   1 #!/usr/bin/env python3
   2 """
   3 Simple static site generator for the pandas web.
   4 
   5 pandas_web.py takes a directory as parameter, and copies all the files into the
   6 target directory after converting markdown files into html and rendering both
   7 markdown and html files with a context. The context is obtained by parsing
   8 the file ``config.yml`` in the root of the source directory.
   9 
  10 The file should contain:
  11 ```
  12 main:
  13   template_path: <path_to_the_jinja2_templates_directory>
  14   base_template: <template_file_all_other_files_will_extend>
  15   ignore:
  16   - <list_of_files_in_the_source_that_will_not_be_copied>
  17   github_repo_url: <organization/repo-name>
  18   context_preprocessors:
  19   - <list_of_functions_that_will_enrich_the_context_parsed_in_this_file>
  20   markdown_extensions:
  21   - <list_of_markdown_extensions_that_will_be_loaded>
  22 ```
  23 
  24 The rest of the items in the file will be added directly to the context.
  25 """

  45 class Preprocessors:
  46     """
  47     Built-in context preprocessors.
  48 
  49     Context preprocessors are functions that receive the context used to
  50     render the templates, and enriches it with additional information.
  51 
  52     The original context is obtained by parsing ``config.yml``, and
  53     anything else needed just be added with context preprocessors.
  54     """
  55 
  56     @staticmethod
  57     def navbar_add_info(context):
  58         """
  59         Items in the main navigation bar can be direct links, or dropdowns with
  60         subitems. This context preprocessor adds a boolean field
  61         ``has_subitems`` that tells which one of them every element is. It
  62         also adds a ``slug`` field to be used as a CSS id.
  63         """
  64         for i, item in enumerate(context["navbar"]):
  65             context["navbar"][i] = dict(
  66                 item,
  67                 has_subitems=isinstance(item["target"], list),
  68                 slug=(item["name"].replace(" ", "-").lower()),
  69             )
  70         return context

  72     @staticmethod
  73     def blog_add_posts(context):
  74         """
  75         Given the blog feed defined in the configuration yaml, this context
  76         preprocessor fetches the posts in the feeds, and returns the relevant
  77         information for them (sorted from newest to oldest).
  78         """
  79         tag_expr = re.compile("<.*?>")
  80         posts = []
  81         # posts from the file system
  82         if context["blog"]["posts_path"]:
  83             posts_path = os.path.join(
  84                 context["source_path"], *context["blog"]["posts_path"].split("/")
  85             )
  86             for fname in os.listdir(posts_path):
  87                 if fname.startswith("index."):
  88                     continue
  89                 link = (
  90                     f"/{context['blog']['posts_path']}"
  91                     f"/{os.path.splitext(fname)[0]}.html"
  92                 )
  93                 md = markdown.Markdown(
  94                     extensions=context["main"]["markdown_extensions"]
  95                 )
  96                 with open(os.path.join(posts_path, fname)) as f:
  97                     html = md.convert(f.read())
  98                 title = md.Meta["title"][0]
  99                 summary = re.sub(tag_expr, "", html)
 100                 try:
 101                     body_position = summary.index(title) + len(title)
 102                 except ValueError:
 103                     raise ValueError(
 104                         f'Blog post "{fname}" should have a markdown header '
 105                         f'corresponding to its "Title" element "{title}"'
 106                     )
 107                 summary = " ".join(summary[body_position:].split(" ")[:30])
 108                 posts.append(
 109                     {
 110                         "title": title,
 111                         "author": context["blog"]["author"],
 112                         "published": datetime.datetime.strptime(
 113                             md.Meta["date"][0], "%Y-%m-%d"
 114                         ),
 115                         "feed": context["blog"]["feed_name"],
 116                         "link": link,
 117                         "description": summary,
 118                         "summary": summary,
 119                     }
 120                 )
 121         # posts from rss feeds
 122         for feed_url in context["blog"]["feed"]:
 123             feed_data = feedparser.parse(feed_url)
 124             for entry in feed_data.entries:
 125                 published = datetime.datetime.fromtimestamp(
 126                     time.mktime(entry.published_parsed)
 127                 )
 128                 summary = re.sub(tag_expr, "", entry.summary)
 129                 posts.append(
 130                     {
 131                         "title": entry.title,
 132                         "author": entry.author,
 133                         "published": published,
 134                         "feed": feed_data["feed"]["title"],
 135                         "link": entry.link,
 136                         "description": entry.description,
 137                         "summary": summary,
 138                     }
 139                 )
 140         posts.sort(key=operator.itemgetter("published"), reverse=True)
 141         context["blog"]["posts"] = posts[: context["blog"]["num_posts"]]
 142         return context

 144     @staticmethod
 145     def maintainers_add_info(context):
 146         """
 147         Given the active maintainers defined in the yaml file, it fetches
 148         the GitHub user information for them.
 149         """
 150         context["maintainers"]["people"] = []
 151         for user in context["maintainers"]["active"]:
 152             resp = requests.get(f"https://api.github.com/users/{user}")
 153             if context["ignore_io_errors"] and resp.status_code == 403:
 154                 return context
 155             resp.raise_for_status()
 156             context["maintainers"]["people"].append(resp.json())
 157         return context
 158 
 159     @staticmethod
 160     def home_add_releases(context):
 161         context["releases"] = []
 162 
 163         github_repo_url = context["main"]["github_repo_url"]
 164         resp = requests.get(f"https://api.github.com/repos/{github_repo_url}/releases")
 165         if context["ignore_io_errors"] and resp.status_code == 403:
 166             return context
 167         resp.raise_for_status()
 168 
 169         for release in resp.json():
 170             if release["prerelease"]:
 171                 continue
 172             published = datetime.datetime.strptime(
 173                 release["published_at"], "%Y-%m-%dT%H:%M:%SZ"
 174             )
 175             context["releases"].append(
 176                 {
 177                     "name": release["tag_name"].lstrip("v"),
 178                     "tag": release["tag_name"],
 179                     "published": published,
 180                     "url": (
 181                         release["assets"][0]["browser_download_url"]
 182                         if release["assets"]
 183                         else ""
 184                     ),
 185                 }
 186             )
 187         return context

 190 def get_callable(obj_as_str: str) -> object:
 191     """
 192     Get a Python object from its string representation.
 193 
 194     For example, for ``sys.stdout.write`` would import the module ``sys``
 195     and return the ``write`` function.
 196     """
 197     components = obj_as_str.split(".")
 198     attrs = []
 199     while components:
 200         try:
 201             obj = importlib.import_module(".".join(components))
 202         except ImportError:
 203             attrs.insert(0, components.pop())
 204         else:
 205             break
 206 
 207     if not obj:
 208         raise ImportError(f'Could not import "{obj_as_str}"')
 209 
 210     for attr in attrs:
 211         obj = getattr(obj, attr)
 212 
 213     return obj

 216 def get_context(config_fname: str, ignore_io_errors: bool, **kwargs):
 217     """
 218     Load the config yaml as the base context, and enrich it with the
 219     information added by the context preprocessors defined in the file.
 220     """
 221     with open(config_fname) as f:
 222         context = yaml.safe_load(f)
 223 
 224     context["source_path"] = os.path.dirname(config_fname)
 225     context["ignore_io_errors"] = ignore_io_errors
 226     context.update(kwargs)
 227 
 228     preprocessors = (
 229         get_callable(context_prep)
 230         for context_prep in context["main"]["context_preprocessors"]
 231     )
 232     for preprocessor in preprocessors:
 233         context = preprocessor(context)
 234         msg = f"{preprocessor.__name__} is missing the return statement"
 235         assert context is not None, msg
 236 
 237     return context

 240 def get_source_files(source_path: str) -> typing.Generator[str, None, None]:
 241     """
 242     Generate the list of files present in the source directory.
 243     """
 244     for root, dirs, fnames in os.walk(source_path):
 245         root = os.path.relpath(root, source_path)
 246         for fname in fnames:
 247             yield os.path.join(root, fname)

 250 def extend_base_template(content: str, base_template: str) -> str:
 251     """
 252     Wrap document to extend the base template, before it is rendered with
 253     Jinja2.
 254     """
 255     result = '{% extends "' + base_template + '" %}'
 256     result += "{% block body %}"
 257     result += content
 258     result += "{% endblock %}"
 259     return result

 262 def main(
 263     source_path: str, target_path: str, base_url: str, ignore_io_errors: bool
 264 ) -> int:
 265     """
 266     Copy every file in the source directory to the target directory.
 267 
 268     For ``.md`` and ``.html`` files, render them with the context
 269     before copyings them. ``.md`` files are transformed to HTML.
 270     """
 271     config_fname = os.path.join(source_path, "config.yml")
 272 
 273     shutil.rmtree(target_path, ignore_errors=True)
 274     os.makedirs(target_path, exist_ok=True)
 275 
 276     sys.stderr.write("Generating context...\n")
 277     context = get_context(config_fname, ignore_io_errors, base_url=base_url)
 278     sys.stderr.write("Context generated\n")
 279 
 280     templates_path = os.path.join(source_path, context["main"]["templates_path"])
 281     jinja_env = jinja2.Environment(loader=jinja2.FileSystemLoader(templates_path))
 282 
 283     for fname in get_source_files(source_path):
 284         if os.path.normpath(fname) in context["main"]["ignore"]:
 285             continue
 286 
 287         sys.stderr.write(f"Processing {fname}\n")
 288         dirname = os.path.dirname(fname)
 289         os.makedirs(os.path.join(target_path, dirname), exist_ok=True)
 290 
 291         extension = os.path.splitext(fname)[-1]
 292         if extension in (".html", ".md"):
 293             with open(os.path.join(source_path, fname)) as f:
 294                 content = f.read()
 295             if extension == ".md":
 296                 body = markdown.markdown(
 297                     content, extensions=context["main"]["markdown_extensions"]
 298                 )
 299                 content = extend_base_template(body, context["main"]["base_template"])
 300             content = jinja_env.from_string(content).render(**context)
 301             fname = os.path.splitext(fname)[0] + ".html"
 302             with open(os.path.join(target_path, fname), "w") as f:
 303                 f.write(content)
 304         else:
 305             shutil.copy(
 306                 os.path.join(source_path, fname), os.path.join(target_path, dirname)
 307             )
 308 
 309 
 310 if __name__ == "__main__":
 311     parser = argparse.ArgumentParser(description="Documentation builder.")
 312     parser.add_argument(
 313         "source_path", help="path to the source directory (must contain config.yml)"
 314     )
 315     parser.add_argument(
 316         "--target-path", default="build", help="directory where to write the output"
 317     )
 318     parser.add_argument(
 319         "--base-url", default="", help="base url where the website is served from"
 320     )
 321     parser.add_argument(
 322         "--ignore-io-errors",
 323         action="store_true",
 324         help="do not fail if errors happen when fetching "
 325         "data from http sources, and those fail "
 326         "(mostly useful to allow github quota errors "
 327         "when running the script locally)",
 328     )
 329     args = parser.parse_args()
 330     sys.exit(
 331         main(args.source_path, args.target_path, args.base_url, args.ignore_io_errors)
 332     )
