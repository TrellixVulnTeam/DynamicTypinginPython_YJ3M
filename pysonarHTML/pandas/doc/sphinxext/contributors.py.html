   1 """Sphinx extension for listing code contributors to a release.
   2 
   3 Usage::
   4 
   5    .. contributors:: v0.23.0..v0.23.1
   6 
   7 This will be replaced with a message indicating the number of
   8 code contributors and commits, and then list each contributor
   9 individually. For development versions (before a tag is available)
  10 use::
  11 
  12     .. contributors:: v0.23.0..v0.23.1|HEAD
  13 
  14 While the v0.23.1 tag does not exist, that will use the HEAD of the
  15 branch as the end of the revision range.
  16 """
  17 from announce import build_components
  18 from docutils import nodes
  19 from docutils.parsers.rst import Directive
  20 import git
  21 
  22 
  23 class ContributorsDirective(Directive):
  24     required_arguments = 1
  25     name = "contributors"
  26 
  27     def run(self):
  28         range_ = self.arguments[0]
  29         if range_.endswith("x..HEAD"):
  30             return [nodes.paragraph(), nodes.bullet_list()]
  31         try:
  32             components = build_components(range_)
  33         except git.GitCommandError as exc:
  34             return [
  35                 self.state.document.reporter.warning(
  36                     f"Cannot find contributors for range {repr(range_)}: {exc}",
  37                     line=self.lineno,
  38                 )
  39             ]
  40         else:
  41             message = nodes.paragraph()
  42             message += nodes.Text(components["author_message"])
  43 
  44             listnode = nodes.bullet_list()
  45 
  46             for author in components["authors"]:
  47                 para = nodes.paragraph()
  48                 para += nodes.Text(author)
  49                 listnode += nodes.list_item("", para)
  50 
  51         return [message, listnode]
  52 
  53 
  54 def setup(app):
  55     app.add_directive("contributors", ContributorsDirective)
  56 
  57     return {"version": "0.1", "parallel_read_safe": True, "parallel_write_safe": True}
