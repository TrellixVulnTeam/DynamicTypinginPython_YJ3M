   1 #!/usr/bin/env python3
   2 # -*- encoding:utf-8 -*-
   3 """
   4 Script to generate contributor and pull request lists
   5 
   6 This script generates contributor and pull request lists for release
   7 announcements using Github v3 protocol. Use requires an authentication token in
   8 order to have sufficient bandwidth, you can get one following the directions at
   9 `<https://help.github.com/articles/creating-an-access-token-for-command-line-use/>_
  10 Don't add any scope, as the default is read access to public information. The
  11 token may be stored in an environment variable as you only get one chance to
  12 see it.
  13 
  14 Usage::
  15 
  16     $ ./scripts/announce.py <token> <revision range>
  17 
  18 The output is utf8 rst.
  19 
  20 Dependencies
  21 ------------
  22 
  23 - gitpython
  24 - pygithub
  25 
  26 Some code was copied from scipy `tools/gh_lists.py` and `tools/authors.py`.
  27 
  28 Examples
  29 --------
  30 
  31 From the bash command line with $GITHUB token.
  32 
  33     $ ./scripts/announce.py $GITHUB v1.11.0..v1.11.1 > announce.rst
  34 
  35 """

  37 import os
  38 import re
  39 import textwrap

  43 UTF8Writer = codecs.getwriter("utf8")
  44 this_repo = Repo(os.path.join(os.path.dirname(__file__), "..", ".."))

  47 A total of %d people contributed patches to this release.  People with a
  48 "+" by their names contributed a patch for the first time.
  49 """

  51 pull_request_msg = """\
  52 A total of %d pull requests were merged for this release.

  56 def get_authors(revision_range):
  57     pat = "^.*\\t(.*)$"
  58     lst_release, cur_release = [r.strip() for r in revision_range.split("..")]
  59 
  60     if "|" in cur_release:
  61         # e.g. v1.0.1|HEAD
  62         maybe_tag, head = cur_release.split("|")
  63         assert head == "HEAD"
  64         if maybe_tag in this_repo.tags:
  65             cur_release = maybe_tag
  66         else:
  67             cur_release = head
  68         revision_range = f"{lst_release}..{cur_release}"
  69 
  70     # authors, in current release and previous to current release.
  71     # We need two passes over the log for cur and prev, one to get the
  72     # "Co-authored by" commits, which come from backports by the bot,
  73     # and one for regular commits.
  74     xpr = re.compile(r"Co-authored-by: (?P<name>[^<]+) ")
  75     cur = set(
  76         xpr.findall(
  77             this_repo.git.log("--grep=Co-authored", "--pretty=%b", revision_range)
  78         )
  79     )
  80     cur |= set(re.findall(pat, this_repo.git.shortlog("-s", revision_range), re.M))
  81 
  82     pre = set(
  83         xpr.findall(this_repo.git.log("--grep=Co-authored", "--pretty=%b", lst_release))
  84     )
  85     pre |= set(re.findall(pat, this_repo.git.shortlog("-s", lst_release), re.M))
  86 
  87     # Homu is the author of auto merges, clean him out.
  88     cur.discard("Homu")
  89     pre.discard("Homu")
  90 
  91     # Append '+' to new authors.
  92     authors = [s + " +" for s in cur - pre] + [s for s in cur & pre]
  93     authors.sort()
  94     return authors
  95

  97 def get_pull_requests(repo, revision_range):
  98     prnums = []
  99 
 100     # From regular merges
 101     merges = this_repo.git.log("--oneline", "--merges", revision_range)
 102     issues = re.findall("Merge pull request \\#(\\d*)", merges)
 103     prnums.extend(int(s) for s in issues)
 104 
 105     # From Homu merges (Auto merges)
 106     issues = re.findall("Auto merge of \\#(\\d*)", merges)
 107     prnums.extend(int(s) for s in issues)
 108 
 109     # From fast forward squash-merges
 110     commits = this_repo.git.log(
 111         "--oneline", "--no-merges", "--first-parent", revision_range
 112     )
 113     issues = re.findall("^.*\\(\\#(\\d+)\\)$", commits, re.M)
 114     prnums.extend(int(s) for s in issues)
 115 
 116     # get PR data from github repo
 117     prnums.sort()
 118     prs = [repo.get_pull(n) for n in prnums]
 119     return prs

 122 def build_components(revision_range, heading="Contributors"):
 123     lst_release, cur_release = [r.strip() for r in revision_range.split("..")]
 124     authors = get_authors(revision_range)
 125 
 126     return {
 127         "heading": heading,
 128         "author_message": author_msg % len(authors),
 129         "authors": authors,
 130     }
 131 
 132 
 133 def build_string(revision_range, heading="Contributors"):
 134     components = build_components(revision_range, heading=heading)
 135     components["uline"] = "=" * len(components["heading"])
 136     components["authors"] = "* " + "\n* ".join(components["authors"])
 137 
 138     # Don't change this to an fstring. It breaks the formatting.
 139     tpl = textwrap.dedent(
 140         """\
 141     {heading}
 142     {uline}
 143 
 144     {author_message}
 145     {authors}"""
 146     ).format(**components)
 147     return tpl

 150 def main(revision_range):
 151     # document authors
 152     text = build_string(revision_range)
 153     print(text)
