 145 err_msg = "Heading capitalization formatted incorrectly. Please correctly capitalize"
 146 
 147 symbols = ("*", "=", "-", "^", "~", "#", '"')

 150 def correct_title_capitalization(title: str) -> str:
 151     """
 152     Algorithm to create the correct capitalization for a given title.
 153 
 154     Parameters
 155     ----------
 156     title : str
 157         Heading string to correct.
 158 
 159     Returns
 160     -------
 161     str
 162         Correctly capitalized heading.
 163     """
 164 
 165     # Skip modification no matter what if title begins by ":" to exclude specific
 166     # syntax that is needed to build links.
 167     if title[0] == ":":
 168         return title
 169 
 170     # Strip all non-word characters from the beginning of the title to the
 171     # first word character.
 172     correct_title: str = re.sub(r"^\W*", "", title).capitalize()
 173 
 174     # Remove a URL from the title. We do this because words in a URL must
 175     # stay lowercase, even if they are a capitalization exception.
 176     removed_https_title = re.sub(r"<https?:\/\/.*[\r\n]*>", "", correct_title)
 177 
 178     # Split a title into a list using non-word character delimiters.
 179     word_list = re.split(r"\W", removed_https_title)
 180 
 181     for word in word_list:
 182         if word.lower() in CAP_EXCEPTIONS_DICT:
 183             correct_title = re.sub(
 184                 rf"\b{word}\b", CAP_EXCEPTIONS_DICT[word.lower()], correct_title
 185             )
 186 
 187     return correct_title

 190 def find_titles(rst_file: str) -> Iterable[Tuple[str, int]]:
 191     """
 192     Algorithm to identify particular text that should be considered headings in an
 193     RST file.
 194 
 195     See <https://thomas-cokelaer.info/tutorials/sphinx/rest_syntax.html> for details
 196     on what constitutes a string as a heading in RST.
 197 
 198     Parameters
 199     ----------
 200     rst_file : str
 201         RST file to scan through for headings.
 202 
 203     Yields
 204     -------
 205     title : str
 206         A heading found in the rst file.
 207 
 208     line_number : int
 209         The corresponding line number of the heading.
 210     """
 211 
 212     with open(rst_file, "r") as fd:
 213         previous_line = ""
 214         for i, line in enumerate(fd):
 215             line = line[:-1]
 216             line_chars = set(line)
 217             if (
 218                 len(line_chars) == 1
 219                 and line_chars.pop() in symbols
 220                 and len(line) == len(previous_line)
 221             ):
 222                 yield re.sub(r"[`\*_]", "", previous_line), i
 223             previous_line = line
 224

 226 def find_rst_files(source_paths: List[str]) -> Iterable[str]:
 227     """
 228     Given the command line arguments of directory paths, this method
 229     yields the strings of the .rst file directories that these paths contain.
 230 
 231     Parameters
 232     ----------
 233     source_paths : str
 234         List of directories to validate, provided through command line arguments.
 235 
 236     Yields
 237     -------
 238     str
 239         Directory address of a .rst files found in command line argument directories.
 240     """
 241 
 242     for directory_address in source_paths:
 243         if not os.path.exists(directory_address):
 244             raise ValueError(
 245                 "Please enter a valid path, pointing to a valid file/directory."
 246             )
 247         elif directory_address.endswith(".rst"):
 248             yield directory_address
 249         else:
 250             for filename in glob.glob(
 251                 pathname=f"{directory_address}/**/*.rst", recursive=True
 252             ):
 253                 yield filename

 256 def main(source_paths: List[str], output_format: str) -> int:
 257     """
 258     The main method to print all headings with incorrect capitalization.
 259 
 260     Parameters
 261     ----------
 262     source_paths : str
 263         List of directories to validate, provided through command line arguments.
 264     output_format : str
 265         Output format of the script.
 266 
 267     Returns
 268     -------
 269     int
 270         Number of incorrect headings found overall.
 271     """
 272 
 273     number_of_errors: int = 0
 274 
 275     for filename in find_rst_files(source_paths):
 276         for title, line_number in find_titles(filename):
 277             if title != correct_title_capitalization(title):
 278                 print(
 279                     f"""{filename}:{line_number}:{err_msg} "{title}" to "{
 280                     correct_title_capitalization(title)}" """
 281                 )
 282                 number_of_errors += 1
 283 
 284     return number_of_errors
