  86 class TestValidator:
  87     def _import_path(self, klass=None, func=None):
  88         """
  89         Build the required import path for tests in this module.
  90 
  91         Parameters
  92         ----------
  93         klass : str
  94             Class name of object in module.
  95         func : str
  96             Function name of object in module.
  97 
  98         Returns
  99         -------
 100         str
 101             Import path of specified object in this module
 102         """
 103         base_path = "scripts.tests.test_validate_docstrings"
 104 
 105         if klass:
 106             base_path = ".".join([base_path, klass])
 107 
 108         if func:
 109             base_path = ".".join([base_path, func])
 110 
 111         return base_path
