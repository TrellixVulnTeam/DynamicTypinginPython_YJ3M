  89 class NonNumericSeriesIndexing:
  90 
  91     params = [
  92         ("string", "datetime", "period"),
  93         ("unique_monotonic_inc", "nonunique_monotonic_inc", "non_monotonic"),
  94     ]
  95     param_names = ["index_dtype", "index_structure"]
  96 
  97     def setup(self, index, index_structure):
  98         N = 10 ** 6
  99         if index == "string":
 100             index = tm.makeStringIndex(N)
 101         elif index == "datetime":
 102             index = date_range("1900", periods=N, freq="s")
 103         elif index == "period":
 104             index = period_range("1900", periods=N, freq="s")
 105         index = index.sort_values()
 106         assert index.is_unique and index.is_monotonic_increasing
 107         if index_structure == "nonunique_monotonic_inc":
 108             index = index.insert(item=index[2], loc=2)[:-1]
 109         elif index_structure == "non_monotonic":
 110             index = index[::2].append(index[1::2])
 111             assert len(index) == N
 112         self.s = Series(np.random.rand(N), index=index)
 113         self.lbl = index[80000]
 114         # warm up index mapping
 115         self.s[self.lbl]
 116 
 117     def time_getitem_label_slice(self, index, index_structure):
 118         self.s[: self.lbl]
 119 
 120     def time_getitem_pos_slice(self, index, index_structure):
 121         self.s[:80000]
 122 
 123     def time_getitem_scalar(self, index, index_structure):
 124         self.s[self.lbl]
 125 
 126     def time_getitem_list_like(self, index, index_structure):
 127         self.s[[self.lbl]]

 244 class CategoricalIndexIndexing:
 245 
 246     params = ["monotonic_incr", "monotonic_decr", "non_monotonic"]
 247     param_names = ["index"]
 248 
 249     def setup(self, index):
 250         N = 10 ** 5
 251         values = list("a" * N + "b" * N + "c" * N)
 252         indices = {
 253             "monotonic_incr": CategoricalIndex(values),
 254             "monotonic_decr": CategoricalIndex(reversed(values)),
 255             "non_monotonic": CategoricalIndex(list("abc" * N)),
 256         }
 257         self.data = indices[index]
 258 
 259         self.int_scalar = 10000
 260         self.int_list = list(range(10000))
 261 
 262         self.cat_scalar = "b"
 263         self.cat_list = ["a", "c"]
 264 
 265     def time_getitem_scalar(self, index):
 266         self.data[self.int_scalar]
 267 
 268     def time_getitem_slice(self, index):
 269         self.data[: self.int_scalar]
 270 
 271     def time_getitem_list_like(self, index):
 272         self.data[[self.int_scalar]]
 273 
 274     def time_getitem_list(self, index):
 275         self.data[self.int_list]
 276 
 277     def time_getitem_bool_array(self, index):
 278         self.data[self.data == self.cat_scalar]
 279 
 280     def time_get_loc_scalar(self, index):
 281         self.data.get_loc(self.cat_scalar)

 311 class IndexSingleRow:
 312     params = [True, False]
 313     param_names = ["unique_cols"]
 314 
 315     def setup(self, unique_cols):
 316         arr = np.arange(10 ** 7).reshape(-1, 10)
 317         df = DataFrame(arr)
 318         dtypes = ["u1", "u2", "u4", "u8", "i1", "i2", "i4", "i8", "f8", "f4"]
 319         for i, d in enumerate(dtypes):
 320             df[i] = df[i].astype(d)
 321 
 322         if not unique_cols:
 323             # GH#33032 single-row lookups with non-unique columns were
 324             #  15x slower than with unique columns
 325             df.columns = ["A", "A"] + list(df.columns[2:])
 326 
 327         self.df = df
 328 
 329     def time_iloc_row(self, unique_cols):
 330         self.df.iloc[10000]
 331 
 332     def time_loc_row(self, unique_cols):
 333         self.df.loc[10000]

 362 class ChainIndexing:
 363 
 364     params = [None, "warn"]
 365     param_names = ["mode"]
 366 
 367     def setup(self, mode):
 368         self.N = 1000000
 369 
 370     def time_chained_indexing(self, mode):
 371         with warnings.catch_warnings(record=True):
 372             with option_context("mode.chained_assignment", mode):
 373                 df = DataFrame({"A": np.arange(self.N), "B": "foo"})
 374                 df2 = df[df.A > self.N // 2]
 375                 df2["C"] = 1.0
