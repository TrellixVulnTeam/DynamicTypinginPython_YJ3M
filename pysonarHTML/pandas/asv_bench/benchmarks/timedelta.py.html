  11 class ToTimedelta:
  12     def setup(self):
  13         self.ints = np.random.randint(0, 60, size=10000)
  14         self.str_days = []
  15         self.str_seconds = []
  16         for i in self.ints:
  17             self.str_days.append(f"{i} days")
  18             self.str_seconds.append(f"00:00:{i:02d}")
  19 
  20     def time_convert_int(self):
