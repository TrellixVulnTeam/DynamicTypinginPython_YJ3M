 186 class AsOf:
 187 
 188     params = ["DataFrame", "Series"]
 189     param_names = ["constructor"]
 190 
 191     def setup(self, constructor):
 192         N = 10000
 193         M = 10
 194         rng = date_range(start="1/1/1990", periods=N, freq="53s")
 195         data = {
 196             "DataFrame": DataFrame(np.random.randn(N, M)),
 197             "Series": Series(np.random.randn(N)),
 198         }
 199         self.ts = data[constructor]
 200         self.ts.index = rng
 201         self.ts2 = self.ts.copy()
 202         self.ts2.iloc[250:5000] = np.nan
 203         self.ts3 = self.ts.copy()
 204         self.ts3.iloc[-5000:] = np.nan
 205         self.dates = date_range(start="1/1/1990", periods=N * 10, freq="5s")
 206         self.date = self.dates[0]
 207         self.date_last = self.dates[-1]
 208         self.date_early = self.date - timedelta(10)
 209 
 210     # test speed of pre-computing NAs.
 211     def time_asof(self, constructor):
 212         self.ts.asof(self.dates)
 213 
 214     # should be roughly the same as above.
 215     def time_asof_nan(self, constructor):
 216         self.ts2.asof(self.dates)
 217 
 218     # test speed of the code path for a scalar index
 219     # without *while* loop
 220     def time_asof_single(self, constructor):
 221         self.ts.asof(self.date)
 222 
 223     # test speed of the code path for a scalar index
 224     # before the start. should be the same as above.
 225     def time_asof_single_early(self, constructor):
 226         self.ts.asof(self.date_early)
 227 
 228     # test the speed of the code path for a scalar index
 229     # with a long *while* loop. should still be much
 230     # faster than pre-computing all the NAs.
 231     def time_asof_nan_single(self, constructor):
 232         self.ts3.asof(self.date_last)
