  49 class Unstack:
  50 
  51     params = ["int", "category"]
  52 
  53     def setup(self, dtype):
  54         m = 100
  55         n = 1000
  56 
  57         levels = np.arange(m)
  58         index = MultiIndex.from_product([levels] * 2)
  59         columns = np.arange(n)
  60         if dtype == "int":
  61             values = np.arange(m * m * n).reshape(m * m, n)
  62         else:
  63             # the category branch is ~20x slower than int. So we
  64             # cut down the size a bit. Now it's only ~3x slower.
  65             n = 50
  66             columns = columns[:n]
  67             indices = np.random.randint(0, 52, size=(m * m, n))
  68             values = np.take(list(string.ascii_letters), indices)
  69             values = [pd.Categorical(v) for v in values.T]
  70 
  71         self.df = DataFrame(values, index, columns)
  72         self.df2 = self.df.iloc[:-1]
  73 
  74     def time_full_product(self, dtype):
  75         self.df.unstack()
  76 
  77     def time_without_last_row(self, dtype):
  78         self.df2.unstack()

 100 class WideToLong:
 101     def setup(self):
 102         nyrs = 20
 103         nidvars = 20
 104         N = 5000
 105         self.letters = list("ABCD")
 106         yrvars = [l + str(num) for l, num in product(self.letters, range(1, nyrs + 1))]
 107         columns = [str(i) for i in range(nidvars)] + yrvars
 108         self.df = DataFrame(np.random.randn(N, nidvars + len(yrvars)), columns=columns)
 109         self.df["id"] = self.df.index

 111     def time_wide_to_long_big(self):
 112         wide_to_long(self.df, self.letters, i="id", j="year")

 115 class PivotTable:
 116     def setup(self):
 117         N = 100000
 118         fac1 = np.array(["A", "B", "C"], dtype="O")
 119         fac2 = np.array(["one", "two"], dtype="O")
 120         ind1 = np.random.randint(0, 3, size=N)
 121         ind2 = np.random.randint(0, 2, size=N)
 122         self.df = DataFrame(
 123             {
 124                 "key1": fac1.take(ind1),
 125                 "key2": fac2.take(ind2),
 126                 "key3": fac2.take(ind2),
 127                 "value1": np.random.randn(N),
 128                 "value2": np.random.randn(N),
 129                 "value3": np.random.randn(N),
 130             }
 131         )
 132         self.df2 = DataFrame(
 133             {"col1": list("abcde"), "col2": list("fghij"), "col3": [1, 2, 3, 4, 5]}
 134         )
 135         self.df2.col1 = self.df2.col1.astype("category")
 136         self.df2.col2 = self.df2.col2.astype("category")
 137 
 138     def time_pivot_table(self):
 139         self.df.pivot_table(index="key1", columns=["key2", "key3"])
 140 
 141     def time_pivot_table_agg(self):
 142         self.df.pivot_table(
 143             index="key1", columns=["key2", "key3"], aggfunc=["sum", "mean"]
 144         )
 145 
 146     def time_pivot_table_margins(self):
 147         self.df.pivot_table(index="key1", columns=["key2", "key3"], margins=True)
 148 
 149     def time_pivot_table_categorical(self):
 150         self.df2.pivot_table(
 151             index="col1", values="col3", columns="col2", aggfunc=np.sum, fill_value=0
 152         )
 153 
 154     def time_pivot_table_categorical_observed(self):
 155         self.df2.pivot_table(
 156             index="col1",
 157             values="col3",
 158             columns="col2",
 159             aggfunc=np.sum,
 160             fill_value=0,
 161             observed=True,
 162         )
 163 
 164     def time_pivot_table_margins_only_column(self):
 165         self.df.pivot_table(columns=["key2", "key3"], margins=True)

 172         fac2 = np.array(["one", "two"], dtype="O")
 173         self.ind1 = np.random.randint(0, 3, size=N)

 175         self.vec1 = fac1.take(self.ind1)
