 150 class JoinNonUnique:
 151     # outer join of non-unique
 152     # GH 6329
 153     def setup(self):
 154         date_index = date_range("01-Jan-2013", "23-Jan-2013", freq="T")
 155         daily_dates = date_index.to_period("D").to_timestamp("S", "S")
 156         self.fracofday = date_index.values - daily_dates.values
 157         self.fracofday = self.fracofday.astype("timedelta64[ns]")
 158         self.fracofday = self.fracofday.astype(np.float64) / 86400000000000.0
 159         self.fracofday = Series(self.fracofday, daily_dates)
 160         index = date_range(date_index.min(), date_index.max(), freq="D")
 161         self.temp = Series(1.0, index)[self.fracofday.index]
 162 
 163     def time_join_non_unique_equal(self):
 164         self.fracofday * self.temp

 211     params = ["inner", "outer", "left", "right"]
 212     param_names = ["how"]
 213 
 214     def setup(self, how):
 215         low, high, n = -1000, 1000, 10 ** 6
 216         self.left = DataFrame(
 217             np.random.randint(low, high, (n, 7)), columns=list("ABCDEFG")
 218         )
 219         self.left["left"] = self.left.sum(axis=1)
 220         self.right = self.left.sample(frac=1).rename({"left": "right"}, axis=1)

 276 class MergeAsof:
 277     params = [["backward", "forward", "nearest"], [None, 5]]
 278     param_names = ["direction", "tolerance"]
 279 
 280     def setup(self, direction, tolerance):
 281         one_count = 200000
 282         two_count = 1000000
 283 
 284         df1 = DataFrame(
 285             {
 286                 "time": np.random.randint(0, one_count / 20, one_count),
 287                 "key": np.random.choice(list(string.ascii_uppercase), one_count),
 288                 "key2": np.random.randint(0, 25, one_count),
 289                 "value1": np.random.randn(one_count),
 290             }
 291         )
 292         df2 = DataFrame(
 293             {
 294                 "time": np.random.randint(0, two_count / 20, two_count),
 295                 "key": np.random.choice(list(string.ascii_uppercase), two_count),
 296                 "key2": np.random.randint(0, 25, two_count),
 297                 "value2": np.random.randn(two_count),
 298             }
 299         )
 300 
 301         df1 = df1.sort_values("time")
 302         df2 = df2.sort_values("time")
 303 
 304         df1["time32"] = np.int32(df1.time)
 305         df2["time32"] = np.int32(df2.time)
 306 
 307         df1["timeu64"] = np.uint64(df1.time)
 308         df2["timeu64"] = np.uint64(df2.time)
 309 
 310         self.df1a = df1[["time", "value1"]]
 311         self.df2a = df2[["time", "value2"]]
 312         self.df1b = df1[["time", "key", "value1"]]
 313         self.df2b = df2[["time", "key", "value2"]]
 314         self.df1c = df1[["time", "key2", "value1"]]
 315         self.df2c = df2[["time", "key2", "value2"]]
 316         self.df1d = df1[["time32", "value1"]]
 317         self.df2d = df2[["time32", "value2"]]
 318         self.df1e = df1[["time", "key", "key2", "value1"]]
 319         self.df2e = df2[["time", "key", "key2", "value2"]]
 320         self.df1f = df1[["timeu64", "value1"]]
 321         self.df2f = df2[["timeu64", "value2"]]
 322 
 323     def time_on_int(self, direction, tolerance):
 324         merge_asof(
 325             self.df1a, self.df2a, on="time", direction=direction, tolerance=tolerance
 326         )
 327 
 328     def time_on_int32(self, direction, tolerance):
 329         merge_asof(
 330             self.df1d, self.df2d, on="time32", direction=direction, tolerance=tolerance
 331         )
 332 
 333     def time_on_uint64(self, direction, tolerance):
 334         merge_asof(
 335             self.df1f, self.df2f, on="timeu64", direction=direction, tolerance=tolerance
 336         )
 337 
 338     def time_by_object(self, direction, tolerance):
 339         merge_asof(
 340             self.df1b,
 341             self.df2b,
 342             on="time",
