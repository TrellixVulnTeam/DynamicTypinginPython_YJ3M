  59 class IsInForObjects:
  60     def setup(self):
  61         self.s_nans = Series(np.full(10 ** 4, np.nan)).astype(np.object)
  62         self.vals_nans = np.full(10 ** 4, np.nan).astype(np.object)
  63         self.s_short = Series(np.arange(2)).astype(np.object)
  64         self.s_long = Series(np.arange(10 ** 5)).astype(np.object)
  65         self.vals_short = np.arange(2).astype(np.object)
  66         self.vals_long = np.arange(10 ** 5).astype(np.object)
  67         # because of nans floats are special:
  68         self.s_long_floats = Series(np.arange(10 ** 5, dtype=np.float)).astype(
  69             np.object
  70         )
  71         self.vals_long_floats = np.arange(10 ** 5, dtype=np.float).astype(np.object)
  72 
  73     def time_isin_nans(self):
  74         # if nan-objects are different objects,
  75         # this has the potential to trigger O(n^2) running time
  76         self.s_nans.isin(self.vals_nans)
  77 
  78     def time_isin_short_series_long_values(self):
  79         # running time dominated by the preprocessing
  80         self.s_short.isin(self.vals_long)
  81 
  82     def time_isin_long_series_short_values(self):
  83         # running time dominated by look-up
  84         self.s_long.isin(self.vals_short)
  85 
  86     def time_isin_long_series_long_values(self):
  87         # no dominating part
  88         self.s_long.isin(self.vals_long)
  89 
  90     def time_isin_long_series_long_values_floats(self):
  91         # no dominating part
  92         self.s_long_floats.isin(self.vals_long_floats)

 158 class Map:
 159 
 160     params = (["dict", "Series", "lambda"], ["object", "category", "int"])
 161     param_names = "mapper"
 162 
 163     def setup(self, mapper, dtype):
 164         map_size = 1000
 165         map_data = Series(map_size - np.arange(map_size), dtype=dtype)
 166 
 167         # construct mapper
 168         if mapper == "Series":
 169             self.map_data = map_data
 170         elif mapper == "dict":
 171             self.map_data = map_data.to_dict()
 172         elif mapper == "lambda":
 173             map_dict = map_data.to_dict()
 174             self.map_data = lambda x: map_dict[x]
 175         else:
 176             raise NotImplementedError
 177 
 178         self.s = Series(np.random.randint(0, map_size, 10000), dtype=dtype)
 179 
 180     def time_map(self, mapper, *args, **kwargs):
 181         self.s.map(self.map_data)

 250 class NanOps:
 251 
 252     params = [
 253         [
 254             "var",
 255             "mean",
 256             "median",
 257             "max",
 258             "min",
 259             "sum",
 260             "std",
 261             "sem",
 262             "argmax",
 263             "skew",
 264             "kurt",
 265             "prod",
 266         ],
 267         [10 ** 3, 10 ** 6],
 268         ["int8", "int32", "int64", "float64", "Int64", "boolean"],
 269     ]
 270     param_names = ["func", "N", "dtype"]
 271 
 272     def setup(self, func, N, dtype):
 273         if func == "argmax" and dtype in {"Int64", "boolean"}:
 274             # Skip argmax for nullable int since this doesn't work yet (GH-24382)
 275             raise NotImplementedError
 276         self.s = Series([1] * N, dtype=dtype)
 277         self.func = getattr(self.s, func)
 278 
 279     def time_func(self, func, N, dtype):
 280         self.func()
