  63 class BaseIO:
  64     """
  65     Base class for IO benchmarks
  66     """
  67 
  68     fname = None
  69 
  70     def remove(self, f):
  71         """Remove created files"""
  72         try:
  73             os.remove(f)
  74         except OSError:
  75             # On Windows, attempting to remove a file that is in use
  76             # causes an exception to be raised
  77             pass
  78 
  79     def teardown(self, *args, **kwargs):
  80         self.remove(self.fname)
