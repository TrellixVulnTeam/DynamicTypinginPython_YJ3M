 211 class CategoricalSlicing:
 212 
 213     params = ["monotonic_incr", "monotonic_decr", "non_monotonic"]
 214     param_names = ["index"]
 215 
 216     def setup(self, index):
 217         N = 10 ** 6
 218         categories = ["a", "b", "c"]
 219         values = [0] * N + [1] * N + [2] * N
 220         if index == "monotonic_incr":
 221             self.data = pd.Categorical.from_codes(values, categories=categories)
 222         elif index == "monotonic_decr":
 223             self.data = pd.Categorical.from_codes(
 224                 list(reversed(values)), categories=categories
 225             )
 226         elif index == "non_monotonic":
 227             self.data = pd.Categorical.from_codes([0, 1, 2] * N, categories=categories)
 228         else:
 229             raise ValueError(f"Invalid index param: {index}")
 230 
 231         self.scalar = 10000
 232         self.list = list(range(10000))
 233         self.cat_scalar = "b"
 234 
 235     def time_getitem_scalar(self, index):
 236         self.data[self.scalar]
 237 
 238     def time_getitem_slice(self, index):
 239         self.data[: self.scalar]
 240 
 241     def time_getitem_list_like(self, index):
 242         self.data[[self.scalar]]
 243 
 244     def time_getitem_list(self, index):
 245         self.data[self.list]
 246 
 247     def time_getitem_bool_array(self, index):
 248         self.data[self.data == self.cat_scalar]
