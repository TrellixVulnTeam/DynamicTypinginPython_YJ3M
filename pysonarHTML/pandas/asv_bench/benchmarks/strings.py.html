 112 class Cat:
 113 
 114     params = ([0, 3], [None, ","], [None, "-"], [0.0, 0.001, 0.15])
 115     param_names = ["other_cols", "sep", "na_rep", "na_frac"]
 116 
 117     def setup(self, other_cols, sep, na_rep, na_frac):
 118         N = 10 ** 5
 119         mask_gen = lambda: np.random.choice([True, False], N, p=[1 - na_frac, na_frac])
 120         self.s = Series(tm.makeStringIndex(N)).where(mask_gen())
 121         if other_cols == 0:
 122             # str.cat self-concatenates only for others=None
 123             self.others = None
 124         else:
 125             self.others = DataFrame(
 126                 {i: tm.makeStringIndex(N).where(mask_gen()) for i in range(other_cols)}
 127             )
 128 
 129     def time_cat(self, other_cols, sep, na_rep, na_frac):
 130         # before the concatenation (one caller + other_cols columns), the total
 131         # expected fraction of rows containing any NaN is:
 132         # reduce(lambda t, _: t + (1 - t) * na_frac, range(other_cols + 1), 0)
 133         # for other_cols=3 and na_frac=0.15, this works out to ~48%
 134         self.s.str.cat(others=self.others, sep=sep, na_rep=na_rep)
