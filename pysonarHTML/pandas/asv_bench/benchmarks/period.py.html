  10 class PeriodIndexConstructor:
  11 
  12     params = [["D"], [True, False]]
  13     param_names = ["freq", "is_offset"]
  14 
  15     def setup(self, freq, is_offset):
  16         self.rng = date_range("1985", periods=1000)
  17         self.rng2 = date_range("1985", periods=1000).to_pydatetime()
  18         self.ints = list(range(2000, 3000))
  19         self.daily_ints = (
  20             date_range("1/1/2000", periods=1000, freq=freq).strftime("%Y%m%d").map(int)
  21         )
  22         if is_offset:
  23             self.freq = to_offset(freq)
  24         else:
  25             self.freq = freq
  26 
  27     def time_from_date_range(self, freq, is_offset):
  28         PeriodIndex(self.rng, freq=freq)
  29 
  30     def time_from_pydatetime(self, freq, is_offset):
  31         PeriodIndex(self.rng2, freq=freq)
  32 
  33     def time_from_ints(self, freq, is_offset):
  34         PeriodIndex(self.ints, freq=freq)
  35 
  36     def time_from_ints_daily(self, freq, is_offset):
  37         PeriodIndex(self.daily_ints, freq=freq)

  56     params = ["index", "series"]
  57     param_names = ["typ"]

  59     def setup(self, typ):
  60         data = [
  61             Period("2011-01", freq="M"),
  62             Period("2011-02", freq="M"),
  63             Period("2011-03", freq="M"),
  64             Period("2011-04", freq="M"),
  65         ]
  66 
  67         if typ == "index":
  68             self.vector = PeriodIndex(data * 1000, freq="M")
  69         elif typ == "series":
  70             self.vector = Series(data * 1000)
