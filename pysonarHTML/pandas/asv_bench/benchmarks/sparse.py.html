   9 def make_array(size, dense_proportion, fill_value, dtype):
  10     dense_size = int(size * dense_proportion)
  11     arr = np.full(size, fill_value, dtype)
  12     indexer = np.random.choice(np.arange(size), dense_size, replace=False)
  13     arr[indexer] = np.random.choice(np.arange(100, dtype=dtype), dense_size)
  14     return arr

 117     def make_block_array(self, length, num_blocks, block_size, fill_value):
 118         arr = np.full(length, fill_value)
 119         indicies = np.random.choice(
 120             np.arange(0, length, block_size), num_blocks, replace=False
 121         )
 122         for ind in indicies:
 123             arr[ind : ind + block_size] = np.random.randint(0, 100, block_size)
 124         return SparseArray(arr, fill_value=fill_value)
