   6 def _get_numeric_engines():
   7     engine_names = [
   8         ("Int64Engine", np.int64),
   9         ("Int32Engine", np.int32),
  10         ("Int16Engine", np.int16),
  11         ("Int8Engine", np.int8),
  12         ("UInt64Engine", np.uint64),
  13         ("UInt32Engine", np.uint32),
  14         ("UInt16engine", np.uint16),
  15         ("UInt8Engine", np.uint8),
  16         ("Float64Engine", np.float64),
  17         ("Float32Engine", np.float32),
  18     ]
  19     return [
  20         (getattr(libindex, engine_name), dtype)
  21         for engine_name, dtype in engine_names
  22         if hasattr(libindex, engine_name)
  23     ]
