 191 class SetOperations:
 192 
 193     params = [
 194         ("monotonic", "non_monotonic"),
 195         ("datetime", "int", "string"),
 196         ("intersection", "union", "symmetric_difference"),
 197     ]
 198     param_names = ["index_structure", "dtype", "method"]
 199 
 200     def setup(self, index_structure, dtype, method):
 201         N = 10 ** 5
 202         level1 = range(1000)
 203 
 204         level2 = date_range(start="1/1/2000", periods=N // 1000)
 205         dates_left = MultiIndex.from_product([level1, level2])
 206 
 207         level2 = range(N // 1000)
 208         int_left = MultiIndex.from_product([level1, level2])
 209 
 210         level2 = tm.makeStringIndex(N // 1000).values
 211         str_left = MultiIndex.from_product([level1, level2])
 212 
 213         data = {
 214             "datetime": dates_left,
 215             "int": int_left,
 216             "string": str_left,
 217         }
 218 
 219         if index_structure == "non_monotonic":
 220             data = {k: mi[::-1] for k, mi in data.items()}
 221 
 222         data = {k: {"left": mi, "right": mi[:-1]} for k, mi in data.items()}
 223         self.left = data[dtype]["left"]
 224         self.right = data[dtype]["right"]
 225 
 226     def time_operation(self, index_structure, dtype, method):
 227         getattr(self.left, method)(self.right)
