  21     def _args_adjust(self):
  22         if is_integer(self.bins):
  23             # create common bin edge
  24             values = self.data._convert(datetime=True)._get_numeric_data()
  25             values = np.ravel(values)
  26             values = values[~isna(values)]
  27 
  28             _, self.bins = np.histogram(
  29                 values, bins=self.bins, range=self.kwds.get("range", None)

  41         bins=None,
  42         bottom=0,

  58     def _make_plot(self):
  59         colors = self._get_colors()
  60         stacking_id = self._get_stacking_id()
  61 
  62         for i, (label, y) in enumerate(self._iter_data()):
  63             ax = self._get_ax(i)
  64

  76             # We allow weights to be a multi-dimensional array, e.g. a (10, 2) array,

 119     def _get_ind(self, y):
 120         if self.ind is None:
 121             # np.nanmax() and np.nanmin() ignores the missing values
 122             sample_range = np.nanmax(y) - np.nanmin(y)
 123             ind = np.linspace(
 124                 np.nanmin(y) - 0.5 * sample_range,
 125                 np.nanmax(y) + 0.5 * sample_range,
 126                 1000,
 127             )
 128         elif is_integer(self.ind):
 129             sample_range = np.nanmax(y) - np.nanmin(y)
 130             ind = np.linspace(
 131                 np.nanmin(y) - 0.5 * sample_range,
 132                 np.nanmax(y) + 0.5 * sample_range,
 133                 self.ind,
 134             )
 135         else:
 136             ind = self.ind
 137         return ind

 139     @classmethod
 140     def _plot(
 141         cls,
 142         ax,
 143         y,
 144         style=None,
 145         bw_method=None,
 146         ind=None,
 147         column_num=None,
 148         stacking_id=None,
 149         **kwds,
 150     ):
 151         from scipy.stats import gaussian_kde
 152 
 153         y = remove_na_arraylike(y)
 154         gkde = gaussian_kde(y, bw_method=bw_method)
 155 
 156         y = gkde.evaluate(ind)
 157         lines = MPLPlot._plot(ax, ind, y, style=style, **kwds)
 158         return lines

 161         kwds["bw_method"] = self.bw_method
 162         kwds["ind"] = self._get_ind(y)
 163         return kwds

 169 def _grouped_plot(
 170     plotf,
 171     data,
 172     column=None,
 173     by=None,
 174     numeric_only=True,
 175     figsize=None,
 176     sharex=True,
 177     sharey=True,
 178     layout=None,
 179     rot=0,
 180     ax=None,
 181     **kwargs,
 182 ):
 183 
 184     if figsize == "default":
 185         # allowed to specify mpl default with 'default'
 186         raise ValueError(
 187             "figsize='default' is no longer supported. "
 188             "Specify figure size by tuple instead"
 189         )
 190 
 191     grouped = data.groupby(by)
 192     if column is not None:
 193         grouped = grouped[column]
 194 
 195     naxes = len(grouped)
 196     fig, axes = _subplots(
 197         naxes=naxes, figsize=figsize, sharex=sharex, sharey=sharey, ax=ax, layout=layout
 198     )
 199 
 200     _axes = _flatten(axes)
 201 
 202     for i, (key, group) in enumerate(grouped):
 203         ax = _axes[i]
 204         if numeric_only and isinstance(group, ABCDataFrame):
 205             group = group._get_numeric_data()
 206         plotf(group, ax, **kwargs)
 207         ax.set_title(pprint_thing(key))
 208 
 209     return fig, axes

 213     data,

 223     grid=True,
 224     xlabelsize=None,
 225     xrot=None,
 226     ylabelsize=None,
 227     yrot=None,
 228     **kwargs,
 229 ):
 230     """
 231     Grouped histogram
 232 
 233     Parameters
 234     ----------
 235     data : Series/DataFrame
 236     column : object, optional
 237     by : object, optional
 238     ax : axes, optional
 239     bins : int, default 50
 240     figsize : tuple, optional
 241     layout : optional
 242     sharex : bool, default False
 243     sharey : bool, default False

 246     kwargs : dict, keyword arguments passed to matplotlib.Axes.hist
 247

 249     -------
 250     collection of Matplotlib Axes
 251     """
 252 
 253     def plot_group(group, ax):
 254         ax.hist(group.dropna().values, bins=bins, **kwargs)
 255 
 256     if xrot is None:
 257         xrot = rot

 260         plot_group,

 282 def hist_series(
 283     self,
 284     by=None,
 285     ax=None,
 286     grid=True,
 287     xlabelsize=None,
 288     xrot=None,
 289     ylabelsize=None,
 290     yrot=None,
 291     figsize=None,
 292     bins=10,
 293     **kwds,
 294 ):
 295     import matplotlib.pyplot as plt
 296 
 297     if by is None:
 298         if kwds.get("layout", None) is not None:
 299             raise ValueError("The 'layout' keyword is not supported when 'by' is None")
 300         # hack until the plotting interface is a bit more unified
 301         fig = kwds.pop(
 302             "figure", plt.gcf() if plt.get_fignums() else plt.figure(figsize=figsize)
 303         )
 304         if figsize is not None and tuple(figsize) != tuple(fig.get_size_inches()):
 305             fig.set_size_inches(*figsize, forward=True)
 306         if ax is None:
 307             ax = fig.gca()
 308         elif ax.get_figure() != fig:
 309             raise AssertionError("passed axis not bound to passed figure")
 310         values = self.dropna().values
 311 
 312         ax.hist(values, bins=bins, **kwds)
 313         ax.grid(grid)
 314         axes = np.array([ax])
 315 
 316         _set_ticks_props(
 317             axes, xlabelsize=xlabelsize, xrot=xrot, ylabelsize=ylabelsize, yrot=yrot
 318         )
 319 
 320     else:
 321         if "figure" in kwds:
 322             raise ValueError(
 323                 "Cannot pass 'figure' when using the "
 324                 "'by' argument, since a new 'Figure' instance will be created"
 325             )
 326         axes = _grouped_hist(
 327             self,
 328             by=by,
 329             ax=ax,
 330             grid=grid,
 331             figsize=figsize,
 332             bins=bins,
 333             xlabelsize=xlabelsize,
 334             xrot=xrot,
 335             ylabelsize=ylabelsize,
 336             yrot=yrot,
 337             **kwds,
 338         )
 339 
 340     if hasattr(axes, "ndim"):
 341         if axes.ndim == 1 and len(axes) == 1:
 342             return axes[0]
 343     return axes

 368             ax=ax,

 383     if column is not None:
 384         if not isinstance(column, (list, np.ndarray, ABCIndexClass)):
 385             column = [column]
