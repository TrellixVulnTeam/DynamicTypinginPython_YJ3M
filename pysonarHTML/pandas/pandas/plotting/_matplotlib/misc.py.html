  14 def scatter_matrix(
  15     frame,
  16     alpha=0.5,
  17     figsize=None,
  18     ax=None,
  19     grid=False,
  20     diagonal="hist",
  21     marker=".",
  22     density_kwds=None,
  23     hist_kwds=None,
  24     range_padding=0.05,
  25     **kwds,
  26 ):
  27     df = frame._get_numeric_data()
  28     n = df.columns.size
  29     naxes = n * n
  30     fig, axes = _subplots(naxes=naxes, figsize=figsize, ax=ax, squeeze=False)
  31 
  32     # no gaps between subplots
  33     fig.subplots_adjust(wspace=0, hspace=0)
  34 
  35     mask = notna(df)
  36 
  37     marker = _get_marker_compat(marker)
  38 
  39     hist_kwds = hist_kwds or {}
  40     density_kwds = density_kwds or {}
  41 
  42     # GH 14855
  43     kwds.setdefault("edgecolors", "none")
  44 
  45     boundaries_list = []
  46     for a in df.columns:
  47         values = df[a].values[mask[a].values]
  48         rmin_, rmax_ = np.min(values), np.max(values)
  49         rdelta_ext = (rmax_ - rmin_) * range_padding / 2.0
  50         boundaries_list.append((rmin_ - rdelta_ext, rmax_ + rdelta_ext))
  51 
  52     for i, a in enumerate(df.columns):
  53         for j, b in enumerate(df.columns):
  54             ax = axes[i, j]
  55 
  56             if i == j:
  57                 values = df[a].values[mask[a].values]
  58 
  59                 # Deal with the diagonal by drawing a histogram there.
  60                 if diagonal == "hist":
  61                     ax.hist(values, **hist_kwds)
  62 
  63                 elif diagonal in ("kde", "density"):
  64                     from scipy.stats import gaussian_kde
  65 
  66                     y = values
  67                     gkde = gaussian_kde(y)
  68                     ind = np.linspace(y.min(), y.max(), 1000)
  69                     ax.plot(ind, gkde.evaluate(ind), **density_kwds)
  70 
  71                 ax.set_xlim(boundaries_list[i])
  72 
  73             else:
  74                 common = (mask[a] & mask[b]).values
  75 
  76                 ax.scatter(
  77                     df[b][common], df[a][common], marker=marker, alpha=alpha, **kwds
  78                 )
  79 
  80                 ax.set_xlim(boundaries_list[j])
  81                 ax.set_ylim(boundaries_list[i])
  82 
  83             ax.set_xlabel(b)
  84             ax.set_ylabel(a)
  85 
  86             if j != 0:
  87                 ax.yaxis.set_visible(False)
  88             if i != n - 1:
  89                 ax.xaxis.set_visible(False)
  90 
  91     if len(df.columns) > 1:
  92         lim1 = boundaries_list[0]
  93         locs = axes[0][1].yaxis.get_majorticklocs()
  94         locs = locs[(lim1[0] <= locs) & (locs <= lim1[1])]
  95         adj = (locs - lim1[0]) / (lim1[1] - lim1[0])
  96 
  97         lim0 = axes[0][0].get_ylim()
  98         adj = adj * (lim0[1] - lim0[0]) + lim0[0]
  99         axes[0][0].yaxis.set_ticks(adj)
 100 
 101         if np.all(locs == locs.astype(int)):
 102             # if all ticks are int
 103             locs = locs.astype(int)
 104         axes[0][0].yaxis.set_ticklabels(locs)
 105 
 106     _set_ticks_props(axes, xlabelsize=8, xrot=90, ylabelsize=8, yrot=0)
 107 
 108     return axes

 117 def radviz(frame, class_column, ax=None, color=None, colormap=None, **kwds):
 118     import matplotlib.pyplot as plt
 119 
 120     def normalize(series):
 121         a = min(series)
 122         b = max(series)
 123         return (series - a) / (b - a)
 124 
 125     n = len(frame)
 126     classes = frame[class_column].drop_duplicates()
 127     class_col = frame[class_column]
 128     df = frame.drop(class_column, axis=1).apply(normalize)
 129 
 130     if ax is None:
 131         ax = plt.gca(xlim=[-1, 1], ylim=[-1, 1])
 132 
 133     to_plot = {}
 134     colors = _get_standard_colors(
 135         num_colors=len(classes), colormap=colormap, color_type="random", color=color
 136     )
 137 
 138     for kls in classes:
 139         to_plot[kls] = [[], []]
 140 
 141     m = len(frame.columns) - 1
 142     s = np.array(
 143         [
 144             (np.cos(t), np.sin(t))
 145             for t in [2.0 * np.pi * (i / float(m)) for i in range(m)]
 146         ]
 147     )
 148 
 149     for i in range(n):
 150         row = df.iloc[i].values
 151         row_ = np.repeat(np.expand_dims(row, axis=1), 2, axis=1)
 152         y = (s * row_).sum(axis=0) / row.sum()
 153         kls = class_col.iat[i]
 154         to_plot[kls][0].append(y[0])
 155         to_plot[kls][1].append(y[1])
 156 
 157     for i, kls in enumerate(classes):
 158         ax.scatter(
 159             to_plot[kls][0],
 160             to_plot[kls][1],
 161             color=colors[i],
 162             label=pprint_thing(kls),
 163             **kwds,
 164         )
 165     ax.legend()
 166 
 167     ax.add_patch(patches.Circle((0.0, 0.0), radius=1.0, facecolor="none"))
 168 
 169     for xy, name in zip(s, df.columns):
 170 
 171         ax.add_patch(patches.Circle(xy, radius=0.025, facecolor="gray"))
 172 
 173         if xy[0] < 0.0 and xy[1] < 0.0:
 174             ax.text(
 175                 xy[0] - 0.025, xy[1] - 0.025, name, ha="right", va="top", size="small"
 176             )
 177         elif xy[0] < 0.0 and xy[1] >= 0.0:
 178             ax.text(
 179                 xy[0] - 0.025,
 180                 xy[1] + 0.025,
 181                 name,
 182                 ha="right",
 183                 va="bottom",
 184                 size="small",
 185             )
 186         elif xy[0] >= 0.0 and xy[1] < 0.0:
 187             ax.text(
 188                 xy[0] + 0.025, xy[1] - 0.025, name, ha="left", va="top", size="small"
 189             )
 190         elif xy[0] >= 0.0 and xy[1] >= 0.0:
 191             ax.text(
 192                 xy[0] + 0.025, xy[1] + 0.025, name, ha="left", va="bottom", size="small"
 193             )
 194 
 195     ax.axis("equal")
 196     return ax

 199 def andrews_curves(
 200     frame, class_column, ax=None, samples=200, color=None, colormap=None, **kwds
 201 ):
 202     import matplotlib.pyplot as plt
 203 
 204     def function(amplitudes):
 205         def f(t):
 206             x1 = amplitudes[0]
 207             result = x1 / np.sqrt(2.0)
 208 
 209             # Take the rest of the coefficients and resize them
 210             # appropriately. Take a copy of amplitudes as otherwise numpy
 211             # deletes the element from amplitudes itself.
 212             coeffs = np.delete(np.copy(amplitudes), 0)
 213             coeffs.resize(int((coeffs.size + 1) / 2), 2)
 214 
 215             # Generate the harmonics and arguments for the sin and cos
 216             # functions.
 217             harmonics = np.arange(0, coeffs.shape[0]) + 1
 218             trig_args = np.outer(harmonics, t)
 219 
 220             result += np.sum(
 221                 coeffs[:, 0, np.newaxis] * np.sin(trig_args)
 222                 + coeffs[:, 1, np.newaxis] * np.cos(trig_args),
 223                 axis=0,
 224             )
 225             return result
 226 
 227         return f
 228 
 229     n = len(frame)
 230     class_col = frame[class_column]
 231     classes = frame[class_column].drop_duplicates()
 232     df = frame.drop(class_column, axis=1)
 233     t = np.linspace(-np.pi, np.pi, samples)
 234     used_legends = set()
 235 
 236     color_values = _get_standard_colors(
 237         num_colors=len(classes), colormap=colormap, color_type="random", color=color
 238     )
 239     colors = dict(zip(classes, color_values))
 240     if ax is None:
 241         ax = plt.gca(xlim=(-np.pi, np.pi))
 242     for i in range(n):
 243         row = df.iloc[i].values
 244         f = function(row)
 245         y = f(t)
 246         kls = class_col.iat[i]
 247         label = pprint_thing(kls)
 248         if label not in used_legends:
 249             used_legends.add(label)
 250             ax.plot(t, y, color=colors[kls], label=label, **kwds)
 251         else:
 252             ax.plot(t, y, color=colors[kls], **kwds)
 253 
 254     ax.legend(loc="upper right")
 255     ax.grid()
 256     return ax

 259 def bootstrap_plot(series, fig=None, size=50, samples=500, **kwds):
 260 
 261     import matplotlib.pyplot as plt
 262 
 263     # TODO: is the failure mentioned below still relevant?
 264     # random.sample(ndarray, int) fails on python 3.3, sigh
 265     data = list(series.values)
 266     samplings = [random.sample(data, size) for _ in range(samples)]
 267 
 268     means = np.array([np.mean(sampling) for sampling in samplings])
 269     medians = np.array([np.median(sampling) for sampling in samplings])
 270     midranges = np.array(
 271         [(min(sampling) + max(sampling)) * 0.5 for sampling in samplings]
 272     )
 273     if fig is None:
 274         fig = plt.figure()
 275     x = list(range(samples))
 276     axes = []
 277     ax1 = fig.add_subplot(2, 3, 1)
 278     ax1.set_xlabel("Sample")
 279     axes.append(ax1)
 280     ax1.plot(x, means, **kwds)
 281     ax2 = fig.add_subplot(2, 3, 2)
 282     ax2.set_xlabel("Sample")
 283     axes.append(ax2)
 284     ax2.plot(x, medians, **kwds)
 285     ax3 = fig.add_subplot(2, 3, 3)
 286     ax3.set_xlabel("Sample")
 287     axes.append(ax3)
 288     ax3.plot(x, midranges, **kwds)
 289     ax4 = fig.add_subplot(2, 3, 4)
 290     ax4.set_xlabel("Mean")
 291     axes.append(ax4)
 292     ax4.hist(means, **kwds)
 293     ax5 = fig.add_subplot(2, 3, 5)
 294     ax5.set_xlabel("Median")
 295     axes.append(ax5)
 296     ax5.hist(medians, **kwds)
 297     ax6 = fig.add_subplot(2, 3, 6)
 298     ax6.set_xlabel("Midrange")
 299     axes.append(ax6)
 300     ax6.hist(midranges, **kwds)
 301     for axis in axes:
 302         plt.setp(axis.get_xticklabels(), fontsize=8)
 303         plt.setp(axis.get_yticklabels(), fontsize=8)
 304     return fig

 307 def parallel_coordinates(
 308     frame,
 309     class_column,
 310     cols=None,
 311     ax=None,
 312     color=None,
 313     use_columns=False,
 314     xticks=None,
 315     colormap=None,
 316     axvlines=True,
 317     axvlines_kwds=None,
 318     sort_labels=False,
 319     **kwds,
 320 ):
 321     import matplotlib.pyplot as plt
 322 
 323     if axvlines_kwds is None:
 324         axvlines_kwds = {"linewidth": 1, "color": "black"}
 325 
 326     n = len(frame)
 327     classes = frame[class_column].drop_duplicates()
 328     class_col = frame[class_column]
 329 
 330     if cols is None:
 331         df = frame.drop(class_column, axis=1)
 332     else:
 333         df = frame[cols]
 334 
 335     used_legends = set()
 336 
 337     ncols = len(df.columns)
 338 
 339     # determine values to use for xticks
 340     if use_columns is True:
 341         if not np.all(np.isreal(list(df.columns))):
 342             raise ValueError("Columns must be numeric to be used as xticks")
 343         x = df.columns
 344     elif xticks is not None:
 345         if not np.all(np.isreal(xticks)):
 346             raise ValueError("xticks specified must be numeric")
 347         elif len(xticks) != ncols:
 348             raise ValueError("Length of xticks must match number of columns")
 349         x = xticks
 350     else:
 351         x = list(range(ncols))
 352 
 353     if ax is None:
 354         ax = plt.gca()
 355 
 356     color_values = _get_standard_colors(
 357         num_colors=len(classes), colormap=colormap, color_type="random", color=color
 358     )
 359 
 360     if sort_labels:
 361         classes = sorted(classes)
 362         color_values = sorted(color_values)
 363     colors = dict(zip(classes, color_values))
 364 
 365     for i in range(n):
 366         y = df.iloc[i].values
 367         kls = class_col.iat[i]
 368         label = pprint_thing(kls)
 369         if label not in used_legends:
 370             used_legends.add(label)
 371             ax.plot(x, y, color=colors[kls], label=label, **kwds)
 372         else:
 373             ax.plot(x, y, color=colors[kls], **kwds)
 374 
 375     if axvlines:
 376         for i in x:
 377             ax.axvline(i, **axvlines_kwds)
 378 
 379     ax.set_xticks(x)
 380     ax.set_xticklabels(df.columns)
 381     ax.set_xlim(x[0], x[-1])
 382     ax.legend(loc="upper right")
 383     ax.grid()
 384     return ax

 387 def lag_plot(series, lag=1, ax=None, **kwds):
 388     # workaround because `c='b'` is hardcoded in matplotlib's scatter method
 389     import matplotlib.pyplot as plt
 390 
 391     kwds.setdefault("c", plt.rcParams["patch.facecolor"])
 392 
 393     data = series.values
 394     y1 = data[:-lag]
 395     y2 = data[lag:]
 396     if ax is None:
 397         ax = plt.gca()
 398     ax.set_xlabel("y(t)")
 399     ax.set_ylabel(f"y(t + {lag})")
 400     ax.scatter(y1, y2, **kwds)
 401     return ax

 404 def autocorrelation_plot(series, ax=None, **kwds):
 405     import matplotlib.pyplot as plt
 406 
 407     n = len(series)
 408     data = np.asarray(series)
 409     if ax is None:
 410         ax = plt.gca(xlim=(1, n), ylim=(-1.0, 1.0))
 411     mean = np.mean(data)
 412     c0 = np.sum((data - mean) ** 2) / float(n)
 413 
 414     def r(h):
 415         return ((data[: n - h] - mean) * (data[h:] - mean)).sum() / float(n) / c0
 416 
 417     x = np.arange(n) + 1
 418     y = [r(loc) for loc in x]
 419     z95 = 1.959963984540054
 420     z99 = 2.5758293035489004
 421     ax.axhline(y=z99 / np.sqrt(n), linestyle="--", color="grey")
 422     ax.axhline(y=z95 / np.sqrt(n), color="grey")
 423     ax.axhline(y=0.0, color="black")
 424     ax.axhline(y=-z95 / np.sqrt(n), color="grey")
 425     ax.axhline(y=-z99 / np.sqrt(n), linestyle="--", color="grey")
 426     ax.set_xlabel("Lag")
 427     ax.set_ylabel("Autocorrelation")
 428     ax.plot(x, y, **kwds)
 429     if "label" in kwds:
 430         ax.legend()
 431     ax.grid()
 432     return ax
