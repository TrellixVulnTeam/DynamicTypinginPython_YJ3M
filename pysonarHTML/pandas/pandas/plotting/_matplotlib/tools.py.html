  46 def _get_layout(nplots, layout=None, layout_type="box"):
  47     if layout is not None:
  48         if not isinstance(layout, (tuple, list)) or len(layout) != 2:
  49             raise ValueError("Layout must be a tuple of (rows, columns)")
  50 
  51         nrows, ncols = layout
  52 
  53         # Python 2 compat
  54         ceil_ = lambda x: int(ceil(x))
  55         if nrows == -1 and ncols > 0:
  56             layout = nrows, ncols = (ceil_(float(nplots) / ncols), ncols)
  57         elif ncols == -1 and nrows > 0:
  58             layout = nrows, ncols = (nrows, ceil_(float(nplots) / nrows))
  59         elif ncols <= 0 and nrows <= 0:
  60             msg = "At least one dimension of layout must be positive"
  61             raise ValueError(msg)
  62 
  63         if nrows * ncols < nplots:
  64             raise ValueError(
  65                 f"Layout of {nrows}x{ncols} must be larger than required size {nplots}"
  66             )
  67 
  68         return layout
  69 
  70     if layout_type == "single":
  71         return (1, 1)
  72     elif layout_type == "horizontal":
  73         return (1, nplots)
  74     elif layout_type == "vertical":
  75         return (nplots, 1)
  76 
  77     layouts = {1: (1, 1), 2: (1, 2), 3: (2, 2), 4: (2, 2)}
  78     try:
  79         return layouts[nplots]
  80     except KeyError:
  81         k = 1
  82         while k ** 2 < nplots:
  83             k += 1
  84 
  85         if (k - 1) * k >= nplots:
  86             return k, (k - 1)
  87         else:
  88             return k, k

  94 def _subplots(
  95     naxes=None,
  96     sharex=False,
  97     sharey=False,
  98     squeeze=True,
  99     subplot_kw=None,
 100     ax=None,
 101     layout=None,
 102     layout_type="box",
 103     **fig_kw,
 104 ):
 105     """
 106     Create a figure with a set of subplots already made.
 107 
 108     This utility wrapper makes it convenient to create common layouts of
 109     subplots, including the enclosing figure object, in a single call.
 110 
 111     Parameters
 112     ----------
 113     naxes : int
 114       Number of required axes. Exceeded axes are set invisible. Default is
 115       nrows * ncols.
 116 
 117     sharex : bool
 118       If True, the X axis will be shared amongst all subplots.
 119 
 120     sharey : bool
 121       If True, the Y axis will be shared amongst all subplots.
 122 
 123     squeeze : bool
 124 
 125       If True, extra dimensions are squeezed out from the returned axis object:
 126         - if only one subplot is constructed (nrows=ncols=1), the resulting
 127         single Axis object is returned as a scalar.
 128         - for Nx1 or 1xN subplots, the returned object is a 1-d numpy object
 129         array of Axis objects are returned as numpy 1-d arrays.
 130         - for NxM subplots with N>1 and M>1 are returned as a 2d array.
 131 
 132       If False, no squeezing is done: the returned axis object is always
 133       a 2-d array containing Axis instances, even if it ends up being 1x1.
 134 
 135     subplot_kw : dict
 136       Dict with keywords passed to the add_subplot() call used to create each
 137       subplots.
 138 
 139     ax : Matplotlib axis object, optional
 140 
 141     layout : tuple
 142       Number of rows and columns of the subplot grid.
 143       If not specified, calculated from naxes and layout_type
 144 
 145     layout_type : {'box', 'horizontal', 'vertical'}, default 'box'
 146       Specify how to layout the subplot grid.
 147 
 148     fig_kw : Other keyword arguments to be passed to the figure() call.
 149         Note that all keywords not recognized above will be
 150         automatically included here.
 151 
 152     Returns
 153     -------
 154     fig, ax : tuple
 155       - fig is the Matplotlib Figure object
 156       - ax can be either a single axis object or an array of axis objects if
 157       more than one subplot was created.  The dimensions of the resulting array
 158       can be controlled with the squeeze keyword, see above.
 159 
 160     Examples
 161     --------
 162     x = np.linspace(0, 2*np.pi, 400)
 163     y = np.sin(x**2)
 164 
 165     # Just a figure and one subplot
 166     f, ax = plt.subplots()
 167     ax.plot(x, y)
 168     ax.set_title('Simple plot')
 169 
 170     # Two subplots, unpack the output array immediately
 171     f, (ax1, ax2) = plt.subplots(1, 2, sharey=True)
 172     ax1.plot(x, y)
 173     ax1.set_title('Sharing Y axis')
 174     ax2.scatter(x, y)
 175 
 176     # Four polar axes
 177     plt.subplots(2, 2, subplot_kw=dict(polar=True))
 178     """
 179     import matplotlib.pyplot as plt
 180 
 181     if subplot_kw is None:
 182         subplot_kw = {}
 183 
 184     if ax is None:
 185         fig = plt.figure(**fig_kw)
 186     else:
 187         if is_list_like(ax):
 188             ax = _flatten(ax)
 189             if layout is not None:
 190                 warnings.warn(
 191                     "When passing multiple axes, layout keyword is ignored", UserWarning
 192                 )
 193             if sharex or sharey:
 194                 warnings.warn(
 195                     "When passing multiple axes, sharex and sharey "
 196                     "are ignored. These settings must be specified when creating axes",
 197                     UserWarning,
 198                     stacklevel=4,
 199                 )
 200             if len(ax) == naxes:
 201                 fig = ax[0].get_figure()
 202                 return fig, ax
 203             else:
 204                 raise ValueError(
 205                     f"The number of passed axes must be {naxes}, the "
 206                     "same as the output plot"
 207                 )
 208 
 209         fig = ax.get_figure()
 210         # if ax is passed and a number of subplots is 1, return ax as it is
 211         if naxes == 1:
 212             if squeeze:
 213                 return fig, ax
 214             else:
 215                 return fig, _flatten(ax)
 216         else:
 217             warnings.warn(
 218                 "To output multiple subplots, the figure containing "
 219                 "the passed axes is being cleared",
 220                 UserWarning,
 221                 stacklevel=4,
 222             )
 223             fig.clear()
 224 
 225     nrows, ncols = _get_layout(naxes, layout=layout, layout_type=layout_type)
 226     nplots = nrows * ncols
 227 
 228     # Create empty object array to hold all axes.  It's easiest to make it 1-d
 229     # so we can just append subplots upon creation, and then
 230     axarr = np.empty(nplots, dtype=object)
 231 
 232     # Create first subplot separately, so we can share it if requested
 233     ax0 = fig.add_subplot(nrows, ncols, 1, **subplot_kw)
 234 
 235     if sharex:
 236         subplot_kw["sharex"] = ax0
 237     if sharey:
 238         subplot_kw["sharey"] = ax0
 239     axarr[0] = ax0
 240 
 241     # Note off-by-one counting because add_subplot uses the MATLAB 1-based
 242     # convention.
 243     for i in range(1, nplots):
 244         kwds = subplot_kw.copy()
 245         # Set sharex and sharey to None for blank/dummy axes, these can
 246         # interfere with proper axis limits on the visible axes if
 247         # they share axes e.g. issue #7528
 248         if i >= naxes:
 249             kwds["sharex"] = None
 250             kwds["sharey"] = None
 251         ax = fig.add_subplot(nrows, ncols, i + 1, **kwds)
 252         axarr[i] = ax
 253 
 254     if naxes != nplots:
 255         for ax in axarr[naxes:]:
 256             ax.set_visible(False)
 257 
 258     _handle_shared_axes(axarr, nplots, naxes, nrows, ncols, sharex, sharey)
 259 
 260     if squeeze:
 261         # Reshape the array to have the final desired dimension (nrow,ncol),
 262         # though discarding unneeded dimensions that equal 1.  If we only have
 263         # one subplot, just return it instead of a 1-element array.
 264         if nplots == 1:
 265             axes = axarr[0]
 266         else:
 267             axes = axarr.reshape(nrows, ncols).squeeze()
 268     else:
 269         # returned axis array will be always 2-d, even if nrows=ncols=1
 270         axes = axarr.reshape(nrows, ncols)
 271 
 272     return fig, axes

 275 def _remove_labels_from_axis(axis):
 276     for t in axis.get_majorticklabels():
 277         t.set_visible(False)
 278 
 279     # set_visible will not be effective if
 280     # minor axis has NullLocator and NullFormatter (default)
 281     if isinstance(axis.get_minor_locator(), ticker.NullLocator):
 282         axis.set_minor_locator(ticker.AutoLocator())
 283     if isinstance(axis.get_minor_formatter(), ticker.NullFormatter):
 284         axis.set_minor_formatter(ticker.FormatStrFormatter(""))
 285     for t in axis.get_minorticklabels():
 286         t.set_visible(False)
 287 
 288     axis.get_label().set_visible(False)

 291 def _handle_shared_axes(axarr, nplots, naxes, nrows, ncols, sharex, sharey):
 292     if nplots > 1:
 293         if compat._mpl_ge_3_2_0():
 294             row_num = lambda x: x.get_subplotspec().rowspan.start
 295             col_num = lambda x: x.get_subplotspec().colspan.start
 296         else:
 297             row_num = lambda x: x.rowNum
 298             col_num = lambda x: x.colNum
 299 
 300         if nrows > 1:
 301             try:
 302                 # first find out the ax layout,
 303                 # so that we can correctly handle 'gaps"
 304                 layout = np.zeros((nrows + 1, ncols + 1), dtype=np.bool)
 305                 for ax in axarr:
 306                     layout[row_num(ax), col_num(ax)] = ax.get_visible()
 307 
 308                 for ax in axarr:
 309                     # only the last row of subplots should get x labels -> all
 310                     # other off layout handles the case that the subplot is
 311                     # the last in the column, because below is no subplot/gap.
 312                     if not layout[row_num(ax) + 1, col_num(ax)]:
 313                         continue
 314                     if sharex or len(ax.get_shared_x_axes().get_siblings(ax)) > 1:
 315                         _remove_labels_from_axis(ax.xaxis)
 316 
 317             except IndexError:
 318                 # if gridspec is used, ax.rowNum and ax.colNum may different
 319                 # from layout shape. in this case, use last_row logic
 320                 for ax in axarr:
 321                     if ax.is_last_row():
 322                         continue
 323                     if sharex or len(ax.get_shared_x_axes().get_siblings(ax)) > 1:
 324                         _remove_labels_from_axis(ax.xaxis)
 325 
 326         if ncols > 1:
 327             for ax in axarr:
 328                 # only the first column should get y labels -> set all other to
 329                 # off as we only have labels in the first column and we always
 330                 # have a subplot there, we can skip the layout test
 331                 if ax.is_first_col():
 332                     continue
 333                 if sharey or len(ax.get_shared_y_axes().get_siblings(ax)) > 1:
 334                     _remove_labels_from_axis(ax.yaxis)

 337 def _flatten(axes):
 338     if not is_list_like(axes):
 339         return np.array([axes])
 340     elif isinstance(axes, (np.ndarray, ABCIndexClass)):
 341         return axes.ravel()
 342     return np.array(axes)

 345 def _set_ticks_props(axes, xlabelsize=None, xrot=None, ylabelsize=None, yrot=None):
 346     import matplotlib.pyplot as plt
 347 
 348     for ax in _flatten(axes):
 349         if xlabelsize is not None:
 350             plt.setp(ax.get_xticklabels(), fontsize=xlabelsize)
 351         if xrot is not None:
 352             plt.setp(ax.get_xticklabels(), rotation=xrot)
 353         if ylabelsize is not None:
 354             plt.setp(ax.get_yticklabels(), fontsize=ylabelsize)
 355         if yrot is not None:
 356             plt.setp(ax.get_yticklabels(), rotation=yrot)
 357     return axes

 360 def _get_all_lines(ax):
 361     lines = ax.get_lines()
 362 
 363     if hasattr(ax, "right_ax"):
 364         lines += ax.right_ax.get_lines()
 365 
 366     if hasattr(ax, "left_ax"):
 367         lines += ax.left_ax.get_lines()
 368 
 369     return lines

 372 def _get_xlim(lines):
 373     left, right = np.inf, -np.inf
 374     for l in lines:
 375         x = l.get_xdata(orig=False)
 376         left = min(np.nanmin(x), left)
 377         right = max(np.nanmax(x), right)
 378     return left, right
