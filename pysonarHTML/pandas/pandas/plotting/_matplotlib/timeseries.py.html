  35 def _maybe_resample(series: "Series", ax, kwargs):
  36     # resample against axes freq if necessary
  37     freq, ax_freq = _get_freq(ax, series)
  38 
  39     if freq is None:  # pragma: no cover
  40         raise ValueError("Cannot use dynamic axis without frequency info")
  41 
  42     # Convert DatetimeIndex to PeriodIndex
  43     if isinstance(series.index, ABCDatetimeIndex):
  44         series = series.to_period(freq=freq)
  45 
  46     if ax_freq is not None and freq != ax_freq:
  47         if is_superperiod(freq, ax_freq):  # upsample input
  48             series = series.copy()
  49             series.index = series.index.asfreq(ax_freq, how="s")  # type: ignore
  50             freq = ax_freq
  51         elif _is_sup(freq, ax_freq):  # one is weekly
  52             how = kwargs.pop("how", "last")
  53             series = getattr(series.resample("D"), how)().dropna()
  54             series = getattr(series.resample(ax_freq), how)().dropna()
  55             freq = ax_freq
  56         elif is_subperiod(freq, ax_freq) or _is_sub(freq, ax_freq):
  57             _upsample_others(ax, freq, kwargs)
  58         else:  # pragma: no cover
  59             raise ValueError("Incompatible frequency conversion")
  60     return freq, series

  63 def _is_sub(f1, f2):
  64     return (f1.startswith("W") and is_subperiod("D", f2)) or (
  65         f2.startswith("W") and is_subperiod(f1, "D")
  66     )
  67 
  68 
  69 def _is_sup(f1, f2):
  70     return (f1.startswith("W") and is_superperiod("D", f2)) or (
  71         f2.startswith("W") and is_superperiod(f1, "D")
  72     )

  75 def _upsample_others(ax, freq, kwargs):
  76     legend = ax.get_legend()
  77     lines, labels = _replot_ax(ax, freq, kwargs)
  78     _replot_ax(ax, freq, kwargs)
  79 
  80     other_ax = None
  81     if hasattr(ax, "left_ax"):
  82         other_ax = ax.left_ax
  83     if hasattr(ax, "right_ax"):
  84         other_ax = ax.right_ax
  85 
  86     if other_ax is not None:
  87         rlines, rlabels = _replot_ax(other_ax, freq, kwargs)
  88         lines.extend(rlines)
  89         labels.extend(rlabels)
  90 
  91     if legend is not None and kwargs.get("legend", True) and len(lines) > 0:
  92         title = legend.get_title().get_text()
  93         if title == "None":
  94             title = None
  95         ax.legend(lines, labels, loc="best", title=title)

  98 def _replot_ax(ax, freq, kwargs):
  99     data = getattr(ax, "_plot_data", None)
 100 
 101     # clear current axes and data
 102     ax._plot_data = []
 103     ax.clear()
 104 
 105     _decorate_axes(ax, freq, kwargs)
 106 
 107     lines = []
 108     labels = []
 109     if data is not None:
 110         for series, plotf, kwds in data:
 111             series = series.copy()
 112             idx = series.index.asfreq(freq, how="S")
 113             series.index = idx
 114             ax._plot_data.append((series, plotf, kwds))
 115 
 116             # for tsplot
 117             if isinstance(plotf, str):
 118                 from pandas.plotting._matplotlib import PLOT_CLASSES
 119 
 120                 plotf = PLOT_CLASSES[plotf]._plot
 121 
 122             lines.append(plotf(ax, series.index._mpl_repr(), series.values, **kwds)[0])
 123             labels.append(pprint_thing(series.name))
 124 
 125     return lines, labels

 128 def _decorate_axes(ax, freq, kwargs):
 129     """Initialize axes for time-series plotting"""
 130     if not hasattr(ax, "_plot_data"):
 131         ax._plot_data = []
 132 
 133     ax.freq = freq
 134     xaxis = ax.get_xaxis()
 135     xaxis.freq = freq
 136     if not hasattr(ax, "legendlabels"):
 137         ax.legendlabels = [kwargs.get("label", None)]
 138     else:
 139         ax.legendlabels.append(kwargs.get("label", None))
 140     ax.view_interval = None
 141     ax.date_axis_info = None

 144 def _get_ax_freq(ax):
 145     """
 146     Get the freq attribute of the ax object if set.
 147     Also checks shared axes (eg when using secondary yaxis, sharex=True
 148     or twinx)
 149     """
 150     ax_freq = getattr(ax, "freq", None)
 151     if ax_freq is None:
 152         # check for left/right ax in case of secondary yaxis
 153         if hasattr(ax, "left_ax"):
 154             ax_freq = getattr(ax.left_ax, "freq", None)
 155         elif hasattr(ax, "right_ax"):
 156             ax_freq = getattr(ax.right_ax, "freq", None)
 157     if ax_freq is None:
 158         # check if a shared ax (sharex/twinx) has already freq set
 159         shared_axes = ax.get_shared_x_axes().get_siblings(ax)
 160         if len(shared_axes) > 1:
 161             for shared_ax in shared_axes:
 162                 ax_freq = getattr(shared_ax, "freq", None)
 163                 if ax_freq is not None:
 164                     break
 165     return ax_freq

 168 def _get_period_alias(freq) -> Optional[str]:

 171     freq = get_period_alias(freqstr)
 172     return freq

 175 def _get_freq(ax, series: "Series"):
 176     # get frequency from data
 177     freq = getattr(series.index, "freq", None)
 178     if freq is None:
 179         freq = getattr(series.index, "inferred_freq", None)
 180         freq = to_offset(freq)
 181 
 182     ax_freq = _get_ax_freq(ax)
 183 
 184     # use axes freq if no data freq
 185     if freq is None:
 186         freq = ax_freq
 187 
 188     # get the period frequency
 189     freq = _get_period_alias(freq)
 190     return freq, ax_freq

 193 def _use_dynamic_x(ax, data: "FrameOrSeriesUnion") -> bool:
 194     freq = _get_index_freq(data.index)
 195     ax_freq = _get_ax_freq(ax)
 196 
 197     if freq is None:  # convert irregular if axes has freq info
 198         freq = ax_freq
 199     else:  # do not use tsplot if irregular was plotted first
 200         if (ax_freq is None) and (len(ax.get_lines()) > 0):
 201             return False
 202 
 203     if freq is None:
 204         return False
 205 
 206     freq = _get_period_alias(freq)
 207 
 208     if freq is None:
 209         return False
 210 
 211     # FIXME: hack this for 0.10.1, creating more technical debt...sigh
 212     if isinstance(data.index, ABCDatetimeIndex):
 213         base = to_offset(freq)._period_dtype_code
 214         x = data.index
 215         if base <= FreqGroup.FR_DAY:
 216             return x[:1].is_normalized
 217         return Period(x[0], freq).to_timestamp().tz_localize(x.tz) == x[0]
 218     return True

 221 def _get_index_freq(index: "Index") -> Optional[DateOffset]:
 222     freq = getattr(index, "freq", None)
 223     if freq is None:
 224         freq = getattr(index, "inferred_freq", None)
 225         if freq == "B":
 226             weekdays = np.unique(index.dayofweek)  # type: ignore
 227             if (5 in weekdays) or (6 in weekdays):
 228                 freq = None
 229 
 230     freq = to_offset(freq)
 231     return freq

 234 def _maybe_convert_index(ax, data):
 235     # tsplot converts automatically, but don't want to convert index
 236     # over and over for DataFrames
 237     if isinstance(data.index, (ABCDatetimeIndex, ABCPeriodIndex)):
 238         freq = data.index.freq
 239 
 240         if freq is None:
 241             # We only get here for DatetimeIndex
 242             freq = data.index.inferred_freq
 243             freq = to_offset(freq)
 244 
 245         if freq is None:
 246             freq = _get_ax_freq(ax)
 247 
 248         if freq is None:
 249             raise ValueError("Could not get frequency alias for plotting")
 250 
 251         freq = _get_period_alias(freq)
 252 
 253         if isinstance(data.index, ABCDatetimeIndex):
 254             data = data.tz_localize(None).to_period(freq=freq)
 255         elif isinstance(data.index, ABCPeriodIndex):
 256             data.index = data.index.asfreq(freq=freq)
 257     return data

 264 def _format_coord(freq, t, y):
 265     time_period = Period(ordinal=int(t), freq=freq)
 266     return f"t = {time_period}  y = {y:8f}"

 269 def format_dateaxis(subplot, freq, index):
 270     """
 271     Pretty-formats the date axis (x-axis).
 272 
 273     Major and minor ticks are automatically set for the frequency of the
 274     current underlying series.  As the dynamic mode is activated by
 275     default, changing the limits of the x axis will intelligently change
 276     the positions of the ticks.
 277     """
 278     from matplotlib import pylab
 279 
 280     # handle index specific formatting
 281     # Note: DatetimeIndex does not use this
 282     # interface. DatetimeIndex uses matplotlib.date directly
 283     if isinstance(index, ABCPeriodIndex):
 284 
 285         majlocator = TimeSeries_DateLocator(
 286             freq, dynamic_mode=True, minor_locator=False, plot_obj=subplot
 287         )
 288         minlocator = TimeSeries_DateLocator(
 289             freq, dynamic_mode=True, minor_locator=True, plot_obj=subplot
 290         )
 291         subplot.xaxis.set_major_locator(majlocator)
 292         subplot.xaxis.set_minor_locator(minlocator)
 293 
 294         majformatter = TimeSeries_DateFormatter(
 295             freq, dynamic_mode=True, minor_locator=False, plot_obj=subplot
 296         )
 297         minformatter = TimeSeries_DateFormatter(
 298             freq, dynamic_mode=True, minor_locator=True, plot_obj=subplot
 299         )
 300         subplot.xaxis.set_major_formatter(majformatter)
 301         subplot.xaxis.set_minor_formatter(minformatter)
 302 
 303         # x and y coord info
 304         subplot.format_coord = functools.partial(_format_coord, freq)
 305 
 306     elif isinstance(index, ABCTimedeltaIndex):
 307         subplot.xaxis.set_major_formatter(TimeSeries_TimedeltaFormatter())
 308     else:
 309         raise TypeError("index type not supported")
 310 
 311     pylab.draw_if_interactive()
