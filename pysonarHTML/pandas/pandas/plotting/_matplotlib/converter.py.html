  93 def register():
  94     pairs = get_pairs()
  95     for type_, cls in pairs:
  96         # Cache previous converter if present
  97         if type_ in units.registry and not isinstance(units.registry[type_], cls):
  98             previous = units.registry[type_]
  99             _mpl_units[type_] = previous
 100         # Replace with pandas converter
 101         units.registry[type_] = cls()
 102 
 103 
 104 def deregister():
 105     # Renamed in pandas.plotting.__init__
 106     for type_, cls in get_pairs():
 107         # We use type to catch our classes directly, no inheritance
 108         if type(units.registry.get(type_)) is cls:
 109             units.registry.pop(type_)
 110 
 111     # restore the old keys
 112     for unit, formatter in _mpl_units.items():
 113         if type(formatter) not in {DatetimeConverter, PeriodConverter, TimeConverter}:
 114             # make it idempotent by excluding ours.
 115             units.registry[unit] = formatter

 165     def __call__(self, x, pos=0) -> str:
 166         """
 167         Return the time of day as a formatted string.
 168 
 169         Parameters
 170         ----------
 171         x : float
 172             The time of day specified as seconds since 00:00 (midnight),
 173             with up to microsecond precision.
 174         pos
 175             Unused
 176 
 177         Returns
 178         -------
 179         str
 180             A string in HH:MM:SS.mmmuuu format. Microseconds,
 181             milliseconds and seconds are only displayed if non-zero.
 182         """
 183         fmt = "%H:%M:%S.%f"
 184         s = int(x)
 185         msus = int(round((x - s) * 1e6))
 186         ms = msus // 1000
 187         us = msus % 1000
 188         m, s = divmod(s, 60)
 189         h, m = divmod(m, 60)
 190         _, h = divmod(h, 24)
 191         if us != 0:
 192             return pydt.time(h, m, s, msus).strftime(fmt)
 193         elif ms != 0:
 194             return pydt.time(h, m, s, msus).strftime(fmt)[:-3]
 195         elif s != 0:
 196             return pydt.time(h, m, s).strftime("%H:%M:%S")
 197 
 198         return pydt.time(h, m).strftime("%H:%M")

 204 class PeriodConverter(dates.DateConverter):
 205     @staticmethod
 206     def convert(values, units, axis):
 207         if is_nested_list_like(values):
 208             values = [PeriodConverter._convert_1d(v, units, axis) for v in values]
 209         else:
 210             values = PeriodConverter._convert_1d(values, units, axis)
 211         return values
 212 
 213     @staticmethod
 214     def _convert_1d(values, units, axis):
 215         if not hasattr(axis, "freq"):
 216             raise TypeError("Axis must have `freq` set to convert to Periods")
 217         valid_types = (str, datetime, Period, pydt.date, pydt.time, np.datetime64)
 218         if isinstance(values, valid_types) or is_integer(values) or is_float(values):
 219             return get_datevalue(values, axis.freq)
 220         elif isinstance(values, PeriodIndex):
 221             return values.asfreq(axis.freq).asi8
 222         elif isinstance(values, Index):
 223             return values.map(lambda x: get_datevalue(x, axis.freq))
 224         elif lib.infer_dtype(values, skipna=False) == "period":
 225             # https://github.com/pandas-dev/pandas/issues/24304
 226             # convert ndarray[period] -> PeriodIndex
 227             return PeriodIndex(values, freq=axis.freq).asi8
 228         elif isinstance(values, (list, tuple, np.ndarray, Index)):
 229             return [get_datevalue(x, axis.freq) for x in values]
 230         return values

 233 def get_datevalue(date, freq):
 234     if isinstance(date, Period):
 235         return date.asfreq(freq).ordinal
 236     elif isinstance(date, (str, datetime, pydt.date, pydt.time, np.datetime64)):
 237         return Period(date, freq).ordinal
 238     elif (
 239         is_integer(date)
 240         or is_float(date)
 241         or (isinstance(date, (np.ndarray, Index)) and (date.size == 1))
 242     ):
 243         return date
 244     elif date is None:
 245         return None
 246     raise ValueError(f"Unrecognizable date '{date}'")

 249 def _dt_to_float_ordinal(dt):
 250     """
 251     Convert :mod:`datetime` to the Gregorian date as UTC float days,
 252     preserving hours, minutes, seconds and microseconds.  Return value
 253     is a :func:`float`.
 254     """
 255     if isinstance(dt, (np.ndarray, Index, Series)) and is_datetime64_ns_dtype(dt):
 256         base = dates.epoch2num(dt.asi8 / 1.0e9)
 257     else:
 258         base = dates.date2num(dt)
 259     return base

 262 # Datetime Conversion
 263 class DatetimeConverter(dates.DateConverter):
 264     @staticmethod
 265     def convert(values, unit, axis):
 266         # values might be a 1-d array, or a list-like of arrays.
 267         if is_nested_list_like(values):
 268             values = [DatetimeConverter._convert_1d(v, unit, axis) for v in values]
 269         else:
 270             values = DatetimeConverter._convert_1d(values, unit, axis)
 271         return values
 272 
 273     @staticmethod
 274     def _convert_1d(values, unit, axis):
 275         def try_parse(values):
 276             try:
 277                 return _dt_to_float_ordinal(tools.to_datetime(values))
 278             except Exception:
 279                 return values
 280 
 281         if isinstance(values, (datetime, pydt.date)):
 282             return _dt_to_float_ordinal(values)
 283         elif isinstance(values, np.datetime64):
 284             return _dt_to_float_ordinal(Timestamp(values))
 285         elif isinstance(values, pydt.time):
 286             return dates.date2num(values)
 287         elif is_integer(values) or is_float(values):
 288             return values
 289         elif isinstance(values, str):
 290             return try_parse(values)
 291         elif isinstance(values, (list, tuple, np.ndarray, Index, Series)):
 292             if isinstance(values, Series):
 293                 # https://github.com/matplotlib/matplotlib/issues/11391
 294                 # Series was skipped. Convert to DatetimeIndex to get asi8
 295                 values = Index(values)
 296             if isinstance(values, Index):
 297                 values = values.values
 298             if not isinstance(values, np.ndarray):
 299                 values = com.asarray_tuplesafe(values)
 300 
 301             if is_integer_dtype(values) or is_float_dtype(values):
 302                 return values
 303 
 304             try:
 305                 values = tools.to_datetime(values)
 306                 if isinstance(values, Index):
 307                     values = _dt_to_float_ordinal(values)
 308                 else:
 309                     values = [_dt_to_float_ordinal(x) for x in values]
 310             except Exception:
 311                 values = _dt_to_float_ordinal(values)
 312 
 313         return values

 340 class PandasAutoDateLocator(dates.AutoDateLocator):
 341     def get_locator(self, dmin, dmax):
 342         """Pick the best locator based on a distance."""
 343         delta = relativedelta(dmax, dmin)
 344 
 345         num_days = (delta.years * 12.0 + delta.months) * 31.0 + delta.days
 346         num_sec = (delta.hours * 60.0 + delta.minutes) * 60.0 + delta.seconds
 347         tot_sec = num_days * 86400.0 + num_sec
 348 
 349         if abs(tot_sec) < self.minticks:
 350             self._freq = -1
 351             locator = MilliSecondLocator(self.tz)
 352             locator.set_axis(self.axis)
 353 
 354             locator.set_view_interval(*self.axis.get_view_interval())
 355             locator.set_data_interval(*self.axis.get_data_interval())
 356             return locator
 357 
 358         return dates.AutoDateLocator.get_locator(self, dmin, dmax)
 359 
 360     def _get_unit(self):
 361         return MilliSecondLocator.get_unit_generic(self._freq)

 382     def __call__(self):
 383         # if no data have been set, this will tank with a ValueError
 384         try:
 385             dmin, dmax = self.viewlim_to_dt()
 386         except ValueError:
 387             return []
 388 
 389         # We need to cap at the endpoints of valid datetime
 390         nmax, nmin = dates.date2num((dmax, dmin))
 391 
 392         num = (nmax - nmin) * 86400 * 1000
 393         max_millis_ticks = 6
 394         for interval in [1, 10, 50, 100, 200, 500]:
 395             if num <= interval * (max_millis_ticks - 1):
 396                 self._interval = interval
 397                 break
 398             else:
 399                 # We went through the whole loop without breaking, default to 1
 400                 self._interval = 1000.0
 401 
 402         estimate = (nmax - nmin) / (self._get_unit() * self._get_interval())
 403 
 404         if estimate > self.MAXTICKS * 2:
 405             raise RuntimeError(
 406                 "MillisecondLocator estimated to generate "
 407                 f"{estimate:d} ticks from {dmin} to {dmax}: exceeds Locator.MAXTICKS"
 408                 f"* 2 ({self.MAXTICKS * 2:d}) "
 409             )
 410 
 411         interval = self._get_interval()
 412         freq = f"{interval}L"
 413         tz = self.tz.tzname(None)
 414         st = _from_ordinal(dates.date2num(dmin))  # strip tz
 415         ed = _from_ordinal(dates.date2num(dmax))
 416         all_dates = date_range(start=st, end=ed, freq=freq, tz=tz).astype(object)
 417 
 418         try:
 419             if len(all_dates) > 0:
 420                 locs = self.raise_if_exceeds(dates.date2num(all_dates))
 421                 return locs
 422         except Exception:  # pragma: no cover
 423             pass
 424 
 425         lims = dates.date2num([dmin, dmax])
 426         return lims

 429         return self._interval

 444 def _from_ordinal(x, tz=None):
 445     ix = int(x)
 446     dt = datetime.fromordinal(ix)
 447     remainder = float(x) - ix
 448     hour, remainder = divmod(24 * remainder, 1)
 449     minute, remainder = divmod(60 * remainder, 1)
 450     second, remainder = divmod(60 * remainder, 1)
 451     microsecond = int(1e6 * remainder)
 452     if microsecond < 10:
 453         microsecond = 0  # compensate for rounding errors
 454     dt = datetime(
 455         dt.year, dt.month, dt.day, int(hour), int(minute), int(second), microsecond
 456     )
 457     if tz is not None:
 458         dt = dt.astimezone(tz)
 459 
 460     if microsecond > 999990:  # compensate for rounding errors
 461         dt += timedelta(microseconds=1e6 - microsecond)
 462 
 463     return dt

 473 def _get_default_annual_spacing(nyears):
 474     """
 475     Returns a default spacing between consecutive ticks for annual data.
 476     """
 477     if nyears < 11:
 478         (min_spacing, maj_spacing) = (1, 1)
 479     elif nyears < 20:
 480         (min_spacing, maj_spacing) = (1, 2)
 481     elif nyears < 50:
 482         (min_spacing, maj_spacing) = (1, 5)
 483     elif nyears < 100:
 484         (min_spacing, maj_spacing) = (5, 10)
 485     elif nyears < 200:
 486         (min_spacing, maj_spacing) = (5, 25)
 487     elif nyears < 600:
 488         (min_spacing, maj_spacing) = (10, 50)
 489     else:
 490         factor = nyears // 1000 + 1
 491         (min_spacing, maj_spacing) = (factor * 20, factor * 100)
 492     return (min_spacing, maj_spacing)

 495 def period_break(dates, period):
 496     """
 497     Returns the indices where the given period changes.
 498 
 499     Parameters
 500     ----------
 501     dates : PeriodIndex
 502         Array of intervals to monitor.
 503     period : string
 504         Name of the period to monitor.
 505     """
 506     current = getattr(dates, period)
 507     previous = getattr(dates - 1 * dates.freq, period)
 508     return np.nonzero(current - previous)[0]

 511 def has_level_label(label_flags, vmin):
 512     """
 513     Returns true if the ``label_flags`` indicate there is at least one label
 514     for this level.

 516     if the minimum view limit is not an exact integer, then the first tick
 517     label won't be shown, so we must adjust for that.
 518     """

 524         return True

 527 def _daily_finder(vmin, vmax, freq: BaseOffset):
 528     dtype_code = freq._period_dtype_code
 529 
 530     periodsperday = -1
 531 
 532     if dtype_code >= FreqGroup.FR_HR:
 533         if dtype_code == FreqGroup.FR_NS:
 534             periodsperday = 24 * 60 * 60 * 1000000000
 535         elif dtype_code == FreqGroup.FR_US:
 536             periodsperday = 24 * 60 * 60 * 1000000
 537         elif dtype_code == FreqGroup.FR_MS:
 538             periodsperday = 24 * 60 * 60 * 1000
 539         elif dtype_code == FreqGroup.FR_SEC:
 540             periodsperday = 24 * 60 * 60
 541         elif dtype_code == FreqGroup.FR_MIN:
 542             periodsperday = 24 * 60
 543         elif dtype_code == FreqGroup.FR_HR:
 544             periodsperday = 24
 545         else:  # pragma: no cover
 546             raise ValueError(f"unexpected frequency: {dtype_code}")
 547         periodsperyear = 365 * periodsperday
 548         periodspermonth = 28 * periodsperday
 549 
 550     elif dtype_code == FreqGroup.FR_BUS:
 551         periodsperyear = 261
 552         periodspermonth = 19
 553     elif dtype_code == FreqGroup.FR_DAY:
 554         periodsperyear = 365
 555         periodspermonth = 28
 556     elif FreqGroup.get_freq_group(dtype_code) == FreqGroup.FR_WK:
 557         periodsperyear = 52
 558         periodspermonth = 3
 559     else:  # pragma: no cover
 560         raise ValueError("unexpected frequency")
 561 
 562     # save this for later usage
 563     vmin_orig = vmin
 564 
 565     (vmin, vmax) = (
 566         Period(ordinal=int(vmin), freq=freq),
 567         Period(ordinal=int(vmax), freq=freq),
 568     )
 569     span = vmax.ordinal - vmin.ordinal + 1
 570     dates_ = period_range(start=vmin, end=vmax, freq=freq)
 571     # Initialize the output
 572     info = np.zeros(
 573         span, dtype=[("val", np.int64), ("maj", bool), ("min", bool), ("fmt", "|S20")]
 574     )
 575     info["val"][:] = dates_.asi8
 576     info["fmt"][:] = ""
 577     info["maj"][[0, -1]] = True
 578     # .. and set some shortcuts
 579     info_maj = info["maj"]
 580     info_min = info["min"]
 581     info_fmt = info["fmt"]
 582 
 583     def first_label(label_flags):
 584         if (label_flags[0] == 0) and (label_flags.size > 1) and ((vmin_orig % 1) > 0.0):
 585             return label_flags[1]
 586         else:
 587             return label_flags[0]
 588 
 589     # Case 1. Less than a month
 590     if span <= periodspermonth:
 591         day_start = period_break(dates_, "day")
 592         month_start = period_break(dates_, "month")
 593 
 594         def _hour_finder(label_interval, force_year_start):
 595             _hour = dates_.hour
 596             _prev_hour = (dates_ - 1 * dates_.freq).hour
 597             hour_start = (_hour - _prev_hour) != 0
 598             info_maj[day_start] = True
 599             info_min[hour_start & (_hour % label_interval == 0)] = True
 600             year_start = period_break(dates_, "year")
 601             info_fmt[hour_start & (_hour % label_interval == 0)] = "%H:%M"
 602             info_fmt[day_start] = "%H:%M\n%d-%b"
 603             info_fmt[year_start] = "%H:%M\n%d-%b\n%Y"
 604             if force_year_start and not has_level_label(year_start, vmin_orig):
 605                 info_fmt[first_label(day_start)] = "%H:%M\n%d-%b\n%Y"
 606 
 607         def _minute_finder(label_interval):
 608             hour_start = period_break(dates_, "hour")
 609             _minute = dates_.minute
 610             _prev_minute = (dates_ - 1 * dates_.freq).minute
 611             minute_start = (_minute - _prev_minute) != 0
 612             info_maj[hour_start] = True
 613             info_min[minute_start & (_minute % label_interval == 0)] = True
 614             year_start = period_break(dates_, "year")
 615             info_fmt = info["fmt"]
 616             info_fmt[minute_start & (_minute % label_interval == 0)] = "%H:%M"
 617             info_fmt[day_start] = "%H:%M\n%d-%b"
 618             info_fmt[year_start] = "%H:%M\n%d-%b\n%Y"
 619 
 620         def _second_finder(label_interval):
 621             minute_start = period_break(dates_, "minute")
 622             _second = dates_.second
 623             _prev_second = (dates_ - 1 * dates_.freq).second
 624             second_start = (_second - _prev_second) != 0
 625             info["maj"][minute_start] = True
 626             info["min"][second_start & (_second % label_interval == 0)] = True
 627             year_start = period_break(dates_, "year")
 628             info_fmt = info["fmt"]
 629             info_fmt[second_start & (_second % label_interval == 0)] = "%H:%M:%S"
 630             info_fmt[day_start] = "%H:%M:%S\n%d-%b"
 631             info_fmt[year_start] = "%H:%M:%S\n%d-%b\n%Y"
 632 
 633         if span < periodsperday / 12000.0:
 634             _second_finder(1)
 635         elif span < periodsperday / 6000.0:
 636             _second_finder(2)
 637         elif span < periodsperday / 2400.0:
 638             _second_finder(5)
 639         elif span < periodsperday / 1200.0:
 640             _second_finder(10)
 641         elif span < periodsperday / 800.0:
 642             _second_finder(15)
 643         elif span < periodsperday / 400.0:
 644             _second_finder(30)
 645         elif span < periodsperday / 150.0:
 646             _minute_finder(1)
 647         elif span < periodsperday / 70.0:
 648             _minute_finder(2)
 649         elif span < periodsperday / 24.0:
 650             _minute_finder(5)
 651         elif span < periodsperday / 12.0:
 652             _minute_finder(15)
 653         elif span < periodsperday / 6.0:
 654             _minute_finder(30)
 655         elif span < periodsperday / 2.5:
 656             _hour_finder(1, False)
 657         elif span < periodsperday / 1.5:
 658             _hour_finder(2, False)
 659         elif span < periodsperday * 1.25:
 660             _hour_finder(3, False)
 661         elif span < periodsperday * 2.5:
 662             _hour_finder(6, True)
 663         elif span < periodsperday * 4:
 664             _hour_finder(12, True)
 665         else:
 666             info_maj[month_start] = True
 667             info_min[day_start] = True
 668             year_start = period_break(dates_, "year")
 669             info_fmt = info["fmt"]
 670             info_fmt[day_start] = "%d"
 671             info_fmt[month_start] = "%d\n%b"
 672             info_fmt[year_start] = "%d\n%b\n%Y"
 673             if not has_level_label(year_start, vmin_orig):
 674                 if not has_level_label(month_start, vmin_orig):
 675                     info_fmt[first_label(day_start)] = "%d\n%b\n%Y"
 676                 else:
 677                     info_fmt[first_label(month_start)] = "%d\n%b\n%Y"
 678 
 679     # Case 2. Less than three months
 680     elif span <= periodsperyear // 4:
 681         month_start = period_break(dates_, "month")
 682         info_maj[month_start] = True
 683         if dtype_code < FreqGroup.FR_HR:
 684             info["min"] = True
 685         else:
 686             day_start = period_break(dates_, "day")
 687             info["min"][day_start] = True
 688         week_start = period_break(dates_, "week")
 689         year_start = period_break(dates_, "year")
 690         info_fmt[week_start] = "%d"
 691         info_fmt[month_start] = "\n\n%b"
 692         info_fmt[year_start] = "\n\n%b\n%Y"
 693         if not has_level_label(year_start, vmin_orig):
 694             if not has_level_label(month_start, vmin_orig):
 695                 info_fmt[first_label(week_start)] = "\n\n%b\n%Y"
 696             else:
 697                 info_fmt[first_label(month_start)] = "\n\n%b\n%Y"
 698     # Case 3. Less than 14 months ...............
 699     elif span <= 1.15 * periodsperyear:
 700         year_start = period_break(dates_, "year")
 701         month_start = period_break(dates_, "month")
 702         week_start = period_break(dates_, "week")
 703         info_maj[month_start] = True
 704         info_min[week_start] = True
 705         info_min[year_start] = False
 706         info_min[month_start] = False
 707         info_fmt[month_start] = "%b"
 708         info_fmt[year_start] = "%b\n%Y"
 709         if not has_level_label(year_start, vmin_orig):
 710             info_fmt[first_label(month_start)] = "%b\n%Y"
 711     # Case 4. Less than 2.5 years ...............
 712     elif span <= 2.5 * periodsperyear:
 713         year_start = period_break(dates_, "year")
 714         quarter_start = period_break(dates_, "quarter")
 715         month_start = period_break(dates_, "month")
 716         info_maj[quarter_start] = True
 717         info_min[month_start] = True
 718         info_fmt[quarter_start] = "%b"
 719         info_fmt[year_start] = "%b\n%Y"
 720     # Case 4. Less than 4 years .................
 721     elif span <= 4 * periodsperyear:
 722         year_start = period_break(dates_, "year")
 723         month_start = period_break(dates_, "month")
 724         info_maj[year_start] = True
 725         info_min[month_start] = True
 726         info_min[year_start] = False
 727 
 728         month_break = dates_[month_start].month
 729         jan_or_jul = month_start[(month_break == 1) | (month_break == 7)]
 730         info_fmt[jan_or_jul] = "%b"
 731         info_fmt[year_start] = "%b\n%Y"
 732     # Case 5. Less than 11 years ................
 733     elif span <= 11 * periodsperyear:
 734         year_start = period_break(dates_, "year")
 735         quarter_start = period_break(dates_, "quarter")
 736         info_maj[year_start] = True
 737         info_min[quarter_start] = True
 738         info_min[year_start] = False
 739         info_fmt[year_start] = "%Y"
 740     # Case 6. More than 12 years ................
 741     else:
 742         year_start = period_break(dates_, "year")
 743         year_break = dates_[year_start].year
 744         nyears = span / periodsperyear
 745         (min_anndef, maj_anndef) = _get_default_annual_spacing(nyears)
 746         major_idx = year_start[(year_break % maj_anndef == 0)]
 747         info_maj[major_idx] = True
 748         minor_idx = year_start[(year_break % min_anndef == 0)]
 749         info_min[minor_idx] = True
 750         info_fmt[major_idx] = "%Y"
 751 
 752     return info

 755 def _monthly_finder(vmin, vmax, freq):
 756     periodsperyear = 12
 757 
 758     vmin_orig = vmin
 759     (vmin, vmax) = (int(vmin), int(vmax))
 760     span = vmax - vmin + 1
 761 
 762     # Initialize the output
 763     info = np.zeros(
 764         span, dtype=[("val", int), ("maj", bool), ("min", bool), ("fmt", "|S8")]
 765     )
 766     info["val"] = np.arange(vmin, vmax + 1)
 767     dates_ = info["val"]
 768     info["fmt"] = ""
 769     year_start = (dates_ % 12 == 0).nonzero()[0]
 770     info_maj = info["maj"]
 771     info_fmt = info["fmt"]
 772 
 773     if span <= 1.15 * periodsperyear:
 774         info_maj[year_start] = True
 775         info["min"] = True
 776 
 777         info_fmt[:] = "%b"
 778         info_fmt[year_start] = "%b\n%Y"
 779 
 780         if not has_level_label(year_start, vmin_orig):
 781             if dates_.size > 1:
 782                 idx = 1
 783             else:
 784                 idx = 0
 785             info_fmt[idx] = "%b\n%Y"
 786 
 787     elif span <= 2.5 * periodsperyear:
 788         quarter_start = (dates_ % 3 == 0).nonzero()
 789         info_maj[year_start] = True
 790         # TODO: Check the following : is it really info['fmt'] ?
 791         info["fmt"][quarter_start] = True
 792         info["min"] = True
 793 
 794         info_fmt[quarter_start] = "%b"
 795         info_fmt[year_start] = "%b\n%Y"
 796 
 797     elif span <= 4 * periodsperyear:
 798         info_maj[year_start] = True
 799         info["min"] = True
 800 
 801         jan_or_jul = (dates_ % 12 == 0) | (dates_ % 12 == 6)
 802         info_fmt[jan_or_jul] = "%b"
 803         info_fmt[year_start] = "%b\n%Y"
 804 
 805     elif span <= 11 * periodsperyear:
 806         quarter_start = (dates_ % 3 == 0).nonzero()
 807         info_maj[year_start] = True
 808         info["min"][quarter_start] = True
 809 
 810         info_fmt[year_start] = "%Y"
 811 
 812     else:
 813         nyears = span / periodsperyear
 814         (min_anndef, maj_anndef) = _get_default_annual_spacing(nyears)
 815         years = dates_[year_start] // 12 + 1
 816         major_idx = year_start[(years % maj_anndef == 0)]
 817         info_maj[major_idx] = True
 818         info["min"][year_start[(years % min_anndef == 0)]] = True
 819 
 820         info_fmt[major_idx] = "%Y"
 821 
 822     return info

 825 def _quarterly_finder(vmin, vmax, freq):
 826     periodsperyear = 4
 827     vmin_orig = vmin
 828     (vmin, vmax) = (int(vmin), int(vmax))
 829     span = vmax - vmin + 1
 830 
 831     info = np.zeros(
 832         span, dtype=[("val", int), ("maj", bool), ("min", bool), ("fmt", "|S8")]
 833     )
 834     info["val"] = np.arange(vmin, vmax + 1)
 835     info["fmt"] = ""
 836     dates_ = info["val"]
 837     info_maj = info["maj"]
 838     info_fmt = info["fmt"]
 839     year_start = (dates_ % 4 == 0).nonzero()[0]
 840 
 841     if span <= 3.5 * periodsperyear:
 842         info_maj[year_start] = True
 843         info["min"] = True
 844 
 845         info_fmt[:] = "Q%q"
 846         info_fmt[year_start] = "Q%q\n%F"
 847         if not has_level_label(year_start, vmin_orig):
 848             if dates_.size > 1:
 849                 idx = 1
 850             else:
 851                 idx = 0
 852             info_fmt[idx] = "Q%q\n%F"
 853 
 854     elif span <= 11 * periodsperyear:
 855         info_maj[year_start] = True
 856         info["min"] = True
 857         info_fmt[year_start] = "%F"
 858 
 859     else:
 860         years = dates_[year_start] // 4 + 1
 861         nyears = span / periodsperyear
 862         (min_anndef, maj_anndef) = _get_default_annual_spacing(nyears)
 863         major_idx = year_start[(years % maj_anndef == 0)]
 864         info_maj[major_idx] = True
 865         info["min"][year_start[(years % min_anndef == 0)]] = True
 866         info_fmt[major_idx] = "%F"
 867

 871 def _annual_finder(vmin, vmax, freq):
 872     (vmin, vmax) = (int(vmin), int(vmax + 1))
 873     span = vmax - vmin + 1
 874 
 875     info = np.zeros(

 879     info["fmt"] = ""

 884     info["maj"][major_idx] = True
 885     info["min"][(dates_ % min_anndef == 0)] = True
 886     info["fmt"][major_idx] = "%Y"

 891 def get_finder(freq: BaseOffset):
 892     dtype_code = freq._period_dtype_code
 893     fgroup = (dtype_code // 1000) * 1000
 894 
 895     if fgroup == FreqGroup.FR_ANN:
 896         return _annual_finder
 897     elif fgroup == FreqGroup.FR_QTR:
 898         return _quarterly_finder
 899     elif dtype_code == FreqGroup.FR_MTH:
 900         return _monthly_finder
 901     elif (dtype_code >= FreqGroup.FR_BUS) or fgroup == FreqGroup.FR_WK:
 902         return _daily_finder
 903     else:  # pragma: no cover
 904         raise NotImplementedError(f"Unsupported frequency: {dtype_code}")

 907 class TimeSeries_DateLocator(Locator):
 908     """
 909     Locates the ticks along an axis controlled by a :class:`Series`.
 910 
 911     Parameters
 912     ----------
 913     freq : {var}
 914         Valid frequency specifier.
 915     minor_locator : {False, True}, optional
 916         Whether the locator is for minor ticks (True) or not.
 917     dynamic_mode : {True, False}, optional
 918         Whether the locator should work in dynamic mode.
 919     base : {int}, optional
 920     quarter : {int}, optional
 921     month : {int}, optional
 922     day : {int}, optional
 923     """
 924 
 925     def __init__(
 926         self,
 927         freq,
 928         minor_locator=False,
 929         dynamic_mode=True,
 930         base=1,
 931         quarter=1,
 932         month=1,
 933         day=1,
 934         plot_obj=None,
 935     ):
 936         freq = to_offset(freq)
 937         self.freq = freq
 938         self.base = base
 939         (self.quarter, self.month, self.day) = (quarter, month, day)
 940         self.isminor = minor_locator
 941         self.isdynamic = dynamic_mode
 942         self.offset = 0
 943         self.plot_obj = plot_obj
 944         self.finder = get_finder(freq)
 945 
 946     def _get_default_locs(self, vmin, vmax):
 947         """Returns the default locations of ticks."""
 948         if self.plot_obj.date_axis_info is None:
 949             self.plot_obj.date_axis_info = self.finder(vmin, vmax, self.freq)
 950 
 951         locator = self.plot_obj.date_axis_info
 952 
 953         if self.isminor:
 954             return np.compress(locator["min"], locator["val"])
 955         return np.compress(locator["maj"], locator["val"])
 956 
 957     def __call__(self):
 958         """Return the locations of the ticks."""
 959         # axis calls Locator.set_axis inside set_m<xxxx>_formatter
 960 
 961         vi = tuple(self.axis.get_view_interval())
 962         if vi != self.plot_obj.view_interval:
 963             self.plot_obj.date_axis_info = None
 964         self.plot_obj.view_interval = vi
 965         vmin, vmax = vi
 966         if vmax < vmin:
 967             vmin, vmax = vmax, vmin
 968         if self.isdynamic:
 969             locs = self._get_default_locs(vmin, vmax)
 970         else:  # pragma: no cover
 971             base = self.base
 972             (d, m) = divmod(vmin, base)
 973             vmin = (d + 1) * base
 974             locs = list(range(vmin, vmax + 1, base))
 975         return locs
 976 
 977     def autoscale(self):
 978         """
 979         Sets the view limits to the nearest multiples of base that contain the
 980         data.
 981         """
 982         # requires matplotlib >= 0.98.0
 983         (vmin, vmax) = self.axis.get_data_interval()
 984 
 985         locs = self._get_default_locs(vmin, vmax)
 986         (vmin, vmax) = locs[[0, -1]]
 987         if vmin == vmax:
 988             vmin -= 1
 989             vmax += 1
 990         return nonsingular(vmin, vmax)

 998 class TimeSeries_DateFormatter(Formatter):
 999     """
1000     Formats the ticks along an axis controlled by a :class:`PeriodIndex`.
1001 
1002     Parameters
1003     ----------
1004     freq : {int, string}
1005         Valid frequency specifier.
1006     minor_locator : {False, True}
1007         Whether the current formatter should apply to minor ticks (True) or
1008         major ticks (False).
1009     dynamic_mode : {True, False}
1010         Whether the formatter works in dynamic mode or not.
1011     """
1012 
1013     def __init__(self, freq, minor_locator=False, dynamic_mode=True, plot_obj=None):
1014         freq = to_offset(freq)
1015         self.format = None
1016         self.freq = freq
1017         self.locs = []
1018         self.formatdict = None
1019         self.isminor = minor_locator
1020         self.isdynamic = dynamic_mode
1021         self.offset = 0
1022         self.plot_obj = plot_obj
1023         self.finder = get_finder(freq)
1024 
1025     def _set_default_format(self, vmin, vmax):
1026         """Returns the default ticks spacing."""
1027         if self.plot_obj.date_axis_info is None:
1028             self.plot_obj.date_axis_info = self.finder(vmin, vmax, self.freq)
1029         info = self.plot_obj.date_axis_info
1030 
1031         if self.isminor:
1032             format = np.compress(info["min"] & np.logical_not(info["maj"]), info)
1033         else:
1034             format = np.compress(info["maj"], info)
1035         self.formatdict = {x: f for (x, _, _, f) in format}
1036         return self.formatdict
1037 
1038     def set_locs(self, locs):
1039         """Sets the locations of the ticks"""
1040         # don't actually use the locs. This is just needed to work with
1041         # matplotlib. Force to use vmin, vmax
1042 
1043         self.locs = locs
1044 
1045         (vmin, vmax) = vi = tuple(self.axis.get_view_interval())
1046         if vi != self.plot_obj.view_interval:
1047             self.plot_obj.date_axis_info = None
1048         self.plot_obj.view_interval = vi
1049         if vmax < vmin:
1050             (vmin, vmax) = (vmax, vmin)
1051         self._set_default_format(vmin, vmax)
1052 
1053     def __call__(self, x, pos=0) -> str:
1054 
1055         if self.formatdict is None:
1056             return ""
1057         else:
1058             fmt = self.formatdict.pop(x, "")
1059             if isinstance(fmt, np.bytes_):
1060                 fmt = fmt.decode("utf-8")
1061             return Period(ordinal=int(x), freq=self.freq).strftime(fmt)

1064 class TimeSeries_TimedeltaFormatter(Formatter):
1065     """
1066     Formats the ticks along an axis controlled by a :class:`TimedeltaIndex`.
1067     """
1068 
1069     @staticmethod
1070     def format_timedelta_ticks(x, pos, n_decimals: int) -> str:
1071         """
1072         Convert seconds to 'D days HH:MM:SS.F'
1073         """
1074         s, ns = divmod(x, 1e9)
1075         m, s = divmod(s, 60)
1076         h, m = divmod(m, 60)
1077         d, h = divmod(h, 24)
1078         decimals = int(ns * 10 ** (n_decimals - 9))
1079         s = f"{int(h):02d}:{int(m):02d}:{int(s):02d}"
1080         if n_decimals > 0:
1081             s += f".{decimals:0{n_decimals}d}"
1082         if d != 0:
1083             s = f"{int(d):d} days {s}"
1084         return s
1085 
1086     def __call__(self, x, pos=0) -> str:
1087         (vmin, vmax) = tuple(self.axis.get_view_interval())
1088         n_decimals = int(np.ceil(np.log10(100 * 1e9 / (vmax - vmin))))
1089         if n_decimals > 9:
1090             n_decimals = 9
1091         return self.format_timedelta_ticks(x, pos, n_decimals)
