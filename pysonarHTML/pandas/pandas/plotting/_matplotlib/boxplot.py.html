  27         # Do not call LinePlot.__init__ which may fill nan

  34     def _args_adjust(self):
  35         if self.subplots:
  36             # Disable label ax sharing. Otherwise, all subplots shows last
  37             # column label
  38             if self.orientation == "vertical":
  39                 self.sharex = False

  43     @classmethod
  44     def _plot(cls, ax, y, column_num=None, return_type="axes", **kwds):
  45         if y.ndim == 2:
  46             y = [remove_na_arraylike(v) for v in y]
  47             # Boxplot fails with empty arrays, so need to add a NaN
  48             #   if any cols are empty
  49             # GH 8181
  50             y = [v if v.size > 0 else np.array([np.nan]) for v in y]
  51         else:
  52             y = remove_na_arraylike(y)
  53         bp = ax.boxplot(y, **kwds)
  54 
  55         if return_type == "dict":
  56             return bp, bp
  57         elif return_type == "both":
  58             return cls.BP(ax=ax, lines=bp), bp
  59         else:
  60             return ax, bp

  62     def _validate_color_args(self):
  63         if "color" in self.kwds:
  64             if self.colormap is not None:
  65                 warnings.warn(
  66                     "'color' and 'colormap' cannot be used "
  67                     "simultaneously. Using 'color'"
  68                 )
  69             self.color = self.kwds.pop("color")
  70 
  71             if isinstance(self.color, dict):
  72                 valid_keys = ["boxes", "whiskers", "medians", "caps"]
  73                 for key, values in self.color.items():
  74                     if key not in valid_keys:
  75                         raise ValueError(
  76                             f"color dict contains invalid key '{key}'. "
  77                             f"The key must be either {valid_keys}"
  78                         )
  79         else:
  80             self.color = None
  81 
  82         # get standard colors for default
  83         colors = _get_standard_colors(num_colors=3, colormap=self.colormap, color=None)
  84         # use 2 colors by default, for box/whisker and median
  85         # flier colors isn't needed here
  86         # because it can be specified by ``sym`` kw
  87         self._boxes_c = colors[0]
  88         self._whiskers_c = colors[0]
  89         self._medians_c = colors[2]
  90         self._caps_c = "k"  # mpl default

  95     def maybe_color_bp(self, bp):
  96         if isinstance(self.color, dict):
  97             boxes = self.color.get("boxes", self._boxes_c)
  98             whiskers = self.color.get("whiskers", self._whiskers_c)
  99             medians = self.color.get("medians", self._medians_c)
 100             caps = self.color.get("caps", self._caps_c)
 101         else:
 102             # Other types are forwarded to matplotlib
 103             # If None, use default colors
 104             boxes = self.color or self._boxes_c
 105             whiskers = self.color or self._whiskers_c
 106             medians = self.color or self._medians_c
 107             caps = self.color or self._caps_c
 108 
 109         # GH 30346, when users specifying those arguments explicitly, our defaults
 110         # for these four kwargs should be overridden; if not, use Pandas settings
 111         if not self.kwds.get("boxprops"):
 112             setp(bp["boxes"], color=boxes, alpha=1)
 113         if not self.kwds.get("whiskerprops"):
 114             setp(bp["whiskers"], color=whiskers, alpha=1)
 115         if not self.kwds.get("medianprops"):
 116             setp(bp["medians"], color=medians, alpha=1)
 117         if not self.kwds.get("capprops"):
 118             setp(bp["caps"], color=caps, alpha=1)

 120     def _make_plot(self):
 121         if self.subplots:
 122             self._return_obj = pd.Series(dtype=object)
 123 
 124             for i, (label, y) in enumerate(self._iter_data()):
 125                 ax = self._get_ax(i)
 126                 kwds = self.kwds.copy()
 127 
 128                 ret, bp = self._plot(
 129                     ax, y, column_num=i, return_type=self.return_type, **kwds
 130                 )
 131                 self.maybe_color_bp(bp)
 132                 self._return_obj[label] = ret
 133 
 134                 label = [pprint_thing(label)]
 135                 self._set_ticklabels(ax, label)
 136         else:
 137             y = self.data.values.T
 138             ax = self._get_ax(0)
 139             kwds = self.kwds.copy()
 140 
 141             ret, bp = self._plot(
 142                 ax, y, column_num=0, return_type=self.return_type, **kwds
 143             )
 144             self.maybe_color_bp(bp)
 145             self._return_obj = ret
 146 
 147             labels = [l for l, _ in self._iter_data()]
 148             labels = [pprint_thing(l) for l in labels]
 149             if not self.use_index:
 150                 labels = [pprint_thing(key) for key in range(len(labels))]
 151             self._set_ticklabels(ax, labels)

 180 def _grouped_plot_by_column(
 181     plotf,
 182     data,
 183     columns=None,
 184     by=None,
 185     numeric_only=True,
 186     grid=False,
 187     figsize=None,
 188     ax=None,
 189     layout=None,
 190     return_type=None,
 191     **kwargs,
 192 ):
 193     grouped = data.groupby(by)
 194     if columns is None:
 195         if not isinstance(by, (list, tuple)):
 196             by = [by]
 197         columns = data._get_numeric_data().columns.difference(by)
 198     naxes = len(columns)
 199     fig, axes = _subplots(
 200         naxes=naxes, sharex=True, sharey=True, figsize=figsize, ax=ax, layout=layout
 201     )
 202 
 203     _axes = _flatten(axes)
 204 
 205     ax_values = []
 206 
 207     for i, col in enumerate(columns):
 208         ax = _axes[i]
 209         gp_col = grouped[col]
 210         keys, values = zip(*gp_col)
 211         re_plotf = plotf(keys, values, ax, **kwargs)
 212         ax.set_title(col)
 213         ax.set_xlabel(pprint_thing(by))
 214         ax_values.append(re_plotf)
 215         ax.grid(grid)
 216 
 217     result = pd.Series(ax_values, index=columns)
 218 
 219     # Return axes in multiplot case, maybe revisit later # 985
 220     if return_type is None:
 221         result = axes
 222 
 223     byline = by[0] if len(by) == 1 else by
 224     fig.suptitle(f"Boxplot grouped by {byline}")
 225     fig.subplots_adjust(bottom=0.15, top=0.9, left=0.1, right=0.9, wspace=0.2)
 226 
 227     return result

 230 def boxplot(
 231     data,
 232     column=None,
 233     by=None,
 234     ax=None,
 235     fontsize=None,
 236     rot=0,
 237     grid=True,
 238     figsize=None,
 239     layout=None,
 240     return_type=None,
 241     **kwds,
 242 ):
 243 
 244     import matplotlib.pyplot as plt
 245 
 246     # validate return_type:
 247     if return_type not in BoxPlot._valid_return_types:
 248         raise ValueError("return_type must be {'axes', 'dict', 'both'}")
 249 
 250     if isinstance(data, pd.Series):
 251         data = data.to_frame("x")
 252         column = "x"
 253 
 254     def _get_colors():
 255         #  num_colors=3 is required as method maybe_color_bp takes the colors
 256         #  in positions 0 and 2.
 257         #  if colors not provided, use same defaults as DataFrame.plot.box
 258         result = _get_standard_colors(num_colors=3)
 259         result = np.take(result, [0, 0, 2])
 260         result = np.append(result, "k")
 261 
 262         colors = kwds.pop("color", None)
 263         if colors:
 264             if is_dict_like(colors):
 265                 # replace colors in result array with user-specified colors
 266                 # taken from the colors dict parameter
 267                 # "boxes" value placed in position 0, "whiskers" in 1, etc.
 268                 valid_keys = ["boxes", "whiskers", "medians", "caps"]
 269                 key_to_index = dict(zip(valid_keys, range(4)))
 270                 for key, value in colors.items():
 271                     if key in valid_keys:
 272                         result[key_to_index[key]] = value
 273                     else:
 274                         raise ValueError(
 275                             f"color dict contains invalid key '{key}'. "
 276                             f"The key must be either {valid_keys}"
 277                         )
 278             else:
 279                 result.fill(colors)
 280 
 281         return result
 282 
 283     def maybe_color_bp(bp, **kwds):
 284         # GH 30346, when users specifying those arguments explicitly, our defaults
 285         # for these four kwargs should be overridden; if not, use Pandas settings
 286         if not kwds.get("boxprops"):
 287             setp(bp["boxes"], color=colors[0], alpha=1)
 288         if not kwds.get("whiskerprops"):
 289             setp(bp["whiskers"], color=colors[1], alpha=1)
 290         if not kwds.get("medianprops"):
 291             setp(bp["medians"], color=colors[2], alpha=1)
 292         if not kwds.get("capprops"):
 293             setp(bp["caps"], color=colors[3], alpha=1)
 294 
 295     def plot_group(keys, values, ax):
 296         keys = [pprint_thing(x) for x in keys]
 297         values = [np.asarray(remove_na_arraylike(v)) for v in values]
 298         bp = ax.boxplot(values, **kwds)
 299         if fontsize is not None:
 300             ax.tick_params(axis="both", labelsize=fontsize)
 301         if kwds.get("vert", 1):
 302             ax.set_xticklabels(keys, rotation=rot)
 303         else:
 304             ax.set_yticklabels(keys, rotation=rot)
 305         maybe_color_bp(bp, **kwds)
 306 
 307         # Return axes in multiplot case, maybe revisit later # 985
 308         if return_type == "dict":
 309             return bp
 310         elif return_type == "both":
 311             return BoxPlot.BP(ax=ax, lines=bp)
 312         else:
 313             return ax
 314 
 315     colors = _get_colors()
 316     if column is None:
 317         columns = None
 318     else:
 319         if isinstance(column, (list, tuple)):
 320             columns = column
 321         else:
 322             columns = [column]
 323 
 324     if by is not None:
 325         # Prefer array return type for 2-D plots to match the subplot layout
 326         # https://github.com/pandas-dev/pandas/pull/12216#issuecomment-241175580
 327         result = _grouped_plot_by_column(
 328             plot_group,
 329             data,
 330             columns=columns,
 331             by=by,
 332             grid=grid,
 333             figsize=figsize,
 334             ax=ax,
 335             layout=layout,
 336             return_type=return_type,
 337         )
 338     else:
 339         if return_type is None:
 340             return_type = "axes"
 341         if layout is not None:
 342             raise ValueError("The 'layout' keyword is not supported when 'by' is None")
 343 
 344         if ax is None:
 345             rc = {"figure.figsize": figsize} if figsize is not None else {}
 346             with plt.rc_context(rc):
 347                 ax = plt.gca()
 348         data = data._get_numeric_data()
 349         if columns is None:
 350             columns = data.columns
 351         else:
 352             data = data[columns]
 353 
 354         result = plot_group(columns, data.values.T, ax)
 355         ax.grid(grid)
 356 
 357     return result

 392 def boxplot_frame_groupby(
 393     grouped,
 394     subplots=True,
 395     column=None,
 396     fontsize=None,
 397     rot=0,
 398     grid=True,
 399     ax=None,
 400     figsize=None,
 401     layout=None,
 402     sharex=False,
 403     sharey=True,
 404     **kwds,
 405 ):
 406     if subplots is True:
 407         naxes = len(grouped)
 408         fig, axes = _subplots(
 409             naxes=naxes,
 410             squeeze=False,
 411             ax=ax,
 412             sharex=sharex,
 413             sharey=sharey,
 414             figsize=figsize,
 415             layout=layout,
 416         )
 417         axes = _flatten(axes)
 418 
 419         ret = pd.Series(dtype=object)
 420 
 421         for (key, group), ax in zip(grouped, axes):
 422             d = group.boxplot(
 423                 ax=ax, column=column, fontsize=fontsize, rot=rot, grid=grid, **kwds
 424             )
 425             ax.set_title(pprint_thing(key))
 426             ret.loc[key] = d
 427         fig.subplots_adjust(bottom=0.15, top=0.9, left=0.1, right=0.9, wspace=0.2)
 428     else:
 429         keys, frames = zip(*grouped)
 430         if grouped.axis == 0:
 431             df = pd.concat(frames, keys=keys, axis=1)
 432         else:
 433             if len(frames) > 1:
 434                 df = frames[0].join(frames[1::])
 435             else:
 436                 df = frames[0]
 437         ret = df.boxplot(
 438             column=column,
 439             fontsize=fontsize,
 440             rot=rot,
 441             grid=grid,
 442             ax=ax,
 443             figsize=figsize,
 444             layout=layout,
 445             **kwds,
 446         )
 447     return ret
