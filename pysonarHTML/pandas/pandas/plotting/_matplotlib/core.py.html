 197     def _validate_color_args(self):
 198         import matplotlib.colors
 199 
 200         if (
 201             "color" in self.kwds
 202             and self.nseries == 1
 203             and not is_list_like(self.kwds["color"])
 204         ):
 205             # support series.plot(color='green')
 206             self.kwds["color"] = [self.kwds["color"]]
 207 
 208         if (
 209             "color" in self.kwds
 210             and isinstance(self.kwds["color"], tuple)
 211             and self.nseries == 1
 212             and len(self.kwds["color"]) in (3, 4)
 213         ):
 214             # support RGB and RGBA tuples in series plot
 215             self.kwds["color"] = [self.kwds["color"]]
 216 
 217         if (
 218             "color" in self.kwds or "colors" in self.kwds
 219         ) and self.colormap is not None:
 220             warnings.warn(
 221                 "'color' and 'colormap' cannot be used simultaneously. Using 'color'"
 222             )
 223 
 224         if "color" in self.kwds and self.style is not None:
 225             if is_list_like(self.style):
 226                 styles = self.style
 227             else:
 228                 styles = [self.style]
 229             # need only a single match
 230             for s in styles:
 231                 for char in s:
 232                     if char in matplotlib.colors.BASE_COLORS:
 233                         raise ValueError(
 234                             "Cannot pass 'style' string with a color symbol and "
 235                             "'color' keyword argument. Please use one or the other or "
 236                             "pass 'style' without a color symbol"
 237                         )

 239     def _iter_data(self, data=None, keep_index=False, fillna=None):
 240         if data is None:
 241             data = self.data
 242         if fillna is not None:
 243             data = data.fillna(fillna)
 244 
 245         for col, values in data.items():
 246             if keep_index is True:
 247                 yield col, values
 248             else:
 249                 yield col, values.values

 281     def _maybe_right_yaxis(self, ax, axes_num):
 282         if not self.on_right(axes_num):
 283             # secondary axes may be passed via ax kw
 284             return self._get_ax_layer(ax)
 285 
 286         if hasattr(ax, "right_ax"):
 287             # if it has right_ax property, ``ax`` must be left axes
 288             return ax.right_ax
 289         elif hasattr(ax, "left_ax"):
 290             # if it has left_ax property, ``ax`` must be right axes
 291             return ax
 292         else:
 293             # otherwise, create twin axes
 294             orig_ax, new_ax = ax, ax.twinx()
 295             # TODO: use Matplotlib public API when available
 296             new_ax._get_lines = orig_ax._get_lines
 297             new_ax._get_patches_for_fill = orig_ax._get_patches_for_fill
 298             orig_ax.right_ax, new_ax.left_ax = new_ax, orig_ax
 299 
 300             if not self._has_plotted_object(orig_ax):  # no data on left y
 301                 orig_ax.get_yaxis().set_visible(False)
 302 
 303             if self.logy is True or self.loglog is True:
 304                 new_ax.set_yscale("log")
 305             elif self.logy == "sym" or self.loglog == "sym":
 306                 new_ax.set_yscale("symlog")
 307             return new_ax

 309     def _setup_subplots(self):
 310         if self.subplots:
 311             fig, axes = _subplots(
 312                 naxes=self.nseries,
 313                 sharex=self.sharex,
 314                 sharey=self.sharey,
 315                 figsize=self.figsize,
 316                 ax=self.ax,
 317                 layout=self.layout,
 318                 layout_type=self._layout_type,
 319             )
 320         else:
 321             if self.ax is None:
 322                 fig = self.plt.figure(figsize=self.figsize)
 323                 axes = fig.add_subplot(111)
 324             else:
 325                 fig = self.ax.get_figure()
 326                 if self.figsize is not None:
 327                     fig.set_size_inches(self.figsize)
 328                 axes = self.ax
 329 
 330         axes = _flatten(axes)
 331 
 332         valid_log = {False, True, "sym", None}
 333         input_log = {self.logx, self.logy, self.loglog}
 334         if input_log - valid_log:
 335             invalid_log = next(iter((input_log - valid_log)))
 336             raise ValueError(
 337                 f"Boolean, None and 'sym' are valid options, '{invalid_log}' is given."
 338             )
 339 
 340         if self.logx is True or self.loglog is True:
 341             [a.set_xscale("log") for a in axes]
 342         elif self.logx == "sym" or self.loglog == "sym":
 343             [a.set_xscale("symlog") for a in axes]
 344 
 345         if self.logy is True or self.loglog is True:
 346             [a.set_yscale("log") for a in axes]
 347         elif self.logy == "sym" or self.loglog == "sym":
 348             [a.set_yscale("symlog") for a in axes]
 349 
 350         self.fig = fig
 351         self.axes = axes

 353     @property
 354     def result(self):
 355         """
 356         Return result axes
 357         """
 358         if self.subplots:
 359             if self.layout is not None and not is_list_like(self.ax):
 360                 return self.axes.reshape(*self.layout)
 361             else:
 362                 return self.axes
 363         else:
 364             sec_true = isinstance(self.secondary_y, bool) and self.secondary_y
 365             all_sec = (
 366                 is_list_like(self.secondary_y) and len(self.secondary_y) == self.nseries
 367             )
 368             if sec_true or all_sec:
 369                 # if all data is plotted on secondary, return right axes
 370                 return self._get_ax_layer(self.axes[0], primary=False)
 371             else:
 372                 return self.axes[0]

 374     def _compute_plot_data(self):
 375         data = self.data
 376 
 377         if isinstance(data, ABCSeries):
 378             label = self.label
 379             if label is None and data.name is None:
 380                 label = "None"
 381             data = data.to_frame(name=label)
 382 
 383         # GH16953, _convert is needed as fallback, for ``Series``
 384         # with ``dtype == object``
 385         data = data._convert(datetime=True, timedelta=True)
 386         include_type = [np.number, "datetime", "datetimetz", "timedelta"]
 387 
 388         # GH23719, allow plotting boolean
 389         if self.include_bool is True:
 390             include_type.append(np.bool_)
 391 
 392         # GH22799, exclude datetime-like type for boxplot
 393         exclude_type = None
 394         if self._kind == "box":
 395             # TODO: change after solving issue 27881
 396             include_type = [np.number]
 397             exclude_type = ["timedelta"]
 398 
 399         # GH 18755, include object and category type for scatter plot
 400         if self._kind == "scatter":
 401             include_type.extend(["object", "category"])
 402 
 403         numeric_data = data.select_dtypes(include=include_type, exclude=exclude_type)
 404 
 405         try:
 406             is_empty = numeric_data.columns.empty
 407         except AttributeError:
 408             is_empty = not len(numeric_data)
 409 
 410         # no non-numeric frames or series allowed
 411         if is_empty:
 412             raise TypeError("no numeric data to plot")
 413 
 414         # GH25587: cast ExtensionArray of pandas (IntegerArray, etc.) to
 415         # np.ndarray before plot.
 416         numeric_data = numeric_data.copy()
 417         for col in numeric_data:
 418             numeric_data[col] = np.asarray(numeric_data[col])
 419 
 420         self.data = numeric_data

 435     def _post_plot_logic_common(self, ax, data):
 436         """Common post process for each axes"""
 437         if self.orientation == "vertical" or self.orientation is None:
 438             self._apply_axis_properties(ax.xaxis, rot=self.rot, fontsize=self.fontsize)
 439             self._apply_axis_properties(ax.yaxis, fontsize=self.fontsize)
 440 
 441             if hasattr(ax, "right_ax"):
 442                 self._apply_axis_properties(ax.right_ax.yaxis, fontsize=self.fontsize)
 443 
 444         elif self.orientation == "horizontal":
 445             self._apply_axis_properties(ax.yaxis, rot=self.rot, fontsize=self.fontsize)
 446             self._apply_axis_properties(ax.xaxis, fontsize=self.fontsize)
 447 
 448             if hasattr(ax, "right_ax"):
 449                 self._apply_axis_properties(ax.right_ax.yaxis, fontsize=self.fontsize)
 450         else:  # pragma no cover
 451             raise ValueError

 457     def _adorn_subplots(self):
 458         """Common post process unrelated to data"""
 459         if len(self.axes) > 0:
 460             all_axes = self._get_subplots()
 461             nrows, ncols = self._get_axes_layout()
 462             _handle_shared_axes(
 463                 axarr=all_axes,
 464                 nplots=len(all_axes),
 465                 naxes=nrows * ncols,
 466                 nrows=nrows,
 467                 ncols=ncols,
 468                 sharex=self.sharex,
 469                 sharey=self.sharey,
 470             )
 471 
 472         for ax in self.axes:
 473             if self.yticks is not None:
 474                 ax.set_yticks(self.yticks)
 475 
 476             if self.xticks is not None:
 477                 ax.set_xticks(self.xticks)
 478 
 479             if self.ylim is not None:
 480                 ax.set_ylim(self.ylim)
 481 
 482             if self.xlim is not None:
 483                 ax.set_xlim(self.xlim)
 484 
 485             ax.grid(self.grid)
 486 
 487         if self.title:
 488             if self.subplots:
 489                 if is_list_like(self.title):
 490                     if len(self.title) != self.nseries:
 491                         raise ValueError(
 492                             "The length of `title` must equal the number "
 493                             "of columns if using `title` of type `list` "
 494                             "and `subplots=True`.\n"
 495                             f"length of title = {len(self.title)}\n"
 496                             f"number of columns = {self.nseries}"
 497                         )
 498 
 499                     for (ax, title) in zip(self.axes, self.title):
 500                         ax.set_title(title)
 501                 else:
 502                     self.fig.suptitle(self.title)
 503             else:
 504                 if is_list_like(self.title):
 505                     msg = (
 506                         "Using `title` of type `list` is not supported "
 507                         "unless `subplots=True` is passed"
 508                     )
 509                     raise ValueError(msg)
 510                 self.axes[0].set_title(self.title)

 512     def _apply_axis_properties(self, axis, rot=None, fontsize=None):
 513         """
 514         Tick creation within matplotlib is reasonably expensive and is
 515         internally deferred until accessed as Ticks are created/destroyed
 516         multiple times per draw. It's therefore beneficial for us to avoid
 517         accessing unless we will act on the Tick.
 518         """
 519         if rot is not None or fontsize is not None:
 520             # rot=0 is a valid setting, hence the explicit None check
 521             labels = axis.get_majorticklabels() + axis.get_minorticklabels()
 522             for label in labels:
 523                 if rot is not None:
 524                     label.set_rotation(rot)
 525                 if fontsize is not None:
 526                     label.set_fontsize(fontsize)

 528     @property
 529     def legend_title(self):
 530         if not isinstance(self.data.columns, ABCMultiIndex):
 531             name = self.data.columns.name
 532             if name is not None:
 533                 name = pprint_thing(name)
 534             return name
 535         else:
 536             stringified = map(pprint_thing, self.data.columns.names)
 537             return ",".join(stringified)

 539     def _add_legend_handle(self, handle, label, index=None):
 540         if label is not None:
 541             if self.mark_right and index is not None:
 542                 if self.on_right(index):
 543                     label = label + " (right)"
 544             self.legend_handles.append(handle)
 545             self.legend_labels.append(label)

 547     def _make_legend(self):
 548         ax, leg, handle = self._get_ax_legend_handle(self.axes[0])
 549 
 550         handles = []
 551         labels = []
 552         title = ""
 553 
 554         if not self.subplots:
 555             if leg is not None:
 556                 title = leg.get_title().get_text()
 557                 # Replace leg.LegendHandles because it misses marker info
 558                 handles.extend(handle)
 559                 labels = [x.get_text() for x in leg.get_texts()]
 560 
 561             if self.legend:
 562                 if self.legend == "reverse":
 563                     self.legend_handles = reversed(self.legend_handles)
 564                     self.legend_labels = reversed(self.legend_labels)
 565 
 566                 handles += self.legend_handles
 567                 labels += self.legend_labels
 568 
 569                 if self.legend_title is not None:
 570                     title = self.legend_title
 571 
 572             if len(handles) > 0:
 573                 ax.legend(handles, labels, loc="best", title=title)
 574 
 575         elif self.subplots and self.legend:
 576             for ax in self.axes:
 577                 if ax.get_visible():
 578                     ax.legend(loc="best")

 580     def _get_ax_legend_handle(self, ax):
 581         """
 582         Take in axes and return ax, legend and handle under different scenarios
 583         """
 584         leg = ax.get_legend()
 585 
 586         # Get handle from axes
 587         handle, _ = ax.get_legend_handles_labels()
 588         other_ax = getattr(ax, "left_ax", None) or getattr(ax, "right_ax", None)
 589         other_leg = None
 590         if other_ax is not None:
 591             other_leg = other_ax.get_legend()
 592         if leg is None and other_leg is not None:
 593             leg = other_leg
 594             ax = other_ax
 595         return ax, leg, handle

 605     def _get_xticks(self, convert_period=False):
 606         index = self.data.index
 607         is_datetype = index.inferred_type in ("datetime", "date", "datetime64", "time")
 608 
 609         if self.use_index:
 610             if convert_period and isinstance(index, ABCPeriodIndex):
 611                 self.data = self.data.reindex(index=index.sort_values())
 612                 x = self.data.index.to_timestamp()._mpl_repr()
 613             elif index.is_numeric():
 614                 """
 615                 Matplotlib supports numeric values or datetime objects as
 616                 xaxis values. Taking LBYL approach here, by the time
 617                 matplotlib raises exception when using non numeric/datetime
 618                 values for xaxis, several actions are already taken by plt.
 619                 """
 620                 x = index._mpl_repr()
 621             elif is_datetype:
 622                 self.data = self.data[notna(self.data.index)]
 623                 self.data = self.data.sort_index()
 624                 x = self.data.index._mpl_repr()
 625             else:
 626                 self._need_to_set_index = True
 627                 x = list(range(len(index)))
 628         else:
 629             x = list(range(len(index)))
 630 
 631         return x

 746     def _parse_errorbars(self, label, err):
 747         """
 748         Look for error keyword arguments and return the actual errorbar data
 749         or return the error DataFrame/dict
 750 
 751         Error bars can be specified in several ways:
 752             Series: the user provides a pandas.Series object of the same
 753                     length as the data
 754             ndarray: provides a np.ndarray of the same length as the data
 755             DataFrame/dict: error values are paired with keys matching the
 756                     key in the plotted DataFrame
 757             str: the name of the column within the plotted DataFrame
 758         """
 759         if err is None:
 760             return None
 761 
 762         def match_labels(data, e):
 763             e = e.reindex(data.index)
 764             return e
 765 
 766         # key-matched DataFrame
 767         if isinstance(err, ABCDataFrame):
 768 
 769             err = match_labels(self.data, err)
 770         # key-matched dict
 771         elif isinstance(err, dict):
 772             pass
 773 
 774         # Series of error values
 775         elif isinstance(err, ABCSeries):
 776             # broadcast error series across data
 777             err = match_labels(self.data, err)
 778             err = np.atleast_2d(err)
 779             err = np.tile(err, (self.nseries, 1))
 780 
 781         # errors are a column in the dataframe
 782         elif isinstance(err, str):
 783             evalues = self.data[err].values
 784             self.data = self.data[self.data.columns.drop(err)]
 785             err = np.atleast_2d(evalues)
 786             err = np.tile(err, (self.nseries, 1))
 787 
 788         elif is_list_like(err):
 789             if is_iterator(err):
 790                 err = np.atleast_2d(list(err))
 791             else:
 792                 # raw error values
 793                 err = np.atleast_2d(err)
 794 
 795             err_shape = err.shape
 796 
 797             # asymmetrical error bars
 798             if err.ndim == 3:
 799                 if (
 800                     (err_shape[0] != self.nseries)
 801                     or (err_shape[1] != 2)
 802                     or (err_shape[2] != len(self.data))
 803                 ):
 804                     raise ValueError(
 805                         "Asymmetrical error bars should be provided "
 806                         f"with the shape ({self.nseries}, 2, {len(self.data)})"
 807                     )
 808 
 809             # broadcast errors to each data series
 810             if len(err) == 1:
 811                 err = np.tile(err, (self.nseries, 1))
 812 
 813         elif is_number(err):
 814             err = np.tile([err], (self.nseries, len(self.data)))
 815 
 816         else:
 817             msg = f"No valid {label} detected"
 818             raise ValueError(msg)
 819 
 820         return err
 821 
 822     def _get_errorbars(self, label=None, index=None, xerr=True, yerr=True):
 823         errors = {}
 824 
 825         for kw, flag in zip(["xerr", "yerr"], [xerr, yerr]):
 826             if flag:
 827                 err = self.errors[kw]
 828                 # user provided label-matched dataframe of errors
 829                 if isinstance(err, (ABCDataFrame, dict)):
 830                     if label is not None and label in err.keys():
 831                         err = err[label]
 832                     else:
 833                         err = None
 834                 elif index is not None and err is not None:
 835                     err = err[index]
 836 
 837                 if err is not None:
 838                     errors[kw] = err
 839         return errors

 895     def _plot_colorbar(self, ax, **kwds):
 896         # Addresses issues #10611 and #10678:
 897         # When plotting scatterplots and hexbinplots in IPython
 898         # inline backend the colorbar axis height tends not to
 899         # exactly match the parent axis height.
 900         # The difference is due to small fractional differences
 901         # in floating points with similar representation.
 902         # To deal with this, this method forces the colorbar
 903         # height to take the height of the parent axes.
 904         # For a more detailed description of the issue
 905         # see the following link:
 906         # https://github.com/ipython/ipython/issues/11215

 908         # GH33389, if ax is used multiple times, we should always
 909         # use the last one which contains the latest information
 910         # about the ax
 911         img = ax.collections[-1]
 912         cbar = self.fig.colorbar(img, ax=ax, **kwds)
 913 
 914         if _mpl_ge_3_0_0():
 915             # The workaround below is no longer necessary.
 916             return
 917 
 918         points = ax.get_position().get_points()
 919         cbar_points = cbar.ax.get_position().get_points()
 920 
 921         cbar.ax.set_position(
 922             [
 923                 cbar_points[0, 0],
 924                 points[0, 1],
 925                 cbar_points[1, 0] - cbar_points[0, 0],
 926                 points[1, 1] - points[0, 1],
 927             ]
 928         )
 929         # To see the discrepancy in axis heights uncomment
 930         # the following two lines:
 931         # print(points[1, 1] - points[0, 1])
