 211 _boxplot_doc = """
 212 Make a box plot from DataFrame columns.
 213 
 214 Make a box-and-whisker plot from DataFrame columns, optionally grouped
 215 by some other columns. A box plot is a method for graphically depicting
 216 groups of numerical data through their quartiles.
 217 The box extends from the Q1 to Q3 quartile values of the data,
 218 with a line at the median (Q2). The whiskers extend from the edges
 219 of box to show the range of the data. By default, they extend no more than
 220 `1.5 * IQR (IQR = Q3 - Q1)` from the edges of the box, ending at the farthest
 221 data point within that interval. Outliers are plotted as separate dots.
 222 
 223 For further details see
 224 Wikipedia's entry for `boxplot <https://en.wikipedia.org/wiki/Box_plot>`_.
 225 
 226 Parameters
 227 ----------
 228 column : str or list of str, optional
 229     Column name or list of names, or vector.
 230     Can be any valid input to :meth:`pandas.DataFrame.groupby`.
 231 by : str or array-like, optional
 232     Column in the DataFrame to :meth:`pandas.DataFrame.groupby`.
 233     One box-plot will be done per value of columns in `by`.
 234 ax : object of class matplotlib.axes.Axes, optional
 235     The matplotlib axes to be used by boxplot.
 236 fontsize : float or str
 237     Tick label font size in points or as a string (e.g., `large`).
 238 rot : int or float, default 0
 239     The rotation angle of labels (in degrees)
 240     with respect to the screen coordinate system.
 241 grid : bool, default True
 242     Setting this to True will show the grid.
 243 figsize : A tuple (width, height) in inches
 244     The size of the figure to create in matplotlib.
 245 layout : tuple (rows, columns), optional
 246     For example, (3, 5) will display the subplots
 247     using 3 columns and 5 rows, starting from the top-left.
 248 return_type : {'axes', 'dict', 'both'} or None, default 'axes'
 249     The kind of object to return. The default is ``axes``.
 250 
 251     * 'axes' returns the matplotlib axes the boxplot is drawn on.
 252     * 'dict' returns a dictionary whose values are the matplotlib
 253       Lines of the boxplot.
 254     * 'both' returns a namedtuple with the axes and dict.
 255     * when grouping with ``by``, a Series mapping columns to
 256       ``return_type`` is returned.
 257 
 258       If ``return_type`` is `None`, a NumPy array
 259       of axes with the same shape as ``layout`` is returned.
 260 %(backend)s\
 261 
 262 **kwargs
 263     All other plotting keyword arguments to be passed to
 264     :func:`matplotlib.pyplot.boxplot`.
 265 
 266 Returns
 267 -------
 268 result
 269     See Notes.
 270 
 271 See Also
 272 --------
 273 Series.plot.hist: Make a histogram.
 274 matplotlib.pyplot.boxplot : Matplotlib equivalent plot.
 275 
 276 Notes
 277 -----
 278 The return type depends on the `return_type` parameter:
 279 
 280 * 'axes' : object of class matplotlib.axes.Axes
 281 * 'dict' : dict of matplotlib.lines.Line2D objects
 282 * 'both' : a namedtuple with structure (ax, lines)
 283 
 284 For data grouped with ``by``, return a Series of the above or a numpy
 285 array:
 286 
 287 * :class:`~pandas.Series`
 288 * :class:`~numpy.array` (for ``return_type = None``)

 292 making up the boxes, caps, fliers, medians, and whiskers is returned.

 296

 310 
 311 .. plot::

 322

 324     :context: close-figs
 325 
 326     >>> df = pd.DataFrame(np.random.randn(10, 3),
 327     ...                   columns=['Col1', 'Col2', 'Col3'])
 328     >>> df['X'] = pd.Series(['A', 'A', 'A', 'A', 'A',

1667     """
1668     Find a pandas plotting backend>
1669 
1670     Parameters
1671     ----------
1672     backend : str
1673         The identifier for the backend. Either an entrypoint item registered
1674         with pkg_resources, or a module name.
1675 
1676     Notes
1677     -----
1678     Modifies _backends with imported backends as a side effect.
1679 
1680     Returns
1681     -------
1682     types.ModuleType
1683         The imported backend.
1684     """
1685     import pkg_resources  # Delay import for performance.
1686 
1687     for entry_point in pkg_resources.iter_entry_points("pandas_plotting_backends"):
1688         if entry_point.name == "matplotlib":
1689             # matplotlib is an optional dependency. When
1690             # missing, this would raise.
1691             continue
1692         _backends[entry_point.name] = entry_point.load()
1693 
1694     try:
1695         return _backends[backend]
1696     except KeyError:
1697         # Fall back to unregistered, module name approach.
1698         try:
1699             module = importlib.import_module(backend)
1700         except ImportError:
1701             # We re-raise later on.
1702             pass
1703         else:
1704             if hasattr(module, "plot"):
1705                 # Validate that the interface is implemented when the option
1706                 # is set, rather than at plot time.
1707                 _backends[backend] = module
1708                 return module
1709 
1710     raise ValueError(
1711         f"Could not find plotting backend '{backend}'. Ensure that you've installed "
1712         f"the package providing the '{backend}' entrypoint, or that the package has a "
1713         "top-level `.plot` method."
1714     )

1717 def _get_plot_backend(backend=None):
1718     """
1719     Return the plotting backend to use (e.g. `pandas.plotting._matplotlib`).
1720 
1721     The plotting system of pandas has been using matplotlib, but the idea here
1722     is that it can also work with other third-party backends. In the future,
1723     this function will return the backend from a pandas option, and all the
1724     rest of the code in this file will use the backend specified there for the
1725     plotting.
1726 
1727     The backend is imported lazily, as matplotlib is a soft dependency, and
1728     pandas can be used without it being installed.
1729     """
1730     backend = backend or get_option("plotting.backend")
1731 
1732     if backend == "matplotlib":
1733         # Because matplotlib is an optional dependency and first-party backend,
1734         # we need to attempt an import here to raise an ImportError if needed.
1735         try:
1736             import pandas.plotting._matplotlib as module
1737         except ImportError:
1738             raise ImportError(
1739                 "matplotlib is required for plotting when the "
1740                 'default backend "matplotlib" is selected.'
1741             ) from None
1742 
1743         _backends["matplotlib"] = module
1744 
1745     if backend in _backends:
1746         return _backends[backend]
1747 
1748     module = _find_backend(backend)
1749     _backends[backend] = module
