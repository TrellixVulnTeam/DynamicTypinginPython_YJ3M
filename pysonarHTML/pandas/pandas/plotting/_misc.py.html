  30 def register():
  31     """
  32     Register pandas formatters and converters with matplotlib.
  33 
  34     This function modifies the global ``matplotlib.units.registry``
  35     dictionary. pandas adds custom converters for
  36 
  37     * pd.Timestamp
  38     * pd.Period
  39     * np.datetime64
  40     * datetime.datetime
  41     * datetime.date
  42     * datetime.time
  43 
  44     See Also
  45     --------
  46     deregister_matplotlib_converters : Remove pandas formatters and converters.
  47     """
  48     plot_backend = _get_plot_backend("matplotlib")
  49     plot_backend.register()

  52 def deregister():
  53     """
  54     Remove pandas formatters and converters.
  55 
  56     Removes the custom converters added by :func:`register`. This
  57     attempts to set the state of the registry back to the state before
  58     pandas registered its own units. Converters for pandas' own types like
  59     Timestamp and Period are removed completely. Converters for types
  60     pandas overwrites, like ``datetime.datetime``, are restored to their
  61     original value.
  62 
  63     See Also
  64     --------
  65     register_matplotlib_converters : Register pandas formatters and converters
  66         with matplotlib.
  67     """
  68     plot_backend = _get_plot_backend("matplotlib")
  69     plot_backend.deregister()

 143 def radviz(frame, class_column, ax=None, color=None, colormap=None, **kwds):
 144     """
 145     Plot a multidimensional dataset in 2D.
 146 
 147     Each Series in the DataFrame is represented as a evenly distributed
 148     slice on a circle. Each data point is rendered in the circle according to
 149     the value on each Series. Highly correlated `Series` in the `DataFrame`
 150     are placed closer on the unit circle.
 151 
 152     RadViz allow to project a N-dimensional data set into a 2D space where the
 153     influence of each dimension can be interpreted as a balance between the
 154     influence of all dimensions.
 155 
 156     More info available at the `original article
 157     <https://citeseerx.ist.psu.edu/viewdoc/summary?doi=10.1.1.135.889>`_
 158     describing RadViz.
 159 
 160     Parameters
 161     ----------
 162     frame : `DataFrame`
 163         Object holding the data.
 164     class_column : str
 165         Column name containing the name of the data point category.
 166     ax : :class:`matplotlib.axes.Axes`, optional
 167         A plot instance to which to add the information.
 168     color : list[str] or tuple[str], optional
 169         Assign a color to each category. Example: ['blue', 'green'].
 170     colormap : str or :class:`matplotlib.colors.Colormap`, default None
 171         Colormap to select colors from. If string, load colormap with that
 172         name from matplotlib.
 173     **kwds
 174         Options to pass to matplotlib scatter plotting method.
 175 
 176     Returns
 177     -------
 178     class:`matplotlib.axes.Axes`
 179 
 180     See Also
 181     --------
 182     plotting.andrews_curves : Plot clustering visualization.
 183 
 184     Examples
 185     --------
 186 
 187     .. plot::
 188         :context: close-figs
 189 
 190         >>> df = pd.DataFrame(
 191         ...     {
 192         ...         'SepalLength': [6.5, 7.7, 5.1, 5.8, 7.6, 5.0, 5.4, 4.6, 6.7, 4.6],
 193         ...         'SepalWidth': [3.0, 3.8, 3.8, 2.7, 3.0, 2.3, 3.0, 3.2, 3.3, 3.6],
 194         ...         'PetalLength': [5.5, 6.7, 1.9, 5.1, 6.6, 3.3, 4.5, 1.4, 5.7, 1.0],
 195         ...         'PetalWidth': [1.8, 2.2, 0.4, 1.9, 2.1, 1.0, 1.5, 0.2, 2.1, 0.2],
 196         ...         'Category': [
 197         ...             'virginica',
 198         ...             'virginica',
 199         ...             'setosa',
 200         ...             'virginica',
 201         ...             'virginica',
 202         ...             'versicolor',
 203         ...             'versicolor',
 204         ...             'setosa',
 205         ...             'virginica',
 206         ...             'setosa'
 207         ...         ]
 208         ...     }
 209         ... )
 210         >>> pd.plotting.radviz(df, 'Category')
 211     """
 212     plot_backend = _get_plot_backend("matplotlib")
 213     return plot_backend.radviz(
 214         frame=frame,
 215         class_column=class_column,
 216         ax=ax,
 217         color=color,
 218         colormap=colormap,
 219         **kwds,
 220     )

 223 def andrews_curves(
 224     frame, class_column, ax=None, samples=200, color=None, colormap=None, **kwargs
 225 ):
 226     """
 227     Generate a matplotlib plot of Andrews curves, for visualising clusters of
 228     multivariate data.
 229 
 230     Andrews curves have the functional form:
 231 
 232     f(t) = x_1/sqrt(2) + x_2 sin(t) + x_3 cos(t) +
 233            x_4 sin(2t) + x_5 cos(2t) + ...
 234 
 235     Where x coefficients correspond to the values of each dimension and t is
 236     linearly spaced between -pi and +pi. Each row of frame then corresponds to
 237     a single curve.
 238 
 239     Parameters
 240     ----------
 241     frame : DataFrame
 242         Data to be plotted, preferably normalized to (0.0, 1.0).
 243     class_column : Name of the column containing class names
 244     ax : matplotlib axes object, default None
 245     samples : Number of points to plot in each curve
 246     color : list or tuple, optional
 247         Colors to use for the different classes.
 248     colormap : str or matplotlib colormap object, default None
 249         Colormap to select colors from. If string, load colormap with that name
 250         from matplotlib.
 251     **kwargs
 252         Options to pass to matplotlib plotting method.
 253 
 254     Returns
 255     -------
 256     class:`matplotlip.axis.Axes`
 257 
 258     Examples
 259     --------
 260 
 261     .. plot::
 262         :context: close-figs
 263 
 264         >>> df = pd.read_csv(
 265         ...     'https://raw.github.com/pandas-dev/'
 266         ...     'pandas/master/pandas/tests/io/data/csv/iris.csv'
 267         ... )
 268         >>> pd.plotting.andrews_curves(df, 'Name')
 269     """
 270     plot_backend = _get_plot_backend("matplotlib")
 271     return plot_backend.andrews_curves(
 272         frame=frame,
 273         class_column=class_column,
 274         ax=ax,
 275         samples=samples,
 276         color=color,
 277         colormap=colormap,
 278         **kwargs,
 279     )

 282 def bootstrap_plot(series, fig=None, size=50, samples=500, **kwds):
 283     """
 284     Bootstrap plot on mean, median and mid-range statistics.
 285 
 286     The bootstrap plot is used to estimate the uncertainty of a statistic
 287     by relaying on random sampling with replacement [1]_. This function will
 288     generate bootstrapping plots for mean, median and mid-range statistics
 289     for the given number of samples of the given size.
 290 
 291     .. [1] "Bootstrapping (statistics)" in \
 292     https://en.wikipedia.org/wiki/Bootstrapping_%28statistics%29
 293 
 294     Parameters
 295     ----------
 296     series : pandas.Series
 297         Series from where to get the samplings for the bootstrapping.
 298     fig : matplotlib.figure.Figure, default None
 299         If given, it will use the `fig` reference for plotting instead of
 300         creating a new one with default parameters.
 301     size : int, default 50
 302         Number of data points to consider during each sampling. It must be
 303         greater or equal than the length of the `series`.
 304     samples : int, default 500
 305         Number of times the bootstrap procedure is performed.
 306     **kwds
 307         Options to pass to matplotlib plotting method.
 308 
 309     Returns
 310     -------
 311     matplotlib.figure.Figure
 312         Matplotlib figure.
 313 
 314     See Also
 315     --------
 316     DataFrame.plot : Basic plotting for DataFrame objects.
 317     Series.plot : Basic plotting for Series objects.
 318 
 319     Examples
 320     --------
 321     This example draws a basic bootstap plot for a Series.
 322 
 323     .. plot::
 324         :context: close-figs
 325 
 326         >>> s = pd.Series(np.random.uniform(size=100))
 327         >>> pd.plotting.bootstrap_plot(s)
 328     """
 329     plot_backend = _get_plot_backend("matplotlib")
 330     return plot_backend.bootstrap_plot(
 331         series=series, fig=fig, size=size, samples=samples, **kwds
 332     )

 488 class _Options(dict):
 489     """
 490     Stores pandas plotting options.
 491 
 492     Allows for parameter aliasing so you can just use parameter names that are
 493     the same as the plot function parameters, but is stored in a canonical
 494     format that makes it easy to breakdown into groups later.
 495     """
 496 
 497     # alias so the names are same as plotting method parameter names
 498     _ALIASES = {"x_compat": "xaxis.compat"}
 499     _DEFAULT_KEYS = ["xaxis.compat"]
 500 
 501     def __init__(self, deprecated=False):
 502         self._deprecated = deprecated
 503         super().__setitem__("xaxis.compat", False)
 504 
 505     def __getitem__(self, key):
 506         key = self._get_canonical_key(key)
 507         if key not in self:
 508             raise ValueError(f"{key} is not a valid pandas plotting option")
 509         return super().__getitem__(key)
 510 
 511     def __setitem__(self, key, value):
 512         key = self._get_canonical_key(key)
 513         return super().__setitem__(key, value)
 514 
 515     def __delitem__(self, key):
 516         key = self._get_canonical_key(key)
 517         if key in self._DEFAULT_KEYS:
 518             raise ValueError(f"Cannot remove default parameter {key}")
 519         return super().__delitem__(key)
 520 
 521     def __contains__(self, key) -> bool:
 522         key = self._get_canonical_key(key)
 523         return super().__contains__(key)
 524 
 525     def reset(self):
 526         """
 527         Reset the option store to its initial state
 528 
 529         Returns
 530         -------
 531         None
 532         """
 533         self.__init__()
 534 
 535     def _get_canonical_key(self, key):
 536         return self._ALIASES.get(key, key)
 537 
 538     @contextmanager
 539     def use(self, key, value):
 540         """
 541         Temporarily set a parameter value using the with statement.
 542         Aliasing allowed.
 543         """
 544         old_value = self[key]
 545         try:
 546             self[key] = value
 547             yield self
 548         finally:
 549             self[key] = old_value
