  20 def load_reduce(self):
  21     stack = self.stack
  22     args = stack.pop()
  23     func = stack[-1]
  24 
  25     if len(args) and type(args[0]) is type:
  26         n = args[0].__name__  # noqa
  27 
  28     try:
  29         stack[-1] = func(*args)
  30         return
  31     except TypeError as err:
  32 
  33         # If we have a deprecated function,
  34         # try to replace and try again.
  35 
  36         msg = "_reconstruct: First argument must be a sub-type of ndarray"
  37 
  38         if msg in str(err):
  39             try:
  40                 cls = args[0]
  41                 stack[-1] = object.__new__(cls)
  42                 return
  43             except TypeError:
  44                 pass
  45         elif args and issubclass(args[0], DateOffset):
  46             # TypeError: object.__new__(Day) is not safe, use Day.__new__()
  47             cls = args[0]
  48             stack[-1] = cls.__new__(*args)
  49             return
  50 
  51         raise

 181 # functions for compat and uses a non-public class of the pickle module.

 183 # error: Name 'pkl._Unpickler' is not defined
 184 class Unpickler(pkl._Unpickler):  # type: ignore
 185     def find_class(self, module, name):
 186         # override superclass
 187         key = (module, name)
 188         module, name = _class_locations_map.get(key, key)
 189         return super().find_class(module, name)

 196 def load_newobj(self):
 197     args = self.stack.pop()
 198     cls = self.stack[-1]
 199 
 200     # compat
 201     if issubclass(cls, Index):
 202         obj = object.__new__(cls)
 203     else:
 204         obj = cls.__new__(cls, *args)
 205 
 206     self.stack[-1] = obj
 207 
 208 
 209 Unpickler.dispatch[pkl.NEWOBJ[0]] = load_newobj

 212 def load_newobj_ex(self):
 213     kwargs = self.stack.pop()
 214     args = self.stack.pop()
 215     cls = self.stack.pop()
 216 
 217     # compat
 218     if issubclass(cls, Index):
 219         obj = object.__new__(cls)
 220     else:
 221         obj = cls.__new__(cls, *args, **kwargs)
 222     self.append(obj)
 223

 225 try:
 226     Unpickler.dispatch[pkl.NEWOBJ_EX[0]] = load_newobj_ex
 227 except (AttributeError, KeyError):
 228     pass

 231 def load(fh, encoding: Optional[str] = None, is_verbose: bool = False):
 232     """
 233     Load a pickle, with a provided encoding,
 234 
 235     Parameters
 236     ----------
 237     fh : a filelike object
 238     encoding : an optional encoding
 239     is_verbose : show exception output
 240     """
 241     try:
 242         fh.seek(0)
 243         if encoding is not None:
 244             up = Unpickler(fh, encoding=encoding)
 245         else:
 246             up = Unpickler(fh)
 247         up.is_verbose = is_verbose
 248 
 249         return up.load()
 250     except (ValueError, TypeError):
 251         raise

 254 def loads(
 255     bytes_object: bytes,
 256     *,
 257     fix_imports: bool = True,
 258     encoding: str = "ASCII",
 259     errors: str = "strict",
 260 ):
 261     """
 262     Analogous to pickle._loads.
 263     """
 264     fd = io.BytesIO(bytes_object)
 265     return Unpickler(
 266         fd, fix_imports=fix_imports, encoding=encoding, errors=errors
 267     ).load()

 270 @contextlib.contextmanager
 271 def patch_pickle():
 272     """
 273     Temporarily patch pickle to use our unpickler.
 274     """
 275     orig_loads = pkl.loads
 276     try:
 277         pkl.loads = loads
 278         yield
 279     finally:
 280         pkl.loads = orig_loads
