  31 def set_function_name(f: F, name: str, cls) -> F:
  32     """
  33     Bind the name/qualname attributes of the function.
  34     """
  35     f.__name__ = name
  36     f.__qualname__ = f"{cls.__name__}.{name}"
  37     f.__module__ = cls.__module__
  38     return f

  42 def is_platform_little_endian() -> bool:

  91     """

 102 def _import_lzma():
 103     """
 104     Importing the `lzma` module.
 105 
 106     Warns
 107     -----
 108     When the `lzma` module is not available.
 109     """
 110     try:
 111         import lzma
 112 
 113         return lzma
 114     except ImportError:
 115         msg = (
 116             "Could not import the lzma module. Your installed Python is incomplete. "
 117             "Attempting to use lzma compression will result in a RuntimeError."
 118         )
 119         warnings.warn(msg)

 122 def _get_lzma_file(lzma):
 123     """
 124     Importing the `LZMAFile` class from the `lzma` module.
 125 
 126     Returns
 127     -------
 128     class
 129         The `LZMAFile` class from the `lzma` module.
 130 
 131     Raises
 132     ------
 133     RuntimeError
 134         If the `lzma` module was not imported correctly, or didn't exist.
 135     """
 136     if lzma is None:
 137         raise RuntimeError(
 138             "lzma module not available. "
 139             "A Python re-install with the proper dependencies, "
 140             "might be required to solve this issue."
 141         )
 142     return lzma.LZMAFile
