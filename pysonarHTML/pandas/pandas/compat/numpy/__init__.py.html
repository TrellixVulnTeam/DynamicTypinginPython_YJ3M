   1 """ support numpy compatibility across versions """
   2 
   3 from distutils.version import LooseVersion
   4 import re
   5 
   6 import numpy as np
   7 
   8 # numpy versioning
   9 _np_version = np.__version__
  10 _nlv = LooseVersion(_np_version)
  11 _np_version_under1p16 = _nlv < LooseVersion("1.16")
  12 _np_version_under1p17 = _nlv < LooseVersion("1.17")
  13 _np_version_under1p18 = _nlv < LooseVersion("1.18")
  14 _np_version_under1p19 = _nlv < LooseVersion("1.19")
  15 _np_version_under1p20 = _nlv < LooseVersion("1.20")
  16 _is_numpy_dev = ".dev" in str(_nlv)
  17 
  18 
  19 if _nlv < "1.15.4":
  20     raise ImportError(
  21         "this version of pandas is incompatible with numpy < 1.15.4\n"
  22         f"your numpy version is {_np_version}.\n"
  23         "Please upgrade numpy to >= 1.15.4 to use this pandas version"
  24     )
  25 
  26 
  27 _tz_regex = re.compile("[+-]0000$")
  28 
  29 
  30 def tz_replacer(s):
  31     if isinstance(s, str):
  32         if s.endswith("Z"):
  33             s = s[:-1]
  34         elif _tz_regex.search(s):
  35             s = s[:-5]
  36     return s
  37 
  38 
  39 def np_datetime64_compat(s, *args, **kwargs):
  40     """
  41     provide compat for construction of strings to numpy datetime64's with
  42     tz-changes in 1.11 that make '2015-01-01 09:00:00Z' show a deprecation
  43     warning, when need to pass '2015-01-01 09:00:00'
  44     """
  45     s = tz_replacer(s)
  46     return np.datetime64(s, *args, **kwargs)
  47 
  48 
  49 def np_array_datetime64_compat(arr, *args, **kwargs):
  50     """
  51     provide compat for construction of an array of strings to a
  52     np.array(..., dtype=np.datetime64(..))
  53     tz-changes in 1.11 that make '2015-01-01 09:00:00Z' show a deprecation
  54     warning, when need to pass '2015-01-01 09:00:00'
  55     """
  56     # is_list_like
  57     if hasattr(arr, "__iter__") and not isinstance(arr, (str, bytes)):
  58         arr = [tz_replacer(s) for s in arr]
  59     else:
  60         arr = tz_replacer(arr)
  61 
  62     return np.array(arr, *args, **kwargs)
