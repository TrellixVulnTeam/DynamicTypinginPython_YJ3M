   1 """
   2 For compatibility with numpy libraries, pandas functions or
   3 methods have to accept '*args' and '**kwargs' parameters to
   4 accommodate numpy arguments that are not actually used or
   5 respected in the pandas implementation.
   6 
   7 To ensure that users do not abuse these parameters, validation
   8 is performed in 'validators.py' to make sure that any extra
   9 parameters passed correspond ONLY to those in the numpy signature.
  10 Part of that validation includes whether or not the user attempted
  11 to pass in non-default values for these extraneous parameters. As we
  12 want to discourage users from relying on these parameters when calling
  13 the pandas implementation, we want them only to pass in the default values
  14 for these parameters.
  15 
  16 This module provides a set of commonly used default arguments for functions
  17 and methods that are spread throughout the codebase. This module will make it
  18 easier to adjust to future upstream changes in the analogous numpy signatures.
  19 """
  20 from collections import OrderedDict
  21 from distutils.version import LooseVersion
  22 from typing import Any, Dict, Optional, Union

  35 class CompatValidator:
  36     def __init__(
  37         self,
  38         defaults,
  39         fname=None,
  40         method: Optional[str] = None,
  41         max_fname_arg_count=None,
  42     ):
  43         self.fname = fname
  44         self.method = method
  45         self.defaults = defaults
  46         self.max_fname_arg_count = max_fname_arg_count
  47 
  48     def __call__(
  49         self,
  50         args,
  51         kwargs,
  52         fname=None,
  53         max_fname_arg_count=None,
  54         method: Optional[str] = None,
  55     ) -> None:
  56         if args or kwargs:
  57             fname = self.fname if fname is None else fname
  58             max_fname_arg_count = (
  59                 self.max_fname_arg_count
  60                 if max_fname_arg_count is None
  61                 else max_fname_arg_count
  62             )
  63             method = self.method if method is None else method
  64 
  65             if method == "args":
  66                 validate_args(fname, args, max_fname_arg_count, self.defaults)
  67             elif method == "kwargs":
  68                 validate_kwargs(fname, kwargs, self.defaults)
  69             elif method == "both":
  70                 validate_args_and_kwargs(
  71                     fname, args, kwargs, max_fname_arg_count, self.defaults
  72                 )
  73             else:
  74                 raise ValueError(f"invalid validation method '{method}'")

  86 def process_skipna(skipna, args):
  87     if isinstance(skipna, ndarray) or skipna is None:
  88         args = (skipna,) + args
  89         skipna = True
  90 
  91     return skipna, args

  94 def validate_argmin_with_skipna(skipna, args, kwargs):
  95     """
  96     If 'Series.argmin' is called via the 'numpy' library,
  97     the third parameter in its signature is 'out', which
  98     takes either an ndarray or 'None', so check if the
  99     'skipna' parameter is either an instance of ndarray or
 100     is None, since 'skipna' itself should be a boolean
 101     """
 102     skipna, args = process_skipna(skipna, args)
 103     validate_argmin(args, kwargs)
 104     return skipna

 317 def validate_window_func(name, args, kwargs) -> None:
 318     numpy_args = ("axis", "dtype", "out")
 319     msg = (
 320         f"numpy operations are not valid with window objects. "
 321         f"Use .{name}() directly instead "
 322     )
 323 
 324     if len(args) > 0:
 325         raise UnsupportedFunctionCall(msg)
 326 
 327     for arg in numpy_args:
 328         if arg in kwargs:
 329             raise UnsupportedFunctionCall(msg)

 362 def validate_groupby_func(name, args, kwargs, allowed=None) -> None:
 363     """
 364     'args' and 'kwargs' should be empty, except for allowed
 365     kwargs because all of
 366     their necessary parameters are explicitly listed in
 367     the function signature
 368     """
 369     if allowed is None:
 370         allowed = []
 371 
 372     kwargs = set(kwargs) - set(allowed)
 373 
 374     if len(args) + len(kwargs) > 0:
 375         raise UnsupportedFunctionCall(
 376             "numpy operations are not valid with groupby. "
 377             f"Use .groupby(...).{name}() instead"
 378         )

 384 def validate_resampler_func(method: str, args, kwargs) -> None:
 385     """
 386     'args' and 'kwargs' should be empty because all of
 387     their necessary parameters are explicitly listed in
 388     the function signature
 389     """
 390     if len(args) + len(kwargs) > 0:
 391         if method in RESAMPLER_NUMPY_OPS:
 392             raise UnsupportedFunctionCall(
 393                 "numpy operations are not valid with resample. "
 394                 f"Use .resample(...).{method}() instead"
 395             )
 396         else:
 397             raise TypeError("too many arguments passed in")

 400 def validate_minmax_axis(axis: Optional[int]) -> None:
 401     """
 402     Ensure that the axis argument passed to min, max, argmin, or argmax is
 403     zero or None, as otherwise it will be incorrectly ignored.
 404 
 405     Parameters
 406     ----------
 407     axis : int or None
 408 
 409     Raises
 410     ------
 411     ValueError
 412     """
 413     ndim = 1  # hard-coded for Index
 414     if axis is None:
 415         return
 416     if axis >= ndim or (axis < 0 and ndim + axis < 0):
 417         raise ValueError(f"`axis` must be fewer than the number of dimensions ({ndim})")
