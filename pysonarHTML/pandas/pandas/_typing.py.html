  65 # FrameOrSeriesUnion  means either a DataFrame or a Series. E.g.
  66 # `def func(a: FrameOrSeriesUnion) -> FrameOrSeriesUnion: ...` means that if a Series
  67 # is passed in, either a Series or DataFrame is returned, and if a DataFrame is passed
  68 # in, either a DataFrame or a Series is returned.
  69 FrameOrSeriesUnion = Union["DataFrame", "Series"]
  70 
  71 # FrameOrSeries is stricter and ensures that the same subclass of NDFrame always is
  72 # used. E.g. `def func(a: FrameOrSeries) -> FrameOrSeries: ...` means that if a
  73 # Series is passed into a function, a Series is always returned and if a DataFrame is
  74 # passed in, a DataFrame is always returned.
  75 FrameOrSeries = TypeVar("FrameOrSeries", bound="NDFrame")
