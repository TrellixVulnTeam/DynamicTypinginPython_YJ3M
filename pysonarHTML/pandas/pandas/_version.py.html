  53 def register_vcs_handler(vcs: str, method: str) -> Callable:  # decorator
  54     def decorate(f: Callable) -> Callable:
  55         if vcs not in HANDLERS:
  56             HANDLERS[vcs] = {}
  57         HANDLERS[vcs][method] = f
  58         return f
  59 
  60     return decorate

  63 def run_command(commands, args, cwd=None, verbose=False, hide_stderr=False):
  64     assert isinstance(commands, list)
  65     p = None
  66     for c in commands:
  67         try:
  68             dispcmd = str([c] + args)
  69             # remember shell=False, so use git.cmd on windows, not just git
  70             p = subprocess.Popen(
  71                 [c] + args,
  72                 cwd=cwd,
  73                 stdout=subprocess.PIPE,
  74                 stderr=(subprocess.PIPE if hide_stderr else None),
  75             )
  76             break
  77         except EnvironmentError:
  78             e = sys.exc_info()[1]
  79             if e.errno == errno.ENOENT:
  80                 continue
  81             if verbose:
  82                 print(f"unable to run {dispcmd}")
  83                 print(e)
  84             return None
  85     else:
  86         if verbose:
  87             print(f"unable to find command, tried {commands}")
  88         return None
  89     stdout = p.communicate()[0].strip().decode()
  90     if p.returncode != 0:
  91         if verbose:
  92             print(f"unable to run {dispcmd} (error)")
  93         return None
  94     return stdout

  97 def versions_from_parentdir(parentdir_prefix, root, verbose):
  98     # Source tarballs conventionally unpack into a directory that includes
  99     # both the project name and a version string.
 100     dirname = os.path.basename(root)
 101     if not dirname.startswith(parentdir_prefix):
 102         if verbose:
 103             print(
 104                 f"guessing rootdir is '{root}', but '{dirname}' "
 105                 f"doesn't start with prefix '{parentdir_prefix}'"
 106             )
 107         raise NotThisMethod("rootdir doesn't start with parentdir_prefix")
 108     return {
 109         "version": dirname[len(parentdir_prefix) :],
 110         "full-revisionid": None,
 111         "dirty": False,
 112         "error": None,
 113     }

 116 @register_vcs_handler("git", "get_keywords")
 117 def git_get_keywords(versionfile_abs):
 118     # the code embedded in _version.py can just fetch the value of these
 119     # keywords. When used from setup.py, we don't want to import _version.py,
 120     # so we do it with a regexp instead. This function is not used from
 121     # _version.py.
 122     keywords = {}
 123     try:
 124         f = open(versionfile_abs, "r")
 125         for line in f.readlines():
 126             if line.strip().startswith("git_refnames ="):
 127                 mo = re.search(r'=\s*"(.*)"', line)
 128                 if mo:
 129                     keywords["refnames"] = mo.group(1)
 130             if line.strip().startswith("git_full ="):
 131                 mo = re.search(r'=\s*"(.*)"', line)
 132                 if mo:
 133                     keywords["full"] = mo.group(1)
 134         f.close()
 135     except EnvironmentError:
 136         pass
 137     return keywords

 140 @register_vcs_handler("git", "keywords")
 141 def git_versions_from_keywords(keywords, tag_prefix, verbose):
 142     if not keywords:
 143         raise NotThisMethod("no keywords at all, weird")
 144     refnames = keywords["refnames"].strip()
 145     if refnames.startswith("$Format"):
 146         if verbose:
 147             print("keywords are unexpanded, not using")
 148         raise NotThisMethod("unexpanded keywords, not a git-archive tarball")
 149     refs = {r.strip() for r in refnames.strip("()").split(",")}
 150     # starting in git-1.8.3, tags are listed as "tag: foo-1.0" instead of
 151     # just "foo-1.0". If we see a "tag: " prefix, prefer those.
 152     TAG = "tag: "
 153     tags = {r[len(TAG) :] for r in refs if r.startswith(TAG)}
 154     if not tags:
 155         # Either we're using git < 1.8.3, or there really are no tags. We use
 156         # a heuristic: assume all version tags have a digit. The old git %d
 157         # expansion behaves like git log --decorate=short and strips out the
 158         # refs/heads/ and refs/tags/ prefixes that would let us distinguish
 159         # between branches and tags. By ignoring refnames without digits, we
 160         # filter out many common branch names like "release" and
 161         # "stabilization", as well as "HEAD" and "master".
 162         tags = {r for r in refs if re.search(r"\d", r)}
 163         if verbose:
 164             print(f"discarding '{','.join(refs - tags)}', no digits")
 165     if verbose:
 166         print(f"likely tags: {','.join(sorted(tags))}")
 167     for ref in sorted(tags):
 168         # sorting will prefer e.g. "2.0" over "2.0rc1"
 169         if ref.startswith(tag_prefix):
 170             r = ref[len(tag_prefix) :]
 171             if verbose:
 172                 print(f"picking {r}")
 173             return {
 174                 "version": r,
 175                 "full-revisionid": keywords["full"].strip(),
 176                 "dirty": False,
 177                 "error": None,
 178             }
 179     # no suitable tags, so version is "0+unknown", but full hex is still there
 180     if verbose:
 181         print("no suitable tags, using unknown + full revision id")
 182     return {
 183         "version": "0+unknown",
 184         "full-revisionid": keywords["full"].strip(),
 185         "dirty": False,
 186         "error": "no suitable tags",
 187     }

 190 @register_vcs_handler("git", "pieces_from_vcs")
 191 def git_pieces_from_vcs(tag_prefix, root, verbose, run_command=run_command):
 192     # this runs 'git' from the root of the source tree. This only gets called
 193     # if the git-archive 'subst' keywords were *not* expanded, and
 194     # _version.py hasn't already been rewritten with a short version string,
 195     # meaning we're inside a checked out source tree.
 196 
 197     if not os.path.exists(os.path.join(root, ".git")):
 198         if verbose:
 199             print(f"no .git in {root}")
 200         raise NotThisMethod("no .git directory")
 201 
 202     GITS = ["git"]
 203     if sys.platform == "win32":
 204         GITS = ["git.cmd", "git.exe"]
 205     # if there is a tag, this yields TAG-NUM-gHEX[-dirty]
 206     # if there are no tags, this yields HEX[-dirty] (no NUM)
 207     describe_out = run_command(
 208         GITS, ["describe", "--tags", "--dirty", "--always", "--long"], cwd=root
 209     )
 210     # --long was added in git-1.5.5
 211     if describe_out is None:
 212         raise NotThisMethod("'git describe' failed")
 213     describe_out = describe_out.strip()
 214     full_out = run_command(GITS, ["rev-parse", "HEAD"], cwd=root)
 215     if full_out is None:
 216         raise NotThisMethod("'git rev-parse' failed")
 217     full_out = full_out.strip()
 218 
 219     pieces = {}
 220     pieces["long"] = full_out
 221     pieces["short"] = full_out[:7]  # maybe improved later
 222     pieces["error"] = None
 223 
 224     # parse describe_out. It will be like TAG-NUM-gHEX[-dirty] or HEX[-dirty]
 225     # TAG might have hyphens.
 226     git_describe = describe_out
 227 
 228     # look for -dirty suffix
 229     dirty = git_describe.endswith("-dirty")
 230     pieces["dirty"] = dirty
 231     if dirty:
 232         git_describe = git_describe[: git_describe.rindex("-dirty")]
 233 
 234     # now we have TAG-NUM-gHEX or HEX
 235 
 236     if "-" in git_describe:
 237         # TAG-NUM-gHEX
 238         mo = re.search(r"^(.+)-(\d+)-g([0-9a-f]+)$", git_describe)
 239         if not mo:
 240             # unparseable. Maybe git-describe is misbehaving?
 241             pieces["error"] = f"unable to parse git-describe output: '{describe_out}'"
 242             return pieces
 243 
 244         # tag
 245         full_tag = mo.group(1)
 246         if not full_tag.startswith(tag_prefix):
 247             msg = f"tag '{full_tag}' doesn't start with prefix '{tag_prefix}'"
 248             if verbose:
 249                 print(msg)
 250             pieces["error"] = msg
 251             return pieces
 252 
 253         pieces["closest-tag"] = full_tag[len(tag_prefix) :]
 254 
 255         # distance: number of commits since tag
 256         pieces["distance"] = int(mo.group(2))
 257 
 258         # commit: short hex revision ID
 259         pieces["short"] = mo.group(3)
 260 
 261     else:
 262         # HEX: no tags
 263         pieces["closest-tag"] = None
 264         count_out = run_command(GITS, ["rev-list", "HEAD", "--count"], cwd=root)
 265         pieces["distance"] = int(count_out)  # total number of commits
 266 
 267     return pieces

 270 def plus_or_dot(pieces):
 271     if "+" in pieces.get("closest-tag", ""):
 272         return "."
 273     return "+"

 276 def render_pep440(pieces):
 277     # now build up version string, with post-release "local version
 278     # identifier". Our goal: TAG[+DISTANCE.gHEX[.dirty]] . Note that if you
 279     # get a tagged build and then dirty it, you'll get TAG+0.gHEX.dirty
 280 
 281     # exceptions:
 282     # 1: no tags. git_describe was just HEX. 0+untagged.DISTANCE.gHEX[.dirty]
 283 
 284     if pieces["closest-tag"]:
 285         rendered = pieces["closest-tag"]
 286         if pieces["distance"] or pieces["dirty"]:
 287             rendered += plus_or_dot(pieces)
 288             rendered += f"{pieces['distance']:d}.g{pieces['short']}"
 289             if pieces["dirty"]:
 290                 rendered += ".dirty"
 291     else:
 292         # exception #1
 293         rendered = f"0+untagged.{pieces['distance']:d}.g{pieces['short']}"
 294         if pieces["dirty"]:
 295             rendered += ".dirty"
 296     return rendered

 299 def render_pep440_pre(pieces):
 300     # TAG[.post.devDISTANCE] . No -dirty
 301 
 302     # exceptions:
 303     # 1: no tags. 0.post.devDISTANCE
 304 
 305     if pieces["closest-tag"]:
 306         rendered = pieces["closest-tag"]
 307         if pieces["distance"]:
 308             rendered += f".post.dev{pieces['distance']:d}"
 309     else:
 310         # exception #1
 311         rendered = f"0.post.dev{pieces['distance']:d}"
 312     return rendered

 315 def render_pep440_post(pieces):
 316     # TAG[.postDISTANCE[.dev0]+gHEX] . The ".dev0" means dirty. Note that
 317     # .dev0 sorts backwards (a dirty tree will appear "older" than the
 318     # corresponding clean one), but you shouldn't be releasing software with
 319     # -dirty anyways.
 320 
 321     # exceptions:
 322     # 1: no tags. 0.postDISTANCE[.dev0]
 323 
 324     if pieces["closest-tag"]:
 325         rendered = pieces["closest-tag"]
 326         if pieces["distance"] or pieces["dirty"]:
 327             rendered += f".post{pieces['distance']:d}"
 328             if pieces["dirty"]:
 329                 rendered += ".dev0"
 330             rendered += plus_or_dot(pieces)
 331             rendered += f"g{pieces['short']}"
 332     else:
 333         # exception #1
 334         rendered = f"0.pos{pieces['distance']:d}"
 335         if pieces["dirty"]:
 336             rendered += ".dev0"
 337         rendered += f"+g{pieces['short']}"
 338     return rendered

 361 def render_git_describe(pieces):
 362     # TAG[-DISTANCE-gHEX][-dirty], like 'git describe --tags --dirty
 363     # --always'
 364 
 365     # exceptions:
 366     # 1: no tags. HEX[-dirty]  (note: no 'g' prefix)
 367 
 368     if pieces["closest-tag"]:
 369         rendered = pieces["closest-tag"]
 370         if pieces["distance"]:
 371             rendered += f"-{pieces['distance']:d}-g{pieces['short']}"
 372     else:
 373         # exception #1
 374         rendered = pieces["short"]
 375     if pieces["dirty"]:
 376         rendered += "-dirty"
 377     return rendered

 398 def render(pieces, style):
 399     if pieces["error"]:
 400         return {
 401             "version": "unknown",
 402             "full-revisionid": pieces.get("long"),
 403             "dirty": None,
 404             "error": pieces["error"],
 405         }
 406 
 407     if not style or style == "default":
 408         style = "pep440"  # the default
 409 
 410     if style == "pep440":
 411         rendered = render_pep440(pieces)
 412     elif style == "pep440-pre":
 413         rendered = render_pep440_pre(pieces)
 414     elif style == "pep440-post":
 415         rendered = render_pep440_post(pieces)
 416     elif style == "pep440-old":
 417         rendered = render_pep440_old(pieces)
 418     elif style == "git-describe":
 419         rendered = render_git_describe(pieces)
 420     elif style == "git-describe-long":
 421         rendered = render_git_describe_long(pieces)
 422     else:
 423         raise ValueError(f"unknown style '{style}'")
 424 
 425     return {
 426         "version": rendered,
 427         "full-revisionid": pieces["long"],
 428         "dirty": pieces["dirty"],
 429         "error": None,
 430     }

 433 def get_versions():
 434     # I am in _version.py, which lives at ROOT/VERSIONFILE_SOURCE. If we have
 435     # __file__, we can work backwards from there to the root. Some
 436     # py2exe/bbfreeze/non-CPython implementations don't do __file__, in which
 437     # case we can only use expanded keywords.
 438 
 439     cfg = get_config()
 440     verbose = cfg.verbose
 441 
 442     try:
 443         return git_versions_from_keywords(get_keywords(), cfg.tag_prefix, verbose)
 444     except NotThisMethod:
 445         pass
 446 
 447     try:
 448         root = os.path.realpath(__file__)
 449         # versionfile_source is the relative path from the top of the source
 450         # tree (where the .git directory might live) to this file. Invert
 451         # this to find the root from __file__.
 452         for i in cfg.versionfile_source.split("/"):
 453             root = os.path.dirname(root)
 454     except NameError:
 455         return {
 456             "version": "0+unknown",
 457             "full-revisionid": None,
 458             "dirty": None,
 459             "error": "unable to find root of source tree",
 460         }
 461 
 462     try:
 463         pieces = git_pieces_from_vcs(cfg.tag_prefix, root, verbose)
 464         return render(pieces, cfg.style)
 465     except NotThisMethod:
 466         pass
 467 
 468     try:
 469         if cfg.parentdir_prefix:
 470             return versions_from_parentdir(cfg.parentdir_prefix, root, verbose)
 471     except NotThisMethod:
 472         pass
 473 
 474     return {
 475         "version": "0+unknown",
 476         "full-revisionid": None,
 477         "dirty": None,
 478         "error": "unable to compute version",
 479     }
