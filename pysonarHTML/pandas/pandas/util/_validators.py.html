  13 def _check_arg_length(fname, args, max_fname_arg_count, compat_args):
  14     """
  15     Checks whether 'args' has length of at most 'compat_args'. Raises
  16     a TypeError if that is not the case, similar to in Python when a
  17     function is called with too many arguments.
  18     """
  19     if max_fname_arg_count < 0:
  20         raise ValueError("'max_fname_arg_count' must be non-negative")
  21 
  22     if len(args) > len(compat_args):
  23         max_arg_count = len(compat_args) + max_fname_arg_count
  24         actual_arg_count = len(args) + max_fname_arg_count
  25         argument = "argument" if max_arg_count == 1 else "arguments"
  26 
  27         raise TypeError(
  28             f"{fname}() takes at most {max_arg_count} {argument} "
  29             f"({actual_arg_count} given)"
  30         )

  33 def _check_for_default_values(fname, arg_val_dict, compat_args):
  34     """
  35     Check that the keys in `arg_val_dict` are mapped to their
  36     default values as specified in `compat_args`.
  37 
  38     Note that this function is to be called only when it has been
  39     checked that arg_val_dict.keys() is a subset of compat_args
  40     """
  41     for key in arg_val_dict:
  42         # try checking equality directly with '=' operator,
  43         # as comparison may have been overridden for the left
  44         # hand object
  45         try:
  46             v1 = arg_val_dict[key]
  47             v2 = compat_args[key]
  48 
  49             # check for None-ness otherwise we could end up
  50             # comparing a numpy array vs None
  51             if (v1 is not None and v2 is None) or (v1 is None and v2 is not None):
  52                 match = False
  53             else:
  54                 match = v1 == v2
  55 
  56             if not is_bool(match):
  57                 raise ValueError("'match' is not a boolean")
  58 
  59         # could not compare them directly, so try comparison
  60         # using the 'is' operator
  61         except ValueError:
  62             match = arg_val_dict[key] is compat_args[key]
  63 
  64         if not match:
  65             raise ValueError(
  66                 f"the '{key}' parameter is not supported in "
  67                 f"the pandas implementation of {fname}()"
  68             )

  71 def validate_args(fname, args, max_fname_arg_count, compat_args):
  72     """
  73     Checks whether the length of the `*args` argument passed into a function
  74     has at most `len(compat_args)` arguments and whether or not all of these
  75     elements in `args` are set to their default values.
  76 
  77     Parameters
  78     ----------
  79     fname : str
  80         The name of the function being passed the `*args` parameter
  81     args : tuple
  82         The `*args` parameter passed into a function
  83     max_fname_arg_count : int
  84         The maximum number of arguments that the function `fname`
  85         can accept, excluding those in `args`. Used for displaying
  86         appropriate error messages. Must be non-negative.
  87     compat_args : dict
  88         A dictionary of keys and their associated default values.
  89         In order to accommodate buggy behaviour in some versions of `numpy`,
  90         where a signature displayed keyword arguments but then passed those
  91         arguments **positionally** internally when calling downstream
  92         implementations, a dict ensures that the original
  93         order of the keyword arguments is enforced.
  94 
  95     Raises
  96     ------
  97     TypeError
  98         If `args` contains more values than there are `compat_args`
  99     ValueError
 100         If `args` contains values that do not correspond to those
 101         of the default values specified in `compat_args`
 102     """
 103     _check_arg_length(fname, args, max_fname_arg_count, compat_args)
 104 
 105     # We do this so that we can provide a more informative
 106     # error message about the parameters that we are not
 107     # supporting in the pandas implementation of 'fname'
 108     kwargs = dict(zip(compat_args, args))
 109     _check_for_default_values(fname, kwargs, compat_args)

 112 def _check_for_invalid_keys(fname, kwargs, compat_args):
 113     """
 114     Checks whether 'kwargs' contains any keys that are not
 115     in 'compat_args' and raises a TypeError if there is one.
 116     """
 117     # set(dict) --> set of the dictionary's keys
 118     diff = set(kwargs) - set(compat_args)
 119 
 120     if diff:
 121         bad_arg = list(diff)[0]
 122         raise TypeError(f"{fname}() got an unexpected keyword argument '{bad_arg}'")

 125 def validate_kwargs(fname, kwargs, compat_args):
 126     """
 127     Checks whether parameters passed to the **kwargs argument in a
 128     function `fname` are valid parameters as specified in `*compat_args`
 129     and whether or not they are set to their default values.
 130 
 131     Parameters
 132     ----------
 133     fname : str
 134         The name of the function being passed the `**kwargs` parameter
 135     kwargs : dict
 136         The `**kwargs` parameter passed into `fname`
 137     compat_args: dict
 138         A dictionary of keys that `kwargs` is allowed to have and their
 139         associated default values
 140 
 141     Raises
 142     ------
 143     TypeError if `kwargs` contains keys not in `compat_args`
 144     ValueError if `kwargs` contains keys in `compat_args` that do not
 145     map to the default values specified in `compat_args`
 146     """
 147     kwds = kwargs.copy()
 148     _check_for_invalid_keys(fname, kwargs, compat_args)
 149     _check_for_default_values(fname, kwds, compat_args)

 152 def validate_args_and_kwargs(fname, args, kwargs, max_fname_arg_count, compat_args):
 153     """
 154     Checks whether parameters passed to the *args and **kwargs argument in a
 155     function `fname` are valid parameters as specified in `*compat_args`
 156     and whether or not they are set to their default values.
 157 
 158     Parameters
 159     ----------
 160     fname: str
 161         The name of the function being passed the `**kwargs` parameter
 162     args: tuple
 163         The `*args` parameter passed into a function
 164     kwargs: dict
 165         The `**kwargs` parameter passed into `fname`
 166     max_fname_arg_count: int
 167         The minimum number of arguments that the function `fname`
 168         requires, excluding those in `args`. Used for displaying
 169         appropriate error messages. Must be non-negative.
 170     compat_args: dict
 171         A dictionary of keys that `kwargs` is allowed to
 172         have and their associated default values.
 173 
 174     Raises
 175     ------
 176     TypeError if `args` contains more values than there are
 177     `compat_args` OR `kwargs` contains keys not in `compat_args`
 178     ValueError if `args` contains values not at the default value (`None`)
 179     `kwargs` contains keys in `compat_args` that do not map to the default
 180     value as specified in `compat_args`
 181 
 182     See Also
 183     --------
 184     validate_args : Purely args validation.
 185     validate_kwargs : Purely kwargs validation.
 186 
 187     """
 188     # Check that the total number of arguments passed in (i.e.
 189     # args and kwargs) does not exceed the length of compat_args
 190     _check_arg_length(
 191         fname, args + tuple(kwargs.values()), max_fname_arg_count, compat_args
 192     )
 193 
 194     # Check there is no overlap with the positional and keyword
 195     # arguments, similar to what is done in actual Python functions
 196     args_dict = dict(zip(compat_args, args))
 197 
 198     for key in args_dict:
 199         if key in kwargs:
 200             raise TypeError(
 201                 f"{fname}() got multiple values for keyword argument '{key}'"
 202             )
 203 
 204     kwargs.update(args_dict)
 205     validate_kwargs(fname, kwargs, compat_args)

 208 def validate_bool_kwarg(value, arg_name):
 209     """ Ensures that argument passed in arg_name is of type bool. """
 210     if not (is_bool(value) or value is None):
 211         raise ValueError(
 212             f'For argument "{arg_name}" expected type bool, received '
 213             f"type {type(value).__name__}."
 214         )
 215     return value

 218 def validate_axis_style_args(data, args, kwargs, arg_name, method_name):
 219     """
 220     Argument handler for mixed index, columns / axis functions
 221 
 222     In an attempt to handle both `.method(index, columns)`, and
 223     `.method(arg, axis=.)`, we have to do some bad things to argument
 224     parsing. This translates all arguments to `{index=., columns=.}` style.
 225 
 226     Parameters
 227     ----------
 228     data : DataFrame
 229     args : tuple
 230         All positional arguments from the user
 231     kwargs : dict
 232         All keyword arguments from the user
 233     arg_name, method_name : str
 234         Used for better error messages
 235 
 236     Returns
 237     -------
 238     kwargs : dict
 239         A dictionary of keyword arguments. Doesn't modify ``kwargs``
 240         inplace, so update them with the return value here.
 241 
 242     Examples
 243     --------
 244     >>> df._validate_axis_style_args((str.upper,), {'columns': id},
 245     ...                              'mapper', 'rename')
 246     {'columns': <function id>, 'index': <method 'upper' of 'str' objects>}
 247 
 248     This emits a warning
 249     >>> df._validate_axis_style_args((str.upper, id), {},
 250     ...                              'mapper', 'rename')
 251     {'columns': <function id>, 'index': <method 'upper' of 'str' objects>}
 252     """
 253     # TODO: Change to keyword-only args and remove all this
 254 
 255     out = {}
 256     # Goal: fill 'out' with index/columns-style arguments
 257     # like out = {'index': foo, 'columns': bar}
 258 
 259     # Start by validating for consistency
 260     if "axis" in kwargs and any(x in kwargs for x in data._AXIS_TO_AXIS_NUMBER):
 261         msg = "Cannot specify both 'axis' and any of 'index' or 'columns'."
 262         raise TypeError(msg)
 263 
 264     # First fill with explicit values provided by the user...
 265     if arg_name in kwargs:
 266         if args:
 267             msg = f"{method_name} got multiple values for argument '{arg_name}'"
 268             raise TypeError(msg)
 269 
 270         axis = data._get_axis_name(kwargs.get("axis", 0))
 271         out[axis] = kwargs[arg_name]
 272 
 273     # More user-provided arguments, now from kwargs
 274     for k, v in kwargs.items():
 275         try:
 276             ax = data._get_axis_name(k)
 277         except ValueError:
 278             pass
 279         else:
 280             out[ax] = v
 281 
 282     # All user-provided kwargs have been handled now.
 283     # Now we supplement with positional arguments, emitting warnings
 284     # when there's ambiguity and raising when there's conflicts
 285 
 286     if len(args) == 0:
 287         pass  # It's up to the function to decide if this is valid
 288     elif len(args) == 1:
 289         axis = data._get_axis_name(kwargs.get("axis", 0))
 290         out[axis] = args[0]
 291     elif len(args) == 2:
 292         if "axis" in kwargs:
 293             # Unambiguously wrong
 294             msg = "Cannot specify both 'axis' and any of 'index' or 'columns'"
 295             raise TypeError(msg)
 296 
 297         msg = (
 298             f"Interpreting call\n\t'.{method_name}(a, b)' as "
 299             f"\n\t'.{method_name}(index=a, columns=b)'.\nUse named "
 300             "arguments to remove any ambiguity. In the future, using "
 301             "positional arguments for 'index' or 'columns' will raise "
 302             "a 'TypeError'."
 303         )
 304         warnings.warn(msg, FutureWarning, stacklevel=4)
 305         out[data._get_axis_name(0)] = args[0]
 306         out[data._get_axis_name(1)] = args[1]
 307     else:
 308         msg = f"Cannot specify all of '{arg_name}', 'index', 'columns'."
 309         raise TypeError(msg)
 310     return out

 313 def validate_fillna_kwargs(value, method, validate_scalar_dict_value=True):
 314     """
 315     Validate the keyword arguments to 'fillna'.
 316 
 317     This checks that exactly one of 'value' and 'method' is specified.
 318     If 'method' is specified, this validates that it's a valid method.
 319 
 320     Parameters
 321     ----------
 322     value, method : object
 323         The 'value' and 'method' keyword arguments for 'fillna'.
 324     validate_scalar_dict_value : bool, default True
 325         Whether to validate that 'value' is a scalar or dict. Specifically,
 326         validate that it is not a list or tuple.
 327 
 328     Returns
 329     -------
 330     value, method : object
 331     """
 332     from pandas.core.missing import clean_fill_method
 333 
 334     if value is None and method is None:
 335         raise ValueError("Must specify a fill 'value' or 'method'.")
 336     elif value is None and method is not None:
 337         method = clean_fill_method(method)
 338 
 339     elif value is not None and method is None:
 340         if validate_scalar_dict_value and isinstance(value, (list, tuple)):
 341             raise TypeError(
 342                 '"value" parameter must be a scalar or dict, but '
 343                 f'you passed a "{type(value).__name__}"'
 344             )
 345 
 346     elif value is not None and method is not None:
 347         raise ValueError("Cannot specify both 'value' and 'method'.")
 348 
 349     return value, method

 352 def validate_percentile(q: Union[float, Iterable[float]]) -> np.ndarray:
 353     """
 354     Validate percentiles (used by describe and quantile).
 355 
 356     This function checks if the given float or iterable of floats is a valid percentile
 357     otherwise raises a ValueError.
 358 
 359     Parameters
 360     ----------
 361     q: float or iterable of floats
 362         A single percentile or an iterable of percentiles.
 363 
 364     Returns
 365     -------
 366     ndarray
 367         An ndarray of the percentiles if valid.
 368 
 369     Raises
 370     ------
 371     ValueError if percentiles are not in given interval([0, 1]).
 372     """
 373     q_arr = np.asarray(q)
 374     msg = (
 375         "percentiles should all be in the interval [0, 1]."
 376         f"Try {q_arr / 100.0} instead."
 377     )
 378     if q_arr.ndim == 0:
 379         if not 0 <= q_arr <= 1:
 380             raise ValueError(msg)
 381     else:
 382         if not all(0 <= qs <= 1 for qs in q_arr):
 383             raise ValueError(msg)
 384     return q_arr
