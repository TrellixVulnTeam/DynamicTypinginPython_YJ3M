   1 """
   2 This module provides decorator functions which can be applied to test objects
   3 in order to skip those objects when certain conditions occur. A sample use case
   4 is to detect if the platform is missing ``matplotlib``. If so, any test objects
   5 which require ``matplotlib`` and decorated with ``@td.skip_if_no_mpl`` will be
   6 skipped by ``pytest`` during the execution of the test suite.
   7 
   8 To illustrate, after importing this module:
   9 
  10 import pandas.util._test_decorators as td
  11 
  12 The decorators can be applied to classes:
  13 
  14 @td.skip_if_some_reason
  15 class Foo:
  16     ...
  17 
  18 Or individual functions:
  19 
  20 @td.skip_if_some_reason
  21 def test_foo():
  22     ...
  23 
  24 For more information, refer to the ``pytest`` documentation on ``skipif``.
  25 """

  41 def safe_import(mod_name: str, min_version: Optional[str] = None):
  42     """
  43     Parameters
  44     ----------
  45     mod_name : str
  46         Name of the module to be imported
  47     min_version : str, default None
  48         Minimum required version of the specified mod_name
  49 
  50     Returns
  51     -------
  52     object
  53         The imported module if successful, or False
  54     """
  55     try:
  56         mod = __import__(mod_name)
  57     except ImportError:
  58         return False
  59 
  60     if not min_version:
  61         return mod
  62     else:
  63         import sys
  64 
  65         try:
  66             version = getattr(sys.modules[mod_name], "__version__")
  67         except AttributeError:
  68             # xlrd uses a capitalized attribute name
  69             version = getattr(sys.modules[mod_name], "__VERSION__")
  70         if version:
  71             from distutils.version import LooseVersion
  72 
  73             if LooseVersion(version) >= LooseVersion(min_version):
  74                 return mod
  75 
  76     return False

  94 def _skip_if_no_mpl():
  95     mod = safe_import("matplotlib")
  96     if mod:
  97         mod.use("Agg", warn=True)
  98     else:
  99         return True

 102 def _skip_if_has_locale():
 103     lang, _ = locale.getlocale()

 105         return True

 108 def _skip_if_not_us_locale():
 109     lang, _ = locale.getlocale()
 110     if lang != "en_US":

 123 def skip_if_installed(package: str) -> Callable:
 124     """
 125     Skip a test if a package is installed.
 126 
 127     Parameters
 128     ----------
 129     package : str
 130         The name of the package.
 131     """
 132     return pytest.mark.skipif(
 133         safe_import(package), reason=f"Skipping because {package} is installed."
 134     )

 137 def skip_if_no(package: str, min_version: Optional[str] = None) -> Callable:
 138     """
 139     Generic function to help skip tests when required packages are not
 140     present on the testing system.
 141 
 142     This function returns a pytest mark with a skip condition that will be
 143     evaluated during test collection. An attempt will be made to import the
 144     specified ``package`` and optionally ensure it meets the ``min_version``
 145 
 146     The mark can be used as either a decorator for a test function or to be
 147     applied to parameters in pytest.mark.parametrize calls or parametrized
 148     fixtures.
 149 
 150     If the import and version check are unsuccessful, then the test function
 151     (or test case when used in conjunction with parametrization) will be
 152     skipped.
 153 
 154     Parameters
 155     ----------
 156     package: str
 157         The name of the required package.
 158     min_version: str or None, default None
 159         Optional minimum version of the package.
 160 
 161     Returns
 162     -------
 163     _pytest.mark.structures.MarkDecorator
 164         a pytest.mark.skipif to use as either a test decorator or a
 165         parametrization mark.
 166     """
 167     msg = f"Could not import '{package}'"
 168     if min_version:
 169         msg += f" satisfying a min_version of {min_version}"
 170     return pytest.mark.skipif(
 171         not safe_import(package, min_version=min_version), reason=msg
 172     )

 199 def skip_if_np_lt(
 200     ver_str: str, reason: Optional[str] = None, *args, **kwds
 201 ) -> Callable:
 202     if reason is None:
 203         reason = f"NumPy {ver_str} or greater required"
 204     return pytest.mark.skipif(
 205         _np_version < LooseVersion(ver_str), reason=reason, *args, **kwds
 206     )

 209 def parametrize_fixture_doc(*args):
 210     """
 211     Intended for use as a decorator for parametrized fixture,
 212     this function will wrap the decorated function with a pytest
 213     ``parametrize_fixture_doc`` mark. That mark will format
 214     initial fixture docstring by replacing placeholders {0}, {1} etc
 215     with parameters passed as arguments.
 216 
 217     Parameters
 218     ----------
 219     args: iterable
 220         Positional arguments for docstring.
 221 
 222     Returns
 223     -------
 224     function
 225         The decorated function wrapped within a pytest
 226         ``parametrize_fixture_doc`` mark
 227     """
 228 
 229     def documented_fixture(fixture):
 230         fixture.__doc__ = fixture.__doc__.format(*args)
 231         return fixture
 232 
 233     return documented_fixture

 236 def check_file_leaks(func) -> Callable:
 237     """
 238     Decorate a test function tot check that we are not leaking file descriptors.
 239     """
 240     psutil = safe_import("psutil")
 241     if not psutil:
 242         return func
 243 
 244     @wraps(func)
 245     def new_func(*args, **kwargs):
 246         proc = psutil.Process()
 247         flist = proc.open_files()
 248 
 249         func(*args, **kwargs)
 250 
 251         flist2 = proc.open_files()
 252         assert flist2 == flist
 253 
 254     return new_func
