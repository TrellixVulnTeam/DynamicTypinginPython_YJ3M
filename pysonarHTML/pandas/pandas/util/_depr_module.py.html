  11 class _DeprecatedModule:
  12     """
  13     Class for mocking deprecated modules.
  14 
  15     Parameters
  16     ----------
  17     deprmod : name of module to be deprecated.
  18     deprmodto : name of module as a replacement, optional.
  19                 If not given, the __module__ attribute will
  20                 be used when needed.
  21     removals : objects or methods in module that will no longer be

  23     moved : dict, optional

  25             objects

  28     def __init__(self, deprmod, deprmodto=None, removals=None, moved=None):
  29         self.deprmod = deprmod
  30         self.deprmodto = deprmodto
  31         self.removals = removals
  32         if self.removals is not None:
  33             self.removals = frozenset(self.removals)
  34         self.moved = moved
  35 
  36         # For introspection purposes.
  37         self.self_dir = frozenset(dir(type(self)))

  39     def __dir__(self) -> Iterable[str]:
  40         deprmodule = self._import_deprmod()
  41         return dir(deprmodule)
  42 
  43     def __repr__(self) -> str:
  44         deprmodule = self._import_deprmod()
  45         return repr(deprmodule)

  49     def __getattr__(self, name: str):

  62         obj = getattr(deprmodule, name)

  64         if self.removals is not None and name in self.removals:

  83                     "a future version.",
  84                     FutureWarning,
  85                     stacklevel=2,
  86                 )

  88                 if deprmodto is None:
  89                     deprmodto = obj.__module__
  90                 # The object is actually located in another module.
  91                 warnings.warn(
  92                     f"{self.deprmod}.{name} is deprecated. Please use "
  93                     f"{deprmodto}.{name} instead.",
  94                     FutureWarning,
  95                     stacklevel=2,
