  11 def deprecate(
  12     name: str,
  13     alternative: Callable[..., Any],
  14     version: str,
  15     alt_name: Optional[str] = None,
  16     klass: Optional[Type[Warning]] = None,
  17     stacklevel: int = 2,
  18     msg: Optional[str] = None,
  19 ) -> Callable[[F], F]:
  20     """
  21     Return a new function that emits a deprecation warning on use.
  22 
  23     To use this method for a deprecated function, another function
  24     `alternative` with the same signature must exist. The deprecated
  25     function will emit a deprecation warning, and in the docstring
  26     it will contain the deprecation directive with the provided version
  27     so it can be detected for future removal.
  28 
  29     Parameters
  30     ----------
  31     name : str
  32         Name of function to deprecate.
  33     alternative : func
  34         Function to use instead.
  35     version : str
  36         Version of pandas in which the method has been deprecated.
  37     alt_name : str, optional
  38         Name to use in preference of alternative.__name__.
  39     klass : Warning, default FutureWarning
  40     stacklevel : int, default 2
  41     msg : str
  42         The message to display in the warning.
  43         Default is '{name} is deprecated. Use {alt_name} instead.'
  44     """
  45     alt_name = alt_name or alternative.__name__
  46     klass = klass or FutureWarning
  47     warning_msg = msg or f"{name} is deprecated, use {alt_name} instead"
  48 
  49     @wraps(alternative)
  50     def wrapper(*args, **kwargs) -> Callable[..., Any]:
  51         warnings.warn(warning_msg, klass, stacklevel=stacklevel)
  52         return alternative(*args, **kwargs)
  53 
  54     # adding deprecated directive to the docstring
  55     msg = msg or f"Use `{alt_name}` instead."
  56     doc_error_msg = (
  57         "deprecate needs a correctly formatted docstring in "
  58         "the target function (should have a one liner short "
  59         "summary, and opening quotes should be in their own "
  60         f"line). Found:\n{alternative.__doc__}"
  61     )
  62 
  63     # when python is running in optimized mode (i.e. `-OO`), docstrings are
  64     # removed, so we check that a docstring with correct formatting is used
  65     # but we allow empty docstrings
  66     if alternative.__doc__:
  67         if alternative.__doc__.count("\n") < 3:
  68             raise AssertionError(doc_error_msg)
  69         empty1, summary, empty2, doc = alternative.__doc__.split("\n", 3)
  70         if empty1 or empty2 and not summary:
  71             raise AssertionError(doc_error_msg)
  72         wrapper.__doc__ = dedent(
  73             f"""
  74         {summary.strip()}
  75 
  76         .. deprecated:: {version}
  77             {msg}
  78 
  79         {dedent(doc)}"""
  80         )
  81 
  82     return wrapper
  83

  85 def deprecate_kwarg(
  86     old_arg_name: str,
  87     new_arg_name: Optional[str],
  88     mapping: Optional[Union[Mapping[Any, Any], Callable[[Any], Any]]] = None,
  89     stacklevel: int = 2,
  90 ) -> Callable[[F], F]:
  91     """
  92     Decorator to deprecate a keyword argument of a function.
  93 
  94     Parameters
  95     ----------
  96     old_arg_name : str
  97         Name of argument in function to deprecate
  98     new_arg_name : str or None
  99         Name of preferred argument in function. Use None to raise warning that
 100         ``old_arg_name`` keyword is deprecated.
 101     mapping : dict or callable
 102         If mapping is present, use it to translate old arguments to
 103         new arguments. A callable must do its own value checking;
 104         values not found in a dict will be forwarded unchanged.
 105 
 106     Examples
 107     --------
 108     The following deprecates 'cols', using 'columns' instead
 109 
 110     >>> @deprecate_kwarg(old_arg_name='cols', new_arg_name='columns')
 111     ... def f(columns=''):
 112     ...     print(columns)
 113     ...
 114     >>> f(columns='should work ok')
 115     should work ok
 116 
 117     >>> f(cols='should raise warning')
 118     FutureWarning: cols is deprecated, use columns instead
 119       warnings.warn(msg, FutureWarning)
 120     should raise warning
 121 
 122     >>> f(cols='should error', columns="can\'t pass do both")
 123     TypeError: Can only specify 'cols' or 'columns', not both
 124 
 125     >>> @deprecate_kwarg('old', 'new', {'yes': True, 'no': False})
 126     ... def f(new=False):
 127     ...     print('yes!' if new else 'no!')
 128     ...
 129     >>> f(old='yes')
 130     FutureWarning: old='yes' is deprecated, use new=True instead
 131       warnings.warn(msg, FutureWarning)
 132     yes!
 133 
 134     To raise a warning that a keyword will be removed entirely in the future
 135 
 136     >>> @deprecate_kwarg(old_arg_name='cols', new_arg_name=None)
 137     ... def f(cols='', another_param=''):
 138     ...     print(cols)
 139     ...
 140     >>> f(cols='should raise warning')
 141     FutureWarning: the 'cols' keyword is deprecated and will be removed in a
 142     future version please takes steps to stop use of 'cols'
 143     should raise warning
 144     >>> f(another_param='should not raise warning')
 145     should not raise warning
 146 
 147     >>> f(cols='should raise warning', another_param='')
 148     FutureWarning: the 'cols' keyword is deprecated and will be removed in a
 149     future version please takes steps to stop use of 'cols'
 150     should raise warning
 151     """
 152     if mapping is not None and not hasattr(mapping, "get") and not callable(mapping):
 153         raise TypeError(
 154             "mapping from old to new argument values must be dict or callable!"
 155         )
 156 
 157     def _deprecate_kwarg(func: F) -> F:
 158         @wraps(func)
 159         def wrapper(*args, **kwargs) -> Callable[..., Any]:
 160             old_arg_value = kwargs.pop(old_arg_name, None)
 161 
 162             if old_arg_value is not None:
 163                 if new_arg_name is None:
 164                     msg = (
 165                         f"the {repr(old_arg_name)} keyword is deprecated and "
 166                         "will be removed in a future version. Please take "
 167                         f"steps to stop the use of {repr(old_arg_name)}"
 168                     )
 169                     warnings.warn(msg, FutureWarning, stacklevel=stacklevel)
 170                     kwargs[old_arg_name] = old_arg_value
 171                     return func(*args, **kwargs)
 172 
 173                 elif mapping is not None:
 174                     if callable(mapping):
 175                         new_arg_value = mapping(old_arg_value)
 176                     else:
 177                         new_arg_value = mapping.get(old_arg_value, old_arg_value)
 178                     msg = (
 179                         f"the {old_arg_name}={repr(old_arg_value)} keyword is "
 180                         "deprecated, use "
 181                         f"{new_arg_name}={repr(new_arg_value)} instead"
 182                     )
 183                 else:
 184                     new_arg_value = old_arg_value
 185                     msg = (
 186                         f"the {repr(old_arg_name)}' keyword is deprecated, "
 187                         f"use {repr(new_arg_name)} instead"
 188                     )
 189 
 190                 warnings.warn(msg, FutureWarning, stacklevel=stacklevel)
 191                 if kwargs.get(new_arg_name) is not None:
 192                     msg = (
 193                         f"Can only specify {repr(old_arg_name)} "
 194                         f"or {repr(new_arg_name)}, not both"
 195                     )
 196                     raise TypeError(msg)
 197                 else:
 198                     kwargs[new_arg_name] = new_arg_value
 199             return func(*args, **kwargs)
 200 
 201         return cast(F, wrapper)
 202 
 203     return _deprecate_kwarg

 206 def _format_argument_list(allow_args: Union[List[str], int]):
 207     """
 208     Convert the allow_args argument (either string or integer) of
 209     `deprecate_nonkeyword_arguments` function to a string describing
 210     it to be inserted into warning message.
 211 
 212     Parameters
 213     ----------
 214     allowed_args : list, tuple or int
 215         The `allowed_args` argument for `deprecate_nonkeyword_arguments`,
 216         but None value is not allowed.
 217 
 218     Returns
 219     -------
 220     s : str
 221         The substring describing the argument list in best way to be
 222         inserted to the warning message.
 223 
 224     Examples
 225     --------
 226     `format_argument_list(0)` -> ''
 227     `format_argument_list(1)` -> 'except for the first argument'
 228     `format_argument_list(2)` -> 'except for the first 2 arguments'
 229     `format_argument_list([])` -> ''
 230     `format_argument_list(['a'])` -> "except for the arguments 'a'"
 231     `format_argument_list(['a', 'b'])` -> "except for the arguments 'a' and 'b'"
 232     `format_argument_list(['a', 'b', 'c'])` ->
 233         "except for the arguments 'a', 'b' and 'c'"
 234     """
 235     if not allow_args:
 236         return ""
 237     elif allow_args == 1:
 238         return " except for the first argument"
 239     elif isinstance(allow_args, int):
 240         return f" except for the first {allow_args} arguments"
 241     elif len(allow_args) == 1:
 242         return f" except for the argument '{allow_args[0]}'"
 243     else:
 244         last = allow_args[-1]
 245         args = ", ".join(["'" + x + "'" for x in allow_args[:-1]])
 246         return f" except for the arguments {args} and '{last}'"

 249 def deprecate_nonkeyword_arguments(
 250     version: str,
 251     allowed_args: Optional[Union[List[str], int]] = None,
 252     stacklevel: int = 2,
 253 ) -> Callable:
 254     """
 255     Decorator to deprecate a use of non-keyword arguments of a function.
 256 
 257     Parameters
 258     ----------
 259     version : str
 260         The version in which positional arguments will become
 261         keyword-only.
 262 
 263     allowed_args : list or int, optional
 264         In case of list, it must be the list of names of some
 265         first arguments of the decorated functions that are
 266         OK to be given as positional arguments. In case of an
 267         integer, this is the number of positional arguments
 268         that will stay positional. In case of None value,
 269         defaults to list of all arguments not having the
 270         default value.
 271 
 272     stacklevel : int, default=2
 273         The stack level for warnings.warn
 274     """
 275 
 276     def decorate(func):
 277         if allowed_args is not None:
 278             allow_args = allowed_args
 279         else:
 280             spec = inspect.getfullargspec(func)
 281             allow_args = spec.args[: -len(spec.defaults)]
 282 
 283         @wraps(func)
 284         def wrapper(*args, **kwargs):
 285             arguments = _format_argument_list(allow_args)
 286             if isinstance(allow_args, (list, tuple)):
 287                 num_allow_args = len(allow_args)
 288             else:
 289                 num_allow_args = allow_args
 290             if len(args) > num_allow_args:
 291                 msg = (
 292                     f"Starting with Pandas version {version} all arguments of "
 293                     f"{func.__name__}{arguments} will be keyword-only"
 294                 )
 295                 warnings.warn(msg, FutureWarning, stacklevel=stacklevel)
 296             return func(*args, **kwargs)
 297 
 298         return wrapper
 299 
 300     return decorate

 303 def rewrite_axis_style_signature(
 304     name: str, extra_params: List[Tuple[str, Any]]
 305 ) -> Callable[..., Any]:
 306     def decorate(func: F) -> F:
 307         @wraps(func)
 308         def wrapper(*args, **kwargs) -> Callable[..., Any]:
 309             return func(*args, **kwargs)
 310 
 311         kind = inspect.Parameter.POSITIONAL_OR_KEYWORD
 312         params = [
 313             inspect.Parameter("self", kind),
 314             inspect.Parameter(name, kind, default=None),
 315             inspect.Parameter("index", kind, default=None),
 316             inspect.Parameter("columns", kind, default=None),
 317             inspect.Parameter("axis", kind, default=None),
 318         ]
 319 
 320         for pname, default in extra_params:
 321             params.append(inspect.Parameter(pname, kind, default=default))
 322 
 323         sig = inspect.Signature(params)
 324 
 325         # https://github.com/python/typing/issues/598
 326         func.__signature__ = sig  # type: ignore
 327         return cast(F, wrapper)
 328 
 329     return decorate

 332 def doc(*docstrings: Union[str, Callable], **params) -> Callable[[F], F]:
 333     """
 334     A decorator take docstring templates, concatenate them and perform string
 335     substitution on it.
 336 
 337     This decorator will add a variable "_docstring_components" to the wrapped
 338     callable to keep track the original docstring template for potential usage.
 339     If it should be consider as a template, it will be saved as a string.
 340     Otherwise, it will be saved as callable, and later user __doc__ and dedent
 341     to get docstring.
 342 
 343     Parameters
 344     ----------
 345     *docstrings : str or callable
 346         The string / docstring / docstring template to be appended in order
 347         after default docstring under callable.
 348     **params
 349         The string which would be used to format docstring template.
 350     """
 351 
 352     def decorator(decorated: F) -> F:
 353         # collecting docstring and docstring templates
 354         docstring_components: List[Union[str, Callable]] = []
 355         if decorated.__doc__:
 356             docstring_components.append(dedent(decorated.__doc__))
 357 
 358         for docstring in docstrings:
 359             if hasattr(docstring, "_docstring_components"):
 360                 docstring_components.extend(
 361                     docstring._docstring_components  # type: ignore
 362                 )
 363             elif isinstance(docstring, str) or docstring.__doc__:
 364                 docstring_components.append(docstring)
 365 
 366         # formatting templates and concatenating docstring
 367         decorated.__doc__ = "".join(
 368             [
 369                 component.format(**params)
 370                 if isinstance(component, str)
 371                 else dedent(component.__doc__ or "")
 372                 for component in docstring_components
 373             ]
 374         )
 375 
 376         decorated._docstring_components = docstring_components  # type: ignore
 377         return decorated
 378 
 379     return decorator

 386 class Substitution:
 387     """
 388     A decorator to take a function's docstring and perform string
 389     substitution on it.
 390 
 391     This decorator should be robust even if func.__doc__ is None
 392     (for example, if -OO was passed to the interpreter)
 393 
 394     Usage: construct a docstring.Substitution with a sequence or
 395     dictionary suitable for performing substitution; then
 396     decorate a suitable function with the constructed object. e.g.
 397 
 398     sub_author_name = Substitution(author='Jason')
 399 
 400     @sub_author_name
 401     def some_function(x):
 402         "%(author)s wrote this function"
 403 
 404     # note that some_function.__doc__ is now "Jason wrote this function"
 405 
 406     One can also use positional arguments.
 407 
 408     sub_first_last_names = Substitution('Edgar Allen', 'Poe')
 409 
 410     @sub_first_last_names
 411     def some_function(x):
 412         "%s %s wrote the Raven"
 413     """
 414 
 415     def __init__(self, *args, **kwargs):
 416         if args and kwargs:
 417             raise AssertionError("Only positional or keyword args are allowed")
 418 
 419         self.params = args or kwargs
 420 
 421     def __call__(self, func: F) -> F:
 422         func.__doc__ = func.__doc__ and func.__doc__ % self.params
 423         return func
 424 
 425     def update(self, *args, **kwargs) -> None:
 426         """
 427         Update self.params with supplied args.
 428         """
 429         if isinstance(self.params, dict):
 430             self.params.update(*args, **kwargs)

 433 class Appender:
 434     """
 435     A function decorator that will append an addendum to the docstring
 436     of the target function.
 437 
 438     This decorator should be robust even if func.__doc__ is None
 439     (for example, if -OO was passed to the interpreter).
 440 
 441     Usage: construct a docstring.Appender with a string to be joined to
 442     the original docstring. An optional 'join' parameter may be supplied
 443     which will be used to join the docstring and addendum. e.g.
 444 
 445     add_copyright = Appender("Copyright (c) 2009", join='\n')
 446 
 447     @add_copyright
 448     def my_dog(has='fleas'):
 449         "This docstring will have a copyright below"
 450         pass
 451     """
 452 
 453     addendum: Optional[str]
 454 
 455     def __init__(self, addendum: Optional[str], join: str = "", indents: int = 0):
 456         if indents > 0:
 457             self.addendum = indent(addendum, indents=indents)
 458         else:
 459             self.addendum = addendum
 460         self.join = join
 461 
 462     def __call__(self, func: F) -> F:
 463         func.__doc__ = func.__doc__ if func.__doc__ else ""
 464         self.addendum = self.addendum if self.addendum else ""
 465         docitems = [func.__doc__, self.addendum]
 466         func.__doc__ = dedent(self.join.join(docitems))
 467         return func

 470 def indent(text: Optional[str], indents: int = 1) -> str:
 471     if not text or not isinstance(text, str):
 472         return ""
 473     jointext = "".join(["\n"] + ["    "] * indents)
 474     return jointext.join(text.split("\n"))
