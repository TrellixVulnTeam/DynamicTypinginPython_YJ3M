   8 class TablePlotter:
   9     """
  10     Layout some DataFrames in vertical/horizontal layout for explanation.
  11     Used in merging.rst
  12     """
  13 
  14     def __init__(
  15         self,
  16         cell_width: float = 0.37,
  17         cell_height: float = 0.25,
  18         font_size: float = 7.5,
  19     ):
  20         self.cell_width = cell_width
  21         self.cell_height = cell_height
  22         self.font_size = font_size
  23 
  24     def _shape(self, df: pd.DataFrame) -> Tuple[int, int]:
  25         """
  26         Calculate table shape considering index levels.
  27         """
  28         row, col = df.shape
  29         return row + df.columns.nlevels, col + df.index.nlevels
  30 
  31     def _get_cells(self, left, right, vertical) -> Tuple[int, int]:
  32         """
  33         Calculate appropriate figure size based on left and right data.
  34         """
  35         if vertical:
  36             # calculate required number of cells
  37             vcells = max(sum(self._shape(l)[0] for l in left), self._shape(right)[0])
  38             hcells = max(self._shape(l)[1] for l in left) + self._shape(right)[1]
  39         else:
  40             vcells = max([self._shape(l)[0] for l in left] + [self._shape(right)[0]])
  41             hcells = sum([self._shape(l)[1] for l in left] + [self._shape(right)[1]])
  42         return hcells, vcells
  43 
  44     def plot(self, left, right, labels=None, vertical: bool = True):
  45         """
  46         Plot left / right DataFrames in specified layout.
  47 
  48         Parameters
  49         ----------
  50         left : list of DataFrames before operation is applied
  51         right : DataFrame of operation result
  52         labels : list of str to be drawn as titles of left DataFrames
  53         vertical : bool, default True
  54             If True, use vertical layout. If False, use horizontal layout.
  55         """
  56         import matplotlib.pyplot as plt
  57         import matplotlib.gridspec as gridspec
  58 
  59         if not isinstance(left, list):
  60             left = [left]
  61         left = [self._conv(l) for l in left]
  62         right = self._conv(right)
  63 
  64         hcells, vcells = self._get_cells(left, right, vertical)
  65 
  66         if vertical:
  67             figsize = self.cell_width * hcells, self.cell_height * vcells
  68         else:
  69             # include margin for titles
  70             figsize = self.cell_width * hcells, self.cell_height * vcells
  71         fig = plt.figure(figsize=figsize)
  72 
  73         if vertical:
  74             gs = gridspec.GridSpec(len(left), hcells)
  75             # left
  76             max_left_cols = max(self._shape(l)[1] for l in left)
  77             max_left_rows = max(self._shape(l)[0] for l in left)
  78             for i, (l, label) in enumerate(zip(left, labels)):
  79                 ax = fig.add_subplot(gs[i, 0:max_left_cols])
  80                 self._make_table(ax, l, title=label, height=1.0 / max_left_rows)
  81             # right
  82             ax = plt.subplot(gs[:, max_left_cols:])
  83             self._make_table(ax, right, title="Result", height=1.05 / vcells)
  84             fig.subplots_adjust(top=0.9, bottom=0.05, left=0.05, right=0.95)
  85         else:
  86             max_rows = max(self._shape(df)[0] for df in left + [right])
  87             height = 1.0 / np.max(max_rows)
  88             gs = gridspec.GridSpec(1, hcells)
  89             # left
  90             i = 0
  91             for l, label in zip(left, labels):
  92                 sp = self._shape(l)
  93                 ax = fig.add_subplot(gs[0, i : i + sp[1]])
  94                 self._make_table(ax, l, title=label, height=height)
  95                 i += sp[1]
  96             # right
  97             ax = plt.subplot(gs[0, i:])
  98             self._make_table(ax, right, title="Result", height=height)
  99             fig.subplots_adjust(top=0.85, bottom=0.05, left=0.05, right=0.95)
 100 
 101         return fig

 103     def _conv(self, data):
 104         """
 105         Convert each input to appropriate for table outplot.
 106         """
 107         if isinstance(data, pd.Series):
 108             if data.name is None:
 109                 data = data.to_frame(name="")
 110             else:
 111                 data = data.to_frame()
 112         data = data.fillna("NaN")
 113         return data
 114 
 115     def _insert_index(self, data):
 116         # insert is destructive
 117         data = data.copy()
 118         idx_nlevels = data.index.nlevels
 119         if idx_nlevels == 1:
 120             data.insert(0, "Index", data.index)
 121         else:
 122             for i in range(idx_nlevels):
 123                 data.insert(i, f"Index{i}", data.index._get_level_values(i))
 124 
 125         col_nlevels = data.columns.nlevels
 126         if col_nlevels > 1:
 127             col = data.columns._get_level_values(0)
 128             values = [
 129                 data.columns._get_level_values(i)._values for i in range(1, col_nlevels)
 130             ]
 131             col_df = pd.DataFrame(values)
 132             data.columns = col_df.columns
 133             data = pd.concat([col_df, data])
 134             data.columns = col
 135         return data
 136 
 137     def _make_table(self, ax, df, title: str, height: Optional[float] = None):
 138         if df is None:
 139             ax.set_visible(False)
 140             return
 141 
 142         import pandas.plotting as plotting
 143 
 144         idx_nlevels = df.index.nlevels
 145         col_nlevels = df.columns.nlevels
 146         # must be convert here to get index levels for colorization
 147         df = self._insert_index(df)
 148         tb = plotting.table(ax, df, loc=9)
 149         tb.set_fontsize(self.font_size)
 150 
 151         if height is None:
 152             height = 1.0 / (len(df) + 1)
 153 
 154         props = tb.properties()
 155         for (r, c), cell in props["celld"].items():
 156             if c == -1:
 157                 cell.set_visible(False)
 158             elif r < col_nlevels and c < idx_nlevels:
 159                 cell.set_visible(False)
 160             elif r < col_nlevels or c < idx_nlevels:
 161                 cell.set_facecolor("#AAAAAA")
 162             cell.set_height(height)
 163 
 164         ax.set_title(title, size=self.font_size)
 165         ax.axis("off")
