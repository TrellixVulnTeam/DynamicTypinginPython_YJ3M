 307     def test_melt_missing_columns_raises(self):
 308         # GH-23575
 309         # This test is to ensure that pandas raises an error if melting is
 310         # attempted with column names absent from the dataframe
 311 
 312         # Generate data
 313         df = pd.DataFrame(np.random.randn(5, 4), columns=list("abcd"))
 314 
 315         # Try to melt with missing `value_vars` column name
 316         msg = "The following '{Var}' are not present in the DataFrame: {Col}"
 317         with pytest.raises(
 318             KeyError, match=msg.format(Var="value_vars", Col="\\['C'\\]")
 319         ):
 320             df.melt(["a", "b"], ["C", "d"])
 321 
 322         # Try to melt with missing `id_vars` column name
 323         with pytest.raises(KeyError, match=msg.format(Var="id_vars", Col="\\['A'\\]")):
 324             df.melt(["A", "b"], ["c", "d"])
 325 
 326         # Multiple missing
 327         with pytest.raises(
 328             KeyError,
 329             match=msg.format(Var="id_vars", Col="\\['not_here', 'or_there'\\]"),
 330         ):
 331             df.melt(["a", "b", "not_here", "or_there"], ["c", "d"])
 332 
 333         # Multiindex melt fails if column is missing from multilevel melt
 334         multi = df.copy()
 335         multi.columns = [list("ABCD"), list("abcd")]
 336         with pytest.raises(KeyError, match=msg.format(Var="id_vars", Col="\\['E'\\]")):
 337             multi.melt([("E", "a")], [("B", "b")])
 338         # Multiindex fails if column is missing from single level melt
 339         with pytest.raises(
 340             KeyError, match=msg.format(Var="value_vars", Col="\\['F'\\]")
 341         ):
 342             multi.melt(["A"], ["F"], col_level=0)
