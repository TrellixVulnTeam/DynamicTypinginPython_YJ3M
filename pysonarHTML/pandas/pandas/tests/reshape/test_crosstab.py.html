 622         # GH 27500
 623         df = DataFrame(
 624             {
 625                 "A": ["foo", "foo", "foo", "foo", "foo", "bar", "bar", "bar", "bar"],
 626                 "B": ["one", "one", "one", "two", "two", "one", "one", "two", "two"],

 634                     "small",
 635                     "small",
 636                     "large",
 637                 ],

 639                 "E": [2, 4, 5, 5, 6, 6, 8, 9, 9],
 640             }
 641         )
 642         # normalize on index
 643         result = crosstab(
 644             [df.A, df.B], df.C, margins=True, margins_name="Sub-Total", normalize=0

 653         )
 654         expected.columns = Index(["large", "small"], dtype="object", name="C")
 655         tm.assert_frame_equal(result, expected)
 656 
 657         # normalize on columns
 658         result = crosstab(
 659             [df.A, df.B], df.C, margins=True, margins_name="Sub-Total", normalize=1
 660         )
 661         expected = DataFrame(
 662             [

 664                 [0.25, 0.2, 0.222222],
 665                 [0.5, 0.2, 0.333333],
 666                 [0, 0.4, 0.222222],
 667             ]
 668         )
 669         expected.columns = Index(
 670             ["large", "small", "Sub-Total"], dtype="object", name="C"
 671         )
 672         expected.index = MultiIndex(
 673             levels=[["bar", "foo"], ["one", "two"]],
 674             codes=[[0, 0, 1, 1], [0, 1, 0, 1]],
 675             names=["A", "B"],
 676         )
 677         tm.assert_frame_equal(result, expected)
 678 
 679         # normalize on both index and column
 680         result = crosstab(
 681             [df.A, df.B], df.C, margins=True, margins_name="Sub-Total", normalize=True
 682         )
 683         expected = DataFrame(
 684             [
 685                 [0.111111, 0.111111, 0.222222],
 686                 [0.111111, 0.111111, 0.222222],
 687                 [0.222222, 0.111111, 0.333333],
 688                 [0.000000, 0.222222, 0.222222],
 689                 [0.444444, 0.555555, 1],
 690             ]
 691         )
 692         expected.columns = Index(
 693             ["large", "small", "Sub-Total"], dtype="object", name="C"
 694         )
 695         expected.index = MultiIndex(
 696             levels=[["Sub-Total", "bar", "foo"], ["", "one", "two"]],
 697             codes=[[1, 1, 2, 2, 0], [1, 2, 1, 2, 0]],
 698             names=["A", "B"],
 699         )
