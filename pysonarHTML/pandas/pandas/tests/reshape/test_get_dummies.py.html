 130             {"a": [1, 0, 0], "b": [0, 1, 0]}, dtype=self.effective_dtype(dtype)

 133             exp = exp.apply(SparseArray, fill_value=0.0)

 141         )
 142         exp_na = exp_na.reindex(["a", "b", np.nan], axis=1)

 146             exp_na = exp_na.apply(SparseArray, fill_value=0.0)
