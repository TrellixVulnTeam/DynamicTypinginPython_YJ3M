 547                 on="index",
 548                 by="ticker",

 550             )

 552         # invalid negative
 553         with pytest.raises(MergeError):
 554             merge_asof(
 555                 trades, quotes, on="time", by="ticker", tolerance=-Timedelta("1s")
 556             )
