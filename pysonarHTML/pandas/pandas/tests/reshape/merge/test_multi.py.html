 124     @pytest.mark.parametrize("sort", [False, True])
 125     def test_left_join_multi_index(self, left, right, sort):
 126         icols = ["1st", "2nd", "3rd"]
 127 
 128         def bind_cols(df):
 129             iord = lambda a: 0 if a != a else ord(a)
 130             f = lambda ts: ts.map(iord) - ord("a")
 131             return f(df["1st"]) + f(df["3rd"]) * 1e2 + df["2nd"].fillna(0) * 1e4
 132 
 133         def run_asserts(left, right, sort):
 134             res = left.join(right, on=icols, how="left", sort=sort)
 135 
 136             assert len(left) < len(res) + 1
 137             assert not res["4th"].isna().any()
 138             assert not res["5th"].isna().any()
 139 
 140             tm.assert_series_equal(res["4th"], -res["5th"], check_names=False)
 141             result = bind_cols(res.iloc[:, :-2])
 142             tm.assert_series_equal(res["4th"], result, check_names=False)
 143             assert result.name is None
 144 
 145             if sort:
 146                 tm.assert_frame_equal(res, res.sort_values(icols, kind="mergesort"))
 147 
 148             out = merge(left, right.reset_index(), on=icols, sort=sort, how="left")
 149 
 150             res.index = np.arange(len(res))
 151             tm.assert_frame_equal(out, res)
 152 
 153         lc = list(map(chr, np.arange(ord("a"), ord("z") + 1)))
 154         left = DataFrame(np.random.choice(lc, (5000, 2)), columns=["1st", "3rd"])
 155         left.insert(1, "2nd", np.random.randint(0, 1000, len(left)))
 156 
 157         i = np.random.permutation(len(left))
 158         right = left.iloc[i].copy()
 159 
 160         left["4th"] = bind_cols(left)
 161         right["5th"] = -bind_cols(right)
 162         right.set_index(icols, inplace=True)
 163 
 164         run_asserts(left, right, sort)
 165 
 166         # inject some nulls
 167         left.loc[1::23, "1st"] = np.nan
 168         left.loc[2::37, "2nd"] = np.nan
 169         left.loc[3::43, "3rd"] = np.nan
 170         left["4th"] = bind_cols(left)
 171 
 172         i = np.random.permutation(len(left))
 173         right = left.iloc[i, :-1]
 174         right["5th"] = -bind_cols(right)
 175         right.set_index(icols, inplace=True)
 176 
 177         run_asserts(left, right, sort)
