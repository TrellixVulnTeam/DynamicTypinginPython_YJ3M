 798 def _check_join(left, right, result, join_col, how="left", lsuffix="_x", rsuffix="_y"):
 799 
 800     # some smoke tests
 801     for c in join_col:
 802         assert result[c].notna().all()
 803 
 804     left_grouped = left.groupby(join_col)
 805     right_grouped = right.groupby(join_col)
 806 
 807     for group_key, group in result.groupby(join_col):
 808         l_joined = _restrict_to_columns(group, left.columns, lsuffix)
 809         r_joined = _restrict_to_columns(group, right.columns, rsuffix)
 810 
 811         try:
 812             lgroup = left_grouped.get_group(group_key)
 813         except KeyError as err:
 814             if how in ("left", "inner"):
 815                 raise AssertionError(
 816                     f"key {group_key} should not have been in the join"
 817                 ) from err
 818 
 819             _assert_all_na(l_joined, left.columns, join_col)
 820         else:
 821             _assert_same_contents(l_joined, lgroup)
 822 
 823         try:
 824             rgroup = right_grouped.get_group(group_key)
 825         except KeyError as err:
 826             if how in ("right", "inner"):
 827                 raise AssertionError(
 828                     f"key {group_key} should not have been in the join"
 829                 ) from err
 830 
 831             _assert_all_na(r_joined, right.columns, join_col)
 832         else:
 833             _assert_same_contents(r_joined, rgroup)
 834

 836 def _restrict_to_columns(group, columns, suffix):
 837     found = [
 838         c for c in group.columns if c in columns or c.replace(suffix, "") in columns
 839     ]
 840 
 841     # filter
 842     group = group.loc[:, found]
 843 
 844     # get rid of suffixes, if any
 845     group = group.rename(columns=lambda x: x.replace(suffix, ""))
 846 
 847     # put in the right order...
 848     group = group.loc[:, columns]
 849 
 850     return group

 853 def _assert_same_contents(join_chunk, source):
 854     NA_SENTINEL = -1234567  # drop_duplicates not so NA-friendly...
 855 
 856     jvalues = join_chunk.fillna(NA_SENTINEL).drop_duplicates().values
 857     svalues = source.fillna(NA_SENTINEL).drop_duplicates().values
 858 
 859     rows = {tuple(row) for row in jvalues}
 860     assert len(rows) == len(source)
 861     assert all(tuple(row) in rows for row in svalues)
 862

 864 def _assert_all_na(join_chunk, source_columns, join_col):
 865     for c in source_columns:
 866         if c in join_col:
 867             continue
 868         assert join_chunk[c].isna().all()

 871 def _join_by_hand(a, b, how="left"):
 872     join_index = a.index.join(b.index, how=how)
 873 
 874     a_re = a.reindex(join_index)
 875     b_re = b.reindex(join_index)
 876 
 877     result_columns = a.columns.append(b.columns)
 878 
 879     for col, s in b_re.items():
 880         a_re[col] = s
 881     return a_re.reindex(columns=result_columns)
