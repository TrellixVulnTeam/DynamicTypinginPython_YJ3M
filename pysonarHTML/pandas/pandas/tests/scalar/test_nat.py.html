 219 def _get_overlap_public_nat_methods(klass, as_tuple=False):
 220     """
 221     Get overlapping public methods between NaT and another class.
 222 
 223     Parameters
 224     ----------
 225     klass : type
 226         The class to compare with NaT
 227     as_tuple : bool, default False
 228         Whether to return a list of tuples of the form (klass, method).
 229 
 230     Returns
 231     -------
 232     overlap : list
 233     """
 234     nat_names = dir(NaT)
 235     klass_names = dir(klass)
 236 
 237     overlap = [
 238         x
 239         for x in nat_names
 240         if x in klass_names and not x.startswith("_") and callable(getattr(klass, x))
 241     ]
 242 
 243     # Timestamp takes precedence over Timedelta in terms of overlap.
 244     if klass is Timedelta:
 245         ts_names = dir(Timestamp)
 246         overlap = [x for x in overlap if x not in ts_names]
 247 
 248     if as_tuple:
 249         overlap = [(klass, method) for method in overlap]
 250 
 251     overlap.sort()
 252     return overlap

 352 @pytest.mark.parametrize("op_name", list(_ops.keys()))
 353 @pytest.mark.parametrize(
 354     "value,val_type",
 355     [
 356         (2, "scalar"),
 357         (1.5, "floating"),
 358         (np.nan, "floating"),
 359         ("foo", "str"),
 360         (timedelta(3600), "timedelta"),
 361         (Timedelta("5s"), "timedelta"),
 362         (datetime(2014, 1, 1), "timestamp"),
 363         (Timestamp("2014-01-01"), "timestamp"),
 364         (Timestamp("2014-01-01", tz="UTC"), "timestamp"),
 365         (Timestamp("2014-01-01", tz="US/Eastern"), "timestamp"),
 366         (pytz.timezone("Asia/Tokyo").localize(datetime(2014, 1, 1)), "timestamp"),
 367     ],
 368 )
 369 def test_nat_arithmetic_scalar(op_name, value, val_type):
 370     # see gh-6873
 371     invalid_ops = {
 372         "scalar": {"right_div_left"},
 373         "floating": {
 374             "right_div_left",
 375             "left_minus_right",
 376             "right_minus_left",
 377             "left_plus_right",
 378             "right_plus_left",
 379         },
 380         "str": set(_ops.keys()),
 381         "timedelta": {"left_times_right", "right_times_left"},
 382         "timestamp": {
 383             "left_times_right",
 384             "right_times_left",
 385             "left_div_right",
 386             "right_div_left",
 387         },
 388     }
 389 
 390     op = _ops[op_name]
 391 
 392     if op_name in invalid_ops.get(val_type, set()):
 393         if (
 394             val_type == "timedelta"
 395             and "times" in op_name
 396             and isinstance(value, Timedelta)
 397         ):
 398             typs = "(Timedelta|NaTType)"
 399             msg = rf"unsupported operand type\(s\) for \*: '{typs}' and '{typs}'"
 400         elif val_type == "str":
 401             # un-specific check here because the message comes from str
 402             #  and varies by method
 403             msg = "|".join(
 404                 [
 405                     "can only concatenate str",
 406                     "unsupported operand type",
 407                     "can't multiply sequence",
 408                     "Can't convert 'NaTType'",
 409                     "must be str, not NaTType",
 410                 ]
 411             )
 412         else:
 413             msg = "unsupported operand type"
 414 
 415         with pytest.raises(TypeError, match=msg):
 416             op(NaT, value)
 417     else:
 418         if val_type == "timedelta" and "div" in op_name:
 419             expected = np.nan
 420         else:
 421             expected = NaT
 422 
 423         assert op(NaT, value) is expected
