  23 class TestOverlaps:
  24     def test_overlaps_self(self, start_shift, closed):
  25         start, shift = start_shift
  26         interval = Interval(start, start + shift, closed)
  27         assert interval.overlaps(interval)
  28 
  29     def test_overlaps_nested(self, start_shift, closed, other_closed):
  30         start, shift = start_shift
  31         interval1 = Interval(start, start + 3 * shift, other_closed)
  32         interval2 = Interval(start + shift, start + 2 * shift, closed)
  33 
  34         # nested intervals should always overlap
  35         assert interval1.overlaps(interval2)
  36 
  37     def test_overlaps_disjoint(self, start_shift, closed, other_closed):
  38         start, shift = start_shift
  39         interval1 = Interval(start, start + shift, other_closed)
  40         interval2 = Interval(start + 2 * shift, start + 3 * shift, closed)
  41 
  42         # disjoint intervals should never overlap
  43         assert not interval1.overlaps(interval2)
  44 
  45     def test_overlaps_endpoint(self, start_shift, closed, other_closed):
  46         start, shift = start_shift
  47         interval1 = Interval(start, start + shift, other_closed)
  48         interval2 = Interval(start + shift, start + 2 * shift, closed)
  49 
  50         # overlap if shared endpoint is closed for both (overlap at a point)
  51         result = interval1.overlaps(interval2)
  52         expected = interval1.closed_right and interval2.closed_left
  53         assert result == expected
  54 
  55     @pytest.mark.parametrize(
  56         "other",
  57         [10, True, "foo", Timedelta("1 day"), Timestamp("2018-01-01")],
  58         ids=lambda x: type(x).__name__,
  59     )
  60     def test_overlaps_invalid_type(self, other):
  61         interval = Interval(0, 1)
  62         msg = f"`other` must be an Interval, got {type(other).__name__}"
  63         with pytest.raises(TypeError, match=msg):
  64             interval.overlaps(other)
