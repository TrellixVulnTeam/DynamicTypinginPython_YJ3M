 238         msg = "Must supply freq for datetime value"
 239         with pytest.raises(ValueError, match=msg):
 240             Period(datetime.now())
 241         with pytest.raises(ValueError, match=msg):
 242             Period(datetime.now().date())
 243 
 244         msg = "Value must be Period, string, integer, or datetime"
 245         with pytest.raises(ValueError, match=msg):
 246             Period(1.6, freq="D")
 247         msg = "Ordinal must be an integer"
 248         with pytest.raises(ValueError, match=msg):
 249             Period(ordinal=1.6, freq="D")
 250         msg = "Only value or ordinal but not both should be given but not both"
 251         with pytest.raises(ValueError, match=msg):
 252             Period(ordinal=2, value=1, freq="D")
 253 
 254         msg = "If value is None, freq cannot be None"
 255         with pytest.raises(ValueError, match=msg):
 256             Period(month=1)
 257 
 258         msg = "Given date string not likely a datetime"
 259         with pytest.raises(ValueError, match=msg):
 260             Period("-2000", "A")
 261         msg = "day is out of range for month"
 262         with pytest.raises(DateParseError, match=msg):
 263             Period("0", "A")
 264         msg = "Unknown datetime string format, unable to parse"
 265         with pytest.raises(DateParseError, match=msg):
