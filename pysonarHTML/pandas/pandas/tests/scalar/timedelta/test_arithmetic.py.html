 254     # TODO: moved from index tests following #24365, may need de-duplication
 255     def test_ops_ndarray(self):
 256         td = Timedelta("1 day")
 257 
 258         # timedelta, timedelta
 259         other = pd.to_timedelta(["1 day"]).values
 260         expected = pd.to_timedelta(["2 days"]).values
 261         tm.assert_numpy_array_equal(td + other, expected)
 262         tm.assert_numpy_array_equal(other + td, expected)
 263         msg = r"unsupported operand type\(s\) for \+: 'Timedelta' and 'int'"
 264         with pytest.raises(TypeError, match=msg):
 265             td + np.array([1])
 266         msg = r"unsupported operand type\(s\) for \+: 'numpy.ndarray' and 'Timedelta'"
 267         with pytest.raises(TypeError, match=msg):
 268             np.array([1]) + td
 269 
 270         expected = pd.to_timedelta(["0 days"]).values
 271         tm.assert_numpy_array_equal(td - other, expected)
 272         tm.assert_numpy_array_equal(-other + td, expected)
 273         msg = r"unsupported operand type\(s\) for -: 'Timedelta' and 'int'"
 274         with pytest.raises(TypeError, match=msg):
 275             td - np.array([1])
 276         msg = r"unsupported operand type\(s\) for -: 'numpy.ndarray' and 'Timedelta'"
 277         with pytest.raises(TypeError, match=msg):
 278             np.array([1]) - td
 279 
 280         expected = pd.to_timedelta(["2 days"]).values
 281         tm.assert_numpy_array_equal(td * np.array([2]), expected)
 282         tm.assert_numpy_array_equal(np.array([2]) * td, expected)
 283         msg = (
 284             "ufunc '?multiply'? cannot use operands with types "
 285             r"dtype\('<m8\[ns\]'\) and dtype\('<m8\[ns\]'\)"
 286         )
 287         with pytest.raises(TypeError, match=msg):
 288             td * other
 289         with pytest.raises(TypeError, match=msg):
 290             other * td
 291 
 292         tm.assert_numpy_array_equal(td / other, np.array([1], dtype=np.float64))
 293         tm.assert_numpy_array_equal(other / td, np.array([1], dtype=np.float64))
 294 
 295         # timedelta, datetime
 296         other = pd.to_datetime(["2000-01-01"]).values
 297         expected = pd.to_datetime(["2000-01-02"]).values
 298         tm.assert_numpy_array_equal(td + other, expected)
 299         tm.assert_numpy_array_equal(other + td, expected)
 300 
 301         expected = pd.to_datetime(["1999-12-31"]).values
 302         tm.assert_numpy_array_equal(-td + other, expected)
 303         tm.assert_numpy_array_equal(other - td, expected)
