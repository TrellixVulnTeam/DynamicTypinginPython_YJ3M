  11     timezones = ["UTC", "Asia/Tokyo", "US/Eastern", "dateutil/US/Pacific"]
  12 
  13     @pytest.mark.parametrize("tz", timezones)
  14     @pytest.mark.parametrize("freq", ["D", "M", "S", "N"])
  15     @pytest.mark.parametrize(
  16         "date", ["2014-03-07", "2014-01-01 09:00", "2014-01-01 00:00:00.000000001"]
  17     )
  18     def test_repr(self, date, freq, tz):
  19         # avoid to match with timezone name
  20         freq_repr = f"'{freq}'"
  21         if tz.startswith("dateutil"):
  22             tz_repr = tz.replace("dateutil", "")
  23         else:
  24             tz_repr = tz
  25 
  26         date_only = Timestamp(date)
  27         assert date in repr(date_only)
  28         assert tz_repr not in repr(date_only)
  29         assert freq_repr not in repr(date_only)
  30         assert date_only == eval(repr(date_only))
  31 
  32         date_tz = Timestamp(date, tz=tz)
  33         assert date in repr(date_tz)
  34         assert tz_repr in repr(date_tz)
  35         assert freq_repr not in repr(date_tz)
  36         assert date_tz == eval(repr(date_tz))
  37 
  38         date_freq = Timestamp(date, freq=freq)
  39         assert date in repr(date_freq)
  40         assert tz_repr not in repr(date_freq)
  41         assert freq_repr in repr(date_freq)
  42         assert date_freq == eval(repr(date_freq))
  43 
  44         date_tz_freq = Timestamp(date, tz=tz, freq=freq)
  45         assert date in repr(date_tz_freq)
  46         assert tz_repr in repr(date_tz_freq)
  47         assert freq_repr in repr(date_tz_freq)
  48         assert date_tz_freq == eval(repr(date_tz_freq))

  50     def test_repr_utcoffset(self):
  51         # This can cause the tz field to be populated, but it's redundant to
  52         # include this information in the date-string.
  53         date_with_utc_offset = Timestamp("2014-03-13 00:00:00-0400", tz=None)
  54         assert "2014-03-13 00:00:00-0400" in repr(date_with_utc_offset)
  55         assert "tzoffset" not in repr(date_with_utc_offset)
  56         assert "pytz.FixedOffset(-240)" in repr(date_with_utc_offset)
  57         expr = repr(date_with_utc_offset).replace(
  58             "'pytz.FixedOffset(-240)'", "pytz.FixedOffset(-240)"
  59         )
  60         assert date_with_utc_offset == eval(expr)
