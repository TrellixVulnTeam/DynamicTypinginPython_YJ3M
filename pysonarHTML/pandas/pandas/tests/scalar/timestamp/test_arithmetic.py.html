 109     def test_addition_subtraction_types(self):
 110         # Assert on the types resulting from Timestamp +/- various date/time
 111         # objects
 112         dt = datetime(2014, 3, 4)
 113         td = timedelta(seconds=1)
 114         # build a timestamp with a frequency, since then it supports
 115         # addition/subtraction of integers
 116         ts = Timestamp(dt, freq="D")
 117 
 118         msg = "Addition/subtraction of integers"
 119         with pytest.raises(TypeError, match=msg):
 120             # GH#22535 add/sub with integers is deprecated
 121             ts + 1
 122         with pytest.raises(TypeError, match=msg):
 123             ts - 1
 124 
 125         # Timestamp + datetime not supported, though subtraction is supported
 126         # and yields timedelta more tests in tseries/base/tests/test_base.py
 127         assert type(ts - dt) == Timedelta
 128         assert type(ts + td) == Timestamp
 129         assert type(ts - td) == Timestamp
 130 
 131         # Timestamp +/- datetime64 not supported, so not tested (could possibly
 132         # assert error raised?)
 133         td64 = np.timedelta64(1, "D")
 134         assert type(ts + td64) == Timestamp
 135         assert type(ts - td64) == Timestamp

 137     @pytest.mark.parametrize(
 138         "freq, td, td64",
 139         [
 140             ("S", timedelta(seconds=1), np.timedelta64(1, "s")),
 141             ("min", timedelta(minutes=1), np.timedelta64(1, "m")),
 142             ("H", timedelta(hours=1), np.timedelta64(1, "h")),
 143             ("D", timedelta(days=1), np.timedelta64(1, "D")),
 144             ("W", timedelta(weeks=1), np.timedelta64(1, "W")),
 145             ("M", None, np.timedelta64(1, "M")),
 146         ],
 147     )
 148     def test_addition_subtraction_preserve_frequency(self, freq, td, td64):
 149         ts = Timestamp("2014-03-05 00:00:00", freq=freq)
 150         original_freq = ts.freq
 151 
 152         assert (ts + 1 * original_freq).freq == original_freq
 153         assert (ts - 1 * original_freq).freq == original_freq
 154 
 155         if td is not None:
 156             # timedelta does not support months as unit
 157             assert (ts + td).freq == original_freq
 158             assert (ts - td).freq == original_freq
 159 
 160         assert (ts + td64).freq == original_freq
 161         assert (ts - td64).freq == original_freq

 216     @pytest.mark.parametrize("shape", [(6,), (2, 3,)])
 217     def test_addsub_m8ndarray(self, shape):
 218         # GH#33296
 219         ts = Timestamp("2020-04-04 15:45")
 220         other = np.arange(6).astype("m8[h]").reshape(shape)
