  23     def test_tz_localize_pushes_out_of_bounds(self):
  24         msg = "^$"
  25         # GH#12677
  26         # tz_localize that pushes away from the boundary is OK
  27         pac = Timestamp.min.tz_localize("US/Pacific")
  28         assert pac.value > Timestamp.min.value
  29         pac.tz_convert("Asia/Tokyo")  # tz_convert doesn't change value
  30         with pytest.raises(OutOfBoundsDatetime, match=msg):
  31             Timestamp.min.tz_localize("Asia/Tokyo")
  32 
  33         # tz_localize that pushes away from the boundary is OK
  34         tokyo = Timestamp.max.tz_localize("Asia/Tokyo")
  35         assert tokyo.value < Timestamp.max.value
  36         tokyo.tz_convert("US/Pacific")  # tz_convert doesn't change value
  37         with pytest.raises(OutOfBoundsDatetime, match=msg):
  38             Timestamp.max.tz_localize("US/Pacific")

  40     def test_tz_localize_ambiguous_bool(self):
  41         # make sure that we are correctly accepting bool values as ambiguous
  42         # GH#14402
  43         ts = Timestamp("2015-11-01 01:00:03")
  44         expected0 = Timestamp("2015-11-01 01:00:03-0500", tz="US/Central")
  45         expected1 = Timestamp("2015-11-01 01:00:03-0600", tz="US/Central")
  46 
  47         msg = "Cannot infer dst time from 2015-11-01 01:00:03"
  48         with pytest.raises(pytz.AmbiguousTimeError, match=msg):
  49             ts.tz_localize("US/Central")
  50 
  51         result = ts.tz_localize("US/Central", ambiguous=True)
  52         assert result == expected0
  53 
  54         result = ts.tz_localize("US/Central", ambiguous=False)
  55         assert result == expected1

  57     def test_tz_localize_ambiguous(self):
  58         ts = Timestamp("2014-11-02 01:00")
  59         ts_dst = ts.tz_localize("US/Eastern", ambiguous=True)
  60         ts_no_dst = ts.tz_localize("US/Eastern", ambiguous=False)
  61 
  62         assert (ts_no_dst.value - ts_dst.value) / 1e9 == 3600
  63         msg = "Cannot infer offset with only one time"
  64         with pytest.raises(ValueError, match=msg):
  65             ts.tz_localize("US/Eastern", ambiguous="infer")
  66 
  67         # GH#8025
  68         msg = "Cannot localize tz-aware Timestamp, use tz_convert for conversions"
  69         with pytest.raises(TypeError, match=msg):
  70             Timestamp("2011-01-01", tz="US/Eastern").tz_localize("Asia/Tokyo")
  71 
  72         msg = "Cannot convert tz-naive Timestamp, use tz_localize to localize"
  73         with pytest.raises(TypeError, match=msg):

 136     def test_tz_localize_ambiguous_compat(self):
 137         # validate that pytz and dateutil are compat for dst
 138         # when the transition happens
 139         naive = Timestamp("2013-10-27 01:00:00")
 140 
 141         pytz_zone = "Europe/London"
 142         dateutil_zone = "dateutil/Europe/London"
 143         result_pytz = naive.tz_localize(pytz_zone, ambiguous=0)
 144         result_dateutil = naive.tz_localize(dateutil_zone, ambiguous=0)
 145         assert result_pytz.value == result_dateutil.value
 146         assert result_pytz.value == 1382835600000000000
 147 
 148         # fixed ambiguous behavior
 149         # see gh-14621
 150         assert result_pytz.to_pydatetime().tzname() == "GMT"
 151         assert result_dateutil.to_pydatetime().tzname() == "BST"
 152         assert str(result_pytz) == str(result_dateutil)
 153 
 154         # 1 hour difference
 155         result_pytz = naive.tz_localize(pytz_zone, ambiguous=1)
 156         result_dateutil = naive.tz_localize(dateutil_zone, ambiguous=1)
 157         assert result_pytz.value == result_dateutil.value
 158         assert result_pytz.value == 1382832000000000000
 159 
 160         # see gh-14621
 161         assert str(result_pytz) == str(result_dateutil)
 162         assert (
 163             result_pytz.to_pydatetime().tzname()
 164             == result_dateutil.to_pydatetime().tzname()
 165         )
