 226     def test_round_int64(self, timestamp, freq):
 227         # check that all rounding modes are accurate to int64 precision
 228         # see GH#22591
 229         dt = Timestamp(timestamp)
 230         unit = to_offset(freq).nanos
 231 
 232         # test floor
 233         result = dt.floor(freq)
 234         assert result.value % unit == 0, f"floor not a {freq} multiple"
 235         assert 0 <= dt.value - result.value < unit, "floor error"
 236 
 237         # test ceil
 238         result = dt.ceil(freq)
 239         assert result.value % unit == 0, f"ceil not a {freq} multiple"
 240         assert 0 <= result.value - dt.value < unit, "ceil error"
 241 
 242         # test round
 243         result = dt.round(freq)
 244         assert result.value % unit == 0, f"round not a {freq} multiple"
 245         assert abs(result.value - dt.value) <= unit // 2, "round error"
 246         if unit % 2 == 0 and abs(result.value - dt.value) == unit // 2:
 247             # round half to even
 248             assert result.value // unit % 2 == 0, "round half to even error"

 402     @td.skip_if_windows
 403     def test_timestamp(self):
 404         # GH#17329
 405         # tz-naive --> treat it as if it were UTC for purposes of timestamp()
 406         ts = Timestamp.now()
 407         uts = ts.replace(tzinfo=utc)
 408         assert ts.timestamp() == uts.timestamp()
 409 
 410         tsc = Timestamp("2014-10-11 11:00:01.12345678", tz="US/Central")
 411         utsc = tsc.tz_convert("UTC")
 412 
 413         # utsc is a different representation of the same time
 414         assert tsc.timestamp() == utsc.timestamp()
 415 
 416         # datetime.timestamp() converts in the local timezone
 417         with tm.set_timezone("UTC"):
 418             # should agree with datetime.timestamp method
 419             dt = ts.to_pydatetime()
 420             assert dt.timestamp() == ts.timestamp()
