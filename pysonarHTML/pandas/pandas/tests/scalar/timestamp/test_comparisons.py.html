 171     def test_timestamp_compare_scalars(self):
 172         # case where ndim == 0
 173         lhs = np.datetime64(datetime(2013, 12, 6))
 174         rhs = Timestamp("now")
 175         nat = Timestamp("nat")
 176 
 177         ops = {"gt": "lt", "lt": "gt", "ge": "le", "le": "ge", "eq": "eq", "ne": "ne"}
 178 
 179         for left, right in ops.items():
 180             left_f = getattr(operator, left)
 181             right_f = getattr(operator, right)
 182             expected = left_f(lhs, rhs)
 183 
 184             result = right_f(rhs, lhs)
 185             assert result == expected
 186 
 187             expected = left_f(rhs, nat)
 188             result = right_f(nat, rhs)
 189             assert result == expected

 219 def test_rich_comparison_with_unsupported_type():
 220     # Comparisons with unsupported objects should return NotImplemented
 221     # (it previously raised TypeError, see #24011)
 222 
 223     class Inf:
 224         def __lt__(self, o):
 225             return False
 226 
 227         def __le__(self, o):
 228             return isinstance(o, Inf)
 229 
 230         def __gt__(self, o):
 231             return not isinstance(o, Inf)
 232 
 233         def __ge__(self, o):
 234             return True
 235 
 236         def __eq__(self, other) -> bool:
 237             return isinstance(other, Inf)
 238 
 239     inf = Inf()
 240     timestamp = Timestamp("2018-11-30")
 241 
 242     for left, right in [(inf, timestamp), (timestamp, inf)]:
 243         assert left > right or left < right
 244         assert left >= right or left <= right
 245         assert not (left == right)
 246         assert left != right
