 364     def test_out_of_bounds_value(self):
 365         one_us = np.timedelta64(1).astype("timedelta64[us]")
 366 
 367         # By definition we can't go out of bounds in [ns], so we
 368         # convert the datetime64s to [us] so we can go out of bounds
 369         min_ts_us = np.datetime64(Timestamp.min).astype("M8[us]")
 370         max_ts_us = np.datetime64(Timestamp.max).astype("M8[us]")
 371 
 372         # No error for the min/max datetimes
 373         Timestamp(min_ts_us)
 374         Timestamp(max_ts_us)
 375 
 376         msg = "Out of bounds"
 377         # One us less than the minimum is an error
 378         with pytest.raises(ValueError, match=msg):
 379             Timestamp(min_ts_us - one_us)
 380 
 381         # One us more than the maximum is an error
 382         with pytest.raises(ValueError, match=msg):
 383             Timestamp(max_ts_us + one_us)
