  13 def test_unique(index_or_series_obj):
  14     obj = index_or_series_obj
  15     obj = np.repeat(obj, range(1, len(obj) + 1))
  16     result = obj.unique()
  17 
  18     # dict.fromkeys preserves the order
  19     unique_values = list(dict.fromkeys(obj.values))
  20     if isinstance(obj, pd.MultiIndex):
  21         expected = pd.MultiIndex.from_tuples(unique_values)
  22         expected.names = obj.names
  23         tm.assert_index_equal(result, expected)
  24     elif isinstance(obj, pd.Index):
  25         expected = pd.Index(unique_values, dtype=obj.dtype)
  26         if is_datetime64tz_dtype(obj.dtype):
  27             expected = expected.normalize()
  28         tm.assert_index_equal(result, expected)
  29     else:
  30         expected = np.array(unique_values)
  31         tm.assert_numpy_array_equal(result, expected)

  34 @pytest.mark.parametrize("null_obj", [np.nan, None])
  35 def test_unique_null(null_obj, index_or_series_obj):
  36     obj = index_or_series_obj
  37 
  38     if not allow_na_ops(obj):
  39         pytest.skip("type doesn't allow for NA operations")
  40     elif len(obj) < 1:
  41         pytest.skip("Test doesn't make sense on empty data")
  42     elif isinstance(obj, pd.MultiIndex):
  43         pytest.skip(f"MultiIndex can't hold '{null_obj}'")
  44 
  45     values = obj.values
  46     if needs_i8_conversion(obj.dtype):
  47         values[0:2] = iNaT
  48     else:
  49         values[0:2] = null_obj
  50 
  51     klass = type(obj)
  52     repeated_values = np.repeat(values, range(1, len(values) + 1))
  53     obj = klass(repeated_values, dtype=obj.dtype)
  54     result = obj.unique()
  55 
  56     unique_values_raw = dict.fromkeys(obj.values)
  57     # because np.nan == np.nan is False, but None == None is True
  58     # np.nan would be duplicated, whereas None wouldn't
  59     unique_values_not_null = [val for val in unique_values_raw if not pd.isnull(val)]
  60     unique_values = [null_obj] + unique_values_not_null
  61 
  62     if isinstance(obj, pd.Index):
  63         expected = pd.Index(unique_values, dtype=obj.dtype)
  64         if is_datetime64tz_dtype(obj.dtype):
  65             result = result.normalize()
  66             expected = expected.normalize()
  67         elif isinstance(obj, pd.CategoricalIndex):
  68             expected = expected.set_categories(unique_values_not_null)
  69         tm.assert_index_equal(result, expected)
  70     else:
  71         expected = np.array(unique_values, dtype=obj.dtype)
  72         tm.assert_numpy_array_equal(result, expected)

  75 def test_nunique(index_or_series_obj):
  76     obj = index_or_series_obj
  77     obj = np.repeat(obj, range(1, len(obj) + 1))
  78     expected = len(obj.unique())
  79     assert obj.nunique(dropna=False) == expected
