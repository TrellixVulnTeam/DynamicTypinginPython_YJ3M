  27 def test_value_counts(index_or_series_obj):
  28     obj = index_or_series_obj
  29     obj = np.repeat(obj, range(1, len(obj) + 1))
  30     result = obj.value_counts()
  31 
  32     counter = collections.Counter(obj)
  33     expected = pd.Series(dict(counter.most_common()), dtype=np.int64, name=obj.name)
  34     expected.index = expected.index.astype(obj.dtype)
  35     if isinstance(obj, pd.MultiIndex):
  36         expected.index = pd.Index(expected.index)
  37 
  38     # TODO: Order of entries with the same count is inconsistent on CI (gh-32449)
  39     if obj.duplicated().any():
  40         result = result.sort_index()
  41         expected = expected.sort_index()
  42     tm.assert_series_equal(result, expected)

  45 @pytest.mark.parametrize("null_obj", [np.nan, None])
  46 def test_value_counts_null(null_obj, index_or_series_obj):
  47     orig = index_or_series_obj
  48     obj = orig.copy()
  49 
  50     if not allow_na_ops(obj):
  51         pytest.skip("type doesn't allow for NA operations")
  52     elif len(obj) < 1:
  53         pytest.skip("Test doesn't make sense on empty data")
  54     elif isinstance(orig, pd.MultiIndex):
  55         pytest.skip(f"MultiIndex can't hold '{null_obj}'")
  56 
  57     values = obj.values
  58     if needs_i8_conversion(obj.dtype):
  59         values[0:2] = iNaT
  60     else:
  61         values[0:2] = null_obj
  62 
  63     klass = type(obj)
  64     repeated_values = np.repeat(values, range(1, len(values) + 1))
  65     obj = klass(repeated_values, dtype=obj.dtype)
  66 
  67     # because np.nan == np.nan is False, but None == None is True
  68     # np.nan would be duplicated, whereas None wouldn't
  69     counter = collections.Counter(obj.dropna())
  70     expected = pd.Series(dict(counter.most_common()), dtype=np.int64)
  71     expected.index = expected.index.astype(obj.dtype)
  72 
  73     result = obj.value_counts()
  74     if obj.duplicated().any():
  75         # TODO:
  76         #  Order of entries with the same count is inconsistent on CI (gh-32449)
  77         expected = expected.sort_index()
  78         result = result.sort_index()
  79     tm.assert_series_equal(result, expected)
  80 
  81     # can't use expected[null_obj] = 3 as
  82     # IntervalIndex doesn't allow assignment
  83     new_entry = pd.Series({np.nan: 3}, dtype=np.int64)
  84     expected = expected.append(new_entry)
  85 
  86     result = obj.value_counts(dropna=False)
  87     if obj.duplicated().any():
  88         # TODO:
  89         #  Order of entries with the same count is inconsistent on CI (gh-32449)
  90         expected = expected.sort_index()
  91         result = result.sort_index()
  92     tm.assert_series_equal(result, expected)

  95 def test_value_counts_inferred(index_or_series):
  96     klass = index_or_series
  97     s_values = ["a", "b", "b", "b", "b", "c", "d", "d", "a", "a"]
  98     s = klass(s_values)
  99     expected = Series([4, 3, 2, 1], index=["b", "a", "d", "c"])
 100     tm.assert_series_equal(s.value_counts(), expected)
 101 
 102     if isinstance(s, Index):
 103         exp = Index(np.unique(np.array(s_values, dtype=np.object_)))
 104         tm.assert_index_equal(s.unique(), exp)
 105     else:
 106         exp = np.unique(np.array(s_values, dtype=np.object_))
 107         tm.assert_numpy_array_equal(s.unique(), exp)
 108 
 109     assert s.nunique() == 4
 110     # don't sort, have to sort after the fact as not sorting is
 111     # platform-dep
 112     hist = s.value_counts(sort=False).sort_values()
 113     expected = Series([3, 1, 4, 2], index=list("acbd")).sort_values()
 114     tm.assert_series_equal(hist, expected)
 115 
 116     # sort ascending
 117     hist = s.value_counts(ascending=True)
 118     expected = Series([1, 2, 3, 4], index=list("cdab"))
 119     tm.assert_series_equal(hist, expected)
 120 
 121     # relative histogram.
 122     hist = s.value_counts(normalize=True)
 123     expected = Series([0.4, 0.3, 0.2, 0.1], index=["b", "a", "d", "c"])
 124     tm.assert_series_equal(hist, expected)

 127 def test_value_counts_bins(index_or_series):
 128     klass = index_or_series
 129     s_values = ["a", "b", "b", "b", "b", "c", "d", "d", "a", "a"]
 130     s = klass(s_values)
 131 
 132     # bins
 133     msg = "bins argument only works with numeric data"
 134     with pytest.raises(TypeError, match=msg):
 135         s.value_counts(bins=1)
 136 
 137     s1 = Series([1, 1, 2, 3])
 138     res1 = s1.value_counts(bins=1)
 139     exp1 = Series({Interval(0.997, 3.0): 4})
 140     tm.assert_series_equal(res1, exp1)
 141     res1n = s1.value_counts(bins=1, normalize=True)
 142     exp1n = Series({Interval(0.997, 3.0): 1.0})
 143     tm.assert_series_equal(res1n, exp1n)
 144 
 145     if isinstance(s1, Index):
 146         tm.assert_index_equal(s1.unique(), Index([1, 2, 3]))
 147     else:
 148         exp = np.array([1, 2, 3], dtype=np.int64)
 149         tm.assert_numpy_array_equal(s1.unique(), exp)
 150 
 151     assert s1.nunique() == 3
 152 
 153     # these return the same
 154     res4 = s1.value_counts(bins=4, dropna=True)
 155     intervals = IntervalIndex.from_breaks([0.997, 1.5, 2.0, 2.5, 3.0])
 156     exp4 = Series([2, 1, 1, 0], index=intervals.take([0, 3, 1, 2]))
 157     tm.assert_series_equal(res4, exp4)
 158 
 159     res4 = s1.value_counts(bins=4, dropna=False)
 160     intervals = IntervalIndex.from_breaks([0.997, 1.5, 2.0, 2.5, 3.0])
 161     exp4 = Series([2, 1, 1, 0], index=intervals.take([0, 3, 1, 2]))
 162     tm.assert_series_equal(res4, exp4)
 163 
 164     res4n = s1.value_counts(bins=4, normalize=True)
 165     exp4n = Series([0.5, 0.25, 0.25, 0], index=intervals.take([0, 3, 1, 2]))
 166     tm.assert_series_equal(res4n, exp4n)
 167 
 168     # handle NA's properly
 169     s_values = ["a", "b", "b", "b", np.nan, np.nan, "d", "d", "a", "a", "b"]
 170     s = klass(s_values)
 171     expected = Series([4, 3, 2], index=["b", "a", "d"])
 172     tm.assert_series_equal(s.value_counts(), expected)
 173 
 174     if isinstance(s, Index):
 175         exp = Index(["a", "b", np.nan, "d"])
 176         tm.assert_index_equal(s.unique(), exp)
 177     else:
 178         exp = np.array(["a", "b", np.nan, "d"], dtype=object)
 179         tm.assert_numpy_array_equal(s.unique(), exp)
 180     assert s.nunique() == 3
 181 
 182     s = klass({}) if klass is dict else klass({}, dtype=object)
 183     expected = Series([], dtype=np.int64)
 184     tm.assert_series_equal(s.value_counts(), expected, check_index_type=False)
 185     # returned dtype differs depending on original
 186     if isinstance(s, Index):
 187         tm.assert_index_equal(s.unique(), Index([]), exact=False)
 188     else:
 189         tm.assert_numpy_array_equal(s.unique(), np.array([]), check_dtype=False)
 190 
 191     assert s.nunique() == 0

 194 def test_value_counts_datetime64(index_or_series):
 195     klass = index_or_series
 196 
 197     # GH 3002, datetime64[ns]
 198     # don't test names though
 199     txt = "\n".join(
 200         [
 201             "xxyyzz20100101PIE",
 202             "xxyyzz20100101GUM",
 203             "xxyyzz20100101EGG",
 204             "xxyyww20090101EGG",
 205             "foofoo20080909PIE",
 206             "foofoo20080909GUM",
 207         ]
 208     )
 209     f = StringIO(txt)
 210     df = pd.read_fwf(
 211         f, widths=[6, 8, 3], names=["person_id", "dt", "food"], parse_dates=["dt"]
 212     )
 213 
 214     s = klass(df["dt"].copy())
 215     s.name = None
 216     idx = pd.to_datetime(
 217         ["2010-01-01 00:00:00", "2008-09-09 00:00:00", "2009-01-01 00:00:00"]
 218     )
 219     expected_s = Series([3, 2, 1], index=idx)
 220     tm.assert_series_equal(s.value_counts(), expected_s)
 221 
 222     expected = np_array_datetime64_compat(
 223         ["2010-01-01 00:00:00", "2009-01-01 00:00:00", "2008-09-09 00:00:00"],
 224         dtype="datetime64[ns]",
 225     )
 226     if isinstance(s, Index):
 227         tm.assert_index_equal(s.unique(), DatetimeIndex(expected))
 228     else:
 229         tm.assert_numpy_array_equal(s.unique(), expected)
 230 
 231     assert s.nunique() == 3
 232 
 233     # with NaT
 234     s = df["dt"].copy()
 235     s = klass(list(s.values) + [pd.NaT])
 236 
 237     result = s.value_counts()
 238     assert result.index.dtype == "datetime64[ns]"
 239     tm.assert_series_equal(result, expected_s)
 240 
 241     result = s.value_counts(dropna=False)
 242     expected_s[pd.NaT] = 1
 243     tm.assert_series_equal(result, expected_s)
 244 
 245     unique = s.unique()
 246     assert unique.dtype == "datetime64[ns]"
 247 
 248     # numpy_array_equal cannot compare pd.NaT
 249     if isinstance(s, Index):
 250         exp_idx = DatetimeIndex(expected.tolist() + [pd.NaT])
 251         tm.assert_index_equal(unique, exp_idx)
 252     else:
 253         tm.assert_numpy_array_equal(unique[:3], expected)
 254         assert pd.isna(unique[3])
 255 
 256     assert s.nunique() == 3
 257     assert s.nunique(dropna=False) == 4
 258 
 259     # timedelta64[ns]
 260     td = df.dt - df.dt + timedelta(1)
 261     td = klass(td, name="dt")
 262 
 263     result = td.value_counts()
 264     expected_s = Series([6], index=[Timedelta("1day")], name="dt")
 265     tm.assert_series_equal(result, expected_s)
 266 
 267     expected = TimedeltaIndex(["1 days"], name="dt")
 268     if isinstance(td, Index):
 269         tm.assert_index_equal(td.unique(), expected)
 270     else:
 271         tm.assert_numpy_array_equal(td.unique(), expected.values)
 272 
 273     td2 = timedelta(1) + (df.dt - df.dt)
 274     td2 = klass(td2, name="dt")
 275     result2 = td2.value_counts()
 276     tm.assert_series_equal(result2, expected_s)
