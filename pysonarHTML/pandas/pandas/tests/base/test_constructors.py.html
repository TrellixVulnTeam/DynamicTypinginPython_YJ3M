  98 class TestConstruction:
  99     # test certain constructor behaviours on dtype inference across Series,
 100     # Index and DataFrame
 101 
 102     @pytest.mark.parametrize(
 103         "klass",
 104         [
 105             Series,
 106             lambda x, **kwargs: DataFrame({"a": x}, **kwargs)["a"],
 107             pytest.param(
 108                 lambda x, **kwargs: DataFrame(x, **kwargs)[0], marks=pytest.mark.xfail
 109             ),
 110             Index,
 111         ],
 112     )
 113     @pytest.mark.parametrize(
 114         "a",
 115         [
 116             np.array(["2263-01-01"], dtype="datetime64[D]"),
 117             np.array([datetime(2263, 1, 1)], dtype=object),
 118             np.array([np.datetime64("2263-01-01", "D")], dtype=object),
 119             np.array(["2263-01-01"], dtype=object),
 120         ],
 121         ids=[
 122             "datetime64[D]",
 123             "object-datetime.datetime",
 124             "object-numpy-scalar",
 125             "object-string",
 126         ],
 127     )
 128     def test_constructor_datetime_outofbound(self, a, klass):
 129         # GH-26853 (+ bug GH-26206 out of bound non-ns unit)
 130 
 131         # No dtype specified (dtype inference)
 132         # datetime64[non-ns] raise error, other cases result in object dtype
 133         # and preserve original data
 134         if a.dtype.kind == "M":
 135             msg = "Out of bounds"
 136             with pytest.raises(pd.errors.OutOfBoundsDatetime, match=msg):
 137                 klass(a)
 138         else:
 139             result = klass(a)
 140             assert result.dtype == "object"
 141             tm.assert_numpy_array_equal(result.to_numpy(), a)
 142 
 143         # Explicit dtype specified
 144         # Forced conversion fails for all -> all cases raise error
 145         msg = "Out of bounds"
 146         with pytest.raises(pd.errors.OutOfBoundsDatetime, match=msg):
 147             klass(a, dtype="datetime64[ns]")
