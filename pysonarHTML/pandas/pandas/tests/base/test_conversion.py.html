 131     def test_iter_box(self):
 132         vals = [Timestamp("2011-01-01"), Timestamp("2011-01-02")]
 133         s = Series(vals)
 134         assert s.dtype == "datetime64[ns]"
 135         for res, exp in zip(s, vals):
 136             assert isinstance(res, Timestamp)
 137             assert res.tz is None
 138             assert res == exp
 139 
 140         vals = [
 141             Timestamp("2011-01-01", tz="US/Eastern"),
 142             Timestamp("2011-01-02", tz="US/Eastern"),
 143         ]
 144         s = Series(vals)
 145 
 146         assert s.dtype == "datetime64[ns, US/Eastern]"
 147         for res, exp in zip(s, vals):
 148             assert isinstance(res, Timestamp)
 149             assert res.tz == exp.tz
 150             assert res == exp
 151 
 152         # timedelta
 153         vals = [Timedelta("1 days"), Timedelta("2 days")]
 154         s = Series(vals)
 155         assert s.dtype == "timedelta64[ns]"
 156         for res, exp in zip(s, vals):
 157             assert isinstance(res, Timedelta)
 158             assert res == exp
 159 
 160         # period
 161         vals = [pd.Period("2011-01-01", freq="M"), pd.Period("2011-01-02", freq="M")]
 162         s = Series(vals)
 163         assert s.dtype == "Period[M]"
 164         for res, exp in zip(s, vals):
 165             assert isinstance(res, pd.Period)
 166             assert res.freq == "M"
