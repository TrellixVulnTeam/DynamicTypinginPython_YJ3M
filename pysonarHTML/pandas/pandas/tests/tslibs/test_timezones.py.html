  15         pytest.skip("UTC: special case in dateutil")
  16 
  17     tz_p = timezones.maybe_get_tz(tz_name)

  96 @pytest.mark.parametrize("ordered", [True, False])
  97 def test_infer_tz_mismatch(infer_setup, ordered):
  98     eastern, _, _, _, start_naive, end_naive = infer_setup
  99     msg = "Inputs must both have the same timezone"
 100 
 101     utc = pytz.utc
 102     start = utc.localize(start_naive)
 103     end = conversion.localize_pydatetime(end_naive, eastern)
 104 
 105     args = (start, end) if ordered else (end, start)
 106 
 107     with pytest.raises(AssertionError, match=msg):
 108         timezones.infer_tzinfo(*args)
