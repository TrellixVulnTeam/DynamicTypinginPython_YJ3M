  18 def test_parse_time_string():
  19     (parsed, reso) = parse_time_string("4Q1984")
  20     (parsed_lower, reso_lower) = parse_time_string("4q1984")
  21 
  22     assert reso == reso_lower
  23     assert parsed == parsed_lower
