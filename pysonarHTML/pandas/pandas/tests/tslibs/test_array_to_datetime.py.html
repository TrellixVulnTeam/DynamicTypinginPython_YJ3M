 119     arr = np.array([invalid_date], dtype="object")

 131         tm.assert_numpy_array_equal(result, expected)

 144 @pytest.mark.parametrize("errors", ["ignore", "coerce"])
 145 def test_coerce_of_invalid_datetimes(errors):
 146     arr = np.array(["01-01-2013", "not_a_date", "1"], dtype=object)
 147     kwargs = dict(values=arr, errors=errors)
 148 
 149     if errors == "ignore":
 150         # Without coercing, the presence of any invalid
 151         # dates prevents any values from being converted.
 152         result, _ = tslib.array_to_datetime(**kwargs)
 153         tm.assert_numpy_array_equal(result, arr)
 154     else:  # coerce.
 155         # With coercing, the invalid dates becomes iNaT
 156         result, _ = tslib.array_to_datetime(arr, errors="coerce")
 157         expected = ["2013-01-01T00:00:00.000000000-0000", iNaT, iNaT]
 158 
 159         tm.assert_numpy_array_equal(
 160             result, np_array_datetime64_compat(expected, dtype="M8[ns]")
 161         )
