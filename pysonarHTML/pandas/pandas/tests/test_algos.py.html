 144     def test_factorize_nan(self):
 145         # nan should map to na_sentinel, not reverse_indexer[na_sentinel]
 146         # rizer.factorize should not raise an exception if na_sentinel indexes
 147         # outside of reverse_indexer
 148         key = np.array([1, 2, 1, np.nan], dtype="O")
 149         rizer = ht.Factorizer(len(key))
 150         for na_sentinel in (-1, 20):
 151             ids = rizer.factorize(key, sort=True, na_sentinel=na_sentinel)
 152             expected = np.array([0, 1, 0, na_sentinel], dtype="int32")
 153             assert len(set(key)) == len(set(expected))
 154             tm.assert_numpy_array_equal(pd.isna(key), expected == na_sentinel)
 155 
 156         # nan still maps to na_sentinel when sort=False
 157         key = np.array([0, np.nan, 1], dtype="O")
 158         na_sentinel = -1
 159 
 160         # TODO(wesm): unused?
 161         ids = rizer.factorize(key, sort=False, na_sentinel=na_sentinel)  # noqa
 162 
 163         expected = np.array([2, -1, 0], dtype="int32")
 164         assert len(set(key)) == len(set(expected))
