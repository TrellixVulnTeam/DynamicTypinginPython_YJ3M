 101     def elementwise_comparison(self, op, array, other):
 102         """
 103         Helper that performs elementwise comparisons between `array` and `other`
 104         """
 105         other = other if is_list_like(other) else [other] * len(array)
 106         return np.array([op(x, y) for x, y in zip(array, other)])
