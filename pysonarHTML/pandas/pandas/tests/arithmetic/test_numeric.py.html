  19 def adjust_negative_zero(zero, expected):
  20     """
  21     Helper to adjust the expected result if we are dividing by -0.0
  22     as opposed to 0.0
  23     """
  24     if np.signbit(np.array(zero)).any():
  25         # All entries in the `zero` fixture should be either
  26         #  all-negative or no-negative.
  27         assert np.signbit(np.array(zero)).all()
  28 
  29         expected *= -1
  30 
  31     return expected
  32
