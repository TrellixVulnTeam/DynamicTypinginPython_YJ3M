  14 class TestDataFrameGroupByPlots(TestPlotBase):
  15     def test_series_groupby_plotting_nominally_works(self):
  16         n = 10
  17         weight = Series(np.random.normal(166, 20, size=n))
  18         height = Series(np.random.normal(60, 10, size=n))
  19         with tm.RNGContext(42):
  20             gender = np.random.choice(["male", "female"], size=n)
  21 
  22         weight.groupby(gender).plot()
  23         tm.close()
  24         height.groupby(gender).hist()
  25         tm.close()
  26         # Regression test for GH8733
  27         height.groupby(gender).plot(alpha=0.5)
  28         tm.close()

  30     def test_plotting_with_float_index_works(self):
