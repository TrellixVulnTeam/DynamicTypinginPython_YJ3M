 116     def test_boxplot_axis_limits(self):
 117         def _check_ax_limits(col, ax):
 118             y_min, y_max = ax.get_ylim()
 119             assert y_min <= col.min()
 120             assert y_max >= col.max()
 121 
 122         df = self.hist_df.copy()
 123         df["age"] = np.random.randint(1, 20, df.shape[0])
 124         # One full row
 125         height_ax, weight_ax = df.boxplot(["height", "weight"], by="category")
 126         _check_ax_limits(df["height"], height_ax)
 127         _check_ax_limits(df["weight"], weight_ax)
 128         assert weight_ax._sharey == height_ax
 129 
 130         # Two rows, one partial
 131         p = df.boxplot(["height", "weight", "age"], by="category")
 132         height_ax, weight_ax, age_ax = p[0, 0], p[0, 1], p[1, 0]
 133         dummy_ax = p[1, 1]
 134 
 135         _check_ax_limits(df["height"], height_ax)
 136         _check_ax_limits(df["weight"], weight_ax)
 137         _check_ax_limits(df["age"], age_ax)
 138         assert weight_ax._sharey == height_ax
 139         assert age_ax._sharey == height_ax
 140         assert dummy_ax._sharey is None
