 111     def test_registry_resets(self):
 112         units = pytest.importorskip("matplotlib.units")
 113         dates = pytest.importorskip("matplotlib.dates")
 114 
 115         # make a copy, to reset to
 116         original = dict(units.registry)
 117 
 118         try:
 119             # get to a known state
 120             units.registry.clear()
 121             date_converter = dates.DateConverter()
 122             units.registry[datetime] = date_converter
 123             units.registry[date] = date_converter
 124 
 125             register_matplotlib_converters()
 126             assert units.registry[date] is not date_converter
 127             deregister_matplotlib_converters()
 128             assert units.registry[date] is date_converter
 129 
 130         finally:
 131             # restore original stater
 132             units.registry.clear()
 133             for k, v in original.items():
 134                 units.registry[k] = v

 218     def test_conversion_outofbounds_datetime(self):
 219         # 2579
 220         values = [date(1677, 1, 1), date(1677, 1, 2)]
 221         rs = self.dtc.convert(values, None, None)
 222         xp = converter.dates.date2num(values)
 223         tm.assert_numpy_array_equal(rs, xp)
 224         rs = self.dtc.convert(values[0], None, None)

 226         assert rs == xp

 228         values = [datetime(1677, 1, 1, 12), datetime(1677, 1, 2, 12)]

 260     def test_resolution(self):
 261         def _assert_less(ts1, ts2):
 262             val1 = self.dtc.convert(ts1, None, None)
 263             val2 = self.dtc.convert(ts2, None, None)
 264             if not val1 < val2:
 265                 raise AssertionError(f"{val1} is not less than {val2}.")
 266 
 267         # Matplotlib's time representation using floats cannot distinguish
 268         # intervals smaller than ~10 microsecond in the common range of years.
 269         ts = Timestamp("2012-1-1")
 270         _assert_less(ts, ts + Second())
 271         _assert_less(ts, ts + Milli())
 272         _assert_less(ts, ts + Micro(50))
