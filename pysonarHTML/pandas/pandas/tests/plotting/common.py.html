  80     def _check_legend_labels(self, axes, labels=None, visible=True):
  81         """
  82         Check each axes has expected legend labels
  83 
  84         Parameters
  85         ----------
  86         axes : matplotlib Axes object, or its list-like
  87         labels : list-like
  88             expected legend labels
  89         visible : bool
  90             expected legend visibility. labels are checked only when visible is
  91             True
  92         """
  93         if visible and (labels is None):
  94             raise ValueError("labels must be specified when visible is True")
  95         axes = self._flatten_visible(axes)
  96         for ax in axes:
  97             if visible:
  98                 assert ax.get_legend() is not None
  99                 self._check_text_labels(ax.get_legend().get_texts(), labels)
 100             else:
 101                 assert ax.get_legend() is None

 165     def _get_colors_mapped(self, series, colors):
 166         unique = series.unique()
 167         # unique and colors length can be differed
 168         # depending on slice value
 169         mapped = dict(zip(unique, colors))
 170         return [mapped[v] for v in series.values]

 172     def _check_colors(
 173         self, collections, linecolors=None, facecolors=None, mapping=None
 174     ):
 175         """
 176         Check each artist has expected line colors and face colors
 177 
 178         Parameters
 179         ----------
 180         collections : list-like
 181             list or collection of target artist
 182         linecolors : list-like which has the same length as collections
 183             list of expected line colors
 184         facecolors : list-like which has the same length as collections
 185             list of expected face colors
 186         mapping : Series
 187             Series used for color grouping key
 188             used for andrew_curves, parallel_coordinates, radviz test
 189         """
 190         from matplotlib.lines import Line2D
 191         from matplotlib.collections import Collection, PolyCollection, LineCollection
 192 
 193         conv = self.colorconverter
 194         if linecolors is not None:
 195 
 196             if mapping is not None:
 197                 linecolors = self._get_colors_mapped(mapping, linecolors)
 198                 linecolors = linecolors[: len(collections)]
 199 
 200             assert len(collections) == len(linecolors)
 201             for patch, color in zip(collections, linecolors):
 202                 if isinstance(patch, Line2D):
 203                     result = patch.get_color()
 204                     # Line2D may contains string color expression
 205                     result = conv.to_rgba(result)
 206                 elif isinstance(patch, (PolyCollection, LineCollection)):
 207                     result = tuple(patch.get_edgecolor()[0])
 208                 else:
 209                     result = patch.get_edgecolor()
 210 
 211                 expected = conv.to_rgba(color)
 212                 assert result == expected
 213 
 214         if facecolors is not None:
 215 
 216             if mapping is not None:
 217                 facecolors = self._get_colors_mapped(mapping, facecolors)
 218                 facecolors = facecolors[: len(collections)]
 219 
 220             assert len(collections) == len(facecolors)
 221             for patch, color in zip(collections, facecolors):
 222                 if isinstance(patch, Collection):
 223                     # returned as list of np.array
 224                     result = patch.get_facecolor()[0]
 225                 else:
 226                     result = patch.get_facecolor()
 227 
 228                 if isinstance(result, np.ndarray):
 229                     result = tuple(result)
 230 
 231                 expected = conv.to_rgba(color)
 232                 assert result == expected

 253     def _check_ticks_props(
 254         self, axes, xlabelsize=None, xrot=None, ylabelsize=None, yrot=None
 255     ):
 256         """
 257         Check each axes has expected tick properties
 258 
 259         Parameters
 260         ----------
 261         axes : matplotlib Axes object, or its list-like
 262         xlabelsize : number
 263             expected xticks font size
 264         xrot : number
 265             expected xticks rotation
 266         ylabelsize : number
 267             expected yticks font size
 268         yrot : number
 269             expected yticks rotation
 270         """
 271         from matplotlib.ticker import NullFormatter
 272 
 273         axes = self._flatten_visible(axes)
 274         for ax in axes:
 275             if xlabelsize or xrot:
 276                 if isinstance(ax.xaxis.get_minor_formatter(), NullFormatter):
 277                     # If minor ticks has NullFormatter, rot / fontsize are not
 278                     # retained
 279                     labels = ax.get_xticklabels()
 280                 else:
 281                     labels = ax.get_xticklabels() + ax.get_xticklabels(minor=True)
 282 
 283                 for label in labels:
 284                     if xlabelsize is not None:
 285                         tm.assert_almost_equal(label.get_fontsize(), xlabelsize)
 286                     if xrot is not None:
 287                         tm.assert_almost_equal(label.get_rotation(), xrot)
 288 
 289             if ylabelsize or yrot:
 290                 if isinstance(ax.yaxis.get_minor_formatter(), NullFormatter):
 291                     labels = ax.get_yticklabels()
 292                 else:
 293                     labels = ax.get_yticklabels() + ax.get_yticklabels(minor=True)
 294 
 295                 for label in labels:
 296                     if ylabelsize is not None:
 297                         tm.assert_almost_equal(label.get_fontsize(), ylabelsize)
 298                     if yrot is not None:
 299                         tm.assert_almost_equal(label.get_rotation(), yrot)

 318     def _check_axes_shape(self, axes, axes_num=None, layout=None, figsize=None):
 319         """
 320         Check expected number of axes is drawn in expected layout
 321 
 322         Parameters
 323         ----------
 324         axes : matplotlib Axes object, or its list-like
 325         axes_num : number
 326             expected number of axes. Unnecessary axes should be set to
 327             invisible.
 328         layout : tuple
 329             expected layout, (expected number of rows , columns)
 330         figsize : tuple
 331             expected figsize. default is matplotlib default
 332         """
 333         from pandas.plotting._matplotlib.tools import _flatten
 334 
 335         if figsize is None:
 336             figsize = self.default_figsize
 337         visible_axes = self._flatten_visible(axes)
 338 
 339         if axes_num is not None:
 340             assert len(visible_axes) == axes_num
 341             for ax in visible_axes:
 342                 # check something drawn on visible axes
 343                 assert len(ax.get_children()) > 0
 344 
 345         if layout is not None:
 346             result = self._get_axes_layout(_flatten(axes))
 347             assert result == layout
 348 
 349         tm.assert_numpy_array_equal(
 350             visible_axes[0].figure.get_size_inches(),
 351             np.array(figsize, dtype=np.float64),
 352         )

 354     def _get_axes_layout(self, axes):
 355         x_set = set()
 356         y_set = set()
 357         for ax in axes:
 358             # check axes coordinates to estimate layout
 359             points = ax.get_position().get_points()
 360             x_set.add(points[0][0])
 361             y_set.add(points[0][1])
 362         return (len(y_set), len(x_set))

 365         """
 366         Flatten axes, and filter only visible
 367 
 368         Parameters
 369         ----------

 371 
 372         """
 373         from pandas.plotting._matplotlib.tools import _flatten
 374 
 375         axes = _flatten(axes)

 379     def _check_has_errorbars(self, axes, xerr=0, yerr=0):
 380         """
 381         Check axes has expected number of errorbars
 382 
 383         Parameters
 384         ----------
 385         axes : matplotlib Axes object, or its list-like
 386         xerr : number
 387             expected number of x errorbar
 388         yerr : number
 389             expected number of y errorbar
 390         """
 391         axes = self._flatten_visible(axes)
 392         for ax in axes:
 393             containers = ax.containers
 394             xerr_count = 0
 395             yerr_count = 0
 396             for c in containers:
 397                 has_xerr = getattr(c, "has_xerr", False)
 398                 has_yerr = getattr(c, "has_yerr", False)
 399                 if has_xerr:
 400                     xerr_count += 1
 401                 if has_yerr:
 402                     yerr_count += 1
 403             assert xerr == xerr_count
 404             assert yerr == yerr_count

 406     def _check_box_return_type(
 407         self, returned, return_type, expected_keys=None, check_ax_title=True
 408     ):
 409         """
 410         Check box returned type is correct
 411 
 412         Parameters
 413         ----------
 414         returned : object to be tested, returned from boxplot
 415         return_type : str
 416             return_type passed to boxplot
 417         expected_keys : list-like, optional
 418             group labels in subplot case. If not passed,
 419             the function checks assuming boxplot uses single ax
 420         check_ax_title : bool
 421             Whether to check the ax.title is the same as expected_key
 422             Intended to be checked by calling from ``boxplot``.
 423             Normal ``plot`` doesn't attach ``ax.title``, it must be disabled.
 424         """
 425         from matplotlib.axes import Axes
 426 
 427         types = {"dict": dict, "axes": Axes, "both": tuple}
 428         if expected_keys is None:
 429             # should be fixed when the returning default is changed
 430             if return_type is None:
 431                 return_type = "dict"
 432 
 433             assert isinstance(returned, types[return_type])
 434             if return_type == "both":
 435                 assert isinstance(returned.ax, Axes)
 436                 assert isinstance(returned.lines, dict)
 437         else:
 438             # should be fixed when the returning default is changed
 439             if return_type is None:
 440                 for r in self._flatten_visible(returned):
 441                     assert isinstance(r, Axes)
 442                 return
 443 
 444             assert isinstance(returned, Series)
 445 
 446             assert sorted(returned.keys()) == sorted(expected_keys)
 447             for key, value in returned.items():
 448                 assert isinstance(value, types[return_type])
 449                 # check returned dict has correct mapping
 450                 if return_type == "axes":
 451                     if check_ax_title:
 452                         assert value.get_title() == key
 453                 elif return_type == "both":
 454                     if check_ax_title:
 455                         assert value.ax.get_title() == key
 456                     assert isinstance(value.ax, Axes)
 457                     assert isinstance(value.lines, dict)
 458                 elif return_type == "dict":
 459                     line = value["medians"][0]
 460                     axes = line.axes
 461                     if check_ax_title:
 462                         assert axes.get_title() == key
 463                 else:
 464                     raise AssertionError

 466     def _check_grid_settings(self, obj, kinds, kws={}):
 467         # Make sure plot defaults to rcParams['axes.grid'] setting, GH 9792
 468 
 469         import matplotlib as mpl
 470 
 471         def is_grid_on():
 472             xticks = self.plt.gca().xaxis.get_major_ticks()
 473             yticks = self.plt.gca().yaxis.get_major_ticks()
 474             # for mpl 2.2.2, gridOn and gridline.get_visible disagree.
 475             # for new MPL, they are the same.
 476 
 477             if self.mpl_ge_3_1_0:
 478                 xoff = all(not g.gridline.get_visible() for g in xticks)
 479                 yoff = all(not g.gridline.get_visible() for g in yticks)
 480             else:
 481                 xoff = all(not g.gridOn for g in xticks)
 482                 yoff = all(not g.gridOn for g in yticks)
 483 
 484             return not (xoff and yoff)
 485 
 486         spndx = 1
 487         for kind in kinds:
 488 
 489             self.plt.subplot(1, 4 * len(kinds), spndx)
 490             spndx += 1
 491             mpl.rc("axes", grid=False)
 492             obj.plot(kind=kind, **kws)
 493             assert not is_grid_on()
 494 
 495             self.plt.subplot(1, 4 * len(kinds), spndx)
 496             spndx += 1
 497             mpl.rc("axes", grid=True)
 498             obj.plot(kind=kind, grid=False, **kws)
 499             assert not is_grid_on()
 500 
 501             if kind != "pie":
 502                 self.plt.subplot(1, 4 * len(kinds), spndx)
 503                 spndx += 1
 504                 mpl.rc("axes", grid=True)
 505                 obj.plot(kind=kind, **kws)
 506                 assert is_grid_on()
 507 
 508                 self.plt.subplot(1, 4 * len(kinds), spndx)
 509                 spndx += 1
 510                 mpl.rc("axes", grid=False)
 511                 obj.plot(kind=kind, grid=True, **kws)
 512                 assert is_grid_on()

 521 def _check_plot_works(f, filterwarnings="always", **kwargs):
 522     import matplotlib.pyplot as plt
 523 
 524     ret = None
 525     with warnings.catch_warnings():
 526         warnings.simplefilter(filterwarnings)
 527         try:
 528             try:
 529                 fig = kwargs["figure"]
 530             except KeyError:
 531                 fig = plt.gcf()
 532 
 533             plt.clf()
 534 
 535             kwargs.get("ax", fig.add_subplot(211))
 536             ret = f(**kwargs)
 537 
 538             tm.assert_is_valid_plot_return_object(ret)
 539 
 540             if f is pd.plotting.bootstrap_plot:
 541                 assert "ax" not in kwargs
 542             else:
 543                 kwargs["ax"] = fig.add_subplot(212)
 544 
 545             ret = f(**kwargs)
 546             tm.assert_is_valid_plot_return_object(ret)
 547 
 548             with tm.ensure_clean(return_filelike=True) as path:
 549                 plt.savefig(path)
 550         finally:
 551             tm.close(fig)
 552 
 553         return ret
