 299     @pytest.mark.slow
 300     def test_grouped_hist_legacy(self):
 301         from matplotlib.patches import Rectangle
 302         from pandas.plotting._matplotlib.hist import _grouped_hist
 303 
 304         df = DataFrame(randn(500, 2), columns=["A", "B"])
 305         df["C"] = np.random.randint(0, 4, 500)
 306         df["D"] = ["X"] * 500
 307 
 308         axes = _grouped_hist(df.A, by=df.C)
 309         self._check_axes_shape(axes, axes_num=4, layout=(2, 2))
 310 
 311         tm.close()
 312         axes = df.hist(by=df.C)
 313         self._check_axes_shape(axes, axes_num=4, layout=(2, 2))
 314 
 315         tm.close()
 316         # group by a key with single value
 317         axes = df.hist(by="D", rot=30)
 318         self._check_axes_shape(axes, axes_num=1, layout=(1, 1))
 319         self._check_ticks_props(axes, xrot=30)
 320 
 321         tm.close()
 322         # make sure kwargs to hist are handled
 323         xf, yf = 20, 18
 324         xrot, yrot = 30, 40
 325 
 326         axes = _grouped_hist(
 327             df.A,
 328             by=df.C,
 329             cumulative=True,
 330             bins=4,
 331             xlabelsize=xf,
 332             xrot=xrot,
 333             ylabelsize=yf,
 334             yrot=yrot,
 335             density=True,
 336         )
 337         # height of last bin (index 5) must be 1.0
 338         for ax in axes.ravel():
 339             rects = [x for x in ax.get_children() if isinstance(x, Rectangle)]
 340             height = rects[-1].get_height()
 341             tm.assert_almost_equal(height, 1.0)
 342         self._check_ticks_props(
 343             axes, xlabelsize=xf, xrot=xrot, ylabelsize=yf, yrot=yrot
 344         )
 345 
 346         tm.close()
 347         axes = _grouped_hist(df.A, by=df.C, log=True)
 348         # scale of y must be 'log'
 349         self._check_ax_scales(axes, yaxis="log")
 350 
 351         tm.close()
 352         # propagate attr exception from matplotlib.Axes.hist
 353         with pytest.raises(AttributeError):
 354             _grouped_hist(df.A, by=df.C, foo="bar")
 355 
 356         msg = "Specify figure size by tuple instead"
 357         with pytest.raises(ValueError, match=msg):
 358             df.hist(by="C", figsize="default")
