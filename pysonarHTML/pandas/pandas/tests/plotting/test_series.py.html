 299     def test_pie_series(self):
 300         # if sum of values is less than 1.0, pie handle them as rate and draw
 301         # semicircle.
 302         series = Series(
 303             np.random.randint(1, 5), index=["a", "b", "c", "d", "e"], name="YLABEL"
 304         )
 305         ax = _check_plot_works(series.plot.pie)
 306         self._check_text_labels(ax.texts, series.index)
 307         assert ax.get_ylabel() == "YLABEL"
 308 
 309         # without wedge labels
 310         ax = _check_plot_works(series.plot.pie, labels=None)
 311         self._check_text_labels(ax.texts, [""] * 5)
 312 
 313         # with less colors than elements
 314         color_args = ["r", "g", "b"]
 315         ax = _check_plot_works(series.plot.pie, colors=color_args)
 316 
 317         color_expected = ["r", "g", "b", "r", "g"]
 318         self._check_colors(ax.patches, facecolors=color_expected)
 319 
 320         # with labels and colors
 321         labels = ["A", "B", "C", "D", "E"]
 322         color_args = ["r", "g", "b", "c", "m"]
 323         ax = _check_plot_works(series.plot.pie, labels=labels, colors=color_args)
 324         self._check_text_labels(ax.texts, labels)
 325         self._check_colors(ax.patches, facecolors=color_args)
 326 
 327         # with autopct and fontsize
 328         ax = _check_plot_works(
 329             series.plot.pie, colors=color_args, autopct="%.2f", fontsize=7
 330         )
 331         pcts = [f"{s*100:.2f}" for s in series.values / float(series.sum())]
 332         expected_texts = list(chain.from_iterable(zip(series.index, pcts)))
 333         self._check_text_labels(ax.texts, expected_texts)
 334         for t in ax.texts:
 335             assert t.get_fontsize() == 7
 336 
 337         # includes negative value
 338         with pytest.raises(ValueError):
 339             series = Series([1, 2, 0, 4, -1], index=["a", "b", "c", "d", "e"])
 340             series.plot.pie()
 341 
 342         # includes nan
 343         series = Series([1, 2, np.nan, 4], index=["a", "b", "c", "d"], name="YLABEL")
 344         ax = _check_plot_works(series.plot.pie)
 345         self._check_text_labels(ax.texts, ["a", "b", "", "d"])
