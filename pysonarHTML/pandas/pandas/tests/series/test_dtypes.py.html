 263     def test_astype_categorical_to_other(self):
 264 
 265         value = np.random.RandomState(0).randint(0, 10000, 100)
 266         df = DataFrame({"value": value})
 267         labels = [f"{i} - {i + 499}" for i in range(0, 10000, 500)]
 268         cat_labels = Categorical(labels, labels)
 269 
 270         df = df.sort_values(by=["value"], ascending=True)
 271         df["value_group"] = pd.cut(
 272             df.value, range(0, 10500, 500), right=False, labels=cat_labels
 273         )
 274 
 275         s = df["value_group"]
 276         expected = s
 277         tm.assert_series_equal(s.astype("category"), expected)
 278         tm.assert_series_equal(s.astype(CategoricalDtype()), expected)
 279         msg = r"could not convert string to float|invalid literal for float\(\)"
 280         with pytest.raises(ValueError, match=msg):
 281             s.astype("float64")
 282 
 283         cat = Series(Categorical(["a", "b", "b", "a", "a", "c", "c", "c"]))
 284         exp = Series(["a", "b", "b", "a", "a", "c", "c", "c"])
 285         tm.assert_series_equal(cat.astype("str"), exp)
 286         s2 = Series(Categorical(["1", "2", "3", "4"]))
 287         exp2 = Series([1, 2, 3, 4]).astype(int)
 288         tm.assert_series_equal(s2.astype("int"), exp2)
 289 
 290         # object don't sort correctly, so just compare that we have the same
 291         # values
 292         def cmp(a, b):
 293             tm.assert_almost_equal(np.sort(np.unique(a)), np.sort(np.unique(b)))
 294 
 295         expected = Series(np.array(s.values), name="value_group")
 296         cmp(s.astype("object"), expected)
 297         cmp(s.astype(np.object_), expected)
 298 
 299         # array conversion
 300         tm.assert_almost_equal(np.array(s), np.array(s.values))
 301 
 302         tm.assert_series_equal(s.astype("category"), s)
 303         tm.assert_series_equal(s.astype(CategoricalDtype()), s)
 304 
 305         roundtrip_expected = s.cat.set_categories(
 306             s.cat.categories.sort_values()
 307         ).cat.remove_unused_categories()
 308         tm.assert_series_equal(
 309             s.astype("object").astype("category"), roundtrip_expected
 310         )
 311         tm.assert_series_equal(
 312             s.astype("object").astype(CategoricalDtype()), roundtrip_expected
 313         )
 314 
 315         # invalid conversion (these are NOT a dtype)
 316         msg = (
 317             "dtype '<class 'pandas.core.arrays.categorical.Categorical'>' "
 318             "not understood"
 319         )
 320 
 321         for invalid in [
 322             lambda x: x.astype(Categorical),
 323             lambda x: x.astype("object").astype(Categorical),
 324         ]:
 325             with pytest.raises(TypeError, match=msg):
 326                 invalid(s)
