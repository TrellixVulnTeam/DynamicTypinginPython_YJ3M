 188     def test_convert_no_arg_error(self):
 189         s = Series(["1.0", "2"])
 190         msg = r"At least one of datetime, numeric or timedelta must be True\."
 191         with pytest.raises(ValueError, match=msg):

 194     def test_convert_preserve_bool(self):
 195         s = Series([1, True, 3, 5], dtype=object)

 206     def test_constructor_no_pandas_array(self):

 212     def test_astype_no_pandas_dtype(self):
 213         # https://github.com/pandas-dev/pandas/pull/24866
 214         ser = pd.Series([1, 2], dtype="int64")
 215         # Don't have PandasDtype in the public API, so we use `.array.dtype`,
 216         # which is a PandasDtype.
 217         result = ser.astype(ser.array.dtype)
 218         tm.assert_series_equal(result, ser)

 235 def test_hasnans_uncached_for_series():
 236     # GH#19700
 237     idx = pd.Index([0, 1])
 238     assert idx.hasnans is False
 239     assert "hasnans" in idx._cache
 240     ser = idx.to_series()
 241     assert ser.hasnans is False
 242     assert not hasattr(ser, "_cache")
 243     ser.iloc[-1] = np.nan
 244     assert ser.hasnans is True
 245     assert Series.hasnans.__doc__ == pd.Index.hasnans.__doc__
