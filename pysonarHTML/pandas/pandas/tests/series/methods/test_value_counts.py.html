 144     def test_value_counts_categorical_with_nan(self):
 145         # see GH#9443
 146 
 147         # sanity check
 148         ser = Series(["a", "b", "a"], dtype="category")
 149         exp = Series([2, 1], index=CategoricalIndex(["a", "b"]))
 150 
 151         res = ser.value_counts(dropna=True)
 152         tm.assert_series_equal(res, exp)
 153 
 154         res = ser.value_counts(dropna=True)
 155         tm.assert_series_equal(res, exp)
 156 
 157         # same Series via two different constructions --> same behaviour
 158         series = [
 159             Series(["a", "b", None, "a", None, None], dtype="category"),
 160             Series(
 161                 Categorical(["a", "b", None, "a", None, None], categories=["a", "b"])
 162             ),
 163         ]
 164 
 165         for ser in series:
 166             # None is a NaN value, so we exclude its count here
 167             exp = Series([2, 1], index=CategoricalIndex(["a", "b"]))
 168             res = ser.value_counts(dropna=True)
 169             tm.assert_series_equal(res, exp)
 170 
 171             # we don't exclude the count of None and sort by counts
 172             exp = Series([3, 2, 1], index=CategoricalIndex([np.nan, "a", "b"]))
 173             res = ser.value_counts(dropna=False)
 174             tm.assert_series_equal(res, exp)
 175 
 176             # When we aren't sorting by counts, and np.nan isn't a
 177             # category, it should be last.
 178             exp = Series([2, 1, 3], index=CategoricalIndex(["a", "b", np.nan]))
 179             res = ser.value_counts(dropna=False, sort=False)
 180             tm.assert_series_equal(res, exp)
