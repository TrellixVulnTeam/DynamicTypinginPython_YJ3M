  20 def test_align(datetime_series, first_slice, second_slice, join_type, fill):
  21     a = datetime_series[slice(*first_slice)]
  22     b = datetime_series[slice(*second_slice)]
  23 
  24     aa, ab = a.align(b, join=join_type, fill_value=fill)
  25 
  26     join_index = a.index.join(b.index, how=join_type)
  27     if fill is not None:
  28         diff_a = aa.index.difference(join_index)
  29         diff_b = ab.index.difference(join_index)
  30         if len(diff_a) > 0:
  31             assert (aa.reindex(diff_a) == fill).all()
  32         if len(diff_b) > 0:
  33             assert (ab.reindex(diff_b) == fill).all()
  34 
  35     ea = a.reindex(join_index)
  36     eb = b.reindex(join_index)
  37 
  38     if fill is not None:
  39         ea = ea.fillna(fill)
  40         eb = eb.fillna(fill)
  41 
  42     tm.assert_series_equal(aa, ea)
  43     tm.assert_series_equal(ab, eb)
  44     assert aa.name == "ts"
  45     assert ea.name == "ts"
  46     assert ab.name == "ts"
  47     assert eb.name == "ts"

  80 def test_align_nocopy(datetime_series):
  81     b = datetime_series[:5].copy()
  82 
  83     # do copy
  84     a = datetime_series.copy()
  85     ra, _ = a.align(b, join="left")
  86     ra[:5] = 5
  87     assert not (a[:5] == 5).any()
  88 
  89     # do not copy
  90     a = datetime_series.copy()
  91     ra, _ = a.align(b, join="left", copy=False)
  92     ra[:5] = 5
  93     assert (a[:5] == 5).all()
  94 
  95     # do copy
  96     a = datetime_series.copy()
  97     b = datetime_series[:5].copy()
  98     _, rb = a.align(b, join="right")
  99     rb[:3] = 5

 104     b = datetime_series[:5].copy()

 120 def test_align_multiindex():
 121     # GH 10665
 122 
 123     midx = pd.MultiIndex.from_product(
 124         [range(2), range(3), range(2)], names=("a", "b", "c")
 125     )
 126     idx = pd.Index(range(2), name="b")
 127     s1 = pd.Series(np.arange(12, dtype="int64"), index=midx)
 128     s2 = pd.Series(np.arange(2, dtype="int64"), index=idx)
 129 
 130     # these must be the same results (but flipped)
 131     res1l, res1r = s1.align(s2, join="left")
 132     res2l, res2r = s2.align(s1, join="right")
 133 
 134     expl = s1
 135     tm.assert_series_equal(expl, res1l)
 136     tm.assert_series_equal(expl, res2r)
 137     expr = pd.Series([0, 0, 1, 1, np.nan, np.nan] * 2, index=midx)
 138     tm.assert_series_equal(expr, res1r)
 139     tm.assert_series_equal(expr, res2l)
 140 
 141     res1l, res1r = s1.align(s2, join="right")
 142     res2l, res2r = s2.align(s1, join="left")
 143 
 144     exp_idx = pd.MultiIndex.from_product(
 145         [range(2), range(2), range(2)], names=("a", "b", "c")
 146     )
 147     expl = pd.Series([0, 1, 2, 3, 6, 7, 8, 9], index=exp_idx)
 148     tm.assert_series_equal(expl, res1l)
 149     tm.assert_series_equal(expl, res2r)
 150     expr = pd.Series([0, 0, 1, 1] * 2, index=exp_idx)
 151     tm.assert_series_equal(expr, res1r)
 152     tm.assert_series_equal(expr, res2l)
