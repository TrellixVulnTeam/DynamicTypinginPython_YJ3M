 269     def test_rank_inf(self, contents, dtype):
 270         dtype_na_map = {
 271             "float64": np.nan,
 272             "float32": np.nan,
 273             "int64": iNaT,
 274             "object": None,
 275         }
 276         # Insert nans at random positions if underlying dtype has missing
 277         # value. Then adjust the expected order by adding nans accordingly
 278         # This is for testing whether rank calculation is affected
 279         # when values are interwined with nan values.
 280         values = np.array(contents, dtype=dtype)
 281         exp_order = np.array(range(len(values)), dtype="float64") + 1.0
 282         if dtype in dtype_na_map:
 283             na_value = dtype_na_map[dtype]
 284             nan_indices = np.random.choice(range(len(values)), 5)
 285             values = np.insert(values, nan_indices, na_value)
 286             exp_order = np.insert(exp_order, nan_indices, np.nan)
 287         # shuffle the testing array and expected results in the same way
 288         random_order = np.random.permutation(len(values))
 289         iseries = Series(values[random_order])
 290         exp = Series(exp_order[random_order], dtype="float64")
 291         iranks = iseries.rank()
 292         tm.assert_series_equal(iranks, exp)

 294     def test_rank_tie_methods(self):
 295         s = self.s
 296 
 297         def _check(s, expected, method="average"):
 298             result = s.rank(method=method)
 299             tm.assert_series_equal(result, Series(expected))
 300 
 301         dtypes = [None, object]
 302         disabled = {(object, "first")}
 303         results = self.results
 304 
 305         for method, dtype in product(results, dtypes):

 309             _check(series, results[method], method=method)

 314     @pytest.mark.parametrize("na_option", ["top", "bottom", "keep"])
 315     def test_rank_tie_methods_on_infs_nans(self, method, na_option, ascending):
 316         dtypes = [
 317             ("object", None, Infinity(), NegInfinity()),
 318             ("float64", np.nan, np.inf, -np.inf),
 319         ]
 320         chunk = 3
 321         disabled = {("object", "first")}
 322 
 323         def _check(s, method, na_option, ascending):
 324             exp_ranks = {
 325                 "average": ([2, 2, 2], [5, 5, 5], [8, 8, 8]),
 326                 "min": ([1, 1, 1], [4, 4, 4], [7, 7, 7]),
 327                 "max": ([3, 3, 3], [6, 6, 6], [9, 9, 9]),
 328                 "first": ([1, 2, 3], [4, 5, 6], [7, 8, 9]),
 329                 "dense": ([1, 1, 1], [2, 2, 2], [3, 3, 3]),
 330             }
 331             ranks = exp_ranks[method]
 332             if na_option == "top":
 333                 order = [ranks[1], ranks[0], ranks[2]]
 334             elif na_option == "bottom":
 335                 order = [ranks[0], ranks[2], ranks[1]]
 336             else:
 337                 order = [ranks[0], [np.nan] * chunk, ranks[1]]
 338             expected = order if ascending else order[::-1]
 339             expected = list(chain.from_iterable(expected))
 340             result = s.rank(method=method, na_option=na_option, ascending=ascending)
 341             tm.assert_series_equal(result, Series(expected, dtype="float64"))
 342 
 343         for dtype, na_value, pos_inf, neg_inf in dtypes:
 344             in_arr = [neg_inf] * chunk + [na_value] * chunk + [pos_inf] * chunk
 345             iseries = Series(in_arr, dtype=dtype)
 346             if (dtype, method) in disabled:
 347                 continue
 348             _check(iseries, method, na_option, ascending)
