  45     def test_reset_index_level(self):
  46         df = DataFrame([[1, 2, 3], [4, 5, 6]], columns=["A", "B", "C"])
  47 
  48         for levels in ["A", "B"], [0, 1]:
  49             # With MultiIndex
  50             s = df.set_index(["A", "B"])["C"]
  51 
  52             result = s.reset_index(level=levels[0])
  53             tm.assert_frame_equal(result, df.set_index("B"))
  54 
  55             result = s.reset_index(level=levels[:1])
  56             tm.assert_frame_equal(result, df.set_index("B"))
  57 
  58             result = s.reset_index(level=levels)
  59             tm.assert_frame_equal(result, df)
  60 
  61             result = df.set_index(["A", "B"]).reset_index(level=levels, drop=True)
  62             tm.assert_frame_equal(result, df[["C"]])
  63 
  64             with pytest.raises(KeyError, match="Level E "):
  65                 s.reset_index(level=["A", "E"])
  66 
  67             # With single-level Index
  68             s = df.set_index("A")["B"]
  69 
  70             result = s.reset_index(level=levels[0])
  71             tm.assert_frame_equal(result, df[["A", "B"]])
  72 
  73             result = s.reset_index(level=levels[:1])
  74             tm.assert_frame_equal(result, df[["A", "B"]])
  75 
  76             result = s.reset_index(level=levels[0], drop=True)
  77             tm.assert_series_equal(result, df["B"])
  78 
  79             with pytest.raises(IndexError, match="Too many levels"):
  80                 s.reset_index(level=[0, 1, 2])
  81 
  82         # Check that .reset_index([],drop=True) doesn't fail
  83         result = Series(range(4)).reset_index([], drop=True)
  84         expected = Series(range(4))
  85         tm.assert_series_equal(result, expected)
