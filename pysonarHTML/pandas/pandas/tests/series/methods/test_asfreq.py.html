  14     def test_asfreq_ts(self):
  15         index = period_range(freq="A", start="1/1/2001", end="12/31/2010")
  16         ts = Series(np.random.randn(len(index)), index=index)
  17         df = DataFrame(np.random.randn(len(index), 3), index=index)
  18 
  19         result = ts.asfreq("D", how="end")
  20         df_result = df.asfreq("D", how="end")
  21         exp_index = index.asfreq("D", how="end")

  39     def test_asfreq(self):
  40         ts = Series(
  41             [0.0, 1.0, 2.0],
  42             index=DatetimeIndex(
  43                 [
  44                     datetime(2009, 10, 30),
  45                     datetime(2009, 11, 30),
  46                     datetime(2009, 12, 31),
  47                 ],
  48                 freq="BM",
  49             ),
  50         )
  51 
  52         daily_ts = ts.asfreq("B")
  53         monthly_ts = daily_ts.asfreq("BM")
  54         tm.assert_series_equal(monthly_ts, ts)
  55 
  56         daily_ts = ts.asfreq("B", method="pad")
  57         monthly_ts = daily_ts.asfreq("BM")
  58         tm.assert_series_equal(monthly_ts, ts)
  59 
  60         daily_ts = ts.asfreq(BDay())
  61         monthly_ts = daily_ts.asfreq(BMonthEnd())
  62         tm.assert_series_equal(monthly_ts, ts)
  63 
  64         result = ts[:0].asfreq("M")
  65         assert len(result) == 0
  66         assert result is not ts
  67 
  68         daily_ts = ts.asfreq("D", fill_value=-1)
  69         result = daily_ts.value_counts().sort_index()
  70         expected = Series([60, 1, 1, 1], index=[-1.0, 2.0, 1.0, 0.0]).sort_index()
  71         tm.assert_series_equal(result, expected)
