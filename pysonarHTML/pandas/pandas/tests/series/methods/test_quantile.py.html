  98     def test_quantile_nan(self):
  99 
 100         # GH 13098
 101         s = pd.Series([1, 2, 3, 4, np.nan])
 102         result = s.quantile(0.5)
 103         expected = 2.5
 104         assert result == expected
 105 
 106         # all nan/empty
 107         s1 = Series([], dtype=object)
 108         cases = [s1, Series([np.nan, np.nan])]

 112             assert np.isnan(res)
 113 
 114             res = s.quantile([0.5])
