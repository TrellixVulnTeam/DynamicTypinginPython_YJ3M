 204     def test_interpolate_index_values(self):
 205         s = Series(np.nan, index=np.sort(np.random.rand(30)))
 206         s[::3] = np.random.randn(10)
 207 
 208         vals = s.index.values.astype(float)
 209 
 210         result = s.interpolate(method="index")
 211 
 212         expected = s.copy()
 213         bad = isna(expected.values)
 214         good = ~bad
 215         expected = Series(
 216             np.interp(vals[bad], vals[good], s.values[good]), index=s.index[bad]
 217         )
 218 
 219         tm.assert_series_equal(result[bad], expected)
 220 
 221         # 'values' is synonymous with 'index' for the method kwarg
 222         other_result = s.interpolate(method="values")
 223 
 224         tm.assert_series_equal(other_result, result)
 225         tm.assert_series_equal(other_result[bad], expected)
