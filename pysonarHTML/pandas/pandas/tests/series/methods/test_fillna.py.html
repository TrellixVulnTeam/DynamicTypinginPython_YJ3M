 123     def test_fillna_categorical_raises(self):
 124         data = ["a", np.nan, "b", np.nan, np.nan]
 125         ser = Series(Categorical(data, categories=["a", "b"]))
 126 
 127         with pytest.raises(ValueError, match="fill value must be in categories"):
 128             ser.fillna("d")
 129 
 130         with pytest.raises(ValueError, match="fill value must be in categories"):
 131             ser.fillna(Series("d"))
 132 
 133         with pytest.raises(ValueError, match="fill value must be in categories"):
 134             ser.fillna({1: "d", 3: "a"})
 135 
 136         msg = '"value" parameter must be a scalar or dict, but you passed a "list"'
 137         with pytest.raises(TypeError, match=msg):
 138             ser.fillna(["a", "b"])
 139 
 140         msg = '"value" parameter must be a scalar or dict, but you passed a "tuple"'
 141         with pytest.raises(TypeError, match=msg):
 142             ser.fillna(("a", "b"))
 143 
 144         msg = (
 145             '"value" parameter must be a scalar, dict '
 146             'or Series, but you passed a "DataFrame"'
 147         )
 148         with pytest.raises(TypeError, match=msg):
 149             ser.fillna(DataFrame({1: ["a"], 3: ["b"]}))

 164     def test_fillna_method_and_limit_invalid(self):
 165 
 166         # related GH#9217, make sure limit is an int and greater than 0
 167         ser = Series([1, 2, 3, None])
 168         msg = (
 169             r"Cannot specify both 'value' and 'method'\.|"
 170             r"Limit must be greater than 0|"
 171             "Limit must be an integer"
 172         )
 173         for limit in [-1, 0, 1.0, 2.0]:
 174             for method in ["backfill", "bfill", "pad", "ffill", None]:
 175                 with pytest.raises(ValueError, match=msg):
 176                     ser.fillna(1, limit=limit, method=method)
