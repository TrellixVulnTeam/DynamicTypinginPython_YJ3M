  92     def test_periodindex(self):
  93         from pandas import period_range, PeriodIndex
  94 
  95         # array or list or dates
  96         N = 50
  97         rng = period_range("1/1/1990", periods=N, freq="H")
  98         ts = Series(np.random.randn(N), index=rng)
  99         ts.iloc[15:30] = np.nan
 100         dates = date_range("1/1/1990", periods=N * 3, freq="37min")
 101 
 102         result = ts.asof(dates)
 103         assert notna(result).all()
 104         lb = ts.index[14]
 105         ub = ts.index[30]
 106 
 107         result = ts.asof(list(dates))
 108         assert notna(result).all()
 109         lb = ts.index[14]
 110         ub = ts.index[30]
 111 
 112         pix = PeriodIndex(result.index.values, freq="H")
 113         mask = (pix >= lb) & (pix < ub)
 114         rs = result[mask]
 115         assert (rs == ts[lb]).all()
 116 
 117         ts.iloc[5:10] = np.nan
 118         ts.iloc[15:20] = np.nan
 119 
 120         val1 = ts.asof(ts.index[7])
 121         val2 = ts.asof(ts.index[19])
 122 
 123         assert val1 == ts[4]
 124         assert val2 == ts[14]
 125 
 126         # accepts strings
 127         val1 = ts.asof(str(ts.index[7]))
 128         assert val1 == ts[4]
 129 
 130         # in there
 131         assert ts.asof(ts.index[3]) == ts[3]
 132 
 133         # no as of value
 134         d = ts.index[0].to_timestamp() - offsets.BDay()
 135         assert isna(ts.asof(d))
 136 
 137         # Mismatched freq
 138         msg = "Input has different freq"
 139         with pytest.raises(IncompatibleFrequency, match=msg):

 161     def test_all_nans(self):
 162         # GH 15713
 163         # series is all nans
 164         result = Series([np.nan]).asof([0])
 165         expected = Series([np.nan])
 166         tm.assert_series_equal(result, expected)

 168         # testing non-default indexes
 169         N = 50
 170         rng = date_range("1/1/1990", periods=N, freq="53s")

 175         tm.assert_series_equal(result, expected)
 176 
 177         # testing scalar input
 178         date = date_range("1/1/1990", periods=N * 3, freq="25s")[0]
 179         result = Series(np.nan, index=rng).asof(date)

 181
