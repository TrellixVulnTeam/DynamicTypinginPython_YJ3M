   9 class TestRename:
  10     def test_rename(self, datetime_series):
  11         ts = datetime_series
  12         renamer = lambda x: x.strftime("%Y%m%d")
  13         renamed = ts.rename(renamer)
  14         assert renamed.index[0] == renamer(ts.index[0])
  15 
  16         # dict
  17         rename_dict = dict(zip(ts.index, renamed.index))
  18         renamed2 = ts.rename(rename_dict)
  19         tm.assert_series_equal(renamed, renamed2)
  20 
  21         # partial dict
  22         s = Series(np.arange(4), index=["a", "b", "c", "d"], dtype="int64")
  23         renamed = s.rename({"b": "foo", "d": "bar"})
  24         tm.assert_index_equal(renamed.index, Index(["a", "foo", "c", "bar"]))
  25 
  26         # index with name
  27         renamer = Series(
  28             np.arange(4), index=Index(["a", "b", "c", "d"], name="name"), dtype="int64"
  29         )
  30         renamed = renamer.rename({})
  31         assert renamed.index.name == renamer.index.name

  40     def test_rename_set_name(self):
