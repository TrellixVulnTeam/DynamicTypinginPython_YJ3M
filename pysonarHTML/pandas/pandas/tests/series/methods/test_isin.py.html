  10     def test_isin(self):
  11         s = Series(["A", "B", "C", "a", "B", "B", "A", "C"])
  12 
  13         result = s.isin(["A", "C"])
  14         expected = Series([True, False, True, False, False, False, True, True])
  15         tm.assert_series_equal(result, expected)
  16 
  17         # GH#16012
  18         # This specific issue has to have a series over 1e6 in len, but the
  19         # comparison array (in_list) must be large enough so that numpy doesn't
  20         # do a manual masking trick that will avoid this issue altogether
  21         s = Series(list("abcdefghijk" * 10 ** 5))
  22         # If numpy doesn't do the manual comparison/mask, these
  23         # unorderable mixed types are what cause the exception in numpy
  24         in_list = [-1, "a", "b", "G", "Y", "Z", "E", "K", "E", "S", "I", "R", "R"] * 6
  25 
  26         assert s.isin(in_list).sum() == 200000

  42     def test_isin_with_i8(self):
  43         # GH#5021
  44 
  45         expected = Series([True, True, False, False, False])
  46         expected2 = Series([False, True, False, False, False])
  47 
  48         # datetime64[ns]
  49         s = Series(date_range("jan-01-2013", "jan-05-2013"))
  50 
  51         result = s.isin(s[0:2])
  52         tm.assert_series_equal(result, expected)
  53 
  54         result = s.isin(s[0:2].values)
  55         tm.assert_series_equal(result, expected)
  56 
  57         # fails on dtype conversion in the first place
  58         result = s.isin(s[0:2].values.astype("datetime64[D]"))
  59         tm.assert_series_equal(result, expected)
  60 
  61         result = s.isin([s[1]])
  62         tm.assert_series_equal(result, expected2)
  63 
  64         result = s.isin([np.datetime64(s[1])])
  65         tm.assert_series_equal(result, expected2)
  66 
  67         result = s.isin(set(s[0:2]))
  68         tm.assert_series_equal(result, expected)
  69 
  70         # timedelta64[ns]
  71         s = Series(pd.to_timedelta(range(5), unit="d"))
  72         result = s.isin(s[0:2])
  73         tm.assert_series_equal(result, expected)
