   8 class TestSeriesArgsort:
   9     def _check_accum_op(self, name, ser, check_dtype=True):
  10         func = getattr(np, name)
  11         tm.assert_numpy_array_equal(
  12             func(ser).values, func(np.array(ser)), check_dtype=check_dtype,
  13         )
  14 
  15         # with missing values
  16         ts = ser.copy()
  17         ts[::2] = np.NaN
  18 
  19         result = func(ts)[1::2]
  20         expected = func(np.array(ts.dropna()))
  21 
  22         tm.assert_numpy_array_equal(result.values, expected, check_dtype=False)

  24     def test_argsort(self, datetime_series):
  25         self._check_accum_op("argsort", datetime_series, check_dtype=False)
  26         argsorted = datetime_series.argsort()
  27         assert issubclass(argsorted.dtype.type, np.integer)

  29         # GH#2967 (introduced bug in 0.11-dev I think)
