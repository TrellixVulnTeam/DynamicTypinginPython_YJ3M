  86     def test_sort_values_categorical(self):
  87 
  88         c = Categorical(["a", "b", "b", "a"], ordered=False)
  89         cat = Series(c.copy())
  90 
  91         # sort in the categories order
  92         expected = Series(
  93             Categorical(["a", "a", "b", "b"], ordered=False), index=[0, 3, 1, 2]
  94         )
  95         result = cat.sort_values()
  96         tm.assert_series_equal(result, expected)
  97 
  98         cat = Series(Categorical(["a", "c", "b", "d"], ordered=True))
  99         res = cat.sort_values()
 100         exp = np.array(["a", "b", "c", "d"], dtype=np.object_)
 101         tm.assert_numpy_array_equal(res.__array__(), exp)
 102 
 103         cat = Series(
 104             Categorical(
 105                 ["a", "c", "b", "d"], categories=["a", "b", "c", "d"], ordered=True
 106             )
 107         )
 108         res = cat.sort_values()
 109         exp = np.array(["a", "b", "c", "d"], dtype=np.object_)
 110         tm.assert_numpy_array_equal(res.__array__(), exp)
 111 
 112         res = cat.sort_values(ascending=False)
 113         exp = np.array(["d", "c", "b", "a"], dtype=np.object_)
 114         tm.assert_numpy_array_equal(res.__array__(), exp)
 115 
 116         raw_cat1 = Categorical(
 117             ["a", "b", "c", "d"], categories=["a", "b", "c", "d"], ordered=False
 118         )
 119         raw_cat2 = Categorical(
 120             ["a", "b", "c", "d"], categories=["d", "c", "b", "a"], ordered=True
 121         )
 122         s = ["a", "b", "c", "d"]
 123         df = DataFrame(
 124             {"unsort": raw_cat1, "sort": raw_cat2, "string": s, "values": [1, 2, 3, 4]}
 125         )
 126 
 127         # Cats must be sorted in a dataframe
 128         res = df.sort_values(by=["string"], ascending=False)
 129         exp = np.array(["d", "c", "b", "a"], dtype=np.object_)
 130         tm.assert_numpy_array_equal(res["sort"].values.__array__(), exp)
 131         assert res["sort"].dtype == "category"
 132 
 133         res = df.sort_values(by=["sort"], ascending=False)
 134         exp = df.sort_values(by=["string"], ascending=True)
 135         tm.assert_series_equal(res["values"], exp["values"])
 136         assert res["sort"].dtype == "category"
 137         assert res["unsort"].dtype == "category"
 138 
 139         # unordered cat, but we allow this
 140         df.sort_values(by=["unsort"], ascending=False)
 141 
 142         # multi-columns sort
 143         # GH#7848
 144         df = DataFrame(
 145             {"id": [6, 5, 4, 3, 2, 1], "raw_grade": ["a", "b", "b", "a", "a", "e"]}
 146         )
 147         df["grade"] = Categorical(df["raw_grade"], ordered=True)
 148         df["grade"] = df["grade"].cat.set_categories(["b", "e", "a"])
 149 
 150         # sorts 'grade' according to the order of the categories
 151         result = df.sort_values(by=["grade"])
 152         expected = df.iloc[[1, 2, 5, 0, 3, 4]]
 153         tm.assert_frame_equal(result, expected)
 154 
 155         # multi
 156         result = df.sort_values(by=["grade", "id"])
 157         expected = df.iloc[[2, 1, 5, 4, 3, 0]]
 158         tm.assert_frame_equal(result, expected)

 160     @pytest.mark.parametrize("inplace", [True, False])
 161     @pytest.mark.parametrize(
 162         "original_list, sorted_list, ignore_index, output_index",
 163         [
 164             ([2, 3, 6, 1], [6, 3, 2, 1], True, [0, 1, 2, 3]),
 165             ([2, 3, 6, 1], [6, 3, 2, 1], False, [2, 1, 0, 3]),
 166         ],
 167     )
 168     def test_sort_values_ignore_index(
 169         self, inplace, original_list, sorted_list, ignore_index, output_index
 170     ):
 171         # GH 30114
 172         ser = Series(original_list)
 173         expected = Series(sorted_list, index=output_index)
 174         kwargs = {"ignore_index": ignore_index, "inplace": inplace}
 175 
 176         if inplace:
 177             result_ser = ser.copy()
 178             result_ser.sort_values(ascending=False, **kwargs)
 179         else:
 180             result_ser = ser.sort_values(ascending=False, **kwargs)
 181 
 182         tm.assert_series_equal(result_ser, expected)
 183         tm.assert_series_equal(ser, Series(original_list))

 186 class TestSeriesSortingKey:
 187     def test_sort_values_key(self):
 188         series = Series(np.array(["Hello", "goodbye"]))
 189 
 190         result = series.sort_values(0)
 191         expected = series
 192         tm.assert_series_equal(result, expected)
 193 
 194         result = series.sort_values(0, key=lambda x: x.str.lower())
 195         expected = series[::-1]
 196         tm.assert_series_equal(result, expected)
 197 
 198     def test_sort_values_key_nan(self):
 199         series = Series(np.array([0, 5, np.nan, 3, 2, np.nan]))
 200 
 201         result = series.sort_values(0)
 202         expected = series.iloc[[0, 4, 3, 1, 2, 5]]
 203         tm.assert_series_equal(result, expected)
 204 
 205         result = series.sort_values(0, key=lambda x: x + 5)
 206         expected = series.iloc[[0, 4, 3, 1, 2, 5]]
 207         tm.assert_series_equal(result, expected)
 208 
 209         result = series.sort_values(0, key=lambda x: -x, ascending=False)
 210         expected = series.iloc[[0, 4, 3, 1, 2, 5]]
 211         tm.assert_series_equal(result, expected)
