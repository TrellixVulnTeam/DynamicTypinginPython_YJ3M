  68 def test_unstack_tuplename_in_multiindex():
  69     # GH 19966
  70     idx = pd.MultiIndex.from_product(
  71         [["a", "b", "c"], [1, 2, 3]], names=[("A", "a"), ("B", "b")]

  74     result = ser.unstack(("A", "a"))
  75

  77         [[1, 1, 1], [1, 1, 1], [1, 1, 1]],
  78         columns=pd.MultiIndex.from_tuples(

  82     )
  83     tm.assert_frame_equal(result, expected)
