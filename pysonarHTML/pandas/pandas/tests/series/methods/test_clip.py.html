  21     def test_clip_types_and_nulls(self):
  22 
  23         sers = [
  24             Series([np.nan, 1.0, 2.0, 3.0]),
  25             Series([None, "a", "b", "c"]),
  26             Series(pd.to_datetime([np.nan, 1, 2, 3], unit="D")),
  27         ]
  28 
  29         for s in sers:
  30             thresh = s[2]
  31             lower = s.clip(lower=thresh)
  32             upper = s.clip(upper=thresh)
  33             assert lower[notna(lower)].min() == thresh
  34             assert upper[notna(upper)].max() == thresh
  35             assert list(isna(s)) == list(isna(lower))
  36             assert list(isna(s)) == list(isna(upper))
