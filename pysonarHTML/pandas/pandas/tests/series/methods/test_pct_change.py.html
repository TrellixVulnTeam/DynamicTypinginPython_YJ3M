   9     def test_pct_change(self, datetime_series):
  10         rs = datetime_series.pct_change(fill_method=None)
  11         tm.assert_series_equal(rs, datetime_series / datetime_series.shift(1) - 1)
  12 
  13         rs = datetime_series.pct_change(2)
  14         filled = datetime_series.fillna(method="pad")
  15         tm.assert_series_equal(rs, filled / filled.shift(2) - 1)
  16 
  17         rs = datetime_series.pct_change(fill_method="bfill", limit=1)
  18         filled = datetime_series.fillna(method="bfill", limit=1)
  19         tm.assert_series_equal(rs, filled / filled.shift(1) - 1)
  20 
  21         rs = datetime_series.pct_change(freq="5D")
  22         filled = datetime_series.fillna(method="pad")
  23         tm.assert_series_equal(
  24             rs, (filled / filled.shift(freq="5D") - 1).reindex_like(filled)
  25         )
