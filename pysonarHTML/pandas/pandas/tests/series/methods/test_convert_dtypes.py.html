 256     @pytest.mark.parametrize("params", product(*[(True, False)] * 4))
 257     def test_convert_dtypes(self, data, maindtype, params, answerdict):
 258         if maindtype is not None:
 259             series = pd.Series(data, dtype=maindtype)
 260         else:
 261             series = pd.Series(data)
 262         answers = {k: a for (kk, a) in answerdict.items() for k in product(*kk)}
 263 
 264         ns = series.convert_dtypes(*params)
 265         expected_dtype = answers[tuple(params)]
 266         expected = pd.Series(series.values, dtype=expected_dtype)
 267         tm.assert_series_equal(ns, expected)
 268 
 269         # Test that it is a copy
 270         copy = series.copy(deep=True)
 271         if is_interval_dtype(ns.dtype) and ns.dtype.subtype.kind in ["i", "u"]:
 272             msg = "Cannot set float NaN to integer-backed IntervalArray"
 273             with pytest.raises(ValueError, match=msg):
 274                 ns[ns.notna()] = np.nan
 275         else:
 276             ns[ns.notna()] = np.nan
 277 
 278         # Make sure original not changed
 279         tm.assert_series_equal(series, copy)

 281     def test_convert_string_dtype(self):
 282         # https://github.com/pandas-dev/pandas/issues/31731 -> converting columns
 283         # that are already string dtype
 284         df = pd.DataFrame(
 285             {"A": ["a", "b", pd.NA], "B": ["ä", "ö", "ü"]}, dtype="string"
 286         )
 287         result = df.convert_dtypes()
 288         tm.assert_frame_equal(df, result)
