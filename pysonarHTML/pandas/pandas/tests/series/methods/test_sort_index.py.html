  15     def test_sort_index(self, datetime_series):
  16         datetime_series.index = datetime_series.index._with_freq(None)
  17 
  18         rindex = list(datetime_series.index)
  19         random.shuffle(rindex)
  20 
  21         random_order = datetime_series.reindex(rindex)
  22         sorted_series = random_order.sort_index()
  23         tm.assert_series_equal(sorted_series, datetime_series)
  24 
  25         # descending
  26         sorted_series = random_order.sort_index(ascending=False)
  27         tm.assert_series_equal(
  28             sorted_series, datetime_series.reindex(datetime_series.index[::-1])
  29         )
  30 
  31         # compat on level
  32         sorted_series = random_order.sort_index(level=0)
  33         tm.assert_series_equal(sorted_series, datetime_series)
  34 
  35         # compat on axis
  36         sorted_series = random_order.sort_index(axis=0)
  37         tm.assert_series_equal(sorted_series, datetime_series)
  38 
  39         msg = "No axis named 1 for object type Series"
  40         with pytest.raises(ValueError, match=msg):
  41             random_order.sort_values(axis=1)
  42 
  43         sorted_series = random_order.sort_index(level=0, axis=0)
  44         tm.assert_series_equal(sorted_series, datetime_series)
  45 
  46         with pytest.raises(ValueError, match=msg):
  47             random_order.sort_index(level=0, axis=1)

  49     def test_sort_index_inplace(self, datetime_series):
  50         datetime_series.index = datetime_series.index._with_freq(None)
  51 
  52         # For GH#11402
  53         rindex = list(datetime_series.index)
  54         random.shuffle(rindex)
  55 
  56         # descending
  57         random_order = datetime_series.reindex(rindex)
  58         result = random_order.sort_index(ascending=False, inplace=True)
  59 
  60         assert result is None
  61         expected = datetime_series.reindex(datetime_series.index[::-1])
  62         expected.index = expected.index._with_freq(None)
  63         tm.assert_series_equal(random_order, expected)
  64 
  65         # ascending
  66         random_order = datetime_series.reindex(rindex)
  67         result = random_order.sort_index(ascending=True, inplace=True)
  68 
  69         assert result is None
  70         expected = datetime_series.copy()
  71         expected.index = expected.index._with_freq(None)
  72         tm.assert_series_equal(random_order, expected)

 151         [

 179     def test_sort_index_ascending_list(self):
 180         # GH#16934
 181 
 182         # Set up a Series with a three level MultiIndex
 183         arrays = [
 184             ["bar", "bar", "baz", "baz", "foo", "foo", "qux", "qux"],
 185             ["one", "two", "one", "two", "one", "two", "one", "two"],
 186             [4, 3, 2, 1, 4, 3, 2, 1],
 187         ]
 188         tuples = zip(*arrays)
 189         mi = MultiIndex.from_tuples(tuples, names=["first", "second", "third"])
 190         ser = Series(range(8), index=mi)
 191 
 192         # Sort with boolean ascending
 193         result = ser.sort_index(level=["third", "first"], ascending=False)
 194         expected = ser.iloc[[4, 0, 5, 1, 6, 2, 7, 3]]
 195         tm.assert_series_equal(result, expected)
 196 
 197         # Sort with list of boolean ascending
 198         result = ser.sort_index(level=["third", "first"], ascending=[False, True])
 199         expected = ser.iloc[[0, 4, 1, 5, 2, 6, 3, 7]]
 200         tm.assert_series_equal(result, expected)
