   9 class TestSeriesAppend:
  10     def test_append(self, datetime_series, string_series, object_series):
  11         appended_series = string_series.append(object_series)
  12         for idx, value in appended_series.items():
  13             if idx in string_series.index:
  14                 assert value == string_series[idx]
  15             elif idx in object_series.index:
  16                 assert value == object_series[idx]
  17             else:
  18                 raise AssertionError("orphaned index!")
  19 
  20         msg = "Indexes have overlapping values:"
  21         with pytest.raises(ValueError, match=msg):
  22             datetime_series.append(datetime_series, verify_integrity=True)

 170     def test_series_append_aware(self):
 171         rng1 = date_range("1/1/2011 01:00", periods=1, freq="H", tz="US/Eastern")
 172         rng2 = date_range("1/1/2011 02:00", periods=1, freq="H", tz="US/Eastern")
 173         ser1 = Series([1], index=rng1)
 174         ser2 = Series([2], index=rng2)
 175         ts_result = ser1.append(ser2)
 176 
 177         exp_index = DatetimeIndex(
 178             ["2011-01-01 01:00", "2011-01-01 02:00"], tz="US/Eastern", freq="H"
 179         )
 180         exp = Series([1, 2], index=exp_index)
 181         tm.assert_series_equal(ts_result, exp)
 182         assert ts_result.index.tz == rng1.tz
 183 
 184         rng1 = date_range("1/1/2011 01:00", periods=1, freq="H", tz="UTC")
 185         rng2 = date_range("1/1/2011 02:00", periods=1, freq="H", tz="UTC")
 186         ser1 = Series([1], index=rng1)
 187         ser2 = Series([2], index=rng2)
 188         ts_result = ser1.append(ser2)
 189 
 190         exp_index = DatetimeIndex(
 191             ["2011-01-01 01:00", "2011-01-01 02:00"], tz="UTC", freq="H"
 192         )
 193         exp = Series([1, 2], index=exp_index)
 194         tm.assert_series_equal(ts_result, exp)
 195         utc = rng1.tz
 196         assert utc == ts_result.index.tz
 197 
 198         # GH#7795
 199         # different tz coerces to object dtype, not UTC
 200         rng1 = date_range("1/1/2011 01:00", periods=1, freq="H", tz="US/Eastern")
 201         rng2 = date_range("1/1/2011 02:00", periods=1, freq="H", tz="US/Central")
 202         ser1 = Series([1], index=rng1)
 203         ser2 = Series([2], index=rng2)
 204         ts_result = ser1.append(ser2)
 205         exp_index = Index(
 206             [
 207                 Timestamp("1/1/2011 01:00", tz="US/Eastern"),
 208                 Timestamp("1/1/2011 02:00", tz="US/Central"),
 209             ]
 210         )
 211         exp = Series([1, 2], index=exp_index)
 212         tm.assert_series_equal(ts_result, exp)
