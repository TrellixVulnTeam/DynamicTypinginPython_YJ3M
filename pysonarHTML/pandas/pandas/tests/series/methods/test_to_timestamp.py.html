  12         series = Series(1, index=index, name="foo")
  13 
  14         exp_index = date_range("1/1/2001", end="12/31/2009", freq="A-DEC")

  16         exp_index = exp_index + Timedelta(1, "D") - Timedelta(1, "ns")
  17         tm.assert_index_equal(result.index, exp_index)
  18         assert result.name == "foo"

  20         exp_index = date_range("1/1/2001", end="1/1/2009", freq="AS-JAN")

  24         def _get_with_delta(delta, freq="A-DEC"):
  25             return date_range(
