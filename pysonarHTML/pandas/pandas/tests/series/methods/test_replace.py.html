 153     def test_replace_mixed_types(self):
 154         s = pd.Series(np.arange(5), dtype="int64")
 155 
 156         def check_replace(to_rep, val, expected):
 157             sc = s.copy()
 158             r = s.replace(to_rep, val)
 159             sc.replace(to_rep, val, inplace=True)
 160             tm.assert_series_equal(expected, r)
 161             tm.assert_series_equal(expected, sc)
 162 
 163         # MUST upcast to float
 164         e = pd.Series([0.0, 1.0, 2.0, 3.0, 4.0])
 165         tr, v = [3], [3.0]
 166         check_replace(tr, v, e)
 167 
 168         # MUST upcast to float
 169         e = pd.Series([0, 1, 2, 3.5, 4])
 170         tr, v = [3], [3.5]
 171         check_replace(tr, v, e)
 172 
 173         # casts to object
 174         e = pd.Series([0, 1, 2, 3.5, "a"])
 175         tr, v = [3, 4], [3.5, "a"]
 176         check_replace(tr, v, e)
 177 
 178         # again casts to object
 179         e = pd.Series([0, 1, 2, 3.5, pd.Timestamp("20130101")])
 180         tr, v = [3, 4], [3.5, pd.Timestamp("20130101")]
 181         check_replace(tr, v, e)
 182 
 183         # casts to object
 184         e = pd.Series([0, 1, 2, 3.5, True], dtype="object")
 185         tr, v = [3, 4], [3.5, True]
 186         check_replace(tr, v, e)
 187 
 188         # test an object with dates + floats + integers + strings
 189         dr = pd.Series(pd.date_range("1/1/2001", "1/10/2001", freq="D"))
