 100     def test_shift2(self):

 104

 157     def test_shift_dst(self):
 158         # GH#13926
 159         dates = date_range("2016-11-06", freq="H", periods=10, tz="US/Eastern")
 160         s = Series(dates)
 161 
 162         res = s.shift(0)
 163         tm.assert_series_equal(res, s)

 165

 184     def test_tshift(self, datetime_series):
 185         # PeriodIndex
 186         ps = tm.makePeriodSeries()
 187         shifted = ps.tshift(1)
 188         unshifted = shifted.tshift(-1)
 189 
 190         tm.assert_series_equal(unshifted, ps)
 191 
 192         shifted2 = ps.tshift(freq="B")
 193         tm.assert_series_equal(shifted, shifted2)
 194 
 195         shifted3 = ps.tshift(freq=BDay())
 196         tm.assert_series_equal(shifted, shifted3)
 197 
 198         msg = "Given freq M does not match PeriodIndex freq B"
 199         with pytest.raises(ValueError, match=msg):
 200             ps.tshift(freq="M")
 201 
 202         # DatetimeIndex
 203         shifted = datetime_series.tshift(1)
 204         unshifted = shifted.tshift(-1)
 205 
 206         tm.assert_series_equal(datetime_series, unshifted)
 207 
 208         shifted2 = datetime_series.tshift(freq=datetime_series.index.freq)
 209         tm.assert_series_equal(shifted, shifted2)
 210 
 211         inferred_ts = Series(
 212             datetime_series.values, Index(np.asarray(datetime_series.index)), name="ts"
 213         )
 214         shifted = inferred_ts.tshift(1)
 215         expected = datetime_series.tshift(1)
 216         expected.index = expected.index._with_freq(None)
 217         tm.assert_series_equal(shifted, expected)
 218 
 219         unshifted = shifted.tshift(-1)
 220         tm.assert_series_equal(unshifted, inferred_ts)
 221 
 222         no_freq = datetime_series[[0, 5, 7]]
 223         msg = "Freq was not given and was not set in the index"
 224         with pytest.raises(ValueError, match=msg):
 225             no_freq.tshift()

 233     def test_shift_object_non_scalar_fill(self):
 234         # shift requires scalar fill_value except for object dtype
 235         ser = Series(range(3))
 236         with pytest.raises(ValueError, match="fill_value must be a scalar"):
 237             ser.shift(1, fill_value=[])
 238 
 239         df = ser.to_frame()
 240         with pytest.raises(ValueError, match="fill_value must be a scalar"):
 241             df.shift(1, fill_value=np.arange(3))
 242 
 243         obj_ser = ser.astype(object)
 244         result = obj_ser.shift(1, fill_value={})
 245         assert result[0] == {}
 246 
 247         obj_df = obj_ser.to_frame()
 248         result = obj_df.shift(1, fill_value={})
 249         assert result.iloc[0, 0] == {}
