 129     def test_corr_callable_method(self, datetime_series):
 130         # simple correlation example
 131         # returns 1 if exact equality, 0 otherwise
 132         my_corr = lambda a, b: 1.0 if (a == b).all() else 0.0
 133 
 134         # simple example
 135         s1 = Series([1, 2, 3, 4, 5])
 136         s2 = Series([5, 4, 3, 2, 1])
 137         expected = 0
 138         tm.assert_almost_equal(s1.corr(s2, method=my_corr), expected)
 139 
 140         # full overlap
 141         tm.assert_almost_equal(
 142             datetime_series.corr(datetime_series, method=my_corr), 1.0
 143         )
 144 
 145         # partial overlap
 146         tm.assert_almost_equal(
 147             datetime_series[:15].corr(datetime_series[5:], method=my_corr), 1.0
 148         )
 149 
 150         # No overlap
 151         assert np.isnan(
 152             datetime_series[::2].corr(datetime_series[1::2], method=my_corr)
 153         )
 154 
 155         # dataframe example
 156         df = pd.DataFrame([s1, s2])
 157         expected = pd.DataFrame([{0: 1.0, 1: 0}, {0: 0, 1: 1.0}])
 158         tm.assert_almost_equal(df.transpose().corr(method=my_corr), expected)
