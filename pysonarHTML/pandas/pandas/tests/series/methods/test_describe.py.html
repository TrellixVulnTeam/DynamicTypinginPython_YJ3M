  62     def test_describe_empty_object(self):
  63         # https://github.com/pandas-dev/pandas/issues/27183
  64         s = Series([None, None], dtype=object)
  65         result = s.describe()
  66         expected = Series(
  67             [0, 0, np.nan, np.nan],
  68             dtype=object,
  69             index=["count", "unique", "top", "freq"],
  70         )
  71         tm.assert_series_equal(result, expected)
  72 
  73         result = s[:0].describe()
  74         tm.assert_series_equal(result, expected)
  75         # ensure NaN, not None
  76         assert np.isnan(result.iloc[2])
  77         assert np.isnan(result.iloc[3])
