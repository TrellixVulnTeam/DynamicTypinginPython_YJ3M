  16     def test_to_period(self):
  17         rng = date_range("1/1/2000", "1/1/2001", freq="D")
  18         ts = Series(np.random.randn(len(rng)), index=rng)
  19 
  20         pts = ts.to_period()
  21         exp = ts.copy()
  22         exp.index = period_range("1/1/2000", "1/1/2001")
  23         tm.assert_series_equal(pts, exp)
  24 
  25         pts = ts.to_period("M")
  26         exp.index = exp.index.asfreq("M")
  27         tm.assert_index_equal(pts.index, exp.index.asfreq("M"))
  28         tm.assert_series_equal(pts, exp)
  29 
  30         # GH#7606 without freq
  31         idx = DatetimeIndex(["2011-01-01", "2011-01-02", "2011-01-03", "2011-01-04"])
  32         exp_idx = PeriodIndex(
  33             ["2011-01-01", "2011-01-02", "2011-01-03", "2011-01-04"], freq="D"
  34         )
  35 
  36         s = Series(np.random.randn(4), index=idx)
  37         expected = s.copy()
  38         expected.index = exp_idx
  39         tm.assert_series_equal(s.to_period(), expected)
  40 
  41         df = DataFrame(np.random.randn(4, 4), index=idx, columns=idx)
  42         expected = df.copy()
  43         expected.index = exp_idx
  44         tm.assert_frame_equal(df.to_period(), expected)
  45 
  46         expected = df.copy()
