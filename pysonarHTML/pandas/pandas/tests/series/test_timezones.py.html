  17         values = [188.5, 328.25]
  18         tzinfo = tzoffset(None, 7200)
  19         index = [
  20             datetime(2012, 5, 11, 11, tzinfo=tzinfo),
  21             datetime(2012, 5, 11, 12, tzinfo=tzinfo),
  22         ]
  23         series = Series(data=values, index=index)
  24
