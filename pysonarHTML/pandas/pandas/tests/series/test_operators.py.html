 380     def test_logical_ops_label_based(self):
 381         # GH#4947
 382         # logical ops should be label based
 383 
 384         a = Series([True, False, True], list("bca"))
 385         b = Series([False, True, False], list("abc"))
 386 
 387         expected = Series([False, True, False], list("abc"))
 388         result = a & b
 389         tm.assert_series_equal(result, expected)
 390 
 391         expected = Series([True, True, False], list("abc"))
 392         result = a | b
 393         tm.assert_series_equal(result, expected)
 394 
 395         expected = Series([True, False, False], list("abc"))
 396         result = a ^ b
 397         tm.assert_series_equal(result, expected)
 398 
 399         # rhs is bigger
 400         a = Series([True, False, True], list("bca"))
 401         b = Series([False, True, False, True], list("abcd"))
 402 
 403         expected = Series([False, True, False, False], list("abcd"))
 404         result = a & b
 405         tm.assert_series_equal(result, expected)
 406 
 407         expected = Series([True, True, False, False], list("abcd"))
 408         result = a | b
 409         tm.assert_series_equal(result, expected)
 410 
 411         # filling
 412 
 413         # vs empty
 414         empty = Series([], dtype=object)
 415 
 416         result = a & empty.copy()
 417         expected = Series([False, False, False], list("bca"))
 418         tm.assert_series_equal(result, expected)
 419 
 420         result = a | empty.copy()

 422         tm.assert_series_equal(result, expected)
 423

 427         tm.assert_series_equal(result, expected)

 481     def test_logical_ops_df_compat(self):
 482         # GH#1134
 483         s1 = pd.Series([True, False, True], index=list("ABC"), name="x")
 484         s2 = pd.Series([True, True, False], index=list("ABD"), name="x")
 485 
 486         exp = pd.Series([True, False, False, False], index=list("ABCD"), name="x")
 487         tm.assert_series_equal(s1 & s2, exp)
 488         tm.assert_series_equal(s2 & s1, exp)
 489 
 490         # True | np.nan => True
 491         exp_or1 = pd.Series([True, True, True, False], index=list("ABCD"), name="x")
 492         tm.assert_series_equal(s1 | s2, exp_or1)
 493         # np.nan | True => np.nan, filled with False
 494         exp_or = pd.Series([True, True, False, False], index=list("ABCD"), name="x")
 495         tm.assert_series_equal(s2 | s1, exp_or)
 496 
 497         # DataFrame doesn't fill nan with False
 498         tm.assert_frame_equal(s1.to_frame() & s2.to_frame(), exp.to_frame())
 499         tm.assert_frame_equal(s2.to_frame() & s1.to_frame(), exp.to_frame())
 500 
 501         exp = pd.DataFrame({"x": [True, True, np.nan, np.nan]}, index=list("ABCD"))
 502         tm.assert_frame_equal(s1.to_frame() | s2.to_frame(), exp_or1.to_frame())
 503         tm.assert_frame_equal(s2.to_frame() | s1.to_frame(), exp_or.to_frame())
 504 
 505         # different length
 506         s3 = pd.Series([True, False, True], index=list("ABC"), name="x")
 507         s4 = pd.Series([True, True, True, True], index=list("ABCD"), name="x")
 508 
 509         exp = pd.Series([True, False, True, False], index=list("ABCD"), name="x")
 510         tm.assert_series_equal(s3 & s4, exp)
 511         tm.assert_series_equal(s4 & s3, exp)
 512 
 513         # np.nan | True => np.nan, filled with False
 514         exp_or1 = pd.Series([True, True, True, False], index=list("ABCD"), name="x")
 515         tm.assert_series_equal(s3 | s4, exp_or1)
 516         # True | np.nan => True
 517         exp_or = pd.Series([True, True, True, True], index=list("ABCD"), name="x")
 518         tm.assert_series_equal(s4 | s3, exp_or)
 519 
 520         tm.assert_frame_equal(s3.to_frame() & s4.to_frame(), exp.to_frame())
 521         tm.assert_frame_equal(s4.to_frame() & s3.to_frame(), exp.to_frame())
 522 
 523         tm.assert_frame_equal(s3.to_frame() | s4.to_frame(), exp_or1.to_frame())
 524         tm.assert_frame_equal(s4.to_frame() | s3.to_frame(), exp_or.to_frame())
