  17 def _check_accum_op(name, series, check_dtype=True):
  18     func = getattr(np, name)
  19     tm.assert_numpy_array_equal(
  20         func(series).values, func(np.array(series)), check_dtype=check_dtype,
  21     )
  22 
  23     # with missing values
  24     ts = series.copy()
  25     ts[::2] = np.NaN
  26 
  27     result = func(ts)[1::2]
  28     expected = func(np.array(ts.dropna()))
  29 
  30     tm.assert_numpy_array_equal(result.values, expected, check_dtype=False)

  40     def test_cummin(self, datetime_series):
  41         tm.assert_numpy_array_equal(
  42             datetime_series.cummin().values,
  43             np.minimum.accumulate(np.array(datetime_series)),
  44         )
  45         ts = datetime_series.copy()
  46         ts[::2] = np.NaN
  47         result = ts.cummin()[1::2]
  48         expected = np.minimum.accumulate(ts.dropna())
  49 
  50         result.index = result.index._with_freq(None)
  51         tm.assert_series_equal(result, expected)

 144     def test_cummethods_bool(self):
 145         # GH#6270
 146 
 147         a = pd.Series([False, False, False, True, True, False, False])
 148         b = ~a
 149         c = pd.Series([False] * len(b))
 150         d = ~c
 151         methods = {
 152             "cumsum": np.cumsum,
 153             "cumprod": np.cumprod,
 154             "cummin": np.minimum.accumulate,
 155             "cummax": np.maximum.accumulate,
 156         }
 157         args = product((a, b, c, d), methods)
 158         for s, method in args:
 159             expected = pd.Series(methods[method](s.values))
 160             result = getattr(s, method)()
 161             tm.assert_series_equal(result, expected)
 162 
 163         e = pd.Series([False, True, np.nan, False])
 164         cse = pd.Series([0, 1, np.nan, 1], dtype=object)
 165         cpe = pd.Series([False, 0, np.nan, 0])
 166         cmin = pd.Series([False, False, np.nan, False])
 167         cmax = pd.Series([False, True, np.nan, True])
 168         expecteds = {"cumsum": cse, "cumprod": cpe, "cummin": cmin, "cummax": cmax}
 169 
 170         for method in methods:
 171             res = getattr(e, method)()
 172             tm.assert_series_equal(res, expecteds[method])
