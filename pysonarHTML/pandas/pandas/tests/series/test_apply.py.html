 203     def test_transform(self, string_series):
 204         # transforming functions
 205 
 206         with np.errstate(all="ignore"):
 207 
 208             f_sqrt = np.sqrt(string_series)
 209             f_abs = np.abs(string_series)
 210 
 211             # ufunc
 212             result = string_series.transform(np.sqrt)
 213             expected = f_sqrt.copy()
 214             tm.assert_series_equal(result, expected)
 215 
 216             result = string_series.apply(np.sqrt)
 217             tm.assert_series_equal(result, expected)
 218 
 219             # list-like
 220             result = string_series.transform([np.sqrt])
 221             expected = f_sqrt.to_frame().copy()
 222             expected.columns = ["sqrt"]
 223             tm.assert_frame_equal(result, expected)
 224 
 225             result = string_series.transform([np.sqrt])
 226             tm.assert_frame_equal(result, expected)
 227 
 228             result = string_series.transform(["sqrt"])
 229             tm.assert_frame_equal(result, expected)
 230 
 231             # multiple items in list
 232             # these are in the order as if we are applying both functions per
 233             # series and then concatting
 234             expected = pd.concat([f_sqrt, f_abs], axis=1)
 235             expected.columns = ["sqrt", "absolute"]
 236             result = string_series.apply([np.sqrt, np.abs])
 237             tm.assert_frame_equal(result, expected)
 238 
 239             result = string_series.transform(["sqrt", "abs"])
 240             expected.columns = ["sqrt", "abs"]
 241             tm.assert_frame_equal(result, expected)
 242 
 243             # dict, provide renaming
 244             expected = pd.concat([f_sqrt, f_abs], axis=1)
 245             expected.columns = ["foo", "bar"]
 246             expected = expected.unstack().rename("series")
 247 
 248             result = string_series.apply({"foo": np.sqrt, "bar": np.abs})
 249             tm.assert_series_equal(result.reindex_like(expected), expected)

 263             with np.errstate(all="ignore"):
