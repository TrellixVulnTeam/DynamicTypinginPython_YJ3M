  16     def test_contiguous_boolean_preserve_freq(self):
  17         rng = date_range("1/1/2000", "3/1/2000", freq="B")
  18 
  19         mask = np.zeros(len(rng), dtype=bool)
  20         mask[10:20] = True
  21 
  22         masked = rng[mask]
