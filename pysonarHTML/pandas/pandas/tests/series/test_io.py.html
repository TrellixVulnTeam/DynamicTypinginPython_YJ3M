 131     def test_to_csv_compression(self, s, encoding, compression):
 132 
 133         with tm.ensure_clean() as filename:
 134 
 135             s.to_csv(filename, compression=compression, encoding=encoding, header=True)
 136             # test the round trip - to_csv -> read_csv
 137             result = pd.read_csv(
 138                 filename,
 139                 compression=compression,
 140                 encoding=encoding,
 141                 index_col=0,
 142                 squeeze=True,
 143             )
 144             tm.assert_series_equal(s, result)
 145 
 146             # test the round trip using file handle - to_csv -> read_csv
 147             f, _handles = get_handle(
 148                 filename, "w", compression=compression, encoding=encoding
 149             )
 150             with f:
 151                 s.to_csv(f, encoding=encoding, header=True)
 152             result = pd.read_csv(
 153                 filename,
 154                 compression=compression,
 155                 encoding=encoding,
 156                 index_col=0,
 157                 squeeze=True,
 158             )
 159             tm.assert_series_equal(s, result)
 160 
 161             # explicitly ensure file was compressed
 162             with tm.decompress_file(filename, compression) as fh:
 163                 text = fh.read().decode(encoding or "utf8")
 164                 assert s.name in text
 165 
 166             with tm.decompress_file(filename, compression) as fh:
 167                 tm.assert_series_equal(
 168                     s, pd.read_csv(fh, index_col=0, squeeze=True, encoding=encoding)
 169                 )
