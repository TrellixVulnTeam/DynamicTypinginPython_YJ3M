  20     def test_matmul(self):
  21         # matmul test is for GH #10259
  22         a = Series(np.random.randn(4), index=["p", "q", "r", "s"])
  23         b = DataFrame(
  24             np.random.randn(3, 4), index=["1", "2", "3"], columns=["p", "q", "r", "s"]
  25         ).T
  26 
  27         # Series @ DataFrame -> Series
  28         result = operator.matmul(a, b)
  29         expected = Series(np.dot(a.values, b.values), index=["1", "2", "3"])
  30         tm.assert_series_equal(result, expected)
  31 
  32         # DataFrame @ Series -> Series
  33         result = operator.matmul(b.T, a)
  34         expected = Series(np.dot(b.T.values, a.T.values), index=["1", "2", "3"])
  35         tm.assert_series_equal(result, expected)
  36 
  37         # Series @ Series -> scalar
  38         result = operator.matmul(a, a)
  39         expected = np.dot(a.values, a.values)
  40         tm.assert_almost_equal(result, expected)
  41 
  42         # GH 21530
  43         # vector (1D np.array) @ Series (__rmatmul__)
  44         result = operator.matmul(a.values, a)
  45         expected = np.dot(a.values, a.values)
  46         tm.assert_almost_equal(result, expected)
  47 
  48         # GH 21530
  49         # vector (1D list) @ Series (__rmatmul__)
  50         result = operator.matmul(a.values.tolist(), a)
  51         expected = np.dot(a.values, a.values)
  52         tm.assert_almost_equal(result, expected)
  53 
  54         # GH 21530
  55         # matrix (2D np.array) @ Series (__rmatmul__)
  56         result = operator.matmul(b.T.values, a)
  57         expected = np.dot(b.T.values, a.values)
  58         tm.assert_almost_equal(result, expected)
  59 
  60         # GH 21530
  61         # matrix (2D nested lists) @ Series (__rmatmul__)
  62         result = operator.matmul(b.T.values.tolist(), a)
  63         expected = np.dot(b.T.values, a.values)
  64         tm.assert_almost_equal(result, expected)
  65 
  66         # mixed dtype DataFrame @ Series
  67         a["p"] = int(a.p)
  68         result = operator.matmul(b.T, a)
  69         expected = Series(np.dot(b.T.values, a.T.values), index=["1", "2", "3"])
  70         tm.assert_series_equal(result, expected)
  71 
  72         # different dtypes DataFrame @ Series
  73         a = a.astype(int)
  74         result = operator.matmul(b.T, a)
  75         expected = Series(np.dot(b.T.values, a.T.values), index=["1", "2", "3"])
  76         tm.assert_series_equal(result, expected)
  77 
  78         msg = r"Dot product shape mismatch, \(4,\) vs \(3,\)"
  79         # exception raised is of type Exception
  80         with pytest.raises(Exception, match=msg):
  81             a.dot(a.values[:3])
  82         msg = "matrices are not aligned"
  83         with pytest.raises(ValueError, match=msg):
  84             a.dot(b.T)

 114     def test_is_monotonic(self):
 115 
 116         s = Series(np.random.randint(0, 10, size=1000))
 117         assert not s.is_monotonic
 118         s = Series(np.arange(1000))
 119         assert s.is_monotonic is True
 120         assert s.is_monotonic_increasing is True
 121         s = Series(np.arange(1000, 0, -1))
 122         assert s.is_monotonic_decreasing is True
 123 
 124         s = Series(pd.date_range("20130101", periods=10))
 125         assert s.is_monotonic is True
 126         assert s.is_monotonic_increasing is True
 127         s = Series(list(reversed(s.tolist())))
 128         assert s.is_monotonic is False
 129         assert s.is_monotonic_decreasing is True

 181     def test_td64_summation_overflow(self):
 182         # GH 9442
 183         s = pd.Series(pd.date_range("20130101", periods=100000, freq="H"))
 184         s[0] += pd.Timedelta("1s 1ms")
 185 
 186         # mean
 187         result = (s - s.min()).mean()
 188         expected = pd.Timedelta((pd.TimedeltaIndex((s - s.min())).asi8 / len(s)).sum())
 189 
 190         # the computation is converted to float so
 191         # might be some loss of precision
 192         assert np.allclose(result.value / 1000, expected.value / 1000)
 193 
 194         # sum
 195         msg = "overflow in timedelta operation"
 196         with pytest.raises(ValueError, match=msg):
 197             (s - s.min()).sum()
 198 
 199         s1 = s[0:10000]
 200         with pytest.raises(ValueError, match=msg):
 201             (s1 - s1.min()).sum()
 202         s2 = s[0:1000]
 203         (s2 - s2.min()).sum()
