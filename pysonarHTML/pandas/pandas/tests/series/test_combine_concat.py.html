  21     def test_concat_empty_series_dtypes_roundtrips(self):
  22 
  23         # round-tripping with self & like self
  24         dtypes = map(np.dtype, ["float64", "int8", "uint8", "bool", "m8[ns]", "M8[ns]"])
  25 
  26         def int_result_type(dtype, dtype2):
  27             typs = {dtype.kind, dtype2.kind}
  28             if not len(typs - {"i", "u", "b"}) and (
  29                 dtype.kind == "i" or dtype2.kind == "i"
  30             ):
  31                 return "i"
  32             elif not len(typs - {"u", "b"}) and (
  33                 dtype.kind == "u" or dtype2.kind == "u"
  34             ):
  35                 return "u"
  36             return None
  37 
  38         def float_result_type(dtype, dtype2):
  39             typs = {dtype.kind, dtype2.kind}
  40             if not len(typs - {"f", "i", "u"}) and (
  41                 dtype.kind == "f" or dtype2.kind == "f"
  42             ):
  43                 return "f"
  44             return None
  45 
  46         def get_result_type(dtype, dtype2):
  47             result = float_result_type(dtype, dtype2)
  48             if result is not None:
  49                 return result
  50             result = int_result_type(dtype, dtype2)
  51             if result is not None:
  52                 return result
  53             return "O"
  54 
  55         for dtype in dtypes:
  56             for dtype2 in dtypes:
  57                 if dtype == dtype2:
  58                     continue
  59 
  60                 expected = get_result_type(dtype, dtype2)
  61                 result = pd.concat([Series(dtype=dtype), Series(dtype=dtype2)]).dtype
  62                 assert result.kind == expected
