  24 def test_unique():
  25     # GH714 also, dtype=float
  26     s = Series([1.2345] * 100)
  27     s[::2] = np.nan
  28     result = s.unique()
  29     assert len(result) == 2
  30 
  31     s = Series([1.2345] * 100, dtype="f4")
  32     s[::2] = np.nan
  33     result = s.unique()
  34     assert len(result) == 2
  35 
  36     # NAs in object arrays #714
  37     s = Series(["foo"] * 100, dtype="O")
  38     s[::2] = np.nan
  39     result = s.unique()
  40     assert len(result) == 2
  41 
  42     # decision about None
  43     s = Series([1, 2, 3, None, None, None], dtype=object)
  44     result = s.unique()
  45     expected = np.array([1, 2, 3, None], dtype=object)
  46     tm.assert_numpy_array_equal(result, expected)
  47 
  48     # GH 18051
  49     s = Series(Categorical([]))
  50     tm.assert_categorical_equal(s.unique(), Categorical([]))

  52     tm.assert_categorical_equal(s.unique(), Categorical([np.nan]))
