  89     def test_operators_combine(self, op, equiv_op, fv):
  90         def _check_fill(meth, op, a, b, fill_value=0):
  91             exp_index = a.index.union(b.index)
  92             a = a.reindex(exp_index)
  93             b = b.reindex(exp_index)
  94 
  95             amask = isna(a)
  96             bmask = isna(b)
  97 
  98             exp_values = []
  99             for i in range(len(exp_index)):
 100                 with np.errstate(all="ignore"):
 101                     if amask[i]:
 102                         if bmask[i]:
 103                             exp_values.append(np.nan)
 104                             continue
 105                         exp_values.append(op(fill_value, b[i]))
 106                     elif bmask[i]:
 107                         if amask[i]:
 108                             exp_values.append(np.nan)
 109                             continue
 110                         exp_values.append(op(a[i], fill_value))
 111                     else:
 112                         exp_values.append(op(a[i], b[i]))
 113 
 114             result = meth(a, b, fill_value=fill_value)
 115             expected = Series(exp_values, exp_index)
 116             tm.assert_series_equal(result, expected)

 432     def test_categorical_comparisons(self):
 433         # GH#8938
 434         # allow equality comparisons
 435         a = Series(list("abc"), dtype="category")
 436         b = Series(list("abc"), dtype="object")
 437         c = Series(["a", "b", "cc"], dtype="object")
 438         d = Series(list("acb"), dtype="object")
 439         e = Categorical(list("abc"))
 440         f = Categorical(list("acb"))
 441 
 442         # vs scalar
 443         assert not (a == "a").all()
 444         assert ((a != "a") == ~(a == "a")).all()
 445 
 446         assert not ("a" == a).all()
 447         assert (a == "a")[0]
 448         assert ("a" == a)[0]
 449         assert not ("a" != a)[0]
 450 
 451         # vs list-like
 452         assert (a == a).all()
 453         assert not (a != a).all()
 454 
 455         assert (a == list(a)).all()
 456         assert (a == b).all()
 457         assert (b == a).all()
 458         assert ((~(a == b)) == (a != b)).all()
 459         assert ((~(b == a)) == (b != a)).all()
 460 
 461         assert not (a == c).all()
 462         assert not (c == a).all()
 463         assert not (a == d).all()
 464         assert not (d == a).all()
 465 
 466         # vs a cat-like
 467         assert (a == e).all()
 468         assert (e == a).all()
 469         assert not (a == f).all()
 470         assert not (f == a).all()
 471 
 472         assert (~(a == e) == (a != e)).all()
 473         assert (~(e == a) == (e != a)).all()
 474         assert (~(a == f) == (a != f)).all()
 475         assert (~(f == a) == (f != a)).all()
 476 
 477         # non-equality is not comparable
 478         msg = "can only compare equality or not"
 479         with pytest.raises(TypeError, match=msg):
 480             a < b
 481         with pytest.raises(TypeError, match=msg):
 482             b < a
 483         with pytest.raises(TypeError, match=msg):
 484             a > b
 485         with pytest.raises(TypeError, match=msg):
 486             b > a
