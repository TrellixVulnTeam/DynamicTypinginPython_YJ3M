  56     def test_binop_maybe_preserve_name(self, datetime_series):
  57         # names match, preserve
  58         result = datetime_series * datetime_series
  59         assert result.name == datetime_series.name
  60         result = datetime_series.mul(datetime_series)
  61         assert result.name == datetime_series.name
  62 
  63         result = datetime_series * datetime_series[:-2]
  64         assert result.name == datetime_series.name
  65 
  66         # names don't match, don't preserve
  67         cp = datetime_series.copy()
  68         cp.name = "something else"
  69         result = datetime_series + cp
  70         assert result.name is None
  71         result = datetime_series.add(cp)
  72         assert result.name is None
  73 
  74         ops = ["add", "sub", "mul", "div", "truediv", "floordiv", "mod", "pow"]
  75         ops = ops + ["r" + op for op in ops]
  76         for op in ops:
  77             # names match, preserve
  78             s = datetime_series.copy()
  79             result = getattr(s, op)(s)
  80             assert result.name == datetime_series.name
  81 
  82             # names don't match, don't preserve
  83             cp = datetime_series.copy()
  84             cp.name = "changed"
  85             result = getattr(s, op)(cp)
  86             assert result.name is None

 325     def test_copy(self):
 326 
 327         for deep in [None, False, True]:
 328             s = Series(np.arange(10), dtype="float64")
 329 
 330             # default deep is True
 331             if deep is None:
 332                 s2 = s.copy()
 333             else:
 334                 s2 = s.copy(deep=deep)
 335 
 336             s2[::2] = np.NaN
 337 
 338             if deep is None or deep is True:
 339                 # Did not modify original Series
 340                 assert np.isnan(s2[0])
 341                 assert not np.isnan(s[0])
 342             else:
 343                 # we DID modify the original Series
 344                 assert np.isnan(s2[0])
 345                 assert np.isnan(s[0])

 390     def test_item(self):
 391         s = Series([1])
 392         result = s.item()
 393         assert result == 1
 394         assert result == s.iloc[0]
 395         assert isinstance(result, int)  # i.e. not np.int64
 396 
 397         ser = Series([0.5], index=[3])
 398         result = ser.item()
 399         assert isinstance(result, float)
 400         assert result == 0.5
 401 
 402         ser = Series([1, 2])
 403         msg = "can only convert an array of size 1"
 404         with pytest.raises(ValueError, match=msg):
 405             ser.item()
 406 
 407         dti = pd.date_range("2016-01-01", periods=2)
 408         with pytest.raises(ValueError, match=msg):
 409             dti.item()
 410         with pytest.raises(ValueError, match=msg):
 411             Series(dti).item()
 412 
 413         val = dti[:1].item()
 414         assert isinstance(val, Timestamp)
 415         val = Series(dti)[:1].item()
 416         assert isinstance(val, Timestamp)
 417 
 418         tdi = dti - dti
 419         with pytest.raises(ValueError, match=msg):
 420             tdi.item()
 421         with pytest.raises(ValueError, match=msg):
 422             Series(tdi).item()
 423 
 424         val = tdi[:1].item()
 425         assert isinstance(val, Timedelta)
 426         val = Series(tdi)[:1].item()
 427         assert isinstance(val, Timedelta)
 428 
 429         # Case where ser[0] would not work
 430         ser = Series(dti, index=[5, 6])
 431         val = ser[:1].item()
 432         assert val == dti[0]
