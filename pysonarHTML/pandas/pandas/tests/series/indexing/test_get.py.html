 139 def test_get_with_default():
 140     # GH#7725
 141     d0 = ["a", "b", "c", "d"]
 142     d1 = np.arange(4, dtype="int64")
 143     others = ["e", 10]
 144 
 145     for data, index in ((d0, d1), (d1, d0)):
 146         s = Series(data, index=index)
 147         for i, d in zip(index, data):
 148             assert s.get(i) == d
 149             assert s.get(i, d) == d
 150             assert s.get(i, "z") == d
 151             for other in others:
 152                 assert s.get(other, "z") == "z"
 153                 assert s.get(other, other) == other
