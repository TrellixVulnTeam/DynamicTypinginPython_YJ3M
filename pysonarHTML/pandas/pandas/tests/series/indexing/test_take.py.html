   8 def test_take():
   9     ser = Series([-1, 5, 6, 2, 4])
  10 
  11     actual = ser.take([1, 3, 4])
  12     expected = Series([5, 2, 4], index=[1, 3, 4])

  14 
  15     actual = ser.take([-1, 3, 4])

  17     tm.assert_series_equal(actual, expected)
  18 
  19     msg = "index {} is out of bounds for( axis 0 with)? size 5"
