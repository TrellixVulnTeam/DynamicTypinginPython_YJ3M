   9 def test_reindex(datetime_series, string_series):
  10     identity = string_series.reindex(string_series.index)
  11 
  12     # __array_interface__ is not defined for older numpies
  13     # and on some pythons
  14     try:
  15         assert np.may_share_memory(string_series.index, identity.index)
  16     except AttributeError:
  17         pass
  18 
  19     assert identity.index.is_(string_series.index)
  20     assert identity.index.identical(string_series.index)
  21 
  22     subIndex = string_series.index[10:20]
  23     subSeries = string_series.reindex(subIndex)
  24 
  25     for idx, val in subSeries.items():
  26         assert val == string_series[idx]
  27 
  28     subIndex2 = datetime_series.index[10:20]
  29     subTS = datetime_series.reindex(subIndex2)
  30 
  31     for idx, val in subTS.items():
  32         assert val == datetime_series[idx]
  33     stuffSeries = datetime_series.reindex(subIndex)
  34 
  35     assert np.isnan(stuffSeries).all()
  36 
  37     # This is extremely important for the Cython code to not screw up
  38     nonContigIndex = datetime_series.index[::2]
  39     subNonContig = datetime_series.reindex(nonContigIndex)
  40     for idx, val in subNonContig.items():
  41         assert val == datetime_series[idx]
  42 
  43     # return a copy the same index here
  44     result = datetime_series.reindex()
  45     assert not (result is datetime_series)

 108 def test_reindex_pad():
 109     s = Series(np.arange(10), dtype="int64")
 110     s2 = s[::2]
 111 
 112     reindexed = s2.reindex(s.index, method="pad")
 113     reindexed2 = s2.reindex(s.index, method="ffill")
 114     tm.assert_series_equal(reindexed, reindexed2)
 115 
 116     expected = Series([0, 0, 2, 2, 4, 4, 6, 6, 8, 8], index=np.arange(10))
 117     tm.assert_series_equal(reindexed, expected)
 118 
 119     # GH4604
 120     s = Series([1, 2, 3, 4, 5], index=["a", "b", "c", "d", "e"])
 121     new_index = ["a", "g", "c", "f"]
 122     expected = Series([1, 1, 3, 3], index=new_index)
 123 
 124     # this changes dtype because the ffill happens after
 125     result = s.reindex(new_index).ffill()
 126     tm.assert_series_equal(result, expected.astype("float64"))
 127 
 128     result = s.reindex(new_index).ffill(downcast="infer")
 129     tm.assert_series_equal(result, expected)
 130 
 131     expected = Series([1, 5, 3, 5], index=new_index)
 132     result = s.reindex(new_index, method="ffill")
 133     tm.assert_series_equal(result, expected)
 134 
 135     # inference of new dtype
 136     s = Series([True, False, False, True], index=list("abcd"))
 137     new_index = "agc"
 138     result = s.reindex(list(new_index)).ffill()
 139     expected = Series([True, True, False], index=list(new_index))
 140     tm.assert_series_equal(result, expected)
 141 
 142     # GH4618 shifted series downcasting
 143     s = Series(False, index=range(0, 5))
 144     result = s.shift(1).fillna(method="bfill")
 145     expected = Series(False, index=range(0, 5))
 146     tm.assert_series_equal(result, expected)

 149 def test_reindex_nearest():
 150     s = Series(np.arange(10, dtype="int64"))
 151     target = [0.1, 0.9, 1.5, 2.0]
 152     result = s.reindex(target, method="nearest")
 153     expected = Series(np.around(target).astype("int64"), target)
 154     tm.assert_series_equal(expected, result)
 155 
 156     result = s.reindex(target, method="nearest", tolerance=0.2)
 157     expected = Series([0, 1, np.nan, 2], target)
 158     tm.assert_series_equal(expected, result)
 159 
 160     result = s.reindex(target, method="nearest", tolerance=[0.3, 0.01, 0.4, 3])
 161     expected = Series([0, np.nan, np.nan, 2], target)
 162     tm.assert_series_equal(expected, result)

 208 def test_reindex_categorical():
 209     index = date_range("20000101", periods=3)
 210 
 211     # reindexing to an invalid Categorical
 212     s = Series(["a", "b", "c"], dtype="category")
 213     result = s.reindex(index)
 214     expected = Series(
 215         Categorical(values=[np.nan, np.nan, np.nan], categories=["a", "b", "c"])
 216     )
 217     expected.index = index
 218     tm.assert_series_equal(result, expected)
 219 
 220     # partial reindexing
 221     expected = Series(Categorical(values=["b", "c"], categories=["a", "b", "c"]))
 222     expected.index = [1, 2]
 223     result = s.reindex([1, 2])
 224     tm.assert_series_equal(result, expected)
 225 
 226     expected = Series(Categorical(values=["c", np.nan], categories=["a", "b", "c"]))
 227     expected.index = [2, 3]
 228     result = s.reindex([2, 3])
 229     tm.assert_series_equal(result, expected)

 232 def test_reindex_fill_value():
 233     # -----------------------------------------------------------
 234     # floats
 235     floats = Series([1.0, 2.0, 3.0])
 236     result = floats.reindex([1, 2, 3])
 237     expected = Series([2.0, 3.0, np.nan], index=[1, 2, 3])
 238     tm.assert_series_equal(result, expected)
 239 
 240     result = floats.reindex([1, 2, 3], fill_value=0)
 241     expected = Series([2.0, 3.0, 0], index=[1, 2, 3])
 242     tm.assert_series_equal(result, expected)
 243 
 244     # -----------------------------------------------------------
 245     # ints
 246     ints = Series([1, 2, 3])
 247 
 248     result = ints.reindex([1, 2, 3])
 249     expected = Series([2.0, 3.0, np.nan], index=[1, 2, 3])
 250     tm.assert_series_equal(result, expected)
 251 
 252     # don't upcast
 253     result = ints.reindex([1, 2, 3], fill_value=0)
 254     expected = Series([2, 3, 0], index=[1, 2, 3])
 255     assert issubclass(result.dtype.type, np.integer)
 256     tm.assert_series_equal(result, expected)
 257 
 258     # -----------------------------------------------------------
 259     # objects
 260     objects = Series([1, 2, 3], dtype=object)
 261 
 262     result = objects.reindex([1, 2, 3])
 263     expected = Series([2, 3, np.nan], index=[1, 2, 3], dtype=object)
 264     tm.assert_series_equal(result, expected)
 265 
 266     result = objects.reindex([1, 2, 3], fill_value="foo")
 267     expected = Series([2, 3, "foo"], index=[1, 2, 3], dtype=object)
 268     tm.assert_series_equal(result, expected)
 269 
 270     # ------------------------------------------------------------
 271     # bools
 272     bools = Series([True, False, True])
 273 
 274     result = bools.reindex([1, 2, 3])
 275     expected = Series([False, True, np.nan], index=[1, 2, 3], dtype=object)
 276     tm.assert_series_equal(result, expected)
 277 
 278     result = bools.reindex([1, 2, 3], fill_value=False)
 279     expected = Series([False, True, False], index=[1, 2, 3])
 280     tm.assert_series_equal(result, expected)
