  48 def test_loc_getitem_setitem_integer_slice_keyerrors():
  49     s = Series(np.random.randn(10), index=list(range(0, 20, 2)))
  50 
  51     # this is OK
  52     cp = s.copy()
  53     cp.iloc[4:10] = 0
  54     assert (cp.iloc[4:10] == 0).all()
  55 
  56     # so is this
  57     cp = s.copy()
  58     cp.iloc[3:11] = 0
  59     assert (cp.iloc[3:11] == 0).values.all()
  60 
  61     result = s.iloc[2:6]
  62     result2 = s.loc[3:11]
  63     expected = s.reindex([4, 6, 8, 10])
  64 
  65     tm.assert_series_equal(result, expected)
  66     tm.assert_series_equal(result2, expected)
  67 
  68     # non-monotonic, raise KeyError
  69     s2 = s.iloc[list(range(5)) + list(range(9, 4, -1))]
  70     with pytest.raises(KeyError, match=r"^3$"):
  71         s2.loc[3:11]
  72     with pytest.raises(KeyError, match=r"^3$"):
