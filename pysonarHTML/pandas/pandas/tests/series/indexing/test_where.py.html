 224

 226

 276 @pytest.mark.parametrize("size", range(2, 6))
 277 @pytest.mark.parametrize(
 278     "mask", [[True, False, False, False, False], [True, False], [False]]
 279 )
 280 @pytest.mark.parametrize(
 281     "item", [2.0, np.nan, np.finfo(np.float).max, np.finfo(np.float).min]
 282 )
 283 # Test numpy arrays, lists and tuples as the input to be
 284 # broadcast
 285 @pytest.mark.parametrize(
 286     "box", [lambda x: np.array([x]), lambda x: [x], lambda x: (x,)]
 287 )
 288 def test_broadcast(size, mask, item, box):
 289     selection = np.resize(mask, size)
 290 
 291     data = np.arange(size, dtype=float)
 292 
 293     # Construct the expected series by taking the source
 294     # data or item based on the selection
 295     expected = Series(
 296         [item if use_item else data[i] for i, use_item in enumerate(selection)]
 297     )
 298 
 299     s = Series(data)
 300     s[selection] = box(item)
 301     tm.assert_series_equal(s, expected)
 302 
 303     s = Series(data)
 304     result = s.where(~selection, box(item))
 305     tm.assert_series_equal(result, expected)
 306 
 307     s = Series(data)
 308     result = s.mask(selection, box(item))
 309     tm.assert_series_equal(result, expected)

 327 def test_where_dups():
 328     # GH 4550
 329     # where crashes with dups in index
 330     s1 = Series(list(range(3)))
 331     s2 = Series(list(range(3)))
 332     comb = pd.concat([s1, s2])
 333     result = comb.where(comb < 2)
 334     expected = Series([0, 1, np.nan, 0, 1, np.nan], index=[0, 1, 2, 0, 1, 2])
 335     tm.assert_series_equal(result, expected)
 336 
 337     # GH 4548

 342 
 343     comb[comb < 2] += 10
