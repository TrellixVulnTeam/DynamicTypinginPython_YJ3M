  51 def test_binary_ufunc_with_array(flip, sparse, ufunc, arrays_for_binary_ufunc):
  52     # Test that ufunc(Series(a), array) == Series(ufunc(a, b))
  53     a1, a2 = arrays_for_binary_ufunc
  54     if sparse:

  57

  59     series = pd.Series(a1, name=name)

  61

  63     series_args = (series, other)  # ufunc(series, array)

  66         array_args = reversed(array_args)
  67         series_args = reversed(series_args)  # ufunc(array, series)
  68 
  69     expected = pd.Series(ufunc(*array_args), name=name)
  70     result = ufunc(*series_args)

 102 @pytest.mark.parametrize("ufunc", BINARY_UFUNCS)
 103 @pytest.mark.parametrize("sparse", SPARSE, ids=SPARSE_IDS)
 104 @pytest.mark.parametrize("shuffle", [True, False], ids=["unaligned", "aligned"])
 105 @pytest.mark.parametrize("flip", [True, False], ids=["flipped", "straight"])
 106 def test_binary_ufunc_with_series(
 107     flip, shuffle, sparse, ufunc, arrays_for_binary_ufunc
 108 ):
 109     # Test that
 110     #   * func(Series(a), Series(b)) == Series(ufunc(a, b))
 111     #   with alignment between the indices
 112     a1, a2 = arrays_for_binary_ufunc
 113     if sparse:
 114         a1 = SparseArray(a1, dtype=pd.SparseDtype("int64", 0))
 115         a2 = SparseArray(a2, dtype=pd.SparseDtype("int64", 0))
 116 
 117     name = "name"  # op(Series, array) preserves the name.
 118     series = pd.Series(a1, name=name)
 119     other = pd.Series(a2, name=name)
 120 
 121     idx = np.random.permutation(len(a1))
 122 
 123     if shuffle:
 124         other = other.take(idx)
 125         if flip:
 126             index = other.align(series)[0].index
 127         else:
 128             index = series.align(other)[0].index
 129     else:
 130         index = series.index
 131 
 132     array_args = (a1, a2)
 133     series_args = (series, other)  # ufunc(series, array)
 134 
 135     if flip:
 136         array_args = tuple(reversed(array_args))
 137         series_args = tuple(reversed(series_args))  # ufunc(array, series)
 138 
 139     expected = pd.Series(ufunc(*array_args), index=index, name=name)
 140     result = ufunc(*series_args)
 141     tm.assert_series_equal(result, expected)

 170 @pytest.mark.parametrize("ufunc", [np.divmod])  # any others?
 171 @pytest.mark.parametrize("sparse", SPARSE, ids=SPARSE_IDS)
 172 @pytest.mark.parametrize("shuffle", SHUFFLE)
 173 @pytest.mark.filterwarnings("ignore:divide by zero:RuntimeWarning")
 174 def test_multiple_output_binary_ufuncs(ufunc, sparse, shuffle, arrays_for_binary_ufunc):
 175     # Test that
 176     #  the same conditions from binary_ufunc_scalar apply to
 177     #  ufuncs with multiple outputs.
 178     if sparse and ufunc is np.divmod:
 179         pytest.skip("sparse divmod not implemented.")
 180 
 181     a1, a2 = arrays_for_binary_ufunc
 182     # work around https://github.com/pandas-dev/pandas/issues/26987
 183     a1[a1 == 0] = 1
 184     a2[a2 == 0] = 1
 185 
 186     if sparse:
 187         a1 = SparseArray(a1, dtype=pd.SparseDtype("int64", 0))
 188         a2 = SparseArray(a2, dtype=pd.SparseDtype("int64", 0))
 189 
 190     s1 = pd.Series(a1)
 191     s2 = pd.Series(a2)
 192 
 193     if shuffle:
 194         # ensure we align before applying the ufunc
 195         s2 = s2.sample(frac=1)
 196 
 197     expected = ufunc(a1, a2)
 198     assert isinstance(expected, tuple)
 199 
 200     result = ufunc(s1, s2)
 201     assert isinstance(result, tuple)
 202     tm.assert_series_equal(result[0], pd.Series(expected[0]))
 203     tm.assert_series_equal(result[1], pd.Series(expected[1]))

 277 def test_object_dtype_ok():
 278     class Thing:
 279         def __init__(self, value):
 280             self.value = value
 281 
 282         def __add__(self, other):
 283             other = getattr(other, "value", other)
 284             return type(self)(self.value + other)
 285 
 286         def __eq__(self, other) -> bool:
 287             return type(other) is Thing and self.value == other.value
 288 
 289         def __repr__(self) -> str:
 290             return f"Thing({self.value})"
 291 
 292     s = pd.Series([Thing(1), Thing(2)])
 293     result = np.add(s, Thing(1))
 294     expected = pd.Series([Thing(2), Thing(3)])
 295     tm.assert_series_equal(result, expected)
