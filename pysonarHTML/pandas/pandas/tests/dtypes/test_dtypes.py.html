  42     def test_hash(self, dtype):
  43         hash(dtype)

  45     def test_equality_invalid(self, dtype):
  46         assert not dtype == "foo"
  47         assert not is_dtype_equal(dtype, np.int64)

  49     def test_numpy_informed(self, dtype):
  50         # npdev 2020-02-02 changed from "data type not understood" to
  51         #  "Cannot interpret 'foo' as a data type"
  52         msg = "|".join(
  53             ["data type not understood", "Cannot interpret '.*' as a data type"]
  54         )
  55         with pytest.raises(TypeError, match=msg):
  56             np.dtype(dtype)
  57 
  58         assert not dtype == np.str_
  59         assert not np.str_ == dtype
  60 
  61     def test_pickle(self, dtype):
  62         # make sure our cache is NOT pickled
  63 
  64         # clear the cache
  65         type(dtype).reset_cache()
  66         assert not len(dtype._cache)
  67 
  68         # force back to the cache
  69         result = tm.round_trip_pickle(dtype)
  70         if not isinstance(dtype, PeriodDtype):
  71             # Because PeriodDtype has a cython class as a base class,
  72             #  it has different pickle semantics, and its cache is re-populated
  73             #  on un-pickling.
  74             assert not len(dtype._cache)
  75         assert result == dtype
