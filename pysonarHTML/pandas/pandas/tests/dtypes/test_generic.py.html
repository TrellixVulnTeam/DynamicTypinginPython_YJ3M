  48 def test_setattr_warnings():
  49     # GH7175 - GOTCHA: You can't use dot notation to add a column...
  50     d = {
  51         "one": pd.Series([1.0, 2.0, 3.0], index=["a", "b", "c"]),
  52         "two": pd.Series([1.0, 2.0, 3.0, 4.0], index=["a", "b", "c", "d"]),
  53     }
  54     df = pd.DataFrame(d)
  55 
  56     with catch_warnings(record=True) as w:
  57         #  successfully add new column
  58         #  this should not raise a warning
  59         df["three"] = df.two + 1
  60         assert len(w) == 0
  61         assert df.three.sum() > df.two.sum()
  62 
  63     with catch_warnings(record=True) as w:
  64         #  successfully modify column in place
  65         #  this should not raise a warning
  66         df.one += 1
  67         assert len(w) == 0
  68         assert df.one.iloc[0] == 2
  69 
  70     with catch_warnings(record=True) as w:
  71         #  successfully add an attribute to a series
  72         #  this should not raise a warning
  73         df.two.not_an_index = [1, 2]
  74         assert len(w) == 0
  75 
  76     with tm.assert_produces_warning(UserWarning):
  77         #  warn when setting column to nonexistent name
  78         df.four = df.two + 2
  79         assert df.four.sum() > df.two.sum()
