  43 def test_upcast():
  44     # GH23823
  45     arr = np.arange(1, 6)
  46     mask = np.array([False, True, False, True, True])
  47     result, changed = maybe_upcast_putmask(arr, mask, other=np.nan)
  48 
  49     expected = np.array([1, np.nan, 3, np.nan, np.nan])
  50     assert changed
  51     tm.assert_numpy_array_equal(result, expected)

  54 def test_upcast_datetime():
