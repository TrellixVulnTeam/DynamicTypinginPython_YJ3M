  66 def _check_promote(dtype, fill_value, expected_dtype, exp_val_for_scalar=None):
  67     """
  68     Auxiliary function to unify testing of scalar/array promotion.
  69 
  70     Parameters
  71     ----------
  72     dtype : dtype
  73         The value to pass on as the first argument to maybe_promote.
  74     fill_value : scalar
  75         The value to pass on as the second argument to maybe_promote as
  76         a scalar.
  77     expected_dtype : dtype
  78         The expected dtype returned by maybe_promote (by design this is the
  79         same regardless of whether fill_value was passed as a scalar or in an
  80         array!).
  81     exp_val_for_scalar : scalar
  82         The expected value for the (potentially upcast) fill_value returned by
  83         maybe_promote.
  84     """
  85     assert is_scalar(fill_value)
  86 
  87     # here, we pass on fill_value as a scalar directly; the expected value
  88     # returned from maybe_promote is fill_value, potentially upcast to the
  89     # returned dtype.
  90     result_dtype, result_fill_value = maybe_promote(dtype, fill_value)
  91     expected_fill_value = exp_val_for_scalar
  92 
  93     assert result_dtype == expected_dtype
  94     _assert_match(result_fill_value, expected_fill_value)

  97 def _assert_match(result_fill_value, expected_fill_value):
  98     # GH#23982/25425 require the same type in addition to equality/NA-ness
  99     res_type = type(result_fill_value)
 100     ex_type = type(expected_fill_value)
 101 
 102     if hasattr(result_fill_value, "dtype"):
 103         # Compare types in a way that is robust to platform-specific
 104         #  idiosyncracies where e.g. sometimes we get "ulonglong" as an alias
 105         #  for "uint64" or "intc" as an alias for "int32"
 106         assert result_fill_value.dtype.kind == expected_fill_value.dtype.kind
 107         assert result_fill_value.dtype.itemsize == expected_fill_value.dtype.itemsize
 108     else:
 109         # On some builds, type comparison fails, e.g. np.int32 != np.int32
 110         assert res_type == ex_type or res_type.__name__ == ex_type.__name__
 111 
 112     match_value = result_fill_value == expected_fill_value
 113 
 114     # Note: type check above ensures that we have the _same_ NA value
 115     # for missing values, None == None (which is checked
 116     # through match_value above), but np.nan != np.nan and pd.NaT != pd.NaT
 117     match_missing = isna(result_fill_value) and isna(expected_fill_value)
 118 
 119     assert match_value or match_missing

 572 @pytest.mark.parametrize("fill_value", [None, np.nan, NaT])
 573 def test_maybe_promote_any_numpy_dtype_with_na(any_numpy_dtype_reduced, fill_value):
 574     dtype = np.dtype(any_numpy_dtype_reduced)
 575 
 576     if is_integer_dtype(dtype) and fill_value is not NaT:
 577         # integer + other missing value (np.nan / None) casts to float
 578         expected_dtype = np.float64
 579         exp_val_for_scalar = np.nan
 580     elif is_object_dtype(dtype) and fill_value is NaT:
 581         # inserting into object does not cast the value
 582         # but *does* cast None to np.nan
 583         expected_dtype = np.dtype(object)
 584         exp_val_for_scalar = fill_value
 585     elif is_datetime_or_timedelta_dtype(dtype):
 586         # datetime / timedelta cast all missing values to dtyped-NaT
 587         expected_dtype = dtype
 588         exp_val_for_scalar = dtype.type("NaT", "ns")
 589     elif fill_value is NaT:
 590         # NaT upcasts everything that's not datetime/timedelta to object
 591         expected_dtype = np.dtype(object)
 592         exp_val_for_scalar = NaT
 593     elif is_float_dtype(dtype) or is_complex_dtype(dtype):
 594         # float / complex + missing value (!= NaT) stays the same
 595         expected_dtype = dtype
 596         exp_val_for_scalar = np.nan
 597     else:
 598         # all other cases cast to object, and use np.nan as missing value
 599         expected_dtype = np.dtype(object)
 600         exp_val_for_scalar = np.nan
 601 
 602     _check_promote(dtype, fill_value, expected_dtype, exp_val_for_scalar)

 605 @pytest.mark.parametrize("dim", [0, 2, 3])
 606 def test_maybe_promote_dimensions(any_numpy_dtype_reduced, dim):
 607     dtype = np.dtype(any_numpy_dtype_reduced)
 608 
 609     # create 0-dim array of given dtype; casts "1" to correct dtype
 610     fill_array = np.array(1, dtype=dtype)
 611 
 612     # expand to desired dimension:
 613     for _ in range(dim):
 614         fill_array = np.expand_dims(fill_array, 0)
 615 
 616     if dtype != object:
 617         # test against 1-dimensional case
 618         with pytest.raises(ValueError, match="fill_value must be a scalar"):
 619             maybe_promote(dtype, np.array([1], dtype=dtype))
 620 
 621         with pytest.raises(ValueError, match="fill_value must be a scalar"):
 622             maybe_promote(dtype, fill_array)
 623 
 624     else:
 625         expected_dtype, expected_missing_value = maybe_promote(
 626             dtype, np.array([1], dtype=dtype)
 627         )
 628         result_dtype, result_missing_value = maybe_promote(dtype, fill_array)
 629         assert result_dtype == expected_dtype
 630         _assert_match(result_missing_value, expected_missing_value)
