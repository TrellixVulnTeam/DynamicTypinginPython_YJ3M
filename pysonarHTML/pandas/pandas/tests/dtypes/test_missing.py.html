 303 def test_array_equivalent():
 304     assert array_equivalent(np.array([np.nan, np.nan]), np.array([np.nan, np.nan]))
 305     assert array_equivalent(
 306         np.array([np.nan, 1, np.nan]), np.array([np.nan, 1, np.nan])
 307     )
 308     assert array_equivalent(
 309         np.array([np.nan, None], dtype="object"),
 310         np.array([np.nan, None], dtype="object"),
 311     )
 312     # Check the handling of nested arrays in array_equivalent_object
 313     assert array_equivalent(
 314         np.array([np.array([np.nan, None], dtype="object"), None], dtype="object"),
 315         np.array([np.array([np.nan, None], dtype="object"), None], dtype="object"),
 316     )
 317     assert array_equivalent(
 318         np.array([np.nan, 1 + 1j], dtype="complex"),
 319         np.array([np.nan, 1 + 1j], dtype="complex"),
 320     )
 321     assert not array_equivalent(
 322         np.array([np.nan, 1 + 1j], dtype="complex"),
 323         np.array([np.nan, 1 + 2j], dtype="complex"),
 324     )
 325     assert not array_equivalent(
 326         np.array([np.nan, 1, np.nan]), np.array([np.nan, 2, np.nan])
 327     )
 328     assert not array_equivalent(np.array(["a", "b", "c", "d"]), np.array(["e", "e"]))
 329     assert array_equivalent(Float64Index([0, np.nan]), Float64Index([0, np.nan]))
 330     assert not array_equivalent(Float64Index([0, np.nan]), Float64Index([1, np.nan]))
 331     assert array_equivalent(DatetimeIndex([0, np.nan]), DatetimeIndex([0, np.nan]))
 332     assert not array_equivalent(DatetimeIndex([0, np.nan]), DatetimeIndex([1, np.nan]))
 333     assert array_equivalent(TimedeltaIndex([0, np.nan]), TimedeltaIndex([0, np.nan]))
 334     assert not array_equivalent(
 335         TimedeltaIndex([0, np.nan]), TimedeltaIndex([1, np.nan])
 336     )
 337     assert array_equivalent(
 338         DatetimeIndex([0, np.nan], tz="US/Eastern"),
 339         DatetimeIndex([0, np.nan], tz="US/Eastern"),
 340     )
 341     assert not array_equivalent(
 342         DatetimeIndex([0, np.nan], tz="US/Eastern"),
 343         DatetimeIndex([1, np.nan], tz="US/Eastern"),
 344     )
 345     assert not array_equivalent(
 346         DatetimeIndex([0, np.nan]), DatetimeIndex([0, np.nan], tz="US/Eastern")
 347     )
 348     assert not array_equivalent(
 349         DatetimeIndex([0, np.nan], tz="CET"),
 350         DatetimeIndex([0, np.nan], tz="US/Eastern"),
 351     )
 352 
 353     assert not array_equivalent(DatetimeIndex([0, np.nan]), TimedeltaIndex([0, np.nan]))
