  50 def test_random_state():
  51     import numpy.random as npr
  52 
  53     # Check with seed
  54     state = com.random_state(5)
  55     assert state.uniform() == npr.RandomState(5).uniform()
  56 
  57     # Check with random state object
  58     state2 = npr.RandomState(10)
  59     assert com.random_state(state2).uniform() == npr.RandomState(10).uniform()
  60 
  61     # check with no arg random state
  62     assert com.random_state() is np.random
  63 
  64     # check array-like
  65     # GH32503
  66     state_arr_like = npr.randint(0, 2 ** 31, size=624, dtype="uint32")
  67     assert (
  68         com.random_state(state_arr_like).uniform()
  69         == npr.RandomState(state_arr_like).uniform()
  70     )
  71 
  72     # Check BitGenerators
  73     # GH32503
  74     if not _np_version_under1p17:
  75         assert (
  76             com.random_state(npr.MT19937(3)).uniform()
  77             == npr.RandomState(npr.MT19937(3)).uniform()
  78         )
  79         assert (
  80             com.random_state(npr.PCG64(11)).uniform()
  81             == npr.RandomState(npr.PCG64(11)).uniform()
  82         )
  83 
  84     # Error for floats or strings
  85     msg = (
  86         "random_state must be an integer, array-like, a BitGenerator, "
  87         "a numpy RandomState, or None"
  88     )
  89     with pytest.raises(ValueError, match=msg):
  90         com.random_state("test")
  91 
  92     with pytest.raises(ValueError, match=msg):
  93         com.random_state(5.5)

 120 def test_standardize_mapping():
 121     # No uninitialized defaultdicts
 122     msg = r"to_dict\(\) only accepts initialized defaultdicts"
 123     with pytest.raises(TypeError, match=msg):
 124         com.standardize_mapping(collections.defaultdict)
 125 
 126     # No non-mapping subtypes, instance
 127     msg = "unsupported type: <class 'list'>"
 128     with pytest.raises(TypeError, match=msg):
 129         com.standardize_mapping([])
 130 
 131     # No non-mapping subtypes, class
 132     with pytest.raises(TypeError, match=msg):
 133         com.standardize_mapping(list)
 134 
 135     fill = {"bad": "data"}
 136     assert com.standardize_mapping(fill) == dict
 137 
 138     # Convert instance to type
 139     assert com.standardize_mapping({}) == dict
 140 
 141     dd = collections.defaultdict(list)
 142     assert isinstance(com.standardize_mapping(dd), partial)
