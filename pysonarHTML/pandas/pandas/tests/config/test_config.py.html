 209     def test_validation(self):
 210         self.cf.register_option("a", 1, "doc", validator=self.cf.is_int)
 211         self.cf.register_option("d", 1, "doc", validator=self.cf.is_nonnegative_int)
 212         self.cf.register_option("b.c", "hullo", "doc2", validator=self.cf.is_text)
 213 
 214         msg = "Value must have type '<class 'int'>'"
 215         with pytest.raises(ValueError, match=msg):
 216             self.cf.register_option("a.b.c.d2", "NO", "doc", validator=self.cf.is_int)
 217 
 218         self.cf.set_option("a", 2)  # int is_int
 219         self.cf.set_option("b.c", "wurld")  # str is_str
 220         self.cf.set_option("d", 2)
 221         self.cf.set_option("d", None)  # non-negative int can be None
 222 
 223         # None not is_int
 224         with pytest.raises(ValueError, match=msg):
 225             self.cf.set_option("a", None)
 226         with pytest.raises(ValueError, match=msg):
 227             self.cf.set_option("a", "ab")
 228 
 229         msg = "Value must be a nonnegative integer or None"
 230         with pytest.raises(ValueError, match=msg):
 231             self.cf.register_option(
 232                 "a.b.c.d3", "NO", "doc", validator=self.cf.is_nonnegative_int
 233             )
 234         with pytest.raises(ValueError, match=msg):
 235             self.cf.register_option(
 236                 "a.b.c.d3", -2, "doc", validator=self.cf.is_nonnegative_int
 237             )
 238 
 239         msg = r"Value must be an instance of <class 'str'>\|<class 'bytes'>"
 240         with pytest.raises(ValueError, match=msg):
 241             self.cf.set_option("b.c", 1)
 242 
 243         validator = self.cf.is_one_of_factory([None, self.cf.is_callable])
 244         self.cf.register_option("b", lambda: None, "doc", validator=validator)
 245         self.cf.set_option("b", "%.1f".format)  # Formatter is callable
 246         self.cf.set_option("b", None)  # Formatter is none (default)
 247         with pytest.raises(ValueError, match="Value must be a callable"):
 248             self.cf.set_option("b", "%.1f")

 284         # we can deprecate non-existent options
 285         self.cf.deprecate_option("foo")
 286

 288         with warnings.catch_warnings(record=True) as w:
 289             warnings.simplefilter("always")

 293             assert "deprecated" in str(w[-1])  # we get the default message
 294 
 295         self.cf.register_option("a", 1, "doc", validator=self.cf.is_int)
 296         self.cf.register_option("b.c", "hullo", "doc2")
 297         self.cf.register_option("foo", "hullo", "doc2")
 298 
 299         self.cf.deprecate_option("a", removal_ver="nifty_ver")
 300         with warnings.catch_warnings(record=True) as w:
 301             warnings.simplefilter("always")
 302             self.cf.get_option("a")
 303 
 304             assert len(w) == 1  # should have raised one warning
 305             assert "eprecated" in str(w[-1])  # we get the default message
 306             assert "nifty_ver" in str(w[-1])  # with the removal_ver quoted
 307 
 308             msg = "Option 'a' has already been defined as deprecated"
 309             with pytest.raises(OptionError, match=msg):
 310                 self.cf.deprecate_option("a")
 311 
 312         self.cf.deprecate_option("b.c", "zounds!")
 313         with warnings.catch_warnings(record=True) as w:
 314             warnings.simplefilter("always")
 315             self.cf.get_option("b.c")
 316 
 317             assert len(w) == 1  # should have raised one warning
 318             assert "zounds!" in str(w[-1])  # we get the custom message
 319 
 320         # test rerouting keys
 321         self.cf.register_option("d.a", "foo", "doc2")
 322         self.cf.register_option("d.dep", "bar", "doc2")

 450     def test_option_context_scope(self):
 451         # Ensure that creating a context does not affect the existing
 452         # environment as it is supposed to be used with the `with` statement.
 453         # See https://github.com/pandas-dev/pandas/issues/8514
 454 
 455         original_value = 60
 456         context_value = 10
 457         option_name = "a"
 458 
 459         self.cf.register_option(option_name, original_value)
 460 
 461         # Ensure creating contexts didn't affect the current context.
 462         ctx = self.cf.option_context(option_name, context_value)
 463         assert self.cf.get_option(option_name) == original_value
 464 
 465         # Ensure the correct value is available inside the context.
 466         with ctx:
 467             assert self.cf.get_option(option_name) == context_value
 468 
 469         # Ensure the current context is reset
 470         assert self.cf.get_option(option_name) == original_value
