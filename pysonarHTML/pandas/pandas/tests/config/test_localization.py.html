  60 @_skip_if_only_one_locale
  61 @pytest.mark.parametrize(
  62     "lang,enc",
  63     [
  64         ("it_CH", "UTF-8"),
  65         ("en_US", "ascii"),
  66         ("zh_CN", "GB2312"),
  67         ("it_IT", "ISO-8859-1"),

  70 def test_set_locale(lang, enc):
  71     if all(x is None for x in _current_locale):
  72         # Not sure why, but on some Travis runs with pytest,
  73         #  getlocale() returned (None, None).
  74         pytest.skip("Current locale is not set.")
  75 
  76     enc = codecs.lookup(enc).name
  77     new_locale = lang, enc
  78 
  79     if not can_set_locale(new_locale):
  80         msg = "unsupported locale setting"
  81 
  82         with pytest.raises(locale.Error, match=msg):
  83             with set_locale(new_locale):
  84                 pass
  85     else:
  86         with set_locale(new_locale) as normalized_locale:
  87             new_lang, new_enc = normalized_locale.split(".")
  88             new_enc = codecs.lookup(enc).name
  89 
  90             normalized_locale = new_lang, new_enc
  91             assert normalized_locale == new_locale
  92 
  93     # Once we exit the "with" statement, locale should be back to what it was.
  94     current_locale = locale.getlocale()
  95     assert current_locale == _current_locale
