  17 def test_groupby_preserves_subclass(obj, groupby_func):
  18     # GH28330 -- preserve subclass through groupby operations
  19 
  20     if isinstance(obj, Series) and groupby_func in {"corrwith"}:
  21         pytest.skip("Not applicable")
  22 
  23     grouped = obj.groupby(np.arange(0, 10))
  24 
  25     # Groups should preserve subclass type
  26     assert isinstance(grouped.get_group(0), type(obj))
  27 
  28     args = []
  29     if groupby_func in {"fillna", "nth"}:
  30         args.append(0)
  31     elif groupby_func == "corrwith":
  32         args.append(obj)
  33     elif groupby_func == "tshift":
  34         args.extend([0, 0])
  35 
  36     result1 = getattr(grouped, groupby_func)(*args)
  37     result2 = grouped.agg(groupby_func, *args)
  38 
  39     # Reduction or transformation kernels should preserve type
  40     slices = {"ngroup", "cumcount", "size"}
  41     if isinstance(obj, DataFrame) and groupby_func in slices:
  42         assert isinstance(result1, obj._constructor_sliced)
  43     else:
  44         assert isinstance(result1, type(obj))
  45 
  46     # Confirm .agg() groupby operations return same results
  47     if isinstance(result1, DataFrame):
  48         tm.assert_frame_equal(result1, result2)
  49     else:
  50         tm.assert_series_equal(result1, result2)
