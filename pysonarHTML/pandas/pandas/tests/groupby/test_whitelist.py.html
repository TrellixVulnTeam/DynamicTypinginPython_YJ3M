 138 
 139     gb = obj.groupby(df.letters)
 140 
 141     f = getattr(type(gb), m)

 145         n = f.__name__

 149 
 150     # qualname
 151     try:

 154         return

 185 @pytest.mark.parametrize("op", AGG_FUNCTIONS)
 186 @pytest.mark.parametrize("level", [0, 1])
 187 @pytest.mark.parametrize("axis", [0, 1])
 188 @pytest.mark.parametrize("skipna", [True, False])
 189 @pytest.mark.parametrize("sort", [True, False])
 190 def test_regression_whitelist_methods(raw_frame, op, level, axis, skipna, sort):
 191     # GH6944
 192     # GH 17537
 193     # explicitly test the whitelist methods
 194 
 195     if axis == 0:
 196         frame = raw_frame
 197     else:
 198         frame = raw_frame.T
 199 
 200     if op in AGG_FUNCTIONS_WITH_SKIPNA:
 201         grouped = frame.groupby(level=level, axis=axis, sort=sort)
 202         result = getattr(grouped, op)(skipna=skipna)
 203         expected = getattr(frame, op)(level=level, axis=axis, skipna=skipna)
 204         if sort:
 205             expected = expected.sort_index(axis=axis, level=level)
 206         tm.assert_frame_equal(result, expected)
 207     else:
 208         grouped = frame.groupby(level=level, axis=axis, sort=sort)
 209         result = getattr(grouped, op)()
 210         expected = getattr(frame, op)(level=level, axis=axis)
 211         if sort:
 212             expected = expected.sort_index(axis=axis, level=level)
 213         tm.assert_frame_equal(result, expected)

 216 def test_groupby_blacklist(df_letters):
 217     df = df_letters
 218     s = df_letters.floats
 219 
 220     blacklist = [
 221         "eval",

 223         "abs",

 225         "mask",
 226         "align",
 227         "groupby",
 228         "clip",
 229         "astype",
 230         "at",
 231         "combine",
 232         "consolidate",
 233         "convert_objects",
 234     ]
 235     to_methods = [method for method in dir(df) if method.startswith("to_")]
 236 
 237     blacklist.extend(to_methods)
 238 
 239     for bl in blacklist:
 240         for obj in (df, s):
 241             gb = obj.groupby(df.letters)
 242 
 243             # e.g., to_csv
 244             defined_but_not_allowed = (
 245                 f"(?:^Cannot.+{repr(bl)}.+'{type(gb).__name__}'.+try "
 246                 f"using the 'apply' method$)"
 247             )
 248 
 249             # e.g., query, eval
 250             not_defined = (
 251                 f"(?:^'{type(gb).__name__}' object has no attribute {repr(bl)}$)"
 252             )
 253 
 254             msg = f"{defined_but_not_allowed}|{not_defined}"
 255 
 256             with pytest.raises(AttributeError, match=msg):
 257                 getattr(gb, bl)

 343 def test_groupby_selection_with_methods(df):
 344     # some methods which require DatetimeIndex
 345     rng = date_range("2014", periods=len(df))
 346     df.index = rng
 347 
 348     g = df.groupby(["A"])[["C"]]
 349     g_exp = df[["C"]].groupby(df["A"])
 350     # TODO check groupby with > 1 col ?
 351 
 352     # methods which are called as .foo()
 353     methods = [
 354         "count",
 355         "corr",
 356         "cummax",
 357         "cummin",
 358         "cumprod",
 359         "describe",
 360         "rank",
 361         "quantile",
 362         "diff",
 363         "shift",
 364         "all",
 365         "any",
 366         "idxmin",
 367         "idxmax",
 368         "ffill",
 369         "bfill",
 370         "pct_change",
 371         "tshift",
 372     ]
 373 
 374     for m in methods:
 375         res = getattr(g, m)()
 376         exp = getattr(g_exp, m)()
 377 
 378         # should always be frames!
 379         tm.assert_frame_equal(res, exp)
 380 
 381     # methods which aren't just .foo()
 382     tm.assert_frame_equal(g.fillna(0), g_exp.fillna(0))
 383     tm.assert_frame_equal(g.dtypes, g_exp.dtypes)
 384     tm.assert_frame_equal(g.apply(lambda x: x.sum()), g_exp.apply(lambda x: x.sum()))
 385 
 386     tm.assert_frame_equal(g.resample("D").mean(), g_exp.resample("D").mean())
 387     tm.assert_frame_equal(g.resample("D").ohlc(), g_exp.resample("D").ohlc())
 388 
 389     tm.assert_frame_equal(
 390         g.filter(lambda x: len(x) == 3), g_exp.filter(lambda x: len(x) == 3)
 391     )

 394 def test_all_methods_categorized(mframe):
 395     grp = mframe.groupby(mframe.iloc[:, 0])
 396     names = {_ for _ in dir(grp) if not _.startswith("_")} - set(mframe.columns)
 397     new_names = set(names)
 398     new_names -= reduction_kernels
 399     new_names -= transformation_kernels
 400     new_names -= groupby_other_methods
 401 
 402     assert not (reduction_kernels & transformation_kernels)
 403     assert not (reduction_kernels & groupby_other_methods)
 404     assert not (transformation_kernels & groupby_other_methods)
 405 
 406     # new public method?
 407     if new_names:
 408         msg = f"""
 409 There are uncatgeorized methods defined on the Grouper class:
 410 {new_names}.
 411 
 412 Was a new method recently added?
 413 
 414 Every public method On Grouper must appear in exactly one the
 415 following three lists defined in pandas.core.groupby.base:
 416 - `reduction_kernels`
 417 - `transformation_kernels`
 418 - `groupby_other_methods`
 419 see the comments in pandas/core/groupby/base.py for guidance on
 420 how to fix this test.
 421         """
 422         raise AssertionError(msg)
 423 
 424     # removed a public method?
 425     all_categorized = reduction_kernels | transformation_kernels | groupby_other_methods
 426     print(names)
 427     print(all_categorized)
 428     if not (names == all_categorized):
 429         msg = f"""
 430 Some methods which are supposed to be on the Grouper class
 431 are missing:
 432 {all_categorized - names}.
 433 
 434 They're still defined in one of the lists that live in pandas/core/groupby/base.py.
 435 If you removed a method, you should update them
 436 """
 437         raise AssertionError(msg)
