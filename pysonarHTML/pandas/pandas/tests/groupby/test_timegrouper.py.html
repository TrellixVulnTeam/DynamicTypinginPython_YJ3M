 337         # GH 6914
 338 
 339         df_original = DataFrame(
 340             {
 341                 "Buyer": "Carl Joe Joe Carl Joe Carl".split(),
 342                 "Quantity": [18, 3, 5, 1, 9, 3],
 343                 "Date": [
 344                     datetime(2013, 9, 1, 13, 0),
 345                     datetime(2013, 9, 1, 13, 5),
 346                     datetime(2013, 10, 1, 20, 0),
 347                     datetime(2013, 10, 3, 10, 0),
 348                     datetime(2013, 12, 2, 12, 0),
 349                     datetime(2013, 9, 2, 14, 0),
 350                 ],
 351             }
 352         )
 353         df_reordered = df_original.sort_values(by="Quantity")
 354 
 355         # single grouping
 356         expected_list = [
 357             df_original.iloc[[0, 1, 5]],
 358             df_original.iloc[[2, 3]],
 359             df_original.iloc[[4]],
 360         ]
 361         dt_list = ["2013-09-30", "2013-10-31", "2013-12-31"]
 362 
 363         for df in [df_original, df_reordered]:
 364             grouped = df.groupby(pd.Grouper(freq="M", key="Date"))
 365             for t, expected in zip(dt_list, expected_list):
 366                 dt = pd.Timestamp(t)
 367                 result = grouped.get_group(dt)
 368                 tm.assert_frame_equal(result, expected)
 369 
 370         # multiple grouping
 371         expected_list = [
 372             df_original.iloc[[1]],

 380             for (b, t), expected in zip(g_list, expected_list):
 381                 dt = pd.Timestamp(t)
 382                 result = grouped.get_group((b, dt))

 387         df_reordered = df_original.sort_values(by="Quantity")

 398                 dt = pd.Timestamp(t)

 416             result.reset_index(drop=True), expected.reset_index(drop=True)

 436     def test_groupby_groups_datetimeindex(self):

 439         ind = pd.date_range(start="2012/1/1", freq="5min", periods=periods)
 440         df = DataFrame(
 441             {"high": np.arange(periods), "low": np.arange(periods)}, index=ind
