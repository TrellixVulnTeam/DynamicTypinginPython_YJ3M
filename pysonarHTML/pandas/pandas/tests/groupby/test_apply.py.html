 148 def test_group_apply_once_per_group(df, group_names):
 149     # GH2936, GH7739, GH10519, GH2656, GH12155, GH20084, GH21417
 150 
 151     # This test should ensure that a function is only evaluated
 152     # once per group. Previously the function has been evaluated twice
 153     # on the first group to check if the Cython index slider is safe to use
 154     # This test ensures that the side effect (append to list) is only triggered
 155     # once per group
 156 
 157     names = []
 158     # cannot parameterize over the functions since they need external
 159     # `names` to detect side effects
 160 
 161     def f_copy(group):
 162         # this takes the fast apply path
 163         names.append(group.name)
 164         return group.copy()
 165 
 166     def f_nocopy(group):
 167         # this takes the slow apply path
 168         names.append(group.name)
 169         return group
 170 
 171     def f_scalar(group):
 172         # GH7739, GH2656
 173         names.append(group.name)
 174         return 0
 175 
 176     def f_none(group):
 177         # GH10519, GH12155, GH21417
 178         names.append(group.name)
 179         return None
 180 
 181     def f_constant_df(group):
 182         # GH2936, GH20084
 183         names.append(group.name)
 184         return DataFrame({"a": [1], "b": [1]})
 185 
 186     for func in [f_copy, f_nocopy, f_scalar, f_none, f_constant_df]:
 187         del names[:]
 188 
 189         df.groupby("a").apply(func)
 190         assert names == group_names
