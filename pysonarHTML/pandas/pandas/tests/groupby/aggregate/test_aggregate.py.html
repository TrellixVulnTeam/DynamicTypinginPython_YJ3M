 269 def test_more_flexible_frame_multi_function(df):
 270     grouped = df.groupby("A")
 271 
 272     exmean = grouped.agg({"C": np.mean, "D": np.mean})
 273     exstd = grouped.agg({"C": np.std, "D": np.std})
 274 
 275     expected = concat([exmean, exstd], keys=["mean", "std"], axis=1)
 276     expected = expected.swaplevel(0, 1, axis=1).sort_index(level=0, axis=1)
 277 
 278     d = {"C": [np.mean, np.std], "D": [np.mean, np.std]}
 279     result = grouped.aggregate(d)
 280 
 281     tm.assert_frame_equal(result, expected)
 282 
 283     # be careful
 284     result = grouped.aggregate({"C": np.mean, "D": [np.mean, np.std]})
 285     expected = grouped.aggregate({"C": np.mean, "D": [np.mean, np.std]})
 286     tm.assert_frame_equal(result, expected)
 287 
 288     def foo(x):
 289         return np.mean(x)
 290 
 291     def bar(x):
 292         return np.std(x, ddof=1)
 293 
 294     # this uses column selection & renaming
 295     msg = r"nested renamer is not supported"
 296     with pytest.raises(SpecificationError, match=msg):
 297         d = dict([["C", np.mean], ["D", dict([["foo", np.mean], ["bar", np.std]])]])
 298         grouped.aggregate(d)
 299 
 300     # But without renaming, these functions are OK
 301     d = {"C": [np.mean], "D": [foo, bar]}
 302     grouped.aggregate(d)
