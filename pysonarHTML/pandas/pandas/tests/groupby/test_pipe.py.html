   8 def test_pipe():
   9     # Test the pipe method of DataFrameGroupBy.
  10     # Issue #17871
  11 
  12     random_state = np.random.RandomState(1234567890)
  13 
  14     df = DataFrame(
  15         {
  16             "A": ["foo", "bar", "foo", "bar", "foo", "bar", "foo", "foo"],

  18             "C": random_state.randn(8),
  19         }
  20     )
  21 
  22     def f(dfgb):
  23         return dfgb.B.max() - dfgb.C.min().min()
