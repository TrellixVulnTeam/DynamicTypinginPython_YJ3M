   9 def test_filter_series():
  10     s = pd.Series([1, 3, 20, 5, 22, 24, 7])
  11     expected_odd = pd.Series([1, 3, 5, 7], index=[0, 1, 3, 6])
  12     expected_even = pd.Series([20, 22, 24], index=[2, 4, 5])
  13     grouper = s.apply(lambda x: x % 2)
  14     grouped = s.groupby(grouper)
  15     tm.assert_series_equal(grouped.filter(lambda x: x.mean() < 10), expected_odd)
  16     tm.assert_series_equal(grouped.filter(lambda x: x.mean() > 10), expected_even)
  17     # Test dropna=False.
  18     tm.assert_series_equal(
  19         grouped.filter(lambda x: x.mean() < 10, dropna=False),
  20         expected_odd.reindex(s.index),
  21     )
  22     tm.assert_series_equal(
  23         grouped.filter(lambda x: x.mean() > 10, dropna=False),
  24         expected_even.reindex(s.index),
  25     )

 170 def test_filter_against_workaround():
 171     np.random.seed(0)
 172     # Series of ints
 173     s = Series(np.random.randint(0, 100, 1000))
 174     grouper = s.apply(lambda x: np.round(x, -1))
 175     grouped = s.groupby(grouper)
 176     f = lambda x: x.mean() > 10
 177 
 178     old_way = s[grouped.transform(f).astype("bool")]
 179     new_way = grouped.filter(f)
 180     tm.assert_series_equal(new_way.sort_values(), old_way.sort_values())
 181 
 182     # Series of floats
 183     s = 100 * Series(np.random.random(1000))
 184     grouper = s.apply(lambda x: np.round(x, -1))
 185     grouped = s.groupby(grouper)
 186     f = lambda x: x.mean() > 10
 187     old_way = s[grouped.transform(f).astype("bool")]
 188     new_way = grouped.filter(f)
 189     tm.assert_series_equal(new_way.sort_values(), old_way.sort_values())
 190 
 191     # Set up DataFrame of ints, floats, strings.
 192     from string import ascii_lowercase
 193 
 194     letters = np.array(list(ascii_lowercase))
 195     N = 1000
 196     random_letters = letters.take(np.random.randint(0, 26, N))
 197     df = DataFrame(
 198         {
 199             "ints": Series(np.random.randint(0, 100, N)),
 200             "floats": N / 10 * Series(np.random.random(N)),
 201             "letters": Series(random_letters),
 202         }
 203     )
 204 
 205     # Group by ints; filter on floats.
 206     grouped = df.groupby("ints")
 207     old_way = df[grouped.floats.transform(lambda x: x.mean() > N / 20).astype("bool")]
 208     new_way = grouped.filter(lambda x: x["floats"].mean() > N / 20)
 209     tm.assert_frame_equal(new_way, old_way)
 210 
 211     # Group by floats (rounded); filter on strings.
 212     grouper = df.floats.apply(lambda x: np.round(x, -1))
 213     grouped = df.groupby(grouper)
 214     old_way = df[grouped.letters.transform(lambda x: len(x) < N / 10).astype("bool")]
 215     new_way = grouped.filter(lambda x: len(x.letters) < N / 10)
 216     tm.assert_frame_equal(new_way, old_way)
 217 
 218     # Group by strings; filter on ints.
 219     grouped = df.groupby("letters")
 220     old_way = df[grouped.ints.transform(lambda x: x.mean() > N / 20).astype("bool")]
 221     new_way = grouped.filter(lambda x: x["ints"].mean() > N / 20)
 222     tm.assert_frame_equal(new_way, old_way)

 252 def test_filter_maintains_ordering():
 253     # Simple case: index is sequential. #4621
 254     df = DataFrame(

 256     )

 259     actual = grouped.filter(lambda x: len(x) > 1)
 260     expected = df.iloc[[1, 2, 4, 7]]

 262

 264     actual = grouped.filter(lambda x: len(x) > 1)
 265     expected = s.iloc[[1, 2, 4, 7]]
 266     tm.assert_series_equal(actual, expected)
 267

 270     s = df["pid"]
 271     grouped = df.groupby("tag")
 272     actual = grouped.filter(lambda x: len(x) > 1)
 273     expected = df.iloc[[1, 2, 4, 7]]
 274     tm.assert_frame_equal(actual, expected)
 275 
 276     grouped = s.groupby(df["tag"])
 277     actual = grouped.filter(lambda x: len(x) > 1)
 278     expected = s.iloc[[1, 2, 4, 7]]
 279     tm.assert_series_equal(actual, expected)
 280 
 281     # Index is shuffled.
 282     SHUFFLED = [4, 6, 7, 2, 1, 0, 5, 3]
 283     df.index = df.index[SHUFFLED]
 284     s = df["pid"]
 285     grouped = df.groupby("tag")
 286     actual = grouped.filter(lambda x: len(x) > 1)
 287     expected = df.iloc[[1, 2, 4, 7]]
