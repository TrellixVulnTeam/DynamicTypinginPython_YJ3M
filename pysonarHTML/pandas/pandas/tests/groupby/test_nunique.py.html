  59 def test_nunique():
  60     df = DataFrame({"A": list("abbacc"), "B": list("abxacc"), "C": list("abbacx")})
  61 
  62     expected = DataFrame({"A": list("abc"), "B": [1, 2, 1], "C": [1, 1, 2]})
  63     result = df.groupby("A", as_index=False).nunique()
  64     tm.assert_frame_equal(result, expected)
  65 
  66     # as_index
  67     expected.index = list("abc")
  68     expected.index.name = "A"
  69     expected = expected.drop(columns="A")
  70     result = df.groupby("A").nunique()
  71     tm.assert_frame_equal(result, expected)
  72 
  73     # with na
  74     result = df.replace({"x": None}).groupby("A").nunique(dropna=False)
  75     tm.assert_frame_equal(result, expected)
  76 
  77     # dropna
  78     expected = DataFrame({"B": [1] * 3, "C": [1] * 3}, index=list("abc"))
  79     expected.index.name = "A"
  80     result = df.replace({"x": None}).groupby("A").nunique()
  81     tm.assert_frame_equal(result, expected)
