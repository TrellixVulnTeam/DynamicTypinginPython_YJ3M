  10 def test_rank_apply():
  11     lev1 = tm.rands_array(10, 100)
  12     lev2 = tm.rands_array(10, 130)
  13     lab1 = np.random.randint(0, 100, size=500)
  14     lab2 = np.random.randint(0, 130, size=500)
  15 
  16     df = DataFrame(
  17         {
  18             "value": np.random.randn(500),
  19             "key1": lev1.take(lab1),
  20             "key2": lev2.take(lab2),
  21         }
  22     )
  23 
  24     result = df.groupby(["key1", "key2"]).value.rank()
  25 
  26     expected = [piece.value.rank() for key, piece in df.groupby(["key1", "key2"])]
  27     expected = concat(expected, axis=0)
  28     expected = expected.reindex(result.index)
  29     tm.assert_series_equal(result, expected)
  30 
  31     result = df.groupby(["key1", "key2"]).value.rank(pct=True)
  32 
  33     expected = [
  34         piece.value.rank(pct=True) for key, piece in df.groupby(["key1", "key2"])
  35     ]
