  33 @pytest.mark.parametrize("q", [0, 0.25, 0.5, 0.75, 1])
  34 def test_quantile(interpolation, a_vals, b_vals, q):
  35     if interpolation == "nearest" and q == 0.5 and b_vals == [4, 3, 2, 1]:
  36         pytest.skip(
  37             "Unclear numpy expectation for nearest result with equidistant data"
  38         )
  39 
  40     a_expected = pd.Series(a_vals).quantile(q, interpolation=interpolation)
  41     b_expected = pd.Series(b_vals).quantile(q, interpolation=interpolation)
  42 
  43     df = DataFrame(
  44         {"key": ["a"] * len(a_vals) + ["b"] * len(b_vals), "val": a_vals + b_vals}
  45     )
  46 
  47     expected = DataFrame(
  48         [a_expected, b_expected], columns=["val"], index=Index(["a", "b"], name="key")
  49     )
  50     result = df.groupby("key").quantile(q, interpolation=interpolation)
  51 
  52     tm.assert_frame_equal(result, expected)

 124 @pytest.mark.parametrize("frame_size", [(2, 3), (100, 10)])
 125 @pytest.mark.parametrize("groupby", [[0], [0, 1]])
 126 @pytest.mark.parametrize("q", [[0.5, 0.6]])
 127 def test_groupby_quantile_with_arraylike_q_and_int_columns(frame_size, groupby, q):
 128     # GH30289
 129     nrow, ncol = frame_size
 130     df = pd.DataFrame(
 131         np.array([ncol * [_ % 4] for _ in range(nrow)]), columns=range(ncol)
 132     )
 133 
 134     idx_levels = [list(range(min(nrow, 4)))] * len(groupby) + [q]
 135     idx_codes = [[x for x in range(min(nrow, 4)) for _ in q]] * len(groupby) + [
 136         list(range(len(q))) * min(nrow, 4)
 137     ]
 138     expected_index = pd.MultiIndex(
 139         levels=idx_levels, codes=idx_codes, names=groupby + [None]
 140     )
 141     expected_values = [
 142         [float(x)] * (ncol - len(groupby)) for x in range(min(nrow, 4)) for _ in q
 143     ]
 144     expected_columns = [x for x in range(ncol) if x not in groupby]
 145     expected = pd.DataFrame(
 146         expected_values, index=expected_index, columns=expected_columns
 147     )
 148     result = df.groupby(groupby).quantile(q)
 149 
 150     tm.assert_frame_equal(result, expected)

 220 @pytest.mark.parametrize("q", [0.5, [0.0, 0.5, 1.0]])
 221 def test_groupby_quantile_nullable_array(values, q):
 222     # https://github.com/pandas-dev/pandas/issues/33136
 223     df = pd.DataFrame({"a": ["x"] * 3 + ["y"] * 3, "b": values})
 224     result = df.groupby("a")["b"].quantile(q)
 225 
 226     if isinstance(q, list):
 227         idx = pd.MultiIndex.from_product((["x", "y"], q), names=["a", None])
 228         true_quantiles = [0.0, 0.5, 1.0]
 229     else:
 230         idx = pd.Index(["x", "y"], name="a")
 231         true_quantiles = [0.5]
 232 
 233     expected = pd.Series(true_quantiles * 2, index=idx, name="b")
 234     tm.assert_series_equal(result, expected)
