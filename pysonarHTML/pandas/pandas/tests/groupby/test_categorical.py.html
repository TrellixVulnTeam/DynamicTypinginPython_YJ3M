  22 def cartesian_product_for_groupers(result, args, names):
  23     """ Reindex to a cartesian production for the groupers,
  24     preserving the nature (Categorical) of each grouper
  25     """
  26 
  27     def f(a):
  28         if isinstance(a, (CategoricalIndex, Categorical)):
  29             categories = a.categories
  30             a = Categorical.from_codes(
  31                 np.arange(len(categories)), categories=categories, ordered=a.ordered
  32             )
  33         return a
  34 
  35     index = MultiIndex.from_product(map(f, args), names=names)
  36     return result.reindex(index).sort_index()
