   8 @pytest.mark.parametrize("by", ["A", "B", ["A", "B"]])
   9 def test_size(df, by):
  10     grouped = df.groupby(by=by)
  11     result = grouped.size()
  12     for key, group in grouped:
  13         assert result[key] == len(group)

  16 @pytest.mark.parametrize("by", ["A", "B", ["A", "B"]])
  17 @pytest.mark.parametrize("sort", [True, False])
  18 def test_size_sort(df, sort, by):
  19     df = DataFrame(np.random.choice(20, (1000, 3)), columns=list("ABC"))
  20     left = df.groupby(by=by, sort=sort).size()
  21     right = df.groupby(by=by, sort=sort)["C"].apply(lambda a: a.shape[0])
  22     tm.assert_series_equal(left, right, check_names=False)
