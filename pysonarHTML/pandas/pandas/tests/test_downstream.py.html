 144 def test_missing_required_dependency():
 145     # GH 23868
 146     # To ensure proper isolation, we pass these flags
 147     # -S : disable site-packages
 148     # -s : disable user site-packages
 149     # -E : disable PYTHON* env vars, especially PYTHONPATH
 150     # https://github.com/MacPython/pandas-wheels/pull/50
 151 
 152     pyexe = sys.executable.replace("\\", "/")
 153     call = [pyexe, "-sSE", "-c", "import pandas"]
 154 
 155     msg = (
 156         rf"Command '\['{pyexe}', '-sSE', '-c', 'import pandas'\]' "
 157         "returned non-zero exit status 1."
 158     )
 159 
 160     with pytest.raises(subprocess.CalledProcessError, match=msg) as exc:
 161         subprocess.check_output(call, stderr=subprocess.STDOUT)
 162 
 163     output = exc.value.stdout.decode()
 164     for name in ["numpy", "pytz", "dateutil"]:
 165         assert name in output
