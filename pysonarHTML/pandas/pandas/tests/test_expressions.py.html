 261     def test_bool_ops_raise_on_arithmetic(self, op_str, opname):
 262         df = DataFrame({"a": np.random.rand(10) > 0.5, "b": np.random.rand(10) > 0.5})
 263 
 264         msg = f"operator {repr(op_str)} not implemented for bool dtypes"
 265         f = getattr(operator, opname)
 266         err_msg = re.escape(msg)
 267 
 268         with pytest.raises(NotImplementedError, match=err_msg):
 269             f(df, df)
 270 
 271         with pytest.raises(NotImplementedError, match=err_msg):
 272             f(df.a, df.b)
 273 
 274         with pytest.raises(NotImplementedError, match=err_msg):
 275             f(df.a, True)
 276 
 277         with pytest.raises(NotImplementedError, match=err_msg):
 278             f(False, df.a)
 279 
 280         with pytest.raises(NotImplementedError, match=err_msg):
 281             f(False, df)
 282 
 283         with pytest.raises(NotImplementedError, match=err_msg):
 284             f(df, True)

 286     @pytest.mark.parametrize(
 287         "op_str,opname", [("+", "add"), ("*", "mul"), ("-", "sub")]
 288     )
 289     def test_bool_ops_warn_on_arithmetic(self, op_str, opname):
 290         n = 10
 291         df = DataFrame({"a": np.random.rand(n) > 0.5, "b": np.random.rand(n) > 0.5})
 292 
 293         subs = {"+": "|", "*": "&", "-": "^"}
 294         sub_funcs = {"|": "or_", "&": "and_", "^": "xor"}
 295 
 296         f = getattr(operator, opname)
 297         fe = getattr(operator, sub_funcs[subs[op_str]])
 298 
 299         if op_str == "-":
 300             # raises TypeError
 301             return
 302 
 303         with tm.use_numexpr(True, min_elements=5):
 304             with tm.assert_produces_warning(check_stacklevel=False):
 305                 r = f(df, df)
 306                 e = fe(df, df)
 307                 tm.assert_frame_equal(r, e)
 308 
 309             with tm.assert_produces_warning(check_stacklevel=False):
 310                 r = f(df.a, df.b)
 311                 e = fe(df.a, df.b)
 312                 tm.assert_series_equal(r, e)
 313 
 314             with tm.assert_produces_warning(check_stacklevel=False):
 315                 r = f(df.a, True)
 316                 e = fe(df.a, True)
 317                 tm.assert_series_equal(r, e)
 318 
 319             with tm.assert_produces_warning(check_stacklevel=False):
 320                 r = f(False, df.a)
 321                 e = fe(False, df.a)
 322                 tm.assert_series_equal(r, e)
 323 
 324             with tm.assert_produces_warning(check_stacklevel=False):
 325                 r = f(False, df)
 326                 e = fe(False, df)
 327                 tm.assert_frame_equal(r, e)
 328 
 329             with tm.assert_produces_warning(check_stacklevel=False):
 330                 r = f(df, True)
 331                 e = fe(df, True)
 332                 tm.assert_frame_equal(r, e)
