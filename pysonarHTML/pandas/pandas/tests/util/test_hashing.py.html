  48 def _check_not_equal_with_index(obj):
  49     """
  50     Check the hash of an object with and without its index is not the same.
  51 
  52     Parameters
  53     ----------
  54     obj : object
  55         The object to hash.
  56     """
  57     if not isinstance(obj, Index):
  58         a = hash_pandas_object(obj, index=True)
  59         b = hash_pandas_object(obj, index=False)
  60 
  61         if len(obj):
  62             assert not (a == b).all()

 201     # Check that categoricals hash consistent with their values,

 209     h3 = hash_pandas_object(s3, categorize=categorize)
 210 
 211     tm.assert_series_equal(h1, h2)
 212     tm.assert_series_equal(h1, h3)

 299 def test_hash_with_tuple():
 300     # GH#28969 array containing a tuple raises on call to arr.astype(str)
 301     #  apparently a numpy bug github.com/numpy/numpy/issues/9441
 302 
 303     df = pd.DataFrame({"data": [tuple("1"), tuple("2")]})
 304     result = hash_pandas_object(df)
 305     expected = pd.Series([10345501319357378243, 8331063931016360761], dtype=np.uint64)
 306     tm.assert_series_equal(result, expected)
 307 
 308     df2 = pd.DataFrame({"data": [tuple([1]), tuple([2])]})
 309     result = hash_pandas_object(df2)
 310     expected = pd.Series([9408946347443669104, 3278256261030523334], dtype=np.uint64)
 311     tm.assert_series_equal(result, expected)
 312 
 313     # require that the elements of such tuples are themselves hashable
 314 
 315     df3 = pd.DataFrame({"data": [tuple([1, []]), tuple([2, {}])]})
 316     with pytest.raises(TypeError, match="unhashable type: 'list'"):
 317         hash_pandas_object(df3)
