 158 def test_numpy_array_equal_copy_flag(other_type, check_same):
 159     a = np.array([1, 2, 3])
 160     msg = None
 161 
 162     if other_type == "same":
 163         other = a.view()
 164     else:
 165         other = a.copy()
 166 
 167     if check_same != other_type:
 168         msg = (
 169             r"array\(\[1, 2, 3\]\) is not array\(\[1, 2, 3\]\)"
 170             if check_same == "same"
 171             else r"array\(\[1, 2, 3\]\) is array\(\[1, 2, 3\]\)"
 172         )
 173 
 174     if msg is not None:
 175         with pytest.raises(AssertionError, match=msg):
 176             tm.assert_numpy_array_equal(a, other, check_same=check_same)
 177     else:
 178         tm.assert_numpy_array_equal(a, other, check_same=check_same)
