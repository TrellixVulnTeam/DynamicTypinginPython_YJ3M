 238         for depr in deprecated_list:
 239             with tm.assert_produces_warning(FutureWarning):
 240                 deprecated = getattr(pd, depr)
 241                 if not compat.PY37:
 242                     if depr == "datetime":
 243                         deprecated.__getattr__(dir(pd.datetime.datetime)[-1])

 299         # avoid cache state affecting the test
