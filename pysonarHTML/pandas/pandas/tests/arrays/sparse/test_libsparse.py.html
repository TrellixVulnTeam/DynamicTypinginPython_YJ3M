 223     def test_intersect(self):
 224         def _check_correct(a, b, expected):
 225             result = a.intersect(b)
 226             assert result.equals(expected)
 227 
 228         def _check_length_exc(a, longer):
 229             msg = "Indices must reference same underlying length"
 230             with pytest.raises(Exception, match=msg):
 231                 a.intersect(longer)
 232 
 233         def _check_case(xloc, xlen, yloc, ylen, eloc, elen):
 234             xindex = BlockIndex(TEST_LENGTH, xloc, xlen)
 235             yindex = BlockIndex(TEST_LENGTH, yloc, ylen)
 236             expected = BlockIndex(TEST_LENGTH, eloc, elen)
 237             longer_index = BlockIndex(TEST_LENGTH + 1, yloc, ylen)
 238 
 239             _check_correct(xindex, yindex, expected)
 240             _check_correct(
 241                 xindex.to_int_index(), yindex.to_int_index(), expected.to_int_index()
 242             )
 243 
 244             _check_length_exc(xindex, longer_index)
 245             _check_length_exc(xindex.to_int_index(), longer_index.to_int_index())
 246 
 247         check_cases(_check_case)
