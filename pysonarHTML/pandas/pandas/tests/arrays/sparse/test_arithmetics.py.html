  32     def _check_numeric_ops(self, a, b, a_dense, b_dense, mix, op):
  33         with np.errstate(invalid="ignore", divide="ignore"):
  34             if mix:
  35                 result = op(a, b_dense).to_dense()
  36             else:
  37                 result = op(a, b).to_dense()
  38 
  39             if op in [operator.truediv, ops.rtruediv]:
  40                 # pandas uses future division
  41                 expected = op(a_dense * 1.0, b_dense)
  42             else:
  43                 expected = op(a_dense, b_dense)
  44 
  45             if op in [operator.floordiv, ops.rfloordiv]:
  46                 # Series sets 1//0 to np.inf, which SparseArray does not do (yet)
  47                 mask = np.isinf(expected)
  48                 if mask.any():
  49                     expected[mask] = np.nan
  50 
  51             self._assert(result, expected)
  52 
  53     def _check_bool_result(self, res):
  54         assert isinstance(res, self._klass)
  55         assert isinstance(res.dtype, SparseDtype)
  56         assert res.dtype.subtype == np.bool
  57         assert isinstance(res.fill_value, bool)

  59     def _check_comparison_ops(self, a, b, a_dense, b_dense):
  60         with np.errstate(invalid="ignore"):
  61             # Unfortunately, trying to wrap the computation of each expected
  62             # value is with np.errstate() is too tedious.
  63             #
  64             # sparse & sparse
  65             self._check_bool_result(a == b)
  66             self._assert((a == b).to_dense(), a_dense == b_dense)
  67 
  68             self._check_bool_result(a != b)
  69             self._assert((a != b).to_dense(), a_dense != b_dense)
  70 
  71             self._check_bool_result(a >= b)
  72             self._assert((a >= b).to_dense(), a_dense >= b_dense)
  73 
  74             self._check_bool_result(a <= b)
  75             self._assert((a <= b).to_dense(), a_dense <= b_dense)
  76 
  77             self._check_bool_result(a > b)
  78             self._assert((a > b).to_dense(), a_dense > b_dense)
  79 
  80             self._check_bool_result(a < b)
  81             self._assert((a < b).to_dense(), a_dense < b_dense)
  82 
  83             # sparse & dense
  84             self._check_bool_result(a == b_dense)
  85             self._assert((a == b_dense).to_dense(), a_dense == b_dense)
  86 
  87             self._check_bool_result(a != b_dense)
  88             self._assert((a != b_dense).to_dense(), a_dense != b_dense)
  89 
  90             self._check_bool_result(a >= b_dense)
  91             self._assert((a >= b_dense).to_dense(), a_dense >= b_dense)
  92 
  93             self._check_bool_result(a <= b_dense)
  94             self._assert((a <= b_dense).to_dense(), a_dense <= b_dense)
  95 
  96             self._check_bool_result(a > b_dense)
  97             self._assert((a > b_dense).to_dense(), a_dense > b_dense)
  98 
  99             self._check_bool_result(a < b_dense)
 100             self._assert((a < b_dense).to_dense(), a_dense < b_dense)

 102     def _check_logical_ops(self, a, b, a_dense, b_dense):
 103         # sparse & sparse
 104         self._check_bool_result(a & b)
 105         self._assert((a & b).to_dense(), a_dense & b_dense)
 106 
 107         self._check_bool_result(a | b)
 108         self._assert((a | b).to_dense(), a_dense | b_dense)
 109         # sparse & dense
 110         self._check_bool_result(a & b_dense)
 111         self._assert((a & b_dense).to_dense(), a_dense & b_dense)
 112 
 113         self._check_bool_result(a | b_dense)
 114         self._assert((a | b_dense).to_dense(), a_dense | b_dense)
