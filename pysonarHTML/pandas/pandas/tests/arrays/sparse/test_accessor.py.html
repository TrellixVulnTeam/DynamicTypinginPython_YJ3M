  28     @pytest.mark.parametrize("format", ["csc", "csr", "coo"])
  29     @pytest.mark.parametrize("labels", [None, list(string.ascii_letters[:10])])

  31     @td.skip_if_no_scipy
  32     def test_from_spmatrix(self, format, labels, dtype):
  33         import scipy.sparse
  34 
  35         sp_dtype = SparseDtype(dtype, np.array(0, dtype=dtype).item())
  36 
  37         mat = scipy.sparse.eye(10, format=format, dtype=dtype)
  38         result = pd.DataFrame.sparse.from_spmatrix(mat, index=labels, columns=labels)
  39         expected = pd.DataFrame(
  40             np.eye(10, dtype=dtype), index=labels, columns=labels
  41         ).astype(sp_dtype)
  42         tm.assert_frame_equal(result, expected)

 106     @pytest.mark.parametrize("dtype", ["int64", "float64"])
 107     @pytest.mark.parametrize("dense_index", [True, False])
 108     @td.skip_if_no_scipy
 109     def test_series_from_coo(self, dtype, dense_index):
 110         import scipy.sparse
 111 
 112         A = scipy.sparse.eye(3, format="coo", dtype=dtype)
 113         result = pd.Series.sparse.from_coo(A, dense_index=dense_index)
 114         index = pd.MultiIndex.from_tuples([(0, 0), (1, 1), (2, 2)])
 115         expected = pd.Series(SparseArray(np.array([1, 1, 1], dtype=dtype)), index=index)
 116         if dense_index:
 117             expected = expected.reindex(pd.MultiIndex.from_product(index.levels))
 118 
 119         tm.assert_series_equal(result, expected)
