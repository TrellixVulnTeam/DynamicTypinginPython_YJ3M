 162 def test_coerce_to_array():
 163     # TODO this is currently not public API
 164     values = np.array([True, False, True, False], dtype="bool")
 165     mask = np.array([False, False, False, True], dtype="bool")
 166     result = BooleanArray(*coerce_to_array(values, mask=mask))
 167     expected = BooleanArray(values, mask)
 168     tm.assert_extension_array_equal(result, expected)
 169     assert result._data is values
 170     assert result._mask is mask
 171     result = BooleanArray(*coerce_to_array(values, mask=mask, copy=True))
 172     expected = BooleanArray(values, mask)
 173     tm.assert_extension_array_equal(result, expected)
 174     assert result._data is not values
 175     assert result._mask is not mask
 176 
 177     # mixed missing from values and mask
 178     values = [True, False, None, False]
 179     mask = np.array([False, False, False, True], dtype="bool")
 180     result = BooleanArray(*coerce_to_array(values, mask=mask))
 181     expected = BooleanArray(
 182         np.array([True, False, True, True]), np.array([False, False, True, True])
 183     )
 184     tm.assert_extension_array_equal(result, expected)
 185     result = BooleanArray(*coerce_to_array(np.array(values, dtype=object), mask=mask))
 186     tm.assert_extension_array_equal(result, expected)
 187     result = BooleanArray(*coerce_to_array(values, mask=mask.tolist()))
 188     tm.assert_extension_array_equal(result, expected)
 189 
 190     # raise errors for wrong dimension
 191     values = np.array([True, False, True, False], dtype="bool")
 192     mask = np.array([False, False, False, True], dtype="bool")
 193 
 194     with pytest.raises(ValueError, match="values must be a 1D list-like"):
 195         coerce_to_array(values.reshape(1, -1))
 196 
 197     with pytest.raises(ValueError, match="mask must be a 1D list-like"):
 198         coerce_to_array(values, mask=mask.reshape(1, -1))

 201 def test_coerce_to_array_from_boolean_array():
 202     # passing BooleanArray to coerce_to_array
 203     values = np.array([True, False, True, False], dtype="bool")
 204     mask = np.array([False, False, False, True], dtype="bool")
 205     arr = BooleanArray(values, mask)
 206     result = BooleanArray(*coerce_to_array(arr))
 207     tm.assert_extension_array_equal(result, arr)
 208     # no copy
 209     assert result._data is arr._data
 210     assert result._mask is arr._mask
 211 
 212     result = BooleanArray(*coerce_to_array(arr), copy=True)
 213     tm.assert_extension_array_equal(result, arr)
 214     assert result._data is not arr._data
 215     assert result._mask is not arr._mask
 216 
 217     with pytest.raises(ValueError, match="cannot pass mask for BooleanArray input"):
 218         coerce_to_array(arr, mask=mask)

 221 def test_coerce_to_numpy_array():
 222     # with missing values -> object dtype
 223     arr = pd.array([True, False, None], dtype="boolean")
 224     result = np.array(arr)
 225     expected = np.array([True, False, pd.NA], dtype="object")
 226     tm.assert_numpy_array_equal(result, expected)
 227 
 228     # also with no missing values -> object dtype
 229     arr = pd.array([True, False, True], dtype="boolean")
 230     result = np.array(arr)
 231     expected = np.array([True, False, True], dtype="object")
 232     tm.assert_numpy_array_equal(result, expected)
 233 
 234     # force bool dtype
 235     result = np.array(arr, dtype="bool")
 236     expected = np.array([True, False, True], dtype="bool")
 237     tm.assert_numpy_array_equal(result, expected)
 238     # with missing values will raise error
 239     arr = pd.array([True, False, None], dtype="boolean")
 240     msg = (
 241         "cannot convert to 'bool'-dtype NumPy array with missing values. "
 242         "Specify an appropriate 'na_value' for this dtype."
 243     )
 244     with pytest.raises(ValueError, match=msg):
 245         np.array(arr, dtype="bool")

 264 @pytest.mark.parametrize("box", [True, False], ids=["series", "array"])
 265 def test_to_numpy(box):
 266     con = pd.Series if box else pd.array
 267     # default (with or without missing values) -> object dtype
 268     arr = con([True, False, True], dtype="boolean")
 269     result = arr.to_numpy()
 270     expected = np.array([True, False, True], dtype="object")
 271     tm.assert_numpy_array_equal(result, expected)
 272 
 273     arr = con([True, False, None], dtype="boolean")
 274     result = arr.to_numpy()
 275     expected = np.array([True, False, pd.NA], dtype="object")
 276     tm.assert_numpy_array_equal(result, expected)
 277 
 278     arr = con([True, False, None], dtype="boolean")
 279     result = arr.to_numpy(dtype="str")
 280     expected = np.array([True, False, pd.NA], dtype="<U5")
 281     tm.assert_numpy_array_equal(result, expected)
 282 
 283     # no missing values -> can convert to bool, otherwise raises
 284     arr = con([True, False, True], dtype="boolean")
 285     result = arr.to_numpy(dtype="bool")
 286     expected = np.array([True, False, True], dtype="bool")
 287     tm.assert_numpy_array_equal(result, expected)
 288 
 289     arr = con([True, False, None], dtype="boolean")
 290     with pytest.raises(ValueError, match="cannot convert to 'bool'-dtype"):
 291         result = arr.to_numpy(dtype="bool")
 292 
 293     # specify dtype and na_value
 294     arr = con([True, False, None], dtype="boolean")
 295     result = arr.to_numpy(dtype=object, na_value=None)
 296     expected = np.array([True, False, None], dtype="object")
 297     tm.assert_numpy_array_equal(result, expected)
 298 
 299     result = arr.to_numpy(dtype=bool, na_value=False)
 300     expected = np.array([True, False, False], dtype="bool")
 301     tm.assert_numpy_array_equal(result, expected)
 302 
 303     result = arr.to_numpy(dtype="int64", na_value=-99)
 304     expected = np.array([1, 0, -99], dtype="int64")
 305     tm.assert_numpy_array_equal(result, expected)
 306 
 307     result = arr.to_numpy(dtype="float64", na_value=np.nan)
 308     expected = np.array([1, 0, np.nan], dtype="float64")
 309     tm.assert_numpy_array_equal(result, expected)
 310 
 311     # converting to int or float without specifying na_value raises
 312     with pytest.raises(ValueError, match="cannot convert to 'int64'-dtype"):
 313         arr.to_numpy(dtype="int64")
 314     with pytest.raises(ValueError, match="cannot convert to 'float64'-dtype"):
 315         arr.to_numpy(dtype="float64")

 318 def test_to_numpy_copy():
 319     # to_numpy can be zero-copy if no missing values
 320     arr = pd.array([True, False, True], dtype="boolean")
 321     result = arr.to_numpy(dtype=bool)
 322     result[0] = False
 323     tm.assert_extension_array_equal(
 324         arr, pd.array([False, False, True], dtype="boolean")
 325     )
 326 
 327     arr = pd.array([True, False, True], dtype="boolean")
 328     result = arr.to_numpy(dtype=bool, copy=True)
 329     result[0] = False
 330     tm.assert_extension_array_equal(arr, pd.array([True, False, True], dtype="boolean"))
