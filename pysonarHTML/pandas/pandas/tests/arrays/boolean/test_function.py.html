  19 def test_ufuncs_binary(ufunc):
  20     # two BooleanArrays
  21     a = pd.array([True, False, None], dtype="boolean")
  22     result = ufunc(a, a)
  23     expected = pd.array(ufunc(a._data, a._data), dtype="boolean")
  24     expected[a._mask] = np.nan
  25     tm.assert_extension_array_equal(result, expected)
  26 
  27     s = pd.Series(a)
  28     result = ufunc(s, a)
  29     expected = pd.Series(ufunc(a._data, a._data), dtype="boolean")
  30     expected[a._mask] = np.nan
  31     tm.assert_series_equal(result, expected)
  32 
  33     # Boolean with numpy array
