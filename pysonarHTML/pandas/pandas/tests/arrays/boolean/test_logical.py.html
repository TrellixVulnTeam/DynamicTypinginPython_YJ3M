  20     def get_op_from_name(self, op_name):
  21         short_opname = op_name.strip("_")
  22         short_opname = short_opname if "xor" in short_opname else short_opname + "_"
  23         try:
  24             op = getattr(operator, short_opname)
  25         except AttributeError:
  26             # Assume it is the reverse operator
  27             rop = getattr(operator, short_opname[1:])
  28             op = lambda x, y: rop(y, x)
  29 
  30         return op

  74     def test_kleene_or(self):
  75         # A clear test of behavior.
  76         a = pd.array([True] * 3 + [False] * 3 + [None] * 3, dtype="boolean")
  77         b = pd.array([True, False, None] * 3, dtype="boolean")
  78         result = a | b
  79         expected = pd.array(
  80             [True, True, True, True, False, None, True, None, None], dtype="boolean"

 208     @pytest.mark.parametrize(
 209         "other", [True, False, pd.NA, [True, False, None] * 3],
 210     )
 211     def test_no_masked_assumptions(self, other, all_logical_operators):
 212         # The logical operations should not assume that masked values are False!
 213         a = pd.arrays.BooleanArray(
 214             np.array([True, True, True, False, False, False, True, False, True]),
 215             np.array([False] * 6 + [True, True, True]),
 216         )
 217         b = pd.array([True] * 3 + [False] * 3 + [None] * 3, dtype="boolean")
 218         if isinstance(other, list):
 219             other = pd.array(other, dtype="boolean")
 220 
 221         result = getattr(a, all_logical_operators)(other)
 222         expected = getattr(b, all_logical_operators)(other)
 223         tm.assert_extension_array_equal(result, expected)
 224 
 225         if isinstance(other, BooleanArray):
 226             other._data[other._mask] = True
 227             a._data[a._mask] = False
 228 
 229             result = getattr(a, all_logical_operators)(other)
 230             expected = getattr(b, all_logical_operators)(other)
 231             tm.assert_extension_array_equal(result, expected)
