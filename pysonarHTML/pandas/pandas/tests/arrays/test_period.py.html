 359 def test_arrow_array(data, freq):
 360     import pyarrow as pa
 361     from pandas.core.arrays._arrow_utils import ArrowPeriodType
 362

 364     result = pa.array(periods)
 365     assert isinstance(result.type, ArrowPeriodType)

 369

 371     result = pa.array(periods, type=pa.int64())

 374     # unsupported conversions
 375     msg = "Not supported to convert PeriodArray to 'double' type"
 376     with pytest.raises(TypeError, match=msg):
 377         pa.array(periods, type="float64")

 379     with pytest.raises(TypeError, match="different 'freq'"):
 380         pa.array(periods, type=ArrowPeriodType("T"))
