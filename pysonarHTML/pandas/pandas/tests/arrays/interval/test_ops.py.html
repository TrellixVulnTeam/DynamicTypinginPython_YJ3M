  34 class TestOverlaps:
  35     def test_overlaps_interval(self, constructor, start_shift, closed, other_closed):
  36         start, shift = start_shift
  37         interval = Interval(start, start + 3 * shift, other_closed)
  38 
  39         # intervals: identical, nested, spanning, partial, adjacent, disjoint
  40         tuples = [
  41             (start, start + 3 * shift),
  42             (start + shift, start + 2 * shift),
  43             (start - shift, start + 4 * shift),
  44             (start + 2 * shift, start + 4 * shift),
  45             (start + 3 * shift, start + 4 * shift),
  46             (start + 4 * shift, start + 5 * shift),
  47         ]
  48         interval_container = constructor.from_tuples(tuples, closed)
  49 
  50         adjacent = interval.closed_right and interval_container.closed_left
  51         expected = np.array([True, True, True, True, adjacent, False])
  52         result = interval_container.overlaps(interval)
  53         tm.assert_numpy_array_equal(result, expected)
