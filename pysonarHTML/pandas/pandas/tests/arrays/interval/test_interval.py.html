 159 def test_arrow_array():
 160     import pyarrow as pa
 161     from pandas.core.arrays._arrow_utils import ArrowIntervalType
 162 
 163     intervals = pd.interval_range(1, 5, freq=1).array
 164 
 165     result = pa.array(intervals)
 166     assert isinstance(result.type, ArrowIntervalType)
 167     assert result.type.closed == intervals.closed
 168     assert result.type.subtype == pa.int64()
 169     assert result.storage.field("left").equals(pa.array([1, 2, 3, 4], type="int64"))
 170     assert result.storage.field("right").equals(pa.array([2, 3, 4, 5], type="int64"))
 171 
 172     expected = pa.array([{"left": i, "right": i + 1} for i in range(1, 5)])
 173     assert result.storage.equals(expected)
 174 
 175     # convert to its storage type
 176     result = pa.array(intervals, type=expected.type)
 177     assert result.equals(expected)
 178 
 179     # unsupported conversions
 180     with pytest.raises(TypeError, match="Not supported to convert IntervalArray"):
 181         pa.array(intervals, type="float64")
 182 
 183     with pytest.raises(TypeError, match="different 'subtype'"):

 223     import pyarrow as pa
 224     from pandas.core.arrays._arrow_utils import ArrowIntervalType
 225

 227     arr[1] = None
 228     df = pd.DataFrame({"a": arr})
 229 
 230     table = pa.table(df)
 231     assert isinstance(table.field("a").type, ArrowIntervalType)
 232     result = table.to_pandas()
 233     assert isinstance(result["a"].dtype, pd.IntervalDtype)
 234     tm.assert_frame_equal(result, df)
 235 
 236     table2 = pa.concat_tables([table, table])
 237     result = table2.to_pandas()
 238     expected = pd.concat([df, df], ignore_index=True)
