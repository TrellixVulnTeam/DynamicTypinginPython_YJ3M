  87         other = pd.Categorical(arr1d, ordered=ordered)
  88         if as_index:
  89             other = pd.CategoricalIndex(other)
  90 
  91         left, right = arr1d, other
  92         if reverse:
  93             left, right = right, left
  94 
  95         ones = np.ones(arr1d.shape, dtype=bool)
  96         zeros = ~ones
  97

  99         tm.assert_numpy_array_equal(result, ones)
