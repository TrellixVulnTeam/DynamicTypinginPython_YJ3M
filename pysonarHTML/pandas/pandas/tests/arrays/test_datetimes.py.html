 134     def test_cmp_dt64_arraylike_tznaive(self, all_compare_operators):
 135         # arbitrary tz-naive DatetimeIndex
 136         opname = all_compare_operators.strip("_")
 137         op = getattr(operator, opname)
 138 
 139         dti = pd.date_range("2016-01-1", freq="MS", periods=9, tz=None)
 140         arr = DatetimeArray(dti)
 141         assert arr.freq == dti.freq
 142         assert arr.tz == dti.tz
 143 
 144         right = dti
 145 
 146         expected = np.ones(len(arr), dtype=bool)
 147         if opname in ["ne", "gt", "lt"]:
 148             # for these the comparisons should be all-False
 149             expected = ~expected
 150 
 151         result = op(arr, arr)
 152         tm.assert_numpy_array_equal(result, expected)
 153         for other in [right, np.array(right)]:
 154             # TODO: add list and tuple, and object-dtype once those
 155             #  are fixed in the constructor
 156             result = op(arr, other)
 157             tm.assert_numpy_array_equal(result, expected)
 158 
 159             result = op(other, arr)
 160             tm.assert_numpy_array_equal(result, expected)
