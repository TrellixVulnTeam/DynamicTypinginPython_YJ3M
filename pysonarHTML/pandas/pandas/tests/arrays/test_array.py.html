 151     a = np.array([1, 2])
 152     # default is to copy
 153     b = pd.array(a, dtype=a.dtype)
 154     assert np.shares_memory(a, b._ndarray) is False

 157     b = pd.array(a, dtype=a.dtype, copy=True)

 276 @register_extension_dtype
 277 class DecimalDtype2(DecimalDtype):
 278     name = "decimal2"
 279 
 280     @classmethod
 281     def construct_array_type(cls):
 282         """
 283         Return the array type associated with this dtype.
 284 
 285         Returns
 286         -------
 287         type
 288         """
 289         return DecimalArray2
 290 
 291 
 292 class DecimalArray2(DecimalArray):
 293     @classmethod
 294     def _from_sequence(cls, scalars, dtype=None, copy=False):
 295         if isinstance(scalars, (pd.Series, pd.Index)):
 296             raise TypeError("scalars should not be of type pd.Series or pd.Index")
 297 
 298         return super()._from_sequence(scalars, dtype=dtype, copy=copy)

 301 def test_array_unboxes(index_or_series):
 302     box = index_or_series
 303 
 304     data = box([decimal.Decimal("1"), decimal.Decimal("2")])
 305     # make sure it works
 306     with pytest.raises(
 307         TypeError, match="scalars should not be of type pd.Series or pd.Index"
 308     ):
 309         DecimalArray2._from_sequence(data)
 310 
 311     result = pd.array(data, dtype="decimal2")
 312     expected = DecimalArray2._from_sequence(data.values)
 313     tm.assert_equal(result, expected)

 334 class TestArrayAnalytics:
 335     def test_searchsorted(self, string_dtype):
 336         arr = pd.array(["a", "b", "c"], dtype=string_dtype)
 337 
 338         result = arr.searchsorted("a", side="left")

 346     def test_searchsorted_numeric_dtypes_scalar(self, any_real_dtype):
 347         arr = pd.array([1, 3, 90], dtype=any_real_dtype)
 348         result = arr.searchsorted(30)
 349         assert is_scalar(result)
 350         assert result == 2
 351 
 352         result = arr.searchsorted([30])
 353         expected = np.array([2], dtype=np.intp)
 354         tm.assert_numpy_array_equal(result, expected)

 356     def test_searchsorted_numeric_dtypes_vector(self, any_real_dtype):
 357         arr = pd.array([1, 3, 90], dtype=any_real_dtype)
 358         result = arr.searchsorted([2, 30])
 359         expected = np.array([1, 2], dtype=np.intp)
 360         tm.assert_numpy_array_equal(result, expected)
