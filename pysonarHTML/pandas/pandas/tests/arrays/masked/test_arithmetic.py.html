 130 def test_error_invalid_object(data, all_arithmetic_operators):
 131     data, _ = data
 132 
 133     op = all_arithmetic_operators
 134     opa = getattr(data, op)
 135 
 136     # 2d -> return NotImplemented
 137     result = opa(pd.DataFrame({"A": data}))
 138     assert result is NotImplemented
 139 
 140     msg = r"can only perform ops with 1-d structures"
 141     with pytest.raises(NotImplementedError, match=msg):
 142         opa(np.arange(len(data)).reshape(-1, len(data)))

 145 def test_error_len_mismatch(data, all_arithmetic_operators):
 146     # operating with a list-like with non-matching length raises
 147     data, scalar = data
 148     op = tm.get_op_from_name(all_arithmetic_operators)
 149 
 150     other = [scalar] * (len(data) - 1)
 151 
 152     for other in [other, np.array(other)]:
 153         with pytest.raises(ValueError, match="Lengths must match"):
 154             op(data, other)
 155 
 156         s = pd.Series(data)
 157         with pytest.raises(ValueError, match="Lengths must match"):
 158             op(s, other)
