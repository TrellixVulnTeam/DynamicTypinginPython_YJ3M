 145     def test_searchsorted(self, ordered):
 146         # https://github.com/pandas-dev/pandas/issues/8420
 147         # https://github.com/pandas-dev/pandas/issues/14522
 148 
 149         cat = Categorical(
 150             ["cheese", "milk", "apple", "bread", "bread"],
 151             categories=["cheese", "milk", "apple", "bread"],
 152             ordered=ordered,
 153         )
 154         ser = Series(cat)
 155 
 156         # Searching for single item argument, side='left' (default)
 157         res_cat = cat.searchsorted("apple")
 158         assert res_cat == 2
 159         assert is_scalar(res_cat)
 160 
 161         res_ser = ser.searchsorted("apple")
 162         assert res_ser == 2
 163         assert is_scalar(res_ser)
 164 
 165         # Searching for single item array, side='left' (default)
 166         res_cat = cat.searchsorted(["bread"])
 167         res_ser = ser.searchsorted(["bread"])
 168         exp = np.array([3], dtype=np.intp)
 169         tm.assert_numpy_array_equal(res_cat, exp)
 170         tm.assert_numpy_array_equal(res_ser, exp)
 171 
 172         # Searching for several items array, side='right'
 173         res_cat = cat.searchsorted(["apple", "bread"], side="right")
 174         res_ser = ser.searchsorted(["apple", "bread"], side="right")
 175         exp = np.array([3, 5], dtype=np.intp)
 176         tm.assert_numpy_array_equal(res_cat, exp)
 177         tm.assert_numpy_array_equal(res_ser, exp)
 178 
 179         # Searching for a single value that is not from the Categorical
 180         with pytest.raises(KeyError, match="cucumber"):
 181             cat.searchsorted("cucumber")
 182         with pytest.raises(KeyError, match="cucumber"):
 183             ser.searchsorted("cucumber")
 184 
 185         # Searching for multiple values one of each is not from the Categorical
 186         with pytest.raises(KeyError, match="cucumber"):
 187             cat.searchsorted(["bread", "cucumber"])
 188         with pytest.raises(KeyError, match="cucumber"):
 189             ser.searchsorted(["bread", "cucumber"])

 191     def test_unique(self):
 192         # categories are reordered based on value when ordered=False
 193         cat = Categorical(["a", "b"])
 194         exp = Index(["a", "b"])
 195         res = cat.unique()
 196         tm.assert_index_equal(res.categories, exp)
 197         tm.assert_categorical_equal(res, cat)
 198 
 199         cat = Categorical(["a", "b", "a", "a"], categories=["a", "b", "c"])
 200         res = cat.unique()
 201         tm.assert_index_equal(res.categories, exp)
 202         tm.assert_categorical_equal(res, Categorical(exp))
 203 
 204         cat = Categorical(["c", "a", "b", "a", "a"], categories=["a", "b", "c"])
 205         exp = Index(["c", "a", "b"])
 206         res = cat.unique()
 207         tm.assert_index_equal(res.categories, exp)
 208         exp_cat = Categorical(exp, categories=["c", "a", "b"])
 209         tm.assert_categorical_equal(res, exp_cat)
 210 
 211         # nan must be removed
 212         cat = Categorical(["b", np.nan, "b", np.nan, "a"], categories=["a", "b", "c"])
 213         res = cat.unique()
 214         exp = Index(["b", "a"])
 215         tm.assert_index_equal(res.categories, exp)
 216         exp_cat = Categorical(["b", np.nan, "a"], categories=["b", "a"])
 217         tm.assert_categorical_equal(res, exp_cat)

 245     def test_unique_index_series(self):
 246         c = Categorical([3, 1, 2, 2, 1], categories=[3, 2, 1])
 247         # Categorical.unique sorts categories by appearance order
 248         # if ordered=False
 249         exp = Categorical([3, 1, 2], categories=[3, 1, 2])
 250         tm.assert_categorical_equal(c.unique(), exp)
 251

 253         tm.assert_categorical_equal(Series(c).unique(), exp)
 254 
 255         c = Categorical([1, 1, 2, 2], categories=[3, 2, 1])
 256         exp = Categorical([1, 2], categories=[1, 2])

 259         tm.assert_categorical_equal(Series(c).unique(), exp)
 260 
 261         c = Categorical([3, 1, 2, 2, 1], categories=[3, 2, 1], ordered=True)
 262         # Categorical.unique keeps categories order if ordered=True

 265

 269     def test_shift(self):
 270         # GH 9416
 271         cat = Categorical(["a", "b", "c", "d", "a"])
 272 
 273         # shift forward
 274         sp1 = cat.shift(1)
 275         xp1 = Categorical([np.nan, "a", "b", "c", "d"])
 276         tm.assert_categorical_equal(sp1, xp1)
 277         tm.assert_categorical_equal(cat[:-1], sp1[1:])
 278 
 279         # shift back
 280         sn2 = cat.shift(-2)
 281         xp2 = Categorical(
 282             ["c", "d", "a", np.nan, np.nan], categories=["a", "b", "c", "d"]
 283         )
 284         tm.assert_categorical_equal(sn2, xp2)
 285         tm.assert_categorical_equal(cat[2:], sn2[:-2])
 286 
 287         # shift by zero
 288         tm.assert_categorical_equal(cat, cat.shift(0))

 295     def test_memory_usage(self):
 296         cat = Categorical([1, 2, 3])
 297 
 298         # .categories is an index, so we include the hashtable
 299         assert 0 < cat.nbytes <= cat.memory_usage()
 300         assert 0 < cat.nbytes <= cat.memory_usage(deep=True)
 301 
 302         cat = Categorical(["foo", "foo", "bar"])
 303         assert cat.memory_usage(deep=True) > cat.nbytes
 304 
 305         if not PYPY:
 306             # sys.getsizeof will call the .memory_usage with
 307             # deep=True, and add on some GC overhead
 308             diff = cat.memory_usage(deep=True) - sys.getsizeof(cat)
 309             assert abs(diff) < 100
