   1 from pandas import Categorical
   2 import pandas._testing as tm
   3 
   4 
   5 class TestCategoricalSubclassing:
   6     def test_constructor(self):
   7         sc = tm.SubclassedCategorical(["a", "b", "c"])
   8         assert isinstance(sc, tm.SubclassedCategorical)
   9         tm.assert_categorical_equal(sc, Categorical(["a", "b", "c"]))
  10 
  11     def test_from_codes(self):
  12         sc = tm.SubclassedCategorical.from_codes([1, 0, 2], ["a", "b", "c"])
  13         assert isinstance(sc, tm.SubclassedCategorical)
  14         exp = Categorical.from_codes([1, 0, 2], ["a", "b", "c"])
  15         tm.assert_categorical_equal(sc, exp)
  16 
  17     def test_map(self):
  18         sc = tm.SubclassedCategorical(["a", "b", "c"])
  19         res = sc.map(lambda x: x.upper())
  20         assert isinstance(res, tm.SubclassedCategorical)
  21         exp = Categorical(["A", "B", "C"])
  22         tm.assert_categorical_equal(res, exp)
