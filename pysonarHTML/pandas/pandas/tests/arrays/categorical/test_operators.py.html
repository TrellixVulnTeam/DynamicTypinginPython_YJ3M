 338     def test_numeric_like_ops(self):
 339 
 340         df = DataFrame({"value": np.random.randint(0, 10000, 100)})
 341         labels = [f"{i} - {i + 499}" for i in range(0, 10000, 500)]
 342         cat_labels = Categorical(labels, labels)
 343 
 344         df = df.sort_values(by=["value"], ascending=True)
 345         df["value_group"] = pd.cut(
 346             df.value, range(0, 10500, 500), right=False, labels=cat_labels
 347         )
 348 
 349         # numeric ops should not succeed
 350         for op, str_rep in [
 351             ("__add__", r"\+"),
 352             ("__sub__", "-"),
 353             ("__mul__", r"\*"),
 354             ("__truediv__", "/"),
 355         ]:
 356             msg = f"Series cannot perform the operation {str_rep}|unsupported operand"
 357             with pytest.raises(TypeError, match=msg):
 358                 getattr(df, op)(df)
 359 
 360         # reduction ops should not succeed (unless specifically defined, e.g.
 361         # min/max)
 362         s = df["value_group"]
 363         for op in ["kurt", "skew", "var", "std", "mean", "sum", "median"]:
 364             msg = f"Categorical cannot perform the operation {op}"
 365             with pytest.raises(TypeError, match=msg):
 366                 getattr(s, op)(numeric_only=False)
 367 
 368         # mad technically works because it takes always the numeric data
 369 
 370         # numpy ops
 371         s = Series(Categorical([1, 2, 3, 4]))
 372         with pytest.raises(
 373             TypeError, match="Categorical cannot perform the operation sum"
 374         ):
 375             np.sum(s)
 376 
 377         # numeric ops on a Series
 378         for op, str_rep in [
 379             ("__add__", r"\+"),
 380             ("__sub__", "-"),
 381             ("__mul__", r"\*"),
 382             ("__truediv__", "/"),
 383         ]:
 384             msg = f"Series cannot perform the operation {str_rep}|unsupported operand"
 385             with pytest.raises(TypeError, match=msg):
 386                 getattr(s, op)(2)
 387 
 388         # invalid ufunc
 389         msg = "Object with dtype category cannot perform the numpy op log"
 390         with pytest.raises(TypeError, match=msg):
 391             np.log(s)
