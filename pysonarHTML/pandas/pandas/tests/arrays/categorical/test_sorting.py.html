  22     def test_numpy_argsort(self):

  40     def test_sort_values(self):
  41 
  42         # unordered cats are sortable
  43         cat = Categorical(["a", "b", "b", "a"], ordered=False)
  44         cat.sort_values()
  45 
  46         cat = Categorical(["a", "c", "b", "d"], ordered=True)
  47 
  48         # sort_values
  49         res = cat.sort_values()
  50         exp = np.array(["a", "b", "c", "d"], dtype=object)
  51         tm.assert_numpy_array_equal(res.__array__(), exp)
  52         tm.assert_index_equal(res.categories, cat.categories)
  53 
  54         cat = Categorical(
  55             ["a", "c", "b", "d"], categories=["a", "b", "c", "d"], ordered=True
  56         )
  57         res = cat.sort_values()
  58         exp = np.array(["a", "b", "c", "d"], dtype=object)
  59         tm.assert_numpy_array_equal(res.__array__(), exp)
  60         tm.assert_index_equal(res.categories, cat.categories)
  61 
  62         res = cat.sort_values(ascending=False)
  63         exp = np.array(["d", "c", "b", "a"], dtype=object)
  64         tm.assert_numpy_array_equal(res.__array__(), exp)
  65         tm.assert_index_equal(res.categories, cat.categories)
  66 
  67         # sort (inplace order)
  68         cat1 = cat.copy()
  69         cat1.sort_values(inplace=True)
  70         exp = np.array(["a", "b", "c", "d"], dtype=object)
  71         tm.assert_numpy_array_equal(cat1.__array__(), exp)
  72         tm.assert_index_equal(res.categories, cat.categories)
  73 
  74         # reverse
  75         cat = Categorical(["a", "c", "c", "b", "d"], ordered=True)
  76         res = cat.sort_values(ascending=False)
  77         exp_val = np.array(["d", "c", "c", "b", "a"], dtype=object)
  78         exp_categories = Index(["a", "b", "c", "d"])
  79         tm.assert_numpy_array_equal(res.__array__(), exp_val)
  80         tm.assert_index_equal(res.categories, exp_categories)

  82     def test_sort_values_na_position(self):

  96 
  97         exp = np.array([np.nan, np.nan, 5.0, 2.0, 2.0])

 100         tm.assert_index_equal(res.categories, exp_categories)
