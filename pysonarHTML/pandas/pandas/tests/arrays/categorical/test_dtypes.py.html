 111     @pytest.mark.parametrize("ordered", [True, False])
 112     def test_astype(self, ordered):
 113         # string
 114         cat = Categorical(list("abbaaccc"), ordered=ordered)
 115         result = cat.astype(object)
 116         expected = np.array(cat)
 117         tm.assert_numpy_array_equal(result, expected)
 118 
 119         msg = "could not convert string to float"
 120         with pytest.raises(ValueError, match=msg):
 121             cat.astype(float)
 122 
 123         # numeric
 124         cat = Categorical([0, 1, 2, 2, 1, 0, 1, 0, 2], ordered=ordered)
 125         result = cat.astype(object)
 126         expected = np.array(cat, dtype=object)
 127         tm.assert_numpy_array_equal(result, expected)
 128 
 129         result = cat.astype(int)
 130         expected = np.array(cat, dtype=np.int)
 131         tm.assert_numpy_array_equal(result, expected)
 132 
 133         result = cat.astype(float)
 134         expected = np.array(cat, dtype=np.float)
 135         tm.assert_numpy_array_equal(result, expected)

 137     @pytest.mark.parametrize("dtype_ordered", [True, False])
 138     @pytest.mark.parametrize("cat_ordered", [True, False])
 139     def test_astype_category(self, dtype_ordered, cat_ordered):
 140         # GH 10696/18593
 141         data = list("abcaacbab")
 142         cat = Categorical(data, categories=list("bac"), ordered=cat_ordered)
 143 
 144         # standard categories
 145         dtype = CategoricalDtype(ordered=dtype_ordered)
 146         result = cat.astype(dtype)
 147         expected = Categorical(data, categories=cat.categories, ordered=dtype_ordered)
 148         tm.assert_categorical_equal(result, expected)
 149 
 150         # non-standard categories
 151         dtype = CategoricalDtype(list("adc"), dtype_ordered)
 152         result = cat.astype(dtype)
 153         expected = Categorical(data, dtype=dtype)
 154         tm.assert_categorical_equal(result, expected)
 155 
 156         if dtype_ordered is False:
 157             # dtype='category' can't specify ordered, so only test once
 158             result = cat.astype("category")
 159             expected = cat
 160             tm.assert_categorical_equal(result, expected)
