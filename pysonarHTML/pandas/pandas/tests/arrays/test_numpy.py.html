 193 def test_ufunc():

 201     e1 = PandasArray(e1)

 203     tm.assert_extension_array_equal(r1, e1)

 236 def test_setitem_preserves_views():
 237     # GH#28150, see also extension test of the same name
 238     arr = PandasArray(np.array([1, 2, 3]))
 239     view1 = arr.view()
 240     view2 = arr[:]
 241     view3 = np.asarray(arr)
 242 
 243     arr[0] = 9
 244     assert view1[0] == 9
 245     assert view2[0] == 9
 246     assert view3[0] == 9
 247 
 248     arr[-1] = 2.5
 249     view1[-1] = 5
 250     assert arr[-1] == 5
