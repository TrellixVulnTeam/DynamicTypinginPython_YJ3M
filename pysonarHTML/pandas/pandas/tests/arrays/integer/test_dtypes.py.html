  22 @pytest.mark.parametrize("op", ["sum", "min", "max", "prod"])
  23 def test_preserve_dtypes(op):
  24     # TODO(#22346): preserve Int64 dtype
  25     # for ops that enable (mean would actually work here
  26     # but generally it is a float return value)
  27     df = pd.DataFrame(
  28         {
  29             "A": ["a", "b", "b"],
  30             "B": [1, None, 3],
  31             "C": integer_array([1, None, 3], dtype="Int64"),
  32         }
  33     )
  34 
  35     # op
  36     result = getattr(df.C, op)()
  37     if op in {"sum", "prod", "min", "max"}:
  38         assert isinstance(result, np.int64)
  39     else:
  40         assert isinstance(result, int)

  45     expected = pd.DataFrame(

  47         index=pd.Index(["a", "b"], name="A"),
  48     )
  49     tm.assert_frame_equal(result, expected)

  97 def test_astype(all_data):
  98     all_data = all_data[:10]
  99 
 100     ints = all_data[~all_data.isna()]
 101     mixed = all_data
 102     dtype = Int8Dtype()
 103 
 104     # coerce to same type - ints
 105     s = pd.Series(ints)
 106     result = s.astype(all_data.dtype)
 107     expected = pd.Series(ints)
 108     tm.assert_series_equal(result, expected)
 109 
 110     # coerce to same other - ints
 111     s = pd.Series(ints)
 112     result = s.astype(dtype)
 113     expected = pd.Series(ints, dtype=dtype)
 114     tm.assert_series_equal(result, expected)
 115 
 116     # coerce to same numpy_dtype - ints
 117     s = pd.Series(ints)
 118     result = s.astype(all_data.dtype.numpy_dtype)
 119     expected = pd.Series(ints._data.astype(all_data.dtype.numpy_dtype))
 120     tm.assert_series_equal(result, expected)
 121 
 122     # coerce to same type - mixed
 123     s = pd.Series(mixed)
 124     result = s.astype(all_data.dtype)
 125     expected = pd.Series(mixed)
 126     tm.assert_series_equal(result, expected)
 127 
 128     # coerce to same other - mixed
 129     s = pd.Series(mixed)
 130     result = s.astype(dtype)
 131     expected = pd.Series(mixed, dtype=dtype)
 132     tm.assert_series_equal(result, expected)
 133 
 134     # coerce to same numpy_dtype - mixed
 135     s = pd.Series(mixed)
 136     msg = r"cannot convert to .*-dtype NumPy array with missing values.*"
 137     with pytest.raises(ValueError, match=msg):
 138         s.astype(all_data.dtype.numpy_dtype)
 139 
 140     # coerce to object
 141     s = pd.Series(mixed)
 142     result = s.astype("object")
 143     expected = pd.Series(np.asarray(mixed))
 144     tm.assert_series_equal(result, expected)

 200 @pytest.mark.parametrize("in_series", [True, False])
 201 def test_to_numpy_na_nan(in_series):
 202     a = pd.array([0, 1, None], dtype="Int64")
 203     if in_series:
 204         a = pd.Series(a)
 205 
 206     result = a.to_numpy(dtype="float64", na_value=np.nan)
 207     expected = np.array([0.0, 1.0, np.nan], dtype="float64")
 208     tm.assert_numpy_array_equal(result, expected)
 209 
 210     result = a.to_numpy(dtype="int64", na_value=-1)
 211     expected = np.array([0, 1, -1], dtype="int64")
 212     tm.assert_numpy_array_equal(result, expected)
 213 
 214     result = a.to_numpy(dtype="bool", na_value=False)
 215     expected = np.array([False, True, False], dtype="bool")
 216     tm.assert_numpy_array_equal(result, expected)
