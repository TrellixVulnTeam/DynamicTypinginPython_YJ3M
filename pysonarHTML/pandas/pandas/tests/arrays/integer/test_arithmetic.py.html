  87     result = a ** 0
  88     expected = pd.array([1, 1, 1, 1, 1], dtype="Int64")
  89     tm.assert_extension_array_equal(result, expected)
  90 
  91     result = a ** 1
  92     expected = pd.array([-1, 0, 1, None, 2], dtype="Int64")
  93     tm.assert_extension_array_equal(result, expected)
  94

 152 def test_error_invalid_values(data, all_arithmetic_operators):
 153 
 154     op = all_arithmetic_operators
 155     s = pd.Series(data)
 156     ops = getattr(s, op)
 157 
 158     # invalid scalars
 159     msg = (
 160         r"(:?can only perform ops with numeric values)"
 161         r"|(:?IntegerArray cannot perform the operation mod)"
 162     )
 163     with pytest.raises(TypeError, match=msg):
 164         ops("foo")
 165     with pytest.raises(TypeError, match=msg):
 166         ops(pd.Timestamp("20180101"))
 167 
 168     # invalid array-likes
 169     with pytest.raises(TypeError, match=msg):
 170         ops(pd.Series("foo", index=s.index))
 171 
 172     if op != "__rpow__":
 173         # TODO(extension)
 174         # rpow with a datetimelike coerces the integer array incorrectly
 175         msg = (
 176             "can only perform ops with numeric values|"
 177             "cannot perform .* with this index type: DatetimeArray|"
 178             "Addition/subtraction of integers and integer-arrays "
 179             "with DatetimeArray is no longer supported. *"
 180         )
 181         with pytest.raises(TypeError, match=msg):
 182             ops(pd.Series(pd.date_range("20180101", periods=len(s))))

 192 def test_arith_coerce_scalar(data, all_arithmetic_operators):
 193     op = tm.get_op_from_name(all_arithmetic_operators)
 194     s = pd.Series(data)
 195     other = 0.01
 196 
 197     result = op(s, other)
 198     expected = op(s.astype(float), other)
 199     # rfloordiv results in nan instead of inf
 200     if all_arithmetic_operators == "__rfloordiv__":
 201         expected[(expected == np.inf) | (expected == -np.inf)] = np.nan
 202 
 203     tm.assert_series_equal(result, expected)

 217 def test_cross_type_arithmetic():
 218 
 219     df = pd.DataFrame(
 220         {
 221             "A": pd.Series([1, 2, np.nan], dtype="Int64"),
 222             "B": pd.Series([1, np.nan, 3], dtype="UInt8"),
 223             "C": [1, 2, 3],
 224         }
 225     )
 226 
 227     result = df.A + df.C
 228     expected = pd.Series([2, 4, np.nan], dtype="Int64")
 229     tm.assert_series_equal(result, expected)
 230 
 231     result = (df.A + df.C) * 3 == 12
 232     expected = pd.Series([False, True, None], dtype="boolean")
 233     tm.assert_series_equal(result, expected)
 234 
 235     result = df.A + df.B
 236     expected = pd.Series([2, np.nan, np.nan], dtype="Int64")
 237     tm.assert_series_equal(result, expected)

 240 @pytest.mark.parametrize("op", ["mean"])
 241 def test_reduce_to_float(op):
 242     # some reduce ops always return float, even if the result
 243     # is a rounded number
 244     df = pd.DataFrame(
 245         {
 246             "A": ["a", "b", "b"],
 247             "B": [1, None, 3],
 248             "C": integer_array([1, None, 3], dtype="Int64"),
 249         }
 250     )
 251 
 252     # op
 253     result = getattr(df.C, op)()
 254     assert isinstance(result, float)
 255 
 256     # groupby
 257     result = getattr(df.groupby("A"), op)()
 258 
 259     expected = pd.DataFrame(
 260         {"B": np.array([1.0, 3.0]), "C": integer_array([1, 3], dtype="Int64")},
 261         index=pd.Index(["a", "b"], name="A"),
 262     )
 263     tm.assert_frame_equal(result, expected)
