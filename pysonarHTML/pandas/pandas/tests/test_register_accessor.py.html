   9 @contextlib.contextmanager
  10 def ensure_removed(obj, attr):
  11     """Ensure that an attribute added to 'obj' during the test is
  12     removed when we're done
  13     """
  14     try:
  15         yield
  16     finally:
  17         try:
  18             delattr(obj, attr)
  19         except AttributeError:
  20             pass
  21         obj._accessors.discard(attr)

  24 class MyAccessor:
  25     def __init__(self, obj):

  37 @pytest.mark.parametrize(
  38     "obj, registrar",
  39     [
  40         (pd.Series, pd.api.extensions.register_series_accessor),
  41         (pd.DataFrame, pd.api.extensions.register_dataframe_accessor),
  42         (pd.Index, pd.api.extensions.register_index_accessor),
  43     ],
  44 )
  45 def test_register(obj, registrar):
  46     with ensure_removed(obj, "mine"):
  47         before = set(dir(obj))
  48         registrar("mine")(MyAccessor)
  49         o = obj([]) if obj is not pd.Series else obj([], dtype=object)
  50         assert o.mine.prop == "item"
  51         after = set(dir(obj))
  52         assert (before ^ after) == {"mine"}
  53         assert "mine" in obj._accessors

  56 def test_accessor_works():

  67 def test_overwrite_warns():
  68     # Need to restore mean
  69     mean = pd.Series.mean
  70     try:
  71         with tm.assert_produces_warning(UserWarning) as w:
  72             pd.api.extensions.register_series_accessor("mean")(MyAccessor)
  73             s = pd.Series([1, 2])
  74             assert s.mean.prop == "item"
  75         msg = str(w[0].message)
  76         assert "mean" in msg
  77         assert "MyAccessor" in msg
  78         assert "Series" in msg
  79     finally:
  80         pd.Series.mean = mean
