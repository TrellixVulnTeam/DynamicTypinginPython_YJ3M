   7 def test_maybe_mangle_lambdas_passthrough():
   8     assert maybe_mangle_lambdas("mean") == "mean"
   9     assert maybe_mangle_lambdas(lambda x: x).__name__ == "<lambda>"
  10     # don't mangel single lambda.
  11     assert maybe_mangle_lambdas([lambda x: x])[0].__name__ == "<lambda>"

  14 def test_maybe_mangle_lambdas_listlike():
  15     aggfuncs = [lambda x: 1, lambda x: 2]
  16     result = maybe_mangle_lambdas(aggfuncs)
  17     assert result[0].__name__ == "<lambda_0>"
  18     assert result[1].__name__ == "<lambda_1>"
  19     assert aggfuncs[0](None) == result[0](None)
  20     assert aggfuncs[1](None) == result[1](None)

  30 def test_maybe_mangle_lambdas_args():
  31     func = {"A": [lambda x, a, b=1: (0, a, b), lambda x: 1]}
  32     result = maybe_mangle_lambdas(func)
  33     assert result["A"][0].__name__ == "<lambda_0>"
  34     assert result["A"][1].__name__ == "<lambda_1>"

  36     assert func["A"][0](0, 1) == (0, 1, 1)

  38     assert func["A"][0](0, 2, b=3) == (0, 2, 3)
