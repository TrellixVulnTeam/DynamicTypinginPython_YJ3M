   6 def assert_offset_equal(offset, base, expected):
   7     actual = offset + base
   8     actual_swapped = base + offset
   9     actual_apply = offset.apply(base)
  10     try:
  11         assert actual == expected
  12         assert actual_swapped == expected
  13         assert actual_apply == expected
  14     except AssertionError as err:
  15         raise AssertionError(
  16             f"\nExpected: {expected}\nActual: {actual}\nFor Offset: {offset})"
  17             f"\nAt Date: {base}"
  18         ) from err
