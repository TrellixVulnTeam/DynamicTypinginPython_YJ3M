 635 def test_bunched_yearends():
 636     # GH#14774 cases with two fiscal year-ends in the same calendar-year
 637     fy = FY5253(n=1, weekday=5, startingMonth=12, variation="nearest")
 638     dt = Timestamp("2004-01-01")
 639     assert fy.rollback(dt) == Timestamp("2002-12-28")
 640     assert (-fy).apply(dt) == Timestamp("2002-12-28")
 641     assert dt - fy == Timestamp("2002-12-28")
 642 
 643     assert fy.rollforward(dt) == Timestamp("2004-01-03")
 644     assert fy.apply(dt) == Timestamp("2004-01-03")
 645     assert fy + dt == Timestamp("2004-01-03")
 646     assert dt + fy == Timestamp("2004-01-03")
 647 
 648     # Same thing, but starting from a Timestamp in the previous year.
 649     dt = Timestamp("2003-12-31")
 650     assert fy.rollback(dt) == Timestamp("2002-12-28")
 651     assert (-fy).apply(dt) == Timestamp("2002-12-28")
 652     assert dt - fy == Timestamp("2002-12-28")
