  88             jim[key].append(a)

 125     def test_nargsort(self):
 126         # np.argsort(items) places NaNs last
 127         items = [np.nan] * 5 + list(range(100)) + [np.nan] * 5
 128         # np.argsort(items2) may not place NaNs first
 129         items2 = np.array(items, dtype="O")
 130 
 131         # mergesort is the most difficult to get right because we want it to be
 132         # stable.
 133 
 134         # According to numpy/core/tests/test_multiarray, """The number of
 135         # sorted items must be greater than ~50 to check the actual algorithm
 136         # because quick and merge sort fall over to insertion sort for small
 137         # arrays."""
 138 
 139         # mergesort, ascending=True, na_position='last'
 140         result = nargsort(items, kind="mergesort", ascending=True, na_position="last")
 141         exp = list(range(5, 105)) + list(range(5)) + list(range(105, 110))
 142         tm.assert_numpy_array_equal(result, np.array(exp), check_dtype=False)
 143 
 144         # mergesort, ascending=True, na_position='first'
 145         result = nargsort(items, kind="mergesort", ascending=True, na_position="first")
 146         exp = list(range(5)) + list(range(105, 110)) + list(range(5, 105))
 147         tm.assert_numpy_array_equal(result, np.array(exp), check_dtype=False)
 148 
 149         # mergesort, ascending=False, na_position='last'
 150         result = nargsort(items, kind="mergesort", ascending=False, na_position="last")
 151         exp = list(range(104, 4, -1)) + list(range(5)) + list(range(105, 110))
 152         tm.assert_numpy_array_equal(result, np.array(exp), check_dtype=False)
 153 
 154         # mergesort, ascending=False, na_position='first'
 155         result = nargsort(items, kind="mergesort", ascending=False, na_position="first")
 156         exp = list(range(5)) + list(range(105, 110)) + list(range(104, 4, -1))
 157         tm.assert_numpy_array_equal(result, np.array(exp), check_dtype=False)
 158 
 159         # mergesort, ascending=True, na_position='last'
 160         result = nargsort(items2, kind="mergesort", ascending=True, na_position="last")
 161         exp = list(range(5, 105)) + list(range(5)) + list(range(105, 110))
 162         tm.assert_numpy_array_equal(result, np.array(exp), check_dtype=False)
 163 
 164         # mergesort, ascending=True, na_position='first'
 165         result = nargsort(items2, kind="mergesort", ascending=True, na_position="first")
 166         exp = list(range(5)) + list(range(105, 110)) + list(range(5, 105))
 167         tm.assert_numpy_array_equal(result, np.array(exp), check_dtype=False)
 168 
 169         # mergesort, ascending=False, na_position='last'
 170         result = nargsort(items2, kind="mergesort", ascending=False, na_position="last")
 171         exp = list(range(104, 4, -1)) + list(range(5)) + list(range(105, 110))
 172         tm.assert_numpy_array_equal(result, np.array(exp), check_dtype=False)
 173 
 174         # mergesort, ascending=False, na_position='first'
 175         result = nargsort(
 176             items2, kind="mergesort", ascending=False, na_position="first"
 177         )
 178         exp = list(range(5)) + list(range(105, 110)) + list(range(104, 4, -1))
 179         tm.assert_numpy_array_equal(result, np.array(exp), check_dtype=False)

 182 class TestMerge:
 183     @pytest.mark.slow
 184     def test_int64_overflow_issues(self):
 185 
 186         # #2690, combinatorial explosion
 187         df1 = DataFrame(np.random.randn(1000, 7), columns=list("ABCDEF") + ["G1"])
 188         df2 = DataFrame(np.random.randn(1000, 7), columns=list("ABCDEF") + ["G2"])
 189 
 190         # it works!
 191         result = merge(df1, df2, how="outer")
 192         assert len(result) == 2000
 193 
 194         low, high, n = -1 << 10, 1 << 10, 1 << 20
 195         left = DataFrame(np.random.randint(low, high, (n, 7)), columns=list("ABCDEFG"))
 196         left["left"] = left.sum(axis=1)
 197 
 198         # one-2-one match
 199         i = np.random.permutation(len(left))
 200         right = left.iloc[i].copy()
 201         right.columns = right.columns[:-1].tolist() + ["right"]
 202         right.index = np.arange(len(right))
 203         right["right"] *= -1
 204 
 205         out = merge(left, right, how="outer")
 206         assert len(out) == len(left)
 207         tm.assert_series_equal(out["left"], -out["right"], check_names=False)
 208         result = out.iloc[:, :-2].sum(axis=1)
 209         tm.assert_series_equal(out["left"], result, check_names=False)
 210         assert result.name is None
 211 
 212         out.sort_values(out.columns.tolist(), inplace=True)
 213         out.index = np.arange(len(out))
 214         for how in ["left", "right", "outer", "inner"]:
 215             tm.assert_frame_equal(out, merge(left, right, how=how, sort=True))
 216 
 217         # check that left merge w/ sort=False maintains left frame order
 218         out = merge(left, right, how="left", sort=False)
 219         tm.assert_frame_equal(left, out[left.columns.tolist()])
 220 
 221         out = merge(right, left, how="left", sort=False)
 222         tm.assert_frame_equal(right, out[right.columns.tolist()])
 223 
 224         # one-2-many/none match
 225         n = 1 << 11
 226         left = DataFrame(
 227             np.random.randint(low, high, (n, 7)).astype("int64"),
 228             columns=list("ABCDEFG"),
 229         )
 230 
 231         # confirm that this is checking what it is supposed to check
 232         shape = left.apply(Series.nunique).values
 233         assert is_int64_overflow_possible(shape)
 234 
 235         # add duplicates to left frame
 236         left = concat([left, left], ignore_index=True)
 237 
 238         right = DataFrame(
 239             np.random.randint(low, high, (n // 2, 7)).astype("int64"),
 240             columns=list("ABCDEFG"),
 241         )
 242 
 243         # add duplicates & overlap with left to the right frame
 244         i = np.random.choice(len(left), n)
 245         right = concat([right, right, left.iloc[i]], ignore_index=True)
 246 
 247         left["left"] = np.random.randn(len(left))
 248         right["right"] = np.random.randn(len(right))
 249 
 250         # shuffle left & right frames
 251         i = np.random.permutation(len(left))
 252         left = left.iloc[i].copy()
 253         left.index = np.arange(len(left))
 254 
 255         i = np.random.permutation(len(right))
 256         right = right.iloc[i].copy()
 257         right.index = np.arange(len(right))
 258 
 259         # manually compute outer merge
 260         ldict, rdict = defaultdict(list), defaultdict(list)
 261 
 262         for idx, row in left.set_index(list("ABCDEFG")).iterrows():
 263             ldict[idx].append(row["left"])
 264 
 265         for idx, row in right.set_index(list("ABCDEFG")).iterrows():
 266             rdict[idx].append(row["right"])
 267 
 268         vals = []
 269         for k, lval in ldict.items():
 270             rval = rdict.get(k, [np.nan])
 271             for lv, rv in product(lval, rval):
 272                 vals.append(k + tuple([lv, rv]))
 273 
 274         for k, rval in rdict.items():
 275             if k not in ldict:
 276                 for rv in rval:
 277                     vals.append(k + tuple([np.nan, rv]))
 278 
 279         def align(df):
 280             df = df.sort_values(df.columns.tolist())
 281             df.index = np.arange(len(df))
 282             return df
 283 
 284         def verify_order(df):
 285             kcols = list("ABCDEFG")
 286             tm.assert_frame_equal(
 287                 df[kcols].copy(), df[kcols].sort_values(kcols, kind="mergesort")
 288             )
 289 
 290         out = DataFrame(vals, columns=list("ABCDEFG") + ["left", "right"])
 291         out = align(out)
 292 
 293         jmask = {
 294             "left": out["left"].notna(),
 295             "right": out["right"].notna(),
 296             "inner": out["left"].notna() & out["right"].notna(),
 297             "outer": np.ones(len(out), dtype="bool"),
 298         }
 299 
 300         for how in "left", "right", "outer", "inner":
 301             mask = jmask[how]
 302             frame = align(out[mask].copy())
 303             assert mask.all() ^ mask.any() or how == "outer"
 304 
 305             for sort in [False, True]:
 306                 res = merge(left, right, how=how, sort=sort)
 307                 if sort:
 308                     verify_order(res)
 309 
 310                 # as in GH9092 dtypes break with outer/right join
 311                 tm.assert_frame_equal(
 312                     frame, align(res), check_dtype=how not in ("right", "outer")
 313                 )
