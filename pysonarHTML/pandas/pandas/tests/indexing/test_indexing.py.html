 441         # GH 3626, an assignment of a sub-df to a df
 442         df = DataFrame(
 443             {
 444                 "FC": ["a", "b", "a", "b", "a", "b"],

 447                 "col2": list(range(6, 12)),
 448             }

 451         df2 = df.copy()
 452 
 453         mask = ~df2.FC.isna()

 466         )
 467 
 468         # frame on rhs
 469         df2.loc[mask, cols] = dft.loc[mask, cols]
 470         tm.assert_frame_equal(df2, expected)
 471 
 472         df2.loc[mask, cols] = dft.loc[mask, cols]
