 343     def test_iloc_setitem_dups(self):
 344 
 345         # GH 6766
 346         # iloc with a mask aligning from another iloc
 347         df1 = DataFrame([{"A": None, "B": 1}, {"A": 2, "B": 2}])
 348         df2 = DataFrame([{"A": 3, "B": 3}, {"A": 4, "B": 4}])
 349         df = concat([df1, df2], axis=1)
 350 
 351         expected = df.fillna(3)
 352         inds = np.isnan(df.iloc[:, 0])
 353         mask = inds[inds].index
 354         df.iloc[mask, 0] = df.iloc[mask, 2]
 355         tm.assert_frame_equal(df, expected)
 356 
 357         # del a dup column across blocks
 358         expected = DataFrame({0: [1, 2], 1: [3, 4]})
 359         expected.columns = ["B", "B"]
 360         del df["A"]
 361         tm.assert_frame_equal(df, expected)
 362 
 363         # assign back to self
 364         df.iloc[[0, 1], [0, 1]] = df.iloc[[0, 1], [0, 1]]
 365         tm.assert_frame_equal(df, expected)
 366 
 367         # reversed x 2
 368         df.iloc[[1, 0], [0, 1]] = df.iloc[[1, 0], [0, 1]].reset_index(drop=True)
 369         df.iloc[[1, 0], [0, 1]] = df.iloc[[1, 0], [0, 1]].reset_index(drop=True)
 370         tm.assert_frame_equal(df, expected)
