 136     def test_indexing_with_datetimeindex_tz(self):
 137 
 138         # GH 12050
 139         # indexing on a series with a datetimeindex with tz
 140         index = date_range("2015-01-01", periods=2, tz="utc")
 141 
 142         ser = Series(range(2), index=index, dtype="int64")
 143 
 144         # list-like indexing
 145 
 146         for sel in (index, list(index)):
 147             # getitem
 148             result = ser[sel]
 149             expected = ser.copy()
 150             if sel is not index:
 151                 expected.index = expected.index._with_freq(None)
 152             tm.assert_series_equal(result, expected)
 153 
 154             # setitem
 155             result = ser.copy()
 156             result[sel] = 1
 157             expected = Series(1, index=index)
 158             tm.assert_series_equal(result, expected)
 159 
 160             # .loc getitem
 161             result = ser.loc[sel]
 162             expected = ser.copy()
 163             if sel is not index:
 164                 expected.index = expected.index._with_freq(None)
 165             tm.assert_series_equal(result, expected)
 166 
 167             # .loc setitem
 168             result = ser.copy()
 169             result.loc[sel] = 1
 170             expected = Series(1, index=index)
 171             tm.assert_series_equal(result, expected)
 172 
 173         # single element indexing
 174 
 175         # getitem
 176         assert ser[index[1]] == 1
 177 
 178         # setitem
 179         result = ser.copy()
 180         result[index[1]] = 5
 181         expected = Series([0, 5], index=index)
 182         tm.assert_series_equal(result, expected)
 183 
 184         # .loc getitem
 185         assert ser.loc[index[1]] == 1
 186 
 187         # .loc setitem
 188         result = ser.copy()
 189         result.loc[index[1]] = 5
 190         expected = Series([0, 5], index=index)
 191         tm.assert_series_equal(result, expected)

 300     def test_loc_getitem_across_dst(self):
 301         # GH 21846
 302         idx = pd.date_range(
 303             "2017-10-29 01:30:00", tz="Europe/Berlin", periods=5, freq="30 min"

 310         t_2 = pd.Timestamp(
