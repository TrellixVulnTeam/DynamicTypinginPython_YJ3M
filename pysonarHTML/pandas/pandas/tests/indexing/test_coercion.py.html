  18 @pytest.fixture(autouse=True, scope="class")
  19 def check_comprehensiveness(request):
  20     # Iterate over combination of dtype, method and klass
  21     # and ensure that each are contained within a collected test
  22     cls = request.cls
  23     combos = itertools.product(cls.klasses, cls.dtypes, [cls.method])
  24 
  25     def has_test(combo):
  26         klass, dtype, method = combo
  27         cls_funcs = request.node.session.items
  28         return any(
  29             klass in x.name and dtype in x.name and method in x.name for x in cls_funcs
  30         )
  31 
  32     for combo in combos:
  33         if not has_test(combo):
  34             raise AssertionError(f"test method is not defined: {cls.__name__}, {combo}")
  35 
  36     yield
