 326 def test_iat_dont_wrap_object_datetimelike():
 327     # GH#32809 .iat calls go through DataFrame._get_value, should not
 328     #  call maybe_box_datetimelike
 329     dti = date_range("2016-01-01", periods=3)
 330     tdi = dti - dti
 331     ser = Series(dti.to_pydatetime(), dtype=object)
 332     ser2 = Series(tdi.to_pytimedelta(), dtype=object)
 333     df = DataFrame({"A": ser, "B": ser2})
 334     assert (df.dtypes == object).all()
 335 
 336     for result in [df.at[0, "A"], df.iat[0, 0], df.loc[0, "A"], df.iloc[0, 0]]:
 337         assert result is ser[0]
 338         assert isinstance(result, datetime)
 339         assert not isinstance(result, Timestamp)
 340 
 341     for result in [df.at[1, "B"], df.iat[1, 1], df.loc[1, "B"], df.iloc[1, 1]]:
 342         assert result is ser2[1]
 343         assert isinstance(result, timedelta)
 344         assert not isinstance(result, Timedelta)
