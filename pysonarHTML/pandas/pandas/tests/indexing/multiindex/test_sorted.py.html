  83     def test_frame_getitem_not_sorted(self, multiindex_dataframe_random_data):
  84         frame = multiindex_dataframe_random_data
  85         df = frame.T
  86         df["foo", "four"] = "foo"
  87

  89

  96 
  97         df = df.T
  98         result = df.xs("foo")
