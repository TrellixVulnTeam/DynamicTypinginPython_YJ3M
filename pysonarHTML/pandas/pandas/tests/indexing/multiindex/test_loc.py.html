 249     @pytest.mark.parametrize(
 250         "indexer_type_1", (list, tuple, set, slice, np.ndarray, Series, Index)
 251     )
 252     @pytest.mark.parametrize(
 253         "indexer_type_2", (list, tuple, set, slice, np.ndarray, Series, Index)
 254     )
 255     def test_loc_getitem_nested_indexer(self, indexer_type_1, indexer_type_2):
 256         # GH #19686
 257         # .loc should work with nested indexers which can be
 258         # any list-like objects (see `pandas.api.types.is_list_like`) or slices
 259 
 260         def convert_nested_indexer(indexer_type, keys):
 261             if indexer_type == np.ndarray:
 262                 return np.array(keys)
 263             if indexer_type == slice:
 264                 return slice(*keys)
 265             return indexer_type(keys)
 266 
 267         a = [10, 20, 30]
 268         b = [1, 2, 3]
 269         index = MultiIndex.from_product([a, b])
 270         df = DataFrame(
 271             np.arange(len(index), dtype="int64"), index=index, columns=["Data"]
 272         )
 273 
 274         keys = ([10, 20], [2, 3])
 275         types = (indexer_type_1, indexer_type_2)
 276 
 277         # check indexers with all the combinations of nested objects
 278         # of all the valid types
 279         indexer = tuple(
 280             convert_nested_indexer(indexer_type, k)
 281             for indexer_type, k in zip(types, keys)
 282         )
 283 
 284         result = df.loc[indexer, "Data"]
 285         expected = Series(
 286             [1, 2, 4, 5], name="Data", index=MultiIndex.from_product(keys)
 287         )
 288 
 289         tm.assert_series_equal(result, expected)
