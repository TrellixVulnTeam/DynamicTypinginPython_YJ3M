 120     def test_multiindex_setitem(self):
 121 
 122         # GH 3738
 123         # setting with a multi-index right hand side
 124         arrays = [
 125             np.array(["bar", "bar", "baz", "qux", "qux", "bar"]),
 126             np.array(["one", "two", "one", "one", "two", "one"]),
 127             np.arange(0, 6, 1),
 128         ]
 129 
 130         df_orig = DataFrame(
 131             np.random.randn(6, 3), index=arrays, columns=["A", "B", "C"]
 132         ).sort_index()
 133 
 134         expected = df_orig.loc[["bar"]] * 2
 135         df = df_orig.copy()
 136         df.loc[["bar"]] *= 2
 137         tm.assert_frame_equal(df.loc[["bar"]], expected)
 138 
 139         # raise because these have differing levels
 140         msg = "cannot align on a multi-index with out specifying the join levels"
 141         with pytest.raises(TypeError, match=msg):
 142             df.loc["bar"] *= 2
 143 
 144         # from SO
 145         # https://stackoverflow.com/questions/24572040/pandas-access-the-level-of-multiindex-for-inplace-operation
 146         df_orig = DataFrame.from_dict(
 147             {
 148                 "price": {
 149                     ("DE", "Coal", "Stock"): 2,
 150                     ("DE", "Gas", "Stock"): 4,
 151                     ("DE", "Elec", "Demand"): 1,
 152                     ("FR", "Gas", "Stock"): 5,
 153                     ("FR", "Solar", "SupIm"): 0,
 154                     ("FR", "Wind", "SupIm"): 0,
 155                 }
 156             }
 157         )
 158         df_orig.index = MultiIndex.from_tuples(
 159             df_orig.index, names=["Sit", "Com", "Type"]
 160         )
 161 
 162         expected = df_orig.copy()
 163         expected.iloc[[0, 2, 3]] *= 2
 164 
 165         idx = pd.IndexSlice
 166         df = df_orig.copy()
 167         df.loc[idx[:, :, "Stock"], :] *= 2
 168         tm.assert_frame_equal(df, expected)
 169 
 170         df = df_orig.copy()
 171         df.loc[idx[:, :, "Stock"], "price"] *= 2
 172         tm.assert_frame_equal(df, expected)

 175 
 176         # GH3777 part 2

 178         # mixed dtype
 179         df = DataFrame(
 180             np.random.randint(5, 10, size=9).reshape(3, 3),
 181             columns=list("abc"),
 182             index=[[4, 4, 8], [8, 10, 12]],
 183         )
 184         df["d"] = np.nan
 185         arr = np.array([0.0, 1.0])
 186 
 187         df.loc[4, "d"] = arr
 188         tm.assert_series_equal(df.loc[4, "d"], Series(arr, index=[8, 10], name="d"))
 189 
 190         # single dtype

 236                 f_index

 265     def test_frame_getitem_setitem_boolean(self, multiindex_dataframe_random_data):
 266         frame = multiindex_dataframe_random_data
 267         df = frame.T.copy()
 268         values = df.values
 269 
 270         result = df[df > 0]
 271         expected = df.where(df > 0)

 273 
 274         df[df > 0] = 5

 276         tm.assert_almost_equal(df.values, values)
 277

 279         values[values == 5] = 0

 286

 314         df = DataFrame(randn(10, 4), columns=[["a", "a", "b", "b"], [0, 1, 0, 1]])
 315 
 316         cp = df.copy()
 317         cp["a"] = cp["b"]
 318         tm.assert_frame_equal(cp["a"], cp["b"])
 319 
 320         # set with ndarray
 321         cp = df.copy()
 322         cp["a"] = cp["b"].values
 323         tm.assert_frame_equal(cp["a"], cp["b"])

 325         # ---------------------------------------
 326         # #1803
 327         columns = MultiIndex.from_tuples([("A", "1"), ("A", "2"), ("B", "1")])
 328         df = DataFrame(index=[1, 3, 5], columns=columns)
 329 
 330         # Works, but adds a column instead of updating the two existing ones
 331         df["A"] = 0.0  # Doesn't work

 339         sliced_a2 = df["A", "2"]
 340         sliced_b1 = df["B", "1"]
 341         tm.assert_series_equal(sliced_a1, sliced_b1, check_names=False)
 342         tm.assert_series_equal(sliced_a2, sliced_b1, check_names=False)
 343         assert sliced_a1.name == ("A", "1")
 344         assert sliced_a2.name == ("A", "2")
