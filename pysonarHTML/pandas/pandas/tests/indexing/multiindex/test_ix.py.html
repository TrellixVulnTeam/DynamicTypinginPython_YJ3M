  22     def test_loc_general(self):
  23 
  24         # GH 2817
  25         data = {
  26             "amount": {0: 700, 1: 600, 2: 222, 3: 333, 4: 444},
  27             "col": {0: 3.5, 1: 3.5, 2: 4.0, 3: 4.0, 4: 4.0},
  28             "year": {0: 2012, 1: 2011, 2: 2012, 3: 2012, 4: 2012},
  29         }
  30         df = DataFrame(data).set_index(keys=["col", "year"])
  31         key = 4.0, 2012
  32 
  33         # emits a PerformanceWarning, ok
  34         with tm.assert_produces_warning(PerformanceWarning):
  35             tm.assert_frame_equal(df.loc[key], df.iloc[2:])
  36 
  37         # this is ok
  38         df.sort_index(inplace=True)
  39         res = df.loc[key]
  40 
  41         # col has float dtype, result should be Float64Index
  42         index = MultiIndex.from_arrays([[4.0] * 3, [2012] * 3], names=["col", "year"])
  43         expected = DataFrame({"amount": [222, 333, 444]}, index=index)
  44         tm.assert_frame_equal(res, expected)
