 395             df.loc[:, 1] = 1
 396 
 397         # these work as they don't really change
 398         # anything but the index
 399         # GH5632
 400         expected = DataFrame(columns=["foo"], index=Index([], dtype="object"))
 401

 403             df = DataFrame(index=Index([], dtype="object"))
 404             df["foo"] = Series([], dtype="object")

 410             df = DataFrame()

 417             df = DataFrame()
 418             df["foo"] = df.index
 419             return df
 420

 422

 429             return df
