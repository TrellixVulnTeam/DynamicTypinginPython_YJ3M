  93     def test_loc_with_slices(self):
  94 
  95         # loc with slices:
  96         #   - Interval objects: only works with exact matches
  97         #   - scalars: only works for non-overlapping, monotonic intervals,
  98         #     and start/stop select location based on the interval that
  99         #     contains them:
 100         #    (slice_loc(start, stop) == (idx.get_loc(start), idx.get_loc(stop))
 101 
 102         s = self.s
 103 
 104         # slice of interval
 105 
 106         expected = s.iloc[:3]
 107         result = s.loc[Interval(0, 1) : Interval(2, 3)]
 108         tm.assert_series_equal(expected, result)
 109         result = s[Interval(0, 1) : Interval(2, 3)]
 110         tm.assert_series_equal(expected, result)
 111 
 112         expected = s.iloc[3:]
 113         result = s.loc[Interval(3, 4) :]
 114         tm.assert_series_equal(expected, result)
 115         result = s[Interval(3, 4) :]
 116         tm.assert_series_equal(expected, result)
 117 
 118         msg = "Interval objects are not currently supported"
 119         with pytest.raises(NotImplementedError, match=msg):
 120             s.loc[Interval(3, 6) :]
 121 
 122         with pytest.raises(NotImplementedError, match=msg):
 123             s[Interval(3, 6) :]
 124 
 125         with pytest.raises(NotImplementedError, match=msg):
 126             s.loc[Interval(3, 4, closed="left") :]
 127 
 128         with pytest.raises(NotImplementedError, match=msg):
 129             s[Interval(3, 4, closed="left") :]
 130 
 131         # slice of scalar
 132 
 133         expected = s.iloc[:3]
 134         tm.assert_series_equal(expected, s.loc[:3])
 135         tm.assert_series_equal(expected, s.loc[:2.5])
 136         tm.assert_series_equal(expected, s.loc[0.1:2.5])
 137         tm.assert_series_equal(expected, s.loc[-1:3])
 138 
 139         tm.assert_series_equal(expected, s[:3])
 140         tm.assert_series_equal(expected, s[:2.5])
 141         tm.assert_series_equal(expected, s[0.1:2.5])

 170     def test_loc_with_overlap(self):
 171 
 172         idx = IntervalIndex.from_tuples([(1, 5), (3, 7)])
 173         s = Series(range(len(idx)), index=idx)
 174 
 175         # scalar
 176         expected = s
 177         result = s.loc[4]
 178         tm.assert_series_equal(expected, result)
 179 
 180         result = s[4]
 181         tm.assert_series_equal(expected, result)
 182 
 183         result = s.loc[[4]]
 184         tm.assert_series_equal(expected, result)
 185 
 186         result = s[[4]]
 187         tm.assert_series_equal(expected, result)
 188 
 189         # interval
 190         expected = 0
 191         result = s.loc[Interval(1, 5)]
 192         result == expected
 193 
 194         result = s[Interval(1, 5)]
 195         result == expected
 196 
 197         expected = s
 198         result = s.loc[[Interval(1, 5), Interval(3, 7)]]
 199         tm.assert_series_equal(expected, result)
 200 
 201         result = s[[Interval(1, 5), Interval(3, 7)]]
 202         tm.assert_series_equal(expected, result)
 203 
 204         with pytest.raises(KeyError, match=re.escape("Interval(3, 5, closed='right')")):
 205             s.loc[Interval(3, 5)]
 206 
 207         with pytest.raises(KeyError, match="^$"):
 208             s.loc[[Interval(3, 5)]]
 209 
 210         with pytest.raises(KeyError, match=re.escape("Interval(3, 5, closed='right')")):
 211             s[Interval(3, 5)]
 212 
 213         with pytest.raises(KeyError, match="^$"):
 214             s[[Interval(3, 5)]]
 215 
 216         # slices with interval (only exact matches)
 217         expected = s
 218         result = s.loc[Interval(1, 5) : Interval(3, 7)]
 219         tm.assert_series_equal(expected, result)
 220 
 221         result = s[Interval(1, 5) : Interval(3, 7)]
 222         tm.assert_series_equal(expected, result)
 223 
 224         msg = "'can only get slices from an IntervalIndex if bounds are"
 225         " non-overlapping and all monotonic increasing or decreasing'"
 226         with pytest.raises(KeyError, match=msg):
 227             s.loc[Interval(1, 6) : Interval(3, 8)]
 228 
 229         with pytest.raises(KeyError, match=msg):
 230             s[Interval(1, 6) : Interval(3, 8)]
 231 
 232         # slices with scalar raise for overlapping intervals
 233         # TODO KeyError is the appropriate error?
 234         with pytest.raises(KeyError, match=msg):
 235             s.loc[1:4]
