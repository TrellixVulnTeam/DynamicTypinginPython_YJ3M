  19 class ListDtype(ExtensionDtype):
  20     type = list
  21     name = "list"
  22     na_value = np.nan
  23 
  24     @classmethod
  25     def construct_array_type(cls) -> Type["ListArray"]:
  26         """
  27         Return the array type associated with this dtype.
  28 
  29         Returns
  30         -------
  31         type
  32         """
  33         return ListArray

  35 
  36 class ListArray(ExtensionArray):
  37     dtype = ListDtype()
  38     __array_priority__ = 1000
  39 
  40     def __init__(self, values, dtype=None, copy=False):
  41         if not isinstance(values, np.ndarray):
  42             raise TypeError("Need to pass a numpy array as values")
  43         for val in values:
  44             if not isinstance(val, self.dtype.type) and not pd.isna(val):
  45                 raise TypeError("All values must be of type " + str(self.dtype.type))
  46         self.data = values
  47 
  48     @classmethod
  49     def _from_sequence(cls, scalars, dtype=None, copy=False):
  50         data = np.empty(len(scalars), dtype=object)
  51         data[:] = scalars
  52         return cls(data)
  53 
  54     def __getitem__(self, item):
  55         if isinstance(item, numbers.Integral):
  56             return self.data[item]
  57         else:
  58             # slice, list-like, mask
  59             return type(self)(self.data[item])
  60 
  61     def __len__(self) -> int:
  62         return len(self.data)
  63 
  64     def isna(self):
  65         return np.array(
  66             [not isinstance(x, list) and np.isnan(x) for x in self.data], dtype=bool
  67         )
  68 
  69     def take(self, indexer, allow_fill=False, fill_value=None):
  70         # re-implement here, since NumPy has trouble setting
  71         # sized objects like UserDicts into scalar slots of
  72         # an ndarary.
  73         indexer = np.asarray(indexer)
  74         msg = (
  75             "Index is out of bounds or cannot do a "
  76             "non-empty take from an empty array."
  77         )
  78 
  79         if allow_fill:
  80             if fill_value is None:
  81                 fill_value = self.dtype.na_value
  82             # bounds check
  83             if (indexer < -1).any():
  84                 raise ValueError
  85             try:
  86                 output = [
  87                     self.data[loc] if loc != -1 else fill_value for loc in indexer
  88                 ]
  89             except IndexError as err:
  90                 raise IndexError(msg) from err
  91         else:
  92             try:
  93                 output = [self.data[loc] for loc in indexer]
  94             except IndexError as err:
  95                 raise IndexError(msg) from err
  96 
  97         return self._from_sequence(output)
  98 
  99     def copy(self):
 100         return type(self)(self.data[:])
 101 
 102     def astype(self, dtype, copy=True):
 103         if isinstance(dtype, type(self.dtype)) and dtype == self.dtype:
 104             if copy:
 105                 return self.copy()
 106             return self
 107         elif pd.api.types.is_string_dtype(dtype) and not pd.api.types.is_object_dtype(
 108             dtype
 109         ):
 110             # numpy has problems with astype(str) for nested elements
 111             return np.array([str(x) for x in self.data], dtype=dtype)
 112         return np.array(self.data, dtype=dtype, copy=copy)
 113 
 114     @classmethod
 115     def _concat_same_type(cls, to_concat):
 116         data = np.concatenate([x.data for x in to_concat])
 117         return cls(data)
