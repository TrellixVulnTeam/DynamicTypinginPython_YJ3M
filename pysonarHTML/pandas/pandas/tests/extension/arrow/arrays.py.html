  70 class ArrowExtensionArray(ExtensionArray):
  71     @classmethod
  72     def from_scalars(cls, values):
  73         arr = pa.chunked_array([pa.array(np.asarray(values))])
  74         return cls(arr)
  75 
  76     @classmethod
  77     def from_array(cls, arr):
  78         assert isinstance(arr, pa.Array)
  79         return cls(pa.chunked_array([arr]))
  80 
  81     @classmethod
  82     def _from_sequence(cls, scalars, dtype=None, copy=False):
  83         return cls.from_scalars(scalars)
  84 
  85     def __repr__(self):
  86         return f"{type(self).__name__}({repr(self._data)})"
  87 
  88     def __getitem__(self, item):
  89         if pd.api.types.is_scalar(item):
  90             return self._data.to_pandas()[item]
  91         else:
  92             vals = self._data.to_pandas()[item]
  93             return type(self).from_scalars(vals)
  94 
  95     def __len__(self):
  96         return len(self._data)
  97 
  98     def astype(self, dtype, copy=True):
  99         # needed to fix this astype for the Series constructor.
 100         if isinstance(dtype, type(self.dtype)) and dtype == self.dtype:
 101             if copy:
 102                 return self.copy()
 103             return self
 104         return super().astype(dtype, copy)
 105 
 106     @property
 107     def dtype(self):
 108         return self._dtype
 109 
 110     def _boolean_op(self, other, op):
 111         if not isinstance(other, type(self)):
 112             raise NotImplementedError()
 113 
 114         result = op(np.array(self._data), np.array(other._data))
 115         return ArrowBoolArray(
 116             pa.chunked_array([pa.array(result, mask=pd.isna(self._data.to_pandas()))])
 117         )
 118 
 119     def __eq__(self, other):
 120         if not isinstance(other, type(self)):
 121             return False
 122 
 123         return self._boolean_op(other, operator.eq)
 124 
 125     def __and__(self, other):
 126         return self._boolean_op(other, operator.and_)
 127 
 128     def __or__(self, other):
 129         return self._boolean_op(other, operator.or_)
 130 
 131     @property
 132     def nbytes(self):
 133         return sum(
 134             x.size
 135             for chunk in self._data.chunks
 136             for x in chunk.buffers()
 137             if x is not None
 138         )
 139 
 140     def isna(self):
 141         nas = pd.isna(self._data.to_pandas())
 142         return type(self).from_scalars(nas)
 143 
 144     def take(self, indices, allow_fill=False, fill_value=None):
 145         data = self._data.to_pandas()
 146 
 147         if allow_fill and fill_value is None:
 148             fill_value = self.dtype.na_value
 149 
 150         result = take(data, indices, fill_value=fill_value, allow_fill=allow_fill)
 151         return self._from_sequence(result, dtype=self.dtype)
 152 
 153     def copy(self):
 154         return type(self)(copy.copy(self._data))
 155 
 156     @classmethod
 157     def _concat_same_type(cls, to_concat):
 158         chunks = list(itertools.chain.from_iterable(x._data.chunks for x in to_concat))
 159         arr = pa.chunked_array(chunks)
 160         return cls(arr)
 161 
 162     def __invert__(self):
 163         return type(self).from_scalars(~self._data.to_pandas())
 164 
 165     def _reduce(self, method, skipna=True, **kwargs):
 166         if skipna:
 167             arr = self[~self.isna()]
 168         else:
 169             arr = self
 170 
 171         try:
 172             op = getattr(arr, method)
 173         except AttributeError as err:
 174             raise TypeError from err
 175         return op(**kwargs)
 176 
 177     def any(self, axis=0, out=None):
 178         # Explicitly return a plain bool to reproduce GH-34660
 179         return bool(self._data.to_pandas().any())
 180 
 181     def all(self, axis=0, out=None):
 182         # Explicitly return a plain bool to reproduce GH-34660
 183         return bool(self._data.to_pandas().all())

 186 class ArrowBoolArray(ArrowExtensionArray):
 187     def __init__(self, values):
 188         if not isinstance(values, pa.ChunkedArray):
 189             raise ValueError
 190 
 191         assert values.type == pa.bool_()
 192         self._data = values
 193         self._dtype = ArrowBoolDtype()
