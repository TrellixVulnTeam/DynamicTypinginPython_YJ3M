  15 def make_data(fill_value):
  16     if np.isnan(fill_value):
  17         data = np.random.uniform(size=100)
  18     else:
  19         data = np.random.randint(1, 100, size=100)
  20         if data[0] == data[1]:
  21             data[0] += 1
  22 
  23     data[2::3] = fill_value
  24     return data

 237     def test_combine_le(self, data_repeated):
 238         # We return a Series[SparseArray].__le__ returns a
 239         # Series[Sparse[bool]]
 240         # rather than Series[bool]
 241         orig_data1, orig_data2 = data_repeated(2)
 242         s1 = pd.Series(orig_data1)
 243         s2 = pd.Series(orig_data2)
 244         result = s1.combine(s2, lambda x1, x2: x1 <= x2)
 245         expected = pd.Series(
 246             SparseArray(
 247                 [a <= b for (a, b) in zip(list(orig_data1), list(orig_data2))],
 248                 fill_value=False,
 249             )
 250         )
 251         self.assert_series_equal(result, expected)
 252 
 253         val = s1.iloc[0]
 254         result = s1.combine(val, lambda x1, x2: x1 <= x2)
 255         expected = pd.Series(
 256             SparseArray([a <= val for a in list(orig_data1)], fill_value=False)
 257         )
 258         self.assert_series_equal(result, expected)

 284     def test_where_series(self, data, na_value):
 285         assert data[0] != data[1]
 286         cls = type(data)
 287         a, b = data[:2]
 288 
 289         ser = pd.Series(cls._from_sequence([a, a, b, b], dtype=data.dtype))
 290 
 291         cond = np.array([True, True, False, False])
 292         result = ser.where(cond)
 293 
 294         new_dtype = SparseDtype("float", 0.0)
 295         expected = pd.Series(
 296             cls._from_sequence([a, a, na_value, na_value], dtype=new_dtype)
 297         )
 298         self.assert_series_equal(result, expected)
 299 
 300         other = cls._from_sequence([a, b, a, b], dtype=data.dtype)
 301         cond = np.array([True, False, True, True])
 302         result = ser.where(cond, other)
 303         expected = pd.Series(cls._from_sequence([a, b, b, b], dtype=data.dtype))
 304         self.assert_series_equal(result, expected)

 389         op = self.get_op_from_name(op_name)
