 181     @pytest.mark.parametrize("obj", ["series", "frame"])
 182     def test_unstack(self, obj):
 183         # GH-13287: can't use base test, since building the expected fails.
 184         data = DatetimeArray._from_sequence(
 185             ["2000", "2001", "2002", "2003"], tz="US/Central"
 186         )
 187         index = pd.MultiIndex.from_product(([["A", "B"], ["a", "b"]]), names=["a", "b"])
 188 
 189         if obj == "series":
 190             ser = pd.Series(data, index=index)
 191             expected = pd.DataFrame(
 192                 {"A": data.take([0, 1]), "B": data.take([2, 3])},
 193                 index=pd.Index(["a", "b"], name="b"),
 194             )
 195             expected.columns.name = "a"
 196 
 197         else:
 198             ser = pd.DataFrame({"A": data, "B": data}, index=index)
 199             expected = pd.DataFrame(
 200                 {
 201                     ("A", "A"): data.take([0, 1]),
 202                     ("A", "B"): data.take([2, 3]),
 203                     ("B", "A"): data.take([0, 1]),
 204                     ("B", "B"): data.take([2, 3]),
 205                 },
 206                 index=pd.Index(["a", "b"], name="b"),
 207             )
 208             expected.columns.names = [None, "a"]
 209 
 210         result = ser.unstack(0)
 211         self.assert_equal(result, expected)
