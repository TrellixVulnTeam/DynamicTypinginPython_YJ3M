 132         # GH 20825

 139             ([a + b for (a, b) in zip(list(orig_data1), list(orig_data2))])

 236 class TestComparisonOps(base.BaseComparisonOpsTests):
 237     def _compare_other(self, s, data, op_name, other):
 238         op = self.get_op_from_name(op_name)
 239         if op_name == "__eq__":
 240             result = op(s, other)
 241             expected = s.combine(other, lambda x, y: x == y)
 242             assert (result == expected).all()
 243 
 244         elif op_name == "__ne__":
 245             result = op(s, other)
 246             expected = s.combine(other, lambda x, y: x != y)
 247             assert (result == expected).all()
 248 
 249         else:
 250             msg = "Unordered Categoricals can only compare equality or not"
 251             with pytest.raises(TypeError, match=msg):
 252                 op(data, other)
 253 
 254     @pytest.mark.parametrize(

 256         [["a", "b"], [0, 1], [pd.Timestamp("2019"), pd.Timestamp("2020")]],
 257     )
