 109 class TestArithmeticOps(base.BaseArithmeticOpsTests):
 110     def check_opname(self, s, op_name, other, exc=None):
 111         # overwriting to indicate ops don't raise an error
 112         super().check_opname(s, op_name, other, exc=None)
 113 
 114     def _check_op(self, s, op, other, op_name, exc=NotImplementedError):
 115         if exc is None:
 116             if s.dtype.is_unsigned_integer and (op_name == "__rsub__"):
 117                 # TODO see https://github.com/pandas-dev/pandas/issues/22023
 118                 pytest.skip("unsigned subtraction gives negative values")
 119 
 120             if (
 121                 hasattr(other, "dtype")
 122                 and not is_extension_array_dtype(other.dtype)
 123                 and pd.api.types.is_integer_dtype(other.dtype)
 124             ):
 125                 # other is np.int64 and would therefore always result in
 126                 # upcasting, so keeping other as same numpy_dtype
 127                 other = other.astype(s.dtype.numpy_dtype)
 128 
 129             result = op(s, other)
 130             expected = s.combine(other, op)
 131 
 132             if op_name in ("__rtruediv__", "__truediv__", "__div__"):
 133                 expected = expected.fillna(np.nan).astype(float)
 134                 if op_name == "__rtruediv__":
 135                     # TODO reverse operators result in object dtype
 136                     result = result.astype(float)
 137             elif op_name.startswith("__r"):
 138                 # TODO reverse operators result in object dtype
 139                 # see https://github.com/pandas-dev/pandas/issues/22024
 140                 expected = expected.astype(s.dtype)
 141                 result = result.astype(s.dtype)
 142             else:
 143                 # combine method result in 'biggest' (int64) dtype
 144                 expected = expected.astype(s.dtype)
 145                 pass
 146 
 147             if (op_name == "__rpow__") and isinstance(other, pd.Series):
 148                 # TODO pow on Int arrays gives different result with NA
 149                 # see https://github.com/pandas-dev/pandas/issues/22022
 150                 result = result.fillna(1)
 151 
 152             self.assert_series_equal(result, expected)
 153         else:
 154             with pytest.raises(exc):
 155                 op(s, other)
 156 
 157     def _check_divmod_op(self, s, op, other, exc=None):
 158         super()._check_divmod_op(s, op, other, None)

 212 class TestMethods(base.BaseMethodsTests):
 213     @pytest.mark.skip(reason="uses nullable integer")
 214     def test_value_counts(self, all_data, dropna):
 215         all_data = all_data[:10]
 216         if dropna:
 217             other = np.array(all_data[~all_data.isna()])
 218         else:
 219             other = all_data
 220 
 221         result = pd.Series(all_data).value_counts(dropna=dropna).sort_index()
 222         expected = pd.Series(other).value_counts(dropna=dropna).sort_index()
 223         expected.index = expected.index.astype(all_data.dtype)
 224 
 225         self.assert_series_equal(result, expected)

 236 class TestNumericReduce(base.BaseNumericReduceTests):
 237     def check_reduce(self, s, op_name, skipna):
 238         # overwrite to ensure pd.NA is tested instead of np.nan
 239         # https://github.com/pandas-dev/pandas/issues/30958
 240         result = getattr(s, op_name)(skipna=skipna)
 241         if not skipna and s.isna().any():
 242             expected = pd.NA
 243         else:
 244             expected = getattr(s.dropna().astype("int64"), op_name)(skipna=skipna)
 245         tm.assert_almost_equal(result, expected)
