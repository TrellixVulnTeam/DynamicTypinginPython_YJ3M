  16 class DummyArray(ExtensionArray):
  17     def __init__(self, data):
  18         self.data = data
  19 
  20     def __array__(self, dtype):
  21         return self.data
  22 
  23     @property
  24     def dtype(self):
  25         return DummyDtype()
  26 
  27     def astype(self, dtype, copy=True):
  28         # we don't support anything but a single dtype
  29         if isinstance(dtype, DummyDtype):
  30             if copy:
  31                 return type(self)(self.data)
  32             return self
  33 
  34         return np.array(self, dtype=dtype, copy=copy)

  68 def test_astype_no_copy():
