  18 @pytest.fixture
  19 def allow_in_pandas(monkeypatch):
  20     """
  21     A monkeypatch to tells pandas to let us in.
  22 
  23     By default, passing a PandasArray to an index / series / frame
  24     constructor will unbox that PandasArray to an ndarray, and treat
  25     it as a non-EA column. We don't want people using EAs without
  26     reason.
  27 
  28     The mechanism for this is a check against ABCPandasArray
  29     in each constructor.
  30 
  31     But, for testing, we need to allow them in pandas. So we patch
  32     the _typ of PandasArray, so that we evade the ABCPandasArray
  33     check.
  34     """
  35     with monkeypatch.context() as m:
  36         m.setattr(PandasArray, "_typ", "extension")
  37         yield
  38
