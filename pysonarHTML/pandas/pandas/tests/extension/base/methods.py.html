 261     @pytest.mark.parametrize("periods", [1, -2])
 262     def test_diff(self, data, periods):
 263         data = data[:5]
 264         if is_bool_dtype(data.dtype):
 265             op = operator.xor
 266         else:
 267             op = operator.sub
 268         try:
 269             # does this array implement ops?
 270             op(data, data)
 271         except Exception:
 272             pytest.skip(f"{type(data)} does not support diff")

 275         expected = pd.Series(op(data, data.shift(periods)))
 276         self.assert_series_equal(result, expected)

 278         df = pd.DataFrame({"A": data, "B": [1.0] * 5})
 279         result = df.diff(periods)
 280         if periods == 1:
 281             b = [np.nan, 0, 0, 0, 0]
 282         else:
 283             b = [0, 0, 0, np.nan, np.nan]
 284         expected = pd.DataFrame({"A": expected, "B": b})

 427     @pytest.mark.parametrize("box", [pd.array, pd.Series, pd.DataFrame])
 428     def test_equals(self, data, na_value, as_series, box):
 429         data2 = type(data)._from_sequence([data[0]] * len(data), dtype=data.dtype)
 430         data_na = type(data)._from_sequence([na_value] * len(data), dtype=data.dtype)
 431 
 432         data = tm.box_expected(data, box, transpose=False)
 433         data2 = tm.box_expected(data2, box, transpose=False)
 434         data_na = tm.box_expected(data_na, box, transpose=False)
 435 
 436         # we are asserting with `is True/False` explicitly, to test that the
 437         # result is an actual Python bool, and not something "truthy"
 438 
 439         assert data.equals(data) is True
 440         assert data.equals(data.copy()) is True
 441 
 442         # unequal other data
 443         assert data.equals(data2) is False
 444         assert data.equals(data_na) is False
 445 
 446         # different length
 447         assert data[:2].equals(data[:3]) is False
 448 
 449         # emtpy are equal
 450         assert data[:0].equals(data[:0]) is True
 451 
 452         # other types
 453         assert data.equals(None) is False
 454         assert data[[0]].equals(data[0]) is False
