  24         df = pd.DataFrame({"A": [1, 1, 2, 2, 3, 3, 1, 4], "B": data_for_grouping})
  25         result = df.groupby("B", as_index=as_index).A.mean()
  26         _, index = pd.factorize(data_for_grouping, sort=True)
  27 
  28         index = pd.Index(index, name="B")

  30         if as_index:

  33             expected = expected.reset_index()
