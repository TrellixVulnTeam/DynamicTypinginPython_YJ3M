  11 class BaseReduceTests(BaseExtensionTests):
  12     """
  13     Reduction specific tests. Generally these only
  14     make sense for numeric/boolean operations.
  15     """
  16 
  17     def check_reduce(self, s, op_name, skipna):
  18         result = getattr(s, op_name)(skipna=skipna)
  19         expected = getattr(s.astype("float64"), op_name)(skipna=skipna)
  20         tm.assert_almost_equal(result, expected)
