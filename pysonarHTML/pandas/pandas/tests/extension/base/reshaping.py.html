 241         df = pd.DataFrame({"A": data[:5], "B": data[:5]})
 242         df.columns = columns
 243         result = df.stack()
 244         expected = df.astype(object).stack()
 245         # we need a second astype(object), in case the constructor inferred
 246         # object -> specialized, as is done for period.
 247         expected = expected.astype(object)
 248 
 249         if isinstance(expected, pd.Series):
 250             assert result.dtype == df.iloc[:, 0].dtype
 251         else:
 252             assert all(result.dtypes == df.iloc[:, 0].dtype)
 253 
 254         result = result.astype(object)
 255         self.assert_equal(result, expected)

 277     @pytest.mark.parametrize("obj", ["series", "frame"])
 278     def test_unstack(self, data, index, obj):
 279         data = data[: len(index)]
 280         if obj == "series":
 281             ser = pd.Series(data, index=index)
 282         else:
 283             ser = pd.DataFrame({"A": data, "B": data}, index=index)
 284 
 285         n = index.nlevels
 286         levels = list(range(n))
 287         # [0, 1, 2]
 288         # [(0,), (1,), (2,), (0, 1), (0, 2), (1, 0), (1, 2), (2, 0), (2, 1)]
 289         combinations = itertools.chain.from_iterable(
 290             itertools.permutations(levels, i) for i in range(1, n)
 291         )
 292 
 293         for level in combinations:
 294             result = ser.unstack(level=level)
 295             assert all(
 296                 isinstance(result[col].array, type(data)) for col in result.columns
 297             )
 298 
 299             if obj == "series":
 300                 # We should get the same result with to_frame+unstack+droplevel
 301                 df = ser.to_frame()
 302 
 303                 alt = df.unstack(level=level).droplevel(0, axis=1)
 304                 self.assert_frame_equal(result, alt)
 305 
 306             expected = ser.astype(object).unstack(level=level)
 307             result = result.astype(object)
 308 
 309             self.assert_frame_equal(result, expected)

 320     def test_transpose(self, data):
 321         df = pd.DataFrame({"A": data[:4], "B": data[:4]}, index=["a", "b", "c", "d"])
 322         result = df.T
 323         expected = pd.DataFrame(
 324             {
 325                 "a": type(data)._from_sequence([data[0]] * 2, dtype=data.dtype),
 326                 "b": type(data)._from_sequence([data[1]] * 2, dtype=data.dtype),
 327                 "c": type(data)._from_sequence([data[2]] * 2, dtype=data.dtype),
 328                 "d": type(data)._from_sequence([data[3]] * 2, dtype=data.dtype),
 329             },
 330             index=["A", "B"],
 331         )
 332         self.assert_frame_equal(result, expected)
 333         self.assert_frame_equal(np.transpose(np.transpose(df)), df)
 334         self.assert_frame_equal(np.transpose(np.transpose(df[["A"]])), df[["A"]])
