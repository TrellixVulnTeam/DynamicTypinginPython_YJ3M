  45     def test_is_extension_array_dtype(self, data):

  51     def test_no_values_attribute(self, data):

  55         assert not hasattr(data, "_values")

  57     def test_is_numeric_honored(self, data):

  61     def test_isna_extension_array(self, data_missing):
  62         # If your `isna` returns an ExtensionArray, you must also implement
  63         # _reduce. At the *very* least, you must implement any and all
  64         na = data_missing.isna()
  65         if is_extension_array_dtype(na):
  66             assert na._reduce("any")
  67             assert na.any()
  68 
  69             assert not na._reduce("all")
  70             assert not na.all()
  71 
  72             assert na.dtype._is_boolean
  73 
  74     def test_copy(self, data):
  75         # GH#27083 removing deep keyword from EA.copy
  76         assert data[0] != data[1]
  77         result = data.copy()
  78 
  79         data[1] = data[0]
  80         assert result[1] != result[0]

  82     def test_view(self, data):
  83         # view with no dtype should return a shallow copy, *not* the same
  84         #  object
  85         assert data[1] != data[0]
  86 
  87         result = data.view()
  88         assert result is not data
  89         assert type(result) == type(data)
  90 
  91         result[1] = result[0]
  92         assert data[1] == data[0]
  93 
  94         # check specifically that the `dtype` kwarg is accepted
  95         data.view(dtype=None)
