  29         value = [data[0]]
  30         if as_array:
  31             value = data._from_sequence(value)
  32

  37         self.assert_series_equal(ser, original)

 181         ser = pd.Series(data)
 182         mask = np.zeros(len(data), dtype=bool)
 183         mask[:2] = True
 184 
 185         if as_callable:
 186             mask2 = lambda x: mask
 187         else:
 188             mask2 = mask
 189 
 190         if setter:
 191             # loc
 192             target = getattr(ser, setter)
 193         else:
 194             # Series.__setitem__

 196 
 197         target[mask2] = data[5:7]

 199         ser[mask2] = data[5:7]

 201         assert ser[1] == data[6]
