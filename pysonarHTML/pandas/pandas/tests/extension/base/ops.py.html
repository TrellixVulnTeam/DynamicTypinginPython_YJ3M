  12 class BaseOpsUtil(BaseExtensionTests):
  13     def get_op_from_name(self, op_name):
  14         return tm.get_op_from_name(op_name)
  15 
  16     def check_opname(self, s, op_name, other, exc=Exception):
  17         op = self.get_op_from_name(op_name)
  18 
  19         self._check_op(s, op, other, op_name, exc)
  20 
  21     def _check_op(self, s, op, other, op_name, exc=NotImplementedError):
  22         if exc is None:
  23             result = op(s, other)
  24             if isinstance(s, pd.DataFrame):
  25                 if len(s.columns) != 1:
  26                     raise NotImplementedError
  27                 expected = s.iloc[:, 0].combine(other, op).to_frame()
  28                 self.assert_frame_equal(result, expected)
  29             else:
  30                 expected = s.combine(other, op)
  31                 self.assert_series_equal(result, expected)
  32         else:
  33             with pytest.raises(exc):
  34                 op(s, other)
  35 
  36     def _check_divmod_op(self, s, op, other, exc=Exception):
  37         # divmod has multiple return values, so check separately
  38         if exc is None:
  39             result_div, result_mod = op(s, other)
  40             if op is divmod:
  41                 expected_div, expected_mod = s // other, s % other
  42             else:
  43                 expected_div, expected_mod = other // s, other % s
  44             self.assert_series_equal(result_div, expected_div)
  45             self.assert_series_equal(result_mod, expected_mod)
  46         else:
  47             with pytest.raises(exc):
  48                 divmod(s, other)

  51 class BaseArithmeticOpsTests(BaseOpsUtil):
  52     """
  53     Various Series and DataFrame arithmetic ops methods.
  54 
  55     Subclasses supporting various ops should set the class variables
  56     to indicate that they support ops of that kind
  57 
  58     * series_scalar_exc = TypeError
  59     * frame_scalar_exc = TypeError
  60     * series_array_exc = TypeError
  61     * divmod_exc = TypeError

  69     def test_arith_series_with_scalar(self, data, all_arithmetic_operators):
  70         # series & scalar

 117     def test_direct_arith_with_series_returns_not_implemented(self, data):
 118         # EAs should return NotImplemented for ops with Series.
 119         # Pandas takes care of unboxing the series and calling the EA's op.
 120         other = pd.Series(data)
 121         if hasattr(data, "__add__"):
 122             result = data.__add__(other)
 123             assert result is NotImplemented
 124         else:
 125             raise pytest.skip(f"{type(data).__name__} does not implement add")

 131     def _compare_other(self, s, data, op_name, other):
 132         op = self.get_op_from_name(op_name)
 133         if op_name == "__eq__":
 134             assert not op(s, other).all()
 135         elif op_name == "__ne__":
 136             assert op(s, other).all()
 137 
 138         else:
 139 
 140             # array
 141             assert getattr(data, op_name)(other) is NotImplemented
 142 
 143             # series
 144             s = pd.Series(data)
 145             with pytest.raises(TypeError):
 146                 op(s, other)
 147 
 148     def test_compare_scalar(self, data, all_compare_operators):
 149         op_name = all_compare_operators
 150         s = pd.Series(data)
 151         self._compare_other(s, data, op_name, 0)
 152 
 153     def test_compare_array(self, data, all_compare_operators):
 154         op_name = all_compare_operators
 155         s = pd.Series(data)
 156         other = pd.Series([data[0]] * len(data))
 157         self._compare_other(s, data, op_name, other)
