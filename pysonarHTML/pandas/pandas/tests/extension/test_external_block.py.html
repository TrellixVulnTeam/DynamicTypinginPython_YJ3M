   9 class CustomBlock(ExtensionBlock):

  11     _holder = np.ndarray

  14     def concat_same_type(self, to_concat, placement=None):
  15         """
  16         Always concatenate disregarding self.ndim as the values are
  17         always 1D in this custom Block
  18         """
  19         values = np.concatenate([blk.values for blk in to_concat])
  20         placement = self.mgr_locs if self.ndim == 2 else slice(len(values))
  21         return self.make_block_same_class(values, placement=placement)
