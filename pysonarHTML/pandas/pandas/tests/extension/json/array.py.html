   1 """
   2 Test extension array for storing nested data in a pandas container.
   3 
   4 The JSONArray stores lists of dictionaries. The storage mechanism is a list,
   5 not an ndarray.
   6 
   7 Note
   8 ----
   9 We currently store lists of UserDicts. Pandas has a few places
  10 internally that specifically check for dicts, and does non-scalar things
  11 in that case. We *want* the dictionaries to be treated as scalars, so we
  12 hack around pandas by using UserDicts.
  13 """
  14 from collections import UserDict, abc
  15 import itertools
  16 import numbers
  17 import random
  18 import string
  19 import sys
  20 from typing import Any, Mapping, Type
  21 
  22 import numpy as np
  23 
  24 from pandas.core.dtypes.common import pandas_dtype
  25 
  26 import pandas as pd
  27 from pandas.api.extensions import ExtensionArray, ExtensionDtype
  28 
  29 
  30 class JSONDtype(ExtensionDtype):
  31     type = abc.Mapping
  32     name = "json"
  33     na_value: Mapping[str, Any] = UserDict()
  34 
  35     @classmethod
  36     def construct_array_type(cls) -> Type["JSONArray"]:
  37         """
  38         Return the array type associated with this dtype.
  39 
  40         Returns
  41         -------
  42         type
  43         """
  44         return JSONArray
  45 
  46 
  47 class JSONArray(ExtensionArray):
  48     dtype = JSONDtype()
  49     __array_priority__ = 1000
  50 
  51     def __init__(self, values, dtype=None, copy=False):
  52         for val in values:
  53             if not isinstance(val, self.dtype.type):
  54                 raise TypeError("All values must be of type " + str(self.dtype.type))
  55         self.data = values
  56 
  57         # Some aliases for common attribute names to ensure pandas supports
  58         # these
  59         self._items = self._data = self.data
  60         # those aliases are currently not working due to assumptions
  61         # in internal code (GH-20735)
  62         # self._values = self.values = self.data
  63 
  64     @classmethod
  65     def _from_sequence(cls, scalars, dtype=None, copy=False):
  66         return cls(scalars)
  67 
  68     @classmethod
  69     def _from_factorized(cls, values, original):
  70         return cls([UserDict(x) for x in values if x != ()])
  71 
  72     def __getitem__(self, item):
  73         if isinstance(item, numbers.Integral):
  74             return self.data[item]
  75         elif isinstance(item, slice) and item == slice(None):
  76             # Make sure we get a view
  77             return type(self)(self.data)
  78         elif isinstance(item, slice):
  79             # slice
  80             return type(self)(self.data[item])
  81         else:
  82             item = pd.api.indexers.check_array_indexer(self, item)
  83             if pd.api.types.is_bool_dtype(item.dtype):
  84                 return self._from_sequence([x for x, m in zip(self, item) if m])
  85             # integer
  86             return type(self)([self.data[i] for i in item])
  87 
  88     def __setitem__(self, key, value):
  89         if isinstance(key, numbers.Integral):
  90             self.data[key] = value
  91         else:
  92             if not isinstance(value, (type(self), abc.Sequence)):
  93                 # broadcast value
  94                 value = itertools.cycle([value])
  95 
  96             if isinstance(key, np.ndarray) and key.dtype == "bool":
  97                 # masking
  98                 for i, (k, v) in enumerate(zip(key, value)):
  99                     if k:
 100                         assert isinstance(v, self.dtype.type)
 101                         self.data[i] = v
 102             else:
 103                 for k, v in zip(key, value):
 104                     assert isinstance(v, self.dtype.type)
 105                     self.data[k] = v
 106 
 107     def __len__(self) -> int:
 108         return len(self.data)

 110     def __eq__(self, other):
 111         return NotImplemented

 114         return NotImplemented

 116     def __array__(self, dtype=None):
 117         if dtype is None:
 118             dtype = object
 119         return np.asarray(self.data, dtype=dtype)
 120 
 121     @property
 122     def nbytes(self) -> int:
 123         return sys.getsizeof(self.data)
 124 
 125     def isna(self):
 126         return np.array([x == self.dtype.na_value for x in self.data], dtype=bool)
 127 
 128     def take(self, indexer, allow_fill=False, fill_value=None):
 129         # re-implement here, since NumPy has trouble setting
 130         # sized objects like UserDicts into scalar slots of
 131         # an ndarary.
 132         indexer = np.asarray(indexer)
 133         msg = (
 134             "Index is out of bounds or cannot do a "
 135             "non-empty take from an empty array."
 136         )
 137 
 138         if allow_fill:
 139             if fill_value is None:
 140                 fill_value = self.dtype.na_value
 141             # bounds check
 142             if (indexer < -1).any():
 143                 raise ValueError
 144             try:
 145                 output = [
 146                     self.data[loc] if loc != -1 else fill_value for loc in indexer
 147                 ]
 148             except IndexError as err:
 149                 raise IndexError(msg) from err
 150         else:
 151             try:
 152                 output = [self.data[loc] for loc in indexer]
 153             except IndexError as err:
 154                 raise IndexError(msg) from err
 155 
 156         return self._from_sequence(output)
 157 
 158     def copy(self):
 159         return type(self)(self.data[:])
 160 
 161     def astype(self, dtype, copy=True):
 162         # NumPy has issues when all the dicts are the same length.
 163         # np.array([UserDict(...), UserDict(...)]) fails,
 164         # but np.array([{...}, {...}]) works, so cast.
 165         from pandas.core.arrays.string_ import StringDtype
 166 
 167         dtype = pandas_dtype(dtype)
 168         # needed to add this check for the Series constructor
 169         if isinstance(dtype, type(self.dtype)) and dtype == self.dtype:
 170             if copy:
 171                 return self.copy()
 172             return self
 173         elif isinstance(dtype, StringDtype):
 174             value = self.astype(str)  # numpy doesn'y like nested dicts
 175             return dtype.construct_array_type()._from_sequence(value, copy=False)
 176 
 177         return np.array([dict(x) for x in self], dtype=dtype, copy=copy)
 178 
 179     def unique(self):
 180         # Parent method doesn't work since np.array will try to infer
 181         # a 2-dim object.
 182         return type(self)(
 183             [dict(x) for x in list({tuple(d.items()) for d in self.data})]
 184         )
 185 
 186     @classmethod
 187     def _concat_same_type(cls, to_concat):
 188         data = list(itertools.chain.from_iterable([x.data for x in to_concat]))
 189         return cls(data)
 190 
 191     def _values_for_factorize(self):
 192         frozen = self._values_for_argsort()
 193         if len(frozen) == 0:
 194             # _factorize_array expects 1-d array, this is a len-0 2-d array.
 195             frozen = frozen.ravel()
 196         return frozen, ()
 197 
 198     def _values_for_argsort(self):
 199         # Disable NumPy's shape inference by including an empty tuple...
 200         # If all the elements of self are the same size P, NumPy will
 201         # cast them to an (N, P) array, instead of an (N,) array of tuples.
 202         frozen = [()] + [tuple(x.items()) for x in self]
 203         return np.array(frozen, dtype=object)[1:]
