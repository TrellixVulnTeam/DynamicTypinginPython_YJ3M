  77 class BaseJSON:
  78     # NumPy doesn't handle an array of equal-length UserDicts.
  79     # The default assert_series_equal eventually does a
  80     # Series.values, which raises. We work around it by
  81     # converting the UserDicts to dicts.
  82     @classmethod
  83     def assert_series_equal(cls, left, right, *args, **kwargs):
  84         if left.dtype.name == "json":
  85             assert left.dtype == right.dtype
  86             left = pd.Series(
  87                 JSONArray(left.values.astype(object)), index=left.index, name=left.name
  88             )
  89             right = pd.Series(
  90                 JSONArray(right.values.astype(object)),
  91                 index=right.index,
  92                 name=right.name,
  93             )
  94         tm.assert_series_equal(left, right, *args, **kwargs)
  95 
  96     @classmethod
  97     def assert_frame_equal(cls, left, right, *args, **kwargs):
  98         obj_type = kwargs.get("obj", "DataFrame")
  99         tm.assert_index_equal(
 100             left.columns,
 101             right.columns,
 102             exact=kwargs.get("check_column_type", "equiv"),
 103             check_names=kwargs.get("check_names", True),
 104             check_exact=kwargs.get("check_exact", False),
 105             check_categorical=kwargs.get("check_categorical", True),
 106             obj=f"{obj_type}.columns",
 107         )
 108 
 109         jsons = (left.dtypes == "json").index
 110 
 111         for col in jsons:
 112             cls.assert_series_equal(left[col], right[col], *args, **kwargs)
 113 
 114         left = left.drop(columns=jsons)
 115         right = right.drop(columns=jsons)
 116         tm.assert_frame_equal(left, right, *args, **kwargs)

 123 class TestInterface(BaseJSON, base.BaseInterfaceTests):
 124     def test_custom_asserts(self):
 125         # This would always trigger the KeyError from trying to put
 126         # an array of equal-length UserDicts inside an ndarray.
 127         data = JSONArray(
 128             [
 129                 collections.UserDict({"a": 1}),
 130                 collections.UserDict({"b": 2}),
 131                 collections.UserDict({"c": 3}),
 132             ]
 133         )
 134         a = pd.Series(data)
 135         self.assert_series_equal(a, a)
 136         self.assert_frame_equal(a.to_frame(), a.to_frame())
 137 
 138         b = pd.Series(data.take([0, 0, 1]))
 139         msg = r"ExtensionArray are different"
 140         with pytest.raises(AssertionError, match=msg):
 141             self.assert_series_equal(a, b)
 142 
 143         with pytest.raises(AssertionError, match=msg):
 144             self.assert_frame_equal(a.to_frame(), b.to_frame())
