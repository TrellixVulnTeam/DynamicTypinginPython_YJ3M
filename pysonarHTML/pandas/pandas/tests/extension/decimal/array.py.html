  47 class DecimalArray(ExtensionArray, ExtensionScalarOpsMixin):
  48     __array_priority__ = 1000
  49 
  50     def __init__(self, values, dtype=None, copy=False, context=None):
  51         for val in values:
  52             if not isinstance(val, decimal.Decimal):
  53                 raise TypeError("All values must be of type " + str(decimal.Decimal))
  54         values = np.asarray(values, dtype=object)
  55 
  56         self._data = values
  57         # Some aliases for common attribute names to ensure pandas supports
  58         # these
  59         self._items = self.data = self._data
  60         # those aliases are currently not working due to assumptions
  61         # in internal code (GH-20735)
  62         # self._values = self.values = self.data
  63         self._dtype = DecimalDtype(context)
  64 
  65     @property
  66     def dtype(self):
  67         return self._dtype

  69     @classmethod
  70     def _from_sequence(cls, scalars, dtype=None, copy=False):
  71         return cls(scalars)
  72 
  73     @classmethod
  74     def _from_sequence_of_strings(cls, strings, dtype=None, copy=False):
  75         return cls._from_sequence([decimal.Decimal(x) for x in strings], dtype, copy)
  76 
  77     @classmethod
  78     def _from_factorized(cls, values, original):
  79         return cls(values)

  81     _HANDLED_TYPES = (decimal.Decimal, numbers.Number, np.ndarray)
  82 
  83     def to_numpy(
  84         self, dtype=None, copy: bool = False, na_value=no_default, decimals=None
  85     ) -> np.ndarray:
  86         result = np.asarray(self, dtype=dtype)
  87         if decimals is not None:
  88             result = np.asarray([round(x, decimals) for x in result])
  89         return result

  91     def __array_ufunc__(self, ufunc, method, *inputs, **kwargs):
  92         #
  93         if not all(
  94             isinstance(t, self._HANDLED_TYPES + (DecimalArray,)) for t in inputs
  95         ):
  96             return NotImplemented
  97 
  98         inputs = tuple(x._data if isinstance(x, DecimalArray) else x for x in inputs)
  99         result = getattr(ufunc, method)(*inputs, **kwargs)
 100 
 101         def reconstruct(x):
 102             if isinstance(x, (decimal.Decimal, numbers.Number)):
 103                 return x
 104             else:
 105                 return DecimalArray._from_sequence(x)
 106 
 107         if isinstance(result, tuple):
 108             return tuple(reconstruct(x) for x in result)
 109         else:
 110             return reconstruct(result)

 112     def __getitem__(self, item):
 113         if isinstance(item, numbers.Integral):
 114             return self._data[item]
 115         else:
 116             # array, slice.
 117             item = pd.api.indexers.check_array_indexer(self, item)
 118             return type(self)(self._data[item])
 119 
 120     def take(self, indexer, allow_fill=False, fill_value=None):
 121         from pandas.api.extensions import take
 122 
 123         data = self._data
 124         if allow_fill and fill_value is None:
 125             fill_value = self.dtype.na_value
 126 
 127         result = take(data, indexer, fill_value=fill_value, allow_fill=allow_fill)
 128         return self._from_sequence(result)
 129 
 130     def copy(self):
 131         return type(self)(self._data.copy())
 132 
 133     def astype(self, dtype, copy=True):
 134         dtype = pandas_dtype(dtype)
 135         if isinstance(dtype, type(self.dtype)):
 136             return type(self)(self._data, context=dtype.context)
 137 
 138         return super().astype(dtype, copy=copy)
 139 
 140     def __setitem__(self, key, value):
 141         if pd.api.types.is_list_like(value):
 142             if pd.api.types.is_scalar(key):
 143                 raise ValueError("setting an array element with a sequence.")
 144             value = [decimal.Decimal(v) for v in value]
 145         else:
 146             value = decimal.Decimal(value)
 147 
 148         key = check_array_indexer(self, key)
 149         self._data[key] = value

 156         n = len(self)

 162         return np.array([x.is_nan() for x in self._data], dtype=bool)

 168     def _formatter(self, boxed=False):
 169         if boxed:

 173     @classmethod
 174     def _concat_same_type(cls, to_concat):

 177     def _reduce(self, name, skipna=True, **kwargs):
 178 
 179         if skipna:
 180             # If we don't have any NAs, we can ignore skipna
 181             if self.isna().any():
 182                 other = self[~self.isna()]
 183                 return other._reduce(name, **kwargs)
 184 
 185         if name == "sum" and len(self) == 0:
 186             # GH#29630 avoid returning int 0 or np.bool_(False) on old numpy
 187             return decimal.Decimal(0)
 188 
 189         try:
 190             op = getattr(self.data, name)
 191         except AttributeError as err:
 192             raise NotImplementedError(
 193                 f"decimal does not support the {name} operation"
 194             ) from err
 195         return op(axis=0)
