  68 class BaseDecimal:
  69     @classmethod
  70     def assert_series_equal(cls, left, right, *args, **kwargs):
  71         def convert(x):
  72             # need to convert array([Decimal(NaN)], dtype='object') to np.NaN
  73             # because Series[object].isnan doesn't recognize decimal(NaN) as
  74             # NA.
  75             try:
  76                 return math.isnan(x)
  77             except TypeError:
  78                 return False
  79 
  80         if left.dtype == "object":
  81             left_na = left.apply(convert)
  82         else:
  83             left_na = left.isna()
  84         if right.dtype == "object":
  85             right_na = right.apply(convert)
  86         else:
  87             right_na = right.isna()
  88 
  89         tm.assert_series_equal(left_na, right_na)
  90         return tm.assert_series_equal(left[~left_na], right[~right_na], *args, **kwargs)
  91 
  92     @classmethod
  93     def assert_frame_equal(cls, left, right, *args, **kwargs):
  94         # TODO(EA): select_dtypes
  95         tm.assert_index_equal(
  96             left.columns,
  97             right.columns,
  98             exact=kwargs.get("check_column_type", "equiv"),
  99             check_names=kwargs.get("check_names", True),
 100             check_exact=kwargs.get("check_exact", False),
 101             check_categorical=kwargs.get("check_categorical", True),
 102             obj=f"{kwargs.get('obj', 'DataFrame')}.columns",
 103         )
 104 
 105         decimals = (left.dtypes == "decimal").index
 106 
 107         for col in decimals:
 108             cls.assert_series_equal(left[col], right[col], *args, **kwargs)
 109 
 110         left = left.drop(columns=decimals)
 111         right = right.drop(columns=decimals)
 112         tm.assert_frame_equal(left, right, *args, **kwargs)

 279         op_name = all_arithmetic_operators

 284         invalidoptrap = context.traps[decimal.InvalidOperation]
 285         context.traps[decimal.DivisionByZero] = 0
 286         context.traps[decimal.InvalidOperation] = 0
 287 
 288         # Decimal supports ops with int, but not float
 289         other = pd.Series([int(d * 100) for d in data])

 293             self.check_opname(s, op_name, s * 2)

 295         self.check_opname(s, op_name, 0)
 296         self.check_opname(s, op_name, 5)
 297         context.traps[decimal.DivisionByZero] = divbyzerotrap

 438 def test_groupby_agg():
 439     # Ensure that the result of agg is inferred to be decimal dtype
 440     # https://github.com/pandas-dev/pandas/issues/29141
 441 
 442     data = make_data()[:5]
 443     df = pd.DataFrame(
 444         {"id1": [0, 0, 0, 1, 1], "id2": [0, 1, 0, 1, 1], "decimals": DecimalArray(data)}
 445     )
 446 
 447     # single key, selected column
 448     expected = pd.Series(to_decimal([data[0], data[3]]))
 449     result = df.groupby("id1")["decimals"].agg(lambda x: x.iloc[0])
 450     tm.assert_series_equal(result, expected, check_names=False)
 451     result = df["decimals"].groupby(df["id1"]).agg(lambda x: x.iloc[0])
 452     tm.assert_series_equal(result, expected, check_names=False)
 453 
 454     # multiple keys, selected column
 455     expected = pd.Series(
 456         to_decimal([data[0], data[1], data[3]]),
 457         index=pd.MultiIndex.from_tuples([(0, 0), (0, 1), (1, 1)]),
 458     )
 459     result = df.groupby(["id1", "id2"])["decimals"].agg(lambda x: x.iloc[0])
 460     tm.assert_series_equal(result, expected, check_names=False)
 461     result = df["decimals"].groupby([df["id1"], df["id2"]]).agg(lambda x: x.iloc[0])
 462     tm.assert_series_equal(result, expected, check_names=False)
 463 
 464     # multiple columns
 465     expected = pd.DataFrame({"id2": [0, 1], "decimals": to_decimal([data[0], data[3]])})
 466     result = df.groupby("id1").agg(lambda x: x.iloc[0])
 467     tm.assert_frame_equal(result, expected, check_names=False)

 472     # https://github.com/pandas-dev/pandas/issues/29141
 473
