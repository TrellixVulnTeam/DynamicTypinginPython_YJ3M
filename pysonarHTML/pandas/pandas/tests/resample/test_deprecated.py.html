 227 def test_resample_with_non_zero_base(start, end, start_freq, end_freq, base, offset):
 228     # GH 23882
 229     s = pd.Series(0, index=pd.period_range(start, end, freq=start_freq))
 230     s = s + np.arange(len(s))
 231     with tm.assert_produces_warning(FutureWarning):
 232         result = s.resample(end_freq, base=base).mean()
 233     result = result.to_timestamp(end_freq)
 234 
 235     # test that the replacement argument 'offset' works
 236     result_offset = s.resample(end_freq, offset=offset).mean()
 237     result_offset = result_offset.to_timestamp(end_freq)
 238     tm.assert_series_equal(result, result_offset)
 239 
 240     # to_timestamp casts 24H -> D
 241     result = result.asfreq(end_freq) if end_freq == "24H" else result
 242     with tm.assert_produces_warning(FutureWarning):
 243         expected = s.to_timestamp().resample(end_freq, base=base).mean()
 244     if end_freq == "M":
 245         # TODO: is non-tick the relevant characteristic? (GH 33815)
 246         expected.index = expected.index._with_freq(None)
 247     tm.assert_series_equal(result, expected)
