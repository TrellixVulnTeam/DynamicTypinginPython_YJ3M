 305 def test_agg():
 306     # test with all three Resampler apis and TimeGrouper
 307 
 308     np.random.seed(1234)
 309     index = date_range(datetime(2005, 1, 1), datetime(2005, 1, 10), freq="D")
 310     index.name = "date"
 311     df = DataFrame(np.random.rand(10, 2), columns=list("AB"), index=index)
 312     df_col = df.reset_index()
 313     df_mult = df_col.copy()
 314     df_mult.index = pd.MultiIndex.from_arrays(
 315         [range(10), df.index], names=["index", "date"]
 316     )
 317     r = df.resample("2D")
 318     cases = [
 319         r,
 320         df_col.resample("2D", on="date"),
 321         df_mult.resample("2D", level="date"),
 322         df.groupby(pd.Grouper(freq="2D")),
 323     ]
 324 
 325     a_mean = r["A"].mean()
 326     a_std = r["A"].std()
 327     a_sum = r["A"].sum()
 328     b_mean = r["B"].mean()
 329     b_std = r["B"].std()
 330     b_sum = r["B"].sum()
 331 
 332     expected = pd.concat([a_mean, a_std, b_mean, b_std], axis=1)
 333     expected.columns = pd.MultiIndex.from_product([["A", "B"], ["mean", "std"]])
 334     for t in cases:
 335         result = t.aggregate([np.mean, np.std])
 336         tm.assert_frame_equal(result, expected)
 337 
 338     expected = pd.concat([a_mean, b_std], axis=1)
 339     for t in cases:
 340         result = t.aggregate({"A": np.mean, "B": np.std})
 341         tm.assert_frame_equal(result, expected, check_like=True)
 342 
 343     expected = pd.concat([a_mean, a_std], axis=1)
 344     expected.columns = pd.MultiIndex.from_tuples([("A", "mean"), ("A", "std")])
 345     for t in cases:
 346         result = t.aggregate({"A": ["mean", "std"]})
 347         tm.assert_frame_equal(result, expected)
 348 
 349     expected = pd.concat([a_mean, a_sum], axis=1)
 350     expected.columns = ["mean", "sum"]
 351     for t in cases:
 352         result = t["A"].aggregate(["mean", "sum"])
 353     tm.assert_frame_equal(result, expected)
 354 
 355     msg = "nested renamer is not supported"
 356     for t in cases:
 357         with pytest.raises(pd.core.base.SpecificationError, match=msg):
 358             t.aggregate({"A": {"mean": "mean", "sum": "sum"}})
 359 
 360     expected = pd.concat([a_mean, a_sum, b_mean, b_sum], axis=1)
 361     expected.columns = pd.MultiIndex.from_tuples(
 362         [("A", "mean"), ("A", "sum"), ("B", "mean2"), ("B", "sum2")]
 363     )
 364     for t in cases:
 365         with pytest.raises(pd.core.base.SpecificationError, match=msg):
 366             t.aggregate(
 367                 {
 368                     "A": {"mean": "mean", "sum": "sum"},
 369                     "B": {"mean2": "mean", "sum2": "sum"},
 370                 }
 371             )
 372 
 373     expected = pd.concat([a_mean, a_std, b_mean, b_std], axis=1)
 374     expected.columns = pd.MultiIndex.from_tuples(
 375         [("A", "mean"), ("A", "std"), ("B", "mean"), ("B", "std")]
 376     )
 377     for t in cases:
 378         result = t.aggregate({"A": ["mean", "std"], "B": ["mean", "std"]})
 379         tm.assert_frame_equal(result, expected, check_like=True)
 380 
 381     expected = pd.concat([a_mean, a_sum, b_mean, b_sum], axis=1)
 382     expected.columns = pd.MultiIndex.from_tuples(
 383         [
 384             ("r1", "A", "mean"),
 385             ("r1", "A", "sum"),
 386             ("r2", "B", "mean"),
 387             ("r2", "B", "sum"),
 388         ]
 389     )

 392 def test_agg_misc():
 393     # test with all three Resampler apis and TimeGrouper
 394 
 395     np.random.seed(1234)
 396     index = date_range(datetime(2005, 1, 1), datetime(2005, 1, 10), freq="D")
 397     index.name = "date"
 398     df = DataFrame(np.random.rand(10, 2), columns=list("AB"), index=index)
 399     df_col = df.reset_index()
 400     df_mult = df_col.copy()
 401     df_mult.index = pd.MultiIndex.from_arrays(
 402         [range(10), df.index], names=["index", "date"]
 403     )
 404 
 405     r = df.resample("2D")
 406     cases = [
 407         r,
 408         df_col.resample("2D", on="date"),
 409         df_mult.resample("2D", level="date"),
 410         df.groupby(pd.Grouper(freq="2D")),
 411     ]
 412 
 413     # passed lambda
 414     for t in cases:
 415         result = t.agg({"A": np.sum, "B": lambda x: np.std(x, ddof=1)})
 416         rcustom = t["B"].apply(lambda x: np.std(x, ddof=1))
 417         expected = pd.concat([r["A"].sum(), rcustom], axis=1)
 418         tm.assert_frame_equal(result, expected, check_like=True)
 419 
 420     # agg with renamers
 421     expected = pd.concat(
 422         [t["A"].sum(), t["B"].sum(), t["A"].mean(), t["B"].mean()], axis=1
 423     )
 424     expected.columns = pd.MultiIndex.from_tuples(
 425         [("result1", "A"), ("result1", "B"), ("result2", "A"), ("result2", "B")]
 426     )
 427 
 428     msg = r"Column\(s\) \['result1', 'result2'\] do not exist"
 429     for t in cases:
 430         with pytest.raises(pd.core.base.SpecificationError, match=msg):
 431             t[["A", "B"]].agg(OrderedDict([("result1", np.sum), ("result2", np.mean)]))
 432 
 433     # agg with different hows
 434     expected = pd.concat(
 435         [t["A"].sum(), t["A"].std(), t["B"].mean(), t["B"].std()], axis=1
 436     )
 437     expected.columns = pd.MultiIndex.from_tuples(
 438         [("A", "sum"), ("A", "std"), ("B", "mean"), ("B", "std")]
 439     )
 440     for t in cases:
 441         result = t.agg(OrderedDict([("A", ["sum", "std"]), ("B", ["mean", "std"])]))
 442         tm.assert_frame_equal(result, expected, check_like=True)
 443 
 444     # equivalent of using a selection list / or not
 445     for t in cases:
 446         result = t[["A", "B"]].agg({"A": ["sum", "std"], "B": ["mean", "std"]})
 447         tm.assert_frame_equal(result, expected, check_like=True)
 448 
 449     msg = "nested renamer is not supported"
 450 
 451     # series like aggs
 452     for t in cases:
 453         with pytest.raises(pd.core.base.SpecificationError, match=msg):
 454             t["A"].agg({"A": ["sum", "std"]})
 455 
 456         with pytest.raises(pd.core.base.SpecificationError, match=msg):
 457             t["A"].agg({"A": ["sum", "std"], "B": ["mean", "std"]})
 458 
 459     # errors
 460     # invalid names in the agg specification
 461     msg = "\"Column 'B' does not exist!\""
 462     for t in cases:
 463         with pytest.raises(KeyError, match=msg):
 464             t[["A"]].agg({"A": ["sum", "std"], "B": ["mean", "std"]})
