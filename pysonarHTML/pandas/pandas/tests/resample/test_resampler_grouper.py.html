 135 def test_groupby_with_origin():
 136     # GH 31809
 137 
 138     freq = "1399min"  # prime number that is smaller than 24h
 139     start, end = "1/1/2000 00:00:00", "1/31/2000 00:00"
 140     middle = "1/15/2000 00:00:00"
 141 
 142     rng = pd.date_range(start, end, freq="1231min")  # prime number
 143     ts = pd.Series(np.random.randn(len(rng)), index=rng)
 144     ts2 = ts[middle:end]
 145 
 146     # proves that grouper without a fixed origin does not work
 147     # when dealing with unusual frequencies
 148     simple_grouper = pd.Grouper(freq=freq)
 149     count_ts = ts.groupby(simple_grouper).agg("count")
 150     count_ts = count_ts[middle:end]
 151     count_ts2 = ts2.groupby(simple_grouper).agg("count")
 152     with pytest.raises(AssertionError):
 153         tm.assert_index_equal(count_ts.index, count_ts2.index)
 154 
 155     # test origin on 1970-01-01 00:00:00
 156     origin = pd.Timestamp(0)
 157     adjusted_grouper = pd.Grouper(freq=freq, origin=origin)
 158     adjusted_count_ts = ts.groupby(adjusted_grouper).agg("count")
 159     adjusted_count_ts = adjusted_count_ts[middle:end]
 160     adjusted_count_ts2 = ts2.groupby(adjusted_grouper).agg("count")
 161     tm.assert_series_equal(adjusted_count_ts, adjusted_count_ts2)
 162 
 163     # test origin on 2049-10-18 20:00:00
 164     origin_future = pd.Timestamp(0) + pd.Timedelta("1399min") * 30_000
 165     adjusted_grouper2 = pd.Grouper(freq=freq, origin=origin_future)
 166     adjusted2_count_ts = ts.groupby(adjusted_grouper2).agg("count")
 167     adjusted2_count_ts = adjusted2_count_ts[middle:end]
 168     adjusted2_count_ts2 = ts2.groupby(adjusted_grouper2).agg("count")
 169     tm.assert_series_equal(adjusted2_count_ts, adjusted2_count_ts2)
 170 
 171     # both grouper use an adjusted timestamp that is a multiple of 1399 min
 172     # they should be equals even if the adjusted_timestamp is in the future
 173     tm.assert_series_equal(adjusted_count_ts, adjusted2_count_ts2)

 205

 212         result = getattr(r, f)()

 283 def test_apply_columns_multilevel():
 284     # GH 16231
 285     cols = pd.MultiIndex.from_tuples([("A", "a", "", "one"), ("B", "b", "i", "two")])
 286     ind = date_range(start="2017-01-01", freq="15Min", periods=8)
 287     df = DataFrame(np.array([0] * 16).reshape(8, 2), index=ind, columns=cols)
