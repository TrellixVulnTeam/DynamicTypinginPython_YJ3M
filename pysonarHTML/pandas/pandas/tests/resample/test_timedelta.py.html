  98 def test_resample_categorical_data_with_timedeltaindex():
  99     # GH #12169
 100     df = DataFrame({"Group_obj": "A"}, index=pd.to_timedelta(list(range(20)), unit="s"))
 101     df["Group"] = df["Group_obj"].astype("category")
 102     result = df.resample("10s").agg(lambda x: (x.value_counts().index[0]))
 103     expected = DataFrame(

 106     )
 107     expected = expected.reindex(["Group_obj", "Group"], axis=1)
 108     expected["Group"] = expected["Group_obj"]
 109     tm.assert_frame_equal(result, expected)

 112 def test_resample_timedelta_values():
 113     # GH 13119
 114     # check that timedelta dtype is preserved when NaT values are
 115     # introduced by the resampling
 116 
 117     times = timedelta_range("1 day", "6 day", freq="4D")
 118     df = DataFrame({"time": times}, index=times)
 119 
 120     times2 = timedelta_range("1 day", "6 day", freq="2D")
 121     exp = Series(times2, index=times2, name="time")
 122     exp.iloc[1] = pd.NaT
 123 
 124     res = df.resample("2D").first()["time"]
 125     tm.assert_series_equal(res, exp)
 126     res = df["time"].resample("2D").first()
 127     tm.assert_series_equal(res, exp)

 130 @pytest.mark.parametrize(
 131     "start, end, freq, resample_freq",
 132     [
 133         ("8H", "21h59min50s", "10S", "3H"),  # GH 30353 example
 134         ("3H", "22H", "1H", "5H"),
 135         ("527D", "5006D", "3D", "10D"),

 138         ("8H", "21h59min50s", "10S", "2H"),
 139         ("0H", "21h59min50s", "10S", "3H"),
 140         ("10D", "85D", "D", "2D"),
 141     ],
 142 )
 143 def test_resample_timedelta_edge_case(start, end, freq, resample_freq):
 144     # GH 33498
 145     # check that the timedelta bins does not contains an extra bin
 146     idx = pd.timedelta_range(start=start, end=end, freq=freq)
 147     s = pd.Series(np.arange(len(idx)), index=idx)
 148     result = s.resample(resample_freq).min()
 149     expected_index = pd.timedelta_range(freq=resample_freq, start=start, end=end)
 150     tm.assert_index_equal(result.index, expected_index)
 151     assert result.index.freq == expected_index.freq
 152     assert not np.isnan(result[-1])
