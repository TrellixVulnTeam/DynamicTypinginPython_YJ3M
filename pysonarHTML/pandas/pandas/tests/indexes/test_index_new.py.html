  27 class TestIndexConstructorInference:
  28     @pytest.mark.parametrize("na_value", [None, np.nan])
  29     @pytest.mark.parametrize("vtype", [list, tuple, iter])
  30     def test_construction_list_tuples_nan(self, na_value, vtype):
  31         # GH#18505 : valid tuples containing NaN
  32         values = [(1, "two"), (3.0, na_value)]
  33         result = Index(vtype(values))
  34         expected = MultiIndex.from_tuples(values)
  35         tm.assert_index_equal(result, expected)
  36 
  37     @pytest.mark.parametrize(
  38         "dtype",
  39         [int, "int64", "int32", "int16", "int8", "uint64", "uint32", "uint16", "uint8"],
  40     )
  41     def test_constructor_int_dtype_float(self, dtype):
  42         # GH#18400
  43         if is_unsigned_integer_dtype(dtype):
  44             index_type = UInt64Index
  45         else:
  46             index_type = Int64Index
  47 
  48         expected = index_type([0, 1, 2, 3])
  49         result = Index([0.0, 1.0, 2.0, 3.0], dtype=dtype)
  50         tm.assert_index_equal(result, expected)
  51 
  52     @pytest.mark.parametrize("cast_index", [True, False])
  53     @pytest.mark.parametrize(
  54         "vals", [[True, False, True], np.array([True, False, True], dtype=bool)]
  55     )
  56     def test_constructor_dtypes_to_object(self, cast_index, vals):
  57         if cast_index:
  58             index = Index(vals, dtype=bool)
  59         else:
  60             index = Index(vals)
  61 
  62         assert type(index) is Index
  63         assert index.dtype == object

  65     def test_constructor_categorical_to_object(self):
  66         # GH#32167 Categorical data and dtype=object should return object-dtype
  67         ci = CategoricalIndex(range(5))
  68         result = Index(ci, dtype=object)
  69         assert not isinstance(result, CategoricalIndex)

  71     def test_constructor_infer_periodindex(self):
  72         xp = period_range("2012-1-1", freq="M", periods=3)
  73         rs = Index(xp)
  74         tm.assert_index_equal(rs, xp)
  75         assert isinstance(rs, PeriodIndex)

  77     @pytest.mark.parametrize("pos", [0, 1])
  78     @pytest.mark.parametrize(
  79         "klass,dtype,ctor",
  80         [
  81             (DatetimeIndex, "datetime64[ns]", np.datetime64("nat")),
  82             (TimedeltaIndex, "timedelta64[ns]", np.timedelta64("nat")),
  83         ],
  84     )
  85     def test_constructor_infer_nat_dt_like(
  86         self, pos, klass, dtype, ctor, nulls_fixture, request
  87     ):
  88         expected = klass([NaT, NaT])
  89         assert expected.dtype == dtype
  90         data = [ctor]
  91         data.insert(pos, nulls_fixture)
  92 
  93         if nulls_fixture is NA:
  94             expected = Index([NA, NaT])
  95             mark = pytest.mark.xfail(reason="Broken with np.NaT ctor; see GH 31884")
  96             request.node.add_marker(mark)
  97 
  98         result = Index(data)
  99         tm.assert_index_equal(result, expected)
 100 
 101         result = Index(np.array(data, dtype=object))
 102         tm.assert_index_equal(result, expected)
 103 
 104     @pytest.mark.parametrize("swap_objs", [True, False])
 105     def test_constructor_mixed_nat_objs_infers_object(self, swap_objs):
 106         # mixed np.datetime64/timedelta64 nat results in object
 107         data = [np.datetime64("nat"), np.timedelta64("nat")]
 108         if swap_objs:
 109             data = data[::-1]
 110 
 111         expected = Index(data, dtype=object)
 112         tm.assert_index_equal(Index(data), expected)
 113         tm.assert_index_equal(Index(np.array(data, dtype=object)), expected)
