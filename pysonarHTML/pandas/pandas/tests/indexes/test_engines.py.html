  31 class TestDatetimeEngine:
  32     @pytest.mark.parametrize(
  33         "scalar",
  34         [
  35             pd.Timedelta(pd.Timestamp("2016-01-01").asm8.view("m8[ns]")),
  36             pd.Timestamp("2016-01-01").value,
  37             pd.Timestamp("2016-01-01").to_pydatetime(),
  38             pd.Timestamp("2016-01-01").to_datetime64(),
  39         ],
  40     )
  41     def test_not_contains_requires_timestamp(self, scalar):
  42         dti1 = pd.date_range("2016-01-01", periods=3)
  43         dti2 = dti1.insert(1, pd.NaT)  # non-monotonic
  44         dti3 = dti1.insert(3, dti1[0])  # non-unique
  45         dti4 = pd.date_range("2016-01-01", freq="ns", periods=2_000_000)
  46         dti5 = dti4.insert(0, dti4[0])  # over size threshold, not unique
  47 
  48         msg = "|".join([re.escape(str(scalar)), re.escape(repr(scalar))])
  49         for dti in [dti1, dti2, dti3, dti4, dti5]:
  50             with pytest.raises(TypeError, match=msg):
  51                 scalar in dti._engine
  52 
  53             with pytest.raises(KeyError, match=msg):
  54                 dti._engine.get_loc(scalar)
