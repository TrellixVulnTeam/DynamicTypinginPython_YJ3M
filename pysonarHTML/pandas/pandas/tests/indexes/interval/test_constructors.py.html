 152     def test_generic_errors(self, constructor):

 154         filler = self.get_kwargs_from_breaks(range(10))
 155

 160 
 161         # unsupported dtype
 162         msg = "dtype must be an IntervalDtype, got int64"

 199     def test_constructor_errors(self):

 220         """mixed int/float left/right results in float for both sides"""
 221         left = np.arange(9, dtype=left_subtype)

 266 class TestFromTuples(Base):
 267     """Tests specific to IntervalIndex.from_tuples"""
 268 
 269     @pytest.fixture
 270     def constructor(self):
 271         return IntervalIndex.from_tuples
 272 
 273     def get_kwargs_from_breaks(self, breaks, closed="right"):
 274         """
 275         converts intervals in breaks format to a dictionary of kwargs to
 276         specific to the format expected by IntervalIndex.from_tuples
 277         """
 278         if len(breaks) == 0:
 279             return {"data": breaks}
 280 
 281         tuples = list(zip(breaks[:-1], breaks[1:]))
 282         if isinstance(breaks, (list, tuple)):
 283             return {"data": tuples}
 284         elif is_categorical_dtype(breaks):
 285             return {"data": breaks._constructor(tuples)}
 286         return {"data": com.asarray_tuplesafe(tuples)}
 287 
 288     def test_constructor_errors(self):
 289         # non-tuple
 290         tuples = [(0, 1), 2, (3, 4)]
 291         msg = "IntervalIndex.from_tuples received an invalid item, 2"
 292         with pytest.raises(TypeError, match=msg.format(t=tuples)):
 293             IntervalIndex.from_tuples(tuples)
 294 
 295         # too few/many items
 296         tuples = [(0, 1), (2,), (3, 4)]
 297         msg = "IntervalIndex.from_tuples requires tuples of length 2, got {t}"
 298         with pytest.raises(ValueError, match=msg.format(t=tuples)):
 299             IntervalIndex.from_tuples(tuples)
 300 
 301         tuples = [(0, 1), (2, 3, 4), (5, 6)]
 302         with pytest.raises(ValueError, match=msg.format(t=tuples)):
 303             IntervalIndex.from_tuples(tuples)
 304 
 305     def test_na_tuples(self):
 306         # tuple (NA, NA) evaluates the same as NA as an element
 307         na_tuple = [(0, 1), (np.nan, np.nan), (2, 3)]
 308         idx_na_tuple = IntervalIndex.from_tuples(na_tuple)
 309         idx_na_element = IntervalIndex.from_tuples([(0, 1), np.nan, (2, 3)])
 310         tm.assert_index_equal(idx_na_tuple, idx_na_element)
