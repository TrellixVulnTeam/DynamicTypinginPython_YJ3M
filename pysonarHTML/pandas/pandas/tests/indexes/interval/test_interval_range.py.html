 148     def test_early_truncation(self, start, end, freq, expected_endpoint):
 149         # index truncates early if freq causes end to be skipped

 151         result_endpoint = result.right[-1]

 191     @pytest.mark.parametrize("freq", [2, 2.0])
 192     @pytest.mark.parametrize("end", [10, 10.0])
 193     @pytest.mark.parametrize("start", [0, 0.0])
 194     def test_float_subtype(self, start, end, freq):
 195         # Has float subtype if any of start/end/freq are float, even if all
 196         # resulting endpoints can safely be upcast to integers
 197 
 198         # defined from start/end/freq
 199         index = interval_range(start=start, end=end, freq=freq)
 200         result = index.dtype.subtype
 201         expected = "int64" if is_integer(start + end + freq) else "float64"
 202         assert result == expected
 203 
 204         # defined from start/periods/freq
 205         index = interval_range(start=start, periods=5, freq=freq)
 206         result = index.dtype.subtype
 207         expected = "int64" if is_integer(start + freq) else "float64"
 208         assert result == expected
 209 
 210         # defined from end/periods/freq
 211         index = interval_range(end=end, periods=5, freq=freq)
 212         result = index.dtype.subtype
 213         expected = "int64" if is_integer(end + freq) else "float64"
 214         assert result == expected
 215 
 216         # GH 20976: linspace behavior defined from start/end/periods
 217         index = interval_range(start=start, end=end, periods=5)
 218         result = index.dtype.subtype
 219         expected = "int64" if is_integer(start + end) else "float64"
 220         assert result == expected

 222     def test_constructor_coverage(self):
 223         # float value for periods
 224         expected = interval_range(start=0, periods=10)
 225         result = interval_range(start=0, periods=10.5)
 226         tm.assert_index_equal(result, expected)
 227 
 228         # equivalent timestamp-like start/end
 229         start, end = Timestamp("2017-01-01"), Timestamp("2017-01-15")
 230         expected = interval_range(start=start, end=end)
 231 
 232         result = interval_range(start=start.to_pydatetime(), end=end.to_pydatetime())
 233         tm.assert_index_equal(result, expected)
 234 
 235         result = interval_range(start=start.asm8, end=end.asm8)
 236         tm.assert_index_equal(result, expected)
 237 
 238         # equivalent freq with timestamp
 239         equiv_freq = [
 240             "D",
 241             Day(),
 242             Timedelta(days=1),
 243             timedelta(days=1),
 244             DateOffset(days=1),
 245         ]
 246         for freq in equiv_freq:
 247             result = interval_range(start=start, end=end, freq=freq)
 248             tm.assert_index_equal(result, expected)
 249 
 250         # equivalent timedelta-like start/end
 251         start, end = Timedelta(days=1), Timedelta(days=10)
 252         expected = interval_range(start=start, end=end)
 253 
 254         result = interval_range(start=start.to_pytimedelta(), end=end.to_pytimedelta())
 255         tm.assert_index_equal(result, expected)
 256 
 257         result = interval_range(start=start.asm8, end=end.asm8)
 258         tm.assert_index_equal(result, expected)
 259 
 260         # equivalent freq with timedelta
 261         equiv_freq = ["D", Day(), Timedelta(days=1), timedelta(days=1)]
 262         for freq in equiv_freq:
 263             result = interval_range(start=start, end=end, freq=freq)
 264             tm.assert_index_equal(result, expected)

 266     def test_errors(self):
 267         # not enough params
 268         msg = (
 269             "Of the four parameters: start, end, periods, and freq, "
 270             "exactly three must be specified"
 271         )
 272 
 273         with pytest.raises(ValueError, match=msg):
 274             interval_range(start=0)
 275 
 276         with pytest.raises(ValueError, match=msg):
 277             interval_range(end=5)
 278 
 279         with pytest.raises(ValueError, match=msg):
 280             interval_range(periods=2)
 281 
 282         with pytest.raises(ValueError, match=msg):
 283             interval_range()
 284 
 285         # too many params
 286         with pytest.raises(ValueError, match=msg):
 287             interval_range(start=0, end=5, periods=6, freq=1.5)
 288 
 289         # mixed units
 290         msg = "start, end, freq need to be type compatible"
 291         with pytest.raises(TypeError, match=msg):
 292             interval_range(start=0, end=Timestamp("20130101"), freq=2)
 293 
 294         with pytest.raises(TypeError, match=msg):
 295             interval_range(start=0, end=Timedelta("1 day"), freq=2)
 296 
 297         with pytest.raises(TypeError, match=msg):
 298             interval_range(start=0, end=10, freq="D")
 299 
 300         with pytest.raises(TypeError, match=msg):
 301             interval_range(start=Timestamp("20130101"), end=10, freq="D")
 302 
 303         with pytest.raises(TypeError, match=msg):
 304             interval_range(
 305                 start=Timestamp("20130101"), end=Timedelta("1 day"), freq="D"
 306             )
 307 
 308         with pytest.raises(TypeError, match=msg):
 309             interval_range(
 310                 start=Timestamp("20130101"), end=Timestamp("20130110"), freq=2
 311             )
 312 
 313         with pytest.raises(TypeError, match=msg):
 314             interval_range(start=Timedelta("1 day"), end=10, freq="D")
 315 
 316         with pytest.raises(TypeError, match=msg):
 317             interval_range(
 318                 start=Timedelta("1 day"), end=Timestamp("20130110"), freq="D"
 319             )
 320 
 321         with pytest.raises(TypeError, match=msg):
 322             interval_range(start=Timedelta("1 day"), end=Timedelta("10 days"), freq=2)
 323 
 324         # invalid periods
 325         msg = "periods must be a number, got foo"
 326         with pytest.raises(TypeError, match=msg):
 327             interval_range(start=0, periods="foo")
 328 
 329         # invalid start
 330         msg = "start must be numeric or datetime-like, got foo"
 331         with pytest.raises(ValueError, match=msg):
 332             interval_range(start="foo", periods=10)
 333 
 334         # invalid end
 335         msg = r"end must be numeric or datetime-like, got \(0, 1\]"
 336         with pytest.raises(ValueError, match=msg):
 337             interval_range(end=Interval(0, 1), periods=10)
 338 
 339         # invalid freq for datetime-like
 340         msg = "freq must be numeric or convertible to DateOffset, got foo"
 341         with pytest.raises(ValueError, match=msg):
 342             interval_range(start=0, end=10, freq="foo")
 343 
 344         with pytest.raises(ValueError, match=msg):
 345             interval_range(start=Timestamp("20130101"), periods=10, freq="foo")
 346 
 347         with pytest.raises(ValueError, match=msg):
 348             interval_range(end=Timedelta("1 day"), periods=10, freq="foo")
 349 
 350         # mixed tz
 351         start = Timestamp("2017-01-01", tz="US/Eastern")
 352         end = Timestamp("2017-01-07", tz="US/Pacific")
 353         msg = "Start and end cannot both be tz-aware with different timezones"
 354         with pytest.raises(TypeError, match=msg):
 355             interval_range(start=start, end=end)
