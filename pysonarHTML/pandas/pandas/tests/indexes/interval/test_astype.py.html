  34     def test_astype_category(self, index):
  35         result = index.astype("category")
  36         expected = CategoricalIndex(index.values)
  37         tm.assert_index_equal(result, expected)
  38 
  39         result = index.astype(CategoricalDtype())
  40         tm.assert_index_equal(result, expected)
  41 
  42         # non-default params
  43         categories = index.dropna().unique().values[:-1]
  44         dtype = CategoricalDtype(categories=categories, ordered=True)
  45         result = index.astype(dtype)
  46         expected = CategoricalIndex(index.values, categories=categories, ordered=True)
  47         tm.assert_index_equal(result, expected)

 206     def test_subtype_datetimelike(self):
 207         # datetime -> timedelta raises
 208         dtype = IntervalDtype("timedelta64[ns]")
 209         msg = "Cannot convert .* to .*; subtypes are incompatible"
 210 
 211         index = interval_range(Timestamp("2018-01-01"), periods=10)
 212         with pytest.raises(TypeError, match=msg):
 213             index.astype(dtype)
 214 
 215         index = interval_range(Timestamp("2018-01-01", tz="CET"), periods=10)
 216         with pytest.raises(TypeError, match=msg):
 217             index.astype(dtype)
 218 
 219         # timedelta -> datetime raises
 220         dtype = IntervalDtype("datetime64[ns]")
 221         index = interval_range(Timedelta("0 days"), periods=10)
 222         with pytest.raises(TypeError, match=msg):
 223             index.astype(dtype)
