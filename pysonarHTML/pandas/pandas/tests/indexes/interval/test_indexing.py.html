  21     def test_get_loc_interval(self, closed, side):
  22 
  23         idx = IntervalIndex.from_tuples([(0, 1), (2, 3)], closed=closed)
  24 
  25         for bound in [[0, 1], [1, 2], [2, 3], [3, 4], [0, 2], [2.5, 3], [-1, 4]]:
  26             # if get_loc is supplied an interval, it should only search
  27             # for exact matches, not overlaps or covers, else KeyError.
  28             msg = re.escape(f"Interval({bound[0]}, {bound[1]}, closed='{side}')")
  29             if closed == side:
  30                 if bound == [0, 1]:
  31                     assert idx.get_loc(Interval(0, 1, closed=side)) == 0
  32                 elif bound == [2, 3]:
  33                     assert idx.get_loc(Interval(2, 3, closed=side)) == 1
  34                 else:
  35                     with pytest.raises(KeyError, match=msg):
  36                         idx.get_loc(Interval(*bound, closed=side))
  37             else:
  38                 with pytest.raises(KeyError, match=msg):
  39                     idx.get_loc(Interval(*bound, closed=side))

  41     @pytest.mark.parametrize("scalar", [-0.5, 0, 0.5, 1, 1.5, 2, 2.5, 3, 3.5])
  42     def test_get_loc_scalar(self, closed, scalar):
  43 
  44         # correct = {side: {query: answer}}.
  45         # If query is not in the dict, that query should raise a KeyError
  46         correct = {
  47             "right": {0.5: 0, 1: 0, 2.5: 1, 3: 1},
  48             "left": {0: 0, 0.5: 0, 2: 1, 2.5: 1},
  49             "both": {0: 0, 0.5: 0, 1: 0, 2: 1, 2.5: 1, 3: 1},
  50             "neither": {0.5: 0, 2.5: 1},
  51         }
  52 
  53         idx = IntervalIndex.from_tuples([(0, 1), (2, 3)], closed=closed)
  54 
  55         # if get_loc is supplied a scalar, it should return the index of
  56         # the interval which contains the scalar, or KeyError.
  57         if scalar in correct[closed].keys():
  58             assert idx.get_loc(scalar) == correct[closed][scalar]
  59         else:
  60             with pytest.raises(KeyError, match=str(scalar)):
  61                 idx.get_loc(scalar)

 330 class TestSliceLocs:
 331     def test_slice_locs_with_interval(self):
 332 
 333         # increasing monotonically
 334         index = IntervalIndex.from_tuples([(0, 2), (1, 3), (2, 4)])
 335 
 336         assert index.slice_locs(start=Interval(0, 2), end=Interval(2, 4)) == (0, 3)
 337         assert index.slice_locs(start=Interval(0, 2)) == (0, 3)
 338         assert index.slice_locs(end=Interval(2, 4)) == (0, 3)
 339         assert index.slice_locs(end=Interval(0, 2)) == (0, 1)
 340         assert index.slice_locs(start=Interval(2, 4), end=Interval(0, 2)) == (2, 1)
 341 
 342         # decreasing monotonically
 343         index = IntervalIndex.from_tuples([(2, 4), (1, 3), (0, 2)])
 344 
 345         assert index.slice_locs(start=Interval(0, 2), end=Interval(2, 4)) == (2, 1)
 346         assert index.slice_locs(start=Interval(0, 2)) == (2, 3)
 347         assert index.slice_locs(end=Interval(2, 4)) == (0, 1)
 348         assert index.slice_locs(end=Interval(0, 2)) == (0, 3)
 349         assert index.slice_locs(start=Interval(2, 4), end=Interval(0, 2)) == (0, 3)
 350 
 351         # sorted duplicates
 352         index = IntervalIndex.from_tuples([(0, 2), (0, 2), (2, 4)])
 353 
 354         assert index.slice_locs(start=Interval(0, 2), end=Interval(2, 4)) == (0, 3)
 355         assert index.slice_locs(start=Interval(0, 2)) == (0, 3)
 356         assert index.slice_locs(end=Interval(2, 4)) == (0, 3)
 357         assert index.slice_locs(end=Interval(0, 2)) == (0, 2)
 358         assert index.slice_locs(start=Interval(2, 4), end=Interval(0, 2)) == (2, 2)
 359 
 360         # unsorted duplicates
 361         index = IntervalIndex.from_tuples([(0, 2), (2, 4), (0, 2)])
 362 
 363         with pytest.raises(
 364             KeyError,
 365             match=re.escape(
 366                 '"Cannot get left slice bound for non-unique label: '
 367                 "Interval(0, 2, closed='right')\""
 368             ),
 369         ):
 370             index.slice_locs(start=Interval(0, 2), end=Interval(2, 4))
 371 
 372         with pytest.raises(
 373             KeyError,
 374             match=re.escape(
 375                 '"Cannot get left slice bound for non-unique label: '
 376                 "Interval(0, 2, closed='right')\""
 377             ),
 378         ):
 379             index.slice_locs(start=Interval(0, 2))
 380 
 381         assert index.slice_locs(end=Interval(2, 4)) == (0, 2)
 382 
 383         with pytest.raises(
 384             KeyError,
 385             match=re.escape(
 386                 '"Cannot get right slice bound for non-unique label: '
 387                 "Interval(0, 2, closed='right')\""
 388             ),
 389         ):
 390             index.slice_locs(end=Interval(0, 2))
 391 
 392         with pytest.raises(
 393             KeyError,
 394             match=re.escape(
 395                 '"Cannot get right slice bound for non-unique label: '
 396                 "Interval(0, 2, closed='right')\""
 397             ),
 398         ):
 399             index.slice_locs(start=Interval(2, 4), end=Interval(0, 2))
 400 
 401         # another unsorted duplicates
 402         index = IntervalIndex.from_tuples([(0, 2), (0, 2), (2, 4), (1, 3)])
 403 
 404         assert index.slice_locs(start=Interval(0, 2), end=Interval(2, 4)) == (0, 3)
 405         assert index.slice_locs(start=Interval(0, 2)) == (0, 4)
 406         assert index.slice_locs(end=Interval(2, 4)) == (0, 3)
 407         assert index.slice_locs(end=Interval(0, 2)) == (0, 2)
 408         assert index.slice_locs(start=Interval(2, 4), end=Interval(0, 2)) == (2, 2)

 410     def test_slice_locs_with_ints_and_floats_succeeds(self):
 411 
 412         # increasing non-overlapping
 413         index = IntervalIndex.from_tuples([(0, 1), (1, 2), (3, 4)])
 414 
 415         assert index.slice_locs(0, 1) == (0, 1)
 416         assert index.slice_locs(0, 2) == (0, 2)
 417         assert index.slice_locs(0, 3) == (0, 2)
 418         assert index.slice_locs(3, 1) == (2, 1)
 419         assert index.slice_locs(3, 4) == (2, 3)
 420         assert index.slice_locs(0, 4) == (0, 3)
 421 
 422         # decreasing non-overlapping
 423         index = IntervalIndex.from_tuples([(3, 4), (1, 2), (0, 1)])
 424         assert index.slice_locs(0, 1) == (3, 3)
 425         assert index.slice_locs(0, 2) == (3, 2)
 426         assert index.slice_locs(0, 3) == (3, 1)
 427         assert index.slice_locs(3, 1) == (1, 3)
 428         assert index.slice_locs(3, 4) == (1, 1)
 429         assert index.slice_locs(0, 4) == (3, 1)
