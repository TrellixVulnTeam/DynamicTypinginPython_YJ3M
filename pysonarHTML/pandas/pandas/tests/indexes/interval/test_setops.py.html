 148     @pytest.mark.parametrize(
 149         "op_name", ["union", "intersection", "difference", "symmetric_difference"]
 150     )
 151     def test_set_incompatible_types(self, closed, op_name, sort):
 152         index = monotonic_index(0, 11, closed=closed)
 153         set_op = getattr(index, op_name)
 154 
 155         # TODO: standardize return type of non-union setops type(self vs other)
 156         # non-IntervalIndex
 157         if op_name == "difference":
 158             expected = index
 159         else:
 160             expected = getattr(index.astype("O"), op_name)(Index([1, 2, 3]))
 161         result = set_op(Index([1, 2, 3]), sort=sort)
 162         tm.assert_index_equal(result, expected)
 163 
 164         # mixed closed
 165         msg = (
 166             "can only do set operations between two IntervalIndex objects "
 167             "that are closed on the same side"
 168         )
 169         for other_closed in {"right", "left", "both", "neither"} - {closed}:
 170             other = monotonic_index(0, 11, closed=other_closed)
 171             with pytest.raises(ValueError, match=msg):
 172                 set_op(other, sort=sort)
 173 
 174         # GH 19016: incompatible dtypes
 175         other = interval_range(Timestamp("20180101"), periods=9, closed=closed)
 176         msg = (
 177             f"can only do {op_name} between two IntervalIndex objects that have "
 178             "compatible dtypes"
 179         )
 180         with pytest.raises(TypeError, match=msg):
 181             set_op(other, sort=sort)
