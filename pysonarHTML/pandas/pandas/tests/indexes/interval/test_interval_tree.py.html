  12 def skipif_32bit(param):
  13     """
  14     Skip parameters in a parametrize on 32bit systems. Specifically used
  15     here to skip leaf_size parameters related to GH 23440.
  16     """
  17     marks = pytest.mark.skipif(
  18         compat.is_platform_32bit(), reason="GH 23440: int type mismatch on 32bit"
  19     )
  20     return pytest.param(param, marks=marks)

 126     @pytest.mark.parametrize(

 129     def test_get_indexer_closed(self, closed, leaf_size):
 130         x = np.arange(1000, dtype="float64")
 131         found = x.astype("intp")
 132         not_found = (-1 * np.ones(1000)).astype("intp")
 133 
 134         tree = IntervalTree(x, x + 0.5, closed=closed, leaf_size=leaf_size)
 135         tm.assert_numpy_array_equal(found, tree.get_indexer(x + 0.25))
 136 
 137         expected = found if tree.closed_left else not_found
 138         tm.assert_numpy_array_equal(expected, tree.get_indexer(x + 0.0))
 139 
 140         expected = found if tree.closed_right else not_found
 141         tm.assert_numpy_array_equal(expected, tree.get_indexer(x + 0.5))
