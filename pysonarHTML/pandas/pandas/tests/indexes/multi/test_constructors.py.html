  62 def test_constructor_mismatched_codes_levels(idx):
  63     codes = [np.array([1]), np.array([2]), np.array([3])]
  64     levels = ["a"]
  65 
  66     msg = "Length of levels and codes must be the same"
  67     with pytest.raises(ValueError, match=msg):
  68         MultiIndex(levels=levels, codes=codes)
  69 
  70     length_error = (
  71         r"On level 0, code max \(3\) >= length of level \(1\)\. "
  72         "NOTE: this index is in an inconsistent state"
  73     )
  74     label_error = r"Unequal code lengths: \[4, 2\]"
  75     code_value_error = r"On level 0, code value \(-2\) < -1"
  76 
  77     # important to check that it's looking at the right thing.
  78     with pytest.raises(ValueError, match=length_error):
  79         MultiIndex(levels=[["a"], ["b"]], codes=[[0, 1, 2, 3], [0, 3, 4, 1]])
  80 
  81     with pytest.raises(ValueError, match=label_error):
  82         MultiIndex(levels=[["a"], ["b"]], codes=[[0, 0, 0, 0], [0, 0]])
  83 
  84     # external API
  85     with pytest.raises(ValueError, match=length_error):
  86         idx.copy().set_levels([["a"], ["b"]])
  87 
  88     with pytest.raises(ValueError, match=label_error):
  89         idx.copy().set_codes([[0, 0, 0, 0], [0, 0]])
  90 
  91     # test set_codes with verify_integrity=False
  92     # the setting should not raise any value error
  93     idx.copy().set_codes(codes=[[0, 0, 0, 0], [0, 0]], verify_integrity=False)
  94 
  95     # code value smaller than -1
  96     with pytest.raises(ValueError, match=code_value_error):
  97         MultiIndex(levels=[["a"], ["b"]], codes=[[0, -2], [0, 0]])
