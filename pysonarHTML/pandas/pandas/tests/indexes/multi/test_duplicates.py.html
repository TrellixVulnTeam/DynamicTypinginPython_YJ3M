 185 def test_has_duplicates_overflow():
 186     # handle int64 overflow if possible
 187     def check(nlevels, with_nulls):
 188         codes = np.tile(np.arange(500), 2)
 189         level = np.arange(500)
 190 
 191         if with_nulls:  # inject some null values
 192             codes[500] = -1  # common nan value
 193             codes = [codes.copy() for i in range(nlevels)]
 194             for i in range(nlevels):
 195                 codes[i][500 + i - nlevels // 2] = -1
 196 
 197             codes += [np.array([-1, 1]).repeat(500)]
 198         else:
 199             codes = [codes] * nlevels + [np.arange(2).repeat(500)]
 200 
 201         levels = [level] * nlevels + [[0, 1]]
 202 
 203         # no dups
 204         mi = MultiIndex(levels=levels, codes=codes)
 205         assert not mi.has_duplicates
 206 
 207         # with a dup
 208         if with_nulls:
 209 
 210             def f(a):
 211                 return np.insert(a, 1000, a[0])
 212 
 213             codes = list(map(f, codes))
 214             mi = MultiIndex(levels=levels, codes=codes)
 215         else:
 216             values = mi.values.tolist()
 217             mi = MultiIndex.from_tuples(values + [values[0]])
 218 
 219         assert mi.has_duplicates
 220 
 221     # no overflow
 222     check(4, False)
 223     check(4, True)
 224 
 225     # overflow possible
 226     check(8, False)
 227     check(8, True)

 257     # GH5873

 280 def test_duplicated_drop_duplicates():
 281     # GH#4060
 282     idx = MultiIndex.from_arrays(([1, 2, 3, 1, 2, 3], [1, 1, 1, 1, 2, 2]))
 283 
 284     expected = np.array([False, False, False, True, False, False], dtype=bool)
 285     duplicated = idx.duplicated()
 286     tm.assert_numpy_array_equal(duplicated, expected)
 287     assert duplicated.dtype == bool
 288     expected = MultiIndex.from_arrays(([1, 2, 3, 2, 3], [1, 1, 1, 2, 2]))
 289     tm.assert_index_equal(idx.drop_duplicates(), expected)
 290 
 291     expected = np.array([True, False, False, False, False, False])
 292     duplicated = idx.duplicated(keep="last")
 293     tm.assert_numpy_array_equal(duplicated, expected)
 294     assert duplicated.dtype == bool
 295     expected = MultiIndex.from_arrays(([2, 3, 1, 2, 3], [1, 1, 1, 2, 2]))
 296     tm.assert_index_equal(idx.drop_duplicates(keep="last"), expected)
 297 
 298     expected = np.array([True, False, False, True, False, False])
 299     duplicated = idx.duplicated(keep=False)
 300     tm.assert_numpy_array_equal(duplicated, expected)
 301     assert duplicated.dtype == bool
