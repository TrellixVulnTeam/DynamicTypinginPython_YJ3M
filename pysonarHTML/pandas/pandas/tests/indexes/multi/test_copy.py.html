   9 def assert_multiindex_copied(copy, original):
  10     # Levels should be (at least, shallow copied)
  11     tm.assert_copy(copy.levels, original.levels)
  12     tm.assert_almost_equal(copy.codes, original.codes)
  13 
  14     # Labels doesn't matter which way copied
  15     tm.assert_almost_equal(copy.codes, original.codes)
  16     assert copy.codes is not original.codes
  17 
  18     # Names doesn't matter which way copied
  19     assert copy.names == original.names
  20     assert copy.names is not original.names
  21 
  22     # Sort order should be copied
  23     assert copy.sortorder == original.sortorder

  76 def test_copy_method_kwargs(deep, kwarg, value):
  77     # gh-12309: Check that the "name" argument as well other kwargs are honored
  78     idx = MultiIndex(
  79         levels=[["foo", "bar"], ["fizz", "buzz"]],
  80         codes=[[0, 0, 0, 1], [0, 0, 1, 1]],
  81         names=["first", "second"],
  82     )
  83     idx_copy = idx.copy(**{kwarg: value, "deep": deep})
  84     if kwarg == "names":
  85         assert getattr(idx_copy, kwarg) == value
  86     else:
  87         assert [list(i) for i in getattr(idx_copy, kwarg)] == value
