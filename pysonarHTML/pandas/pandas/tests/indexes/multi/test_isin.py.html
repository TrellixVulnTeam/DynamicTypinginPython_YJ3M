  10 @pytest.mark.skipif(not PYPY, reason="tuples cmp recursively on PyPy")
  11 def test_isin_nan_pypy():
  12     idx = MultiIndex.from_arrays([["foo", "bar"], [1.0, np.nan]])

  19 def test_isin():
  20     values = [("foo", 2), ("bar", 3), ("quux", 4)]
  21 
  22     idx = MultiIndex.from_arrays([["qux", "baz", "foo", "bar"], np.arange(4)])
  23     result = idx.isin(values)
  24     expected = np.array([False, False, True, True])
  25     tm.assert_numpy_array_equal(result, expected)
  26 
  27     # empty, return dtype bool
  28     idx = MultiIndex.from_arrays([[], []])
  29     result = idx.isin(values)
  30     assert len(result) == 0
  31     assert result.dtype == np.bool_

  43 def test_isin_level_kwarg():
  44     idx = MultiIndex.from_arrays([["qux", "baz", "foo", "bar"], np.arange(4)])
  45 
  46     vals_0 = ["foo", "bar", "quux"]
  47     vals_1 = [2, 3, 10]
  48 
  49     expected = np.array([False, False, True, True])
  50     tm.assert_numpy_array_equal(expected, idx.isin(vals_0, level=0))
  51     tm.assert_numpy_array_equal(expected, idx.isin(vals_0, level=-2))
  52 
  53     tm.assert_numpy_array_equal(expected, idx.isin(vals_1, level=1))
  54     tm.assert_numpy_array_equal(expected, idx.isin(vals_1, level=-1))
  55 
  56     msg = "Too many levels: Index has only 2 levels, not 6"
  57     with pytest.raises(IndexError, match=msg):
  58         idx.isin(vals_0, level=5)
  59     msg = "Too many levels: Index has only 2 levels, -5 is not a valid level number"
  60     with pytest.raises(IndexError, match=msg):
  61         idx.isin(vals_0, level=-5)
  62 
  63     with pytest.raises(KeyError, match=r"'Level 1\.0 not found'"):
  64         idx.isin(vals_0, level=1.0)
  65     with pytest.raises(KeyError, match=r"'Level -1\.0 not found'"):
  66         idx.isin(vals_1, level=-1.0)
  67     with pytest.raises(KeyError, match="'Level A not found'"):
  68         idx.isin(vals_1, level="A")
  69 
  70     idx.names = ["A", "B"]
  71     tm.assert_numpy_array_equal(expected, idx.isin(vals_0, level="A"))
  72     tm.assert_numpy_array_equal(expected, idx.isin(vals_1, level="B"))
  73 
  74     with pytest.raises(KeyError, match="'Level C not found'"):
  75         idx.isin(vals_1, level="C")
