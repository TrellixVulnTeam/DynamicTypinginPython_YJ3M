 145     def test_get_indexer(self):
 146         major_axis = Index(np.arange(4))
 147         minor_axis = Index(np.arange(2))
 148 
 149         major_codes = np.array([0, 0, 1, 2, 2, 3, 3], dtype=np.intp)
 150         minor_codes = np.array([0, 1, 0, 0, 1, 0, 1], dtype=np.intp)
 151 
 152         index = MultiIndex(
 153             levels=[major_axis, minor_axis], codes=[major_codes, minor_codes]
 154         )
 155         idx1 = index[:5]
 156         idx2 = index[[1, 3, 5]]
 157 
 158         r1 = idx1.get_indexer(idx2)
 159         tm.assert_almost_equal(r1, np.array([1, 3, -1], dtype=np.intp))
 160 
 161         r1 = idx2.get_indexer(idx1, method="pad")
 162         e1 = np.array([-1, 0, 0, 1, 1], dtype=np.intp)
 163         tm.assert_almost_equal(r1, e1)
 164 
 165         r2 = idx2.get_indexer(idx1[::-1], method="pad")
 166         tm.assert_almost_equal(r2, e1[::-1])
 167 
 168         rffill1 = idx2.get_indexer(idx1, method="ffill")
 169         tm.assert_almost_equal(r1, rffill1)
 170 
 171         r1 = idx2.get_indexer(idx1, method="backfill")
 172         e1 = np.array([0, 0, 1, 1, 2], dtype=np.intp)
 173         tm.assert_almost_equal(r1, e1)
 174 
 175         r2 = idx2.get_indexer(idx1[::-1], method="backfill")
 176         tm.assert_almost_equal(r2, e1[::-1])
 177 
 178         rbfill1 = idx2.get_indexer(idx1, method="bfill")
 179         tm.assert_almost_equal(r1, rbfill1)
 180 
 181         # pass non-MultiIndex
 182         r1 = idx1.get_indexer(idx2.values)
 183         rexp1 = idx1.get_indexer(idx2)
 184         tm.assert_almost_equal(r1, rexp1)
 185 
 186         r1 = idx1.get_indexer([1, 2, 3])
 187         assert (r1 == [-1, -1, -1]).all()
 188 
 189         # create index with duplicates
 190         idx1 = Index(list(range(10)) + list(range(10)))
 191         idx2 = Index(list(range(20)))
 192 
 193         msg = "Reindexing only valid with uniquely valued Index objects"
 194         with pytest.raises(InvalidIndexError, match=msg):
 195             idx1.get_indexer(idx2)

 245         # test getting an indexer for another index with different methods
 246         # confirms that getting an indexer without a filling method, getting an

 249         # several levels in the MultiIndex into which they are getting an indexer

 252         # mult_idx_1:
 253         #  0: -1 0
 254         #  1:    2
 255         #  2:    3

 267         #  0: 0 1
 268         #  1:   3
 269         #  2:   4

 282         backfill_indexer = mult_idx_1.get_indexer(mult_idx_2, method="bfill")

 791 def test_pyint_engine():
 792     # GH#18519 : when combinations of codes cannot be represented in 64
 793     # bits, the index underlying the MultiIndex engine works with Python
 794     # integers, rather than uint64.
 795     N = 5
 796     keys = [
 797         tuple(l)
 798         for l in [
 799             [0] * 10 * N,
 800             [1] * 10 * N,
 801             [2] * 10 * N,
 802             [np.nan] * N + [2] * 9 * N,
 803             [0] * N + [2] * 9 * N,
 804             [np.nan] * N + [2] * 8 * N + [0] * N,
 805         ]
 806     ]
 807     # Each level contains 4 elements (including NaN), so it is represented
 808     # in 2 bits, for a total of 2*N*10 = 100 > 64 bits. If we were using a
 809     # 64 bit engine and truncating the first levels, the fourth and fifth
 810     # keys would collide; if truncating the last levels, the fifth and
 811     # sixth; if rotating bits rather than shifting, the third and fifth.
 812 
 813     for idx in range(len(keys)):
 814         index = MultiIndex.from_tuples(keys)
 815         assert index.get_loc(keys[idx]) == idx
 816 
 817         expected = np.arange(idx + 1, dtype=np.intp)
 818         result = index.get_indexer([keys[i] for i in expected])
 819         tm.assert_numpy_array_equal(result, expected)
 820 
 821     # With missing key:
 822     idces = range(len(keys))
 823     expected = np.array([-1] + list(idces), dtype=np.intp)
 824     missing = tuple([0, 1] * 5 * N)
 825     result = index.get_indexer([missing] + [keys[i] for i in idces])
 826     tm.assert_numpy_array_equal(result, expected)
