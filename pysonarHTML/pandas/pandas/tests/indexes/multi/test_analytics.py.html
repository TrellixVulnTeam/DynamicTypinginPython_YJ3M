 176 @pytest.mark.parametrize(
 177     "mapper",
 178     [
 179         lambda values, idx: {i: e for e, i in zip(values, idx)},
 180         lambda values, idx: pd.Series(values, idx),
 181     ],
 182 )
 183 def test_map_dictlike(idx, mapper):
 184 
 185     if isinstance(idx, (pd.CategoricalIndex, pd.IntervalIndex)):
 186         pytest.skip(f"skipping tests for {type(idx)}")
 187 
 188     identity = mapper(idx.values, idx)
 189 
 190     # we don't infer to UInt64 for a dict
 191     if isinstance(idx, pd.UInt64Index) and isinstance(identity, dict):
 192         expected = idx.astype("int64")
 193     else:
 194         expected = idx
 195 
 196     result = idx.map(identity)
 197     tm.assert_index_equal(result, expected)
 198 
 199     # empty mappable
 200     expected = pd.Index([np.nan] * len(idx))
 201     result = idx.map(mapper(expected, idx))
 202     tm.assert_index_equal(result, expected)
