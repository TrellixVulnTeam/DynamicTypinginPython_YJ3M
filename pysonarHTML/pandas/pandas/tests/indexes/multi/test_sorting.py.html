  69 def test_numpy_argsort(idx):
  70     result = np.argsort(idx)
  71     expected = idx.argsort()
  72     tm.assert_numpy_array_equal(result, expected)
  73 
  74     # these are the only two types that perform
  75     # pandas compatibility input validation - the
  76     # rest already perform separate (or no) such
  77     # validation via their 'values' attribute as
  78     # defined in pandas.core.indexes/base.py - they
  79     # cannot be changed at the moment due to
  80     # backwards compatibility concerns
  81     if isinstance(type(idx), (CategoricalIndex, RangeIndex)):
  82         msg = "the 'axis' parameter is not supported"
  83         with pytest.raises(ValueError, match=msg):
  84             np.argsort(idx, axis=1)
  85 
  86         msg = "the 'kind' parameter is not supported"
  87         with pytest.raises(ValueError, match=msg):
  88             np.argsort(idx, kind="mergesort")
  89 
  90         msg = "the 'order' parameter is not supported"
  91         with pytest.raises(ValueError, match=msg):
  92             np.argsort(idx, order=("a", "b"))

  95 def test_unsortedindex():
  96     # GH 11897
  97     mi = pd.MultiIndex.from_tuples(
  98         [("z", "a"), ("x", "a"), ("y", "b"), ("x", "b"), ("y", "a"), ("z", "b")],
  99         names=["one", "two"],
 100     )
 101     df = pd.DataFrame([[i, 10 * i] for i in range(6)], index=mi, columns=["one", "two"])
 102 
 103     # GH 16734: not sorted, but no real slicing
 104     result = df.loc(axis=0)["z", "a"]
 105     expected = df.iloc[0]
 106     tm.assert_series_equal(result, expected)
 107 
 108     msg = (
 109         "MultiIndex slicing requires the index to be lexsorted: "
 110         r"slicing on levels \[1\], lexsort depth 0"
 111     )
 112     with pytest.raises(UnsortedIndexError, match=msg):
 113         df.loc(axis=0)["z", slice("a")]
 114     df.sort_index(inplace=True)
 115     assert len(df.loc(axis=0)["z", :]) == 2
 116 
 117     with pytest.raises(KeyError, match="'q'"):
 118         df.loc(axis=0)["q", :]

 121 def test_unsortedindex_doc_examples():
 122     # https://pandas.pydata.org/pandas-docs/stable/advanced.html#sorting-a-multiindex  # noqa
 123     dfm = DataFrame(
 124         {"jim": [0, 0, 1, 1], "joe": ["x", "x", "z", "y"], "jolie": np.random.rand(4)}
 125     )
 126 
 127     dfm = dfm.set_index(["jim", "joe"])
 128     with tm.assert_produces_warning(PerformanceWarning):
 129         dfm.loc[(1, "z")]
 130 
 131     msg = r"Key length \(2\) was greater than MultiIndex lexsort depth \(1\)"
 132     with pytest.raises(UnsortedIndexError, match=msg):
 133         dfm.loc[(0, "y"):(1, "z")]
 134 
 135     assert not dfm.index.is_lexsorted()
 136     assert dfm.index.lexsort_depth == 1
 137 
 138     # sort it
 139     dfm = dfm.sort_index()
 140     dfm.loc[(1, "z")]
 141     dfm.loc[(0, "y"):(1, "z")]
 142 
 143     assert dfm.index.is_lexsorted()
 144     assert dfm.index.lexsort_depth == 2

 147 def test_reconstruct_sort():
 148 
 149     # starts off lexsorted & monotonic
 150     mi = MultiIndex.from_arrays([["A", "A", "B", "B", "B"], [1, 2, 1, 2, 3]])
 151     assert mi.is_lexsorted()
 152     assert mi.is_monotonic
 153 
 154     recons = mi._sort_levels_monotonic()
 155     assert recons.is_lexsorted()
 156     assert recons.is_monotonic
 157     assert mi is recons
 158 
 159     assert mi.equals(recons)
 160     assert Index(mi.values).equals(Index(recons.values))
 161 
 162     # cannot convert to lexsorted
 163     mi = pd.MultiIndex.from_tuples(
 164         [("z", "a"), ("x", "a"), ("y", "b"), ("x", "b"), ("y", "a"), ("z", "b")],

 166     )
 167     assert not mi.is_lexsorted()
 168     assert not mi.is_monotonic
 169 
 170     recons = mi._sort_levels_monotonic()
 171     assert not recons.is_lexsorted()
 172     assert not recons.is_monotonic
 173 
 174     assert mi.equals(recons)
 175     assert Index(mi.values).equals(Index(recons.values))
 176 
 177     # cannot convert to lexsorted
 178     mi = MultiIndex(
 179         levels=[["b", "d", "a"], [1, 2, 3]],
 180         codes=[[0, 1, 0, 2], [2, 0, 0, 1]],
 181         names=["col1", "col2"],
 182     )
 183     assert not mi.is_lexsorted()
 184     assert not mi.is_monotonic

 187     assert not recons.is_lexsorted()
 188     assert not recons.is_monotonic

 194 def test_reconstruct_remove_unused():
 195     # xref to GH 2770
 196     df = DataFrame(
 197         [["deleteMe", 1, 9], ["keepMe", 2, 9], ["keepMeToo", 3, 9]],
 198         columns=["first", "second", "third"],
 199     )
 200     df2 = df.set_index(["first", "second"], drop=False)
 201     df2 = df2[df2["first"] != "deleteMe"]
 202 
 203     # removed levels are there
 204     expected = MultiIndex(
 205         levels=[["deleteMe", "keepMe", "keepMeToo"], [1, 2, 3]],
 206         codes=[[1, 2], [1, 2]],
 207         names=["first", "second"],
 208     )
 209     result = df2.index
 210     tm.assert_index_equal(result, expected)
 211 
 212     expected = MultiIndex(
 213         levels=[["keepMe", "keepMeToo"], [2, 3]],
 214         codes=[[0, 1], [0, 1]],
 215         names=["first", "second"],
 216     )
 217     result = df2.index.remove_unused_levels()
 218     tm.assert_index_equal(result, expected)
 219 
 220     # idempotent
 221     result2 = result.remove_unused_levels()
 222     tm.assert_index_equal(result2, expected)
 223     assert result2.is_(result)

 229 def test_remove_unused_levels_large(first_type, second_type):
 230     # GH16556
 231 
 232     # because tests should be deterministic (and this test in particular
 233     # checks that levels are removed, which is not the case for every
 234     # random input):
 235     rng = np.random.RandomState(4)  # seed is arbitrary value that works
 236 
 237     size = 1 << 16
 238     df = DataFrame(
 239         dict(
 240             first=rng.randint(0, 1 << 13, size).astype(first_type),
 241             second=rng.randint(0, 1 << 10, size).astype(second_type),
 242             third=rng.rand(size),
 243         )
 244     )
 245     df = df.groupby(["first", "second"]).sum()
 246     df = df[df.third < 0.1]
 247 
 248     result = df.index.remove_unused_levels()
 249     assert len(result.levels[0]) < len(df.index.levels[0])
 250     assert len(result.levels[1]) < len(df.index.levels[1])
 251     assert result.equals(df.index)
 252 
 253     expected = df.reset_index().set_index(["first", "second"]).index
 254     tm.assert_index_equal(result, expected)
