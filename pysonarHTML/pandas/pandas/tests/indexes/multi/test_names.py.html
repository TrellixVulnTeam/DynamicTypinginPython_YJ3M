  29 def test_changing_names(idx):
  30     assert [level.name for level in idx.levels] == ["first", "second"]
  31 
  32     view = idx.view()
  33     copy = idx.copy()
  34     shallow_copy = idx._shallow_copy()

  36     # changing names should not change level names on object

  47     shallow_copy.names = [name + "c" for name in shallow_copy.names]
  48     check_level_names(idx, ["firsta", "seconda"])

  79 def test_names(idx, index_names):
  80 
  81     # names are assigned in setup
  82     assert index_names == ["first", "second"]
  83     level_names = [level.name for level in idx.levels]
  84     assert level_names == index_names
  85 
  86     # setting bad names on existing
  87     index = idx
  88     with pytest.raises(ValueError, match="^Length of names"):
  89         setattr(index, "names", list(index.names) + ["third"])
  90     with pytest.raises(ValueError, match="^Length of names"):
  91         setattr(index, "names", [])
  92 
  93     # initializing with bad names (should always be equivalent)
  94     major_axis, minor_axis = idx.levels
  95     major_codes, minor_codes = idx.codes
  96     with pytest.raises(ValueError, match="^Length of names"):
  97         MultiIndex(
  98             levels=[major_axis, minor_axis],
  99             codes=[major_codes, minor_codes],
 100             names=["first"],
 101         )
 102     with pytest.raises(ValueError, match="^Length of names"):
 103         MultiIndex(
 104             levels=[major_axis, minor_axis],
 105             codes=[major_codes, minor_codes],
 106             names=["first", "second", "third"],
 107         )
 108 
 109     # names are assigned on index, but not transferred to the levels
 110     index.names = ["a", "b"]
 111     level_names = [level.name for level in index.levels]
 112     assert level_names == ["a", "b"]
