   9 def assert_matching(actual, expected, check_dtype=False):
  10     # avoid specifying internal representation
  11     # as much as possible
  12     assert len(actual) == len(expected)
  13     for act, exp in zip(actual, expected):
  14         act = np.asarray(act)
  15         exp = np.asarray(exp)
  16         tm.assert_numpy_array_equal(act, exp, check_dtype=check_dtype)

  83 def test_set_levels(idx):
  84     # side note - you probably wouldn't want to use levels and codes
  85     # directly like this - but it is possible.
  86     levels = idx.levels
  87     new_levels = [[lev + "a" for lev in level] for level in levels]
  88 
  89     # level changing [w/o mutation]
  90     ind2 = idx.set_levels(new_levels)
  91     assert_matching(ind2.levels, new_levels)
  92     assert_matching(idx.levels, levels)
  93 
  94     # level changing [w/ mutation]
  95     ind2 = idx.copy()
  96     inplace_return = ind2.set_levels(new_levels, inplace=True)
  97     assert inplace_return is None
  98     assert_matching(ind2.levels, new_levels)
  99 
 100     # level changing specific level [w/o mutation]
 101     ind2 = idx.set_levels(new_levels[0], level=0)
 102     assert_matching(ind2.levels, [new_levels[0], levels[1]])
 103     assert_matching(idx.levels, levels)
 104 
 105     ind2 = idx.set_levels(new_levels[1], level=1)
 106     assert_matching(ind2.levels, [levels[0], new_levels[1]])
 107     assert_matching(idx.levels, levels)
 108 
 109     # level changing multiple levels [w/o mutation]
 110     ind2 = idx.set_levels(new_levels, level=[0, 1])
 111     assert_matching(ind2.levels, new_levels)
 112     assert_matching(idx.levels, levels)
 113 
 114     # level changing specific level [w/ mutation]
 115     ind2 = idx.copy()
 116     inplace_return = ind2.set_levels(new_levels[0], level=0, inplace=True)
 117     assert inplace_return is None
 118     assert_matching(ind2.levels, [new_levels[0], levels[1]])
 119     assert_matching(idx.levels, levels)
 120 
 121     ind2 = idx.copy()
 122     inplace_return = ind2.set_levels(new_levels[1], level=1, inplace=True)
 123     assert inplace_return is None
 124     assert_matching(ind2.levels, [levels[0], new_levels[1]])
 125     assert_matching(idx.levels, levels)
 126 
 127     # level changing multiple levels [w/ mutation]
 128     ind2 = idx.copy()
 129     inplace_return = ind2.set_levels(new_levels, level=[0, 1], inplace=True)
 130     assert inplace_return is None
 131     assert_matching(ind2.levels, new_levels)
 132     assert_matching(idx.levels, levels)
 133 
 134     # illegal level changing should not change levels
 135     # GH 13754
 136     original_index = idx.copy()
 137     for inplace in [True, False]:
 138         with pytest.raises(ValueError, match="^On"):
 139             idx.set_levels(["c"], level=0, inplace=inplace)
 140         assert_matching(idx.levels, original_index.levels, check_dtype=True)
 141 
 142         with pytest.raises(ValueError, match="^On"):
 143             idx.set_codes([0, 1, 2, 3, 4, 5], level=0, inplace=inplace)
 144         assert_matching(idx.codes, original_index.codes, check_dtype=True)
 145 
 146         with pytest.raises(TypeError, match="^Levels"):
 147             idx.set_levels("c", level=0, inplace=inplace)
 148         assert_matching(idx.levels, original_index.levels, check_dtype=True)
 149 
 150         with pytest.raises(TypeError, match="^Codes"):
 151             idx.set_codes(1, level=0, inplace=inplace)
 152         assert_matching(idx.codes, original_index.codes, check_dtype=True)

 155 def test_set_codes(idx):
 156     # side note - you probably wouldn't want to use levels and codes
 157     # directly like this - but it is possible.
 158     codes = idx.codes
 159     major_codes, minor_codes = codes
 160     major_codes = [(x + 1) % 3 for x in major_codes]
 161     minor_codes = [(x + 1) % 1 for x in minor_codes]
 162     new_codes = [major_codes, minor_codes]
 163 
 164     # changing codes w/o mutation
 165     ind2 = idx.set_codes(new_codes)
 166     assert_matching(ind2.codes, new_codes)
 167     assert_matching(idx.codes, codes)
 168 
 169     # changing label w/ mutation
 170     ind2 = idx.copy()
 171     inplace_return = ind2.set_codes(new_codes, inplace=True)
 172     assert inplace_return is None
 173     assert_matching(ind2.codes, new_codes)
 174 
 175     # codes changing specific level w/o mutation
 176     ind2 = idx.set_codes(new_codes[0], level=0)
 177     assert_matching(ind2.codes, [new_codes[0], codes[1]])
 178     assert_matching(idx.codes, codes)
 179 
 180     ind2 = idx.set_codes(new_codes[1], level=1)
 181     assert_matching(ind2.codes, [codes[0], new_codes[1]])
 182     assert_matching(idx.codes, codes)
 183 
 184     # codes changing multiple levels w/o mutation
 185     ind2 = idx.set_codes(new_codes, level=[0, 1])
 186     assert_matching(ind2.codes, new_codes)
 187     assert_matching(idx.codes, codes)
 188 
 189     # label changing specific level w/ mutation
 190     ind2 = idx.copy()
 191     inplace_return = ind2.set_codes(new_codes[0], level=0, inplace=True)
 192     assert inplace_return is None
 193     assert_matching(ind2.codes, [new_codes[0], codes[1]])
 194     assert_matching(idx.codes, codes)
 195 
 196     ind2 = idx.copy()
 197     inplace_return = ind2.set_codes(new_codes[1], level=1, inplace=True)
 198     assert inplace_return is None
 199     assert_matching(ind2.codes, [codes[0], new_codes[1]])
 200     assert_matching(idx.codes, codes)
 201 
 202     # codes changing multiple levels [w/ mutation]
 203     ind2 = idx.copy()
 204     inplace_return = ind2.set_codes(new_codes, level=[0, 1], inplace=True)
 205     assert inplace_return is None
 206     assert_matching(ind2.codes, new_codes)
 207     assert_matching(idx.codes, codes)

 210     ind = pd.MultiIndex.from_tuples([(0, i) for i in range(130)])
 211     new_codes = range(129, -1, -1)
 212     expected = pd.MultiIndex.from_tuples([(0, i) for i in new_codes])
 213 
 214     # [w/o mutation]
 215     result = ind.set_codes(codes=new_codes, level=1)
 216     assert result.equals(expected)
 217 
 218     # [w/ mutation]
 219     result = ind.copy()
 220     result.set_codes(codes=new_codes, level=1, inplace=True)
 221     assert result.equals(expected)

 239         idx.set_levels(levels[0])

 241     # shouldn't scalar data error, instead should demand list-like
 242     with pytest.raises(TypeError, match="list of lists-like"):
 243         idx.set_codes(codes[0])

 247         idx.set_names(names[0])

 303 def test_set_value_keeps_names():
 304     # motivating example from #3742
 305     lev1 = ["hans", "hans", "hans", "grethe", "grethe", "grethe"]
 306     lev2 = ["1", "2", "3"] * 2
 307     idx = pd.MultiIndex.from_arrays([lev1, lev2], names=["Name", "Number"])
 308     df = pd.DataFrame(
 309         np.random.randn(6, 4), columns=["one", "two", "three", "four"], index=idx
 310     )
 311     df = df.sort_index()
 312     assert df._is_copy is None
 313     assert df.index.names == ("Name", "Number")
 314     df.at[("grethe", "4"), "one"] = 99.34
 315     assert df._is_copy is None
 316     assert df.index.names == ("Name", "Number")
