  37 def test_take_fill_value():
  38     # GH 12631
  39     vals = [["A", "B"], [pd.Timestamp("2011-01-01"), pd.Timestamp("2011-01-02")]]
  40     idx = pd.MultiIndex.from_product(vals, names=["str", "dt"])
  41 
  42     result = idx.take(np.array([1, 0, -1]))
  43     exp_vals = [
  44         ("A", pd.Timestamp("2011-01-02")),
  45         ("A", pd.Timestamp("2011-01-01")),
  46         ("B", pd.Timestamp("2011-01-02")),
  47     ]
  48     expected = pd.MultiIndex.from_tuples(exp_vals, names=["str", "dt"])
  49     tm.assert_index_equal(result, expected)
  50 
  51     # fill_value
  52     result = idx.take(np.array([1, 0, -1]), fill_value=True)
  53     exp_vals = [
  54         ("A", pd.Timestamp("2011-01-02")),
  55         ("A", pd.Timestamp("2011-01-01")),
  56         (np.nan, pd.NaT),
  57     ]
  58     expected = pd.MultiIndex.from_tuples(exp_vals, names=["str", "dt"])
  59     tm.assert_index_equal(result, expected)
  60 
  61     # allow_fill=False
  62     result = idx.take(np.array([1, 0, -1]), allow_fill=False, fill_value=True)
  63     exp_vals = [
  64         ("A", pd.Timestamp("2011-01-02")),
  65         ("A", pd.Timestamp("2011-01-01")),
  66         ("B", pd.Timestamp("2011-01-02")),
  67     ]
  68     expected = pd.MultiIndex.from_tuples(exp_vals, names=["str", "dt"])
  69     tm.assert_index_equal(result, expected)
  70 
  71     msg = "When allow_fill=True and fill_value is not None, all indices must be >= -1"
  72     with pytest.raises(ValueError, match=msg):
  73         idx.take(np.array([1, 0, -2]), fill_value=True)
  74     with pytest.raises(ValueError, match=msg):
  75         idx.take(np.array([1, 0, -5]), fill_value=True)
  76 
  77     msg = "index -5 is out of bounds for( axis 0 with)? size 4"
  78     with pytest.raises(IndexError, match=msg):
  79         idx.take(np.array([1, -5]))
