  11 def test_drop(idx):
  12     dropped = idx.drop([("foo", "two"), ("qux", "one")])
  13 
  14     index = MultiIndex.from_tuples([("foo", "two"), ("qux", "one")])
  15     dropped2 = idx.drop(index)
  16 
  17     expected = idx[[0, 2, 3, 5]]
  18     tm.assert_index_equal(dropped, expected)
  19     tm.assert_index_equal(dropped2, expected)
  20 
  21     dropped = idx.drop(["bar"])
  22     expected = idx[[0, 1, 3, 4, 5]]
  23     tm.assert_index_equal(dropped, expected)
  24 
  25     dropped = idx.drop("foo")
  26     expected = idx[[2, 3, 4, 5]]
  27     tm.assert_index_equal(dropped, expected)
  28 
  29     index = MultiIndex.from_tuples([("bar", "two")])
  30     with pytest.raises(KeyError, match=r"^10$"):
  31         idx.drop([("bar", "two")])
  32     with pytest.raises(KeyError, match=r"^10$"):
  33         idx.drop(index)
  34     with pytest.raises(KeyError, match=r"^'two'$"):
  35         idx.drop(["foo", "two"])
  36 
  37     # partially correct argument
  38     mixed_index = MultiIndex.from_tuples([("qux", "one"), ("bar", "two")])
  39     with pytest.raises(KeyError, match=r"^10$"):
  40         idx.drop(mixed_index)
  41 
  42     # error='ignore'
  43     dropped = idx.drop(index, errors="ignore")
  44     expected = idx[[0, 1, 2, 3, 4, 5]]
  45     tm.assert_index_equal(dropped, expected)
  46 
  47     dropped = idx.drop(mixed_index, errors="ignore")
  48     expected = idx[[0, 1, 2, 3, 5]]
  49     tm.assert_index_equal(dropped, expected)
  50 
  51     dropped = idx.drop(["foo", "two"], errors="ignore")
  52     expected = idx[[2, 3, 4, 5]]
  53     tm.assert_index_equal(dropped, expected)
  54 
  55     # mixed partial / full drop
  56     dropped = idx.drop(["foo", ("qux", "one")])
  57     expected = idx[[2, 3, 5]]
  58     tm.assert_index_equal(dropped, expected)
  59 
  60     # mixed partial / full drop / error='ignore'
  61     mixed_index = ["foo", ("qux", "one"), "two"]
  62     with pytest.raises(KeyError, match=r"^'two'$"):
  63         idx.drop(mixed_index)
  64     dropped = idx.drop(mixed_index, errors="ignore")
  65     expected = idx[[2, 3, 5]]
  66     tm.assert_index_equal(dropped, expected)

  69 def test_droplevel_with_names(idx):
  70     index = idx[idx.get_loc("foo")]
  71     dropped = index.droplevel(0)
  72     assert dropped.name == "second"
  73 
  74     index = MultiIndex(
  75         levels=[Index(range(4)), Index(range(4)), Index(range(4))],
  76         codes=[
  77             np.array([0, 0, 1, 2, 2, 2, 3, 3]),
  78             np.array([0, 1, 0, 0, 0, 1, 0, 1]),
  79             np.array([1, 0, 1, 1, 0, 0, 1, 0]),
  80         ],
  81         names=["one", "two", "three"],
  82     )
  83     dropped = index.droplevel(0)
  84     assert dropped.names == ("two", "three")
  85 
  86     dropped = index.droplevel("two")
  87     expected = index.droplevel(1)
  88     assert dropped.equals(expected)

  91 def test_droplevel_list():
  92     index = MultiIndex(
  93         levels=[Index(range(4)), Index(range(4)), Index(range(4))],
  94         codes=[
  95             np.array([0, 0, 1, 2, 2, 2, 3, 3]),
  96             np.array([0, 1, 0, 0, 0, 1, 0, 1]),
  97             np.array([1, 0, 1, 1, 0, 0, 1, 0]),
  98         ],
  99         names=["one", "two", "three"],
 100     )
 101 
 102     dropped = index[:2].droplevel(["three", "one"])
 103     expected = index[:2].droplevel(2).droplevel(0)
 104     assert dropped.equals(expected)
 105 
 106     dropped = index[:2].droplevel([])
 107     expected = index[:2]
 108     assert dropped.equals(expected)
 109 
 110     msg = (
 111         "Cannot remove 3 levels from an index with 3 levels: "
 112         "at least one level must be left"
 113     )
 114     with pytest.raises(ValueError, match=msg):
 115         index[:2].droplevel(["one", "two", "three"])
 116 
 117     with pytest.raises(KeyError, match="'Level four not found'"):
 118         index[:2].droplevel(["one", "four"])

 121 def test_drop_not_lexsorted():
 122     # GH 12078
 123 
 124     # define the lexsorted version of the multi-index
 125     tuples = [("a", ""), ("b1", "c1"), ("b2", "c2")]
 126     lexsorted_mi = MultiIndex.from_tuples(tuples, names=["b", "c"])
 127     assert lexsorted_mi.is_lexsorted()
 128 
 129     # and the not-lexsorted version
 130     df = pd.DataFrame(
 131         columns=["a", "b", "c", "d"], data=[[1, "b1", "c1", 3], [1, "b2", "c2", 4]]
 132     )
 133     df = df.pivot_table(index="a", columns=["b", "c"], values="d")
 134     df = df.reset_index()
 135     not_lexsorted_mi = df.columns
 136     assert not not_lexsorted_mi.is_lexsorted()
 137 
 138     # compare the results
 139     tm.assert_index_equal(lexsorted_mi, not_lexsorted_mi)
 140     with tm.assert_produces_warning(PerformanceWarning):
 141         tm.assert_index_equal(lexsorted_mi.drop("a"), not_lexsorted_mi.drop("a"))
