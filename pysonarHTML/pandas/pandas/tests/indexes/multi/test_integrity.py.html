 201 def test_metadata_immutable(idx):
 202     levels, codes = idx.levels, idx.codes
 203     # shouldn't be able to set at either the top level or base level
 204     mutable_regex = re.compile("does not support mutable operations")
 205     with pytest.raises(TypeError, match=mutable_regex):
 206         levels[0] = levels[0]
 207     with pytest.raises(TypeError, match=mutable_regex):
 208         levels[0][0] = levels[0][0]
 209     # ditto for labels
 210     with pytest.raises(TypeError, match=mutable_regex):
 211         codes[0] = codes[0]
 212     with pytest.raises(ValueError, match="assignment destination is read-only"):
 213         codes[0][0] = codes[0][0]
 214     # and for names
 215     names = idx.names
 216     with pytest.raises(TypeError, match=mutable_regex):
 217         names[0] = names[0]
