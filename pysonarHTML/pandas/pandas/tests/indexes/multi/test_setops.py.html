 121 def test_difference(idx, sort):
 122 
 123     first = idx
 124     result = first.difference(idx[-3:], sort=sort)
 125     vals = idx[:-3].values
 126 
 127     if sort is None:
 128         vals = sorted(vals)
 129 
 130     expected = MultiIndex.from_tuples(vals, sortorder=0, names=idx.names)
 131 
 132     assert isinstance(result, MultiIndex)
 133     assert result.equals(expected)
 134     assert result.names == idx.names
 135     tm.assert_index_equal(result, expected)
 136 
 137     # empty difference: reflexive
 138     result = idx.difference(idx, sort=sort)
 139     expected = idx[:0]
 140     assert result.equals(expected)
 141     assert result.names == idx.names
 142 
 143     # empty difference: superset
 144     result = idx[-3:].difference(idx, sort=sort)
 145     expected = idx[:0]
 146     assert result.equals(expected)
 147     assert result.names == idx.names
 148 
 149     # empty difference: degenerate
 150     result = idx[:0].difference(idx, sort=sort)
 151     expected = idx[:0]
 152     assert result.equals(expected)
 153     assert result.names == idx.names
 154 
 155     # names not the same
 156     chunklet = idx[-3:]
 157     chunklet.names = ["foo", "baz"]
 158     result = first.difference(chunklet, sort=sort)
 159     assert result.names == (None, None)
 160 
 161     # empty, but non-equal
 162     result = idx.difference(idx.sortlevel(1)[0], sort=sort)
 163     assert len(result) == 0
 164 
 165     # raise Exception called with non-MultiIndex
 166     result = first.difference(first.values, sort=sort)
 167     assert result.equals(first[:0])
 168 
 169     # name from empty array
 170     result = first.difference([], sort=sort)
 171     assert first.equals(result)
 172     assert first.names == result.names
 173 
 174     # name from non-empty array
 175     result = first.difference([("foo", "one")], sort=sort)
 176     expected = pd.MultiIndex.from_tuples(
 177         [("bar", "one"), ("baz", "two"), ("foo", "two"), ("qux", "one"), ("qux", "two")]
 178     )
 179     expected.names = first.names
 180     assert first.names == result.names
 181 
 182     msg = "other must be a MultiIndex or a list of tuples"
 183     with pytest.raises(TypeError, match=msg):
 184         first.difference([1, 2, 3, 4, 5], sort=sort)
