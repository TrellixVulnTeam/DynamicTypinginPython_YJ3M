 141     def test_drop_duplicates_metadata(self, freq_sample):

 155     @pytest.mark.parametrize(
 156         "keep, expected, index",
 157         [
 158             ("first", np.concatenate(([False] * 10, [True] * 5)), np.arange(0, 10)),
 159             ("last", np.concatenate(([True] * 5, [False] * 10)), np.arange(5, 15)),
 160             (
 161                 False,
 162                 np.concatenate(([True] * 5, [False] * 5, [True] * 5)),
 163                 np.arange(5, 10),
 164             ),
 165         ],
 166     )
 167     def test_drop_duplicates(self, freq_sample, keep, expected, index):
 168         # to check Index/Series compat
 169         idx = pd.timedelta_range("1 day", periods=10, freq=freq_sample, name="idx")
 170         idx = idx.append(idx[:5])
 171 
 172         tm.assert_numpy_array_equal(idx.duplicated(keep=keep), expected)
 173         expected = idx[~expected]
 174 
 175         result = idx.drop_duplicates(keep=keep)
 176         tm.assert_index_equal(result, expected)
 177 
 178         result = Series(idx).drop_duplicates(keep=keep)
 179         tm.assert_series_equal(result, Series(expected, index=index))

 231     def test_equals(self):
 232         # GH 13107
 233         idx = pd.TimedeltaIndex(["1 days", "2 days", "NaT"])
 234         assert idx.equals(idx)
 235         assert idx.equals(idx.copy())
 236         assert idx.equals(idx.astype(object))
 237         assert idx.astype(object).equals(idx)
 238         assert idx.astype(object).equals(idx.astype(object))
 239         assert not idx.equals(list(idx))
 240         assert not idx.equals(pd.Series(idx))
 241 
 242         idx2 = pd.TimedeltaIndex(["2 days", "1 days", "NaT"])
 243         assert not idx.equals(idx2)
 244         assert not idx.equals(idx2.copy())
 245         assert not idx.equals(idx2.astype(object))
 246         assert not idx.astype(object).equals(idx2)
 247         assert not idx.astype(object).equals(idx2.astype(object))
 248         assert not idx.equals(list(idx2))
 249         assert not idx.equals(pd.Series(idx2))
 250 
 251         # Check that we dont raise OverflowError on comparisons outside the
 252         #  implementation range
 253         oob = pd.Index([timedelta(days=10 ** 6)] * 3, dtype=object)
 254         assert not idx.equals(oob)
 255         assert not idx2.equals(oob)
 256 
 257         # FIXME: oob.apply(np.timedelta64) incorrectly overflows
 258         oob2 = pd.Index([np.timedelta64(x) for x in oob], dtype=object)
 259         assert not idx.equals(oob2)
 260         assert not idx2.equals(oob2)

 298     def test_freq_view_safe(self):
 299         # Setting the freq for one TimedeltaIndex shouldn't alter the freq
 300         #  for another that views the same data
 301 
 302         tdi = TimedeltaIndex(["0 days", "2 days", "4 days"], freq="2D")
 303         tda = tdi._data
 304 
 305         tdi2 = TimedeltaIndex(tda)._with_freq(None)
 306         assert tdi2.freq is None
 307 
 308         # Original was not altered
 309         assert tdi.freq == "2D"
 310         assert tda.freq == "2D"
