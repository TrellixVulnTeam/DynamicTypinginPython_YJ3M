  69     def test_factorize(self):

 156     def test_fields(self):
 157         rng = timedelta_range("1 days, 10:11:12.100123456", periods=2, freq="s")
 158         tm.assert_index_equal(rng.days, Index([1, 1], dtype="int64"))
 159         tm.assert_index_equal(
 160             rng.seconds,
 161             Index([10 * 3600 + 11 * 60 + 12, 10 * 3600 + 11 * 60 + 13], dtype="int64"),
 162         )
 163         tm.assert_index_equal(
 164             rng.microseconds, Index([100 * 1000 + 123, 100 * 1000 + 123], dtype="int64")
 165         )
 166         tm.assert_index_equal(rng.nanoseconds, Index([456, 456], dtype="int64"))
 167 
 168         msg = "'TimedeltaIndex' object has no attribute '{}'"
 169         with pytest.raises(AttributeError, match=msg.format("hours")):
 170             rng.hours
 171         with pytest.raises(AttributeError, match=msg.format("minutes")):
 172             rng.minutes
 173         with pytest.raises(AttributeError, match=msg.format("milliseconds")):
 174             rng.milliseconds
 175 
 176         # with nat
 177         s = Series(rng)
 178         s[1] = np.nan
 179 
 180         tm.assert_series_equal(s.dt.days, Series([1, np.nan], index=[0, 1]))
 181         tm.assert_series_equal(
 182             s.dt.seconds, Series([10 * 3600 + 11 * 60 + 12, np.nan], index=[0, 1])
 183         )
 184 
 185         # preserve name (GH15589)
 186         rng.name = "name"
 187         assert rng.days.name == "name"

 189     def test_freq_conversion(self):
 190 
 191         # doc example
 192 
 193         # series
 194         td = Series(date_range("20130101", periods=4)) - Series(
 195             date_range("20121201", periods=4)
 196         )
 197         td[2] += timedelta(minutes=5, seconds=3)
 198         td[3] = np.nan
 199 
 200         result = td / np.timedelta64(1, "D")
 201         expected = Series([31, 31, (31 * 86400 + 5 * 60 + 3) / 86400.0, np.nan])
 202         tm.assert_series_equal(result, expected)
 203 
 204         result = td.astype("timedelta64[D]")
 205         expected = Series([31, 31, 31, np.nan])
 206         tm.assert_series_equal(result, expected)
 207 
 208         result = td / np.timedelta64(1, "s")
 209         expected = Series([31 * 86400, 31 * 86400, 31 * 86400 + 5 * 60 + 3, np.nan])
 210         tm.assert_series_equal(result, expected)
 211 
 212         result = td.astype("timedelta64[s]")
 213         tm.assert_series_equal(result, expected)
 214 
 215         # tdi
 216         td = TimedeltaIndex(td)
 217 
 218         result = td / np.timedelta64(1, "D")
 219         expected = Index([31, 31, (31 * 86400 + 5 * 60 + 3) / 86400.0, np.nan])
 220         tm.assert_index_equal(result, expected)
 221 
 222         result = td.astype("timedelta64[D]")
 223         expected = Index([31, 31, 31, np.nan])
 224         tm.assert_index_equal(result, expected)
 225 
 226         result = td / np.timedelta64(1, "s")
 227         expected = Index([31 * 86400, 31 * 86400, 31 * 86400 + 5 * 60 + 3, np.nan])
 228         tm.assert_index_equal(result, expected)
 229 
 230         result = td.astype("timedelta64[s]")
 231         tm.assert_index_equal(result, expected)
