  15 class TestVectorizedTimedelta:
  16     def test_tdi_total_seconds(self):
  17         # GH#10939
  18         # test index
  19         rng = timedelta_range("1 days, 10:11:12.100123456", periods=2, freq="s")
  20         expt = [
  21             1 * 86400 + 10 * 3600 + 11 * 60 + 12 + 100123456.0 / 1e9,
  22             1 * 86400 + 10 * 3600 + 11 * 60 + 13 + 100123456.0 / 1e9,
  23         ]
  24         tm.assert_almost_equal(rng.total_seconds(), Index(expt))
  25 
  26         # test Series
  27         ser = Series(rng)
  28         s_expt = Series(expt, index=[0, 1])
  29         tm.assert_series_equal(ser.dt.total_seconds(), s_expt)
  30 
  31         # with nat
  32         ser[1] = np.nan
  33         s_expt = Series(
  34             [1 * 86400 + 10 * 3600 + 11 * 60 + 12 + 100123456.0 / 1e9, np.nan],
  35             index=[0, 1],
  36         )
  37         tm.assert_series_equal(ser.dt.total_seconds(), s_expt)
  38 
  39         # with both nat
  40         ser = Series([np.nan, np.nan], dtype="timedelta64[ns]")
  41         tm.assert_series_equal(
  42             ser.dt.total_seconds(), Series([np.nan, np.nan], index=[0, 1])
  43         )

  45     def test_tdi_round(self):
  46         td = pd.timedelta_range(start="16801 days", periods=5, freq="30Min")
  47         elt = td[1]
  48 
  49         expected_rng = TimedeltaIndex(
  50             [
  51                 Timedelta("16801 days 00:00:00"),
  52                 Timedelta("16801 days 00:00:00"),
  53                 Timedelta("16801 days 01:00:00"),
  54                 Timedelta("16801 days 02:00:00"),
  55                 Timedelta("16801 days 02:00:00"),
  56             ]
  57         )
  58         expected_elt = expected_rng[1]
  59 
  60         tm.assert_index_equal(td.round(freq="H"), expected_rng)
  61         assert elt.round(freq="H") == expected_elt
  62 
  63         msg = INVALID_FREQ_ERR_MSG
  64         with pytest.raises(ValueError, match=msg):
  65             td.round(freq="foo")
  66         with pytest.raises(ValueError, match=msg):
  67             elt.round(freq="foo")
  68 
  69         msg = "<MonthEnd> is a non-fixed frequency"
  70         with pytest.raises(ValueError, match=msg):
  71             td.round(freq="M")
  72         with pytest.raises(ValueError, match=msg):
  73             elt.round(freq="M")

  83     def test_tdi_round_invalid(self, freq, msg):
  84         t1 = timedelta_range("1 days", periods=3, freq="1 min 2 s 3 us")

  87             t1.round(freq)
  88         with pytest.raises(ValueError, match=msg):

  90             t1._data.round(freq)

  93     def test_round(self):
  94         t1 = timedelta_range("1 days", periods=3, freq="1 min 2 s 3 us")
  95         t2 = -1 * t1
  96         t1a = timedelta_range("1 days", periods=3, freq="1 min 2 s")
  97         t1c = TimedeltaIndex([1, 1, 1], unit="D")
  98 
  99         # note that negative times round DOWN! so don't give whole numbers
 100         for (freq, s1, s2) in [
 101             ("N", t1, t2),
 102             ("U", t1, t2),
 103             (
 104                 "L",
 105                 t1a,
 106                 TimedeltaIndex(
 107                     ["-1 days +00:00:00", "-2 days +23:58:58", "-2 days +23:57:56"],
 108                 ),
 109             ),
 110             (
 111                 "S",
 112                 t1a,
 113                 TimedeltaIndex(
 114                     ["-1 days +00:00:00", "-2 days +23:58:58", "-2 days +23:57:56"],
 115                 ),
 116             ),
 117             ("12T", t1c, TimedeltaIndex(["-1 days", "-1 days", "-1 days"],),),
 118             ("H", t1c, TimedeltaIndex(["-1 days", "-1 days", "-1 days"],),),
 119             ("d", t1c, TimedeltaIndex([-1, -1, -1], unit="D")),
 120         ]:
 121 
 122             r1 = t1.round(freq)
 123             tm.assert_index_equal(r1, s1)
 124             r2 = t2.round(freq)
 125             tm.assert_index_equal(r2, s2)
