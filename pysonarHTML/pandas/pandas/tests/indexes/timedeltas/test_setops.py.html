 111     def test_intersection_equal(self, sort):
 112         # GH 24471 Test intersection outcome given the sort keyword
 113         # for equal indicies intersection should return the original index
 114         first = timedelta_range("1 day", periods=4, freq="h")
 115         second = timedelta_range("1 day", periods=4, freq="h")
 116         intersect = first.intersection(second, sort=sort)
 117         if sort is None:
 118             tm.assert_index_equal(intersect, second.sort_values())
 119         assert tm.equalContents(intersect, second)
 120 
 121         # Corner cases
 122         inter = first.intersection(first, sort=sort)
 123         assert inter is first

 173     @pytest.mark.parametrize(
 174         "rng, expected",
 175         # part intersection works
 176         [
 177             (
 178                 TimedeltaIndex(["5 hour", "2 hour", "4 hour", "9 hour"], name="idx"),
 179                 TimedeltaIndex(["2 hour", "4 hour"], name="idx"),
 180             ),
 181             # reordered part intersection
 182             (
 183                 TimedeltaIndex(["2 hour", "5 hour", "5 hour", "1 hour"], name="other"),
 184                 TimedeltaIndex(["1 hour", "2 hour"], name=None),
 185             ),
 186             # reversed index
 187             (
 188                 TimedeltaIndex(["1 hour", "2 hour", "4 hour", "3 hour"], name="idx")[

 190                 ],
 191                 TimedeltaIndex(["1 hour", "2 hour", "4 hour", "3 hour"], name="idx"),

 193         ],

 195     def test_intersection_non_monotonic(self, rng, expected, sort):
 196         # 24471 non-monotonic
 197         base = TimedeltaIndex(["1 hour", "2 hour", "4 hour", "3 hour"], name="idx")
 198         result = base.intersection(rng, sort=sort)
 199         if sort is None:
 200             expected = expected.sort_values()
 201         tm.assert_index_equal(result, expected)
 202         assert result.name == expected.name
 203 
 204         # if reversed order, frequency is still the same
 205         if all(base == rng[::-1]) and sort is None:
 206             assert isinstance(result.freq, Hour)
 207         else:
 208             assert result.freq is None

 211 class TestTimedeltaIndexDifference:
 212     def test_difference_freq(self, sort):
 213         # GH14323: Difference of TimedeltaIndex should not preserve frequency
 214 
 215         index = timedelta_range("0 days", "5 days", freq="D")
 216 
 217         other = timedelta_range("1 days", "4 days", freq="D")
 218         expected = TimedeltaIndex(["0 days", "5 days"], freq=None)
 219         idx_diff = index.difference(other, sort)
 220         tm.assert_index_equal(idx_diff, expected)
 221         tm.assert_attr_equal("freq", idx_diff, expected)
 222 
 223         other = timedelta_range("2 days", "5 days", freq="D")
 224         idx_diff = index.difference(other, sort)
 225         expected = TimedeltaIndex(["0 days", "1 days"], freq=None)
 226         tm.assert_index_equal(idx_diff, expected)
 227         tm.assert_attr_equal("freq", idx_diff, expected)
