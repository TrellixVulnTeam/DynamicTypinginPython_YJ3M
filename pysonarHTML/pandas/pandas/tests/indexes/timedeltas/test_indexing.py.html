 244     def test_take_fill_value(self):
 245         # GH 12631
 246         idx = TimedeltaIndex(["1 days", "2 days", "3 days"], name="xxx")
 247         result = idx.take(np.array([1, 0, -1]))
 248         expected = TimedeltaIndex(["2 days", "1 days", "3 days"], name="xxx")
 249         tm.assert_index_equal(result, expected)

 251         # fill_value
 252         result = idx.take(np.array([1, 0, -1]), fill_value=True)

 254         tm.assert_index_equal(result, expected)

 257         result = idx.take(np.array([1, 0, -1]), allow_fill=False, fill_value=True)
 258         expected = TimedeltaIndex(["2 days", "1 days", "3 days"], name="xxx")
 259         tm.assert_index_equal(result, expected)
 260 
 261         msg = (
 262             "When allow_fill=True and fill_value is not None, "
 263             "all indices must be >= -1"
 264         )
 265         with pytest.raises(ValueError, match=msg):
 266             idx.take(np.array([1, 0, -2]), fill_value=True)
 267         with pytest.raises(ValueError, match=msg):
 268             idx.take(np.array([1, 0, -5]), fill_value=True)
 269 
 270         msg = "index -5 is out of bounds for (axis 0 with )?size 3"
