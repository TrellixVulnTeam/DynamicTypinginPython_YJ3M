  47         cases = {
  48             (0, 1, 2): expected_0_2,
  49             (7, 8, 9): expected_7_9,

  51         }
  52         for n, expected in cases.items():

  55             assert result.name == expected.name
  56             assert result.freq == expected.freq
  57 
  58             result = idx.delete(slice(n[0], n[-1] + 1))
