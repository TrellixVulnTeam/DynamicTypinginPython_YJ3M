  78     def test_float64_ns_rounded(self):
  79         # GH#23539 without specifying a unit, floats are regarded as nanos,
  80         #  and fractional portions are truncated
  81         tdi = TimedeltaIndex([2.3, 9.7])
  82         expected = TimedeltaIndex([2, 9])
  83         tm.assert_index_equal(tdi, expected)
  84 
  85         # integral floats are non-lossy
  86         tdi = TimedeltaIndex([2.0, 9.0])
  87         expected = TimedeltaIndex([2, 9])
  88         tm.assert_index_equal(tdi, expected)
  89 
  90         # NaNs get converted to NaT
  91         tdi = TimedeltaIndex([2.0, np.nan])
  92         expected = TimedeltaIndex([pd.Timedelta(nanoseconds=2), pd.NaT])
  93         tm.assert_index_equal(tdi, expected)

 162     def test_constructor_coverage(self):
 163         rng = timedelta_range("1 days", periods=10.5)
 164         exp = timedelta_range("1 days", periods=10)
 165         tm.assert_index_equal(rng, exp)
 166 
 167         msg = "periods must be a number, got foo"
 168         with pytest.raises(TypeError, match=msg):
 169             timedelta_range(start="1 days", periods="foo", freq="D")
 170 
 171         msg = (
 172             r"TimedeltaIndex\(\) must be called with a collection of some kind, "
 173             "'1 days' was passed"
 174         )
 175         with pytest.raises(TypeError, match=msg):
 176             TimedeltaIndex("1 days")
 177 
 178         # generator expression
 179         gen = (timedelta(i) for i in range(10))
 180         result = TimedeltaIndex(gen)
 181         expected = TimedeltaIndex([timedelta(i) for i in range(10)])
 182         tm.assert_index_equal(result, expected)
 183 
 184         # NumPy string array
 185         strings = np.array(["1 days", "2 days", "3 days"])
 186         result = TimedeltaIndex(strings)
 187         expected = to_timedelta([1, 2, 3], unit="d")
 188         tm.assert_index_equal(result, expected)
 189 
 190         from_ints = TimedeltaIndex(expected.asi8)
 191         tm.assert_index_equal(from_ints, expected)
 192 
 193         # non-conforming freq
 194         msg = (
 195             "Inferred frequency None from passed values does not conform to "
 196             "passed frequency D"
 197         )
 198         with pytest.raises(ValueError, match=msg):
 199             TimedeltaIndex(["1 days", "2 days", "4 days"], freq="D")
 200 
 201         msg = (
 202             "Of the four parameters: start, end, periods, and freq, exactly "
 203             "three must be specified"
 204         )
 205         with pytest.raises(ValueError, match=msg):
 206             timedelta_range(periods=10, freq="D")
