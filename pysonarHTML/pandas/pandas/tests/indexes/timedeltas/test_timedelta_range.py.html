  13         expected = to_timedelta(np.arange(5), unit="D")
  14         result = timedelta_range("0 days", periods=5, freq="D")

  17         expected = to_timedelta(np.arange(11), unit="D")

  33     @pytest.mark.parametrize(

  36     def test_linspace_behavior(self, periods, freq):
  37         # GH 20976
  38         result = timedelta_range(start="0 days", end="4 days", periods=periods)

  40         tm.assert_index_equal(result, expected)

  65     @pytest.mark.parametrize(

  75     )
  76     def test_timedelta_range_freq_divide_end(self, start, end, freq, expected_periods):
  77         # GH 33498 only the cases where `(end % freq) == 0` used to fail
  78         res = timedelta_range(start=start, end=end, freq=freq)
  79         assert Timedelta(start) == res[0]
  80         assert Timedelta(end) >= res[-1]
  81         assert len(res) == expected_periods
