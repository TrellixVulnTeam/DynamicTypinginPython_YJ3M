 143     def test_period_astype_to_timestamp(self):
 144         pi = PeriodIndex(["2011-01", "2011-02", "2011-03"], freq="M")
 145 
 146         exp = DatetimeIndex(["2011-01-01", "2011-02-01", "2011-03-01"], freq="MS")
 147         res = pi.astype("datetime64[ns]")
 148         tm.assert_index_equal(res, exp)
 149         assert res.freq == exp.freq
 150 
 151         exp = DatetimeIndex(["2011-01-31", "2011-02-28", "2011-03-31"])
 152         exp = exp + Timedelta(1, "D") - Timedelta(1, "ns")
 153         res = pi.astype("datetime64[ns]", how="end")
 154         tm.assert_index_equal(res, exp)
 155         assert res.freq == exp.freq
 156 
 157         exp = DatetimeIndex(["2011-01-01", "2011-02-01", "2011-03-01"], tz="US/Eastern")
 158         res = pi.astype("datetime64[ns, US/Eastern]")
 159         tm.assert_index_equal(res, exp)
 160         assert res.freq == exp.freq
 161 
 162         exp = DatetimeIndex(["2011-01-31", "2011-02-28", "2011-03-31"], tz="US/Eastern")
 163         exp = exp + Timedelta(1, "D") - Timedelta(1, "ns")
 164         res = pi.astype("datetime64[ns, US/Eastern]", how="end")
 165         tm.assert_index_equal(res, exp)
 166         assert res.freq == exp.freq
