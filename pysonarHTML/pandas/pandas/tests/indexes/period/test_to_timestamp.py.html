  56     def test_to_timestamp_quarterly_bug(self):
  57         years = np.arange(1960, 2000).repeat(4)
  58         quarters = np.tile(list(range(1, 5)), 40)
  59 
  60         pindex = PeriodIndex(year=years, quarter=quarters)
  61 
  62         stamps = pindex.to_timestamp("D", "end")
  63         expected = DatetimeIndex([x.to_timestamp("D", "end") for x in pindex])
  64         tm.assert_index_equal(stamps, expected)

  67     def test_to_timestamp_pi_mult(self):
  68         idx = PeriodIndex(["2011-01", "NaT", "2011-02"], freq="2M", name="idx")
  69 
  70         result = idx.to_timestamp()
  71         expected = DatetimeIndex(["2011-01-01", "NaT", "2011-02-01"], name="idx")
  72         tm.assert_index_equal(result, expected)
  73 
  74         result = idx.to_timestamp(how="E")
  75         expected = DatetimeIndex(["2011-02-28", "NaT", "2011-03-31"], name="idx")
  76         expected = expected + Timedelta(1, "D") - Timedelta(1, "ns")
  77         tm.assert_index_equal(result, expected)

  79     def test_to_timestamp_pi_combined(self):

  81 
  82         result = idx.to_timestamp()
  83         expected = DatetimeIndex(["2011-01-01 00:00", "2011-01-02 01:00"], name="idx")
  84         tm.assert_index_equal(result, expected)
  85 
  86         result = idx.to_timestamp(how="E")
  87         expected = DatetimeIndex(
  88             ["2011-01-02 00:59:59", "2011-01-03 01:59:59"], name="idx"

  90         expected = expected + Timedelta(1, "s") - Timedelta(1, "ns")
