  38     def test_construction_from_period(self):
  39         # upsampling
  40         start, end = Period("2017Q1", freq="Q"), Period("2018Q1", freq="Q")
  41         expected = date_range(
  42             start="2017-03-31", end="2018-03-31", freq="M", name="foo"
  43         ).to_period()
  44         result = period_range(start=start, end=end, freq="M", name="foo")
  45         tm.assert_index_equal(result, expected)
  46 
  47         # downsampling
  48         start, end = Period("2017-1", freq="M"), Period("2019-12", freq="M")
  49         expected = date_range(
  50             start="2017-01-31", end="2019-12-31", freq="Q", name="foo"
  51         ).to_period()
  52         result = period_range(start=start, end=end, freq="Q", name="foo")
  53         tm.assert_index_equal(result, expected)
  54 
  55         # test for issue # 21793

  74     def test_errors(self):
  75         # not enough params
  76         msg = (
  77             "Of the three parameters: start, end, and periods, "
  78             "exactly two must be specified"
  79         )
  80         with pytest.raises(ValueError, match=msg):
  81             period_range(start="2017Q1")
  82 
  83         with pytest.raises(ValueError, match=msg):
  84             period_range(end="2017Q1")
  85 
  86         with pytest.raises(ValueError, match=msg):
  87             period_range(periods=5)
  88 
  89         with pytest.raises(ValueError, match=msg):
  90             period_range()
  91 
  92         # too many params
  93         with pytest.raises(ValueError, match=msg):
  94             period_range(start="2017Q1", end="2018Q1", periods=8, freq="Q")
  95 
  96         # start/end NaT
  97         msg = "start and end must not be NaT"
  98         with pytest.raises(ValueError, match=msg):
  99             period_range(start=NaT, end="2018Q1")
 100 
 101         with pytest.raises(ValueError, match=msg):
 102             period_range(start="2017Q1", end=NaT)
 103 
 104         # invalid periods param
 105         msg = "periods must be a number, got foo"
 106         with pytest.raises(TypeError, match=msg):
 107             period_range(start="2017Q1", periods="foo")
