 123     def test_order_compat(self):
 124         def _check_freq(index, expected_index):
 125             if isinstance(index, PeriodIndex):
 126                 assert index.freq == expected_index.freq
 127 
 128         pidx = PeriodIndex(["2011", "2012", "2013"], name="pidx", freq="A")
 129         # for compatibility check
 130         iidx = Index([2011, 2012, 2013], name="idx")
 131         for idx in [pidx, iidx]:
 132             ordered = idx.sort_values()
 133             tm.assert_index_equal(ordered, idx)
 134             _check_freq(ordered, idx)
 135 
 136             ordered = idx.sort_values(ascending=False)
 137             tm.assert_index_equal(ordered, idx[::-1])
 138             _check_freq(ordered, idx[::-1])
 139 
 140             ordered, indexer = idx.sort_values(return_indexer=True)
 141             tm.assert_index_equal(ordered, idx)
 142             tm.assert_numpy_array_equal(indexer, np.array([0, 1, 2]), check_dtype=False)
 143             _check_freq(ordered, idx)
 144 
 145             ordered, indexer = idx.sort_values(return_indexer=True, ascending=False)
 146             tm.assert_index_equal(ordered, idx[::-1])
 147             tm.assert_numpy_array_equal(indexer, np.array([2, 1, 0]), check_dtype=False)
 148             _check_freq(ordered, idx[::-1])
 149 
 150         pidx = PeriodIndex(
 151             ["2011", "2013", "2015", "2012", "2011"], name="pidx", freq="A"
 152         )
 153         pexpected = PeriodIndex(
 154             ["2011", "2011", "2012", "2013", "2015"], name="pidx", freq="A"
 155         )
 156         # for compatibility check
 157         iidx = Index([2011, 2013, 2015, 2012, 2011], name="idx")
 158         iexpected = Index([2011, 2011, 2012, 2013, 2015], name="idx")
 159         for idx, expected in [(pidx, pexpected), (iidx, iexpected)]:
 160             ordered = idx.sort_values()
 161             tm.assert_index_equal(ordered, expected)
 162             _check_freq(ordered, idx)
 163 
 164             ordered = idx.sort_values(ascending=False)
 165             tm.assert_index_equal(ordered, expected[::-1])
 166             _check_freq(ordered, idx)
 167 
 168             ordered, indexer = idx.sort_values(return_indexer=True)
 169             tm.assert_index_equal(ordered, expected)
 170 
 171             exp = np.array([0, 4, 3, 1, 2])
 172             tm.assert_numpy_array_equal(indexer, exp, check_dtype=False)
 173             _check_freq(ordered, idx)
 174 
 175             ordered, indexer = idx.sort_values(return_indexer=True, ascending=False)
 176             tm.assert_index_equal(ordered, expected[::-1])
 177 
 178             exp = np.array([2, 1, 3, 4, 0])
 179             tm.assert_numpy_array_equal(indexer, exp, check_dtype=False)
 180             _check_freq(ordered, idx)
 181 
 182         pidx = PeriodIndex(["2011", "2013", "NaT", "2011"], name="pidx", freq="D")
 183 
 184         result = pidx.sort_values()

 186         tm.assert_index_equal(result, expected)

 293     @pytest.mark.parametrize("freq", ["D", "M"])
 294     def test_equals(self, freq):
 295         # GH#13107
 296         idx = pd.PeriodIndex(["2011-01-01", "2011-01-02", "NaT"], freq=freq)
 297         assert idx.equals(idx)
 298         assert idx.equals(idx.copy())
 299         assert idx.equals(idx.astype(object))
 300         assert idx.astype(object).equals(idx)
 301         assert idx.astype(object).equals(idx.astype(object))
 302         assert not idx.equals(list(idx))
 303         assert not idx.equals(pd.Series(idx))
 304 
 305         idx2 = pd.PeriodIndex(["2011-01-01", "2011-01-02", "NaT"], freq="H")
 306         assert not idx.equals(idx2)
 307         assert not idx.equals(idx2.copy())
 308         assert not idx.equals(idx2.astype(object))
 309         assert not idx.astype(object).equals(idx2)
 310         assert not idx.equals(list(idx2))
 311         assert not idx.equals(pd.Series(idx2))
 312 
 313         # same internal, different tz
 314         idx3 = pd.PeriodIndex._simple_new(
 315             idx._values._simple_new(idx._values.asi8, freq="H")
 316         )
 317         tm.assert_numpy_array_equal(idx.asi8, idx3.asi8)
 318         assert not idx.equals(idx3)
 319         assert not idx.equals(idx3.copy())
 320         assert not idx.equals(idx3.astype(object))
 321         assert not idx.astype(object).equals(idx3)
 322         assert not idx.equals(list(idx3))
