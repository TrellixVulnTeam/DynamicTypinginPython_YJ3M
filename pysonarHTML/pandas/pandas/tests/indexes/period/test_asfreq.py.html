  98     def test_asfreq_combined_pi(self):
  99         pi = PeriodIndex(["2001-01-01 00:00", "2001-01-02 02:00", "NaT"], freq="H")
 100         exp = PeriodIndex(["2001-01-01 00:00", "2001-01-02 02:00", "NaT"], freq="25H")
 101         for freq, how in zip(["1D1H", "1H1D"], ["S", "E"]):
 102             result = pi.asfreq(freq, how=how)
 103             tm.assert_index_equal(result, exp)
 104             assert result.freq == exp.freq
 105 
 106         for freq in ["1D1H", "1H1D"]:
 107             pi = PeriodIndex(["2001-01-01 00:00", "2001-01-02 02:00", "NaT"], freq=freq)
 108             result = pi.asfreq("H")
 109             exp = PeriodIndex(["2001-01-02 00:00", "2001-01-03 02:00", "NaT"], freq="H")
 110             tm.assert_index_equal(result, exp)
 111             assert result.freq == exp.freq
 112 
 113             pi = PeriodIndex(["2001-01-01 00:00", "2001-01-02 02:00", "NaT"], freq=freq)
 114             result = pi.asfreq("H", how="S")
 115             exp = PeriodIndex(["2001-01-01 00:00", "2001-01-02 02:00", "NaT"], freq="H")
 116             tm.assert_index_equal(result, exp)

 119     def test_astype_asfreq(self):
 120         pi1 = PeriodIndex(["2011-01-01", "2011-02-01", "2011-03-01"], freq="D")
 121         exp = PeriodIndex(["2011-01", "2011-02", "2011-03"], freq="M")
 122         tm.assert_index_equal(pi1.asfreq("M"), exp)
 123         tm.assert_index_equal(pi1.astype("period[M]"), exp)
 124 
 125         exp = PeriodIndex(["2011-01", "2011-02", "2011-03"], freq="3M")
 126         tm.assert_index_equal(pi1.asfreq("3M"), exp)
 127         tm.assert_index_equal(pi1.astype("period[3M]"), exp)
