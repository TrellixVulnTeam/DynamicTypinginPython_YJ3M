  93     def test_partial_slice_doesnt_require_monotonicity(self):
  94         # See also: DatetimeIndex test ofm the same name
  95         dti = date_range("2014-01-01", periods=30, freq="30D")
  96         pi = dti.to_period("D")
  97 
  98         ser_montonic = Series(np.arange(30), index=pi)
  99 
 100         shuffler = list(range(0, 30, 2)) + list(range(1, 31, 2))
 101         ser = ser_montonic[shuffler]
 102         nidx = ser.index
 103 
 104         # Manually identified locations of year==2014
 105         indexer_2014 = np.array(
 106             [0, 1, 2, 3, 4, 5, 6, 15, 16, 17, 18, 19, 20], dtype=np.intp
 107         )
 108         assert (nidx[indexer_2014].year == 2014).all()
 109         assert not (nidx[~indexer_2014].year == 2014).any()
 110 
 111         result = nidx.get_loc("2014")
 112         tm.assert_numpy_array_equal(result, indexer_2014)
 113 
 114         expected = ser[indexer_2014]
 115 
 116         with tm.assert_produces_warning(FutureWarning):
 117             result = nidx.get_value(ser, "2014")
 118         tm.assert_series_equal(result, expected)
 119 
 120         result = ser.loc["2014"]
 121         tm.assert_series_equal(result, expected)
 122 
 123         result = ser["2014"]
 124         tm.assert_series_equal(result, expected)
 125 
 126         # Manually identified locations where ser.index is within Mat 2015
 127         indexer_may2015 = np.array([23], dtype=np.intp)
 128         assert nidx[23].year == 2015 and nidx[23].month == 5
 129 
 130         result = nidx.get_loc("May 2015")
 131         tm.assert_numpy_array_equal(result, indexer_may2015)
 132 
 133         expected = ser[indexer_may2015]
 134 
 135         with tm.assert_produces_warning(FutureWarning):
 136             result = nidx.get_value(ser, "May 2015")
 137         tm.assert_series_equal(result, expected)
 138 
 139         result = ser.loc["May 2015"]
 140         tm.assert_series_equal(result, expected)
 141 
 142         result = ser["May 2015"]
 143         tm.assert_series_equal(result, expected)
