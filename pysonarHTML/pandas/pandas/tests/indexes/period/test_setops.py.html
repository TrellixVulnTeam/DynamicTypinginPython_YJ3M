 136     def test_union_misc(self, sort):
 137         index = period_range("1/1/2000", "1/20/2000", freq="D")
 138 
 139         result = index[:-5].union(index[10:], sort=sort)
 140         tm.assert_index_equal(result, index)
 141 
 142         # not in order
 143         result = _permute(index[:-5]).union(_permute(index[10:]), sort=sort)
 144         if sort is None:
 145             tm.assert_index_equal(result, index)
 146         assert tm.equalContents(result, index)
 147 
 148         # raise if different frequencies
 149         index = period_range("1/1/2000", "1/20/2000", freq="D")
 150         index2 = period_range("1/1/2000", "1/20/2000", freq="W-WED")
 151         msg = r"Input has different freq=W-WED from PeriodIndex\(freq=D\)"
 152         with pytest.raises(IncompatibleFrequency, match=msg):
 153             index.union(index2, sort=sort)

 167     def test_intersection(self, sort):
 168         index = period_range("1/1/2000", "1/20/2000", freq="D")
 169 
 170         result = index[:-5].intersection(index[10:], sort=sort)
 171         tm.assert_index_equal(result, index[10:-5])
 172 
 173         # not in order
 174         left = _permute(index[:-5])
 175         right = _permute(index[10:])
 176         result = left.intersection(right, sort=sort)
 177         if sort is None:
 178             tm.assert_index_equal(result, index[10:-5])
 179         assert tm.equalContents(result, index[10:-5])
 180 
 181         # raise if different frequencies
 182         index = period_range("1/1/2000", "1/20/2000", freq="D")
 183         index2 = period_range("1/1/2000", "1/20/2000", freq="W-WED")
 184         msg = r"Input has different freq=W-WED from PeriodIndex\(freq=D\)"
 185         with pytest.raises(IncompatibleFrequency, match=msg):
 186             index.intersection(index2, sort=sort)
 187 
 188         index3 = period_range("1/1/2000", "1/20/2000", freq="2D")
 189         msg = r"Input has different freq=2D from PeriodIndex\(freq=D\)"
 190         with pytest.raises(IncompatibleFrequency, match=msg):
 191             index.intersection(index3, sort=sort)

 193     def test_intersection_cases(self, sort):
 194         base = period_range("6/1/2000", "6/30/2000", freq="D", name="idx")
 195 
 196         # if target has the same name, it is preserved
 197         rng2 = period_range("5/15/2000", "6/20/2000", freq="D", name="idx")
 198         expected2 = period_range("6/1/2000", "6/20/2000", freq="D", name="idx")
 199 
 200         # if target name is different, it will be reset
 201         rng3 = period_range("5/15/2000", "6/20/2000", freq="D", name="other")
 202         expected3 = period_range("6/1/2000", "6/20/2000", freq="D", name=None)
 203 
 204         rng4 = period_range("7/1/2000", "7/31/2000", freq="D", name="idx")
 205         expected4 = PeriodIndex([], name="idx", freq="D")
 206 
 207         for (rng, expected) in [
 208             (rng2, expected2),
 209             (rng3, expected3),
 210             (rng4, expected4),
 211         ]:
 212             result = base.intersection(rng, sort=sort)
 213             tm.assert_index_equal(result, expected)
 214             assert result.name == expected.name
 215             assert result.freq == expected.freq
 216 
 217         # non-monotonic
 218         base = PeriodIndex(
 219             ["2011-01-05", "2011-01-04", "2011-01-02", "2011-01-03"],
 220             freq="D",
 221             name="idx",
 222         )
 223 
 224         rng2 = PeriodIndex(
 225             ["2011-01-04", "2011-01-02", "2011-02-02", "2011-02-03"],
 226             freq="D",
 227             name="idx",
 228         )
 229         expected2 = PeriodIndex(["2011-01-04", "2011-01-02"], freq="D", name="idx")
 230 
 231         rng3 = PeriodIndex(
 232             ["2011-01-04", "2011-01-02", "2011-02-02", "2011-02-03"],
 233             freq="D",
 234             name="other",
 235         )
 236         expected3 = PeriodIndex(["2011-01-04", "2011-01-02"], freq="D", name=None)
 237 
 238         rng4 = period_range("7/1/2000", "7/31/2000", freq="D", name="idx")
 239         expected4 = PeriodIndex([], freq="D", name="idx")
 240 
 241         for (rng, expected) in [
 242             (rng2, expected2),
 243             (rng3, expected3),
 244             (rng4, expected4),
 245         ]:
 246             result = base.intersection(rng, sort=sort)
 247             if sort is None:
 248                 expected = expected.sort_values()
 249             tm.assert_index_equal(result, expected)
 250             assert result.name == expected.name
 251             assert result.freq == "D"
 252 
 253         # empty same freq
 254         rng = date_range("6/1/2000", "6/15/2000", freq="T")
 255         result = rng[0:0].intersection(rng)
 256         assert len(result) == 0
 257 
 258         result = rng.intersection(rng[0:0])
 259         assert len(result) == 0

 325     def test_difference_freq(self, sort):
 326         # GH14323: difference of Period MUST preserve frequency
 327         # but the ability to union results must be preserved
 328 
 329         index = period_range("20160920", "20160925", freq="D")
 330 
 331         other = period_range("20160921", "20160924", freq="D")
 332         expected = PeriodIndex(["20160920", "20160925"], freq="D")
 333         idx_diff = index.difference(other, sort)
 334         tm.assert_index_equal(idx_diff, expected)
 335         tm.assert_attr_equal("freq", idx_diff, expected)
 336 
 337         other = period_range("20160922", "20160925", freq="D")
 338         idx_diff = index.difference(other, sort)
 339         expected = PeriodIndex(["20160920", "20160921"], freq="D")
 340         tm.assert_index_equal(idx_diff, expected)
 341         tm.assert_attr_equal("freq", idx_diff, expected)
