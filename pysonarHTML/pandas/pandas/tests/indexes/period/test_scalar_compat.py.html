   3 from pandas import Timedelta, date_range, period_range

   7 class TestPeriodIndexOps:
   8     def test_start_time(self):
   9         # GH#17157
  10         index = period_range(freq="M", start="2016-01-01", end="2016-05-31")
  11         expected_index = date_range("2016-01-01", end="2016-05-31", freq="MS")

  14     def test_end_time(self):
  15         # GH#17157
  16         index = period_range(freq="M", start="2016-01-01", end="2016-05-31")
  17         expected_index = date_range("2016-01-01", end="2016-05-31", freq="M")
  18         expected_index += Timedelta(1, "D") - Timedelta(1, "ns")
  19         tm.assert_index_equal(index.end_time, expected_index)

  21     def test_end_time_business_friday(self):
