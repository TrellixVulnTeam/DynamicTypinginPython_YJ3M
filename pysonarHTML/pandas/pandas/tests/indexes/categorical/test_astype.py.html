   9     def test_astype(self):
  10         ci = CategoricalIndex(list("aabbca"), categories=list("cab"), ordered=False)
  11 
  12         result = ci.astype(object)
  13         tm.assert_index_equal(result, Index(np.array(ci)))
  14 
  15         # this IS equal, but not the same class
  16         assert result.equals(ci)
  17         assert isinstance(result, Index)
  18         assert not isinstance(result, CategoricalIndex)
  19 
  20         # interval
  21         ii = IntervalIndex.from_arrays(left=[-0.001, 2.0], right=[2, 4], closed="right")
  22 
  23         ci = CategoricalIndex(
  24             Categorical.from_codes([0, 1, -1], categories=ii, ordered=True)
  25         )
  26 
  27         result = ci.astype("interval")
  28         expected = ii.take([0, 1, -1])
  29         tm.assert_index_equal(result, expected)
  30 
  31         result = IntervalIndex(result.values)
  32         tm.assert_index_equal(result, expected)
