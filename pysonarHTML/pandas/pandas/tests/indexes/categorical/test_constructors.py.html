  83     def test_construction_with_dtype(self):
  84 
  85         # specify dtype
  86         ci = CategoricalIndex(list("aabbca"), categories=list("abc"), ordered=False)
  87 
  88         result = Index(np.array(ci), dtype="category")
  89         tm.assert_index_equal(result, ci, exact=True)
  90 
  91         result = Index(np.array(ci).tolist(), dtype="category")
  92         tm.assert_index_equal(result, ci, exact=True)
  93 
  94         # these are generally only equal when the categories are reordered
  95         ci = CategoricalIndex(list("aabbca"), categories=list("cab"), ordered=False)
  96 
  97         result = Index(np.array(ci), dtype="category").reorder_categories(ci.categories)
  98         tm.assert_index_equal(result, ci, exact=True)
  99 
 100         # make sure indexes are handled
 101         expected = CategoricalIndex([0, 1, 2], categories=[0, 1, 2], ordered=True)
 102         idx = Index(range(3))
 103         result = CategoricalIndex(idx, categories=idx, ordered=True)
 104         tm.assert_index_equal(result, expected, exact=True)

 112     def test_construction_with_categorical_dtype(self):
 113         # construction with CategoricalDtype
 114         # GH#18109
 115         data, cats, ordered = "a a b b".split(), "c b a".split(), True
 116         dtype = CategoricalDtype(categories=cats, ordered=ordered)
 117 
 118         result = CategoricalIndex(data, dtype=dtype)
 119         expected = CategoricalIndex(data, categories=cats, ordered=ordered)
 120         tm.assert_index_equal(result, expected, exact=True)
 121 
 122         # GH#19032
 123         result = Index(data, dtype=dtype)
 124         tm.assert_index_equal(result, expected, exact=True)
 125 
 126         # error when combining categories/ordered and dtype kwargs
 127         msg = "Cannot specify `categories` or `ordered` together with `dtype`."
 128         with pytest.raises(ValueError, match=msg):
 129             CategoricalIndex(data, categories=cats, dtype=dtype)
 130 
 131         with pytest.raises(ValueError, match=msg):
 132             Index(data, categories=cats, dtype=dtype)
 133 
 134         with pytest.raises(ValueError, match=msg):
 135             CategoricalIndex(data, ordered=ordered, dtype=dtype)
 136 
 137         with pytest.raises(ValueError, match=msg):
 138             Index(data, ordered=ordered, dtype=dtype)
