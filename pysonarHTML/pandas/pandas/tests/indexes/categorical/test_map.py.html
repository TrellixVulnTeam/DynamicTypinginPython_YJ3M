  10     @pytest.mark.parametrize(
  11         "data, categories",
  12         [
  13             (list("abcbca"), list("cab")),
  14             (pd.interval_range(0, 3).repeat(3), pd.interval_range(0, 3)),
  15         ],
  16         ids=["string", "interval"],
  17     )
  18     def test_map_str(self, data, categories, ordered):
  19         # GH 31202 - override base class since we want to maintain categorical/ordered
  20         index = CategoricalIndex(data, categories=categories, ordered=ordered)
  21         result = index.map(str)
  22         expected = CategoricalIndex(
  23             map(str, data), categories=map(str, categories), ordered=ordered
  24         )
  25         tm.assert_index_equal(result, expected)

  27     def test_map(self):
  28         ci = pd.CategoricalIndex(list("ABABC"), categories=list("CBA"), ordered=True)
  29         result = ci.map(lambda x: x.lower())
  30         exp = pd.CategoricalIndex(list("ababc"), categories=list("cba"), ordered=True)
  31         tm.assert_index_equal(result, exp)
  32 
  33         ci = pd.CategoricalIndex(
  34             list("ABABC"), categories=list("BAC"), ordered=False, name="XXX"
  35         )
  36         result = ci.map(lambda x: x.lower())
  37         exp = pd.CategoricalIndex(
  38             list("ababc"), categories=list("bac"), ordered=False, name="XXX"
  39         )
  40         tm.assert_index_equal(result, exp)
  41 
  42         # GH 12766: Return an index not an array
  43         tm.assert_index_equal(
  44             ci.map(lambda x: 1), Index(np.array([1] * 5, dtype=np.int64), name="XXX")
  45         )
  46 
  47         # change categories dtype
  48         ci = pd.CategoricalIndex(list("ABABC"), categories=list("BAC"), ordered=False)
  49 
  50         def f(x):
  51             return {"A": 10, "B": 20, "C": 30}.get(x)
  52 
  53         result = ci.map(f)
  54         exp = pd.CategoricalIndex(
  55             [10, 20, 10, 20, 30], categories=[20, 10, 30], ordered=False

  58 
  59         result = ci.map(pd.Series([10, 20, 30], index=["A", "B", "C"]))

  65     def test_map_with_categorical_series(self):
  66         # GH 12756
  67         a = pd.Index([1, 2, 3, 4])
  68         b = pd.Series(["even", "odd", "even", "odd"], dtype="category")
  69         c = pd.Series(["even", "odd", "even", "odd"])
  70 
  71         exp = CategoricalIndex(["odd", "even", "odd", np.nan])
  72         tm.assert_index_equal(a.map(b), exp)
  73         exp = pd.Index(["odd", "even", "odd", np.nan])
  74         tm.assert_index_equal(a.map(c), exp)
