 100

 104         # append cats with the same categories

 106         tm.assert_index_equal(result, ci, exact=True)

 123         # with objects

 125         expected = CategoricalIndex(list("aabbcaca"), categories=categories)
 126         tm.assert_index_equal(result, expected, exact=True)
 127 
 128         # invalid objects
 129         msg = "cannot append a non-category item to a CategoricalIndex"
 130         with pytest.raises(TypeError, match=msg):
 131             ci.append(Index(["a", "d"]))

 134         result = Index(["c", "a"]).append(ci)

 374     def test_equals_categorical(self):
 375         ci1 = CategoricalIndex(["a", "b"], categories=["a", "b"], ordered=True)
 376         ci2 = CategoricalIndex(["a", "b"], categories=["a", "b", "c"], ordered=True)
 377 
 378         assert ci1.equals(ci1)
 379         assert not ci1.equals(ci2)
 380         assert ci1.equals(ci1.astype(object))
 381         assert ci1.astype(object).equals(ci1)
 382 
 383         assert (ci1 == ci1).all()
 384         assert not (ci1 != ci1).all()
 385         assert not (ci1 > ci1).all()
 386         assert not (ci1 < ci1).all()
 387         assert (ci1 <= ci1).all()
 388         assert (ci1 >= ci1).all()
 389 
 390         assert not (ci1 == 1).all()
 391         assert (ci1 == Index(["a", "b"])).all()
 392         assert (ci1 == ci1.values).all()
 393 
 394         # invalid comparisons
 395         with pytest.raises(ValueError, match="Lengths must match"):
 396             ci1 == Index(["a", "b", "c"])
 397 
 398         msg = (
 399             "categorical index comparisons must have the same categories "
 400             "and ordered attributes"
 401             "|"
 402             "Categoricals can only be compared if 'categories' are the same. "
 403             "Categories are different lengths"
 404             "|"
 405             "Categoricals can only be compared if 'ordered' is the same"
 406         )
 407         with pytest.raises(TypeError, match=msg):
 408             ci1 == ci2
 409         with pytest.raises(TypeError, match=msg):
 410             ci1 == Categorical(ci1.values, ordered=False)
 411         with pytest.raises(TypeError, match=msg):
 412             ci1 == Categorical(ci1.values, categories=list("abc"))
 413 
 414         # tests
 415         # make sure that we are testing for category inclusion properly
 416         ci = CategoricalIndex(list("aabca"), categories=["c", "a", "b"])
 417         assert not ci.equals(list("aabca"))
 418         # Same categories, but different order
 419         # Unordered
 420         assert ci.equals(CategoricalIndex(list("aabca")))
 421         # Ordered
 422         assert not ci.equals(CategoricalIndex(list("aabca"), ordered=True))
 423         assert ci.equals(ci.copy())
 424 
 425         ci = CategoricalIndex(list("aabca") + [np.nan], categories=["c", "a", "b"])
 426         assert not ci.equals(list("aabca"))
 427         assert not ci.equals(CategoricalIndex(list("aabca")))
 428         assert ci.equals(ci.copy())
 429 
 430         ci = CategoricalIndex(list("aabca") + [np.nan], categories=["c", "a", "b"])
 431         assert not ci.equals(list("aabca") + [np.nan])
 432         assert ci.equals(CategoricalIndex(list("aabca") + [np.nan]))
 433         assert not ci.equals(CategoricalIndex(list("aabca") + [np.nan], ordered=True))
 434         assert ci.equals(ci.copy())

 451     @pytest.mark.parametrize(
 452         "dtype, engine_type",
 453         [
 454             (np.int8, libindex.Int8Engine),
 455             (np.int16, libindex.Int16Engine),
 456             (np.int32, libindex.Int32Engine),
 457             (np.int64, libindex.Int64Engine),
 458         ],
 459     )
 460     def test_engine_type(self, dtype, engine_type):
 461         if dtype != np.int64:
 462             # num. of uniques required to push CategoricalIndex.codes to a
 463             # dtype (128 categories required for .codes dtype to be int16 etc.)
 464             num_uniques = {np.int8: 1, np.int16: 128, np.int32: 32768}[dtype]
 465             ci = pd.CategoricalIndex(range(num_uniques))
 466         else:
 467             # having 2**32 - 2**31 categories would be very memory-intensive,
 468             # so we cheat a bit with the dtype
 469             ci = pd.CategoricalIndex(range(32768))  # == 2**16 - 2**(16 - 1)
 470             ci.values._codes = ci.values._codes.astype("int64")
 471         assert np.issubdtype(ci.codes.dtype, dtype)
 472         assert isinstance(ci._engine, engine_type)
