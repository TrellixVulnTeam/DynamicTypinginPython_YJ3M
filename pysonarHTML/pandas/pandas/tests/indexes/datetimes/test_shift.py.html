 107     @pytest.mark.parametrize(
 108         "shift, result_time",
 109         [
 110             [0, "2014-11-14 00:00:00"],
 111             [-1, "2014-11-13 23:00:00"],
 112             [1, "2014-11-14 01:00:00"],

 114     )
 115     def test_dti_shift_near_midnight(self, shift, result_time):
 116         # GH 8616
 117         dt = datetime(2014, 11, 14, 0)
 118         dt_est = pytz.timezone("EST").localize(dt)
 119         s = Series(data=[1], index=[dt_est])
 120         result = s.shift(shift, freq="H")
 121         expected = Series(1, index=DatetimeIndex([result_time], tz="EST"))
 122         tm.assert_series_equal(result, expected)

 130     @pytest.mark.parametrize("freq", ["B", "C"])
 131     def test_shift_bday(self, freq):
 132         rng = date_range(START, END, freq=freq)
 133         shifted = rng.shift(5)
 134         assert shifted[0] == rng[5]
 135         assert shifted.freq == rng.freq
 136 
 137         shifted = rng.shift(-5)
 138         assert shifted[5] == rng[0]
 139         assert shifted.freq == rng.freq
 140 
 141         shifted = rng.shift(0)
 142         assert shifted[0] == rng[0]
 143         assert shifted.freq == rng.freq

 145     def test_shift_bmonth(self):
 146         rng = date_range(START, END, freq=pd.offsets.BMonthEnd())
 147         shifted = rng.shift(1, freq=pd.offsets.BDay())
 148         assert shifted[0] == rng[0] + pd.offsets.BDay()
 149

 153             assert shifted[0] == rng[0] + pd.offsets.CDay()
