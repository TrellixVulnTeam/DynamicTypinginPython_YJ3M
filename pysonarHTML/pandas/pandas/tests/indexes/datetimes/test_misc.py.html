 295     @pytest.mark.parametrize(
 296         "time_locale", [None] if tm.get_locales() is None else [None] + tm.get_locales()
 297     )
 298     def test_datetime_name_accessors(self, time_locale):
 299         # Test Monday -> Sunday and January -> December, in that sequence
 300         if time_locale is None:
 301             # If the time_locale is None, day-name and month_name should
 302             # return the english attributes
 303             expected_days = [
 304                 "Monday",
 305                 "Tuesday",
 306                 "Wednesday",
 307                 "Thursday",
 308                 "Friday",
 309                 "Saturday",
 310                 "Sunday",
 311             ]
 312             expected_months = [
 313                 "January",
 314                 "February",
 315                 "March",
 316                 "April",
 317                 "May",
 318                 "June",
 319                 "July",
 320                 "August",
 321                 "September",
 322                 "October",
 323                 "November",
 324                 "December",
 325             ]
 326         else:
 327             with tm.set_locale(time_locale, locale.LC_TIME):
 328                 expected_days = calendar.day_name[:]
 329                 expected_months = calendar.month_name[1:]
 330 
 331         # GH#11128
 332         dti = pd.date_range(freq="D", start=datetime(1998, 1, 1), periods=365)
 333         english_days = [
 334             "Monday",
 335             "Tuesday",
 336             "Wednesday",
 337             "Thursday",
 338             "Friday",
 339             "Saturday",
 340             "Sunday",
 341         ]
 342         for day, name, eng_name in zip(range(4, 11), expected_days, english_days):
 343             name = name.capitalize()
 344             assert dti.day_name(locale=time_locale)[day] == name
 345             ts = Timestamp(datetime(2016, 4, day))
 346             assert ts.day_name(locale=time_locale) == name
 347         dti = dti.append(DatetimeIndex([pd.NaT]))
 348         assert np.isnan(dti.day_name(locale=time_locale)[-1])
 349         ts = Timestamp(pd.NaT)
 350         assert np.isnan(ts.day_name(locale=time_locale))
 351 
 352         # GH#12805
 353         dti = pd.date_range(freq="M", start="2012", end="2013")
 354         result = dti.month_name(locale=time_locale)
 355         expected = Index([month.capitalize() for month in expected_months])
 356 
 357         # work around different normalization schemes
 358         # https://github.com/pandas-dev/pandas/issues/22342
 359         result = result.str.normalize("NFD")
 360         expected = expected.str.normalize("NFD")
 361 
 362         tm.assert_index_equal(result, expected)
 363 
 364         for date, expected in zip(dti, expected_months):
 365             result = date.month_name(locale=time_locale)
 366             expected = expected.capitalize()
 367 
 368             result = unicodedata.normalize("NFD", result)
 369             expected = unicodedata.normalize("NFD", result)
 370 
 371             assert result == expected
 372         dti = dti.append(DatetimeIndex([pd.NaT]))
 373         assert np.isnan(dti.month_name(locale=time_locale)[-1])
