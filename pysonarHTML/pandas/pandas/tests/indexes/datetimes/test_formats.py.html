  13 def test_to_native_types():
  14     index = pd.date_range(freq="1D", periods=3, start="2017-01-01")
  15 
  16     # First, with no arguments.
  17     expected = np.array(["2017-01-01", "2017-01-02", "2017-01-03"], dtype=object)
  18 
  19     result = index.to_native_types()
  20     tm.assert_numpy_array_equal(result, expected)
  21 
  22     # No NaN values, so na_rep has no effect
  23     result = index.to_native_types(na_rep="pandas")
  24     tm.assert_numpy_array_equal(result, expected)
  25 
  26     # Make sure slicing works
  27     expected = np.array(["2017-01-01", "2017-01-03"], dtype=object)
  28 
  29     result = index.to_native_types([0, 2])
  30     tm.assert_numpy_array_equal(result, expected)
  31 
  32     # Make sure date formatting works
  33     expected = np.array(["01-2017-01", "01-2017-02", "01-2017-03"], dtype=object)
  34 
  35     result = index.to_native_types(date_format="%m-%Y-%d")
  36     tm.assert_numpy_array_equal(result, expected)
  37 
  38     # NULL object handling should work
  39     index = DatetimeIndex(["2017-01-01", pd.NaT, "2017-01-03"])
  40     expected = np.array(["2017-01-01", "NaT", "2017-01-03"], dtype=object)
  41 
  42     result = index.to_native_types()
  43     tm.assert_numpy_array_equal(result, expected)
  44 
  45     expected = np.array(["2017-01-01", "pandas", "2017-01-03"], dtype=object)
  46 
  47     result = index.to_native_types(na_rep="pandas")
  48     tm.assert_numpy_array_equal(result, expected)
