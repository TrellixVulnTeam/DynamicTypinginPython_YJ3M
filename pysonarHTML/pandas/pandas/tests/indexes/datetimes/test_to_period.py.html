  23     def test_dti_to_period(self):
  24         dti = date_range(start="1/1/2005", end="12/1/2005", freq="M")
  25         pi1 = dti.to_period()
  26         pi2 = dti.to_period(freq="D")
  27         pi3 = dti.to_period(freq="3D")
  28 
  29         assert pi1[0] == Period("Jan 2005", freq="M")
  30         assert pi2[0] == Period("1/31/2005", freq="D")
  31         assert pi3[0] == Period("1/31/2005", freq="3D")
  32 
  33         assert pi1[-1] == Period("Nov 2005", freq="M")
  34         assert pi2[-1] == Period("11/30/2005", freq="D")
  35         assert pi3[-1], Period("11/30/2005", freq="3D")
  36 
  37         tm.assert_index_equal(pi1, period_range("1/1/2005", "11/1/2005", freq="M"))
  38         tm.assert_index_equal(
  39             pi2, period_range("1/1/2005", "11/1/2005", freq="M").asfreq("D")
  40         )
  41         tm.assert_index_equal(
  42             pi3, period_range("1/1/2005", "11/1/2005", freq="M").asfreq("3D")

  45     @pytest.mark.parametrize("month", MONTHS)
  46     def test_to_period_quarterly(self, month):
  47         # make sure we can make the round trip
  48         freq = f"Q-{month}"
  49         rng = period_range("1989Q3", "1991Q3", freq=freq)
  50         stamps = rng.to_timestamp()
  51         result = stamps.to_period(freq)
  52         tm.assert_index_equal(rng, result)

 102     def test_period_dt64_round_trip(self):
 103         dti = date_range("1/1/2000", "1/7/2002", freq="B")
 104         pi = dti.to_period()

 111     def test_to_period_millisecond(self):
 112         index = DatetimeIndex(
 113             [
 114                 Timestamp("2007-01-01 10:11:12.123456Z"),
 115                 Timestamp("2007-01-01 10:11:13.789123Z"),
 116             ]
 117         )
 118 
 119         with tm.assert_produces_warning(UserWarning):
 120             # warning that timezone info will be lost
 121             period = index.to_period(freq="L")
 122         assert 2 == len(period)
 123         assert period[0] == Period("2007-01-01 10:11:12.123Z", "L")
 124         assert period[1] == Period("2007-01-01 10:11:13.789Z", "L")

 172     def test_to_period_nofreq(self):
 173         idx = DatetimeIndex(["2000-01-01", "2000-01-02", "2000-01-04"])
 174         msg = "You must pass a freq argument as current index has none."
 175         with pytest.raises(ValueError, match=msg):
 176             idx.to_period()
 177 
 178         idx = DatetimeIndex(["2000-01-01", "2000-01-02", "2000-01-03"], freq="infer")
 179         assert idx.freqstr == "D"
 180         expected = PeriodIndex(["2000-01-01", "2000-01-02", "2000-01-03"], freq="D")
 181         tm.assert_index_equal(idx.to_period(), expected)
 182 
 183         # GH#7606
 184         idx = DatetimeIndex(["2000-01-01", "2000-01-02", "2000-01-03"])
 185         assert idx.freqstr is None
 186         tm.assert_index_equal(idx.to_period(), expected)
