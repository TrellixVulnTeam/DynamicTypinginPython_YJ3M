 104         rng = date_range(freq="D", start=datetime(2005, 1, 1), periods=500)
 105         s = Series(np.arange(len(rng)), index=rng)
 106 
 107         result = s["2005-05":"2006-02"]

 117         tm.assert_series_equal(result, expected)
 118 
 119         result = s["2005-1-1"]
 120         assert result == s.iloc[0]
 121 
 122         with pytest.raises(KeyError, match=r"^'2004-12-31'$"):
 123             s["2004-12-31"]

 181     def test_partial_slicing_dataframe(self):
 182         # GH14856
 183         # Test various combinations of string slicing resolution vs.
 184         # index resolution
 185         # - If string resolution is less precise than index resolution,
 186         # string is considered a slice
 187         # - If string resolution is equal to or more precise than index
 188         # resolution, string is considered an exact match
 189         formats = [
 190             "%Y",
 191             "%Y-%m",
 192             "%Y-%m-%d",
 193             "%Y-%m-%d %H",
 194             "%Y-%m-%d %H:%M",
 195             "%Y-%m-%d %H:%M:%S",
 196         ]
 197         resolutions = ["year", "month", "day", "hour", "minute", "second"]
 198         for rnum, resolution in enumerate(resolutions[2:], 2):
 199             # we check only 'day', 'hour', 'minute' and 'second'
 200             unit = Timedelta("1 " + resolution)
 201             middate = datetime(2012, 1, 1, 0, 0, 0)
 202             index = DatetimeIndex([middate - unit, middate, middate + unit])
 203             values = [1, 2, 3]
 204             df = DataFrame({"a": values}, index, dtype=np.int64)
 205             assert df.index.resolution == resolution
 206 
 207             # Timestamp with the same resolution as index
 208             # Should be exact match for Series (return scalar)
 209             # and raise KeyError for Frame
 210             for timestamp, expected in zip(index, values):
 211                 ts_string = timestamp.strftime(formats[rnum])
 212                 # make ts_string as precise as index
 213                 result = df["a"][ts_string]
 214                 assert isinstance(result, np.int64)
 215                 assert result == expected
 216                 msg = fr"^'{ts_string}'$"
 217                 with pytest.raises(KeyError, match=msg):
 218                     df[ts_string]
 219 
 220             # Timestamp with resolution less precise than index
 221             for fmt in formats[:rnum]:
 222                 for element, theslice in [[0, slice(None, 1)], [1, slice(1, None)]]:
 223                     ts_string = index[element].strftime(fmt)
 224 
 225                     # Series should return slice
 226                     result = df["a"][ts_string]
 227                     expected = df["a"][theslice]
 228                     tm.assert_series_equal(result, expected)
 229 
 230                     # Frame should return slice as well
 231                     result = df[ts_string]
 232                     expected = df[theslice]
 233                     tm.assert_frame_equal(result, expected)
 234 
 235             # Timestamp with resolution more precise than index
 236             # Compatible with existing key
 237             # Should return scalar for Series
 238             # and raise KeyError for Frame
 239             for fmt in formats[rnum + 1 :]:
 240                 ts_string = index[1].strftime(fmt)
 241                 result = df["a"][ts_string]
 242                 assert isinstance(result, np.int64)
 243                 assert result == 2
 244                 msg = fr"^'{ts_string}'$"
 245                 with pytest.raises(KeyError, match=msg):
 246                     df[ts_string]
 247 
 248             # Not compatible with existing key
 249             # Should raise KeyError
 250             for fmt, res in list(zip(formats, resolutions))[rnum + 1 :]:
 251                 ts = index[1] + Timedelta("1 " + res)
 252                 ts_string = ts.strftime(fmt)
 253                 msg = fr"^'{ts_string}'$"
 254                 with pytest.raises(KeyError, match=msg):
 255                     df["a"][ts_string]
 256                 with pytest.raises(KeyError, match=msg):
 257                     df[ts_string]

 259     def test_partial_slicing_with_multiindex(self):
 260 
 261         # GH 4758
 262         # partial string indexing with a multi-index buggy
 263         df = DataFrame(
 264             {
 265                 "ACCOUNT": ["ACCT1", "ACCT1", "ACCT1", "ACCT2"],
 266                 "TICKER": ["ABC", "MNP", "XYZ", "XYZ"],
 267                 "val": [1, 2, 3, 4],
 268             },
 269             index=date_range("2013-06-19 09:30:00", periods=4, freq="5T"),
 270         )
 271         df_multi = df.set_index(["ACCOUNT", "TICKER"], append=True)
 272 
 273         expected = DataFrame(
 274             [[1]], index=Index(["ABC"], name="TICKER"), columns=["val"]
 275         )
 276         result = df_multi.loc[("2013-06-19 09:30:00", "ACCT1")]
 277         tm.assert_frame_equal(result, expected)
 278 
 279         expected = df_multi.loc[
 280             (Timestamp("2013-06-19 09:30:00", tz=None), "ACCT1", "ABC")
 281         ]
 282         result = df_multi.loc[("2013-06-19 09:30:00", "ACCT1", "ABC")]
 283         tm.assert_series_equal(result, expected)
 284 
 285         # this is an IndexingError as we don't do partial string selection on
 286         # multi-levels.
 287         msg = "Too many indexers"
 288         with pytest.raises(IndexingError, match=msg):
 289             df_multi.loc[("2013-06-19", "ACCT1", "ABC")]
 290 
 291         # GH 4294
 292         # partial slice on a series mi
 293         s = DataFrame(
 294             np.random.rand(1000, 1000), index=date_range("2000-1-1", periods=1000)
 295         ).stack()
 296 
 297         s2 = s[:-1].copy()
 298         expected = s2["2000-1-4"]
 299         result = s2[Timestamp("2000-1-4")]
 300         tm.assert_series_equal(result, expected)
 301 
 302         result = s[Timestamp("2000-1-4")]
 303         expected = s["2000-1-4"]
 304         tm.assert_series_equal(result, expected)
 305 
 306         df2 = DataFrame(s)
 307         expected = df2.xs("2000-1-4")
 308         result = df2.loc[Timestamp("2000-1-4")]
 309         tm.assert_frame_equal(result, expected)

 311     def test_partial_slice_doesnt_require_monotonicity(self):
 312         # For historical reasons.
 313         s = Series(np.arange(10), date_range("2014-01-01", periods=10))
 314 
 315         nonmonotonic = s[[3, 5, 4]]
 316         expected = nonmonotonic.iloc[:0]
 317         timestamp = Timestamp("2014-01-10")
 318 
 319         tm.assert_series_equal(nonmonotonic["2014-01-10":], expected)
 320         with pytest.raises(KeyError, match=r"Timestamp\('2014-01-10 00:00:00'\)"):
 321             nonmonotonic[timestamp:]
 322 
 323         tm.assert_series_equal(nonmonotonic.loc["2014-01-10":], expected)
 324         with pytest.raises(KeyError, match=r"Timestamp\('2014-01-10 00:00:00'\)"):
 325             nonmonotonic.loc[timestamp:]
