 322     def test_equals(self):
 323         # GH 13107
 324         idx = pd.DatetimeIndex(["2011-01-01", "2011-01-02", "NaT"])
 325         assert idx.equals(idx)
 326         assert idx.equals(idx.copy())
 327         assert idx.equals(idx.astype(object))
 328         assert idx.astype(object).equals(idx)
 329         assert idx.astype(object).equals(idx.astype(object))
 330         assert not idx.equals(list(idx))
 331         assert not idx.equals(pd.Series(idx))
 332 
 333         idx2 = pd.DatetimeIndex(["2011-01-01", "2011-01-02", "NaT"], tz="US/Pacific")
 334         assert not idx.equals(idx2)
 335         assert not idx.equals(idx2.copy())
 336         assert not idx.equals(idx2.astype(object))
 337         assert not idx.astype(object).equals(idx2)
 338         assert not idx.equals(list(idx2))
 339         assert not idx.equals(pd.Series(idx2))
 340 
 341         # same internal, different tz
 342         idx3 = pd.DatetimeIndex(idx.asi8, tz="US/Pacific")
 343         tm.assert_numpy_array_equal(idx.asi8, idx3.asi8)
 344         assert not idx.equals(idx3)
 345         assert not idx.equals(idx3.copy())
 346         assert not idx.equals(idx3.astype(object))
 347         assert not idx.astype(object).equals(idx3)
 348         assert not idx.equals(list(idx3))
 349         assert not idx.equals(pd.Series(idx3))
 350 
 351         # check that we do not raise when comparing with OutOfBounds objects
 352         oob = pd.Index([datetime(2500, 1, 1)] * 3, dtype=object)
 353         assert not idx.equals(oob)
 354         assert not idx2.equals(oob)
 355         assert not idx3.equals(oob)
 356 
 357         # check that we do not raise when comparing with OutOfBounds dt64
 358         oob2 = oob.map(np.datetime64)
 359         assert not idx.equals(oob2)
 360         assert not idx2.equals(oob2)
 361         assert not idx3.equals(oob2)
