  86     def test_outer_join(self, freq):
  87         # should just behave as union
  88         start, end = datetime(2009, 1, 1), datetime(2010, 1, 1)
  89         rng = date_range(start=start, end=end, freq=freq)
  90 
  91         # overlapping
  92         left = rng[:10]
  93         right = rng[5:10]
  94 
  95         the_join = left.join(right, how="outer")
  96         assert isinstance(the_join, DatetimeIndex)
  97 
  98         # non-overlapping, gap in middle
  99         left = rng[:5]
 100         right = rng[10:]
 101 
 102         the_join = left.join(right, how="outer")
 103         assert isinstance(the_join, DatetimeIndex)
 104         assert the_join.freq is None
 105 
 106         # non-overlapping, no gap
 107         left = rng[:5]
 108         right = rng[5:10]
 109 
 110         the_join = left.join(right, how="outer")
 111         assert isinstance(the_join, DatetimeIndex)

 120     def test_naive_aware_conflicts(self):
 121         start, end = datetime(2009, 1, 1), datetime(2010, 1, 1)
 122         naive = date_range(start, end, freq=BDay(), tz=None)
 123         aware = date_range(start, end, freq=BDay(), tz="Asia/Hong_Kong")

 126         with pytest.raises(TypeError, match=msg):

 132     @pytest.mark.parametrize("tz", [None, "US/Pacific"])
 133     def test_join_preserves_freq(self, tz):
 134         # GH#32157
 135         dti = date_range("2016-01-01", periods=10, tz=tz)
 136         result = dti[:5].join(dti[5:], how="outer")
 137         assert result.freq == dti.freq
 138         tm.assert_index_equal(result, dti)
 139 
 140         result = dti[:5].join(dti[6:], how="outer")
 141         assert result.freq is None
 142         expected = dti.delete(5)
 143         tm.assert_index_equal(result, expected)
