 214     def test_index_convert_to_datetime_array(self):
 215         def _check_rng(rng):
 216             converted = rng.to_pydatetime()
 217             assert isinstance(converted, np.ndarray)
 218             for x, stamp in zip(converted, rng):
 219                 assert isinstance(x, datetime)
 220                 assert x == stamp.to_pydatetime()
 221                 assert x.tzinfo == stamp.tzinfo
 222 
 223         rng = date_range("20090415", "20090519")
 224         rng_eastern = date_range("20090415", "20090519", tz="US/Eastern")
 225         rng_utc = date_range("20090415", "20090519", tz="utc")
 226 
 227         _check_rng(rng)
 228         _check_rng(rng_eastern)
 229         _check_rng(rng_utc)
