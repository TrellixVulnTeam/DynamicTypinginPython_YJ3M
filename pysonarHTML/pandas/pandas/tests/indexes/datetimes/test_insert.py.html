  41     def test_insert(self):
  42         idx = DatetimeIndex(["2000-01-04", "2000-01-01", "2000-01-02"], name="idx")
  43 
  44         result = idx.insert(2, datetime(2000, 1, 5))
  45         exp = DatetimeIndex(
  46             ["2000-01-04", "2000-01-01", "2000-01-05", "2000-01-02"], name="idx"
  47         )
  48         tm.assert_index_equal(result, exp)
  49 
  50         # insertion of non-datetime should coerce to object index
  51         result = idx.insert(1, "inserted")
  52         expected = Index(
  53             [
  54                 datetime(2000, 1, 4),
  55                 "inserted",
  56                 datetime(2000, 1, 1),
  57                 datetime(2000, 1, 2),
  58             ],
  59             name="idx",
  60         )
  61         assert not isinstance(result, DatetimeIndex)
  62         tm.assert_index_equal(result, expected)
  63         assert result.name == expected.name
  64 
  65         idx = date_range("1/1/2000", periods=3, freq="M", name="idx")
  66 
  67         # preserve freq
  68         expected_0 = DatetimeIndex(
  69             ["1999-12-31", "2000-01-31", "2000-02-29", "2000-03-31"],

  71             freq="M",

  73         expected_3 = DatetimeIndex(
  74             ["2000-01-31", "2000-02-29", "2000-03-31", "2000-04-30"],
  75             name="idx",
  76             freq="M",
  77         )
  78 
  79         # reset freq to None
  80         expected_1_nofreq = DatetimeIndex(
  81             ["2000-01-31", "2000-01-31", "2000-02-29", "2000-03-31"],

  84         )
  85         expected_3_nofreq = DatetimeIndex(
  86             ["2000-01-31", "2000-02-29", "2000-03-31", "2000-01-02"],

  91         cases = [
  92             (0, datetime(1999, 12, 31), expected_0),
