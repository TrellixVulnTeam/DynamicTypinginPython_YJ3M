  60     def test_time_loc(self):  # GH8667
  61         from datetime import time
  62         from pandas._libs.index import _SIZE_CUTOFF
  63 
  64         ns = _SIZE_CUTOFF + np.array([-100, 100], dtype=np.int64)
  65         key = time(15, 11, 30)
  66         start = key.hour * 3600 + key.minute * 60 + key.second
  67         step = 24 * 3600
  68 
  69         for n in ns:
  70             idx = pd.date_range("2014-11-26", periods=n, freq="S")
  71             ts = pd.Series(np.random.randn(n), index=idx)

  73 
  74             tm.assert_numpy_array_equal(ts.index.get_loc(key), i, check_dtype=False)
  75             tm.assert_series_equal(ts[key], ts.iloc[i])
  76 
  77             left, right = ts.copy(), ts.copy()
  78             left[key] *= -10
  79             right.iloc[i] *= -10
  80             tm.assert_series_equal(left, right)

  82     def test_time_overflow_for_32bit_machines(self):
  83         # GH8943.  On some machines NumPy defaults to np.int32 (for example,
  84         # 32-bit Linux machines).  In the function _generate_regular_range
  85         # found in tseries/index.py, `periods` gets multiplied by `strides`
  86         # (which has value 1e9) and since the max value for np.int32 is ~2e9,
  87         # and since those machines won't promote np.int32 to np.int64, we get
  88         # overflow.
  89         periods = np.int_(1000)
  90 
  91         idx1 = pd.date_range(start="2000", periods=periods, freq="S")
  92         assert len(idx1) == periods
  93 
  94         idx2 = pd.date_range(end="2000", periods=periods, freq="S")
  95         assert len(idx2) == periods

 262     def test_factorize(self):
 263         idx1 = DatetimeIndex(
 264             ["2014-01", "2014-01", "2014-02", "2014-02", "2014-03", "2014-03"]
 265         )
 266 
 267         exp_arr = np.array([0, 0, 1, 1, 2, 2], dtype=np.intp)
 268         exp_idx = DatetimeIndex(["2014-01", "2014-02", "2014-03"])
 269 
 270         arr, idx = idx1.factorize()
 271         tm.assert_numpy_array_equal(arr, exp_arr)
 272         tm.assert_index_equal(idx, exp_idx)
 273 
 274         arr, idx = idx1.factorize(sort=True)
 275         tm.assert_numpy_array_equal(arr, exp_arr)
 276         tm.assert_index_equal(idx, exp_idx)
 277 
 278         # tz must be preserved
 279         idx1 = idx1.tz_localize("Asia/Tokyo")
 280         exp_idx = exp_idx.tz_localize("Asia/Tokyo")
 281 
 282         arr, idx = idx1.factorize()
 283         tm.assert_numpy_array_equal(arr, exp_arr)
 284         tm.assert_index_equal(idx, exp_idx)
 285 
 286         idx2 = pd.DatetimeIndex(
 287             ["2014-03", "2014-03", "2014-02", "2014-01", "2014-03", "2014-01"]
 288         )
 289 
 290         exp_arr = np.array([2, 2, 1, 0, 2, 0], dtype=np.intp)
 291         exp_idx = DatetimeIndex(["2014-01", "2014-02", "2014-03"])
 292         arr, idx = idx2.factorize(sort=True)
 293         tm.assert_numpy_array_equal(arr, exp_arr)
 294         tm.assert_index_equal(idx, exp_idx)
 295 
 296         exp_arr = np.array([0, 0, 1, 2, 0, 2], dtype=np.intp)
 297         exp_idx = DatetimeIndex(["2014-03", "2014-02", "2014-01"])
 298         arr, idx = idx2.factorize()
 299         tm.assert_numpy_array_equal(arr, exp_arr)
 300         tm.assert_index_equal(idx, exp_idx)
 301 
 302         # freq must be preserved
 303         idx3 = date_range("2000-01", periods=4, freq="M", tz="Asia/Tokyo")
 304         exp_arr = np.array([0, 1, 2, 3], dtype=np.intp)
 305         arr, idx = idx3.factorize()
 306         tm.assert_numpy_array_equal(arr, exp_arr)
 307         tm.assert_index_equal(idx, idx3)
