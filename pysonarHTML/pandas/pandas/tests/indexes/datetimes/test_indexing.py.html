 405     def test_get_loc(self):
 406         idx = pd.date_range("2000-01-01", periods=3)
 407 
 408         assert idx.get_loc("2000-01-01", method="nearest") == 0
 409         assert idx.get_loc("2000-01-01T12", method="nearest") == 1
 410 
 411         assert idx.get_loc("2000-01-01T12", method="nearest", tolerance="1 day") == 1
 412         assert (
 413             idx.get_loc("2000-01-01T12", method="nearest", tolerance=pd.Timedelta("1D"))
 414             == 1
 415         )
 416         assert (
 417             idx.get_loc(
 418                 "2000-01-01T12", method="nearest", tolerance=np.timedelta64(1, "D")
 419             )
 420             == 1
 421         )
 422         assert (
 423             idx.get_loc("2000-01-01T12", method="nearest", tolerance=timedelta(1)) == 1
 424         )
 425         with pytest.raises(ValueError, match="unit abbreviation w/o a number"):
 426             idx.get_loc("2000-01-01T12", method="nearest", tolerance="foo")
 427         with pytest.raises(KeyError, match="'2000-01-01T03'"):
 428             idx.get_loc("2000-01-01T03", method="nearest", tolerance="2 hours")
 429         with pytest.raises(
 430             ValueError, match="tolerance size must match target index size"
 431         ):
 432             idx.get_loc(
 433                 "2000-01-01",
 434                 method="nearest",
 435                 tolerance=[
 436                     pd.Timedelta("1day").to_timedelta64(),
 437                     pd.Timedelta("1day").to_timedelta64(),
 438                 ],
 439             )
 440 
 441         assert idx.get_loc("2000", method="nearest") == slice(0, 3)
 442         assert idx.get_loc("2000-01", method="nearest") == slice(0, 3)
 443 
 444         assert idx.get_loc("1999", method="nearest") == 0
 445         assert idx.get_loc("2001", method="nearest") == 2
 446 
 447         with pytest.raises(KeyError, match="'1999'"):
 448             idx.get_loc("1999", method="pad")
 449         with pytest.raises(KeyError, match="'2001'"):
 450             idx.get_loc("2001", method="backfill")
 451 
 452         with pytest.raises(KeyError, match="'foobar'"):
 453             idx.get_loc("foobar")
 454         with pytest.raises(InvalidIndexError, match=r"slice\(None, 2, None\)"):
 455             idx.get_loc(slice(2))
 456 
 457         idx = pd.to_datetime(["2000-01-01", "2000-01-04"])
 458         assert idx.get_loc("2000-01-02", method="nearest") == 0
 459         assert idx.get_loc("2000-01-03", method="nearest") == 1
 460         assert idx.get_loc("2000-01", method="nearest") == slice(0, 2)
 461 
 462         # time indexing
 463         idx = pd.date_range("2000-01-01", periods=24, freq="H")
 464         tm.assert_numpy_array_equal(
 465             idx.get_loc(time(12)), np.array([12]), check_dtype=False
 466         )
 467         tm.assert_numpy_array_equal(
 468             idx.get_loc(time(12, 30)), np.array([]), check_dtype=False
 469         )
 470         msg = "cannot yet lookup inexact labels when key is a time object"
 471         with pytest.raises(NotImplementedError, match=msg):
 472             idx.get_loc(time(12, 30), method="pad")

 542         tm.assert_numpy_array_equal(idx.get_indexer(idx), exp)
 543 
 544         target = idx[0] + pd.to_timedelta(["-1 hour", "12 hours", "1 day 1 hour"])
 545         tm.assert_numpy_array_equal(
 546             idx.get_indexer(target, "pad"), np.array([-1, 0, 1], dtype=np.intp)
 547         )
 548         tm.assert_numpy_array_equal(
 549             idx.get_indexer(target, "backfill"), np.array([0, 1, 2], dtype=np.intp)
 550         )
 551         tm.assert_numpy_array_equal(
 552             idx.get_indexer(target, "nearest"), np.array([0, 1, 1], dtype=np.intp)
 553         )
 554         tm.assert_numpy_array_equal(
 555             idx.get_indexer(target, "nearest", tolerance=pd.Timedelta("1 hour")),
 556             np.array([0, -1, 1], dtype=np.intp),
 557         )
 558         tol_raw = [
 559             pd.Timedelta("1 hour"),
 560             pd.Timedelta("1 hour"),
 561             pd.Timedelta("1 hour").to_timedelta64(),
 562         ]
 563         tm.assert_numpy_array_equal(
 564             idx.get_indexer(
 565                 target, "nearest", tolerance=[np.timedelta64(x) for x in tol_raw]
 566             ),
 567             np.array([0, -1, 1], dtype=np.intp),
 568         )
 569         tol_bad = [
 570             pd.Timedelta("2 hour").to_timedelta64(),
 571             pd.Timedelta("1 hour").to_timedelta64(),
 572             "foo",
 573         ]
 574         with pytest.raises(ValueError, match="abbreviation w/o a number"):
 575             idx.get_indexer(target, "nearest", tolerance=tol_bad)
 576         with pytest.raises(ValueError, match="abbreviation w/o a number"):
 577             idx.get_indexer(idx[[0]], method="nearest", tolerance="foo")
