 512     def test_constructor_coverage(self):
 513         rng = date_range("1/1/2000", periods=10.5)
 514         exp = date_range("1/1/2000", periods=10)
 515         tm.assert_index_equal(rng, exp)
 516 
 517         msg = "periods must be a number, got foo"
 518         with pytest.raises(TypeError, match=msg):
 519             date_range(start="1/1/2000", periods="foo", freq="D")
 520 
 521         msg = "DatetimeIndex\\(\\) must be called with a collection"
 522         with pytest.raises(TypeError, match=msg):
 523             DatetimeIndex("1/1/2000")
 524 
 525         # generator expression
 526         gen = (datetime(2000, 1, 1) + timedelta(i) for i in range(10))
 527         result = DatetimeIndex(gen)
 528         expected = DatetimeIndex(
 529             [datetime(2000, 1, 1) + timedelta(i) for i in range(10)]
 530         )
 531         tm.assert_index_equal(result, expected)
 532 
 533         # NumPy string array
 534         strings = np.array(["2000-01-01", "2000-01-02", "2000-01-03"])
 535         result = DatetimeIndex(strings)
 536         expected = DatetimeIndex(strings.astype("O"))
 537         tm.assert_index_equal(result, expected)
 538 
 539         from_ints = DatetimeIndex(expected.asi8)
 540         tm.assert_index_equal(from_ints, expected)
 541 
 542         # string with NaT
 543         strings = np.array(["2000-01-01", "2000-01-02", "NaT"])
 544         result = DatetimeIndex(strings)
 545         expected = DatetimeIndex(strings.astype("O"))
 546         tm.assert_index_equal(result, expected)
 547 
 548         from_ints = DatetimeIndex(expected.asi8)
 549         tm.assert_index_equal(from_ints, expected)
 550 
 551         # non-conforming
 552         msg = (
 553             "Inferred frequency None from passed values does not conform "
 554             "to passed frequency D"
 555         )
 556         with pytest.raises(ValueError, match=msg):
 557             DatetimeIndex(["2000-01-01", "2000-01-02", "2000-01-04"], freq="D")
 558 
 559         msg = (
 560             "Of the four parameters: start, end, periods, and freq, exactly "
 561             "three must be specified"
 562         )
 563         with pytest.raises(ValueError, match=msg):
 564             date_range(start="2011-01-01", freq="b")
 565         with pytest.raises(ValueError, match=msg):
