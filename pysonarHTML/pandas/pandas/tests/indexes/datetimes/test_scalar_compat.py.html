 103     def test_round(self, tz_naive_fixture):
 104         tz = tz_naive_fixture
 105         rng = date_range(start="2016-01-01", periods=5, freq="30Min", tz=tz)
 106         elt = rng[1]
 107 
 108         expected_rng = DatetimeIndex(
 109             [
 110                 Timestamp("2016-01-01 00:00:00", tz=tz, freq="30T"),
 111                 Timestamp("2016-01-01 00:00:00", tz=tz, freq="30T"),
 112                 Timestamp("2016-01-01 01:00:00", tz=tz, freq="30T"),
 113                 Timestamp("2016-01-01 02:00:00", tz=tz, freq="30T"),
 114                 Timestamp("2016-01-01 02:00:00", tz=tz, freq="30T"),
 115             ]
 116         )
 117         expected_elt = expected_rng[1]
 118 
 119         tm.assert_index_equal(rng.round(freq="H"), expected_rng)
 120         assert elt.round(freq="H") == expected_elt
 121 
 122         msg = INVALID_FREQ_ERR_MSG
 123         with pytest.raises(ValueError, match=msg):
 124             rng.round(freq="foo")
 125         with pytest.raises(ValueError, match=msg):
 126             elt.round(freq="foo")
 127 
 128         msg = "<MonthEnd> is a non-fixed frequency"
 129         with pytest.raises(ValueError, match=msg):
 130             rng.round(freq="M")
 131         with pytest.raises(ValueError, match=msg):
 132             elt.round(freq="M")
 133 
 134         # GH#14440 & GH#15578
 135         index = DatetimeIndex(["2016-10-17 12:00:00.0015"], tz=tz)
 136         result = index.round("ms")
 137         expected = DatetimeIndex(["2016-10-17 12:00:00.002000"], tz=tz)
 138         tm.assert_index_equal(result, expected)
 139 
 140         for freq in ["us", "ns"]:
 141             tm.assert_index_equal(index, index.round(freq))
 142 
 143         index = DatetimeIndex(["2016-10-17 12:00:00.00149"], tz=tz)
 144         result = index.round("ms")
 145         expected = DatetimeIndex(["2016-10-17 12:00:00.001000"], tz=tz)
 146         tm.assert_index_equal(result, expected)
 147 
 148         index = DatetimeIndex(["2016-10-17 12:00:00.001501031"])
 149         result = index.round("10ns")
 150         expected = DatetimeIndex(["2016-10-17 12:00:00.001501030"])
 151         tm.assert_index_equal(result, expected)
 152

 218     @pytest.mark.parametrize(
 219         "start, index_freq, periods",
 220         [("2018-01-01", "12H", 25), ("2018-01-01 0:0:0.124999", "1ns", 1000)],

 222     @pytest.mark.parametrize(
 223         "round_freq",
 224         [
 225             "2ns",
 226             "3ns",
 227             "4ns",
 228             "5ns",
 229             "6ns",
 230             "7ns",
 231             "250ns",
 232             "500ns",
 233             "750ns",
 234             "1us",
 235             "19us",
 236             "250us",
 237             "500us",
 238             "750us",
 239             "1s",
 240             "2s",
 241             "3s",
 242             "12H",
 243             "1D",
 244         ],
 245     )
 246     def test_round_int64(self, start, index_freq, periods, round_freq):
 247         dt = date_range(start=start, freq=index_freq, periods=periods)
 248         unit = to_offset(round_freq).nanos
 249 
 250         # test floor
 251         result = dt.floor(round_freq)
 252         diff = dt.asi8 - result.asi8
 253         mod = result.asi8 % unit
 254         assert (mod == 0).all(), f"floor not a {round_freq} multiple"
 255         assert (0 <= diff).all() and (diff < unit).all(), "floor error"
 256 
 257         # test ceil
 258         result = dt.ceil(round_freq)
 259         diff = result.asi8 - dt.asi8
 260         mod = result.asi8 % unit
 261         assert (mod == 0).all(), f"ceil not a {round_freq} multiple"
 262         assert (0 <= diff).all() and (diff < unit).all(), "ceil error"
 263 
 264         # test round
 265         result = dt.round(round_freq)
 266         diff = abs(result.asi8 - dt.asi8)
 267         mod = result.asi8 % unit
 268         assert (mod == 0).all(), f"round not a {round_freq} multiple"
 269         assert (diff <= unit // 2).all(), "round error"
 270         if unit % 2 == 0:
 271             assert (
 272                 result.asi8[diff == unit // 2] % 2 == 0
 273             ).all(), "round half to even error"

 278     def test_normalize(self):
 279         rng = date_range("1/1/2000 9:30", periods=10, freq="D")
 280 
 281         result = rng.normalize()
 282         expected = date_range("1/1/2000", periods=10, freq="D")
 283         tm.assert_index_equal(result, expected)
 284 
 285         arr_ns = np.array([1380585623454345752, 1380585612343234312]).astype(
 286             "datetime64[ns]"
 287         )
 288         rng_ns = DatetimeIndex(arr_ns)
 289         rng_ns_normalized = rng_ns.normalize()
 290 
 291         arr_ns = np.array([1380585600000000000, 1380585600000000000]).astype(
 292             "datetime64[ns]"
 293         )
 294         expected = DatetimeIndex(arr_ns)
 295         tm.assert_index_equal(rng_ns_normalized, expected)
 296 
 297         assert result.is_normalized
 298         assert not rng.is_normalized
