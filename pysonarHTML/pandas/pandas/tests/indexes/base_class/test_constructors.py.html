   1 import numpy as np
   2 import pytest
   3 
   4 from pandas import Index, MultiIndex
   5 
   6 
   7 class TestIndexConstructor:
   8     # Tests for the Index constructor, specifically for cases that do
   9     #  not return a subclass
  10 
  11     @pytest.mark.parametrize("value", [1, np.int64(1)])
  12     def test_constructor_corner(self, value):
  13         # corner case
  14         msg = (
  15             r"Index\(\.\.\.\) must be called with a collection of some "
  16             f"kind, {value} was passed"
  17         )
  18         with pytest.raises(TypeError, match=msg):
  19             Index(value)
  20 
  21     @pytest.mark.parametrize("index_vals", [[("A", 1), "B"], ["B", ("A", 1)]])
  22     def test_construction_list_mixed_tuples(self, index_vals):
  23         # see gh-10697: if we are constructing from a mixed list of tuples,
  24         # make sure that we are independent of the sorting order.
  25         index = Index(index_vals)
  26         assert isinstance(index, Index)
  27         assert not isinstance(index, MultiIndex)
  28 
  29     def test_constructor_wrong_kwargs(self):
  30         # GH #19348
  31         with pytest.raises(TypeError, match="Unexpected keyword arguments {'foo'}"):
  32             Index([], foo="bar")
  33 
  34     @pytest.mark.xfail(reason="see GH#21311: Index doesn't enforce dtype argument")
  35     def test_constructor_cast(self):
  36         msg = "could not convert string to float"
  37         with pytest.raises(ValueError, match=msg):
  38             Index(["a", "b", "c"], dtype=float)
