  20     def test_insert(self):
  21 
  22         # GH 7256
  23         # validate neg/pos inserts
  24         result = Index(["b", "c", "d"])
  25 
  26         # test 0th element
  27         tm.assert_index_equal(Index(["a", "b", "c", "d"]), result.insert(0, "a"))
  28 
  29         # test Nth element that follows Python list behavior
  30         tm.assert_index_equal(Index(["b", "c", "e", "d"]), result.insert(-1, "e"))
  31 
  32         # test loc +/- neq (0, -1)
  33         tm.assert_index_equal(result.insert(1, "z"), result.insert(-2, "z"))
  34 
  35         # test empty
  36         null_index = Index([])
  37         tm.assert_index_equal(Index(["a"]), null_index.insert(0, "a"))
  38 
  39     @pytest.mark.parametrize(
  40         "pos,expected",
  41         [
  42             (0, Index(["b", "c", "d"], name="index")),
  43             (-1, Index(["a", "b", "c"], name="index")),

  46     def test_delete(self, pos, expected):
  47         index = Index(["a", "b", "c", "d"], name="index")

  52     def test_append_multiple(self):
  53         index = Index(["a", "b", "c", "d", "e", "f"])
  54 
  55         foos = [index[:2], index[2:4], index[4:]]
  56         result = foos[0].append(foos[1:])

  59         # empty
