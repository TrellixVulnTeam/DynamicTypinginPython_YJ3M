  21     def test_setops_preserve_object_dtype(self):
  22         idx = pd.Index([1, 2, 3], dtype=object)
  23         result = idx.intersection(idx[1:])
  24         expected = idx[1:]
  25         tm.assert_index_equal(result, expected)
  26 
  27         # if other is not monotonic increasing, intersection goes through
