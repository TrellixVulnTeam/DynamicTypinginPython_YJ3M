 253     def test_ensure_copied_data(self, indices):
 254         # Check the "copy" argument of each Index.__new__ is honoured
 255         # GH12309
 256         init_kwargs = {}
 257         if isinstance(indices, PeriodIndex):
 258             # Needs "freq" specification:
 259             init_kwargs["freq"] = indices.freq
 260         elif isinstance(indices, (RangeIndex, MultiIndex, CategoricalIndex)):
 261             # RangeIndex cannot be initialized from data
 262             # MultiIndex and CategoricalIndex are tested separately
 263             return
 264 
 265         index_type = type(indices)
 266         result = index_type(indices.values, copy=True, **init_kwargs)
 267         if is_datetime64tz_dtype(indices.dtype):
 268             result = result.tz_localize("UTC").tz_convert(indices.tz)
 269         if isinstance(indices, (DatetimeIndex, TimedeltaIndex)):
 270             indices = indices._with_freq(None)
 271 
 272         tm.assert_index_equal(indices, result)
 273 
 274         if isinstance(indices, PeriodIndex):
 275             # .values an object array of Period, thus copied
 276             result = index_type(ordinal=indices.asi8, copy=False, **init_kwargs)
 277             tm.assert_numpy_array_equal(indices.asi8, result.asi8, check_same="same")
 278         elif isinstance(indices, IntervalIndex):
 279             # checked in test_interval.py
 280             pass
 281         else:
 282             result = index_type(indices.values, copy=False, **init_kwargs)
 283             tm.assert_numpy_array_equal(
 284                 indices.values, result.values, check_same="same"

 298         result3 = indices.memory_usage(deep=True)

 573     def test_equals_op(self):
 574         # GH9947, GH10637
 575         index_a = self.create_index()
 576         if isinstance(index_a, PeriodIndex):
 577             pytest.skip("Skip check for PeriodIndex")
 578 
 579         n = len(index_a)
 580         index_b = index_a[0:-1]
 581         index_c = index_a[0:-1].append(index_a[-2:-1])
 582         index_d = index_a[0:1]
 583 
 584         msg = "Lengths must match|could not be broadcast"
 585         with pytest.raises(ValueError, match=msg):
 586             index_a == index_b
 587         expected1 = np.array([True] * n)
 588         expected2 = np.array([True] * (n - 1) + [False])
 589         tm.assert_numpy_array_equal(index_a == index_a, expected1)
 590         tm.assert_numpy_array_equal(index_a == index_c, expected2)
 591 
 592         # test comparisons with numpy arrays
 593         array_a = np.array(index_a)
 594         array_b = np.array(index_a[0:-1])
 595         array_c = np.array(index_a[0:-1].append(index_a[-2:-1]))
 596         array_d = np.array(index_a[0:1])
 597         with pytest.raises(ValueError, match=msg):
 598             index_a == array_b
 599         tm.assert_numpy_array_equal(index_a == array_a, expected1)
 600         tm.assert_numpy_array_equal(index_a == array_c, expected2)
 601 
 602         # test comparisons with Series
 603         series_a = Series(array_a)
 604         series_b = Series(array_b)
 605         series_c = Series(array_c)
 606         series_d = Series(array_d)
 607         with pytest.raises(ValueError, match=msg):
 608             index_a == series_b
 609 
 610         tm.assert_numpy_array_equal(index_a == series_a, expected1)
 611         tm.assert_numpy_array_equal(index_a == series_c, expected2)
 612 
 613         # cases where length is 1 for one of them
 614         with pytest.raises(ValueError, match="Lengths must match"):
 615             index_a == index_d
 616         with pytest.raises(ValueError, match="Lengths must match"):
 617             index_a == series_d
 618         with pytest.raises(ValueError, match="Lengths must match"):
 619             index_a == array_d
 620         msg = "Can only compare identically-labeled Series objects"
 621         with pytest.raises(ValueError, match=msg):
 622             series_a == series_d
 623         with pytest.raises(ValueError, match="Lengths must match"):
 624             series_a == array_d
 625
