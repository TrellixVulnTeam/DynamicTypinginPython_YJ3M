 194     def test_get_unique_index(self, indices):
 195         # MultiIndex tested separately
 196         if not len(indices) or isinstance(indices, MultiIndex):
 197             pytest.skip("Skip check for empty Index and MultiIndex")
 198 
 199         idx = indices[[0] * 5]
 200         idx_unique = indices[[0]]
 201 
 202         # We test against `idx_unique`, so first we make sure it's unique
 203         # and doesn't contain nans.
 204         assert idx_unique.is_unique is True
 205         try:
 206             assert idx_unique.hasnans is False
 207         except NotImplementedError:
 208             pass
 209 
 210         for dropna in [False, True]:
 211             result = idx._get_unique_index(dropna=dropna)
 212             tm.assert_index_equal(result, idx_unique)
 213 
 214         # nans:
 215         if not indices._can_hold_na:
 216             pytest.skip("Skip na-check if index cannot hold na")
 217 
 218         if is_period_dtype(indices.dtype):
 219             vals = indices[[0] * 5]._data
 220             vals[0] = pd.NaT
 221         elif needs_i8_conversion(indices.dtype):
 222             vals = indices.asi8[[0] * 5]
 223             vals[0] = iNaT
 224         else:
 225             vals = indices.values[[0] * 5]
 226             vals[0] = np.nan
 227 
 228         vals_unique = vals[:2]
 229         idx_nan = indices._shallow_copy(vals)
 230         idx_unique_nan = indices._shallow_copy(vals_unique)
 231         assert idx_unique_nan.is_unique is True
 232 
 233         assert idx_nan.dtype == indices.dtype
 234         assert idx_unique_nan.dtype == indices.dtype
 235 
 236         for dropna, expected in zip([False, True], [idx_unique_nan, idx_unique]):
 237             for i in [idx_nan, idx_unique_nan]:
 238                 result = i._get_unique_index(dropna=dropna)
 239                 tm.assert_index_equal(result, expected)

 251     def test_searchsorted_monotonic(self, indices):
 252         # GH17271
 253         # not implemented for tuple searches in MultiIndex
 254         # or Intervals searches in IntervalIndex
 255         if isinstance(indices, (MultiIndex, pd.IntervalIndex)):
 256             pytest.skip("Skip check for MultiIndex/IntervalIndex")
 257 
 258         # nothing to test if the index is empty
 259         if indices.empty:
 260             pytest.skip("Skip check for empty Index")
 261         value = indices[0]
 262 
 263         # determine the expected results (handle dupes for 'right')
 264         expected_left, expected_right = 0, (indices == value).argmin()
 265         if expected_right == 0:
 266             # all values are the same, expected_right should be length
 267             expected_right = len(indices)
 268 
 269         # test _searchsorted_monotonic in all cases
 270         # test searchsorted only for increasing
 271         if indices.is_monotonic_increasing:
 272             ssm_left = indices._searchsorted_monotonic(value, side="left")
 273             assert expected_left == ssm_left
 274 
 275             ssm_right = indices._searchsorted_monotonic(value, side="right")
 276             assert expected_right == ssm_right
 277 
 278             ss_left = indices.searchsorted(value, side="left")
 279             assert expected_left == ss_left
 280 
 281             ss_right = indices.searchsorted(value, side="right")
 282             assert expected_right == ss_right
 283 
 284         elif indices.is_monotonic_decreasing:
 285             ssm_left = indices._searchsorted_monotonic(value, side="left")
 286             assert expected_left == ssm_left
 287 
 288             ssm_right = indices._searchsorted_monotonic(value, side="right")
 289             assert expected_right == ssm_right
 290         else:
 291             # non-monotonic should raise.
 292             with pytest.raises(ValueError):
 293                 indices._searchsorted_monotonic(value, side="left")

 301     def test_drop_duplicates(self, indices, keep):
 302         if isinstance(indices, MultiIndex):
 303             pytest.skip("MultiIndex is tested separately")
 304         if isinstance(indices, RangeIndex):
 305             pytest.skip(
 306                 "RangeIndex is tested in test_drop_duplicates_no_duplicates "
 307                 "as it cannot hold duplicates"
 308             )
 309         if len(indices) == 0:
 310             pytest.skip(
 311                 "empty index is tested in test_drop_duplicates_no_duplicates "
 312                 "as it cannot hold duplicates"
 313             )
 314 
 315         # make unique index
 316         holder = type(indices)
 317         unique_values = list(set(indices))
 318         unique_idx = holder(unique_values)
 319 
 320         # make duplicated index
 321         n = len(unique_idx)
 322         duplicated_selection = np.random.choice(n, int(n * 1.5))
 323         idx = holder(unique_idx.values[duplicated_selection])
 324 
 325         # Series.duplicated is tested separately
 326         expected_duplicated = (
 327             pd.Series(duplicated_selection).duplicated(keep=keep).values
 328         )
 329         tm.assert_numpy_array_equal(idx.duplicated(keep=keep), expected_duplicated)
 330 
 331         # Series.drop_duplicates is tested separately
 332         expected_dropped = holder(pd.Series(idx).drop_duplicates(keep=keep))
 333         tm.assert_index_equal(idx.drop_duplicates(keep=keep), expected_dropped)

 335     def test_drop_duplicates_no_duplicates(self, indices):
 336         if isinstance(indices, MultiIndex):
 337             pytest.skip("MultiIndex is tested separately")
 338 
 339         # make unique index
 340         if isinstance(indices, RangeIndex):
 341             # RangeIndex cannot have duplicates
 342             unique_idx = indices
 343         else:
 344             holder = type(indices)
 345             unique_values = list(set(indices))
 346             unique_idx = holder(unique_values)
 347 
 348         # check on unique index
 349         expected_duplicated = np.array([False] * len(unique_idx), dtype="bool")
 350         tm.assert_numpy_array_equal(unique_idx.duplicated(), expected_duplicated)
 351         result_dropped = unique_idx.drop_duplicates()
 352         tm.assert_index_equal(result_dropped, unique_idx)
 353         # validate shallow copy
 354         assert result_dropped is not unique_idx

 373     @pytest.mark.parametrize(
 374         "dtype",
 375         ["int64", "uint64", "float64", "category", "datetime64[ns]", "timedelta64[ns]"],
 376     )
 377     @pytest.mark.parametrize("copy", [True, False])
 378     def test_astype_preserves_name(self, indices, dtype, copy):
 379         # https://github.com/pandas-dev/pandas/issues/32013
 380         if isinstance(indices, MultiIndex):
 381             indices.names = ["idx" + str(i) for i in range(indices.nlevels)]
 382         else:
 383             indices.name = "idx"
 384 
 385         try:
 386             # Some of these conversions cannot succeed so we use a try / except
 387             if copy:
 388                 result = indices.copy(dtype=dtype)
 389             else:
 390                 result = indices.astype(dtype)
 391         except (ValueError, TypeError, NotImplementedError, SystemError):
 392             return
 393 
 394         if isinstance(indices, MultiIndex):
 395             assert result.names == indices.names
 396         else:
 397             assert result.name == indices.name
