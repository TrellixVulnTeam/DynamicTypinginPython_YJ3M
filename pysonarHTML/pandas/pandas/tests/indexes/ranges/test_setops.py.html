  12         # intersect with Int64Index
  13         index = RangeIndex(start=0, stop=20, step=2)
  14         other = Index(np.arange(1, 6))
  15         result = index.intersection(other, sort=sort)
  16         expected = Index(np.sort(np.intersect1d(index.values, other.values)))
  17         tm.assert_index_equal(result, expected)
  18

  21             np.sort(np.asarray(np.intersect1d(index.values, other.values)))

  81     def test_union_noncomparable(self, sort):
  82         # corner case, non-Int64Index
  83         index = RangeIndex(start=0, stop=20, step=2)
  84         other = Index([datetime.now() + timedelta(i) for i in range(4)], dtype=object)
  85         result = index.union(other, sort=sort)
  86         expected = Index(np.concatenate((index, other)))
  87         tm.assert_index_equal(result, expected)
  88 
  89         result = other.union(index, sort=sort)
  90         expected = Index(np.concatenate((other, index)))
  91         tm.assert_index_equal(result, expected)

 228     def test_union_sorted(self, unions):
 229 
 230         idx1, idx2, expected_sorted, expected_notsorted = unions
 231 
 232         res1 = idx1.union(idx2, sort=None)
 233         tm.assert_index_equal(res1, expected_sorted, exact=True)
 234 
 235         res1 = idx1.union(idx2, sort=False)
 236         tm.assert_index_equal(res1, expected_notsorted, exact=True)
 237 
 238         res2 = idx2.union(idx1, sort=None)
 239         res3 = idx1._int64index.union(idx2, sort=None)
 240         tm.assert_index_equal(res2, expected_sorted, exact=True)
 241         tm.assert_index_equal(res3, expected_sorted)
