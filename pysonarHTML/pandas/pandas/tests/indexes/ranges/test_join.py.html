 129     def test_join_non_int_index(self):
 130         index = RangeIndex(start=0, stop=20, step=2)
 131         other = Index([3, 6, 7, 8, 10], dtype=object)
 132 
 133         outer = index.join(other, how="outer")
 134         outer2 = other.join(index, how="outer")
 135         expected = Index([0, 2, 3, 4, 6, 7, 8, 10, 12, 14, 16, 18])
 136         tm.assert_index_equal(outer, outer2)
 137         tm.assert_index_equal(outer, expected)
 138 
 139         inner = index.join(other, how="inner")
 140         inner2 = other.join(index, how="inner")
 141         expected = Index([6, 8, 10])
 142         tm.assert_index_equal(inner, inner2)
 143         tm.assert_index_equal(inner, expected)
 144 
 145         left = index.join(other, how="left")
 146         tm.assert_index_equal(left, index.astype(object))
 147 
 148         left2 = other.join(index, how="left")
 149         tm.assert_index_equal(left2, other)
 150 
 151         right = index.join(other, how="right")
 152         tm.assert_index_equal(right, other)
 153 
 154         right2 = other.join(index, how="right")
 155         tm.assert_index_equal(right2, index.astype(object))
