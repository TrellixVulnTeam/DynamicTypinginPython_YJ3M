  10 class TestRangeIndexConstructors:
  11     @pytest.mark.parametrize("name", [None, "foo"])
  12     @pytest.mark.parametrize(
  13         "args, kwargs, start, stop, step",
  14         [
  15             ((5,), dict(), 0, 5, 1),
  16             ((1, 5), dict(), 1, 5, 1),
  17             ((1, 5, 2), dict(), 1, 5, 2),
  18             ((0,), dict(), 0, 0, 1),
  19             ((0, 0), dict(), 0, 0, 1),
  20             (tuple(), dict(start=0), 0, 0, 1),
  21             (tuple(), dict(stop=0), 0, 0, 1),
  22         ],
  23     )
  24     def test_constructor(self, args, kwargs, start, stop, step, name):
  25         result = RangeIndex(*args, name=name, **kwargs)
  26         expected = Index(np.arange(start, stop, step, dtype=np.int64), name=name)
  27         assert isinstance(result, RangeIndex)
  28         assert result.name is name
  29         assert result._range == range(start, stop, step)
  30         tm.assert_index_equal(result, expected)

  32     def test_constructor_invalid_args(self):
  33         msg = "RangeIndex\\(\\.\\.\\.\\) must be called with integers"

  71     def test_constructor_same(self):
  72 
  73         # pass thru w and w/o copy
  74         index = RangeIndex(1, 5, 2)
  75         result = RangeIndex(index, copy=False)
  76         assert result.identical(index)
  77 
  78         result = RangeIndex(index, copy=True)
  79         tm.assert_index_equal(result, index, exact=True)
  80 
  81         result = RangeIndex(index)
  82         tm.assert_index_equal(result, index, exact=True)
  83 
  84         with pytest.raises(
  85             ValueError,
  86             match="Incorrect `dtype` passed: expected signed integer, received float64",
  87         ):
  88             RangeIndex(index, dtype="float64")

  90     def test_constructor_range(self):
  91 
  92         msg = "Value needs to be a scalar value, was type range"
  93         with pytest.raises(TypeError, match=msg):
  94             result = RangeIndex(range(1, 5, 2))
  95 
  96         result = RangeIndex.from_range(range(1, 5, 2))
  97         expected = RangeIndex(1, 5, 2)
  98         tm.assert_index_equal(result, expected, exact=True)
  99 
 100         result = RangeIndex.from_range(range(5, 6))
 101         expected = RangeIndex(5, 6, 1)
 102         tm.assert_index_equal(result, expected, exact=True)
 103 
 104         # an invalid range
 105         result = RangeIndex.from_range(range(5, 1))
 106         expected = RangeIndex(0, 0, 1)
 107         tm.assert_index_equal(result, expected, exact=True)
 108 
 109         result = RangeIndex.from_range(range(5))
 110         expected = RangeIndex(0, 5, 1)
 111         tm.assert_index_equal(result, expected, exact=True)
 112 
 113         result = Index(range(1, 5, 2))
 114         expected = RangeIndex(1, 5, 2)
 115         tm.assert_index_equal(result, expected, exact=True)
 116 
 117         with pytest.raises(
 118             ValueError,
 119             match="Incorrect `dtype` passed: expected signed integer, received float64",
 120         ):
 121             Index(range(1, 5, 2), dtype="float64")
 122         msg = r"^from_range\(\) got an unexpected keyword argument"
 123         with pytest.raises(TypeError, match=msg):
 124             RangeIndex.from_range(range(10), copy=True)

 126     def test_constructor_name(self):
 127         # GH#12288
 128         orig = RangeIndex(10)
 129         orig.name = "original"
 130 
 131         copy = RangeIndex(orig)
 132         copy.name = "copy"
 133 
 134         assert orig.name == "original"
 135         assert copy.name == "copy"
 136 
 137         new = Index(copy)
 138         assert new.name == "copy"
 139 
 140         new.name = "new"
 141         assert orig.name == "original"
 142         assert copy.name == "copy"
 143         assert new.name == "new"

 145     def test_constructor_corner(self):
 146         arr = np.array([1, 2, 3, 4], dtype=object)
 147         index = RangeIndex(1, 5)
 148         assert index.values.dtype == np.int64
 149         tm.assert_index_equal(index, Index(arr))
 150 
 151         # non-int raise Exception
 152         with pytest.raises(TypeError, match=r"Wrong type \<class 'str'\>"):
 153             RangeIndex("1", "10", "1")
 154         with pytest.raises(TypeError, match=r"Wrong type \<class 'float'\>"):
 155             RangeIndex(1.1, 10.2, 1.3)
 156 
 157         # invalid passed type
 158         with pytest.raises(
 159             ValueError,
 160             match="Incorrect `dtype` passed: expected signed integer, received float64",
 161         ):
 162             RangeIndex(1, 5, dtype="float64")
