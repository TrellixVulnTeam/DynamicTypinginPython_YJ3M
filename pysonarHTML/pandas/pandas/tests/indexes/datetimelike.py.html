  61     def test_map_callable(self):
  62         index = self.create_index()
  63         expected = index + index.freq
  64         result = index.map(lambda x: x + x.freq)
  65         tm.assert_index_equal(result, expected)
  66 
  67         # map to NaT
  68         result = index.map(lambda x: pd.NaT if x == index[0] else x)
  69         expected = pd.Index([pd.NaT] + index[1:].tolist())
  70         tm.assert_index_equal(result, expected)

  72     @pytest.mark.parametrize(
  73         "mapper",
  74         [
  75             lambda values, index: {i: e for e, i in zip(values, index)},
  76             lambda values, index: pd.Series(values, index, dtype=object),
  77         ],
  78     )
  79     def test_map_dictlike(self, mapper):
  80         index = self.create_index()
  81         expected = index + index.freq
  82 
  83         # don't compare the freqs
  84         if isinstance(expected, (pd.DatetimeIndex, pd.TimedeltaIndex)):
  85             expected = expected._with_freq(None)
  86 
  87         result = index.map(mapper(expected, index))
  88         tm.assert_index_equal(result, expected)
  89 
  90         expected = pd.Index([pd.NaT] + index[1:].tolist())
  91         result = index.map(mapper(expected, index))
  92         tm.assert_index_equal(result, expected)
  93 
  94         # empty map; these map to np.nan because we cannot know
  95         # to re-infer things
  96         expected = pd.Index([np.nan] * len(index))
  97         result = index.map(mapper([], []))
  98         tm.assert_index_equal(result, expected)
