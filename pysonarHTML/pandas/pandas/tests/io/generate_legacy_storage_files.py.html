   1 """
   2 self-contained to write legacy storage pickle files
   3 
   4 To use this script. Create an environment where you want
   5 generate pickles, say its for 0.20.3, with your pandas clone
   6 in ~/pandas
   7 
   8 . activate pandas_0.20.3
   9 cd ~/
  10 
  11 $ python pandas/pandas/tests/io/generate_legacy_storage_files.py \
  12     pandas/pandas/tests/io/data/legacy_pickle/0.20.3/ pickle
  13 
  14 This script generates a storage file for the current arch, system,
  15 and python version
  16   pandas version: 0.20.3
  17   output dir    : pandas/pandas/tests/io/data/legacy_pickle/0.20.3/
  18   storage format: pickle
  19 created pickle file: 0.20.3_x86_64_darwin_3.5.2.pickle
  20 
  21 The idea here is you are using the *current* version of the
  22 generate_legacy_storage_files with an *older* version of pandas to
  23 generate a pickle file. We will then check this file into a current
  24 branch, and test using test_pickle.py. This will load the *older*
  25 pickles and test versus the current data that is generated
  26 (with master). These are then compared.
  27 
  28 If we have cases where we changed the signature (e.g. we renamed
  29 offset -> freq in Timestamp). Then we have to conditionally execute
  30 in the generate_legacy_storage_files.py to make it
  31 run under the older AND the newer version.

  94 def _create_sp_series():
  95     nan = np.nan
  96 
  97     # nan-based
  98     arr = np.arange(15, dtype=np.float64)
  99     arr[7:12] = nan
 100     arr[-1:] = nan
 101 
 102     bseries = Series(SparseArray(arr, kind="block"))
 103     bseries.name = "bseries"

 107 def _create_sp_tsseries():
 108     nan = np.nan
 109 
 110     # nan-based
 111     arr = np.arange(15, dtype=np.float64)

 115     date_index = bdate_range("1/1/2011", periods=len(arr))
 116     bseries = Series(SparseArray(arr, kind="block"), index=date_index)
 117     bseries.name = "btsseries"

 123 
 124     data = {

 135 def create_data():
 136     """ create the pickle data """
 137     data = {
 138         "A": [0.0, 1.0, 2.0, 3.0, np.nan],
 139         "B": [0, 1, 0, 1, 0],
 140         "C": ["foo1", "foo2", "foo3", "foo4", "foo5"],

 144 
 145     scalars = dict(timestamp=Timestamp("20130101"), period=Period("2012", "M"))
 146 
 147     index = dict(
 148         int=Index(np.arange(10)),

 150         period=period_range("2013-01-01", freq="M", periods=10),

 156     index["range"] = RangeIndex(10)
 157

 163     mi = dict(

 184         mi=Series(
 185             np.arange(5).astype(np.float64),

 191         cat=Series(Categorical(["foo", "bar", "baz"])),

 200         float=DataFrame({"A": series["float"], "B": series["float"] + 1}),

 211                         ]
 212                     )
 213                 ),
 214                 names=["first", "second"],
 215             ),
 216         ),
 217         dup=DataFrame(

 220         cat_onecol=DataFrame({"A": Categorical(["foo", "bar"])}),

 305 def platform_name():
 306     return "_".join(
 307         [
 308             str(pandas.__version__),
 309             str(pl.machine()),
 310             str(pl.system().lower()),
 311             str(pl.python_version()),
 312         ]

 316 def write_legacy_pickles(output_dir):
 317 
 318     version = pandas.__version__
 319 
 320     print(
 321         "This script generates a storage file for the current arch, system, "
 322         "and python version"
 323     )
 324     print(f"  pandas version: {version}")
 325     print(f"  output dir    : {output_dir}")
 326     print("  storage format: pickle")
 327 
 328     pth = f"{platform_name()}.pickle"
 329 
 330     fh = open(os.path.join(output_dir, pth), "wb")
 331     pickle.dump(create_pickle_data(), fh, pickle.HIGHEST_PROTOCOL)
 332     fh.close()
 333 
 334     print(f"created pickle file: {pth}")

 337 def write_legacy_file():
 338     # force our cwd to be the first searched
 339     sys.path.insert(0, ".")
 340 
 341     if not (3 <= len(sys.argv) <= 4):
 342         exit(
 343             "Specify output directory and storage type: generate_legacy_"
 344             "storage_files.py <output_dir> <storage_type> "
 345         )
 346 
 347     output_dir = str(sys.argv[1])
 348     storage_type = str(sys.argv[2])
 349 
 350     if storage_type == "pickle":
 351         write_legacy_pickles(output_dir=output_dir)
 352     else:
 353         exit("storage_type must be one of {'pickle'}")
 354 
 355 
 356 if __name__ == "__main__":
 357     write_legacy_file()
