  17 def numeric_as_float(data):
  18     for v in data.columns:
  19         if data[v].dtype is np.dtype("int64"):
  20             data[v] = data[v].astype(np.float64)

  33     def test1_basic(self):
  34         # Tests with DEMO_G.xpt (all numeric file)
  35 
  36         # Compare to this
  37         data_csv = pd.read_csv(self.file01.replace(".xpt", ".csv"))
  38         numeric_as_float(data_csv)
  39 
  40         # Read full file
  41         data = read_sas(self.file01, format="xport")
  42         tm.assert_frame_equal(data, data_csv)
  43         num_rows = data.shape[0]
  44 
  45         # Test reading beyond end of file
  46         reader = read_sas(self.file01, format="xport", iterator=True)
  47         data = reader.read(num_rows + 100)
  48         assert data.shape[0] == num_rows
  49         reader.close()
  50 
  51         # Test incremental read with `read` method.
  52         reader = read_sas(self.file01, format="xport", iterator=True)
  53         data = reader.read(10)
  54         reader.close()
  55         tm.assert_frame_equal(data, data_csv.iloc[0:10, :])
  56 
  57         # Test incremental read with `get_chunk` method.
  58         reader = read_sas(self.file01, format="xport", chunksize=10)
  59         data = reader.get_chunk()
  60         reader.close()
  61         tm.assert_frame_equal(data, data_csv.iloc[0:10, :])
  62 
  63         # Test read in loop
  64         m = 0
  65         reader = read_sas(self.file01, format="xport", chunksize=100)
  66         for x in reader:
  67             m += x.shape[0]
  68         reader.close()
  69         assert m == num_rows
  70 
  71         # Read full file with `read_sas` method
  72         data = read_sas(self.file01)
  73         tm.assert_frame_equal(data, data_csv)

  99     def test1_incremental(self):
