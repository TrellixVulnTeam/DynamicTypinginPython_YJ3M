  43 def test_to_jsonl():
  44     # GH9180
  45     df = DataFrame([[1, 2], [1, 2]], columns=["a", "b"])
  46     result = df.to_json(orient="records", lines=True)
  47     expected = '{"a":1,"b":2}\n{"a":1,"b":2}'
  48     assert result == expected
  49 
  50     df = DataFrame([["foo}", "bar"], ['foo"', "bar"]], columns=["a", "b"])
  51     result = df.to_json(orient="records", lines=True)
  52     expected = '{"a":"foo}","b":"bar"}\n{"a":"foo\\"","b":"bar"}'
  53     assert result == expected
  54     tm.assert_frame_equal(read_json(result, lines=True), df)
