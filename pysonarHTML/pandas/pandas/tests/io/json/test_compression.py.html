   9 def test_compression_roundtrip(compression):
  10     df = pd.DataFrame(
  11         [[0.123456, 0.234567, 0.567567], [12.32112, 123123.2, 321321.2]],
  12         index=["A", "B"],
  13         columns=["X", "Y", "Z"],
  14     )
  15 
  16     with tm.ensure_clean() as path:
  17         df.to_json(path, compression=compression)
  18         tm.assert_frame_equal(df, pd.read_json(path, compression=compression))
  19 
  20         # explicitly ensure file was compressed.
  21         with tm.decompress_file(path, compression) as fh:
  22             result = fh.read().decode("utf8")
  23         tm.assert_frame_equal(df, pd.read_json(result))
