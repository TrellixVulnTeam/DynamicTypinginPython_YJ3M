 165     def test_roundtrip_str_axes(self, orient, convert_axes, numpy, dtype):
 166         df = DataFrame(
 167             np.zeros((200, 4)),
 168             columns=[str(i) for i in range(4)],
 169             index=[str(i) for i in range(200)],
 170             dtype=dtype,
 171         )
 172 
 173         # TODO: do we even need to support U3 dtypes?
 174         if numpy and dtype == "U3" and orient != "split":
 175             pytest.xfail("Can't decode directly to array")
 176 
 177         data = df.to_json(orient=orient)
 178         result = pd.read_json(
 179             data, orient=orient, convert_axes=convert_axes, numpy=numpy, dtype=dtype
 180         )
 181 
 182         expected = df.copy()
 183         if not dtype:
 184             expected = expected.astype(np.int64)
 185 
 186         # index columns, and records orients cannot fully preserve the string
 187         # dtype for axes as the index and column labels are used as keys in
 188         # JSON objects. JSON keys are by definition strings, so there's no way
 189         # to disambiguate whether those keys actually were strings or numeric
 190         # beforehand and numeric wins out.
 191         # TODO: Split should be able to support this
 192         if convert_axes and (orient in ("split", "index", "columns")):
 193             expected.columns = expected.columns.astype(np.int64)
 194             expected.index = expected.index.astype(np.int64)
 195         elif orient == "records" and convert_axes:
 196             expected.columns = expected.columns.astype(np.int64)
 197 
 198         assert_json_roundtrip_equal(result, expected, orient)
