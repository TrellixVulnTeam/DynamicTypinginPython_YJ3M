 212     def teardown_method(self, method):
 213         # if setup fails, there may not be a connection to close.
 214         if hasattr(self, "conn"):
 215             for tbl in self._get_all_tables():
 216                 self.drop_table(tbl)
 217             self._close_conn()

 220 class MySQLMixIn(MixInBase):
 221     def drop_table(self, table_name):
 222         cur = self.conn.cursor()
 223         cur.execute(f"DROP TABLE IF EXISTS {sql._get_valid_mysql_name(table_name)}")
 224         self.conn.commit()
 225 
 226     def _get_all_tables(self):
 227         cur = self.conn.cursor()
 228         cur.execute("SHOW TABLES")
 229         return [table[0] for table in cur.fetchall()]
 230 
 231     def _close_conn(self):
 232         from pymysql.err import Error
 233 
 234         try:
 235             self.conn.close()
 236         except Error:
 237             pass
