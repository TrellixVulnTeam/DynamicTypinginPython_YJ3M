  45     def assert_equal_style(cell1, cell2, engine):
  46         if engine in ["xlsxwriter", "openpyxl"]:
  47             pytest.xfail(
  48                 reason=(f"GH25351: failing on some attribute comparisons in {engine}")
  49             )
  50         # TODO: should find a better way to check equality
  51         assert cell1.alignment.__dict__ == cell2.alignment.__dict__
  52         assert cell1.border.__dict__ == cell2.border.__dict__
  53         assert cell1.fill.__dict__ == cell2.fill.__dict__
  54         assert cell1.font.__dict__ == cell2.font.__dict__
  55         assert cell1.number_format == cell2.number_format
  56         assert cell1.protection.__dict__ == cell2.protection.__dict__

  58     def custom_converter(css):
  59         # use bold iff there is custom style attached to the cell
  60         if css.strip(" \n;"):
  61             return {"font": {"bold": True}}
  62         return {}
