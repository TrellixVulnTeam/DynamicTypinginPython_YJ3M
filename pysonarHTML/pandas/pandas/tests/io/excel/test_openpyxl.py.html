  17 def test_to_excel_styleconverter(ext):
  18     from openpyxl import styles
  19 
  20     hstyle = {
  21         "font": {"color": "00FF0000", "bold": True},
  22         "borders": {"top": "thin", "right": "thin", "bottom": "thin", "left": "thin"},
  23         "alignment": {"horizontal": "center", "vertical": "top"},
  24         "fill": {"patternType": "solid", "fgColor": {"rgb": "006666FF", "tint": 0.3}},
  25         "number_format": {"format_code": "0.00"},
  26         "protection": {"locked": True, "hidden": False},
  27     }
  28 
  29     font_color = styles.Color("00FF0000")
  30     font = styles.Font(bold=True, color=font_color)
  31     side = styles.Side(style=styles.borders.BORDER_THIN)
  32     border = styles.Border(top=side, right=side, bottom=side, left=side)
  33     alignment = styles.Alignment(horizontal="center", vertical="top")
  34     fill_color = styles.Color(rgb="006666FF", tint=0.3)
  35     fill = styles.PatternFill(patternType="solid", fgColor=fill_color)
  36 
  37     number_format = "0.00"
  38 
  39     protection = styles.Protection(locked=True, hidden=False)
  40 
  41     kw = _OpenpyxlWriter._convert_to_style_kwargs(hstyle)
  42     assert kw["font"] == font
  43     assert kw["border"] == border
  44     assert kw["alignment"] == alignment
  45     assert kw["fill"] == fill
  46     assert kw["number_format"] == number_format
  47     assert kw["protection"] == protection

  50 def test_write_cells_merge_styled(ext):
  51     from pandas.io.formats.excel import ExcelCell
  52 
  53     sheet_name = "merge_styled"
  54 
  55     sty_b1 = {"font": {"color": "00FF0000"}}
  56     sty_a2 = {"font": {"color": "0000FF00"}}
  57 
  58     initial_cells = [
  59         ExcelCell(col=1, row=0, val=42, style=sty_b1),
  60         ExcelCell(col=0, row=1, val=99, style=sty_a2),
  61     ]
  62 
  63     sty_merged = {"font": {"color": "000000FF", "bold": True}}
  64     sty_kwargs = _OpenpyxlWriter._convert_to_style_kwargs(sty_merged)
  65     openpyxl_sty_merged = sty_kwargs["font"]
  66     merge_cells = [
  67         ExcelCell(
  68             col=0, row=0, val="pandas", mergestart=1, mergeend=1, style=sty_merged
  69         )
  70     ]
  71 
  72     with tm.ensure_clean(ext) as path:
  73         writer = _OpenpyxlWriter(path)
  74         writer.write_cells(initial_cells, sheet_name=sheet_name)
  75         writer.write_cells(merge_cells, sheet_name=sheet_name)
  76 
  77         wks = writer.sheets[sheet_name]
  78         xcell_b1 = wks["B1"]
  79         xcell_a2 = wks["A2"]
  80         assert xcell_b1.font == openpyxl_sty_merged
  81         assert xcell_a2.font == openpyxl_sty_merged

  85     "mode,expected", [("w", ["baz"]), ("a", ["foo", "bar", "baz"])]

  87 def test_write_append_mode(ext, mode, expected):
  88     df = DataFrame([1], columns=["baz"])
  89 
  90     with tm.ensure_clean(ext) as f:
  91         wb = openpyxl.Workbook()
  92         wb.worksheets[0].title = "foo"
  93         wb.worksheets[0]["A1"].value = "foo"
  94         wb.create_sheet("bar")
  95         wb.worksheets[1]["A1"].value = "bar"
  96         wb.save(f)
  97 
  98         writer = ExcelWriter(f, engine="openpyxl", mode=mode)
  99         df.to_excel(writer, sheet_name="baz", index=False)
 100         writer.save()
 101 
 102         wb2 = openpyxl.load_workbook(f)
 103         result = [sheet.title for sheet in wb2.worksheets]
 104         assert result == expected
 105 
 106         for index, cell_value in enumerate(expected):
 107             assert wb2.worksheets[index]["A1"].value == cell_value
