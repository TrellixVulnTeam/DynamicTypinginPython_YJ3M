  64 def _is_valid_engine_ext_pair(engine, read_ext: str) -> bool:
  65     """
  66     Filter out invalid (engine, ext) pairs instead of skipping, as that
  67     produces 500+ pytest.skips.
  68     """
  69     engine = engine.values[0]
  70     if engine == "openpyxl" and read_ext == ".xls":
  71         return False
  72     if engine == "odf" and read_ext != ".ods":
  73         return False
  74     if read_ext == ".ods" and engine != "odf":
  75         return False
  76     if engine == "pyxlsb" and read_ext != ".xlsb":
  77         return False
  78     if read_ext == ".xlsb" and engine != "pyxlsb":
  79         return False
  80     return True

  83 def _transfer_marks(engine, read_ext):
  84     """
  85     engine gives us a pytest.param objec with some marks, read_ext is just
  86     a string.  We need to generate a new pytest.param inheriting the marks.
  87     """
  88     values = engine.values + (read_ext,)
  89     new_param = pytest.param(values, marks=engine.marks)
  90     return new_param
