  48 def compare_element(result, expected, typ, version=None):
  49     if isinstance(expected, Index):
  50         tm.assert_index_equal(expected, result)
  51         return
  52 
  53     if typ.startswith("sp_"):
  54         comparator = tm.assert_equal
  55         comparator(result, expected)
  56     elif typ == "timestamp":
  57         if expected is pd.NaT:
  58             assert result is pd.NaT
  59         else:
  60             assert result == expected
  61             assert result.freq == expected.freq
  62     else:
  63         comparator = getattr(tm, f"assert_{typ}_equal", tm.assert_almost_equal)
  64         comparator(result, expected)
  65

  67 def compare(data, vf, version):
  68 
  69     data = pd.read_pickle(vf)
  70 
  71     m = globals()
  72     for typ, dv in data.items():
  73         for dt, result in dv.items():
  74             expected = data[typ][dt]
  75 
  76             # use a specific comparator
  77             # if available
  78             comparator = f"compare_{typ}_{dt}"
  79 
  80             comparator = m.get(comparator, m["compare_element"])
  81             comparator(result, expected, typ, version)
  82     return data

 156 def test_round_trip_current(current_pickle_data):
 157     def python_pickler(obj, path):
 158         with open(path, "wb") as fh:
 159             pickle.dump(obj, fh, protocol=-1)
 160 
 161     def python_unpickler(path):
 162         with open(path, "rb") as fh:
 163             fh.seek(0)
 164             return pickle.load(fh)
 165 
 166     data = current_pickle_data
 167     for typ, dv in data.items():
 168         for dt, expected in dv.items():
 169 
 170             for writer in [pd.to_pickle, python_pickler]:
 171                 if writer is None:
 172                     continue
 173 
 174                 with tm.ensure_clean() as path:
 175 
 176                     # test writing with each pickler
 177                     writer(expected, path)
 178 
 179                     # test reading with each unpickler
 180                     result = pd.read_pickle(path)
 181                     compare_element(result, expected, typ)
 182 
 183                     result = python_unpickler(path)
 184                     compare_element(result, expected, typ)

 237 class TestCompression:
 238 
 239     _compression_to_extension = {
 240         None: ".none",
 241         "gzip": ".gz",
 242         "bz2": ".bz2",
 243         "zip": ".zip",
 244         "xz": ".xz",
 245     }
 246 
 247     def compress_file(self, src_path, dest_path, compression):
 248         if compression is None:
 249             shutil.copyfile(src_path, dest_path)
 250             return
 251 
 252         if compression == "gzip":
 253             f = gzip.open(dest_path, "w")
 254         elif compression == "bz2":
 255             f = bz2.BZ2File(dest_path, "w")
 256         elif compression == "zip":
 257             with zipfile.ZipFile(dest_path, "w", compression=zipfile.ZIP_DEFLATED) as f:
 258                 f.write(src_path, os.path.basename(src_path))
 259         elif compression == "xz":
 260             f = _get_lzma_file(lzma)(dest_path, "w")
 261         else:
 262             msg = f"Unrecognized compression type: {compression}"
 263             raise ValueError(msg)
 264 
 265         if compression != "zip":
 266             with open(src_path, "rb") as fh, f:
 267                 f.write(fh.read())

 269     def test_write_explicit(self, compression, get_random_path):
 270         base = get_random_path
 271         path1 = base + ".compressed"
 272         path2 = base + ".raw"
 273 
 274         with tm.ensure_clean(path1) as p1, tm.ensure_clean(path2) as p2:
 275             df = tm.makeDataFrame()
 276 
 277             # write to compressed file
 278             df.to_pickle(p1, compression=compression)
 279 
 280             # decompress
 281             with tm.decompress_file(p1, compression=compression) as f:
 282                 with open(p2, "wb") as fh:
 283                     fh.write(f.read())
 284 
 285             # read decompressed file
 286             df2 = pd.read_pickle(p2, compression=None)
 287
