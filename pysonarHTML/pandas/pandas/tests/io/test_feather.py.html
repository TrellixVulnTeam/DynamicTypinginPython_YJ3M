  23 class TestFeather:
  24     def check_error_on_write(self, df, exc):
  25         # check that we are raising the exception
  26         # on writing
  27 
  28         with pytest.raises(exc):
  29             with tm.ensure_clean() as path:
  30                 to_feather(df, path)
  31 
  32     def check_round_trip(self, df, expected=None, write_kwargs={}, **read_kwargs):
  33 
  34         if expected is None:
  35             expected = df
  36 
  37         with tm.ensure_clean() as path:
  38             to_feather(df, path, **write_kwargs)
  39 
  40             result = read_feather(path, **read_kwargs)
  41             tm.assert_frame_equal(result, expected)
  42 
  43     def test_error(self):
  44 
  45         for obj in [
  46             pd.Series([1, 2, 3]),
  47             1,
  48             "foo",
  49             pd.Timestamp("20130101"),
  50             np.array([1, 2, 3]),
  51         ]:
  52             self.check_error_on_write(obj, ValueError)

  54     def test_basic(self):
  55 
  56         df = pd.DataFrame(
  57             {
  58                 "string": list("abc"),
  59                 "int": list(range(1, 4)),
  60                 "uint": np.arange(3, 6).astype("u1"),
  61                 "float": np.arange(4.0, 7.0, dtype="float64"),
  62                 "float_with_null": [1.0, np.nan, 3],
  63                 "bool": [True, False, True],
  64                 "bool_with_null": [True, np.nan, False],
  65                 "cat": pd.Categorical(list("abc")),
  66                 "dt": pd.DatetimeIndex(
  67                     list(pd.date_range("20130101", periods=3)), freq=None
  68                 ),
  69                 "dttz": pd.DatetimeIndex(
  70                     list(pd.date_range("20130101", periods=3, tz="US/Eastern")),
  71                     freq=None,
  72                 ),
  73                 "dt_with_null": [
  74                     pd.Timestamp("20130101"),
  75                     pd.NaT,
  76                     pd.Timestamp("20130103"),
  77                 ],
  78                 "dtns": pd.DatetimeIndex(
  79                     list(pd.date_range("20130101", periods=3, freq="ns")), freq=None,
  80                 ),
  81             }
  82         )
  83         if pyarrow_version >= LooseVersion("0.16.1.dev"):
  84             df["periods"] = pd.period_range("2013", freq="M", periods=3)
  85             df["timedeltas"] = pd.timedelta_range("1 day", periods=3)
  86             # TODO temporary disable due to regression in pyarrow 0.17.1
  87             # https://github.com/pandas-dev/pandas/issues/34255
  88             # df["intervals"] = pd.interval_range(0, 3, 3)
  89 
  90         assert df.dttz.dtype.tz.zone == "US/Eastern"
  91         self.check_round_trip(df)

 130     def test_write_with_index(self):
 131 
 132         df = pd.DataFrame({"A": [1, 2, 3]})
 133         self.check_round_trip(df)
 134 
 135         # non-default index
 136         for index in [
 137             [2, 3, 4],
 138             pd.date_range("20130101", periods=3),
 139             list("abc"),
 140             [1, 3, 4],
 141             pd.MultiIndex.from_tuples([("a", 1), ("a", 2), ("b", 1)]),
 142         ]:
 143 
 144             df.index = index
 145             self.check_error_on_write(df, ValueError)
 146 
 147         # index with meta-data
 148         df.index = [0, 1, 2]
 149         df.index.name = "foo"
 150         self.check_error_on_write(df, ValueError)
 151 
 152         # column multi-index
 153         df.index = [0, 1, 2]
 154         df.columns = pd.MultiIndex.from_tuples([("a", 1)])
 155         self.check_error_on_write(df, ValueError)
