 126 def check_round_trip(
 127     df,
 128     engine=None,
 129     path=None,
 130     write_kwargs=None,
 131     read_kwargs=None,
 132     expected=None,
 133     check_names=True,
 134     check_like=False,
 135     repeat=2,
 136 ):
 137     """Verify parquet serializer and deserializer produce the same results.
 138 
 139     Performs a pandas to disk and disk to pandas round trip,
 140     then compares the 2 resulting DataFrames to verify equality.
 141 
 142     Parameters
 143     ----------
 144     df: Dataframe
 145     engine: str, optional
 146         'pyarrow' or 'fastparquet'
 147     path: str, optional
 148     write_kwargs: dict of str:str, optional
 149     read_kwargs: dict of str:str, optional
 150     expected: DataFrame, optional
 151         Expected deserialization result, otherwise will be equal to `df`
 152     check_names: list of str, optional
 153         Closed set of column names to be compared
 154     check_like: bool, optional
 155         If True, ignore the order of index & columns.
 156     repeat: int, optional
 157         How many times to repeat the test
 158     """
 159     write_kwargs = write_kwargs or {"compression": None}
 160     read_kwargs = read_kwargs or {}
 161 
 162     if expected is None:
 163         expected = df
 164 
 165     if engine:
 166         write_kwargs["engine"] = engine
 167         read_kwargs["engine"] = engine
 168 
 169     def compare(repeat):
 170         for _ in range(repeat):
 171             df.to_parquet(path, **write_kwargs)
 172             with catch_warnings(record=True):
 173                 actual = read_parquet(path, **read_kwargs)
 174 
 175             tm.assert_frame_equal(
 176                 expected, actual, check_names=check_names, check_like=check_like
 177             )
 178 
 179     if path is None:
 180         with tm.ensure_clean() as path:
 181             compare(repeat)
 182     else:
 183         compare(repeat)

 232 def test_get_engine_auto_error_message():
 233     # Expect different error messages from get_engine(engine="auto")
 234     # if engines aren't installed vs. are installed but bad version
 235     from pandas.compat._optional import VERSIONS
 236 
 237     # Do we have engines installed, but a bad version of them?
 238     pa_min_ver = VERSIONS.get("pyarrow")
 239     fp_min_ver = VERSIONS.get("fastparquet")
 240     have_pa_bad_version = (
 241         False
 242         if not _HAVE_PYARROW
 243         else LooseVersion(pyarrow.__version__) < LooseVersion(pa_min_ver)
 244     )
 245     have_fp_bad_version = (
 246         False
 247         if not _HAVE_FASTPARQUET
 248         else LooseVersion(fastparquet.__version__) < LooseVersion(fp_min_ver)
 249     )
 250     # Do we have usable engines installed?
 251     have_usable_pa = _HAVE_PYARROW and not have_pa_bad_version
 252     have_usable_fp = _HAVE_FASTPARQUET and not have_fp_bad_version
 253 
 254     if not have_usable_pa and not have_usable_fp:
 255         # No usable engines found.
 256         if have_pa_bad_version:
 257             match = f"Pandas requires version .{pa_min_ver}. or newer of .pyarrow."
 258             with pytest.raises(ImportError, match=match):
 259                 get_engine("auto")
 260         else:
 261             match = "Missing optional dependency .pyarrow."
 262             with pytest.raises(ImportError, match=match):
 263                 get_engine("auto")
 264 
 265         if have_fp_bad_version:
 266             match = f"Pandas requires version .{fp_min_ver}. or newer of .fastparquet."
 267             with pytest.raises(ImportError, match=match):
 268                 get_engine("auto")
 269         else:
 270             match = "Missing optional dependency .fastparquet."
 271             with pytest.raises(ImportError, match=match):
 272                 get_engine("auto")

 433     def test_write_ignoring_index(self, engine):
 434         # ENH 20768
 435         # Ensure index=False omits the index from the written Parquet file.
 436         df = pd.DataFrame({"a": [1, 2, 3], "b": ["q", "r", "s"]})
 437 
 438         write_kwargs = {"compression": None, "index": False}

 440         # Because we're dropping the index, we expect the loaded dataframe to

 442         expected = df.reset_index(drop=True)

 444         check_round_trip(df, engine, write_kwargs=write_kwargs, expected=expected)
 445 
 446         # Ignore custom index
 447         df = pd.DataFrame(
 448             {"a": [1, 2, 3], "b": ["q", "r", "s"]}, index=["zyx", "wvu", "tsr"]
 449         )

 451         check_round_trip(df, engine, write_kwargs=write_kwargs, expected=expected)
 452 
 453         # Ignore multi-indexes as well.
 454         arrays = [
 455             ["bar", "bar", "baz", "baz", "foo", "foo", "qux", "qux"],
 456             ["one", "two", "one", "two", "one", "two", "one", "two"],

 460         )
 461
