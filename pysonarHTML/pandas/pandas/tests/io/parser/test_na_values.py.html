  82 def test_default_na_values(all_parsers):
  83     _NA_VALUES = {
  84         "-1.#IND",
  85         "1.#QNAN",
  86         "1.#IND",
  87         "-1.#QNAN",
  88         "#N/A",
  89         "N/A",
  90         "n/a",
  91         "NA",
  92         "<NA>",
  93         "#NA",
  94         "NULL",
  95         "null",
  96         "NaN",
  97         "nan",
  98         "-NaN",
  99         "-nan",
 100         "#N/A N/A",
 101         "",
 102     }
 103     assert _NA_VALUES == STR_NA_VALUES
 104 
 105     parser = all_parsers
 106     nv = len(_NA_VALUES)
 107 
 108     def f(i, v):
 109         if i == 0:
 110             buf = ""
 111         elif i > 0:
 112             buf = "".join([","] * i)
 113 
 114         buf = f"{buf}{v}"
 115 
 116         if i < nv - 1:
 117             joined = "".join([","] * (nv - i - 1))
 118             buf = f"{buf}{joined}"
 119 
 120         return buf
 121 
 122     data = StringIO("\n".join(f(i, v) for i, v in enumerate(_NA_VALUES)))
 123     expected = DataFrame(np.nan, columns=range(nv), index=range(nv))
 124 
 125     result = parser.read_csv(data, header=None)
 126     tm.assert_frame_equal(result, expected)
