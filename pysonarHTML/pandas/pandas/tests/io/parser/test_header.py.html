 501 @pytest.mark.parametrize("index_col", [None, [0]])
 502 @pytest.mark.parametrize(
 503     "columns", [None, (["", "Unnamed"]), (["Unnamed", ""]), (["Unnamed", "NotUnnamed"])]
 504 )
 505 def test_multi_index_unnamed(all_parsers, index_col, columns):
 506     # see gh-23687
 507     #
 508     # When specifying a multi-index header, make sure that
 509     # we don't error just because one of the rows in our header
 510     # has ALL column names containing the string "Unnamed". The
 511     # correct condition to check is whether the row contains
 512     # ALL columns that did not have names (and instead were given
 513     # placeholder ones).
 514     parser = all_parsers
 515     header = [0, 1]
 516 
 517     if index_col is None:
 518         data = ",".join(columns or ["", ""]) + "\n0,1\n2,3\n4,5\n"
 519     else:
 520         data = ",".join([""] + (columns or ["", ""])) + "\n,0,1\n0,2,3\n1,4,5\n"
 521 
 522     if columns is None:
 523         msg = (
 524             r"Passed header=\[0,1\] are too "
 525             r"many rows for this multi_index of columns"
 526         )
 527         with pytest.raises(ParserError, match=msg):
 528             parser.read_csv(StringIO(data), header=header, index_col=index_col)
 529     else:
 530         result = parser.read_csv(StringIO(data), header=header, index_col=index_col)
 531         template = "Unnamed: {i}_level_0"
 532         exp_columns = []
 533 
 534         for i, col in enumerate(columns):
 535             if not col:  # Unnamed.
 536                 col = template.format(i=i if index_col is None else i + 1)
 537 
 538             exp_columns.append(col)
 539 
 540         columns = MultiIndex.from_tuples(zip(exp_columns, ["0", "1"]))
 541         expected = DataFrame([[2, 3], [4, 5]], columns=columns)
 542         tm.assert_frame_equal(result, expected)
