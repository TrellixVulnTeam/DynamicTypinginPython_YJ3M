 259     #
 260     # Turns out "T060" is parsable as a datetime slice!
 261     tz_list = [1, 10, 20, 30, 60, 80, 100]
 262     widths = [16] + [8] * len(tz_list)
 263     names = ["SST"] + [f"T{z:03d}" for z in tz_list[1:]]
 264 
 265     data = """  2009164202000   9.5403  9.4105  8.6571  7.8372  6.0612  5.8843  5.5192
 266 2009164203000   9.5435  9.2010  8.6167  7.8176  6.0804  5.8728  5.4869
 267 2009164204000   9.5873  9.1326  8.4694  7.5889  6.0422  5.8526  5.4657
 268 2009164205000   9.5810  9.0896  8.4009  7.4652  6.0322  5.8189  5.4379
 269 2009164210000   9.6034  9.0897  8.3822  7.4905  6.0908  5.7904  5.4039
 270 """
 271 
 272     result = read_fwf(
 273         StringIO(data),
 274         index_col=0,
 275         header=None,
 276         names=names,
 277         widths=widths,
 278         parse_dates=True,
 279         date_parser=lambda s: datetime.strptime(s, "%Y%j%H%M%S"),
 280     )
 281     expected = DataFrame(
 282         [
 283             [9.5403, 9.4105, 8.6571, 7.8372, 6.0612, 5.8843, 5.5192],
 284             [9.5435, 9.2010, 8.6167, 7.8176, 6.0804, 5.8728, 5.4869],

 287             [9.6034, 9.0897, 8.3822, 7.4905, 6.0908, 5.7904, 5.4039],
 288         ],
 289         index=DatetimeIndex(
 290             [
 291                 "2009-06-13 20:20:00",
 292                 "2009-06-13 20:30:00",
 293                 "2009-06-13 20:40:00",
 294                 "2009-06-13 20:50:00",

 303 def test_fwf_for_uint8():
 304     data = """1421302965.213420    PRI=3 PGN=0xef00      DST=0x17 SRC=0x28    04 154 00 00 00 00 00 127
 305 1421302964.226776    PRI=6 PGN=0xf002               SRC=0x47    243 00 00 255 247 00 00 71"""  # noqa
 306     df = read_fwf(
 307         StringIO(data),
 308         colspecs=[(0, 17), (25, 26), (33, 37), (49, 51), (58, 62), (63, 1000)],
 309         names=["time", "pri", "pgn", "dst", "src", "data"],
 310         converters={
 311             "pgn": lambda x: int(x, 16),

 313             "dst": lambda x: int(x, 16),
 314             "data": lambda x: len(x.split(" ")),
 315         },
 316     )
 317 
 318     expected = DataFrame(
 319         [
 320             [1421302965.213420, 3, 61184, 23, 40, 8],
 321             [1421302964.226776, 6, 61442, None, 71, 8],
 322         ],
 323         columns=["time", "pri", "pgn", "dst", "src", "data"],
 324     )
 325     expected["dst"] = expected["dst"].astype(object)
 326     tm.assert_frame_equal(df, expected)
