  97     def test_parse_public_s3_bucket_chunked(self, tips_df):
  98         # Read with a chunksize
  99         chunksize = 5
 100         for ext, comp in [("", None), (".gz", "gzip"), (".bz2", "bz2")]:
 101             df_reader = read_csv(
 102                 "s3://pandas-test/tips.csv" + ext, chunksize=chunksize, compression=comp
 103             )
 104             assert df_reader.chunksize == chunksize
 105             for i_chunk in [0, 1, 2]:
 106                 # Read a couple of chunks and make sure we see them
 107                 # properly.
 108                 df = df_reader.get_chunk()
 109                 assert isinstance(df, DataFrame)
 110                 assert not df.empty
 111                 true_df = tips_df.iloc[chunksize * i_chunk : chunksize * (i_chunk + 1)]
 112                 tm.assert_frame_equal(true_df, df)

 213     def test_read_csv_chunked_download(self, s3_resource, caplog):
 214         # 8 MB, S3FS usees 5MB chunks
 215         import s3fs
 216 
 217         df = DataFrame(np.random.randn(100000, 4), columns=list("abcd"))
 218         str_buf = StringIO()
 219 
 220         df.to_csv(str_buf)
 221 
 222         buf = BytesIO(str_buf.getvalue().encode("utf-8"))
 223 
 224         s3_resource.Bucket("pandas-test").put_object(Key="large-file.csv", Body=buf)
 225 
 226         # Possibly some state leaking in between tests.
 227         # If we don't clear this cache, we saw `GetObject operation: Forbidden`.
 228         # Presumably the s3fs instance is being cached, with the directory listing
 229         # from *before* we add the large-file.csv in the pandas-test bucket.
 230         s3fs.S3FileSystem.clear_instance_cache()
 231 
 232         with caplog.at_level(logging.DEBUG, logger="s3fs"):
 233             read_csv("s3://pandas-test/large-file.csv", nrows=5)
