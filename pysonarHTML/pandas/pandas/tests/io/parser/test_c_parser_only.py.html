 150 def test_precise_conversion(c_parser_only):
 151     from decimal import Decimal
 152 
 153     parser = c_parser_only
 154 
 155     normal_errors = []
 156     precise_errors = []
 157 
 158     # test numbers between 1 and 2
 159     for num in np.linspace(1.0, 2.0, num=500):
 160         # 25 decimal digits of precision
 161         text = f"a\n{num:.25}"
 162 
 163         normal_val = float(parser.read_csv(StringIO(text))["a"][0])
 164         precise_val = float(
 165             parser.read_csv(StringIO(text), float_precision="high")["a"][0]
 166         )
 167         roundtrip_val = float(
 168             parser.read_csv(StringIO(text), float_precision="round_trip")["a"][0]
 169         )
 170         actual_val = Decimal(text[2:])
 171 
 172         def error(val):
 173             return abs(Decimal(f"{val:.100}") - actual_val)
 174 
 175         normal_errors.append(error(normal_val))
 176         precise_errors.append(error(precise_val))
 177 
 178         # round-trip should match float()
 179         assert roundtrip_val == float(text[2:])
 180 
 181     assert sum(precise_errors) <= sum(normal_errors)
 182     assert max(precise_errors) <= max(normal_errors)

 312     # to stress the system memory allocator, to cause it to move the
 313     # stream buffer and either let the OS reclaim the region, or let
 314     # other memory requests of parser otherwise modify the contents
 315     # of memory space, where it was formally located.
 316     # This test is designed to cause a `segfault` with unpatched
 317     # `tokenizer.c`. Sometimes the test fails on `segfault`, other
 318     # times it fails due to memory corruption, which causes the
 319     # loaded DataFrame to differ from the expected one.

 321     parser = c_parser_only

 324     # approx 1.5KiB).

 326         """9999-9,99:99,,,,ZZ,ZZ,,,ZZZ-ZZZZ,.Z-ZZZZ,-9.99,,,9.99,Z"""
 327         """ZZZZ,,-99,9,ZZZ-ZZZZ,ZZ-ZZZZ,,9.99,ZZZ-ZZZZZ,ZZZ-ZZZZZ,"""
 328         """ZZZ-ZZZZ,ZZZ-ZZZZ,ZZZ-ZZZZ,ZZZ-ZZZZ,ZZZ-ZZZZ,ZZZ-ZZZZ,9"""

 330         """9,9,99,99,999,999,ZZZZZ,ZZZ-ZZZZZ,ZZZ-ZZZZ,9,ZZ-ZZZZ,9."""
 331         """99,ZZ-ZZZZ,ZZ-ZZZZ,,,,ZZZZ,,,ZZ,ZZ,,,,,,,,,,,,,9,,,999."""
 332         """99,999.99,,,ZZZZZ,,,Z9,,,,,,,ZZZ,ZZZ,,,,,,,,,,,ZZZZZ,ZZ"""
 333         """ZZZ,ZZZ-ZZZZZZ,ZZZ-ZZZZZZ,ZZ-ZZZZ,ZZ-ZZZZ,ZZ-ZZZZ,ZZ-ZZ"""
 334         """ZZ,,,999999,999999,ZZZ,ZZZ,,,ZZZ,ZZZ,999.99,999.99,,,,Z"""
 335         """ZZ-ZZZ,ZZZ-ZZZ,-9.99,-9.99,9,9,,99,,9.99,9.99,9,9,9.99,"""
 336         """9.99,,,,9.99,9.99,,99,,99,9.99,9.99,,,ZZZ,ZZZ,,999.99,,"""
 337         """999.99,ZZZ,ZZZ-ZZZZ,ZZZ-ZZZZ,,,ZZZZZ,ZZZZZ,ZZZ,ZZZ,9,9,"""
 338         """,,,,,ZZZ-ZZZZ,ZZZ999Z,,,999.99,,999.99,ZZZ-ZZZZ,,,9.999"""
 339         """,9.999,9.999,9.999,-9.999,-9.999,-9.999,-9.999,9.999,9."""
 340         """999,9.999,9.999,9.999,9.999,9.999,9.999,99999,ZZZ-ZZZZ,"""
 341         """,9.99,ZZZ,,,,,,,,ZZZ,,,,,9,,,,9,,,,,,,,,,ZZZ-ZZZZ,ZZZ-Z"""

 343         """,999,,,,ZZ-ZZZZ,ZZZ,ZZZ,ZZZ-ZZZZ,ZZZ-ZZZZ,,,99.99,99.99"""
 344         """,,,9.99,9.99,9.99,9.99,ZZZ-ZZZZ,,,ZZZ-ZZZZZ,,,,,-9.99,-"""
 345         """9.99,-9.99,-9.99,,,,,,,,,ZZZ-ZZZZ,,9,9.99,9.99,99ZZ,,-9"""
 346         """.99,-9.99,ZZZ-ZZZZ,,,,,,,ZZZ-ZZZZ,9.99,9.99,9999,,,,,,,"""
 347         """,,,-9.9,Z/Z-ZZZZ,999.99,9.99,,999.99,ZZ-ZZZZ,ZZ-ZZZZ,9."""
 348         """99,9.99,9.99,9.99,9.99,9.99,,ZZZ-ZZZZZ,ZZZ-ZZZZZ,ZZZ-ZZ"""
 349         """ZZZ,ZZZ-ZZZZZ,ZZZ-ZZZZZ,ZZZ,ZZZ,ZZZ,ZZZ,9.99,,,-9.99,ZZ"""
 350         """-ZZZZ,-999.99,,-9999,,999.99,,,,999.99,99.99,,,ZZ-ZZZZZ"""
 351         """ZZZ,ZZ-ZZZZ-ZZZZZZZ,,,,ZZ-ZZ-ZZZZZZZZ,ZZZZZZZZ,ZZZ-ZZZZ"""

 357     # Set the number of lines so that a call to `parser_trim_buffers`
 358     # is triggered: after a couple of full chunks are consumed a
 359     # relatively small 'residual' chunk would cause reallocation
