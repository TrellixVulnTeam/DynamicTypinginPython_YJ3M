  19 def test_skip_rows_bug(all_parsers, skiprows):
  20     # see gh-505
  21     parser = all_parsers
  22     text = """#foo,a,b,c
  23 #foo,a,b,c
  24 #foo,a,b,c
  25 #foo,a,b,c
  26 #foo,a,b,c

  31 """
  32     result = parser.read_csv(

  34     )
  35     index = Index(
  36         [datetime(2000, 1, 1), datetime(2000, 1, 2), datetime(2000, 1, 3)], name=0

  41     )

  47     parser = all_parsers

  49         [",".join([str(i), str(i + 1), str(i + 2)]) for i in range(10)]

  53     )

  55     result = parser.read_csv(StringIO(data), skiprows=[6, 8])
  56     condensed_result = parser.read_csv(StringIO(condensed_data))

 176 @pytest.mark.parametrize(
 177     "line_terminator", ["\n", "\r\n", "\r"]  # "LF"  # "CRLF"  # "CR"
 178 )
 179 def test_skiprows_lineterminator(all_parsers, line_terminator):
 180     # see gh-9079
 181     parser = all_parsers
 182     data = "\n".join(
 183         [
 184             "SMOSMANIA ThetaProbe-ML2X ",
 185             "2007/01/01 01:00   0.2140 U M ",
 186             "2007/01/01 02:00   0.2141 M O ",
 187             "2007/01/01 04:00   0.2142 D M ",
 188         ]
 189     )
 190     expected = DataFrame(
 191         [
 192             ["2007/01/01", "01:00", 0.2140, "U", "M"],
 193             ["2007/01/01", "02:00", 0.2141, "M", "O"],
 194             ["2007/01/01", "04:00", 0.2142, "D", "M"],
 195         ],
 196         columns=["date", "time", "var", "flag", "oflag"],
 197     )
 198 
 199     if parser.engine == "python" and line_terminator == "\r":
 200         pytest.skip("'CR' not respect with the Python parser yet")
 201 
 202     data = data.replace("\n", line_terminator)
 203     result = parser.read_csv(
 204         StringIO(data),
 205         skiprows=1,
 206         delim_whitespace=True,
 207         names=["date", "time", "var", "flag", "oflag"],
 208     )
 209     tm.assert_frame_equal(result, expected)
