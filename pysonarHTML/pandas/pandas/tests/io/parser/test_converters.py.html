  26 @pytest.mark.parametrize("column", [3, "D"])

  30 def test_converters(all_parsers, column, converter):
  31     parser = all_parsers
  32     data = """A,B,C,D

  34 b,3,4,01/02/2009
  35 c,4,5,01/03/2009
  36 """
  37     result = parser.read_csv(StringIO(data), converters={column: converter})

  40     expected["D"] = expected["D"].map(converter)
  41

  58 def test_converters_euro_decimal_format(all_parsers):
  59     # see gh-583
  60     converters = dict()
  61     parser = all_parsers
  62 
  63     data = """Id;Number1;Number2;Text1;Text2;Number3
  64 1;1521,1541;187101,9543;ABC;poi;4,7387
  65 2;121,12;14897,76;DEF;uyt;0,3773
  66 3;878,158;108013,434;GHI;rez;2,7356"""
  67     converters["Number1"] = converters["Number2"] = converters[
  68         "Number3"
  69     ] = lambda x: float(x.replace(",", "."))
  70 
  71     result = parser.read_csv(StringIO(data), sep=";", converters=converters)
  72     expected = DataFrame(
  73         [
  74             [1, 1521.1541, 187101.9543, "ABC", "poi", 4.7387],
  75             [2, 121.12, 14897.76, "DEF", "uyt", 0.3773],
  76             [3, 878.158, 108013.434, "GHI", "rez", 2.7356],
  77         ],
  78         columns=["Id", "Number1", "Number2", "Text1", "Text2", "Number3"],
  79     )
  80     tm.assert_frame_equal(result, expected)

  83 def test_converters_corner_with_nans(all_parsers):
  84     parser = all_parsers
  85     data = """id,score,days
  86 1,2,12
  87 2,2-5,
  88 3,,14+
  89 4,6-12,2"""
  90 
  91     # Example converters.
  92     def convert_days(x):
  93         x = x.strip()
  94 
  95         if not x:
  96             return np.nan
  97 
  98         is_plus = x.endswith("+")
  99 
 100         if is_plus:
 101             x = int(x[:-1]) + 1
 102         else:
 103             x = int(x)
 104 
 105         return x
 106 
 107     def convert_days_sentinel(x):
 108         x = x.strip()
 109 
 110         if not x:
 111             return np.nan
 112 
 113         is_plus = x.endswith("+")
 114 
 115         if is_plus:
 116             x = int(x[:-1]) + 1
 117         else:
 118             x = int(x)
 119 
 120         return x
 121 
 122     def convert_score(x):
 123         x = x.strip()
 124 
 125         if not x:
 126             return np.nan
 127 
 128         if x.find("-") > 0:
 129             val_min, val_max = map(int, x.split("-"))
 130             val = 0.5 * (val_min + val_max)
 131         else:
 132             val = float(x)
 133 
 134         return val
 135 
 136     results = []
 137 
 138     for day_converter in [convert_days, convert_days_sentinel]:
 139         result = parser.read_csv(
 140             StringIO(data),
 141             converters={"score": convert_score, "days": day_converter},
 142             na_values=["", None],
 143         )
 144         assert pd.isna(result["days"][1])
 145         results.append(result)
 146 
 147     tm.assert_frame_equal(results[0], results[1])
