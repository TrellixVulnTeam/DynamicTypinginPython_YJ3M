  26 def test_override_set_noconvert_columns():
  27     # see gh-17351
  28     #
  29     # Usecols needs to be sorted in _set_noconvert_columns based
  30     # on the test_usecols_with_parse_dates test from test_usecols.py
  31     class MyTextFileReader(TextFileReader):
  32         def __init__(self):
  33             self._currow = 0
  34             self.squeeze = False
  35 
  36     class MyCParserWrapper(CParserWrapper):
  37         def _set_noconvert_columns(self):
  38             if self.usecols_dtype == "integer":
  39                 # self.usecols is a set, which is documented as unordered
  40                 # but in practice, a CPython set of integers is sorted.
  41                 # In other implementations this assumption does not hold.
  42                 # The following code simulates a different order, which
  43                 # before GH 17351 would cause the wrong columns to be
  44                 # converted via the parse_dates parameter
  45                 self.usecols = list(self.usecols)
  46                 self.usecols.reverse()
  47             return CParserWrapper._set_noconvert_columns(self)
  48 
  49     data = """a,b,c,d,e
  50 0,1,20140101,0900,4
  51 0,1,20140102,1000,4"""
  52 
  53     parse_dates = [[1, 2]]
  54     cols = {
  55         "a": [0, 0],
  56         "c_d": [Timestamp("2014-01-01 09:00:00"), Timestamp("2014-01-02 10:00:00")],
  57     }
  58     expected = DataFrame(cols, columns=["c_d", "a"])
  59 
  60     parser = MyTextFileReader()
  61     parser.options = {
  62         "usecols": [0, 2, 3],
  63         "parse_dates": parse_dates,
  64         "delimiter": ",",
  65     }
  66     parser._engine = MyCParserWrapper(StringIO(data), **parser.options)
  67 
  68     result = parser.read()
  69     tm.assert_frame_equal(result, expected)

  85 def test_bad_stream_exception(all_parsers, csv_dir_path):
  86     # see gh-13652
  87     #
  88     # This test validates that both the Python engine and C engine will
  89     # raise UnicodeDecodeError instead of C engine raising ParserError
  90     # and swallowing the exception that caused read to fail.
  91     path = os.path.join(csv_dir_path, "sauron.SHIFT_JIS.csv")
  92     codec = codecs.lookup("utf-8")
  93     utf8 = codecs.lookup("utf-8")
  94     parser = all_parsers
  95     msg = "'utf-8' codec can't decode byte"
  96 
  97     # Stream must be binary UTF8.
  98     with open(path, "rb") as handle, codecs.StreamRecoder(
  99         handle, utf8.encode, utf8.decode, codec.streamreader, codec.streamwriter
 100     ) as stream:
 101 
 102         with pytest.raises(UnicodeDecodeError, match=msg):
 103             parser.read_csv(stream)
