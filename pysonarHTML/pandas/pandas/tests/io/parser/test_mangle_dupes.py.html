 114 def test_mangled_unnamed_placeholders(all_parsers):
 115     # xref gh-13017
 116     orig_key = "0"
 117     parser = all_parsers
 118 
 119     orig_value = [1, 2, 3]
 120     df = DataFrame({orig_key: orig_value})
 121 
 122     # This test recursively updates `df`.
 123     for i in range(3):
 124         expected = DataFrame()
 125 
 126         for j in range(i + 1):
 127             expected["Unnamed: 0" + ".1" * j] = [0, 1, 2]
 128 
 129         expected[orig_key] = orig_value
 130         df = parser.read_csv(StringIO(df.to_csv()))
 131 
 132         tm.assert_frame_equal(df, expected)
