 197     def test_numpy_string_dtype(self):
 198         data = """\
 199 a,1
 200 aa,2
 201 aaa,3
 202 aaaa,4
 203 aaaaa,5"""
 204 
 205         def _make_reader(**kwds):
 206             return TextReader(StringIO(data), delimiter=",", header=None, **kwds)
 207 
 208         reader = _make_reader(dtype="S5,i4")
 209         result = reader.read()
 210 
 211         assert result[0].dtype == "S5"
 212 
 213         ex_values = np.array(["a", "aa", "aaa", "aaaa", "aaaaa"], dtype="S5")
 214         assert (result[0] == ex_values).all()
 215         assert result[1].dtype == "i4"
