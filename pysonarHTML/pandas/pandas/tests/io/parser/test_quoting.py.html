  91 @pytest.mark.parametrize(
  92     "kwargs,exp_data",
  93     [
  94         (dict(), [[1, 2, "foo"]]),  # Test default.
  95         # QUOTE_MINIMAL only applies to CSV writing, so no effect on reading.
  96         (dict(quotechar='"', quoting=csv.QUOTE_MINIMAL), [[1, 2, "foo"]]),
  97         # QUOTE_MINIMAL only applies to CSV writing, so no effect on reading.
  98         (dict(quotechar='"', quoting=csv.QUOTE_ALL), [[1, 2, "foo"]]),
  99         # QUOTE_NONE tells the reader to do no special handling
 100         # of quote characters and leave them alone.
 101         (dict(quotechar='"', quoting=csv.QUOTE_NONE), [[1, 2, '"foo"']]),
 102         # QUOTE_NONNUMERIC tells the reader to cast
 103         # all non-quoted fields to float
 104         (dict(quotechar='"', quoting=csv.QUOTE_NONNUMERIC), [[1.0, 2.0, "foo"]]),
 105     ],
 106 )
 107 def test_quoting_various(all_parsers, kwargs, exp_data):
 108     data = '1,2,"foo"'
 109     parser = all_parsers
 110     columns = ["a", "b", "c"]
 111 
 112     result = parser.read_csv(StringIO(data), names=columns, **kwargs)
 113     expected = DataFrame(exp_data, columns=columns)
 114     tm.assert_frame_equal(result, expected)

 140 @pytest.mark.parametrize("balanced", [True, False])
 141 def test_unbalanced_quoting(all_parsers, balanced):
 142     # see gh-22789.
 143     parser = all_parsers
 144     data = 'a,b,c\n1,2,"3'
 145 
 146     if balanced:
 147         # Re-balance the quoting and read in without errors.
 148         expected = DataFrame([[1, 2, 3]], columns=["a", "b", "c"])
 149         result = parser.read_csv(StringIO(data + '"'))
 150         tm.assert_frame_equal(result, expected)
 151     else:
 152         msg = (
 153             "EOF inside string starting at row 1"
 154             if parser.engine == "c"
 155             else "unexpected end of data"
 156         )
 157 
 158         with pytest.raises(ParserError, match=msg):
 159             parser.read_csv(StringIO(data))
