  14 @pytest.mark.parametrize("na_values", [None, ["NaN"]])
  15 def test_comment(all_parsers, na_values):
  16     parser = all_parsers
  17     data = """A,B,C
  18 1,2.,4.#hello world
  19 5.,NaN,10.0
  20 """
  21     expected = DataFrame(
  22         [[1.0, 2.0, 4.0], [5.0, np.nan, 10.0]], columns=["A", "B", "C"]
  23     )
  24     result = parser.read_csv(StringIO(data), comment="#", na_values=na_values)
  25     tm.assert_frame_equal(result, expected)

  28 @pytest.mark.parametrize(
  29     "read_kwargs", [dict(), dict(lineterminator="*"), dict(delim_whitespace=True)]
  30 )
  31 def test_line_comment(all_parsers, read_kwargs):
  32     parser = all_parsers
  33     data = """# empty
  34 A,B,C
  35 1,2.,4.#hello world
  36 #ignore this line
  37 5.,NaN,10.0
  38 """
  39     if read_kwargs.get("delim_whitespace"):
  40         data = data.replace(",", " ")
  41     elif read_kwargs.get("lineterminator"):
  42         if parser.engine != "c":
  43             pytest.skip("Custom terminator not supported with Python engine")
  44 
  45         data = data.replace("\n", read_kwargs.get("lineterminator"))
  46 
  47     read_kwargs["comment"] = "#"
  48     result = parser.read_csv(StringIO(data), **read_kwargs)
  49 
  50     expected = DataFrame(
  51         [[1.0, 2.0, 4.0], [5.0, np.nan, 10.0]], columns=["A", "B", "C"]
  52     )
  53     tm.assert_frame_equal(result, expected)

  56 def test_comment_skiprows(all_parsers):
  57     parser = all_parsers
  58     data = """# empty
  59 random line
  60 # second empty line
  61 1,2,3
  62 A,B,C
  63 1,2.,4.
  64 5.,NaN,10.0
  65 """
  66     # This should ignore the first four lines (including comments).
  67     expected = DataFrame(
  68         [[1.0, 2.0, 4.0], [5.0, np.nan, 10.0]], columns=["A", "B", "C"]
  69     )
  70     result = parser.read_csv(StringIO(data), comment="#", skiprows=4)
  71     tm.assert_frame_equal(result, expected)

  91 def test_comment_skiprows_header(all_parsers):
  92     parser = all_parsers
  93     data = """# empty
  94 # second empty line
  95 # third empty line
  96 X,Y,Z
  97 1,2,3
  98 A,B,C
  99 1,2.,4.
 100 5.,NaN,10.0
 101 """
 102     # Skiprows should skip the first 4 lines (including comments),
 103     # while header should start from the second non-commented line,
 104     # starting with line 5.
 105     expected = DataFrame(
 106         [[1.0, 2.0, 4.0], [5.0, np.nan, 10.0]], columns=["A", "B", "C"]
 107     )
 108     result = parser.read_csv(StringIO(data), comment="#", skiprows=4, header=1)
 109     tm.assert_frame_equal(result, expected)

 112 @pytest.mark.parametrize("comment_char", ["#", "~", "&", "^", "*", "@"])
 113 def test_custom_comment_char(all_parsers, comment_char):
 114     parser = all_parsers
 115     data = "a,b,c\n1,2,3#ignore this!\n4,5,6#ignorethistoo"
 116     result = parser.read_csv(
 117         StringIO(data.replace("#", comment_char)), comment=comment_char
 118     )
 119 
 120     expected = DataFrame([[1, 2, 3], [4, 5, 6]], columns=["a", "b", "c"])
 121     tm.assert_frame_equal(result, expected)
