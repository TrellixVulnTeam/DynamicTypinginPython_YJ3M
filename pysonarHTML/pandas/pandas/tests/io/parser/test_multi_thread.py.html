  39 def test_multi_thread_string_io_read_csv(all_parsers):
  40     # see gh-11786
  41     parser = all_parsers
  42     max_row_range = 10000
  43     num_files = 100
  44 
  45     bytes_to_df = [
  46         "\n".join([f"{i:d},{i:d},{i:d}" for i in range(max_row_range)]).encode()
  47         for _ in range(num_files)
  48     ]
  49     files = [BytesIO(b) for b in bytes_to_df]
  50 
  51     # Read all files in many threads.
  52     pool = ThreadPool(8)
  53 
  54     results = pool.map(parser.read_csv, files)
  55     first_result = results[0]
  56 
  57     for result in results:
  58         tm.assert_frame_equal(first_result, result)

  61 def _generate_multi_thread_dataframe(parser, path, num_rows, num_tasks):
  62     """
  63     Generate a DataFrame via multi-thread.
  64 
  65     Parameters
  66     ----------
  67     parser : BaseParser
  68         The parser object to use for reading the data.
  69     path : str
  70         The location of the CSV file to read.
  71     num_rows : int
  72         The number of rows to read per task.
  73     num_tasks : int
  74         The number of tasks to use for reading this DataFrame.
  75 
  76     Returns
  77     -------
  78     df : DataFrame
  79     """
  80 
  81     def reader(arg):
  82         """
  83         Create a reader for part of the CSV.
  84 
  85         Parameters
  86         ----------
  87         arg : tuple
  88             A tuple of the following:
  89 
  90             * start : int
  91                 The starting row to start for parsing CSV
  92             * nrows : int
  93                 The number of rows to read.
  94 
  95         Returns
  96         -------
  97         df : DataFrame
  98         """
  99         start, nrows = arg
 100 
 101         if not start:
 102             return parser.read_csv(
 103                 path, index_col=0, header=0, nrows=nrows, parse_dates=["date"]
 104             )
 105 
 106         return parser.read_csv(
 107             path,
 108             index_col=0,
 109             header=None,
 110             skiprows=int(start) + 1,
 111             nrows=nrows,
 112             parse_dates=[9],
 113         )
 114 
 115     tasks = [
 116         (num_rows * i // num_tasks, num_rows // num_tasks) for i in range(num_tasks)
 117     ]
 118 
 119     pool = ThreadPool(processes=num_tasks)
 120     results = pool.map(reader, tasks)
 121 
 122     header = results[0].columns
 123 
 124     for r in results[1:]:
 125         r.columns = header
 126 
 127     final_dataframe = pd.concat(results)
 128     return final_dataframe
