 103 def test_compression_warning(compression_only):
 104     # Assert that passing a file object to to_csv while explicitly specifying a
 105     # compression protocol triggers a RuntimeWarning, as per GH21227.
 106     df = pd.DataFrame(
 107         100 * [[0.123456, 0.234567, 0.567567], [12.32112, 123123.2, 321321.2]],
 108         columns=["X", "Y", "Z"],
 109     )
 110     with tm.ensure_clean() as path:
 111         f, handles = icom.get_handle(path, "w", compression=compression_only)
 112         with tm.assert_produces_warning(RuntimeWarning, check_stacklevel=False):
 113             with f:
 114                 df.to_csv(f, compression=compression_only)
