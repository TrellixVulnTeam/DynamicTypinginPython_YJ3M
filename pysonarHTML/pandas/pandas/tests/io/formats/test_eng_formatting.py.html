  41     def compare_all(self, formatter, in_out):
  42         """
  43         Parameters:
  44         -----------
  45         formatter: EngFormatter under test
  46         in_out: list of tuples. Each tuple = (number, expected_formatting)
  47 
  48         It is tested if 'formatter(number) == expected_formatting'.
  49         *number* should be >= 0 because formatter(-number) == fmt is also
  50         tested. *fmt* is derived from *expected_formatting*
  51         """
  52         for input, output in in_out:
  53             self.compare(formatter, input, output)
  54             self.compare(formatter, -input, "-" + output[1:])

 172     def test_rounding(self):
 173         formatter = fmt.EngFormatter(accuracy=3, use_eng_prefix=True)
 174         in_out = [
 175             (5.55555, " 5.556"),
 176             (55.5555, " 55.556"),
 177             (555.555, " 555.555"),
 178             (5555.55, " 5.556k"),
 179             (55555.5, " 55.556k"),
 180             (555555, " 555.555k"),
 181         ]
 182         self.compare_all(formatter, in_out)
 183 
 184         formatter = fmt.EngFormatter(accuracy=1, use_eng_prefix=True)
 185         in_out = [
 186             (5.55555, " 5.6"),
 187             (55.5555, " 55.6"),
 188             (555.555, " 555.6"),
 189             (5555.55, " 5.6k"),
 190             (55555.5, " 55.6k"),
 191             (555555, " 555.6k"),
 192         ]
 193         self.compare_all(formatter, in_out)
 194 
 195         formatter = fmt.EngFormatter(accuracy=0, use_eng_prefix=True)
 196         in_out = [
 197             (5.55555, " 6"),
 198             (55.5555, " 56"),
 199             (555.555, " 556"),
 200             (5555.55, " 6k"),
 201             (55555.5, " 56k"),
 202             (555555, " 556k"),
 203         ]
 204         self.compare_all(formatter, in_out)
 205 
 206         formatter = fmt.EngFormatter(accuracy=3, use_eng_prefix=True)
 207         result = formatter(0)
 208         assert result == " 0.000"

 210     def test_nan(self):
 211         # Issue #11981
 212 
 213         formatter = fmt.EngFormatter(accuracy=1, use_eng_prefix=True)
 214         result = formatter(np.nan)
 215         assert result == "NaN"
 216 
 217         df = pd.DataFrame(
 218             {
 219                 "a": [1.5, 10.3, 20.5],
 220                 "b": [50.3, 60.67, 70.12],
 221                 "c": [100.2, 101.33, 120.33],
 222             }
 223         )
 224         pt = df.pivot_table(values="a", index="b", columns="c")
 225         fmt.set_eng_float_format(accuracy=1)
 226         result = pt.to_string()
 227         assert "NaN" in result
 228         tm.reset_display_options()
