 104 def test_info_memory():
 105     # https://github.com/pandas-dev/pandas/issues/21056
 106     df = DataFrame({"a": Series([1, 2], dtype="i8")})
 107     buf = StringIO()
 108     df.info(buf=buf)
 109     result = buf.getvalue()
 110     bytes = float(df.memory_usage().sum())
 111     expected = textwrap.dedent(
 112         f"""\
 113     <class 'pandas.core.frame.DataFrame'>
 114     RangeIndex: 2 entries, 0 to 1
 115     Data columns (total 1 columns):
 116      #   Column  Non-Null Count  Dtype
 117     ---  ------  --------------  -----
 118      0   a       2 non-null      int64
 119     dtypes: int64(1)
 120     memory usage: {bytes} bytes
 121     """
 122     )
 123     assert result == expected

 192 def test_info_max_cols():
 193     df = DataFrame(np.random.randn(10, 5))
 194     for len_, verbose in [(5, None), (5, False), (12, True)]:
 195         # For verbose always      ^ setting  ^ summarize ^ full output
 196         with option_context("max_info_columns", 4):
 197             buf = StringIO()
 198             df.info(buf=buf, verbose=verbose)
 199             res = buf.getvalue()
 200             assert len(res.strip().split("\n")) == len_
 201 
 202     for len_, verbose in [(12, None), (5, False), (12, True)]:
 203 
 204         # max_cols not exceeded
 205         with option_context("max_info_columns", 5):
 206             buf = StringIO()
 207             df.info(buf=buf, verbose=verbose)
 208             res = buf.getvalue()
 209             assert len(res.strip().split("\n")) == len_
 210 
 211     for len_, max_cols in [(12, 5), (5, 4)]:
 212         # setting truncates
 213         with option_context("max_info_columns", 4):
 214             buf = StringIO()
 215             df.info(buf=buf, max_cols=max_cols)
 216             res = buf.getvalue()
 217             assert len(res.strip().split("\n")) == len_
 218 
 219         # setting wouldn't truncate
 220         with option_context("max_info_columns", 5):
 221             buf = StringIO()
 222             df.info(buf=buf, max_cols=max_cols)

 227 def test_info_memory_usage():
 228     # Ensure memory usage is displayed, when asserted, on the last line
 229     dtypes = [
 230         "int64",
 231         "float64",
 232         "datetime64[ns]",
 233         "timedelta64[ns]",
 234         "complex128",
 235         "object",
 236         "bool",
 237     ]
 238     data = {}
 239     n = 10
 240     for i, dtype in enumerate(dtypes):
 241         data[i] = np.random.randint(2, size=n).astype(dtype)
 242     df = DataFrame(data)
 243     buf = StringIO()
 244 
 245     # display memory usage case
 246     df.info(buf=buf, memory_usage=True)
 247     res = buf.getvalue().splitlines()
 248     assert "memory usage: " in res[-1]
 249 
 250     # do not display memory usage case
 251     df.info(buf=buf, memory_usage=False)
 252     res = buf.getvalue().splitlines()
 253     assert "memory usage: " not in res[-1]
 254 
 255     df.info(buf=buf, memory_usage=True)
 256     res = buf.getvalue().splitlines()
 257 
 258     # memory usage is a lower bound, so print it as XYZ+ MB
 259     assert re.match(r"memory usage: [^+]+\+", res[-1])
 260 
 261     df.iloc[:, :5].info(buf=buf, memory_usage=True)
 262     res = buf.getvalue().splitlines()
 263 
 264     # excluded column with object dtype, so estimate is accurate
 265     assert not re.match(r"memory usage: [^+]+\+", res[-1])
 266 
 267     # Test a DataFrame with duplicate columns
 268     dtypes = ["int64", "int64", "int64", "float64"]
 269     data = {}
 270     n = 100
 271     for i, dtype in enumerate(dtypes):
 272         data[i] = np.random.randint(2, size=n).astype(dtype)
 273     df = DataFrame(data)
 274     df.columns = dtypes
 275 
 276     df_with_object_index = DataFrame({"a": [1]}, index=["foo"])
 277     df_with_object_index.info(buf=buf, memory_usage=True)
 278     res = buf.getvalue().splitlines()
 279     assert re.match(r"memory usage: [^+]+\+", res[-1])
 280 
 281     df_with_object_index.info(buf=buf, memory_usage="deep")
 282     res = buf.getvalue().splitlines()
 283     assert re.match(r"memory usage: [^+]+$", res[-1])
 284 
 285     # Ensure df size is as expected
 286     # (cols * rows * bytes) + index size
 287     df_size = df.memory_usage().sum()
 288     exp_size = len(dtypes) * n * 8 + df.index.nbytes
 289     assert df_size == exp_size
 290 
 291     # Ensure number of cols in memory_usage is the same as df
 292     size_df = np.size(df.columns.values) + 1  # index=True; default
 293     assert size_df == np.size(df.memory_usage())
 294 
 295     # assert deep works only on object
 296     assert df.memory_usage().sum() == df.memory_usage(deep=True).sum()
 297 
 298     # test for validity
 299     DataFrame(1, index=["a"], columns=["A"]).memory_usage(index=True)
 300     DataFrame(1, index=["a"], columns=["A"]).index.nbytes
 301     df = DataFrame(
 302         data=1, index=MultiIndex.from_product([["a"], range(1000)]), columns=["A"],
 303     )
 304     df.index.nbytes
 305     df.memory_usage(index=True)
 306     df.index.values.nbytes
 307 
 308     mem = df.memory_usage(deep=True).sum()
 309     assert mem > 0

 336 @pytest.mark.skipif(PYPY, reason="PyPy getsizeof() fails by design")
 337 def test_usage_via_getsizeof():
 338     df = DataFrame(
 339         data=1, index=MultiIndex.from_product([["a"], range(1000)]), columns=["A"],
 340     )
 341     mem = df.memory_usage(deep=True).sum()
 342     # sys.getsizeof will call the .memory_usage with

 377 def test_info_memory_usage_bug_on_multiindex():
 378     # GH 14308
 379     # memory usage introspection should not materialize .values
 380 
 381     def memory_usage(f):
 382         return f.memory_usage(deep=True).sum()
 383 
 384     N = 100
 385     M = len(uppercase)
 386     index = MultiIndex.from_product(
 387         [list(uppercase), date_range("20160101", periods=N)], names=["id", "date"],
 388     )
 389     df = DataFrame({"value": np.random.randn(N * M)}, index=index)
 390 
 391     unstacked = df.unstack("id")
 392     assert df.values.nbytes == unstacked.values.nbytes
 393     assert memory_usage(df) > memory_usage(unstacked)
 394 
 395     # high upper bound
 396     assert memory_usage(unstacked) - memory_usage(df) < 2000
