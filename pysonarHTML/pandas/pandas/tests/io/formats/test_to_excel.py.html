 213 def test_css_to_excel_multiple():
 214     convert = CSSToExcelConverter()
 215     actual = convert(
 216         """
 217         font-weight: bold;
 218         text-decoration: underline;
 219         color: red;
 220         border-width: thin;
 221         text-align: center;
 222         vertical-align: top;
 223         unused: something;
 224     """
 225     )
 226     assert {
 227         "font": {"bold": True, "underline": "single", "color": "FF0000"},
 228         "border": {
 229             "top": {"style": "thin"},
 230             "right": {"style": "thin"},
 231             "bottom": {"style": "thin"},
 232             "left": {"style": "thin"},
 233         },
 234         "alignment": {"horizontal": "center", "vertical": "top"},
 235     } == actual

 262 @pytest.mark.parametrize(
 263     "input_color,output_color",
 264     (
 265         list(CSSToExcelConverter.NAMED_COLORS.items())
 266         + [("#" + rgb, rgb) for rgb in CSSToExcelConverter.NAMED_COLORS.values()]
 267         + [("#F0F", "FF00FF"), ("#ABC", "AABBCC")]
 268     ),
 269 )
 270 def test_css_to_excel_good_colors(input_color, output_color):
 271     # see gh-18392
 272     css = (
 273         f"border-top-color: {input_color}; "
 274         f"border-right-color: {input_color}; "
 275         f"border-bottom-color: {input_color}; "
 276         f"border-left-color: {input_color}; "
 277         f"background-color: {input_color}; "
 278         f"color: {input_color}"
 279     )
 280 
 281     expected = dict()
 282 
 283     expected["fill"] = {"patternType": "solid", "fgColor": output_color}
 284 
 285     expected["font"] = {"color": output_color}
 286 
 287     expected["border"] = {
 288         k: {"color": output_color} for k in ("top", "right", "bottom", "left")
 289     }
 290 
 291     with tm.assert_produces_warning(None):
 292         convert = CSSToExcelConverter()
 293         assert expected == convert(css)

 296 @pytest.mark.parametrize("input_color", [None, "not-a-color"])
 297 def test_css_to_excel_bad_colors(input_color):
 298     # see gh-18392
 299     css = (
 300         f"border-top-color: {input_color}; "
 301         f"border-right-color: {input_color}; "
 302         f"border-bottom-color: {input_color}; "
 303         f"border-left-color: {input_color}; "
 304         f"background-color: {input_color}; "
 305         f"color: {input_color}"
 306     )
 307 
 308     expected = dict()
 309 
 310     if input_color is not None:
 311         expected["fill"] = {"patternType": "solid"}
 312 
 313     with tm.assert_produces_warning(CSSWarning):
 314         convert = CSSToExcelConverter()
 315         assert expected == convert(css)
