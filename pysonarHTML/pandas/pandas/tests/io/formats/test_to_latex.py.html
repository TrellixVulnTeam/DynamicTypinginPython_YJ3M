 242     def test_to_latex_multiindex_dupe_level(self):
 243         # see gh-14484
 244         #
 245         # If an index is repeated in subsequent rows, it should be
 246         # replaced with a blank in the created table. This should
 247         # ONLY happen if all higher order indices (to the left) are

 255 \toprule
 256   &   &  col \\
 257 \midrule
 258 A & c &  NaN \\
 259 B & c &  NaN \\
 260 \bottomrule
 261 \end{tabular}
 262 """
 263         assert result == expected
