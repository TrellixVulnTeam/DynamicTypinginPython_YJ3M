 131     def test_publishes(self):
 132 
 133         df = pd.DataFrame({"A": [1, 2]})
 134         objects = [df["A"], df, df]  # dataframe / series
 135         expected_keys = [
 136             {"text/plain", "application/vnd.dataresource+json"},
 137             {"text/plain", "text/html", "application/vnd.dataresource+json"},
 138         ]
 139 
 140         opt = pd.option_context("display.html.table_schema", True)
 141         for obj, expected in zip(objects, expected_keys):
 142             with opt:
 143                 formatted = self.display_formatter.format(obj)
 144             assert set(formatted[0].keys()) == expected
 145 
 146         with_latex = pd.option_context("display.latex.repr", True)
 147 
 148         with opt, with_latex:
 149             formatted = self.display_formatter.format(obj)
 150 
 151         expected = {
 152             "text/plain",
 153             "text/html",
 154             "text/latex",
 155             "application/vnd.dataresource+json",
 156         }
 157         assert set(formatted[0].keys()) == expected
