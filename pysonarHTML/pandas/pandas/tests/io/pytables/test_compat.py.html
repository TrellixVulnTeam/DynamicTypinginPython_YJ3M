  44 class TestReadPyTablesHDF5:
  45     """
  46     A group of tests which covers reading HDF5 files written by plain PyTables
  47     (not written by pandas).
  48 
  49     Was introduced for regression-testing issue 11188.
  50     """
  51 
  52     def test_read_complete(self, pytables_hdf5_file):
  53         path, objname, df = pytables_hdf5_file
  54         result = pd.read_hdf(path, key=objname)
  55         expected = df
  56         tm.assert_frame_equal(result, expected)
  57 
  58     def test_read_with_start(self, pytables_hdf5_file):
  59         path, objname, df = pytables_hdf5_file
  60         # This is a regression test for pandas-dev/pandas/issues/11188
  61         result = pd.read_hdf(path, key=objname, start=1)
  62         expected = df[1:].reset_index(drop=True)
  63         tm.assert_frame_equal(result, expected)
  64
