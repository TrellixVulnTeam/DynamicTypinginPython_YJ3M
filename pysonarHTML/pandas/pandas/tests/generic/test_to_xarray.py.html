  13     def test_to_xarray_index_types(self, indices):
  14         if isinstance(indices, pd.MultiIndex):
  15             pytest.skip("MultiIndex is tested separately")
  16         if len(indices) == 0:
  17             pytest.skip("Test doesn't make sense for empty index")
  18 
  19         from xarray import Dataset

  22             {
  23                 "a": list("abc"),
