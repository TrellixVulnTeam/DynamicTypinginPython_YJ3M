 113         def finalize(self, other, method=None, **kwargs):
 114 
 115             for name in self._metadata:
 116                 if method == "merge":
 117                     left, right = other.left, other.right
 118                     value = getattr(left, name, "") + "|" + getattr(right, name, "")
 119                     object.__setattr__(self, name, value)

 121                     object.__setattr__(self, name, getattr(other, name, ""))
 122 
 123             return self
