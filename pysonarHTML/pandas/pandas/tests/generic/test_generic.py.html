  28     def _construct(self, shape, value=None, dtype=None, **kwargs):
  29         """
  30         construct an object for the given shape
  31         if value is specified use that if its a scalar
  32         if value is an array, repeat it as needed
  33         """
  34         if isinstance(shape, int):
  35             shape = tuple([shape] * self._ndim)
  36         if value is not None:
  37             if is_scalar(value):
  38                 if value == "empty":
  39                     arr = None
  40                     dtype = np.float64
  41 
  42                     # remove the info axis
  43                     kwargs.pop(self._typ._info_axis_name, None)
  44                 else:
  45                     arr = np.empty(shape, dtype=dtype)
  46                     arr.fill(value)
  47             else:
  48                 fshape = np.prod(shape)
  49                 arr = value.ravel()
  50                 new_shape = fshape / arr.shape[0]
  51                 if fshape % arr.shape[0] != 0:
  52                     raise Exception("invalid value passed in _construct")
  53 
  54                 arr = np.repeat(arr, new_shape).reshape(shape)
  55         else:
  56             arr = np.random.randn(*shape)
  57         return self._typ(arr, dtype=dtype, **kwargs)

 203     def test_metadata_propagation(self):
 204         # check that the metadata matches up on the resulting ops
 205 
 206         o = self._construct(shape=3)
 207         o.name = "foo"
 208         o2 = self._construct(shape=3)
 209         o2.name = "bar"
 210 
 211         # ----------
 212         # preserving
 213         # ----------
 214 
 215         # simple ops with scalars
 216         for op in ["__add__", "__sub__", "__truediv__", "__mul__"]:
 217             result = getattr(o, op)(1)
 218             self.check_metadata(o, result)
 219 
 220         # ops with like
 221         for op in ["__add__", "__sub__", "__truediv__", "__mul__"]:
 222             result = getattr(o, op)(o)
 223             self.check_metadata(o, result)
 224 
 225         # simple boolean
 226         for op in ["__eq__", "__le__", "__ge__"]:
 227             v1 = getattr(o, op)(o)
 228             self.check_metadata(o, v1)
 229             self.check_metadata(o, v1 & v1)
 230             self.check_metadata(o, v1 | v1)
 231 
 232         # combine_first
 233         result = o.combine_first(o2)
 234         self.check_metadata(o, result)
 235 
 236         # ---------------------------
 237         # non-preserving (by default)
 238         # ---------------------------
 239 
 240         # add non-like
 241         result = o + o2
 242         self.check_metadata(result)
 243 
 244         # simple boolean
 245         for op in ["__eq__", "__le__", "__ge__"]:
 246 
 247             # this is a name matching op
 248             v1 = getattr(o, op)(o)
 249             v2 = getattr(o, op)(o2)
 250             self.check_metadata(v2)
 251             self.check_metadata(v1 & v2)

 281 
 282         o = self._construct(shape=10)
 283 
 284         ###

 287 
 288         # Check for stability when receives seed or random state -- run 10

 290         for test in range(10):

 292             self._compare(
 293                 o.sample(n=4, random_state=seed), o.sample(n=4, random_state=seed)

 295 
 296             self._compare(
 297                 o.sample(frac=0.7, random_state=seed),
 298                 o.sample(frac=0.7, random_state=seed),
 299             )
 300 
 301             self._compare(
 302                 o.sample(n=4, random_state=np.random.RandomState(test)),
 303                 o.sample(n=4, random_state=np.random.RandomState(test)),
 304             )
 305 
 306             self._compare(
 307                 o.sample(frac=0.7, random_state=np.random.RandomState(test)),
 308                 o.sample(frac=0.7, random_state=np.random.RandomState(test)),
 309             )
 310 
 311             self._compare(
 312                 o.sample(
 313                     frac=2, replace=True, random_state=np.random.RandomState(test)
 314                 ),
 315                 o.sample(
 316                     frac=2, replace=True, random_state=np.random.RandomState(test)
 317                 ),
 318             )
 319 
 320             os1, os2 = [], []
 321             for _ in range(2):
 322                 np.random.seed(test)
 323                 os1.append(o.sample(n=4))
 324                 os2.append(o.sample(frac=0.7))
 325             self._compare(*os1)
 326             self._compare(*os2)
 327 
 328         # Check for error when random_state argument invalid.
 329         with pytest.raises(ValueError):
 330             o.sample(random_state="astring!")
 331 
 332         ###

 334         ###

 336         # Giving both frac and N throws error
 337         with pytest.raises(ValueError):
 338             o.sample(n=3, frac=0.3)
 339

 341         with pytest.raises(ValueError):
 342             o.sample(n=-3)
 343         with pytest.raises(ValueError):
 344             o.sample(frac=-0.3)
 345

 347         with pytest.raises(ValueError):
 348             o.sample(n=3.2)
