  70     def test_nonzero_single_element(self):
  71 
  72         # allow single item via bool method
  73         s = Series([True])
  74         assert s.bool()
  75 
  76         s = Series([False])
  77         assert not s.bool()
  78 
  79         msg = "The truth value of a Series is ambiguous"
  80         # single item nan to raise
  81         for s in [Series([np.nan]), Series([pd.NaT]), Series([True]), Series([False])]:
  82             with pytest.raises(ValueError, match=msg):
  83                 bool(s)
  84 
  85         msg = "bool cannot act on a non-boolean single element Series"
  86         for s in [Series([np.nan]), Series([pd.NaT])]:
  87             with pytest.raises(ValueError, match=msg):
  88                 s.bool()
  89 
  90         # multiple bool are still an error
  91         msg = "The truth value of a Series is ambiguous"
  92         for s in [Series([True, True]), Series([False, False])]:
  93             with pytest.raises(ValueError, match=msg):
  94                 bool(s)
  95             with pytest.raises(ValueError, match=msg):
  96                 s.bool()
  97 
  98         # single non-bool are an error
  99         for s in [Series([1]), Series([0]), Series(["a"]), Series([0.0])]:
 100             msg = "The truth value of a Series is ambiguous"
 101             with pytest.raises(ValueError, match=msg):
 102                 bool(s)
 103             msg = "bool cannot act on a non-boolean single element Series"
 104             with pytest.raises(ValueError, match=msg):
 105                 s.bool()

 107     def test_metadata_propagation_indiv(self):
 108         # check that the metadata matches up on the resulting ops
 109 
 110         o = Series(range(3), range(3))
 111         o.name = "foo"
 112         o2 = Series(range(3), range(3))
 113         o2.name = "bar"
 114 
 115         result = o.T
 116         self.check_metadata(o, result)
 117 
 118         # resample
 119         ts = Series(
 120             np.random.rand(1000),
 121             index=date_range("20130101", periods=1000, freq="s"),
 122             name="foo",
 123         )
 124         result = ts.resample("1T").mean()
 125         self.check_metadata(ts, result)
 126 
 127         result = ts.resample("1T").min()
 128         self.check_metadata(ts, result)
 129 
 130         result = ts.resample("1T").apply(lambda x: x.sum())
 131         self.check_metadata(ts, result)
 132 
 133         _metadata = Series._metadata
 134         _finalize = Series.__finalize__
 135         Series._metadata = ["name", "filename"]
 136         o.filename = "foo"
 137         o2.filename = "bar"
 138 
 139         def finalize(self, other, method=None, **kwargs):
 140             for name in self._metadata:
 141                 if method == "concat" and name == "filename":
 142                     value = "+".join(
 143                         [getattr(o, name) for o in other.objs if getattr(o, name, None)]
 144                     )
 145                     object.__setattr__(self, name, value)
 146                 else:
 147                     object.__setattr__(self, name, getattr(other, name, None))
 148 
 149             return self
 150 
 151         Series.__finalize__ = finalize
 152 
 153         result = pd.concat([o, o2])
 154         assert result.filename == "foo+bar"
 155         assert result.name is None
 156 
 157         # reset
 158         Series._metadata = _metadata
 159         Series.__finalize__ = _finalize  # FIXME: use monkeypatch
