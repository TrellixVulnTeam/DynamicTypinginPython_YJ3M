 138             lambda x, y: x.ewm(com=3).corr(y, pairwise=False),
 139         ],
 140     )
 141     def test_no_pairwise_with_other(self, f):
 142 
 143         # DataFrame with another DataFrame, pairwise=False
 144         results = [
 145             f(df, self.df2) if df.columns.is_unique else None for df in self.df1s
 146         ]
 147         for (df, result) in zip(self.df1s, results):
 148             if result is not None:
 149                 with warnings.catch_warnings(record=True):
 150                     warnings.simplefilter("ignore", RuntimeWarning)
 151                     # we can have int and str columns
 152                     expected_index = df.index.union(self.df2.index)
 153                     expected_columns = df.columns.union(self.df2.columns)
 154                 tm.assert_index_equal(result.index, expected_index)
 155                 tm.assert_index_equal(result.columns, expected_columns)
 156             else:
 157                 with pytest.raises(ValueError, match="'arg1' columns are not unique"):
 158                     f(df, self.df2)
 159                 with pytest.raises(ValueError, match="'arg2' columns are not unique"):
 160                     f(self.df2, df)
