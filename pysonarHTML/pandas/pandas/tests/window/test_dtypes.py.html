 111     def _create_dtype_data(self, dtype):
 112         sr1 = Series(np.arange(5), dtype=dtype)
 113         sr2 = Series(np.arange(10, 0, -2), dtype=dtype)
 114         sr3 = sr1.copy()
 115         sr3[3] = np.NaN
 116         df = DataFrame(np.arange(10).reshape((5, 2)), dtype=dtype)
 117 
 118         data = {"sr1": sr1, "sr2": sr2, "sr3": sr3, "df": df}
 119 
 120         return data

 122     def _create_data(self):
 123         self.data = self._create_dtype_data(self.dtype)
 124         self.expects = self.get_expects()

 126     def test_dtypes(self):
 127         self._create_data()
 128         for f_name, d_name in product(self.funcs.keys(), self.data.keys()):
 129 
 130             f = self.funcs[f_name]
 131             d = self.data[d_name]
 132             exp = self.expects[d_name][f_name]
 133             self.check_dtypes(f, f_name, d, d_name, exp)
 134 
 135     def check_dtypes(self, f, f_name, d, d_name, exp):
 136         roll = d.rolling(window=self.window)
 137         result = f(roll)
 138         tm.assert_almost_equal(result, exp)

 214 class DatetimeLike(Dtype):

 216 
 217         roll = d.rolling(window=self.window)
 218         if f_name == "count":
 219             result = f(roll)
 220             tm.assert_almost_equal(result, exp)

 222         else:
 223             with pytest.raises(DataError):
