  28 def test_indexer_constructor_arg():

  30     use_expanding = [True, False, True, False, True]
  31     df = DataFrame({"values": range(5)})

  33     class CustomIndexer(BaseIndexer):
  34         def get_window_bounds(self, num_values, min_periods, center, closed):
  35             start = np.empty(num_values, dtype=np.int64)
  36             end = np.empty(num_values, dtype=np.int64)
  37             for i in range(num_values):
  38                 if self.use_expanding[i]:
  39                     start[i] = 0

  55     class CustomIndexer(BaseIndexer):
  56         def get_window_bounds(self, num_values, min_periods, center, closed):
  57             start = np.empty(num_values, dtype=np.int64)
  58             end = np.empty(num_values, dtype=np.int64)
  59             for i in range(num_values):
  60                 if center and min_periods == 1 and closed == "both" and i == 2:
  61                     start[i] = 0
  62                     end[i] = num_values

  64                     start[i] = i
  65                     end[i] = i + self.window_size

 139 def test_rolling_forward_window(constructor, func, np_func, expected, np_kwargs):
 140     # GH 32865
 141     values = np.arange(10)
 142     values[5] = 100.0
 143 
 144     indexer = FixedForwardWindowIndexer(window_size=3)
 145 
 146     match = "Forward-looking windows can't have center=True"
 147     with pytest.raises(ValueError, match=match):
 148         rolling = constructor(values).rolling(window=indexer, center=True)
 149         result = getattr(rolling, func)()
 150 
 151     match = "Forward-looking windows don't support setting the closed argument"
 152     with pytest.raises(ValueError, match=match):
 153         rolling = constructor(values).rolling(window=indexer, closed="right")
 154         result = getattr(rolling, func)()
 155 
 156     rolling = constructor(values).rolling(window=indexer, min_periods=2)
 157     result = getattr(rolling, func)()
 158 
 159     # Check that the function output matches the explicitly provided array
 160     expected = constructor(expected)
 161     tm.assert_equal(result, expected)
 162 
 163     # Check that the rolling function output matches applying an alternative
 164     # function to the rolling window object
 165     expected2 = constructor(rolling.apply(lambda x: np_func(x, **np_kwargs)))
 166     tm.assert_equal(result, expected2)
 167 
 168     # Check that the function output matches applying an alternative function
 169     # if min_periods isn't specified
 170     rolling3 = constructor(values).rolling(window=indexer)
 171     result3 = getattr(rolling3, func)()
 172     expected3 = constructor(rolling3.apply(lambda x: np_func(x, **np_kwargs)))
 173     tm.assert_equal(result3, expected3)
