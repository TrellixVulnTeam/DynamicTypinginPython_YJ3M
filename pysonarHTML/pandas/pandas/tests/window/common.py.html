  60 def moments_consistency_is_constant(x, is_constant, min_periods, count, mean, corr):
  61     count_x = count(x)
  62     mean_x = mean(x)
  63     # check that correlation of a series with itself is either 1 or NaN
  64     corr_x_x = corr(x, x)
  65 
  66     if is_constant:
  67         exp = x.max() if isinstance(x, Series) else x.max().max()
  68

  90 def moments_consistency_var_data(
  91     x, is_constant, min_periods, count, mean, var_unbiased, var_biased
  92 ):
  93     count_x = count(x)
  94     mean_x = mean(x)
  95     for var in [var_biased, var_unbiased]:
  96         var_x = var(x)
  97         assert not (var_x < 0).any().any()
  98 
  99         if var is var_biased:
 100             # check that biased var(x) == mean(x^2) - mean(x)^2
 101             mean_x2 = mean(x * x)
 102             tm.assert_equal(var_x, mean_x2 - (mean_x * mean_x))
 103 
 104         if is_constant:
 105             # check that variance of constant series is identically 0
 106             assert not (var_x > 0).any().any()
 107             expected = x * np.nan
 108             expected[count_x >= max(min_periods, 1)] = 0.0
 109             if var is var_unbiased:
 110                 expected[count_x < 2] = np.nan
 111             tm.assert_equal(var_x, expected)

 137 def moments_consistency_series_data(
 138     x,
 139     corr,
 140     mean,
 141     std_biased,
 142     std_unbiased,
 143     cov_unbiased,
 144     var_unbiased,
 145     var_biased,
 146     cov_biased,
 147 ):
 148     if isinstance(x, Series):
 149         y = x
 150         mean_x = mean(x)
 151         if not x.isna().equals(y.isna()):
 152             # can only easily test two Series with similar
 153             # structure
 154             pass
 155 
 156         # check that cor(x, y) is symmetric
 157         corr_x_y = corr(x, y)
 158         corr_y_x = corr(y, x)
 159         tm.assert_equal(corr_x_y, corr_y_x)
 160 
 161         for (std, var, cov) in [
 162             (std_biased, var_biased, cov_biased),
 163             (std_unbiased, var_unbiased, cov_unbiased),
 164         ]:
 165             var_x = var(x)
 166             std_x = std(x)
 167 
 168             if cov:
 169                 # check that cov(x, y) is symmetric
 170                 cov_x_y = cov(x, y)
 171                 cov_y_x = cov(y, x)
 172                 tm.assert_equal(cov_x_y, cov_y_x)
 173 
 174                 # check that cov(x, y) == (var(x+y) - var(x) -
 175                 # var(y)) / 2
 176                 var_x_plus_y = var(x + y)
 177                 var_y = var(y)
 178                 tm.assert_equal(cov_x_y, 0.5 * (var_x_plus_y - var_x - var_y))
 179 
 180                 # check that corr(x, y) == cov(x, y) / (std(x) *
 181                 # std(y))
 182                 std_y = std(y)
 183                 tm.assert_equal(corr_x_y, cov_x_y / (std_x * std_y))
 184 
 185                 if cov is cov_biased:
 186                     # check that biased cov(x, y) == mean(x*y) -
 187                     # mean(x)*mean(y)
 188                     mean_y = mean(y)
 189                     mean_x_times_y = mean(x * y)
 190                     tm.assert_equal(cov_x_y, mean_x_times_y - (mean_x * mean_y))
