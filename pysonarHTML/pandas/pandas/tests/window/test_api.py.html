  73 def test_agg():
  74     df = DataFrame({"A": range(5), "B": range(0, 10, 2)})
  75 
  76     r = df.rolling(window=3)
  77     a_mean = r["A"].mean()
  78     a_std = r["A"].std()
  79     a_sum = r["A"].sum()
  80     b_mean = r["B"].mean()
  81     b_std = r["B"].std()
  82 
  83     result = r.aggregate([np.mean, np.std])
  84     expected = concat([a_mean, a_std, b_mean, b_std], axis=1)
  85     expected.columns = pd.MultiIndex.from_product([["A", "B"], ["mean", "std"]])
  86     tm.assert_frame_equal(result, expected)
  87 
  88     result = r.aggregate({"A": np.mean, "B": np.std})
  89 
  90     expected = concat([a_mean, b_std], axis=1)
  91     tm.assert_frame_equal(result, expected, check_like=True)
  92 
  93     result = r.aggregate({"A": ["mean", "std"]})
  94     expected = concat([a_mean, a_std], axis=1)
  95     expected.columns = pd.MultiIndex.from_tuples([("A", "mean"), ("A", "std")])
  96     tm.assert_frame_equal(result, expected)
  97 
  98     result = r["A"].aggregate(["mean", "sum"])
  99     expected = concat([a_mean, a_sum], axis=1)
 100     expected.columns = ["mean", "sum"]
 101     tm.assert_frame_equal(result, expected)
 102 
 103     msg = "nested renamer is not supported"
 104     with pytest.raises(SpecificationError, match=msg):
 105         # using a dict with renaming
 106         r.aggregate({"A": {"mean": "mean", "sum": "sum"}})
 107 
 108     with pytest.raises(SpecificationError, match=msg):
 109         r.aggregate(
 110             {
 111                 "A": {"mean": "mean", "sum": "sum"},
 112                 "B": {"mean2": "mean", "sum2": "sum"},
 113             }
 114         )
 115 
 116     result = r.aggregate({"A": ["mean", "std"], "B": ["mean", "std"]})
 117     expected = concat([a_mean, a_std, b_mean, b_std], axis=1)
 118 
 119     exp_cols = [("A", "mean"), ("A", "std"), ("B", "mean"), ("B", "std")]
 120     expected.columns = pd.MultiIndex.from_tuples(exp_cols)
 121     tm.assert_frame_equal(result, expected, check_like=True)

 315 def test_multiple_agg_funcs(func, window_size, expected_vals):
 316     # GH 15072
 317     df = pd.DataFrame(
 318         [
 319             ["A", 10, 20],
 320             ["A", 20, 30],
 321             ["A", 30, 40],
 322             ["B", 10, 30],
 323             ["B", 30, 40],
 324             ["B", 40, 80],
 325             ["B", 80, 90],
 326         ],
 327         columns=["stock", "low", "high"],
 328     )
 329 
 330     f = getattr(df.groupby("stock"), func)
 331     if window_size:
 332         window = f(window_size)
 333     else:
 334         window = f()
 335 
 336     index = pd.MultiIndex.from_tuples(
 337         [("A", 0), ("A", 1), ("A", 2), ("B", 3), ("B", 4), ("B", 5), ("B", 6)],
 338         names=["stock", None],
 339     )
 340     columns = pd.MultiIndex.from_tuples(
 341         [("low", "mean"), ("low", "max"), ("high", "mean"), ("high", "min")]
 342     )
 343     expected = pd.DataFrame(expected_vals, index=index, columns=columns)
 344 
 345     result = window.agg(
 346         OrderedDict((("low", ["mean", "max"]), ("high", ["mean", "min"])))
 347     )
 348 
 349     tm.assert_frame_equal(result, expected)
