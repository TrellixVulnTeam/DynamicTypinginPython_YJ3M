 166 def test_ewm_domain_checks(arr):
 167     # GH 12492
 168     s = Series(arr)
 169     msg = "comass must satisfy: comass >= 0"
 170     with pytest.raises(ValueError, match=msg):
 171         s.ewm(com=-0.1)
 172     s.ewm(com=0.0)
 173     s.ewm(com=0.1)
 174 
 175     msg = "span must satisfy: span >= 1"
 176     with pytest.raises(ValueError, match=msg):
 177         s.ewm(span=-0.1)
 178     with pytest.raises(ValueError, match=msg):
 179         s.ewm(span=0.0)
 180     with pytest.raises(ValueError, match=msg):
 181         s.ewm(span=0.9)
 182     s.ewm(span=1.0)
 183     s.ewm(span=1.1)
 184 
 185     msg = "halflife must satisfy: halflife > 0"
 186     with pytest.raises(ValueError, match=msg):
 187         s.ewm(halflife=-0.1)
 188     with pytest.raises(ValueError, match=msg):
 189         s.ewm(halflife=0.0)
 190     s.ewm(halflife=0.1)
 191 
 192     msg = "alpha must satisfy: 0 < alpha <= 1"
 193     with pytest.raises(ValueError, match=msg):
 194         s.ewm(alpha=-0.1)
