  91 def test_rolling_apply_consistency(
  92     consistency_data, base_functions, no_nan_functions, window, min_periods, center
  93 ):
  94     x, is_constant, no_nans = consistency_data
  95 
  96     with warnings.catch_warnings():
  97         warnings.filterwarnings(
  98             "ignore", message=".*(empty slice|0 for slice).*", category=RuntimeWarning,
  99         )
 100         # test consistency between rolling_xyz() and either (a)
 101         # rolling_apply of Series.xyz(), or (b) rolling_apply of
 102         # np.nanxyz()
 103         functions = base_functions
 104 
 105         # GH 8269
 106         if no_nans:
 107             functions = no_nan_functions + base_functions
 108         for (f, require_min_periods, name) in functions:
 109             rolling_f = getattr(
 110                 x.rolling(window=window, center=center, min_periods=min_periods), name,
 111             )
 112 
 113             if (
 114                 require_min_periods
 115                 and (min_periods is not None)
 116                 and (min_periods < require_min_periods)
 117             ):
 118                 continue
 119 
 120             if name == "count":
 121                 rolling_f_result = rolling_f()
 122                 rolling_apply_f_result = x.rolling(
 123                     window=window, min_periods=min_periods, center=center
 124                 ).apply(func=f, raw=True)
 125             else:
 126                 if name in ["cov", "corr"]:
 127                     rolling_f_result = rolling_f(pairwise=False)
 128                 else:
 129                     rolling_f_result = rolling_f()
 130                 rolling_apply_f_result = x.rolling(
 131                     window=window, min_periods=min_periods, center=center
 132                 ).apply(func=f, raw=True)
 133 
 134             # GH 9422
 135             if name in ["sum", "prod"]:
 136                 tm.assert_equal(rolling_f_result, rolling_apply_f_result)

 628 @pytest.mark.slow
 629 @pytest.mark.parametrize(
 630     "window,min_periods,center", list(_rolling_consistency_cases())
 631 )
 632 def test_rolling_consistency(consistency_data, window, min_periods, center):
 633     x, is_constant, no_nans = consistency_data
 634     # suppress warnings about empty slices, as we are deliberately testing
 635     # with empty/0-length Series/DataFrames
 636     with warnings.catch_warnings():
 637         warnings.filterwarnings(
 638             "ignore", message=".*(empty slice|0 for slice).*", category=RuntimeWarning,
 639         )
 640 
 641         # test consistency between different rolling_* moments
 642         moments_consistency_mock_mean(
 643             x=x,
 644             mean=lambda x: (
 645                 x.rolling(window=window, min_periods=min_periods, center=center).mean()
 646             ),
 647             mock_mean=lambda x: (
 648                 x.rolling(window=window, min_periods=min_periods, center=center)
 649                 .sum()
 650                 .divide(
 651                     x.rolling(
 652                         window=window, min_periods=min_periods, center=center
 653                     ).count()
 654                 )
 655             ),
 656         )
 657 
 658         moments_consistency_is_constant(
 659             x=x,
 660             is_constant=is_constant,
 661             min_periods=min_periods,
 662             count=lambda x: (
 663                 x.rolling(window=window, min_periods=min_periods, center=center).count()
 664             ),
 665             mean=lambda x: (
 666                 x.rolling(window=window, min_periods=min_periods, center=center).mean()
 667             ),
 668             corr=lambda x, y: (
 669                 x.rolling(window=window, min_periods=min_periods, center=center).corr(y)
 670             ),
 671         )
 672 
 673         moments_consistency_var_debiasing_factors(
 674             x=x,
 675             var_unbiased=lambda x: (
 676                 x.rolling(window=window, min_periods=min_periods, center=center).var()
 677             ),
 678             var_biased=lambda x: (
 679                 x.rolling(window=window, min_periods=min_periods, center=center).var(
 680                     ddof=0
 681                 )
 682             ),
 683             var_debiasing_factors=lambda x: (
 684                 x.rolling(window=window, min_periods=min_periods, center=center)
 685                 .count()
 686                 .divide(
 687                     (
 688                         x.rolling(
 689                             window=window, min_periods=min_periods, center=center
 690                         ).count()
 691                         - 1.0
 692                     ).replace(0.0, np.nan)
 693                 )
 694             ),
 695         )
