 163     consistency_data, base_functions, no_nan_functions, min_periods
 164 ):
 165     x, is_constant, no_nans = consistency_data
 166 
 167     with warnings.catch_warnings():
 168         warnings.filterwarnings(
 169             "ignore", message=".*(empty slice|0 for slice).*", category=RuntimeWarning,
 170         )
 171         # test consistency between expanding_xyz() and either (a)
 172         # expanding_apply of Series.xyz(), or (b) expanding_apply of
 173         # np.nanxyz()
 174         functions = base_functions
 175 
 176         # GH 8269
 177         if no_nans:
 178             functions = base_functions + no_nan_functions
 179         for (f, require_min_periods, name) in functions:
 180             expanding_f = getattr(x.expanding(min_periods=min_periods), name)
 181 
 182             if (
 183                 require_min_periods
 184                 and (min_periods is not None)
 185                 and (min_periods < require_min_periods)
 186             ):
 187                 continue
 188 
 189             if name == "count":
 190                 expanding_f_result = expanding_f()
 191                 expanding_apply_f_result = x.expanding(min_periods=0).apply(
 192                     func=f, raw=True
 193                 )
 194             else:
 195                 if name in ["cov", "corr"]:
 196                     expanding_f_result = expanding_f(pairwise=False)
 197                 else:
 198                     expanding_f_result = expanding_f()
 199                 expanding_apply_f_result = x.expanding(min_periods=min_periods).apply(
 200                     func=f, raw=True
 201                 )
 202 
 203             # GH 9422
 204             if name in ["sum", "prod"]:
 205                 tm.assert_equal(expanding_f_result, expanding_apply_f_result)
