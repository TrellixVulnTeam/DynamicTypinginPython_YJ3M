  17 def _check_moment_func(
  18     static_comp,
  19     name,
  20     raw,
  21     has_min_periods=True,
  22     has_center=True,
  23     has_time_rule=True,

  25     zero_min_periods_equal=True,
  26     series=None,
  27     frame=None,
  28     **kwargs,
  29 ):
  30 
  31     # inject raw
  32     if name == "apply":
  33         kwargs = copy.copy(kwargs)
  34         kwargs["raw"] = raw
  35 
  36     def get_result(obj, window, min_periods=None, center=False):
  37         r = obj.rolling(window=window, min_periods=min_periods, center=center)
  38         return getattr(r, name)(**kwargs)
  39 
  40     series_result = get_result(series, window=50)
  41     assert isinstance(series_result, Series)
  42     tm.assert_almost_equal(series_result.iloc[-1], static_comp(series[-50:]))
  43 
  44     frame_result = get_result(frame, window=50)
  45     assert isinstance(frame_result, DataFrame)
  46     tm.assert_series_equal(
  47         frame_result.iloc[-1, :],
  48         frame.iloc[-50:, :].apply(static_comp, axis=0, raw=raw),
  49         check_names=False,
  50     )
  51 
  52     # check time_rule works
  53     if has_time_rule:
  54         win = 25
  55         minp = 10
  56         ser = series[::2].resample("B").mean()
  57         frm = frame[::2].resample("B").mean()
  58 
  59         if has_min_periods:
  60             series_result = get_result(ser, window=win, min_periods=minp)
  61             frame_result = get_result(frm, window=win, min_periods=minp)
  62         else:
  63             series_result = get_result(ser, window=win, min_periods=0)
  64             frame_result = get_result(frm, window=win, min_periods=0)
  65 
  66         last_date = series_result.index[-1]
  67         prev_date = last_date - 24 * offsets.BDay()
  68 
  69         trunc_series = series[::2].truncate(prev_date, last_date)
  70         trunc_frame = frame[::2].truncate(prev_date, last_date)
  71 
  72         tm.assert_almost_equal(series_result[-1], static_comp(trunc_series))
  73 
  74         tm.assert_series_equal(
  75             frame_result.xs(last_date),
  76             trunc_frame.apply(static_comp, raw=raw),
  77             check_names=False,
  78         )

  83     obj[-10:] = np.NaN

  85         result = get_result(obj, 50, min_periods=30)
  86         tm.assert_almost_equal(result.iloc[-1], static_comp(obj[10:-10]))
  87 
  88         # min_periods is working correctly
  89         result = get_result(obj, 20, min_periods=15)
  90         assert isna(result.iloc[23])
  91         assert not isna(result.iloc[24])
  92

 762 @pytest.mark.parametrize("q", [0.0, 0.1, 0.5, 0.9, 1.0])
 763 def test_rolling_quantile(q, raw, series, frame):
 764     def scoreatpercentile(a, per):
 765         values = np.sort(a, axis=0)
 766 
 767         idx = int(per / 1.0 * (values.shape[0] - 1))
 768 
 769         if idx == values.shape[0] - 1:
 770             retval = values[-1]
 771 
 772         else:
 773             qlow = float(idx) / float(values.shape[0] - 1)
 774             qhig = float(idx + 1) / float(values.shape[0] - 1)
 775             vlow = values[idx]
 776             vhig = values[idx + 1]
 777             retval = vlow + (vhig - vlow) * (per - qlow) / (qhig - qlow)
 778 
 779         return retval
 780 
 781     def quantile_func(x):
 782         return scoreatpercentile(x, q)
 783 
 784     _check_moment_func(
 785         quantile_func, name="quantile", quantile=q, raw=raw, series=series, frame=frame
 786     )
