  48 def test_ewm_consistency(consistency_data, min_periods, adjust, ignore_na):
  49     def _weights(s, com, adjust, ignore_na):
  50         if isinstance(s, DataFrame):
  51             if not len(s.columns):
  52                 return DataFrame(index=s.index, columns=s.columns)
  53             w = concat(
  54                 [
  55                     _weights(s.iloc[:, i], com=com, adjust=adjust, ignore_na=ignore_na)
  56                     for i, _ in enumerate(s.columns)
  57                 ],
  58                 axis=1,
  59             )
  60             w.index = s.index
  61             w.columns = s.columns
  62             return w
  63 
  64         w = Series(np.nan, index=s.index)
  65         alpha = 1.0 / (1.0 + com)
  66         if ignore_na:
  67             w[s.notna()] = _weights(
  68                 s[s.notna()], com=com, adjust=adjust, ignore_na=False
  69             )
  70         elif adjust:
  71             for i in range(len(s)):
  72                 if s.iat[i] == s.iat[i]:
  73                     w.iat[i] = pow(1.0 / (1.0 - alpha), i)
  74         else:
  75             sum_wts = 0.0
  76             prev_i = -1
  77             for i in range(len(s)):
  78                 if s.iat[i] == s.iat[i]:
  79                     if prev_i == -1:
  80                         w.iat[i] = 1.0
  81                     else:
  82                         w.iat[i] = alpha * sum_wts / pow(1.0 - alpha, i - prev_i)
  83                     sum_wts += w.iat[i]
  84                     prev_i = i
  85         return w
  86 
  87     def _variance_debiasing_factors(s, com, adjust, ignore_na):
  88         weights = _weights(s, com=com, adjust=adjust, ignore_na=ignore_na)
  89         cum_sum = weights.cumsum().fillna(method="ffill")
  90         cum_sum_sq = (weights * weights).cumsum().fillna(method="ffill")
  91         numerator = cum_sum * cum_sum
  92         denominator = numerator - cum_sum_sq
  93         denominator[denominator <= 0.0] = np.nan
  94         return numerator / denominator
  95 
  96     def _ewma(s, com, min_periods, adjust, ignore_na):
  97         weights = _weights(s, com=com, adjust=adjust, ignore_na=ignore_na)
  98         result = (
  99             s.multiply(weights).cumsum().divide(weights.cumsum()).fillna(method="ffill")
 100         )
 101         result[
 102             s.expanding().count() < (max(min_periods, 1) if min_periods else 1)
 103         ] = np.nan
 104         return result
