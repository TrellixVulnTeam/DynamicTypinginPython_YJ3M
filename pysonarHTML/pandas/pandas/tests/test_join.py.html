 293 def test_merge_join_categorical_multiindex():
 294     # From issue 16627
 295     a = {
 296         "Cat1": Categorical(["a", "b", "a", "c", "a", "b"], ["a", "b", "c"]),
 297         "Int1": [0, 1, 0, 1, 0, 0],
 298     }
 299     a = DataFrame(a)
 300 
 301     b = {
 302         "Cat": Categorical(["a", "b", "c", "a", "b", "c"], ["a", "b", "c"]),
 303         "Int": [0, 0, 0, 1, 1, 1],
 304         "Factor": [1.1, 1.2, 1.3, 1.4, 1.5, 1.6],
 305     }
 306     b = DataFrame(b).set_index(["Cat", "Int"])["Factor"]
 307 
 308     expected = merge(
 309         a,
 310         b.reset_index(),
 311         left_on=["Cat1", "Int1"],
 312         right_on=["Cat", "Int"],
 313         how="left",
 314     )
 315     result = a.join(b, on=["Cat1", "Int1"])
 316     expected = expected.drop(["Cat", "Int"], axis=1)
 317     tm.assert_frame_equal(expected, result)
 318 
 319     # Same test, but with ordered categorical
 320     a = {
 321         "Cat1": Categorical(
 322             ["a", "b", "a", "c", "a", "b"], ["b", "a", "c"], ordered=True
 323         ),
 324         "Int1": [0, 1, 0, 1, 0, 0],
 325     }
 326     a = DataFrame(a)
 327 
 328     b = {
 329         "Cat": Categorical(
 330             ["a", "b", "c", "a", "b", "c"], ["b", "a", "c"], ordered=True
 331         ),
 332         "Int": [0, 0, 0, 1, 1, 1],
 333         "Factor": [1.1, 1.2, 1.3, 1.4, 1.5, 1.6],
 334     }
 335     b = DataFrame(b).set_index(["Cat", "Int"])["Factor"]
 336 
 337     expected = merge(
 338         a,
 339         b.reset_index(),
 340         left_on=["Cat1", "Int1"],
 341         right_on=["Cat", "Int"],
 342         how="left",
 343     )
 344     result = a.join(b, on=["Cat1", "Int1"])
 345     expected = expected.drop(["Cat", "Int"], axis=1)
 346     tm.assert_frame_equal(expected, result)
