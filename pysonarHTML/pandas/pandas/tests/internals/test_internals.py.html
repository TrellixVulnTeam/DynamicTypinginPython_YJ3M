  37 def get_numeric_mat(shape):
  38     arr = np.arange(shape[0])
  39     return np.lib.stride_tricks.as_strided(
  40         x=arr, shape=shape, strides=(arr.itemsize,) + (0,) * (len(shape) - 1)
  41     ).copy()

  47 def create_block(typestr, placement, item_shape=None, num_offset=0):
  48     """
  49     Supported typestr:
  50 
  51         * float, f8, f4, f2
  52         * int, i8, i4, i2, i1
  53         * uint, u8, u4, u2, u1
  54         * complex, c16, c8
  55         * bool
  56         * object, string, O
  57         * datetime, dt, M8[ns], M8[ns, tz]
  58         * timedelta, td, m8[ns]
  59         * sparse (SparseArray with fill_value=0.0)
  60         * sparse_na (SparseArray with fill_value=np.nan)
  61         * category, category2
  62 
  63     """
  64     placement = BlockPlacement(placement)
  65     num_items = len(placement)
  66 
  67     if item_shape is None:
  68         item_shape = (N,)
  69 
  70     shape = (num_items,) + item_shape
  71 
  72     mat = get_numeric_mat(shape)
  73 
  74     if typestr in (
  75         "float",
  76         "f8",
  77         "f4",
  78         "f2",
  79         "int",
  80         "i8",
  81         "i4",
  82         "i2",
  83         "i1",
  84         "uint",
  85         "u8",
  86         "u4",
  87         "u2",
  88         "u1",
  89     ):
  90         values = mat.astype(typestr) + num_offset
  91     elif typestr in ("complex", "c16", "c8"):
  92         values = 1.0j * (mat.astype(typestr) + num_offset)
  93     elif typestr in ("object", "string", "O"):
  94         values = np.reshape([f"A{i:d}" for i in mat.ravel() + num_offset], shape)
  95     elif typestr in ("b", "bool"):
  96         values = np.ones(shape, dtype=np.bool_)
  97     elif typestr in ("datetime", "dt", "M8[ns]"):
  98         values = (mat * 1e9).astype("M8[ns]")
  99     elif typestr.startswith("M8[ns"):
 100         # datetime with tz
 101         m = re.search(r"M8\[ns,\s*(\w+\/?\w*)\]", typestr)
 102         assert m is not None, f"incompatible typestr -> {typestr}"
 103         tz = m.groups()[0]
 104         assert num_items == 1, "must have only 1 num items for a tz-aware"
 105         values = DatetimeIndex(np.arange(N) * 1e9, tz=tz)
 106     elif typestr in ("timedelta", "td", "m8[ns]"):
 107         values = (mat * 1).astype("m8[ns]")
 108     elif typestr in ("category",):
 109         values = Categorical([1, 1, 2, 2, 3, 3, 3, 3, 4, 4])
 110     elif typestr in ("category2",):
 111         values = Categorical(["a", "a", "a", "a", "b", "b", "c", "c", "c", "d"])
 112     elif typestr in ("sparse", "sparse_na"):
 113         # FIXME: doesn't support num_rows != 10
 114         assert shape[-1] == 10
 115         assert all(s == 1 for s in shape[:-1])
 116         if typestr.endswith("_na"):
 117             fill_value = np.nan
 118         else:
 119             fill_value = 0.0
 120         values = SparseArray(
 121             [fill_value, fill_value, 1, 2, 3, fill_value, 4, 5, fill_value, 6],
 122             fill_value=fill_value,
 123         )
 124         arr = values.sp_values.view()
 125         arr += num_offset - 1
 126     else:
 127         raise ValueError(f'Unsupported typestr: "{typestr}"')
 128 
 129     return make_block(values, placement=placement, ndim=len(shape))

 142 def create_mgr(descr, item_shape=None):
 143     """
 144     Construct BlockManager from string description.
 145 
 146     String description syntax looks similar to np.matrix initializer.  It looks
 147     like this::
 148 
 149         a,b,c: f8; d,e,f: i8
 150 
 151     Rules are rather simple:
 152 
 153     * see list of supported datatypes in `create_block` method
 154     * components are semicolon-separated
 155     * each component is `NAME,NAME,NAME: DTYPE_ID`
 156     * whitespace around colons & semicolons are removed
 157     * components with same DTYPE_ID are combined into single block
 158     * to force multiple blocks with same dtype, use '-SUFFIX'::
 159 
 160         'a:f8-1; b:f8-2; c:f8-foobar'
 161 
 162     """
 163     if item_shape is None:
 164         item_shape = (N,)
 165 
 166     offset = 0
 167     mgr_items = []
 168     block_placements = OrderedDict()
 169     for d in descr.split(";"):
 170         d = d.strip()
 171         if not len(d):
 172             continue
 173         names, blockstr = d.partition(":")[::2]
 174         blockstr = blockstr.strip()
 175         names = names.strip().split(",")
 176 
 177         mgr_items.extend(names)
 178         placement = list(np.arange(len(names)) + offset)
 179         try:
 180             block_placements[blockstr].extend(placement)
 181         except KeyError:
 182             block_placements[blockstr] = placement
 183         offset += len(names)
 184 
 185     mgr_items = Index(mgr_items)
 186 
 187     blocks = []
 188     num_offset = 0
 189     for blockstr, placement in block_placements.items():
 190         typestr = blockstr.split("-")[0]
 191         blocks.append(
 192             create_block(
 193                 typestr, placement, item_shape=item_shape, num_offset=num_offset
 194             )
 195         )
 196         num_offset += len(placement)
 197 
 198     return BlockManager(
 199         sorted(blocks, key=lambda b: b.mgr_locs[0]),
 200         [mgr_items] + [np.arange(n) for n in item_shape],
 201     )

 468     def test_convert(self):
 469         def _compare(old_mgr, new_mgr):
 470             """ compare the blocks, numeric compare ==, object don't """
 471             old_blocks = set(old_mgr.blocks)
 472             new_blocks = set(new_mgr.blocks)
 473             assert len(old_blocks) == len(new_blocks)
 474 
 475             # compare non-numeric
 476             for b in old_blocks:
 477                 found = False
 478                 for nb in new_blocks:
 479                     if (b.values == nb.values).all():
 480                         found = True
 481                         break
 482                 assert found
 483 
 484             for b in new_blocks:
 485                 found = False
 486                 for ob in old_blocks:
 487                     if (b.values == ob.values).all():
 488                         found = True
 489                         break
 490                 assert found
 491 
 492         # noops
 493         mgr = create_mgr("f: i8; g: f8")
 494         new_mgr = mgr.convert()
 495         _compare(mgr, new_mgr)
 496 
 497         # convert
 498         mgr = create_mgr("a,b,foo: object; f: i8; g: f8")
 499         mgr.iset(0, np.array(["1"] * N, dtype=np.object_))
 500         mgr.iset(1, np.array(["2."] * N, dtype=np.object_))
 501         mgr.iset(2, np.array(["foo."] * N, dtype=np.object_))
 502         new_mgr = mgr.convert(numeric=True)
 503         assert new_mgr.iget(0).dtype == np.int64
 504         assert new_mgr.iget(1).dtype == np.float64
 505         assert new_mgr.iget(2).dtype == np.object_
 506         assert new_mgr.iget(3).dtype == np.int64
 507         assert new_mgr.iget(4).dtype == np.float64
 508 
 509         mgr = create_mgr(
 510             "a,b,foo: object; f: i4; bool: bool; dt: datetime; i: i8; g: f8; h: f2"
 511         )
 512         mgr.iset(0, np.array(["1"] * N, dtype=np.object_))
 513         mgr.iset(1, np.array(["2."] * N, dtype=np.object_))

 772             mat = mgr.as_array()
 773 
 774             # we maybe using an ndarray to test slicing and
 775             # might not be the full length of the axis
 776             if isinstance(slobj, np.ndarray):
 777                 ax = mgr.axes[axis]
 778                 if len(ax) and len(slobj) and len(slobj) != len(ax):
