 259 def test_really_large_in_arr(large_val, signed, transform, multiple_elts, errors):
 260     # see gh-24910
 261     kwargs = dict(errors=errors) if errors is not None else dict()
 262     val = -large_val if signed else large_val
 263     val = transform(val)
 264 
 265     extra_elt = "string"
 266     arr = [val] + multiple_elts * [extra_elt]
 267 
 268     val_is_string = isinstance(val, str)
 269     coercing = errors == "coerce"
 270 
 271     if errors in (None, "raise") and (val_is_string or multiple_elts):
 272         if val_is_string:
 273             msg = "Integer out of range. at position 0"
 274         else:
 275             msg = 'Unable to parse string "string" at position 1'
 276 
 277         with pytest.raises(ValueError, match=msg):
 278             to_numeric(arr, **kwargs)
 279     else:
 280         result = to_numeric(arr, **kwargs)
 281 
 282         exp_val = float(val) if (coercing and val_is_string) else val
 283         expected = [exp_val]
 284 
 285         if multiple_elts:
 286             if coercing:
 287                 expected.append(np.nan)
 288                 exp_dtype = float
 289             else:
 290                 expected.append(extra_elt)
 291                 exp_dtype = object
 292         else:
 293             exp_dtype = float if isinstance(exp_val, (int, float)) else object
 294 
 295         tm.assert_almost_equal(result, np.array(expected, dtype=exp_dtype))

 298 def test_really_large_in_arr_consistent(large_val, signed, multiple_elts, errors):
 299     # see gh-24910
 300     #
 301     # Even if we discover that we have to hold float, does not mean
 302     # we should be lenient on subsequent elements that fail to be integer.
 303     kwargs = dict(errors=errors) if errors is not None else dict()
 304     arr = [str(-large_val if signed else large_val)]
 305 
 306     if multiple_elts:
 307         arr.insert(0, large_val)

 313         with pytest.raises(ValueError, match=msg):
 314             to_numeric(arr, **kwargs)
 315     else:
 316         result = to_numeric(arr, **kwargs)
 317 
 318         if errors == "coerce":
 319             expected = [float(i) for i in arr]
 320             exp_dtype = float
 321         else:
 322             expected = arr
 323             exp_dtype = object
