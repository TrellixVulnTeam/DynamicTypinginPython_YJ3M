  60     def test_to_timedelta_dataframe(self):
  61         # GH 11776
  62         arr = np.arange(10).reshape(2, 5)
  63         df = pd.DataFrame(np.arange(10).reshape(2, 5))
  64         for arg in (arr, df):
  65             with pytest.raises(TypeError, match="1-d array"):
  66                 to_timedelta(arg)
  67             for errors in ["ignore", "raise", "coerce"]:
  68                 with pytest.raises(TypeError, match="1-d array"):
  69                     to_timedelta(arg, errors=errors)

  71     def test_to_timedelta_invalid(self):
  72 
  73         # bad value for errors parameter
  74         msg = "errors must be one of"
  75         with pytest.raises(ValueError, match=msg):
  76             to_timedelta(["foo"], errors="never")
  77 
  78         # these will error
  79         msg = "invalid unit abbreviation: foo"
  80         with pytest.raises(ValueError, match=msg):
  81             to_timedelta([1, 2], unit="foo")
  82         with pytest.raises(ValueError, match=msg):
  83             to_timedelta(1, unit="foo")
  84 
  85         # time not supported ATM
  86         msg = (
  87             "Value must be Timedelta, string, integer, float, timedelta or convertible"
  88         )
  89         with pytest.raises(ValueError, match=msg):
  90             to_timedelta(time(second=1))
  91         assert to_timedelta(time(second=1), errors="coerce") is pd.NaT
  92 
  93         msg = "unit abbreviation w/o a number"
  94         with pytest.raises(ValueError, match=msg):
  95             to_timedelta(["foo", "bar"])
  96         tm.assert_index_equal(
  97             TimedeltaIndex([pd.NaT, pd.NaT]),
  98             to_timedelta(["foo", "bar"], errors="coerce"),
  99         )
 100 
 101         tm.assert_index_equal(
 102             TimedeltaIndex(["1 day", pd.NaT, "1 min"]),
 103             to_timedelta(["1 day", "bar", "1 min"], errors="coerce"),
 104         )
 105 
 106         # gh-13613: these should not error because errors='ignore'
 107         invalid_data = "apple"
 108         assert invalid_data == to_timedelta(invalid_data, errors="ignore")
 109 
 110         invalid_data = ["apple", "1 days"]
 111         tm.assert_numpy_array_equal(
 112             np.array(invalid_data, dtype=object),
 113             to_timedelta(invalid_data, errors="ignore"),
 114         )
 115 
 116         invalid_data = pd.Index(["apple", "1 days"])
 117         tm.assert_index_equal(invalid_data, to_timedelta(invalid_data, errors="ignore"))
 118 
 119         invalid_data = Series(["apple", "1 days"])
 120         tm.assert_series_equal(
 121             invalid_data, to_timedelta(invalid_data, errors="ignore")
 122         )

 133     def test_to_timedelta_on_missing_values(self):
 134         # GH5438
 135         timedelta_NaT = np.timedelta64("NaT")
 136 
 137         actual = pd.to_timedelta(Series(["00:00:01", np.nan]))
 138         expected = Series(
 139             [np.timedelta64(1000000000, "ns"), timedelta_NaT], dtype="<m8[ns]"
 140         )
 141         tm.assert_series_equal(actual, expected)
 142 
 143         actual = pd.to_timedelta(Series(["00:00:01", pd.NaT]))
 144         tm.assert_series_equal(actual, expected)
 145 
 146         actual = pd.to_timedelta(np.nan)
 147         assert actual.value == timedelta_NaT.astype("int64")
 148 
 149         actual = pd.to_timedelta(pd.NaT)
 150         assert actual.value == timedelta_NaT.astype("int64")

 152     def test_to_timedelta_float(self):
 153         # https://github.com/pandas-dev/pandas/issues/25077
 154         arr = np.arange(0, 1, 1e-6)[-10:]
 155         result = pd.to_timedelta(arr, unit="s")
 156         expected_asi8 = np.arange(999990000, int(1e9), 1000, dtype="int64")
 157         tm.assert_numpy_array_equal(result.asi8, expected_asi8)
