 154     def test_unstack_fill(self):
 155 
 156         # GH #9746: fill_value keyword argument for Series
 157         # and DataFrame unstack
 158 
 159         # From a series
 160         data = Series([1, 2, 4, 5], dtype=np.int16)
 161         data.index = MultiIndex.from_tuples(
 162             [("x", "a"), ("x", "b"), ("y", "b"), ("z", "a")]
 163         )
 164 
 165         result = data.unstack(fill_value=-1)
 166         expected = DataFrame(
 167             {"a": [1, -1, 5], "b": [2, 4, -1]}, index=["x", "y", "z"], dtype=np.int16
 168         )
 169         tm.assert_frame_equal(result, expected)
 170 
 171         # From a series with incorrect data type for fill_value
 172         result = data.unstack(fill_value=0.5)
 173         expected = DataFrame(
 174             {"a": [1, 0.5, 5], "b": [2, 4, 0.5]}, index=["x", "y", "z"], dtype=np.float
 175         )
 176         tm.assert_frame_equal(result, expected)
 177 
 178         # GH #13971: fill_value when unstacking multiple levels:
 179         df = DataFrame(
 180             {"x": ["a", "a", "b"], "y": ["j", "k", "j"], "z": [0, 1, 2], "w": [0, 1, 2]}
 181         ).set_index(["x", "y", "z"])
 182         unstacked = df.unstack(["x", "y"], fill_value=0)
 183         key = ("w", "b", "j")
 184         expected = unstacked[key]
 185         result = pd.Series([0, 0, 2], index=unstacked.index, name=key)
 186         tm.assert_series_equal(result, expected)
 187 
 188         stacked = unstacked.stack(["x", "y"])
 189         stacked.index = stacked.index.reorder_levels(df.index.names)
 190         # Workaround for GH #17886 (unnecessarily casts to float):
 191         stacked = stacked.astype(np.int64)
 192         result = stacked.loc[df.index]
 193         tm.assert_frame_equal(result, df)
 194 
 195         # From a series
 196         s = df["w"]
 197         result = s.unstack(["x", "y"], fill_value=0)
 198         expected = unstacked["w"]
 199         tm.assert_frame_equal(result, expected)
