  28 def assert_stat_op_calc(
  29     opname,
  30     alternative,
  31     frame,
  32     has_skipna=True,
  33     check_dtype=True,
  34     check_dates=False,
  35     check_less_precise=False,
  36     skipna_alternative=None,
  37 ):
  38     """
  39     Check that operator opname works as advertised on frame
  40 
  41     Parameters
  42     ----------
  43     opname : string
  44         Name of the operator to test on frame
  45     alternative : function
  46         Function that opname is tested against; i.e. "frame.opname()" should
  47         equal "alternative(frame)".
  48     frame : DataFrame
  49         The object that the tests are executed on
  50     has_skipna : bool, default True
  51         Whether the method "opname" has the kwarg "skip_na"
  52     check_dtype : bool, default True
  53         Whether the dtypes of the result of "frame.opname()" and
  54         "alternative(frame)" should be checked.
  55     check_dates : bool, default false
  56         Whether opname should be tested on a Datetime Series
  57     check_less_precise : bool, default False
  58         Whether results should only be compared approximately;
  59         passed on to tm.assert_series_equal
  60     skipna_alternative : function, default None
  61         NaN-safe version of alternative
  62     """
  63     f = getattr(frame, opname)
  64 
  65     if check_dates:
  66         expected_warning = FutureWarning if opname in ["mean", "median"] else None
  67         df = DataFrame({"b": date_range("1/1/2001", periods=2)})
  68         with tm.assert_produces_warning(expected_warning):
  69             result = getattr(df, opname)()
  70         assert isinstance(result, Series)
  71 
  72         df["a"] = range(len(df))
  73         with tm.assert_produces_warning(expected_warning):
  74             result = getattr(df, opname)()
  75         assert isinstance(result, Series)
  76         assert len(result)
  77 
  78     if has_skipna:
  79 
  80         def wrapper(x):
  81             return alternative(x.values)
  82 
  83         skipna_wrapper = tm._make_skipna_wrapper(alternative, skipna_alternative)
  84         result0 = f(axis=0, skipna=False)
  85         result1 = f(axis=1, skipna=False)
  86         tm.assert_series_equal(
  87             result0,
  88             frame.apply(wrapper),
  89             check_dtype=check_dtype,
  90             check_less_precise=check_less_precise,
  91         )
  92         # HACK: win32
  93         tm.assert_series_equal(
  94             result1,
  95             frame.apply(wrapper, axis=1),
  96             check_dtype=False,
  97             check_less_precise=check_less_precise,
  98         )
  99     else:
 100         skipna_wrapper = alternative
 101 
 102     result0 = f(axis=0)
 103     result1 = f(axis=1)
 104     tm.assert_series_equal(
 105         result0,
 106         frame.apply(skipna_wrapper),
 107         check_dtype=check_dtype,
 108         check_less_precise=check_less_precise,
 109     )
 110 
 111     if opname in ["sum", "prod"]:
 112         expected = frame.apply(skipna_wrapper, axis=1)
 113         tm.assert_series_equal(
 114             result1, expected, check_dtype=False, check_less_precise=check_less_precise
 115         )
 116 
 117     # check dtypes
 118     if check_dtype:
 119         lcd_dtype = frame.values.dtype
 120         assert lcd_dtype == result0.dtype
 121         assert lcd_dtype == result1.dtype
 122 
 123     # bad axis
 124     with pytest.raises(ValueError, match="No axis named 2"):
 125         f(axis=2)
 126 
 127     # all NA case
 128     if has_skipna:
 129         all_na = frame * np.NaN
 130         r0 = getattr(all_na, opname)(axis=0)
 131         r1 = getattr(all_na, opname)(axis=1)
 132         if opname in ["sum", "prod"]:
 133             unit = 1 if opname == "prod" else 0  # result for empty sum/prod
 134             expected = pd.Series(unit, index=r0.index, dtype=r0.dtype)
 135             tm.assert_series_equal(r0, expected)
 136             expected = pd.Series(unit, index=r1.index, dtype=r1.dtype)
 137             tm.assert_series_equal(r1, expected)

 166 def assert_bool_op_calc(opname, alternative, frame, has_skipna=True):
 167     """
 168     Check that bool operator opname works as advertised on frame
 169 
 170     Parameters
 171     ----------
 172     opname : string
 173         Name of the operator to test on frame
 174     alternative : function
 175         Function that opname is tested against; i.e. "frame.opname()" should
 176         equal "alternative(frame)".
 177     frame : DataFrame
 178         The object that the tests are executed on
 179     has_skipna : bool, default True
 180         Whether the method "opname" has the kwarg "skip_na"
 181     """
 182     f = getattr(frame, opname)
 183 
 184     if has_skipna:
 185 
 186         def skipna_wrapper(x):
 187             nona = x.dropna().values
 188             return alternative(nona)
 189 
 190         def wrapper(x):
 191             return alternative(x.values)
 192 
 193         result0 = f(axis=0, skipna=False)
 194         result1 = f(axis=1, skipna=False)
 195 
 196         tm.assert_series_equal(result0, frame.apply(wrapper))
 197         tm.assert_series_equal(
 198             result1, frame.apply(wrapper, axis=1), check_dtype=False
 199         )  # HACK: win32
 200     else:
 201         skipna_wrapper = alternative
 202         wrapper = alternative
 203 
 204     result0 = f(axis=0)
 205     result1 = f(axis=1)
 206 
 207     tm.assert_series_equal(result0, frame.apply(skipna_wrapper))
 208     tm.assert_series_equal(
 209         result1, frame.apply(skipna_wrapper, axis=1), check_dtype=False
 210     )
 211 
 212     # bad axis
 213     with pytest.raises(ValueError, match="No axis named 2"):
 214         f(axis=2)
 215 
 216     # all NA case
 217     if has_skipna:
 218         all_na = frame * np.NaN
 219         r0 = getattr(all_na, opname)(axis=0)
 220         r1 = getattr(all_na, opname)(axis=1)
 221         if opname == "any":
 222             assert not r0.any()
 223             assert not r1.any()
 224         else:
 225             assert r0.all()
 226             assert r1.all()

 292     def test_stat_op_calc(self, float_frame_with_na, mixed_float_frame):
 293         def count(s):
 294             return notna(s).sum()
 295 
 296         def nunique(s):
 297             return len(algorithms.unique1d(s.dropna()))
 298 
 299         def mad(x):
 300             return np.abs(x - x.mean()).mean()
 301 
 302         def var(x):
 303             return np.var(x, ddof=1)
 304 
 305         def std(x):
 306             return np.std(x, ddof=1)
 307 
 308         def sem(x):
 309             return np.std(x, ddof=1) / np.sqrt(len(x))
 310 
 311         def skewness(x):
 312             from scipy.stats import skew  # noqa:F811
 313 
 314             if len(x) < 3:
 315                 return np.nan
 316             return skew(x, bias=False)
 317 
 318         def kurt(x):
 319             from scipy.stats import kurtosis  # noqa:F811
 320 
 321             if len(x) < 4:
 322                 return np.nan
 323             return kurtosis(x, bias=False)
 324 
 325         assert_stat_op_calc(
 326             "nunique",
 327             nunique,
 328             float_frame_with_na,
 329             has_skipna=False,
 330             check_dtype=False,
 331             check_dates=True,
 332         )
 333 
 334         # GH#32571 check_less_precise is needed on apparently-random
 335         #  py37-npdev builds and OSX-PY36-min_version builds
 336         # mixed types (with upcasting happening)
 337         assert_stat_op_calc(
 338             "sum",
 339             np.sum,
 340             mixed_float_frame.astype("float32"),
 341             check_dtype=False,
 342             check_less_precise=True,
 343         )
 344 
 345         assert_stat_op_calc(
 346             "sum", np.sum, float_frame_with_na, skipna_alternative=np.nansum
 347         )
 348         assert_stat_op_calc("mean", np.mean, float_frame_with_na, check_dates=True)
 349         assert_stat_op_calc(
 350             "product", np.prod, float_frame_with_na, skipna_alternative=np.nanprod
 351         )
 352 
 353         assert_stat_op_calc("mad", mad, float_frame_with_na)
 354         assert_stat_op_calc("var", var, float_frame_with_na)
 355         assert_stat_op_calc("std", std, float_frame_with_na)
 356         assert_stat_op_calc("sem", sem, float_frame_with_na)
 357 
 358         assert_stat_op_calc(
 359             "count",
 360             count,
 361             float_frame_with_na,
 362             has_skipna=False,
 363             check_dtype=False,
 364             check_dates=True,
 365         )
 366 
 367         try:
 368             from scipy import skew, kurtosis  # noqa:F401
 369 
 370             assert_stat_op_calc("skew", skewness, float_frame_with_na)
 371             assert_stat_op_calc("kurt", kurt, float_frame_with_na)
 372         except ImportError:
 373             pass
