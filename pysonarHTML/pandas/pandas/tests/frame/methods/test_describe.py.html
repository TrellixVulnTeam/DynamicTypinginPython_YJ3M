  87     def test_describe_categorical(self):
  88         df = DataFrame({"value": np.random.randint(0, 10000, 100)})
  89         labels = [f"{i} - {i + 499}" for i in range(0, 10000, 500)]
  90         cat_labels = Categorical(labels, labels)
  91 
  92         df = df.sort_values(by=["value"], ascending=True)
  93         df["value_group"] = pd.cut(
  94             df.value, range(0, 10500, 500), right=False, labels=cat_labels
  95         )
  96         cat = df
  97 
  98         # Categoricals should not show up together with numerical columns
  99         result = cat.describe()
 100         assert len(result.columns) == 1
 101 
 102         # In a frame, describe() for the cat should be the same as for string
 103         # arrays (count, unique, top, freq)
 104 
 105         cat = Categorical(
 106             ["a", "b", "b", "b"], categories=["a", "b", "c"], ordered=True
 107         )
 108         s = Series(cat)
 109         result = s.describe()
 110         expected = Series([4, 2, "b", 3], index=["count", "unique", "top", "freq"])
 111         tm.assert_series_equal(result, expected)
 112 
 113         cat = Series(Categorical(["a", "b", "c", "c"]))
 114         df3 = DataFrame({"cat": cat, "s": ["a", "b", "c", "c"]})
 115         result = df3.describe()
 116         tm.assert_numpy_array_equal(result["cat"].values, result["s"].values)

 118     def test_describe_empty_categorical_column(self):
 119         # GH#26397
 120         # Ensure the index of an an empty categorical DataFrame column
 121         # also contains (count, unique, top, freq)
 122         df = pd.DataFrame({"empty_col": Categorical([])})
 123         result = df.describe()

 165     def test_describe_datetime_columns(self):
 166         columns = pd.DatetimeIndex(
 167             ["2011-01-01", "2011-02-01", "2011-03-01"],
 168             freq="MS",
 169             tz="US/Eastern",
 170             name="XXX",
 171         )
 172         df = DataFrame(
 173             {
 174                 0: [10, 20, 30, 40, 50],
 175                 1: [10, 20, 30, 40, 50],
 176                 2: ["A", 0, None, "X", 1],
 177             }
 178         )
 179         df.columns = columns
 180         result = df.describe()
 181 
 182         exp_columns = pd.DatetimeIndex(
 183             ["2011-01-01", "2011-02-01"], freq="MS", tz="US/Eastern", name="XXX"
 184         )
 185         expected = DataFrame(
 186             {
 187                 0: [5, 30, df.iloc[:, 0].std(), 10, 20, 30, 40, 50],
 188                 1: [5, 30, df.iloc[:, 1].std(), 10, 20, 30, 40, 50],
 189             },
 190             index=["count", "mean", "std", "min", "25%", "50%", "75%", "max"],
 191         )
 192         expected.columns = exp_columns
 193         tm.assert_frame_equal(result, expected)
 194         assert result.columns.freq == "MS"
 195         assert result.columns.tz == expected.columns.tz
