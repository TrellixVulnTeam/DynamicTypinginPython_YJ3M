  24     def test_pct_change(self, datetime_frame):
  25         rs = datetime_frame.pct_change(fill_method=None)
  26         tm.assert_frame_equal(rs, datetime_frame / datetime_frame.shift(1) - 1)
  27 
  28         rs = datetime_frame.pct_change(2)
  29         filled = datetime_frame.fillna(method="pad")
  30         tm.assert_frame_equal(rs, filled / filled.shift(2) - 1)
  31 
  32         rs = datetime_frame.pct_change(fill_method="bfill", limit=1)
  33         filled = datetime_frame.fillna(method="bfill", limit=1)
  34         tm.assert_frame_equal(rs, filled / filled.shift(1) - 1)
  35 
  36         rs = datetime_frame.pct_change(freq="5D")
  37         filled = datetime_frame.fillna(method="pad")
  38         tm.assert_frame_equal(
  39             rs, (filled / filled.shift(freq="5D") - 1).reindex_like(filled)
  40         )
