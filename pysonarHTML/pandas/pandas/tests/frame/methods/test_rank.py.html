 218     def test_rank_methods_frame(self):
 219         import scipy.stats  # noqa:F401
 220         from scipy.stats import rankdata
 221 
 222         xs = np.random.randint(0, 21, (100, 26))
 223         xs = (xs - 10.0) / 10.0
 224         cols = [chr(ord("z") - i) for i in range(xs.shape[1])]
 225 
 226         for vals in [xs, xs + 1e6, xs * 1e-6]:
 227             df = DataFrame(vals, columns=cols)
 228 
 229             for ax in [0, 1]:
 230                 for m in ["average", "min", "max", "first", "dense"]:
 231                     result = df.rank(axis=ax, method=m)
 232                     sprank = np.apply_along_axis(
 233                         rankdata, ax, vals, m if m != "first" else "ordinal"
 234                     )
 235                     sprank = sprank.astype(np.float64)
 236                     expected = DataFrame(sprank, columns=cols).astype("float64")
 237                     tm.assert_frame_equal(result, expected)

 239     @pytest.mark.parametrize("dtype", ["O", "f8", "i8"])
 240     def test_rank_descending(self, method, dtype):
 241 
 242         if "i" in dtype:
 243             df = self.df.dropna()
 244         else:
 245             df = self.df.astype(dtype)
 246 
 247         res = df.rank(ascending=False)
 248         expected = (df.max() - df).rank()
 249         tm.assert_frame_equal(res, expected)
 250 
 251         if method == "first" and dtype == "O":
 252             return
 253 
 254         expected = (df.max() - df).rank(method=method)
 255 
 256         if dtype != "O":
 257             res2 = df.rank(method=method, ascending=False, numeric_only=True)
 258             tm.assert_frame_equal(res2, expected)
 259 
 260         res3 = df.rank(method=method, ascending=False, numeric_only=False)
 261         tm.assert_frame_equal(res3, expected)

 263     @pytest.mark.parametrize("axis", [0, 1])
 264     @pytest.mark.parametrize("dtype", [None, object])
 265     def test_rank_2d_tie_methods(self, method, axis, dtype):
 266         df = self.df
 267 
 268         def _check2d(df, expected, method="average", axis=0):
 269             exp_df = DataFrame({"A": expected, "B": expected})
 270 
 271             if axis == 1:
 272                 df = df.T
 273                 exp_df = exp_df.T
 274 
 275             result = df.rank(method=method, axis=axis)
 276             tm.assert_frame_equal(result, exp_df)
 277 
 278         disabled = {(object, "first")}
 279         if (dtype, method) in disabled:
 280             return
 281         frame = df if dtype is None else df.astype(dtype)
 282         _check2d(frame, self.results[method], method=method, axis=axis)
