 289     def test_sort_values_nat_values_in_int_column(self):
 290 
 291         # GH#14922: "sorting with large float and multiple columns incorrect"
 292 
 293         # cause was that the int64 value NaT was considered as "na". Which is
 294         # only correct for datetime64 columns.
 295 
 296         int_values = (2, int(NaT))
 297         float_values = (2.0, -1.797693e308)
 298 
 299         df = DataFrame(
 300             dict(int=int_values, float=float_values), columns=["int", "float"]
 301         )
 302 
 303         df_reversed = DataFrame(
 304             dict(int=int_values[::-1], float=float_values[::-1]),
 305             columns=["int", "float"],
 306             index=[1, 0],
 307         )
 308 
 309         # NaT is not a "na" for int64 columns, so na_position must not
 310         # influence the result:
 311         df_sorted = df.sort_values(["int", "float"], na_position="last")
 312         tm.assert_frame_equal(df_sorted, df_reversed)
 313 
 314         df_sorted = df.sort_values(["int", "float"], na_position="first")
 315         tm.assert_frame_equal(df_sorted, df_reversed)
 316 
 317         # reverse sorting order
 318         df_sorted = df.sort_values(["int", "float"], ascending=False)
 319         tm.assert_frame_equal(df_sorted, df)
 320 
 321         # and now check if NaT is still considered as "na" for datetime64
 322         # columns:
 323         df = DataFrame(
 324             dict(datetime=[Timestamp("2016-01-01"), NaT], float=float_values),
 325             columns=["datetime", "float"],
 326         )
 327 
 328         df_reversed = DataFrame(
 329             dict(datetime=[NaT, Timestamp("2016-01-01")], float=float_values[::-1]),
 330             columns=["datetime", "float"],
 331             index=[1, 0],
 332         )
 333 
 334         df_sorted = df.sort_values(["datetime", "float"], na_position="first")
 335         tm.assert_frame_equal(df_sorted, df_reversed)
 336 
 337         df_sorted = df.sort_values(["datetime", "float"], na_position="last")
 338         tm.assert_frame_equal(df_sorted, df)
 339 
 340         # Ascending should not affect the results.
 341         df_sorted = df.sort_values(["datetime", "float"], ascending=False)
 342         tm.assert_frame_equal(df_sorted, df)
