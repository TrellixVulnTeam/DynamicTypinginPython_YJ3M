 181     def test_to_dict_tz(self):
 182         # GH#18372 When converting to dict with orient='records' columns of
 183         # datetime that are tz-aware were not converted to required arrays
 184         data = [
 185             (datetime(2017, 11, 18, 21, 53, 0, 219225, tzinfo=pytz.utc),),
 186             (datetime(2017, 11, 18, 22, 6, 30, 61810, tzinfo=pytz.utc),),
 187         ]
 188         df = DataFrame(list(data), columns=["d"])
 189 
 190         result = df.to_dict(orient="records")
 191         expected = [
 192             {"d": Timestamp("2017-11-18 21:53:00.219225+0000", tz=pytz.utc)},
 193             {"d": Timestamp("2017-11-18 22:06:30.061810+0000", tz=pytz.utc)},
 194         ]
 195         tm.assert_dict_equal(result[0], expected[0])
 196         tm.assert_dict_equal(result[1], expected[1])
