  24     def test_append_series_dict(self):
  25         df = DataFrame(np.random.randn(5, 4), columns=["foo", "bar", "baz", "qux"])
  26 
  27         series = df.loc[4]
  28         msg = "Indexes have overlapping values"
  29         with pytest.raises(ValueError, match=msg):
  30             df.append(series, verify_integrity=True)
  31 
  32         series.name = None
  33         msg = "Can only append a Series if ignore_index=True"
  34         with pytest.raises(TypeError, match=msg):
  35             df.append(series, verify_integrity=True)
  36 
  37         result = df.append(series[::-1], ignore_index=True)
  38         expected = df.append(
  39             DataFrame({0: series[::-1]}, index=df.columns).T, ignore_index=True
  40         )
  41         tm.assert_frame_equal(result, expected)
  42 
  43         # dict
  44         result = df.append(series.to_dict(), ignore_index=True)
  45         tm.assert_frame_equal(result, expected)
  46 
  47         result = df.append(series[::-1][:3], ignore_index=True)
  48         expected = df.append(
  49             DataFrame({0: series[::-1][:3]}).T, ignore_index=True, sort=True
  50         )
  51         tm.assert_frame_equal(result, expected.loc[:, result.columns])
  52 
  53         msg = "Can only append a dict if ignore_index=True"
  54         with pytest.raises(TypeError, match=msg):
  55             df.append(series.to_dict())
  56 
  57         # can append when name set
  58         row = df.loc[4]
  59         row.name = 5
  60         result = df.append(row)
  61         expected = df.append(df[-1:], ignore_index=True)
  62         tm.assert_frame_equal(result, expected)

 126     def test_append_dtypes(self):
 127 
 128         # GH 5754
 129         # row appends of different dtypes (so need to do by-item)
 130         # can sometimes infer the correct type
 131 
 132         df1 = DataFrame({"bar": Timestamp("20130101")}, index=range(5))
 133         df2 = DataFrame()
 134         result = df1.append(df2)
 135         expected = df1.copy()
 136         tm.assert_frame_equal(result, expected)
 137 
 138         df1 = DataFrame({"bar": Timestamp("20130101")}, index=range(1))
 139         df2 = DataFrame({"bar": "foo"}, index=range(1, 2))
 140         result = df1.append(df2)
 141         expected = DataFrame({"bar": [Timestamp("20130101"), "foo"]})
 142         tm.assert_frame_equal(result, expected)
 143 
 144         df1 = DataFrame({"bar": Timestamp("20130101")}, index=range(1))
 145         df2 = DataFrame({"bar": np.nan}, index=range(1, 2))
 146         result = df1.append(df2)
 147         expected = DataFrame(
 148             {"bar": Series([Timestamp("20130101"), np.nan], dtype="M8[ns]")}
 149         )
 150         tm.assert_frame_equal(result, expected)
 151 
 152         df1 = DataFrame({"bar": Timestamp("20130101")}, index=range(1))
 153         df2 = DataFrame({"bar": np.nan}, index=range(1, 2), dtype=object)
 154         result = df1.append(df2)
 155         expected = DataFrame(
 156             {"bar": Series([Timestamp("20130101"), np.nan], dtype="M8[ns]")}
 157         )
 158         tm.assert_frame_equal(result, expected)
 159 
 160         df1 = DataFrame({"bar": np.nan}, index=range(1))
 161         df2 = DataFrame({"bar": Timestamp("20130101")}, index=range(1, 2))
 162         result = df1.append(df2)
 163         expected = DataFrame(
 164             {"bar": Series([np.nan, Timestamp("20130101")], dtype="M8[ns]")}
 165         )
 166         tm.assert_frame_equal(result, expected)
 167 
 168         df1 = DataFrame({"bar": Timestamp("20130101")}, index=range(1))
 169         df2 = DataFrame({"bar": 1}, index=range(1, 2), dtype=object)
 170         result = df1.append(df2)
 171         expected = DataFrame({"bar": Series([Timestamp("20130101"), 1])})
 172         tm.assert_frame_equal(result, expected)
