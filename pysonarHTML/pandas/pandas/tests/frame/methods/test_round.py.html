 126     def test_round_numpy(self):
 127         # GH#12600
 128         df = DataFrame([[1.53, 1.36], [0.06, 7.01]])
 129         out = np.round(df, decimals=0)
 130         expected = DataFrame([[2.0, 1.0], [0.0, 7.0]])
 131         tm.assert_frame_equal(out, expected)
 132 
 133         msg = "the 'out' parameter is not supported"
 134         with pytest.raises(ValueError, match=msg):
 135             np.round(df, decimals=0, out=df)

 145     def test_round_mixed_type(self):
 146         # GH#11885
 147         df = DataFrame(
 148             {
 149                 "col1": [1.1, 2.2, 3.3, 4.4],
 150                 "col2": ["1", "a", "c", "f"],
 151                 "col3": date_range("20111111", periods=4),
 152             }
 153         )
 154         round_0 = DataFrame(
 155             {
 156                 "col1": [1.0, 2.0, 3.0, 4.0],
 157                 "col2": ["1", "a", "c", "f"],
 158                 "col3": date_range("20111111", periods=4),
 159             }
 160         )
 161         tm.assert_frame_equal(df.round(), round_0)
 162         tm.assert_frame_equal(df.round(1), df)
 163         tm.assert_frame_equal(df.round({"col1": 1}), df)
 164         tm.assert_frame_equal(df.round({"col1": 0}), round_0)
 165         tm.assert_frame_equal(df.round({"col1": 0, "col2": 1}), round_0)
 166         tm.assert_frame_equal(df.round({"col3": 1}), df)

 168     def test_round_with_duplicate_columns(self):
 169         # GH#11611
 170 
 171         df = pd.DataFrame(
 172             np.random.random([3, 3]),
 173             columns=["A", "B", "C"],
 174             index=["first", "second", "third"],
 175         )
 176 
 177         dfs = pd.concat((df, df), axis=1)
 178         rounded = dfs.round()
 179         tm.assert_index_equal(rounded.index, dfs.index)
 180
