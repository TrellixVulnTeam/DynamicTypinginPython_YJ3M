  20 def test_drop_duplicates():
  21     df = DataFrame(
  22         {
  23             "AAA": ["foo", "bar", "foo", "bar", "foo", "bar", "bar", "foo"],
  24             "B": ["one", "one", "two", "two", "two", "two", "one", "two"],
  25             "C": [1, 1, 2, 2, 2, 2, 1, 2],
  26             "D": range(8),
  27         }
  28     )
  29     # single column
  30     result = df.drop_duplicates("AAA")
  31     expected = df[:2]
  32     tm.assert_frame_equal(result, expected)
  33 
  34     result = df.drop_duplicates("AAA", keep="last")
  35     expected = df.loc[[6, 7]]
  36     tm.assert_frame_equal(result, expected)
  37 
  38     result = df.drop_duplicates("AAA", keep=False)
  39     expected = df.loc[[]]
  40     tm.assert_frame_equal(result, expected)
  41     assert len(result) == 0
  42 
  43     # multi column
  44     expected = df.loc[[0, 1, 2, 3]]
  45     result = df.drop_duplicates(np.array(["AAA", "B"]))
  46     tm.assert_frame_equal(result, expected)
  47     result = df.drop_duplicates(["AAA", "B"])
  48     tm.assert_frame_equal(result, expected)
  49 
  50     result = df.drop_duplicates(("AAA", "B"), keep="last")
  51     expected = df.loc[[0, 5, 6, 7]]
  52     tm.assert_frame_equal(result, expected)
  53 
  54     result = df.drop_duplicates(("AAA", "B"), keep=False)
  55     expected = df.loc[[0]]
  56     tm.assert_frame_equal(result, expected)
  57 
  58     # consider everything
  59     df2 = df.loc[:, ["AAA", "B", "C"]]
  60 
  61     result = df2.drop_duplicates()
  62     # in this case only
  63     expected = df2.drop_duplicates(["AAA", "B"])
  64     tm.assert_frame_equal(result, expected)
  65 
  66     result = df2.drop_duplicates(keep="last")
  67     expected = df2.drop_duplicates(["AAA", "B"], keep="last")
  68     tm.assert_frame_equal(result, expected)
  69 
  70     result = df2.drop_duplicates(keep=False)
  71     expected = df2.drop_duplicates(["AAA", "B"], keep=False)
  72     tm.assert_frame_equal(result, expected)
  73 
  74     # integers
  75     result = df.drop_duplicates("C")
  76     expected = df.iloc[[0, 2]]
  77     tm.assert_frame_equal(result, expected)
  78     result = df.drop_duplicates("C", keep="last")
  79     expected = df.iloc[[-2, -1]]
  80     tm.assert_frame_equal(result, expected)
  81 
  82     df["E"] = df["C"].astype("int8")
  83     result = df.drop_duplicates("E")
  84     expected = df.iloc[[0, 2]]
  85     tm.assert_frame_equal(result, expected)
  86     result = df.drop_duplicates("E", keep="last")
  87     expected = df.iloc[[-2, -1]]
  88     tm.assert_frame_equal(result, expected)
  89

  93     tm.assert_frame_equal(df.drop_duplicates(), expected)
  94

 325 def test_drop_duplicates_inplace():
 326     orig = DataFrame(
 327         {
 328             "A": ["foo", "bar", "foo", "bar", "foo", "bar", "bar", "foo"],
 329             "B": ["one", "one", "two", "two", "two", "two", "one", "two"],
 330             "C": [1, 1, 2, 2, 2, 2, 1, 2],
 331             "D": range(8),
 332         }
 333     )
 334     # single column
 335     df = orig.copy()
 336     df.drop_duplicates("A", inplace=True)
 337     expected = orig[:2]
 338     result = df
 339     tm.assert_frame_equal(result, expected)
 340 
 341     df = orig.copy()
 342     df.drop_duplicates("A", keep="last", inplace=True)
 343     expected = orig.loc[[6, 7]]
 344     result = df
 345     tm.assert_frame_equal(result, expected)
 346 
 347     df = orig.copy()
 348     df.drop_duplicates("A", keep=False, inplace=True)
 349     expected = orig.loc[[]]
 350     result = df
 351     tm.assert_frame_equal(result, expected)
 352     assert len(df) == 0
 353 
 354     # multi column
 355     df = orig.copy()
 356     df.drop_duplicates(["A", "B"], inplace=True)
 357     expected = orig.loc[[0, 1, 2, 3]]
 358     result = df
 359     tm.assert_frame_equal(result, expected)
 360 
 361     df = orig.copy()
 362     df.drop_duplicates(["A", "B"], keep="last", inplace=True)
 363     expected = orig.loc[[0, 5, 6, 7]]
 364     result = df
 365     tm.assert_frame_equal(result, expected)
 366 
 367     df = orig.copy()
 368     df.drop_duplicates(["A", "B"], keep=False, inplace=True)
 369     expected = orig.loc[[0]]
 370     result = df
 371     tm.assert_frame_equal(result, expected)
 372 
 373     # consider everything
 374     orig2 = orig.loc[:, ["A", "B", "C"]].copy()
 375 
 376     df2 = orig2.copy()
 377     df2.drop_duplicates(inplace=True)
 378     # in this case only
 379     expected = orig2.drop_duplicates(["A", "B"])
 380     result = df2
 381     tm.assert_frame_equal(result, expected)
 382 
 383     df2 = orig2.copy()
 384     df2.drop_duplicates(keep="last", inplace=True)
 385     expected = orig2.drop_duplicates(["A", "B"], keep="last")
 386     result = df2
 387     tm.assert_frame_equal(result, expected)
 388 
 389     df2 = orig2.copy()
 390     df2.drop_duplicates(keep=False, inplace=True)
 391     expected = orig2.drop_duplicates(["A", "B"], keep=False)
 392     result = df2
 393     tm.assert_frame_equal(result, expected)

 396 @pytest.mark.parametrize("inplace", [True, False])
 397 @pytest.mark.parametrize(
 398     "origin_dict, output_dict, ignore_index, output_index",
 399     [
 400         ({"A": [2, 2, 3]}, {"A": [2, 3]}, True, [0, 1]),
 401         ({"A": [2, 2, 3]}, {"A": [2, 3]}, False, [0, 2]),
 402         ({"A": [2, 2, 3], "B": [2, 2, 4]}, {"A": [2, 3], "B": [2, 4]}, True, [0, 1]),
 403         ({"A": [2, 2, 3], "B": [2, 2, 4]}, {"A": [2, 3], "B": [2, 4]}, False, [0, 2]),
 404     ],
 405 )
 406 def test_drop_duplicates_ignore_index(
 407     inplace, origin_dict, output_dict, ignore_index, output_index
 408 ):
 409     # GH 30114
 410     df = DataFrame(origin_dict)
 411     expected = DataFrame(output_dict, index=output_index)
 412 
 413     if inplace:
 414         result_df = df.copy()
 415         result_df.drop_duplicates(ignore_index=ignore_index, inplace=inplace)
 416     else:
 417         result_df = df.drop_duplicates(ignore_index=ignore_index, inplace=inplace)
 418 
 419     tm.assert_frame_equal(result_df, expected)
 420     tm.assert_frame_equal(df, DataFrame(origin_dict))
