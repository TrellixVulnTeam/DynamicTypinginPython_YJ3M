 187     # Add list-of-list constructor because list is ambiguous -> lambda
 188     # also test index name if append=True (name is duplicate here for B)
 189     @pytest.mark.parametrize(
 190         "box",
 191         [
 192             Series,
 193             Index,
 194             np.array,
 195             list,
 196             lambda x: [list(x)],
 197             lambda x: MultiIndex.from_arrays([x]),

 200     @pytest.mark.parametrize(
 201         "append, index_name", [(True, None), (True, "B"), (True, "test"), (False, None)]
 202     )
 203     @pytest.mark.parametrize("drop", [True, False])
 204     def test_set_index_pass_single_array(
 205         self, frame_of_index_cols, drop, append, index_name, box
 206     ):
 207         df = frame_of_index_cols
 208         df.index.name = index_name
 209 
 210         key = box(df["B"])
 211         if box == list:
 212             # list of strings gets interpreted as list of keys
 213             msg = "['one', 'two', 'three', 'one', 'two']"
 214             with pytest.raises(KeyError, match=msg):
 215                 df.set_index(key, drop=drop, append=append)
 216         else:
 217             # np.array/list-of-list "forget" the name of B
 218             name_mi = getattr(key, "names", None)
 219             name = [getattr(key, "name", None)] if name_mi is None else name_mi
 220 
 221             result = df.set_index(key, drop=drop, append=append)
 222 
 223             # only valid column keys are dropped
 224             # since B is always passed as array above, nothing is dropped
 225             expected = df.set_index(["B"], drop=False, append=append)
 226             expected.index.names = [index_name] + name if append else name
 227 
 228             tm.assert_frame_equal(result, expected)

 292         self, frame_of_index_cols, drop, append, index_name, box1, box2
 293     ):
 294         df = frame_of_index_cols
 295         df.index.name = index_name
 296 
 297         keys = [box1(df["A"]), box2(df["A"])]
 298         result = df.set_index(keys, drop=drop, append=append)
 299 
 300         # if either box is iter, it has been consumed; re-read

 313             )
 314             else drop
 315         )
 316         # to test against already-tested behaviour, we add sequentially,
 317         # hence second append always True; must wrap keys in list, otherwise
 318         # box = list would be interpreted as keys
 319         expected = df.set_index([keys[0]], drop=first_drop, append=append)
 320         expected = expected.set_index([keys[1]], drop=drop, append=True)
 321         tm.assert_frame_equal(result, expected)

 471 
 472         # actual example discussed in GH 24984 was e.g. for shapely.geometry
 473         # objects (e.g. a collection of Points) that can be both hashable and

 494         result = df.set_index([thing2])
 495         tm.assert_frame_equal(result, expected)
 496 
 497         # missing key
 498         thing3 = Thing(["Three", "pink"])
 499         msg = r"frozenset\(\{'Three', 'pink'\}\)"

 501             # missing label directly
