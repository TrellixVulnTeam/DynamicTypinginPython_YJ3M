  31     def test_basic(self, date_range_frame):
  32         df = date_range_frame
  33         N = 50
  34         df.loc[df.index[15:30], "A"] = np.nan
  35         dates = date_range("1/1/1990", periods=N * 3, freq="25s")
  36 
  37         result = df.asof(dates)
  38         assert result.notna().all(1).all()
  39         lb = df.index[14]
  40         ub = df.index[30]

  42         dates = list(dates)
  43 
  44         result = df.asof(dates)

  51     def test_subset(self, date_range_frame):
  52         N = 10
  53         df = date_range_frame.iloc[:N].copy()
  54         df.loc[df.index[4:8], "A"] = np.nan
  55         dates = date_range("1/1/1990", periods=N * 3, freq="25s")
  56 
  57         # with a subset of A should be the same
  58         result = df.asof(dates, subset="A")
  59         expected = df.asof(dates)
  60         tm.assert_frame_equal(result, expected)
  61 
  62         # same with A/B
  63         result = df.asof(dates, subset=["A", "B"])
  64         expected = df.asof(dates)
  65         tm.assert_frame_equal(result, expected)
  66 
  67         # B gives df.asof
  68         result = df.asof(dates, subset="B")
  69         expected = df.resample("25s", closed="right").ffill().reindex(dates)
  70         expected.iloc[20:] = 9
  71 
  72         tm.assert_frame_equal(result, expected)

  99     def test_all_nans(self, date_range_frame):
 100         # GH 15713
 101         # DataFrame is all nans
 102         result = DataFrame([np.nan]).asof([0])
 103         expected = DataFrame([np.nan])
 104         tm.assert_frame_equal(result, expected)
 105 
 106         # testing non-default indexes, multiple inputs
 107         N = 150
 108         rng = date_range_frame.index
 109         dates = date_range("1/1/1990", periods=N, freq="25s")
 110         result = DataFrame(np.nan, index=rng, columns=["A"]).asof(dates)
 111         expected = DataFrame(np.nan, index=dates, columns=["A"])
 112         tm.assert_frame_equal(result, expected)
 113 
 114         # testing multiple columns
 115         dates = date_range("1/1/1990", periods=N, freq="25s")
 116         result = DataFrame(np.nan, index=rng, columns=["A", "B", "C"]).asof(dates)
 117         expected = DataFrame(np.nan, index=dates, columns=["A", "B", "C"])
 118         tm.assert_frame_equal(result, expected)
 119 
 120         # testing scalar input
 121         result = DataFrame(np.nan, index=[1, 2], columns=["A", "B"]).asof([3])
 122         expected = DataFrame(np.nan, index=[3], columns=["A", "B"])
 123         tm.assert_frame_equal(result, expected)
 124 
 125         result = DataFrame(np.nan, index=[1, 2], columns=["A", "B"]).asof(3)
 126         expected = Series(np.nan, index=["A", "B"], name=3)
 127         tm.assert_series_equal(result, expected)
