  12         mapping = {"A": "a", "B": "b", "C": "c", "D": "d"}
  13 
  14         renamed = float_frame.rename(columns=mapping)
  15         renamed2 = float_frame.rename(columns=str.lower)
  16 
  17         tm.assert_frame_equal(renamed, renamed2)
  18         tm.assert_frame_equal(
  19             renamed2.rename(columns=str.upper), float_frame, check_names=False
  20         )
  21 
  22         # index
  23         data = {"A": {"foo": 0, "bar": 1}}
  24

  26         df = DataFrame(data)

  29 
  30         renamed = df.rename(index=str.upper)
  31         tm.assert_index_equal(renamed.index, Index(["BAR", "FOO"]))
  32

  35             float_frame.rename()
  36 
  37         # partial columns
  38         renamed = float_frame.rename(columns={"C": "foo", "D": "bar"})
  39         tm.assert_index_equal(renamed.columns, Index(["A", "B", "foo", "bar"]))
  40 
  41         # other axis
  42         renamed = float_frame.T.rename(index={"C": "foo", "D": "bar"})

 146     def test_rename_inplace(self, float_frame):

 148         assert "C" in float_frame
 149         assert "foo" not in float_frame
 150

 156         assert "foo" in float_frame

 263     def test_rename_axis_style_raises(self):
 264         # see gh-12392
 265         df = DataFrame({"A": [1, 2], "B": [1, 2]}, index=["0", "1"])
 266 
 267         # Named target and axis
 268         over_spec_msg = "Cannot specify both 'axis' and any of 'index' or 'columns'"
 269         with pytest.raises(TypeError, match=over_spec_msg):
 270             df.rename(index=str.lower, axis=1)
 271 
 272         with pytest.raises(TypeError, match=over_spec_msg):
 273             df.rename(index=str.lower, axis="columns")
 274 
 275         with pytest.raises(TypeError, match=over_spec_msg):
 276             df.rename(columns=str.lower, axis="columns")
 277 
 278         with pytest.raises(TypeError, match=over_spec_msg):
 279             df.rename(index=str.lower, axis=0)
 280 
 281         # Multiple targets and axis
 282         with pytest.raises(TypeError, match=over_spec_msg):
 283             df.rename(str.lower, index=str.lower, axis="columns")
 284 
 285         # Too many targets
 286         over_spec_msg = "Cannot specify both 'mapper' and any of 'index' or 'columns'"
 287         with pytest.raises(TypeError, match=over_spec_msg):
 288             df.rename(str.lower, index=str.lower, columns=str.lower)
 289 
 290         # Duplicates
 291         with pytest.raises(TypeError, match="multiple values"):
 292             df.rename(id, mapper=id)
