  40     def test_asfreq_fillvalue(self):
  41         # test for fill value during upsampling, related to issue 3715
  42 
  43         # setup
  44         rng = date_range("1/1/2016", periods=10, freq="2S")
  45         ts = Series(np.arange(len(rng)), index=rng)
  46         df = DataFrame({"one": ts})
  47 
  48         # insert pre-existing missing value
  49         df.loc["2016-01-01 00:00:08", "one"] = None
  50 
  51         actual_df = df.asfreq(freq="1S", fill_value=9.0)
  52         expected_df = df.asfreq(freq="1S").fillna(9.0)
  53         expected_df.loc["2016-01-01 00:00:08", "one"] = None
  54         tm.assert_frame_equal(expected_df, actual_df)
  55 
  56         expected_series = ts.asfreq(freq="1S").fillna(9.0)
  57         actual_series = ts.asfreq(freq="1S", fill_value=9.0)
  58         tm.assert_series_equal(expected_series, actual_series)
