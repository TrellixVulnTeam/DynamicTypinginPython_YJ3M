  55     def test_reset_index(self, float_frame):
  56         stacked = float_frame.stack()[::2]
  57         stacked = DataFrame({"foo": stacked, "bar": stacked})
  58 
  59         names = ["first", "second"]
  60         stacked.index.names = names
  61         deleveled = stacked.reset_index()
  62         for i, (lev, level_codes) in enumerate(
  63             zip(stacked.index.levels, stacked.index.codes)
  64         ):
  65             values = lev.take(level_codes)
  66             name = names[i]
  67             tm.assert_index_equal(values, Index(deleveled[name]))
  68 
  69         stacked.index.names = [None, None]
  70         deleveled2 = stacked.reset_index()
  71         tm.assert_series_equal(
  72             deleveled["first"], deleveled2["level_0"], check_names=False
  73         )
  74         tm.assert_series_equal(
  75             deleveled["second"], deleveled2["level_1"], check_names=False
  76         )
  77 
  78         # default name assigned
  79         rdf = float_frame.reset_index()
  80         exp = Series(float_frame.index.values, name="index")
  81         tm.assert_series_equal(rdf["index"], exp)
  82 
  83         # default name assigned, corner case
  84         df = float_frame.copy()
  85         df["index"] = "foo"
  86         rdf = df.reset_index()
  87         exp = Series(float_frame.index.values, name="level_0")
  88         tm.assert_series_equal(rdf["level_0"], exp)
  89 
  90         # but this is ok
  91         float_frame.index.name = "index"
  92         deleveled = float_frame.reset_index()
  93         tm.assert_series_equal(deleveled["index"], Series(float_frame.index))
  94         tm.assert_index_equal(deleveled.index, Index(np.arange(len(deleveled))))
  95 
  96         # preserve column names
  97         float_frame.columns.name = "columns"
  98         resetted = float_frame.reset_index()
  99         assert resetted.columns.name == "columns"
 100 
 101         # only remove certain columns
 102         df = float_frame.reset_index().set_index(["index", "A", "B"])
 103         rs = df.reset_index(["A", "B"])
 104 
 105         # TODO should reset_index check_names ?
 106         tm.assert_frame_equal(rs, float_frame, check_names=False)
 107 
 108         rs = df.reset_index(["index", "A", "B"])
 109         tm.assert_frame_equal(rs, float_frame.reset_index(), check_names=False)
 110 
 111         rs = df.reset_index(["index", "A", "B"])
 112         tm.assert_frame_equal(rs, float_frame.reset_index(), check_names=False)
 113 
 114         rs = df.reset_index("A")
 115         xp = float_frame.reset_index().set_index(["index", "B"])
 116         tm.assert_frame_equal(rs, xp, check_names=False)
 117 
 118         # test resetting in place
 119         df = float_frame.copy()
 120         resetted = float_frame.reset_index()
 121         df.reset_index(inplace=True)
 122         tm.assert_frame_equal(df, resetted, check_names=False)
 123 
 124         df = float_frame.reset_index().set_index(["index", "A", "B"])
 125         rs = df.reset_index("A", drop=True)
 126         xp = float_frame.copy()
 127         del xp["A"]
 128         xp = xp.set_index(["B"], append=True)
 129         tm.assert_frame_equal(rs, xp, check_names=False)

 142     def test_reset_index_level(self):
 143         df = DataFrame([[1, 2, 3, 4], [5, 6, 7, 8]], columns=["A", "B", "C", "D"])
 144 
 145         for levels in ["A", "B"], [0, 1]:
 146             # With MultiIndex
 147             result = df.set_index(["A", "B"]).reset_index(level=levels[0])
 148             tm.assert_frame_equal(result, df.set_index("B"))
 149 
 150             result = df.set_index(["A", "B"]).reset_index(level=levels[:1])
 151             tm.assert_frame_equal(result, df.set_index("B"))
 152 
 153             result = df.set_index(["A", "B"]).reset_index(level=levels)
 154             tm.assert_frame_equal(result, df)
 155 
 156             result = df.set_index(["A", "B"]).reset_index(level=levels, drop=True)
 157             tm.assert_frame_equal(result, df[["C", "D"]])
 158 
 159             # With single-level Index (GH 16263)
 160             result = df.set_index("A").reset_index(level=levels[0])
 161             tm.assert_frame_equal(result, df)
 162 
 163             result = df.set_index("A").reset_index(level=levels[:1])
 164             tm.assert_frame_equal(result, df)
 165 
 166             result = df.set_index(["A"]).reset_index(level=levels[0], drop=True)
 167             tm.assert_frame_equal(result, df[["B", "C", "D"]])
 168 
 169         # Missing levels - for both MultiIndex and single-level Index:
 170         for idx_lev in ["A", "B"], ["A"]:
 171             with pytest.raises(KeyError, match=r"(L|l)evel \(?E\)?"):
 172                 df.set_index(idx_lev).reset_index(level=["A", "E"])
 173             with pytest.raises(IndexError, match="Too many levels"):
 174                 df.set_index(idx_lev).reset_index(level=[0, 1, 2])

 189     def test_reset_index_multiindex_col(self):
 190         vals = np.random.randn(3, 3).astype(object)
 191         idx = ["x", "y", "z"]
 192         full = np.hstack(([[x] for x in idx], vals))
 193         df = DataFrame(
 194             vals,
 195             Index(idx, name="a"),
 196             columns=[["b", "b", "c"], ["mean", "median", "mean"]],
 197         )
 198         rs = df.reset_index()
 199         xp = DataFrame(
 200             full, columns=[["a", "b", "b", "c"], ["", "mean", "median", "mean"]]
 201         )
 202         tm.assert_frame_equal(rs, xp)
 203 
 204         rs = df.reset_index(col_fill=None)
 205         xp = DataFrame(
 206             full, columns=[["a", "b", "b", "c"], ["a", "mean", "median", "mean"]]
 207         )
 208         tm.assert_frame_equal(rs, xp)
 209 
 210         rs = df.reset_index(col_level=1, col_fill="blah")
 211         xp = DataFrame(
 212             full, columns=[["blah", "b", "b", "c"], ["a", "mean", "median", "mean"]]
 213         )
 214         tm.assert_frame_equal(rs, xp)
 215 
 216         df = DataFrame(
 217             vals,
 218             MultiIndex.from_arrays([[0, 1, 2], ["x", "y", "z"]], names=["d", "a"]),
 219             columns=[["b", "b", "c"], ["mean", "median", "mean"]],
 220         )
 221         rs = df.reset_index("a")
 222         xp = DataFrame(
 223             full,
 224             Index([0, 1, 2], name="d"),
 225             columns=[["a", "b", "b", "c"], ["", "mean", "median", "mean"]],
 226         )
 227         tm.assert_frame_equal(rs, xp)
 228 
 229         rs = df.reset_index("a", col_fill=None)
 230         xp = DataFrame(
 231             full,
 232             Index(range(3), name="d"),
 233             columns=[["a", "b", "b", "c"], ["a", "mean", "median", "mean"]],
 234         )
 235         tm.assert_frame_equal(rs, xp)
 236 
 237         rs = df.reset_index("a", col_fill="blah", col_level=1)
 238         xp = DataFrame(
 239             full,
 240             Index(range(3), name="d"),
 241             columns=[["blah", "b", "b", "c"], ["a", "mean", "median", "mean"]],
 242         )
 243         tm.assert_frame_equal(rs, xp)
