  27     def test_combine_first(self, float_frame):
  28         # disjoint
  29         head, tail = float_frame[:5], float_frame[5:]
  30 
  31         combined = head.combine_first(tail)
  32         reordered_frame = float_frame.reindex(combined.index)
  33         tm.assert_frame_equal(combined, reordered_frame)
  34         assert tm.equalContents(combined.columns, float_frame.columns)
  35         tm.assert_series_equal(combined["A"], reordered_frame["A"])
  36 
  37         # same index
  38         fcopy = float_frame.copy()
  39         fcopy["A"] = 1
  40         del fcopy["C"]
  41 
  42         fcopy2 = float_frame.copy()
  43         fcopy2["B"] = 0
  44         del fcopy2["D"]
  45 
  46         combined = fcopy.combine_first(fcopy2)
  47 
  48         assert (combined["A"] == 1).all()
  49         tm.assert_series_equal(combined["B"], fcopy["B"])
  50         tm.assert_series_equal(combined["C"], fcopy2["C"])
  51         tm.assert_series_equal(combined["D"], fcopy["D"])
  52 
  53         # overlap
  54         head, tail = reordered_frame[:10].copy(), reordered_frame
  55         head["A"] = 1
  56 
  57         combined = head.combine_first(tail)
  58         assert (combined["A"][:10] == 1).all()
  59 
  60         # reverse overlap
  61         tail["A"][:10] = 0
  62         combined = tail.combine_first(head)
  63         assert (combined["A"][:10] == 0).all()
  64 
  65         # no overlap
  66         f = float_frame[:10]
  67         g = float_frame[10:]
  68         combined = f.combine_first(g)
  69         tm.assert_series_equal(combined["A"].reindex(f.index), f["A"])
  70         tm.assert_series_equal(combined["A"].reindex(g.index), g["A"])
  71 
  72         # corner cases
  73         comb = float_frame.combine_first(DataFrame())
  74         tm.assert_frame_equal(comb, float_frame)
  75 
  76         comb = DataFrame().combine_first(float_frame)
  77         tm.assert_frame_equal(comb, float_frame)
  78 
  79         comb = float_frame.combine_first(DataFrame(index=["faz", "boo"]))
  80         assert "faz" in comb.index
  81 
  82         # #2525
  83         df = DataFrame({"a": [1]}, index=[datetime(2012, 1, 1)])
  84         df2 = DataFrame(columns=["b"])
  85         result = df.combine_first(df2)
  86         assert "b" in result

 299     def test_combine_first_period(self):
 300         data1 = pd.PeriodIndex(["2011-01", "NaT", "2011-03", "2011-04"], freq="M")
 301         df1 = pd.DataFrame({"P": data1}, index=[1, 3, 5, 7])
 302         data2 = pd.PeriodIndex(["2012-01-01", "2012-02", "2012-03"], freq="M")
 303         df2 = pd.DataFrame({"P": data2}, index=[2, 4, 5])
 304 
 305         res = df1.combine_first(df2)
 306         exp_dts = pd.PeriodIndex(
 307             ["2011-01", "2012-01", "NaT", "2012-02", "2011-03", "2011-04"], freq="M"
 308         )
 309         exp = pd.DataFrame({"P": exp_dts}, index=[1, 2, 3, 4, 5, 7])
 310         tm.assert_frame_equal(res, exp)
 311         assert res["P"].dtype == data1.dtype
 312 
 313         # different freq
 314         dts2 = pd.PeriodIndex(["2012-01-01", "2012-01-02", "2012-01-03"], freq="D")
 315         df2 = pd.DataFrame({"P": dts2}, index=[2, 4, 5])
 316 
 317         res = df1.combine_first(df2)
 318         exp_dts = [
 319             pd.Period("2011-01", freq="M"),
 320             pd.Period("2012-01-01", freq="D"),
 321             pd.NaT,
 322             pd.Period("2012-01-02", freq="D"),
 323             pd.Period("2011-03", freq="M"),
 324             pd.Period("2011-04", freq="M"),
 325         ]
 326         exp = pd.DataFrame({"P": exp_dts}, index=[1, 2, 3, 4, 5, 7])
 327         tm.assert_frame_equal(res, exp)
 328         assert res["P"].dtype == "object"
