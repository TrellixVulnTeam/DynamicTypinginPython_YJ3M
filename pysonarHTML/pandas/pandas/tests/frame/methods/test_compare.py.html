  94 def test_compare_with_equal_nulls():
  95     # We want to make sure two NaNs are considered the same
  96     # and dropped where applicable
  97     df = pd.DataFrame(
  98         {"col1": ["a", "b", "c"], "col2": [1.0, 2.0, np.nan], "col3": [1.0, 2.0, 3.0]},
  99         columns=["col1", "col2", "col3"],
 100     )
 101     df2 = df.copy()
 102     df2.loc[0, "col1"] = "c"
 103 
 104     result = df.compare(df2)
 105     indices = pd.Index([0])
 106     columns = pd.MultiIndex.from_product([["col1"], ["self", "other"]])

 108     tm.assert_frame_equal(result, expected)

 134 @pytest.mark.parametrize("align_axis", [0, 1])
 135 def test_compare_multi_index(align_axis):
 136     df = pd.DataFrame(
 137         {"col1": ["a", "b", "c"], "col2": [1.0, 2.0, np.nan], "col3": [1.0, 2.0, 3.0]}
 138     )
 139     df.columns = pd.MultiIndex.from_arrays([["a", "a", "b"], ["col1", "col2", "col3"]])
 140     df.index = pd.MultiIndex.from_arrays([["x", "x", "y"], [0, 1, 2]])
 141 
 142     df2 = df.copy()
 143     df2.iloc[0, 0] = "c"
 144     df2.iloc[2, 2] = 4.0
 145 
 146     result = df.compare(df2, align_axis=align_axis)
 147 
 148     if align_axis == 0:
 149         indices = pd.MultiIndex.from_arrays(
 150             [["x", "x", "y", "y"], [0, 0, 2, 2], ["self", "other", "self", "other"]]
 151         )
 152         columns = pd.MultiIndex.from_arrays([["a", "b"], ["col1", "col3"]])
 153         data = [["a", np.nan], ["c", np.nan], [np.nan, 3.0], [np.nan, 4.0]]
 154     else:
 155         indices = pd.MultiIndex.from_arrays([["x", "y"], [0, 2]])
 156         columns = pd.MultiIndex.from_arrays(
 157             [
 158                 ["a", "a", "b", "b"],
 159                 ["col1", "col1", "col3", "col3"],
 160                 ["self", "other", "self", "other"],
 161             ]
 162         )
 163         data = [["a", "c", np.nan, np.nan], [np.nan, np.nan, 3.0, 4.0]]
 164 
 165     expected = pd.DataFrame(data=data, index=indices, columns=columns)
 166     tm.assert_frame_equal(result, expected)

 169 def test_compare_unaligned_objects():
 170     # test DataFrames with different indices
 171     msg = "Can only compare identically-labeled DataFrame objects"
 172     with pytest.raises(ValueError, match=msg):
 173         df1 = pd.DataFrame([1, 2, 3], index=["a", "b", "c"])
 174         df2 = pd.DataFrame([1, 2, 3], index=["a", "b", "d"])
 175         df1.compare(df2)
 176 
 177     # test DataFrames with different shapes
 178     msg = "Can only compare identically-labeled DataFrame objects"
 179     with pytest.raises(ValueError, match=msg):
 180         df1 = pd.DataFrame(np.ones((3, 3)))
 181         df2 = pd.DataFrame(np.zeros((2, 1)))
 182         df1.compare(df2)
