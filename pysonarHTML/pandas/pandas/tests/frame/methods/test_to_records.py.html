 338     def test_to_records_dict_like(self):
 339         # see GH#18146
 340         class DictLike:
 341             def __init__(self, **kwargs):
 342                 self.d = kwargs.copy()
 343 
 344             def __getitem__(self, key):
 345                 return self.d.__getitem__(key)
 346 
 347             def __contains__(self, key) -> bool:
 348                 return key in self.d
 349 
 350             def keys(self):
 351                 return self.d.keys()
 352 
 353         df = DataFrame({"A": [1, 2], "B": [0.2, 1.5], "C": ["a", "bc"]})
 354 
 355         dtype_mappings = dict(
 356             column_dtypes=DictLike(**{"A": np.int8, "B": np.float32}),
 357             index_dtypes="<U2",
 358         )
 359 
 360         result = df.to_records(**dtype_mappings)
 361         expected = np.rec.array(
 362             [("0", "1", "0.2", "a"), ("1", "2", "1.5", "bc")],
 363             dtype=[("index", "<U2"), ("A", "i1"), ("B", "<f4"), ("C", "O")],
 364         )
 365         tm.assert_almost_equal(result, expected)
