  25 def _check_cast(df, v):
  26     """
  27     Check if all dtypes of df are equal to v
  28     """
  29     assert all(s.dtype.name == v for _, s in df.items())

 175     @pytest.mark.parametrize("dtype_class", [dict, Series])
 176     def test_astype_dict_like(self, dtype_class):
 177         # GH7271 & GH16717
 178         a = Series(date_range("2010-01-04", periods=5))
 179         b = Series(range(5))
 180         c = Series([0.0, 0.2, 0.4, 0.6, 0.8])
 181         d = Series(["1.0", "2", "3.14", "4", "5.4"])
 182         df = DataFrame({"a": a, "b": b, "c": c, "d": d})
 183         original = df.copy(deep=True)
 184 
 185         # change type of a subset of columns
 186         dt1 = dtype_class({"b": "str", "d": "float32"})
 187         result = df.astype(dt1)
 188         expected = DataFrame(
 189             {
 190                 "a": a,
 191                 "b": Series(["0", "1", "2", "3", "4"]),
 192                 "c": c,
 193                 "d": Series([1.0, 2.0, 3.14, 4.0, 5.4], dtype="float32"),
 194             }
 195         )
 196         tm.assert_frame_equal(result, expected)
 197         tm.assert_frame_equal(df, original)
 198 
 199         dt2 = dtype_class({"b": np.float32, "c": "float32", "d": np.float64})
 200         result = df.astype(dt2)
 201         expected = DataFrame(
 202             {
 203                 "a": a,
 204                 "b": Series([0.0, 1.0, 2.0, 3.0, 4.0], dtype="float32"),
 205                 "c": Series([0.0, 0.2, 0.4, 0.6, 0.8], dtype="float32"),
 206                 "d": Series([1.0, 2.0, 3.14, 4.0, 5.4], dtype="float64"),
 207             }
 208         )
 209         tm.assert_frame_equal(result, expected)
 210         tm.assert_frame_equal(df, original)
 211 
 212         # change all columns
 213         dt3 = dtype_class({"a": str, "b": str, "c": str, "d": str})
 214         tm.assert_frame_equal(df.astype(dt3), df.astype(str))
 215         tm.assert_frame_equal(df, original)
 216 
 217         # error should be raised when using something other than column labels
 218         # in the keys of the dtype dict
 219         dt4 = dtype_class({"b": str, 2: str})
 220         dt5 = dtype_class({"e": str})
 221         msg = "Only a column name can be used for the key in a dtype mappings argument"
 222         with pytest.raises(KeyError, match=msg):
 223             df.astype(dt4)
 224         with pytest.raises(KeyError, match=msg):
 225             df.astype(dt5)
 226         tm.assert_frame_equal(df, original)
 227 
 228         # if the dtypes provided are the same as the original dtypes, the
 229         # resulting DataFrame should be the same as the original DataFrame
 230         dt6 = dtype_class({col: df[col].dtype for col in df.columns})
 231         equiv = df.astype(dt6)
 232         tm.assert_frame_equal(df, equiv)
 233         tm.assert_frame_equal(df, original)
 234 
 235         # GH#16717
 236         # if dtypes provided is empty, the resulting DataFrame
 237         # should be the same as the original DataFrame
 238         dt7 = dtype_class({}) if dtype_class is dict else dtype_class({}, dtype=object)
 239         equiv = df.astype(dt7)
 240         tm.assert_frame_equal(df, equiv)
 241         tm.assert_frame_equal(df, original)
