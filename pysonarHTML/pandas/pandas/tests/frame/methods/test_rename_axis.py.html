  40     def test_rename_axis_mapper(self):
  41         # GH#19978
  42         mi = MultiIndex.from_product([["a", "b", "c"], [1, 2]], names=["ll", "nn"])
  43         df = DataFrame(

  46 
  47         # Test for rename of the Index object of columns

  49         tm.assert_index_equal(result.columns, Index(["x", "y"], name="cols"))
  50 
  51         # Test for rename of the Index object of columns using dict
  52         result = result.rename_axis(columns={"cols": "new"}, axis=1)
  53         tm.assert_index_equal(result.columns, Index(["x", "y"], name="new"))
  54 
  55         # Test for renaming index using dict
  56         result = df.rename_axis(index={"ll": "foo"})
  57         assert result.index.names == ["foo", "nn"]
  58 
  59         # Test for renaming index using a function

  61         assert result.index.names == ["LL", "NN"]
  62 
  63         # Test for renaming index providing complete list
  64         result = df.rename_axis(index=["foo", "goo"])
  65         assert result.index.names == ["foo", "goo"]
  66 
  67         # Test for changing index and columns at same time
