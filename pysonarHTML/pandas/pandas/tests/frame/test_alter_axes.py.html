 138     def test_reindex_api_equivalence(self):
 139         # equivalence of the labels/axis and index/columns API's
 140         df = DataFrame(
 141             [[1, 2, 3], [3, 4, 5], [5, 6, 7]],
 142             index=["a", "b", "c"],
 143             columns=["d", "e", "f"],
 144         )
 145 
 146         res1 = df.reindex(["b", "a"])
 147         res2 = df.reindex(index=["b", "a"])
 148         res3 = df.reindex(labels=["b", "a"])
 149         res4 = df.reindex(labels=["b", "a"], axis=0)
 150         res5 = df.reindex(["b", "a"], axis=0)
 151         for res in [res2, res3, res4, res5]:
 152             tm.assert_frame_equal(res1, res)
 153 
 154         res1 = df.reindex(columns=["e", "d"])
 155         res2 = df.reindex(["e", "d"], axis=1)
 156         res3 = df.reindex(labels=["e", "d"], axis=1)
 157         for res in [res2, res3]:
 158             tm.assert_frame_equal(res1, res)
 159 
 160         res1 = df.reindex(index=["b", "a"], columns=["e", "d"])
 161         res2 = df.reindex(columns=["e", "d"], index=["b", "a"])
 162         res3 = df.reindex(labels=["b", "a"], axis=0).reindex(labels=["e", "d"], axis=1)
 163         for res in [res2, res3]:
 164             tm.assert_frame_equal(res1, res)

 222 class TestIntervalIndex:
 223     def test_setitem(self):
 224 
 225         df = DataFrame({"A": range(10)})
 226         s = cut(df.A, 5)
 227         assert isinstance(s.cat.categories, IntervalIndex)
 228 
 229         # B & D end up as Categoricals
 230         # the remainer are converted to in-line objects
 231         # contining an IntervalIndex.values
 232         df["B"] = s
 233         df["C"] = np.array(s)
 234         df["D"] = s.values
 235         df["E"] = np.array(s.values)
 236 
 237         assert is_categorical_dtype(df["B"].dtype)
 238         assert is_interval_dtype(df["B"].cat.categories)
 239         assert is_categorical_dtype(df["D"].dtype)
 240         assert is_interval_dtype(df["D"].cat.categories)
 241 
 242         assert is_object_dtype(df["C"])
 243         assert is_object_dtype(df["E"])
 244 
 245         # they compare equal as Index
 246         # when converted to numpy objects
 247         c = lambda x: Index(np.array(x))
 248         tm.assert_index_equal(c(df.B), c(df.B), check_names=False)
 249         tm.assert_index_equal(c(df.B), c(df.C), check_names=False)
 250         tm.assert_index_equal(c(df.B), c(df.D), check_names=False)
 251         tm.assert_index_equal(c(df.B), c(df.D), check_names=False)
 252 
 253         # B & D are the same Series
 254         tm.assert_series_equal(df["B"], df["B"], check_names=False)
 255         tm.assert_series_equal(df["B"], df["D"], check_names=False)
 256 
 257         # C & E are the same Series
 258         tm.assert_series_equal(df["C"], df["C"], check_names=False)
 259         tm.assert_series_equal(df["C"], df["E"], check_names=False)
