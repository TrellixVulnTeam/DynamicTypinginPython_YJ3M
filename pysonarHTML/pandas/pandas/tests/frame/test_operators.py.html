 132     def test_logical_ops_empty_frame(self):
 133         # GH#5808
 134         # empty frames, non-mixed dtype
 135         df = DataFrame(index=[1])
 136 
 137         result = df & df
 138         tm.assert_frame_equal(result, df)
 139 
 140         result = df | df
 141         tm.assert_frame_equal(result, df)
 142 
 143         df2 = DataFrame(index=[1, 2])
 144         result = df & df2
 145         tm.assert_frame_equal(result, df2)
 146 
 147         dfa = DataFrame(index=[1], columns=["A"])
 148 
 149         result = dfa & dfa
 150         expected = DataFrame(False, index=[1], columns=["A"])
 151         tm.assert_frame_equal(result, expected)

 190     def test_logical_operators(self):
 191         def _check_bin_op(op):
 192             result = op(df1, df2)
 193             expected = DataFrame(
 194                 op(df1.values, df2.values), index=df1.index, columns=df1.columns

 230     def test_logical_with_nas(self):
 231         d = DataFrame({"a": [np.nan, False], "b": [True, True]})
 232 
 233         # GH4947
 234         # bool comparisons should return bool
 235         result = d["a"] | d["b"]
 236         expected = Series([False, True])
 237         tm.assert_series_equal(result, expected)
 238 
 239         # GH4604, automatic casting here
 240         result = d["a"].fillna(False) | d["b"]
 241         expected = Series([True, True])
 242         tm.assert_series_equal(result, expected)
 243 
 244         result = d["a"].fillna(False, downcast=False) | d["b"]
 245         expected = Series([True, True])
 246         tm.assert_series_equal(result, expected)
