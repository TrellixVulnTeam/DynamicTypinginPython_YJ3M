  19         df = DataFrame({"A": date_range("2000", periods=4, tz=tz)})

  55         test1 = DataFrame(
  56             np.zeros((6, 3)),
  57             index=date_range(

  66             columns=range(3, 6),

  75     def test_frame_align_aware(self):
  76         idx1 = date_range("2001", periods=5, freq="H", tz="US/Eastern")
  77         idx2 = date_range("2001", periods=5, freq="2H", tz="US/Eastern")
  78         df1 = DataFrame(np.random.randn(len(idx1), 3), idx1)
  79         df2 = DataFrame(np.random.randn(len(idx2), 3), idx2)
  80         new1, new2 = df1.align(df2)
  81         assert df1.index.tz == new1.index.tz
  82         assert df2.index.tz == new2.index.tz
  83 
  84         # different timezones convert to UTC
  85 
  86         # frame with frame
  87         df1_central = df1.tz_convert("US/Central")
  88         new1, new2 = df1.align(df1_central)
  89         assert new1.index.tz == pytz.UTC
  90         assert new2.index.tz == pytz.UTC
  91 
  92         # frame with Series
  93         new1, new2 = df1.align(df1_central[0], axis=0)
  94         assert new1.index.tz == pytz.UTC
  95         assert new2.index.tz == pytz.UTC
  96 
  97         df1[0].align(df1_central, axis=0)
  98         assert new1.index.tz == pytz.UTC
  99         assert new2.index.tz == pytz.UTC

 101     @pytest.mark.parametrize("tz", ["US/Eastern", "dateutil/US/Eastern"])
 102     def test_frame_no_datetime64_dtype(self, tz):
 103         # after GH#7822
 104         # these retain the timezones on dict construction
 105         dr = date_range("2011/1/1", "2012/1/1", freq="W-FRI")
 106         dr_tz = dr.tz_localize(tz)
 107         df = DataFrame({"A": "foo", "B": dr_tz}, index=dr)
 108         tz_expected = DatetimeTZDtype("ns", dr_tz.tzinfo)
 109         assert df["B"].dtype == tz_expected
 110 
 111         # GH#2810 (with timezones)
 112         datetimes_naive = [ts.to_pydatetime() for ts in dr]
 113         datetimes_with_tz = [ts.to_pydatetime() for ts in dr_tz]
 114         df = DataFrame({"dr": dr})
 115         df["dr_tz"] = dr_tz
 116         df["datetimes_naive"] = datetimes_naive
 117         df["datetimes_with_tz"] = datetimes_with_tz
 118         result = df.dtypes
 119         expected = Series(
 120             [
 121                 np.dtype("datetime64[ns]"),
 122                 DatetimeTZDtype(tz=tz),
 123                 np.dtype("datetime64[ns]"),
 124                 DatetimeTZDtype(tz=tz),
 125             ],
 126             index=["dr", "dr_tz", "datetimes_naive", "datetimes_with_tz"],
 127         )
 128         tm.assert_series_equal(result, expected)

 130     @pytest.mark.parametrize("tz", ["US/Eastern", "dateutil/US/Eastern"])
 131     def test_frame_reset_index(self, tz):
 132         dr = date_range("2012-06-02", periods=10, tz=tz)
 133         df = DataFrame(np.random.randn(len(dr)), dr)
 134         roundtripped = df.reset_index().set_index("index")
 135         xp = df.index.tz
 136         rs = roundtripped.index.tz
 137         assert xp == rs

 139     @pytest.mark.parametrize("tz", [None, "America/New_York"])
 140     def test_boolean_compare_transpose_tzindex_with_dst(self, tz):
 141         # GH 19970
 142         idx = date_range("20161101", "20161130", freq="4H", tz=tz)
