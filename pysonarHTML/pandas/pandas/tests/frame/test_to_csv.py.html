 184     def test_to_csv_new_dupe_cols(self):
 185         import pandas as pd
 186 
 187         def _check_df(df, cols=None):
 188             with tm.ensure_clean() as path:
 189                 df.to_csv(path, columns=cols, chunksize=chunksize)
 190                 rs_c = pd.read_csv(path, index_col=0)
 191 
 192                 # we wrote them in a different order
 193                 # so compare them in that order
 194                 if cols is not None:
 195 
 196                     if df.columns.is_unique:
 197                         rs_c.columns = cols
 198                     else:
 199                         indexer, missing = df.columns.get_indexer_non_unique(cols)
 200                         rs_c.columns = df.columns.take(indexer)
 201 
 202                     for c in cols:
 203                         obj_df = df[c]
 204                         obj_rs = rs_c[c]
 205                         if isinstance(obj_df, Series):
 206                             tm.assert_series_equal(obj_df, obj_rs)
 207                         else:
 208                             tm.assert_frame_equal(obj_df, obj_rs, check_names=False)
 209 
 210                 # wrote in the same order
 211                 else:
 212                     rs_c.columns = df.columns
 213                     tm.assert_frame_equal(df, rs_c, check_names=False)
 214 
 215         chunksize = 5
 216         N = int(chunksize * 2.5)
 217 
 218         # dupe cols
 219         df = tm.makeCustomDataframe(N, 3)
 220         df.columns = ["a", "a", "b"]
 221         _check_df(df, None)
 222 
 223         # dupe cols with selection
 224         cols = ["b", "a"]
