  79 def test_join_index(float_frame):
  80     # left / right
  81 
  82     f = float_frame.loc[float_frame.index[:10], ["A", "B"]]
  83     f2 = float_frame.loc[float_frame.index[5:], ["C", "D"]].iloc[::-1]
  84 
  85     joined = f.join(f2)
  86     tm.assert_index_equal(f.index, joined.index)
  87     expected_columns = Index(["A", "B", "C", "D"])
  88     tm.assert_index_equal(joined.columns, expected_columns)
  89 
  90     joined = f.join(f2, how="left")
  91     tm.assert_index_equal(joined.index, f.index)
  92     tm.assert_index_equal(joined.columns, expected_columns)
  93 
  94     joined = f.join(f2, how="right")
  95     tm.assert_index_equal(joined.index, f2.index)
  96     tm.assert_index_equal(joined.columns, expected_columns)
  97 
  98     # inner
  99 
 100     joined = f.join(f2, how="inner")
 101     tm.assert_index_equal(joined.index, f.index[5:10])
 102     tm.assert_index_equal(joined.columns, expected_columns)
 103 
 104     # outer
 105 
 106     joined = f.join(f2, how="outer")
 107     tm.assert_index_equal(joined.index, float_frame.index.sort_values())
 108     tm.assert_index_equal(joined.columns, expected_columns)
 109 
 110     with pytest.raises(ValueError, match="join method"):
 111         f.join(f2, how="foo")
 112 
 113     # corner case - overlapping columns
 114     msg = "columns overlap but no suffix"
 115     for how in ("outer", "left", "inner"):
 116         with pytest.raises(ValueError, match=msg):
 117             float_frame.join(float_frame, how=how)

 120 def test_join_index_more(float_frame):
 121     af = float_frame.loc[:, ["A", "B"]]
 122     bf = float_frame.loc[::2, ["C", "D"]]
 123 
 124     expected = af.copy()
 125     expected["C"] = float_frame["C"][::2]
 126     expected["D"] = float_frame["D"][::2]
 127 
 128     result = af.join(bf)
 129     tm.assert_frame_equal(result, expected)

 132     tm.assert_frame_equal(result, expected[::2])
 133 
 134     result = bf.join(af, how="right")
 135     tm.assert_frame_equal(result, expected.loc[:, result.columns])

 151 def test_join_overlap(float_frame):
 152     df1 = float_frame.loc[:, ["A", "B", "C"]]
 153     df2 = float_frame.loc[:, ["B", "C", "D"]]
 154 
 155     joined = df1.join(df2, lsuffix="_df1", rsuffix="_df2")
 156     df1_suf = df1.loc[:, ["B", "C"]].add_suffix("_df1")
 157     df2_suf = df2.loc[:, ["B", "C"]].add_suffix("_df2")
 158 
 159     no_overlap = float_frame.loc[:, ["A", "D"]]
 160     expected = df1_suf.join(df2_suf).join(no_overlap)
 161 
 162     # column order not necessarily sorted
 163     tm.assert_frame_equal(joined, expected.loc[:, joined.columns])

 201 @pytest.mark.parametrize("sort_kw", [True, False])
 202 def test_suppress_future_warning_with_sort_kw(sort_kw):
 203     a = DataFrame({"col1": [1, 2]}, index=["c", "a"])
 204 
 205     b = DataFrame({"col2": [4, 5]}, index=["b", "a"])
 206 
 207     c = DataFrame({"col3": [7, 8]}, index=["a", "b"])
 208 
 209     expected = DataFrame(
 210         {
 211             "col1": {"a": 2.0, "b": float("nan"), "c": 1.0},
 212             "col2": {"a": 5.0, "b": 4.0, "c": float("nan")},
 213             "col3": {"a": 7.0, "b": 8.0, "c": float("nan")},
 214         }
 215     )
 216     if sort_kw is False:
 217         expected = expected.reindex(index=["c", "a", "b"])
 218 
 219     with tm.assert_produces_warning(None, check_stacklevel=False):
 220         result = a.join([b, c], how="outer", sort=sort_kw)
 221     tm.assert_frame_equal(result, expected)

 236     def test_join_multiindex_leftright(self):
 237         # GH 10741
 238         df1 = pd.DataFrame(
 239             [
 240                 ["a", "x", 0.471780],
 241                 ["a", "y", 0.774908],
 242                 ["a", "z", 0.563634],
 243                 ["b", "x", -0.353756],
 244                 ["b", "y", 0.368062],
 245                 ["b", "z", -1.721840],
 246                 ["c", "x", 1],
 247                 ["c", "y", 2],
 248                 ["c", "z", 3],
 249             ],
 250             columns=["first", "second", "value1"],
 251         ).set_index(["first", "second"])
 252 
 253         df2 = pd.DataFrame(
 254             [["a", 10], ["b", 20]], columns=["first", "value2"]
 255         ).set_index(["first"])
 256 
 257         exp = pd.DataFrame(
 258             [
 259                 [0.471780, 10],
 260                 [0.774908, 10],
 261                 [0.563634, 10],
 262                 [-0.353756, 20],
 263                 [0.368062, 20],
 264                 [-1.721840, 20],
 265                 [1.000000, np.nan],
 266                 [2.000000, np.nan],
 267                 [3.000000, np.nan],
 268             ],
 269             index=df1.index,
 270             columns=["value1", "value2"],
 271         )
 272 
 273         # these must be the same results (but columns are flipped)
 274         tm.assert_frame_equal(df1.join(df2, how="left"), exp)
 275         tm.assert_frame_equal(df2.join(df1, how="right"), exp[["value2", "value1"]])
 276 
 277         exp_idx = pd.MultiIndex.from_product(
 278             [["a", "b"], ["x", "y", "z"]], names=["first", "second"]
 279         )
 280         exp = pd.DataFrame(
 281             [
 282                 [0.471780, 10],
 283                 [0.774908, 10],
 284                 [0.563634, 10],
 285                 [-0.353756, 20],
 286                 [0.368062, 20],
 287                 [-1.721840, 20],
 288             ],
 289             index=exp_idx,
 290             columns=["value1", "value2"],
 291         )
 292 
 293         tm.assert_frame_equal(df1.join(df2, how="right"), exp)
 294         tm.assert_frame_equal(df2.join(df1, how="left"), exp[["value2", "value1"]])
