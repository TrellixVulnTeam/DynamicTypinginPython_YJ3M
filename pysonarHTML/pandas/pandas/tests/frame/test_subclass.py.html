  12     def test_frame_subclassing_and_slicing(self):
  13         # Subclass frame and ensure it returns the right class on slicing it
  14         # In reference to PR 9632
  15 
  16         class CustomSeries(Series):
  17             @property
  18             def _constructor(self):
  19                 return CustomSeries
  20 
  21             def custom_series_function(self):
  22                 return "OK"
  23 
  24         class CustomDataFrame(DataFrame):
  25             """
  26             Subclasses pandas DF, fills DF with simulation results, adds some
  27             custom plotting functions.
  28             """
  29 
  30             def __init__(self, *args, **kw):
  31                 super().__init__(*args, **kw)
  32 
  33             @property
  34             def _constructor(self):
  35                 return CustomDataFrame
  36 
  37             _constructor_sliced = CustomSeries
  38 
  39             def custom_frame_function(self):
  40                 return "OK"
  41 
  42         data = {"col1": range(10), "col2": range(10)}
  43         cdf = CustomDataFrame(data)
  44 
  45         # Did we get back our own DF class?
  46         assert isinstance(cdf, CustomDataFrame)
  47 
  48         # Do we get back our own Series class after selecting a column?
  49         cdf_series = cdf.col1
  50         assert isinstance(cdf_series, CustomSeries)
  51         assert cdf_series.custom_series_function() == "OK"
  52 
  53         # Do we get back our own DF class after slicing row-wise?
  54         cdf_rows = cdf[1:5]
  55         assert isinstance(cdf_rows, CustomDataFrame)
  56         assert cdf_rows.custom_frame_function() == "OK"
  57 
  58         # Make sure sliced part of multi-index frame is custom class
  59         mcol = pd.MultiIndex.from_tuples([("A", "A"), ("A", "B")])
  60         cdf_multi = CustomDataFrame([[0, 1], [2, 3]], columns=mcol)
  61         assert isinstance(cdf_multi["A"], CustomDataFrame)
  62 
  63         mcol = pd.MultiIndex.from_tuples([("A", ""), ("B", "")])
  64         cdf_multi2 = CustomDataFrame([[0, 1], [2, 3]], columns=mcol)
  65         assert isinstance(cdf_multi2["A"], CustomSeries)
