  14 def _check_cast(df, v):
  15     """
  16     Check if all dtypes of df are equal to v
  17     """
  18     assert all(s.dtype.name == v for _, s in df.items())

  22     def test_concat_empty_dataframe_dtypes(self):
  23         df = DataFrame(columns=list("abc"))
  24         df["a"] = df["a"].astype(np.bool_)
  25         df["b"] = df["b"].astype(np.int32)
  26         df["c"] = df["c"].astype(np.float64)
  27 
  28         result = pd.concat([df, df])
  29         assert result["a"].dtype == np.bool_
  30         assert result["b"].dtype == np.int32
  31         assert result["c"].dtype == np.float64
  32 
  33         result = pd.concat([df, df.astype(np.float64)])
  34         assert result["a"].dtype == np.object_
  35         assert result["b"].dtype == np.float64
  36         assert result["c"].dtype == np.float64

 126     def test_timedeltas(self):
 127         df = DataFrame(
 128             dict(
 129                 A=Series(date_range("2012-1-1", periods=3, freq="D")),
 130                 B=Series([timedelta(days=i) for i in range(3)]),
 131             )
 132         )
 133         result = df.dtypes
 134         expected = Series(
 135             [np.dtype("datetime64[ns]"), np.dtype("timedelta64[ns]")], index=list("AB")
 136         )
 137         tm.assert_series_equal(result, expected)
 138 
 139         df["C"] = df["A"] + df["B"]
 140         result = df.dtypes
 141         expected = Series(
 142             [
 143                 np.dtype("datetime64[ns]"),
 144                 np.dtype("timedelta64[ns]"),
 145                 np.dtype("datetime64[ns]"),
 146             ],
 147             index=list("ABC"),
 148         )
 149         tm.assert_series_equal(result, expected)
 150 
 151         # mixed int types
 152         df["D"] = 1
 153         result = df.dtypes
 154         expected = Series(
 155             [
 156                 np.dtype("datetime64[ns]"),
 157                 np.dtype("timedelta64[ns]"),
 158                 np.dtype("datetime64[ns]"),
 159                 np.dtype("int64"),
 160             ],
 161             index=list("ABCD"),
 162         )
 163         tm.assert_series_equal(result, expected)

 245     def test_str_to_small_float_conversion_type(self):
 246         # GH 20388
 247         np.random.seed(13)
 248         col_data = [str(np.random.random() * 1e-12) for _ in range(5)]
 249         result = pd.DataFrame(col_data, columns=["A"])
 250         expected = pd.DataFrame(col_data, columns=["A"], dtype=object)
 251         tm.assert_frame_equal(result, expected)
 252         # change the dtype of the elements from object to float one by one
 253         result.loc[result.index, "A"] = [float(x) for x in col_data]
 254         expected = pd.DataFrame(col_data, columns=["A"], dtype=float)
 255         tm.assert_frame_equal(result, expected)
