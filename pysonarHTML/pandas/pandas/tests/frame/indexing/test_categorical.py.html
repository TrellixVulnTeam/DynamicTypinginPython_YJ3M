 360     def test_setitem_single_row_categorical(self):
 361         # GH 25495
 362         df = DataFrame({"Alpha": ["a"], "Numeric": [0]})

 370     def test_loc_indexing_preserves_index_category_dtype(self):
 371         # GH 15166
 372         df = DataFrame(
 373             data=np.arange(2, 22, 2),
 374             index=pd.MultiIndex(
 375                 levels=[pd.CategoricalIndex(["a", "b"]), range(10)],
 376                 codes=[[0] * 5 + [1] * 5, range(10)],
 377                 names=["Index1", "Index2"],
 378             ),
 379         )
 380 
 381         expected = pd.CategoricalIndex(
 382             ["a", "b"],
 383             categories=["a", "b"],
 384             ordered=False,
 385             name="Index1",
 386             dtype="category",
 387         )
 388 
 389         result = df.index.levels[0]
 390         tm.assert_index_equal(result, expected)
 391 
 392         result = df.loc[["a"]].index.levels[0]
 393         tm.assert_index_equal(result, expected)
