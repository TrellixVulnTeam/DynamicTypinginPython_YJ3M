  25 def _safe_add(df):
  26     # only add to the numeric items
  27     def is_ok(s):
  28         return (
  29             issubclass(s.dtype.type, (np.integer, np.floating)) and s.dtype != "uint8"
  30         )
  31 
  32     return DataFrame(dict((c, s + 1) if is_ok(s) else (c, s) for c, s in df.items()))

  86     def test_where_alignment(self, where_frame, float_string_frame):
  87         # aligning
  88         def _check_align(df, cond, other, check_dtypes=True):
  89             rs = df.where(cond, other)
  90             for i, k in enumerate(rs.columns):
  91                 result = rs[k]
  92                 d = df[k].values
  93                 c = cond[k].reindex(df[k].index).fillna(False).values
  94 
  95                 if is_scalar(other):
  96                     o = other
  97                 else:
  98                     if isinstance(other, np.ndarray):
  99                         o = Series(other[:, i], index=result.index).values
 100                     else:
 101                         o = other[k].values
 102 
 103                 new_values = d if c.all() else np.where(c, d, o)
 104                 expected = Series(new_values, index=result.index, name=k)
 105 
 106                 # since we can't always have the correct numpy dtype
 107                 # as numpy doesn't know how to downcast, don't check
 108                 tm.assert_series_equal(result, expected, check_dtype=False)
 109 
 110             # dtypes
 111             # can't check dtype when other is an ndarray
 112 
 113             if check_dtypes and not isinstance(other, np.ndarray):
 114                 assert (rs.dtypes == df.dtypes).all()
 115 
 116         df = where_frame
 117         if df is float_string_frame:
 118             msg = "'>' not supported between instances of 'str' and 'int'"
 119             with pytest.raises(TypeError, match=msg):
 120                 df > 0
 121             return
 122 
 123         # other is a frame
 124         cond = (df > 0)[1:]
 125         _check_align(df, cond, _safe_add(df))
 126 
 127         # check other is ndarray
 128         cond = df > 0
 129         _check_align(df, cond, (_safe_add(df).values))
 130 
 131         # integers are upcast, so don't check the dtypes
 132         cond = df > 0
 133         check_dtypes = all(not issubclass(s.type, np.integer) for s in df.dtypes)
 134         _check_align(df, cond, np.nan, check_dtypes=check_dtypes)

 157     def test_where_set(self, where_frame, float_string_frame):
 158         # where inplace
 159 
 160         def _check_set(df, cond, check_dtypes=True):
 161             dfi = df.copy()
 162             econd = cond.reindex_like(df).fillna(True)
 163             expected = dfi.mask(~econd)
 164 
 165             dfi.where(cond, np.nan, inplace=True)
 166             tm.assert_frame_equal(dfi, expected)
 167 
 168             # dtypes (and confirm upcasts)x
 169             if check_dtypes:
 170                 for k, v in df.dtypes.items():
 171                     if issubclass(v.type, np.integer) and not cond[k].all():
 172                         v = np.dtype("float64")
 173                     assert dfi[k].dtype == v
 174 
 175         df = where_frame
 176         if df is float_string_frame:
 177             msg = "'>' not supported between instances of 'str' and 'int'"
 178             with pytest.raises(TypeError, match=msg):
 179                 df > 0
 180             return
 181 
 182         cond = df > 0
 183         _check_set(df, cond)
 184 
 185         cond = df >= 0
 186         _check_set(df, cond)
 187 
 188         # aligning
 189         cond = (df >= 0)[1:]
 190         _check_set(df, cond)
