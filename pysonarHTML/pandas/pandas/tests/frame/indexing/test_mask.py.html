  11 class TestDataFrameMask:
  12     def test_mask(self):
  13         df = DataFrame(np.random.randn(5, 3))
  14         cond = df > 0
  15 
  16         rs = df.where(cond, np.nan)
  17         tm.assert_frame_equal(rs, df.mask(df <= 0))
  18         tm.assert_frame_equal(rs, df.mask(~cond))
  19 
  20         other = DataFrame(np.random.randn(5, 3))
  21         rs = df.where(cond, other)
  22         tm.assert_frame_equal(rs, df.mask(df <= 0, other))
  23         tm.assert_frame_equal(rs, df.mask(~cond, other))
  24 
  25         # see GH#21891
  26         df = DataFrame([1, 2])
  27         res = df.mask([[True], [False]])
  28 
  29         exp = DataFrame([np.nan, 2])
  30         tm.assert_frame_equal(res, exp)

  32     def test_mask_inplace(self):
  33         # GH#8801
  34         df = DataFrame(np.random.randn(5, 3))
  35         cond = df > 0
  36 
  37         rdf = df.copy()
  38 
  39         rdf.where(cond, inplace=True)
  40         tm.assert_frame_equal(rdf, df.where(cond))
  41         tm.assert_frame_equal(rdf, df.mask(~cond))
  42 
  43         rdf = df.copy()
  44         rdf.where(cond, -df, inplace=True)
  45         tm.assert_frame_equal(rdf, df.where(cond, -df))
  46         tm.assert_frame_equal(rdf, df.mask(~cond, -df))

  55     def test_mask_callable(self):
  56         # GH#12533
  57         df = DataFrame([[1, 2, 3], [4, 5, 6], [7, 8, 9]])
  58         result = df.mask(lambda x: x > 4, lambda x: x + 1)
  59         exp = DataFrame([[1, 2, 3], [4, 6, 7], [8, 9, 10]])
  60         tm.assert_frame_equal(result, exp)
  61         tm.assert_frame_equal(result, df.mask(df > 4, df + 1))
  62 
  63         # return ndarray and scalar
  64         result = df.mask(lambda x: (x % 2 == 0).values, lambda x: 99)
  65         exp = DataFrame([[1, 99, 3], [99, 5, 99], [7, 99, 9]])
  66         tm.assert_frame_equal(result, exp)
  67         tm.assert_frame_equal(result, df.mask(df % 2 == 0, 99))
  68 
  69         # chain
  70         result = (df + 2).mask(lambda x: x > 8, lambda x: x + 10)
  71         exp = DataFrame([[3, 4, 5], [6, 7, 8], [19, 20, 21]])
  72         tm.assert_frame_equal(result, exp)
  73         tm.assert_frame_equal(result, (df + 2).mask((df + 2) > 8, (df + 2) + 10))

  75     def test_mask_dtype_conversion(self):
  76         # GH#3733
  77         df = DataFrame(data=np.random.randn(100, 50))
  78         df = df.where(df > 0)  # create nans
  79         bools = df > 0
  80         mask = isna(df)
  81         expected = bools.astype(float).mask(mask)
  82         result = bools.mask(mask)
  83         tm.assert_frame_equal(result, expected)
