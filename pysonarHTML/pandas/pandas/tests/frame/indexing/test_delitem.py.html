   9 class TestDataFrameDelItem:
  10     def test_delitem(self, float_frame):
  11         del float_frame["A"]
  12         assert "A" not in float_frame
  13 
  14     def test_delitem_multiindex(self):
  15         midx = MultiIndex.from_product([["A", "B"], [1, 2]])
  16         df = DataFrame(np.random.randn(4, 4), columns=midx)
  17         assert len(df.columns) == 4
  18         assert ("A",) in df.columns
  19         assert "A" in df.columns
  20 
  21         result = df["A"]
  22         assert isinstance(result, DataFrame)
  23         del df["A"]
  24 
  25         assert len(df.columns) == 2
  26 
  27         # A still in the levels, BUT get a KeyError if trying
  28         # to delete
  29         assert ("A",) not in df.columns
  30         with pytest.raises(KeyError, match=re.escape("('A',)")):
  31             del df[("A",)]
  32 
  33         # behavior of dropped/deleted MultiIndex levels changed from
  34         # GH 2770 to GH 19027: MultiIndex no longer '.__contains__'
  35         # levels which are dropped/deleted
  36         assert "A" not in df.columns
  37         with pytest.raises(KeyError, match=re.escape("('A',)")):
  38             del df["A"]
  39 
  40     def test_delitem_corner(self, float_frame):
  41         f = float_frame.copy()
  42         del f["D"]
  43         assert len(f.columns) == 3
  44         with pytest.raises(KeyError, match=r"^'D'$"):
  45             del f["D"]
  46         del f["B"]
  47         assert len(f.columns) == 2

  49     def test_delitem_col_still_multiindex(self):
  50         arrays = [["a", "b", "c", "top"], ["", "", "", "OD"], ["", "", "", "wx"]]
  51 
  52         tuples = sorted(zip(*arrays))
  53         index = MultiIndex.from_tuples(tuples)
  54 
  55         df = DataFrame(np.random.randn(3, 4), columns=index)
  56         del df[("a", "", "")]
  57         assert isinstance(df.columns, MultiIndex)
