  49         buf = StringIO()
  50

  52         repr(float_frame)
  53         float_frame.info(verbose=False, buf=buf)

  57         float_frame.reindex(columns=["A", "B"]).info(verbose=False, buf=buf)
  58 
  59         # exhausting cases in DataFrame.info
  60 
  61         # columns but no index
  62         no_index = DataFrame(columns=[0, 1, 3])
  63         repr(no_index)

  65         # no columns or index
  66         DataFrame().info(buf=buf)

 190     def test_repr_categorical_dates_periods(self):

 202         assert repr(df) == exp
