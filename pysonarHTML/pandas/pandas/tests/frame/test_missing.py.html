 327     def test_na_actions_categorical(self):
 328 
 329         cat = Categorical([1, 2, 3, np.nan], categories=[1, 2, 3])
 330         vals = ["a", "b", np.nan, "d"]
 331         df = DataFrame({"cats": cat, "vals": vals})
 332         cat2 = Categorical([1, 2, 3, 3], categories=[1, 2, 3])
 333         vals2 = ["a", "b", "b", "d"]
 334         df_exp_fill = DataFrame({"cats": cat2, "vals": vals2})
 335         cat3 = Categorical([1, 2, 3], categories=[1, 2, 3])
 336         vals3 = ["a", "b", np.nan]
 337         df_exp_drop_cats = DataFrame({"cats": cat3, "vals": vals3})
 338         cat4 = Categorical([1, 2], categories=[1, 2, 3])
 339         vals4 = ["a", "b"]
 340         df_exp_drop_all = DataFrame({"cats": cat4, "vals": vals4})
 341 
 342         # fillna
 343         res = df.fillna(value={"cats": 3, "vals": "b"})
 344         tm.assert_frame_equal(res, df_exp_fill)
 345 
 346         with pytest.raises(ValueError, match=("fill value must be in categories")):
 347             df.fillna(value={"cats": 4, "vals": "c"})
 348 
 349         res = df.fillna(method="pad")
 350         tm.assert_frame_equal(res, df_exp_fill)
 351 
 352         # dropna
 353         res = df.dropna(subset=["cats"])
 354         tm.assert_frame_equal(res, df_exp_drop_cats)
 355 
 356         res = df.dropna()
 357         tm.assert_frame_equal(res, df_exp_drop_all)
 358 
 359         # make sure that fillna takes missing values into account
 360         c = Categorical([np.nan, "b", np.nan], categories=["a", "b"])
 361         df = pd.DataFrame({"cats": c, "vals": [1, 2, 3]})
 362 
 363         cat_exp = Categorical(["a", "b", "a"], categories=["a", "b"])
 364         df_exp = DataFrame({"cats": cat_exp, "vals": [1, 2, 3]})
 365 
 366         res = df.fillna("a")
 367         tm.assert_frame_equal(res, df_exp)
