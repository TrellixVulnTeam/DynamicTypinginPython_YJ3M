  39     def test_reindex(self, float_frame):
  40         datetime_series = tm.makeTimeSeries(nper=30)
  41 
  42         newFrame = float_frame.reindex(datetime_series.index)
  43 
  44         for col in newFrame.columns:
  45             for idx, val in newFrame[col].items():
  46                 if idx in float_frame.index:
  47                     if np.isnan(val):
  48                         assert np.isnan(float_frame[col][idx])
  49                     else:
  50                         assert val == float_frame[col][idx]
  51                 else:
  52                     assert np.isnan(val)
  53 
  54         for col, series in newFrame.items():
  55             assert tm.equalContents(series.index, newFrame.index)
  56         emptyFrame = float_frame.reindex(Index([]))
  57         assert len(emptyFrame.index) == 0
  58 
  59         # Cython code should be unit-tested directly
  60         nonContigFrame = float_frame.reindex(datetime_series.index[::2])
  61 
  62         for col in nonContigFrame.columns:
  63             for idx, val in nonContigFrame[col].items():
  64                 if idx in float_frame.index:
  65                     if np.isnan(val):
  66                         assert np.isnan(float_frame[col][idx])
  67                     else:
  68                         assert val == float_frame[col][idx]
  69                 else:
  70                     assert np.isnan(val)
  71 
  72         for col, series in nonContigFrame.items():
  73             assert tm.equalContents(series.index, nonContigFrame.index)
  74 
  75         # corner cases
  76 
  77         # Same index, copies values but not index if copy=False
  78         newFrame = float_frame.reindex(float_frame.index, copy=False)
  79         assert newFrame.index is float_frame.index
  80 
  81         # length zero
  82         newFrame = float_frame.reindex([])
  83         assert newFrame.empty
  84         assert len(newFrame.columns) == len(float_frame.columns)
  85 
  86         # length zero with columns reindexed with non-empty index
  87         newFrame = float_frame.reindex([])
  88         newFrame = newFrame.reindex(float_frame.index)
  89         assert len(newFrame.index) == len(float_frame.index)
  90         assert len(newFrame.columns) == len(float_frame.columns)
  91 
  92         # pass non-Index
  93         newFrame = float_frame.reindex(list(datetime_series.index))
  94         expected = datetime_series.index._with_freq(None)
  95         tm.assert_index_equal(newFrame.index, expected)
  96 
  97         # copy with no axes
  98         result = float_frame.reindex()
  99         tm.assert_frame_equal(result, float_frame)
 100         assert result is not float_frame

 354     def test_reindex_api_equivalence(self):
 355         # https://github.com/pandas-dev/pandas/issues/12392
 356         # equivalence of the labels/axis and index/columns API's
 357         df = DataFrame(
 358             [[1, 2, 3], [3, 4, 5], [5, 6, 7]],
 359             index=["a", "b", "c"],
 360             columns=["d", "e", "f"],
 361         )
 362 
 363         res1 = df.reindex(["b", "a"])
 364         res2 = df.reindex(index=["b", "a"])
 365         res3 = df.reindex(labels=["b", "a"])
 366         res4 = df.reindex(labels=["b", "a"], axis=0)
 367         res5 = df.reindex(["b", "a"], axis=0)
 368         for res in [res2, res3, res4, res5]:
 369             tm.assert_frame_equal(res1, res)
 370 
 371         res1 = df.reindex(columns=["e", "d"])
 372         res2 = df.reindex(["e", "d"], axis=1)
 373         res3 = df.reindex(labels=["e", "d"], axis=1)
 374         for res in [res2, res3]:
 375             tm.assert_frame_equal(res1, res)
 376 
 377         with tm.assert_produces_warning(FutureWarning) as m:
 378             res1 = df.reindex(["b", "a"], ["e", "d"])
 379         assert "reindex" in str(m[0].message)
 380         res2 = df.reindex(columns=["e", "d"], index=["b", "a"])
 381         res3 = df.reindex(labels=["b", "a"], axis=0).reindex(labels=["e", "d"], axis=1)
 382         for res in [res2, res3]:
 383             tm.assert_frame_equal(res1, res)
