  15 def next_monday(dt):
  16     """
  17     If holiday falls on Saturday, use following Monday instead;
  18     if holiday falls on Sunday, use Monday instead
  19     """
  20     if dt.weekday() == 5:
  21         return dt + timedelta(2)
  22     elif dt.weekday() == 6:
  23         return dt + timedelta(1)
  24     return dt

  27 def next_monday_or_tuesday(dt):
  28     """
  29     For second holiday of two adjacent ones!
  30     If holiday falls on Saturday, use following Monday instead;
  31     if holiday falls on Sunday or Monday, use following Tuesday instead
  32     (because Monday is already taken by adjacent holiday on the day before)
  33     """
  34     dow = dt.weekday()
  35     if dow == 5 or dow == 6:
  36         return dt + timedelta(2)
  37     elif dow == 0:
  38         return dt + timedelta(1)
  39     return dt

  43     """
  44     If holiday falls on Saturday or Sunday, use previous Friday instead.
  45     """
  46     if dt.weekday() == 5:
  47         return dt - timedelta(1)
  48     elif dt.weekday() == 6:
  49         return dt - timedelta(2)

  62 def weekend_to_monday(dt):
  63     """
  64     If holiday falls on Sunday or Saturday,
  65     use day thereafter (Monday) instead.
  66     Needed for holidays such as Christmas observation in Europe
  67     """
  68     if dt.weekday() == 6:
  69         return dt + timedelta(1)
  70     elif dt.weekday() == 5:
  71         return dt + timedelta(2)
  72     return dt

  75 def nearest_workday(dt):
  76     """
  77     If holiday falls on Saturday, use day before (Friday) instead;
  78     if holiday falls on Sunday, use day thereafter (Monday) instead.
  79     """
  80     if dt.weekday() == 5:
  81         return dt - timedelta(1)
  82     elif dt.weekday() == 6:
  83         return dt + timedelta(1)
  84     return dt

  87 def next_workday(dt):
  88     """
  89     returns next weekday used for observances
  90     """
  91     dt += timedelta(days=1)
  92     while dt.weekday() > 4:
  93         # Mon-Fri are 0-4
  94         dt += timedelta(days=1)
  95     return dt

  98 def previous_workday(dt):
  99     """
 100     returns previous weekday used for observances
 101     """
 102     dt -= timedelta(days=1)
 103     while dt.weekday() > 4:
 104         # Mon-Fri are 0-4
 105         dt -= timedelta(days=1)
 106     return dt

 130     def __init__(
 131         self,
 132         name,
 133         year=None,
 134         month=None,
 135         day=None,
 136         offset=None,
 137         observance=None,
 138         start_date=None,
 139         end_date=None,
 140         days_of_week=None,
 141     ):
 142         """
 143         Parameters
 144         ----------
 145         name : str
 146             Name of the holiday , defaults to class name
 147         offset : array of pandas.tseries.offsets or
 148                 class from pandas.tseries.offsets
 149             computes offset from date
 150         observance: function
 151             computes when holiday is given a pandas Timestamp
 152         days_of_week:
 153             provide a tuple of days e.g  (0,1,2,3,) for Monday Through Thursday
 154             Monday=0,..,Sunday=6
 155 
 156         Examples
 157         --------
 158         >>> from pandas.tseries.holiday import Holiday, nearest_workday
 159         >>> from dateutil.relativedelta import MO
 160 
 161         >>> USMemorialDay = Holiday(
 162         ...     "Memorial Day", month=5, day=31, offset=pd.DateOffset(weekday=MO(-1))
 163         ... )
 164         >>> USMemorialDay
 165         Holiday: Memorial Day (month=5, day=31, offset=<DateOffset: weekday=MO(-1)>)
 166 
 167         >>> USLaborDay = Holiday(
 168         ...     "Labor Day", month=9, day=1, offset=pd.DateOffset(weekday=MO(1))
 169         ... )
 170         >>> USLaborDay
 171         Holiday: Labor Day (month=9, day=1, offset=<DateOffset: weekday=MO(+1)>)
 172 
 173         >>> July3rd = Holiday("July 3rd", month=7, day=3)
 174         >>> July3rd
 175         Holiday: July 3rd (month=7, day=3, )
 176 
 177         >>> NewYears = Holiday(
 178         ...     "New Years Day", month=1,  day=1, observance=nearest_workday
 179         ... )
 180         >>> NewYears  # doctest: +SKIP
 181         Holiday: New Years Day (
 182             month=1, day=1, observance=<function nearest_workday at 0x66545e9bc440>
 183         )
 184 
 185         >>> July3rd = Holiday("July 3rd", month=7, day=3, days_of_week=(0, 1, 2, 3))
 186         >>> July3rd
 187         Holiday: July 3rd (month=7, day=3, )
 188         """
 189         if offset is not None and observance is not None:
 190             raise NotImplementedError("Cannot use both offset and observance.")
 191 
 192         self.name = name
 193         self.year = year
 194         self.month = month
 195         self.day = day
 196         self.offset = offset
 197         self.start_date = (
 198             Timestamp(start_date) if start_date is not None else start_date
 199         )
 200         self.end_date = Timestamp(end_date) if end_date is not None else end_date
 201         self.observance = observance
 202         assert days_of_week is None or type(days_of_week) == tuple
 203         self.days_of_week = days_of_week

 220     def dates(self, start_date, end_date, return_name=False):
 221         """
 222         Calculate holidays observed between start date and end date
 223 
 224         Parameters
 225         ----------
 226         start_date : starting date, datetime-like, optional
 227         end_date : ending date, datetime-like, optional
 228         return_name : bool, optional, default=False
 229             If True, return a series that has dates and holiday names.
 230             False will only return dates.
 231         """
 232         start_date = Timestamp(start_date)
 233         end_date = Timestamp(end_date)
 234 
 235         filter_start_date = start_date
 236         filter_end_date = end_date
 237 
 238         if self.year is not None:
 239             dt = Timestamp(datetime(self.year, self.month, self.day))
 240             if return_name:
 241                 return Series(self.name, index=[dt])
 242             else:
 243                 return [dt]
 244 
 245         dates = self._reference_dates(start_date, end_date)
 246         holiday_dates = self._apply_rule(dates)
 247         if self.days_of_week is not None:
 248             holiday_dates = holiday_dates[
 249                 np.in1d(holiday_dates.dayofweek, self.days_of_week)
 250             ]
 251 
 252         if self.start_date is not None:
 253             filter_start_date = max(
 254                 self.start_date.tz_localize(filter_start_date.tz), filter_start_date
 255             )
 256         if self.end_date is not None:
 257             filter_end_date = min(
 258                 self.end_date.tz_localize(filter_end_date.tz), filter_end_date
 259             )
 260         holiday_dates = holiday_dates[
 261             (holiday_dates >= filter_start_date) & (holiday_dates <= filter_end_date)
 262         ]
 263         if return_name:
 264             return Series(self.name, index=holiday_dates)
 265         return holiday_dates

 267     def _reference_dates(self, start_date, end_date):
 268         """
 269         Get reference dates for the holiday.
 270 
 271         Return reference dates for the holiday also returning the year
 272         prior to the start_date and year following the end_date.  This ensures
 273         that any offsets to be applied will yield the holidays within
 274         the passed in dates.
 275         """
 276         if self.start_date is not None:
 277             start_date = self.start_date.tz_localize(start_date.tz)
 278 
 279         if self.end_date is not None:
 280             end_date = self.end_date.tz_localize(start_date.tz)
 281 
 282         year_offset = DateOffset(years=1)
 283         reference_start_date = Timestamp(
 284             datetime(start_date.year - 1, self.month, self.day)
 285         )
 286 
 287         reference_end_date = Timestamp(
 288             datetime(end_date.year + 1, self.month, self.day)
 289         )
 290         # Don't process unnecessary holidays
 291         dates = date_range(
 292             start=reference_start_date,
 293             end=reference_end_date,
 294             freq=year_offset,
 295             tz=start_date.tz,
 296         )
 297 
 298         return dates

 300     def _apply_rule(self, dates):
 301         """
 302         Apply the given offset/observance to a DatetimeIndex of dates.
 303 
 304         Parameters
 305         ----------
 306         dates : DatetimeIndex
 307             Dates to apply the given offset/observance rule
 308 
 309         Returns
 310         -------
 311         Dates with rules applied
 312         """
 313         if self.observance is not None:
 314             return dates.map(lambda d: self.observance(d))
 315 
 316         if self.offset is not None:
 317             if not isinstance(self.offset, list):
 318                 offsets = [self.offset]
 319             else:
 320                 offsets = self.offset
 321             for offset in offsets:
 322 
 323                 # if we are adding a non-vectorized value
 324                 # ignore the PerformanceWarnings:
 325                 with warnings.catch_warnings():
 326                     warnings.simplefilter("ignore", PerformanceWarning)
 327                     dates += offset
 328         return dates

 361 class AbstractHolidayCalendar(metaclass=HolidayCalendarMetaClass):
 362     """
 363     Abstract interface to create holidays following certain rules.
 364     """
 365 
 366     rules: List[Holiday] = []
 367     start_date = Timestamp(datetime(1970, 1, 1))
 368     end_date = Timestamp(datetime(2200, 12, 31))
 369     _cache = None
 370 
 371     def __init__(self, name=None, rules=None):
 372         """
 373         Initializes holiday object with a given set a rules.  Normally
 374         classes just have the rules defined within them.
 375 
 376         Parameters
 377         ----------
 378         name : str
 379             Name of the holiday calendar, defaults to class name
 380         rules : array of Holiday objects
 381             A set of rules used to create the holidays.
 382         """
 383         super().__init__()
 384         if name is None:
 385             name = type(self).__name__
 386         self.name = name
 387 
 388         if rules is not None:
 389             self.rules = rules
 390 
 391     def rule_from_name(self, name):
 392         for rule in self.rules:
 393             if rule.name == name:
 394                 return rule
 395 
 396         return None

 398     def holidays(self, start=None, end=None, return_name=False):
 399         """
 400         Returns a curve with holidays between start_date and end_date
 401 
 402         Parameters
 403         ----------
 404         start : starting date, datetime-like, optional
 405         end : ending date, datetime-like, optional
 406         return_name : bool, optional
 407             If True, return a series that has dates and holiday names.
 408             False will only return a DatetimeIndex of dates.
 409 
 410         Returns
 411         -------
 412             DatetimeIndex of holidays
 413         """
 414         if self.rules is None:
 415             raise Exception(
 416                 f"Holiday Calendar {self.name} does not have any rules specified"
 417             )
 418 
 419         if start is None:
 420             start = AbstractHolidayCalendar.start_date
 421 
 422         if end is None:
 423             end = AbstractHolidayCalendar.end_date
 424 
 425         start = Timestamp(start)
 426         end = Timestamp(end)
 427 
 428         # If we don't have a cache or the dates are outside the prior cache, we
 429         # get them again
 430         if self._cache is None or start < self._cache[0] or end > self._cache[1]:
 431             holidays = [rule.dates(start, end, return_name=True) for rule in self.rules]
 432             if holidays:
 433                 holidays = concat(holidays)
 434             else:
 435                 holidays = Series(index=DatetimeIndex([]), dtype=object)
 436 
 437             self._cache = (start, end, holidays.sort_index())
 438 
 439         holidays = self._cache[2]
 440         holidays = holidays[start:end]
 441 
 442         if return_name:
 443             return holidays
 444         else:
 445             return holidays.index

 447     @staticmethod
 448     def merge_class(base, other):
 449         """
 450         Merge holiday calendars together. The base calendar
 451         will take precedence to other. The merge will be done
 452         based on each holiday's name.
 453 
 454         Parameters
 455         ----------
 456         base : AbstractHolidayCalendar
 457           instance/subclass or array of Holiday objects
 458         other : AbstractHolidayCalendar
 459           instance/subclass or array of Holiday objects
 460         """
 461         try:
 462             other = other.rules
 463         except AttributeError:
 464             pass
 465 
 466         if not isinstance(other, list):
 467             other = [other]
 468         other_holidays = {holiday.name: holiday for holiday in other}
 469 
 470         try:
 471             base = base.rules
 472         except AttributeError:
 473             pass
 474 
 475         if not isinstance(base, list):
 476             base = [base]
 477         base_holidays = {holiday.name: holiday for holiday in base}
 478 
 479         other_holidays.update(base_holidays)
 480         return list(other_holidays.values())
 481 
 482     def merge(self, other, inplace=False):
 483         """
 484         Merge holiday calendars together.  The caller's class
 485         rules take precedence.  The merge will be done
 486         based on each holiday's name.
 487 
 488         Parameters
 489         ----------
 490         other : holiday calendar
 491         inplace : bool (default=False)
 492             If True set rule_table to holidays, else return array of Holidays
 493         """
 494         holidays = self.merge_class(self, other)
 495         if inplace:
 496             self.rules = holidays
 497         else:
 498             return holidays

 547 def HolidayCalendarFactory(name, base, other, base_class=AbstractHolidayCalendar):
 548     rules = AbstractHolidayCalendar.merge_class(base, other)
 549     calendar_class = type(name, (base_class,), {"rules": rules, "name": name})
 550     return calendar_class
