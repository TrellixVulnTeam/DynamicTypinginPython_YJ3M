 123 def infer_freq(index, warn: bool = True) -> Optional[str]:
 124     """
 125     Infer the most likely frequency given the input index. If the frequency is
 126     uncertain, a warning will be printed.
 127 
 128     Parameters
 129     ----------
 130     index : DatetimeIndex or TimedeltaIndex
 131       If passed a Series will use the values of the series (NOT THE INDEX).
 132     warn : bool, default True
 133 
 134     Returns
 135     -------
 136     str or None
 137         None if no discernible frequency.
 138 
 139     Raises
 140     ------
 141     TypeError
 142         If the index is not datetime-like.
 143     ValueError
 144         If there are fewer than three values.
 145     """
 146     import pandas as pd
 147 
 148     if isinstance(index, ABCSeries):
 149         values = index._values
 150         if not (
 151             is_datetime64_dtype(values)
 152             or is_timedelta64_dtype(values)
 153             or values.dtype == object
 154         ):
 155             raise TypeError(
 156                 "cannot infer freq from a non-convertible dtype "
 157                 f"on a Series of {index.dtype}"
 158             )
 159         index = values
 160 
 161     inferer: _FrequencyInferer
 162 
 163     if not hasattr(index, "dtype"):
 164         pass
 165     elif is_period_dtype(index.dtype):
 166         raise TypeError(
 167             "PeriodIndex given. Check the `freq` attribute "
 168             "instead of using infer_freq."
 169         )
 170     elif is_timedelta64_dtype(index.dtype):
 171         # Allow TimedeltaIndex and TimedeltaArray
 172         inferer = _TimedeltaFrequencyInferer(index, warn=warn)
 173         return inferer.get_freq()
 174 
 175     if isinstance(index, pd.Index) and not isinstance(index, pd.DatetimeIndex):
 176         if isinstance(index, (pd.Int64Index, pd.Float64Index)):
 177             raise TypeError(
 178                 f"cannot infer freq from a non-convertible index type {type(index)}"
 179             )
 180         index = index._values
 181 
 182     if not isinstance(index, pd.DatetimeIndex):
 183         index = pd.DatetimeIndex(index)
 184 
 185     inferer = _FrequencyInferer(index, warn=warn)
 186     return inferer.get_freq()
 187

 189 class _FrequencyInferer:
 190     """
 191     Not sure if I can avoid the state machine here
 192     """
 193 
 194     def __init__(self, index, warn: bool = True):
 195         self.index = index
 196         self.i8values = index.asi8
 197 
 198         # This moves the values, which are implicitly in UTC, to the
 199         # the timezone so they are in local time
 200         if hasattr(index, "tz"):
 201             if index.tz is not None:
 202                 self.i8values = tz_convert(self.i8values, UTC, index.tz)
 203 
 204         self.warn = warn
 205 
 206         if len(index) < 3:
 207             raise ValueError("Need at least 3 dates to infer frequency")
 208 
 209         self.is_monotonic = (
 210             self.index._is_monotonic_increasing or self.index._is_monotonic_decreasing
 211         )
 212 
 213     @cache_readonly
 214     def deltas(self):
 215         return unique_deltas(self.i8values)
 216 
 217     @cache_readonly
 218     def deltas_asi8(self):
 219         # NB: we cannot use self.i8values here because we may have converted
 220         #  the tz in __init__
 221         return unique_deltas(self.index.asi8)
 222 
 223     @cache_readonly
 224     def is_unique(self) -> bool:
 225         return len(self.deltas) == 1
 226 
 227     @cache_readonly
 228     def is_unique_asi8(self) -> bool:
 229         return len(self.deltas_asi8) == 1
 230 
 231     def get_freq(self) -> Optional[str]:
 232         """
 233         Find the appropriate frequency string to describe the inferred
 234         frequency of self.i8values
 235 
 236         Returns
 237         -------
 238         str or None
 239         """
 240         if not self.is_monotonic or not self.index._is_unique:
 241             return None
 242 
 243         delta = self.deltas[0]
 244         if _is_multiple(delta, _ONE_DAY):
 245             return self._infer_daily_rule()
 246 
 247         # Business hourly, maybe. 17: one day / 65: one weekend
 248         if self.hour_deltas in ([1, 17], [1, 65], [1, 17, 65]):
 249             return "BH"
 250         # Possibly intraday frequency.  Here we use the
 251         # original .asi8 values as the modified values
 252         # will not work around DST transitions.  See #8772
 253         elif not self.is_unique_asi8:
 254             return None
 255 
 256         delta = self.deltas_asi8[0]
 257         if _is_multiple(delta, _ONE_HOUR):
 258             # Hours
 259             return _maybe_add_count("H", delta / _ONE_HOUR)
 260         elif _is_multiple(delta, _ONE_MINUTE):
 261             # Minutes
 262             return _maybe_add_count("T", delta / _ONE_MINUTE)
 263         elif _is_multiple(delta, _ONE_SECOND):
 264             # Seconds
 265             return _maybe_add_count("S", delta / _ONE_SECOND)
 266         elif _is_multiple(delta, _ONE_MILLI):
 267             # Milliseconds
 268             return _maybe_add_count("L", delta / _ONE_MILLI)
 269         elif _is_multiple(delta, _ONE_MICRO):
 270             # Microseconds
 271             return _maybe_add_count("U", delta / _ONE_MICRO)
 272         else:
 273             # Nanoseconds
 274             return _maybe_add_count("N", delta)
 275 
 276     @cache_readonly
 277     def day_deltas(self):
 278         return [x / _ONE_DAY for x in self.deltas]
 279 
 280     @cache_readonly
 281     def hour_deltas(self):
 282         return [x / _ONE_HOUR for x in self.deltas]
 283 
 284     @cache_readonly
 285     def fields(self):
 286         return build_field_sarray(self.i8values)
 287 
 288     @cache_readonly
 289     def rep_stamp(self):
 290         return Timestamp(self.i8values[0])
 291 
 292     def month_position_check(self):
 293         return month_position_check(self.fields, self.index.dayofweek)
 294 
 295     @cache_readonly
 296     def mdiffs(self):
 297         nmonths = self.fields["Y"] * 12 + self.fields["M"]
 298         return unique_deltas(nmonths.astype("i8"))
 299 
 300     @cache_readonly
 301     def ydiffs(self):
 302         return unique_deltas(self.fields["Y"].astype("i8"))
 303 
 304     def _infer_daily_rule(self) -> Optional[str]:
 305         annual_rule = self._get_annual_rule()
 306         if annual_rule:
 307             nyears = self.ydiffs[0]
 308             month = MONTH_ALIASES[self.rep_stamp.month]
 309             alias = f"{annual_rule}-{month}"
 310             return _maybe_add_count(alias, nyears)
 311 
 312         quarterly_rule = self._get_quarterly_rule()
 313         if quarterly_rule:
 314             nquarters = self.mdiffs[0] / 3
 315             mod_dict = {0: 12, 2: 11, 1: 10}
 316             month = MONTH_ALIASES[mod_dict[self.rep_stamp.month % 3]]
 317             alias = f"{quarterly_rule}-{month}"
 318             return _maybe_add_count(alias, nquarters)
 319 
 320         monthly_rule = self._get_monthly_rule()
 321         if monthly_rule:
 322             return _maybe_add_count(monthly_rule, self.mdiffs[0])
 323 
 324         if self.is_unique:
 325             days = self.deltas[0] / _ONE_DAY
 326             if days % 7 == 0:
 327                 # Weekly
 328                 day = int_to_weekday[self.rep_stamp.weekday()]
 329                 return _maybe_add_count(f"W-{day}", days / 7)
 330             else:
 331                 return _maybe_add_count("D", days)
 332 
 333         if self._is_business_daily():
 334             return "B"
 335 
 336         wom_rule = self._get_wom_rule()
 337         if wom_rule:
 338             return wom_rule
 339 
 340         return None
 341 
 342     def _get_annual_rule(self) -> Optional[str]:
 343         if len(self.ydiffs) > 1:
 344             return None
 345 
 346         if len(unique(self.fields["M"])) > 1:
 347             return None
 348 
 349         pos_check = self.month_position_check()
 350         return {"cs": "AS", "bs": "BAS", "ce": "A", "be": "BA"}.get(pos_check)
 351 
 352     def _get_quarterly_rule(self) -> Optional[str]:
 353         if len(self.mdiffs) > 1:
 354             return None
 355 
 356         if not self.mdiffs[0] % 3 == 0:
 357             return None
 358 
 359         pos_check = self.month_position_check()
 360         return {"cs": "QS", "bs": "BQS", "ce": "Q", "be": "BQ"}.get(pos_check)
 361 
 362     def _get_monthly_rule(self) -> Optional[str]:
 363         if len(self.mdiffs) > 1:
 364             return None
 365         pos_check = self.month_position_check()
 366         return {"cs": "MS", "bs": "BMS", "ce": "M", "be": "BM"}.get(pos_check)
 367 
 368     def _is_business_daily(self) -> bool:
 369         # quick check: cannot be business daily
 370         if self.day_deltas != [1, 3]:
 371             return False
 372 
 373         # probably business daily, but need to confirm
 374         first_weekday = self.index[0].weekday()
 375         shifts = np.diff(self.index.asi8)
 376         shifts = np.floor_divide(shifts, _ONE_DAY)
 377         weekdays = np.mod(first_weekday + np.cumsum(shifts), 7)
 378         return np.all(
 379             ((weekdays == 0) & (shifts == 3))
 380             | ((weekdays > 0) & (weekdays <= 4) & (shifts == 1))
 381         )
 382 
 383     def _get_wom_rule(self) -> Optional[str]:
 384         # FIXME: dont leave commented-out
 385         #         wdiffs = unique(np.diff(self.index.week))
 386         # We also need -47, -49, -48 to catch index spanning year boundary
 387         #     if not lib.ismember(wdiffs, set([4, 5, -47, -49, -48])).all():
 388         #         return None
 389 
 390         weekdays = unique(self.index.weekday)
 391         if len(weekdays) > 1:
 392             return None
 393 
 394         week_of_months = unique((self.index.day - 1) // 7)
 395         # Only attempt to infer up to WOM-4. See #9425
 396         week_of_months = week_of_months[week_of_months < 4]
 397         if len(week_of_months) == 0 or len(week_of_months) > 1:
 398             return None
 399 
 400         # get which week
 401         week = week_of_months[0] + 1
 402         wd = int_to_weekday[weekdays[0]]
 403 
 404         return f"WOM-{week}{wd}"

 407 class _TimedeltaFrequencyInferer(_FrequencyInferer):
 408     def _infer_daily_rule(self):
 409         if self.is_unique:
 410             days = self.deltas[0] / _ONE_DAY
 411             if days % 7 == 0:
 412                 # Weekly
 413                 wd = int_to_weekday[self.rep_stamp.weekday()]
 414                 alias = f"W-{wd}"
 415                 return _maybe_add_count(alias, days / 7)
 416             else:
 417                 return _maybe_add_count("D", days)

 420 def _is_multiple(us, mult: int) -> bool:
 421     return us % mult == 0

 424 def _maybe_add_count(base: str, count: float) -> str:
 425     if count != 1:
 426         assert count == int(count)
 427         count = int(count)
 428         return f"{count}{base}"
 429     else:
 430         return base

 437 def is_subperiod(source, target) -> bool:
 438     """
 439     Returns True if downsampling is possible between source and target
 440     frequencies
 441 
 442     Parameters
 443     ----------
 444     source : str or DateOffset
 445         Frequency converting from
 446     target : str or DateOffset
 447         Frequency converting to
 448 
 449     Returns
 450     -------
 451     bool
 452     """
 453 
 454     if target is None or source is None:
 455         return False
 456     source = _maybe_coerce_freq(source)
 457     target = _maybe_coerce_freq(target)
 458 
 459     if _is_annual(target):
 460         if _is_quarterly(source):
 461             return _quarter_months_conform(
 462                 get_rule_month(source), get_rule_month(target)
 463             )
 464         return source in {"D", "C", "B", "M", "H", "T", "S", "L", "U", "N"}
 465     elif _is_quarterly(target):
 466         return source in {"D", "C", "B", "M", "H", "T", "S", "L", "U", "N"}
 467     elif _is_monthly(target):
 468         return source in {"D", "C", "B", "H", "T", "S", "L", "U", "N"}
 469     elif _is_weekly(target):
 470         return source in {target, "D", "C", "B", "H", "T", "S", "L", "U", "N"}
 471     elif target == "B":
 472         return source in {"B", "H", "T", "S", "L", "U", "N"}
 473     elif target == "C":
 474         return source in {"C", "H", "T", "S", "L", "U", "N"}
 475     elif target == "D":
 476         return source in {"D", "H", "T", "S", "L", "U", "N"}
 477     elif target == "H":
 478         return source in {"H", "T", "S", "L", "U", "N"}
 479     elif target == "T":
 480         return source in {"T", "S", "L", "U", "N"}
 481     elif target == "S":
 482         return source in {"S", "L", "U", "N"}
 483     elif target == "L":
 484         return source in {"L", "U", "N"}
 485     elif target == "U":
 486         return source in {"U", "N"}
 487     elif target == "N":
 488         return source in {"N"}
 489     else:
 490         return False

 493 def is_superperiod(source, target) -> bool:
 494     """
 495     Returns True if upsampling is possible between source and target
 496     frequencies
 497 
 498     Parameters
 499     ----------
 500     source : str or DateOffset
 501         Frequency converting from
 502     target : str or DateOffset
 503         Frequency converting to
 504 
 505     Returns
 506     -------
 507     bool
 508     """
 509     if target is None or source is None:
 510         return False
 511     source = _maybe_coerce_freq(source)
 512     target = _maybe_coerce_freq(target)
 513 
 514     if _is_annual(source):
 515         if _is_annual(target):
 516             return get_rule_month(source) == get_rule_month(target)
 517 
 518         if _is_quarterly(target):
 519             smonth = get_rule_month(source)
 520             tmonth = get_rule_month(target)
 521             return _quarter_months_conform(smonth, tmonth)
 522         return target in {"D", "C", "B", "M", "H", "T", "S", "L", "U", "N"}
 523     elif _is_quarterly(source):
 524         return target in {"D", "C", "B", "M", "H", "T", "S", "L", "U", "N"}
 525     elif _is_monthly(source):
 526         return target in {"D", "C", "B", "H", "T", "S", "L", "U", "N"}
 527     elif _is_weekly(source):
 528         return target in {source, "D", "C", "B", "H", "T", "S", "L", "U", "N"}
 529     elif source == "B":
 530         return target in {"D", "C", "B", "H", "T", "S", "L", "U", "N"}
 531     elif source == "C":
 532         return target in {"D", "C", "B", "H", "T", "S", "L", "U", "N"}
 533     elif source == "D":
 534         return target in {"D", "C", "B", "H", "T", "S", "L", "U", "N"}
 535     elif source == "H":
 536         return target in {"H", "T", "S", "L", "U", "N"}
 537     elif source == "T":
 538         return target in {"T", "S", "L", "U", "N"}
 539     elif source == "S":
 540         return target in {"S", "L", "U", "N"}
 541     elif source == "L":
 542         return target in {"L", "U", "N"}
 543     elif source == "U":
 544         return target in {"U", "N"}
 545     elif source == "N":
 546         return target in {"N"}
 547     else:
 548         return False
