  62 class Styler:
  63     """
  64     Helps style a DataFrame or Series according to the data with HTML and CSS.
  65 
  66     Parameters
  67     ----------
  68     data : Series or DataFrame
  69         Data to be styled - either a Series or DataFrame.
  70     precision : int
  71         Precision to round floats to, defaults to pd.options.display.precision.
  72     table_styles : list-like, default None
  73         List of {selector: (attr, value)} dicts; see Notes.
  74     uuid : str, default None
  75         A unique identifier to avoid CSS collisions; generated automatically.
  76     caption : str, default None
  77         Caption to attach to the table.
  78     table_attributes : str, default None
  79         Items that show up in the opening ``<table>`` tag
  80         in addition to automatic (by default) id.
  81     cell_ids : bool, default True
  82         If True, each cell will have an ``id`` attribute in their HTML tag.
  83         The ``id`` takes the form ``T_<uuid>_row<num_row>_col<num_col>``
  84         where ``<uuid>`` is the unique identifier, ``<num_row>`` is the row
  85         number and ``<num_col>`` is the column number.
  86     na_rep : str, optional
  87         Representation for missing values.
  88         If ``na_rep`` is None, no special formatting is applied.
  89 
  90         .. versionadded:: 1.0.0
  91 
  92     Attributes
  93     ----------
  94     env : Jinja2 jinja2.Environment
  95     template : Jinja2 Template
  96     loader : Jinja2 Loader
  97 
  98     See Also
  99     --------
 100     DataFrame.style : Return a Styler object containing methods for building
 101         a styled HTML representation for the DataFrame.
 102 
 103     Notes
 104     -----
 105     Most styling will be done by passing style functions into
 106     ``Styler.apply`` or ``Styler.applymap``. Style functions should
 107     return values with strings containing CSS ``'attr: value'`` that will
 108     be applied to the indicated cells.
 109 
 110     If using in the Jupyter notebook, Styler has defined a ``_repr_html_``
 111     to automatically render itself. Otherwise call Styler.render to get
 112     the generated HTML.
 113 
 114     CSS classes are attached to the generated HTML
 115 
 116     * Index and Column names include ``index_name`` and ``level<k>``
 117       where `k` is its level in a MultiIndex
 118     * Index label cells include
 119 
 120       * ``row_heading``
 121       * ``row<n>`` where `n` is the numeric position of the row
 122       * ``level<k>`` where `k` is the level in a MultiIndex
 123 
 124     * Column label cells include
 125       * ``col_heading``
 126       * ``col<n>`` where `n` is the numeric position of the column
 127       * ``level<k>`` where `k` is the level in a MultiIndex
 128 
 129     * Blank cells include ``blank``
 130     * Data cells include ``data``
 131     """
 132 
 133     loader = jinja2.PackageLoader("pandas", "io/formats/templates")
 134     env = jinja2.Environment(loader=loader, trim_blocks=True)
 135     template = env.get_template("html.tpl")

 137     def __init__(
 138         self,
 139         data: FrameOrSeriesUnion,
 140         precision: Optional[int] = None,
 141         table_styles: Optional[List[Dict[str, List[Tuple[str, str]]]]] = None,
 142         uuid: Optional[str] = None,
 143         caption: Optional[str] = None,
 144         table_attributes: Optional[str] = None,
 145         cell_ids: bool = True,
 146         na_rep: Optional[str] = None,
 147     ):
 148         self.ctx: DefaultDict[Tuple[int, int], List[str]] = defaultdict(list)
 149         self._todo: List[Tuple[Callable, Tuple, Dict]] = []
 150 
 151         if not isinstance(data, (pd.Series, pd.DataFrame)):
 152             raise TypeError("``data`` must be a Series or DataFrame")
 153         if data.ndim == 1:
 154             data = data.to_frame()
 155         if not data.index.is_unique or not data.columns.is_unique:
 156             raise ValueError("style is not supported for non-unique indices.")
 157 
 158         self.data = data
 159         self.index = data.index
 160         self.columns = data.columns
 161 
 162         self.uuid = uuid
 163         self.table_styles = table_styles
 164         self.caption = caption
 165         if precision is None:
 166             precision = get_option("display.precision")
 167         self.precision = precision
 168         self.table_attributes = table_attributes
 169         self.hidden_index = False
 170         self.hidden_columns: Sequence[int] = []
 171         self.cell_ids = cell_ids
 172         self.na_rep = na_rep
 173 
 174         # display_funcs maps (row, col) -> formatting function
 175 
 176         def default_display_func(x):
 177             if self.na_rep is not None and pd.isna(x):
 178                 return self.na_rep
 179             elif is_float(x):
 180                 display_format = f"{x:.{self.precision}f}"
 181                 return display_format
 182             else:
 183                 return x
 184 
 185         self._display_funcs: DefaultDict[
 186             Tuple[int, int], Callable[[Any], str]
 187         ] = defaultdict(lambda: default_display_func)

 238     def _translate(self):
 239         """
 240         Convert the DataFrame in `self.data` and the attrs from `_build_styles`
 241         into a dictionary of {head, body, uuid, cellstyle}.
 242         """
 243         table_styles = self.table_styles or []
 244         caption = self.caption
 245         ctx = self.ctx
 246         precision = self.precision
 247         hidden_index = self.hidden_index
 248         hidden_columns = self.hidden_columns
 249         uuid = self.uuid or str(uuid1()).replace("-", "_")
 250         ROW_HEADING_CLASS = "row_heading"
 251         COL_HEADING_CLASS = "col_heading"
 252         INDEX_NAME_CLASS = "index_name"
 253 
 254         DATA_CLASS = "data"
 255         BLANK_CLASS = "blank"
 256         BLANK_VALUE = ""
 257 
 258         def format_attr(pair):
 259             return f"{pair['key']}={pair['value']}"
 260 
 261         # for sparsifying a MultiIndex
 262         idx_lengths = _get_level_lengths(self.index)
 263         col_lengths = _get_level_lengths(self.columns, hidden_columns)
 264 
 265         cell_context = dict()
 266 
 267         n_rlvls = self.data.index.nlevels
 268         n_clvls = self.data.columns.nlevels
 269         rlabels = self.data.index.tolist()
 270         clabels = self.data.columns.tolist()
 271 
 272         if n_rlvls == 1:
 273             rlabels = [[x] for x in rlabels]
 274         if n_clvls == 1:
 275             clabels = [[x] for x in clabels]
 276         clabels = list(zip(*clabels))
 277 
 278         cellstyle_map = defaultdict(list)
 279         head = []
 280 
 281         for r in range(n_clvls):
 282             # Blank for Index columns...
 283             row_es = [
 284                 {
 285                     "type": "th",
 286                     "value": BLANK_VALUE,
 287                     "display_value": BLANK_VALUE,
 288                     "is_visible": not hidden_index,
 289                     "class": " ".join([BLANK_CLASS]),
 290                 }
 291             ] * (n_rlvls - 1)
 292 
 293             # ... except maybe the last for columns.names
 294             name = self.data.columns.names[r]
 295             cs = [
 296                 BLANK_CLASS if name is None else INDEX_NAME_CLASS,
 297                 f"level{r}",
 298             ]
 299             name = BLANK_VALUE if name is None else name
 300             row_es.append(
 301                 {
 302                     "type": "th",
 303                     "value": name,
 304                     "display_value": name,
 305                     "class": " ".join(cs),
 306                     "is_visible": not hidden_index,
 307                 }
 308             )
 309 
 310             if clabels:
 311                 for c, value in enumerate(clabels[r]):
 312                     cs = [
 313                         COL_HEADING_CLASS,
 314                         f"level{r}",
 315                         f"col{c}",
 316                     ]
 317                     cs.extend(
 318                         cell_context.get("col_headings", {}).get(r, {}).get(c, [])
 319                     )
 320                     es = {
 321                         "type": "th",
 322                         "value": value,
 323                         "display_value": value,
 324                         "class": " ".join(cs),
 325                         "is_visible": _is_visible(c, r, col_lengths),
 326                     }
 327                     colspan = col_lengths.get((r, c), 0)
 328                     if colspan > 1:
 329                         es["attributes"] = [
 330                             format_attr({"key": "colspan", "value": colspan})
 331                         ]
 332                     row_es.append(es)
 333                 head.append(row_es)
 334 
 335         if (
 336             self.data.index.names
 337             and com.any_not_none(*self.data.index.names)
 338             and not hidden_index
 339         ):
 340             index_header_row = []
 341 
 342             for c, name in enumerate(self.data.index.names):
 343                 cs = [INDEX_NAME_CLASS, f"level{c}"]
 344                 name = "" if name is None else name
 345                 index_header_row.append(
 346                     {"type": "th", "value": name, "class": " ".join(cs)}
 347                 )
 348 
 349             index_header_row.extend(
 350                 [{"type": "th", "value": BLANK_VALUE, "class": " ".join([BLANK_CLASS])}]
 351                 * (len(clabels[0]) - len(hidden_columns))
 352             )
 353 
 354             head.append(index_header_row)
 355 
 356         body = []
 357         for r, idx in enumerate(self.data.index):
 358             row_es = []
 359             for c, value in enumerate(rlabels[r]):
 360                 rid = [
 361                     ROW_HEADING_CLASS,
 362                     f"level{c}",
 363                     f"row{r}",
 364                 ]
 365                 es = {
 366                     "type": "th",
 367                     "is_visible": (_is_visible(r, c, idx_lengths) and not hidden_index),
 368                     "value": value,
 369                     "display_value": value,
 370                     "id": "_".join(rid[1:]),
 371                     "class": " ".join(rid),
 372                 }
 373                 rowspan = idx_lengths.get((c, r), 0)
 374                 if rowspan > 1:
 375                     es["attributes"] = [
 376                         format_attr({"key": "rowspan", "value": rowspan})
 377                     ]
 378                 row_es.append(es)
 379 
 380             for c, col in enumerate(self.data.columns):
 381                 cs = [DATA_CLASS, f"row{r}", f"col{c}"]
 382                 cs.extend(cell_context.get("data", {}).get(r, {}).get(c, []))
 383                 formatter = self._display_funcs[(r, c)]
 384                 value = self.data.iloc[r, c]
 385                 row_dict = {
 386                     "type": "td",
 387                     "value": value,
 388                     "class": " ".join(cs),
 389                     "display_value": formatter(value),
 390                     "is_visible": (c not in hidden_columns),
 391                 }
 392                 # only add an id if the cell has a style
 393                 if self.cell_ids or not (len(ctx[r, c]) == 1 and ctx[r, c][0] == ""):
 394                     row_dict["id"] = "_".join(cs[1:])
 395                 row_es.append(row_dict)
 396                 props = []
 397                 for x in ctx[r, c]:
 398                     # have to handle empty styles like ['']
 399                     if x.count(":"):
 400                         props.append(tuple(x.split(":")))
 401                     else:
 402                         props.append(("", ""))
 403                 cellstyle_map[tuple(props)].append(f"row{r}_col{c}")
 404             body.append(row_es)
 405 
 406         cellstyle = [
 407             {"props": list(props), "selectors": selectors}
 408             for props, selectors in cellstyle_map.items()
 409         ]
 410 
 411         table_attr = self.table_attributes
 412         use_mathjax = get_option("display.html.use_mathjax")
 413         if not use_mathjax:
 414             table_attr = table_attr or ""
 415             if 'class="' in table_attr:
 416                 table_attr = table_attr.replace('class="', 'class="tex2jax_ignore ')
 417             else:
 418                 table_attr += ' class="tex2jax_ignore"'
 419 
 420         return dict(
 421             head=head,
 422             cellstyle=cellstyle,
 423             body=body,
 424             uuid=uuid,
 425             precision=precision,
 426             table_styles=table_styles,
 427             caption=caption,
 428             table_attributes=table_attr,
 429         )

 431     def format(self, formatter, subset=None, na_rep: Optional[str] = None) -> "Styler":
 432         """
 433         Format the text display value of cells.
 434 
 435         Parameters
 436         ----------
 437         formatter : str, callable, dict or None
 438             If ``formatter`` is None, the default formatter is used.
 439         subset : IndexSlice
 440             An argument to ``DataFrame.loc`` that restricts which elements
 441             ``formatter`` is applied to.
 442         na_rep : str, optional
 443             Representation for missing values.
 444             If ``na_rep`` is None, no special formatting is applied.
 445 
 446             .. versionadded:: 1.0.0
 447 
 448         Returns
 449         -------
 450         self : Styler
 451 
 452         Notes
 453         -----
 454         ``formatter`` is either an ``a`` or a dict ``{column name: a}`` where
 455         ``a`` is one of
 456 
 457         - str: this will be wrapped in: ``a.format(x)``
 458         - callable: called with the value of an individual cell
 459 
 460         The default display value for numeric values is the "general" (``g``)
 461         format with ``pd.options.display.precision`` precision.
 462 
 463         Examples
 464         --------
 465         >>> df = pd.DataFrame(np.random.randn(4, 2), columns=['a', 'b'])
 466         >>> df.style.format("{:.2%}")
 467         >>> df['c'] = ['a', 'b', 'c', 'd']
 468         >>> df.style.format({'c': str.upper})
 469         """
 470         if formatter is None:
 471             assert self._display_funcs.default_factory is not None
 472             formatter = self._display_funcs.default_factory()
 473 
 474         if subset is None:
 475             row_locs = range(len(self.data))
 476             col_locs = range(len(self.data.columns))
 477         else:
 478             subset = _non_reducing_slice(subset)
 479             if len(subset) == 1:
 480                 subset = subset, self.data.columns
 481 
 482             sub_df = self.data.loc[subset]
 483             row_locs = self.data.index.get_indexer_for(sub_df.index)
 484             col_locs = self.data.columns.get_indexer_for(sub_df.columns)
 485 
 486         if is_dict_like(formatter):
 487             for col, col_formatter in formatter.items():
 488                 # formatter must be callable, so '{}' are converted to lambdas
 489                 col_formatter = _maybe_wrap_formatter(col_formatter, na_rep)
 490                 col_num = self.data.columns.get_indexer_for([col])[0]
 491 
 492                 for row_num in row_locs:
 493                     self._display_funcs[(row_num, col_num)] = col_formatter
 494         else:
 495             # single scalar to format all cells with
 496             formatter = _maybe_wrap_formatter(formatter, na_rep)
 497             locs = product(*(row_locs, col_locs))
 498             for i, j in locs:
 499                 self._display_funcs[(i, j)] = formatter
 500         return self

 502     def render(self, **kwargs) -> str:
 503         """
 504         Render the built up styles to HTML.
 505 
 506         Parameters
 507         ----------
 508         **kwargs
 509             Any additional keyword arguments are passed
 510             through to ``self.template.render``.
 511             This is useful when you need to provide
 512             additional variables for a custom template.
 513 
 514         Returns
 515         -------
 516         rendered : str
 517             The rendered HTML.
 518 
 519         Notes
 520         -----
 521         ``Styler`` objects have defined the ``_repr_html_`` method
 522         which automatically calls ``self.render()`` when it's the
 523         last item in a Notebook cell. When calling ``Styler.render()``
 524         directly, wrap the result in ``IPython.display.HTML`` to view
 525         the rendered HTML in the notebook.
 526 
 527         Pandas uses the following keys in render. Arguments passed
 528         in ``**kwargs`` take precedence, so think carefully if you want
 529         to override them:
 530 
 531         * head
 532         * cellstyle
 533         * body
 534         * uuid
 535         * precision
 536         * table_styles
 537         * caption
 538         * table_attributes
 539         """
 540         self._compute()
 541         # TODO: namespace all the pandas keys
 542         d = self._translate()
 543         # filter out empty styles, every cell will have a class
 544         # but the list of props may just be [['', '']].
 545         # so we have the nested anys below
 546         trimmed = [x for x in d["cellstyle"] if any(any(y) for y in x["props"])]
 547         d["cellstyle"] = trimmed
 548         d.update(kwargs)
 549         return self.template.render(**d)

 551     def _update_ctx(self, attrs: DataFrame) -> None:
 552         """
 553         Update the state of the Styler.
 554 
 555         Collects a mapping of {index_label: ['<property>: <value>']}.
 556 
 557         Parameters
 558         ----------
 559         attrs : DataFrame
 560             should contain strings of '<property>: <value>;<prop2>: <val2>'
 561             Whitespace shouldn't matter and the final trailing ';' shouldn't
 562             matter.
 563         """
 564         for row_label, v in attrs.iterrows():
 565             for col_label, col in v.items():
 566                 i = self.index.get_indexer([row_label])[0]
 567                 j = self.columns.get_indexer([col_label])[0]
 568                 for pair in col.rstrip(";").split(";"):
 569                     self.ctx[(i, j)].append(pair)
 570 
 571     def _copy(self, deepcopy: bool = False) -> "Styler":

 573             self.data,
 574             precision=self.precision,
 575             caption=self.caption,
 576             uuid=self.uuid,

 578             na_rep=self.na_rep,
 579         )
 580         if deepcopy:
 581             styler.ctx = copy.deepcopy(self.ctx)
 582             styler._todo = copy.deepcopy(self._todo)
 583         else:
 584             styler.ctx = self.ctx
 585             styler._todo = self._todo
 586         return styler

 606     def _compute(self):
 607         """
 608         Execute the style functions built up in `self._todo`.
 609 
 610         Relies on the conventions that all style functions go through
 611         .apply or .applymap. The append styles to apply as tuples of
 612 
 613         (application method, *args, **kwargs)
 614         """
 615         r = self
 616         for func, args, kwargs in self._todo:
 617             r = func(self)(*args, **kwargs)
 618         return r

 620     def _apply(
 621         self,
 622         func: Callable[..., "Styler"],
 623         axis: Optional[Axis] = 0,
 624         subset=None,
 625         **kwargs,
 626     ) -> "Styler":
 627         subset = slice(None) if subset is None else subset
 628         subset = _non_reducing_slice(subset)
 629         data = self.data.loc[subset]
 630         if axis is not None:
 631             result = data.apply(func, axis=axis, result_type="expand", **kwargs)
 632             result.columns = data.columns
 633         else:
 634             result = func(data, **kwargs)
 635             if not isinstance(result, pd.DataFrame):
 636                 raise TypeError(
 637                     f"Function {repr(func)} must return a DataFrame when "
 638                     f"passed to `Styler.apply` with axis=None"
 639                 )
 640             if not (
 641                 result.index.equals(data.index) and result.columns.equals(data.columns)
 642             ):
 643                 raise ValueError(
 644                     f"Result of {repr(func)} must have identical "
 645                     f"index and columns as the input"
 646                 )
 647 
 648         result_shape = result.shape
 649         expected_shape = self.data.loc[subset].shape
 650         if result_shape != expected_shape:
 651             raise ValueError(
 652                 f"Function {repr(func)} returned the wrong shape.\n"
 653                 f"Result has shape: {result.shape}\n"
 654                 f"Expected shape:   {expected_shape}"
 655             )
 656         self._update_ctx(result)
 657         return self

 659     def apply(
 660         self,
 661         func: Callable[..., "Styler"],
 662         axis: Optional[Axis] = 0,
 663         subset=None,
 664         **kwargs,
 665     ) -> "Styler":
 666         """
 667         Apply a function column-wise, row-wise, or table-wise.
 668 
 669         Updates the HTML representation with the result.
 670 
 671         Parameters
 672         ----------
 673         func : function
 674             ``func`` should take a Series or DataFrame (depending
 675             on ``axis``), and return an object with the same shape.
 676             Must return a DataFrame with identical index and
 677             column labels when ``axis=None``.
 678         axis : {0 or 'index', 1 or 'columns', None}, default 0

1097     @staticmethod
1098     def _background_gradient(
1099         s,
1100         cmap="PuBu",
1101         low: float = 0,
1102         high: float = 0,
1103         text_color_threshold: float = 0.408,
1104         vmin: Optional[float] = None,
1105         vmax: Optional[float] = None,
1106     ):
1107         """
1108         Color background in a range according to the data.
1109         """
1110         if (
1111             not isinstance(text_color_threshold, (float, int))
1112             or not 0 <= text_color_threshold <= 1
1113         ):
1114             msg = "`text_color_threshold` must be a value from 0 to 1."
1115             raise ValueError(msg)
1116 
1117         with _mpl(Styler.background_gradient) as (plt, colors):
1118             smin = np.nanmin(s.to_numpy()) if vmin is None else vmin
1119             smax = np.nanmax(s.to_numpy()) if vmax is None else vmax
1120             rng = smax - smin
1121             # extend lower / upper bounds, compresses color range
1122             norm = colors.Normalize(smin - (rng * low), smax + (rng * high))
1123             # matplotlib colors.Normalize modifies inplace?
1124             # https://github.com/matplotlib/matplotlib/issues/5427
1125             rgbas = plt.cm.get_cmap(cmap)(norm(s.to_numpy(dtype=float)))
1126 
1127             def relative_luminance(rgba) -> float:
1128                 """
1129                 Calculate relative luminance of a color.
1130 
1131                 The calculation adheres to the W3C standards
1132                 (https://www.w3.org/WAI/GL/wiki/Relative_luminance)
1133 
1134                 Parameters
1135                 ----------
1136                 color : rgb or rgba tuple
1137 
1138                 Returns
1139                 -------
1140                 float
1141                     The relative luminance as a value from 0 to 1
1142                 """
1143                 r, g, b = (
1144                     x / 12.92 if x <= 0.03928 else ((x + 0.055) / 1.055 ** 2.4)
1145                     for x in rgba[:3]
1146                 )
1147                 return 0.2126 * r + 0.7152 * g + 0.0722 * b
1148 
1149             def css(rgba) -> str:
1150                 dark = relative_luminance(rgba) < text_color_threshold
1151                 text_color = "#f1f1f1" if dark else "#000000"
1152                 return f"background-color: {colors.rgb2hex(rgba)};color: {text_color};"
1153 
1154             if s.ndim == 1:
1155                 return [css(rgba) for rgba in rgbas]
1156             else:
1157                 return pd.DataFrame(
1158                     [[css(rgba) for rgba in row] for row in rgbas],
1159                     index=s.index,
1160                     columns=s.columns,
1161                 )

1188     @staticmethod
1189     def _bar(
1190         s,
1191         align: str,
1192         colors: List[str],
1193         width: float = 100,
1194         vmin: Optional[float] = None,
1195         vmax: Optional[float] = None,
1196     ):
1197         """
1198         Draw bar chart in dataframe cells.
1199         """
1200         # Get input value range.
1201         smin = np.nanmin(s.to_numpy()) if vmin is None else vmin
1202         smax = np.nanmax(s.to_numpy()) if vmax is None else vmax
1203         if align == "mid":
1204             smin = min(0, smin)
1205             smax = max(0, smax)
1206         elif align == "zero":
1207             # For "zero" mode, we want the range to be symmetrical around zero.
1208             smax = max(abs(smin), abs(smax))
1209             smin = -smax
1210         # Transform to percent-range of linear-gradient
1211         normed = width * (s.to_numpy(dtype=float) - smin) / (smax - smin + 1e-12)
1212         zero = -width * smin / (smax - smin + 1e-12)
1213 
1214         def css_bar(start: float, end: float, color: str) -> str:
1215             """
1216             Generate CSS code to draw a bar from start to end.
1217             """
1218             css = "width: 10em; height: 80%;"
1219             if end > start:
1220                 css += "background: linear-gradient(90deg,"
1221                 if start > 0:
1222                     css += f" transparent {start:.1f}%, {color} {start:.1f}%, "
1223                 e = min(end, width)
1224                 css += f"{color} {e:.1f}%, transparent {e:.1f}%)"
1225             return css
1226 
1227         def css(x):
1228             if pd.isna(x):
1229                 return ""
1230 
1231             # avoid deprecated indexing `colors[x > zero]`
1232             color = colors[1] if x > zero else colors[0]
1233 
1234             if align == "left":
1235                 return css_bar(0, x, color)
1236             else:
1237                 return css_bar(min(x, zero), max(x, zero), color)
1238 
1239         if s.ndim == 1:
1240             return [css(x) for x in normed]
1241         else:
1242             return pd.DataFrame(
1243                 [[css(x) for x in row] for row in normed],
1244                 index=s.index,
1245                 columns=s.columns,
1246             )

1439     def pipe(self, func: Callable, *args, **kwargs):
1440         """
1441         Apply ``func(self, *args, **kwargs)``, and return the result.
1442 
1443         .. versionadded:: 0.24.0
1444 
1445         Parameters
1446         ----------
1447         func : function
1448             Function to apply to the Styler.  Alternatively, a
1449             ``(callable, keyword)`` tuple where ``keyword`` is a string
1450             indicating the keyword of ``callable`` that expects the Styler.
1451         *args : optional
1452             Arguments passed to `func`.
1453         **kwargs : optional
1454             A dictionary of keyword arguments passed into ``func``.
1455 
1456         Returns
1457         -------
1458         object :
1459             The value returned by ``func``.
1460 
1461         See Also
1462         --------
1463         DataFrame.pipe : Analogous method for DataFrame.
1464         Styler.apply : Apply a function row-wise, column-wise, or table-wise to
1465             modify the dataframe's styling.
1466 
1467         Notes
1468         -----
1469         Like :meth:`DataFrame.pipe`, this method can simplify the
1470         application of several user-defined functions to a styler.  Instead
1471         of writing:
1472 
1473         .. code-block:: python
1474 
1475             f(g(df.style.set_precision(3), arg1=a), arg2=b, arg3=c)
1476 
1477         users can write:
1478 
1479         .. code-block:: python
1480 
1481             (df.style.set_precision(3)
1482                .pipe(g, arg1=a)
1483                .pipe(f, arg2=b, arg3=c))
1484 
1485         In particular, this allows users to define functions that take a
1486         styler object, along with other parameters, and return the styler after
1487         making styling changes (such as calling :meth:`Styler.apply` or
1488         :meth:`Styler.set_properties`).  Using ``.pipe``, these user-defined
1489         style "transformations" can be interleaved with calls to the built-in
1490         Styler interface.
1491 
1492         Examples
1493         --------
1494         >>> def format_conversion(styler):
1495         ...     return (styler.set_properties(**{'text-align': 'right'})
1496         ...                   .format({'conversion': '{:.1%}'}))
1497 
1498         The user-defined ``format_conversion`` function above can be called
1499         within a sequence of other style modifications:
1500 
1501         >>> df = pd.DataFrame({'trial': list(range(5)),
1502         ...                    'conversion': [0.75, 0.85, np.nan, 0.7, 0.72]})
1503         >>> (df.style
1504         ...    .highlight_min(subset=['conversion'], color='yellow')
1505         ...    .pipe(format_conversion)
1506         ...    .set_caption("Results with minimum conversion highlighted."))
1507         """
1508         return com.pipe(self, func, *args, **kwargs)

1518 def _get_level_lengths(index, hidden_elements=None):
1519     """
1520     Given an index, find the level length for each element.
1521 
1522     Optional argument is a list of index positions which
1523     should not be visible.
1524 
1525     Result is a dictionary of (level, initial_position): span
1526     """
1527     levels = index.format(sparsify=lib.no_default, adjoin=False, names=False)
1528 
1529     if hidden_elements is None:
1530         hidden_elements = []
1531 
1532     lengths = {}
1533     if index.nlevels == 1:
1534         for i, value in enumerate(levels):
1535             if i not in hidden_elements:
1536                 lengths[(0, i)] = 1
1537         return lengths
1538 
1539     for i, lvl in enumerate(levels):
1540         for j, row in enumerate(lvl):
1541             if not get_option("display.multi_sparse"):
1542                 lengths[(i, j)] = 1
1543             elif (row is not lib.no_default) and (j not in hidden_elements):
1544                 last_label = j
1545                 lengths[(i, last_label)] = 1
1546             elif row is not lib.no_default:
1547                 # even if its hidden, keep track of it in case
1548                 # length >1 and later elements are visible
1549                 last_label = j
1550                 lengths[(i, last_label)] = 0
1551             elif j not in hidden_elements:
1552                 lengths[(i, last_label)] += 1
1553 
1554     non_zero_lengths = {
1555         element: length for element, length in lengths.items() if length >= 1
1556     }
1557 
1558     return non_zero_lengths

1561 def _maybe_wrap_formatter(
1562     formatter: Union[Callable, str], na_rep: Optional[str]
1563 ) -> Callable:
1564     if isinstance(formatter, str):
1565         formatter_func = lambda x: formatter.format(x)
1566     elif callable(formatter):
1567         formatter_func = formatter
1568     else:
1569         msg = f"Expected a template string or callable, got {formatter} instead"
1570         raise TypeError(msg)
1571 
1572     if na_rep is None:
1573         return formatter_func
1574     elif isinstance(na_rep, str):
1575         return lambda x: na_rep if pd.isna(x) else formatter_func(x)
1576     else:
1577         msg = f"Expected a string, got {na_rep} instead"
