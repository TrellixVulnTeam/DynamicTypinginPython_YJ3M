  98     def write_th(
  99         self, s: Any, header: bool = False, indent: int = 0, tags: Optional[str] = None
 100     ) -> None:
 101         """
 102         Method for writing a formatted <th> cell.
 103 
 104         If col_space is set on the formatter then that is used for
 105         the value of min-width.
 106 
 107         Parameters
 108         ----------
 109         s : object
 110             The data to be written inside the cell.
 111         header : bool, default False
 112             Set to True if the <th> is for use inside <thead>.  This will
 113             cause min-width to be set if there is one.
 114         indent : int, default 0
 115             The indentation level of the cell.
 116         tags : str, default None
 117             Tags to include in the cell.
 118 
 119         Returns
 120         -------
 121         A written <th> cell.
 122         """
 123         if header and self.fmt.col_space is not None:
 124             tags = tags or ""
 125             tags += f'style="min-width: {self.fmt.col_space};"'
 126 
 127         self._write_cell(s, kind="th", indent=indent, tags=tags)

 129     def write_td(self, s: Any, indent: int = 0, tags: Optional[str] = None) -> None:

 132     def _write_cell(
 133         self, s: Any, kind: str = "td", indent: int = 0, tags: Optional[str] = None
 134     ) -> None:
 135         if tags is not None:
 136             start_tag = f"<{kind} {tags}>"
 137         else:
 138             start_tag = f"<{kind}>"
 139 
 140         if self.escape:
 141             # escape & first to prevent double escaping of &
 142             esc = {"&": r"&amp;", "<": r"&lt;", ">": r"&gt;"}
 143         else:
 144             esc = {}
 145 
 146         rs = pprint_thing(s, escape_chars=esc).strip()
 147 
 148         if self.render_links and is_url(rs):
 149             rs_unescaped = pprint_thing(s, escape_chars={}).strip()
 150             start_tag += f'<a href="{rs_unescaped}" target="_blank">'
 151             end_a = "</a>"
 152         else:
 153             end_a = ""
 154 
 155         self.write(f"{start_tag}{rs}{end_a}</{kind}>", indent)

 157     def write_tr(
 158         self,
 159         line: Iterable,
 160         indent: int = 0,
 161         indent_delta: int = 0,
 162         header: bool = False,
 163         align: Optional[str] = None,
 164         tags: Optional[Dict[int, str]] = None,
 165         nindex_levels: int = 0,
 166     ) -> None:
 167         if tags is None:
 168             tags = {}
 169 
 170         if align is None:
 171             self.write("<tr>", indent)
 172         else:
 173             self.write(f'<tr style="text-align: {align};">', indent)
 174         indent += indent_delta
 175 
 176         for i, s in enumerate(line):
 177             val_tag = tags.get(i, None)
 178             if header or (self.bold_rows and i < nindex_levels):
 179                 self.write_th(s, indent=indent, header=header, tags=val_tag)
 180             else:
 181                 self.write_td(s, indent, tags=val_tag)
 182 
 183         indent -= indent_delta
 184         self.write("</tr>", indent)

 200     def _write_table(self, indent: int = 0) -> None:
 201         _classes = ["dataframe"]  # Default class.
 202         use_mathjax = get_option("display.html.use_mathjax")
 203         if not use_mathjax:
 204             _classes.append("tex2jax_ignore")
 205         if self.classes is not None:
 206             if isinstance(self.classes, str):
 207                 self.classes = self.classes.split()
 208             if not isinstance(self.classes, (list, tuple)):
 209                 raise TypeError(
 210                     "classes must be a string, list, "
 211                     f"or tuple, not {type(self.classes)}"
 212                 )
 213             _classes.extend(self.classes)
 214 
 215         if self.table_id is None:
 216             id_section = ""
 217         else:
 218             id_section = f' id="{self.table_id}"'
 219 
 220         self.write(
 221             f'<table border="{self.border}" class="{" ".join(_classes)}"{id_section}>',
 222             indent,
 223         )
 224 
 225         if self.fmt.header or self.show_row_idx_names:
 226             self._write_header(indent + self.indent_delta)
 227 
 228         self._write_body(indent + self.indent_delta)
 229 
 230         self.write("</table>", indent)

 232     def _write_col_header(self, indent: int) -> None:
 233         truncate_h = self.fmt.truncate_h
 234         if isinstance(self.columns, MultiIndex):
 235             template = 'colspan="{span:d}" halign="left"'
 236 
 237             if self.fmt.sparsify:
 238                 # GH3547
 239                 sentinel = lib.no_default
 240             else:
 241                 sentinel = False
 242             levels = self.columns.format(sparsify=sentinel, adjoin=False, names=False)
 243             level_lengths = get_level_lengths(levels, sentinel)
 244             inner_lvl = len(level_lengths) - 1
 245             for lnum, (records, values) in enumerate(zip(level_lengths, levels)):
 246                 if truncate_h:
 247                     # modify the header lines
 248                     ins_col = self.fmt.tr_col_num
 249                     if self.fmt.sparsify:
 250                         recs_new = {}
 251                         # Increment tags after ... col.
 252                         for tag, span in list(records.items()):
 253                             if tag >= ins_col:
 254                                 recs_new[tag + 1] = span
 255                             elif tag + span > ins_col:
 256                                 recs_new[tag] = span + 1
 257                                 if lnum == inner_lvl:
 258                                     values = (
 259                                         values[:ins_col] + ("...",) + values[ins_col:]
 260                                     )
 261                                 else:
 262                                     # sparse col headers do not receive a ...
 263                                     values = (
 264                                         values[:ins_col]
 265                                         + (values[ins_col - 1],)
 266                                         + values[ins_col:]
 267                                     )
 268                             else:
 269                                 recs_new[tag] = span
 270                             # if ins_col lies between tags, all col headers
 271                             # get ...
 272                             if tag + span == ins_col:
 273                                 recs_new[ins_col] = 1
 274                                 values = values[:ins_col] + ("...",) + values[ins_col:]
 275                         records = recs_new
 276                         inner_lvl = len(level_lengths) - 1
 277                         if lnum == inner_lvl:
 278                             records[ins_col] = 1
 279                     else:
 280                         recs_new = {}
 281                         for tag, span in list(records.items()):
 282                             if tag >= ins_col:
 283                                 recs_new[tag + 1] = span
 284                             else:
 285                                 recs_new[tag] = span
 286                         recs_new[ins_col] = 1
 287                         records = recs_new
 288                         values = values[:ins_col] + ["..."] + values[ins_col:]
 289 
 290                 # see gh-22579
 291                 # Column Offset Bug with to_html(index=False) with
 292                 # MultiIndex Columns and Index.
 293                 # Initially fill row with blank cells before column names.
 294                 # TODO: Refactor to remove code duplication with code
 295                 # block below for standard columns index.
 296                 row = [""] * (self.row_levels - 1)
 297                 if self.fmt.index or self.show_col_idx_names:
 298                     # see gh-22747
 299                     # If to_html(index_names=False) do not show columns
 300                     # index names.
 301                     # TODO: Refactor to use _get_column_name_list from
 302                     # DataFrameFormatter class and create a
 303                     # _get_formatted_column_labels function for code
 304                     # parity with DataFrameFormatter class.
 305                     if self.fmt.show_index_names:
 306                         name = self.columns.names[lnum]
 307                         row.append(pprint_thing(name or ""))
 308                     else:
 309                         row.append("")
 310 
 311                 tags = {}
 312                 j = len(row)
 313                 for i, v in enumerate(values):
 314                     if i in records:
 315                         if records[i] > 1:
 316                             tags[j] = template.format(span=records[i])
 317                     else:
 318                         continue
 319                     j += 1
 320                     row.append(v)
 321                 self.write_tr(row, indent, self.indent_delta, tags=tags, header=True)
 322         else:
 323             # see gh-22579
 324             # Column misalignment also occurs for
 325             # a standard index when the columns index is named.
 326             # Initially fill row with blank cells before column names.
 327             # TODO: Refactor to remove code duplication with code block
 328             # above for columns MultiIndex.
 329             row = [""] * (self.row_levels - 1)
 330             if self.fmt.index or self.show_col_idx_names:
 331                 # see gh-22747
 332                 # If to_html(index_names=False) do not show columns
 333                 # index names.
 334                 # TODO: Refactor to use _get_column_name_list from
 335                 # DataFrameFormatter class.
 336                 if self.fmt.show_index_names:
 337                     row.append(self.columns.name or "")
 338                 else:
 339                     row.append("")
 340             row.extend(self._get_columns_formatted_values())
 341             align = self.fmt.justify
 342 
 343             if truncate_h:
 344                 ins_col = self.row_levels + self.fmt.tr_col_num
 345                 row.insert(ins_col, "...")
 346 
 347             self.write_tr(row, indent, self.indent_delta, header=True, align=align)

 349     def _write_row_header(self, indent: int) -> None:
 350         truncate_h = self.fmt.truncate_h
 351         row = [x if x is not None else "" for x in self.frame.index.names] + [""] * (
 352             self.ncols + (1 if truncate_h else 0)
 353         )
 354         self.write_tr(row, indent, self.indent_delta, header=True)

 367     def _get_formatted_values(self) -> Dict[int, List[str]]:

 372     def _write_body(self, indent: int) -> None:
 373         self.write("<tbody>", indent)
 374         fmt_values = self._get_formatted_values()
 375 
 376         # write values
 377         if self.fmt.index and isinstance(self.frame.index, MultiIndex):
 378             self._write_hierarchical_rows(fmt_values, indent + self.indent_delta)
 379         else:
 380             self._write_regular_rows(fmt_values, indent + self.indent_delta)

 382         self.write("</tbody>", indent)

 384     def _write_regular_rows(
 385         self, fmt_values: Mapping[int, List[str]], indent: int
 386     ) -> None:
 387         truncate_h = self.fmt.truncate_h
 388         truncate_v = self.fmt.truncate_v
 389 
 390         nrows = len(self.fmt.tr_frame)
 391 
 392         if self.fmt.index:
 393             fmt = self.fmt._get_formatter("__index__")
 394             if fmt is not None:
 395                 index_values = self.fmt.tr_frame.index.map(fmt)
 396             else:
 397                 index_values = self.fmt.tr_frame.index.format()
 398 
 399         row: List[str] = []
 400         for i in range(nrows):
 401 
 402             if truncate_v and i == (self.fmt.tr_row_num):
 403                 str_sep_row = ["..."] * len(row)
 404                 self.write_tr(
 405                     str_sep_row,
 406                     indent,
 407                     self.indent_delta,
 408                     tags=None,
 409                     nindex_levels=self.row_levels,
 410                 )
 411 
 412             row = []
 413             if self.fmt.index:
 414                 row.append(index_values[i])
 415             # see gh-22579
 416             # Column misalignment also occurs for
 417             # a standard index when the columns index is named.
 418             # Add blank cell before data cells.
 419             elif self.show_col_idx_names:
 420                 row.append("")
 421             row.extend(fmt_values[j][i] for j in range(self.ncols))
 422 
 423             if truncate_h:
 424                 dot_col_ix = self.fmt.tr_col_num + self.row_levels
 425                 row.insert(dot_col_ix, "...")
 426             self.write_tr(
 427                 row, indent, self.indent_delta, tags=None, nindex_levels=self.row_levels
 428             )

 430     def _write_hierarchical_rows(
 431         self, fmt_values: Mapping[int, List[str]], indent: int
 432     ) -> None:
 433         template = 'rowspan="{span}" valign="top"'
 434 
 435         truncate_h = self.fmt.truncate_h
 436         truncate_v = self.fmt.truncate_v
 437         frame = self.fmt.tr_frame
 438         nrows = len(frame)
 439 
 440         idx_values = frame.index.format(sparsify=False, adjoin=False, names=False)
 441         idx_values = list(zip(*idx_values))
 442 
 443         if self.fmt.sparsify:
 444             # GH3547
 445             sentinel = lib.no_default
 446             levels = frame.index.format(sparsify=sentinel, adjoin=False, names=False)
 447 
 448             level_lengths = get_level_lengths(levels, sentinel)
 449             inner_lvl = len(level_lengths) - 1
 450             if truncate_v:
 451                 # Insert ... row and adjust idx_values and
 452                 # level_lengths to take this into account.
 453                 ins_row = self.fmt.tr_row_num
 454                 # cast here since if truncate_v is True, self.fmt.tr_row_num is not None
 455                 ins_row = cast(int, ins_row)
 456                 inserted = False
 457                 for lnum, records in enumerate(level_lengths):
 458                     rec_new = {}
 459                     for tag, span in list(records.items()):
 460                         if tag >= ins_row:
 461                             rec_new[tag + 1] = span
 462                         elif tag + span > ins_row:
 463                             rec_new[tag] = span + 1
 464 
 465                             # GH 14882 - Make sure insertion done once
 466                             if not inserted:
 467                                 dot_row = list(idx_values[ins_row - 1])
 468                                 dot_row[-1] = "..."
 469                                 idx_values.insert(ins_row, tuple(dot_row))
 470                                 inserted = True
 471                             else:
 472                                 dot_row = list(idx_values[ins_row])
 473                                 dot_row[inner_lvl - lnum] = "..."
 474                                 idx_values[ins_row] = tuple(dot_row)
 475                         else:
 476                             rec_new[tag] = span
 477                         # If ins_row lies between tags, all cols idx cols
 478                         # receive ...
 479                         if tag + span == ins_row:
 480                             rec_new[ins_row] = 1
 481                             if lnum == 0:
 482                                 idx_values.insert(
 483                                     ins_row, tuple(["..."] * len(level_lengths))
 484                                 )
 485 
 486                             # GH 14882 - Place ... in correct level
 487                             elif inserted:
 488                                 dot_row = list(idx_values[ins_row])
 489                                 dot_row[inner_lvl - lnum] = "..."
 490                                 idx_values[ins_row] = tuple(dot_row)
 491                     level_lengths[lnum] = rec_new
 492 
 493                 level_lengths[inner_lvl][ins_row] = 1
 494                 for ix_col in range(len(fmt_values)):
 495                     fmt_values[ix_col].insert(ins_row, "...")
 496                 nrows += 1
 497 
 498             for i in range(nrows):
 499                 row = []
 500                 tags = {}
 501 
 502                 sparse_offset = 0
 503                 j = 0
 504                 for records, v in zip(level_lengths, idx_values[i]):
 505                     if i in records:
 506                         if records[i] > 1:
 507                             tags[j] = template.format(span=records[i])
 508                     else:
 509                         sparse_offset += 1
 510                         continue
 511 
 512                     j += 1
 513                     row.append(v)
 514 
 515                 row.extend(fmt_values[j][i] for j in range(self.ncols))
 516                 if truncate_h:
 517                     row.insert(
 518                         self.row_levels - sparse_offset + self.fmt.tr_col_num, "..."
 519                     )
 520                 self.write_tr(
 521                     row,
 522                     indent,
 523                     self.indent_delta,
 524                     tags=tags,
 525                     nindex_levels=len(levels) - sparse_offset,
 526                 )
 527         else:
 528             row = []
 529             for i in range(len(frame)):
 530                 if truncate_v and i == (self.fmt.tr_row_num):
 531                     str_sep_row = ["..."] * len(row)
 532                     self.write_tr(
 533                         str_sep_row,
 534                         indent,
 535                         self.indent_delta,
 536                         tags=None,
 537                         nindex_levels=self.row_levels,
 538                     )
 539 
 540                 idx_values = list(
 541                     zip(*frame.index.format(sparsify=False, adjoin=False, names=False))
 542                 )
 543                 row = []
 544                 row.extend(idx_values[i])
 545                 row.extend(fmt_values[j][i] for j in range(self.ncols))
 546                 if truncate_h:
 547                     row.insert(self.row_levels + self.fmt.tr_col_num, "...")
 548                 self.write_tr(
 549                     row,
 550                     indent,
 551                     self.indent_delta,
 552                     tags=None,
 553                     nindex_levels=frame.index.nlevels,
 554                 )

 557 class NotebookFormatter(HTMLFormatter):
 558     """
 559     Internal class for formatting output data in html for display in Jupyter
 560     Notebooks. This class is intended for functionality specific to
 561     DataFrame._repr_html_() and DataFrame.to_html(notebook=True)
 562     """
 563 
 564     def _get_formatted_values(self) -> Dict[int, List[str]]:
 565         return {i: self.fmt._format_col(i) for i in range(self.ncols)}
 566 
 567     def _get_columns_formatted_values(self) -> List[str]:
 568         return self.columns.format()
 569 
 570     def write_style(self) -> None:
 571         # We use the "scoped" attribute here so that the desired
 572         # style properties for the data frame are not then applied
 573         # throughout the entire notebook.
 574         template_first = """\
 575             <style scoped>"""
 576         template_last = """\
 577             </style>"""
 578         template_select = """\
 579                 .dataframe %s {
 580                     %s: %s;
 581                 }"""
 582         element_props = [
 583             ("tbody tr th:only-of-type", "vertical-align", "middle"),
 584             ("tbody tr th", "vertical-align", "top"),
 585         ]
 586         if isinstance(self.columns, MultiIndex):
 587             element_props.append(("thead tr th", "text-align", "left"))
 588             if self.show_row_idx_names:
 589                 element_props.append(
 590                     ("thead tr:last-of-type th", "text-align", "right")
 591                 )
 592         else:
 593             element_props.append(("thead th", "text-align", "right"))
 594         template_mid = "\n\n".join(map(lambda t: template_select % t, element_props))
 595         template = dedent("\n".join((template_first, template_mid, template_last)))
 596         self.write(template)
 597 
 598     def render(self) -> List[str]:
 599         self.write("<div>")
 600         self.write_style()
 601         super().render()
 602         self.write("</div>")
 603         return self.elements
