 176     def _get_footer(self) -> str:
 177         footer = ""
 178 
 179         if self.length:
 180             if footer:
 181                 footer += ", "
 182             footer += f"Length: {len(self.categorical)}"
 183 
 184         level_info = self.categorical._repr_categories_info()
 185 
 186         # Levels are added in a newline
 187         if footer:

 189         footer += level_info

 203 
 204         if len(categorical) == 0:
 205             if self.footer:
 206                 return self._get_footer()

 256     def _chk_truncate(self) -> None:
 257         from pandas.core.reshape.concat import concat
 258 
 259         self.tr_row_num: Optional[int]
 260 
 261         min_rows = self.min_rows
 262         max_rows = self.max_rows
 263         # truncation determined by max_rows, actual truncated number of rows
 264         # used below by min_rows
 265         truncate_v = max_rows and (len(self.series) > max_rows)
 266         series = self.series
 267         if truncate_v:
 268             max_rows = cast(int, max_rows)
 269             if min_rows:
 270                 # if min_rows is set (not None or 0), set max_rows to minimum
 271                 # of both
 272                 max_rows = min(min_rows, max_rows)
 273             if max_rows == 1:
 274                 row_num = max_rows
 275                 series = series.iloc[:max_rows]
 276             else:
 277                 row_num = max_rows // 2
 278                 series = concat((series.iloc[:row_num], series.iloc[-row_num:]))
 279             self.tr_row_num = row_num
 280         else:
 281             self.tr_row_num = None
 282         self.tr_series = series
 283         self.truncate_v = truncate_v
 284 
 285     def _get_footer(self) -> str:
 286         name = self.series.name
 287         footer = ""
 288 
 289         if getattr(self.series.index, "freq", None) is not None:
 290             assert isinstance(
 291                 self.series.index, (DatetimeIndex, PeriodIndex, TimedeltaIndex)
 292             )
 293             footer += f"Freq: {self.series.index.freqstr}"
 294 
 295         if self.name is not False and name is not None:
 296             if footer:
 297                 footer += ", "
 298 
 299             series_name = pprint_thing(name, escape_chars=("\t", "\r", "\n"))
 300             footer += f"Name: {series_name}"
 301 
 302         if self.length is True or (self.length == "truncate" and self.truncate_v):
 303             if footer:
 304                 footer += ", "
 305             footer += f"Length: {len(self.series)}"
 306 
 307         if self.dtype is not False and self.dtype is not None:
 308             dtype_name = getattr(self.tr_series.dtype, "name", None)
 309             if dtype_name:
 310                 if footer:
 311                     footer += ", "
 312                 footer += f"dtype: {pprint_thing(dtype_name)}"
 313 
 314         # level infos are added to the end and in a new line, like it is done
 315         # for Categoricals
 316         if is_categorical_dtype(self.tr_series.dtype):
 317             level_info = self.tr_series._values._repr_categories_info()
 318             if footer:
 319                 footer += "\n"
 320             footer += level_info
 321 
 322         return str(footer)

 324     def _get_formatted_index(self) -> Tuple[List[str], bool]:
 325         index = self.tr_series.index
 326         is_multi = isinstance(index, MultiIndex)
 327 
 328         if is_multi:
 329             have_header = any(name for name in index.names)
 330             fmt_index = index.format(names=True)
 331         else:
 332             have_header = index.name is not None
 333             fmt_index = index.format(name=True)
 334         return fmt_index, have_header
 335 
 336     def _get_formatted_values(self) -> List[str]:
 337         return format_array(
 338             self.tr_series._values,
 339             None,
 340             float_format=self.float_format,
 341             na_rep=self.na_rep,
 342         )
 343 
 344     def to_string(self) -> str:
 345         series = self.tr_series
 346         footer = self._get_footer()
 347 
 348         if len(series) == 0:
 349             return f"{type(self.series).__name__}([], {footer})"
 350 
 351         fmt_index, have_header = self._get_formatted_index()
 352         fmt_values = self._get_formatted_values()
 353 
 354         if self.truncate_v:
 355             n_header_rows = 0
 356             row_num = self.tr_row_num
 357             row_num = cast(int, row_num)
 358             width = self.adj.len(fmt_values[row_num - 1])
 359             if width > 3:
 360                 dot_str = "..."
 361             else:
 362                 dot_str = ".."
 363             # Series uses mode=center because it has single value columns
 364             # DataFrame uses mode=left
 365             dot_str = self.adj.justify([dot_str], width, mode="center")[0]
 366             fmt_values.insert(row_num + n_header_rows, dot_str)
 367             fmt_index.insert(row_num + 1, "")
 368 
 369         if self.index:
 370             result = self.adj.adjoin(3, *[fmt_index[1:], fmt_values])
 371         else:
 372             result = self.adj.adjoin(3, fmt_values)
 373 
 374         if self.header and have_header:
 375             result = fmt_index[0] + "\n" + result
 376 
 377         if footer:
 378             result += "\n" + footer
 379 
 380         return str("".join(result))

 397 class EastAsianTextAdjustment(TextAdjustment):
 398     def __init__(self):
 399         super().__init__()
 400         if get_option("display.unicode.ambiguous_as_wide"):
 401             self.ambiguous_width = 2
 402         else:
 403             self.ambiguous_width = 1
 404 
 405         # Definition of East Asian Width
 406         # https://unicode.org/reports/tr11/
 407         # Ambiguous width can be changed by option
 408         self._EAW_MAP = {"Na": 1, "N": 1, "W": 2, "F": 2, "H": 1}
 409 
 410     def len(self, text: str) -> int:
 411         """
 412         Calculate display width considering unicode East Asian Width
 413         """
 414         if not isinstance(text, str):
 415             return len(text)
 416 
 417         return sum(
 418             self._EAW_MAP.get(east_asian_width(c), self.ambiguous_width) for c in text
 419         )
 420 
 421     def justify(
 422         self, texts: Iterable[str], max_len: int, mode: str = "right"
 423     ) -> List[str]:
 424         # re-calculate padding space per str considering East Asian Width
 425         def _get_pad(t):
 426             return max_len - self.len(t) + len(t)
 427 
 428         if mode == "left":
 429             return [x.ljust(_get_pad(x)) for x in texts]
 430         elif mode == "center":
 431             return [x.center(_get_pad(x)) for x in texts]
 432         else:
 433             return [x.rjust(_get_pad(x)) for x in texts]
 434 
 435 
 436 def _get_adjustment() -> TextAdjustment:
 437     use_east_asian_width = get_option("display.unicode.east_asian_width")
 438     if use_east_asian_width:
 439         return EastAsianTextAdjustment()
 440     else:
 441         return TextAdjustment()

 469     @contextmanager
 470     def get_buffer(
 471         self, buf: Optional[FilePathOrBuffer[str]], encoding: Optional[str] = None
 472     ):
 473         """
 474         Context manager to open, yield and close buffer for filenames or Path-like
 475         objects, otherwise yield buf unchanged.
 476         """
 477         if buf is not None:
 478             buf = stringify_path(buf)
 479         else:
 480             buf = StringIO()
 481 
 482         if encoding is None:
 483             encoding = "utf-8"
 484         elif not isinstance(buf, str):
 485             raise ValueError("buf is not a file name and encoding is specified.")
 486 
 487         if hasattr(buf, "write"):
 488             yield buf
 489         elif isinstance(buf, str):
 490             with open(buf, "w", encoding=encoding, newline="") as f:
 491                 # GH#30034 open instead of codecs.open prevents a file leak
 492                 #  if we have an invalid encoding argument.
 493                 # newline="" is needed to roundtrip correctly on
 494                 #  windows test_to_latex_filename
 495                 yield f
 496         else:
 497             raise TypeError("buf is not a file name and it has no write method")
 498 
 499     def write_result(self, buf: IO[str]) -> None:
 500         """
 501         Write the result of serialization to buf.
 502         """
 503         raise AbstractMethodError(self)
 504 
 505     def get_result(
 506         self,
 507         buf: Optional[FilePathOrBuffer[str]] = None,
 508         encoding: Optional[str] = None,
 509     ) -> Optional[str]:
 510         """
 511         Perform serialization. Write to buf or return as string if buf is None.
 512         """
 513         with self.get_buffer(buf, encoding=encoding) as f:
 514             self.write_result(buf=f)
 515             if buf is None:
 516                 return f.getvalue()
 517             return None

 606     def _chk_truncate(self) -> None:
 607         """
 608         Checks whether the frame should be truncated. If so, slices
 609         the frame up.
 610         """
 611         from pandas.core.reshape.concat import concat
 612 
 613         # Cut the data to the information actually printed
 614         max_cols = self.max_cols
 615         max_rows = self.max_rows
 616         self.max_rows_adj: Optional[int]
 617         max_rows_adj: Optional[int]
 618 
 619         if max_cols == 0 or max_rows == 0:  # assume we are in the terminal
 620             (w, h) = get_terminal_size()
 621             self.w = w
 622             self.h = h
 623             if self.max_rows == 0:
 624                 dot_row = 1
 625                 prompt_row = 1
 626                 if self.show_dimensions:
 627                     show_dimension_rows = 3
 628                 # assume we only get here if self.header is boolean.
 629                 # i.e. not to_latex() where self.header may be List[str]
 630                 self.header = cast(bool, self.header)
 631                 n_add_rows = self.header + dot_row + show_dimension_rows + prompt_row
 632                 # rows available to fill with actual data
 633                 max_rows_adj = self.h - n_add_rows
 634                 self.max_rows_adj = max_rows_adj
 635 
 636             # Format only rows and columns that could potentially fit the
 637             # screen
 638             if max_cols == 0 and len(self.frame.columns) > w:
 639                 max_cols = w
 640             if max_rows == 0 and len(self.frame) > h:
 641                 max_rows = h
 642 
 643         if not hasattr(self, "max_rows_adj"):
 644             if max_rows:
 645                 if (len(self.frame) > max_rows) and self.min_rows:
 646                     # if truncated, set max_rows showed to min_rows
 647                     max_rows = min(self.min_rows, max_rows)
 648             self.max_rows_adj = max_rows

 853     def _join_multiline(self, *args) -> str:
 854         lwidth = self.line_width
 855         adjoin_width = 1
 856         strcols = list(args)
 857         if self.index:

 860

 862             np.array([self.adj.len(x) for x in col]).max() if len(col) > 0 else 0
 863             for col in strcols

 865

1187     def _format_strings(self) -> List[str]:
1188         if self.float_format is None:
1189             float_format = get_option("display.float_format")
1190             if float_format is None:
1191                 precision = get_option("display.precision")
1192                 float_format = lambda x: f"{x: .{precision:d}g}"
1193         else:
1194             float_format = self.float_format
1195 
1196         formatter = (
1197             self.formatter
1198             if self.formatter is not None
1199             else (lambda x: pprint_thing(x, escape_chars=("\t", "\r", "\n")))
1200         )
1201 
1202         def _format(x):
1203             if self.na_rep is not None and is_scalar(x) and isna(x):
1204                 try:
1205                     # try block for np.isnat specifically
1206                     # determine na_rep if x is None or NaT-like
1207                     if x is None:
1208                         return "None"
1209                     elif x is NA:
1210                         return str(NA)
1211                     elif x is NaT or np.isnat(x):
1212                         return "NaT"
1213                 except (TypeError, ValueError):
1214                     # np.isnat only handles datetime or timedelta objects
1215                     pass
1216                 return self.na_rep
1217             elif isinstance(x, PandasObject):
1218                 return str(x)
1219             else:
1220                 # object dtype
1221                 return str(formatter(x))
1222 
1223         vals = extract_array(self.values, extract_numpy=True)
1224 
1225         is_float_type = (
1226             lib.map_infer(vals, is_float)
1227             # vals may have 2 or more dimensions
1228             & np.all(notna(vals), axis=tuple(range(1, len(vals.shape))))
1229         )
1230         leading_space = self.leading_space
1231         if leading_space is None:
1232             leading_space = is_float_type.any()
1233 
1234         fmt_values = []
1235         for i, v in enumerate(vals):
1236             if not is_float_type[i] and leading_space:
1237                 fmt_values.append(f" {_format(v)}")
1238             elif is_float_type[i]:
1239                 fmt_values.append(float_format(v))
1240             else:
1241                 if leading_space is False:
1242                     # False specifically, so that the default is
1243                     # to include a space if we get here.
1244                     tpl = "{v}"
1245                 else:
1246                     tpl = " {v}"
1247                 fmt_values.append(tpl.format(v=_format(v)))
1248 
1249         return fmt_values

1279         # we are going to compose different functions, to first convert to
1280         # a string, then replace the decimal symbol, and finally chop according
1281         # to the threshold
1282 
1283         # when there is no float_format, we use str instead of '%g'
1284         # because str(0.0) = '0.0' while '%g' % 0.0 = '0'
1285         if float_format:
1286 
1287             def base_formatter(v):
1288                 return float_format(value=v) if notna(v) else self.na_rep
1289 
1290         else:
1291 
1292             def base_formatter(v):
1293                 return str(v) if notna(v) else self.na_rep
1294 
1295         if self.decimal != ".":
1296 
1297             def decimal_formatter(v):
1298                 return base_formatter(v).replace(".", self.decimal, 1)
1299 
1300         else:
1301             decimal_formatter = base_formatter
1302 
1303         if threshold is None:
1304             return decimal_formatter
1305 
1306         def formatter(value):
1307             if notna(value):
1308                 if abs(value) > threshold:
1309                     return decimal_formatter(value)
1310                 else:
1311                     return decimal_formatter(0.0)
1312             else:
1313                 return self.na_rep
1314 
1315         return formatter

1317     def get_result_as_array(self) -> np.ndarray:
1318         """

1320         the parameters given at initialisation, as a numpy array
1321         """
1322         if self.formatter is not None:
1323             return np.array([self.formatter(x) for x in self.values])
1324 
1325         if self.fixed_width:
1326             threshold = get_option("display.chop_threshold")
1327         else:
1328             threshold = None
1329 
1330         # if we have a fixed_width, we'll need to try different float_format
1331         def format_values_with(float_format):
1332             formatter = self._value_formatter(float_format, threshold)
1333 
1334             # default formatter leaves a space to the left when formatting
1335             # floats, must be consistent for left-justifying NaNs (GH #25061)
1336             if self.justify == "left":
1337                 na_rep = " " + self.na_rep
1338             else:
1339                 na_rep = self.na_rep
1340

1342             values = self.values
1343             is_complex = is_complex_dtype(values)
1344             mask = isna(values)
1345             values = np.array(values, dtype="object")
1346             values[mask] = na_rep
1347             imask = (~mask).ravel()
1348             values.flat[imask] = np.array(
1349                 [formatter(val) for val in values.ravel()[imask]]
1350             )
1351 
1352             if self.fixed_width:
1353                 if is_complex:
1354                     result = _trim_zeros_complex(values, na_rep)
1355                 else:
1356                     result = _trim_zeros_float(values, na_rep)
1357                 return np.asarray(result, dtype="object")
1358 
1359             return values
1360 
1361         # There is a special default string when we are fixed-width
1362         # The default is otherwise to use str instead of a formatting string
1363         float_format: Optional[FloatFormatType]
1364         if self.float_format is None:
1365             if self.fixed_width:
1366                 float_format = partial(
1367                     "{value: .{digits:d}f}".format, digits=self.digits
1368                 )
1369             else:
1370                 float_format = self.float_format
1371         else:
1372             float_format = lambda value: self.float_format % value
1373 
1374         formatted_values = format_values_with(float_format)
1375 
1376         if not self.fixed_width:
1377             return formatted_values
1378 
1379         # we need do convert to engineering format if some values are too small
1380         # and would appear as 0, or if some values are too big and take too
1381         # much space
1382 
1383         if len(formatted_values) > 0:
1384             maxlen = max(len(x) for x in formatted_values)
1385             too_long = maxlen > self.digits + 6
1386         else:
1387             too_long = False
1388 
1389         with np.errstate(invalid="ignore"):
1390             abs_vals = np.abs(self.values)
1391             # this is pretty arbitrary for now
1392             # large values: more that 8 characters including decimal symbol
1393             # and first digit, hence > 1e6
1394             has_large_values = (abs_vals > 1e6).any()
1395             has_small_values = (
1396                 (abs_vals < 10 ** (-self.digits)) & (abs_vals > 0)
1397             ).any()
1398 
1399         if has_small_values or (too_long and has_large_values):
1400             float_format = partial("{value: .{digits:d}e}".format, digits=self.digits)
1401             formatted_values = format_values_with(float_format)
1402 
1403         return formatted_values

1407         if self.formatter is not None:
1408             return [self.formatter(x) for x in self.values]

1475 def format_percentiles(
1476     percentiles: Union[
1477         np.ndarray, List[Union[int, float]], List[float], List[Union[str, float]]
1478     ]
1479 ) -> List[str]:
1480     """
1481     Outputs rounded and formatted percentiles.
1482 
1483     Parameters
1484     ----------
1485     percentiles : list-like, containing floats from interval [0,1]
1486 
1487     Returns
1488     -------
1489     formatted : list of strings
1490 
1491     Notes
1492     -----
1493     Rounding precision is chosen so that: (1) if any two elements of
1494     ``percentiles`` differ, they remain different after rounding
1495     (2) no entry is *rounded* to 0% or 100%.
1496     Any non-integer is always rounded to at least 1 decimal place.
1497 
1498     Examples
1499     --------
1500     Keeps all entries different after rounding:
1501 
1502     >>> format_percentiles([0.01999, 0.02001, 0.5, 0.666666, 0.9999])
1503     ['1.999%', '2.001%', '50%', '66.667%', '99.99%']
1504 
1505     No element is rounded to 0% or 100% (unless already equal to it).
1506     Duplicates are allowed:
1507 
1508     >>> format_percentiles([0, 0.5, 0.02001, 0.5, 0.666666, 0.9999])
1509     ['0%', '50%', '2.0%', '50%', '66.67%', '99.99%']
1510     """
1511     percentiles = np.asarray(percentiles)
1512 
1513     # It checks for np.NaN as well
1514     with np.errstate(invalid="ignore"):
1515         if (
1516             not is_numeric_dtype(percentiles)
1517             or not np.all(percentiles >= 0)
1518             or not np.all(percentiles <= 1)
1519         ):
1520             raise ValueError("percentiles should all be in the interval [0,1]")
1521 
1522     percentiles = 100 * percentiles
1523     int_idx = np.isclose(percentiles.astype(int), percentiles)
1524 
1525     if np.all(int_idx):
1526         out = percentiles.astype(int).astype(str)
1527         return [i + "%" for i in out]
1528 
1529     unique_pcts = np.unique(percentiles)
1530     to_begin = unique_pcts[0] if unique_pcts[0] > 0 else None
1531     to_end = 100 - unique_pcts[-1] if unique_pcts[-1] < 100 else None
1532 
1533     # Least precision that keeps percentiles unique after rounding
1534     prec = -np.floor(
1535         np.log10(np.min(np.ediff1d(unique_pcts, to_begin=to_begin, to_end=to_end)))
1536     ).astype(int)
1537     prec = max(1, prec)
1538     out = np.empty_like(percentiles, dtype=object)
1539     out[int_idx] = percentiles[int_idx].astype(int).astype(str)
1540     out[~int_idx] = percentiles[~int_idx].round(prec).astype(str)
1541     return [i + "%" for i in out]

1770 class EngFormatter:
1771     """
1772     Formats float values according to engineering format.
1773 
1774     Based on matplotlib.ticker.EngFormatter
1775     """
1776 
1777     # The SI engineering prefixes
1778     ENG_PREFIXES = {
1779         -24: "y",
1780         -21: "z",
1781         -18: "a",
1782         -15: "f",
1783         -12: "p",
1784         -9: "n",
1785         -6: "u",
1786         -3: "m",
1787         0: "",
1788         3: "k",
1789         6: "M",
1790         9: "G",
1791         12: "T",
1792         15: "P",
1793         18: "E",
1794         21: "Z",
1795         24: "Y",
1796     }
1797 
1798     def __init__(self, accuracy: Optional[int] = None, use_eng_prefix: bool = False):
1799         self.accuracy = accuracy
1800         self.use_eng_prefix = use_eng_prefix
1801 
1802     def __call__(self, num: Union[int, float]) -> str:
1803         """
1804         Formats a number in engineering notation, appending a letter
1805         representing the power of 1000 of the original number. Some examples:
1806 
1807         >>> format_eng(0)       # for self.accuracy = 0
1808         ' 0'
1809 
1810         >>> format_eng(1000000) # for self.accuracy = 1,
1811                                 #     self.use_eng_prefix = True
1812         ' 1.0M'
1813 
1814         >>> format_eng("-1e-6") # for self.accuracy = 2
1815                                 #     self.use_eng_prefix = False
1816         '-1.00E-06'
1817 
1818         @param num: the value to represent
1819         @type num: either a numeric value or a string that can be converted to
1820                    a numeric value (as per decimal.Decimal constructor)
1821 
1822         @return: engineering formatted string
1823         """
1824         dnum = decimal.Decimal(str(num))
1825 
1826         if decimal.Decimal.is_nan(dnum):
1827             return "NaN"
1828 
1829         if decimal.Decimal.is_infinite(dnum):
1830             return "inf"
1831 
1832         sign = 1
1833 
1834         if dnum < 0:  # pragma: no cover
1835             sign = -1
1836             dnum = -dnum
1837 
1838         if dnum != 0:
1839             pow10 = decimal.Decimal(int(math.floor(dnum.log10() / 3) * 3))
1840         else:
1841             pow10 = decimal.Decimal(0)
1842 
1843         pow10 = pow10.min(max(self.ENG_PREFIXES.keys()))
1844         pow10 = pow10.max(min(self.ENG_PREFIXES.keys()))
1845         int_pow10 = int(pow10)
1846 
1847         if self.use_eng_prefix:
1848             prefix = self.ENG_PREFIXES[int_pow10]
1849         else:
1850             if int_pow10 < 0:
1851                 prefix = f"E-{-int_pow10:02d}"
1852             else:
1853                 prefix = f"E+{int_pow10:02d}"
1854 
1855         mant = sign * dnum / (10 ** pow10)
1856 
1857         if self.accuracy is None:  # pragma: no cover
1858             format_str = "{mant: g}{prefix}"
1859         else:
1860             format_str = f"{{mant: .{self.accuracy:d}f}}{{prefix}}"
1861 
1862         formatted = format_str.format(mant=mant, prefix=prefix)
1863 
1864         return formatted
