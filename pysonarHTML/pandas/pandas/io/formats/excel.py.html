  43 class CSSToExcelConverter:
  44     """
  45     A callable for converting CSS declarations to ExcelWriter styles
  46 
  47     Supports parts of CSS 2.2, with minimal CSS 3.0 support (e.g. text-shadow),
  48     focusing on font styling, backgrounds, borders and alignment.
  49 
  50     Operates by first computing CSS styles in a fairly generic
  51     way (see :meth:`compute_css`) then determining Excel style
  52     properties from CSS properties (see :meth:`build_xlstyle`).
  53 
  54     Parameters
  55     ----------
  56     inherited : str, optional
  57         CSS declarations understood to be the containing scope for the
  58         CSS processed by :meth:`__call__`.
  59     """
  60 
  61     # NB: Most of the methods here could be classmethods, as only __init__
  62     #     and __call__ make use of instance attributes.  We leave them as
  63     #     instancemethods so that users can easily experiment with extensions
  64     #     without monkey-patching.
  65 
  66     def __init__(self, inherited: Optional[str] = None):
  67         if inherited is not None:
  68             inherited = self.compute_css(inherited)
  69 
  70         self.inherited = inherited
  71 
  72     compute_css = CSSResolver()

  74     def __call__(self, declarations_str: str) -> Dict[str, Dict[str, str]]:
  75         """
  76         Convert CSS declarations to ExcelWriter style.
  77 
  78         Parameters
  79         ----------
  80         declarations_str : str
  81             List of CSS declarations.
  82             e.g. "font-weight: bold; background: blue"
  83 
  84         Returns
  85         -------
  86         xlstyle : dict
  87             A style as interpreted by ExcelWriter when found in
  88             ExcelCell.style.
  89         """
  90         # TODO: memoize?
  91         properties = self.compute_css(declarations_str, self.inherited)
  92         return self.build_xlstyle(properties)
  93 
  94     def build_xlstyle(self, props: Dict[str, str]) -> Dict[str, Dict[str, str]]:
  95         out = {
  96             "alignment": self.build_alignment(props),
  97             "border": self.build_border(props),
  98             "fill": self.build_fill(props),
  99             "font": self.build_font(props),
 100             "number_format": self.build_number_format(props),
 101         }
 102 
 103         # TODO: handle cell width and height: needs support in pandas.io.excel
 104 
 105         def remove_none(d: Dict[str, str]) -> None:
 106             """Remove key where value is None, through nested dicts"""
 107             for k, v in list(d.items()):
 108                 if v is None:
 109                     del d[k]
 110                 elif isinstance(v, dict):
 111                     remove_none(v)
 112                     if not v:
 113                         del d[k]
 114 
 115         remove_none(out)
 116         return out

 140     def build_border(self, props: Dict) -> Dict[str, Dict[str, str]]:
 141         return {
 142             side: {

 144                     props.get(f"border-{side}-style"),
 145                     props.get(f"border-{side}-width"),

 148             }
 149             for side in ["top", "right", "bottom", "left"]

 152     def _border_style(self, style: Optional[str], width):
 153         # convert styles and widths to openxml, one of:
 154         #       'dashDot'
 155         #       'dashDotDot'
 156         #       'dashed'
 157         #       'dotted'
 158         #       'double'
 159         #       'hair'
 160         #       'medium'
 161         #       'mediumDashDot'
 162         #       'mediumDashDotDot'
 163         #       'mediumDashed'
 164         #       'slantDashDot'
 165         #       'thick'
 166         #       'thin'
 167         if width is None and style is None:
 168             return None
 169         if style == "none" or style == "hidden":
 170             return None
 171 
 172         if width is None:
 173             width = "2pt"
 174         width = float(width[:-2])
 175         if width < 1e-5:
 176             return None
 177         elif width < 1.3:
 178             width_name = "thin"
 179         elif width < 2.8:
 180             width_name = "medium"
 181         else:
 182             width_name = "thick"
 183 
 184         if style in (None, "groove", "ridge", "inset", "outset"):
 185             # not handled
 186             style = "solid"
 187 
 188         if style == "double":
 189             return "double"
 190         if style == "solid":
 191             return width_name
 192         if style == "dotted":
 193             if width_name in ("hair", "thin"):
 194                 return "dotted"
 195             return "mediumDashDotDot"
 196         if style == "dashed":
 197             if width_name in ("hair", "thin"):
 198                 return "dashed"
 199             return "mediumDashed"

 225     def build_font(self, props) -> Dict[str, Optional[Union[bool, int, str]]]:
 226         size = props.get("font-size")
 227         if size is not None:
 228             assert size.endswith("pt")
 229             size = float(size[:-2])
 230 
 231         font_names_tmp = re.findall(
 232             r"""(?x)
 233             (
 234             "(?:[^"]|\\")+"
 235             |
 236             '(?:[^']|\\')+'
 237             |
 238             [^'",]+
 239             )(?=,|\s*$)
 240         """,
 241             props.get("font-family", ""),
 242         )
 243         font_names = []
 244         for name in font_names_tmp:
 245             if name[:1] == '"':
 246                 name = name[1:-1].replace('\\"', '"')
 247             elif name[:1] == "'":
 248                 name = name[1:-1].replace("\\'", "'")
 249             else:
 250                 name = name.strip()
 251             if name:
 252                 font_names.append(name)
 253 
 254         family = None
 255         for name in font_names:
 256             if name == "serif":
 257                 family = 1  # roman
 258                 break
 259             elif name == "sans-serif":
 260                 family = 2  # swiss
 261                 break
 262             elif name == "cursive":
 263                 family = 4  # script
 264                 break
 265             elif name == "fantasy":
 266                 family = 5  # decorative
 267                 break
 268 
 269         decoration = props.get("text-decoration")
 270         if decoration is not None:
 271             decoration = decoration.split()
 272         else:
 273             decoration = ()
 274 
 275         return {
 276             "name": font_names[0] if font_names else None,
 277             "family": family,
 278             "size": size,
 279             "bold": self.BOLD_MAP.get(props.get("font-weight")),
 280             "italic": self.ITALIC_MAP.get(props.get("font-style")),
 281             "underline": ("single" if "underline" in decoration else None),
 282             "strike": ("line-through" in decoration) or None,
 283             "color": self.color_to_excel(props.get("color")),
 284             # shadow if nonzero digit before shadow color
 285             "shadow": (
 286                 bool(re.search("^[^#(]*[1-9]", props["text-shadow"]))
 287                 if "text-shadow" in props
 288                 else None
 289             ),
 290             # FIXME: dont leave commented-out
 291             # 'vertAlign':,
 292             # 'charset': ,
 293             # 'scheme': ,
 294             # 'outline': ,
 295             # 'condense': ,
 296         }

 321     def color_to_excel(self, val: Optional[str]):
 322         if val is None:
 323             return None
 324         if val.startswith("#") and len(val) == 7:
 325             return val[1:].upper()
 326         if val.startswith("#") and len(val) == 4:
 327             return (val[1] * 2 + val[2] * 2 + val[3] * 2).upper()
 328         try:
 329             return self.NAMED_COLORS[val]
 330         except KeyError:
 331             warnings.warn(f"Unhandled color format: {repr(val)}", CSSWarning)

 337 class ExcelFormatter:
 338     """
 339     Class for formatting a DataFrame to a list of ExcelCells,
 340 
 341     Parameters
 342     ----------
 343     df : DataFrame or Styler
 344     na_rep: na representation
 345     float_format : string, default None
 346             Format string for floating point numbers
 347     cols : sequence, optional
 348         Columns to write
 349     header : boolean or list of string, default True
 350         Write out column names. If a list of string is given it is
 351         assumed to be aliases for the column names
 352     index : boolean, default True
 353         output row names (index)
 354     index_label : string or sequence, default None
 355             Column label for index column(s) if desired. If None is given, and
 356             `header` and `index` are True, then the index names are used. A
 357             sequence should be given if the DataFrame uses MultiIndex.
 358     merge_cells : boolean, default False
 359             Format MultiIndex and Hierarchical Rows as merged cells.
 360     inf_rep : string, default `'inf'`
 361         representation for np.inf values (which aren't representable in Excel)
 362         A `'-'` sign will be added in front of -inf.
 363     style_converter : callable, optional
 364         This translates Styler styles (CSS) into ExcelWriter styles.
 365         Defaults to ``CSSToExcelConverter()``.
 366         It should have signature css_declarations string -> excel style.
 367         This is only called for body cells.
 368     """
 369 
 370     max_rows = 2 ** 20
 371     max_cols = 2 ** 14
 372 
 373     def __init__(
 374         self,
 375         df,
 376         na_rep: str = "",
 377         float_format: Optional[str] = None,
 378         cols: Optional[Sequence[Label]] = None,
 379         header: Union[Sequence[Label], bool] = True,
 380         index: bool = True,
 381         index_label: Optional[Union[Label, Sequence[Label]]] = None,
 382         merge_cells: bool = False,
 383         inf_rep: str = "inf",
 384         style_converter: Optional[Callable] = None,
 385     ):
 386         self.rowcounter = 0
 387         self.na_rep = na_rep
 388         if not isinstance(df, DataFrame):
 389             self.styler = df
 390             df = df.data
 391             if style_converter is None:
 392                 style_converter = CSSToExcelConverter()
 393             self.style_converter = style_converter
 394         else:
 395             self.styler = None
 396         self.df = df
 397         if cols is not None:
 398 
 399             # all missing, raise
 400             if not len(Index(cols) & df.columns):
 401                 raise KeyError("passes columns are not ALL present dataframe")
 402 
 403             if len(Index(cols) & df.columns) != len(cols):
 404                 # Deprecated in GH#17295, enforced in 1.0.0
 405                 raise KeyError("Not all names specified in 'columns' are found")
 406 
 407             self.df = df.reindex(columns=cols)
 408 
 409         self.columns = self.df.columns
 410         self.float_format = float_format
 411         self.index = index
 412         self.index_label = index_label
 413         self.header = header
 414         self.merge_cells = merge_cells
 415         self.inf_rep = inf_rep

 419         return {
 420             "font": {"bold": True},
 421             "borders": {
 422                 "top": "thin",
 423                 "right": "thin",
 424                 "bottom": "thin",
 425                 "left": "thin",
 426             },
 427             "alignment": {"horizontal": "center", "vertical": "top"},

 430     def _format_value(self, val):
 431         if is_scalar(val) and missing.isna(val):
 432             val = self.na_rep
 433         elif is_float(val):
 434             if missing.isposinf_scalar(val):
 435                 val = self.inf_rep
 436             elif missing.isneginf_scalar(val):
 437                 val = f"-{self.inf_rep}"
 438             elif self.float_format is not None:
 439                 val = float(self.float_format % val)
 440         if getattr(val, "tzinfo", None) is not None:
 441             raise ValueError(
 442                 "Excel does not support datetimes with "
 443                 "timezones. Please ensure that datetimes "
 444                 "are timezone unaware before writing to Excel."
 445             )
 446         return val

 448     def _format_header_mi(self):
 449         if self.columns.nlevels > 1:
 450             if not self.index:
 451                 raise NotImplementedError(
 452                     "Writing to Excel with MultiIndex columns and no "
 453                     "index ('index'=False) is not yet implemented."
 454                 )
 455 
 456         has_aliases = isinstance(self.header, (tuple, list, np.ndarray, ABCIndex))
 457         if not (has_aliases or self.header):
 458             return
 459 
 460         columns = self.columns
 461         level_strs = columns.format(
 462             sparsify=self.merge_cells, adjoin=False, names=False
 463         )
 464         level_lengths = get_level_lengths(level_strs)
 465         coloffset = 0
 466         lnum = 0
 467 
 468         if self.index and isinstance(self.df.index, MultiIndex):
 469             coloffset = len(self.df.index[0]) - 1
 470 
 471         if self.merge_cells:
 472             # Format multi-index as a merged cells.
 473             for lnum in range(len(level_lengths)):
 474                 name = columns.names[lnum]
 475                 yield ExcelCell(lnum, coloffset, name, self.header_style)
 476 
 477             for lnum, (spans, levels, level_codes) in enumerate(
 478                 zip(level_lengths, columns.levels, columns.codes)
 479             ):
 480                 values = levels.take(level_codes)
 481                 for i in spans:
 482                     if spans[i] > 1:
 483                         yield ExcelCell(
 484                             lnum,
 485                             coloffset + i + 1,
 486                             values[i],
 487                             self.header_style,
 488                             lnum,
 489                             coloffset + i + spans[i],
 490                         )
 491                     else:
 492                         yield ExcelCell(
 493                             lnum, coloffset + i + 1, values[i], self.header_style
 494                         )
 495         else:
 496             # Format in legacy format with dots to indicate levels.
 497             for i, values in enumerate(zip(*level_strs)):
 498                 v = ".".join(map(pprint_thing, values))
 499                 yield ExcelCell(lnum, coloffset + i + 1, v, self.header_style)
 500 
 501         self.rowcounter = lnum
 502 
 503     def _format_header_regular(self):
 504         has_aliases = isinstance(self.header, (tuple, list, np.ndarray, ABCIndex))
 505         if has_aliases or self.header:
 506             coloffset = 0
 507 
 508             if self.index:
 509                 coloffset = 1
 510                 if isinstance(self.df.index, MultiIndex):
 511                     coloffset = len(self.df.index[0])
 512 
 513             colnames = self.columns
 514             if has_aliases:
 515                 if len(self.header) != len(self.columns):
 516                     raise ValueError(
 517                         f"Writing {len(self.columns)} cols but got {len(self.header)} "
 518                         "aliases"
 519                     )
 520                 else:
 521                     colnames = self.header
 522 
 523             for colindex, colname in enumerate(colnames):
 524                 yield ExcelCell(
 525                     self.rowcounter, colindex + coloffset, colname, self.header_style
 526                 )

 528     def _format_header(self):
 529         if isinstance(self.columns, MultiIndex):
 530             gen = self._format_header_mi()
 531         else:
 532             gen = self._format_header_regular()
 533 
 534         gen2 = ()
 535         if self.df.index.names:
 536             row = [x if x is not None else "" for x in self.df.index.names] + [
 537                 ""
 538             ] * len(self.columns)
 539             if reduce(lambda x, y: x and y, map(lambda x: x != "", row)):
 540                 gen2 = (
 541                     ExcelCell(self.rowcounter, colindex, val, self.header_style)
 542                     for colindex, val in enumerate(row)
 543                 )
 544                 self.rowcounter += 1
 545         return itertools.chain(gen, gen2)

 547     def _format_body(self):
 548         if isinstance(self.df.index, MultiIndex):
 549             return self._format_hierarchical_rows()
 550         else:
 551             return self._format_regular_rows()

 553     def _format_regular_rows(self):
 554         has_aliases = isinstance(self.header, (tuple, list, np.ndarray, ABCIndex))
 555         if has_aliases or self.header:
 556             self.rowcounter += 1
 557 
 558         # output index and index_label?
 559         if self.index:
 560             # check aliases
 561             # if list only take first as this is not a MultiIndex
 562             if self.index_label and isinstance(
 563                 self.index_label, (list, tuple, np.ndarray, Index)
 564             ):
 565                 index_label = self.index_label[0]
 566             # if string good to go
 567             elif self.index_label and isinstance(self.index_label, str):
 568                 index_label = self.index_label
 569             else:
 570                 index_label = self.df.index.names[0]
 571 
 572             if isinstance(self.columns, MultiIndex):
 573                 self.rowcounter += 1
 574 
 575             if index_label and self.header is not False:
 576                 yield ExcelCell(self.rowcounter - 1, 0, index_label, self.header_style)
 577 
 578             # write index_values
 579             index_values = self.df.index
 580             if isinstance(self.df.index, PeriodIndex):
 581                 index_values = self.df.index.to_timestamp()
 582 
 583             for idx, idxval in enumerate(index_values):
 584                 yield ExcelCell(self.rowcounter + idx, 0, idxval, self.header_style)
 585 
 586             coloffset = 1
 587         else:
 588             coloffset = 0
 589 
 590         for cell in self._generate_body(coloffset):
 591             yield cell

 593     def _format_hierarchical_rows(self):
 594         has_aliases = isinstance(self.header, (tuple, list, np.ndarray, ABCIndex))
 595         if has_aliases or self.header:
 596             self.rowcounter += 1
 597 
 598         gcolidx = 0
 599 
 600         if self.index:
 601             index_labels = self.df.index.names
 602             # check for aliases
 603             if self.index_label and isinstance(
 604                 self.index_label, (list, tuple, np.ndarray, Index)
 605             ):
 606                 index_labels = self.index_label
 607 
 608             # MultiIndex columns require an extra row
 609             # with index names (blank if None) for
 610             # unambiguous round-trip, unless not merging,
 611             # in which case the names all go on one row Issue #11328
 612             if isinstance(self.columns, MultiIndex) and self.merge_cells:
 613                 self.rowcounter += 1
 614 
 615             # if index labels are not empty go ahead and dump
 616             if com.any_not_none(*index_labels) and self.header is not False:
 617 
 618                 for cidx, name in enumerate(index_labels):
 619                     yield ExcelCell(self.rowcounter - 1, cidx, name, self.header_style)
 620 
 621             if self.merge_cells:
 622                 # Format hierarchical rows as merged cells.
 623                 level_strs = self.df.index.format(
 624                     sparsify=True, adjoin=False, names=False
 625                 )
 626                 level_lengths = get_level_lengths(level_strs)
 627 
 628                 for spans, levels, level_codes in zip(
 629                     level_lengths, self.df.index.levels, self.df.index.codes
 630                 ):
 631 
 632                     values = levels.take(
 633                         level_codes, allow_fill=levels._can_hold_na, fill_value=True
 634                     )
 635 
 636                     for i in spans:
 637                         if spans[i] > 1:
 638                             yield ExcelCell(
 639                                 self.rowcounter + i,
 640                                 gcolidx,
 641                                 values[i],
 642                                 self.header_style,
 643                                 self.rowcounter + i + spans[i] - 1,
 644                                 gcolidx,
 645                             )
 646                         else:
 647                             yield ExcelCell(
 648                                 self.rowcounter + i,
 649                                 gcolidx,
 650                                 values[i],
 651                                 self.header_style,
 652                             )
 653                     gcolidx += 1
 654 
 655             else:
 656                 # Format hierarchical rows with non-merged values.
 657                 for indexcolvals in zip(*self.df.index):
 658                     for idx, indexcolval in enumerate(indexcolvals):
 659                         yield ExcelCell(
 660                             self.rowcounter + idx,
 661                             gcolidx,
 662                             indexcolval,
 663                             self.header_style,
 664                         )
 665                     gcolidx += 1
 666 
 667         for cell in self._generate_body(gcolidx):
 668             yield cell

 670     def _generate_body(self, coloffset: int):
 671         if self.styler is None:
 672             styles = None
 673         else:
 674             styles = self.styler._compute().ctx
 675             if not styles:
 676                 styles = None
 677         xlstyle = None
 678 
 679         # Write the body of the frame data series by series.
 680         for colidx in range(len(self.columns)):
 681             series = self.df.iloc[:, colidx]
 682             for i, val in enumerate(series):
 683                 if styles is not None:
 684                     xlstyle = self.style_converter(";".join(styles[i, colidx]))
 685                 yield ExcelCell(self.rowcounter + i, colidx + coloffset, val, xlstyle)
 686 
 687     def get_formatted_cells(self):
 688         for cell in itertools.chain(self._format_header(), self._format_body()):
 689             cell.val = self._format_value(cell.val)
 690             yield cell

 692     def write(
 693         self,
 694         writer,
 695         sheet_name="Sheet1",
 696         startrow=0,
 697         startcol=0,
 698         freeze_panes=None,
 699         engine=None,
 700     ):
 701         """
 702         writer : string or ExcelWriter object
 703             File path or existing ExcelWriter
 704         sheet_name : string, default 'Sheet1'
 705             Name of sheet which will contain DataFrame
 706         startrow :
 707             upper left cell row to dump data frame
 708         startcol :
 709             upper left cell column to dump data frame
 710         freeze_panes : tuple of integer (length 2), default None
 711             Specifies the one-based bottommost row and rightmost column that
 712             is to be frozen
 713         engine : string, default None
 714             write engine to use if writer is a path - you can also set this
 715             via the options ``io.excel.xlsx.writer``, ``io.excel.xls.writer``,
 716             and ``io.excel.xlsm.writer``.
 717         """
 718         from pandas.io.excel import ExcelWriter
 719 
 720         num_rows, num_cols = self.df.shape
 721         if num_rows > self.max_rows or num_cols > self.max_cols:
 722             raise ValueError(
 723                 f"This sheet is too large! Your sheet size is: {num_rows}, {num_cols} "
 724                 f"Max sheet size is: {self.max_rows}, {self.max_cols}"
 725             )
 726 
 727         if isinstance(writer, ExcelWriter):
 728             need_save = False
 729         else:
 730             writer = ExcelWriter(stringify_path(writer), engine=engine)
 731             need_save = True
 732 
 733         formatted_cells = self.get_formatted_cells()
 734         writer.write_cells(
 735             formatted_cells,
 736             sheet_name,
 737             startrow=startrow,
 738             startcol=startcol,
 739             freeze_panes=freeze_panes,
 740         )
 741         if need_save:
 742             writer.save()
