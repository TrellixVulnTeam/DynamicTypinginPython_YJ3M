  13 class LatexFormatter(TableFormatter):
  14     """
  15     Used to render a DataFrame to a LaTeX tabular/longtable environment output.
  16 
  17     Parameters
  18     ----------
  19     formatter : `DataFrameFormatter`
  20     column_format : str, default None
  21         The columns format as specified in `LaTeX table format
  22         <https://en.wikibooks.org/wiki/LaTeX/Tables>`__ e.g 'rcl' for 3 columns
  23     longtable : boolean, default False
  24         Use a longtable environment instead of tabular.
  25 
  26     See Also
  27     --------
  28     HTMLFormatter
  29     """
  30 
  31     def __init__(
  32         self,
  33         formatter: DataFrameFormatter,
  34         column_format: Optional[str] = None,
  35         longtable: bool = False,
  36         multicolumn: bool = False,
  37         multicolumn_format: Optional[str] = None,
  38         multirow: bool = False,
  39         caption: Optional[str] = None,
  40         label: Optional[str] = None,
  41     ):
  42         self.fmt = formatter
  43         self.frame = self.fmt.frame
  44         self.bold_rows = self.fmt.bold_rows
  45         self.column_format = column_format
  46         self.longtable = longtable
  47         self.multicolumn = multicolumn
  48         self.multicolumn_format = multicolumn_format
  49         self.multirow = multirow
  50         self.caption = caption
  51         self.label = label
  52         self.escape = self.fmt.escape
  53 
  54     def write_result(self, buf: IO[str]) -> None:
  55         """
  56         Render a DataFrame to a LaTeX tabular, longtable, or table/tabular
  57         environment output.
  58         """
  59         # string representation of the columns
  60         if len(self.frame.columns) == 0 or len(self.frame.index) == 0:
  61             info_line = (
  62                 f"Empty {type(self.frame).__name__}\n"
  63                 f"Columns: {self.frame.columns}\n"
  64                 f"Index: {self.frame.index}"
  65             )
  66             strcols = [[info_line]]
  67         else:
  68             strcols = self.fmt._to_str_columns()
  69 
  70         def get_col_type(dtype):
  71             if issubclass(dtype.type, np.number):
  72                 return "r"
  73             else:
  74                 return "l"
  75 
  76         # reestablish the MultiIndex that has been joined by _to_str_column
  77         if self.fmt.index and isinstance(self.frame.index, ABCMultiIndex):
  78             out = self.frame.index.format(
  79                 adjoin=False,
  80                 sparsify=self.fmt.sparsify,
  81                 names=self.fmt.has_index_names,
  82                 na_rep=self.fmt.na_rep,
  83             )
  84 
  85             # index.format will sparsify repeated entries with empty strings
  86             # so pad these with some empty space
  87             def pad_empties(x):
  88                 for pad in reversed(x):
  89                     if pad:
  90                         break
  91                 return [x[0]] + [i if i else " " * len(pad) for i in x[1:]]
  92 
  93             out = (pad_empties(i) for i in out)
  94 
  95             # Add empty spaces for each column level
  96             clevels = self.frame.columns.nlevels
  97             out = [[" " * len(i[-1])] * clevels + i for i in out]
  98 
  99             # Add the column names to the last index column
 100             cnames = self.frame.columns.names
 101             if any(cnames):
 102                 new_names = [i if i else "{}" for i in cnames]
 103                 out[self.frame.index.nlevels - 1][:clevels] = new_names
 104 
 105             # Get rid of old multiindex column and add new ones
 106             strcols = out + strcols[1:]
 107 
 108         if self.column_format is None:
 109             dtypes = self.frame.dtypes._values
 110             column_format = "".join(map(get_col_type, dtypes))
 111             if self.fmt.index:
 112                 index_format = "l" * self.frame.index.nlevels
 113                 column_format = index_format + column_format
 114         elif not isinstance(self.column_format, str):  # pragma: no cover
 115             raise AssertionError(
 116                 f"column_format must be str or unicode, not {type(column_format)}"
 117             )
 118         else:
 119             column_format = self.column_format
 120 
 121         if self.longtable:
 122             self._write_longtable_begin(buf, column_format)
 123         else:
 124             self._write_tabular_begin(buf, column_format)
 125 
 126         buf.write("\\toprule\n")
 127 
 128         ilevels = self.frame.index.nlevels
 129         clevels = self.frame.columns.nlevels
 130         nlevels = clevels
 131         if self.fmt.has_index_names and self.fmt.show_index_names:
 132             nlevels += 1
 133         strrows = list(zip(*strcols))
 134         self.clinebuf: List[List[int]] = []
 135 
 136         for i, row in enumerate(strrows):
 137             if i == nlevels and self.fmt.header:
 138                 buf.write("\\midrule\n")  # End of header
 139                 if self.longtable:
 140                     buf.write("\\endhead\n")
 141                     buf.write("\\midrule\n")
 142                     buf.write(
 143                         f"\\multicolumn{{{len(row)}}}{{r}}"
 144                         "{{Continued on next page}} \\\\\n"
 145                     )
 146                     buf.write("\\midrule\n")
 147                     buf.write("\\endfoot\n\n")
 148                     buf.write("\\bottomrule\n")
 149                     buf.write("\\endlastfoot\n")
 150             if self.escape:
 151                 # escape backslashes first
 152                 crow = [
 153                     (
 154                         x.replace("\\", "\\textbackslash ")
 155                         .replace("_", "\\_")
 156                         .replace("%", "\\%")
 157                         .replace("$", "\\$")
 158                         .replace("#", "\\#")
 159                         .replace("{", "\\{")
 160                         .replace("}", "\\}")
 161                         .replace("~", "\\textasciitilde ")
 162                         .replace("^", "\\textasciicircum ")
 163                         .replace("&", "\\&")
 164                         if (x and x != "{}")
 165                         else "{}"
 166                     )
 167                     for x in row
 168                 ]
 169             else:
 170                 crow = [x if x else "{}" for x in row]
 171             if self.bold_rows and self.fmt.index:
 172                 # bold row labels
 173                 crow = [
 174                     f"\\textbf{{{x}}}"
 175                     if j < ilevels and x.strip() not in ["", "{}"]
 176                     else x
 177                     for j, x in enumerate(crow)
 178                 ]
 179             if i < clevels and self.fmt.header and self.multicolumn:
 180                 # sum up columns to multicolumns
 181                 crow = self._format_multicolumn(crow, ilevels)
 182             if i >= nlevels and self.fmt.index and self.multirow and ilevels > 1:
 183                 # sum up rows to multirows
 184                 crow = self._format_multirow(crow, ilevels, i, strrows)
 185             buf.write(" & ".join(crow))
 186             buf.write(" \\\\\n")
 187             if self.multirow and i < len(strrows) - 1:
 188                 self._print_cline(buf, i, len(strcols))
 189 
 190         if self.longtable:
 191             self._write_longtable_end(buf)
 192         else:
 193             self._write_tabular_end(buf)
 194 
 195     def _format_multicolumn(self, row: List[str], ilevels: int) -> List[str]:
 196         r"""
 197         Combine columns belonging to a group to a single multicolumn entry
 198         according to self.multicolumn_format
 199 
 200         e.g.:
 201         a &  &  & b & c &
 202         will become
 203         \multicolumn{3}{l}{a} & b & \multicolumn{2}{l}{c}
 204         """
 205         row2 = list(row[:ilevels])
 206         ncol = 1
 207         coltext = ""
 208 
 209         def append_col():
 210             # write multicolumn if needed
 211             if ncol > 1:
 212                 row2.append(
 213                     f"\\multicolumn{{{ncol:d}}}{{{self.multicolumn_format}}}"
 214                     f"{{{coltext.strip()}}}"
 215                 )
 216             # don't modify where not needed
 217             else:
 218                 row2.append(coltext)
 219 
 220         for c in row[ilevels:]:
 221             # if next col has text, write the previous
 222             if c.strip():
 223                 if coltext:
 224                     append_col()
 225                 coltext = c
 226                 ncol = 1
 227             # if not, add it to the previous multicolumn
 228             else:
 229                 ncol += 1
 230         # write last column name
 231         if coltext:
 232             append_col()
 233         return row2

 235     def _format_multirow(
 236         self, row: List[str], ilevels: int, i: int, rows: List[Tuple[str, ...]]
 237     ) -> List[str]:
 238         r"""
 239         Check following rows, whether row should be a multirow
 240 
 241         e.g.:     becomes:
 242         a & 0 &   \multirow{2}{*}{a} & 0 &
 243           & 1 &     & 1 &
 244         b & 0 &   \cline{1-2}
 245                   b & 0 &
 246         """
 247         for j in range(ilevels):
 248             if row[j].strip():
 249                 nrow = 1
 250                 for r in rows[i + 1 :]:
 251                     if not r[j].strip():
 252                         nrow += 1
 253                     else:
 254                         break
 255                 if nrow > 1:
 256                     # overwrite non-multirow entry
 257                     row[j] = f"\\multirow{{{nrow:d}}}{{*}}{{{row[j].strip()}}}"
 258                     # save when to end the current block with \cline
 259                     self.clinebuf.append([i + nrow - 1, j + 1])
 260         return row
 261 
 262     def _print_cline(self, buf: IO[str], i: int, icol: int) -> None:
 263         """
 264         Print clines after multirow-blocks are finished.
 265         """
 266         for cl in self.clinebuf:
 267             if cl[0] == i:
 268                 buf.write(f"\\cline{{{cl[1]:d}-{icol:d}}}\n")
 269         # remove entries that have been written to buffer
 270         self.clinebuf = [x for x in self.clinebuf if x[0] != i]

 272     def _write_tabular_begin(self, buf, column_format: str):
 273         """
 274         Write the beginning of a tabular environment or
 275         nested table/tabular environments including caption and label.
 276 
 277         Parameters
 278         ----------
 279         buf : string or file handle
 280             File path or object. If not specified, the result is returned as
 281             a string.
 282         column_format : str
 283             The columns format as specified in `LaTeX table format
 284             <https://en.wikibooks.org/wiki/LaTeX/Tables>`__ e.g 'rcl'
 285             for 3 columns
 286         """
 287         if self.caption is not None or self.label is not None:
 288             # then write output in a nested table/tabular environment
 289             if self.caption is None:
 290                 caption_ = ""
 291             else:
 292                 caption_ = f"\n\\caption{{{self.caption}}}"
 293 
 294             if self.label is None:
 295                 label_ = ""
 296             else:
 297                 label_ = f"\n\\label{{{self.label}}}"
 298 
 299             buf.write(f"\\begin{{table}}\n\\centering{caption_}{label_}\n")
 300         else:
 301             # then write output only in a tabular environment
 302             pass
 303 
 304         buf.write(f"\\begin{{tabular}}{{{column_format}}}\n")
 305 
 306     def _write_tabular_end(self, buf):
 307         """
 308         Write the end of a tabular environment or nested table/tabular
 309         environment.
 310 
 311         Parameters
 312         ----------
 313         buf : string or file handle
 314             File path or object. If not specified, the result is returned as
 315             a string.
 316 
 317         """
 318         buf.write("\\bottomrule\n")
 319         buf.write("\\end{tabular}\n")
 320         if self.caption is not None or self.label is not None:
 321             buf.write("\\end{table}\n")
 322         else:
 323             pass

 325     def _write_longtable_begin(self, buf, column_format: str):
 326         """
 327         Write the beginning of a longtable environment including caption and
 328         label if provided by user.
 329 
 330         Parameters
 331         ----------
 332         buf : string or file handle
 333             File path or object. If not specified, the result is returned as
 334             a string.
 335         column_format : str
 336             The columns format as specified in `LaTeX table format
 337             <https://en.wikibooks.org/wiki/LaTeX/Tables>`__ e.g 'rcl'
 338             for 3 columns
 339         """
 340         buf.write(f"\\begin{{longtable}}{{{column_format}}}\n")
 341 
 342         if self.caption is not None or self.label is not None:
 343             if self.caption is None:
 344                 pass
 345             else:
 346                 buf.write(f"\\caption{{{self.caption}}}")
 347 
 348             if self.label is None:
 349                 pass
 350             else:
 351                 buf.write(f"\\label{{{self.label}}}")
 352 
 353             # a double-backslash is required at the end of the line
 354             # as discussed here:
 355             # https://tex.stackexchange.com/questions/219138
 356             buf.write("\\\\\n")
 357         else:
 358             pass
 359 
 360     @staticmethod
 361     def _write_longtable_end(buf):
 362         """
 363         Write the end of a longtable environment.
 364 
 365         Parameters
 366         ----------
 367         buf : string or file handle
 368             File path or object. If not specified, the result is returned as
 369             a string.
 370 
 371         """
 372         buf.write("\\end{longtable}\n")
