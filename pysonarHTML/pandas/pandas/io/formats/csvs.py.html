  35         self,
  36         obj,
  37         path_or_buf: Optional[FilePathOrBuffer[str]] = None,
  38         sep: str = ",",
  39         na_rep: str = "",
  40         float_format: Optional[str] = None,
  41         cols=None,
  42         header: Union[bool, Sequence[Hashable]] = True,
  43         index: bool = True,
  44         index_label: Optional[Union[bool, Hashable, Sequence[Hashable]]] = None,
  45         mode: str = "w",
  46         encoding: Optional[str] = None,
  47         errors: str = "strict",
  48         compression: Union[str, Mapping[str, str], None] = "infer",
  49         quoting: Optional[int] = None,
  50         line_terminator="\n",
  51         chunksize: Optional[int] = None,
  52         quotechar='"',
  53         date_format: Optional[str] = None,
  54         doublequote: bool = True,
  55         escapechar: Optional[str] = None,
  56         decimal=".",
  57     ):
  58         self.obj = obj
  59 
  60         if path_or_buf is None:
  61             path_or_buf = StringIO()
  62 
  63         # Extract compression mode as given, if dict
  64         compression, self.compression_args = get_compression_method(compression)
  65 
  66         self.path_or_buf, _, _, self.should_close = get_filepath_or_buffer(
  67             path_or_buf, encoding=encoding, compression=compression, mode=mode
  68         )
  69         self.sep = sep
  70         self.na_rep = na_rep
  71         self.float_format = float_format
  72         self.decimal = decimal
  73 
  74         self.header = header
  75         self.index = index
  76         self.index_label = index_label
  77         self.mode = mode
  78         if encoding is None:
  79             encoding = "utf-8"
  80         self.encoding = encoding
  81         self.errors = errors
  82         self.compression = infer_compression(self.path_or_buf, compression)
  83 
  84         if quoting is None:
  85             quoting = csvlib.QUOTE_MINIMAL
  86         self.quoting = quoting
  87 
  88         if quoting == csvlib.QUOTE_NONE:
  89             # prevents crash in _csv
  90             quotechar = None
  91         self.quotechar = quotechar
  92 
  93         self.doublequote = doublequote
  94         self.escapechar = escapechar
  95 
  96         self.line_terminator = line_terminator or os.linesep
  97 
  98         self.date_format = date_format
  99 
 100         self.has_mi_columns = isinstance(obj.columns, ABCMultiIndex)
 101 
 102         # validate mi options
 103         if self.has_mi_columns:
 104             if cols is not None:
 105                 raise TypeError("cannot specify cols with a MultiIndex on the columns")
 106 
 107         if cols is not None:
 108             if isinstance(cols, ABCIndexClass):
 109                 cols = cols.to_native_types(
 110                     na_rep=na_rep,
 111                     float_format=float_format,
 112                     date_format=date_format,
 113                     quoting=self.quoting,
 114                 )
 115             else:
 116                 cols = list(cols)
 117             self.obj = self.obj.loc[:, cols]
 118 
 119         # update columns to include possible multiplicity of dupes
 120         # and make sure sure cols is just a list of labels
 121         cols = self.obj.columns
 122         if isinstance(cols, ABCIndexClass):
 123             cols = cols.to_native_types(
 124                 na_rep=na_rep,
 125                 float_format=float_format,
 126                 date_format=date_format,
 127                 quoting=self.quoting,
 128             )
 129         else:
 130             cols = list(cols)
 131

 133         self.cols = cols
 134 
 135         # preallocate data 2d list
 136         ncols = self.obj.shape[-1]
 137         self.data = [None] * ncols
 138

 140             chunksize = (100000 // (len(self.cols) or 1)) or 1

 158     def save(self) -> None:
 159         """
 160         Create the writer & save.
 161         """
 162         # GH21227 internal compression is not used when file-like passed.
 163         if self.compression and hasattr(self.path_or_buf, "write"):
 164             warnings.warn(
 165                 "compression has no effect when passing file-like object as input.",
 166                 RuntimeWarning,
 167                 stacklevel=2,
 168             )
 169 
 170         # when zip compression is called.
 171         is_zip = isinstance(self.path_or_buf, ZipFile) or (
 172             not hasattr(self.path_or_buf, "write") and self.compression == "zip"
 173         )
 174 
 175         if is_zip:
 176             # zipfile doesn't support writing string to archive. uses string
 177             # buffer to receive csv writing and dump into zip compression
 178             # file handle. GH21241, GH21118
 179             f = StringIO()
 180             close = False
 181         elif hasattr(self.path_or_buf, "write"):
 182             f = self.path_or_buf
 183             close = False
 184         else:
 185             f, handles = get_handle(
 186                 self.path_or_buf,
 187                 self.mode,
 188                 encoding=self.encoding,
 189                 errors=self.errors,
 190                 compression=dict(self.compression_args, method=self.compression),
 191             )
 192             close = True
 193 
 194         try:
 195             # Note: self.encoding is irrelevant here
 196             self.writer = csvlib.writer(
 197                 f,
 198                 lineterminator=self.line_terminator,
 199                 delimiter=self.sep,
 200                 quoting=self.quoting,
 201                 doublequote=self.doublequote,
 202                 escapechar=self.escapechar,
 203                 quotechar=self.quotechar,
 204             )
 205 
 206             self._save()
 207 
 208         finally:
 209             if is_zip:
 210                 # GH17778 handles zip compression separately.
 211                 buf = f.getvalue()
 212                 if hasattr(self.path_or_buf, "write"):
 213                     self.path_or_buf.write(buf)
 214                 else:
 215                     compression = dict(self.compression_args, method=self.compression)
 216 
 217                     f, handles = get_handle(
 218                         self.path_or_buf,
 219                         self.mode,
 220                         encoding=self.encoding,
 221                         errors=self.errors,
 222                         compression=compression,
 223                     )
 224                     f.write(buf)
 225                     close = True
 226             if close:
 227                 f.close()
 228                 for _fh in handles:
 229                     _fh.close()
 230             elif self.should_close:
 231                 f.close()

 233     def _save_header(self):
 234         writer = self.writer
 235         obj = self.obj
 236         index_label = self.index_label
 237         cols = self.cols
 238         has_mi_columns = self.has_mi_columns
 239         header = self.header
 240         encoded_labels: List[str] = []
 241 
 242         has_aliases = isinstance(header, (tuple, list, np.ndarray, ABCIndexClass))
 243         if not (has_aliases or self.header):
 244             return
 245         if has_aliases:
 246             if len(header) != len(cols):
 247                 raise ValueError(
 248                     f"Writing {len(cols)} cols but got {len(header)} aliases"
 249                 )
 250             else:
 251                 write_cols = header
 252         else:
 253             write_cols = cols
 254 
 255         if self.index:
 256             # should write something for index label
 257             if index_label is not False:
 258                 if index_label is None:
 259                     if isinstance(obj.index, ABCMultiIndex):
 260                         index_label = []
 261                         for i, name in enumerate(obj.index.names):
 262                             if name is None:
 263                                 name = ""
 264                             index_label.append(name)
 265                     else:
 266                         index_label = obj.index.name
 267                         if index_label is None:
 268                             index_label = [""]
 269                         else:
 270                             index_label = [index_label]
 271                 elif not isinstance(
 272                     index_label, (list, tuple, np.ndarray, ABCIndexClass)
 273                 ):
 274                     # given a string for a DF with Index
 275                     index_label = [index_label]
 276 
 277                 encoded_labels = list(index_label)
 278             else:
 279                 encoded_labels = []
 280 
 281         if not has_mi_columns or has_aliases:
 282             encoded_labels += list(write_cols)
 283             writer.writerow(encoded_labels)
 284         else:
 285             # write out the mi
 286             columns = obj.columns
 287 
 288             # write out the names for each level, then ALL of the values for
 289             # each level
 290             for i in range(columns.nlevels):
 291 
 292                 # we need at least 1 index column to write our col names
 293                 col_line = []
 294                 if self.index:
 295 
 296                     # name is the first column
 297                     col_line.append(columns.names[i])
 298 
 299                     if isinstance(index_label, list) and len(index_label) > 1:
 300                         col_line.extend([""] * (len(index_label) - 1))
 301 
 302                 col_line.extend(columns._get_level_values(i))
 303 
 304                 writer.writerow(col_line)
 305 
 306             # Write out the index line if it's not empty.
 307             # Otherwise, we will print out an extraneous
 308             # blank line between the mi and the data rows.
 309             if encoded_labels and set(encoded_labels) != {""}:
 310                 encoded_labels.extend([""] * len(columns))
 311                 writer.writerow(encoded_labels)

 313     def _save(self) -> None:
 314         self._save_header()
 315 
 316         nrows = len(self.data_index)
 317 
 318         # write in chunksize bites
 319         chunksize = self.chunksize
 320         chunks = int(nrows / chunksize) + 1
 321 
 322         for i in range(chunks):
 323             start_i = i * chunksize
 324             end_i = min((i + 1) * chunksize, nrows)
 325             if start_i >= end_i:
 326                 break
 327 
 328             self._save_chunk(start_i, end_i)
 329 
 330     def _save_chunk(self, start_i: int, end_i: int) -> None:
 331         data_index = self.data_index
 332 
 333         # create the data for a chunk
 334         slicer = slice(start_i, end_i)
 335 
 336         df = self.obj.iloc[slicer]
 337         blocks = df._mgr.blocks
 338 
 339         for i in range(len(blocks)):
 340             b = blocks[i]
 341             d = b.to_native_types(
 342                 na_rep=self.na_rep,
 343                 float_format=self.float_format,
 344                 decimal=self.decimal,
 345                 date_format=self.date_format,
 346                 quoting=self.quoting,
 347             )
 348 
 349             for col_loc, col in zip(b.mgr_locs, d):
 350                 # self.data is a preallocated list
 351                 self.data[col_loc] = col
 352 
 353         ix = data_index.to_native_types(
 354             slicer=slicer,
 355             na_rep=self.na_rep,
 356             float_format=self.float_format,
 357             decimal=self.decimal,
 358             date_format=self.date_format,
 359             quoting=self.quoting,
 360         )
 361 
 362         libwriters.write_csv_rows(self.data, ix, self.nlevels, self.cols, self.writer)
