  17 def _side_expander(prop_fmt: str):
  18     def expand(self, prop, value: str):
  19         tokens = value.split()
  20         try:
  21             mapping = self.SIDE_SHORTHANDS[len(tokens)]
  22         except KeyError:
  23             warnings.warn(
  24                 f'Could not expand "{prop}: {value}"', CSSWarning,
  25             )
  26             return
  27         for key, idx in zip(self.SIDES, mapping):
  28             yield prop_fmt.format(key), tokens[idx]
  29 
  30     return expand

  34     """
  35     A callable for parsing and resolving CSS to atomic properties.
  36     """
  37 
  38     def __call__(self, declarations_str, inherited=None):
  39         """
  40         The given declarations to atomic properties.
  41 
  42         Parameters
  43         ----------
  44         declarations_str : str
  45             A list of CSS declarations
  46         inherited : dict, optional
  47             Atomic properties indicating the inherited style context in which
  48             declarations_str is to be resolved. ``inherited`` should already
  49             be resolved, i.e. valid output of this method.
  50 
  51         Returns
  52         -------
  53         dict
  54             Atomic CSS 2.2 properties.
  55 
  56         Examples
  57         --------
  58         >>> resolve = CSSResolver()
  59         >>> inherited = {'font-family': 'serif', 'font-weight': 'bold'}
  60         >>> out = resolve('''
  61         ...               border-color: BLUE RED;
  62         ...               font-size: 1em;
  63         ...               font-size: 2em;
  64         ...               font-weight: normal;
  65         ...               font-weight: inherit;
  66         ...               ''', inherited)
  67         >>> sorted(out.items())  # doctest: +NORMALIZE_WHITESPACE
  68         [('border-bottom-color', 'blue'),
  69          ('border-left-color', 'red'),
  70          ('border-right-color', 'red'),
  71          ('border-top-color', 'blue'),
  72          ('font-family', 'serif'),
  73          ('font-size', '24pt'),
  74          ('font-weight', 'bold')]
  75         """
  76         props = dict(self.atomize(self.parse(declarations_str)))
  77         if inherited is None:
  78             inherited = {}
  79 
  80         # 1. resolve inherited, initial
  81         for prop, val in inherited.items():
  82             if prop not in props:
  83                 props[prop] = val
  84 
  85         for prop, val in list(props.items()):
  86             if val == "inherit":
  87                 val = inherited.get(prop, "initial")
  88             if val == "initial":
  89                 val = None
  90 
  91             if val is None:
  92                 # we do not define a complete initial stylesheet
  93                 del props[prop]
  94             else:
  95                 props[prop] = val
  96 
  97         # 2. resolve relative font size
  98         if props.get("font-size"):
  99             if "font-size" in inherited:
 100                 em_pt = inherited["font-size"]
 101                 assert em_pt[-2:] == "pt"
 102                 em_pt = float(em_pt[:-2])
 103             else:
 104                 em_pt = None
 105             props["font-size"] = self.size_to_pt(
 106                 props["font-size"], em_pt, conversions=self.FONT_SIZE_RATIOS
 107             )
 108 
 109             font_size = float(props["font-size"][:-2])
 110         else:
 111             font_size = None
 112 
 113         # 3. TODO: resolve other font-relative units
 114         for side in self.SIDES:
 115             prop = f"border-{side}-width"
 116             if prop in props:
 117                 props[prop] = self.size_to_pt(
 118                     props[prop], em_pt=font_size, conversions=self.BORDER_WIDTH_RATIOS
 119                 )
 120             for prop in [
 121                 f"margin-{side}",
 122                 f"padding-{side}",
 123             ]:
 124                 if prop in props:
 125                     # TODO: support %
 126                     props[prop] = self.size_to_pt(
 127                         props[prop], em_pt=font_size, conversions=self.MARGIN_RATIOS
 128                     )
 129 
 130         return props

 176     def size_to_pt(self, in_val, em_pt=None, conversions=UNIT_RATIOS):
 177         def _error():
 178             warnings.warn(f"Unhandled size: {repr(in_val)}", CSSWarning)
 179             return self.size_to_pt("1!!default", conversions=conversions)
 180 
 181         try:
 182             val, unit = re.match(r"^(\S*?)([a-zA-Z%!].*)", in_val).groups()
 183         except AttributeError:
 184             return _error()
 185         if val == "":
 186             # hack for 'large' etc.
 187             val = 1
 188         else:
 189             try:
 190                 val = float(val)
 191             except ValueError:
 192                 return _error()
 193 
 194         while unit != "pt":
 195             if unit == "em":
 196                 if em_pt is None:
 197                     unit = "rem"
 198                 else:
 199                     val *= em_pt
 200                     unit = "pt"
 201                 continue
 202 
 203             try:
 204                 unit, mul = conversions[unit]
 205             except KeyError:
 206                 return _error()
 207             val *= mul
 208 
 209         val = round(val, 5)
 210         if int(val) == val:
 211             size_fmt = f"{int(val):d}pt"
 212         else:
 213             size_fmt = f"{val:f}pt"
 214         return size_fmt

 216     def atomize(self, declarations):
 217         for prop, value in declarations:
 218             attr = "expand_" + prop.replace("-", "_")
 219             try:
 220                 expand = getattr(self, attr)
 221             except AttributeError:
 222                 yield prop, value
 223             else:
 224                 for prop, value in expand(prop, value):
 225                     yield prop, value

 241     def parse(self, declarations_str: str):
 242         """
 243         Generates (prop, value) pairs from declarations.
 244 
 245         In a future version may generate parsed tokens from tinycss/tinycss2
 246 
 247         Parameters
 248         ----------
 249         declarations_str : str
 250         """
 251         for decl in declarations_str.split(";"):
 252             if not decl.strip():
 253                 continue
 254             prop, sep, val = decl.partition(":")
 255             prop = prop.strip().lower()
 256             # TODO: don't lowercase case sensitive parts of values (strings)
 257             val = val.strip().lower()
 258             if sep:
 259                 yield prop, val
 260             else:
 261                 warnings.warn(
 262                     f"Ill-formatted attribute: expected a colon in {repr(decl)}",
 263                     CSSWarning,
 264                 )
