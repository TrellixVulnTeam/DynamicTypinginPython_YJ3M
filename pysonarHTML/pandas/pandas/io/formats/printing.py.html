  29 def adjoin(space: int, *lists: List[str], **kwargs) -> str:
  30     """
  31     Glues together two sets of strings using the amount of space requested.
  32     The idea is to prettify.
  33 
  34     ----------
  35     space : int
  36         number of spaces for padding
  37     lists : str
  38         list of str which being joined
  39     strlen : callable
  40         function used to calculate the length of each str. Needed for unicode
  41         handling.
  42     justfunc : callable
  43         function used to justify str. Needed for unicode handling.
  44     """
  45     strlen = kwargs.pop("strlen", len)
  46     justfunc = kwargs.pop("justfunc", justify)
  47 
  48     out_lines = []
  49     newLists = []
  50     lengths = [max(map(strlen, x)) + space for x in lists[:-1]]
  51     # not the last one
  52     lengths.append(max(map(len, lists[-1])))
  53     maxLen = max(map(len, lists))
  54     for i, lst in enumerate(lists):
  55         nl = justfunc(lst, lengths[i], mode="left")
  56         nl.extend([" " * lengths[i]] * (maxLen - len(lst)))
  57         newLists.append(nl)
  58     toJoin = zip(*newLists)
  59     for lines in toJoin:
  60         out_lines.append("".join(lines))
  61     return "\n".join(out_lines)

  64 def justify(texts: Iterable[str], max_len: int, mode: str = "right") -> List[str]:
  65     """
  66     Perform ljust, center, rjust against string or list-like
  67     """
  68     if mode == "left":
  69         return [x.ljust(max_len) for x in texts]
  70     elif mode == "center":
  71         return [x.center(max_len) for x in texts]
  72     else:
  73         return [x.rjust(max_len) for x in texts]

  76 # Unicode consolidation
  77 # ---------------------
  78 #
  79 # pprinting utility functions for generating Unicode text or
  80 # bytes(3.x)/str(2.x) representations of objects.
  81 # Try to use these as much as possible rather than rolling your own.
  82 #
  83 # When to use
  84 # -----------
  85 #
  86 # 1) If you're writing code internal to pandas (no I/O directly involved),
  87 #    use pprint_thing().
  88 #
  89 #    It will always return unicode text which can handled by other
  90 #    parts of the package without breakage.
  91 #
  92 # 2) if you need to write something out to file, use
  93 #    pprint_thing_encoded(encoding).
  94 #
  95 #    If no encoding is specified, it defaults to utf-8. Since encoding pure
  96 #    ascii with utf-8 is a no-op you can safely use the default utf-8 if you're
  97 #    working with straight ascii.

 100 def _pprint_seq(
 101     seq: Sequence, _nest_lvl: int = 0, max_seq_items: Optional[int] = None, **kwds
 102 ) -> str:
 103     """
 104     internal. pprinter for iterables. you should probably use pprint_thing()
 105     rather than calling this directly.
 106 
 107     bounds length of printed sequence, depending on options
 108     """
 109     if isinstance(seq, set):
 110         fmt = "{{{body}}}"
 111     else:
 112         fmt = "[{body}]" if hasattr(seq, "__setitem__") else "({body})"
 113 
 114     if max_seq_items is False:
 115         nitems = len(seq)
 116     else:
 117         nitems = max_seq_items or get_option("max_seq_items") or len(seq)
 118 
 119     s = iter(seq)
 120     # handle sets, no slicing
 121     r = [
 122         pprint_thing(next(s), _nest_lvl + 1, max_seq_items=max_seq_items, **kwds)
 123         for i in range(min(nitems, len(seq)))
 124     ]
 125     body = ", ".join(r)
 126 
 127     if nitems < len(seq):
 128         body += ", ..."
 129     elif isinstance(seq, tuple) and len(seq) == 1:
 130         body += ","
 131 
 132     return fmt.format(body=body)

 135 def _pprint_dict(
 136     seq: Mapping, _nest_lvl: int = 0, max_seq_items: Optional[int] = None, **kwds
 137 ) -> str:
 138     """
 139     internal. pprinter for iterables. you should probably use pprint_thing()
 140     rather than calling this directly.
 141     """
 142     fmt = "{{{things}}}"
 143     pairs = []
 144 
 145     pfmt = "{key}: {val}"
 146 
 147     if max_seq_items is False:
 148         nitems = len(seq)
 149     else:
 150         nitems = max_seq_items or get_option("max_seq_items") or len(seq)
 151 
 152     for k, v in list(seq.items())[:nitems]:
 153         pairs.append(
 154             pfmt.format(
 155                 key=pprint_thing(k, _nest_lvl + 1, max_seq_items=max_seq_items, **kwds),
 156                 val=pprint_thing(v, _nest_lvl + 1, max_seq_items=max_seq_items, **kwds),
 157             )
 158         )
 159 
 160     if nitems < len(seq):
 161         return fmt.format(things=", ".join(pairs) + ", ...")
 162     else:
 163         return fmt.format(things=", ".join(pairs))

 166 def pprint_thing(
 167     thing: Any,
 168     _nest_lvl: int = 0,
 169     escape_chars: Optional[EscapeChars] = None,
 170     default_escapes: bool = False,
 171     quote_strings: bool = False,
 172     max_seq_items: Optional[int] = None,
 173 ) -> str:
 174     """
 175     This function is the sanctioned way of converting objects
 176     to a string representation and properly handles nested sequences.
 177 
 178     Parameters
 179     ----------
 180     thing : anything to be formatted
 181     _nest_lvl : internal use only. pprint_thing() is mutually-recursive
 182         with pprint_sequence, this argument is used to keep track of the
 183         current nesting level, and limit it.
 184     escape_chars : list or dict, optional
 185         Characters to escape. If a dict is passed the values are the
 186         replacements
 187     default_escapes : bool, default False
 188         Whether the input escape characters replaces or adds to the defaults
 189     max_seq_items : int or None, default None
 190         Pass through to other pretty printers to limit sequence printing
 191 
 192     Returns
 193     -------
 194     str
 195     """
 196 
 197     def as_escaped_string(
 198         thing: Any, escape_chars: Optional[EscapeChars] = escape_chars
 199     ) -> str:
 200         translate = {"\t": r"\t", "\n": r"\n", "\r": r"\r"}
 201         if isinstance(escape_chars, dict):
 202             if default_escapes:
 203                 translate.update(escape_chars)
 204             else:
 205                 translate = escape_chars
 206             escape_chars = list(escape_chars.keys())
 207         else:
 208             escape_chars = escape_chars or tuple()
 209 
 210         result = str(thing)
 211         for c in escape_chars:
 212             result = result.replace(c, translate[c])
 213         return result
 214 
 215     if hasattr(thing, "__next__"):
 216         return str(thing)
 217     elif isinstance(thing, dict) and _nest_lvl < get_option(
 218         "display.pprint_nest_depth"
 219     ):
 220         result = _pprint_dict(
 221             thing, _nest_lvl, quote_strings=True, max_seq_items=max_seq_items
 222         )
 223     elif is_sequence(thing) and _nest_lvl < get_option("display.pprint_nest_depth"):
 224         result = _pprint_seq(
 225             thing,
 226             _nest_lvl,
 227             escape_chars=escape_chars,
 228             quote_strings=quote_strings,
 229             max_seq_items=max_seq_items,
 230         )
 231     elif isinstance(thing, str) and quote_strings:
 232         result = f"'{as_escaped_string(thing)}'"
 233     else:
 234         result = as_escaped_string(thing)
 235 
 236     return result

 239 def pprint_thing_encoded(
 240     object, encoding: str = "utf-8", errors: str = "replace"
 241 ) -> bytes:
 242     value = pprint_thing(object)  # get unicode representation of object
 243     return value.encode(encoding, errors)

 246 def _enable_data_resource_formatter(enable: bool) -> None:
 247     if "IPython" not in sys.modules:
 248         # definitely not in IPython
 249         return
 250     from IPython import get_ipython
 251 
 252     ip = get_ipython()
 253     if ip is None:
 254         # still not in IPython
 255         return
 256 
 257     formatters = ip.display_formatter.formatters
 258     mimetype = "application/vnd.dataresource+json"
 259 
 260     if enable:
 261         if mimetype not in formatters:
 262             # define tableschema formatter
 263             from IPython.core.formatters import BaseFormatter
 264 
 265             class TableSchemaFormatter(BaseFormatter):
 266                 print_method = "_repr_data_resource_"
 267                 _return_type = (dict,)
 268 
 269             # register it:
 270             formatters[mimetype] = TableSchemaFormatter()
 271         # enable it if it's been disabled:
 272         formatters[mimetype].enabled = True
 273     else:
 274         # unregister tableschema mime-type
 275         if mimetype in formatters:
 276             formatters[mimetype].enabled = False
 277 
 278 
 279 default_pprint = lambda x, max_seq_items=None: pprint_thing(
 280     x, escape_chars=("\t", "\r", "\n"), quote_strings=True, max_seq_items=max_seq_items

 284 def format_object_summary(
 285     obj,
 286     formatter: Callable,
 287     is_justify: bool = True,
 288     name: Optional[str] = None,
 289     indent_for_name: bool = True,
 290     line_break_each_value: bool = False,
 291 ) -> str:
 292     """
 293     Return the formatted obj as a unicode string
 294 
 295     Parameters
 296     ----------
 297     obj : object
 298         must be iterable and support __getitem__
 299     formatter : callable
 300         string formatter for an element
 301     is_justify : boolean
 302         should justify the display
 303     name : name, optional
 304         defaults to the class name of the obj
 305     indent_for_name : bool, default True
 306         Whether subsequent lines should be be indented to
 307         align with the name.
 308     line_break_each_value : bool, default False
 309         If True, inserts a line break for each value of ``obj``.
 310         If False, only break lines when the a line of values gets wider
 311         than the display width.
 312 
 313         .. versionadded:: 0.25.0
 314 
 315     Returns
 316     -------
 317     summary string
 318     """
 319     from pandas.io.formats.console import get_console_size
 320     from pandas.io.formats.format import _get_adjustment
 321 
 322     display_width, _ = get_console_size()
 323     if display_width is None:
 324         display_width = get_option("display.width") or 80
 325     if name is None:
 326         name = type(obj).__name__
 327 
 328     if indent_for_name:
 329         name_len = len(name)
 330         space1 = f'\n{(" " * (name_len + 1))}'
 331         space2 = f'\n{(" " * (name_len + 2))}'
 332     else:
 333         space1 = "\n"
 334         space2 = "\n "  # space for the opening '['
 335 
 336     n = len(obj)
 337     if line_break_each_value:
 338         # If we want to vertically align on each value of obj, we need to
 339         # separate values by a line break and indent the values
 340         sep = ",\n " + " " * len(name)
 341     else:
 342         sep = ","
 343     max_seq_items = get_option("display.max_seq_items") or n
 344 
 345     # are we a truncated display
 346     is_truncated = n > max_seq_items
 347 
 348     # adj can optionally handle unicode eastern asian width
 349     adj = _get_adjustment()
 350 
 351     def _extend_line(
 352         s: str, line: str, value: str, display_width: int, next_line_prefix: str
 353     ) -> Tuple[str, str]:
 354 
 355         if adj.len(line.rstrip()) + adj.len(value.rstrip()) >= display_width:
 356             s += line.rstrip()
 357             line = next_line_prefix
 358         line += value
 359         return s, line
 360 
 361     def best_len(values: List[str]) -> int:
 362         if values:
 363             return max(adj.len(x) for x in values)
 364         else:
 365             return 0
 366 
 367     close = ", "
 368 
 369     if n == 0:
 370         summary = f"[]{close}"
 371     elif n == 1 and not line_break_each_value:
 372         first = formatter(obj[0])
 373         summary = f"[{first}]{close}"
 374     elif n == 2 and not line_break_each_value:
 375         first = formatter(obj[0])
 376         last = formatter(obj[-1])
 377         summary = f"[{first}, {last}]{close}"
 378     else:
 379 
 380         if n > max_seq_items:
 381             n = min(max_seq_items // 2, 10)
 382             head = [formatter(x) for x in obj[:n]]
 383             tail = [formatter(x) for x in obj[-n:]]
 384         else:
 385             head = []
 386             tail = [formatter(x) for x in obj]
 387 
 388         # adjust all values to max length if needed
 389         if is_justify:
 390             if line_break_each_value:
 391                 # Justify each string in the values of head and tail, so the
 392                 # strings will right align when head and tail are stacked
 393                 # vertically.
 394                 head, tail = _justify(head, tail)
 395             elif is_truncated or not (
 396                 len(", ".join(head)) < display_width
 397                 and len(", ".join(tail)) < display_width
 398             ):
 399                 # Each string in head and tail should align with each other
 400                 max_length = max(best_len(head), best_len(tail))
 401                 head = [x.rjust(max_length) for x in head]
 402                 tail = [x.rjust(max_length) for x in tail]
 403             # If we are not truncated and we are only a single
 404             # line, then don't justify
 405 
 406         if line_break_each_value:
 407             # Now head and tail are of type List[Tuple[str]]. Below we
 408             # convert them into List[str], so there will be one string per
 409             # value. Also truncate items horizontally if wider than
 410             # max_space
 411             max_space = display_width - len(space2)
 412             value = tail[0]
 413             for max_items in reversed(range(1, len(value) + 1)):
 414                 pprinted_seq = _pprint_seq(value, max_seq_items=max_items)
 415                 if len(pprinted_seq) < max_space:
 416                     break
 417             head = [_pprint_seq(x, max_seq_items=max_items) for x in head]
 418             tail = [_pprint_seq(x, max_seq_items=max_items) for x in tail]
 419 
 420         summary = ""
 421         line = space2
 422 
 423         for max_items in range(len(head)):
 424             word = head[max_items] + sep + " "
 425             summary, line = _extend_line(summary, line, word, display_width, space2)
 426 
 427         if is_truncated:
 428             # remove trailing space of last line
 429             summary += line.rstrip() + space2 + "..."
 430             line = space2
 431 
 432         for max_items in range(len(tail) - 1):
 433             word = tail[max_items] + sep + " "
 434             summary, line = _extend_line(summary, line, word, display_width, space2)
 435 
 436         # last value: no sep added + 1 space of width used for trailing ','
 437         summary, line = _extend_line(summary, line, tail[-1], display_width - 2, space2)
 438         summary += line
 439 
 440         # right now close is either '' or ', '
 441         # Now we want to include the ']', but not the maybe space.
 442         close = "]" + close.rstrip(" ")
 443         summary += close
 444 
 445         if len(summary) > (display_width) or line_break_each_value:
 446             summary += space1
 447         else:  # one row
 448             summary += " "
 449 
 450         # remove initial space
 451         summary = "[" + summary[len(space2) :]
 452 
 453     return summary

 456 def _justify(
 457     head: List[Sequence[str]], tail: List[Sequence[str]]
 458 ) -> Tuple[List[Tuple[str, ...]], List[Tuple[str, ...]]]:
 459     """
 460     Justify items in head and tail, so they are right-aligned when stacked.
 461 
 462     Parameters
 463     ----------
 464     head : list-like of list-likes of strings
 465     tail : list-like of list-likes of strings
 466 
 467     Returns
 468     -------
 469     tuple of list of tuples of strings
 470         Same as head and tail, but items are right aligned when stacked
 471         vertically.
 472 
 473     Examples
 474     --------
 475     >>> _justify([['a', 'b']], [['abc', 'abcd']])
 476     ([('  a', '   b')], [('abc', 'abcd')])
 477     """
 478     combined = head + tail
 479 
 480     # For each position for the sequences in ``combined``,
 481     # find the length of the largest string.
 482     max_length = [0] * len(combined[0])
 483     for inner_seq in combined:
 484         length = [len(item) for item in inner_seq]
 485         max_length = [max(x, y) for x, y in zip(max_length, length)]
 486 
 487     # justify each item in each list-like in head and tail using max_length
 488     head = [
 489         tuple(x.rjust(max_len) for x, max_len in zip(seq, max_length)) for seq in head
 490     ]
 491     tail = [
 492         tuple(x.rjust(max_len) for x, max_len in zip(seq, max_length)) for seq in tail
 493     ]
 494     # https://github.com/python/mypy/issues/4975
 495     # error: Incompatible return value type (got "Tuple[List[Sequence[str]],
 496     #  List[Sequence[str]]]", expected "Tuple[List[Tuple[str, ...]],
 497     #  List[Tuple[str, ...]]]")
 498     return head, tail  # type: ignore

 501 def format_object_attrs(
 502     obj: Sequence, include_dtype: bool = True
 503 ) -> List[Tuple[str, Union[str, int]]]:
 504     """
 505     Return a list of tuples of the (attr, formatted_value)
 506     for common attrs, including dtype, name, length
 507 
 508     Parameters
 509     ----------
 510     obj : object
 511         must be iterable
 512     include_dtype : bool
 513         If False, dtype won't be in the returned list
 514 
 515     Returns
 516     -------
 517     list of 2-tuple
 518 
 519     """
 520     attrs: List[Tuple[str, Union[str, int]]] = []
 521     if hasattr(obj, "dtype") and include_dtype:
 522         # error: "Sequence[Any]" has no attribute "dtype"
 523         attrs.append(("dtype", f"'{obj.dtype}'"))  # type: ignore
 524     if getattr(obj, "name", None) is not None:
 525         # error: "Sequence[Any]" has no attribute "name"
 526         attrs.append(("name", default_pprint(obj.name)))  # type: ignore
 527     # error: "Sequence[Any]" has no attribute "names"
 528     elif getattr(obj, "names", None) is not None and any(obj.names):  # type: ignore
 529         # error: "Sequence[Any]" has no attribute "names"
 530         attrs.append(("names", default_pprint(obj.names)))  # type: ignore
 531     max_seq_items = get_option("display.max_seq_items") or len(obj)
 532     if len(obj) > max_seq_items:
 533         attrs.append(("length", len(obj)))
 534     return attrs
