  16 def _put_str(s: Union[str, Dtype], space: int) -> str:
  17     """
  18     Make string of specified length, padding to the right if necessary.
  19 
  20     Parameters
  21     ----------
  22     s : Union[str, Dtype]
  23         String to be formatted.
  24     space : int
  25         Length to force string to be of.
  26 
  27     Returns
  28     -------
  29     str
  30         String coerced to given length.
  31 
  32     Examples
  33     --------
  34     >>> pd.io.formats.info._put_str("panda", 6)
  35     'panda '
  36     >>> pd.io.formats.info._put_str("panda", 4)
  37     'pand'
  38     """
  39     return str(s)[:space].ljust(space)

  42 def _get_ids_and_dtypes(data: FrameOrSeries) -> Tuple["Index", "Series"]:
  43     """
  44     Get DataFrame's columns and dtypes.
  45 
  46     Parameters
  47     ----------
  48     data : DataFrame
  49         Object that `info` was called on.
  50 
  51     Returns
  52     -------
  53     ids : Index
  54         DataFrame's columns.
  55     dtypes : Series
  56         Dtype of each of the DataFrame's columns.
  57     """
  58     ids = data.columns
  59     dtypes = data.dtypes
  60     return ids, dtypes

  63 def info(
  64     data: FrameOrSeries,
  65     verbose: Optional[bool] = None,
  66     buf: Optional[IO[str]] = None,
  67     max_cols: Optional[int] = None,
  68     memory_usage: Optional[Union[bool, str]] = None,
  69     null_counts: Optional[bool] = None,
  70 ) -> None:
  71     """
  72     Print a concise summary of a %(klass)s.
  73 
  74     This method prints information about a %(klass)s including
  75     the index dtype%(type_sub)s, non-null values and memory usage.
  76 
  77     Parameters
  78     ----------
  79     data : %(klass)s
  80         %(klass)s to print information about.
  81     verbose : bool, optional
  82         Whether to print the full summary. By default, the setting in
  83         ``pandas.options.display.max_info_columns`` is followed.
  84     buf : writable buffer, defaults to sys.stdout
  85         Where to send the output. By default, the output is printed to
  86         sys.stdout. Pass a writable buffer if you need to further process
  87         the output.
  88     %(max_cols_sub)s
  89     memory_usage : bool, str, optional
  90         Specifies whether total memory usage of the %(klass)s
  91         elements (including the index) should be displayed. By default,
  92         this follows the ``pandas.options.display.memory_usage`` setting.
  93 
  94         True always show memory usage. False never shows memory usage.
  95         A value of 'deep' is equivalent to "True with deep introspection".
  96         Memory usage is shown in human-readable units (base-2
  97         representation). Without deep introspection a memory estimation is
  98         made based in column dtype and number of rows assuming values
  99         consume the same memory amount for corresponding dtypes. With deep
 100         memory introspection, a real memory usage calculation is performed
 101         at the cost of computational resources.
 102     null_counts : bool, optional
 103         Whether to show the non-null counts. By default, this is shown
 104         only if the %(klass)s is smaller than
 105         ``pandas.options.display.max_info_rows`` and
 106         ``pandas.options.display.max_info_columns``. A value of True always
 107         shows the counts, and False never shows the counts.
 108 
 109     Returns
 110     -------
 111     None
 112         This method prints a summary of a %(klass)s and returns None.
 113 
 114     See Also
 115     --------
 116     %(see_also_sub)s
 117 
 118     Examples
 119     --------
 120     %(examples_sub)s
 121     """
 122     if buf is None:  # pragma: no cover
 123         buf = sys.stdout
 124 
 125     lines = []
 126 
 127     lines.append(str(type(data)))
 128     lines.append(data.index._summary())
 129 
 130     ids, dtypes = _get_ids_and_dtypes(data)
 131     col_count = len(ids)
 132 
 133     if col_count == 0:
 134         lines.append(f"Empty {type(data).__name__}")
 135         fmt.buffer_put_lines(buf, lines)
 136         return
 137 
 138     # hack
 139     if max_cols is None:
 140         max_cols = get_option("display.max_info_columns", col_count + 1)
 141 
 142     max_rows = get_option("display.max_info_rows", len(data) + 1)
 143 
 144     if null_counts is None:
 145         show_counts = (col_count <= max_cols) and (len(data) < max_rows)
 146     else:
 147         show_counts = null_counts
 148     exceeds_info_cols = col_count > max_cols
 149 
 150     def _verbose_repr():
 151         lines.append(f"Data columns (total {col_count} columns):")
 152 
 153         id_head = " # "
 154         column_head = "Column"
 155         col_space = 2
 156 
 157         max_col = max(len(pprint_thing(k)) for k in ids)
 158         len_column = len(pprint_thing(column_head))
 159         space = max(max_col, len_column) + col_space
 160 
 161         max_id = len(pprint_thing(col_count))
 162         len_id = len(pprint_thing(id_head))
 163         space_num = max(max_id, len_id) + col_space
 164 
 165         header = _put_str(id_head, space_num) + _put_str(column_head, space)
 166         if show_counts:
 167             counts = data.count()
 168             if col_count != len(counts):  # pragma: no cover
 169                 raise AssertionError(
 170                     f"Columns must equal counts ({col_count} != {len(counts)})"
 171                 )
 172             count_header = "Non-Null Count"
 173             len_count = len(count_header)
 174             non_null = " non-null"
 175             max_count = max(len(pprint_thing(k)) for k in counts) + len(non_null)
 176             space_count = max(len_count, max_count) + col_space
 177             count_temp = "{count}" + non_null
 178         else:
 179             count_header = ""
 180             space_count = len(count_header)
 181             len_count = space_count
 182             count_temp = "{count}"
 183 
 184         dtype_header = "Dtype"
 185         len_dtype = len(dtype_header)
 186         max_dtypes = max(len(pprint_thing(k)) for k in dtypes)
 187         space_dtype = max(len_dtype, max_dtypes)
 188         header += _put_str(count_header, space_count) + _put_str(
 189             dtype_header, space_dtype
 190         )
 191 
 192         lines.append(header)
 193         lines.append(
 194             _put_str("-" * len_id, space_num)
 195             + _put_str("-" * len_column, space)
 196             + _put_str("-" * len_count, space_count)
 197             + _put_str("-" * len_dtype, space_dtype)
 198         )
 199 
 200         for i, col in enumerate(ids):
 201             dtype = dtypes[i]
 202             col = pprint_thing(col)
 203 
 204             line_no = _put_str(f" {i}", space_num)
 205             count = ""
 206             if show_counts:
 207                 count = counts[i]
 208 
 209             lines.append(
 210                 line_no
 211                 + _put_str(col, space)
 212                 + _put_str(count_temp.format(count=count), space_count)
 213                 + _put_str(dtype, space_dtype)
 214             )
 215 
 216     def _non_verbose_repr():
 217         lines.append(ids._summary(name="Columns"))
 218 
 219     def _sizeof_fmt(num, size_qualifier):
 220         # returns size in human readable format
 221         for x in ["bytes", "KB", "MB", "GB", "TB"]:
 222             if num < 1024.0:
 223                 return f"{num:3.1f}{size_qualifier} {x}"
 224             num /= 1024.0
 225         return f"{num:3.1f}{size_qualifier} PB"
 226 
 227     if verbose:
 228         _verbose_repr()
 229     elif verbose is False:  # specifically set to False, not necessarily None
 230         _non_verbose_repr()
 231     else:
 232         if exceeds_info_cols:
 233             _non_verbose_repr()
 234         else:
 235             _verbose_repr()
 236 
 237     # groupby dtype.name to collect e.g. Categorical columns
 238     counts = dtypes.value_counts().groupby(lambda x: x.name).sum()
 239     collected_dtypes = [f"{k[0]}({k[1]:d})" for k in sorted(counts.items())]
 240     lines.append(f"dtypes: {', '.join(collected_dtypes)}")
 241 
 242     if memory_usage is None:
 243         memory_usage = get_option("display.memory_usage")
 244     if memory_usage:
 245         # append memory usage of df to display
 246         size_qualifier = ""
 247         if memory_usage == "deep":
 248             deep = True
 249         else:
 250             # size_qualifier is just a best effort; not guaranteed to catch
 251             # all cases (e.g., it misses categorical data even with object
 252             # categories)
 253             deep = False
 254             if "object" in counts or data.index._is_memory_usage_qualified():
 255                 size_qualifier = "+"
 256         mem_usage = data.memory_usage(index=True, deep=deep).sum()
 257         lines.append(f"memory usage: {_sizeof_fmt(mem_usage, size_qualifier)}\n")
 258     fmt.buffer_put_lines(buf, lines)
