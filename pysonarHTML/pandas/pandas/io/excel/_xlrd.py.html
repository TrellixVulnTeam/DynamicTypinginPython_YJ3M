  49     def get_sheet_data(self, sheet, convert_float):
  50         from xlrd import (
  51             xldate,
  52             XL_CELL_DATE,
  53             XL_CELL_ERROR,
  54             XL_CELL_BOOLEAN,
  55             XL_CELL_NUMBER,
  56         )
  57 
  58         epoch1904 = self.book.datemode
  59 
  60         def _parse_cell(cell_contents, cell_typ):
  61             """
  62             converts the contents of the cell into a pandas appropriate object
  63             """
  64             if cell_typ == XL_CELL_DATE:
  65 
  66                 # Use the newer xlrd datetime handling.
  67                 try:
  68                     cell_contents = xldate.xldate_as_datetime(cell_contents, epoch1904)
  69                 except OverflowError:
  70                     return cell_contents
  71 
  72                 # Excel doesn't distinguish between dates and time,
  73                 # so we treat dates on the epoch as times only.
  74                 # Also, Excel supports 1900 and 1904 epochs.
  75                 year = (cell_contents.timetuple())[0:3]
  76                 if (not epoch1904 and year == (1899, 12, 31)) or (
  77                     epoch1904 and year == (1904, 1, 1)
  78                 ):
  79                     cell_contents = time(
  80                         cell_contents.hour,
  81                         cell_contents.minute,
  82                         cell_contents.second,
  83                         cell_contents.microsecond,
  84                     )
  85 
  86             elif cell_typ == XL_CELL_ERROR:
  87                 cell_contents = np.nan
  88             elif cell_typ == XL_CELL_BOOLEAN:
  89                 cell_contents = bool(cell_contents)
  90             elif convert_float and cell_typ == XL_CELL_NUMBER:
  91                 # GH5394 - Excel 'numbers' are always floats
  92                 # it's a minimal perf hit and less surprising
  93                 val = int(cell_contents)
  94                 if val == cell_contents:
  95                     cell_contents = val
  96             return cell_contents
  97 
  98         data = []
  99 
 100         for i in range(sheet.nrows):
 101             row = [
 102                 _parse_cell(value, typ)
 103                 for value, typ in zip(sheet.row_values(i), sheet.row_types(i))
 104             ]
 105             data.append(row)
 106 
 107         return data
