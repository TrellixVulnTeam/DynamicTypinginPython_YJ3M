  51     def _convert_cell(self, cell, convert_float: bool) -> Scalar:
  52         # TODO: there is no way to distinguish between floats and datetimes in pyxlsb
  53         # This means that there is no way to read datetime types from an xlsb file yet
  54         if cell.v is None:
  55             return ""  # Prevents non-named columns from not showing up as Unnamed: i
  56         if isinstance(cell.v, float) and convert_float:
  57             val = int(cell.v)
  58             if val == cell.v:
  59                 return val
  60             else:
  61                 return float(cell.v)
  62 
  63         return cell.v
  64 
  65     def get_sheet_data(self, sheet, convert_float: bool) -> List[List[Scalar]]:
  66         return [
  67             [self._convert_cell(c, convert_float) for c in r]
  68             for r in sheet.rows(sparse=False)
  69         ]
