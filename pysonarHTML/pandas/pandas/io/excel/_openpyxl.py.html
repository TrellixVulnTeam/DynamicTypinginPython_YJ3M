  47         """

  50         Parameters

  58             for nk, nv in value.items():

  61                         xls_style.borders.__getattribute__(nk).__setattr__(

  70     @classmethod
  71     def _convert_to_style_kwargs(cls, style_dict):
  72         """
  73         Convert a style_dict to a set of kwargs suitable for initializing
  74         or updating-on-copy an openpyxl v2 style object.
  75 
  76         Parameters
  77         ----------
  78         style_dict : dict
  79             A dict with zero or more of the following keys (or their synonyms).
  80                 'font'
  81                 'fill'
  82                 'border' ('borders')
  83                 'alignment'
  84                 'number_format'
  85                 'protection'
  86 
  87         Returns
  88         -------
  89         style_kwargs : dict
  90             A dict with the same, normalized keys as ``style_dict`` but each
  91             value has been replaced with a native openpyxl style object of the
  92             appropriate class.
  93         """
  94         _style_key_map = {"borders": "border"}
  95 
  96         style_kwargs = {}
  97         for k, v in style_dict.items():
  98             if k in _style_key_map:
  99                 k = _style_key_map[k]
 100             _conv_to_x = getattr(cls, f"_convert_to_{k}", lambda x: None)
 101             new_v = _conv_to_x(v)
 102             if new_v:
 103                 style_kwargs[k] = new_v
 104 
 105         return style_kwargs

 141         Parameters

 203     @classmethod
 204     def _convert_to_fill(cls, fill_dict):
 205         """
 206         Convert ``fill_dict`` to an openpyxl v2 Fill object.
 207 
 208         Parameters
 209         ----------
 210         fill_dict : dict
 211             A dict with one or more of the following keys (or their synonyms),
 212                 'fill_type' ('patternType', 'patterntype')
 213                 'start_color' ('fgColor', 'fgcolor')
 214                 'end_color' ('bgColor', 'bgcolor')
 215             or one or more of the following keys (or their synonyms).
 216                 'type' ('fill_type')
 217                 'degree'
 218                 'left'
 219                 'right'
 220                 'top'
 221                 'bottom'
 222                 'stop'
 223 
 224         Returns
 225         -------
 226         fill : openpyxl.styles.Fill
 227         """
 228         from openpyxl.styles import PatternFill, GradientFill
 229 
 230         _pattern_fill_key_map = {
 231             "patternType": "fill_type",
 232             "patterntype": "fill_type",
 233             "fgColor": "start_color",
 234             "fgcolor": "start_color",
 235             "bgColor": "end_color",
 236             "bgcolor": "end_color",
 237         }
 238 
 239         _gradient_fill_key_map = {"fill_type": "type"}
 240 
 241         pfill_kwargs = {}
 242         gfill_kwargs = {}
 243         for k, v in fill_dict.items():
 244             pk = gk = None
 245             if k in _pattern_fill_key_map:
 246                 pk = _pattern_fill_key_map[k]
 247             if k in _gradient_fill_key_map:
 248                 gk = _gradient_fill_key_map[k]
 249             if pk in ["start_color", "end_color"]:
 250                 v = cls._convert_to_color(v)
 251             if gk == "stop":
 252                 v = cls._convert_to_stop(v)
 253             if pk:
 254                 pfill_kwargs[pk] = v
 255             elif gk:
 256                 gfill_kwargs[gk] = v
 257             else:
 258                 pfill_kwargs[k] = v
 259                 gfill_kwargs[k] = v
 260 
 261         try:
 262             return PatternFill(**pfill_kwargs)
 263         except TypeError:
 264             return GradientFill(**gfill_kwargs)

 401     def write_cells(
 402         self, cells, sheet_name=None, startrow=0, startcol=0, freeze_panes=None
 403     ):
 404         # Write the frame cells using openpyxl.
 405         sheet_name = self._get_sheet_name(sheet_name)
 406 
 407         _style_cache = {}
 408 
 409         if sheet_name in self.sheets:
 410             wks = self.sheets[sheet_name]
 411         else:
 412             wks = self.book.create_sheet()
 413             wks.title = sheet_name
 414             self.sheets[sheet_name] = wks
 415 
 416         if _validate_freeze_panes(freeze_panes):
 417             wks.freeze_panes = wks.cell(
 418                 row=freeze_panes[0] + 1, column=freeze_panes[1] + 1
 419             )
 420 
 421         for cell in cells:
 422             xcell = wks.cell(
 423                 row=startrow + cell.row + 1, column=startcol + cell.col + 1
 424             )
 425             xcell.value, fmt = self._value_with_fmt(cell.val)
 426             if fmt:
 427                 xcell.number_format = fmt
 428 
 429             style_kwargs = {}
 430             if cell.style:
 431                 key = str(cell.style)
 432                 style_kwargs = _style_cache.get(key)
 433                 if style_kwargs is None:
 434                     style_kwargs = self._convert_to_style_kwargs(cell.style)
 435                     _style_cache[key] = style_kwargs
 436 
 437             if style_kwargs:
 438                 for k, v in style_kwargs.items():
 439                     setattr(xcell, k, v)
 440 
 441             if cell.mergestart is not None and cell.mergeend is not None:
 442 
 443                 wks.merge_cells(
 444                     start_row=startrow + cell.row + 1,
 445                     start_column=startcol + cell.col + 1,
 446                     end_column=startcol + cell.mergeend + 1,
 447                     end_row=startrow + cell.mergestart + 1,
 448                 )
 449 
 450                 # When cells are merged only the top-left cell is preserved
 451                 # The behaviour of the other cells in a merged range is
 452                 # undefined
 453                 if style_kwargs:
 454                     first_row = startrow + cell.row + 1
 455                     last_row = startrow + cell.mergestart + 1
 456                     first_col = startcol + cell.col + 1
 457                     last_col = startcol + cell.mergeend + 1
 458 
 459                     for row in range(first_row, last_row + 1):
 460                         for col in range(first_col, last_col + 1):
 461                             if row == first_row and col == first_col:
 462                                 # Ignore first cell. It is already handled.
 463                                 continue
 464                             xcell = wks.cell(column=col, row=row)
 465                             for k, v in style_kwargs.items():
 466                                 setattr(xcell, k, v)

 510     def _convert_cell(self, cell, convert_float: bool) -> Scalar:
 511 
 512         # TODO: replace with openpyxl constants
 513         if cell.is_date:
 514             return cell.value
 515         elif cell.data_type == "e":
 516             return np.nan
 517         elif cell.data_type == "b":
 518             return bool(cell.value)
 519         elif cell.value is None:
 520             return ""  # compat with xlrd
 521         elif cell.data_type == "n":
 522             # GH5394
 523             if convert_float:
 524                 val = int(cell.value)
 525                 if val == cell.value:
 526                     return val
 527             else:
 528                 return float(cell.value)
 529 
 530         return cell.value
 531 
 532     def get_sheet_data(self, sheet, convert_float: bool) -> List[List[Scalar]]:
 533         data: List[List[Scalar]] = []
 534         for row in sheet.rows:
 535             data.append([self._convert_cell(cell, convert_float) for cell in row])
 536 
 537         return data
