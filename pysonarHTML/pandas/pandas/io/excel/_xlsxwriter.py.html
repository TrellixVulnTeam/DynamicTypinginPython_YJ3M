  78     @classmethod
  79     def convert(cls, style_dict, num_format_str=None):
  80         """
  81         converts a style_dict to an xlsxwriter format dict
  82 
  83         Parameters
  84         ----------
  85         style_dict : style dictionary to convert
  86         num_format_str : optional number format string
  87         """
  88         # Create a XlsxWriter format object.
  89         props = {}
  90 
  91         if num_format_str is not None:
  92             props["num_format"] = num_format_str
  93 
  94         if style_dict is None:
  95             return props
  96 
  97         if "borders" in style_dict:
  98             style_dict = style_dict.copy()
  99             style_dict["border"] = style_dict.pop("borders")
 100 
 101         for style_group_key, style_group in style_dict.items():
 102             for src, dst in cls.STYLE_MAPPING.get(style_group_key, []):
 103                 # src is a sequence of keys into a nested dict
 104                 # dst is a flat key
 105                 if dst in props:
 106                     continue
 107                 v = style_group
 108                 for k in src:
 109                     try:
 110                         v = v[k]
 111                     except (KeyError, TypeError):
 112                         break
 113                 else:
 114                     props[dst] = v
 115 
 116         if isinstance(props.get("pattern"), str):
 117             # TODO: support other fill patterns
 118             props["pattern"] = 0 if props["pattern"] == "none" else 1
 119 
 120         for k in ["border", "top", "right", "bottom", "left"]:
 121             if isinstance(props.get(k), str):
 122                 try:
 123                     props[k] = [
 124                         "none",
 125                         "thin",
 126                         "medium",
 127                         "dashed",
 128                         "dotted",
 129                         "thick",
 130                         "double",
 131                         "hair",
 132                         "mediumDashed",
 133                         "dashDot",
 134                         "mediumDashDot",
 135                         "dashDotDot",
 136                         "mediumDashDotDot",
 137                         "slantDashDot",
 138                     ].index(props[k])
 139                 except ValueError:
 140                     props[k] = 2
 141 
 142         if isinstance(props.get("font_script"), str):
 143             props["font_script"] = ["baseline", "superscript", "subscript"].index(
 144                 props["font_script"]
 145             )
 146 
 147         if isinstance(props.get("underline"), str):
 148             props["underline"] = {
 149                 "none": 0,
 150                 "single": 1,
 151                 "double": 2,
 152                 "singleAccounting": 33,
 153                 "doubleAccounting": 34,
 154             }[props["underline"]]
 155 
 156         return props

 163     def __init__(
 164         self,
 165         path,
 166         engine=None,
 167         date_format=None,
 168         datetime_format=None,
 169         mode="w",
 170         **engine_kwargs,
 171     ):
 172         # Use the xlsxwriter module as the Excel writer.
 173         import xlsxwriter
 174 
 175         if mode == "a":
 176             raise ValueError("Append mode is not supported with xlsxwriter!")
 177 
 178         super().__init__(
 179             path,
 180             engine=engine,
 181             date_format=date_format,
 182             datetime_format=datetime_format,
 183             mode=mode,
 184             **engine_kwargs,
 185         )
 186 
 187         self.book = xlsxwriter.Workbook(path, **engine_kwargs)
 188 
 189     def save(self):
 190         """
 191         Save workbook to disk.
 192         """
 193         return self.book.close()

 195     def write_cells(
 196         self, cells, sheet_name=None, startrow=0, startcol=0, freeze_panes=None
 197     ):
 198         # Write the frame cells using xlsxwriter.
 199         sheet_name = self._get_sheet_name(sheet_name)
 200 
 201         if sheet_name in self.sheets:
 202             wks = self.sheets[sheet_name]
 203         else:
 204             wks = self.book.add_worksheet(sheet_name)
 205             self.sheets[sheet_name] = wks
 206 
 207         style_dict = {"null": None}
 208 
 209         if _validate_freeze_panes(freeze_panes):
 210             wks.freeze_panes(*(freeze_panes))
 211 
 212         for cell in cells:
 213             val, fmt = self._value_with_fmt(cell.val)
 214 
 215             stylekey = json.dumps(cell.style)
 216             if fmt:
 217                 stylekey += fmt
 218 
 219             if stylekey in style_dict:
 220                 style = style_dict[stylekey]
 221             else:
 222                 style = self.book.add_format(_XlsxStyler.convert(cell.style, fmt))
 223                 style_dict[stylekey] = style
 224 
 225             if cell.mergestart is not None and cell.mergeend is not None:
 226                 wks.merge_range(
 227                     startrow + cell.row,
 228                     startcol + cell.col,
 229                     startrow + cell.mergestart,
 230                     startcol + cell.mergeend,
 231                     val,
 232                     style,
 233                 )
 234             else:
 235                 wks.write(startrow + cell.row, startcol + cell.col, val, style)
