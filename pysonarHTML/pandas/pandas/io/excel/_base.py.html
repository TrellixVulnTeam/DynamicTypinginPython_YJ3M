  34     """
  35 Read an Excel file into a pandas DataFrame.
  36 
  37 Supports `xls`, `xlsx`, `xlsm`, `xlsb`, `odf`, `ods` and `odt` file extensions
  38 read from a local filesystem or URL. Supports an option to read
  39 a single sheet or a list of sheets.
  40 
  41 Parameters
  42 ----------
  43 io : str, bytes, ExcelFile, xlrd.Book, path object, or file-like object
  44     Any valid string path is acceptable. The string could be a URL. Valid
  45     URL schemes include http, ftp, s3, and file. For file URLs, a host is
  46     expected. A local file could be: ``file://localhost/path/to/table.xlsx``.
  47 
  48     If you want to pass in a path object, pandas accepts any ``os.PathLike``.
  49 
  50     By file-like object, we refer to objects with a ``read()`` method,
  51     such as a file handler (e.g. via builtin ``open`` function)
  52     or ``StringIO``.
  53 sheet_name : str, int, list, or None, default 0
  54     Strings are used for sheet names. Integers are used in zero-indexed
  55     sheet positions. Lists of strings/integers are used to request
  56     multiple sheets. Specify None to get all sheets.
  57 
  58     Available cases:
  59 
  60     * Defaults to ``0``: 1st sheet as a `DataFrame`
  61     * ``1``: 2nd sheet as a `DataFrame`
  62     * ``"Sheet1"``: Load sheet with name "Sheet1"
  63     * ``[0, 1, "Sheet5"]``: Load first, second and sheet named "Sheet5"
  64       as a dict of `DataFrame`
  65     * None: All sheets.
  66 
  67 header : int, list of int, default 0
  68     Row (0-indexed) to use for the column labels of the parsed
  69     DataFrame. If a list of integers is passed those row positions will
  70     be combined into a ``MultiIndex``. Use None if there is no header.
  71 names : array-like, default None
  72     List of column names to use. If file contains no header row,
  73     then you should explicitly pass header=None.
  74 index_col : int, list of int, default None
  75     Column (0-indexed) to use as the row labels of the DataFrame.
  76     Pass None if there is no such column.  If a list is passed,
  77     those columns will be combined into a ``MultiIndex``.  If a
  78     subset of data is selected with ``usecols``, index_col
  79     is based on the subset.
  80 usecols : int, str, list-like, or callable default None
  81     * If None, then parse all columns.
  82     * If str, then indicates comma separated list of Excel column letters
  83       and column ranges (e.g. "A:E" or "A,C,E:F"). Ranges are inclusive of
  84       both sides.
  85     * If list of int, then indicates list of column numbers to be parsed.
  86     * If list of string, then indicates list of column names to be parsed.
  87 
  88       .. versionadded:: 0.24.0
  89 
  90     * If callable, then evaluate each column name against it and parse the
  91       column if the callable returns ``True``.
  92 
  93     Returns a subset of the columns according to behavior above.
  94 
  95       .. versionadded:: 0.24.0
  96 
  97 squeeze : bool, default False
  98     If the parsed data only contains one column then return a Series.
  99 dtype : Type name or dict of column -> type, default None
 100     Data type for data or columns. E.g. {'a': np.float64, 'b': np.int32}
 101     Use `object` to preserve data as stored in Excel and not interpret dtype.
 102     If converters are specified, they will be applied INSTEAD
 103     of dtype conversion.
 104 engine : str, default None
 105     If io is not a buffer or path, this must be set to identify io.
 106     Supported engines: "xlrd", "openpyxl", "odf", "pyxlsb", default "xlrd".
 107     Engine compatibility :
 108     - "xlrd" supports most old/new Excel file formats.
 109     - "openpyxl" supports newer Excel file formats.
 110     - "odf" supports OpenDocument file formats (.odf, .ods, .odt).
 111     - "pyxlsb" supports Binary Excel files.
 112 converters : dict, default None
 113     Dict of functions for converting values in certain columns. Keys can
 114     either be integers or column labels, values are functions that take one
 115     input argument, the Excel cell content, and return the transformed
 116     content.
 117 true_values : list, default None
 118     Values to consider as True.
 119 false_values : list, default None
 120     Values to consider as False.
 121 skiprows : list-like
 122     Rows to skip at the beginning (0-indexed).
 123 nrows : int, default None
 124     Number of rows to parse.
 125 
 126     .. versionadded:: 0.23.0
 127 
 128 na_values : scalar, str, list-like, or dict, default None
 129     Additional strings to recognize as NA/NaN. If dict passed, specific
 130     per-column NA values. By default the following values are interpreted
 131     as NaN: '"""
 132     + fill("', '".join(sorted(STR_NA_VALUES)), 70, subsequent_indent="    ")
 133     + """'.
 134 keep_default_na : bool, default True
 135     Whether or not to include the default NaN values when parsing the data.
 136     Depending on whether `na_values` is passed in, the behavior is as follows:
 137 
 138     * If `keep_default_na` is True, and `na_values` are specified, `na_values`
 139       is appended to the default NaN values used for parsing.
 140     * If `keep_default_na` is True, and `na_values` are not specified, only
 141       the default NaN values are used for parsing.
 142     * If `keep_default_na` is False, and `na_values` are specified, only
 143       the NaN values specified `na_values` are used for parsing.
 144     * If `keep_default_na` is False, and `na_values` are not specified, no
 145       strings will be parsed as NaN.
 146 
 147     Note that if `na_filter` is passed in as False, the `keep_default_na` and
 148     `na_values` parameters will be ignored.
 149 na_filter : bool, default True
 150     Detect missing value markers (empty strings and the value of na_values). In
 151     data without any NAs, passing na_filter=False can improve the performance
 152     of reading a large file.
 153 verbose : bool, default False
 154     Indicate number of NA values placed in non-numeric columns.
 155 parse_dates : bool, list-like, or dict, default False
 156     The behavior is as follows:
 157 
 158     * bool. If True -> try parsing the index.
 159     * list of int or names. e.g. If [1, 2, 3] -> try parsing columns 1, 2, 3
 160       each as a separate date column.
 161     * list of lists. e.g.  If [[1, 3]] -> combine columns 1 and 3 and parse as
 162       a single date column.
 163     * dict, e.g. {'foo' : [1, 3]} -> parse columns 1, 3 as date and call
 164       result 'foo'
 165 
 166     If a column or index contains an unparseable date, the entire column or
 167     index will be returned unaltered as an object data type. If you don`t want to
 168     parse some cells as date just change their type in Excel to "Text".
 169     For non-standard datetime parsing, use ``pd.to_datetime`` after ``pd.read_excel``.
 170 
 171     Note: A fast-path exists for iso8601-formatted dates.
 172 date_parser : function, optional
 173     Function to use for converting a sequence of string columns to an array of
 174     datetime instances. The default uses ``dateutil.parser.parser`` to do the
 175     conversion. Pandas will try to call `date_parser` in three different ways,
 176     advancing to the next if an exception occurs: 1) Pass one or more arrays
 177     (as defined by `parse_dates`) as arguments; 2) concatenate (row-wise) the
 178     string values from the columns defined by `parse_dates` into a single array
 179     and pass that; and 3) call `date_parser` once for each row using one or
 180     more strings (corresponding to the columns defined by `parse_dates`) as
 181     arguments.
 182 thousands : str, default None
 183     Thousands separator for parsing string columns to numeric.  Note that
 184     this parameter is only necessary for columns stored as TEXT in Excel,
 185     any numeric columns will automatically be parsed, regardless of display
 186     format.
 187 comment : str, default None
 188     Comments out remainder of line. Pass a character or characters to this
 189     argument to indicate comments in the input file. Any data between the
 190     comment string and the end of the current line is ignored.
 191 skipfooter : int, default 0
 192     Rows at the end to skip (0-indexed).
 193 convert_float : bool, default True
 194     Convert integral floats to int (i.e., 1.0 --> 1). If False, all numeric
 195     data will be read in as floats: Excel stores all numbers as floats
 196     internally.
 197 mangle_dupe_cols : bool, default True
 198     Duplicate columns will be specified as 'X', 'X.1', ...'X.N', rather than
 199     'X'...'X'. Passing in False will cause data to be overwritten if there
 200     are duplicate names in the columns.
 201 
 202 Returns
 203 -------
 204 DataFrame or dict of DataFrames
 205     DataFrame from the passed in Excel file. See notes in sheet_name
 206     argument for more information on when a dict of DataFrames is returned.
 207

 337 class _BaseExcelReader(metaclass=abc.ABCMeta):
 338     def __init__(self, filepath_or_buffer):
 339         # If filepath_or_buffer is a url, load the data into a BytesIO
 340         if is_url(filepath_or_buffer):
 341             filepath_or_buffer = BytesIO(urlopen(filepath_or_buffer).read())
 342         elif not isinstance(filepath_or_buffer, (ExcelFile, self._workbook_class)):
 343             filepath_or_buffer, _, _, _ = get_filepath_or_buffer(filepath_or_buffer)
 344 
 345         if isinstance(filepath_or_buffer, self._workbook_class):
 346             self.book = filepath_or_buffer
 347         elif hasattr(filepath_or_buffer, "read"):
 348             # N.B. xlrd.Book has a read attribute too
 349             filepath_or_buffer.seek(0)
 350             self.book = self.load_workbook(filepath_or_buffer)
 351         elif isinstance(filepath_or_buffer, str):
 352             self.book = self.load_workbook(filepath_or_buffer)
 353         elif isinstance(filepath_or_buffer, bytes):
 354             self.book = self.load_workbook(BytesIO(filepath_or_buffer))
 355         else:
 356             raise ValueError(
 357                 "Must explicitly set engine if not passing in buffer or path for io."
 358             )

 389     def parse(
 390         self,
 391         sheet_name=0,
 392         header=0,
 393         names=None,
 394         index_col=None,
 395         usecols=None,
 396         squeeze=False,
 397         dtype=None,
 398         true_values=None,
 399         false_values=None,
 400         skiprows=None,
 401         nrows=None,
 402         na_values=None,
 403         verbose=False,
 404         parse_dates=False,
 405         date_parser=None,
 406         thousands=None,
 407         comment=None,
 408         skipfooter=0,
 409         convert_float=True,
 410         mangle_dupe_cols=True,
 411         **kwds,
 412     ):
 413 
 414         validate_header_arg(header)
 415 
 416         ret_dict = False
 417 
 418         # Keep sheetname to maintain backwards compatibility.
 419         if isinstance(sheet_name, list):
 420             sheets = sheet_name
 421             ret_dict = True
 422         elif sheet_name is None:
 423             sheets = self.sheet_names
 424             ret_dict = True
 425         else:
 426             sheets = [sheet_name]
 427 
 428         # handle same-type duplicates.
 429         sheets = list(dict.fromkeys(sheets).keys())
 430 
 431         output = {}
 432 
 433         for asheetname in sheets:
 434             if verbose:
 435                 print(f"Reading sheet {asheetname}")
 436 
 437             if isinstance(asheetname, str):
 438                 sheet = self.get_sheet_by_name(asheetname)
 439             else:  # assume an integer if not a string
 440                 sheet = self.get_sheet_by_index(asheetname)
 441 
 442             data = self.get_sheet_data(sheet, convert_float)
 443             usecols = _maybe_convert_usecols(usecols)
 444 
 445             if not data:
 446                 output[asheetname] = DataFrame()
 447                 continue
 448 
 449             if is_list_like(header) and len(header) == 1:
 450                 header = header[0]
 451 
 452             # forward fill and pull out names for MultiIndex column
 453             header_names = None
 454             if header is not None and is_list_like(header):
 455                 header_names = []
 456                 control_row = [True] * len(data[0])
 457 
 458                 for row in header:
 459                     if is_integer(skiprows):
 460                         row += skiprows
 461 
 462                     data[row], control_row = _fill_mi_header(data[row], control_row)
 463 
 464                     if index_col is not None:
 465                         header_name, _ = _pop_header_name(data[row], index_col)
 466                         header_names.append(header_name)
 467 
 468             if is_list_like(index_col):
 469                 # Forward fill values for MultiIndex index.
 470                 if not is_list_like(header):
 471                     offset = 1 + header
 472                 else:
 473                     offset = 1 + max(header)
 474 
 475                 # Check if we have an empty dataset
 476                 # before trying to collect data.
 477                 if offset < len(data):
 478                     for col in index_col:
 479                         last = data[offset][col]
 480 
 481                         for row in range(offset + 1, len(data)):
 482                             if data[row][col] == "" or data[row][col] is None:
 483                                 data[row][col] = last
 484                             else:
 485                                 last = data[row][col]
 486 
 487             has_index_names = is_list_like(header) and len(header) > 1
 488 
 489             # GH 12292 : error when read one empty column from excel file
 490             try:
 491                 parser = TextParser(
 492                     data,
 493                     names=names,
 494                     header=header,
 495                     index_col=index_col,
 496                     has_index_names=has_index_names,
 497                     squeeze=squeeze,
 498                     dtype=dtype,
 499                     true_values=true_values,
 500                     false_values=false_values,
 501                     skiprows=skiprows,
 502                     nrows=nrows,
 503                     na_values=na_values,
 504                     parse_dates=parse_dates,
 505                     date_parser=date_parser,
 506                     thousands=thousands,
 507                     comment=comment,
 508                     skipfooter=skipfooter,
 509                     usecols=usecols,
 510                     mangle_dupe_cols=mangle_dupe_cols,
 511                     **kwds,
 512                 )
 513 
 514                 output[asheetname] = parser.read(nrows=nrows)
 515 
 516                 if not squeeze or isinstance(output[asheetname], DataFrame):
 517                     if header_names:
 518                         output[asheetname].columns = output[
 519                             asheetname
 520                         ].columns.set_names(header_names)
 521 
 522             except EmptyDataError:
 523                 # No Data, return an empty DataFrame
 524                 output[asheetname] = DataFrame()
 525 
 526         if ret_dict:
 527             return output
 528         else:
 529             return output[asheetname]

 532 class ExcelWriter(metaclass=abc.ABCMeta):
 533     """
 534     Class for writing DataFrame objects into excel sheets.
 535 
 536     Default is to use xlwt for xls, openpyxl for xlsx.
 537     See DataFrame.to_excel for typical usage.
 538 
 539     Parameters
 540     ----------
 541     path : str
 542         Path to xls or xlsx file.
 543     engine : str (optional)
 544         Engine to use for writing. If None, defaults to
 545         ``io.excel.<extension>.writer``.  NOTE: can only be passed as a keyword
 546         argument.
 547     date_format : str, default None
 548         Format string for dates written into Excel files (e.g. 'YYYY-MM-DD').
 549     datetime_format : str, default None
 550         Format string for datetime objects written into Excel files.
 551         (e.g. 'YYYY-MM-DD HH:MM:SS').
 552     mode : {'w', 'a'}, default 'w'
 553         File mode to use (write or append).
 554 
 555         .. versionadded:: 0.24.0
 556 
 557     Attributes
 558     ----------
 559     None
 560 
 561     Methods
 562     -------
 563     None
 564 
 565     Notes
 566     -----
 567     None of the methods and properties are considered public.
 568 
 569     For compatibility with CSV writers, ExcelWriter serializes lists
 570     and dicts to strings before writing.
 571 
 572     Examples
 573     --------
 574     Default usage:
 575 
 576     >>> with ExcelWriter('path_to_file.xlsx') as writer:
 577     ...     df.to_excel(writer)
 578 
 579     To write to separate sheets in a single file:
 580 
 581     >>> with ExcelWriter('path_to_file.xlsx') as writer:
 582     ...     df1.to_excel(writer, sheet_name='Sheet1')
 583     ...     df2.to_excel(writer, sheet_name='Sheet2')
 584 
 585     You can set the date format or datetime format:
 586 
 587     >>> with ExcelWriter('path_to_file.xlsx',
 588                           date_format='YYYY-MM-DD',
 589                           datetime_format='YYYY-MM-DD HH:MM:SS') as writer:
 590     ...     df.to_excel(writer)
 591 
 592     You can also append to an existing Excel file:
 593 
 594     >>> with ExcelWriter('path_to_file.xlsx', mode='a') as writer:
 595     ...     df.to_excel(writer, sheet_name='Sheet3')
 596     """
 597 
 598     # Defining an ExcelWriter implementation (see abstract methods for more...)
 599 
 600     # - Mandatory
 601     #   - ``write_cells(self, cells, sheet_name=None, startrow=0, startcol=0)``
 602     #     --> called to write additional DataFrames to disk
 603     #   - ``supported_extensions`` (tuple of supported extensions), used to
 604     #      check that engine supports the given extension.
 605     #   - ``engine`` - string that gives the engine name. Necessary to
 606     #     instantiate class directly and bypass ``ExcelWriterMeta`` engine
 607     #     lookup.
 608     #   - ``save(self)`` --> called to save file to disk
 609     # - Mostly mandatory (i.e. should at least exist)
 610     #   - book, cur_sheet, path
 611 
 612     # - Optional:
 613     #   - ``__init__(self, path, engine=None, **kwargs)`` --> always called
 614     #     with path as first argument.
 615 
 616     # You also need to register the class with ``register_writer()``.
 617     # Technically, ExcelWriter implementations don't need to subclass
 618     # ExcelWriter.
 619     def __new__(cls, path, engine=None, **kwargs):
 620         # only switch class if generic(ExcelWriter)
 621 
 622         if cls is ExcelWriter:
 623             if engine is None or (isinstance(engine, str) and engine == "auto"):
 624                 if isinstance(path, str):
 625                     ext = os.path.splitext(path)[-1][1:]
 626                 else:
 627                     ext = "xlsx"
 628 
 629                 try:
 630                     engine = config.get_option(f"io.excel.{ext}.writer")
 631                     if engine == "auto":
 632                         engine = _get_default_writer(ext)
 633                 except KeyError as err:
 634                     raise ValueError(f"No engine for filetype: '{ext}'") from err
 635             cls = get_writer(engine)
 636 
 637         return object.__new__(cls)
 638 
 639     # declare external properties you can count on
 640     book = None
 641     curr_sheet = None
 642     path = None
 643 
 644     @property
 645     @abc.abstractmethod
 646     def supported_extensions(self):
 647         """Extensions that writer engine supports."""

 725     def _value_with_fmt(self, val):
 726         """
 727         Convert numpy types to Python types for the Excel writers.
 728 
 729         Parameters
 730         ----------
 731         val : object
 732             Value to be written into cells
 733 
 734         Returns
 735         -------
 736         Tuple with the first element being the converted value and the second
 737             being an optional format
 738         """
 739         fmt = None
 740 
 741         if is_integer(val):
 742             val = int(val)
 743         elif is_float(val):
 744             val = float(val)
 745         elif is_bool(val):
 746             val = bool(val)
 747         elif isinstance(val, datetime.datetime):
 748             fmt = self.datetime_format
 749         elif isinstance(val, datetime.date):
 750             fmt = self.date_format
 751         elif isinstance(val, datetime.timedelta):
 752             val = val.total_seconds() / float(86400)
 753             fmt = "0"
 754         else:
 755             val = str(val)
 756 
 757         return val, fmt

 759     @classmethod
 760     def check_extension(cls, ext):
 761         """
 762         checks that path's extension against the Writer's supported
 763         extensions.  If it isn't supported, raises UnsupportedFiletypeError.
 764         """
 765         if ext.startswith("."):
 766             ext = ext[1:]
 767         if not any(ext in extension for extension in cls.supported_extensions):
 768             raise ValueError(f"Invalid extension for engine '{cls.engine}': '{ext}'")
 769         else:
 770             return True
