  66     def get_sheet_data(self, sheet, convert_float: bool) -> List[List[Scalar]]:
  67         """
  68         Parse an ODF Table into a list of lists
  69         """
  70         from odf.table import CoveredTableCell, TableCell, TableRow
  71 
  72         covered_cell_name = CoveredTableCell().qname
  73         table_cell_name = TableCell().qname
  74         cell_names = {covered_cell_name, table_cell_name}
  75 
  76         sheet_rows = sheet.getElementsByType(TableRow)
  77         empty_rows = 0
  78         max_row_len = 0
  79 
  80         table: List[List[Scalar]] = []
  81 
  82         for i, sheet_row in enumerate(sheet_rows):
  83             sheet_cells = [x for x in sheet_row.childNodes if x.qname in cell_names]
  84             empty_cells = 0
  85             table_row: List[Scalar] = []
  86 
  87             for j, sheet_cell in enumerate(sheet_cells):
  88                 if sheet_cell.qname == table_cell_name:
  89                     value = self._get_cell_value(sheet_cell, convert_float)
  90                 else:
  91                     value = self.empty_value
  92 
  93                 column_repeat = self._get_column_repeat(sheet_cell)
  94 
  95                 # Queue up empty values, writing only if content succeeds them
  96                 if value == self.empty_value:
  97                     empty_cells += column_repeat
  98                 else:
  99                     table_row.extend([self.empty_value] * empty_cells)
 100                     empty_cells = 0
 101                     table_row.extend([value] * column_repeat)
 102 
 103             if max_row_len < len(table_row):
 104                 max_row_len = len(table_row)
 105 
 106             row_repeat = self._get_row_repeat(sheet_row)
 107             if self._is_empty_row(sheet_row):
 108                 empty_rows += row_repeat
 109             else:
 110                 # add blank rows to our table
 111                 table.extend([[self.empty_value]] * empty_rows)
 112                 empty_rows = 0
 113                 for _ in range(row_repeat):
 114                     table.append(table_row)
 115 
 116         # Make our table square
 117         for row in table:
 118             if len(row) < max_row_len:
 119                 row.extend([self.empty_value] * (max_row_len - len(row)))
 120 
 121         return table

 123     def _get_row_repeat(self, row) -> int:
 124         """
 125         Return number of times this row was repeated
 126         Repeating an empty row appeared to be a common way
 127         of representing sparse rows in the table.
 128         """
 129         from odf.namespaces import TABLENS
 130 
 131         return int(row.attributes.get((TABLENS, "number-rows-repeated"), 1))

 133     def _get_column_repeat(self, cell) -> int:

 138     def _is_empty_row(self, row) -> bool:
 139         """
 140         Helper function to find empty rows
 141         """
 142         for column in row.childNodes:
 143             if len(column.childNodes) > 0:
 144                 return False
 145 
 146         return True

 148     def _get_cell_value(self, cell, convert_float: bool) -> Scalar:
 149         from odf.namespaces import OFFICENS
 150 
 151         cell_type = cell.attributes.get((OFFICENS, "value-type"))
 152         if cell_type == "boolean":
 153             if str(cell) == "TRUE":
 154                 return True
 155             return False
 156         if cell_type is None:
 157             return self.empty_value
 158         elif cell_type == "float":
 159             # GH5394
 160             cell_value = float(cell.attributes.get((OFFICENS, "value")))
 161 
 162             if cell_value == 0.0:  # NA handling
 163                 return str(cell)
 164 
 165             if convert_float:
 166                 val = int(cell_value)
 167                 if val == cell_value:
 168                     return val
 169             return cell_value
 170         elif cell_type == "percentage":
 171             cell_value = cell.attributes.get((OFFICENS, "value"))
 172             return float(cell_value)
 173         elif cell_type == "string":
 174             return self._get_cell_string_value(cell)
 175         elif cell_type == "currency":
 176             cell_value = cell.attributes.get((OFFICENS, "value"))
 177             return float(cell_value)
 178         elif cell_type == "date":
 179             cell_value = cell.attributes.get((OFFICENS, "date-value"))
 180             return pd.to_datetime(cell_value)
 181         elif cell_type == "time":
 182             result = pd.to_datetime(str(cell))
 183             result = cast(pd.Timestamp, result)
 184             return result.time()
 185         else:
 186             raise ValueError(f"Unrecognized type {cell_type}")
 187 
 188     def _get_cell_string_value(self, cell) -> str:
 189         """
 190         Find and decode OpenDocument text:s tags that represent
 191         a run length encoded sequence of space characters.
 192         """
 193         from odf.element import Text, Element
 194         from odf.text import S, P
 195         from odf.namespaces import TEXTNS
 196 
 197         text_p = P().qname
 198         text_s = S().qname
 199 
 200         p = cell.childNodes[0]
 201 
 202         value = []
 203         if p.qname == text_p:
 204             for k, fragment in enumerate(p.childNodes):
 205                 if isinstance(fragment, Text):
 206                     value.append(fragment.data)
 207                 elif isinstance(fragment, Element):
 208                     if fragment.qname == text_s:
 209                         spaces = int(fragment.attributes.get((TEXTNS, "c"), 1))
 210                     value.append(" " * spaces)
 211         return "".join(value)
