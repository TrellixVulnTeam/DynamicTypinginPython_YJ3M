 136 def _parse_date(datestr: str) -> datetime:
 137     """ Given a date in xport format, return Python date. """
 138     try:
 139         # e.g. "16FEB11:10:07:55"
 140         return datetime.strptime(datestr, "%d%b%y:%H:%M:%S")
 141     except ValueError:
 142         return pd.NaT

 145 def _split_line(s: str, parts):
 146     """
 147     Parameters
 148     ----------
 149     s: str
 150         Fixed-length string to split
 151     parts: list of (name, length) pairs
 152         Used to break up string, name '_' will be filtered from output.
 153 
 154     Returns
 155     -------
 156     Dict of name:contents of string at given location.
 157     """
 158     out = {}
 159     start = 0
 160     for name, length in parts:
 161         out[name] = s[start : start + length].strip()
 162         start += length
 163     del out["_"]
 164     return out

 167 def _handle_truncated_float_vec(vec, nbytes):
 168     # This feature is not well documented, but some SAS XPORT files
 169     # have 2-7 byte "truncated" floats.  To read these truncated
 170     # floats, pad them with zeros on the right to make 8 byte floats.
 171     #
 172     # References:
 173     # https://github.com/jcushman/xport/pull/3
 174     # The R "foreign" library
 175 
 176     if nbytes != 8:
 177         vec1 = np.zeros(len(vec), np.dtype("S8"))
 178         dtype = np.dtype(f"S{nbytes},S{8 - nbytes}")
 179         vec2 = vec1.view(dtype=dtype)
 180         vec2["f0"] = vec
 181         return vec2
 182 
 183     return vec

 186 def _parse_float_vec(vec):
 187     """
 188     Parse a vector of float values representing IBM 8 byte floats into
 189     native 8 byte floats.
 190     """
 191     dtype = np.dtype(">u4,>u4")
 192     vec1 = vec.view(dtype=dtype)
 193     xport1 = vec1["f0"]
 194     xport2 = vec1["f1"]
 195 
 196     # Start by setting first half of ieee number to first half of IBM
 197     # number sans exponent
 198     ieee1 = xport1 & 0x00FFFFFF
 199 
 200     # The fraction bit to the left of the binary point in the ieee
 201     # format was set and the number was shifted 0, 1, 2, or 3
 202     # places. This will tell us how to adjust the ibm exponent to be a
 203     # power of 2 ieee exponent and how to shift the fraction bits to
 204     # restore the correct magnitude.
 205     shift = np.zeros(len(vec), dtype=np.uint8)
 206     shift[np.where(xport1 & 0x00200000)] = 1
 207     shift[np.where(xport1 & 0x00400000)] = 2
 208     shift[np.where(xport1 & 0x00800000)] = 3
 209 
 210     # shift the ieee number down the correct number of places then
 211     # set the second half of the ieee number to be the second half
 212     # of the ibm number shifted appropriately, ored with the bits
 213     # from the first half that would have been shifted in if we
 214     # could shift a double. All we are worried about are the low
 215     # order 3 bits of the first half since we're only shifting by
 216     # 1, 2, or 3.
 217     ieee1 >>= shift
 218     ieee2 = (xport2 >> shift) | ((xport1 & 0x00000007) << (29 + (3 - shift)))
 219 
 220     # clear the 1 bit to the left of the binary point
 221     ieee1 &= 0xFFEFFFFF
 222 
 223     # set the exponent of the ieee number to be the actual exponent
 224     # plus the shift count + 1023. Or this into the first half of the
 225     # ieee number. The ibm exponent is excess 64 but is adjusted by 65
 226     # since during conversion to ibm format the exponent is
 227     # incremented by 1 and the fraction bits left 4 positions to the
 228     # right of the radix point.  (had to add >> 24 because C treats &
 229     # 0x7f as 0x7f000000 and Python doesn't)
 230     ieee1 |= ((((((xport1 >> 24) & 0x7F) - 65) << 2) + shift + 1023) << 20) | (
 231         xport1 & 0x80000000
 232     )
 233 
 234     ieee = np.empty((len(ieee1),), dtype=">u4,>u4")
 235     ieee["f0"] = ieee1
 236     ieee["f1"] = ieee2
 237     ieee = ieee.view(dtype=">f8")
 238     ieee = ieee.astype("f8")
 239 
 240     return ieee

 246     def __init__(
 247         self, filepath_or_buffer, index=None, encoding="ISO-8859-1", chunksize=None
 248     ):
 249 
 250         self._encoding = encoding
 251         self._lines_read = 0
 252         self._index = index
 253         self._chunksize = chunksize
 254 
 255         if isinstance(filepath_or_buffer, str):
 256             (
 257                 filepath_or_buffer,
 258                 encoding,
 259                 compression,
 260                 should_close,
 261             ) = get_filepath_or_buffer(filepath_or_buffer, encoding=encoding)
 262 
 263         if isinstance(filepath_or_buffer, (str, bytes)):
 264             self.filepath_or_buffer = open(filepath_or_buffer, "rb")
 265         else:
 266             # Since xport files include non-text byte sequences, xport files
 267             # should already be opened in binary mode in Python 3.
 268             self.filepath_or_buffer = filepath_or_buffer
 269 
 270         self._read_header()
 271 
 272     def close(self):
 273         self.filepath_or_buffer.close()
 274 
 275     def _get_row(self):
 276         return self.filepath_or_buffer.read(80).decode()
 277 
 278     def _read_header(self):
 279         self.filepath_or_buffer.seek(0)
 280 
 281         # read file header
 282         line1 = self._get_row()
 283         if line1 != _correct_line1:
 284             self.close()
 285             raise ValueError("Header record is not an XPORT file.")
 286 
 287         line2 = self._get_row()
 288         fif = [["prefix", 24], ["version", 8], ["OS", 8], ["_", 24], ["created", 16]]
 289         file_info = _split_line(line2, fif)
 290         if file_info["prefix"] != "SAS     SAS     SASLIB":
 291             self.close()
 292             raise ValueError("Header record has invalid prefix.")
 293         file_info["created"] = _parse_date(file_info["created"])
 294         self.file_info = file_info
 295 
 296         line3 = self._get_row()
 297         file_info["modified"] = _parse_date(line3[:16])
 298 
 299         # read member header
 300         header1 = self._get_row()
 301         header2 = self._get_row()
 302         headflag1 = header1.startswith(_correct_header1)
 303         headflag2 = header2 == _correct_header2
 304         if not (headflag1 and headflag2):
 305             self.close()
 306             raise ValueError("Member header not found")
 307         # usually 140, could be 135
 308         fieldnamelength = int(header1[-5:-2])
 309 
 310         # member info
 311         mem = [
 312             ["prefix", 8],
 313             ["set_name", 8],
 314             ["sasdata", 8],
 315             ["version", 8],
 316             ["OS", 8],
 317             ["_", 24],
 318             ["created", 16],
 319         ]
 320         member_info = _split_line(self._get_row(), mem)
 321         mem = [["modified", 16], ["_", 16], ["label", 40], ["type", 8]]
 322         member_info.update(_split_line(self._get_row(), mem))
 323         member_info["modified"] = _parse_date(member_info["modified"])
 324         member_info["created"] = _parse_date(member_info["created"])
 325         self.member_info = member_info
 326 
 327         # read field names
 328         types = {1: "numeric", 2: "char"}
 329         fieldcount = int(self._get_row()[54:58])
 330         datalength = fieldnamelength * fieldcount
 331         # round up to nearest 80
 332         if datalength % 80:
 333             datalength += 80 - datalength % 80
 334         fielddata = self.filepath_or_buffer.read(datalength)
 335         fields = []
 336         obs_length = 0
 337         while len(fielddata) >= fieldnamelength:
 338             # pull data for one field
 339             field, fielddata = (
 340                 fielddata[:fieldnamelength],
 341                 fielddata[fieldnamelength:],
 342             )
 343 
 344             # rest at end gets ignored, so if field is short, pad out
 345             # to match struct pattern below
 346             field = field.ljust(140)
 347 
 348             fieldstruct = struct.unpack(">hhhh8s40s8shhh2s8shhl52s", field)
 349             field = dict(zip(_fieldkeys, fieldstruct))
 350             del field["_"]
 351             field["ntype"] = types[field["ntype"]]
 352             fl = field["field_length"]
 353             if field["ntype"] == "numeric" and ((fl < 2) or (fl > 8)):
 354                 self.close()
 355                 msg = f"Floating field width {fl} is not between 2 and 8."
 356                 raise TypeError(msg)
 357 
 358             for k, v in field.items():
 359                 try:
 360                     field[k] = v.strip()
 361                 except AttributeError:
 362                     pass
 363 
 364             obs_length += field["field_length"]
 365             fields += [field]
 366 
 367         header = self._get_row()
 368         if not header == _correct_obs_header:
 369             self.close()
 370             raise ValueError("Observation header not found.")
 371 
 372         self.fields = fields
 373         self.record_length = obs_length
 374         self.record_start = self.filepath_or_buffer.tell()
 375 
 376         self.nobs = self._record_count()
 377         self.columns = [x["name"].decode() for x in self.fields]
 378 
 379         # Setup the dtype.
 380         dtypel = [
 381             ("s" + str(i), "S" + str(field["field_length"]))
 382             for i, field in enumerate(self.fields)
 383         ]
 384         dtype = np.dtype(dtypel)
 385         self._dtype = dtype
 386 
 387     def __next__(self):
 388         return self.read(nrows=self._chunksize or 1)
 389 
 390     def _record_count(self) -> int:
 391         """
 392         Get number of records in file.
 393 
 394         This is maybe suboptimal because we have to seek to the end of
 395         the file.
 396 
 397         Side effect: returns file position to record_start.
 398         """
 399         self.filepath_or_buffer.seek(0, 2)
 400         total_records_length = self.filepath_or_buffer.tell() - self.record_start
 401 
 402         if total_records_length % 80 != 0:
 403             warnings.warn("xport file may be corrupted")
 404 
 405         if self.record_length > 80:
 406             self.filepath_or_buffer.seek(self.record_start)
 407             return total_records_length // self.record_length
 408 
 409         self.filepath_or_buffer.seek(-80, 2)
 410         last_card = self.filepath_or_buffer.read(80)
 411         last_card = np.frombuffer(last_card, dtype=np.uint64)
 412 
 413         # 8 byte blank
 414         ix = np.flatnonzero(last_card == 2314885530818453536)
 415 
 416         if len(ix) == 0:
 417             tail_pad = 0
 418         else:
 419             tail_pad = 8 * len(ix)
 420 
 421         self.filepath_or_buffer.seek(self.record_start)
 422 
 423         return (total_records_length - tail_pad) // self.record_length

 425     def get_chunk(self, size=None):
 426         """
 427         Reads lines from Xport file and returns as dataframe
 428 
 429         Parameters
 430         ----------
 431         size : int, defaults to None
 432             Number of lines to read.  If None, reads whole file.
 433 
 434         Returns
 435         -------
 436         DataFrame
 437         """
 438         if size is None:
 439             size = self._chunksize
 440         return self.read(nrows=size)

 442     def _missing_double(self, vec):
 443         v = vec.view(dtype="u1,u1,u2,u4")
 444         miss = (v["f1"] == 0) & (v["f2"] == 0) & (v["f3"] == 0)
 445         miss1 = (
 446             ((v["f0"] >= 0x41) & (v["f0"] <= 0x5A))
 447             | (v["f0"] == 0x5F)
 448             | (v["f0"] == 0x2E)
 449         )
 450         miss &= miss1
 451         return miss

 453     @Appender(_read_method_doc)
 454     def read(self, nrows=None):
 455 
 456         if nrows is None:
 457             nrows = self.nobs
 458 
 459         read_lines = min(nrows, self.nobs - self._lines_read)
 460         read_len = read_lines * self.record_length
 461         if read_len <= 0:
 462             self.close()
 463             raise StopIteration
 464         raw = self.filepath_or_buffer.read(read_len)
 465         data = np.frombuffer(raw, dtype=self._dtype, count=read_lines)
 466 
 467         df = pd.DataFrame(index=range(read_lines))
 468         for j, x in enumerate(self.columns):
 469             vec = data["s" + str(j)]
 470             ntype = self.fields[j]["ntype"]
 471             if ntype == "numeric":
 472                 vec = _handle_truncated_float_vec(vec, self.fields[j]["field_length"])
 473                 miss = self._missing_double(vec)
 474                 v = _parse_float_vec(vec)
 475                 v[miss] = np.nan
 476             elif self.fields[j]["ntype"] == "char":
 477                 v = [y.rstrip() for y in vec]
 478 
 479                 if self._encoding is not None:
 480                     v = [y.decode(self._encoding) for y in v]
 481 
 482             df[x] = v
 483 
 484         if self._index is None:
 485             df.index = range(self._lines_read, self._lines_read + read_lines)
 486         else:
 487             df = df.set_index(self._index)
 488 
 489         self._lines_read += read_lines
 490 
 491         return df
