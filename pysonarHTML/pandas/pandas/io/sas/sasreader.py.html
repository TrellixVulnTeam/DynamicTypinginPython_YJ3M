  54 def read_sas(
  55     filepath_or_buffer: FilePathOrBuffer,
  56     format: Optional[str] = None,
  57     index: Optional[Label] = None,
  58     encoding: Optional[str] = None,
  59     chunksize: Optional[int] = None,
  60     iterator: bool = False,
  61 ) -> Union["DataFrame", ReaderBase]:
  62     """
  63     Read SAS files stored as either XPORT or SAS7BDAT format files.
  64 
  65     Parameters
  66     ----------
  67     filepath_or_buffer : str, path object or file-like object
  68         Any valid string path is acceptable. The string could be a URL. Valid
  69         URL schemes include http, ftp, s3, and file. For file URLs, a host is
  70         expected. A local file could be:
  71         ``file://localhost/path/to/table.sas``.
  72 
  73         If you want to pass in a path object, pandas accepts any
  74         ``os.PathLike``.
  75 
  76         By file-like object, we refer to objects with a ``read()`` method,
  77         such as a file handler (e.g. via builtin ``open`` function)
  78         or ``StringIO``.
  79     format : str {'xport', 'sas7bdat'} or None
  80         If None, file format is inferred from file extension. If 'xport' or
  81         'sas7bdat', uses the corresponding format.
  82     index : identifier of index column, defaults to None
  83         Identifier of column that should be used as index of the DataFrame.
  84     encoding : str, default is None
  85         Encoding for text data.  If None, text data are stored as raw bytes.
  86     chunksize : int
  87         Read file `chunksize` lines at a time, returns iterator.
  88     iterator : bool, defaults to False
  89         If True, returns an iterator for reading the file incrementally.
  90 
  91     Returns
  92     -------
  93     DataFrame if iterator=False and chunksize=None, else SAS7BDATReader
  94     or XportReader
  95     """
  96     if format is None:
  97         buffer_error_msg = (
  98             "If this is a buffer object rather "
  99             "than a string name, you must specify a format string"
 100         )
 101         filepath_or_buffer = stringify_path(filepath_or_buffer)
 102         if not isinstance(filepath_or_buffer, str):
 103             raise ValueError(buffer_error_msg)
 104         fname = filepath_or_buffer.lower()
 105         if fname.endswith(".xpt"):
 106             format = "xport"
 107         elif fname.endswith(".sas7bdat"):
 108             format = "sas7bdat"
 109         else:
 110             raise ValueError("unable to infer format of SAS file")
 111 
 112     reader: ReaderBase
 113     if format.lower() == "xport":
 114         from pandas.io.sas.sas_xport import XportReader
 115 
 116         reader = XportReader(
 117             filepath_or_buffer, index=index, encoding=encoding, chunksize=chunksize
 118         )
 119     elif format.lower() == "sas7bdat":
 120         from pandas.io.sas.sas7bdat import SAS7BDATReader
 121 
 122         reader = SAS7BDATReader(
 123             filepath_or_buffer, index=index, encoding=encoding, chunksize=chunksize
 124         )
 125     else:
 126         raise ValueError("unknown SAS format")
 127 
 128     if iterator or chunksize:
 129         return reader
 130 
 131     data = reader.read()
 132     reader.close()
 133     return data
