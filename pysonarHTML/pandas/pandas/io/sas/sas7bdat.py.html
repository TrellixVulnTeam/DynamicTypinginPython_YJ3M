  32 def _convert_datetimes(sas_datetimes: pd.Series, unit: str) -> pd.Series:
  33     """
  34     Convert to Timestamp if possible, otherwise to datetime.datetime.
  35     SAS float64 lacks precision for more than ms resolution so the fit
  36     to datetime.datetime is ok.
  37 
  38     Parameters
  39     ----------
  40     sas_datetimes : {Series, Sequence[float]}
  41        Dates or datetimes in SAS
  42     unit : {str}
  43        "d" if the floats represent dates, "s" for datetimes
  44 
  45     Returns
  46     -------
  47     Series
  48        Series of datetime64 dtype or datetime.datetime.
  49     """
  50     try:
  51         return pd.to_datetime(sas_datetimes, unit=unit, origin="1960-01-01")
  52     except OutOfBoundsDatetime:
  53         if unit == "s":
  54             return sas_datetimes.apply(
  55                 lambda sas_float: datetime(1960, 1, 1) + timedelta(seconds=sas_float)
  56             )
  57         elif unit == "d":
  58             return sas_datetimes.apply(
  59                 lambda sas_float: datetime(1960, 1, 1) + timedelta(days=sas_float)
  60             )
  61         else:
  62             raise ValueError("unit must be 'd' or 's'")

  74 class SAS7BDATReader(ReaderBase, abc.Iterator):
  75     """
  76     Read SAS files in SAS7BDAT format.
  77 
  78     Parameters
  79     ----------
  80     path_or_buf : path name or buffer
  81         Name of SAS file or file-like object pointing to SAS file
  82         contents.
  83     index : column identifier, defaults to None

  85     convert_dates : boolean, defaults to True
  86         Attempt to convert dates to Pandas datetime values.  Note that
  87         some rarely used SAS date formats may be unsupported.
  88     blank_missing : boolean, defaults to True

 169     def _get_properties(self):
 170 
 171         # Check magic number
 172         self._path_or_buf.seek(0)
 173         self._cached_page = self._path_or_buf.read(288)
 174         if self._cached_page[0 : len(const.magic)] != const.magic:
 175             self.close()
 176             raise ValueError("magic number mismatch (not a SAS file?)")
 177 
 178         # Get alignment information
 179         align1, align2 = 0, 0
 180         buf = self._read_bytes(const.align_1_offset, const.align_1_length)
 181         if buf == const.u64_byte_checker_value:
 182             align2 = const.align_2_value
 183             self.U64 = True
 184             self._int_length = 8
 185             self._page_bit_offset = const.page_bit_offset_x64
 186             self._subheader_pointer_length = const.subheader_pointer_length_x64
 187         else:
 188             self.U64 = False
 189             self._page_bit_offset = const.page_bit_offset_x86
 190             self._subheader_pointer_length = const.subheader_pointer_length_x86
 191             self._int_length = 4
 192         buf = self._read_bytes(const.align_2_offset, const.align_2_length)
 193         if buf == const.align_1_checker_value:
 194             align1 = const.align_2_value
 195         total_align = align1 + align2
 196 
 197         # Get endianness information
 198         buf = self._read_bytes(const.endianness_offset, const.endianness_length)
 199         if buf == b"\x01":
 200             self.byte_order = "<"
 201         else:
 202             self.byte_order = ">"
 203 
 204         # Get encoding information
 205         buf = self._read_bytes(const.encoding_offset, const.encoding_length)[0]
 206         if buf in const.encoding_names:
 207             self.file_encoding = const.encoding_names[buf]
 208         else:
 209             self.file_encoding = f"unknown (code={buf})"
 210 
 211         # Get platform information
 212         buf = self._read_bytes(const.platform_offset, const.platform_length)
 213         if buf == b"1":
 214             self.platform = "unix"
 215         elif buf == b"2":
 216             self.platform = "windows"
 217         else:
 218             self.platform = "unknown"
 219 
 220         buf = self._read_bytes(const.dataset_offset, const.dataset_length)
 221         self.name = buf.rstrip(b"\x00 ")
 222         if self.convert_header_text:
 223             self.name = self.name.decode(self.encoding or self.default_encoding)
 224 
 225         buf = self._read_bytes(const.file_type_offset, const.file_type_length)
 226         self.file_type = buf.rstrip(b"\x00 ")
 227         if self.convert_header_text:
 228             self.file_type = self.file_type.decode(
 229                 self.encoding or self.default_encoding
 230             )
 231 
 232         # Timestamp is epoch 01/01/1960
 233         epoch = datetime(1960, 1, 1)
 234         x = self._read_float(
 235             const.date_created_offset + align1, const.date_created_length
 236         )
 237         self.date_created = epoch + pd.to_timedelta(x, unit="s")
 238         x = self._read_float(
 239             const.date_modified_offset + align1, const.date_modified_length
 240         )
 241         self.date_modified = epoch + pd.to_timedelta(x, unit="s")
 242 
 243         self.header_length = self._read_int(
 244             const.header_size_offset + align1, const.header_size_length
 245         )
 246 
 247         # Read the rest of the header into cached_page.
 248         buf = self._path_or_buf.read(self.header_length - 288)
 249         self._cached_page += buf
 250         if len(self._cached_page) != self.header_length:
 251             self.close()
 252             raise ValueError("The SAS7BDAT file appears to be truncated.")
 253 
 254         self._page_length = self._read_int(
 255             const.page_size_offset + align1, const.page_size_length
 256         )
 257         self._page_count = self._read_int(
 258             const.page_count_offset + align1, const.page_count_length
 259         )
 260 
 261         buf = self._read_bytes(
 262             const.sas_release_offset + total_align, const.sas_release_length
 263         )
 264         self.sas_release = buf.rstrip(b"\x00 ")
 265         if self.convert_header_text:
 266             self.sas_release = self.sas_release.decode(
 267                 self.encoding or self.default_encoding
 268             )
 269 
 270         buf = self._read_bytes(
 271             const.sas_server_type_offset + total_align, const.sas_server_type_length
 272         )
 273         self.server_type = buf.rstrip(b"\x00 ")
 274         if self.convert_header_text:
 275             self.server_type = self.server_type.decode(
 276                 self.encoding or self.default_encoding
 277             )
 278 
 279         buf = self._read_bytes(
 280             const.os_version_number_offset + total_align, const.os_version_number_length
 281         )
 282         self.os_version = buf.rstrip(b"\x00 ")
 283         if self.convert_header_text:
 284             self.os_version = self.os_version.decode(
 285                 self.encoding or self.default_encoding
 286             )
 287 
 288         buf = self._read_bytes(const.os_name_offset + total_align, const.os_name_length)
 289         buf = buf.rstrip(b"\x00 ")
 290         if len(buf) > 0:
 291             self.os_name = buf.decode(self.encoding or self.default_encoding)
 292         else:
 293             buf = self._read_bytes(
 294                 const.os_maker_offset + total_align, const.os_maker_length
 295             )
 296             self.os_name = buf.rstrip(b"\x00 ")
 297             if self.convert_header_text:
 298                 self.os_name = self.os_name.decode(
 299                     self.encoding or self.default_encoding
 300                 )

 302     def __next__(self):
 303         da = self.read(nrows=self.chunksize or 1)
 304         if da is None:
 305             raise StopIteration
 306         return da

 308     # Read a single float of the given width (4 or 8).
 309     def _read_float(self, offset, width):
 310         if width not in (4, 8):
 311             self.close()
 312             raise ValueError("invalid float width")
 313         buf = self._read_bytes(offset, width)
 314         fd = "f" if width == 4 else "d"
 315         return struct.unpack(self.byte_order + fd, buf)[0]
 316 
 317     # Read a single signed integer of the given width (1, 2, 4 or 8).
 318     def _read_int(self, offset, width):
 319         if width not in (1, 2, 4, 8):
 320             self.close()
 321             raise ValueError("invalid int width")
 322         buf = self._read_bytes(offset, width)
 323         it = {1: "b", 2: "h", 4: "l", 8: "q"}[width]
 324         iv = struct.unpack(self.byte_order + it, buf)[0]
 325         return iv
 326 
 327     def _read_bytes(self, offset, length):
 328         if self._cached_page is None:
 329             self._path_or_buf.seek(offset)
 330             buf = self._path_or_buf.read(length)
 331             if len(buf) < length:
 332                 self.close()
 333                 msg = f"Unable to read {length:d} bytes from file position {offset:d}."
 334                 raise ValueError(msg)
 335             return buf
 336         else:
 337             if offset + length > len(self._cached_page):
 338                 self.close()
 339                 raise ValueError("The cached page is too small.")
 340             return self._cached_page[offset : offset + length]
 341 
 342     def _parse_metadata(self):
 343         done = False
 344         while not done:
 345             self._cached_page = self._path_or_buf.read(self._page_length)
 346             if len(self._cached_page) <= 0:
 347                 break
 348             if len(self._cached_page) != self._page_length:
 349                 self.close()
 350                 raise ValueError("Failed to read a meta data page from the SAS file.")
 351             done = self._process_page_meta()
 352 
 353     def _process_page_meta(self):
 354         self._read_page_header()
 355         pt = [const.page_meta_type, const.page_amd_type] + const.page_mix_types
 356         if self._current_page_type in pt:
 357             self._process_page_metadata()
 358         is_data_page = self._current_page_type & const.page_data_type
 359         is_mix_page = self._current_page_type in const.page_mix_types
 360         return (
 361             is_data_page
 362             or is_mix_page
 363             or self._current_page_data_subheader_pointers != []
 364         )
 365 
 366     def _read_page_header(self):
 367         bit_offset = self._page_bit_offset
 368         tx = const.page_type_offset + bit_offset
 369         self._current_page_type = self._read_int(tx, const.page_type_length)
 370         tx = const.block_count_offset + bit_offset
 371         self._current_page_block_count = self._read_int(tx, const.block_count_length)
 372         tx = const.subheader_count_offset + bit_offset
 373         self._current_page_subheaders_count = self._read_int(
 374             tx, const.subheader_count_length
 375         )
 376 
 377     def _process_page_metadata(self):
 378         bit_offset = self._page_bit_offset
 379 
 380         for i in range(self._current_page_subheaders_count):
 381             pointer = self._process_subheader_pointers(
 382                 const.subheader_pointers_offset + bit_offset, i
 383             )
 384             if pointer.length == 0:
 385                 continue
 386             if pointer.compression == const.truncated_subheader_id:
 387                 continue
 388             subheader_signature = self._read_subheader_signature(pointer.offset)
 389             subheader_index = self._get_subheader_index(
 390                 subheader_signature, pointer.compression, pointer.ptype
 391             )
 392             self._process_subheader(subheader_index, pointer)
 393 
 394     def _get_subheader_index(self, signature, compression, ptype):
 395         index = const.subheader_signature_to_index.get(signature)
 396         if index is None:
 397             f1 = (compression == const.compressed_subheader_id) or (compression == 0)
 398             f2 = ptype == const.compressed_subheader_type
 399             if (self.compression != "") and f1 and f2:
 400                 index = const.SASIndex.data_subheader_index
 401             else:
 402                 self.close()
 403                 raise ValueError("Unknown subheader signature")
 404         return index
 405 
 406     def _process_subheader_pointers(self, offset, subheader_pointer_index):
 407 
 408         subheader_pointer_length = self._subheader_pointer_length
 409         total_offset = offset + subheader_pointer_length * subheader_pointer_index
 410 
 411         subheader_offset = self._read_int(total_offset, self._int_length)
 412         total_offset += self._int_length
 413 
 414         subheader_length = self._read_int(total_offset, self._int_length)
 415         total_offset += self._int_length
 416 
 417         subheader_compression = self._read_int(total_offset, 1)
 418         total_offset += 1
 419 
 420         subheader_type = self._read_int(total_offset, 1)
 421 
 422         x = _subheader_pointer()
 423         x.offset = subheader_offset
 424         x.length = subheader_length
 425         x.compression = subheader_compression
 426         x.ptype = subheader_type
 427 
 428         return x

 430     def _read_subheader_signature(self, offset):
 431         subheader_signature = self._read_bytes(offset, self._int_length)
 432         return subheader_signature

 434     def _process_subheader(self, subheader_index, pointer):
 435         offset = pointer.offset
 436         length = pointer.length
 437 
 438         if subheader_index == const.SASIndex.row_size_index:
 439             processor = self._process_rowsize_subheader
 440         elif subheader_index == const.SASIndex.column_size_index:
 441             processor = self._process_columnsize_subheader
 442         elif subheader_index == const.SASIndex.column_text_index:
 443             processor = self._process_columntext_subheader
 444         elif subheader_index == const.SASIndex.column_name_index:
 445             processor = self._process_columnname_subheader
 446         elif subheader_index == const.SASIndex.column_attributes_index:
 447             processor = self._process_columnattributes_subheader
 448         elif subheader_index == const.SASIndex.format_and_label_index:
 449             processor = self._process_format_subheader
 450         elif subheader_index == const.SASIndex.column_list_index:
 451             processor = self._process_columnlist_subheader
 452         elif subheader_index == const.SASIndex.subheader_counts_index:
 453             processor = self._process_subheader_counts
 454         elif subheader_index == const.SASIndex.data_subheader_index:
 455             self._current_page_data_subheader_pointers.append(pointer)
 456             return
 457         else:
 458             raise ValueError("unknown subheader index")
 459 
 460         processor(offset, length)

 462     def _process_rowsize_subheader(self, offset, length):
 463 
 464         int_len = self._int_length
 465         lcs_offset = offset
 466         lcp_offset = offset
 467         if self.U64:
 468             lcs_offset += 682
 469             lcp_offset += 706
 470         else:
 471             lcs_offset += 354
 472             lcp_offset += 378
 473 
 474         self.row_length = self._read_int(
 475             offset + const.row_length_offset_multiplier * int_len, int_len
 476         )
 477         self.row_count = self._read_int(
 478             offset + const.row_count_offset_multiplier * int_len, int_len
 479         )
 480         self.col_count_p1 = self._read_int(
 481             offset + const.col_count_p1_multiplier * int_len, int_len
 482         )
 483         self.col_count_p2 = self._read_int(
 484             offset + const.col_count_p2_multiplier * int_len, int_len
 485         )
 486         mx = const.row_count_on_mix_page_offset_multiplier * int_len
 487         self._mix_page_row_count = self._read_int(offset + mx, int_len)
 488         self._lcs = self._read_int(lcs_offset, 2)
 489         self._lcp = self._read_int(lcp_offset, 2)

 491     def _process_columnsize_subheader(self, offset, length):
 492         int_len = self._int_length
 493         offset += int_len
 494         self.column_count = self._read_int(offset, int_len)
 495         if self.col_count_p1 + self.col_count_p2 != self.column_count:
 496             print(
 497                 f"Warning: column count mismatch ({self.col_count_p1} + "
 498                 f"{self.col_count_p2} != {self.column_count})\n"
 499             )

 505     def _process_columntext_subheader(self, offset, length):
 506 
 507         offset += self._int_length
 508         text_block_size = self._read_int(offset, const.text_block_size_length)
 509 
 510         buf = self._read_bytes(offset, text_block_size)
 511         cname_raw = buf[0:text_block_size].rstrip(b"\x00 ")
 512         cname = cname_raw
 513         if self.convert_header_text:
 514             cname = cname.decode(self.encoding or self.default_encoding)
 515         self.column_names_strings.append(cname)
 516 
 517         if len(self.column_names_strings) == 1:
 518             compression_literal = ""
 519             for cl in const.compression_literals:
 520                 if cl in cname_raw:
 521                     compression_literal = cl
 522             self.compression = compression_literal
 523             offset -= self._int_length
 524 
 525             offset1 = offset + 16
 526             if self.U64:
 527                 offset1 += 4
 528 
 529             buf = self._read_bytes(offset1, self._lcp)
 530             compression_literal = buf.rstrip(b"\x00")
 531             if compression_literal == "":
 532                 self._lcs = 0
 533                 offset1 = offset + 32
 534                 if self.U64:
 535                     offset1 += 4
 536                 buf = self._read_bytes(offset1, self._lcp)
 537                 self.creator_proc = buf[0 : self._lcp]
 538             elif compression_literal == const.rle_compression:
 539                 offset1 = offset + 40
 540                 if self.U64:
 541                     offset1 += 4
 542                 buf = self._read_bytes(offset1, self._lcp)
 543                 self.creator_proc = buf[0 : self._lcp]
 544             elif self._lcs > 0:
 545                 self._lcp = 0
 546                 offset1 = offset + 16
 547                 if self.U64:
 548                     offset1 += 4
 549                 buf = self._read_bytes(offset1, self._lcs)
 550                 self.creator_proc = buf[0 : self._lcp]
 551             if self.convert_header_text:
 552                 if hasattr(self, "creator_proc"):
 553                     self.creator_proc = self.creator_proc.decode(
 554                         self.encoding or self.default_encoding
 555                     )

 557     def _process_columnname_subheader(self, offset, length):
 558         int_len = self._int_length
 559         offset += int_len
 560         column_name_pointers_count = (length - 2 * int_len - 12) // 8
 561         for i in range(column_name_pointers_count):
 562             text_subheader = (
 563                 offset
 564                 + const.column_name_pointer_length * (i + 1)
 565                 + const.column_name_text_subheader_offset
 566             )
 567             col_name_offset = (
 568                 offset
 569                 + const.column_name_pointer_length * (i + 1)
 570                 + const.column_name_offset_offset
 571             )
 572             col_name_length = (
 573                 offset
 574                 + const.column_name_pointer_length * (i + 1)
 575                 + const.column_name_length_offset
 576             )
 577 
 578             idx = self._read_int(
 579                 text_subheader, const.column_name_text_subheader_length
 580             )
 581             col_offset = self._read_int(
 582                 col_name_offset, const.column_name_offset_length
 583             )
 584             col_len = self._read_int(col_name_length, const.column_name_length_length)
 585 
 586             name_str = self.column_names_strings[idx]
 587             self.column_names.append(name_str[col_offset : col_offset + col_len])

 589     def _process_columnattributes_subheader(self, offset, length):
 590         int_len = self._int_length
 591         column_attributes_vectors_count = (length - 2 * int_len - 12) // (int_len + 8)
 592         for i in range(column_attributes_vectors_count):
 593             col_data_offset = (
 594                 offset + int_len + const.column_data_offset_offset + i * (int_len + 8)
 595             )
 596             col_data_len = (
 597                 offset
 598                 + 2 * int_len
 599                 + const.column_data_length_offset
 600                 + i * (int_len + 8)
 601             )
 602             col_types = (
 603                 offset + 2 * int_len + const.column_type_offset + i * (int_len + 8)
 604             )

 667     def read(self, nrows=None):
 668 
 669         if (nrows is None) and (self.chunksize is not None):
 670             nrows = self.chunksize
 671         elif nrows is None:
 672             nrows = self.row_count
 673 
 674         if len(self._column_types) == 0:
 675             self.close()
 676             raise EmptyDataError("No columns to parse from file")
 677 
 678         if self._current_row_in_file_index >= self.row_count:
 679             return None
 680 
 681         m = self.row_count - self._current_row_in_file_index
 682         if nrows > m:
 683             nrows = m
 684 
 685         nd = self._column_types.count(b"d")
 686         ns = self._column_types.count(b"s")
 687 
 688         self._string_chunk = np.empty((ns, nrows), dtype=np.object)
 689         self._byte_chunk = np.zeros((nd, 8 * nrows), dtype=np.uint8)
 690 
 691         self._current_row_in_chunk_index = 0
 692         p = Parser(self)
 693         p.read(nrows)
 694 
 695         rslt = self._chunk_to_dataframe()
 696         if self.index is not None:
 697             rslt = rslt.set_index(self.index)
 698 
 699         return rslt

 701     def _read_next_page(self):
 702         self._current_page_data_subheader_pointers = []
 703         self._cached_page = self._path_or_buf.read(self._page_length)
 704         if len(self._cached_page) <= 0:
 705             return True
 706         elif len(self._cached_page) != self._page_length:
 707             self.close()
 708             msg = (
 709                 "failed to read complete page from file (read "
 710                 f"{len(self._cached_page):d} of {self._page_length:d} bytes)"
 711             )
 712             raise ValueError(msg)
 713 
 714         self._read_page_header()
 715         page_type = self._current_page_type
 716         if page_type == const.page_meta_type:
 717             self._process_page_metadata()
 718 
 719         is_data_page = page_type & const.page_data_type
 720         pt = [const.page_meta_type] + const.page_mix_types
 721         if not is_data_page and self._current_page_type not in pt:
 722             return self._read_next_page()
 723 
 724         return False

 726     def _chunk_to_dataframe(self):
 727 
 728         n = self._current_row_in_chunk_index
 729         m = self._current_row_in_file_index
 730         ix = range(m - n, m)
 731         rslt = pd.DataFrame(index=ix)
 732 
 733         js, jb = 0, 0
 734         for j in range(self.column_count):
 735 
 736             name = self.column_names[j]
 737 
 738             if self._column_types[j] == b"d":
 739                 rslt[name] = self._byte_chunk[jb, :].view(dtype=self.byte_order + "d")
 740                 rslt[name] = np.asarray(rslt[name], dtype=np.float64)
 741                 if self.convert_dates:
 742                     if self.column_formats[j] in const.sas_date_formats:
 743                         rslt[name] = _convert_datetimes(rslt[name], "d")
 744                     elif self.column_formats[j] in const.sas_datetime_formats:
 745                         rslt[name] = _convert_datetimes(rslt[name], "s")
 746                 jb += 1
 747             elif self._column_types[j] == b"s":
 748                 rslt[name] = self._string_chunk[js, :]
 749                 if self.convert_text and (self.encoding is not None):
 750                     rslt[name] = rslt[name].str.decode(
 751                         self.encoding or self.default_encoding
 752                     )
 753                 if self.blank_missing:
 754                     ii = rslt[name].str.len() == 0
 755                     rslt.loc[ii, name] = np.nan
 756                 js += 1
 757             else:
 758                 self.close()
 759                 raise ValueError(f"unknown column type {self._column_types[j]}")
 760 
 761         return rslt
