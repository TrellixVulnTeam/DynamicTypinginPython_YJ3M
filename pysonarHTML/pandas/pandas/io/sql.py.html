  80 def _handle_date_column(col, utc=None, format=None):
  81     if isinstance(format, dict):
  82         return to_datetime(col, errors="ignore", **format)
  83     else:
  84         # Allow passing of formatting string for integers
  85         # GH17855
  86         if format is None and (
  87             issubclass(col.dtype.type, np.floating)
  88             or issubclass(col.dtype.type, np.integer)
  89         ):
  90             format = "s"
  91         if format in ["D", "d", "h", "m", "s", "ms", "us", "ns"]:
  92             return to_datetime(col, errors="coerce", unit=format, utc=utc)
  93         elif is_datetime64tz_dtype(col.dtype):
  94             # coerce to UTC timezone
  95             # GH11216
  96             return to_datetime(col, utc=True)
  97         else:
  98             return to_datetime(col, errors="coerce", format=format, utc=utc)

 101 def _parse_date_columns(data_frame, parse_dates):
 102     """
 103     Force non-datetime columns to be read as such.
 104     Supports both string formatted and integer timestamp columns.
 105     """
 106     parse_dates = _process_parse_dates_argument(parse_dates)
 107 
 108     # we want to coerce datetime64_tz dtypes for now to UTC
 109     # we could in theory do a 'nice' conversion from a FixedOffset tz
 110     # GH11216
 111     for col_name, df_col in data_frame.items():
 112         if is_datetime64tz_dtype(df_col.dtype) or col_name in parse_dates:
 113             try:
 114                 fmt = parse_dates[col_name]
 115             except TypeError:
 116                 fmt = None
 117             data_frame[col_name] = _handle_date_column(df_col, format=fmt)
 118 
 119     return data_frame

 194 def read_sql_table(
 195     table_name,
 196     con,
 197     schema=None,
 198     index_col=None,
 199     coerce_float=True,
 200     parse_dates=None,
 201     columns=None,
 202     chunksize: Optional[int] = None,
 203 ) -> Union[DataFrame, Iterator[DataFrame]]:
 204     """
 205     Read SQL database table into a DataFrame.
 206 
 207     Given a table name and a SQLAlchemy connectable, returns a DataFrame.
 208     This function does not support DBAPI connections.
 209 
 210     Parameters
 211     ----------
 212     table_name : str
 213         Name of SQL table in database.
 214     con : SQLAlchemy connectable or str
 215         A database URI could be provided as as str.
 216         SQLite DBAPI connection mode not supported.
 217     schema : str, default None
 218         Name of SQL schema in database to query (if database flavor
 219         supports this). Uses default schema if None (default).
 220     index_col : str or list of str, optional, default: None
 221         Column(s) to set as index(MultiIndex).
 222     coerce_float : bool, default True
 223         Attempts to convert values of non-string, non-numeric objects (like
 224         decimal.Decimal) to floating point. Can result in loss of Precision.
 225     parse_dates : list or dict, default None
 226         - List of column names to parse as dates.
 227         - Dict of ``{column_name: format string}`` where format string is
 228           strftime compatible in case of parsing string times or is one of
 229           (D, s, ns, ms, us) in case of parsing integer timestamps.
 230         - Dict of ``{column_name: arg dict}``, where the arg dict corresponds
 231           to the keyword arguments of :func:`pandas.to_datetime`
 232           Especially useful with databases without native Datetime support,
 233           such as SQLite.
 234     columns : list, default None
 235         List of column names to select from SQL table.
 236     chunksize : int, default None
 237         If specified, returns an iterator where `chunksize` is the number of
 238         rows to include in each chunk.
 239 
 240     Returns
 241     -------
 242     DataFrame or Iterator[DataFrame]
 243         A SQL table is returned as two-dimensional data structure with labeled
 244         axes.
 245 
 246     See Also
 247     --------
 248     read_sql_query : Read SQL query into a DataFrame.
 249     read_sql : Read SQL query or database table into a DataFrame.
 250 
 251     Notes
 252     -----
 253     Any datetime values with time zone information will be converted to UTC.
 254 
 255     Examples
 256     --------
 257     >>> pd.read_sql_table('table_name', 'postgres:///db_name')  # doctest:+SKIP
 258     """
 259     con = _engine_builder(con)
 260     if not _is_sqlalchemy_connectable(con):
 261         raise NotImplementedError(
 262             "read_sql_table only supported for SQLAlchemy connectable."
 263         )
 264     import sqlalchemy
 265     from sqlalchemy.schema import MetaData
 266 
 267     meta = MetaData(con, schema=schema)
 268     try:
 269         meta.reflect(only=[table_name], views=True)
 270     except sqlalchemy.exc.InvalidRequestError as err:
 271         raise ValueError(f"Table {table_name} not found") from err
 272 
 273     pandas_sql = SQLDatabase(con, meta=meta)
 274     table = pandas_sql.read_table(
 275         table_name,
 276         index_col=index_col,
 277         coerce_float=coerce_float,
 278         parse_dates=parse_dates,
 279         columns=columns,
 280         chunksize=chunksize,
 281     )
 282 
 283     if table is not None:
 284         return table
 285     else:
 286         raise ValueError(f"Table {table_name} not found", con)

 429     ``read_sql_query`` (for backward compatibility). It will delegate
 430     to the specific function depending on the provided input. A SQL query
 431     will be routed to ``read_sql_query``, while a database table name will
 432     be routed to ``read_sql_table``. Note that the delegated function might
 433     have more specific notes about their functionality not listed here.
 434

 436     ----------
 437     sql : str or SQLAlchemy Selectable (select or text object)
 438         SQL query to be executed or a table name.

 444     index_col : str or list of str, optional, default: None
 445         Column(s) to set as index(MultiIndex).
 446     coerce_float : bool, default True

 448         decimal.Decimal) to floating point, useful for SQL result sets.
 449     params : list, tuple or dict, optional, default: None
 450         List of parameters to pass to execute method.  The syntax used
 451         to pass parameters is database driver dependent. Check your
 452         database driver documentation for which of the five syntax styles,
 453         described in PEP 249's paramstyle, is supported.
 454         Eg. for psycopg2, uses %(name)s so use params={'name' : 'value'}.
 455     parse_dates : list or dict, default: None
 456         - List of column names to parse as dates.
 457         - Dict of ``{column_name: format string}`` where format string is
 458           strftime compatible in case of parsing string times, or is one of
 459           (D, s, ns, ms, us) in case of parsing integer timestamps.
 460         - Dict of ``{column_name: arg dict}``, where the arg dict corresponds
 461           to the keyword arguments of :func:`pandas.to_datetime`
 462           Especially useful with databases without native Datetime support,
 463           such as SQLite.
 464     columns : list, default: None
 465         List of column names to select from SQL table (only used when reading
 466         a table).
 467     chunksize : int, default None
 468         If specified, return an iterator where `chunksize` is the
 469         number of rows to include in each chunk.
 470 
 471     Returns
 472     -------
 473     DataFrame or Iterator[DataFrame]
 474 
 475     See Also
 476     --------
 477     read_sql_table : Read SQL database table into a DataFrame.

 479     """

 484             sql,

 519 def to_sql(
 520     frame,
 521     name,
 522     con,
 523     schema=None,
 524     if_exists="fail",
 525     index=True,
 526     index_label=None,
 527     chunksize=None,
 528     dtype=None,
 529     method=None,
 530 ) -> None:
 531     """
 532     Write records stored in a DataFrame to a SQL database.
 533 
 534     Parameters
 535     ----------
 536     frame : DataFrame, Series
 537     name : str
 538         Name of SQL table.
 539     con : SQLAlchemy connectable(engine/connection) or database string URI
 540         or sqlite3 DBAPI2 connection
 541         Using SQLAlchemy makes it possible to use any DB supported by that
 542         library.
 543         If a DBAPI2 object, only sqlite3 is supported.
 544     schema : str, optional
 545         Name of SQL schema in database to write to (if database flavor
 546         supports this). If None, use default schema (default).
 547     if_exists : {'fail', 'replace', 'append'}, default 'fail'
 548         - fail: If table exists, do nothing.
 549         - replace: If table exists, drop it, recreate it, and insert data.
 550         - append: If table exists, insert data. Create if does not exist.
 551     index : boolean, default True
 552         Write DataFrame index as a column.
 553     index_label : str or sequence, optional
 554         Column label for index column(s). If None is given (default) and
 555         `index` is True, then the index names are used.
 556         A sequence should be given if the DataFrame uses MultiIndex.
 557     chunksize : int, optional
 558         Specify the number of rows in each batch to be written at a time.
 559         By default, all rows will be written at once.
 560     dtype : dict or scalar, optional
 561         Specifying the datatype for columns. If a dictionary is used, the
 562         keys should be the column names and the values should be the
 563         SQLAlchemy types or strings for the sqlite3 fallback mode. If a
 564         scalar is provided, it will be applied to all columns.
 565     method : {None, 'multi', callable}, optional
 566         Controls the SQL insertion clause used:
 567 
 568         - None : Uses standard SQL ``INSERT`` clause (one per row).
 569         - 'multi': Pass multiple values in a single ``INSERT`` clause.
 570         - callable with signature ``(pd_table, conn, keys, data_iter)``.
 571 
 572         Details and a sample callable implementation can be found in the
 573         section :ref:`insert method <io.sql.method>`.
 574 
 575         .. versionadded:: 0.24.0
 576     """
 577     if if_exists not in ("fail", "replace", "append"):
 578         raise ValueError(f"'{if_exists}' is not valid for if_exists")
 579 
 580     pandas_sql = pandasSQL_builder(con, schema=schema)
 581 
 582     if isinstance(frame, Series):
 583         frame = frame.to_frame()
 584     elif not isinstance(frame, DataFrame):
 585         raise NotImplementedError(
 586             "'frame' argument should be either a Series or a DataFrame"
 587         )
 588 
 589     pandas_sql.to_sql(
 590         frame,
 591         name,
 592         if_exists=if_exists,
 593         index=index,
 594         index_label=index_label,
 595         schema=schema,
 596         chunksize=chunksize,
 597         dtype=dtype,
 598         method=method,
 599     )

 796     def insert(self, chunksize=None, method=None):
 797 
 798         # set insert method
 799         if method is None:
 800             exec_insert = self._execute_insert
 801         elif method == "multi":
 802             exec_insert = self._execute_insert_multi
 803         elif callable(method):
 804             exec_insert = partial(method, self)
 805         else:
 806             raise ValueError(f"Invalid parameter `method`: {method}")
 807 
 808         keys, data_list = self.insert_data()
 809 
 810         nrows = len(self.frame)
 811 
 812         if nrows == 0:
 813             return
 814 
 815         if chunksize is None:
 816             chunksize = nrows
 817         elif chunksize == 0:
 818             raise ValueError("chunksize argument should be non-zero")
 819 
 820         chunks = int(nrows / chunksize) + 1
 821 
 822         with self.pd_sql.run_transaction() as conn:
 823             for i in range(chunks):
 824                 start_i = i * chunksize

 826                 if start_i >= end_i:
 827                     break

 829                 chunk_iter = zip(*[arr[start_i:end_i] for arr in data_list])

 967     def _harmonize_columns(self, parse_dates=None):
 968         """
 969         Make the DataFrame's column types align with the SQL table
 970         column types.
 971         Need to work around limited NA value support. Floats are always
 972         fine, ints must always be floats if there are Null values.
 973         Booleans are hard because converting bool column with None replaces
 974         all Nones with false. Therefore only convert bool if there are no
 975         NA values.
 976         Datetimes should already be converted to np.datetime64 if supported,
 977         but here we also force conversion if required.
 978         """
 979         parse_dates = _process_parse_dates_argument(parse_dates)
 980 
 981         for sql_col in self.table.columns:
 982             col_name = sql_col.name
 983             try:
 984                 df_col = self.frame[col_name]
 985 
 986                 # Handle date parsing upfront; don't try to convert columns
 987                 # twice
 988                 if col_name in parse_dates:
 989                     try:
 990                         fmt = parse_dates[col_name]
 991                     except TypeError:
 992                         fmt = None
 993                     self.frame[col_name] = _handle_date_column(df_col, format=fmt)
 994                     continue
 995 
 996                 # the type the dataframe column should have
 997                 col_type = self._get_dtype(sql_col.type)
 998 
 999                 if (
1000                     col_type is datetime
1001                     or col_type is date
1002                     or col_type is DatetimeTZDtype
1003                 ):
1004                     # Convert tz-aware Datetime SQL columns to UTC
1005                     utc = col_type is DatetimeTZDtype
1006                     self.frame[col_name] = _handle_date_column(df_col, utc=utc)
1007                 elif col_type is float:
1008                     # floats support NA, can always convert!
1009                     self.frame[col_name] = df_col.astype(col_type, copy=False)
1010 
1011                 elif len(df_col) == df_col.count():
1012                     # No NA values, can convert ints and bools
1013                     if col_type is np.dtype("int64") or col_type is bool:
1014                         self.frame[col_name] = df_col.astype(col_type, copy=False)
1015             except KeyError:
1016                 pass  # this column not in results

1018     def _sqlalchemy_type(self, col):
1019 
1020         dtype = self.dtype or {}
1021         if col.name in dtype:
1022             return self.dtype[col.name]
1023 
1024         # Infer type of column, while ignoring missing values.
1025         # Needed for inserting typed data containing NULLs, GH 8778.
1026         col_type = lib.infer_dtype(col, skipna=True)
1027 
1028         from sqlalchemy.types import (
1029             BigInteger,
1030             Integer,
1031             Float,
1032             Text,
1033             Boolean,
1034             DateTime,
1035             Date,
1036             Time,
1037             TIMESTAMP,
1038         )
1039 
1040         if col_type == "datetime64" or col_type == "datetime":
1041             # GH 9086: TIMESTAMP is the suggested type if the column contains
1042             # timezone information
1043             try:
1044                 if col.dt.tz is not None:
1045                     return TIMESTAMP(timezone=True)
1046             except AttributeError:
1047                 # The column is actually a DatetimeIndex
1048                 # GH 26761 or an Index with date-like data e.g. 9999-01-01
1049                 if getattr(col, "tz", None) is not None:
1050                     return TIMESTAMP(timezone=True)
1051             return DateTime
1052         if col_type == "timedelta64":
1053             warnings.warn(
1054                 "the 'timedelta' type is not supported, and will be "
1055                 "written as integer values (ns frequency) to the database.",
1056                 UserWarning,
1057                 stacklevel=8,
1058             )
1059             return BigInteger
1060         elif col_type == "floating":
1061             if col.dtype == "float32":
1062                 return Float(precision=23)
1063             else:
1064                 return Float(precision=53)
1065         elif col_type == "integer":
1066             if col.dtype == "int32":
1067                 return Integer
1068             else:
1069                 return BigInteger
1070         elif col_type == "boolean":
1071             return Boolean
1072         elif col_type == "date":
1073             return Date
1074         elif col_type == "time":
1075             return Time
1076         elif col_type == "complex":
1077             raise ValueError("Complex datatypes not supported")
1078 
1079         return Text

1081     def _get_dtype(self, sqltype):
1082         from sqlalchemy.types import Integer, Float, Boolean, DateTime, Date, TIMESTAMP
1083 
1084         if isinstance(sqltype, Float):
1085             return float
1086         elif isinstance(sqltype, Integer):
1087             # TODO: Refine integer size.
1088             return np.dtype("int64")
1089         elif isinstance(sqltype, TIMESTAMP):
1090             # we have a timezone capable type
1091             if not sqltype.timezone:
1092                 return datetime
1093             return DatetimeTZDtype
1094         elif isinstance(sqltype, DateTime):
1095             # Caution: np.datetime64 is also a subclass of np.number.
1096             return datetime
1097         elif isinstance(sqltype, Date):
1098             return date
1099         elif isinstance(sqltype, Boolean):
