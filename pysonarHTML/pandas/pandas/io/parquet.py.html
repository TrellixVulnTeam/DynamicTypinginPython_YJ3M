  19 def get_engine(engine: str) -> "BaseImpl":
  20     """ return our implementation """
  21     if engine == "auto":
  22         engine = get_option("io.parquet.engine")
  23 
  24     if engine == "auto":
  25         # try engines in this order
  26         engine_classes = [PyArrowImpl, FastParquetImpl]
  27 
  28         error_msgs = ""
  29         for engine_class in engine_classes:
  30             try:
  31                 return engine_class()
  32             except ImportError as err:
  33                 error_msgs += "\n - " + str(err)
  34 
  35         raise ImportError(
  36             "Unable to find a usable engine; "
  37             "tried using: 'pyarrow', 'fastparquet'.\n"
  38             "A suitable version of "
  39             "pyarrow or fastparquet is required for parquet "
  40             "support.\n"
  41             "Trying to import the above resulted in these errors:"
  42             f"{error_msgs}"
  43         )
  44 
  45     if engine == "pyarrow":
  46         return PyArrowImpl()
  47     elif engine == "fastparquet":
  48         return FastParquetImpl()
  49 
  50     raise ValueError("engine must be one of 'pyarrow', 'fastparquet'")

  53 class BaseImpl:
  54     @staticmethod
  55     def validate_dataframe(df: DataFrame):
  56 
  57         if not isinstance(df, DataFrame):
  58             raise ValueError("to_parquet only supports IO with DataFrames")
  59 
  60         # must have value column names (strings only)
  61         if df.columns.inferred_type not in {"string", "empty"}:
  62             raise ValueError("parquet must have string column names")
  63 
  64         # index level names must be strings
  65         valid_names = all(
  66             isinstance(name, str) for name in df.index.names if name is not None
  67         )
  68         if not valid_names:
  69             raise ValueError("Index level names must be strings")
  70 
  71     def write(self, df: DataFrame, path, compression, **kwargs):
  72         raise AbstractMethodError(self)
  73 
  74     def read(self, path, columns=None, **kwargs):
  75         raise AbstractMethodError(self)

  78 class PyArrowImpl(BaseImpl):
  79     def __init__(self):
  80         import_optional_dependency(
  81             "pyarrow", extra="pyarrow is required for parquet support."
  82         )
  83         import pyarrow.parquet
  84 
  85         # import utils to register the pyarrow extension types
  86         import pandas.core.arrays._arrow_utils  # noqa
  87 
  88         self.api = pyarrow
  89 
  90     def write(
  91         self,
  92         df: DataFrame,
  93         path,
  94         compression="snappy",
  95         index: Optional[bool] = None,
  96         partition_cols=None,
  97         **kwargs,
  98     ):
  99         self.validate_dataframe(df)
 100         file_obj_or_path, _, _, should_close = get_filepath_or_buffer(path, mode="wb")
 101 
 102         from_pandas_kwargs: Dict[str, Any] = {"schema": kwargs.pop("schema", None)}
 103         if index is not None:
 104             from_pandas_kwargs["preserve_index"] = index
 105 
 106         table = self.api.Table.from_pandas(df, **from_pandas_kwargs)
 107         # write_to_dataset does not support a file-like object when
 108         # a directory path is used, so just pass the path string.
 109         if partition_cols is not None:
 110             self.api.parquet.write_to_dataset(
 111                 table,
 112                 path,
 113                 compression=compression,
 114                 partition_cols=partition_cols,
 115                 **kwargs,
 116             )
 117         else:
 118             self.api.parquet.write_table(
 119                 table, file_obj_or_path, compression=compression, **kwargs
 120             )
 121         if should_close:
 122             file_obj_or_path.close()
 123 
 124     def read(self, path, columns=None, **kwargs):
 125         fs = get_fs_for_path(path)
 126         should_close = None
 127         # Avoid calling get_filepath_or_buffer for s3/gcs URLs since
 128         # since it returns an S3File which doesn't support dir reads in arrow
 129         if not fs:
 130             path, _, _, should_close = get_filepath_or_buffer(path)
 131 
 132         kwargs["use_pandas_metadata"] = True
 133         result = self.api.parquet.read_table(
 134             path, columns=columns, filesystem=fs, **kwargs
 135         ).to_pandas()
 136         if should_close:
 137             path.close()
 138 
 139         return result

 142 class FastParquetImpl(BaseImpl):
 143     def __init__(self):
 144         # since pandas is a dependency of fastparquet
 145         # we need to import on first use
 146         fastparquet = import_optional_dependency(
 147             "fastparquet", extra="fastparquet is required for parquet support."
 148         )
 149         self.api = fastparquet
 150 
 151     def write(
 152         self,
 153         df: DataFrame,
 154         path,
 155         compression="snappy",
 156         index=None,
 157         partition_cols=None,
 158         **kwargs,
 159     ):
 160         self.validate_dataframe(df)
 161         # thriftpy/protocol/compact.py:339:
 162         # DeprecationWarning: tostring() is deprecated.
 163         # Use tobytes() instead.
 164 
 165         if "partition_on" in kwargs and partition_cols is not None:
 166             raise ValueError(
 167                 "Cannot use both partition_on and "
 168                 "partition_cols. Use partition_cols for partitioning data"
 169             )
 170         elif "partition_on" in kwargs:
 171             partition_cols = kwargs.pop("partition_on")
 172 
 173         if partition_cols is not None:
 174             kwargs["file_scheme"] = "hive"
 175 
 176         if is_s3_url(path) or is_gcs_url(path):
 177             # if path is s3:// or gs:// we need to open the file in 'wb' mode.
 178             # TODO: Support 'ab'
 179 
 180             path, _, _, _ = get_filepath_or_buffer(path, mode="wb")
 181             # And pass the opened file to the fastparquet internal impl.
 182             kwargs["open_with"] = lambda path, _: path
 183         else:
 184             path, _, _, _ = get_filepath_or_buffer(path)
 185 
 186         with catch_warnings(record=True):
 187             self.api.write(
 188                 path,
 189                 df,
 190                 compression=compression,
 191                 write_index=index,
 192                 partition_on=partition_cols,
 193                 **kwargs,
 194             )
 195 
 196     def read(self, path, columns=None, **kwargs):
 197         if is_s3_url(path):
 198             from pandas.io.s3 import get_file_and_filesystem
 199 
 200             # When path is s3:// an S3File is returned.
 201             # We need to retain the original path(str) while also
 202             # pass the S3File().open function to fastparquet impl.
 203             s3, filesystem = get_file_and_filesystem(path)
 204             try:
 205                 parquet_file = self.api.ParquetFile(path, open_with=filesystem.open)
 206             finally:
 207                 s3.close()
 208         else:
 209             path, _, _, _ = get_filepath_or_buffer(path)
 210             parquet_file = self.api.ParquetFile(path)
 211 
 212         return parquet_file.to_pandas(columns=columns, **kwargs)

 215 def to_parquet(
 216     df: DataFrame,
 217     path,
 218     engine: str = "auto",
 219     compression="snappy",
 220     index: Optional[bool] = None,
 221     partition_cols=None,
 222     **kwargs,
 223 ):
 224     """
 225     Write a DataFrame to the parquet format.
 226 
 227     Parameters
 228     ----------
 229     df : DataFrame
 230     path : str
 231         File path or Root Directory path. Will be used as Root Directory path
 232         while writing a partitioned dataset.
 233 
 234         .. versionchanged:: 0.24.0
 235 
 236     engine : {'auto', 'pyarrow', 'fastparquet'}, default 'auto'
 237         Parquet library to use. If 'auto', then the option
 238         ``io.parquet.engine`` is used. The default ``io.parquet.engine``
 239         behavior is to try 'pyarrow', falling back to 'fastparquet' if
 240         'pyarrow' is unavailable.
 241     compression : {'snappy', 'gzip', 'brotli', None}, default 'snappy'
 242         Name of the compression to use. Use ``None`` for no compression.
 243     index : bool, default None
 244         If ``True``, include the dataframe's index(es) in the file output. If
 245         ``False``, they will not be written to the file.
 246         If ``None``, similar to ``True`` the dataframe's index(es)
 247         will be saved. However, instead of being saved as values,
 248         the RangeIndex will be stored as a range in the metadata so it
 249         doesn't require much space and is faster. Other indexes will
 250         be included as columns in the file output.
 251 
 252         .. versionadded:: 0.24.0
 253 
 254     partition_cols : str or list, optional, default None
 255         Column names by which to partition the dataset
 256         Columns are partitioned in the order they are given
 257 
 258         .. versionadded:: 0.24.0
 259 
 260     kwargs
 261         Additional keyword arguments passed to the engine
 262     """
 263     if isinstance(partition_cols, str):
 264         partition_cols = [partition_cols]
 265     impl = get_engine(engine)
 266     return impl.write(
 267         df,
 268         path,
 269         compression=compression,
 270         index=index,
 271         partition_cols=partition_cols,
 272         **kwargs,
 273     )

 276 def read_parquet(path, engine: str = "auto", columns=None, **kwargs):
 277     """
 278     Load a parquet object from the file path, returning a DataFrame.
 279 
 280     Parameters
 281     ----------
 282     path : str, path object or file-like object
 283         Any valid string path is acceptable. The string could be a URL. Valid
 284         URL schemes include http, ftp, s3, and file. For file URLs, a host is
 285         expected. A local file could be:
 286         ``file://localhost/path/to/table.parquet``.
 287         A file URL can also be a path to a directory that contains multiple
 288         partitioned parquet files. Both pyarrow and fastparquet support
 289         paths to directories as well as file URLs. A directory path could be:
 290         ``file://localhost/path/to/tables`` or ``s3://bucket/partition_dir``
 291 
 292         If you want to pass in a path object, pandas accepts any
 293         ``os.PathLike``.
 294 
 295         By file-like object, we refer to objects with a ``read()`` method,
 296         such as a file handler (e.g. via builtin ``open`` function)
 297         or ``StringIO``.
 298     engine : {'auto', 'pyarrow', 'fastparquet'}, default 'auto'
 299         Parquet library to use. If 'auto', then the option
 300         ``io.parquet.engine`` is used. The default ``io.parquet.engine``
 301         behavior is to try 'pyarrow', falling back to 'fastparquet' if
 302         'pyarrow' is unavailable.
 303     columns : list, default=None
 304         If not None, only these columns will be read from the file.
 305     **kwargs
 306         Any additional kwargs are passed to the engine.
 307 
 308     Returns
 309     -------
 310     DataFrame
 311     """
 312     impl = get_engine(engine)
 313     return impl.read(path, columns=columns, **kwargs)
