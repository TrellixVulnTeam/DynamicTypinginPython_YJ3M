  34 def as_json_table_type(x: DtypeObj) -> str:
  35     """
  36     Convert a NumPy / pandas type to its corresponding json_table.
  37 
  38     Parameters
  39     ----------
  40     x : np.dtype or ExtensionDtype
  41 
  42     Returns
  43     -------
  44     str
  45         the Table Schema data types
  46 
  47     Notes
  48     -----
  49     This table shows the relationship between NumPy / pandas dtypes,
  50     and Table Schema dtypes.
  51 
  52     ==============  =================
  53     Pandas type     Table Schema type
  54     ==============  =================
  55     int64           integer
  56     float64         number
  57     bool            boolean
  58     datetime64[ns]  datetime
  59     timedelta64[ns] duration
  60     object          str
  61     categorical     any
  62     =============== =================
  63     """
  64     if is_integer_dtype(x):
  65         return "integer"
  66     elif is_bool_dtype(x):
  67         return "boolean"
  68     elif is_numeric_dtype(x):
  69         return "number"
  70     elif is_datetime64_dtype(x) or is_datetime64tz_dtype(x) or is_period_dtype(x):
  71         return "datetime"
  72     elif is_timedelta64_dtype(x):
  73         return "duration"
  74     elif is_categorical_dtype(x):
  75         return "any"
  76     elif is_string_dtype(x):
  77         return "string"
  78     else:
  79         return "any"

  82 def set_default_names(data):
  83     """Sets index names to 'index' for regular, or 'level_x' for Multi"""
  84     if com.all_not_none(*data.index.names):
  85         nms = data.index.names
  86         if len(nms) == 1 and data.index.name == "index":
  87             warnings.warn("Index name of 'index' is not round-trippable")
  88         elif len(nms) > 1 and any(x.startswith("level_") for x in nms):
  89             warnings.warn("Index names beginning with 'level_' are not round-trippable")
  90         return data
  91 
  92     data = data.copy()
  93     if data.index.nlevels > 1:
  94         names = [
  95             name if name is not None else f"level_{i}"
  96             for i, name in enumerate(data.index.names)
  97         ]
  98         data.index.names = names
  99     else:
 100         data.index.name = data.index.name or "index"
 101     return data

 104 def convert_pandas_type_to_json_field(arr):
 105     dtype = arr.dtype
 106     if arr.name is None:
 107         name = "values"
 108     else:
 109         name = arr.name
 110     field: Dict[str, JSONSerializable] = {
 111         "name": name,
 112         "type": as_json_table_type(dtype),
 113     }
 114 
 115     if is_categorical_dtype(dtype):
 116         cats = dtype.categories
 117         ordered = dtype.ordered
 118 
 119         field["constraints"] = {"enum": list(cats)}
 120         field["ordered"] = ordered
 121     elif is_period_dtype(dtype):
 122         field["freq"] = dtype.freq.freqstr
 123     elif is_datetime64tz_dtype(dtype):
 124         field["tz"] = dtype.tz.zone
 125     return field

 128 def convert_json_field_to_pandas_type(field):
 129     """
 130     Converts a JSON field descriptor into its corresponding NumPy / pandas type
 131 
 132     Parameters
 133     ----------
 134     field
 135         A JSON field descriptor
 136 
 137     Returns
 138     -------
 139     dtype
 140 
 141     Raises
 142     ------
 143     ValueError
 144         If the type of the provided field is unknown or currently unsupported
 145 
 146     Examples
 147     --------
 148     >>> convert_json_field_to_pandas_type({'name': 'an_int',
 149                                            'type': 'integer'})
 150     'int64'
 151     >>> convert_json_field_to_pandas_type({'name': 'a_categorical',
 152                                            'type': 'any',
 153                                            'constraints': {'enum': [
 154                                                           'a', 'b', 'c']},
 155                                            'ordered': True})
 156     'CategoricalDtype(categories=['a', 'b', 'c'], ordered=True)'
 157     >>> convert_json_field_to_pandas_type({'name': 'a_datetime',
 158                                            'type': 'datetime'})
 159     'datetime64[ns]'
 160     >>> convert_json_field_to_pandas_type({'name': 'a_datetime_with_tz',
 161                                            'type': 'datetime',
 162                                            'tz': 'US/Central'})
 163     'datetime64[ns, US/Central]'
 164     """
 165     typ = field["type"]
 166     if typ == "string":
 167         return "object"
 168     elif typ == "integer":
 169         return "int64"
 170     elif typ == "number":
 171         return "float64"
 172     elif typ == "boolean":
 173         return "bool"
 174     elif typ == "duration":
 175         return "timedelta64"
 176     elif typ == "datetime":
 177         if field.get("tz"):
 178             return f"datetime64[ns, {field['tz']}]"
 179         else:
 180             return "datetime64[ns]"
 181     elif typ == "any":
 182         if "constraints" in field and "ordered" in field:
 183             return CategoricalDtype(
 184                 categories=field["constraints"]["enum"], ordered=field["ordered"]
 185             )
 186         else:
 187             return "object"
 188 
 189     raise ValueError(f"Unsupported or invalid field type: {typ}")

 192 def build_table_schema(
 193     data: FrameOrSeries,
 194     index: bool = True,
 195     primary_key: Optional[bool] = None,
 196     version: bool = True,
 197 ) -> Dict[str, JSONSerializable]:
 198     """
 199     Create a Table schema from ``data``.
 200 
 201     Parameters
 202     ----------
 203     data : Series, DataFrame
 204     index : bool, default True
 205         Whether to include ``data.index`` in the schema.
 206     primary_key : bool or None, default True
 207         Column names to designate as the primary key.
 208         The default `None` will set `'primaryKey'` to the index
 209         level or levels if the index is unique.
 210     version : bool, default True
 211         Whether to include a field `pandas_version` with the version
 212         of pandas that generated the schema.
 213 
 214     Returns
 215     -------
 216     schema : dict
 217 
 218     Notes
 219     -----
 220     See `Table Schema
 221     <https://pandas.pydata.org/docs/user_guide/io.html#table-schema>`__ for
 222     conversion types.
 223     Timedeltas as converted to ISO8601 duration format with
 224     9 decimal places after the seconds field for nanosecond precision.
 225 
 226     Categoricals are converted to the `any` dtype, and use the `enum` field
 227     constraint to list the allowed values. The `ordered` attribute is included
 228     in an `ordered` field.
 229 
 230     Examples
 231     --------
 232     >>> df = pd.DataFrame(
 233     ...     {'A': [1, 2, 3],
 234     ...      'B': ['a', 'b', 'c'],
 235     ...      'C': pd.date_range('2016-01-01', freq='d', periods=3),
 236     ...     }, index=pd.Index(range(3), name='idx'))
 237     >>> build_table_schema(df)
 238     {'fields': [{'name': 'idx', 'type': 'integer'},
 239     {'name': 'A', 'type': 'integer'},
 240     {'name': 'B', 'type': 'string'},
 241     {'name': 'C', 'type': 'datetime'}],
 242     'pandas_version': '0.20.0',
 243     'primaryKey': ['idx']}
 244     """
 245     if index is True:
 246         data = set_default_names(data)
 247 
 248     schema: Dict[str, Any] = {}
 249     fields = []
 250 
 251     if index:
 252         if data.index.nlevels > 1:
 253             data.index = cast("MultiIndex", data.index)
 254             for level, name in zip(data.index.levels, data.index.names):
 255                 new_field = convert_pandas_type_to_json_field(level)
 256                 new_field["name"] = name
 257                 fields.append(new_field)
 258         else:
 259             fields.append(convert_pandas_type_to_json_field(data.index))
 260 
 261     if data.ndim > 1:
 262         for column, s in data.items():
 263             fields.append(convert_pandas_type_to_json_field(s))
 264     else:
 265         fields.append(convert_pandas_type_to_json_field(data))
 266 
 267     schema["fields"] = fields
 268     if index and data.index.is_unique and primary_key is None:
 269         if data.index.nlevels == 1:
 270             schema["primaryKey"] = [data.index.name]
 271         else:
 272             schema["primaryKey"] = data.index.names
 273     elif primary_key is not None:
 274         schema["primaryKey"] = primary_key
 275 
 276     if version:
 277         schema["pandas_version"] = "0.20.0"
 278     return schema

 281 def parse_table_schema(json, precise_float):
 282     """
 283     Builds a DataFrame from a given schema
 284 
 285     Parameters
 286     ----------
 287     json :
 288         A JSON table schema
 289     precise_float : boolean
 290         Flag controlling precision when decoding string to double values, as
 291         dictated by ``read_json``
 292 
 293     Returns
 294     -------
 295     df : DataFrame
 296 
 297     Raises
 298     ------
 299     NotImplementedError
 300         If the JSON table schema contains either timezone or timedelta data
 301 
 302     Notes
 303     -----
 304         Because :func:`DataFrame.to_json` uses the string 'index' to denote a
 305         name-less :class:`Index`, this function sets the name of the returned
 306         :class:`DataFrame` to ``None`` when said string is encountered with a
 307         normal :class:`Index`. For a :class:`MultiIndex`, the same limitation
 308         applies to any strings beginning with 'level_'. Therefore, an
 309         :class:`Index` name of 'index'  and :class:`MultiIndex` names starting
 310         with 'level_' are not supported.
 311 
 312     See Also
 313     --------
 314     build_table_schema : Inverse function.
 315     pandas.read_json
 316     """
 317     table = loads(json, precise_float=precise_float)
 318     col_order = [field["name"] for field in table["schema"]["fields"]]
 319     df = DataFrame(table["data"], columns=col_order)[col_order]
 320 
 321     dtypes = {
 322         field["name"]: convert_json_field_to_pandas_type(field)
 323         for field in table["schema"]["fields"]
 324     }
 325 
 326     # Cannot directly use as_type with timezone data on object; raise for now
 327     if any(str(x).startswith("datetime64[ns, ") for x in dtypes.values()):
 328         raise NotImplementedError('table="orient" can not yet read timezone data')
 329 
 330     # No ISO constructor for Timedelta as of yet, so need to raise
 331     if "timedelta64" in dtypes.values():
 332         raise NotImplementedError(
 333             'table="orient" can not yet read ISO-formatted Timedelta data'
 334         )
 335 
 336     df = df.astype(dtypes)
 337 
 338     if "primaryKey" in table["schema"]:
 339         df = df.set_index(table["schema"]["primaryKey"])
 340         if len(df.index.names) == 1:
 341             if df.index.name == "index":
 342                 df.index.name = None
 343         else:
 344             df.index.names = [
 345                 None if x.startswith("level_") else x for x in df.index.names
 346             ]
 347 
 348     return df
