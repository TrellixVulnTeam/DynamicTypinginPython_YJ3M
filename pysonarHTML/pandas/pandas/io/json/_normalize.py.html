  18 def convert_to_line_delimits(s):
  19     """
  20     Helper function that converts JSON lists to line delimited JSON.
  21     """
  22     # Determine we have a JSON list to turn to lines otherwise just return the
  23     # json object, only lists can
  24     if not s[0] == "[" and s[-1] == "]":
  25         return s
  26     s = s[1:-1]
  27 
  28     return convert_json_to_lines(s)

  31 def nested_to_record(
  32     ds,
  33     prefix: str = "",
  34     sep: str = ".",
  35     level: int = 0,
  36     max_level: Optional[int] = None,
  37 ):
  38     """
  39     A simplified json_normalize
  40 
  41     Converts a nested dict into a flat dict ("record"), unlike json_normalize,
  42     it does not attempt to extract a subset of the data.
  43 
  44     Parameters
  45     ----------
  46     ds : dict or list of dicts
  47     prefix: the prefix, optional, default: ""
  48     sep : str, default '.'
  49         Nested records will generate names separated by sep,
  50         e.g., for sep='.', { 'foo' : { 'bar' : 0 } } -> foo.bar
  51     level: int, optional, default: 0
  52         The number of levels in the json string.
  53 
  54     max_level: int, optional, default: None
  55         The max depth to normalize.
  56 
  57         .. versionadded:: 0.25.0
  58 
  59     Returns
  60     -------
  61     d - dict or list of dicts, matching `ds`
  62 
  63     Examples
  64     --------
  65     IN[52]: nested_to_record(dict(flat1=1,dict1=dict(c=1,d=2),
  66                                   nested=dict(e=dict(c=1,d=2),d=2)))
  67     Out[52]:
  68     {'dict1.c': 1,
  69      'dict1.d': 2,
  70      'flat1': 1,
  71      'nested.d': 2,
  72      'nested.e.c': 1,
  73      'nested.e.d': 2}
  74     """
  75     singleton = False
  76     if isinstance(ds, dict):
  77         ds = [ds]
  78         singleton = True
  79     new_ds = []
  80     for d in ds:
  81         new_d = copy.deepcopy(d)
  82         for k, v in d.items():
  83             # each key gets renamed with prefix
  84             if not isinstance(k, str):
  85                 k = str(k)
  86             if level == 0:
  87                 newkey = k
  88             else:
  89                 newkey = prefix + sep + k
  90 
  91             # flatten if type is dict and
  92             # current dict level  < maximum level provided and
  93             # only dicts gets recurse-flattened
  94             # only at level>1 do we rename the rest of the keys
  95             if not isinstance(v, dict) or (
  96                 max_level is not None and level >= max_level
  97             ):
  98                 if level != 0:  # so we skip copying for top level, common case
  99                     v = new_d.pop(k)
 100                     new_d[newkey] = v
 101                 continue
 102             else:
 103                 v = new_d.pop(k)
 104                 new_d.update(nested_to_record(v, newkey, sep, level + 1, max_level))
 105         new_ds.append(new_d)
 106 
 107     if singleton:
 108         return new_ds[0]
 109     return new_ds

 112 def _json_normalize(
 113     data: Union[Dict, List[Dict]],
 114     record_path: Optional[Union[str, List]] = None,
 115     meta: Optional[Union[str, List[Union[str, List[str]]]]] = None,
 116     meta_prefix: Optional[str] = None,
 117     record_prefix: Optional[str] = None,
 118     errors: str = "raise",
 119     sep: str = ".",
 120     max_level: Optional[int] = None,
 121 ) -> "DataFrame":
 122     """
 123     Normalize semi-structured JSON data into a flat table.
 124 
 125     Parameters
 126     ----------
 127     data : dict or list of dicts
 128         Unserialized JSON objects.
 129     record_path : str or list of str, default None
 130         Path in each object to list of records. If not passed, data will be
 131         assumed to be an array of records.
 132     meta : list of paths (str or list of str), default None
 133         Fields to use as metadata for each record in resulting table.
 134     meta_prefix : str, default None
 135         If True, prefix records with dotted (?) path, e.g. foo.bar.field if
 136         meta is ['foo', 'bar'].
 137     record_prefix : str, default None
 138         If True, prefix records with dotted (?) path, e.g. foo.bar.field if
 139         path to records is ['foo', 'bar'].
 140     errors : {'raise', 'ignore'}, default 'raise'
 141         Configures error handling.
 142 
 143         * 'ignore' : will ignore KeyError if keys listed in meta are not
 144           always present.
 145         * 'raise' : will raise KeyError if keys listed in meta are not
 146           always present.
 147     sep : str, default '.'
 148         Nested records will generate names separated by sep.
 149         e.g., for sep='.', {'foo': {'bar': 0}} -> foo.bar.
 150     max_level : int, default None
 151         Max number of levels(depth of dict) to normalize.
 152         if None, normalizes all levels.
 153 
 154         .. versionadded:: 0.25.0
 155 
 156     Returns
 157     -------
 158     frame : DataFrame
 159     Normalize semi-structured JSON data into a flat table.
 160 
 161     Examples
 162     --------
 163     >>> data = [{'id': 1, 'name': {'first': 'Coleen', 'last': 'Volk'}},
 164     ...         {'name': {'given': 'Mose', 'family': 'Regner'}},
 165     ...         {'id': 2, 'name': 'Faye Raker'}]
 166     >>> pandas.json_normalize(data)
 167         id        name name.family name.first name.given name.last
 168     0  1.0         NaN         NaN     Coleen        NaN      Volk
 169     1  NaN         NaN      Regner        NaN       Mose       NaN
 170     2  2.0  Faye Raker         NaN        NaN        NaN       NaN
 171 
 172     >>> data = [{'id': 1,
 173     ...          'name': "Cole Volk",
 174     ...          'fitness': {'height': 130, 'weight': 60}},
 175     ...         {'name': "Mose Reg",
 176     ...          'fitness': {'height': 130, 'weight': 60}},
 177     ...         {'id': 2, 'name': 'Faye Raker',
 178     ...          'fitness': {'height': 130, 'weight': 60}}]
 179     >>> json_normalize(data, max_level=0)
 180                 fitness                 id        name
 181     0   {'height': 130, 'weight': 60}  1.0   Cole Volk
 182     1   {'height': 130, 'weight': 60}  NaN    Mose Reg
 183     2   {'height': 130, 'weight': 60}  2.0  Faye Raker
 184 
 185     Normalizes nested data up to level 1.
 186 
 187     >>> data = [{'id': 1,
 188     ...          'name': "Cole Volk",
 189     ...          'fitness': {'height': 130, 'weight': 60}},
 190     ...         {'name': "Mose Reg",
 191     ...          'fitness': {'height': 130, 'weight': 60}},
 192     ...         {'id': 2, 'name': 'Faye Raker',
 193     ...          'fitness': {'height': 130, 'weight': 60}}]
 194     >>> json_normalize(data, max_level=1)
 195       fitness.height  fitness.weight   id    name
 196     0   130              60          1.0    Cole Volk
 197     1   130              60          NaN    Mose Reg
 198     2   130              60          2.0    Faye Raker
 199 
 200     >>> data = [{'state': 'Florida',
 201     ...          'shortname': 'FL',
 202     ...          'info': {'governor': 'Rick Scott'},
 203     ...          'counties': [{'name': 'Dade', 'population': 12345},
 204     ...                       {'name': 'Broward', 'population': 40000},
 205     ...                       {'name': 'Palm Beach', 'population': 60000}]},
 206     ...         {'state': 'Ohio',
 207     ...          'shortname': 'OH',
 208     ...          'info': {'governor': 'John Kasich'},
 209     ...          'counties': [{'name': 'Summit', 'population': 1234},
 210     ...                       {'name': 'Cuyahoga', 'population': 1337}]}]
 211     >>> result = json_normalize(data, 'counties', ['state', 'shortname',
 212     ...                                            ['info', 'governor']])
 213     >>> result
 214              name  population    state shortname info.governor
 215     0        Dade       12345   Florida    FL    Rick Scott
 216     1     Broward       40000   Florida    FL    Rick Scott
 217     2  Palm Beach       60000   Florida    FL    Rick Scott
 218     3      Summit        1234   Ohio       OH    John Kasich
 219     4    Cuyahoga        1337   Ohio       OH    John Kasich
 220 
 221     >>> data = {'A': [1, 2]}
 222     >>> json_normalize(data, 'A', record_prefix='Prefix.')
 223         Prefix.0
 224     0          1
 225     1          2
 226 
 227     Returns normalized data with columns prefixed with the given string.
 228     """
 229 
 230     def _pull_field(
 231         js: Dict[str, Any], spec: Union[List, str]
 232     ) -> Union[Scalar, Iterable]:
 233         """Internal function to pull field"""
 234         result = js
 235         if isinstance(spec, list):
 236             for field in spec:
 237                 result = result[field]
 238         else:
 239             result = result[spec]
 240         return result
 241 
 242     def _pull_records(js: Dict[str, Any], spec: Union[List, str]) -> List:
 243         """
 244         Internal function to pull field for records, and similar to
 245         _pull_field, but require to return list. And will raise error
 246         if has non iterable value.
 247         """
 248         result = _pull_field(js, spec)
 249 
 250         # GH 31507 GH 30145, GH 26284 if result is not list, raise TypeError if not
 251         # null, otherwise return an empty list
 252         if not isinstance(result, list):
 253             if pd.isnull(result):
 254                 result = []
 255             else:
 256                 raise TypeError(
 257                     f"{js} has non list value {result} for path {spec}. "
 258                     "Must be list or null."
 259                 )
 260         return result
 261 
 262     if isinstance(data, list) and not data:
 263         return DataFrame()
 264 
 265     # A bit of a hackjob
 266     if isinstance(data, dict):
 267         data = [data]
 268 
 269     if record_path is None:
 270         if any([isinstance(x, dict) for x in y.values()] for y in data):
 271             # naive normalization, this is idempotent for flat records
 272             # and potentially will inflate the data considerably for
 273             # deeply nested structures:
 274             #  {VeryLong: { b: 1,c:2}} -> {VeryLong.b:1 ,VeryLong.c:@}
 275             #
 276             # TODO: handle record value which are lists, at least error
 277             #       reasonably
 278             data = nested_to_record(data, sep=sep, max_level=max_level)
 279         return DataFrame(data)
 280     elif not isinstance(record_path, list):
 281         record_path = [record_path]
 282 
 283     if meta is None:
 284         meta = []
 285     elif not isinstance(meta, list):
 286         meta = [meta]
 287 
 288     _meta = [m if isinstance(m, list) else [m] for m in meta]
 289 
 290     # Disastrously inefficient for now
 291     records: List = []
 292     lengths = []
 293 
 294     meta_vals: DefaultDict = defaultdict(list)
 295     meta_keys = [sep.join(val) for val in _meta]
 296 
 297     def _recursive_extract(data, path, seen_meta, level=0):
 298         if isinstance(data, dict):
 299             data = [data]
 300         if len(path) > 1:
 301             for obj in data:
 302                 for val, key in zip(_meta, meta_keys):
 303                     if level + 1 == len(val):
 304                         seen_meta[key] = _pull_field(obj, val[-1])
 305 
 306                 _recursive_extract(obj[path[0]], path[1:], seen_meta, level=level + 1)
 307         else:
 308             for obj in data:
 309                 recs = _pull_records(obj, path[0])
 310                 recs = [
 311                     nested_to_record(r, sep=sep, max_level=max_level)
 312                     if isinstance(r, dict)
 313                     else r
 314                     for r in recs
 315                 ]
 316 
 317                 # For repeating the metadata later
 318                 lengths.append(len(recs))
 319                 for val, key in zip(_meta, meta_keys):
 320                     if level + 1 > len(val):
 321                         meta_val = seen_meta[key]
 322                     else:
 323                         try:
 324                             meta_val = _pull_field(obj, val[level:])
 325                         except KeyError as e:
 326                             if errors == "ignore":
 327                                 meta_val = np.nan
 328                             else:
 329                                 raise KeyError(
 330                                     "Try running with errors='ignore' as key "
 331                                     f"{e} is not always present"
 332                                 ) from e
 333                     meta_vals[key].append(meta_val)
 334                 records.extend(recs)
 335 
 336     _recursive_extract(data, record_path, {}, level=0)
 337 
 338     result = DataFrame(records)
 339 
 340     if record_prefix is not None:
 341         result = result.rename(columns=lambda x: f"{record_prefix}{x}")
 342 
 343     # Data types, a problem
 344     for k, v in meta_vals.items():
 345         if meta_prefix is not None:
 346             k = meta_prefix + k
 347 
 348         if k in result:
 349             raise ValueError(
 350                 f"Conflicting metadata name {k}, need distinguishing prefix "
 351             )
 352         result[k] = np.array(v, dtype=object).repeat(lengths)
 353     return result
