 248 class JSONTableWriter(FrameWriter):
 249     _default_orient = "records"
 250 
 251     def __init__(
 252         self,
 253         obj,
 254         orient: Optional[str],
 255         date_format: str,
 256         double_precision: int,
 257         ensure_ascii: bool,
 258         date_unit: str,
 259         index: bool,
 260         default_handler: Optional[Callable[[Any], JSONSerializable]] = None,
 261         indent: int = 0,
 262     ):
 263         """
 264         Adds a `schema` attribute with the Table Schema, resets
 265         the index (can't do in caller, because the schema inference needs
 266         to know what the index is, forces orient to records, and forces
 267         date_format to 'iso'.
 268         """
 269         super().__init__(
 270             obj,
 271             orient,
 272             date_format,
 273             double_precision,
 274             ensure_ascii,
 275             date_unit,
 276             index,
 277             default_handler=default_handler,
 278             indent=indent,
 279         )
 280 
 281         if date_format != "iso":
 282             msg = (
 283                 "Trying to write with `orient='table'` and "
 284                 f"`date_format='{date_format}'`. Table Schema requires dates "
 285                 "to be formatted with `date_format='iso'`"
 286             )
 287             raise ValueError(msg)
 288 
 289         self.schema = build_table_schema(obj, index=self.index)
 290 
 291         # NotImplemented on a column MultiIndex
 292         if obj.ndim == 2 and isinstance(obj.columns, MultiIndex):
 293             raise NotImplementedError("orient='table' is not supported for MultiIndex")
 294 
 295         # TODO: Do this timedelta properly in objToJSON.c See GH #15137
 296         if (
 297             (obj.ndim == 1)
 298             and (obj.name in set(obj.index.names))
 299             or len(obj.columns & obj.index.names)
 300         ):
 301             msg = "Overlapping names between the index and columns"
 302             raise ValueError(msg)
 303 
 304         obj = obj.copy()
 305         timedeltas = obj.select_dtypes(include=["timedelta"]).columns
 306         if len(timedeltas):
 307             obj[timedeltas] = obj[timedeltas].applymap(lambda x: x.isoformat())
 308         # Convert PeriodIndex to datetimes before serializing
 309         if is_period_dtype(obj.index.dtype):
 310             obj.index = obj.index.to_timestamp()
 311 
 312         # exclude index from obj if index=False
 313         if not self.index:
 314             self.obj = obj.reset_index(drop=True)
 315         else:
 316             self.obj = obj.reset_index(drop=False)
 317         self.date_format = "iso"
 318         self.orient = "records"
 319         self.index = index
 320

 325         double_precision,

 336             double_precision,
 337             ensure_ascii,
 338             date_unit,
 339             iso_dates,

 341             indent,
 342         )

 344         return serialized

 351 def read_json(
 352     path_or_buf=None,
 353     orient=None,
 354     typ="frame",
 355     dtype=None,
 356     convert_axes=None,
 357     convert_dates=True,
 358     keep_default_dates: bool = True,
 359     numpy: bool = False,
 360     precise_float: bool = False,
 361     date_unit=None,
 362     encoding=None,
 363     lines: bool = False,
 364     chunksize: Optional[int] = None,
 365     compression="infer",
 366     nrows: Optional[int] = None,
 367 ):
 368     """
 369     Convert a JSON string to pandas object.
 370 
 371     Parameters
 372     ----------
 373     path_or_buf : a valid JSON str, path object or file-like object
 374         Any valid string path is acceptable. The string could be a URL. Valid
 375         URL schemes include http, ftp, s3, and file. For file URLs, a host is
 376         expected. A local file could be:
 377         ``file://localhost/path/to/table.json``.
 378 
 379         If you want to pass in a path object, pandas accepts any
 380         ``os.PathLike``.
 381 
 382         By file-like object, we refer to objects with a ``read()`` method,
 383         such as a file handler (e.g. via builtin ``open`` function)
 384         or ``StringIO``.
 385     orient : str
 386         Indication of expected JSON string format.
 387         Compatible JSON strings can be produced by ``to_json()`` with a
 388         corresponding orient value.
 389         The set of possible orients is:
 390 
 391         - ``'split'`` : dict like
 392           ``{index -> [index], columns -> [columns], data -> [values]}``
 393         - ``'records'`` : list like
 394           ``[{column -> value}, ... , {column -> value}]``
 395         - ``'index'`` : dict like ``{index -> {column -> value}}``
 396         - ``'columns'`` : dict like ``{column -> {index -> value}}``
 397         - ``'values'`` : just the values array
 398 
 399         The allowed and default values depend on the value
 400         of the `typ` parameter.
 401 
 402         * when ``typ == 'series'``,
 403 
 404           - allowed orients are ``{'split','records','index'}``
 405           - default is ``'index'``
 406           - The Series index must be unique for orient ``'index'``.
 407 
 408         * when ``typ == 'frame'``,
 409 
 410           - allowed orients are ``{'split','records','index',
 411             'columns','values', 'table'}``
 412           - default is ``'columns'``
 413           - The DataFrame index must be unique for orients ``'index'`` and
 414             ``'columns'``.
 415           - The DataFrame columns must be unique for orients ``'index'``,
 416             ``'columns'``, and ``'records'``.
 417 
 418         .. versionadded:: 0.23.0
 419            'table' as an allowed value for the ``orient`` argument
 420 
 421     typ : {'frame', 'series'}, default 'frame'
 422         The type of object to recover.
 423 
 424     dtype : bool or dict, default None
 425         If True, infer dtypes; if a dict of column to dtype, then use those;
 426         if False, then don't infer dtypes at all, applies only to the data.
 427 
 428         For all ``orient`` values except ``'table'``, default is True.
 429 
 430         .. versionchanged:: 0.25.0
 431 
 432            Not applicable for ``orient='table'``.
 433 
 434     convert_axes : bool, default None
 435         Try to convert the axes to the proper dtypes.
 436 
 437         For all ``orient`` values except ``'table'``, default is True.
 438 
 439         .. versionchanged:: 0.25.0
 440 
 441            Not applicable for ``orient='table'``.
 442 
 443     convert_dates : bool or list of str, default True
 444         If True then default datelike columns may be converted (depending on
 445         keep_default_dates).
 446         If False, no dates will be converted.
 447         If a list of column names, then those columns will be converted and
 448         default datelike columns may also be converted (depending on
 449         keep_default_dates).
 450 
 451     keep_default_dates : bool, default True
 452         If parsing dates (convert_dates is not False), then try to parse the
 453         default datelike columns.
 454         A column label is datelike if
 455 
 456         * it ends with ``'_at'``,
 457 
 458         * it ends with ``'_time'``,
 459 
 460         * it begins with ``'timestamp'``,
 461 
 462         * it is ``'modified'``, or
 463 
 464         * it is ``'date'``.
 465 
 466     numpy : bool, default False
 467         Direct decoding to numpy arrays. Supports numeric data only, but
 468         non-numeric column and index labels are supported. Note also that the
 469         JSON ordering MUST be the same for each term if numpy=True.
 470 
 471         .. deprecated:: 1.0.0
 472 
 473     precise_float : bool, default False
 474         Set to enable usage of higher precision (strtod) function when
 475         decoding string to double values. Default (False) is to use fast but
 476         less precise builtin functionality.
 477 
 478     date_unit : str, default None
 479         The timestamp unit to detect if converting dates. The default behaviour
 480         is to try and detect the correct precision, but if this is not desired
 481         then pass one of 's', 'ms', 'us' or 'ns' to force parsing only seconds,
 482         milliseconds, microseconds or nanoseconds respectively.
 483 
 484     encoding : str, default is 'utf-8'
 485         The encoding to use to decode py3 bytes.
 486 
 487     lines : bool, default False
 488         Read the file as a json object per line.
 489 
 490     chunksize : int, optional
 491         Return JsonReader object for iteration.
 492         See the `line-delimited json docs
 493         <https://pandas.pydata.org/pandas-docs/stable/user_guide/io.html#line-delimited-json>`_
 494         for more information on ``chunksize``.
 495         This can only be passed if `lines=True`.
 496         If this is None, the file will be read into memory all at once.
 497 
 498     compression : {'infer', 'gzip', 'bz2', 'zip', 'xz', None}, default 'infer'
 499         For on-the-fly decompression of on-disk data. If 'infer', then use
 500         gzip, bz2, zip or xz if path_or_buf is a string ending in
 501         '.gz', '.bz2', '.zip', or 'xz', respectively, and no decompression
 502         otherwise. If using 'zip', the ZIP file must contain only one data
 503         file to be read in. Set to None for no decompression.
 504 
 505     nrows : int, optional
 506         The number of lines from the line-delimited jsonfile that has to be read.
 507         This can only be passed if `lines=True`.
 508         If this is None, all the rows will be returned.
 509 
 510         .. versionadded:: 1.1
 511 
 512     Returns
 513     -------
 514     Series or DataFrame
 515         The type returned depends on the value of `typ`.
 516 
 517     See Also
 518     --------
 519     DataFrame.to_json : Convert a DataFrame to a JSON string.
 520     Series.to_json : Convert a Series to a JSON string.
 521 
 522     Notes
 523     -----
 524     Specific to ``orient='table'``, if a :class:`DataFrame` with a literal
 525     :class:`Index` name of `index` gets written with :func:`to_json`, the
 526     subsequent read operation will incorrectly set the :class:`Index` name to
 527     ``None``. This is because `index` is also used by :func:`DataFrame.to_json`
 528     to denote a missing :class:`Index` name, and the subsequent
 529     :func:`read_json` operation cannot distinguish between the two. The same
 530     limitation is encountered with a :class:`MultiIndex` and any names
 531     beginning with ``'level_'``.
 532 
 533     Examples
 534     --------
 535     >>> df = pd.DataFrame([['a', 'b'], ['c', 'd']],
 536     ...                   index=['row 1', 'row 2'],
 537     ...                   columns=['col 1', 'col 2'])
 538 
 539     Encoding/decoding a Dataframe using ``'split'`` formatted JSON:
 540 
 541     >>> df.to_json(orient='split')
 542     '{"columns":["col 1","col 2"],
 543       "index":["row 1","row 2"],
 544       "data":[["a","b"],["c","d"]]}'
 545     >>> pd.read_json(_, orient='split')
 546           col 1 col 2
 547     row 1     a     b
 548     row 2     c     d
 549 
 550     Encoding/decoding a Dataframe using ``'index'`` formatted JSON:
 551 
 552     >>> df.to_json(orient='index')
 553     '{"row 1":{"col 1":"a","col 2":"b"},"row 2":{"col 1":"c","col 2":"d"}}'
 554     >>> pd.read_json(_, orient='index')
 555           col 1 col 2
 556     row 1     a     b
 557     row 2     c     d
 558 
 559     Encoding/decoding a Dataframe using ``'records'`` formatted JSON.
 560     Note that index labels are not preserved with this encoding.
 561 
 562     >>> df.to_json(orient='records')
 563     '[{"col 1":"a","col 2":"b"},{"col 1":"c","col 2":"d"}]'
 564     >>> pd.read_json(_, orient='records')
 565       col 1 col 2
 566     0     a     b
 567     1     c     d
 568 
 569     Encoding with Table Schema
 570 
 571     >>> df.to_json(orient='table')
 572     '{"schema": {"fields": [{"name": "index", "type": "string"},
 573                             {"name": "col 1", "type": "string"},
 574                             {"name": "col 2", "type": "string"}],
 575                     "primaryKey": "index",
 576                     "pandas_version": "0.20.0"},
 577         "data": [{"index": "row 1", "col 1": "a", "col 2": "b"},
 578                 {"index": "row 2", "col 1": "c", "col 2": "d"}]}'
 579     """
 580     if orient == "table" and dtype:
 581         raise ValueError("cannot pass both dtype and orient='table'")
 582     if orient == "table" and convert_axes:
 583         raise ValueError("cannot pass both convert_axes and orient='table'")
 584 
 585     if dtype is None and orient != "table":
 586         dtype = True
 587     if convert_axes is None and orient != "table":
 588         convert_axes = True
 589     if encoding is None:
 590         encoding = "utf-8"
 591 
 592     compression = infer_compression(path_or_buf, compression)
 593     filepath_or_buffer, _, compression, should_close = get_filepath_or_buffer(
 594         path_or_buf, encoding=encoding, compression=compression
 595     )
 596 
 597     json_reader = JsonReader(
 598         filepath_or_buffer,
 599         orient=orient,
 600         typ=typ,
 601         dtype=dtype,
 602         convert_axes=convert_axes,
 603         convert_dates=convert_dates,
 604         keep_default_dates=keep_default_dates,
 605         numpy=numpy,
 606         precise_float=precise_float,
 607         date_unit=date_unit,
 608         encoding=encoding,
 609         lines=lines,
 610         chunksize=chunksize,
 611         compression=compression,
 612         nrows=nrows,
 613     )
 614 
 615     if chunksize:
 616         return json_reader
 617 
 618     result = json_reader.read()
 619     if should_close:
 620         filepath_or_buffer.close()
 621 
 622     return result

 625 class JsonReader(abc.Iterator):
 626     """
 627     JsonReader provides an interface for reading in a JSON file.
 628 
 629     If initialized with ``lines=True`` and ``chunksize``, can be iterated over
 630     ``chunksize`` lines at a time. Otherwise, calling ``read`` reads in the
 631     whole document.
 632     """
 633 
 634     def __init__(
 635         self,
 636         filepath_or_buffer,
 637         orient,
 638         typ,
 639         dtype,
 640         convert_axes,
 641         convert_dates,
 642         keep_default_dates: bool,
 643         numpy: bool,
 644         precise_float: bool,
 645         date_unit,
 646         encoding,
 647         lines: bool,
 648         chunksize: Optional[int],
 649         compression,
 650         nrows: Optional[int],
 651     ):
 652 
 653         self.orient = orient
 654         self.typ = typ
 655         self.dtype = dtype
 656         self.convert_axes = convert_axes
 657         self.convert_dates = convert_dates
 658         self.keep_default_dates = keep_default_dates
 659         self.numpy = numpy
 660         self.precise_float = precise_float
 661         self.date_unit = date_unit
 662         self.encoding = encoding
 663         self.compression = compression
 664         self.lines = lines
 665         self.chunksize = chunksize
 666         self.nrows_seen = 0
 667         self.should_close = False
 668         self.nrows = nrows
 669 
 670         if self.chunksize is not None:
 671             self.chunksize = _validate_integer("chunksize", self.chunksize, 1)
 672             if not self.lines:
 673                 raise ValueError("chunksize can only be passed if lines=True")
 674         if self.nrows is not None:
 675             self.nrows = _validate_integer("nrows", self.nrows, 0)
 676             if not self.lines:
 677                 raise ValueError("nrows can only be passed if lines=True")
 678 
 679         data = self._get_data_from_filepath(filepath_or_buffer)
 680         self.data = self._preprocess_data(data)
 681 
 682     def _preprocess_data(self, data):
 683         """
 684         At this point, the data either has a `read` attribute (e.g. a file
 685         object or a StringIO) or is a string that is a JSON document.
 686 
 687         If self.chunksize, we prepare the data for the `__next__` method.
 688         Otherwise, we read it into memory for the `read` method.
 689         """
 690         if hasattr(data, "read") and (not self.chunksize or not self.nrows):
 691             data = data.read()
 692         if not hasattr(data, "read") and (self.chunksize or self.nrows):
 693             data = StringIO(data)
 694 
 695         return data

 697     def _get_data_from_filepath(self, filepath_or_buffer):
 698         """
 699         The function read_json accepts three input types:
 700             1. filepath (string-like)
 701             2. file-like object (e.g. open file object, StringIO)
 702             3. JSON string
 703 
 704         This method turns (1) into (2) to simplify the rest of the processing.
 705         It returns input types (2) and (3) unchanged.
 706         """
 707         data = filepath_or_buffer
 708 
 709         exists = False
 710         if isinstance(data, str):
 711             try:
 712                 exists = os.path.exists(filepath_or_buffer)
 713             # gh-5874: if the filepath is too long will raise here
 714             except (TypeError, ValueError):
 715                 pass
 716 
 717         if exists or self.compression is not None:
 718             data, _ = get_handle(
 719                 filepath_or_buffer,
 720                 "r",
 721                 encoding=self.encoding,
 722                 compression=self.compression,
 723             )
 724             self.should_close = True
 725             self.open_stream = data
 726 
 727         return data

 729     def _combine_lines(self, lines) -> str:
 730         """
 731         Combines a list of JSON objects into one JSON object.
 732         """
 733         lines = filter(None, map(lambda x: x.strip(), lines))
 734         return "[" + ",".join(lines) + "]"

 736     def read(self):
 737         """
 738         Read the whole JSON input into a pandas object.
 739         """
 740         if self.lines:
 741             if self.chunksize:
 742                 obj = concat(self)
 743             elif self.nrows:
 744                 lines = list(islice(self.data, self.nrows))
 745                 lines_json = self._combine_lines(lines)
 746                 obj = self._get_object_parser(lines_json)
 747             else:
 748                 data = ensure_str(self.data)
 749                 data = data.split("\n")
 750                 obj = self._get_object_parser(self._combine_lines(data))
 751         else:
 752             obj = self._get_object_parser(self.data)
 753         self.close()
 754         return obj

 756     def _get_object_parser(self, json):
 757         """
 758         Parses a json document into a pandas object.
 759         """
 760         typ = self.typ
 761         dtype = self.dtype
 762         kwargs = {
 763             "orient": self.orient,
 764             "dtype": self.dtype,
 765             "convert_axes": self.convert_axes,
 766             "convert_dates": self.convert_dates,
 767             "keep_default_dates": self.keep_default_dates,
 768             "numpy": self.numpy,
 769             "precise_float": self.precise_float,
 770             "date_unit": self.date_unit,
 771         }
 772         obj = None
 773         if typ == "frame":
 774             obj = FrameParser(json, **kwargs).parse()
 775 
 776         if typ == "series" or obj is None:
 777             if not isinstance(dtype, bool):
 778                 kwargs["dtype"] = dtype
 779             obj = SeriesParser(json, **kwargs).parse()
 780 
 781         return obj

 783     def close(self):
 784         """
 785         If we opened a stream earlier, in _get_data_from_filepath, we should
 786         close it.
 787 
 788         If an open stream or file was passed, we leave it open.
 789         """
 790         if self.should_close:
 791             try:
 792                 self.open_stream.close()
 793             except (IOError, AttributeError):
 794                 pass

 796     def __next__(self):
 797         if self.nrows:
 798             if self.nrows_seen >= self.nrows:
 799                 self.close()
 800                 raise StopIteration
 801 
 802         lines = list(islice(self.data, self.chunksize))
 803         if lines:
 804             lines_json = self._combine_lines(lines)
 805             obj = self._get_object_parser(lines_json)
 806 
 807             # Make sure that the returned objects have the right index.
 808             obj.index = range(self.nrows_seen, self.nrows_seen + len(obj))
 809             self.nrows_seen += len(obj)
 810 
 811             return obj
 812 
 813         self.close()
 814         raise StopIteration

 909     def _try_convert_data(self, name, data, use_dtypes=True, convert_dates=True):
 910         """
 911         Try to parse a ndarray like into a column by inferring dtype.
 912         """
 913         # don't try to coerce, unless a force conversion
 914         if use_dtypes:
 915             if not self.dtype:
 916                 return data, False
 917             elif self.dtype is True:
 918                 pass
 919             else:
 920                 # dtype to force
 921                 dtype = (
 922                     self.dtype.get(name) if isinstance(self.dtype, dict) else self.dtype
 923                 )
 924                 if dtype is not None:
 925                     try:
 926                         dtype = np.dtype(dtype)
 927                         return data.astype(dtype), True
 928                     except (TypeError, ValueError):
 929                         return data, False
 930 
 931         if convert_dates:
 932             new_data, result = self._try_convert_to_date(data)
 933             if result:
 934                 return new_data, True
 935 
 936         result = False
 937 
 938         if data.dtype == "object":
 939 
 940             # try float
 941             try:
 942                 data = data.astype("float64")
 943                 result = True
 944             except (TypeError, ValueError):
 945                 pass
 946 
 947         if data.dtype.kind == "f":
 948 
 949             if data.dtype != "float64":
 950 
 951                 # coerce floats to 64
 952                 try:
 953                     data = data.astype("float64")
 954                     result = True
 955                 except (TypeError, ValueError):
 956                     pass
 957 
 958         # don't coerce 0-len data
 959         if len(data) and (data.dtype == "float" or data.dtype == "object"):
 960 
 961             # coerce ints if we can
 962             try:
 963                 new_data = data.astype("int64")
 964                 if (new_data == data).all():
 965                     data = new_data
 966                     result = True
 967             except (TypeError, ValueError, OverflowError):
 968                 pass
 969 
 970         # coerce ints to 64
 971         if data.dtype == "int":
 972 
 973             # coerce floats to 64
 974             try:
 975                 data = data.astype("int64")
 976                 result = True
 977             except (TypeError, ValueError):
 978                 pass
 979 
 980         return data, result
 981 
 982     def _try_convert_to_date(self, data):
 983         """
 984         Try to parse a ndarray like into a date column.
 985 
 986         Try to coerce object in epoch/iso formats and integer/float in epoch
 987         formats. Return a boolean if parsing was successful.
 988         """
 989         # no conversion on empty
 990         if not len(data):
 991             return data, False
 992 
 993         new_data = data
 994         if new_data.dtype == "object":
 995             try:
 996                 new_data = data.astype("int64")
 997             except (TypeError, ValueError, OverflowError):
 998                 pass
 999 
1000         # ignore numbers that are out of range
1001         if issubclass(new_data.dtype.type, np.number):
1002             in_range = (
1003                 isna(new_data._values)
1004                 | (new_data > self.min_stamp)
1005                 | (new_data._values == iNaT)
1006             )
1007             if not in_range.all():
1008                 return data, False
1009 
1010         date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
1011         for date_unit in date_units:
1012             try:
1013                 new_data = to_datetime(new_data, errors="raise", unit=date_unit)
1014             except (ValueError, OverflowError, TypeError):
1015                 continue
1016             return new_data, True
1017         return data, False

1138     def _process_converter(self, f, filt=None):
1139         """
1140         Take a conversion function and possibly recreate the frame.
1141         """
1142         if filt is None:
1143             filt = lambda col, c: True
1144 
1145         needs_new_obj = False
1146         new_obj = dict()
1147         for i, (col, c) in enumerate(self.obj.items()):
1148             if filt(col, c):
1149                 new_data, result = f(col, c)
1150                 if result:
1151                     c = new_data
1152                     needs_new_obj = True
1153             new_obj[i] = c
1154 
1155         if needs_new_obj:
1156 
1157             # possibly handle dup columns
1158             new_obj = DataFrame(new_obj, index=self.obj.index)
1159             new_obj.columns = self.obj.columns
1160             self.obj = new_obj

1172     def _try_convert_dates(self):
1173         if self.obj is None:
1174             return
1175 
1176         # our columns to parse
1177         convert_dates = self.convert_dates
1178         if convert_dates is True:
1179             convert_dates = []
1180         convert_dates = set(convert_dates)
1181 
1182         def is_ok(col) -> bool:
1183             """
1184             Return if this col is ok to try for a date parse.
1185             """
1186             if not isinstance(col, str):
1187                 return False
1188 
1189             col_lower = col.lower()
1190             if (
1191                 col_lower.endswith("_at")
1192                 or col_lower.endswith("_time")
1193                 or col_lower == "modified"
1194                 or col_lower == "date"
1195                 or col_lower == "datetime"
1196                 or col_lower.startswith("timestamp")
1197             ):
1198                 return True
1199             return False
1200 
1201         self._process_converter(
1202             lambda col, c: self._try_convert_to_date(c),
1203             lambda col, c: (
1204                 (self.keep_default_dates and is_ok(col)) or col in convert_dates
1205             ),
1206         )
