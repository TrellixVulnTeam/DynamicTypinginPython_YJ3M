   2 Pyperclip
   3 
   4 A cross-platform clipboard module for Python,
   5 with copy & paste functions for plain text.
   6 By Al Sweigart al@inventwithpython.com
   7 BSD License
   8 
   9 Usage:
  10   import pyperclip
  11   pyperclip.copy('The text to be copied to the clipboard.')
  12   spam = pyperclip.paste()
  13 
  14   if not pyperclip.is_available():
  15     print("Copy functionality unavailable!")
  16 
  17 On Windows, no additional modules are needed.

  94 def _stringifyText(text) -> str:
  95     acceptedTypes = (str, int, float, bool)
  96     if not isinstance(text, acceptedTypes):
  97         raise PyperclipException(
  98             f"only str, int, float, and bool values "
  99             f"can be copied to the clipboard, not {type(text).__name__}"
 100         )
 101     return str(text)

 166 def init_xclip_clipboard():
 167     DEFAULT_SELECTION = "c"
 168     PRIMARY_SELECTION = "p"
 169 
 170     def copy_xclip(text, primary=False):
 171         text = _stringifyText(text)  # Converts non-str values to str.
 172         selection = DEFAULT_SELECTION
 173         if primary:
 174             selection = PRIMARY_SELECTION
 175         p = subprocess.Popen(
 176             ["xclip", "-selection", selection], stdin=subprocess.PIPE, close_fds=True
 177         )
 178         p.communicate(input=text.encode(ENCODING))
 179

 181         selection = DEFAULT_SELECTION
 182         if primary:
 183             selection = PRIMARY_SELECTION
 184         p = subprocess.Popen(
 185             ["xclip", "-selection", selection, "-o"],
 186             stdout=subprocess.PIPE,
 187             stderr=subprocess.PIPE,
 188             close_fds=True,
 189         )
 190         stdout, stderr = p.communicate()
 191         # Intentionally ignore extraneous output on stderr when clipboard is empty
 192         return stdout.decode(ENCODING)

 225     def copy_klipper(text):
 226         text = _stringifyText(text)  # Converts non-str values to str.
 227         p = subprocess.Popen(
 228             [
 229                 "qdbus",
 230                 "org.kde.klipper",
 231                 "/klipper",
 232                 "setClipboardContents",
 233                 text.encode(ENCODING),
 234             ],
 235             stdin=subprocess.PIPE,
 236             close_fds=True,

 238         p.communicate(input=None)

 240     def paste_klipper():
 241         p = subprocess.Popen(
 242             ["qdbus", "org.kde.klipper", "/klipper", "getClipboardContents"],
 243             stdout=subprocess.PIPE,
 244             close_fds=True,
 245         )
 246         stdout, stderr = p.communicate()
 247 
 248         # Workaround for https://bugs.kde.org/show_bug.cgi?id=342874
 249         # TODO: https://github.com/asweigart/pyperclip/issues/43
 250         clipboardContents = stdout.decode(ENCODING)
 251         # even if blank, Klipper will append a newline at the end
 252         assert len(clipboardContents) > 0
 253         # make sure that newline is there
 254         assert clipboardContents.endswith("\n")
 255         if clipboardContents.endswith("\n"):
 256             clipboardContents = clipboardContents[:-1]
 257         return clipboardContents

 262 def init_dev_clipboard_clipboard():
 263     def copy_dev_clipboard(text):
 264         text = _stringifyText(text)  # Converts non-str values to str.
 265         if text == "":
 266             warnings.warn(
 267                 "Pyperclip cannot copy a blank string to the clipboard on Cygwin."
 268                 "This is effectively a no-op."
 269             )
 270         if "\r" in text:
 271             warnings.warn("Pyperclip cannot handle \\r characters on Cygwin.")
 272 
 273         with open("/dev/clipboard", "wt") as fo:
 274             fo.write(text)
 275 
 276     def paste_dev_clipboard() -> str:
 277         with open("/dev/clipboard", "rt") as fo:
 278             content = fo.read()
 279         return content
 280 
 281     return copy_dev_clipboard, paste_dev_clipboard

 295 # Windows-related clipboard functions:
 296 class CheckedCall:
 297     def __init__(self, f):
 298         super().__setattr__("f", f)
 299 
 300     def __call__(self, *args):
 301         ret = self.f(*args)
 302         if not ret and get_errno():
 303             raise PyperclipWindowsException("Error calling " + self.f.__name__)
 304         return ret
 305 
 306     def __setattr__(self, key, value):
 307         setattr(self.f, key, value)

 310 def init_windows_clipboard():
 311     global HGLOBAL, LPVOID, DWORD, LPCSTR, INT
 312     global HWND, HINSTANCE, HMENU, BOOL, UINT, HANDLE
 313     from ctypes.wintypes import (
 314         HGLOBAL,
 315         LPVOID,
 316         DWORD,
 317         LPCSTR,
 318         INT,
 319         HWND,
 320         HINSTANCE,
 321         HMENU,
 322         BOOL,
 323         UINT,
 324         HANDLE,
 325     )
 326 
 327     windll = ctypes.windll
 328     msvcrt = ctypes.CDLL("msvcrt")
 329 
 330     safeCreateWindowExA = CheckedCall(windll.user32.CreateWindowExA)
 331     safeCreateWindowExA.argtypes = [
 332         DWORD,
 333         LPCSTR,
 334         LPCSTR,
 335         DWORD,
 336         INT,
 337         INT,
 338         INT,
 339         INT,
 340         HWND,
 341         HMENU,
 342         HINSTANCE,
 343         LPVOID,
 344     ]
 345     safeCreateWindowExA.restype = HWND
 346 
 347     safeDestroyWindow = CheckedCall(windll.user32.DestroyWindow)
 348     safeDestroyWindow.argtypes = [HWND]
 349     safeDestroyWindow.restype = BOOL
 350 
 351     OpenClipboard = windll.user32.OpenClipboard
 352     OpenClipboard.argtypes = [HWND]
 353     OpenClipboard.restype = BOOL
 354 
 355     safeCloseClipboard = CheckedCall(windll.user32.CloseClipboard)
 356     safeCloseClipboard.argtypes = []
 357     safeCloseClipboard.restype = BOOL
 358 
 359     safeEmptyClipboard = CheckedCall(windll.user32.EmptyClipboard)
 360     safeEmptyClipboard.argtypes = []
 361     safeEmptyClipboard.restype = BOOL
 362 
 363     safeGetClipboardData = CheckedCall(windll.user32.GetClipboardData)
 364     safeGetClipboardData.argtypes = [UINT]
 365     safeGetClipboardData.restype = HANDLE
 366 
 367     safeSetClipboardData = CheckedCall(windll.user32.SetClipboardData)
 368     safeSetClipboardData.argtypes = [UINT, HANDLE]
 369     safeSetClipboardData.restype = HANDLE
 370 
 371     safeGlobalAlloc = CheckedCall(windll.kernel32.GlobalAlloc)
 372     safeGlobalAlloc.argtypes = [UINT, c_size_t]
 373     safeGlobalAlloc.restype = HGLOBAL
 374 
 375     safeGlobalLock = CheckedCall(windll.kernel32.GlobalLock)
 376     safeGlobalLock.argtypes = [HGLOBAL]
 377     safeGlobalLock.restype = LPVOID
 378 
 379     safeGlobalUnlock = CheckedCall(windll.kernel32.GlobalUnlock)
 380     safeGlobalUnlock.argtypes = [HGLOBAL]
 381     safeGlobalUnlock.restype = BOOL
 382 
 383     wcslen = CheckedCall(msvcrt.wcslen)
 384     wcslen.argtypes = [c_wchar_p]
 385     wcslen.restype = UINT
 386 
 387     GMEM_MOVEABLE = 0x0002
 388     CF_UNICODETEXT = 13
 389 
 390     @contextlib.contextmanager
 391     def window():
 392         """
 393         Context that provides a valid Windows hwnd.
 394         """
 395         # we really just need the hwnd, so setting "STATIC"
 396         # as predefined lpClass is just fine.
 397         hwnd = safeCreateWindowExA(
 398             0, b"STATIC", None, 0, 0, 0, 0, 0, None, None, None, None
 399         )
 400         try:
 401             yield hwnd
 402         finally:
 403             safeDestroyWindow(hwnd)
 404 
 405     @contextlib.contextmanager
 406     def clipboard(hwnd):
 407         """
 408         Context manager that opens the clipboard and prevents
 409         other applications from modifying the clipboard content.
 410         """
 411         # We may not get the clipboard handle immediately because
 412         # some other application is accessing it (?)
 413         # We try for at least 500ms to get the clipboard.
 414         t = time.time() + 0.5
 415         success = False
 416         while time.time() < t:
 417             success = OpenClipboard(hwnd)
 418             if success:
 419                 break
 420             time.sleep(0.01)
 421         if not success:
 422             raise PyperclipWindowsException("Error calling OpenClipboard")
 423 
 424         try:
 425             yield
 426         finally:
 427             safeCloseClipboard()
 428 
 429     def copy_windows(text):
 430         # This function is heavily based on
 431         # http://msdn.com/ms649016#_win32_Copying_Information_to_the_Clipboard
 432 
 433         text = _stringifyText(text)  # Converts non-str values to str.
 434 
 435         with window() as hwnd:
 436             # http://msdn.com/ms649048
 437             # If an application calls OpenClipboard with hwnd set to NULL,
 438             # EmptyClipboard sets the clipboard owner to NULL;
 439             # this causes SetClipboardData to fail.
 440             # => We need a valid hwnd to copy something.
 441             with clipboard(hwnd):
 442                 safeEmptyClipboard()
 443 
 444                 if text:
 445                     # http://msdn.com/ms649051
 446                     # If the hMem parameter identifies a memory object,
 447                     # the object must have been allocated using the
 448                     # function with the GMEM_MOVEABLE flag.
 449                     count = wcslen(text) + 1
 450                     handle = safeGlobalAlloc(GMEM_MOVEABLE, count * sizeof(c_wchar))
 451                     locked_handle = safeGlobalLock(handle)
 452 
 453                     ctypes.memmove(
 454                         c_wchar_p(locked_handle),
 455                         c_wchar_p(text),
 456                         count * sizeof(c_wchar),
 457                     )
 458 
 459                     safeGlobalUnlock(handle)
 460                     safeSetClipboardData(CF_UNICODETEXT, handle)
 461 
 462     def paste_windows():
 463         with clipboard(None):
 464             handle = safeGetClipboardData(CF_UNICODETEXT)
 465             if not handle:
 466                 # GetClipboardData may return NULL with errno == NO_ERROR
 467                 # if the clipboard is empty.
 468                 # (Also, it may return a handle to an empty buffer,
 469                 # but technically that's not empty)
 470                 return ""
 471             return c_wchar_p(handle).value
 472 
 473     return copy_windows, paste_windows

 496 # Automatic detection of clipboard mechanisms
 497 # and importing is done in determine_clipboard():
 498 def determine_clipboard():
 499     """
 500     Determine the OS/platform and set the copy() and paste() functions
 501     accordingly.
 502     """
 503     global Foundation, AppKit, qtpy, PyQt4, PyQt5
 504 
 505     # Setup for the CYGWIN platform:
 506     if (
 507         "cygwin" in platform.system().lower()
 508     ):  # Cygwin has a variety of values returned by platform.system(),
 509         # such as 'CYGWIN_NT-6.1'
 510         # FIXME: pyperclip currently does not support Cygwin,
 511         # see https://github.com/asweigart/pyperclip/issues/55
 512         if os.path.exists("/dev/clipboard"):
 513             warnings.warn(
 514                 "Pyperclip's support for Cygwin is not perfect,"
 515                 "see https://github.com/asweigart/pyperclip/issues/55"
 516             )
 517             return init_dev_clipboard_clipboard()
 518 
 519     # Setup for the WINDOWS platform:
 520     elif os.name == "nt" or platform.system() == "Windows":
 521         return init_windows_clipboard()
 522 
 523     if platform.system() == "Linux":
 524         with open("/proc/version", "r") as f:
 525             if "Microsoft" in f.read():
 526                 return init_wsl_clipboard()
 527 
 528     # Setup for the MAC OS X platform:
 529     if os.name == "mac" or platform.system() == "Darwin":
 530         try:
 531             import Foundation  # check if pyobjc is installed
 532             import AppKit
 533         except ImportError:
 534             return init_osx_pbcopy_clipboard()
 535         else:
 536             return init_osx_pyobjc_clipboard()
 537 
 538     # Setup for the LINUX platform:
 539     if HAS_DISPLAY:
 540         if _executable_exists("xsel"):
 541             return init_xsel_clipboard()
 542         if _executable_exists("xclip"):
 543             return init_xclip_clipboard()
 544         if _executable_exists("klipper") and _executable_exists("qdbus"):
 545             return init_klipper_clipboard()
 546 
 547         try:
 548             # qtpy is a small abstraction layer that lets you write applications
 549             # using a single api call to either PyQt or PySide.
 550             # https://pypi.python.org/project/QtPy
 551             import qtpy  # check if qtpy is installed
 552         except ImportError:
 553             # If qtpy isn't installed, fall back on importing PyQt4.
 554             try:
 555                 import PyQt5  # check if PyQt5 is installed
 556             except ImportError:
 557                 try:
 558                     import PyQt4  # check if PyQt4 is installed
 559                 except ImportError:
 560                     pass  # We want to fail fast for all non-ImportError exceptions.
 561                 else:
 562                     return init_qt_clipboard()
 563             else:
 564                 return init_qt_clipboard()
 565         else:
 566             return init_qt_clipboard()
 567 
 568     return init_no_clipboard()

 571 def set_clipboard(clipboard):
 572     """
 573     Explicitly sets the clipboard mechanism. The "clipboard mechanism" is how
 574     the copy() and paste() functions interact with the operating system to
 575     implement the copy/paste feature. The clipboard parameter must be one of:
 576         - pbcopy
 577         - pbobjc (default on Mac OS X)
 578         - qt
 579         - xclip
 580         - xsel
 581         - klipper
 582         - windows (default on Windows)
 583         - no (this is what is set when no clipboard mechanism can be found)
 584     """
 585     global copy, paste
 586 
 587     clipboard_types = {
 588         "pbcopy": init_osx_pbcopy_clipboard,
 589         "pyobjc": init_osx_pyobjc_clipboard,
 590         "qt": init_qt_clipboard,  # TODO - split this into 'qtpy', 'pyqt4', and 'pyqt5'
 591         "xclip": init_xclip_clipboard,
 592         "xsel": init_xsel_clipboard,
 593         "klipper": init_klipper_clipboard,
 594         "windows": init_windows_clipboard,
 595         "no": init_no_clipboard,
 596     }
 597 
 598     if clipboard not in clipboard_types:
 599         allowed_clipboard_types = [repr(_) for _ in clipboard_types.keys()]
 600         raise ValueError(
 601             f"Argument must be one of {', '.join(allowed_clipboard_types)}"
 602         )
 603 
 604     # Sets pyperclip's copy() and paste() functions:
 605     copy, paste = clipboard_types[clipboard]()

 608 def lazy_load_stub_copy(text):
 609     """
 610     A stub function for copy(), which will load the real copy() function when
 611     called so that the real copy() function is used for later calls.
 612 
 613     This allows users to import pyperclip without having determine_clipboard()
 614     automatically run, which will automatically select a clipboard mechanism.
 615     This could be a problem if it selects, say, the memory-heavy PyQt4 module
 616     but the user was just going to immediately call set_clipboard() to use a
 617     different clipboard mechanism.
 618 
 619     The lazy loading this stub function implements gives the user a chance to
 620     call set_clipboard() to pick another clipboard mechanism. Or, if the user
 621     simply calls copy() or paste() without calling set_clipboard() first,
 622     will fall back on whatever clipboard mechanism that determine_clipboard()
 623     automatically chooses.
 624     """
 625     global copy, paste
 626     copy, paste = determine_clipboard()
 627     return copy(text)

 632     A stub function for paste(), which will load the real paste() function when
 633     called so that the real paste() function is used for later calls.
 634 
 635     This allows users to import pyperclip without having determine_clipboard()
 636     automatically run, which will automatically select a clipboard mechanism.
 637     This could be a problem if it selects, say, the memory-heavy PyQt4 module
 638     but the user was just going to immediately call set_clipboard() to use a
 639     different clipboard mechanism.
 640 
 641     The lazy loading this stub function implements gives the user a chance to
 642     call set_clipboard() to pick another clipboard mechanism. Or, if the user
 643     simply calls copy() or paste() without calling set_clipboard() first,
 644     will fall back on whatever clipboard mechanism that determine_clipboard()
