  96 def stringify_path(
  97     filepath_or_buffer: FilePathOrBuffer[AnyStr],
  98 ) -> FilePathOrBuffer[AnyStr]:
  99     """
 100     Attempt to convert a path-like object to a string.
 101 
 102     Parameters
 103     ----------
 104     filepath_or_buffer : object to be converted
 105 
 106     Returns
 107     -------
 108     str_filepath_or_buffer : maybe a string version of the object
 109 
 110     Notes
 111     -----
 112     Objects supporting the fspath protocol (python 3.6+) are coerced
 113     according to its __fspath__ method.
 114 
 115     For backwards compatibility with older pythons, pathlib.Path and
 116     py.path objects are specially coerced.
 117 
 118     Any other object is passed through unchanged, which includes bytes,
 119     strings, buffers, or anything else that's not even path-like.
 120     """
 121     if hasattr(filepath_or_buffer, "__fspath__"):
 122         # https://github.com/python/mypy/issues/1424
 123         return filepath_or_buffer.__fspath__()  # type: ignore
 124     elif isinstance(filepath_or_buffer, pathlib.Path):
 125         return str(filepath_or_buffer)
 126     return _expand_user(filepath_or_buffer)

 153 def get_fs_for_path(filepath: str):
 154     """
 155     Get appropriate filesystem given a filepath.
 156     Supports s3fs, gcs and local file system.
 157 
 158     Parameters
 159     ----------
 160     filepath : str
 161         File path. e.g s3://bucket/object, /local/path, gcs://pandas/obj
 162 
 163     Returns
 164     -------
 165     s3fs.S3FileSystem, gcsfs.GCSFileSystem, None
 166         Appropriate FileSystem to use. None for local filesystem.
 167     """
 168     if is_s3_url(filepath):
 169         from pandas.io import s3
 170 
 171         return s3.get_fs()
 172     elif is_gcs_url(filepath):
 173         from pandas.io import gcs
 174 
 175         return gcs.get_fs()
 176     else:
 177         return None

 180 def get_filepath_or_buffer(
 181     filepath_or_buffer: FilePathOrBuffer,
 182     encoding: Optional[str] = None,
 183     compression: Optional[str] = None,
 184     mode: Optional[str] = None,
 185 ):
 186     """
 187     If the filepath_or_buffer is a url, translate and return the buffer.
 188     Otherwise passthrough.
 189 
 190     Parameters
 191     ----------
 192     filepath_or_buffer : a url, filepath (str, py.path.local or pathlib.Path),
 193                          or buffer
 194     compression : {{'gzip', 'bz2', 'zip', 'xz', None}}, optional
 195     encoding : the encoding to use to decode bytes, default is 'utf-8'
 196     mode : str, optional
 197 
 198     Returns
 199     -------
 200     Tuple[FilePathOrBuffer, str, str, bool]
 201         Tuple containing the filepath or buffer, the encoding, the compression
 202         and should_close.
 203     """
 204     filepath_or_buffer = stringify_path(filepath_or_buffer)
 205 
 206     if isinstance(filepath_or_buffer, str) and is_url(filepath_or_buffer):
 207         req = urlopen(filepath_or_buffer)
 208         content_encoding = req.headers.get("Content-Encoding", None)
 209         if content_encoding == "gzip":
 210             # Override compression based on Content-Encoding header
 211             compression = "gzip"
 212         reader = BytesIO(req.read())
 213         req.close()
 214         return reader, encoding, compression, True
 215 
 216     if is_s3_url(filepath_or_buffer):
 217         from pandas.io import s3
 218 
 219         return s3.get_filepath_or_buffer(
 220             filepath_or_buffer, encoding=encoding, compression=compression, mode=mode
 221         )
 222 
 223     if is_gcs_url(filepath_or_buffer):
 224         from pandas.io import gcs
 225 
 226         return gcs.get_filepath_or_buffer(
 227             filepath_or_buffer, encoding=encoding, compression=compression, mode=mode
 228         )
 229 
 230     if isinstance(filepath_or_buffer, (str, bytes, mmap.mmap)):
 231         return _expand_user(filepath_or_buffer), None, compression, False
 232 
 233     if not is_file_like(filepath_or_buffer):
 234         msg = f"Invalid file path or buffer object type: {type(filepath_or_buffer)}"
 235         raise ValueError(msg)
 236 
 237     return filepath_or_buffer, None, compression, False

 261 def get_compression_method(
 262     compression: Optional[Union[str, Mapping[str, str]]]
 263 ) -> Tuple[Optional[str], Dict[str, str]]:
 264     """
 265     Simplifies a compression argument to a compression method string and
 266     a mapping containing additional arguments.
 267 
 268     Parameters
 269     ----------
 270     compression : str or mapping
 271         If string, specifies the compression method. If mapping, value at key
 272         'method' specifies compression method.
 273 
 274     Returns
 275     -------
 276     tuple of ({compression method}, Optional[str]
 277               {compression arguments}, Dict[str, str])
 278 
 279     Raises
 280     ------
 281     ValueError on mapping missing 'method' key
 282     """
 283     if isinstance(compression, Mapping):
 284         compression_args = dict(compression)
 285         try:
 286             compression = compression_args.pop("method")
 287         except KeyError as err:
 288             raise ValueError("If mapping, compression must have key 'method'") from err
 289     else:
 290         compression_args = {}
 291     return compression, compression_args

 294 def infer_compression(
 295     filepath_or_buffer: FilePathOrBuffer, compression: Optional[str]
 296 ) -> Optional[str]:
 297     """
 298     Get the compression method for filepath_or_buffer. If compression='infer',
 299     the inferred compression method is returned. Otherwise, the input
 300     compression method is returned unchanged, unless it's invalid, in which
 301     case an error is raised.
 302 
 303     Parameters
 304     ----------
 305     filepath_or_buffer : str or file handle
 306         File path or object.
 307     compression : {'infer', 'gzip', 'bz2', 'zip', 'xz', None}
 308         If 'infer' and `filepath_or_buffer` is path-like, then detect
 309         compression from the following extensions: '.gz', '.bz2', '.zip',
 310         or '.xz' (otherwise no compression).
 311 
 312     Returns
 313     -------
 314     string or None
 315 
 316     Raises
 317     ------
 318     ValueError on invalid compression specified.
 319     """
 320     # No compression has been explicitly specified
 321     if compression is None:
 322         return None
 323 
 324     # Infer compression
 325     if compression == "infer":
 326         # Convert all path types (e.g. pathlib.Path) to strings
 327         filepath_or_buffer = stringify_path(filepath_or_buffer)
 328         if not isinstance(filepath_or_buffer, str):
 329             # Cannot infer compression of a buffer, assume no compression
 330             return None
 331 
 332         # Infer compression from the filename/URL extension
 333         for compression, extension in _compression_to_extension.items():
 334             if filepath_or_buffer.endswith(extension):
 335                 return compression
 336         return None
 337 
 338     # Compression has been specified. Check that it's valid
 339     if compression in _compression_to_extension:
 340         return compression
 341 
 342     msg = f"Unrecognized compression type: {compression}"
 343     valid = ["infer", None] + sorted(_compression_to_extension)
 344     msg += f"\nValid compression types are {valid}"
 345     raise ValueError(msg)

 348 def get_handle(
 349     path_or_buf,
 350     mode: str,
 351     encoding=None,
 352     compression: Optional[Union[str, Mapping[str, Any]]] = None,
 353     memory_map: bool = False,
 354     is_text: bool = True,
 355     errors=None,
 356 ):
 357     """
 358     Get file handle for given path/buffer and mode.
 359 
 360     Parameters
 361     ----------
 362     path_or_buf : str or file handle
 363         File path or object.
 364     mode : str
 365         Mode to open path_or_buf with.
 366     encoding : str or None
 367         Encoding to use.
 368     compression : str or dict, default None
 369         If string, specifies compression mode. If dict, value at key 'method'
 370         specifies compression mode. Compression mode must be one of {'infer',
 371         'gzip', 'bz2', 'zip', 'xz', None}. If compression mode is 'infer'
 372         and `filepath_or_buffer` is path-like, then detect compression from
 373         the following extensions: '.gz', '.bz2', '.zip', or '.xz' (otherwise
 374         no compression). If dict and compression mode is one of
 375         {'zip', 'gzip', 'bz2'}, or inferred as one of the above,
 376         other entries passed as additional compression options.
 377 
 378         .. versionchanged:: 1.0.0
 379 
 380            May now be a dict with key 'method' as compression mode
 381            and other keys as compression options if compression
 382            mode is 'zip'.
 383 
 384         .. versionchanged:: 1.1.0
 385 
 386            Passing compression options as keys in dict is now
 387            supported for compression modes 'gzip' and 'bz2' as well as 'zip'.
 388 
 389     memory_map : boolean, default False
 390         See parsers._parser_params for more information.
 391     is_text : boolean, default True
 392         whether file/buffer is in text format (csv, json, etc.), or in binary
 393         mode (pickle, etc.).
 394     errors : str, default 'strict'
 395         Specifies how encoding and decoding errors are to be handled.
 396         See the errors argument for :func:`open` for a full list
 397         of options.
 398 
 399         .. versionadded:: 1.1.0
 400 
 401     Returns
 402     -------
 403     f : file-like
 404         A file-like object.
 405     handles : list of file-like objects
 406         A list of file-like object that were opened in this function.
 407     """
 408     need_text_wrapping: Tuple[Type["IOBase"], ...]
 409     try:
 410         from s3fs import S3File
 411 
 412         need_text_wrapping = (BufferedIOBase, RawIOBase, S3File)
 413     except ImportError:
 414         need_text_wrapping = (BufferedIOBase, RawIOBase)
 415 
 416     handles: List[IO] = list()
 417     f = path_or_buf
 418 
 419     # Convert pathlib.Path/py.path.local or string
 420     path_or_buf = stringify_path(path_or_buf)
 421     is_path = isinstance(path_or_buf, str)
 422 
 423     compression, compression_args = get_compression_method(compression)
 424     if is_path:
 425         compression = infer_compression(path_or_buf, compression)
 426 
 427     if compression:
 428 
 429         # GH33398 the type ignores here seem related to mypy issue #5382;
 430         # it may be possible to remove them once that is resolved.
 431 
 432         # GZ Compression
 433         if compression == "gzip":
 434             if is_path:
 435                 f = gzip.open(
 436                     path_or_buf, mode, **compression_args  # type: ignore
 437                 )
 438             else:
 439                 f = gzip.GzipFile(
 440                     fileobj=path_or_buf, **compression_args  # type: ignore
 441                 )
 442 
 443         # BZ Compression
 444         elif compression == "bz2":
 445             if is_path:
 446                 f = bz2.BZ2File(
 447                     path_or_buf, mode, **compression_args  # type: ignore
 448                 )
 449             else:
 450                 f = bz2.BZ2File(path_or_buf, **compression_args)  # type: ignore
 451 
 452         # ZIP Compression
 453         elif compression == "zip":
 454             zf = _BytesZipFile(path_or_buf, mode, **compression_args)
 455             # Ensure the container is closed as well.
 456             handles.append(zf)
 457             if zf.mode == "w":
 458                 f = zf
 459             elif zf.mode == "r":
 460                 zip_names = zf.namelist()
 461                 if len(zip_names) == 1:
 462                     f = zf.open(zip_names.pop())
 463                 elif len(zip_names) == 0:
 464                     raise ValueError(f"Zero files found in ZIP file {path_or_buf}")
 465                 else:
 466                     raise ValueError(
 467                         "Multiple files found in ZIP file. "
 468                         f"Only one file per ZIP: {zip_names}"
 469                     )
 470 
 471         # XZ Compression
 472         elif compression == "xz":
 473             f = _get_lzma_file(lzma)(path_or_buf, mode)
 474 
 475         # Unrecognized Compression
 476         else:
 477             msg = f"Unrecognized compression type: {compression}"
 478             raise ValueError(msg)
 479 
 480         handles.append(f)
 481 
 482     elif is_path:
 483         if encoding:
 484             # Encoding
 485             f = open(path_or_buf, mode, encoding=encoding, errors=errors, newline="")
 486         elif is_text:
 487             # No explicit encoding
 488             f = open(path_or_buf, mode, errors="replace", newline="")
 489         else:
 490             # Binary mode
 491             f = open(path_or_buf, mode)
 492         handles.append(f)
 493 
 494     # Convert BytesIO or file objects passed with an encoding
 495     if is_text and (compression or isinstance(f, need_text_wrapping)):
 496         from io import TextIOWrapper
 497 
 498         g = TextIOWrapper(f, encoding=encoding, errors=errors, newline="")
 499         if not isinstance(f, (BufferedIOBase, RawIOBase)):
 500             handles.append(g)
 501         f = g
 502 
 503     if memory_map and hasattr(f, "fileno"):
 504         try:
 505             wrapped = _MMapWrapper(f)
 506             f.close()
 507             f = wrapped
 508         except Exception:
 509             # we catch any errors that may have occurred
 510             # because that is consistent with the lower-level
 511             # functionality of the C engine (pd.read_csv), so
 512             # leave the file handler as is then
 513             pass
 514 
 515     return f, handles

 518 class _BytesZipFile(zipfile.ZipFile, BytesIO):  # type: ignore
 519     """
 520     Wrapper for standard library class ZipFile and allow the returned file-like
 521     handle to accept byte strings via `write` method.
 522 
 523     BytesIO provides attributes of file-like object and ZipFile.writestr writes
 524     bytes strings into a member of the archive.
 525     """
 526 
 527     # GH 17778
 528     def __init__(
 529         self,
 530         file: FilePathOrBuffer,
 531         mode: str,
 532         archive_name: Optional[str] = None,
 533         **kwargs,
 534     ):
 535         if mode in ["wb", "rb"]:
 536             mode = mode.replace("b", "")
 537         self.archive_name = archive_name
 538         super().__init__(file, mode, zipfile.ZIP_DEFLATED, **kwargs)
 539 
 540     def write(self, data):
 541         archive_name = self.filename
 542         if self.archive_name is not None:
 543             archive_name = self.archive_name
 544         super().writestr(archive_name, data)
 545 
 546     @property
 547     def closed(self):
 548         return self.fp is None

 551 class _MMapWrapper(abc.Iterator):
 552     """
 553     Wrapper for the Python's mmap class so that it can be properly read in
 554     by Python's csv.reader class.
 555 
 556     Parameters
 557     ----------
 558     f : file object
 559         File object to be mapped onto memory. Must support the 'fileno'
 560         method or have an equivalent attribute
 561 
 562     """
 563 
 564     def __init__(self, f: IO):
 565         self.mmap = mmap.mmap(f.fileno(), 0, access=mmap.ACCESS_READ)
 566 
 567     def __getattr__(self, name: str):
 568         return getattr(self.mmap, name)
 569 
 570     def __iter__(self) -> "_MMapWrapper":
 571         return self
 572 
 573     def __next__(self) -> str:
 574         newbytes = self.mmap.readline()
 575 
 576         # readline returns bytes, not str, but Python's CSV reader
 577         # expects str, so convert the output to str before continuing
 578         newline = newbytes.decode("utf-8")
 579 
 580         # mmap doesn't raise if reading past the allocated
 581         # data but instead returns an empty string, so raise
 582         # if that is returned
 583         if newline == "":
 584             raise StopIteration
 585         return newline
