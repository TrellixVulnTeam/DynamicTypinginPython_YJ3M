 222     """
 223     Open a compressed file and return a file object.
 224 
 225     Parameters
 226     ----------
 227     path : str
 228         The path where the file is read from.
 229 
 230     compression : {'gzip', 'bz2', 'zip', 'xz', None}
 231         Name of the decompression to use
 232 
 233     Returns
 234     -------
 235     file object
 236     """
 237     if compression is None:
 238         f = open(path, "rb")
 239     elif compression == "gzip":
 240         f = gzip.open(path, "rb")
 241     elif compression == "bz2":

 251             raise ValueError(f"ZIP file {path} error. Only one file per ZIP.")
 252     else:
 253         raise ValueError(f"Unrecognized compression type: {compression}")
 254 
 255     try:
 256         yield f
 257     finally:
 258         f.close()
 259         if compression == "zip":

 306 def assert_almost_equal(
 307     left,
 308     right,
 309     check_dtype: Union[bool, str] = "equiv",
 310     check_less_precise: Union[bool, int] = False,
 311     **kwargs,
 312 ):
 313     """
 314     Check that the left and right objects are approximately equal.
 315 
 316     By approximately equal, we refer to objects that are numbers or that
 317     contain numbers which may be equivalent to specific levels of precision.
 318 
 319     Parameters
 320     ----------
 321     left : object
 322     right : object
 323     check_dtype : bool or {'equiv'}, default 'equiv'
 324         Check dtype if both a and b are the same type. If 'equiv' is passed in,
 325         then `RangeIndex` and `Int64Index` are also considered equivalent
 326         when doing type checking.
 327     check_less_precise : bool or int, default False
 328         Specify comparison precision. 5 digits (False) or 3 digits (True)
 329         after decimal points are compared. If int, then specify the number
 330         of digits to compare.
 331 
 332         When comparing two numbers, if the first number has magnitude less
 333         than 1e-5, we compare the two numbers directly and check whether
 334         they are equivalent within the specified precision. Otherwise, we
 335         compare the **ratio** of the second number to the first number and
 336         check whether it is equivalent to 1 within the specified precision.
 337     """
 338     if isinstance(left, pd.Index):
 339         assert_index_equal(
 340             left,
 341             right,
 342             check_exact=False,
 343             exact=check_dtype,
 344             check_less_precise=check_less_precise,
 345             **kwargs,
 346         )
 347 
 348     elif isinstance(left, pd.Series):
 349         assert_series_equal(
 350             left,
 351             right,
 352             check_exact=False,
 353             check_dtype=check_dtype,
 354             check_less_precise=check_less_precise,
 355             **kwargs,
 356         )
 357 
 358     elif isinstance(left, pd.DataFrame):
 359         assert_frame_equal(
 360             left,
 361             right,
 362             check_exact=False,
 363             check_dtype=check_dtype,
 364             check_less_precise=check_less_precise,
 365             **kwargs,
 366         )
 367 
 368     else:
 369         # Other sequences.
 370         if check_dtype:
 371             if is_number(left) and is_number(right):
 372                 # Do not compare numeric classes, like np.float64 and float.
 373                 pass
 374             elif is_bool(left) and is_bool(right):
 375                 # Do not compare bool classes, like np.bool_ and bool.
 376                 pass
 377             else:
 378                 if isinstance(left, np.ndarray) or isinstance(right, np.ndarray):
 379                     obj = "numpy array"
 380                 else:
 381                     obj = "Input"
 382                 assert_class_equal(left, right, obj=obj)
 383         _testing.assert_almost_equal(
 384             left,
 385             right,
 386             check_dtype=check_dtype,
 387             check_less_precise=check_less_precise,
 388             **kwargs,

 487     """
 488     Gets a temporary path and agrees to remove on close.
 489 
 490     Parameters
 491     ----------
 492     filename : str (optional)

 494         scope. if passed, creates temporary file with filename as ending.

 497         savefig and other functions which want to append extensions.

 503         argument to either function.
 504 
 505     See Also
 506     --------
 507     tempfile.TemporaryFile
 508     tempfile.mkstemp
 509     """
 510     filename = filename or ""
 511     fd = None
 512 
 513     kwargs["suffix"] = filename
 514 
 515     if return_filelike:
 516         f = tempfile.TemporaryFile(**kwargs)
 517 
 518         try:
 519             yield f
 520         finally:
 521             f.close()
 522     else:
 523         # Don't generate tempfile if using a path with directory specified.
 524         if len(os.path.dirname(filename)):
 525             raise ValueError("Can't pass a qualified name to ensure_clean()")
 526 
 527         try:
 528             fd, filename = tempfile.mkstemp(**kwargs)
 529         except UnicodeEncodeError:
 530             import pytest
 531 
 532             pytest.skip("no unicode file names on this system")
 533 
 534         try:
 535             yield filename
 536         finally:
 537             try:
 538                 os.close(fd)
 539             except OSError:
 540                 print(f"Couldn't close file descriptor: {fd} (file: {filename})")
 541             try:
 542                 if os.path.exists(filename):

 594 def assert_index_equal(
 595     left: Index,
 596     right: Index,
 597     exact: Union[bool, str] = "equiv",
 598     check_names: bool = True,
 599     check_less_precise: Union[bool, int] = False,
 600     check_exact: bool = True,
 601     check_categorical: bool = True,
 602     obj: str = "Index",
 603 ) -> None:
 604     """
 605     Check that left and right Index are equal.
 606 
 607     Parameters
 608     ----------
 609     left : Index
 610     right : Index
 611     exact : bool or {'equiv'}, default 'equiv'
 612         Whether to check the Index class, dtype and inferred_type
 613         are identical. If 'equiv', then RangeIndex can be substituted for
 614         Int64Index as well.
 615     check_names : bool, default True
 616         Whether to check the names attribute.
 617     check_less_precise : bool or int, default False
 618         Specify comparison precision. Only used when check_exact is False.
 619         5 digits (False) or 3 digits (True) after decimal points are compared.
 620         If int, then specify the digits to compare.
 621     check_exact : bool, default True
 622         Whether to compare number exactly.
 623     check_categorical : bool, default True
 624         Whether to compare internal Categorical exactly.
 625     obj : str, default 'Index'
 626         Specify object name being compared, internally used to show appropriate
 627         assertion message.
 628     """
 629     __tracebackhide__ = True
 630 
 631     def _check_types(l, r, obj="Index"):
 632         if exact:
 633             assert_class_equal(l, r, exact=exact, obj=obj)
 634 
 635             # Skip exact dtype checking when `check_categorical` is False
 636             if check_categorical:
 637                 assert_attr_equal("dtype", l, r, obj=obj)
 638 
 639             # allow string-like to have different inferred_types
 640             if l.inferred_type in ("string"):
 641                 assert r.inferred_type in ("string")
 642             else:
 643                 assert_attr_equal("inferred_type", l, r, obj=obj)
 644 
 645     def _get_ilevel_values(index, level):
 646         # accept level number only
 647         unique = index.levels[level]
 648         level_codes = index.codes[level]
 649         filled = take_1d(unique._values, level_codes, fill_value=unique._na_value)
 650         values = unique._shallow_copy(filled, name=index.names[level])
 651         return values
 652 
 653     # instance validation
 654     _check_isinstance(left, right, Index)
 655 
 656     # class / dtype comparison
 657     _check_types(left, right, obj=obj)
 658 
 659     # level comparison
 660     if left.nlevels != right.nlevels:
 661         msg1 = f"{obj} levels are different"
 662         msg2 = f"{left.nlevels}, {left}"
 663         msg3 = f"{right.nlevels}, {right}"
 664         raise_assert_detail(obj, msg1, msg2, msg3)
 665

 947 def assert_numpy_array_equal(
 948     left,
 949     right,
 950     strict_nan=False,

 953     check_same=None,
 954     obj="numpy array",
 955     index_values=None,
 956 ):
 957     """
 958     Check that 'np.ndarray' is equivalent.
 959 
 960     Parameters
 961     ----------
 962     left, right : numpy.ndarray or iterable
 963         The two arrays to be compared.
 964     strict_nan : bool, default False
 965         If True, consider NaN and None to be different.
 966     check_dtype : bool, default True
 967         Check dtype if both a and b are np.ndarray.
 968     err_msg : str, default None
 969         If provided, used as assertion message.
 970     check_same : None|'copy'|'same', default None
 971         Ensure left and right refer/do not refer to the same memory area.
 972     obj : str, default 'numpy array'
 973         Specify object name being compared, internally used to show appropriate
 974         assertion message.
 975     index_values : numpy.ndarray, default None
 976         optional index (shared by both left and right), used in output.
 977     """
 978     __tracebackhide__ = True
 979 
 980     # instance validation
 981     # Show a detailed error message when classes are different
 982     assert_class_equal(left, right, obj=obj)
 983     # both classes must be an np.ndarray
 984     _check_isinstance(left, right, np.ndarray)
 985 
 986     def _get_base(obj):
 987         return obj.base if getattr(obj, "base", None) is not None else obj
 988 
 989     left_base = _get_base(left)
 990     right_base = _get_base(right)
 991 
 992     if check_same == "same":
 993         if left_base is not right_base:
 994             raise AssertionError(f"{repr(left_base)} is not {repr(right_base)}")
 995     elif check_same == "copy":
 996         if left_base is right_base:
 997             raise AssertionError(f"{repr(left_base)} is {repr(right_base)}")
 998 
 999     def _raise(left, right, err_msg):
1000         if err_msg is None:
1001             if left.shape != right.shape:
1002                 raise_assert_detail(
1003                     obj, f"{obj} shapes are different", left.shape, right.shape,
1004                 )
1005 
1006             diff = 0
1007             for l, r in zip(left, right):
1008                 # count up differences
1009                 if not array_equivalent(l, r, strict_nan=strict_nan):
1010                     diff += 1
1011 
1012             diff = diff * 100.0 / left.size
1013             msg = f"{obj} values are different ({np.round(diff, 5)} %)"
1014             raise_assert_detail(obj, msg, left, right, index_values=index_values)
1015 
1016         raise AssertionError(err_msg)
1017 
1018     # compare shape and values
1019     if not array_equivalent(left, right, strict_nan=strict_nan):
1020         _raise(left, right, err_msg)
1021 
1022     if check_dtype:
1023         if isinstance(left, np.ndarray) and isinstance(right, np.ndarray):
1024             assert_attr_equal("dtype", left, right, obj=obj)

1301         Whether to check the DataFrame dtype is identical.
1302     check_index_type : bool or {'equiv'}, default 'equiv'
1303         Whether to check the Index class, dtype and inferred_type
1304         are identical.
1305     check_column_type : bool or {'equiv'}, default 'equiv'
1306         Whether to check the columns class, dtype and inferred_type
1307         are identical. Is passed as the ``exact`` argument of
1308         :func:`assert_index_equal`.
1309     check_frame_type : bool, default True
1310         Whether to check the DataFrame class is identical.
1311     check_less_precise : bool or int, default False
1312         Specify comparison precision. Only used when check_exact is False.
1313         5 digits (False) or 3 digits (True) after decimal points are compared.
1314         If int, then specify the digits to compare.
1315 
1316         When comparing two numbers, if the first number has magnitude less
1317         than 1e-5, we compare the two numbers directly and check whether
1318         they are equivalent within the specified precision. Otherwise, we
1319         compare the **ratio** of the second number to the first number and
1320         check whether it is equivalent to 1 within the specified precision.
1321     check_names : bool, default True
1322         Whether to check that the `names` attribute for both the `index`
1323         and `column` attributes of the DataFrame is identical.
1324     by_blocks : bool, default False
1325         Specify how to compare internal data. If False, compare by columns.
1326         If True, compare by blocks.
1327     check_exact : bool, default False
1328         Whether to compare number exactly.
1329     check_datetimelike_compat : bool, default False
1330         Compare datetime-like which is comparable ignoring dtype.
1331     check_categorical : bool, default True
1332         Whether to compare internal Categorical exactly.
1333     check_like : bool, default False
1334         If True, ignore the order of index & columns.
1335         Note: index labels must match their respective rows
1336         (same as in columns) - same labels must be with the same data.

1343         assertion message.

1367     ...

1381     _check_isinstance(left, right, DataFrame)

1869 def makeCustomIndex(
1870     nentries, nlevels, prefix="#", names=False, ndupe_l=None, idx_type=None
1871 ):
1872     """
1873     Create an index/multindex with given dimensions, levels, names, etc'
1874 
1875     nentries - number of entries in index
1876     nlevels - number of levels (> 1 produces multindex)
1877     prefix - a string prefix for labels
1878     names - (Optional), bool or list of strings. if True will use default
1879        names, if false will use no names, if a list is given, the name of
1880        each level in the index will be taken from the list.
1881     ndupe_l - (Optional), list of ints, the number of rows for which the
1882        label will repeated at the corresponding level, you can specify just
1883        the first few, the rest will use the default ndupe_l of 1.
1884        len(ndupe_l) <= nlevels.
1885     idx_type - "i"/"f"/"s"/"u"/"dt"/"p"/"td".
1886        If idx_type is not None, `idx_nlevels` must be 1.
1887        "i"/"f" creates an integer/float index,
1888        "s"/"u" creates a string/unicode index
1889        "dt" create a datetime index.
1890        "td" create a datetime index.
1891 
1892         if unspecified, string labels will be generated.
1893     """
1894     if ndupe_l is None:
1895         ndupe_l = [1] * nlevels
1896     assert is_sequence(ndupe_l) and len(ndupe_l) <= nlevels
1897     assert names is None or names is False or names is True or len(names) is nlevels
1898     assert idx_type is None or (
1899         idx_type in ("i", "f", "s", "u", "dt", "p", "td") and nlevels == 1
1900     )
1901 
1902     if names is True:
1903         # build default names
1904         names = [prefix + str(i) for i in range(nlevels)]
1905     if names is False:
1906         # pass None to index constructor for no name
1907         names = None
1908 
1909     # make singleton case uniform
1910     if isinstance(names, str) and nlevels == 1:
1911         names = [names]
1912 
1913     # specific 1D index type requested?
1914     idx_func = dict(
1915         i=makeIntIndex,
1916         f=makeFloatIndex,
1917         s=makeStringIndex,
1918         u=makeUnicodeIndex,
1919         dt=makeDateIndex,
1920         td=makeTimedeltaIndex,
1921         p=makePeriodIndex,
1922     ).get(idx_type)
1923     if idx_func:
1924         idx = idx_func(nentries)
1925         # but we need to fill in the name
1926         if names:
1927             idx.name = names[0]
1928         return idx
1929     elif idx_type is not None:
1930         raise ValueError(
1931             f"{repr(idx_type)} is not a legal value for `idx_type`, "
1932             "use  'i'/'f'/'s'/'u'/'dt'/'p'/'td'."
1933         )
1934 
1935     if len(ndupe_l) < nlevels:
1936         ndupe_l.extend([1] * (nlevels - len(ndupe_l)))
1937     assert len(ndupe_l) == nlevels
1938 
1939     assert all(x > 0 for x in ndupe_l)
