 144     def get_result(self):
 145         """ compute the results """
 146         # dispatch to agg
 147         if is_list_like(self.f) or is_dict_like(self.f):
 148             return self.obj.aggregate(self.f, axis=self.axis, *self.args, **self.kwds)
 149 
 150         # all empty
 151         if len(self.columns) == 0 and len(self.index) == 0:
 152             return self.apply_empty_result()
 153 
 154         # string dispatch
 155         if isinstance(self.f, str):
 156             # Support for `frame.transform('method')`
 157             # Some methods (shift, etc.) require the axis argument, others
 158             # don't, so inspect and insert if necessary.
 159             func = getattr(self.obj, self.f)
 160             sig = inspect.getfullargspec(func)
 161             if "axis" in sig.args:
 162                 self.kwds["axis"] = self.axis
 163             return func(*self.args, **self.kwds)
 164 
 165         # ufunc
 166         elif isinstance(self.f, np.ufunc):
 167             with np.errstate(all="ignore"):
 168                 results = self.obj._mgr.apply("apply", func=self.f)
 169             # _constructor will retain self.index and self.columns
 170             return self.obj._constructor(data=results)
 171 
 172         # broadcasting
 173         if self.result_type == "broadcast":
 174             return self.apply_broadcast(self.obj)
 175 
 176         # one axis empty
 177         elif not all(self.obj.shape):
 178             return self.apply_empty_result()
 179 
 180         # raw
 181         elif self.raw:
 182             return self.apply_raw()
 183 
 184         return self.apply_standard()

 186     def apply_empty_result(self):
 187         """
 188         we have an empty result; at least 1 axis is 0
 189 
 190         we will try to apply the function to an empty
 191         series in order to see if this is a reduction function
 192         """
 193         # we are not asked to reduce or infer reduction
 194         # so just return a copy of the existing object
 195         if self.result_type not in ["reduce", None]:
 196             return self.obj.copy()
 197 
 198         # we may need to infer
 199         should_reduce = self.result_type == "reduce"
 200 
 201         from pandas import Series
 202 
 203         if not should_reduce:
 204             try:
 205                 r = self.f(Series([], dtype=np.float64))
 206             except Exception:
 207                 pass
 208             else:
 209                 should_reduce = not isinstance(r, Series)
 210 
 211         if should_reduce:
 212             if len(self.agg_axis):
 213                 r = self.f(Series([], dtype=np.float64))
 214             else:
 215                 r = np.nan
 216 
 217             return self.obj._constructor_sliced(r, index=self.agg_axis)
 218         else:
 219             return self.obj.copy()

 221     def apply_raw(self):
 222         """ apply to the values as a numpy array """
 223         result, reduction_success = libreduction.compute_reduction(
 224             self.values, self.f, axis=self.axis
 225         )
 226 
 227         # We expect np.apply_along_axis to give a two-dimensional result, or raise.
 228         if not reduction_success:
 229             result = np.apply_along_axis(self.f, self.axis, self.values)
 230 
 231         # TODO: mixed type case
 232         if result.ndim == 2:
 233             return self.obj._constructor(result, index=self.index, columns=self.columns)
 234         else:
 235             return self.obj._constructor_sliced(result, index=self.agg_axis)
 236 
 237     def apply_broadcast(self, target: "DataFrame") -> "DataFrame":
 238         result_values = np.empty_like(target.values)
 239 
 240         # axis which we want to compare compliance
 241         result_compare = target.shape[0]
 242 
 243         for i, col in enumerate(target.columns):
 244             res = self.f(target[col])
 245             ares = np.asarray(res).ndim
 246 
 247             # must be a scalar or 1d
 248             if ares > 1:
 249                 raise ValueError("too many dims to broadcast")
 250             elif ares == 1:
 251 
 252                 # must match return dim
 253                 if result_compare != len(res):
 254                     raise ValueError("cannot broadcast result")
 255 
 256             result_values[:, i] = res
 257 
 258         # we *always* preserve the original index / columns
 259         result = self.obj._constructor(
 260             result_values, index=target.index, columns=target.columns
 261         )
 262         return result

 264     def apply_standard(self):
 265 
 266         # partial result that may be returned from reduction
 267         partial_result = None
 268 
 269         # try to reduce first (by default)
 270         # this only matters if the reduction in values is of different dtype
 271         # e.g. if we want to apply to a SparseFrame, then can't directly reduce
 272 
 273         # we cannot reduce using non-numpy dtypes,
 274         # as demonstrated in gh-12244
 275         if (
 276             self.result_type in ["reduce", None]
 277             and not self.dtypes.apply(is_extension_array_dtype).any()
 278             # Disallow dtypes where setting _index_data will break
 279             #  ExtensionArray values, see GH#31182
 280             and not self.dtypes.apply(lambda x: x.kind in ["m", "M"]).any()
 281             # Disallow complex_internals since libreduction shortcut raises a TypeError
 282             and not self.agg_axis._has_complex_internals
 283         ):
 284 
 285             values = self.values
 286             index = self.obj._get_axis(self.axis)
 287             labels = self.agg_axis
 288             empty_arr = np.empty(len(index), dtype=values.dtype)
 289 
 290             # Preserve subclass for e.g. test_subclassed_apply
 291             dummy = self.obj._constructor_sliced(
 292                 empty_arr, index=index, dtype=values.dtype
 293             )
 294 
 295             try:
 296                 result, reduction_success = libreduction.compute_reduction(
 297                     values, self.f, axis=self.axis, dummy=dummy, labels=labels
 298                 )
 299             except TypeError:
 300                 # e.g. test_apply_ignore_failures we just ignore
 301                 if not self.ignore_failures:
 302                     raise
 303             except ZeroDivisionError:
 304                 # reached via numexpr; fall back to python implementation
 305                 pass
 306             else:
 307                 if reduction_success:
 308                     return self.obj._constructor_sliced(result, index=labels)
 309 
 310                 # no exceptions - however reduction was unsuccessful,
 311                 # use the computed function result for first element
 312                 partial_result = result[0]
 313                 if isinstance(partial_result, ABCSeries):
 314                     partial_result = partial_result.infer_objects()
 315 
 316         # compute the result using the series generator,
 317         # use the result computed while trying to reduce if available.
 318         results, res_index = self.apply_series_generator(partial_result)
 319 
 320         # wrap results
 321         return self.wrap_results(results, res_index)
 322 
 323     def apply_series_generator(self, partial_result=None) -> Tuple[ResType, "Index"]:
 324         series_gen = self.series_generator
 325         res_index = self.result_index
 326 
 327         results = {}
 328 
 329         # If a partial result was already computed,
 330         # use it instead of running on the first element again
 331         series_gen_enumeration = enumerate(series_gen)
 332         if partial_result is not None:
 333             i, v = next(series_gen_enumeration)
 334             results[i] = partial_result
 335 
 336         if self.ignore_failures:
 337             successes = []
 338             for i, v in series_gen_enumeration:
 339                 try:
 340                     results[i] = self.f(v)
 341                 except Exception:
 342                     pass
 343                 else:
 344                     successes.append(i)
 345 
 346             # so will work with MultiIndex
 347             if len(successes) < len(res_index):
 348                 res_index = res_index.take(successes)
 349 
 350         else:
 351             for i, v in series_gen_enumeration:
 352 
 353                 results[i] = self.f(v)
 354 
 355         return results, res_index

 357     def wrap_results(
 358         self, results: ResType, res_index: "Index"
 359     ) -> Union["Series", "DataFrame"]:
 360         from pandas import Series
 361 
 362         # see if we can infer the results
 363         if len(results) > 0 and 0 in results and is_sequence(results[0]):
 364 
 365             return self.wrap_results_for_axis(results, res_index)
 366 
 367         # dict of scalars
 368 
 369         # the default dtype of an empty Series will be `object`, but this
 370         # code can be hit by df.mean() where the result should have dtype
 371         # float64 even if it's an empty Series.
 372         constructor_sliced = self.obj._constructor_sliced
 373         if constructor_sliced is Series:
 374             result = create_series_with_explicit_dtype(
 375                 results, dtype_if_empty=np.float64
 376             )
 377         else:
 378             result = constructor_sliced(results)
 379         result.index = res_index
 380 
 381         return result

 402     def wrap_results_for_axis(
 403         self, results: ResType, res_index: "Index"
 404     ) -> "DataFrame":
 405         """ return the results for the rows """
 406         result = self.obj._constructor(data=results)
 407 
 408         if not isinstance(results[0], ABCSeries):
 409             if len(result.index) == len(self.res_columns):
 410                 result.index = self.res_columns
 411 
 412         if len(result.columns) == len(res_index):
 413             result.columns = res_index
 414 
 415         return result

 418 class FrameColumnApply(FrameApply):
 419     axis = 1
 420 
 421     def apply_broadcast(self, target: "DataFrame") -> "DataFrame":
 422         result = super().apply_broadcast(target.T)
 423         return result.T

 425     @property
 426     def series_generator(self):
 427         constructor = self.obj._constructor_sliced
 428         return (
 429             constructor(arr, index=self.columns, name=name)
 430             for i, (arr, name) in enumerate(zip(self.values, self.index))
 431         )
 432 
 433     @property
 434     def result_index(self) -> "Index":
 435         return self.index
 436 
 437     @property
 438     def result_columns(self) -> "Index":
 439         return self.columns
 440 
 441     def wrap_results_for_axis(
 442         self, results: ResType, res_index: "Index"
 443     ) -> Union["Series", "DataFrame"]:
 444         """ return the results for the columns """
 445         result: Union["Series", "DataFrame"]
 446 
 447         # we have requested to expand
 448         if self.result_type == "expand":
 449             result = self.infer_to_same_shape(results, res_index)
 450 
 451         # we have a non-series and don't want inference
 452         elif not isinstance(results[0], ABCSeries):
 453             from pandas import Series
 454 
 455             result = Series(results)
 456             result.index = res_index
 457 
 458         # we may want to infer results
 459         else:
 460             result = self.infer_to_same_shape(results, res_index)
 461 
 462         return result

 464     def infer_to_same_shape(self, results: ResType, res_index: "Index") -> "DataFrame":
 465         """ infer the results to the same shape as the input object """
 466         result = self.obj._constructor(data=results)
 467         result = result.T
 468 
 469         # set the index
 470         result.index = res_index
 471 
 472         # infer dtypes
 473         result = result.infer_objects()
 474 
 475         return result
