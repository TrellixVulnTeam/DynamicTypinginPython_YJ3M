   6 def cartesian_product(X):
   7     """
   8     Numpy version of itertools.product.
   9     Sometimes faster (for large inputs)...
  10 
  11     Parameters
  12     ----------
  13     X : list-like of list-likes
  14 
  15     Returns
  16     -------
  17     product : list of ndarrays
  18 
  19     Examples
  20     --------
  21     >>> cartesian_product([list('ABC'), [1, 2]])
  22     [array(['A', 'A', 'B', 'B', 'C', 'C'], dtype='<U1'), array([1, 2, 1, 2, 1, 2])]
  23 
  24     See Also
  25     --------
  26     itertools.product : Cartesian product of input iterables.  Equivalent to
  27         nested for-loops.
  28     """
  29     msg = "Input must be a list-like of list-likes"
  30     if not is_list_like(X):
  31         raise TypeError(msg)
  32     for x in X:
  33         if not is_list_like(x):
  34             raise TypeError(msg)
  35 
  36     if len(X) == 0:
  37         return []
  38 
  39     lenX = np.fromiter((len(x) for x in X), dtype=np.intp)
  40     cumprodX = np.cumproduct(lenX)
  41 
  42     a = np.roll(cumprodX, 1)
  43     a[0] = 1
  44 
  45     if cumprodX[-1] != 0:
  46         b = cumprodX[-1] / cumprodX
  47     else:
  48         # if any factor is empty, the cartesian product is empty
  49         b = np.zeros_like(cumprodX)
  50 
  51     return [_tile_compat(np.repeat(x, b[i]), np.product(a[i])) for i, x in enumerate(X)]

  54 def _tile_compat(arr, num: int):
  55     """
  56     Index compat for np.tile.
  57 
  58     Notes
  59     -----
  60     Does not support multi-dimensional `num`.
  61     """
  62     if isinstance(arr, np.ndarray):
  63         return np.tile(arr, num)
  64 
  65     # Otherwise we have an Index
  66     taker = np.tile(np.arange(len(arr)), num)
  67     return arr.take(taker)
