 288 def _compute_grand_margin(data, values, aggfunc, margins_name: str = "All"):
 289 
 290     if values:
 291         grand_margin = {}
 292         for k, v in data[values].items():
 293             try:
 294                 if isinstance(aggfunc, str):
 295                     grand_margin[k] = getattr(v, aggfunc)()
 296                 elif isinstance(aggfunc, dict):
 297                     if isinstance(aggfunc[k], str):
 298                         grand_margin[k] = getattr(v, aggfunc[k])()
 299                     else:
 300                         grand_margin[k] = aggfunc[k](v)
 301                 else:
 302                     grand_margin[k] = aggfunc(v)
 303             except TypeError:
 304                 pass
 305         return grand_margin
 306     else:
 307         return {margins_name: aggfunc(data.index)}

 310 def _generate_marginal_results(
 311     table, data, values, rows, cols, aggfunc, observed, margins_name: str = "All",
 312 ):
 313     if len(cols) > 0:
 314         # need to "interleave" the margins
 315         table_pieces = []
 316         margin_keys = []
 317 
 318         def _all_key(key):
 319             return (key, margins_name) + ("",) * (len(cols) - 1)
 320 
 321         if len(rows) > 0:
 322             margin = data[rows + values].groupby(rows, observed=observed).agg(aggfunc)
 323             cat_axis = 1
 324 
 325             for key, piece in table.groupby(level=0, axis=cat_axis, observed=observed):
 326                 all_key = _all_key(key)
 327 
 328                 # we are going to mutate this, so need to copy!
 329                 piece = piece.copy()
 330                 try:
 331                     piece[all_key] = margin[key]
 332                 except TypeError:
 333 
 334                     # we cannot reshape, so coerce the axis
 335                     piece.set_axis(
 336                         piece._get_axis(cat_axis)._to_safe_for_reshape(),
 337                         axis=cat_axis,
 338                         inplace=True,
 339                     )
 340                     piece[all_key] = margin[key]
 341 
 342                 table_pieces.append(piece)
 343                 margin_keys.append(all_key)
 344         else:
 345             from pandas import DataFrame
 346 
 347             cat_axis = 0
 348             for key, piece in table.groupby(level=0, axis=cat_axis, observed=observed):
 349                 if len(cols) > 1:
 350                     all_key = _all_key(key)
 351                 else:
 352                     all_key = margins_name
 353                 table_pieces.append(piece)
 354                 # GH31016 this is to calculate margin for each group, and assign
 355                 # corresponded key as index
 356                 transformed_piece = DataFrame(piece.apply(aggfunc)).T
 357                 transformed_piece.index = Index([all_key], name=piece.index.name)
 358 
 359                 # append piece for margin into table_piece
 360                 table_pieces.append(transformed_piece)
 361                 margin_keys.append(all_key)
 362 
 363         result = concat(table_pieces, axis=cat_axis)
 364 
 365         if len(rows) == 0:
 366             return result
 367     else:
 368         result = table
 369         margin_keys = table.columns
 370 
 371     if len(cols) > 0:
 372         row_margin = data[cols + values].groupby(cols, observed=observed).agg(aggfunc)
 373         row_margin = row_margin.stack()
 374 
 375         # slight hack
 376         new_order = [len(cols)] + list(range(len(cols)))
 377         row_margin.index = row_margin.index.reorder_levels(new_order)
 378     else:
 379         row_margin = Series(np.nan, index=result.columns)
 380 
 381     return result, margin_keys, row_margin

 384 def _generate_marginal_results_without_values(
 385     table: "DataFrame", data, rows, cols, aggfunc, observed, margins_name: str = "All"
 386 ):
 387     if len(cols) > 0:
 388         # need to "interleave" the margins
 389         margin_keys: Union[List, Index] = []
 390 
 391         def _all_key():
 392             if len(cols) == 1:
 393                 return margins_name
 394             return (margins_name,) + ("",) * (len(cols) - 1)
 395 
 396         if len(rows) > 0:
 397             margin = data[rows].groupby(rows, observed=observed).apply(aggfunc)
 398             all_key = _all_key()
 399             table[all_key] = margin
 400             result = table
 401             margin_keys.append(all_key)
 402 
 403         else:
 404             margin = data.groupby(level=0, axis=0, observed=observed).apply(aggfunc)
 405             all_key = _all_key()
 406             table[all_key] = margin
 407             result = table
 408             margin_keys.append(all_key)
 409             return result
 410     else:
 411         result = table
 412         margin_keys = table.columns
 413 
 414     if len(cols):
 415         row_margin = data[cols].groupby(cols, observed=observed).apply(aggfunc)
 416     else:
 417         row_margin = Series(np.nan, index=result.columns)
 418

 422 def _convert_by(by):
 423     if by is None:
 424         by = []
 425     elif (
 426         is_scalar(by)
 427         or isinstance(by, (np.ndarray, Index, ABCSeries, Grouper))
 428         or hasattr(by, "__call__")
 429     ):
 430         by = [by]
 431     else:
 432         by = list(by)

 436 @Substitution("\ndata : DataFrame")
 437 @Appender(_shared_docs["pivot"], indents=1)
 438 def pivot(
 439     data: "DataFrame",
 440     index: Optional[Union[Label, Sequence[Label]]] = None,
 441     columns: Optional[Union[Label, Sequence[Label]]] = None,
 442     values: Optional[Union[Label, Sequence[Label]]] = None,
 443 ) -> "DataFrame":
 444     if columns is None:
 445         raise TypeError("pivot() missing 1 required argument: 'columns'")
 446 
 447     columns = com.convert_to_list_like(columns)
 448 
 449     if values is None:
 450         if index is not None:
 451             cols = com.convert_to_list_like(index)
 452         else:
 453             cols = []
 454         cols.extend(columns)
 455 
 456         append = index is None
 457         indexed = data.set_index(cols, append=append)
 458     else:
 459         if index is None:
 460             index = [Series(data.index, name=data.index.name)]
 461         else:
 462             index = com.convert_to_list_like(index)
 463             index = [data[idx] for idx in index]
 464 
 465         data_columns = [data[col] for col in columns]
 466         index.extend(data_columns)
 467         index = MultiIndex.from_arrays(index)
 468 
 469         if is_list_like(values) and not isinstance(values, tuple):
 470             # Exclude tuple because it is seen as a single column name
 471             values = cast(Sequence[Label], values)
 472             indexed = data._constructor(
 473                 data[values]._values, index=index, columns=values
 474             )
 475         else:
 476             indexed = data._constructor_sliced(data[values]._values, index=index)
 477     return indexed.unstack(columns)

 480 def crosstab(
 481     index,
 482     columns,
 483     values=None,
 484     rownames=None,
 485     colnames=None,
 486     aggfunc=None,
 487     margins=False,
 488     margins_name: str = "All",
 489     dropna: bool = True,
 490     normalize=False,
 491 ) -> "DataFrame":
 492     """
 493     Compute a simple cross tabulation of two (or more) factors. By default
 494     computes a frequency table of the factors unless an array of values and an
 495     aggregation function are passed.
 496 
 497     Parameters
 498     ----------
 499     index : array-like, Series, or list of arrays/Series
 500         Values to group by in the rows.
 501     columns : array-like, Series, or list of arrays/Series
 502         Values to group by in the columns.
 503     values : array-like, optional
 504         Array of values to aggregate according to the factors.
 505         Requires `aggfunc` be specified.
 506     rownames : sequence, default None
 507         If passed, must match number of row arrays passed.
 508     colnames : sequence, default None
 509         If passed, must match number of column arrays passed.
 510     aggfunc : function, optional
 511         If specified, requires `values` be specified as well.
 512     margins : bool, default False
 513         Add row/column margins (subtotals).
 514     margins_name : str, default 'All'
 515         Name of the row/column that will contain the totals
 516         when margins is True.
 517     dropna : bool, default True
 518         Do not include columns whose entries are all NaN.
 519     normalize : bool, {'all', 'index', 'columns'}, or {0,1}, default False
 520         Normalize by dividing all values by the sum of values.
 521 
 522         - If passed 'all' or `True`, will normalize over all values.
 523         - If passed 'index' will normalize over each row.
 524         - If passed 'columns' will normalize over each column.
 525         - If margins is `True`, will also normalize margin values.
 526 
 527     Returns
 528     -------
 529     DataFrame
 530         Cross tabulation of the data.
 531 
 532     See Also
 533     --------
 534     DataFrame.pivot : Reshape data based on column values.
 535     pivot_table : Create a pivot table as a DataFrame.
 536 
 537     Notes
 538     -----
 539     Any Series passed will have their name attributes used unless row or column
 540     names for the cross-tabulation are specified.
 541 
 542     Any input passed containing Categorical data will have **all** of its
 543     categories included in the cross-tabulation, even if the actual data does
 544     not contain any instances of a particular category.
 545 
 546     In the event that there aren't overlapping indexes an empty DataFrame will
 547     be returned.
 548 
 549     Examples
 550     --------
 551     >>> a = np.array(["foo", "foo", "foo", "foo", "bar", "bar",
 552     ...               "bar", "bar", "foo", "foo", "foo"], dtype=object)
 553     >>> b = np.array(["one", "one", "one", "two", "one", "one",
 554     ...               "one", "two", "two", "two", "one"], dtype=object)
 555     >>> c = np.array(["dull", "dull", "shiny", "dull", "dull", "shiny",
 556     ...               "shiny", "dull", "shiny", "shiny", "shiny"],
 557     ...              dtype=object)
 558     >>> pd.crosstab(a, [b, c], rownames=['a'], colnames=['b', 'c'])
 559     b   one        two
 560     c   dull shiny dull shiny
 561     a
 562     bar    1     2    1     0
 563     foo    2     2    1     2
 564 
 565     Here 'c' and 'f' are not represented in the data and will not be
 566     shown in the output because dropna is True by default. Set
 567     dropna=False to preserve categories with no data.
 568 
 569     >>> foo = pd.Categorical(['a', 'b'], categories=['a', 'b', 'c'])
 570     >>> bar = pd.Categorical(['d', 'e'], categories=['d', 'e', 'f'])
 571     >>> pd.crosstab(foo, bar)
 572     col_0  d  e
 573     row_0
 574     a      1  0
 575     b      0  1
 576     >>> pd.crosstab(foo, bar, dropna=False)
 577     col_0  d  e  f
 578     row_0
 579     a      1  0  0
 580     b      0  1  0
 581     c      0  0  0
 582     """
 583     index = com.maybe_make_list(index)
 584     columns = com.maybe_make_list(columns)
 585 
 586     rownames = _get_names(index, rownames, prefix="row")
 587     colnames = _get_names(columns, colnames, prefix="col")
 588 
 589     common_idx = None
 590     pass_objs = [x for x in index + columns if isinstance(x, (ABCSeries, ABCDataFrame))]
 591     if pass_objs:
 592         common_idx = get_objs_combined_axis(pass_objs, intersect=True, sort=False)
 593 
 594     data: Dict = {}
 595     data.update(zip(rownames, index))
 596     data.update(zip(colnames, columns))
 597 
 598     if values is None and aggfunc is not None:
 599         raise ValueError("aggfunc cannot be used without values.")
 600 
 601     if values is not None and aggfunc is None:
 602         raise ValueError("values cannot be used without an aggfunc.")
 603 
 604     from pandas import DataFrame
 605 
 606     df = DataFrame(data, index=common_idx)
 607     original_df_cols = df.columns
 608 
 609     if values is None:
 610         df["__dummy__"] = 0
 611         kwargs = {"aggfunc": len, "fill_value": 0}
 612     else:
 613         df["__dummy__"] = values
 614         kwargs = {"aggfunc": aggfunc}
 615 
 616     table = df.pivot_table(
 617         ["__dummy__"],
 618         index=rownames,
 619         columns=colnames,
 620         margins=margins,
 621         margins_name=margins_name,
 622         dropna=dropna,
 623         **kwargs,
 624     )
 625 
 626     # GH18321, after pivoting, an extra top level of column index of `__dummy__` is
 627     # created, and this extra level should not be included in the further steps
 628     if not table.empty:
 629         cols_diff = df.columns.difference(original_df_cols)[0]
 630         table = table[cols_diff]
 631 
 632     # Post-process
 633     if normalize is not False:
 634         table = _normalize(
 635             table, normalize=normalize, margins=margins, margins_name=margins_name
 636         )
 637 
 638     return table

 641 def _normalize(table, normalize, margins: bool, margins_name="All"):
 642 
 643     if not isinstance(normalize, (bool, str)):
 644         axis_subs = {0: "index", 1: "columns"}
 645         try:
 646             normalize = axis_subs[normalize]
 647         except KeyError as err:
 648             raise ValueError("Not a valid normalize argument") from err
 649 
 650     if margins is False:
 651 
 652         # Actual Normalizations
 653         normalizers: Dict[Union[bool, str], Callable] = {
 654             "all": lambda x: x / x.sum(axis=1).sum(axis=0),
 655             "columns": lambda x: x / x.sum(),
 656             "index": lambda x: x.div(x.sum(axis=1), axis=0),
 657         }
 658 
 659         normalizers[True] = normalizers["all"]
 660 
 661         try:
 662             f = normalizers[normalize]
 663         except KeyError as err:
 664             raise ValueError("Not a valid normalize argument") from err
 665 
 666         table = f(table)
 667         table = table.fillna(0)
 668 
 669     elif margins is True:
 670         # keep index and column of pivoted table
 671         table_index = table.index
 672         table_columns = table.columns
 673 
 674         # check if margin name is in (for MI cases) or equal to last
 675         # index/column and save the column and index margin
 676         if (margins_name not in table.iloc[-1, :].name) | (
 677             margins_name != table.iloc[:, -1].name
 678         ):
 679             raise ValueError(f"{margins_name} not in pivoted DataFrame")
 680         column_margin = table.iloc[:-1, -1]
 681         index_margin = table.iloc[-1, :-1]
 682 
 683         # keep the core table
 684         table = table.iloc[:-1, :-1]
 685 
 686         # Normalize core
 687         table = _normalize(table, normalize=normalize, margins=False)
 688 
 689         # Fix Margins
 690         if normalize == "columns":
 691             column_margin = column_margin / column_margin.sum()
 692             table = concat([table, column_margin], axis=1)
 693             table = table.fillna(0)
 694             table.columns = table_columns
 695 
 696         elif normalize == "index":
 697             index_margin = index_margin / index_margin.sum()
 698             table = table.append(index_margin)
 699             table = table.fillna(0)
 700             table.index = table_index
 701 
 702         elif normalize == "all" or normalize is True:
 703             column_margin = column_margin / column_margin.sum()
 704             index_margin = index_margin / index_margin.sum()
 705             index_margin.loc[margins_name] = 1
 706             table = concat([table, column_margin], axis=1)
 707             table = table.append(index_margin)
 708 
 709             table = table.fillna(0)
 710             table.index = table_index
 711             table.columns = table_columns
 712 
 713         else:
 714             raise ValueError("Not a valid normalize argument")
 715 
 716     else:
 717         raise ValueError("Not a valid margins argument")
 718 
 719     return table

 722 def _get_names(arrs, names, prefix: str = "row"):
 723     if names is None:
 724         names = []

 733         if not isinstance(names, list):
 734             names = list(names)
