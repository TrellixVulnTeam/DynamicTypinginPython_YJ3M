  38 class _Unstacker:
  39     """
  40     Helper class to unstack data / pivot with multi-level index
  41 
  42     Parameters
  43     ----------
  44     index : MultiIndex
  45     level : int or str, default last level
  46         Level to "unstack". Accepts a name for the level.
  47     fill_value : scalar, optional
  48         Default value to fill in missing values if subgroups do not have the
  49         same set of labels. By default, missing values will be replaced with
  50         the default fill value for that data type, NaN for float, NaT for
  51         datetimelike, etc. For integer types, by default data will converted to
  52         float and missing values will be set to NaN.
  53     constructor : object
  54         Pandas ``DataFrame`` or subclass used to create unstacked
  55         response.  If None, DataFrame will be used.
  56 
  57     Examples
  58     --------
  59     >>> index = pd.MultiIndex.from_tuples([('one', 'a'), ('one', 'b'),
  60     ...                                    ('two', 'a'), ('two', 'b')])
  61     >>> s = pd.Series(np.arange(1, 5, dtype=np.int64), index=index)
  62     >>> s
  63     one  a    1
  64          b    2
  65     two  a    3
  66          b    4
  67     dtype: int64
  68 
  69     >>> s.unstack(level=-1)
  70          a  b
  71     one  1  2
  72     two  3  4
  73 
  74     >>> s.unstack(level=0)
  75        one  two
  76     a    1    3
  77     b    2    4
  78 
  79     Returns
  80     -------
  81     unstacked : DataFrame
  82     """
  83 
  84     def __init__(
  85         self, index: MultiIndex, level=-1, constructor=None,
  86     ):
  87 
  88         if constructor is None:
  89             constructor = DataFrame
  90         self.constructor = constructor
  91 
  92         self.index = index.remove_unused_levels()
  93 
  94         self.level = self.index._get_level_number(level)
  95 
  96         # when index includes `nan`, need to lift levels/strides by 1
  97         self.lift = 1 if -1 in self.index.codes[self.level] else 0
  98 
  99         # Note: the "pop" below alters these in-place.
 100         self.new_index_levels = list(self.index.levels)
 101         self.new_index_names = list(self.index.names)
 102 
 103         self.removed_name = self.new_index_names.pop(self.level)
 104         self.removed_level = self.new_index_levels.pop(self.level)
 105         self.removed_level_full = index.levels[self.level]
 106 
 107         # Bug fix GH 20601
 108         # If the data frame is too big, the number of unique index combination
 109         # will cause int32 overflow on windows environments.
 110         # We want to check and raise an error before this happens
 111         num_rows = np.max([index_level.size for index_level in self.new_index_levels])
 112         num_columns = self.removed_level.size
 113 
 114         # GH20601: This forces an overflow if the number of cells is too high.
 115         num_cells = np.multiply(num_rows, num_columns, dtype=np.int32)
 116 
 117         if num_rows > 0 and num_columns > 0 and num_cells <= 0:
 118             raise ValueError("Unstacked DataFrame is too big, causing int32 overflow")
 119 
 120         self._make_selectors()
 121 
 122     @cache_readonly
 123     def _indexer_and_to_sort(self):
 124         v = self.level
 125 
 126         codes = list(self.index.codes)
 127         levs = list(self.index.levels)
 128         to_sort = codes[:v] + codes[v + 1 :] + [codes[v]]
 129         sizes = [len(x) for x in levs[:v] + levs[v + 1 :] + [levs[v]]]
 130 
 131         comp_index, obs_ids = get_compressed_ids(to_sort, sizes)
 132         ngroups = len(obs_ids)
 133 
 134         indexer = libalgos.groupsort_indexer(comp_index, ngroups)[0]
 135         indexer = ensure_platform_int(indexer)
 136 
 137         return indexer, to_sort

 139     @cache_readonly
 140     def sorted_labels(self):
 141         indexer, to_sort = self._indexer_and_to_sort
 142         return [l.take(indexer) for l in to_sort]

 144     def _make_sorted_values(self, values: np.ndarray) -> np.ndarray:
 145         indexer, _ = self._indexer_and_to_sort
 146 
 147         sorted_values = algos.take_nd(values, indexer, axis=0)
 148         return sorted_values

 150     def _make_selectors(self):
 151         new_levels = self.new_index_levels
 152 
 153         # make the mask
 154         remaining_labels = self.sorted_labels[:-1]
 155         level_sizes = [len(x) for x in new_levels]
 156 
 157         comp_index, obs_ids = get_compressed_ids(remaining_labels, level_sizes)
 158         ngroups = len(obs_ids)
 159 
 160         comp_index = ensure_platform_int(comp_index)
 161         stride = self.index.levshape[self.level] + self.lift
 162         self.full_shape = ngroups, stride
 163 
 164         selector = self.sorted_labels[-1] + stride * comp_index + self.lift
 165         mask = np.zeros(np.prod(self.full_shape), dtype=bool)
 166         mask.put(selector, True)
 167 
 168         if mask.sum() < len(self.index):
 169             raise ValueError("Index contains duplicate entries, cannot reshape")
 170 
 171         self.group_index = comp_index
 172         self.mask = mask
 173         self.unique_groups = obs_ids
 174         self.compressor = comp_index.searchsorted(np.arange(ngroups))

 176     def get_result(self, values, value_columns, fill_value):
 177 
 178         if values.ndim == 1:
 179             values = values[:, np.newaxis]
 180 
 181         if value_columns is None and values.shape[1] != 1:  # pragma: no cover
 182             raise ValueError("must pass column labels for multi-column data")
 183 
 184         values, _ = self.get_new_values(values, fill_value)
 185         columns = self.get_new_columns(value_columns)
 186         index = self.new_index
 187 
 188         return self.constructor(values, index=index, columns=columns)

 190     def get_new_values(self, values, fill_value=None):
 191 
 192         if values.ndim == 1:
 193             values = values[:, np.newaxis]
 194 
 195         sorted_values = self._make_sorted_values(values)
 196 
 197         # place the values
 198         length, width = self.full_shape
 199         stride = values.shape[1]
 200         result_width = width * stride
 201         result_shape = (length, result_width)
 202         mask = self.mask
 203         mask_all = mask.all()
 204 
 205         # we can simply reshape if we don't have a mask
 206         if mask_all and len(values):
 207             # TODO: Under what circumstances can we rely on sorted_values
 208             #  matching values?  When that holds, we can slice instead
 209             #  of take (in particular for EAs)
 210             new_values = (
 211                 sorted_values.reshape(length, width, stride)
 212                 .swapaxes(1, 2)
 213                 .reshape(result_shape)
 214             )
 215             new_mask = np.ones(result_shape, dtype=bool)
 216             return new_values, new_mask
 217 
 218         # if our mask is all True, then we can use our existing dtype
 219         if mask_all:
 220             dtype = values.dtype
 221             new_values = np.empty(result_shape, dtype=dtype)
 222         else:
 223             dtype, fill_value = maybe_promote(values.dtype, fill_value)
 224             new_values = np.empty(result_shape, dtype=dtype)
 225             new_values.fill(fill_value)
 226 
 227         new_mask = np.zeros(result_shape, dtype=bool)
 228 
 229         name = np.dtype(dtype).name
 230 
 231         # we need to convert to a basic dtype
 232         # and possibly coerce an input to our output dtype
 233         # e.g. ints -> floats
 234         if needs_i8_conversion(values.dtype):
 235             sorted_values = sorted_values.view("i8")
 236             new_values = new_values.view("i8")
 237         elif is_bool_dtype(values.dtype):
 238             sorted_values = sorted_values.astype("object")
 239             new_values = new_values.astype("object")
 240         else:
 241             sorted_values = sorted_values.astype(name, copy=False)
 242 
 243         # fill in our values & mask
 244         libreshape.unstack(
 245             sorted_values,
 246             mask.view("u1"),
 247             stride,
 248             length,
 249             width,
 250             new_values,
 251             new_mask.view("u1"),
 252         )
 253 
 254         # reconstruct dtype if needed
 255         if needs_i8_conversion(values.dtype):
 256             new_values = new_values.view(values.dtype)
 257 
 258         return new_values, new_mask
 259 
 260     def get_new_columns(self, value_columns):
 261         if value_columns is None:
 262             if self.lift == 0:
 263                 return self.removed_level._shallow_copy(name=self.removed_name)
 264 
 265             lev = self.removed_level.insert(0, item=self.removed_level._na_value)
 266             return lev.rename(self.removed_name)
 267 
 268         stride = len(self.removed_level) + self.lift
 269         width = len(value_columns)
 270         propagator = np.repeat(np.arange(width), stride)
 271         if isinstance(value_columns, MultiIndex):
 272             new_levels = value_columns.levels + (self.removed_level_full,)
 273             new_names = value_columns.names + (self.removed_name,)
 274 
 275             new_codes = [lab.take(propagator) for lab in value_columns.codes]
 276         else:
 277             new_levels = [value_columns, self.removed_level_full]
 278             new_names = [value_columns.name, self.removed_name]
 279             new_codes = [propagator]
 280 
 281         # The two indices differ only if the unstacked level had unused items:
 282         if len(self.removed_level_full) != len(self.removed_level):
 283             # In this case, we remap the new codes to the original level:
 284             repeater = self.removed_level_full.get_indexer(self.removed_level)
 285             if self.lift:
 286                 repeater = np.insert(repeater, 0, -1)
 287         else:
 288             # Otherwise, we just use each level item exactly once:
 289             repeater = np.arange(stride) - self.lift
 290 
 291         # The entire level is then just a repetition of the single chunk:
 292         new_codes.append(np.tile(repeater, width))
 293         return MultiIndex(
 294             levels=new_levels, codes=new_codes, names=new_names, verify_integrity=False
 295         )

 297     @cache_readonly
 298     def new_index(self):
 299         # Does not depend on values or value_columns
 300         result_codes = [lab.take(self.compressor) for lab in self.sorted_labels[:-1]]
 301 
 302         # construct the new index
 303         if len(self.new_index_levels) == 1:
 304             level, level_codes = self.new_index_levels[0], result_codes[0]
 305             if (level_codes == -1).any():
 306                 level = level.insert(len(level), level._na_value)
 307             return level.take(level_codes).rename(self.new_index_names[0])
 308 
 309         return MultiIndex(
 310             levels=self.new_index_levels,
 311             codes=result_codes,
 312             names=self.new_index_names,
 313             verify_integrity=False,
 314         )

 317 def _unstack_multiple(data, clocs, fill_value=None):
 318     if len(clocs) == 0:
 319         return data
 320 
 321     # NOTE: This doesn't deal with hierarchical columns yet
 322 
 323     index = data.index
 324 
 325     # GH 19966 Make sure if MultiIndexed index has tuple name, they will be
 326     # recognised as a whole
 327     if clocs in index.names:
 328         clocs = [clocs]
 329     clocs = [index._get_level_number(i) for i in clocs]
 330 
 331     rlocs = [i for i in range(index.nlevels) if i not in clocs]
 332 
 333     clevels = [index.levels[i] for i in clocs]
 334     ccodes = [index.codes[i] for i in clocs]
 335     cnames = [index.names[i] for i in clocs]
 336     rlevels = [index.levels[i] for i in rlocs]
 337     rcodes = [index.codes[i] for i in rlocs]
 338     rnames = [index.names[i] for i in rlocs]
 339 
 340     shape = [len(x) for x in clevels]
 341     group_index = get_group_index(ccodes, shape, sort=False, xnull=False)
 342 
 343     comp_ids, obs_ids = compress_group_index(group_index, sort=False)
 344     recons_codes = decons_obs_group_ids(comp_ids, obs_ids, shape, ccodes, xnull=False)
 345 
 346     if not rlocs:
 347         # Everything is in clocs, so the dummy df has a regular index
 348         dummy_index = Index(obs_ids, name="__placeholder__")
 349     else:
 350         dummy_index = MultiIndex(
 351             levels=rlevels + [obs_ids],
 352             codes=rcodes + [comp_ids],
 353             names=rnames + ["__placeholder__"],
 354             verify_integrity=False,
 355         )
 356 
 357     if isinstance(data, Series):
 358         dummy = data.copy()
 359         dummy.index = dummy_index
 360 
 361         unstacked = dummy.unstack("__placeholder__", fill_value=fill_value)
 362         new_levels = clevels
 363         new_names = cnames
 364         new_codes = recons_codes
 365     else:
 366         if isinstance(data.columns, MultiIndex):
 367             result = data
 368             for i in range(len(clocs)):
 369                 val = clocs[i]
 370                 result = result.unstack(val, fill_value=fill_value)
 371                 clocs = [v if v < val else v - 1 for v in clocs]
 372 
 373             return result
 374 
 375         dummy = data.copy()
 376         dummy.index = dummy_index
 377 
 378         unstacked = dummy.unstack("__placeholder__", fill_value=fill_value)
 379         if isinstance(unstacked, Series):
 380             unstcols = unstacked.index
 381         else:
 382             unstcols = unstacked.columns
 383         assert isinstance(unstcols, MultiIndex)  # for mypy
 384         new_levels = [unstcols.levels[0]] + clevels
 385         new_names = [data.columns.name] + cnames
 386 
 387         new_codes = [unstcols.codes[0]]
 388         for rec in recons_codes:
 389             new_codes.append(rec.take(unstcols.codes[-1]))
 390 
 391     new_columns = MultiIndex(
 392         levels=new_levels, codes=new_codes, names=new_names, verify_integrity=False
 393     )
 394 
 395     if isinstance(unstacked, Series):
 396         unstacked.index = new_columns
 397     else:
 398         unstacked.columns = new_columns
 399 
 400     return unstacked

 403 def unstack(obj, level, fill_value=None):
 404     if isinstance(level, (tuple, list)):
 405         if len(level) != 1:
 406             # _unstack_multiple only handles MultiIndexes,
 407             # and isn't needed for a single level
 408             return _unstack_multiple(obj, level, fill_value=fill_value)
 409         else:
 410             level = level[0]
 411 
 412     # Prioritize integer interpretation (GH #21677):
 413     if not is_integer(level) and not level == "__placeholder__":
 414         level = obj.index._get_level_number(level)
 415 
 416     if isinstance(obj, DataFrame):
 417         if isinstance(obj.index, MultiIndex):
 418             return _unstack_frame(obj, level, fill_value=fill_value)
 419         else:
 420             return obj.T.stack(dropna=False)
 421     else:
 422         if is_extension_array_dtype(obj.dtype):
 423             return _unstack_extension_series(obj, level, fill_value)
 424         unstacker = _Unstacker(
 425             obj.index, level=level, constructor=obj._constructor_expanddim,
 426         )
 427         return unstacker.get_result(
 428             obj.values, value_columns=None, fill_value=fill_value
 429         )

 432 def _unstack_frame(obj, level, fill_value=None):
 433     if not obj._can_fast_transpose:
 434         unstacker = _Unstacker(obj.index, level=level)
 435         mgr = obj._mgr.unstack(unstacker, fill_value=fill_value)
 436         return obj._constructor(mgr)
 437     else:
 438         return _Unstacker(
 439             obj.index, level=level, constructor=obj._constructor,
 440         ).get_result(obj._values, value_columns=obj.columns, fill_value=fill_value)

 443 def _unstack_extension_series(series, level, fill_value):
 444     """
 445     Unstack an ExtensionArray-backed Series.
 446 
 447     The ExtensionDtype is preserved.
 448 
 449     Parameters
 450     ----------
 451     series : Series
 452         A Series with an ExtensionArray for values
 453     level : Any
 454         The level name or number.
 455     fill_value : Any
 456         The user-level (not physical storage) fill value to use for
 457         missing values introduced by the reshape. Passed to
 458         ``series.values.take``.
 459 
 460     Returns
 461     -------
 462     DataFrame
 463         Each column of the DataFrame will have the same dtype as
 464         the input Series.
 465     """
 466     # Defer to the logic in ExtensionBlock._unstack
 467     df = series.to_frame()
 468     result = df.unstack(level=level, fill_value=fill_value)
 469     return result.droplevel(level=0, axis=1)

 472 def stack(frame, level=-1, dropna=True):
 473     """
 474     Convert DataFrame to Series with multi-level Index. Columns become the
 475     second level of the resulting hierarchical index
 476 
 477     Returns
 478     -------
 479     stacked : Series
 480     """
 481 
 482     def factorize(index):
 483         if index.is_unique:
 484             return index, np.arange(len(index))
 485         codes, categories = factorize_from_iterable(index)
 486         return categories, codes
 487 
 488     N, K = frame.shape
 489 
 490     # Will also convert negative level numbers and check if out of bounds.
 491     level_num = frame.columns._get_level_number(level)
 492 
 493     if isinstance(frame.columns, MultiIndex):
 494         return _stack_multi_columns(frame, level_num=level_num, dropna=dropna)
 495     elif isinstance(frame.index, MultiIndex):
 496         new_levels = list(frame.index.levels)
 497         new_codes = [lab.repeat(K) for lab in frame.index.codes]
 498 
 499         clev, clab = factorize(frame.columns)
 500         new_levels.append(clev)
 501         new_codes.append(np.tile(clab, N).ravel())
 502 
 503         new_names = list(frame.index.names)
 504         new_names.append(frame.columns.name)
 505         new_index = MultiIndex(
 506             levels=new_levels, codes=new_codes, names=new_names, verify_integrity=False
 507         )
 508     else:
 509         levels, (ilab, clab) = zip(*map(factorize, (frame.index, frame.columns)))
 510         codes = ilab.repeat(K), np.tile(clab, N).ravel()
 511         new_index = MultiIndex(
 512             levels=levels,
 513             codes=codes,
 514             names=[frame.index.name, frame.columns.name],
 515             verify_integrity=False,
 516         )
 517 
 518     if frame._is_homogeneous_type:
 519         # For homogeneous EAs, frame._values will coerce to object. So
 520         # we concatenate instead.
 521         dtypes = list(frame.dtypes._values)
 522         dtype = dtypes[0]
 523 
 524         if is_extension_array_dtype(dtype):
 525             arr = dtype.construct_array_type()
 526             new_values = arr._concat_same_type(
 527                 [col._values for _, col in frame.items()]
 528             )
 529             new_values = _reorder_for_extension_array_stack(new_values, N, K)
 530         else:
 531             # homogeneous, non-EA
 532             new_values = frame._values.ravel()
 533 
 534     else:
 535         # non-homogeneous
 536         new_values = frame._values.ravel()
 537 
 538     if dropna:
 539         mask = notna(new_values)
 540         new_values = new_values[mask]
 541         new_index = new_index[mask]
 542 
 543     return frame._constructor_sliced(new_values, index=new_index)

 546 def stack_multiple(frame, level, dropna=True):
 547     # If all passed levels match up to column names, no
 548     # ambiguity about what to do
 549     if all(lev in frame.columns.names for lev in level):
 550         result = frame
 551         for lev in level:
 552             result = stack(result, lev, dropna=dropna)
 553 
 554     # Otherwise, level numbers may change as each successive level is stacked
 555     elif all(isinstance(lev, int) for lev in level):
 556         # As each stack is done, the level numbers decrease, so we need
 557         #  to account for that when level is a sequence of ints
 558         result = frame
 559         # _get_level_number() checks level numbers are in range and converts
 560         # negative numbers to positive
 561         level = [frame.columns._get_level_number(lev) for lev in level]
 562 
 563         # Can't iterate directly through level as we might need to change
 564         # values as we go
 565         for index in range(len(level)):
 566             lev = level[index]
 567             result = stack(result, lev, dropna=dropna)
 568             # Decrement all level numbers greater than current, as these
 569             # have now shifted down by one
 570             updated_level = []
 571             for other in level:
 572                 if other > lev:
 573                     updated_level.append(other - 1)
 574                 else:
 575                     updated_level.append(other)
 576             level = updated_level
 577 
 578     else:
 579         raise ValueError(
 580             "level should contain all level names or all level "
 581             "numbers, not a mixture of the two."
 582         )
 583 
 584     return result

 587 def _stack_multi_columns(frame, level_num=-1, dropna=True):
 588     def _convert_level_number(level_num, columns):
 589         """
 590         Logic for converting the level number to something we can safely pass
 591         to swaplevel:
 592 
 593         We generally want to convert the level number into a level name, except
 594         when columns do not have names, in which case we must leave as a level
 595         number
 596         """
 597         if level_num in columns.names:
 598             return columns.names[level_num]
 599         else:
 600             if columns.names[level_num] is None:
 601                 return level_num
 602             else:
 603                 return columns.names[level_num]
 604 
 605     this = frame.copy()
 606 
 607     # this makes life much simpler
 608     if level_num != frame.columns.nlevels - 1:
 609         # roll levels to put selected level at end
 610         roll_columns = this.columns
 611         for i in range(level_num, frame.columns.nlevels - 1):
 612             # Need to check if the ints conflict with level names
 613             lev1 = _convert_level_number(i, roll_columns)
 614             lev2 = _convert_level_number(i + 1, roll_columns)
 615             roll_columns = roll_columns.swaplevel(lev1, lev2)
 616         this.columns = roll_columns
 617 
 618     if not this.columns.is_lexsorted():
 619         # Workaround the edge case where 0 is one of the column names,
 620         # which interferes with trying to sort based on the first
 621         # level
 622         level_to_sort = _convert_level_number(0, this.columns)
 623         this = this.sort_index(level=level_to_sort, axis=1)
 624 
 625     # tuple list excluding level for grouping columns
 626     if len(frame.columns.levels) > 2:
 627         tuples = list(
 628             zip(
 629                 *[
 630                     lev.take(level_codes)
 631                     for lev, level_codes in zip(
 632                         this.columns.levels[:-1], this.columns.codes[:-1]
 633                     )
 634                 ]
 635             )
 636         )
 637         unique_groups = [key for key, _ in itertools.groupby(tuples)]
 638         new_names = this.columns.names[:-1]
 639         new_columns = MultiIndex.from_tuples(unique_groups, names=new_names)
 640     else:
 641         new_columns = this.columns.levels[0]._shallow_copy(name=this.columns.names[0])
 642         unique_groups = new_columns
 643 
 644     # time to ravel the values
 645     new_data = {}
 646     level_vals = this.columns.levels[-1]
 647     level_codes = sorted(set(this.columns.codes[-1]))
 648     level_vals_used = level_vals[level_codes]
 649     levsize = len(level_codes)
 650     drop_cols = []
 651     for key in unique_groups:
 652         try:
 653             loc = this.columns.get_loc(key)
 654         except KeyError:
 655             drop_cols.append(key)
 656             continue
 657 
 658         # can make more efficient?
 659         # we almost always return a slice
 660         # but if unsorted can get a boolean
 661         # indexer
 662         if not isinstance(loc, slice):
 663             slice_len = len(loc)
 664         else:
 665             slice_len = loc.stop - loc.start
 666 
 667         if slice_len != levsize:
 668             chunk = this.loc[:, this.columns[loc]]
 669             chunk.columns = level_vals.take(chunk.columns.codes[-1])
 670             value_slice = chunk.reindex(columns=level_vals_used).values
 671         else:
 672             if frame._is_homogeneous_type and is_extension_array_dtype(
 673                 frame.dtypes.iloc[0]
 674             ):
 675                 dtype = this[this.columns[loc]].dtypes.iloc[0]
 676                 subset = this[this.columns[loc]]
 677 
 678                 value_slice = dtype.construct_array_type()._concat_same_type(
 679                     [x._values for _, x in subset.items()]
 680                 )
 681                 N, K = this.shape
 682                 idx = np.arange(N * K).reshape(K, N).T.ravel()
 683                 value_slice = value_slice.take(idx)
 684 
 685             elif frame._is_mixed_type:
 686                 value_slice = this[this.columns[loc]].values
 687             else:
 688                 value_slice = this.values[:, loc]
 689 
 690         if value_slice.ndim > 1:
 691             # i.e. not extension
 692             value_slice = value_slice.ravel()
 693 
 694         new_data[key] = value_slice
 695 
 696     if len(drop_cols) > 0:
 697         new_columns = new_columns.difference(drop_cols)
 698 
 699     N = len(this)
 700 
 701     if isinstance(this.index, MultiIndex):
 702         new_levels = list(this.index.levels)
 703         new_names = list(this.index.names)
 704         new_codes = [lab.repeat(levsize) for lab in this.index.codes]
 705     else:
 706         old_codes, old_levels = factorize_from_iterable(this.index)
 707         new_levels = [old_levels]
 708         new_codes = [old_codes.repeat(levsize)]
 709         new_names = [this.index.name]  # something better?
 710 
 711     new_levels.append(level_vals)
 712     new_codes.append(np.tile(level_codes, N))
 713     new_names.append(frame.columns.names[level_num])
 714 
 715     new_index = MultiIndex(
 716         levels=new_levels, codes=new_codes, names=new_names, verify_integrity=False
 717     )
 718 
 719     result = frame._constructor(new_data, index=new_index, columns=new_columns)
 720 
 721     # more efficient way to go about this? can do the whole masking biz but
 722     # will only save a small amount of time...
 723     if dropna:
 724         result = result.dropna(axis=0, how="all")
 725 
 726     return result

 729 def get_dummies(
 730     data,
 731     prefix=None,
 732     prefix_sep="_",
 733     dummy_na=False,
 734     columns=None,
 735     sparse=False,
 736     drop_first=False,
 737     dtype=None,
 738 ) -> "DataFrame":
 739     """
 740     Convert categorical variable into dummy/indicator variables.
 741 
 742     Parameters
 743     ----------
 744     data : array-like, Series, or DataFrame
 745         Data of which to get dummy indicators.
 746     prefix : str, list of str, or dict of str, default None
 747         String to append DataFrame column names.
 748         Pass a list with length equal to the number of columns
 749         when calling get_dummies on a DataFrame. Alternatively, `prefix`
 750         can be a dictionary mapping column names to prefixes.
 751     prefix_sep : str, default '_'
 752         If appending prefix, separator/delimiter to use. Or pass a
 753         list or dictionary as with `prefix`.
 754     dummy_na : bool, default False
 755         Add a column to indicate NaNs, if False NaNs are ignored.
 756     columns : list-like, default None
 757         Column names in the DataFrame to be encoded.
 758         If `columns` is None then all the columns with
 759         `object` or `category` dtype will be converted.
 760     sparse : bool, default False
 761         Whether the dummy-encoded columns should be backed by
 762         a :class:`SparseArray` (True) or a regular NumPy array (False).
 763     drop_first : bool, default False
 764         Whether to get k-1 dummies out of k categorical levels by removing the
 765         first level.
 766     dtype : dtype, default np.uint8
 767         Data type for new columns. Only a single dtype is allowed.
 768 
 769         .. versionadded:: 0.23.0
 770 
 771     Returns
 772     -------
 773     DataFrame
 774         Dummy-coded data.
 775 
 776     See Also
 777     --------
 778     Series.str.get_dummies : Convert Series to dummy codes.
 779 
 780     Examples
 781     --------
 782     >>> s = pd.Series(list('abca'))
 783 
 784     >>> pd.get_dummies(s)
 785        a  b  c
 786     0  1  0  0
 787     1  0  1  0
 788     2  0  0  1
 789     3  1  0  0
 790 
 791     >>> s1 = ['a', 'b', np.nan]
 792 
 793     >>> pd.get_dummies(s1)
 794        a  b
 795     0  1  0
 796     1  0  1
 797     2  0  0
 798 
 799     >>> pd.get_dummies(s1, dummy_na=True)
 800        a  b  NaN
 801     0  1  0    0
 802     1  0  1    0
 803     2  0  0    1
 804 
 805     >>> df = pd.DataFrame({'A': ['a', 'b', 'a'], 'B': ['b', 'a', 'c'],
 806     ...                    'C': [1, 2, 3]})
 807 
 808     >>> pd.get_dummies(df, prefix=['col1', 'col2'])
 809        C  col1_a  col1_b  col2_a  col2_b  col2_c
 810     0  1       1       0       0       1       0
 811     1  2       0       1       1       0       0
 812     2  3       1       0       0       0       1
 813 
 814     >>> pd.get_dummies(pd.Series(list('abcaa')))
 815        a  b  c
 816     0  1  0  0
 817     1  0  1  0
 818     2  0  0  1
 819     3  1  0  0
 820     4  1  0  0
 821 
 822     >>> pd.get_dummies(pd.Series(list('abcaa')), drop_first=True)
 823        b  c
 824     0  0  0
 825     1  1  0
 826     2  0  1
 827     3  0  0
 828     4  0  0
 829 
 830     >>> pd.get_dummies(pd.Series(list('abc')), dtype=float)
 831          a    b    c
 832     0  1.0  0.0  0.0
 833     1  0.0  1.0  0.0
 834     2  0.0  0.0  1.0
 835     """
 836     from pandas.core.reshape.concat import concat
 837 
 838     dtypes_to_encode = ["object", "category"]
 839 
 840     if isinstance(data, DataFrame):
 841         # determine columns being encoded
 842         if columns is None:
 843             data_to_encode = data.select_dtypes(include=dtypes_to_encode)
 844         elif not is_list_like(columns):
 845             raise TypeError("Input must be a list-like for parameter `columns`")
 846         else:
 847             data_to_encode = data[columns]
 848 
 849         # validate prefixes and separator to avoid silently dropping cols
 850         def check_len(item, name):
 851 
 852             if is_list_like(item):
 853                 if not len(item) == data_to_encode.shape[1]:
 854                     len_msg = (
 855                         f"Length of '{name}' ({len(item)}) did not match the "
 856                         "length of the columns being encoded "
 857                         f"({data_to_encode.shape[1]})."
 858                     )
 859                     raise ValueError(len_msg)
 860 
 861         check_len(prefix, "prefix")
 862         check_len(prefix_sep, "prefix_sep")
 863 
 864         if isinstance(prefix, str):
 865             prefix = itertools.cycle([prefix])
 866         if isinstance(prefix, dict):
 867             prefix = [prefix[col] for col in data_to_encode.columns]
 868 
 869         if prefix is None:
 870             prefix = data_to_encode.columns
 871 
 872         # validate separators
 873         if isinstance(prefix_sep, str):
 874             prefix_sep = itertools.cycle([prefix_sep])
 875         elif isinstance(prefix_sep, dict):
 876             prefix_sep = [prefix_sep[col] for col in data_to_encode.columns]
 877 
 878         with_dummies: List[DataFrame]
 879         if data_to_encode.shape == data.shape:
 880             # Encoding the entire df, do not prepend any dropped columns
 881             with_dummies = []
 882         elif columns is not None:
 883             # Encoding only cols specified in columns. Get all cols not in
 884             # columns to prepend to result.
 885             with_dummies = [data.drop(columns, axis=1)]
 886         else:
 887             # Encoding only object and category dtype columns. Get remaining
 888             # columns to prepend to result.
 889             with_dummies = [data.select_dtypes(exclude=dtypes_to_encode)]
 890 
 891         for (col, pre, sep) in zip(data_to_encode.items(), prefix, prefix_sep):
 892             # col is (column_name, column), use just column data here
 893             dummy = _get_dummies_1d(
 894                 col[1],
 895                 prefix=pre,
 896                 prefix_sep=sep,
 897                 dummy_na=dummy_na,
 898                 sparse=sparse,
 899                 drop_first=drop_first,
 900                 dtype=dtype,
 901             )
 902             with_dummies.append(dummy)
 903         result = concat(with_dummies, axis=1)
 904     else:
 905         result = _get_dummies_1d(
 906             data,
 907             prefix,
 908             prefix_sep,
 909             dummy_na,
 910             sparse=sparse,
 911             drop_first=drop_first,
 912             dtype=dtype,
 913         )
 914     return result

 917 def _get_dummies_1d(
 918     data,
 919     prefix,
 920     prefix_sep="_",
 921     dummy_na=False,
 922     sparse=False,
 923     drop_first=False,
 924     dtype=None,
 925 ):
 926     from pandas.core.reshape.concat import concat
 927 
 928     # Series avoids inconsistent NaN handling
 929     codes, levels = factorize_from_iterable(Series(data))
 930 
 931     if dtype is None:
 932         dtype = np.uint8
 933     dtype = np.dtype(dtype)
 934 
 935     if is_object_dtype(dtype):
 936         raise ValueError("dtype=object is not a valid dtype for get_dummies")
 937 
 938     def get_empty_frame(data) -> DataFrame:
 939         if isinstance(data, Series):
 940             index = data.index
 941         else:
 942             index = np.arange(len(data))
 943         return DataFrame(index=index)
 944 
 945     # if all NaN
 946     if not dummy_na and len(levels) == 0:
 947         return get_empty_frame(data)
 948 
 949     codes = codes.copy()
 950     if dummy_na:
 951         codes[codes == -1] = len(levels)
 952         levels = np.append(levels, np.nan)
 953 
 954     # if dummy_na, we just fake a nan level. drop_first will drop it again
 955     if drop_first and len(levels) == 1:
 956         return get_empty_frame(data)
 957 
 958     number_of_cols = len(levels)
 959 
 960     if prefix is None:
 961         dummy_cols = levels
 962     else:
 963         dummy_cols = [f"{prefix}{prefix_sep}{level}" for level in levels]
 964 
 965     index: Optional[Index]
 966     if isinstance(data, Series):
 967         index = data.index
 968     else:
 969         index = None
 970 
 971     if sparse:
 972 
 973         fill_value: Union[bool, float, int]
 974         if is_integer_dtype(dtype):
 975             fill_value = 0
 976         elif dtype == bool:
 977             fill_value = False
 978         else:
 979             fill_value = 0.0
 980 
 981         sparse_series = []
 982         N = len(data)
 983         sp_indices: List[List] = [[] for _ in range(len(dummy_cols))]
 984         mask = codes != -1
 985         codes = codes[mask]
 986         n_idx = np.arange(N)[mask]
 987 
 988         for ndx, code in zip(n_idx, codes):
 989             sp_indices[code].append(ndx)
 990 
 991         if drop_first:
 992             # remove first categorical level to avoid perfect collinearity
 993             # GH12042
 994             sp_indices = sp_indices[1:]
 995             dummy_cols = dummy_cols[1:]
 996         for col, ixs in zip(dummy_cols, sp_indices):
 997             sarr = SparseArray(
 998                 np.ones(len(ixs), dtype=dtype),
 999                 sparse_index=IntIndex(N, ixs),
1000                 fill_value=fill_value,
1001                 dtype=dtype,
1002             )
1003             sparse_series.append(Series(data=sarr, index=index, name=col))
1004 
1005         out = concat(sparse_series, axis=1, copy=False)
1006         return out
1007 
1008     else:
1009         dummy_mat = np.eye(number_of_cols, dtype=dtype).take(codes, axis=0)
1010 
1011         if not dummy_na:
1012             # reset NaN GH4446
1013             dummy_mat[codes == -1] = 0
1014 
1015         if drop_first:
1016             # remove first GH12042
1017             dummy_mat = dummy_mat[:, 1:]
1018             dummy_cols = dummy_cols[1:]
1019         return DataFrame(dummy_mat, index=index, columns=dummy_cols)

1022 def _reorder_for_extension_array_stack(arr, n_rows: int, n_columns: int):
1023     """
1024     Re-orders the values when stacking multiple extension-arrays.
1025 
1026     The indirect stacking method used for EAs requires a followup
1027     take to get the order correct.
1028 
1029     Parameters
1030     ----------
1031     arr : ExtensionArray
1032     n_rows, n_columns : int
1033         The number of rows and columns in the original DataFrame.
1034 
1035     Returns
1036     -------
1037     taken : ExtensionArray
1038         The original `arr` with elements re-ordered appropriately
1039 
1040     Examples
1041     --------
1042     >>> arr = np.array(['a', 'b', 'c', 'd', 'e', 'f'])
1043     >>> _reorder_for_extension_array_stack(arr, 2, 3)
1044     array(['a', 'c', 'e', 'b', 'd', 'f'], dtype='<U1')
1045 
1046     >>> _reorder_for_extension_array_stack(arr, 3, 2)
1047     array(['a', 'd', 'b', 'e', 'c', 'f'], dtype='<U1')
1048     """
1049     # final take to get the order correct.
1050     # idx is an indexer like
1051     # [c0r0, c1r0, c2r0, ...,
1052     #  c0r1, c1r1, c2r1, ...]
1053     idx = np.arange(n_rows * n_columns).reshape(n_columns, n_rows).T.ravel()
1054     return arr.take(idx)
