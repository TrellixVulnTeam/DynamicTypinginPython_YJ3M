  67 def concat(
  68     objs: Union[Iterable[FrameOrSeries], Mapping[Label, FrameOrSeries]],
  69     axis=0,
  70     join="outer",
  71     ignore_index: bool = False,
  72     keys=None,
  73     levels=None,
  74     names=None,
  75     verify_integrity: bool = False,
  76     sort: bool = False,
  77     copy: bool = True,
  78 ) -> FrameOrSeriesUnion:
  79     """
  80     Concatenate pandas objects along a particular axis with optional set logic
  81     along the other axes.
  82 
  83     Can also add a layer of hierarchical indexing on the concatenation axis,
  84     which may be useful if the labels are the same (or overlapping) on
  85     the passed axis number.
  86 
  87     Parameters
  88     ----------
  89     objs : a sequence or mapping of Series or DataFrame objects
  90         If a mapping is passed, the sorted keys will be used as the `keys`
  91         argument, unless it is passed, in which case the values will be
  92         selected (see below). Any None objects will be dropped silently unless
  93         they are all None in which case a ValueError will be raised.
  94     axis : {0/'index', 1/'columns'}, default 0
  95         The axis to concatenate along.
  96     join : {'inner', 'outer'}, default 'outer'
  97         How to handle indexes on other axis (or axes).
  98     ignore_index : bool, default False
  99         If True, do not use the index values along the concatenation axis. The
 100         resulting axis will be labeled 0, ..., n - 1. This is useful if you are
 101         concatenating objects where the concatenation axis does not have
 102         meaningful indexing information. Note the index values on the other
 103         axes are still respected in the join.
 104     keys : sequence, default None
 105         If multiple levels passed, should contain tuples. Construct
 106         hierarchical index using the passed keys as the outermost level.
 107     levels : list of sequences, default None
 108         Specific levels (unique values) to use for constructing a
 109         MultiIndex. Otherwise they will be inferred from the keys.
 110     names : list, default None
 111         Names for the levels in the resulting hierarchical index.
 112     verify_integrity : bool, default False
 113         Check whether the new concatenated axis contains duplicates. This can
 114         be very expensive relative to the actual data concatenation.
 115     sort : bool, default False
 116         Sort non-concatenation axis if it is not already aligned when `join`
 117         is 'outer'.
 118         This has no effect when ``join='inner'``, which already preserves
 119         the order of the non-concatenation axis.
 120 
 121         .. versionadded:: 0.23.0
 122         .. versionchanged:: 1.0.0
 123 
 124            Changed to not sort by default.
 125 
 126     copy : bool, default True
 127         If False, do not copy data unnecessarily.
 128 
 129     Returns
 130     -------
 131     object, type of objs
 132         When concatenating all ``Series`` along the index (axis=0), a
 133         ``Series`` is returned. When ``objs`` contains at least one
 134         ``DataFrame``, a ``DataFrame`` is returned. When concatenating along
 135         the columns (axis=1), a ``DataFrame`` is returned.
 136 
 137     See Also
 138     --------
 139     Series.append : Concatenate Series.
 140     DataFrame.append : Concatenate DataFrames.
 141     DataFrame.join : Join DataFrames using indexes.
 142     DataFrame.merge : Merge DataFrames by indexes or columns.
 143 
 144     Notes
 145     -----
 146     The keys, levels, and names arguments are all optional.
 147 
 148     A walkthrough of how this method fits in with other tools for combining
 149     pandas objects can be found `here
 150     <https://pandas.pydata.org/pandas-docs/stable/user_guide/merging.html>`__.
 151 
 152     Examples
 153     --------
 154     Combine two ``Series``.
 155 
 156     >>> s1 = pd.Series(['a', 'b'])
 157     >>> s2 = pd.Series(['c', 'd'])
 158     >>> pd.concat([s1, s2])
 159     0    a
 160     1    b
 161     0    c
 162     1    d
 163     dtype: object
 164 
 165     Clear the existing index and reset it in the result
 166     by setting the ``ignore_index`` option to ``True``.
 167 
 168     >>> pd.concat([s1, s2], ignore_index=True)
 169     0    a
 170     1    b
 171     2    c
 172     3    d
 173     dtype: object
 174 
 175     Add a hierarchical index at the outermost level of
 176     the data with the ``keys`` option.
 177 
 178     >>> pd.concat([s1, s2], keys=['s1', 's2'])
 179     s1  0    a
 180         1    b
 181     s2  0    c
 182         1    d
 183     dtype: object
 184 
 185     Label the index keys you create with the ``names`` option.
 186 
 187     >>> pd.concat([s1, s2], keys=['s1', 's2'],
 188     ...           names=['Series name', 'Row ID'])
 189     Series name  Row ID
 190     s1           0         a
 191                  1         b
 192     s2           0         c
 193                  1         d
 194     dtype: object
 195 
 196     Combine two ``DataFrame`` objects with identical columns.
 197 
 198     >>> df1 = pd.DataFrame([['a', 1], ['b', 2]],
 199     ...                    columns=['letter', 'number'])
 200     >>> df1
 201       letter  number
 202     0      a       1
 203     1      b       2
 204     >>> df2 = pd.DataFrame([['c', 3], ['d', 4]],
 205     ...                    columns=['letter', 'number'])
 206     >>> df2
 207       letter  number
 208     0      c       3
 209     1      d       4
 210     >>> pd.concat([df1, df2])
 211       letter  number
 212     0      a       1
 213     1      b       2
 214     0      c       3
 215     1      d       4
 216 
 217     Combine ``DataFrame`` objects with overlapping columns
 218     and return everything. Columns outside the intersection will
 219     be filled with ``NaN`` values.
 220 
 221     >>> df3 = pd.DataFrame([['c', 3, 'cat'], ['d', 4, 'dog']],
 222     ...                    columns=['letter', 'number', 'animal'])
 223     >>> df3
 224       letter  number animal
 225     0      c       3    cat
 226     1      d       4    dog
 227     >>> pd.concat([df1, df3], sort=False)
 228       letter  number animal
 229     0      a       1    NaN
 230     1      b       2    NaN
 231     0      c       3    cat
 232     1      d       4    dog
 233 
 234     Combine ``DataFrame`` objects with overlapping columns
 235     and return only those that are shared by passing ``inner`` to
 236     the ``join`` keyword argument.
 237 
 238     >>> pd.concat([df1, df3], join="inner")
 239       letter  number
 240     0      a       1
 241     1      b       2
 242     0      c       3
 243     1      d       4
 244 
 245     Combine ``DataFrame`` objects horizontally along the x axis by
 246     passing in ``axis=1``.
 247 
 248     >>> df4 = pd.DataFrame([['bird', 'polly'], ['monkey', 'george']],
 249     ...                    columns=['animal', 'name'])
 250     >>> pd.concat([df1, df4], axis=1)
 251       letter  number  animal    name
 252     0      a       1    bird   polly
 253     1      b       2  monkey  george
 254 
 255     Prevent the result from including duplicate index values with the
 256     ``verify_integrity`` option.
 257 
 258     >>> df5 = pd.DataFrame([1], index=['a'])
 259     >>> df5
 260        0
 261     a  1
 262     >>> df6 = pd.DataFrame([2], index=['a'])
 263     >>> df6
 264        0
 265     a  2
 266     >>> pd.concat([df5, df6], verify_integrity=True)
 267     Traceback (most recent call last):
 268         ...
 269     ValueError: Indexes have overlapping values: ['a']
 270     """
 271     op = _Concatenator(
 272         objs,
 273         axis=axis,
 274         ignore_index=ignore_index,
 275         join=join,
 276         keys=keys,
 277         levels=levels,
 278         names=names,
 279         verify_integrity=verify_integrity,
 280         copy=copy,
 281         sort=sort,
 282     )
 283 
 284     return op.get_result()

 287 class _Concatenator:
 288     """
 289     Orchestrates a concatenation operation for BlockManagers
 290     """
 291 
 292     def __init__(
 293         self,
 294         objs,
 295         axis=0,
 296         join: str = "outer",
 297         keys=None,
 298         levels=None,
 299         names=None,
 300         ignore_index: bool = False,
 301         verify_integrity: bool = False,
 302         copy: bool = True,
 303         sort=False,
 304     ):
 305         if isinstance(objs, (NDFrame, str)):
 306             raise TypeError(
 307                 "first argument must be an iterable of pandas "
 308                 f'objects, you passed an object of type "{type(objs).__name__}"'
 309             )
 310 
 311         if join == "outer":
 312             self.intersect = False
 313         elif join == "inner":
 314             self.intersect = True
 315         else:  # pragma: no cover
 316             raise ValueError(
 317                 "Only can inner (intersect) or outer (union) join the other axis"
 318             )
 319 
 320         if isinstance(objs, abc.Mapping):
 321             if keys is None:
 322                 keys = list(objs.keys())
 323             objs = [objs[k] for k in keys]
 324         else:
 325             objs = list(objs)
 326 
 327         if len(objs) == 0:
 328             raise ValueError("No objects to concatenate")
 329 
 330         if keys is None:
 331             objs = list(com.not_none(*objs))
 332         else:
 333             # #1649
 334             clean_keys = []
 335             clean_objs = []
 336             for k, v in zip(keys, objs):
 337                 if v is None:
 338                     continue
 339                 clean_keys.append(k)
 340                 clean_objs.append(v)
 341             objs = clean_objs
 342             name = getattr(keys, "name", None)
 343             keys = Index(clean_keys, name=name)
 344 
 345         if len(objs) == 0:
 346             raise ValueError("All objects passed were None")
 347 
 348         # consolidate data & figure out what our result ndim is going to be
 349         ndims = set()
 350         for obj in objs:
 351             if not isinstance(obj, (Series, DataFrame)):
 352                 msg = (
 353                     f"cannot concatenate object of type '{type(obj)}'; "
 354                     "only Series and DataFrame objs are valid"
 355                 )
 356                 raise TypeError(msg)
 357 
 358             # consolidate
 359             obj._consolidate(inplace=True)
 360             ndims.add(obj.ndim)
 361 
 362         # get the sample
 363         # want the highest ndim that we have, and must be non-empty
 364         # unless all objs are empty
 365         sample = None
 366         if len(ndims) > 1:
 367             max_ndim = max(ndims)
 368             for obj in objs:
 369                 if obj.ndim == max_ndim and np.sum(obj.shape):
 370                     sample = obj
 371                     break
 372 
 373         else:
 374             # filter out the empties if we have not multi-index possibilities
 375             # note to keep empty Series as it affect to result columns / name
 376             non_empties = [
 377                 obj for obj in objs if sum(obj.shape) > 0 or isinstance(obj, Series)
 378             ]
 379 
 380             if len(non_empties) and (
 381                 keys is None and names is None and levels is None and not self.intersect
 382             ):
 383                 objs = non_empties
 384                 sample = objs[0]
 385 
 386         if sample is None:
 387             sample = objs[0]
 388         self.objs = objs
 389 
 390         # Standardize axis parameter to int
 391         if isinstance(sample, Series):
 392             axis = DataFrame._get_axis_number(axis)
 393         else:
 394             axis = sample._get_axis_number(axis)
 395 
 396         # Need to flip BlockManager axis in the DataFrame special case
 397         self._is_frame = isinstance(sample, ABCDataFrame)
 398         if self._is_frame:
 399             axis = DataFrame._get_block_manager_axis(axis)
 400 
 401         self._is_series = isinstance(sample, ABCSeries)
 402         if not 0 <= axis <= sample.ndim:
 403             raise AssertionError(
 404                 f"axis must be between 0 and {sample.ndim}, input was {axis}"
 405             )
 406 
 407         # if we have mixed ndims, then convert to highest ndim
 408         # creating column numbers as needed
 409         if len(ndims) > 1:
 410             current_column = 0
 411             max_ndim = sample.ndim
 412             self.objs, objs = [], self.objs
 413             for obj in objs:
 414 
 415                 ndim = obj.ndim
 416                 if ndim == max_ndim:
 417                     pass
 418 
 419                 elif ndim != max_ndim - 1:
 420                     raise ValueError(
 421                         "cannot concatenate unaligned mixed "
 422                         "dimensional NDFrame objects"
 423                     )
 424 
 425                 else:
 426                     name = getattr(obj, "name", None)
 427                     if ignore_index or name is None:
 428                         name = current_column
 429                         current_column += 1
 430 
 431                     # doing a row-wise concatenation so need everything
 432                     # to line up
 433                     if self._is_frame and axis == 1:
 434                         name = 0
 435                     obj = sample._constructor({name: obj})
 436 
 437                 self.objs.append(obj)
 438 
 439         # note: this is the BlockManager axis (since DataFrame is transposed)
 440         self.bm_axis = axis
 441         self.axis = 1 - self.bm_axis if self._is_frame else 0
 442         self.keys = keys
 443         self.names = names or getattr(keys, "names", None)
 444         self.levels = levels
 445         self.sort = sort
 446 
 447         self.ignore_index = ignore_index
 448         self.verify_integrity = verify_integrity
 449         self.copy = copy
 450 
 451         self.new_axes = self._get_new_axes()
 452 
 453     def get_result(self):
 454 
 455         # series only
 456         if self._is_series:
 457 
 458             # stack blocks
 459             if self.bm_axis == 0:
 460                 name = com.consensus_name_attr(self.objs)
 461                 cons = self.objs[0]._constructor
 462 
 463                 arrs = [ser._values for ser in self.objs]
 464 
 465                 res = concat_compat(arrs, axis=0)
 466                 result = cons(res, index=self.new_axes[0], name=name, dtype=res.dtype)
 467                 return result.__finalize__(self, method="concat")
 468 
 469             # combine as columns in a frame
 470             else:
 471                 data = dict(zip(range(len(self.objs)), self.objs))
 472 
 473                 # GH28330 Preserves subclassed objects through concat
 474                 cons = self.objs[0]._constructor_expanddim
 475 
 476                 index, columns = self.new_axes
 477                 df = cons(data, index=index)
 478                 df.columns = columns
 479                 return df.__finalize__(self, method="concat")
 480 
 481         # combine block managers
 482         else:
 483             mgrs_indexers = []
 484             for obj in self.objs:
 485                 indexers = {}
 486                 for ax, new_labels in enumerate(self.new_axes):
 487                     # ::-1 to convert BlockManager ax to DataFrame ax
 488                     if ax == self.bm_axis:
 489                         # Suppress reindexing on concat axis
 490                         continue
 491 
 492                     # 1-ax to convert BlockManager axis to DataFrame axis
 493                     obj_labels = obj.axes[1 - ax]
 494                     if not new_labels.equals(obj_labels):
 495                         indexers[ax] = obj_labels.reindex(new_labels)[1]
 496 
 497                 mgrs_indexers.append((obj._mgr, indexers))
 498 
 499             new_data = concatenate_block_managers(
 500                 mgrs_indexers, self.new_axes, concat_axis=self.bm_axis, copy=self.copy
 501             )
 502             if not self.copy:
 503                 new_data._consolidate_inplace()
 504 
 505             cons = self.objs[0]._constructor
 506             return cons(new_data).__finalize__(self, method="concat")
 507 
 508     def _get_result_dim(self) -> int:
 509         if self._is_series and self.bm_axis == 1:
 510             return 2
 511         else:
 512             return self.objs[0].ndim
 513 
 514     def _get_new_axes(self) -> List[Index]:
 515         ndim = self._get_result_dim()
 516         return [
 517             self._get_concat_axis() if i == self.bm_axis else self._get_comb_axis(i)
 518             for i in range(ndim)
 519         ]

 521     def _get_comb_axis(self, i: int) -> Index:
 522         data_axis = self.objs[0]._get_block_manager_axis(i)
 523         return get_objs_combined_axis(
 524             self.objs,
 525             axis=data_axis,
 526             intersect=self.intersect,
 527             sort=self.sort,
 528             copy=self.copy,
 529         )

 531     def _get_concat_axis(self) -> Index:
 532         """
 533         Return index to be used along concatenation axis.
 534         """
 535         if self._is_series:
 536             if self.bm_axis == 0:
 537                 indexes = [x.index for x in self.objs]
 538             elif self.ignore_index:
 539                 idx = ibase.default_index(len(self.objs))
 540                 return idx
 541             elif self.keys is None:
 542                 names: List[Label] = [None] * len(self.objs)
 543                 num = 0
 544                 has_names = False
 545                 for i, x in enumerate(self.objs):
 546                     if not isinstance(x, Series):
 547                         raise TypeError(
 548                             f"Cannot concatenate type 'Series' with "
 549                             f"object of type '{type(x).__name__}'"
 550                         )
 551                     if x.name is not None:
 552                         names[i] = x.name
 553                         has_names = True
 554                     else:
 555                         names[i] = num
 556                         num += 1
 557                 if has_names:
 558                     return Index(names)
 559                 else:
 560                     return ibase.default_index(len(self.objs))
 561             else:
 562                 return ensure_index(self.keys).set_names(self.names)
 563         else:
 564             indexes = [x.axes[self.axis] for x in self.objs]
 565 
 566         if self.ignore_index:
 567             idx = ibase.default_index(sum(len(i) for i in indexes))
 568             return idx
 569 
 570         if self.keys is None:
 571             concat_axis = _concat_indexes(indexes)
 572         else:
 573             concat_axis = _make_concat_multiindex(
 574                 indexes, self.keys, self.levels, self.names
 575             )
 576 
 577         self._maybe_check_integrity(concat_axis)
 578 
 579         return concat_axis
 580 
 581     def _maybe_check_integrity(self, concat_index: Index):
 582         if self.verify_integrity:
 583             if not concat_index.is_unique:
 584                 overlap = concat_index[concat_index.duplicated()].unique()
 585                 raise ValueError(f"Indexes have overlapping values: {overlap}")
 586 
 587 
 588 def _concat_indexes(indexes) -> Index:
 589     return indexes[0].append(indexes[1:])

 592 def _make_concat_multiindex(indexes, keys, levels=None, names=None) -> MultiIndex:
 593 
 594     if (levels is None and isinstance(keys[0], tuple)) or (
 595         levels is not None and len(levels) > 1
 596     ):
 597         zipped = list(zip(*keys))
 598         if names is None:
 599             names = [None] * len(zipped)
 600 
 601         if levels is None:
 602             _, levels = factorize_from_iterables(zipped)
 603         else:
 604             levels = [ensure_index(x) for x in levels]
 605     else:
 606         zipped = [keys]
 607         if names is None:
 608             names = [None]
 609 
 610         if levels is None:
 611             levels = [ensure_index(keys)]
 612         else:
 613             levels = [ensure_index(x) for x in levels]
 614 
 615     if not all_indexes_same(indexes):
 616         codes_list = []
 617 
 618         # things are potentially different sizes, so compute the exact codes
 619         # for each level and pass those to MultiIndex.from_arrays
 620 
 621         for hlevel, level in zip(zipped, levels):
 622             to_concat = []
 623             for key, index in zip(hlevel, indexes):
 624                 mask = level == key
 625                 if not mask.any():
 626                     raise ValueError(f"Key {key} not in level {level}")
 627                 i = np.nonzero(level == key)[0][0]
 628 
 629                 to_concat.append(np.repeat(i, len(index)))
 630             codes_list.append(np.concatenate(to_concat))
 631 
 632         concat_index = _concat_indexes(indexes)
 633 
 634         # these go at the end
 635         if isinstance(concat_index, MultiIndex):
 636             levels.extend(concat_index.levels)
 637             codes_list.extend(concat_index.codes)
 638         else:
 639             codes, categories = factorize_from_iterable(concat_index)
 640             levels.append(categories)
 641             codes_list.append(codes)
 642 
 643         if len(names) == len(levels):
 644             names = list(names)
 645         else:
 646             # make sure that all of the passed indices have the same nlevels
 647             if not len({idx.nlevels for idx in indexes}) == 1:
 648                 raise AssertionError(
 649                     "Cannot concat indices that do not have the same number of levels"
 650                 )
 651 
 652             # also copies
 653             names = names + get_consensus_names(indexes)
 654 
 655         return MultiIndex(
 656             levels=levels, codes=codes_list, names=names, verify_integrity=False
 657         )
 658 
 659     new_index = indexes[0]
 660     n = len(new_index)
 661     kpieces = len(indexes)
 662 
 663     # also copies
 664     new_names = list(names)
 665     new_levels = list(levels)
 666 
 667     # construct codes
 668     new_codes = []
 669 
 670     # do something a bit more speedy
 671 
 672     for hlevel, level in zip(zipped, levels):
 673         hlevel = ensure_index(hlevel)
 674         mapped = level.get_indexer(hlevel)
 675 
 676         mask = mapped == -1
 677         if mask.any():
 678             raise ValueError(f"Values not found in passed level: {hlevel[mask]!s}")
 679 
 680         new_codes.append(np.repeat(mapped, n))
 681 
 682     if isinstance(new_index, MultiIndex):
 683         new_levels.extend(new_index.levels)
 684         new_codes.extend([np.tile(lab, kpieces) for lab in new_index.codes])
 685     else:
 686         new_levels.append(new_index)
 687         new_codes.append(np.tile(np.arange(n), kpieces))
 688 
 689     if len(new_names) < len(new_levels):
 690         new_names.extend(new_index.names)
 691 
 692     return MultiIndex(
 693         levels=new_levels, codes=new_codes, names=new_names, verify_integrity=False
 694     )
