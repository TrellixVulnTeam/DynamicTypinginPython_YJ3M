  20 @Appender(
  21     _shared_docs["melt"]
  22     % dict(caller="pd.melt(df, ", versionadded="", other="DataFrame.melt")
  23 )
  24 def melt(
  25     frame: DataFrame,
  26     id_vars=None,
  27     value_vars=None,
  28     var_name=None,
  29     value_name="value",
  30     col_level=None,
  31 ) -> DataFrame:
  32     # TODO: what about the existing index?
  33     # If multiindex, gather names of columns on all level for checking presence
  34     # of `id_vars` and `value_vars`
  35     if isinstance(frame.columns, MultiIndex):
  36         cols = [x for c in frame.columns for x in c]
  37     else:
  38         cols = list(frame.columns)
  39 
  40     if id_vars is not None:
  41         if not is_list_like(id_vars):
  42             id_vars = [id_vars]
  43         elif isinstance(frame.columns, MultiIndex) and not isinstance(id_vars, list):
  44             raise ValueError(
  45                 "id_vars must be a list of tuples when columns are a MultiIndex"
  46             )
  47         else:
  48             # Check that `id_vars` are in frame
  49             id_vars = list(id_vars)
  50             missing = Index(com.flatten(id_vars)).difference(cols)
  51             if not missing.empty:
  52                 raise KeyError(
  53                     "The following 'id_vars' are not present "
  54                     f"in the DataFrame: {list(missing)}"
  55                 )
  56     else:
  57         id_vars = []
  58 
  59     if value_vars is not None:
  60         if not is_list_like(value_vars):
  61             value_vars = [value_vars]
  62         elif isinstance(frame.columns, MultiIndex) and not isinstance(value_vars, list):
  63             raise ValueError(
  64                 "value_vars must be a list of tuples when columns are a MultiIndex"
  65             )
  66         else:
  67             value_vars = list(value_vars)
  68             # Check that `value_vars` are in frame
  69             missing = Index(com.flatten(value_vars)).difference(cols)
  70             if not missing.empty:
  71                 raise KeyError(
  72                     "The following 'value_vars' are not present in "
  73                     f"the DataFrame: {list(missing)}"
  74                 )
  75         if col_level is not None:
  76             idx = frame.columns.get_level_values(col_level).get_indexer(
  77                 id_vars + value_vars
  78             )
  79         else:
  80             idx = frame.columns.get_indexer(id_vars + value_vars)
  81         frame = frame.iloc[:, idx]
  82     else:
  83         frame = frame.copy()
  84 
  85     if col_level is not None:  # allow list or other?
  86         # frame is a copy
  87         frame.columns = frame.columns.get_level_values(col_level)
  88 
  89     if var_name is None:
  90         if isinstance(frame.columns, MultiIndex):
  91             if len(frame.columns.names) == len(set(frame.columns.names)):
  92                 var_name = frame.columns.names
  93             else:
  94                 var_name = [f"variable_{i}" for i in range(len(frame.columns.names))]
  95         else:
  96             var_name = [
  97                 frame.columns.name if frame.columns.name is not None else "variable"
  98             ]
  99     if isinstance(var_name, str):
 100         var_name = [var_name]
 101 
 102     N, K = frame.shape
 103     K -= len(id_vars)
 104 
 105     mdata = {}
 106     for col in id_vars:
 107         id_data = frame.pop(col)
 108         if is_extension_array_dtype(id_data):
 109             id_data = concat([id_data] * K, ignore_index=True)
 110         else:
 111             id_data = np.tile(id_data._values, K)
 112         mdata[col] = id_data
 113 
 114     mcolumns = id_vars + var_name + [value_name]
 115 
 116     mdata[value_name] = frame._values.ravel("F")
 117     for i, col in enumerate(var_name):
 118         # asanyarray will keep the columns as an Index
 119         mdata[col] = np.asanyarray(frame.columns._get_level_values(i)).repeat(N)
 120 
 121     return frame._constructor(mdata, columns=mcolumns)

 124 @deprecate_kwarg(old_arg_name="label", new_arg_name=None)
 125 def lreshape(data: DataFrame, groups, dropna: bool = True, label=None) -> DataFrame:
 126     """
 127     Reshape long-format data to wide. Generalized inverse of DataFrame.pivot
 128 
 129     Parameters
 130     ----------
 131     data : DataFrame
 132     groups : dict
 133         {new_name : list_of_columns}
 134     dropna : boolean, default True
 135 
 136     Examples
 137     --------
 138     >>> data = pd.DataFrame({'hr1': [514, 573], 'hr2': [545, 526],
 139     ...                      'team': ['Red Sox', 'Yankees'],
 140     ...                      'year1': [2007, 2007], 'year2': [2008, 2008]})
 141     >>> data
 142        hr1  hr2     team  year1  year2
 143     0  514  545  Red Sox   2007   2008
 144     1  573  526  Yankees   2007   2008
 145 
 146     >>> pd.lreshape(data, {'year': ['year1', 'year2'], 'hr': ['hr1', 'hr2']})
 147           team  year   hr
 148     0  Red Sox  2007  514
 149     1  Yankees  2007  573
 150     2  Red Sox  2008  545
 151     3  Yankees  2008  526
 152 
 153     Returns
 154     -------
 155     reshaped : DataFrame
 156     """
 157     if isinstance(groups, dict):
 158         keys = list(groups.keys())
 159         values = list(groups.values())
 160     else:
 161         keys, values = zip(*groups)
 162 
 163     all_cols = list(set.union(*[set(x) for x in values]))
 164     id_cols = list(data.columns.difference(all_cols))
 165 
 166     K = len(values[0])
 167 
 168     for seq in values:
 169         if len(seq) != K:
 170             raise ValueError("All column lists must be same length")
 171 
 172     mdata = {}
 173     pivot_cols = []
 174 
 175     for target, names in zip(keys, values):
 176         to_concat = [data[col]._values for col in names]
 177 
 178         mdata[target] = concat_compat(to_concat)
 179         pivot_cols.append(target)
 180 
 181     for col in id_cols:
 182         mdata[col] = np.tile(data[col]._values, K)
 183 
 184     if dropna:
 185         mask = np.ones(len(mdata[pivot_cols[0]]), dtype=bool)
 186         for c in pivot_cols:
 187             mask &= notna(mdata[c])
 188         if not mask.all():
 189             mdata = {k: v[mask] for k, v in mdata.items()}
 190 
 191     return data._constructor(mdata, columns=id_cols + pivot_cols)

 194 def wide_to_long(
 195     df: DataFrame, stubnames, i, j, sep: str = "", suffix: str = r"\d+"
 196 ) -> DataFrame:
 197     r"""
 198     Wide panel to long format. Less flexible but more user-friendly than melt.
 199 
 200     With stubnames ['A', 'B'], this function expects to find one or more
 201     group of columns with format
 202     A-suffix1, A-suffix2,..., B-suffix1, B-suffix2,...
 203     You specify what you want to call this suffix in the resulting long format
 204     with `j` (for example `j='year'`)
 205 
 206     Each row of these wide variables are assumed to be uniquely identified by
 207     `i` (can be a single column name or a list of column names)
 208 
 209     All remaining variables in the data frame are left intact.
 210 
 211     Parameters
 212     ----------
 213     df : DataFrame
 214         The wide-format DataFrame.
 215     stubnames : str or list-like
 216         The stub name(s). The wide format variables are assumed to
 217         start with the stub names.
 218     i : str or list-like
 219         Column(s) to use as id variable(s).
 220     j : str
 221         The name of the sub-observation variable. What you wish to name your
 222         suffix in the long format.
 223     sep : str, default ""
 224         A character indicating the separation of the variable names
 225         in the wide format, to be stripped from the names in the long format.
 226         For example, if your column names are A-suffix1, A-suffix2, you
 227         can strip the hyphen by specifying `sep='-'`.
 228     suffix : str, default '\\d+'
 229         A regular expression capturing the wanted suffixes. '\\d+' captures
 230         numeric suffixes. Suffixes with no numbers could be specified with the
 231         negated character class '\\D+'. You can also further disambiguate
 232         suffixes, for example, if your wide variables are of the form
 233         A-one, B-two,.., and you have an unrelated column A-rating, you can
 234         ignore the last one by specifying `suffix='(!?one|two)'`.
 235 
 236         .. versionchanged:: 0.23.0
 237             When all suffixes are numeric, they are cast to int64/float64.
 238 
 239     Returns
 240     -------
 241     DataFrame
 242         A DataFrame that contains each stub name as a variable, with new index
 243         (i, j).
 244 
 245     Notes
 246     -----
 247     All extra variables are left untouched. This simply uses
 248     `pandas.melt` under the hood, but is hard-coded to "do the right thing"
 249     in a typical case.
 250 
 251     Examples
 252     --------
 253     >>> np.random.seed(123)
 254     >>> df = pd.DataFrame({"A1970" : {0 : "a", 1 : "b", 2 : "c"},
 255     ...                    "A1980" : {0 : "d", 1 : "e", 2 : "f"},
 256     ...                    "B1970" : {0 : 2.5, 1 : 1.2, 2 : .7},
 257     ...                    "B1980" : {0 : 3.2, 1 : 1.3, 2 : .1},
 258     ...                    "X"     : dict(zip(range(3), np.random.randn(3)))
 259     ...                   })
 260     >>> df["id"] = df.index
 261     >>> df
 262       A1970 A1980  B1970  B1980         X  id
 263     0     a     d    2.5    3.2 -1.085631   0
 264     1     b     e    1.2    1.3  0.997345   1
 265     2     c     f    0.7    0.1  0.282978   2
 266     >>> pd.wide_to_long(df, ["A", "B"], i="id", j="year")
 267     ... # doctest: +NORMALIZE_WHITESPACE
 268                     X  A    B
 269     id year
 270     0  1970 -1.085631  a  2.5
 271     1  1970  0.997345  b  1.2
 272     2  1970  0.282978  c  0.7
 273     0  1980 -1.085631  d  3.2
 274     1  1980  0.997345  e  1.3
 275     2  1980  0.282978  f  0.1
 276 
 277     With multiple id columns
 278 
 279     >>> df = pd.DataFrame({
 280     ...     'famid': [1, 1, 1, 2, 2, 2, 3, 3, 3],
 281     ...     'birth': [1, 2, 3, 1, 2, 3, 1, 2, 3],
 282     ...     'ht1': [2.8, 2.9, 2.2, 2, 1.8, 1.9, 2.2, 2.3, 2.1],
 283     ...     'ht2': [3.4, 3.8, 2.9, 3.2, 2.8, 2.4, 3.3, 3.4, 2.9]
 284     ... })
 285     >>> df
 286        famid  birth  ht1  ht2
 287     0      1      1  2.8  3.4
 288     1      1      2  2.9  3.8
 289     2      1      3  2.2  2.9
 290     3      2      1  2.0  3.2
 291     4      2      2  1.8  2.8
 292     5      2      3  1.9  2.4
 293     6      3      1  2.2  3.3
 294     7      3      2  2.3  3.4
 295     8      3      3  2.1  2.9
 296     >>> l = pd.wide_to_long(df, stubnames='ht', i=['famid', 'birth'], j='age')

 420     def get_var_names(df, stub: str, sep: str, suffix: str) -> List[str]:
 421         regex = fr"^{re.escape(stub)}{re.escape(sep)}{suffix}$"
 422         pattern = re.compile(regex)
 423         return [col for col in df.columns if pattern.match(col)]
 424 
 425     def melt_stub(df, stub: str, i, j, value_vars, sep: str):
 426         newdf = melt(
 427             df,
 428             id_vars=i,
 429             value_vars=value_vars,
 430             value_name=stub.rstrip(sep),
 431             var_name=j,
 432         )
 433         newdf[j] = Categorical(newdf[j])
 434         newdf[j] = newdf[j].str.replace(re.escape(stub + sep), "")
 435 
 436         # GH17627 Cast numerics suffixes to int/float
 437         newdf[j] = to_numeric(newdf[j], errors="ignore")
 438 
 439         return newdf.set_index(i + [j])
 440 
 441     if not is_list_like(stubnames):
 442         stubnames = [stubnames]
 443     else:
 444         stubnames = list(stubnames)
 445 
 446     if any(col in stubnames for col in df.columns):
 447         raise ValueError("stubname can't be identical to a column name")
 448 
 449     if not is_list_like(i):
 450         i = [i]
 451     else:
 452         i = list(i)
 453 
 454     if df[i].duplicated().any():
 455         raise ValueError("the id variables need to uniquely identify each row")
 456 
 457     value_vars = [get_var_names(df, stub, sep, suffix) for stub in stubnames]
 458 
 459     value_vars_flattened = [e for sublist in value_vars for e in sublist]
 460     id_vars = list(set(df.columns.tolist()).difference(value_vars_flattened))
 461 
 462     _melted = [melt_stub(df, s, i, j, v, sep) for s, v in zip(stubnames, value_vars)]
 463     melted = _melted[0].join(_melted[1:], how="outer")
 464 
 465     if len(i) == 1:
 466         new = df[id_vars].set_index(i).join(melted)
 467         return new
 468 
 469     new = df[id_vars].merge(melted.reset_index(), on=i).set_index(i + [j])
 470 
 471     return new
