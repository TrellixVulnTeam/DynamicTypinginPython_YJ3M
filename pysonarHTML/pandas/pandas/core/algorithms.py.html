  69 def _ensure_data(
  70     values, dtype: Optional[DtypeObj] = None
  71 ) -> Tuple[np.ndarray, DtypeObj]:
  72     """
  73     routine to ensure that our data is of the correct
  74     input dtype for lower-level routines
  75 
  76     This will coerce:
  77     - ints -> int64
  78     - uint -> uint64
  79     - bool -> uint64 (TODO this should be uint8)
  80     - datetimelike -> i8
  81     - datetime64tz -> i8 (in local tz)
  82     - categorical -> codes
  83 
  84     Parameters
  85     ----------
  86     values : array-like
  87     dtype : pandas_dtype, optional
  88         coerce to this dtype
  89 
  90     Returns
  91     -------
  92     values : ndarray
  93     pandas_dtype : np.dtype or ExtensionDtype
  94     """
  95 
  96     if not isinstance(values, ABCMultiIndex):
  97         # extract_array would raise
  98         values = extract_array(values, extract_numpy=True)
  99 
 100     # we check some simple dtypes first
 101     if is_object_dtype(dtype):
 102         return ensure_object(np.asarray(values)), np.dtype("object")
 103     elif is_object_dtype(values) and dtype is None:
 104         return ensure_object(np.asarray(values)), np.dtype("object")
 105 
 106     try:
 107         if is_bool_dtype(values) or is_bool_dtype(dtype):
 108             # we are actually coercing to uint64
 109             # until our algos support uint8 directly (see TODO)
 110             return np.asarray(values).astype("uint64"), np.dtype("bool")
 111         elif is_signed_integer_dtype(values) or is_signed_integer_dtype(dtype):
 112             return ensure_int64(values), np.dtype("int64")
 113         elif is_unsigned_integer_dtype(values) or is_unsigned_integer_dtype(dtype):
 114             return ensure_uint64(values), np.dtype("uint64")
 115         elif is_float_dtype(values) or is_float_dtype(dtype):
 116             return ensure_float64(values), np.dtype("float64")
 117         elif is_complex_dtype(values) or is_complex_dtype(dtype):
 118 
 119             # ignore the fact that we are casting to float
 120             # which discards complex parts
 121             with catch_warnings():
 122                 simplefilter("ignore", np.ComplexWarning)
 123                 values = ensure_float64(values)
 124             return values, np.dtype("float64")
 125 
 126     except (TypeError, ValueError, OverflowError):
 127         # if we are trying to coerce to a dtype
 128         # and it is incompatible this will fall through to here
 129         return ensure_object(values), np.dtype("object")
 130 
 131     # datetimelike
 132     vals_dtype = getattr(values, "dtype", None)
 133     if needs_i8_conversion(vals_dtype) or needs_i8_conversion(dtype):
 134         if is_period_dtype(vals_dtype) or is_period_dtype(dtype):
 135             from pandas import PeriodIndex
 136 
 137             values = PeriodIndex(values)
 138             dtype = values.dtype
 139         elif is_timedelta64_dtype(vals_dtype) or is_timedelta64_dtype(dtype):
 140             from pandas import TimedeltaIndex
 141 
 142             values = TimedeltaIndex(values)
 143             dtype = values.dtype
 144         else:
 145             # Datetime
 146             if values.ndim > 1 and is_datetime64_ns_dtype(vals_dtype):
 147                 # Avoid calling the DatetimeIndex constructor as it is 1D only
 148                 # Note: this is reached by DataFrame.rank calls GH#27027
 149                 # TODO(EA2D): special case not needed with 2D EAs
 150                 asi8 = values.view("i8")
 151                 dtype = values.dtype
 152                 return asi8, dtype
 153 
 154             from pandas import DatetimeIndex
 155 
 156             values = DatetimeIndex(values)
 157             dtype = values.dtype
 158 
 159         return values.asi8, dtype
 160 
 161     elif is_categorical_dtype(vals_dtype) and (
 162         is_categorical_dtype(dtype) or dtype is None
 163     ):
 164         values = values.codes
 165         dtype = pandas_dtype("category")
 166 
 167         # we are actually coercing to int64
 168         # until our algos support int* directly (not all do)
 169         values = ensure_int64(values)
 170 
 171         return values, dtype
 172 
 173     # we have failed, return object
 174     values = np.asarray(values, dtype=np.object)
 175     return ensure_object(values), np.dtype("object")

 178 def _reconstruct_data(
 179     values: ArrayLike, dtype: DtypeObj, original: AnyArrayLike
 180 ) -> ArrayLike:
 181     """
 182     reverse of _ensure_data
 183 
 184     Parameters
 185     ----------
 186     values : np.ndarray or ExtensionArray
 187     dtype : np.ndtype or ExtensionDtype
 188     original : AnyArrayLike
 189 
 190     Returns
 191     -------
 192     ExtensionArray or np.ndarray
 193     """
 194     if is_extension_array_dtype(dtype):
 195         values = dtype.construct_array_type()._from_sequence(values)
 196     elif is_bool_dtype(dtype):
 197         values = values.astype(dtype, copy=False)
 198 
 199         # we only support object dtypes bool Index
 200         if isinstance(original, ABCIndexClass):
 201             values = values.astype(object, copy=False)
 202     elif dtype is not None:
 203         if is_datetime64_dtype(dtype):
 204             dtype = "datetime64[ns]"
 205         elif is_timedelta64_dtype(dtype):
 206             dtype = "timedelta64[ns]"
 207 
 208         values = values.astype(dtype, copy=False)
 209 
 210     return values

 213 def _ensure_arraylike(values):
 214     """
 215     ensure that we are arraylike if not already
 216     """
 217     if not is_array_like(values):
 218         inferred = lib.infer_dtype(values, skipna=False)

 220             if isinstance(values, tuple):

 275 
 276     Parameters
 277     ----------
 278     values : ndarray
 279     ndtype : str
 280

 282     -------
 283     str

 301 def unique(values):
 302     """
 303     Hash table-based unique. Uniques are returned in order
 304     of appearance. This does NOT sort.
 305 
 306     Significantly faster than numpy.unique. Includes NA values.
 307 
 308     Parameters
 309     ----------
 310     values : 1d array-like
 311 
 312     Returns
 313     -------
 314     numpy.ndarray or ExtensionArray
 315 
 316         The return can be:
 317 
 318         * Index : when the input is an Index
 319         * Categorical : when the input is a Categorical dtype
 320         * ndarray : when the input is a Series/ndarray
 321 
 322         Return numpy.ndarray or ExtensionArray.
 323 
 324     See Also
 325     --------
 326     Index.unique : Return unique values from an Index.
 327     Series.unique : Return unique values of Series object.
 328 
 329     Examples
 330     --------
 331     >>> pd.unique(pd.Series([2, 1, 3, 3]))
 332     array([2, 1, 3])
 333 
 334     >>> pd.unique(pd.Series([2] + [1] * 5))
 335     array([2, 1])
 336 
 337     >>> pd.unique(pd.Series([pd.Timestamp('20160101'),
 338     ...                     pd.Timestamp('20160101')]))
 339     array(['2016-01-01T00:00:00.000000000'], dtype='datetime64[ns]')
 340 
 341     >>> pd.unique(pd.Series([pd.Timestamp('20160101', tz='US/Eastern'),
 342     ...                      pd.Timestamp('20160101', tz='US/Eastern')]))
 343     array([Timestamp('2016-01-01 00:00:00-0500', tz='US/Eastern')],
 344           dtype=object)
 345 
 346     >>> pd.unique(pd.Index([pd.Timestamp('20160101', tz='US/Eastern'),
 347     ...                     pd.Timestamp('20160101', tz='US/Eastern')]))
 348     DatetimeIndex(['2016-01-01 00:00:00-05:00'],
 349     ...           dtype='datetime64[ns, US/Eastern]', freq=None)
 350 
 351     >>> pd.unique(list('baabc'))
 352     array(['b', 'a', 'c'], dtype=object)
 353 
 354     An unordered Categorical will return categories in the
 355     order of appearance.
 356 
 357     >>> pd.unique(pd.Series(pd.Categorical(list('baabc'))))
 358     [b, a, c]
 359     Categories (3, object): [b, a, c]
 360 
 361     >>> pd.unique(pd.Series(pd.Categorical(list('baabc'),
 362     ...                                    categories=list('abc'))))
 363     [b, a, c]
 364     Categories (3, object): [b, a, c]
 365 
 366     An ordered Categorical preserves the category ordering.
 367 
 368     >>> pd.unique(pd.Series(pd.Categorical(list('baabc'),
 369     ...                                    categories=list('abc'),
 370     ...                                    ordered=True)))
 371     [b, a, c]
 372     Categories (3, object): [a < b < c]
 373 
 374     An array of tuples
 375 
 376     >>> pd.unique([('a', 'b'), ('b', 'a'), ('a', 'c'), ('b', 'a')])
 377     array([('a', 'b'), ('b', 'a'), ('a', 'c')], dtype=object)
 378     """
 379     values = _ensure_arraylike(values)
 380 
 381     if is_extension_array_dtype(values):
 382         # Dispatch to extension dtype's unique.
 383         return values.unique()
 384 
 385     original = values
 386     htable, values = _get_hashtable_algo(values)
 387 
 388     table = htable(len(values))
 389     uniques = table.unique(values)
 390     uniques = _reconstruct_data(uniques, original.dtype, original)
 391     return uniques

 397 def isin(comps: AnyArrayLike, values: AnyArrayLike) -> np.ndarray:
 398     """
 399     Compute the isin boolean array.
 400 
 401     Parameters
 402     ----------
 403     comps : array-like
 404     values : array-like
 405 
 406     Returns
 407     -------
 408     ndarray[bool]
 409         Same length as `comps`.
 410     """
 411     if not is_list_like(comps):
 412         raise TypeError(
 413             "only list-like objects are allowed to be passed "
 414             f"to isin(), you passed a [{type(comps).__name__}]"
 415         )
 416     if not is_list_like(values):
 417         raise TypeError(
 418             "only list-like objects are allowed to be passed "
 419             f"to isin(), you passed a [{type(values).__name__}]"
 420         )
 421 
 422     if not isinstance(values, (ABCIndex, ABCSeries, ABCExtensionArray, np.ndarray)):
 423         values = construct_1d_object_array_from_listlike(list(values))
 424         # TODO: could use ensure_arraylike here
 425 
 426     comps = extract_array(comps, extract_numpy=True)
 427     if is_categorical_dtype(comps):
 428         # TODO(extension)
 429         # handle categoricals
 430         return comps.isin(values)  # type: ignore
 431 
 432     comps, dtype = _ensure_data(comps)
 433     values, _ = _ensure_data(values, dtype=dtype)
 434 
 435     # faster for larger cases to use np.in1d
 436     f = htable.ismember_object
 437 
 438     # GH16012
 439     # Ensure np.in1d doesn't get object types or it *may* throw an exception
 440     if len(comps) > 1_000_000 and not is_object_dtype(comps):
 441         f = np.in1d
 442     elif is_integer_dtype(comps):
 443         try:
 444             values = values.astype("int64", copy=False)
 445             comps = comps.astype("int64", copy=False)
 446             f = htable.ismember_int64
 447         except (TypeError, ValueError, OverflowError):
 448             values = values.astype(object)
 449             comps = comps.astype(object)
 450 
 451     elif is_float_dtype(comps):
 452         try:
 453             values = values.astype("float64", copy=False)
 454             comps = comps.astype("float64", copy=False)
 455             f = htable.ismember_float64
 456         except (TypeError, ValueError):
 457             values = values.astype(object)
 458             comps = comps.astype(object)
 459 
 460     return f(comps, values)

 463 def _factorize_array(
 464     values, na_sentinel: int = -1, size_hint=None, na_value=None, mask=None,
 465 ) -> Tuple[np.ndarray, np.ndarray]:
 466     """
 467     Factorize an array-like to codes and uniques.
 468 
 469     This doesn't do any coercion of types or unboxing before factorization.
 470 
 471     Parameters
 472     ----------
 473     values : ndarray
 474     na_sentinel : int, default -1
 475     size_hint : int, optional
 476         Passed through to the hashtable's 'get_labels' method
 477     na_value : object, optional
 478         A value in `values` to consider missing. Note: only use this
 479         parameter when you know that you don't have any values pandas would
 480         consider missing in the array (NaN for float data, iNaT for
 481         datetimes, etc.).
 482     mask : ndarray[bool], optional
 483         If not None, the mask is used as indicator for missing values
 484         (True = missing, False = valid) instead of `na_value` or
 485         condition "val != val".
 486 
 487     Returns
 488     -------
 489     codes : ndarray
 490     uniques : ndarray
 491     """
 492     hash_klass, values = _get_data_algo(values)
 493 
 494     table = hash_klass(size_hint or len(values))
 495     uniques, codes = table.factorize(
 496         values, na_sentinel=na_sentinel, na_value=na_value, mask=mask
 497     )
 498 
 499     codes = ensure_platform_int(codes)
 500     return codes, uniques

 504     values=dedent(
 505         """\

 514         Sort `uniques` and shuffle `codes` to maintain the

 525 def factorize(
 526     values,
 527     sort: bool = False,
 528     na_sentinel: int = -1,
 529     size_hint: Optional[int] = None,
 530     dropna: bool = True,
 531 ) -> Tuple[np.ndarray, Union[np.ndarray, ABCIndex]]:
 532     """
 533     Encode the object as an enumerated type or categorical variable.
 534 
 535     This method is useful for obtaining a numeric representation of an
 536     array when all that matters is identifying distinct values. `factorize`
 537     is available as both a top-level function :func:`pandas.factorize`,
 538     and as a method :meth:`Series.factorize` and :meth:`Index.factorize`.
 539 
 540     Parameters
 541     ----------
 542     {values}{sort}
 543     na_sentinel : int, default -1
 544         Value to mark "not found".
 545     {size_hint}\
 546 
 547     Returns
 548     -------
 549     codes : ndarray
 550         An integer ndarray that's an indexer into `uniques`.
 551         ``uniques.take(codes)`` will have the same values as `values`.
 552     uniques : ndarray, Index, or Categorical
 553         The unique valid values. When `values` is Categorical, `uniques`
 554         is a Categorical. When `values` is some other pandas object, an
 555         `Index` is returned. Otherwise, a 1-D ndarray is returned.
 556 
 557         .. note ::
 558 
 559            Even if there's a missing value in `values`, `uniques` will
 560            *not* contain an entry for it.
 561 
 562     See Also
 563     --------
 564     cut : Discretize continuous-valued array.
 565     unique : Find the unique value in an array.
 566 
 567     Examples
 568     --------
 569     These examples all show factorize as a top-level method like
 570     ``pd.factorize(values)``. The results are identical for methods like
 571     :meth:`Series.factorize`.
 572 
 573     >>> codes, uniques = pd.factorize(['b', 'b', 'a', 'c', 'b'])
 574     >>> codes
 575     array([0, 0, 1, 2, 0]...)
 576     >>> uniques
 577     array(['b', 'a', 'c'], dtype=object)
 578 
 579     With ``sort=True``, the `uniques` will be sorted, and `codes` will be
 580     shuffled so that the relationship is the maintained.
 581 
 582     >>> codes, uniques = pd.factorize(['b', 'b', 'a', 'c', 'b'], sort=True)
 583     >>> codes
 584     array([1, 1, 0, 2, 1]...)
 585     >>> uniques
 586     array(['a', 'b', 'c'], dtype=object)
 587 
 588     Missing values are indicated in `codes` with `na_sentinel`
 589     (``-1`` by default). Note that missing values are never
 590     included in `uniques`.
 591 
 592     >>> codes, uniques = pd.factorize(['b', None, 'a', 'c', 'b'])
 593     >>> codes
 594     array([ 0, -1,  1,  2,  0]...)
 595     >>> uniques
 596     array(['b', 'a', 'c'], dtype=object)
 597 
 598     Thus far, we've only factorized lists (which are internally coerced to
 599     NumPy arrays). When factorizing pandas objects, the type of `uniques`
 600     will differ. For Categoricals, a `Categorical` is returned.
 601 
 602     >>> cat = pd.Categorical(['a', 'a', 'c'], categories=['a', 'b', 'c'])
 603     >>> codes, uniques = pd.factorize(cat)
 604     >>> codes
 605     array([0, 0, 1]...)
 606     >>> uniques
 607     [a, c]
 608     Categories (3, object): [a, b, c]
 609 
 610     Notice that ``'b'`` is in ``uniques.categories``, despite not being
 611     present in ``cat.values``.
 612 
 613     For all other pandas objects, an Index of the appropriate type is
 614     returned.
 615 
 616     >>> cat = pd.Series(['a', 'a', 'c'])
 617     >>> codes, uniques = pd.factorize(cat)
 618     >>> codes
 619     array([0, 0, 1]...)
 620     >>> uniques
 621     Index(['a', 'c'], dtype='object')
 622     """
 623     # Implementation notes: This method is responsible for 3 things
 624     # 1.) coercing data to array-like (ndarray, Index, extension array)
 625     # 2.) factorizing codes and uniques
 626     # 3.) Maybe boxing the uniques in an Index
 627     #
 628     # Step 2 is dispatched to extension types (like Categorical). They are
 629     # responsible only for factorization. All data coercion, sorting and boxing
 630     # should happen here.
 631 
 632     values = _ensure_arraylike(values)
 633     original = values
 634 
 635     if is_extension_array_dtype(values.dtype):
 636         values = extract_array(values)
 637         codes, uniques = values.factorize(na_sentinel=na_sentinel)
 638         dtype = original.dtype
 639     else:
 640         values, dtype = _ensure_data(values)
 641 
 642         if original.dtype.kind in ["m", "M"]:
 643             na_value = na_value_for_dtype(original.dtype)
 644         else:
 645             na_value = None
 646 
 647         codes, uniques = _factorize_array(
 648             values, na_sentinel=na_sentinel, size_hint=size_hint, na_value=na_value
 649         )
 650 
 651     if sort and len(uniques) > 0:
 652         uniques, codes = safe_sort(
 653             uniques, codes, na_sentinel=na_sentinel, assume_unique=True, verify=False
 654         )
 655 
 656     code_is_na = codes == na_sentinel
 657     if not dropna and code_is_na.any():
 658         # na_value is set based on the dtype of uniques, and compat set to False is
 659         # because we do not want na_value to be 0 for integers
 660         na_value = na_value_for_dtype(uniques.dtype, compat=False)
 661         uniques = np.append(uniques, [na_value])
 662         codes = np.where(code_is_na, len(uniques) - 1, codes)
 663 
 664     uniques = _reconstruct_data(uniques, dtype, original)
 665 
 666     # return original tenor
 667     if isinstance(original, ABCIndexClass):
 668         uniques = original._shallow_copy(uniques, name=None)
 669     elif isinstance(original, ABCSeries):
 670         from pandas import Index
 671 
 672         uniques = Index(uniques)
 673 
 674     return codes, uniques

 929 def checked_add_with_arr(arr, b, arr_mask=None, b_mask=None):
 930     """
 931     Perform array addition that checks for underflow and overflow.
 932 
 933     Performs the addition of an int64 array and an int64 integer (or array)
 934     but checks that they do not result in overflow first. For elements that
 935     are indicated to be NaN, whether or not there is overflow for that element
 936     is automatically ignored.
 937 
 938     Parameters
 939     ----------
 940     arr : array addend.
 941     b : array or scalar addend.
 942     arr_mask : boolean array or None
 943         array indicating which elements to exclude from checking
 944     b_mask : boolean array or boolean or None
 945         array or scalar indicating which element(s) to exclude from checking
 946 
 947     Returns
 948     -------
 949     sum : An array for elements x + b for each element x in arr if b is
 950           a scalar or an array for elements x + y for each element pair
 951           (x, y) in (arr, b).
 952 
 953     Raises
 954     ------
 955     OverflowError if any x + y exceeds the maximum or minimum int64 value.
 956     """
 957     # For performance reasons, we broadcast 'b' to the new array 'b2'
 958     # so that it has the same size as 'arr'.
 959     b2 = np.broadcast_to(b, arr.shape)
 960     if b_mask is not None:
 961         # We do the same broadcasting for b_mask as well.
 962         b2_mask = np.broadcast_to(b_mask, arr.shape)
 963     else:
 964         b2_mask = None
 965 
 966     # For elements that are NaN, regardless of their value, we should
 967     # ignore whether they overflow or not when doing the checked add.
 968     if arr_mask is not None and b2_mask is not None:
 969         not_nan = np.logical_not(arr_mask | b2_mask)
 970     elif arr_mask is not None:
 971         not_nan = np.logical_not(arr_mask)
 972     elif b_mask is not None:
 973         not_nan = np.logical_not(b2_mask)
 974     else:
 975         not_nan = np.empty(arr.shape, dtype=bool)
 976         not_nan.fill(True)
 977 
 978     # gh-14324: For each element in 'arr' and its corresponding element
 979     # in 'b2', we check the sign of the element in 'b2'. If it is positive,
 980     # we then check whether its sum with the element in 'arr' exceeds
 981     # np.iinfo(np.int64).max. If so, we have an overflow error. If it
 982     # it is negative, we then check whether its sum with the element in
 983     # 'arr' exceeds np.iinfo(np.int64).min. If so, we have an overflow
 984     # error as well.
 985     mask1 = b2 > 0
 986     mask2 = b2 < 0
 987 
 988     if not mask1.any():
 989         to_raise = ((np.iinfo(np.int64).min - b2 > arr) & not_nan).any()
 990     elif not mask2.any():
 991         to_raise = ((np.iinfo(np.int64).max - b2 < arr) & not_nan).any()
 992     else:
 993         to_raise = (
 994             ((np.iinfo(np.int64).max - b2[mask1] < arr[mask1]) & not_nan[mask1]).any()
 995             or (
 996                 (np.iinfo(np.int64).min - b2[mask2] > arr[mask2]) & not_nan[mask2]
 997             ).any()
 998         )
 999 
1000     if to_raise:
1001         raise OverflowError("Overflow in int64 addition")
1002     return arr + b

1005 def quantile(x, q, interpolation_method="fraction"):
1006     """
1007     Compute sample quantile or quantiles of the input array. For example, q=0.5
1008     computes the median.
1009 
1010     The `interpolation_method` parameter supports three values, namely
1011     `fraction` (default), `lower` and `higher`. Interpolation is done only,
1012     if the desired quantile lies between two data points `i` and `j`. For
1013     `fraction`, the result is an interpolated value between `i` and `j`;
1014     for `lower`, the result is `i`, for `higher` the result is `j`.
1015 
1016     Parameters
1017     ----------
1018     x : ndarray
1019         Values from which to extract score.
1020     q : scalar or array
1021         Percentile at which to extract score.
1022     interpolation_method : {'fraction', 'lower', 'higher'}, optional
1023         This optional parameter specifies the interpolation method to use,
1024         when the desired quantile lies between two data points `i` and `j`:
1025 
1026         - fraction: `i + (j - i)*fraction`, where `fraction` is the
1027                     fractional part of the index surrounded by `i` and `j`.
1028         -lower: `i`.
1029         - higher: `j`.
1030 
1031     Returns
1032     -------
1033     score : float
1034         Score at percentile.
1035 
1036     Examples
1037     --------
1038     >>> from scipy import stats
1039     >>> a = np.arange(100)
1040     >>> stats.scoreatpercentile(a, 50)
1041     49.5
1042 
1043     """
1044     x = np.asarray(x)
1045     mask = isna(x)
1046 
1047     x = x[~mask]
1048 
1049     values = np.sort(x)
1050 
1051     def _interpolate(a, b, fraction):
1052         """
1053         Returns the point at the given fraction between a and b, where
1054         'fraction' must be between 0 and 1.
1055         """
1056         return a + (b - a) * fraction
1057 
1058     def _get_score(at):
1059         if len(values) == 0:
1060             return np.nan
1061 
1062         idx = at * (len(values) - 1)
1063         if idx % 1 == 0:
1064             score = values[int(idx)]
1065         else:
1066             if interpolation_method == "fraction":
1067                 score = _interpolate(values[int(idx)], values[int(idx) + 1], idx % 1)
1068             elif interpolation_method == "lower":
1069                 score = values[np.floor(idx)]
1070             elif interpolation_method == "higher":
1071                 score = values[np.ceil(idx)]
1072             else:
1073                 raise ValueError(
1074                     "interpolation_method can only be 'fraction' "
1075                     ", 'lower' or 'higher'"
1076                 )
1077 
1078         return score
1079 
1080     if is_scalar(q):
1081         return _get_score(q)
1082     else:
1083         q = np.asarray(q, np.float64)
1084         result = [_get_score(x) for x in q]
1085         result = np.array(result, dtype=np.float64)
1086         return result

1493 def take(arr, indices, axis: int = 0, allow_fill: bool = False, fill_value=None):
1494     """
1495     Take elements from an array.
1496 
1497     .. versionadded:: 0.23.0
1498 
1499     Parameters
1500     ----------
1501     arr : sequence
1502         Non array-likes (sequences without a dtype) are coerced
1503         to an ndarray.
1504     indices : sequence of integers
1505         Indices to be taken.
1506     axis : int, default 0
1507         The axis over which to select values.
1508     allow_fill : bool, default False
1509         How to handle negative values in `indices`.
1510 
1511         * False: negative values in `indices` indicate positional indices
1512           from the right (the default). This is similar to :func:`numpy.take`.
1513 
1514         * True: negative values in `indices` indicate
1515           missing values. These values are set to `fill_value`. Any other
1516           other negative values raise a ``ValueError``.
1517 
1518     fill_value : any, optional
1519         Fill value to use for NA-indices when `allow_fill` is True.
1520         This may be ``None``, in which case the default NA value for
1521         the type (``self.dtype.na_value``) is used.
1522 
1523         For multi-dimensional `arr`, each *element* is filled with
1524         `fill_value`.
1525 
1526     Returns
1527     -------
1528     ndarray or ExtensionArray
1529         Same type as the input.
1530 
1531     Raises
1532     ------
1533     IndexError
1534         When `indices` is out of bounds for the array.
1535     ValueError
1536         When the indexer contains negative values other than ``-1``
1537         and `allow_fill` is True.
1538 
1539     Notes
1540     -----
1541     When `allow_fill` is False, `indices` may be whatever dimensionality
1542     is accepted by NumPy for `arr`.
1543 
1544     When `allow_fill` is True, `indices` should be 1-D.
1545 
1546     See Also
1547     --------
1548     numpy.take : Take elements from an array along an axis.
1549 
1550     Examples
1551     --------
1552     >>> from pandas.api.extensions import take
1553 
1554     With the default ``allow_fill=False``, negative numbers indicate
1555     positional indices from the right.
1556 
1557     >>> take(np.array([10, 20, 30]), [0, 0, -1])
1558     array([10, 10, 30])
1559 
1560     Setting ``allow_fill=True`` will place `fill_value` in those positions.
1561 
1562     >>> take(np.array([10, 20, 30]), [0, 0, -1], allow_fill=True)
1563     array([10., 10., nan])
1564 
1565     >>> take(np.array([10, 20, 30]), [0, 0, -1], allow_fill=True,
1566     ...      fill_value=-10)
1567     array([ 10,  10, -10])
1568     """
1569     if not is_array_like(arr):
1570         arr = np.asarray(arr)
1571 
1572     indices = np.asarray(indices, dtype=np.intp)
1573 
1574     if allow_fill:
1575         # Pandas style, -1 means NA
1576         validate_indices(indices, arr.shape[axis])
1577         result = take_1d(
1578             arr, indices, axis=axis, allow_fill=True, fill_value=fill_value
1579         )
1580     else:
1581         # NumPy style
1582         result = arr.take(indices, axis=axis)
1583     return result

1587     arr, indexer, axis: int = 0, out=None, fill_value=np.nan, allow_fill: bool = True
1588 ):
1589     """
1590     Specialized Cython take which sets NaN values in one pass
1591 
1592     This dispatches to ``take`` defined on ExtensionArrays. It does not
1593     currently dispatch to ``SparseArray.take`` for sparse ``arr``.
1594 
1595     Parameters
1596     ----------
1597     arr : array-like
1598         Input array.
1599     indexer : ndarray
1600         1-D array of indices to take, subarrays corresponding to -1 value
1601         indices are filed with fill_value
1602     axis : int, default 0
1603         Axis to take from
1604     out : ndarray or None, default None
1605         Optional output array, must be appropriate type to hold input and
1606         fill_value together, if indexer has any -1 value entries; call
1607         maybe_promote to determine this type for any fill_value
1608     fill_value : any, default np.nan
1609         Fill value to replace -1 values with
1610     allow_fill : boolean, default True
1611         If False, indexer is assumed to contain no -1 values so no filling
1612         will be done.  This short-circuits computation of a mask.  Result is
1613         undefined if allow_fill == False and -1 is present in indexer.
1614 
1615     Returns
1616     -------
1617     subarray : array-like
1618         May be the same type as the input, or cast to an ndarray.
1619     """
1620     mask_info = None
1621 
1622     if is_extension_array_dtype(arr):
1623         return arr.take(indexer, fill_value=fill_value, allow_fill=allow_fill)
1624 
1625     arr = extract_array(arr)
1626     arr = np.asarray(arr)
1627 
1628     if indexer is None:
1629         indexer = np.arange(arr.shape[axis], dtype=np.int64)
1630         dtype, fill_value = arr.dtype, arr.dtype.type()
1631     else:
1632         indexer = ensure_int64(indexer, copy=False)
1633         if not allow_fill:
1634             dtype, fill_value = arr.dtype, arr.dtype.type()
1635             mask_info = None, False
1636         else:
1637             # check for promotion based on types only (do this first because
1638             # it's faster than computing a mask)
1639             dtype, fill_value = maybe_promote(arr.dtype, fill_value)
1640             if dtype != arr.dtype and (out is None or out.dtype != dtype):
1641                 # check if promotion is actually required based on indexer
1642                 mask = indexer == -1
1643                 needs_masking = mask.any()
1644                 mask_info = mask, needs_masking
1645                 if needs_masking:
1646                     if out is not None and out.dtype != dtype:
1647                         raise TypeError("Incompatible type for fill_value")
1648                 else:
1649                     # if not, then depromote, set fill_value to dummy
1650                     # (it won't be used but we don't want the cython code
1651                     # to crash when trying to cast it to dtype)
1652                     dtype, fill_value = arr.dtype, arr.dtype.type()
1653 
1654     flip_order = False
1655     if arr.ndim == 2:
1656         if arr.flags.f_contiguous:
1657             flip_order = True
1658 
1659     if flip_order:
1660         arr = arr.T
1661         axis = arr.ndim - axis - 1
1662         if out is not None:
1663             out = out.T
1664 
1665     # at this point, it's guaranteed that dtype can hold both the arr values
1666     # and the fill_value
1667     if out is None:
1668         out_shape_ = list(arr.shape)
1669         out_shape_[axis] = len(indexer)
1670         out_shape = tuple(out_shape_)
1671         if arr.flags.f_contiguous and axis == arr.ndim - 1:
1672             # minor tweak that can make an order-of-magnitude difference
1673             # for dataframes initialized directly from 2-d ndarrays
1674             # (s.t. df.values is c-contiguous and df._mgr.blocks[0] is its
1675             # f-contiguous transpose)
1676             out = np.empty(out_shape, dtype=dtype, order="F")
1677         else:
1678             out = np.empty(out_shape, dtype=dtype)
1679 
1680     func = _get_take_nd_function(
1681         arr.ndim, arr.dtype, out.dtype, axis=axis, mask_info=mask_info
1682     )
1683     func(arr, indexer, out, fill_value)
1684 
1685     if flip_order:
1686         out = out.T
1687     return out

1753 def searchsorted(arr, value, side="left", sorter=None):
1754     """
1755     Find indices where elements should be inserted to maintain order.
1756 
1757     .. versionadded:: 0.25.0
1758 
1759     Find the indices into a sorted array `arr` (a) such that, if the
1760     corresponding elements in `value` were inserted before the indices,
1761     the order of `arr` would be preserved.
1762 
1763     Assuming that `arr` is sorted:
1764 
1765     ======  ================================
1766     `side`  returned index `i` satisfies
1767     ======  ================================
1768     left    ``arr[i-1] < value <= self[i]``
1769     right   ``arr[i-1] <= value < self[i]``
1770     ======  ================================
1771 
1772     Parameters
1773     ----------
1774     arr: array-like
1775         Input array. If `sorter` is None, then it must be sorted in
1776         ascending order, otherwise `sorter` must be an array of indices
1777         that sort it.
1778     value : array_like
1779         Values to insert into `arr`.
1780     side : {'left', 'right'}, optional
1781         If 'left', the index of the first suitable location found is given.
1782         If 'right', return the last such index.  If there is no suitable
1783         index, return either 0 or N (where N is the length of `self`).
1784     sorter : 1-D array_like, optional
1785         Optional array of integer indices that sort array a into ascending
1786         order. They are typically the result of argsort.
1787 
1788     Returns
1789     -------
1790     array of ints
1791         Array of insertion points with the same shape as `value`.
1792 
1793     See Also
1794     --------
1795     numpy.searchsorted : Similar method from NumPy.
1796     """
1797     if sorter is not None:
1798         sorter = ensure_platform_int(sorter)
1799 
1800     if (
1801         isinstance(arr, np.ndarray)
1802         and is_integer_dtype(arr)
1803         and (is_integer(value) or is_integer_dtype(value))
1804     ):
1805         # if `arr` and `value` have different dtypes, `arr` would be
1806         # recast by numpy, causing a slow search.
1807         # Before searching below, we therefore try to give `value` the
1808         # same dtype as `arr`, while guarding against integer overflows.
1809         iinfo = np.iinfo(arr.dtype.type)
1810         value_arr = np.array([value]) if is_scalar(value) else np.array(value)
1811         if (value_arr >= iinfo.min).all() and (value_arr <= iinfo.max).all():
1812             # value within bounds, so no overflow, so can convert value dtype
1813             # to dtype of arr
1814             dtype = arr.dtype
1815         else:
1816             dtype = value_arr.dtype
1817 
1818         if is_scalar(value):
1819             value = dtype.type(value)
1820         else:
1821             value = array(value, dtype=dtype)
1822     elif not (
1823         is_object_dtype(arr) or is_numeric_dtype(arr) or is_categorical_dtype(arr)
1824     ):
1825         # E.g. if `arr` is an array with dtype='datetime64[ns]'
1826         # and `value` is a pd.Timestamp, we may need to convert value
1827         value_ser = array([value]) if is_scalar(value) else array(value)
1828         value = value_ser[0] if is_scalar(value) else value_ser
1829         if isinstance(value, Timestamp) and value.tzinfo is None:
1830             value = value.to_datetime64()
1831 
1832     result = arr.searchsorted(value, side=side, sorter=sorter)
1833     return result

1884                 "TypeError. Convert to a suitable dtype prior to calling 'diff'.",
1885                 FutureWarning,
1886                 stacklevel=stacklevel,
1887             )
1888             arr = np.asarray(arr)
1889             dtype = arr.dtype
1890 
1891     is_timedelta = False
1892     is_bool = False
1893     if needs_i8_conversion(arr.dtype):
1894         dtype = np.float64
1895         arr = arr.view("i8")
1896         na = iNaT
1897         is_timedelta = True
1898 
1899     elif is_bool_dtype(dtype):
1900         dtype = np.object_
1901         is_bool = True
1902 
1903     elif is_integer_dtype(dtype):
1904         dtype = np.float64
1905 
1906     dtype = np.dtype(dtype)
1907     out_arr = np.empty(arr.shape, dtype=dtype)
1908 
1909     na_indexer = [slice(None)] * arr.ndim
1910     na_indexer[axis] = slice(None, n) if n >= 0 else slice(n, None)
1911     out_arr[tuple(na_indexer)] = na
1912 
1913     if arr.ndim == 2 and arr.dtype.name in _diff_special:
1914         # TODO: can diff_2d dtype specialization troubles be fixed by defining
1915         #  out_arr inside diff_2d?
1916         algos.diff_2d(arr, out_arr, n, axis)
1917     else:
1918         # To keep mypy happy, _res_indexer is a list while res_indexer is
1919         #  a tuple, ditto for lag_indexer.
1920         _res_indexer = [slice(None)] * arr.ndim
1921         _res_indexer[axis] = slice(n, None) if n >= 0 else slice(None, n)
1922         res_indexer = tuple(_res_indexer)
1923 
1924         _lag_indexer = [slice(None)] * arr.ndim
1925         _lag_indexer[axis] = slice(None, -n) if n > 0 else slice(-n, None)
1926         lag_indexer = tuple(_lag_indexer)
1927 
1928         # need to make sure that we account for na for datelike/timedelta
1929         # we don't actually want to subtract these i8 numbers
1930         if is_timedelta:
1931             res = arr[res_indexer]
1932             lag = arr[lag_indexer]
1933 
1934             mask = (arr[res_indexer] == na) | (arr[lag_indexer] == na)
1935             if mask.any():
1936                 res = res.copy()
1937                 res[mask] = 0
1938                 lag = lag.copy()
1939                 lag[mask] = 0
1940 
1941             result = res - lag
1942             result[mask] = na
1943             out_arr[res_indexer] = result
1944         elif is_bool:
1945             out_arr[res_indexer] = arr[res_indexer] ^ arr[lag_indexer]
1946         else:
1947             out_arr[res_indexer] = arr[res_indexer] - arr[lag_indexer]
1948 
1949     if is_timedelta:
1950         out_arr = out_arr.astype("int64").view("timedelta64[ns]")
1951 
1952     return out_arr

1958 # Note: safe_sort is in algorithms.py instead of sorting.py because it is
1959 #  low-dependency, is used in this module, and used private methods from
1960 #  this module.
1961 def safe_sort(
1962     values,
1963     codes=None,
1964     na_sentinel: int = -1,
1965     assume_unique: bool = False,
1966     verify: bool = True,
1967 ) -> Union[np.ndarray, Tuple[np.ndarray, np.ndarray]]:
1968     """
1969     Sort ``values`` and reorder corresponding ``codes``.
1970 
1971     ``values`` should be unique if ``codes`` is not None.
1972     Safe for use with mixed types (int, str), orders ints before strs.
1973 
1974     Parameters
1975     ----------
1976     values : list-like
1977         Sequence; must be unique if ``codes`` is not None.
1978     codes : list_like, optional
1979         Indices to ``values``. All out of bound indices are treated as
1980         "not found" and will be masked with ``na_sentinel``.
1981     na_sentinel : int, default -1
1982         Value in ``codes`` to mark "not found".
1983         Ignored when ``codes`` is None.
1984     assume_unique : bool, default False
1985         When True, ``values`` are assumed to be unique, which can speed up
1986         the calculation. Ignored when ``codes`` is None.
1987     verify : bool, default True
1988         Check if codes are out of bound for the values and put out of bound
1989         codes equal to na_sentinel. If ``verify=False``, it is assumed there
1990         are no out of bound codes. Ignored when ``codes`` is None.
1991 
1992         .. versionadded:: 0.25.0
1993 
1994     Returns
1995     -------
1996     ordered : ndarray
1997         Sorted ``values``
1998     new_codes : ndarray
1999         Reordered ``codes``; returned when ``codes`` is not None.
2000 
2001     Raises
2002     ------
2003     TypeError
2004         * If ``values`` is not list-like or if ``codes`` is neither None
2005         nor list-like
2006         * If ``values`` cannot be sorted
2007     ValueError
2008         * If ``codes`` is not None and ``values`` contain duplicates.
2009     """
2010     if not is_list_like(values):
2011         raise TypeError(
2012             "Only list-like objects are allowed to be passed to safe_sort as values"
2013         )
2014 
2015     if not isinstance(values, np.ndarray) and not is_extension_array_dtype(values):
2016         # don't convert to string types
2017         dtype, _ = infer_dtype_from_array(values)
2018         values = np.asarray(values, dtype=dtype)
2019 
2020     def sort_mixed(values):
2021         # order ints before strings, safe in py3
2022         str_pos = np.array([isinstance(x, str) for x in values], dtype=bool)
2023         nums = np.sort(values[~str_pos])
2024         strs = np.sort(values[str_pos])
2025         return np.concatenate([nums, np.asarray(strs, dtype=object)])
2026 
2027     sorter = None
2028     if (
2029         not is_extension_array_dtype(values)
2030         and lib.infer_dtype(values, skipna=False) == "mixed-integer"
2031     ):
2032         # unorderable in py3 if mixed str/int
2033         ordered = sort_mixed(values)
2034     else:
2035         try:
2036             sorter = values.argsort()
2037             ordered = values.take(sorter)
2038         except TypeError:
2039             # try this anyway
2040             ordered = sort_mixed(values)
2041 
2042     # codes:
2043 
2044     if codes is None:
2045         return ordered
2046 
2047     if not is_list_like(codes):
2048         raise TypeError(
2049             "Only list-like objects or None are allowed to "
2050             "be passed to safe_sort as codes"
2051         )
2052     codes = ensure_platform_int(np.asarray(codes))
2053 
2054     if not assume_unique and not len(unique(values)) == len(values):
2055         raise ValueError("values should be unique if codes is not None")
2056 
2057     if sorter is None:
2058         # mixed types
2059         hash_klass, values = _get_data_algo(values)
2060         t = hash_klass(len(values))
2061         t.map_locations(values)
2062         sorter = ensure_platform_int(t.lookup(ordered))
2063 
2064     if na_sentinel == -1:
2065         # take_1d is faster, but only works for na_sentinels of -1
2066         order2 = sorter.argsort()
2067         new_codes = take_1d(order2, codes, fill_value=-1)
2068         if verify:
2069             mask = (codes < -len(values)) | (codes >= len(values))
2070         else:
2071             mask = None
2072     else:
2073         reverse_indexer = np.empty(len(sorter), dtype=np.int_)
2074         reverse_indexer.put(sorter, np.arange(len(sorter)))
2075         # Out of bound indices will be masked with `na_sentinel` next, so we
2076         # may deal with them here without performance loss using `mode='wrap'`
2077         new_codes = reverse_indexer.take(codes, mode="wrap")
2078 
2079         mask = codes == na_sentinel
2080         if verify:
2081             mask = mask | (codes < -len(values)) | (codes >= len(values))
2082 
2083     if mask is not None:
2084         np.putmask(new_codes, mask, na_sentinel)
2085 
2086     return ordered, ensure_platform_int(new_codes)
