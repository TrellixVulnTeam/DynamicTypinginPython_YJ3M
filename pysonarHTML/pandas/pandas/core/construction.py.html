  56 def array(
  57     data: Union[Sequence[object], AnyArrayLike],
  58     dtype: Optional[Dtype] = None,
  59     copy: bool = True,
  60 ) -> "ExtensionArray":
  61     """
  62     Create an array.
  63 
  64     .. versionadded:: 0.24.0
  65 
  66     Parameters
  67     ----------
  68     data : Sequence of objects
  69         The scalars inside `data` should be instances of the
  70         scalar type for `dtype`. It's expected that `data`
  71         represents a 1-dimensional array of data.
  72 
  73         When `data` is an Index or Series, the underlying array
  74         will be extracted from `data`.
  75 
  76     dtype : str, np.dtype, or ExtensionDtype, optional
  77         The dtype to use for the array. This may be a NumPy
  78         dtype or an extension type registered with pandas using
  79         :meth:`pandas.api.extensions.register_extension_dtype`.
  80 
  81         If not specified, there are two possibilities:
  82 
  83         1. When `data` is a :class:`Series`, :class:`Index`, or
  84            :class:`ExtensionArray`, the `dtype` will be taken
  85            from the data.
  86         2. Otherwise, pandas will attempt to infer the `dtype`
  87            from the data.
  88 
  89         Note that when `data` is a NumPy array, ``data.dtype`` is
  90         *not* used for inferring the array type. This is because
  91         NumPy cannot represent all the types of data that can be
  92         held in extension arrays.
  93 
  94         Currently, pandas will infer an extension dtype for sequences of
  95 
  96         ============================== =====================================
  97         Scalar Type                    Array Type
  98         ============================== =====================================
  99         :class:`pandas.Interval`       :class:`pandas.arrays.IntervalArray`
 100         :class:`pandas.Period`         :class:`pandas.arrays.PeriodArray`
 101         :class:`datetime.datetime`     :class:`pandas.arrays.DatetimeArray`
 102         :class:`datetime.timedelta`    :class:`pandas.arrays.TimedeltaArray`
 103         :class:`int`                   :class:`pandas.arrays.IntegerArray`
 104         :class:`str`                   :class:`pandas.arrays.StringArray`
 105         :class:`bool`                  :class:`pandas.arrays.BooleanArray`
 106         ============================== =====================================
 107 
 108         For all other cases, NumPy's usual inference rules will be used.
 109 
 110         .. versionchanged:: 1.0.0
 111 
 112            Pandas infers nullable-integer dtype for integer data,
 113            string dtype for string data, and nullable-boolean dtype
 114            for boolean data.
 115 
 116     copy : bool, default True
 117         Whether to copy the data, even if not necessary. Depending
 118         on the type of `data`, creating the new array may require
 119         copying data, even if ``copy=False``.
 120 
 121     Returns
 122     -------
 123     ExtensionArray
 124         The newly created array.
 125 
 126     Raises
 127     ------
 128     ValueError
 129         When `data` is not 1-dimensional.
 130 
 131     See Also
 132     --------
 133     numpy.array : Construct a NumPy array.
 134     Series : Construct a pandas Series.
 135     Index : Construct a pandas Index.
 136     arrays.PandasArray : ExtensionArray wrapping a NumPy array.
 137     Series.array : Extract the array stored within a Series.
 138 
 139     Notes
 140     -----
 141     Omitting the `dtype` argument means pandas will attempt to infer the
 142     best array type from the values in the data. As new array types are
 143     added by pandas and 3rd party libraries, the "best" array type may
 144     change. We recommend specifying `dtype` to ensure that
 145 
 146     1. the correct array type for the data is returned
 147     2. the returned array type doesn't change as new extension types
 148        are added by pandas and third-party libraries
 149 
 150     Additionally, if the underlying memory representation of the returned
 151     array matters, we recommend specifying the `dtype` as a concrete object
 152     rather than a string alias or allowing it to be inferred. For example,
 153     a future version of pandas or a 3rd-party library may include a
 154     dedicated ExtensionArray for string data. In this event, the following
 155     would no longer return a :class:`arrays.PandasArray` backed by a NumPy
 156     array.
 157 
 158     >>> pd.array(['a', 'b'], dtype=str)
 159     <PandasArray>
 160     ['a', 'b']
 161     Length: 2, dtype: str32
 162 
 163     This would instead return the new ExtensionArray dedicated for string
 164     data. If you really need the new array to be backed by a  NumPy array,
 165     specify that in the dtype.
 166 
 167     >>> pd.array(['a', 'b'], dtype=np.dtype("<U1"))
 168     <PandasArray>
 169     ['a', 'b']
 170     Length: 2, dtype: str32
 171 
 172     Finally, Pandas has arrays that mostly overlap with NumPy
 173 
 174       * :class:`arrays.DatetimeArray`
 175       * :class:`arrays.TimedeltaArray`
 176 
 177     When data with a ``datetime64[ns]`` or ``timedelta64[ns]`` dtype is
 178     passed, pandas will always return a ``DatetimeArray`` or ``TimedeltaArray``
 179     rather than a ``PandasArray``. This is for symmetry with the case of
 180     timezone-aware data, which NumPy does not natively support.
 181 
 182     >>> pd.array(['2015', '2016'], dtype='datetime64[ns]')
 183     <DatetimeArray>
 184     ['2015-01-01 00:00:00', '2016-01-01 00:00:00']
 185     Length: 2, dtype: datetime64[ns]
 186 
 187     >>> pd.array(["1H", "2H"], dtype='timedelta64[ns]')
 188     <TimedeltaArray>
 189     ['0 days 01:00:00', '0 days 02:00:00']
 190     Length: 2, dtype: timedelta64[ns]
 191 
 192     Examples
 193     --------
 194     If a dtype is not specified, pandas will infer the best dtype from the values.
 195     See the description of `dtype` for the types pandas infers for.
 196 
 197     >>> pd.array([1, 2])
 198     <IntegerArray>
 199     [1, 2]
 200     Length: 2, dtype: Int64
 201 
 202     >>> pd.array([1, 2, np.nan])
 203     <IntegerArray>
 204     [1, 2, <NA>]
 205     Length: 3, dtype: Int64
 206 
 207     >>> pd.array(["a", None, "c"])
 208     <StringArray>
 209     ['a', <NA>, 'c']
 210     Length: 3, dtype: string
 211 
 212     >>> pd.array([pd.Period('2000', freq="D"), pd.Period("2000", freq="D")])
 213     <PeriodArray>
 214     ['2000-01-01', '2000-01-01']
 215     Length: 2, dtype: period[D]
 216 
 217     You can use the string alias for `dtype`
 218 
 219     >>> pd.array(['a', 'b', 'a'], dtype='category')
 220     [a, b, a]
 221     Categories (2, object): [a, b]
 222 
 223     Or specify the actual dtype
 224 
 225     >>> pd.array(['a', 'b', 'a'],
 226     ...          dtype=pd.CategoricalDtype(['a', 'b', 'c'], ordered=True))
 227     [a, b, a]
 228     Categories (3, object): [a < b < c]
 229 
 230     If pandas does not infer a dedicated extension type a
 231     :class:`arrays.PandasArray` is returned.
 232 
 233     >>> pd.array([1.1, 2.2])
 234     <PandasArray>
 235     [1.1, 2.2]
 236     Length: 2, dtype: float64
 237 
 238     As mentioned in the "Notes" section, new extension types may be added
 239     in the future (by pandas or 3rd party libraries), causing the return
 240     value to no longer be a :class:`arrays.PandasArray`. Specify the `dtype`
 241     as a NumPy dtype if you need to ensure there's no future change in
 242     behavior.
 243 
 244     >>> pd.array([1, 2], dtype=np.dtype("int32"))
 245     <PandasArray>
 246     [1, 2]
 247     Length: 2, dtype: int32
 248 
 249     `data` must be 1-dimensional. A ValueError is raised when the input
 250     has the wrong dimensionality.
 251 
 252     >>> pd.array(1)
 253     Traceback (most recent call last):
 254       ...
 255     ValueError: Cannot pass scalar '1' to 'pandas.array'.
 256     """
 257     from pandas.core.arrays import (
 258         period_array,
 259         BooleanArray,
 260         IntegerArray,
 261         IntervalArray,
 262         PandasArray,
 263         DatetimeArray,
 264         TimedeltaArray,
 265         StringArray,
 266     )
 267 
 268     if lib.is_scalar(data):
 269         msg = f"Cannot pass scalar '{data}' to 'pandas.array'."
 270         raise ValueError(msg)
 271 
 272     if dtype is None and isinstance(
 273         data, (ABCSeries, ABCIndexClass, ABCExtensionArray)
 274     ):
 275         dtype = data.dtype
 276 
 277     data = extract_array(data, extract_numpy=True)
 278 
 279     # this returns None for not-found dtypes.
 280     if isinstance(dtype, str):
 281         dtype = registry.find(dtype) or dtype
 282 
 283     if is_extension_array_dtype(dtype):
 284         cls = cast(ExtensionDtype, dtype).construct_array_type()
 285         return cls._from_sequence(data, dtype=dtype, copy=copy)
 286 
 287     if dtype is None:
 288         inferred_dtype = lib.infer_dtype(data, skipna=True)
 289         if inferred_dtype == "period":
 290             try:
 291                 return period_array(data, copy=copy)
 292             except IncompatibleFrequency:
 293                 # We may have a mixture of frequencies.
 294                 # We choose to return an ndarray, rather than raising.
 295                 pass
 296         elif inferred_dtype == "interval":
 297             try:
 298                 return IntervalArray(data, copy=copy)
 299             except ValueError:
 300                 # We may have a mixture of `closed` here.
 301                 # We choose to return an ndarray, rather than raising.
 302                 pass
 303 
 304         elif inferred_dtype.startswith("datetime"):
 305             # datetime, datetime64
 306             try:
 307                 return DatetimeArray._from_sequence(data, copy=copy)
 308             except ValueError:
 309                 # Mixture of timezones, fall back to PandasArray
 310                 pass
 311 
 312         elif inferred_dtype.startswith("timedelta"):
 313             # timedelta, timedelta64
 314             return TimedeltaArray._from_sequence(data, copy=copy)
 315 
 316         elif inferred_dtype == "string":
 317             return StringArray._from_sequence(data, copy=copy)
 318 
 319         elif inferred_dtype == "integer":
 320             return IntegerArray._from_sequence(data, copy=copy)
 321 
 322         elif inferred_dtype == "boolean":
 323             return BooleanArray._from_sequence(data, copy=copy)
 324 
 325     # Pandas overrides NumPy for
 326     #   1. datetime64[ns]
 327     #   2. timedelta64[ns]
 328     # so that a DatetimeArray is returned.
 329     if is_datetime64_ns_dtype(dtype):
 330         return DatetimeArray._from_sequence(data, dtype=dtype, copy=copy)
 331     elif is_timedelta64_ns_dtype(dtype):
 332         return TimedeltaArray._from_sequence(data, dtype=dtype, copy=copy)
 333 
 334     result = PandasArray._from_sequence(data, dtype=dtype, copy=copy)
 335     return result

 338 def extract_array(obj, extract_numpy: bool = False):
 339     """
 340     Extract the ndarray or ExtensionArray from a Series or Index.
 341 
 342     For all other types, `obj` is just returned as is.
 343 
 344     Parameters
 345     ----------
 346     obj : object
 347         For Series / Index, the underlying ExtensionArray is unboxed.
 348         For Numpy-backed ExtensionArrays, the ndarray is extracted.
 349 
 350     extract_numpy : bool, default False
 351         Whether to extract the ndarray from a PandasArray
 352 
 353     Returns
 354     -------
 355     arr : object
 356 
 357     Examples
 358     --------
 359     >>> extract_array(pd.Series(['a', 'b', 'c'], dtype='category'))
 360     [a, b, c]
 361     Categories (3, object): [a, b, c]
 362 
 363     Other objects like lists, arrays, and DataFrames are just passed through.
 364 
 365     >>> extract_array([1, 2, 3])
 366     [1, 2, 3]
 367 
 368     For an ndarray-backed Series / Index a PandasArray is returned.
 369 
 370     >>> extract_array(pd.Series([1, 2, 3]))
 371     <PandasArray>
 372     [1, 2, 3]
 373     Length: 3, dtype: int64
 374 
 375     To extract all the way down to the ndarray, pass ``extract_numpy=True``.
 376 
 377     >>> extract_array(pd.Series([1, 2, 3]), extract_numpy=True)
 378     array([1, 2, 3])
 379     """
 380     if isinstance(obj, (ABCIndexClass, ABCSeries)):
 381         obj = obj.array
 382 
 383     if extract_numpy and isinstance(obj, ABCPandasArray):
 384         obj = obj.to_numpy()
 385 
 386     return obj

 389 def sanitize_array(
 390     data,
 391     index: Optional["Index"],
 392     dtype: Optional[DtypeObj] = None,
 393     copy: bool = False,
 394     raise_cast_failure: bool = False,
 395 ) -> ArrayLike:
 396     """
 397     Sanitize input data to an ndarray or ExtensionArray, copy if specified,
 398     coerce to the dtype if specified.
 399     """
 400 
 401     if isinstance(data, ma.MaskedArray):
 402         mask = ma.getmaskarray(data)
 403         if mask.any():
 404             data, fill_value = maybe_upcast(data, copy=True)
 405             data.soften_mask()  # set hardmask False if it was True
 406             data[mask] = fill_value
 407         else:
 408             data = data.copy()
 409 
 410     # extract ndarray or ExtensionArray, ensure we have no PandasArray
 411     data = extract_array(data, extract_numpy=True)
 412 
 413     # GH#846
 414     if isinstance(data, np.ndarray):
 415 
 416         if dtype is not None and is_float_dtype(data.dtype) and is_integer_dtype(dtype):
 417             # possibility of nan -> garbage
 418             try:
 419                 subarr = _try_cast(data, dtype, copy, True)
 420             except ValueError:
 421                 if copy:
 422                     subarr = data.copy()
 423                 else:
 424                     subarr = np.array(data, copy=False)
 425         else:
 426             # we will try to copy be-definition here
 427             subarr = _try_cast(data, dtype, copy, raise_cast_failure)
 428 
 429     elif isinstance(data, ABCExtensionArray):
 430         # it is already ensured above this is not a PandasArray
 431         subarr = data
 432 
 433         if dtype is not None:
 434             subarr = subarr.astype(dtype, copy=copy)
 435         elif copy:
 436             subarr = subarr.copy()
 437         return subarr
 438 
 439     elif isinstance(data, (list, tuple)) and len(data) > 0:
 440         if dtype is not None:
 441             subarr = _try_cast(data, dtype, copy, raise_cast_failure)
 442         else:
 443             subarr = maybe_convert_platform(data)
 444 
 445         subarr = maybe_cast_to_datetime(subarr, dtype)
 446 
 447     elif isinstance(data, range):
 448         # GH#16804
 449         arr = np.arange(data.start, data.stop, data.step, dtype="int64")
 450         subarr = _try_cast(arr, dtype, copy, raise_cast_failure)
 451     elif isinstance(data, abc.Set):
 452         raise TypeError("Set type is unordered")
 453     elif lib.is_scalar(data) and index is not None and dtype is not None:
 454         data = maybe_cast_to_datetime(data, dtype)
 455         if not lib.is_scalar(data):
 456             data = data[0]
 457         subarr = construct_1d_arraylike_from_scalar(data, len(index), dtype)
 458     else:
 459         subarr = _try_cast(data, dtype, copy, raise_cast_failure)
 460 
 461     # scalar like, GH
 462     if getattr(subarr, "ndim", 0) == 0:
 463         if isinstance(data, list):  # pragma: no cover
 464             subarr = np.array(data, dtype=object)
 465         elif index is not None:
 466             value = data
 467 
 468             # figure out the dtype from the value (upcast if necessary)
 469             if dtype is None:
 470                 dtype, value = infer_dtype_from_scalar(value)
 471             else:
 472                 # need to possibly convert the value here
 473                 value = maybe_cast_to_datetime(value, dtype)
 474 
 475             subarr = construct_1d_arraylike_from_scalar(value, len(index), dtype)
 476 
 477         else:
 478             return subarr.item()
 479 
 480     # the result that we want
 481     elif subarr.ndim == 1:
 482         if index is not None:
 483 
 484             # a 1-element ndarray
 485             if len(subarr) != len(index) and len(subarr) == 1:
 486                 subarr = construct_1d_arraylike_from_scalar(
 487                     subarr[0], len(index), subarr.dtype
 488                 )
 489 
 490     elif subarr.ndim > 1:
 491         if isinstance(data, np.ndarray):
 492             raise Exception("Data must be 1-dimensional")
 493         else:
 494             subarr = com.asarray_tuplesafe(data, dtype=dtype)
 495 
 496     if not (is_extension_array_dtype(subarr.dtype) or is_extension_array_dtype(dtype)):
 497         # This is to prevent mixed-type Series getting all casted to
 498         # NumPy string type, e.g. NaN --> '-1#IND'.
 499         if issubclass(subarr.dtype.type, str):
 500             # GH#16605
 501             # If not empty convert the data to dtype
 502             # GH#19853: If data is a scalar, subarr has already the result
 503             if not lib.is_scalar(data):
 504                 if not np.all(isna(data)):
 505                     data = np.array(data, dtype=dtype, copy=False)
 506                 subarr = np.array(data, dtype=object, copy=copy)
 507 
 508         if is_object_dtype(subarr.dtype) and not is_object_dtype(dtype):
 509             inferred = lib.infer_dtype(subarr, skipna=False)
 510             if inferred in {"interval", "period"}:
 511                 subarr = array(subarr)
 512 
 513     return subarr

 516 def _try_cast(
 517     arr, dtype: Optional[DtypeObj], copy: bool, raise_cast_failure: bool,
 518 ):
 519     """
 520     Convert input to numpy ndarray and optionally cast to a given dtype.
 521 
 522     Parameters
 523     ----------
 524     arr : ndarray, scalar, list, tuple, iterator (catchall)
 525         Excludes: ExtensionArray, Series, Index.
 526     dtype : np.dtype, ExtensionDtype or None
 527     copy : bool
 528         If False, don't copy the data if not needed.
 529     raise_cast_failure : bool
 530         If True, and if a dtype is specified, raise errors during casting.
 531         Otherwise an object array is returned.
 532     """
 533     # perf shortcut as this is the most common case
 534     if isinstance(arr, np.ndarray):
 535         if maybe_castable(arr) and not copy and dtype is None:
 536             return arr
 537 
 538     if isinstance(dtype, ExtensionDtype) and dtype.kind != "M":
 539         # create an extension array from its dtype
 540         # DatetimeTZ case needs to go through maybe_cast_to_datetime
 541         array_type = dtype.construct_array_type()._from_sequence
 542         subarr = array_type(arr, dtype=dtype, copy=copy)
 543         return subarr
 544 
 545     try:
 546         # GH#15832: Check if we are requesting a numeric dtype and
 547         # that we can convert the data to the requested dtype.
 548         if is_integer_dtype(dtype):
 549             # this will raise if we have e.g. floats
 550             maybe_cast_to_integer_array(arr, dtype)
 551             subarr = arr
 552         else:
 553             subarr = maybe_cast_to_datetime(arr, dtype)
 554 
 555         # Take care in creating object arrays (but iterators are not
 556         # supported):
 557         if is_object_dtype(dtype) and (
 558             is_list_like(subarr)
 559             and not (is_iterator(subarr) or isinstance(subarr, np.ndarray))
 560         ):
 561             subarr = construct_1d_object_array_from_listlike(subarr)
 562         elif not is_extension_array_dtype(subarr):
 563             subarr = construct_1d_ndarray_preserving_na(subarr, dtype, copy=copy)
 564     except OutOfBoundsDatetime:
 565         # in case of out of bound datetime64 -> always raise
 566         raise
 567     except (ValueError, TypeError):
 568         if dtype is not None and raise_cast_failure:
 569             raise
 570         else:
 571             subarr = np.array(arr, dtype=object, copy=copy)
 572     return subarr

 575 def is_empty_data(data: Any) -> bool:
 576     """
 577     Utility to check if a Series is instantiated with empty data,
 578     which does not contain dtype information.
 579 
 580     Parameters
 581     ----------
 582     data : array-like, Iterable, dict, or scalar value
 583         Contains data stored in Series.
 584 
 585     Returns
 586     -------
 587     bool
 588     """
 589     is_none = data is None
 590     is_list_like_without_dtype = is_list_like(data) and not hasattr(data, "dtype")
 591     is_simple_empty = is_list_like_without_dtype and not data
 592     return is_none or is_simple_empty

 595 def create_series_with_explicit_dtype(
 596     data: Any = None,
 597     index: Optional[Union[ArrayLike, "Index"]] = None,
 598     dtype: Optional[Dtype] = None,
 599     name: Optional[str] = None,
 600     copy: bool = False,
 601     fastpath: bool = False,
 602     dtype_if_empty: Dtype = object,
 603 ) -> "Series":
 604     """
 605     Helper to pass an explicit dtype when instantiating an empty Series.
 606 
 607     This silences a DeprecationWarning described in GitHub-17261.
 608 
 609     Parameters
 610     ----------
 611     data : Mirrored from Series.__init__
 612     index : Mirrored from Series.__init__
 613     dtype : Mirrored from Series.__init__
 614     name : Mirrored from Series.__init__
 615     copy : Mirrored from Series.__init__
 616     fastpath : Mirrored from Series.__init__
 617     dtype_if_empty : str, numpy.dtype, or ExtensionDtype
 618         This dtype will be passed explicitly if an empty Series will
 619         be instantiated.
 620 
 621     Returns
 622     -------
 623     Series
 624     """
 625     from pandas.core.series import Series
 626 
 627     if is_empty_data(data) and dtype is None:
 628         dtype = dtype_if_empty
 629     return Series(
 630         data=data, index=index, dtype=dtype, name=name, copy=copy, fastpath=fastpath
 631     )
