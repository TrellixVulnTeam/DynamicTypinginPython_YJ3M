 121 def _coerce_method(converter):
 122     """
 123     Install the scalar coercion methods.
 124     """
 125 
 126     def wrapper(self):
 127         if len(self) == 1:
 128             return converter(self.iloc[0])
 129         raise TypeError(f"cannot convert the series to {converter}")

 131     wrapper.__name__ = f"__{converter.__name__}__"
 132     return wrapper

 139 class Series(base.IndexOpsMixin, generic.NDFrame):
 140     """
 141     One-dimensional ndarray with axis labels (including time series).
 142 
 143     Labels need not be unique but must be a hashable type. The object
 144     supports both integer- and label-based indexing and provides a host of
 145     methods for performing operations involving the index. Statistical
 146     methods from ndarray have been overridden to automatically exclude
 147     missing data (currently represented as NaN).
 148 
 149     Operations between Series (+, -, /, *, **) align values based on their
 150     associated index values-- they need not be the same length. The result
 151     index will be the sorted union of the two indexes.
 152 
 153     Parameters
 154     ----------
 155     data : array-like, Iterable, dict, or scalar value
 156         Contains data stored in Series.
 157 
 158         .. versionchanged:: 0.23.0
 159            If data is a dict, argument order is maintained for Python 3.6
 160            and later.
 161 
 162     index : array-like or Index (1d)
 163         Values must be hashable and have the same length as `data`.
 164         Non-unique index values are allowed. Will default to
 165         RangeIndex (0, 1, 2, ..., n) if not provided. If both a dict and index
 166         sequence are used, the index will override the keys found in the
 167         dict.
 168     dtype : str, numpy.dtype, or ExtensionDtype, optional
 169         Data type for the output Series. If not specified, this will be
 170         inferred from `data`.
 171         See the :ref:`user guide <basics.dtypes>` for more usages.
 172     name : str, optional
 173         The name to give to the Series.
 174     copy : bool, default False
 175         Copy input data.
 176     """
 177

 201     def __init__(
 202         self, data=None, index=None, dtype=None, name=None, copy=False, fastpath=False
 203     ):
 204 
 205         if (
 206             isinstance(data, SingleBlockManager)
 207             and index is None
 208             and dtype is None
 209             and copy is False
 210         ):
 211             # GH#33357 called with just the SingleBlockManager
 212             NDFrame.__init__(self, data)
 213             self.name = name
 214             return
 215 
 216         # we are called internally, so short-circuit
 217         if fastpath:
 218 
 219             # data is an ndarray, index is defined
 220             if not isinstance(data, SingleBlockManager):
 221                 data = SingleBlockManager.from_array(data, index)
 222             if copy:
 223                 data = data.copy()
 224             if index is None:
 225                 index = data.index
 226 
 227         else:
 228 
 229             name = ibase.maybe_extract_name(name, data, type(self))
 230 
 231             if is_empty_data(data) and dtype is None:
 232                 # gh-17261
 233                 warnings.warn(
 234                     "The default dtype for empty Series will be 'object' instead "
 235                     "of 'float64' in a future version. Specify a dtype explicitly "
 236                     "to silence this warning.",
 237                     DeprecationWarning,
 238                     stacklevel=2,
 239                 )
 240                 # uncomment the line below when removing the DeprecationWarning
 241                 # dtype = np.dtype(object)
 242 
 243             if index is not None:
 244                 index = ensure_index(index)
 245 
 246             if data is None:
 247                 data = {}
 248             if dtype is not None:
 249                 dtype = self._validate_dtype(dtype)
 250 
 251             if isinstance(data, MultiIndex):
 252                 raise NotImplementedError(
 253                     "initializing a Series from a MultiIndex is not supported"
 254                 )
 255             elif isinstance(data, Index):
 256 
 257                 if dtype is not None:
 258                     # astype copies
 259                     data = data.astype(dtype)
 260                 else:
 261                     # GH#24096 we need to ensure the index remains immutable
 262                     data = data._values.copy()
 263                 copy = False
 264 
 265             elif isinstance(data, np.ndarray):
 266                 if len(data.dtype):
 267                     # GH#13296 we are dealing with a compound dtype, which
 268                     #  should be treated as 2D
 269                     raise ValueError(
 270                         "Cannot construct a Series from an ndarray with "
 271                         "compound dtype.  Use DataFrame instead."
 272                     )
 273             elif isinstance(data, Series):
 274                 if index is None:
 275                     index = data.index
 276                 else:
 277                     data = data.reindex(index, copy=copy)
 278                     copy = False
 279                 data = data._mgr
 280             elif is_dict_like(data):
 281                 data, index = self._init_dict(data, index, dtype)
 282                 dtype = None
 283                 copy = False
 284             elif isinstance(data, SingleBlockManager):
 285                 if index is None:
 286                     index = data.index
 287                 elif not data.index.equals(index) or copy:
 288                     # GH#19275 SingleBlockManager input should only be called
 289                     # internally
 290                     raise AssertionError(
 291                         "Cannot pass both SingleBlockManager "
 292                         "`data` argument and a different "
 293                         "`index` argument. `copy` must be False."
 294                     )
 295 
 296             elif is_extension_array_dtype(data):
 297                 pass
 298             elif isinstance(data, (set, frozenset)):
 299                 raise TypeError(f"'{type(data).__name__}' type is unordered")
 300             else:
 301                 data = com.maybe_iterable_to_list(data)
 302 
 303             if index is None:
 304                 if not is_list_like(data):
 305                     data = [data]
 306                 index = ibase.default_index(len(data))
 307             elif is_list_like(data):
 308 
 309                 # a scalar numpy array is list-like but doesn't
 310                 # have a proper length
 311                 try:
 312                     if len(index) != len(data):
 313                         raise ValueError(
 314                             f"Length of passed values is {len(data)}, "
 315                             f"index implies {len(index)}."
 316                         )
 317                 except TypeError:
 318                     pass
 319 
 320             # create/copy the manager
 321             if isinstance(data, SingleBlockManager):
 322                 if dtype is not None:
 323                     data = data.astype(dtype=dtype, errors="ignore", copy=copy)
 324                 elif copy:
 325                     data = data.copy()
 326             else:
 327                 data = sanitize_array(data, index, dtype, copy, raise_cast_failure=True)
 328 
 329                 data = SingleBlockManager.from_array(data, index)
 330 
 331         generic.NDFrame.__init__(self, data)
 332         self.name = name
 333         self._set_axis(0, index, fastpath=True)

 335     def _init_dict(self, data, index=None, dtype=None):
 336         """
 337         Derive the "_mgr" and "index" attributes of a new Series from a
 338         dictionary input.
 339 
 340         Parameters
 341         ----------
 342         data : dict or dict-like
 343             Data used to populate the new Series.
 344         index : Index or index-like, default None
 345             Index for the new Series: if None, use dict keys.
 346         dtype : dtype, default None
 347             The dtype for the new Series: if None, infer from data.
 348 
 349         Returns
 350         -------
 351         _data : BlockManager for the new Series
 352         index : index for the new Series
 353         """
 354         # Looking for NaN in dict doesn't work ({np.nan : 1}[float('nan')]
 355         # raises KeyError), so we iterate the entire dict, and align
 356         if data:
 357             keys, values = zip(*data.items())
 358             values = list(values)
 359         elif index is not None:
 360             # fastpath for Series(data=None). Just use broadcasting a scalar
 361             # instead of reindexing.
 362             values = na_value_for_dtype(dtype)
 363             keys = index
 364         else:
 365             keys, values = [], []
 366 
 367         # Input is now list-like, so rely on "standard" construction:
 368 
 369         # TODO: passing np.float64 to not break anything yet. See GH-17261
 370         s = create_series_with_explicit_dtype(
 371             values, index=keys, dtype=dtype, dtype_if_empty=np.float64
 372         )
 373 
 374         # Now we just make sure the order is respected, if any
 375         if data and index is not None:
 376             s = s.reindex(index, copy=False)
 377         return s._mgr, s.index

 398     def _set_axis(self, axis: int, labels, fastpath: bool = False) -> None:
 399         """
 400         Override generic, we want to set the _typ here.
 401 
 402         This is called from the cython code when we set the `index` attribute
 403         directly, e.g. `series.index = [1, 2, 3]`.
 404         """
 405         if not fastpath:
 406             labels = ensure_index(labels)
 407 
 408         is_all_dates = labels.is_all_dates
 409         if is_all_dates:
 410             if not isinstance(labels, (DatetimeIndex, PeriodIndex, TimedeltaIndex)):
 411                 try:
 412                     labels = DatetimeIndex(labels)
 413                     # need to set here because we changed the index
 414                     if fastpath:
 415                         self._mgr.set_axis(axis, labels)
 416                 except (tslibs.OutOfBoundsDatetime, ValueError):
 417                     # labels may exceeds datetime bounds,
 418                     # or not be a DatetimeIndex
 419                     pass
 420 
 421         object.__setattr__(self, "_index", labels)
 422         if not fastpath:
 423             # The ensure_index call above ensures we have an Index object
 424             self._mgr.set_axis(axis, labels)

 540     @property

 543         Return the internal repr of this data (defined by Block.interval_values).

 601     def view(self, dtype=None) -> "Series":
 602         """
 603         Create a new view of the Series.
 604 
 605         This function will return a new Series with a view of the same
 606         underlying values in memory, optionally reinterpreted with a new data
 607         type. The new data type must preserve the same size in bytes as to not
 608         cause index misalignment.
 609 
 610         Parameters
 611         ----------
 612         dtype : data type
 613             Data type object or one of their string representations.
 614 
 615         Returns
 616         -------
 617         Series
 618             A new Series object as a view of the same data in memory.
 619 
 620         See Also
 621         --------
 622         numpy.ndarray.view : Equivalent numpy function to create a new view of
 623             the same data in memory.
 624 
 625         Notes
 626         -----
 627         Series are instantiated with ``dtype=float64`` by default. While
 628         ``numpy.ndarray.view()`` will return a view with the same data type as
 629         the original array, ``Series.view()`` (without specified dtype)
 630         will try using ``float64`` and may fail if the original data type size
 631         in bytes is not the same.
 632 
 633         Examples
 634         --------
 635         >>> s = pd.Series([-2, -1, 0, 1, 2], dtype='int8')
 636         >>> s
 637         0   -2
 638         1   -1
 639         2    0
 640         3    1
 641         4    2
 642         dtype: int8
 643 
 644         The 8 bit signed integer representation of `-1` is `0b11111111`, but
 645         the same bytes represent 255 if read as an 8 bit unsigned integer:
 646 
 647         >>> us = s.view('uint8')
 648         >>> us
 649         0    254
 650         1    255
 651         2      0
 652         3      1
 653         4      2
 654         dtype: uint8
 655 
 656         The views share the same underlying values:
 657 
 658         >>> us[0] = 128
 659         >>> s
 660         0   -128
 661         1     -1
 662         2      0
 663         3      1
 664         4      2
 665         dtype: int8
 666         """
 667         return self._constructor(
 668             self._values.view(dtype), index=self.index
 669         ).__finalize__(self, method="view")

 675     def __array_ufunc__(
 676         self, ufunc: Callable, method: str, *inputs: Any, **kwargs: Any
 677     ):
 678         # TODO: handle DataFrame
 679         cls = type(self)
 680 
 681         # for binary ops, use our custom dunder methods
 682         result = ops.maybe_dispatch_ufunc_to_dunder_op(
 683             self, ufunc, method, *inputs, **kwargs
 684         )
 685         if result is not NotImplemented:
 686             return result
 687 
 688         # Determine if we should defer.
 689         no_defer = (np.ndarray.__array_ufunc__, cls.__array_ufunc__)
 690 
 691         for item in inputs:
 692             higher_priority = (
 693                 hasattr(item, "__array_priority__")
 694                 and item.__array_priority__ > self.__array_priority__
 695             )
 696             has_array_ufunc = (
 697                 hasattr(item, "__array_ufunc__")
 698                 and type(item).__array_ufunc__ not in no_defer
 699                 and not isinstance(item, self._HANDLED_TYPES)
 700             )
 701             if higher_priority or has_array_ufunc:
 702                 return NotImplemented
 703 
 704         # align all the inputs.
 705         names = [getattr(x, "name") for x in inputs if hasattr(x, "name")]
 706         types = tuple(type(x) for x in inputs)
 707         # TODO: dataframe
 708         alignable = [x for x, t in zip(inputs, types) if issubclass(t, Series)]
 709 
 710         if len(alignable) > 1:
 711             # This triggers alignment.
 712             # At the moment, there aren't any ufuncs with more than two inputs
 713             # so this ends up just being x1.index | x2.index, but we write
 714             # it to handle *args.
 715             index = alignable[0].index
 716             for s in alignable[1:]:
 717                 index |= s.index
 718             inputs = tuple(
 719                 x.reindex(index) if issubclass(t, Series) else x
 720                 for x, t in zip(inputs, types)
 721             )
 722         else:
 723             index = self.index
 724 
 725         inputs = tuple(extract_array(x, extract_numpy=True) for x in inputs)
 726         result = getattr(ufunc, method)(*inputs, **kwargs)
 727 
 728         name = names[0] if len(set(names)) == 1 else None
 729 
 730         def construct_return(result):
 731             if lib.is_scalar(result):
 732                 return result
 733             elif result.ndim > 1:
 734                 # e.g. np.subtract.outer
 735                 if method == "outer":
 736                     # GH#27198
 737                     raise NotImplementedError
 738                 return result
 739             return self._constructor(result, index=index, name=name, copy=False)
 740 
 741         if type(result) is tuple:
 742             # multiple return values
 743             return tuple(construct_return(x) for x in result)
 744         elif method == "at":
 745             # no return value
 746             return None
 747         else:
 748             return construct_return(result)

 910     def _get_with(self, key):
 911         # other: fancy integer or otherwise
 912         if isinstance(key, slice):
 913             # _convert_slice_indexer to determine if this slice is positional
 914             #  or label based, and if the latter, convert to positional
 915             slobj = self.index._convert_slice_indexer(key, kind="getitem")
 916             return self._slice(slobj)
 917         elif isinstance(key, ABCDataFrame):
 918             raise TypeError(
 919                 "Indexing a Series with DataFrame is not "
 920                 "supported, use the appropriate DataFrame column"
 921             )
 922         elif isinstance(key, tuple):
 923             return self._get_values_tuple(key)
 924 
 925         elif not is_list_like(key):
 926             # e.g. scalars that aren't recognized by lib.is_scalar, GH#32684
 927             return self.loc[key]
 928 
 929         if not isinstance(key, (list, np.ndarray, ExtensionArray, Series, Index)):
 930             key = list(key)
 931 
 932         if isinstance(key, Index):
 933             key_type = key.inferred_type
 934         else:
 935             key_type = lib.infer_dtype(key, skipna=False)
 936 
 937         # Note: The key_type == "boolean" case should be caught by the
 938         #  com.is_bool_indexer check in __getitem__

 941             #  (i.e. self.iloc) or label-based (i.e. self.loc)
 942             if not self.index._should_fallback_to_positional():
 943                 return self.loc[key]
 944             else:
 945                 return self.iloc[key]

1164     def reset_index(self, level=None, drop=False, name=None, inplace=False):
1165         """
1166         Generate a new DataFrame or Series with the index reset.
1167 
1168         This is useful when the index needs to be treated as a column, or
1169         when the index is meaningless and needs to be reset to the default
1170         before another operation.
1171 
1172         Parameters
1173         ----------
1174         level : int, str, tuple, or list, default optional
1175             For a Series with a MultiIndex, only remove the specified levels
1176             from the index. Removes all levels by default.
1177         drop : bool, default False
1178             Just reset the index, without inserting it as a column in
1179             the new DataFrame.
1180         name : object, optional
1181             The name to use for the column containing the original Series
1182             values. Uses ``self.name`` by default. This argument is ignored
1183             when `drop` is True.
1184         inplace : bool, default False
1185             Modify the Series in place (do not create a new object).
1186 
1187         Returns
1188         -------
1189         Series or DataFrame
1190             When `drop` is False (the default), a DataFrame is returned.
1191             The newly created columns will come first in the DataFrame,
1192             followed by the original Series values.
1193             When `drop` is True, a `Series` is returned.
1194             In either case, if ``inplace=True``, no value is returned.
1195 
1196         See Also
1197         --------
1198         DataFrame.reset_index: Analogous function for DataFrame.
1199 
1200         Examples
1201         --------
1202         >>> s = pd.Series([1, 2, 3, 4], name='foo',
1203         ...               index=pd.Index(['a', 'b', 'c', 'd'], name='idx'))
1204 
1205         Generate a DataFrame with default index.
1206 
1207         >>> s.reset_index()
1208           idx  foo
1209         0   a    1
1210         1   b    2
1211         2   c    3
1212         3   d    4
1213 
1214         To specify the name of the new column use `name`.
1215 
1216         >>> s.reset_index(name='values')
1217           idx  values
1218         0   a       1
1219         1   b       2
1220         2   c       3
1221         3   d       4
1222 
1223         To generate a new Series with the default set `drop` to True.
1224 
1225         >>> s.reset_index(drop=True)
1226         0    1
1227         1    2
1228         2    3
1229         3    4
1230         Name: foo, dtype: int64
1231 
1232         To update the Series in place, without generating a new one
1233         set `inplace` to True. Note that it also requires ``drop=True``.
1234 
1235         >>> s.reset_index(inplace=True, drop=True)
1236         >>> s
1237         0    1
1238         1    2
1239         2    3
1240         3    4
1241         Name: foo, dtype: int64
1242 
1243         The `level` parameter is interesting for Series with a multi-level
1244         index.
1245 
1246         >>> arrays = [np.array(['bar', 'bar', 'baz', 'baz']),
1247         ...           np.array(['one', 'two', 'one', 'two'])]
1248         >>> s2 = pd.Series(
1249         ...     range(4), name='foo',
1250         ...     index=pd.MultiIndex.from_arrays(arrays,
1251         ...                                     names=['a', 'b']))
1252 
1253         To remove a specific level from the Index, use `level`.
1254 
1255         >>> s2.reset_index(level='a')
1256                a  foo
1257         b
1258         one  bar    0
1259         two  bar    1
1260         one  baz    2
1261         two  baz    3
1262 
1263         If `level` is not set, all levels are removed from the Index.
1264 
1265         >>> s2.reset_index()
1266              a    b  foo
1267         0  bar  one    0
1268         1  bar  two    1
1269         2  baz  one    2
1270         3  baz  two    3
1271         """
1272         inplace = validate_bool_kwarg(inplace, "inplace")
1273         if drop:
1274             new_index = ibase.default_index(len(self))
1275             if level is not None:
1276                 if not isinstance(level, (tuple, list)):
1277                     level = [level]
1278                 level = [self.index._get_level_number(lev) for lev in level]
1279                 if len(level) < self.index.nlevels:
1280                     new_index = self.index.droplevel(level)
1281 
1282             if inplace:
1283                 self.index = new_index
1284                 # set name if it was passed, otherwise, keep the previous name
1285                 self.name = name or self.name
1286             else:
1287                 return self._constructor(
1288                     self._values.copy(), index=new_index
1289                 ).__finalize__(self, method="reset_index")
1290         elif inplace:
1291             raise TypeError(
1292                 "Cannot reset_index inplace on a Series to create a DataFrame"
1293             )
1294         else:
1295             df = self.to_frame(name)
1296             return df.reset_index(level=level, drop=drop)
