 124 def pin_whitelisted_properties(klass: Type[FrameOrSeries], whitelist: FrozenSet[str]):
 125     """
 126     Create GroupBy member defs for DataFrame/Series names in a whitelist.
 127 
 128     Parameters
 129     ----------
 130     klass : DataFrame or Series class
 131         class where members are defined.
 132     whitelist : frozenset[str]
 133         Set of names of klass methods to be constructed
 134 
 135     Returns
 136     -------
 137     class decorator
 138 
 139     Notes
 140     -----
 141     Since we don't want to override methods explicitly defined in the
 142     base class, any such name is skipped.
 143     """
 144 
 145     def pinner(cls):
 146         for name in whitelist:
 147             if hasattr(cls, name):
 148                 # don't override anything that was explicitly defined
 149                 #  in the base class
 150                 continue
 151 
 152             prop = generate_property(name, klass)
 153             setattr(cls, name, prop)
 154 
 155         return cls
 156 
 157     return pinner

 287     def _aggregate_multiple_funcs(self, arg):
 288         if isinstance(arg, dict):
 289 
 290             # show the deprecation, but only if we
 291             # have not shown a higher level one
 292             # GH 15931
 293             if isinstance(self._selected_obj, Series):
 294                 raise SpecificationError("nested renamer is not supported")
 295 
 296             columns = list(arg.keys())
 297             arg = arg.items()
 298         elif any(isinstance(x, (tuple, list)) for x in arg):
 299             arg = [(x, x) if not isinstance(x, (tuple, list)) else x for x in arg]
 300 
 301             # indicated column order
 302             columns = next(zip(*arg))
 303         else:
 304             # list of functions / function names
 305             columns = []
 306             for f in arg:
 307                 columns.append(com.get_callable_name(f) or f)

 312         for name, func in arg:
 313             obj = self

 469     def transform(self, func, *args, engine="cython", engine_kwargs=None, **kwargs):
 470         func = self._get_cython_func(func) or func
 471 
 472         if not isinstance(func, str):
 473             return self._transform_general(
 474                 func, *args, engine=engine, engine_kwargs=engine_kwargs, **kwargs
 475             )
 476 
 477         elif func not in base.transform_kernel_whitelist:
 478             msg = f"'{func}' is not a valid function name for transform(name)"
 479             raise ValueError(msg)
 480         elif func in base.cythonized_kernels:
 481             # cythonized transform or canned "agg+broadcast"
 482             return getattr(self, func)(*args, **kwargs)
 483         elif func in base.transformation_kernels:
 484             return getattr(self, func)(*args, **kwargs)
 485 
 486         # If func is a reduction, we need to broadcast the
 487         # result to the whole group. Compute func result
 488         # and deal with possible broadcasting below.
 489         result = getattr(self, func)(*args, **kwargs)
 490         return self._transform_fast(result, func)

 492     def _transform_general(
 493         self, func, *args, engine="cython", engine_kwargs=None, **kwargs
 494     ):
 495         """
 496         Transform with a non-str `func`.
 497         """
 498 
 499         if engine == "numba":
 500             numba_func, cache_key = generate_numba_func(
 501                 func, engine_kwargs, kwargs, "groupby_transform"
 502             )
 503 
 504         klass = type(self._selected_obj)
 505 
 506         results = []
 507         for name, group in self:
 508             object.__setattr__(group, "name", name)
 509             if engine == "numba":
 510                 values, index = split_for_numba(group)
 511                 res = numba_func(values, index, *args)
 512                 if cache_key not in NUMBA_FUNC_CACHE:
 513                     NUMBA_FUNC_CACHE[cache_key] = numba_func
 514             else:
 515                 res = func(group, *args, **kwargs)
 516 
 517             if isinstance(res, (ABCDataFrame, ABCSeries)):
 518                 res = res._values
 519 
 520             indexer = self._get_index(name)
 521             ser = klass(res, indexer)
 522             results.append(ser)
 523 
 524         # check for empty "results" to avoid concat ValueError
 525         if results:
 526             from pandas.core.reshape.concat import concat
 527 
 528             result = concat(results).sort_index()
 529         else:
 530             result = self.obj._constructor(dtype=np.float64)
 531 
 532         # we will only try to coerce the result type if
 533         # we have a numeric dtype, as these are *always* user-defined funcs
 534         # the cython take a different path (and casting)
 535         dtype = self._selected_obj.dtype
 536         if is_numeric_dtype(dtype):
 537             result = maybe_downcast_to_dtype(result, dtype)
 538 
 539         result.name = self._selected_obj.name
 540         result.index = self._selected_obj.index
 541         return result

 556     def filter(self, func, dropna=True, *args, **kwargs):
 557         """
 558         Return a copy of a Series excluding elements from groups that
 559         do not satisfy the boolean criterion specified by func.
 560 
 561         Parameters
 562         ----------
 563         func : function
 564             To apply to each group. Should return True or False.
 565         dropna : Drop groups that do not pass the filter. True by default;
 566             if False, groups that evaluate False are filled with NaNs.
 567 
 568         Examples
 569         --------
 570         >>> df = pd.DataFrame({'A' : ['foo', 'bar', 'foo', 'bar',
 571         ...                           'foo', 'bar'],
 572         ...                    'B' : [1, 2, 3, 4, 5, 6],
 573         ...                    'C' : [2.0, 5., 8., 1., 2., 9.]})
 574         >>> grouped = df.groupby('A')
 575         >>> df.groupby('A').B.filter(lambda x: x.mean() > 3.)
 576         1    2
 577         3    4
 578         5    6
 579         Name: B, dtype: int64
 580 
 581         Returns
 582         -------
 583         filtered : Series
 584         """
 585         if isinstance(func, str):
 586             wrapper = lambda x: getattr(x, func)(*args, **kwargs)
 587         else:
 588             wrapper = lambda x: func(x, *args, **kwargs)
 589 
 590         # Interpret np.nan as False.
 591         def true_and_notna(x, *args, **kwargs) -> bool:
 592             b = wrapper(x, *args, **kwargs)
 593             return b and notna(b)
 594 
 595         try:
 596             indices = [
 597                 self._get_index(name) for name, group in self if true_and_notna(group)
 598             ]
 599         except (ValueError, TypeError) as err:
 600             raise TypeError("the filter must return a boolean result") from err
 601 
 602         filtered = self._apply_filter(indices, dropna)
 603         return filtered

 605     def nunique(self, dropna: bool = True) -> Series:
 606         """
 607         Return number of unique elements in the group.
 608 
 609         Returns
 610         -------
 611         Series
 612             Number of unique values within each group.
 613         """
 614         ids, _, _ = self.grouper.group_info
 615 
 616         val = self.obj._values
 617 
 618         codes, _ = algorithms.factorize(val, sort=False)
 619         sorter = np.lexsort((codes, ids))
 620         codes = codes[sorter]
 621         ids = ids[sorter]
 622 
 623         # group boundaries are where group ids change
 624         # unique observations are where sorted values change
 625         idx = np.r_[0, 1 + np.nonzero(ids[1:] != ids[:-1])[0]]
 626         inc = np.r_[1, codes[1:] != codes[:-1]]
 627 
 628         # 1st item of each group is a new unique observation
 629         mask = codes == -1
 630         if dropna:
 631             inc[idx] = 1
 632             inc[mask] = 0
 633         else:
 634             inc[mask & np.r_[False, mask[:-1]]] = 0
 635             inc[idx] = 1
 636 
 637         out = np.add.reduceat(inc, idx).astype("int64", copy=False)
 638         if len(ids):
 639             # NaN/NaT group exists if the head of ids is -1,
 640             # so remove it from res and exclude its index from idx
 641             if ids[0] == -1:
 642                 res = out[1:]
 643                 idx = idx[np.flatnonzero(idx)]
 644             else:
 645                 res = out
 646         else:
 647             res = out[1:]
 648         ri = self.grouper.result_index
 649 
 650         # we might have duplications among the bins
 651         if len(res) != len(ri):
 652             res, out = np.zeros(len(ri), dtype=out.dtype), res
 653             res[ids[idx]] = out
 654 
 655         result = self.obj._constructor(res, index=ri, name=self._selection_name)
 656         return self._reindex_output(result, fill_value=0)
