  65 class BaseGrouper:
  66     """
  67     This is an internal Grouper class, which actually holds
  68     the generated groups
  69 
  70     Parameters
  71     ----------
  72     axis : Index
  73     groupings : Sequence[Grouping]
  74         all the grouping instances to handle in this grouper
  75         for example for grouper list to groupby, need to pass the list
  76     sort : bool, default True
  77         whether this grouper will give sorted result or not

  79     mutated : bool, default False

 121     def get_iterator(self, data: FrameOrSeries, axis: int = 0):
 122         """
 123         Groupby iterator
 124 
 125         Returns
 126         -------
 127         Generator yielding sequence of (name, subsetted object)
 128         for each group
 129         """
 130         splitter = self._get_splitter(data, axis=axis)
 131         keys = self._get_group_keys()
 132         for key, (i, group) in zip(keys, splitter):
 133             yield key, group
 134 
 135     def _get_splitter(self, data: FrameOrSeries, axis: int = 0) -> "DataSplitter":
 136         comp_ids, _, ngroups = self.group_info
 137         return get_splitter(data, comp_ids, ngroups, axis=axis)

 139     def _get_grouper(self):
 140         """
 141         We are a grouper as part of another's groupings.
 142 
 143         We have a specific method of grouping, so cannot
 144         convert to a Index for our grouper.
 145         """
 146         return self.groupings[0].grouper

 148     def _get_group_keys(self):
 149         if len(self.groupings) == 1:
 150             return self.levels[0]
 151         else:
 152             comp_ids, _, ngroups = self.group_info
 153 
 154             # provide "flattened" iterator for multi-group setting
 155             return get_flattened_iterator(comp_ids, ngroups, self.levels, self.codes)

 157     def apply(self, f: F, data: FrameOrSeries, axis: int = 0):
 158         mutated = self.mutated
 159         splitter = self._get_splitter(data, axis=axis)
 160         group_keys = self._get_group_keys()
 161         result_values = None
 162 
 163         sdata: FrameOrSeries = splitter._get_sorted_data()
 164         if sdata.ndim == 2 and np.any(sdata.dtypes.apply(is_extension_array_dtype)):
 165             # calling splitter.fast_apply will raise TypeError via apply_frame_axis0
 166             #  if we pass EA instead of ndarray
 167             #  TODO: can we have a workaround for EAs backed by ndarray?
 168             pass
 169 
 170         elif (
 171             com.get_callable_name(f) not in base.plotting_methods
 172             and isinstance(splitter, FrameSplitter)
 173             and axis == 0
 174             # fast_apply/libreduction doesn't allow non-numpy backed indexes
 175             and not sdata.index._has_complex_internals
 176         ):
 177             try:
 178                 result_values, mutated = splitter.fast_apply(f, sdata, group_keys)
 179 
 180             except libreduction.InvalidApply as err:
 181                 # This Exception is raised if `f` triggers an exception
 182                 # but it is preferable to raise the exception in Python.
 183                 if "Let this error raise above us" not in str(err):
 184                     # TODO: can we infer anything about whether this is
 185                     #  worth-retrying in pure-python?
 186                     raise
 187 
 188             else:
 189                 # If the fast apply path could be used we can return here.
 190                 # Otherwise we need to fall back to the slow implementation.
 191                 if len(result_values) == len(group_keys):
 192                     return group_keys, result_values, mutated
 193 
 194         for key, (i, group) in zip(group_keys, splitter):
 195             object.__setattr__(group, "name", key)
 196 
 197             # result_values is None if fast apply path wasn't taken
 198             # or fast apply aborted with an unexpected exception.
 199             # In either case, initialize the result list and perform
 200             # the slow iteration.
 201             if result_values is None:
 202                 result_values = []
 203 
 204             # If result_values is not None we're in the case that the
 205             # fast apply loop was broken prematurely but we have
 206             # already the result for the first group which we can reuse.
 207             elif i == 0:
 208                 continue
 209 
 210             # group might be modified
 211             group_axes = group.axes
 212             res = f(group)
 213             if not _is_indexed_like(res, group_axes):
 214                 mutated = True
 215             result_values.append(res)
 216 
 217         return group_keys, result_values, mutated

 241     def size(self) -> Series:
 242         """
 243         Compute group sizes.
 244         """
 245         ids, _, ngroup = self.group_info
 246         ids = ensure_platform_int(ids)
 247         if ngroup:
 248             out = np.bincount(ids[ids != -1], minlength=ngroup)
 249         else:
 250             out = []
 251         return Series(out, index=self.result_index, dtype="int64")

 276     @cache_readonly
 277     def codes_info(self) -> np.ndarray:
 278         # return the codes of items in original grouped axis
 279         codes, _, _ = self.group_info

 285     def _get_compressed_codes(self) -> Tuple[np.ndarray, np.ndarray]:
 286         all_codes = self.codes
 287         if len(all_codes) > 1:
 288             group_index = get_group_index(all_codes, self.shape, sort=True, xnull=True)
 289             return compress_group_index(group_index, sort=self.sort)

 291         ping = self.groupings[0]

 304     @cache_readonly
 305     def result_index(self) -> Index:
 306         if not self.compressed and len(self.groupings) == 1:
 307             return self.groupings[0].result_index.rename(self.names[0])
 308 
 309         codes = self.reconstructed_codes
 310         levels = [ping.result_index for ping in self.groupings]
 311         result = MultiIndex(
 312             levels=levels, codes=codes, verify_integrity=False, names=self.names
 313         )

 316     def get_group_levels(self) -> List[Index]:
 317         if not self.compressed and len(self.groupings) == 1:
 318             return [self.groupings[0].result_index]
 319 
 320         name_list = []
 321         for ping, codes in zip(self.groupings, self.reconstructed_codes):
 322             codes = ensure_platform_int(codes)
 323             levels = ping.result_index.take(codes)

 325             name_list.append(levels)

 358     def _is_builtin_func(self, arg):

 362         """
 363         return SelectionMixin._builtin_table.get(arg, arg)

 365     def _get_cython_function(
 366         self, kind: str, how: str, values: np.ndarray, is_numeric: bool
 367     ):
 368 
 369         dtype_str = values.dtype.name
 370         ftype = self._cython_functions[kind][how]
 371 
 372         # see if there is a fused-type version of function
 373         # only valid for numeric
 374         f = getattr(libgroupby, ftype, None)
 375         if f is not None and is_numeric:
 376             return f
 377 
 378         # otherwise find dtype-specific version, falling back to object
 379         for dt in [dtype_str, "object"]:
 380             f2 = getattr(libgroupby, f"{ftype}_{dt}", None)
 381             if f2 is not None:
 382                 return f2
 383 
 384         if hasattr(f, "__signatures__"):
 385             # inspect what fused types are implemented
 386             if dtype_str == "object" and "object" not in f.__signatures__:
 387                 # disallow this function so we get a NotImplementedError below
 388                 #  instead of a TypeError at runtime
 389                 f = None
 390 
 391         func = f
 392 
 393         if func is None:
 394             raise NotImplementedError(
 395                 f"function is not implemented for this dtype: "
 396                 f"[how->{how},dtype->{dtype_str}]"
 397             )
 398 
 399         return func

 401     def _get_cython_func_and_vals(
 402         self, kind: str, how: str, values: np.ndarray, is_numeric: bool
 403     ):
 404         """
 405         Find the appropriate cython function, casting if necessary.
 406 
 407         Parameters
 408         ----------
 409         kind : str
 410         how : str
 411         values : np.ndarray
 412         is_numeric : bool
 413 
 414         Returns
 415         -------
 416         func : callable
 417         values : np.ndarray
 418         """
 419         try:
 420             func = self._get_cython_function(kind, how, values, is_numeric)
 421         except NotImplementedError:
 422             if is_numeric:
 423                 try:
 424                     values = ensure_float64(values)
 425                 except TypeError:
 426                     if lib.infer_dtype(values, skipna=False) == "complex":
 427                         values = values.astype(complex)
 428                     else:
 429                         raise
 430                 func = self._get_cython_function(kind, how, values, is_numeric)
 431             else:
 432                 raise
 433         return func, values

 435     def _cython_operation(
 436         self, kind: str, values, how: str, axis: int, min_count: int = -1, **kwargs
 437     ) -> Tuple[np.ndarray, Optional[List[str]]]:
 438         """
 439         Returns the values of a cython operation as a Tuple of [data, names].
 440 
 441         Names is only useful when dealing with 2D results, like ohlc
 442         (see self._name_functions).
 443         """
 444         assert kind in ["transform", "aggregate"]
 445         orig_values = values
 446 
 447         if values.ndim > 2:
 448             raise NotImplementedError("number of dimensions is currently limited to 2")
 449         elif values.ndim == 2:
 450             # Note: it is *not* the case that axis is always 0 for 1-dim values,
 451             #  as we can have 1D ExtensionArrays that we need to treat as 2D
 452             assert axis == 1, axis
 453 
 454         # can we do this operation with our cython functions
 455         # if not raise NotImplementedError
 456 
 457         # we raise NotImplemented if this is an invalid operation
 458         # entirely, e.g. adding datetimes
 459 
 460         # categoricals are only 1d, so we
 461         # are not setup for dim transforming
 462         if is_categorical_dtype(values.dtype) or is_sparse(values.dtype):
 463             raise NotImplementedError(f"{values.dtype} dtype not supported")
 464         elif is_datetime64_any_dtype(values.dtype):
 465             if how in ["add", "prod", "cumsum", "cumprod"]:
 466                 raise NotImplementedError(
 467                     f"datetime64 type does not support {how} operations"
 468                 )
 469         elif is_timedelta64_dtype(values.dtype):
 470             if how in ["prod", "cumprod"]:
 471                 raise NotImplementedError(
 472                     f"timedelta64 type does not support {how} operations"
 473                 )
 474 
 475         if is_datetime64tz_dtype(values.dtype):
 476             # Cast to naive; we'll cast back at the end of the function
 477             # TODO: possible need to reshape?
 478             # TODO(EA2D):kludge can be avoided when 2D EA is allowed.
 479             values = values.view("M8[ns]")
 480 
 481         is_datetimelike = needs_i8_conversion(values.dtype)
 482         is_numeric = is_numeric_dtype(values.dtype)
 483 
 484         if is_datetimelike:
 485             values = values.view("int64")
 486             is_numeric = True
 487         elif is_bool_dtype(values.dtype):
 488             values = ensure_int_or_float(values)
 489         elif is_integer_dtype(values):
 490             # we use iNaT for the missing value on ints
 491             # so pre-convert to guard this condition
 492             if (values == iNaT).any():
 493                 values = ensure_float64(values)
 494             else:
 495                 values = ensure_int_or_float(values)
 496         elif is_numeric and not is_complex_dtype(values):
 497             values = ensure_float64(values)
 498         else:
 499             values = values.astype(object)
 500 
 501         arity = self._cython_arity.get(how, 1)
 502 
 503         vdim = values.ndim
 504         swapped = False
 505         if vdim == 1:
 506             values = values[:, None]
 507             out_shape = (self.ngroups, arity)
 508         else:
 509             if axis > 0:
 510                 swapped = True
 511                 assert axis == 1, axis
 512                 values = values.T
 513             if arity > 1:
 514                 raise NotImplementedError(
 515                     "arity of more than 1 is not supported for the 'how' argument"
 516                 )
 517             out_shape = (self.ngroups,) + values.shape[1:]
 518 
 519         func, values = self._get_cython_func_and_vals(kind, how, values, is_numeric)
 520 
 521         if how == "rank":
 522             out_dtype = "float"
 523         else:
 524             if is_numeric:
 525                 out_dtype = f"{values.dtype.kind}{values.dtype.itemsize}"
 526             else:
 527                 out_dtype = "object"
 528 
 529         codes, _, _ = self.group_info
 530 
 531         if kind == "aggregate":
 532             result = _maybe_fill(
 533                 np.empty(out_shape, dtype=out_dtype), fill_value=np.nan
 534             )
 535             counts = np.zeros(self.ngroups, dtype=np.int64)
 536             result = self._aggregate(result, counts, values, codes, func, min_count)
 537         elif kind == "transform":
 538             result = _maybe_fill(
 539                 np.empty_like(values, dtype=out_dtype), fill_value=np.nan
 540             )
 541 
 542             # TODO: min_count
 543             result = self._transform(
 544                 result, values, codes, func, is_datetimelike, **kwargs
 545             )
 546 
 547         if is_integer_dtype(result) and not is_datetimelike:
 548             mask = result == iNaT
 549             if mask.any():
 550                 result = result.astype("float64")
 551                 result[mask] = np.nan
 552 
 553         if kind == "aggregate" and self._filter_empty_groups and not counts.all():
 554             assert result.ndim != 2
 555             result = result[counts > 0]
 556 
 557         if vdim == 1 and arity == 1:
 558             result = result[:, 0]
 559 
 560         names: Optional[List[str]] = self._name_functions.get(how, None)
 561 
 562         if swapped:
 563             result = result.swapaxes(0, axis)
 564 
 565         if is_datetime64tz_dtype(orig_values.dtype) or is_period_dtype(
 566             orig_values.dtype
 567         ):
 568             # We need to use the constructors directly for these dtypes
 569             # since numpy won't recognize them
 570             # https://github.com/pandas-dev/pandas/issues/31471
 571             result = type(orig_values)(result.astype(np.int64), dtype=orig_values.dtype)
 572         elif is_datetimelike and kind == "aggregate":
 573             result = result.astype(orig_values.dtype)
 574 
 575         if is_extension_array_dtype(orig_values.dtype):
 576             result = maybe_cast_result(result=result, obj=orig_values, how=how)
 577 
 578         return result, names

 590     def _aggregate(
 591         self, result, counts, values, comp_ids, agg_func, min_count: int = -1,
 592     ):
 593         if agg_func is libgroupby.group_nth:
 594             # different signature from the others
 595             # TODO: should we be using min_count instead of hard-coding it?
 596             agg_func(result, counts, values, comp_ids, rank=1, min_count=-1)
 597         else:
 598             agg_func(result, counts, values, comp_ids, min_count)
 599 
 600         return result

 611     def agg_series(
 612         self,
 613         obj: Series,
 614         func: F,
 615         *args,
 616         engine: str = "cython",
 617         engine_kwargs=None,
 618         **kwargs,
 619     ):
 620         # Caller is responsible for checking ngroups != 0
 621         assert self.ngroups != 0
 622 
 623         if engine == "numba":
 624             return self._aggregate_series_pure_python(
 625                 obj, func, *args, engine=engine, engine_kwargs=engine_kwargs, **kwargs
 626             )
 627         if len(obj) == 0:
 628             # SeriesGrouper would raise if we were to call _aggregate_series_fast
 629             return self._aggregate_series_pure_python(obj, func)
 630 
 631         elif is_extension_array_dtype(obj.dtype):
 632             # _aggregate_series_fast would raise TypeError when
 633             #  calling libreduction.Slider
 634             # In the datetime64tz case it would incorrectly cast to tz-naive
 635             # TODO: can we get a performant workaround for EAs backed by ndarray?
 636             return self._aggregate_series_pure_python(obj, func)
 637 
 638         elif obj.index._has_complex_internals:
 639             # Preempt TypeError in _aggregate_series_fast
 640             return self._aggregate_series_pure_python(obj, func)
 641 
 642         try:
 643             return self._aggregate_series_fast(obj, func)
 644         except ValueError as err:
 645             if "Function does not reduce" in str(err):
 646                 # raised in libreduction
 647                 pass
 648             else:
 649                 raise
 650         return self._aggregate_series_pure_python(obj, func)
 651 
 652     def _aggregate_series_fast(self, obj: Series, func: F):
 653         # At this point we have already checked that
 654         #  - obj.index is not a MultiIndex
 655         #  - obj is backed by an ndarray, not ExtensionArray
 656         #  - len(obj) > 0
 657         #  - ngroups != 0
 658         func = self._is_builtin_func(func)
 659 
 660         group_index, _, ngroups = self.group_info
 661 
 662         # avoids object / Series creation overhead
 663         dummy = obj.iloc[:0]
 664         indexer = get_group_index_sorter(group_index, ngroups)
 665         obj = obj.take(indexer)
 666         group_index = algorithms.take_nd(group_index, indexer, allow_fill=False)
 667         grouper = libreduction.SeriesGrouper(obj, func, group_index, ngroups, dummy)
 668         result, counts = grouper.get_result()
 669         return result, counts
 670 
 671     def _aggregate_series_pure_python(
 672         self,
 673         obj: Series,
 674         func: F,
 675         *args,
 676         engine: str = "cython",
 677         engine_kwargs=None,
 678         **kwargs,
 679     ):
 680 
 681         if engine == "numba":
 682             numba_func, cache_key = generate_numba_func(
 683                 func, engine_kwargs, kwargs, "groupby_agg"
 684             )
 685 
 686         group_index, _, ngroups = self.group_info
 687 
 688         counts = np.zeros(ngroups, dtype=int)
 689         result = None
 690 
 691         splitter = get_splitter(obj, group_index, ngroups, axis=0)
 692 
 693         for label, group in splitter:
 694             if engine == "numba":
 695                 values, index = split_for_numba(group)
 696                 res = numba_func(values, index, *args)
 697                 if cache_key not in NUMBA_FUNC_CACHE:
 698                     NUMBA_FUNC_CACHE[cache_key] = numba_func
 699             else:
 700                 res = func(group, *args, **kwargs)
 701 
 702             if result is None:
 703                 if isinstance(res, (Series, Index, np.ndarray)):
 704                     if len(res) == 1:
 705                         # e.g. test_agg_lambda_with_timezone lambda e: e.head(1)
 706                         # FIXME: are we potentially losing important res.index info?
 707                         res = res.item()
 708                     else:
 709                         raise ValueError("Function does not reduce")
 710                 result = np.empty(ngroups, dtype="O")
 711 
 712             counts[label] = group.shape[0]
 713             result[label] = res
 714 
 715         assert result is not None
 716         result = lib.maybe_convert_objects(result, try_float=0)
 717         # TODO: maybe_cast_to_extension_array?
 718 
 719         return result, counts

 722 class BinGrouper(BaseGrouper):
 723     """
 724     This is an internal Grouper class
 725 
 726     Parameters
 727     ----------
 728     bins : the split index of binlabels to group the item of axis
 729     binlabels : the label list
 730     filter_empty : boolean, default False
 731     mutated : boolean, default False
 732     indexer : a intp array
 733 
 734     Examples
 735     --------
 736     bins: [2, 4, 6, 8, 10]
 737     binlabels: DatetimeIndex(['2005-01-01', '2005-01-03',
 738         '2005-01-05', '2005-01-07', '2005-01-09'],
 739         dtype='datetime64[ns]', freq='2D')
 740 
 741     the group_info, which contains the label of each item in grouped
 742     axis, the index of label in label list, group number, is
 743 
 744     (array([0, 0, 1, 1, 2, 2, 3, 3, 4, 4]), array([0, 1, 2, 3, 4]), 5)
 745 
 746     means that, the grouped axis has 10 items, can be grouped into 5
 747     labels, the first and second items belong to the first label, the
 748     third and forth items belong to the second label, and so on
 749 
 750     """
 751 
 752     def __init__(
 753         self,
 754         bins,
 755         binlabels,
 756         filter_empty: bool = False,
 757         mutated: bool = False,
 758         indexer=None,
 759     ):
 760         self.bins = ensure_int64(bins)
 761         self.binlabels = ensure_index(binlabels)
 762         self._filter_empty_groups = filter_empty
 763         self.mutated = mutated
 764         self.indexer = indexer
 765 
 766         # These lengths must match, otherwise we could call agg_series
 767         #  with empty self.bins, which would raise in libreduction.
 768         assert len(self.binlabels) == len(self.bins)
 769 
 770     @cache_readonly
 771     def groups(self):
 772         """ dict {group name -> group labels} """
 773         # this is mainly for compat
 774         # GH 3881
 775         result = {
 776             key: value
 777             for key, value in zip(self.binlabels, self.bins)
 778             if key is not NaT
 779         }
 780         return result
 781 
 782     @property
 783     def nkeys(self) -> int:
 784         return 1
 785 
 786     def _get_grouper(self):
 787         """
 788         We are a grouper as part of another's groupings.
 789 
 790         We have a specific method of grouping, so cannot
 791         convert to a Index for our grouper.
 792         """
 793         return self

 795     def get_iterator(self, data: FrameOrSeries, axis: int = 0):
 796         """
 797         Groupby iterator
 798 
 799         Returns
 800         -------
 801         Generator yielding sequence of (name, subsetted object)
 802         for each group
 803         """
 804         if axis == 0:
 805             slicer = lambda start, edge: data.iloc[start:edge]
 806         else:
 807             slicer = lambda start, edge: data.iloc[:, start:edge]
 808 
 809         length = len(data.axes[axis])
 810 
 811         start = 0
 812         for edge, label in zip(self.bins, self.binlabels):
 813             if label is not NaT:
 814                 yield label, slicer(start, edge)
 815             start = edge
 816 
 817         if start < length:
 818             yield self.binlabels[-1], slicer(start, None)
 819 
 820     @cache_readonly
 821     def indices(self):
 822         indices = collections.defaultdict(list)
 823 
 824         i = 0
 825         for label, bin in zip(self.binlabels, self.bins):
 826             if i < bin:
 827                 if label is not NaT:
 828                     indices[label] = list(range(i, bin))
 829                 i = bin
 830         return indices

 832     @cache_readonly
 833     def group_info(self):
 834         ngroups = self.ngroups
 835         obs_group_ids = np.arange(ngroups)
 836         rep = np.diff(np.r_[0, self.bins])
 837 
 838         rep = ensure_platform_int(rep)
 839         if ngroups == len(self.bins):
 840             comp_ids = np.repeat(np.arange(ngroups), rep)
 841         else:
 842             comp_ids = np.repeat(np.r_[-1, np.arange(ngroups)], rep)
 843 
 844         return (
 845             comp_ids.astype("int64", copy=False),
 846             obs_group_ids.astype("int64", copy=False),
 847             ngroups,
 848         )

 850     @cache_readonly
 851     def reconstructed_codes(self) -> List[np.ndarray]:
 852         # get unique result indices, and prepend 0 as groupby starts from the first
 853         return [np.r_[0, np.flatnonzero(self.bins[1:] != self.bins[:-1]) + 1]]

 855     @cache_readonly
 856     def result_index(self):
 857         if len(self.binlabels) != 0 and isna(self.binlabels[0]):
 858             return self.binlabels[1:]
 859 
 860         return self.binlabels

 864         return [self.binlabels]

 877     def agg_series(
 878         self,
 879         obj: Series,
 880         func: F,
 881         *args,
 882         engine: str = "cython",
 883         engine_kwargs=None,
 884         **kwargs,
 885     ):
 886         # Caller is responsible for checking ngroups != 0
 887         assert self.ngroups != 0
 888         assert len(self.bins) > 0  # otherwise we'd get IndexError in get_result
 889 
 890         if is_extension_array_dtype(obj.dtype):
 891             # preempt SeriesBinGrouper from raising TypeError
 892             return self._aggregate_series_pure_python(obj, func)
 893 
 894         dummy = obj[:0]
 895         grouper = libreduction.SeriesBinGrouper(obj, func, self.bins, dummy)
 896         return grouper.get_result()

 900     if isinstance(obj, Series):
 901         if len(axes) > 1:
 902             return False

 914 class DataSplitter:
 915     def __init__(self, data: FrameOrSeries, labels, ngroups: int, axis: int = 0):
 916         self.data = data
 917         self.labels = ensure_int64(labels)
 918         self.ngroups = ngroups
 919 
 920         self.axis = axis
 921         assert isinstance(axis, int), axis
 922 
 923     @cache_readonly
 924     def slabels(self):
 925         # Sorted labels
 926         return algorithms.take_nd(self.labels, self.sort_idx, allow_fill=False)
 927 
 928     @cache_readonly
 929     def sort_idx(self):
 930         # Counting sort indexer
 931         return get_group_index_sorter(self.labels, self.ngroups)
 932 
 933     def __iter__(self):
 934         sdata = self._get_sorted_data()
 935 
 936         if self.ngroups == 0:
 937             # we are inside a generator, rather than raise StopIteration
 938             # we merely return signal the end
 939             return
 940 
 941         starts, ends = lib.generate_slices(self.slabels, self.ngroups)
 942 
 943         for i, (start, end) in enumerate(zip(starts, ends)):
 944             yield i, self._chop(sdata, slice(start, end))
 945 
 946     def _get_sorted_data(self) -> FrameOrSeries:
 947         return self.data.take(self.sort_idx, axis=self.axis)
 948 
 949     def _chop(self, sdata, slice_obj: slice) -> NDFrame:
 950         raise AbstractMethodError(self)
 951

 953 class SeriesSplitter(DataSplitter):
 954     def _chop(self, sdata: Series, slice_obj: slice) -> Series:
 955         # fastpath equivalent to `sdata.iloc[slice_obj]`
 956         mgr = sdata._mgr.get_slice(slice_obj)
 957         return type(sdata)(mgr, name=sdata.name, fastpath=True)
 958 
 959 
 960 class FrameSplitter(DataSplitter):
 961     def fast_apply(self, f: F, sdata: FrameOrSeries, names):
 962         # must return keys::list, values::list, mutated::bool
 963         starts, ends = lib.generate_slices(self.slabels, self.ngroups)
 964         return libreduction.apply_frame_axis0(sdata, f, names, starts, ends)
 965 
 966     def _chop(self, sdata: DataFrame, slice_obj: slice) -> DataFrame:
 967         # Fastpath equivalent to:
 968         # if self.axis == 0:
 969         #     return sdata.iloc[slice_obj]
 970         # else:
 971         #     return sdata.iloc[:, slice_obj]
 972         mgr = sdata._mgr.get_slice(slice_obj, axis=1 - self.axis)
 973         return type(sdata)(mgr)

 976 def get_splitter(
 977     data: FrameOrSeries, labels: np.ndarray, ngroups: int, axis: int = 0
 978 ) -> DataSplitter:
 979     if isinstance(data, Series):
 980         klass: Type[DataSplitter] = SeriesSplitter
 981     else:
 982         # i.e. DataFrame
 983         klass = FrameSplitter
 984 
 985     return klass(data, labels, ngroups, axis)
