  35 class Grouper:
  36     """
  37     A Grouper allows the user to specify a groupby instruction for an object.
  38 
  39     This specification will select a column via the key parameter, or if the
  40     level and/or axis parameters are given, a level of the index of the target
  41     object.
  42 
  43     If `axis` and/or `level` are passed as keywords to both `Grouper` and
  44     `groupby`, the values passed to `Grouper` take precedence.
  45 
  46     Parameters
  47     ----------
  48     key : str, defaults to None
  49         Groupby key, which selects the grouping column of the target.
  50     level : name/number, defaults to None
  51         The level for the target index.
  52     freq : str / frequency object, defaults to None
  53         This will groupby the specified frequency if the target selection
  54         (via key or level) is a datetime-like object. For full specification
  55         of available frequencies, please see `here
  56         <https://pandas.pydata.org/pandas-docs/stable/user_guide/timeseries.html#offset-aliases>`_.
  57     axis : str, int, defaults to 0
  58         Number/name of the axis.
  59     sort : bool, default to False
  60         Whether to sort the resulting labels.
  61     closed : {'left' or 'right'}
  62         Closed end of interval. Only when `freq` parameter is passed.
  63     label : {'left' or 'right'}
  64         Interval boundary to use for labeling.
  65         Only when `freq` parameter is passed.
  66     convention : {'start', 'end', 'e', 's'}
  67         If grouper is PeriodIndex and `freq` parameter is passed.
  68     base : int, default 0
  69         Only when `freq` parameter is passed.
  70         For frequencies that evenly subdivide 1 day, the "origin" of the
  71         aggregated intervals. For example, for '5min' frequency, base could
  72         range from 0 through 4. Defaults to 0.
  73 
  74         .. deprecated:: 1.1.0
  75             The new arguments that you should use are 'offset' or 'origin'.
  76 
  77     loffset : str, DateOffset, timedelta object
  78         Only when `freq` parameter is passed.
  79 
  80         .. deprecated:: 1.1.0
  81             loffset is only working for ``.resample(...)`` and not for
  82             Grouper (:issue:`28302`).
  83             However, loffset is also deprecated for ``.resample(...)``
  84             See: :class:`DataFrame.resample`
  85 
  86     origin : {'epoch', 'start', 'start_day'}, Timestamp or str, default 'start_day'
  87         The timestamp on which to adjust the grouping. The timezone of origin must
  88         match the timezone of the index.
  89         If a timestamp is not used, these values are also supported:
  90 
  91         - 'epoch': `origin` is 1970-01-01
  92         - 'start': `origin` is the first value of the timeseries
  93         - 'start_day': `origin` is the first day at midnight of the timeseries
  94 
  95         .. versionadded:: 1.1.0
  96 
  97     offset : Timedelta or str, default is None
  98         An offset timedelta added to the origin.
  99 
 100         .. versionadded:: 1.1.0
 101 
 102     Returns
 103     -------
 104     A specification for a groupby instruction
 105 
 106     Examples
 107     --------
 108     Syntactic sugar for ``df.groupby('A')``
 109 
 110     >>> df = pd.DataFrame(
 111     ...     {
 112     ...         "Animal": ["Falcon", "Parrot", "Falcon", "Falcon", "Parrot"],
 113     ...         "Speed": [100, 5, 200, 300, 15],
 114     ...     }
 115     ... )
 116     >>> df
 117        Animal  Speed
 118     0  Falcon    100
 119     1  Parrot      5
 120     2  Falcon    200
 121     3  Falcon    300
 122     4  Parrot     15
 123     >>> df.groupby(pd.Grouper(key="Animal")).mean()
 124             Speed
 125     Animal
 126     Falcon    200
 127     Parrot     10
 128 
 129     Specify a resample operation on the column 'Publish date'
 130 
 131     >>> df = pd.DataFrame(
 132     ...    {
 133     ...        "Publish date": [
 134     ...             pd.Timestamp("2000-01-02"),
 135     ...             pd.Timestamp("2000-01-02"),
 136     ...             pd.Timestamp("2000-01-09"),
 137     ...             pd.Timestamp("2000-01-16")
 138     ...         ],
 139     ...         "ID": [0, 1, 2, 3],
 140     ...         "Price": [10, 20, 30, 40]
 141     ...     }
 142     ... )
 143     >>> df
 144       Publish date  ID  Price
 145     0   2000-01-02   0     10
 146     1   2000-01-02   1     20
 147     2   2000-01-09   2     30
 148     3   2000-01-16   3     40
 149     >>> df.groupby(pd.Grouper(key="Publish date", freq="1W")).mean()
 150                    ID  Price
 151     Publish date
 152     2000-01-02    0.5   15.0
 153     2000-01-09    2.0   30.0
 154     2000-01-16    3.0   40.0
 155 
 156     If you want to adjust the start of the bins based on a fixed timestamp:
 157 
 158     >>> start, end = '2000-10-01 23:30:00', '2000-10-02 00:30:00'
 159     >>> rng = pd.date_range(start, end, freq='7min')
 160     >>> ts = pd.Series(np.arange(len(rng)) * 3, index=rng)
 161     >>> ts
 162     2000-10-01 23:30:00     0
 163     2000-10-01 23:37:00     3
 164     2000-10-01 23:44:00     6
 165     2000-10-01 23:51:00     9
 166     2000-10-01 23:58:00    12
 167     2000-10-02 00:05:00    15
 168     2000-10-02 00:12:00    18
 169     2000-10-02 00:19:00    21
 170     2000-10-02 00:26:00    24
 171     Freq: 7T, dtype: int64
 172 
 173     >>> ts.groupby(pd.Grouper(freq='17min')).sum()
 174     2000-10-01 23:14:00     0
 175     2000-10-01 23:31:00     9
 176     2000-10-01 23:48:00    21
 177     2000-10-02 00:05:00    54
 178     2000-10-02 00:22:00    24
 179     Freq: 17T, dtype: int64
 180 
 181     >>> ts.groupby(pd.Grouper(freq='17min', origin='epoch')).sum()
 182     2000-10-01 23:18:00     0
 183     2000-10-01 23:35:00    18
 184     2000-10-01 23:52:00    27
 185     2000-10-02 00:09:00    39
 186     2000-10-02 00:26:00    24
 187     Freq: 17T, dtype: int64
 188 
 189     >>> ts.groupby(pd.Grouper(freq='17min', origin='2000-01-01')).sum()
 190     2000-10-01 23:24:00     3
 191     2000-10-01 23:41:00    15
 192     2000-10-01 23:58:00    45
 193     2000-10-02 00:15:00    45
 194     Freq: 17T, dtype: int64
 195 
 196     If you want to adjust the start of the bins with an `offset` Timedelta, the two
 197     following lines are equivalent:
 198 
 199     >>> ts.groupby(pd.Grouper(freq='17min', origin='start')).sum()
 200     2000-10-01 23:30:00     9
 201     2000-10-01 23:47:00    21
 202     2000-10-02 00:04:00    54
 203     2000-10-02 00:21:00    24
 204     Freq: 17T, dtype: int64
 205 
 206     >>> ts.groupby(pd.Grouper(freq='17min', offset='23h30min')).sum()
 207     2000-10-01 23:30:00     9
 208     2000-10-01 23:47:00    21
 209     2000-10-02 00:04:00    54
 210     2000-10-02 00:21:00    24
 211     Freq: 17T, dtype: int64
 212 
 213     To replace the use of the deprecated `base` argument, you can now use `offset`,
 214     in this example it is equivalent to have `base=2`:
 215 
 216     >>> ts.groupby(pd.Grouper(freq='17min', offset='2min')).sum()
 217     2000-10-01 23:16:00     0
 218     2000-10-01 23:33:00     9
 219     2000-10-01 23:50:00    36
 220     2000-10-02 00:07:00    39
 221     2000-10-02 00:24:00    24
 222     Freq: 17T, dtype: int64
 223     """

 227     def __new__(cls, *args, **kwargs):
 228         if kwargs.get("freq") is not None:
 229             from pandas.core.resample import TimeGrouper
 230 
 231             # Deprecation warning of `base` and `loffset` since v1.1.0:
 232             # we are raising the warning here to be able to set the `stacklevel`
 233             # properly since we need to raise the `base` and `loffset` deprecation
 234             # warning from three different cases:
 235             #   core/generic.py::NDFrame.resample
 236             #   core/groupby/groupby.py::GroupBy.resample
 237             #   core/groupby/grouper.py::Grouper
 238             # raising these warnings from TimeGrouper directly would fail the test:
 239             #   tests/resample/test_deprecated.py::test_deprecating_on_loffset_and_base
 240 
 241             # hacky way to set the stacklevel: if cls is TimeGrouper it means
 242             # that the call comes from a pandas internal call of resample,
 243             # otherwise it comes from pd.Grouper
 244             stacklevel = 4 if cls is TimeGrouper else 2
 245             if kwargs.get("base", None) is not None:
 246                 warnings.warn(
 247                     "'base' in .resample() and in Grouper() is deprecated.\n"
 248                     "The new arguments that you should use are 'offset' or 'origin'.\n"
 249                     '\n>>> df.resample(freq="3s", base=2)\n'
 250                     "\nbecomes:\n"
 251                     '\n>>> df.resample(freq="3s", offset="2s")\n',
 252                     FutureWarning,
 253                     stacklevel=stacklevel,
 254                 )
 255 
 256             if kwargs.get("loffset", None) is not None:
 257                 warnings.warn(
 258                     "'loffset' in .resample() and in Grouper() is deprecated.\n"
 259                     '\n>>> df.resample(freq="3s", loffset="8H")\n'
 260                     "\nbecomes:\n"
 261                     "\n>>> from pandas.tseries.frequencies import to_offset"
 262                     '\n>>> df = df.resample(freq="3s").mean()'
 263                     '\n>>> df.index = df.index.to_timestamp() + to_offset("8H")\n',
 264                     FutureWarning,
 265                     stacklevel=stacklevel,
 266                 )

 268             cls = TimeGrouper
 269         return super().__new__(cls)

 291     def _get_grouper(self, obj, validate: bool = True):
 292         """
 293         Parameters
 294         ----------
 295         obj : the subject object
 296         validate : boolean, default True
 297             if True, validate the grouper
 298 
 299         Returns
 300         -------
 301         a tuple of binner, grouper, obj (possibly sorted)
 302         """
 303         self._set_grouper(obj)
 304         self.grouper, _, self.obj = get_grouper(
 305             self.obj,
 306             [self.key],
 307             axis=self.axis,
 308             level=self.level,
 309             sort=self.sort,
 310             validate=validate,
 311             dropna=self.dropna,
 312         )
 313         return self.binner, self.grouper, self.obj

 315     def _set_grouper(self, obj: FrameOrSeries, sort: bool = False):
 316         """
 317         given an object and the specifications, setup the internal grouper
 318         for this particular specification
 319 
 320         Parameters
 321         ----------
 322         obj : Series or DataFrame
 323         sort : bool, default False
 324             whether the resulting grouper should be sorted
 325         """
 326         assert obj is not None
 327 
 328         if self.key is not None and self.level is not None:
 329             raise ValueError("The Grouper cannot specify both a key and a level!")
 330 
 331         # Keep self.grouper value before overriding
 332         if self._grouper is None:
 333             self._grouper = self.grouper
 334 
 335         # the key must be a valid info item
 336         if self.key is not None:
 337             key = self.key
 338             # The 'on' is already defined
 339             if getattr(self.grouper, "name", None) == key and isinstance(
 340                 obj, ABCSeries
 341             ):
 342                 ax = self._grouper.take(obj.index)
 343             else:
 344                 if key not in obj._info_axis:
 345                     raise KeyError(f"The grouper name {key} is not found")
 346                 ax = Index(obj[key], name=key)
 347 
 348         else:
 349             ax = obj._get_axis(self.axis)
 350             if self.level is not None:
 351                 level = self.level
 352 
 353                 # if a level is given it must be a mi level or
 354                 # equivalent to the axis name
 355                 if isinstance(ax, MultiIndex):
 356                     level = ax._get_level_number(level)
 357                     ax = Index(ax._get_level_values(level), name=ax.names[level])
 358 
 359                 else:
 360                     if level not in (0, ax.name):
 361                         raise ValueError(f"The level {level} is not valid")
 362 
 363         # possibly sort
 364         if (self.sort or sort) and not ax.is_monotonic:
 365             # use stable sort to support first, last, nth
 366             indexer = self.indexer = ax.argsort(kind="mergesort")
 367             ax = ax.take(indexer)
 368             obj = obj.take(indexer, axis=self.axis)
 369 
 370         self.obj = obj
 371         self.grouper = ax
 372         return self.grouper

 389 class Grouping:
 390     """
 391     Holds the grouping information for a single key
 392 
 393     Parameters
 394     ----------
 395     index : Index
 396     grouper :
 397     obj Union[DataFrame, Series]:
 398     name : Label
 399     level :
 400     observed : bool, default False
 401         If we are a Categorical, use the observed values
 402     in_axis : if the Grouping is a column in self.obj and hence among
 403         Groupby.exclusions list
 404 
 405     Returns
 406     -------
 407     **Attributes**:
 408       * indices : dict of {group -> index_list}
 409       * codes : ndarray, group codes
 410       * group_index : unique groups
 411       * groups : dict of {group -> label_list}
 412     """
 413 
 414     def __init__(
 415         self,
 416         index: Index,
 417         grouper=None,
 418         obj: Optional[FrameOrSeries] = None,
 419         name=None,
 420         level=None,
 421         sort: bool = True,
 422         observed: bool = False,
 423         in_axis: bool = False,
 424         dropna: bool = True,
 425     ):
 426         self.name = name
 427         self.level = level
 428         self.grouper = _convert_grouper(index, grouper)
 429         self.all_grouper = None
 430         self.index = index
 431         self.sort = sort
 432         self.obj = obj
 433         self.observed = observed
 434         self.in_axis = in_axis
 435         self.dropna = dropna
 436 
 437         # right place for this?
 438         if isinstance(grouper, (Series, Index)) and name is None:
 439             self.name = grouper.name
 440 
 441         if isinstance(grouper, MultiIndex):
 442             self.grouper = grouper._values
 443 
 444         # we have a single grouper which may be a myriad of things,
 445         # some of which are dependent on the passing in level
 446 
 447         if level is not None:
 448             if not isinstance(level, int):
 449                 if level not in index.names:
 450                     raise AssertionError(f"Level {level} not in index")
 451                 level = index.names.index(level)
 452 
 453             if self.name is None:
 454                 self.name = index.names[level]
 455 
 456             (
 457                 self.grouper,
 458                 self._codes,
 459                 self._group_index,
 460             ) = index._get_grouper_for_level(self.grouper, level)
 461 
 462         # a passed Grouper like, directly get the grouper in the same way
 463         # as single grouper groupby, use the group_info to get codes
 464         elif isinstance(self.grouper, Grouper):
 465             # get the new grouper; we already have disambiguated
 466             # what key/level refer to exactly, don't need to
 467             # check again as we have by this point converted these
 468             # to an actual value (rather than a pd.Grouper)
 469             _, grouper, _ = self.grouper._get_grouper(self.obj, validate=False)
 470             if self.name is None:
 471                 self.name = grouper.result_index.name
 472             self.obj = self.grouper.obj
 473             self.grouper = grouper._get_grouper()
 474 
 475         else:
 476             if self.grouper is None and self.name is not None and self.obj is not None:
 477                 self.grouper = self.obj[self.name]
 478 
 479             elif isinstance(self.grouper, (list, tuple)):
 480                 self.grouper = com.asarray_tuplesafe(self.grouper)
 481 
 482             # a passed Categorical
 483             elif is_categorical_dtype(self.grouper):
 484 
 485                 self.grouper, self.all_grouper = recode_for_groupby(
 486                     self.grouper, self.sort, observed
 487                 )
 488                 categories = self.grouper.categories
 489 
 490                 # we make a CategoricalIndex out of the cat grouper
 491                 # preserving the categories / ordered attributes
 492                 self._codes = self.grouper.codes
 493                 if observed:
 494                     codes = algorithms.unique1d(self.grouper.codes)
 495                     codes = codes[codes != -1]
 496                     if sort or self.grouper.ordered:
 497                         codes = np.sort(codes)
 498                 else:
 499                     codes = np.arange(len(categories))
 500 
 501                 self._group_index = CategoricalIndex(
 502                     Categorical.from_codes(
 503                         codes=codes, categories=categories, ordered=self.grouper.ordered
 504                     ),
 505                     name=self.name,
 506                 )
 507 
 508             # we are done
 509             if isinstance(self.grouper, Grouping):
 510                 self.grouper = self.grouper.grouper
 511 
 512             # no level passed
 513             elif not isinstance(
 514                 self.grouper, (Series, Index, ExtensionArray, np.ndarray)
 515             ):
 516                 if getattr(self.grouper, "ndim", 1) != 1:
 517                     t = self.name or str(type(self.grouper))
 518                     raise ValueError(f"Grouper for '{t}' not 1-dimensional")
 519                 self.grouper = self.index.map(self.grouper)
 520                 if not (
 521                     hasattr(self.grouper, "__len__")
 522                     and len(self.grouper) == len(self.index)
 523                 ):
 524                     grper = pprint_thing(self.grouper)
 525                     errmsg = (
 526                         "Grouper result violates len(labels) == "
 527                         f"len(data)\nresult: {grper}"
 528                     )
 529                     self.grouper = None  # Try for sanity
 530                     raise AssertionError(errmsg)
 531 
 532         # if we have a date/time-like grouper, make sure that we have
 533         # Timestamps like
 534         if getattr(self.grouper, "dtype", None) is not None:
 535             if is_datetime64_dtype(self.grouper):
 536                 self.grouper = self.grouper.astype("datetime64[ns]")
 537             elif is_timedelta64_dtype(self.grouper):
 538 
 539                 self.grouper = self.grouper.astype("timedelta64[ns]")
 540 
 541     def __repr__(self) -> str:
 542         return f"Grouping({self.name})"
 543 
 544     def __iter__(self):
 545         return iter(self.indices)

 547     _codes: Optional[np.ndarray] = None
 548     _group_index: Optional[Index] = None

 550     @property
 551     def ngroups(self) -> int:
 552         return len(self.group_index)
 553 
 554     @cache_readonly
 555     def indices(self):
 556         # we have a list of groupers
 557         if isinstance(self.grouper, ops.BaseGrouper):
 558             return self.grouper.indices
 559 
 560         values = Categorical(self.grouper)
 561         return values._reverse_indexer()
 562 
 563     @property
 564     def codes(self) -> np.ndarray:
 565         if self._codes is None:
 566             self._make_codes()
 567         return self._codes
 568 
 569     @cache_readonly
 570     def result_index(self) -> Index:
 571         if self.all_grouper is not None:
 572             return recode_from_groupby(self.all_grouper, self.sort, self.group_index)
 573         return self.group_index
 574 
 575     @property
 576     def group_index(self) -> Index:
 577         if self._group_index is None:
 578             self._make_codes()
 579         assert self._group_index is not None
 580         return self._group_index
 581 
 582     def _make_codes(self) -> None:
 583         if self._codes is None or self._group_index is None:
 584             # we have a list of groupers
 585             if isinstance(self.grouper, ops.BaseGrouper):
 586                 codes = self.grouper.codes_info
 587                 uniques = self.grouper.result_index
 588             else:
 589                 codes, uniques = algorithms.factorize(
 590                     self.grouper, sort=self.sort, dropna=self.dropna
 591                 )
 592                 uniques = Index(uniques, name=self.name)
 593             self._codes = codes
 594             self._group_index = uniques
 595 
 596     @cache_readonly
 597     def groups(self) -> Dict[Hashable, np.ndarray]:
 598         return self.index.groupby(Categorical.from_codes(self.codes, self.group_index))

 601 def get_grouper(
 602     obj: FrameOrSeries,
 603     key=None,
 604     axis: int = 0,
 605     level=None,
 606     sort: bool = True,
 607     observed: bool = False,
 608     mutated: bool = False,
 609     validate: bool = True,
 610     dropna: bool = True,
 611 ) -> "Tuple[ops.BaseGrouper, List[Hashable], FrameOrSeries]":
 612     """
 613     Create and return a BaseGrouper, which is an internal
 614     mapping of how to create the grouper indexers.
 615     This may be composed of multiple Grouping objects, indicating
 616     multiple groupers
 617 
 618     Groupers are ultimately index mappings. They can originate as:
 619     index mappings, keys to columns, functions, or Groupers
 620 
 621     Groupers enable local references to axis,level,sort, while
 622     the passed in axis, level, and sort are 'global'.
 623 
 624     This routine tries to figure out what the passing in references
 625     are and then creates a Grouping for each one, combined into
 626     a BaseGrouper.
 627 
 628     If observed & we have a categorical grouper, only show the observed
 629     values.
 630 
 631     If validate, then check for key/level overlaps.
 632 
 633     """
 634     group_axis = obj._get_axis(axis)
 635 
 636     # validate that the passed single level is compatible with the passed
 637     # axis of the object
 638     if level is not None:
 639         # TODO: These if-block and else-block are almost same.
 640         # MultiIndex instance check is removable, but it seems that there are
 641         # some processes only for non-MultiIndex in else-block,
 642         # eg. `obj.index.name != level`. We have to consider carefully whether
 643         # these are applicable for MultiIndex. Even if these are applicable,
 644         # we need to check if it makes no side effect to subsequent processes
 645         # on the outside of this condition.
 646         # (GH 17621)
 647         if isinstance(group_axis, MultiIndex):
 648             if is_list_like(level) and len(level) == 1:
 649                 level = level[0]
 650 
 651             if key is None and is_scalar(level):
 652                 # Get the level values from group_axis
 653                 key = group_axis.get_level_values(level)
 654                 level = None
 655 
 656         else:
 657             # allow level to be a length-one list-like object
 658             # (e.g., level=[0])
 659             # GH 13901
 660             if is_list_like(level):
 661                 nlevels = len(level)
 662                 if nlevels == 1:
 663                     level = level[0]
 664                 elif nlevels == 0:
 665                     raise ValueError("No group keys passed!")
 666                 else:
 667                     raise ValueError("multiple levels only valid with MultiIndex")
 668 
 669             if isinstance(level, str):
 670                 if obj._get_axis(axis).name != level:
 671                     raise ValueError(
 672                         f"level name {level} is not the name "
 673                         f"of the {obj._get_axis_name(axis)}"
 674                     )
 675             elif level > 0 or level < -1:
 676                 raise ValueError("level > 0 or level < -1 only valid with MultiIndex")
 677 
 678             # NOTE: `group_axis` and `group_axis.get_level_values(level)`
 679             # are same in this section.
 680             level = None
 681             key = group_axis
 682 
 683     # a passed-in Grouper, directly convert
 684     if isinstance(key, Grouper):
 685         binner, grouper, obj = key._get_grouper(obj, validate=False)
 686         if key.key is None:
 687             return grouper, [], obj
 688         else:
 689             return grouper, [key.key], obj
 690 
 691     # already have a BaseGrouper, just return it
 692     elif isinstance(key, ops.BaseGrouper):
 693         return key, [], obj
 694 
 695     if not isinstance(key, list):
 696         keys = [key]
 697         match_axis_length = False
 698     else:
 699         keys = key
 700         match_axis_length = len(keys) == len(group_axis)
 701 
 702     # what are we after, exactly?
 703     any_callable = any(callable(g) or isinstance(g, dict) for g in keys)
 704     any_groupers = any(isinstance(g, Grouper) for g in keys)
 705     any_arraylike = any(
 706         isinstance(g, (list, tuple, Series, Index, np.ndarray)) for g in keys
 707     )
 708 
 709     # is this an index replacement?
 710     if (
 711         not any_callable
 712         and not any_arraylike
 713         and not any_groupers
 714         and match_axis_length
 715         and level is None
 716     ):
 717         if isinstance(obj, DataFrame):
 718             all_in_columns_index = all(
 719                 g in obj.columns or g in obj.index.names for g in keys
 720             )
 721         else:
 722             assert isinstance(obj, Series)
 723             all_in_columns_index = all(g in obj.index.names for g in keys)
 724 
 725         if not all_in_columns_index:
 726             keys = [com.asarray_tuplesafe(keys)]
 727 
 728     if isinstance(level, (tuple, list)):
 729         if key is None:
 730             keys = [None] * len(level)
 731         levels = level
 732     else:
 733         levels = [level] * len(keys)
 734 
 735     groupings: List[Grouping] = []
 736     exclusions: List[Hashable] = []
 737 
 738     # if the actual grouper should be obj[key]
 739     def is_in_axis(key) -> bool:
 740         if not _is_label_like(key):
 741             # items -> .columns for DataFrame, .index for Series
 742             items = obj.axes[-1]
 743             try:
 744                 items.get_loc(key)
 745             except (KeyError, TypeError, InvalidIndexError):
 746                 # TypeError shows up here if we pass e.g. Int64Index
 747                 return False
 748 
 749         return True
 750 
 751     # if the grouper is obj[name]
 752     def is_in_obj(gpr) -> bool:
 753         if not hasattr(gpr, "name"):
 754             return False
 755         try:
 756             return gpr is obj[gpr.name]
 757         except (KeyError, IndexError, ValueError):
 758             # TODO: ValueError: Given date string not likely a datetime.
 759             # should be KeyError?
 760             return False
 761 
 762     for i, (gpr, level) in enumerate(zip(keys, levels)):
 763 
 764         if is_in_obj(gpr):  # df.groupby(df['name'])
 765             in_axis, name = True, gpr.name
 766             exclusions.append(name)
 767 
 768         elif is_in_axis(gpr):  # df.groupby('name')
 769             if gpr in obj:
 770                 if validate:
 771                     obj._check_label_or_level_ambiguity(gpr, axis=axis)
 772                 in_axis, name, gpr = True, gpr, obj[gpr]
 773                 exclusions.append(name)
 774             elif obj._is_level_reference(gpr, axis=axis):
 775                 in_axis, name, level, gpr = False, None, gpr, None
 776             else:
 777                 raise KeyError(gpr)
 778         elif isinstance(gpr, Grouper) and gpr.key is not None:
 779             # Add key to exclusions
 780             exclusions.append(gpr.key)
 781             in_axis, name = False, None
 782         else:
 783             in_axis, name = False, None
 784 
 785         if is_categorical_dtype(gpr) and len(gpr) != obj.shape[axis]:
 786             raise ValueError(
 787                 f"Length of grouper ({len(gpr)}) and axis ({obj.shape[axis]}) "
 788                 "must be same length"
 789             )
 790 
 791         # create the Grouping
 792         # allow us to passing the actual Grouping as the gpr
 793         ping = (
 794             Grouping(
 795                 group_axis,
 796                 gpr,
 797                 obj=obj,
 798                 name=name,
 799                 level=level,
 800                 sort=sort,
 801                 observed=observed,
 802                 in_axis=in_axis,
 803                 dropna=dropna,
 804             )
 805             if not isinstance(gpr, Grouping)
 806             else gpr
 807         )
 808 
 809         groupings.append(ping)
 810 
 811     if len(groupings) == 0 and len(obj):
 812         raise ValueError("No group keys passed!")
 813     elif len(groupings) == 0:
 814         groupings.append(Grouping(Index([], dtype="int"), np.array([], dtype=np.intp)))
 815 
 816     # create the internals grouper
 817     grouper = ops.BaseGrouper(group_axis, groupings, sort=sort, mutated=mutated)
 818     return grouper, exclusions, obj

 821 def _is_label_like(val) -> bool:
 822     return isinstance(val, (str, tuple)) or (val is not None and is_scalar(val))

 825 def _convert_grouper(axis: Index, grouper):
 826     if isinstance(grouper, dict):
 827         return grouper.get
 828     elif isinstance(grouper, Series):
 829         if grouper.index.equals(axis):
 830             return grouper._values
 831         else:
 832             return grouper.reindex(axis)._values
 833     elif isinstance(grouper, (list, Series, Index, np.ndarray)):
 834         if len(grouper) != len(axis):
 835             raise ValueError("Grouper and axis must be same length")
 836         return grouper
 837     else:
 838         return grouper
