 573     def _get_indices(self, names):
 574         """
 575         Safe get multiple indices, translate keys for
 576         datelike to underlying repr.
 577         """
 578 
 579         def get_converter(s):
 580             # possibly convert to the actual key types
 581             # in the indices, could be a Timestamp or a np.datetime64
 582             if isinstance(s, datetime.datetime):
 583                 return lambda key: Timestamp(key)
 584             elif isinstance(s, np.datetime64):
 585                 return lambda key: Timestamp(key).asm8
 586             else:
 587                 return lambda key: key
 588 
 589         if len(names) == 0:
 590             return []
 591 
 592         if len(self.indices) > 0:
 593             index_sample = next(iter(self.indices))
 594         else:
 595             index_sample = None  # Dummy sample
 596 
 597         name_sample = names[0]
 598         if isinstance(index_sample, tuple):
 599             if not isinstance(name_sample, tuple):
 600                 msg = "must supply a tuple to get_group with multiple grouping keys"
 601                 raise ValueError(msg)
 602             if not len(name_sample) == len(index_sample):
 603                 try:
 604                     # If the original grouper was a tuple
 605                     return [self.indices[name] for name in names]
 606                 except KeyError as err:
 607                     # turns out it wasn't a tuple
 608                     msg = (
 609                         "must supply a same-length tuple to get_group "
 610                         "with multiple grouping keys"
 611                     )
 612                     raise ValueError(msg) from err
 613 
 614             converters = [get_converter(s) for s in index_sample]
 615             names = (tuple(f(n) for f, n in zip(converters, name)) for name in names)
 616 
 617         else:
 618             converter = get_converter(index_sample)
 619             names = (converter(name) for name in names)
 620 
 621         return [self.indices.get(name, []) for name in names]

 901     def _cumcount_array(self, ascending: bool = True):
 902         """
 903         Parameters
 904         ----------
 905         ascending : bool, default True
 906             If False, number in reverse, from length of group - 1 to 0.
 907 
 908         Notes
 909         -----
 910         this is currently implementing sort=False
 911         (though the default is sort=True) for groupby in general
 912         """
 913         ids, _, ngroups = self.grouper.group_info
 914         sorter = get_group_index_sorter(ids, ngroups)
 915         ids, count = ids[sorter], len(ids)
 916 
 917         if count == 0:
 918             return np.empty(0, dtype=np.int64)
 919 
 920         run = np.r_[True, ids[:-1] != ids[1:]]
 921         rep = np.diff(np.r_[np.nonzero(run)[0], count])
 922         out = (~run).cumsum()
 923 
 924         if ascending:
 925             out -= np.repeat(out[run], rep)
 926         else:
 927             out = np.repeat(out[np.r_[run[1:], True]], rep) - out
 928 
 929         rev = np.empty(count, dtype=np.intp)
 930         rev[sorter] = np.arange(count, dtype=np.intp)
 931         return out[rev].astype(np.int64, copy=False)

2618         we have a fully re-indexed output to the levels. This means expanding
2619         the output space to accommodate all values in the cartesian product of
2620         our groups, regardless of whether they were observed in the data or
2621         not. This will expand the output space if there are missing groups.

2638         """
2639         groupings = self.grouper.groupings
2640         if groupings is None:
2641             return output
2642         elif len(groupings) == 1:
2643             return output
2644 
2645         # if we only care about the observed values
2646         # we are done
2647         elif self.observed:
2648             return output
2649 
2650         # reindexing only applies to a Categorical grouper
2651         elif not any(
2652             isinstance(ping.grouper, (Categorical, CategoricalIndex))
2653             for ping in groupings
2654         ):
2655             return output
2656 
2657         levels_list = [ping.group_index for ping in groupings]
2658         index, _ = MultiIndex.from_product(
2659             levels_list, names=self.grouper.names

2664                 self.obj._get_axis_name(self.axis): index,
