  58 def cat_core(list_of_columns: List, sep: str):
  59     """
  60     Auxiliary function for :meth:`str.cat`
  61 
  62     Parameters
  63     ----------
  64     list_of_columns : list of numpy arrays
  65         List of arrays to be concatenated with sep;
  66         these arrays may not contain NaNs!
  67     sep : string
  68         The separator string for concatenating the columns.
  69 
  70     Returns
  71     -------
  72     nd.array
  73         The concatenation of list_of_columns with sep.
  74     """
  75     if sep == "":
  76         # no need to interleave sep if it is empty
  77         arr_of_cols = np.asarray(list_of_columns, dtype=object)
  78         return np.sum(arr_of_cols, axis=0)
  79     list_with_sep = [sep] * (2 * len(list_of_columns) - 1)
  80     list_with_sep[::2] = list_of_columns
  81     arr_with_sep = np.asarray(list_with_sep, dtype=object)
  82     return np.sum(arr_with_sep, axis=0)

  85 def cat_safe(list_of_columns: List, sep: str):
  86     """
  87     Auxiliary function for :meth:`str.cat`.
  88 
  89     Same signature as cat_core, but handles TypeErrors in concatenation, which
  90     happen if the arrays in list_of columns have the wrong dtypes or content.
  91 
  92     Parameters
  93     ----------
  94     list_of_columns : list of numpy arrays
  95         List of arrays to be concatenated with sep;
  96         these arrays may not contain NaNs!
  97     sep : string
  98         The separator string for concatenating the columns.
  99 
 100     Returns
 101     -------
 102     nd.array
 103         The concatenation of list_of_columns with sep.
 104     """
 105     try:
 106         result = cat_core(list_of_columns, sep)
 107     except TypeError:
 108         # if there are any non-string values (wrong dtype or hidden behind
 109         # object dtype), np.sum will fail; catch and return with better message
 110         for column in list_of_columns:
 111             dtype = lib.infer_dtype(column, skipna=True)
 112             if dtype not in ["string", "empty"]:
 113                 raise TypeError(
 114                     "Concatenation requires list-likes containing only "
 115                     "strings (or missing values). Offending values found in "
 116                     f"column {dtype}"
 117                 ) from None
 118     return result

 133 def _map_stringarray(
 134     func: Callable[[str], Any], arr: "StringArray", na_value: Any, dtype: Dtype
 135 ) -> ArrayLike:
 136     """
 137     Map a callable over valid elements of a StringArray.
 138 
 139     Parameters
 140     ----------
 141     func : Callable[[str], Any]
 142         Apply to each valid element.
 143     arr : StringArray
 144     na_value : Any
 145         The value to use for missing values. By default, this is
 146         the original value (NA).
 147     dtype : Dtype
 148         The result dtype to use. Specifying this avoids an intermediate
 149         object-dtype allocation.
 150 
 151     Returns
 152     -------
 153     ArrayLike
 154         An ExtensionArray for integer or string dtypes, otherwise
 155         an ndarray.
 156 
 157     """
 158     from pandas.arrays import IntegerArray, StringArray, BooleanArray
 159 
 160     mask = isna(arr)
 161 
 162     assert isinstance(arr, StringArray)
 163     arr = np.asarray(arr)
 164 
 165     if is_integer_dtype(dtype) or is_bool_dtype(dtype):
 166         constructor: Union[Type[IntegerArray], Type[BooleanArray]]
 167         if is_integer_dtype(dtype):
 168             constructor = IntegerArray
 169         else:
 170             constructor = BooleanArray
 171 
 172         na_value_is_na = isna(na_value)
 173         if na_value_is_na:
 174             na_value = 1
 175         result = lib.map_infer_mask(
 176             arr,
 177             func,
 178             mask.view("uint8"),
 179             convert=False,
 180             na_value=na_value,
 181             dtype=np.dtype(dtype),
 182         )
 183 
 184         if not na_value_is_na:
 185             mask[:] = False
 186 
 187         return constructor(result, mask)
 188 
 189     elif is_string_dtype(dtype) and not is_object_dtype(dtype):
 190         # i.e. StringDtype
 191         result = lib.map_infer_mask(
 192             arr, func, mask.view("uint8"), convert=False, na_value=na_value
 193         )
 194         return StringArray(result)
 195     else:
 196         # This is when the result type is object. We reach this when
 197         # -> We know the result type is truly object (e.g. .encode returns bytes
 198         #    or .findall returns a list).
 199         # -> We don't know the result type. E.g. `.get` can return anything.
 200         return lib.map_infer_mask(arr, func, mask.view("uint8"))

 203 def _map_object(f, arr, na_mask=False, na_value=np.nan, dtype=np.dtype(object)):
 204     if not len(arr):
 205         return np.ndarray(0, dtype=dtype)
 206 
 207     if isinstance(arr, ABCSeries):
 208         arr = arr._values  # TODO: extract_array?
 209     if not isinstance(arr, np.ndarray):
 210         arr = np.asarray(arr, dtype=object)
 211     if na_mask:
 212         mask = isna(arr)
 213         convert = not np.all(mask)
 214         try:
 215             result = lib.map_infer_mask(arr, f, mask.view(np.uint8), convert)
 216         except (TypeError, AttributeError) as e:
 217             # Reraise the exception if callable `f` got wrong number of args.
 218             # The user may want to be warned by this, instead of getting NaN
 219             p_err = (
 220                 r"((takes)|(missing)) (?(2)from \d+ to )?\d+ "
 221                 r"(?(3)required )positional arguments?"
 222             )
 223 
 224             if len(e.args) >= 1 and re.search(p_err, e.args[0]):
 225                 # FIXME: this should be totally avoidable
 226                 raise e
 227 
 228             def g(x):
 229                 try:
 230                     return f(x)
 231                 except (TypeError, AttributeError):
 232                     return na_value
 233 
 234             return _map_object(g, arr, dtype=dtype)
 235         if na_value is not np.nan:
 236             np.putmask(result, mask, na_value)
 237             if result.dtype == object:
 238                 result = lib.maybe_convert_objects(result)
 239         return result
 240     else:
 241         return lib.map_infer(arr, f)

 244 def str_count(arr, pat, flags=0):
 245     """
 246     Count occurrences of pattern in each string of the Series/Index.
 247 
 248     This function is used to count the number of times a particular regex
 249     pattern is repeated in each of the string elements of the
 250     :class:`~pandas.Series`.
 251 
 252     Parameters
 253     ----------
 254     pat : str
 255         Valid regular expression.
 256     flags : int, default 0, meaning no flags
 257         Flags for the `re` module. For a complete list, `see here
 258         <https://docs.python.org/3/howto/regex.html#compilation-flags>`_.
 259     **kwargs
 260         For compatibility with other string methods. Not used.
 261 
 262     Returns
 263     -------
 264     Series or Index
 265         Same type as the calling object containing the integer counts.
 266 
 267     See Also
 268     --------
 269     re : Standard library module for regular expressions.
 270     str.count : Standard library version, without regular expression support.
 271 
 272     Notes
 273     -----
 274     Some characters need to be escaped when passing in `pat`.
 275     eg. ``'$'`` has a special meaning in regex and must be escaped when
 276     finding this literal character.
 277 
 278     Examples
 279     --------
 280     >>> s = pd.Series(['A', 'B', 'Aaba', 'Baca', np.nan, 'CABA', 'cat'])
 281     >>> s.str.count('a')
 282     0    0.0
 283     1    0.0
 284     2    2.0
 285     3    2.0
 286     4    NaN
 287     5    0.0
 288     6    1.0
 289     dtype: float64
 290 
 291     Escape ``'$'`` to find the literal dollar sign.
 292 
 293     >>> s = pd.Series(['$', 'B', 'Aab$', '$$ca', 'C$B$', 'cat'])
 294     >>> s.str.count('\\$')
 295     0    1
 296     1    0
 297     2    1
 298     3    2
 299     4    2
 300     5    0
 301     dtype: int64
 302 
 303     This is also available on Index
 304 
 305     >>> pd.Index(['A', 'A', 'Aaba', 'cat']).str.count('a')
 306     Int64Index([0, 0, 2, 1], dtype='int64')
 307     """
 308     regex = re.compile(pat, flags=flags)
 309     f = lambda x: len(regex.findall(x))
 310     return _na_map(f, arr, dtype="int64")

 313 def str_contains(arr, pat, case=True, flags=0, na=np.nan, regex=True):
 314     """
 315     Test if pattern or regex is contained within a string of a Series or Index.
 316 
 317     Return boolean Series or Index based on whether a given pattern or regex is
 318     contained within a string of a Series or Index.
 319 
 320     Parameters
 321     ----------
 322     pat : str
 323         Character sequence or regular expression.
 324     case : bool, default True
 325         If True, case sensitive.
 326     flags : int, default 0 (no flags)
 327         Flags to pass through to the re module, e.g. re.IGNORECASE.
 328     na : default NaN
 329         Fill value for missing values.
 330     regex : bool, default True
 331         If True, assumes the pat is a regular expression.
 332 
 333         If False, treats the pat as a literal string.
 334 
 335     Returns
 336     -------
 337     Series or Index of boolean values
 338         A Series or Index of boolean values indicating whether the
 339         given pattern is contained within the string of each element
 340         of the Series or Index.
 341 
 342     See Also
 343     --------
 344     match : Analogous, but stricter, relying on re.match instead of re.search.
 345     Series.str.startswith : Test if the start of each string element matches a
 346         pattern.
 347     Series.str.endswith : Same as startswith, but tests the end of string.
 348 
 349     Examples
 350     --------
 351     Returning a Series of booleans using only a literal pattern.
 352 
 353     >>> s1 = pd.Series(['Mouse', 'dog', 'house and parrot', '23', np.NaN])
 354     >>> s1.str.contains('og', regex=False)
 355     0    False
 356     1     True
 357     2    False
 358     3    False
 359     4      NaN
 360     dtype: object
 361 
 362     Returning an Index of booleans using only a literal pattern.
 363 
 364     >>> ind = pd.Index(['Mouse', 'dog', 'house and parrot', '23.0', np.NaN])
 365     >>> ind.str.contains('23', regex=False)
 366     Index([False, False, False, True, nan], dtype='object')
 367 
 368     Specifying case sensitivity using `case`.
 369 
 370     >>> s1.str.contains('oG', case=True, regex=True)
 371     0    False
 372     1    False
 373     2    False
 374     3    False
 375     4      NaN
 376     dtype: object
 377 
 378     Specifying `na` to be `False` instead of `NaN` replaces NaN values
 379     with `False`. If Series or Index does not contain NaN values
 380     the resultant dtype will be `bool`, otherwise, an `object` dtype.
 381 
 382     >>> s1.str.contains('og', na=False, regex=True)
 383     0    False
 384     1     True
 385     2    False
 386     3    False
 387     4    False
 388     dtype: bool
 389 
 390     Returning 'house' or 'dog' when either expression occurs in a string.
 391 
 392     >>> s1.str.contains('house|dog', regex=True)
 393     0    False
 394     1     True
 395     2     True
 396     3    False
 397     4      NaN
 398     dtype: object
 399 
 400     Ignoring case sensitivity using `flags` with regex.
 401 
 402     >>> import re
 403     >>> s1.str.contains('PARROT', flags=re.IGNORECASE, regex=True)
 404     0    False
 405     1    False
 406     2     True
 407     3    False
 408     4      NaN
 409     dtype: object
 410 
 411     Returning any digit using regular expression.
 412 
 413     >>> s1.str.contains('\\d', regex=True)
 414     0    False
 415     1    False
 416     2    False
 417     3     True
 418     4      NaN
 419     dtype: object
 420 
 421     Ensure `pat` is a not a literal pattern when `regex` is set to True.
 422     Note in the following example one might expect only `s2[1]` and `s2[3]` to
 423     return `True`. However, '.0' as a regex matches any character
 424     followed by a 0.
 425 
 426     >>> s2 = pd.Series(['40', '40.0', '41', '41.0', '35'])
 427     >>> s2.str.contains('.0', regex=True)
 428     0     True
 429     1     True
 430     2    False
 431     3     True
 432     4    False
 433     dtype: bool
 434     """
 435     if regex:
 436         if not case:
 437             flags |= re.IGNORECASE
 438 
 439         regex = re.compile(pat, flags=flags)
 440 
 441         if regex.groups > 0:
 442             warnings.warn(
 443                 "This pattern has match groups. To actually get the "
 444                 "groups, use str.extract.",
 445                 UserWarning,
 446                 stacklevel=3,
 447             )
 448 
 449         f = lambda x: regex.search(x) is not None
 450     else:
 451         if case:
 452             f = lambda x: pat in x
 453         else:
 454             upper_pat = pat.upper()
 455             f = lambda x: upper_pat in x
 456             uppered = _na_map(lambda x: x.upper(), arr)
 457             return _na_map(f, uppered, na, dtype=np.dtype(bool))
 458     return _na_map(f, arr, na, dtype=np.dtype(bool))

 571 def str_replace(arr, pat, repl, n=-1, case=None, flags=0, regex=True):
 572     r"""
 573     Replace each occurrence of pattern/regex in the Series/Index.
 574 
 575     Equivalent to :meth:`str.replace` or :func:`re.sub`, depending on the regex value.
 576 
 577     Parameters
 578     ----------
 579     pat : str or compiled regex
 580         String can be a character sequence or regular expression.
 581     repl : str or callable
 582         Replacement string or a callable. The callable is passed the regex
 583         match object and must return a replacement string to be used.
 584         See :func:`re.sub`.
 585     n : int, default -1 (all)
 586         Number of replacements to make from start.
 587     case : bool, default None
 588         Determines if replace is case sensitive:
 589 
 590         - If True, case sensitive (the default if `pat` is a string)
 591         - Set to False for case insensitive
 592         - Cannot be set if `pat` is a compiled regex.
 593 
 594     flags : int, default 0 (no flags)
 595         Regex module flags, e.g. re.IGNORECASE. Cannot be set if `pat` is a compiled
 596         regex.
 597     regex : bool, default True
 598         Determines if assumes the passed-in pattern is a regular expression:
 599 
 600         - If True, assumes the passed-in pattern is a regular expression.
 601         - If False, treats the pattern as a literal string
 602         - Cannot be set to False if `pat` is a compiled regex or `repl` is
 603           a callable.
 604 
 605         .. versionadded:: 0.23.0
 606 
 607     Returns
 608     -------
 609     Series or Index of object
 610         A copy of the object with all matching occurrences of `pat` replaced by
 611         `repl`.
 612 
 613     Raises
 614     ------
 615     ValueError
 616         * if `regex` is False and `repl` is a callable or `pat` is a compiled
 617           regex
 618         * if `pat` is a compiled regex and `case` or `flags` is set
 619 
 620     Notes
 621     -----
 622     When `pat` is a compiled regex, all flags should be included in the
 623     compiled regex. Use of `case`, `flags`, or `regex=False` with a compiled
 624     regex will raise an error.
 625 
 626     Examples
 627     --------
 628     When `pat` is a string and `regex` is True (the default), the given `pat`
 629     is compiled as a regex. When `repl` is a string, it replaces matching
 630     regex patterns as with :meth:`re.sub`. NaN value(s) in the Series are
 631     left as is:
 632 
 633     >>> pd.Series(['foo', 'fuz', np.nan]).str.replace('f.', 'ba', regex=True)
 634     0    bao
 635     1    baz
 636     2    NaN
 637     dtype: object
 638 
 639     When `pat` is a string and `regex` is False, every `pat` is replaced with
 640     `repl` as with :meth:`str.replace`:
 641 
 642     >>> pd.Series(['f.o', 'fuz', np.nan]).str.replace('f.', 'ba', regex=False)
 643     0    bao
 644     1    fuz
 645     2    NaN
 646     dtype: object
 647 
 648     When `repl` is a callable, it is called on every `pat` using
 649     :func:`re.sub`. The callable should expect one positional argument
 650     (a regex object) and return a string.
 651 
 652     To get the idea:
 653 
 654     >>> pd.Series(['foo', 'fuz', np.nan]).str.replace('f', repr)
 655     0    <re.Match object; span=(0, 1), match='f'>oo
 656     1    <re.Match object; span=(0, 1), match='f'>uz
 657     2                                            NaN
 658     dtype: object
 659 
 660     Reverse every lowercase alphabetic word:
 661 
 662     >>> repl = lambda m: m.group(0)[::-1]
 663     >>> pd.Series(['foo 123', 'bar baz', np.nan]).str.replace(r'[a-z]+', repl)
 664     0    oof 123
 665     1    rab zab
 666     2        NaN
 667     dtype: object
 668 
 669     Using regex groups (extract second group and swap case):
 670 
 671     >>> pat = r"(?P<one>\w+) (?P<two>\w+) (?P<three>\w+)"
 672     >>> repl = lambda m: m.group('two').swapcase()
 673     >>> pd.Series(['One Two Three', 'Foo Bar Baz']).str.replace(pat, repl)
 674     0    tWO
 675     1    bAR
 676     dtype: object
 677 
 678     Using a compiled regex with flags
 679

 689     if not (isinstance(repl, str) or callable(repl)):
 690         raise TypeError("repl must be a string or callable")

 694         if is_compiled_re:
 695             if (case is not None) or (flags != 0):

 697                     "case and flags cannot be set when pat is a compiled regex"

 699         else:
 700             # not a compiled regex

 702             if case is None:

 974 def str_extract(arr, pat, flags=0, expand=True):
 975     r"""
 976     Extract capture groups in the regex `pat` as columns in a DataFrame.
 977 
 978     For each subject string in the Series, extract groups from the
 979     first match of regular expression `pat`.
 980 
 981     Parameters
 982     ----------
 983     pat : str
 984         Regular expression pattern with capturing groups.
 985     flags : int, default 0 (no flags)
 986         Flags from the ``re`` module, e.g. ``re.IGNORECASE``, that
 987         modify regular expression matching for things like case,
 988         spaces, etc. For more details, see :mod:`re`.
 989     expand : bool, default True
 990         If True, return DataFrame with one column per capture group.
 991         If False, return a Series/Index if there is one capture group
 992         or DataFrame if there are multiple capture groups.
 993 
 994     Returns
 995     -------
 996     DataFrame or Series or Index
 997         A DataFrame with one row for each subject string, and one
 998         column for each group. Any capture group names in regular
 999         expression pat will be used for column names; otherwise
1000         capture group numbers will be used. The dtype of each result
1001         column is always object, even when no match is found. If
1002         ``expand=False`` and pat has only one capture group, then
1003         return a Series (if subject is a Series) or Index (if subject
1004         is an Index).
1005 
1006     See Also
1007     --------
1008     extractall : Returns all matches (not just the first match).
1009 
1010     Examples
1011     --------
1012     A pattern with two groups will return a DataFrame with two columns.
1013     Non-matches will be NaN.
1014 
1015     >>> s = pd.Series(['a1', 'b2', 'c3'])
1016     >>> s.str.extract(r'([ab])(\d)')
1017          0    1
1018     0    a    1
1019     1    b    2
1020     2  NaN  NaN
1021 
1022     A pattern may contain optional groups.
1023 
1024     >>> s.str.extract(r'([ab])?(\d)')
1025          0  1
1026     0    a  1
1027     1    b  2
1028     2  NaN  3
1029 
1030     Named groups will become column names in the result.
1031 
1032     >>> s.str.extract(r'(?P<letter>[ab])(?P<digit>\d)')
1033       letter digit
1034     0      a     1
1035     1      b     2
1036     2    NaN   NaN
1037 
1038     A pattern with one group will return a DataFrame with one column
1039     if expand=True.
1040 
1041     >>> s.str.extract(r'[ab](\d)', expand=True)
1042          0
1043     0    1
1044     1    2
1045     2  NaN
1046 
1047     A pattern with one group will return a Series if expand=False.
1048 
1049     >>> s.str.extract(r'[ab](\d)', expand=False)
1050     0      1
1051     1      2
1052     2    NaN
1053     dtype: object
1054     """
1055     if not isinstance(expand, bool):
1056         raise ValueError("expand must be True or False")
1057     if expand:
1058         return _str_extract_frame(arr._orig, pat, flags=flags)
1059     else:
1060         result, name = _str_extract_noexpand(arr._parent, pat, flags=flags)
1061         return arr._wrap_result(result, name=name, expand=expand)

1064 def str_extractall(arr, pat, flags=0):
1065     r"""
1066     Extract capture groups in the regex `pat` as columns in DataFrame.
1067 
1068     For each subject string in the Series, extract groups from all
1069     matches of regular expression pat. When each subject string in the
1070     Series has exactly one match, extractall(pat).xs(0, level='match')
1071     is the same as extract(pat).
1072 
1073     Parameters
1074     ----------
1075     pat : str
1076         Regular expression pattern with capturing groups.
1077     flags : int, default 0 (no flags)
1078         A ``re`` module flag, for example ``re.IGNORECASE``. These allow
1079         to modify regular expression matching for things like case, spaces,
1080         etc. Multiple flags can be combined with the bitwise OR operator,
1081         for example ``re.IGNORECASE | re.MULTILINE``.
1082 
1083     Returns
1084     -------
1085     DataFrame
1086         A ``DataFrame`` with one row for each match, and one column for each
1087         group. Its rows have a ``MultiIndex`` with first levels that come from
1088         the subject ``Series``. The last level is named 'match' and indexes the
1089         matches in each item of the ``Series``. Any capture group names in
1090         regular expression pat will be used for column names; otherwise capture
1091         group numbers will be used.
1092 
1093     See Also
1094     --------
1095     extract : Returns first match only (not all matches).
1096 
1097     Examples
1098     --------
1099     A pattern with one group will return a DataFrame with one column.
1100     Indices with no matches will not appear in the result.
1101 
1102     >>> s = pd.Series(["a1a2", "b1", "c1"], index=["A", "B", "C"])
1103     >>> s.str.extractall(r"[ab](\d)")
1104              0
1105       match
1106     A 0      1
1107       1      2
1108     B 0      1
1109 
1110     Capture group names are used for column names of the result.
1111 
1112     >>> s.str.extractall(r"[ab](?P<digit>\d)")
1113             digit
1114       match
1115     A 0         1
1116       1         2
1117     B 0         1
1118 
1119     A pattern with two groups will return a DataFrame with two columns.
1120 
1121     >>> s.str.extractall(r"(?P<letter>[ab])(?P<digit>\d)")
1122             letter digit
1123       match
1124     A 0          a     1
1125       1          a     2
1126     B 0          b     1
1127 
1128     Optional groups that do not match are NaN in the result.
1129 
1130     >>> s.str.extractall(r"(?P<letter>[ab])?(?P<digit>\d)")
1131             letter digit
1132       match
1133     A 0          a     1
1134       1          a     2
1135     B 0          b     1
1136     C 0        NaN     1
1137     """
1138     regex = re.compile(pat, flags=flags)
1139     # the regex must contain capture groups.
1140     if regex.groups == 0:
1141         raise ValueError("pattern contains no capture groups")
1142 
1143     if isinstance(arr, ABCIndexClass):
1144         arr = arr.to_series().reset_index(drop=True)
1145 
1146     names = dict(zip(regex.groupindex.values(), regex.groupindex.keys()))
1147     columns = [names.get(1 + i, i) for i in range(regex.groups)]
1148     match_list = []
1149     index_list = []
1150     is_mi = arr.index.nlevels > 1
1151 
1152     for subject_key, subject in arr.items():
1153         if isinstance(subject, str):
1154 
1155             if not is_mi:
1156                 subject_key = (subject_key,)
1157 
1158             for match_i, match_tuple in enumerate(regex.findall(subject)):
1159                 if isinstance(match_tuple, str):
1160                     match_tuple = (match_tuple,)
1161                 na_tuple = [np.NaN if group == "" else group for group in match_tuple]
1162                 match_list.append(na_tuple)
1163                 result_key = tuple(subject_key + (match_i,))
1164                 index_list.append(result_key)
1165 
1166     from pandas import MultiIndex
1167 
1168     index = MultiIndex.from_tuples(index_list, names=arr.index.names + ["match"])
1169     dtype = _result_dtype(arr)
1170 
1171     result = arr._constructor_expanddim(
1172         match_list, index=index, columns=columns, dtype=dtype
1173     )
1174     return result

1942 def forbid_nonstring_types(forbidden, name=None):
1943     """
1944     Decorator to forbid specific types for a method of StringMethods.
1945 
1946     For calling `.str.{method}` on a Series or Index, it is necessary to first
1947     initialize the :class:`StringMethods` object, and then call the method.
1948     However, different methods allow different input types, and so this can not
1949     be checked during :meth:`StringMethods.__init__`, but must be done on a
1950     per-method basis. This decorator exists to facilitate this process, and
1951     make it explicit which (inferred) types are disallowed by the method.
1952 
1953     :meth:`StringMethods.__init__` allows the *union* of types its different
1954     methods allow (after skipping NaNs; see :meth:`StringMethods._validate`),
1955     namely: ['string', 'empty', 'bytes', 'mixed', 'mixed-integer'].
1956 
1957     The default string types ['string', 'empty'] are allowed for all methods.
1958     For the additional types ['bytes', 'mixed', 'mixed-integer'], each method
1959     then needs to forbid the types it is not intended for.
1960 
1961     Parameters
1962     ----------
1963     forbidden : list-of-str or None
1964         List of forbidden non-string types, may be one or more of
1965         `['bytes', 'mixed', 'mixed-integer']`.
1966     name : str, default None
1967         Name of the method to use in the error message. By default, this is
1968         None, in which case the name from the method being wrapped will be
1969         copied. However, for working with further wrappers (like _pat_wrapper
1970         and _noarg_wrapper), it is necessary to specify the name.
1971 
1972     Returns
1973     -------
1974     func : wrapper
1975         The method to which the decorator is applied, with an added check that
1976         enforces the inferred type to not be in the list of forbidden types.
1977 
1978     Raises
1979     ------
1980     TypeError
1981         If the inferred type of the underlying data is in `forbidden`.
1982     """
1983     # deal with None
1984     forbidden = [] if forbidden is None else forbidden
1985 
1986     allowed_types = {"string", "empty", "bytes", "mixed", "mixed-integer"} - set(
1987         forbidden
1988     )
1989 
1990     def _forbid_nonstring_types(func):
1991         func_name = func.__name__ if name is None else name
1992 
1993         @wraps(func)
1994         def wrapper(self, *args, **kwargs):
1995             if self._inferred_dtype not in allowed_types:
1996                 msg = (
1997                     f"Cannot use .str.{func_name} with values of "
1998                     f"inferred dtype '{self._inferred_dtype}'."
1999                 )
2000                 raise TypeError(msg)
2001             return func(self, *args, **kwargs)
2002 
2003         wrapper.__name__ = func_name
2004         return wrapper
2005 
2006     return _forbid_nonstring_types
