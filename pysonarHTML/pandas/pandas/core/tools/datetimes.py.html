  71 def _guess_datetime_format_for_array(arr, **kwargs):
  72     # Try to guess the format based on the first non-NaN element
  73     non_nan_elements = notna(arr).nonzero()[0]
  74     if len(non_nan_elements):
  75         return _guess_datetime_format(arr[non_nan_elements[0]], **kwargs)

  78 def should_cache(
  79     arg: ArrayConvertible, unique_share: float = 0.7, check_count: Optional[int] = None
  80 ) -> bool:
  81     """
  82     Decides whether to do caching.
  83 
  84     If the percent of unique elements among `check_count` elements less
  85     than `unique_share * 100` then we can do caching.
  86 
  87     Parameters
  88     ----------
  89     arg: listlike, tuple, 1-d array, Series
  90     unique_share: float, default=0.7, optional
  91         0 < unique_share < 1
  92     check_count: int, optional
  93         0 <= check_count <= len(arg)
  94 
  95     Returns
  96     -------
  97     do_caching: bool
  98 
  99     Notes
 100     -----
 101     By default for a sequence of less than 50 items in size, we don't do
 102     caching; for the number of elements less than 5000, we take ten percent of
 103     all elements to check for a uniqueness share; if the sequence size is more
 104     than 5000, then we check only the first 500 elements.
 105     All constants were chosen empirically by.
 106     """
 107     do_caching = True
 108 
 109     # default realization
 110     if check_count is None:
 111         # in this case, the gain from caching is negligible
 112         if len(arg) <= 50:
 113             return False
 114 
 115         if len(arg) <= 5000:
 116             check_count = int(len(arg) * 0.1)
 117         else:
 118             check_count = 500
 119     else:
 120         assert (
 121             0 <= check_count <= len(arg)
 122         ), "check_count must be in next bounds: [0; len(arg)]"
 123         if check_count == 0:
 124             return False
 125 
 126     assert 0 < unique_share < 1, "unique_share must be in next bounds: (0; 1)"
 127 
 128     unique_elements = set(islice(arg, check_count))
 129     if len(unique_elements) > check_count * unique_share:
 130         do_caching = False
 131     return do_caching

 134 def _maybe_cache(
 135     arg: ArrayConvertible,
 136     format: Optional[str],
 137     cache: bool,
 138     convert_listlike: Callable,
 139 ) -> "Series":
 140     """
 141     Create a cache of unique dates from an array of dates
 142 
 143     Parameters
 144     ----------
 145     arg : listlike, tuple, 1-d array, Series
 146     format : string
 147         Strftime format to parse time
 148     cache : boolean
 149         True attempts to create a cache of converted values
 150     convert_listlike : function
 151         Conversion function to apply on dates
 152 
 153     Returns
 154     -------
 155     cache_array : Series
 156         Cache of converted, unique dates. Can be empty
 157     """
 158     from pandas import Series
 159 
 160     cache_array = Series(dtype=object)
 161 
 162     if cache:
 163         # Perform a quicker unique check
 164         if not should_cache(arg):
 165             return cache_array
 166 
 167         unique_dates = unique(arg)
 168         if len(unique_dates) < len(arg):
 169             cache_dates = convert_listlike(unique_dates, format)
 170             cache_array = Series(cache_dates, index=unique_dates)
 171     return cache_array

 229 def _return_parsed_timezone_results(result, timezones, tz, name):
 230     """
 231     Return results from array_strptime if a %z or %Z directive was passed.
 232 
 233     Parameters
 234     ----------
 235     result : ndarray
 236         int64 date representations of the dates
 237     timezones : ndarray
 238         pytz timezone objects
 239     tz : object
 240         None or pytz timezone object
 241     name : string, default None
 242         Name for a DatetimeIndex
 243 
 244     Returns
 245     -------
 246     tz_result : Index-like of parsed dates with timezone
 247     """
 248     tz_results = np.array(
 249         [Timestamp(res).tz_localize(zone) for res, zone in zip(result, timezones)]
 250     )
 251     if tz is not None:
 252         # Convert to the same tz
 253         tz_results = np.array([tz_result.tz_convert(tz) for tz_result in tz_results])
 254 
 255     return Index(tz_results, name=name)

 258 def _convert_listlike_datetimes(
 259     arg,
 260     format: Optional[str],
 261     name: Label = None,
 262     tz: Optional[Timezone] = None,
 263     unit: Optional[str] = None,
 264     errors: Optional[str] = None,
 265     infer_datetime_format: Optional[bool] = None,
 266     dayfirst: Optional[bool] = None,
 267     yearfirst: Optional[bool] = None,
 268     exact: Optional[bool] = None,
 269 ):
 270     """
 271     Helper function for to_datetime. Performs the conversions of 1D listlike
 272     of dates
 273 
 274     Parameters
 275     ----------
 276     arg : list, tuple, ndarray, Series, Index
 277         date to be parsed
 278     name : object
 279         None or string for the Index name
 280     tz : object
 281         None or 'utc'
 282     unit : string
 283         None or string of the frequency of the passed data
 284     errors : string
 285         error handing behaviors from to_datetime, 'raise', 'coerce', 'ignore'
 286     infer_datetime_format : boolean
 287         inferring format behavior from to_datetime
 288     dayfirst : boolean
 289         dayfirst parsing behavior from to_datetime
 290     yearfirst : boolean
 291         yearfirst parsing behavior from to_datetime
 292     exact : boolean
 293         exact format matching behavior from to_datetime
 294 
 295     Returns
 296     -------
 297     Index-like of parsed dates
 298     """
 299 
 300     if isinstance(arg, (list, tuple)):
 301         arg = np.array(arg, dtype="O")
 302 
 303     arg_dtype = getattr(arg, "dtype", None)
 304     # these are shortcutable
 305     if is_datetime64tz_dtype(arg_dtype):
 306         if not isinstance(arg, (DatetimeArray, DatetimeIndex)):
 307             return DatetimeIndex(arg, tz=tz, name=name)
 308         if tz == "utc":
 309             # error: Item "DatetimeIndex" of "Union[DatetimeArray, DatetimeIndex]" has
 310             # no attribute "tz_convert"
 311             arg = arg.tz_convert(None).tz_localize(tz)  # type: ignore
 312         return arg
 313 
 314     elif is_datetime64_ns_dtype(arg_dtype):
 315         if not isinstance(arg, (DatetimeArray, DatetimeIndex)):
 316             try:
 317                 return DatetimeIndex(arg, tz=tz, name=name)
 318             except ValueError:
 319                 pass
 320         elif tz:
 321             # DatetimeArray, DatetimeIndex
 322             return arg.tz_localize(tz)
 323 
 324         return arg
 325 
 326     elif unit is not None:
 327         if format is not None:
 328             raise ValueError("cannot specify both format and unit")
 329         arg = getattr(arg, "_values", arg)
 330 
 331         # GH 30050 pass an ndarray to tslib.array_with_unit_to_datetime
 332         # because it expects an ndarray argument
 333         if isinstance(arg, IntegerArray):
 334             result = arg.astype(f"datetime64[{unit}]")
 335             tz_parsed = None
 336         else:
 337 
 338             result, tz_parsed = tslib.array_with_unit_to_datetime(
 339                 arg, unit, errors=errors
 340             )
 341 
 342         if errors == "ignore":
 343 
 344             result = Index(result, name=name)
 345         else:
 346             result = DatetimeIndex(result, name=name)
 347         # GH 23758: We may still need to localize the result with tz
 348         # GH 25546: Apply tz_parsed first (from arg), then tz (from caller)
 349         # result will be naive but in UTC
 350         try:
 351             result = result.tz_localize("UTC").tz_convert(tz_parsed)
 352         except AttributeError:
 353             # Regular Index from 'ignore' path
 354             return result
 355         if tz is not None:
 356             if result.tz is None:
 357                 result = result.tz_localize(tz)
 358             else:
 359                 result = result.tz_convert(tz)
 360         return result
 361     elif getattr(arg, "ndim", 1) > 1:
 362         raise TypeError(
 363             "arg must be a string, datetime, list, tuple, 1-d array, or Series"
 364         )
 365 
 366     # warn if passing timedelta64, raise for PeriodDtype
 367     # NB: this must come after unit transformation
 368     orig_arg = arg
 369     try:
 370         arg, _ = maybe_convert_dtype(arg, copy=False)
 371     except TypeError:
 372         if errors == "coerce":
 373             result = np.array(["NaT"], dtype="datetime64[ns]").repeat(len(arg))
 374             return DatetimeIndex(result, name=name)
 375         elif errors == "ignore":
 376             result = Index(arg, name=name)
 377             return result
 378         raise
 379 
 380     arg = ensure_object(arg)
 381     require_iso8601 = False
 382 
 383     if infer_datetime_format and format is None:
 384         format = _guess_datetime_format_for_array(arg, dayfirst=dayfirst)
 385 
 386     if format is not None:
 387         # There is a special fast-path for iso8601 formatted
 388         # datetime strings, so in those cases don't use the inferred
 389         # format because this path makes process slower in this
 390         # special case
 391         format_is_iso8601 = _format_is_iso(format)
 392         if format_is_iso8601:
 393             require_iso8601 = not infer_datetime_format
 394             format = None
 395 
 396     tz_parsed = None
 397     result = None
 398 
 399     if format is not None:
 400         try:
 401             # shortcut formatting here
 402             if format == "%Y%m%d":
 403                 try:
 404                     # pass orig_arg as float-dtype may have been converted to
 405                     # datetime64[ns]
 406                     orig_arg = ensure_object(orig_arg)
 407                     result = _attempt_YYYYMMDD(orig_arg, errors=errors)
 408                 except (ValueError, TypeError, tslibs.OutOfBoundsDatetime) as err:
 409                     raise ValueError(
 410                         "cannot convert the input to '%Y%m%d' date format"
 411                     ) from err
 412 
 413             # fallback
 414             if result is None:
 415                 try:
 416                     result, timezones = array_strptime(
 417                         arg, format, exact=exact, errors=errors
 418                     )
 419                     if "%Z" in format or "%z" in format:
 420                         return _return_parsed_timezone_results(
 421                             result, timezones, tz, name
 422                         )
 423                 except tslibs.OutOfBoundsDatetime:
 424                     if errors == "raise":
 425                         raise
 426                     elif errors == "coerce":
 427                         result = np.empty(arg.shape, dtype="M8[ns]")
 428                         iresult = result.view("i8")
 429                         iresult.fill(tslibs.iNaT)
 430                     else:
 431                         result = arg
 432                 except ValueError:
 433                     # if format was inferred, try falling back
 434                     # to array_to_datetime - terminate here
 435                     # for specified formats
 436                     if not infer_datetime_format:
 437                         if errors == "raise":
 438                             raise
 439                         elif errors == "coerce":
 440                             result = np.empty(arg.shape, dtype="M8[ns]")
 441                             iresult = result.view("i8")
 442                             iresult.fill(tslibs.iNaT)
 443                         else:
 444                             result = arg
 445         except ValueError as e:
 446             # Fallback to try to convert datetime objects if timezone-aware
 447             #  datetime objects are found without passing `utc=True`
 448             try:
 449                 values, tz = conversion.datetime_to_datetime64(arg)
 450                 dta = DatetimeArray(values, dtype=tz_to_dtype(tz))
 451                 return DatetimeIndex._simple_new(dta, name=name)
 452             except (ValueError, TypeError):
 453                 raise e
 454 
 455     if result is None:
 456         assert format is None or infer_datetime_format
 457         utc = tz == "utc"
 458         result, tz_parsed = objects_to_datetime64ns(
 459             arg,
 460             dayfirst=dayfirst,
 461             yearfirst=yearfirst,
 462             utc=utc,
 463             errors=errors,
 464             require_iso8601=require_iso8601,
 465             allow_object=True,
 466         )
 467 
 468     if tz_parsed is not None:
 469         # We can take a shortcut since the datetime64 numpy array
 470         # is in UTC
 471         dta = DatetimeArray(result, dtype=tz_to_dtype(tz_parsed))
 472         return DatetimeIndex._simple_new(dta, name=name)
 473 
 474     utc = tz == "utc"
 475     return _box_as_indexlike(result, utc=utc, name=name)

 478 def _adjust_to_origin(arg, origin, unit):
 479     """
 480     Helper function for to_datetime.
 481     Adjust input argument to the specified origin
 482 
 483     Parameters
 484     ----------
 485     arg : list, tuple, ndarray, Series, Index
 486         date to be adjusted
 487     origin : 'julian' or Timestamp
 488         origin offset for the arg
 489     unit : string
 490         passed unit from to_datetime, must be 'D'
 491 
 492     Returns
 493     -------
 494     ndarray or scalar of adjusted date(s)
 495     """
 496     if origin == "julian":
 497         original = arg
 498         j0 = Timestamp(0).to_julian_date()
 499         if unit != "D":
 500             raise ValueError("unit must be 'D' for origin='julian'")
 501         try:
 502             arg = arg - j0
 503         except TypeError as err:
 504             raise ValueError(
 505                 "incompatible 'arg' type for given 'origin'='julian'"
 506             ) from err
 507 
 508         # preemptively check this for a nice range
 509         j_max = Timestamp.max.to_julian_date() - j0
 510         j_min = Timestamp.min.to_julian_date() - j0
 511         if np.any(arg > j_max) or np.any(arg < j_min):
 512             raise tslibs.OutOfBoundsDatetime(
 513                 f"{original} is Out of Bounds for origin='julian'"
 514             )
 515     else:
 516         # arg must be numeric
 517         if not (
 518             (is_scalar(arg) and (is_integer(arg) or is_float(arg)))
 519             or is_numeric_dtype(np.asarray(arg))
 520         ):
 521             raise ValueError(
 522                 f"'{arg}' is not compatible with origin='{origin}'; "
 523                 "it must be numeric with a unit specified"
 524             )
 525 
 526         # we are going to offset back to unix / epoch time
 527         try:
 528             offset = Timestamp(origin)
 529         except tslibs.OutOfBoundsDatetime as err:
 530             raise tslibs.OutOfBoundsDatetime(
 531                 f"origin {origin} is Out of Bounds"
 532             ) from err
 533         except ValueError as err:
 534             raise ValueError(
 535                 f"origin {origin} cannot be converted to a Timestamp"
 536             ) from err
 537 
 538         if offset.tz is not None:
 539             raise ValueError(f"origin offset {offset} must be tz-naive")
 540         offset -= Timestamp(0)
 541 
 542         # convert the offset to the unit of the arg
 543         # this should be lossless in terms of precision
 544         offset = offset // tslibs.Timedelta(1, unit=unit)
 545 
 546         # scalars & ndarray-like can handle the addition
 547         if is_list_like(arg) and not isinstance(arg, (ABCSeries, Index, np.ndarray)):
 548             arg = np.asarray(arg)
 549         arg = arg + offset
 550     return arg

 604 def to_datetime(
 605     arg: DatetimeScalarOrArrayConvertible,
 606     errors: str = "raise",
 607     dayfirst: bool = False,
 608     yearfirst: bool = False,
 609     utc: Optional[bool] = None,
 610     format: Optional[str] = None,
 611     exact: bool = True,
 612     unit: Optional[str] = None,
 613     infer_datetime_format: bool = False,
 614     origin="unix",
 615     cache: bool = True,
 616 ) -> Union[DatetimeIndex, "Series", DatetimeScalar, "NaTType"]:
 617     """
 618     Convert argument to datetime.
 619 
 620     Parameters
 621     ----------
 622     arg : int, float, str, datetime, list, tuple, 1-d array, Series, DataFrame/dict-like
 623         The object to convert to a datetime.
 624     errors : {'ignore', 'raise', 'coerce'}, default 'raise'
 625         - If 'raise', then invalid parsing will raise an exception.
 626         - If 'coerce', then invalid parsing will be set as NaT.
 627         - If 'ignore', then invalid parsing will return the input.
 628     dayfirst : bool, default False
 629         Specify a date parse order if `arg` is str or its list-likes.
 630         If True, parses dates with the day first, eg 10/11/12 is parsed as
 631         2012-11-10.
 632         Warning: dayfirst=True is not strict, but will prefer to parse
 633         with day first (this is a known bug, based on dateutil behavior).
 634     yearfirst : bool, default False
 635         Specify a date parse order if `arg` is str or its list-likes.
 636 
 637         - If True parses dates with the year first, eg 10/11/12 is parsed as
 638           2010-11-12.
 639         - If both dayfirst and yearfirst are True, yearfirst is preceded (same
 640           as dateutil).
 641 
 642         Warning: yearfirst=True is not strict, but will prefer to parse
 643         with year first (this is a known bug, based on dateutil behavior).
 644     utc : bool, default None
 645         Return UTC DatetimeIndex if True (converting any tz-aware
 646         datetime.datetime objects as well).
 647     format : str, default None
 648         The strftime to parse time, eg "%d/%m/%Y", note that "%f" will parse
 649         all the way up to nanoseconds.
 650         See strftime documentation for more information on choices:
 651         https://docs.python.org/3/library/datetime.html#strftime-and-strptime-behavior.
 652     exact : bool, True by default
 653         Behaves as:
 654         - If True, require an exact format match.
 655         - If False, allow the format to match anywhere in the target string.
 656 
 657     unit : str, default 'ns'
 658         The unit of the arg (D,s,ms,us,ns) denote the unit, which is an
 659         integer or float number. This will be based off the origin.
 660         Example, with unit='ms' and origin='unix' (the default), this
 661         would calculate the number of milliseconds to the unix epoch start.
 662     infer_datetime_format : bool, default False
 663         If True and no `format` is given, attempt to infer the format of the
 664         datetime strings based on the first non-NaN element,
 665         and if it can be inferred, switch to a faster method of parsing them.
 666         In some cases this can increase the parsing speed by ~5-10x.
 667     origin : scalar, default 'unix'
 668         Define the reference date. The numeric values would be parsed as number
 669         of units (defined by `unit`) since this reference date.
 670 
 671         - If 'unix' (or POSIX) time; origin is set to 1970-01-01.
 672         - If 'julian', unit must be 'D', and origin is set to beginning of
 673           Julian Calendar. Julian day number 0 is assigned to the day starting
 674           at noon on January 1, 4713 BC.
 675         - If Timestamp convertible, origin is set to Timestamp identified by
 676           origin.
 677     cache : bool, default True
 678         If True, use a cache of unique, converted dates to apply the datetime
 679         conversion. May produce significant speed-up when parsing duplicate
 680         date strings, especially ones with timezone offsets. The cache is only
 681         used when there are at least 50 values. The presence of out-of-bounds
 682         values will render the cache unusable and may slow down parsing.
 683 
 684         .. versionadded:: 0.23.0
 685 
 686         .. versionchanged:: 0.25.0
 687             - changed default value from False to True.
 688 
 689     Returns
 690     -------
 691     datetime
 692         If parsing succeeded.
 693         Return type depends on input:
 694 
 695         - list-like: DatetimeIndex
 696         - Series: Series of datetime64 dtype
 697         - scalar: Timestamp
 698 
 699         In case when it is not possible to return designated types (e.g. when
 700         any element of input is before Timestamp.min or after Timestamp.max)
 701         return will have datetime.datetime type (or corresponding
 702         array/Series).
 703 
 704     See Also
 705     --------
 706     DataFrame.astype : Cast argument to a specified dtype.
 707     to_timedelta : Convert argument to timedelta.
 708     convert_dtypes : Convert dtypes.
 709 
 710     Examples
 711     --------
 712     Assembling a datetime from multiple columns of a DataFrame. The keys can be
 713     common abbreviations like ['year', 'month', 'day', 'minute', 'second',
 714     'ms', 'us', 'ns']) or plurals of the same
 715 
 716     >>> df = pd.DataFrame({'year': [2015, 2016],
 717     ...                    'month': [2, 3],
 718     ...                    'day': [4, 5]})
 719     >>> pd.to_datetime(df)
 720     0   2015-02-04
 721     1   2016-03-05
 722     dtype: datetime64[ns]
 723 
 724     If a date does not meet the `timestamp limitations
 725     <https://pandas.pydata.org/pandas-docs/stable/user_guide/timeseries.html
 726     #timeseries-timestamp-limits>`_, passing errors='ignore'
 727     will return the original input instead of raising any exception.
 728 
 729     Passing errors='coerce' will force an out-of-bounds date to NaT,
 730     in addition to forcing non-dates (or non-parseable dates) to NaT.
 731 
 732     >>> pd.to_datetime('13000101', format='%Y%m%d', errors='ignore')
 733     datetime.datetime(1300, 1, 1, 0, 0)
 734     >>> pd.to_datetime('13000101', format='%Y%m%d', errors='coerce')
 735     NaT
 736 
 737     Passing infer_datetime_format=True can often-times speedup a parsing
 738     if its not an ISO8601 format exactly, but in a regular format.
 739 
 740     >>> s = pd.Series(['3/11/2000', '3/12/2000', '3/13/2000'] * 1000)
 741     >>> s.head()
 742     0    3/11/2000
 743     1    3/12/2000
 744     2    3/13/2000
 745     3    3/11/2000
 746     4    3/12/2000
 747     dtype: object
 748 
 749     >>> %timeit pd.to_datetime(s, infer_datetime_format=True)  # doctest: +SKIP
 750     100 loops, best of 3: 10.4 ms per loop
 751 
 752     >>> %timeit pd.to_datetime(s, infer_datetime_format=False)  # doctest: +SKIP
 753     1 loop, best of 3: 471 ms per loop
 754 
 755     Using a unix epoch time
 756 
 757     >>> pd.to_datetime(1490195805, unit='s')
 758     Timestamp('2017-03-22 15:16:45')
 759     >>> pd.to_datetime(1490195805433502912, unit='ns')
 760     Timestamp('2017-03-22 15:16:45.433502912')
 761 
 762     .. warning:: For float arg, precision rounding might happen. To prevent
 763         unexpected behavior use a fixed-width exact type.
 764 
 765     Using a non-unix epoch origin
 766 
 767     >>> pd.to_datetime([1, 2, 3], unit='D',
 768     ...                origin=pd.Timestamp('1960-01-01'))
 769     DatetimeIndex(['1960-01-02', '1960-01-03', '1960-01-04'], \
 770 dtype='datetime64[ns]', freq=None)
 771     """
 772     if arg is None:
 773         return None
 774 
 775     if origin != "unix":
 776         arg = _adjust_to_origin(arg, origin, unit)
 777 
 778     tz = "utc" if utc else None
 779     convert_listlike = partial(
 780         _convert_listlike_datetimes,
 781         tz=tz,
 782         unit=unit,
 783         dayfirst=dayfirst,
 784         yearfirst=yearfirst,
 785         errors=errors,
 786         exact=exact,
 787         infer_datetime_format=infer_datetime_format,
 788     )
 789 
 790     if isinstance(arg, Timestamp):
 791         result = arg
 792         if tz is not None:
 793             if arg.tz is not None:
 794                 result = result.tz_convert(tz)
 795             else:
 796                 result = result.tz_localize(tz)
 797     elif isinstance(arg, ABCSeries):
 798         cache_array = _maybe_cache(arg, format, cache, convert_listlike)
 799         if not cache_array.empty:
 800             result = arg.map(cache_array)
 801         else:
 802             values = convert_listlike(arg._values, format)
 803             result = arg._constructor(values, index=arg.index, name=arg.name)
 804     elif isinstance(arg, (ABCDataFrame, abc.MutableMapping)):
 805         result = _assemble_from_unit_mappings(arg, errors, tz)
 806     elif isinstance(arg, Index):
 807         cache_array = _maybe_cache(arg, format, cache, convert_listlike)
 808         if not cache_array.empty:
 809             result = _convert_and_box_cache(arg, cache_array, name=arg.name)
 810         else:
 811             result = convert_listlike(arg, format, name=arg.name)
 812     elif is_list_like(arg):
 813         try:
 814             cache_array = _maybe_cache(arg, format, cache, convert_listlike)
 815         except tslibs.OutOfBoundsDatetime:
 816             # caching attempts to create a DatetimeIndex, which may raise
 817             # an OOB. If that's the desired behavior, then just reraise...
 818             if errors == "raise":
 819                 raise
 820             # ... otherwise, continue without the cache.
 821             from pandas import Series
 822 
 823             cache_array = Series([], dtype=object)  # just an empty array
 824         if not cache_array.empty:
 825             result = _convert_and_box_cache(arg, cache_array)
 826         else:
 827             result = convert_listlike(arg, format)
 828     else:
 829         result = convert_listlike(np.array([arg]), format)[0]
 830 
 831     return result

 860 def _assemble_from_unit_mappings(arg, errors, tz):
 861     """
 862     assemble the unit specified fields from the arg (DataFrame)
 863     Return a Series for actual parsing
 864 
 865     Parameters
 866     ----------
 867     arg : DataFrame
 868     errors : {'ignore', 'raise', 'coerce'}, default 'raise'
 869 
 870         - If 'raise', then invalid parsing will raise an exception
 871         - If 'coerce', then invalid parsing will be set as NaT
 872         - If 'ignore', then invalid parsing will return the input
 873     tz : None or 'utc'
 874 
 875     Returns
 876     -------
 877     Series
 878     """
 879     from pandas import to_timedelta, to_numeric, DataFrame
 880 
 881     arg = DataFrame(arg)
 882     if not arg.columns.is_unique:
 883         raise ValueError("cannot assemble with duplicate keys")
 884 
 885     # replace passed unit with _unit_map
 886     def f(value):
 887         if value in _unit_map:
 888             return _unit_map[value]
 889 
 890         # m is case significant
 891         if value.lower() in _unit_map:
 892             return _unit_map[value.lower()]
 893 
 894         return value
 895 
 896     unit = {k: f(k) for k in arg.keys()}
 897     unit_rev = {v: k for k, v in unit.items()}
 898 
 899     # we require at least Ymd
 900     required = ["year", "month", "day"]
 901     req = sorted(set(required) - set(unit_rev.keys()))
 902     if len(req):
 903         _required = ",".join(req)
 904         raise ValueError(
 905             "to assemble mappings requires at least that "
 906             f"[year, month, day] be specified: [{_required}] is missing"
 907         )
 908 
 909     # keys we don't recognize
 910     excess = sorted(set(unit_rev.keys()) - set(_unit_map.values()))
 911     if len(excess):
 912         _excess = ",".join(excess)
 913         raise ValueError(
 914             f"extra keys have been passed to the datetime assemblage: [{_excess}]"
 915         )
 916 
 917     def coerce(values):
 918         # we allow coercion to if errors allows
 919         values = to_numeric(values, errors=errors)
 920 
 921         # prevent overflow in case of int8 or int16
 922         if is_integer_dtype(values):
 923             values = values.astype("int64", copy=False)
 924         return values
 925 
 926     values = (
 927         coerce(arg[unit_rev["year"]]) * 10000
 928         + coerce(arg[unit_rev["month"]]) * 100
 929         + coerce(arg[unit_rev["day"]])
 930     )
 931     try:
 932         values = to_datetime(values, format="%Y%m%d", errors=errors, utc=tz)
 933     except (TypeError, ValueError) as err:
 934         raise ValueError(f"cannot assemble the datetimes: {err}") from err
 935 
 936     for u in ["h", "m", "s", "ms", "us", "ns"]:
 937         value = unit_rev.get(u)
 938         if value is not None and value in arg:
 939             try:
 940                 values += to_timedelta(coerce(arg[value]), unit=u, errors=errors)
 941             except (TypeError, ValueError) as err:
 942                 raise ValueError(
 943                     f"cannot assemble the datetimes [{value}]: {err}"
 944                 ) from err
 945     return values

 948 def _attempt_YYYYMMDD(arg, errors):
 949     """
 950     try to parse the YYYYMMDD/%Y%m%d format, try to deal with NaT-like,
 951     arg is a passed in as an object dtype, but could really be ints/strings
 952     with nan-like/or floats (e.g. with nan)
 953 
 954     Parameters
 955     ----------
 956     arg : passed value
 957     errors : 'raise','ignore','coerce'
 958     """
 959 
 960     def calc(carg):
 961         # calculate the actual result
 962         carg = carg.astype(object)
 963         parsed = parsing.try_parse_year_month_day(
 964             carg / 10000, carg / 100 % 100, carg % 100
 965         )
 966         return tslib.array_to_datetime(parsed, errors=errors)[0]
 967 
 968     def calc_with_mask(carg, mask):
 969         result = np.empty(carg.shape, dtype="M8[ns]")
 970         iresult = result.view("i8")
 971         iresult[~mask] = tslibs.iNaT
 972 
 973         masked_result = calc(carg[mask].astype(np.float64).astype(np.int64))
 974         result[mask] = masked_result.astype("M8[ns]")
 975         return result
 976 
 977     # try intlike / strings that are ints
 978     try:
 979         return calc(arg.astype(np.int64))
 980     except (ValueError, OverflowError, TypeError):
 981         pass
 982 
 983     # a float with actual np.nan
 984     try:
 985         carg = arg.astype(np.float64)
 986         return calc_with_mask(carg, notna(carg))
 987     except (ValueError, OverflowError, TypeError):
 988         pass
 989 
 990     # string with NaN-like
 991     try:
 992         mask = ~algorithms.isin(arg, list(tslibs.nat_strings))
 993         return calc_with_mask(arg, mask)
 994     except (ValueError, OverflowError, TypeError):
 995         pass
 996 
 997     return None
