  23 def _get_method_wrappers(cls):
  24     """
  25     Find the appropriate operation-wrappers to use when defining flex/special
  26     arithmetic, boolean, and comparison operations with the given class.
  27 
  28     Parameters
  29     ----------
  30     cls : class
  31 
  32     Returns
  33     -------
  34     arith_flex : function or None
  35     comp_flex : function or None
  36     arith_special : function
  37     comp_special : function
  38     bool_special : function
  39 
  40     Notes
  41     -----
  42     None is only returned for SparseArray
  43     """
  44     # TODO: make these non-runtime imports once the relevant functions
  45     #  are no longer in __init__
  46     from pandas.core.ops import (
  47         _arith_method_FRAME,
  48         _arith_method_SERIES,
  49         _bool_method_SERIES,
  50         _comp_method_FRAME,
  51         _comp_method_SERIES,
  52         _flex_comp_method_FRAME,
  53         _flex_method_SERIES,
  54     )
  55 
  56     if issubclass(cls, ABCSeries):
  57         # Just Series
  58         arith_flex = _flex_method_SERIES
  59         comp_flex = _flex_method_SERIES
  60         arith_special = _arith_method_SERIES
  61         comp_special = _comp_method_SERIES
  62         bool_special = _bool_method_SERIES
  63     elif issubclass(cls, ABCDataFrame):
  64         arith_flex = _arith_method_FRAME
  65         comp_flex = _flex_comp_method_FRAME
  66         arith_special = _arith_method_FRAME
  67         comp_special = _comp_method_FRAME
  68         bool_special = _arith_method_FRAME
  69     return arith_flex, comp_flex, arith_special, comp_special, bool_special

  72 def add_special_arithmetic_methods(cls):
  73     """
  74     Adds the full suite of special arithmetic methods (``__add__``,
  75     ``__sub__``, etc.) to the class.
  76 
  77     Parameters
  78     ----------
  79     cls : class
  80         special methods will be defined and pinned to this class
  81     """
  82     _, _, arith_method, comp_method, bool_method = _get_method_wrappers(cls)
  83     new_methods = _create_methods(
  84         cls, arith_method, comp_method, bool_method, special=True
  85     )
  86     # inplace operators (I feel like these should get passed an `inplace=True`
  87     # or just be removed
  88 
  89     def _wrap_inplace_method(method):
  90         """
  91         return an inplace wrapper for this method
  92         """
  93 
  94         def f(self, other):
  95             result = method(self, other)
  96             # Delete cacher
  97             self._reset_cacher()
  98             # this makes sure that we are aligned like the input
  99             # we are updating inplace so we want to ignore is_copy
 100             self._update_inplace(
 101                 result.reindex_like(self, copy=False), verify_is_copy=False
 102             )
 103 
 104             return self
 105 
 106         name = method.__name__.strip("__")
 107         f.__name__ = f"__i{name}__"
 108         return f
 109 
 110     new_methods.update(
 111         dict(
 112             __iadd__=_wrap_inplace_method(new_methods["__add__"]),
 113             __isub__=_wrap_inplace_method(new_methods["__sub__"]),
 114             __imul__=_wrap_inplace_method(new_methods["__mul__"]),
 115             __itruediv__=_wrap_inplace_method(new_methods["__truediv__"]),
 116             __ifloordiv__=_wrap_inplace_method(new_methods["__floordiv__"]),
 117             __imod__=_wrap_inplace_method(new_methods["__mod__"]),
 118             __ipow__=_wrap_inplace_method(new_methods["__pow__"]),
 119         )
 120     )
 121 
 122     new_methods.update(
 123         dict(
 124             __iand__=_wrap_inplace_method(new_methods["__and__"]),
 125             __ior__=_wrap_inplace_method(new_methods["__or__"]),
 126             __ixor__=_wrap_inplace_method(new_methods["__xor__"]),
 127         )
 128     )
 129 
 130     _add_methods(cls, new_methods=new_methods)

 133 def add_flex_arithmetic_methods(cls):
 134     """
 135     Adds the full suite of flex arithmetic methods (``pow``, ``mul``, ``add``)
 136     to the class.
 137 
 138     Parameters
 139     ----------
 140     cls : class
 141         flex methods will be defined and pinned to this class
 142     """
 143     flex_arith_method, flex_comp_method, _, _, _ = _get_method_wrappers(cls)
 144     new_methods = _create_methods(
 145         cls, flex_arith_method, flex_comp_method, bool_method=None, special=False
 146     )
 147     new_methods.update(
 148         dict(
 149             multiply=new_methods["mul"],
 150             subtract=new_methods["sub"],
 151             divide=new_methods["div"],
 152         )
 153     )
 154     # opt out of bool flex methods for now
 155     assert not any(kname in new_methods for kname in ("ror_", "rxor", "rand_"))
 156 
 157     _add_methods(cls, new_methods=new_methods)

 160 def _create_methods(cls, arith_method, comp_method, bool_method, special):
 161     # creates actual methods based upon arithmetic, comp and bool method
 162     # constructors.
 163 
 164     have_divmod = issubclass(cls, ABCSeries)
 165     # divmod is available for Series
 166 
 167     new_methods = dict(
 168         add=arith_method(cls, operator.add, special),
 169         radd=arith_method(cls, radd, special),
 170         sub=arith_method(cls, operator.sub, special),
 171         mul=arith_method(cls, operator.mul, special),
 172         truediv=arith_method(cls, operator.truediv, special),
 173         floordiv=arith_method(cls, operator.floordiv, special),
 174         # Causes a floating point exception in the tests when numexpr enabled,
 175         # so for now no speedup
 176         mod=arith_method(cls, operator.mod, special),
 177         pow=arith_method(cls, operator.pow, special),
 178         # not entirely sure why this is necessary, but previously was included
 179         # so it's here to maintain compatibility
 180         rmul=arith_method(cls, rmul, special),
 181         rsub=arith_method(cls, rsub, special),
 182         rtruediv=arith_method(cls, rtruediv, special),
 183         rfloordiv=arith_method(cls, rfloordiv, special),
 184         rpow=arith_method(cls, rpow, special),
 185         rmod=arith_method(cls, rmod, special),
 186     )
 187     new_methods["div"] = new_methods["truediv"]
 188     new_methods["rdiv"] = new_methods["rtruediv"]
 189     if have_divmod:
 190         # divmod doesn't have an op that is supported by numexpr
 191         new_methods["divmod"] = arith_method(cls, divmod, special)
 192         new_methods["rdivmod"] = arith_method(cls, rdivmod, special)
 193 
 194     new_methods.update(
 195         dict(
 196             eq=comp_method(cls, operator.eq, special),
 197             ne=comp_method(cls, operator.ne, special),
 198             lt=comp_method(cls, operator.lt, special),
 199             gt=comp_method(cls, operator.gt, special),
 200             le=comp_method(cls, operator.le, special),
 201             ge=comp_method(cls, operator.ge, special),
 202         )
 203     )
 204 
 205     if bool_method:
 206         new_methods.update(
 207             dict(
 208                 and_=bool_method(cls, operator.and_, special),
 209                 or_=bool_method(cls, operator.or_, special),
 210                 xor=bool_method(cls, operator.xor, special),
 211                 rand_=bool_method(cls, rand_, special),
 212                 ror_=bool_method(cls, ror_, special),
 213                 rxor=bool_method(cls, rxor, special),
 214             )
 215         )
 216 
 217     if special:
 218         dunderize = lambda x: f"__{x.strip('_')}__"
 219     else:
 220         dunderize = lambda x: x
 221     new_methods = {dunderize(k): v for k, v in new_methods.items()}
 222     return new_methods

 225 def _add_methods(cls, new_methods):
 226     for name, method in new_methods.items():
 227         setattr(cls, name, method)
