  38 def comp_method_OBJECT_ARRAY(op, x, y):
  39     if isinstance(y, list):
  40         y = construct_1d_object_array_from_listlike(y)
  41 
  42     if isinstance(y, (np.ndarray, ABCSeries, ABCIndex)):
  43         # Note: these checks can be for ABCIndex and not ABCIndexClass
  44         #  because that is the only object-dtype class.
  45         if not is_object_dtype(y.dtype):
  46             y = y.astype(np.object_)
  47 
  48         if isinstance(y, (ABCSeries, ABCIndex)):
  49             y = y._values
  50 
  51         if x.shape != y.shape:
  52             raise ValueError("Shapes must match", x.shape, y.shape)
  53         result = libops.vec_compare(x.ravel(), y.ravel(), op)
  54     else:
  55         result = libops.scalar_compare(x.ravel(), y, op)
  56     return result.reshape(x.shape)

  59 def masked_arith_op(x: np.ndarray, y, op):
  60     """
  61     If the given arithmetic operation fails, attempt it again on
  62     only the non-null elements of the input array(s).
  63 
  64     Parameters
  65     ----------
  66     x : np.ndarray
  67     y : np.ndarray, Series, Index
  68     op : binary operator
  69     """
  70     # For Series `x` is 1D so ravel() is a no-op; calling it anyway makes
  71     # the logic valid for both Series and DataFrame ops.
  72     xrav = x.ravel()
  73     assert isinstance(x, np.ndarray), type(x)
  74     if isinstance(y, np.ndarray):
  75         dtype = find_common_type([x.dtype, y.dtype])
  76         result = np.empty(x.size, dtype=dtype)
  77 
  78         if len(x) != len(y):
  79             raise ValueError(x.shape, y.shape)
  80         else:
  81             ymask = notna(y)
  82 
  83         # NB: ravel() is only safe since y is ndarray; for e.g. PeriodIndex
  84         #  we would get int64 dtype, see GH#19956
  85         yrav = y.ravel()
  86         mask = notna(xrav) & ymask.ravel()
  87 
  88         # See GH#5284, GH#5035, GH#19448 for historical reference
  89         if mask.any():
  90             with np.errstate(all="ignore"):
  91                 result[mask] = op(xrav[mask], yrav[mask])
  92 
  93     else:
  94         if not is_scalar(y):
  95             raise TypeError(
  96                 f"Cannot broadcast np.ndarray with operand of type { type(y) }"
  97             )
  98 
  99         # mask is only meaningful for x
 100         result = np.empty(x.size, dtype=x.dtype)
 101         mask = notna(xrav)
 102 
 103         # 1 ** np.nan is 1. So we have to unmask those.
 104         if op is pow:
 105             mask = np.where(x == 1, False, mask)
 106         elif op is rpow:
 107             mask = np.where(y == 1, False, mask)
 108 
 109         if mask.any():
 110             with np.errstate(all="ignore"):
 111                 result[mask] = op(xrav[mask], y)
 112 
 113     result, _ = maybe_upcast_putmask(result, ~mask, np.nan)
 114     result = result.reshape(x.shape)  # 2D compat
 115     return result

 118 def na_arithmetic_op(left, right, op, is_cmp: bool = False):
 119     """
 120     Return the result of evaluating op on the passed in values.
 121 
 122     If native types are not compatible, try coercion to object dtype.
 123 
 124     Parameters
 125     ----------
 126     left : np.ndarray
 127     right : np.ndarray or scalar
 128     is_cmp : bool, default False
 129         If this a comparison operation.
 130 
 131     Returns
 132     -------
 133     array-like
 134 
 135     Raises
 136     ------
 137     TypeError : invalid operation
 138     """
 139     import pandas.core.computation.expressions as expressions
 140 
 141     try:
 142         result = expressions.evaluate(op, left, right)
 143     except TypeError:
 144         if is_cmp:
 145             # numexpr failed on comparison op, e.g. ndarray[float] > datetime
 146             #  In this case we do not fall back to the masked op, as that
 147             #  will handle complex numbers incorrectly, see GH#32047
 148             raise
 149         result = masked_arith_op(left, right, op)
 150 
 151     if is_cmp and (is_scalar(result) or result is NotImplemented):
 152         # numpy returned a scalar instead of operating element-wise
 153         # e.g. numeric array vs str
 154         return invalid_comparison(left, right, op)
 155 
 156     return missing.dispatch_fill_zeros(op, left, right, result)

 159 def arithmetic_op(left: ArrayLike, right: Any, op):
 160     """
 161     Evaluate an arithmetic operation `+`, `-`, `*`, `/`, `//`, `%`, `**`, ...
 162 
 163     Parameters
 164     ----------
 165     left : np.ndarray or ExtensionArray
 166     right : object
 167         Cannot be a DataFrame or Index.  Series is *not* excluded.
 168     op : {operator.add, operator.sub, ...}
 169         Or one of the reversed variants from roperator.
 170 
 171     Returns
 172     -------
 173     ndarray or ExtensionArray
 174         Or a 2-tuple of these in the case of divmod or rdivmod.
 175     """
 176 
 177     # NB: We assume that extract_array has already been called
 178     #  on `left` and `right`.
 179     lvalues = maybe_upcast_datetimelike_array(left)
 180     rvalues = maybe_upcast_datetimelike_array(right)
 181     rvalues = maybe_upcast_for_op(rvalues, lvalues.shape)
 182 
 183     if should_extension_dispatch(lvalues, rvalues) or isinstance(rvalues, Timedelta):
 184         # Timedelta is included because numexpr will fail on it, see GH#31457
 185         res_values = op(lvalues, rvalues)
 186 
 187     else:
 188         with np.errstate(all="ignore"):
 189             res_values = na_arithmetic_op(lvalues, rvalues, op)
 190 
 191     return res_values

 194 def comparison_op(left: ArrayLike, right: Any, op) -> ArrayLike:
 195     """
 196     Evaluate a comparison operation `=`, `!=`, `>=`, `>`, `<=`, or `<`.
 197 
 198     Parameters
 199     ----------
 200     left : np.ndarray or ExtensionArray
 201     right : object
 202         Cannot be a DataFrame, Series, or Index.
 203     op : {operator.eq, operator.ne, operator.gt, operator.ge, operator.lt, operator.le}
 204 
 205     Returns
 206     -------
 207     ndarray or ExtensionArray
 208     """
 209     # NB: We assume extract_array has already been called on left and right
 210     lvalues = maybe_upcast_datetimelike_array(left)
 211     rvalues = right
 212 
 213     rvalues = lib.item_from_zerodim(rvalues)
 214     if isinstance(rvalues, list):
 215         # TODO: same for tuples?
 216         rvalues = np.asarray(rvalues)
 217 
 218     if isinstance(rvalues, (np.ndarray, ABCExtensionArray)):
 219         # TODO: make this treatment consistent across ops and classes.
 220         #  We are not catching all listlikes here (e.g. frozenset, tuple)
 221         #  The ambiguous case is object-dtype.  See GH#27803
 222         if len(lvalues) != len(rvalues):
 223             raise ValueError(
 224                 "Lengths must match to compare", lvalues.shape, rvalues.shape
 225             )
 226 
 227     if should_extension_dispatch(lvalues, rvalues):
 228         # Call the method on lvalues
 229         res_values = op(lvalues, rvalues)
 230 
 231     elif is_scalar(rvalues) and isna(rvalues):
 232         # numpy does not like comparisons vs None
 233         if op is operator.ne:
 234             res_values = np.ones(lvalues.shape, dtype=bool)
 235         else:
 236             res_values = np.zeros(lvalues.shape, dtype=bool)
 237 
 238     elif is_object_dtype(lvalues.dtype):
 239         res_values = comp_method_OBJECT_ARRAY(op, lvalues, rvalues)
 240 
 241     else:
 242         with warnings.catch_warnings():
 243             # suppress warnings from numpy about element-wise comparison
 244             warnings.simplefilter("ignore", DeprecationWarning)
 245             with np.errstate(all="ignore"):
 246                 res_values = na_arithmetic_op(lvalues, rvalues, op, is_cmp=True)
 247 
 248     return res_values

 251 def na_logical_op(x: np.ndarray, y, op):
 252     try:
 253         # For exposition, write:
 254         #  yarr = isinstance(y, np.ndarray)
 255         #  yint = is_integer(y) or (yarr and y.dtype.kind == "i")
 256         #  ybool = is_bool(y) or (yarr and y.dtype.kind == "b")
 257         #  xint = x.dtype.kind == "i"
 258         #  xbool = x.dtype.kind == "b"
 259         # Then Cases where this goes through without raising include:
 260         #  (xint or xbool) and (yint or bool)
 261         result = op(x, y)
 262     except TypeError:
 263         if isinstance(y, np.ndarray):
 264             # bool-bool dtype operations should be OK, should not get here
 265             assert not (is_bool_dtype(x.dtype) and is_bool_dtype(y.dtype))
 266             x = ensure_object(x)
 267             y = ensure_object(y)
 268             result = libops.vec_binop(x.ravel(), y.ravel(), op)
 269         else:
 270             # let null fall thru
 271             assert lib.is_scalar(y)
 272             if not isna(y):
 273                 y = bool(y)
 274             try:
 275                 result = libops.scalar_binop(x, y, op)
 276             except (
 277                 TypeError,
 278                 ValueError,
 279                 AttributeError,
 280                 OverflowError,
 281                 NotImplementedError,
 282             ) as err:
 283                 typ = type(y).__name__
 284                 raise TypeError(
 285                     f"Cannot perform '{op.__name__}' with a dtyped [{x.dtype}] array "
 286                     f"and scalar of type [{typ}]"
 287                 ) from err
 288 
 289     return result.reshape(x.shape)

 292 def logical_op(left: ArrayLike, right: Any, op) -> ArrayLike:
 293     """
 294     Evaluate a logical operation `|`, `&`, or `^`.
 295 
 296     Parameters
 297     ----------
 298     left : np.ndarray or ExtensionArray
 299     right : object
 300         Cannot be a DataFrame, Series, or Index.
 301     op : {operator.and_, operator.or_, operator.xor}
 302         Or one of the reversed variants from roperator.
 303 
 304     Returns
 305     -------
 306     ndarray or ExtensionArray
 307     """
 308     fill_int = lambda x: x
 309 
 310     def fill_bool(x, left=None):
 311         # if `left` is specifically not-boolean, we do not cast to bool
 312         if x.dtype.kind in ["c", "f", "O"]:
 313             # dtypes that can hold NA
 314             mask = isna(x)
 315             if mask.any():
 316                 x = x.astype(object)
 317                 x[mask] = False
 318 
 319         if left is None or is_bool_dtype(left.dtype):
 320             x = x.astype(bool)
 321         return x
 322 
 323     is_self_int_dtype = is_integer_dtype(left.dtype)
 324 
 325     right = lib.item_from_zerodim(right)
 326     if is_list_like(right) and not hasattr(right, "dtype"):
 327         # e.g. list, tuple
 328         right = construct_1d_object_array_from_listlike(right)
 329 
 330     # NB: We assume extract_array has already been called on left and right
 331     lvalues = maybe_upcast_datetimelike_array(left)
 332     rvalues = right
 333 
 334     if should_extension_dispatch(lvalues, rvalues):
 335         # Call the method on lvalues
 336         res_values = op(lvalues, rvalues)
 337 
 338     else:
 339         if isinstance(rvalues, np.ndarray):
 340             is_other_int_dtype = is_integer_dtype(rvalues.dtype)
 341             rvalues = rvalues if is_other_int_dtype else fill_bool(rvalues, lvalues)
 342 
 343         else:
 344             # i.e. scalar
 345             is_other_int_dtype = lib.is_integer(rvalues)
 346 
 347         # For int vs int `^`, `|`, `&` are bitwise operators and return
 348         #   integer dtypes.  Otherwise these are boolean ops
 349         filler = fill_int if is_self_int_dtype and is_other_int_dtype else fill_bool
 350 
 351         res_values = na_logical_op(lvalues, rvalues, op)
 352         res_values = filler(res_values)  # type: ignore
 353 
 354     return res_values

 357 def get_array_op(op):
 358     """
 359     Return a binary array operation corresponding to the given operator op.
 360 
 361     Parameters
 362     ----------
 363     op : function
 364         Binary operator from operator or roperator module.
 365 
 366     Returns
 367     -------
 368     functools.partial
 369     """
 370     if isinstance(op, partial):
 371         # We get here via dispatch_to_series in DataFrame case
 372         # TODO: avoid getting here
 373         return op
 374 
 375     op_name = op.__name__.strip("_").lstrip("r")
 376     if op_name == "arith_op":
 377         # Reached via DataFrame._combine_frame
 378         return op
 379 
 380     if op_name in {"eq", "ne", "lt", "le", "gt", "ge"}:
 381         return partial(comparison_op, op=op)
 382     elif op_name in {"and", "or", "xor", "rand", "ror", "rxor"}:
 383         return partial(logical_op, op=op)
 384     elif op_name in {
 385         "add",
 386         "sub",
 387         "mul",
 388         "truediv",
 389         "floordiv",
 390         "mod",
 391         "divmod",
 392         "pow",
 393     }:
 394         return partial(arithmetic_op, op=op)
 395     else:
 396         raise NotImplementedError(op_name)

 399 def maybe_upcast_datetimelike_array(obj: ArrayLike) -> ArrayLike:
 400     """
 401     If we have an ndarray that is either datetime64 or timedelta64, wrap in EA.
 402 
 403     Parameters
 404     ----------
 405     obj : ndarray or ExtensionArray
 406 
 407     Returns
 408     -------
 409     ndarray or ExtensionArray
 410     """
 411     if isinstance(obj, np.ndarray):
 412         if obj.dtype.kind == "m":
 413             from pandas.core.arrays import TimedeltaArray
 414 
 415             return TimedeltaArray._from_sequence(obj)
 416         if obj.dtype.kind == "M":
 417             from pandas.core.arrays import DatetimeArray
 418 
 419             return DatetimeArray._from_sequence(obj)
 420 
 421     return obj

 424 def maybe_upcast_for_op(obj, shape: Tuple[int, ...]):
 425     """
 426     Cast non-pandas objects to pandas types to unify behavior of arithmetic
 427     and comparison operations.
 428 
 429     Parameters
 430     ----------
 431     obj: object
 432     shape : tuple[int]
 433 
 434     Returns
 435     -------
 436     out : object
 437 
 438     Notes
 439     -----
 440     Be careful to call this *after* determining the `name` attribute to be
 441     attached to the result of the arithmetic operation.
 442     """
 443     from pandas.core.arrays import DatetimeArray, TimedeltaArray
 444 
 445     if type(obj) is timedelta:
 446         # GH#22390  cast up to Timedelta to rely on Timedelta
 447         # implementation; otherwise operation against numeric-dtype
 448         # raises TypeError
 449         return Timedelta(obj)
 450     elif isinstance(obj, np.datetime64):
 451         # GH#28080 numpy casts integer-dtype to datetime64 when doing
 452         #  array[int] + datetime64, which we do not allow
 453         if isna(obj):
 454             # Avoid possible ambiguities with pd.NaT
 455             obj = obj.astype("datetime64[ns]")
 456             right = np.broadcast_to(obj, shape)
 457             return DatetimeArray(right)
 458 
 459         return Timestamp(obj)
 460 
 461     elif isinstance(obj, np.timedelta64):
 462         if isna(obj):
 463             # wrapping timedelta64("NaT") in Timedelta returns NaT,
 464             #  which would incorrectly be treated as a datetime-NaT, so
 465             #  we broadcast and wrap in a TimedeltaArray
 466             obj = obj.astype("timedelta64[ns]")
 467             right = np.broadcast_to(obj, shape)
 468             return TimedeltaArray(right)
 469 
 470         # In particular non-nanosecond timedelta64 needs to be cast to
 471         #  nanoseconds, or else we get undesired behavior like
 472         #  np.timedelta64(3, 'D') / 2 == np.timedelta64(1, 'D')
 473         return Timedelta(obj)
 474 
 475     elif isinstance(obj, np.ndarray) and obj.dtype.kind == "m":
 476         # GH#22390 Unfortunately we need to special-case right-hand
 477         # timedelta64 dtypes because numpy casts integer dtypes to
 478         # timedelta64 when operating with timedelta64
 479         return TimedeltaArray._from_sequence(obj)
 480     return obj
