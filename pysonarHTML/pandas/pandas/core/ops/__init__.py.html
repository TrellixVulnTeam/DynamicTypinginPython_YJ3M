  87 def get_op_result_name(left, right):
  88     """
  89     Find the appropriate name to pin to an operation result.  This result
  90     should always be either an Index or a Series.
  91 
  92     Parameters
  93     ----------
  94     left : {Series, Index}
  95     right : object
  96 
  97     Returns
  98     -------
  99     name : object
 100         Usually a string
 101     """
 102     # `left` is always a Series when called from within ops
 103     if isinstance(right, (ABCSeries, ABCIndexClass)):
 104         name = _maybe_match_name(left, right)
 105     else:
 106         name = left.name
 107     return name

 110 def _maybe_match_name(a, b):
 111     """
 112     Try to find a name to attach to the result of an operation between
 113     a and b.  If only one of these has a `name` attribute, return that
 114     name.  Otherwise return a consensus name if they match of None if
 115     they have different names.
 116 
 117     Parameters
 118     ----------
 119     a : object
 120     b : object
 121 
 122     Returns
 123     -------
 124     name : str or None
 125 
 126     See Also
 127     --------
 128     pandas.core.common.consensus_name_attr
 129     """
 130     a_has = hasattr(a, "name")
 131     b_has = hasattr(b, "name")
 132     if a_has and b_has:
 133         if a.name == b.name:
 134             return a.name
 135         else:
 136             # TODO: what if they both have np.nan for their names?
 137             return None
 138     elif a_has:
 139         return a.name
 140     elif b_has:
 141         return b.name
 142     return None

 148 def _get_frame_op_default_axis(name: str) -> Optional[str]:
 149     """
 150     Only DataFrame cares about default_axis, specifically:
 151     special methods have default_axis=None and flex methods
 152     have default_axis='columns'.
 153 
 154     Parameters
 155     ----------
 156     name : str
 157 
 158     Returns
 159     -------
 160     default_axis: str or None
 161     """
 162     if name.replace("__r", "__") in ["__and__", "__or__", "__xor__"]:
 163         # bool methods
 164         return "columns"
 165     elif name.startswith("__"):
 166         # __add__, __mul__, ...
 167         return None
 168     else:
 169         # add, mul, ...
 170         return "columns"

 197 def fill_binop(left, right, fill_value):
 198     """
 199     If a non-None fill_value is given, replace null entries in left and right
 200     with this value, but only in positions where _one_ of left/right is null,
 201     not both.
 202 
 203     Parameters
 204     ----------
 205     left : array-like
 206     right : array-like
 207     fill_value : object
 208 
 209     Returns
 210     -------
 211     left : array-like
 212     right : array-like
 213 
 214     Notes
 215     -----
 216     Makes copies if fill_value is not None and NAs are present.
 217     """
 218     if fill_value is not None:
 219         left_mask = isna(left)
 220         right_mask = isna(right)
 221 
 222         # one but not both
 223         mask = left_mask ^ right_mask
 224 
 225         if left_mask.any():
 226             # Avoid making a copy if we can
 227             left = left.copy()
 228             left[left_mask & mask] = fill_value
 229 
 230         if right_mask.any():
 231             # Avoid making a copy if we can
 232             right = right.copy()
 233             right[right_mask & mask] = fill_value
 234 
 235     return left, right

 242 def dispatch_to_series(left, right, func, axis: Optional[int] = None):
 243     """
 244     Evaluate the frame operation func(left, right) by evaluating
 245     column-by-column, dispatching to the Series implementation.
 246 
 247     Parameters
 248     ----------
 249     left : DataFrame
 250     right : scalar, Series, or DataFrame
 251     func : arithmetic or comparison operator
 252     axis : {None, 0, 1}
 253 
 254     Returns
 255     -------
 256     DataFrame
 257     """
 258     # Get the appropriate array-op to apply to each column/block's values.
 259     array_op = get_array_op(func)
 260 
 261     right = lib.item_from_zerodim(right)
 262     if not is_list_like(right):
 263         # i.e. scalar, faster than checking np.ndim(right) == 0
 264         bm = left._mgr.apply(array_op, right=right)
 265         return type(left)(bm)
 266 
 267     elif isinstance(right, ABCDataFrame):
 268         assert left.index.equals(right.index)
 269         assert left.columns.equals(right.columns)
 270         # TODO: The previous assertion `assert right._indexed_same(left)`
 271         #  fails in cases with empty columns reached via
 272         #  _frame_arith_method_with_reindex
 273 
 274         bm = left._mgr.operate_blockwise(right._mgr, array_op)
 275         return type(left)(bm)
 276 
 277     elif isinstance(right, ABCSeries) and axis == 1:
 278         # axis=1 means we want to operate row-by-row
 279         assert right.index.equals(left.columns)
 280 
 281         right = right._values
 282         # maybe_align_as_frame ensures we do not have an ndarray here
 283         assert not isinstance(right, np.ndarray)
 284 
 285         arrays = [array_op(l, r) for l, r in zip(left._iter_column_arrays(), right)]
 286 
 287     elif isinstance(right, ABCSeries):
 288         assert right.index.equals(left.index)  # Handle other cases later
 289         right = right._values
 290 
 291         arrays = [array_op(l, right) for l in left._iter_column_arrays()]
 292 
 293     else:
 294         # Remaining cases have less-obvious dispatch rules
 295         raise NotImplementedError(right)
 296 
 297     return type(left)._from_arrays(
 298         arrays, left.columns, left.index, verify_integrity=False
 299     )

 306 def _align_method_SERIES(left: "Series", right, align_asobject: bool = False):
 307     """ align lhs and rhs Series """
 308     # ToDo: Different from _align_method_FRAME, list, tuple and ndarray
 309     # are not coerced here
 310     # because Series has inconsistencies described in #13637
 311 
 312     if isinstance(right, ABCSeries):
 313         # avoid repeated alignment
 314         if not left.index.equals(right.index):
 315 
 316             if align_asobject:
 317                 # to keep original value's dtype for bool ops
 318                 left = left.astype(object)
 319                 right = right.astype(object)
 320 
 321             left, right = left.align(right, copy=False)
 322 
 323     return left, right

 326 def _arith_method_SERIES(cls, op, special):
 327     """
 328     Wrapper function for Series arithmetic operations, to avoid
 329     code duplication.
 330     """
 331     assert special  # non-special uses _flex_method_SERIES
 332     op_name = _get_op_name(op, special)
 333 
 334     @unpack_zerodim_and_defer(op_name)
 335     def wrapper(left, right):
 336 
 337         left, right = _align_method_SERIES(left, right)
 338         res_name = get_op_result_name(left, right)
 339 
 340         lvalues = extract_array(left, extract_numpy=True)
 341         rvalues = extract_array(right, extract_numpy=True)
 342         result = arithmetic_op(lvalues, rvalues, op)
 343 
 344         return left._construct_result(result, name=res_name)
 345 
 346     wrapper.__name__ = op_name
 347     return wrapper

 350 def _comp_method_SERIES(cls, op, special):
 351     """
 352     Wrapper function for Series arithmetic operations, to avoid
 353     code duplication.
 354     """
 355     assert special  # non-special uses _flex_method_SERIES
 356     op_name = _get_op_name(op, special)
 357 
 358     @unpack_zerodim_and_defer(op_name)
 359     def wrapper(self, other):
 360 
 361         res_name = get_op_result_name(self, other)
 362 
 363         if isinstance(other, ABCSeries) and not self._indexed_same(other):
 364             raise ValueError("Can only compare identically-labeled Series objects")

 367         rvalues = extract_array(other, extract_numpy=True)
 368 
 369         res_values = comparison_op(lvalues, rvalues, op)
 370 
 371         return self._construct_result(res_values, name=res_name)

 400 def _flex_method_SERIES(cls, op, special):
 401     assert not special  # "special" also means "not flex"
 402     name = _get_op_name(op, special)
 403     doc = _make_flex_doc(name, "series")
 404 
 405     @Appender(doc)
 406     def flex_wrapper(self, other, level=None, fill_value=None, axis=0):
 407         # validate axis
 408         if axis is not None:
 409             self._get_axis_number(axis)
 410 
 411         if isinstance(other, ABCSeries):
 412             return self._binop(other, op, level=level, fill_value=fill_value)
 413         elif isinstance(other, (np.ndarray, list, tuple)):
 414             if len(other) != len(self):
 415                 raise ValueError("Lengths must be equal")
 416             other = self._constructor(other, self.index)
 417             return self._binop(other, op, level=level, fill_value=fill_value)
 418         else:
 419             if fill_value is not None:
 420                 self = self.fillna(fill_value)
 421 
 422             return op(self, other)
 423 
 424     flex_wrapper.__name__ = name
 425     return flex_wrapper

 432 def _align_method_FRAME(
 433     left, right, axis, flex: Optional[bool] = False, level: Level = None
 434 ):
 435     """
 436     Convert rhs to meet lhs dims if input is list, tuple or np.ndarray.
 437 
 438     Parameters
 439     ----------
 440     left : DataFrame
 441     right : Any
 442     axis: int, str, or None
 443     flex: bool or None, default False
 444         Whether this is a flex op, in which case we reindex.
 445         None indicates not to check for alignment.
 446     level : int or level name, default None
 447 
 448     Returns
 449     -------
 450     left : DataFrame
 451     right : Any
 452     """
 453 
 454     def to_series(right):
 455         msg = "Unable to coerce to Series, length must be {req_len}: given {given_len}"
 456         if axis is not None and left._get_axis_name(axis) == "index":
 457             if len(left.index) != len(right):
 458                 raise ValueError(
 459                     msg.format(req_len=len(left.index), given_len=len(right))
 460                 )
 461             right = left._constructor_sliced(right, index=left.index)
 462         else:
 463             if len(left.columns) != len(right):
 464                 raise ValueError(
 465                     msg.format(req_len=len(left.columns), given_len=len(right))
 466                 )
 467             right = left._constructor_sliced(right, index=left.columns)
 468         return right
 469 
 470     if isinstance(right, np.ndarray):
 471 
 472         if right.ndim == 1:
 473             right = to_series(right)
 474 
 475         elif right.ndim == 2:
 476             if right.shape == left.shape:
 477                 right = left._constructor(right, index=left.index, columns=left.columns)
 478 
 479             elif right.shape[0] == left.shape[0] and right.shape[1] == 1:
 480                 # Broadcast across columns
 481                 right = np.broadcast_to(right, left.shape)
 482                 right = left._constructor(right, index=left.index, columns=left.columns)
 483 
 484             elif right.shape[1] == left.shape[1] and right.shape[0] == 1:
 485                 # Broadcast along rows
 486                 right = to_series(right[0, :])
 487 
 488             else:
 489                 raise ValueError(
 490                     "Unable to coerce to DataFrame, shape "
 491                     f"must be {left.shape}: given {right.shape}"
 492                 )
 493 
 494         elif right.ndim > 2:
 495             raise ValueError(
 496                 "Unable to coerce to Series/DataFrame, "
 497                 f"dimension must be <= 2: {right.shape}"
 498             )
 499 
 500     elif is_list_like(right) and not isinstance(right, (ABCSeries, ABCDataFrame)):
 501         # GH17901
 502         right = to_series(right)
 503 
 504     if flex is not None and isinstance(right, ABCDataFrame):
 505         if not left._indexed_same(right):
 506             if flex:
 507                 left, right = left.align(right, join="outer", level=level, copy=False)
 508             else:
 509                 raise ValueError(
 510                     "Can only compare identically-labeled DataFrame objects"
 511                 )
 512     elif isinstance(right, ABCSeries):
 513         # axis=1 is default for DataFrame-with-Series op
 514         axis = left._get_axis_number(axis) if axis is not None else 1
 515         left, right = left.align(
 516             right, join="outer", axis=axis, level=level, copy=False
 517         )
 518         right = _maybe_align_series_as_frame(left, right, axis)
 519 
 520     return left, right

 523 def _should_reindex_frame_op(
 524     left: "DataFrame", right, op, axis, default_axis, fill_value, level
 525 ) -> bool:
 526     """
 527     Check if this is an operation between DataFrames that will need to reindex.
 528     """
 529     assert isinstance(left, ABCDataFrame)
 530 
 531     if op is operator.pow or op is rpow:
 532         # GH#32685 pow has special semantics for operating with null values
 533         return False
 534 
 535     if not isinstance(right, ABCDataFrame):
 536         return False
 537 
 538     if fill_value is None and level is None and axis is default_axis:
 539         # TODO: any other cases we should handle here?
 540         cols = left.columns.intersection(right.columns)
 541         if not (cols.equals(left.columns) and cols.equals(right.columns)):
 542             return True
 543 
 544     return False

 547 def _frame_arith_method_with_reindex(
 548     left: "DataFrame", right: "DataFrame", op
 549 ) -> "DataFrame":
 550     """
 551     For DataFrame-with-DataFrame operations that require reindexing,
 552     operate only on shared columns, then reindex.
 553 
 554     Parameters
 555     ----------
 556     left : DataFrame
 557     right : DataFrame
 558     op : binary operator
 559 
 560     Returns
 561     -------
 562     DataFrame
 563     """
 564     # GH#31623, only operate on shared columns
 565     cols = left.columns.intersection(right.columns)
 566 
 567     new_left = left[cols]
 568     new_right = right[cols]
 569     result = op(new_left, new_right)
 570 
 571     # Do the join on the columns instead of using _align_method_FRAME
 572     #  to avoid constructing two potentially large/sparse DataFrames
 573     join_columns, _, _ = left.columns.join(
 574         right.columns, how="outer", level=None, return_indexers=True
 575     )
 576     return result.reindex(join_columns, axis=1)

 579 def _maybe_align_series_as_frame(frame: "DataFrame", series: "Series", axis: int):
 580     """
 581     If the Series operand is not EA-dtype, we can broadcast to 2D and operate
 582     blockwise.
 583     """
 584     rvalues = series._values
 585     if not isinstance(rvalues, np.ndarray):
 586         # TODO(EA2D): no need to special-case with 2D EAs
 587         if rvalues.dtype == "datetime64[ns]" or rvalues.dtype == "timedelta64[ns]":
 588             # We can losslessly+cheaply cast to ndarray
 589             rvalues = np.asarray(rvalues)
 590         else:
 591             return series
 592 
 593     if axis == 0:
 594         rvalues = rvalues.reshape(-1, 1)
 595     else:
 596         rvalues = rvalues.reshape(1, -1)
 597 
 598     rvalues = np.broadcast_to(rvalues, frame.shape)
 599     return type(frame)(rvalues, index=frame.index, columns=frame.columns)

 602 def _arith_method_FRAME(cls: Type["DataFrame"], op, special: bool):
 603     # This is the only function where `special` can be either True or False
 604     op_name = _get_op_name(op, special)
 605     default_axis = _get_frame_op_default_axis(op_name)
 606 
 607     na_op = get_array_op(op)
 608 
 609     if op_name in _op_descriptions:
 610         # i.e. include "add" but not "__add__"
 611         doc = _make_flex_doc(op_name, "dataframe")
 612     else:
 613         doc = _arith_doc_FRAME % op_name
 614 
 615     @Appender(doc)
 616     def f(self, other, axis=default_axis, level=None, fill_value=None):
 617 
 618         if _should_reindex_frame_op(
 619             self, other, op, axis, default_axis, fill_value, level
 620         ):
 621             return _frame_arith_method_with_reindex(self, other, op)
 622 
 623         if isinstance(other, ABCSeries) and fill_value is not None:
 624             # TODO: We could allow this in cases where we end up going
 625             #  through the DataFrame path
 626             raise NotImplementedError(f"fill_value {fill_value} not supported.")
 627 
 628         axis = self._get_axis_number(axis) if axis is not None else 1
 629 
 630         # TODO: why are we passing flex=True instead of flex=not special?
 631         #  15 tests fail if we pass flex=not special instead
 632         self, other = _align_method_FRAME(self, other, axis, flex=True, level=level)
 633 
 634         if isinstance(other, ABCDataFrame):
 635             # Another DataFrame
 636             new_data = self._combine_frame(other, na_op, fill_value)
 637 
 638         elif isinstance(other, ABCSeries):
 639             new_data = dispatch_to_series(self, other, op, axis=axis)
 640         else:
 641             # in this case we always have `np.ndim(other) == 0`
 642             if fill_value is not None:
 643                 self = self.fillna(fill_value)
 644 
 645             new_data = dispatch_to_series(self, other, op)
 646 
 647         return self._construct_result(new_data)
 648 
 649     f.__name__ = op_name
 650 
 651     return f

 654 def _flex_comp_method_FRAME(cls: Type["DataFrame"], op, special: bool):
 655     assert not special  # "special" also means "not flex"
 656     op_name = _get_op_name(op, special)
 657     default_axis = _get_frame_op_default_axis(op_name)
 658     assert default_axis == "columns", default_axis  # because we are not "special"
 659 
 660     doc = _flex_comp_doc_FRAME.format(
 661         op_name=op_name, desc=_op_descriptions[op_name]["desc"]
 662     )
 663 
 664     @Appender(doc)
 665     def f(self, other, axis=default_axis, level=None):
 666         axis = self._get_axis_number(axis) if axis is not None else 1
 667 
 668         self, other = _align_method_FRAME(self, other, axis, flex=True, level=level)
 669 
 670         new_data = dispatch_to_series(self, other, op, axis=axis)
 671         return self._construct_result(new_data)
 672 
 673     f.__name__ = op_name
 674 
 675     return f

 678 def _comp_method_FRAME(cls: Type["DataFrame"], op, special: bool):
 679     assert special  # "special" also means "not flex"
 680     op_name = _get_op_name(op, special)
 681 
 682     @Appender(f"Wrapper for comparison method {op_name}")
 683     def f(self, other):
 684         axis = 1  # only relevant for Series other case
 685 
 686         self, other = _align_method_FRAME(self, other, axis, level=None, flex=False)
 687 
 688         # See GH#4537 for discussion of scalar op behavior
 689         new_data = dispatch_to_series(self, other, op, axis=axis)
 690         return self._construct_result(new_data)
 691 
 692     f.__name__ = op_name
 693 
 694     return f
