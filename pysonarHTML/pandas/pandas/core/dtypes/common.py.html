  72 def ensure_float(arr):
  73     """
  74     Ensure that an array object has a float dtype if possible.
  75 
  76     Parameters
  77     ----------
  78     arr : array-like
  79         The array whose data type we want to enforce as float.
  80 
  81     Returns
  82     -------
  83     float_arr : The original array cast to the float dtype if
  84                 possible. Otherwise, the original array is returned.
  85     """
  86     if issubclass(arr.dtype.type, (np.integer, np.bool_)):
  87         arr = arr.astype(float)
  88     return arr

 100 def ensure_str(value: Union[bytes, Any]) -> str:
 101     """
 102     Ensure that bytes and non-strings get converted into ``str`` objects.
 103     """
 104     if isinstance(value, bytes):
 105         value = value.decode("utf-8")
 106     elif not isinstance(value, str):
 107         value = str(value)
 108     return value

 111 def ensure_int_or_float(arr: ArrayLike, copy: bool = False) -> np.array:
 112     """
 113     Ensure that an dtype array of some integer dtype
 114     has an int64 dtype if possible.
 115     If it's not possible, potentially because of overflow,
 116     convert the array to float64 instead.
 117 
 118     Parameters
 119     ----------
 120     arr : array-like
 121           The array whose data type we want to enforce.
 122     copy: bool
 123           Whether to copy the original array or reuse
 124           it in place, if possible.
 125 
 126     Returns
 127     -------
 128     out_arr : The input array cast as int64 if
 129               possible without overflow.
 130               Otherwise the input array cast to float64.
 131 
 132     Notes
 133     -----
 134     If the array is explicitly of type uint64 the type
 135     will remain unchanged.
 136     """
 137     # TODO: GH27506 potential bug with ExtensionArrays
 138     try:
 139         return arr.astype("int64", copy=copy, casting="safe")  # type: ignore
 140     except TypeError:
 141         pass
 142     try:
 143         return arr.astype("uint64", copy=copy, casting="safe")  # type: ignore
 144     except TypeError:
 145         if is_extension_array_dtype(arr.dtype):
 146             return arr.to_numpy(dtype="float64", na_value=np.nan)
 147         return arr.astype("float64", copy=copy)

 150 def ensure_python_int(value: Union[int, np.integer]) -> int:
 151     """
 152     Ensure that a value is a python int.
 153 
 154     Parameters
 155     ----------
 156     value: int or numpy.integer
 157 
 158     Returns
 159     -------
 160     int
 161 
 162     Raises
 163     ------
 164     TypeError: if the value isn't an int or can't be converted to one.
 165     """
 166     if not is_scalar(value):
 167         raise TypeError(
 168             f"Value needs to be a scalar value, was type {type(value).__name__}"
 169         )
 170     try:
 171         new_value = int(value)
 172         assert new_value == value
 173     except (TypeError, ValueError, AssertionError) as err:
 174         raise TypeError(f"Wrong type {type(value)} for value {value}") from err
 175     return new_value

 183 def classes_and_not_datetimelike(*klasses) -> Callable:
 184     """
 185     evaluate if the tipo is a subclass of the klasses
 186     and not a datetimelike
 187     """
 188     return lambda tipo: (
 189         issubclass(tipo, klasses)
 190         and not issubclass(tipo, (np.datetime64, np.timedelta64))

 224 def is_sparse(arr) -> bool:
 225     """
 226     Check whether an array-like is a 1-D pandas sparse array.
 227 
 228     Check that the one-dimensional array-like is a pandas sparse array.
 229     Returns True if it is a pandas sparse array, not another type of
 230     sparse array.
 231 
 232     Parameters
 233     ----------
 234     arr : array-like
 235         Array-like to check.
 236 
 237     Returns
 238     -------
 239     bool
 240         Whether or not the array-like is a pandas sparse array.
 241 
 242     Examples
 243     --------
 244     Returns `True` if the parameter is a 1-D pandas sparse array.
 245 
 246     >>> is_sparse(pd.arrays.SparseArray([0, 0, 1, 0]))
 247     True
 248     >>> is_sparse(pd.Series(pd.arrays.SparseArray([0, 0, 1, 0])))
 249     True
 250 
 251     Returns `False` if the parameter is not sparse.
 252 
 253     >>> is_sparse(np.array([0, 0, 1, 0]))
 254     False
 255     >>> is_sparse(pd.Series([0, 1, 0, 0]))
 256     False
 257 
 258     Returns `False` if the parameter is not a pandas sparse array.
 259 
 260     >>> from scipy.sparse import bsr_matrix
 261     >>> is_sparse(bsr_matrix([0, 1, 0, 0]))
 262     False
 263 
 264     Returns `False` if the parameter has more than one dimension.
 265     """
 266     from pandas.core.arrays.sparse import SparseDtype
 267 
 268     dtype = getattr(arr, "dtype", arr)
 269     return isinstance(dtype, SparseDtype)

 566 def is_string_dtype(arr_or_dtype) -> bool:
 567     """
 568     Check whether the provided array or dtype is of the string dtype.
 569 
 570     Parameters
 571     ----------
 572     arr_or_dtype : array-like
 573         The array or dtype to check.
 574 
 575     Returns
 576     -------
 577     boolean
 578         Whether or not the array or dtype is of the string dtype.
 579 
 580     Examples
 581     --------
 582     >>> is_string_dtype(str)
 583     True
 584     >>> is_string_dtype(object)
 585     True
 586     >>> is_string_dtype(int)
 587     False
 588     >>>
 589     >>> is_string_dtype(np.array(['a', 'b']))
 590     True
 591     >>> is_string_dtype(pd.Series([1, 2]))
 592     False
 593     """
 594     # TODO: gh-15585: consider making the checks stricter.
 595     def condition(dtype) -> bool:
 596         return dtype.kind in ("O", "S", "U") and not is_excluded_dtype(dtype)
 597 
 598     def is_excluded_dtype(dtype) -> bool:
 599         """
 600         These have kind = "O" but aren't string dtypes so need to be explicitly excluded
 601         """
 602         is_excluded_checks = (is_period_dtype, is_interval_dtype, is_categorical_dtype)
 603         return any(is_excluded(dtype) for is_excluded in is_excluded_checks)
 604 
 605     return _is_dtype(arr_or_dtype, condition)

 608 def is_dtype_equal(source, target) -> bool:
 609     """
 610     Check if two dtypes are equal.
 611 
 612     Parameters
 613     ----------
 614     source : The first dtype to compare
 615     target : The second dtype to compare
 616 
 617     Returns
 618     -------
 619     boolean
 620         Whether or not the two dtypes are equal.
 621 
 622     Examples
 623     --------
 624     >>> is_dtype_equal(int, float)
 625     False
 626     >>> is_dtype_equal("int", int)
 627     True
 628     >>> is_dtype_equal(object, "category")
 629     False
 630     >>> is_dtype_equal(CategoricalDtype(), "category")
 631     True
 632     >>> is_dtype_equal(DatetimeTZDtype(tz="UTC"), "datetime64")
 633     False
 634     """
 635     try:
 636         source = _get_dtype(source)
 637         target = _get_dtype(target)
 638         return source == target
 639     except (TypeError, AttributeError):
 640 
 641         # invalid comparison
 642         # object == category will hit this
 643         return False

1064 def is_numeric_v_string_like(a, b):
1065     """
1066     Check if we are comparing a string-like object to a numeric ndarray.
1067     NumPy doesn't like to compare such objects, especially numeric arrays
1068     and scalar string-likes.
1069 
1070     Parameters
1071     ----------
1072     a : array-like, scalar
1073         The first object to check.
1074     b : array-like, scalar
1075         The second object to check.
1076

1078     -------
1079     boolean
1080         Whether we return a comparing a string-like object to a numeric array.
1081 
1082     Examples
1083     --------

1085     False
1086     >>> is_numeric_v_string_like("foo", "foo")
1087     False
1088     >>> is_numeric_v_string_like(1, "foo")  # non-array numeric
1089     False

1091     True
1092     >>> is_numeric_v_string_like("foo", np.array([1]))  # symmetric check
1093     True

1102     """
1103     is_a_array = isinstance(a, np.ndarray)
1104     is_b_array = isinstance(b, np.ndarray)

1107     is_b_numeric_array = is_b_array and is_numeric_dtype(b)

1109     is_b_string_array = is_b_array and is_string_like_dtype(b)
1110 
1111     is_a_scalar_string_like = not is_a_array and isinstance(a, str)
1112     is_b_scalar_string_like = not is_b_array and isinstance(b, str)
1113 
1114     return (
1115         (is_a_numeric_array and is_b_scalar_string_like)
1116         or (is_b_numeric_array and is_a_scalar_string_like)

1118         or (is_b_numeric_array and is_a_string_array)

1403     classes represented within the pandas library and not ones external
1404     to it like scipy sparse matrices), and datetime-like arrays.
1405 
1406     Parameters
1407     ----------
1408     arr : array-like
1409         The array-like to check.
1410 
1411     Returns
1412     -------
1413     boolean
1414         Whether or not the array-like is of a pandas extension class instance.
1415 
1416     Examples
1417     --------
1418     >>> is_extension_type([1, 2, 3])
1419     False
1420     >>> is_extension_type(np.array([1, 2, 3]))
1421     False
1422     >>>
1423     >>> cat = pd.Categorical([1, 2, 3])
1424     >>>
1425     >>> is_extension_type(cat)
1426     True
1427     >>> is_extension_type(pd.Series(cat))
1428     True
1429     >>> is_extension_type(pd.arrays.SparseArray([1, 2, 3]))
1430     True
1431     >>> from scipy.sparse import bsr_matrix
1432     >>> is_extension_type(bsr_matrix([1, 2, 3]))
1433     False
1434     >>> is_extension_type(pd.DatetimeIndex([1, 2, 3]))
1435     False
1436     >>> is_extension_type(pd.DatetimeIndex([1, 2, 3], tz="US/Eastern"))
1437     True
1438     >>>
1439     >>> dtype = DatetimeTZDtype("ns", tz="US/Eastern")
1440     >>> s = pd.Series([], dtype=dtype)
1441     >>> is_extension_type(s)
1442     True
1443     """
1444     warnings.warn(
1445         "'is_extension_type' is deprecated and will be removed in a future "
1446         "version.  Use 'is_extension_array_dtype' instead.",
1447         FutureWarning,
1448         stacklevel=2,
1449     )
1450 
1451     if is_categorical_dtype(arr):
1452         return True
1453     elif is_sparse(arr):
1454         return True
1455     elif is_datetime64tz_dtype(arr):
1456         return True
1457     return False

1460 def is_extension_array_dtype(arr_or_dtype) -> bool:
1461     """
1462     Check if an object is a pandas extension array type.
1463

1465 
1466     Parameters
1467     ----------

1471 
1472     Returns
1473     -------

1485     * Period
1486     * DatetimeArray
1487     * TimedeltaArray

1541 def _is_dtype(arr_or_dtype, condition) -> bool:
1542     """
1543     Return a boolean if the condition is satisfied for the arr_or_dtype.
1544 
1545     Parameters
1546     ----------
1547     arr_or_dtype : array-like, str, np.dtype, or ExtensionArrayType
1548         The array-like or dtype object whose dtype we want to extract.

1550 
1551     Returns
1552     -------
1553     bool
1554 
1555     """
1556     if arr_or_dtype is None:
1557         return False
1558     try:
1559         dtype = _get_dtype(arr_or_dtype)
1560     except (TypeError, ValueError, UnicodeEncodeError):
1561         return False
1562     return condition(dtype)

1565 def _get_dtype(arr_or_dtype) -> DtypeObj:
1566     """
1567     Get the dtype instance associated with an array
1568     or dtype object.
1569 
1570     Parameters
1571     ----------
1572     arr_or_dtype : array-like
1573         The array-like or dtype object whose dtype we want to extract.
1574 
1575     Returns
1576     -------
1577     obj_dtype : The extract dtype instance from the
1578                 passed in array or dtype object.
1579 
1580     Raises
1581     ------
1582     TypeError : The passed in object is None.
1583     """
1584     if arr_or_dtype is None:
1585         raise TypeError("Cannot deduce dtype from null object")
1586 
1587     # fastpath
1588     elif isinstance(arr_or_dtype, np.dtype):
1589         return arr_or_dtype
1590     elif isinstance(arr_or_dtype, type):
1591         return np.dtype(arr_or_dtype)
1592

1594     elif hasattr(arr_or_dtype, "dtype"):
1595         arr_or_dtype = arr_or_dtype.dtype

1600 def _is_dtype_type(arr_or_dtype, condition) -> bool:
1601     """
1602     Return a boolean if the condition is satisfied for the arr_or_dtype.
1603 
1604     Parameters
1605     ----------
1606     arr_or_dtype : array-like
1607         The array-like or dtype object whose dtype we want to extract.
1608     condition : callable[Union[np.dtype, ExtensionDtypeType]]
1609 
1610     Returns
1611     -------
1612     bool : if the condition is satisfied for the arr_or_dtype
1613     """
1614     if arr_or_dtype is None:
1615         return condition(type(None))
1616 
1617     # fastpath
1618     if isinstance(arr_or_dtype, np.dtype):
1619         return condition(arr_or_dtype.type)
1620     elif isinstance(arr_or_dtype, type):
1621         if issubclass(arr_or_dtype, ExtensionDtype):
1622             arr_or_dtype = arr_or_dtype.type
1623         return condition(np.dtype(arr_or_dtype).type)
1624 
1625     # if we have an array-like
1626     if hasattr(arr_or_dtype, "dtype"):
1627         arr_or_dtype = arr_or_dtype.dtype
1628 
1629     # we are not possibly a dtype
1630     elif is_list_like(arr_or_dtype):
1631         return condition(type(None))
1632 
1633     try:
1634         tipo = pandas_dtype(arr_or_dtype).type
1635     except (TypeError, ValueError, UnicodeEncodeError):
1636         if is_scalar(arr_or_dtype):
1637             return condition(type(None))
1638 
1639         return False
1640 
1641     return condition(tipo)

1644 def infer_dtype_from_object(dtype):
1645     """
1646     Get a numpy dtype.type-style object for a dtype object.
1647 
1648     This methods also includes handling of the datetime64[ns] and
1649     datetime64[ns, TZ] objects.
1650 
1651     If no dtype can be found, we return ``object``.
1652 
1653     Parameters
1654     ----------
1655     dtype : dtype, type
1656         The dtype object whose numpy dtype.type-style
1657         object we want to extract.
1658 
1659     Returns
1660     -------
1661     dtype_object : The extracted numpy dtype.type-style object.
1662     """
1663     if isinstance(dtype, type) and issubclass(dtype, np.generic):
1664         # Type object from a dtype
1665         return dtype
1666     elif isinstance(dtype, (np.dtype, ExtensionDtype)):
1667         # dtype object
1668         try:
1669             _validate_date_like_dtype(dtype)
1670         except TypeError:
1671             # Should still pass if we don't have a date-like
1672             pass
1673         return dtype.type
1674 
1675     try:
1676         dtype = pandas_dtype(dtype)
1677     except TypeError:
1678         pass
1679 
1680     if is_extension_array_dtype(dtype):
1681         return dtype.type
1682     elif isinstance(dtype, str):
1683 
1684         # TODO(jreback)
1685         # should deprecate these
1686         if dtype in ["datetimetz", "datetime64tz"]:
1687             return DatetimeTZDtype.type
1688         elif dtype in ["period"]:
1689             raise NotImplementedError
1690 
1691         if dtype == "datetime" or dtype == "timedelta":
1692             dtype += "64"
1693         try:
1694             return infer_dtype_from_object(getattr(np, dtype))
1695         except (AttributeError, TypeError):
1696             # Handles cases like _get_dtype(int) i.e.,
1697             # Python objects that are valid dtypes
1698             # (unlike user-defined types, in general)
1699             #
1700             # TypeError handles the float16 type code of 'e'
1701             # further handle internal types
1702             pass
1703 
1704     return infer_dtype_from_object(np.dtype(dtype))
1705

1707 def _validate_date_like_dtype(dtype) -> None:
1708     """
1709     Check whether the dtype is a date-like dtype. Raises an error if invalid.
1710 
1711     Parameters
1712     ----------
1713     dtype : dtype, type
1714         The dtype to check.
1715 
1716     Raises
1717     ------
1718     TypeError : The dtype could not be casted to a date-like dtype.
1719     ValueError : The dtype is an illegal date-like dtype (e.g. the
1720                  the frequency provided is too specific)
1721     """
1722     try:
1723         typ = np.datetime_data(dtype)[0]
1724     except ValueError as e:
1725         raise TypeError(e) from e
1726     if typ != "generic" and typ != "ns":
1727         raise ValueError(
1728             f"{repr(dtype.name)} is too specific of a frequency, "
1729             f"try passing {repr(dtype.type.__name__)}"
1730         )
1731

1733 def pandas_dtype(dtype) -> DtypeObj:
1734     """
1735     Convert input into a pandas only dtype object or a numpy dtype object.
1736 
1737     Parameters
1738     ----------
1739     dtype : object to be converted
1740 
1741     Returns
1742     -------
1743     np.dtype or a pandas dtype
1744 
1745     Raises
1746     ------
1747     TypeError if not a dtype
1748     """
1749     # short-circuit
1750     if isinstance(dtype, np.ndarray):
1751         return dtype.dtype
1752     elif isinstance(dtype, (np.dtype, ExtensionDtype)):
1753         return dtype
1754 
1755     # registered extension types
1756     result = registry.find(dtype)
1757     if result is not None:
1758         return result
1759 
1760     # try a numpy dtype
1761     # raise a consistent TypeError if failed
1762     try:
1763         npdtype = np.dtype(dtype)
1764     except SyntaxError as err:
1765         # np.dtype uses `eval` which can raise SyntaxError
1766         raise TypeError(f"data type '{dtype}' not understood") from err
1767 
1768     # Any invalid dtype (such as pd.Timestamp) should raise an error.
1769     # np.dtype(invalid_type).kind = 0 for such objects. However, this will
1770     # also catch some valid dtypes such as object, np.object_ and 'object'
1771     # which we safeguard against by catching them earlier and returning
1772     # np.dtype(valid_dtype) before this condition is evaluated.
1773     if is_hashable(dtype) and dtype in [object, np.object_, "object", "O"]:
1774         # check hashability to avoid errors/DeprecationWarning when we get
1775         # here and `dtype` is an array
1776         return npdtype
1777     elif npdtype.kind == "O":
1778         raise TypeError(f"dtype '{dtype}' not understood")
1779 
1780     return npdtype
