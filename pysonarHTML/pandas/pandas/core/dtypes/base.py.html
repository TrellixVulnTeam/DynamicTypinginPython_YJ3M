  18 class ExtensionDtype:
  19     """
  20     A custom data type, to be paired with an ExtensionArray.
  21 
  22     .. versionadded:: 0.23.0
  23 
  24     See Also
  25     --------
  26     extensions.register_extension_dtype
  27     extensions.ExtensionArray
  28 
  29     Notes
  30     -----
  31     The interface includes the following abstract methods that must
  32     be implemented by subclasses:
  33 
  34     * type
  35     * name
  36 
  37     The following attributes and methods influence the behavior of the dtype in
  38     pandas operations
  39 
  40     * _is_numeric
  41     * _is_boolean
  42     * _get_common_dtype
  43 
  44     Optionally one can override construct_array_type for construction
  45     with the name of this dtype via the Registry. See
  46     :meth:`extensions.register_extension_dtype`.
  47 
  48     * construct_array_type
  49 
  50     The `na_value` class attribute can be used to set the default NA value
  51     for this type. :attr:`numpy.nan` is used by default.
  52 
  53     ExtensionDtypes are required to be hashable. The base class provides
  54     a default implementation, which relies on the ``_metadata`` class
  55     attribute. ``_metadata`` should be a tuple containing the strings
  56     that define your data type. For example, with ``PeriodDtype`` that's
  57     the ``freq`` attribute.
  58 
  59     **If you have a parametrized dtype you should set the ``_metadata``
  60     class property**.
  61 
  62     Ideally, the attributes in ``_metadata`` will match the
  63     parameters to your ``ExtensionDtype.__init__`` (if any). If any of
  64     the attributes in ``_metadata`` don't implement the standard
  65     ``__eq__`` or ``__hash__``, the default implementations here will not
  66     work.
  67 
  68     .. versionchanged:: 0.24.0
  69 
  70        Added ``_metadata``, ``__hash__``, and changed the default definition
  71        of ``__eq__``.
  72 
  73     For interaction with Apache Arrow (pyarrow), a ``__from_arrow__`` method
  74     can be implemented: this method receives a pyarrow Array or ChunkedArray
  75     as only argument and is expected to return the appropriate pandas
  76     ExtensionArray for this dtype and the passed values::
  77 
  78         class ExtensionDtype:
  79 
  80             def __from_arrow__(
  81                 self, array: Union[pyarrow.Array, pyarrow.ChunkedArray]
  82             ) -> ExtensionArray:
  83                 ...
  84 
  85     This class does not inherit from 'abc.ABCMeta' for performance reasons.
  86     Methods and properties required by the interface raise
  87     ``pandas.errors.AbstractMethodError`` and no ``register`` method is
  88     provided for registering virtual subclasses.
  89     """
  90 
  91     _metadata: Tuple[str, ...] = ()
  92 
  93     def __str__(self) -> str:
  94         return self.name
  95 
  96     def __eq__(self, other: Any) -> bool:
  97         """
  98         Check whether 'other' is equal to self.
  99 
 100         By default, 'other' is considered equal if either
 101 
 102         * it's a string matching 'self.name'.
 103         * it's an instance of this type and all of the
 104           the attributes in ``self._metadata`` are equal between
 105           `self` and `other`.
 106 
 107         Parameters
 108         ----------
 109         other : Any
 110 
 111         Returns
 112         -------
 113         bool
 114         """
 115         if isinstance(other, str):
 116             try:
 117                 other = self.construct_from_string(other)
 118             except TypeError:
 119                 return False
 120         if isinstance(other, type(self)):
 121             return all(
 122                 getattr(self, attr) == getattr(other, attr) for attr in self._metadata
 123             )
 124         return False
 125 
 126     def __hash__(self) -> int:
 127         return hash(tuple(getattr(self, attr) for attr in self._metadata))
 128 
 129     def __ne__(self, other: Any) -> bool:
 130         return not self.__eq__(other)
 131 
 132     @property
 133     def na_value(self) -> object:
 134         """
 135         Default NA value to use for this type.
 136 
 137         This is used in e.g. ExtensionArray.take. This should be the
 138         user-facing "boxed" version of the NA value, not the physical NA value
 139         for storage.  e.g. for JSONArray, this is an empty dictionary.
 140         """
 141         return np.nan
 142 
 143     @property
 144     def type(self) -> Type:
 145         """
 146         The scalar type for the array, e.g. ``int``
 147 
 148         It's expected ``ExtensionArray[item]`` returns an instance
 149         of ``ExtensionDtype.type`` for scalar ``item``, assuming
 150         that value is valid (not NA). NA values do not need to be
 151         instances of `type`.
 152         """
 153         raise AbstractMethodError(self)
 154 
 155     @property
 156     def kind(self) -> str:
 157         """
 158         A character code (one of 'biufcmMOSUV'), default 'O'
 159 
 160         This should match the NumPy dtype used when the array is
 161         converted to an ndarray, which is probably 'O' for object if
 162         the extension type cannot be represented as a built-in NumPy
 163         type.
 164 
 165         See Also
 166         --------
 167         numpy.dtype.kind
 168         """
 169         return "O"
 170 
 171     @property
 172     def name(self) -> str:
 173         """
 174         A string identifying the data type.
 175 
 176         Will be used for display in, e.g. ``Series.dtype``
 177         """
 178         raise AbstractMethodError(self)
 179 
 180     @property
 181     def names(self) -> Optional[List[str]]:
 182         """
 183         Ordered list of field names, or None if there are no fields.
 184 
 185         This is for compatibility with NumPy arrays, and may be removed in the
 186         future.
 187         """
 188         return None

 190     @classmethod
 191     def construct_array_type(cls) -> Type["ExtensionArray"]:
 192         """
 193         Return the array type associated with this dtype.
 194 
 195         Returns
 196         -------
 197         type
 198         """
 199         raise NotImplementedError

 201     @classmethod
 202     def construct_from_string(cls, string: str):
 203         r"""
 204         Construct this type from a string.
 205 
 206         This is useful mainly for data types that accept parameters.
 207         For example, a period dtype accepts a frequency parameter that
 208         can be set as ``period[H]`` (where H means hourly frequency).
 209 
 210         By default, in the abstract class, just the name of the type is
 211         expected. But subclasses can overwrite this method to accept
 212         parameters.
 213 
 214         Parameters
 215         ----------
 216         string : str
 217             The name of the type, for example ``category``.
 218 
 219         Returns
 220         -------
 221         ExtensionDtype
 222             Instance of the dtype.
 223 
 224         Raises
 225         ------
 226         TypeError
 227             If a class cannot be constructed from this 'string'.
 228 
 229         Examples
 230         --------
 231         For extension dtypes with arguments the following may be an
 232         adequate implementation.
 233 
 234         >>> @classmethod
 235         ... def construct_from_string(cls, string):
 236         ...     pattern = re.compile(r"^my_type\[(?P<arg_name>.+)\]$")
 237         ...     match = pattern.match(string)
 238         ...     if match:
 239         ...         return cls(**match.groupdict())
 240         ...     else:
 241         ...         raise TypeError(
 242         ...             f"Cannot construct a '{cls.__name__}' from '{string}'"
 243         ...         )
 244         """
 245         if not isinstance(string, str):
 246             raise TypeError(
 247                 f"'construct_from_string' expects a string, got {type(string)}"
 248             )
 249         # error: Non-overlapping equality check (left operand type: "str", right
 250         #  operand type: "Callable[[ExtensionDtype], str]")  [comparison-overlap]
 251         assert isinstance(cls.name, str), (cls, type(cls.name))
 252         if string != cls.name:
 253             raise TypeError(f"Cannot construct a '{cls.__name__}' from '{string}'")
 254         return cls()
 255 
 256     @classmethod
 257     def is_dtype(cls, dtype: object) -> bool:
 258         """
 259         Check if we match 'dtype'.
 260 
 261         Parameters
 262         ----------
 263         dtype : object
 264             The object to check.
 265 
 266         Returns
 267         -------
 268         bool
 269 
 270         Notes
 271         -----
 272         The default implementation is True if
 273 
 274         1. ``cls.construct_from_string(dtype)`` is an instance
 275            of ``cls``.
 276         2. ``dtype`` is an object and is an instance of ``cls``
 277         3. ``dtype`` has a ``dtype`` attribute, and any of the above
 278            conditions is true for ``dtype.dtype``.
 279         """
 280         dtype = getattr(dtype, "dtype", dtype)
 281 
 282         if isinstance(dtype, (ABCSeries, ABCIndexClass, ABCDataFrame, np.dtype)):
 283             # https://github.com/pandas-dev/pandas/issues/22960
 284             # avoid passing data to `construct_from_string`. This could
 285             # cause a FutureWarning from numpy about failing elementwise
 286             # comparison from, e.g., comparing DataFrame == 'category'.
 287             return False
 288         elif dtype is None:
 289             return False
 290         elif isinstance(dtype, cls):
 291             return True
 292         if isinstance(dtype, str):
 293             try:
 294                 return cls.construct_from_string(dtype) is not None
 295             except TypeError:
 296                 return False
 297         return False
 298 
 299     @property
 300     def _is_numeric(self) -> bool:
 301         """
 302         Whether columns with this dtype should be considered numeric.
 303 
 304         By default ExtensionDtypes are assumed to be non-numeric.
 305         They'll be excluded from operations that exclude non-numeric
 306         columns, like (groupby) reductions, plotting, etc.
 307         """
 308         return False
 309 
 310     @property
 311     def _is_boolean(self) -> bool:
 312         """
 313         Whether this dtype should be considered boolean.
 314 
 315         By default, ExtensionDtypes are assumed to be non-numeric.
 316         Setting this to True will affect the behavior of several places,
 317         e.g.
 318 
 319         * is_bool
 320         * boolean indexing
 321 
 322         Returns
 323         -------
 324         bool
 325         """
 326         return False

 328     def _get_common_dtype(self, dtypes: List[DtypeObj]) -> Optional[DtypeObj]:
 329         """
 330         Return the common dtype, if one exists.
 331 
 332         Used in `find_common_type` implementation. This is for example used
 333         to determine the resulting dtype in a concat operation.
 334 
 335         If no common dtype exists, return None (which gives the other dtypes
 336         the chance to determine a common dtype). If all dtypes in the list
 337         return None, then the common dtype will be "object" dtype (this means
 338         it is never needed to return "object" dtype from this method itself).
 339 
 340         Parameters
 341         ----------
 342         dtypes : list of dtypes
 343             The dtypes for which to determine a common dtype. This is a list
 344             of np.dtype or ExtensionDtype instances.
 345 
 346         Returns
 347         -------
 348         Common dtype (np.dtype or ExtensionDtype) or None
 349         """
 350         if len(set(dtypes)) == 1:
 351             # only itself
 352             return self
 353         else:
 354             return None
