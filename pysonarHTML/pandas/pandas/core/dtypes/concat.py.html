  29 def get_dtype_kinds(l):
  30     """
  31     Parameters
  32     ----------
  33     l : list of arrays
  34 
  35     Returns
  36     -------
  37     a set of kinds that exist in this list of arrays
  38     """
  39     typs = set()
  40     for arr in l:
  41 
  42         dtype = arr.dtype
  43         if is_categorical_dtype(dtype):
  44             typ = "category"
  45         elif is_sparse(dtype):
  46             typ = "sparse"
  47         elif isinstance(arr, ABCRangeIndex):
  48             typ = "range"
  49         elif is_datetime64tz_dtype(dtype):
  50             # if to_concat contains different tz,
  51             # the result must be object dtype
  52             typ = str(dtype)
  53         elif is_datetime64_dtype(dtype):
  54             typ = "datetime"
  55         elif is_timedelta64_dtype(dtype):
  56             typ = "timedelta"
  57         elif is_object_dtype(dtype):
  58             typ = "object"
  59         elif is_bool_dtype(dtype):
  60             typ = "bool"
  61         elif is_extension_array_dtype(dtype):
  62             typ = str(dtype)
  63         else:
  64             typ = dtype.kind
  65         typs.add(typ)
  66     return typs

  69 def _cast_to_common_type(arr: ArrayLike, dtype: DtypeObj) -> ArrayLike:
  70     """
  71     Helper function for `arr.astype(common_dtype)` but handling all special
  72     cases.
  73     """
  74     if (
  75         is_categorical_dtype(arr.dtype)
  76         and isinstance(dtype, np.dtype)
  77         and np.issubdtype(dtype, np.integer)
  78     ):
  79         # problem case: categorical of int -> gives int as result dtype,
  80         # but categorical can contain NAs -> fall back to object dtype
  81         try:
  82             return arr.astype(dtype, copy=False)
  83         except ValueError:
  84             return arr.astype(object, copy=False)
  85 
  86     if is_sparse(arr) and not is_sparse(dtype):
  87         # problem case: SparseArray.astype(dtype) doesn't follow the specified
  88         # dtype exactly, but converts this to Sparse[dtype] -> first manually
  89         # convert to dense array
  90         arr = cast(SparseArray, arr)
  91         return arr.to_dense().astype(dtype, copy=False)
  92 
  93     if (
  94         isinstance(arr, np.ndarray)
  95         and arr.dtype.kind in ["m", "M"]
  96         and dtype is np.dtype("object")
  97     ):
  98         # wrap datetime-likes in EA to ensure astype(object) gives Timestamp/Timedelta
  99         # this can happen when concat_compat is called directly on arrays (when arrays
 100         # are not coming from Index/Series._values), eg in BlockManager.quantile
 101         arr = array(arr)
 102 
 103     if is_extension_array_dtype(dtype):
 104         if isinstance(arr, np.ndarray):
 105             # numpy's astype cannot handle ExtensionDtypes
 106             return array(arr, dtype=dtype, copy=False)
 107     return arr.astype(dtype, copy=False)

 110 def concat_compat(to_concat, axis: int = 0):
 111     """
 112     provide concatenation of an array of arrays each of which is a single
 113     'normalized' dtypes (in that for example, if it's object, then it is a
 114     non-datetimelike and provide a combined dtype for the resulting array that
 115     preserves the overall dtype if possible)
 116 
 117     Parameters
 118     ----------
 119     to_concat : array of arrays
 120     axis : axis to provide concatenation
 121 
 122     Returns
 123     -------
 124     a single array, preserving the combined dtypes
 125     """
 126     # filter empty arrays
 127     # 1-d dtypes always are included here
 128     def is_nonempty(x) -> bool:
 129         if x.ndim <= axis:
 130             return True
 131         return x.shape[axis] > 0
 132 
 133     # If all arrays are empty, there's nothing to convert, just short-cut to
 134     # the concatenation, #3121.
 135     #
 136     # Creating an empty array directly is tempting, but the winnings would be
 137     # marginal given that it would still require shape & dtype calculation and
 138     # np.concatenate which has them both implemented is compiled.
 139     non_empties = [x for x in to_concat if is_nonempty(x)]
 140     if non_empties and axis == 0:
 141         to_concat = non_empties
 142 
 143     typs = get_dtype_kinds(to_concat)
 144     _contains_datetime = any(typ.startswith("datetime") for typ in typs)
 145 
 146     all_empty = not len(non_empties)
 147     single_dtype = len({x.dtype for x in to_concat}) == 1
 148     any_ea = any(is_extension_array_dtype(x.dtype) for x in to_concat)
 149 
 150     if any_ea and axis == 0:
 151         if not single_dtype:
 152             target_dtype = find_common_type([x.dtype for x in to_concat])
 153             to_concat = [_cast_to_common_type(arr, target_dtype) for arr in to_concat]
 154 
 155         if isinstance(to_concat[0], ExtensionArray):
 156             cls = type(to_concat[0])
 157             return cls._concat_same_type(to_concat)
 158         else:
 159             return np.concatenate(to_concat)
 160 
 161     elif _contains_datetime or "timedelta" in typs:
 162         return concat_datetime(to_concat, axis=axis, typs=typs)
 163 
 164     elif any_ea and axis == 1:
 165         to_concat = [np.atleast_2d(x.astype("object")) for x in to_concat]
 166         return np.concatenate(to_concat, axis=axis)
 167 
 168     elif all_empty:
 169         # we have all empties, but may need to coerce the result dtype to
 170         # object if we have non-numeric type operands (numpy would otherwise
 171         # cast this to float)
 172         typs = get_dtype_kinds(to_concat)
 173         if len(typs) != 1:
 174 
 175             if not len(typs - {"i", "u", "f"}) or not len(typs - {"bool", "i", "u"}):
 176                 # let numpy coerce
 177                 pass
 178             else:
 179                 # coerce to object
 180                 to_concat = [x.astype("object") for x in to_concat]
 181 
 182     return np.concatenate(to_concat, axis=axis)

 185 def union_categoricals(
 186     to_union, sort_categories: bool = False, ignore_order: bool = False
 187 ):
 188     """
 189     Combine list-like of Categorical-like, unioning categories.
 190 
 191     All categories must have the same dtype.
 192 
 193     Parameters
 194     ----------
 195     to_union : list-like
 196         Categorical, CategoricalIndex, or Series with dtype='category'.
 197     sort_categories : bool, default False
 198         If true, resulting categories will be lexsorted, otherwise
 199         they will be ordered as they appear in the data.
 200     ignore_order : bool, default False
 201         If true, the ordered attribute of the Categoricals will be ignored.
 202         Results in an unordered categorical.
 203 
 204     Returns
 205     -------
 206     Categorical
 207 
 208     Raises
 209     ------
 210     TypeError
 211         - all inputs do not have the same dtype
 212         - all inputs do not have the same ordered property
 213         - all inputs are ordered and their categories are not identical
 214         - sort_categories=True and Categoricals are ordered
 215     ValueError
 216         Empty list of categoricals passed
 217 
 218     Notes
 219     -----
 220     To learn more about categories, see `link
 221     <https://pandas.pydata.org/pandas-docs/stable/user_guide/categorical.html#unioning>`__
 222 
 223     Examples
 224     --------
 225     >>> from pandas.api.types import union_categoricals
 226 
 227     If you want to combine categoricals that do not necessarily have
 228     the same categories, `union_categoricals` will combine a list-like
 229     of categoricals. The new categories will be the union of the
 230     categories being combined.
 231 
 232     >>> a = pd.Categorical(["b", "c"])
 233     >>> b = pd.Categorical(["a", "b"])
 234     >>> union_categoricals([a, b])
 235     [b, c, a, b]
 236     Categories (3, object): [b, c, a]
 237 
 238     By default, the resulting categories will be ordered as they appear
 239     in the `categories` of the data. If you want the categories to be
 240     lexsorted, use `sort_categories=True` argument.
 241 
 242     >>> union_categoricals([a, b], sort_categories=True)
 243     [b, c, a, b]
 244     Categories (3, object): [a, b, c]
 245 
 246     `union_categoricals` also works with the case of combining two
 247     categoricals of the same categories and order information (e.g. what
 248     you could also `append` for).
 249 
 250     >>> a = pd.Categorical(["a", "b"], ordered=True)
 251     >>> b = pd.Categorical(["a", "b", "a"], ordered=True)
 252     >>> union_categoricals([a, b])
 253     [a, b, a, b, a]
 254     Categories (2, object): [a < b]
 255 
 256     Raises `TypeError` because the categories are ordered and not identical.
 257 
 258     >>> a = pd.Categorical(["a", "b"], ordered=True)
 259     >>> b = pd.Categorical(["a", "b", "c"], ordered=True)
 260     >>> union_categoricals([a, b])
 261     Traceback (most recent call last):
 262         ...
 263     TypeError: to union ordered Categoricals, all categories must be the same
 264 
 265     New in version 0.20.0
 266 
 267     Ordered categoricals with different categories or orderings can be
 268     combined by using the `ignore_ordered=True` argument.
 269 
 270     >>> a = pd.Categorical(["a", "b", "c"], ordered=True)
 271     >>> b = pd.Categorical(["c", "b", "a"], ordered=True)
 272     >>> union_categoricals([a, b], ignore_order=True)
 273     [a, b, c, c, b, a]
 274     Categories (3, object): [a, b, c]
 275 
 276     `union_categoricals` also works with a `CategoricalIndex`, or `Series`
 277     containing categorical data, but note that the resulting array will
 278     always be a plain `Categorical`
 279 
 280     >>> a = pd.Series(["b", "c"], dtype='category')
 281     >>> b = pd.Series(["a", "b"], dtype='category')
 282     >>> union_categoricals([a, b])
 283     [b, c, a, b]
 284     Categories (3, object): [b, c, a]
 285     """
 286     from pandas import Categorical
 287     from pandas.core.arrays.categorical import recode_for_categories
 288 
 289     if len(to_union) == 0:
 290         raise ValueError("No Categoricals to union")
 291 
 292     def _maybe_unwrap(x):
 293         if isinstance(x, (ABCCategoricalIndex, ABCSeries)):
 294             return x._values
 295         elif isinstance(x, Categorical):
 296             return x
 297         else:
 298             raise TypeError("all components to combine must be Categorical")
 299 
 300     to_union = [_maybe_unwrap(x) for x in to_union]
 301     first = to_union[0]
 302 
 303     if not all(
 304         is_dtype_equal(other.categories.dtype, first.categories.dtype)
 305         for other in to_union[1:]
 306     ):
 307         raise TypeError("dtype of categories must be the same")
 308 
 309     ordered = False
 310     if all(first.is_dtype_equal(other) for other in to_union[1:]):
 311         # identical categories - fastpath
 312         categories = first.categories
 313         ordered = first.ordered
 314 
 315         if all(first.categories.equals(other.categories) for other in to_union[1:]):
 316             new_codes = np.concatenate([c.codes for c in to_union])
 317         else:
 318             codes = [first.codes] + [
 319                 recode_for_categories(other.codes, other.categories, first.categories)
 320                 for other in to_union[1:]
 321             ]
 322             new_codes = np.concatenate(codes)
 323 
 324         if sort_categories and not ignore_order and ordered:
 325             raise TypeError("Cannot use sort_categories=True with ordered Categoricals")
 326 
 327         if sort_categories and not categories.is_monotonic_increasing:
 328             categories = categories.sort_values()
 329             indexer = categories.get_indexer(first.categories)
 330 
 331             from pandas.core.algorithms import take_1d
 332 
 333             new_codes = take_1d(indexer, new_codes, fill_value=-1)
 334     elif ignore_order or all(not c.ordered for c in to_union):
 335         # different categories - union and recode
 336         cats = first.categories.append([c.categories for c in to_union[1:]])
 337         categories = cats.unique()
 338         if sort_categories:
 339             categories = categories.sort_values()
 340 
 341         new_codes = [
 342             recode_for_categories(c.codes, c.categories, categories) for c in to_union
 343         ]
 344         new_codes = np.concatenate(new_codes)
 345     else:
 346         # ordered - to show a proper error message
 347         if all(c.ordered for c in to_union):
 348             msg = "to union ordered Categoricals, all categories must be the same"
 349             raise TypeError(msg)
 350         else:
 351             raise TypeError("Categorical.ordered must be the same")
 352 
 353     if ignore_order:
 354         ordered = False
 355 
 356     return Categorical(new_codes, categories=categories, ordered=ordered, fastpath=True)

 359 def _concatenate_2d(to_concat, axis: int):
 360     # coerce to 2d if needed & concatenate
 361     if axis == 1:
 362         to_concat = [np.atleast_2d(x) for x in to_concat]
 363     return np.concatenate(to_concat, axis=axis)

 366 def concat_datetime(to_concat, axis=0, typs=None):
 367     """
 368     provide concatenation of an datetimelike array of arrays each of which is a
 369     single M8[ns], datetime64[ns, tz] or m8[ns] dtype
 370 
 371     Parameters
 372     ----------
 373     to_concat : array of arrays
 374     axis : axis to provide concatenation
 375     typs : set of to_concat dtypes
 376 
 377     Returns
 378     -------
 379     a single array, preserving the combined dtypes
 380     """
 381     if typs is None:
 382         typs = get_dtype_kinds(to_concat)
 383 
 384     to_concat = [_wrap_datetimelike(x) for x in to_concat]
 385     single_dtype = len({x.dtype for x in to_concat}) == 1
 386 
 387     # multiple types, need to coerce to object
 388     if not single_dtype:
 389         # wrap_datetimelike ensures that astype(object) wraps in Timestamp/Timedelta
 390         return _concatenate_2d([x.astype(object) for x in to_concat], axis=axis)
 391 
 392     if axis == 1:
 393         # TODO(EA2D): kludge not necessary with 2D EAs
 394         to_concat = [x.reshape(1, -1) if x.ndim == 1 else x for x in to_concat]
 395 
 396     result = type(to_concat[0])._concat_same_type(to_concat, axis=axis)
 397 
 398     if result.ndim == 2 and is_extension_array_dtype(result.dtype):
 399         # TODO(EA2D): kludge not necessary with 2D EAs
 400         assert result.shape[0] == 1
 401         result = result[0]
 402     return result

 405 def _wrap_datetimelike(arr):
 406     """
 407     Wrap datetime64 and timedelta64 ndarrays in DatetimeArray/TimedeltaArray.
 408 
 409     DTA/TDA handle .astype(object) correctly.
 410     """
 411     from pandas.core.construction import array as pd_array, extract_array
 412 
 413     arr = extract_array(arr, extract_numpy=True)
 414     if isinstance(arr, np.ndarray) and arr.dtype.kind in ["m", "M"]:
 415         arr = pd_array(arr)
 416     return arr
