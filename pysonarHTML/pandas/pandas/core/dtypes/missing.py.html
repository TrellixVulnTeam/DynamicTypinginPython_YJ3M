  47 def isna(obj):
  48     """
  49     Detect missing values for an array-like object.
  50 
  51     This function takes a scalar or array-like object and indicates
  52     whether values are missing (``NaN`` in numeric arrays, ``None`` or ``NaN``
  53     in object arrays, ``NaT`` in datetimelike).
  54 
  55     Parameters
  56     ----------
  57     obj : scalar or array-like
  58         Object to check for null or missing values.
  59 
  60     Returns
  61     -------
  62     bool or array-like of bool
  63         For scalar input, returns a scalar boolean.
  64         For array input, returns an array of boolean indicating whether each
  65         corresponding element is missing.
  66 
  67     See Also
  68     --------
  69     notna : Boolean inverse of pandas.isna.
  70     Series.isna : Detect missing values in a Series.
  71     DataFrame.isna : Detect missing values in a DataFrame.
  72     Index.isna : Detect missing values in an Index.
  73 
  74     Examples
  75     --------
  76     Scalar arguments (including strings) result in a scalar boolean.
  77 
  78     >>> pd.isna('dog')
  79     False
  80 
  81     >>> pd.isna(pd.NA)
  82     True
  83 
  84     >>> pd.isna(np.nan)
  85     True
  86 
  87     ndarrays result in an ndarray of booleans.
  88 
  89     >>> array = np.array([[1, np.nan, 3], [4, 5, np.nan]])
  90     >>> array
  91     array([[ 1., nan,  3.],
  92            [ 4.,  5., nan]])
  93     >>> pd.isna(array)
  94     array([[False,  True, False],
  95            [False, False,  True]])
  96 
  97     For indexes, an ndarray of booleans is returned.
  98 
  99     >>> index = pd.DatetimeIndex(["2017-07-05", "2017-07-06", None,
 100     ...                           "2017-07-08"])
 101     >>> index
 102     DatetimeIndex(['2017-07-05', '2017-07-06', 'NaT', '2017-07-08'],
 103                   dtype='datetime64[ns]', freq=None)
 104     >>> pd.isna(index)
 105     array([False, False,  True, False])
 106 
 107     For Series and DataFrame, the same type is returned, containing booleans.
 108 
 109     >>> df = pd.DataFrame([['ant', 'bee', 'cat'], ['dog', None, 'fly']])
 110     >>> df
 111          0     1    2
 112     0  ant   bee  cat
 113     1  dog  None  fly
 114     >>> pd.isna(df)
 115            0      1      2
 116     0  False  False  False
 117     1  False   True  False
 118 
 119     >>> pd.isna(df[1])
 120     0    False
 121     1     True
 122     Name: 1, dtype: bool
 123     """
 124     return _isna(obj)

 130 def _isna(obj, inf_as_na: bool = False):
 131     """
 132     Detect missing values, treating None, NaN or NA as null. Infinite
 133     values will also be treated as null if inf_as_na is True.
 134 
 135     Parameters
 136     ----------
 137     obj: ndarray or object value
 138         Input array or scalar value.
 139     inf_as_na: bool
 140         Whether to treat infinity as null.
 141 
 142     Returns
 143     -------
 144     boolean ndarray or boolean
 145     """
 146     if is_scalar(obj):
 147         if inf_as_na:
 148             return libmissing.checknull_old(obj)
 149         else:
 150             return libmissing.checknull(obj)
 151     # hack (for now) because MI registers as ndarray
 152     elif isinstance(obj, ABCMultiIndex):
 153         raise NotImplementedError("isna is not defined for MultiIndex")
 154     elif isinstance(obj, type):
 155         return False
 156     elif isinstance(obj, (ABCSeries, np.ndarray, ABCIndexClass, ABCExtensionArray)):
 157         return _isna_ndarraylike(obj, inf_as_na=inf_as_na)
 158     elif isinstance(obj, ABCDataFrame):
 159         return obj.isna()
 160     elif isinstance(obj, list):
 161         return _isna_ndarraylike(np.asarray(obj, dtype=object), inf_as_na=inf_as_na)
 162     elif hasattr(obj, "__array__"):
 163         return _isna_ndarraylike(np.asarray(obj), inf_as_na=inf_as_na)
 164     else:
 165         return False

 168 def _use_inf_as_na(key):
 169     """
 170     Option change callback for na/inf behaviour.
 171 
 172     Choose which replacement for numpy.isnan / -numpy.isfinite is used.
 173 
 174     Parameters
 175     ----------
 176     flag: bool
 177         True means treat None, NaN, INF, -INF as null (old way),
 178         False means None and NaN are null, but INF, -INF are not null
 179         (new way).
 180 
 181     Notes
 182     -----
 183     This approach to setting global module values is discussed and
 184     approved here:
 185 
 186     * https://stackoverflow.com/questions/4859217/
 187       programmatically-creating-variables-in-python/4859312#4859312
 188     """
 189     inf_as_na = get_option(key)
 190     globals()["_isna"] = partial(_isna, inf_as_na=inf_as_na)

 193 def _isna_ndarraylike(obj, inf_as_na: bool = False):
 194     """
 195     Return an array indicating which values of the input array are NaN / NA.
 196 
 197     Parameters
 198     ----------
 199     obj: array-like
 200         The input array whose elements are to be checked.
 201     inf_as_na: bool
 202         Whether or not to treat infinite values as NA.
 203 
 204     Returns
 205     -------
 206     array-like
 207         Array of boolean values denoting the NA status of each element.
 208     """
 209     values = getattr(obj, "_values", obj)
 210     dtype = values.dtype
 211 
 212     if is_extension_array_dtype(dtype):
 213         if inf_as_na and is_categorical_dtype(dtype):
 214             result = libmissing.isnaobj_old(values.to_numpy())
 215         else:
 216             result = values.isna()
 217     elif is_string_dtype(dtype):
 218         result = _isna_string_dtype(values, dtype, inf_as_na=inf_as_na)
 219     elif needs_i8_conversion(dtype):
 220         # this is the NaT pattern
 221         result = values.view("i8") == iNaT
 222     else:
 223         if inf_as_na:
 224             result = ~np.isfinite(values)
 225         else:
 226             result = np.isnan(values)
 227 
 228     # box
 229     if isinstance(obj, ABCSeries):
 230         result = obj._constructor(result, index=obj.index, name=obj.name, copy=False)
 231 
 232     return result

 235 def _isna_string_dtype(
 236     values: np.ndarray, dtype: np.dtype, inf_as_na: bool
 237 ) -> np.ndarray:
 238     # Working around NumPy ticket 1542
 239     shape = values.shape
 240 
 241     if is_string_like_dtype(dtype):
 242         result = np.zeros(values.shape, dtype=bool)
 243     else:
 244         result = np.empty(shape, dtype=bool)
 245         if inf_as_na:
 246             vec = libmissing.isnaobj_old(values.ravel())
 247         else:
 248             vec = libmissing.isnaobj(values.ravel())
 249 
 250         result[...] = vec.reshape(shape)
 251 
 252     return result

 255 def notna(obj):
 256     """
 257     Detect non-missing values for an array-like object.
 258 
 259     This function takes a scalar or array-like object and indicates
 260     whether values are valid (not missing, which is ``NaN`` in numeric
 261     arrays, ``None`` or ``NaN`` in object arrays, ``NaT`` in datetimelike).
 262 
 263     Parameters
 264     ----------
 265     obj : array-like or object value
 266         Object to check for *not* null or *non*-missing values.
 267 
 268     Returns
 269     -------
 270     bool or array-like of bool
 271         For scalar input, returns a scalar boolean.
 272         For array input, returns an array of boolean indicating whether each
 273         corresponding element is valid.
 274 
 275     See Also
 276     --------
 277     isna : Boolean inverse of pandas.notna.
 278     Series.notna : Detect valid values in a Series.
 279     DataFrame.notna : Detect valid values in a DataFrame.
 280     Index.notna : Detect valid values in an Index.
 281 
 282     Examples
 283     --------
 284     Scalar arguments (including strings) result in a scalar boolean.
 285 
 286     >>> pd.notna('dog')
 287     True
 288 
 289     >>> pd.notna(pd.NA)
 290     False
 291 
 292     >>> pd.notna(np.nan)
 293     False
 294 
 295     ndarrays result in an ndarray of booleans.
 296 
 297     >>> array = np.array([[1, np.nan, 3], [4, 5, np.nan]])
 298     >>> array
 299     array([[ 1., nan,  3.],
 300            [ 4.,  5., nan]])
 301     >>> pd.notna(array)
 302     array([[ True, False,  True],
 303            [ True,  True, False]])
 304 
 305     For indexes, an ndarray of booleans is returned.
 306 
 307     >>> index = pd.DatetimeIndex(["2017-07-05", "2017-07-06", None,
 308     ...                          "2017-07-08"])
 309     >>> index
 310     DatetimeIndex(['2017-07-05', '2017-07-06', 'NaT', '2017-07-08'],
 311                   dtype='datetime64[ns]', freq=None)
 312     >>> pd.notna(index)
 313     array([ True,  True, False,  True])
 314 
 315     For Series and DataFrame, the same type is returned, containing booleans.
 316 
 317     >>> df = pd.DataFrame([['ant', 'bee', 'cat'], ['dog', None, 'fly']])
 318     >>> df
 319          0     1    2
 320     0  ant   bee  cat
 321     1  dog  None  fly
 322     >>> pd.notna(df)
 323           0      1     2
 324     0  True   True  True
 325     1  True  False  True
 326 
 327     >>> pd.notna(df[1])
 328     0     True
 329     1    False
 330     Name: 1, dtype: bool
 331     """
 332     res = isna(obj)
 333     if is_scalar(res):
 334         return not res
 335     return ~res

 341 def _isna_compat(arr, fill_value=np.nan) -> bool:
 342     """
 343     Parameters
 344     ----------
 345     arr: a numpy array
 346     fill_value: fill value, default to np.nan
 347 
 348     Returns
 349     -------
 350     True if we can fill using this fill_value
 351     """
 352     dtype = arr.dtype
 353     if isna(fill_value):
 354         return not (is_bool_dtype(dtype) or is_integer_dtype(dtype))
 355     return True

 358 def array_equivalent(left, right, strict_nan: bool = False) -> bool:
 359     """
 360     True if two arrays, left and right, have equal non-NaN elements, and NaNs
 361     in corresponding locations.  False otherwise. It is assumed that left and
 362     right are NumPy arrays of the same dtype. The behavior of this function
 363     (particularly with respect to NaNs) is not defined if the dtypes are
 364     different.
 365 
 366     Parameters
 367     ----------
 368     left, right : ndarrays
 369     strict_nan : bool, default False
 370         If True, consider NaN and None to be different.
 371 
 372     Returns
 373     -------
 374     b : bool
 375         Returns True if the arrays are equivalent.
 376 
 377     Examples
 378     --------
 379     >>> array_equivalent(
 380     ...     np.array([1, 2, np.nan]),
 381     ...     np.array([1, 2, np.nan]))
 382     True
 383     >>> array_equivalent(
 384     ...     np.array([1, np.nan, 2]),
 385     ...     np.array([1, 2, np.nan]))
 386     False
 387     """
 388     left, right = np.asarray(left), np.asarray(right)
 389 
 390     # shape compat
 391     if left.shape != right.shape:
 392         return False
 393 
 394     # Object arrays can contain None, NaN and NaT.
 395     # string dtypes must be come to this path for NumPy 1.7.1 compat
 396     if is_string_dtype(left.dtype) or is_string_dtype(right.dtype):
 397 
 398         if not strict_nan:
 399             # isna considers NaN and None to be equivalent.
 400             return lib.array_equivalent_object(
 401                 ensure_object(left.ravel()), ensure_object(right.ravel())
 402             )
 403 
 404         for left_value, right_value in zip(left, right):
 405             if left_value is NaT and right_value is not NaT:
 406                 return False
 407 
 408             elif left_value is libmissing.NA and right_value is not libmissing.NA:
 409                 return False
 410 
 411             elif isinstance(left_value, float) and np.isnan(left_value):
 412                 if not isinstance(right_value, float) or not np.isnan(right_value):
 413                     return False
 414             else:
 415                 try:
 416                     if np.any(np.asarray(left_value != right_value)):
 417                         return False
 418                 except TypeError as err:
 419                     if "Cannot compare tz-naive" in str(err):
 420                         # tzawareness compat failure, see GH#28507
 421                         return False
 422                     elif "boolean value of NA is ambiguous" in str(err):
 423                         return False
 424                     raise
 425         return True
 426 
 427     # NaNs can occur in float and complex arrays.
 428     if is_float_dtype(left.dtype) or is_complex_dtype(left.dtype):
 429 
 430         # empty
 431         if not (np.prod(left.shape) and np.prod(right.shape)):
 432             return True
 433         return ((left == right) | (isna(left) & isna(right))).all()
 434 
 435     elif is_datetimelike_v_numeric(left, right):
 436         # GH#29553 avoid numpy deprecation warning
 437         return False
 438 
 439     elif needs_i8_conversion(left.dtype) or needs_i8_conversion(right.dtype):
 440         # datetime64, timedelta64, Period
 441         if not is_dtype_equal(left.dtype, right.dtype):
 442             return False
 443 
 444         left = left.view("i8")
 445         right = right.view("i8")
 446 
 447     # if we have structured dtypes, compare first
 448     if left.dtype.type is np.void or right.dtype.type is np.void:
 449         if left.dtype != right.dtype:
 450             return False
 451 
 452     return np.array_equal(left, right)

 455 def _infer_fill_value(val):
 456     """
 457     infer the fill value for the nan/NaT from the provided
 458     scalar/ndarray/list-like if we are a NaT, return the correct dtyped
 459     element to provide proper block construction
 460     """
 461     if not is_list_like(val):
 462         val = [val]
 463     val = np.array(val, copy=False)
 464     if needs_i8_conversion(val.dtype):
 465         return np.array("NaT", dtype=val.dtype)
 466     elif is_object_dtype(val.dtype):
 467         dtype = lib.infer_dtype(ensure_object(val), skipna=False)
 468         if dtype in ["datetime", "datetime64"]:
 469             return np.array("NaT", dtype=DT64NS_DTYPE)
 470         elif dtype in ["timedelta", "timedelta64"]:
 471             return np.array("NaT", dtype=TD64NS_DTYPE)
 472     return np.nan

 475 def _maybe_fill(arr, fill_value=np.nan):
 476     """
 477     if we have a compatible fill_value and arr dtype, then fill
 478     """
 479     if _isna_compat(arr, fill_value):
 480         arr.fill(fill_value)
 481     return arr

 484 def na_value_for_dtype(dtype, compat: bool = True):
 485     """
 486     Return a dtype compat na value
 487 
 488     Parameters
 489     ----------
 490     dtype : string / dtype
 491     compat : bool, default True
 492 
 493     Returns
 494     -------
 495     np.dtype or a pandas dtype
 496 
 497     Examples
 498     --------
 499     >>> na_value_for_dtype(np.dtype('int64'))
 500     0
 501     >>> na_value_for_dtype(np.dtype('int64'), compat=False)
 502     nan
 503     >>> na_value_for_dtype(np.dtype('float64'))
 504     nan
 505     >>> na_value_for_dtype(np.dtype('bool'))
 506     False
 507     >>> na_value_for_dtype(np.dtype('datetime64[ns]'))
 508     NaT
 509     """
 510     dtype = pandas_dtype(dtype)
 511 
 512     if is_extension_array_dtype(dtype):
 513         return dtype.na_value
 514     if needs_i8_conversion(dtype):
 515         return NaT
 516     elif is_float_dtype(dtype):
 517         return np.nan
 518     elif is_integer_dtype(dtype):
 519         if compat:
 520             return 0
 521         return np.nan
 522     elif is_bool_dtype(dtype):
 523         if compat:
 524             return False
 525         return np.nan
 526     return np.nan

 529 def remove_na_arraylike(arr):
 530     """
 531     Return array-like containing only true/non-NaN values, possibly empty.
 532     """
 533     if is_extension_array_dtype(arr):
 534         return arr[notna(arr)]
 535     else:
 536         return arr[notna(np.asarray(arr))]

 539 def is_valid_nat_for_dtype(obj, dtype: DtypeObj) -> bool:
 540     """
 541     isna check that excludes incompatible dtypes
 542 
 543     Parameters
 544     ----------
 545     obj : object
 546     dtype : np.datetime64, np.timedelta64, DatetimeTZDtype, or PeriodDtype
 547 
 548     Returns
 549     -------
 550     bool
 551     """
 552     if not lib.is_scalar(obj) or not isna(obj):
 553         return False
 554     if dtype.kind == "M":
 555         return not isinstance(obj, np.timedelta64)
 556     if dtype.kind == "m":
 557         return not isinstance(obj, np.datetime64)
 558 
 559     # must be PeriodDType
 560     return not isinstance(obj, (np.datetime64, np.timedelta64))
