 114 def maybe_downcast_to_dtype(result, dtype):
 115     """
 116     try to cast to the specified dtype (e.g. convert back to bool/int
 117     or could be an astype of float64->float32
 118     """
 119     do_round = False
 120 
 121     if is_scalar(result):
 122         return result
 123     elif isinstance(result, ABCDataFrame):
 124         # occurs in pivot_table doctest
 125         return result
 126 
 127     if isinstance(dtype, str):
 128         if dtype == "infer":
 129             inferred_type = lib.infer_dtype(ensure_object(result), skipna=False)
 130             if inferred_type == "boolean":
 131                 dtype = "bool"
 132             elif inferred_type == "integer":
 133                 dtype = "int64"
 134             elif inferred_type == "datetime64":
 135                 dtype = "datetime64[ns]"
 136             elif inferred_type == "timedelta64":
 137                 dtype = "timedelta64[ns]"
 138 
 139             # try to upcast here
 140             elif inferred_type == "floating":
 141                 dtype = "int64"
 142                 if issubclass(result.dtype.type, np.number):
 143                     do_round = True
 144 
 145             else:
 146                 dtype = "object"
 147 
 148         dtype = np.dtype(dtype)
 149 
 150     converted = maybe_downcast_numeric(result, dtype, do_round)
 151     if converted is not result:
 152         return converted
 153 
 154     # a datetimelike
 155     # GH12821, iNaT is casted to float
 156     if dtype.kind in ["M", "m"] and result.dtype.kind in ["i", "f"]:
 157         if hasattr(dtype, "tz"):
 158             # not a numpy dtype
 159             if dtype.tz:
 160                 # convert to datetime and change timezone
 161                 from pandas import to_datetime
 162 
 163                 result = to_datetime(result).tz_localize("utc")
 164                 result = result.tz_convert(dtype.tz)
 165         else:
 166             result = result.astype(dtype)
 167 
 168     elif dtype.type is Period:
 169         # TODO(DatetimeArray): merge with previous elif
 170         from pandas.core.arrays import PeriodArray
 171 
 172         try:
 173             return PeriodArray(result, freq=dtype.freq)
 174         except TypeError:
 175             # e.g. TypeError: int() argument must be a string, a
 176             #  bytes-like object or a number, not 'Period
 177             pass
 178 
 179     return result

 182 def maybe_downcast_numeric(result, dtype, do_round: bool = False):
 183     """
 184     Subset of maybe_downcast_to_dtype restricted to numeric dtypes.
 185 
 186     Parameters
 187     ----------
 188     result : ndarray or ExtensionArray
 189     dtype : np.dtype or ExtensionDtype
 190     do_round : bool
 191 
 192     Returns
 193     -------
 194     ndarray or ExtensionArray
 195     """
 196     if not isinstance(dtype, np.dtype):
 197         # e.g. SparseDtype has no itemsize attr
 198         return result
 199 
 200     if isinstance(result, list):
 201         # reached via groupby.agg._ohlc; really this should be handled earlier
 202         result = np.array(result)
 203 
 204     def trans(x):
 205         if do_round:
 206             return x.round()
 207         return x
 208 
 209     if dtype.kind == result.dtype.kind:
 210         # don't allow upcasts here (except if empty)
 211         if result.dtype.itemsize <= dtype.itemsize and result.size:
 212             return result
 213 
 214     if is_bool_dtype(dtype) or is_integer_dtype(dtype):
 215 
 216         if not result.size:
 217             # if we don't have any elements, just astype it
 218             return trans(result).astype(dtype)
 219 
 220         # do a test on the first element, if it fails then we are done
 221         r = result.ravel()
 222         arr = np.array([r[0]])
 223 
 224         if isna(arr).any():
 225             # if we have any nulls, then we are done
 226             return result
 227 
 228         elif not isinstance(r[0], (np.integer, np.floating, np.bool, int, float, bool)):
 229             # a comparable, e.g. a Decimal may slip in here
 230             return result
 231 
 232         if (
 233             issubclass(result.dtype.type, (np.object_, np.number))
 234             and notna(result).all()
 235         ):
 236             new_result = trans(result).astype(dtype)
 237             if new_result.dtype.kind == "O" or result.dtype.kind == "O":
 238                 # np.allclose may raise TypeError on object-dtype
 239                 if (new_result == result).all():
 240                     return new_result
 241             else:
 242                 if np.allclose(new_result, result, rtol=0):
 243                     return new_result
 244 
 245     elif (
 246         issubclass(dtype.type, np.floating)
 247         and not is_bool_dtype(result.dtype)
 248         and not is_string_dtype(result.dtype)
 249     ):
 250         return result.astype(dtype)
 251 
 252     return result

 255 def maybe_cast_result(result, obj: "Series", numeric_only: bool = False, how: str = ""):
 256     """
 257     Try casting result to a different type if appropriate
 258 
 259     Parameters
 260     ----------
 261     result : array-like
 262         Result to cast.
 263     obj : Series
 264         Input Series from which result was calculated.
 265     numeric_only : bool, default False
 266         Whether to cast only numerics or datetimes as well.
 267     how : str, default ""
 268         How the result was computed.
 269 
 270     Returns
 271     -------
 272     result : array-like
 273         result maybe casted to the dtype.
 274     """
 275     if obj.ndim > 1:
 276         dtype = obj._values.dtype
 277     else:
 278         dtype = obj.dtype
 279     dtype = maybe_cast_result_dtype(dtype, how)
 280 
 281     if not is_scalar(result):
 282         if (
 283             is_extension_array_dtype(dtype)
 284             and not is_categorical_dtype(dtype)
 285             and dtype.kind != "M"
 286         ):
 287             # We have to special case categorical so as not to upcast
 288             # things like counts back to categorical
 289             cls = dtype.construct_array_type()
 290             result = maybe_cast_to_extension_array(cls, result, dtype=dtype)
 291 
 292         elif numeric_only and is_numeric_dtype(dtype) or not numeric_only:
 293             result = maybe_downcast_to_dtype(result, dtype)
 294 
 295     return result

 298 def maybe_cast_result_dtype(dtype: DtypeObj, how: str) -> DtypeObj:
 299     """
 300     Get the desired dtype of a result based on the
 301     input dtype and how it was computed.
 302 
 303     Parameters
 304     ----------
 305     dtype : DtypeObj
 306         Input dtype.
 307     how : str
 308         How the result was computed.
 309 
 310     Returns
 311     -------
 312     DtypeObj
 313         The desired dtype of the result.
 314     """
 315     from pandas.core.arrays.boolean import BooleanDtype
 316     from pandas.core.arrays.integer import Int64Dtype
 317 
 318     if how in ["add", "cumsum", "sum"] and (dtype == np.dtype(np.bool)):
 319         return np.dtype(np.int64)
 320     elif how in ["add", "cumsum", "sum"] and isinstance(dtype, BooleanDtype):
 321         return Int64Dtype()
 322     return dtype

 325 def maybe_cast_to_extension_array(cls: Type["ExtensionArray"], obj, dtype=None):
 326     """
 327     Call to `_from_sequence` that returns the object unchanged on Exception.
 328 
 329     Parameters
 330     ----------
 331     cls : class, subclass of ExtensionArray
 332     obj : arraylike
 333         Values to pass to cls._from_sequence
 334     dtype : ExtensionDtype, optional
 335 
 336     Returns
 337     -------
 338     ExtensionArray or obj
 339     """
 340     from pandas.core.arrays.string_ import StringArray
 341 
 342     assert isinstance(cls, type), f"must pass a type: {cls}"
 343     assertion_msg = f"must pass a subclass of ExtensionArray: {cls}"
 344     assert issubclass(cls, ABCExtensionArray), assertion_msg
 345 
 346     # Everything can be be converted to StringArrays, but we may not want to convert
 347     if issubclass(cls, StringArray) and lib.infer_dtype(obj) != "string":
 348         return obj
 349 
 350     try:
 351         result = cls._from_sequence(obj, dtype=dtype)
 352     except Exception:
 353         # We can't predict what downstream EA constructors may raise
 354         result = obj
 355     return result

 358 def maybe_upcast_putmask(result: np.ndarray, mask: np.ndarray, other):
 359     """
 360     A safe version of putmask that potentially upcasts the result.
 361 
 362     The result is replaced with the first N elements of other,
 363     where N is the number of True values in mask.
 364     If the length of other is shorter than N, other will be repeated.
 365 
 366     Parameters
 367     ----------
 368     result : ndarray
 369         The destination array. This will be mutated in-place if no upcasting is
 370         necessary.
 371     mask : boolean ndarray
 372     other : scalar
 373         The source value.
 374 
 375     Returns
 376     -------
 377     result : ndarray
 378     changed : bool
 379         Set to true if the result array was upcasted.
 380 
 381     Examples
 382     --------
 383     >>> arr = np.arange(1, 6)
 384     >>> mask = np.array([False, True, False, True, True])
 385     >>> result, _ = maybe_upcast_putmask(arr, mask, False)
 386     >>> result
 387     array([1, 0, 3, 0, 0])
 388     """
 389     if not isinstance(result, np.ndarray):
 390         raise ValueError("The result input must be a ndarray.")
 391     if not is_scalar(other):
 392         # We _could_ support non-scalar other, but until we have a compelling
 393         #  use case, we assume away the possibility.
 394         raise ValueError("other must be a scalar")
 395 
 396     if mask.any():
 397         # Two conversions for date-like dtypes that can't be done automatically
 398         # in np.place:
 399         #   NaN -> NaT
 400         #   integer or integer array -> date-like array
 401         if result.dtype.kind in ["m", "M"]:
 402             if is_scalar(other):
 403                 if isna(other):
 404                     other = result.dtype.type("nat")
 405                 elif is_integer(other):
 406                     other = np.array(other, dtype=result.dtype)
 407             elif is_integer_dtype(other):
 408                 other = np.array(other, dtype=result.dtype)
 409 
 410         def changeit():
 411             # we are forced to change the dtype of the result as the input
 412             # isn't compatible
 413             r, _ = maybe_upcast(result, fill_value=other, copy=True)
 414             np.place(r, mask, other)
 415 
 416             return r, True
 417 
 418         # we want to decide whether place will work
 419         # if we have nans in the False portion of our mask then we need to
 420         # upcast (possibly), otherwise we DON't want to upcast (e.g. if we
 421         # have values, say integers, in the success portion then it's ok to not
 422         # upcast)
 423         new_dtype, _ = maybe_promote(result.dtype, other)
 424         if new_dtype != result.dtype:
 425 
 426             # we have a scalar or len 0 ndarray
 427             # and its nan and we are changing some values
 428             if isna(other):
 429                 return changeit()
 430 
 431         try:
 432             np.place(result, mask, other)
 433         except TypeError:
 434             # e.g. int-dtype result and float-dtype other
 435             return changeit()
 436 
 437     return result, False

 440 def maybe_promote(dtype, fill_value=np.nan):
 441     """
 442     Find the minimal dtype that can hold both the given dtype and fill_value.
 443 
 444     Parameters
 445     ----------
 446     dtype : np.dtype or ExtensionDtype
 447     fill_value : scalar, default np.nan
 448 
 449     Returns
 450     -------
 451     dtype
 452         Upcasted from dtype argument if necessary.
 453     fill_value
 454         Upcasted from fill_value argument if necessary.
 455     """
 456     if not is_scalar(fill_value) and not is_object_dtype(dtype):
 457         # with object dtype there is nothing to promote, and the user can
 458         #  pass pretty much any weird fill_value they like
 459         raise ValueError("fill_value must be a scalar")
 460 
 461     # if we passed an array here, determine the fill value by dtype
 462     if isinstance(fill_value, np.ndarray):
 463         if issubclass(fill_value.dtype.type, (np.datetime64, np.timedelta64)):
 464             fill_value = fill_value.dtype.type("NaT", "ns")
 465         else:
 466 
 467             # we need to change to object type as our
 468             # fill_value is of object type
 469             if fill_value.dtype == np.object_:
 470                 dtype = np.dtype(np.object_)
 471             fill_value = np.nan
 472 
 473         if dtype == np.object_ or dtype.kind in ["U", "S"]:
 474             # We treat string-like dtypes as object, and _always_ fill
 475             #  with np.nan
 476             fill_value = np.nan
 477             dtype = np.dtype(np.object_)
 478 
 479     # returns tuple of (dtype, fill_value)
 480     if issubclass(dtype.type, np.datetime64):
 481         if isinstance(fill_value, datetime) and fill_value.tzinfo is not None:
 482             # Trying to insert tzaware into tznaive, have to cast to object
 483             dtype = np.dtype(np.object_)
 484         elif is_integer(fill_value) or (is_float(fill_value) and not isna(fill_value)):
 485             dtype = np.dtype(np.object_)
 486         else:
 487             try:
 488                 fill_value = tslibs.Timestamp(fill_value).to_datetime64()
 489             except (TypeError, ValueError):
 490                 dtype = np.dtype(np.object_)
 491     elif issubclass(dtype.type, np.timedelta64):
 492         if (
 493             is_integer(fill_value)
 494             or (is_float(fill_value) and not np.isnan(fill_value))
 495             or isinstance(fill_value, str)
 496         ):
 497             # TODO: What about str that can be a timedelta?
 498             dtype = np.dtype(np.object_)
 499         else:
 500             try:
 501                 fv = tslibs.Timedelta(fill_value)
 502             except ValueError:
 503                 dtype = np.dtype(np.object_)
 504             else:
 505                 if fv is NaT:
 506                     # NaT has no `to_timedelta64` method
 507                     fill_value = np.timedelta64("NaT", "ns")
 508                 else:
 509                     fill_value = fv.to_timedelta64()
 510     elif is_datetime64tz_dtype(dtype):
 511         if isna(fill_value):
 512             fill_value = NaT
 513         elif not isinstance(fill_value, datetime):
 514             dtype = np.dtype(np.object_)
 515         elif fill_value.tzinfo is None:
 516             dtype = np.dtype(np.object_)
 517         elif not tz_compare(fill_value.tzinfo, dtype.tz):
 518             # TODO: sure we want to cast here?
 519             dtype = np.dtype(np.object_)
 520 
 521     elif is_extension_array_dtype(dtype) and isna(fill_value):
 522         fill_value = dtype.na_value
 523 
 524     elif is_float(fill_value):
 525         if issubclass(dtype.type, np.bool_):
 526             dtype = np.dtype(np.object_)
 527 
 528         elif issubclass(dtype.type, np.integer):
 529             dtype = np.dtype(np.float64)
 530 
 531         elif dtype.kind == "f":
 532             mst = np.min_scalar_type(fill_value)
 533             if mst > dtype:
 534                 # e.g. mst is np.float64 and dtype is np.float32
 535                 dtype = mst
 536 
 537         elif dtype.kind == "c":
 538             mst = np.min_scalar_type(fill_value)
 539             dtype = np.promote_types(dtype, mst)
 540 
 541     elif is_bool(fill_value):
 542         if not issubclass(dtype.type, np.bool_):
 543             dtype = np.dtype(np.object_)
 544 
 545     elif is_integer(fill_value):
 546         if issubclass(dtype.type, np.bool_):
 547             dtype = np.dtype(np.object_)
 548 
 549         elif issubclass(dtype.type, np.integer):
 550             if not np.can_cast(fill_value, dtype):
 551                 # upcast to prevent overflow
 552                 mst = np.min_scalar_type(fill_value)
 553                 dtype = np.promote_types(dtype, mst)
 554                 if dtype.kind == "f":
 555                     # Case where we disagree with numpy
 556                     dtype = np.dtype(np.object_)
 557 
 558     elif is_complex(fill_value):
 559         if issubclass(dtype.type, np.bool_):
 560             dtype = np.dtype(np.object_)
 561 
 562         elif issubclass(dtype.type, (np.integer, np.floating)):
 563             mst = np.min_scalar_type(fill_value)
 564             dtype = np.promote_types(dtype, mst)
 565 
 566         elif dtype.kind == "c":
 567             mst = np.min_scalar_type(fill_value)
 568             if mst > dtype:
 569                 # e.g. mst is np.complex128 and dtype is np.complex64
 570                 dtype = mst
 571 
 572     elif fill_value is None:
 573         if is_float_dtype(dtype) or is_complex_dtype(dtype):
 574             fill_value = np.nan
 575         elif is_integer_dtype(dtype):
 576             dtype = np.float64
 577             fill_value = np.nan
 578         elif is_datetime_or_timedelta_dtype(dtype):
 579             fill_value = dtype.type("NaT", "ns")
 580         else:
 581             dtype = np.dtype(np.object_)
 582             fill_value = np.nan
 583     else:
 584         dtype = np.dtype(np.object_)
 585 
 586     # in case we have a string that looked like a number
 587     if is_extension_array_dtype(dtype):
 588         pass
 589     elif issubclass(np.dtype(dtype).type, (bytes, str)):
 590         dtype = np.dtype(np.object_)
 591 
 592     fill_value = _ensure_dtype_type(fill_value, dtype)
 593     return dtype, fill_value

 596 def _ensure_dtype_type(value, dtype):
 597     """
 598     Ensure that the given value is an instance of the given dtype.
 599 
 600     e.g. if out dtype is np.complex64, we should have an instance of that
 601     as opposed to a python complex object.
 602 
 603     Parameters
 604     ----------
 605     value : object
 606     dtype : np.dtype or ExtensionDtype
 607 
 608     Returns
 609     -------
 610     object
 611     """
 612     # Start with exceptions in which we do _not_ cast to numpy types
 613     if is_extension_array_dtype(dtype):
 614         return value
 615     elif dtype == np.object_:
 616         return value
 617     elif isna(value):
 618         # e.g. keep np.nan rather than try to cast to np.float32(np.nan)
 619         return value
 620 
 621     return dtype.type(value)

 641 def infer_dtype_from_scalar(val, pandas_dtype: bool = False) -> Tuple[DtypeObj, Any]:
 642     """
 643     Interpret the dtype from a scalar.
 644 
 645     Parameters
 646     ----------
 647     pandas_dtype : bool, default False
 648         whether to infer dtype including pandas extension types.
 649         If False, scalar belongs to pandas extension types is inferred as
 650         object
 651     """
 652     dtype = np.dtype(object)
 653 
 654     # a 1-element ndarray
 655     if isinstance(val, np.ndarray):
 656         msg = "invalid ndarray passed to infer_dtype_from_scalar"
 657         if val.ndim != 0:
 658             raise ValueError(msg)
 659 
 660         dtype = val.dtype
 661         val = val.item()
 662 
 663     elif isinstance(val, str):
 664 
 665         # If we create an empty array using a string to infer
 666         # the dtype, NumPy will only allocate one character per entry
 667         # so this is kind of bad. Alternately we could use np.repeat
 668         # instead of np.empty (but then you still don't want things
 669         # coming out as np.str_!
 670 
 671         dtype = np.dtype(object)
 672 
 673     elif isinstance(val, (np.datetime64, datetime)):
 674         val = tslibs.Timestamp(val)
 675         if val is tslibs.NaT or val.tz is None:
 676             dtype = np.dtype("M8[ns]")
 677         else:
 678             if pandas_dtype:
 679                 dtype = DatetimeTZDtype(unit="ns", tz=val.tz)
 680             else:
 681                 # return datetimetz as object
 682                 return np.dtype(object), val
 683         val = val.value
 684 
 685     elif isinstance(val, (np.timedelta64, timedelta)):
 686         val = tslibs.Timedelta(val).value
 687         dtype = np.dtype("m8[ns]")
 688 
 689     elif is_bool(val):
 690         dtype = np.dtype(np.bool_)
 691 
 692     elif is_integer(val):
 693         if isinstance(val, np.integer):
 694             dtype = np.dtype(type(val))
 695         else:
 696             dtype = np.dtype(np.int64)
 697 
 698     elif is_float(val):
 699         if isinstance(val, np.floating):
 700             dtype = np.dtype(type(val))
 701         else:
 702             dtype = np.dtype(np.float64)
 703 
 704     elif is_complex(val):
 705         dtype = np.dtype(np.complex_)
 706 
 707     elif pandas_dtype:
 708         if lib.is_period(val):
 709             dtype = PeriodDtype(freq=val.freq)
 710             val = val.ordinal
 711         elif lib.is_interval(val):
 712             subtype = infer_dtype_from_scalar(val.left, pandas_dtype=True)[0]
 713             dtype = IntervalDtype(subtype=subtype)
 714 
 715     return dtype, val

 719 def infer_dtype_from_array(arr, pandas_dtype: bool = False) -> Tuple[DtypeObj, Any]:
 720     """
 721     Infer the dtype from an array.
 722 
 723     Parameters
 724     ----------
 725     arr : array
 726     pandas_dtype : bool, default False
 727         whether to infer dtype including pandas extension types.
 728         If False, array belongs to pandas extension types
 729         is inferred as object
 730 
 731     Returns
 732     -------
 733     tuple (numpy-compat/pandas-compat dtype, array)
 734 
 735     Notes
 736     -----
 737     if pandas_dtype=False. these infer to numpy dtypes
 738     exactly with the exception that mixed / object dtypes
 739     are not coerced by stringifying or conversion
 740 
 741     if pandas_dtype=True. datetime64tz-aware/categorical
 742     types will retain there character.
 743 
 744     Examples
 745     --------
 746     >>> np.asarray([1, '1'])
 747     array(['1', '1'], dtype='<U21')
 748 
 749     >>> infer_dtype_from_array([1, '1'])
 750     (dtype('O'), [1, '1'])
 751     """
 752     if isinstance(arr, np.ndarray):
 753         return arr.dtype, arr
 754 
 755     if not is_list_like(arr):
 756         arr = [arr]
 757 
 758     if pandas_dtype and is_extension_array_dtype(arr):
 759         return arr.dtype, arr
 760 
 761     elif isinstance(arr, ABCSeries):
 762         return arr.dtype, np.asarray(arr)
 763 
 764     # don't force numpy coerce with nan's
 765     inferred = lib.infer_dtype(arr, skipna=False)
 766     if inferred in ["string", "bytes", "mixed", "mixed-integer"]:
 767         return (np.dtype(np.object_), arr)
 768 
 769     arr = np.asarray(arr)
 770     return arr.dtype, arr

 807 def maybe_upcast(values, fill_value=np.nan, dtype=None, copy: bool = False):
 808     """
 809     Provide explicit type promotion and coercion.
 810 
 811     Parameters
 812     ----------
 813     values : ndarray or ExtensionArray
 814         The array that we want to maybe upcast.
 815     fill_value : what we want to fill with
 816     dtype : if None, then use the dtype of the values, else coerce to this type
 817     copy : bool, default True
 818         If True always make a copy even if no upcast is required.
 819     """
 820     if not is_scalar(fill_value) and not is_object_dtype(values.dtype):
 821         # We allow arbitrary fill values for object dtype
 822         raise ValueError("fill_value must be a scalar")
 823 
 824     if is_extension_array_dtype(values):
 825         if copy:
 826             values = values.copy()
 827     else:
 828         if dtype is None:
 829             dtype = values.dtype
 830         new_dtype, fill_value = maybe_promote(dtype, fill_value)
 831         if new_dtype != values.dtype:
 832             values = values.astype(new_dtype)
 833         elif copy:
 834             values = values.copy()
 835 
 836     return values, fill_value

 861 def coerce_to_dtypes(result, dtypes):
 862     """
 863     given a dtypes and a result set, coerce the result elements to the
 864     dtypes
 865     """

 867         raise AssertionError("_coerce_to_dtypes requires equal len arrays")

 870         if np.any(isna(r)):
 871             pass

 891 def astype_nansafe(arr, dtype, copy: bool = True, skipna: bool = False):
 892     """
 893     Cast the elements of an array to a given dtype a nan-safe manner.
 894 
 895     Parameters
 896     ----------
 897     arr : ndarray
 898     dtype : np.dtype
 899     copy : bool, default True
 900         If False, a view will be attempted but may fail, if
 901         e.g. the item sizes don't align.
 902     skipna: bool, default False
 903         Whether or not we should skip NaN when casting as a string-type.
 904 
 905     Raises
 906     ------
 907     ValueError
 908         The dtype was a datetime64/timedelta64 dtype, but it had no unit.
 909     """
 910     # dispatch on extension dtype if needed
 911     if is_extension_array_dtype(dtype):
 912         return dtype.construct_array_type()._from_sequence(arr, dtype=dtype, copy=copy)
 913 
 914     if not isinstance(dtype, np.dtype):
 915         dtype = pandas_dtype(dtype)
 916 
 917     if issubclass(dtype.type, str):
 918         return lib.astype_str(arr.ravel(), skipna=skipna).reshape(arr.shape)
 919 
 920     elif is_datetime64_dtype(arr):
 921         if is_object_dtype(dtype):
 922             return tslib.ints_to_pydatetime(arr.view(np.int64))
 923         elif dtype == np.int64:
 924             if isna(arr).any():
 925                 raise ValueError("Cannot convert NaT values to integer")
 926             return arr.view(dtype)
 927 
 928         # allow frequency conversions
 929         if dtype.kind == "M":
 930             return arr.astype(dtype)
 931 
 932         raise TypeError(f"cannot astype a datetimelike from [{arr.dtype}] to [{dtype}]")
 933 
 934     elif is_timedelta64_dtype(arr):
 935         if is_object_dtype(dtype):
 936             return tslibs.ints_to_pytimedelta(arr.view(np.int64))
 937         elif dtype == np.int64:
 938             if isna(arr).any():
 939                 raise ValueError("Cannot convert NaT values to integer")
 940             return arr.view(dtype)
 941 
 942         if dtype not in [INT64_DTYPE, TD64NS_DTYPE]:
 943 
 944             # allow frequency conversions
 945             # we return a float here!
 946             if dtype.kind == "m":
 947                 mask = isna(arr)
 948                 result = arr.astype(dtype).astype(np.float64)
 949                 result[mask] = np.nan
 950                 return result
 951         elif dtype == TD64NS_DTYPE:
 952             return arr.astype(TD64NS_DTYPE, copy=copy)
 953 
 954         raise TypeError(f"cannot astype a timedelta from [{arr.dtype}] to [{dtype}]")
 955 
 956     elif np.issubdtype(arr.dtype, np.floating) and np.issubdtype(dtype, np.integer):
 957 
 958         if not np.isfinite(arr).all():
 959             raise ValueError("Cannot convert non-finite values (NA or inf) to integer")
 960 
 961     elif is_object_dtype(arr):
 962 
 963         # work around NumPy brokenness, #1987
 964         if np.issubdtype(dtype.type, np.integer):
 965             return lib.astype_intsafe(arr.ravel(), dtype).reshape(arr.shape)
 966 
 967         # if we have a datetime/timedelta array of objects
 968         # then coerce to a proper dtype and recall astype_nansafe
 969 
 970         elif is_datetime64_dtype(dtype):
 971             from pandas import to_datetime
 972 
 973             return astype_nansafe(to_datetime(arr).values, dtype, copy=copy)
 974         elif is_timedelta64_dtype(dtype):
 975             from pandas import to_timedelta
 976 
 977             return astype_nansafe(to_timedelta(arr)._values, dtype, copy=copy)
 978 
 979     if dtype.name in ("datetime64", "timedelta64"):
 980         msg = (
 981             f"The '{dtype.name}' dtype has no unit. Please pass in "
 982             f"'{dtype.name}[ns]' instead."
 983         )
 984         raise ValueError(msg)
 985 
 986     if copy or is_object_dtype(arr) or is_object_dtype(dtype):
 987         # Explicit copy, or required since NumPy can't view from / to object.
 988         return arr.astype(dtype, copy=True)
 989 
 990     return arr.view(dtype)

 993 def maybe_convert_objects(values: np.ndarray, convert_numeric: bool = True):
 994     """
 995     If we have an object dtype array, try to coerce dates and/or numbers.
 996 
 997     Parameters
 998     ----------
 999     values : ndarray
1000     convert_numeric : bool, default True
1001 
1002     Returns
1003     -------
1004     ndarray or DatetimeIndex
1005     """
1006     validate_bool_kwarg(convert_numeric, "convert_numeric")
1007 
1008     orig_values = values
1009 
1010     # convert dates
1011     if is_object_dtype(values.dtype):
1012         values = lib.maybe_convert_objects(values, convert_datetime=True)
1013 
1014     # convert timedeltas
1015     if is_object_dtype(values.dtype):
1016         values = lib.maybe_convert_objects(values, convert_timedelta=True)
1017 
1018     # convert to numeric
1019     if is_object_dtype(values.dtype):
1020         if convert_numeric:
1021             try:
1022                 new_values = lib.maybe_convert_numeric(
1023                     values, set(), coerce_numeric=True
1024                 )
1025             except (ValueError, TypeError):
1026                 pass
1027             else:
1028                 # if we are all nans then leave me alone
1029                 if not isna(new_values).all():
1030                     values = new_values
1031 
1032         else:
1033             # soft-conversion
1034             values = lib.maybe_convert_objects(values)
1035 
1036     if values is orig_values:
1037         values = values.copy()
1038 
1039     return values

1042 def soft_convert_objects(
1043     values: np.ndarray,
1044     datetime: bool = True,
1045     numeric: bool = True,
1046     timedelta: bool = True,
1047     coerce: bool = False,
1048     copy: bool = True,
1049 ):
1050     """ if we have an object dtype, try to coerce dates and/or numbers """
1051     validate_bool_kwarg(datetime, "datetime")
1052     validate_bool_kwarg(numeric, "numeric")
1053     validate_bool_kwarg(timedelta, "timedelta")
1054     validate_bool_kwarg(coerce, "coerce")
1055     validate_bool_kwarg(copy, "copy")
1056 
1057     conversion_count = sum((datetime, numeric, timedelta))
1058     if conversion_count == 0:
1059         raise ValueError("At least one of datetime, numeric or timedelta must be True.")
1060     elif conversion_count > 1 and coerce:
1061         raise ValueError(
1062             "Only one of 'datetime', 'numeric' or "
1063             "'timedelta' can be True when when coerce=True."
1064         )
1065 
1066     if not is_object_dtype(values.dtype):
1067         # If not object, do not attempt conversion
1068         values = values.copy() if copy else values
1069         return values
1070 
1071     # If 1 flag is coerce, ensure 2 others are False
1072     if coerce:
1073         # Immediate return if coerce
1074         if datetime:
1075             from pandas import to_datetime
1076 
1077             return to_datetime(values, errors="coerce").to_numpy()
1078         elif timedelta:
1079             from pandas import to_timedelta
1080 
1081             return to_timedelta(values, errors="coerce").to_numpy()
1082         elif numeric:
1083             from pandas import to_numeric
1084 
1085             return to_numeric(values, errors="coerce")
1086 
1087     # Soft conversions
1088     if datetime:
1089         # GH 20380, when datetime is beyond year 2262, hence outside
1090         # bound of nanosecond-resolution 64-bit integers.
1091         try:
1092             values = lib.maybe_convert_objects(values, convert_datetime=True)
1093         except OutOfBoundsDatetime:
1094             pass
1095 
1096     if timedelta and is_object_dtype(values.dtype):
1097         # Object check to ensure only run if previous did not convert
1098         values = lib.maybe_convert_objects(values, convert_timedelta=True)
1099 
1100     if numeric and is_object_dtype(values.dtype):
1101         try:
1102             converted = lib.maybe_convert_numeric(values, set(), coerce_numeric=True)
1103         except (ValueError, TypeError):
1104             pass
1105         else:
1106             # If all NaNs, then do not-alter
1107             values = converted if not isna(converted).all() else values
1108             values = values.copy() if copy else values
1109 
1110     return values

1113 def convert_dtypes(
1114     input_array,
1115     convert_string: bool = True,
1116     convert_integer: bool = True,
1117     convert_boolean: bool = True,
1118 ) -> Dtype:
1119     """
1120     Convert objects to best possible type, and optionally,
1121     to types supporting ``pd.NA``.
1122 
1123     Parameters
1124     ----------
1125     input_array : ExtensionArray or PandasArray
1126     convert_string : bool, default True
1127         Whether object dtypes should be converted to ``StringDtype()``.
1128     convert_integer : bool, default True
1129         Whether, if possible, conversion can be done to integer extension types.
1130     convert_boolean : bool, defaults True
1131         Whether object dtypes should be converted to ``BooleanDtypes()``.
1132 
1133     Returns
1134     -------
1135     dtype
1136         new dtype
1137     """
1138     is_extension = is_extension_array_dtype(input_array.dtype)
1139     if (convert_string or convert_integer or convert_boolean) and not is_extension:
1140         try:
1141             inferred_dtype = lib.infer_dtype(input_array)
1142         except ValueError:
1143             # Required to catch due to Period.  Can remove once GH 23553 is fixed
1144             inferred_dtype = input_array.dtype
1145 
1146         if not convert_string and is_string_dtype(inferred_dtype):
1147             inferred_dtype = input_array.dtype
1148 
1149         if convert_integer:
1150             target_int_dtype = "Int64"
1151 
1152             if is_integer_dtype(input_array.dtype):
1153                 from pandas.core.arrays.integer import _dtypes
1154 
1155                 inferred_dtype = _dtypes.get(input_array.dtype.name, target_int_dtype)
1156             if not is_integer_dtype(input_array.dtype) and is_numeric_dtype(
1157                 input_array.dtype
1158             ):
1159                 inferred_dtype = target_int_dtype
1160 
1161         else:
1162             if is_integer_dtype(inferred_dtype):
1163                 inferred_dtype = input_array.dtype
1164 
1165         if convert_boolean:
1166             if is_bool_dtype(input_array.dtype):
1167                 inferred_dtype = "boolean"
1168         else:
1169             if isinstance(inferred_dtype, str) and inferred_dtype == "boolean":
1170                 inferred_dtype = input_array.dtype
1171 
1172     else:
1173         inferred_dtype = input_array.dtype
1174 
1175     return inferred_dtype

1192 def maybe_infer_to_datetimelike(value, convert_dates: bool = False):
1193     """
1194     we might have a array (or single object) that is datetime like,
1195     and no dtype is passed don't change the value unless we find a
1196     datetime/timedelta set
1197 
1198     this is pretty strict in that a datetime/timedelta is REQUIRED
1199     in addition to possible nulls/string likes
1200 
1201     Parameters
1202     ----------
1203     value : np.array / Series / Index / list-like
1204     convert_dates : bool, default False
1205        if True try really hard to convert dates (such as datetime.date), other
1206        leave inferred dtype 'date' alone
1207 
1208     """
1209     # TODO: why not timedelta?
1210     if isinstance(
1211         value, (ABCDatetimeIndex, ABCPeriodIndex, ABCDatetimeArray, ABCPeriodArray)
1212     ):
1213         return value
1214     elif isinstance(value, ABCSeries):
1215         if isinstance(value._values, ABCDatetimeIndex):
1216             return value._values
1217 
1218     v = value
1219 
1220     if not is_list_like(v):
1221         v = [v]
1222     v = np.array(v, copy=False)
1223 
1224     # we only care about object dtypes
1225     if not is_object_dtype(v):
1226         return value
1227 
1228     shape = v.shape
1229     if not v.ndim == 1:
1230         v = v.ravel()
1231 
1232     if not len(v):
1233         return value
1234 
1235     def try_datetime(v):
1236         # safe coerce to datetime64
1237         try:
1238             # GH19671
1239             v = tslib.array_to_datetime(v, require_iso8601=True, errors="raise")[0]
1240         except ValueError:
1241 
1242             # we might have a sequence of the same-datetimes with tz's
1243             # if so coerce to a DatetimeIndex; if they are not the same,
1244             # then these stay as object dtype, xref GH19671
1245             from pandas._libs.tslibs import conversion
1246             from pandas import DatetimeIndex
1247 
1248             try:
1249 
1250                 values, tz = conversion.datetime_to_datetime64(v)
1251                 return DatetimeIndex(values).tz_localize("UTC").tz_convert(tz=tz)
1252             except (ValueError, TypeError):
1253                 pass
1254 
1255         except Exception:
1256             pass
1257 
1258         return v.reshape(shape)
1259 
1260     def try_timedelta(v):
1261         # safe coerce to timedelta64
1262 
1263         # will try first with a string & object conversion
1264         from pandas import to_timedelta
1265 
1266         try:
1267             td_values = to_timedelta(v)
1268         except ValueError:
1269             return v.reshape(shape)
1270         else:
1271             return np.asarray(td_values).reshape(shape)
1272 
1273     inferred_type = lib.infer_datetimelike_array(ensure_object(v))
1274 
1275     if inferred_type == "date" and convert_dates:
1276         value = try_datetime(v)
1277     elif inferred_type == "datetime":
1278         value = try_datetime(v)
1279     elif inferred_type == "timedelta":
1280         value = try_timedelta(v)
1281     elif inferred_type == "nat":
1282 
1283         # if all NaT, return as datetime
1284         if isna(v).all():
1285             value = try_datetime(v)
1286         else:
1287 
1288             # We have at least a NaT and a string
1289             # try timedelta first to avoid spurious datetime conversions
1290             # e.g. '00:00:01' is a timedelta but technically is also a datetime
1291             value = try_timedelta(v)
1292             if lib.infer_dtype(value, skipna=False) in ["mixed"]:
1293                 # cannot skip missing values, as NaT implies that the string
1294                 # is actually a datetime
1295                 value = try_datetime(v)
1296 
1297     return value

1300 def maybe_cast_to_datetime(value, dtype, errors: str = "raise"):
1301     """
1302     try to cast the array/value to a datetimelike dtype, converting float
1303     nan to iNaT
1304     """
1305     from pandas.core.tools.timedeltas import to_timedelta
1306     from pandas.core.tools.datetimes import to_datetime
1307 
1308     if dtype is not None:
1309         if isinstance(dtype, str):
1310             dtype = np.dtype(dtype)
1311 
1312         is_datetime64 = is_datetime64_dtype(dtype)
1313         is_datetime64tz = is_datetime64tz_dtype(dtype)
1314         is_timedelta64 = is_timedelta64_dtype(dtype)
1315 
1316         if is_datetime64 or is_datetime64tz or is_timedelta64:
1317 
1318             # Force the dtype if needed.
1319             msg = (
1320                 f"The '{dtype.name}' dtype has no unit. "
1321                 f"Please pass in '{dtype.name}[ns]' instead."
1322             )
1323 
1324             if is_datetime64 and not is_dtype_equal(dtype, DT64NS_DTYPE):
1325 
1326                 # pandas supports dtype whose granularity is less than [ns]
1327                 # e.g., [ps], [fs], [as]
1328                 if dtype <= np.dtype("M8[ns]"):
1329                     if dtype.name == "datetime64":
1330                         raise ValueError(msg)
1331                     dtype = DT64NS_DTYPE
1332                 else:
1333                     raise TypeError(f"cannot convert datetimelike to dtype [{dtype}]")
1334             elif is_datetime64tz:
1335 
1336                 # our NaT doesn't support tz's
1337                 # this will coerce to DatetimeIndex with
1338                 # a matching dtype below
1339                 if is_scalar(value) and isna(value):
1340                     value = [value]
1341 
1342             elif is_timedelta64 and not is_dtype_equal(dtype, TD64NS_DTYPE):
1343 
1344                 # pandas supports dtype whose granularity is less than [ns]
1345                 # e.g., [ps], [fs], [as]
1346                 if dtype <= np.dtype("m8[ns]"):
1347                     if dtype.name == "timedelta64":
1348                         raise ValueError(msg)
1349                     dtype = TD64NS_DTYPE
1350                 else:
1351                     raise TypeError(f"cannot convert timedeltalike to dtype [{dtype}]")
1352 
1353             if is_scalar(value):
1354                 if value == iNaT or isna(value):
1355                     value = iNaT
1356             else:
1357                 value = np.array(value, copy=False)
1358 
1359                 # have a scalar array-like (e.g. NaT)
1360                 if value.ndim == 0:
1361                     value = iNaT
1362 
1363                 # we have an array of datetime or timedeltas & nulls
1364                 elif np.prod(value.shape) or not is_dtype_equal(value.dtype, dtype):
1365                     try:
1366                         if is_datetime64:
1367                             value = to_datetime(value, errors=errors)
1368                             # GH 25843: Remove tz information since the dtype
1369                             # didn't specify one
1370                             if value.tz is not None:
1371                                 value = value.tz_localize(None)
1372                             value = value._values
1373                         elif is_datetime64tz:
1374                             # The string check can be removed once issue #13712
1375                             # is solved. String data that is passed with a
1376                             # datetime64tz is assumed to be naive which should
1377                             # be localized to the timezone.
1378                             is_dt_string = is_string_dtype(value.dtype)
1379                             value = to_datetime(value, errors=errors).array
1380                             if is_dt_string:
1381                                 # Strings here are naive, so directly localize
1382                                 value = value.tz_localize(dtype.tz)
1383                             else:
1384                                 # Numeric values are UTC at this point,
1385                                 # so localize and convert
1386                                 value = value.tz_localize("UTC").tz_convert(dtype.tz)
1387                         elif is_timedelta64:
1388                             value = to_timedelta(value, errors=errors)._values
1389                     except OutOfBoundsDatetime:
1390                         raise
1391                     except (AttributeError, ValueError, TypeError):
1392                         pass
1393 
1394         # coerce datetimelike to object
1395         elif is_datetime64_dtype(
1396             getattr(value, "dtype", None)
1397         ) and not is_datetime64_dtype(dtype):
1398             if is_object_dtype(dtype):
1399                 if value.dtype != DT64NS_DTYPE:
1400                     value = value.astype(DT64NS_DTYPE)
1401                 ints = np.asarray(value).view("i8")
1402                 return tslib.ints_to_pydatetime(ints)
1403 
1404             # we have a non-castable dtype that was passed
1405             raise TypeError(f"Cannot cast datetime64 to {dtype}")
1406 
1407     else:
1408 
1409         is_array = isinstance(value, np.ndarray)
1410 
1411         # catch a datetime/timedelta that is not of ns variety
1412         # and no coercion specified
1413         if is_array and value.dtype.kind in ["M", "m"]:
1414             dtype = value.dtype
1415 
1416             if dtype.kind == "M" and dtype != DT64NS_DTYPE:
1417                 value = tslibs.conversion.ensure_datetime64ns(value)
1418 
1419             elif dtype.kind == "m" and dtype != TD64NS_DTYPE:
1420                 value = to_timedelta(value)
1421 
1422         # only do this if we have an array and the dtype of the array is not
1423         # setup already we are not an integer/object, so don't bother with this
1424         # conversion
1425         elif not (
1426             is_array
1427             and not (
1428                 issubclass(value.dtype.type, np.integer) or value.dtype == np.object_
1429             )
1430         ):
1431             value = maybe_infer_to_datetimelike(value)
1432 
1433     return value

1436 def find_common_type(types: List[DtypeObj]) -> DtypeObj:
1437     """
1438     Find a common data type among the given dtypes.
1439 
1440     Parameters
1441     ----------
1442     types : list of dtypes
1443 
1444     Returns
1445     -------
1446     pandas extension or numpy dtype
1447 
1448     See Also
1449     --------
1450     numpy.find_common_type
1451 
1452     """
1453     if len(types) == 0:
1454         raise ValueError("no types given")
1455 
1456     first = types[0]
1457 
1458     # workaround for find_common_type([np.dtype('datetime64[ns]')] * 2)
1459     # => object
1460     if all(is_dtype_equal(first, t) for t in types[1:]):
1461         return first
1462 
1463     # get unique types (dict.fromkeys is used as order-preserving set())
1464     types = list(dict.fromkeys(types).keys())
1465 
1466     if any(isinstance(t, ExtensionDtype) for t in types):
1467         for t in types:
1468             if isinstance(t, ExtensionDtype):
1469                 res = t._get_common_dtype(types)
1470                 if res is not None:
1471                     return res
1472         return np.dtype("object")
1473 
1474     # take lowest unit
1475     if all(is_datetime64_dtype(t) for t in types):
1476         return np.dtype("datetime64[ns]")
1477     if all(is_timedelta64_dtype(t) for t in types):
1478         return np.dtype("timedelta64[ns]")
1479 
1480     # don't mix bool / int or float or complex
1481     # this is different from numpy, which casts bool with float/int as int
1482     has_bools = any(is_bool_dtype(t) for t in types)
1483     if has_bools:
1484         for t in types:
1485             if is_integer_dtype(t) or is_float_dtype(t) or is_complex_dtype(t):
1486                 return np.object
1487 
1488     return np.find_common_type(types, [])

1585 ) -> np.ndarray:

1588

1598     -------

1626 def maybe_cast_to_integer_array(arr, dtype, copy: bool = False):
1627     """
1628     Takes any dtype and returns the casted version, raising for when data is
1629     incompatible with integer/unsigned integer dtypes.
1630 
1631     .. versionadded:: 0.24.0
1632 
1633     Parameters
1634     ----------
1635     arr : array-like
1636         The array to cast.
1637     dtype : str, np.dtype
1638         The integer dtype to cast the array to.
1639     copy: bool, default False
1640         Whether to make a copy of the array before returning.
1641 
1642     Returns
1643     -------
1644     ndarray
1645         Array of integer or unsigned integer dtype.
1646 
1647     Raises
1648     ------
1649     OverflowError : the dtype is incompatible with the data
1650     ValueError : loss of precision has occurred during casting
1651 
1652     Examples
1653     --------
1654     If you try to coerce negative values to unsigned integers, it raises:
1655 
1656     >>> pd.Series([-1], dtype="uint64")
1657     Traceback (most recent call last):
1658         ...
1659     OverflowError: Trying to coerce negative values to unsigned integers
1660 
1661     Also, if you try to coerce float values to integers, it raises:
1662 
1663     >>> pd.Series([1, 2, 3.5], dtype="int64")
1664     Traceback (most recent call last):
1665         ...
1666     ValueError: Trying to coerce float values to integers
1667     """
1668     try:
1669         if not hasattr(arr, "astype"):
1670             casted = np.array(arr, dtype=dtype, copy=copy)
1671         else:
1672             casted = arr.astype(dtype, copy=copy)
1673     except OverflowError as err:
1674         raise OverflowError(
1675             "The elements provided in the data cannot all be "
1676             f"casted to the dtype {dtype}"
1677         ) from err
1678 
1679     if np.array_equal(arr, casted):
1680         return casted
1681 
1682     # We do this casting to allow for proper
1683     # data and dtype checking.
1684     #
1685     # We didn't do this earlier because NumPy
1686     # doesn't handle `uint64` correctly.
1687     arr = np.asarray(arr)
1688 
1689     if is_unsigned_integer_dtype(dtype) and (arr < 0).any():
1690         raise OverflowError("Trying to coerce negative values to unsigned integers")
1691 
1692     if is_integer_dtype(dtype) and (is_float_dtype(arr) or is_object_dtype(arr)):
1693         raise ValueError("Trying to coerce float values to integers")

1696 def convert_scalar_for_putitemlike(scalar, dtype: np.dtype):
1697     """
1698     Convert datetimelike scalar if we are setting into a datetime64
1699     or timedelta64 ndarray.
1700 
1701     Parameters
1702     ----------
1703     scalar : scalar
1704     dtype : np.dtype
1705 
1706     Returns
1707     -------
1708     scalar
1709     """
1710     if dtype.kind == "m":
1711         if isinstance(scalar, (timedelta, np.timedelta64)):
1712             # We have to cast after asm8 in case we have NaT
1713             return Timedelta(scalar).asm8.view("timedelta64[ns]")
1714         elif scalar is None or scalar is NaT or (is_float(scalar) and np.isnan(scalar)):
1715             return np.timedelta64("NaT", "ns")
1716     if dtype.kind == "M":
1717         if isinstance(scalar, (date, np.datetime64)):
1718             # Note: we include date, not just datetime
1719             return Timestamp(scalar).to_datetime64()
1720         elif scalar is None or scalar is NaT or (is_float(scalar) and np.isnan(scalar)):
1721             return np.datetime64("NaT", "ns")
1722     else:
1723         validate_numeric_casting(dtype, scalar)
1724     return scalar

1727 def validate_numeric_casting(dtype: np.dtype, value):
1728     """
1729     Check that we can losslessly insert the given value into an array
1730     with the given dtype.
1731 
1732     Parameters
1733     ----------
1734     dtype : np.dtype
1735     value : scalar
1736 
1737     Raises
1738     ------
1739     ValueError
1740     """
1741     if issubclass(dtype.type, (np.integer, np.bool_)):
1742         if is_float(value) and np.isnan(value):
1743             raise ValueError("Cannot assign nan to integer series")
1744 
1745     if issubclass(dtype.type, (np.integer, np.floating, np.complex)) and not issubclass(
1746         dtype.type, np.bool_
1747     ):
1748         if is_bool(value):
1749             raise ValueError("Cannot assign bool to float/integer series")
