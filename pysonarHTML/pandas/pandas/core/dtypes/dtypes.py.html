  43 def register_extension_dtype(cls: Type[ExtensionDtype]) -> Type[ExtensionDtype]:
  44     """
  45     Register an ExtensionType with pandas as class decorator.
  46 
  47     .. versionadded:: 0.24.0
  48 
  49     This enables operations like ``.astype(name)`` for the name
  50     of the ExtensionDtype.
  51 
  52     Returns
  53     -------
  54     callable
  55         A class decorator.
  56 
  57     Examples
  58     --------
  59     >>> from pandas.api.extensions import register_extension_dtype
  60     >>> from pandas.api.extensions import ExtensionDtype
  61     >>> @register_extension_dtype
  62     ... class MyExtensionDtype(ExtensionDtype):
  63     ...     pass
  64     """
  65     registry.register(cls)
  66     return cls

  69 class Registry:
  70     """
  71     Registry for dtype inference.
  72 
  73     The registry allows one to map a string repr of a extension
  74     dtype to an extension dtype. The string alias can be used in several
  75     places, including
  76 
  77     * Series and Index constructors
  78     * :meth:`pandas.array`
  79     * :meth:`pandas.Series.astype`
  80 
  81     Multiple extension types can be registered.
  82     These are tried in order.
  83     """
  84 
  85     def __init__(self):
  86         self.dtypes: List[Type[ExtensionDtype]] = []
  87 
  88     def register(self, dtype: Type[ExtensionDtype]) -> None:
  89         """
  90         Parameters
  91         ----------
  92         dtype : ExtensionDtype class
  93         """
  94         if not issubclass(dtype, ExtensionDtype):
  95             raise ValueError("can only register pandas extension dtypes")
  96 
  97         self.dtypes.append(dtype)
  98 
  99     def find(
 100         self, dtype: Union[Type[ExtensionDtype], str]
 101     ) -> Optional[Type[ExtensionDtype]]:
 102         """
 103         Parameters
 104         ----------
 105         dtype : Type[ExtensionDtype] or str
 106 
 107         Returns
 108         -------
 109         return the first matching dtype, otherwise return None
 110         """
 111         if not isinstance(dtype, str):
 112             dtype_type = dtype
 113             if not isinstance(dtype, type):
 114                 dtype_type = type(dtype)
 115             if issubclass(dtype_type, ExtensionDtype):
 116                 return dtype
 117 
 118             return None
 119 
 120         for dtype_type in self.dtypes:
 121             try:
 122                 return dtype_type.construct_from_string(dtype)
 123             except TypeError:
 124                 pass
 125 
 126         return None
 127

 189 class CategoricalDtype(PandasExtensionDtype, ExtensionDtype):
 190     """
 191     Type for categorical data with the categories and orderedness.
 192 
 193     Parameters
 194     ----------
 195     categories : sequence, optional
 196         Must be unique, and must not contain any nulls.
 197         The categories are stored in an Index,
 198         and if an index is provided the dtype of that index will be used.
 199     ordered : bool or None, default False
 200         Whether or not this categorical is treated as a ordered categorical.
 201         None can be used to maintain the ordered value of existing categoricals when
 202         used in operations that combine categoricals, e.g. astype, and will resolve to
 203         False if there is no existing ordered to maintain.
 204 
 205     Attributes
 206     ----------
 207     categories
 208     ordered
 209 
 210     Methods
 211     -------
 212     None
 213 
 214     See Also
 215     --------
 216     Categorical : Represent a categorical variable in classic R / S-plus fashion.
 217 
 218     Notes
 219     -----
 220     This class is useful for specifying the type of a ``Categorical``
 221     independent of the values. See :ref:`categorical.categoricaldtype`
 222     for more.
 223 
 224     Examples
 225     --------
 226     >>> t = pd.CategoricalDtype(categories=['b', 'a'], ordered=True)
 227     >>> pd.Series(['a', 'b', 'a', 'c'], dtype=t)
 228     0      a
 229     1      b
 230     2      a
 231     3    NaN
 232     dtype: category
 233     Categories (2, object): [b < a]
 234 
 235     An empty CategoricalDtype with a specific dtype can be created
 236     by providing an empty index. As follows,
 237 
 238     >>> pd.CategoricalDtype(pd.DatetimeIndex([])).categories.dtype
 239     dtype('<M8[ns]')
 240     """
 241 
 242     # TODO: Document public vs. private API
 243     name = "category"
 244     type: Type[CategoricalDtypeType] = CategoricalDtypeType
 245     kind: str_type = "O"
 246     str = "|O08"
 247     base = np.dtype("O")
 248     _metadata = ("categories", "ordered")
 249     _cache: Dict[str_type, PandasExtensionDtype] = {}

 262     @classmethod
 263     def _from_categorical_dtype(
 264         cls, dtype: "CategoricalDtype", categories=None, ordered: Ordered = None
 265     ) -> "CategoricalDtype":
 266         if categories is ordered is None:
 267             return dtype
 268         if categories is None:
 269             categories = dtype.categories
 270         if ordered is None:
 271             ordered = dtype.ordered
 272         return cls(categories, ordered)

 274     @classmethod
 275     def _from_values_or_dtype(
 276         cls,
 277         values=None,
 278         categories=None,
 279         ordered: Optional[bool] = None,
 280         dtype: Optional["CategoricalDtype"] = None,
 281     ) -> "CategoricalDtype":
 282         """
 283         Construct dtype from the input parameters used in :class:`Categorical`.
 284 
 285         This constructor method specifically does not do the factorization
 286         step, if that is needed to find the categories. This constructor may
 287         therefore return ``CategoricalDtype(categories=None, ordered=None)``,
 288         which may not be useful. Additional steps may therefore have to be
 289         taken to create the final dtype.
 290 
 291         The return dtype is specified from the inputs in this prioritized
 292         order:
 293         1. if dtype is a CategoricalDtype, return dtype
 294         2. if dtype is the string 'category', create a CategoricalDtype from
 295            the supplied categories and ordered parameters, and return that.
 296         3. if values is a categorical, use value.dtype, but override it with
 297            categories and ordered if either/both of those are not None.
 298         4. if dtype is None and values is not a categorical, construct the
 299            dtype from categories and ordered, even if either of those is None.
 300 
 301         Parameters
 302         ----------
 303         values : list-like, optional
 304             The list-like must be 1-dimensional.
 305         categories : list-like, optional
 306             Categories for the CategoricalDtype.
 307         ordered : bool, optional
 308             Designating if the categories are ordered.
 309         dtype : CategoricalDtype or the string "category", optional
 310             If ``CategoricalDtype``, cannot be used together with
 311             `categories` or `ordered`.
 312 
 313         Returns
 314         -------
 315         CategoricalDtype
 316 
 317         Examples
 318         --------
 319         >>> pd.CategoricalDtype._from_values_or_dtype()
 320         CategoricalDtype(categories=None, ordered=None)
 321         >>> pd.CategoricalDtype._from_values_or_dtype(
 322         ...     categories=['a', 'b'], ordered=True
 323         ... )
 324         CategoricalDtype(categories=['a', 'b'], ordered=True)
 325         >>> dtype1 = pd.CategoricalDtype(['a', 'b'], ordered=True)
 326         >>> dtype2 = pd.CategoricalDtype(['x', 'y'], ordered=False)
 327         >>> c = pd.Categorical([0, 1], dtype=dtype1, fastpath=True)
 328         >>> pd.CategoricalDtype._from_values_or_dtype(
 329         ...     c, ['x', 'y'], ordered=True, dtype=dtype2
 330         ... )
 331         Traceback (most recent call last):
 332             ...
 333         ValueError: Cannot specify `categories` or `ordered` together with
 334         `dtype`.
 335 
 336         The supplied dtype takes precedence over values' dtype:
 337 
 338         >>> pd.CategoricalDtype._from_values_or_dtype(c, dtype=dtype2)
 339         CategoricalDtype(categories=['x', 'y'], ordered=False)
 340         """
 341 
 342         if dtype is not None:
 343             # The dtype argument takes precedence over values.dtype (if any)
 344             if isinstance(dtype, str):
 345                 if dtype == "category":
 346                     dtype = CategoricalDtype(categories, ordered)
 347                 else:
 348                     raise ValueError(f"Unknown dtype {repr(dtype)}")
 349             elif categories is not None or ordered is not None:
 350                 raise ValueError(
 351                     "Cannot specify `categories` or `ordered` together with `dtype`."
 352                 )
 353             elif not isinstance(dtype, CategoricalDtype):
 354                 raise ValueError(f"Cannot not construct CategoricalDtype from {dtype}")
 355         elif cls.is_dtype(values):
 356             # If no "dtype" was passed, use the one from "values", but honor
 357             # the "ordered" and "categories" arguments
 358             dtype = values.dtype._from_categorical_dtype(
 359                 values.dtype, categories, ordered
 360             )
 361         else:
 362             # If dtype=None and values is not categorical, create a new dtype.
 363             # Note: This could potentially have categories=None and
 364             # ordered=None.
 365             dtype = CategoricalDtype(categories, ordered)
 366 
 367         return dtype

 371         """
 372         Construct a CategoricalDtype from a string.
 373 
 374         Parameters

 400     def _finalize(self, categories, ordered: Ordered, fastpath: bool = False) -> None:
 401 
 402         if ordered is not None:
 403             self.validate_ordered(ordered)

 405         if categories is not None:
 406             categories = self.validate_categories(categories, fastpath=fastpath)

 409         self._ordered = ordered

 418     def __hash__(self) -> int:
 419         # _hash_categories returns a uint64, so use the negative
 420         # space for when we have unknown categories to avoid a conflict
 421         if self.categories is None:
 422             if self.ordered:
 423                 return -1
 424             else:
 425                 return -2
 426         # We *do* want to include the real self.ordered here
 427         return int(self._hash_categories(self.categories, self.ordered))

 429     def __eq__(self, other: Any) -> bool:
 430         """
 431         Rules for CDT equality:
 432         1) Any CDT is equal to the string 'category'
 433         2) Any CDT is equal to itself
 434         3) Any CDT is equal to a CDT with categories=None regardless of ordered
 435         4) A CDT with ordered=True is only equal to another CDT with
 436            ordered=True and identical categories in the same order
 437         5) A CDT with ordered={False, None} is only equal to another CDT with
 438            ordered={False, None} and identical categories, but same order is
 439            not required. There is no distinction between False/None.
 440         6) Any other comparison returns False
 441         """
 442         if isinstance(other, str):
 443             return other == self.name
 444         elif other is self:
 445             return True
 446         elif not (hasattr(other, "ordered") and hasattr(other, "categories")):
 447             return False
 448         elif self.categories is None or other.categories is None:
 449             # We're forced into a suboptimal corner thanks to math and
 450             # backwards compatibility. We require that `CDT(...) == 'category'`
 451             # for all CDTs **including** `CDT(None, ...)`. Therefore, *all*
 452             # CDT(., .) = CDT(None, False) and *all*
 453             # CDT(., .) = CDT(None, True).
 454             return True
 455         elif self.ordered or other.ordered:
 456             # At least one has ordered=True; equal if both have ordered=True
 457             # and the same values for categories in the same order.
 458             return (self.ordered == other.ordered) and self.categories.equals(
 459                 other.categories
 460             )
 461         else:
 462             # Neither has ordered=True; equal if both have the same categories,
 463             # but same order is not necessary.  There is no distinction between
 464             # ordered=False and ordered=None: CDT(., False) and CDT(., None)
 465             # will be equal if they have the same categories.
 466             if (
 467                 self.categories.dtype == other.categories.dtype
 468                 and self.categories.equals(other.categories)
 469             ):
 470                 # Check and see if they happen to be identical categories
 471                 return True
 472             return hash(self) == hash(other)

 481     @staticmethod
 482     def _hash_categories(categories, ordered: Ordered = True) -> int:
 483         from pandas.core.util.hashing import (
 484             hash_array,
 485             _combine_hash_arrays,
 486             hash_tuples,
 487         )
 488         from pandas.core.dtypes.common import is_datetime64tz_dtype, DT64NS_DTYPE
 489 
 490         if len(categories) and isinstance(categories[0], tuple):
 491             # assumes if any individual category is a tuple, then all our. ATM
 492             # I don't really want to support just some of the categories being
 493             # tuples.
 494             categories = list(categories)  # breaks if a np.array of categories
 495             cat_array = hash_tuples(categories)
 496         else:
 497             if categories.dtype == "O":
 498                 if len({type(x) for x in categories}) != 1:
 499                     # TODO: hash_array doesn't handle mixed types. It casts
 500                     # everything to a str first, which means we treat
 501                     # {'1', '2'} the same as {'1', 2}
 502                     # find a better solution
 503                     hashed = hash((tuple(categories), ordered))
 504                     return hashed
 505 
 506             if is_datetime64tz_dtype(categories.dtype):
 507                 # Avoid future warning.
 508                 categories = categories.astype(DT64NS_DTYPE)
 509 
 510             cat_array = hash_array(np.asarray(categories), categorize=False)
 511         if ordered:
 512             cat_array = np.vstack(
 513                 [cat_array, np.arange(len(cat_array), dtype=cat_array.dtype)]
 514             )
 515         else:
 516             cat_array = [cat_array]
 517         hashed = _combine_hash_arrays(iter(cat_array), num_items=len(cat_array))
 518         return np.bitwise_xor.reduce(hashed)

 552     @staticmethod
 553     def validate_categories(categories, fastpath: bool = False):
 554         """
 555         Validates that we have good categories
 556 
 557         Parameters
 558         ----------
 559         categories : array-like
 560         fastpath : bool
 561             Whether to skip nan and uniqueness checks
 562 
 563         Returns
 564         -------
 565         categories : Index
 566         """
 567         from pandas.core.indexes.base import Index
 568 
 569         if not fastpath and not is_list_like(categories):
 570             raise TypeError(
 571                 f"Parameter 'categories' must be list-like, was {repr(categories)}"
 572             )
 573         elif not isinstance(categories, ABCIndexClass):
 574             categories = Index(categories, tupleize_cols=False)
 575 
 576         if not fastpath:
 577 
 578             if categories.hasnans:
 579                 raise ValueError("Categorical categories cannot be null")
 580 
 581             if not categories.is_unique:
 582                 raise ValueError("Categorical categories must be unique")
 583 
 584         if isinstance(categories, ABCCategoricalIndex):
 585             categories = categories.categories
 586 
 587         return categories

 589     def update_dtype(
 590         self, dtype: Union[str_type, "CategoricalDtype"]
 591     ) -> "CategoricalDtype":
 592         """
 593         Returns a CategoricalDtype with categories and ordered taken from dtype
 594         if specified, otherwise falling back to self if unspecified
 595 
 596         Parameters
 597         ----------
 598         dtype : CategoricalDtype
 599 
 600         Returns
 601         -------
 602         new_dtype : CategoricalDtype
 603         """
 604         if isinstance(dtype, str) and dtype == "category":
 605             # dtype='category' should not change anything
 606             return self
 607         elif not self.is_dtype(dtype):
 608             raise ValueError(
 609                 f"a CategoricalDtype must be passed to perform an update, "
 610                 f"got {repr(dtype)}"
 611             )
 612         else:
 613             # from here on, dtype is a CategoricalDtype
 614             dtype = cast(CategoricalDtype, dtype)
 615 
 616         # update categories/ordered unless they've been explicitly passed as None
 617         new_categories = (
 618             dtype.categories if dtype.categories is not None else self.categories
 619         )
 620         new_ordered = dtype.ordered if dtype.ordered is not None else self.ordered
 621 
 622         return CategoricalDtype(new_categories, new_ordered)

 644     def _get_common_dtype(self, dtypes: List[DtypeObj]) -> Optional[DtypeObj]:
 645         from pandas.core.arrays.sparse import SparseDtype
 646 
 647         # check if we have all categorical dtype with identical categories
 648         if all(isinstance(x, CategoricalDtype) for x in dtypes):
 649             first = dtypes[0]
 650             if all(first == other for other in dtypes[1:]):
 651                 return first
 652 
 653         # special case non-initialized categorical
 654         # TODO we should figure out the expected return value in general
 655         non_init_cats = [
 656             isinstance(x, CategoricalDtype) and x.categories is None for x in dtypes
 657         ]
 658         if all(non_init_cats):
 659             return self
 660         elif any(non_init_cats):
 661             return None
 662 
 663         # categorical is aware of Sparse -> extract sparse subdtypes
 664         dtypes = [x.subtype if isinstance(x, SparseDtype) else x for x in dtypes]
 665         # extract the categories' dtype
 666         non_cat_dtypes = [
 667             x.categories.dtype if isinstance(x, CategoricalDtype) else x for x in dtypes
 668         ]
 669         # TODO should categorical always give an answer?
 670         from pandas.core.dtypes.cast import find_common_type
 671 
 672         return find_common_type(non_cat_dtypes)
