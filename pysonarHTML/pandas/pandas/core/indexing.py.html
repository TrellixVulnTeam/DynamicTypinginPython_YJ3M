  99     def iloc(self) -> "_iLocIndexer":
 100         """
 101         Purely integer-location based indexing for selection by position.
 102 
 103         ``.iloc[]`` is primarily integer position based (from ``0`` to
 104         ``length-1`` of the axis), but may also be used with a boolean
 105         array.
 106 
 107         Allowed inputs are:
 108 
 109         - An integer, e.g. ``5``.
 110         - A list or array of integers, e.g. ``[4, 3, 0]``.
 111         - A slice object with ints, e.g. ``1:7``.
 112         - A boolean array.
 113         - A ``callable`` function with one argument (the calling Series or
 114           DataFrame) and that returns valid output for indexing (one of the above).
 115           This is useful in method chains, when you don't have a reference to the
 116           calling object, but would like to base your selection on some value.
 117 
 118         ``.iloc`` will raise ``IndexError`` if a requested indexer is
 119         out-of-bounds, except *slice* indexers which allow out-of-bounds
 120         indexing (this conforms with python/numpy *slice* semantics).
 121 
 122         See more at :ref:`Selection by Position <indexing.integer>`.
 123 
 124         See Also
 125         --------
 126         DataFrame.iat : Fast integer location scalar accessor.
 127         DataFrame.loc : Purely label-location based indexer for selection by label.
 128         Series.iloc : Purely integer-location based indexing for
 129                        selection by position.
 130 
 131         Examples
 132         --------
 133         >>> mydict = [{'a': 1, 'b': 2, 'c': 3, 'd': 4},
 134         ...           {'a': 100, 'b': 200, 'c': 300, 'd': 400},
 135         ...           {'a': 1000, 'b': 2000, 'c': 3000, 'd': 4000 }]
 136         >>> df = pd.DataFrame(mydict)
 137         >>> df
 138               a     b     c     d
 139         0     1     2     3     4
 140         1   100   200   300   400
 141         2  1000  2000  3000  4000
 142 
 143         **Indexing just the rows**
 144 
 145         With a scalar integer.
 146 
 147         >>> type(df.iloc[0])
 148         <class 'pandas.core.series.Series'>
 149         >>> df.iloc[0]
 150         a    1
 151         b    2
 152         c    3
 153         d    4
 154         Name: 0, dtype: int64
 155 
 156         With a list of integers.
 157 
 158         >>> df.iloc[[0]]
 159            a  b  c  d
 160         0  1  2  3  4
 161         >>> type(df.iloc[[0]])
 162         <class 'pandas.core.frame.DataFrame'>
 163 
 164         >>> df.iloc[[0, 1]]
 165              a    b    c    d
 166         0    1    2    3    4
 167         1  100  200  300  400
 168 
 169         With a `slice` object.
 170 
 171         >>> df.iloc[:3]
 172               a     b     c     d
 173         0     1     2     3     4
 174         1   100   200   300   400
 175         2  1000  2000  3000  4000
 176 
 177         With a boolean mask the same length as the index.
 178 
 179         >>> df.iloc[[True, False, True]]
 180               a     b     c     d
 181         0     1     2     3     4
 182         2  1000  2000  3000  4000
 183 
 184         With a callable, useful in method chains. The `x` passed
 185         to the ``lambda`` is the DataFrame being sliced. This selects
 186         the rows whose index label even.
 187 
 188         >>> df.iloc[lambda x: x.index % 2 == 0]
 189               a     b     c     d
 190         0     1     2     3     4
 191         2  1000  2000  3000  4000
 192 
 193         **Indexing both axes**
 194 
 195         You can mix the indexer types for the index and columns. Use ``:`` to
 196         select the entire axis.
 197 
 198         With scalar integers.
 199 
 200         >>> df.iloc[0, 1]
 201         2
 202 
 203         With lists of integers.
 204 
 205         >>> df.iloc[[0, 2], [1, 3]]
 206               b     d
 207         0     2     4
 208         2  2000  4000
 209 
 210         With `slice` objects.
 211 
 212         >>> df.iloc[1:3, 0:3]
 213               a     b     c
 214         1   100   200   300
 215         2  1000  2000  3000
 216 
 217         With a boolean array whose length matches the columns.
 218 
 219         >>> df.iloc[:, [True, False, True, False]]
 220               a     c
 221         0     1     3
 222         1   100   300
 223         2  1000  3000
 224 
 225         With a callable function that expects the Series or DataFrame.
 226 
 227         >>> df.iloc[:, lambda df: [0, 2]]
 228               a     c
 229         0     1     3
 230         1   100   300
 231         2  1000  3000
 232         """
 233         return _iLocIndexer("iloc", self)

 235     @property
 236     def loc(self) -> "_LocIndexer":
 237         """
 238         Access a group of rows and columns by label(s) or a boolean array.
 239 
 240         ``.loc[]`` is primarily label based, but may also be used with a
 241         boolean array.
 242 
 243         Allowed inputs are:
 244 
 245         - A single label, e.g. ``5`` or ``'a'``, (note that ``5`` is
 246           interpreted as a *label* of the index, and **never** as an
 247           integer position along the index).
 248         - A list or array of labels, e.g. ``['a', 'b', 'c']``.
 249         - A slice object with labels, e.g. ``'a':'f'``.
 250 
 251           .. warning:: Note that contrary to usual python slices, **both** the
 252               start and the stop are included
 253 
 254         - A boolean array of the same length as the axis being sliced,
 255           e.g. ``[True, False, True]``.
 256         - A ``callable`` function with one argument (the calling Series or
 257           DataFrame) and that returns valid output for indexing (one of the above)
 258 
 259         See more at :ref:`Selection by Label <indexing.label>`
 260 
 261         Raises
 262         ------
 263         KeyError
 264             If any items are not found.
 265 
 266         See Also
 267         --------
 268         DataFrame.at : Access a single value for a row/column label pair.
 269         DataFrame.iloc : Access group of rows and columns by integer position(s).
 270         DataFrame.xs : Returns a cross-section (row(s) or column(s)) from the
 271             Series/DataFrame.
 272         Series.loc : Access group of values using labels.
 273 
 274         Examples
 275         --------
 276         **Getting values**
 277 
 278         >>> df = pd.DataFrame([[1, 2], [4, 5], [7, 8]],
 279         ...      index=['cobra', 'viper', 'sidewinder'],
 280         ...      columns=['max_speed', 'shield'])
 281         >>> df
 282                     max_speed  shield
 283         cobra               1       2
 284         viper               4       5
 285         sidewinder          7       8
 286 
 287         Single label. Note this returns the row as a Series.
 288 
 289         >>> df.loc['viper']
 290         max_speed    4
 291         shield       5
 292         Name: viper, dtype: int64
 293 
 294         List of labels. Note using ``[[]]`` returns a DataFrame.
 295 
 296         >>> df.loc[['viper', 'sidewinder']]
 297                     max_speed  shield
 298         viper               4       5
 299         sidewinder          7       8
 300 
 301         Single label for row and column
 302 
 303         >>> df.loc['cobra', 'shield']
 304         2
 305 
 306         Slice with labels for row and single label for column. As mentioned
 307         above, note that both the start and stop of the slice are included.
 308 
 309         >>> df.loc['cobra':'viper', 'max_speed']
 310         cobra    1
 311         viper    4
 312         Name: max_speed, dtype: int64
 313 
 314         Boolean list with the same length as the row axis
 315 
 316         >>> df.loc[[False, False, True]]
 317                     max_speed  shield
 318         sidewinder          7       8
 319 
 320         Conditional that returns a boolean Series
 321 
 322         >>> df.loc[df['shield'] > 6]
 323                     max_speed  shield
 324         sidewinder          7       8
 325 
 326         Conditional that returns a boolean Series with column labels specified
 327 
 328         >>> df.loc[df['shield'] > 6, ['max_speed']]
 329                     max_speed
 330         sidewinder          7
 331 
 332         Callable that returns a boolean Series
 333 
 334         >>> df.loc[lambda df: df['shield'] == 8]
 335                     max_speed  shield
 336         sidewinder          7       8
 337 
 338         **Setting values**
 339 
 340         Set value for all items matching the list of labels
 341 
 342         >>> df.loc[['viper', 'sidewinder'], ['shield']] = 50
 343         >>> df
 344                     max_speed  shield
 345         cobra               1       2
 346         viper               4      50
 347         sidewinder          7      50
 348 
 349         Set value for an entire row
 350 
 351         >>> df.loc['cobra'] = 10
 352         >>> df
 353                     max_speed  shield
 354         cobra              10      10
 355         viper               4      50
 356         sidewinder          7      50
 357 
 358         Set value for an entire column
 359 
 360         >>> df.loc[:, 'max_speed'] = 30
 361         >>> df
 362                     max_speed  shield
 363         cobra              30      10
 364         viper              30      50
 365         sidewinder         30      50
 366 
 367         Set value for rows matching callable condition
 368 
 369         >>> df.loc[df['shield'] > 35] = 0
 370         >>> df
 371                     max_speed  shield
 372         cobra              30      10
 373         viper               0       0
 374         sidewinder          0       0
 375 
 376         **Getting values on a DataFrame with an index that has integer labels**
 377 
 378         Another example using integers for the index
 379 
 380         >>> df = pd.DataFrame([[1, 2], [4, 5], [7, 8]],
 381         ...      index=[7, 8, 9], columns=['max_speed', 'shield'])
 382         >>> df
 383            max_speed  shield
 384         7          1       2
 385         8          4       5
 386         9          7       8
 387 
 388         Slice with integer labels for rows. As mentioned above, note that both
 389         the start and stop of the slice are included.
 390 
 391         >>> df.loc[7:9]
 392            max_speed  shield
 393         7          1       2
 394         8          4       5
 395         9          7       8
 396 
 397         **Getting values with a MultiIndex**
 398 
 399         A number of examples using a DataFrame with a MultiIndex
 400 
 401         >>> tuples = [
 402         ...    ('cobra', 'mark i'), ('cobra', 'mark ii'),
 403         ...    ('sidewinder', 'mark i'), ('sidewinder', 'mark ii'),
 404         ...    ('viper', 'mark ii'), ('viper', 'mark iii')
 405         ... ]
 406         >>> index = pd.MultiIndex.from_tuples(tuples)
 407         >>> values = [[12, 2], [0, 4], [10, 20],
 408         ...         [1, 4], [7, 1], [16, 36]]
 409         >>> df = pd.DataFrame(values, columns=['max_speed', 'shield'], index=index)
 410         >>> df
 411                              max_speed  shield
 412         cobra      mark i           12       2
 413                    mark ii           0       4
 414         sidewinder mark i           10      20
 415                    mark ii           1       4
 416         viper      mark ii           7       1
 417                    mark iii         16      36
 418 
 419         Single label. Note this returns a DataFrame with a single index.
 420 
 421         >>> df.loc['cobra']
 422                  max_speed  shield
 423         mark i          12       2
 424         mark ii          0       4
 425 
 426         Single index tuple. Note this returns a Series.
 427 
 428         >>> df.loc[('cobra', 'mark ii')]
 429         max_speed    0
 430         shield       4
 431         Name: (cobra, mark ii), dtype: int64
 432 
 433         Single label for row and column. Similar to passing in a tuple, this
 434         returns a Series.
 435 
 436         >>> df.loc['cobra', 'mark i']
 437         max_speed    12
 438         shield        2
 439         Name: (cobra, mark i), dtype: int64
 440 
 441         Single tuple. Note using ``[[]]`` returns a DataFrame.
 442 
 443         >>> df.loc[[('cobra', 'mark ii')]]
 444                        max_speed  shield
 445         cobra mark ii          0       4
 446 
 447         Single tuple for the index with a single label for the column
 448 
 449         >>> df.loc[('cobra', 'mark i'), 'shield']
 450         2
 451 
 452         Slice from index tuple to single label
 453 
 454         >>> df.loc[('cobra', 'mark i'):'viper']
 455                              max_speed  shield
 456         cobra      mark i           12       2
 457                    mark ii           0       4
 458         sidewinder mark i           10      20
 459                    mark ii           1       4
 460         viper      mark ii           7       1
 461                    mark iii         16      36
 462 
 463         Slice from index tuple to index tuple
 464 
 465         >>> df.loc[('cobra', 'mark i'):('viper', 'mark ii')]
 466                             max_speed  shield
 467         cobra      mark i          12       2
 468                    mark ii          0       4
 469         sidewinder mark i          10      20
 470                    mark ii          1       4
 471         viper      mark ii          7       1
 472         """
 473         return _LocIndexer("loc", self)

 577     def __call__(self, axis=None):
 578         # we need to return a copy of ourselves
 579         new_self = type(self)(self.name, self.obj)
 580 
 581         if axis is not None:
 582             axis = self.obj._get_axis_number(axis)
 583         new_self.axis = axis
 584         return new_self

 586     def _get_setitem_indexer(self, key):
 587         """
 588         Convert a potentially-label-based key into a positional indexer.
 589         """
 590         if self.name == "loc":
 591             self._ensure_listlike_indexer(key)
 592 
 593         if self.axis is not None:
 594             return self._convert_tuple(key, is_setter=True)
 595 
 596         ax = self.obj._get_axis(0)
 597 
 598         if isinstance(ax, ABCMultiIndex) and self.name != "iloc":
 599             try:
 600                 return ax.get_loc(key)
 601             except (TypeError, KeyError, InvalidIndexError):
 602                 # TypeError e.g. passed a bool
 603                 pass
 604 
 605         if isinstance(key, tuple):
 606             try:
 607                 return self._convert_tuple(key, is_setter=True)
 608             except IndexingError:
 609                 pass
 610 
 611         if isinstance(key, range):
 612             return list(key)
 613 
 614         try:
 615             return self._convert_to_indexer(key, axis=0, is_setter=True)
 616         except TypeError as e:
 617 
 618             # invalid indexer type vs 'other' indexing errors
 619             if "cannot do" in str(e):
 620                 raise
 621             elif "unhashable type" in str(e):
 622                 raise
 623             raise IndexingError(key) from e

 625     def _ensure_listlike_indexer(self, key, axis=None):
 626         """
 627         Ensure that a list-like of column labels are all present by adding them if
 628         they do not already exist.
 629 
 630         Parameters
 631         ----------
 632         key : list-like of column labels
 633             Target labels.
 634         axis : key axis if known
 635         """
 636         column_axis = 1
 637 
 638         # column only exists in 2-dimensional DataFrame
 639         if self.ndim != 2:
 640             return
 641 
 642         if isinstance(key, tuple):
 643             # key may be a tuple if we are .loc
 644             # in that case, set key to the column part of key
 645             key = key[column_axis]
 646             axis = column_axis
 647 
 648         if (
 649             axis == column_axis
 650             and not isinstance(self.obj.columns, ABCMultiIndex)
 651             and is_list_like_indexer(key)
 652             and not com.is_bool_indexer(key)
 653             and all(is_hashable(k) for k in key)
 654         ):
 655             for k in key:
 656                 if k not in self.obj:
 657                     self.obj[k] = np.nan
 658 
 659     def __setitem__(self, key, value):
 660         if isinstance(key, tuple):
 661             key = tuple(com.apply_if_callable(x, self.obj) for x in key)
 662         else:
 663             key = com.apply_if_callable(key, self.obj)
 664         indexer = self._get_setitem_indexer(key)
 665         self._has_valid_setitem_indexer(key)
 666 
 667         iloc = self if self.name == "iloc" else self.obj.iloc
 668         iloc._setitem_with_indexer(indexer, value)
