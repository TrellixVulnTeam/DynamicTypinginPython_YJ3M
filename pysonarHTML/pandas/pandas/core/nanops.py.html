  52 class disallow:
  53     def __init__(self, *dtypes):
  54         super().__init__()
  55         self.dtypes = tuple(pandas_dtype(dtype).type for dtype in dtypes)
  56 
  57     def check(self, obj) -> bool:
  58         return hasattr(obj, "dtype") and issubclass(obj.dtype.type, self.dtypes)
  59 
  60     def __call__(self, f: F) -> F:
  61         @functools.wraps(f)
  62         def _f(*args, **kwargs):
  63             obj_iter = itertools.chain(args, kwargs.values())
  64             if any(self.check(obj) for obj in obj_iter):
  65                 f_name = f.__name__.replace("nan", "")
  66                 raise TypeError(
  67                     f"reduction operation '{f_name}' not allowed for this dtype"
  68                 )
  69             try:
  70                 with np.errstate(invalid="ignore"):
  71                     return f(*args, **kwargs)
  72             except ValueError as e:
  73                 # we want to transform an object array
  74                 # ValueError message to the more typical TypeError
  75                 # e.g. this is normally a disallowed function on
  76                 # object arrays that contain strings
  77                 if is_object_dtype(args[0]):
  78                     raise TypeError(e) from e
  79                 raise
  80 
  81         return cast(F, _f)

  84 class bottleneck_switch:
  85     def __init__(self, name=None, **kwargs):
  86         self.name = name
  87         self.kwargs = kwargs
  88 
  89     def __call__(self, alt: F) -> F:
  90         bn_name = self.name or alt.__name__
  91 
  92         try:
  93             bn_func = getattr(bn, bn_name)
  94         except (AttributeError, NameError):  # pragma: no cover
  95             bn_func = None
  96 
  97         @functools.wraps(alt)
  98         def f(
  99             values: np.ndarray, axis: Optional[int] = None, skipna: bool = True, **kwds
 100         ):
 101             if len(self.kwargs) > 0:
 102                 for k, v in self.kwargs.items():
 103                     if k not in kwds:
 104                         kwds[k] = v
 105 
 106             if values.size == 0 and kwds.get("min_count") is None:
 107                 # We are empty, returning NA for our type
 108                 # Only applies for the default `min_count` of None
 109                 # since that affects how empty arrays are handled.
 110                 # TODO(GH-18976) update all the nanops methods to
 111                 # correctly handle empty inputs and remove this check.
 112                 # It *may* just be `var`
 113                 return _na_for_min_count(values, axis)
 114 
 115             if _USE_BOTTLENECK and skipna and _bn_ok_dtype(values.dtype, bn_name):
 116                 if kwds.get("mask", None) is None:
 117                     # `mask` is not recognised by bottleneck, would raise
 118                     #  TypeError if called
 119                     kwds.pop("mask", None)
 120                     result = bn_func(values, axis=axis, **kwds)
 121 
 122                     # prefer to treat inf/-inf as NA, but must compute the func
 123                     # twice :(
 124                     if _has_infs(result):
 125                         result = alt(values, axis=axis, skipna=skipna, **kwds)
 126                 else:
 127                     result = alt(values, axis=axis, skipna=skipna, **kwds)
 128             else:
 129                 result = alt(values, axis=axis, skipna=skipna, **kwds)
 130 
 131             return result
 132 
 133         return cast(F, f)

 136 def _bn_ok_dtype(dtype: DtypeObj, name: str) -> bool:
 137     # Bottleneck chokes on datetime64, PeriodDtype (or and EA)
 138     if not is_object_dtype(dtype) and not needs_i8_conversion(dtype):
 139 
 140         # GH 15507
 141         # bottleneck does not properly upcast during the sum
 142         # so can overflow
 143 
 144         # GH 9422
 145         # further we also want to preserve NaN when all elements
 146         # are NaN, unlike bottleneck/numpy which consider this
 147         # to be 0
 148         if name in ["nansum", "nanprod"]:
 149             return False
 150 
 151         return True
 152     return False

 155 def _has_infs(result) -> bool:
 156     if isinstance(result, np.ndarray):
 157         if result.dtype == "f8":
 158             return lib.has_infs_f8(result.ravel("K"))
 159         elif result.dtype == "f4":
 160             return lib.has_infs_f4(result.ravel("K"))
 161     try:
 162         return np.isinf(result).any()
 163     except (TypeError, NotImplementedError):
 164         # if it doesn't support infs, then it can't have infs
 165         return False

 168 def _get_fill_value(
 169     dtype: DtypeObj, fill_value: Optional[Scalar] = None, fill_value_typ=None
 170 ):
 171     """ return the correct fill value for the dtype of the values """
 172     if fill_value is not None:
 173         return fill_value
 174     if _na_ok_dtype(dtype):
 175         if fill_value_typ is None:
 176             return np.nan
 177         else:
 178             if fill_value_typ == "+inf":
 179                 return np.inf
 180             else:
 181                 return -np.inf
 182     else:
 183         if fill_value_typ is None:
 184             return iNaT
 185         else:
 186             if fill_value_typ == "+inf":
 187                 # need the max int here
 188                 return _int64_max
 189             else:
 190                 return iNaT

 193 def _maybe_get_mask(
 194     values: np.ndarray, skipna: bool, mask: Optional[np.ndarray]
 195 ) -> Optional[np.ndarray]:
 196     """
 197     Compute a mask if and only if necessary.
 198 
 199     This function will compute a mask iff it is necessary. Otherwise,
 200     return the provided mask (potentially None) when a mask does not need to be
 201     computed.
 202 
 203     A mask is never necessary if the values array is of boolean or integer
 204     dtypes, as these are incapable of storing NaNs. If passing a NaN-capable
 205     dtype that is interpretable as either boolean or integer data (eg,
 206     timedelta64), a mask must be provided.
 207 
 208     If the skipna parameter is False, a new mask will not be computed.
 209 
 210     The mask is computed using isna() by default. Setting invert=True selects
 211     notna() as the masking function.
 212 
 213     Parameters
 214     ----------
 215     values : ndarray
 216         input array to potentially compute mask for
 217     skipna : bool
 218         boolean for whether NaNs should be skipped
 219     mask : Optional[ndarray]
 220         nan-mask if known
 221 
 222     Returns
 223     -------
 224     Optional[np.ndarray]
 225     """
 226     if mask is None:
 227         if is_bool_dtype(values.dtype) or is_integer_dtype(values.dtype):
 228             # Boolean data cannot contain nulls, so signal via mask being None
 229             return None
 230 
 231         if skipna:
 232             mask = isna(values)
 233 
 234     return mask

 237 def _get_values(
 238     values: np.ndarray,
 239     skipna: bool,
 240     fill_value: Any = None,
 241     fill_value_typ: Optional[str] = None,
 242     mask: Optional[np.ndarray] = None,
 243 ) -> Tuple[np.ndarray, Optional[np.ndarray], np.dtype, np.dtype, Any]:
 244     """
 245     Utility to get the values view, mask, dtype, dtype_max, and fill_value.
 246 
 247     If both mask and fill_value/fill_value_typ are not None and skipna is True,
 248     the values array will be copied.
 249 
 250     For input arrays of boolean or integer dtypes, copies will only occur if a
 251     precomputed mask, a fill_value/fill_value_typ, and skipna=True are
 252     provided.
 253 
 254     Parameters
 255     ----------
 256     values : ndarray
 257         input array to potentially compute mask for
 258     skipna : bool
 259         boolean for whether NaNs should be skipped
 260     fill_value : Any
 261         value to fill NaNs with
 262     fill_value_typ : str
 263         Set to '+inf' or '-inf' to handle dtype-specific infinities
 264     mask : Optional[np.ndarray]
 265         nan-mask if known
 266 
 267     Returns
 268     -------
 269     values : ndarray
 270         Potential copy of input value array
 271     mask : Optional[ndarray[bool]]
 272         Mask for values, if deemed necessary to compute
 273     dtype : np.dtype
 274         dtype for values
 275     dtype_max : np.dtype
 276         platform independent dtype
 277     fill_value : Any
 278         fill value used
 279     """
 280     # In _get_values is only called from within nanops, and in all cases
 281     #  with scalar fill_value.  This guarantee is important for the
 282     #  maybe_upcast_putmask call below
 283     assert is_scalar(fill_value)
 284     values = extract_array(values, extract_numpy=True)
 285 
 286     mask = _maybe_get_mask(values, skipna, mask)
 287 
 288     dtype = values.dtype
 289 
 290     if needs_i8_conversion(values.dtype):
 291         # changing timedelta64/datetime64 to int64 needs to happen after
 292         #  finding `mask` above
 293         values = np.asarray(values.view("i8"))
 294 
 295     dtype_ok = _na_ok_dtype(dtype)
 296 
 297     # get our fill value (in case we need to provide an alternative
 298     # dtype for it)
 299     fill_value = _get_fill_value(
 300         dtype, fill_value=fill_value, fill_value_typ=fill_value_typ
 301     )
 302 
 303     if skipna and (mask is not None) and (fill_value is not None):
 304         values = values.copy()
 305         if dtype_ok and mask.any():
 306             np.putmask(values, mask, fill_value)
 307 
 308         # promote if needed
 309         else:
 310             values, _ = maybe_upcast_putmask(values, mask, fill_value)
 311 
 312     # return a platform independent precision dtype
 313     dtype_max = dtype
 314     if is_integer_dtype(dtype) or is_bool_dtype(dtype):
 315         dtype_max = np.dtype(np.int64)
 316     elif is_float_dtype(dtype):
 317         dtype_max = np.dtype(np.float64)
 318 
 319     return values, mask, dtype, dtype_max, fill_value
 320 
 321 
 322 def _na_ok_dtype(dtype: DtypeObj) -> bool:
 323     if needs_i8_conversion(dtype):
 324         return False
 325     return not issubclass(dtype.type, np.integer)

 328 def _wrap_results(result, dtype: DtypeObj, fill_value=None):
 329     """ wrap our results if needed """
 330     if is_datetime64_any_dtype(dtype):
 331         if fill_value is None:
 332             # GH#24293
 333             fill_value = iNaT
 334         if not isinstance(result, np.ndarray):
 335             tz = getattr(dtype, "tz", None)
 336             assert not isna(fill_value), "Expected non-null fill_value"
 337             if result == fill_value:
 338                 result = np.nan
 339             result = Timestamp(result, tz=tz)
 340         else:
 341             # If we have float dtype, taking a view will give the wrong result
 342             result = result.astype(dtype)
 343     elif is_timedelta64_dtype(dtype):
 344         if not isinstance(result, np.ndarray):
 345             if result == fill_value:
 346                 result = np.nan
 347 
 348             # raise if we have a timedelta64[ns] which is too large
 349             if np.fabs(result) > _int64_max:
 350                 raise ValueError("overflow in timedelta operation")
 351 
 352             result = Timedelta(result, unit="ns")
 353         else:
 354             result = result.astype("m8[ns]").view(dtype)
 355 
 356     return result

 359 def _na_for_min_count(
 360     values: np.ndarray, axis: Optional[int]
 361 ) -> Union[Scalar, np.ndarray]:
 362     """
 363     Return the missing value for `values`.
 364 
 365     Parameters
 366     ----------
 367     values : ndarray
 368     axis : int or None
 369         axis for the reduction, required if values.ndim > 1.
 370 
 371     Returns
 372     -------
 373     result : scalar or ndarray
 374         For 1-D values, returns a scalar of the correct missing type.
 375         For 2-D values, returns a 1-D array where each element is missing.
 376     """
 377     # we either return np.nan or pd.NaT
 378     if is_numeric_dtype(values):
 379         values = values.astype("float64")
 380     fill_value = na_value_for_dtype(values.dtype)
 381 
 382     if values.ndim == 1:
 383         return fill_value
 384     else:
 385         assert axis is not None  # assertion to make mypy happy
 386         result_shape = values.shape[:axis] + values.shape[axis + 1 :]
 387         # calling np.full with dtype parameter throws an ValueError when called
 388         # with dtype=np.datetime64 and and fill_value=pd.NaT
 389         try:
 390             result = np.full(result_shape, fill_value, dtype=values.dtype)
 391         except ValueError:
 392             result = np.full(result_shape, fill_value)
 393         return result

 515 @disallow(PeriodDtype)
 516 @bottleneck_switch()
 517 def nanmean(
 518     values: np.ndarray,
 519     axis: Optional[int] = None,
 520     skipna: bool = True,
 521     mask: Optional[np.ndarray] = None,
 522 ) -> float:
 523     """
 524     Compute the mean of the element along an axis ignoring NaNs
 525 
 526     Parameters
 527     ----------
 528     values : ndarray
 529     axis: int, optional
 530     skipna : bool, default True
 531     mask : ndarray[bool], optional
 532         nan-mask if known
 533 
 534     Returns
 535     -------
 536     float
 537         Unless input is a float array, in which case use the same
 538         precision as the input array.
 539 
 540     Examples
 541     --------
 542     >>> import pandas.core.nanops as nanops
 543     >>> s = pd.Series([1, 2, np.nan])
 544     >>> nanops.nanmean(s)
 545     1.5
 546     """
 547     values, mask, dtype, dtype_max, _ = _get_values(
 548         values, skipna, fill_value=0, mask=mask
 549     )
 550     dtype_sum = dtype_max
 551     dtype_count = np.float64
 552     # not using needs_i8_conversion because that includes period
 553     if (
 554         is_integer_dtype(dtype)
 555         or is_datetime64_any_dtype(dtype)
 556         or is_timedelta64_dtype(dtype)
 557     ):
 558         dtype_sum = np.float64
 559     elif is_float_dtype(dtype):
 560         dtype_sum = dtype
 561         dtype_count = dtype
 562     count = _get_counts(values.shape, mask, axis, dtype=dtype_count)
 563     the_sum = _ensure_numeric(values.sum(axis, dtype=dtype_sum))
 564 
 565     if axis is not None and getattr(the_sum, "ndim", False):
 566         count = cast(np.ndarray, count)
 567         with np.errstate(all="ignore"):
 568             # suppress division by zero warnings
 569             the_mean = the_sum / count
 570         ct_mask = count == 0
 571         if ct_mask.any():
 572             the_mean[ct_mask] = np.nan
 573     else:
 574         the_mean = the_sum / count if count > 0 else np.nan
 575 
 576     return _wrap_results(the_mean, dtype)

 579 @bottleneck_switch()
 580 def nanmedian(values, axis=None, skipna=True, mask=None):
 581     """
 582     Parameters
 583     ----------
 584     values : ndarray
 585     axis: int, optional
 586     skipna : bool, default True
 587     mask : ndarray[bool], optional
 588         nan-mask if known
 589 
 590     Returns
 591     -------
 592     result : float
 593         Unless input is a float array, in which case use the same
 594         precision as the input array.
 595 
 596     Examples
 597     --------
 598     >>> import pandas.core.nanops as nanops
 599     >>> s = pd.Series([1, np.nan, 2, 2])
 600     >>> nanops.nanmedian(s)
 601     2.0
 602     """
 603 
 604     def get_median(x):
 605         mask = notna(x)
 606         if not skipna and not mask.all():
 607             return np.nan
 608         return np.nanmedian(x[mask])
 609 
 610     values, mask, dtype, _, _ = _get_values(values, skipna, mask=mask)
 611     if not is_float_dtype(values.dtype):
 612         try:
 613             values = values.astype("f8")
 614         except ValueError as err:
 615             # e.g. "could not convert string to float: 'a'"
 616             raise TypeError from err
 617         if mask is not None:
 618             values[mask] = np.nan
 619 
 620     if axis is None:
 621         values = values.ravel("K")
 622 
 623     notempty = values.size
 624 
 625     # an array from a frame
 626     if values.ndim > 1:
 627 
 628         # there's a non-empty array to apply over otherwise numpy raises
 629         if notempty:
 630             if not skipna:
 631                 return _wrap_results(
 632                     np.apply_along_axis(get_median, axis, values), dtype
 633                 )
 634 
 635             # fastpath for the skipna case
 636             return _wrap_results(np.nanmedian(values, axis), dtype)
 637 
 638         # must return the correct shape, but median is not defined for the
 639         # empty set so return nans of shape "everything but the passed axis"
 640         # since "axis" is where the reduction would occur if we had a nonempty
 641         # array
 642         shp = np.array(values.shape)
 643         dims = np.arange(values.ndim)
 644         ret = np.empty(shp[dims != axis])
 645         ret.fill(np.nan)
 646         return _wrap_results(ret, dtype)
 647 
 648     # otherwise return a scalar value
 649     return _wrap_results(get_median(values) if notempty else np.nan, dtype)

 652 def _get_counts_nanvar(
 653     value_counts: Tuple[int],
 654     mask: Optional[np.ndarray],
 655     axis: Optional[int],
 656     ddof: int,
 657     dtype: Dtype = float,
 658 ) -> Tuple[Union[int, np.ndarray], Union[int, np.ndarray]]:
 659     """
 660     Get the count of non-null values along an axis, accounting
 661     for degrees of freedom.
 662 
 663     Parameters
 664     ----------
 665     values_shape : Tuple[int]
 666         shape tuple from values ndarray, used if mask is None
 667     mask : Optional[ndarray[bool]]
 668         locations in values that should be considered missing
 669     axis : Optional[int]
 670         axis to count along
 671     ddof : int
 672         degrees of freedom
 673     dtype : type, optional
 674         type to use for count
 675 
 676     Returns
 677     -------
 678     count : scalar or array
 679     d : scalar or array
 680     """
 681     dtype = _get_dtype(dtype)
 682     count = _get_counts(value_counts, mask, axis, dtype=dtype)
 683     d = count - dtype.type(ddof)
 684 
 685     # always return NaN, never inf
 686     if is_scalar(count):

 760     >>> import pandas.core.nanops as nanops
 761     >>> s = pd.Series([1, np.nan, 2, 3])
 762     >>> nanops.nanvar(s)
 763     1.0
 764     """

 766     dtype = values.dtype
 767     mask = _maybe_get_mask(values, skipna, mask)
 768     if is_any_int_dtype(dtype):
 769         values = values.astype("f8")

 973 @disallow("M8", "m8")
 974 def nanskew(
 975     values: np.ndarray,
 976     axis: Optional[int] = None,
 977     skipna: bool = True,
 978     mask: Optional[np.ndarray] = None,
 979 ) -> float:
 980     """
 981     Compute the sample skewness.
 982 
 983     The statistic computed here is the adjusted Fisher-Pearson standardized
 984     moment coefficient G1. The algorithm computes this coefficient directly
 985     from the second and third central moment.
 986 
 987     Parameters
 988     ----------
 989     values : ndarray
 990     axis: int, optional
 991     skipna : bool, default True
 992     mask : ndarray[bool], optional
 993         nan-mask if known
 994 
 995     Returns
 996     -------
 997     result : float64
 998         Unless input is a float array, in which case use the same
 999         precision as the input array.
1000 
1001     Examples
1002     --------
1003     >>> import pandas.core.nanops as nanops
1004     >>> s = pd.Series([1, np.nan, 1, 2])
1005     >>> nanops.nanskew(s)
1006     1.7320508075688787
1007     """
1008     values = extract_array(values, extract_numpy=True)
1009     mask = _maybe_get_mask(values, skipna, mask)
1010     if not is_float_dtype(values.dtype):
1011         values = values.astype("f8")
1012         count = _get_counts(values.shape, mask, axis)
1013     else:
1014         count = _get_counts(values.shape, mask, axis, dtype=values.dtype)
1015 
1016     if skipna and mask is not None:
1017         values = values.copy()
1018         np.putmask(values, mask, 0)
1019 
1020     mean = values.sum(axis, dtype=np.float64) / count
1021     if axis is not None:
1022         mean = np.expand_dims(mean, axis)
1023 
1024     adjusted = values - mean
1025     if skipna and mask is not None:
1026         np.putmask(adjusted, mask, 0)
1027     adjusted2 = adjusted ** 2
1028     adjusted3 = adjusted2 * adjusted
1029     m2 = adjusted2.sum(axis, dtype=np.float64)
1030     m3 = adjusted3.sum(axis, dtype=np.float64)
1031 
1032     # floating point error
1033     #
1034     # #18044 in _libs/windows.pyx calc_skew follow this behavior
1035     # to fix the fperr to treat m2 <1e-14 as zero
1036     m2 = _zero_out_fperr(m2)
1037     m3 = _zero_out_fperr(m3)
1038 
1039     with np.errstate(invalid="ignore", divide="ignore"):
1040         result = (count * (count - 1) ** 0.5 / (count - 2)) * (m3 / m2 ** 1.5)
1041 
1042     dtype = values.dtype
1043     if is_float_dtype(dtype):
1044         result = result.astype(dtype)
1045 
1046     if isinstance(result, np.ndarray):
1047         result = np.where(m2 == 0, 0, result)
1048         result[count < 3] = np.nan
1049         return result
1050     else:
1051         result = 0 if m2 == 0 else result
1052         if count < 3:
1053             return np.nan
1054         return result

1063 ) -> float:
1064     """
1065     Compute the sample excess kurtosis
1066 
1067     The statistic computed here is the adjusted Fisher-Pearson standardized
1068     moment coefficient G2, computed directly from the second and fourth
1069     central moment.
1070 
1071     Parameters
1072     ----------
1073     values : ndarray
1074     axis: int, optional

1076     mask : ndarray[bool], optional
1077         nan-mask if known
1078 
1079     Returns
1080     -------
1081     result : float64
1082         Unless input is a float array, in which case use the same
1083         precision as the input array.
1084

1086     --------
1087     >>> import pandas.core.nanops as nanops
1088     >>> s = pd.Series([1, np.nan, 1, 3, 2])
1089     >>> nanops.nankurt(s)
1090     -1.2892561983471076
1091     """
1092     values = extract_array(values, extract_numpy=True)
1093     mask = _maybe_get_mask(values, skipna, mask)
1094     if not is_float_dtype(values.dtype):
1095         values = values.astype("f8")
1096         count = _get_counts(values.shape, mask, axis)
1097     else:
1098         count = _get_counts(values.shape, mask, axis, dtype=values.dtype)
1099 
1100     if skipna and mask is not None:
1101         values = values.copy()
1102         np.putmask(values, mask, 0)
1103 
1104     mean = values.sum(axis, dtype=np.float64) / count
1105     if axis is not None:
1106         mean = np.expand_dims(mean, axis)
1107 
1108     adjusted = values - mean
1109     if skipna and mask is not None:
1110         np.putmask(adjusted, mask, 0)
1111     adjusted2 = adjusted ** 2
1112     adjusted4 = adjusted2 ** 2
1113     m2 = adjusted2.sum(axis, dtype=np.float64)
1114     m4 = adjusted4.sum(axis, dtype=np.float64)
1115 
1116     with np.errstate(invalid="ignore", divide="ignore"):
1117         adj = 3 * (count - 1) ** 2 / ((count - 2) * (count - 3))
1118         numer = count * (count + 1) * (count - 1) * m4
1119         denom = (count - 2) * (count - 3) * m2 ** 2
1120 
1121     # floating point error
1122     #
1123     # #18044 in _libs/windows.pyx calc_kurt follow this behavior
1124     # to fix the fperr to treat denom <1e-14 as zero
1125     numer = _zero_out_fperr(numer)
1126     denom = _zero_out_fperr(denom)
1127 
1128     if not isinstance(denom, np.ndarray):
1129         # if ``denom`` is a scalar, check these corner cases first before
1130         # doing division
1131         if count < 4:
1132             return np.nan
1133         if denom == 0:
1134             return 0
1135 
1136     with np.errstate(invalid="ignore", divide="ignore"):
1137         result = numer / denom - adj
1138 
1139     dtype = values.dtype
1140     if is_float_dtype(dtype):
1141         result = result.astype(dtype)
1142 
1143     if isinstance(result, np.ndarray):
1144         result = np.where(denom == 0, 0, result)
1145         result[count < 4] = np.nan
1146 
1147     return result

1357         from scipy.stats import kendalltau
1358

1360             return kendalltau(a, b)[0]
1361

1364         from scipy.stats import spearmanr
1365 
1366         def func(a, b):
1367             return spearmanr(a, b)[0]
1368

1404 def _ensure_numeric(x):
1405     if isinstance(x, np.ndarray):
1406         if is_integer_dtype(x) or is_bool_dtype(x):
1407             x = x.astype(np.float64)
1408         elif is_object_dtype(x):
1409             try:
1410                 x = x.astype(np.complex128)
1411             except (TypeError, ValueError):
1412                 try:
1413                     x = x.astype(np.float64)
1414                 except ValueError as err:
1415                     # GH#29941 we get here with object arrays containing strs
1416                     raise TypeError(f"Could not convert {x} to numeric") from err
1417             else:
1418                 if not np.any(np.imag(x)):
1419                     x = x.real
1420     elif not (is_float(x) or is_integer(x) or is_complex(x)):
1421         try:
1422             x = float(x)
1423         except ValueError:
1424             # e.g. "1+1j" or "foo"
1425             try:
1426                 x = complex(x)
1427             except ValueError as err:
1428                 # e.g. "foo"
1429                 raise TypeError(f"Could not convert {x} to numeric") from err
1430     return x

1498     q,
1499     axis: int,
1500     na_value,
1501     mask: np.ndarray,
1502     ndim: int,
1503     interpolation,
1504 ):
1505     """
1506     Wrapper for np.percentile that skips missing values.
1507 
1508     Parameters
1509     ----------
1510     values : array over which to find quantiles
1511     q : scalar or array of quantile indices to find
1512     axis : {0, 1}
1513     na_value : scalar
1514         value to return for empty or all-null values
1515     mask : ndarray[bool]
1516         locations in values that should be considered missing
1517     ndim : {1, 2}
1518     interpolation : str
1519 
1520     Returns
1521     -------
1522     quantiles : scalar or array
1523     """
1524     if values.dtype.kind in ["m", "M"]:
1525         # need to cast to integer to avoid rounding errors in numpy
1526         result = nanpercentile(
1527             values.view("i8"), q, axis, na_value.view("i8"), mask, ndim, interpolation
1528         )
1529 
1530         # Note: we have to do do `astype` and not view because in general we
1531         #  have float result at this point, not i8
1532         return result.astype(values.dtype)
1533 
1534     if not lib.is_scalar(mask) and mask.any():

1548                 for (val, m) in zip(list(values), list(mask))

1556 def na_accum_func(values: ArrayLike, accum_func, skipna: bool) -> ArrayLike:
1557     """
1558     Cumulative function with skipna support.
1559 
1560     Parameters
1561     ----------
1562     values : np.ndarray or ExtensionArray
1563     accum_func : {np.cumprod, np.maximum.accumulate, np.cumsum, np.minimum.accumulate}
1564     skipna : bool
1565 
1566     Returns
1567     -------
1568     np.ndarray or ExtensionArray
1569     """
1570     mask_a, mask_b = {
1571         np.cumprod: (1.0, np.nan),
1572         np.maximum.accumulate: (-np.inf, np.nan),
1573         np.cumsum: (0.0, np.nan),
1574         np.minimum.accumulate: (np.inf, np.nan),
1575     }[accum_func]
1576 
1577     # We will be applying this function to block values
1578     if values.dtype.kind in ["m", "M"]:
1579         # GH#30460, GH#29058
1580         # numpy 1.18 started sorting NaTs at the end instead of beginning,
1581         #  so we need to work around to maintain backwards-consistency.
1582         orig_dtype = values.dtype
1583 
1584         # We need to define mask before masking NaTs
1585         mask = isna(values)
1586 
1587         if accum_func == np.minimum.accumulate:
1588             # Note: the accum_func comparison fails as an "is" comparison
1589             y = values.view("i8")
1590             y[mask] = np.iinfo(np.int64).max
1591             changed = True
1592         else:
1593             y = values
1594             changed = False
1595 
1596         result = accum_func(y.view("i8"), axis=0)
1597         if skipna:
1598             result[mask] = iNaT
1599         elif accum_func == np.minimum.accumulate:
1600             # Restore NaTs that we masked previously
1601             nz = (~np.asarray(mask)).nonzero()[0]
1602             if len(nz):
1603                 # everything up to the first non-na entry stays NaT
1604                 result[: nz[0]] = iNaT
1605 
1606         if changed:
1607             # restore NaT elements
1608             y[mask] = iNaT  # TODO: could try/finally for this?
1609 
1610         if isinstance(values, np.ndarray):
1611             result = result.view(orig_dtype)
1612         else:
1613             # DatetimeArray
1614             result = type(values)._from_sequence(result, dtype=orig_dtype)
1615 
1616     elif skipna and not issubclass(values.dtype.type, (np.integer, np.bool_)):
1617         vals = values.copy()
1618         mask = isna(vals)
1619         vals[mask] = mask_a
1620         result = accum_func(vals, axis=0)
1621         result[mask] = mask_b
1622     else:
1623         result = accum_func(values, axis=0)
1624 
1625     return result
