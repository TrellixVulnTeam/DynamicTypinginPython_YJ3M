  45 def flatten(l):
  46     """
  47     Flatten an arbitrarily nested sequence.
  48 
  49     Parameters
  50     ----------
  51     l : sequence
  52         The non string sequence to flatten
  53 
  54     Notes
  55     -----
  56     This doesn't consider strings sequences.
  57 
  58     Returns
  59     -------
  60     flattened : generator
  61     """
  62     for el in l:
  63         if _iterable_not_string(el):
  64             for s in flatten(el):
  65                 yield s
  66         else:
  67             yield el

  70 def consensus_name_attr(objs):
  71     name = objs[0].name
  72     for obj in objs[1:]:
  73         try:
  74             if obj.name != name:
  75                 name = None
  76         except ValueError:
  77             name = None
  78     return name

  81 def maybe_box_datetimelike(value, dtype=None):
  82     # turn a datetime like into a Timestamp/timedelta as needed
  83     if dtype == object:
  84         # If we dont have datetime64/timedelta64 dtype, we dont want to
  85         #  box datetimelike scalars
  86         return value
  87 
  88     if isinstance(value, (np.datetime64, datetime)):
  89         value = tslibs.Timestamp(value)
  90     elif isinstance(value, (np.timedelta64, timedelta)):
  91         value = tslibs.Timedelta(value)
  92 
  93     return value

  96 def is_bool_indexer(key: Any) -> bool:
  97     """
  98     Check whether `key` is a valid boolean indexer.
  99 
 100     Parameters
 101     ----------
 102     key : Any
 103         Only list-likes may be considered boolean indexers.
 104         All other types are not considered a boolean indexer.
 105         For array-like input, boolean ndarrays or ExtensionArrays
 106         with ``_is_boolean`` set are considered boolean indexers.
 107 
 108     Returns
 109     -------
 110     bool
 111         Whether `key` is a valid boolean indexer.
 112 
 113     Raises
 114     ------
 115     ValueError
 116         When the array is an object-dtype ndarray or ExtensionArray
 117         and contains missing values.
 118 
 119     See Also
 120     --------
 121     check_array_indexer : Check that `key` is a valid array to index,
 122         and convert to an ndarray.
 123     """
 124     if isinstance(key, (ABCSeries, np.ndarray, ABCIndex)) or (
 125         is_array_like(key) and is_extension_array_dtype(key.dtype)
 126     ):
 127         if key.dtype == np.object_:
 128             key = np.asarray(key)
 129 
 130             if not lib.is_bool_array(key):
 131                 na_msg = "Cannot mask with non-boolean array containing NA / NaN values"
 132                 if isna(key).any():
 133                     raise ValueError(na_msg)
 134                 return False
 135             return True
 136         elif is_bool_dtype(key.dtype):
 137             return True
 138     elif isinstance(key, list):
 139         try:
 140             arr = np.asarray(key)
 141             return arr.dtype == np.bool_ and len(arr) == len(key)
 142         except TypeError:  # pragma: no cover
 143             return False
 144 
 145     return False

 148 def cast_scalar_indexer(val, warn_float=False):
 149     """
 150     To avoid numpy DeprecationWarnings, cast float to integer where valid.
 151 
 152     Parameters
 153     ----------
 154     val : scalar
 155     warn_float : bool, default False
 156         If True, issue deprecation warning for a float indexer.
 157 
 158     Returns
 159     -------
 160     outval : scalar
 161     """
 162     # assumes lib.is_scalar(val)
 163     if lib.is_float(val) and val.is_integer():
 164         if warn_float:
 165             warnings.warn(
 166                 "Indexing with a float is deprecated, and will raise an IndexError "
 167                 "in pandas 2.0. You can manually convert to an integer key instead.",
 168                 FutureWarning,
 169                 stacklevel=3,
 170             )
 171         return int(val)
 172     return val

 217 def asarray_tuplesafe(values, dtype=None):
 218 
 219     if not (isinstance(values, (list, tuple)) or hasattr(values, "__array__")):
 220         values = list(values)
 221     elif isinstance(values, ABCIndexClass):
 222         return values._values
 223 
 224     if isinstance(values, list) and dtype in [np.object_, object]:
 225         return construct_1d_object_array_from_listlike(values)
 226 
 227     result = np.asarray(values, dtype=dtype)
 228 
 229     if issubclass(result.dtype.type, str):
 230         result = np.asarray(values, dtype=object)
 231 
 232     if result.ndim == 2:
 233         # Avoid building an array of arrays:
 234         values = [tuple(x) for x in values]
 235         result = construct_1d_object_array_from_listlike(values)
 236 
 237     return result

 240 def index_labels_to_array(labels, dtype=None):
 241     """
 242     Transform label or iterable of labels to array, for use in Index.
 243 
 244     Parameters
 245     ----------
 246     dtype : dtype
 247         If specified, use as dtype of the resulting array, otherwise infer.
 248 
 249     Returns
 250     -------
 251     array
 252     """
 253     if isinstance(labels, (str, tuple)):
 254         labels = [labels]
 255 
 256     if not isinstance(labels, (list, np.ndarray)):
 257         try:
 258             labels = list(labels)
 259         except TypeError:  # non-iterable
 260             labels = [labels]
 261 
 262     labels = asarray_tuplesafe(labels, dtype=dtype)
 263 
 264     return labels

 273 def maybe_iterable_to_list(obj: Union[Iterable[T], T]) -> Union[Collection[T], T]:

 282 def is_null_slice(obj) -> bool:
 283     """
 284     We have a null slice.
 285     """
 286     return (
 287         isinstance(obj, slice)
 288         and obj.start is None
 289         and obj.stop is None
 290         and obj.step is None

 294 def is_true_slices(l):
 295     """
 296     Find non-trivial slices in "l": return a list of booleans with same length.
 297     """
 298     return [isinstance(k, slice) and not is_null_slice(k) for k in l]

 311 def get_callable_name(obj):
 312     # typical case has name
 313     if hasattr(obj, "__name__"):
 314         return getattr(obj, "__name__")
 315     # some objects don't; could recurse
 316     if isinstance(obj, partial):
 317         return get_callable_name(obj.func)
 318     # fall back to class name
 319     if hasattr(obj, "__call__"):
 320         return type(obj).__name__
 321     # everything failed (probably because the argument
 322     # wasn't actually callable); we return None
 323     # instead of the empty string in this case to allow
 324     # distinguishing between no name and a name of ''
 325     return None

 328 def apply_if_callable(maybe_callable, obj, **kwargs):
 329     """
 330     Evaluate possibly callable input using obj and kwargs if it is callable,

 338     """

 362 def standardize_mapping(into):
 363     """
 364     Helper function to standardize a supplied mapping.
 365 
 366     Parameters
 367     ----------
 368     into : instance or subclass of collections.abc.Mapping
 369         Must be a class, an initialized collections.defaultdict,
 370         or an instance of a collections.abc.Mapping subclass.
 371 
 372     Returns
 373     -------
 374     mapping : a collections.abc.Mapping subclass or other constructor
 375         a callable object that can accept an iterator to create
 376         the desired Mapping.
 377 
 378     See Also
 379     --------
 380     DataFrame.to_dict
 381     Series.to_dict
 382     """
 383     if not inspect.isclass(into):
 384         if isinstance(into, defaultdict):
 385             return partial(defaultdict, into.default_factory)
 386         into = type(into)
 387     if not issubclass(into, abc.Mapping):
 388         raise TypeError(f"unsupported type: {into}")
 389     elif into == defaultdict:
 390         raise TypeError("to_dict() only accepts initialized defaultdicts")
 391     return into

 394 def random_state(state=None):
 395     """
 396     Helper function for processing random_state arguments.
 397 
 398     Parameters
 399     ----------
 400     state : int, array-like, BitGenerator (NumPy>=1.17), np.random.RandomState, None.
 401         If receives an int, array-like, or BitGenerator, passes to
 402         np.random.RandomState() as seed.
 403         If receives an np.random.RandomState object, just returns object.
 404         If receives `None`, returns np.random.
 405         If receives anything else, raises an informative ValueError.
 406 
 407         ..versionchanged:: 1.1.0
 408 
 409             array-like and BitGenerator (for NumPy>=1.18) object now passed to
 410             np.random.RandomState() as seed
 411 
 412         Default None.
 413 
 414     Returns
 415     -------
 416     np.random.RandomState
 417 
 418     """
 419     if (
 420         is_integer(state)
 421         or is_array_like(state)
 422         or (not _np_version_under1p18 and isinstance(state, np.random.BitGenerator))
 423     ):
 424         return np.random.RandomState(state)
 425     elif isinstance(state, np.random.RandomState):
 426         return state
 427     elif state is None:
 428         return np.random
 429     else:
 430         raise ValueError(
 431             (
 432                 "random_state must be an integer, array-like, a BitGenerator, "
 433                 "a numpy RandomState, or None"
 434             )
 435         )

 438 def pipe(obj, func, *args, **kwargs):
 439     """
 440     Apply a function ``func`` to object ``obj`` either by passing obj as the
 441     first argument to the function or, in the case that the func is a tuple,
 442     interpret the first element of the tuple as a function and pass the obj to
 443     that function as a keyword argument whose key is the value of the second
 444     element of the tuple.
 445 
 446     Parameters
 447     ----------
 448     func : callable or tuple of (callable, str)
 449         Function to apply to this object or, alternatively, a
 450         ``(callable, data_keyword)`` tuple where ``data_keyword`` is a
 451         string indicating the keyword of `callable`` that expects the
 452         object.
 453     *args : iterable, optional
 454         Positional arguments passed into ``func``.
 455     **kwargs : dict, optional
 456         A dictionary of keyword arguments passed into ``func``.
 457 
 458     Returns
 459     -------
 460     object : the return type of ``func``.
 461     """
 462     if isinstance(func, tuple):
 463         func, target = func
 464         if target in kwargs:
 465             msg = f"{target} is both the pipe target and a keyword argument"
 466             raise ValueError(msg)
 467         kwargs[target] = obj
 468         return func(*args, **kwargs)
 469     else:
 470         return func(obj, *args, **kwargs)

 473 def get_rename_function(mapper):
 474     """
 475     Returns a function that will map names/labels, dependent if mapper
 476     is a dict, Series or just a function.
 477     """
 478     if isinstance(mapper, (abc.Mapping, ABCSeries)):
 479 
 480         def f(x):
 481             if x in mapper:
 482                 return mapper[x]
 483             else:
 484                 return x
 485 
 486     else:
 487         f = mapper
 488 
 489     return f

 492 def convert_to_list_like(
 493     values: Union[Scalar, Iterable, AnyArrayLike]
 494 ) -> Union[List, AnyArrayLike]:
 495     """
 496     Convert list-like or scalar input to list-like. List, numpy and pandas array-like
 497     inputs are returned unmodified whereas others are converted to list.
 498     """
 499     if isinstance(values, (list, np.ndarray, ABCIndex, ABCSeries, ABCExtensionArray)):
 500         return values
 501     elif isinstance(values, abc.Iterable) and not isinstance(values, str):
 502         return list(values)
 503 
 504     return [values]
