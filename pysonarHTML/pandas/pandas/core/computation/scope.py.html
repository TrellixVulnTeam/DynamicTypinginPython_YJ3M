  33 def _replacer(x) -> str:
  34     """
  35     Replace a number with its hexadecimal representation. Used to tag
  36     temporary variables with their calling scope's id.
  37     """
  38     # get the hex repr of the binary char and remove 0x and pad by pad_size
  39     # zeros
  40     try:
  41         hexin = ord(x)
  42     except TypeError:
  43         # bytes literals masquerade as ints when iterating in py3
  44         hexin = x
  45 
  46     return hex(hexin)

  49 def _raw_hex_id(obj) -> str:
  50     """Return the padded hexadecimal id of ``obj``."""
  51     # interpret as a pointer since that's what really what id returns
  52     packed = struct.pack("@P", id(obj))
  53     return "".join(_replacer(x) for x in packed)

  87 class Scope:
  88     """
  89     Object to hold scope, with a few bells to deal with some custom syntax
  90     and contexts added by pandas.
  91 
  92     Parameters
  93     ----------
  94     level : int
  95     global_dict : dict or None, optional, default None
  96     local_dict : dict or Scope or None, optional, default None
  97     resolvers : list-like or None, optional, default None
  98     target : object
  99 
 100     Attributes
 101     ----------
 102     level : int
 103     scope : DeepChainMap
 104     target : object
 105     temps : dict
 106     """
 107 
 108     __slots__ = ["level", "scope", "target", "resolvers", "temps"]
 109 
 110     def __init__(
 111         self, level, global_dict=None, local_dict=None, resolvers=(), target=None
 112     ):
 113         self.level = level + 1
 114 
 115         # shallow copy because we don't want to keep filling this up with what
 116         # was there before if there are multiple calls to Scope/_ensure_scope
 117         self.scope = DeepChainMap(_DEFAULT_GLOBALS.copy())
 118         self.target = target
 119 
 120         if isinstance(local_dict, Scope):
 121             self.scope.update(local_dict.scope)
 122             if local_dict.target is not None:
 123                 self.target = local_dict.target
 124             self._update(local_dict.level)
 125 
 126         frame = sys._getframe(self.level)
 127 
 128         try:
 129             # shallow copy here because we don't want to replace what's in
 130             # scope when we align terms (alignment accesses the underlying
 131             # numpy array of pandas objects)
 132             self.scope = self.scope.new_child((global_dict or frame.f_globals).copy())
 133             if not isinstance(local_dict, Scope):
 134                 self.scope = self.scope.new_child((local_dict or frame.f_locals).copy())
 135         finally:
 136             del frame
 137 
 138         # assumes that resolvers are going from outermost scope to inner
 139         if isinstance(local_dict, Scope):
 140             resolvers += tuple(local_dict.resolvers.maps)
 141         self.resolvers = DeepChainMap(*resolvers)
 142         self.temps = {}
 143 
 144     def __repr__(self) -> str:
 145         scope_keys = _get_pretty_string(list(self.scope.keys()))
 146         res_keys = _get_pretty_string(list(self.resolvers.keys()))
 147         unicode_str = f"{type(self).__name__}(scope={scope_keys}, resolvers={res_keys})"
 148         return unicode_str

 150     @property
 151     def has_resolvers(self) -> bool:
 152         """
 153         Return whether we have any extra scope.
 154 
 155         For example, DataFrames pass Their columns as resolvers during calls to
 156         ``DataFrame.eval()`` and ``DataFrame.query()``.
 157 
 158         Returns
 159         -------
 160         hr : bool
 161         """
 162         return bool(len(self.resolvers))

 164     def resolve(self, key: str, is_local: bool):
 165         """
 166         Resolve a variable name in a possibly local context.
 167 
 168         Parameters
 169         ----------
 170         key : str
 171             A variable name
 172         is_local : bool
 173             Flag indicating whether the variable is local or not (prefixed with
 174             the '@' symbol)
 175 
 176         Returns
 177         -------
 178         value : object
 179             The value of a particular variable
 180         """
 181         try:
 182             # only look for locals in outer scope
 183             if is_local:
 184                 return self.scope[key]
 185 
 186             # not a local variable so check in resolvers if we have them
 187             if self.has_resolvers:
 188                 return self.resolvers[key]
 189 
 190             # if we're here that means that we have no locals and we also have
 191             # no resolvers
 192             assert not is_local and not self.has_resolvers
 193             return self.scope[key]
 194         except KeyError:
 195             try:
 196                 # last ditch effort we look in temporaries
 197                 # these are created when parsing indexing expressions
 198                 # e.g., df[df > 0]
 199                 return self.temps[key]
 200             except KeyError as err:
 201                 # runtime import because ops imports from scope
 202                 from pandas.core.computation.ops import UndefinedVariableError
 203 
 204                 raise UndefinedVariableError(key, is_local) from err
 205 
 206     def swapkey(self, old_key: str, new_key: str, new_value=None):
 207         """
 208         Replace a variable name, with a potentially new value.
 209 
 210         Parameters
 211         ----------
 212         old_key : str
 213             Current variable name to replace
 214         new_key : str
 215             New variable name to replace `old_key` with
 216         new_value : object
 217             Value to be replaced along with the possible renaming
 218         """
 219         if self.has_resolvers:
 220             maps = self.resolvers.maps + self.scope.maps
 221         else:
 222             maps = self.scope.maps
 223 
 224         maps.append(self.temps)
 225 
 226         for mapping in maps:
 227             if old_key in mapping:
 228                 mapping[new_key] = new_value
 229                 return

 231     def _get_vars(self, stack, scopes: List[str]):
 232         """
 233         Get specifically scoped variables from a list of stack frames.
 234 
 235         Parameters
 236         ----------
 237         stack : list
 238             A list of stack frames as returned by ``inspect.stack()``
 239         scopes : sequence of strings
 240             A sequence containing valid stack frame attribute names that
 241             evaluate to a dictionary. For example, ('locals', 'globals')
 242         """
 243         variables = itertools.product(scopes, stack)
 244         for scope, (frame, _, _, _, _, _) in variables:
 245             try:
 246                 d = getattr(frame, "f_" + scope)
 247                 self.scope = self.scope.new_child(d)
 248             finally:
 249                 # won't remove it, but DECREF it
 250                 # in Py3 this probably isn't necessary since frame won't be
 251                 # scope after the loop
 252                 del frame
 253 
 254     def _update(self, level: int):
 255         """
 256         Update the current scope by going back `level` levels.
 257 
 258         Parameters
 259         ----------
 260         level : int
 261         """
 262         sl = level + 1
 263 
 264         # add sl frames to the scope starting with the
 265         # most distant and overwriting with more current
 266         # makes sure that we can capture variable scope
 267         stack = inspect.stack()
 268 
 269         try:
 270             self._get_vars(stack[:sl], scopes=["locals"])
 271         finally:
 272             del stack[:], stack

 274     def add_tmp(self, value) -> str:
 275         """
 276         Add a temporary variable to the scope.
 277 
 278         Parameters
 279         ----------
 280         value : object
 281             An arbitrary object to be assigned to a temporary variable.
 282 
 283         Returns
 284         -------
 285         str
 286             The name of the temporary variable created.
 287         """
 288         name = f"{type(value).__name__}_{self.ntemps}_{_raw_hex_id(self)}"
 289 
 290         # add to inner most scope
 291         assert name not in self.temps
 292         self.temps[name] = value
 293         assert name in self.temps
 294 
 295         # only increment if the variable gets put in the scope
 296         return name
 297 
 298     @property
 299     def ntemps(self) -> int:
 300         """The number of temporary variables in this scope"""
 301         return len(self.temps)
 302 
 303     @property
 304     def full_scope(self):
 305         """
 306         Return the full scope for use with passing to engines transparently
 307         as a mapping.
 308 
 309         Returns
 310         -------
 311         vars : DeepChainMap
 312             All variables in this scope.
 313         """
 314         maps = [self.temps] + self.resolvers.maps + self.scope.maps
 315         return DeepChainMap(*maps)
