  38 def _rewrite_assign(tok: Tuple[int, str]) -> Tuple[int, str]:
  39     """
  40     Rewrite the assignment operator for PyTables expressions that use ``=``
  41     as a substitute for ``==``.
  42 
  43     Parameters
  44     ----------
  45     tok : tuple of int, str
  46         ints correspond to the all caps constants in the tokenize module
  47 
  48     Returns
  49     -------
  50     tuple of int, str
  51         Either the input or token or the replacement values
  52     """
  53     toknum, tokval = tok
  54     return toknum, "==" if tokval == "=" else tokval

  57 def _replace_booleans(tok: Tuple[int, str]) -> Tuple[int, str]:
  58     """
  59     Replace ``&`` with ``and`` and ``|`` with ``or`` so that bitwise
  60     precedence is changed to boolean precedence.
  61 
  62     Parameters
  63     ----------
  64     tok : tuple of int, str
  65         ints correspond to the all caps constants in the tokenize module
  66 
  67     Returns
  68     -------
  69     tuple of int, str
  70         Either the input or token or the replacement values
  71     """
  72     toknum, tokval = tok
  73     if toknum == tokenize.OP:
  74         if tokval == "&":
  75             return tokenize.NAME, "and"
  76         elif tokval == "|":
  77             return tokenize.NAME, "or"
  78         return toknum, tokval
  79     return toknum, tokval

  82 def _replace_locals(tok: Tuple[int, str]) -> Tuple[int, str]:
  83     """
  84     Replace local variables with a syntactically valid name.
  85 
  86     Parameters
  87     ----------
  88     tok : tuple of int, str
  89         ints correspond to the all caps constants in the tokenize module
  90 
  91     Returns
  92     -------
  93     tuple of int, str
  94         Either the input or token or the replacement values
  95 
  96     Notes
  97     -----
  98     This is somewhat of a hack in that we rewrite a string such as ``'@a'`` as
  99     ``'__pd_eval_local_a'`` by telling the tokenizer that ``__pd_eval_local_``
 100     is a ``tokenize.OP`` and to replace the ``'@'`` symbol with it.
 101     """
 102     toknum, tokval = tok
 103     if toknum == tokenize.OP and tokval == "@":
 104         return tokenize.OP, _LOCAL_TAG
 105     return toknum, tokval

 123 def _preparse(
 124     source: str,
 125     f=_compose(
 126         _replace_locals, _replace_booleans, _rewrite_assign, clean_backtick_quoted_toks
 127     ),
 128 ) -> str:
 129     """
 130     Compose a collection of tokenization functions.
 131 
 132     Parameters
 133     ----------
 134     source : str
 135         A Python source code string
 136     f : callable
 137         This takes a tuple of (toknum, tokval) as its argument and returns a
 138         tuple with the same structure but possibly different elements. Defaults
 139         to the composition of ``_rewrite_assign``, ``_replace_booleans``, and
 140         ``_replace_locals``.
 141 
 142     Returns
 143     -------
 144     str
 145         Valid Python source code
 146 
 147     Notes
 148     -----
 149     The `f` parameter can be any callable that takes *and* returns input of the
 150     form ``(toknum, tokval)``, where ``toknum`` is one of the constants from
 151     the ``tokenize`` module and ``tokval`` is a string.
 152     """
 153     assert callable(f), "f must be callable"
 154     return tokenize.untokenize((f(x) for x in tokenize_string(source)))

 258 def disallow(nodes: Set[str]) -> Callable[[Type[_T]], Type[_T]]:
 259     """
 260     Decorator to disallow certain nodes from parsing. Raises a
 261     NotImplementedError instead.
 262 
 263     Returns
 264     -------
 265     callable
 266     """
 267 
 268     def disallowed(cls: Type[_T]) -> Type[_T]:
 269         cls.unsupported_nodes = ()
 270         for node in nodes:
 271             new_method = _node_not_implemented(node)
 272             name = f"visit_{node}"
 273             cls.unsupported_nodes += (name,)
 274             setattr(cls, name, new_method)
 275         return cls
 276 
 277     return disallowed

 305 def add_ops(op_classes):
 306     """
 307     Decorator to add default implementation of ops.
 308     """
 309 
 310     def f(cls):
 311         for op_attr_name, op_class in op_classes.items():
 312             ops = getattr(cls, f"{op_attr_name}_ops")
 313             ops_map = getattr(cls, f"{op_attr_name}_op_nodes_map")
 314             for op in ops:
 315                 op_node = ops_map[op]
 316                 if op_node is not None:
 317                     made_op = _op_maker(op_class, op)
 318                     setattr(cls, f"visit_{op_node}", made_op)
 319         return cls
 320 
 321     return f

 325 @add_ops(_op_classes)
 326 class BaseExprVisitor(ast.NodeVisitor):
 327     """
 328     Custom ast walker. Parsers of other engines should subclass this class
 329     if necessary.
 330 
 331     Parameters
 332     ----------
 333     env : Scope
 334     engine : str
 335     parser : str
 336     preparser : callable
 337     """
 338 
 339     const_type: Type[Term] = Constant
 340     term_type = Term
 341 
 342     binary_ops = _cmp_ops_syms + _bool_ops_syms + _arith_ops_syms
 343     binary_op_nodes = (
 344         "Gt",
 345         "Lt",
 346         "GtE",
 347         "LtE",
 348         "Eq",
 349         "NotEq",
 350         "In",
 351         "NotIn",
 352         "BitAnd",
 353         "BitOr",
 354         "And",
 355         "Or",
 356         "Add",
 357         "Sub",
 358         "Mult",
 359         None,
 360         "Pow",
 361         "FloorDiv",
 362         "Mod",
 363     )
 364     binary_op_nodes_map = dict(zip(binary_ops, binary_op_nodes))
 365 
 366     unary_ops = _unary_ops_syms
 367     unary_op_nodes = "UAdd", "USub", "Invert", "Not"
 368     unary_op_nodes_map = dict(zip(unary_ops, unary_op_nodes))
 369 
 370     rewrite_map = {
 371         ast.Eq: ast.In,
 372         ast.NotEq: ast.NotIn,
 373         ast.In: ast.In,
 374         ast.NotIn: ast.NotIn,
 375     }
 376 
 377     unsupported_nodes: Tuple[str, ...]
 378 
 379     def __init__(self, env, engine, parser, preparser=_preparse):
 380         self.env = env
 381         self.engine = engine
 382         self.parser = parser
 383         self.preparser = preparser
 384         self.assigner = None
 385 
 386     def visit(self, node, **kwargs):
 387         if isinstance(node, str):
 388             clean = self.preparser(node)
 389             try:
 390                 node = ast.fix_missing_locations(ast.parse(clean))
 391             except SyntaxError as e:
 392                 if any(iskeyword(x) for x in clean.split()):
 393                     e.msg = "Python keyword not valid identifier in numexpr query"
 394                 raise e
 395 
 396         method = "visit_" + type(node).__name__
 397         visitor = getattr(self, method)
 398         return visitor(node, **kwargs)
 399 
 400     def visit_Module(self, node, **kwargs):
 401         if len(node.body) != 1:
 402             raise SyntaxError("only a single expression is allowed")
 403         expr = node.body[0]
 404         return self.visit(expr, **kwargs)
 405 
 406     def visit_Expr(self, node, **kwargs):
 407         return self.visit(node.value, **kwargs)

 409     def _rewrite_membership_op(self, node, left, right):
 410         # the kind of the operator (is actually an instance)
 411         op_instance = node.op
 412         op_type = type(op_instance)
 413 
 414         # must be two terms and the comparison operator must be ==/!=/in/not in
 415         if is_term(left) and is_term(right) and op_type in self.rewrite_map:
 416 
 417             left_list, right_list = map(_is_list, (left, right))
 418             left_str, right_str = map(_is_str, (left, right))
 419 
 420             # if there are any strings or lists in the expression
 421             if left_list or right_list or left_str or right_str:
 422                 op_instance = self.rewrite_map[op_type]()
 423 
 424             # pop the string variable out of locals and replace it with a list
 425             # of one string, kind of a hack
 426             if right_str:
 427                 name = self.env.add_tmp([right.value])
 428                 right = self.term_type(name, self.env)
 429 
 430             if left_str:
 431                 name = self.env.add_tmp([left.value])
 432                 left = self.term_type(name, self.env)
 433 
 434         op = self.visit(op_instance)
 435         return op, op_instance, left, right
 436 
 437     def _maybe_transform_eq_ne(self, node, left=None, right=None):
 438         if left is None:
 439             left = self.visit(node.left, side="left")
 440         if right is None:
 441             right = self.visit(node.right, side="right")
 442         op, op_class, left, right = self._rewrite_membership_op(node, left, right)
 443         return op, op_class, left, right

 445     def _maybe_downcast_constants(self, left, right):
 446         f32 = np.dtype(np.float32)
 447         if (
 448             left.is_scalar
 449             and hasattr(left, "value")
 450             and not right.is_scalar
 451             and right.return_type == f32
 452         ):
 453             # right is a float32 array, left is a scalar
 454             name = self.env.add_tmp(np.float32(left.value))
 455             left = self.term_type(name, self.env)
 456         if (
 457             right.is_scalar
 458             and hasattr(right, "value")
 459             and not left.is_scalar
 460             and left.return_type == f32
 461         ):
 462             # left is a float32 array, right is a scalar
 463             name = self.env.add_tmp(np.float32(right.value))
 464             right = self.term_type(name, self.env)
 465 
 466         return left, right

 468     def _maybe_eval(self, binop, eval_in_python):
 469         # eval `in` and `not in` (for now) in "partial" python space
 470         # things that can be evaluated in "eval" space will be turned into
 471         # temporary variables. for example,
 472         # [1,2] in a + 2 * b
 473         # in that case a + 2 * b will be evaluated using numexpr, and the "in"
 474         # call will be evaluated using isin (in python space)
 475         return binop.evaluate(
 476             self.env, self.engine, self.parser, self.term_type, eval_in_python
 477         )
 478 
 479     def _maybe_evaluate_binop(
 480         self,
 481         op,
 482         op_class,
 483         lhs,
 484         rhs,
 485         eval_in_python=("in", "not in"),
 486         maybe_eval_in_python=("==", "!=", "<", ">", "<=", ">="),
 487     ):
 488         res = op(lhs, rhs)
 489 
 490         if res.has_invalid_return_type:
 491             raise TypeError(
 492                 f"unsupported operand type(s) for {res.op}: "
 493                 f"'{lhs.type}' and '{rhs.type}'"
 494             )
 495 
 496         if self.engine != "pytables":
 497             if (
 498                 res.op in _cmp_ops_syms
 499                 and getattr(lhs, "is_datetime", False)
 500                 or getattr(rhs, "is_datetime", False)
 501             ):
 502                 # all date ops must be done in python bc numexpr doesn't work
 503                 # well with NaT
 504                 return self._maybe_eval(res, self.binary_ops)
 505 
 506         if res.op in eval_in_python:
 507             # "in"/"not in" ops are always evaluated in python
 508             return self._maybe_eval(res, eval_in_python)
 509         elif self.engine != "pytables":
 510             if (
 511                 getattr(lhs, "return_type", None) == object
 512                 or getattr(rhs, "return_type", None) == object
 513             ):
 514                 # evaluate "==" and "!=" in python if either of our operands
 515                 # has an object return type
 516                 return self._maybe_eval(res, eval_in_python + maybe_eval_in_python)
 517         return res
 518 
 519     def visit_BinOp(self, node, **kwargs):
 520         op, op_class, left, right = self._maybe_transform_eq_ne(node)
 521         left, right = self._maybe_downcast_constants(left, right)
 522         return self._maybe_evaluate_binop(op, op_class, left, right)

 548     def visit_List(self, node, **kwargs):
 549         name = self.env.add_tmp([self.visit(e)(self.env) for e in node.elts])
 550         return self.term_type(name, self.env)

 554     def visit_Index(self, node, **kwargs):
 555         """ df.index[4] """

 558     def visit_Subscript(self, node, **kwargs):
 559         import pandas as pd
 560 
 561         value = self.visit(node.value)
 562         slobj = self.visit(node.slice)
 563         result = pd.eval(
 564             slobj, local_dict=self.env, engine=self.engine, parser=self.parser
 565         )
 566         try:
 567             # a Term instance
 568             v = value.value[result]
 569         except AttributeError:
 570             # an Op instance
 571             lhs = pd.eval(
 572                 value, local_dict=self.env, engine=self.engine, parser=self.parser
 573             )
 574             v = lhs[result]
 575         name = self.env.add_tmp(v)
 576         return self.term_type(name, env=self.env)
 577 
 578     def visit_Slice(self, node, **kwargs):
 579         """ df.index[slice(4,6)] """
 580         lower = node.lower
 581         if lower is not None:
 582             lower = self.visit(lower).value
 583         upper = node.upper
 584         if upper is not None:
 585             upper = self.visit(upper).value
 586         step = node.step
 587         if step is not None:
 588             step = self.visit(step).value
 589 
 590         return slice(lower, upper, step)

 592     def visit_Assign(self, node, **kwargs):
 593         """
 594         support a single assignment node, like
 595 
 596         c = a + b
 597 
 598         set the assigner at the top level, must be a Name node which
 599         might or might not exist in the resolvers
 600 
 601         """
 602         if len(node.targets) != 1:
 603             raise SyntaxError("can only assign a single expression")
 604         if not isinstance(node.targets[0], ast.Name):
 605             raise SyntaxError("left hand side of an assignment must be a single name")
 606         if self.env.target is None:
 607             raise ValueError("cannot assign without a target object")
 608 
 609         try:
 610             assigner = self.visit(node.targets[0], **kwargs)
 611         except UndefinedVariableError:
 612             assigner = node.targets[0].id
 613 
 614         self.assigner = getattr(assigner, "name", assigner)
 615         if self.assigner is None:
 616             raise SyntaxError(
 617                 "left hand side of an assignment must be a single resolvable name"
 618             )
 619 
 620         return self.visit(node.value, **kwargs)
 621 
 622     def visit_Attribute(self, node, **kwargs):
 623         attr = node.attr
 624         value = node.value
 625 
 626         ctx = node.ctx
 627         if isinstance(ctx, ast.Load):
 628             # resolve the value
 629             resolved = self.visit(value).value
 630             try:
 631                 v = getattr(resolved, attr)
 632                 name = self.env.add_tmp(v)
 633                 return self.term_type(name, self.env)
 634             except AttributeError:
 635                 # something like datetime.datetime where scope is overridden
 636                 if isinstance(value, ast.Name) and value.id == attr:
 637                     return resolved
 638                 raise
 639 
 640         raise ValueError(f"Invalid Attribute context {type(ctx).__name__}")
 641 
 642     def visit_Call(self, node, side=None, **kwargs):
 643 
 644         if isinstance(node.func, ast.Attribute) and node.func.attr != "__call__":
 645             res = self.visit_Attribute(node.func)
 646         elif not isinstance(node.func, ast.Name):
 647             raise TypeError("Only named functions are supported")
 648         else:
 649             try:
 650                 res = self.visit(node.func)
 651             except UndefinedVariableError:
 652                 # Check if this is a supported function name
 653                 try:
 654                     res = FuncNode(node.func.id)
 655                 except ValueError:
 656                     # Raise original error
 657                     raise
 658 
 659         if res is None:
 660             raise ValueError(f"Invalid function call {node.func.id}")
 661         if hasattr(res, "value"):
 662             res = res.value
 663 
 664         if isinstance(res, FuncNode):
 665 
 666             new_args = [self.visit(arg) for arg in node.args]
 667 
 668             if node.keywords:
 669                 raise TypeError(
 670                     f'Function "{res.name}" does not support keyword arguments'
 671                 )
 672 
 673             return res(*new_args)
 674 
 675         else:
 676 
 677             new_args = [self.visit(arg).value for arg in node.args]
 678 
 679             for key in node.keywords:
 680                 if not isinstance(key, ast.keyword):
 681                     raise ValueError(f"keyword error in function call '{node.func.id}'")
 682 
 683                 if key.arg:
 684                     kwargs[key.arg] = self.visit(key.value).value
 685 
 686             return self.const_type(res(*new_args, **kwargs), self.env)
 687 
 688     def translate_In(self, op):
 689         return op
 690 
 691     def visit_Compare(self, node, **kwargs):
 692         ops = node.ops
 693         comps = node.comparators
 694 
 695         # base case: we have something like a CMP b
 696         if len(comps) == 1:
 697             op = self.translate_In(ops[0])
 698             binop = ast.BinOp(op=op, left=node.left, right=comps[0])
 699             return self.visit(binop)
 700 
 701         # recursive case: we have a chained comparison, a CMP b CMP c, etc.
 702         left = node.left
 703         values = []
 704         for op, comp in zip(ops, comps):
 705             new_node = self.visit(
 706                 ast.Compare(comparators=[comp], left=left, ops=[self.translate_In(op)])
 707             )
 708             left = comp
 709             values.append(new_node)
 710         return self.visit(ast.BoolOp(op=ast.And(), values=values))
 711 
 712     def _try_visit_binop(self, bop):
 713         if isinstance(bop, (Op, Term)):
 714             return bop
 715         return self.visit(bop)
 716 
 717     def visit_BoolOp(self, node, **kwargs):
 718         def visitor(x, y):
 719             lhs = self._try_visit_binop(x)
 720             rhs = self._try_visit_binop(y)
 721 
 722             op, op_class, lhs, rhs = self._maybe_transform_eq_ne(node, lhs, rhs)
 723             return self._maybe_evaluate_binop(op, node.op, lhs, rhs)
 724 
 725         operands = node.values
 726         return reduce(visitor, operands)

 757 class Expr:
 758     """
 759     Object encapsulating an expression.
 760 
 761     Parameters
 762     ----------
 763     expr : str
 764     engine : str, optional, default 'numexpr'
 765     parser : str, optional, default 'pandas'
 766     env : Scope, optional, default None
 767     level : int, optional, default 2
 768     """
 769 
 770     env: Scope
 771     engine: str
 772     parser: str
 773 
 774     def __init__(
 775         self,
 776         expr,
 777         engine: str = "numexpr",
 778         parser: str = "pandas",
 779         env: Optional[Scope] = None,
 780         level: int = 0,
 781     ):
 782         self.expr = expr
 783         self.env = env or Scope(level=level + 1)
 784         self.engine = engine
 785         self.parser = parser
 786         self._visitor = _parsers[parser](self.env, self.engine, self.parser)
 787         self.terms = self.parse()
 788 
 789     @property
 790     def assigner(self):
 791         return getattr(self._visitor, "assigner", None)
 792 
 793     def __call__(self):
 794         return self.terms(self.env)
 795 
 796     def __repr__(self) -> str:
 797         return printing.pprint_thing(self.terms)
 798 
 799     def __len__(self) -> int:
 800         return len(self.expr)
 801 
 802     def parse(self):
 803         """
 804         Parse an expression.
 805         """
 806         return self._visitor.visit(self.expr)
 807 
 808     @property
 809     def names(self):
 810         """
 811         Get the names in an expression.
 812         """
 813         if is_term(self.terms):
 814             return frozenset([self.terms.name])
 815         return frozenset(term.name for term in com.flatten(self.terms))
