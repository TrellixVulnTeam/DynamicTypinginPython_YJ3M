  69 class Term:
  70     def __new__(cls, name, env, side=None, encoding=None):
  71         klass = Constant if not isinstance(name, str) else cls
  72         supr_new = super(Term, klass).__new__
  73         return supr_new(klass)
  74 
  75     is_local: bool
  76 
  77     def __init__(self, name, env, side=None, encoding=None):
  78         # name is a str for Term, but may be something else for subclasses
  79         self._name = name
  80         self.env = env
  81         self.side = side
  82         tname = str(name)
  83         self.is_local = tname.startswith(_LOCAL_TAG) or tname in _DEFAULT_GLOBALS
  84         self._value = self._resolve_name()
  85         self.encoding = encoding

  87     @property
  88     def local_name(self) -> str:
  89         return self.name.replace(_LOCAL_TAG, "")
  90 
  91     def __repr__(self) -> str:
  92         return pprint_thing(self.name)
  93 
  94     def __call__(self, *args, **kwargs):
  95         return self.value
  96 
  97     def evaluate(self, *args, **kwargs):
  98         return self

 100     def _resolve_name(self):
 101         res = self.env.resolve(self.local_name, is_local=self.is_local)
 102         self.update(res)
 103 
 104         if hasattr(res, "ndim") and res.ndim > 2:
 105             raise NotImplementedError(
 106                 "N-dimensional objects, where N > 2, are not supported with eval"
 107             )
 108         return res
 109 
 110     def update(self, value):
 111         """
 112         search order for local (i.e., @variable) variables:
 113 
 114         scope, key_variable
 115         [('locals', 'local_name'),
 116          ('globals', 'local_name'),
 117          ('locals', 'key'),
 118          ('globals', 'key')]
 119         """
 120         key = self.name
 121 
 122         # if it's a variable name (otherwise a constant)
 123         if isinstance(key, str):
 124             self.env.swapkey(self.local_name, key, new_value=value)
 125 
 126         self.value = value
 127 
 128     @property
 129     def is_scalar(self) -> bool:
 130         return is_scalar(self._value)

 133     def type(self):
 134         try:
 135             # potentially very slow for large, mixed dtype frames
 136             return self._value.values.dtype
 137         except AttributeError:
 138             try:
 139                 # ndarray
 140                 return self._value.dtype
 141             except AttributeError:
 142                 # scalar
 143                 return type(self._value)

 197 class Op:
 198     """
 199     Hold an operator of arbitrary arity.
 200     """
 201 
 202     op: str
 203 
 204     def __init__(self, op: str, operands: Iterable[Union[Term, "Op"]], encoding=None):
 205         self.op = _bool_op_map.get(op, op)
 206         self.operands = operands
 207         self.encoding = encoding
 208 
 209     def __iter__(self):
 210         return iter(self.operands)
 211 
 212     def __repr__(self) -> str:
 213         """
 214         Print a generic n-ary operator and its operands using infix notation.
 215         """
 216         # recurse over the operands
 217         parened = (f"({pprint_thing(opr)})" for opr in self.operands)
 218         return pprint_thing(f" {self.op} ".join(parened))
 219 
 220     @property
 221     def return_type(self):
 222         # clobber types to bool if the op is a boolean operator
 223         if self.op in (_cmp_ops_syms + _bool_ops_syms):
 224             return np.bool_
 225         return result_type_many(*(term.type for term in com.flatten(self)))
 226 
 227     @property
 228     def has_invalid_return_type(self) -> bool:
 229         types = self.operand_types
 230         obj_dtype_set = frozenset([np.dtype("object")])
 231         return self.return_type == object and types - obj_dtype_set
 232 
 233     @property
 234     def operand_types(self):
 235         return frozenset(term.type for term in com.flatten(self))
 236 
 237     @property
 238     def is_scalar(self) -> bool:
 239         return all(operand.is_scalar for operand in self.operands)
 240 
 241     @property
 242     def is_datetime(self) -> bool:
 243         try:
 244             t = self.return_type.type
 245         except AttributeError:
 246             t = self.return_type
 247 
 248         return issubclass(t, (datetime, np.datetime64))

 251 def _in(x, y):
 252     """
 253     Compute the vectorized membership of ``x in y`` if possible, otherwise
 254     use Python.
 255     """
 256     try:
 257         return x.isin(y)
 258     except AttributeError:
 259         if is_list_like(x):
 260             try:
 261                 return y.isin(x)
 262             except AttributeError:
 263                 pass
 264         return x in y
 265 
 266 
 267 def _not_in(x, y):
 268     """
 269     Compute the vectorized membership of ``x not in y`` if possible,
 270     otherwise use Python.
 271     """
 272     try:
 273         return ~x.isin(y)
 274     except AttributeError:
 275         if is_list_like(x):
 276             try:
 277                 return ~y.isin(x)
 278             except AttributeError:
 279                 pass
 280         return x not in y

 324 def _cast_inplace(terms, acceptable_dtypes, dtype):
 325     """
 326     Cast an expression inplace.
 327 
 328     Parameters
 329     ----------
 330     terms : Op
 331         The expression that should cast.
 332     acceptable_dtypes : list of acceptable numpy.dtype
 333         Will not cast if term's dtype in this list.
 334     dtype : str or numpy.dtype
 335         The dtype to cast to.
 336     """
 337     dt = np.dtype(dtype)
 338     for term in terms:
 339         if term.type in acceptable_dtypes:
 340             continue
 341 
 342         try:
 343             new_value = term.value.astype(dt)
 344         except AttributeError:
 345             new_value = dt.type(term.value)
 346         term.update(new_value)

 353 class BinOp(Op):
 354     """
 355     Hold a binary operator and its operands.
 356 
 357     Parameters
 358     ----------
 359     op : str
 360     lhs : Term or Op
 361     rhs : Term or Op
 362     """
 363 
 364     def __init__(self, op: str, lhs, rhs):
 365         super().__init__(op, (lhs, rhs))
 366         self.lhs = lhs
 367         self.rhs = rhs
 368 
 369         self._disallow_scalar_only_bool_ops()
 370 
 371         self.convert_values()
 372 
 373         try:
 374             self.func = _binary_ops_dict[op]
 375         except KeyError as err:
 376             # has to be made a list for python3
 377             keys = list(_binary_ops_dict.keys())
 378             raise ValueError(
 379                 f"Invalid binary operator {repr(op)}, valid operators are {keys}"
 380             ) from err
 381 
 382     def __call__(self, env):
 383         """
 384         Recursively evaluate an expression in Python space.
 385 
 386         Parameters
 387         ----------
 388         env : Scope
 389 
 390         Returns
 391         -------
 392         object
 393             The result of an evaluated expression.
 394         """
 395         # recurse over the left/right nodes
 396         left = self.lhs(env)
 397         right = self.rhs(env)
 398 
 399         return self.func(left, right)
 400 
 401     def evaluate(self, env, engine: str, parser, term_type, eval_in_python):
 402         """
 403         Evaluate a binary operation *before* being passed to the engine.
 404 
 405         Parameters
 406         ----------
 407         env : Scope
 408         engine : str
 409         parser : str
 410         term_type : type
 411         eval_in_python : list
 412 
 413         Returns
 414         -------
 415         term_type
 416             The "pre-evaluated" expression as an instance of ``term_type``
 417         """
 418         if engine == "python":
 419             res = self(env)
 420         else:
 421             # recurse over the left/right nodes
 422 
 423             left = self.lhs.evaluate(
 424                 env,
 425                 engine=engine,
 426                 parser=parser,
 427                 term_type=term_type,
 428                 eval_in_python=eval_in_python,
 429             )
 430 
 431             right = self.rhs.evaluate(
 432                 env,
 433                 engine=engine,
 434                 parser=parser,
 435                 term_type=term_type,
 436                 eval_in_python=eval_in_python,
 437             )
 438 
 439             # base cases
 440             if self.op in eval_in_python:
 441                 res = self.func(left.value, right.value)
 442             else:
 443                 from pandas.core.computation.eval import eval
 444 
 445                 res = eval(self, local_dict=env, engine=engine, parser=parser)
 446 
 447         name = env.add_tmp(res)
 448         return term_type(name, env=env)
 449 
 450     def convert_values(self):
 451         """
 452         Convert datetimes to a comparable value in an expression.
 453         """
 454 
 455         def stringify(value):
 456             encoder: Callable
 457             if self.encoding is not None:
 458                 encoder = partial(pprint_thing_encoded, encoding=self.encoding)
 459             else:
 460                 encoder = pprint_thing
 461             return encoder(value)
 462 
 463         lhs, rhs = self.lhs, self.rhs
 464 
 465         if is_term(lhs) and lhs.is_datetime and is_term(rhs) and rhs.is_scalar:
 466             v = rhs.value
 467             if isinstance(v, (int, float)):
 468                 v = stringify(v)
 469             v = Timestamp(_ensure_decoded(v))
 470             if v.tz is not None:
 471                 v = v.tz_convert("UTC")
 472             self.rhs.update(v)
 473 
 474         if is_term(rhs) and rhs.is_datetime and is_term(lhs) and lhs.is_scalar:
 475             v = lhs.value
 476             if isinstance(v, (int, float)):
 477                 v = stringify(v)
 478             v = Timestamp(_ensure_decoded(v))
 479             if v.tz is not None:
 480                 v = v.tz_convert("UTC")
 481             self.lhs.update(v)
 482 
 483     def _disallow_scalar_only_bool_ops(self):
 484         if (
 485             (self.lhs.is_scalar or self.rhs.is_scalar)
 486             and self.op in _bool_ops_dict
 487             and (
 488                 not (
 489                     issubclass(self.rhs.return_type, (bool, np.bool_))
 490                     and issubclass(self.lhs.return_type, (bool, np.bool_))
 491                 )
 492             )
 493         ):
 494             raise NotImplementedError("cannot evaluate scalar only bool ops")

 501 class Div(BinOp):
 502     """
 503     Div operator to special case casting.
 504 
 505     Parameters
 506     ----------
 507     lhs, rhs : Term or Op
 508         The Terms or Ops in the ``/`` expression.
 509     """
 510 
 511     def __init__(self, lhs, rhs):
 512         super().__init__("/", lhs, rhs)
 513 
 514         if not isnumeric(lhs.return_type) or not isnumeric(rhs.return_type):
 515             raise TypeError(
 516                 f"unsupported operand type(s) for {self.op}: "
 517                 f"'{lhs.return_type}' and '{rhs.return_type}'"
 518             )
 519 
 520         # do not upcast float32s to float64 un-necessarily
 521         acceptable_dtypes = [np.float32, np.float_]
 522         _cast_inplace(com.flatten(self), acceptable_dtypes, np.float_)

 530 class UnaryOp(Op):
 531     """
 532     Hold a unary operator and its operands.
 533 
 534     Parameters
 535     ----------
 536     op : str
 537         The token used to represent the operator.
 538     operand : Term or Op
 539         The Term or Op operand to the operator.
 540 
 541     Raises
 542     ------
 543     ValueError
 544         * If no function associated with the passed operator token is found.
 545     """
 546 
 547     def __init__(self, op: str, operand):
 548         super().__init__(op, (operand,))
 549         self.operand = operand
 550 
 551         try:
 552             self.func = _unary_ops_dict[op]
 553         except KeyError as err:
 554             raise ValueError(
 555                 f"Invalid unary operator {repr(op)}, "
 556                 f"valid operators are {_unary_ops_syms}"
 557             ) from err
 558 
 559     def __call__(self, env):
 560         operand = self.operand(env)
 561         return self.func(operand)
 562 
 563     def __repr__(self) -> str:
 564         return pprint_thing(f"{self.op}({self.operand})")
 565 
 566     @property
 567     def return_type(self) -> np.dtype:
 568         operand = self.operand
 569         if operand.return_type == np.dtype("bool"):
 570             return np.dtype("bool")
 571         if isinstance(operand, Op) and (
 572             operand.op in _cmp_ops_dict or operand.op in _bool_ops_dict
 573         ):
 574             return np.dtype("bool")
 575         return np.dtype("int")

 578 class MathCall(Op):
 579     def __init__(self, func, args):
 580         super().__init__(func.name, args)
 581         self.func = func
 582 
 583     def __call__(self, env):
 584         operands = [op(env) for op in self.operands]
 585         with np.errstate(all="ignore"):
 586             return self.func.func(*operands)
 587 
 588     def __repr__(self) -> str:
 589         operands = map(str, self.operands)
 590         return pprint_thing(f"{self.op}({','.join(operands)})")

 593 class FuncNode:
 594     def __init__(self, name: str):
 595         from pandas.core.computation.check import _NUMEXPR_INSTALLED, _NUMEXPR_VERSION
 596 
 597         if name not in _mathops or (
 598             _NUMEXPR_INSTALLED
 599             and _NUMEXPR_VERSION < LooseVersion("2.6.9")
 600             and name in ("floor", "ceil")
 601         ):
 602             raise ValueError(f'"{name}" is not a supported function')
 603 
 604         self.name = name
 605         self.func = getattr(np, name)
 606 
 607     def __call__(self, *args):
 608         return MathCall(self, args)
