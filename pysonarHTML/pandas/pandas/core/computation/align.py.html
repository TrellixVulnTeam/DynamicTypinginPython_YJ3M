  46     """
  47     Check a sequence of terms for instances of PandasObject.
  48     """
  49     return any(isinstance(term.value, PandasObject) for term in terms)

  52 def _filter_special_cases(f):
  53     @wraps(f)
  54     def wrapper(terms):
  55         # single unary operand
  56         if len(terms) == 1:
  57             return _align_core_single_unary_op(terms[0])
  58 
  59         term_values = (term.value for term in terms)
  60 
  61         # we don't have any pandas objects
  62         if not _any_pandas_objects(terms):
  63             return result_type_many(*term_values), None
  64 
  65         return f(terms)
  66 
  67     return wrapper

  70 @_filter_special_cases
  71 def _align_core(terms):
  72     term_index = [i for i, term in enumerate(terms) if hasattr(term.value, "axes")]
  73     term_dims = [terms[i].value.ndim for i in term_index]
  74 
  75     from pandas import Series
  76 
  77     ndims = Series(dict(zip(term_index, term_dims)))
  78 
  79     # initial axes are the axes of the largest-axis'd term
  80     biggest = terms[ndims.idxmax()].value
  81     typ = biggest._constructor
  82     axes = biggest.axes
  83     naxes = len(axes)
  84     gt_than_one_axis = naxes > 1
  85 
  86     for value in (terms[i].value for i in term_index):
  87         is_series = isinstance(value, ABCSeries)
  88         is_series_and_gt_one_axis = is_series and gt_than_one_axis
  89 
  90         for axis, items in enumerate(value.axes):
  91             if is_series_and_gt_one_axis:
  92                 ax, itm = naxes - 1, value.index
  93             else:
  94                 ax, itm = axis, items
  95 
  96             if not axes[ax].is_(itm):
  97                 axes[ax] = axes[ax].join(itm, how="outer")
  98 
  99     for i, ndim in ndims.items():
 100         for axis, items in zip(range(ndim), axes):
 101             ti = terms[i].value
 102 
 103             if hasattr(ti, "reindex"):
 104                 transpose = isinstance(ti, ABCSeries) and naxes > 1
 105                 reindexer = axes[naxes - 1] if transpose else items
 106 
 107                 term_axis_size = len(ti.axes[axis])
 108                 reindexer_size = len(reindexer)
 109 
 110                 ordm = np.log10(max(1, abs(reindexer_size - term_axis_size)))
 111                 if ordm >= 1 and reindexer_size >= 10000:
 112                     w = (
 113                         f"Alignment difference on axis {axis} is larger "
 114                         f"than an order of magnitude on term {repr(terms[i].name)}, "
 115                         f"by more than {ordm:.4g}; performance may suffer"
 116                     )
 117                     warnings.warn(w, category=PerformanceWarning, stacklevel=6)
 118 
 119                 f = partial(ti.reindex, reindexer, axis=axis, copy=False)
 120 
 121                 terms[i].update(f())
 122 
 123         terms[i].update(terms[i].value.values)
 124 
 125     return typ, _zip_axes_from_type(typ, axes)

 128 def align_terms(terms):
 129     """
 130     Align a set of terms.
 131     """
 132     try:
 133         # flatten the parse tree (a nested list, really)
 134         terms = list(com.flatten(terms))
 135     except TypeError:
 136         # can't iterate so it must just be a constant or single variable
 137         if isinstance(terms.value, (ABCSeries, ABCDataFrame)):
 138             typ = type(terms.value)
 139             return typ, _zip_axes_from_type(typ, terms.value.axes)
 140         return np.result_type(terms.type), None
 141 
 142     # if all resolved variables are numeric scalars
 143     if all(term.is_scalar for term in terms):
 144         return result_type_many(*(term.value for term in terms)).type, None
 145 
 146     # perform the main alignment
 147     typ, axes = _align_core(terms)
 148     return typ, axes

 151 def reconstruct_object(typ, obj, axes, dtype):
 152     """
 153     Reconstruct an object given its type, raw value, and possibly empty
 154     (None) axes.
 155 
 156     Parameters
 157     ----------
 158     typ : object
 159         A type
 160     obj : object
 161         The value to use in the type constructor
 162     axes : dict
 163         The axes to use to construct the resulting pandas object
 164 
 165     Returns
 166     -------
 167     ret : typ
 168         An object of type ``typ`` with the value `obj` and possible axes
 169         `axes`.
 170     """
 171     try:
 172         typ = typ.type
 173     except AttributeError:
 174         pass
 175 
 176     res_t = np.result_type(obj.dtype, dtype)
 177 
 178     if not isinstance(typ, partial) and issubclass(typ, PandasObject):
 179         return typ(obj, dtype=res_t, **axes)
 180 
 181     # special case for pathological things like ~True/~False
 182     if hasattr(res_t, "type") and typ == np.bool_ and res_t != np.bool_:
 183         ret_value = res_t.type(obj)
 184     else:
 185         ret_value = typ(obj).astype(res_t)
 186         # The condition is to distinguish 0-dim array (returned in case of
 187         # scalar) and 1 element array
 188         # e.g. np.array(0) and np.array([0])
 189         if len(obj.shape) == 1 and len(obj) == 1:
 190             if not isinstance(ret_value, np.ndarray):
 191                 ret_value = np.array([ret_value]).astype(res_t)
 192 
 193     return ret_value
