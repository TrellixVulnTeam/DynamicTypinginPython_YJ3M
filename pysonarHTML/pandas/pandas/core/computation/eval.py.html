  20 def _check_engine(engine: Optional[str]) -> str:
  21     """
  22     Make sure a valid engine is passed.
  23 
  24     Parameters
  25     ----------
  26     engine : str
  27         String to validate.
  28 
  29     Raises
  30     ------
  31     KeyError
  32       * If an invalid engine is passed.
  33     ImportError
  34       * If numexpr was requested but doesn't exist.
  35 
  36     Returns
  37     -------
  38     str
  39         Engine name.
  40     """
  41     from pandas.core.computation.check import _NUMEXPR_INSTALLED
  42 
  43     if engine is None:
  44         engine = "numexpr" if _NUMEXPR_INSTALLED else "python"
  45 
  46     if engine not in _engines:
  47         valid_engines = list(_engines.keys())
  48         raise KeyError(
  49             f"Invalid engine '{engine}' passed, valid engines are {valid_engines}"
  50         )
  51 
  52     # TODO: validate this in a more general way (thinking of future engines
  53     # that won't necessarily be import-able)
  54     # Could potentially be done on engine instantiation
  55     if engine == "numexpr":
  56         if not _NUMEXPR_INSTALLED:
  57             raise ImportError(
  58                 "'numexpr' is not installed or an unsupported version. Cannot use "
  59                 "engine='numexpr' for query/eval if 'numexpr' is not installed"
  60             )
  61 
  62     return engine

  84 def _check_resolvers(resolvers):
  85     if resolvers is not None:
  86         for resolver in resolvers:
  87             if not hasattr(resolver, "__getitem__"):
  88                 name = type(resolver).__name__
  89                 raise TypeError(
  90                     f"Resolver of type '{name}' does not "
  91                     "implement the __getitem__ method"
  92                 )

 113 def _convert_expression(expr) -> str:
 114     """
 115     Convert an object to an expression.
 116 
 117     This function converts an object to an expression (a unicode string) and
 118     checks to make sure it isn't empty after conversion. This is used to
 119     convert operators to their string representation for recursive calls to
 120     :func:`~pandas.eval`.
 121 
 122     Parameters
 123     ----------
 124     expr : object
 125         The object to be converted to a string.
 126 
 127     Returns
 128     -------
 129     str
 130         The string representation of an object.
 131 
 132     Raises
 133     ------
 134     ValueError
 135       * If the expression is empty.
 136     """
 137     s = pprint_thing(expr)
 138     _check_expression(s)
 139     return s

 142 def _check_for_locals(expr: str, stack_level: int, parser: str):
 143 
 144     at_top_of_stack = stack_level == 0
 145     not_pandas_parser = parser != "pandas"
 146 
 147     if not_pandas_parser:
 148         msg = "The '@' prefix is only supported by the pandas parser"
 149     elif at_top_of_stack:
 150         msg = (
 151             "The '@' prefix is not allowed in top-level eval calls.\n"
 152             "please refer to your variables by name without the '@' prefix."
 153         )
 154 
 155     if at_top_of_stack or not_pandas_parser:
 156         for toknum, tokval in tokenize_string(expr):
 157             if toknum == tokenize.OP and tokval == "@":
 158                 raise SyntaxError(msg)

 161 def eval(
 162     expr,
 163     parser="pandas",
 164     engine: Optional[str] = None,
 165     truediv=no_default,
 166     local_dict=None,
 167     global_dict=None,
 168     resolvers=(),
 169     level=0,
 170     target=None,
 171     inplace=False,
 172 ):
 173     """
 174     Evaluate a Python expression as a string using various backends.
 175 
 176     The following arithmetic operations are supported: ``+``, ``-``, ``*``,
 177     ``/``, ``**``, ``%``, ``//`` (python engine only) along with the following
 178     boolean operations: ``|`` (or), ``&`` (and), and ``~`` (not).
 179     Additionally, the ``'pandas'`` parser allows the use of :keyword:`and`,
 180     :keyword:`or`, and :keyword:`not` with the same semantics as the
 181     corresponding bitwise operators.  :class:`~pandas.Series` and
 182     :class:`~pandas.DataFrame` objects are supported and behave as they would
 183     with plain ol' Python evaluation.
 184 
 185     Parameters
 186     ----------
 187     expr : str
 188         The expression to evaluate. This string cannot contain any Python
 189         `statements
 190         <https://docs.python.org/3/reference/simple_stmts.html#simple-statements>`__,
 191         only Python `expressions
 192         <https://docs.python.org/3/reference/simple_stmts.html#expression-statements>`__.
 193     parser : {'pandas', 'python'}, default 'pandas'
 194         The parser to use to construct the syntax tree from the expression. The
 195         default of ``'pandas'`` parses code slightly different than standard
 196         Python. Alternatively, you can parse an expression using the
 197         ``'python'`` parser to retain strict Python semantics.  See the
 198         :ref:`enhancing performance <enhancingperf.eval>` documentation for
 199         more details.
 200     engine : {'python', 'numexpr'}, default 'numexpr'
 201 
 202         The engine used to evaluate the expression. Supported engines are
 203 
 204         - None         : tries to use ``numexpr``, falls back to ``python``
 205         - ``'numexpr'``: This default engine evaluates pandas objects using
 206                          numexpr for large speed ups in complex expressions
 207                          with large frames.
 208         - ``'python'``: Performs operations as if you had ``eval``'d in top
 209                         level python. This engine is generally not that useful.
 210 
 211         More backends may be available in the future.
 212 
 213     truediv : bool, optional
 214         Whether to use true division, like in Python >= 3.
 215         deprecated:: 1.0.0
 216 
 217     local_dict : dict or None, optional
 218         A dictionary of local variables, taken from locals() by default.
 219     global_dict : dict or None, optional
 220         A dictionary of global variables, taken from globals() by default.
 221     resolvers : list of dict-like or None, optional
 222         A list of objects implementing the ``__getitem__`` special method that
 223         you can use to inject an additional collection of namespaces to use for
 224         variable lookup. For example, this is used in the
 225         :meth:`~DataFrame.query` method to inject the
 226         ``DataFrame.index`` and ``DataFrame.columns``
 227         variables that refer to their respective :class:`~pandas.DataFrame`
 228         instance attributes.
 229     level : int, optional
 230         The number of prior stack frames to traverse and add to the current
 231         scope. Most users will **not** need to change this parameter.
 232     target : object, optional, default None
 233         This is the target object for assignment. It is used when there is
 234         variable assignment in the expression. If so, then `target` must
 235         support item assignment with string keys, and if a copy is being
 236         returned, it must also support `.copy()`.
 237     inplace : bool, default False
 238         If `target` is provided, and the expression mutates `target`, whether
 239         to modify `target` inplace. Otherwise, return a copy of `target` with
 240         the mutation.
 241 
 242     Returns
 243     -------
 244     ndarray, numeric scalar, DataFrame, Series
 245 
 246     Raises
 247     ------
 248     ValueError
 249         There are many instances where such an error can be raised:
 250 
 251         - `target=None`, but the expression is multiline.
 252         - The expression is multiline, but not all them have item assignment.
 253           An example of such an arrangement is this:
 254 
 255           a = b + 1
 256           a + 2
 257 
 258           Here, there are expressions on different lines, making it multiline,
 259           but the last line has no variable assigned to the output of `a + 2`.
 260         - `inplace=True`, but the expression is missing item assignment.
 261         - Item assignment is provided, but the `target` does not support
 262           string item assignment.
 263         - Item assignment is provided and `inplace=False`, but the `target`
 264           does not support the `.copy()` method
 265 
 266     See Also
 267     --------
 268     DataFrame.query : Evaluates a boolean expression to query the columns
 269             of a frame.
 270     DataFrame.eval : Evaluate a string describing operations on
 271             DataFrame columns.
 272 
 273     Notes
 274     -----
 275     The ``dtype`` of any objects involved in an arithmetic ``%`` operation are
 276     recursively cast to ``float64``.
 277 
 278     See the :ref:`enhancing performance <enhancingperf.eval>` documentation for
 279     more details.
 280 
 281     Examples
 282     --------
 283     >>> df = pd.DataFrame({"animal": ["dog", "pig"], "age": [10, 20]})
 284     >>> df
 285       animal  age
 286     0    dog   10
 287     1    pig   20
 288 
 289     We can add a new column using ``pd.eval``:
 290 
 291     >>> pd.eval("double_age = df.age * 2", target=df)
 292       animal  age  double_age
 293     0    dog   10          20
 294     1    pig   20          40
 295     """
 296     inplace = validate_bool_kwarg(inplace, "inplace")
 297 
 298     if truediv is not no_default:
 299         warnings.warn(
 300             (
 301                 "The `truediv` parameter in pd.eval is deprecated and "
 302                 "will be removed in a future version."
 303             ),
 304             FutureWarning,
 305             stacklevel=2,
 306         )
 307 
 308     if isinstance(expr, str):
 309         _check_expression(expr)
 310         exprs = [e.strip() for e in expr.splitlines() if e.strip() != ""]
 311     else:
 312         exprs = [expr]
 313     multi_line = len(exprs) > 1
 314 
 315     if multi_line and target is None:
 316         raise ValueError(
 317             "multi-line expressions are only valid in the "
 318             "context of data, use DataFrame.eval"
 319         )
 320     engine = _check_engine(engine)
 321     _check_parser(parser)
 322     _check_resolvers(resolvers)
 323 
 324     ret = None
 325     first_expr = True
 326     target_modified = False
 327 
 328     for expr in exprs:
 329         expr = _convert_expression(expr)
 330         _check_for_locals(expr, level, parser)
 331 
 332         # get our (possibly passed-in) scope
 333         env = ensure_scope(
 334             level + 1,
 335             global_dict=global_dict,
 336             local_dict=local_dict,
 337             resolvers=resolvers,
 338             target=target,
 339         )
 340 
 341         parsed_expr = Expr(expr, engine=engine, parser=parser, env=env)
 342 
 343         # construct the engine and evaluate the parsed expression
 344         eng = _engines[engine]
 345         eng_inst = eng(parsed_expr)
 346         ret = eng_inst.evaluate()
 347 
 348         if parsed_expr.assigner is None:
 349             if multi_line:
 350                 raise ValueError(
 351                     "Multi-line expressions are only valid "
 352                     "if all expressions contain an assignment"
 353                 )
 354             elif inplace:
 355                 raise ValueError("Cannot operate inplace if there is no assignment")
 356 
 357         # assign if needed
 358         assigner = parsed_expr.assigner
 359         if env.target is not None and assigner is not None:
 360             target_modified = True
 361 
 362             # if returning a copy, copy only on the first assignment
 363             if not inplace and first_expr:
 364                 try:
 365                     target = env.target.copy()
 366                 except AttributeError as err:
 367                     raise ValueError("Cannot return a copy of the target") from err
 368             else:
 369                 target = env.target
 370 
 371             # TypeError is most commonly raised (e.g. int, list), but you
 372             # get IndexError if you try to do this assignment on np.ndarray.
 373             # we will ignore numpy warnings here; e.g. if trying
 374             # to use a non-numeric indexer
 375             try:
 376                 with warnings.catch_warnings(record=True):
 377                     # TODO: Filter the warnings we actually care about here.
 378                     target[assigner] = ret
 379             except (TypeError, IndexError) as err:
 380                 raise ValueError("Cannot assign expression output to target") from err
 381 
 382             if not resolvers:
 383                 resolvers = ({assigner: ret},)
 384             else:
 385                 # existing resolver needs updated to handle
 386                 # case of mutating existing column in copy
 387                 for resolver in resolvers:
 388                     if assigner in resolver:
 389                         resolver[assigner] = ret
 390                         break
 391                 else:
 392                     resolvers += ({assigner: ret},)
 393 
 394             ret = None
 395             first_expr = False
 396 
 397     # We want to exclude `inplace=None` as being False.
 398     if inplace is False:
 399         return target if target_modified else ret
