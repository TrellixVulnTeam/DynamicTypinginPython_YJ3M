  61 def _evaluate_standard(op, op_str, a, b):

  71 def _can_use_numexpr(op, op_str, a, b, dtype_check):
  72     """ return a boolean if we WILL be using numexpr """
  73     if op_str is not None:
  74 
  75         # required min elements (otherwise we are adding overhead)
  76         if np.prod(a.shape) > _MIN_ELEMENTS:
  77             # check for dtype compatibility
  78             dtypes = set()
  79             for o in [a, b]:
  80                 # Series implements dtypes, check for dimension count as well
  81                 if hasattr(o, "dtypes") and o.ndim > 1:
  82                     s = o.dtypes.value_counts()
  83                     if len(s) > 1:
  84                         return False
  85                     dtypes |= set(s.index.astype(str))
  86                 # ndarray and Series Case
  87                 elif hasattr(o, "dtype"):
  88                     dtypes |= {o.dtype.name}
  89 
  90             # allowed are a superset
  91             if not len(dtypes) or _ALLOWED_DTYPES[dtype_check] >= dtypes:
  92                 return True
  93 
  94     return False

  97 def _evaluate_numexpr(op, op_str, a, b):
  98     result = None
  99 
 100     if _can_use_numexpr(op, op_str, a, b, "evaluate"):
 101         is_reversed = op.__name__.strip("_").startswith("r")
 102         if is_reversed:
 103             # we were originally called by a reversed op method
 104             a, b = b, a
 105 
 106         a_value = a
 107         b_value = b
 108 
 109         result = ne.evaluate(
 110             f"a_value {op_str} b_value",
 111             local_dict={"a_value": a_value, "b_value": b_value},
 112             casting="safe",
 113         )
 114 
 115     if _TEST_MODE:
 116         _store_test_result(result is not None)
 117 
 118     if result is None:
 119         result = _evaluate_standard(op, op_str, a, b)
 120 
 121     return result

 156 def _where_standard(cond, a, b):
 157     # Caller is responsible for extracting ndarray if necessary
 158     return np.where(cond, a, b)

 161 def _where_numexpr(cond, a, b):
 162     # Caller is responsible for extracting ndarray if necessary
 163     result = None
 164 
 165     if _can_use_numexpr(None, "where", a, b, "where"):
 166 
 167         result = ne.evaluate(
 168             "where(cond_value, a_value, b_value)",
 169             local_dict={"cond_value": cond, "a_value": a, "b_value": b},
 170             casting="safe",
 171         )
 172 
 173     if result is None:
 174         result = _where_standard(cond, a, b)
 175 
 176     return result

 183 def _has_bool_dtype(x):
 184     if isinstance(x, ABCDataFrame):
 185         return "bool" in x.dtypes
 186     try:
 187         return x.dtype == bool
 188     except AttributeError:
 189         return isinstance(x, (bool, np.bool_))

 192 def _bool_arith_check(
 193     op_str, a, b, not_allowed=frozenset(("/", "//", "**")), unsupported=None
 194 ):
 195     if unsupported is None:
 196         unsupported = {"+": "|", "*": "&", "-": "^"}
 197 
 198     if _has_bool_dtype(a) and _has_bool_dtype(b):
 199         if op_str in unsupported:
 200             warnings.warn(
 201                 f"evaluating in Python space because the {repr(op_str)} "
 202                 "operator is not supported by numexpr for "
 203                 f"the bool dtype, use {repr(unsupported[op_str])} instead"
 204             )
 205             return False
 206 
 207         if op_str in not_allowed:
 208             raise NotImplementedError(
 209                 f"operator {repr(op_str)} not implemented for bool dtypes"
 210             )
 211     return True

 214 def evaluate(op, a, b, use_numexpr: bool = True):
 215     """
 216     Evaluate and return the expression of the op on a and b.
 217 
 218     Parameters
 219     ----------
 220     op : the actual operand
 221     a : left operand
 222     b : right operand
 223     use_numexpr : bool, default True
 224         Whether to try to use numexpr.
 225     """
 226     op_str = _op_str_mapping[op]
 227     if op_str is not None:
 228         use_numexpr = use_numexpr and _bool_arith_check(op_str, a, b)
 229         if use_numexpr:
 230             return _evaluate(op, op_str, a, b)  # type: ignore
 231     return _evaluate_standard(op, op_str, a, b)

 234 def where(cond, a, b, use_numexpr=True):
 235     """
 236     Evaluate the where condition cond on a and b.
 237 
 238     Parameters
 239     ----------
 240     cond : np.ndarray[bool]
 241     a : return if cond is True
 242     b : return if cond is False
 243     use_numexpr : bool, default True
 244         Whether to try to use numexpr.
 245     """
 246     return _where(cond, a, b) if use_numexpr else _where_standard(cond, a, b)

 249 def set_test_mode(v=True):
 250     """
 251     Keeps track of whether numexpr was used.  Stores an additional ``True``
 252     for every successful use of evaluate with numexpr since the last
 253     ``get_test_result``
