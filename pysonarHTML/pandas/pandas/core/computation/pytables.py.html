  80 class BinOp(ops.BinOp):
  81 
  82     _max_selectors = 31
  83 
  84     op: str
  85     queryables: Dict[str, Any]
  86 
  87     def __init__(self, op: str, lhs, rhs, queryables: Dict[str, Any], encoding):
  88         super().__init__(op, lhs, rhs)
  89         self.queryables = queryables
  90         self.encoding = encoding
  91         self.condition = None
  92 
  93     def _disallow_scalar_only_bool_ops(self):
  94         pass

  96     def prune(self, klass):
  97         def pr(left, right):
  98             """ create and return a new specialized BinOp from myself """
  99             if left is None:
 100                 return right
 101             elif right is None:
 102                 return left
 103 
 104             k = klass
 105             if isinstance(left, ConditionBinOp):
 106                 if isinstance(right, ConditionBinOp):
 107                     k = JointConditionBinOp
 108                 elif isinstance(left, k):
 109                     return left
 110                 elif isinstance(right, k):
 111                     return right
 112 
 113             elif isinstance(left, FilterBinOp):
 114                 if isinstance(right, FilterBinOp):
 115                     k = JointFilterBinOp
 116                 elif isinstance(left, k):
 117                     return left
 118                 elif isinstance(right, k):
 119                     return right
 120 
 121             return k(
 122                 self.op, left, right, queryables=self.queryables, encoding=self.encoding
 123             ).evaluate()
 124 
 125         left, right = self.lhs, self.rhs
 126 
 127         if is_term(left) and is_term(right):
 128             res = pr(left.value, right.value)
 129         elif not is_term(left) and is_term(right):
 130             res = pr(left.prune(klass), right.value)
 131         elif is_term(left) and not is_term(right):
 132             res = pr(left.value, right.prune(klass))
 133         elif not (is_term(left) or is_term(right)):
 134             res = pr(left.prune(klass), right.prune(klass))
 135 
 136         return res
 137 
 138     def conform(self, rhs):
 139         """ inplace conform rhs """
 140         if not is_list_like(rhs):
 141             rhs = [rhs]
 142         if isinstance(rhs, np.ndarray):
 143             rhs = rhs.ravel()
 144         return rhs

 174     def generate(self, v) -> str:
 175         """ create and return the op string for this TermValue """
 176         val = v.tostring(self.encoding)
 177         return f"({self.lhs} {self.op} {val})"
 178 
 179     def convert_value(self, v) -> "TermValue":
 180         """
 181         convert the expression that is in the term to something that is
 182         accepted by pytables
 183         """
 184 
 185         def stringify(value):
 186             if self.encoding is not None:
 187                 encoder = partial(pprint_thing_encoded, encoding=self.encoding)
 188             else:
 189                 encoder = pprint_thing
 190             return encoder(value)
 191 
 192         kind = _ensure_decoded(self.kind)
 193         meta = _ensure_decoded(self.meta)
 194         if kind == "datetime64" or kind == "datetime":
 195             if isinstance(v, (int, float)):
 196                 v = stringify(v)
 197             v = _ensure_decoded(v)
 198             v = Timestamp(v)
 199             if v.tz is not None:
 200                 v = v.tz_convert("UTC")
 201             return TermValue(v, v.value, kind)
 202         elif kind == "timedelta64" or kind == "timedelta":
 203             if isinstance(v, str):
 204                 v = Timedelta(v).value
 205             else:
 206                 v = Timedelta(v, unit="s").value
 207             return TermValue(int(v), v, kind)
 208         elif meta == "category":
 209             metadata = extract_array(self.metadata, extract_numpy=True)
 210             result = metadata.searchsorted(v, side="left")
 211 
 212             # result returns 0 if v is first element or if v is not in metadata
 213             # check that metadata contains v
 214             if not result and v not in metadata:
 215                 result = -1
 216             return TermValue(result, result, "integer")
 217         elif kind == "integer":
 218             v = int(float(v))
 219             return TermValue(v, v, kind)
 220         elif kind == "float":
 221             v = float(v)
 222             return TermValue(v, v, kind)
 223         elif kind == "bool":
 224             if isinstance(v, str):
 225                 v = not v.strip().lower() in [
 226                     "false",
 227                     "f",
 228                     "no",
 229                     "n",
 230                     "none",
 231                     "0",
 232                     "[]",
 233                     "{}",
 234                     "",
 235                 ]
 236             else:
 237                 v = bool(v)
 238             return TermValue(v, v, kind)
 239         elif isinstance(v, str):
 240             # string quoting
 241             return TermValue(v, stringify(v), "string")
 242         else:
 243             raise TypeError(f"Cannot compare {v} of type {type(v)} to {kind} column")

 249 class FilterBinOp(BinOp):
 250     filter: Optional[Tuple[Any, Any, pd.Index]] = None
 251 
 252     def __repr__(self) -> str:
 253         if self.filter is None:
 254             return "Filter: Not Initialized"
 255         return pprint_thing(f"[Filter : [{self.filter[0]}] -> [{self.filter[1]}]")
 256 
 257     def invert(self):
 258         """ invert the filter """
 259         if self.filter is not None:
 260             f = list(self.filter)
 261             f[1] = self.generate_filter_op(invert=True)
 262             self.filter = tuple(f)
 263         return self
 264 
 265     def format(self):
 266         """ return the actual filter format """
 267         return [self.filter]
 268 
 269     def evaluate(self):
 270 
 271         if not self.is_valid:
 272             raise ValueError(f"query term is not valid [{self}]")
 273 
 274         rhs = self.conform(self.rhs)
 275         values = list(rhs)
 276 
 277         if self.is_in_table:
 278 
 279             # if too many values to create the expression, use a filter instead
 280             if self.op in ["==", "!="] and len(values) > self._max_selectors:
 281 
 282                 filter_op = self.generate_filter_op()
 283                 self.filter = (self.lhs, filter_op, pd.Index(values))
 284 
 285                 return self
 286             return None
 287 
 288         # equality conditions
 289         if self.op in ["==", "!="]:
 290 
 291             filter_op = self.generate_filter_op()
 292             self.filter = (self.lhs, filter_op, pd.Index(values))
 293 
 294         else:
 295             raise TypeError(
 296                 f"passing a filterable condition to a non-table indexer [{self}]"
 297             )
 298 
 299         return self
 300 
 301     def generate_filter_op(self, invert: bool = False):
 302         if (self.op == "!=" and not invert) or (self.op == "==" and invert):
 303             return lambda axis, vals: ~axis.isin(vals)
 304         else:
 305             return lambda axis, vals: axis.isin(vals)

 316 class ConditionBinOp(BinOp):
 317     def __repr__(self) -> str:
 318         return pprint_thing(f"[Condition : [{self.condition}]]")
 319 
 320     def invert(self):
 321         """ invert the condition """
 322         # if self.condition is not None:
 323         #    self.condition = "~(%s)" % self.condition
 324         # return self
 325         raise NotImplementedError(
 326             "cannot use an invert condition when passing to numexpr"
 327         )
 328 
 329     def format(self):
 330         """ return the actual ne format """
 331         return self.condition
 332 
 333     def evaluate(self):
 334 
 335         if not self.is_valid:
 336             raise ValueError(f"query term is not valid [{self}]")
 337 
 338         # convert values if we are in the table
 339         if not self.is_in_table:
 340             return None
 341 
 342         rhs = self.conform(self.rhs)
 343         values = [self.convert_value(v) for v in rhs]
 344 
 345         # equality conditions
 346         if self.op in ["==", "!="]:
 347 
 348             # too many values to create the expression?
 349             if len(values) <= self._max_selectors:
 350                 vs = [self.generate(v) for v in values]
 351                 self.condition = f"({' | '.join(vs)})"
 352 
 353             # use a filter after reading
 354             else:
 355                 return None
 356         else:
 357             self.condition = self.generate(values[0])
 358 
 359         return self

 368 class UnaryOp(ops.UnaryOp):
 369     def prune(self, klass):
 370 
 371         if self.op != "~":
 372             raise NotImplementedError("UnaryOp only support invert type ops")
 373 
 374         operand = self.operand
 375         operand = operand.prune(klass)
 376 
 377         if operand is not None:
 378             if issubclass(klass, ConditionBinOp):
 379                 if operand.condition is not None:
 380                     return operand.invert()
 381             elif issubclass(klass, FilterBinOp):
 382                 if operand.filter is not None:
 383                     return operand.invert()
 384 
 385         return None

 388 class PyTablesExprVisitor(BaseExprVisitor):
 389     const_type = Constant
 390     term_type = Term
 391 
 392     def __init__(self, env, engine, parser, **kwargs):
 393         super().__init__(env, engine, parser)
 394         for bin_op in self.binary_ops:
 395             bin_node = self.binary_op_nodes_map[bin_op]
 396             setattr(
 397                 self,
 398                 f"visit_{bin_node}",
 399                 lambda node, bin_op=bin_op: partial(BinOp, bin_op, **kwargs),
 400             )
 401 
 402     def visit_UnaryOp(self, node, **kwargs):
 403         if isinstance(node.op, (ast.Not, ast.Invert)):
 404             return UnaryOp("~", self.visit(node.operand))
 405         elif isinstance(node.op, ast.USub):
 406             return self.const_type(-self.visit(node.operand).value, self.env)
 407         elif isinstance(node.op, ast.UAdd):
 408             raise NotImplementedError("Unary addition not supported")
 409 
 410     def visit_Index(self, node, **kwargs):
 411         return self.visit(node.value).value
 412 
 413     def visit_Assign(self, node, **kwargs):
 414         cmpr = ast.Compare(
 415             ops=[ast.Eq()], left=node.targets[0], comparators=[node.value]
 416         )
 417         return self.visit(cmpr)
 418 
 419     def visit_Subscript(self, node, **kwargs):
 420         # only allow simple subscripts
 421 
 422         value = self.visit(node.value)
 423         slobj = self.visit(node.slice)
 424         try:
 425             value = value.value
 426         except AttributeError:
 427             pass
 428 
 429         try:
 430             return self.const_type(value[slobj], self.env)
 431         except TypeError as err:
 432             raise ValueError(
 433                 f"cannot subscript {repr(value)} with {repr(slobj)}"
 434             ) from err
 435 
 436     def visit_Attribute(self, node, **kwargs):
 437         attr = node.attr
 438         value = node.value
 439 
 440         ctx = type(node.ctx)
 441         if ctx == ast.Load:
 442             # resolve the value
 443             resolved = self.visit(value)
 444 
 445             # try to get the value to see if we are another expression
 446             try:
 447                 resolved = resolved.value
 448             except (AttributeError):
 449                 pass
 450 
 451             try:
 452                 return self.term_type(getattr(resolved, attr), self.env)
 453             except AttributeError:
 454 
 455                 # something like datetime.datetime where scope is overridden
 456                 if isinstance(value, ast.Name) and value.id == attr:
 457                     return resolved
 458 
 459         raise ValueError(f"Invalid Attribute context {ctx.__name__}")
 460 
 461     def translate_In(self, op):
 462         return ast.Eq() if isinstance(op, ast.In) else op
 463 
 464     def _rewrite_membership_op(self, node, left, right):
 465         return self.visit(node.op), node.op, left, right

 468 def _validate_where(w):
 469     """
 470     Validate that the where statement is of the right type.
 471 
 472     The type may either be String, Expr, or list-like of Exprs.
 473 
 474     Parameters
 475     ----------
 476     w : String term expression, Expr, or list-like of Exprs.
 477 
 478     Returns
 479     -------
 480     where : The original where clause if the check was successful.
 481 
 482     Raises
 483     ------
 484     TypeError : An invalid data type was passed in for w (e.g. dict).
 485     """
 486     if not (isinstance(w, (PyTablesExpr, str)) or is_list_like(w)):
 487         raise TypeError(
 488             "where must be passed as a string, PyTablesExpr, "
 489             "or list-like of PyTablesExpr"
 490         )
 491 
 492     return w

 495 class PyTablesExpr(expr.Expr):
 496     """
 497     Hold a pytables-like expression, comprised of possibly multiple 'terms'.
 498 
 499     Parameters
 500     ----------
 501     where : string term expression, PyTablesExpr, or list-like of PyTablesExprs
 502     queryables : a "kinds" map (dict of column name -> kind), or None if column
 503         is non-indexable
 504     encoding : an encoding that will encode the query terms
 505 
 506     Returns
 507     -------
 508     a PyTablesExpr object
 509 
 510     Examples
 511     --------
 512     'index>=date'
 513     "columns=['A', 'D']"
 514     'columns=A'
 515     'columns==A'
 516     "~(columns=['A','B'])"
 517     'index>df.index[3] & string="bar"'
 518     '(index>df.index[3] & index<=df.index[6]) | string="bar"'
 519     "ts>=Timestamp('2012-02-01')"
 520     "major_axis>=20130101"
 521     """
 522 
 523     _visitor: Optional[PyTablesExprVisitor]
 524     env: PyTablesScope
 525 
 526     def __init__(
 527         self,
 528         where,
 529         queryables: Optional[Dict[str, Any]] = None,
 530         encoding=None,
 531         scope_level: int = 0,
 532     ):
 533 
 534         where = _validate_where(where)
 535 
 536         self.encoding = encoding
 537         self.condition = None
 538         self.filter = None
 539         self.terms = None
 540         self._visitor = None
 541 
 542         # capture the environment if needed
 543         local_dict: DeepChainMap[Any, Any] = DeepChainMap()
 544 
 545         if isinstance(where, PyTablesExpr):
 546             local_dict = where.env.scope
 547             _where = where.expr
 548 
 549         elif isinstance(where, (list, tuple)):
 550             where = list(where)
 551             for idx, w in enumerate(where):
 552                 if isinstance(w, PyTablesExpr):
 553                     local_dict = w.env.scope
 554                 else:
 555                     w = _validate_where(w)
 556                     where[idx] = w
 557             _where = " & ".join((f"({w})" for w in com.flatten(where)))
 558         else:
 559             _where = where
 560 
 561         self.expr = _where
 562         self.env = PyTablesScope(scope_level + 1, local_dict=local_dict)
 563 
 564         if queryables is not None and isinstance(self.expr, str):
 565             self.env.queryables.update(queryables)
 566             self._visitor = PyTablesExprVisitor(
 567                 self.env,
 568                 queryables=queryables,
 569                 parser="pytables",
 570                 engine="pytables",
 571                 encoding=encoding,
 572             )
 573             self.terms = self.parse()
 574 
 575     def __repr__(self) -> str:
 576         if self.terms is not None:
 577             return pprint_thing(self.terms)
 578         return pprint_thing(self.expr)
 579 
 580     def evaluate(self):
 581         """ create and return the numexpr condition and filter """
 582         try:
 583             self.condition = self.terms.prune(ConditionBinOp)
 584         except AttributeError as err:
 585             raise ValueError(
 586                 f"cannot process expression [{self.expr}], [{self}] "
 587                 "is not a valid condition"
 588             ) from err
 589         try:
 590             self.filter = self.terms.prune(FilterBinOp)
 591         except AttributeError as err:
 592             raise ValueError(
 593                 f"cannot process expression [{self.expr}], [{self}] "
 594                 "is not a valid filter"
 595             ) from err
 596 
 597         return self.condition, self.filter

 600 class TermValue:
 601     """ hold a term value the we use to construct a condition/filter """
 602 
 603     def __init__(self, value, converted, kind: str):
 604         assert isinstance(kind, str), kind
 605         self.value = value
 606         self.converted = converted
 607         self.kind = kind
 608 
 609     def tostring(self, encoding) -> str:
 610         """ quote the string if not encoded else encode and return """
 611         if self.kind == "string":
 612             if encoding is not None:
 613                 return str(self.converted)
 614             return f'"{self.converted}"'
 615         elif self.kind == "float":
 616             # python 2 str(float) is not always
 617             # round-trippable so use repr()
 618             return repr(self.converted)
 619         return str(self.converted)

 622 def maybe_expression(s) -> bool:
 623     """ loose checking if s is a pytables-acceptable expression """
 624     if not isinstance(s, str):
 625         return False
 626     ops = PyTablesExprVisitor.binary_ops + PyTablesExprVisitor.unary_ops + ("=",)
 627 
 628     # make sure we have an op at least
 629     return any(op in s for op in ops)
