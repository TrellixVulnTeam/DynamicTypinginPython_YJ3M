  24 def is_valid_positional_slice(slc: slice) -> bool:
  25     """
  26     Check if a slice object can be interpreted as a positional indexer.
  27 
  28     Parameters
  29     ----------
  30     slc : slice
  31 
  32     Returns
  33     -------
  34     bool
  35 
  36     Notes
  37     -----
  38     A valid positional slice may also be interpreted as a label-based slice
  39     depending on the index being sliced.
  40     """
  41 
  42     def is_int_or_none(val):
  43         return val is None or is_integer(val)
  44 
  45     return (
  46         is_int_or_none(slc.start)
  47         and is_int_or_none(slc.stop)
  48         and is_int_or_none(slc.step)
  49     )

  52 def is_list_like_indexer(key) -> bool:
  53     """
  54     Check if we have a list-like indexer that is *not* a NamedTuple.
  55 
  56     Parameters
  57     ----------
  58     key : object
  59 
  60     Returns
  61     -------
  62     bool
  63     """
  64     # allow a list_like, but exclude NamedTuples which can be indexers
  65     return is_list_like(key) and not (isinstance(key, tuple) and type(key) is not tuple)

  68 def is_scalar_indexer(indexer, ndim: int) -> bool:
  69     """
  70     Return True if we are all scalar indexers.
  71 
  72     Parameters
  73     ----------
  74     indexer : object
  75     ndim : int
  76         Number of dimensions in the object being indexed.
  77 
  78     Returns
  79     -------
  80     bool
  81     """
  82     if isinstance(indexer, tuple):
  83         if len(indexer) == ndim:
  84             return all(
  85                 is_integer(x) or (isinstance(x, np.ndarray) and x.ndim == len(x) == 1)
  86                 for x in indexer
  87             )
  88     return False

  91 def is_empty_indexer(indexer, arr_value: np.ndarray) -> bool:
  92     """
  93     Check if we have an empty indexer.
  94 
  95     Parameters
  96     ----------
  97     indexer : object
  98     arr_value : np.ndarray
  99 
 100     Returns
 101     -------
 102     bool
 103     """
 104     if is_list_like(indexer) and not len(indexer):
 105         return True
 106     if arr_value.ndim == 1:
 107         if not isinstance(indexer, tuple):
 108             indexer = tuple([indexer])
 109         return any(isinstance(idx, np.ndarray) and len(idx) == 0 for idx in indexer)
 110     return False

 117 def check_setitem_lengths(indexer, value, values) -> None:
 118     """
 119     Validate that value and indexer are the same length.
 120 
 121     An special-case is allowed for when the indexer is a boolean array
 122     and the number of true values equals the length of ``value``. In
 123     this case, no exception is raised.
 124 
 125     Parameters
 126     ----------
 127     indexer : sequence
 128         Key for the setitem.
 129     value : array-like
 130         Value for the setitem.
 131     values : array-like
 132         Values being set into.
 133 
 134     Returns
 135     -------
 136     None
 137 
 138     Raises
 139     ------
 140     ValueError
 141         When the indexer is an ndarray or list and the lengths don't match.
 142     """
 143     # boolean with truth values == len of the value is ok too
 144     if isinstance(indexer, (np.ndarray, list)):
 145         if is_list_like(value) and len(indexer) != len(value):
 146             if not (
 147                 isinstance(indexer, np.ndarray)
 148                 and indexer.dtype == np.bool_
 149                 and len(indexer[indexer]) == len(value)
 150             ):
 151                 raise ValueError(
 152                     "cannot set using a list-like indexer "
 153                     "with a different length than the value"
 154                 )
 155 
 156     elif isinstance(indexer, slice):
 157         # slice
 158         if is_list_like(value) and len(values):
 159             if len(value) != length_of_indexer(indexer, values):
 160                 raise ValueError(
 161                     "cannot set using a slice indexer with a "
 162                     "different length than the value"
 163                 )

 166 def validate_indices(indices: np.ndarray, n: int) -> None:
 167     """
 168     Perform bounds-checking for an indexer.
 169 
 170     -1 is allowed for indicating missing values.
 171 
 172     Parameters
 173     ----------
 174     indices : ndarray
 175     n : int
 176         Length of the array being indexed.
 177 
 178     Raises
 179     ------
 180     ValueError
 181 
 182     Examples
 183     --------
 184     >>> validate_indices([1, 2], 3)
 185     # OK
 186     >>> validate_indices([1, -2], 3)
 187     ValueError
 188     >>> validate_indices([1, 2, 3], 3)
 189     IndexError
 190     >>> validate_indices([-1, -1], 0)
 191     # OK
 192     >>> validate_indices([0, 1], 0)
 193     IndexError
 194     """
 195     if len(indices):
 196         min_idx = indices.min()
 197         if min_idx < -1:
 198             msg = f"'indices' contains values less than allowed ({min_idx} < -1)"
 199             raise ValueError(msg)
 200 
 201         max_idx = indices.max()
 202         if max_idx >= n:
 203             raise IndexError("indices are out-of-bounds")

 210 def maybe_convert_indices(indices, n: int):
 211     """
 212     Attempt to convert indices into valid, positive indices.
 213 
 214     If we have negative indices, translate to positive here.
 215     If we have indices that are out-of-bounds, raise an IndexError.
 216 
 217     Parameters
 218     ----------
 219     indices : array-like
 220         Array of indices that we are to convert.
 221     n : int
 222         Number of elements in the array that we are indexing.
 223 
 224     Returns
 225     -------
 226     array-like
 227         An array-like of positive indices that correspond to the ones
 228         that were passed in initially to this function.
 229 
 230     Raises
 231     ------
 232     IndexError
 233         One of the converted indices either exceeded the number of,
 234         elements (specified by `n`), or was still negative.
 235     """
 236     if isinstance(indices, list):
 237         indices = np.array(indices)
 238         if len(indices) == 0:
 239             # If `indices` is empty, np.array will return a float,
 240             # and will cause indexing errors.
 241             return np.empty(0, dtype=np.intp)
 242 
 243     mask = indices < 0
 244     if mask.any():
 245         indices = indices.copy()
 246         indices[mask] += n
 247 
 248     mask = (indices >= n) | (indices < 0)
 249     if mask.any():
 250         raise IndexError("indices are out-of-bounds")
 251     return indices

 258 def length_of_indexer(indexer, target=None) -> int:
 259     """
 260     Return the expected length of target[indexer]
 261 
 262     Returns
 263     -------
 264     int
 265     """
 266     if target is not None and isinstance(indexer, slice):
 267         target_len = len(target)
 268         start = indexer.start
 269         stop = indexer.stop
 270         step = indexer.step
 271         if start is None:
 272             start = 0
 273         elif start < 0:
 274             start += target_len
 275         if stop is None or stop > target_len:
 276             stop = target_len
 277         elif stop < 0:
 278             stop += target_len
 279         if step is None:
 280             step = 1
 281         elif step < 0:
 282             start, stop = stop + 1, start + 1
 283             step = -step
 284         return (stop - start + step - 1) // step
 285     elif isinstance(indexer, (ABCSeries, ABCIndexClass, np.ndarray, list)):
 286         if isinstance(indexer, list):
 287             indexer = np.array(indexer)
 288 
 289         if indexer.dtype == bool:
 290             # GH#25774
 291             return indexer.sum()
 292         return len(indexer)
 293     elif not is_list_like_indexer(indexer):
 294         return 1
 295     raise AssertionError("cannot find the length of the indexer")

 298 def deprecate_ndim_indexing(result):
 299     """
 300     Helper function to raise the deprecation warning for multi-dimensional
 301     indexing on 1D Series/Index.
 302 
 303     GH#27125 indexer like idx[:, None] expands dim, but we cannot do that
 304     and keep an index, so we currently return ndarray, which is deprecated
 305     (Deprecation GH#30588).
 306     """
 307     if np.ndim(result) > 1:
 308         warnings.warn(
 309             "Support for multi-dimensional indexing (e.g. `index[:, None]`) "
 310             "on an Index is deprecated and will be removed in a future "
 311             "version.  Convert to a numpy array before indexing instead.",
 312             DeprecationWarning,
 313             stacklevel=3,
 314         )

 317 def unpack_1tuple(tup):
 318     """
 319     If we have a length-1 tuple/list that contains a slice, unpack to just
 320     the slice.
 321 
 322     Notes
 323     -----
 324     The list case is deprecated.
 325     """
 326     if len(tup) == 1 and isinstance(tup[0], slice):
 327         # if we don't have a MultiIndex, we may still be able to handle
 328         #  a 1-tuple.  see test_1tuple_without_multiindex
 329 
 330         if isinstance(tup, list):
 331             # GH#31299
 332             warnings.warn(
 333                 "Indexing with a single-item list containing a "
 334                 "slice is deprecated and will raise in a future "
 335                 "version.  Pass a tuple instead.",
 336                 FutureWarning,
 337                 stacklevel=3,
 338             )
 339 
 340         return tup[0]
 341     return tup

 348 def check_array_indexer(array: AnyArrayLike, indexer: Any) -> Any:
 349     """
 350     Check if `indexer` is a valid array indexer for `array`.
 351 
 352     For a boolean mask, `array` and `indexer` are checked to have the same
 353     length. The dtype is validated, and if it is an integer or boolean
 354     ExtensionArray, it is checked if there are missing values present, and
 355     it is converted to the appropriate numpy array. Other dtypes will raise
 356     an error.
 357 
 358     Non-array indexers (integer, slice, Ellipsis, tuples, ..) are passed
 359     through as is.
 360 
 361     .. versionadded:: 1.0.0
 362 
 363     Parameters
 364     ----------
 365     array : array-like
 366         The array that is being indexed (only used for the length).
 367     indexer : array-like or list-like
 368         The array-like that's used to index. List-like input that is not yet
 369         a numpy array or an ExtensionArray is converted to one. Other input
 370         types are passed through as is.
 371 
 372     Returns
 373     -------
 374     numpy.ndarray
 375         The validated indexer as a numpy array that can be used to index.
 376 
 377     Raises
 378     ------
 379     IndexError
 380         When the lengths don't match.
 381     ValueError
 382         When `indexer` cannot be converted to a numpy ndarray to index
 383         (e.g. presence of missing values).
 384 
 385     See Also
 386     --------
 387     api.types.is_bool_dtype : Check if `key` is of boolean dtype.
 388 
 389     Examples
 390     --------
 391     When checking a boolean mask, a boolean ndarray is returned when the
 392     arguments are all valid.
 393 
 394     >>> mask = pd.array([True, False])
 395     >>> arr = pd.array([1, 2])
 396     >>> pd.api.indexers.check_array_indexer(arr, mask)
 397     array([ True, False])
 398 
 399     An IndexError is raised when the lengths don't match.
 400 
 401     >>> mask = pd.array([True, False, True])
 402     >>> pd.api.indexers.check_array_indexer(arr, mask)
 403     Traceback (most recent call last):
 404     ...
 405     IndexError: Boolean index has wrong length: 3 instead of 2.
 406 
 407     NA values in a boolean array are treated as False.
 408 
 409     >>> mask = pd.array([True, pd.NA])
 410     >>> pd.api.indexers.check_array_indexer(arr, mask)
 411     array([ True, False])
 412 
 413     A numpy boolean mask will get passed through (if the length is correct):
 414 
 415     >>> mask = np.array([True, False])
 416     >>> pd.api.indexers.check_array_indexer(arr, mask)
 417     array([ True, False])
 418 
 419     Similarly for integer indexers, an integer ndarray is returned when it is
 420     a valid indexer, otherwise an error is  (for integer indexers, a matching
 421     length is not required):
 422 
 423     >>> indexer = pd.array([0, 2], dtype="Int64")
 424     >>> arr = pd.array([1, 2, 3])
 425     >>> pd.api.indexers.check_array_indexer(arr, indexer)
 426     array([0, 2])
 427 
 428     >>> indexer = pd.array([0, pd.NA], dtype="Int64")
 429     >>> pd.api.indexers.check_array_indexer(arr, indexer)
 430     Traceback (most recent call last):
 431     ...
 432     ValueError: Cannot index with an integer indexer containing NA values
 433 
 434     For non-integer/boolean dtypes, an appropriate error is raised:
 435 
 436     >>> indexer = np.array([0., 2.], dtype="float64")
 437     >>> pd.api.indexers.check_array_indexer(arr, indexer)
 438     Traceback (most recent call last):
 439     ...
 440     IndexError: arrays used as indices must be of integer or boolean type
 441     """
 442     from pandas.core.construction import array as pd_array
 443 
 444     # whatever is not an array-like is returned as-is (possible valid array
 445     # indexers that are not array-like: integer, slice, Ellipsis, None)
 446     # In this context, tuples are not considered as array-like, as they have
 447     # a specific meaning in indexing (multi-dimensional indexing)
 448     if is_list_like(indexer):
 449         if isinstance(indexer, tuple):
 450             return indexer
 451     else:
 452         return indexer
 453 
 454     # convert list-likes to array
 455     if not is_array_like(indexer):
 456         indexer = pd_array(indexer)
 457         if len(indexer) == 0:
 458             # empty list is converted to float array by pd.array
 459             indexer = np.array([], dtype=np.intp)
 460 
 461     dtype = indexer.dtype
 462     if is_bool_dtype(dtype):
 463         if is_extension_array_dtype(dtype):
 464             indexer = indexer.to_numpy(dtype=bool, na_value=False)
 465         else:
 466             indexer = np.asarray(indexer, dtype=bool)
 467 
 468         # GH26658
 469         if len(indexer) != len(array):
 470             raise IndexError(
 471                 f"Boolean index has wrong length: "
 472                 f"{len(indexer)} instead of {len(array)}"
 473             )
 474     elif is_integer_dtype(dtype):
 475         try:
 476             indexer = np.asarray(indexer, dtype=np.intp)
 477         except ValueError as err:
 478             raise ValueError(
 479                 "Cannot index with an integer indexer containing NA values"
 480             ) from err
 481     else:
 482         raise IndexError("arrays used as indices must be of integer or boolean type")
 483 
 484     return indexer
