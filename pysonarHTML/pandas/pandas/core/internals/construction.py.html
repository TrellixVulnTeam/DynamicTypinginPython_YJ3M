  60 def arrays_to_mgr(
  61     arrays,
  62     arr_names,
  63     index,
  64     columns,
  65     dtype: Optional[DtypeObj] = None,
  66     verify_integrity: bool = True,
  67 ):
  68     """
  69     Segregate Series based on type and coerce into matrices.
  70 
  71     Needs to handle a lot of exceptional cases.
  72     """
  73     arr_names = ensure_index(arr_names)
  74 
  75     if verify_integrity:
  76         # figure out the index, if necessary
  77         if index is None:
  78             index = extract_index(arrays)
  79         else:
  80             index = ensure_index(index)
  81 
  82         # don't force copy because getting jammed in an ndarray anyway
  83         arrays = _homogenize(arrays, index, dtype)
  84 
  85         columns = ensure_index(columns)
  86     else:
  87         columns = ensure_index(columns)
  88         index = ensure_index(index)
  89 
  90     # from BlockManager perspective
  91     axes = [columns, index]
  92 
  93     return create_block_manager_from_arrays(arrays, arr_names, axes)

  96 def masked_rec_array_to_mgr(
  97     data, index, columns, dtype: Optional[DtypeObj], copy: bool
  98 ):
  99     """
 100     Extract from a masked rec array and create the manager.
 101     """
 102     # essentially process a record array then fill it
 103     fill_value = data.fill_value
 104     fdata = ma.getdata(data)
 105     if index is None:
 106         index = get_names_from_index(fdata)
 107         if index is None:
 108             index = ibase.default_index(len(data))
 109     index = ensure_index(index)
 110 
 111     if columns is not None:
 112         columns = ensure_index(columns)
 113     arrays, arr_columns = to_arrays(fdata, columns)
 114 
 115     # fill if needed
 116     new_arrays = []
 117     for fv, arr, col in zip(fill_value, arrays, arr_columns):
 118         # TODO: numpy docs suggest fv must be scalar, but could it be
 119         #  non-scalar for object dtype?
 120         assert lib.is_scalar(fv), fv
 121         mask = ma.getmaskarray(data[col])
 122         if mask.any():
 123             arr, fv = maybe_upcast(arr, fill_value=fv, copy=True)
 124             arr[mask] = fv
 125         new_arrays.append(arr)
 126 
 127     # create the manager
 128     arrays, arr_columns = reorder_arrays(new_arrays, arr_columns, columns)
 129     if columns is None:
 130         columns = arr_columns
 131 
 132     mgr = arrays_to_mgr(arrays, arr_columns, index, columns, dtype)
 133 
 134     if copy:
 135         mgr = mgr.copy()
 136     return mgr

 143 def init_ndarray(values, index, columns, dtype: Optional[DtypeObj], copy: bool):
 144     # input must be a ndarray, list, Series, index
 145 
 146     if isinstance(values, ABCSeries):
 147         if columns is None:
 148             if values.name is not None:
 149                 columns = [values.name]
 150         if index is None:
 151             index = values.index
 152         else:
 153             values = values.reindex(index)
 154 
 155         # zero len case (GH #2234)
 156         if not len(values) and columns is not None and len(columns):
 157             values = np.empty((0, 1), dtype=object)
 158 
 159     # we could have a categorical type passed or coerced to 'category'
 160     # recast this to an arrays_to_mgr
 161     if is_categorical_dtype(getattr(values, "dtype", None)) or is_categorical_dtype(
 162         dtype
 163     ):
 164 
 165         if not hasattr(values, "dtype"):
 166             values = _prep_ndarray(values, copy=copy)
 167             values = values.ravel()
 168         elif copy:
 169             values = values.copy()
 170 
 171         index, columns = _get_axes(len(values), 1, index, columns)
 172         return arrays_to_mgr([values], columns, index, columns, dtype=dtype)
 173     elif is_extension_array_dtype(values) or is_extension_array_dtype(dtype):
 174         # GH#19157
 175 
 176         if isinstance(values, np.ndarray) and values.ndim > 1:
 177             # GH#12513 a EA dtype passed with a 2D array, split into
 178             #  multiple EAs that view the values
 179             values = [values[:, n] for n in range(values.shape[1])]
 180         else:
 181             values = [values]
 182 
 183         if columns is None:
 184             columns = Index(range(len(values)))
 185 
 186         return arrays_to_mgr(values, columns, index, columns, dtype=dtype)
 187 
 188     # by definition an array here
 189     # the dtypes will be coerced to a single dtype
 190     values = _prep_ndarray(values, copy=copy)
 191 
 192     if dtype is not None:
 193         if not is_dtype_equal(values.dtype, dtype):
 194             try:
 195                 values = values.astype(dtype)
 196             except Exception as orig:
 197                 # e.g. ValueError when trying to cast object dtype to float64
 198                 raise ValueError(
 199                     f"failed to cast to '{dtype}' (Exception was: {orig})"
 200                 ) from orig
 201 
 202     # _prep_ndarray ensures that values.ndim == 2 at this point
 203     index, columns = _get_axes(
 204         values.shape[0], values.shape[1], index=index, columns=columns
 205     )
 206     values = values.T
 207 
 208     # if we don't have a dtype specified, then try to convert objects
 209     # on the entire block; this is to convert if we have datetimelike's
 210     # embedded in an object type
 211     if dtype is None and is_object_dtype(values):
 212 
 213         if values.ndim == 2 and values.shape[0] != 1:
 214             # transpose and separate blocks
 215 
 216             dvals_list = [maybe_infer_to_datetimelike(row) for row in values]
 217             for n in range(len(dvals_list)):
 218                 if isinstance(dvals_list[n], np.ndarray):
 219                     dvals_list[n] = dvals_list[n].reshape(1, -1)
 220 
 221             from pandas.core.internals.blocks import make_block
 222 
 223             # TODO: What about re-joining object columns?
 224             block_values = [
 225                 make_block(dvals_list[n], placement=[n]) for n in range(len(dvals_list))
 226             ]
 227 
 228         else:
 229             datelike_vals = maybe_infer_to_datetimelike(values)
 230             block_values = [datelike_vals]
 231     else:
 232         block_values = [values]
 233 
 234     return create_block_manager_from_blocks(block_values, [columns, index])

 237 def init_dict(data: Dict, index, columns, dtype: Optional[DtypeObj] = None):
 238     """
 239     Segregate Series based on type and coerce into matrices.
 240     Needs to handle a lot of exceptional cases.
 241     """
 242     arrays: Union[Sequence[Any], "Series"]
 243 
 244     if columns is not None:
 245         from pandas.core.series import Series  # noqa:F811
 246 
 247         arrays = Series(data, index=columns, dtype=object)
 248         data_names = arrays.index
 249 
 250         missing = arrays.isna()
 251         if index is None:
 252             # GH10856
 253             # raise ValueError if only scalars in dict
 254             index = extract_index(arrays[~missing])
 255         else:
 256             index = ensure_index(index)
 257 
 258         # no obvious "empty" int column
 259         if missing.any() and not is_integer_dtype(dtype):
 260             if dtype is None or (
 261                 not is_extension_array_dtype(dtype)
 262                 and np.issubdtype(dtype, np.flexible)
 263             ):
 264                 # GH#1783
 265                 nan_dtype = np.dtype(object)
 266             else:
 267                 nan_dtype = dtype
 268             val = construct_1d_arraylike_from_scalar(np.nan, len(index), nan_dtype)
 269             arrays.loc[missing] = [val] * missing.sum()
 270 
 271     else:
 272         keys = list(data.keys())
 273         columns = data_names = Index(keys)
 274         arrays = [com.maybe_iterable_to_list(data[k]) for k in keys]
 275         # GH#24096 need copy to be deep for datetime64tz case
 276         # TODO: See if we can avoid these copies
 277         arrays = [
 278             arr if not isinstance(arr, ABCIndexClass) else arr._data for arr in arrays
 279         ]
 280         arrays = [
 281             arr if not is_datetime64tz_dtype(arr) else arr.copy() for arr in arrays
 282         ]
 283     return arrays_to_mgr(arrays, data_names, index, columns, dtype=dtype)

 289 def _prep_ndarray(values, copy: bool = True) -> np.ndarray:
 290     if not isinstance(values, (np.ndarray, ABCSeries, Index)):
 291         if len(values) == 0:
 292             return np.empty((0, 0), dtype=object)
 293         elif isinstance(values, range):
 294             arr = np.arange(values.start, values.stop, values.step, dtype="int64")
 295             return arr[..., np.newaxis]
 296 
 297         def convert(v):
 298             return maybe_convert_platform(v)
 299 
 300         # we could have a 1-dim or 2-dim list here
 301         # this is equiv of np.asarray, but does object conversion
 302         # and platform dtype preservation
 303         try:
 304             if is_list_like(values[0]) or hasattr(values[0], "len"):
 305                 values = np.array([convert(v) for v in values])
 306             elif isinstance(values[0], np.ndarray) and values[0].ndim == 0:
 307                 # GH#21861
 308                 values = np.array([convert(v) for v in values])
 309             else:
 310                 values = convert(values)
 311         except (ValueError, TypeError):
 312             values = convert(values)
 313 
 314     else:
 315 
 316         # drop subclass info, do not copy data
 317         values = np.asarray(values)
 318         if copy:
 319             values = values.copy()
 320 
 321     if values.ndim == 1:
 322         values = values.reshape((values.shape[0], 1))
 323     elif values.ndim != 2:
 324         raise ValueError("Must pass 2-d input")
 325 
 326     return values

 329 def _homogenize(data, index, dtype: Optional[DtypeObj]):
 330     oindex = None
 331     homogenized = []
 332 
 333     for val in data:
 334         if isinstance(val, ABCSeries):
 335             if dtype is not None:
 336                 val = val.astype(dtype)
 337             if val.index is not index:
 338                 # Forces alignment. No need to copy data since we
 339                 # are putting it into an ndarray later
 340                 val = val.reindex(index, copy=False)
 341         else:
 342             if isinstance(val, dict):
 343                 if oindex is None:
 344                     oindex = index.astype("O")
 345 
 346                 if isinstance(index, (ABCDatetimeIndex, ABCTimedeltaIndex)):
 347                     val = com.dict_compat(val)
 348                 else:
 349                     val = dict(val)
 350                 val = lib.fast_multiget(val, oindex._values, default=np.nan)
 351             val = sanitize_array(
 352                 val, index, dtype=dtype, copy=False, raise_cast_failure=False
 353             )
 354 
 355         homogenized.append(val)
 356 
 357     return homogenized

 360 def extract_index(data) -> Index:
 361     """
 362     Try to infer an Index from the passed data, raise ValueError on failure.
 363     """
 364     index = None
 365     if len(data) == 0:
 366         index = Index([])
 367     elif len(data) > 0:
 368         raw_lengths = []
 369         indexes = []
 370 
 371         have_raw_arrays = False
 372         have_series = False
 373         have_dicts = False
 374 
 375         for val in data:
 376             if isinstance(val, ABCSeries):
 377                 have_series = True
 378                 indexes.append(val.index)
 379             elif isinstance(val, dict):
 380                 have_dicts = True
 381                 indexes.append(list(val.keys()))
 382             elif is_list_like(val) and getattr(val, "ndim", 1) == 1:
 383                 have_raw_arrays = True
 384                 raw_lengths.append(len(val))
 385 
 386         if not indexes and not raw_lengths:
 387             raise ValueError("If using all scalar values, you must pass an index")
 388 
 389         if have_series:
 390             index = union_indexes(indexes)
 391         elif have_dicts:
 392             index = union_indexes(indexes, sort=False)
 393 
 394         if have_raw_arrays:
 395             lengths = list(set(raw_lengths))
 396             if len(lengths) > 1:
 397                 raise ValueError("arrays must all be same length")
 398 
 399             if have_dicts:
 400                 raise ValueError(
 401                     "Mixing dicts with non-Series may lead to ambiguous ordering."
 402                 )
 403 
 404             if have_series:
 405                 assert index is not None  # for mypy
 406                 if lengths[0] != len(index):
 407                     msg = (
 408                         f"array length {lengths[0]} does not match index "
 409                         f"length {len(index)}"
 410                     )
 411                     raise ValueError(msg)
 412             else:
 413                 index = ibase.default_index(lengths[0])
 414 
 415     return ensure_index(index)

 418 def reorder_arrays(arrays, arr_columns, columns):
 419     # reorder according to the columns
 420     if (
 421         columns is not None
 422         and len(columns)
 423         and arr_columns is not None
 424         and len(arr_columns)
 425     ):
 426         indexer = ensure_index(arr_columns).get_indexer(columns)
 427         arr_columns = ensure_index([arr_columns[i] for i in indexer])
 428         arrays = [arrays[i] for i in indexer]
 429     return arrays, arr_columns

 432 def get_names_from_index(data):
 433     has_some_name = any(getattr(s, "name", None) is not None for s in data)
 434     if not has_some_name:
 435         return ibase.default_index(len(data))
 436 
 437     index = list(range(len(data)))
 438     count = 0
 439     for i, s in enumerate(data):
 440         n = getattr(s, "name", None)
 441         if n is not None:
 442             index[i] = n
 443         else:
 444             index[i] = f"Unnamed {count}"
 445             count += 1
 446 
 447     return index

 466 def dataclasses_to_dicts(data):
 467     """
 468     Converts a list of dataclass instances to a list of dictionaries.

 498 def to_arrays(
 499     data, columns, coerce_float: bool = False, dtype: Optional[DtypeObj] = None
 500 ):
 501     """
 502     Return list of arrays, columns.
 503     """
 504     if isinstance(data, ABCDataFrame):
 505         if columns is not None:
 506             arrays = [
 507                 data._ixs(i, axis=1).values
 508                 for i, col in enumerate(data.columns)
 509                 if col in columns
 510             ]
 511         else:
 512             columns = data.columns
 513             arrays = [data._ixs(i, axis=1).values for i in range(len(columns))]
 514 
 515         return arrays, columns
 516 
 517     if not len(data):
 518         if isinstance(data, np.ndarray):
 519             columns = data.dtype.names
 520             if columns is not None:
 521                 return [[]] * len(columns), columns
 522         return [], []  # columns if columns is not None else []
 523     if isinstance(data[0], (list, tuple)):
 524         return _list_to_arrays(data, columns, coerce_float=coerce_float, dtype=dtype)
 525     elif isinstance(data[0], abc.Mapping):
 526         return _list_of_dict_to_arrays(
 527             data, columns, coerce_float=coerce_float, dtype=dtype
 528         )
 529     elif isinstance(data[0], ABCSeries):
 530         return _list_of_series_to_arrays(
 531             data, columns, coerce_float=coerce_float, dtype=dtype
 532         )
 533     elif isinstance(data[0], Categorical):
 534         if columns is None:
 535             columns = ibase.default_index(len(data))
 536         return data, columns
 537     elif (
 538         isinstance(data, (np.ndarray, ABCSeries, Index))
 539         and data.dtype.names is not None
 540     ):
 541 
 542         columns = list(data.dtype.names)
 543         arrays = [data[k] for k in columns]
 544         return arrays, columns
 545     else:
 546         # last ditch effort
 547         data = [tuple(x) for x in data]
 548         return _list_to_arrays(data, columns, coerce_float=coerce_float, dtype=dtype)

 551 def _list_to_arrays(
 552     data: List[Scalar],
 553     columns: Union[Index, List],
 554     coerce_float: bool = False,
 555     dtype: Optional[DtypeObj] = None,
 556 ) -> Tuple[List[Scalar], Union[Index, List[Axis]]]:
 557     if len(data) > 0 and isinstance(data[0], tuple):
 558         content = list(lib.to_object_array_tuples(data).T)
 559     else:
 560         # list of lists
 561         content = list(lib.to_object_array(data).T)
 562     # gh-26429 do not raise user-facing AssertionError
 563     try:
 564         columns = _validate_or_indexify_columns(content, columns)
 565         result = _convert_object_array(content, dtype=dtype, coerce_float=coerce_float)
 566     except AssertionError as e:
 567         raise ValueError(e) from e
 568     return result, columns

 571 def _list_of_series_to_arrays(
 572     data: List,
 573     columns: Union[Index, List],
 574     coerce_float: bool = False,
 575     dtype: Optional[DtypeObj] = None,
 576 ) -> Tuple[List[Scalar], Union[Index, List[Axis]]]:
 577     if columns is None:
 578         # We know pass_data is non-empty because data[0] is a Series
 579         pass_data = [x for x in data if isinstance(x, (ABCSeries, ABCDataFrame))]
 580         columns = get_objs_combined_axis(pass_data, sort=False)
 581 
 582     indexer_cache: Dict[int, Scalar] = {}
 583 
 584     aligned_values = []
 585     for s in data:
 586         index = getattr(s, "index", None)
 587         if index is None:
 588             index = ibase.default_index(len(s))
 589 
 590         if id(index) in indexer_cache:
 591             indexer = indexer_cache[id(index)]
 592         else:
 593             indexer = indexer_cache[id(index)] = index.get_indexer(columns)
 594 
 595         values = extract_array(s, extract_numpy=True)
 596         aligned_values.append(algorithms.take_1d(values, indexer))
 597 
 598     values = np.vstack(aligned_values)
 599 
 600     if values.dtype == np.object_:
 601         content = list(values.T)
 602         columns = _validate_or_indexify_columns(content, columns)
 603         content = _convert_object_array(content, dtype=dtype, coerce_float=coerce_float)
 604         return content, columns
 605     else:
 606         return values.T, columns

 609 def _list_of_dict_to_arrays(
 610     data: List,
 611     columns: Union[Index, List],
 612     coerce_float: bool = False,
 613     dtype: Optional[DtypeObj] = None,
 614 ) -> Tuple[List[Scalar], Union[Index, List[Axis]]]:
 615     """
 616     Convert list of dicts to numpy arrays
 617 
 618     if `columns` is not passed, column names are inferred from the records
 619     - for OrderedDict and dicts, the column names match
 620       the key insertion-order from the first record to the last.
 621     - For other kinds of dict-likes, the keys are lexically sorted.
 622 
 623     Parameters
 624     ----------
 625     data : iterable
 626         collection of records (OrderedDict, dict)
 627     columns: iterables or None
 628     coerce_float : bool
 629     dtype : np.dtype
 630 
 631     Returns
 632     -------
 633     tuple
 634         arrays, columns
 635     """
 636     if columns is None:
 637         gen = (list(x.keys()) for x in data)
 638         sort = not any(isinstance(d, dict) for d in data)
 639         columns = lib.fast_unique_multiple_list_gen(gen, sort=sort)
 640 
 641     # assure that they are of the base dict class and not of derived
 642     # classes
 643     data = [(type(d) is dict) and d or dict(d) for d in data]
 644 
 645     content = list(lib.dicts_to_array(data, list(columns)).T)
 646     columns = _validate_or_indexify_columns(content, columns)
 647     content = _convert_object_array(content, dtype=dtype, coerce_float=coerce_float)
 648     return content, columns

 651 def _validate_or_indexify_columns(
 652     content: List, columns: Optional[Union[Index, List]]
 653 ) -> Union[Index, List[Axis]]:
 654     """
 655     If columns is None, make numbers as column names; Otherwise, validate that
 656     columns have valid length.
 657 
 658     Parameters
 659     ----------
 660     content: list of data
 661     columns: Iterable or None
 662 
 663     Returns
 664     -------
 665     columns: If columns is Iterable, return as is; If columns is None, assign
 666     positional column index value as columns.
 667 
 668     Raises
 669     ------
 670     1. AssertionError when content is not composed of list of lists, and if
 671         length of columns is not equal to length of content.
 672     2. ValueError when content is list of lists, but length of each sub-list
 673         is not equal
 674     3. ValueError when content is list of lists, but length of sub-list is
 675         not equal to length of content
 676     """
 677     if columns is None:
 678         columns = ibase.default_index(len(content))
 679     else:
 680 
 681         # Add mask for data which is composed of list of lists
 682         is_mi_list = isinstance(columns, list) and all(
 683             isinstance(col, list) for col in columns
 684         )
 685 
 686         if not is_mi_list and len(columns) != len(content):  # pragma: no cover
 687             # caller's responsibility to check for this...
 688             raise AssertionError(
 689                 f"{len(columns)} columns passed, passed data had "
 690                 f"{len(content)} columns"
 691             )
 692         elif is_mi_list:
 693 
 694             # check if nested list column, length of each sub-list should be equal
 695             if len({len(col) for col in columns}) > 1:
 696                 raise ValueError(
 697                     "Length of columns passed for MultiIndex columns is different"
 698                 )
 699 
 700             # if columns is not empty and length of sublist is not equal to content
 701             elif columns and len(columns[0]) != len(content):
 702                 raise ValueError(
 703                     f"{len(columns[0])} columns passed, passed data had "
 704                     f"{len(content)} columns"
 705                 )
 706     return columns

 709 def _convert_object_array(
 710     content: List[Scalar], coerce_float: bool = False, dtype: Optional[DtypeObj] = None
 711 ) -> List[Scalar]:
 712     """
 713     Internal function ot convert object array.
 714 
 715     Parameters
 716     ----------
 717     content: list of processed data records
 718     coerce_float: bool, to coerce floats or not, default is False
 719     dtype: np.dtype, default is None
 720 
 721     Returns
 722     -------
 723     arrays: casted content if not object dtype, otherwise return as is in list.
 724     """
 725     # provide soft conversion of object dtypes
 726     def convert(arr):
 727         if dtype != np.dtype("O"):
 728             arr = lib.maybe_convert_objects(arr, try_float=coerce_float)
 729             arr = maybe_cast_to_datetime(arr, dtype)
 730         return arr
 731 
 732     arrays = [convert(arr) for arr in content]
 733 
 734     return arrays

 741 def sanitize_index(data, index: Index):

 744     through a non-Index.
 745     """
