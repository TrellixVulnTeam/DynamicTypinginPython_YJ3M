 180         if axes is None:

 234     def _rebuild_blknos_and_blklocs(self) -> None:
 235         """
 236         Update mgr._blknos / mgr._blklocs.
 237         """
 238         new_blknos = np.empty(self.shape[0], dtype=np.int64)
 239         new_blklocs = np.empty(self.shape[0], dtype=np.int64)
 240         new_blknos.fill(-1)
 241         new_blklocs.fill(-1)

 244             rl = blk.mgr_locs
 245             new_blknos[rl.indexer] = blkno

 330     def reduce(self, func, *args, **kwargs):
 331         # If 2D, we assume that we're operating column-wise
 332         if self.ndim == 1:
 333             # we'll be returning a scalar
 334             blk = self.blocks[0]
 335             return func(blk.values, *args, **kwargs)
 336 
 337         res = {}
 338         for blk in self.blocks:
 339             bres = func(blk.values, *args, **kwargs)
 340 
 341             if np.ndim(bres) == 0:
 342                 # EA
 343                 assert blk.shape[0] == 1
 344                 new_res = zip(blk.mgr_locs.as_array, [bres])
 345             else:
 346                 assert bres.ndim == 1, bres.shape
 347                 assert blk.shape[0] == len(bres), (blk.shape, bres.shape, args, kwargs)
 348                 new_res = zip(blk.mgr_locs.as_array, bres)
 349 
 350             nr = dict(new_res)
 351             assert not any(key in res for key in nr)
 352             res.update(nr)
 353 
 354         return res

 362     def apply(self: T, f, align_keys=None, **kwargs) -> T:
 363         """
 364         Iterate over the blocks, collect and create a new BlockManager.
 365 
 366         Parameters
 367         ----------
 368         f : str or callable
 369             Name of the Block method to apply.
 370 
 371         Returns
 372         -------
 373         BlockManager
 374         """
 375         assert "filter" not in kwargs
 376 
 377         align_keys = align_keys or []
 378         result_blocks: List[Block] = []
 379         # fillna: Series/DataFrame is responsible for making sure value is aligned
 380 
 381         aligned_args = {k: kwargs[k] for k in align_keys}
 382 
 383         for b in self.blocks:
 384 
 385             if aligned_args:
 386 
 387                 for k, obj in aligned_args.items():
 388                     if isinstance(obj, (ABCSeries, ABCDataFrame)):
 389                         # The caller is responsible for ensuring that
 390                         #  obj.axes[-1].equals(self.items)
 391                         if obj.ndim == 1:
 392                             kwargs[k] = obj.iloc[b.mgr_locs.indexer]._values
 393                         else:
 394                             kwargs[k] = obj.iloc[:, b.mgr_locs.indexer]._values
 395                     else:
 396                         # otherwise we have an ndarray
 397                         kwargs[k] = obj[b.mgr_locs.indexer]
 398 
 399             if callable(f):
 400                 applied = b.apply(f, **kwargs)
 401             else:
 402                 applied = getattr(b, f)(**kwargs)
 403             result_blocks = _extend_blocks(applied, result_blocks)
 404 
 405         if len(result_blocks) == 0:
 406             return self.make_empty(self.axes)
 407 
 408         return type(self).from_blocks(result_blocks, self.axes)

 410     def quantile(
 411         self,
 412         axis: int = 0,
 413         transposed: bool = False,
 414         interpolation="linear",
 415         qs=None,
 416         numeric_only=None,
 417     ) -> "BlockManager":
 418         """
 419         Iterate over blocks applying quantile reduction.
 420         This routine is intended for reduction type operations and
 421         will do inference on the generated blocks.
 422 
 423         Parameters
 424         ----------
 425         axis: reduction axis, default 0
 426         transposed: bool, default False
 427             we are holding transposed data
 428         interpolation : type of interpolation, default 'linear'
 429         qs : a scalar or list of the quantiles to be computed
 430         numeric_only : ignored
 431 
 432         Returns
 433         -------
 434         BlockManager
 435         """
 436         # Series dispatches to DataFrame for quantile, which allows us to
 437         #  simplify some of the code here and in the blocks
 438         assert self.ndim >= 2
 439 
 440         def get_axe(block, qs, axes):
 441             # Because Series dispatches to DataFrame, we will always have
 442             #  block.ndim == 2
 443             from pandas import Float64Index
 444 
 445             if is_list_like(qs):
 446                 ax = Float64Index(qs)
 447             else:
 448                 ax = axes[0]
 449             return ax
 450 
 451         axes, blocks = [], []
 452         for b in self.blocks:
 453             block = b.quantile(axis=axis, qs=qs, interpolation=interpolation)
 454 
 455             axe = get_axe(b, qs, axes=self.axes)
 456 
 457             axes.append(axe)
 458             blocks.append(block)
 459 
 460         # note that some DatetimeTZ, Categorical are always ndim==1
 461         ndim = {b.ndim for b in blocks}
 462         assert 0 not in ndim, ndim
 463 
 464         if 2 in ndim:
 465 
 466             new_axes = list(self.axes)
 467 
 468             # multiple blocks that are reduced
 469             if len(blocks) > 1:
 470                 new_axes[1] = axes[0]
 471 
 472                 # reset the placement to the original
 473                 for b, sb in zip(blocks, self.blocks):
 474                     b.mgr_locs = sb.mgr_locs
 475 
 476             else:
 477                 new_axes[axis] = Index(np.concatenate([ax._values for ax in axes]))
 478 
 479             if transposed:
 480                 new_axes = new_axes[::-1]
 481                 blocks = [
 482                     b.make_block(b.values.T, placement=np.arange(b.shape[1]))
 483                     for b in blocks
 484                 ]
 485 
 486             return type(self)(blocks, new_axes)
 487 
 488         # single block, i.e. ndim == {1}
 489         values = concat_compat([b.values for b in blocks])
 490 
 491         # compute the orderings of our original data
 492         if len(self.blocks) > 1:
 493 
 494             indexer = np.empty(len(self.axes[0]), dtype=np.intp)
 495             i = 0
 496             for b in self.blocks:
 497                 for j in b.mgr_locs:
 498                     indexer[j] = i
 499                     i = i + 1
 500 
 501             values = values.take(indexer)
 502 
 503         return SingleBlockManager(
 504             make_block(values, ndim=1, placement=np.arange(len(values))), axes[0],
 505         )

 596     def replace_list(
 597         self, src_list, dest_list, inplace: bool = False, regex: bool = False
 598     ) -> "BlockManager":
 599         """ do a list replace """
 600         inplace = validate_bool_kwarg(inplace, "inplace")
 601 
 602         # figure out our mask apriori to avoid repeated replacements
 603         values = self.as_array()
 604 
 605         def comp(s, regex=False):
 606             """
 607             Generate a bool array by perform an equality check, or perform
 608             an element-wise regular expression matching
 609             """
 610             if isna(s):
 611                 return isna(values)
 612 
 613             s = com.maybe_box_datetimelike(s)
 614             return _compare_or_regex_search(values, s, regex)
 615 
 616         masks = [comp(s, regex) for s in src_list]
 617 
 618         result_blocks = []
 619         src_len = len(src_list) - 1
 620         for blk in self.blocks:
 621 
 622             # its possible to get multiple result blocks here
 623             # replace ALWAYS will return a list
 624             rb = [blk if inplace else blk.copy()]
 625             for i, (s, d) in enumerate(zip(src_list, dest_list)):
 626                 new_rb: List[Block] = []
 627                 for b in rb:
 628                     m = masks[i][b.mgr_locs.indexer]
 629                     convert = i == src_len  # only convert once at the end
 630                     result = b._replace_coerce(
 631                         mask=m,
 632                         to_replace=s,
 633                         value=d,
 634                         inplace=inplace,
 635                         convert=convert,
 636                         regex=regex,
 637                     )
 638                     if m.any() or convert:
 639                         new_rb = _extend_blocks(result, new_rb)
 640                     else:
 641                         new_rb.append(b)
 642                 rb = new_rb
 643             result_blocks.extend(rb)
 644 
 645         bm = type(self).from_blocks(result_blocks, self.axes)
 646         bm._consolidate_inplace()
 647         return bm

 712     def _combine(self, blocks: List[Block], copy: bool = True) -> "BlockManager":
 713         """ return a new manager with the blocks """
 714         if len(blocks) == 0:
 715             return self.make_empty()

 717         # FIXME: optimization potential
 718         indexer = np.sort(np.concatenate([b.mgr_locs.as_array for b in blocks]))
 719         inv_indexer = lib.get_reverse_indexer(indexer, self.shape[0])

 721         new_blocks = []

 723             b = b.copy(deep=copy)

 725             new_blocks.append(b)

 781     def as_array(
 782         self,
 783         transpose: bool = False,
 784         dtype=None,
 785         copy: bool = False,
 786         na_value=lib.no_default,
 787     ) -> np.ndarray:
 788         """
 789         Convert the blockmanager data into an numpy array.
 790 
 791         Parameters
 792         ----------
 793         transpose : bool, default False
 794             If True, transpose the return array.
 795         dtype : object, default None
 796             Data type of the return array.
 797         copy : bool, default False
 798             If True then guarantee that a copy is returned. A value of
 799             False does not guarantee that the underlying data is not
 800             copied.
 801         na_value : object, default lib.no_default
 802             Value to be used as the missing value sentinel.
 803 
 804         Returns
 805         -------
 806         arr : ndarray
 807         """
 808         if len(self.blocks) == 0:
 809             arr = np.empty(self.shape, dtype=float)
 810             return arr.transpose() if transpose else arr
 811 
 812         # We want to copy when na_value is provided to avoid
 813         # mutating the original object
 814         copy = copy or na_value is not lib.no_default
 815 
 816         if self._is_single_block and self.blocks[0].is_extension:
 817             # Avoid implicit conversion of extension blocks to object
 818             arr = (
 819                 self.blocks[0]
 820                 .values.to_numpy(dtype=dtype, na_value=na_value)
 821                 .reshape(self.blocks[0].shape)
 822             )
 823         elif self._is_single_block or not self.is_mixed_type:
 824             arr = np.asarray(self.blocks[0].get_values())
 825             if dtype:
 826                 arr = arr.astype(dtype, copy=False)
 827         else:
 828             arr = self._interleave(dtype=dtype, na_value=na_value)
 829             # The underlying data was copied within _interleave
 830             copy = False
 831 
 832         if copy:
 833             arr = arr.copy()
 834 
 835         if na_value is not lib.no_default:
 836             arr[isna(arr)] = na_value
 837 
 838         return arr.transpose() if transpose else arr

 840     def _interleave(self, dtype=None, na_value=lib.no_default) -> np.ndarray:
 841         """
 842         Return ndarray from blocks with specified item order
 843         Items must be contained in the blocks
 844         """
 845         if not dtype:
 846             dtype = _interleaved_dtype(self.blocks)
 847 
 848         # TODO: https://github.com/pandas-dev/pandas/issues/22791
 849         # Give EAs some input on what happens here. Sparse needs this.
 850         if isinstance(dtype, SparseDtype):
 851             dtype = dtype.subtype
 852         elif is_extension_array_dtype(dtype):
 853             dtype = "object"
 854 
 855         result = np.empty(self.shape, dtype=dtype)
 856 
 857         itemmask = np.zeros(self.shape[0])
 858 
 859         for blk in self.blocks:
 860             rl = blk.mgr_locs
 861             if blk.is_extension:
 862                 # Avoid implicit conversion of extension blocks to object
 863                 arr = blk.values.to_numpy(dtype=dtype, na_value=na_value)
 864             else:
 865                 arr = blk.get_values(dtype)
 866             result[rl.indexer] = arr
 867             itemmask[rl.indexer] = 1
 868 
 869         if not itemmask.all():
 870             raise AssertionError("Some items were not contained in blocks")
 871 
 872         return result

 980     def idelete(self, indexer):
 981         """
 982         Delete selected locations in-place (new block and array, same BlockManager)
 983         """
 984         is_deleted = np.zeros(self.shape[0], dtype=np.bool_)
 985         is_deleted[indexer] = True
 986         ref_loc_offset = -is_deleted.cumsum()
 987 
 988         is_blk_deleted = [False] * len(self.blocks)
 989 
 990         if isinstance(indexer, int):
 991             affected_start = indexer
 992         else:
 993             affected_start = is_deleted.nonzero()[0][0]
 994 
 995         for blkno, _ in _fast_count_smallints(self.blknos[affected_start:]):
 996             blk = self.blocks[blkno]
 997             bml = blk.mgr_locs
 998             blk_del = is_deleted[bml.indexer].nonzero()[0]
 999 
1000             if len(blk_del) == len(bml):
1001                 is_blk_deleted[blkno] = True
1002                 continue
1003             elif len(blk_del) != 0:
1004                 blk.delete(blk_del)
1005                 bml = blk.mgr_locs
1006 
1007             blk.mgr_locs = bml.add(ref_loc_offset[bml.indexer])
1008 
1009         # FIXME: use Index.delete as soon as it uses fastpath=True
1010         self.axes[0] = self.items[~is_deleted]
1011         self.blocks = tuple(
1012             b for blkno, b in enumerate(self.blocks) if not is_blk_deleted[blkno]
1013         )

1016     def iset(self, loc: Union[int, slice, np.ndarray], value):
1017         """
1018         Set new item in-place. Does not consolidate. Adds new Block if not
1019         contained in the current set of items
1020         """
1021         # FIXME: refactor, clearly separate broadcasting & zip-like assignment
1022         #        can prob also fix the various if tests for sparse/categorical
1023         if self._blklocs is None and self.ndim > 1:
1024             self._rebuild_blknos_and_blklocs()
1025 
1026         value_is_extension_type = is_extension_array_dtype(value)
1027 
1028         # categorical/sparse/datetimetz
1029         if value_is_extension_type:
1030 
1031             def value_getitem(placement):
1032                 return value
1033 
1034         else:
1035             if value.ndim == self.ndim - 1:
1036                 value = _safe_reshape(value, (1,) + value.shape)
1037 
1038                 def value_getitem(placement):
1039                     return value
1040 
1041             else:
1042 
1043                 def value_getitem(placement):
1044                     return value[placement.indexer]
1045 
1046             if value.shape[1:] != self.shape[1:]:
1047                 raise AssertionError(
1048                     "Shape of new values must be compatible with manager shape"
1049                 )
1050 
1051         if lib.is_integer(loc):
1052             # We have 6 tests where loc is _not_ an int.
1053             # In this case, get_blkno_placements will yield only one tuple,
1054             #  containing (self._blknos[loc], BlockPlacement(slice(0, 1, 1)))
1055             loc = [loc]
1056 
1057         # Accessing public blknos ensures the public versions are initialized
1058         blknos = self.blknos[loc]
1059         blklocs = self.blklocs[loc].copy()
1060 
1061         unfit_mgr_locs = []

1067             if blk.should_store(value):
1068                 blk.set(blk_locs, value_getitem(val_locs))
1069             else:
1070                 unfit_mgr_locs.append(blk.mgr_locs.as_array[blk_locs])
1071                 unfit_val_locs.append(val_locs)
1072 
1073                 # If all block items are unfit, schedule the block for removal.
1074                 if len(val_locs) == len(blk.mgr_locs):
1075                     removed_blknos.append(blkno)
1076                 else:
1077                     blk.delete(blk_locs)
1078                     self._blklocs[blk.mgr_locs.indexer] = np.arange(len(blk))
1079 
1080         if len(removed_blknos):
1081             # Remove blocks & update blknos accordingly
1082             is_deleted = np.zeros(self.nblocks, dtype=np.bool_)
1083             is_deleted[removed_blknos] = True
1084 
1085             new_blknos = np.empty(self.nblocks, dtype=np.int64)
1086             new_blknos.fill(-1)
1087             new_blknos[~is_deleted] = np.arange(self.nblocks - len(removed_blknos))
1088             self._blknos = new_blknos[self._blknos]
1089             self.blocks = tuple(
1090                 blk for i, blk in enumerate(self.blocks) if i not in set(removed_blknos)
1091             )
1092 
1093         if unfit_val_locs:
1094             unfit_mgr_locs = np.concatenate(unfit_mgr_locs)
1095             unfit_count = len(unfit_mgr_locs)
1096 
1097             new_blocks: List[Block] = []
1098             if value_is_extension_type:
1099                 # This code (ab-)uses the fact that EA blocks contain only
1100                 # one item.
1101                 # TODO(EA2D): special casing unnecessary with 2D EAs
1102                 new_blocks.extend(
1103                     make_block(
1104                         values=value,
1105                         ndim=self.ndim,
1106                         placement=slice(mgr_loc, mgr_loc + 1),
1107                     )
1108                     for mgr_loc in unfit_mgr_locs
1109                 )
1110 
1111                 self._blknos[unfit_mgr_locs] = np.arange(unfit_count) + len(self.blocks)
1112                 self._blklocs[unfit_mgr_locs] = 0
1113 
1114             else:
1115                 # unfit_val_locs contains BlockPlacement objects
1116                 unfit_val_items = unfit_val_locs[0].append(unfit_val_locs[1:])
1117 
1118                 new_blocks.append(
1119                     make_block(
1120                         values=value_getitem(unfit_val_items),
1121                         ndim=self.ndim,
1122                         placement=unfit_mgr_locs,
1123                     )
1124                 )
1125 
1126                 self._blknos[unfit_mgr_locs] = len(self.blocks)
1127                 self._blklocs[unfit_mgr_locs] = np.arange(unfit_count)

1892 def _compare_or_regex_search(
1893     a: ArrayLike, b: Scalar, regex: bool = False
1894 ) -> Union[ArrayLike, bool]:
1895     """
1896     Compare two array_like inputs of the same shape or two scalar values
1897 
1898     Calls operator.eq or re.search, depending on regex argument. If regex is
1899     True, perform an element-wise regex matching.
1900 
1901     Parameters
1902     ----------
1903     a : array_like
1904     b : scalar
1905     regex : bool, default False
1906 
1907     Returns
1908     -------
1909     mask : array_like of bool
1910     """
1911 
1912     def _check_comparison_types(
1913         result: Union[ArrayLike, bool], a: ArrayLike, b: Scalar,
1914     ):
1915         """
1916         Raises an error if the two arrays (a,b) cannot be compared.
1917         Otherwise, returns the comparison result as expected.
1918         """
1919         if is_scalar(result) and isinstance(a, np.ndarray):
1920             type_names = [type(a).__name__, type(b).__name__]
1921 
1922             if isinstance(a, np.ndarray):
1923                 type_names[0] = f"ndarray(dtype={a.dtype})"
1924 
1925             raise TypeError(
1926                 f"Cannot compare types {repr(type_names[0])} and {repr(type_names[1])}"
1927             )
1928 
1929     if not regex:
1930         op = lambda x: operator.eq(x, b)
1931     else:
1932         op = np.vectorize(
1933             lambda x: bool(re.search(b, x))
1934             if isinstance(x, str) and isinstance(b, str)
1935             else False
1936         )
1937 
1938     # GH#32621 use mask to avoid comparing to NAs
1939     if isinstance(a, np.ndarray) and not isinstance(b, np.ndarray):
1940         mask = np.reshape(~(isna(a)), a.shape)
1941     if isinstance(a, np.ndarray):
1942         a = a[mask]
1943 
1944     if is_datetimelike_v_numeric(a, b) or is_numeric_v_string_like(a, b):
1945         # GH#29553 avoid deprecation warnings from numpy
1946         _check_comparison_types(False, a, b)
1947         return False
1948 
1949     result = op(a)
1950 
1951     if isinstance(result, np.ndarray):
1952         # The shape of the mask can differ to that of the result
1953         # since we may compare only a subset of a's or b's elements
1954         tmp = np.zeros(mask.shape, dtype=np.bool)
1955         tmp[mask] = result
