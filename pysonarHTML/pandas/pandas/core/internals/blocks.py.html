 115     def __init__(self, values, placement, ndim=None):
 116         self.ndim = self._check_ndim(values, ndim)

 126     def _check_ndim(self, values, ndim):
 127         """
 128         ndim inference and validation.
 129 
 130         Infers ndim from 'values' if not provided to __init__.
 131         Validates that values.ndim and ndim are consistent if and only if
 132         the class variable '_validate_ndim' is True.
 133 
 134         Parameters
 135         ----------
 136         values : array-like
 137         ndim : int or None
 138 
 139         Returns
 140         -------
 141         ndim : int
 142 
 143         Raises
 144         ------
 145         ValueError : the number of dimensions do not match
 146         """
 147         if ndim is None:
 148             ndim = values.ndim
 149 
 150         if self._validate_ndim and values.ndim != ndim:
 151             raise ValueError(
 152                 "Wrong number of dimensions. "
 153                 f"values.ndim != ndim [{values.ndim} != {ndim}]"
 154             )
 155         return ndim

 333     def apply(self, func, **kwargs) -> List["Block"]:
 334         """
 335         apply the function to my values; return a block if we are not
 336         one
 337         """
 338         with np.errstate(all="ignore"):
 339             result = func(self.values, **kwargs)
 340 
 341         return self._split_op_result(result)

 343     def _split_op_result(self, result) -> List["Block"]:
 344         # See also: split_and_operate
 345         if is_extension_array_dtype(result) and result.ndim > 1:
 346             # TODO(EA2D): unnecessary with 2D EAs
 347             # if we get a 2D ExtensionArray, we need to split it into 1D pieces
 348             nbs = []
 349             for i, loc in enumerate(self.mgr_locs):
 350                 vals = result[i]
 351                 block = self.make_block(values=vals, placement=[loc])
 352                 nbs.append(block)
 353             return nbs
 354 
 355         if not isinstance(result, Block):
 356             result = self.make_block(result)
 357 
 358         return [result]

 360     def fillna(
 361         self, value, limit=None, inplace: bool = False, downcast=None
 362     ) -> List["Block"]:
 363         """
 364         fillna on the block with the value. If we fail, then convert to
 365         ObjectBlock and try again
 366         """
 367         inplace = validate_bool_kwarg(inplace, "inplace")
 368 
 369         mask = isna(self.values)
 370         if limit is not None:
 371             limit = libalgos._validate_limit(None, limit=limit)
 372             mask[mask.cumsum(self.ndim - 1) > limit] = False
 373 
 374         if not self._can_hold_na:
 375             if inplace:
 376                 return [self]
 377             else:
 378                 return [self.copy()]
 379 
 380         if self._can_hold_element(value):
 381             # equivalent: _try_coerce_args(value) would not raise
 382             blocks = self.putmask(mask, value, inplace=inplace)
 383             return self._maybe_downcast(blocks, downcast)
 384 
 385         # we can't process the value, but nothing to do
 386         if not mask.any():
 387             return [self] if inplace else [self.copy()]
 388 
 389         # operate column-by-column
 390         def f(mask, val, idx):
 391             block = self.coerce_to_target_dtype(value)
 392 
 393             # slice out our block
 394             if idx is not None:
 395                 # i.e. self.ndim == 2
 396                 block = block.getitem_block(slice(idx, idx + 1))
 397             return block.fillna(value, limit=limit, inplace=inplace, downcast=None)
 398 
 399         return self.split_and_operate(None, f, inplace)
 400 
 401     def split_and_operate(self, mask, f, inplace: bool) -> List["Block"]:
 402         """
 403         split the block per-column, and apply the callable f
 404         per-column, return a new block for each. Handle
 405         masking which will not change a block unless needed.
 406 
 407         Parameters
 408         ----------
 409         mask : 2-d boolean mask
 410         f : callable accepting (1d-mask, 1d values, indexer)
 411         inplace : boolean
 412 
 413         Returns
 414         -------
 415         list of blocks
 416         """
 417         if mask is None:
 418             mask = np.broadcast_to(True, shape=self.shape)
 419 
 420         new_values = self.values
 421 
 422         def make_a_block(nv, ref_loc):
 423             if isinstance(nv, list):
 424                 assert len(nv) == 1, nv
 425                 assert isinstance(nv[0], Block)
 426                 block = nv[0]
 427             else:
 428                 # Put back the dimension that was taken from it and make
 429                 # a block out of the result.
 430                 nv = _block_shape(nv, ndim=self.ndim)
 431                 block = self.make_block(values=nv, placement=ref_loc)
 432             return block
 433 
 434         # ndim == 1
 435         if self.ndim == 1:
 436             if mask.any():
 437                 nv = f(mask, new_values, None)
 438             else:
 439                 nv = new_values if inplace else new_values.copy()
 440             block = make_a_block(nv, self.mgr_locs)
 441             return [block]
 442 
 443         # ndim > 1
 444         new_blocks = []
 445         for i, ref_loc in enumerate(self.mgr_locs):
 446             m = mask[i]
 447             v = new_values[i]
 448 
 449             # need a new block
 450             if m.any():
 451                 nv = f(m, v, i)
 452             else:
 453                 nv = v if inplace else v.copy()
 454 
 455             block = make_a_block(nv, [ref_loc])
 456             new_blocks.append(block)
 457 
 458         return new_blocks

 460     def _maybe_downcast(self, blocks: List["Block"], downcast=None) -> List["Block"]:
 461 
 462         # no need to downcast our float
 463         # unless indicated
 464         if downcast is None and (
 465             self.is_float or self.is_timedelta or self.is_datetime
 466         ):
 467             return blocks
 468 
 469         return _extend_blocks([b.downcast(downcast) for b in blocks])

 471     def downcast(self, dtypes=None):
 472         """ try to downcast each item to the dict of dtypes if present """
 473         # turn it off completely
 474         if dtypes is False:
 475             return self
 476 
 477         values = self.values
 478 
 479         # single block handling
 480         if self._is_single_block:
 481 
 482             # try to cast all non-floats here
 483             if dtypes is None:
 484                 dtypes = "infer"
 485 
 486             nv = maybe_downcast_to_dtype(values, dtypes)
 487             return self.make_block(nv)
 488 
 489         # ndim > 1
 490         if dtypes is None:
 491             return self
 492 
 493         if not (dtypes == "infer" or isinstance(dtypes, dict)):
 494             raise ValueError(
 495                 "downcast must have a dictionary or 'infer' as its argument"
 496             )
 497         elif dtypes != "infer":
 498             raise AssertionError("dtypes as dict is not supported yet")
 499 
 500         # operate column-by-column
 501         # this is expensive as it splits the blocks items-by-item
 502         def f(mask, val, idx):
 503             val = maybe_downcast_to_dtype(val, dtype="infer")
 504             return val
 505 
 506         return self.split_and_operate(None, f, False)

 508     def astype(self, dtype, copy: bool = False, errors: str = "raise"):
 509         """
 510         Coerce to the new dtype.
 511 
 512         Parameters
 513         ----------
 514         dtype : str, dtype convertible
 515         copy : bool, default False
 516             copy if indicated
 517         errors : str, {'raise', 'ignore'}, default 'ignore'
 518             - ``raise`` : allow exceptions to be raised
 519             - ``ignore`` : suppress exceptions. On error return original object
 520 
 521         Returns
 522         -------
 523         Block
 524         """
 525         errors_legal_values = ("raise", "ignore")
 526 
 527         if errors not in errors_legal_values:
 528             invalid_arg = (
 529                 "Expected value of kwarg 'errors' to be one of "
 530                 f"{list(errors_legal_values)}. Supplied value is '{errors}'"
 531             )
 532             raise ValueError(invalid_arg)
 533 
 534         if inspect.isclass(dtype) and issubclass(dtype, ExtensionDtype):
 535             msg = (
 536                 f"Expected an instance of {dtype.__name__}, "
 537                 "but got the class instead. Try instantiating 'dtype'."
 538             )
 539             raise TypeError(msg)
 540 
 541         if dtype is not None:
 542             dtype = pandas_dtype(dtype)
 543 
 544         # may need to convert to categorical
 545         if is_categorical_dtype(dtype):
 546 
 547             if is_categorical_dtype(self.values.dtype):
 548                 # GH 10696/18593: update an existing categorical efficiently
 549                 return self.make_block(self.values.astype(dtype, copy=copy))
 550 
 551             return self.make_block(Categorical(self.values, dtype=dtype))
 552 
 553         dtype = pandas_dtype(dtype)
 554 
 555         # astype processing
 556         if is_dtype_equal(self.dtype, dtype):
 557             if copy:
 558                 return self.copy()
 559             return self
 560 
 561         # force the copy here
 562         if self.is_extension:
 563             # TODO: Should we try/except this astype?
 564             values = self.values.astype(dtype)
 565         else:
 566             if issubclass(dtype.type, str):
 567 
 568                 # use native type formatting for datetime/tz/timedelta
 569                 if self.is_datelike:
 570                     values = self.to_native_types()
 571 
 572                 # astype formatting
 573                 else:
 574                     # Because we have neither is_extension nor is_datelike,
 575                     #  self.values already has the correct shape
 576                     values = self.values
 577 
 578             else:
 579                 values = self.get_values(dtype=dtype)
 580 
 581             # _astype_nansafe works fine with 1-d only
 582             vals1d = values.ravel()
 583             try:
 584                 values = astype_nansafe(vals1d, dtype, copy=True)
 585             except (ValueError, TypeError):
 586                 # e.g. astype_nansafe can fail on object-dtype of strings
 587                 #  trying to convert to float
 588                 if errors == "raise":
 589                     raise
 590                 newb = self.copy() if copy else self
 591                 return newb
 592 
 593         # TODO(EA2D): special case not needed with 2D EAs
 594         if isinstance(values, np.ndarray):
 595             values = values.reshape(self.shape)
 596 
 597         newb = make_block(values, placement=self.mgr_locs, ndim=self.ndim)
 598

 671     def replace(
 672         self,
 673         to_replace,
 674         value,
 675         inplace: bool = False,
 676         regex: bool = False,
 677         convert: bool = True,
 678     ):
 679         """
 680         replace the to_replace value with value, possible to create new
 681         blocks here this is just a call to putmask. regex is not used here.
 682         It is used in ObjectBlocks.  It is here for API compatibility.
 683         """
 684         inplace = validate_bool_kwarg(inplace, "inplace")
 685         original_to_replace = to_replace
 686 
 687         # If we cannot replace with own dtype, convert to ObjectBlock and
 688         # retry
 689         if not self._can_hold_element(to_replace):
 690             if not isinstance(to_replace, list):
 691                 if inplace:
 692                     return [self]
 693                 return [self.copy()]
 694 
 695             to_replace = [x for x in to_replace if self._can_hold_element(x)]
 696             if not len(to_replace):
 697                 # GH#28084 avoid costly checks since we can infer
 698                 #  that there is nothing to replace in this block
 699                 if inplace:
 700                     return [self]
 701                 return [self.copy()]
 702 
 703             if len(to_replace) == 1:
 704                 # _can_hold_element checks have reduced this back to the
 705                 #  scalar case and we can avoid a costly object cast
 706                 return self.replace(
 707                     to_replace[0], value, inplace=inplace, regex=regex, convert=convert,
 708                 )
 709 
 710             # GH 22083, TypeError or ValueError occurred within error handling
 711             # causes infinite loop. Cast and retry only if not objectblock.
 712             if is_object_dtype(self):
 713                 raise AssertionError
 714 
 715             # try again with a compatible block
 716             block = self.astype(object)
 717             return block.replace(
 718                 to_replace=to_replace,
 719                 value=value,
 720                 inplace=inplace,
 721                 regex=regex,
 722                 convert=convert,
 723             )
 724 
 725         values = self.values
 726         if lib.is_scalar(to_replace) and isinstance(values, np.ndarray):
 727             # The only non-DatetimeLike class that also has a non-trivial
 728             #  try_coerce_args is ObjectBlock, but that overrides replace,
 729             #  so does not get here.
 730             to_replace = convert_scalar_for_putitemlike(to_replace, values.dtype)
 731 
 732         mask = missing.mask_missing(values, to_replace)
 733 
 734         try:
 735             blocks = self.putmask(mask, value, inplace=inplace)
 736             # Note: it is _not_ the case that self._can_hold_element(value)
 737             #  is always true at this point.  In particular, that can fail
 738             #  for:
 739             #   "2u" with bool-dtype, float-dtype
 740             #   0.5 with int64-dtype
 741             #   np.nan with int64-dtype
 742         except (TypeError, ValueError):
 743             # GH 22083, TypeError or ValueError occurred within error handling
 744             # causes infinite loop. Cast and retry only if not objectblock.
 745             if is_object_dtype(self):
 746                 raise
 747 
 748             assert not self._can_hold_element(value), value
 749 
 750             # try again with a compatible block
 751             block = self.astype(object)
 752             return block.replace(
 753                 to_replace=original_to_replace,
 754                 value=value,
 755                 inplace=inplace,
 756                 regex=regex,
 757                 convert=convert,
 758             )
 759         if convert:
 760             blocks = [b.convert(numeric=False, copy=not inplace) for b in blocks]
 761         return blocks

 767     def setitem(self, indexer, value):
 768         """
 769         Attempt self.values[indexer] = value, possibly creating a new array.
 770 
 771         Parameters
 772         ----------
 773         indexer : tuple, list-like, array-like, slice
 774             The subset of self.values to set
 775         value : object
 776             The value being set
 777 
 778         Returns
 779         -------
 780         Block
 781 
 782         Notes
 783         -----
 784         `indexer` is a direct slice/positional indexer. `value` must
 785         be a compatible shape.
 786         """
 787         transpose = self.ndim == 2
 788 
 789         if isinstance(indexer, np.ndarray) and indexer.ndim > self.ndim:
 790             raise ValueError(f"Cannot set values with ndim > {self.ndim}")
 791 
 792         # coerce None values, if appropriate
 793         if value is None:
 794             if self.is_numeric:
 795                 value = np.nan
 796 
 797         # coerce if block dtype can store value
 798         values = self.values
 799         if self._can_hold_element(value):
 800             # We only get here for non-Extension Blocks, so _try_coerce_args
 801             #  is only relevant for DatetimeBlock and TimedeltaBlock
 802             if lib.is_scalar(value):
 803                 value = convert_scalar_for_putitemlike(value, values.dtype)
 804 
 805         else:
 806             # current dtype cannot store value, coerce to common dtype
 807 
 808             if hasattr(value, "dtype"):
 809                 dtype = value.dtype
 810 
 811             elif lib.is_scalar(value) and not isna(value):
 812                 dtype, _ = infer_dtype_from_scalar(value, pandas_dtype=True)
 813 
 814             else:
 815                 # e.g. we are bool dtype and value is nan
 816                 # TODO: watch out for case with listlike value and scalar/empty indexer
 817                 dtype, _ = maybe_promote(np.array(value).dtype)
 818                 return self.astype(dtype).setitem(indexer, value)
 819 
 820             dtype = find_common_type([values.dtype, dtype])
 821             assert not is_dtype_equal(self.dtype, dtype)
 822             # otherwise should have _can_hold_element
 823 
 824             return self.astype(dtype).setitem(indexer, value)
 825 
 826         # value must be storable at this moment
 827         if is_extension_array_dtype(getattr(value, "dtype", None)):
 828             # We need to be careful not to allow through strings that
 829             #  can be parsed to EADtypes
 830             is_ea_value = True
 831             arr_value = value
 832         else:
 833             is_ea_value = False
 834             arr_value = np.array(value)
 835 
 836         if transpose:
 837             values = values.T
 838 
 839         # length checking
 840         check_setitem_lengths(indexer, value, values)
 841         exact_match = (
 842             len(arr_value.shape)
 843             and arr_value.shape[0] == values.shape[0]
 844             and arr_value.size == values.size
 845         )
 846         if is_empty_indexer(indexer, arr_value):
 847             # GH#8669 empty indexers
 848             pass
 849 
 850         elif is_scalar_indexer(indexer, self.ndim):
 851             # setting a single element for each dim and with a rhs that could
 852             #  be e.g. a list; see GH#6043
 853             values[indexer] = value
 854 
 855         elif exact_match and is_categorical_dtype(arr_value.dtype):
 856             # GH25495 - If the current dtype is not categorical,
 857             # we need to create a new categorical block
 858             values[indexer] = value
 859             return self.make_block(Categorical(self.values, dtype=arr_value.dtype))
 860 
 861         elif exact_match and is_ea_value:
 862             # GH#32395 if we're going to replace the values entirely, just
 863             #  substitute in the new array
 864             return self.make_block(arr_value)
 865 
 866         # if we are an exact match (ex-broadcasting),
 867         # then use the resultant dtype
 868         elif exact_match:
 869             # We are setting _all_ of the array's values, so can cast to new dtype
 870             values[indexer] = value
 871 
 872             values = values.astype(arr_value.dtype, copy=False)
 873 
 874         # set
 875         else:
 876             values[indexer] = value
 877 
 878         if transpose:
 879             values = values.T
 880         block = self.make_block(values)
 881         return block

 883     def putmask(
 884         self, mask, new, inplace: bool = False, axis: int = 0, transpose: bool = False,
 885     ) -> List["Block"]:
 886         """
 887         putmask the data to the block; it is possible that we may create a
 888         new dtype of block
 889 
 890         Return the resulting block(s).
 891 
 892         Parameters
 893         ----------
 894         mask : np.ndarray[bool], SparseArray[bool], or BooleanArray
 895         new : a ndarray/object
 896         inplace : bool, default False
 897             Perform inplace modification.
 898         axis : int
 899         transpose : bool, default False
 900             Set to True if self is stored with axes reversed.
 901 
 902         Returns
 903         -------
 904         List[Block]
 905         """
 906         mask = _extract_bool_array(mask)
 907         assert not isinstance(new, (ABCIndexClass, ABCSeries, ABCDataFrame))
 908 
 909         new_values = self.values  # delay copy if possible.
 910         # if we are passed a scalar None, convert it here
 911         if not is_list_like(new) and isna(new) and not self.is_object:
 912             # FIXME: make sure we have compatible NA
 913             new = self.fill_value
 914 
 915         if self._can_hold_element(new):
 916             # We only get here for non-Extension Blocks, so _try_coerce_args
 917             #  is only relevant for DatetimeBlock and TimedeltaBlock
 918             if lib.is_scalar(new):
 919                 new = convert_scalar_for_putitemlike(new, self.values.dtype)
 920 
 921             if transpose:
 922                 new_values = new_values.T
 923 
 924             # If the default repeat behavior in np.putmask would go in the
 925             # wrong direction, then explicitly repeat and reshape new instead
 926             if getattr(new, "ndim", 0) >= 1:
 927                 if self.ndim - 1 == new.ndim and axis == 1:
 928                     new = np.repeat(new, new_values.shape[-1]).reshape(self.shape)
 929                 new = new.astype(new_values.dtype)
 930 
 931             if new_values is self.values and not inplace:
 932                 new_values = new_values.copy()
 933             # we require exact matches between the len of the
 934             # values we are setting (or is compat). np.putmask
 935             # doesn't check this and will simply truncate / pad
 936             # the output, but we want sane error messages
 937             #
 938             # TODO: this prob needs some better checking
 939             # for 2D cases
 940             if (
 941                 is_list_like(new)
 942                 and np.any(mask[mask])
 943                 and getattr(new, "ndim", 1) == 1
 944             ):
 945                 if mask[mask].shape[-1] == len(new):
 946                     # GH 30567
 947                     # If length of ``new`` is less than the length of ``new_values``,
 948                     # `np.putmask` would first repeat the ``new`` array and then
 949                     # assign the masked values hence produces incorrect result.
 950                     # `np.place` on the other hand uses the ``new`` values at it is
 951                     # to place in the masked locations of ``new_values``
 952                     np.place(new_values, mask, new)
 953                 elif mask.shape[-1] == len(new) or len(new) == 1:
 954                     np.putmask(new_values, mask, new)
 955                 else:
 956                     raise ValueError("cannot assign mismatch length to masked array")
 957             else:
 958                 np.putmask(new_values, mask, new)
 959 
 960         # maybe upcast me
 961         elif mask.any():
 962             if transpose:
 963                 mask = mask.T
 964                 if isinstance(new, np.ndarray):
 965                     new = new.T
 966                 axis = new_values.ndim - axis - 1
 967 
 968             # Pseudo-broadcast
 969             if getattr(new, "ndim", 0) >= 1:
 970                 if self.ndim - 1 == new.ndim:
 971                     new_shape = list(new.shape)
 972                     new_shape.insert(axis, 1)
 973                     new = new.reshape(tuple(new_shape))
 974 
 975             # operate column-by-column
 976             def f(mask, val, idx):
 977 
 978                 if idx is None:
 979                     # ndim==1 case.
 980                     n = new
 981                 else:
 982 
 983                     if isinstance(new, np.ndarray):
 984                         n = np.squeeze(new[idx % new.shape[0]])
 985                     else:
 986                         n = np.array(new)
 987 
 988                     # type of the new block
 989                     dtype, _ = maybe_promote(n.dtype)
 990 
 991                     # we need to explicitly astype here to make a copy
 992                     n = n.astype(dtype)
 993 
 994                 nv = _putmask_smart(val, mask, n)
 995                 return nv
 996 
 997             new_blocks = self.split_and_operate(mask, f, inplace)
 998             return new_blocks
 999 
1000         if inplace:
1001             return [self]
1002 
1003         if transpose:
1004             if new_values is None:
1005                 new_values = self.values if inplace else self.values.copy()
1006             new_values = new_values.T
1007 
1008         return [self.make_block(new_values)]

1010     def coerce_to_target_dtype(self, other):
1011         """
1012         coerce the current block to a dtype compat for other
1013         we will return a block, possibly object, and not raise
1014 
1015         we can also safely try to coerce to the same dtype
1016         and will receive the same block
1017         """
1018         # if we cannot then coerce to object
1019         dtype, _ = infer_dtype_from(other, pandas_dtype=True)
1020 
1021         if is_dtype_equal(self.dtype, dtype):
1022             return self
1023 
1024         if self.is_bool or is_object_dtype(dtype) or is_bool_dtype(dtype):
1025             # we don't upcast to bool
1026             return self.astype(object)
1027 
1028         elif (self.is_float or self.is_complex) and (
1029             is_integer_dtype(dtype) or is_float_dtype(dtype)
1030         ):
1031             # don't coerce float/complex to int
1032             return self
1033 
1034         elif (
1035             self.is_datetime
1036             or is_datetime64_dtype(dtype)
1037             or is_datetime64tz_dtype(dtype)
1038         ):
1039 
1040             # not a datetime
1041             if not (
1042                 (is_datetime64_dtype(dtype) or is_datetime64tz_dtype(dtype))
1043                 and self.is_datetime
1044             ):
1045                 return self.astype(object)
1046 
1047             # don't upcast timezone with different timezone or no timezone
1048             mytz = getattr(self.dtype, "tz", None)
1049             othertz = getattr(dtype, "tz", None)
1050 
1051             if not tz_compare(mytz, othertz):
1052                 return self.astype(object)
1053 
1054             raise AssertionError(
1055                 f"possible recursion in coerce_to_target_dtype: {self} {other}"
1056             )
1057 
1058         elif self.is_timedelta or is_timedelta64_dtype(dtype):
1059 
1060             # not a timedelta
1061             if not (is_timedelta64_dtype(dtype) and self.is_timedelta):
1062                 return self.astype(object)
1063 
1064             raise AssertionError(
1065                 f"possible recursion in coerce_to_target_dtype: {self} {other}"
1066             )
1067 
1068         try:
1069             return self.astype(dtype)
1070         except (ValueError, TypeError, OverflowError):
1071             return self.astype(object)

2662 def get_block_type(values, dtype=None):
2663     """
2664     Find the appropriate Block subclass to use for the given values and dtype.
2665 
2666     Parameters
2667     ----------
2668     values : ndarray-like
2669     dtype : numpy or pandas dtype
2670 
2671     Returns
2672     -------
2673     cls : class, subclass of Block
2674     """
2675     dtype = dtype or values.dtype
2676     vtype = dtype.type
2677 
2678     if is_sparse(dtype):
2679         # Need this first(ish) so that Sparse[datetime] is sparse
2680         cls = ExtensionBlock
2681     elif is_categorical_dtype(values.dtype):
2682         cls = CategoricalBlock
2683     elif issubclass(vtype, np.datetime64):
2684         assert not is_datetime64tz_dtype(values.dtype)
2685         cls = DatetimeBlock
2686     elif is_datetime64tz_dtype(values.dtype):
2687         cls = DatetimeTZBlock
2688     elif is_interval_dtype(dtype) or is_period_dtype(dtype):
2689         cls = ObjectValuesExtensionBlock
2690     elif is_extension_array_dtype(values.dtype):
2691         cls = ExtensionBlock
2692     elif issubclass(vtype, np.floating):
2693         cls = FloatBlock
2694     elif issubclass(vtype, np.timedelta64):
2695         assert issubclass(vtype, np.integer)
2696         cls = TimeDeltaBlock
2697     elif issubclass(vtype, np.complexfloating):
2698         cls = ComplexBlock
2699     elif issubclass(vtype, np.integer):
2700         cls = IntBlock
2701     elif dtype == np.bool_:
2702         cls = BoolBlock
2703     else:
2704         cls = ObjectBlock
2705     return cls

2787 def _putmask_smart(v: np.ndarray, mask: np.ndarray, n) -> np.ndarray:
2788     """
2789     Return a new ndarray, try to preserve dtype if possible.
2790 
2791     Parameters
2792     ----------
2793     v : np.ndarray
2794         `values`, updated in-place.
2795     mask : np.ndarray[bool]
2796         Applies to both sides (array like).
2797     n : `new values` either scalar or an array like aligned with `values`
2798 
2799     Returns
2800     -------
2801     values : ndarray with updated values
2802         this *may* be a copy of the original
2803 
2804     See Also
2805     --------
2806     ndarray.putmask
2807     """
2808     # we cannot use np.asarray() here as we cannot have conversions
2809     # that numpy does when numeric are mixed with strings
2810 
2811     # n should be the length of the mask or a scalar here
2812     if not is_list_like(n):
2813         n = np.repeat(n, len(mask))
2814 
2815     # see if we are only masking values that if putted
2816     # will work in the current dtype
2817     try:
2818         nn = n[mask]
2819     except TypeError:
2820         # TypeError: only integer scalar arrays can be converted to a scalar index
2821         pass
2822     else:
2823         # make sure that we have a nullable type
2824         # if we have nulls
2825         if not _isna_compat(v, nn[0]):
2826             pass
2827         elif not (is_float_dtype(nn.dtype) or is_integer_dtype(nn.dtype)):
2828             # only compare integers/floats
2829             pass
2830         elif not (is_float_dtype(v.dtype) or is_integer_dtype(v.dtype)):
2831             # only compare integers/floats
2832             pass
2833         else:
2834 
2835             # we ignore ComplexWarning here
2836             with warnings.catch_warnings(record=True):
2837                 warnings.simplefilter("ignore", np.ComplexWarning)
2838                 nn_at = nn.astype(v.dtype)
2839 
2840             comp = nn == nn_at
2841             if is_list_like(comp) and comp.all():
2842                 nv = v.copy()
2843                 nv[mask] = nn_at
2844                 return nv
2845 
2846     n = np.asarray(n)
2847 
2848     def _putmask_preserve(nv, n):
2849         try:
2850             nv[mask] = n[mask]
2851         except (IndexError, ValueError):
2852             nv[mask] = n
2853         return nv
2854 
2855     # preserves dtype if possible
2856     if v.dtype.kind == n.dtype.kind:
2857         return _putmask_preserve(v, n)
2858 
2859     # change the dtype if needed
2860     dtype, _ = maybe_promote(n.dtype)
2861 
2862     v = v.astype(dtype)
2863 
2864     return _putmask_preserve(v, n)
