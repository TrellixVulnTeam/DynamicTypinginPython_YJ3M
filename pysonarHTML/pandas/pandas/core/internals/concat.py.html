  31 def concatenate_block_managers(
  32     mgrs_indexers, axes, concat_axis: int, copy: bool
  33 ) -> BlockManager:
  34     """
  35     Concatenate block managers into one.
  36 
  37     Parameters
  38     ----------
  39     mgrs_indexers : list of (BlockManager, {axis: indexer,...}) tuples
  40     axes : list of Index
  41     concat_axis : int
  42     copy : bool
  43 
  44     Returns
  45     -------
  46     BlockManager
  47     """
  48     concat_plans = [
  49         _get_mgr_concatenation_plan(mgr, indexers) for mgr, indexers in mgrs_indexers
  50     ]
  51     concat_plan = _combine_concat_plans(concat_plans, concat_axis)
  52     blocks = []
  53 
  54     for placement, join_units in concat_plan:
  55 
  56         if len(join_units) == 1 and not join_units[0].indexers:
  57             b = join_units[0].block
  58             values = b.values
  59             if copy:
  60                 values = values.copy()
  61             else:
  62                 values = values.view()
  63             b = b.make_block_same_class(values, placement=placement)
  64         elif _is_uniform_join_units(join_units):
  65             blk = join_units[0].block
  66             vals = [ju.block.values for ju in join_units]
  67 
  68             if not blk.is_extension:
  69                 values = concat_compat(vals, axis=blk.ndim - 1)
  70             else:
  71                 # TODO(EA2D): special-casing not needed with 2D EAs
  72                 values = concat_compat(vals)
  73                 if not isinstance(values, ExtensionArray):
  74                     values = values.reshape(1, len(values))
  75 
  76             b = make_block(values, placement=placement, ndim=blk.ndim)
  77         else:
  78             b = make_block(
  79                 _concatenate_join_units(join_units, concat_axis, copy=copy),
  80                 placement=placement,
  81             )
  82         blocks.append(b)
  83 
  84     return BlockManager(blocks, axes)

  87 def _get_mgr_concatenation_plan(mgr, indexers):
  88     """
  89     Construct concatenation plan for given block manager and indexers.
  90 
  91     Parameters
  92     ----------
  93     mgr : BlockManager
  94     indexers : dict of {axis: indexer}
  95 
  96     Returns
  97     -------
  98     plan : list of (BlockPlacement, JoinUnit) tuples
  99 
 100     """
 101     # Calculate post-reindex shape , save for item axis which will be separate
 102     # for each block anyway.
 103     mgr_shape = list(mgr.shape)
 104     for ax, indexer in indexers.items():
 105         mgr_shape[ax] = len(indexer)
 106     mgr_shape = tuple(mgr_shape)
 107 
 108     if 0 in indexers:
 109         ax0_indexer = indexers.pop(0)
 110         blknos = algos.take_1d(mgr.blknos, ax0_indexer, fill_value=-1)
 111         blklocs = algos.take_1d(mgr.blklocs, ax0_indexer, fill_value=-1)
 112     else:
 113 
 114         if mgr._is_single_block:
 115             blk = mgr.blocks[0]
 116             return [(blk.mgr_locs, JoinUnit(blk, mgr_shape, indexers))]
 117 
 118         ax0_indexer = None
 119         blknos = mgr.blknos
 120         blklocs = mgr.blklocs
 121 
 122     plan = []
 123     for blkno, placements in libinternals.get_blkno_placements(blknos, group=False):
 124 
 125         assert placements.is_slice_like
 126 
 127         join_unit_indexers = indexers.copy()
 128 
 129         shape = list(mgr_shape)
 130         shape[0] = len(placements)
 131         shape = tuple(shape)
 132 
 133         if blkno == -1:
 134             unit = JoinUnit(None, shape)
 135         else:
 136             blk = mgr.blocks[blkno]
 137             ax0_blk_indexer = blklocs[placements.indexer]
 138 
 139             unit_no_ax0_reindexing = (
 140                 len(placements) == len(blk.mgr_locs)
 141                 and
 142                 # Fastpath detection of join unit not
 143                 # needing to reindex its block: no ax0
 144                 # reindexing took place and block
 145                 # placement was sequential before.
 146                 (
 147                     (
 148                         ax0_indexer is None
 149                         and blk.mgr_locs.is_slice_like
 150                         and blk.mgr_locs.as_slice.step == 1
 151                     )
 152                     or
 153                     # Slow-ish detection: all indexer locs
 154                     # are sequential (and length match is
 155                     # checked above).
 156                     (np.diff(ax0_blk_indexer) == 1).all()
 157                 )
 158             )
 159 
 160             # Omit indexer if no item reindexing is required.
 161             if unit_no_ax0_reindexing:
 162                 join_unit_indexers.pop(0, None)
 163             else:
 164                 join_unit_indexers[0] = ax0_blk_indexer
 165 
 166             unit = JoinUnit(blk, shape, join_unit_indexers)
 167 
 168         plan.append((placements, unit))
 169 
 170     return plan

 173 class JoinUnit:
 174     def __init__(self, block, shape, indexers=None):
 175         # Passing shape explicitly is required for cases when block is None.
 176         if indexers is None:
 177             indexers = {}
 178         self.block = block
 179         self.indexers = indexers
 180         self.shape = shape
 181 
 182     def __repr__(self) -> str:
 183         return f"{type(self).__name__}({repr(self.block)}, {self.indexers})"
 184 
 185     @cache_readonly
 186     def needs_filling(self):
 187         for indexer in self.indexers.values():
 188             # FIXME: cache results of indexer == -1 checks.
 189             if (indexer == -1).any():
 190                 return True
 191 
 192         return False
 193 
 194     @cache_readonly
 195     def dtype(self):
 196         if self.block is None:
 197             raise AssertionError("Block is None, no dtype")
 198 
 199         if not self.needs_filling:
 200             return self.block.dtype
 201         else:
 202             return _get_dtype(maybe_promote(self.block.dtype, self.block.fill_value)[0])
 203 
 204     @cache_readonly
 205     def is_na(self):
 206         if self.block is None:
 207             return True
 208 
 209         if not self.block._can_hold_na:
 210             return False
 211 
 212         # Usually it's enough to check but a small fraction of values to see if
 213         # a block is NOT null, chunks should help in such cases.  1000 value
 214         # was chosen rather arbitrarily.
 215         values = self.block.values
 216         if self.block.is_categorical:
 217             values_flat = values.categories
 218         elif is_sparse(self.block.values.dtype):
 219             return False
 220         elif self.block.is_extension:
 221             # TODO(EA2D): no need for special case with 2D EAs
 222             values_flat = values
 223         else:
 224             values_flat = values.ravel(order="K")
 225         total_len = values_flat.shape[0]
 226         chunk_len = max(total_len // 40, 1000)
 227         for i in range(0, total_len, chunk_len):
 228             if not isna(values_flat[i : i + chunk_len]).all():
 229                 return False
 230 
 231         return True
 232 
 233     def get_reindexed_values(self, empty_dtype, upcasted_na):
 234         if upcasted_na is None:
 235             # No upcasting is necessary
 236             fill_value = self.block.fill_value
 237             values = self.block.get_values()
 238         else:
 239             fill_value = upcasted_na
 240 
 241             if self.is_na:
 242                 if getattr(self.block, "is_object", False):
 243                     # we want to avoid filling with np.nan if we are
 244                     # using None; we already know that we are all
 245                     # nulls
 246                     values = self.block.values.ravel(order="K")
 247                     if len(values) and values[0] is None:
 248                         fill_value = None
 249 
 250                 if getattr(self.block, "is_datetimetz", False) or is_datetime64tz_dtype(
 251                     empty_dtype
 252                 ):
 253                     if self.block is None:
 254                         array = empty_dtype.construct_array_type()
 255                         # TODO(EA2D): special case unneeded with 2D EAs
 256                         return array(
 257                             np.full(self.shape[1], fill_value.value), dtype=empty_dtype
 258                         )
 259                 elif getattr(self.block, "is_categorical", False):
 260                     pass
 261                 elif getattr(self.block, "is_extension", False):
 262                     pass
 263                 else:
 264                     missing_arr = np.empty(self.shape, dtype=empty_dtype)
 265                     missing_arr.fill(fill_value)
 266                     return missing_arr
 267 
 268             if (not self.indexers) and (not self.block._can_consolidate):
 269                 # preserve these for validation in concat_compat
 270                 return self.block.values
 271 
 272             if self.block.is_bool and not self.block.is_categorical:
 273                 # External code requested filling/upcasting, bool values must
 274                 # be upcasted to object to avoid being upcasted to numeric.
 275                 values = self.block.astype(np.object_).values
 276             elif self.block.is_extension:
 277                 values = self.block.values
 278             else:
 279                 # No dtype upcasting is done here, it will be performed during
 280                 # concatenation itself.
 281                 values = self.block.values
 282 
 283         if not self.indexers:
 284             # If there's no indexing to be done, we want to signal outside
 285             # code that this array must be copied explicitly.  This is done
 286             # by returning a view and checking `retval.base`.
 287             values = values.view()
 288 
 289         else:
 290             for ax, indexer in self.indexers.items():
 291                 values = algos.take_nd(values, indexer, axis=ax, fill_value=fill_value)
 292 
 293         return values
 294

 296 def _concatenate_join_units(join_units, concat_axis, copy):
 297     """
 298     Concatenate values from several join units along selected axis.
 299     """
 300     if concat_axis == 0 and len(join_units) > 1:
 301         # Concatenating join units along ax0 is handled in _merge_blocks.
 302         raise AssertionError("Concatenating join units along axis0")
 303 
 304     empty_dtype, upcasted_na = _get_empty_dtype_and_na(join_units)
 305 
 306     to_concat = [
 307         ju.get_reindexed_values(empty_dtype=empty_dtype, upcasted_na=upcasted_na)
 308         for ju in join_units
 309     ]
 310 
 311     if len(to_concat) == 1:
 312         # Only one block, nothing to concatenate.
 313         concat_values = to_concat[0]
 314         if copy:
 315             if isinstance(concat_values, np.ndarray):
 316                 # non-reindexed (=not yet copied) arrays are made into a view
 317                 # in JoinUnit.get_reindexed_values
 318                 if concat_values.base is not None:
 319                     concat_values = concat_values.copy()
 320             else:
 321                 concat_values = concat_values.copy()
 322     else:
 323         concat_values = concat_compat(to_concat, axis=concat_axis)
 324 
 325     return concat_values

 328 def _get_empty_dtype_and_na(join_units):
 329     """
 330     Return dtype and N/A values to use when concatenating specified units.
 331 
 332     Returned N/A value may be None which means there was no casting involved.
 333 
 334     Returns
 335     -------
 336     dtype
 337     na
 338     """
 339     if len(join_units) == 1:
 340         blk = join_units[0].block
 341         if blk is None:
 342             return np.dtype(np.float64), np.nan
 343 
 344     if _is_uniform_reindex(join_units):
 345         # FIXME: integrate property
 346         empty_dtype = join_units[0].block.dtype
 347         upcasted_na = join_units[0].block.fill_value
 348         return empty_dtype, upcasted_na
 349 
 350     has_none_blocks = False
 351     dtypes = [None] * len(join_units)
 352     for i, unit in enumerate(join_units):
 353         if unit.block is None:
 354             has_none_blocks = True
 355         else:
 356             dtypes[i] = unit.dtype
 357 
 358     upcast_classes = defaultdict(list)
 359     null_upcast_classes = defaultdict(list)
 360     for dtype, unit in zip(dtypes, join_units):
 361         if dtype is None:
 362             continue
 363 
 364         if is_categorical_dtype(dtype):
 365             upcast_cls = "category"
 366         elif is_datetime64tz_dtype(dtype):
 367             upcast_cls = "datetimetz"
 368         elif issubclass(dtype.type, np.bool_):
 369             upcast_cls = "bool"
 370         elif issubclass(dtype.type, np.object_):
 371             upcast_cls = "object"
 372         elif is_datetime64_dtype(dtype):
 373             upcast_cls = "datetime"
 374         elif is_timedelta64_dtype(dtype):
 375             upcast_cls = "timedelta"
 376         elif is_sparse(dtype):
 377             upcast_cls = dtype.subtype.name
 378         elif is_extension_array_dtype(dtype):
 379             upcast_cls = "object"
 380         elif is_float_dtype(dtype) or is_numeric_dtype(dtype):
 381             upcast_cls = dtype.name
 382         else:
 383             upcast_cls = "float"
 384 
 385         # Null blocks should not influence upcast class selection, unless there
 386         # are only null blocks, when same upcasting rules must be applied to
 387         # null upcast classes.
 388         if unit.is_na:
 389             null_upcast_classes[upcast_cls].append(dtype)
 390         else:
 391             upcast_classes[upcast_cls].append(dtype)
 392 
 393     if not upcast_classes:
 394         upcast_classes = null_upcast_classes
 395 
 396     # TODO: de-duplicate with maybe_promote?
 397     # create the result
 398     if "object" in upcast_classes:
 399         return np.dtype(np.object_), np.nan
 400     elif "bool" in upcast_classes:
 401         if has_none_blocks:
 402             return np.dtype(np.object_), np.nan
 403         else:
 404             return np.dtype(np.bool_), None
 405     elif "category" in upcast_classes:
 406         return np.dtype(np.object_), np.nan
 407     elif "datetimetz" in upcast_classes:
 408         # GH-25014. We use NaT instead of iNaT, since this eventually
 409         # ends up in DatetimeArray.take, which does not allow iNaT.
 410         dtype = upcast_classes["datetimetz"]
 411         return dtype[0], NaT
 412     elif "datetime" in upcast_classes:
 413         return np.dtype("M8[ns]"), np.datetime64("NaT", "ns")
 414     elif "timedelta" in upcast_classes:
 415         return np.dtype("m8[ns]"), np.timedelta64("NaT", "ns")
 416     else:  # pragma
 417         try:
 418             g = np.find_common_type(upcast_classes, [])
 419         except TypeError:
 420             # At least one is an ExtensionArray
 421             return np.dtype(np.object_), np.nan
 422         else:
 423             if is_float_dtype(g):
 424                 return g, g.type(np.nan)
 425             elif is_numeric_dtype(g):
 426                 if has_none_blocks:
 427                     return np.dtype(np.float64), np.nan
 428                 else:
 429                     return g, None
 430 
 431     msg = "invalid dtype determination in get_concat_dtype"
 432     raise AssertionError(msg)

 435 def _is_uniform_join_units(join_units: List[JoinUnit]) -> bool:
 436     """
 437     Check if the join units consist of blocks of uniform type that can
 438     be concatenated using Block.concat_same_type instead of the generic
 439     _concatenate_join_units (which uses `concat_compat`).
 440 
 441     """
 442     # TODO: require dtype match in addition to same type?  e.g. DatetimeTZBlock
 443     #  cannot necessarily join
 444     return (
 445         # all blocks need to have the same type
 446         all(isinstance(ju.block, type(join_units[0].block)) for ju in join_units)
 447         and  # noqa
 448         # no blocks that would get missing values (can lead to type upcasts)
 449         # unless we're an extension dtype.
 450         all(not ju.is_na or ju.block.is_extension for ju in join_units)
 451         and
 452         # no blocks with indexers (as then the dimensions do not fit)
 453         all(not ju.indexers for ju in join_units)
 454         and
 455         # only use this path when there is something to concatenate
 456         len(join_units) > 1
 457     )
 458 
 459 
 460 def _is_uniform_reindex(join_units) -> bool:
 461     return (
 462         # TODO: should this be ju.block._can_hold_na?
 463         all(ju.block and ju.block.is_extension for ju in join_units)
 464         and len({ju.block.dtype.name for ju in join_units}) == 1
 465     )

 468 def _trim_join_unit(join_unit, length):
 469     """
 470     Reduce join_unit's shape along item axis to length.
 471 
 472     Extra items that didn't fit are returned as a separate block.
 473     """
 474     if 0 not in join_unit.indexers:
 475         extra_indexers = join_unit.indexers
 476 
 477         if join_unit.block is None:
 478             extra_block = None
 479         else:
 480             extra_block = join_unit.block.getitem_block(slice(length, None))
 481             join_unit.block = join_unit.block.getitem_block(slice(length))
 482     else:
 483         extra_block = join_unit.block
 484 
 485         extra_indexers = copy.copy(join_unit.indexers)
 486         extra_indexers[0] = extra_indexers[0][length:]
 487         join_unit.indexers[0] = join_unit.indexers[0][:length]
 488 
 489     extra_shape = (join_unit.shape[0] - length,) + join_unit.shape[1:]
 490     join_unit.shape = (length,) + join_unit.shape[1:]
 491 
 492     return JoinUnit(block=extra_block, indexers=extra_indexers, shape=extra_shape)

 495 def _combine_concat_plans(plans, concat_axis):
 496     """
 497     Combine multiple concatenation plans into one.
 498 
 499     existing_plan is updated in-place.
 500     """
 501     if len(plans) == 1:
 502         for p in plans[0]:
 503             yield p[0], [p[1]]
 504 
 505     elif concat_axis == 0:
 506         offset = 0
 507         for plan in plans:
 508             last_plc = None
 509 
 510             for plc, unit in plan:
 511                 yield plc.add(offset), [unit]
 512                 last_plc = plc
 513 
 514             if last_plc is not None:
 515                 offset += last_plc.as_slice.stop
 516 
 517     else:
 518         num_ended = [0]
 519 
 520         def _next_or_none(seq):
 521             retval = next(seq, None)
 522             if retval is None:
 523                 num_ended[0] += 1
 524             return retval
 525 
 526         plans = list(map(iter, plans))
 527         next_items = list(map(_next_or_none, plans))
 528 
 529         while num_ended[0] != len(next_items):
 530             if num_ended[0] > 0:
 531                 raise ValueError("Plan shapes are not aligned")
 532 
 533             placements, units = zip(*next_items)
 534 
 535             lengths = list(map(len, placements))
 536             min_len, max_len = min(lengths), max(lengths)
 537 
 538             if min_len == max_len:
 539                 yield placements[0], units
 540                 next_items[:] = map(_next_or_none, plans)
 541             else:
 542                 yielded_placement = None
 543                 yielded_units = [None] * len(next_items)
 544                 for i, (plc, unit) in enumerate(next_items):
 545                     yielded_units[i] = unit
 546                     if len(plc) > min_len:
 547                         # _trim_join_unit updates unit in place, so only
 548                         # placement needs to be sliced to skip min_len.
 549                         next_items[i] = (plc[min_len:], _trim_join_unit(unit, min_len))
 550                     else:
 551                         yielded_placement = plc
 552                         next_items[i] = _next_or_none(plans[i])
 553 
 554                 yield yielded_placement, yielded_units
