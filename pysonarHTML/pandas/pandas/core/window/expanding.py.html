  11 class Expanding(_Rolling_and_Expanding):
  12     """
  13     Provide expanding transformations.
  14 
  15     Parameters
  16     ----------
  17     min_periods : int, default 1
  18         Minimum number of observations in window required to have a value
  19         (otherwise result is NA).
  20     center : bool, default False
  21         Set the labels at the center of the window.
  22     axis : int or str, default 0
  23 
  24     Returns
  25     -------
  26     a Window sub-classed for the particular operation
  27 
  28     See Also
  29     --------
  30     rolling : Provides rolling window calculations.
  31     ewm : Provides exponential weighted functions.
  32 
  33     Notes
  34     -----
  35     By default, the result is set to the right edge of the window. This can be
  36     changed to the center of the window by setting ``center=True``.
  37 
  38     Examples
  39     --------
  40     >>> df = pd.DataFrame({"B": [0, 1, 2, np.nan, 4]})
  41     >>> df
  42          B
  43     0  0.0
  44     1  1.0
  45     2  2.0
  46     3  NaN
  47     4  4.0
  48 
  49     >>> df.expanding(2).sum()
  50          B
  51     0  NaN
  52     1  1.0
  53     2  3.0
  54     3  3.0
  55     4  7.0
  56     """
  57 
  58     _attributes = ["min_periods", "center", "axis"]
  59 
  60     def __init__(self, obj, min_periods=1, center=False, axis=0, **kwargs):
  61         super().__init__(obj=obj, min_periods=min_periods, center=center, axis=axis)
  62 
  63     @property
  64     def _constructor(self):
  65         return Expanding

  67     def _get_window(self, other=None, **kwargs):
  68         """
  69         Get the window length over which to perform some operation.
  70 
  71         Parameters
  72         ----------
  73         other : object, default None
  74             The other object that is involved in the operation.
  75             Such an object is involved for operations like covariance.
  76 
  77         Returns
  78         -------
  79         window : int
  80             The window length.
  81         """
  82         axis = self.obj._get_axis(self.axis)
  83         length = len(axis) + (other is not None) * len(axis)
  84 
  85         other = self.min_periods or -1
  86         return max(length, other)
