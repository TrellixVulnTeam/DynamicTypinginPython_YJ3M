  79 def calculate_min_periods(
  80     window: int,
  81     min_periods: Optional[int],
  82     num_values: int,
  83     required_min_periods: int,
  84     floor: int,
  85 ) -> int:
  86     """
  87     Calculate final minimum periods value for rolling aggregations.
  88 
  89     Parameters
  90     ----------
  91     window : passed window value
  92     min_periods : passed min periods value
  93     num_values : total number of values
  94     required_min_periods : required min periods per aggregation function
  95     floor : required min periods per aggregation function
  96 
  97     Returns
  98     -------
  99     min_periods : int
 100     """
 101     if min_periods is None:
 102         min_periods = window
 103     else:
 104         min_periods = max(required_min_periods, min_periods)
 105     if min_periods > window:
 106         raise ValueError(f"min_periods {min_periods} must be <= window {window}")
 107     elif min_periods > num_values:
 108         min_periods = num_values + 1
 109     elif min_periods < 0:
 110         raise ValueError("min_periods must be >= 0")
 111     return max(min_periods, floor)

 339     def _prep_values(self, values: Optional[np.ndarray] = None) -> np.ndarray:
 340         """Convert input to numpy arrays for Cython routines"""
 341         if values is None:
 342             values = extract_array(self._selected_obj, extract_numpy=True)
 343 
 344         # GH #12373 : rolling functions error on float32 data
 345         # make sure the data is coerced to float64
 346         if is_float_dtype(values.dtype):
 347             values = ensure_float64(values)
 348         elif is_integer_dtype(values.dtype):
 349             values = ensure_float64(values)
 350         elif needs_i8_conversion(values.dtype):
 351             raise NotImplementedError(
 352                 f"ops for {self._window_type} for this "
 353                 f"dtype {values.dtype} are not implemented"
 354             )
 355         else:
 356             try:
 357                 values = ensure_float64(values)
 358             except (ValueError, TypeError) as err:
 359                 raise TypeError(f"cannot handle this type -> {values.dtype}") from err
 360 
 361         # Convert inf to nan for C funcs
 362         inf = np.isinf(values)
 363         if inf.any():
 364             values = np.where(inf, np.nan, values)
 365 
 366         return values

 386     def _wrap_results(self, results, blocks, obj, exclude=None) -> FrameOrSeries:
 387         """
 388         Wrap the results.
 389 
 390         Parameters
 391         ----------
 392         results : list of ndarrays
 393         blocks : list of blocks
 394         obj : conformed data (may be resampled)
 395         exclude: list of columns to exclude, default to None
 396         """
 397         from pandas import Series, concat
 398 
 399         final = []
 400         for result, block in zip(results, blocks):
 401 
 402             result = self._wrap_result(result, block=block, obj=obj)
 403             if result.ndim == 1:
 404                 return result
 405             final.append(result)
 406 
 407         # if we have an 'on' column
 408         # we want to put it back into the results
 409         # in the same location
 410         columns = self._selected_obj.columns
 411         if self.on is not None and not self._on.equals(obj.index):
 412 
 413             name = self._on.name
 414             final.append(Series(self._on, index=obj.index, name=name))
 415 
 416             if self._selection is not None:
 417 
 418                 selection = ensure_index(self._selection)
 419 
 420                 # need to reorder to include original location of
 421                 # the on column (if its not already there)
 422                 if name not in selection:
 423                     columns = self.obj.columns
 424                     indexer = columns.get_indexer(selection.tolist() + [name])
 425                     columns = columns.take(sorted(indexer))
 426 
 427         # exclude nuisance columns so that they are not reindexed
 428         if exclude is not None and exclude:
 429             columns = [c for c in columns if c not in exclude]
 430 
 431             if not columns:
 432                 raise DataError("No numeric types to aggregate")
 433 
 434         if not len(final):
 435             return obj.astype("float64")
 436         return concat(final, axis=1).reindex(columns=columns, copy=False)

 438     def _center_window(self, result, window) -> np.ndarray:
 439         """
 440         Center the result in the window.
 441         """
 442         if self.axis > result.ndim - 1:
 443             raise ValueError("Requested axis is larger then no. of argument dimensions")
 444 
 445         offset = calculate_center_offset(window)
 446         if offset > 0:
 447             lead_indexer = [slice(None)] * result.ndim
 448             lead_indexer[self.axis] = slice(offset, None)
 449             result = np.copy(result[tuple(lead_indexer)])
 450         return result

 492     def _apply(
 493         self,
 494         func: Callable,
 495         center: bool,
 496         require_min_periods: int = 0,
 497         floor: int = 1,
 498         is_weighted: bool = False,
 499         name: Optional[str] = None,
 500         use_numba_cache: bool = False,
 501         **kwargs,
 502     ):
 503         """
 504         Rolling statistical measure using supplied function.
 505 
 506         Designed to be used with passed-in Cython array-based functions.
 507 
 508         Parameters
 509         ----------
 510         func : callable function to apply
 511         center : bool
 512         require_min_periods : int
 513         floor : int
 514         is_weighted : bool
 515         name : str,
 516             compatibility with groupby.rolling
 517         use_numba_cache : bool
 518             whether to cache a numba compiled function. Only available for numba
 519             enabled methods (so far only apply)
 520         **kwargs
 521             additional arguments for rolling function and window function
 522 
 523         Returns
 524         -------
 525         y : type of input
 526         """
 527         win_type = self._get_win_type(kwargs)
 528         window = self._get_window(win_type=win_type)
 529 
 530         blocks, obj = self._create_blocks(self._selected_obj)
 531         block_list = list(blocks)
 532         window_indexer = self._get_window_indexer(window)
 533 
 534         results = []
 535         exclude: List[Scalar] = []
 536         for i, b in enumerate(blocks):
 537             try:
 538                 values = self._prep_values(b.values)
 539 
 540             except (TypeError, NotImplementedError) as err:
 541                 if isinstance(obj, ABCDataFrame):
 542                     exclude.extend(b.columns)
 543                     del block_list[i]
 544                     continue
 545                 else:
 546                     raise DataError("No numeric types to aggregate") from err
 547 
 548             if values.size == 0:
 549                 results.append(values.copy())
 550                 continue
 551 
 552             # calculation function
 553             offset = calculate_center_offset(window) if center else 0
 554             additional_nans = np.array([np.nan] * offset)
 555 
 556             if not is_weighted:
 557 
 558                 def calc(x):
 559                     x = np.concatenate((x, additional_nans))
 560                     if not isinstance(self.window, BaseIndexer):
 561                         min_periods = calculate_min_periods(
 562                             window, self.min_periods, len(x), require_min_periods, floor
 563                         )
 564                     else:
 565                         min_periods = calculate_min_periods(
 566                             window_indexer.window_size,
 567                             self.min_periods,
 568                             len(x),
 569                             require_min_periods,
 570                             floor,
 571                         )
 572                     start, end = window_indexer.get_window_bounds(
 573                         num_values=len(x),
 574                         min_periods=self.min_periods,
 575                         center=self.center,
 576                         closed=self.closed,
 577                     )
 578                     return func(x, start, end, min_periods)
 579 
 580             else:
 581 
 582                 def calc(x):
 583                     x = np.concatenate((x, additional_nans))
 584                     return func(x, window, self.min_periods)
 585 
 586             with np.errstate(all="ignore"):
 587                 if values.ndim > 1:
 588                     result = np.apply_along_axis(calc, self.axis, values)
 589                 else:
 590                     result = calc(values)
 591                     result = np.asarray(result)
 592 
 593             if use_numba_cache:
 594                 NUMBA_FUNC_CACHE[(kwargs["original_func"], "rolling_apply")] = func
 595 
 596             if center:
 597                 result = self._center_window(result, window)
 598 
 599             results.append(result)
 600 
 601         return self._wrap_results(results, block_list, obj, exclude)
 602

 604         result, how = self._aggregate(func, *args, **kwargs)
 605         if result is None:
