  29 def get_center_of_mass(
  30     comass: Optional[float],
  31     span: Optional[float],
  32     halflife: Optional[float],
  33     alpha: Optional[float],
  34 ) -> float:
  35     valid_count = com.count_not_none(comass, span, halflife, alpha)
  36     if valid_count > 1:
  37         raise ValueError("comass, span, halflife, and alpha are mutually exclusive")
  38 
  39     # Convert to center of mass; domain checks ensure 0 < alpha <= 1
  40     if comass is not None:
  41         if comass < 0:
  42             raise ValueError("comass must satisfy: comass >= 0")
  43     elif span is not None:
  44         if span < 1:
  45             raise ValueError("span must satisfy: span >= 1")
  46         comass = (span - 1) / 2.0
  47     elif halflife is not None:
  48         if halflife <= 0:
  49             raise ValueError("halflife must satisfy: halflife > 0")
  50         decay = 1 - np.exp(np.log(0.5) / halflife)
  51         comass = 1 / decay - 1
  52     elif alpha is not None:
  53         if alpha <= 0 or alpha > 1:
  54             raise ValueError("alpha must satisfy: 0 < alpha <= 1")
  55         comass = (1.0 - alpha) / alpha
  56     else:
  57         raise ValueError("Must pass one of comass, span, halflife, or alpha")
  58 
  59     return float(comass)

  62 class EWM(_Rolling):
  63     r"""
  64     Provide exponential weighted (EW) functions.
  65 
  66     Available EW functions: ``mean()``, ``var()``, ``std()``, ``corr()``, ``cov()``.
  67 
  68     Exactly one parameter: ``com``, ``span``, ``halflife``, or ``alpha`` must be
  69     provided.
  70 
  71     Parameters
  72     ----------
  73     com : float, optional
  74         Specify decay in terms of center of mass,
  75         :math:`\alpha = 1 / (1 + com)`, for :math:`com \geq 0`.
  76     span : float, optional
  77         Specify decay in terms of span,
  78         :math:`\alpha = 2 / (span + 1)`, for :math:`span \geq 1`.
  79     halflife : float, optional
  80         Specify decay in terms of half-life,
  81         :math:`\alpha = 1 - \exp\left(-\ln(2) / halflife\right)`, for
  82         :math:`halflife > 0`.
  83     alpha : float, optional
  84         Specify smoothing factor :math:`\alpha` directly,
  85         :math:`0 < \alpha \leq 1`.
  86     min_periods : int, default 0
  87         Minimum number of observations in window required to have a value
  88         (otherwise result is NA).
  89     adjust : bool, default True
  90         Divide by decaying adjustment factor in beginning periods to account
  91         for imbalance in relative weightings (viewing EWMA as a moving average).
  92 
  93         - When ``adjust=True`` (default), the EW function is calculated using weights
  94           :math:`w_i = (1 - \alpha)^i`. For example, the EW moving average of the series
  95           [:math:`x_0, x_1, ..., x_t`] would be:
  96 
  97         .. math::
  98             y_t = \frac{x_t + (1 - \alpha)x_{t-1} + (1 - \alpha)^2 x_{t-2} + ... + (1 -
  99             \alpha)^t x_0}{1 + (1 - \alpha) + (1 - \alpha)^2 + ... + (1 - \alpha)^t}
 100 
 101         - When ``adjust=False``, the exponentially weighted function is calculated
 102           recursively:
 103 
 104         .. math::
 105             \begin{split}
 106                 y_0 &= x_0\\
 107                 y_t &= (1 - \alpha) y_{t-1} + \alpha x_t,
 108             \end{split}
 109     ignore_na : bool, default False
 110         Ignore missing values when calculating weights; specify ``True`` to reproduce
 111         pre-0.15.0 behavior.
 112 
 113         - When ``ignore_na=False`` (default), weights are based on absolute positions.
 114           For example, the weights of :math:`x_0` and :math:`x_2` used in calculating
 115           the final weighted average of [:math:`x_0`, None, :math:`x_2`] are
 116           :math:`(1-\alpha)^2` and :math:`1` if ``adjust=True``, and
 117           :math:`(1-\alpha)^2` and :math:`\alpha` if ``adjust=False``.
 118 
 119         - When ``ignore_na=True`` (reproducing pre-0.15.0 behavior), weights are based
 120           on relative positions. For example, the weights of :math:`x_0` and :math:`x_2`
 121           used in calculating the final weighted average of
 122           [:math:`x_0`, None, :math:`x_2`] are :math:`1-\alpha` and :math:`1` if
 123           ``adjust=True``, and :math:`1-\alpha` and :math:`\alpha` if ``adjust=False``.
 124     axis : {0, 1}, default 0
 125         The axis to use. The value 0 identifies the rows, and 1
 126         identifies the columns.
 127 
 128     Returns
 129     -------
 130     DataFrame
 131         A Window sub-classed for the particular operation.
 132 
 133     See Also
 134     --------
 135     rolling : Provides rolling window calculations.
 136     expanding : Provides expanding transformations.
 137 
 138     Notes
 139     -----
 140 
 141     More details can be found at:
 142     :ref:`Exponentially weighted windows <stats.moments.exponentially_weighted>`.
 143 
 144     Examples
 145     --------
 146     >>> df = pd.DataFrame({'B': [0, 1, 2, np.nan, 4]})
 147     >>> df
 148          B
 149     0  0.0
 150     1  1.0
 151     2  2.0
 152     3  NaN
 153     4  4.0
 154 
 155     >>> df.ewm(com=0.5).mean()
 156               B
 157     0  0.000000
 158     1  0.750000
 159     2  1.615385
 160     3  1.615385
 161     4  3.670213
 162     """
 163 
 164     _attributes = ["com", "min_periods", "adjust", "ignore_na", "axis"]
 165

 174         adjust: bool = True,

 183         self.axis = axis
 184         self.on = None

 230     def _apply(self, func):
 231         """
 232         Rolling statistical measure using supplied function. Designed to be
 233         used with passed-in Cython array-based functions.
 234 
 235         Parameters
 236         ----------
 237         func : str/callable to apply
 238 
 239         Returns
 240         -------
 241         y : same type as input argument
 242         """
 243         blocks, obj = self._create_blocks(self._selected_obj)
 244         block_list = list(blocks)
 245 
 246         results = []
 247         exclude = []
 248         for i, b in enumerate(blocks):
 249             try:
 250                 values = self._prep_values(b.values)
 251 
 252             except (TypeError, NotImplementedError) as err:
 253                 if isinstance(obj, ABCDataFrame):
 254                     exclude.extend(b.columns)
 255                     del block_list[i]
 256                     continue
 257                 else:
 258                     raise DataError("No numeric types to aggregate") from err
 259 
 260             if values.size == 0:
 261                 results.append(values.copy())
 262                 continue
 263 
 264             results.append(np.apply_along_axis(func, self.axis, values))
 265 
 266         return self._wrap_results(results, block_list, obj, exclude)

 268     @Substitution(name="ewm", func_name="mean")
 269     @Appender(_doc_template)
 270     def mean(self, *args, **kwargs):
 271         """
 272         Exponential weighted moving average.
 273 
 274         Parameters
 275         ----------
 276         *args, **kwargs
 277             Arguments and keyword arguments to be passed into func.
 278         """
 279         nv.validate_window_func("mean", args, kwargs)
 280         window_func = self._get_roll_func("ewma")
 281         window_func = partial(
 282             window_func,
 283             com=self.com,
 284             adjust=self.adjust,
 285             ignore_na=self.ignore_na,
 286             minp=self.min_periods,
 287         )
 288         return self._apply(window_func)

 302     @Substitution(name="ewm", func_name="var")
 303     @Appender(_doc_template)
 304     @Appender(_bias_template)
 305     def var(self, bias: bool = False, *args, **kwargs):
 306         """
 307         Exponential weighted moving variance.
 308         """
 309         nv.validate_window_func("var", args, kwargs)
 310 
 311         def f(arg):
 312             return window_aggregations.ewmcov(
 313                 arg, arg, self.com, self.adjust, self.ignore_na, self.min_periods, bias,
 314             )
 315 
 316         return self._apply(f)

 318     @Substitution(name="ewm", func_name="cov")
 319     @Appender(_doc_template)
 320     def cov(
 321         self,
 322         other: Optional[Union[np.ndarray, FrameOrSeries]] = None,
 323         pairwise: Optional[bool] = None,
 324         bias: bool = False,
 325         **kwargs,
 326     ):
 327         """
 328         Exponential weighted sample covariance.
 329 
 330         Parameters
 331         ----------
 332         other : Series, DataFrame, or ndarray, optional
 333             If not supplied then will default to self and produce pairwise
 334             output.
 335         pairwise : bool, default None
 336             If False then only matching columns between self and other will be
 337             used and the output will be a DataFrame.
 338             If True then all pairwise combinations will be calculated and the
 339             output will be a MultiIndex DataFrame in the case of DataFrame
 340             inputs. In the case of missing elements, only complete pairwise
 341             observations will be used.
 342         bias : bool, default False
 343             Use a standard estimation bias correction.
 344         **kwargs
 345            Keyword arguments to be passed into func.
 346         """
 347         if other is None:
 348             other = self._selected_obj
 349             # only default unset
 350             pairwise = True if pairwise is None else pairwise
 351         other = self._shallow_copy(other)
 352 
 353         def _get_cov(X, Y):
 354             X = self._shallow_copy(X)
 355             Y = self._shallow_copy(Y)
 356             cov = window_aggregations.ewmcov(
 357                 X._prep_values(),
 358                 Y._prep_values(),
 359                 self.com,
 360                 self.adjust,
 361                 self.ignore_na,
 362                 self.min_periods,
 363                 bias,
 364             )
 365             return X._wrap_result(cov)
 366 
 367         return _flex_binary_moment(
 368             self._selected_obj, other._selected_obj, _get_cov, pairwise=bool(pairwise)
 369         )

 371     @Substitution(name="ewm", func_name="corr")
 372     @Appender(_doc_template)
 373     def corr(
 374         self,
 375         other: Optional[Union[np.ndarray, FrameOrSeries]] = None,
 376         pairwise: Optional[bool] = None,
 377         **kwargs,
 378     ):
 379         """
 380         Exponential weighted sample correlation.
 381 
 382         Parameters
 383         ----------
 384         other : Series, DataFrame, or ndarray, optional
 385             If not supplied then will default to self and produce pairwise
 386             output.
 387         pairwise : bool, default None
 388             If False then only matching columns between self and other will be
 389             used and the output will be a DataFrame.
 390             If True then all pairwise combinations will be calculated and the
 391             output will be a MultiIndex DataFrame in the case of DataFrame
 392             inputs. In the case of missing elements, only complete pairwise
 393             observations will be used.
 394         **kwargs
 395            Keyword arguments to be passed into func.
 396         """
 397         if other is None:
 398             other = self._selected_obj
 399             # only default unset
 400             pairwise = True if pairwise is None else pairwise
 401         other = self._shallow_copy(other)
 402 
 403         def _get_corr(X, Y):
 404             X = self._shallow_copy(X)
 405             Y = self._shallow_copy(Y)
 406 
 407             def _cov(x, y):
 408                 return window_aggregations.ewmcov(
 409                     x, y, self.com, self.adjust, self.ignore_na, self.min_periods, 1,
 410                 )
 411 
 412             x_values = X._prep_values()
 413             y_values = Y._prep_values()
 414             with np.errstate(all="ignore"):
 415                 cov = _cov(x_values, y_values)
 416                 x_var = _cov(x_values, x_values)
 417                 y_var = _cov(y_values, y_values)
 418                 corr = cov / zsqrt(x_var * y_var)
 419             return X._wrap_result(corr)
 420 
 421         return _flex_binary_moment(
 422             self._selected_obj, other._selected_obj, _get_corr, pairwise=bool(pairwise)
 423         )
