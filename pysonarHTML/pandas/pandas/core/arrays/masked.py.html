  44 class BaseMaskedArray(ExtensionArray, ExtensionOpsMixin):
  45     """
  46     Base class for masked arrays (which use _data and _mask to store the data).
  47 
  48     numpy based
  49     """
  50 
  51     # The value used to fill '_data' to avoid upcasting
  52     _internal_fill_value: Scalar
  53 
  54     def __init__(self, values: np.ndarray, mask: np.ndarray, copy: bool = False):
  55         # values is supposed to already be validated in the subclass
  56         if not (isinstance(mask, np.ndarray) and mask.dtype == np.bool_):
  57             raise TypeError(
  58                 "mask should be boolean numpy array. Use "
  59                 "the 'pd.array' function instead"
  60             )
  61         if not values.ndim == 1:
  62             raise ValueError("values must be a 1D array")
  63         if not mask.ndim == 1:
  64             raise ValueError("mask must be a 1D array")
  65 
  66         if copy:
  67             values = values.copy()
  68             mask = mask.copy()
  69 
  70         self._data = values
  71         self._mask = mask
  72 
  73     @property
  74     def dtype(self) -> BaseMaskedDtype:
  75         raise AbstractMethodError(self)
  76 
  77     def __getitem__(self, item):
  78         if is_integer(item):
  79             if self._mask[item]:
  80                 return self.dtype.na_value
  81             return self._data[item]
  82 
  83         item = check_array_indexer(self, item)
  84 
  85         return type(self)(self._data[item], self._mask[item])
  86 
  87     def _coerce_to_array(self, values) -> Tuple[np.ndarray, np.ndarray]:
  88         raise AbstractMethodError(self)
  89 
  90     def __setitem__(self, key, value) -> None:
  91         _is_scalar = is_scalar(value)
  92         if _is_scalar:
  93             value = [value]
  94         value, mask = self._coerce_to_array(value)
  95 
  96         if _is_scalar:
  97             value = value[0]
  98             mask = mask[0]
  99 
 100         key = check_array_indexer(self, key)
 101         self._data[key] = value
 102         self._mask[key] = mask
 103 
 104     def __iter__(self):
 105         for i in range(len(self)):
 106             if self._mask[i]:
 107                 yield self.dtype.na_value
 108             else:
 109                 yield self._data[i]
 110 
 111     def __len__(self) -> int:
 112         return len(self._data)

 114     def __invert__(self: BaseMaskedArrayT) -> BaseMaskedArrayT:
 115         return type(self)(~self._data, self._mask)

 117     def to_numpy(
 118         self, dtype=None, copy: bool = False, na_value: Scalar = lib.no_default,
 119     ) -> np.ndarray:
 120         """
 121         Convert to a NumPy Array.
 122 
 123         By default converts to an object-dtype NumPy array. Specify the `dtype` and
 124         `na_value` keywords to customize the conversion.
 125 
 126         Parameters
 127         ----------
 128         dtype : dtype, default object
 129             The numpy dtype to convert to.
 130         copy : bool, default False
 131             Whether to ensure that the returned value is a not a view on
 132             the array. Note that ``copy=False`` does not *ensure* that
 133             ``to_numpy()`` is no-copy. Rather, ``copy=True`` ensure that
 134             a copy is made, even if not strictly necessary. This is typically
 135             only possible when no missing values are present and `dtype`
 136             is the equivalent numpy dtype.
 137         na_value : scalar, optional
 138              Scalar missing value indicator to use in numpy array. Defaults
 139              to the native missing value indicator of this array (pd.NA).
 140 
 141         Returns
 142         -------
 143         numpy.ndarray
 144 
 145         Examples
 146         --------
 147         An object-dtype is the default result
 148 
 149         >>> a = pd.array([True, False, pd.NA], dtype="boolean")
 150         >>> a.to_numpy()
 151         array([True, False, <NA>], dtype=object)
 152 
 153         When no missing values are present, an equivalent dtype can be used.
 154 
 155         >>> pd.array([True, False], dtype="boolean").to_numpy(dtype="bool")
 156         array([ True, False])
 157         >>> pd.array([1, 2], dtype="Int64").to_numpy("int64")
 158         array([1, 2])
 159 
 160         However, requesting such dtype will raise a ValueError if
 161         missing values are present and the default missing value :attr:`NA`
 162         is used.
 163 
 164         >>> a = pd.array([True, False, pd.NA], dtype="boolean")
 165         >>> a
 166         <BooleanArray>
 167         [True, False, <NA>]
 168         Length: 3, dtype: boolean
 169 
 170         >>> a.to_numpy(dtype="bool")
 171         Traceback (most recent call last):
 172         ...
 173         ValueError: cannot convert to bool numpy array in presence of missing values
 174 
 175         Specify a valid `na_value` instead
 176 
 177         >>> a.to_numpy(dtype="bool", na_value=False)
 178         array([ True, False, False])
 179         """
 180         if na_value is lib.no_default:
 181             na_value = libmissing.NA
 182         if dtype is None:
 183             dtype = object
 184         if self._hasna:
 185             if (
 186                 not (is_object_dtype(dtype) or is_string_dtype(dtype))
 187                 and na_value is libmissing.NA
 188             ):
 189                 raise ValueError(
 190                     f"cannot convert to '{dtype}'-dtype NumPy array "
 191                     "with missing values. Specify an appropriate 'na_value' "
 192                     "for this dtype."
 193                 )
 194             # don't pass copy to astype -> always need a copy since we are mutating
 195             data = self._data.astype(dtype)
 196             data[self._mask] = na_value
 197         else:
 198             data = self._data.astype(dtype, copy=copy)
 199         return data

 204         """
 205         the array interface, return my values
 206         We return an object array here to preserve our scalar values

 242     def take(
 243         self: BaseMaskedArrayT,
 244         indexer,
 245         allow_fill: bool = False,
 246         fill_value: Optional[Scalar] = None,
 247     ) -> BaseMaskedArrayT:
 248         # we always fill with 1 internally
 249         # to avoid upcasting
 250         data_fill_value = self._internal_fill_value if isna(fill_value) else fill_value
 251         result = take(
 252             self._data, indexer, fill_value=data_fill_value, allow_fill=allow_fill
 253         )
 254 
 255         mask = take(self._mask, indexer, fill_value=True, allow_fill=allow_fill)
 256 
 257         # if we are filling
 258         # we only fill where the indexer is null
 259         # not existing missing values
 260         # TODO(jreback) what if we have a non-na float as a fill value?
 261         if allow_fill and notna(fill_value):
 262             fill_mask = np.asarray(indexer) == -1
 263             result[fill_mask] = fill_value
 264             mask = mask ^ fill_mask
 265 
 266         return type(self)(result, mask, copy=False)
 267 
 268     def copy(self: BaseMaskedArrayT) -> BaseMaskedArrayT:
 269         data, mask = self._data, self._mask
 270         data = data.copy()
 271         mask = mask.copy()
 272         return type(self)(data, mask, copy=False)

 274     @doc(ExtensionArray.factorize)
 275     def factorize(self, na_sentinel: int = -1) -> Tuple[np.ndarray, ExtensionArray]:
 276         arr = self._data
 277         mask = self._mask
 278 
 279         codes, uniques = _factorize_array(arr, na_sentinel=na_sentinel, mask=mask)
 280 
 281         # the hashtables don't handle all different types of bits
 282         uniques = uniques.astype(self.dtype.numpy_dtype, copy=False)
 283         uniques = type(self)(uniques, np.zeros(len(uniques), dtype=bool))
 284         return codes, uniques

 286     def value_counts(self, dropna: bool = True) -> "Series":
 287         """
 288         Returns a Series containing counts of each unique value.
 289 
 290         Parameters
 291         ----------
 292         dropna : bool, default True
 293             Don't include counts of missing values.
 294 
 295         Returns
 296         -------
 297         counts : Series
 298 
 299         See Also
 300         --------
 301         Series.value_counts
 302         """
 303         from pandas import Index, Series
 304         from pandas.arrays import IntegerArray
 305 
 306         # compute counts on the data with no nans
 307         data = self._data[~self._mask]
 308         value_counts = Index(data).value_counts()
 309 
 310         # TODO(extension)
 311         # if we have allow Index to hold an ExtensionArray
 312         # this is easier
 313         index = value_counts.index._values.astype(object)
 314 
 315         # if we want nans, count the mask
 316         if dropna:
 317             counts = value_counts._values
 318         else:
 319             counts = np.empty(len(value_counts) + 1, dtype="int64")
 320             counts[:-1] = value_counts
 321             counts[-1] = self._mask.sum()
 322 
 323             index = Index(
 324                 np.concatenate([index, np.array([self.dtype.na_value], dtype=object)]),
 325                 dtype=object,
 326             )
 327 
 328         mask = np.zeros(len(counts), dtype="bool")
 329         counts = IntegerArray(counts, mask)
 330 
 331         return Series(counts, index=index)

 333     def _reduce(self, name: str, skipna: bool = True, **kwargs):
 334         data = self._data
 335         mask = self._mask
 336 
 337         if name in {"sum", "prod", "min", "max"}:
 338             op = getattr(masked_reductions, name)
 339             return op(data, mask, skipna=skipna, **kwargs)
 340 
 341         # coerce to a nan-aware float if needed
 342         # (we explicitly use NaN within reductions)
 343         if self._hasna:
 344             data = self.to_numpy("float64", na_value=np.nan)
 345 
 346         op = getattr(nanops, "nan" + name)
 347         result = op(data, axis=0, skipna=skipna, mask=mask, **kwargs)
 348 
 349         if np.isnan(result):
 350             return libmissing.NA
 351 
 352         return result
