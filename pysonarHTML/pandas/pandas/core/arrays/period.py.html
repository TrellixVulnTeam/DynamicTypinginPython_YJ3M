  64 class PeriodArray(dtl.DatetimeLikeArrayMixin, dtl.DatelikeOps):
  65     """
  66     Pandas ExtensionArray for storing Period data.
  67 
  68     Users should use :func:`period_array` to create new instances.
  69 
  70     Parameters
  71     ----------
  72     values : Union[PeriodArray, Series[period], ndarray[int], PeriodIndex]
  73         The data to store. These should be arrays that can be directly
  74         converted to ordinals without inference or copy (PeriodArray,
  75         ndarray[int64]), or a box around such an array (Series[period],
  76         PeriodIndex).
  77     freq : str or DateOffset
  78         The `freq` to use for the array. Mostly applicable when `values`
  79         is an ndarray of integers, when `freq` is required. When `values`
  80         is a PeriodArray (or box around), it's checked that ``values.freq``
  81         matches `freq`.
  82     dtype : PeriodDtype, optional
  83         A PeriodDtype instance from which to extract a `freq`. If both
  84         `freq` and `dtype` are specified, then the frequencies must match.
  85     copy : bool, default False
  86         Whether to copy the ordinals before storing.
  87 
  88     Attributes
  89     ----------
  90     None
  91 
  92     Methods
  93     -------
  94     None
  95 
  96     See Also
  97     --------
  98     period_array : Create a new PeriodArray.
  99     PeriodIndex : Immutable Index for period data.
 100 
 101     Notes
 102     -----
 103     There are two components to a PeriodArray
 104 
 105     - ordinals : integer ndarray
 106     - freq : pd.tseries.offsets.Offset
 107 
 108     The values are physically stored as a 1-D ndarray of integers. These are
 109     called "ordinals" and represent some kind of offset from a base.
 110 
 111     The `freq` indicates the span covered by each element of the array.
 112     All elements in the PeriodArray have the same `freq`.
 113     """

 149     def __init__(self, values, freq=None, dtype=None, copy=False):
 150         freq = validate_dtype_freq(dtype, freq)
 151 
 152         if freq is not None:
 153             freq = Period._maybe_convert_freq(freq)
 154 
 155         if isinstance(values, ABCSeries):
 156             values = values._values
 157             if not isinstance(values, type(self)):
 158                 raise TypeError("Incorrect dtype")
 159 
 160         elif isinstance(values, ABCPeriodIndex):
 161             values = values._values
 162 
 163         if isinstance(values, type(self)):
 164             if freq is not None and freq != values.freq:
 165                 raise raise_on_incompatible(values, freq)
 166             values, freq = values._data, values.freq
 167 
 168         values = np.array(values, dtype="int64", copy=copy)
 169         self._data = values
 170         if freq is None:
 171             raise ValueError("freq is not specified and cannot be inferred")
 172         self._dtype = PeriodDtype(freq)

 182     def _from_sequence(
 183         cls: Type["PeriodArray"],
 184         scalars: Union[Sequence[Optional[Period]], AnyArrayLike],
 185         dtype: Optional[PeriodDtype] = None,
 186         copy: bool = False,
 187     ) -> "PeriodArray":
 188         if dtype:
 189             freq = dtype.freq
 190         else:
 191             freq = None
 192 
 193         if isinstance(scalars, cls):
 194             validate_dtype_freq(scalars.dtype, freq)
 195             if copy:
 196                 scalars = scalars.copy()
 197             return scalars
 198 
 199         periods = np.asarray(scalars, dtype=object)
 200         if copy:
 201             periods = periods.copy()
 202 
 203         freq = freq or libperiod.extract_freq(periods)
 204         ordinals = libperiod.extract_ordinals(periods, freq)
 205         return cls(ordinals, freq=freq)

 297     def __arrow_array__(self, type=None):
 298         """
 299         Convert myself into a pyarrow Array.
 300         """
 301         import pyarrow
 302         from pandas.core.arrays._arrow_utils import ArrowPeriodType
 303 
 304         if type is not None:
 305             if pyarrow.types.is_integer(type):
 306                 return pyarrow.array(self._data, mask=self.isna(), type=type)
 307             elif isinstance(type, ArrowPeriodType):
 308                 # ensure we have the same freq
 309                 if self.freqstr != type.freq:
 310                     raise TypeError(
 311                         "Not supported to convert PeriodArray to array with different "
 312                         f"'freq' ({self.freqstr} vs {type.freq})"

 315                 raise TypeError(
 316                     f"Not supported to convert PeriodArray to '{type}' type"
 317                 )

 319         period_type = ArrowPeriodType(self.freqstr)
 320         storage_array = pyarrow.array(self._data, mask=self.isna(), type="int64")
 321         return pyarrow.ExtensionArray.from_storage(period_type, storage_array)

 412     def to_timestamp(self, freq=None, how="start"):
 413         """
 414         Cast to DatetimeArray/Index.
 415 
 416         Parameters
 417         ----------
 418         freq : str or DateOffset, optional
 419             Target frequency. The default is 'D' for week or longer,
 420             'S' otherwise.
 421         how : {'s', 'e', 'start', 'end'}
 422             Whether to use the start or end of the time period being converted.
 423 
 424         Returns
 425         -------
 426         DatetimeArray/Index
 427         """
 428         from pandas.core.arrays import DatetimeArray
 429 
 430         how = libperiod.validate_end_alias(how)
 431 
 432         end = how == "E"
 433         if end:
 434             if freq == "B" or self.freq == "B":
 435                 # roll forward to ensure we land on B date
 436                 adjust = Timedelta(1, "D") - Timedelta(1, "ns")
 437                 return self.to_timestamp(how="start") + adjust
 438             else:
 439                 adjust = Timedelta(1, "ns")
 440                 return (self + self.freq).to_timestamp(how="start") - adjust
 441 
 442         if freq is None:
 443             base = self.freq._period_dtype_code
 444             freq = libfrequencies.get_to_timestamp_base(base)
 445             base = freq
 446         else:
 447             freq = Period._maybe_convert_freq(freq)
 448             base = freq._period_dtype_code
 449 
 450         new_data = self.asfreq(freq, how=how)
 451 
 452         new_data = libperiod.periodarr_to_dt64arr(new_data.asi8, base)
 453         return DatetimeArray(new_data)._with_freq("infer")

 457     def _time_shift(self, periods, freq=None):
 458         """
 459         Shift each value by `periods`.
 460 
 461         Note this is different from ExtensionArray.shift, which
 462         shifts the *position* of each element, padding the end with
 463         missing values.
 464 
 465         Parameters
 466         ----------
 467         periods : int
 468             Number of periods to shift by.
 469         freq : pandas.DateOffset, pandas.Timedelta, or str
 470             Frequency increment to shift by.
 471         """
 472         if freq is not None:
 473             raise TypeError(
 474                 "`freq` argument is not supported for "
 475                 f"{type(self).__name__}._time_shift"
 476             )
 477         values = self.asi8 + periods * self.freq.n
 478         if self._hasnans:
 479             values[self._isnan] = iNaT
 480         return type(self)(values, freq=self.freq)

 486     def asfreq(self, freq=None, how: str = "E") -> "PeriodArray":
 487         """
 488         Convert the Period Array/Index to the specified frequency `freq`.
 489 
 490         Parameters
 491         ----------
 492         freq : str
 493             A frequency.
 494         how : str {'E', 'S'}
 495             Whether the elements should be aligned to the end
 496             or start within pa period.
 497 
 498             * 'E', 'END', or 'FINISH' for end,
 499             * 'S', 'START', or 'BEGIN' for start.
 500 
 501             January 31st ('END') vs. January 1st ('START') for example.
 502 
 503         Returns
 504         -------
 505         Period Array/Index
 506             Constructed with the new frequency.
 507 
 508         Examples
 509         --------
 510         >>> pidx = pd.period_range('2010-01-01', '2015-01-01', freq='A')
 511         >>> pidx
 512         PeriodIndex(['2010', '2011', '2012', '2013', '2014', '2015'],
 513         dtype='period[A-DEC]', freq='A-DEC')
 514 
 515         >>> pidx.asfreq('M')
 516         PeriodIndex(['2010-12', '2011-12', '2012-12', '2013-12', '2014-12',
 517         '2015-12'], dtype='period[M]', freq='M')
 518 
 519         >>> pidx.asfreq('M', how='S')
 520         PeriodIndex(['2010-01', '2011-01', '2012-01', '2013-01', '2014-01',
 521         '2015-01'], dtype='period[M]', freq='M')
 522         """
 523         how = libperiod.validate_end_alias(how)
 524 
 525         freq = Period._maybe_convert_freq(freq)
 526 
 527         base1 = self.freq._period_dtype_code
 528         base2 = freq._period_dtype_code
 529 
 530         asi8 = self.asi8
 531         # self.freq.n can't be negative or 0
 532         end = how == "E"
 533         if end:
 534             ordinal = asi8 + self.freq.n - 1
 535         else:
 536             ordinal = asi8
 537 
 538         new_data = period_asfreq_arr(ordinal, base1, base2, end)
 539 
 540         if self._hasnans:
 541             new_data[self._isnan] = iNaT
 542 
 543         return type(self)(new_data, freq=freq)

 553     def _format_native_types(self, na_rep="NaT", date_format=None, **kwargs):
 554         """
 555         actually format my specific types
 556         """
 557         values = self.astype(object)
 558 
 559         if date_format:
 560             formatter = lambda dt: dt.strftime(date_format)
 561         else:
 562             formatter = lambda dt: str(dt)
 563 
 564         if self._hasnans:
 565             mask = self._isnan
 566             values[mask] = na_rep
 567             imask = ~mask
 568             values[imask] = np.array([formatter(dt) for dt in values[imask]])
 569         else:
 570             values = np.array([formatter(dt) for dt in values])

 591     def _sub_period(self, other):
 592         # If the operation is well-defined, we return an object-Index
 593         # of DateOffsets.  Null entries are filled with pd.NaT
 594         self._check_compatible_with(other)
 595         asi8 = self.asi8
 596         new_data = asi8 - other.ordinal
 597         new_data = np.array([self.freq * x for x in new_data])
 598 
 599         if self._hasnans:
 600             new_data[self._isnan] = NaT
 601 
 602         return new_data

 604     def _sub_period_array(self, other):
 605         """
 606         Subtract a Period Array/Index from self.  This is only valid if self
 607         is itself a Period Array/Index, raises otherwise.  Both objects must
 608         have the same frequency.
 609 
 610         Parameters
 611         ----------
 612         other : PeriodIndex or PeriodArray
 613 
 614         Returns
 615         -------
 616         result : np.ndarray[object]
 617             Array of DateOffset objects; nulls represented by NaT.
 618         """
 619         if self.freq != other.freq:
 620             msg = DIFFERENT_FREQ.format(
 621                 cls=type(self).__name__, own_freq=self.freqstr, other_freq=other.freqstr
 622             )
 623             raise IncompatibleFrequency(msg)
 624 
 625         new_values = algos.checked_add_with_arr(
 626             self.asi8, -other.asi8, arr_mask=self._isnan, b_mask=other._isnan
 627         )
 628 
 629         new_values = np.array([self.freq.base * x for x in new_values])
 630         if self._hasnans or other._hasnans:
 631             mask = (self._isnan) | (other._isnan)
 632             new_values[mask] = NaT
 633         return new_values

 635     def _addsub_int_array(
 636         self, other: np.ndarray, op: Callable[[Any, Any], Any],
 637     ) -> "PeriodArray":
 638         """
 639         Add or subtract array of integers; equivalent to applying
 640         `_time_shift` pointwise.
 641 
 642         Parameters
 643         ----------
 644         other : np.ndarray[integer-dtype]
 645         op : {operator.add, operator.sub}
 646 
 647         Returns
 648         -------
 649         result : PeriodArray
 650         """
 651         assert op in [operator.add, operator.sub]
 652         if op is operator.sub:
 653             other = -other
 654         res_values = algos.checked_add_with_arr(self.asi8, other, arr_mask=self._isnan)
 655         res_values = res_values.view("i8")
 656         res_values[self._isnan] = iNaT
 657         return type(self)(res_values, freq=self.freq)

 659     def _add_offset(self, other: DateOffset):
 660         assert not isinstance(other, Tick)
 661 
 662         if other.base != self.freq.base:
 663             raise raise_on_incompatible(self, other)
 664 
 665         # Note: when calling parent class's _add_timedeltalike_scalar,
 666         #  it will call delta_to_nanoseconds(delta).  Because delta here
 667         #  is an integer, delta_to_nanoseconds will return it unchanged.
 668         result = super()._add_timedeltalike_scalar(other.n)
 669         return type(self)(result, freq=self.freq)

 671     def _add_timedeltalike_scalar(self, other):
 672         """
 673         Parameters
 674         ----------
 675         other : timedelta, Tick, np.timedelta64
 676 
 677         Returns
 678         -------
 679         PeriodArray
 680         """
 681         if not isinstance(self.freq, Tick):
 682             # We cannot add timedelta-like to non-tick PeriodArray
 683             raise raise_on_incompatible(self, other)
 684 
 685         if notna(other):
 686             # special handling for np.timedelta64("NaT"), avoid calling
 687             #  _check_timedeltalike_freq_compat as that would raise TypeError
 688             other = self._check_timedeltalike_freq_compat(other)
 689 
 690         # Note: when calling parent class's _add_timedeltalike_scalar,
 691         #  it will call delta_to_nanoseconds(delta).  Because delta here
 692         #  is an integer, delta_to_nanoseconds will return it unchanged.
 693         return super()._add_timedeltalike_scalar(other)
 694 
 695     def _add_timedelta_arraylike(self, other):
 696         """
 697         Parameters
 698         ----------
 699         other : TimedeltaArray or ndarray[timedelta64]
 700 
 701         Returns
 702         -------
 703         result : ndarray[int64]
 704         """
 705         if not isinstance(self.freq, Tick):
 706             # We cannot add timedelta-like to non-tick PeriodArray
 707             raise TypeError(
 708                 f"Cannot add or subtract timedelta64[ns] dtype from {self.dtype}"
 709             )
 710 
 711         if not np.all(isna(other)):
 712             delta = self._check_timedeltalike_freq_compat(other)
 713         else:
 714             # all-NaT TimedeltaIndex is equivalent to a single scalar td64 NaT
 715             return self + np.timedelta64("NaT")
 716 
 717         ordinals = self._addsub_int_array(delta, operator.add).asi8
 718         return type(self)(ordinals, dtype=self.dtype)
 719 
 720     def _check_timedeltalike_freq_compat(self, other):
 721         """
 722         Arithmetic operations with timedelta-like scalars or array `other`
 723         are only valid if `other` is an integer multiple of `self.freq`.
 724         If the operation is valid, find that integer multiple.  Otherwise,
 725         raise because the operation is invalid.
 726 
 727         Parameters
 728         ----------
 729         other : timedelta, np.timedelta64, Tick,
 730                 ndarray[timedelta64], TimedeltaArray, TimedeltaIndex
 731 
 732         Returns
 733         -------
 734         multiple : int or ndarray[int64]
 735 
 736         Raises
 737         ------
 738         IncompatibleFrequency
 739         """
 740         assert isinstance(self.freq, Tick)  # checked by calling function
 741         base_nanos = self.freq.base.nanos
 742 
 743         if isinstance(other, (timedelta, np.timedelta64, Tick)):
 744             nanos = delta_to_nanoseconds(other)
 745 
 746         elif isinstance(other, np.ndarray):
 747             # numpy timedelta64 array; all entries must be compatible
 748             assert other.dtype.kind == "m"
 749             if other.dtype != TD64NS_DTYPE:
 750                 # i.e. non-nano unit
 751                 # TODO: disallow unit-less timedelta64
 752                 other = other.astype(TD64NS_DTYPE)
 753             nanos = other.view("i8")
 754         else:
 755             # TimedeltaArray/Index
 756             nanos = other.asi8
 757 
 758         if np.all(nanos % base_nanos == 0):
 759             # nanos being added is an integer multiple of the
 760             #  base-frequency to self.freq
 761             delta = nanos // base_nanos
 762             # delta is the integer (or integer-array) number of periods
 763             # by which will be added to self.
 764             return delta
 765 
 766         raise raise_on_incompatible(self, other)
 767

 769 def raise_on_incompatible(left, right):
 770     """
 771     Helper function to render a consistent error message when raising
 772     IncompatibleFrequency.
 773 
 774     Parameters
 775     ----------
 776     left : PeriodArray
 777     right : None, DateOffset, Period, ndarray, or timedelta-like
 778 
 779     Returns
 780     -------
 781     IncompatibleFrequency
 782         Exception to be raised by the caller.
 783     """
 784     # GH#24283 error message format depends on whether right is scalar
 785     if isinstance(right, (np.ndarray, ABCTimedeltaArray)) or right is None:
 786         other_freq = None
 787     elif isinstance(right, (ABCPeriodIndex, PeriodArray, Period, DateOffset)):
 788         other_freq = right.freqstr
 789     else:
 790         other_freq = delta_to_tick(Timedelta(right)).freqstr
 791 
 792     msg = DIFFERENT_FREQ.format(
 793         cls=type(left).__name__, own_freq=left.freqstr, other_freq=other_freq
 794     )
 795     return IncompatibleFrequency(msg)

 802 def period_array(
 803     data: Union[Sequence[Optional[Period]], AnyArrayLike],
 804     freq: Optional[Union[str, Tick]] = None,
 805     copy: bool = False,
 806 ) -> PeriodArray:
 807     """
 808     Construct a new PeriodArray from a sequence of Period scalars.
 809 
 810     Parameters
 811     ----------
 812     data : Sequence of Period objects
 813         A sequence of Period objects. These are required to all have
 814         the same ``freq.`` Missing values can be indicated by ``None``
 815         or ``pandas.NaT``.
 816     freq : str, Tick, or Offset
 817         The frequency of every element of the array. This can be specified
 818         to avoid inferring the `freq` from `data`.
 819     copy : bool, default False
 820         Whether to ensure a copy of the data is made.
 821 
 822     Returns
 823     -------
 824     PeriodArray
 825 
 826     See Also
 827     --------
 828     PeriodArray
 829     pandas.PeriodIndex
 830 
 831     Examples
 832     --------
 833     >>> period_array([pd.Period('2017', freq='A'),
 834     ...               pd.Period('2018', freq='A')])
 835     <PeriodArray>
 836     ['2017', '2018']
 837     Length: 2, dtype: period[A-DEC]
 838 
 839     >>> period_array([pd.Period('2017', freq='A'),
 840     ...               pd.Period('2018', freq='A'),
 841     ...               pd.NaT])
 842     <PeriodArray>
 843     ['2017', '2018', 'NaT']
 844     Length: 3, dtype: period[A-DEC]
 845 
 846     Integers that look like years are handled
 847 
 848     >>> period_array([2000, 2001, 2002], freq='D')
 849     <PeriodArray>
 850     ['2000-01-01', '2001-01-01', '2002-01-01']
 851     Length: 3, dtype: period[D]
 852 
 853     Datetime-like strings may also be passed
 854 
 855     >>> period_array(['2000-Q1', '2000-Q2', '2000-Q3', '2000-Q4'], freq='Q')
 856     <PeriodArray>
 857     ['2000Q1', '2000Q2', '2000Q3', '2000Q4']
 858     Length: 4, dtype: period[Q-DEC]
 859     """
 860     data_dtype = getattr(data, "dtype", None)
 861 
 862     if is_datetime64_dtype(data_dtype):
 863         return PeriodArray._from_datetime64(data, freq)
 864     if is_period_dtype(data_dtype):
 865         return PeriodArray(data, freq)
 866 
 867     # other iterable of some kind
 868     if not isinstance(data, (np.ndarray, list, tuple, ABCSeries)):
 869         data = list(data)
 870 
 871     data = np.asarray(data)
 872 
 873     dtype: Optional[PeriodDtype]
 874     if freq:
 875         dtype = PeriodDtype(freq)
 876     else:
 877         dtype = None
 878 
 879     if is_float_dtype(data) and len(data) > 0:
 880         raise TypeError("PeriodIndex does not allow floating point in construction")
 881 
 882     data = ensure_object(data)
 883 
 884     return PeriodArray._from_sequence(data, dtype=dtype)

 887 def validate_dtype_freq(dtype, freq):
 888     """
 889     If both a dtype and a freq are available, ensure they match.  If only
 890     dtype is available, extract the implied freq.
 891 
 892     Parameters
 893     ----------
 894     dtype : dtype
 895     freq : DateOffset or None
 896 
 897     Returns
 898     -------
 899     freq : DateOffset
 900 
 901     Raises
 902     ------
 903     ValueError : non-period dtype
 904     IncompatibleFrequency : mismatch between dtype and freq
 905     """
 906     if freq is not None:
 907         freq = to_offset(freq)
 908 
 909     if dtype is not None:
 910         dtype = pandas_dtype(dtype)
 911         if not is_period_dtype(dtype):
 912             raise ValueError("dtype must be PeriodDtype")
 913         if freq is None:
 914             freq = dtype.freq
 915         elif freq != dtype.freq:
 916             raise IncompatibleFrequency("specified freq and dtype are different")
 917     return freq

 920 def dt64arr_to_periodarr(data, freq, tz=None):
 921     """
 922     Convert an datetime-like array to values Period ordinals.
 923 
 924     Parameters
 925     ----------
 926     data : Union[Series[datetime64[ns]], DatetimeIndex, ndarray[datetime64ns]]
 927     freq : Optional[Union[str, Tick]]
 928         Must match the `freq` on the `data` if `data` is a DatetimeIndex
 929         or Series.
 930     tz : Optional[tzinfo]
 931 
 932     Returns
 933     -------
 934     ordinals : ndarray[int]
 935     freq : Tick
 936         The frequency extracted from the Series or DatetimeIndex if that's
 937         used.
 938 
 939     """
 940     if data.dtype != np.dtype("M8[ns]"):
 941         raise ValueError(f"Wrong dtype: {data.dtype}")
 942 
 943     if freq is None:
 944         if isinstance(data, ABCIndexClass):
 945             data, freq = data._values, data.freq
 946         elif isinstance(data, ABCSeries):
 947             data, freq = data._values, data.dt.freq
 948 
 949     freq = Period._maybe_convert_freq(freq)
 950 
 951     if isinstance(data, (ABCIndexClass, ABCSeries)):
 952         data = data._values
 953 
 954     base = freq._period_dtype_code
 955     return libperiod.dt64arr_to_periodarr(data.view("i8"), base, tz), freq

 958 def _get_ordinal_range(start, end, periods, freq, mult=1):
 959     if com.count_not_none(start, end, periods) != 2:
 960         raise ValueError(
 961             "Of the three parameters: start, end, and periods, "
 962             "exactly two must be specified"
 963         )
 964 
 965     if freq is not None:
 966         freq = to_offset(freq)
 967         mult = freq.n
 968 
 969     if start is not None:
 970         start = Period(start, freq)
 971     if end is not None:
 972         end = Period(end, freq)
 973 
 974     is_start_per = isinstance(start, Period)
 975     is_end_per = isinstance(end, Period)
 976 
 977     if is_start_per and is_end_per and start.freq != end.freq:
 978         raise ValueError("start and end must have same freq")
 979     if start is NaT or end is NaT:
 980         raise ValueError("start and end must not be NaT")
 981 
 982     if freq is None:
 983         if is_start_per:
 984             freq = start.freq
 985         elif is_end_per:
 986             freq = end.freq
 987         else:  # pragma: no cover
 988             raise ValueError("Could not infer freq from start/end")
 989 
 990     if periods is not None:
 991         periods = periods * mult
 992         if start is None:
 993             data = np.arange(
 994                 end.ordinal - periods + mult, end.ordinal + 1, mult, dtype=np.int64
 995             )
 996         else:
 997             data = np.arange(
 998                 start.ordinal, start.ordinal + periods, mult, dtype=np.int64
 999             )
1000     else:
1001         data = np.arange(start.ordinal, end.ordinal + 1, mult, dtype=np.int64)
1002 
1003     return data, freq

1052 def _make_field_arrays(*fields):
1053     length = None
1054     for x in fields:
1055         if isinstance(x, (list, np.ndarray, ABCSeries)):
1056             if length is not None and len(x) != length:
1057                 raise ValueError("Mismatched Period array lengths")
1058             elif length is None:
1059                 length = len(x)
1060 
1061     arrays = [
1062         np.asarray(x)
1063         if isinstance(x, (np.ndarray, list, ABCSeries))
1064         else np.repeat(x, length)
1065         for x in fields
1066     ]
1067 
1068     return arrays
