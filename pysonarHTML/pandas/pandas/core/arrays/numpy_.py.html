  29 class PandasDtype(ExtensionDtype):
  30     """
  31     A Pandas ExtensionDtype for NumPy dtypes.
  32 
  33     .. versionadded:: 0.24.0
  34 
  35     This is mostly for internal compatibility, and is not especially
  36     useful on its own.
  37 
  38     Parameters
  39     ----------
  40     dtype : object
  41         Object to be converted to a NumPy data type object.
  42 
  43     See Also
  44     --------
  45     numpy.dtype
  46     """
  47 
  48     _metadata = ("_dtype",)
  49 
  50     def __init__(self, dtype: object):
  51         self._dtype = np.dtype(dtype)
  52 
  53     def __repr__(self) -> str:
  54         return f"PandasDtype({repr(self.name)})"
  55 
  56     @property
  57     def numpy_dtype(self) -> np.dtype:
  58         """
  59         The NumPy dtype this PandasDtype wraps.
  60         """
  61         return self._dtype
  62 
  63     @property
  64     def name(self) -> str:
  65         """
  66         A bit-width name for this data-type.
  67         """
  68         return self._dtype.name
  69 
  70     @property
  71     def type(self) -> Type[np.generic]:
  72         """
  73         The type object used to instantiate a scalar of this NumPy data-type.
  74         """
  75         return self._dtype.type
  76 
  77     @property
  78     def _is_numeric(self) -> bool:
  79         # exclude object, str, unicode, void.
  80         return self.kind in set("biufc")
  81 
  82     @property
  83     def _is_boolean(self) -> bool:
  84         return self.kind == "b"
  85 
  86     @classmethod
  87     def construct_from_string(cls, string: str) -> "PandasDtype":
  88         try:
  89             dtype = np.dtype(string)
  90         except TypeError as err:
  91             if not isinstance(string, str):
  92                 msg = f"'construct_from_string' expects a string, got {type(string)}"
  93             else:
  94                 msg = f"Cannot construct a 'PandasDtype' from '{string}'"
  95             raise TypeError(msg) from err
  96         return cls(dtype)
  97 
  98     @classmethod
  99     def construct_array_type(cls) -> Type["PandasArray"]:
 100         """
 101         Return the array type associated with this dtype.
 102 
 103         Returns
 104         -------
 105         type
 106         """
 107         return PandasArray
 108 
 109     @property
 110     def kind(self) -> str:
 111         """
 112         A character code (one of 'biufcmMOSUV') identifying the general kind of data.
 113         """
 114         return self._dtype.kind
 115 
 116     @property
 117     def itemsize(self) -> int:
 118         """
 119         The element size of this data-type object.
 120         """
 121         return self._dtype.itemsize

 130     .. versionadded:: 0.24.0

 133     useful on its own.

 135     Parameters
 136     ----------
 137     values : ndarray
 138         The NumPy ndarray to wrap. Must be 1-dimensional.

 151     # If you're wondering why pd.Series(cls) doesn't put the array in an
 152     # ExtensionBlock, search for `ABCPandasArray`. We check for
 153     # that _typ to ensure that that users don't unnecessarily use EAs inside
 154     # pandas internals, which turns off things like block consolidation.
 155     _typ = "npy_extension"
 156     __array_priority__ = 1000
 157     _ndarray: np.ndarray

 159     # ------------------------------------------------------------------------

 162     def __init__(self, values: Union[np.ndarray, "PandasArray"], copy: bool = False):
 163         if isinstance(values, type(self)):
 164             values = values._ndarray
 165         if not isinstance(values, np.ndarray):
 166             raise ValueError(
 167                 f"'values' must be a NumPy array, not {type(values).__name__}"
 168             )
 169 
 170         if values.ndim != 1:
 171             raise ValueError("PandasArray must be 1-dimensional.")
 172 
 173         if copy:
 174             values = values.copy()
 175 
 176         self._ndarray = values
 177         self._dtype = PandasDtype(values.dtype)

 179     @classmethod
 180     def _from_sequence(cls, scalars, dtype=None, copy: bool = False) -> "PandasArray":
 181         if isinstance(dtype, PandasDtype):
 182             dtype = dtype._dtype
 183 
 184         result = np.asarray(scalars, dtype=dtype)
 185         if copy and result is scalars:
 186             result = result.copy()
 187         return cls(result)

 189     @classmethod
 190     def _from_factorized(cls, values, original) -> "PandasArray":
 191         return cls(values)

 195         return cls(np.concatenate(to_concat))

 213     _HANDLED_TYPES = (np.ndarray, numbers.Number)
 214 
 215     def __array_ufunc__(self, ufunc, method: str, *inputs, **kwargs):
 216         # Lightly modified version of
 217         # https://numpy.org/doc/stable/reference/generated/numpy.lib.mixins.NDArrayOperatorsMixin.html
 218         # The primary modification is not boxing scalar return values
 219         # in PandasArray, since pandas' ExtensionArrays are 1-d.
 220         out = kwargs.get("out", ())
 221         for x in inputs + out:
 222             # Only support operations with instances of _HANDLED_TYPES.
 223             # Use PandasArray instead of type(self) for isinstance to
 224             # allow subclasses that don't override __array_ufunc__ to
 225             # handle PandasArray objects.
 226             if not isinstance(x, self._HANDLED_TYPES + (PandasArray,)):
 227                 return NotImplemented
 228 
 229         # Defer to the implementation of the ufunc on unwrapped values.
 230         inputs = tuple(x._ndarray if isinstance(x, PandasArray) else x for x in inputs)
 231         if out:
 232             kwargs["out"] = tuple(
 233                 x._ndarray if isinstance(x, PandasArray) else x for x in out
 234             )
 235         result = getattr(ufunc, method)(*inputs, **kwargs)
 236 
 237         if type(result) is tuple and len(result):
 238             # multiple return values
 239             if not lib.is_scalar(result[0]):
 240                 # re-box array-like results
 241                 return tuple(type(self)(x) for x in result)
 242             else:
 243                 # but not scalar reductions
 244                 return result
 245         elif method == "at":
 246             # no return value
 247             return None
 248         else:
 249             # one return value
 250             if not lib.is_scalar(result):
 251                 # re-box array-like results, but not scalar reductions
 252                 result = type(self)(result)
 253             return result

 258     def __getitem__(self, item):
 259         if isinstance(item, type(self)):
 260             item = item._ndarray
 261 
 262         item = check_array_indexer(self, item)
 263 
 264         result = self._ndarray[item]
 265         if not lib.is_scalar(item):
 266             result = type(self)(result)
 267         return result
 268 
 269     def __setitem__(self, key, value) -> None:
 270         value = extract_array(value, extract_numpy=True)
 271 
 272         key = check_array_indexer(self, key)
 273         scalar_value = lib.is_scalar(value)
 274 
 275         if not scalar_value:
 276             value = np.asarray(value, dtype=self._ndarray.dtype)
 277 
 278         self._ndarray[key] = value

 283     def fillna(
 284         self, value=None, method: Optional[str] = None, limit: Optional[int] = None,
 285     ) -> "PandasArray":
 286         # TODO(_values_for_fillna): remove this
 287         value, method = validate_fillna_kwargs(value, method)
 288 
 289         mask = self.isna()
 290 
 291         if is_array_like(value):
 292             if len(value) != len(self):
 293                 raise ValueError(
 294                     f"Length of 'value' does not match. Got ({len(value)}) "
 295                     f" expected {len(self)}"
 296                 )
 297             value = value[mask]
 298 
 299         if mask.any():
 300             if method is not None:
 301                 func = pad_1d if method == "pad" else backfill_1d
 302                 new_values = func(self._ndarray, limit=limit, mask=mask)
 303                 new_values = self._from_sequence(new_values, dtype=self.dtype)
 304             else:
 305                 # fill with value
 306                 new_values = self.copy()
 307                 new_values[mask] = value
 308         else:
 309             new_values = self.copy()
 310         return new_values

 414     def to_numpy(
 415         self, dtype=None, copy: bool = False, na_value=lib.no_default
 416     ) -> np.ndarray:
 417         result = np.asarray(self._ndarray, dtype=dtype)
 418 
 419         if (copy or na_value is not lib.no_default) and result is self._ndarray:
 420             result = result.copy()

 422         if na_value is not lib.no_default:
 423             result[self.isna()] = na_value

 437     @classmethod
 438     def _create_arithmetic_method(cls, op):
 439         def arithmetic_method(self, other):
 440             if isinstance(other, (ABCIndexClass, ABCSeries)):
 441                 return NotImplemented
 442 
 443             elif isinstance(other, cls):
 444                 other = other._ndarray
 445 
 446             with np.errstate(all="ignore"):
 447                 result = op(self._ndarray, other)
 448 
 449             if op is divmod:
 450                 a, b = result
 451                 return cls(a), cls(b)
 452 
 453             return cls(result)
 454 
 455         return compat.set_function_name(arithmetic_method, f"__{op.__name__}__", cls)
