  39 class _IntegerDtype(BaseMaskedDtype):
  40     """
  41     An ExtensionDtype to hold a single size & kind of integer dtype.
  42 
  43     These specific implementations are subclasses of the non-public
  44     _IntegerDtype. For example we have Int8Dtype to represent signed int 8s.
  45 
  46     The attributes name & type are set when these subclasses are created.
  47     """
  48 
  49     name: str
  50     base = None
  51     type: Type
  52 
  53     def __repr__(self) -> str:
  54         sign = "U" if self.is_unsigned_integer else ""
  55         return f"{sign}Int{8 * self.itemsize}Dtype()"
  56 
  57     @cache_readonly
  58     def is_signed_integer(self) -> bool:
  59         return self.kind == "i"
  60 
  61     @cache_readonly
  62     def is_unsigned_integer(self) -> bool:
  63         return self.kind == "u"
  64 
  65     @property
  66     def _is_numeric(self) -> bool:
  67         return True
  68 
  69     @cache_readonly
  70     def numpy_dtype(self) -> np.dtype:
  71         """ Return an instance of our numpy dtype """
  72         return np.dtype(self.type)
  73 
  74     @cache_readonly
  75     def kind(self) -> str:
  76         return self.numpy_dtype.kind
  77 
  78     @cache_readonly
  79     def itemsize(self) -> int:
  80         """ Return the number of bytes in this dtype """
  81         return self.numpy_dtype.itemsize

  83     @classmethod
  84     def construct_array_type(cls) -> Type["IntegerArray"]:
  85         """
  86         Return the array type associated with this dtype.
  87 
  88         Returns
  89         -------
  90         type
  91         """
  92         return IntegerArray

  94     def _get_common_dtype(self, dtypes: List[DtypeObj]) -> Optional[DtypeObj]:
  95         # for now only handle other integer types
  96         if not all(isinstance(t, _IntegerDtype) for t in dtypes):
  97             return None
  98         np_dtype = np.find_common_type(
  99             [t.numpy_dtype for t in dtypes], []  # type: ignore
 100         )
 101         if np.issubdtype(np_dtype, np.integer):
 102             return _dtypes[str(np_dtype)]
 103         return None

 105     def __from_arrow__(
 106         self, array: Union["pyarrow.Array", "pyarrow.ChunkedArray"]
 107     ) -> "IntegerArray":
 108         """
 109         Construct IntegerArray from pyarrow Array/ChunkedArray.
 110         """
 111         import pyarrow  # noqa: F811
 112         from pandas.core.arrays._arrow_utils import pyarrow_array_to_numpy_and_mask
 113 
 114         pyarrow_type = pyarrow.from_numpy_dtype(self.type)
 115         if not array.type.equals(pyarrow_type):
 116             array = array.cast(pyarrow_type)
 117 
 118         if isinstance(array, pyarrow.Array):
 119             chunks = [array]
 120         else:
 121             # pyarrow.ChunkedArray
 122             chunks = array.chunks
 123 
 124         results = []
 125         for arr in chunks:
 126             data, mask = pyarrow_array_to_numpy_and_mask(arr, dtype=self.type)
 127             int_arr = IntegerArray(data.copy(), ~mask, copy=False)
 128             results.append(int_arr)
 129 
 130         return IntegerArray._concat_same_type(results)

 133 def integer_array(values, dtype=None, copy: bool = False,) -> "IntegerArray":
 134     """
 135     Infer and return an integer array of the values.
 136 
 137     Parameters
 138     ----------
 139     values : 1D list-like
 140     dtype : dtype, optional
 141         dtype to coerce
 142     copy : bool, default False
 143 
 144     Returns
 145     -------
 146     IntegerArray
 147 
 148     Raises
 149     ------
 150     TypeError if incompatible types
 151     """
 152     values, mask = coerce_to_array(values, dtype=dtype, copy=copy)
 153     return IntegerArray(values, mask)

 156 def safe_cast(values, dtype, copy: bool):
 157     """
 158     Safely cast the values to the dtype if they
 159     are equivalent, meaning floats must be equivalent to the
 160     ints.
 161 
 162     """
 163     try:
 164         return values.astype(dtype, casting="safe", copy=copy)
 165     except TypeError as err:
 166 
 167         casted = values.astype(dtype, copy=copy)
 168         if (casted == values).all():
 169             return casted
 170 
 171         raise TypeError(
 172             f"cannot safely cast non-equivalent {values.dtype} to {np.dtype(dtype)}"
 173         ) from err

 176 def coerce_to_array(
 177     values, dtype, mask=None, copy: bool = False,
 178 ) -> Tuple[np.ndarray, np.ndarray]:
 179     """
 180     Coerce the input values array to numpy arrays with a mask
 181 
 182     Parameters
 183     ----------
 184     values : 1D list-like
 185     dtype : integer dtype
 186     mask : bool 1D array, optional
 187     copy : bool, default False
 188         if True, copy the input
 189 
 190     Returns
 191     -------
 192     tuple of (values, mask)
 193     """
 194     # if values is integer numpy array, preserve it's dtype
 195     if dtype is None and hasattr(values, "dtype"):
 196         if is_integer_dtype(values.dtype):
 197             dtype = values.dtype
 198 
 199     if dtype is not None:
 200         if isinstance(dtype, str) and (
 201             dtype.startswith("Int") or dtype.startswith("UInt")
 202         ):
 203             # Avoid DeprecationWarning from NumPy about np.dtype("Int64")
 204             # https://github.com/numpy/numpy/pull/7476
 205             dtype = dtype.lower()
 206 
 207         if not issubclass(type(dtype), _IntegerDtype):
 208             try:
 209                 dtype = _dtypes[str(np.dtype(dtype))]
 210             except KeyError as err:
 211                 raise ValueError(f"invalid dtype specified {dtype}") from err
 212 
 213     if isinstance(values, IntegerArray):
 214         values, mask = values._data, values._mask
 215         if dtype is not None:
 216             values = values.astype(dtype.numpy_dtype, copy=False)
 217 
 218         if copy:
 219             values = values.copy()
 220             mask = mask.copy()
 221         return values, mask
 222 
 223     values = np.array(values, copy=copy)
 224     if is_object_dtype(values):
 225         inferred_type = lib.infer_dtype(values, skipna=True)
 226         if inferred_type == "empty":
 227             values = np.empty(len(values))
 228             values.fill(np.nan)
 229         elif inferred_type not in [
 230             "floating",
 231             "integer",
 232             "mixed-integer",
 233             "integer-na",
 234             "mixed-integer-float",
 235         ]:
 236             raise TypeError(f"{values.dtype} cannot be converted to an IntegerDtype")
 237 
 238     elif is_bool_dtype(values) and is_integer_dtype(dtype):
 239         values = np.array(values, dtype=int, copy=copy)
 240 
 241     elif not (is_integer_dtype(values) or is_float_dtype(values)):
 242         raise TypeError(f"{values.dtype} cannot be converted to an IntegerDtype")
 243 
 244     if mask is None:
 245         mask = isna(values)
 246     else:
 247         assert len(mask) == len(values)
 248 
 249     if not values.ndim == 1:
 250         raise TypeError("values must be a 1D list-like")
 251     if not mask.ndim == 1:
 252         raise TypeError("mask must be a 1D list-like")
 253 
 254     # infer dtype if needed
 255     if dtype is None:
 256         dtype = np.dtype("int64")
 257     else:
 258         dtype = dtype.type
 259 
 260     # if we are float, let's make sure that we can
 261     # safely cast
 262 
 263     # we copy as need to coerce here
 264     if mask.any():
 265         values = values.copy()
 266         values[mask] = 1
 267         values = safe_cast(values, dtype, copy=False)
 268     else:
 269         values = safe_cast(values, dtype, copy=False)
 270 
 271     return values, mask

 274 class IntegerArray(BaseMaskedArray):
 275     """
 276     Array of integer (optional missing) values.
 277 
 278     .. versionadded:: 0.24.0
 279 
 280     .. versionchanged:: 1.0.0
 281 
 282        Now uses :attr:`pandas.NA` as the missing value rather
 283        than :attr:`numpy.nan`.
 284 
 285     .. warning::
 286 
 287        IntegerArray is currently experimental, and its API or internal
 288        implementation may change without warning.
 289 
 290     We represent an IntegerArray with 2 numpy arrays:
 291 
 292     - data: contains a numpy integer array of the appropriate dtype
 293     - mask: a boolean array holding a mask on the data, True is missing
 294 
 295     To construct an IntegerArray from generic array-like input, use
 296     :func:`pandas.array` with one of the integer dtypes (see examples).
 297 
 298     See :ref:`integer_na` for more.
 299 
 300     Parameters
 301     ----------
 302     values : numpy.ndarray
 303         A 1-d integer-dtype array.
 304     mask : numpy.ndarray
 305         A 1-d boolean-dtype array indicating missing values.
 306     copy : bool, default False
 307         Whether to copy the `values` and `mask`.
 308 
 309     Attributes
 310     ----------
 311     None
 312 
 313     Methods
 314     -------
 315     None
 316 
 317     Returns
 318     -------
 319     IntegerArray
 320 
 321     Examples
 322     --------
 323     Create an IntegerArray with :func:`pandas.array`.
 324 
 325     >>> int_array = pd.array([1, None, 3], dtype=pd.Int32Dtype())
 326     >>> int_array
 327     <IntegerArray>
 328     [1, <NA>, 3]
 329     Length: 3, dtype: Int32
 330 
 331     String aliases for the dtypes are also available. They are capitalized.
 332 
 333     >>> pd.array([1, None, 3], dtype='Int32')
 334     <IntegerArray>
 335     [1, <NA>, 3]
 336     Length: 3, dtype: Int32
 337 
 338     >>> pd.array([1, None, 3], dtype='UInt16')
 339     <IntegerArray>
 340     [1, <NA>, 3]
 341     Length: 3, dtype: UInt16
 342     """
 343 
 344     # The value used to fill '_data' to avoid upcasting
 345     _internal_fill_value = 1
 346 
 347     @cache_readonly
 348     def dtype(self) -> _IntegerDtype:
 349         return _dtypes[str(self._data.dtype)]
 350 
 351     def __init__(self, values: np.ndarray, mask: np.ndarray, copy: bool = False):
 352         if not (isinstance(values, np.ndarray) and values.dtype.kind in ["i", "u"]):
 353             raise TypeError(
 354                 "values should be integer numpy array. Use "
 355                 "the 'pd.array' function instead"
 356             )
 357         super().__init__(values, mask, copy=copy)
 358 
 359     @classmethod
 360     def _from_sequence(cls, scalars, dtype=None, copy: bool = False) -> "IntegerArray":
 361         return integer_array(scalars, dtype=dtype, copy=copy)
 362 
 363     @classmethod
 364     def _from_sequence_of_strings(
 365         cls, strings, dtype=None, copy: bool = False
 366     ) -> "IntegerArray":
 367         scalars = to_numeric(strings, errors="raise")
 368         return cls._from_sequence(scalars, dtype, copy)

 370     _HANDLED_TYPES = (np.ndarray, numbers.Number)
 371 
 372     def __array_ufunc__(self, ufunc, method: str, *inputs, **kwargs):
 373         # For IntegerArray inputs, we apply the ufunc to ._data
 374         # and mask the result.
 375         if method == "reduce":
 376             # Not clear how to handle missing values in reductions. Raise.
 377             raise NotImplementedError("The 'reduce' method is not supported.")
 378         out = kwargs.get("out", ())
 379 
 380         for x in inputs + out:
 381             if not isinstance(x, self._HANDLED_TYPES + (IntegerArray,)):
 382                 return NotImplemented
 383 
 384         # for binary ops, use our custom dunder methods
 385         result = ops.maybe_dispatch_ufunc_to_dunder_op(
 386             self, ufunc, method, *inputs, **kwargs
 387         )
 388         if result is not NotImplemented:
 389             return result
 390 
 391         mask = np.zeros(len(self), dtype=bool)
 392         inputs2 = []
 393         for x in inputs:
 394             if isinstance(x, IntegerArray):
 395                 mask |= x._mask
 396                 inputs2.append(x._data)
 397             else:
 398                 inputs2.append(x)
 399 
 400         def reconstruct(x):
 401             # we don't worry about scalar `x` here, since we
 402             # raise for reduce up above.
 403 
 404             if is_integer_dtype(x.dtype):
 405                 m = mask.copy()
 406                 return IntegerArray(x, m)
 407             else:
 408                 x[mask] = np.nan
 409             return x
 410 
 411         result = getattr(ufunc, method)(*inputs2, **kwargs)
 412         if isinstance(result, tuple):
 413             tuple(reconstruct(x) for x in result)
 414         else:
 415             return reconstruct(result)

 420     def astype(self, dtype, copy: bool = True) -> ArrayLike:
 421         """
 422         Cast to a NumPy array or ExtensionArray with 'dtype'.
 423 
 424         Parameters
 425         ----------
 426         dtype : str or dtype
 427             Typecode or data-type to which the array is cast.
 428         copy : bool, default True
 429             Whether to copy the data, even if not necessary. If False,
 430             a copy is made only if the old dtype does not match the
 431             new dtype.
 432 
 433         Returns
 434         -------
 435         ndarray or ExtensionArray
 436             NumPy ndarray, BooleanArray or IntegerArray with 'dtype' for its dtype.
 437 
 438         Raises
 439         ------
 440         TypeError
 441             if incompatible type with an IntegerDtype, equivalent of same_kind
 442             casting
 443         """
 444         from pandas.core.arrays.boolean import BooleanDtype
 445         from pandas.core.arrays.string_ import StringDtype
 446 
 447         dtype = pandas_dtype(dtype)
 448 
 449         # if we are astyping to an existing IntegerDtype we can fastpath
 450         if isinstance(dtype, _IntegerDtype):
 451             result = self._data.astype(dtype.numpy_dtype, copy=False)
 452             return dtype.construct_array_type()(result, mask=self._mask, copy=False)
 453         elif isinstance(dtype, BooleanDtype):
 454             result = self._data.astype("bool", copy=False)
 455             return dtype.construct_array_type()(result, mask=self._mask, copy=False)
 456         elif isinstance(dtype, StringDtype):
 457             return dtype.construct_array_type()._from_sequence(self, copy=False)
 458 
 459         # coerce
 460         if is_float_dtype(dtype):
 461             # In astype, we consider dtype=float to also mean na_value=np.nan
 462             na_value = np.nan
 463         elif is_datetime64_dtype(dtype):
 464             na_value = np.datetime64("NaT")
 465         else:
 466             na_value = lib.no_default
 467 
 468         return self.to_numpy(dtype=dtype, na_value=na_value, copy=False)

 470     def _values_for_argsort(self) -> np.ndarray:
 471         """
 472         Return values for sorting.
 473 
 474         Returns
 475         -------
 476         ndarray
 477             The transformed values should maintain the ordering between values
 478             within the array.
 479 
 480         See Also
 481         --------
 482         ExtensionArray.argsort
 483         """
 484         data = self._data.copy()
 485         if self._mask.any():
 486             data[self._mask] = data.min() - 1
 487         return data

 489     @classmethod
 490     def _create_comparison_method(cls, op):
 491         op_name = op.__name__
 492 
 493         @unpack_zerodim_and_defer(op.__name__)
 494         def cmp_method(self, other):
 495             from pandas.arrays import BooleanArray
 496 
 497             mask = None
 498 
 499             if isinstance(other, (BooleanArray, IntegerArray)):
 500                 other, mask = other._data, other._mask
 501 
 502             elif is_list_like(other):
 503                 other = np.asarray(other)
 504                 if other.ndim > 1:
 505                     raise NotImplementedError(
 506                         "can only perform ops with 1-d structures"
 507                     )
 508                 if len(self) != len(other):
 509                     raise ValueError("Lengths must match to compare")
 510 
 511             if other is libmissing.NA:
 512                 # numpy does not handle pd.NA well as "other" scalar (it returns
 513                 # a scalar False instead of an array)
 514                 # This may be fixed by NA.__array_ufunc__. Revisit this check
 515                 # once that's implemented.
 516                 result = np.zeros(self._data.shape, dtype="bool")
 517                 mask = np.ones(self._data.shape, dtype="bool")
 518             else:
 519                 with warnings.catch_warnings():
 520                     # numpy may show a FutureWarning:
 521                     #     elementwise comparison failed; returning scalar instead,
 522                     #     but in the future will perform elementwise comparison
 523                     # before returning NotImplemented. We fall back to the correct
 524                     # behavior today, so that should be fine to ignore.
 525                     warnings.filterwarnings("ignore", "elementwise", FutureWarning)
 526                     with np.errstate(all="ignore"):
 527                         method = getattr(self._data, f"__{op_name}__")
 528                         result = method(other)
 529 
 530                     if result is NotImplemented:
 531                         result = invalid_comparison(self._data, other, op)
 532 
 533             # nans propagate
 534             if mask is None:
 535                 mask = self._mask.copy()
 536             else:
 537                 mask = self._mask | mask
 538 
 539             return BooleanArray(result, mask)
 540 
 541         name = f"__{op.__name__}__"
 542         return set_function_name(cmp_method, name, cls)
 543 
 544     def sum(self, skipna=True, min_count=0, **kwargs):
 545         nv.validate_sum((), kwargs)
 546         result = masked_reductions.sum(
 547             values=self._data, mask=self._mask, skipna=skipna, min_count=min_count
 548         )
 549         return result

 551     def _maybe_mask_result(self, result, mask, other, op_name: str):
 552         """
 553         Parameters
 554         ----------
 555         result : array-like
 556         mask : array-like bool
 557         other : scalar or array-like
 558         op_name : str
 559         """
 560         # if we have a float operand we are by-definition
 561         # a float result
 562         # or our op is a divide
 563         if (is_float_dtype(other) or is_float(other)) or (
 564             op_name in ["rtruediv", "truediv"]
 565         ):
 566             result[mask] = np.nan
 567             return result
 568 
 569         return type(self)(result, mask, copy=False)
 570 
 571     @classmethod
 572     def _create_arithmetic_method(cls, op):
 573         op_name = op.__name__
 574 
 575         @unpack_zerodim_and_defer(op.__name__)
 576         def integer_arithmetic_method(self, other):
 577 
 578             omask = None
 579 
 580             if getattr(other, "ndim", 0) > 1:
 581                 raise NotImplementedError("can only perform ops with 1-d structures")
 582 
 583             if isinstance(other, IntegerArray):
 584                 other, omask = other._data, other._mask
 585 
 586             elif is_list_like(other):
 587                 other = np.asarray(other)
 588                 if other.ndim > 1:
 589                     raise NotImplementedError(
 590                         "can only perform ops with 1-d structures"
 591                     )
 592                 if len(self) != len(other):
 593                     raise ValueError("Lengths must match")
 594                 if not (is_float_dtype(other) or is_integer_dtype(other)):
 595                     raise TypeError("can only perform ops with numeric values")
 596 
 597             else:
 598                 if not (is_float(other) or is_integer(other) or other is libmissing.NA):
 599                     raise TypeError("can only perform ops with numeric values")
 600 
 601             if omask is None:
 602                 mask = self._mask.copy()
 603                 if other is libmissing.NA:
 604                     mask |= True
 605             else:
 606                 mask = self._mask | omask
 607 
 608             if op_name == "pow":
 609                 # 1 ** x is 1.
 610                 mask = np.where((self._data == 1) & ~self._mask, False, mask)
 611                 # x ** 0 is 1.
 612                 if omask is not None:
 613                     mask = np.where((other == 0) & ~omask, False, mask)
 614                 elif other is not libmissing.NA:
 615                     mask = np.where(other == 0, False, mask)
 616 
 617             elif op_name == "rpow":
 618                 # 1 ** x is 1.
 619                 if omask is not None:
 620                     mask = np.where((other == 1) & ~omask, False, mask)
 621                 elif other is not libmissing.NA:
 622                     mask = np.where(other == 1, False, mask)
 623                 # x ** 0 is 1.
 624                 mask = np.where((self._data == 0) & ~self._mask, False, mask)
 625 
 626             if other is libmissing.NA:
 627                 result = np.ones_like(self._data)
 628             else:
 629                 with np.errstate(all="ignore"):
 630                     result = op(self._data, other)
 631 
 632             # divmod returns a tuple
 633             if op_name == "divmod":
 634                 div, mod = result
 635                 return (
 636                     self._maybe_mask_result(div, mask, other, "floordiv"),
 637                     self._maybe_mask_result(mod, mask, other, "mod"),
 638                 )
 639 
 640             return self._maybe_mask_result(result, mask, other, op_name)
 641 
 642         name = f"__{op.__name__}__"
 643         return set_function_name(integer_arithmetic_method, name, cls)
