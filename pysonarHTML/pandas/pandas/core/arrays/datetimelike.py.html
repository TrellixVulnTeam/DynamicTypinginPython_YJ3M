  70 def _datetimelike_array_cmp(cls, op):
  71     """
  72     Wrap comparison operations to convert Timestamp/Timedelta/Period-like to
  73     boxed scalars/arrays.
  74     """
  75     opname = f"__{op.__name__}__"
  76     nat_result = opname == "__ne__"
  77 
  78     class InvalidComparison(Exception):
  79         pass
  80 
  81     def _validate_comparison_value(self, other):
  82         if isinstance(other, str):
  83             try:
  84                 # GH#18435 strings get a pass from tzawareness compat
  85                 other = self._scalar_from_string(other)
  86             except ValueError:
  87                 # failed to parse as Timestamp/Timedelta/Period
  88                 raise InvalidComparison(other)
  89 
  90         if isinstance(other, self._recognized_scalars) or other is NaT:
  91             other = self._scalar_type(other)
  92             self._check_compatible_with(other)
  93 
  94         elif not is_list_like(other):
  95             raise InvalidComparison(other)
  96 
  97         elif len(other) != len(self):
  98             raise ValueError("Lengths must match")
  99 
 100         else:
 101             try:
 102                 other = self._validate_listlike(other, opname, allow_object=True)
 103             except TypeError as err:
 104                 raise InvalidComparison(other) from err
 105 
 106         return other
 107 
 108     @unpack_zerodim_and_defer(opname)
 109     def wrapper(self, other):
 110         if self.ndim > 1 and getattr(other, "shape", None) == self.shape:
 111             # TODO: handle 2D-like listlikes
 112             return op(self.ravel(), other.ravel()).reshape(self.shape)
 113 
 114         try:
 115             other = _validate_comparison_value(self, other)
 116         except InvalidComparison:
 117             return invalid_comparison(self, other, op)
 118 
 119         dtype = getattr(other, "dtype", None)
 120         if is_object_dtype(dtype):
 121             # We have to use comp_method_OBJECT_ARRAY instead of numpy
 122             #  comparison otherwise it would fail to raise when
 123             #  comparing tz-aware and tz-naive
 124             with np.errstate(all="ignore"):
 125                 result = ops.comp_method_OBJECT_ARRAY(op, self.astype(object), other)
 126             return result
 127 
 128         other_i8 = self._unbox(other)
 129         result = op(self.asi8, other_i8)
 130 
 131         o_mask = isna(other)
 132         if self._hasnans | np.any(o_mask):
 133             result[self._isnan | o_mask] = nat_result
 134 
 135         return result
 136 
 137     return set_function_name(wrapper, opname, cls)

 380     def _round(self, freq, mode, ambiguous, nonexistent):
 381         # round the local times
 382         if is_datetime64tz_dtype(self.dtype):
 383             # operate on naive timestamps, then convert back to aware
 384             naive = self.tz_localize(None)
 385             result = naive._round(freq, mode, ambiguous, nonexistent)
 386             aware = result.tz_localize(
 387                 self.tz, ambiguous=ambiguous, nonexistent=nonexistent
 388             )
 389             return aware
 390 
 391         values = self.view("i8")
 392         result = round_nsint64(values, mode, freq)
 393         result = self._maybe_mask_results(result, fill_value=NaT)
 394         return self._simple_new(result, dtype=self.dtype)

 529     def __getitem__(self, key):
 530         """
 531         This getitem defers to the underlying array, which by-definition can
 532         only handle list-likes, slices, and integer scalars
 533         """
 534 
 535         if lib.is_integer(key):
 536             # fast-path
 537             result = self._data[key]
 538             if self.ndim == 1:
 539                 return self._box_func(result)
 540             return self._simple_new(result, dtype=self.dtype)
 541 
 542         if com.is_bool_indexer(key):
 543             # first convert to boolean, because check_array_indexer doesn't
 544             # allow object dtype
 545             if is_object_dtype(key):
 546                 key = np.asarray(key, dtype=bool)
 547 
 548             key = check_array_indexer(self, key)
 549             key = lib.maybe_booleans_to_slice(key.view(np.uint8))
 550         elif isinstance(key, list) and len(key) == 1 and isinstance(key[0], slice):
 551             # see https://github.com/pandas-dev/pandas/issues/31299, need to allow
 552             # this for now (would otherwise raise in check_array_indexer)
 553             pass
 554         else:
 555             key = check_array_indexer(self, key)
 556 
 557         freq = self._get_getitem_freq(key)
 558         result = self._data[key]
 559         if lib.is_scalar(result):
 560             return self._box_func(result)
 561         return self._simple_new(result, dtype=self.dtype, freq=freq)

 563     def _get_getitem_freq(self, key):
 564         """
 565         Find the `freq` attribute to assign to the result of a __getitem__ lookup.
 566         """
 567         is_period = is_period_dtype(self.dtype)
 568         if is_period:
 569             freq = self.freq
 570         else:
 571             freq = None
 572             if isinstance(key, slice):
 573                 if self.freq is not None and key.step is not None:
 574                     freq = key.step * self.freq
 575                 else:
 576                     freq = self.freq
 577             elif key is Ellipsis:
 578                 # GH#21282 indexing with Ellipsis is similar to a full slice,
 579                 #  should preserve `freq` attribute
 580                 freq = self.freq
 581         return freq

 583     def __setitem__(
 584         self,
 585         key: Union[int, Sequence[int], Sequence[bool], slice],
 586         value: Union[NaTType, Any, Sequence[Any]],
 587     ) -> None:
 588         # I'm fudging the types a bit here. "Any" above really depends
 589         # on type(self). For PeriodArray, it's Period (or stuff coercible
 590         # to a period in from_sequence). For DatetimeArray, it's Timestamp...
 591         # I don't know if mypy can do that, possibly with Generics.
 592         # https://mypy.readthedocs.io/en/latest/generics.html
 593         if is_list_like(value):
 594             is_slice = isinstance(key, slice)
 595 
 596             if lib.is_scalar(key):
 597                 raise ValueError("setting an array element with a sequence.")
 598 
 599             if not is_slice:
 600                 key = cast(Sequence, key)
 601                 if len(key) != len(value) and not com.is_bool_indexer(key):
 602                     msg = (
 603                         f"shape mismatch: value array of length '{len(key)}' "
 604                         "does not match indexing result of length "
 605                         f"'{len(value)}'."
 606                     )
 607                     raise ValueError(msg)
 608                 elif not len(key):
 609                     return
 610 
 611         value = self._validate_setitem_value(value)
 612         key = check_array_indexer(self, key)
 613         self._data[key] = value
 614         self._maybe_clear_freq()

 623         #   1. PeriodArray.astype handles period -> period

 625         #   3. DatetimeArray.astype handles datetime -> period
 626         dtype = pandas_dtype(dtype)

1349     def _time_shift(self, periods, freq=None):
1350         """
1351         Shift each value by `periods`.
1352 
1353         Note this is different from ExtensionArray.shift, which
1354         shifts the *position* of each element, padding the end with
1355         missing values.
1356 
1357         Parameters
1358         ----------
1359         periods : int
1360             Number of periods to shift by.
1361         freq : pandas.DateOffset, pandas.Timedelta, or str
1362             Frequency increment to shift by.
1363         """
1364         if freq is not None and freq != self.freq:
1365             if isinstance(freq, str):
1366                 freq = to_offset(freq)
1367             offset = periods * freq
1368             result = self + offset
1369             return result
1370 
1371         if periods == 0:
1372             # immutable so OK
1373             return self.copy()
1374 
1375         if self.freq is None:
1376             raise NullFrequencyError("Cannot shift with no freq")
1377 
1378         start = self[0] + periods * self.freq
1379         end = self[-1] + periods * self.freq
1380 
1381         # Note: in the DatetimeTZ case, _generate_range will infer the
1382         #  appropriate timezone from `start` and `end`, so tz does not need
1383         #  to be passed explicitly.
1384         return self._generate_range(start=start, end=end, periods=None, freq=self.freq)
