 198

 200         left = ensure_index(left, copy=copy)

 205             dtype = pandas_dtype(dtype)

 467     def _validate(self):
 468         """
 469         Verify that the IntervalArray is valid.
 470 
 471         Checks that
 472 
 473         * closed is valid
 474         * left and right match lengths
 475         * left and right have the same missing values
 476         * left is always below right
 477         """
 478         if self.closed not in _VALID_CLOSED:
 479             msg = f"invalid option for 'closed': {self.closed}"
 480             raise ValueError(msg)
 481         if len(self.left) != len(self.right):
 482             msg = "left and right must have the same length"
 483             raise ValueError(msg)
 484         left_mask = notna(self.left)
 485         right_mask = notna(self.right)
 486         if not (left_mask == right_mask).all():
 487             msg = (
 488                 "missing values must be missing in the same "
 489                 "location both left and right sides"
 490             )
 491             raise ValueError(msg)
 492         if not (self.left[left_mask] <= self.right[left_mask]).all():
 493             msg = "left side of interval must be <= right side"
 494             raise ValueError(msg)

 505     def __getitem__(self, value):
 506         value = check_array_indexer(self, value)
 507         left = self.left[value]
 508         right = self.right[value]
 509 
 510         # scalar
 511         if not isinstance(left, ABCIndexClass):
 512             if is_scalar(left) and isna(left):
 513                 return self._fill_value
 514             if np.ndim(left) > 1:
 515                 # GH#30588 multi-dimensional indexer disallowed
 516                 raise ValueError("multi-dimensional indexing not allowed")
 517             return Interval(left, right, self.closed)
 518 
 519         return self._shallow_copy(left, right)

 521     def __setitem__(self, key, value):
 522         # na value: need special casing to set directly on numpy arrays
 523         needs_float_conversion = False
 524         if is_scalar(value) and isna(value):
 525             if is_integer_dtype(self.dtype.subtype):
 526                 # can't set NaN on a numpy integer array
 527                 needs_float_conversion = True
 528             elif is_datetime64_any_dtype(self.dtype.subtype):
 529                 # need proper NaT to set directly on the numpy array
 530                 value = np.datetime64("NaT")
 531             elif is_timedelta64_dtype(self.dtype.subtype):
 532                 # need proper NaT to set directly on the numpy array
 533                 value = np.timedelta64("NaT")
 534             value_left, value_right = value, value
 535 
 536         # scalar interval
 537         elif is_interval_dtype(value) or isinstance(value, Interval):
 538             self._check_closed_matches(value, name="value")
 539             value_left, value_right = value.left, value.right
 540 
 541         else:
 542             # list-like of intervals
 543             try:
 544                 array = IntervalArray(value)
 545                 value_left, value_right = array.left, array.right
 546             except TypeError as err:
 547                 # wrong type: not interval or NA
 548                 msg = f"'value' should be an interval type, got {type(value)} instead."
 549                 raise TypeError(msg) from err
 550 
 551         if needs_float_conversion:
 552             raise ValueError("Cannot set float NaN to integer-backed IntervalArray")
 553 
 554         key = check_array_indexer(self, key)
 555 
 556         # Need to ensure that left and right are updated atomically, so we're
 557         # forced to copy, update the copy, and swap in the new values.
 558         left = self.left.copy(deep=True)
 559         left._values[key] = value_left
 560         self._left = left
 561 
 562         right = self.right.copy(deep=True)
 563         right._values[key] = value_right
 564         self._right = right

 566     def __eq__(self, other):
 567         # ensure pandas array for list-like and eliminate non-interval scalars
 568         if is_list_like(other):
 569             if len(self) != len(other):
 570                 raise ValueError("Lengths must match to compare")
 571             other = array(other)
 572         elif not isinstance(other, Interval):
 573             # non-interval scalar -> no matches
 574             return np.zeros(len(self), dtype=bool)
 575 
 576         # determine the dtype of the elements we want to compare
 577         if isinstance(other, Interval):
 578             other_dtype = pandas_dtype("interval")
 579         elif not is_categorical_dtype(other.dtype):
 580             other_dtype = other.dtype
 581         else:
 582             # for categorical defer to categories for dtype
 583             other_dtype = other.categories.dtype
 584 
 585             # extract intervals if we have interval categories with matching closed
 586             if is_interval_dtype(other_dtype):
 587                 if self.closed != other.categories.closed:
 588                     return np.zeros(len(self), dtype=bool)
 589                 other = other.categories.take(other.codes)
 590 
 591         # interval-like -> need same closed and matching endpoints
 592         if is_interval_dtype(other_dtype):
 593             if self.closed != other.closed:
 594                 return np.zeros(len(self), dtype=bool)
 595             return (self.left == other.left) & (self.right == other.right)
 596 
 597         # non-interval/non-object dtype -> no matches
 598         if not is_object_dtype(other_dtype):
 599             return np.zeros(len(self), dtype=bool)
 600 
 601         # object dtype -> iteratively check for intervals
 602         result = np.zeros(len(self), dtype=bool)
 603         for i, obj in enumerate(other):
 604             # need object to be an Interval with same closed and endpoints
 605             if (
 606                 isinstance(obj, Interval)
 607                 and self.closed == obj.closed
 608                 and self.left[i] == obj.left
 609                 and self.right[i] == obj.right
 610             ):
 611                 result[i] = True
 612 
 613         return result

 615     def fillna(self, value=None, method=None, limit=None):
 616         """
 617         Fill NA/NaN values using the specified method.
 618 
 619         Parameters
 620         ----------
 621         value : scalar, dict, Series
 622             If a scalar value is passed it is used to fill all missing values.
 623             Alternatively, a Series or dict can be used to fill in different
 624             values for each index. The value should not be a list. The
 625             value(s) passed should be either Interval objects or NA/NaN.
 626         method : {'backfill', 'bfill', 'pad', 'ffill', None}, default None
 627             (Not implemented yet for IntervalArray)
 628             Method to use for filling holes in reindexed Series
 629         limit : int, default None
 630             (Not implemented yet for IntervalArray)
 631             If method is specified, this is the maximum number of consecutive
 632             NaN values to forward/backward fill. In other words, if there is
 633             a gap with more than this number of consecutive NaNs, it will only
 634             be partially filled. If method is not specified, this is the
 635             maximum number of entries along the entire axis where NaNs will be
 636             filled.
 637 
 638         Returns
 639         -------
 640         filled : IntervalArray with NA/NaN filled
 641         """
 642         if method is not None:
 643             raise TypeError("Filling by method is not supported for IntervalArray.")
 644         if limit is not None:
 645             raise TypeError("limit is not supported for IntervalArray.")
 646 
 647         if not isinstance(value, Interval):
 648             msg = (
 649                 "'IntervalArray.fillna' only supports filling with a "
 650                 f"scalar 'pandas.Interval'. Got a '{type(value).__name__}' instead."
 651             )
 652             raise TypeError(msg)
 653 
 654         self._check_closed_matches(value, name="value")
 655 
 656         left = self.left.fillna(value=value.left)
 657         right = self.right.fillna(value=value.right)
 658         return self._shallow_copy(left, right)

 664     def astype(self, dtype, copy=True):
 665         """
 666         Cast to an ExtensionArray or NumPy array with dtype 'dtype'.
 667 
 668         Parameters
 669         ----------
 670         dtype : str or dtype

 672

 674             Whether to copy the data, even if not necessary. If False,
 675             a copy is made only if the old dtype does not match the
 676             new dtype.
 677 
 678         Returns
 679         -------
 680         array : ExtensionArray or ndarray
 681             ExtensionArray or NumPy ndarray with 'dtype' for its dtype.
 682         """
 683         from pandas.core.arrays.string_ import StringDtype
 684 
 685         if dtype is not None:
 686             dtype = pandas_dtype(dtype)
 687 
 688         if is_interval_dtype(dtype):
 689             if dtype == self.dtype:
 690                 return self.copy() if copy else self
 691 
 692             # need to cast to different subtype
 693             try:
 694                 new_left = self.left.astype(dtype.subtype)
 695                 new_right = self.right.astype(dtype.subtype)
 696             except TypeError as err:
 697                 msg = (
 698                     f"Cannot convert {self.dtype} to {dtype}; subtypes are incompatible"
 699                 )
 700                 raise TypeError(msg) from err
 701             return self._shallow_copy(new_left, new_right)
 702         elif is_categorical_dtype(dtype):
 703             return Categorical(np.asarray(self))
 704         elif isinstance(dtype, StringDtype):
 705             return dtype.construct_array_type()._from_sequence(self, copy=False)
 706 
 707         # TODO: This try/except will be repeated.
 708         try:
 709             return np.asarray(self).astype(dtype, copy=copy)
 710         except (TypeError, ValueError) as err:
 711             msg = f"Cannot cast {type(self).__name__} to dtype {dtype}"
 712             raise TypeError(msg) from err

 715     def _concat_same_type(cls, to_concat):
 716         """
 717         Concatenate multiple IntervalArray
 718 
 719         Parameters
 720         ----------
 721         to_concat : sequence of IntervalArray
 722 
 723         Returns
 724         -------
 725         IntervalArray
 726         """

 729             raise ValueError("Intervals must all be closed on the same side.")

 731

 775     def shift(self, periods: int = 1, fill_value: object = None) -> "IntervalArray":
 776         if not len(self) or periods == 0:
 777             return self.copy()
 778 
 779         if isna(fill_value):
 780             fill_value = self.dtype.na_value
 781 
 782         # ExtensionArray.shift doesn't work for two reasons
 783         # 1. IntervalArray.dtype.na_value may not be correct for the dtype.
 784         # 2. IntervalArray._from_sequence only accepts NaN for missing values,
 785         #    not other values like NaT
 786 
 787         empty_len = min(abs(periods), len(self))
 788         if isna(fill_value):
 789             fill_value = self.left._na_value
 790             empty = IntervalArray.from_breaks([fill_value] * (empty_len + 1))
 791         else:
 792             empty = self._from_sequence([fill_value] * empty_len)
 793 
 794         if periods > 0:
 795             a = empty
 796             b = self[:-periods]
 797         else:
 798             a = self[abs(periods) :]
 799             b = empty
 800         return self._concat_same_type([a, b])

 802     def take(self, indices, allow_fill=False, fill_value=None, axis=None, **kwargs):
 803         """
 804         Take elements from the IntervalArray.
 805 
 806         Parameters
 807         ----------
 808         indices : sequence of integers
 809             Indices to be taken.
 810 
 811         allow_fill : bool, default False
 812             How to handle negative values in `indices`.
 813 
 814             * False: negative values in `indices` indicate positional indices
 815               from the right (the default). This is similar to
 816               :func:`numpy.take`.
 817 
 818             * True: negative values in `indices` indicate
 819               missing values. These values are set to `fill_value`. Any other
 820               other negative values raise a ``ValueError``.
 821 
 822         fill_value : Interval or NA, optional
 823             Fill value to use for NA-indices when `allow_fill` is True.
 824             This may be ``None``, in which case the default NA value for
 825             the type, ``self.dtype.na_value``, is used.
 826 
 827             For many ExtensionArrays, there will be two representations of
 828             `fill_value`: a user-facing "boxed" scalar, and a low-level
 829             physical NA value. `fill_value` should be the user-facing version,
 830             and the implementation should handle translating that to the
 831             physical version for processing the take if necessary.
 832 
 833         axis : any, default None
 834             Present for compat with IntervalIndex; does nothing.
 835 
 836         Returns
 837         -------
 838         IntervalArray
 839 
 840         Raises
 841         ------
 842         IndexError
 843             When the indices are out of bounds for the array.
 844         ValueError
 845             When `indices` contains negative values other than ``-1``
 846             and `allow_fill` is True.
 847         """
 848         nv.validate_take(tuple(), kwargs)
 849 
 850         fill_left = fill_right = fill_value
 851         if allow_fill:
 852             if fill_value is None:
 853                 fill_left = fill_right = self.left._na_value
 854             elif is_interval(fill_value):
 855                 self._check_closed_matches(fill_value, name="fill_value")
 856                 fill_left, fill_right = fill_value.left, fill_value.right
 857             elif not is_scalar(fill_value) and notna(fill_value):
 858                 msg = (
 859                     "'IntervalArray.fillna' only supports filling with a "
 860                     "'scalar pandas.Interval or NA'. "
 861                     f"Got a '{type(fill_value).__name__}' instead."
 862                 )
 863                 raise ValueError(msg)
 864 
 865         left_take = take(
 866             self.left, indices, allow_fill=allow_fill, fill_value=fill_left
 867         )
 868         right_take = take(
 869             self.right, indices, allow_fill=allow_fill, fill_value=fill_right
 870         )
 871 
 872         return self._shallow_copy(left_take, right_take)

 922                 summary = f"[{head_str} ... {tail_str}]"
 923             else:
 924                 tail = [formatter(x) for x in self]
 925                 tail_str = ", ".join(tail)
 926                 summary = f"[{tail_str}]"

1063     )
1064     def is_non_overlapping_monotonic(self):
1065         # must be increasing  (e.g., [0, 1), [1, 2), [2, 3), ... )
1066         # or decreasing (e.g., [-1, 0), [-2, -1), [-3, -2), ...)
1067         # we already require left <= right
1068 
1069         # strict inequality for closed == 'both'; equality implies overlapping
1070         # at a point when both sides of intervals are included
1071         if self.closed == "both":
1072             return bool(
1073                 (self.right[:-1] < self.left[1:]).all()
1074                 or (self.left[:-1] > self.right[1:]).all()
1075             )
1076 
1077         # non-strict inequality when closed != 'both'; at least one side is
1078         # not included in the intervals, so equality does not imply overlapping
1079         return bool(
1080             (self.right[:-1] <= self.left[1:]).all()
1081             or (self.left[:-1] >= self.right[1:]).all()
1082         )

1103     def __arrow_array__(self, type=None):
1104         """
1105         Convert myself into a pyarrow Array.
1106         """
1107         import pyarrow
1108         from pandas.core.arrays._arrow_utils import ArrowIntervalType
1109 
1110         try:
1111             subtype = pyarrow.from_numpy_dtype(self.dtype.subtype)
1112         except TypeError as err:
1113             raise TypeError(
1114                 f"Conversion to arrow with subtype '{self.dtype.subtype}' "
1115                 "is not supported"
1116             ) from err
1117         interval_type = ArrowIntervalType(subtype, self.closed)
1118         storage_array = pyarrow.StructArray.from_arrays(
1119             [
1120                 pyarrow.array(self.left, type=subtype, from_pandas=True),
1121                 pyarrow.array(self.right, type=subtype, from_pandas=True),
1122             ],
1123             names=["left", "right"],
1124         )
1125         mask = self.isna()
1126         if mask.any():
1127             # if there are missing values, set validity bitmap also on the array level
1128             null_bitmap = pyarrow.array(~mask).buffers()[1]
1129             storage_array = pyarrow.StructArray.from_buffers(
1130                 storage_array.type,
1131                 len(storage_array),
1132                 [null_bitmap],
1133                 children=[storage_array.field(0), storage_array.field(1)],
1134             )
1135 
1136         if type is not None:
1137             if type.equals(interval_type.storage_type):
1138                 return storage_array
1139             elif isinstance(type, ArrowIntervalType):
1140                 # ensure we have the same subtype and closed attributes
1141                 if not type.equals(interval_type):
1142                     raise TypeError(
1143                         "Not supported to convert IntervalArray to type with "
1144                         f"different 'subtype' ({self.dtype.subtype} vs {type.subtype}) "
1145                         f"and 'closed' ({self.closed} vs {type.closed}) attributes"
1146                     )
1147             else:

1149                     f"Not supported to convert IntervalArray to '{type}' type"
1150                 )
1151

1305             raise NotImplementedError
1306         elif not isinstance(other, Interval):
1307             msg = f"`other` must be Interval-like, got {type(other).__name__}"
1308             raise TypeError(msg)
1309 
1310         # equality is okay if both endpoints are closed (overlap at a point)
1311         op1 = le if (self.closed_left and other.closed_right) else lt
1312         op2 = le if (other.closed_left and self.closed_right) else lt
1313 
1314         # overlaps is equivalent negation of two interval being disjoint:
1315         # disjoint = (A.left > B.right) or (B.left > A.right)
1316         # (simplifying the negation allows this to be done in less operations)
1317         return op1(self.left, other.right) & op2(other.left, self.right)

1323     Wrapper around maybe_convert_platform that alters the default return
