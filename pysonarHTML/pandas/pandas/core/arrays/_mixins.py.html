   3 import numpy as np

   5 from pandas.compat.numpy import function as nv
   6 from pandas.errors import AbstractMethodError
   7 from pandas.util._decorators import cache_readonly

   9 from pandas.core.algorithms import take, unique
  10 from pandas.core.arrays.base import ExtensionArray
  11 
  12 _T = TypeVar("_T", bound="NDArrayBackedExtensionArray")
  13 
  14 
  15 class NDArrayBackedExtensionArray(ExtensionArray):
  16     """
  17     ExtensionArray that is backed by a single NumPy ndarray.
  18     """
  19 
  20     _ndarray: np.ndarray
  21 
  22     def _from_backing_data(self: _T, arr: np.ndarray) -> _T:
  23         """
  24         Construct a new ExtensionArray `new_array` with `arr` as its _ndarray.
  25 
  26         This should round-trip:
  27             self == self._from_backing_data(self._ndarray)
  28         """
  29         raise AbstractMethodError(self)
  30 
  31     # ------------------------------------------------------------------------
  32 
  33     def take(
  34         self: _T,
  35         indices: Sequence[int],
  36         allow_fill: bool = False,
  37         fill_value: Any = None,
  38     ) -> _T:
  39         if allow_fill:
  40             fill_value = self._validate_fill_value(fill_value)
  41 
  42         new_data = take(
  43             self._ndarray, indices, allow_fill=allow_fill, fill_value=fill_value,
  44         )
  45         return self._from_backing_data(new_data)
  46 
  47     def _validate_fill_value(self, fill_value):
  48         """
  49         If a fill_value is passed to `take` convert it to a representation
  50         suitable for self._ndarray, raising ValueError if this is not possible.
  51 
  52         Parameters
  53         ----------
  54         fill_value : object
  55 
  56         Returns
  57         -------
  58         fill_value : native representation
  59 
  60         Raises
  61         ------
  62         ValueError
  63         """
  64         raise AbstractMethodError(self)
  65 
  66     # ------------------------------------------------------------------------
  67 
  68     # TODO: make this a cache_readonly; for that to work we need to remove

  70     @property
  71     def shape(self) -> Tuple[int, ...]:
  72         return self._ndarray.shape
  73 
  74     def __len__(self) -> int:
  75         return self.shape[0]
  76 
  77     @cache_readonly
  78     def ndim(self) -> int:
  79         return len(self.shape)
  80 
  81     @cache_readonly
  82     def size(self) -> int:
  83         return np.prod(self.shape)
  84 
  85     @cache_readonly
  86     def nbytes(self) -> int:
  87         return self._ndarray.nbytes
  88 
  89     def reshape(self: _T, *args, **kwargs) -> _T:
  90         new_data = self._ndarray.reshape(*args, **kwargs)
  91         return self._from_backing_data(new_data)
  92 
  93     def ravel(self: _T, *args, **kwargs) -> _T:
  94         new_data = self._ndarray.ravel(*args, **kwargs)
  95         return self._from_backing_data(new_data)
  96 
  97     @property
  98     def T(self: _T) -> _T:
  99         new_data = self._ndarray.T
 100         return self._from_backing_data(new_data)
 101 
 102     # ------------------------------------------------------------------------
 103 
 104     def copy(self: _T) -> _T:
 105         new_data = self._ndarray.copy()
 106         return self._from_backing_data(new_data)
 107 
 108     def repeat(self: _T, repeats, axis=None) -> _T:
 109         """
 110         Repeat elements of an array.
 111 
 112         See Also
 113         --------
 114         numpy.ndarray.repeat
 115         """
 116         nv.validate_repeat(tuple(), dict(axis=axis))
 117         new_data = self._ndarray.repeat(repeats, axis=axis)
 118         return self._from_backing_data(new_data)
 119 
 120     def unique(self: _T) -> _T:
 121         new_data = unique(self._ndarray)
 122         return self._from_backing_data(new_data)
