 137     def __init__(self, values, dtype=TD64NS_DTYPE, freq=lib.no_default, copy=False):
 138         values = extract_array(values)
 139 
 140         inferred_freq = getattr(values, "_freq", None)
 141         explicit_none = freq is None
 142         freq = freq if freq is not lib.no_default else None
 143 
 144         if isinstance(values, type(self)):
 145             if explicit_none:
 146                 # dont inherit from values
 147                 pass
 148             elif freq is None:
 149                 freq = values.freq
 150             elif freq and values.freq:
 151                 freq = to_offset(freq)
 152                 freq, _ = dtl.validate_inferred_freq(freq, values.freq, False)
 153             values = values._data
 154 
 155         if not isinstance(values, np.ndarray):
 156             msg = (
 157                 f"Unexpected type '{type(values).__name__}'. 'values' must be a "
 158                 "TimedeltaArray ndarray, or Series or Index containing one of those."
 159             )
 160             raise ValueError(msg)
 161         if values.ndim not in [1, 2]:
 162             raise ValueError("Only 1-dimensional input arrays are supported.")
 163 
 164         if values.dtype == "i8":
 165             # for compat with datetime/timedelta/period shared methods,
 166             #  we can sometimes get here with int64 values.  These represent
 167             #  nanosecond UTC (or tz-naive) unix timestamps
 168             values = values.view(TD64NS_DTYPE)
 169 
 170         _validate_td64_dtype(values.dtype)
 171         dtype = _validate_td64_dtype(dtype)
 172 
 173         if freq == "infer":
 174             msg = (
 175                 "Frequency inference not allowed in TimedeltaArray.__init__. "
 176                 "Use 'pd.array()' instead."
 177             )
 178             raise ValueError(msg)
 179 
 180         if copy:
 181             values = values.copy()
 182         if freq:
 183             freq = to_offset(freq)
 184 
 185         self._data = values
 186         self._dtype = dtype
 187         self._freq = freq
 188 
 189         if inferred_freq is None and freq is not None:
 190             type(self)._validate_frequency(self, freq)

 192     @classmethod
 193     def _simple_new(cls, values, freq=None, dtype=TD64NS_DTYPE):
 194         assert dtype == TD64NS_DTYPE, dtype
 195         assert isinstance(values, np.ndarray), type(values)
 196         if values.dtype != TD64NS_DTYPE:
 197             assert values.dtype == "i8"
 198             values = values.view(TD64NS_DTYPE)
 199 
 200         result = object.__new__(cls)
 201         result._data = values
 202         result._freq = to_offset(freq)
 203         result._dtype = TD64NS_DTYPE
 204         return result
 205 
 206     @classmethod
 207     def _from_sequence(
 208         cls, data, dtype=TD64NS_DTYPE, copy=False, freq=lib.no_default, unit=None
 209     ):
 210         if dtype:
 211             _validate_td64_dtype(dtype)
 212 
 213         explicit_none = freq is None
 214         freq = freq if freq is not lib.no_default else None
 215 
 216         freq, freq_infer = dtl.maybe_infer_freq(freq)
 217 
 218         data, inferred_freq = sequence_to_td64ns(data, copy=copy, unit=unit)
 219         freq, freq_infer = dtl.validate_inferred_freq(freq, inferred_freq, freq_infer)
 220         if explicit_none:
 221             freq = None
 222 
 223         result = cls._simple_new(data, freq=freq)
 224 
 225         if inferred_freq is None and freq is not None:
 226             # this condition precludes `freq_infer`
 227             cls._validate_frequency(result, freq)
 228 
 229         elif freq_infer:
 230             # Set _freq directly to bypass duplicative _validate_frequency
 231             # check.
 232             result._freq = to_offset(result.inferred_freq)
 233 
 234         return result

 236     @classmethod
 237     def _generate_range(cls, start, end, periods, freq, closed=None):
 238 
 239         periods = dtl.validate_periods(periods)
 240         if freq is None and any(x is None for x in [periods, start, end]):
 241             raise ValueError("Must provide freq argument if no data is supplied")
 242 
 243         if com.count_not_none(start, end, periods, freq) != 3:
 244             raise ValueError(
 245                 "Of the four parameters: start, end, periods, "
 246                 "and freq, exactly three must be specified"
 247             )
 248 
 249         if start is not None:
 250             start = Timedelta(start)
 251 
 252         if end is not None:
 253             end = Timedelta(end)
 254 
 255         left_closed, right_closed = dtl.validate_endpoints(closed)
 256 
 257         if freq is not None:
 258             index = generate_regular_range(start, end, periods, freq)
 259         else:
 260             index = np.linspace(start.value, end.value, periods).astype("i8")
 261             if len(index) >= 2:
 262                 # Infer a frequency
 263                 td = Timedelta(index[1] - index[0])
 264                 freq = to_offset(td)
 265 
 266         if not left_closed:
 267             index = index[1:]
 268         if not right_closed:
 269             index = index[:-1]
 270 
 271         return cls._simple_new(index, freq=freq)

 295     def astype(self, dtype, copy=True):
 296         # We handle
 297         #   --> timedelta64[ns]
 298         #   --> timedelta64
 299         # DatetimeLikeArrayMixin super call handles other cases
 300         dtype = pandas_dtype(dtype)
 301 
 302         if is_timedelta64_dtype(dtype) and not is_timedelta64_ns_dtype(dtype):
 303             # by pandas convention, converting to non-nano timedelta64
 304             #  returns an int64-dtyped array with ints representing multiples
 305             #  of the desired timedelta unit.  This is essentially division
 306             if self._hasnans:
 307                 # avoid double-copying
 308                 result = self._data.astype(dtype, copy=False)
 309                 values = self._maybe_mask_results(
 310                     result, fill_value=None, convert="float64"
 311                 )
 312                 return values
 313             result = self._data.astype(dtype, copy=copy)
 314             return result.astype("i8")
 315         elif is_timedelta64_ns_dtype(dtype):
 316             if copy:
 317                 return self.copy()
 318             return self
 319         return dtl.DatetimeLikeArrayMixin.astype(self, dtype, copy=copy)

 325         self,
 326         axis=None,
 327         dtype=None,
 328         out=None,
 329         keepdims: bool = False,
 330         initial=None,

 344         )
 345         return Timedelta(result)

 427     def _add_datetimelike_scalar(self, other):
 428         # adding a timedeltaindex to a datetimelike
 429         from pandas.core.arrays import DatetimeArray
 430 
 431         assert other is not NaT
 432         other = Timestamp(other)
 433         if other is NaT:
 434             # In this case we specifically interpret NaT as a datetime, not
 435             # the timedelta interpretation we would get by returning self + NaT
 436             result = self.asi8.view("m8[ms]") + NaT.to_datetime64()
 437             return DatetimeArray(result)
 438 
 439         i8 = self.asi8
 440         result = checked_add_with_arr(i8, other.value, arr_mask=self._isnan)
 441         result = self._maybe_mask_results(result)
 442         dtype = DatetimeTZDtype(tz=other.tz) if other.tz else DT64NS_DTYPE
 443         return DatetimeArray(result, dtype=dtype, freq=self.freq)

 457     @unpack_zerodim_and_defer("__mul__")
 458     def __mul__(self, other):
 459         if is_scalar(other):
 460             # numpy will accept float and int, raise TypeError for others
 461             result = self._data * other
 462             freq = None
 463             if self.freq is not None and not isna(other):
 464                 freq = self.freq * other
 465             return type(self)(result, freq=freq)
 466 
 467         if not hasattr(other, "dtype"):
 468             # list, tuple
 469             other = np.array(other)
 470         if len(other) != len(self) and not is_timedelta64_dtype(other.dtype):
 471             # Exclude timedelta64 here so we correctly raise TypeError
 472             #  for that instead of ValueError
 473             raise ValueError("Cannot multiply with unequal lengths")
 474 
 475         if is_object_dtype(other.dtype):
 476             # this multiplication will succeed only if all elements of other
 477             #  are int or float scalars, so we will end up with
 478             #  timedelta64[ns]-dtyped result
 479             result = [self[n] * other[n] for n in range(len(self))]
 480             result = np.array(result)
 481             return type(self)(result)
 482 
 483         # numpy will accept float or int dtype, raise TypeError for others
 484         result = self._data * other
 485         return type(self)(result)

 489     @unpack_zerodim_and_defer("__truediv__")
 490     def __truediv__(self, other):
 491         # timedelta / X is well-defined for timedelta-like or numeric X
 492 
 493         if isinstance(other, (timedelta, np.timedelta64, Tick)):
 494             other = Timedelta(other)
 495             if other is NaT:
 496                 # specifically timedelta64-NaT
 497                 result = np.empty(self.shape, dtype=np.float64)
 498                 result.fill(np.nan)
 499                 return result
 500 
 501             # otherwise, dispatch to Timedelta implementation
 502             return self._data / other
 503 
 504         elif lib.is_scalar(other):
 505             # assume it is numeric
 506             result = self._data / other
 507             freq = None
 508             if self.freq is not None:
 509                 # Tick division is not implemented, so operate on Timedelta
 510                 freq = self.freq.delta / other
 511             return type(self)(result, freq=freq)
 512 
 513         if not hasattr(other, "dtype"):
 514             # e.g. list, tuple
 515             other = np.array(other)
 516 
 517         if len(other) != len(self):
 518             raise ValueError("Cannot divide vectors with unequal lengths")
 519 
 520         elif is_timedelta64_dtype(other.dtype):
 521             # let numpy handle it
 522             return self._data / other
 523 
 524         elif is_object_dtype(other.dtype):
 525             # We operate on raveled arrays to avoid problems in inference
 526             #  on NaT
 527             srav = self.ravel()
 528             orav = other.ravel()
 529             result = [srav[n] / orav[n] for n in range(len(srav))]
 530             result = np.array(result).reshape(self.shape)
 531 
 532             # We need to do dtype inference in order to keep DataFrame ops
 533             #  behavior consistent with Series behavior
 534             inferred = lib.infer_dtype(result)
 535             if inferred == "timedelta":
 536                 flat = result.ravel()
 537                 result = type(self)._from_sequence(flat).reshape(result.shape)
 538             elif inferred == "floating":
 539                 result = result.astype(float)
 540 
 541             return result
 542 
 543         else:
 544             result = self._data / other
 545             return type(self)(result)
 546 
 547     @unpack_zerodim_and_defer("__rtruediv__")
 548     def __rtruediv__(self, other):
 549         # X / timedelta is defined only for timedelta-like X
 550         if isinstance(other, (timedelta, np.timedelta64, Tick)):
 551             other = Timedelta(other)
 552             if other is NaT:
 553                 # specifically timedelta64-NaT
 554                 result = np.empty(self.shape, dtype=np.float64)
 555                 result.fill(np.nan)
 556                 return result
 557 
 558             # otherwise, dispatch to Timedelta implementation
 559             return other / self._data
 560 
 561         elif lib.is_scalar(other):
 562             raise TypeError(
 563                 f"Cannot divide {type(other).__name__} by {type(self).__name__}"
 564             )
 565 
 566         if not hasattr(other, "dtype"):
 567             # e.g. list, tuple
 568             other = np.array(other)
 569 
 570         if len(other) != len(self):
 571             raise ValueError("Cannot divide vectors with unequal lengths")
 572 
 573         elif is_timedelta64_dtype(other.dtype):
 574             # let numpy handle it
 575             return other / self._data
 576 
 577         elif is_object_dtype(other.dtype):
 578             # Note: unlike in __truediv__, we do not _need_ to do type
 579             #  inference on the result.  It does not raise, a numeric array
 580             #  is returned.  GH#23829
 581             result = [other[n] / self[n] for n in range(len(self))]
 582             return np.array(result)
 583 
 584         else:
 585             raise TypeError(
 586                 f"Cannot divide {other.dtype} data by {type(self).__name__}"
 587             )

 589     @unpack_zerodim_and_defer("__floordiv__")
 590     def __floordiv__(self, other):
 591 
 592         if is_scalar(other):
 593             if isinstance(other, (timedelta, np.timedelta64, Tick)):
 594                 other = Timedelta(other)
 595                 if other is NaT:
 596                     # treat this specifically as timedelta-NaT
 597                     result = np.empty(self.shape, dtype=np.float64)
 598                     result.fill(np.nan)
 599                     return result
 600 
 601                 # dispatch to Timedelta implementation
 602                 result = other.__rfloordiv__(self._data)
 603                 return result
 604 
 605             # at this point we should only have numeric scalars; anything
 606             #  else will raise
 607             result = self.asi8 // other
 608             result[self._isnan] = iNaT
 609             freq = None
 610             if self.freq is not None:
 611                 # Note: freq gets division, not floor-division
 612                 freq = self.freq / other
 613                 if freq.nanos == 0 and self.freq.nanos != 0:
 614                     # e.g. if self.freq is Nano(1) then dividing by 2
 615                     #  rounds down to zero
 616                     freq = None
 617             return type(self)(result.view("m8[ns]"), freq=freq)
 618 
 619         if not hasattr(other, "dtype"):
 620             # list, tuple
 621             other = np.array(other)
 622         if len(other) != len(self):
 623             raise ValueError("Cannot divide with unequal lengths")
 624 
 625         elif is_timedelta64_dtype(other.dtype):
 626             other = type(self)(other)
 627 
 628             # numpy timedelta64 does not natively support floordiv, so operate
 629             #  on the i8 values
 630             result = self.asi8 // other.asi8
 631             mask = self._isnan | other._isnan
 632             if mask.any():
 633                 result = result.astype(np.int64)
 634                 result[mask] = np.nan
 635             return result
 636 
 637         elif is_object_dtype(other.dtype):
 638             result = [self[n] // other[n] for n in range(len(self))]
 639             result = np.array(result)
 640             if lib.infer_dtype(result, skipna=False) == "timedelta":
 641                 result, _ = sequence_to_td64ns(result)
 642                 return type(self)(result)
 643             return result
 644 
 645         elif is_integer_dtype(other.dtype) or is_float_dtype(other.dtype):
 646             result = self._data // other
 647             return type(self)(result)
 648 
 649         else:
 650             dtype = getattr(other, "dtype", type(other).__name__)
 651             raise TypeError(f"Cannot divide {dtype} by {type(self).__name__}")
 652 
 653     @unpack_zerodim_and_defer("__rfloordiv__")
 654     def __rfloordiv__(self, other):
 655 
 656         if is_scalar(other):
 657             if isinstance(other, (timedelta, np.timedelta64, Tick)):
 658                 other = Timedelta(other)
 659                 if other is NaT:
 660                     # treat this specifically as timedelta-NaT
 661                     result = np.empty(self.shape, dtype=np.float64)
 662                     result.fill(np.nan)
 663                     return result
 664 
 665                 # dispatch to Timedelta implementation
 666                 result = other.__floordiv__(self._data)
 667                 return result
 668 
 669             raise TypeError(
 670                 f"Cannot divide {type(other).__name__} by {type(self).__name__}"
 671             )
 672 
 673         if not hasattr(other, "dtype"):
 674             # list, tuple
 675             other = np.array(other)
 676 
 677         if len(other) != len(self):
 678             raise ValueError("Cannot divide with unequal lengths")
 679 
 680         elif is_timedelta64_dtype(other.dtype):
 681             other = type(self)(other)
 682 
 683             # numpy timedelta64 does not natively support floordiv, so operate
 684             #  on the i8 values
 685             result = other.asi8 // self.asi8
 686             mask = self._isnan | other._isnan
 687             if mask.any():
 688                 result = result.astype(np.int64)
 689                 result[mask] = np.nan
 690             return result
 691 
 692         elif is_object_dtype(other.dtype):
 693             result = [other[n] // self[n] for n in range(len(self))]
 694             result = np.array(result)
 695             return result
 696 
 697         else:
 698             dtype = getattr(other, "dtype", type(other).__name__)
 699             raise TypeError(f"Cannot divide {dtype} by {type(self).__name__}")
 700 
 701     @unpack_zerodim_and_defer("__mod__")
 702     def __mod__(self, other):
 703         # Note: This is a naive implementation, can likely be optimized
 704         if isinstance(other, (timedelta, np.timedelta64, Tick)):
 705             other = Timedelta(other)
 706         return self - (self // other) * other
 707 
 708     @unpack_zerodim_and_defer("__rmod__")
 709     def __rmod__(self, other):
 710         # Note: This is a naive implementation, can likely be optimized
 711         if isinstance(other, (timedelta, np.timedelta64, Tick)):
 712             other = Timedelta(other)
 713         return other - (other // self) * self
 714 
 715     @unpack_zerodim_and_defer("__divmod__")
 716     def __divmod__(self, other):
 717         # Note: This is a naive implementation, can likely be optimized
 718         if isinstance(other, (timedelta, np.timedelta64, Tick)):
 719             other = Timedelta(other)
 720 
 721         res1 = self // other
 722         res2 = self - res1 * other
 723         return res1, res2
 724 
 725     @unpack_zerodim_and_defer("__rdivmod__")
 726     def __rdivmod__(self, other):
 727         # Note: This is a naive implementation, can likely be optimized
 728         if isinstance(other, (timedelta, np.timedelta64, Tick)):
 729             other = Timedelta(other)
 730 
 731         res1 = other // self
 732         res2 = other - res1 * self
 733         return res1, res2

 735     def __neg__(self):
 736         if self.freq is not None:
 737             return type(self)(-self._data, freq=-self.freq)
 738         return type(self)(-self._data)

 741         return type(self)(self._data, freq=self.freq)

 744         # Note: freq is not preserved

 750     def total_seconds(self):
 751         """
 752         Return total duration of each element expressed in seconds.
 753 
 754         This method is available directly on TimedeltaArray, TimedeltaIndex
 755         and on Series containing timedelta values under the ``.dt`` namespace.
 756 
 757         Returns
 758         -------
 759         seconds : [ndarray, Float64Index, Series]
 760             When the calling object is a TimedeltaArray, the return type
 761             is ndarray.  When the calling object is a TimedeltaIndex,
 762             the return type is a Float64Index. When the calling object
 763             is a Series, the return type is Series of type `float64` whose
 764             index is the same as the original.
 765 
 766         See Also
 767         --------
 768         datetime.timedelta.total_seconds : Standard library version
 769             of this method.
 770         TimedeltaIndex.components : Return a DataFrame with components of
 771             each Timedelta.
 772 
 773         Examples
 774         --------
 775         **Series**
 776 
 777         >>> s = pd.Series(pd.to_timedelta(np.arange(5), unit='d'))
 778         >>> s
 779         0   0 days
 780         1   1 days
 781         2   2 days
 782         3   3 days
 783         4   4 days
 784         dtype: timedelta64[ns]
 785 
 786         >>> s.dt.total_seconds()
 787         0         0.0
 788         1     86400.0
 789         2    172800.0
 790         3    259200.0
 791         4    345600.0
 792         dtype: float64
 793 
 794         **TimedeltaIndex**
 795 
 796         >>> idx = pd.to_timedelta(np.arange(5), unit='d')
 797         >>> idx
 798         TimedeltaIndex(['0 days', '1 days', '2 days', '3 days', '4 days'],
 799                        dtype='timedelta64[ns]', freq=None)
 800 
 801         >>> idx.total_seconds()
 802         Float64Index([0.0, 86400.0, 172800.0, 259200.00000000003, 345600.0],
 803                      dtype='float64')
 804         """
 805         return self._maybe_mask_results(1e-9 * self.asi8, fill_value=None)

 839         seconds, milliseconds, microseconds, nanoseconds) of the Timedeltas.
 840 
 841         Returns
 842         -------
 843         a DataFrame
 844         """
 845         from pandas import DataFrame
 846 
 847         columns = [
 848             "days",
 849             "hours",
 850             "minutes",
 851             "seconds",
 852             "milliseconds",
 853             "microseconds",
 854             "nanoseconds",
 855         ]
 856         hasnans = self._hasnans
 857         if hasnans:
 858 
 859             def f(x):
 860                 if isna(x):
 861                     return [np.nan] * len(columns)
 862                 return x.components
 863 
 864         else:
 865 
 866             def f(x):
 867                 return x.components
 868

 870         if not hasnans:
 871             result = result.astype("int64")

 879 def sequence_to_td64ns(data, copy=False, unit=None, errors="raise"):
 880     """
 881     Parameters
 882     ----------
 883     data : list-like
 884     copy : bool, default False
 885     unit : str, default "ns"
 886         The timedelta unit to treat integers as multiples of.
 887         Must be un-specifed if the data contains a str.
 888     errors : {"raise", "coerce", "ignore"}, default "raise"
 889         How to handle elements that cannot be converted to timedelta64[ns].
 890         See ``pandas.to_timedelta`` for details.
 891 
 892     Returns
 893     -------
 894     converted : numpy.ndarray
 895         The sequence converted to a numpy array with dtype ``timedelta64[ns]``.
 896     inferred_freq : Tick or None
 897         The inferred frequency of the sequence.
 898 
 899     Raises
 900     ------
 901     ValueError : Data cannot be converted to timedelta64[ns].
 902 
 903     Notes
 904     -----
 905     Unlike `pandas.to_timedelta`, if setting ``errors=ignore`` will not cause
 906     errors to be ignored; they are caught and subsequently ignored at a
 907     higher level.
 908     """
 909     inferred_freq = None
 910     if unit is not None:
 911         unit = parse_timedelta_unit(unit)
 912 
 913     # Unwrap whatever we have into a np.ndarray
 914     if not hasattr(data, "dtype"):
 915         # e.g. list, tuple
 916         if np.ndim(data) == 0:
 917             # i.e. generator
 918             data = list(data)
 919         data = np.array(data, copy=False)
 920     elif isinstance(data, ABCSeries):
 921         data = data._values
 922     elif isinstance(data, (ABCTimedeltaIndex, TimedeltaArray)):
 923         inferred_freq = data.freq
 924         data = data._data
 925 
 926     # Convert whatever we have into timedelta64[ns] dtype
 927     if is_object_dtype(data.dtype) or is_string_dtype(data.dtype):
 928         # no need to make a copy, need to convert if string-dtyped
 929         data = objects_to_td64ns(data, unit=unit, errors=errors)
 930         copy = False
 931 
 932     elif is_integer_dtype(data.dtype):
 933         # treat as multiples of the given unit
 934         data, copy_made = ints_to_td64ns(data, unit=unit)
 935         copy = copy and not copy_made
 936 
 937     elif is_float_dtype(data.dtype):
 938         # cast the unit, multiply base/frac separately
 939         # to avoid precision issues from float -> int
 940         mask = np.isnan(data)
 941         m, p = precision_from_unit(unit or "ns")
 942         base = data.astype(np.int64)
 943         frac = data - base
 944         if p:
 945             frac = np.round(frac, p)
 946         data = (base * m + (frac * m).astype(np.int64)).view("timedelta64[ns]")
 947         data[mask] = iNaT
 948         copy = False
 949 
 950     elif is_timedelta64_dtype(data.dtype):
 951         if data.dtype != TD64NS_DTYPE:
 952             # non-nano unit
 953             # TODO: watch out for overflows
 954             data = data.astype(TD64NS_DTYPE)
 955             copy = False
 956 
 957     else:
 958         # This includes datetime64-dtype, see GH#23539, GH#29794
 959         raise TypeError(f"dtype {data.dtype} cannot be converted to timedelta64[ns]")
 960 
 961     data = np.array(data, copy=copy)
 962 
 963     assert data.dtype == "m8[ns]", data
 964     return data, inferred_freq

 967 def ints_to_td64ns(data, unit="ns"):
 968     """
 969     Convert an ndarray with integer-dtype to timedelta64[ns] dtype, treating
 970     the integers as multiples of the given timedelta unit.
 971 
 972     Parameters
 973     ----------
 974     data : numpy.ndarray with integer-dtype
 975     unit : str, default "ns"
 976         The timedelta unit to treat integers as multiples of.
 977 
 978     Returns
 979     -------
 980     numpy.ndarray : timedelta64[ns] array converted from data
 981     bool : whether a copy was made
 982     """
 983     copy_made = False
 984     unit = unit if unit is not None else "ns"
 985 
 986     if data.dtype != np.int64:
 987         # converting to int64 makes a copy, so we can avoid
 988         # re-copying later
 989         data = data.astype(np.int64)
 990         copy_made = True
 991 
 992     if unit != "ns":
 993         dtype_str = f"timedelta64[{unit}]"
 994         data = data.view(dtype_str)
 995 
 996         # TODO: watch out for overflows when converting from lower-resolution
 997         data = data.astype("timedelta64[ns]")
 998         # the astype conversion makes a copy, so we can avoid re-copying later
 999         copy_made = True
1000 
1001     else:
1002         data = data.view("timedelta64[ns]")
1003 
1004     return data, copy_made
