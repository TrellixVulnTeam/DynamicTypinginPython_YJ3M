  28     """
  29     Extension dtype for string data.
  30 
  31     .. versionadded:: 1.0.0
  32 
  33     .. warning::

  35        StringDtype is considered experimental. The implementation and
  36        parts of the API may change without warning.
  37 
  38        In particular, StringDtype.na_value may change to no longer be
  39        ``numpy.nan``.
  40 
  41     Attributes
  42     ----------
  43     None
  44 
  45     Methods
  46     -------
  47     None
  48 
  49     Examples
  50     --------
  51     >>> pd.StringDtype()
  52     StringDtype
  53     """

  55     name = "string"
  56 
  57     #: StringDtype.na_value uses pandas.NA
  58     na_value = libmissing.NA

  60     @property
  61     def type(self) -> Type[str]:
  62         return str
  63 
  64     @classmethod
  65     def construct_array_type(cls) -> Type["StringArray"]:
  66         """
  67         Return the array type associated with this dtype.
  68 
  69         Returns
  70         -------
  71         type

  73         return StringArray

  78     def __from_arrow__(
  79         self, array: Union["pyarrow.Array", "pyarrow.ChunkedArray"]
  80     ) -> "StringArray":
  81         """
  82         Construct StringArray from pyarrow Array/ChunkedArray.
  83         """
  84         import pyarrow  # noqa: F811
  85 
  86         if isinstance(array, pyarrow.Array):
  87             chunks = [array]
  88         else:
  89             # pyarrow.ChunkedArray
  90             chunks = array.chunks
  91 
  92         results = []
  93         for arr in chunks:
  94             # using _from_sequence to ensure None is converted to NA
  95             str_arr = StringArray._from_sequence(np.array(arr))
  96             results.append(str_arr)
  97 
  98         return StringArray._concat_same_type(results)

 101 class StringArray(PandasArray):
 102     """
 103     Extension array for string data.
 104 
 105     .. versionadded:: 1.0.0
 106 
 107     .. warning::
 108 
 109        StringArray is considered experimental. The implementation and
 110        parts of the API may change without warning.
 111 
 112     Parameters
 113     ----------
 114     values : array-like
 115         The array of data.
 116 
 117         .. warning::
 118 
 119            Currently, this expects an object-dtype ndarray
 120            where the elements are Python strings or :attr:`pandas.NA`.
 121            This may change without warning in the future. Use
 122            :meth:`pandas.array` with ``dtype="string"`` for a stable way of
 123            creating a `StringArray` from any sequence.
 124 
 125     copy : bool, default False
 126         Whether to copy the array of data.
 127 
 128     Attributes
 129     ----------
 130     None
 131 
 132     Methods
 133     -------
 134     None
 135 
 136     See Also
 137     --------
 138     array
 139         The recommended function for creating a StringArray.
 140     Series.str
 141         The string methods are available on Series backed by
 142         a StringArray.
 143 
 144     Notes
 145     -----
 146     StringArray returns a BooleanArray for comparison methods.
 147 
 148     Examples
 149     --------
 150     >>> pd.array(['This is', 'some text', None, 'data.'], dtype="string")
 151     <StringArray>
 152     ['This is', 'some text', <NA>, 'data.']
 153     Length: 4, dtype: string
 154 
 155     Unlike arrays instantiated with ``dtype="object"``, ``StringArray``
 156     will convert the values to strings.
 157 
 158     >>> pd.array(['1', 1], dtype="object")
 159     <PandasArray>
 160     ['1', 1]
 161     Length: 2, dtype: object
 162     >>> pd.array(['1', 1], dtype="string")
 163     <StringArray>
 164     ['1', '1']
 165     Length: 2, dtype: string
 166 
 167     However, instantiating StringArrays directly with non-strings will raise an error.
 168 
 169     For comparison methods, `StringArray` returns a :class:`pandas.BooleanArray`:
 170 
 171     >>> pd.array(["a", None, "c"], dtype="string") == "a"
 172     <BooleanArray>
 173     [True, <NA>, False]
 174     Length: 3, dtype: boolean
 175     """
 176 
 177     # undo the PandasArray hack
 178     _typ = "extension"
 179 
 180     def __init__(self, values, copy=False):
 181         values = extract_array(values)
 182         skip_validation = isinstance(values, type(self))
 183 
 184         super().__init__(values, copy=copy)
 185         self._dtype = StringDtype()
 186         if not skip_validation:
 187             self._validate()
 188 
 189     def _validate(self):
 190         """Validate that we only store NA or strings."""
 191         if len(self._ndarray) and not lib.is_string_array(self._ndarray, skipna=True):
 192             raise ValueError("StringArray requires a sequence of strings or pandas.NA")
 193         if self._ndarray.dtype != "object":
 194             raise ValueError(
 195                 "StringArray requires a sequence of strings or pandas.NA. Got "
 196                 f"'{self._ndarray.dtype}' dtype instead."
 197             )
 198 
 199     @classmethod
 200     def _from_sequence(cls, scalars, dtype=None, copy=False):
 201         if dtype:
 202             assert dtype == "string"
 203 
 204         result = np.asarray(scalars, dtype="object")
 205         if copy and result is scalars:
 206             result = result.copy()
 207 
 208         # Standardize all missing-like values to NA
 209         # TODO: it would be nice to do this in _validate / lib.is_string_array
 210         # We are already doing a scan over the values there.
 211         na_values = isna(result)
 212         has_nans = na_values.any()
 213         if has_nans and result is scalars:
 214             # force a copy now, if we haven't already
 215             result = result.copy()
 216 
 217         # convert to str, then to object to avoid dtype like '<U3', then insert na_value
 218         result = np.asarray(result, dtype=str)
 219         result = np.asarray(result, dtype="object")
 220         if has_nans:
 221             result[na_values] = StringDtype.na_value
 222 
 223         return cls(result)
 224 
 225     @classmethod
 226     def _from_sequence_of_strings(cls, strings, dtype=None, copy=False):
 227         return cls._from_sequence(strings, dtype=dtype, copy=copy)
 228 
 229     def __arrow_array__(self, type=None):
 230         """
 231         Convert myself into a pyarrow Array.
 232         """
 233         import pyarrow as pa
 234 
 235         if type is None:
 236             type = pa.string()
 237 
 238         values = self._ndarray.copy()
 239         values[self.isna()] = None
 240         return pa.array(values, type=type, from_pandas=True)
 241 
 242     def _values_for_factorize(self):
 243         arr = self._ndarray.copy()
 244         mask = self.isna()
 245         arr[mask] = -1
 246         return arr, -1
 247 
 248     def __setitem__(self, key, value):
 249         value = extract_array(value, extract_numpy=True)
 250         if isinstance(value, type(self)):
 251             # extract_array doesn't extract PandasArray subclasses
 252             value = value._ndarray
 253 
 254         key = check_array_indexer(self, key)
 255         scalar_key = lib.is_scalar(key)
 256         scalar_value = lib.is_scalar(value)
 257         if scalar_key and not scalar_value:
 258             raise ValueError("setting an array element with a sequence.")
 259 
 260         # validate new items
 261         if scalar_value:
 262             if isna(value):
 263                 value = StringDtype.na_value
 264             elif not isinstance(value, str):
 265                 raise ValueError(
 266                     f"Cannot set non-string value '{value}' into a StringArray."
 267                 )
 268         else:
 269             if not is_array_like(value):
 270                 value = np.asarray(value, dtype=object)
 271             if len(value) and not lib.is_string_array(value, skipna=True):
 272                 raise ValueError("Must provide strings.")
 273 
 274         super().__setitem__(key, value)
 275 
 276     def fillna(self, value=None, method=None, limit=None):
 277         # TODO: validate dtype
 278         return super().fillna(value, method, limit)
 279 
 280     def astype(self, dtype, copy=True):
 281         dtype = pandas_dtype(dtype)
 282         if isinstance(dtype, StringDtype):
 283             if copy:
 284                 return self.copy()
 285             return self
 286         elif isinstance(dtype, _IntegerDtype):
 287             arr = self._ndarray.copy()
 288             mask = self.isna()
 289             arr[mask] = 0
 290             values = arr.astype(dtype.numpy_dtype)
 291             return IntegerArray(values, mask, copy=False)
 292 
 293         return super().astype(dtype, copy)
 294 
 295     def _reduce(self, name, skipna=True, **kwargs):
 296         if name in ["min", "max"]:
 297             return getattr(self, name)(skipna=skipna)
 298 
 299         raise TypeError(f"Cannot perform reduction '{name}' with string dtype")
 300 
 301     def value_counts(self, dropna=False):
 302         from pandas import value_counts
 303 
 304         return value_counts(self._ndarray, dropna=dropna).astype("Int64")
 305 
 306     def memory_usage(self, deep=False):
 307         result = self._ndarray.nbytes
 308         if deep:
 309             return result + lib.memory_usage_of_objects(self._ndarray)
 310         return result

 312     # Override parent because we have different return types.
 313     @classmethod
 314     def _create_arithmetic_method(cls, op):
 315         # Note: this handles both arithmetic and comparison methods.
 316         def method(self, other):
 317             from pandas.arrays import BooleanArray
 318 
 319             assert op.__name__ in ops.ARITHMETIC_BINOPS | ops.COMPARISON_BINOPS
 320 
 321             if isinstance(other, (ABCIndexClass, ABCSeries, ABCDataFrame)):
 322                 return NotImplemented
 323 
 324             elif isinstance(other, cls):
 325                 other = other._ndarray
 326 
 327             mask = isna(self) | isna(other)
 328             valid = ~mask
 329 
 330             if not lib.is_scalar(other):
 331                 if len(other) != len(self):
 332                     # prevent improper broadcasting when other is 2D
 333                     raise ValueError(
 334                         f"Lengths of operands do not match: {len(self)} != {len(other)}"
 335                     )
 336 
 337                 other = np.asarray(other)
 338                 other = other[valid]
 339 
 340             if op.__name__ in ops.ARITHMETIC_BINOPS:
 341                 result = np.empty_like(self._ndarray, dtype="object")
 342                 result[mask] = StringDtype.na_value
 343                 result[valid] = op(self._ndarray[valid], other)
 344                 return StringArray(result)
 345             else:
 346                 # logical
 347                 result = np.zeros(len(self._ndarray), dtype="bool")
 348                 result[valid] = op(self._ndarray[valid], other)
 349                 return BooleanArray(result, mask)
 350 
 351         return compat.set_function_name(method, f"__{op.__name__}__", cls)
 352 
 353     @classmethod
 354     def _add_arithmetic_ops(cls):
 355         cls.__add__ = cls._create_arithmetic_method(operator.add)
 356         cls.__radd__ = cls._create_arithmetic_method(ops.radd)
 357 
 358         cls.__mul__ = cls._create_arithmetic_method(operator.mul)
 359         cls.__rmul__ = cls._create_arithmetic_method(ops.rmul)
 360 
 361     _create_comparison_method = _create_arithmetic_method
