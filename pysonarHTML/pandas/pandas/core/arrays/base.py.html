  36 class ExtensionArray:
  37     """
  38     Abstract base class for custom 1-D array types.
  39 
  40     pandas will recognize instances of this class as proper arrays
  41     with a custom type and will not attempt to coerce them to objects. They
  42     may be stored directly inside a :class:`DataFrame` or :class:`Series`.
  43 
  44     .. versionadded:: 0.23.0
  45 
  46     Attributes
  47     ----------
  48     dtype
  49     nbytes
  50     ndim
  51     shape
  52 
  53     Methods
  54     -------
  55     argsort
  56     astype
  57     copy
  58     dropna
  59     factorize
  60     fillna
  61     equals
  62     isna
  63     ravel
  64     repeat
  65     searchsorted
  66     shift
  67     take
  68     unique
  69     view
  70     _concat_same_type
  71     _formatter
  72     _from_factorized
  73     _from_sequence
  74     _from_sequence_of_strings
  75     _reduce
  76     _values_for_argsort
  77     _values_for_factorize
  78 
  79     Notes
  80     -----
  81     The interface includes the following abstract methods that must be
  82     implemented by subclasses:
  83 
  84     * _from_sequence
  85     * _from_factorized
  86     * __getitem__
  87     * __len__
  88     * __eq__
  89     * dtype
  90     * nbytes
  91     * isna
  92     * take
  93     * copy
  94     * _concat_same_type
  95 
  96     A default repr displaying the type, (truncated) data, length,
  97     and dtype is provided. It can be customized or replaced by
  98     by overriding:
  99 
 100     * __repr__ : A default repr for the ExtensionArray.
 101     * _formatter : Print scalars inside a Series or DataFrame.
 102 
 103     Some methods require casting the ExtensionArray to an ndarray of Python
 104     objects with ``self.astype(object)``, which may be expensive. When
 105     performance is a concern, we highly recommend overriding the following
 106     methods:
 107 
 108     * fillna
 109     * dropna
 110     * unique
 111     * factorize / _values_for_factorize
 112     * argsort / _values_for_argsort
 113     * searchsorted
 114 
 115     The remaining methods implemented on this class should be performant,
 116     as they only compose abstract methods. Still, a more efficient
 117     implementation may be available, and these methods can be overridden.
 118 
 119     One can implement methods to handle array reductions.
 120 
 121     * _reduce
 122 
 123     One can implement methods to handle parsing from strings that will be used
 124     in methods such as ``pandas.io.parsers.read_csv``.
 125 
 126     * _from_sequence_of_strings
 127 
 128     This class does not inherit from 'abc.ABCMeta' for performance reasons.
 129     Methods and properties required by the interface raise
 130     ``pandas.errors.AbstractMethodError`` and no ``register`` method is
 131     provided for registering virtual subclasses.
 132 
 133     ExtensionArrays are limited to 1 dimension.
 134 
 135     They may be backed by none, one, or many NumPy arrays. For example,
 136     ``pandas.Categorical`` is an extension array backed by two arrays,
 137     one for codes and one for categories. An array of IPv6 address may
 138     be backed by a NumPy structured array with two fields, one for the
 139     lower 64 bits and one for the upper 64 bits. Or they may be backed
 140     by some other storage type, like Python lists. Pandas makes no
 141     assumptions on how the data are stored, just that it can be converted
 142     to a NumPy array.
 143     The ExtensionArray interface does not impose any rules on how this data
 144     is stored. However, currently, the backing data cannot be stored in
 145     attributes called ``.values`` or ``._values`` to ensure full compatibility
 146     with pandas internals. But other names as ``.data``, ``._data``,
 147     ``._items``, ... can be freely used.
 148 
 149     If implementing NumPy's ``__array_ufunc__`` interface, pandas expects
 150     that
 151 
 152     1. You defer by returning ``NotImplemented`` when any Series are present
 153        in `inputs`. Pandas will extract the arrays and call the ufunc again.
 154     2. You define a ``_HANDLED_TYPES`` tuple as an attribute on the class.
 155        Pandas inspect this to determine whether the ufunc is valid for the
 156        types present.
 157 
 158     See :ref:`extending.extension.ufunc` for more.
 159 
 160     By default, ExtensionArrays are not hashable.  Immutable subclasses may
 161     override this behavior.
 162     """

 273     def __setitem__(self, key: Union[int, np.ndarray], value: Any) -> None:
 274         """
 275         Set one or more values inplace.
 276 
 277         This method is not required to satisfy the pandas extension array
 278         interface.
 279 
 280         Parameters
 281         ----------
 282         key : int, ndarray, or slice
 283             When called from, e.g. ``Series.__setitem__``, ``key`` will be
 284             one of
 285 
 286             * scalar int
 287             * ndarray of integers.
 288             * boolean ndarray
 289             * slice object
 290 
 291         value : ExtensionDtype.type, Sequence[ExtensionDtype.type], or object
 292             value or values to be set of ``key``.
 293 
 294         Returns
 295         -------
 296         None
 297         """
 298         # Some notes to the ExtensionArray implementor who may have ended up
 299         # here. While this method is not required for the interface, if you
 300         # *do* choose to implement __setitem__, then some semantics should be
 301         # observed:
 302         #
 303         # * Setting multiple values : ExtensionArrays should support setting
 304         #   multiple values at once, 'key' will be a sequence of integers and
 305         #  'value' will be a same-length sequence.
 306         #
 307         # * Broadcasting : For a sequence 'key' and a scalar 'value',
 308         #   each position in 'key' should be set to 'value'.
 309         #
 310         # * Coercion : Most users will expect basic coercion to work. For
 311         #   example, a string like '2018-01-01' is coerced to a datetime
 312         #   when setting on a datetime64ns array. In general, if the
 313         #   __init__ method coerces that value, then so should __setitem__
 314         # Note, also, that Series/DataFrame.where internally use __setitem__
 315         # on a copy of the data.
 316         raise NotImplementedError(f"{type(self)} does not implement __setitem__.")

 356     def to_numpy(
 357         self, dtype=None, copy: bool = False, na_value=lib.no_default
 358     ) -> np.ndarray:
 359         """
 360         Convert to a NumPy ndarray.
 361 
 362         .. versionadded:: 1.0.0
 363 
 364         This is similar to :meth:`numpy.asarray`, but may provide additional control
 365         over how the conversion is done.
 366 
 367         Parameters
 368         ----------
 369         dtype : str or numpy.dtype, optional
 370             The dtype to pass to :meth:`numpy.asarray`.
 371         copy : bool, default False
 372             Whether to ensure that the returned value is a not a view on
 373             another array. Note that ``copy=False`` does not *ensure* that
 374             ``to_numpy()`` is no-copy. Rather, ``copy=True`` ensure that
 375             a copy is made, even if not strictly necessary.
 376         na_value : Any, optional
 377             The value to use for missing values. The default value depends
 378             on `dtype` and the type of the array.
 379 
 380         Returns
 381         -------
 382         numpy.ndarray
 383         """
 384         result = np.asarray(self, dtype=dtype)
 385         if copy or na_value is not lib.no_default:
 386             result = result.copy()
 387         if na_value is not lib.no_default:
 388             result[self.isna()] = na_value
 389         return result

 536     def fillna(self, value=None, method=None, limit=None):
 537         """
 538         Fill NA/NaN values using the specified method.
 539 
 540         Parameters
 541         ----------
 542         value : scalar, array-like
 543             If a scalar value is passed it is used to fill all missing values.
 544             Alternatively, an array-like 'value' can be given. It's expected
 545             that the array-like have the same length as 'self'.
 546         method : {'backfill', 'bfill', 'pad', 'ffill', None}, default None
 547             Method to use for filling holes in reindexed Series
 548             pad / ffill: propagate last valid observation forward to next valid
 549             backfill / bfill: use NEXT valid observation to fill gap.
 550         limit : int, default None
 551             If method is specified, this is the maximum number of consecutive
 552             NaN values to forward/backward fill. In other words, if there is
 553             a gap with more than this number of consecutive NaNs, it will only
 554             be partially filled. If method is not specified, this is the
 555             maximum number of entries along the entire axis where NaNs will be
 556             filled.
 557 
 558         Returns
 559         -------
 560         ExtensionArray
 561             With NA/NaN filled.
 562         """
 563         value, method = validate_fillna_kwargs(value, method)
 564 
 565         mask = self.isna()
 566 
 567         if is_array_like(value):
 568             if len(value) != len(self):
 569                 raise ValueError(
 570                     f"Length of 'value' does not match. Got ({len(value)}) "
 571                     f"expected {len(self)}"
 572                 )
 573             value = value[mask]
 574 
 575         if mask.any():
 576             if method is not None:
 577                 func = pad_1d if method == "pad" else backfill_1d
 578                 new_values = func(self.astype(object), limit=limit, mask=mask)
 579                 new_values = self._from_sequence(new_values, dtype=self.dtype)
 580             else:
 581                 # fill with value
 582                 new_values = self.copy()
 583                 new_values[mask] = value
 584         else:
 585             new_values = self.copy()
 586         return new_values

 598     def shift(self, periods: int = 1, fill_value: object = None) -> "ExtensionArray":
 599         """
 600         Shift values by desired number.
 601 
 602         Newly introduced missing values are filled with
 603         ``self.dtype.na_value``.
 604 
 605         .. versionadded:: 0.24.0
 606 
 607         Parameters
 608         ----------
 609         periods : int, default 1
 610             The number of periods to shift. Negative values are allowed
 611             for shifting backwards.
 612 
 613         fill_value : object, optional
 614             The scalar value to use for newly introduced missing values.
 615             The default is ``self.dtype.na_value``.
 616 
 617             .. versionadded:: 0.24.0
 618 
 619         Returns
 620         -------
 621         ExtensionArray
 622             Shifted.
 623 
 624         Notes
 625         -----
 626         If ``self`` is empty or ``periods`` is 0, a copy of ``self`` is
 627         returned.
 628 
 629         If ``periods > len(self)``, then an array of size
 630         len(self) is returned, with all values filled with
 631         ``self.dtype.na_value``.
 632         """
 633         # Note: this implementation assumes that `self.dtype.na_value` can be
 634         # stored in an instance of your ExtensionArray with `self.dtype`.
 635         if not len(self) or periods == 0:
 636             return self.copy()
 637 
 638         if isna(fill_value):
 639             fill_value = self.dtype.na_value
 640 
 641         empty = self._from_sequence(
 642             [fill_value] * min(abs(periods), len(self)), dtype=self.dtype
 643         )
 644         if periods > 0:
 645             a = empty
 646             b = self[:-periods]
 647         else:
 648             a = self[abs(periods) :]
 649             b = empty
 650         return self._concat_same_type([a, b])

 663     def searchsorted(self, value, side="left", sorter=None):
 664         """
 665         Find indices where elements should be inserted to maintain order.
 666 
 667         .. versionadded:: 0.24.0
 668 
 669         Find the indices into a sorted array `self` (a) such that, if the
 670         corresponding elements in `value` were inserted before the indices,
 671         the order of `self` would be preserved.
 672 
 673         Assuming that `self` is sorted:
 674 
 675         ======  ================================
 676         `side`  returned index `i` satisfies
 677         ======  ================================
 678         left    ``self[i-1] < value <= self[i]``
 679         right   ``self[i-1] <= value < self[i]``
 680         ======  ================================
 681 
 682         Parameters
 683         ----------
 684         value : array_like
 685             Values to insert into `self`.
 686         side : {'left', 'right'}, optional
 687             If 'left', the index of the first suitable location found is given.
 688             If 'right', return the last such index.  If there is no suitable
 689             index, return either 0 or N (where N is the length of `self`).
 690         sorter : 1-D array_like, optional
 691             Optional array of integer indices that sort array a into ascending
 692             order. They are typically the result of argsort.
 693 
 694         Returns
 695         -------
 696         array of ints
 697             Array of insertion points with the same shape as `value`.
 698 
 699         See Also
 700         --------
 701         numpy.searchsorted : Similar method from NumPy.
 702         """
 703         # Note: the base tests provided by pandas only test the basics.
 704         # We do not test
 705         # 1. Values outside the range of the `data_for_sorting` fixture
 706         # 2. Values between the values in the `data_for_sorting` fixture
 707         # 3. Missing values.
 708         arr = self.astype(object)
 709         return arr.searchsorted(value, side=side, sorter=sorter)

 711     def equals(self, other: "ExtensionArray") -> bool:
 712         """
 713         Return if another array is equivalent to this array.
 714 
 715         Equivalent means that both arrays have the same shape and dtype, and
 716         all values compare equal. Missing values in the same location are
 717         considered equal (in contrast with normal equality).
 718 
 719         Parameters
 720         ----------
 721         other : ExtensionArray
 722             Array to compare to this Array.
 723 
 724         Returns
 725         -------
 726         boolean
 727             Whether the arrays are equivalent.
 728         """
 729         if not type(self) == type(other):
 730             return False
 731         elif not self.dtype == other.dtype:
 732             return False
 733         elif not len(self) == len(other):
 734             return False
 735         else:
 736             equal_values = self == other
 737             if isinstance(equal_values, ExtensionArray):
 738                 # boolean array with NA -> fill with False
 739                 equal_values = equal_values.fillna(False)
 740             equal_na = self.isna() & other.isna()
 741             return bool((equal_values | equal_na).all())

 767     def factorize(self, na_sentinel: int = -1) -> Tuple[np.ndarray, "ExtensionArray"]:
 768         """
 769         Encode the extension array as an enumerated type.
 770 
 771         Parameters
 772         ----------
 773         na_sentinel : int, default -1
 774             Value to use in the `codes` array to indicate missing values.
 775 
 776         Returns
 777         -------
 778         codes : ndarray
 779             An integer NumPy array that's an indexer into the original
 780             ExtensionArray.
 781         uniques : ExtensionArray
 782             An ExtensionArray containing the unique values of `self`.
 783

 785

 787                the ExtensionArray if there are any missing values present
 788                in `self`.

 791         --------
 792         factorize : Top-level factorize method that dispatches here.
 793 
 794         Notes
 795         -----
 796         :meth:`pandas.factorize` offers a `sort` keyword as well.
 797         """
 798         # Implementer note: There are two ways to override the behavior of
 799         # pandas.factorize
 800         # 1. _values_for_factorize and _from_factorize.

 870     def take(
 871         self, indices: Sequence[int], allow_fill: bool = False, fill_value: Any = None
 872     ) -> "ExtensionArray":
 873         """
 874         Take elements from an array.
 875 
 876         Parameters
 877         ----------
 878         indices : sequence of int
 879             Indices to be taken.
 880         allow_fill : bool, default False
 881             How to handle negative values in `indices`.
 882 
 883             * False: negative values in `indices` indicate positional indices
 884               from the right (the default). This is similar to
 885               :func:`numpy.take`.
 886 
 887             * True: negative values in `indices` indicate
 888               missing values. These values are set to `fill_value`. Any other
 889               other negative values raise a ``ValueError``.
 890 
 891         fill_value : any, optional
 892             Fill value to use for NA-indices when `allow_fill` is True.
 893             This may be ``None``, in which case the default NA value for
 894             the type, ``self.dtype.na_value``, is used.
 895 
 896             For many ExtensionArrays, there will be two representations of
 897             `fill_value`: a user-facing "boxed" scalar, and a low-level
 898             physical NA value. `fill_value` should be the user-facing version,
 899             and the implementation should handle translating that to the
 900             physical version for processing the take if necessary.
 901 
 902         Returns
 903         -------
 904         ExtensionArray
 905 
 906         Raises
 907         ------
 908         IndexError
 909             When the indices are out of bounds for the array.
 910         ValueError
 911             When `indices` contains negative values other than ``-1``
 912             and `allow_fill` is True.
 913 
 914         See Also
 915         --------
 916         numpy.take
 917         api.extensions.take
 918 
 919         Notes
 920         -----
 921         ExtensionArray.take is called by ``Series.__getitem__``, ``.loc``,
 922         ``iloc``, when `indices` is a sequence of values. Additionally,
 923         it's called by :meth:`Series.reindex`, or any other method
 924         that causes realignment, with a `fill_value`.
 925 
 926         Examples
 927         --------
 928         Here's an example implementation, which relies on casting the
 929         extension array to object dtype. This uses the helper method
 930         :func:`pandas.api.extensions.take`.
 931 
 932         .. code-block:: python
 933 
 934            def take(self, indices, allow_fill=False, fill_value=None):
 935                from pandas.core.algorithms import take
 936 
 937                # If the ExtensionArray is backed by an ndarray, then
 938                # just pass that here instead of coercing to object.
 939                data = self.astype(object)
 940 
 941                if allow_fill and fill_value is None:
 942                    fill_value = self.dtype.na_value
 943 
 944                # fill value should always be translated from the scalar
 945                # type for the array, to the physical storage type for
 946                # the data, before passing to take.
 947 
 948                result = take(data, indices, fill_value=fill_value,
 949                              allow_fill=allow_fill)
 950                return self._from_sequence(result, dtype=self.dtype)
 951         """
 952         # Implementer note: The `fill_value` parameter should be a user-facing
 953         # value, an instance of self.dtype.type. When passed `fill_value=None`,
 954         # the default of `self.dtype.na_value` should be used.
 955         # This may differ from the physical storage type your ExtensionArray
 956         # uses. In this case, your implementation is responsible for casting
 957         # the user-facing type to the storage type, before using
 958         # pandas.api.extensions.take
 959         raise AbstractMethodError(self)

1168 class ExtensionScalarOpsMixin(ExtensionOpsMixin):
1169     """
1170     A mixin for defining  ops on an ExtensionArray.
1171 
1172     It is assumed that the underlying scalar objects have the operators
1173     already defined.
1174 
1175     Notes
1176     -----
1177     If you have defined a subclass MyExtensionArray(ExtensionArray), then
1178     use MyExtensionArray(ExtensionArray, ExtensionScalarOpsMixin) to
1179     get the arithmetic operators.  After the definition of MyExtensionArray,
1180     insert the lines
1181 
1182     MyExtensionArray._add_arithmetic_ops()
1183     MyExtensionArray._add_comparison_ops()
1184 
1185     to link the operators to your class.
1186 
1187     .. note::
1188 
1189        You may want to set ``__array_priority__`` if you want your
1190        implementation to be called when involved in binary operations
1191        with NumPy arrays.
1192     """
1193 
1194     @classmethod
1195     def _create_method(cls, op, coerce_to_dtype=True, result_dtype=None):
1196         """
1197         A class method that returns a method that will correspond to an
1198         operator for an ExtensionArray subclass, by dispatching to the
1199         relevant operator defined on the individual elements of the
1200         ExtensionArray.
1201 
1202         Parameters
1203         ----------
1204         op : function
1205             An operator that takes arguments op(a, b)
1206         coerce_to_dtype : bool, default True
1207             boolean indicating whether to attempt to convert
1208             the result to the underlying ExtensionArray dtype.
1209             If it's not possible to create a new ExtensionArray with the
1210             values, an ndarray is returned instead.
1211 
1212         Returns
1213         -------
1214         Callable[[Any, Any], Union[ndarray, ExtensionArray]]
1215             A method that can be bound to a class. When used, the method
1216             receives the two arguments, one of which is the instance of
1217             this class, and should return an ExtensionArray or an ndarray.
1218 
1219             Returning an ndarray may be necessary when the result of the
1220             `op` cannot be stored in the ExtensionArray. The dtype of the
1221             ndarray uses NumPy's normal inference rules.
1222 
1223         Examples
1224         --------
1225         Given an ExtensionArray subclass called MyExtensionArray, use
1226 
1227             __add__ = cls._create_method(operator.add)
1228 
1229         in the class definition of MyExtensionArray to create the operator
1230         for addition, that will be based on the operator implementation
1231         of the underlying elements of the ExtensionArray
1232         """
1233 
1234         def _binop(self, other):
1235             def convert_values(param):
1236                 if isinstance(param, ExtensionArray) or is_list_like(param):
1237                     ovalues = param
1238                 else:  # Assume its an object
1239                     ovalues = [param] * len(self)
1240                 return ovalues
1241 
1242             if isinstance(other, (ABCSeries, ABCIndexClass)):
1243                 # rely on pandas to unbox and dispatch to us
1244                 return NotImplemented
1245 
1246             lvalues = self
1247             rvalues = convert_values(other)
1248 
1249             # If the operator is not defined for the underlying objects,
1250             # a TypeError should be raised
1251             res = [op(a, b) for (a, b) in zip(lvalues, rvalues)]
1252 
1253             def _maybe_convert(arr):
1254                 if coerce_to_dtype:
1255                     # https://github.com/pandas-dev/pandas/issues/22850
1256                     # We catch all regular exceptions here, and fall back
1257                     # to an ndarray.
1258                     res = maybe_cast_to_extension_array(type(self), arr)
1259                     if not isinstance(res, type(self)):
1260                         # exception raised in _from_sequence; ensure we have ndarray
1261                         res = np.asarray(arr)
1262                 else:
1263                     res = np.asarray(arr, dtype=result_dtype)
1264                 return res
1265 
1266             if op.__name__ in {"divmod", "rdivmod"}:
1267                 a, b = zip(*res)
1268                 return _maybe_convert(a), _maybe_convert(b)
1269 
1270             return _maybe_convert(res)
1271 
1272         op_name = f"__{op.__name__}__"
1273         return set_function_name(_binop, op_name, cls)
1274 
1275     @classmethod
1276     def _create_arithmetic_method(cls, op):
1277         return cls._create_method(op)
1278 
1279     @classmethod
1280     def _create_comparison_method(cls, op):
1281         return cls._create_method(op, coerce_to_dtype=False, result_dtype=bool)
