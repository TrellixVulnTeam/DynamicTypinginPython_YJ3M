  15 def generate_regular_range(
  16     start: Union[Timestamp, Timedelta],
  17     end: Union[Timestamp, Timedelta],
  18     periods: int,
  19     freq: DateOffset,
  20 ):
  21     """
  22     Generate a range of dates or timestamps with the spans between dates
  23     described by the given `freq` DateOffset.
  24 
  25     Parameters
  26     ----------
  27     start : Timedelta, Timestamp or None
  28         First point of produced date range.
  29     end : Timedelta, Timestamp or None
  30         Last point of produced date range.
  31     periods : int
  32         Number of periods in produced date range.
  33     freq : Tick
  34         Describes space between dates in produced date range.
  35 
  36     Returns
  37     -------
  38     ndarray[np.int64] Representing nanoseconds.
  39     """
  40     start = start.value if start is not None else None
  41     end = end.value if end is not None else None
  42     stride = freq.nanos
  43 
  44     if periods is None:
  45         b = start
  46         # cannot just use e = Timestamp(end) + 1 because arange breaks when
  47         # stride is too large, see GH10887
  48         e = b + (end - b) // stride * stride + stride // 2 + 1
  49     elif start is not None:
  50         b = start
  51         e = _generate_range_overflow_safe(b, periods, stride, side="start")
  52     elif end is not None:
  53         e = end + stride
  54         b = _generate_range_overflow_safe(e, periods, stride, side="end")
  55     else:
  56         raise ValueError(
  57             "at least 'start' or 'end' should be specified if a 'period' is given."
  58         )
  59 
  60     with np.errstate(over="raise"):
  61         # If the range is sufficiently large, np.arange may overflow
  62         #  and incorrectly return an empty array if not caught.
  63         try:
  64             values = np.arange(b, e, stride, dtype=np.int64)
  65         except FloatingPointError:
  66             xdr = [b]
  67             while xdr[-1] != e:
  68                 xdr.append(xdr[-1] + stride)
  69             values = np.array(xdr[:-1], dtype=np.int64)
  70     return values

  73 def _generate_range_overflow_safe(
  74     endpoint: int, periods: int, stride: int, side: str = "start"
  75 ) -> int:
  76     """
  77     Calculate the second endpoint for passing to np.arange, checking
  78     to avoid an integer overflow.  Catch OverflowError and re-raise
  79     as OutOfBoundsDatetime.
  80 
  81     Parameters
  82     ----------
  83     endpoint : int
  84         nanosecond timestamp of the known endpoint of the desired range
  85     periods : int
  86         number of periods in the desired range
  87     stride : int
  88         nanoseconds between periods in the desired range
  89     side : {'start', 'end'}
  90         which end of the range `endpoint` refers to
  91 
  92     Returns
  93     -------
  94     other_end : int
  95 
  96     Raises
  97     ------
  98     OutOfBoundsDatetime
  99     """
 100     # GH#14187 raise instead of incorrectly wrapping around
 101     assert side in ["start", "end"]
 102 
 103     i64max = np.uint64(np.iinfo(np.int64).max)
 104     msg = f"Cannot generate range with {side}={endpoint} and periods={periods}"
 105 
 106     with np.errstate(over="raise"):
 107         # if periods * strides cannot be multiplied within the *uint64* bounds,
 108         #  we cannot salvage the operation by recursing, so raise
 109         try:
 110             addend = np.uint64(periods) * np.uint64(np.abs(stride))
 111         except FloatingPointError as err:
 112             raise OutOfBoundsDatetime(msg) from err
 113 
 114     if np.abs(addend) <= i64max:
 115         # relatively easy case without casting concerns
 116         return _generate_range_overflow_safe_signed(endpoint, periods, stride, side)
 117 
 118     elif (endpoint > 0 and side == "start" and stride > 0) or (
 119         endpoint < 0 and side == "end" and stride > 0
 120     ):
 121         # no chance of not-overflowing
 122         raise OutOfBoundsDatetime(msg)
 123 
 124     elif side == "end" and endpoint > i64max and endpoint - stride <= i64max:
 125         # in _generate_regular_range we added `stride` thereby overflowing
 126         #  the bounds.  Adjust to fix this.
 127         return _generate_range_overflow_safe(
 128             endpoint - stride, periods - 1, stride, side
 129         )
 130 
 131     # split into smaller pieces
 132     mid_periods = periods // 2
 133     remaining = periods - mid_periods
 134     assert 0 < remaining < periods, (remaining, periods, endpoint, stride)
 135 
 136     midpoint = _generate_range_overflow_safe(endpoint, mid_periods, stride, side)
 137     return _generate_range_overflow_safe(midpoint, remaining, stride, side)
 138

 140 def _generate_range_overflow_safe_signed(
 141     endpoint: int, periods: int, stride: int, side: str
 142 ) -> int:
 143     """
 144     A special case for _generate_range_overflow_safe where `periods * stride`
 145     can be calculated without overflowing int64 bounds.
 146     """
 147     assert side in ["start", "end"]
 148     if side == "end":
 149         stride *= -1
 150 
 151     with np.errstate(over="raise"):
 152         addend = np.int64(periods) * np.int64(stride)
 153         try:
 154             # easy case with no overflows
 155             return np.int64(endpoint) + addend
 156         except (FloatingPointError, OverflowError):
 157             # with endpoint negative and addend positive we risk
 158             #  FloatingPointError; with reversed signed we risk OverflowError
 159             pass
 160 
 161         # if stride and endpoint had opposite signs, then endpoint + addend
 162         #  should never overflow.  so they must have the same signs
 163         assert (stride > 0 and endpoint >= 0) or (stride < 0 and endpoint <= 0)
 164 
 165         if stride > 0:
 166             # watch out for very special case in which we just slightly
 167             #  exceed implementation bounds, but when passing the result to
 168             #  np.arange will get a result slightly within the bounds
 169             result = np.uint64(endpoint) + np.uint64(addend)
 170             i64max = np.uint64(np.iinfo(np.int64).max)
 171             assert result > i64max
 172             if result <= i64max + np.uint64(stride):
 173                 return result
 174 
 175     raise OutOfBoundsDatetime(
 176         f"Cannot generate range with {side}={endpoint} and periods={periods}"
 177     )
