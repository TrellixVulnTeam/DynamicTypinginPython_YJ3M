  48     @classmethod
  49     def from_coo(cls, A, dense_index=False):
  50         """
  51         Create a Series with sparse values from a scipy.sparse.coo_matrix.
  52 
  53         Parameters
  54         ----------
  55         A : scipy.sparse.coo_matrix
  56         dense_index : bool, default False
  57             If False (default), the SparseSeries index consists of only the
  58             coords of the non-null entries of the original coo_matrix.
  59             If True, the SparseSeries index consists of the full sorted
  60             (row, col) coordinates of the coo_matrix.
  61 
  62         Returns
  63         -------
  64         s : Series
  65             A Series with sparse values.
  66 
  67         Examples
  68         --------
  69         >>> from scipy import sparse
  70 
  71         >>> A = sparse.coo_matrix(
  72         ...     ([3.0, 1.0, 2.0], ([1, 0, 0], [0, 2, 3])), shape=(3, 4)
  73         ... )
  74         >>> A
  75         <3x4 sparse matrix of type '<class 'numpy.float64'>'
  76         with 3 stored elements in COOrdinate format>
  77 
  78         >>> A.todense()
  79         matrix([[0., 0., 1., 2.],
  80         [3., 0., 0., 0.],
  81         [0., 0., 0., 0.]])
  82 
  83         >>> ss = pd.Series.sparse.from_coo(A)
  84         >>> ss
  85         0  2    1.0
  86            3    2.0
  87         1  0    3.0
  88         dtype: Sparse[float64, nan]
  89         """
  90         from pandas.core.arrays.sparse.scipy_sparse import _coo_to_sparse_series
  91         from pandas import Series
  92 
  93         result = _coo_to_sparse_series(A, dense_index=dense_index)
  94         result = Series(result.array, index=result.index, copy=False)
  95 
  96         return result

  98     def to_coo(self, row_levels=(0,), column_levels=(1,), sort_labels=False):
  99         """
 100         Create a scipy.sparse.coo_matrix from a Series with MultiIndex.
 101 
 102         Use row_levels and column_levels to determine the row and column
 103         coordinates respectively. row_levels and column_levels are the names
 104         (labels) or numbers of the levels. {row_levels, column_levels} must be
 105         a partition of the MultiIndex level names (or numbers).
 106 
 107         Parameters
 108         ----------
 109         row_levels : tuple/list
 110         column_levels : tuple/list
 111         sort_labels : bool, default False
 112             Sort the row and column labels before forming the sparse matrix.
 113 
 114         Returns
 115         -------
 116         y : scipy.sparse.coo_matrix
 117         rows : list (row labels)
 118         columns : list (column labels)
 119 
 120         Examples
 121         --------
 122         >>> s = pd.Series([3.0, np.nan, 1.0, 3.0, np.nan, np.nan])
 123         >>> s.index = pd.MultiIndex.from_tuples(
 124         ...     [
 125         ...         (1, 2, "a", 0),
 126         ...         (1, 2, "a", 1),
 127         ...         (1, 1, "b", 0),
 128         ...         (1, 1, "b", 1),
 129         ...         (2, 1, "b", 0),
 130         ...         (2, 1, "b", 1)
 131         ...     ],
 132         ...     names=["A", "B", "C", "D"],
 133         ... )
 134         >>> s
 135         A  B  C  D
 136         1  2  a  0    3.0
 137                  1    NaN
 138            1  b  0    1.0
 139                  1    3.0
 140         2  1  b  0    NaN
 141                  1    NaN
 142         dtype: float64
 143 
 144         >>> ss = s.astype("Sparse")
 145         >>> ss
 146         A  B  C  D
 147         1  2  a  0    3.0
 148                  1    NaN
 149            1  b  0    1.0
 150                  1    3.0
 151         2  1  b  0    NaN
 152                  1    NaN
 153         dtype: Sparse[float64, nan]
 154 
 155         >>> A, rows, columns = ss.sparse.to_coo(
 156         ...     row_levels=["A", "B"], column_levels=["C", "D"], sort_labels=True
 157         ... )
 158         >>> A
 159         <3x4 sparse matrix of type '<class 'numpy.float64'>'
 160         with 3 stored elements in COOrdinate format>
 161         >>> A.todense()
 162         matrix([[0., 0., 1., 3.],
 163         [3., 0., 0., 0.],
 164         [0., 0., 0., 0.]])
 165 
 166         >>> rows
 167         [(1, 1), (1, 2), (2, 1)]
 168         >>> columns
 169         [('a', 0), ('a', 1), ('b', 0), ('b', 1)]
 170         """
 171         from pandas.core.arrays.sparse.scipy_sparse import _sparse_series_to_coo
 172 
 173         A, rows, columns = _sparse_series_to_coo(
 174             self._parent, row_levels, column_levels, sort_labels=sort_labels
 175         )
 176         return A, rows, columns

 178     def to_dense(self):
 179         """
 180         Convert a Series from sparse values to dense.
 181 
 182         .. versionadded:: 0.25.0
 183 
 184         Returns
 185         -------
 186         Series:
 187             A Series with the same values, stored as a dense array.
 188 
 189         Examples
 190         --------
 191         >>> series = pd.Series(pd.arrays.SparseArray([0, 1, 0]))
 192         >>> series
 193         0    0
 194         1    1
 195         2    0
 196         dtype: Sparse[int64, 0]
 197 
 198         >>> series.sparse.to_dense()
 199         0    0
 200         1    1
 201         2    0
 202         dtype: int64
 203         """
 204         from pandas import Series
 205 
 206         return Series(
 207             self._parent.array.to_dense(),
 208             index=self._parent.index,
 209             name=self._parent.name,
 210         )

 225     @classmethod
 226     def from_spmatrix(cls, data, index=None, columns=None):
 227         """
 228         Create a new DataFrame from a scipy sparse matrix.
 229 
 230         .. versionadded:: 0.25.0
 231 
 232         Parameters
 233         ----------
 234         data : scipy.sparse.spmatrix
 235             Must be convertible to csc format.
 236         index, columns : Index, optional
 237             Row and column labels to use for the resulting DataFrame.
 238             Defaults to a RangeIndex.
 239 
 240         Returns
 241         -------
 242         DataFrame
 243             Each column of the DataFrame is stored as a
 244             :class:`arrays.SparseArray`.
 245 
 246         Examples
 247         --------
 248         >>> import scipy.sparse
 249         >>> mat = scipy.sparse.eye(3)
 250         >>> pd.DataFrame.sparse.from_spmatrix(mat)
 251              0    1    2
 252         0  1.0  0.0  0.0
 253         1  0.0  1.0  0.0
 254         2  0.0  0.0  1.0
 255         """
 256         from pandas import DataFrame
 257         from pandas._libs.sparse import IntIndex
 258 
 259         data = data.tocsc()
 260         index, columns = cls._prep_index(data, index, columns)
 261         n_rows, n_columns = data.shape
 262         # We need to make sure indices are sorted, as we create
 263         # IntIndex with no input validation (i.e. check_integrity=False ).
 264         # Indices may already be sorted in scipy in which case this adds
 265         # a small overhead.
 266         data.sort_indices()
 267         indices = data.indices
 268         indptr = data.indptr
 269         array_data = data.data
 270         dtype = SparseDtype(array_data.dtype, 0)
 271         arrays = []
 272         for i in range(n_columns):
 273             sl = slice(indptr[i], indptr[i + 1])
 274             idx = IntIndex(n_rows, indices[sl], check_integrity=False)
 275             arr = SparseArray._simple_new(array_data[sl], idx, dtype)
 276             arrays.append(arr)
 277         return DataFrame._from_arrays(
 278             arrays, columns=columns, index=index, verify_integrity=False
 279         )

 281     def to_dense(self):
 282         """
 283         Convert a DataFrame with sparse values to dense.
 284 
 285         .. versionadded:: 0.25.0
 286 
 287         Returns
 288         -------
 289         DataFrame
 290             A DataFrame with the same values stored as dense arrays.
 291 
 292         Examples
 293         --------
 294         >>> df = pd.DataFrame({"A": pd.arrays.SparseArray([0, 1, 0])})
 295         >>> df.sparse.to_dense()
 296            A
 297         0  0
 298         1  1
 299         2  0
 300         """
 301         from pandas import DataFrame
 302 
 303         data = {k: v.array.to_dense() for k, v in self._parent.items()}
 304         return DataFrame(data, index=self._parent.index, columns=self._parent.columns)

 306     def to_coo(self):
 307         """
 308         Return the contents of the frame as a sparse SciPy COO matrix.
 309 
 310         .. versionadded:: 0.25.0
 311 
 312         Returns
 313         -------
 314         coo_matrix : scipy.sparse.spmatrix
 315             If the caller is heterogeneous and contains booleans or objects,
 316             the result will be of dtype=object. See Notes.
 317 
 318         Notes
 319         -----
 320         The dtype will be the lowest-common-denominator type (implicit
 321         upcasting); that is to say if the dtypes (even of numeric types)
 322         are mixed, the one that accommodates all will be chosen.
 323 
 324         e.g. If the dtypes are float16 and float32, dtype will be upcast to
 325         float32. By numpy.find_common_type convention, mixing int64 and
 326         and uint64 will result in a float64 dtype.
 327         """
 328         import_optional_dependency("scipy")
 329         from scipy.sparse import coo_matrix
 330 
 331         dtype = find_common_type(self._parent.dtypes)
 332         if isinstance(dtype, SparseDtype):
 333             dtype = dtype.subtype
 334 
 335         cols, rows, datas = [], [], []
 336         for col, name in enumerate(self._parent):
 337             s = self._parent[name]
 338             row = s.array.sp_index.to_int_index().indices
 339             cols.append(np.repeat(col, len(row)))
 340             rows.append(row)
 341             datas.append(s.array.sp_values.astype(dtype, copy=False))
 342 
 343         cols = np.concatenate(cols)
 344         rows = np.concatenate(rows)
 345         datas = np.concatenate(datas)
 346         return coo_matrix((datas, (rows, cols)), shape=self._parent.shape)

 348     @property
 349     def density(self) -> float:
 350         """
 351         Ratio of non-sparse points to total (dense) data points.
 352         """
 353         return np.mean([column.array.density for _, column in self._parent.items()])

 355     @staticmethod
 356     def _prep_index(data, index, columns):
 357         import pandas.core.indexes.base as ibase
 358         from pandas.core.indexes.api import ensure_index
 359 
 360         N, K = data.shape
 361         if index is None:
 362             index = ibase.default_index(N)
 363         else:
 364             index = ensure_index(index)
 365         if columns is None:
 366             columns = ibase.default_index(K)
 367         else:
 368             columns = ensure_index(columns)
 369 
 370         if len(columns) != K:
 371             raise ValueError(f"Column length mismatch: {len(columns)} vs. {K}")
 372         if len(index) != N:
 373             raise ValueError(f"Index length mismatch: {len(index)} vs. {N}")
 374         return index, columns
