  63 def _get_fill(arr: "SparseArray") -> np.ndarray:
  64     """

  68     ----------

  72     -------

  87 def _sparse_array_op(
  88     left: "SparseArray", right: "SparseArray", op: Callable, name: str
  89 ) -> Any:
  90     """
  91     Perform a binary operation between two arrays.
  92 
  93     Parameters
  94     ----------
  95     left : Union[SparseArray, ndarray]
  96     right : Union[SparseArray, ndarray]
  97     op : Callable
  98         The binary operation to perform
  99     name str
 100         Name of the callable.
 101 
 102     Returns
 103     -------
 104     SparseArray
 105     """
 106     if name.startswith("__"):
 107         # For lookups in _libs.sparse we need non-dunder op name
 108         name = name[2:-2]
 109 
 110     # dtype used to find corresponding sparse method
 111     ltype = left.dtype.subtype
 112     rtype = right.dtype.subtype
 113 
 114     if not is_dtype_equal(ltype, rtype):
 115         subtype = find_common_type([ltype, rtype])
 116         ltype = SparseDtype(subtype, left.fill_value)
 117         rtype = SparseDtype(subtype, right.fill_value)
 118 
 119         # TODO(GH-23092): pass copy=False. Need to fix astype_nansafe
 120         left = left.astype(ltype)
 121         right = right.astype(rtype)
 122         dtype = ltype.subtype
 123     else:
 124         dtype = ltype
 125 
 126     # dtype the result must have
 127     result_dtype = None
 128 
 129     if left.sp_index.ngaps == 0 or right.sp_index.ngaps == 0:
 130         with np.errstate(all="ignore"):
 131             result = op(left.to_dense(), right.to_dense())
 132             fill = op(_get_fill(left), _get_fill(right))
 133 
 134         if left.sp_index.ngaps == 0:
 135             index = left.sp_index
 136         else:
 137             index = right.sp_index
 138     elif left.sp_index.equals(right.sp_index):
 139         with np.errstate(all="ignore"):
 140             result = op(left.sp_values, right.sp_values)
 141             fill = op(_get_fill(left), _get_fill(right))
 142         index = left.sp_index
 143     else:
 144         if name[0] == "r":
 145             left, right = right, left
 146             name = name[1:]
 147 
 148         if name in ("and", "or", "xor") and dtype == "bool":
 149             opname = f"sparse_{name}_uint8"
 150             # to make template simple, cast here
 151             left_sp_values = left.sp_values.view(np.uint8)
 152             right_sp_values = right.sp_values.view(np.uint8)
 153             result_dtype = np.bool
 154         else:
 155             opname = f"sparse_{name}_{dtype}"
 156             left_sp_values = left.sp_values
 157             right_sp_values = right.sp_values
 158 
 159         sparse_op = getattr(splib, opname)
 160 
 161         with np.errstate(all="ignore"):
 162             result, index, fill = sparse_op(
 163                 left_sp_values,
 164                 left.sp_index,
 165                 left.fill_value,
 166                 right_sp_values,
 167                 right.sp_index,
 168                 right.fill_value,
 169             )
 170 
 171     if result_dtype is None:
 172         result_dtype = result.dtype
 173 
 174     return _wrap_result(name, result, index, fill, dtype=result_dtype)

 177 def _wrap_result(name, data, sparse_index, fill_value, dtype=None):
 178     """
 179     wrap op result to have correct dtype
 180     """
 181     if name.startswith("__"):
 182         # e.g. __eq__ --> eq
 183         name = name[2:-2]
 184 
 185     if name in ("eq", "ne", "lt", "gt", "le", "ge"):
 186         dtype = np.bool
 187 
 188     fill_value = lib.item_from_zerodim(fill_value)
 189 
 190     if is_bool_dtype(dtype):
 191         # fill_value may be np.bool_
 192         fill_value = bool(fill_value)
 193     return SparseArray(
 194         data, sparse_index=sparse_index, fill_value=fill_value, dtype=dtype
 195     )

 198 class SparseArray(PandasObject, ExtensionArray, ExtensionOpsMixin):
 199     """
 200     An ExtensionArray for storing sparse data.
 201 
 202     .. versionchanged:: 0.24.0
 203 
 204        Implements the ExtensionArray interface.
 205 
 206     Parameters
 207     ----------
 208     data : array-like
 209         A dense array of values to store in the SparseArray. This may contain
 210         `fill_value`.
 211     sparse_index : SparseIndex, optional
 212     index : Index
 213     fill_value : scalar, optional
 214         Elements in `data` that are `fill_value` are not stored in the
 215         SparseArray. For memory savings, this should be the most common value
 216         in `data`. By default, `fill_value` depends on the dtype of `data`:
 217 
 218         =========== ==========
 219         data.dtype  na_value
 220         =========== ==========
 221         float       ``np.nan``
 222         int         ``0``
 223         bool        False
 224         datetime64  ``pd.NaT``
 225         timedelta64 ``pd.NaT``
 226         =========== ==========
 227 
 228         The fill value is potentially specified in three ways. In order of
 229         precedence, these are
 230 
 231         1. The `fill_value` argument
 232         2. ``dtype.fill_value`` if `fill_value` is None and `dtype` is
 233            a ``SparseDtype``
 234         3. ``data.dtype.fill_value`` if `fill_value` is None and `dtype`
 235            is not a ``SparseDtype`` and `data` is a ``SparseArray``.
 236 
 237     kind : {'int', 'block'}, default 'int'
 238         The type of storage for sparse locations.
 239 
 240         * 'block': Stores a `block` and `block_length` for each
 241           contiguous *span* of sparse values. This is best when
 242           sparse data tends to be clumped together, with large
 243           regions of ``fill-value`` values between sparse values.
 244         * 'integer': uses an integer to store the location of
 245           each sparse value.
 246 
 247     dtype : np.dtype or SparseDtype, optional
 248         The dtype to use for the SparseArray. For numpy dtypes, this
 249         determines the dtype of ``self.sp_values``. For SparseDtype,
 250         this determines ``self.sp_values`` and ``self.fill_value``.
 251     copy : bool, default False
 252         Whether to explicitly copy the incoming `data` array.
 253 
 254     Attributes
 255     ----------
 256     None
 257 
 258     Methods
 259     -------
 260     None
 261 
 262     Examples
 263     --------
 264     >>> from pandas.arrays import SparseArray
 265     >>> arr = SparseArray([0, 0, 1, 2])
 266     >>> arr
 267     [0, 0, 1, 2]
 268     Fill: 0
 269     IntIndex
 270     Indices: array([2, 3], dtype=int32)
 271     """
 272 
 273     _subtyp = "sparse_array"  # register ABCSparseArray
 274     _deprecations = PandasObject._deprecations | frozenset(["get_values"])
 275     _sparse_index: SparseIndex

 277     def __init__(
 278         self,
 279         data,
 280         sparse_index=None,
 281         index=None,
 282         fill_value=None,
 283         kind="integer",
 284         dtype=None,
 285         copy=False,
 286     ):
 287 
 288         if fill_value is None and isinstance(dtype, SparseDtype):
 289             fill_value = dtype.fill_value
 290 
 291         if isinstance(data, type(self)):
 292             # disable normal inference on dtype, sparse_index, & fill_value
 293             if sparse_index is None:
 294                 sparse_index = data.sp_index
 295             if fill_value is None:
 296                 fill_value = data.fill_value
 297             if dtype is None:
 298                 dtype = data.dtype
 299             # TODO: make kind=None, and use data.kind?
 300             data = data.sp_values
 301 
 302         # Handle use-provided dtype
 303         if isinstance(dtype, str):
 304             # Two options: dtype='int', regular numpy dtype
 305             # or dtype='Sparse[int]', a sparse dtype
 306             try:
 307                 dtype = SparseDtype.construct_from_string(dtype)
 308             except TypeError:
 309                 dtype = pandas_dtype(dtype)
 310 
 311         if isinstance(dtype, SparseDtype):
 312             if fill_value is None:
 313                 fill_value = dtype.fill_value
 314             dtype = dtype.subtype
 315 
 316         if index is not None and not is_scalar(data):
 317             raise Exception("must only pass scalars with an index")
 318 
 319         if is_scalar(data):
 320             if index is not None:
 321                 if data is None:
 322                     data = np.nan
 323 
 324             if index is not None:
 325                 npoints = len(index)
 326             elif sparse_index is None:
 327                 npoints = 1
 328             else:
 329                 npoints = sparse_index.length
 330 
 331             dtype = infer_dtype_from_scalar(data)[0]
 332             data = construct_1d_arraylike_from_scalar(data, npoints, dtype)
 333 
 334         if dtype is not None:
 335             dtype = pandas_dtype(dtype)
 336 
 337         # TODO: disentangle the fill_value dtype inference from
 338         # dtype inference
 339         if data is None:
 340             # TODO: What should the empty dtype be? Object or float?
 341             data = np.array([], dtype=dtype)
 342 
 343         if not is_array_like(data):
 344             try:
 345                 # probably shared code in sanitize_series
 346 
 347                 data = sanitize_array(data, index=None)
 348             except ValueError:
 349                 # NumPy may raise a ValueError on data like [1, []]
 350                 # we retry with object dtype here.
 351                 if dtype is None:
 352                     dtype = object
 353                     data = np.atleast_1d(np.asarray(data, dtype=dtype))
 354                 else:
 355                     raise
 356 
 357         if copy:
 358             # TODO: avoid double copy when dtype forces cast.
 359             data = data.copy()
 360 
 361         if fill_value is None:
 362             fill_value_dtype = data.dtype if dtype is None else dtype
 363             if fill_value_dtype is None:
 364                 fill_value = np.nan
 365             else:
 366                 fill_value = na_value_for_dtype(fill_value_dtype)
 367 
 368         if isinstance(data, type(self)) and sparse_index is None:
 369             sparse_index = data._sparse_index
 370             sparse_values = np.asarray(data.sp_values, dtype=dtype)
 371         elif sparse_index is None:
 372             data = extract_array(data, extract_numpy=True)
 373             if not isinstance(data, np.ndarray):
 374                 # EA
 375                 if is_datetime64tz_dtype(data.dtype):
 376                     warnings.warn(
 377                         f"Creating SparseArray from {data.dtype} data "
 378                         "loses timezone information.  Cast to object before "
 379                         "sparse to retain timezone information.",
 380                         UserWarning,
 381                         stacklevel=2,
 382                     )
 383                     data = np.asarray(data, dtype="datetime64[ns]")
 384                 data = np.asarray(data)
 385             sparse_values, sparse_index, fill_value = make_sparse(
 386                 data, kind=kind, fill_value=fill_value, dtype=dtype
 387             )
 388         else:
 389             sparse_values = np.asarray(data, dtype=dtype)
 390             if len(sparse_values) != sparse_index.npoints:
 391                 raise AssertionError(
 392                     f"Non array-like type {type(sparse_values)} must "
 393                     "have the same length as the index"
 394                 )
 395         self._sparse_index = sparse_index
 396         self._sparse_values = sparse_values
 397         self._dtype = SparseDtype(sparse_values.dtype, fill_value)

 409     @classmethod
 410     def from_spmatrix(cls, data):
 411         """
 412         Create a SparseArray from a scipy.sparse matrix.
 413 
 414         .. versionadded:: 0.25.0
 415 
 416         Parameters
 417         ----------
 418         data : scipy.sparse.sp_matrix
 419             This should be a SciPy sparse matrix where the size
 420             of the second dimension is 1. In other words, a
 421             sparse matrix with a single column.
 422 
 423         Returns
 424         -------
 425         SparseArray
 426 
 427         Examples
 428         --------
 429         >>> import scipy.sparse
 430         >>> mat = scipy.sparse.coo_matrix((4, 1))
 431         >>> pd.arrays.SparseArray.from_spmatrix(mat)
 432         [0.0, 0.0, 0.0, 0.0]
 433         Fill: 0.0
 434         IntIndex
 435         Indices: array([], dtype=int32)
 436         """
 437         length, ncol = data.shape
 438 
 439         if ncol != 1:
 440             raise ValueError(f"'data' must have a single column, not '{ncol}'")
 441 
 442         # our sparse index classes require that the positions be strictly
 443         # increasing. So we need to sort loc, and arr accordingly.
 444         data = data.tocsc()
 445         data.sort_indices()
 446         arr = data.data
 447         idx = data.indices
 448 
 449         zero = np.array(0, dtype=arr.dtype).item()
 450         dtype = SparseDtype(arr.dtype, zero)
 451         index = IntIndex(length, idx)
 452 
 453         return cls._simple_new(arr, index, dtype)

 455     def __array__(self, dtype=None, copy=True) -> np.ndarray:
 456         fill_value = self.fill_value
 457 
 458         if self.sp_index.ngaps == 0:
 459             # Compat for na dtype and int values.
 460             return self.sp_values
 461         if dtype is None:
 462             # Can NumPy represent this type?
 463             # If not, `np.result_type` will raise. We catch that
 464             # and return object.
 465             if is_datetime64_any_dtype(self.sp_values.dtype):
 466                 # However, we *do* special-case the common case of
 467                 # a datetime64 with pandas NaT.
 468                 if fill_value is NaT:
 469                     # Can't put pd.NaT in a datetime64[ns]
 470                     fill_value = np.datetime64("NaT")
 471             try:
 472                 dtype = np.result_type(self.sp_values.dtype, type(fill_value))
 473             except TypeError:
 474                 dtype = object
 475 
 476         out = np.full(self.shape, fill_value, dtype=dtype)
 477         out[self.sp_index.to_int_index().indices] = self.sp_values
 478         return out

 601     def fillna(self, value=None, method=None, limit=None):
 602         """
 603         Fill missing values with `value`.
 604 
 605         Parameters
 606         ----------
 607         value : scalar, optional
 608         method : str, optional
 609 
 610             .. warning::
 611 
 612                Using 'method' will result in high memory use,
 613                as all `fill_value` methods will be converted to
 614                an in-memory ndarray
 615 
 616         limit : int, optional
 617 
 618         Returns
 619         -------
 620         SparseArray
 621 
 622         Notes
 623         -----
 624         When `value` is specified, the result's ``fill_value`` depends on
 625         ``self.fill_value``. The goal is to maintain low-memory use.
 626 
 627         If ``self.fill_value`` is NA, the result dtype will be
 628         ``SparseDtype(self.dtype, fill_value=value)``. This will preserve
 629         amount of memory used before and after filling.
 630 
 631         When ``self.fill_value`` is not NA, the result dtype will be
 632         ``self.dtype``. Again, this preserves the amount of memory used.
 633         """
 634         if (method is None and value is None) or (
 635             method is not None and value is not None
 636         ):
 637             raise ValueError("Must specify one of 'method' or 'value'.")
 638 
 639         elif method is not None:
 640             msg = "fillna with 'method' requires high memory usage."
 641             warnings.warn(msg, PerformanceWarning)
 642             filled = interpolate_2d(np.asarray(self), method=method, limit=limit)
 643             return type(self)(filled, fill_value=self.fill_value)
 644 
 645         else:
 646             new_values = np.where(isna(self.sp_values), value, self.sp_values)
 647 
 648             if self._null_fill_value:
 649                 # This is essentially just updating the dtype.
 650                 new_dtype = SparseDtype(self.dtype.subtype, fill_value=value)
 651             else:
 652                 new_dtype = self.dtype
 653 
 654         return self._simple_new(new_values, self._sparse_index, new_dtype)

 656     def shift(self, periods=1, fill_value=None):
 657 
 658         if not len(self) or periods == 0:
 659             return self.copy()
 660 
 661         if isna(fill_value):
 662             fill_value = self.dtype.na_value
 663 
 664         subtype = np.result_type(fill_value, self.dtype.subtype)
 665 
 666         if subtype != self.dtype.subtype:
 667             # just coerce up front
 668             arr = self.astype(SparseDtype(subtype, self.fill_value))
 669         else:
 670             arr = self
 671 
 672         empty = self._from_sequence(
 673             [fill_value] * min(abs(periods), len(self)), dtype=arr.dtype
 674         )
 675 
 676         if periods > 0:
 677             a = empty
 678             b = arr[:-periods]
 679         else:
 680             a = arr[abs(periods) :]
 681             b = empty
 682         return arr._concat_same_type([a, b])

 684     def _first_fill_value_loc(self):
 685         """
 686         Get the location of the first missing value.
 687 
 688         Returns
 689         -------
 690         int
 691         """
 692         if len(self) == 0 or self.sp_index.npoints == len(self):
 693             return -1
 694 
 695         indices = self.sp_index.to_int_index().indices
 696         if not len(indices) or indices[0] > 0:
 697             return 0
 698 
 699         diff = indices[1:] - indices[:-1]
 700         return np.searchsorted(diff, 2) + 1

 702     def unique(self):

 723     def value_counts(self, dropna=True):
 724         """
 725         Returns a Series containing counts of unique values.
 726 
 727         Parameters
 728         ----------
 729         dropna : boolean, default True
 730             Don't include counts of NaN, even if NaN is in sp_values.
 731 
 732         Returns
 733         -------
 734         counts : Series
 735         """
 736         from pandas import Index, Series
 737 
 738         keys, counts = algos._value_counts_arraylike(self.sp_values, dropna=dropna)
 739         fcounts = self.sp_index.ngaps
 740         if fcounts > 0:
 741             if self._null_fill_value and dropna:
 742                 pass
 743             else:
 744                 if self._null_fill_value:
 745                     mask = isna(keys)
 746                 else:
 747                     mask = keys == self.fill_value
 748 
 749                 if mask.any():
 750                     counts[mask] += fcounts
 751                 else:
 752                     keys = np.insert(keys, 0, self.fill_value)
 753                     counts = np.insert(counts, 0, fcounts)
 754 
 755         if not isinstance(keys, ABCIndexClass):
 756             keys = Index(keys)
 757         result = Series(counts, index=keys)
 758         return result

 764     def __getitem__(self, key):
 765         # avoid mypy issues when importing at the top-level
 766         from pandas.core.indexing import check_bool_indexer
 767 
 768         if isinstance(key, tuple):
 769             if len(key) > 1:
 770                 raise IndexError("too many indices for array.")
 771             key = key[0]
 772 
 773         if is_integer(key):
 774             return self._get_val_at(key)
 775         elif isinstance(key, tuple):
 776             data_slice = self.to_dense()[key]
 777         elif isinstance(key, slice):
 778             # special case to preserve dtypes
 779             if key == slice(None):
 780                 return self.copy()
 781             # TODO: this logic is surely elsewhere
 782             # TODO: this could be more efficient
 783             indices = np.arange(len(self), dtype=np.int32)[key]
 784             return self.take(indices)
 785         else:
 786             # TODO: I think we can avoid densifying when masking a
 787             # boolean SparseArray with another. Need to look at the
 788             # key's fill_value for True / False, and then do an intersection
 789             # on the indices of the sp_values.
 790             if isinstance(key, SparseArray):
 791                 if is_bool_dtype(key):
 792                     key = key.to_dense()
 793                 else:
 794                     key = np.asarray(key)
 795 
 796             key = check_array_indexer(self, key)
 797 
 798             if com.is_bool_indexer(key):
 799                 key = check_bool_indexer(self, key)
 800 
 801                 return self.take(np.arange(len(key), dtype=np.int32)[key])
 802             elif hasattr(key, "__len__"):
 803                 return self.take(key)
 804             else:
 805                 raise ValueError(f"Cannot slice with '{key}'")
 806

 809     def _get_val_at(self, loc):
 810         n = len(self)
 811         if loc < 0:
 812             loc += n
 813 
 814         if loc >= n or loc < 0:
 815             raise IndexError("Out of bounds access")
 816 
 817         sp_loc = self.sp_index.lookup(loc)
 818         if sp_loc == -1:
 819             return self.fill_value
 820         else:
 821             val = self.sp_values[sp_loc]
 822             val = com.maybe_box_datetimelike(val, self.sp_values.dtype)
 823             return val

 825     def take(self, indices, allow_fill=False, fill_value=None) -> "SparseArray":
 826         if is_scalar(indices):
 827             raise ValueError(f"'indices' must be an array, not a scalar '{indices}'.")
 828         indices = np.asarray(indices, dtype=np.int32)
 829 
 830         if indices.size == 0:
 831             result = np.array([], dtype="object")
 832             kwargs = {"dtype": self.dtype}
 833         elif allow_fill:
 834             result = self._take_with_fill(indices, fill_value=fill_value)
 835             kwargs = {}
 836         else:
 837             result = self._take_without_fill(indices)
 838             kwargs = {"dtype": self.dtype}
 839 
 840         return type(self)(result, fill_value=self.fill_value, kind=self.kind, **kwargs)

 842     def _take_with_fill(self, indices, fill_value=None) -> np.ndarray:
 843         if fill_value is None:
 844             fill_value = self.dtype.na_value
 845 
 846         if indices.min() < -1:
 847             raise ValueError(
 848                 "Invalid value in 'indices'. Must be between -1 "
 849                 "and the length of the array."
 850             )
 851 
 852         if indices.max() >= len(self):
 853             raise IndexError("out of bounds value in 'indices'.")
 854 
 855         if len(self) == 0:
 856             # Empty... Allow taking only if all empty
 857             if (indices == -1).all():
 858                 dtype = np.result_type(self.sp_values, type(fill_value))
 859                 taken = np.empty_like(indices, dtype=dtype)
 860                 taken.fill(fill_value)
 861                 return taken
 862             else:
 863                 raise IndexError("cannot do a non-empty take from an empty axes.")
 864 
 865         sp_indexer = self.sp_index.lookup_array(indices)
 866 
 867         if self.sp_index.npoints == 0:
 868             # Avoid taking from the empty self.sp_values
 869             taken = np.full(
 870                 sp_indexer.shape,
 871                 fill_value=fill_value,
 872                 dtype=np.result_type(type(fill_value)),
 873             )
 874         else:
 875             taken = self.sp_values.take(sp_indexer)
 876 
 877             # sp_indexer may be -1 for two reasons
 878             # 1.) we took for an index of -1 (new)
 879             # 2.) we took a value that was self.fill_value (old)
 880             new_fill_indices = indices == -1
 881             old_fill_indices = (sp_indexer == -1) & ~new_fill_indices
 882 
 883             # Fill in two steps.
 884             # Old fill values
 885             # New fill values
 886             # potentially coercing to a new dtype at each stage.
 887 
 888             m0 = sp_indexer[old_fill_indices] < 0
 889             m1 = sp_indexer[new_fill_indices] < 0
 890 
 891             result_type = taken.dtype
 892 
 893             if m0.any():
 894                 result_type = np.result_type(result_type, type(self.fill_value))
 895                 taken = taken.astype(result_type)
 896                 taken[old_fill_indices] = self.fill_value
 897 
 898             if m1.any():
 899                 result_type = np.result_type(result_type, type(fill_value))
 900                 taken = taken.astype(result_type)
 901                 taken[new_fill_indices] = fill_value
 902 
 903         return taken

 905     def _take_without_fill(self, indices) -> Union[np.ndarray, "SparseArray"]:
 906         to_shift = indices < 0
 907         indices = indices.copy()
 908 
 909         n = len(self)
 910 
 911         if (indices.max() >= n) or (indices.min() < -n):
 912             if n == 0:
 913                 raise IndexError("cannot do a non-empty take from an empty axes.")
 914             else:
 915                 raise IndexError("out of bounds value in 'indices'.")
 916 
 917         if to_shift.any():
 918             indices[to_shift] += n
 919 
 920         if self.sp_index.npoints == 0:
 921             # edge case in take...
 922             # I think just return
 923             out = np.full(
 924                 indices.shape,
 925                 self.fill_value,
 926                 dtype=np.result_type(type(self.fill_value)),
 927             )
 928             arr, sp_index, fill_value = make_sparse(out, fill_value=self.fill_value)
 929             return type(self)(arr, sparse_index=sp_index, fill_value=fill_value)
 930 
 931         sp_indexer = self.sp_index.lookup_array(indices)
 932         taken = self.sp_values.take(sp_indexer)
 933         fillable = sp_indexer < 0
 934 
 935         if fillable.any():
 936             # TODO: may need to coerce array to fill value
 937             result_type = np.result_type(taken, type(self.fill_value))
 938             taken = taken.astype(result_type)
 939             taken[fillable] = self.fill_value
 940 
 941         return taken

 955     @classmethod
 956     def _concat_same_type(cls, to_concat):
 957         fill_value = to_concat[0].fill_value
 958 
 959         values = []
 960         length = 0
 961 
 962         if to_concat:
 963             sp_kind = to_concat[0].kind
 964         else:
 965             sp_kind = "integer"
 966 
 967         if sp_kind == "integer":
 968             indices = []
 969 
 970             for arr in to_concat:
 971                 idx = arr.sp_index.to_int_index().indices.copy()
 972                 idx += length  # TODO: wraparound
 973                 length += arr.sp_index.length
 974 
 975                 values.append(arr.sp_values)
 976                 indices.append(idx)
 977 
 978             data = np.concatenate(values)
 979             indices = np.concatenate(indices)
 980             sp_index = IntIndex(length, indices)
 981 
 982         else:
 983             # when concatenating block indices, we don't claim that you'll
 984             # get an identical index as concating the values and then
 985             # creating a new index. We don't want to spend the time trying
 986             # to merge blocks across arrays in `to_concat`, so the resulting
 987             # BlockIndex may have more blocs.
 988             blengths = []
 989             blocs = []
 990 
 991             for arr in to_concat:
 992                 idx = arr.sp_index.to_block_index()
 993 
 994                 values.append(arr.sp_values)
 995                 blocs.append(idx.blocs.copy() + length)
 996                 blengths.append(idx.blengths)
 997                 length += arr.sp_index.length
 998 
 999             data = np.concatenate(values)
1000             blocs = np.concatenate(blocs)
1001             blengths = np.concatenate(blengths)
1002 
1003             sp_index = BlockIndex(length, blocs, blengths)
1004 
1005         return cls(data, sparse_index=sp_index, fill_value=fill_value)

1007     def astype(self, dtype=None, copy=True):
1008         """
1009         Change the dtype of a SparseArray.
1010 
1011         The output will always be a SparseArray. To convert to a dense
1012         ndarray with a certain dtype, use :meth:`numpy.asarray`.
1013 
1014         Parameters
1015         ----------
1016         dtype : np.dtype or ExtensionDtype
1017             For SparseDtype, this changes the dtype of
1018             ``self.sp_values`` and the ``self.fill_value``.
1019 
1020             For other dtypes, this only changes the dtype of
1021             ``self.sp_values``.
1022 
1023         copy : bool, default True
1024             Whether to ensure a copy is made, even if not necessary.
1025 
1026         Returns
1027         -------
1028         SparseArray
1029 
1030         Examples
1031         --------
1032         >>> arr = pd.arrays.SparseArray([0, 0, 1, 2])
1033         >>> arr
1034         [0, 0, 1, 2]
1035         Fill: 0
1036         IntIndex
1037         Indices: array([2, 3], dtype=int32)
1038 
1039         >>> arr.astype(np.dtype('int32'))
1040         [0, 0, 1, 2]
1041         Fill: 0
1042         IntIndex
1043         Indices: array([2, 3], dtype=int32)
1044 
1045         Using a NumPy dtype with a different kind (e.g. float) will coerce
1046         just ``self.sp_values``.
1047 
1048         >>> arr.astype(np.dtype('float64'))
1049         ... # doctest: +NORMALIZE_WHITESPACE
1050         [0.0, 0.0, 1.0, 2.0]
1051         Fill: 0.0
1052         IntIndex
1053         Indices: array([2, 3], dtype=int32)
1054 
1055         Use a SparseDtype if you wish to be change the fill value as well.
1056 
1057         >>> arr.astype(SparseDtype("float64", fill_value=np.nan))
1058         ... # doctest: +NORMALIZE_WHITESPACE
1059         [nan, nan, 1.0, 2.0]
1060         Fill: nan
1061         IntIndex
1062         Indices: array([2, 3], dtype=int32)
1063         """
1064         dtype = self.dtype.update_dtype(dtype)
1065         subtype = dtype._subtype_with_str
1066         # TODO copy=False is broken for astype_nansafe with int -> float, so cannot
1067         # passthrough copy keyword: https://github.com/pandas-dev/pandas/issues/34456
1068         sp_values = astype_nansafe(self.sp_values, subtype, copy=True)
1069         if sp_values is self.sp_values and copy:
1070             sp_values = sp_values.copy()
1071 
1072         return self._simple_new(sp_values, self.sp_index, dtype)

1075         """
1076         Map categories using input correspondence (dict, Series, or function).
1077 
1078         Parameters
1079         ----------

1081             The correspondence from old values to new.

1339         )
1340         if result is not NotImplemented:
1341             return result
1342 
1343         if len(inputs) == 1:
1344             # No alignment necessary.
1345             sp_values = getattr(ufunc, method)(self.sp_values, **kwargs)
1346             fill_value = getattr(ufunc, method)(self.fill_value, **kwargs)
1347 
1348             if isinstance(sp_values, tuple):
1349                 # multiple outputs. e.g. modf
1350                 arrays = tuple(
1351                     self._simple_new(
1352                         sp_value, self.sp_index, SparseDtype(sp_value.dtype, fv)
1353                     )
1354                     for sp_value, fv in zip(sp_values, fill_value)
1355                 )
1356                 return arrays
1357             elif is_scalar(sp_values):
1358                 # e.g. reductions
1359                 return sp_values
1360 
1361             return self._simple_new(
1362                 sp_values, self.sp_index, SparseDtype(sp_values.dtype, fill_value)
1363             )
1364 
1365         result = getattr(ufunc, method)(*[np.asarray(x) for x in inputs], **kwargs)
1366         if out:
1367             if len(out) == 1:
1368                 out = out[0]
1369             return out
1370 
1371         if type(result) is tuple:
1372             return tuple(type(self)(x) for x in result)
1373         elif method == "at":
1374             # no return value
1375             return None
1376         else:
1377             return type(self)(result)
1378

1380         return np.abs(self)

1387     def _create_unary_method(cls, op) -> Callable[["SparseArray"], "SparseArray"]:
1388         def sparse_unary_method(self) -> "SparseArray":
1389             fill_value = op(np.array(self.fill_value)).item()
1390             values = op(self.sp_values)

1397     @classmethod
1398     def _create_arithmetic_method(cls, op):
1399         op_name = op.__name__
1400 
1401         @unpack_zerodim_and_defer(op_name)
1402         def sparse_arithmetic_method(self, other):
1403 
1404             if isinstance(other, SparseArray):
1405                 return _sparse_array_op(self, other, op, op_name)
1406 
1407             elif is_scalar(other):
1408                 with np.errstate(all="ignore"):
1409                     fill = op(_get_fill(self), np.asarray(other))
1410                     result = op(self.sp_values, other)
1411 
1412                 if op_name == "divmod":
1413                     left, right = result
1414                     lfill, rfill = fill
1415                     return (
1416                         _wrap_result(op_name, left, self.sp_index, lfill),
1417                         _wrap_result(op_name, right, self.sp_index, rfill),
1418                     )
1419 
1420                 return _wrap_result(op_name, result, self.sp_index, fill)
1421 
1422             else:
1423                 other = np.asarray(other)
1424                 with np.errstate(all="ignore"):
1425                     # TODO: look into _wrap_result
1426                     if len(self) != len(other):
1427                         raise AssertionError(
1428                             (f"length mismatch: {len(self)} vs. {len(other)}")
1429                         )
1430                     if not isinstance(other, SparseArray):
1431                         dtype = getattr(other, "dtype", None)
1432                         other = SparseArray(
1433                             other, fill_value=self.fill_value, dtype=dtype
1434                         )
1435                     return _sparse_array_op(self, other, op, op_name)
1436 
1437         name = f"__{op.__name__}__"
1438         return compat.set_function_name(sparse_arithmetic_method, name, cls)

1440     @classmethod
1441     def _create_comparison_method(cls, op):
1442         op_name = op.__name__
1443         if op_name in {"and_", "or_"}:
1444             op_name = op_name[:-1]
1445 
1446         @unpack_zerodim_and_defer(op_name)
1447         def cmp_method(self, other):
1448 
1449             if not is_scalar(other) and not isinstance(other, type(self)):
1450                 # convert list-like to ndarray
1451                 other = np.asarray(other)
1452 
1453             if isinstance(other, np.ndarray):
1454                 # TODO: make this more flexible than just ndarray...
1455                 if len(self) != len(other):
1456                     raise AssertionError(
1457                         f"length mismatch: {len(self)} vs. {len(other)}"
1458                     )
1459                 other = SparseArray(other, fill_value=self.fill_value)
1460 
1461             if isinstance(other, SparseArray):
1462                 return _sparse_array_op(self, other, op, op_name)
1463             else:
1464                 with np.errstate(all="ignore"):
1465                     fill_value = op(self.fill_value, other)
1466                     result = op(self.sp_values, other)
1467 
1468                 return type(self)(
1469                     result,
1470                     sparse_index=self.sp_index,
1471                     fill_value=fill_value,
1472                     dtype=np.bool_,
1473                 )
1474 
1475         name = f"__{op.__name__}__"
1476         return compat.set_function_name(cmp_method, name, cls)

1511 def make_sparse(arr: np.ndarray, kind="block", fill_value=None, dtype=None, copy=False):
1512     """
1513     Convert ndarray to sparse format
1514 
1515     Parameters
1516     ----------
1517     arr : ndarray
1518     kind : {'block', 'integer'}
1519     fill_value : NaN or another value
1520     dtype : np.dtype, optional
1521     copy : bool, default False
1522 
1523     Returns
1524     -------
1525     (sparse_values, index, fill_value) : (ndarray, SparseIndex, Scalar)
1526     """
1527     assert isinstance(arr, np.ndarray)
1528 
1529     if arr.ndim > 1:
1530         raise TypeError("expected dimension <= 1 data")
1531 
1532     if fill_value is None:
1533         fill_value = na_value_for_dtype(arr.dtype)
1534 
1535     if isna(fill_value):
1536         mask = notna(arr)
1537     else:
1538         # cast to object comparison to be safe
1539         if is_string_dtype(arr.dtype):
1540             arr = arr.astype(object)
1541 
1542         if is_object_dtype(arr.dtype):
1543             # element-wise equality check method in numpy doesn't treat
1544             # each element type, eg. 0, 0.0, and False are treated as
1545             # same. So we have to check the both of its type and value.
1546             mask = splib.make_mask_object_ndarray(arr, fill_value)
1547         else:
1548             mask = arr != fill_value
1549 
1550     length = len(arr)
1551     if length != len(mask):
1552         # the arr is a SparseArray
1553         indices = mask.sp_index.indices
1554     else:
1555         indices = mask.nonzero()[0].astype(np.int32)
1556 
1557     index = _make_index(length, indices, kind)
1558     sparsified_values = arr[mask]
1559     if dtype is not None:
1560         sparsified_values = astype_nansafe(sparsified_values, dtype=dtype)
1561     # TODO: copy
1562     return sparsified_values, index, fill_value
1563 
1564 
1565 def _make_index(length, indices, kind):
1566 
1567     if kind == "block" or isinstance(kind, BlockIndex):
1568         locs, lens = splib.get_blocks(indices)
1569         index = BlockIndex(length, locs, lens)
1570     elif kind == "integer" or isinstance(kind, IntIndex):
1571         index = IntIndex(length, indices)
1572     else:  # pragma: no cover
1573         raise ValueError("must be block or integer type")
1574     return index
