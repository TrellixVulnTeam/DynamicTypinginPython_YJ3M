  10 def _check_is_partition(parts, whole):
  11     whole = set(whole)
  12     parts = [set(x) for x in parts]
  13     if set.intersection(*parts) != set():
  14         raise ValueError("Is not a partition because intersection is not null.")
  15     if set.union(*parts) != whole:
  16         raise ValueError("Is not a partition because union is not the whole.")

  19 def _to_ijv(ss, row_levels=(0,), column_levels=(1,), sort_labels=False):
  20     """
  21     For arbitrary (MultiIndexed) sparse Series return
  22     (v, i, j, ilabels, jlabels) where (v, (i, j)) is suitable for
  23     passing to scipy.sparse.coo constructor.
  24     """
  25     # index and column levels must be a partition of the index
  26     _check_is_partition([row_levels, column_levels], range(ss.index.nlevels))
  27 
  28     # from the sparse Series: get the labels and data for non-null entries
  29     values = ss.array._valid_sp_values
  30 
  31     nonnull_labels = ss.dropna()
  32 
  33     def get_indexers(levels):
  34         """ Return sparse coords and dense labels for subset levels """
  35         # TODO: how to do this better? cleanly slice nonnull_labels given the
  36         # coord
  37         values_ilabels = [tuple(x[i] for i in levels) for x in nonnull_labels.index]
  38         if len(levels) == 1:
  39             values_ilabels = [x[0] for x in values_ilabels]
  40 
  41         # # performance issues with groupby ###################################
  42         # TODO: these two lines can replace the code below but
  43         # groupby is too slow (in some cases at least)
  44         # labels_to_i = ss.groupby(level=levels, sort=sort_labels).first()
  45         # labels_to_i[:] = np.arange(labels_to_i.shape[0])
  46 
  47         def _get_label_to_i_dict(labels, sort_labels=False):
  48             """
  49             Return dict of unique labels to number.
  50             Optionally sort by label.
  51             """
  52             labels = Index(map(tuple, labels)).unique().tolist()  # squish
  53             if sort_labels:
  54                 labels = sorted(labels)
  55             return {k: i for i, k in enumerate(labels)}
  56 
  57         def _get_index_subset_to_coord_dict(index, subset, sort_labels=False):
  58             ilabels = list(zip(*[index._get_level_values(i) for i in subset]))
  59             labels_to_i = _get_label_to_i_dict(ilabels, sort_labels=sort_labels)
  60             labels_to_i = Series(labels_to_i)
  61             if len(subset) > 1:
  62                 labels_to_i.index = MultiIndex.from_tuples(labels_to_i.index)
  63                 labels_to_i.index.names = [index.names[i] for i in subset]
  64             else:
  65                 labels_to_i.index = Index(x[0] for x in labels_to_i.index)
  66                 labels_to_i.index.name = index.names[subset[0]]
  67 
  68             labels_to_i.name = "value"
  69             return labels_to_i
  70 
  71         labels_to_i = _get_index_subset_to_coord_dict(
  72             ss.index, levels, sort_labels=sort_labels
  73         )
  74         # #####################################################################
  75         # #####################################################################
  76 
  77         i_coord = labels_to_i[values_ilabels].tolist()

  79 
  80         return i_coord, i_labels
  81 
  82     i_coord, i_labels = get_indexers(row_levels)
  83     j_coord, j_labels = get_indexers(column_levels)
  84 
  85     return values, i_coord, j_coord, i_labels, j_labels

  88 def _sparse_series_to_coo(ss, row_levels=(0,), column_levels=(1,), sort_labels=False):
  89     """
  90     Convert a sparse Series to a scipy.sparse.coo_matrix using index
  91     levels row_levels, column_levels as the row and column
  92     labels respectively. Returns the sparse_matrix, row and column labels.
  93     """
  94     import scipy.sparse
  95 
  96     if ss.index.nlevels < 2:
  97         raise ValueError("to_coo requires MultiIndex with nlevels > 2")
  98     if not ss.index.is_unique:
  99         raise ValueError(
 100             "Duplicate index entries are not allowed in to_coo transformation."
 101         )
 102 
 103     # to keep things simple, only rely on integer indexing (not labels)
 104     row_levels = [ss.index._get_level_number(x) for x in row_levels]
 105     column_levels = [ss.index._get_level_number(x) for x in column_levels]
 106 
 107     v, i, j, rows, columns = _to_ijv(
 108         ss, row_levels=row_levels, column_levels=column_levels, sort_labels=sort_labels
 109     )
 110     sparse_matrix = scipy.sparse.coo_matrix(
 111         (v, (i, j)), shape=(len(rows), len(columns))
 112     )
 113     return sparse_matrix, rows, columns

 116 def _coo_to_sparse_series(A, dense_index: bool = False):
 117     """
 118     Convert a scipy.sparse.coo_matrix to a SparseSeries.
 119 
 120     Parameters
 121     ----------
 122     A : scipy.sparse.coo.coo_matrix
 123     dense_index : bool, default False
 124 
 125     Returns
 126     -------
 127     Series
 128 
 129     Raises
 130     ------
 131     TypeError if A is not a coo_matrix
 132     """
 133     from pandas import SparseDtype
 134 
 135     try:
 136         s = Series(A.data, MultiIndex.from_arrays((A.row, A.col)))
 137     except AttributeError as err:
 138         raise TypeError(
 139             f"Expected coo_matrix. Got {type(A).__name__} instead."
 140         ) from err
 141     s = s.sort_index()
 142     s = s.astype(SparseDtype(s.dtype))
 143     if dense_index:
 144         # is there a better constructor method to use here?
 145         i = range(A.shape[0])
 146         j = range(A.shape[1])
 147         ind = MultiIndex.from_product([i, j])
 148         s = s.reindex(ind)
 149     return s
