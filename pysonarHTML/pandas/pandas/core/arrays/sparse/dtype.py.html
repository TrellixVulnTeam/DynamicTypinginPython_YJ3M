  29 @register_extension_dtype
  30 class SparseDtype(ExtensionDtype):
  31     """
  32     Dtype for data stored in :class:`SparseArray`.
  33 
  34     This dtype implements the pandas ExtensionDtype interface.
  35 
  36     .. versionadded:: 0.24.0
  37 
  38     Parameters
  39     ----------
  40     dtype : str, ExtensionDtype, numpy.dtype, type, default numpy.float64
  41         The dtype of the underlying array storing the non-fill value values.
  42     fill_value : scalar, optional
  43         The scalar value not stored in the SparseArray. By default, this
  44         depends on `dtype`.
  45 
  46         =========== ==========
  47         dtype       na_value
  48         =========== ==========
  49         float       ``np.nan``
  50         int         ``0``
  51         bool        ``False``
  52         datetime64  ``pd.NaT``
  53         timedelta64 ``pd.NaT``
  54         =========== ==========
  55 
  56         The default value may be overridden by specifying a `fill_value`.
  57 
  58     Attributes
  59     ----------
  60     None
  61 
  62     Methods
  63     -------
  64     None
  65     """
  66 
  67     # We include `_is_na_fill_value` in the metadata to avoid hash collisions
  68     # between SparseDtype(float, 0.0) and SparseDtype(float, nan).
  69     # Without is_na_fill_value in the comparison, those would be equal since
  70     # hash(nan) is (sometimes?) 0.
  71     _metadata = ("_dtype", "_fill_value", "_is_na_fill_value")
  72 
  73     def __init__(self, dtype: Dtype = np.float64, fill_value: Any = None):
  74 
  75         if isinstance(dtype, type(self)):
  76             if fill_value is None:
  77                 fill_value = dtype.fill_value
  78             dtype = dtype.subtype
  79 
  80         dtype = pandas_dtype(dtype)
  81         if is_string_dtype(dtype):
  82             dtype = np.dtype("object")
  83 
  84         if fill_value is None:
  85             fill_value = na_value_for_dtype(dtype)
  86 
  87         if not is_scalar(fill_value):
  88             raise ValueError(f"fill_value must be a scalar. Got {fill_value} instead")
  89         self._dtype = dtype
  90         self._fill_value = fill_value
  91 
  92     def __hash__(self):
  93         # Python3 doesn't inherit __hash__ when a base class overrides
  94         # __eq__, so we explicitly do it here.
  95         return super().__hash__()
  96 
  97     def __eq__(self, other: Any) -> bool:
  98         # We have to override __eq__ to handle NA values in _metadata.
  99         # The base class does simple == checks, which fail for NA.
 100         if isinstance(other, str):
 101             try:
 102                 other = self.construct_from_string(other)
 103             except TypeError:
 104                 return False
 105 
 106         if isinstance(other, type(self)):
 107             subtype = self.subtype == other.subtype
 108             if self._is_na_fill_value:
 109                 # this case is complicated by two things:
 110                 # SparseDtype(float, float(nan)) == SparseDtype(float, np.nan)
 111                 # SparseDtype(float, np.nan)     != SparseDtype(float, pd.NaT)
 112                 # i.e. we want to treat any floating-point NaN as equal, but
 113                 # not a floating-point NaN and a datetime NaT.
 114                 fill_value = (
 115                     other._is_na_fill_value
 116                     and isinstance(self.fill_value, type(other.fill_value))
 117                     or isinstance(other.fill_value, type(self.fill_value))
 118                 )
 119             else:
 120                 fill_value = self.fill_value == other.fill_value
 121 
 122             return subtype and fill_value
 123         return False
 124 
 125     @property
 126     def fill_value(self):
 127         """
 128         The fill value of the array.
 129 
 130         Converting the SparseArray to a dense ndarray will fill the
 131         array with this value.
 132 
 133         .. warning::
 134 
 135            It's possible to end up with a SparseArray that has ``fill_value``
 136            values in ``sp_values``. This can occur, for example, when setting
 137            ``SparseArray.fill_value`` directly.
 138         """
 139         return self._fill_value

 188     @classmethod
 189     def construct_from_string(cls, string: str) -> "SparseDtype":
 190         """
 191         Construct a SparseDtype from a string form.
 192 
 193         Parameters
 194         ----------
 195         string : str
 196             Can take the following forms.
 197 
 198             string           dtype
 199             ================ ============================
 200             'int'            SparseDtype[np.int64, 0]
 201             'Sparse'         SparseDtype[np.float64, nan]
 202             'Sparse[int]'    SparseDtype[np.int64, 0]
 203             'Sparse[int, 0]' SparseDtype[np.int64, 0]
 204             ================ ============================
 205 
 206             It is not possible to specify non-default fill values
 207             with a string. An argument like ``'Sparse[int, 1]'``
 208             will raise a ``TypeError`` because the default fill value
 209             for integers is 0.
 210 
 211         Returns
 212         -------
 213         SparseDtype
 214         """
 215         if not isinstance(string, str):
 216             raise TypeError(
 217                 f"'construct_from_string' expects a string, got {type(string)}"
 218             )
 219         msg = f"Cannot construct a 'SparseDtype' from '{string}'"
 220         if string.startswith("Sparse"):
 221             try:
 222                 sub_type, has_fill_value = cls._parse_subtype(string)
 223             except ValueError as err:
 224                 raise TypeError(msg) from err
 225             else:
 226                 result = SparseDtype(sub_type)
 227                 msg = (
 228                     f"Cannot construct a 'SparseDtype' from '{string}'.\n\nIt "
 229                     "looks like the fill_value in the string is not "
 230                     "the default for the dtype. Non-default fill_values "
 231                     "are not supported. Use the 'SparseDtype()' "
 232                     "constructor instead."
 233                 )
 234                 if has_fill_value and str(result) != string:
 235                     raise TypeError(msg)
 236                 return result
 237         else:
 238             raise TypeError(msg)
 239 
 240     @staticmethod
 241     def _parse_subtype(dtype: str) -> Tuple[str, bool]:
 242         """
 243         Parse a string to get the subtype
 244 
 245         Parameters
 246         ----------
 247         dtype : str
 248             A string like
 249 
 250             * Sparse[subtype]
 251             * Sparse[subtype, fill_value]
 252 
 253         Returns
 254         -------
 255         subtype : str
 256 
 257         Raises
 258         ------
 259         ValueError
 260             When the subtype cannot be extracted.
 261         """
 262         xpr = re.compile(r"Sparse\[(?P<subtype>[^,]*)(, )?(?P<fill_value>.*?)?\]$")
 263         m = xpr.match(dtype)
 264         has_fill_value = False
 265         if m:
 266             subtype = m.groupdict()["subtype"]
 267             has_fill_value = bool(m.groupdict()["fill_value"])
 268         elif dtype == "Sparse":
 269             subtype = "float64"
 270         else:
 271             raise ValueError(f"Cannot parse {dtype}")
 272         return subtype, has_fill_value
 273 
 274     @classmethod
 275     def is_dtype(cls, dtype: object) -> bool:
 276         dtype = getattr(dtype, "dtype", dtype)
 277         if isinstance(dtype, str) and dtype.startswith("Sparse"):
 278             sub_type, _ = cls._parse_subtype(dtype)
 279             dtype = np.dtype(sub_type)
 280         elif isinstance(dtype, cls):
 281             return True
 282         return isinstance(dtype, np.dtype) or dtype == "Sparse"

 284     def update_dtype(self, dtype):
 285         """
 286         Convert the SparseDtype to a new dtype.
 287 
 288         This takes care of converting the ``fill_value``.
 289 
 290         Parameters
 291         ----------
 292         dtype : Union[str, numpy.dtype, SparseDtype]
 293             The new dtype to use.
 294 
 295             * For a SparseDtype, it is simply returned
 296             * For a NumPy dtype (or str), the current fill value
 297               is converted to the new dtype, and a SparseDtype
 298               with `dtype` and the new fill value is returned.
 299 
 300         Returns
 301         -------
 302         SparseDtype
 303             A new SparseDtype with the correct `dtype` and fill value
 304             for that `dtype`.
 305 
 306         Raises
 307         ------
 308         ValueError
 309             When the current fill value cannot be converted to the
 310             new `dtype` (e.g. trying to convert ``np.nan`` to an
 311             integer dtype).
 312 
 313 
 314         Examples
 315         --------
 316         >>> SparseDtype(int, 0).update_dtype(float)
 317         Sparse[float64, 0.0]
 318 
 319         >>> SparseDtype(int, 1).update_dtype(SparseDtype(float, np.nan))
 320         Sparse[float64, nan]
 321         """
 322         cls = type(self)
 323         dtype = pandas_dtype(dtype)
 324 
 325         if not isinstance(dtype, cls):
 326             if is_extension_array_dtype(dtype):
 327                 raise TypeError("sparse arrays of extension dtypes not supported")
 328 
 329             fill_value = astype_nansafe(np.array(self.fill_value), dtype).item()
 330             dtype = cls(dtype, fill_value=fill_value)
 331 
 332         return dtype

 334     @property
 335     def _subtype_with_str(self):
 336         """
 337         Whether the SparseDtype's subtype should be considered ``str``.
 338 
 339         Typically, pandas will store string data in an object-dtype array.
 340         When converting values to a dtype, e.g. in ``.astype``, we need to
 341         be more specific, we need the actual underlying type.
 342 
 343         Returns
 344         -------
 345         >>> SparseDtype(int, 1)._subtype_with_str
 346         dtype('int64')
 347 
 348         >>> SparseDtype(object, 1)._subtype_with_str
 349         dtype('O')
 350 
 351         >>> dtype = SparseDtype(str, '')
 352         >>> dtype.subtype
 353         dtype('O')
 354 
 355         >>> dtype._subtype_with_str
 356         <class 'str'>
 357         """
 358         if isinstance(self.fill_value, str):
 359             return type(self.fill_value)
 360         return self.subtype

 362     def _get_common_dtype(self, dtypes: List[DtypeObj]) -> Optional[DtypeObj]:
 363         # TODO for now only handle SparseDtypes and numpy dtypes => extend
 364         # with other compatibtle extension dtypes
 365         if any(
 366             isinstance(x, ExtensionDtype) and not isinstance(x, SparseDtype)
 367             for x in dtypes
 368         ):
 369             return None
 370 
 371         fill_values = [x.fill_value for x in dtypes if isinstance(x, SparseDtype)]
 372         fill_value = fill_values[0]
 373 
 374         # np.nan isn't a singleton, so we may end up with multiple
 375         # NaNs here, so we ignore tha all NA case too.
 376         if not (len(set(fill_values)) == 1 or isna(fill_values).all()):
 377             warnings.warn(
 378                 "Concatenating sparse arrays with multiple fill "
 379                 f"values: '{fill_values}'. Picking the first and "
 380                 "converting the rest.",
 381                 PerformanceWarning,
 382                 stacklevel=6,
 383             )
 384 
 385         np_dtypes = [x.subtype if isinstance(x, SparseDtype) else x for x in dtypes]
 386         return SparseDtype(np.find_common_type(np_dtypes, []), fill_value=fill_value)
