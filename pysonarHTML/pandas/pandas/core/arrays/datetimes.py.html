 211     def __init__(self, values, dtype=DT64NS_DTYPE, freq=None, copy=False):
 212         if isinstance(values, (ABCSeries, ABCIndexClass)):
 213             values = values._values
 214 
 215         inferred_freq = getattr(values, "_freq", None)
 216 
 217         if isinstance(values, type(self)):
 218             # validation
 219             dtz = getattr(dtype, "tz", None)
 220             if dtz and values.tz is None:
 221                 dtype = DatetimeTZDtype(tz=dtype.tz)
 222             elif dtz and values.tz:
 223                 if not timezones.tz_compare(dtz, values.tz):
 224                     msg = (
 225                         "Timezone of the array and 'dtype' do not match. "
 226                         f"'{dtz}' != '{values.tz}'"
 227                     )
 228                     raise TypeError(msg)
 229             elif values.tz:
 230                 dtype = values.dtype
 231 
 232             if freq is None:
 233                 freq = values.freq
 234             values = values._data
 235 
 236         if not isinstance(values, np.ndarray):
 237             raise ValueError(
 238                 f"Unexpected type '{type(values).__name__}'. 'values' must be "
 239                 "a DatetimeArray ndarray, or Series or Index containing one of those."
 240             )
 241         if values.ndim not in [1, 2]:
 242             raise ValueError("Only 1-dimensional input arrays are supported.")
 243 
 244         if values.dtype == "i8":
 245             # for compat with datetime/timedelta/period shared methods,
 246             #  we can sometimes get here with int64 values.  These represent
 247             #  nanosecond UTC (or tz-naive) unix timestamps
 248             values = values.view(DT64NS_DTYPE)
 249 
 250         if values.dtype != DT64NS_DTYPE:
 251             raise ValueError(
 252                 "The dtype of 'values' is incorrect. Must be 'datetime64[ns]'. "
 253                 f"Got {values.dtype} instead."
 254             )
 255 
 256         dtype = _validate_dt64_dtype(dtype)
 257 
 258         if freq == "infer":
 259             raise ValueError(
 260                 "Frequency inference not allowed in DatetimeArray.__init__. "
 261                 "Use 'pd.array()' instead."
 262             )
 263 
 264         if copy:
 265             values = values.copy()
 266         if freq:
 267             freq = to_offset(freq)
 268         if getattr(dtype, "tz", None):
 269             # https://github.com/pandas-dev/pandas/issues/18595
 270             # Ensure that we have a standard timezone for pytz objects.
 271             # Without this, things like adding an array of timedeltas and
 272             # a  tz-aware Timestamp (with a tz specific to its datetime) will
 273             # be incorrect(ish?) for the array as a whole
 274             dtype = DatetimeTZDtype(tz=timezones.tz_standardize(dtype.tz))
 275 
 276         self._data = values
 277         self._dtype = dtype
 278         self._freq = freq
 279 
 280         if inferred_freq is None and freq is not None:
 281             type(self)._validate_frequency(self, freq)

 284     def _simple_new(cls, values, freq=None, dtype=DT64NS_DTYPE):
 285         assert isinstance(values, np.ndarray)
 286         if values.dtype != DT64NS_DTYPE:
 287             assert values.dtype == "i8"
 288             values = values.view(DT64NS_DTYPE)
 289 
 290         result = object.__new__(cls)
 291         result._data = values
 292         result._freq = freq
 293         result._dtype = dtype
 294         return result

 296     @classmethod
 297     def _from_sequence(
 298         cls,
 299         data,
 300         dtype=None,
 301         copy=False,
 302         tz=None,
 303         freq=lib.no_default,
 304         dayfirst=False,
 305         yearfirst=False,
 306         ambiguous="raise",
 307     ):
 308         explicit_none = freq is None
 309         freq = freq if freq is not lib.no_default else None
 310 
 311         freq, freq_infer = dtl.maybe_infer_freq(freq)
 312 
 313         subarr, tz, inferred_freq = sequence_to_dt64ns(
 314             data,
 315             dtype=dtype,
 316             copy=copy,
 317             tz=tz,
 318             dayfirst=dayfirst,
 319             yearfirst=yearfirst,
 320             ambiguous=ambiguous,
 321         )
 322 
 323         freq, freq_infer = dtl.validate_inferred_freq(freq, inferred_freq, freq_infer)
 324         if explicit_none:
 325             freq = None
 326 
 327         dtype = tz_to_dtype(tz)
 328         result = cls._simple_new(subarr, freq=freq, dtype=dtype)
 329 
 330         if inferred_freq is None and freq is not None:
 331             # this condition precludes `freq_infer`
 332             cls._validate_frequency(result, freq, ambiguous=ambiguous)
 333 
 334         elif freq_infer:
 335             # Set _freq directly to bypass duplicative _validate_frequency
 336             # check.
 337             result._freq = to_offset(result.inferred_freq)
 338 
 339         return result

 341     @classmethod
 342     def _generate_range(
 343         cls,
 344         start,
 345         end,
 346         periods,
 347         freq,
 348         tz=None,
 349         normalize=False,
 350         ambiguous="raise",
 351         nonexistent="raise",
 352         closed=None,
 353     ):
 354 
 355         periods = dtl.validate_periods(periods)
 356         if freq is None and any(x is None for x in [periods, start, end]):
 357             raise ValueError("Must provide freq argument if no data is supplied")
 358 
 359         if com.count_not_none(start, end, periods, freq) != 3:
 360             raise ValueError(
 361                 "Of the four parameters: start, end, periods, "
 362                 "and freq, exactly three must be specified"
 363             )
 364         freq = to_offset(freq)
 365 
 366         if start is not None:
 367             start = Timestamp(start)
 368 
 369         if end is not None:
 370             end = Timestamp(end)
 371 
 372         if start is NaT or end is NaT:
 373             raise ValueError("Neither `start` nor `end` can be NaT")
 374 
 375         left_closed, right_closed = dtl.validate_endpoints(closed)
 376         start, end, _normalized = _maybe_normalize_endpoints(start, end, normalize)
 377         tz = _infer_tz_from_endpoints(start, end, tz)
 378 
 379         if tz is not None:
 380             # Localize the start and end arguments
 381             start_tz = None if start is None else start.tz
 382             end_tz = None if end is None else end.tz
 383             start = _maybe_localize_point(
 384                 start, start_tz, start, freq, tz, ambiguous, nonexistent
 385             )
 386             end = _maybe_localize_point(
 387                 end, end_tz, end, freq, tz, ambiguous, nonexistent
 388             )
 389         if freq is not None:
 390             # We break Day arithmetic (fixed 24 hour) here and opt for
 391             # Day to mean calendar day (23/24/25 hour). Therefore, strip
 392             # tz info from start and day to avoid DST arithmetic
 393             if isinstance(freq, Day):
 394                 if start is not None:
 395                     start = start.tz_localize(None)
 396                 if end is not None:
 397                     end = end.tz_localize(None)
 398 
 399             if isinstance(freq, Tick):
 400                 values = generate_regular_range(start, end, periods, freq)
 401             else:
 402                 xdr = generate_range(start=start, end=end, periods=periods, offset=freq)
 403                 values = np.array([x.value for x in xdr], dtype=np.int64)
 404 
 405             _tz = start.tz if start is not None else end.tz
 406             index = cls._simple_new(values, freq=freq, dtype=tz_to_dtype(_tz))
 407 
 408             if tz is not None and index.tz is None:
 409                 arr = tzconversion.tz_localize_to_utc(
 410                     index.asi8, tz, ambiguous=ambiguous, nonexistent=nonexistent
 411                 )
 412 
 413                 index = cls(arr)
 414 
 415                 # index is localized datetime64 array -> have to convert
 416                 # start/end as well to compare
 417                 if start is not None:
 418                     start = start.tz_localize(tz).asm8
 419                 if end is not None:
 420                     end = end.tz_localize(tz).asm8
 421         else:
 422             # Create a linearly spaced date_range in local time
 423             # Nanosecond-granularity timestamps aren't always correctly
 424             # representable with doubles, so we limit the range that we
 425             # pass to np.linspace as much as possible
 426             arr = (
 427                 np.linspace(0, end.value - start.value, periods, dtype="int64")
 428                 + start.value
 429             )
 430             dtype = tz_to_dtype(tz)
 431             index = cls._simple_new(
 432                 arr.astype("M8[ns]", copy=False), freq=None, dtype=dtype
 433             )
 434 
 435         if not left_closed and len(index) and index[0] == start:
 436             index = index[1:]
 437         if not right_closed and len(index) and index[-1] == end:
 438             index = index[:-1]
 439 
 440         dtype = tz_to_dtype(tz)
 441         return cls._simple_new(index.asi8, freq=freq, dtype=dtype)

 553     def __iter__(self):
 554         """
 555         Return an iterator over the boxed values
 556 
 557         Yields
 558         ------
 559         tstamp : Timestamp
 560         """
 561 
 562         # convert in chunks of 10k for efficiency
 563         data = self.asi8
 564         length = len(self)
 565         chunksize = 10000
 566         chunks = int(length / chunksize) + 1
 567         for i in range(chunks):
 568             start_i = i * chunksize
 569             end_i = min((i + 1) * chunksize, length)
 570             converted = tslib.ints_to_pydatetime(
 571                 data[start_i:end_i], tz=self.tz, freq=self.freq, box="timestamp"
 572             )
 573             for v in converted:

 576     def astype(self, dtype, copy=True):
 577         # We handle
 578         #   --> datetime
 579         #   --> period
 580         # DatetimeLikeArrayMixin Super handles the rest.
 581         dtype = pandas_dtype(dtype)
 582 
 583         if is_datetime64_ns_dtype(dtype) and not is_dtype_equal(dtype, self.dtype):
 584             # GH#18951: datetime64_ns dtype but not equal means different tz
 585             new_tz = getattr(dtype, "tz", None)
 586             if getattr(self.dtype, "tz", None) is None:
 587                 return self.tz_localize(new_tz)
 588             result = self.tz_convert(new_tz)
 589             if copy:
 590                 result = result.copy()
 591             if new_tz is None:
 592                 # Do we want .astype('datetime64[ns]') to be an ndarray.
 593                 # The astype in Block._astype expects this to return an
 594                 # ndarray, but we could maybe work around it there.
 595                 result = result._data
 596             return result
 597         elif is_datetime64tz_dtype(self.dtype) and is_dtype_equal(self.dtype, dtype):
 598             if copy:
 599                 return self.copy()
 600             return self
 601         elif is_period_dtype(dtype):
 602             return self.to_period(freq=dtype.freq)
 603         return dtl.DatetimeLikeArrayMixin.astype(self, dtype, copy)

 655         """subtract DatetimeArray/Index or ndarray[datetime64]"""
 656         if len(self) != len(other):
 657             raise ValueError("cannot add indices of unequal length")
 658 
 659         if isinstance(other, np.ndarray):
 660             assert is_datetime64_dtype(other)
 661             other = type(self)(other)
 662 
 663         if not self._has_same_tz(other):
 664             # require tz compat
 665             raise TypeError(
 666                 f"{type(self).__name__} subtraction must have the same "
 667                 "timezones or no timezones"
 668             )
 669 
 670         self_i8 = self.asi8
 671         other_i8 = other.asi8
 672         arr_mask = self._isnan | other._isnan
 673         new_values = checked_add_with_arr(self_i8, -other_i8, arr_mask=arr_mask)
 674         if self._hasnans or other._hasnans:
 675             new_values[arr_mask] = iNaT
 676         return new_values.view("timedelta64[ns]")
 677 
 678     def _add_offset(self, offset):
 679         if self.ndim == 2:
 680             return self.ravel()._add_offset(offset).reshape(self.shape)
 681 
 682         assert not isinstance(offset, Tick)
 683         try:
 684             if self.tz is not None:
 685                 values = self.tz_localize(None)
 686             else:
 687                 values = self
 688             result = offset.apply_index(values)
 689             result = DatetimeArray._simple_new(result)
 690             result = result.tz_localize(self.tz)
 691 
 692         except NotImplementedError:
 693             warnings.warn(
 694                 "Non-vectorized DateOffset being applied to Series or DatetimeIndex",
 695                 PerformanceWarning,
 696             )
 697             result = self.astype("O") + offset
 698             if not len(self):
 699                 # GH#30336 _from_sequence won't be able to infer self.tz
 700                 return type(self)._from_sequence(result).tz_localize(self.tz)
 701 
 702         return type(self)._from_sequence(result)

 704     def _sub_datetimelike_scalar(self, other):
 705         # subtract a datetime from myself, yielding a ndarray[timedelta64[ns]]
 706         assert isinstance(other, (datetime, np.datetime64))
 707         assert other is not NaT
 708         other = Timestamp(other)
 709         if other is NaT:
 710             return self - NaT
 711 
 712         if not self._has_same_tz(other):
 713             # require tz compat
 714             raise TypeError(
 715                 "Timestamp subtraction must have the same timezones or no timezones"
 716             )
 717 
 718         i8 = self.asi8
 719         result = checked_add_with_arr(i8, -other.value, arr_mask=self._isnan)
 720         result = self._maybe_mask_results(result)
 721         return result.view("timedelta64[ns]")

 735     def tz_convert(self, tz):
 736         """
 737         Convert tz-aware Datetime Array/Index from one time zone to another.
 738 
 739         Parameters
 740         ----------
 741         tz : str, pytz.timezone, dateutil.tz.tzfile or None
 742             Time zone for time. Corresponding timestamps would be converted
 743             to this time zone of the Datetime Array/Index. A `tz` of None will
 744             convert to UTC and remove the timezone information.
 745 
 746         Returns
 747         -------
 748         Array or Index
 749 
 750         Raises
 751         ------
 752         TypeError
 753             If Datetime Array/Index is tz-naive.
 754 
 755         See Also
 756         --------
 757         DatetimeIndex.tz : A timezone that has a variable offset from UTC.
 758         DatetimeIndex.tz_localize : Localize tz-naive DatetimeIndex to a
 759             given time zone, or remove timezone from a tz-aware DatetimeIndex.
 760 
 761         Examples
 762         --------
 763         With the `tz` parameter, we can change the DatetimeIndex
 764         to other time zones:
 765 
 766         >>> dti = pd.date_range(start='2014-08-01 09:00',
 767         ...                     freq='H', periods=3, tz='Europe/Berlin')
 768 
 769         >>> dti
 770         DatetimeIndex(['2014-08-01 09:00:00+02:00',
 771                        '2014-08-01 10:00:00+02:00',
 772                        '2014-08-01 11:00:00+02:00'],
 773                       dtype='datetime64[ns, Europe/Berlin]', freq='H')
 774 
 775         >>> dti.tz_convert('US/Central')
 776         DatetimeIndex(['2014-08-01 02:00:00-05:00',
 777                        '2014-08-01 03:00:00-05:00',
 778                        '2014-08-01 04:00:00-05:00'],
 779                       dtype='datetime64[ns, US/Central]', freq='H')
 780 
 781         With the ``tz=None``, we can remove the timezone (after converting
 782         to UTC if necessary):
 783 
 784         >>> dti = pd.date_range(start='2014-08-01 09:00', freq='H',
 785         ...                     periods=3, tz='Europe/Berlin')
 786 
 787         >>> dti
 788         DatetimeIndex(['2014-08-01 09:00:00+02:00',
 789                        '2014-08-01 10:00:00+02:00',
 790                        '2014-08-01 11:00:00+02:00'],
 791                         dtype='datetime64[ns, Europe/Berlin]', freq='H')
 792 
 793         >>> dti.tz_convert(None)
 794         DatetimeIndex(['2014-08-01 07:00:00',
 795                        '2014-08-01 08:00:00',
 796                        '2014-08-01 09:00:00'],
 797                         dtype='datetime64[ns]', freq='H')
 798         """
 799         tz = timezones.maybe_get_tz(tz)
 800 
 801         if self.tz is None:
 802             # tz naive, use tz_localize
 803             raise TypeError(
 804                 "Cannot convert tz-naive timestamps, use tz_localize to localize"
 805             )
 806 
 807         # No conversion since timestamps are all UTC to begin with
 808         dtype = tz_to_dtype(tz)
 809         return self._simple_new(self.asi8, dtype=dtype, freq=self.freq)

 811     def tz_localize(self, tz, ambiguous="raise", nonexistent="raise"):
 812         """
 813         Localize tz-naive Datetime Array/Index to tz-aware
 814         Datetime Array/Index.
 815 
 816         This method takes a time zone (tz) naive Datetime Array/Index object
 817         and makes this time zone aware. It does not move the time to another
 818         time zone.
 819         Time zone localization helps to switch from time zone aware to time
 820         zone unaware objects.
 821 
 822         Parameters
 823         ----------
 824         tz : str, pytz.timezone, dateutil.tz.tzfile or None
 825             Time zone to convert timestamps to. Passing ``None`` will
 826             remove the time zone information preserving local time.
 827         ambiguous : 'infer', 'NaT', bool array, default 'raise'
 828             When clocks moved backward due to DST, ambiguous times may arise.
 829             For example in Central European Time (UTC+01), when going from
 830             03:00 DST to 02:00 non-DST, 02:30:00 local time occurs both at
 831             00:30:00 UTC and at 01:30:00 UTC. In such a situation, the
 832             `ambiguous` parameter dictates how ambiguous times should be
 833             handled.
 834 
 835             - 'infer' will attempt to infer fall dst-transition hours based on
 836               order
 837             - bool-ndarray where True signifies a DST time, False signifies a
 838               non-DST time (note that this flag is only applicable for
 839               ambiguous times)
 840             - 'NaT' will return NaT where there are ambiguous times
 841             - 'raise' will raise an AmbiguousTimeError if there are ambiguous
 842               times.
 843 
 844         nonexistent : 'shift_forward', 'shift_backward, 'NaT', timedelta, \
 845 default 'raise'
 846             A nonexistent time does not exist in a particular timezone
 847             where clocks moved forward due to DST.
 848 
 849             - 'shift_forward' will shift the nonexistent time forward to the
 850               closest existing time
 851             - 'shift_backward' will shift the nonexistent time backward to the
 852               closest existing time
 853             - 'NaT' will return NaT where there are nonexistent times
 854             - timedelta objects will shift nonexistent times by the timedelta
 855             - 'raise' will raise an NonExistentTimeError if there are
 856               nonexistent times.
 857 
 858             .. versionadded:: 0.24.0
 859 
 860         Returns
 861         -------
 862         Same type as self
 863             Array/Index converted to the specified time zone.
 864 
 865         Raises
 866         ------
 867         TypeError
 868             If the Datetime Array/Index is tz-aware and tz is not None.
 869 
 870         See Also
 871         --------
 872         DatetimeIndex.tz_convert : Convert tz-aware DatetimeIndex from
 873             one time zone to another.
 874 
 875         Examples
 876         --------
 877         >>> tz_naive = pd.date_range('2018-03-01 09:00', periods=3)
 878         >>> tz_naive
 879         DatetimeIndex(['2018-03-01 09:00:00', '2018-03-02 09:00:00',
 880                        '2018-03-03 09:00:00'],
 881                       dtype='datetime64[ns]', freq='D')
 882 
 883         Localize DatetimeIndex in US/Eastern time zone:
 884 
 885         >>> tz_aware = tz_naive.tz_localize(tz='US/Eastern')
 886         >>> tz_aware
 887         DatetimeIndex(['2018-03-01 09:00:00-05:00',
 888                        '2018-03-02 09:00:00-05:00',
 889                        '2018-03-03 09:00:00-05:00'],
 890                       dtype='datetime64[ns, US/Eastern]', freq=None)
 891 
 892         With the ``tz=None``, we can remove the time zone information
 893         while keeping the local time (not converted to UTC):
 894 
 895         >>> tz_aware.tz_localize(None)
 896         DatetimeIndex(['2018-03-01 09:00:00', '2018-03-02 09:00:00',
 897                        '2018-03-03 09:00:00'],
 898                       dtype='datetime64[ns]', freq=None)
 899 
 900         Be careful with DST changes. When there is sequential data, pandas can
 901         infer the DST time:
 902 
 903         >>> s = pd.to_datetime(pd.Series(['2018-10-28 01:30:00',
 904         ...                               '2018-10-28 02:00:00',
 905         ...                               '2018-10-28 02:30:00',
 906         ...                               '2018-10-28 02:00:00',
 907         ...                               '2018-10-28 02:30:00',
 908         ...                               '2018-10-28 03:00:00',
 909         ...                               '2018-10-28 03:30:00']))
 910         >>> s.dt.tz_localize('CET', ambiguous='infer')
 911         0   2018-10-28 01:30:00+02:00
 912         1   2018-10-28 02:00:00+02:00
 913         2   2018-10-28 02:30:00+02:00
 914         3   2018-10-28 02:00:00+01:00
 915         4   2018-10-28 02:30:00+01:00
 916         5   2018-10-28 03:00:00+01:00
 917         6   2018-10-28 03:30:00+01:00
 918         dtype: datetime64[ns, CET]
 919 
 920         In some cases, inferring the DST is impossible. In such cases, you can
 921         pass an ndarray to the ambiguous parameter to set the DST explicitly
 922 
 923         >>> s = pd.to_datetime(pd.Series(['2018-10-28 01:20:00',
 924         ...                               '2018-10-28 02:36:00',
 925         ...                               '2018-10-28 03:46:00']))
 926         >>> s.dt.tz_localize('CET', ambiguous=np.array([True, True, False]))
 927         0   2018-10-28 01:20:00+02:00
 928         1   2018-10-28 02:36:00+02:00
 929         2   2018-10-28 03:46:00+01:00
 930         dtype: datetime64[ns, CET]
 931 
 932         If the DST transition causes nonexistent times, you can shift these
 933         dates forward or backwards with a timedelta object or `'shift_forward'`
 934         or `'shift_backwards'`.
 935 
 936         >>> s = pd.to_datetime(pd.Series(['2015-03-29 02:30:00',
 937         ...                               '2015-03-29 03:30:00']))
 938         >>> s.dt.tz_localize('Europe/Warsaw', nonexistent='shift_forward')
 939         0   2015-03-29 03:00:00+02:00
 940         1   2015-03-29 03:30:00+02:00
 941         dtype: datetime64[ns, Europe/Warsaw]
 942 
 943         >>> s.dt.tz_localize('Europe/Warsaw', nonexistent='shift_backward')
 944         0   2015-03-29 01:59:59.999999999+01:00
 945         1   2015-03-29 03:30:00+02:00
 946         dtype: datetime64[ns, Europe/Warsaw]
 947 
 948         >>> s.dt.tz_localize('Europe/Warsaw', nonexistent=pd.Timedelta('1H'))
 949         0   2015-03-29 03:30:00+02:00
 950         1   2015-03-29 03:30:00+02:00
 951         dtype: datetime64[ns, Europe/Warsaw]
 952         """
 953         nonexistent_options = ("raise", "NaT", "shift_forward", "shift_backward")
 954         if nonexistent not in nonexistent_options and not isinstance(
 955             nonexistent, timedelta
 956         ):
 957             raise ValueError(
 958                 "The nonexistent argument must be one of 'raise', "
 959                 "'NaT', 'shift_forward', 'shift_backward' or "
 960                 "a timedelta object"
 961             )
 962 
 963         if self.tz is not None:
 964             if tz is None:
 965                 new_dates = tzconversion.tz_convert(self.asi8, timezones.UTC, self.tz)
 966             else:
 967                 raise TypeError("Already tz-aware, use tz_convert to convert.")
 968         else:
 969             tz = timezones.maybe_get_tz(tz)
 970             # Convert to UTC
 971 
 972             new_dates = tzconversion.tz_localize_to_utc(
 973                 self.asi8, tz, ambiguous=ambiguous, nonexistent=nonexistent
 974             )
 975         new_dates = new_dates.view(DT64NS_DTYPE)
 976         dtype = tz_to_dtype(tz)
 977 
 978         freq = None
 979         if timezones.is_utc(tz) or (len(self) == 1 and not isna(new_dates[0])):
 980             # we can preserve freq
 981             # TODO: Also for fixed-offsets
 982             freq = self.freq
 983         elif tz is None and self.tz is None:
 984             # no-op
 985             freq = self.freq
 986         return self._simple_new(new_dates, dtype=dtype, freq=freq)

1830         """
1831         Convert Datetime Array to float64 ndarray of Julian Dates.
1832         0 Julian date is noon January 1, 4713 BC.
1833         https://en.wikipedia.org/wiki/Julian_day
1834         """

1837         year = np.asarray(self.year)
1838         month = np.asarray(self.month)
1839         day = np.asarray(self.day)
1840         testarr = month < 3
1841         year[testarr] -= 1
1842         month[testarr] += 12

1844             day
1845             + np.fix((153 * month - 457) / 5)
1846             + 365 * year
1847             + np.floor(year / 4)
1848             - np.floor(year / 100)
1849             + np.floor(year / 400)
1850             + 1_721_118.5
1851             + (
1852                 self.hour
1853                 + self.minute / 60.0
1854                 + self.second / 3600.0
1855                 + self.microsecond / 3600.0 / 1e6
1856                 + self.nanosecond / 3600.0 / 1e9

1866 def sequence_to_dt64ns(
1867     data,
1868     dtype=None,
1869     copy=False,
1870     tz=None,
1871     dayfirst=False,
1872     yearfirst=False,
1873     ambiguous="raise",
1874 ):
1875     """
1876     Parameters
1877     ----------
1878     data : list-like
1879     dtype : dtype, str, or None, default None
1880     copy : bool, default False
1881     tz : tzinfo, str, or None, default None
1882     dayfirst : bool, default False
1883     yearfirst : bool, default False
1884     ambiguous : str, bool, or arraylike, default 'raise'
1885         See pandas._libs.tslibs.tzconversion.tz_localize_to_utc.
1886 
1887     Returns
1888     -------
1889     result : numpy.ndarray
1890         The sequence converted to a numpy array with dtype ``datetime64[ns]``.
1891     tz : tzinfo or None
1892         Either the user-provided tzinfo or one inferred from the data.
1893     inferred_freq : Tick or None
1894         The inferred frequency of the sequence.
1895 
1896     Raises
1897     ------
1898     TypeError : PeriodDType data is passed
1899     """
1900 
1901     inferred_freq = None
1902 
1903     dtype = _validate_dt64_dtype(dtype)
1904 
1905     if not hasattr(data, "dtype"):
1906         # e.g. list, tuple
1907         if np.ndim(data) == 0:
1908             # i.e. generator
1909             data = list(data)
1910         data = np.asarray(data)
1911         copy = False
1912     elif isinstance(data, ABCSeries):
1913         data = data._values
1914     if isinstance(data, ABCPandasArray):
1915         data = data.to_numpy()
1916 
1917     if hasattr(data, "freq"):
1918         # i.e. DatetimeArray/Index
1919         inferred_freq = data.freq
1920 
1921     # if dtype has an embedded tz, capture it
1922     tz = validate_tz_from_dtype(dtype, tz)
1923 
1924     if isinstance(data, ABCIndexClass):
1925         if data.nlevels > 1:
1926             # Without this check, data._data below is None
1927             raise TypeError("Cannot create a DatetimeArray from a MultiIndex.")
1928         data = data._data
1929 
1930     # By this point we are assured to have either a numpy array or Index
1931     data, copy = maybe_convert_dtype(data, copy)
1932     data_dtype = getattr(data, "dtype", None)
1933 
1934     if is_object_dtype(data_dtype) or is_string_dtype(data_dtype):
1935         # TODO: We do not have tests specific to string-dtypes,
1936         #  also complex or categorical or other extension
1937         copy = False
1938         if lib.infer_dtype(data, skipna=False) == "integer":
1939             data = data.astype(np.int64)
1940         else:
1941             # data comes back here as either i8 to denote UTC timestamps
1942             #  or M8[ns] to denote wall times
1943             data, inferred_tz = objects_to_datetime64ns(
1944                 data, dayfirst=dayfirst, yearfirst=yearfirst
1945             )
1946             tz = maybe_infer_tz(tz, inferred_tz)
1947         data_dtype = data.dtype
1948 
1949     # `data` may have originally been a Categorical[datetime64[ns, tz]],
1950     # so we need to handle these types.
1951     if is_datetime64tz_dtype(data_dtype):
1952         # DatetimeArray -> ndarray
1953         tz = maybe_infer_tz(tz, data.tz)
1954         result = data._data
1955 
1956     elif is_datetime64_dtype(data_dtype):
1957         # tz-naive DatetimeArray or ndarray[datetime64]
1958         data = getattr(data, "_data", data)
1959         if data.dtype != DT64NS_DTYPE:
1960             data = conversion.ensure_datetime64ns(data)
1961 
1962         if tz is not None:
1963             # Convert tz-naive to UTC
1964             tz = timezones.maybe_get_tz(tz)
1965             data = tzconversion.tz_localize_to_utc(
1966                 data.view("i8"), tz, ambiguous=ambiguous
1967             )
1968             data = data.view(DT64NS_DTYPE)
