  63 def _cat_compare_op(op):
  64     opname = f"__{op.__name__}__"
  65 
  66     @unpack_zerodim_and_defer(opname)
  67     def func(self, other):
  68         if is_list_like(other) and len(other) != len(self):
  69             # TODO: Could this fail if the categories are listlike objects?
  70             raise ValueError("Lengths must match.")
  71 
  72         if not self.ordered:
  73             if opname in ["__lt__", "__gt__", "__le__", "__ge__"]:
  74                 raise TypeError(
  75                     "Unordered Categoricals can only compare equality or not"
  76                 )
  77         if isinstance(other, Categorical):
  78             # Two Categoricals can only be be compared if the categories are
  79             # the same (maybe up to ordering, depending on ordered)
  80 
  81             msg = "Categoricals can only be compared if 'categories' are the same."
  82             if len(self.categories) != len(other.categories):
  83                 raise TypeError(msg + " Categories are different lengths")
  84             elif self.ordered and not (self.categories == other.categories).all():
  85                 raise TypeError(msg)
  86             elif not set(self.categories) == set(other.categories):
  87                 raise TypeError(msg)
  88 
  89             if not (self.ordered == other.ordered):
  90                 raise TypeError(
  91                     "Categoricals can only be compared if 'ordered' is the same"
  92                 )
  93             if not self.ordered and not self.categories.equals(other.categories):
  94                 # both unordered and different order
  95                 other_codes = _get_codes_for_values(other, self.categories)
  96             else:
  97                 other_codes = other._codes
  98 
  99             f = getattr(self._codes, opname)
 100             ret = f(other_codes)
 101             mask = (self._codes == -1) | (other_codes == -1)
 102             if mask.any():
 103                 # In other series, the leads to False, so do that here too
 104                 if opname == "__ne__":
 105                     ret[(self._codes == -1) & (other_codes == -1)] = True
 106                 else:
 107                     ret[mask] = False
 108             return ret
 109 
 110         if is_scalar(other):
 111             if other in self.categories:
 112                 i = self.categories.get_loc(other)
 113                 ret = getattr(self._codes, opname)(i)
 114 
 115                 if opname not in {"__eq__", "__ge__", "__gt__"}:
 116                     # check for NaN needed if we are not equal or larger
 117                     mask = self._codes == -1
 118                     ret[mask] = False
 119                 return ret
 120             else:
 121                 if opname == "__eq__":
 122                     return np.zeros(len(self), dtype=bool)
 123                 elif opname == "__ne__":
 124                     return np.ones(len(self), dtype=bool)
 125                 else:
 126                     raise TypeError(
 127                         f"Cannot compare a Categorical for op {opname} with a "
 128                         "scalar, which is not a category."
 129                     )
 130         else:
 131             # allow categorical vs object dtype array comparisons for equality
 132             # these are only positional comparisons
 133             if opname not in ["__eq__", "__ne__"]:
 134                 raise TypeError(
 135                     f"Cannot compare a Categorical for op {opname} with "
 136                     f"type {type(other)}.\nIf you want to compare values, "
 137                     "use 'np.asarray(cat) <op> other'."
 138                 )
 139 
 140             if isinstance(other, ExtensionArray) and needs_i8_conversion(other.dtype):
 141                 # We would return NotImplemented here, but that messes up
 142                 #  ExtensionIndex's wrapped methods
 143                 return op(other, self)
 144             return getattr(np.array(self), opname)(np.array(other))
 145 
 146     func.__name__ = opname
 147 
 148     return func

 151 def contains(cat, key, container):
 152     """
 153     Helper for membership check for ``key`` in ``cat``.
 154 
 155     This is a helper method for :method:`__contains__`
 156     and :class:`CategoricalIndex.__contains__`.
 157 
 158     Returns True if ``key`` is in ``cat.categories`` and the
 159     location of ``key`` in ``categories`` is in ``container``.
 160 
 161     Parameters
 162     ----------
 163     cat : :class:`Categorical`or :class:`categoricalIndex`
 164     key : a hashable object
 165         The key to check membership for.
 166     container : Container (e.g. list-like or mapping)
 167         The container to check for membership in.
 168 
 169     Returns
 170     -------
 171     is_in : bool
 172         True if ``key`` is in ``self.categories`` and location of
 173         ``key`` in ``categories`` is in ``container``, else False.
 174 
 175     Notes
 176     -----
 177     This method does not check for NaN values. Do that separately
 178     before calling this method.
 179     """
 180     hash(key)
 181 
 182     # get location of key in categories.
 183     # If a KeyError, the key isn't in categories, so logically
 184     #  can't be in container either.
 185     try:
 186         loc = cat.categories.get_loc(key)
 187     except (KeyError, TypeError):
 188         return False
 189 
 190     # loc is the location of key in categories, but also the *value*
 191     # for key in container. So, `key` may be in categories,
 192     # but still not in `container`. Example ('b' in categories,
 193     # but not in values):
 194     # 'b' in Categorical(['a'], categories=['a', 'b'])  # False
 195     if is_scalar(loc):
 196         return loc in container
 197     else:
 198         # if categories is an IntervalIndex, loc is an array.
 199         return any(loc_ in container for loc_ in loc)

 202 class Categorical(NDArrayBackedExtensionArray, PandasObject):
 203     """
 204     Represent a categorical variable in classic R / S-plus fashion.
 205 
 206     `Categoricals` can only take on only a limited, and usually fixed, number
 207     of possible values (`categories`). In contrast to statistical categorical
 208     variables, a `Categorical` might have an order, but numerical operations
 209     (additions, divisions, ...) are not possible.
 210 
 211     All values of the `Categorical` are either in `categories` or `np.nan`.
 212     Assigning values outside of `categories` will raise a `ValueError`. Order
 213     is defined by the order of the `categories`, not lexical order of the
 214     values.
 215 
 216     Parameters
 217     ----------
 218     values : list-like
 219         The values of the categorical. If categories are given, values not in
 220         categories will be replaced with NaN.
 221     categories : Index-like (unique), optional
 222         The unique categories for this categorical. If not given, the
 223         categories are assumed to be the unique values of `values` (sorted, if
 224         possible, otherwise in the order in which they appear).
 225     ordered : bool, default False
 226         Whether or not this categorical is treated as a ordered categorical.
 227         If True, the resulting categorical will be ordered.
 228         An ordered categorical respects, when sorted, the order of its
 229         `categories` attribute (which in turn is the `categories` argument, if
 230         provided).
 231     dtype : CategoricalDtype
 232         An instance of ``CategoricalDtype`` to use for this categorical.
 233 
 234     Attributes
 235     ----------
 236     categories : Index
 237         The categories of this categorical
 238     codes : ndarray
 239         The codes (integer positions, which point to the categories) of this
 240         categorical, read only.
 241     ordered : bool
 242         Whether or not this Categorical is ordered.
 243     dtype : CategoricalDtype
 244         The instance of ``CategoricalDtype`` storing the ``categories``
 245         and ``ordered``.
 246 
 247     Methods
 248     -------
 249     from_codes
 250     __array__
 251 
 252     Raises
 253     ------
 254     ValueError
 255         If the categories do not validate.
 256     TypeError
 257         If an explicit ``ordered=True`` is given but no `categories` and the
 258         `values` are not sortable.
 259 
 260     See Also
 261     --------
 262     CategoricalDtype : Type for categorical data.
 263     CategoricalIndex : An Index with an underlying ``Categorical``.
 264 
 265     Notes
 266     -----
 267     See the `user guide
 268     <https://pandas.pydata.org/pandas-docs/stable/user_guide/categorical.html>`_
 269     for more.
 270 
 271     Examples
 272     --------
 273     >>> pd.Categorical([1, 2, 3, 1, 2, 3])
 274     [1, 2, 3, 1, 2, 3]
 275     Categories (3, int64): [1, 2, 3]
 276 
 277     >>> pd.Categorical(['a', 'b', 'c', 'a', 'b', 'c'])
 278     [a, b, c, a, b, c]
 279     Categories (3, object): [a, b, c]
 280 
 281     Ordered `Categoricals` can be sorted according to the custom order
 282     of the categories and can have a min and max value.
 283 
 284     >>> c = pd.Categorical(['a', 'b', 'c', 'a', 'b', 'c'], ordered=True,
 285     ...                    categories=['c', 'b', 'a'])
 286     >>> c
 287     [a, b, c, a, b, c]
 288     Categories (3, object): [c < b < a]
 289     >>> c.min()
 290     'c'
 291     """
 292 
 293     # For comparisons, so that numpy uses our implementation if the compare
 294     # ops, which raise
 295     __array_priority__ = 1000
 296     _dtype = CategoricalDtype(ordered=False)
 297     # tolist is not actually deprecated, just suppressed in the __dir__
 298     _deprecations = PandasObject._deprecations | frozenset(["tolist"])
 299     _typ = "categorical"

 301     def __init__(
 302         self, values, categories=None, ordered=None, dtype=None, fastpath=False
 303     ):
 304 
 305         dtype = CategoricalDtype._from_values_or_dtype(
 306             values, categories, ordered, dtype
 307         )
 308         # At this point, dtype is always a CategoricalDtype, but
 309         # we may have dtype.categories be None, and we need to
 310         # infer categories in a factorization step further below
 311 
 312         if fastpath:
 313             self._codes = coerce_indexer_dtype(values, dtype.categories)
 314             self._dtype = self._dtype.update_dtype(dtype)
 315             return
 316 
 317         # null_mask indicates missing values we want to exclude from inference.
 318         # This means: only missing values in list-likes (not arrays/ndframes).
 319         null_mask = np.array(False)
 320 
 321         # sanitize input
 322         if is_categorical_dtype(values):
 323             if dtype.categories is None:
 324                 dtype = CategoricalDtype(values.categories, dtype.ordered)
 325         elif not isinstance(values, (ABCIndexClass, ABCSeries)):
 326             # sanitize_array coerces np.nan to a string under certain versions
 327             # of numpy
 328             values = maybe_infer_to_datetimelike(values, convert_dates=True)
 329             if not isinstance(values, np.ndarray):
 330                 values = com.convert_to_list_like(values)
 331 
 332                 # By convention, empty lists result in object dtype:
 333                 sanitize_dtype = np.dtype("O") if len(values) == 0 else None
 334                 null_mask = isna(values)
 335                 if null_mask.any():
 336                     values = [values[idx] for idx in np.where(~null_mask)[0]]
 337                 values = sanitize_array(values, None, dtype=sanitize_dtype)
 338 
 339         if dtype.categories is None:
 340             try:
 341                 codes, categories = factorize(values, sort=True)
 342             except TypeError as err:
 343                 codes, categories = factorize(values, sort=False)
 344                 if dtype.ordered:
 345                     # raise, as we don't have a sortable data structure and so
 346                     # the user should give us one by specifying categories
 347                     raise TypeError(
 348                         "'values' is not ordered, please "
 349                         "explicitly specify the categories order "
 350                         "by passing in a categories argument."
 351                     ) from err
 352             except ValueError as err:
 353 
 354                 # TODO(EA2D)
 355                 raise NotImplementedError(
 356                     "> 1 ndim Categorical are not supported at this time"
 357                 ) from err
 358 
 359             # we're inferring from values
 360             dtype = CategoricalDtype(categories, dtype.ordered)
 361 
 362         elif is_categorical_dtype(values.dtype):
 363             old_codes = (
 364                 values._values.codes if isinstance(values, ABCSeries) else values.codes
 365             )
 366             codes = recode_for_categories(
 367                 old_codes, values.dtype.categories, dtype.categories
 368             )
 369 
 370         else:
 371             codes = _get_codes_for_values(values, dtype.categories)
 372 
 373         if null_mask.any():
 374             # Reinsert -1 placeholders for previously removed missing values
 375             full_codes = -np.ones(null_mask.shape, dtype=codes.dtype)
 376             full_codes[~null_mask] = codes
 377             codes = full_codes
 378 
 379         self._dtype = self._dtype.update_dtype(dtype)
 380         self._codes = coerce_indexer_dtype(codes, dtype.categories)

 497     @classmethod
 498     def _from_inferred_categories(
 499         cls, inferred_categories, inferred_codes, dtype, true_values=None
 500     ):
 501         """
 502         Construct a Categorical from inferred values.
 503 
 504         For inferred categories (`dtype` is None) the categories are sorted.
 505         For explicit `dtype`, the `inferred_categories` are cast to the
 506         appropriate type.
 507 
 508         Parameters
 509         ----------
 510         inferred_categories : Index
 511         inferred_codes : Index
 512         dtype : CategoricalDtype or 'category'
 513         true_values : list, optional
 514             If none are provided, the default ones are
 515             "True", "TRUE", and "true."
 516 
 517         Returns
 518         -------
 519         Categorical
 520         """
 521         from pandas import Index, to_numeric, to_datetime, to_timedelta
 522 
 523         cats = Index(inferred_categories)
 524         known_categories = (
 525             isinstance(dtype, CategoricalDtype) and dtype.categories is not None
 526         )
 527 
 528         if known_categories:
 529             # Convert to a specialized type with `dtype` if specified.
 530             if dtype.categories.is_numeric():
 531                 cats = to_numeric(inferred_categories, errors="coerce")
 532             elif is_datetime64_dtype(dtype.categories):
 533                 cats = to_datetime(inferred_categories, errors="coerce")
 534             elif is_timedelta64_dtype(dtype.categories):
 535                 cats = to_timedelta(inferred_categories, errors="coerce")
 536             elif dtype.categories.is_boolean():
 537                 if true_values is None:
 538                     true_values = ["True", "TRUE", "true"]
 539 
 540                 cats = cats.isin(true_values)
 541 
 542         if known_categories:
 543             # Recode from observation order to dtype.categories order.
 544             categories = dtype.categories
 545             codes = recode_for_categories(inferred_codes, cats, categories)
 546         elif not cats.is_monotonic_increasing:
 547             # Sort categories and recode for unknown categories.
 548             unsorted = cats.copy()
 549             categories = cats.sort_values()
 550 
 551             codes = recode_for_categories(inferred_codes, unsorted, categories)
 552             dtype = CategoricalDtype(categories, ordered=False)
 553         else:
 554             dtype = CategoricalDtype(cats, ordered=False)
 555             codes = inferred_codes
 556 
 557         return cls(codes, dtype=dtype, fastpath=True)

 559     @classmethod
 560     def from_codes(cls, codes, categories=None, ordered=None, dtype=None):
 561         """
 562         Make a Categorical type from codes and categories or dtype.
 563 
 564         This constructor is useful if you already have codes and
 565         categories/dtype and so do not need the (computation intensive)
 566         factorization step, which is usually done on the constructor.
 567 
 568         If your data does not follow this convention, please use the normal
 569         constructor.
 570 
 571         Parameters
 572         ----------
 573         codes : array-like of int
 574             An integer array, where each integer points to a category in
 575             categories or dtype.categories, or else is -1 for NaN.
 576         categories : index-like, optional
 577             The categories for the categorical. Items need to be unique.
 578             If the categories are not given here, then they must be provided
 579             in `dtype`.
 580         ordered : bool, optional
 581             Whether or not this categorical is treated as an ordered
 582             categorical. If not given here or in `dtype`, the resulting
 583             categorical will be unordered.
 584         dtype : CategoricalDtype or "category", optional
 585             If :class:`CategoricalDtype`, cannot be used together with
 586             `categories` or `ordered`.
 587 
 588             .. versionadded:: 0.24.0
 589 
 590                When `dtype` is provided, neither `categories` nor `ordered`
 591                should be provided.
 592 
 593         Returns
 594         -------
 595         Categorical
 596 
 597         Examples
 598         --------
 599         >>> dtype = pd.CategoricalDtype(['a', 'b'], ordered=True)
 600         >>> pd.Categorical.from_codes(codes=[0, 1, 0, 1], dtype=dtype)
 601         [a, b, a, b]
 602         Categories (2, object): [a < b]
 603         """
 604         dtype = CategoricalDtype._from_values_or_dtype(
 605             categories=categories, ordered=ordered, dtype=dtype
 606         )
 607         if dtype.categories is None:
 608             msg = (
 609                 "The categories must be provided in 'categories' or "
 610                 "'dtype'. Both were None."
 611             )
 612             raise ValueError(msg)
 613 
 614         if is_extension_array_dtype(codes) and is_integer_dtype(codes):
 615             # Avoid the implicit conversion of Int to object
 616             if isna(codes).any():
 617                 raise ValueError("codes cannot contain NA values")
 618             codes = codes.to_numpy(dtype=np.int64)
 619         else:
 620             codes = np.asarray(codes)

 757     def set_categories(self, new_categories, ordered=None, rename=False, inplace=False):
 758         """
 759         Set the categories to the specified new_categories.
 760 
 761         `new_categories` can include new categories (which will result in
 762         unused categories) or remove old categories (which results in values
 763         set to NaN). If `rename==True`, the categories will simple be renamed
 764         (less or more items than in old categories will result in values set to
 765         NaN or in unused categories respectively).
 766 
 767         This method can be used to perform more than one action of adding,
 768         removing, and reordering simultaneously and is therefore faster than
 769         performing the individual steps via the more specialised methods.
 770 
 771         On the other hand this methods does not do checks (e.g., whether the
 772         old categories are included in the new categories on a reorder), which
 773         can result in surprising changes, for example when using special string
 774         dtypes, which does not considers a S1 string equal to a single char
 775         python string.
 776 
 777         Parameters
 778         ----------
 779         new_categories : Index-like
 780            The categories in new order.
 781         ordered : bool, default False
 782            Whether or not the categorical is treated as a ordered categorical.
 783            If not given, do not change the ordered information.
 784         rename : bool, default False
 785            Whether or not the new_categories should be considered as a rename
 786            of the old categories or as reordered categories.
 787         inplace : bool, default False
 788            Whether or not to reorder the categories in-place or return a copy
 789            of this categorical with reordered categories.
 790 
 791         Returns
 792         -------
 793         Categorical with reordered categories or None if inplace.
 794 
 795         Raises
 796         ------
 797         ValueError
 798             If new_categories does not validate as categories
 799 
 800         See Also
 801         --------
 802         rename_categories : Rename categories.
 803         reorder_categories : Reorder categories.
 804         add_categories : Add new categories.
 805         remove_categories : Remove the specified categories.
 806         remove_unused_categories : Remove categories which are not used.
 807         """
 808         inplace = validate_bool_kwarg(inplace, "inplace")
 809         if ordered is None:
 810             ordered = self.dtype.ordered
 811         new_dtype = CategoricalDtype(new_categories, ordered=ordered)
 812 
 813         cat = self if inplace else self.copy()
 814         if rename:
 815             if cat.dtype.categories is not None and len(new_dtype.categories) < len(
 816                 cat.dtype.categories
 817             ):
 818                 # remove all _codes which are larger and set to -1/NaN
 819                 cat._codes[cat._codes >= len(new_dtype.categories)] = -1
 820         else:
 821             codes = recode_for_categories(
 822                 cat.codes, cat.categories, new_dtype.categories
 823             )
 824             cat._codes = codes
 825         cat._dtype = new_dtype
 826 
 827         if not inplace:
 828             return cat

 830     def rename_categories(self, new_categories, inplace=False):
 831         """
 832         Rename categories.
 833 
 834         Parameters
 835         ----------
 836         new_categories : list-like, dict-like or callable
 837 
 838             New categories which will replace old categories.
 839 
 840             * list-like: all items must be unique and the number of items in
 841               the new categories must match the existing number of categories.
 842 
 843             * dict-like: specifies a mapping from
 844               old categories to new. Categories not contained in the mapping
 845               are passed through and extra categories in the mapping are
 846               ignored.
 847 
 848             * callable : a callable that is called on all items in the old
 849               categories and whose return values comprise the new categories.
 850 
 851             .. versionadded:: 0.23.0.
 852 
 853         inplace : bool, default False
 854             Whether or not to rename the categories inplace or return a copy of
 855             this categorical with renamed categories.
 856 
 857         Returns
 858         -------
 859         cat : Categorical or None
 860            With ``inplace=False``, the new categorical is returned.
 861            With ``inplace=True``, there is no return value.
 862 
 863         Raises
 864         ------
 865         ValueError
 866             If new categories are list-like and do not have the same number of
 867             items than the current categories or do not validate as categories
 868 
 869         See Also
 870         --------
 871         reorder_categories : Reorder categories.
 872         add_categories : Add new categories.
 873         remove_categories : Remove the specified categories.
 874         remove_unused_categories : Remove categories which are not used.
 875         set_categories : Set the categories to the specified ones.
 876 
 877         Examples
 878         --------
 879         >>> c = pd.Categorical(['a', 'a', 'b'])
 880         >>> c.rename_categories([0, 1])
 881         [0, 0, 1]
 882         Categories (2, int64): [0, 1]
 883 
 884         For dict-like ``new_categories``, extra keys are ignored and
 885         categories not in the dictionary are passed through
 886 
 887         >>> c.rename_categories({'a': 'A', 'c': 'C'})
 888         [A, A, b]
 889         Categories (2, object): [A, b]
 890 
 891         You may also provide a callable to create the new categories
 892 
 893         >>> c.rename_categories(lambda x: x.upper())
 894         [A, A, B]
 895         Categories (2, object): [A, B]
 896         """
 897         inplace = validate_bool_kwarg(inplace, "inplace")
 898         cat = self if inplace else self.copy()
 899 
 900         if is_dict_like(new_categories):
 901             cat.categories = [new_categories.get(item, item) for item in cat.categories]
 902         elif callable(new_categories):
 903             cat.categories = [new_categories(item) for item in cat.categories]
 904         else:
 905             cat.categories = new_categories
 906         if not inplace:
 907             return cat

 955 
 956         `new_categories` will be included at the last/highest place in the
 957         categories and will be unused directly after this call.
 958 
 959         Parameters
 960         ----------
 961         new_categories : category or list-like of category
 962            The new categories to be included.
 963         inplace : bool, default False
 964            Whether or not to add the categories inplace or return a copy of
 965            this categorical with added categories.
 966

 968         -------
 969         cat : Categorical with new categories added or None if inplace.

 972         ------

1005 
1006         `removals` must be included in the old categories. Values which were in
1007         the removed categories will be set to NaN
1008 
1009         Parameters
1010         ----------
1011         removals : category or list of categories
1012            The categories which should be removed.
1013         inplace : bool, default False
1014            Whether or not to remove the categories inplace or return a copy of
1015            this categorical with removed categories.
1016

1019         cat : Categorical with removed categories or None if inplace.

1023         ValueError

1027         --------
1028         rename_categories : Rename categories.

1030         add_categories : Add new categories.
1031         remove_unused_categories : Remove categories which are not used.

1037

1093     def map(self, mapper):
1094         """
1095         Map categories using input correspondence (dict, Series, or function).
1096 
1097         Maps the categories to new categories. If the mapping correspondence is
1098         one-to-one the result is a :class:`~pandas.Categorical` which has the
1099         same order property as the original, otherwise a :class:`~pandas.Index`
1100         is returned. NaN values are unaffected.
1101 
1102         If a `dict` or :class:`~pandas.Series` is used any unmapped category is
1103         mapped to `NaN`. Note that if this happens an :class:`~pandas.Index`
1104         will be returned.
1105 
1106         Parameters
1107         ----------
1108         mapper : function, dict, or Series
1109             Mapping correspondence.
1110 
1111         Returns
1112         -------
1113         pandas.Categorical or pandas.Index
1114             Mapped categorical.
1115 
1116         See Also
1117         --------
1118         CategoricalIndex.map : Apply a mapping correspondence on a
1119             :class:`~pandas.CategoricalIndex`.
1120         Index.map : Apply a mapping correspondence on an
1121             :class:`~pandas.Index`.
1122         Series.map : Apply a mapping correspondence on a
1123             :class:`~pandas.Series`.
1124         Series.apply : Apply more complex functions on a
1125             :class:`~pandas.Series`.
1126 
1127         Examples
1128         --------
1129         >>> cat = pd.Categorical(['a', 'b', 'c'])
1130         >>> cat
1131         [a, b, c]
1132         Categories (3, object): [a, b, c]
1133         >>> cat.map(lambda x: x.upper())
1134         [A, B, C]
1135         Categories (3, object): [A, B, C]
1136         >>> cat.map({'a': 'first', 'b': 'second', 'c': 'third'})
1137         [first, second, third]
1138         Categories (3, object): [first, second, third]
1139 
1140         If the mapping is one-to-one the ordering of the categories is
1141         preserved:
1142 
1143         >>> cat = pd.Categorical(['a', 'b', 'c'], ordered=True)
1144         >>> cat
1145         [a, b, c]
1146         Categories (3, object): [a < b < c]
1147         >>> cat.map({'a': 3, 'b': 2, 'c': 1})
1148         [3, 2, 1]
1149         Categories (3, int64): [3 < 2 < 1]
1150 
1151         If the mapping is not one-to-one an :class:`~pandas.Index` is returned:
1152 
1153         >>> cat.map({'a': 'first', 'b': 'second', 'c': 'first'})
1154         Index(['first', 'second', 'first'], dtype='object')
1155 
1156         If a `dict` is used, all unmapped categories are mapped to `NaN` and
1157         the result is an :class:`~pandas.Index`:
1158 
1159         >>> cat.map({'a': 'first', 'b': 'second'})
1160         Index(['first', 'second', nan], dtype='object')
1161         """
1162         new_categories = self.categories.map(mapper)
1163         try:
1164             return self.from_codes(
1165                 self._codes.copy(), categories=new_categories, ordered=self.ordered
1166             )
1167         except ValueError:
1168             # NA values are represented in self._codes with -1
1169             # np.take causes NA values to take final element in new_categories
1170             if np.any(self._codes == -1):
1171                 new_categories = new_categories.insert(len(new_categories), np.nan)
1172             return np.take(new_categories, self._codes)

1647     def fillna(self, value=None, method=None, limit=None):
1648         """
1649         Fill NA/NaN values using the specified method.
1650 
1651         Parameters
1652         ----------
1653         value : scalar, dict, Series
1654             If a scalar value is passed it is used to fill all missing values.
1655             Alternatively, a Series or dict can be used to fill in different
1656             values for each index. The value should not be a list. The
1657             value(s) passed should either be in the categories or should be
1658             NaN.
1659         method : {'backfill', 'bfill', 'pad', 'ffill', None}, default None
1660             Method to use for filling holes in reindexed Series
1661             pad / ffill: propagate last valid observation forward to next valid
1662             backfill / bfill: use NEXT valid observation to fill gap
1663         limit : int, default None
1664             (Not implemented yet for Categorical!)
1665             If method is specified, this is the maximum number of consecutive
1666             NaN values to forward/backward fill. In other words, if there is
1667             a gap with more than this number of consecutive NaNs, it will only
1668             be partially filled. If method is not specified, this is the
1669             maximum number of entries along the entire axis where NaNs will be
1670             filled.
1671 
1672         Returns
1673         -------
1674         filled : Categorical with NA/NaN filled
1675         """
1676         value, method = validate_fillna_kwargs(
1677             value, method, validate_scalar_dict_value=False
1678         )
1679 
1680         if value is None:
1681             value = np.nan
1682         if limit is not None:
1683             raise NotImplementedError(
1684                 "specifying a limit for fillna has not been implemented yet"
1685             )
1686 
1687         codes = self._codes
1688 
1689         # pad / bfill
1690         if method is not None:
1691 
1692             # TODO: dispatch when self.categories is EA-dtype
1693             values = np.asarray(self).reshape(-1, len(self))
1694             values = interpolate_2d(values, method, 0, None, value).astype(
1695                 self.categories.dtype
1696             )[0]
1697             codes = _get_codes_for_values(values, self.categories)
1698 
1699         else:
1700 
1701             # If value is a dict or a Series (a dict value has already
1702             # been converted to a Series)
1703             if isinstance(value, (np.ndarray, Categorical, ABCSeries)):
1704                 # We get ndarray or Categorical if called via Series.fillna,
1705                 #  where it will unwrap another aligned Series before getting here
1706 
1707                 mask = ~algorithms.isin(value, self.categories)
1708                 if not isna(value[mask]).all():
1709                     raise ValueError("fill value must be in categories")
1710 
1711                 values_codes = _get_codes_for_values(value, self.categories)
1712                 indexer = np.where(codes == -1)
1713                 codes = codes.copy()
1714                 codes[indexer] = values_codes[indexer]
1715 
1716             # If value is not a dict or Series it should be a scalar
1717             elif is_hashable(value):
1718                 if not isna(value) and value not in self.categories:
1719                     raise ValueError("fill value must be in categories")
1720 
1721                 mask = codes == -1
1722                 if mask.any():
1723                     codes = codes.copy()
1724                     if isna(value):
1725                         codes[mask] = -1
1726                     else:
1727                         codes[mask] = self.categories.get_loc(value)
1728 
1729             else:
1730                 raise TypeError(
1731                     f"'value' parameter must be a scalar, dict "
1732                     f"or Series, but you passed a {type(value).__name__}"
1733                 )
1734 
1735         return self._constructor(codes, dtype=self.dtype, fastpath=True)

1737     def take(self: _T, indexer, allow_fill: bool = False, fill_value=None) -> _T:
1738         """
1739         Take elements from the Categorical.
1740 
1741         Parameters
1742         ----------
1743         indexer : sequence of int
1744             The indices in `self` to take. The meaning of negative values in
1745             `indexer` depends on the value of `allow_fill`.
1746         allow_fill : bool, default False
1747             How to handle negative values in `indexer`.
1748 
1749             * False: negative values in `indices` indicate positional indices
1750               from the right. This is similar to
1751               :func:`numpy.take`.
1752 
1753             * True: negative values in `indices` indicate missing values
1754               (the default). These values are set to `fill_value`. Any other
1755               other negative values raise a ``ValueError``.
1756 
1757             .. versionchanged:: 1.0.0
1758 
1759                Default value changed from ``True`` to ``False``.
1760 
1761         fill_value : object
1762             The value to use for `indices` that are missing (-1), when
1763             ``allow_fill=True``. This should be the category, i.e. a value
1764             in ``self.categories``, not a code.
1765 
1766         Returns
1767         -------
1768         Categorical
1769             This Categorical will have the same categories and ordered as
1770             `self`.
1771 
1772         See Also
1773         --------
1774         Series.take : Similar method for Series.
1775         numpy.ndarray.take : Similar method for NumPy arrays.
1776 
1777         Examples
1778         --------
1779         >>> cat = pd.Categorical(['a', 'a', 'b'])
1780         >>> cat
1781         [a, a, b]
1782         Categories (2, object): [a, b]
1783 
1784         Specify ``allow_fill==False`` to have negative indices mean indexing
1785         from the right.
1786 
1787         >>> cat.take([0, -1, -2], allow_fill=False)
1788         [a, b, a]
1789         Categories (2, object): [a, b]
1790 
1791         With ``allow_fill=True``, indices equal to ``-1`` mean "missing"

1800 
1801         >>> cat.take([0, -1, -1], allow_fill=True, fill_value='a')
