 119 def coerce_to_array(
 120     values, mask=None, copy: bool = False
 121 ) -> Tuple[np.ndarray, np.ndarray]:
 122     """
 123     Coerce the input values array to numpy arrays with a mask.
 124 
 125     Parameters
 126     ----------
 127     values : 1D list-like
 128     mask : bool 1D array, optional
 129     copy : bool, default False
 130         if True, copy the input
 131 
 132     Returns
 133     -------
 134     tuple of (values, mask)
 135     """
 136     if isinstance(values, BooleanArray):
 137         if mask is not None:
 138             raise ValueError("cannot pass mask for BooleanArray input")
 139         values, mask = values._data, values._mask
 140         if copy:
 141             values = values.copy()
 142             mask = mask.copy()
 143         return values, mask
 144 
 145     mask_values = None
 146     if isinstance(values, np.ndarray) and values.dtype == np.bool_:
 147         if copy:
 148             values = values.copy()
 149     elif isinstance(values, np.ndarray) and is_numeric_dtype(values.dtype):
 150         mask_values = isna(values)
 151 
 152         values_bool = np.zeros(len(values), dtype=bool)
 153         values_bool[~mask_values] = values[~mask_values].astype(bool)
 154 
 155         if not np.all(
 156             values_bool[~mask_values].astype(values.dtype) == values[~mask_values]
 157         ):
 158             raise TypeError("Need to pass bool-like values")
 159 
 160         values = values_bool
 161     else:
 162         values_object = np.asarray(values, dtype=object)
 163 
 164         inferred_dtype = lib.infer_dtype(values_object, skipna=True)
 165         integer_like = ("floating", "integer", "mixed-integer-float")
 166         if inferred_dtype not in ("boolean", "empty") + integer_like:
 167             raise TypeError("Need to pass bool-like values")
 168 
 169         mask_values = isna(values_object)
 170         values = np.zeros(len(values), dtype=bool)
 171         values[~mask_values] = values_object[~mask_values].astype(bool)
 172 
 173         # if the values were integer-like, validate it were actually 0/1's
 174         if inferred_dtype in integer_like:
 175             if not np.all(
 176                 values[~mask_values].astype(float)
 177                 == values_object[~mask_values].astype(float)
 178             ):
 179                 raise TypeError("Need to pass bool-like values")
 180 
 181     if mask is None and mask_values is None:
 182         mask = np.zeros(len(values), dtype=bool)
 183     elif mask is None:
 184         mask = mask_values
 185     else:
 186         if isinstance(mask, np.ndarray) and mask.dtype == np.bool_:
 187             if mask_values is not None:
 188                 mask = mask | mask_values
 189             else:
 190                 if copy:
 191                     mask = mask.copy()
 192         else:
 193             mask = np.array(mask, dtype=bool)
 194             if mask_values is not None:
 195                 mask = mask | mask_values
 196 
 197     if not values.ndim == 1:
 198         raise ValueError("values must be a 1D list-like")
 199     if not mask.ndim == 1:
 200         raise ValueError("mask must be a 1D list-like")
 201 
 202     return values, mask

 205 class BooleanArray(BaseMaskedArray):
 206     """
 207     Array of boolean (True/False) data with missing values.
 208 
 209     This is a pandas Extension array for boolean data, under the hood
 210     represented by 2 numpy arrays: a boolean array with the data and
 211     a boolean array with the mask (True indicating missing).
 212 
 213     BooleanArray implements Kleene logic (sometimes called three-value
 214     logic) for logical operations. See :ref:`boolean.kleene` for more.
 215 
 216     To construct an BooleanArray from generic array-like input, use
 217     :func:`pandas.array` specifying ``dtype="boolean"`` (see examples
 218     below).
 219 
 220     .. versionadded:: 1.0.0
 221 
 222     .. warning::
 223 
 224        BooleanArray is considered experimental. The implementation and
 225        parts of the API may change without warning.
 226 
 227     Parameters
 228     ----------
 229     values : numpy.ndarray
 230         A 1-d boolean-dtype array with the data.
 231     mask : numpy.ndarray
 232         A 1-d boolean-dtype array indicating missing values (True
 233         indicates missing).
 234     copy : bool, default False
 235         Whether to copy the `values` and `mask` arrays.
 236 
 237     Attributes
 238     ----------
 239     None
 240 
 241     Methods
 242     -------
 243     None
 244 
 245     Returns
 246     -------
 247     BooleanArray
 248 
 249     Examples
 250     --------
 251     Create an BooleanArray with :func:`pandas.array`:
 252 
 253     >>> pd.array([True, False, None], dtype="boolean")
 254     <BooleanArray>
 255     [True, False, <NA>]
 256     Length: 3, dtype: boolean
 257     """
 258 
 259     # The value used to fill '_data' to avoid upcasting
 260     _internal_fill_value = False

 262     def __init__(self, values: np.ndarray, mask: np.ndarray, copy: bool = False):
 263         if not (isinstance(values, np.ndarray) and values.dtype == np.bool_):
 264             raise TypeError(
 265                 "values should be boolean numpy array. Use "
 266                 "the 'pd.array' function instead"
 267             )
 268         self._dtype = BooleanDtype()
 269         super().__init__(values, mask, copy=copy)

 283     def _from_sequence_of_strings(
 284         cls, strings: List[str], dtype=None, copy: bool = False
 285     ) -> "BooleanArray":
 286         def map_string(s):
 287             if isna(s):
 288                 return s
 289             elif s in ["True", "TRUE", "true"]:
 290                 return True
 291             elif s in ["False", "FALSE", "false"]:
 292                 return False
 293             else:
 294                 raise ValueError(f"{s} cannot be cast to bool")
 295 
 296         scalars = [map_string(x) for x in strings]
 297         return cls._from_sequence(scalars, dtype, copy)

 301     def __array_ufunc__(self, ufunc, method: str, *inputs, **kwargs):
 302         # For BooleanArray inputs, we apply the ufunc to ._data
 303         # and mask the result.
 304         if method == "reduce":
 305             # Not clear how to handle missing values in reductions. Raise.
 306             raise NotImplementedError("The 'reduce' method is not supported.")
 307         out = kwargs.get("out", ())
 308 
 309         for x in inputs + out:
 310             if not isinstance(x, self._HANDLED_TYPES + (BooleanArray,)):
 311                 return NotImplemented
 312 
 313         # for binary ops, use our custom dunder methods
 314         result = ops.maybe_dispatch_ufunc_to_dunder_op(
 315             self, ufunc, method, *inputs, **kwargs
 316         )
 317         if result is not NotImplemented:
 318             return result
 319 
 320         mask = np.zeros(len(self), dtype=bool)
 321         inputs2 = []
 322         for x in inputs:
 323             if isinstance(x, BooleanArray):
 324                 mask |= x._mask
 325                 inputs2.append(x._data)
 326             else:
 327                 inputs2.append(x)
 328 
 329         def reconstruct(x):
 330             # we don't worry about scalar `x` here, since we
 331             # raise for reduce up above.
 332 
 333             if is_bool_dtype(x.dtype):
 334                 m = mask.copy()
 335                 return BooleanArray(x, m)
 336             else:
 337                 x[mask] = np.nan
 338             return x
 339 
 340         result = getattr(ufunc, method)(*inputs2, **kwargs)
 341         if isinstance(result, tuple):
 342             tuple(reconstruct(x) for x in result)
 343         else:
 344             return reconstruct(result)

 349     def astype(self, dtype, copy: bool = True) -> ArrayLike:
 350         """
 351         Cast to a NumPy array or ExtensionArray with 'dtype'.
 352 
 353         Parameters
 354         ----------
 355         dtype : str or dtype
 356             Typecode or data-type to which the array is cast.
 357         copy : bool, default True
 358             Whether to copy the data, even if not necessary. If False,
 359             a copy is made only if the old dtype does not match the
 360             new dtype.
 361 
 362         Returns
 363         -------
 364         ndarray or ExtensionArray
 365             NumPy ndarray, BooleanArray or IntegerArray with 'dtype' for its dtype.
 366 
 367         Raises
 368         ------
 369         TypeError
 370             if incompatible type with an BooleanDtype, equivalent of same_kind
 371             casting
 372         """
 373         from pandas.core.arrays.string_ import StringDtype
 374 
 375         dtype = pandas_dtype(dtype)
 376 
 377         if isinstance(dtype, BooleanDtype):
 378             values, mask = coerce_to_array(self, copy=copy)
 379             return BooleanArray(values, mask, copy=False)
 380         elif isinstance(dtype, StringDtype):
 381             return dtype.construct_array_type()._from_sequence(self, copy=False)
 382 
 383         if is_bool_dtype(dtype):
 384             # astype_nansafe converts np.nan to True
 385             if self._hasna:
 386                 raise ValueError("cannot convert float NaN to bool")
 387             else:
 388                 return self._data.astype(dtype, copy=copy)
 389         if is_extension_array_dtype(dtype) and is_integer_dtype(dtype):
 390             from pandas.core.arrays import IntegerArray
 391 
 392             return IntegerArray(
 393                 self._data.astype(dtype.numpy_dtype), self._mask.copy(), copy=False
 394             )
 395         # for integer, error if there are missing values
 396         if is_integer_dtype(dtype):
 397             if self._hasna:
 398                 raise ValueError("cannot convert NA to integer")
 399         # for float dtype, ensure we use np.nan before casting (numpy cannot
 400         # deal with pd.NA)
 401         na_value = self._na_value
 402         if is_float_dtype(dtype):
 403             na_value = np.nan
 404         # coerce
 405         return self.to_numpy(dtype=dtype, na_value=na_value, copy=False)

 425     def any(self, skipna: bool = True, **kwargs):
 426         """
 427         Return whether any element is True.
 428 
 429         Returns False unless there is at least one element that is True.
 430         By default, NAs are skipped. If ``skipna=False`` is specified and
 431         missing values are present, similar :ref:`Kleene logic <boolean.kleene>`
 432         is used as for logical operations.
 433 
 434         Parameters
 435         ----------
 436         skipna : bool, default True
 437             Exclude NA values. If the entire array is NA and `skipna` is
 438             True, then the result will be False, as for an empty array.
 439             If `skipna` is False, the result will still be True if there is
 440             at least one element that is True, otherwise NA will be returned
 441             if there are NA's present.
 442         **kwargs : any, default None
 443             Additional keywords have no effect but might be accepted for
 444             compatibility with NumPy.
 445 
 446         Returns
 447         -------
 448         bool or :attr:`pandas.NA`
 449 
 450         See Also
 451         --------
 452         numpy.any : Numpy version of this method.
 453         BooleanArray.all : Return whether all elements are True.
 454 
 455         Examples
 456         --------
 457         The result indicates whether any element is True (and by default
 458         skips NAs):
 459 
 460         >>> pd.array([True, False, True]).any()
 461         True
 462         >>> pd.array([True, False, pd.NA]).any()
 463         True
 464         >>> pd.array([False, False, pd.NA]).any()
 465         False
 466         >>> pd.array([], dtype="boolean").any()
 467         False
 468         >>> pd.array([pd.NA], dtype="boolean").any()
 469         False
 470 
 471         With ``skipna=False``, the result can be NA if this is logically
 472         required (whether ``pd.NA`` is True or False influences the result):
 473 
 474         >>> pd.array([True, False, pd.NA]).any(skipna=False)
 475         True
 476         >>> pd.array([False, False, pd.NA]).any(skipna=False)
 477         <NA>
 478         """
 479         kwargs.pop("axis", None)
 480         nv.validate_any((), kwargs)
 481 
 482         values = self._data.copy()
 483         np.putmask(values, self._mask, False)
 484         result = values.any()
 485         if skipna:
 486             return result
 487         else:
 488             if result or len(self) == 0 or not self._mask.any():
 489                 return result
 490             else:
 491                 return self.dtype.na_value

 560     @classmethod
 561     def _create_logical_method(cls, op):
 562         def logical_method(self, other):
 563             if isinstance(other, (ABCDataFrame, ABCSeries, ABCIndexClass)):
 564                 # Rely on pandas to unbox and dispatch to us.
 565                 return NotImplemented
 566 
 567             assert op.__name__ in {"or_", "ror_", "and_", "rand_", "xor", "rxor"}
 568             other = lib.item_from_zerodim(other)
 569             other_is_booleanarray = isinstance(other, BooleanArray)
 570             other_is_scalar = lib.is_scalar(other)
 571             mask = None
 572 
 573             if other_is_booleanarray:
 574                 other, mask = other._data, other._mask
 575             elif is_list_like(other):
 576                 other = np.asarray(other, dtype="bool")
 577                 if other.ndim > 1:
 578                     raise NotImplementedError(
 579                         "can only perform ops with 1-d structures"
 580                     )
 581                 other, mask = coerce_to_array(other, copy=False)
 582             elif isinstance(other, np.bool_):
 583                 other = other.item()
 584 
 585             if other_is_scalar and not (other is libmissing.NA or lib.is_bool(other)):
 586                 raise TypeError(
 587                     "'other' should be pandas.NA or a bool. "
 588                     f"Got {type(other).__name__} instead."
 589                 )
 590 
 591             if not other_is_scalar and len(self) != len(other):
 592                 raise ValueError("Lengths must match to compare")
 593 
 594             if op.__name__ in {"or_", "ror_"}:
 595                 result, mask = ops.kleene_or(self._data, other, self._mask, mask)
 596             elif op.__name__ in {"and_", "rand_"}:
 597                 result, mask = ops.kleene_and(self._data, other, self._mask, mask)
 598             elif op.__name__ in {"xor", "rxor"}:
 599                 result, mask = ops.kleene_xor(self._data, other, self._mask, mask)
 600 
 601             return BooleanArray(result, mask)
 602 
 603         name = f"__{op.__name__}__"
 604         return set_function_name(logical_method, name, cls)

 606     @classmethod
 607     def _create_comparison_method(cls, op):
 608         def cmp_method(self, other):
 609             from pandas.arrays import IntegerArray
 610 
 611             if isinstance(
 612                 other, (ABCDataFrame, ABCSeries, ABCIndexClass, IntegerArray)
 613             ):
 614                 # Rely on pandas to unbox and dispatch to us.
 615                 return NotImplemented
 616 
 617             other = lib.item_from_zerodim(other)
 618             mask = None
 619 
 620             if isinstance(other, BooleanArray):
 621                 other, mask = other._data, other._mask
 622 
 623             elif is_list_like(other):
 624                 other = np.asarray(other)
 625                 if other.ndim > 1:
 626                     raise NotImplementedError(
 627                         "can only perform ops with 1-d structures"
 628                     )
 629                 if len(self) != len(other):
 630                     raise ValueError("Lengths must match to compare")
 631 
 632             if other is libmissing.NA:
 633                 # numpy does not handle pd.NA well as "other" scalar (it returns
 634                 # a scalar False instead of an array)
 635                 result = np.zeros_like(self._data)
 636                 mask = np.ones_like(self._data)
 637             else:
 638                 # numpy will show a DeprecationWarning on invalid elementwise
 639                 # comparisons, this will raise in the future
 640                 with warnings.catch_warnings():
 641                     warnings.filterwarnings("ignore", "elementwise", FutureWarning)
 642                     with np.errstate(all="ignore"):
 643                         result = op(self._data, other)
 644 
 645                 # nans propagate
 646                 if mask is None:
 647                     mask = self._mask.copy()
 648                 else:
 649                     mask = self._mask | mask
 650 
 651             return BooleanArray(result, mask, copy=False)
 652 
 653         name = f"__{op.__name__}"
 654         return set_function_name(cmp_method, name, cls)

 663     def _maybe_mask_result(self, result, mask, other, op_name: str):
 664         """
 665         Parameters
 666         ----------
 667         result : array-like
 668         mask : array-like bool
 669         other : scalar or array-like
 670         op_name : str

 673         # a float result

 676             op_name in ["rtruediv", "truediv"]

 678             result[mask] = np.nan

 681         if is_bool_dtype(result):
 682             return BooleanArray(result, mask, copy=False)
 683 
 684         elif is_integer_dtype(result):
 685             from pandas.core.arrays import IntegerArray
 686 
 687             return IntegerArray(result, mask, copy=False)
 688         else:
 689             result[mask] = np.nan

 723                 mask = self._mask | mask
 724 
 725             if other is libmissing.NA:
 726                 # if other is NA, the result will be all NA and we can't run the
 727                 # actual op, so we need to choose the resulting dtype manually
 728                 if op_name in {"floordiv", "rfloordiv", "mod", "rmod", "pow", "rpow"}:
 729                     dtype = "int8"
 730                 else:
 731                     dtype = "bool"
 732                 result = np.zeros(len(self._data), dtype=dtype)
 733             else:
 734                 with np.errstate(all="ignore"):
 735                     result = op(self._data, other)
 736 
 737             # divmod returns a tuple
 738             if op_name == "divmod":
 739                 div, mod = result
 740                 return (
 741                     self._maybe_mask_result(div, mask, other, "floordiv"),
 742                     self._maybe_mask_result(mod, mask, other, "mod"),
 743                 )
 744 
 745             return self._maybe_mask_result(result, mask, other, op_name)
 746 
 747         name = f"__{op_name}__"
 748         return set_function_name(boolean_arithmetic_method, name, cls)
