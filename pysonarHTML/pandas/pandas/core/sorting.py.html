  24 def get_group_index(labels, shape, sort: bool, xnull: bool):
  25     """
  26     For the particular label_list, gets the offsets into the hypothetical list
  27     representing the totally ordered cartesian product of all possible label
  28     combinations, *as long as* this space fits within int64 bounds;
  29     otherwise, though group indices identify unique combinations of
  30     labels, they cannot be deconstructed.
  31     - If `sort`, rank of returned ids preserve lexical ranks of labels.
  32       i.e. returned id's can be used to do lexical sort on labels;
  33     - If `xnull` nulls (-1 labels) are passed through.
  34 
  35     Parameters
  36     ----------
  37     labels : sequence of arrays
  38         Integers identifying levels at each location
  39     shape : sequence of ints
  40         Number of unique levels at each location
  41     sort : bool
  42         If the ranks of returned ids should match lexical ranks of labels
  43     xnull : bool
  44         If true nulls are excluded. i.e. -1 values in the labels are
  45         passed through.
  46 
  47     Returns
  48     -------
  49     An array of type int64 where two elements are equal if their corresponding
  50     labels are equal at all location.
  51 
  52     Notes
  53     -----
  54     The length of `labels` and `shape` must be identical.
  55     """
  56 
  57     def _int64_cut_off(shape) -> int:
  58         acc = 1
  59         for i, mul in enumerate(shape):
  60             acc *= int(mul)
  61             if not acc < _INT64_MAX:
  62                 return i
  63         return len(shape)
  64 
  65     def maybe_lift(lab, size):
  66         # promote nan values (assigned -1 label in lab array)
  67         # so that all output values are non-negative
  68         return (lab + 1, size + 1) if (lab == -1).any() else (lab, size)
  69 
  70     labels = map(ensure_int64, labels)
  71     if not xnull:
  72         labels, shape = map(list, zip(*map(maybe_lift, labels, shape)))
  73 
  74     labels = list(labels)
  75     shape = list(shape)
  76 
  77     # Iteratively process all the labels in chunks sized so less
  78     # than _INT64_MAX unique int ids will be required for each chunk
  79     while True:
  80         # how many levels can be done without overflow:
  81         nlev = _int64_cut_off(shape)
  82 
  83         # compute flat ids for the first `nlev` levels
  84         stride = np.prod(shape[1:nlev], dtype="i8")
  85         out = stride * labels[0].astype("i8", subok=False, copy=False)
  86 
  87         for i in range(1, nlev):
  88             if shape[i] == 0:
  89                 stride = 0
  90             else:
  91                 stride //= shape[i]
  92             out += labels[i] * stride
  93 
  94         if xnull:  # exclude nulls
  95             mask = labels[0] == -1
  96             for lab in labels[1:nlev]:
  97                 mask |= lab == -1
  98             out[mask] = -1
  99 
 100         if nlev == len(shape):  # all levels done!
 101             break
 102 
 103         # compress what has been done so far in order to avoid overflow
 104         # to retain lexical ranks, obs_ids should be sorted
 105         comp_ids, obs_ids = compress_group_index(out, sort=sort)
 106 
 107         labels = [comp_ids] + labels[nlev:]
 108         shape = [len(obs_ids)] + shape[nlev:]
 109 
 110     return out

 113 def get_compressed_ids(labels, sizes):
 114     """
 115     Group_index is offsets into cartesian product of all possible labels. This
 116     space can be huge, so this function compresses it, by computing offsets
 117     (comp_ids) into the list of unique labels (obs_group_ids).
 118 
 119     Parameters
 120     ----------
 121     labels : list of label arrays
 122     sizes : list of size of the levels
 123 
 124     Returns
 125     -------
 126     tuple of (comp_ids, obs_group_ids)
 127     """
 128     ids = get_group_index(labels, sizes, sort=True, xnull=False)
 129     return compress_group_index(ids, sort=True)

 132 def is_int64_overflow_possible(shape) -> bool:
 133     the_prod = 1
 134     for x in shape:
 135         the_prod *= int(x)
 136 
 137     return the_prod >= _INT64_MAX

 140 def decons_group_index(comp_labels, shape):
 141     # reconstruct labels
 142     if is_int64_overflow_possible(shape):
 143         # at some point group indices are factorized,
 144         # and may not be deconstructed here! wrong path!
 145         raise ValueError("cannot deconstruct factorized group indices!")
 146 
 147     label_list = []
 148     factor = 1
 149     y = 0
 150     x = comp_labels
 151     for i in reversed(range(len(shape))):
 152         labels = (x - y) % (factor * shape[i]) // factor
 153         np.putmask(labels, comp_labels < 0, -1)
 154         label_list.append(labels)
 155         y = labels * factor
 156         factor *= shape[i]
 157     return label_list[::-1]

 160 def decons_obs_group_ids(comp_ids, obs_ids, shape, labels, xnull: bool):
 161     """
 162     Reconstruct labels from observed group ids.
 163 
 164     Parameters
 165     ----------
 166     xnull : bool
 167         If nulls are excluded; i.e. -1 labels are passed through.
 168     """
 169     if not xnull:
 170         lift = np.fromiter(((a == -1).any() for a in labels), dtype="i8")
 171         shape = np.asarray(shape, dtype="i8") + lift
 172 
 173     if not is_int64_overflow_possible(shape):
 174         # obs ids are deconstructable! take the fast route!
 175         out = decons_group_index(obs_ids, shape)
 176         return out if xnull or not lift.any() else [x - y for x, y in zip(out, lift)]
 177 
 178     i = unique_label_indices(comp_ids)
 179     i8copy = lambda a: a.astype("i8", subok=False, copy=True)
 180     return [i8copy(lab[i]) for lab in labels]

 183 def indexer_from_factorized(labels, shape, compress: bool = True):
 184     ids = get_group_index(labels, shape, sort=True, xnull=False)
 185 
 186     if not compress:
 187         ngroups = (ids.size and ids.max()) + 1
 188     else:
 189         ids, obs = compress_group_index(ids, sort=True)
 190         ngroups = len(obs)
 191 
 192     return get_group_index_sorter(ids, ngroups)

 195 def lexsort_indexer(
 196     keys, orders=None, na_position: str = "last", key: Optional[Callable] = None
 197 ):
 198     """
 199     Performs lexical sorting on a set of keys
 200 
 201     Parameters
 202     ----------
 203     keys : sequence of arrays
 204         Sequence of ndarrays to be sorted by the indexer
 205     orders : boolean or list of booleans, optional
 206         Determines the sorting order for each element in keys. If a list,
 207         it must be the same length as keys. This determines whether the
 208         corresponding element in keys should be sorted in ascending
 209         (True) or descending (False) order. if bool, applied to all
 210         elements as above. if None, defaults to True.
 211     na_position : {'first', 'last'}, default 'last'
 212         Determines placement of NA elements in the sorted list ("last" or "first")
 213     key : Callable, optional
 214         Callable key function applied to every element in keys before sorting
 215 
 216         .. versionadded:: 1.0.0
 217     """
 218     from pandas.core.arrays import Categorical
 219 
 220     labels = []
 221     shape = []
 222     if isinstance(orders, bool):
 223         orders = [orders] * len(keys)
 224     elif orders is None:
 225         orders = [True] * len(keys)
 226 
 227     keys = [ensure_key_mapped(k, key) for k in keys]
 228 
 229     for k, order in zip(keys, orders):
 230         # we are already a Categorical
 231         if is_categorical_dtype(k):
 232             cat = k
 233 
 234         # create the Categorical
 235         else:
 236             cat = Categorical(k, ordered=True)
 237 
 238         if na_position not in ["last", "first"]:
 239             raise ValueError(f"invalid na_position: {na_position}")
 240 
 241         n = len(cat.categories)
 242         codes = cat.codes.copy()
 243 
 244         mask = cat.codes == -1
 245         if order:  # ascending
 246             if na_position == "last":
 247                 codes = np.where(mask, n, codes)
 248             elif na_position == "first":
 249                 codes += 1
 250         else:  # not order means descending
 251             if na_position == "last":
 252                 codes = np.where(mask, n, n - codes - 1)
 253             elif na_position == "first":
 254                 codes = np.where(mask, 0, n - codes)
 255         if mask.any():
 256             n += 1
 257 
 258         shape.append(n)
 259         labels.append(codes)
 260 
 261     return indexer_from_factorized(labels, shape)

 264 def nargsort(
 265     items,
 266     kind: str = "quicksort",
 267     ascending: bool = True,
 268     na_position: str = "last",
 269     key: Optional[Callable] = None,
 270 ):
 271     """
 272     Intended to be a drop-in replacement for np.argsort which handles NaNs.
 273 
 274     Adds ascending, na_position, and key parameters.
 275 
 276     (GH #6399, #5231, #27237)
 277 
 278     Parameters
 279     ----------
 280     kind : str, default 'quicksort'
 281     ascending : bool, default True
 282     na_position : {'first', 'last'}, default 'last'
 283     key : Optional[Callable], default None
 284     """
 285 
 286     if key is not None:
 287         items = ensure_key_mapped(items, key)
 288         return nargsort(
 289             items, kind=kind, ascending=ascending, na_position=na_position, key=None
 290         )
 291 
 292     items = extract_array(items)
 293     mask = np.asarray(isna(items))
 294 
 295     if is_extension_array_dtype(items):
 296         items = items._values_for_argsort()
 297     else:
 298         items = np.asanyarray(items)
 299 
 300     idx = np.arange(len(items))
 301     non_nans = items[~mask]
 302     non_nan_idx = idx[~mask]
 303 
 304     nan_idx = np.nonzero(mask)[0]
 305     if not ascending:
 306         non_nans = non_nans[::-1]
 307         non_nan_idx = non_nan_idx[::-1]
 308     indexer = non_nan_idx[non_nans.argsort(kind=kind)]
 309     if not ascending:
 310         indexer = indexer[::-1]
 311     # Finally, place the NaNs at the end or the beginning according to
 312     # na_position
 313     if na_position == "last":
 314         indexer = np.concatenate([indexer, nan_idx])
 315     elif na_position == "first":
 316         indexer = np.concatenate([nan_idx, indexer])
 317     else:
 318         raise ValueError(f"invalid na_position: {na_position}")
 319     return indexer

 322 def ensure_key_mapped_multiindex(index, key: Callable, level=None):
 323     """
 324     Returns a new MultiIndex in which key has been applied
 325     to all levels specified in level (or all levels if level
 326     is None). Used for key sorting for MultiIndex.
 327 
 328     Parameters
 329     ----------
 330     index : MultiIndex
 331         Index to which to apply the key function on the
 332         specified levels.
 333     key : Callable
 334         Function that takes an Index and returns an Index of
 335         the same shape. This key is applied to each level
 336         separately. The name of the level can be used to
 337         distinguish different levels for application.
 338     level : list-like, int or str, default None
 339         Level or list of levels to apply the key function to.
 340         If None, key function is applied to all levels. Other
 341         levels are left unchanged.
 342 
 343     Returns
 344     -------
 345     labels : MultiIndex
 346         Resulting MultiIndex with modified levels.
 347     """
 348     from pandas.core.indexes.api import MultiIndex
 349 
 350     if level is not None:
 351         if isinstance(level, (str, int)):
 352             sort_levels = [level]
 353         else:
 354             sort_levels = level
 355 
 356         sort_levels = [index._get_level_number(lev) for lev in sort_levels]
 357     else:
 358         sort_levels = list(range(index.nlevels))  # satisfies mypy
 359 
 360     mapped = [
 361         ensure_key_mapped(index._get_level_values(level), key)
 362         if level in sort_levels
 363         else index._get_level_values(level)
 364         for level in range(index.nlevels)
 365     ]
 366 
 367     labels = MultiIndex.from_arrays(mapped)
 368 
 369     return labels

 372 def ensure_key_mapped(values, key: Optional[Callable], levels=None):
 373     """
 374     Applies a callable key function to the values function and checks
 375     that the resulting value has the same shape. Can be called on Index
 376     subclasses, Series, DataFrames, or ndarrays.
 377 
 378     Parameters
 379     ----------
 380     values : Series, DataFrame, Index subclass, or ndarray
 381     key : Optional[Callable], key to be called on the values array
 382     levels : Optional[List], if values is a MultiIndex, list of levels to
 383     apply the key to.
 384     """
 385     from pandas.core.indexes.api import Index
 386 
 387     if not key:
 388         return values
 389 
 390     if isinstance(values, ABCMultiIndex):
 391         return ensure_key_mapped_multiindex(values, key, level=levels)
 392 
 393     result = key(values.copy())
 394     if len(result) != len(values):
 395         raise ValueError(
 396             "User-provided `key` function must not change the shape of the array."
 397         )
 398 
 399     try:
 400         if isinstance(
 401             values, Index
 402         ):  # convert to a new Index subclass, not necessarily the same
 403             result = Index(result)
 404         else:
 405             type_of_values = type(values)
 406             result = type_of_values(result)  # try to revert to original type otherwise
 407     except TypeError:
 408         raise TypeError(
 409             f"User-provided `key` function returned an invalid type {type(result)} \
 410             which could not be converted to {type(values)}."
 411         )
 412 
 413     return result

 416 class _KeyMapper:
 417     """
 418     Map compressed group id -> key tuple.
 419     """
 420 
 421     def __init__(self, comp_ids, ngroups: int, levels, labels):
 422         self.levels = levels
 423         self.labels = labels
 424         self.comp_ids = comp_ids.astype(np.int64)
 425 
 426         self.k = len(labels)
 427         self.tables = [hashtable.Int64HashTable(ngroups) for _ in range(self.k)]
 428 
 429         self._populate_tables()
 430 
 431     def _populate_tables(self):
 432         for labs, table in zip(self.labels, self.tables):
 433             table.map(self.comp_ids, labs.astype(np.int64))
 434 
 435     def get_key(self, comp_id):
 436         return tuple(
 437             level[table.get_item(comp_id)]
 438             for table, level in zip(self.tables, self.levels)
 439         )

 442 def get_flattened_iterator(comp_ids, ngroups, levels, labels):
 443     # provide "flattened" iterator for multi-group setting
 444     mapper = _KeyMapper(comp_ids, ngroups, levels, labels)
 445     return [mapper.get_key(i) for i in range(ngroups)]

 448 def get_indexer_dict(label_list, keys):
 449     """
 450     Returns
 451     -------
 452     dict
 453         Labels mapped to indexers.
 454     """
 455     shape = [len(x) for x in keys]
 456 
 457     group_index = get_group_index(label_list, shape, sort=True, xnull=True)
 458     ngroups = (
 459         ((group_index.size and group_index.max()) + 1)
 460         if is_int64_overflow_possible(shape)
 461         else np.prod(shape, dtype="i8")
 462     )
 463 
 464     sorter = get_group_index_sorter(group_index, ngroups)
 465 
 466     sorted_labels = [lab.take(sorter) for lab in label_list]
 467     group_index = group_index.take(sorter)
 468 
 469     return lib.indices_fast(sorter, group_index, keys, sorted_labels)

 476 def get_group_index_sorter(group_index, ngroups: int):
 477     """
 478     algos.groupsort_indexer implements `counting sort` and it is at least
 479     O(ngroups), where
 480         ngroups = prod(shape)
 481         shape = map(len, keys)
 482     that is, linear in the number of combinations (cartesian product) of unique
 483     values of groupby keys. This can be huge when doing multi-key groupby.
 484     np.argsort(kind='mergesort') is O(count x log(count)) where count is the
 485     length of the data-frame;
 486     Both algorithms are `stable` sort and that is necessary for correctness of
 487     groupby operations. e.g. consider:
 488         df.groupby(key)[col].transform('first')
 489     """
 490     count = len(group_index)
 491     alpha = 0.0  # taking complexities literally; there may be
 492     beta = 1.0  # some room for fine-tuning these parameters
 493     do_groupsort = count > 0 and ((alpha + beta * ngroups) < (count * np.log(count)))
 494     if do_groupsort:
 495         sorter, _ = algos.groupsort_indexer(ensure_int64(group_index), ngroups)
 496         return ensure_platform_int(sorter)
 497     else:
 498         return group_index.argsort(kind="mergesort")

 501 def compress_group_index(group_index, sort: bool = True):
 502     """
 503     Group_index is offsets into cartesian product of all possible labels. This
 504     space can be huge, so this function compresses it, by computing offsets
 505     (comp_ids) into the list of unique labels (obs_group_ids).
 506     """
 507     size_hint = min(len(group_index), hashtable._SIZE_HINT_LIMIT)
 508     table = hashtable.Int64HashTable(size_hint)
 509 
 510     group_index = ensure_int64(group_index)
 511 
 512     # note, group labels come out ascending (ie, 1,2,3 etc)
 513     comp_ids, obs_group_ids = table.get_labels_groupby(group_index)
 514 
 515     if sort and len(obs_group_ids) > 0:
 516         obs_group_ids, comp_ids = _reorder_by_uniques(obs_group_ids, comp_ids)
 517 
 518     return comp_ids, obs_group_ids

 521 def _reorder_by_uniques(uniques, labels):
 522     # sorter is index where elements ought to go
 523     sorter = uniques.argsort()
 524 
 525     # reverse_indexer is where elements came from
 526     reverse_indexer = np.empty(len(sorter), dtype=np.int64)
 527     reverse_indexer.put(sorter, np.arange(len(sorter)))
 528 
 529     mask = labels < 0
 530 
 531     # move labels to right locations (ie, unsort ascending labels)
 532     labels = algorithms.take_nd(reverse_indexer, labels, allow_fill=False)
 533     np.putmask(labels, mask, -1)
 534 
 535     # sort observed ids
 536     uniques = algorithms.take_nd(uniques, sorter, allow_fill=False)
 537 
 538     return uniques, labels
