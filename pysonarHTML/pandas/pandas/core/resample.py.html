 350     def _groupby_and_aggregate(self, how, grouper=None, *args, **kwargs):
 351         """
 352         Re-evaluate the obj with a groupby aggregation.
 353         """
 354         if grouper is None:
 355             self._set_binner()
 356             grouper = self.grouper
 357 
 358         obj = self._selected_obj
 359 
 360         grouped = get_groupby(obj, by=None, grouper=grouper, axis=self.axis)
 361 
 362         try:
 363             if isinstance(obj, ABCDataFrame) and callable(how):
 364                 # Check if the function is reducing or not.
 365                 result = grouped._aggregate_item_by_item(how, *args, **kwargs)
 366             else:
 367                 result = grouped.aggregate(how, *args, **kwargs)
 368         except DataError:
 369             # we have a non-reducing function; try to evaluate
 370             result = grouped.apply(how, *args, **kwargs)
 371         except ValueError as err:
 372             if "Must produce aggregated value" in str(err):
 373                 # raised in _aggregate_named
 374                 pass
 375             elif "len(index) != len(labels)" in str(err):
 376                 # raised in libgroupby validation
 377                 pass

 381                 #  _apply_to_column_groupbys (ohlc) or DataFrameGroupBy.nunique
 382                 pass
 383             else:
 384                 raise
 385 
 386             # we have a non-reducing function
 387             # try to evaluate
 388             result = grouped.apply(how, *args, **kwargs)
 389 
 390         result = self._apply_loffset(result)
 391         return self._wrap_result(result)

 459     def nearest(self, limit=None):
 460         """
 461         Resample by using the nearest value.
 462 
 463         When resampling data, missing values may appear (e.g., when the
 464         resampling frequency is higher than the original frequency).
 465         The `nearest` method will replace ``NaN`` values that appeared in
 466         the resampled data with the value from the nearest member of the
 467         sequence, based on the index value.
 468         Missing values that existed in the original data will not be modified.
 469         If `limit` is given, fill only this many values in each direction for
 470         each of the original values.
 471 
 472         Parameters
 473         ----------
 474         limit : int, optional

 482

 486         pad : Forward fill ``NaN`` values.

 491         ...               index=pd.date_range('20180101',

 623     def fillna(self, method, limit=None):
 624         """
 625         Fill missing values introduced by upsampling.
 626 
 627         In statistics, imputation is the process of replacing missing data with
 628         substituted values [1]_. When resampling data, missing values may
 629         appear (e.g., when the resampling frequency is higher than the original
 630         frequency).
 631 
 632         Missing values that existed in the original data will
 633         not be modified.
 634 
 635         Parameters
 636         ----------
 637         method : {'pad', 'backfill', 'ffill', 'bfill', 'nearest'}
 638             Method to use for filling holes in resampled data
 639 
 640             * 'pad' or 'ffill': use previous valid observation to fill gap
 641               (forward fill).
 642             * 'backfill' or 'bfill': use next valid observation to fill gap.
 643             * 'nearest': use nearest valid observation to fill gap.
 644 
 645         limit : int, optional
 646             Limit of how many consecutive missing values to fill.
 647 
 648         Returns
 649         -------
 650         Series or DataFrame
 651             An upsampled Series or DataFrame with missing values filled.
 652 
 653         See Also
 654         --------
 655         backfill : Backward fill NaN values in the resampled data.
 656         pad : Forward fill NaN values in the resampled data.
 657         nearest : Fill NaN values in the resampled data
 658             with nearest neighbor starting from center.

1067         fill_value : scalar, default None

1668 def _get_timestamp_range_edges(
1669     first, last, freq, closed="left", origin="start_day", offset=None
1670 ):
1671     """
1672     Adjust the `first` Timestamp to the preceding Timestamp that resides on
1673     the provided offset. Adjust the `last` Timestamp to the following
1674     Timestamp that resides on the provided offset. Input Timestamps that
1675     already reside on the offset will be adjusted depending on the type of
1676     offset and the `closed` parameter.
1677 
1678     Parameters
1679     ----------
1680     first : pd.Timestamp
1681         The beginning Timestamp of the range to be adjusted.
1682     last : pd.Timestamp
1683         The ending Timestamp of the range to be adjusted.
1684     freq : pd.DateOffset
1685         The dateoffset to which the Timestamps will be adjusted.
1686     closed : {'right', 'left'}, default None
1687         Which side of bin interval is closed.
1688     origin : {'epoch', 'start', 'start_day'} or Timestamp, default 'start_day'
1689         The timestamp on which to adjust the grouping. The timezone of origin must
1690         match the timezone of the index.
1691         If a timestamp is not used, these values are also supported:
1692 
1693         - 'epoch': `origin` is 1970-01-01
1694         - 'start': `origin` is the first value of the timeseries
1695         - 'start_day': `origin` is the first day at midnight of the timeseries
1696     offset : pd.Timedelta, default is None
1697         An offset timedelta added to the origin.
1698 
1699     Returns
1700     -------
1701     A tuple of length 2, containing the adjusted pd.Timestamp objects.
1702     """
1703     if isinstance(freq, Tick):
1704         index_tz = first.tz
1705         if isinstance(origin, Timestamp) and (origin.tz is None) != (index_tz is None):
1706             raise ValueError("The origin must have the same timezone as the index.")
1707         elif origin == "epoch":
1708             # set the epoch based on the timezone to have similar bins results when
1709             # resampling on the same kind of indexes on different timezones
1710             origin = Timestamp("1970-01-01", tz=index_tz)
1711 
1712         if isinstance(freq, Day):
1713             # _adjust_dates_anchored assumes 'D' means 24H, but first/last
1714             # might contain a DST transition (23H, 24H, or 25H).
1715             # So "pretend" the dates are naive when adjusting the endpoints
1716             first = first.tz_localize(None)
1717             last = last.tz_localize(None)
1718             if isinstance(origin, Timestamp):
1719                 origin = origin.tz_localize(None)
1720 
1721         first, last = _adjust_dates_anchored(
1722             first, last, freq, closed=closed, origin=origin, offset=offset
1723         )
1724         if isinstance(freq, Day):
1725             first = first.tz_localize(index_tz)
1726             last = last.tz_localize(index_tz)
1727     else:
1728         first = first.normalize()
1729         last = last.normalize()
1730 
1731         if closed == "left":
1732             first = Timestamp(freq.rollback(first))
1733         else:
1734             first = Timestamp(first - freq)
1735 
1736         last = Timestamp(last + freq)
1737 
1738     return first, last

1741 def _get_period_range_edges(
1742     first, last, freq, closed="left", origin="start_day", offset=None
1743 ):
1744     """
1745     Adjust the provided `first` and `last` Periods to the respective Period of
1746     the given offset that encompasses them.
1747 
1748     Parameters
1749     ----------
1750     first : pd.Period
1751         The beginning Period of the range to be adjusted.
1752     last : pd.Period
1753         The ending Period of the range to be adjusted.
1754     freq : pd.DateOffset
1755         The freq to which the Periods will be adjusted.
1756     closed : {'right', 'left'}, default None
1757         Which side of bin interval is closed.
1758     origin : {'epoch', 'start', 'start_day'}, Timestamp, default 'start_day'
1759         The timestamp on which to adjust the grouping. The timezone of origin must
1760         match the timezone of the index.
1761 
1762         If a timestamp is not used, these values are also supported:
1763 
1764         - 'epoch': `origin` is 1970-01-01
1765         - 'start': `origin` is the first value of the timeseries
1766         - 'start_day': `origin` is the first day at midnight of the timeseries
1767     offset : pd.Timedelta, default is None
1768         An offset timedelta added to the origin.
1769 
1770     Returns
1771     -------
1772     A tuple of length 2, containing the adjusted pd.Period objects.
1773     """
1774     if not all(isinstance(obj, Period) for obj in [first, last]):
1775         raise TypeError("'first' and 'last' must be instances of type Period")
1776 
1777     # GH 23882
1778     first = first.to_timestamp()
1779     last = last.to_timestamp()
1780     adjust_first = not freq.is_on_offset(first)
1781     adjust_last = freq.is_on_offset(last)
1782 
1783     first, last = _get_timestamp_range_edges(
1784         first, last, freq, closed=closed, origin=origin, offset=offset
1785     )
1786

1788     last = (last - int(adjust_last) * freq).to_period(freq)

1792 def _adjust_dates_anchored(
1793     first, last, freq, closed="right", origin="start_day", offset=None
1794 ):
1795     # First and last offsets should be calculated from the start day to fix an
1796     # error cause by resampling across multiple days when a one day period is
1797     # not a multiple of the frequency. See GH 8683
1798     # To handle frequencies that are not multiple or divisible by a day we let
1799     # the possibility to define a fixed origin timestamp. See GH 31809
1800     origin_nanos = 0  # origin == "epoch"
1801     if origin == "start_day":
1802         origin_nanos = first.normalize().value
1803     elif origin == "start":
1804         origin_nanos = first.value
1805     elif isinstance(origin, Timestamp):
1806         origin_nanos = origin.value
1807     origin_nanos += offset.value if offset else 0
1808 
1809     # GH 10117 & GH 19375. If first and last contain timezone information,
1810     # Perform the calculation in UTC in order to avoid localizing on an
1811     # Ambiguous or Nonexistent time.
1812     first_tzinfo = first.tzinfo
1813     last_tzinfo = last.tzinfo
1814     if first_tzinfo is not None:
1815         first = first.tz_convert("UTC")
1816     if last_tzinfo is not None:
1817         last = last.tz_convert("UTC")
1818 
1819     foffset = (first.value - origin_nanos) % freq.nanos
1820     loffset = (last.value - origin_nanos) % freq.nanos
1821 
1822     if closed == "right":
1823         if foffset > 0:
1824             # roll back
1825             fresult = first.value - foffset
1826         else:
1827             fresult = first.value - freq.nanos
1828 
1829         if loffset > 0:
1830             # roll forward
1831             lresult = last.value + (freq.nanos - loffset)
1832         else:
1833             # already the end of the road
1834             lresult = last.value
1835     else:  # closed == 'left'
1836         if foffset > 0:
1837             fresult = first.value - foffset
1838         else:
1839             # start of the road
1840             fresult = first.value
1841 
1842         if loffset > 0:
1843             # roll forward
1844             lresult = last.value + (freq.nanos - loffset)
1845         else:
1846             lresult = last.value + freq.nanos
1847     fresult = Timestamp(fresult)
1848     lresult = Timestamp(lresult)
1849     if first_tzinfo is not None:
1850         fresult = fresult.tz_localize("UTC").tz_convert(first_tzinfo)
1851     if last_tzinfo is not None:
1852         lresult = lresult.tz_localize("UTC").tz_convert(last_tzinfo)
1853     return fresult, lresult
