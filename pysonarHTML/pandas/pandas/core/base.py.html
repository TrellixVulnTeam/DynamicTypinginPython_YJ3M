  67     def _reset_cache(self, key: Optional[str] = None) -> None:
  68         """
  69         Reset cached properties. If ``key`` is passed, only clears that key.
  70         """
  71         if getattr(self, "_cache", None) is None:
  72             return
  73         if key is None:
  74             self._cache.clear()
  75         else:
  76             self._cache.pop(key, None)

  78     def __sizeof__(self):
  79         """
  80         Generates the total memory usage for an object that returns
  81         either a value or Series of values
  82         """
  83         if hasattr(self, "memory_usage"):
  84             mem = self.memory_usage(deep=True)
  85             return int(mem if is_scalar(mem) else mem.sum())

  87         # no memory_usage attribute, so fall back to object's 'sizeof'
  88         return super().__sizeof__()

  91 class NoNewAttributesMixin:
  92     """
  93     Mixin which prevents adding new attributes.
  94 
  95     Prevents additional attributes via xxx.attribute = "something" after a
  96     call to `self.__freeze()`. Mainly used to prevent the user from using
  97     wrong attributes on an accessor (`Series.cat/.str/.dt`).
  98 
  99     If you really want to add a new attribute at a later time, you need to use
 100     `object.__setattr__(self, key, value)`.
 101     """
 102 
 103     def _freeze(self):
 104         """
 105         Prevents setting additional attributes.
 106         """
 107         object.__setattr__(self, "__frozen", True)
 108 
 109     # prevent adding any attribute via s.xxx.new_attribute = ...
 110     def __setattr__(self, key: str, value):
 111         # _cache is used by a decorator
 112         # We need to check both 1.) cls.__dict__ and 2.) getattr(self, key)
 113         # because
 114         # 1.) getattr is false for attributes that raise errors
 115         # 2.) cls.__dict__ doesn't traverse into base classes
 116         if getattr(self, "__frozen", False) and not (
 117             key == "_cache"
 118             or key in type(self).__dict__
 119             or getattr(self, key, None) is not None
 120         ):
 121             raise AttributeError(f"You cannot add any new attribute '{key}'")
 122         object.__setattr__(self, key, value)

 212     def __getitem__(self, key):
 213         if self._selection is not None:
 214             raise IndexError(f"Column(s) {self._selection} already selected")
 215 
 216         if isinstance(key, (list, tuple, ABCSeries, ABCIndexClass, np.ndarray)):
 217             if len(self.obj.columns.intersection(key)) != len(key):
 218                 bad_keys = list(set(key).difference(self.obj.columns))
 219                 raise KeyError(f"Columns not found: {str(bad_keys)[1:-1]}")
 220             return self._gotitem(list(key), ndim=2)
 221 
 222         elif not getattr(self, "as_index", False):
 223             if key not in self.obj.columns:
 224                 raise KeyError(f"Column not found: {key}")
 225             return self._gotitem(key, ndim=2)
 226 
 227         else:
 228             if key not in self.obj:
 229                 raise KeyError(f"Column not found: {key}")
 230             return self._gotitem(key, ndim=1)

 252     def _try_aggregate_string_function(self, arg: str, *args, **kwargs):
 253         """
 254         if arg is a string, then try to operate on it:
 255         - try to find a function (or attribute) on ourselves
 256         - try to find a numpy function
 257         - raise
 258         """
 259         assert isinstance(arg, str)
 260 
 261         f = getattr(self, arg, None)
 262         if f is not None:
 263             if callable(f):
 264                 return f(*args, **kwargs)
 265 
 266             # people may try to aggregate on a non-callable attribute
 267             # but don't let them think they can pass args to it
 268             assert len(args) == 0
 269             assert len([kwarg for kwarg in kwargs if kwarg not in ["axis"]]) == 0
 270             return f
 271 
 272         f = getattr(np, arg, None)
 273         if f is not None:
 274             if hasattr(self, "__array__"):
 275                 # in particular exclude Window
 276                 return f(self, *args, **kwargs)
 277 
 278         raise AttributeError(
 279             f"'{arg}' is not a valid function for '{type(self).__name__}' object"
 280         )

 282     def _aggregate(self, arg, *args, **kwargs):
 283         """
 284         provide an implementation for the aggregators
 285 
 286         Parameters
 287         ----------
 288         arg : string, dict, function
 289         *args : args to pass on to the function
 290         **kwargs : kwargs to pass on to the function
 291 
 292         Returns
 293         -------
 294         tuple of result, how
 295 
 296         Notes
 297         -----
 298         how can be a string describe the required post-processing, or
 299         None if not required
 300         """
 301         is_aggregator = lambda x: isinstance(x, (list, tuple, dict))
 302 
 303         _axis = kwargs.pop("_axis", None)
 304         if _axis is None:
 305             _axis = getattr(self, "axis", 0)
 306 
 307         if isinstance(arg, str):
 308             return self._try_aggregate_string_function(arg, *args, **kwargs), None
 309 
 310         if isinstance(arg, dict):
 311             # aggregate based on the passed dict
 312             if _axis != 0:  # pragma: no cover
 313                 raise ValueError("Can only pass dict with axis=0")
 314 
 315             obj = self._selected_obj
 316 
 317             # if we have a dict of any non-scalars
 318             # eg. {'A' : ['mean']}, normalize all to
 319             # be list-likes
 320             if any(is_aggregator(x) for x in arg.values()):
 321                 new_arg = {}
 322                 for k, v in arg.items():
 323                     if not isinstance(v, (tuple, list, dict)):
 324                         new_arg[k] = [v]
 325                     else:
 326                         new_arg[k] = v
 327 
 328                     # the keys must be in the columns
 329                     # for ndim=2, or renamers for ndim=1
 330 
 331                     # ok for now, but deprecated
 332                     # {'A': { 'ra': 'mean' }}
 333                     # {'A': { 'ra': ['mean'] }}
 334                     # {'ra': ['mean']}
 335 
 336                     # not ok
 337                     # {'ra' : { 'A' : 'mean' }}
 338                     if isinstance(v, dict):
 339                         raise SpecificationError("nested renamer is not supported")
 340                     elif isinstance(obj, ABCSeries):
 341                         raise SpecificationError("nested renamer is not supported")
 342                     elif isinstance(obj, ABCDataFrame) and k not in obj.columns:
 343                         raise KeyError(f"Column '{k}' does not exist!")
 344 
 345                 arg = new_arg
 346 
 347             else:
 348                 # deprecation of renaming keys
 349                 # GH 15931
 350                 keys = list(arg.keys())
 351                 if isinstance(obj, ABCDataFrame) and len(
 352                     obj.columns.intersection(keys)
 353                 ) != len(keys):
 354                     cols = sorted(set(keys) - set(obj.columns.intersection(keys)))
 355                     raise SpecificationError(f"Column(s) {cols} do not exist")
 356 
 357             from pandas.core.reshape.concat import concat
 358 
 359             def _agg_1dim(name, how, subset=None):
 360                 """
 361                 aggregate a 1-dim with how
 362                 """
 363                 colg = self._gotitem(name, ndim=1, subset=subset)
 364                 if colg.ndim != 1:
 365                     raise SpecificationError(
 366                         "nested dictionary is ambiguous in aggregation"
 367                     )
 368                 return colg.aggregate(how)
 369 
 370             def _agg_2dim(how):
 371                 """
 372                 aggregate a 2-dim with how
 373                 """
 374                 colg = self._gotitem(self._selection, ndim=2, subset=obj)
 375                 return colg.aggregate(how)
 376 
 377             def _agg(arg, func):
 378                 """
 379                 run the aggregations over the arg with func
 380                 return a dict
 381                 """
 382                 result = {}
 383                 for fname, agg_how in arg.items():
 384                     result[fname] = func(fname, agg_how)
 385                 return result
 386 
 387             # set the final keys
 388             keys = list(arg.keys())
 389             result = {}
 390 
 391             if self._selection is not None:
 392 
 393                 sl = set(self._selection_list)
 394 
 395                 # we are a Series like object,
 396                 # but may have multiple aggregations
 397                 if len(sl) == 1:
 398 
 399                     result = _agg(
 400                         arg, lambda fname, agg_how: _agg_1dim(self._selection, agg_how)
 401                     )
 402 
 403                 # we are selecting the same set as we are aggregating
 404                 elif not len(sl - set(keys)):
 405 
 406                     result = _agg(arg, _agg_1dim)
 407 
 408                 # we are a DataFrame, with possibly multiple aggregations
 409                 else:
 410 
 411                     result = _agg(arg, _agg_2dim)
 412 
 413             # no selection
 414             else:
 415 
 416                 try:
 417                     result = _agg(arg, _agg_1dim)
 418                 except SpecificationError:
 419 
 420                     # we are aggregating expecting all 1d-returns
 421                     # but we have 2d
 422                     result = _agg(arg, _agg_2dim)
 423 
 424             # combine results
 425 
 426             def is_any_series() -> bool:
 427                 # return a boolean if we have *any* nested series
 428                 return any(isinstance(r, ABCSeries) for r in result.values())
 429 
 430             def is_any_frame() -> bool:
 431                 # return a boolean if we have *any* nested series
 432                 return any(isinstance(r, ABCDataFrame) for r in result.values())
 433 
 434             if isinstance(result, list):
 435                 return concat(result, keys=keys, axis=1, sort=True), True
 436 
 437             elif is_any_frame():
 438                 # we have a dict of DataFrames
 439                 # return a MI DataFrame
 440 
 441                 keys_to_use = [k for k in keys if not result[k].empty]
 442                 # Have to check, if at least one DataFrame is not empty.
 443                 keys_to_use = keys_to_use if keys_to_use != [] else keys
 444                 return (
 445                     concat([result[k] for k in keys_to_use], keys=keys_to_use, axis=1),
 446                     True,
 447                 )
 448 
 449             elif isinstance(self, ABCSeries) and is_any_series():
 450 
 451                 # we have a dict of Series
 452                 # return a MI Series
 453                 try:
 454                     result = concat(result)
 455                 except TypeError as err:
 456                     # we want to give a nice error here if
 457                     # we have non-same sized objects, so
 458                     # we don't automatically broadcast
 459 
 460                     raise ValueError(
 461                         "cannot perform both aggregation "
 462                         "and transformation operations "
 463                         "simultaneously"
 464                     ) from err
 465 
 466                 return result, True
 467 
 468             # fall thru
 469             from pandas import DataFrame, Series
 470 
 471             try:
 472                 result = DataFrame(result)
 473             except ValueError:
 474 
 475                 # we have a dict of scalars
 476                 result = Series(result, name=getattr(self, "name", None))
 477 
 478             return result, True
 479         elif is_list_like(arg):
 480             # we require a list, but not an 'str'
 481             return self._aggregate_multiple_funcs(arg, _axis=_axis), None
 482         else:
 483             result = None
 484 
 485         f = self._get_cython_func(arg)
 486         if f and not args and not kwargs:
 487             return getattr(self, f)(), None
 488 
 489         # caller can react
 490         return result, True
 491 
 492     def _aggregate_multiple_funcs(self, arg, _axis):
 493         from pandas.core.reshape.concat import concat
 494 
 495         if _axis != 0:
 496             raise NotImplementedError("axis other than 0 is not supported")
 497 
 498         if self._selected_obj.ndim == 1:
 499             obj = self._selected_obj
 500         else:
 501             obj = self._obj_with_exclusions
 502 
 503         results = []
 504         keys = []
 505 
 506         # degenerate case
 507         if obj.ndim == 1:
 508             for a in arg:
 509                 colg = self._gotitem(obj.name, ndim=1, subset=obj)
 510                 try:
 511                     new_res = colg.aggregate(a)
 512 
 513                 except TypeError:
 514                     pass
 515                 else:
 516                     results.append(new_res)
 517 
 518                     # make sure we find a good name
 519                     name = com.get_callable_name(a) or a
 520                     keys.append(name)
 521 
 522         # multiples
 523         else:
 524             for index, col in enumerate(obj):
 525                 colg = self._gotitem(col, ndim=1, subset=obj.iloc[:, index])
 526                 try:
 527                     new_res = colg.aggregate(arg)
 528                 except (TypeError, DataError):
 529                     pass
 530                 except ValueError as err:
 531                     # cannot aggregate
 532                     if "Must produce aggregated value" in str(err):
 533                         # raised directly in _aggregate_named
 534                         pass
 535                     elif "no results" in str(err):
 536                         # raised directly in _aggregate_multiple_funcs
 537                         pass
 538                     else:
 539                         raise
 540                 else:
 541                     results.append(new_res)
 542                     keys.append(col)
 543 
 544         # if we are empty
 545         if not len(results):
 546             raise ValueError("no results")
 547 
 548         try:
 549             return concat(results, keys=keys, axis=1, sort=False)
 550         except TypeError as err:
 551 
 552             # we are concatting non-NDFrame objects,
 553             # e.g. a list of scalars
 554 
 555             from pandas import Series
 556 
 557             result = Series(results, index=keys, name=self.name)
 558             if is_nested_object(result):
 559                 raise ValueError(
 560                     "cannot combine transform and aggregation operations"
 561                 ) from err
 562             return result

 581     def _shallow_copy(self, obj, **kwargs):
 582         """
 583         return a new object with the replacement attributes
 584         """
 585         if isinstance(obj, self._constructor):
 586             obj = obj.obj

 684     @property
 685     def array(self) -> ExtensionArray:
 686         """
 687         The ExtensionArray of the data backing this Series or Index.
 688 
 689         .. versionadded:: 0.24.0
 690 
 691         Returns
 692         -------
 693         ExtensionArray
 694             An ExtensionArray of the values stored within. For extension
 695             types, this is the actual array. For NumPy native types, this
 696             is a thin (no copy) wrapper around :class:`numpy.ndarray`.
 697 
 698             ``.array`` differs ``.values`` which may require converting the
 699             data to a different form.
 700 
 701         See Also
 702         --------
 703         Index.to_numpy : Similar method that always returns a NumPy array.
 704         Series.to_numpy : Similar method that always returns a NumPy array.
 705 
 706         Notes
 707         -----
 708         This table lays out the different array types for each extension
 709         dtype within pandas.
 710 
 711         ================== =============================
 712         dtype              array type
 713         ================== =============================
 714         category           Categorical
 715         period             PeriodArray
 716         interval           IntervalArray
 717         IntegerNA          IntegerArray
 718         string             StringArray
 719         boolean            BooleanArray
 720         datetime64[ns, tz] DatetimeArray
 721         ================== =============================
 722 
 723         For any 3rd-party extension types, the array type will be an
 724         ExtensionArray.
 725 
 726         For all remaining dtypes ``.array`` will be a
 727         :class:`arrays.NumpyExtensionArray` wrapping the actual ndarray
 728         stored within. If you absolutely need a NumPy array (possibly with

 751     def to_numpy(self, dtype=None, copy=False, na_value=lib.no_default, **kwargs):
 752         """
 753         A NumPy ndarray representing the values in this Series or Index.
 754 
 755         .. versionadded:: 0.24.0
 756 
 757         Parameters
 758         ----------
 759         dtype : str or numpy.dtype, optional
 760             The dtype to pass to :meth:`numpy.asarray`.
 761         copy : bool, default False
 762             Whether to ensure that the returned value is not a view on
 763             another array. Note that ``copy=False`` does not *ensure* that
 764             ``to_numpy()`` is no-copy. Rather, ``copy=True`` ensure that
 765             a copy is made, even if not strictly necessary.
 766         na_value : Any, optional
 767             The value to use for missing values. The default value depends
 768             on `dtype` and the type of the array.
 769 
 770             .. versionadded:: 1.0.0
 771 
 772         **kwargs
 773             Additional keywords passed through to the ``to_numpy`` method
 774             of the underlying array (for extension arrays).
 775 
 776             .. versionadded:: 1.0.0
 777 
 778         Returns
 779         -------
 780         numpy.ndarray
 781 
 782         See Also
 783         --------
 784         Series.array : Get the actual data stored within.
 785         Index.array : Get the actual data stored within.
 786         DataFrame.to_numpy : Similar method for DataFrame.
 787 
 788         Notes
 789         -----
 790         The returned array will be the same up to equality (values equal
 791         in `self` will be equal in the returned array; likewise for values
 792         that are not equal). When `self` contains an ExtensionArray, the
 793         dtype may be different. For example, for a category-dtype Series,
 794         ``to_numpy()`` will return a NumPy array and the categorical dtype
 795         will be lost.
 796 
 797         For NumPy dtypes, this will be a reference to the actual data stored
 798         in this Series or Index (assuming ``copy=False``). Modifying the result
 799         in place will modify the data stored in the Series or Index (not that
 800         we recommend doing that).
 801 
 802         For extension types, ``to_numpy()`` *may* require copying data and
 803         coercing the result to a NumPy type (possibly object), which may be
 804         expensive. When you need a no-copy reference to the underlying data,
 805         :attr:`Series.array` should be used instead.
 806 
 807         This table lays out the different dtypes and default return types of
 808         ``to_numpy()`` for various dtypes within pandas.
 809 
 810         ================== ================================
 811         dtype              array type
 812         ================== ================================
 813         category[T]        ndarray[T] (same dtype as input)
 814         period             ndarray[object] (Periods)
 815         interval           ndarray[object] (Intervals)
 816         IntegerNA          ndarray[object]
 817         datetime64[ns]     datetime64[ns]
 818         datetime64[ns, tz] ndarray[object] (Timestamps)
 819         ================== ================================
 820 
 821         Examples
 822         --------
 823         >>> ser = pd.Series(pd.Categorical(['a', 'b', 'a']))
 824         >>> ser.to_numpy()
 825         array(['a', 'b', 'a'], dtype=object)
 826 
 827         Specify the `dtype` to control how datetime-aware data is represented.
 828         Use ``dtype=object`` to return an ndarray of pandas :class:`Timestamp`
 829         objects, each with the correct ``tz``.
 830 
 831         >>> ser = pd.Series(pd.date_range('2000', periods=2, tz="CET"))
 832         >>> ser.to_numpy(dtype=object)
 833         array([Timestamp('2000-01-01 00:00:00+0100', tz='CET', freq='D'),

 835               dtype=object)
 836

 838         datetime64 values. The values are converted to UTC and the timezone
 839         info is dropped.
 840 
 841         >>> ser.to_numpy(dtype="datetime64[ns]")
 842         ... # doctest: +ELLIPSIS
 843         array(['1999-12-31T23:00:00.000000000', '2000-01-01T23:00:00...'],
 844               dtype='datetime64[ns]')
 845         """
 846         if is_extension_array_dtype(self.dtype):
 847             return self.array.to_numpy(dtype, copy=copy, na_value=na_value, **kwargs)
 848         elif kwargs:
 849             bad_keys = list(kwargs.keys())[0]
 850             raise TypeError(
 851                 f"to_numpy() got an unexpected keyword argument '{bad_keys}'"
 852             )
 853 
 854         result = np.asarray(self._values, dtype=dtype)
 855         # TODO(GH-24345): Avoid potential double copy
 856         if copy or na_value is not lib.no_default:
 857             result = result.copy()
 858             if na_value is not lib.no_default:
 859                 result[self.isna()] = na_value
 860         return result

1086     def _map_values(self, mapper, na_action=None):
1087         """
1088         An internal function that maps values using the input
1089         correspondence (which can be a dict, Series, or function).
1090 
1091         Parameters
1092         ----------
1093         mapper : function, dict, or Series
1094             The input correspondence object
1095         na_action : {None, 'ignore'}
1096             If 'ignore', propagate NA values, without passing them to the
1097             mapping function
1098 
1099         Returns
1100         -------
1101         Union[Index, MultiIndex], inferred
1102             The output of the mapping function applied to the index.
1103             If the function returns a tuple with more than one element
1104             a MultiIndex will be returned.
1105         """
1106         # we can fastpath dict/Series to an efficient map
1107         # as we know that we are not going to have to yield
1108         # python types
1109         if is_dict_like(mapper):
1110             if isinstance(mapper, dict) and hasattr(mapper, "__missing__"):
1111                 # If a dictionary subclass defines a default value method,
1112                 # convert mapper to a lookup function (GH #15999).
1113                 dict_with_default = mapper
1114                 mapper = lambda x: dict_with_default[x]
1115             else:
1116                 # Dictionary does not have a default. Thus it's safe to
1117                 # convert to an Series for efficiency.
1118                 # we specify the keys here to handle the
1119                 # possibility that they are tuples
1120 
1121                 # The return value of mapping with an empty mapper is
1122                 # expected to be pd.Series(np.nan, ...). As np.nan is
1123                 # of dtype float64 the return value of this method should
1124                 # be float64 as well
1125                 mapper = create_series_with_explicit_dtype(
1126                     mapper, dtype_if_empty=np.float64
1127                 )
1128 
1129         if isinstance(mapper, ABCSeries):
1130             # Since values were input this means we came from either
1131             # a dict or a series and mapper should be an index
1132             if is_categorical_dtype(self.dtype):
1133                 # use the built in categorical series mapper which saves
1134                 # time by mapping the categories instead of all values
1135                 return self._values.map(mapper)
1136 
1137             values = self._values
1138 
1139             indexer = mapper.index.get_indexer(values)
1140             new_values = algorithms.take_1d(mapper._values, indexer)
1141 
1142             return new_values
1143 
1144         # we must convert to python types
1145         if is_extension_array_dtype(self.dtype) and hasattr(self._values, "map"):
1146             # GH#23179 some EAs do not have `map`
1147             values = self._values
1148             if na_action is not None:
1149                 raise NotImplementedError
1150             map_f = lambda values, f: values.map(f)
1151         else:
1152             values = self.astype(object)._values
1153             if na_action == "ignore":
1154 
1155                 def map_f(values, f):
1156                     return lib.map_infer_mask(values, f, isna(values).view(np.uint8))
1157 
1158             elif na_action is None:
1159                 map_f = lib.map_infer
1160             else:
1161                 msg = (
1162                     "na_action must either be 'ignore' or None, "
1163                     f"{na_action} was passed"
1164                 )
1165                 raise ValueError(msg)
1166 
1167         # mapper is a function
1168         new_values = map_f(values, mapper)
1169 
1170         return new_values
