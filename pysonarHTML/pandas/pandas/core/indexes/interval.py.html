  79 def _get_next_label(label):
  80     dtype = getattr(label, "dtype", type(label))
  81     if isinstance(label, (Timestamp, Timedelta)):
  82         dtype = "datetime64"
  83     if is_datetime_or_timedelta_dtype(dtype) or is_datetime64tz_dtype(dtype):
  84         return label + np.timedelta64(1, "ns")
  85     elif is_integer_dtype(dtype):
  86         return label + 1
  87     elif is_float_dtype(dtype):
  88         return np.nextafter(label, np.infty)
  89     else:
  90         raise TypeError(f"cannot determine next label for type {repr(type(label))}")
  91 
  92 
  93 def _get_prev_label(label):
  94     dtype = getattr(label, "dtype", type(label))
  95     if isinstance(label, (Timestamp, Timedelta)):
  96         dtype = "datetime64"
  97     if is_datetime_or_timedelta_dtype(dtype) or is_datetime64tz_dtype(dtype):
  98         return label - np.timedelta64(1, "ns")
  99     elif is_integer_dtype(dtype):
 100         return label - 1
 101     elif is_float_dtype(dtype):
 102         return np.nextafter(label, -np.infty)
 103     else:
 104         raise TypeError(f"cannot determine next label for type {repr(type(label))}")

 121     def __init__(self, op_name):
 122         self.op_name = op_name
 123 
 124     def __call__(self, setop):
 125         def func(intvidx_self, other, sort=False):
 126             intvidx_self._assert_can_do_setop(other)
 127             other = ensure_index(other)
 128 
 129             if not isinstance(other, IntervalIndex):
 130                 result = getattr(intvidx_self.astype(object), self.op_name)(other)
 131                 if self.op_name in ("difference",):
 132                     result = result.astype(intvidx_self.dtype)
 133                 return result
 134             elif intvidx_self.closed != other.closed:
 135                 raise ValueError(
 136                     "can only do set operations between two IntervalIndex "
 137                     "objects that are closed on the same side"
 138                 )
 139 
 140             # GH 19016: ensure set op will not return a prohibited dtype
 141             subtypes = [intvidx_self.dtype.subtype, other.dtype.subtype]
 142             common_subtype = find_common_type(subtypes)
 143             if is_object_dtype(common_subtype):
 144                 raise TypeError(
 145                     f"can only do {self.op_name} between two IntervalIndex "
 146                     "objects that have compatible dtypes"
 147                 )
 148 
 149             return setop(intvidx_self, other, sort)
 150 
 151         return func

 438     @cache_readonly
 439     def is_unique(self):
 440         """
 441         Return True if the IntervalIndex contains unique elements, else False.
 442         """
 443         left = self.left
 444         right = self.right
 445 
 446         if self.isna().sum() > 1:
 447             return False
 448 
 449         if left.is_unique or right.is_unique:
 450             return True
 451 
 452         seen_pairs = set()
 453         check_idx = np.where(left.duplicated(keep=False))[0]
 454         for idx in check_idx:
 455             pair = (left[idx], right[idx])
 456             if pair in seen_pairs:
 457                 return False
 458             seen_pairs.add(pair)
 459 
 460         return True

 555     def _needs_i8_conversion(self, key) -> bool:
 556         """
 557         Check if a given key needs i8 conversion. Conversion is necessary for
 558         Timestamp, Timedelta, DatetimeIndex, and TimedeltaIndex keys. An
 559         Interval-like requires conversion if it's endpoints are one of the
 560         aforementioned types.
 561 
 562         Assumes that any list-like data has already been cast to an Index.
 563 
 564         Parameters
 565         ----------
 566         key : scalar or Index-like
 567             The key that should be checked for i8 conversion
 568 
 569         Returns
 570         -------
 571         bool
 572         """
 573         if is_interval_dtype(key) or isinstance(key, Interval):
 574             return self._needs_i8_conversion(key.left)
 575 
 576         i8_types = (Timestamp, Timedelta, DatetimeIndex, TimedeltaIndex)
 577         return isinstance(key, i8_types)

 579     def _maybe_convert_i8(self, key):
 580         """
 581         Maybe convert a given key to it's equivalent i8 value(s). Used as a
 582         preprocessing step prior to IntervalTree queries (self._engine), which
 583         expects numeric data.
 584 
 585         Parameters
 586         ----------
 587         key : scalar or list-like
 588             The key that should maybe be converted to i8.
 589 
 590         Returns
 591         -------
 592         scalar or list-like
 593             The original key if no conversion occurred, int if converted scalar,
 594             Int64Index if converted list-like.
 595         """
 596         original = key
 597         if is_list_like(key):
 598             key = ensure_index(key)
 599 
 600         if not self._needs_i8_conversion(key):
 601             return original
 602 
 603         scalar = is_scalar(key)
 604         if is_interval_dtype(key) or isinstance(key, Interval):
 605             # convert left/right and reconstruct
 606             left = self._maybe_convert_i8(key.left)
 607             right = self._maybe_convert_i8(key.right)
 608             constructor = Interval if scalar else IntervalIndex.from_arrays
 609             return constructor(left, right, closed=self.closed)
 610 
 611         if scalar:
 612             # Timestamp/Timedelta
 613             key_dtype, key_i8 = infer_dtype_from_scalar(key, pandas_dtype=True)
 614         else:
 615             # DatetimeIndex/TimedeltaIndex
 616             key_dtype, key_i8 = key.dtype, Index(key.asi8)
 617             if key.hasnans:
 618                 # convert NaT from it's i8 value to np.nan so it's not viewed
 619                 # as a valid value, maybe causing errors (e.g. is_overlapping)
 620                 key_i8 = key_i8.where(~key._isnan)
 621 
 622         # ensure consistency with IntervalIndex subtype
 623         subtype = self.dtype.subtype
 624 
 625         if not is_dtype_equal(subtype, key_dtype):
 626             raise ValueError(
 627                 f"Cannot index an IntervalIndex of subtype {subtype} with "
 628                 f"values of dtype {key_dtype}"
 629             )
 630 
 631         return key_i8

 644     def _searchsorted_monotonic(self, label, side, exclude_label=False):
 645         if not self.is_non_overlapping_monotonic:
 646             raise KeyError(
 647                 "can only get slices from an IntervalIndex if bounds are "
 648                 "non-overlapping and all monotonic increasing or decreasing"
 649             )
 650 
 651         if isinstance(label, IntervalMixin):
 652             raise NotImplementedError("Interval objects are not currently supported")
 653 
 654         # GH 20921: "not is_monotonic_increasing" for the second condition
 655         # instead of "is_monotonic_decreasing" to account for single element
 656         # indexes being both increasing and decreasing
 657         if (side == "left" and self.left.is_monotonic_increasing) or (
 658             side == "right" and not self.left.is_monotonic_increasing
 659         ):
 660             sub_idx = self.right
 661             if self.open_right or exclude_label:
 662                 label = _get_next_label(label)
 663         else:
 664             sub_idx = self.left
 665             if self.open_left or exclude_label:
 666                 label = _get_prev_label(label)
 667 
 668         return sub_idx._searchsorted_monotonic(label, side)

 678         key : label
 679         method : {None}, optional
 680             * default: matches where the label is within an interval only.
 681 
 682         Returns
 683         -------
 684         int if unique index, slice if monotonic index, else mask
 685 
 686         Examples
 687         --------
 688         >>> i1, i2 = pd.Interval(0, 1), pd.Interval(1, 2)
 689         >>> index = pd.IntervalIndex([i1, i2])
 690         >>> index.get_loc(1)
 691         0
 692 
 693         You can also supply a point inside an interval.
 694 
 695         >>> index.get_loc(1.5)
 696         1
 697 
 698         If a label is in several intervals, you get the locations of all the
 699         relevant intervals.
 700 
 701         >>> i3 = pd.Interval(0, 2)
 702         >>> overlapping_index = pd.IntervalIndex([i1, i2, i3])
 703         >>> overlapping_index.get_loc(0.5)
 704         array([ True, False,  True])
 705 
 706         Only exact matches will be returned if an interval is provided.
 707 
 708         >>> index.get_loc(pd.Interval(0, 1))
 709         0
 710         """
 711         self._check_method(method)
 712 
 713         if not is_scalar(key):
 714             raise InvalidIndexError(key)
 715 
 716         if isinstance(key, Interval):
 717             if self.closed != key.closed:
 718                 raise KeyError(key)
 719             mask = (self.left == key.left) & (self.right == key.right)
 720         else:
 721             # assume scalar
 722             op_left = le if self.closed_left else lt
 723             op_right = le if self.closed_right else lt
 724             try:
 725                 mask = op_left(self.left, key) & op_right(key, self.right)
 726             except TypeError as err:
 727                 # scalar is not comparable to II subtype --> invalid label
 728                 raise KeyError(key) from err
 729 
 730         matches = mask.sum()
 731         if matches == 0:
 732             raise KeyError(key)
 733         elif matches == 1:
 734             return mask.argmax()
 735         return lib.maybe_booleans_to_slice(mask.view("u1"))

 753     @Appender(_index_shared_docs["get_indexer"])
 754     def get_indexer(
 755         self,
 756         target: AnyArrayLike,
 757         method: Optional[str] = None,
 758         limit: Optional[int] = None,
 759         tolerance: Optional[Any] = None,
 760     ) -> np.ndarray:
 761 
 762         self._check_method(method)
 763 
 764         if self.is_overlapping:
 765             raise InvalidIndexError(
 766                 "cannot handle overlapping indices; "
 767                 "use IntervalIndex.get_indexer_non_unique"
 768             )
 769 
 770         target_as_index = ensure_index(target)
 771 
 772         if isinstance(target_as_index, IntervalIndex):
 773             # equal indexes -> 1:1 positional match
 774             if self.equals(target_as_index):
 775                 return np.arange(len(self), dtype="intp")
 776 
 777             # different closed or incompatible subtype -> no matches
 778             common_subtype = find_common_type(
 779                 [self.dtype.subtype, target_as_index.dtype.subtype]
 780             )
 781             if self.closed != target_as_index.closed or is_object_dtype(common_subtype):
 782                 return np.repeat(np.intp(-1), len(target_as_index))
 783 
 784             # non-overlapping -> at most one match per interval in target_as_index
 785             # want exact matches -> need both left/right to match, so defer to
 786             # left/right get_indexer, compare elementwise, equality -> match
 787             left_indexer = self.left.get_indexer(target_as_index.left)
 788             right_indexer = self.right.get_indexer(target_as_index.right)
 789             indexer = np.where(left_indexer == right_indexer, left_indexer, -1)
 790         elif is_categorical_dtype(target_as_index.dtype):
 791             # get an indexer for unique categories then propagate to codes via take_1d
 792             categories_indexer = self.get_indexer(target_as_index.categories)
 793             indexer = take_1d(categories_indexer, target_as_index.codes, fill_value=-1)
 794         elif not is_object_dtype(target_as_index):
 795             # homogeneous scalar index: use IntervalTree
 796             target_as_index = self._maybe_convert_i8(target_as_index)
 797             indexer = self._engine.get_indexer(target_as_index.values)
 798         else:
 799             # heterogeneous scalar index: defer elementwise to get_loc
 800             # (non-overlapping so get_loc guarantees scalar of KeyError)
 801             indexer = []
 802             for key in target_as_index:
 803                 try:
 804                     loc = self.get_loc(key)
 805                 except KeyError:
 806                     loc = -1
 807                 except InvalidIndexError as err:
 808                     # i.e. non-scalar key
 809                     raise TypeError(key) from err
 810                 indexer.append(loc)
 811 
 812         return ensure_platform_int(indexer)

 814     @Appender(_index_shared_docs["get_indexer_non_unique"] % _index_doc_kwargs)
 815     def get_indexer_non_unique(
 816         self, target: AnyArrayLike
 817     ) -> Tuple[np.ndarray, np.ndarray]:
 818         target_as_index = ensure_index(target)
 819 
 820         # check that target_as_index IntervalIndex is compatible
 821         if isinstance(target_as_index, IntervalIndex):
 822             common_subtype = find_common_type(
 823                 [self.dtype.subtype, target_as_index.dtype.subtype]
 824             )
 825             if self.closed != target_as_index.closed or is_object_dtype(common_subtype):
 826                 # different closed or incompatible subtype -> no matches
 827                 return (
 828                     np.repeat(-1, len(target_as_index)),
 829                     np.arange(len(target_as_index)),
 830                 )
 831 
 832         if is_object_dtype(target_as_index) or isinstance(
 833             target_as_index, IntervalIndex
 834         ):
 835             # target_as_index might contain intervals: defer elementwise to get_loc
 836             indexer, missing = [], []
 837             for i, key in enumerate(target_as_index):
 838                 try:
 839                     locs = self.get_loc(key)
 840                     if isinstance(locs, slice):
 841                         locs = np.arange(locs.start, locs.stop, locs.step, dtype="intp")
 842                     locs = np.array(locs, ndmin=1)
 843                 except KeyError:
 844                     missing.append(i)
 845                     locs = np.array([-1])
 846                 indexer.append(locs)
 847             indexer = np.concatenate(indexer)
 848         else:
 849             target_as_index = self._maybe_convert_i8(target_as_index)
 850             indexer, missing = self._engine.get_indexer_non_unique(
 851                 target_as_index.values
 852             )
 853 
 854         return ensure_platform_int(indexer), ensure_platform_int(missing)

1050     def _intersection_unique(self, other: "IntervalIndex") -> "IntervalIndex":
1051         """
1052         Used when the IntervalIndex does not have any common endpoint,
1053         no mater left or right.

1057         ----------

1174 def interval_range(
1175     start=None, end=None, periods=None, freq=None, name=None, closed="right"
1176 ):
1177     """
1178     Return a fixed frequency IntervalIndex.
1179 
1180     Parameters
1181     ----------
1182     start : numeric or datetime-like, default None
1183         Left bound for generating intervals.
1184     end : numeric or datetime-like, default None
1185         Right bound for generating intervals.
1186     periods : int, default None
1187         Number of periods to generate.
1188     freq : numeric, str, or DateOffset, default None
1189         The length of each interval. Must be consistent with the type of start
1190         and end, e.g. 2 for numeric, or '5H' for datetime-like.  Default is 1
1191         for numeric and 'D' for datetime-like.
1192     name : str, default None
1193         Name of the resulting IntervalIndex.
1194     closed : {'left', 'right', 'both', 'neither'}, default 'right'
1195         Whether the intervals are closed on the left-side, right-side, both
1196         or neither.
1197 
1198     Returns
1199     -------
1200     IntervalIndex
1201 
1202     See Also
1203     --------
1204     IntervalIndex : An Index of intervals that are all closed on the same side.
1205 
1206     Notes
1207     -----
1208     Of the four parameters ``start``, ``end``, ``periods``, and ``freq``,
1209     exactly three must be specified. If ``freq`` is omitted, the resulting
1210     ``IntervalIndex`` will have ``periods`` linearly spaced elements between
1211     ``start`` and ``end``, inclusively.
1212 
1213     To learn more about datetime-like frequency strings, please see `this link
1214     <https://pandas.pydata.org/pandas-docs/stable/user_guide/timeseries.html#offset-aliases>`__.
1215 
1216     Examples
1217     --------
1218     Numeric ``start`` and  ``end`` is supported.
1219 
1220     >>> pd.interval_range(start=0, end=5)
1221     IntervalIndex([(0, 1], (1, 2], (2, 3], (3, 4], (4, 5]],
1222                   closed='right', dtype='interval[int64]')
1223 
1224     Additionally, datetime-like input is also supported.
1225 
1226     >>> pd.interval_range(start=pd.Timestamp('2017-01-01'),
1227     ...                   end=pd.Timestamp('2017-01-04'))
1228     IntervalIndex([(2017-01-01, 2017-01-02], (2017-01-02, 2017-01-03],
1229                    (2017-01-03, 2017-01-04]],
1230                   closed='right', dtype='interval[datetime64[ns]]')
1231 
1232     The ``freq`` parameter specifies the frequency between the left and right.
1233     endpoints of the individual intervals within the ``IntervalIndex``.  For
1234     numeric ``start`` and ``end``, the frequency must also be numeric.
1235 
1236     >>> pd.interval_range(start=0, periods=4, freq=1.5)
1237     IntervalIndex([(0.0, 1.5], (1.5, 3.0], (3.0, 4.5], (4.5, 6.0]],
1238                   closed='right', dtype='interval[float64]')
1239 
1240     Similarly, for datetime-like ``start`` and ``end``, the frequency must be
1241     convertible to a DateOffset.
1242 
1243     >>> pd.interval_range(start=pd.Timestamp('2017-01-01'),
1244     ...                   periods=3, freq='MS')
1245     IntervalIndex([(2017-01-01, 2017-02-01], (2017-02-01, 2017-03-01],
1246                    (2017-03-01, 2017-04-01]],
1247                   closed='right', dtype='interval[datetime64[ns]]')
1248 
1249     Specify ``start``, ``end``, and ``periods``; the frequency is generated
1250     automatically (linearly spaced).
1251 
1252     >>> pd.interval_range(start=0, end=6, periods=4)
1253     IntervalIndex([(0.0, 1.5], (1.5, 3.0], (3.0, 4.5], (4.5, 6.0]],
1254               closed='right',
1255               dtype='interval[float64]')
1256 
1257     The ``closed`` parameter specifies which endpoints of the individual
1258     intervals within the ``IntervalIndex`` are closed.
1259 
1260     >>> pd.interval_range(end=5, periods=4, closed='both')
1261     IntervalIndex([[1, 2], [2, 3], [3, 4], [4, 5]],
1262                   closed='both', dtype='interval[int64]')
1263     """
1264     start = com.maybe_box_datetimelike(start)
1265     end = com.maybe_box_datetimelike(end)
1266     endpoint = start if start is not None else end
1267 
1268     if freq is None and com.any_none(periods, start, end):
1269         freq = 1 if is_number(endpoint) else "D"
1270 
1271     if com.count_not_none(start, end, periods, freq) != 3:
1272         raise ValueError(
1273             "Of the four parameters: start, end, periods, and "
1274             "freq, exactly three must be specified"
1275         )
1276 
1277     if not _is_valid_endpoint(start):
1278         raise ValueError(f"start must be numeric or datetime-like, got {start}")
1279     elif not _is_valid_endpoint(end):
1280         raise ValueError(f"end must be numeric or datetime-like, got {end}")
1281 
1282     if is_float(periods):
1283         periods = int(periods)
1284     elif not is_integer(periods) and periods is not None:
1285         raise TypeError(f"periods must be a number, got {periods}")
1286 
1287     if freq is not None and not is_number(freq):
1288         try:
1289             freq = to_offset(freq)
1290         except ValueError as err:
1291             raise ValueError(
1292                 f"freq must be numeric or convertible to DateOffset, got {freq}"
1293             ) from err
1294 
1295     # verify type compatibility
1296     if not all(
1297         [
1298             _is_type_compatible(start, end),
1299             _is_type_compatible(start, freq),
1300             _is_type_compatible(end, freq),
1301         ]
1302     ):
1303         raise TypeError("start, end, freq need to be type compatible")
1304 
1305     # +1 to convert interval count to breaks count (n breaks = n-1 intervals)
1306     if periods is not None:
1307         periods += 1
1308 
1309     if is_number(endpoint):
1310         # force consistency between start/end/freq (lower end if freq skips it)
