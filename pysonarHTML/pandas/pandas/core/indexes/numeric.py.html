  41 class NumericIndex(Index):
  42     """
  43     Provide numeric type operations.
  44 
  45     This is an abstract class.
  46     """
  47 
  48     _is_numeric_dtype = True
  49 
  50     def __new__(cls, data=None, dtype=None, copy=False, name=None):
  51         cls._validate_dtype(dtype)
  52         name = maybe_extract_name(name, data, cls)
  53 
  54         # Coerce to ndarray if not already ndarray or Index
  55         if not isinstance(data, (np.ndarray, Index)):
  56             if is_scalar(data):
  57                 raise cls._scalar_data_error(data)
  58 
  59             # other iterable of some kind
  60             if not isinstance(data, (ABCSeries, list, tuple)):
  61                 data = list(data)
  62 
  63             data = np.asarray(data, dtype=dtype)
  64 
  65         if issubclass(data.dtype.type, str):
  66             cls._string_data_error(data)
  67 
  68         if copy or not is_dtype_equal(data.dtype, cls._default_dtype):
  69             subarr = np.array(data, dtype=cls._default_dtype, copy=copy)
  70             cls._assert_safe_casting(data, subarr)
  71         else:
  72             subarr = data
  73 
  74         if subarr.ndim > 1:
  75             # GH#13601, GH#20285, GH#27125
  76             raise ValueError("Index data must be 1-dimensional")
  77 
  78         subarr = np.asarray(subarr)
  79         return cls._simple_new(subarr, name=name)
  80 
  81     @classmethod
  82     def _validate_dtype(cls, dtype: Dtype) -> None:
  83         if dtype is None:
  84             return
  85         validation_metadata = {
  86             "int64index": (is_signed_integer_dtype, "signed integer"),
  87             "uint64index": (is_unsigned_integer_dtype, "unsigned integer"),
  88             "float64index": (is_float_dtype, "float"),
  89             "rangeindex": (is_signed_integer_dtype, "signed integer"),
  90         }
  91 
  92         validation_func, expected = validation_metadata[cls._typ]
  93         if not validation_func(dtype):
  94             raise ValueError(
  95                 f"Incorrect `dtype` passed: expected {expected}, received {dtype}"
  96             )
  97 
  98     @doc(Index._maybe_cast_slice_bound)
  99     def _maybe_cast_slice_bound(self, label, side, kind):
 100         assert kind in ["loc", "getitem", None]
 101 
 102         # we will try to coerce to integers
 103         return self._maybe_cast_indexer(label)
 104 
 105     @doc(Index._shallow_copy)
 106     def _shallow_copy(self, values=None, name: Label = lib.no_default):
 107         if values is not None and not self._can_hold_na and values.dtype.kind == "f":
 108             name = self.name if name is lib.no_default else name
 109             # Ensure we are not returning an Int64Index with float data:
 110             return Float64Index._simple_new(values, name=name)
 111         return super()._shallow_copy(values=values, name=name)
 112 
 113     def _convert_for_op(self, value):
 114         """
 115         Convert value to be insertable to ndarray.
 116         """
 117         if is_bool(value) or is_bool_dtype(value):
 118             # force conversion to object
 119             # so we don't lose the bools
 120             raise TypeError
 121 
 122         return value
 123 
 124     def _convert_tolerance(self, tolerance, target):
 125         tolerance = np.asarray(tolerance)
 126         if target.size != tolerance.size and tolerance.size > 1:
 127             raise ValueError("list-like tolerance size must match target index size")
 128         if not np.issubdtype(tolerance.dtype, np.number):
 129             if tolerance.ndim > 0:
 130                 raise ValueError(
 131                     f"tolerance argument for {type(self).__name__} must contain "
 132                     "numeric elements if it is list type"
 133                 )
 134             else:
 135                 raise ValueError(
 136                     f"tolerance argument for {type(self).__name__} must be numeric "
 137                     f"if it is a scalar: {repr(tolerance)}"
 138                 )
 139         return tolerance
 140 
 141     @classmethod
 142     def _assert_safe_casting(cls, data, subarr):
 143         """
 144         Subclasses need to override this only if the process of casting data
 145         from some accepted dtype to the internal dtype(s) bears the risk of
 146         truncation (e.g. float to int).
 147         """
 148         pass
 149 
 150     @property
 151     def is_all_dates(self) -> bool:
 152         """
 153         Checks that all the labels are datetime objects.
 154         """
 155         return False
 156 
 157     @doc(Index.insert)
 158     def insert(self, loc: int, item):
 159         # treat NA values as nans:
 160         if is_scalar(item) and isna(item):
 161             item = self._na_value
 162         return super().insert(loc, item)
 163 
 164     def _union(self, other, sort):
 165         # Right now, we treat union(int, float) a bit special.
 166         # See https://github.com/pandas-dev/pandas/issues/26778 for discussion
 167         # We may change union(int, float) to go to object.
 168         # float | [u]int -> float  (the special case)
 169         # <T>   | <T>    -> T
 170         # <T>   | <U>    -> object
 171         needs_cast = (is_integer_dtype(self.dtype) and is_float_dtype(other.dtype)) or (
 172             is_integer_dtype(other.dtype) and is_float_dtype(self.dtype)
 173         )
 174         if needs_cast:
 175             first = self.astype("float")
 176             second = other.astype("float")
 177             return first._union(second, sort)
 178         else:
 179             return super()._union(other, sort)

 218 class IntegerIndex(NumericIndex):
 219     """
 220     This is an abstract class for Int64Index, UInt64Index.
 221     """
 222 
 223     _default_dtype: np.dtype
 224 
 225     def __contains__(self, key) -> bool:
 226         """
 227         Check if key is a float and has a decimal. If it has, return False.
 228         """
 229         hash(key)
 230         try:
 231             if is_float(key) and int(key) != key:
 232                 return False
 233             return key in self._engine
 234         except (OverflowError, TypeError, ValueError):
 235             return False
 236 
 237     @property
 238     def inferred_type(self) -> str:
 239         """
 240         Always 'integer' for ``Int64Index`` and ``UInt64Index``
 241         """
 242         return "integer"
 243 
 244     @property
 245     def asi8(self) -> np.ndarray:
 246         # do not cache or you'll create a memory leak
 247         return self._values.view(self._default_dtype)

 294     @doc(Index._convert_arr_indexer)
 295     def _convert_arr_indexer(self, keyarr):
 296         # Cast the indexer to uint64 if possible so that the values returned
 297         # from indexing are also uint64.
 298         dtype = None
 299         if is_integer_dtype(keyarr) or (
 300             lib.infer_dtype(keyarr, skipna=False) == "integer"
 301         ):
 302             dtype = np.uint64
 303 
 304         return com.asarray_tuplesafe(keyarr, dtype=dtype)

 306     @doc(Index._convert_index_indexer)
 307     def _convert_index_indexer(self, keyarr):
 308         # Cast the indexer to uint64 if possible so
 309         # that the values returned from indexing are
 310         # also uint64.
 311         if keyarr.is_integer():
 312             return keyarr.astype(np.uint64)
 313         return keyarr

 342 class Float64Index(NumericIndex):
 343     __doc__ = _num_index_shared_docs["class_descr"] % _float64_descr_args
 344 
 345     _typ = "float64index"
 346     _engine_type = libindex.Float64Engine
 347     _default_dtype = np.float64
 348 
 349     @property
 350     def inferred_type(self) -> str:
 351         """
 352         Always 'floating' for ``Float64Index``
 353         """
 354         return "floating"
 355 
 356     @doc(Index.astype)
 357     def astype(self, dtype, copy=True):
 358         dtype = pandas_dtype(dtype)
 359         if needs_i8_conversion(dtype):
 360             raise TypeError(
 361                 f"Cannot convert Float64Index to dtype {dtype}; integer "
 362                 "values are required for conversion"
 363             )
 364         elif is_integer_dtype(dtype) and not is_extension_array_dtype(dtype):
 365             # TODO(jreback); this can change once we have an EA Index type
 366             # GH 13149
 367             arr = astype_nansafe(self._values, dtype=dtype)
 368             return Int64Index(arr, name=self.name)
 369         return super().astype(dtype, copy=copy)

 389         self, na_rep="", float_format=None, decimal=".", quoting=None, **kwargs

 403     def equals(self, other) -> bool:
 404         """
 405         Determines if two Index objects contain the same elements.
 406         """
 407         if self is other:
 408             return True
 409 
 410         if not isinstance(other, Index):
 411             return False
 412 
 413         # need to compare nans locations and make sure that they are the same
 414         # since nans don't compare equal this is a bit tricky
 415         try:
 416             if not isinstance(other, Float64Index):
 417                 other = self._constructor(other)
 418             if not is_dtype_equal(self.dtype, other.dtype) or self.shape != other.shape:
 419                 return False
 420             left, right = self._values, other._values
 421             return ((left == right) | (self._isnan & other._isnan)).all()
 422         except (TypeError, ValueError):
 423             return False
 424 
 425     def __contains__(self, other: Any) -> bool:
 426         hash(other)
 427         if super().__contains__(other):
 428             return True
 429 
 430         return is_float(other) and np.isnan(other) and self.hasnans
 431 
 432     @doc(Index.get_loc)
 433     def get_loc(self, key, method=None, tolerance=None):
 434         if is_bool(key):
 435             # Catch this to avoid accidentally casting to 1.0
 436             raise KeyError(key)
 437 
 438         if is_float(key) and np.isnan(key):
 439             nan_idxs = self._nan_idxs
 440             if not len(nan_idxs):
 441                 raise KeyError(key)
 442             elif len(nan_idxs) == 1:
 443                 return nan_idxs[0]
 444             return nan_idxs
 445 
 446         return super().get_loc(key, method=method, tolerance=tolerance)
 447 
 448     @cache_readonly
