  71 class CategoricalIndex(ExtensionIndex, accessor.PandasDelegate):
  72     """
  73     Index based on an underlying :class:`Categorical`.
  74 
  75     CategoricalIndex, like Categorical, can only take on a limited,
  76     and usually fixed, number of possible values (`categories`). Also,
  77     like Categorical, it might have an order, but numerical operations
  78     (additions, divisions, ...) are not possible.
  79 
  80     Parameters
  81     ----------
  82     data : array-like (1-dimensional)
  83         The values of the categorical. If `categories` are given, values not in
  84         `categories` will be replaced with NaN.
  85     categories : index-like, optional
  86         The categories for the categorical. Items need to be unique.
  87         If the categories are not given here (and also not in `dtype`), they
  88         will be inferred from the `data`.
  89     ordered : bool, optional
  90         Whether or not this categorical is treated as an ordered
  91         categorical. If not given here or in `dtype`, the resulting
  92         categorical will be unordered.
  93     dtype : CategoricalDtype or "category", optional
  94         If :class:`CategoricalDtype`, cannot be used together with
  95         `categories` or `ordered`.
  96     copy : bool, default False
  97         Make a copy of input ndarray.
  98     name : object, optional
  99         Name to be stored in the index.
 100 
 101     Attributes
 102     ----------
 103     codes
 104     categories
 105     ordered
 106 
 107     Methods
 108     -------
 109     rename_categories
 110     reorder_categories
 111     add_categories
 112     remove_categories
 113     remove_unused_categories
 114     set_categories
 115     as_ordered
 116     as_unordered
 117     map
 118 
 119     Raises
 120     ------
 121     ValueError
 122         If the categories do not validate.
 123     TypeError
 124         If an explicit ``ordered=True`` is given but no `categories` and the
 125         `values` are not sortable.
 126 
 127     See Also
 128     --------
 129     Index : The base pandas Index type.
 130     Categorical : A categorical array.
 131     CategoricalDtype : Type for categorical data.
 132 
 133     Notes
 134     -----
 135     See the `user guide
 136     <https://pandas.pydata.org/pandas-docs/stable/user_guide/advanced.html#categoricalindex>`_
 137     for more.
 138 
 139     Examples
 140     --------
 141     >>> pd.CategoricalIndex(["a", "b", "c", "a", "b", "c"])
 142     CategoricalIndex(['a', 'b', 'c', 'a', 'b', 'c'],
 143                      categories=['a', 'b', 'c'], ordered=False, dtype='category')
 144 
 145     ``CategoricalIndex`` can also be instantiated from a ``Categorical``:
 146 
 147     >>> c = pd.Categorical(["a", "b", "c", "a", "b", "c"])
 148     >>> pd.CategoricalIndex(c)
 149     CategoricalIndex(['a', 'b', 'c', 'a', 'b', 'c'],
 150                      categories=['a', 'b', 'c'], ordered=False, dtype='category')
 151 
 152     Ordered ``CategoricalIndex`` can have a min and max value.
 153 
 154     >>> ci = pd.CategoricalIndex(
 155     ...     ["a", "b", "c", "a", "b", "c"], ordered=True, categories=["c", "b", "a"]
 156     ... )
 157     >>> ci
 158     CategoricalIndex(['a', 'b', 'c', 'a', 'b', 'c'],
 159                      categories=['c', 'b', 'a'], ordered=True, dtype='category')
 160     >>> ci.min()
 161     'c'

 186     def __new__(
 187         cls, data=None, categories=None, ordered=None, dtype=None, copy=False, name=None
 188     ):
 189 
 190         dtype = CategoricalDtype._from_values_or_dtype(data, categories, ordered, dtype)
 191 
 192         name = maybe_extract_name(name, data, cls)
 193 
 194         if not is_categorical_dtype(data):
 195             # don't allow scalars
 196             # if data is None, then categories must be provided
 197             if is_scalar(data):
 198                 if data is not None or categories is None:
 199                     raise cls._scalar_data_error(data)
 200                 data = []
 201 
 202         assert isinstance(dtype, CategoricalDtype), dtype
 203         data = extract_array(data, extract_numpy=True)
 204 
 205         if not isinstance(data, Categorical):
 206             data = Categorical(data, dtype=dtype)
 207         elif isinstance(dtype, CategoricalDtype) and dtype != data.dtype:
 208             # we want to silently ignore dtype='category'
 209             data = data._set_dtype(dtype)
 210 
 211         data = data.copy() if copy else data
 212 
 213         return cls._simple_new(data, name=name)

 215     def _create_from_codes(self, codes, dtype=None, name=None):
 216         """
 217         *this is an internal non-public method*
 218 
 219         create the correct categorical from codes
 220 
 221         Parameters
 222         ----------
 223         codes : new codes
 224         dtype: CategoricalDtype, defaults to existing
 225         name : optional name attribute, defaults to existing
 226 
 227         Returns
 228         -------
 229         CategoricalIndex
 230         """
 231         if dtype is None:
 232             dtype = self.dtype
 233         if name is None:
 234             name = self.name
 235         cat = Categorical.from_codes(codes, dtype=dtype)
 236         return CategoricalIndex(cat, name=name)

 262     def _is_dtype_compat(self, other) -> bool:
 263         """
 264         *this is an internal non-public method*
 265 
 266         provide a comparison between the dtype of self and other (coercing if
 267         needed)
 268 
 269         Raises
 270         ------
 271         TypeError if the dtypes are not compatible
 272         """
 273         if is_categorical_dtype(other):
 274             if isinstance(other, CategoricalIndex):
 275                 other = other._values
 276             if not other.is_dtype_equal(self):
 277                 raise TypeError(
 278                     "categories must match existing categories when appending"
 279                 )
 280         else:
 281             values = other
 282             if not is_list_like(values):
 283                 values = [values]
 284             cat = Categorical(other, dtype=self.dtype)
 285             other = CategoricalIndex(cat)
 286             if not other.isin(values).all():
 287                 raise TypeError(
 288                     "cannot append a non-category item to a CategoricalIndex"
 289                 )
 290 
 291         return other

 293     def equals(self, other) -> bool:
 294         """
 295         Determine if two CategoricalIndex objects contain the same elements.
 296 
 297         Returns
 298         -------
 299         bool
 300             If two CategoricalIndex objects have equal elements True,
 301             otherwise False.
 302         """
 303         if self.is_(other):
 304             return True
 305 
 306         if not isinstance(other, Index):
 307             return False
 308 
 309         try:
 310             other = self._is_dtype_compat(other)
 311             if isinstance(other, type(self)):
 312                 other = other._data
 313             return self._data.equals(other)
 314         except (TypeError, ValueError):
 315             pass
 316 
 317         return False

 444     def reindex(self, target, method=None, level=None, limit=None, tolerance=None):
 445         """
 446         Create index with target's values (move/add/delete values as necessary)
 447 
 448         Returns
 449         -------
 450         new_index : pd.Index
 451             Resulting index
 452         indexer : np.ndarray or None
 453             Indices of output values in original index
 454 
 455         """
 456         if method is not None:
 457             raise NotImplementedError(
 458                 "argument method is not implemented for CategoricalIndex.reindex"
 459             )
 460         if level is not None:
 461             raise NotImplementedError(
 462                 "argument level is not implemented for CategoricalIndex.reindex"
 463             )
 464         if limit is not None:
 465             raise NotImplementedError(
 466                 "argument limit is not implemented for CategoricalIndex.reindex"
 467             )
 468 
 469         target = ibase.ensure_index(target)
 470 
 471         missing: List[int]
 472         if self.equals(target):
 473             indexer = None
 474             missing = []
 475         else:
 476             indexer, missing = self.get_indexer_non_unique(np.array(target))
 477 
 478         if len(self.codes) and indexer is not None:
 479             new_target = self.take(indexer)
 480         else:
 481             new_target = target
 482 
 483         # filling in missing if needed
 484         if len(missing):
 485             cats = self.categories.get_indexer(target)
 486 
 487             if (cats == -1).any():
 488                 # coerce to a regular index here!
 489                 result = Index(np.array(self), name=self.name)
 490                 new_target, indexer, _ = result._reindex_non_unique(np.array(target))
 491             else:
 492 
 493                 codes = new_target.codes.copy()
 494                 codes[indexer == -1] = cats[missing]
 495                 new_target = self._create_from_codes(codes)
 496 
 497         # we always want to return an Index type here
 498         # to be consistent with .reindex for other index types (e.g. they don't
 499         # coerce based on the actual values, only on the dtype)
 500         # unless we had an initial Categorical to begin with
 501         # in which case we are going to conform to the passed Categorical
 502         new_target = np.asarray(new_target)
 503         if is_categorical_dtype(target):
 504             new_target = target._shallow_copy(new_target, name=self.name)
 505         else:
 506             new_target = Index(new_target, name=self.name)
 507 
 508         return new_target, indexer

 510     def _reindex_non_unique(self, target):
 511         """
 512         reindex from a non-unique; which CategoricalIndex's are almost
 513         always
 514         """
 515         new_target, indexer = self.reindex(target)
 516         new_indexer = None
 517 
 518         check = indexer == -1
 519         if check.any():
 520             new_indexer = np.arange(len(self.take(indexer)))
 521             new_indexer[check] = -1
 522 
 523         cats = self.categories.get_indexer(target)
 524         if not (cats == -1).any():
 525             # .reindex returns normal Index. Revert to CategoricalIndex if
 526             # all targets are included in my categories
 527             new_target = self._shallow_copy(new_target)
 528 
 529         return new_target, indexer, new_indexer

 531     @Appender(_index_shared_docs["get_indexer"] % _index_doc_kwargs)
 532     def get_indexer(self, target, method=None, limit=None, tolerance=None):
 533         method = missing.clean_reindex_fill_method(method)
 534         target = ibase.ensure_index(target)
 535 
 536         if self.is_unique and self.equals(target):
 537             return np.arange(len(self), dtype="intp")
 538 
 539         if method == "pad" or method == "backfill":
 540             raise NotImplementedError(
 541                 "method='pad' and method='backfill' not "
 542                 "implemented yet for CategoricalIndex"
 543             )
 544         elif method == "nearest":
 545             raise NotImplementedError(
 546                 "method='nearest' not implemented yet for CategoricalIndex"
 547             )
 548 
 549         if isinstance(target, CategoricalIndex) and self._values.is_dtype_equal(target):
 550             if self._values.equals(target._values):
 551                 # we have the same codes
 552                 codes = target.codes
 553             else:
 554                 codes = recode_for_categories(
 555                     target.codes, target.categories, self._values.categories
 556                 )
 557         else:
 558             if isinstance(target, CategoricalIndex):
 559                 code_indexer = self.categories.get_indexer(target.categories)
 560                 codes = take_1d(code_indexer, target.codes, fill_value=-1)
 561             else:
 562                 codes = self.categories.get_indexer(target)
 563 
 564         indexer, _ = self._engine.get_indexer_non_unique(codes)
 565         return ensure_platform_int(indexer)

 567     @Appender(_index_shared_docs["get_indexer_non_unique"] % _index_doc_kwargs)
 568     def get_indexer_non_unique(self, target):
 569         target = ibase.ensure_index(target)
 570 
 571         if isinstance(target, CategoricalIndex):
 572             # Indexing on codes is more efficient if categories are the same:
 573             if target.categories is self.categories:
 574                 target = target.codes
 575                 indexer, missing = self._engine.get_indexer_non_unique(target)
 576                 return ensure_platform_int(indexer), missing
 577             target = target._values
 578 
 579         codes = self.categories.get_indexer(target)
 580         indexer, missing = self._engine.get_indexer_non_unique(codes)
 581         return ensure_platform_int(indexer), missing

 629     def map(self, mapper):
 630         """
 631         Map values using input correspondence (a dict, Series, or function).
 632 
 633         Maps the values (their categories, not the codes) of the index to new
 634         categories. If the mapping correspondence is one-to-one the result is a
 635         :class:`~pandas.CategoricalIndex` which has the same order property as
 636         the original, otherwise an :class:`~pandas.Index` is returned.
 637 
 638         If a `dict` or :class:`~pandas.Series` is used any unmapped category is
 639         mapped to `NaN`. Note that if this happens an :class:`~pandas.Index`
 640         will be returned.
 641 
 642         Parameters
 643         ----------
 644         mapper : function, dict, or Series
 645             Mapping correspondence.
 646 
 647         Returns
 648         -------
 649         pandas.CategoricalIndex or pandas.Index
 650             Mapped index.
 651 
 652         See Also
 653         --------
 654         Index.map : Apply a mapping correspondence on an
 655             :class:`~pandas.Index`.
 656         Series.map : Apply a mapping correspondence on a
 657             :class:`~pandas.Series`.
 658         Series.apply : Apply more complex functions on a
 659             :class:`~pandas.Series`.
 660 
 661         Examples
 662         --------
 663         >>> idx = pd.CategoricalIndex(['a', 'b', 'c'])
 664         >>> idx
 665         CategoricalIndex(['a', 'b', 'c'], categories=['a', 'b', 'c'],
 666                           ordered=False, dtype='category')
 667         >>> idx.map(lambda x: x.upper())
 668         CategoricalIndex(['A', 'B', 'C'], categories=['A', 'B', 'C'],
 669                          ordered=False, dtype='category')
 670         >>> idx.map({'a': 'first', 'b': 'second', 'c': 'third'})
 671         CategoricalIndex(['first', 'second', 'third'], categories=['first',
 672                          'second', 'third'], ordered=False, dtype='category')
 673 
 674         If the mapping is one-to-one the ordering of the categories is
 675         preserved:
 676 
 677         >>> idx = pd.CategoricalIndex(['a', 'b', 'c'], ordered=True)
 678         >>> idx
 679         CategoricalIndex(['a', 'b', 'c'], categories=['a', 'b', 'c'],
 680                          ordered=True, dtype='category')
 681         >>> idx.map({'a': 3, 'b': 2, 'c': 1})
 682         CategoricalIndex([3, 2, 1], categories=[3, 2, 1], ordered=True,
 683                          dtype='category')
 684 
 685         If the mapping is not one-to-one an :class:`~pandas.Index` is returned:
 686 
 687         >>> idx.map({'a': 'first', 'b': 'second', 'c': 'first'})
 688         Index(['first', 'second', 'first'], dtype='object')
 689 
 690         If a `dict` is used, all unmapped categories are mapped to `NaN` and
 691         the result is an :class:`~pandas.Index`:
 692 
 693         >>> idx.map({'a': 'first', 'b': 'second'})
 694         Index(['first', 'second', nan], dtype='object')
 695         """
 696         mapped = self._values.map(mapper)
 697         return Index(mapped, name=self.name)
 698 
 699     def delete(self, loc):
 700         """
 701         Make new Index with passed location(-s) deleted
 702 
 703         Returns
 704         -------
 705         new_index : Index
 706         """
 707         return self._create_from_codes(np.delete(self.codes, loc))
 708 
 709     def insert(self, loc: int, item):
 710         """
 711         Make new Index inserting new item at location. Follows
 712         Python list.append semantics for negative values
 713 
 714         Parameters
 715         ----------
 716         loc : int
 717         item : object
 718 
 719         Returns
 720         -------
 721         new_index : Index
 722 
 723         Raises
 724         ------
 725         ValueError if the item is not in the categories
 726 
 727         """
 728         code = self.categories.get_indexer([item])
 729         if (code == -1) and not (is_scalar(item) and isna(item)):
 730             raise TypeError(
 731                 "cannot insert an item into a CategoricalIndex "
 732                 "that is not already an existing category"
 733             )
 734 
 735         codes = self.codes
 736         codes = np.concatenate((codes[:loc], code, codes[loc:]))
 737         return self._create_from_codes(codes)
 738 
 739     def _concat(self, to_concat, name):
 740         # if calling index is category, don't check dtype of others
 741         codes = np.concatenate([self._is_dtype_compat(c).codes for c in to_concat])
 742         result = self._create_from_codes(codes, name=name)
 743         # if name is None, _create_from_codes sets self.name
 744         result.name = name
 745         return result
 746 
 747     def _delegate_method(self, name: str, *args, **kwargs):
 748         """ method delegation to the ._values """
 749         method = getattr(self._values, name)
 750         if "inplace" in kwargs:
 751             raise ValueError("cannot use inplace with CategoricalIndex")
 752         res = method(*args, **kwargs)
 753         if is_scalar(res):
 754             return res
 755         return CategoricalIndex(res, name=self.name)
 756 
 757     def _wrap_joined_index(
 758         self, joined: np.ndarray, other: "CategoricalIndex"
 759     ) -> "CategoricalIndex":
 760         name = get_op_result_name(self, other)
 761         return self._create_from_codes(joined, name=name)
