  73 class MultiIndexUIntEngine(libindex.BaseMultiIndexCodesEngine, libindex.UInt64Engine):
  74     """
  75     This class manages a MultiIndex by mapping label combinations to positive
  76     integers.
  77     """
  78 
  79     _base = libindex.UInt64Engine

  81     def _codes_to_ints(self, codes):
  82         """
  83         Transform combination(s) of uint64 in one uint64 (each), in a strictly
  84         monotonic way (i.e. respecting the lexicographic order of integer
  85         combinations): see BaseMultiIndexCodesEngine documentation.
  86 
  87         Parameters
  88         ----------
  89         codes : 1- or 2-dimensional array of dtype uint64
  90             Combinations of integers (one per row)
  91 
  92         Returns
  93         -------
  94         scalar or 1-dimensional array, of dtype uint64
  95             Integer(s) representing one combination (each).
  96         """
  97         # Shift the representation of each level by the pre-calculated number
  98         # of bits:
  99         codes <<= self.offsets
 100 
 101         # Now sum and OR are in fact interchangeable. This is a simple
 102         # composition of the (disjunct) significant bits of each level (i.e.
 103         # each column in "codes") in a single positive integer:
 104         if codes.ndim == 1:
 105             # Single key
 106             return np.bitwise_or.reduce(codes)
 107 
 108         # Multiple keys
 109         return np.bitwise_or.reduce(codes, axis=1)

 121     def _codes_to_ints(self, codes):
 122         """
 123         Transform combination(s) of uint64 in one Python integer (each), in a
 124         strictly monotonic way (i.e. respecting the lexicographic order of
 125         integer combinations): see BaseMultiIndexCodesEngine documentation.
 126 
 127         Parameters
 128         ----------
 129         codes : 1- or 2-dimensional array of dtype uint64
 130             Combinations of integers (one per row)
 131 
 132         Returns
 133         -------
 134         int, or 1-dimensional array of dtype object
 135             Integer(s) representing one combination (each).
 136         """
 137         # Shift the representation of each level by the pre-calculated number
 138         # of bits. Since this can overflow uint64, first make sure we are
 139         # working with Python integers:
 140         codes = codes.astype("object") << self.offsets
 141 
 142         # Now sum and OR are in fact interchangeable. This is a simple
 143         # composition of the (disjunct) significant bits of each level (i.e.
 144         # each column in "codes") in a single positive integer (per row):
 145         if codes.ndim == 1:
 146             # Single key
 147             return np.bitwise_or.reduce(codes)
 148 
 149         # Multiple keys
 150         return np.bitwise_or.reduce(codes, axis=1)

 299     def _validate_codes(self, level: List, code: List):
 300         """
 301         Reassign code values as -1 if their corresponding levels are NaN.
 302 
 303         Parameters
 304         ----------
 305         code : list
 306             Code to reassign.
 307         level : list
 308             Level to check for missing values (NaN, NaT, None).
 309 
 310         Returns
 311         -------
 312         new code where code value = -1 if it corresponds
 313         to a level with missing values (NaN, NaT, None).
 314         """
 315         null_mask = isna(level)
 316         if np.any(null_mask):
 317             code = np.where(null_mask[code], -1, code)
 318         return code

 320     def _verify_integrity(
 321         self, codes: Optional[List] = None, levels: Optional[List] = None
 322     ):
 323         """
 324         Parameters
 325         ----------
 326         codes : optional list
 327             Codes to check for validity. Defaults to current codes.
 328         levels : optional list
 329             Levels to check for validity. Defaults to current levels.
 330 
 331         Raises
 332         ------
 333         ValueError
 334             If length of levels and codes don't match, if the codes for any
 335             level would exceed level bounds, or there are any duplicate levels.
 336 
 337         Returns
 338         -------
 339         new codes where code value = -1 if it corresponds to a
 340         NaN level.
 341         """
 342         # NOTE: Currently does not check, among other things, that cached
 343         # nlevels matches nor that sortorder matches actually sortorder.
 344         codes = codes or self.codes
 345         levels = levels or self.levels
 346 
 347         if len(levels) != len(codes):
 348             raise ValueError(
 349                 "Length of levels and codes must match. NOTE: "
 350                 "this index is in an inconsistent state."
 351             )
 352         codes_length = len(codes[0])
 353         for i, (level, level_codes) in enumerate(zip(levels, codes)):
 354             if len(level_codes) != codes_length:
 355                 raise ValueError(
 356                     f"Unequal code lengths: {[len(code_) for code_ in codes]}"
 357                 )
 358             if len(level_codes) and level_codes.max() >= len(level):
 359                 raise ValueError(
 360                     f"On level {i}, code max ({level_codes.max()}) >= length of "
 361                     f"level ({len(level)}). NOTE: this index is in an "
 362                     "inconsistent state"
 363                 )
 364             if len(level_codes) and level_codes.min() < -1:
 365                 raise ValueError(f"On level {i}, code value ({level_codes.min()}) < -1")
 366             if not level.is_unique:
 367                 raise ValueError(
 368                     f"Level values must be unique: {list(level)} on level {i}"
 369                 )
 370         if self.sortorder is not None:
 371             if self.sortorder > self._lexsort_depth():
 372                 raise ValueError(
 373                     "Value for sortorder must be inferior or equal to actual "
 374                     f"lexsort_depth: sortorder {self.sortorder} "
 375                     f"with lexsort_depth {self._lexsort_depth()}"
 376                 )
 377 
 378         codes = [
 379             self._validate_codes(level, code) for level, code in zip(levels, codes)
 380         ]
 381         new_codes = FrozenList(codes)
 382         return new_codes

 385     def from_arrays(cls, arrays, sortorder=None, names=lib.no_default) -> "MultiIndex":
 386         """
 387         Convert arrays to MultiIndex.
 388 
 389         Parameters
 390         ----------
 391         arrays : list / sequence of array-likes
 392             Each array-like gives one level's value for each data point.
 393             len(arrays) is the number of levels.
 394         sortorder : int or None
 395             Level of sortedness (must be lexicographically sorted by that
 396             level).
 397         names : list / sequence of str, optional
 398             Names for the levels in the index.
 399 
 400         Returns
 401         -------
 402         MultiIndex
 403 
 404         See Also
 405         --------
 406         MultiIndex.from_tuples : Convert list of tuples to MultiIndex.
 407         MultiIndex.from_product : Make a MultiIndex from cartesian product
 408                                   of iterables.
 409         MultiIndex.from_frame : Make a MultiIndex from a DataFrame.
 410 
 411         Examples
 412         --------
 413         >>> arrays = [[1, 1, 2, 2], ['red', 'blue', 'red', 'blue']]
 414         >>> pd.MultiIndex.from_arrays(arrays, names=('number', 'color'))
 415         MultiIndex([(1,  'red'),
 416                     (1, 'blue'),
 417                     (2,  'red'),
 418                     (2, 'blue')],
 419                    names=['number', 'color'])
 420         """
 421         error_msg = "Input must be a list / sequence of array-likes."
 422         if not is_list_like(arrays):
 423             raise TypeError(error_msg)
 424         elif is_iterator(arrays):
 425             arrays = list(arrays)
 426 
 427         # Check if elements of array are list-like
 428         for array in arrays:
 429             if not is_list_like(array):
 430                 raise TypeError(error_msg)
 431 
 432         # Check if lengths of all arrays are equal or not,
 433         # raise ValueError, if not
 434         for i in range(1, len(arrays)):
 435             if len(arrays[i]) != len(arrays[i - 1]):
 436                 raise ValueError("all arrays must be same length")
 437 
 438         codes, levels = factorize_from_iterables(arrays)
 439         if names is lib.no_default:
 440             names = [getattr(arr, "name", None) for arr in arrays]
 441 
 442         return MultiIndex(
 443             levels=levels,
 444             codes=codes,
 445             sortorder=sortorder,
 446             names=names,
 447             verify_integrity=False,
 448         )

 982     @cache_readonly
 983     def _engine(self):
 984         # Calculate the number of bits needed to represent labels in each
 985         # level, as log2 of their sizes (including -1 for NaN):
 986         sizes = np.ceil(np.log2([len(l) + 1 for l in self.levels]))
 987 
 988         # Sum bit counts, starting from the _right_....
 989         lev_bits = np.cumsum(sizes[::-1])[::-1]
 990 
 991         # ... in order to obtain offsets such that sorting the combination of
 992         # shifted codes (one for each level, resulting in a unique integer) is
 993         # equivalent to sorting lexicographically the codes themselves. Notice
 994         # that each level needs to be shifted by the number of bits needed to
 995         # represent the _previous_ ones:
 996         offsets = np.concatenate([lev_bits[1:], [0]]).astype("uint64")
 997 
 998         # Check the total number of bits needed for our representation:
 999         if lev_bits[0] > 64:
1000             # The levels would overflow a 64 bit uint - use Python integers:
1001             return MultiIndexPyIntEngine(self.levels, self.codes, offsets)
1002         return MultiIndexUIntEngine(self.levels, self.codes, offsets)

1726     def _sort_levels_monotonic(self):
1727         """
1728         This is an *internal* function.
1729 
1730         Create a new MultiIndex from the current to monotonically sorted
1731         items IN the levels. This does not actually make the entire MultiIndex
1732         monotonic, JUST the levels.
1733 
1734         The resulting MultiIndex will have the same outward
1735         appearance, meaning the same .values and ordering. It will also
1736         be .equals() to the original.
1737 
1738         Returns
1739         -------
1740         MultiIndex
1741 
1742         Examples
1743         --------
1744         >>> mi = pd.MultiIndex(levels=[['a', 'b'], ['bb', 'aa']],
1745         ...                    codes=[[0, 0, 1, 1], [0, 1, 0, 1]])
1746         >>> mi
1747         MultiIndex([('a', 'bb'),
1748                     ('a', 'aa'),
1749                     ('b', 'bb'),
1750                     ('b', 'aa')],
1751                    )
1752 
1753         >>> mi.sort_values()
1754         MultiIndex([('a', 'aa'),
1755                     ('a', 'bb'),
1756                     ('b', 'aa'),
1757                     ('b', 'bb')],
1758                    )
1759         """
1760         if self.is_lexsorted() and self.is_monotonic:
1761             return self
1762 
1763         new_levels = []

1767 
1768             if not lev.is_monotonic:
1769                 try:
1770                     # indexer to reorder the levels
1771                     indexer = lev.argsort()
1772                 except TypeError:
1773                     pass
1774                 else:
1775                     lev = lev.take(indexer)
1776 
1777                     # indexer to reorder the level codes
1778                     indexer = ensure_int64(indexer)
1779                     ri = lib.get_reverse_indexer(indexer, len(indexer))
1780                     level_codes = algos.take_1d(ri, level_codes)
1781 
1782             new_levels.append(lev)
1783             new_codes.append(level_codes)
1784 
1785         return MultiIndex(

1787             new_codes,
1788             names=self.names,

1793     def remove_unused_levels(self):
1794         """
1795         Create new MultiIndex from current that removes unused levels.
1796 
1797         Unused level(s) means levels that are not expressed in the
1798         labels. The resulting MultiIndex will have the same outward
1799         appearance, meaning the same .values and ordering. It will
1800         also be .equals() to the original.
1801 
1802         Returns
1803         -------
1804         MultiIndex
1805 
1806         Examples
1807         --------
1808         >>> mi = pd.MultiIndex.from_product([range(2), list('ab')])
1809         >>> mi
1810         MultiIndex([(0, 'a'),
1811                     (0, 'b'),
1812                     (1, 'a'),
1813                     (1, 'b')],
1814                    )
1815 
1816         >>> mi[2:]
1817         MultiIndex([(1, 'a'),
1818                     (1, 'b')],
1819                    )
1820 
1821         The 0 from the first level is not represented
1822         and can be removed
1823 
1824         >>> mi2 = mi[2:].remove_unused_levels()
1825         >>> mi2.levels
1826         FrozenList([[1], ['a', 'b']])
1827         """
1828         new_levels = []
1829         new_codes = []
1830 
1831         changed = False
1832         for lev, level_codes in zip(self.levels, self.codes):
1833 
1834             # Since few levels are typically unused, bincount() is more
1835             # efficient than unique() - however it only accepts positive values
1836             # (and drops order):
1837             uniques = np.where(np.bincount(level_codes + 1) > 0)[0] - 1
1838             has_na = int(len(uniques) and (uniques[0] == -1))
1839 
1840             if len(uniques) != len(lev) + has_na:
1841                 # We have unused levels
1842                 changed = True
1843 
1844                 # Recalculate uniques, now preserving order.
1845                 # Can easily be cythonized by exploiting the already existing
1846                 # "uniques" and stop parsing "level_codes" when all items
1847                 # are found:
1848                 uniques = algos.unique(level_codes)
1849                 if has_na:
1850                     na_idx = np.where(uniques == -1)[0]
1851                     # Just ensure that -1 is in first position:
1852                     uniques[[0, na_idx[0]]] = uniques[[na_idx[0], 0]]
1853 
1854                 # codes get mapped from uniques to 0:len(uniques)
1855                 # -1 (if present) is mapped to last position
1856                 code_mapping = np.zeros(len(lev) + has_na)
1857                 # ... and reassigned value -1:
1858                 code_mapping[uniques] = np.arange(len(uniques)) - has_na
1859 
1860                 level_codes = code_mapping[level_codes]
1861 
1862                 # new levels are simple
1863                 lev = lev.take(uniques[has_na:])
1864 
1865             new_levels.append(lev)
1866             new_codes.append(level_codes)
1867 
1868         result = self.view()
1869 
1870         if changed:
1871             result._reset_identity()
1872             result._set_levels(new_levels, validate=False)
1873             result._set_codes(new_codes, validate=False)
1874 
1875         return result

2880     def _get_level_indexer(self, key, level=0, indexer=None):
2881         # return an indexer, boolean array or a slice showing where the key is
2882         # in the totality of values
2883         # if the indexer is provided, then use this
2884 
2885         level_index = self.levels[level]
2886         level_codes = self.codes[level]
2887 
2888         def convert_indexer(start, stop, step, indexer=indexer, codes=level_codes):
2889             # given the inputs and the codes/indexer, compute an indexer set
2890             # if we have a provided indexer, then this need not consider
2891             # the entire labels set
2892

2894             if indexer is not None and len(indexer) != len(codes):
2895 
2896                 # we have an indexer which maps the locations in the labels
2897                 # that we have already selected (and is not an indexer for the
2898                 # entire set) otherwise this is wasteful so we only need to
2899                 # examine locations that are in this set the only magic here is
2900                 # that the result are the mappings to the set that we have
2901                 # selected
2902                 from pandas import Series
2903 
2904                 mapper = Series(indexer)
2905                 indexer = codes.take(ensure_platform_int(indexer))
2906                 result = Series(Index(indexer).isin(r).nonzero()[0])
2907                 m = result.map(mapper)
2908                 m = np.asarray(m)
2909 
2910             else:
2911                 m = np.zeros(len(codes), dtype=bool)
2912                 m[np.in1d(codes, r, assume_unique=Index(codes).is_unique)] = True
2913 
2914             return m
2915 
2916         if isinstance(key, slice):
2917             # handle a slice, returning a slice if we can
2918             # otherwise a boolean indexer
2919 
2920             try:
2921                 if key.start is not None:
2922                     start = level_index.get_loc(key.start)
2923                 else:
2924                     start = 0
2925                 if key.stop is not None:
2926                     stop = level_index.get_loc(key.stop)
2927                 else:
2928                     stop = len(level_index) - 1
2929                 step = key.step
2930             except KeyError:
2931 
2932                 # we have a partial slice (like looking up a partial date
2933                 # string)
2934                 start = stop = level_index.slice_indexer(
2935                     key.start, key.stop, key.step, kind="loc"
2936                 )
2937                 step = start.step
2938 
2939             if isinstance(start, slice) or isinstance(stop, slice):
2940                 # we have a slice for start and/or stop
2941                 # a partial date slicer on a DatetimeIndex generates a slice
2942                 # note that the stop ALREADY includes the stopped point (if
2943                 # it was a string sliced)
2944                 start = getattr(start, "start", start)
2945                 stop = getattr(stop, "stop", stop)
2946                 return convert_indexer(start, stop, step)
2947 
2948             elif level > 0 or self.lexsort_depth == 0 or step is not None:

2950                 # searching as when we are using a slice
2951                 # so include the stop+1 (so we include stop)
2952                 return convert_indexer(start, stop + 1, step)
2953             else:
2954                 # sorted, so can return slice object -> view
2955                 i = level_codes.searchsorted(start, side="left")
2956                 j = level_codes.searchsorted(stop, side="right")
2957                 return slice(i, j, step)
2958 
2959         else:
2960 
2961             code = self._get_loc_single_level_index(level_index, key)
2962 
2963             if level > 0 or self.lexsort_depth == 0:
2964                 # Desired level is not sorted
2965                 locs = np.array(level_codes == code, dtype=bool, copy=False)
2966                 if not locs.any():
2967                     # The label is present in self.levels[level] but unused:
2968                     raise KeyError(key)
2969                 return locs
2970 
2971             i = level_codes.searchsorted(code, side="left")
2972             j = level_codes.searchsorted(code, side="right")
2973             if i == j:
2974                 # The label is present in self.levels[level] but unused:
2975                 raise KeyError(key)

3080 
3081             elif isinstance(k, slice):
3082

3108     def _reorder_indexer(
3109         self,
3110         seq: Tuple[Union[Scalar, Iterable, AnyArrayLike], ...],
3111         indexer: Int64Index,
3112     ) -> Int64Index:
3113         """
3114         Reorder an indexer of a MultiIndex (self) so that the label are in the
3115         same order as given in seq
3116 
3117         Parameters
3118         ----------
3119         seq : label/slice/list/mask or a sequence of such
3120         indexer: an Int64Index indexer of self
3121 
3122         Returns
3123         -------
3124         indexer : a sorted Int64Index indexer of self ordered as seq
3125         """
3126         # If the index is lexsorted and the list_like label in seq are sorted
3127         # then we do not need to sort
3128         if self.is_lexsorted():
3129             need_sort = False
3130             for i, k in enumerate(seq):
3131                 if is_list_like(k):
3132                     if not need_sort:
3133                         k_codes = self.levels[i].get_indexer(k)
3134                         k_codes = k_codes[k_codes >= 0]  # Filter absent keys
3135                         # True if the given codes are not ordered
3136                         need_sort = (k_codes[:-1] > k_codes[1:]).any()
3137             # Bail out if both index and seq are sorted
3138             if not need_sort:
3139                 return indexer
3140 
3141         n = len(self)
3142         keys: Tuple[np.ndarray, ...] = tuple()
3143         # For each level of the sequence in seq, map the level codes with the
3144         # order they appears in a list-like sequence
3145         # This mapping is then use to reorder the indexer
3146         for i, k in enumerate(seq):
3147             if com.is_bool_indexer(k):
3148                 new_order = np.arange(n)[indexer]
3149             elif is_list_like(k):
3150                 # Generate a map with all level codes as sorted initially
3151                 key_order_map = np.ones(len(self.levels[i]), dtype=np.uint64) * len(
3152                     self.levels[i]
3153                 )
3154                 # Set order as given in the indexer list
3155                 level_indexer = self.levels[i].get_indexer(k)
3156                 level_indexer = level_indexer[level_indexer >= 0]  # Filter absent keys
3157                 key_order_map[level_indexer] = np.arange(len(level_indexer))
3158 
3159                 new_order = key_order_map[self.codes[i][indexer]]
3160             else:
3161                 # For all other case, use the same order as the level
3162                 new_order = np.arange(n)[indexer]
3163             keys = (new_order,) + keys
3164 
3165         # Find the reordering using lexsort on the keys mapping
3166         ind = np.lexsort(keys)
3167         return indexer[ind]
