 173     def to_pydatetime(self) -> np.ndarray:
 174         """
 175         Return the data as an array of native Python datetime objects.
 176 
 177         Timezone information is retained if present.
 178 
 179         .. warning::
 180 
 181            Python's datetime uses microsecond resolution, which is lower than
 182            pandas (nanosecond). The values are truncated.
 183 
 184         Returns
 185         -------
 186         numpy.ndarray
 187             Object dtype array containing native Python datetime objects.
 188 
 189         See Also
 190         --------
 191         datetime.datetime : Standard library value for a datetime.
 192 
 193         Examples
 194         --------
 195         >>> s = pd.Series(pd.date_range('20180310', periods=2))
 196         >>> s
 197         0   2018-03-10
 198         1   2018-03-11
 199         dtype: datetime64[ns]
 200 
 201         >>> s.dt.to_pydatetime()
 202         array([datetime.datetime(2018, 3, 10, 0, 0),
 203                datetime.datetime(2018, 3, 11, 0, 0)], dtype=object)
 204 
 205         pandas' nanosecond precision is truncated to microseconds.
 206 
 207         >>> s = pd.Series(pd.date_range('20180310', periods=2, freq='ns'))
 208         >>> s
 209         0   2018-03-10 00:00:00.000000000
 210         1   2018-03-10 00:00:00.000000001
 211         dtype: datetime64[ns]
 212 
 213         >>> s.dt.to_pydatetime()
 214         array([datetime.datetime(2018, 3, 10, 0, 0),
 215                datetime.datetime(2018, 3, 10, 0, 0)], dtype=object)
 216         """
 217         return self._get_values().to_pydatetime()

 448 class CombinedDatetimelikeProperties(
 449     DatetimeProperties, TimedeltaProperties, PeriodProperties
 450 ):
 451     def __new__(cls, data: "Series"):
 452         # CombinedDatetimelikeProperties isn't really instantiated. Instead
 453         # we need to choose which parent (datetime or timedelta) is
 454         # appropriate. Since we're checking the dtypes anyway, we'll just
 455         # do all the validation here.
 456 
 457         if not isinstance(data, ABCSeries):
 458             raise TypeError(
 459                 f"cannot convert an object of type {type(data)} to a datetimelike index"
 460             )
 461 
 462         orig = data if is_categorical_dtype(data.dtype) else None
 463         if orig is not None:
 464             data = data._constructor(
 465                 orig.array,
 466                 name=orig.name,
 467                 copy=False,
 468                 dtype=orig._values.categories.dtype,
 469             )
 470 
 471         if is_datetime64_dtype(data.dtype):
 472             return DatetimeProperties(data, orig)
 473         elif is_datetime64tz_dtype(data.dtype):
 474             return DatetimeProperties(data, orig)
 475         elif is_timedelta64_dtype(data.dtype):
 476             return TimedeltaProperties(data, orig)
 477         elif is_period_dtype(data.dtype):
 478             return PeriodProperties(data, orig)
 479 
 480         raise AttributeError("Can only use .dt accessor with datetimelike values")
