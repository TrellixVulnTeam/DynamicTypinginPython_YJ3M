  21 def inherit_from_data(name: str, delegate, cache: bool = False, wrap: bool = False):
  22     """
  23     Make an alias for a method of the underlying ExtensionArray.
  24 
  25     Parameters
  26     ----------
  27     name : str
  28         Name of an attribute the class should inherit from its EA parent.
  29     delegate : class
  30     cache : bool, default False
  31         Whether to convert wrapped properties into cache_readonly
  32     wrap : bool, default False
  33         Whether to wrap the inherited result in an Index.
  34 
  35     Returns
  36     -------
  37     attribute, method, property, or cache_readonly
  38     """
  39     attr = getattr(delegate, name)
  40 
  41     if isinstance(attr, property):
  42         if cache:
  43 
  44             def cached(self):
  45                 return getattr(self._data, name)
  46 
  47             cached.__name__ = name
  48             cached.__doc__ = attr.__doc__
  49             method = cache_readonly(cached)
  50 
  51         else:
  52 
  53             def fget(self):
  54                 result = getattr(self._data, name)
  55                 if wrap:
  56                     if isinstance(result, type(self._data)):
  57                         return type(self)._simple_new(result, name=self.name)
  58                     elif isinstance(result, ABCDataFrame):
  59                         return result.set_index(self)
  60                     return Index(result, name=self.name)
  61                 return result
  62 
  63             def fset(self, value):
  64                 setattr(self._data, name, value)
  65 
  66             fget.__name__ = name
  67             fget.__doc__ = attr.__doc__
  68 
  69             method = property(fget, fset)
  70 
  71     elif not callable(attr):
  72         # just a normal attribute, no wrapping
  73         method = attr
  74 
  75     else:
  76 
  77         def method(self, *args, **kwargs):
  78             result = attr(self._data, *args, **kwargs)
  79             if wrap:
  80                 if isinstance(result, type(self._data)):
  81                     return type(self)._simple_new(result, name=self.name)
  82                 elif isinstance(result, ABCDataFrame):
  83                     return result.set_index(self)
  84                 return Index(result, name=self.name)
  85             return result
  86 
  87         method.__name__ = name
  88         method.__doc__ = attr.__doc__
  89     return method

  92 def inherit_names(names: List[str], delegate, cache: bool = False, wrap: bool = False):
  93     """
  94     Class decorator to pin attributes from an ExtensionArray to a Index subclass.
  95 
  96     Parameters
  97     ----------
  98     names : List[str]
  99     delegate : class
 100     cache : bool, default False
 101     wrap : bool, default False
 102         Whether to wrap the inherited result in an Index.
 103     """
 104 
 105     def wrapper(cls):
 106         for name in names:
 107             meth = inherit_from_data(name, delegate, cache=cache, wrap=wrap)
 108             setattr(cls, name, meth)
 109 
 110         return cls
 111 
 112     return wrapper

 115 def _make_wrapped_comparison_op(opname: str):
 116     """
 117     Create a comparison method that dispatches to ``._data``.
 118     """
 119 
 120     def wrapper(self, other):
 121         if isinstance(other, ABCSeries):
 122             # the arrays defer to Series for comparison ops but the indexes
 123             #  don't, so we have to unwrap here.
 124             other = other._values
 125 
 126         other = _maybe_unwrap_index(other)
 127 
 128         op = getattr(self._data, opname)
 129         return op(other)
 130 
 131     wrapper.__name__ = opname
 132     return wrapper
 133

 135 def make_wrapped_arith_op(opname: str):
 136     def method(self, other):
 137         if (
 138             isinstance(other, Index)
 139             and is_object_dtype(other.dtype)
 140             and type(other) is not Index
 141         ):
 142             # We return NotImplemented for object-dtype index *subclasses* so they have
 143             # a chance to implement ops before we unwrap them.
 144             # See https://github.com/pandas-dev/pandas/issues/31109
 145             return NotImplemented
 146         meth = getattr(self._data, opname)
 147         result = meth(_maybe_unwrap_index(other))
 148         return _wrap_arithmetic_op(self, other, result)
 149 
 150     method.__name__ = opname
 151     return method
 152 
 153 
 154 def _wrap_arithmetic_op(self, other, result):
 155     if result is NotImplemented:
 156         return NotImplemented
 157 
 158     if isinstance(result, tuple):
 159         # divmod, rdivmod
 160         assert len(result) == 2
 161         return (
 162             _wrap_arithmetic_op(self, other, result[0]),
 163             _wrap_arithmetic_op(self, other, result[1]),
 164         )
 165 
 166     if not isinstance(result, Index):
 167         # Index.__new__ will choose appropriate subclass for dtype
 168         result = Index(result)
 169 
 170     res_name = get_op_result_name(self, other)
 171     result.name = res_name
 172     return result

 175 def _maybe_unwrap_index(obj):
 176     """
 177     If operating against another Index object, we need to unwrap the underlying
 178     data before deferring to the DatetimeArray/TimedeltaArray/PeriodArray
 179     implementation, otherwise we will incorrectly return NotImplemented.
 180 
 181     Parameters
 182     ----------
 183     obj : object
 184 
 185     Returns
 186     -------
 187     unwrapped object
 188     """
 189     if isinstance(obj, Index):
 190         return obj._data
 191     return obj

 194 class ExtensionIndex(Index):
 195     """
 196     Index subclass for indexes backed by ExtensionArray.
 197     """
 198 
 199     # The base class already passes through to _data:
 200     #  size, __len__, dtype
 201 
 202     _data: ExtensionArray
 203 
 204     __eq__ = _make_wrapped_comparison_op("__eq__")
 205     __ne__ = _make_wrapped_comparison_op("__ne__")
 206     __lt__ = _make_wrapped_comparison_op("__lt__")
 207     __gt__ = _make_wrapped_comparison_op("__gt__")
 208     __le__ = _make_wrapped_comparison_op("__le__")
 209     __ge__ = _make_wrapped_comparison_op("__ge__")
 210 
 211     # ---------------------------------------------------------------------
 212     # NDarray-Like Methods
 213 
 214     def __getitem__(self, key):
 215         result = self._data[key]
 216         if isinstance(result, type(self._data)):
 217             if result.ndim == 1:
 218                 return type(self)(result, name=self.name)
 219             # Unpack to ndarray for MPL compat
 220             result = result._data
 221 
 222         # Includes cases where we get a 2D ndarray back for MPL compat
 223         deprecate_ndim_indexing(result)
 224         return result
 225 
 226     # ---------------------------------------------------------------------
 227 
 228     def _get_engine_target(self) -> np.ndarray:
 229         # NB: _values_for_argsort happens to match the desired engine targets
 230         #  for all of our existing EA-backed indexes, but in general
 231         #  cannot be relied upon to exist.
 232         return self._data._values_for_argsort()
 233 
 234     def repeat(self, repeats, axis=None):
 235         nv.validate_repeat(tuple(), dict(axis=axis))
 236         result = self._data.repeat(repeats, axis=axis)
 237         return self._shallow_copy(result)
 238 
 239     def insert(self, loc: int, item):
 240         # ExtensionIndex subclasses must override Index.insert
 241         raise AbstractMethodError(self)
 242 
 243     def _get_unique_index(self, dropna=False):
 244         if self.is_unique and not dropna:
 245             return self
 246 
 247         result = self._data.unique()
 248         if dropna and self.hasnans:
 249             result = result[~result.isna()]
 250         return self._shallow_copy(result)
 251 
 252     @doc(Index.map)
 253     def map(self, mapper, na_action=None):
 254         # Try to run function on index first, and then on elements of index
 255         # Especially important for group-by functionality
 256         try:
 257             result = mapper(self)
 258 
 259             # Try to use this result if we can
 260             if isinstance(result, np.ndarray):
 261                 result = Index(result)
 262 
 263             if not isinstance(result, Index):
 264                 raise TypeError("The map function must return an Index object")
 265             return result
 266         except Exception:
 267             return self.astype(object).map(mapper)
 268 
 269     @doc(Index.astype)
 270     def astype(self, dtype, copy=True):
 271         if is_dtype_equal(self.dtype, dtype) and copy is False:
 272             # Ensure that self.astype(self.dtype) is self
 273             return self
 274 
 275         new_values = self._data.astype(dtype, copy=copy)
 276 
 277         # pass copy=False because any copying will be done in the
 278         #  _data.astype call above
 279         return Index(new_values, dtype=new_values.dtype, name=self.name, copy=False)
