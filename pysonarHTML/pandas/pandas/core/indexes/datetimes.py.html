  34 def _new_DatetimeIndex(cls, d):
  35     """
  36     This is called upon unpickling, rather than the default which doesn't
  37     have arguments and breaks __new__
  38     """
  39     if "data" in d and not isinstance(d["data"], DatetimeIndex):
  40         # Avoid need to verify integrity by calling simple_new directly
  41         data = d.pop("data")
  42         if not isinstance(data, DatetimeArray):
  43             # For backward compat with older pickles, we may need to construct
  44             #  a DatetimeArray to adapt to the newer _simple_new signature
  45             tz = d.pop("tz")
  46             freq = d.pop("freq")
  47             dta = DatetimeArray._simple_new(data, dtype=tz_to_dtype(tz), freq=freq)
  48         else:
  49             dta = data
  50             for key in ["tz", "freq"]:
  51                 # These are already stored in our DatetimeArray; if they are
  52                 #  also in the pickle and don't match, we have a problem.
  53                 if key in d:
  54                     assert d.pop(key) == getattr(dta, key)
  55         result = cls._simple_new(dta, **d)
  56     else:
  57         with warnings.catch_warnings():
  58             # TODO: If we knew what was going in to **d, we might be able to
  59             #  go through _simple_new instead
  60             warnings.simplefilter("ignore")
  61             result = cls.__new__(cls, **d)
  62 
  63     return result

 101 class DatetimeIndex(DatetimeTimedeltaMixin):
 102     """
 103     Immutable ndarray-like of datetime64 data.
 104 
 105     Represented internally as int64, and which can be boxed to Timestamp objects
 106     that are subclasses of datetime and carry metadata.
 107 
 108     Parameters
 109     ----------
 110     data : array-like (1-dimensional), optional
 111         Optional datetime-like data to construct index with.
 112     freq : str or pandas offset object, optional
 113         One of pandas date offset strings or corresponding objects. The string
 114         'infer' can be passed in order to set the frequency of the index as the
 115         inferred frequency upon creation.
 116     tz : pytz.timezone or dateutil.tz.tzfile or datetime.tzinfo or str
 117         Set the Timezone of the data.
 118     normalize : bool, default False
 119         Normalize start/end dates to midnight before generating date range.
 120     closed : {'left', 'right'}, optional
 121         Set whether to include `start` and `end` that are on the
 122         boundary. The default includes boundary points on either end.
 123     ambiguous : 'infer', bool-ndarray, 'NaT', default 'raise'
 124         When clocks moved backward due to DST, ambiguous times may arise.
 125         For example in Central European Time (UTC+01), when going from 03:00
 126         DST to 02:00 non-DST, 02:30:00 local time occurs both at 00:30:00 UTC
 127         and at 01:30:00 UTC. In such a situation, the `ambiguous` parameter
 128         dictates how ambiguous times should be handled.
 129 
 130         - 'infer' will attempt to infer fall dst-transition hours based on
 131           order
 132         - bool-ndarray where True signifies a DST time, False signifies a
 133           non-DST time (note that this flag is only applicable for ambiguous
 134           times)
 135         - 'NaT' will return NaT where there are ambiguous times
 136         - 'raise' will raise an AmbiguousTimeError if there are ambiguous times.
 137     dayfirst : bool, default False
 138         If True, parse dates in `data` with the day first order.
 139     yearfirst : bool, default False
 140         If True parse dates in `data` with the year first order.
 141     dtype : numpy.dtype or DatetimeTZDtype or str, default None
 142         Note that the only NumPy dtype allowed is ‘datetime64[ns]’.
 143     copy : bool, default False
 144         Make a copy of input ndarray.
 145     name : label, default None
 146         Name to be stored in the index.

 328     def _is_comparable_dtype(self, dtype: DtypeObj) -> bool:
 329         """
 330         Can we compare values of the given dtype to our own?
 331         """
 332         if not is_datetime64_any_dtype(dtype):
 333             return False
 334         if self.tz is not None:
 335             # If we have tz, we can compare to tzaware
 336             return is_datetime64tz_dtype(dtype)
 337         # if we dont have tz, we can only compare to tznaive
 338         return is_datetime64_dtype(dtype)

 357     def union_many(self, others):
 358         """
 359         A bit of a hack to accelerate unioning a collection of indexes.
 360         """
 361         this = self
 362 
 363         for other in others:
 364             if not isinstance(this, DatetimeIndex):
 365                 this = Index.union(this, other)
 366                 continue
 367 
 368             if not isinstance(other, DatetimeIndex):
 369                 try:
 370                     other = DatetimeIndex(other)
 371                 except TypeError:
 372                     pass
 373 
 374             this, other = this._maybe_utc_convert(other)
 375 
 376             if this._can_fast_union(other):
 377                 this = this._fast_union(other)
 378             else:
 379                 this = Index.union(this, other)
 380         return this

 390     def to_series(self, keep_tz=lib.no_default, index=None, name=None):
 391         """
 392         Create a Series with both index and values equal to the index keys
 393         useful with map for returning an indexer based on an index.
 394 
 395         Parameters
 396         ----------
 397         keep_tz : optional, defaults True
 398             Return the data keeping the timezone.
 399 
 400             If keep_tz is True:
 401 
 402               If the timezone is not set, the resulting
 403               Series will have a datetime64[ns] dtype.
 404 
 405               Otherwise the Series will have an datetime64[ns, tz] dtype; the
 406               tz will be preserved.
 407 
 408             If keep_tz is False:
 409 
 410               Series will have a datetime64[ns] dtype. TZ aware
 411               objects will have the tz removed.
 412 
 413             .. versionchanged:: 1.0.0
 414                 The default value is now True.  In a future version,
 415                 this keyword will be removed entirely.  Stop passing the
 416                 argument to obtain the future behavior and silence the warning.
 417 
 418         index : Index, optional
 419             Index of resulting Series. If None, defaults to original index.
 420         name : str, optional
 421             Name of resulting Series. If None, defaults to name of original
 422             index.
 423 
 424         Returns
 425         -------
 426         Series
 427         """
 428         from pandas import Series
 429 
 430         if index is None:
 431             index = self._shallow_copy()
 432         if name is None:
 433             name = self.name
 434 
 435         if keep_tz is not lib.no_default:
 436             if keep_tz:
 437                 warnings.warn(
 438                     "The 'keep_tz' keyword in DatetimeIndex.to_series "
 439                     "is deprecated and will be removed in a future version.  "
 440                     "You can stop passing 'keep_tz' to silence this warning.",
 441                     FutureWarning,
 442                     stacklevel=2,
 443                 )
 444             else:
 445                 warnings.warn(
 446                     "Specifying 'keep_tz=False' is deprecated and this "
 447                     "option will be removed in a future release. If "
 448                     "you want to remove the timezone information, you "
 449                     "can do 'idx.tz_convert(None)' before calling "
 450                     "'to_series'.",
 451                     FutureWarning,
 452                     stacklevel=2,
 453                 )
 454         else:
 455             keep_tz = True
 456 
 457         if keep_tz and self.tz is not None:
 458             # preserve the tz & copy
 459             values = self.copy(deep=True)
 460         else:
 461             values = self._values.view("M8[ns]").copy()
 462 
 463         return Series(values, index=index, name=name)

 465     def snap(self, freq="S"):
 466         """
 467         Snap time stamps to nearest occurring frequency.
 468 
 469         Returns
 470         -------
 471         DatetimeIndex
 472         """
 473         # Superdumb, punting on any optimizing
 474         freq = to_offset(freq)
 475 
 476         snapped = np.empty(len(self), dtype=DT64NS_DTYPE)
 477 
 478         for i, v in enumerate(self):
 479             s = v
 480             if not freq.is_on_offset(s):
 481                 t0 = freq.rollback(s)
 482                 t1 = freq.rollforward(s)
 483                 if abs(s - t0) < abs(t1 - s):
 484                     s = t0
 485                 else:
 486                     s = t1
 487             snapped[i] = s
 488 
 489         dta = DatetimeArray(snapped, dtype=self.dtype)
 490         return DatetimeIndex._simple_new(dta, name=self.name)

 492     def _parsed_string_to_bounds(self, reso: Resolution, parsed: datetime):
 493         """
 494         Calculate datetime bounds for parsed time string and its resolution.
 495 
 496         Parameters
 497         ----------
 498         reso : str
 499             Resolution provided by parsed string.
 500         parsed : datetime
 501             Datetime from parsed string.
 502 
 503         Returns
 504         -------
 505         lower, upper: pd.Timestamp
 506         """
 507         assert isinstance(reso, Resolution), (type(reso), reso)
 508         valid_resos = {
 509             "year",
 510             "month",
 511             "quarter",
 512             "day",
 513             "hour",
 514             "minute",
 515             "second",
 516             "minute",
 517             "second",
 518             "microsecond",
 519         }
 520         if reso.attrname not in valid_resos:
 521             raise KeyError
 522 
 523         grp = reso.freq_group
 524         per = Period(parsed, freq=grp)
 525         start, end = per.start_time, per.end_time
 526 
 527         # GH 24076
 528         # If an incoming date string contained a UTC offset, need to localize
 529         # the parsed date to this offset first before aligning with the index's
 530         # timezone
 531         if parsed.tzinfo is not None:
 532             if self.tz is None:
 533                 raise ValueError(
 534                     "The index must be timezone aware when indexing "
 535                     "with a date string with a UTC offset"
 536                 )
 537             start = start.tz_localize(parsed.tzinfo).tz_convert(self.tz)
 538             end = end.tz_localize(parsed.tzinfo).tz_convert(self.tz)
 539         elif self.tz is not None:
 540             start = start.tz_localize(self.tz)
 541             end = end.tz_localize(self.tz)
 542         return start, end

 562     def get_loc(self, key, method=None, tolerance=None):
 563         """
 564         Get integer location for requested label
 565 
 566         Returns
 567         -------
 568         loc : int
 569         """
 570         if not is_scalar(key):
 571             raise InvalidIndexError(key)
 572 
 573         orig_key = key
 574         if is_valid_nat_for_dtype(key, self.dtype):
 575             key = NaT
 576 
 577         if isinstance(key, self._data._recognized_scalars):
 578             # needed to localize naive datetimes
 579             key = self._maybe_cast_for_get_loc(key)
 580 
 581         elif isinstance(key, str):
 582             try:
 583                 return self._get_string_slice(key)
 584             except (TypeError, KeyError, ValueError, OverflowError):
 585                 pass
 586 
 587             try:
 588                 key = self._maybe_cast_for_get_loc(key)
 589             except ValueError as err:
 590                 raise KeyError(key) from err
 591 
 592         elif isinstance(key, timedelta):
 593             # GH#20464
 594             raise TypeError(
 595                 f"Cannot index {type(self).__name__} with {type(key).__name__}"
 596             )
 597 
 598         elif isinstance(key, time):
 599             if method is not None:
 600                 raise NotImplementedError(
 601                     "cannot yet lookup inexact labels when key is a time object"
 602                 )
 603             return self.indexer_at_time(key)
 604 
 605         else:
 606             # unrecognized type
 607             raise KeyError(key)
 608 
 609         try:
 610             return Index.get_loc(self, key, method, tolerance)
 611         except KeyError as err:
 612             raise KeyError(orig_key) from err
 613 
 614     def _maybe_cast_for_get_loc(self, key) -> Timestamp:
 615         # needed to localize naive datetimes
 616         key = Timestamp(key)
 617         if key.tzinfo is None:
 618             key = key.tz_localize(self.tz)
 619         else:
 620             key = key.tz_convert(self.tz)
 621         return key

 623     def _maybe_cast_slice_bound(self, label, side: str, kind):
 624         """
 625         If label is a string, cast it to datetime according to resolution.
 626 
 627         Parameters
 628         ----------
 629         label : object
 630         side : {'left', 'right'}
 631         kind : {'loc', 'getitem'} or None
 632 
 633         Returns
 634         -------
 635         label : object
 636 
 637         Notes
 638         -----
 639         Value of `side` parameter should be validated in caller.
 640         """
 641         assert kind in ["loc", "getitem", None]
 642 
 643         if is_float(label) or isinstance(label, time) or is_integer(label):
 644             self._invalid_indexer("slice", label)
 645 
 646         if isinstance(label, str):
 647             freq = getattr(self, "freqstr", getattr(self, "inferred_freq", None))
 648             parsed, reso = parsing.parse_time_string(label, freq)
 649             reso = Resolution.from_attrname(reso)
 650             lower, upper = self._parsed_string_to_bounds(reso, parsed)
 651             # lower, upper form the half-open interval:
 652             #   [parsed, parsed + 1 freq)
 653             # because label may be passed to searchsorted
 654             # the bounds need swapped if index is reverse sorted and has a
 655             # length > 1 (is_monotonic_decreasing gives True for empty
 656             # and length 1 index)
 657             if self._is_strictly_monotonic_decreasing and len(self) > 1:
 658                 return upper if side == "left" else lower
 659             return lower if side == "left" else upper
 660         else:
 661             return label

 663     def _get_string_slice(self, key: str, use_lhs: bool = True, use_rhs: bool = True):
 664         freq = getattr(self, "freqstr", getattr(self, "inferred_freq", None))
 665         parsed, reso = parsing.parse_time_string(key, freq)
 666         reso = Resolution.from_attrname(reso)
 667         loc = self._partial_date_slice(reso, parsed, use_lhs=use_lhs, use_rhs=use_rhs)
 668         return loc

 670     def slice_indexer(self, start=None, end=None, step=None, kind=None):
 671         """
 672         Return indexer for specified label slice.
 673         Index.slice_indexer, customized to handle time slicing.
 674 
 675         In addition to functionality provided by Index.slice_indexer, does the
 676         following:
 677 
 678         - if both `start` and `end` are instances of `datetime.time`, it
 679           invokes `indexer_between_time`
 680         - if `start` and `end` are both either string or None perform
 681           value-based selection in non-monotonic cases.
 682 
 683         """
 684         # For historical reasons DatetimeIndex supports slices between two
 685         # instances of datetime.time as if it were applying a slice mask to
 686         # an array of (self.hour, self.minute, self.seconds, self.microsecond).
 687         if isinstance(start, time) and isinstance(end, time):
 688             if step is not None and step != 1:
 689                 raise ValueError("Must have step size of 1 with time slices")
 690             return self.indexer_between_time(start, end)
 691 
 692         if isinstance(start, time) or isinstance(end, time):
 693             raise KeyError("Cannot mix time and non-time slice keys")
 694 
 695         # Pandas supports slicing with dates, treated as datetimes at midnight.
 696         # https://github.com/pandas-dev/pandas/issues/31501
 697         if isinstance(start, date) and not isinstance(start, datetime):
 698             start = datetime.combine(start, time(0, 0))
 699         if isinstance(end, date) and not isinstance(end, datetime):
 700             end = datetime.combine(end, time(0, 0))
 701 
 702         try:
 703             return Index.slice_indexer(self, start, end, step, kind=kind)
 704         except KeyError:
 705             # For historical reasons DatetimeIndex by default supports
 706             # value-based partial (aka string) slices on non-monotonic arrays,
 707             # let's try that.
 708             if (start is None or isinstance(start, str)) and (
 709                 end is None or isinstance(end, str)
 710             ):
 711                 mask = True
 712                 if start is not None:
 713                     start_casted = self._maybe_cast_slice_bound(start, "left", kind)
 714                     mask = start_casted <= self
 715 
 716                 if end is not None:
 717                     end_casted = self._maybe_cast_slice_bound(end, "right", kind)
 718                     mask = (self <= end_casted) & mask
 719 
 720                 indexer = mask.nonzero()[0][::step]
 721                 if len(indexer) == len(self):
 722                     return slice(None)
 723                 else:
 724                     return indexer
 725             else:
 726                 raise

 731         return typ == self.inferred_type or typ == "datetime"

 734     def inferred_type(self) -> str:
 735         # b/c datetime is represented as microseconds since the epoch, make
 736         # sure we can't have ambiguous indexing
 737         return "datetime64"

 739     def indexer_at_time(self, time, asof=False):
 740         """
 741         Return index locations of values at particular time of day
 742         (e.g. 9:30AM).
 743 
 744         Parameters
 745         ----------
 746         time : datetime.time or str
 747             Time passed in either as object (datetime.time) or as string in
 748             appropriate format ("%H:%M", "%H%M", "%I:%M%p", "%I%M%p",
 749             "%H:%M:%S", "%H%M%S", "%I:%M:%S%p", "%I%M%S%p").
 750 
 751         Returns
 752         -------
 753         values_at_time : array of integers
 754 
 755         See Also
 756         --------
 757         indexer_between_time : Get index locations of values between particular
 758             times of day.
 759         DataFrame.at_time : Select values at particular time of day.
 760         """
 761         if asof:
 762             raise NotImplementedError("'asof' argument is not supported")
 763 
 764         if isinstance(time, str):
 765             from dateutil.parser import parse
 766 
 767             time = parse(time).time()
 768 
 769         if time.tzinfo:
 770             if self.tz is None:
 771                 raise ValueError("Index must be timezone aware.")
 772             time_micros = self.tz_convert(time.tzinfo)._get_time_micros()
 773         else:
 774             time_micros = self._get_time_micros()
 775         micros = _time_to_micros(time)
 776         return (micros == time_micros).nonzero()[0]
 777 
 778     def indexer_between_time(
 779         self, start_time, end_time, include_start=True, include_end=True
 780     ):
 781         """
 782         Return index locations of values between particular times of day
 783         (e.g., 9:00-9:30AM).
 784 
 785         Parameters
 786         ----------
 787         start_time, end_time : datetime.time, str
 788             Time passed either as object (datetime.time) or as string in
 789             appropriate format ("%H:%M", "%H%M", "%I:%M%p", "%I%M%p",
 790             "%H:%M:%S", "%H%M%S", "%I:%M:%S%p","%I%M%S%p").
 791         include_start : bool, default True
 792         include_end : bool, default True
 793 
 794         Returns
 795         -------
 796         values_between_time : array of integers
 797 
 798         See Also
 799         --------
 800         indexer_at_time : Get index locations of values at particular time of day.
 801         DataFrame.between_time : Select values between particular times of day.
 802         """
 803         start_time = to_time(start_time)
 804         end_time = to_time(end_time)
 805         time_micros = self._get_time_micros()
 806         start_micros = _time_to_micros(start_time)
 807         end_micros = _time_to_micros(end_time)
 808 
 809         if include_start and include_end:
 810             lop = rop = operator.le
 811         elif include_start:
 812             lop = operator.le
 813             rop = operator.lt
 814         elif include_end:
 815             lop = operator.lt
 816             rop = operator.le
 817         else:
 818             lop = rop = operator.lt
 819 
 820         if start_time <= end_time:
 821             join_op = operator.and_
 822         else:
 823             join_op = operator.or_
 824 
 825         mask = join_op(lop(start_micros, time_micros), rop(time_micros, end_micros))
 826 
 827         return mask.nonzero()[0]

 834 def date_range(
 835     start=None,
 836     end=None,
 837     periods=None,
 838     freq=None,
 839     tz=None,
 840     normalize=False,
 841     name=None,
 842     closed=None,
 843     **kwargs,
 844 ) -> DatetimeIndex:
 845     """
 846     Return a fixed frequency DatetimeIndex.
 847 
 848     Parameters
 849     ----------
 850     start : str or datetime-like, optional
 851         Left bound for generating dates.
 852     end : str or datetime-like, optional
 853         Right bound for generating dates.
 854     periods : int, optional
 855         Number of periods to generate.
 856     freq : str or DateOffset, default 'D'
 857         Frequency strings can have multiples, e.g. '5H'. See
 858         :ref:`here <timeseries.offset_aliases>` for a list of
 859         frequency aliases.
 860     tz : str or tzinfo, optional
 861         Time zone name for returning localized DatetimeIndex, for example
 862         'Asia/Hong_Kong'. By default, the resulting DatetimeIndex is
 863         timezone-naive.
 864     normalize : bool, default False
 865         Normalize start/end dates to midnight before generating date range.
 866     name : str, default None
 867         Name of the resulting DatetimeIndex.
 868     closed : {None, 'left', 'right'}, optional
 869         Make the interval closed with respect to the given frequency to
 870         the 'left', 'right', or both sides (None, the default).
 871     **kwargs
 872         For compatibility. Has no effect on the result.
 873 
 874     Returns
 875     -------
 876     rng : DatetimeIndex
 877 
 878     See Also
 879     --------
 880     DatetimeIndex : An immutable container for datetimes.
 881     timedelta_range : Return a fixed frequency TimedeltaIndex.
 882     period_range : Return a fixed frequency PeriodIndex.
 883     interval_range : Return a fixed frequency IntervalIndex.
 884 
 885     Notes
 886     -----
 887     Of the four parameters ``start``, ``end``, ``periods``, and ``freq``,
 888     exactly three must be specified. If ``freq`` is omitted, the resulting
 889     ``DatetimeIndex`` will have ``periods`` linearly spaced elements between
 890     ``start`` and ``end`` (closed on both sides).
 891 
 892     To learn more about the frequency strings, please see `this link
 893     <https://pandas.pydata.org/pandas-docs/stable/user_guide/timeseries.html#offset-aliases>`__.
 894 
 895     Examples
 896     --------
 897     **Specifying the values**
 898 
 899     The next four examples generate the same `DatetimeIndex`, but vary
 900     the combination of `start`, `end` and `periods`.
 901 
 902     Specify `start` and `end`, with the default daily frequency.
 903 
 904     >>> pd.date_range(start='1/1/2018', end='1/08/2018')
 905     DatetimeIndex(['2018-01-01', '2018-01-02', '2018-01-03', '2018-01-04',
 906                    '2018-01-05', '2018-01-06', '2018-01-07', '2018-01-08'],
 907                   dtype='datetime64[ns]', freq='D')
 908 
 909     Specify `start` and `periods`, the number of periods (days).
 910 
 911     >>> pd.date_range(start='1/1/2018', periods=8)
 912     DatetimeIndex(['2018-01-01', '2018-01-02', '2018-01-03', '2018-01-04',
 913                    '2018-01-05', '2018-01-06', '2018-01-07', '2018-01-08'],
 914                   dtype='datetime64[ns]', freq='D')
 915 
 916     Specify `end` and `periods`, the number of periods (days).
 917 
 918     >>> pd.date_range(end='1/1/2018', periods=8)
 919     DatetimeIndex(['2017-12-25', '2017-12-26', '2017-12-27', '2017-12-28',
 920                    '2017-12-29', '2017-12-30', '2017-12-31', '2018-01-01'],
 921                   dtype='datetime64[ns]', freq='D')
 922 
 923     Specify `start`, `end`, and `periods`; the frequency is generated
 924     automatically (linearly spaced).
 925 
 926     >>> pd.date_range(start='2018-04-24', end='2018-04-27', periods=3)
 927     DatetimeIndex(['2018-04-24 00:00:00', '2018-04-25 12:00:00',
 928                    '2018-04-27 00:00:00'],
 929                   dtype='datetime64[ns]', freq=None)
 930 
 931     **Other Parameters**
 932 
 933     Changed the `freq` (frequency) to ``'M'`` (month end frequency).
 934 
 935     >>> pd.date_range(start='1/1/2018', periods=5, freq='M')
 936     DatetimeIndex(['2018-01-31', '2018-02-28', '2018-03-31', '2018-04-30',
 937                    '2018-05-31'],
 938                   dtype='datetime64[ns]', freq='M')
 939 
 940     Multiples are allowed
 941 
 942     >>> pd.date_range(start='1/1/2018', periods=5, freq='3M')
 943     DatetimeIndex(['2018-01-31', '2018-04-30', '2018-07-31', '2018-10-31',
 944                    '2019-01-31'],
 945                   dtype='datetime64[ns]', freq='3M')
 946 
 947     `freq` can also be specified as an Offset object.
 948 
 949     >>> pd.date_range(start='1/1/2018', periods=5, freq=pd.offsets.MonthEnd(3))
 950     DatetimeIndex(['2018-01-31', '2018-04-30', '2018-07-31', '2018-10-31',
 951                    '2019-01-31'],
 952                   dtype='datetime64[ns]', freq='3M')
 953 
 954     Specify `tz` to set the timezone.
 955 
 956     >>> pd.date_range(start='1/1/2018', periods=5, tz='Asia/Tokyo')
 957     DatetimeIndex(['2018-01-01 00:00:00+09:00', '2018-01-02 00:00:00+09:00',
 958                    '2018-01-03 00:00:00+09:00', '2018-01-04 00:00:00+09:00',
 959                    '2018-01-05 00:00:00+09:00'],
 960                   dtype='datetime64[ns, Asia/Tokyo]', freq='D')
 961 
 962     `closed` controls whether to include `start` and `end` that are on the
 963     boundary. The default includes boundary points on either end.
 964 
 965     >>> pd.date_range(start='2017-01-01', end='2017-01-04', closed=None)
 966     DatetimeIndex(['2017-01-01', '2017-01-02', '2017-01-03', '2017-01-04'],
 967                   dtype='datetime64[ns]', freq='D')
 968 
 969     Use ``closed='left'`` to exclude `end` if it falls on the boundary.
 970 
 971     >>> pd.date_range(start='2017-01-01', end='2017-01-04', closed='left')
 972     DatetimeIndex(['2017-01-01', '2017-01-02', '2017-01-03'],
 973                   dtype='datetime64[ns]', freq='D')
 974 
 975     Use ``closed='right'`` to exclude `start` if it falls on the boundary.
 976 
 977     >>> pd.date_range(start='2017-01-01', end='2017-01-04', closed='right')
 978     DatetimeIndex(['2017-01-02', '2017-01-03', '2017-01-04'],
 979                   dtype='datetime64[ns]', freq='D')
 980     """
 981     if freq is None and com.any_none(periods, start, end):
 982         freq = "D"
 983 
 984     dtarr = DatetimeArray._generate_range(
 985         start=start,
 986         end=end,
 987         periods=periods,
 988         freq=freq,
 989         tz=tz,
 990         normalize=normalize,
 991         closed=closed,
 992         **kwargs,
 993     )
 994     return DatetimeIndex._simple_new(dtarr, name=name)

 997 def bdate_range(
 998     start=None,
 999     end=None,
1000     periods=None,
1001     freq="B",
1002     tz=None,
1003     normalize=True,
1004     name=None,
1005     weekmask=None,
1006     holidays=None,
1007     closed=None,
1008     **kwargs,
1009 ) -> DatetimeIndex:
1010     """
1011     Return a fixed frequency DatetimeIndex, with business day as the default
1012     frequency.
1013 
1014     Parameters
1015     ----------
1016     start : str or datetime-like, default None
1017         Left bound for generating dates.
1018     end : str or datetime-like, default None
1019         Right bound for generating dates.
1020     periods : int, default None
1021         Number of periods to generate.
1022     freq : str or DateOffset, default 'B' (business daily)
1023         Frequency strings can have multiples, e.g. '5H'.
1024     tz : str or None
1025         Time zone name for returning localized DatetimeIndex, for example
1026         Asia/Beijing.
1027     normalize : bool, default False
1028         Normalize start/end dates to midnight before generating date range.
1029     name : str, default None
1030         Name of the resulting DatetimeIndex.
1031     weekmask : str or None, default None
1032         Weekmask of valid business days, passed to ``numpy.busdaycalendar``,
1033         only used when custom frequency strings are passed.  The default
1034         value None is equivalent to 'Mon Tue Wed Thu Fri'.
1035     holidays : list-like or None, default None
1036         Dates to exclude from the set of valid business days, passed to
1037         ``numpy.busdaycalendar``, only used when custom frequency strings
1038         are passed.
1039     closed : str, default None
1040         Make the interval closed with respect to the given frequency to
1041         the 'left', 'right', or both sides (None).
1042     **kwargs
1043         For compatibility. Has no effect on the result.
1044 
1045     Returns
1046     -------
1047     DatetimeIndex
1048 
1049     Notes
1050     -----
1051     Of the four parameters: ``start``, ``end``, ``periods``, and ``freq``,
1052     exactly three must be specified.  Specifying ``freq`` is a requirement
1053     for ``bdate_range``.  Use ``date_range`` if specifying ``freq`` is not
1054     desired.
1055 
1056     To learn more about the frequency strings, please see `this link
1057     <https://pandas.pydata.org/pandas-docs/stable/user_guide/timeseries.html#offset-aliases>`__.
1058 
1059     Examples
1060     --------
1061     Note how the two weekend days are skipped in the result.
1062 
1063     >>> pd.bdate_range(start='1/1/2018', end='1/08/2018')
1064     DatetimeIndex(['2018-01-01', '2018-01-02', '2018-01-03', '2018-01-04',
1065                '2018-01-05', '2018-01-08'],
1066               dtype='datetime64[ns]', freq='B')
1067     """
1068     if freq is None:
1069         msg = "freq must be specified for bdate_range; use date_range instead"
1070         raise TypeError(msg)
1071 
1072     if isinstance(freq, str) and freq.startswith("C"):
1073         try:
1074             weekmask = weekmask or "Mon Tue Wed Thu Fri"
1075             freq = prefix_mapping[freq](holidays=holidays, weekmask=weekmask)
1076         except (KeyError, TypeError) as err:
1077             msg = f"invalid custom frequency string: {freq}"
1078             raise ValueError(msg) from err
1079     elif holidays or weekmask:
1080         msg = (
1081             "a custom frequency string is required when holidays or "
1082             f"weekmask are passed, got frequency {freq}"
1083         )
1084         raise ValueError(msg)
1085 
1086     return date_range(
1087         start=start,
1088         end=end,
1089         periods=periods,
1090         freq=freq,
1091         tz=tz,
1092         normalize=normalize,
1093         name=name,
1094         closed=closed,
1095         **kwargs,
1096     )

1099 def _time_to_micros(time):
1100     seconds = time.hour * 60 * 60 + 60 * time.minute + time.second
1101     return 1000000 * seconds + time.microsecond
