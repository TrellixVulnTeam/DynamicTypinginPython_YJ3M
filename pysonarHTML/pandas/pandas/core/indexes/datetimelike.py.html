  54 def _join_i8_wrapper(joinf, with_indexers: bool = True):
  55     """
  56     Create the join wrapper methods.
  57     """
  58 
  59     @staticmethod  # type: ignore
  60     def wrapper(left, right):
  61         if isinstance(left, (np.ndarray, ABCIndex, ABCSeries, DatetimeLikeArrayMixin)):
  62             left = left.view("i8")
  63         if isinstance(right, (np.ndarray, ABCIndex, ABCSeries, DatetimeLikeArrayMixin)):
  64             right = right.view("i8")
  65 
  66         results = joinf(left, right)
  67         if with_indexers:
  68             # dtype should be timedelta64[ns] for TimedeltaIndex
  69             #  and datetime64[ns] for DatetimeIndex
  70             dtype = left.dtype.base
  71 
  72             join_index, left_indexer, right_indexer = results
  73             join_index = join_index.view(dtype)
  74             return join_index, left_indexer, right_indexer
  75         return results
  76 
  77     return wrapper

 115     def __array_wrap__(self, result, context=None):
 116         """
 117         Gets called after a ufunc.
 118         """
 119         result = lib.item_from_zerodim(result)
 120         if is_bool_dtype(result) or lib.is_scalar(result):
 121             return result
 122 
 123         attrs = self._get_attributes_dict()
 124         if not is_period_dtype(self.dtype) and attrs["freq"]:
 125             # no need to infer if freq is None
 126             attrs["freq"] = "infer"
 127         return Index(result, **attrs)

 131     def equals(self, other) -> bool:
 132         """
 133         Determines if two Index objects contain the same elements.
 134         """
 135         if self.is_(other):
 136             return True
 137 
 138         if not isinstance(other, ABCIndexClass):
 139             return False
 140         elif not isinstance(other, type(self)):
 141             try:
 142                 other = type(self)(other)
 143             except (ValueError, TypeError, OverflowError):
 144                 # e.g.
 145                 #  ValueError -> cannot parse str entry, or OutOfBoundsDatetime
 146                 #  TypeError  -> trying to convert IntervalIndex to DatetimeIndex
 147                 #  OverflowError -> Index([very_large_timedeltas])
 148                 return False
 149 
 150         if not is_dtype_equal(self.dtype, other.dtype):
 151             # have different timezone
 152             return False
 153 
 154         return np.array_equal(self.asi8, other.asi8)

 167     def sort_values(self, return_indexer=False, ascending=True, key=None):
 168         """
 169         Return sorted copy of Index.

 171         idx = ensure_key_mapped(self, key)

 174         if not ascending:
 175             _as = _as[::-1]
 176         sorted_index = self.take(_as)

 178         if return_indexer:
 179             return sorted_index, _as

 226     def min(self, axis=None, skipna=True, *args, **kwargs):
 227         """
 228         Return the minimum value of the Index or minimum along
 229         an axis.
 230 
 231         See Also
 232         --------
 233         numpy.ndarray.min
 234         Series.min : Return the minimum value in a Series.
 235         """
 236         nv.validate_min(args, kwargs)
 237         nv.validate_minmax_axis(axis)
 238 
 239         if not len(self):
 240             return self._na_value
 241 
 242         i8 = self.asi8
 243         try:
 244             # quick check
 245             if len(i8) and self.is_monotonic:
 246                 if i8[0] != iNaT:
 247                     return self._box_func(i8[0])
 248 
 249             if self.hasnans:
 250                 if skipna:
 251                     min_stamp = self[~self._isnan].asi8.min()
 252                 else:
 253                     return self._na_value
 254             else:
 255                 min_stamp = i8.min()
 256             return self._box_func(min_stamp)
 257         except ValueError:
 258             return self._na_value

 260     def argmin(self, axis=None, skipna=True, *args, **kwargs):
 261         """
 262         Returns the indices of the minimum values along an axis.
 263 
 264         See `numpy.ndarray.argmin` for more information on the
 265         `axis` parameter.
 266 
 267         See Also
 268         --------
 269         numpy.ndarray.argmin
 270         """
 271         nv.validate_argmin(args, kwargs)
 272         nv.validate_minmax_axis(axis)
 273 
 274         i8 = self.asi8
 275         if self.hasnans:
 276             mask = self._isnan
 277             if mask.all() or not skipna:
 278                 return -1
 279             i8 = i8.copy()
 280             i8[mask] = np.iinfo("int64").max
 281         return i8.argmin()

 372     def _partial_date_slice(
 373         self,
 374         reso: Resolution,
 375         parsed: datetime,
 376         use_lhs: bool = True,
 377         use_rhs: bool = True,
 378     ):
 379         """
 380         Parameters
 381         ----------
 382         reso : Resolution
 383         parsed : datetime
 384         use_lhs : bool, default True
 385         use_rhs : bool, default True
 386 
 387         Returns
 388         -------
 389         slice or ndarray[intp]
 390         """
 391         self._validate_partial_date_slice(reso)
 392 
 393         t1, t2 = self._parsed_string_to_bounds(reso, parsed)
 394         i8vals = self.asi8
 395         unbox = self._data._unbox_scalar
 396 
 397         if self.is_monotonic:
 398 
 399             if len(self) and (
 400                 (use_lhs and t1 < self[0] and t2 < self[0])
 401                 or ((use_rhs and t1 > self[-1] and t2 > self[-1]))
 402             ):
 403                 # we are out of range
 404                 raise KeyError
 405 
 406             # TODO: does this depend on being monotonic _increasing_?
 407 
 408             # a monotonic (sorted) series can be sliced
 409             # Use asi8.searchsorted to avoid re-validating Periods/Timestamps
 410             left = i8vals.searchsorted(unbox(t1), side="left") if use_lhs else None
 411             right = i8vals.searchsorted(unbox(t2), side="right") if use_rhs else None
 412             return slice(left, right)
 413 
 414         else:
 415             lhs_mask = (i8vals >= unbox(t1)) if use_lhs else True
 416             rhs_mask = (i8vals <= unbox(t2)) if use_rhs else True
 417 
 418             # try to find the dates
 419             return (lhs_mask & rhs_mask).nonzero()[0]

 441     def isin(self, values, level=None):
 442         """
 443         Compute boolean array of whether each index value is found in the
 444         passed set of values.
 445 
 446         Parameters
 447         ----------
 448         values : set or sequence of values
 449 
 450         Returns
 451         -------
 452         is_contained : ndarray (boolean dtype)
 453         """
 454         if level is not None:
 455             self._validate_index_level(level)
 456 
 457         if not isinstance(values, type(self)):
 458             try:
 459                 values = type(self)(values)
 460             except ValueError:
 461                 return self.astype(object).isin(values)
 462 
 463         return algorithms.isin(self.asi8, values.asi8)

 510     def shift(self, periods=1, freq=None):
 511         """
 512         Shift index by desired number of time frequency increments.
 513 
 514         This method is for shifting the values of datetime-like indexes
 515         by a specified time increment a given number of times.
 516 
 517         Parameters
 518         ----------
 519         periods : int, default 1
 520             Number of periods (or increments) to shift by,
 521             can be positive or negative.
 522 
 523             .. versionchanged:: 0.24.0
 524 
 525         freq : pandas.DateOffset, pandas.Timedelta or string, optional
 526             Frequency increment to shift by.
 527             If None, the index is shifted by its own `freq` attribute.
 528             Offset aliases are valid strings, e.g., 'D', 'W', 'M' etc.
 529 
 530         Returns
 531         -------
 532         pandas.DatetimeIndex
 533             Shifted index.
 534 
 535         See Also
 536         --------
 537         Index.shift : Shift values of Index.
 538         PeriodIndex.shift : Shift values of PeriodIndex.
 539         """
 540         arr = self._data.view()
 541         arr._freq = self.freq
 542         result = arr._time_shift(periods, freq=freq)
 543         return type(self)(result, name=self.name)

 548     def delete(self, loc):
 549         new_i8s = np.delete(self.asi8, loc)
 550 
 551         freq = None
 552         if is_period_dtype(self.dtype):
 553             freq = self.freq
 554         elif is_integer(loc):
 555             if loc in (0, -len(self), -1, len(self) - 1):
 556                 freq = self.freq
 557         else:
 558             if is_list_like(loc):
 559                 loc = lib.maybe_indices_to_slice(ensure_int64(np.array(loc)), len(self))
 560             if isinstance(loc, slice) and loc.step in (1, None):
 561                 if loc.start in (0, None) or loc.stop in (len(self), None):
 562                     freq = self.freq
 563 
 564         arr = type(self._data)._simple_new(new_i8s, dtype=self.dtype, freq=freq)
 565         return type(self)._simple_new(arr, name=self.name)

 656 
 657             .. versionchanged:: 0.25.0
 658 
 659                The `sort` keyword is added
 660 
 661         Returns
 662         -------
 663         y : Index or same type as self
 664         """
 665         self._validate_sort_keyword(sort)
 666         self._assert_can_do_setop(other)
 667         res_name = get_op_result_name(self, other)
 668 
 669         if self.equals(other):
 670             return self._get_reconciled_name_object(other)
 671 
 672         if len(self) == 0:
 673             return self.copy()
 674         if len(other) == 0:
 675             return other.copy()
 676 
 677         if not isinstance(other, type(self)):
 678             result = Index.intersection(self, other, sort=sort)
 679             if isinstance(result, type(self)):
 680                 if result.freq is None:
 681                     # TODO: no tests rely on this; needed?
 682                     result = result._with_freq("infer")
 683             assert result.name == res_name
 684             return result
 685 
 686         elif not self._can_fast_intersect(other):
 687             result = Index.intersection(self, other, sort=sort)
 688             assert result.name == res_name
 689             # We need to invalidate the freq because Index.intersection
 690             #  uses _shallow_copy on a view of self._data, which will preserve
 691             #  self.freq if we're not careful.
 692             result = result._with_freq(None)._with_freq("infer")
 693             return result
 694 
 695         # to make our life easier, "sort" the two ranges
 696         if self[0] <= other[0]:
 697             left, right = self, other
 698         else:
 699             left, right = other, self
 700 
 701         # after sorting, the intersection always starts with the right index
 702         # and ends with the index of which the last elements is smallest
 703         end = min(left[-1], right[-1])
 704         start = right[0]
 705 
 706         if end < start:
 707             return type(self)(data=[], dtype=self.dtype, freq=self.freq, name=res_name)
 708         else:
 709             lslice = slice(*left.slice_locs(start, end))
 710             left_chunk = left._values[lslice]
 711             return type(self)._simple_new(left_chunk, name=res_name)
 712 
 713     def _can_fast_intersect(self: _T, other: _T) -> bool:
 714         if self.freq is None:
 715             return False
 716 
 717         elif other.freq != self.freq:
 718             return False
 719 
 720         elif not self.is_monotonic_increasing:
 721             # Because freq is not None, we must then be monotonic decreasing
 722             return False
 723 
 724         elif self.freq.is_anchored():
 725             # this along with matching freqs ensure that we "line up",
 726             #  so intersection will preserve freq
 727             return True
 728 
 729         elif isinstance(self.freq, Tick):
 730             # We "line up" if and only if the difference between two of our points
 731             #  is a multiple of our freq
 732             diff = self[0] - other[0]
 733             remainder = diff % self.freq.delta
 734             return remainder == Timedelta(0)
 735 
 736         return True

 738     def _can_fast_union(self: _T, other: _T) -> bool:
 739         # Assumes that type(self) == type(other), as per the annotation
 740         # The ability to fast_union also implies that `freq` should be
 741         #  retained on union.
 742         if not isinstance(other, type(self)):
 743             return False
 744 
 745         freq = self.freq
 746 
 747         if freq is None or freq != other.freq:
 748             return False
 749 
 750         if not self.is_monotonic_increasing:
 751             # Because freq is not None, we must then be monotonic decreasing
 752             # TODO: do union on the reversed indexes?
 753             return False
 754 
 755         if len(self) == 0 or len(other) == 0:
 756             return True
 757 
 758         # to make our life easier, "sort" the two ranges
 759         if self[0] <= other[0]:
 760             left, right = self, other
 761         else:
 762             left, right = other, self
 763 
 764         right_start = right[0]
 765         left_end = left[-1]
 766 
 767         # Only need to "adjoin", not overlap
 768         return (right_start == left_end + freq) or right_start in left
 769 
 770     def _fast_union(self, other, sort=None):
 771         if len(other) == 0:
 772             return self.view(type(self))
 773 
 774         if len(self) == 0:
 775             return other.view(type(self))
 776 
 777         # to make our life easier, "sort" the two ranges
 778         if self[0] <= other[0]:
 779             left, right = self, other
 780         elif sort is False:
 781             # TDIs are not in the "correct" order and we don't want
 782             #  to sort but want to remove overlaps
 783             left, right = self, other
 784             left_start = left[0]
 785             loc = right.searchsorted(left_start, side="left")
 786             right_chunk = right._values[:loc]
 787             dates = concat_compat((left._values, right_chunk))
 788             # With sort being False, we can't infer that result.freq == self.freq
 789             # TODO: no tests rely on the _with_freq("infer"); needed?
 790             result = self._shallow_copy(dates)._with_freq("infer")
 791             return result
 792         else:
 793             left, right = other, self
 794 
 795         left_end = left[-1]
 796         right_end = right[-1]
 797 
 798         # concatenate
 799         if left_end < right_end:
 800             loc = right.searchsorted(left_end, side="right")
 801             right_chunk = right._values[loc:]
 802             dates = concat_compat([left._values, right_chunk])
 803             # The can_fast_union check ensures that the result.freq
 804             #  should match self.freq
 805             dates = type(self._data)(dates, freq=self.freq)
 806             result = type(self)._simple_new(dates, name=self.name)
 807             return result
 808         else:
 809             return left
 810 
 811     def _union(self, other, sort):
 812         if not len(other) or self.equals(other) or not len(self):
 813             return super()._union(other, sort=sort)
 814 
 815         # We are called by `union`, which is responsible for this validation
 816         assert isinstance(other, type(self))
 817 
 818         this, other = self._maybe_utc_convert(other)
 819 
 820         if this._can_fast_union(other):
 821             result = this._fast_union(other, sort=sort)
 822             if sort is None:
 823                 # In the case where sort is None, _can_fast_union
 824                 #  implies that result.freq should match self.freq
 825                 assert result.freq == self.freq, (result.freq, self.freq)
 826             elif result.freq is None:
 827                 # TODO: no tests rely on this; needed?
 828                 result = result._with_freq("infer")
 829             return result
 830         else:
 831             i8self = Int64Index._simple_new(self.asi8, name=self.name)
 832             i8other = Int64Index._simple_new(other.asi8, name=other.name)
 833             i8result = i8self._union(i8other, sort=sort)
 834             result = type(self)(i8result, dtype=self.dtype, freq="infer")
 835             return result

 849         self, other, how: str = "left", level=None, return_indexers=False, sort=False
 850     ):
 851         """
 852         See Index.join
 853         """
 854         if self._is_convertible_to_index_for_join(other):
 855             try:
 856                 other = type(self)(other)
 857             except (TypeError, ValueError):
 858                 pass
 859 
 860         this, other = self._maybe_utc_convert(other)
 861         return Index.join(
 862             this,
 863             other,
 864             how=how,
 865             level=level,
 866             return_indexers=return_indexers,
 867             sort=sort,
 868         )

 870     def _maybe_utc_convert(self, other):
 871         this = self
 872         if not hasattr(self, "tz"):
 873             return this, other
 874 
 875         if isinstance(other, type(self)):
 876             if self.tz is not None:
 877                 if other.tz is None:
 878                     raise TypeError("Cannot join tz-naive with tz-aware DatetimeIndex")
 879             elif other.tz is not None:
 880                 raise TypeError("Cannot join tz-naive with tz-aware DatetimeIndex")
 881 
 882             if not timezones.tz_compare(self.tz, other.tz):
 883                 this = self.tz_convert("UTC")
 884                 other = other.tz_convert("UTC")
 885         return this, other

 887     @classmethod
 888     def _is_convertible_to_index_for_join(cls, other: Index) -> bool:
 889         """
 890         return a boolean whether I can attempt conversion to a
 891         DatetimeIndex/TimedeltaIndex
 892         """
 893         if isinstance(other, cls):
 894             return False
 895         elif len(other) > 0 and other.inferred_type not in (
 896             "floating",
 897             "mixed-integer",
 898             "integer",
 899             "integer-na",
 900             "mixed-integer-float",
 901             "mixed",
 902         ):
 903             return True
 904         return False

 909     def insert(self, loc, item):
 910         """
 911         Make new Index inserting new item at location
 912 
 913         Parameters
 914         ----------
 915         loc : int
 916         item : object
 917             if not either a Python datetime or a numpy integer-like, returned
 918             Index dtype will be object rather than datetime.
 919 
 920         Returns
 921         -------
 922         new_index : Index
 923         """
 924         if isinstance(item, str):
 925             # TODO: Why are strings special?
 926             # TODO: Should we attempt _scalar_from_string?
 927             return self.astype(object).insert(loc, item)
 928 
 929         item = self._data._validate_insert_value(item)
 930 
 931         freq = None
 932         # check freq can be preserved on edge cases
 933         if self.freq is not None:
 934             if self.size:
 935                 if item is NaT:
 936                     pass
 937                 elif (loc == 0 or loc == -len(self)) and item + self.freq == self[0]:
 938                     freq = self.freq
 939                 elif (loc == len(self)) and item - self.freq == self[-1]:
 940                     freq = self.freq
 941             else:
 942                 # Adding a single item to an empty index may preserve freq
 943                 if self.freq.is_on_offset(item):
 944                     freq = self.freq
 945 
 946         item = self._data._unbox_scalar(item)
 947 
 948         new_i8s = np.concatenate([self[:loc].asi8, [item], self[loc:].asi8])
 949         arr = type(self._data)._simple_new(new_i8s, dtype=self.dtype, freq=freq)
 950         return type(self)._simple_new(arr, name=self.name)
