  65 def get_objs_combined_axis(
  66     objs, intersect: bool = False, axis=0, sort: bool = True, copy: bool = False
  67 ) -> Index:
  68     """
  69     Extract combined index: return intersection or union (depending on the
  70     value of "intersect") of indexes on given axis, or None if all objects
  71     lack indexes (e.g. they are numpy arrays).
  72 
  73     Parameters
  74     ----------
  75     objs : list
  76         Series or DataFrame objects, may be mix of the two.
  77     intersect : bool, default False
  78         If True, calculate the intersection between indexes. Otherwise,
  79         calculate the union.
  80     axis : {0 or 'index', 1 or 'outer'}, default 0
  81         The axis to extract indexes from.
  82     sort : bool, default True
  83         Whether the result index should come out sorted or not.
  84     copy : bool, default False
  85         If True, return a copy of the combined index.
  86 
  87     Returns
  88     -------
  89     Index
  90     """
  91     obs_idxes = [obj._get_axis(axis) for obj in objs]
  92     return _get_combined_index(obs_idxes, intersect=intersect, sort=sort, copy=copy)

  95 def _get_distinct_objs(objs: List[Index]) -> List[Index]:
  96     """
  97     Return a list with distinct elements of "objs" (different ids).
  98     Preserves order.
  99     """
 100     ids: Set[int] = set()
 101     res = []
 102     for obj in objs:
 103         if id(obj) not in ids:
 104             ids.add(id(obj))
 105             res.append(obj)
 106     return res

 109 def _get_combined_index(
 110     indexes: List[Index],
 111     intersect: bool = False,
 112     sort: bool = False,
 113     copy: bool = False,
 114 ) -> Index:
 115     """
 116     Return the union or intersection of indexes.
 117 
 118     Parameters
 119     ----------
 120     indexes : list of Index or list objects
 121         When intersect=True, do not accept list of lists.
 122     intersect : bool, default False
 123         If True, calculate the intersection between indexes. Otherwise,
 124         calculate the union.
 125     sort : bool, default False
 126         Whether the result index should come out sorted or not.
 127     copy : bool, default False
 128         If True, return a copy of the combined index.
 129 
 130     Returns
 131     -------
 132     Index
 133     """
 134     # TODO: handle index names!
 135     indexes = _get_distinct_objs(indexes)
 136     if len(indexes) == 0:
 137         index = Index([])
 138     elif len(indexes) == 1:
 139         index = indexes[0]
 140     elif intersect:
 141         index = indexes[0]
 142         for other in indexes[1:]:
 143             index = index.intersection(other)
 144     else:
 145         index = union_indexes(indexes, sort=sort)
 146         index = ensure_index(index)
 147 
 148     if sort:
 149         try:
 150             index = index.sort_values()
 151         except TypeError:
 152             pass
 153 
 154     # GH 29879
 155     if copy:
 156         index = index.copy()
 157 
 158     return index

 161 def union_indexes(indexes, sort=True) -> Index:
 162     """
 163     Return the union of indexes.
 164 
 165     The behavior of sort and names is not consistent.
 166 
 167     Parameters
 168     ----------
 169     indexes : list of Index or list objects
 170     sort : bool, default True
 171         Whether the result index should come out sorted or not.
 172 
 173     Returns
 174     -------
 175     Index
 176     """
 177     if len(indexes) == 0:
 178         raise AssertionError("Must have at least 1 Index to union")
 179     if len(indexes) == 1:
 180         result = indexes[0]
 181         if isinstance(result, list):
 182             result = Index(sorted(result))
 183         return result
 184 
 185     indexes, kind = _sanitize_and_check(indexes)
 186 
 187     def _unique_indices(inds) -> Index:
 188         """
 189         Convert indexes to lists and concatenate them, removing duplicates.
 190 
 191         The final dtype is inferred.
 192 
 193         Parameters
 194         ----------
 195         inds : list of Index or list objects
 196 
 197         Returns
 198         -------
 199         Index
 200         """
 201 
 202         def conv(i):
 203             if isinstance(i, Index):
 204                 i = i.tolist()
 205             return i
 206 
 207         return Index(lib.fast_unique_multiple_list([conv(i) for i in inds], sort=sort))
 208 
 209     if kind == "special":
 210         result = indexes[0]
 211 
 212         if hasattr(result, "union_many"):
 213             # DatetimeIndex
 214             return result.union_many(indexes[1:])
 215         else:
 216             for other in indexes[1:]:
 217                 result = result.union(other)
 218             return result
 219     elif kind == "array":
 220         index = indexes[0]
 221         for other in indexes[1:]:
 222             if not index.equals(other):
 223                 return _unique_indices(indexes)
 224 
 225         name = get_consensus_names(indexes)[0]
 226         if name != index.name:
 227             index = index._shallow_copy(name=name)
 228         return index
 229     else:  # kind='list'
 230         return _unique_indices(indexes)

 233 def _sanitize_and_check(indexes):
 234     """
 235     Verify the type of indexes and convert lists to Index.
 236 
 237     Cases:
 238 
 239     - [list, list, ...]: Return ([list, list, ...], 'list')
 240     - [list, Index, ...]: Return _sanitize_and_check([Index, Index, ...])
 241         Lists are sorted and converted to Index.
 242     - [Index, Index, ...]: Return ([Index, Index, ...], TYPE)
 243         TYPE = 'special' if at least one special type, 'array' otherwise.
 244 
 245     Parameters
 246     ----------
 247     indexes : list of Index or list objects
 248 
 249     Returns
 250     -------
 251     sanitized_indexes : list of Index or list objects
 252     type : {'list', 'array', 'special'}
 253     """
 254     kinds = list({type(index) for index in indexes})
 255 
 256     if list in kinds:
 257         if len(kinds) > 1:
 258             indexes = [
 259                 Index(list(x)) if not isinstance(x, Index) else x for x in indexes
 260             ]
 261             kinds.remove(list)
 262         else:
 263             return indexes, "list"
 264 
 265     if len(kinds) > 1 or Index not in kinds:
 266         return indexes, "special"
 267     else:
 268         return indexes, "array"

 271 def get_consensus_names(indexes):
 272     """
 273     Give a consensus 'names' to indexes.
 274 
 275     If there's exactly one non-empty 'names', return this,
 276     otherwise, return empty.
 277 
 278     Parameters
 279     ----------
 280     indexes : list of Index objects
 281 
 282     Returns
 283     -------
 284     list
 285         A list representing the consensus 'names' found.
 286     """
 287     # find the non-none names, need to tupleify to make
 288     # the set hashable, then reverse on return
 289     consensus_names = {tuple(i.names) for i in indexes if com.any_not_none(*i.names)}
 290     if len(consensus_names) == 1:
 291         return list(list(consensus_names)[0])
 292     return [None] * indexes[0].nlevels

 295 def all_indexes_same(indexes):
 296     """
 297     Determine if all indexes contain the same elements.
 298 
 299     Parameters
 300     ----------
 301     indexes : list of Index objects
 302 
 303     Returns
 304     -------
 305     bool
 306         True if all indexes contain the same elements, False otherwise.
 307     """
 308     first = indexes[0]
 309     for index in indexes[1:]:
 310         if not first.equals(index):
 311             return False
 312     return True
