  41 class RangeIndex(Int64Index):
  42     """
  43     Immutable Index implementing a monotonic integer range.
  44 
  45     RangeIndex is a memory-saving special case of Int64Index limited to
  46     representing monotonic ranges. Using RangeIndex may in some instances
  47     improve computing speed.
  48 
  49     This is the default index type used
  50     by DataFrame and Series when no explicit index is provided by the user.
  51 
  52     Parameters
  53     ----------
  54     start : int (default: 0), or other RangeIndex instance
  55         If int and "stop" is not given, interpreted as "stop" instead.
  56     stop : int (default: 0)
  57     step : int (default: 1)
  58     name : object, optional
  59         Name to be stored in the index.
  60     copy : bool, default False
  61         Unused, accepted for homogeneity with other index types.
  62 
  63     Attributes
  64     ----------
  65     start
  66     stop
  67     step
  68 
  69     Methods
  70     -------
  71     from_range
  72 
  73     See Also
  74     --------
  75     Index : The base pandas Index type.
  76     Int64Index : Index of int64 data.
  77     """

 357     @Appender(_index_shared_docs["get_indexer"])
 358     def get_indexer(self, target, method=None, limit=None, tolerance=None):
 359         if com.any_not_none(method, tolerance, limit) or not is_list_like(target):
 360             return super().get_indexer(
 361                 target, method=method, tolerance=tolerance, limit=limit
 362             )
 363 
 364         if self.step > 0:
 365             start, stop, step = self.start, self.stop, self.step
 366         else:
 367             # GH 28678: work on reversed range for simplicity
 368             reverse = self._range[::-1]
 369             start, stop, step = reverse.start, reverse.stop, reverse.step
 370 
 371         target_array = np.asarray(target)
 372         if not (is_integer_dtype(target_array) and target_array.ndim == 1):
 373             # checks/conversions/roundings are delegated to general method
 374             return super().get_indexer(target, method=method, tolerance=tolerance)
 375 
 376         locs = target_array - start
 377         valid = (locs % step == 0) & (locs >= 0) & (target_array < stop)
 378         locs[~valid] = -1
 379         locs[valid] = locs[valid] / step
 380 
 381         if step != self.step:
 382             # We reversed this range: transform to original locs
 383             locs[valid] = len(self) - 1 - locs[valid]
 384         return ensure_platform_int(locs)

 456     def intersection(self, other, sort=False):
 457         """
 458         Form the intersection of two Index objects.
 459 
 460         Parameters
 461         ----------
 462         other : Index or array-like
 463         sort : False or None, default False
 464             Sort the resulting index if possible
 465 
 466             .. versionadded:: 0.24.0
 467 
 468             .. versionchanged:: 0.24.1
 469 
 470                Changed the default to ``False`` to match the behaviour
 471                from before 0.24.0.
 472 
 473         Returns
 474         -------
 475         intersection : Index
 476         """
 477         self._validate_sort_keyword(sort)
 478 
 479         if self.equals(other):
 480             return self._get_reconciled_name_object(other)
 481 
 482         if not isinstance(other, RangeIndex):
 483             return super().intersection(other, sort=sort)
 484 
 485         if not len(self) or not len(other):
 486             return self._simple_new(_empty_range)
 487 
 488         first = self._range[::-1] if self.step < 0 else self._range
 489         second = other._range[::-1] if other.step < 0 else other._range
 490 
 491         # check whether intervals intersect
 492         # deals with in- and decreasing ranges
 493         int_low = max(first.start, second.start)
 494         int_high = min(first.stop, second.stop)
 495         if int_high <= int_low:
 496             return self._simple_new(_empty_range)
 497 
 498         # Method hint: linear Diophantine equation
 499         # solve intersection problem
 500         # performance hint: for identical step sizes, could use
 501         # cheaper alternative
 502         gcd, s, t = self._extended_gcd(first.step, second.step)
 503 
 504         # check whether element sets intersect
 505         if (first.start - second.start) % gcd:
 506             return self._simple_new(_empty_range)
 507 
 508         # calculate parameters for the RangeIndex describing the
 509         # intersection disregarding the lower bounds
 510         tmp_start = first.start + (second.start - first.start) * first.step // gcd * s
 511         new_step = first.step * second.step // gcd
 512         new_range = range(tmp_start, int_high, new_step)
 513         new_index = self._simple_new(new_range)
 514 
 515         # adjust index to limiting interval
 516         new_start = new_index._min_fitting_element(int_low)
 517         new_range = range(new_start, new_index.stop, new_index.step)
 518         new_index = self._simple_new(new_range)
 519 
 520         if (self.step < 0 and other.step < 0) is not (new_index.step < 0):
 521             new_index = new_index[::-1]
 522         if sort is None:
 523             new_index = new_index.sort_values()
 524         return new_index
 525 
 526     def _min_fitting_element(self, lower_limit: int) -> int:
 527         """Returns the smallest element greater than or equal to the limit"""
 528         no_steps = -(-(lower_limit - self.start) // abs(self.step))
 529         return self.start + abs(self.step) * no_steps
 530 
 531     def _max_fitting_element(self, upper_limit: int) -> int:
 532         """Returns the largest element smaller than or equal to the limit"""
 533         no_steps = (upper_limit - self.start) // abs(self.step)
 534         return self.start + abs(self.step) * no_steps

 536     def _extended_gcd(self, a, b):
 537         """
 538         Extended Euclidean algorithms to solve Bezout's identity:
 539            a*x + b*y = gcd(x, y)
 540         Finds one particular solution for x, y: s, t
 541         Returns: gcd, s, t
 542         """
 543         s, old_s = 0, 1
 544         t, old_t = 1, 0
 545         r, old_r = b, a
 546         while r:
 547             quotient = old_r // r
 548             old_r, r = r, old_r - quotient * r
 549             old_s, s = s, old_s - quotient * s
 550             old_t, t = t, old_t - quotient * t
 551         return old_r, old_s, old_t

 553     def _union(self, other, sort):
 554         """
 555         Form the union of two Index objects and sorts if possible
 556 
 557         Parameters
 558         ----------
 559         other : Index or array-like
 560 
 561         sort : False or None, default None
 562             Whether to sort resulting index. ``sort=None`` returns a
 563             monotonically increasing ``RangeIndex`` if possible or a sorted
 564             ``Int64Index`` if not. ``sort=False`` always returns an
 565             unsorted ``Int64Index``
 566 
 567             .. versionadded:: 0.25.0
 568 
 569         Returns
 570         -------
 571         union : Index
 572         """
 573         if not len(other) or self.equals(other) or not len(self):
 574             return super()._union(other, sort=sort)
 575 
 576         if isinstance(other, RangeIndex) and sort is None:
 577             start_s, step_s = self.start, self.step
 578             end_s = self.start + self.step * (len(self) - 1)
 579             start_o, step_o = other.start, other.step
 580             end_o = other.start + other.step * (len(other) - 1)
 581             if self.step < 0:
 582                 start_s, step_s, end_s = end_s, -step_s, start_s
 583             if other.step < 0:
 584                 start_o, step_o, end_o = end_o, -step_o, start_o
 585             if len(self) == 1 and len(other) == 1:
 586                 step_s = step_o = abs(self.start - other.start)
 587             elif len(self) == 1:
 588                 step_s = step_o
 589             elif len(other) == 1:
 590                 step_o = step_s
 591             start_r = min(start_s, start_o)
 592             end_r = max(end_s, end_o)
 593             if step_o == step_s:
 594                 if (
 595                     (start_s - start_o) % step_s == 0
 596                     and (start_s - end_o) <= step_s
 597                     and (start_o - end_s) <= step_s
 598                 ):
 599                     return type(self)(start_r, end_r + step_s, step_s)
 600                 if (
 601                     (step_s % 2 == 0)
 602                     and (abs(start_s - start_o) <= step_s / 2)
 603                     and (abs(end_s - end_o) <= step_s / 2)
 604                 ):
 605                     return type(self)(start_r, end_r + step_s / 2, step_s / 2)
 606             elif step_o % step_s == 0:
 607                 if (
 608                     (start_o - start_s) % step_s == 0
 609                     and (start_o + step_s >= start_s)
 610                     and (end_o - step_s <= end_s)
 611                 ):
 612                     return type(self)(start_r, end_r + step_s, step_s)
 613             elif step_s % step_o == 0:
 614                 if (
 615                     (start_s - start_o) % step_o == 0
 616                     and (start_s + step_o >= start_o)
 617                     and (end_s - step_o <= end_o)
 618                 ):
 619                     return type(self)(start_r, end_r + step_o, step_o)
 620         return self._int64index._union(other, sort=sort)

 630     def _concat(self, indexes, name):
 631         """
 632         Overriding parent method for the case of all RangeIndex instances.
 633 
 634         When all members of "indexes" are of type RangeIndex: result will be
 635         RangeIndex if possible, Int64Index otherwise. E.g.:
 636         indexes = [RangeIndex(3), RangeIndex(3, 6)] -> RangeIndex(6)
 637         indexes = [RangeIndex(3), RangeIndex(4, 6)] -> Int64Index([0,1,2,4,5])
 638         """
 639         if not all(isinstance(x, RangeIndex) for x in indexes):
 640             return super()._concat(indexes, name)
 641 
 642         start = step = next_ = None
 643 
 644         # Filter the empty indexes
 645         non_empty_indexes = [obj for obj in indexes if len(obj)]
 646 
 647         for obj in non_empty_indexes:
 648             rng: range = obj._range
 649 
 650             if start is None:
 651                 # This is set by the first non-empty index
 652                 start = rng.start
 653                 if step is None and len(rng) > 1:
 654                     step = rng.step
 655             elif step is None:
 656                 # First non-empty index had only one element
 657                 if rng.start == start:
 658                     result = Int64Index(np.concatenate([x._values for x in indexes]))
 659                     return result.rename(name)
 660 
 661                 step = rng.start - start
 662 
 663             non_consecutive = (step != rng.step and len(rng) > 1) or (
 664                 next_ is not None and rng.start != next_
 665             )
 666             if non_consecutive:
 667                 result = Int64Index(np.concatenate([x._values for x in indexes]))
 668                 return result.rename(name)
 669 
 670             if step is not None:
 671                 next_ = rng[-1] + step
 672 
 673         if non_empty_indexes:
 674             # Get the stop value from "next" or alternatively
 675             # from the last non-empty index
 676             stop = non_empty_indexes[-1].stop if next_ is None else next_
 677             return RangeIndex(start, stop, step).rename(name)
 678 
 679         # Here all "indexes" had 0 length, i.e. were empty.
 680         # In this case return an empty range index.
 681         return RangeIndex(0, 0).rename(name)

 684         """
 685         return the length of the RangeIndex
 686         """
 687         return len(self._range)

 693     def __getitem__(self, key):
 694         """
 695         Conserve RangeIndex type for scalar and slice keys.
 696         """
 697         if isinstance(key, slice):
 698             new_range = self._range[key]
 699             return self._simple_new(new_range, name=self.name)
 700         elif is_integer(key):
 701             new_key = int(key)
 702             try:
 703                 return self._range[new_key]
 704             except IndexError as err:
 705                 raise IndexError(
 706                     f"index {key} is out of bounds for axis 0 with size {len(self)}"
 707                 ) from err
 708         elif is_scalar(key):
 709             raise IndexError(
 710                 "only integers, slices (`:`), "
 711                 "ellipsis (`...`), numpy.newaxis (`None`) "
 712                 "and integer or boolean "
 713                 "arrays are valid indices"
 714             )
 715         # fall back to Int64Index
 716         return super().__getitem__(key)

 718     @unpack_zerodim_and_defer("__floordiv__")
 719     def __floordiv__(self, other):
 720 
 721         if is_integer(other) and other != 0:
 722             if len(self) == 0 or self.start % other == 0 and self.step % other == 0:
 723                 start = self.start // other
 724                 step = self.step // other
 725                 stop = start + len(self) * step
 726                 new_range = range(start, stop, step or 1)
 727                 return self._simple_new(new_range, name=self.name)
 728             if len(self) == 1:
 729                 start = self.start // other
 730                 new_range = range(start, start + 1, 1)
 731                 return self._simple_new(new_range, name=self.name)
 732         return self._int64index // other
 733 
 734     def all(self) -> bool:
 735         return 0 not in self._range
 736 
 737     def any(self) -> bool:
 738         return any(self._range)

 740     @classmethod
 741     def _add_numeric_methods_binary(cls):
 742         """ add in numeric methods, specialized to RangeIndex """
 743 
 744         def _make_evaluate_binop(op, step=False):
 745             """
 746             Parameters
 747             ----------
 748             op : callable that accepts 2 params
 749                 perform the binary op
 750             step : callable, optional, default to False
 751                 op to apply to the step parm if not None
 752                 if False, use the existing step
 753             """
 754 
 755             @unpack_zerodim_and_defer(op.__name__)
 756             def _evaluate_numeric_binop(self, other):
 757                 if isinstance(other, ABCTimedeltaIndex):
 758                     # Defer to TimedeltaIndex implementation
 759                     return NotImplemented
 760                 elif isinstance(other, (timedelta, np.timedelta64)):
 761                     # GH#19333 is_integer evaluated True on timedelta64,
 762                     # so we need to catch these explicitly
 763                     return op(self._int64index, other)
 764                 elif is_timedelta64_dtype(other):
 765                     # Must be an np.ndarray; GH#22390
 766                     return op(self._int64index, other)
 767 
 768                 other = extract_array(other, extract_numpy=True)
 769                 attrs = self._get_attributes_dict()
 770 
 771                 left, right = self, other
 772 
 773                 try:
 774                     # apply if we have an override
 775                     if step:
 776                         with np.errstate(all="ignore"):
 777                             rstep = step(left.step, right)
 778 
 779                         # we don't have a representable op
 780                         # so return a base index
 781                         if not is_integer(rstep) or not rstep:
 782                             raise ValueError
 783 
 784                     else:
 785                         rstep = left.step
 786 
 787                     with np.errstate(all="ignore"):
 788                         rstart = op(left.start, right)
 789                         rstop = op(left.stop, right)
 790 
 791                     result = type(self)(rstart, rstop, rstep, **attrs)
 792 
 793                     # for compat with numpy / Int64Index
 794                     # even if we can represent as a RangeIndex, return
 795                     # as a Float64Index if we have float-like descriptors
 796                     if not all(is_integer(x) for x in [rstart, rstop, rstep]):
 797                         result = result.astype("float64")
 798 
 799                     return result
 800 
 801                 except (ValueError, TypeError, ZeroDivisionError):
 802                     # Defer to Int64Index implementation
 803                     return op(self._int64index, other)
 804                     # TODO: Do attrs get handled reliably?
 805 
 806             name = f"__{op.__name__}__"
 807             return compat.set_function_name(_evaluate_numeric_binop, name, cls)
 808 
 809         cls.__add__ = _make_evaluate_binop(operator.add)
 810         cls.__radd__ = _make_evaluate_binop(ops.radd)
 811         cls.__sub__ = _make_evaluate_binop(operator.sub)
 812         cls.__rsub__ = _make_evaluate_binop(ops.rsub)
 813         cls.__mul__ = _make_evaluate_binop(operator.mul, step=operator.mul)
 814         cls.__rmul__ = _make_evaluate_binop(ops.rmul, step=ops.rmul)
 815         cls.__truediv__ = _make_evaluate_binop(operator.truediv, step=operator.truediv)
 816         cls.__rtruediv__ = _make_evaluate_binop(ops.rtruediv, step=ops.rtruediv)
