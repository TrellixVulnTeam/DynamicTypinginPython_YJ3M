  53 class TimedeltaIndex(DatetimeTimedeltaMixin):
  54     """
  55     Immutable ndarray of timedelta64 data, represented internally as int64, and
  56     which can be boxed to timedelta objects.
  57 
  58     Parameters
  59     ----------
  60     data  : array-like (1-dimensional), optional
  61         Optional timedelta-like data to construct index with.
  62     unit : unit of the arg (D,h,m,s,ms,us,ns) denote the unit, optional
  63         Which is an integer/float number.
  64     freq : str or pandas offset object, optional
  65         One of pandas date offset strings or corresponding objects. The string
  66         'infer' can be passed in order to set the frequency of the index as the
  67         inferred frequency upon creation.
  68     copy  : bool
  69         Make a copy of input ndarray.
  70     name : object
  71         Name to be stored in the index.
  72 
  73     Attributes
  74     ----------
  75     days
  76     seconds
  77     microseconds
  78     nanoseconds
  79     components
  80     inferred_freq
  81 
  82     Methods
  83     -------
  84     to_pytimedelta
  85     to_series
  86     round
  87     floor
  88     ceil
  89     to_frame
  90     mean
  91 
  92     See Also
  93     --------
  94     Index : The base pandas Index type.
  95     Timedelta : Represents a duration between two dates or times.
  96     DatetimeIndex : Index of datetime64 data.
  97     PeriodIndex : Index of Period data.
  98     timedelta_range : Create a fixed-frequency TimedeltaIndex.
  99 
 100     Notes
 101     -----
 102     To learn more about the frequency strings, please see `this link
 103     <https://pandas.pydata.org/pandas-docs/stable/user_guide/timeseries.html#offset-aliases>`__.
 104     """

 119     def __new__(
 120         cls,
 121         data=None,
 122         unit=None,
 123         freq=lib.no_default,
 124         closed=None,
 125         dtype=TD64NS_DTYPE,
 126         copy=False,
 127         name=None,
 128     ):
 129         name = maybe_extract_name(name, data, cls)
 130 
 131         if is_scalar(data):
 132             raise TypeError(
 133                 f"{cls.__name__}() must be called with a "
 134                 f"collection of some kind, {repr(data)} was passed"
 135             )
 136 
 137         if unit in {"Y", "y", "M"}:
 138             raise ValueError(
 139                 "Units 'M' and 'Y' are no longer supported, as they do not "
 140                 "represent unambiguous timedelta values durations."
 141             )
 142 
 143         if isinstance(data, TimedeltaArray) and freq is lib.no_default:
 144             if copy:
 145                 data = data.copy()
 146             return cls._simple_new(data, name=name)
 147 
 148         if isinstance(data, TimedeltaIndex) and freq is lib.no_default and name is None:
 149             if copy:
 150                 return data.copy()
 151             else:
 152                 return data._shallow_copy()
 153 
 154         # - Cases checked above all return/raise before reaching here - #
 155 
 156         tdarr = TimedeltaArray._from_sequence(
 157             data, freq=freq, unit=unit, dtype=dtype, copy=copy
 158         )
 159         return cls._simple_new(tdarr, name=name)

 161     @classmethod
 162     def _simple_new(cls, values: TimedeltaArray, name: Label = None):
 163         assert isinstance(values, TimedeltaArray)
 164 
 165         result = object.__new__(cls)
 166         result._data = values
 167         result._name = name
 168         result._cache = {}
 169         # For groupby perf. See note in indexes/base about _index_data
 170         result._index_data = values._data
 171 
 172         result._reset_identity()
 173         return result

 186     @doc(Index.astype)
 187     def astype(self, dtype, copy=True):
 188         dtype = pandas_dtype(dtype)
 189         if is_timedelta64_dtype(dtype) and not is_timedelta64_ns_dtype(dtype):
 190             # Have to repeat the check for 'timedelta64' (not ns) dtype
 191             #  so that we can return a numeric index, since pandas will return
 192             #  a TimedeltaIndex when dtype='timedelta'
 193             result = self._data.astype(dtype, copy=copy)
 194             if self.hasnans:
 195                 return Index(result, name=self.name)
 196             return Index(result.astype("i8"), name=self.name)
 197         return DatetimeIndexOpsMixin.astype(self, dtype, copy=copy)

 199     def _is_comparable_dtype(self, dtype: DtypeObj) -> bool:
 200         """
 201         Can we compare values of the given dtype to our own?
 202         """
 203         return is_timedelta64_dtype(dtype)

 205     def get_loc(self, key, method=None, tolerance=None):
 206         """
 207         Get integer location for requested label
 208 
 209         Returns
 210         -------
 211         loc : int, slice, or ndarray[int]
 212         """
 213         if not is_scalar(key):
 214             raise InvalidIndexError(key)
 215 
 216         msg = str(key)
 217         try:
 218             key = self._data._validate_scalar(key, msg, cast_str=True)
 219         except TypeError as err:
 220             raise KeyError(key) from err
 221 
 222         return Index.get_loc(self, key, method, tolerance)
 223 
 224     def _maybe_cast_slice_bound(self, label, side: str, kind):
 225         """
 226         If label is a string, cast it to timedelta according to resolution.
 227 
 228         Parameters
 229         ----------
 230         label : object
 231         side : {'left', 'right'}
 232         kind : {'loc', 'getitem'} or None
 233 
 234         Returns
 235         -------
 236         label : object
 237         """
 238         assert kind in ["loc", "getitem", None]
 239 
 240         if isinstance(label, str):
 241             parsed = Timedelta(label)
 242             lbound = parsed.round(parsed.resolution_string)
 243             if side == "left":
 244                 return lbound
 245             else:
 246                 return lbound + to_offset(parsed.resolution_string) - Timedelta(1, "ns")
 247         elif is_integer(label) or is_float(label):
 248             self._invalid_indexer("slice", label)
 249 
 250         return label
 251 
 252     def is_type_compatible(self, typ) -> bool:
 253         return typ == self.inferred_type or typ == "timedelta"

 263 def timedelta_range(
 264     start=None, end=None, periods=None, freq=None, name=None, closed=None
 265 ) -> TimedeltaIndex:
 266     """
 267     Return a fixed frequency TimedeltaIndex, with day as the default
 268     frequency.
 269 
 270     Parameters
 271     ----------
 272     start : str or timedelta-like, default None
 273         Left bound for generating timedeltas.
 274     end : str or timedelta-like, default None
 275         Right bound for generating timedeltas.
 276     periods : int, default None
 277         Number of periods to generate.
 278     freq : str or DateOffset, default 'D'
 279         Frequency strings can have multiples, e.g. '5H'.
 280     name : str, default None
 281         Name of the resulting TimedeltaIndex.
 282     closed : str, default None
 283         Make the interval closed with respect to the given frequency to
 284         the 'left', 'right', or both sides (None).
 285 
 286     Returns
 287     -------
 288     rng : TimedeltaIndex
 289 
 290     Notes
 291     -----
 292     Of the four parameters ``start``, ``end``, ``periods``, and ``freq``,
 293     exactly three must be specified. If ``freq`` is omitted, the resulting
 294     ``TimedeltaIndex`` will have ``periods`` linearly spaced elements between
 295     ``start`` and ``end`` (closed on both sides).
 296 
 297     To learn more about the frequency strings, please see `this link
 298     <https://pandas.pydata.org/pandas-docs/stable/user_guide/timeseries.html#offset-aliases>`__.
 299 
 300     Examples
 301     --------
 302     >>> pd.timedelta_range(start='1 day', periods=4)
 303     TimedeltaIndex(['1 days', '2 days', '3 days', '4 days'],
 304                    dtype='timedelta64[ns]', freq='D')
 305 
 306     The ``closed`` parameter specifies which endpoint is included.  The default
 307     behavior is to include both endpoints.
 308 
 309     >>> pd.timedelta_range(start='1 day', periods=4, closed='right')
 310     TimedeltaIndex(['2 days', '3 days', '4 days'],
 311                    dtype='timedelta64[ns]', freq='D')
 312 
 313     The ``freq`` parameter specifies the frequency of the TimedeltaIndex.
 314     Only fixed frequencies can be passed, non-fixed frequencies such as
 315     'M' (month end) will raise.
 316 
 317     >>> pd.timedelta_range(start='1 day', end='2 days', freq='6H')
 318     TimedeltaIndex(['1 days 00:00:00', '1 days 06:00:00', '1 days 12:00:00',
 319                     '1 days 18:00:00', '2 days 00:00:00'],
 320                    dtype='timedelta64[ns]', freq='6H')
 321 
 322     Specify ``start``, ``end``, and ``periods``; the frequency is generated
 323     automatically (linearly spaced).
 324 
 325     >>> pd.timedelta_range(start='1 day', end='5 days', periods=4)
 326     TimedeltaIndex(['1 days 00:00:00', '2 days 08:00:00', '3 days 16:00:00',
 327                 '5 days 00:00:00'],
 328                dtype='timedelta64[ns]', freq='32H')
 329     """
 330     if freq is None and com.any_none(periods, start, end):
 331         freq = "D"
 332 
 333     freq, _ = dtl.maybe_infer_freq(freq)
 334     tdarr = TimedeltaArray._generate_range(start, end, periods, freq, closed=closed)
 335     return TimedeltaIndex._simple_new(tdarr, name=name)
