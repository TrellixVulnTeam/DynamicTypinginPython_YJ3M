  54 def _new_PeriodIndex(cls, **d):
  55     # GH13277 for unpickling
  56     values = d.pop("data")
  57     if values.dtype == "int64":
  58         freq = d.pop("freq", None)
  59         values = PeriodArray(values, freq=freq)

 164     def __new__(
 165         cls,
 166         data=None,
 167         ordinal=None,
 168         freq=None,
 169         tz=None,
 170         dtype=None,
 171         copy=False,
 172         name=None,
 173         **fields,
 174     ):
 175 
 176         valid_field_set = {
 177             "year",
 178             "month",
 179             "day",
 180             "quarter",
 181             "hour",
 182             "minute",
 183             "second",
 184         }
 185 
 186         if not set(fields).issubset(valid_field_set):
 187             argument = list(set(fields) - valid_field_set)[0]
 188             raise TypeError(f"__new__() got an unexpected keyword argument {argument}")
 189 
 190         name = maybe_extract_name(name, data, cls)
 191 
 192         if data is None and ordinal is None:
 193             # range-based.
 194             data, freq2 = PeriodArray._generate_range(None, None, None, freq, fields)
 195             # PeriodArray._generate range does validation that fields is
 196             # empty when really using the range-based constructor.
 197             freq = freq2
 198 
 199             data = PeriodArray(data, freq=freq)
 200         else:
 201             freq = validate_dtype_freq(dtype, freq)
 202 
 203             # PeriodIndex allow PeriodIndex(period_index, freq=different)
 204             # Let's not encourage that kind of behavior in PeriodArray.
 205 
 206             if freq and isinstance(data, cls) and data.freq != freq:
 207                 # TODO: We can do some of these with no-copy / coercion?
 208                 # e.g. D -> 2D seems to be OK
 209                 data = data.asfreq(freq)
 210 
 211             if data is None and ordinal is not None:
 212                 # we strangely ignore `ordinal` if data is passed.
 213                 ordinal = np.asarray(ordinal, dtype=np.int64)
 214                 data = PeriodArray(ordinal, freq)
 215             else:
 216                 # don't pass copy here, since we copy later.
 217                 data = period_array(data=data, freq=freq)
 218 
 219         if copy:
 220             data = data.copy()
 221 
 222         return cls._simple_new(data, name=name)

 224     @classmethod
 225     def _simple_new(cls, values: PeriodArray, name: Label = None):
 226         """
 227         Create a new PeriodIndex.
 228 
 229         Parameters
 230         ----------
 231         values : PeriodArray
 232             Values that can be converted to a PeriodArray without inference
 233             or coercion.
 234         """
 235         assert isinstance(values, PeriodArray), type(values)
 236 
 237         result = object.__new__(cls)
 238         result._data = values
 239         # For groupby perf. See note in indexes/base about _index_data
 240         result._index_data = values._data
 241         result.name = name
 242         result._cache = {}
 243         result._reset_identity()
 244         return result

 268     def _maybe_convert_timedelta(self, other):
 269         """
 270         Convert timedelta-like input to an integer multiple of self.freq
 271 
 272         Parameters
 273         ----------
 274         other : timedelta, np.timedelta64, DateOffset, int, np.ndarray
 275 
 276         Returns
 277         -------
 278         converted : int, np.ndarray[int64]
 279 
 280         Raises
 281         ------
 282         IncompatibleFrequency : if the input cannot be written as a multiple
 283             of self.freq.  Note IncompatibleFrequency subclasses ValueError.
 284         """
 285         if isinstance(other, (timedelta, np.timedelta64, Tick, np.ndarray)):
 286             if isinstance(self.freq, Tick):
 287                 # _check_timedeltalike_freq_compat will raise if incompatible
 288                 delta = self._data._check_timedeltalike_freq_compat(other)
 289                 return delta
 290         elif isinstance(other, DateOffset):
 291             if other.base == self.freq.base:
 292                 return other.n
 293 
 294             raise raise_on_incompatible(self, other)
 295         elif is_integer(other):
 296             # integer is passed to .shift via
 297             # _add_datetimelike_methods basically
 298             # but ufunc may pass integer to _add_delta
 299             return other
 300 
 301         # raise when input doesn't have freq
 302         raise raise_on_incompatible(self, None)

 304     def _is_comparable_dtype(self, dtype: DtypeObj) -> bool:
 305         """
 306         Can we compare values of the given dtype to our own?
 307         """
 308         if not isinstance(dtype, PeriodDtype):
 309             return False
 310         return dtype.freq == self.freq

 326     @doc(Index.__contains__)
 327     def __contains__(self, key: Any) -> bool:
 328         if isinstance(key, Period):
 329             if key.freq != self.freq:
 330                 return False
 331             else:
 332                 return key.ordinal in self._engine
 333         else:
 334             hash(key)
 335             try:
 336                 self.get_loc(key)
 337                 return True
 338             except KeyError:

 348     def __array_wrap__(self, result, context=None):
 349         """
 350         Gets called after a ufunc. Needs additional handling as
 351         PeriodIndex stores internal data as int dtype
 352 
 353         Replace this to __numpy_ufunc__ in future version
 354         """
 355         if isinstance(context, tuple) and len(context) > 0:
 356             func = context[0]
 357             if func is np.add:
 358                 pass
 359             elif func is np.subtract:
 360                 name = self.name
 361                 left = context[1][0]
 362                 right = context[1][1]
 363                 if isinstance(left, PeriodIndex) and isinstance(right, PeriodIndex):
 364                     name = left.name if left.name == right.name else None
 365                     return Index(result, name=name)
 366                 elif isinstance(left, Period) or isinstance(right, Period):
 367                     return Index(result, name=name)
 368             elif isinstance(func, np.ufunc):
 369                 if "M->M" not in func.types:
 370                     msg = f"ufunc '{func.__name__}' not supported for the PeriodIndex"
 371                     # This should be TypeError, but TypeError cannot be raised
 372                     # from here because numpy catches.
 373                     raise ValueError(msg)
 374 
 375         if is_bool_dtype(result):
 376             return result
 377         # the result is object dtype array of Period
 378         # cannot pass _simple_new as it is
 379         return type(self)(result, freq=self.freq, name=self.name)

 381     def asof_locs(self, where, mask: np.ndarray) -> np.ndarray:
 382         """
 383         where : array of timestamps
 384         mask : array of booleans where data is not NA
 385         """
 386         where_idx = where
 387         if isinstance(where_idx, DatetimeIndex):
 388             where_idx = PeriodIndex(where_idx._values, freq=self.freq)
 389         elif not isinstance(where_idx, PeriodIndex):
 390             raise TypeError("asof_locs `where` must be DatetimeIndex or PeriodIndex")
 391         elif where_idx.freq != self.freq:
 392             raise raise_on_incompatible(self, where_idx)
 393 
 394         locs = self.asi8[mask].searchsorted(where_idx.asi8, side="right")
 395 
 396         locs = np.where(locs > 0, locs - 1, 0)
 397         result = np.arange(len(self))[mask].take(locs)
 398 
 399         first = mask.argmax()
 400         result[(locs == 0) & (where_idx.asi8 < self.asi8[first])] = -1
 401 
 402         return result

 416     @property
 417     def is_full(self) -> bool:
 418         """
 419         Returns True if this PeriodIndex is range-like in that all Periods
 420         between start and end are present, in order.
 421         """
 422         if len(self) == 0:
 423             return True
 424         if not self.is_monotonic:
 425             raise ValueError("Index is not monotonic")
 426         values = self.asi8
 427         return ((values[1:] - values[:-1]) < 2).all()

 435     @Appender(_index_shared_docs["get_indexer"] % _index_doc_kwargs)
 436     def get_indexer(self, target, method=None, limit=None, tolerance=None):
 437         target = ensure_index(target)
 438 
 439         if isinstance(target, PeriodIndex):
 440             if target.freq != self.freq:
 441                 # No matches
 442                 no_matches = -1 * np.ones(self.shape, dtype=np.intp)
 443                 return no_matches
 444 
 445             target = target.asi8
 446             self_index = self._int64index
 447         else:
 448             self_index = self
 449 
 450         if tolerance is not None:
 451             tolerance = self._convert_tolerance(tolerance, target)
 452             if self_index is not self:
 453                 # convert tolerance to i8
 454                 tolerance = self._maybe_convert_timedelta(tolerance)
 455 
 456         return Index.get_indexer(self_index, target, method, limit, tolerance)

 471     def get_loc(self, key, method=None, tolerance=None):
 472         """
 473         Get integer location for requested label.
 474 
 475         Parameters
 476         ----------
 477         key : Period, NaT, str, or datetime
 478             String or datetime key must be parsable as Period.
 479 
 480         Returns
 481         -------
 482         loc : int or ndarray[int64]
 483 
 484         Raises
 485         ------
 486         KeyError
 487             Key is not present in the index.
 488         TypeError
 489             If key is listlike or otherwise not hashable.
 490         """
 491         orig_key = key
 492 
 493         if not is_scalar(key):
 494             raise InvalidIndexError(key)
 495 
 496         if isinstance(key, str):
 497 
 498             try:
 499                 loc = self._get_string_slice(key)
 500                 return loc
 501             except (TypeError, ValueError):
 502                 pass
 503 
 504             try:
 505                 asdt, reso = parse_time_string(key, self.freq)
 506             except DateParseError as err:
 507                 # A string with invalid format
 508                 raise KeyError(f"Cannot interpret '{key}' as period") from err
 509 
 510             reso = Resolution.from_attrname(reso)
 511             grp = reso.freq_group
 512             freqn = self.dtype.freq_group
 513 
 514             # _get_string_slice will handle cases where grp < freqn
 515             assert grp >= freqn
 516 
 517             # BusinessDay is a bit strange. It has a *lower* code, but we never parse
 518             # a string as "BusinessDay" resolution, just Day.
 519             if grp == freqn or (
 520                 reso == Resolution.RESO_DAY and self.dtype.freq.name == "B"
 521             ):
 522                 key = Period(asdt, freq=self.freq)
 523                 loc = self.get_loc(key, method=method, tolerance=tolerance)
 524                 return loc
 525             elif method is None:
 526                 raise KeyError(key)
 527             else:
 528                 key = asdt
 529 
 530         elif is_integer(key):
 531             # Period constructor will cast to string, which we dont want
 532             raise KeyError(key)
 533 
 534         try:
 535             key = Period(key, freq=self.freq)
 536         except ValueError as err:
 537             # we cannot construct the Period
 538             raise KeyError(orig_key) from err
 539 
 540         try:
 541             return Index.get_loc(self, key, method, tolerance)
 542         except KeyError as err:
 543             raise KeyError(orig_key) from err

 545     def _maybe_cast_slice_bound(self, label, side: str, kind: str):
 546         """
 547         If label is a string or a datetime, cast it to Period.ordinal according
 548         to resolution.
 549 
 550         Parameters
 551         ----------
 552         label : object
 553         side : {'left', 'right'}
 554         kind : {'loc', 'getitem'}
 555 
 556         Returns
 557         -------
 558         bound : Period or object
 559 
 560         Notes
 561         -----
 562         Value of `side` parameter should be validated in caller.
 563 
 564         """
 565         assert kind in ["loc", "getitem"]
 566 
 567         if isinstance(label, datetime):
 568             return Period(label, freq=self.freq)
 569         elif isinstance(label, str):
 570             try:
 571                 parsed, reso = parse_time_string(label, self.freq)
 572                 reso = Resolution.from_attrname(reso)
 573                 bounds = self._parsed_string_to_bounds(reso, parsed)
 574                 return bounds[0 if side == "left" else 1]
 575             except ValueError as err:
 576                 # string cannot be parsed as datetime-like
 577                 # TODO: we need tests for this case
 578                 raise KeyError(label) from err
 579         elif is_integer(label) or is_float(label):
 580             self._invalid_indexer("slice", label)
 581 
 582         return label

 609     def insert(self, loc, item):
 610         if not isinstance(item, Period) or self.freq != item.freq:
 611             return self.astype(object).insert(loc, item)
 612 
 613         i8result = np.concatenate(
 614             (self[:loc].asi8, np.array([item.ordinal]), self[loc:].asi8)
 615         )
 616         arr = type(self._data)._simple_new(i8result, dtype=self.dtype)
 617         return type(self)._simple_new(arr, name=self.name)

 619     def join(self, other, how="left", level=None, return_indexers=False, sort=False):
 620         """
 621         See Index.join
 622         """
 623         self._assert_can_do_setop(other)
 624 
 625         if not isinstance(other, PeriodIndex):
 626             return self.astype(object).join(
 627                 other, how=how, level=level, return_indexers=return_indexers, sort=sort
 628             )
 629 
 630         # _assert_can_do_setop ensures we have matching dtype
 631         result = Int64Index.join(
 632             self,
 633             other,
 634             how=how,
 635             level=level,
 636             return_indexers=return_indexers,
 637             sort=sort,
 638         )
 639         return result

 644     def _assert_can_do_setop(self, other):
 645         super()._assert_can_do_setop(other)
 646 
 647         # *Can't* use PeriodIndexes of different freqs

 652     def _setop(self, other, sort, opname: str):
 653         """
 654         Perform a set operation by dispatching to the Int64Index implementation.
 655         """
 656         self._validate_sort_keyword(sort)
 657         self._assert_can_do_setop(other)
 658         res_name = get_op_result_name(self, other)
 659         other = ensure_index(other)
 660 
 661         i8self = Int64Index._simple_new(self.asi8)
 662         i8other = Int64Index._simple_new(other.asi8)
 663         i8result = getattr(i8self, opname)(i8other, sort=sort)
 664 
 665         parr = type(self._data)(np.asarray(i8result, dtype=np.int64), dtype=self.dtype)
 666         result = type(self)._simple_new(parr, name=res_name)
 667         return result

 669     def intersection(self, other, sort=False):
 670         self._validate_sort_keyword(sort)
 671         self._assert_can_do_setop(other)
 672         other = ensure_index(other)
 673 
 674         if self.equals(other):
 675             return self._get_reconciled_name_object(other)
 676 
 677         elif is_object_dtype(other.dtype):
 678             return self.astype("O").intersection(other, sort=sort)
 679 
 680         elif not is_dtype_equal(self.dtype, other.dtype):
 681             # We can infer that the intersection is empty.
 682             # assert_can_do_setop ensures that this is not just a mismatched freq
 683             this = self[:0].astype("O")
 684             other = other[:0].astype("O")
 685             return this.intersection(other, sort=sort)
 686 
 687         return self._setop(other, sort, opname="intersection")

 689     def difference(self, other, sort=None):
 690         self._validate_sort_keyword(sort)
 691         self._assert_can_do_setop(other)
 692         other = ensure_index(other)
 693 
 694         if self.equals(other):
 695             # pass an empty PeriodArray with the appropriate dtype
 696             return type(self)._simple_new(self._data[:0], name=self.name)
 697 
 698         if is_object_dtype(other):
 699             return self.astype(object).difference(other).astype(self.dtype)
 700 
 701         elif not is_dtype_equal(self.dtype, other.dtype):
 702             return self
 703 
 704         return self._setop(other, sort, opname="difference")
 705 
 706     def _union(self, other, sort):
 707         if not len(other) or self.equals(other) or not len(self):
 708             return super()._union(other, sort=sort)
 709 
 710         # We are called by `union`, which is responsible for this validation
 711         assert isinstance(other, type(self))
 712 
 713         if not is_dtype_equal(self.dtype, other.dtype):
 714             this = self.astype("O")
 715             other = other.astype("O")
 716             return this._union(other, sort=sort)
 717 
 718         return self._setop(other, sort, opname="_union")
 719 
 720     # ------------------------------------------------------------------------
 721 
 722     def memory_usage(self, deep=False):
 723         result = super().memory_usage(deep=deep)
 724         if hasattr(self, "_cache") and "_int64index" in self._cache:
 725             result += self._int64index.memory_usage(deep=deep)
 726         return result

 733 def period_range(
 734     start=None, end=None, periods=None, freq=None, name=None
 735 ) -> PeriodIndex:
 736     """
 737     Return a fixed frequency PeriodIndex.
 738 
 739     The day (calendar) is the default frequency.
 740 
 741     Parameters
 742     ----------
 743     start : str or period-like, default None
 744         Left bound for generating periods.
 745     end : str or period-like, default None
 746         Right bound for generating periods.
 747     periods : int, default None
 748         Number of periods to generate.
 749     freq : str or DateOffset, optional
 750         Frequency alias. By default the freq is taken from `start` or `end`
 751         if those are Period objects. Otherwise, the default is ``"D"`` for
 752         daily frequency.
 753     name : str, default None
 754         Name of the resulting PeriodIndex.
 755 
 756     Returns
 757     -------
 758     PeriodIndex
 759 
 760     Notes
 761     -----
 762     Of the three parameters: ``start``, ``end``, and ``periods``, exactly two
 763     must be specified.
 764 
 765     To learn more about the frequency strings, please see `this link
 766     <https://pandas.pydata.org/pandas-docs/stable/user_guide/timeseries.html#offset-aliases>`__.
 767 
 768     Examples
 769     --------
 770     >>> pd.period_range(start='2017-01-01', end='2018-01-01', freq='M')
 771     PeriodIndex(['2017-01', '2017-02', '2017-03', '2017-04', '2017-05', '2017-06',
 772              '2017-07', '2017-08', '2017-09', '2017-10', '2017-11', '2017-12',
 773              '2018-01'],
 774             dtype='period[M]', freq='M')
 775 
 776     If ``start`` or ``end`` are ``Period`` objects, they will be used as anchor
 777     endpoints for a ``PeriodIndex`` with frequency matching that of the
 778     ``period_range`` constructor.
 779 
 780     >>> pd.period_range(start=pd.Period('2017Q1', freq='Q'),
 781     ...                 end=pd.Period('2017Q2', freq='Q'), freq='M')
 782     PeriodIndex(['2017-03', '2017-04', '2017-05', '2017-06'],
 783                 dtype='period[M]', freq='M')
 784     """
 785     if com.count_not_none(start, end, periods) != 2:
 786         raise ValueError(
 787             "Of the three parameters: start, end, and periods, "
 788             "exactly two must be specified"
 789         )
 790     if freq is None and (not isinstance(start, Period) and not isinstance(end, Period)):
 791         freq = "D"
 792 
 793     data, freq = PeriodArray._generate_range(start, end, periods, freq, fields={})
 794     data = PeriodArray(data, freq=freq)
 795     return PeriodIndex(data, name=name)
