  26 def mask_missing(arr, values_to_mask):
  27     """
  28     Return a masking array of same size/shape as arr
  29     with entries equaling any member of values_to_mask set to True
  30     """
  31     dtype, values_to_mask = infer_dtype_from_array(values_to_mask)
  32 
  33     try:
  34         values_to_mask = np.array(values_to_mask, dtype=dtype)
  35 
  36     except Exception:
  37         values_to_mask = np.array(values_to_mask, dtype=object)
  38 
  39     na_mask = isna(values_to_mask)
  40     nonna = values_to_mask[~na_mask]
  41 
  42     mask = None
  43     for x in nonna:
  44         if mask is None:
  45             if is_numeric_v_string_like(arr, x):
  46                 # GH#29553 prevent numpy deprecation warnings
  47                 mask = False
  48             else:
  49                 mask = arr == x
  50 
  51             # if x is a string and arr is not, then we get False and we must
  52             # expand the mask to size arr.shape
  53             if is_scalar(mask):
  54                 mask = np.zeros(arr.shape, dtype=bool)
  55         else:
  56             if is_numeric_v_string_like(arr, x):
  57                 # GH#29553 prevent numpy deprecation warnings
  58                 mask |= False
  59             else:
  60                 mask |= arr == x
  61 
  62     if na_mask.any():
  63         if mask is None:
  64             mask = isna(arr)
  65         else:
  66             mask |= isna(arr)
  67 
  68     # GH 21977
  69     if mask is None:
  70         mask = np.zeros(arr.shape, dtype=bool)
  71 
  72     return mask

  75 def clean_fill_method(method, allow_nearest=False):
  76     # asfreq is compat for resampling
  77     if method in [None, "asfreq"]:
  78         return None
  79 
  80     if isinstance(method, str):
  81         method = method.lower()
  82         if method == "ffill":
  83             method = "pad"
  84         elif method == "bfill":
  85             method = "backfill"
  86 
  87     valid_methods = ["pad", "backfill"]
  88     expecting = "pad (ffill) or backfill (bfill)"
  89     if allow_nearest:
  90         valid_methods.append("nearest")
  91         expecting = "pad (ffill), backfill (bfill) or nearest"
  92     if method not in valid_methods:
  93         raise ValueError(f"Invalid fill method. Expecting {expecting}. Got {method}")
  94     return method

 121 def clean_interp_method(method: str, **kwargs) -> str:
 122     order = kwargs.get("order")
 123 
 124     if method in ("spline", "polynomial") and order is None:
 125         raise ValueError("You must specify the order of the spline or polynomial.")
 126 
 127     valid = NP_METHODS + SP_METHODS
 128     if method not in valid:
 129         raise ValueError(f"method must be one of {valid}. Got '{method}' instead.")
 130 
 131     return method

 134 def find_valid_index(values, how: str):
 135     """
 136     Retrieves the index of the first valid value.
 137 
 138     Parameters
 139     ----------
 140     values : ndarray or ExtensionArray
 141     how : {'first', 'last'}
 142         Use this parameter to change between the first or last valid index.
 143 
 144     Returns
 145     -------
 146     int or None
 147     """
 148     assert how in ["first", "last"]
 149 
 150     if len(values) == 0:  # early stop
 151         return None
 152 
 153     is_valid = ~isna(values)
 154 
 155     if values.ndim == 2:
 156         is_valid = is_valid.any(1)  # reduce axis 1
 157 
 158     if how == "first":
 159         idxpos = is_valid[::].argmax()
 160 
 161     if how == "last":
 162         idxpos = len(values) - 1 - is_valid[::-1].argmax()
 163 
 164     chk_notna = is_valid[idxpos]
 165 
 166     if not chk_notna:
 167         return None
 168     return idxpos

 171 def interpolate_1d(
 172     xvalues: np.ndarray,
 173     yvalues: np.ndarray,
 174     method: Optional[str] = "linear",
 175     limit: Optional[int] = None,
 176     limit_direction: str = "forward",
 177     limit_area: Optional[str] = None,
 178     fill_value: Optional[Any] = None,
 179     bounds_error: bool = False,
 180     order: Optional[int] = None,
 181     **kwargs,
 182 ):
 183     """
 184     Logic for the 1-d interpolation.  The result should be 1-d, inputs
 185     xvalues and yvalues will each be 1-d arrays of the same length.
 186 
 187     Bounds_error is currently hardcoded to False since non-scipy ones don't
 188     take it as an argument.
 189     """
 190     invalid = isna(yvalues)
 191     valid = ~invalid
 192 
 193     if not valid.any():
 194         # have to call np.asarray(xvalues) since xvalues could be an Index
 195         # which can't be mutated
 196         result = np.empty_like(np.asarray(xvalues), dtype=np.float64)
 197         result.fill(np.nan)
 198         return result
 199 
 200     if valid.all():
 201         return yvalues
 202 
 203     if method == "time":
 204         if not getattr(xvalues, "is_all_dates", None):
 205             # if not issubclass(xvalues.dtype.type, np.datetime64):
 206             raise ValueError(
 207                 "time-weighted interpolation only works "
 208                 "on Series or DataFrames with a "
 209                 "DatetimeIndex"
 210             )
 211         method = "values"
 212 
 213     valid_limit_directions = ["forward", "backward", "both"]
 214     limit_direction = limit_direction.lower()
 215     if limit_direction not in valid_limit_directions:
 216         raise ValueError(
 217             "Invalid limit_direction: expecting one of "
 218             f"{valid_limit_directions}, got '{limit_direction}'."
 219         )
 220 
 221     if limit_area is not None:
 222         valid_limit_areas = ["inside", "outside"]
 223         limit_area = limit_area.lower()
 224         if limit_area not in valid_limit_areas:
 225             raise ValueError(
 226                 f"Invalid limit_area: expecting one of {valid_limit_areas}, got "
 227                 f"{limit_area}."
 228             )
 229 
 230     # default limit is unlimited GH #16282
 231     limit = algos._validate_limit(nobs=None, limit=limit)
 232 
 233     # These are sets of index pointers to invalid values... i.e. {0, 1, etc...
 234     all_nans = set(np.flatnonzero(invalid))
 235     start_nans = set(range(find_valid_index(yvalues, "first")))
 236     end_nans = set(range(1 + find_valid_index(yvalues, "last"), len(valid)))
 237     mid_nans = all_nans - start_nans - end_nans
 238 
 239     # Like the sets above, preserve_nans contains indices of invalid values,
 240     # but in this case, it is the final set of indices that need to be
 241     # preserved as NaN after the interpolation.
 242 
 243     # For example if limit_direction='forward' then preserve_nans will
 244     # contain indices of NaNs at the beginning of the series, and NaNs that
 245     # are more than'limit' away from the prior non-NaN.
 246 
 247     # set preserve_nans based on direction using _interp_limit
 248     preserve_nans: Union[List, Set]
 249     if limit_direction == "forward":
 250         preserve_nans = start_nans | set(_interp_limit(invalid, limit, 0))
 251     elif limit_direction == "backward":
 252         preserve_nans = end_nans | set(_interp_limit(invalid, 0, limit))
 253     else:
 254         # both directions... just use _interp_limit
 255         preserve_nans = set(_interp_limit(invalid, limit, limit))
 256 
 257     # if limit_area is set, add either mid or outside indices
 258     # to preserve_nans GH #16284
 259     if limit_area == "inside":
 260         # preserve NaNs on the outside
 261         preserve_nans |= start_nans | end_nans
 262     elif limit_area == "outside":
 263         # preserve NaNs on the inside
 264         preserve_nans |= mid_nans
 265 
 266     # sort preserve_nans and covert to list
 267     preserve_nans = sorted(preserve_nans)
 268 
 269     yvalues = getattr(yvalues, "values", yvalues)
 270     result = yvalues.copy()
 271 
 272     # xvalues to pass to NumPy/SciPy
 273 
 274     xvalues = getattr(xvalues, "values", xvalues)
 275     if method == "linear":
 276         inds = xvalues
 277     else:
 278         inds = np.asarray(xvalues)
 279 
 280         # hack for DatetimeIndex, #1646
 281         if needs_i8_conversion(inds.dtype):
 282             inds = inds.view(np.int64)
 283 
 284         if method in ("values", "index"):
 285             if inds.dtype == np.object_:
 286                 inds = lib.maybe_convert_objects(inds)
 287 
 288     if method in NP_METHODS:
 289         # np.interp requires sorted X values, #21037
 290         indexer = np.argsort(inds[valid])
 291         result[invalid] = np.interp(
 292             inds[invalid], inds[valid][indexer], yvalues[valid][indexer]
 293         )
 294     else:
 295         result[invalid] = _interpolate_scipy_wrapper(
 296             inds[valid],
 297             yvalues[valid],
 298             inds[invalid],
 299             method=method,
 300             fill_value=fill_value,
 301             bounds_error=bounds_error,
 302             order=order,
 303             **kwargs,
 304         )
 305 
 306     result[preserve_nans] = np.nan
 307     return result

 310 def _interpolate_scipy_wrapper(
 311     x, y, new_x, method, fill_value=None, bounds_error=False, order=None, **kwargs
 312 ):
 313     """
 314     Passed off to scipy.interpolate.interp1d. method is scipy's kind.
 315     Returns an array interpolated at new_x.  Add any new methods to
 316     the list in _clean_interp_method.
 317     """
 318     extra = f"{method} interpolation requires SciPy."
 319     import_optional_dependency("scipy", extra=extra)
 320     from scipy import interpolate
 321 
 322     new_x = np.asarray(new_x)
 323 
 324     # ignores some kwargs that could be passed along.
 325     alt_methods = {
 326         "barycentric": interpolate.barycentric_interpolate,
 327         "krogh": interpolate.krogh_interpolate,
 328         "from_derivatives": _from_derivatives,
 329         "piecewise_polynomial": _from_derivatives,
 330     }
 331 
 332     if getattr(x, "is_all_dates", False):
 333         # GH 5975, scipy.interp1d can't handle datetime64s
 334         x, new_x = x._values.astype("i8"), new_x.astype("i8")
 335 
 336     if method == "pchip":
 337         alt_methods["pchip"] = interpolate.pchip_interpolate
 338     elif method == "akima":
 339         alt_methods["akima"] = _akima_interpolate
 340     elif method == "cubicspline":
 341         alt_methods["cubicspline"] = _cubicspline_interpolate
 342 
 343     interp1d_methods = [
 344         "nearest",
 345         "zero",
 346         "slinear",
 347         "quadratic",
 348         "cubic",
 349         "polynomial",
 350     ]
 351     if method in interp1d_methods:
 352         if method == "polynomial":
 353             method = order
 354         terp = interpolate.interp1d(
 355             x, y, kind=method, fill_value=fill_value, bounds_error=bounds_error
 356         )
 357         new_y = terp(new_x)
 358     elif method == "spline":
 359         # GH #10633, #24014
 360         if isna(order) or (order <= 0):
 361             raise ValueError(
 362                 f"order needs to be specified and greater than 0; got order: {order}"
 363             )
 364         terp = interpolate.UnivariateSpline(x, y, k=order, **kwargs)
 365         new_y = terp(new_x)
 366     else:
 367         # GH 7295: need to be able to write for some reason
 368         # in some circumstances: check all three
 369         if not x.flags.writeable:
 370             x = x.copy()
 371         if not y.flags.writeable:
 372             y = y.copy()
 373         if not new_x.flags.writeable:
 374             new_x = new_x.copy()
 375         method = alt_methods[method]
 376         new_y = method(x, y, new_x, **kwargs)
 377     return new_y

 380 def _from_derivatives(xi, yi, x, order=None, der=0, extrapolate=False):
 381     """
 382     Convenience function for interpolate.BPoly.from_derivatives.
 383 
 384     Construct a piecewise polynomial in the Bernstein basis, compatible
 385     with the specified values and derivatives at breakpoints.
 386 
 387     Parameters
 388     ----------
 389     xi : array_like
 390         sorted 1D array of x-coordinates
 391     yi : array_like or list of array-likes
 392         yi[i][j] is the j-th derivative known at xi[i]
 393     order: None or int or array_like of ints. Default: None.
 394         Specifies the degree of local polynomials. If not None, some
 395         derivatives are ignored.
 396     der : int or list
 397         How many derivatives to extract; None for all potentially nonzero
 398         derivatives (that is a number equal to the number of points), or a
 399         list of derivatives to extract. This number includes the function
 400         value as 0th derivative.
 401      extrapolate : bool, optional
 402         Whether to extrapolate to ouf-of-bounds points based on first and last
 403         intervals, or to return NaNs. Default: True.
 404 
 405     See Also
 406     --------
 407     scipy.interpolate.BPoly.from_derivatives
 408 
 409     Returns
 410     -------
 411     y : scalar or array_like
 412         The result, of length R or length M or M by R.
 413     """
 414     from scipy import interpolate
 415 
 416     # return the method for compat with scipy version & backwards compat
 417     method = interpolate.BPoly.from_derivatives
 418     m = method(xi, yi.reshape(-1, 1), orders=order, extrapolate=extrapolate)
 419 
 420     return m(x)

 423 def _akima_interpolate(xi, yi, x, der=0, axis=0):
 424     """
 425     Convenience function for akima interpolation.
 426     xi and yi are arrays of values used to approximate some function f,
 427     with ``yi = f(xi)``.
 428 
 429     See `Akima1DInterpolator` for details.
 430 
 431     Parameters
 432     ----------
 433     xi : array_like
 434         A sorted list of x-coordinates, of length N.
 435     yi : array_like
 436         A 1-D array of real values.  `yi`'s length along the interpolation
 437         axis must be equal to the length of `xi`. If N-D array, use axis
 438         parameter to select correct axis.
 439     x : scalar or array_like
 440         Of length M.
 441     der : int, optional
 442         How many derivatives to extract; None for all potentially
 443         nonzero derivatives (that is a number equal to the number
 444         of points), or a list of derivatives to extract. This number
 445         includes the function value as 0th derivative.
 446     axis : int, optional
 447         Axis in the yi array corresponding to the x-coordinate values.
 448 
 449     See Also
 450     --------
 451     scipy.interpolate.Akima1DInterpolator
 452 
 453     Returns
 454     -------
 455     y : scalar or array_like
 456         The result, of length R or length M or M by R,
 457 
 458     """
 459     from scipy import interpolate
 460 
 461     P = interpolate.Akima1DInterpolator(xi, yi, axis=axis)
 462 
 463     return P(x, nu=der)

 466 def _cubicspline_interpolate(xi, yi, x, axis=0, bc_type="not-a-knot", extrapolate=None):
 467     """
 468     Convenience function for cubic spline data interpolator.
 469 
 470     See `scipy.interpolate.CubicSpline` for details.
 471 
 472     Parameters
 473     ----------
 474     xi : array_like, shape (n,)
 475         1-d array containing values of the independent variable.
 476         Values must be real, finite and in strictly increasing order.
 477     yi : array_like
 478         Array containing values of the dependent variable. It can have
 479         arbitrary number of dimensions, but the length along ``axis``
 480         (see below) must match the length of ``x``. Values must be finite.
 481     x : scalar or array_like, shape (m,)
 482     axis : int, optional
 483         Axis along which `y` is assumed to be varying. Meaning that for
 484         ``x[i]`` the corresponding values are ``np.take(y, i, axis=axis)``.
 485         Default is 0.
 486     bc_type : string or 2-tuple, optional
 487         Boundary condition type. Two additional equations, given by the
 488         boundary conditions, are required to determine all coefficients of
 489         polynomials on each segment [2]_.
 490         If `bc_type` is a string, then the specified condition will be applied
 491         at both ends of a spline. Available conditions are:
 492         * 'not-a-knot' (default): The first and second segment at a curve end
 493           are the same polynomial. It is a good default when there is no
 494           information on boundary conditions.
 495         * 'periodic': The interpolated functions is assumed to be periodic
 496           of period ``x[-1] - x[0]``. The first and last value of `y` must be
 497           identical: ``y[0] == y[-1]``. This boundary condition will result in
 498           ``y'[0] == y'[-1]`` and ``y''[0] == y''[-1]``.
 499         * 'clamped': The first derivative at curves ends are zero. Assuming
 500           a 1D `y`, ``bc_type=((1, 0.0), (1, 0.0))`` is the same condition.
 501         * 'natural': The second derivative at curve ends are zero. Assuming
 502           a 1D `y`, ``bc_type=((2, 0.0), (2, 0.0))`` is the same condition.
 503         If `bc_type` is a 2-tuple, the first and the second value will be
 504         applied at the curve start and end respectively. The tuple values can
 505         be one of the previously mentioned strings (except 'periodic') or a
 506         tuple `(order, deriv_values)` allowing to specify arbitrary
 507         derivatives at curve ends:
 508         * `order`: the derivative order, 1 or 2.
 509         * `deriv_value`: array_like containing derivative values, shape must
 510           be the same as `y`, excluding ``axis`` dimension. For example, if
 511           `y` is 1D, then `deriv_value` must be a scalar. If `y` is 3D with
 512           the shape (n0, n1, n2) and axis=2, then `deriv_value` must be 2D
 513           and have the shape (n0, n1).
 514     extrapolate : {bool, 'periodic', None}, optional
 515         If bool, determines whether to extrapolate to out-of-bounds points
 516         based on first and last intervals, or to return NaNs. If 'periodic',
 517         periodic extrapolation is used. If None (default), ``extrapolate`` is
 518         set to 'periodic' for ``bc_type='periodic'`` and to True otherwise.
 519 
 520     See Also
 521     --------
 522     scipy.interpolate.CubicHermiteSpline
 523 
 524     Returns
 525     -------
 526     y : scalar or array_like
 527         The result, of shape (m,)
 528 
 529     References
 530     ----------
 531     .. [1] `Cubic Spline Interpolation
 532             <https://en.wikiversity.org/wiki/Cubic_Spline_Interpolation>`_
 533             on Wikiversity.
 534     .. [2] Carl de Boor, "A Practical Guide to Splines", Springer-Verlag, 1978.
 535     """
 536     from scipy import interpolate
 537 
 538     P = interpolate.CubicSpline(
 539         xi, yi, axis=axis, bc_type=bc_type, extrapolate=extrapolate
 540     )
 541 
 542     return P(x)

 545 def interpolate_2d(
 546     values, method="pad", axis=0, limit=None, fill_value=None, dtype=None
 547 ):
 548     """
 549     Perform an actual interpolation of values, values will be make 2-d if
 550     needed fills inplace, returns the result.
 551     """
 552     orig_values = values
 553 
 554     transf = (lambda x: x) if axis == 0 else (lambda x: x.T)
 555 
 556     # reshape a 1 dim if needed
 557     ndim = values.ndim
 558     if values.ndim == 1:
 559         if axis != 0:  # pragma: no cover
 560             raise AssertionError("cannot interpolate on a ndim == 1 with axis != 0")
 561         values = values.reshape(tuple((1,) + values.shape))
 562 
 563     if fill_value is None:
 564         mask = None
 565     else:  # todo create faster fill func without masking
 566         mask = mask_missing(transf(values), fill_value)
 567 
 568     method = clean_fill_method(method)
 569     if method == "pad":
 570         values = transf(pad_2d(transf(values), limit=limit, mask=mask, dtype=dtype))
 571     else:
 572         values = transf(
 573             backfill_2d(transf(values), limit=limit, mask=mask, dtype=dtype)
 574         )
 575 
 576     # reshape back
 577     if ndim == 1:
 578         values = values[0]
 579 
 580     if orig_values.dtype.kind == "M":
 581         # convert float back to datetime64
 582         values = values.astype(orig_values.dtype)
 583 
 584     return values

 587 def _cast_values_for_fillna(values, dtype):
 588     """
 589     Cast values to a dtype that algos.pad and algos.backfill can handle.
 590     """
 591     # TODO: for int-dtypes we make a copy, but for everything else this
 592     #  alters the values in-place.  Is this intentional?
 593 
 594     if (
 595         is_datetime64_dtype(dtype)
 596         or is_datetime64tz_dtype(dtype)
 597         or is_timedelta64_dtype(dtype)
 598     ):
 599         values = values.view(np.int64)
 600 
 601     elif is_integer_dtype(values):
 602         # NB: this check needs to come after the datetime64 check above
 603         values = ensure_float64(values)
 604 
 605     return values

 608 def _fillna_prep(values, mask=None, dtype=None):
 609     # boilerplate for pad_1d, backfill_1d, pad_2d, backfill_2d
 610     if dtype is None:
 611         dtype = values.dtype
 612 
 613     if mask is None:
 614         # This needs to occur before datetime/timedeltas are cast to int64
 615         mask = isna(values)
 616 
 617     values = _cast_values_for_fillna(values, dtype)
 618 
 619     mask = mask.view(np.uint8)
 620     return values, mask
 621

 623 def pad_1d(values, limit=None, mask=None, dtype=None):
 624     values, mask = _fillna_prep(values, mask, dtype)
 625     algos.pad_inplace(values, mask, limit=limit)
 626     return values
 627 
 628 
 629 def backfill_1d(values, limit=None, mask=None, dtype=None):
 630     values, mask = _fillna_prep(values, mask, dtype)
 631     algos.backfill_inplace(values, mask, limit=limit)
 632     return values
 633 
 634 
 635 def pad_2d(values, limit=None, mask=None, dtype=None):
 636     values, mask = _fillna_prep(values, mask, dtype)
 637 
 638     if np.all(values.shape):
 639         algos.pad_2d_inplace(values, mask, limit=limit)
 640     else:
 641         # for test coverage
 642         pass
 643     return values
 644 
 645 
 646 def backfill_2d(values, limit=None, mask=None, dtype=None):
 647     values, mask = _fillna_prep(values, mask, dtype)
 648 
 649     if np.all(values.shape):
 650         algos.backfill_2d_inplace(values, mask, limit=limit)
 651     else:
 652         # for test coverage
 653         pass
 654     return values

 669 def _interp_limit(invalid, fw_limit, bw_limit):
 670     """
 671     Get indexers of values that won't be filled
 672     because they exceed the limits.
 673 
 674     Parameters
 675     ----------
 676     invalid : boolean ndarray
 677     fw_limit : int or None
 678         forward limit to index
 679     bw_limit : int or None
 680         backward limit to index
 681 
 682     Returns
 683     -------
 684     set of indexers
 685 
 686     Notes
 687     -----
 688     This is equivalent to the more readable, but slower
 689 
 690     .. code-block:: python
 691 
 692         def _interp_limit(invalid, fw_limit, bw_limit):
 693             for x in np.where(invalid)[0]:
 694                 if invalid[max(0, x - fw_limit):x + bw_limit + 1].all():
 695                     yield x
 696     """
 697     # handle forward first; the backward direction is the same except
 698     # 1. operate on the reversed array
 699     # 2. subtract the returned indices from N - 1
 700     N = len(invalid)
 701     f_idx = set()
 702     b_idx = set()
 703 
 704     def inner(invalid, limit):
 705         limit = min(limit, N)
 706         windowed = _rolling_window(invalid, limit + 1).all(1)
 707         idx = set(np.where(windowed)[0] + limit) | set(
 708             np.where((~invalid[: limit + 1]).cumsum() == 0)[0]
 709         )
 710         return idx
 711 
 712     if fw_limit is not None:
 713 
 714         if fw_limit == 0:
 715             f_idx = set(np.where(invalid)[0])
 716         else:
 717             f_idx = inner(invalid, fw_limit)
 718 
 719     if bw_limit is not None:
 720 
 721         if bw_limit == 0:
 722             # then we don't even need to care about backwards
 723             # just use forwards
 724             return f_idx
 725         else:
 726             b_idx = list(inner(invalid[::-1], bw_limit))
 727             b_idx = set(N - 1 - np.asarray(b_idx))
 728             if fw_limit == 0:
 729                 return b_idx
 730 
 731     return f_idx & b_idx
 732 
 733 
 734 def _rolling_window(a, window):
 735     """
 736     [True, True, False, True, False], 2 ->
 737 
 738     [
 739         [True,  True],
 740         [True, False],
 741         [False, True],
 742         [True, False],
 743     ]
 744     """
 745     # https://stackoverflow.com/a/6811241
 746     shape = a.shape[:-1] + (a.shape[-1] - window + 1, window)
 747     strides = a.strides + (a.strides[-1],)
 748     return np.lib.stride_tricks.as_strided(a, shape=shape, strides=strides)
