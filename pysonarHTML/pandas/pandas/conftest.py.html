  62 def pytest_runtest_setup(item):
  63     if "slow" in item.keywords and item.config.getoption("--skip-slow"):
  64         pytest.skip("skipping due to --skip-slow")
  65 
  66     if "slow" not in item.keywords and item.config.getoption("--only-slow"):
  67         pytest.skip("skipping due to --only-slow")
  68 
  69     if "network" in item.keywords and item.config.getoption("--skip-network"):
  70         pytest.skip("skipping due to --skip-network")
  71 
  72     if "db" in item.keywords and item.config.getoption("--skip-db"):
  73         pytest.skip("skipping due to --skip-db")
  74 
  75     if "high_memory" in item.keywords and not item.config.getoption(
  76         "--run-high-memory"
  77     ):
  78         pytest.skip("skipping high memory test since --run-high-memory was not set")
