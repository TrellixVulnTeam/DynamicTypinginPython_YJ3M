  12 _initial_defencoding = None

  15 def detect_console_encoding() -> str:
  16     """
  17     Try to find the most capable encoding supported by the console.
  18     slightly modified from the way IPython handles the same issue.
  19     """
  20     global _initial_defencoding
  21 
  22     encoding = None
  23     try:
  24         encoding = sys.stdout.encoding or sys.stdin.encoding
  25     except (AttributeError, IOError):
  26         pass
  27 
  28     # try again for something better
  29     if not encoding or "ascii" in encoding.lower():
  30         try:
  31             encoding = locale.getpreferredencoding()
  32         except locale.Error:
  33             # can be raised by locale.setlocale(), which is
  34             #  called by getpreferredencoding
  35             #  (on some systems, see stdlib locale docs)
  36             pass
  37 
  38     # when all else fails. this will usually be "ascii"
  39     if not encoding or "ascii" in encoding.lower():
  40         encoding = sys.getdefaultencoding()
  41 
  42     # GH#3360, save the reported defencoding at import time
  43     # MPL backends may change it. Make available for debugging.
  44     if not _initial_defencoding:
  45         _initial_defencoding = sys.getdefaultencoding()
  46 
  47     return encoding
  48 
  49 
  50 pc_encoding_doc = """
  51 : str/unicode
  52     Defaults to the detected encoding of the console.
  53     Specifies the encoding to be used for strings returned by to_string,
  54     these are generally strings meant to be displayed on the console.
  55 """
  56 
  57 with cf.config_prefix("display"):
  58     cf.register_option(
  59         "encoding", detect_console_encoding(), pc_encoding_doc, validator=cf.is_text
  60     )
