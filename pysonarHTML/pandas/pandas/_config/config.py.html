   1 """
   2 The config module holds package-wide configurables and provides
   3 a uniform API for working with them.
   4 
   5 Overview
   6 ========
   7 
   8 This module supports the following requirements:
   9 - options are referenced using keys in dot.notation, e.g. "x.y.option - z".
  10 - keys are case-insensitive.
  11 - functions should accept partial/regex keys, when unambiguous.
  12 - options can be registered by modules at import time.
  13 - options can be registered at init-time (via core.config_init)
  14 - options have a default value, and (optionally) a description and
  15   validation function associated with them.
  16 - options can be deprecated, in which case referencing them
  17   should produce a warning.
  18 - deprecated options can optionally be rerouted to a replacement
  19   so that accessing a deprecated option reroutes to a differently
  20   named option.
  21 - options can be reset to their default value.
  22 - all option can be reset to their default value at once.
  23 - all options in a certain sub - namespace can be reset at once.
  24 - the user can set / get / reset or ask for the description of an option.
  25 - a developer can register and mark an option as deprecated.
  26 - you can register a callback to be invoked when the option value
  27   is set or reset. Changing the stored value is considered misuse, but
  28   is not verboten.
  29 
  30 Implementation
  31 ==============
  32 
  33 - Data is stored using nested dictionaries, and should be accessed
  34   through the provided API.
  35 
  36 - "Registered options" and "Deprecated options" have metadata associated
  37   with them, which are stored in auxiliary dictionaries keyed on the
  38   fully-qualified key, e.g. "x.y.z.option".
  39 
  40 - the config_init module is imported by the package's __init__.py file.
  41   placing any register_option() calls there will ensure those options
  42   are available as soon as pandas is loaded. If you use register_option
  43   in a module, it will only be available after that module is imported,
  44   which you should be aware of.
  45 
  46 - `config_prefix` is a context_manager (for use with the `with` keyword)
  47   which can save developers some typing, see the docstring.
  48 
  49 """

  86 def _get_single_key(pat: str, silent: bool) -> str:
  87     keys = _select_options(pat)
  88     if len(keys) == 0:
  89         if not silent:
  90             _warn_if_deprecated(pat)
  91         raise OptionError(f"No such keys(s): {repr(pat)}")
  92     if len(keys) > 1:
  93         raise OptionError("Pattern matched multiple keys")
  94     key = keys[0]
  95 
  96     if not silent:
  97         _warn_if_deprecated(key)
  98 
  99     key = _translate_key(key)
 100 
 101     return key

 104 def _get_option(pat: str, silent: bool = False):
 105     key = _get_single_key(pat, silent)

 108     root, k = _get_root(key)

 112 def _set_option(*args, **kwargs) -> None:
 113     # must at least 1 arg deal with constraints later
 114     nargs = len(args)
 115     if not nargs or nargs % 2 != 0:
 116         raise ValueError("Must provide an even number of non-keyword arguments")
 117 
 118     # default to false
 119     silent = kwargs.pop("silent", False)
 120 
 121     if kwargs:
 122         kwarg = list(kwargs.keys())[0]
 123         raise TypeError(f'_set_option() got an unexpected keyword argument "{kwarg}"')
 124 
 125     for k, v in zip(args[::2], args[1::2]):
 126         key = _get_single_key(k, silent)
 127 
 128         o = _get_registered_option(key)
 129         if o and o.validator:
 130             o.validator(v)
 131 
 132         # walk the nested dict
 133         root, k = _get_root(key)
 134         root[k] = v
 135 
 136         if o.cb:
 137             if silent:
 138                 with warnings.catch_warnings(record=True):
 139                     o.cb(key)
 140             else:
 141                 o.cb(key)

 144 def _describe_option(pat: str = "", _print_desc: bool = True):
 145 
 146     keys = _select_options(pat)
 147     if len(keys) == 0:
 148         raise OptionError("No such keys(s)")
 149 
 150     s = "\n".join([_build_option_description(k) for k in keys])
 151 
 152     if _print_desc:
 153         print(s)
 154     else:
 155         return s

 158 def _reset_option(pat: str, silent: bool = False) -> None:
 159 
 160     keys = _select_options(pat)
 161 
 162     if len(keys) == 0:
 163         raise OptionError("No such keys(s)")
 164 
 165     if len(keys) > 1 and len(pat) < 4 and pat != "all":
 166         raise ValueError(
 167             "You must specify at least 4 characters when "
 168             "resetting multiple keys, use the special keyword "
 169             '"all" to reset all the options to their default value'
 170         )
 171 
 172     for k in keys:
 173         _set_option(k, _registered_options[k].defval, silent=silent)

 177     key = _get_single_key(pat, silent=True)

 181 class DictWrapper:
 182     """ provide attribute-style access to a nested dict"""
 183 
 184     def __init__(self, d: Dict[str, Any], prefix: str = ""):
 185         object.__setattr__(self, "d", d)
 186         object.__setattr__(self, "prefix", prefix)
 187 
 188     def __setattr__(self, key: str, val: Any) -> None:
 189         prefix = object.__getattribute__(self, "prefix")
 190         if prefix:
 191             prefix += "."
 192         prefix += key
 193         # you can't set new keys
 194         # can you can't overwrite subtrees
 195         if key in self.d and not isinstance(self.d[key], dict):
 196             _set_option(prefix, val)
 197         else:
 198             raise OptionError("You can only set the value of existing options")
 199 
 200     def __getattr__(self, key: str):
 201         prefix = object.__getattribute__(self, "prefix")
 202         if prefix:
 203             prefix += "."
 204         prefix += key
 205         try:
 206             v = object.__getattribute__(self, "d")[key]
 207         except KeyError as err:
 208             raise OptionError("No such option") from err
 209         if isinstance(v, dict):
 210             return DictWrapper(v, prefix)
 211         else:
 212             return _get_option(prefix)
 213 
 214     def __dir__(self) -> Iterable[str]:
 215         return list(self.d.keys())

 229         self.__doc_tmpl__ = doc_tmpl
 230         self.__func__ = func
 231

 382 class option_context(ContextDecorator):
 383     """
 384     Context manager to temporarily set options in the `with` statement context.
 385 
 386     You need to invoke as ``option_context(pat, val, [(pat, val), ...])``.
 387 
 388     Examples
 389     --------
 390     >>> with option_context('display.max_rows', 10, 'display.max_columns', 5):
 391     ...     ...
 392     """
 393 
 394     def __init__(self, *args):
 395         if not (len(args) % 2 == 0 and len(args) >= 2):
 396             raise ValueError(
 397                 "Need to invoke as option_context(pat, val, [(pat, val), ...])."
 398             )
 399 
 400         self.ops = list(zip(args[::2], args[1::2]))
 401 
 402     def __enter__(self):
 403         self.undo = [(pat, _get_option(pat, silent=True)) for pat, val in self.ops]
 404 
 405         for pat, val in self.ops:
 406             _set_option(pat, val, silent=True)
 407 
 408     def __exit__(self, *args):
 409         if self.undo:
 410             for pat, val in self.undo:
 411                 _set_option(pat, val, silent=True)

 414 def register_option(
 415     key: str,
 416     defval: object,
 417     doc: str = "",
 418     validator: Optional[Callable[[Any], Any]] = None,
 419     cb: Optional[Callable[[str], Any]] = None,
 420 ) -> None:
 421     """
 422     Register an option in the package-wide pandas config object
 423 
 424     Parameters
 425     ----------
 426     key : str
 427         Fully-qualified key, e.g. "x.y.option - z".
 428     defval : object
 429         Default value of the option.
 430     doc : str
 431         Description of the option.
 432     validator : Callable, optional
 433         Function of a single argument, should raise `ValueError` if
 434         called with a value which is not a legal value for the option.
 435     cb
 436         a function of a single argument "key", which is called
 437         immediately after an option value is set/reset. key is
 438         the full name of the option.
 439 
 440     Raises
 441     ------
 442     ValueError if `validator` is specified and `defval` is not a valid value.
 443 
 444     """
 445     import tokenize
 446     import keyword
 447 
 448     key = key.lower()
 449 
 450     if key in _registered_options:
 451         raise OptionError(f"Option '{key}' has already been registered")
 452     if key in _reserved_keys:
 453         raise OptionError(f"Option '{key}' is a reserved key")
 454 
 455     # the default value should be legal
 456     if validator:
 457         validator(defval)
 458 
 459     # walk the nested dict, creating dicts as needed along the path
 460     path = key.split(".")
 461 
 462     for k in path:
 463         # NOTE: tokenize.Name is not a public constant
 464         # error: Module has no attribute "Name"  [attr-defined]
 465         if not re.match("^" + tokenize.Name + "$", k):  # type: ignore
 466             raise ValueError(f"{k} is not a valid identifier")
 467         if keyword.iskeyword(k):
 468             raise ValueError(f"{k} is a python keyword")
 469 
 470     cursor = _global_config
 471     msg = "Path prefix to option '{option}' is already an option"
 472 
 473     for i, p in enumerate(path[:-1]):
 474         if not isinstance(cursor, dict):
 475             raise OptionError(msg.format(option=".".join(path[:i])))
 476         if p not in cursor:
 477             cursor[p] = {}
 478         cursor = cursor[p]
 479 
 480     if not isinstance(cursor, dict):
 481         raise OptionError(msg.format(option=".".join(path[:-1])))
 482 
 483     cursor[path[-1]] = defval  # initialize
 484 
 485     # save the option metadata
 486     _registered_options[key] = RegisteredOption(
 487         key=key, defval=defval, doc=doc, validator=validator, cb=cb
 488     )

 491 def deprecate_option(
 492     key: str, msg: Optional[str] = None, rkey: Optional[str] = None, removal_ver=None
 493 ) -> None:
 494     """
 495     Mark option `key` as deprecated, if code attempts to access this option,
 496     a warning will be produced, using `msg` if given, or a default message
 497     if not.
 498     if `rkey` is given, any access to the key will be re-routed to `rkey`.
 499 
 500     Neither the existence of `key` nor that if `rkey` is checked. If they
 501     do not exist, any subsequence access will fail as usual, after the
 502     deprecation warning is given.
 503 
 504     Parameters
 505     ----------
 506     key : str
 507         Name of the option to be deprecated.
 508         must be a fully-qualified option name (e.g "x.y.z.rkey").
 509     msg : str, optional
 510         Warning message to output when the key is referenced.
 511         if no message is given a default message will be emitted.
 512     rkey : str, optional
 513         Name of an option to reroute access to.
 514         If specified, any referenced `key` will be
 515         re-routed to `rkey` including set/get/reset.
 516         rkey must be a fully-qualified option name (e.g "x.y.z.rkey").
 517         used by the default message if no `msg` is specified.
 518     removal_ver : optional
 519         Specifies the version in which this option will
 520         be removed. used by the default message if no `msg` is specified.
 521 
 522     Raises
 523     ------
 524     OptionError
 525         If the specified key has already been deprecated.
 526     """
 527     key = key.lower()
 528 
 529     if key in _deprecated_options:
 530         raise OptionError(f"Option '{key}' has already been defined as deprecated.")
 531 
 532     _deprecated_options[key] = DeprecatedOption(key, msg, rkey, removal_ver)

 539 def _select_options(pat: str) -> List[str]:
 540     """
 541     returns a list of keys matching `pat`
 542 
 543     if pat=="all", returns all registered options
 544     """
 545     # short-circuit for exact key
 546     if pat in _registered_options:
 547         return [pat]
 548 
 549     # else look through all of them
 550     keys = sorted(_registered_options.keys())
 551     if pat == "all":  # reserved key
 552         return keys
 553 
 554     return [k for k in keys if re.search(pat, k, re.I)]

 557 def _get_root(key: str) -> Tuple[Dict[str, Any], str]:
 558     path = key.split(".")
 559     cursor = _global_config
 560     for p in path[:-1]:
 561         cursor = cursor[p]
 562     return cursor, path[-1]

 598 def _translate_key(key: str) -> str:
 599     """
 600     if key id deprecated and a replacement key defined, will return the
 601     replacement key, otherwise returns `key` as - is

 610 def _warn_if_deprecated(key: str) -> bool:
 611     """
 612     Checks if `key` is a deprecated option and if so, prints a warning.
 613 
 614     Returns
 615     -------
 616     bool - True if `key` is deprecated, False otherwise.
 617     """
 618     d = _get_deprecated_option(key)
 619     if d:
 620         if d.msg:
 621             print(d.msg)
 622             warnings.warn(d.msg, FutureWarning)
 623         else:
 624             msg = f"'{key}' is deprecated"
 625             if d.removal_ver:
 626                 msg += f" and will be removed in {d.removal_ver}"
 627             if d.rkey:
 628                 msg += f", please use '{d.rkey}' instead."
 629             else:
 630                 msg += ", please refrain from using it."
 631 
 632             warnings.warn(msg, FutureWarning)
 633         return True
 634     return False

 637 def _build_option_description(k: str) -> str:
 638     """ Builds a formatted description of a registered option and prints it """
 639     o = _get_registered_option(k)
 640     d = _get_deprecated_option(k)
 641 
 642     s = f"{k} "
 643 
 644     if o.doc:
 645         s += "\n".join(o.doc.strip().split("\n"))
 646     else:
 647         s += "No description available."
 648 
 649     if o:
 650         s += f"\n    [default: {o.defval}] [currently: {_get_option(k, True)}]"
 651 
 652     if d:
 653         rkey = d.rkey if d.rkey else ""
 654         s += "\n    (Deprecated"
 655         s += f", use `{rkey}` instead."
 656         s += ")"
 657 
 658     return s

 661 def pp_options_list(keys: Iterable[str], width=80, _print: bool = False):
 662     """ Builds a concise listing of available options, grouped by prefix """
 663     from textwrap import wrap
 664     from itertools import groupby
 665 
 666     def pp(name: str, ks: Iterable[str]) -> List[str]:
 667         pfx = "- " + name + ".[" if name else ""
 668         ls = wrap(
 669             ", ".join(ks),
 670             width,
 671             initial_indent=pfx,
 672             subsequent_indent="  ",
 673             break_long_words=False,
 674         )
 675         if ls and ls[-1] and name:
 676             ls[-1] = ls[-1] + "]"
 677         return ls
 678 
 679     ls: List[str] = []
 680     singles = [x for x in sorted(keys) if x.find(".") < 0]
 681     if singles:
 682         ls += pp("", singles)
 683     keys = [x for x in keys if x.find(".") >= 0]
 684 
 685     for k, g in groupby(sorted(keys), lambda x: x[: x.rfind(".")]):
 686         ks = [x[len(k) + 1 :] for x in list(g)]
 687         ls += pp(k, ks)
 688     s = "\n".join(ls)
 689     if _print:
 690         print(s)
 691     else:
 692         return s

 699 @contextmanager
 700 def config_prefix(prefix):
 701     """
 702     contextmanager for multiple invocations of API with a common prefix
 703 
 704     supported API functions: (register / get / set )__option
 705 
 706     Warning: This is not thread - safe, and won't work properly if you import
 707     the API functions into your module using the "from x import y" construct.
 708 
 709     Example
 710     -------
 711     import pandas._config.config as cf
 712     with cf.config_prefix("display.font"):
 713         cf.register_option("color", "red")
 714         cf.register_option("size", " 5 pt")
 715         cf.set_option(size, " 6 pt")
 716         cf.get_option(size)
 717         ...
 718 
 719         etc'
 720 
 721     will register options "display.font.color", "display.font.size", set the
 722     value of "display.font.size"... and so on.
 723     """
 724     # Note: reset_option relies on set_option, and on key directly
 725     # it does not fit in to this monkey-patching scheme
 726 
 727     global register_option, get_option, set_option, reset_option
 728 
 729     def wrap(func: F) -> F:
 730         def inner(key: str, *args, **kwds):
 731             pkey = f"{prefix}.{key}"
 732             return func(pkey, *args, **kwds)
 733 
 734         return cast(F, inner)
 735 
 736     _register_option = register_option
 737     _get_option = get_option

 752 def is_type_factory(_type: Type[Any]) -> Callable[[Any], None]:
 753     """
 754 
 755     Parameters
 756     ----------
 757     `_type` - a type to be compared against (e.g. type(x) == `_type`)
 758 
 759     Returns
 760     -------
 761     validator - a function of a single argument x , which raises
 762                 ValueError if type(x) is not equal to `_type`
 763 
 764     """
 765 
 766     def inner(x) -> None:
 767         if type(x) != _type:
 768             raise ValueError(f"Value must have type '{_type}'")
 769 
 770     return inner

 773 def is_instance_factory(_type) -> Callable[[Any], None]:
 774     """
 775 
 776     Parameters
 777     ----------
 778     `_type` - the type to be checked against
 779 
 780     Returns
 781     -------
 782     validator - a function of a single argument x , which raises
 783                 ValueError if x is not an instance of `_type`
 784 
 785     """
 786     if isinstance(_type, (tuple, list)):
 787         _type = tuple(_type)
 788         type_repr = "|".join(map(str, _type))
 789     else:
 790         type_repr = f"'{_type}'"
 791 
 792     def inner(x) -> None:
 793         if not isinstance(x, _type):
 794             raise ValueError(f"Value must be an instance of {type_repr}")
 795 
 796     return inner

 799 def is_one_of_factory(legal_values) -> Callable[[Any], None]:
 800 
 801     callables = [c for c in legal_values if callable(c)]
 802     legal_values = [c for c in legal_values if not callable(c)]
 803 
 804     def inner(x) -> None:
 805         if x not in legal_values:
 806 
 807             if not any(c(x) for c in callables):
 808                 uvals = [str(lval) for lval in legal_values]
 809                 pp_values = "|".join(uvals)
 810                 msg = f"Value must be one of {pp_values}"
 811                 if len(callables):
 812                     msg += " or a callable"
 813                 raise ValueError(msg)
 814 
 815     return inner

 818 def is_nonnegative_int(value: Optional[int]) -> None:
 819     """
 820     Verify that value is None or a positive int.
 821 
 822     Parameters
 823     ----------
 824     value : None or int
 825             The `value` to be checked.
 826 
 827     Raises
 828     ------
 829     ValueError
 830         When the value is not None or is a negative integer
 831     """
 832     if value is None:
 833         return
 834 
 835     elif isinstance(value, int):
 836         if value >= 0:
 837             return
 838 
 839     msg = "Value must be a nonnegative integer or None"
 840     raise ValueError(msg)
