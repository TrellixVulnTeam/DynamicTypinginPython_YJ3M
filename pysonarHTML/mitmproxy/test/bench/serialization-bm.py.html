  13 class StreamTester:
  14 
  15     """
  16     Generates a constant stream of flows and
  17     measure protobuf dumping throughput.
  18     """
  19 
  20     def __init__(self):
  21         self.dbh = None
  22         self.streaming = False
  23         self.tf = None
  24         self.out = None
  25         self.hot_flows = []
  26         self.results = []
  27         self._flushes = 0
  28         self._stream_period = 0.001
  29         self._flush_period = 3.0
  30         self._flush_rate = 150
  31         self._target = 2000
  32         self.loop = asyncio.get_event_loop()
  33         self.queue = asyncio.Queue(maxsize=self._flush_rate * 3, loop=self.loop)
  34         self.temp = tempfile.NamedTemporaryFile()
  35 
  36     def load(self, loader):
  37         loader.add_option(
  38             "testflow_size",
  39             int,
  40             1000,
  41             "Length in bytes of test flow content"
  42         )
  43         loader.add_option(
  44             "benchmark_save_path",
  45             typing.Optional[str],
  46             None,
  47             "Destination for the stats result file"
  48         )
  49 
  50     def _log(self, msg):
  51         if self.out:
  52             self.out.write(msg + '\n')
  53         else:
  54             ctx.log(msg)
  55 
  56     def running(self):
  57         if not self.streaming:
  58             ctx.log("<== Serialization Benchmark Enabled ==>")
  59             self.tf = tflow.tflow()
  60             self.tf.request.content = b'A' * ctx.options.testflow_size
  61             ctx.log(f"With content size: {len(self.tf.request.content)} B")
  62             if ctx.options.benchmark_save_path:
  63                 ctx.log(f"Storing results to {ctx.options.benchmark_save_path}")
  64                 self.out = open(ctx.options.benchmark_save_path, "w")
  65             self.dbh = db.DBHandler(self.temp.name, mode='write')
  66             self.streaming = True
  67             tasks = (self.stream, self.writer, self.stats)
  68             self.loop.create_task(asyncio.gather(*(t() for t in tasks)))
  69 
  70     async def stream(self):
  71         while True:
  72             await self.queue.put(self.tf)
  73             await asyncio.sleep(self._stream_period)
  74 
  75     async def writer(self):
  76         while True:
  77             await asyncio.sleep(self._flush_period)
  78             count = 1
  79             f = await self.queue.get()
  80             self.hot_flows.append(f)
  81             while count < self._flush_rate:
  82                 try:
  83                     self.hot_flows.append(self.queue.get_nowait())
  84                     count += 1
  85                 except asyncio.QueueEmpty:
  86                     pass
  87             start = time.perf_counter()
  88             n = self._fflush()
  89             end = time.perf_counter()
  90             self._log(f"dumps/time ratio: {n} / {end-start} -> {n/(end-start)}")
  91             self.results.append(n / (end - start))
  92             self._flushes += n
  93             self._log(f"Flows dumped: {self._flushes}")
  94             ctx.log(f"Progress: {min(100.0, 100.0 * (self._flushes / self._target))}%")
  95 
  96     async def stats(self):
  97         while True:
  98             await asyncio.sleep(1.0)
  99             if self._flushes >= self._target:
 100                 self._log(f"AVG : {mean(self.results)}")
 101                 ctx.log(f"<== Benchmark Ended. Shutting down... ==>")
 102                 if self.out:
 103                     self.out.close()
 104                 self.temp.close()
 105                 ctx.master.shutdown()
 106 
 107     def _fflush(self):
 108         self.dbh.store(self.hot_flows)
 109         n = len(self.hot_flows)
 110         self.hot_flows = []
 111         return n
