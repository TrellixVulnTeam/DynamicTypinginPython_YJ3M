  25     def test_modify_form(self, tdata):
  26         with taddons.context() as tctx:
  27             sc = tctx.script(tdata.path("../examples/addons/http-modify-form.py"))
  28 
  29             form_header = Headers(content_type="application/x-www-form-urlencoded")
  30             f = tflow.tflow(req=tutils.treq(headers=form_header))
  31             sc.request(f)
  32 
  33             assert f.request.urlencoded_form["mitmproxy"] == "rocks"
  34 
  35             f.request.headers["content-type"] = ""
  36             sc.request(f)
  37             assert list(f.request.urlencoded_form.items()) == [("foo", "bar")]
