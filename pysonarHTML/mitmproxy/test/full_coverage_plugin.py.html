  29 def pytest_configure(config):
  30     global enable_coverage
  31     global no_full_cov
  32 
  33     enable_coverage = (
  34         config.getoption('file_or_dir') and len(config.getoption('file_or_dir')) == 0 and
  35         config.getoption('full_cov') and len(config.getoption('full_cov')) > 0 and
  36         config.pluginmanager.getplugin("_cov") is not None and
  37         config.pluginmanager.getplugin("_cov").cov_controller is not None and
  38         config.pluginmanager.getplugin("_cov").cov_controller.cov is not None
  39     )
  40 
  41     c = configparser.ConfigParser()
  42     c.read(os.path.join(here, "..", "setup.cfg"))
  43     fs = c['tool:full_coverage']['exclude'].split('\n')
  44     no_full_cov = config.option.no_full_cov + [f.strip() for f in fs]

  47 @pytest.hookimpl(hookwrapper=True)
  48 def pytest_runtestloop(session):
  49     global enable_coverage
  50     global coverage_values
  51     global coverage_passed
  52     global no_full_cov
  53 
  54     if not enable_coverage:
  55         yield
  56         return
  57 
  58     cov = session.config.pluginmanager.getplugin("_cov").cov_controller.cov
  59 
  60     if os.name == 'nt':
  61         cov.exclude('pragma: windows no cover')
  62 
  63     if sys.platform == 'darwin':
  64         cov.exclude('pragma: osx no cover')
  65 
  66     if os.environ.get("OPENSSL") == "old":
  67         cov.exclude('pragma: openssl-old no cover')
  68 
  69     yield
  70 
  71     coverage_values = dict([(name, 0) for name in session.config.option.full_cov])
  72 
  73     prefix = os.getcwd()
  74 
  75     excluded_files = [os.path.normpath(f) for f in no_full_cov]
  76     measured_files = [os.path.normpath(os.path.relpath(f, prefix)) for f in cov.get_data().measured_files()]
  77     measured_files = [f for f in measured_files if not any(f.startswith(excluded_f) for excluded_f in excluded_files)]
  78 
  79     for name in coverage_values.keys():
  80         files = [f for f in measured_files if f.startswith(os.path.normpath(name))]
  81         try:
  82             with open(os.devnull, 'w') as null:
  83                 overall = cov.report(files, ignore_errors=True, file=null)
  84                 singles = [(s, cov.report(s, ignore_errors=True, file=null)) for s in files]
  85                 coverage_values[name] = (overall, singles)
  86         except:
  87             pass
  88 
  89     if any(v < 100 for v, _ in coverage_values.values()):
  90         # make sure we get the EXIT_TESTSFAILED exit code
  91         session.testsfailed += 1
  92         coverage_passed = False

  95 def pytest_terminal_summary(terminalreporter, exitstatus, config):
  96     global enable_coverage
  97     global coverage_values
  98     global coverage_passed
  99     global no_full_cov
 100 
 101     if not enable_coverage:
 102         return
 103 
 104     terminalreporter.write('\n')
 105     if not coverage_passed:
 106         markup = {'red': True, 'bold': True}
 107         msg = "FAIL: Full test coverage not reached!\n"
 108         terminalreporter.write(msg, **markup)
 109 
 110         for name in sorted(coverage_values.keys()):
 111             msg = 'Coverage for {}: {:.2f}%\n'.format(name, coverage_values[name][0])
 112             if coverage_values[name][0] < 100:
 113                 markup = {'red': True, 'bold': True}
 114                 for s, v in sorted(coverage_values[name][1]):
 115                     if v < 100:
 116                         msg += '  {}: {:.2f}%\n'.format(s, v)
 117             else:
 118                 markup = {'green': True}
 119             terminalreporter.write(msg, **markup)
 120     else:
 121         msg = 'SUCCESS: Full test coverage reached in modules and files:\n'
 122         msg += '{}\n\n'.format('\n'.join(config.option.full_cov))
 123         terminalreporter.write(msg, green=True)
 124 
 125     msg = '\nExcluded files:\n'
 126     for s in sorted(no_full_cov):
 127         msg += "  {}\n".format(s)
 128     terminalreporter.write(msg)
