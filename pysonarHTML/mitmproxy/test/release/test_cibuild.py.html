 253 @pytest.mark.parametrize("version, tag, ok", [
 254     ("3.0.0.dev", "", True),  # regular snapshot
 255     ("3.0.0.dev", "v3.0.0", False),  # forgot to remove ".dev" on bump
 256     ("3.0.0", "", False),  # forgot to re-add ".dev"
 257     ("3.0.0", "v4.0.0", False),  # version mismatch
 258     ("3.0.0", "v3.0.0", True),  # regular release
 259     ("3.0.0.rc1", "v3.0.0.rc1", False),  # non-canonical.
 260     ("3.0.0.dev", "anyname", True),  # tagged test/dev release
 261     ("3.0.0", "3.0.0", False),  # tagged, but without v prefix
 262 ])
 263 def test_buildenviron_check_version(version, tag, ok, tmpdir):
 264     tmpdir.mkdir("mitmproxy").join("version.py").write(f'VERSION = "{version}"')
 265 
 266     be = cibuild.BuildEnviron(
 267         root_dir=tmpdir,
 268         travis_tag=tag,
 269         travis_branch=tag or "branch",
 270     )
 271     if ok:
 272         be.check_version()
 273     else:
 274         with pytest.raises(ValueError):
 275             be.check_version()

 278 def test_bool_from_env(monkeypatch):
 279     monkeypatch.setenv("FOO", "1")
 280     assert cibuild.bool_from_env("FOO")
 281 
 282     monkeypatch.setenv("FOO", "0")
 283     assert not cibuild.bool_from_env("FOO")
 284 
 285     monkeypatch.setenv("FOO", "false")
 286     assert not cibuild.bool_from_env("FOO")
 287 
 288     monkeypatch.setenv("FOO", "")
 289     assert not cibuild.bool_from_env("FOO")
 290 
 291     monkeypatch.delenv("FOO")
 292     assert not cibuild.bool_from_env("FOO")
