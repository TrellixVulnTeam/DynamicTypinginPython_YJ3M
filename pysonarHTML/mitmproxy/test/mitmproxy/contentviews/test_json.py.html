   8 def test_parse_json():
   9     assert json.parse_json(b'{"foo": 1}')
  10     assert json.parse_json(b'null') is None
  11     assert json.parse_json(b"moo") is json.PARSE_ERROR
  12     assert json.parse_json(b'{"foo" : "\xe4\xb8\x96\xe7\x95\x8c"}')  # utf8 with chinese characters
  13     assert json.parse_json(b'{"foo" : "\xFF"}') is json.PARSE_ERROR

  30 def test_view_json():
  31     v = full_eval(json.ViewJSON())
  32     assert v(b"null")
  33     assert v(b"{}")
  34     assert not v(b"{")
  35     assert v(b"[1, 2, 3, 4, 5]")
  36     assert v(b'{"foo" : 3}')
  37     assert v(b'{"foo": true, "nullvalue": null}')
