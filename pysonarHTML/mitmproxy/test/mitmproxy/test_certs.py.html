 153     def test_simple(self, tdata):
 154         with open(tdata.path("mitmproxy/net/data/text_cert"), "rb") as f:
 155             d = f.read()
 156         c1 = certs.Cert.from_pem(d)

 162             d = f.read()
 163         c2 = certs.Cert.from_pem(d)
 164         assert c2.cn == b"www.inode.co.nz"
 165         assert len(c2.altnames) == 2
 166         assert c2.digest("sha1")
 167         assert c2.notbefore
 168         assert c2.notafter
 169         assert c2.subject
 170         assert c2.keyinfo == ("RSA", 2048)
 171         assert c2.serial
 172         assert c2.issuer
 173         assert c2.to_pem()
 174         assert c2.has_expired is not None
 175 
 176         assert c1 != c2
