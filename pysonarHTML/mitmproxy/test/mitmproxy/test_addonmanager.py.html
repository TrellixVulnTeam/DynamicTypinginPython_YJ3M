 162 def test_nesting():
 163     o = options.Options()
 164     m = master.Master(o)
 165     a = addonmanager.AddonManager(m)
 166 
 167     a.add(
 168         TAddon(
 169             "one",
 170             addons=[
 171                 TAddon("two"),
 172                 TAddon("three", addons=[TAddon("four")])
 173             ]
 174         )
 175     )
 176     assert len(a.chain) == 1
 177     assert a.get("one")
 178     assert a.get("two")
 179     assert a.get("three")
 180     assert a.get("four")
 181 
 182     a.trigger("running")
 183     assert a.get("one").running_called
 184     assert a.get("two").running_called
 185     assert a.get("three").running_called
 186     assert a.get("four").running_called
 187 
 188     a.remove(a.get("three"))
 189     assert not a.get("three")
 190     assert not a.get("four")
