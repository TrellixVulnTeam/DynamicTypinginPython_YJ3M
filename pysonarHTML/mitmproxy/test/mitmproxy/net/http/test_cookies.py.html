 224 def test_parse_set_cookie_header():
 225     def set_cookie_equal(obs, exp):
 226         assert obs[0] == exp[0]
 227         assert obs[1] == exp[1]
 228         assert obs[2].items(multi=True) == exp[2]
 229 
 230     vals = [
 231         [
 232             "", []
 233         ],
 234         [
 235             ";", []
 236         ],
 237         [
 238             "=uno",
 239             [
 240                 ("", "uno", ())
 241             ]
 242         ],
 243         [
 244             "one=uno",
 245             [
 246                 ("one", "uno", ())
 247             ]
 248         ],
 249         [
 250             "one=uno; foo=bar",
 251             [
 252                 ("one", "uno", (("foo", "bar"),))
 253             ]
 254         ],
 255         [
 256             "one=uno; foo=bar; foo=baz",
 257             [
 258                 ("one", "uno", (("foo", "bar"), ("foo", "baz")))
 259             ]
 260         ],
 261         # Comma Separated Variant of Set-Cookie Headers
 262         [
 263             "foo=bar, doo=dar",
 264             [
 265                 ("foo", "bar", ()),
 266                 ("doo", "dar", ()),
 267             ]
 268         ],
 269         [
 270             "foo=bar; path=/, doo=dar; roo=rar; zoo=zar",
 271             [
 272                 ("foo", "bar", (("path", "/"),)),
 273                 ("doo", "dar", (("roo", "rar"), ("zoo", "zar"))),
 274             ]
 275         ],
 276         [
 277             "foo=bar; expires=Mon, 24 Aug 2037",
 278             [
 279                 ("foo", "bar", (("expires", "Mon, 24 Aug 2037"),)),
 280             ]
 281         ],
 282         [
 283             "foo=bar; expires=Mon, 24 Aug 2037 00:00:00 GMT, doo=dar",
 284             [
 285                 ("foo", "bar", (("expires", "Mon, 24 Aug 2037 00:00:00 GMT"),)),
 286                 ("doo", "dar", ()),
 287             ]
 288         ],
 289     ]
 290     for s, expected in vals:
 291         ret = cookies.parse_set_cookie_header(s)
 292         if expected:
 293             for i in range(len(expected)):
 294                 set_cookie_equal(ret[i], expected[i])
 295 
 296             s2 = cookies.format_set_cookie_header(ret)
 297             ret2 = cookies.parse_set_cookie_header(s2)
 298             for i in range(len(expected)):
 299                 set_cookie_equal(ret2[i], expected[i])
 300         else:
 301             assert not ret

 304 def test_refresh_cookie():
 305 
 306     # Invalid expires format, sent to us by Reddit.
 307     c = "rfoo=bar; Domain=reddit.com; expires=Thu, 31 Dec 2037 23:59:59 GMT; Path=/"
 308     assert cookies.refresh_set_cookie_header(c, 60)
 309 
 310     c = "MOO=BAR; Expires=Tue, 08-Mar-2011 00:20:38 GMT; Path=foo.com; Secure"
 311     assert "00:21:38" in cookies.refresh_set_cookie_header(c, 60)
 312 
 313     c = "rfoo=bar; Domain=reddit.com; expires=Thu, 31 Dec 2037; Path=/"
 314     assert "expires" not in cookies.refresh_set_cookie_header(c, 60)
 315 
 316     c = "foo,bar"
 317     with pytest.raises(ValueError):
 318         cookies.refresh_set_cookie_header(c, 60)
 319 
 320     # https://github.com/mitmproxy/mitmproxy/issues/773
 321     c = ">=A"
 322     assert cookies.refresh_set_cookie_header(c, 60)
 323 
 324     # https://github.com/mitmproxy/mitmproxy/issues/1118
 325     c = "foo:bar=bla"
 326     assert cookies.refresh_set_cookie_header(c, 0)
 327     c = "foo/bar=bla"
 328     assert cookies.refresh_set_cookie_header(c, 0)
 329 
 330     # https://github.com/mitmproxy/mitmproxy/issues/2250
 331     c = ""
 332     assert cookies.refresh_set_cookie_header(c, 60) == ""

 335 @mock.patch('time.time')
 336 def test_get_expiration_ts(*args):
 337     # Freeze time
 338     now_ts = 17
 339     time.time.return_value = now_ts
 340 
 341     CA = cookies.CookieAttrs
 342     F = cookies.get_expiration_ts
 343 
 344     assert F(CA([("Expires", "Thu, 01-Jan-1970 00:00:00 GMT")])) == 0
 345     assert F(CA([("Expires", "Mon, 24-Aug-2037 00:00:00 GMT")])) == 2134684800
 346 
 347     assert F(CA([("Max-Age", "0")])) == now_ts
 348     assert F(CA([("Max-Age", "31")])) == now_ts + 31

 351 def test_is_expired():
 352     CA = cookies.CookieAttrs
 353 
 354     # A cookie can be expired
 355     # by setting the expire time in the past
 356     assert cookies.is_expired(CA([("Expires", "Thu, 01-Jan-1970 00:00:00 GMT")]))
 357 
 358     # or by setting Max-Age to 0
 359     assert cookies.is_expired(CA([("Max-Age", "0")]))
 360 
 361     # or both
 362     assert cookies.is_expired(CA([("Expires", "Thu, 01-Jan-1970 00:00:00 GMT"), ("Max-Age", "0")]))
 363 
 364     assert not cookies.is_expired(CA([("Expires", "Mon, 24-Aug-2037 00:00:00 GMT")]))
 365     assert not cookies.is_expired(CA([("Max-Age", "1")]))
 366     assert not cookies.is_expired(CA([("Expires", "Wed, 15-Jul-2037 00:00:00 GMT"), ("Max-Age", "1")]))
 367 
 368     assert not cookies.is_expired(CA([("Max-Age", "nan")]))
 369     assert not cookies.is_expired(CA([("Expires", "false")]))
