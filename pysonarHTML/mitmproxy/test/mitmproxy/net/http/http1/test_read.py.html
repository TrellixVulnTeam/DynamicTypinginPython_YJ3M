 161 def test_expected_http_body_size():
 162     # Expect: 100-continue
 163     assert expected_http_body_size(
 164         treq(headers=Headers(expect="100-continue", content_length="42")),
 165         expect_continue_as_0=True
 166     ) == 0
 167     # Expect: 100-continue
 168     assert expected_http_body_size(
 169         treq(headers=Headers(expect="100-continue", content_length="42")),
 170         expect_continue_as_0=False
 171     ) == 42
 172 
 173     # http://tools.ietf.org/html/rfc7230#section-3.3
 174     assert expected_http_body_size(
 175         treq(method=b"HEAD"),
 176         tresp(headers=Headers(content_length="42"))
 177     ) == 0
 178     assert expected_http_body_size(
 179         treq(method=b"CONNECT"),
 180         tresp()
 181     ) == 0
 182     for code in (100, 204, 304):
 183         assert expected_http_body_size(
 184             treq(),
 185             tresp(status_code=code)
 186         ) == 0
 187 
 188     # chunked
 189     assert expected_http_body_size(
 190         treq(headers=Headers(transfer_encoding="chunked")),
 191     ) is None
 192 
 193     # explicit length
 194     for val in (b"foo", b"-7"):
 195         with pytest.raises(exceptions.HttpSyntaxException):
 196             expected_http_body_size(
 197                 treq(headers=Headers(content_length=val))
 198             )
 199     assert expected_http_body_size(
 200         treq(headers=Headers(content_length="42"))
 201     ) == 42
 202 
 203     # more than 1 content-length headers with same value
 204     assert expected_http_body_size(
 205         treq(headers=Headers([(b'content-length', b'42'), (b'content-length', b'42')]))
 206     ) == 42
 207 
 208     # more than 1 content-length headers with conflicting value
 209     with pytest.raises(exceptions.HttpSyntaxException):
 210         expected_http_body_size(
 211             treq(headers=Headers([(b'content-length', b'42'), (b'content-length', b'45')]))
 212         )
 213 
 214     # no length
 215     assert expected_http_body_size(
 216         treq(headers=Headers())
 217     ) == 0
 218     assert expected_http_body_size(
 219         treq(headers=Headers()), tresp(headers=Headers())
 220     ) == -1

 240 def test_read_request_line():
 241     def t(b):
 242         return _read_request_line(BytesIO(b))
 243 
 244     assert (t(b"GET / HTTP/1.1") ==
 245             ("relative", b"GET", None, None, None, b"/", b"HTTP/1.1"))
 246     assert (t(b"OPTIONS * HTTP/1.1") ==
 247             ("relative", b"OPTIONS", None, None, None, b"*", b"HTTP/1.1"))
 248     assert (t(b"CONNECT foo:42 HTTP/1.1") ==
 249             ("authority", b"CONNECT", None, b"foo", 42, None, b"HTTP/1.1"))
 250     assert (t(b"GET http://foo:42/bar HTTP/1.1") ==
 251             ("absolute", b"GET", b"http", b"foo", 42, b"/bar", b"HTTP/1.1"))
 252 
 253     with pytest.raises(exceptions.HttpSyntaxException):
 254         t(b"GET / WTF/1.1")
 255     with pytest.raises(exceptions.HttpSyntaxException):
 256         t(b"this is not http")
 257     with pytest.raises(exceptions.HttpReadDisconnect):
 258         t(b"")

 362 def test_read_chunked():
 363     req = treq(content=None)
 364     req.headers["Transfer-Encoding"] = "chunked"
 365 
 366     data = b"1\r\na\r\n0\r\n"
 367     with pytest.raises(exceptions.HttpSyntaxException):
 368         b"".join(_read_chunked(BytesIO(data)))
 369 
 370     data = b"1\r\na\r\n0\r\n\r\n"
 371     assert b"".join(_read_chunked(BytesIO(data))) == b"a"
 372 
 373     data = b"\r\n\r\n1\r\na\r\n1\r\nb\r\n0\r\n\r\n"
 374     assert b"".join(_read_chunked(BytesIO(data))) == b"ab"
 375 
 376     data = b"\r\n"
 377     with pytest.raises(Exception, match="closed prematurely"):
 378         b"".join(_read_chunked(BytesIO(data)))
 379 
 380     data = b"1\r\nfoo"
 381     with pytest.raises(Exception, match="Malformed chunked body"):
 382         b"".join(_read_chunked(BytesIO(data)))
 383 
 384     data = b"foo\r\nfoo"
 385     with pytest.raises(exceptions.HttpSyntaxException):
 386         b"".join(_read_chunked(BytesIO(data)))
 387 
 388     data = b"5\r\naaaaa\r\n0\r\n\r\n"
 389     with pytest.raises(Exception, match="too large"):
 390         b"".join(_read_chunked(BytesIO(data), limit=2))
