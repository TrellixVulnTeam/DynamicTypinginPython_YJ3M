  54 def test_assemble_body():
  55     c = list(assemble_body(Headers(), [b"body"]))
  56     assert c == [b"body"]
  57 
  58     c = list(assemble_body(Headers(transfer_encoding="chunked"), [b"123456789a", b""]))
  59     assert c == [b"a\r\n123456789a\r\n", b"0\r\n\r\n"]
  60 
  61     c = list(assemble_body(Headers(transfer_encoding="chunked"), [b"123456789a"]))
  62     assert c == [b"a\r\n123456789a\r\n", b"0\r\n\r\n"]

  65 def test_assemble_request_line():
  66     assert _assemble_request_line(treq().data) == b"GET /path HTTP/1.1"
  67 
  68     authority_request = treq(method=b"CONNECT", first_line_format="authority").data
  69     assert _assemble_request_line(authority_request) == b"CONNECT address:22 HTTP/1.1"
  70 
  71     absolute_request = treq(first_line_format="absolute").data
  72     assert _assemble_request_line(absolute_request) == b"GET http://address:22/path HTTP/1.1"
  73 
  74     with pytest.raises(RuntimeError):
  75         _assemble_request_line(treq(first_line_format="invalid_form").data)
