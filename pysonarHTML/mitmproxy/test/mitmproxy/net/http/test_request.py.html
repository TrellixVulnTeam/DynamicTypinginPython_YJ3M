  42         r = Request.make("GET", "https://example.com/")
  43         assert r.method == "GET"
  44         assert r.scheme == "https"

  88     def test_host(self):
  89         request = treq()
  90         assert request.host == request.data.host.decode("idna")
  91 
  92         # Test IDNA encoding
  93         # Set str, get raw bytes
  94         request.host = "ídna.example"
  95         assert request.data.host == b"xn--dna-qma.example"
  96         # Set raw bytes, get decoded
  97         request.data.host = b"xn--idn-gla.example"
  98         assert request.host == "idná.example"
  99         # Set bytes, get raw bytes
 100         request.host = b"xn--dn-qia9b.example"
 101         assert request.data.host == b"xn--dn-qia9b.example"
 102         # IDNA encoding is not bijective
 103         request.host = "fußball"
 104         assert request.host == "fussball"
 105 
 106         # Don't fail on garbage
 107         request.data.host = b"foo\xFF\x00bar"
 108         assert request.host.startswith("foo")
 109         assert request.host.endswith("bar")
 110         # foo.bar = foo.bar should not cause any side effects.
 111         d = request.host
 112         request.host = d
 113         assert request.data.host == b"foo\xFF\x00bar"

 145     def test_modify_host_header(self):
 146         h1 = treq()
 147         assert "host" not in h1.headers
 148         assert ":authority" not in h1.headers
 149         h1.host_header = "example.com"
 150         assert "host" in h1.headers
 151         assert ":authority" not in h1.headers
 152         h1.host_header = None
 153         assert "host" not in h1.headers
 154 
 155         h2 = treq(http_version=b"HTTP/2.0")
 156         h2.host_header = "example.org"
 157         assert "host" not in h2.headers
 158         assert ":authority" in h2.headers
 159         del h2.host_header
 160         assert ":authority" not in h2.headers
 161 
 162         both_hdrs = treq(headers=(
 163             (b":authority", b"example.com"),

 165         ))
 166         both_hdrs.host_header = "foo.example.com"
 167         assert both_hdrs.headers["Host"] == "foo.example.com"
 168         assert both_hdrs.headers[":authority"] == "foo.example.com"
