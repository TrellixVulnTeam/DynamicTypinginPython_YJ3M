  67     def test_reason(self):
  68         resp = tresp()
  69         assert resp.reason == "OK"
  70 
  71         resp.reason = "ABC"
  72         assert resp.data.reason == b"ABC"
  73 
  74         resp.reason = b"DEF"
  75         assert resp.data.reason == b"DEF"
  76 
  77         resp.reason = None
  78         assert resp.data.reason is None
  79 
  80         resp.data.reason = b'cr\xe9e'
  81         assert resp.reason == "crée"
  82 
  83         # HTTP2 responses do not contain a reason phrase and self.data.reason will be None.
  84         # This should render to an empty reason phrase so that functions
  85         # expecting a string work properly.
  86         resp.data.reason = None
  87         assert resp.reason == ""

 104     def test_get_cookies_simple(self):
 105         resp = tresp()
 106         resp.headers = Headers(set_cookie="cookiename=cookievalue")
 107         result = resp.cookies
 108         assert len(result) == 1
 109         assert "cookiename" in result
 110         assert result["cookiename"] == ("cookievalue", CookieAttrs())
 111 
 112     def test_get_cookies_with_parameters(self):
 113         resp = tresp()
 114         cookie = "cookiename=cookievalue;domain=example.com;expires=Wed Oct  21 16:29:41 2015;path=/; HttpOnly"
 115         resp.headers = Headers(set_cookie=cookie)
 116         result = resp.cookies
 117         assert len(result) == 1
 118         assert "cookiename" in result
 119         assert result["cookiename"][0] == "cookievalue"
 120         attrs = result["cookiename"][1]
 121         assert len(attrs) == 4
 122         assert attrs["domain"] == "example.com"
 123         assert attrs["expires"] == "Wed Oct  21 16:29:41 2015"
 124         assert attrs["path"] == "/"
 125         assert attrs["httponly"] == ""

 127     def test_get_cookies_no_value(self):

 136     def test_get_cookies_twocookies(self):
 137         resp = tresp()
 138         resp.headers = Headers([
 139             [b"Set-Cookie", b"cookiename=cookievalue"],
 140             [b"Set-Cookie", b"othercookie=othervalue"]
 141         ])
 142         result = resp.cookies
 143         assert len(result) == 2
 144         assert "cookiename" in result
 145         assert result["cookiename"] == ("cookievalue", CookieAttrs())
 146         assert "othercookie" in result
 147         assert result["othercookie"] == ("othervalue", CookieAttrs())

 149     def test_set_cookies(self):
 150         resp = tresp()
 151         resp.cookies["foo"] = ("bar", {})
 152         assert len(resp.cookies) == 1
 153         assert resp.cookies["foo"] == ("bar", CookieAttrs())
 154         resp.cookies = [["one", ("uno", CookieAttrs())], ["two", ("due", CookieAttrs())]]
 155         assert list(resp.cookies.keys()) == ["one", "two"]
