   6 def test_decode():
   7     boundary = 'somefancyboundary'
   8     headers = Headers(
   9         content_type='multipart/form-data; boundary=' + boundary
  10     )
  11     content = (
  12         "--{0}\n"
  13         "Content-Disposition: form-data; name=\"field1\"\n\n"
  14         "value1\n"
  15         "--{0}\n"
  16         "Content-Disposition: form-data; name=\"field2\"\n\n"
  17         "value2\n"
  18         "--{0}--".format(boundary).encode()
  19     )
  20 
  21     form = multipart.decode(headers, content)
  22 
  23     assert len(form) == 2
  24     assert form[0] == (b"field1", b"value1")
  25     assert form[1] == (b"field2", b"value2")
  26

  31     result = multipart.decode(headers, content)
  32     assert result == []

  40 def test_encode():
  41     data = [("file".encode('utf-8'), "shell.jpg".encode('utf-8')),
  42             ("file_size".encode('utf-8'), "1000".encode('utf-8'))]
  43     headers = Headers(
  44         content_type='multipart/form-data; boundary=127824672498'
  45     )
  46     content = multipart.encode(headers, data)
  47 
  48     assert b'Content-Disposition: form-data; name="file"' in content
  49     assert b'Content-Type: text/plain; charset=utf-8\r\n\r\nshell.jpg\r\n\r\n--127824672498\r\n' in content
  50     assert b'1000\r\n\r\n--127824672498--\r\n'
  51     assert len(content) == 252
  52 
  53     with pytest.raises(ValueError, match=r"boundary found in encoded string"):
  54         multipart.encode(headers, [("key".encode('utf-8'), "--127824672498".encode('utf-8'))])
  55
