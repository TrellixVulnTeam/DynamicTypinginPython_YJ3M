  76 def test_cache():
  77     decode_gzip = mock.MagicMock()
  78     decode_gzip.return_value = b"decoded"
  79     encode_gzip = mock.MagicMock()
  80     encode_gzip.return_value = b"encoded"
  81 
  82     with mock.patch.dict(encoding.custom_decode, gzip=decode_gzip):
  83         with mock.patch.dict(encoding.custom_encode, gzip=encode_gzip):
  84             assert encoding.decode(b"encoded", "gzip") == b"decoded"
  85             assert decode_gzip.call_count == 1
  86 
  87             # should be cached
  88             assert encoding.decode(b"encoded", "gzip") == b"decoded"
  89             assert decode_gzip.call_count == 1
  90 
  91             # the other way around as well
  92             assert encoding.encode(b"decoded", "gzip") == b"encoded"
  93             assert encode_gzip.call_count == 0
  94 
  95             # different encoding
  96             decode_gzip.return_value = b"bar"
  97             assert encoding.encode(b"decoded", "deflate") != b"decoded"
  98             assert encode_gzip.call_count == 0
  99 
 100             # This is not in the cache anymore
 101             assert encoding.encode(b"decoded", "gzip") == b"encoded"
 102             assert encode_gzip.call_count == 1
