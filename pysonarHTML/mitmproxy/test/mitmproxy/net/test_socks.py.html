   9 def test_client_greeting():
  10     raw = tutils.treader(b"\x05\x02\x00\xBE\xEF")
  11     out = BytesIO()
  12     msg = socks.ClientGreeting.from_file(raw)
  13     msg.assert_socks5()
  14     msg.to_file(out)
  15 
  16     assert out.getvalue() == raw.getvalue()[:-1]
  17     assert msg.ver == 5
  18     assert len(msg.methods) == 2
  19     assert 0xBE in msg.methods
  20     assert 0xEF not in msg.methods

  88 def test_username_password_auth():
  89     raw = tutils.treader(b"\x01\x03usr\x03psd\xBE\xEF")
  90     out = BytesIO()
  91     auth = socks.UsernamePasswordAuth.from_file(raw)
  92     auth.assert_authver1()
  93     assert raw.read(2) == b"\xBE\xEF"
  94     auth.to_file(out)
  95 
  96     assert out.getvalue() == raw.getvalue()[:-2]
  97     assert auth.ver == socks.USERNAME_PASSWORD_VERSION.DEFAULT
  98     assert auth.username == "usr"

 164 def test_message_ipv6():
 165     # Test ATYP=0x04 (IPV6)
 166     ipv6_addr = u"2001:db8:85a3:8d3:1319:8a2e:370:7344"
 167 
 168     raw = tutils.treader(
 169         b"\x05\x01\x00\x04" +
 170         ipaddress.IPv6Address(ipv6_addr).packed +
 171         b"\xDE\xAD\xBE\xEF")
 172     out = BytesIO()
 173     msg = socks.Message.from_file(raw)
 174     assert raw.read(2) == b"\xBE\xEF"
 175     msg.to_file(out)
 176 
 177     assert out.getvalue() == raw.getvalue()[:-2]
 178     assert msg.addr[0] == ipv6_addr
