 155     @pytest.mark.parametrize("masked", [True, False])
 156     @pytest.mark.parametrize("length", [100, 50000, 150000])
 157     def test_serialization_bijection(self, masked, length):
 158         frame = websockets.Frame(
 159             os.urandom(length),
 160             fin=True,
 161             opcode=websockets.OPCODE.TEXT,
 162             mask=int(masked),
 163             masking_key=(os.urandom(4) if masked else None)
 164         )
 165         serialized = bytes(frame)
 166         assert frame == websockets.Frame.from_bytes(serialized)
