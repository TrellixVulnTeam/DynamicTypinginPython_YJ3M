  25 def sign(cert: str, subject: str):
  26     with open(f"openssl-{cert}.conf", "w") as f:
  27         f.write(textwrap.dedent(f"""
  28         authorityKeyIdentifier=keyid,issuer
  29         basicConstraints=CA:FALSE
  30         keyUsage = digitalSignature, keyEncipherment
  31         subjectAltName = {subject}
  32         """))
  33     do(f"openssl x509 -req -in {cert}.csr "
  34        f"-CA {ROOT_CA}.crt "
  35        f"-CAkey {ROOT_CA}.key "
  36        f"-CAcreateserial "
  37        f"-days 7300 "
  38        f"-sha256 "
  39        f"-extfile \"openssl-{cert}.conf\" "
  40        f"-out {cert}.crt"
  41        )
  42     os.remove(f"openssl-{cert}.conf")

  45 def mkcert(cert, subject):
  46     genrsa(cert)
  47     do(f"openssl req -new -nodes -batch "
  48        f"-key {cert}.key "
  49        f"-addext \"subjectAltName = {subject}\" "
  50        f"-out {cert}.csr"
  51        )
  52     sign(cert, subject)
  53     os.remove(f"{cert}.csr")
