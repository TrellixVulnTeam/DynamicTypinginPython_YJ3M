   6 def test_parse():
   7     assert server_spec.parse("example.com") == ("https", ("example.com", 443))
   8     assert server_spec.parse("example.com") == ("https", ("example.com", 443))
   9     assert server_spec.parse("http://example.com") == ("http", ("example.com", 80))
  10     assert server_spec.parse("http://127.0.0.1") == ("http", ("127.0.0.1", 80))
  11     assert server_spec.parse("http://[::1]") == ("http", ("::1", 80))
  12     assert server_spec.parse("http://[::1]/") == ("http", ("::1", 80))
  13     assert server_spec.parse("https://[::1]/") == ("https", ("::1", 443))
  14     assert server_spec.parse("http://[::1]:8080") == ("http", ("::1", 8080))
  15
