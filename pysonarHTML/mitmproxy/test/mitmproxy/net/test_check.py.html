   3 from mitmproxy.net import check
   4

   6 def test_is_valid_host():
   7     assert not check.is_valid_host(b"")
   8     assert not check.is_valid_host(b"xn--ke.ws")
   9     assert check.is_valid_host(b"one.two")
  10     assert not check.is_valid_host(b"one" * 255)
  11     assert check.is_valid_host(b"one.two.")
  12     # Allow underscore
  13     assert check.is_valid_host(b"one_two")
  14     assert check.is_valid_host(b"::1")
  15 
  16     # IP Address Validations
  17     assert check.is_valid_host(b'127.0.0.1')
  18     assert check.is_valid_host(b'2001:0db8:85a3:0000:0000:8a2e:0370:7334')
  19     assert check.is_valid_host(b'2001:db8:85a3:0:0:8a2e:370:7334')
  20     assert check.is_valid_host(b'2001:db8:85a3::8a2e:370:7334')
  21     assert not check.is_valid_host(b'2001:db8::85a3::7334')
  22     assert check.is_valid_host(b'2001-db8-85a3-8d3-1319-8a2e-370-7348.ipv6-literal.net')
  23 
  24     # TLD must be between 2 and 63 chars
  25     assert check.is_valid_host(b'example.tl')
  26     assert check.is_valid_host(b'example.tld')
  27     assert check.is_valid_host(b'example.' + b"x" * 63)
  28     assert not check.is_valid_host(b'example.' + b"x" * 64)
  29 
  30     # misc characters test
  31     assert not check.is_valid_host(b'ex@mple')
  32     assert not check.is_valid_host(b'ex@mple.com')
  33     assert not check.is_valid_host(b'example..com')
  34     assert not check.is_valid_host(b'.example.com')
  35     assert not check.is_valid_host(b'@.example.com')
  36     assert not check.is_valid_host(b'!.example.com')
  37 
  38     # Every label must be between 1 and 63 chars
  39     assert not check.is_valid_host(b'.tld')
  40     assert check.is_valid_host(b'x' * 1 + b'.tld')
  41     assert check.is_valid_host(b'x' * 30 + b'.tld')
  42     assert not check.is_valid_host(b'x' * 64 + b'.tld')
  43     assert check.is_valid_host(b'x' * 1 + b'.example.tld')
  44     assert check.is_valid_host(b'x' * 30 + b'.example.tld')
  45     assert not check.is_valid_host(b'x' * 64 + b'.example.tld')
  46 
  47     # Misc Underscore Test Cases
  48     assert check.is_valid_host(b'_example')
  49     assert check.is_valid_host(b'_example_')
  50     assert check.is_valid_host(b'example_')
  51     assert check.is_valid_host(b'_a.example.tld')
  52     assert check.is_valid_host(b'a_.example.tld')
  53     assert check.is_valid_host(b'_a_.example.tld')
  54 
  55     # Misc Dash/Hyphen/Minus Test Cases
  56     assert check.is_valid_host(b'-example')
  57     assert check.is_valid_host(b'-example_')
  58     assert check.is_valid_host(b'example-')
  59     assert check.is_valid_host(b'-a.example.tld')
  60     assert check.is_valid_host(b'a-.example.tld')
  61     assert check.is_valid_host(b'-a-.example.tld')
  62 
  63     # Misc Combo Test Cases
  64     assert check.is_valid_host(b'api-.example.com')
  65     assert check.is_valid_host(b'__a.example-site.com')
  66     assert check.is_valid_host(b'_-a.example-site.com')
  67     assert check.is_valid_host(b'_a_.example-site.com')
  68     assert check.is_valid_host(b'-a-.example-site.com')
  69     assert check.is_valid_host(b'api-.a.example.com')
  70     assert check.is_valid_host(b'api-._a.example.com')
  71     assert check.is_valid_host(b'api-.a_.example.com')
  72     assert check.is_valid_host(b'api-.ab.example.com')
