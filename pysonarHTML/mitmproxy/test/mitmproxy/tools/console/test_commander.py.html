  29     def test_cycle(self):
  30         tests = [
  31             [
  32                 "",
  33                 ["a", "b", "c"],
  34                 ["a", "b", "c", "a"],
  35                 ["c", "b", "a", "c"],
  36                 ["a", "c", "a", "c"]
  37             ],
  38             [
  39                 "xxx",
  40                 ["a", "b", "c"],
  41                 ["xxx", "xxx", "xxx"],
  42                 ["xxx", "xxx", "xxx"],
  43                 ["xxx", "xxx", "xxx"]
  44             ],
  45             [
  46                 "b",
  47                 ["a", "b", "ba", "bb", "c"],
  48                 ["b", "ba", "bb", "b"],
  49                 ["bb", "ba", "b", "bb"],
  50                 ["b", "bb", "b", "bb"]
  51             ],
  52         ]
  53         for start, opts, cycle, cycle_reverse, cycle_mix in tests:
  54             c = commander.ListCompleter(start, opts)
  55             for expected in cycle:
  56                 assert c.cycle() == expected
  57             for expected in cycle_reverse:
  58                 assert c.cycle(False) == expected
  59             forward = True
  60             for expected in cycle_mix:
  61                 assert c.cycle(forward) == expected
  62                 forward = not forward

 292 class TestCommandBuffer:
 293 
 294     def test_backspace(self):
 295         tests = [
 296             [("", 0), ("", 0)],
 297             [("1", 0), ("1", 0)],
 298             [("1", 1), ("", 0)],
 299             [("123", 3), ("12", 2)],
 300             [("123", 2), ("13", 1)],
 301             [("123", 0), ("123", 0)],
 302         ]
 303         with taddons.context() as commander_tctx:
 304             for start, output in tests:
 305                 cb = commander.CommandBuffer(commander_tctx.master)
 306                 cb.text, cb.cursor = start[0], start[1]
 307                 cb.backspace()
 308                 assert cb.text == output[0]
 309                 assert cb.cursor == output[1]
 310 
 311     def test_left(self):
 312         cursors = [3, 2, 1, 0, 0]
 313         with taddons.context() as commander_tctx:
 314             cb = commander.CommandBuffer(commander_tctx.master)
 315             cb.text, cb.cursor = "abcd", 4
 316             for c in cursors:
 317                 cb.left()
 318                 assert cb.cursor == c

 320     def test_right(self):
 321         cursors = [1, 2, 3, 4, 4]
 322         with taddons.context() as commander_tctx:
 323             cb = commander.CommandBuffer(commander_tctx.master)
 324             cb.text, cb.cursor = "abcd", 0
 325             for c in cursors:
 326                 cb.right()
 327                 assert cb.cursor == c

 329     def test_insert(self):
 330         tests = [
 331             [("", 0), ("x", 1)],
 332             [("a", 0), ("xa", 1)],
 333             [("xa", 2), ("xax", 3)],
 334         ]
 335         with taddons.context() as commander_tctx:
 336             for start, output in tests:
 337                 cb = commander.CommandBuffer(commander_tctx.master)
 338                 cb.text, cb.cursor = start[0], start[1]
 339                 cb.insert("x")
 340                 assert cb.text == output[0]
 341                 assert cb.cursor == output[1]

 343     def test_cycle_completion(self):
 344         with taddons.context() as commander_tctx:
 345             cb = commander.CommandBuffer(commander_tctx.master)
 346             cb.text = "foo bar"
 347             cb.cursor = len(cb.text)
 348             cb.cycle_completion()
 349 
 350             ce = commander.CommandEdit(commander_tctx.master, "se")
 351             ce.keypress(1, 'tab')
 352             ce.update()
 353             ret = ce.cbuf.render()
 354             assert ret == [
 355                 ('commander_command', 'set'),
 356                 ('text', ' '),
 357                 ('commander_hint', 'option '),
 358                 ('commander_hint', 'value '),
 359             ]

 361     def test_render(self):
 362         with taddons.context() as commander_tctx:
 363             cb = commander.CommandBuffer(commander_tctx.master)
 364             cb.text = "foo"
 365             assert cb.render()
 366 
 367             cb.text = "set view_filter '~bq test'"
 368             ret = cb.render()
 369             assert ret == [
 370                 ('commander_command', 'set'),
 371                 ('text', ' '),
 372                 ('text', 'view_filter'),
 373                 ('text', ' '),
 374                 ('text', "'~bq test'"),
 375             ]
 376 
 377             cb.text = "set"
 378             ret = cb.render()
 379             assert ret == [
 380                 ('commander_command', 'set'),
 381                 ('text', ' '),
 382                 ('commander_hint', 'option '),
 383                 ('commander_hint', 'value '),
