  12 def test_bind():

  14         km = keymap.Keymap(tctx.master)
  15         km.executor = mock.Mock()
  16

  40 def test_join():
  41     with taddons.context() as tctx:
  42         km = keymap.Keymap(tctx.master)
  43         km.add("key", "str", ["options"], "help1")
  44         km.add("key", "str", ["commands"])
  45 
  46         assert len(km.bindings) == 1
  47         assert len(km.bindings[0].contexts) == 2
  48         assert km.bindings[0].help == "help1"
  49         km.add("key", "str", ["commands"], "help2")
  50         assert len(km.bindings) == 1
  51         assert len(km.bindings[0].contexts) == 2
  52         assert km.bindings[0].help == "help2"
  53 
  54         assert km.get("commands", "key")
  55         km.unbind(km.bindings[0])
  56         assert len(km.bindings) == 0
  57         assert not km.get("commands", "key")

  75 def test_load_path(tmpdir):
  76     dst = str(tmpdir.join("conf"))
  77 
  78     kmc = keymap.KeymapConfig()
  79     with taddons.context(kmc) as tctx:
  80         km = keymap.Keymap(tctx.master)
  81         tctx.master.keymap = km
  82 
  83         with open(dst, 'wb') as f:
  84             f.write(b"\xff\xff\xff")
  85         with pytest.raises(keymap.KeyBindingError, match="expected UTF8"):
  86             kmc.load_path(km, dst)
  87 
  88         with open(dst, 'w') as f:
  89             f.write("'''")
  90         with pytest.raises(keymap.KeyBindingError):
  91             kmc.load_path(km, dst)
  92 
  93         with open(dst, 'w') as f:
  94             f.write(
  95                 """
  96                     -   key: key1
  97                         ctx: [unknown]
  98                         cmd: >
  99                             foo bar
 100                             foo bar
 101                 """
 102             )
 103         with pytest.raises(keymap.KeyBindingError):
 104             kmc.load_path(km, dst)
 105 
 106         with open(dst, 'w') as f:
 107             f.write(
 108                 """
 109                     -   key: key1
 110                         ctx: [chooser]
 111                         help: one
 112                         cmd: >
 113                             foo bar
 114                             foo bar
 115                 """
 116             )
 117         kmc.load_path(km, dst)
 118         assert(km.get("chooser", "key1"))
 119 
 120         with open(dst, 'w') as f:
 121             f.write(
 122                 """
 123                     -   key: key2
 124                         ctx: [flowlist]
 125                         cmd: foo
 126                     -   key: key2
 127                         ctx: [flowview]
 128                         cmd: bar
 129                 """
 130             )
 131         kmc.load_path(km, dst)
 132         assert(km.get("flowlist", "key2"))
 133         assert(km.get("flowview", "key2"))
 134 
 135         km.add("key123", "str", ["flowlist", "flowview"])
 136         with open(dst, 'w') as f:
 137             f.write(
 138                 """
 139                     -   key: key123
 140                         ctx: [options]
 141                         cmd: foo
 142                 """
 143             )
 144         kmc.load_path(km, dst)
 145         assert(km.get("flowlist", "key123"))
 146         assert(km.get("flowview", "key123"))
 147         assert(km.get("options", "key123"))

 150 def test_parse():
 151     kmc = keymap.KeymapConfig()
 152     with taddons.context(kmc):
 153         assert kmc.parse("") == []
 154         assert kmc.parse("\n\n\n   \n") == []
 155         with pytest.raises(keymap.KeyBindingError, match="expected a list of keys"):
 156             kmc.parse("key: val")
 157         with pytest.raises(keymap.KeyBindingError, match="expected a list of keys"):
 158             kmc.parse("val")
 159         with pytest.raises(keymap.KeyBindingError, match="Unknown key attributes"):
 160             kmc.parse(
 161                 """
 162                     -   key: key1
 163                         nonexistent: bar
 164                 """
 165             )
 166         with pytest.raises(keymap.KeyBindingError, match="Missing required key attributes"):
 167             kmc.parse(
 168                 """
 169                     -   help: key1
 170                 """
 171             )
 172         with pytest.raises(keymap.KeyBindingError, match="Invalid type for cmd"):
 173             kmc.parse(
 174                 """
 175                     -   key: key1
 176                         cmd: [ cmd ]
 177                 """
 178             )
 179         with pytest.raises(keymap.KeyBindingError, match="Invalid type for ctx"):
 180             kmc.parse(
 181                 """
 182                     -   key: key1
 183                         ctx: foo
 184                         cmd: cmd
 185                 """
 186             )
 187         assert kmc.parse(
 188             """
 189                 -   key: key1
 190                     ctx: [one, two]
 191                     help: one
 192                     cmd: >
 193                         foo bar
 194                         foo bar
 195             """
 196         ) == [{"key": "key1", "ctx": ["one", "two"], "help": "one", "cmd": "foo bar foo bar\n"}]
