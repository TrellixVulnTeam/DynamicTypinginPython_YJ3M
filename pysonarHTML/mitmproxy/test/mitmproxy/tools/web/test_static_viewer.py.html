  36 def test_save_flows(tmpdir):
  37     flows = [tflow.tflow(req=True, resp=None), tflow.tflow(req=True, resp=True)]
  38     static_viewer.save_flows(tmpdir, flows)
  39     assert tmpdir.join('flows.json').check(file=1)
  40     assert tmpdir.join('flows.json').read() == json.dumps([flow_to_json(f) for f in flows])

  43 @mock.patch('mitmproxy.ctx.log')
  44 def test_save_flows_content(ctx, tmpdir):
  45     flows = [tflow.tflow(req=True, resp=None), tflow.tflow(req=True, resp=True)]
  46     with mock.patch('time.time', mock.Mock(side_effect=[1, 2, 2] * 4)):
  47         static_viewer.save_flows_content(tmpdir, flows)
  48     flows_path = tmpdir.join('flows')
  49     assert len(flows_path.listdir()) == len(flows)
  50     for p in flows_path.listdir():
  51         assert p.join('request').check(dir=1)
  52         assert p.join('response').check(dir=1)
  53         assert p.join('request/content.data').check(file=1)
  54         assert p.join('request/content').check(dir=1)
  55         assert p.join('response/content.data').check(file=1)
  56         assert p.join('response/content').check(dir=1)
  57         assert p.join('request/content/Auto.json').check(file=1)
  58         assert p.join('response/content/Auto.json').check(file=1)

  61 @pytest.mark.asyncio
  62 async def test_static_viewer(tmpdir):
  63     s = static_viewer.StaticViewer()
  64     rf = readfile.ReadFile()
  65     sa = save.Save()
  66     with taddons.context(rf) as tctx:
  67         sa.save([tflow.tflow(resp=True)], str(tmpdir.join('foo')))
  68         tctx.master.addons.add(s)
  69         tctx.configure(s, web_static_viewer=str(tmpdir), rfile=str(tmpdir.join('foo')))
  70         assert tmpdir.join('index.html').check(file=1)
  71         assert tmpdir.join('static').check(dir=1)
  72         assert tmpdir.join('flows').check(dir=1)
