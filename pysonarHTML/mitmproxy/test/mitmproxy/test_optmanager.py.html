 146 def test_subscribe():
 147     o = TO()
 148     r = Rec()
 149 
 150     # pytest.raises keeps a reference here that interferes with the cleanup test
 151     # further down.
 152     try:
 153         o.subscribe(r, ["unknown"])
 154     except exceptions.OptionsError:
 155         pass
 156     else:
 157         raise AssertionError
 158 
 159     assert len(o.changed.receivers) == 0
 160 
 161     o.subscribe(r, ["two"])
 162     o.one = 2
 163     assert not r.called
 164     o.two = 3
 165     assert r.called
 166 
 167     assert len(o.changed.receivers) == 1
 168     del r
 169     o.two = 4
 170     assert len(o.changed.receivers) == 0

 172     class binder:
 173         def __init__(self):
 174             self.o = TO()
 175             self.called = False
 176             self.o.subscribe(self.bound, ["two"])

 179             self.called = True

 182     t.o.one = 3

 188 def test_rollback():
 189     o = TO()
 190 
 191     rec = []
 192 
 193     def sub(opts, updated):
 194         rec.append(copy.copy(opts))
 195 
 196     recerr = []
 197 
 198     def errsub(opts, **kwargs):
 199         recerr.append(kwargs)
 200 
 201     def err(opts, updated):
 202         if opts.one == 10:
 203             raise exceptions.OptionsError()
 204         if opts.bool is True:
 205             raise exceptions.OptionsError()
 206 
 207     o.changed.connect(sub)
 208     o.changed.connect(err)
 209     o.errored.connect(errsub)
 210 
 211     assert o.one is None
 212     with pytest.raises(exceptions.OptionsError):
 213         o.one = 10
 214     assert o.one is None
 215     with pytest.raises(exceptions.OptionsError):
 216         o.bool = True
 217     assert o.bool is False
 218     assert isinstance(recerr[0]["exc"], exceptions.OptionsError)
 219     assert o.one is None
 220     assert o.bool is False
 221     assert len(rec) == 4
 222     assert rec[0].one == 10
 223     assert rec[1].one is None
 224     assert rec[2].bool is True
 225     assert rec[3].bool is False
 226 
 227     with pytest.raises(exceptions.OptionsError):
 228         with o.rollback({"one"}, reraise=True):
 229             raise exceptions.OptionsError()
