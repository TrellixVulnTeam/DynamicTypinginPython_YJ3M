  14 def test_parse_size():
  15     assert human.parse_size("0") == 0
  16     assert human.parse_size("0b") == 0
  17     assert human.parse_size("1") == 1
  18     assert human.parse_size("1k") == 1024
  19     assert human.parse_size("1m") == 1024**2
  20     assert human.parse_size("1g") == 1024**3
  21     with pytest.raises(ValueError):
  22         human.parse_size("1f")
  23     with pytest.raises(ValueError):
  24         human.parse_size("ak")
  25     assert human.parse_size(None) is None

  28 def test_pretty_size():

  53 def test_format_address():
  54     assert human.format_address(("::1", "54010", "0", "0")) == "[::1]:54010"
  55     assert human.format_address(("::ffff:127.0.0.1", "54010", "0", "0")) == "127.0.0.1:54010"
  56     assert human.format_address(("127.0.0.1", "54010")) == "127.0.0.1:54010"
  57     assert human.format_address(("example.com", "54010")) == "example.com:54010"
  58     assert human.format_address(("::", "8080")) == "*:8080"
  59     assert human.format_address(("0.0.0.0", "8080")) == "*:8080"
  60     assert human.format_address(None) == "<no address>"
