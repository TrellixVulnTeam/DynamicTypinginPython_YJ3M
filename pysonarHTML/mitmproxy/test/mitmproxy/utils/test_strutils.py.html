   6 def test_always_bytes():
   7     assert strutils.always_bytes(bytes(range(256))) == bytes(range(256))
   8     assert strutils.always_bytes("foo") == b"foo"
   9     with pytest.raises(ValueError):
  10         strutils.always_bytes(u"\u2605", "ascii")
  11     with pytest.raises(TypeError):
  12         strutils.always_bytes(42, "ascii")

  44 def test_bytes_to_escaped_str():
  45     assert strutils.bytes_to_escaped_str(b"foo") == "foo"
  46     assert strutils.bytes_to_escaped_str(b"\b") == r"\x08"
  47     assert strutils.bytes_to_escaped_str(br"&!?=\)") == r"&!?=\\)"
  48     assert strutils.bytes_to_escaped_str(b'\xc3\xbc') == r"\xc3\xbc"
  49     assert strutils.bytes_to_escaped_str(b"'") == r"'"
  50     assert strutils.bytes_to_escaped_str(b'"') == r'"'
  51 
  52     assert strutils.bytes_to_escaped_str(b"'", escape_single_quotes=True) == r"\'"
  53     assert strutils.bytes_to_escaped_str(b'"', escape_single_quotes=True) == r'"'
  54 
  55     assert strutils.bytes_to_escaped_str(b"\r\n\t") == "\\r\\n\\t"
  56     assert strutils.bytes_to_escaped_str(b"\r\n\t", True) == "\r\n\t"
  57 
  58     assert strutils.bytes_to_escaped_str(b"\n", True) == "\n"
  59     assert strutils.bytes_to_escaped_str(b"\\n", True) == "\\ \\ n".replace(" ", "")
  60     assert strutils.bytes_to_escaped_str(b"\\\n", True) == "\\ \\ \n".replace(" ", "")
  61     assert strutils.bytes_to_escaped_str(b"\\\\n", True) == "\\ \\ \\ \\ n".replace(" ", "")
  62 
  63     with pytest.raises(ValueError):
  64         strutils.bytes_to_escaped_str(u"such unicode")

 102     "'" + strutils.SINGLELINE_CONTENT + strutils.NO_ESCAPE + "'",
 103     '"' + strutils.SINGLELINE_CONTENT + strutils.NO_ESCAPE + '"'

 107 def test_split_special_areas():
 108     assert strutils.split_special_areas("foo", ESCAPE_QUOTES) == ["foo"]
 109     assert strutils.split_special_areas("foo 'bar' baz", ESCAPE_QUOTES) == ["foo ", "'bar'", " baz"]
 110     assert strutils.split_special_areas(
 111         """foo 'b\\'a"r' baz""",
 112         ESCAPE_QUOTES
 113     ) == ["foo ", "'b\\'a\"r'", " baz"]
 114     assert strutils.split_special_areas(
 115         "foo\n/*bar\nbaz*/\nqux",
 116         [r'/\*[\s\S]+?\*/']
 117     ) == ["foo\n", "/*bar\nbaz*/", "\nqux"]
 118     assert strutils.split_special_areas(
 119         "foo\n//bar\nbaz",
 120         [r'//.+$']
 121     ) == ["foo\n", "//bar", "\nbaz"]

 124 def test_escape_special_areas():
 125     assert strutils.escape_special_areas('foo "bar" baz', ESCAPE_QUOTES, "*") == 'foo "bar" baz'
 126     esc = strutils.escape_special_areas('foo "b*r" b*z', ESCAPE_QUOTES, "*")
 127     assert esc == 'foo "b\ue02ar" b*z'
 128     assert strutils.unescape_special_areas(esc) == 'foo "b*r" b*z'
