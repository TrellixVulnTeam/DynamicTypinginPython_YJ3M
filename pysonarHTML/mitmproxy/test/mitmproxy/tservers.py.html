  89 class ProxyThread(threading.Thread):
  90 
  91     def __init__(self, masterclass, options):
  92         threading.Thread.__init__(self)
  93         self.masterclass = masterclass
  94         self.options = options
  95         self.tmaster = None
  96         self.event_loop = None
  97         controller.should_exit = False
  98 
  99     @property
 100     def port(self):
 101         return self.tmaster.server.address[1]
 102 
 103     @property
 104     def tlog(self):
 105         return self.tmaster.logs
 106 
 107     def shutdown(self):
 108         self.tmaster.shutdown()
 109 
 110     def run(self):
 111         self.event_loop = asyncio.new_event_loop()
 112         asyncio.set_event_loop(self.event_loop)
 113         self.tmaster = self.masterclass(self.options)
 114         self.tmaster.addons.add(core.Core())
 115         self.name = "ProxyThread (%s)" % human.format_address(self.tmaster.server.address)
 116         self.tmaster.run()
 117 
 118     def set_addons(self, *addons):
 119         self.tmaster.reset(addons)
 120 
 121     def start(self):
 122         super().start()
 123         while True:
 124             if self.tmaster:
 125                 break
 126             time.sleep(0.01)

 343 class HTTPUpstreamProxyTest(HTTPProxyTest):
 344     """
 345     Chain three instances of mitmproxy in a row to test upstream mode.
 346     Proxy order is cls.proxy -> cls.chain[0] -> cls.chain[1]
 347     cls.proxy and cls.chain[0] are in upstream mode,
 348     cls.chain[1] is in regular mode.
 349     """
 350     chain = None
 351     n = 2
 352 
 353     @classmethod
 354     def setup_class(cls):
 355         # We need to initialize the chain first so that the normal server gets a correct config.
 356         cls.chain = []
 357         for _ in range(cls.n):
 358             opts = cls.get_options()
 359             proxy = ProxyThread(cls.masterclass, opts)
 360             proxy.start()
 361             cls.chain.insert(0, proxy)
 362             while True:
 363                 if proxy.event_loop and proxy.event_loop.is_running():
 364                     break
 365 
 366         super().setup_class()
 367 
 368     @classmethod
 369     def teardown_class(cls):
 370         super().teardown_class()
 371         for proxy in cls.chain:
 372             proxy.shutdown()
 373 
 374     def setup(self):
 375         super().setup()
 376         for proxy in self.chain:
 377             proxy.tmaster.reset(self.addons())
 378 
 379     @classmethod
 380     def get_options(cls):
 381         opts = super().get_options()
 382         if cls.chain:  # First proxy is in normal mode.
 383             s = "http://127.0.0.1:%s" % cls.chain[0].port
 384             opts.update(
 385                 mode="upstream:" + s,
 386             )
 387         return opts
