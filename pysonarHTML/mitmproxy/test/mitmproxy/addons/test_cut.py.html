 101 def test_cut_save(tmpdir):
 102     f = str(tmpdir.join("path"))
 103     v = view.View()
 104     c = cut.Cut()
 105     with taddons.context() as tctx:
 106         tctx.master.addons.add(v, c)
 107         v.add([tflow.tflow(resp=True)])
 108 
 109         tctx.command(c.save, "@all", "request.method", f)
 110         assert qr(f) == b"GET"
 111         tctx.command(c.save, "@all", "request.content", f)
 112         assert qr(f) == b"content"
 113         tctx.command(c.save, "@all", "request.content", "+" + f)
 114         assert qr(f) == b"content\ncontent"
 115 
 116         v.add([tflow.tflow(resp=True)])
 117         tctx.command(c.save, "@all", "request.method", f)
 118         assert qr(f).splitlines() == [b"GET", b"GET"]
 119         tctx.command(c.save, "@all", "request.method,request.content", f)
 120         assert qr(f).splitlines() == [b"GET,content", b"GET,content"]

 143 def test_cut():
 144     c = cut.Cut()
 145     with taddons.context():
 146         tflows = [tflow.tflow(resp=True)]
 147         assert c.cut(tflows, ["request.method"]) == [["GET"]]
 148         assert c.cut(tflows, ["request.scheme"]) == [["http"]]
 149         assert c.cut(tflows, ["request.host"]) == [["address"]]
 150         assert c.cut(tflows, ["request.port"]) == [["22"]]
 151         assert c.cut(tflows, ["request.path"]) == [["/path"]]
 152         assert c.cut(tflows, ["request.url"]) == [["http://address:22/path"]]
 153         assert c.cut(tflows, ["request.content"]) == [[b"content"]]
 154         assert c.cut(tflows, ["request.header[header]"]) == [["qvalue"]]
 155         assert c.cut(tflows, ["request.header[unknown]"]) == [[""]]
 156 
 157         assert c.cut(tflows, ["response.status_code"]) == [["200"]]
 158         assert c.cut(tflows, ["response.reason"]) == [["OK"]]
 159         assert c.cut(tflows, ["response.content"]) == [[b"message"]]
 160         assert c.cut(tflows, ["response.header[header-response]"]) == [["svalue"]]
 161         assert c.cut(tflows, ["moo"]) == [[""]]
 162         with pytest.raises(exceptions.CommandError):
 163             assert c.cut(tflows, ["__dict__"]) == [[""]]
 164 
 165     with taddons.context():
 166         tflows = [tflow.tflow(resp=False)]
 167         assert c.cut(tflows, ["response.reason"]) == [[""]]
 168         assert c.cut(tflows, ["response.header[key]"]) == [[""]]
 169 
 170     c = cut.Cut()
 171     with taddons.context():
 172         tflows = [tflow.ttcpflow()]
 173         assert c.cut(tflows, ["request.method"]) == [[""]]
 174         assert c.cut(tflows, ["response.status"]) == [[""]]
