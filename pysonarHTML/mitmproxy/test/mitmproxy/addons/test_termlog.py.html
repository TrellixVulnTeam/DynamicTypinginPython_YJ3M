   9 class TestTermLog:
  10     @pytest.mark.usefixtures('capfd')
  11     @pytest.mark.parametrize('outfile, expected_out, expected_err', [
  12         (None, ['one', 'three'], ['four']),
  13         (sys.stdout, ['one', 'three', 'four'], []),
  14         (sys.stderr, [], ['one', 'three', 'four']),
  15     ])
  16     def test_output(self, outfile, expected_out, expected_err, capfd):
  17         t = termlog.TermLog(outfile=outfile)
  18         with taddons.context(t) as tctx:
  19             tctx.options.termlog_verbosity = "info"
  20             tctx.configure(t)
  21             t.log(log.LogEntry("one", "info"))
  22             t.log(log.LogEntry("two", "debug"))
  23             t.log(log.LogEntry("three", "warn"))
  24             t.log(log.LogEntry("four", "error"))
  25         out, err = capfd.readouterr()
  26         assert out.strip().splitlines() == expected_out
  27         assert err.strip().splitlines() == expected_err
