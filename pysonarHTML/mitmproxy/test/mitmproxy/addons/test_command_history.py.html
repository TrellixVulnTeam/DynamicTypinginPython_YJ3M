   8     def test_load_and_save(self, tmpdir):
   9         history_file = tmpdir.join('command_history')
  10         commands = ["cmd1", "cmd2", "cmd3"]
  11         with open(history_file, 'w') as f:
  12             f.write("\n".join(commands))
  13 
  14         ch = command_history.CommandHistory()
  15         ch.VACUUM_SIZE = 4
  16         with taddons.context(ch) as tctx:
  17             tctx.options.confdir = str(tmpdir)
  18             assert ch.history == commands
  19             ch.add_command("cmd4")
  20             ch.done()
  21 
  22         with open(history_file, "r") as f:
  23             assert f.read() == "cmd3\ncmd4\n"

 171     def test_multiple_instances(self, tmpdir):
 172         ch = command_history.CommandHistory()
 173         with taddons.context(ch) as tctx:
 174             tctx.options.confdir = str(tmpdir)
 175 
 176         instances = [
 177             command_history.CommandHistory(),
 178             command_history.CommandHistory(),
 179             command_history.CommandHistory()
 180         ]
 181 
 182         for i in instances:
 183             i.configure('command_history')
 184             saved_commands = i.get_history()
 185             assert saved_commands == []
 186 
 187         instances[0].add_command('cmd1')
 188         saved_commands = instances[0].get_history()
 189         assert saved_commands == ['cmd1']
 190 
 191         # These instances haven't yet added a new command, so they haven't
 192         # yet reloaded their commands from the command file.
 193         # This is expected, because if the user is filtering a command on
 194         # another window, we don't want to interfere with that
 195         saved_commands = instances[1].get_history()
 196         assert saved_commands == []
 197         saved_commands = instances[2].get_history()
 198         assert saved_commands == []
 199 
 200         # Since the second instanced added a new command, its list of
 201         # saved commands has been updated to have the commands from the
 202         # first instance + its own commands
 203         instances[1].add_command('cmd2')
 204         saved_commands = instances[1].get_history()
 205         assert saved_commands == ['cmd2']
 206 
 207         saved_commands = instances[0].get_history()
 208         assert saved_commands == ['cmd1']
 209 
 210         # Third instance is still empty as it has not yet ran any command
 211         saved_commands = instances[2].get_history()
 212         assert saved_commands == []
 213 
 214         instances[2].add_command('cmd3')
 215         saved_commands = instances[2].get_history()
 216         assert saved_commands == ['cmd3']
 217 
 218         instances[0].add_command('cmd4')
 219         saved_commands = instances[0].get_history()
 220         assert saved_commands == ['cmd1', 'cmd4']
 221 
 222         instances.append(command_history.CommandHistory())
 223         instances[3].configure('command_history')
 224         saved_commands = instances[3].get_history()
 225         assert saved_commands == ['cmd1', 'cmd2', 'cmd3', 'cmd4']
 226 
 227         instances[0].add_command('cmd_before_close')
 228         instances.pop(0).done()
 229 
 230         saved_commands = instances[0].get_history()
 231         assert saved_commands == ['cmd2']
 232 
 233         instances[0].add_command('new_cmd')
 234         saved_commands = instances[0].get_history()
 235         assert saved_commands == ['cmd2', 'new_cmd']
 236 
 237         instances.pop(0).done()
 238         instances.pop(0).done()
 239         instances.pop(0).done()
 240 
 241         _path = os.path.join(tctx.options.confdir, 'command_history')
 242         lines = open(_path, 'r').readlines()
 243         saved_commands = [cmd.strip() for cmd in lines]
 244         assert saved_commands == ['cmd1', 'cmd2', 'cmd3', 'cmd4', 'cmd_before_close', 'new_cmd']
 245 
 246         instances = [
 247             command_history.CommandHistory(),
 248             command_history.CommandHistory()
 249         ]
 250 
 251         for i in instances:
 252             i.configure('command_history')
 253             i.clear_history()
 254             saved_commands = i.get_history()
 255             assert saved_commands == []
 256 
 257         instances[0].add_command('cmd1')
 258         instances[0].add_command('cmd2')
 259         instances[1].add_command('cmd3')
 260         instances[1].add_command('cmd4')
 261         instances[1].add_command('cmd5')
 262 
 263         saved_commands = instances[1].get_history()
 264         assert saved_commands == ['cmd3', 'cmd4', 'cmd5']
 265 
 266         instances.pop().done()
 267         instances.pop().done()
 268 
 269         _path = os.path.join(tctx.options.confdir, 'command_history')
 270         lines = open(_path, 'r').readlines()
 271         saved_commands = [cmd.strip() for cmd in lines]
 272         assert saved_commands == ['cmd1', 'cmd2', 'cmd3', 'cmd4', 'cmd5']
