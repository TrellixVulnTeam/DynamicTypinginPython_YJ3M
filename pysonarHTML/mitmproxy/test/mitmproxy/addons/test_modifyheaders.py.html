   9 def test_parse_modify_spec():
  10     spec = parse_modify_spec("/foo/bar/voing", True)
  11     assert spec.matches.pattern == "foo"
  12     assert spec.subject == b"bar"
  13     assert spec.read_replacement() == b"voing"
  14 
  15     spec = parse_modify_spec("/foo/bar/vo/ing/", False)
  16     assert spec.matches.pattern == "foo"
  17     assert spec.subject == b"bar"
  18     assert spec.read_replacement() == b"vo/ing/"
  19 
  20     spec = parse_modify_spec("/bar/voing", False)
  21     assert spec.matches(tflow.tflow())
  22     assert spec.subject == b"bar"
  23     assert spec.read_replacement() == b"voing"
  24

 127 class TestModifyHeadersFile:
 128     def test_simple(self, tmpdir):
 129         mh = ModifyHeaders()
 130         with taddons.context(mh) as tctx:
 131             tmpfile = tmpdir.join("replacement")
 132             tmpfile.write("two")
 133             tctx.configure(
 134                 mh,
 135                 modify_headers=["/~q/one/@" + str(tmpfile)]
 136             )
 137             f = tflow.tflow()
 138             f.request.headers["one"] = "xxx"
 139             mh.request(f)
 140             assert f.request.headers["one"] == "two"
