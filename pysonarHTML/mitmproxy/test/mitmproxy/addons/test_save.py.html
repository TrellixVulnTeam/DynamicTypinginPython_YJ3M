  12 def test_configure(tmpdir):
  13     sa = save.Save()
  14     with taddons.context(sa) as tctx:
  15         with pytest.raises(exceptions.OptionsError):
  16             tctx.configure(sa, save_stream_file=str(tmpdir))
  17         with pytest.raises(Exception, match="Invalid filter"):
  18             tctx.configure(
  19                 sa, save_stream_file=str(tmpdir.join("foo")), save_stream_filter="~~"
  20             )
  21         tctx.configure(sa, save_stream_filter="foo")
  22         assert sa.filt
  23         tctx.configure(sa, save_stream_filter=None)
  24         assert not sa.filt

  27 def rd(p):
  28     with open(p, "rb") as f:
  29         x = io.FlowReader(f)
  30         return list(x.stream())

  59 def test_save_command(tmpdir):
  60     sa = save.Save()
  61     with taddons.context() as tctx:
  62         p = str(tmpdir.join("foo"))
  63         sa.save([tflow.tflow(resp=True)], p)
  64         assert len(rd(p)) == 1
  65         sa.save([tflow.tflow(resp=True)], p)
  66         assert len(rd(p)) == 1
  67         sa.save([tflow.tflow(resp=True)], "+" + p)
  68         assert len(rd(p)) == 2
  69 
  70         with pytest.raises(exceptions.CommandError):
  71             sa.save([tflow.tflow(resp=True)], str(tmpdir))
  72 
  73         v = view.View()
  74         tctx.master.addons.add(v)
  75         tctx.master.addons.add(sa)
  76         tctx.master.commands.execute("save.file @shown %s" % p)
