 416 def test_signals():
 417     v = view.View()
 418     rec_add = Record()
 419     rec_update = Record()
 420     rec_remove = Record()
 421     rec_refresh = Record()
 422 
 423     def clearrec():
 424         rec_add.calls = []
 425         rec_update.calls = []
 426         rec_remove.calls = []
 427         rec_refresh.calls = []
 428 
 429     v.sig_view_add.connect(rec_add)
 430     v.sig_view_update.connect(rec_update)
 431     v.sig_view_remove.connect(rec_remove)
 432     v.sig_view_refresh.connect(rec_refresh)
 433 
 434     assert not any([rec_add, rec_update, rec_remove, rec_refresh])
 435 
 436     # Simple add
 437     v.add([tft()])
 438     assert rec_add
 439     assert not any([rec_update, rec_remove, rec_refresh])
 440 
 441     # Filter change triggers refresh
 442     clearrec()
 443     v.set_filter(flowfilter.parse("~m put"))
 444     assert rec_refresh
 445     assert not any([rec_update, rec_add, rec_remove])
 446 
 447     v.set_filter(flowfilter.parse("~m get"))
 448 
 449     # An update that results in a flow being added to the view
 450     clearrec()
 451     v[0].request.method = "PUT"
 452     v.update([v[0]])
 453     assert rec_remove
 454     assert not any([rec_update, rec_refresh, rec_add])
 455 
 456     # An update that does not affect the view just sends update
 457     v.set_filter(flowfilter.parse("~m put"))
 458     clearrec()
 459     v.update([v[0]])
 460     assert rec_update
 461     assert not any([rec_remove, rec_refresh, rec_add])
 462 
 463     # An update for a flow in state but not view does not do anything
 464     f = v[0]
 465     v.set_filter(flowfilter.parse("~m get"))
 466     assert not len(v)
 467     clearrec()
 468     v.update([f])
 469     assert not any([rec_add, rec_update, rec_remove, rec_refresh])
