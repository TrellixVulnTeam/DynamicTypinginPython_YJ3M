  66 def test_flow_set():
  67     sa = core.Core()
  68     with taddons.context(loadcore=False):
  69         f = tflow.tflow(resp=True)
  70         assert sa.flow_set_options()
  71 
  72         assert f.request.method != "post"
  73         sa.flow_set([f], "method", "post")
  74         assert f.request.method == "POST"
  75 
  76         assert f.request.host != "testhost"
  77         sa.flow_set([f], "host", "testhost")
  78         assert f.request.host == "testhost"
  79 
  80         assert f.request.path != "/test/path"
  81         sa.flow_set([f], "path", "/test/path")
  82         assert f.request.path == "/test/path"
  83 
  84         assert f.request.url != "http://foo.com/bar"
  85         sa.flow_set([f], "url", "http://foo.com/bar")
  86         assert f.request.url == "http://foo.com/bar"
  87         with pytest.raises(exceptions.CommandError):
  88             sa.flow_set([f], "url", "oink")
  89 
  90         assert f.response.status_code != 404
  91         sa.flow_set([f], "status_code", "404")
  92         assert f.response.status_code == 404
  93         assert f.response.reason == "Not Found"
  94         with pytest.raises(exceptions.CommandError):
  95             sa.flow_set([f], "status_code", "oink")
  96 
  97         assert f.response.reason != "foo"
  98         sa.flow_set([f], "reason", "foo")
  99         assert f.response.reason == "foo"

 151         with open(p, 'a') as f:

 157 def test_validation_simple():
 158     sa = core.Core()
 159     with taddons.context() as tctx:
 160         with pytest.raises(exceptions.OptionsError):
 161             tctx.configure(sa, body_size_limit = "invalid")
 162         tctx.configure(sa, body_size_limit = "1m")
 163 
 164         with pytest.raises(exceptions.OptionsError, match="mutually exclusive"):
 165             tctx.configure(
 166                 sa,
 167                 add_upstream_certs_to_client_chain = True,
 168                 upstream_cert = False
 169             )
 170         with pytest.raises(exceptions.OptionsError, match="requires certificate verification to be disabled"):
 171             tctx.configure(
 172                 sa,
 173                 add_upstream_certs_to_client_chain = True,
 174                 ssl_insecure = False
 175             )
 176         with pytest.raises(exceptions.OptionsError, match="Invalid mode"):
 177             tctx.configure(
 178                 sa,
 179                 mode = "Flibble"
 180             )
