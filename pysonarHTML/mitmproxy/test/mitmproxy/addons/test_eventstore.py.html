   6 def test_simple():
   7     store = eventstore.EventStore()
   8     assert not store.data
   9 
  10     sig_add = mock.Mock(spec=lambda: 42)
  11     sig_refresh = mock.Mock(spec=lambda: 42)
  12     store.sig_add.connect(sig_add)
  13     store.sig_refresh.connect(sig_refresh)
  14 
  15     assert not sig_add.called
  16     assert not sig_refresh.called
  17 
  18     # test .log()
  19     store.log(log.LogEntry("test", "info"))
  20     assert store.data
  21 
  22     assert sig_add.called
  23     assert not sig_refresh.called
  24 
  25     # test .clear()
  26     sig_add.reset_mock()

  35 def test_max_size():
  36     store = eventstore.EventStore(3)
  37     assert store.size == 3
  38     store.log(log.LogEntry("foo", "info"))
  39     store.log(log.LogEntry("bar", "info"))
  40     store.log(log.LogEntry("baz", "info"))
  41     assert len(store.data) == 3
  42     assert ["foo", "bar", "baz"] == [x.msg for x in store.data]
  43 
  44     # overflow
  45     store.log(log.LogEntry("boo", "info"))
  46     assert len(store.data) == 3
  47     assert ["bar", "baz", "boo"] == [x.msg for x in store.data]
