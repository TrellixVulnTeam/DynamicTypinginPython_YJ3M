  54 @pytest.mark.asyncio
  55 async def test_block_global(block_global, block_private, should_be_killed, address):
  56     ar = block.Block()
  57     with taddons.context(ar) as tctx:
  58         tctx.options.block_global = block_global
  59         tctx.options.block_private = block_private
  60         with mock.patch('mitmproxy.proxy.protocol.base.Layer') as layer:
  61             layer.client_conn.address = address
  62             ar.clientconnect(layer)
  63             if should_be_killed:
  64                 assert layer.reply.kill.called
  65                 assert await tctx.master.await_log("killed", "warn")
  66             else:
  67                 assert not layer.reply.kill.called
