  15 def test_configure():
  16     d = dumper.Dumper()
  17     with taddons.context(d) as ctx:
  18         ctx.configure(d, dumper_filter="~b foo")
  19         assert d.filter
  20 
  21         f = tflow.tflow(resp=True)
  22         assert not d.match(f)

 116 def test_echo_body():

 131 def test_echo_request_line():
 132     sio = io.StringIO()
 133     sio_err = io.StringIO()
 134     d = dumper.Dumper(sio, sio_err)
 135     with taddons.context(d) as ctx:
 136         ctx.configure(d, flow_detail=3, showhost=True)
 137         f = tflow.tflow(client_conn=None, server_conn=True, resp=True)
 138         f.request.is_replay = True
 139         d._echo_request_line(f)
 140         assert "[replay]" in sio.getvalue()
 141         sio.truncate(0)
 142 
 143         f = tflow.tflow(client_conn=None, server_conn=True, resp=True)
 144         f.request.is_replay = False
 145         d._echo_request_line(f)
 146         assert "[replay]" not in sio.getvalue()
 147         sio.truncate(0)
 148 
 149         f = tflow.tflow(client_conn=None, server_conn=True, resp=True)
 150         f.request.http_version = "nonstandard"
 151         d._echo_request_line(f)
 152         assert "nonstandard" in sio.getvalue()
 153         sio.truncate(0)
 154 
 155         ctx.configure(d, flow_detail=0, showhost=True)
 156         f = tflow.tflow(client_conn=None, server_conn=True, resp=True)
 157         terminalWidth = max(shutil.get_terminal_size()[0] - 25, 50)
 158         f.request.url = "http://address:22/" + ("x" * terminalWidth) + "textToBeTruncated"
 159         d._echo_request_line(f)
 160         assert "textToBeTruncated" not in sio.getvalue()
 161         sio.truncate(0)

 183         ctx.configure(d, flow_detail=3, showhost=True)
