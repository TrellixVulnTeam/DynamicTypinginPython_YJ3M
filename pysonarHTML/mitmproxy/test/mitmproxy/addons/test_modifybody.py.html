   8 class TestModifyBody:
   9     def test_configure(self):
  10         mb = modifybody.ModifyBody()
  11         with taddons.context(mb) as tctx:
  12             tctx.configure(mb, modify_body=["one/two/three"])
  13             with pytest.raises(Exception, match="Cannot parse modify_body"):
  14                 tctx.configure(mb, modify_body=["/"])
  15             tctx.configure(mb, modify_body=["/a/b/c/"])

  17     def test_simple(self):
  18         mb = modifybody.ModifyBody()
  19         with taddons.context(mb) as tctx:
  20             tctx.configure(
  21                 mb,
  22                 modify_body=[
  23                     "/~q/foo/bar",
  24                     "/~s/foo/bar",
  25                 ]
  26             )
  27             f = tflow.tflow()
  28             f.request.content = b"foo"
  29             mb.request(f)

  31 
  32             f = tflow.tflow(resp=True)
  33             f.response.content = b"foo"

  55 class TestModifyBodyFile:
  56     def test_simple(self, tmpdir):
  57         mb = modifybody.ModifyBody()
  58         with taddons.context(mb) as tctx:
  59             tmpfile = tmpdir.join("replacement")
  60             tmpfile.write("bar")
  61             tctx.configure(
  62                 mb,
  63                 modify_body=["/~q/foo/@" + str(tmpfile)]
  64             )
  65             f = tflow.tflow()
  66             f.request.content = b"foo"
  67             mb.request(f)
  68             assert f.request.content == b"bar"
