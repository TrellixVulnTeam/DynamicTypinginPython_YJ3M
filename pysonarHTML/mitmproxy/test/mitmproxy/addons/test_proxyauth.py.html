 133     def test_authenticate(self):
 134         up = proxyauth.ProxyAuth()
 135         with taddons.context(up, loadcore=False) as ctx:
 136             ctx.configure(up, proxyauth="any", mode="regular")
 137 
 138             f = tflow.tflow()
 139             assert not f.response
 140             up.authenticate(f)
 141             assert f.response.status_code == 407
 142 
 143             f = tflow.tflow()
 144             f.request.headers["Proxy-Authorization"] = proxyauth.mkauth(
 145                 "test", "test"
 146             )
 147             up.authenticate(f)
 148             assert not f.response
 149             assert not f.request.headers.get("Proxy-Authorization")
 150 
 151             f = tflow.tflow()
 152             ctx.configure(up, mode="reverse")
 153             assert not f.response
 154             up.authenticate(f)
 155             assert f.response.status_code == 401
 156 
 157             f = tflow.tflow()
 158             f.request.headers["Authorization"] = proxyauth.mkauth(
 159                 "test", "test"
 160             )
 161             up.authenticate(f)
 162             assert not f.response
 163             assert not f.request.headers.get("Authorization")

 165     def test_configure(self, tdata):
 166         up = proxyauth.ProxyAuth()
 167         with taddons.context(up) as ctx:
 168             with pytest.raises(exceptions.OptionsError):
 169                 ctx.configure(up, proxyauth="foo")
 170 
 171             ctx.configure(up, proxyauth="foo:bar")
 172             assert up.singleuser == ["foo", "bar"]
 173 
 174             ctx.configure(up, proxyauth=None)
 175             assert up.singleuser is None
 176 
 177             ctx.configure(up, proxyauth="any")
 178             assert up.nonanonymous
 179             ctx.configure(up, proxyauth=None)
 180             assert not up.nonanonymous
 181 
 182             with mock.patch('ldap3.Server', return_value="ldap://fake_server:389 - cleartext"):
 183                 with mock.patch('ldap3.Connection', return_value="test"):
 184                     ctx.configure(up, proxyauth="ldap:localhost:cn=default,dc=cdhdt,dc=com:password:ou=application,dc=cdhdt,dc=com")
 185                     assert up.ldapserver
 186                     ctx.configure(up, proxyauth="ldaps:localhost:cn=default,dc=cdhdt,dc=com:password:ou=application,dc=cdhdt,dc=com")
 187                     assert up.ldapserver
 188 
 189             with pytest.raises(exceptions.OptionsError):
 190                 ctx.configure(up, proxyauth="ldap:test:test:test")
 191 
 192             with pytest.raises(exceptions.OptionsError):
 193                 ctx.configure(up, proxyauth="ldap:fake_serveruid=?dc=example,dc=com:person")
 194 
 195             with pytest.raises(exceptions.OptionsError):
 196                 ctx.configure(up, proxyauth="ldapssssssss:fake_server:dn:password:tree")
 197 
 198             with pytest.raises(exceptions.OptionsError):
 199                 ctx.configure(
 200                     up,
 201                     proxyauth= "@" + tdata.path("mitmproxy/net/data/server.crt")
 202                 )
 203             with pytest.raises(exceptions.OptionsError):
 204                 ctx.configure(up, proxyauth="@nonexistent")
 205 
 206             ctx.configure(
 207                 up,
 208                 proxyauth= "@" + tdata.path(
 209                     "mitmproxy/net/data/htpasswd"
 210                 )
 211             )
 212             assert up.htpasswd
 213             assert up.htpasswd.check_password("test", "test")
 214             assert not up.htpasswd.check_password("test", "foo")
 215             ctx.configure(up, proxyauth=None)
 216             assert not up.htpasswd
 217 
 218             with pytest.raises(exceptions.OptionsError):
 219                 ctx.configure(up, proxyauth="any", mode="transparent")
 220             with pytest.raises(exceptions.OptionsError):
 221                 ctx.configure(up, proxyauth="any", mode="socks5")

 223     def test_handlers(self):
 224         up = proxyauth.ProxyAuth()
 225         with taddons.context(up) as ctx:
 226             ctx.configure(up, proxyauth="any", mode="regular")
 227 
 228             f = tflow.tflow()
 229             assert not f.response
 230             up.requestheaders(f)
 231             assert f.response.status_code == 407
 232 
 233             f = tflow.tflow()
 234             f.request.method = "CONNECT"
 235             assert not f.response
 236             up.http_connect(f)
 237             assert f.response.status_code == 407
 238 
 239             f = tflow.tflow()
 240             f.request.method = "CONNECT"
 241             f.request.headers["Proxy-Authorization"] = proxyauth.mkauth(
 242                 "test", "test"
 243             )
 244             up.http_connect(f)
 245             assert not f.response
 246 
 247             f2 = tflow.tflow(client_conn=f.client_conn)
 248             up.requestheaders(f2)
 249             assert not f2.response
 250             assert f2.metadata["proxyauth"] == ('test', 'test')
