  27 class TBase(tservers.HTTPProxyTest):
  28     @staticmethod
  29     def wait_response(flow):
  30         """
  31         Race condition: We don't want to replay the flow while it is still live.
  32         """
  33         s = time.time()
  34         while True:
  35             if flow.response or flow.error:
  36                 flow.server_conn.close()
  37                 break
  38             time.sleep(0.001)
  39             if time.time() - s > 5:
  40                 raise RuntimeError("Flow is live for too long.")

  43     def reset(f):
  44         f.live = False
  45         f.repsonse = False
  46         f.error = False

  51     def test_replay(self):
  52         cr = self.master.addons.get("clientplayback")
  53 
  54         assert self.pathod("304").status_code == 304
  55         assert len(self.master.state.flows) == 1
  56         l = self.master.state.flows[-1]
  57         assert l.response.status_code == 304
  58         l.request.path = "/p/305"
  59         l.response = None
  60         l.live = False
  61         l.intercepted = False
  62 
  63         cr.start_replay([l])
  64         self.wait_response(l)
  65         assert l.response.status_code == 305
  66 
  67         # Disconnect error
  68         cr.stop_replay()
  69         self.reset(l)
  70         l.request.path = "/p/305:d0"
  71         cr.start_replay([l])
  72         self.wait_response(l)
  73         if isinstance(self, tservers.HTTPUpstreamProxyTest):
  74             assert l.response.status_code == 502
  75         else:
  76             assert l.error
  77 
  78         # # Port error
  79         cr.stop_replay()
  80         self.reset(l)
  81         l.request.port = 1
  82         # In upstream mode, we get a 502 response from the upstream proxy server.
  83         # In upstream mode with ssl, the replay will fail as we cannot establish
  84         # SSL with the upstream proxy.
  85         cr.start_replay([l])
  86         self.wait_response(l)
  87         if isinstance(self, tservers.HTTPUpstreamProxyTest):
  88             assert l.response.status_code == 502
  89         else:
  90             assert l.error

 106     def test_load_file(self, tmpdir):
 107         cp = clientplayback.ClientPlayback()
 108         with taddons.context(cp):
 109             fpath = str(tmpdir.join("flows"))

 167     def test_http2(self):
 168         cp = clientplayback.ClientPlayback()
 169         with taddons.context(cp):
 170             req = tutils.treq(
 171                 headers = net_http.Headers(
 172                     (
 173                         (b":authority", b"foo"),
 174                         (b"header", b"qvalue"),
 175                         (b"content-length", b"7")
 176                     )
 177                 )
 178             )
 179             f = tflow.tflow(req=req)
 180             f.request.http_version = "HTTP/2.0"
 181             cp.start_replay([f])
 182             assert f.request.http_version == "HTTP/1.1"
 183             assert ":authority" not in f.request.headers
