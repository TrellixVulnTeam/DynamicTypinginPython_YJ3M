  10 def test_configure():
  11     up = upstream_auth.UpstreamAuth()
  12     with taddons.context(up) as tctx:
  13         tctx.configure(up, upstream_auth="test:test")
  14         assert up.auth == b"Basic" + b" " + base64.b64encode(b"test:test")
  15
