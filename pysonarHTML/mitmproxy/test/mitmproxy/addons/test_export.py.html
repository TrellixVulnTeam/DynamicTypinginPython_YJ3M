  56 class TestExportCurlCommand:
  57     def test_get(self, get_request):
  58         result = """curl -H 'header: qvalue' 'http://address:22/path?a=foo&a=bar&b=baz'"""
  59         assert export.curl_command(get_request) == result
  60 
  61     def test_post(self, post_request):
  62         post_request.request.content = b'nobinarysupport'
  63         result = "curl -X POST http://address:22/path -d nobinarysupport"
  64         assert export.curl_command(post_request) == result
  65 
  66     def test_fails_with_binary_data(self, post_request):
  67         # shlex.quote doesn't support a bytes object
  68         # see https://github.com/python/cpython/pull/10871
  69         post_request.request.headers["Content-Type"] = "application/json; charset=utf-8"
  70         with pytest.raises(exceptions.CommandError):
  71             export.curl_command(post_request)
  72 
  73     def test_patch(self, patch_request):
  74         result = """curl -H 'header: qvalue' -X PATCH 'http://address:22/path?query=param' -d content"""
  75         assert export.curl_command(patch_request) == result
  76 
  77     def test_tcp(self, tcp_flow):
  78         with pytest.raises(exceptions.CommandError):
  79             export.curl_command(tcp_flow)
  80 
  81     def test_escape_single_quotes_in_body(self):
  82         request = tflow.tflow(
  83             req=tutils.treq(
  84                 method=b'POST',
  85                 headers=(),
  86                 content=b"'&#"
  87             )
  88         )
  89         command = export.curl_command(request)
  90         assert shlex.split(command)[-2] == '-d'
  91         assert shlex.split(command)[-1] == "'&#"
  92 
  93     def test_strip_unnecessary(self, get_request):
  94         get_request.request.headers.clear()
  95         get_request.request.headers["host"] = "address"
  96         get_request.request.headers[":authority"] = "address"
  97         get_request.request.headers["accept-encoding"] = "br"
  98         result = """curl --compressed 'http://address:22/path?a=foo&a=bar&b=baz'"""
  99         assert export.curl_command(get_request) == result

 195 def test_export(tmpdir):
 196     f = str(tmpdir.join("path"))
 197     e = export.Export()
 198     with taddons.context():
 199         assert e.formats() == ["curl", "httpie", "raw", "raw_request", "raw_response"]
 200         with pytest.raises(exceptions.CommandError):
 201             e.file("nonexistent", tflow.tflow(resp=True), f)

 204         assert qr(f)
 205         os.unlink(f)
