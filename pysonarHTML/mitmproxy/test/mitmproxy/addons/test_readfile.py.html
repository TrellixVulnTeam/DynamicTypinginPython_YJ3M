  41 class TestReadFile:
  42     def test_configure(self):
  43         rf = readfile.ReadFile()
  44         with taddons.context(rf) as tctx:
  45             tctx.configure(rf, readfile_filter="~q")
  46             with pytest.raises(Exception, match="Invalid readfile filter"):
  47                 tctx.configure(rf, readfile_filter="~~")

  49     @pytest.mark.asyncio
  50     async def test_read(self, tmpdir, data, corrupt_data):
  51         rf = readfile.ReadFile()
  52         with taddons.context(rf) as tctx:
  53             assert not rf.reading()
  54 
  55             tf = tmpdir.join("tfile")
  56 
  57             with asynctest.patch('mitmproxy.master.Master.load_flow') as mck:
  58                 tf.write(data.getvalue())
  59                 tctx.configure(
  60                     rf,
  61                     rfile = str(tf),
  62                     readfile_filter = ".*"
  63                 )
  64                 assert not mck.awaited
  65                 rf.running()
  66                 await asyncio.sleep(0)
  67                 assert mck.awaited
  68 
  69             tf.write(corrupt_data.getvalue())
  70             tctx.configure(rf, rfile=str(tf))
  71             rf.running()
  72             assert await tctx.master.await_log("corrupted")
  73 
  74     @pytest.mark.asyncio
  75     async def test_corrupt(self, corrupt_data):
  76         rf = readfile.ReadFile()
  77         with taddons.context(rf) as tctx:
  78             with pytest.raises(exceptions.FlowReadException):
  79                 await rf.load_flows(io.BytesIO(b"qibble"))
  80 
  81             tctx.master.clear()
  82             with pytest.raises(exceptions.FlowReadException):
  83                 await rf.load_flows(corrupt_data)
  84             assert await tctx.master.await_log("file corrupted")
  85
