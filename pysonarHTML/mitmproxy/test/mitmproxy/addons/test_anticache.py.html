   7 class TestAntiCache:
   8     def test_simple(self):
   9         sa = anticache.AntiCache()
  10         with taddons.context(sa) as tctx:
  11             f = tflow.tflow(resp=True)
  12             f.request.headers["if-modified-since"] = "test"
  13             f.request.headers["if-none-match"] = "test"
  14 
  15             sa.request(f)
  16             assert "if-modified-since" in f.request.headers
  17             assert "if-none-match" in f.request.headers
  18 
  19             tctx.configure(sa, anticache = True)
  20             sa.request(f)
  21             assert "if-modified-since" not in f.request.headers
  22             assert "if-none-match" not in f.request.headers
