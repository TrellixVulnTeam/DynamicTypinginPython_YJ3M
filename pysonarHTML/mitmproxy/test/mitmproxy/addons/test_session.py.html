  24     @staticmethod
  25     def start_session(fp=None):
  26         s = session.Session()
  27         with taddons.context() as tctx:
  28             tctx.master.addons.add(s)
  29             tctx.options.session_path = None
  30             tctx.options.view_filter = None
  31         # To make tests quicker
  32         if fp:
  33             s._flush_period = fp
  34             s._FP_DEFAULT = fp
  35         s.running()
  36         return s

  65     def test_session_load_existing(self, tdata):
  66         path = tdata.path('mitmproxy/data/') + '/test_le.sqlite'
  67         if os.path.isfile(path):
  68             os.remove(path)
  69         con = sqlite3.connect(path)
  70         script_path = pkg_data.path("io/sql/session_create.sql")
  71         with open(script_path) as f:
  72             qry = f.read()
  73         with con:
  74             con.executescript(qry)
  75             blob = b'blob_of_data'
  76             con.execute(f'INSERT INTO FLOW VALUES(1, "{blob}");')
  77         con.close()
  78         session.SessionDB(path)
  79         con = sqlite3.connect(path)
  80         with con:
  81             cur = con.cursor()
  82             cur.execute('SELECT * FROM FLOW;')
  83             rows = cur.fetchall()
  84             assert len(rows) == 1
  85         con.close()
  86         os.remove(path)

 144     def test_storage_filter(self):
 145         s = self.start_session()
 146         s.request(self.tft(method="get"))
 147         s.request(self.tft(method="put"))
 148         s.request(self.tft(method="get"))
 149         s.request(self.tft(method="put"))
 150         assert len(s._view) == 4
 151         with taddons.context() as tctx:
 152             tctx.master.addons.add(s)
 153             tctx.options.view_filter = '~m get'
 154         s.configure({"view_filter"})
 155         assert [f.request.method for f in s.load_view()] == ["GET", "GET"]
 156         assert s.store_count() == 4
 157         with pytest.raises(CommandError):
 158             s.set_filter("~notafilter")
 159         s.set_filter(None)
 160         assert len(s._view) == 4

 163     async def test_storage_flush_with_specials(self):
 164         s = self.start_session(fp=0.5)
 165         f = self.tft()
 166         s.request(f)
 167         await asyncio.sleep(1)
 168         assert len(s._hot_store) == 0
 169         f.response = http.HTTPResponse.wrap(tutils.tresp())
 170         s.response(f)
 171         assert len(s._hot_store) == 1
 172         assert s.load_storage() == [f]
 173         await asyncio.sleep(1)
 174         assert all([lflow.__dict__ == flow.__dict__ for lflow, flow in list(zip(s.load_storage(), [f]))])
 175 
 176         f.server_conn.via = tflow.tserver_conn()
 177         s.request(f)
 178         await asyncio.sleep(0.6)
 179         assert len(s._hot_store) == 0
 180         assert all([lflow.__dict__ == flow.__dict__ for lflow, flow in list(zip(s.load_storage(), [f]))])
 181 
 182         flows = [self.tft() for _ in range(500)]
 183         s.update(flows)
 184         await asyncio.sleep(0.6)
 185         assert s._flush_period == s._FP_DEFAULT * s._FP_DECREMENT
 186         await asyncio.sleep(3)
 187         assert s._flush_period == s._FP_DEFAULT

 224     @pytest.mark.asyncio
 225     async def test_storage_order(self):
 226         s = self.start_session(fp=0.5)
 227         s.request(self.tft(method="GET", start=4))
 228         s.request(self.tft(method="PUT", start=2))
 229         s.request(self.tft(method="GET", start=3))
 230         s.request(self.tft(method="PUT", start=1))
 231         assert [i.request.timestamp_start for i in s.load_view()] == [1, 2, 3, 4]
 232         await asyncio.sleep(1.0)
 233         assert [i.request.timestamp_start for i in s.load_view()] == [1, 2, 3, 4]
 234         with taddons.context() as tctx:
 235             tctx.master.addons.add(s)
 236             tctx.options.view_order = "method"
 237         s.configure({"view_order"})
 238         assert [i.request.method for i in s.load_view()] == ["GET", "GET", "PUT", "PUT"]
 239 
 240         s.set_order("time")
 241         assert [i.request.timestamp_start for i in s.load_view()] == [1, 2, 3, 4]
 242 
 243         with pytest.raises(CommandError):
 244             s.set_order("not_an_order")
