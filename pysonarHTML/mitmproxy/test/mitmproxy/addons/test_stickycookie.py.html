  16     def test_config(self):
  17         sc = stickycookie.StickyCookie()
  18         with taddons.context(sc) as tctx:
  19             with pytest.raises(Exception, match="invalid filter"):
  20                 tctx.configure(sc, stickycookie="~b")

  22             tctx.configure(sc, stickycookie="foo")

  24             tctx.configure(sc, stickycookie=None)

  27     def test_simple(self):
  28         sc = stickycookie.StickyCookie()
  29         with taddons.context(sc) as tctx:
  30             tctx.configure(sc, stickycookie=".*")
  31             f = tflow.tflow(resp=True)
  32             f.response.headers["set-cookie"] = "foo=bar"
  33             sc.request(f)
  34 
  35             f.reply.acked = False
  36             sc.response(f)
  37 
  38             assert sc.jar
  39             assert "cookie" not in f.request.headers
  40 
  41             f = f.copy()
  42             sc.request(f)
  43             assert f.request.headers["cookie"] == "foo=bar"

  45     def _response(self, sc, cookie, host):
  46         f = tflow.tflow(req=ntutils.treq(host=host, port=80), resp=True)
  47         f.response.headers["Set-Cookie"] = cookie
  48         sc.response(f)
  49         return f

  51     def test_response(self):
  52         sc = stickycookie.StickyCookie()
  53         with taddons.context(sc) as tctx:
  54             tctx.configure(sc, stickycookie=".*")
  55 
  56             c = "SSID=mooo; domain=.google.com, FOO=bar; Domain=.google.com; Path=/; " \
  57                 "Expires=Wed, 13-Jan-2021 22:23:01 GMT; Secure; "
  58 
  59             self._response(sc, c, "host")
  60             assert not sc.jar.keys()
  61 
  62             self._response(sc, c, "www.google.com")
  63             assert sc.jar.keys()
  64 
  65             sc.jar.clear()
  66             self._response(sc, "SSID=mooo", "www.google.com")
  67             assert list(sc.jar.keys())[0] == ('www.google.com', 80, '/')

  69     def test_response_multiple(self):
  70         sc = stickycookie.StickyCookie()
  71         with taddons.context(sc) as tctx:
  72             tctx.configure(sc, stickycookie=".*")
  73 
  74             # Test setting of multiple cookies
  75             c1 = "somecookie=test; Path=/"
  76             c2 = "othercookie=helloworld; Path=/"
  77             f = self._response(sc, c1, "www.google.com")
  78             f.response.headers["Set-Cookie"] = c2
  79             sc.response(f)
  80             googlekey = list(sc.jar.keys())[0]
  81             assert len(sc.jar[googlekey].keys()) == 2

  83     def test_response_weird(self):
  84         sc = stickycookie.StickyCookie()
  85         with taddons.context(sc) as tctx:
  86             tctx.configure(sc, stickycookie=".*")
  87 
  88             # Test setting of weird cookie keys
  89             f = tflow.tflow(req=ntutils.treq(host="www.google.com", port=80), resp=True)
  90             cs = [
  91                 "foo/bar=hello",
  92                 "foo:bar=world",
  93                 "foo@bar=fizz",
  94             ]
  95             for c in cs:
  96                 f.response.headers["Set-Cookie"] = c
  97                 sc.response(f)
  98             googlekey = list(sc.jar.keys())[0]
  99             assert len(sc.jar[googlekey].keys()) == len(cs)

 101     def test_response_overwrite(self):
 102         sc = stickycookie.StickyCookie()
 103         with taddons.context(sc) as tctx:
 104             tctx.configure(sc, stickycookie=".*")
 105 
 106             # Test overwriting of a cookie value
 107             c1 = "somecookie=helloworld; Path=/"
 108             c2 = "somecookie=newvalue; Path=/"
 109             f = self._response(sc, c1, "www.google.com")
 110             f.response.headers["Set-Cookie"] = c2
 111             sc.response(f)
 112             googlekey = list(sc.jar.keys())[0]
 113             assert len(sc.jar[googlekey]) == 1
 114             assert sc.jar[googlekey]["somecookie"] == "newvalue"
