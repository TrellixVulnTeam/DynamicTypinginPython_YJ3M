  45     This only succeeds if they get assigned different basenames.
  46 
  47     """
  48     ns = script.load_script(
  49         tdata.path(
  50             "mitmproxy/data/addonscripts/addon.py"

 103     @pytest.mark.asyncio
 104     async def test_reload(self, tmpdir):
 105         with taddons.context() as tctx:
 106             f = tmpdir.join("foo.py")
 107             f.ensure(file=True)
 108             f.write("\n")
 109             sc = script.Script(str(f), True)
 110             tctx.configure(sc)
 111             assert await tctx.master.await_log("Loading")
 112 
 113             tctx.master.clear()
 114             for i in range(20):
 115                 f.write("\n")
 116                 if tctx.master.has_log("Loading"):
 117                     break
 118                 await asyncio.sleep(0.1)
 119             else:
 120                 raise AssertionError("No reload seen")

 166 class TestCutTraceback:
 167     def raise_(self, i):
 168         if i > 0:
 169             self.raise_(i - 1)
 170         raise RuntimeError()
 171 
 172     def test_simple(self):
 173         try:
 174             self.raise_(4)
 175         except RuntimeError:
 176             tb = sys.exc_info()[2]
 177             tb_cut = addonmanager.cut_traceback(tb, "test_simple")
 178             assert len(traceback.extract_tb(tb_cut)) == 5
 179 
 180             tb_cut2 = addonmanager.cut_traceback(tb, "nonexistent")
 181             assert len(traceback.extract_tb(tb_cut2)) == len(traceback.extract_tb(tb))

 265     @pytest.mark.asyncio
 266     async def test_order(self, tdata):
 267         rec = tdata.path("mitmproxy/data/addonscripts/recorder")
 268         sc = script.ScriptLoader()
 269         sc.is_running = True
 270         with taddons.context(sc) as tctx:
 271             tctx.configure(
 272                 sc,
 273                 scripts = [
 274                     "%s/a.py" % rec,
 275                     "%s/b.py" % rec,
 276                     "%s/c.py" % rec,
 277                 ]
 278             )
 279             await tctx.master.await_log("configure")
 280             debug = [i.msg for i in tctx.master.logs if i.level == "debug"]
 281             assert debug == [
 282                 'a load',
 283                 'a running',
 284                 'a configure',
 285 
 286                 'b load',
 287                 'b running',
 288                 'b configure',
 289 
 290                 'c load',
 291                 'c running',
 292                 'c configure',
 293             ]
 294 
 295             tctx.master.clear()
 296             tctx.configure(
 297                 sc,
 298                 scripts = [
 299                     "%s/c.py" % rec,
 300                     "%s/a.py" % rec,
 301                     "%s/b.py" % rec,
 302                 ]
 303             )
 304 
 305             await tctx.master.await_log("c configure")
 306             debug = [i.msg for i in tctx.master.logs if i.level == "debug"]
 307             assert debug == [
 308                 'c configure',
 309                 'a configure',
 310                 'b configure',
 311             ]
 312 
 313             tctx.master.clear()
 314             tctx.configure(
 315                 sc,
 316                 scripts = [
 317                     "%s/e.py" % rec,
 318                     "%s/a.py" % rec,
 319                 ]
 320             )
 321             await tctx.master.await_log("Loading")
 322             debug = [i.msg for i in tctx.master.logs if i.level == "debug"]
 323             assert debug == [
 324                 'c done',
 325                 'b done',
 326                 'a configure',
 327                 'e load',
 328                 'e running',
 329                 'e configure',
 330             ]
