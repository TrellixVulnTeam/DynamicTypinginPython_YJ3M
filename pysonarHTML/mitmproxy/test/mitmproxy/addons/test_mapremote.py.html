  10     def test_configure(self):
  11         mr = mapremote.MapRemote()
  12         with taddons.context(mr) as tctx:
  13             tctx.configure(mr, map_remote=["one/two/three"])
  14             with pytest.raises(Exception, match="Cannot parse map_remote .* Invalid number"):
  15                 tctx.configure(mr, map_remote=["/"])
  16             with pytest.raises(Exception, match="Cannot parse map_remote .* Invalid filter"):
  17                 tctx.configure(mr, map_remote=["/~b/two/three"])
  18             with pytest.raises(Exception, match="Cannot parse map_remote .* Invalid regular expression"):
  19                 tctx.configure(mr, map_remote=["/foo/+/three"])
  20             tctx.configure(mr, map_remote=["/a/b/c/"])

  22     def test_simple(self):
  23         mr = mapremote.MapRemote()
  24         with taddons.context(mr) as tctx:
  25             tctx.configure(
  26                 mr,
  27                 map_remote=[
  28                     ":example.org/images/:mitmproxy.org/img/",
  29                 ]
  30             )
  31             f = tflow.tflow()
  32             f.request.url = b"https://example.org/images/test.jpg"
  33             mr.request(f)
  34             assert f.request.url == "https://mitmproxy.org/img/test.jpg"

  51             tmpfile = tmpdir.join("replacement")
  52             tmpfile.write("mitmproxy.org")

  55                 map_remote=["|example.org|@" + str(tmpfile)]

  57             f = tflow.tflow()

  60             assert f.request.url == "https://mitmproxy.org/test"
