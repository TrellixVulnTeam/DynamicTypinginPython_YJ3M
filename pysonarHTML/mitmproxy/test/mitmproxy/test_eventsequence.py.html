   7 @pytest.mark.parametrize("resp, err", [
   8     (False, False),
   9     (True, False),
  10     (False, True),
  11     (True, True),
  12 ])
  13 def test_http_flow(resp, err):
  14     f = tflow.tflow(resp=resp, err=err)
  15     i = eventsequence.iterate(f)
  16     assert next(i) == ("requestheaders", f)
  17     assert next(i) == ("request", f)
  18     if resp:
  19         assert next(i) == ("responseheaders", f)
  20         assert next(i) == ("response", f)
  21     if err:
  22         assert next(i) == ("error", f)

  25 @pytest.mark.parametrize("err", [False, True])
  26 def test_websocket_flow(err):
  27     f = tflow.twebsocketflow(err=err)
  28     i = eventsequence.iterate(f)
  29     assert next(i) == ("websocket_start", f)
  30     assert len(f.messages) == 0
  31     assert next(i) == ("websocket_message", f)
  32     assert len(f.messages) == 1
  33     assert next(i) == ("websocket_message", f)
  34     assert len(f.messages) == 2
  35     assert next(i) == ("websocket_message", f)
  36     assert len(f.messages) == 3
  37     if err:
  38         assert next(i) == ("websocket_error", f)
  39     assert next(i) == ("websocket_end", f)
