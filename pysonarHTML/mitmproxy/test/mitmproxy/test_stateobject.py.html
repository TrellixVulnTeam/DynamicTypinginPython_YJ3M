   8 class TObject(StateObject):
   9     def __init__(self, x):
  10         self.x = x
  11 
  12     @classmethod
  13     def from_state(cls, state):
  14         obj = cls(None)
  15         obj.set_state(state)
  16         return obj

  19 class Child(TObject):
  20     _stateobject_attributes = dict(

  24     def __eq__(self, other):

  58 def test_simple():
  59     a = Child(42)

  67 def test_serializable_child():
  68     child = Child(42)
  69     a = TSerializableChild(child)
  70     assert a.get_state() == {
  71         "x": {"x": 42}
  72     }
  73     a.set_state({
  74         "x": {"x": 43}
  75     })
  76     assert a.x.x == 43
  77     assert a.x is child
  78     b = a.copy()
  79     assert a.x == b.x

  83 def test_tuple():
  84     a = TTuple((42, Child(43)))
