  17     def test_roundtrip(self):
  18         sio = io.BytesIO()
  19         f = tflow.tflow()
  20         f.marked = True
  21         f.request.content = bytes(range(256))
  22         w = mitmproxy.io.FlowWriter(sio)
  23         w.add(f)
  24 
  25         sio.seek(0)
  26         r = mitmproxy.io.FlowReader(sio)
  27         l = list(r.stream())
  28         assert len(l) == 1
  29 
  30         f2 = l[0]
  31         assert f2.get_state() == f.get_state()
  32         assert f2.request == f.request
  33         assert f2.marked

  96 class TestFlowMaster:
  97     @pytest.mark.asyncio
  98     async def test_load_http_flow_reverse(self):
  99         opts = options.Options(
 100             mode="reverse:https://use-this-domain"
 101         )
 102         s = tservers.TestState()
 103         with taddons.context(s, options=opts) as ctx:
 104             f = tflow.tflow(resp=True)
 105             await ctx.master.load_flow(f)
 106             assert s.flows[0].request.host == "use-this-domain"
 107 
 108     @pytest.mark.asyncio
 109     async def test_load_websocket_flow(self):
 110         opts = options.Options(
 111             mode="reverse:https://use-this-domain"
 112         )
 113         s = tservers.TestState()
 114         with taddons.context(s, options=opts) as ctx:
 115             f = tflow.twebsocketflow()
 116             await ctx.master.load_flow(f.handshake_flow)
 117             await ctx.master.load_flow(f)
 118             assert s.flows[0].request.host == "use-this-domain"
 119             assert s.flows[1].handshake_flow == f.handshake_flow
 120             assert len(s.flows[1].messages) == len(f.messages)
 121 
 122     @pytest.mark.asyncio
 123     async def test_all(self):
 124         opts = options.Options(
 125             mode="reverse:https://use-this-domain"
 126         )
 127         s = tservers.TestState()
 128         with taddons.context(s, options=opts) as ctx:
 129             f = tflow.tflow(req=None)
 130             await ctx.master.addons.handle_lifecycle("clientconnect", f.client_conn)
 131             f.request = http.HTTPRequest.wrap(mitmproxy.test.tutils.treq())
 132             await ctx.master.addons.handle_lifecycle("request", f)
 133             assert len(s.flows) == 1
 134 
 135             f.response = http.HTTPResponse.wrap(mitmproxy.test.tutils.tresp())
 136             await ctx.master.addons.handle_lifecycle("response", f)
 137             assert len(s.flows) == 1
 138 
 139             await ctx.master.addons.handle_lifecycle("clientdisconnect", f.client_conn)
 140 
 141             f.error = flow.Error("msg")
 142             await ctx.master.addons.handle_lifecycle("error", f)

 147     def test_getset_state(self):
 148         e = flow.Error("Error")
 149         state = e.get_state()
 150         assert flow.Error.from_state(state).get_state() == e.get_state()
 151 
 152         assert e.copy()
 153 
 154         e2 = flow.Error("bar")
 155         assert not e == e2
 156         e.set_state(e2.get_state())
 157         assert e.get_state() == e2.get_state()
 158 
 159         e3 = e.copy()
 160         assert e3.get_state() == e.get_state()
