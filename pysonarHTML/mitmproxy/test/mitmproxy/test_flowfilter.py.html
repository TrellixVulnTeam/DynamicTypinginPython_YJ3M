  12     def _dump(self, x):
  13         c = io.StringIO()
  14         x.dump(fp=c)
  15         assert c.getvalue()

  17     def test_parse_err(self):
  18         assert flowfilter.parse("~h [") is None

  20     def test_simple(self):
  21         assert not flowfilter.parse("~b")
  22         assert flowfilter.parse("~q")
  23         assert flowfilter.parse("~c 10")
  24         assert flowfilter.parse("~m foobar")
  25         assert flowfilter.parse("~u foobar")
  26         assert flowfilter.parse("~q ~c 10")
  27         p = flowfilter.parse("~q ~c 10")
  28         self._dump(p)
  29         assert len(p.lst) == 2

  31     def test_non_ascii(self):
  32         assert flowfilter.parse("~s шгн")

  34     def test_naked_url(self):
  35         a = flowfilter.parse("foobar ~h rex")
  36         assert a.lst[0].expr == "foobar"
  37         assert a.lst[1].expr == "rex"
  38         self._dump(a)

  40     def test_quoting(self):
  41         a = flowfilter.parse("~u 'foo ~u bar' ~u voing")
  42         assert a.lst[0].expr == "foo ~u bar"
  43         assert a.lst[1].expr == "voing"
  44         self._dump(a)
  45 
  46         a = flowfilter.parse("~u foobar")
  47         assert a.expr == "foobar"
  48 
  49         a = flowfilter.parse(r"~u 'foobar\"\''")
  50         assert a.expr == "foobar\"'"
  51 
  52         a = flowfilter.parse(r'~u "foo \'bar"')
  53         assert a.expr == "foo 'bar"
  54 
  55     def test_nesting(self):
  56         a = flowfilter.parse("(~u foobar & ~h voing)")
  57         assert a.lst[0].expr == "foobar"
  58         self._dump(a)
  59 
  60     def test_not(self):
  61         a = flowfilter.parse("!~h test")
  62         assert a.itm.expr == "test"
  63         a = flowfilter.parse("!(~u test & ~h bar)")
  64         assert a.itm.lst[0].expr == "test"
  65         self._dump(a)
  66 
  67     def test_binaryops(self):
  68         a = flowfilter.parse("~u foobar | ~h voing")
  69         isinstance(a, flowfilter.FOr)
  70         self._dump(a)
  71 
  72         a = flowfilter.parse("~u foobar & ~h voing")
  73         isinstance(a, flowfilter.FAnd)
  74         self._dump(a)
