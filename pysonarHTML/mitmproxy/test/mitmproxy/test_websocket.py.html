  12     def test_copy(self):
  13         f = tflow.twebsocketflow()
  14         f.get_state()
  15         f2 = f.copy()
  16         a = f.get_state()
  17         b = f2.get_state()
  18         del a["id"]
  19         del b["id"]
  20         assert a == b
  21         assert not f == f2
  22         assert f is not f2
  23 
  24         assert f.client_key == f2.client_key
  25         assert f.client_protocol == f2.client_protocol
  26         assert f.client_extensions == f2.client_extensions
  27         assert f.server_accept == f2.server_accept
  28         assert f.server_protocol == f2.server_protocol
  29         assert f.server_extensions == f2.server_extensions
  30         assert f.messages is not f2.messages
  31         assert f.handshake_flow is not f2.handshake_flow
  32 
  33         for m in f.messages:
  34             m2 = m.copy()
  35             m2.set_state(m2.get_state())
  36             assert m is not m2
  37             assert m.get_state() == m2.get_state()
  38 
  39         f = tflow.twebsocketflow(err=True)
  40         f2 = f.copy()
  41         assert f is not f2
  42         assert f.handshake_flow is not f2.handshake_flow
  43         assert f.error.get_state() == f2.error.get_state()

  46     def test_kill(self):
  47         f = tflow.twebsocketflow()
  48         with pytest.raises(ControlException):
  49             f.intercept()
  50             f.resume()
  51             f.kill()
  52 
  53         f = tflow.twebsocketflow()
  54         f.intercept()
  55         assert f.killable
  56         f.kill()
  57         assert not f.killable
  58         assert f.reply.value == Kill
