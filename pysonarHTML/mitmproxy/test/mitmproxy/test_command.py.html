 419 def test_simple():
 420     with taddons.context() as tctx:
 421         c = command.CommandManager(tctx.master)
 422         a = TAddon()
 423         c.add("one.two", a.cmd1)
 424         assert (c.commands["one.two"].help == "cmd1 help")
 425         assert (c.execute("one.two foo") == "ret foo")
 426         assert (c.execute("one.two \"foo\"") == "ret foo")
 427         assert (c.execute("one.two 'foo bar'") == "ret foo bar")
 428         assert (c.call("one.two", "foo") == "ret foo")
 429         with pytest.raises(exceptions.CommandError, match="Unknown"):
 430             c.execute("nonexistent")
 431         with pytest.raises(exceptions.CommandError, match="Invalid"):
 432             c.execute("")
 433         with pytest.raises(exceptions.CommandError, match="argument mismatch"):
 434             c.execute("one.two too many args")
 435         with pytest.raises(exceptions.CommandError, match="Unknown"):
 436             c.call("nonexistent")
 437         with pytest.raises(exceptions.CommandError, match="Unknown"):
 438             c.execute("\\")
 439         with pytest.raises(exceptions.CommandError, match="Unknown"):
 440             c.execute(r"\'")
 441         with pytest.raises(exceptions.CommandError, match="Unknown"):
 442             c.execute(r"\"")
 443         with pytest.raises(exceptions.CommandError, match="Unknown"):
 444             c.execute(r"\"")
 445 
 446         c.add("empty", a.empty)
 447         c.execute("empty")
 448 
 449         fp = io.StringIO()
 450         c.dump(fp)
 451         assert fp.getvalue()

 454 def test_typename():
 455     assert command.typename(str) == "str"
 456     assert command.typename(typing.Sequence[flow.Flow]) == "flow[]"
 457 
 458     assert command.typename(mitmproxy.types.Data) == "data[][]"
 459     assert command.typename(mitmproxy.types.CutSpec) == "cut[]"
 460 
 461     assert command.typename(flow.Flow) == "flow"
 462     assert command.typename(typing.Sequence[str]) == "str[]"
 463 
 464     assert command.typename(mitmproxy.types.Choice("foo")) == "choice"
 465     assert command.typename(mitmproxy.types.Path) == "path"
 466     assert command.typename(mitmproxy.types.Cmd) == "cmd"
 467 
 468     with pytest.raises(exceptions.CommandError, match="missing type annotation"):
 469         command.typename(inspect._empty)
 470     with pytest.raises(exceptions.CommandError, match="unsupported type"):
 471         command.typename(None)
