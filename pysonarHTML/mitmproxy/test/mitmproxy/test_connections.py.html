 183     @pytest.mark.parametrize("sni", [
 184         None,
 185         "example.com"
 186     ])
 187     def test_tls_with_sni(self, sni, tdata):
 188         address = ('127.0.0.1', 0)
 189         sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
 190         sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
 191         sock.bind(address)
 192         sock.listen()
 193         address = sock.getsockname()
 194 
 195         def client_run():
 196             ctx = ssl.create_default_context()
 197             ctx.check_hostname = False
 198             ctx.verify_mode = ssl.CERT_NONE
 199             s = socket.create_connection(address)
 200             s = ctx.wrap_socket(s, server_hostname=sni)
 201             s.send(b'foobar')
 202             # we need to wait for the test to finish successfully before calling .close() on Windows.
 203             # The workaround here is to signal completion by sending data the other way around.
 204             s.recv(3)
 205             s.close()
 206         threading.Thread(target=client_run).start()
 207 
 208         connection, client_address = sock.accept()
 209         c = connections.ClientConnection(connection, client_address, None)
 210 
 211         cert = tdata.path("mitmproxy/net/data/server.crt")
 212         with open(tdata.path("mitmproxy/net/data/server.key")) as f:
 213             raw_key = f.read()
 214         key = OpenSSL.crypto.load_privatekey(
 215             OpenSSL.crypto.FILETYPE_PEM,
 216             raw_key)
 217         c.convert_to_tls(cert, key)
 218         assert c.connected()
 219         assert c.sni == sni
 220         assert c.tls_established
 221         assert c.rfile.read(6) == b'foobar'
 222         c.wfile.send(b"foo")
 223         c.finish()
 224         sock.close()
