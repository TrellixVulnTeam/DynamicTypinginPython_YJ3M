  31 def get_random_object(random=random, depth=0):
  32     """Generate a random serializable object."""
  33     #  The probability of generating a scalar value increases as the depth increase.
  34     #  This ensures that we bottom out eventually.
  35     if random.randint(depth, 10) <= 4:
  36         what = random.randint(0, 1)
  37         if what == 0:
  38             n = random.randint(0, 10)
  39             l = []
  40             for _ in range(n):
  41                 l.append(get_random_object(random, depth + 1))
  42             return l
  43         if what == 1:
  44             n = random.randint(0, 10)
  45             d = {}
  46             for _ in range(n):
  47                 n = random.randint(0, 100)
  48                 k = str([random.randint(32, 126) for _ in range(n)])
  49                 d[k] = get_random_object(random, depth + 1)
  50             return d
  51     else:
  52         what = random.randint(0, 4)
  53         if what == 0:
  54             return None
  55         if what == 1:
  56             return True
  57         if what == 2:
  58             return False
  59         if what == 3:
  60             if random.randint(0, 1) == 0:
  61                 return random.randint(0, MAXINT)
  62             else:
  63                 return -1 * random.randint(0, MAXINT)
  64         n = random.randint(0, 100)
  65         return bytes([random.randint(32, 126) for _ in range(n)])

  96 class Test_FileLoading(unittest.TestCase):

  98     def test_roundtrip_file_examples(self):
  99         for data, expect in FORMAT_EXAMPLES.items():
 100             s = io.BytesIO()
 101             s.write(data)
 102             s.write(b'OK')
 103             s.seek(0)
 104             self.assertEqual(expect, tnetstring.load(s))
 105             self.assertEqual(b'OK', s.read())
 106             s = io.BytesIO()
 107             tnetstring.dump(expect, s)
 108             s.write(b'OK')
 109             s.seek(0)

 111             self.assertEqual(b'OK', s.read())
