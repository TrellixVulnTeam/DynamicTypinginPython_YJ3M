 126         # send CONNECT request

 157     def _send_request(self,
 158                       wfile,
 159                       h2_conn,
 160                       stream_id=1,
 161                       headers=None,
 162                       body=b'',
 163                       end_stream=None,
 164                       priority_exclusive=None,
 165                       priority_depends_on=None,
 166                       priority_weight=None,
 167                       streaming=False):
 168         if headers is None:
 169             headers = []
 170         if end_stream is None:
 171             end_stream = (len(body) == 0)
 172 
 173         h2_conn.send_headers(
 174             stream_id=stream_id,
 175             headers=headers,
 176             end_stream=end_stream,
 177             priority_exclusive=priority_exclusive,
 178             priority_depends_on=priority_depends_on,
 179             priority_weight=priority_weight,
 180         )
 181         if body:
 182             h2_conn.send_data(stream_id, body)
 183             if not streaming:
 184                 h2_conn.end_stream(stream_id)
 185         wfile.write(h2_conn.data_to_send())
 186         wfile.flush()

 843     def test_max_concurrent_streams(self):
 844         h2_conn = self.setup_connection()

 846         for stream_id in new_streams:
 847             # this will exceed MAX_CONCURRENT_STREAMS on the server connection
 848             # and cause mitmproxy to throttle stream creation to the server
 849             self._send_request(self.client.wfile, h2_conn, stream_id=stream_id, headers=[

 859             try:
 860                 header, body = http2.read_raw_frame(self.client.rfile)
 861                 events = h2_conn.receive_data(b''.join([header, body]))
 862             except:
 863                 break
 864             self.client.wfile.write(h2_conn.data_to_send())
 865             self.client.wfile.flush()

 867             for event in events:

 869                     ended_streams += 1
 870
