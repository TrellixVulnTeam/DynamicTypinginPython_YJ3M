 211     def test_connection_close(self):
 212         # Add a body, so we have a content-length header, which combined with
 213         # HTTP1.1 means the connection is kept alive.
 214         response = '%s/p/200:b@1' % self.server.urlbase
 215 
 216         # Lets sanity check that the connection does indeed stay open by
 217         # issuing two requests over the same connection
 218         p = self.pathoc()
 219         with p.connect():
 220             assert p.request("get:'%s'" % response)
 221             assert p.request("get:'%s'" % response)
 222 
 223         # Now check that the connection is closed as the client specifies
 224         p = self.pathoc()
 225         with p.connect():
 226             assert p.request("get:'%s':h'Connection'='close'" % response)
 227             # There's a race here, which means we can get any of a number of errors.
 228             # Rather than introduce yet another sleep into the test suite, we just
 229             # relax the Exception specification.
 230             with pytest.raises(Exception):
 231                 p.request("get:'%s'" % response)

 897     def _host_pattern_on(self, attr):
 898         """
 899         Updates config.check_tcp or check_filter, depending on attr.
 900         """
 901         assert not hasattr(self, "_ignore_%s_backup" % attr)
 902         backup = []
 903         handle = attr
 904         attr = "filter" if attr in ["allow", "ignore"] else attr
 905         for proxy in self.chain:
 906             old_matcher = getattr(
 907                 proxy.tmaster.server.config,
 908                 "check_%s" %
 909                 attr)
 910             backup.append(old_matcher)
 911             setattr(
 912                 proxy.tmaster.server.config,
 913                 "check_%s" % attr,
 914                 HostMatcher(handle, [".+:%s" % self.server.port] + old_matcher.patterns)
 915             )
 916 
 917         setattr(self, "_ignore_%s_backup" % attr, backup)
 918 
 919     def _host_pattern_off(self, attr):
 920         attr = "filter" if attr in ["allow", "ignore"] else attr
 921         backup = getattr(self, "_ignore_%s_backup" % attr)
 922         for proxy in reversed(self.chain):
 923             setattr(
 924                 proxy.tmaster.server.config,
 925                 "check_%s" % attr,
 926                 backup.pop()
 927             )
 928 
 929         assert not backup
 930         delattr(self, "_ignore_%s_backup" % attr)
 931

 933         super()._ignore_on()
 934         self._host_pattern_on("ignore")
