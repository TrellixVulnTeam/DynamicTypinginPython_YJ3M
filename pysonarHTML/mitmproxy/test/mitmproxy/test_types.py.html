  16 @contextlib.contextmanager
  17 def chdir(path: str):
  18     old_dir = os.getcwd()
  19     os.chdir(path)
  20     yield
  21     os.chdir(old_dir)

  66 def test_path(tdata, monkeypatch):
  67     with taddons.context() as tctx:
  68         b = mitmproxy.types._PathType()
  69         assert b.parse(tctx.master.commands, mitmproxy.types.Path, "/foo") == "/foo"
  70         assert b.parse(tctx.master.commands, mitmproxy.types.Path, "/bar") == "/bar"
  71         monkeypatch.setenv("HOME", "/home/test")
  72         monkeypatch.setenv("USERPROFILE", "/home/test")
  73         assert b.parse(tctx.master.commands, mitmproxy.types.Path, "~/mitm") == "/home/test/mitm"
  74         assert b.is_valid(tctx.master.commands, mitmproxy.types.Path, "foo") is True
  75         assert b.is_valid(tctx.master.commands, mitmproxy.types.Path, "~/mitm") is True
  76         assert b.is_valid(tctx.master.commands, mitmproxy.types.Path, 3) is False
  77 
  78         def normPathOpts(prefix, match):
  79             ret = []
  80             for s in b.completion(tctx.master.commands, mitmproxy.types.Path, match):
  81                 s = s[len(prefix):]
  82                 s = s.replace(os.sep, "/")
  83                 ret.append(s)
  84             return ret
  85 
  86         cd = os.path.normpath(tdata.path("mitmproxy/completion"))
  87         assert normPathOpts(cd, cd) == ['/aaa', '/aab', '/aac', '/bbb/']
  88         assert normPathOpts(cd, os.path.join(cd, "a")) == ['/aaa', '/aab', '/aac']
  89         with chdir(cd):
  90             assert normPathOpts("", "./") == ['./aaa', './aab', './aac', './bbb/']
  91             assert normPathOpts("", "") == ['./aaa', './aab', './aac', './bbb/']
  92         assert b.completion(
  93             tctx.master.commands, mitmproxy.types.Path, "nonexistent"
  94         ) == ["nonexistent"]
