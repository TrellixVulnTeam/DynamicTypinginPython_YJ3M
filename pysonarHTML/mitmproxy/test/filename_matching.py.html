   9 def check_src_files_have_test():
  10     missing_test_files = []
  11 
  12     excluded = ['mitmproxy/contrib/', 'mitmproxy/io/proto/',
  13                 'mitmproxy/test/', 'mitmproxy/tools/', 'mitmproxy/platform/']
  14     src_files = glob.glob('mitmproxy/**/*.py', recursive=True) + glob.glob('pathod/**/*.py', recursive=True)
  15     src_files = [f for f in src_files if os.path.basename(f) != '__init__.py']
  16     src_files = [f for f in src_files if not any(os.path.normpath(p) in f for p in excluded)]
  17     for f in src_files:
  18         p = os.path.join("test", os.path.dirname(f), "test_" + os.path.basename(f))
  19         if not os.path.isfile(p):
  20             missing_test_files.append((f, p))
  21 
  22     return missing_test_files

  25 def check_test_files_have_src():
  26     unknown_test_files = []
  27 
  28     excluded = ['test/mitmproxy/data/', 'test/mitmproxy/net/data/', '/tservers.py', '/conftest.py']
  29     test_files = glob.glob('test/mitmproxy/**/*.py', recursive=True) + glob.glob('test/pathod/**/*.py', recursive=True)
  30     test_files = [f for f in test_files if os.path.basename(f) != '__init__.py']
  31     test_files = [f for f in test_files if not any(os.path.normpath(p) in f for p in excluded)]
  32     for f in test_files:
  33         p = os.path.join(re.sub('^test/', '', os.path.dirname(f)), re.sub('^test_', '', os.path.basename(f)))
  34         if not os.path.isfile(p):
  35             unknown_test_files.append((f, p))
  36 
  37     return unknown_test_files
