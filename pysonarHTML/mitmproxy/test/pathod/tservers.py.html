 111     def pathoc(
 112         self,
 113         specs,
 114         timeout=None,
 115         connect_to=None,
 116         ssl=None,
 117         ws_read_limit=None,
 118         use_http2=False,
 119     ):
 120         """
 121             Returns a (messages, text log) tuple.
 122         """
 123         if ssl is None:
 124             ssl = self.ssl
 125         logfp = io.StringIO()
 126         c = pathoc.Pathoc(
 127             ("localhost", self.d.port),
 128             ssl=ssl,
 129             ws_read_limit=ws_read_limit,
 130             timeout=timeout,
 131             fp=logfp,
 132             use_http2=use_http2,
 133         )
 134         with c.connect(connect_to):
 135             ret = []
 136             for i in specs:
 137                 resp = c.request(i)
 138                 if resp:
 139                     ret.append(resp)
 140             for frm in c.wait():
 141                 ret.append(frm)
 142             c.stop()
 143             return ret, logfp.getvalue()
