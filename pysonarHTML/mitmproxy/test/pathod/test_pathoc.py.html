  21 class PathocTestDaemon(tservers.DaemonTests):
  22     def tval(self, requests, timeout=None, showssl=False, **kwargs):
  23         s = io.StringIO()
  24         c = pathoc.Pathoc(
  25             ("127.0.0.1", self.d.port),
  26             ssl=self.ssl,
  27             fp=s,
  28             **kwargs
  29         )
  30         with c.connect(showssl=showssl, fp=s):
  31             if timeout:
  32                 c.settimeout(timeout)
  33             for i in requests:
  34                 r = next(language.parse_pathoc(i))
  35                 if kwargs.get("explain"):
  36                     r = r.freeze(language.Settings())
  37                 try:
  38                     c.request(r)
  39                 except exceptions.NetlibException:
  40                     pass
  41         self.d.wait_for_silence()
  42         return s.getvalue()
