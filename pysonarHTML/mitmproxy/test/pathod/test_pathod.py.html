  18     def test_logging(self):
  19         s = io.StringIO()
  20         p = pathod.Pathod(("127.0.0.1", 0), logfp=s)
  21         assert len(p.get_log()) == 0
  22         id = p.add_log(dict(s="foo"))
  23         assert p.log_by_id(id)
  24         assert len(p.get_log()) == 1
  25         p.clear_log()
  26         assert len(p.get_log()) == 0
  27 
  28         for _ in range(p.LOGBUF + 1):
  29             p.add_log(dict(s="foo"))
  30         assert len(p.get_log()) <= p.LOGBUF
