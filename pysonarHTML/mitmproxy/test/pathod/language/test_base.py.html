  41     def test_expr(self):
  42         v = base.TokValueLiteral("foo")
  43         assert v.expr()
  44         assert v.val == b"foo"
  45 
  46         v = base.TokValueLiteral("foo\n")
  47         assert v.expr()
  48         assert v.val == b"foo\n"
  49         assert repr(v)
  50 
  51     def test_spec(self):
  52         v = base.TokValueLiteral("foo")
  53         assert v.spec() == r"'foo'"
  54 
  55         v = base.TokValueLiteral("f\x00oo")
  56         assert v.spec() == repr(v) == r"'f\x00oo'"
  57 
  58         v = base.TokValueLiteral('"')
  59         assert v.spec() == repr(v) == """  '"'  """.strip()
  60 
  61         # While pyparsing has a escChar argument for QuotedString,
  62         # escChar only performs scapes single-character escapes and does not work for e.g. r"\x02".
  63         # Thus, we cannot use that option, which means we cannot have single quotes in strings.
  64         # To fix this, we represent single quotes as r"\x07".
  65         v = base.TokValueLiteral("'")
  66         assert v.spec() == r"'\x27'"
  67 
  68     def roundtrip(self, spec):
  69         e = base.TokValueLiteral.expr()
  70         v = base.TokValueLiteral(spec)
  71         v2 = e.parseString(v.spec())
  72         assert v.val == v2[0].val
  73         assert v.spec() == v2[0].spec()
