 191     def test_simple(self):
 192         r = parse_response('200:r:h"foo"="bar"')
 193         assert r.status_code.string() == b"200"
 194         assert len(r.headers) == 1
 195         assert r.headers[0].values(default_settings()) == (b"foo", b"bar")
 196         assert r.body is None
 197 
 198         r = parse_response('200:r:h"foo"="bar":bfoobar:h"bla"="fasel"')
 199         assert r.status_code.string() == b"200"
 200         assert len(r.headers) == 2
 201         assert r.headers[0].values(default_settings()) == (b"foo", b"bar")
 202         assert r.headers[1].values(default_settings()) == (b"bla", b"fasel")
 203         assert r.body.string() == b"foobar"
