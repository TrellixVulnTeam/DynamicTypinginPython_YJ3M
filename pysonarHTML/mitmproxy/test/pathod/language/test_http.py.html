 154     def test_response(self):
 155         r = next(language.parse_pathod("400:m'msg'"))
 156         assert r.status_code.string() == b"400"
 157         assert r.reason.string() == b"msg"
 158 
 159         r = next(language.parse_pathod("400:m'msg':b@100b"))
 160         assert r.reason.string() == b"msg"
 161         assert r.body.values({})
 162         assert str(r)
 163 
 164         r = next(language.parse_pathod("200"))
 165         assert r.status_code.string() == b"200"
 166         assert not r.reason
 167         assert b"OK" in [i[:] for i in r.preamble({})]

 248     def test_parse_stress(self):
 249         # While larger values are known to work on linux, len() technically
 250         # returns an int and a python 2.7 int on windows has 32bit precision.

 252         # tests.
 253         r = next(language.parse_pathod("400:b@1g"))
 254         assert r.length({})
