  62     def test_flags(self):
  63         wf = parse_request("wf:fin:mask:rsv1:rsv2:rsv3")
  64         frm = mitmproxy.net.websockets.Frame.from_bytes(tservers.render(wf))
  65         assert frm.header.fin
  66         assert frm.header.mask
  67         assert frm.header.rsv1
  68         assert frm.header.rsv2
  69         assert frm.header.rsv3
  70 
  71         wf = parse_request("wf:-fin:-mask:-rsv1:-rsv2:-rsv3")
  72         frm = mitmproxy.net.websockets.Frame.from_bytes(tservers.render(wf))
  73         assert not frm.header.fin
  74         assert not frm.header.mask
  75         assert not frm.header.rsv1
  76         assert not frm.header.rsv2
  77         assert not frm.header.rsv3

  79     def fr(self, spec, **kwargs):

  82         return mitmproxy.net.websockets.Frame.from_bytes(tservers.render(wf, settings))

  99     def test_construction_2(self):
 100         # Simple server frame
 101         frm = self.fr("wf:b'foo'")
 102         assert not frm.header.mask
 103         assert not frm.header.masking_key
 104 
 105         # Simple client frame
 106         frm = self.fr("wf:b'foo'", is_client=True)
 107         assert frm.header.mask
 108         assert frm.header.masking_key
 109         frm = self.fr("wf:b'foo':k'abcd'", is_client=True)
 110         assert frm.header.mask
 111         assert frm.header.masking_key == b'abcd'
 112 
 113         # Server frame, mask explicitly set
 114         frm = self.fr("wf:b'foo':mask")
 115         assert frm.header.mask
 116         assert frm.header.masking_key
 117         frm = self.fr("wf:b'foo':k'abcd'")
 118         assert frm.header.mask
 119         assert frm.header.masking_key == b'abcd'
 120 
 121         # Client frame, mask explicitly unset
 122         frm = self.fr("wf:b'foo':-mask", is_client=True)
 123         assert not frm.header.mask
 124         assert not frm.header.masking_key
 125
