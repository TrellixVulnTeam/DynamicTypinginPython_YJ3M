   6 def test_send_chunk():
   7     v = b"foobarfoobar"
   8     for bs in range(1, len(v) + 2):
   9         s = io.BytesIO()
  10         writer.send_chunk(s, v, bs, 0, len(v))
  11         assert s.getvalue() == v
  12         for start in range(len(v)):
  13             for end in range(len(v)):
  14                 s = io.BytesIO()
  15                 writer.send_chunk(s, v, bs, start, end)
  16                 assert s.getvalue() == v[start:end]

  57 def test_write_values_pauses():
  58     tst = "".join(str(i) for i in range(10)).encode()
  59     for i in range(2, 10):
  60         s = io.BytesIO()
  61         writer.write_values(
  62             s, [tst], [(2, "pause", 0), (1, "pause", 0)], blocksize=i
  63         )
  64         assert s.getvalue() == tst
  65 
  66     for i in range(2, 10):
  67         s = io.BytesIO()
  68         writer.write_values(s, [tst], [(1, "pause", 0)], blocksize=i)
  69         assert s.getvalue() == tst
  70 
  71     tst = [tst] * 5
  72     for i in range(2, 10):
  73         s = io.BytesIO()
  74         writer.write_values(s, tst[:], [(1, "pause", 0)], blocksize=i)
  75         assert s.getvalue() == b"".join(tst)
