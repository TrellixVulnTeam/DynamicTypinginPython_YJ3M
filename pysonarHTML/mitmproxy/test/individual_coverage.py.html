  14 def run_tests(src, test, fail):
  15     stderr = io.StringIO()
  16     stdout = io.StringIO()
  17     with contextlib.redirect_stderr(stderr):
  18         with contextlib.redirect_stdout(stdout):
  19             e = pytest.main([
  20                 '-qq',
  21                 '--disable-pytest-warnings',
  22                 '--cov', src.replace('.py', '').replace('/', '.'),
  23                 '--cov-fail-under', '100',
  24                 '--cov-report', 'term-missing:skip-covered',
  25                 '-o', 'faulthandler_timeout=0',
  26                 test
  27             ])
  28 
  29     if e == 0:
  30         if fail:
  31             print("FAIL DUE TO UNEXPECTED SUCCESS:", src, "Please remove this file from setup.cfg tool:individual_coverage/exclude.")
  32             e = 42
  33         else:
  34             print("Success:", src)
  35     else:
  36         if fail:
  37             print("Ignoring allowed fail:", src)
  38             e = 0
  39         else:
  40             cov = [l for l in stdout.getvalue().split("\n") if (src in l) or ("was never imported" in l)]
  41             if len(cov) == 1:
  42                 print("FAIL:", cov[0])
  43             else:
  44                 print("FAIL:", src, test, stdout.getvalue(), stdout.getvalue())
  45                 print(stderr.getvalue())
  46                 print(stdout.getvalue())
  47 
  48     sys.exit(e)
  49

  51 def start_pytest(src, test, fail):
  52     # run pytest in a new process, otherwise imports and modules might conflict
  53     proc = multiprocessing.Process(target=run_tests, args=(src, test, fail))
  54     proc.start()
  55     proc.join()
  56     return (src, test, proc.exitcode)

  59 def main():
  60     c = configparser.ConfigParser()
  61     c.read('setup.cfg')
  62     fs = c['tool:individual_coverage']['exclude'].strip().split('\n')
  63     no_individual_cov = [f.strip() for f in fs]
  64 
  65     excluded = ['mitmproxy/contrib/', 'mitmproxy/test/', 'mitmproxy/tools/', 'mitmproxy/platform/']
  66     src_files = glob.glob('mitmproxy/**/*.py', recursive=True) + glob.glob('pathod/**/*.py', recursive=True)
  67     src_files = [f for f in src_files if os.path.basename(f) != '__init__.py']
  68     src_files = [f for f in src_files if not any(os.path.normpath(p) in f for p in excluded)]
  69 
  70     ps = []
  71     for src in sorted(src_files):
  72         test = os.path.join("test", os.path.dirname(src), "test_" + os.path.basename(src))
  73         if os.path.isfile(test):
  74             ps.append((src, test, src in no_individual_cov))
  75 
  76     result = list(itertools.starmap(start_pytest, ps))
  77 
  78     if any(e != 0 for _, _, e in result):
  79         sys.exit(1)
  80         pass
