 127     def archive(self, path):
 128         # ZipFile and tarfile have slightly different APIs. Fix that.
 129         if self.system == "Windows":
 130             a = zipfile.ZipFile(path, "w")
 131             a.add = a.write
 132             return a
 133         else:
 134             return tarfile.open(path, "w:gz")

 136     def archive_name(self, bdist: str) -> str:
 137         if self.system == "Windows":
 138             ext = "zip"
 139         else:
 140             ext = "tar.gz"
 141         return "{project}-{version}-{platform}.{ext}".format(
 142             project=bdist,
 143             version=self.version,
 144             platform=self.platform_tag,
 145             ext=ext
 146         )

 209     def check_version(self) -> None:
 210         """
 211         Check that version numbers match our conventions.
 212         Raises a ValueError if there is a mismatch.
 213         """
 214         with open(pathlib.Path(self.root_dir) / "mitmproxy" / "version.py") as f:
 215             contents = f.read()
 216         match = re.search(r'^VERSION = "(.+?)"', contents, re.M)
 217         assert match
 218         version = match.group(1)
 219 
 220         if self.is_prod_release:
 221             # For production releases, we require strict version equality
 222             if self.version != version:
 223                 raise ValueError(f"Tag is {self.tag}, but mitmproxy/version.py is {version}.")
 224         elif not self.is_maintenance_branch:
 225             # Commits on maintenance branches don't need the dev suffix. This
 226             # allows us to incorporate and test commits between tagged releases.
 227             # For snapshots, we only ensure that mitmproxy/version.py contains a
 228             # dev release.
 229             version_info = parver.Version.parse(version)
 230             if not version_info.is_devrelease:
 231                 raise ValueError(f"Non-production releases must have dev suffix: {version}")
 232 
 233     @property
 234     def is_maintenance_branch(self) -> bool:
 235         """
 236             Is this an untagged commit on a maintenance branch?
 237         """
 238         if not self.tag and self.branch and re.match(r"v\d+\.x", self.branch):
 239             return True
 240         return False

 246     @property
 247     def is_prod_release(self) -> bool:
 248         if not (self.tag and self.tag.startswith("v")):
 249             return False
 250         try:
 251             v = parver.Version.parse(self.version, strict=True)
 252         except (parver.ParseError, BuildError):
 253             return False
 254         return not v.is_prerelease

 316     @property
 317     def version(self) -> str:
 318         if self.tag:
 319             if self.tag.startswith("v"):
 320                 try:
 321                     parver.Version.parse(self.tag[1:], strict=True)
 322                 except parver.ParseError:
 323                     return self.tag
 324                 return self.tag[1:]
 325             return self.tag
 326         elif self.branch:
 327             return self.branch
 328         else:
 329             raise BuildError("We're on neither a tag nor a branch - could not establish version")

 372 def build_pyinstaller(be: BuildEnviron):  # pragma: no cover
 373     click.echo("Building pyinstaller package...")
 374 
 375     PYINSTALLER_SPEC = os.path.join(be.release_dir, "specs")
 376     # PyInstaller 3.2 does not bundle pydivert's Windivert binaries
 377     PYINSTALLER_HOOKS = os.path.abspath(os.path.join(be.release_dir, "hooks"))
 378     PYINSTALLER_TEMP = os.path.abspath(os.path.join(be.build_dir, "pyinstaller"))
 379     PYINSTALLER_DIST = os.path.abspath(os.path.join(be.build_dir, "binaries", be.platform_tag))
 380 
 381     # https://virtualenv.pypa.io/en/latest/userguide.html#windows-notes
 382     # scripts and executables on Windows go in ENV\Scripts\ instead of ENV/bin/
 383     if platform.system() == "Windows":
 384         PYINSTALLER_ARGS = [
 385             # PyInstaller < 3.2 does not handle Python 3.5's ucrt correctly.
 386             "-p", r"C:\Program Files (x86)\Windows Kits\10\Redist\ucrt\DLLs\x86",
 387         ]
 388     else:
 389         PYINSTALLER_ARGS = []
 390 
 391     if os.path.exists(PYINSTALLER_TEMP):
 392         shutil.rmtree(PYINSTALLER_TEMP)
 393     if os.path.exists(PYINSTALLER_DIST):
 394         shutil.rmtree(PYINSTALLER_DIST)
 395 
 396     for bdist, tools in sorted(be.bdists.items()):
 397         with be.archive(os.path.join(be.dist_dir, be.archive_name(bdist))) as archive:
 398             for tool in tools:
 399                 # We can't have a folder and a file with the same name.
 400                 if tool == "mitmproxy":
 401                     tool = "mitmproxy_main"
 402                 # This is PyInstaller, so it messes up paths.
 403                 # We need to make sure that we are in the spec folder.
 404                 with chdir(PYINSTALLER_SPEC):
 405                     click.echo("Building PyInstaller %s binary..." % tool)
 406                     excludes = []
 407                     if tool != "mitmweb":
 408                         excludes.append("mitmproxy.tools.web")
 409                     if tool != "mitmproxy_main":
 410                         excludes.append("mitmproxy.tools.console")
 411 
 412                     subprocess.check_call(
 413                         [
 414                             "pyinstaller",
 415                             "--clean",
 416                             "--workpath", PYINSTALLER_TEMP,
 417                             "--distpath", PYINSTALLER_DIST,
 418                             "--additional-hooks-dir", PYINSTALLER_HOOKS,
 419                             "--onefile",
 420                             "--console",
 421                             "--icon", "icon.ico",
 422                             # This is PyInstaller, so setting a
 423                             # different log level obviously breaks it :-)
 424                             # "--log-level", "WARN",
 425                         ]
 426                         + [x for e in excludes for x in ["--exclude-module", e]]
 427                         + PYINSTALLER_ARGS
 428                         + [tool]
 429                     )
 430                     # Delete the spec file - we're good without.
 431                     os.remove("{}.spec".format(tool))
 432 
 433                 # Test if it works at all O:-)
 434                 executable = os.path.join(PYINSTALLER_DIST, tool)
 435                 if platform.system() == "Windows":
 436                     executable += ".exe"
 437 
 438                 # Remove _main suffix from mitmproxy executable
 439                 if "_main" in executable:
 440                     shutil.move(
 441                         executable,
 442                         executable.replace("_main", "")
 443                     )
 444                     executable = executable.replace("_main", "")
 445 
 446                 click.echo("> %s --version" % executable)
 447                 click.echo(subprocess.check_output([executable, "--version"]).decode())
 448 
 449                 archive.add(executable, os.path.basename(executable))
 450         click.echo("Packed {}.".format(be.archive_name(bdist)))

 453 def build_wininstaller(be: BuildEnviron):  # pragma: no cover
 454     if not be.build_key:
 455         click.echo("Cannot build windows installer without secret key.")
 456         return
 457     click.echo("Building wininstaller package...")
 458 
 459     IB_VERSION = "20.3.0"
 460     IB_DIR = pathlib.Path(be.release_dir) / "installbuilder"
 461     IB_SETUP = IB_DIR / "setup" / f"{IB_VERSION}-installer.exe"
 462     IB_CLI = fr"C:\Program Files (x86)\VMware InstallBuilder Enterprise {IB_VERSION}\bin\builder-cli.exe"
 463     IB_LICENSE = IB_DIR / "license.xml"
 464 
 465     if not os.path.isfile(IB_CLI):
 466         if not os.path.isfile(IB_SETUP):
 467             click.echo("Downloading InstallBuilder...")
 468 
 469             def report(block, blocksize, total):
 470                 done = block * blocksize
 471                 if round(100 * done / total) != round(100 * (done - blocksize) / total):
 472                     click.secho(f"Downloading... {round(100 * done / total)}%")
 473 
 474             urllib.request.urlretrieve(
 475                 f"https://installbuilder.com/installbuilder-enterprise-{IB_VERSION}-windows-installer.exe",
 476                 IB_SETUP.with_suffix(".tmp"),
 477                 reporthook=report
 478             )
 479             shutil.move(str(IB_SETUP.with_suffix(".tmp")), str(IB_SETUP))
 480 
 481         click.echo("Install InstallBuilder...")
 482         subprocess.run([str(IB_SETUP), "--mode", "unattended", "--unattendedmodeui", "none"], check=True)
 483         assert os.path.isfile(IB_CLI)
 484 
 485     click.echo("Decrypt InstallBuilder license...")
 486     f = cryptography.fernet.Fernet(be.build_key.encode())
 487     with open(IB_LICENSE.with_suffix(".xml.enc"), "rb") as infile, \
 488             open(IB_LICENSE, "wb") as outfile:
 489         outfile.write(f.decrypt(infile.read()))
 490 
 491     click.echo("Run InstallBuilder...")
 492     subprocess.run([
 493         IB_CLI,
 494         "build",
 495         str(IB_DIR / "mitmproxy.xml"),
 496         "windows",
 497         "--license", str(IB_LICENSE),
 498         "--setvars", f"project.version={be.version}",
 499         "--verbose"
 500     ], check=True)
 501     assert os.path.isfile(
 502         os.path.join(be.dist_dir, f"mitmproxy-{be.version}-windows-installer.exe"))

 513 @cli.command("build")
 514 def build():  # pragma: no cover
 515     """
 516         Build a binary distribution
 517     """
 518     be = BuildEnviron.from_env()
 519     be.dump_info()
 520 
 521     be.check_version()
 522     os.makedirs(be.dist_dir, exist_ok=True)
 523 
 524     if be.should_build_wheel:
 525         build_wheel(be)
 526     if be.should_build_docker:
 527         build_docker_image(be)
 528     if be.should_build_pyinstaller:
 529         build_pyinstaller(be)
 530     if be.should_build_wininstaller:
 531         build_wininstaller(be)

 534 @cli.command("upload")
 535 def upload():  # pragma: no cover
 536     """
 537         Upload build artifacts
 538 
 539         Uploads the wheels package to PyPi.
 540         Uploads the Pyinstaller and wheels packages to the snapshot server.
 541         Pushes the Docker image to Docker Hub.
 542     """
 543     be = BuildEnviron.from_env()
 544     be.dump_info()
 545 
 546     if be.is_pull_request:
 547         click.echo("Refusing to upload artifacts from a pull request!")
 548         return
 549 
 550     if be.should_upload_aws:
 551         num_files = len([name for name in os.listdir(be.dist_dir) if os.path.isfile(name)])
 552         click.echo(f"Uploading {num_files} files to AWS dir {be.upload_dir}...")
 553         subprocess.check_call([
 554             "aws", "s3", "cp",
 555             "--acl", "public-read",
 556             be.dist_dir + "/",
 557             "s3://snapshots.mitmproxy.org/{}/".format(be.upload_dir),
 558             "--recursive",
 559         ])
 560 
 561     if be.should_upload_pypi:
 562         whl = glob.glob(os.path.join(be.dist_dir, 'mitmproxy-*-py3-none-any.whl'))[0]
 563         click.echo("Uploading {} to PyPi...".format(whl))
 564         subprocess.check_call(["twine", "upload", whl])
 565 
 566     if be.should_upload_docker:
 567         click.echo("Uploading Docker image to tag={}...".format(be.docker_tag))
 568         subprocess.check_call([
 569             "docker",
 570             "login",
 571             "-u", be.docker_username,
 572             "-p", be.docker_password,
 573         ])
 574         subprocess.check_call(["docker", "push", be.docker_tag])
 575         if be.is_prod_release:
 576             subprocess.check_call(["docker", "tag", be.docker_tag, "mitmproxy/mitmproxy:latest"])
 577             subprocess.check_call(["docker", "push", "mitmproxy/mitmproxy:latest"])
