  14 """
  15 Hook for cryptography module from the Python Cryptography Authority.
  16 """
  17 
  18 import os.path
  19 import glob
  20 
  21 from PyInstaller.compat import EXTENSION_SUFFIXES
  22 from PyInstaller.utils.hooks import collect_submodules, get_module_file_attribute
  23 from PyInstaller.utils.hooks import copy_metadata
  24 
  25 # get the package data so we can load the backends
  26 datas = copy_metadata('cryptography')
  27 
  28 # Add the backends as hidden imports
  29 hiddenimports = collect_submodules('cryptography.hazmat.backends')
  30 
  31 # Add the OpenSSL FFI binding modules as hidden imports
  32 hiddenimports += collect_submodules('cryptography.hazmat.bindings.openssl') + ['_cffi_backend']
  33 
  34 
  35 # Include the cffi extensions as binaries in a subfolder named like the package.
  36 # The cffi verifier expects to find them inside the package directory for
  37 # the main module. We cannot use hiddenimports because that would add the modules
  38 # outside the package.
  39 binaries = []
  40 cryptography_dir = os.path.dirname(get_module_file_attribute('cryptography'))
  41 for ext in EXTENSION_SUFFIXES:
  42     ffimods = glob.glob(os.path.join(cryptography_dir, '*_cffi_*%s*' % ext))
  43     for f in ffimods:
  44         binaries.append((f, 'cryptography'))
