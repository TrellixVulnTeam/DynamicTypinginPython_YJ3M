  10 
  11 This script automatically scans all visited webpages for XSS and SQLi vulnerabilities.
  12 
  13 Usage: mitmproxy -s xss_scanner.py
  14 
  15 This script scans for vulnerabilities by injecting a fuzzing payload (see PAYLOAD below) into 4 different places
  16 and examining the HTML to look for XSS and SQLi injection vulnerabilities. The XSS scanning functionality works by
  17 looking to see whether it is possible to inject HTML based off of of where the payload appears in the page and what
  18 characters are escaped. In addition, it also looks for any script tags that load javascript from unclaimed domains.
  19 The SQLi scanning functionality works by using regular expressions to look for errors from a number of different
  20 common databases. Since it is only looking for errors, it will not find blind SQLi vulnerabilities.
  21 
  22 The 4 places it injects the payload into are:
  23 1. URLs         (e.g. https://example.com/ -> https://example.com/PAYLOAD/)
  24 2. Queries      (e.g. https://example.com/index.html?a=b -> https://example.com/index.html?a=PAYLOAD)
  25 3. Referers     (e.g. The referer changes from https://example.com to PAYLOAD)
  26 4. User Agents  (e.g. The UA changes from Chrome to PAYLOAD)
  27 
  28 Reports from this script show up in the event log (viewable by pressing e) and formatted like:
  29 
  30 ===== XSS Found ====
  31 XSS URL: http://daviddworken.com/vulnerableUA.php
  32 Injection Point: User Agent
  33 Suggested Exploit: <script>alert(0)</script>
  34 Line: 1029zxcs'd"ao<ac>so[sb]po(pc)se;sl/bsl\eq=3847asd
  35

  89 def find_unclaimed_URLs(body, requestUrl):
  90     """ Look for unclaimed URLs in script tags and log them if found"""
  91     def getValue(attrs: List[Tuple[str, str]], attrName: str) -> Optional[str]:
  92         for name, value in attrs:
  93             if attrName == name:
  94                 return value
  95         return None
  96 
  97     class ScriptURLExtractor(HTMLParser):
  98         script_URLs: List[str] = []
  99 
 100         def handle_starttag(self, tag, attrs):
 101             if (tag == "script" or tag == "iframe") and "src" in [name for name, value in attrs]:
 102                 self.script_URLs.append(getValue(attrs, "src"))
 103             if tag == "link" and getValue(attrs, "rel") == "stylesheet" and "href" in [name for name, value in attrs]:
 104                 self.script_URLs.append(getValue(attrs, "href"))
 105 
 106     parser = ScriptURLExtractor()
 107     parser.feed(body)
 108     for url in parser.script_URLs:
 109         url_parser = urlparse(url)
 110         domain = url_parser.netloc
 111         try:
 112             socket.gethostbyname(domain)
 113         except socket.gaierror:
 114             ctx.log.error(f"XSS found in {requestUrl} due to unclaimed URL \"{url}\".")

 117 def test_end_of_URL_injection(original_body: str, request_URL: str, cookies: Cookies) -> VulnData:
 118     """ Test the given URL for XSS via injection onto the end of the URL and
 119         log the XSS if found """
 120     parsed_URL = urlparse(request_URL)
 121     path = parsed_URL.path
 122     if path != "" and path[-1] != "/":  # ensure the path ends in a /
 123         path += "/"
 124     path += FULL_PAYLOAD.decode('utf-8')  # the path must be a string while the payload is bytes
 125     url = parsed_URL._replace(path=path).geturl()
 126     body = requests.get(url, cookies=cookies).text.lower()
 127     xss_info = get_XSS_data(body, url, "End of URL")
 128     sqli_info = get_SQLi_data(body, original_body, url, "End of URL")
 129     return xss_info, sqli_info

 150 def test_query_injection(original_body: str, request_URL: str, cookies: Cookies):
 151     """ Test the given URL for XSS via injection into URL queries and
 152         log the XSS if found """
 153     parsed_URL = urlparse(request_URL)
 154     query_string = parsed_URL.query
 155     # queries is a list of parameters where each parameter is set to the payload
 156     queries = [query.split("=")[0] + "=" + FULL_PAYLOAD.decode('utf-8') for query in query_string.split("&")]
 157     new_query_string = "&".join(queries)
 158     new_URL = parsed_URL._replace(query=new_query_string).geturl()
 159     body = requests.get(new_URL, cookies=cookies).text.lower()
 160     xss_info = get_XSS_data(body, new_URL, "Query")
 161     sqli_info = get_SQLi_data(body, original_body, new_URL, "Query")
 162     return xss_info, sqli_info

 189 def get_SQLi_data(new_body: str, original_body: str, request_URL: str, injection_point: str) -> Optional[SQLiData]:
 190     """ Return a SQLiDict if there is a SQLi otherwise return None
 191         String String URL String -> (SQLiDict or None) """
 192     # Regexes taken from Damn Small SQLi Scanner: https://github.com/stamparm/DSSS/blob/master/dsss.py#L17
 193     DBMS_ERRORS = {
 194         "MySQL": (r"SQL syntax.*MySQL", r"Warning.*mysql_.*", r"valid MySQL result", r"MySqlClient\."),
 195         "PostgreSQL": (r"PostgreSQL.*ERROR", r"Warning.*\Wpg_.*", r"valid PostgreSQL result", r"Npgsql\."),
 196         "Microsoft SQL Server": (r"Driver.* SQL[\-\_\ ]*Server", r"OLE DB.* SQL Server", r"(\W|\A)SQL Server.*Driver",
 197                                  r"Warning.*mssql_.*", r"(\W|\A)SQL Server.*[0-9a-fA-F]{8}",
 198                                  r"(?s)Exception.*\WSystem\.Data\.SqlClient\.", r"(?s)Exception.*\WRoadhouse\.Cms\."),
 199         "Microsoft Access": (r"Microsoft Access Driver", r"JET Database Engine", r"Access Database Engine"),
 200         "Oracle": (r"\bORA-[0-9][0-9][0-9][0-9]", r"Oracle error", r"Oracle.*Driver", r"Warning.*\Woci_.*", r"Warning.*\Wora_.*"),
 201         "IBM DB2": (r"CLI Driver.*DB2", r"DB2 SQL error", r"\bdb2_\w+\("),
 202         "SQLite": (r"SQLite/JDBCDriver", r"SQLite.Exception", r"System.Data.SQLite.SQLiteException", r"Warning.*sqlite_.*",
 203                    r"Warning.*SQLite3::", r"\[SQLITE_ERROR\]"),
 204         "Sybase": (r"(?i)Warning.*sybase.*", r"Sybase message", r"Sybase.*Server message.*"),
 205     }
 206     for dbms, regexes in DBMS_ERRORS.items():
 207         for regex in regexes:  # type: ignore
 208             if re.search(regex, new_body, re.IGNORECASE) and not re.search(regex, original_body, re.IGNORECASE):
 209                 return SQLiData(request_URL,
 210                                 injection_point,
 211                                 regex,
 212                                 dbms)
 213     return None

 216 # A qc is either ' or "
 217 def inside_quote(qc: str, substring_bytes: bytes, text_index: int, body_bytes: bytes) -> bool:
 218     """ Whether the Numberth occurrence of the first string in the second
 219         string is inside quotes as defined by the supplied QuoteChar """
 220     substring = substring_bytes.decode('utf-8')
 221     body = body_bytes.decode('utf-8')
 222     num_substrings_found = 0
 223     in_quote = False
 224     for index, char in enumerate(body):
 225         # Whether the next chunk of len(substring) chars is the substring
 226         next_part_is_substring = (
 227             (not (index + len(substring) > len(body))) and
 228             (body[index:index + len(substring)] == substring)
 229         )
 230         # Whether this char is escaped with a \
 231         is_not_escaped = (
 232             (index - 1 < 0 or index - 1 > len(body)) or
 233             (body[index - 1] != "\\")
 234         )
 235         if char == qc and is_not_escaped:
 236             in_quote = not in_quote
 237         if next_part_is_substring:
 238             if num_substrings_found == text_index:
 239                 return in_quote
 240             num_substrings_found += 1
 241     return False

 244 def paths_to_text(html: str, string: str) -> List[str]:
 245     """ Return list of Paths to a given str in the given HTML tree
 246           - Note that it does a BFS """
 247 
 248     def remove_last_occurence_of_sub_string(string: str, substr: str) -> str:
 249         """ Delete the last occurrence of substr from str
 250         String String -> String
 251         """
 252         index = string.rfind(substr)
 253         return string[:index] + string[index + len(substr):]
 254 
 255     class PathHTMLParser(HTMLParser):
 256         currentPath = ""
 257         paths: List[str] = []
 258 
 259         def handle_starttag(self, tag, attrs):
 260             self.currentPath += ("/" + tag)
 261 
 262         def handle_endtag(self, tag):
 263             self.currentPath = remove_last_occurence_of_sub_string(self.currentPath, "/" + tag)
 264 
 265         def handle_data(self, data):
 266             if string in data:
 267                 self.paths.append(self.currentPath)
 268 
 269     parser = PathHTMLParser()
 270     parser.feed(html)
 271     return parser.paths

 274 def get_XSS_data(body: Union[str, bytes], request_URL: str, injection_point: str) -> Optional[XSSData]:
 275     """ Return a XSSDict if there is a XSS otherwise return None """
 276     def in_script(text, index, body) -> bool:
 277         """ Whether the Numberth occurrence of the first string in the second
 278             string is inside a script tag """
 279         paths = paths_to_text(body.decode('utf-8'), text.decode("utf-8"))
 280         try:
 281             path = paths[index]
 282             return "script" in path
 283         except IndexError:
 284             return False
 285 
 286     def in_HTML(text: bytes, index: int, body: bytes) -> bool:
 287         """ Whether the Numberth occurrence of the first string in the second
 288             string is inside the HTML but not inside a script tag or part of
 289             a HTML attribute"""
 290         # if there is a < then lxml will interpret that as a tag, so only search for the stuff before it
 291         text = text.split(b"<")[0]
 292         paths = paths_to_text(body.decode('utf-8'), text.decode("utf-8"))
 293         try:
 294             path = paths[index]
 295             return "script" not in path
 296         except IndexError:
 297             return False
 298 
 299     def inject_javascript_handler(html: str) -> bool:
 300         """ Whether you can inject a Javascript:alert(0) as a link """
 301         class injectJSHandlerHTMLParser(HTMLParser):
 302             injectJSHandler = False
 303 
 304             def handle_starttag(self, tag, attrs):
 305                 for name, value in attrs:
 306                     if name == "href" and value.startswith(FRONT_WALL.decode('utf-8')):
 307                         self.injectJSHandler = True
 308 
 309         parser = injectJSHandlerHTMLParser()
 310         parser.feed(html)
 311         return parser.injectJSHandler
 312     # Only convert the body to bytes if needed
 313     if isinstance(body, str):
 314         body = bytes(body, 'utf-8')
 315     # Regex for between 24 and 72 (aka 24*3) characters encapsulated by the walls
 316     regex = re.compile(b"""%s.{24,72}?%s""" % (FRONT_WALL, BACK_WALL))
 317     matches = regex.findall(body)
 318     for index, match in enumerate(matches):
 319         # Where the string is injected into the HTML
 320         in_script_val = in_script(match, index, body)
 321         in_HTML_val = in_HTML(match, index, body)
 322         in_tag = not in_script_val and not in_HTML_val
 323         in_single_quotes = inside_quote("'", match, index, body)
 324         in_double_quotes = inside_quote('"', match, index, body)
 325         # Whether you can inject:
 326         inject_open_angle = b"ao<ac" in match  # open angle brackets
 327         inject_close_angle = b"ac>so" in match  # close angle brackets
 328         inject_single_quotes = b"s'd" in match  # single quotes
 329         inject_double_quotes = b'd"ao' in match  # double quotes
 330         inject_slash = b"sl/bsl" in match  # forward slashes
 331         inject_semi = b"se;sl" in match  # semicolons
 332         inject_equals = b"eq=" in match  # equals sign
 333         if in_script_val and inject_slash and inject_open_angle and inject_close_angle:  # e.g. <script>PAYLOAD</script>
 334             return XSSData(request_URL,
 335                            injection_point,
 336                            '</script><script>alert(0)</script><script>',
 337                            match.decode('utf-8'))
 338         elif in_script_val and in_single_quotes and inject_single_quotes and inject_semi:  # e.g. <script>t='PAYLOAD';</script>
 339             return XSSData(request_URL,
 340                            injection_point,
 341                            "';alert(0);g='",
 342                            match.decode('utf-8'))
 343         elif in_script_val and in_double_quotes and inject_double_quotes and inject_semi:  # e.g. <script>t="PAYLOAD";</script>
 344             return XSSData(request_URL,
 345                            injection_point,
 346                            '";alert(0);g="',
 347                            match.decode('utf-8'))
 348         elif in_tag and in_single_quotes and inject_single_quotes and inject_open_angle and inject_close_angle and inject_slash:
 349             # e.g. <a href='PAYLOAD'>Test</a>
 350             return XSSData(request_URL,
 351                            injection_point,
 352                            "'><script>alert(0)</script>",
 353                            match.decode('utf-8'))
 354         elif in_tag and in_double_quotes and inject_double_quotes and inject_open_angle and inject_close_angle and inject_slash:
 355             # e.g. <a href="PAYLOAD">Test</a>
 356             return XSSData(request_URL,
 357                            injection_point,
 358                            '"><script>alert(0)</script>',
 359                            match.decode('utf-8'))
 360         elif in_tag and not in_double_quotes and not in_single_quotes and inject_open_angle and inject_close_angle and inject_slash:
 361             # e.g. <a href=PAYLOAD>Test</a>
 362             return XSSData(request_URL,
 363                            injection_point,
 364                            '><script>alert(0)</script>',
 365                            match.decode('utf-8'))
 366         elif inject_javascript_handler(body.decode('utf-8')):  # e.g. <html><a href=PAYLOAD>Test</a>
 367             return XSSData(request_URL,
 368                            injection_point,
 369                            'Javascript:alert(0)',
 370                            match.decode('utf-8'))
 371         elif in_tag and in_double_quotes and inject_double_quotes and inject_equals:  # e.g. <a href="PAYLOAD">Test</a>
 372             return XSSData(request_URL,
 373                            injection_point,
 374                            '" onmouseover="alert(0)" t="',
 375                            match.decode('utf-8'))
 376         elif in_tag and in_single_quotes and inject_single_quotes and inject_equals:  # e.g. <a href='PAYLOAD'>Test</a>
 377             return XSSData(request_URL,
 378                            injection_point,
 379                            "' onmouseover='alert(0)' t='",
 380                            match.decode('utf-8'))
 381         elif in_tag and not in_single_quotes and not in_double_quotes and inject_equals:  # e.g. <a href=PAYLOAD>Test</a>
 382             return XSSData(request_URL,
 383                            injection_point,
 384                            " onmouseover=alert(0) t=",
 385                            match.decode('utf-8'))
 386         elif in_HTML_val and not in_script_val and inject_open_angle and inject_close_angle and inject_slash:  # e.g. <html>PAYLOAD</html>
 387             return XSSData(request_URL,
 388                            injection_point,
 389                            '<script>alert(0)</script>',
 390                            match.decode('utf-8'))
 391         else:
 392             return None
 393     return None

 396 # response is mitmproxy's entry point
 397 def response(flow: http.HTTPFlow) -> None:
 398     assert flow.response
 399     cookies_dict = get_cookies(flow)
 400     resp = flow.response.get_text(strict=False)
 401     assert resp
 402     # Example: http://xss.guru/unclaimedScriptTag.html
 403     find_unclaimed_URLs(resp, flow.request.url)
 404     results = test_end_of_URL_injection(resp, flow.request.url, cookies_dict)
 405     log_XSS_data(results[0])
 406     log_SQLi_data(results[1])
 407     # Example: https://daviddworken.com/vulnerableReferer.php
 408     results = test_referer_injection(resp, flow.request.url, cookies_dict)
 409     log_XSS_data(results[0])
 410     log_SQLi_data(results[1])
 411     # Example: https://daviddworken.com/vulnerableUA.php
 412     results = test_user_agent_injection(resp, flow.request.url, cookies_dict)
 413     log_XSS_data(results[0])
 414     log_SQLi_data(results[1])
 415     if "?" in flow.request.url:
 416         # Example: https://daviddworken.com/vulnerable.php?name=
 417         results = test_query_injection(resp, flow.request.url, cookies_dict)
 418         log_XSS_data(results[0])
 419         log_SQLi_data(results[1])
