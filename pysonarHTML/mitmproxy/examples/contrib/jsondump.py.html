   1 """
   2 This script serializes the entire traffic dump, including websocket traffic,
   3 as JSON, and either sends it to a URL or writes to a file. The serialization
   4 format is optimized for Elasticsearch; the script can be used to send all
   5 captured traffic to Elasticsearch directly.
   6 
   7 Usage:
   8 
   9     mitmproxy
  10         --mode reverse:http://example.com/
  11         -s examples/complex/jsondump.py
  12 
  13 Configuration:
  14 
  15     Send to a URL:
  16 
  17         cat > ~/.mitmproxy/config.yaml <<EOF
  18         dump_destination: "https://elastic.search.local/my-index/my-type"
  19         # Optional Basic auth:
  20         dump_username: "never-gonna-give-you-up"
  21         dump_password: "never-gonna-let-you-down"
  22         # Optional base64 encoding of content fields
  23         # to store as binary fields in Elasticsearch:
  24         dump_encodecontent: true
  25         EOF
  26 
  27     Dump to a local file:
  28 
  29         cat > ~/.mitmproxy/config.yaml <<EOF
  30         dump_destination: "/user/rastley/output.log"
  31         EOF
  32 """

  45 class JSONDumper:
  46     """
  47     JSONDumper performs JSON serialization and some extra processing
  48     for out-of-the-box Elasticsearch support, and then either writes
  49     the result to a file or sends it to a URL.
  50     """
  51     def __init__(self):
  52         self.outfile = None
  53         self.transformations = None
  54         self.encode = None
  55         self.url = None
  56         self.lock = None
  57         self.auth = None
  58         self.queue = Queue()
  59 
  60     def done(self):
  61         self.queue.join()
  62         if self.outfile:
  63             self.outfile.close()
  64 
  65     fields = {
  66         'timestamp': (
  67             ('error', 'timestamp'),
  68 
  69             ('request', 'timestamp_start'),
  70             ('request', 'timestamp_end'),
  71 
  72             ('response', 'timestamp_start'),
  73             ('response', 'timestamp_end'),
  74 
  75             ('client_conn', 'timestamp_start'),
  76             ('client_conn', 'timestamp_end'),
  77             ('client_conn', 'timestamp_tls_setup'),
  78 
  79             ('server_conn', 'timestamp_start'),
  80             ('server_conn', 'timestamp_end'),
  81             ('server_conn', 'timestamp_tls_setup'),
  82             ('server_conn', 'timestamp_tcp_setup'),
  83         ),
  84         'ip': (
  85             ('server_conn', 'source_address'),
  86             ('server_conn', 'ip_address'),
  87             ('server_conn', 'address'),
  88             ('client_conn', 'address'),
  89         ),
  90         'ws_messages': (
  91             ('messages', ),
  92         ),
  93         'headers': (
  94             ('request', 'headers'),
  95             ('response', 'headers'),
  96         ),
  97         'content': (
  98             ('request', 'content'),
  99             ('response', 'content'),

 103     def _init_transformations(self):
 104         self.transformations = [
 105             {
 106                 'fields': self.fields['headers'],
 107                 'func': dict,
 108             },
 109             {
 110                 'fields': self.fields['timestamp'],
 111                 'func': lambda t: int(t * 1000),
 112             },
 113             {
 114                 'fields': self.fields['ip'],
 115                 'func': lambda addr: {
 116                     'host': addr[0].replace('::ffff:', ''),
 117                     'port': addr[1],
 118                 },
 119             },
 120             {
 121                 'fields': self.fields['ws_messages'],
 122                 'func': lambda ms: [{
 123                     'type': m[0],
 124                     'from_client': m[1],
 125                     'content': base64.b64encode(bytes(m[2], 'utf-8')) if self.encode else m[2],
 126                     'timestamp': int(m[3] * 1000),
 127                 } for m in ms],
 128             }
 129         ]
 130 
 131         if self.encode:
 132             self.transformations.append({
 133                 'fields': self.fields['content'],
 134                 'func': base64.b64encode,
 135             })

 137     @staticmethod
 138     def transform_field(obj, path, func):
 139         """
 140         Apply a transformation function `func` to a value
 141         under the specified `path` in the `obj` dictionary.
 142         """
 143         for key in path[:-1]:
 144             if not (key in obj and obj[key]):
 145                 return
 146             obj = obj[key]
 147         if path[-1] in obj and obj[path[-1]]:
 148             obj[path[-1]] = func(obj[path[-1]])
 149 
 150     @classmethod
 151     def convert_to_strings(cls, obj):
 152         """
 153         Recursively convert all list/dict elements of type `bytes` into strings.
 154         """
 155         if isinstance(obj, dict):
 156             return {cls.convert_to_strings(key): cls.convert_to_strings(value)
 157                     for key, value in obj.items()}
 158         elif isinstance(obj, list) or isinstance(obj, tuple):
 159             return [cls.convert_to_strings(element) for element in obj]
 160         elif isinstance(obj, bytes):
 161             return str(obj)[2:-1]
 162         return obj

 164     def worker(self):
 165         while True:
 166             frame = self.queue.get()
 167             self.dump(frame)
 168             self.queue.task_done()
 169 
 170     def dump(self, frame):
 171         """
 172         Transform and dump (write / send) a data frame.
 173         """
 174         for tfm in self.transformations:
 175             for field in tfm['fields']:
 176                 self.transform_field(frame, field, tfm['func'])
 177         frame = self.convert_to_strings(frame)
 178 
 179         if self.outfile:
 180             self.lock.acquire()
 181             self.outfile.write(json.dumps(frame) + "\n")
 182             self.lock.release()
 183         else:
 184             requests.post(self.url, json=frame, auth=(self.auth or None))
 185 
 186     @staticmethod
 187     def load(loader):
 188         """
 189         Extra options to be specified in `~/.mitmproxy/config.yaml`.
 190         """
 191         loader.add_option('dump_encodecontent', bool, False,
 192                           'Encode content as base64.')
 193         loader.add_option('dump_destination', str, 'jsondump.out',
 194                           'Output destination: path to a file or URL.')
 195         loader.add_option('dump_username', str, '',
 196                           'Basic auth username for URL destinations.')
 197         loader.add_option('dump_password', str, '',
 198                           'Basic auth password for URL destinations.')
 199 
 200     def configure(self, _):
 201         """
 202         Determine the destination type and path, initialize the output
 203         transformation rules.
 204         """
 205         self.encode = ctx.options.dump_encodecontent
 206 
 207         if ctx.options.dump_destination.startswith('http'):
 208             self.outfile = None
 209             self.url = ctx.options.dump_destination
 210             ctx.log.info('Sending all data frames to %s' % self.url)
 211             if ctx.options.dump_username and ctx.options.dump_password:
 212                 self.auth = (ctx.options.dump_username, ctx.options.dump_password)
 213                 ctx.log.info('HTTP Basic auth enabled.')
 214         else:
 215             self.outfile = open(ctx.options.dump_destination, 'a')
 216             self.url = None
 217             self.lock = Lock()
 218             ctx.log.info('Writing all data frames to %s' % ctx.options.dump_destination)
 219 
 220         self._init_transformations()
 221 
 222         for i in range(FILE_WORKERS if self.outfile else HTTP_WORKERS):
 223             t = Thread(target=self.worker)
 224             t.daemon = True
 225             t.start()
 226 
 227     def response(self, flow):
 228         """
 229         Dump request/response pairs.
 230         """
 231         self.queue.put(flow.get_state())
 232 
 233     def error(self, flow):
 234         """
 235         Dump errors.
 236         """
 237         self.queue.put(flow.get_state())

 239     def websocket_end(self, flow):
 240         """
 241         Dump websocket messages once the connection ends.
 242 
 243         Alternatively, you can replace `websocket_end` with
 244         `websocket_message` if you want the messages to be
 245         dumped one at a time with full metadata. Warning:
 246         this takes up _a lot_ of space.
 247         """
 248         self.queue.put(flow.get_state())

 250     def websocket_error(self, flow):
 251         """
 252         Dump websocket errors.
 253         """
 254         self.queue.put(flow.get_state())
