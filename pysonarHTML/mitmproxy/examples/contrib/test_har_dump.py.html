  47     def test_format_cookies(self, tdata):
  48         with taddons.context() as tctx:
  49             a = tctx.script(tdata.path("../examples/complex/har_dump.py"))
  50 
  51             CA = cookies.CookieAttrs
  52 
  53             f = a.format_cookies([("n", "v", CA([("k", "v")]))])[0]
  54             assert f['name'] == "n"
  55             assert f['value'] == "v"
  56             assert not f['httpOnly']
  57             assert not f['secure']
  58 
  59             f = a.format_cookies([("n", "v", CA([("httponly", None), ("secure", None)]))])[0]
  60             assert f['httpOnly']

  62 
  63             f = a.format_cookies([("n", "v", CA([("expires", "Mon, 24-Aug-2037 00:00:00 GMT")]))])[0]
  64             assert f['expires']
