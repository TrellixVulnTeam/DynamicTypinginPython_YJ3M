   2 from bs4 import BeautifulSoup
   3 from mitmproxy import ctx, http
   4 
   5 
   6 class Injector:
   7     def load(self, loader):
   8         loader.add_option(
   9             "iframe", str, "", "IFrame to inject"
  10         )
  11 
  12     def response(self, flow: http.HTTPFlow) -> None:
  13         if ctx.options.iframe:
  14             html = BeautifulSoup(flow.response.content, "html.parser")
  15             if html.body:
  16                 iframe = html.new_tag(
  17                     "iframe",
  18                     src=ctx.options.iframe,
  19                     frameborder=0,
  20                     height=0,
  21                     width=0)
  22                 html.body.insert(0, iframe)
  23                 flow.response.content = str(html).encode("utf8")
  24
