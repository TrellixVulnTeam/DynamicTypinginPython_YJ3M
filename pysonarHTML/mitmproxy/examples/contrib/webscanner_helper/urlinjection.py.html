  12 class InjectionGenerator:
  13     """Abstract class for an generator of the injection content in order to inject the URL index."""
  14     ENCODING = "UTF8"
  15 
  16     @abc.abstractmethod
  17     def inject(self, index, flow: HTTPFlow):
  18         """Injects the given URL index into the given flow."""
  19         pass

  22 class HTMLInjection(InjectionGenerator):
  23     """Injects the URL index either by creating a new HTML page or by appending is to an existing page."""
  24 
  25     def __init__(self, insert: bool = False):
  26         """Initializes the HTMLInjection.
  27 
  28         Args:
  29             insert: boolean to decide whether to insert the URL index to an existing page (True) or to create a new
  30                 page containing the URL index.
  31         """
  32         self.insert = insert
  33 
  34     @classmethod
  35     def _form_html(cls, url):
  36         return f"<form action=\"{url}\" method=\"POST\"></form>"
  37 
  38     @classmethod
  39     def _link_html(cls, url):
  40         return f"<a href=\"{url}\">link to {url}</a>"
  41 
  42     @classmethod
  43     def index_html(cls, index):
  44         link_htmls = []
  45         for scheme_netloc, paths in index.items():
  46             for path, methods in paths.items():
  47                 url = scheme_netloc + path
  48                 if "POST" in methods:
  49                     link_htmls.append(cls._form_html(url))
  50 
  51                 if "GET" in methods:
  52                     link_htmls.append(cls._link_html(url))
  53         return "</ br>".join(link_htmls)
  54 
  55     @classmethod
  56     def landing_page(cls, index):
  57         return (
  58                 "<head><meta charset=\"UTF-8\"></head><body>"
  59                 + cls.index_html(index)
  60                 + "</body>"
  61         )
  62 
  63     def inject(self, index, flow: HTTPFlow):
  64         if flow.response is not None:
  65             if flow.response.status_code != 404 and not self.insert:
  66                 logger.warning(
  67                     f"URL '{flow.request.url}' didn't return 404 status, "
  68                     f"index page would overwrite valid page.")
  69             elif self.insert:
  70                 content = (flow.response
  71                            .content
  72                            .decode(self.ENCODING, "backslashreplace"))
  73                 if "</body>" in content:
  74                     content = content.replace("</body>", self.index_html(index) + "</body>")
  75                 else:
  76                     content += self.index_html(index)
  77                 flow.response.content = content.encode(self.ENCODING)
  78             else:
  79                 flow.response.content = (self.landing_page(index)
  80                                          .encode(self.ENCODING))

  83 class RobotsInjection(InjectionGenerator):
  84     """Injects the URL index by creating a new robots.txt including the URLs."""
  85 
  86     def __init__(self, directive="Allow"):
  87         self.directive = directive
  88 
  89     @classmethod
  90     def robots_txt(cls, index, directive="Allow"):
  91         lines = ["User-agent: *"]
  92         for scheme_netloc, paths in index.items():
  93             for path, methods in paths.items():
  94                 lines.append(directive + ": " + path)
  95         return "\n".join(lines)
  96 
  97     def inject(self, index, flow: HTTPFlow):
  98         if flow.response is not None:
  99             if flow.response.status_code != 404:
 100                 logger.warning(
 101                     f"URL '{flow.request.url}' didn't return 404 status, "
 102                     f"index page would overwrite valid page.")
 103             else:
 104                 flow.response.content = self.robots_txt(index,
 105                                                         self.directive).encode(
 106                     self.ENCODING)

 109 class SitemapInjection(InjectionGenerator):
 110     """Injects the URL index by creating a new sitemap including the URLs."""
 111 
 112     @classmethod
 113     def sitemap(cls, index):
 114         lines = [
 115             "<?xml version=\"1.0\" encoding=\"UTF-8\"?><urlset xmlns=\"http://www.sitemaps.org/schemas/sitemap/0.9\">"]
 116         for scheme_netloc, paths in index.items():
 117             for path, methods in paths.items():
 118                 url = scheme_netloc + path
 119                 lines.append(f"<url><loc>{html.escape(url)}</loc></url>")
 120         lines.append("</urlset>")
 121         return "\n".join(lines)
 122 
 123     def inject(self, index, flow: HTTPFlow):
 124         if flow.response is not None:
 125             if flow.response.status_code != 404:
 126                 logger.warning(
 127                     f"URL '{flow.request.url}' didn't return 404 status, "
 128                     f"index page would overwrite valid page.")
 129             else:
 130                 flow.response.content = self.sitemap(index).encode(self.ENCODING)

 133 class UrlInjectionAddon:
 134     """ The UrlInjection add-on can be used in combination with web application scanners to improve their crawling
 135     performance.
 136 
 137     The given URls will be injected into the web application. With this, web application scanners can find pages to
 138     crawl much easier. Depending on the Injection generator, the URLs will be injected at different places of the
 139     web application. It is possible to create a landing page which includes the URL (HTMLInjection()), to inject the
 140     URLs to an existing page (HTMLInjection(insert=True)), to create a robots.txt containing the URLs
 141     (RobotsInjection()) or to create a sitemap.xml which includes the URLS (SitemapInjection()).
 142     It is necessary that the web application scanner can find the newly created page containing the URL index. For
 143     example, the newly created page can be set as starting point for the web application scanner.
 144     The URL index needed for the injection can be generated by the UrlIndex Add-on.
 145     """
 146 
 147     def __init__(self, flt: str, url_index_file: str,
 148                  injection_gen: InjectionGenerator):
 149         """Initializes the UrlIndex add-on.
 150 
 151         Args:
 152             flt: mitmproxy filter to decide on which pages the URLs will be injected (str).
 153             url_index_file: Path to the file which includes the URL index in JSON format (e.g. generated by the UrlIndexAddon), given
 154                 as str.
 155             injection_gen: InjectionGenerator that should be used to inject the URLs into the web application.
 156         """
 157         self.name = f"{self.__class__.__name__}-{injection_gen.__class__.__name__}-{self.__hash__()}"
 158         self.flt = flowfilter.parse(flt)
 159         self.injection_gen = injection_gen
 160         with open(url_index_file, "r") as f:
 161             self.url_store = json.load(f)
 162 
 163     def response(self, flow: HTTPFlow):
 164         """Checks if the response matches the filter and such should be injected.
 165         Injects the URL index if appropriate.
 166         """
 167         if flow.response is not None:
 168             if self.flt is not None and self.flt(flow):
 169                 self.injection_gen.inject(self.url_store, flow)
 170                 flow.response.status_code = 200
 171                 flow.response.headers["content-type"] = "text/html"
 172                 logger.debug(f"Set status code to 200 and set content to logged "
 173                              f"urls. Method: {self.injection_gen}")
