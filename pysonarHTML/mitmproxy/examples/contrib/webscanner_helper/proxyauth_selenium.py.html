  25 def randomString(string_length=10):
  26     """Generate a random string of fixed length """

  28     return ''.join(random.choice(letters) for i in range(string_length))

  42 class SeleniumAddon:
  43     """ This Addon can be used in combination with web application scanners in order to help them to authenticate
  44     against a web application.
  45 
  46     Since the authentication is highly dependant on the web application, this add-on includes the abstract method
  47     *login*. In order to use the add-on, a class for the web application inheriting from SeleniumAddon needs to be
  48     created. This class needs to include the concrete selenium actions necessary to authenticate against the web
  49     application. In addition, an authentication oracle which inherits from AuthorizationOracle should be created.
  50     """
  51     def __init__(self, fltr: str, domain: str,
  52                  auth_oracle: AuthorizationOracle):
  53         self.filter = flowfilter.parse(fltr)
  54         self.auth_oracle = auth_oracle
  55         self.domain = domain
  56         self.browser = None
  57         self.set_cookies = False
  58 
  59         options = webdriver.FirefoxOptions()
  60         options.headless = True
  61 
  62         profile = webdriver.FirefoxProfile()
  63         profile.set_preference('network.proxy.type', 0)
  64         self.browser = webdriver.Firefox(firefox_profile=profile,
  65                                          options=options)
  66         self.cookies: List[Dict[str, str]] = []
  67 
  68     def _login(self, flow):
  69         self.cookies = self.login(flow)
  70         self.browser.get("about:blank")
  71         self._set_request_cookies(flow)
  72         self.set_cookies = True
  73 
  74     def request(self, flow: mitmproxy.http.HTTPFlow):
  75         if flow.request.is_replay:
  76             logger.warning("Caught replayed request: " + str(flow))
  77         if (not self.filter or self.filter(flow)) and self.auth_oracle.is_unauthorized_request(flow):
  78             logger.debug("unauthorized request detected, perform login")
  79             self._login(flow)
  80 
  81     # has to be concurrent because replay.client is blocking and replayed flows
  82     # will also call response
  83     @concurrent
  84     def response(self, flow: mitmproxy.http.HTTPFlow):
  85         if flow.response and (self.filter is None or self.filter(flow)):
  86             if self.auth_oracle.is_unauthorized_response(flow):
  87                 self._login(flow)
  88                 new_flow = flow.copy()
  89                 if master and hasattr(master, 'commands'):
  90                     # cast necessary for mypy
  91                     cast(Any, master).commands.call("replay.client", [new_flow])
  92                     count = 0
  93                     while new_flow.response is None and count < 10:
  94                         logger.error("waiting since " + str(count) + " ...")
  95                         count = count + 1
  96                         time.sleep(1)
  97                     if new_flow.response:
  98                         flow.response = new_flow.response
  99                 else:
 100                     logger.warning("Could not call 'replay.client' command since master was not initialized yet.")
 101 
 102             if self.set_cookies and flow.response:
 103                 logger.debug("set set-cookie header for response")
 104                 self._set_set_cookie_headers(flow)
 105                 self.set_cookies = False
 106 
 107     def done(self):
 108         self.browser.close()

 110     def _set_set_cookie_headers(self, flow: mitmproxy.http.HTTPFlow):
 111         if flow.response and self.cookies:
 112             for cookie in self.cookies:
 113                 parts = [f"{cookie['name']}={cookie['value']}"]
 114                 for k, v in cookie_key_name.items():
 115                     if k in cookie and isinstance(cookie[k], str):
 116                         parts.append(f"{v}={cookie[k]}")
 117                     elif k in cookie and isinstance(cookie[k], bool) and cookie[k]:
 118                         parts.append(cookie[k])
 119                 encoded_c = "; ".join(parts)
 120                 flow.response.headers["set-cookie"] = encoded_c
 121 
 122     def _set_request_cookies(self, flow: mitmproxy.http.HTTPFlow):
 123         if self.cookies:
 124             cookies = "; ".join(
 125                 map(lambda c: f"{c['name']}={c['value']}", self.cookies))
 126             flow.request.headers["cookie"] = cookies
