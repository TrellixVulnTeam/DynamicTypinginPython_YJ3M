  96 class TestUrlInjectionAddon:
  97 
  98     def test_init(self, tmpdir):
  99         tmpfile = tmpdir.join("tmpfile")
 100         with open(tmpfile, "w") as tfile:
 101             json.dump(index, tfile)
 102         flt = f"~u .*/site.html$"
 103         url_injection = UrlInjectionAddon(f"~u .*/site.html$", tmpfile, HTMLInjection(insert=True))
 104         assert "http://example.com:80" in url_injection.url_store
 105         fltr = flowfilter.parse(flt)
 106         f = tflow.tflow(resp=tutils.tresp())
 107         f.request.url = "http://example.com/site.html"
 108         assert fltr(f)
 109         assert "http://example.com:80" not in str(f.response.content)
 110         url_injection.response(f)
 111         assert "http://example.com:80" in str(f.response.content)
