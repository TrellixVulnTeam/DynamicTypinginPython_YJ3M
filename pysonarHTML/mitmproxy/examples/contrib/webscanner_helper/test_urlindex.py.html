  47 class TestJSONUrlIndexWriter:

  49     def test_load(self, tmpdir):
  50         tmpfile = tmpdir.join("tmpfile")
  51         with open(tmpfile, "w") as tfile:
  52             tfile.write(
  53                 "{\"http://example.com:80\": {\"/\": {\"GET\": [301]}}, \"http://www.example.com:80\": {\"/\": {\"GET\": [302]}}}")
  54         writer = JSONUrlIndexWriter(filename=tmpfile)
  55         writer.load()
  56         assert 'http://example.com:80' in writer.host_urls
  57         assert '/' in writer.host_urls['http://example.com:80']
  58         assert 'GET' in writer.host_urls['http://example.com:80']['/']
  59         assert 301 in writer.host_urls['http://example.com:80']['/']['GET']
