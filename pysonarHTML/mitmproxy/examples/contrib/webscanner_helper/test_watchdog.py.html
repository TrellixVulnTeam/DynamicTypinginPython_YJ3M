  16     def test_init_file(self, tmpdir):

  35     def test_serverconnect(self, tmpdir):
  36         event = multiprocessing.Event()
  37         w = WatchdogAddon(event, Path(tmpdir), timeout=10)
  38         with mock.patch('mitmproxy.connections.ServerConnection.settimeout') as mock_set_timeout:
  39             w.serverconnect(ServerConnection("127.0.0.1"))
  40         mock_set_timeout.assert_called()

  49     def test_trigger(self, tmpdir):
  50         event = multiprocessing.Event()
  51         w = WatchdogAddon(event, Path(tmpdir))
  52         f = tflow.tflow(resp=tutils.tresp())
  53         f.error = "Test Error"
  54 
  55         with mock.patch.object(logger, 'error') as mock_error:
  56             open_mock = mock.mock_open()
  57             with mock.patch("pathlib.Path.open", open_mock, create=True):
  58                 w.error(f)
  59             mock_error.assert_called()
  60             open_mock.assert_called()

  62     def test_trigger_http_synatx(self, tmpdir):
  63         event = multiprocessing.Event()
  64         w = WatchdogAddon(event, Path(tmpdir))
  65         f = tflow.tflow(resp=tutils.tresp())
  66         f.error = HttpSyntaxException()

  68 
  69         with mock.patch.object(logger, 'error') as mock_error:
  70             open_mock = mock.mock_open()
  71             with mock.patch("pathlib.Path.open", open_mock, create=True):
  72                 w.error(f)
  73             assert not mock_error.called
  74             assert not open_mock.called

  76     def test_timeout(self, tmpdir):
  77         event = multiprocessing.Event()
  78         w = WatchdogAddon(event, Path(tmpdir))
  79 
  80         assert w.not_in_timeout(None, None)
  81         assert w.not_in_timeout(time.time, None)
  82         with mock.patch('time.time', return_value=5):
  83             assert not w.not_in_timeout(3, 20)
  84             assert w.not_in_timeout(3, 1)
