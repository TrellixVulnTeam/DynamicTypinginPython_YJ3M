  44 class SetEncoder(json.JSONEncoder):
  45     def default(self, obj):
  46         if isinstance(obj, set):
  47             return list(obj)
  48         return json.JSONEncoder.default(self, obj)

  51 class JSONUrlIndexWriter(UrlIndexWriter):
  52     """Writes seen URLs as JSON."""
  53 
  54     def __init__(self, *args, **kwargs):
  55         super().__init__(*args, **kwargs)
  56         self.host_urls = {}
  57 
  58     def load(self):
  59         if self.filepath.exists():
  60             with self.filepath.open("r") as f:
  61                 self.host_urls = json.load(f)
  62             for host in self.host_urls.keys():
  63                 for path, methods in self.host_urls[host].items():
  64                     for method, codes in methods.items():
  65                         self.host_urls[host][path] = {method: set(codes)}
  66 
  67     def add_url(self, flow: HTTPFlow):
  68         req = flow.request
  69         res = flow.response
  70 
  71         if req is not None and res is not None:
  72             urls = self.host_urls.setdefault(f"{req.scheme}://{req.host}:{req.port}", dict())
  73             methods = urls.setdefault(req.path, {})
  74             codes = methods.setdefault(req.method, set())
  75             codes.add(res.status_code)
  76 
  77     def save(self):
  78         with self.filepath.open("w") as f:
  79             json.dump(self.host_urls, f, cls=SetEncoder)

  82 class TextUrlIndexWriter(UrlIndexWriter):
  83     """Writes seen URLs as text."""
  84 
  85     def load(self):
  86         pass
  87 
  88     def add_url(self, flow: HTTPFlow):
  89         res = flow.response
  90         req = flow.request
  91         if res is not None and req is not None:
  92             with self.filepath.open("a+") as f:
  93                 f.write(f"{datetime.datetime.utcnow().isoformat()} STATUS: {res.status_code} METHOD: "
  94                         f"{req.method} URL:{req.url}\n")

  97         pass

 111 class UrlIndexAddon:
 112     """Add-on to write seen URLs, either as JSON or as text.
 113 
 114     For example, these URLs can be injected in a web application to improve the crawling of web application scanners.
 115     The injection can be done using the URLInjection Add-on.
 116     """
 117 
 118     index_filter: Optional[Union[str, flowfilter.TFilter]]
 119     writer: UrlIndexWriter
 120 
 121     OPT_FILEPATH = "URLINDEX_FILEPATH"
 122     OPT_APPEND = "URLINDEX_APPEND"
 123     OPT_INDEX_FILTER = "URLINDEX_FILTER"
 124 
 125     def __init__(self, file_path: Union[str, Path], append: bool = True,
 126                  index_filter: Union[str, flowfilter.TFilter] = filter_404, index_format: str = "json"):
 127         """ Initializes the urlindex add-on.
 128 
 129         Args:
 130             file_path: Path to file to which the URL index will be written. Can either be given as str or Path.
 131             append: Bool to decide whether to append new URLs to the given file (as opposed to overwrite the contents
 132                 of the file)
 133             index_filer: A mitmproxy filter with which the seen URLs will be filtered before being written. Can either
 134                 be given as str or as flowfilter.TFilter
 135             index_format: The format of the URL index, can either be "json" or "text".
 136         """
 137 
 138         if isinstance(index_filter, str):
 139             self.index_filter = flowfilter.parse(index_filter)
 140             if self.index_filter is None:
 141                 raise ValueError("Invalid filter expression.")
 142         else:
 143             self.index_filter = index_filter
 144 
 145         file_path = Path(file_path)
 146         try:
 147             self.writer = WRITER[index_format.lower()](file_path)
 148         except KeyError:
 149             raise ValueError(f"Format '{index_format}' is not supported.")
 150 
 151         if not append and file_path.exists():
 152             file_path.unlink()
 153 
 154         self.writer.load()
 155 
 156     def response(self, flow: HTTPFlow):
 157         """Checks if the response should be included in the URL based on the index_filter and adds it to the URL index
 158             if appropriate.
 159         """
 160         if isinstance(self.index_filter, str) or self.index_filter is None:
 161             raise ValueError("Invalid filter expression.")
 162         else:
 163             if self.index_filter(flow):
 164                 self.writer.add_url(flow)
 165 
 166     def done(self):
 167         """Writes the URL index."""
 168         self.writer.save()
