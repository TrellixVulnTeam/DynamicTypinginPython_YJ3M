  14 class MappingAddonConfig:
  15     HTML_PARSER = "html.parser"

  18 class MappingAddon:
  19     """ The mapping add-on can be used in combination with web application scanners to reduce their false positives.
  20 
  21     Many web application scanners produce false positives caused by dynamically changing content of web applications
  22     such as the current time or current measurements. When testing for injection vulnerabilities, web application
  23     scanners are tricked into thinking they changed the content with the injected payload. In realty, the content of
  24     the web application changed notwithstanding the scanner's input. When the mapping add-on is used to map the content
  25     to a fixed value, these false positives can be avoided.
  26     """
  27 
  28     OPT_MAPPING_FILE = "mapping_file"
  29     """File where urls and css selector to mapped content is stored.
  30 
  31     Elements will be replaced with the content given in this file. If the content is none it will be set to the first
  32     seen value.
  33 
  34     Example:
  35 
  36         {
  37             "http://10.10.10.10": {
  38                 "body": "My Text"
  39             },
  40             "URL": {
  41                 "css selector": "Replace with this"
  42             }
  43         }
  44     """
  45 
  46     OPT_MAP_PERSISTENT = "map_persistent"
  47     """Whether to store all new content in the configuration file."""
  48 
  49     def __init__(self, filename: str, persistent: bool = False) -> None:
  50         """ Initializes the mapping add-on
  51 
  52         Args:
  53             filename: str that provides the name of the file in which the urls and css selectors to mapped content is
  54                 stored.
  55             persistent: bool that indicates whether to store all new content in the configuration file.
  56 
  57         Example:
  58             The file in which the mapping config is given should be in the following format:
  59             {
  60                 "http://10.10.10.10": {
  61                     "body": "My Text"
  62                 },
  63                 "<URL>": {
  64                     "<css selector>": "Replace with this"
  65                 }
  66             }
  67         """
  68         self.filename = filename
  69         self.persistent = persistent
  70         self.logger = logging.getLogger(self.__class__.__name__)
  71         with open(filename, "r") as f:
  72             self.mapping_templates = URLDict.load(f)
  73 
  74     def load(self, loader):
  75         loader.add_option(
  76             self.OPT_MAPPING_FILE, str, "",
  77             "File where replacement configuration is stored."
  78         )
  79         loader.add_option(
  80             self.OPT_MAP_PERSISTENT, bool, False,
  81             "Whether to store all new content in the configuration file."
  82         )
  83 
  84     def configure(self, updated):
  85         if self.OPT_MAPPING_FILE in updated:
  86             self.filename = updated[self.OPT_MAPPING_FILE]
  87             with open(self.filename, "r") as f:
  88                 self.mapping_templates = URLDict.load(f)
  89 
  90         if self.OPT_MAP_PERSISTENT in updated:
  91             self.persistent = updated[self.OPT_MAP_PERSISTENT]
  92 
  93     def replace(self, soup: BeautifulSoup, css_sel: str, replace: BeautifulSoup) -> None:
  94         """Replaces the content of soup that matches the css selector with the given replace content."""
  95         for content in soup.select(css_sel):
  96             self.logger.debug(f"replace \"{content}\" with \"{replace}\"")
  97             content.replace_with(copy.copy(replace))
  98 
  99     def apply_template(self, soup: BeautifulSoup, template: Dict[str, typing.Union[BeautifulSoup]]) -> None:
 100         """Applies the given mapping template to the given soup."""
 101         for css_sel, replace in template.items():
 102             mapped = soup.select(css_sel)
 103             if not mapped:
 104                 self.logger.warning(f"Could not find \"{css_sel}\", can not freeze anything.")
 105             else:
 106                 self.replace(soup, css_sel, BeautifulSoup(replace, features=MappingAddonConfig.HTML_PARSER))
 107 
 108     def response(self, flow: HTTPFlow) -> None:
 109         """If a response is received, check if we should replace some content. """
 110         try:
 111             templates = self.mapping_templates[flow]
 112             res = flow.response
 113             if res is not None:
 114                 encoding = res.headers.get("content-encoding", "utf-8")
 115                 content_type = res.headers.get("content-type", "text/html")
 116 
 117                 if "text/html" in content_type and encoding == "utf-8":
 118                     content = BeautifulSoup(res.content, MappingAddonConfig.HTML_PARSER)
 119                     for template in templates:
 120                         self.apply_template(content, template)
 121                     res.content = content.encode(encoding)
 122                 else:
 123                     self.logger.warning(f"Unsupported content type '{content_type}' or content encoding '{encoding}'")
 124         except KeyError:
 125             pass
 126 
 127     def done(self) -> None:
 128         """Dumps all new content into the configuration file if self.persistent is set."""
 129         if self.persistent:
 130 
 131             # make sure that all items are strings and not soups.
 132             def value_dumper(value):
 133                 store = {}
 134                 if value is None:
 135                     return "None"
 136                 try:
 137                     for css_sel, soup in value.items():
 138                         store[css_sel] = str(soup)
 139                 except:
 140                     raise RuntimeError(value)
 141                 return store
 142 
 143             with open(self.filename, "w") as f:
 144                 self.mapping_templates.dump(f, value_dumper)
