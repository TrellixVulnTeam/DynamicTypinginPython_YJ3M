   1 """
   2 This inline script allows conditional TLS Interception based
   3 on a user-defined strategy.
   4 
   5 Example:
   6 
   7     > mitmdump -s tls_passthrough.py
   8 
   9     1. curl --proxy http://localhost:8080 https://example.com --insecure
  10     // works - we'll also see the contents in mitmproxy
  11 
  12     2. curl --proxy http://localhost:8080 https://example.com --insecure
  13     // still works - we'll also see the contents in mitmproxy
  14 
  15     3. curl --proxy http://localhost:8080 https://example.com
  16     // fails with a certificate error, which we will also see in mitmproxy
  17 
  18     4. curl --proxy http://localhost:8080 https://example.com
  19     // works again, but mitmproxy does not intercept and we do *not* see the contents
  20 
  21 Authors: Maximilian Hils, Matthew Tuusberg
  22 """
  23 import collections
  24 import random
  25 
  26 from enum import Enum
  27 
  28 import mitmproxy
  29 from mitmproxy import ctx
  30 from mitmproxy.exceptions import TlsProtocolException
  31 from mitmproxy.proxy.protocol import TlsLayer, RawTCPLayer
  32 
  33 
  34 class InterceptionResult(Enum):
  35     success = True
  36     failure = False
  37     skipped = None
  38 
  39 
  40 class _TlsStrategy:
  41     """
  42     Abstract base class for interception strategies.
  43     """
  44 
  45     def __init__(self):
  46         # A server_address -> interception results mapping
  47         self.history = collections.defaultdict(lambda: collections.deque(maxlen=200))
  48 
  49     def should_intercept(self, server_address):
  50         """
  51         Returns:
  52             True, if we should attempt to intercept the connection.
  53             False, if we want to employ pass-through instead.
  54         """
  55         raise NotImplementedError()
  56 
  57     def record_success(self, server_address):
  58         self.history[server_address].append(InterceptionResult.success)
  59 
  60     def record_failure(self, server_address):
  61         self.history[server_address].append(InterceptionResult.failure)
  62 
  63     def record_skipped(self, server_address):
  64         self.history[server_address].append(InterceptionResult.skipped)
  65 
  66 
  67 class ConservativeStrategy(_TlsStrategy):
  68     """
  69     Conservative Interception Strategy - only intercept if there haven't been any failed attempts
  70     in the history.
  71     """
  72 
  73     def should_intercept(self, server_address):
  74         if InterceptionResult.failure in self.history[server_address]:
  75             return False
  76         return True
  77 
  78 
  79 class ProbabilisticStrategy(_TlsStrategy):
  80     """
  81     Fixed probability that we intercept a given connection.
  82     """
  83 
  84     def __init__(self, p):
  85         self.p = p
  86         super(ProbabilisticStrategy, self).__init__()
  87 
  88     def should_intercept(self, server_address):
  89         return random.uniform(0, 1) < self.p
  90

  92 class TlsFeedback(TlsLayer):
  93     """
  94     Monkey-patch _establish_tls_with_client to get feedback if TLS could be established
  95     successfully on the client connection (which may fail due to cert pinning).
  96     """
  97 
  98     def _establish_tls_with_client(self):
  99         server_address = self.server_conn.address
 100 
 101         try:
 102             super(TlsFeedback, self)._establish_tls_with_client()
 103         except TlsProtocolException as e:
 104             tls_strategy.record_failure(server_address)
 105             raise e
 106         else:
 107             tls_strategy.record_success(server_address)

 110 # inline script hooks below.
 111 
 112 tls_strategy = None
 113 
 114 
 115 def load(l):
 116     l.add_option(
 117         "tlsstrat", int, 0, "TLS passthrough strategy (0-100)",
 118     )
 119 
 120 
 121 def configure(updated):
 122     global tls_strategy
 123     if ctx.options.tlsstrat > 0:
 124         tls_strategy = ProbabilisticStrategy(float(ctx.options.tlsstrat) / 100.0)
 125     else:
 126         tls_strategy = ConservativeStrategy()
 127 
 128 
 129 def next_layer(next_layer):
 130     """
 131     This hook does the actual magic - if the next layer is planned to be a TLS layer,
 132     we check if we want to enter pass-through mode instead.
 133     """
 134     if isinstance(next_layer, TlsLayer) and next_layer._client_tls:
 135         server_address = next_layer.server_conn.address
 136 
 137         if tls_strategy.should_intercept(server_address):
 138             # We try to intercept.
 139             # Monkey-Patch the layer to get feedback from the TLSLayer if interception worked.
 140             next_layer.__class__ = TlsFeedback
 141         else:
 142             # We don't intercept - reply with a pass-through layer and add a "skipped" entry.
 143             mitmproxy.ctx.log("TLS passthrough for %s" % repr(next_layer.server_conn.address), "info")
 144             next_layer_replacement = RawTCPLayer(next_layer.ctx, ignore=True)
 145             next_layer.reply.send(next_layer_replacement)
 146             tls_strategy.record_skipped(server_address)
