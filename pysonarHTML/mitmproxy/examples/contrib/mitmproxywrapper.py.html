  19     def __init__(self, port, extra_arguments=None):

  23     def run_networksetup_command(self, *arguments):
  24         return subprocess.check_output(
  25             ['sudo', 'networksetup'] + list(arguments))

  27     def proxy_state_for_service(self, service):
  28         state = self.run_networksetup_command(
  29             '-getwebproxy',
  30             service).splitlines()
  31         return dict([re.findall(r'([^:]+): (.*)', line)[0] for line in state])
  32 
  33     def enable_proxy_for_service(self, service):
  34         print('Enabling proxy on {}...'.format(service))
  35         for subcommand in ['-setwebproxy', '-setsecurewebproxy']:
  36             self.run_networksetup_command(
  37                 subcommand, service, '127.0.0.1', str(
  38                     self.port))
  39 
  40     def disable_proxy_for_service(self, service):
  41         print('Disabling proxy on {}...'.format(service))
  42         for subcommand in ['-setwebproxystate', '-setsecurewebproxystate']:
  43             self.run_networksetup_command(subcommand, service, 'Off')
  44 
  45     def interface_name_to_service_name_map(self):
  46         order = self.run_networksetup_command('-listnetworkserviceorder')
  47         mapping = re.findall(
  48             r'\(\d+\)\s(.*)$\n\(.*Device: (.+)\)$',
  49             order,
  50             re.MULTILINE)
  51         return dict([(b, a) for (a, b) in mapping])
  52 
  53     def run_command_with_input(self, command, input):
  54         popen = subprocess.Popen(
  55             command,
  56             stdin=subprocess.PIPE,
  57             stdout=subprocess.PIPE)
  58         (stdout, stderr) = popen.communicate(input)
  59         return stdout

  61     def primary_interace_name(self):
  62         scutil_script = 'get State:/Network/Global/IPv4\nd.show\n'
  63         stdout = self.run_command_with_input('/usr/sbin/scutil', scutil_script)
  64         interface, = re.findall(r'PrimaryInterface\s*:\s*(.+)', stdout)
  65         return interface
  66 
  67     def primary_service_name(self):
  68         return self.interface_name_to_service_name_map()[
  69             self.primary_interace_name()]

  71     def proxy_enabled_for_service(self, service):
  72         return self.proxy_state_for_service(service)['Enabled'] == 'Yes'
  73 
  74     def toggle_proxy(self):
  75         new_state = not self.proxy_enabled_for_service(
  76             self.primary_service_name())
  77         for service_name in self.connected_service_names():
  78             if self.proxy_enabled_for_service(service_name) and not new_state:
  79                 self.disable_proxy_for_service(service_name)
  80             elif not self.proxy_enabled_for_service(service_name) and new_state:
  81                 self.enable_proxy_for_service(service_name)
  82 
  83     def connected_service_names(self):
  84         scutil_script = 'list\n'
  85         stdout = self.run_command_with_input('/usr/sbin/scutil', scutil_script)
  86         service_ids = re.findall(r'State:/Network/Service/(.+)/IPv4', stdout)
  87 
  88         service_names = []
  89         for service_id in service_ids:
  90             scutil_script = 'show Setup:/Network/Service/{}\n'.format(
  91                 service_id)
  92             stdout = self.run_command_with_input(
  93                 '/usr/sbin/scutil',
  94                 scutil_script)
  95             service_name, = re.findall(r'UserDefinedName\s*:\s*(.+)', stdout)
  96             service_names.append(service_name)
  97 
  98         return service_names

 100     def wrap_mitmproxy(self):
 101         with self.wrap_proxy():
 102             cmd = ['mitmproxy', '-p', str(self.port)]
 103             if self.extra_arguments:
 104                 cmd.extend(self.extra_arguments)
 105             subprocess.check_call(cmd)

 107     def wrap_honeyproxy(self):
 108         with self.wrap_proxy():
 109             popen = subprocess.Popen('honeyproxy.sh')
 110             try:
 111                 popen.wait()
 112             except KeyboardInterrupt:
 113                 popen.terminate()

 115     @contextlib.contextmanager
 116     def wrap_proxy(self):
 117         connected_service_names = self.connected_service_names()
 118         for service_name in connected_service_names:
 119             if not self.proxy_enabled_for_service(service_name):
 120                 self.enable_proxy_for_service(service_name)
 121 
 122         yield
 123 
 124         for service_name in connected_service_names:
 125             if self.proxy_enabled_for_service(service_name):
 126                 self.disable_proxy_for_service(service_name)
