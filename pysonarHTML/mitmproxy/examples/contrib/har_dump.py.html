  56 def response(flow):
  57     """
  58        Called when a server response has been received.
  59     """
  60 
  61     # -1 indicates that these values do not apply to current request
  62     ssl_time = -1
  63     connect_time = -1
  64 
  65     if flow.server_conn and flow.server_conn not in SERVERS_SEEN:
  66         connect_time = (flow.server_conn.timestamp_tcp_setup -
  67                         flow.server_conn.timestamp_start)
  68 
  69         if flow.server_conn.timestamp_tls_setup is not None:
  70             ssl_time = (flow.server_conn.timestamp_tls_setup -
  71                         flow.server_conn.timestamp_tcp_setup)
  72 
  73         SERVERS_SEEN.add(flow.server_conn)
  74 
  75     # Calculate raw timings from timestamps. DNS timings can not be calculated
  76     # for lack of a way to measure it. The same goes for HAR blocked.
  77     # mitmproxy will open a server connection as soon as it receives the host
  78     # and port from the client connection. So, the time spent waiting is actually
  79     # spent waiting between request.timestamp_end and response.timestamp_start
  80     # thus it correlates to HAR wait instead.
  81     timings_raw = {
  82         'send': flow.request.timestamp_end - flow.request.timestamp_start,
  83         'receive': flow.response.timestamp_end - flow.response.timestamp_start,
  84         'wait': flow.response.timestamp_start - flow.request.timestamp_end,
  85         'connect': connect_time,
  86         'ssl': ssl_time,
  87     }
  88 
  89     # HAR timings are integers in ms, so we re-encode the raw timings to that format.
  90     timings = {
  91         k: int(1000 * v) if v != -1 else -1
  92         for k, v in timings_raw.items()
  93     }
  94 
  95     # full_time is the sum of all timings.
  96     # Timings set to -1 will be ignored as per spec.
  97     full_time = sum(v for v in timings.values() if v > -1)
  98 
  99     started_date_time = datetime.fromtimestamp(flow.request.timestamp_start, timezone.utc).isoformat()
 100 
 101     # Response body size and encoding
 102     response_body_size = len(flow.response.raw_content) if flow.response.raw_content else 0
 103     response_body_decoded_size = len(flow.response.content) if flow.response.content else 0
 104     response_body_compression = response_body_decoded_size - response_body_size
 105 
 106     entry = {
 107         "startedDateTime": started_date_time,
 108         "time": full_time,
 109         "request": {
 110             "method": flow.request.method,
 111             "url": flow.request.url,
 112             "httpVersion": flow.request.http_version,
 113             "cookies": format_request_cookies(flow.request.cookies.fields),
 114             "headers": name_value(flow.request.headers),
 115             "queryString": name_value(flow.request.query or {}),
 116             "headersSize": len(str(flow.request.headers)),
 117             "bodySize": len(flow.request.content),
 118         },
 119         "response": {
 120             "status": flow.response.status_code,
 121             "statusText": flow.response.reason,
 122             "httpVersion": flow.response.http_version,
 123             "cookies": format_response_cookies(flow.response.cookies.fields),
 124             "headers": name_value(flow.response.headers),
 125             "content": {
 126                 "size": response_body_size,
 127                 "compression": response_body_compression,
 128                 "mimeType": flow.response.headers.get('Content-Type', '')
 129             },
 130             "redirectURL": flow.response.headers.get('Location', ''),
 131             "headersSize": len(str(flow.response.headers)),
 132             "bodySize": response_body_size,
 133         },
 134         "cache": {},
 135         "timings": timings,
 136     }
 137 
 138     # Store binary data as base64
 139     if strutils.is_mostly_bin(flow.response.content):
 140         entry["response"]["content"]["text"] = base64.b64encode(flow.response.content).decode()
 141         entry["response"]["content"]["encoding"] = "base64"
 142     else:
 143         entry["response"]["content"]["text"] = flow.response.get_text(strict=False)
 144 
 145     if flow.request.method in ["POST", "PUT", "PATCH"]:
 146         params = [
 147             {"name": a, "value": b}
 148             for a, b in flow.request.urlencoded_form.items(multi=True)
 149         ]
 150         entry["request"]["postData"] = {
 151             "mimeType": flow.request.headers.get("Content-Type", ""),
 152             "text": flow.request.get_text(strict=False),
 153             "params": params
 154         }
 155 
 156     if flow.server_conn.connected():
 157         entry["serverIPAddress"] = str(flow.server_conn.ip_address[0])
 158 
 159     HAR["log"]["entries"].append(entry)

 162 def done():
 163     """
 164         Called once on script shutdown, after any other events.
 165     """
 166     if ctx.options.hardump:
 167         json_dump: str = json.dumps(HAR, indent=2)
 168 
 169         if ctx.options.hardump == '-':
 170             mitmproxy.ctx.log(json_dump)
 171         else:
 172             raw: bytes = json_dump.encode()
 173             if ctx.options.hardump.endswith('.zhar'):
 174                 raw = zlib.compress(raw, 9)
 175 
 176             with open(os.path.expanduser(ctx.options.hardump), "wb") as f:
 177                 f.write(raw)
 178 
 179             mitmproxy.ctx.log("HAR dump finished (wrote %s bytes to file)" % len(json_dump))

 182 def format_cookies(cookie_list):
 183     rv = []
 184 
 185     for name, value, attrs in cookie_list:
 186         cookie_har = {
 187             "name": name,
 188             "value": value,
 189         }
 190 
 191         # HAR only needs some attributes
 192         for key in ["path", "domain", "comment"]:
 193             if key in attrs:
 194                 cookie_har[key] = attrs[key]
 195 
 196         # These keys need to be boolean!
 197         for key in ["httpOnly", "secure"]:
 198             cookie_har[key] = bool(key in attrs)
 199 
 200         # Expiration time needs to be formatted
 201         expire_ts = cookies.get_expiration_ts(attrs)
 202         if expire_ts is not None:
 203             cookie_har["expires"] = datetime.fromtimestamp(expire_ts, timezone.utc).isoformat()
 204 
 205         rv.append(cookie_har)
 206 
 207     return rv

 210 def format_request_cookies(fields):
 211     return format_cookies(cookies.group_cookies(fields))
 212 
 213 
 214 def format_response_cookies(fields):
 215     return format_cookies((c[0], c[1][0], c[1][1]) for c in fields)

 218 def name_value(obj):
 219     """
 220         Convert (key, value) pairs to HAR format.
 221     """
 222     return [{"name": k, "value": v} for k, v in obj.items()]
