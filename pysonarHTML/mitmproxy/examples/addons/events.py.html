  10 class Events:
  11     # Network lifecycle
  12     def clientconnect(self, layer: mitmproxy.proxy.protocol.Layer):
  13         """
  14             A client has connected to mitmproxy. Note that a connection can
  15             correspond to multiple HTTP requests.
  16         """
  17 
  18     def clientdisconnect(self, layer: mitmproxy.proxy.protocol.Layer):
  19         """
  20             A client has disconnected from mitmproxy.
  21         """
  22 
  23     def serverconnect(self, conn: mitmproxy.connections.ServerConnection):
  24         """
  25             Mitmproxy has connected to a server. Note that a connection can
  26             correspond to multiple requests.
  27         """
  28 
  29     def serverdisconnect(self, conn: mitmproxy.connections.ServerConnection):
  30         """
  31             Mitmproxy has disconnected from a server.
  32         """
  33 
  34     def next_layer(self, layer: mitmproxy.proxy.protocol.Layer):
  35         """
  36             Network layers are being switched. You may change which layer will
  37             be used by returning a new layer object from this event.
  38         """
  39 
  40     # General lifecycle
  41     def configure(self, updated: typing.Set[str]):
  42         """
  43             Called when configuration changes. The updated argument is a
  44             set-like object containing the keys of all changed options. This
  45             event is called during startup with all options in the updated set.
  46         """
  47 
  48     def done(self):
  49         """
  50             Called when the addon shuts down, either by being removed from
  51             the mitmproxy instance, or when mitmproxy itself shuts down. On
  52             shutdown, this event is called after the event loop is
  53             terminated, guaranteeing that it will be the final event an addon
  54             sees. Note that log handlers are shut down at this point, so
  55             calls to log functions will produce no output.
  56         """
  57 
  58     def load(self, entry: mitmproxy.addonmanager.Loader):
  59         """
  60             Called when an addon is first loaded. This event receives a Loader
  61             object, which contains methods for adding options and commands. This
  62             method is where the addon configures itself.
  63         """
  64 
  65     def log(self, entry: mitmproxy.log.LogEntry):
  66         """
  67             Called whenever a new log entry is created through the mitmproxy
  68             context. Be careful not to log from this event, which will cause an
  69             infinite loop!
  70         """
  71 
  72     def running(self):
  73         """
  74             Called when the proxy is completely up and running. At this point,
  75             you can expect the proxy to be bound to a port, and all addons to be
  76             loaded.
  77         """
  78 
  79     def update(self, flows: typing.Sequence[mitmproxy.flow.Flow]):
  80         """
  81             Update is called when one or more flow objects have been modified,
  82             usually from a different addon.
  83         """
