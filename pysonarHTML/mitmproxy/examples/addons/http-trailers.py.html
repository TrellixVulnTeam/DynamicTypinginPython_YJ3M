   1 """
   2 This script simply prints all received HTTP Trailers.
   3 
   4 HTTP requests and responses can container trailing headers which are sent after
   5 the body is fully transmitted. Such trailers need to be announced in the initial
   6 headers by name, so the receiving endpoint can wait and read them after the
   7 body.
   8 """
   9 
  10 from mitmproxy import http
  11 from mitmproxy.net.http import Headers
  12 
  13 
  14 def request(flow: http.HTTPFlow):
  15     if flow.request.trailers:
  16         print("HTTP Trailers detected! Request contains:", flow.request.trailers)
  17 
  18 
  19 def response(flow: http.HTTPFlow):
  20     if flow.response.trailers:
  21         print("HTTP Trailers detected! Response contains:", flow.response.trailers)
  22 
  23     if flow.request.path == "/inject_trailers":
  24         flow.response.headers["trailer"] = "x-my-injected-trailer-header"
  25         flow.response.trailers = Headers([
  26             (b"x-my-injected-trailer-header", b"foobar")
  27         ])
  28         print("Injected a new trailer...", flow.response.headers["trailer"])
