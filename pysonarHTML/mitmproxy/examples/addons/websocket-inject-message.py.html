   1 """
   2 Inject a WebSocket message into a running connection.
   3 
   4 This example shows how to inject a WebSocket message to the client.
   5 Every new WebSocket connection will trigger a new asyncio task that
   6 periodically injects a new message to the client.
   7 """
   8 import asyncio
   9 import mitmproxy.websocket
  10 
  11 
  12 class InjectWebSocketMessage:
  13     async def inject(self, flow: mitmproxy.websocket.WebSocketFlow):
  14         i = 0
  15         while not flow.ended and not flow.error:
  16             await asyncio.sleep(5)
  17             flow.inject_message(flow.client_conn, f'This is the #{i} injected message!')
  18             i += 1
  19 
  20     def websocket_start(self, flow):
  21         asyncio.get_event_loop().create_task(self.inject(flow))
