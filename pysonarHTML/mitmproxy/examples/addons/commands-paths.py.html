   1 """Handle file paths as command arguments."""
   2 import typing
   3 
   4 from mitmproxy import command
   5 from mitmproxy import ctx
   6 from mitmproxy import flow
   7 from mitmproxy import types
   8 
   9 
  10 class MyAddon:
  11     @command.command("myaddon.histogram")
  12     def histogram(
  13         self,
  14         flows: typing.Sequence[flow.Flow],
  15         path: types.Path,
  16     ) -> None:
  17         totals = {}
  18         for f in flows:
  19             totals[f.request.host] = totals.setdefault(f.request.host, 0) + 1
  20 
  21         with open(path, "w+") as fp:
  22             for cnt, dom in sorted([(v, k) for (k, v) in totals.items()]):
  23                 fp.write("%s: %s\n" % (cnt, dom))
  24 
  25         ctx.log.alert("done")
  26
