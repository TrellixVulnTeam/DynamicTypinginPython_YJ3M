   1 """React to configuration changes."""
   2 import typing
   3 
   4 from mitmproxy import ctx
   5 from mitmproxy import exceptions
   6 
   7 
   8 class AddHeader:
   9     def load(self, loader):
  10         loader.add_option(
  11             name = "addheader",
  12             typespec = typing.Optional[int],
  13             default = None,
  14             help = "Add a header to responses",
  15         )
  16 
  17     def configure(self, updates):
  18         if "addheader" in updates:
  19             if ctx.options.addheader is not None and ctx.options.addheader > 100:
  20                 raise exceptions.OptionsError("addheader must be <= 100")
  21 
  22     def response(self, flow):
  23         if ctx.options.addheader is not None:
  24             flow.response.headers["addheader"] = str(ctx.options.addheader)
