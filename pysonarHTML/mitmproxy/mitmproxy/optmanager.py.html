  23 class _Option:
  24     __slots__ = ("name", "typespec", "value", "_default", "choices", "help")
  25 
  26     def __init__(
  27         self,
  28         name: str,
  29         typespec: type,
  30         default: typing.Any,
  31         help: str,
  32         choices: typing.Optional[typing.Sequence[str]]
  33     ) -> None:
  34         typecheck.check_option_type(name, default, typespec)
  35         self.name = name
  36         self.typespec = typespec
  37         self._default = default
  38         self.value = unset
  39         self.help = textwrap.dedent(help).strip().replace("\n", " ")
  40         self.choices = choices
  41 
  42     def __repr__(self):
  43         return "{value} [{type}]".format(value=self.current(), type=self.typespec)
  44 
  45     @property
  46     def default(self):
  47         return copy.deepcopy(self._default)
  48 
  49     def current(self) -> typing.Any:
  50         if self.value is unset:
  51             v = self.default
  52         else:
  53             v = self.value
  54         return copy.deepcopy(v)
  55 
  56     def set(self, value: typing.Any) -> None:
  57         typecheck.check_option_type(self.name, value, self.typespec)
  58         self.value = value
  59 
  60     def reset(self) -> None:
  61         self.value = unset
  62 
  63     def has_changed(self) -> bool:
  64         return self.current() != self.default
  65 
  66     def __eq__(self, other) -> bool:
  67         for i in self.__slots__:
  68             if getattr(self, i) != getattr(other, i):
  69                 return False
  70         return True
  71 
  72     def __deepcopy__(self, _):
  73         o = _Option(
  74             self.name, self.typespec, self.default, self.help, self.choices
  75         )
  76         if self.has_changed():
  77             o.value = self.current()
  78         return o

  81 class OptManager:
  82     """
  83         OptManager is the base class from which Options objects are derived.
  84 
  85         .changed is a blinker Signal that triggers whenever options are
  86         updated. If any handler in the chain raises an exceptions.OptionsError
  87         exception, all changes are rolled back, the exception is suppressed,
  88         and the .errored signal is notified.
  89 
  90         Optmanager always returns a deep copy of options to ensure that
  91         mutation doesn't change the option state inadvertently.
  92     """
  93     def __init__(self):
  94         self.deferred: typing.Dict[str, str] = {}
  95         self.changed = blinker.Signal()
  96         self.errored = blinker.Signal()
  97         # Options must be the last attribute here - after that, we raise an
  98         # error for attribute assigment to unknown options.
  99         self._options: typing.Dict[str, typing.Any] = {}
 100 
 101     def add_option(
 102         self,
 103         name: str,
 104         typespec: type,
 105         default: typing.Any,
 106         help: str,
 107         choices: typing.Optional[typing.Sequence[str]] = None
 108     ) -> None:
 109         self._options[name] = _Option(name, typespec, default, help, choices)
 110         self.changed.send(self, updated={name})
 111 
 112     @contextlib.contextmanager
 113     def rollback(self, updated, reraise=False):
 114         old = copy.deepcopy(self._options)
 115         try:
 116             yield
 117         except exceptions.OptionsError as e:
 118             # Notify error handlers
 119             self.errored.send(self, exc=e)
 120             # Rollback
 121             self.__dict__["_options"] = old
 122             self.changed.send(self, updated=updated)
 123             if reraise:
 124                 raise e
 125 
 126     def subscribe(self, func, opts):
 127         """
 128             Subscribe a callable to the .changed signal, but only for a
 129             specified list of options. The callable should accept arguments
 130             (options, updated), and may raise an OptionsError.
 131 
 132             The event will automatically be unsubscribed if the callable goes out of scope.
 133         """
 134         for i in opts:
 135             if i not in self._options:
 136                 raise exceptions.OptionsError("No such option: %s" % i)
 137 
 138         # We reuse blinker's safe reference functionality to cope with weakrefs
 139         # to bound methods.
 140         func = blinker._saferef.safe_ref(func)
 141 
 142         @functools.wraps(func)
 143         def _call(options, updated):
 144             if updated.intersection(set(opts)):
 145                 f = func()
 146                 if f:
 147                     f(options, updated)
 148                 else:
 149                     self.changed.disconnect(_call)
 150 
 151         # Our wrapper function goes out of scope immediately, so we have to set
 152         # weakrefs to false. This means we need to keep our own weakref, and
 153         # clean up the hook when it's gone.
 154         self.changed.connect(_call, weak=False)
 155 
 156     def __eq__(self, other):
 157         if isinstance(other, OptManager):
 158             return self._options == other._options
 159         return False
 160 
 161     def __deepcopy__(self, memodict = None):
 162         o = OptManager()
 163         o.__dict__["_options"] = copy.deepcopy(self._options, memodict)
 164         return o

 166     __copy__ = __deepcopy__

 168     def __getattr__(self, attr):
 169         if attr in self._options:
 170             return self._options[attr].current()
 171         else:
 172             raise AttributeError("No such option: %s" % attr)
 173 
 174     def __setattr__(self, attr, value):
 175         # This is slightly tricky. We allow attributes to be set on the instance
 176         # until we have an _options attribute. After that, assignment is sent to
 177         # the update function, and will raise an error for unknown options.
 178         opts = self.__dict__.get("_options")
 179         if not opts:
 180             super().__setattr__(attr, value)
 181         else:
 182             self.update(**{attr: value})
 183 
 184     def keys(self):
 185         return set(self._options.keys())
 186 
 187     def items(self):
 188         return self._options.items()
 189 
 190     def __contains__(self, k):
 191         return k in self._options

 193     def reset(self):
 194         """
 195             Restore defaults for all options.
 196         """
 197         for o in self._options.values():
 198             o.reset()
 199         self.changed.send(self, updated=set(self._options.keys()))
 200 
 201     def update_known(self, **kwargs):
 202         """
 203             Update and set all known options from kwargs. Returns a dictionary
 204             of unknown options.
 205         """
 206         known, unknown = {}, {}
 207         for k, v in kwargs.items():
 208             if k in self._options:
 209                 known[k] = v
 210             else:
 211                 unknown[k] = v
 212         updated = set(known.keys())
 213         if updated:
 214             with self.rollback(updated, reraise=True):
 215                 for k, v in known.items():
 216                     self._options[k].set(v)
 217                 self.changed.send(self, updated=updated)
 218         return unknown
 219 
 220     def update_defer(self, **kwargs):
 221         unknown = self.update_known(**kwargs)
 222         self.deferred.update(unknown)
 223 
 224     def update(self, **kwargs):
 225         u = self.update_known(**kwargs)
 226         if u:
 227             raise KeyError("Unknown options: %s" % ", ".join(u.keys()))
 228 
 229     def setter(self, attr):
 230         """
 231             Generate a setter for a given attribute. This returns a callable
 232             taking a single argument.
 233         """
 234         if attr not in self._options:
 235             raise KeyError("No such option: %s" % attr)
 236 
 237         def setter(x):
 238             setattr(self, attr, x)
 239         return setter
 240 
 241     def toggler(self, attr):
 242         """
 243             Generate a toggler for a boolean attribute. This returns a callable
 244             that takes no arguments.
 245         """
 246         if attr not in self._options:
 247             raise KeyError("No such option: %s" % attr)
 248         o = self._options[attr]
 249         if o.typespec != bool:
 250             raise ValueError("Toggler can only be used with boolean options")
 251 
 252         def toggle():
 253             setattr(self, attr, not getattr(self, attr))
 254         return toggle
 255 
 256     def default(self, option: str) -> typing.Any:
 257         return self._options[option].default

 259     def has_changed(self, option):
 260         """
 261             Has the option changed from the default?
 262         """
 263         return self._options[option].has_changed()

 265     def merge(self, opts):
 266         """
 267             Merge a dict of options into this object. Options that have None
 268             value are ignored. Lists and tuples are appended to the current
 269             option value.
 270         """
 271         toset = {}
 272         for k, v in opts.items():
 273             if v is not None:
 274                 if isinstance(v, (list, tuple)):
 275                     toset[k] = getattr(self, k) + v
 276                 else:
 277                     toset[k] = v
 278         self.update(**toset)

 280     def __repr__(self):
 281         options = pprint.pformat(self._options, indent=4).strip(" {}")
 282         if "\n" in options:
 283             options = "\n    " + options + "\n"
 284         return "{mod}.{cls}({{{options}}})".format(
 285             mod=type(self).__module__,
 286             cls=type(self).__name__,
 287             options=options
 288         )

 290     def set(self, *spec, defer=False):
 291         """
 292             Takes a list of set specification in standard form (option=value).
 293             Options that are known are updated immediately. If defer is true,
 294             options that are not known are deferred, and will be set once they
 295             are added.
 296         """
 297         vals = {}
 298         unknown = {}
 299         for i in spec:
 300             parts = i.split("=", maxsplit=1)
 301             if len(parts) == 1:
 302                 optname, optval = parts[0], None
 303             else:
 304                 optname, optval = parts[0], parts[1]
 305             if optname in self._options:
 306                 vals[optname] = self.parse_setval(self._options[optname], optval)
 307             else:
 308                 unknown[optname] = optval
 309         if defer:
 310             self.deferred.update(unknown)
 311         elif unknown:
 312             raise exceptions.OptionsError("Unknown options: %s" % ", ".join(unknown.keys()))
 313         self.update(**vals)
 314 
 315     def process_deferred(self):
 316         """
 317             Processes options that were deferred in previous calls to set, and
 318             have since been added.
 319         """
 320         update = {}
 321         for optname, optval in self.deferred.items():
 322             if optname in self._options:
 323                 optval = self.parse_setval(self._options[optname], optval)
 324                 update[optname] = optval
 325         self.update(**update)
 326         for k in update.keys():
 327             del self.deferred[k]

 329     def parse_setval(self, o: _Option, optstr: typing.Optional[str]) -> typing.Any:
 330         """
 331             Convert a string to a value appropriate for the option type.
 332         """
 333         if o.typespec in (str, typing.Optional[str]):
 334             return optstr
 335         elif o.typespec in (int, typing.Optional[int]):
 336             if optstr:
 337                 try:
 338                     return int(optstr)
 339                 except ValueError:
 340                     raise exceptions.OptionsError("Not an integer: %s" % optstr)
 341             elif o.typespec == int:
 342                 raise exceptions.OptionsError("Option is required: %s" % o.name)
 343             else:
 344                 return None
 345         elif o.typespec == bool:
 346             if optstr == "toggle":
 347                 return not o.current()
 348             if not optstr or optstr == "true":
 349                 return True
 350             elif optstr == "false":
 351                 return False
 352             else:
 353                 raise exceptions.OptionsError(
 354                     "Boolean must be \"true\", \"false\", or have the value " "omitted (a synonym for \"true\")."
 355                 )
 356         elif o.typespec == typing.Sequence[str]:
 357             if not optstr:
 358                 return []
 359             else:
 360                 return getattr(self, o.name) + [optstr]
 361         raise NotImplementedError("Unsupported option type: %s", o.typespec)
 362 
 363     def make_parser(self, parser, optname, metavar=None, short=None):
 364         """
 365             Auto-Create a command-line parser entry for a named option. If the
 366             option does not exist, it is ignored.
 367         """
 368         if optname not in self._options:
 369             return
 370 
 371         o = self._options[optname]
 372 
 373         def mkf(l, s):
 374             l = l.replace("_", "-")
 375             f = ["--%s" % l]
 376             if s:
 377                 f.append("-" + s)
 378             return f
 379 
 380         flags = mkf(optname, short)
 381 
 382         if o.typespec == bool:
 383             g = parser.add_mutually_exclusive_group(required=False)
 384             onf = mkf(optname, None)
 385             offf = mkf("no-" + optname, None)
 386             # The short option for a bool goes to whatever is NOT the default
 387             if short:
 388                 if o.default:
 389                     offf = mkf("no-" + optname, short)
 390                 else:
 391                     onf = mkf(optname, short)
 392             g.add_argument(
 393                 *offf,
 394                 action="store_false",
 395                 dest=optname,
 396             )
 397             g.add_argument(
 398                 *onf,
 399                 action="store_true",
 400                 dest=optname,
 401                 help=o.help
 402             )
 403             parser.set_defaults(**{optname: None})
 404         elif o.typespec in (int, typing.Optional[int]):
 405             parser.add_argument(
 406                 *flags,
 407                 action="store",
 408                 type=int,
 409                 dest=optname,
 410                 help=o.help,
 411                 metavar=metavar,
 412             )
 413         elif o.typespec in (str, typing.Optional[str]):
 414             parser.add_argument(
 415                 *flags,
 416                 action="store",
 417                 type=str,
 418                 dest=optname,
 419                 help=o.help,
 420                 metavar=metavar,
 421                 choices=o.choices
 422             )
 423         elif o.typespec == typing.Sequence[str]:
 424             parser.add_argument(
 425                 *flags,
 426                 action="append",
 427                 type=str,
 428                 dest=optname,
 429                 help=o.help + " May be passed multiple times.",
 430                 metavar=metavar,
 431                 choices=o.choices,
 432             )
 433         else:
 434             raise ValueError("Unsupported option type: %s", o.typespec)

 437 def dump_defaults(opts):
 438     """
 439         Dumps an annotated file with all options.
 440     """
 441     # Sort data
 442     s = ruamel.yaml.comments.CommentedMap()
 443     for k in sorted(opts.keys()):
 444         o = opts._options[k]
 445         s[k] = o.default
 446         txt = o.help.strip()
 447 
 448         if o.choices:
 449             txt += " Valid values are %s." % ", ".join(repr(c) for c in o.choices)
 450         else:
 451             t = typecheck.typespec_to_str(o.typespec)
 452             txt += " Type %s." % t
 453 
 454         txt = "\n".join(textwrap.wrap(txt))
 455         s.yaml_set_comment_before_after_key(k, before="\n" + txt)
 456     return ruamel.yaml.round_trip_dump(s)

 459 def dump_dicts(opts, keys: typing.List[str]=None):
 460     """
 461         Dumps the options into a list of dict object.
 462 
 463         Return: A list like: { "anticache": { type: "bool", default: false, value: true, help: "help text"} }
 464     """
 465     options_dict = {}
 466     keys = keys if keys else opts.keys()
 467     for k in sorted(keys):
 468         o = opts._options[k]
 469         t = typecheck.typespec_to_str(o.typespec)
 470         option = {
 471             'type': t,
 472             'default': o.default,
 473             'value': o.current(),
 474             'help': o.help,
 475             'choices': o.choices
 476         }
 477         options_dict[k] = option
 478     return options_dict

 481 def parse(text):
 482     if not text:
 483         return {}
 484     try:
 485         data = ruamel.yaml.load(text, ruamel.yaml.RoundTripLoader)
 486     except ruamel.yaml.error.YAMLError as v:
 487         if hasattr(v, "problem_mark"):
 488             snip = v.problem_mark.get_snippet()
 489             raise exceptions.OptionsError(
 490                 "Config error at line %s:\n%s\n%s" %
 491                 (v.problem_mark.line + 1, snip, v.problem)
 492             )
 493         else:
 494             raise exceptions.OptionsError("Could not parse options.")
 495     if isinstance(data, str):
 496         raise exceptions.OptionsError("Config error - no keys found.")
 497     elif data is None:
 498         return {}
 499     return data

 511 def load_paths(opts: OptManager, *paths: str) -> None:
 512     """
 513         Load paths in order. Each path takes precedence over the previous
 514         path. Paths that don't exist are ignored, errors raise an
 515         OptionsError.
 516     """
 517     for p in paths:
 518         p = os.path.expanduser(p)
 519         if os.path.exists(p) and os.path.isfile(p):
 520             with open(p, "rt", encoding="utf8") as f:
 521                 try:
 522                     txt = f.read()
 523                 except UnicodeDecodeError as e:
 524                     raise exceptions.OptionsError(
 525                         "Error reading %s: %s" % (p, e)
 526                     )
 527             try:
 528                 load(opts, txt)
 529             except exceptions.OptionsError as e:
 530                 raise exceptions.OptionsError(
 531                     "Error reading %s: %s" % (p, e)
 532                 )

 535 def serialize(opts: OptManager, text: str, defaults: bool = False) -> str:
 536     """
 537         Performs a round-trip serialization. If text is not None, it is
 538         treated as a previous serialization that should be modified
 539         in-place.
 540 
 541         - If "defaults" is False, only options with non-default values are
 542             serialized. Default values in text are preserved.
 543         - Unknown options in text are removed.
 544         - Raises OptionsError if text is invalid.
 545     """
 546     data = parse(text)
 547     for k in opts.keys():
 548         if defaults or opts.has_changed(k):
 549             data[k] = getattr(opts, k)
 550     for k in list(data.keys()):
 551         if k not in opts._options:
 552             del data[k]
 553     ret = ruamel.yaml.round_trip_dump(data)
 554     assert ret
 555     return ret

 558 def save(opts: OptManager, path: str, defaults: bool =False) -> None:
 559     """
 560         Save to path. If the destination file exists, modify it in-place.
 561 
 562         Raises OptionsError if the existing data is corrupt.
 563     """
 564     path = os.path.expanduser(path)
 565     if os.path.exists(path) and os.path.isfile(path):
 566         with open(path, "rt", encoding="utf8") as f:
 567             try:
 568                 data = f.read()
 569             except UnicodeDecodeError as e:
 570                 raise exceptions.OptionsError(
 571                     "Error trying to modify %s: %s" % (path, e)
 572                 )
 573     else:
 574         data = ""
 575     data = serialize(opts, data, defaults)
 576     with open(path, "wt", encoding="utf8") as f:
 577         f.write(data)
