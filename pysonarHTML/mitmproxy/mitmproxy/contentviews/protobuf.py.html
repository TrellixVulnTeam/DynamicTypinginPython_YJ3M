   8 def write_buf(out, field_tag, body, indent_level):
   9     if body is not None:
  10         out.write("{: <{level}}{}: {}\n".format('', field_tag, body if isinstance(body, int) else str(body, 'utf-8'),
  11                                                 level=indent_level))
  12     elif field_tag is not None:
  13         out.write(' ' * indent_level + str(field_tag) + " {\n")
  14     else:
  15         out.write(' ' * indent_level + "}\n")

  18 def format_pbuf(raw):
  19     out = io.StringIO()
  20     stack = []
  21 
  22     try:
  23         buf = google_protobuf.GoogleProtobuf(KaitaiStream(io.BytesIO(raw)))
  24     except:
  25         return False
  26     stack.extend([(pair, 0) for pair in buf.pairs[::-1]])
  27 
  28     while len(stack):
  29         pair, indent_level = stack.pop()
  30 
  31         if pair.wire_type == pair.WireTypes.group_start:
  32             body = None
  33         elif pair.wire_type == pair.WireTypes.group_end:
  34             body = None
  35             pair._m_field_tag = None
  36         elif pair.wire_type == pair.WireTypes.len_delimited:
  37             body = pair.value.body
  38         elif pair.wire_type == pair.WireTypes.varint:
  39             body = pair.value.value
  40         else:
  41             body = pair.value
  42 
  43         try:
  44             next_buf = google_protobuf.GoogleProtobuf(KaitaiStream(io.BytesIO(body)))
  45             stack.extend([(pair, indent_level + 2) for pair in next_buf.pairs[::-1]])
  46             write_buf(out, pair.field_tag, None, indent_level)
  47         except:
  48             write_buf(out, pair.field_tag, body, indent_level)
  49 
  50         if stack:
  51             prev_level = stack[-1][1]
  52         else:
  53             prev_level = 0
  54 
  55         if prev_level < indent_level:
  56             levels = int((indent_level - prev_level) / 2)
  57             for i in range(1, levels + 1):
  58                 write_buf(out, None, None, indent_level - i * 2)
  59 
  60     return out.getvalue()
