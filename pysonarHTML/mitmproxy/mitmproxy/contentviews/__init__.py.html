   1 """
   2 Mitmproxy Content Views
   3 =======================
   4 
   5 mitmproxy includes a set of content views which can be used to
   6 format/decode/highlight data. While they are currently used for HTTP message
   7 bodies only, the may be used in other contexts in the future, e.g. to decode
   8 protobuf messages sent as WebSocket frames.
   9 
  10 Thus, the View API is very minimalistic. The only arguments are `data` and
  11 `**metadata`, where `data` is the actual content (as bytes). The contents on
  12 metadata depend on the protocol in use. For HTTP, the message headers are
  13 passed as the ``headers`` keyword argument. For HTTP requests, the query
  14 parameters are passed as the ``query`` keyword argument.
  15 """
  16 import traceback
  17 from typing import Dict, Optional  # noqa
  18 from typing import List  # noqa

  33 def get(name: str) -> Optional[View]:
  34     for i in views:
  35         if i.name.lower() == name.lower():
  36             return i
  37     return None

  40 def add(view: View) -> None:
  41     # TODO: auto-select a different name (append an integer?)
  42     for i in views:
  43         if i.name == view.name:
  44             raise exceptions.ContentViewException("Duplicate view: " + view.name)
  45 
  46     views.append(view)
  47 
  48     for ct in view.content_types:
  49         l = content_types_map.setdefault(ct, [])
  50         l.append(view)
  51 
  52 
  53 def remove(view: View) -> None:
  54     for ct in view.content_types:
  55         l = content_types_map.setdefault(ct, [])
  56         l.remove(view)
  57 
  58         if not len(l):
  59             del content_types_map[ct]
  60 
  61     views.remove(view)

  64 def safe_to_print(lines, encoding="utf8"):
  65     """
  66     Wraps a content generator so that each text portion is a *safe to print* unicode string.
  67     """
  68     for line in lines:
  69         clean_line = []
  70         for (style, text) in line:
  71             if isinstance(text, bytes):
  72                 text = text.decode(encoding, "replace")
  73             text = strutils.escape_control_characters(text)
  74             clean_line.append((style, text))
  75         yield clean_line

  78 def get_message_content_view(viewname, message, flow):
  79     """
  80     Like get_content_view, but also handles message encoding.
  81     """
  82     viewmode = get(viewname)
  83     if not viewmode:
  84         viewmode = get("auto")
  85     try:
  86         content = message.content
  87     except ValueError:
  88         content = message.raw_content
  89         enc = "[cannot decode]"
  90     else:
  91         if isinstance(message, http.Message) and content != message.raw_content:
  92             enc = "[decoded {}]".format(
  93                 message.headers.get("content-encoding")
  94             )
  95         else:
  96             enc = None
  97 
  98     if content is None:
  99         return "", iter([[("error", "content missing")]]), None
 100 
 101     metadata = {}
 102     if isinstance(message, http.Request):
 103         metadata["query"] = message.query
 104     if isinstance(message, http.Message):
 105         metadata["headers"] = message.headers
 106     metadata["message"] = message
 107     metadata["flow"] = flow
 108 
 109     description, lines, error = get_content_view(
 110         viewmode, content, **metadata
 111     )
 112 
 113     if enc:
 114         description = "{} {}".format(enc, description)
 115 
 116     return description, lines, error

 119 def get_tcp_content_view(viewname: str, data: bytes):
 120     viewmode = get(viewname)
 121     if not viewmode:
 122         viewmode = get("auto")
 123 
 124     # https://github.com/mitmproxy/mitmproxy/pull/3970#issuecomment-623024447
 125     assert viewmode
 126 
 127     description, lines, error = get_content_view(viewmode, data)
 128 
 129     return description, lines, error

 132 def get_content_view(viewmode: View, data: bytes, **metadata):
 133     """
 134         Args:
 135             viewmode: the view to use.
 136             data, **metadata: arguments passed to View instance.
 137 
 138         Returns:
 139             A (description, content generator, error) tuple.
 140             If the content view raised an exception generating the view,
 141             the exception is returned in error and the flow is formatted in raw mode.
 142             In contrast to calling the views directly, text is always safe-to-print unicode.
 143     """
 144     try:
 145         ret = viewmode(data, **metadata)
 146         if ret is None:
 147             ret = "Couldn't parse: falling back to Raw", get("Raw")(data, **metadata)[1]
 148         desc, content = ret
 149         error = None
 150     # Third-party viewers can fail in unexpected ways...
 151     except Exception:
 152         desc = "Couldn't parse: falling back to Raw"
 153         raw = get("Raw")
 154         assert raw
 155         content = raw(data, **metadata)[1]
 156         error = "{} Content viewer failed: \n{}".format(
 157             getattr(viewmode, "name"),
 158             traceback.format_exc()
 159         )
 160 
 161     return desc, safe_to_print(content), error
