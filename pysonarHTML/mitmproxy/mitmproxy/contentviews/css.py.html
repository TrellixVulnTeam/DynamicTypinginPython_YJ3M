   1 import re
   2 import time

   4 from mitmproxy.contentviews import base
   5 from mitmproxy.utils import strutils
   6 
   7 """
   8 A custom CSS prettifier. Compared to other prettifiers, its main features are:
   9 
  10 - Implemented in pure Python.
  11 - Modifies whitespace only.
  12 - Works with any input.
  13 - Considerably faster than e.g. cssutils.
  14 """
  15 
  16 CSS_SPECIAL_AREAS = (
  17     "'" + strutils.SINGLELINE_CONTENT + strutils.NO_ESCAPE + "'",
  18     '"' + strutils.SINGLELINE_CONTENT + strutils.NO_ESCAPE + '"',
  19     r"/\*" + strutils.MULTILINE_CONTENT + r"\*/",
  20     "//" + strutils.SINGLELINE_CONTENT + "$"
  21 )
  22 CSS_SPECIAL_CHARS = "{};:"
  23 
  24 
  25 def beautify(data: str, indent: str = "    "):
  26     """Beautify a string containing CSS code"""
  27     data = strutils.escape_special_areas(
  28         data.strip(),
  29         CSS_SPECIAL_AREAS,
  30         CSS_SPECIAL_CHARS,
  31     )
  32 
  33     # Add newlines
  34     data = re.sub(r"\s*;\s*", ";\n", data)
  35     data = re.sub(r"\s*{\s*", " {\n", data)
  36     data = re.sub(r"\s*}\s*", "\n}\n\n", data)
  37 
  38     # Fix incorrect ":" placement
  39     data = re.sub(r"\s*:\s*(?=[^{]+})", ": ", data)
  40     # Fix no space after ","
  41     data = re.sub(r"\s*,\s*", ", ", data)
  42 
  43     # indent
  44     data = re.sub("\n[ \t]+", "\n", data)
  45     data = re.sub("\n(?![}\n])(?=[^{]*})", "\n" + indent, data)
  46 
  47     data = strutils.unescape_special_areas(data)
  48     return data.rstrip("\n") + "\n"
  49 
  50 
  51 class ViewCSS(base.View):
  52     name = "CSS"
  53     content_types = [
  54         "text/css"
  55     ]
  56 
  57     def __call__(self, data, **metadata):
  58         data = data.decode("utf8", "surrogateescape")
  59         beautified = beautify(data)
  60         return "CSS", base.format_text(beautified)
