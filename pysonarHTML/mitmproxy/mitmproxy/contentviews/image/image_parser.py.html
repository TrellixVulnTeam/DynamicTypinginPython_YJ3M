  14 def parse_png(data: bytes) -> Metadata:
  15     img = png.Png(KaitaiStream(io.BytesIO(data)))
  16     parts = [
  17         ('Format', 'Portable network graphics'),
  18         ('Size', "{0} x {1} px".format(img.ihdr.width, img.ihdr.height))
  19     ]
  20     for chunk in img.chunks:
  21         if chunk.type == 'gAMA':
  22             parts.append(('gamma', str(chunk.body.gamma_int / 100000)))
  23         elif chunk.type == 'pHYs':
  24             aspectx = chunk.body.pixels_per_unit_x
  25             aspecty = chunk.body.pixels_per_unit_y
  26             parts.append(('aspect', "{0} x {1}".format(aspectx, aspecty)))
  27         elif chunk.type == 'tEXt':
  28             parts.append((chunk.body.keyword, chunk.body.text))
  29         elif chunk.type == 'iTXt':
  30             parts.append((chunk.body.keyword, chunk.body.text))
  31         elif chunk.type == 'zTXt':
  32             parts.append((chunk.body.keyword, chunk.body.text_datastream.decode('iso8859-1')))
  33     return parts

  36 def parse_gif(data: bytes) -> Metadata:
  37     img = gif.Gif(KaitaiStream(io.BytesIO(data)))
  38     descriptor = img.logical_screen_descriptor
  39     parts = [
  40         ('Format', 'Compuserve GIF'),
  41         ('Version', "GIF{}".format(img.hdr.version)),
  42         ('Size', "{} x {} px".format(descriptor.screen_width, descriptor.screen_height)),
  43         ('background', str(descriptor.bg_color_index))
  44     ]
  45     ext_blocks = []
  46     for block in img.blocks:
  47         if block.block_type.name == 'extension':
  48             ext_blocks.append(block)
  49     comment_blocks = []
  50     for block in ext_blocks:
  51         if block.body.label._name_ == 'comment':
  52             comment_blocks.append(block)
  53     for block in comment_blocks:
  54         entries = block.body.body.entries
  55         for entry in entries:
  56             comment = entry.bytes
  57             if comment != b'':
  58                 parts.append(('comment', str(comment)))
  59     return parts

  62 def parse_jpeg(data: bytes) -> Metadata:
  63     img = jpeg.Jpeg(KaitaiStream(io.BytesIO(data)))
  64     parts = [
  65         ('Format', 'JPEG (ISO 10918)')
  66     ]
  67     for segment in img.segments:
  68         if segment.marker._name_ == 'sof0':
  69             parts.append(('Size', "{0} x {1} px".format(segment.data.image_width, segment.data.image_height)))
  70         if segment.marker._name_ == 'app0':
  71             parts.append(('jfif_version', "({0}, {1})".format(segment.data.version_major, segment.data.version_minor)))
  72             parts.append(('jfif_density', "({0}, {1})".format(segment.data.density_x, segment.data.density_y)))
  73             parts.append(('jfif_unit', str(segment.data.density_units._value_)))
  74         if segment.marker._name_ == 'com':
  75             parts.append(('comment', str(segment.data)))
  76         if segment.marker._name_ == 'app1':
  77             if hasattr(segment.data, 'body'):
  78                 for field in segment.data.body.data.body.ifd0.fields:
  79                     if field.data is not None:
  80                         parts.append((field.tag._name_, field.data.decode('UTF-8').strip('\x00')))
  81     return parts

  84 def parse_ico(data: bytes) -> Metadata:
  85     img = ico.Ico(KaitaiStream(io.BytesIO(data)))
  86     parts = [
  87         ('Format', 'ICO'),
  88         ('Number of images', str(img.num_images)),
  89     ]
  90 
  91     for i, image in enumerate(img.images):
  92         parts.append(
  93             (
  94                 'Image {}'.format(i + 1), "Size: {} x {}\n"
  95                                           "{: >18}Bits per pixel: {}\n"
  96                                           "{: >18}PNG: {}".format(256 if not image.width else image.width,
  97                                                                   256 if not image.height else image.height,
  98                                                                   '', image.bpp,
  99                                                                   '', image.is_png)
 100             )
 101         )
 102 
 103     return parts
