   7 DELIMITERS = '{};\n'
   8 SPECIAL_AREAS = (
   9     r"(?<=[^\w\s)])\s*/(?:[^\n/]|(?<!\\)(?:\\\\)*\\/)+?/(?=[gimsuy]{0,6}\s*(?:[;,).\n]|$))",
  10     r"'" + strutils.MULTILINE_CONTENT_LINE_CONTINUATION + strutils.NO_ESCAPE + "'",
  11     r'"' + strutils.MULTILINE_CONTENT_LINE_CONTINUATION + strutils.NO_ESCAPE + '"',
  12     r'`' + strutils.MULTILINE_CONTENT + strutils.NO_ESCAPE + '`',
  13     r"/\*" + strutils.MULTILINE_CONTENT + r"\*/",
  14     r"//" + strutils.SINGLELINE_CONTENT + "$",
  15     r"for\(" + strutils.SINGLELINE_CONTENT + r"\)",
  16 )

  19 def beautify(data):
  20     data = strutils.escape_special_areas(
  21         data,
  22         SPECIAL_AREAS,
  23         DELIMITERS
  24     )
  25 
  26     data = re.sub(r"\s*{\s*(?!};)", " {\n", data)
  27     data = re.sub(r"\s*;\s*", ";\n", data)
  28     data = re.sub(r"(?<!{)\s*}(;)?\s*", r"\n}\1\n", data)
  29 
  30     beautified = io.StringIO()
  31     indent_level = 0
  32 
  33     for line in data.splitlines(True):
  34         if line.endswith("{\n"):
  35             beautified.write(" " * 2 * indent_level + line)
  36             indent_level += 1
  37         elif line.startswith("}"):
  38             indent_level -= 1
  39             beautified.write(" " * 2 * indent_level + line)
  40         else:
  41             beautified.write(" " * 2 * indent_level + line)
  42 
  43     data = strutils.unescape_special_areas(beautified.getvalue())
  44     return data
