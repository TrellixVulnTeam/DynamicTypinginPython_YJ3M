  21 class WebSocketLayer(base.Layer):
  22     """
  23         WebSocket layer to intercept, modify, and forward WebSocket messages.
  24 
  25         Only version 13 is supported (as specified in RFC6455).
  26         Only HTTP/1.1-initiated connections are supported.
  27 
  28         The client starts by sending an Upgrade-request.
  29         In order to determine the handshake and negotiate the correct protocol
  30         and extensions, the Upgrade-request is forwarded to the server.
  31         The response from the server is then parsed and negotiated settings are extracted.
  32         Finally the handshake is completed by forwarding the server-response to the client.
  33         After that, only WebSocket frames are exchanged.
  34 
  35         PING/PONG frames pass through and must be answered by the other endpoint.
  36 
  37         CLOSE frames are forwarded before this WebSocketLayer terminates.
  38 
  39         This layer is transparent to any negotiated extensions.
  40         This layer is transparent to any negotiated subprotocols.
  41         Only raw frames are forwarded to the other endpoint.
  42 
  43         WebSocket messages are stored in a WebSocketFlow.
  44     """
  45 
  46     def __init__(self, ctx, handshake_flow):
  47         super().__init__(ctx)
  48         self.handshake_flow = handshake_flow
  49         self.flow: WebSocketFlow = None
  50 
  51         self.client_frame_buffer = []
  52         self.server_frame_buffer = []
  53 
  54         self.connections: dict[object, WSConnection] = {}
  55 
  56         client_extensions = []
  57         server_extensions = []
  58         if 'Sec-WebSocket-Extensions' in handshake_flow.response.headers:
  59             if PerMessageDeflate.name in handshake_flow.response.headers['Sec-WebSocket-Extensions']:
  60                 client_extensions = [PerMessageDeflate()]
  61                 server_extensions = [PerMessageDeflate()]
  62         self.connections[self.client_conn] = WSConnection(ConnectionType.SERVER)
  63         self.connections[self.server_conn] = WSConnection(ConnectionType.CLIENT)
  64 
  65         if client_extensions:
  66             client_extensions[0].finalize(handshake_flow.response.headers['Sec-WebSocket-Extensions'])
  67         if server_extensions:
  68             server_extensions[0].finalize(handshake_flow.response.headers['Sec-WebSocket-Extensions'])
  69 
  70         request = Request(extensions=client_extensions, host=handshake_flow.request.host, target=handshake_flow.request.path)
  71         data = self.connections[self.server_conn].send(request)
  72         self.connections[self.client_conn].receive_data(data)
  73 
  74         event = next(self.connections[self.client_conn].events())
  75         assert isinstance(event, events.Request)
  76 
  77         data = self.connections[self.client_conn].send(AcceptConnection(extensions=server_extensions))
  78         self.connections[self.server_conn].receive_data(data)
  79         assert isinstance(next(self.connections[self.server_conn].events()), events.AcceptConnection)
  80 
  81     def _handle_event(self, event, source_conn, other_conn, is_server):
  82         if isinstance(event, events.Message):
  83             return self._handle_message(event, source_conn, other_conn, is_server)
  84         elif isinstance(event, events.Ping):
  85             return self._handle_ping(event, source_conn, other_conn, is_server)
  86         elif isinstance(event, events.Pong):
  87             return self._handle_pong(event, source_conn, other_conn, is_server)
  88         elif isinstance(event, events.CloseConnection):
  89             return self._handle_close_connection(event, source_conn, other_conn, is_server)
  90 
  91         # fail-safe for unhandled events
  92         return True  # pragma: no cover
  93 
  94     def _handle_message(self, event, source_conn, other_conn, is_server):
  95         fb = self.server_frame_buffer if is_server else self.client_frame_buffer
  96         fb.append(event.data)
  97 
  98         if event.message_finished:
  99             original_chunk_sizes = [len(f) for f in fb]
 100 
 101             if isinstance(event, events.TextMessage):
 102                 message_type = wsproto.frame_protocol.Opcode.TEXT
 103                 payload = ''.join(fb)
 104             else:
 105                 message_type = wsproto.frame_protocol.Opcode.BINARY
 106                 payload = b''.join(fb)
 107 
 108             fb.clear()
 109 
 110             websocket_message = WebSocketMessage(message_type, not is_server, payload)
 111             length = len(websocket_message.content)
 112             self.flow.messages.append(websocket_message)
 113             self.channel.ask("websocket_message", self.flow)
 114 
 115             if not self.flow.stream and not websocket_message.killed:
 116                 def get_chunk(payload):
 117                     if len(payload) == length:
 118                         # message has the same length, we can reuse the same sizes
 119                         pos = 0
 120                         for s in original_chunk_sizes:
 121                             yield (payload[pos:pos + s], True if pos + s == length else False)
 122                             pos += s
 123                     else:
 124                         # just re-chunk everything into 4kB frames
 125                         # header len = 4 bytes without masking key and 8 bytes with masking key
 126                         chunk_size = 4092 if is_server else 4088
 127                         chunks = range(0, len(payload), chunk_size)
 128                         for i in chunks:
 129                             yield (payload[i:i + chunk_size], True if i + chunk_size >= len(payload) else False)
 130 
 131                 for chunk, final in get_chunk(websocket_message.content):
 132                     data = self.connections[other_conn].send(Message(data=chunk, message_finished=final))
 133                     other_conn.send(data)
 134 
 135         if self.flow.stream:
 136             data = self.connections[other_conn].send(Message(data=event.data, message_finished=event.message_finished))
 137             other_conn.send(data)
 138         return True
 139 
 140     def _handle_ping(self, event, source_conn, other_conn, is_server):
 141         # Use event.response to create the approprate Pong response
 142         data = self.connections[other_conn].send(Ping())
 143         other_conn.send(data)
 144         data = self.connections[source_conn].send(event.response())
 145         source_conn.send(data)
 146         self.log(
 147             "Ping Received from {}".format("server" if is_server else "client"),
 148             "info",
 149             [strutils.bytes_to_escaped_str(bytes(event.payload))]
 150         )
 151         return True
 152 
 153     def _handle_pong(self, event, source_conn, other_conn, is_server):
 154         self.log(
 155             "Pong Received from {}".format("server" if is_server else "client"),
 156             "info",
 157             [strutils.bytes_to_escaped_str(bytes(event.payload))]
 158         )
 159         return True
 160 
 161     def _handle_close_connection(self, event, source_conn, other_conn, is_server):
 162         self.flow.close_sender = "server" if is_server else "client"
 163         self.flow.close_code = event.code
 164         self.flow.close_reason = event.reason
 165 
 166         data = self.connections[other_conn].send(CloseConnection(code=event.code, reason=event.reason))
 167         other_conn.send(data)
 168         data = self.connections[source_conn].send(event.response())
 169         source_conn.send(data)
 170 
 171         return False

 173     def _inject_messages(self, endpoint, message_queue):
 174         while True:
 175             try:
 176                 payload = message_queue.get_nowait()
 177                 data = self.connections[endpoint].send(Message(data=payload, message_finished=True))
 178                 endpoint.send(data)
 179             except queue.Empty:
 180                 break

 182     def __call__(self):
 183         self.flow = WebSocketFlow(self.client_conn, self.server_conn, self.handshake_flow)
 184         self.flow.metadata['websocket_handshake'] = self.handshake_flow.id
 185         self.handshake_flow.metadata['websocket_flow'] = self.flow.id
 186         self.channel.ask("websocket_start", self.flow)
 187 
 188         conns = [c.connection for c in self.connections.keys()]
 189         close_received = False
 190 
 191         try:
 192             while not self.channel.should_exit.is_set():
 193                 self._inject_messages(self.client_conn, self.flow._inject_messages_client)
 194                 self._inject_messages(self.server_conn, self.flow._inject_messages_server)
 195 
 196                 r = tcp.ssl_read_select(conns, 0.1)
 197                 for conn in r:
 198                     source_conn = self.client_conn if conn == self.client_conn.connection else self.server_conn
 199                     other_conn = self.server_conn if conn == self.client_conn.connection else self.client_conn
 200                     is_server = (source_conn == self.server_conn)
 201 
 202                     frame = websockets.Frame.from_file(source_conn.rfile)
 203                     data = self.connections[source_conn].receive_data(bytes(frame))
 204                     source_conn.send(data)
 205 
 206                     if close_received:
 207                         return
 208 
 209                     for event in self.connections[source_conn].events():
 210                         if not self._handle_event(event, source_conn, other_conn, is_server):
 211                             if not close_received:
 212                                 close_received = True
 213         except (socket.error, exceptions.TcpException, SSL.Error) as e:
 214             s = 'server' if is_server else 'client'
 215             self.flow.error = flow.Error("WebSocket connection closed unexpectedly by {}: {}".format(s, repr(e)))
 216             self.channel.tell("websocket_error", self.flow)
 217         finally:
 218             self.flow.ended = True
 219             self.channel.tell("websocket_end", self.flow)
