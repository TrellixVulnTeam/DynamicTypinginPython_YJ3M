 210 class TlsLayer(base.Layer):
 211     """
 212     The TLS layer implements transparent TLS connections.
 213 
 214     It exposes the following API to child layers:
 215 
 216         - :py:meth:`set_server_tls` to modify TLS settings for the server connection.
 217         - :py:attr:`server_tls`, :py:attr:`server_sni` as read-only attributes describing the current TLS settings for
 218           the server connection.
 219     """
 220 
 221     def __init__(self, ctx, client_tls, server_tls, custom_server_sni=None):
 222         super().__init__(ctx)
 223         self._client_tls = client_tls
 224         self._server_tls = server_tls
 225 
 226         self._custom_server_sni = custom_server_sni
 227         self._client_hello: Optional[net_tls.ClientHello] = None
 228 
 229     def __call__(self):
 230         """
 231         The strategy for establishing TLS is as follows:
 232             First, we determine whether we need the server cert to establish ssl with the client.
 233             If so, we first connect to the server and then to the client.
 234             If not, we only connect to the client and do the server handshake lazily.
 235 
 236         An additional complexity is that we need to mirror SNI and ALPN from the client when connecting to the server.
 237         We manually peek into the connection and parse the ClientHello message to obtain these values.
 238         """
 239         if self._client_tls:
 240             # Peek into the connection, read the initial client hello and parse it to obtain SNI and ALPN values.
 241             try:
 242                 self._client_hello = net_tls.ClientHello.from_file(self.client_conn.rfile)
 243             except exceptions.TlsProtocolException as e:
 244                 self.log("Cannot parse Client Hello: %s" % repr(e), "error")
 245                 # Without knowning the ClientHello we cannot proceed in this connection.
 246                 return
 247 
 248         # Do we need to do a server handshake now?
 249         # There are two reasons why we would want to establish TLS with the server now:
 250         #  1. If we already have an existing server connection and server_tls is True,
 251         #     we need to establish TLS now because .connect() will not be called anymore.
 252         #  2. We may need information from the server connection for the client handshake.
 253         #
 254         # A couple of factors influence (2):
 255         #  2.1 There actually is (or will be) a TLS-enabled upstream connection
 256         #  2.2 An upstream connection is not wanted by the user if --no-upstream-cert is passed.
 257         #  2.3 An upstream connection is implied by add_upstream_certs_to_client_chain
 258         #  2.4 The client wants to negotiate an alternative protocol in its handshake, we need to find out
 259         #      what is supported by the server
 260         #  2.5 The client did not sent a SNI value, we don't know the certificate subject.
 261         client_tls_requires_server_connection = (
 262             self._server_tls and
 263             self.config.options.upstream_cert and
 264             (
 265                 self.config.options.add_upstream_certs_to_client_chain or
 266                 self._client_tls and (
 267                     self._client_hello.alpn_protocols or
 268                     not self._client_hello.sni
 269                 )
 270             )
 271         )
 272         establish_server_tls_now = (
 273             (self.server_conn.connected() and self._server_tls) or
 274             client_tls_requires_server_connection
 275         )
 276 
 277         if self._client_tls and establish_server_tls_now:
 278             self._establish_tls_with_client_and_server()
 279         elif self._client_tls:
 280             self._establish_tls_with_client()
 281         elif establish_server_tls_now:
 282             self._establish_tls_with_server()
 283 
 284         layer = self.ctx.next_layer(self)
 285         layer()
 286 
 287     def __repr__(self):  # pragma: no cover
 288         if self._client_tls and self._server_tls:
 289             return "TlsLayer(client and server)"
 290         elif self._client_tls:
 291             return "TlsLayer(client)"
 292         elif self._server_tls:
 293             return "TlsLayer(server)"
 294         else:
 295             return "TlsLayer(inactive)"
 296 
 297     def connect(self):
 298         if not self.server_conn.connected():
 299             self.ctx.connect()
 300         if self._server_tls and not self.server_conn.tls_established:
 301             self._establish_tls_with_server()
 302 
 303     def set_server_tls(self, server_tls: bool, sni: Union[str, None, bool] = None) -> None:
 304         """
 305         Set the TLS settings for the next server connection that will be established.
 306         This function will not alter an existing connection.
 307 
 308         Args:
 309             server_tls: Shall we establish TLS with the server?
 310             sni: ``str`` for a custom SNI value,
 311                 ``None`` for the client SNI value,
 312                 ``False`` if no SNI value should be sent.
 313         """
 314         self._server_tls = server_tls
 315         self._custom_server_sni = sni
 316 
 317     @property
 318     def server_tls(self):
 319         """
 320         ``True``, if the next server connection that will be established should be upgraded to TLS.
 321         """
 322         return self._server_tls
 323 
 324     @property
 325     def server_sni(self) -> Optional[str]:
 326         """
 327         The Server Name Indication we want to send with the next server TLS handshake.
 328         """
 329         if self._custom_server_sni is False:
 330             return None
 331         elif self._custom_server_sni:
 332             return self._custom_server_sni
 333         elif self._client_hello and self._client_hello.sni:
 334             return self._client_hello.sni.decode("idna")
 335         else:
 336             return None
 337 
 338     @property
 339     def alpn_for_client_connection(self):
 340         return self.server_conn.get_alpn_proto_negotiated()
 341 
 342     def __alpn_select_callback(self, conn_, options):
 343         # This gets triggered if we haven't established an upstream connection yet.
 344         default_alpn = b'http/1.1'
 345 
 346         if self.alpn_for_client_connection in options:
 347             choice = bytes(self.alpn_for_client_connection)
 348         elif default_alpn in options:
 349             choice = bytes(default_alpn)
 350         else:
 351             choice = options[0]
 352         self.log("ALPN for client: %s" % choice, "debug")
 353         return choice
 354 
 355     def _establish_tls_with_client_and_server(self):
 356         try:
 357             self.ctx.connect()
 358             self._establish_tls_with_server()
 359         except Exception:
 360             # If establishing TLS with the server fails, we try to establish TLS with the client nonetheless
 361             # to send an error message over TLS.
 362             try:
 363                 self._establish_tls_with_client()
 364             except:
 365                 pass
 366             raise
 367 
 368         self._establish_tls_with_client()
 369 
 370     def _establish_tls_with_client(self):
 371         self.log("Establish TLS with client", "debug")
 372         cert, key, chain_file = self._find_cert()
 373 
 374         if self.config.options.add_upstream_certs_to_client_chain:
 375             extra_certs = self.server_conn.server_certs
 376         else:
 377             extra_certs = None
 378 
 379         try:
 380             tls_method, tls_options = net_tls.VERSION_CHOICES[self.config.options.ssl_version_client]
 381             self.client_conn.convert_to_tls(
 382                 cert, key,
 383                 method=tls_method,
 384                 options=tls_options,
 385                 cipher_list=self.config.options.ciphers_client or DEFAULT_CLIENT_CIPHERS,
 386                 dhparams=self.config.certstore.dhparams,
 387                 chain_file=chain_file,
 388                 alpn_select_callback=self.__alpn_select_callback,
 389                 extra_chain_certs=extra_certs,
 390             )
 391             # Some TLS clients will not fail the handshake,
 392             # but will immediately throw an "unexpected eof" error on the first read.
 393             # The reason for this might be difficult to find, so we try to peek here to see if it
 394             # raises ann error.
 395             self.client_conn.rfile.peek(1)
 396         except exceptions.TlsException as e:
 397             sni_str = self._client_hello.sni and self._client_hello.sni.decode("idna")
 398             raise exceptions.ClientHandshakeException(
 399                 "Cannot establish TLS with client (sni: {sni}): {e}".format(
 400                     sni=sni_str, e=repr(e)
 401                 ),
 402                 sni_str or repr(self.server_conn.address)
 403             )
 404 
 405     def _establish_tls_with_server(self):
 406         self.log("Establish TLS with server", "debug")
 407         try:
 408             alpn = None
 409             if self._client_tls:
 410                 if self._client_hello.alpn_protocols:
 411                     # We only support http/1.1 and h2.
 412                     # If the server only supports spdy (next to http/1.1), it may select that
 413                     # and mitmproxy would enter TCP passthrough mode, which we want to avoid.
 414                     alpn = [
 415                         x for x in self._client_hello.alpn_protocols if
 416                         not (x.startswith(b"h2-") or x.startswith(b"spdy"))
 417                     ]
 418                 if alpn and b"h2" in alpn and not self.config.options.http2:
 419                     alpn.remove(b"h2")
 420 
 421             if self.client_conn.tls_established and self.client_conn.get_alpn_proto_negotiated():
 422                 # If the client has already negotiated an ALP, then force the
 423                 # server to use the same. This can only happen if the host gets
 424                 # changed after the initial connection was established. E.g.:
 425                 #   * the client offers http/1.1 and h2,
 426                 #   * the initial host is only capable of http/1.1,
 427                 #   * then the first server connection negotiates http/1.1,
 428                 #   * but after the server_conn change, the new host offers h2
 429                 #   * which results in garbage because the layers don' match.
 430                 alpn = [self.client_conn.get_alpn_proto_negotiated()]
 431 
 432             # We pass through the list of ciphers send by the client, because some HTTP/2 servers
 433             # will select a non-HTTP/2 compatible cipher from our default list and then hang up
 434             # because it's incompatible with h2. :-)
 435             ciphers_server = self.config.options.ciphers_server
 436             if not ciphers_server and self._client_tls:
 437                 ciphers_server = []
 438                 for id in self._client_hello.cipher_suites:
 439                     if id in CIPHER_ID_NAME_MAP.keys():
 440                         ciphers_server.append(CIPHER_ID_NAME_MAP[id])
 441                 ciphers_server = ':'.join(ciphers_server)
 442 
 443             args = net_tls.client_arguments_from_options(self.config.options)
 444             args["cipher_list"] = ciphers_server
 445             self.server_conn.establish_tls(
 446                 sni=self.server_sni,
 447                 alpn_protos=alpn,
 448                 **args
 449             )
 450             tls_cert_err = self.server_conn.ssl_verification_error
 451             if tls_cert_err is not None:
 452                 self.log(str(tls_cert_err), "warn")
 453                 self.log("Ignoring server verification error, continuing with connection", "warn")
 454         except exceptions.InvalidCertificateException as e:
 455             raise exceptions.InvalidServerCertificate(str(e))
 456         except exceptions.TlsException as e:
 457             raise exceptions.TlsProtocolException(
 458                 "Cannot establish TLS with {host}:{port} (sni: {sni}): {e}".format(
 459                     host=self.server_conn.address[0],
 460                     port=self.server_conn.address[1],
 461                     sni=self.server_sni,
 462                     e=repr(e)
 463                 )
 464             )
 465 
 466         proto = self.alpn_for_client_connection.decode() if self.alpn_for_client_connection else '-'
 467         self.log("ALPN selected by server: {}".format(proto), "debug")
 468 
 469     def _find_cert(self):
 470         """
 471         This function determines the Common Name (CN), Subject Alternative Names (SANs) and Organization Name
 472         our certificate should have and then fetches a matching cert from the certstore.
 473         """
 474         host = None
 475         sans = set()
 476         organization = None
 477 
 478         # In normal operation, the server address should always be known at this point.
 479         # However, we may just want to establish TLS so that we can send an error message to the client,
 480         # in which case the address can be None.
 481         if self.server_conn.address:
 482             host = self.server_conn.address[0].encode("idna")
 483 
 484         # Should we incorporate information from the server certificate?
 485         use_upstream_cert = (
 486             self.server_conn and
 487             self.server_conn.tls_established and
 488             self.config.options.upstream_cert
 489         )
 490         if use_upstream_cert:
 491             upstream_cert = self.server_conn.cert
 492             sans.update(upstream_cert.altnames)
 493             if upstream_cert.cn:
 494                 sans.add(host)
 495                 host = upstream_cert.cn.decode("utf8").encode("idna")
 496             if upstream_cert.organization:
 497                 organization = upstream_cert.organization
 498         # Also add SNI values.
 499         if self._client_hello.sni:
 500             sans.add(self._client_hello.sni)
 501         if self._custom_server_sni:
 502             sans.add(self._custom_server_sni.encode("idna"))
 503 
 504         # RFC 2818: If a subjectAltName extension of type dNSName is present, that MUST be used as the identity.
 505         # In other words, the Common Name is irrelevant then.
 506         if host:
 507             sans.add(host)
 508         return self.config.certstore.get_cert(host, list(sans), organization)
