  23 class SafeH2Connection(connection.H2Connection):
  24 
  25     def __init__(self, conn, *args, **kwargs):
  26         super().__init__(*args, **kwargs)
  27         self.conn = conn
  28         self.lock = threading.RLock()
  29 
  30     def safe_acknowledge_received_data(self, acknowledged_size: int, stream_id: int):
  31         if acknowledged_size == 0:
  32             return
  33 
  34         with self.lock:
  35             self.acknowledge_received_data(acknowledged_size, stream_id)
  36             self.conn.send(self.data_to_send())
  37 
  38     def safe_reset_stream(self, stream_id: int, error_code: int):
  39         with self.lock:
  40             try:
  41                 self.reset_stream(stream_id, error_code)
  42             except h2.exceptions.StreamClosedError:  # pragma: no cover
  43                 # stream is already closed - good
  44                 pass
  45             self.conn.send(self.data_to_send())
  46 
  47     def safe_update_settings(self, new_settings: Dict[int, Any]):
  48         with self.lock:
  49             self.update_settings(new_settings)
  50             self.conn.send(self.data_to_send())
  51 
  52     def safe_send_headers(self, raise_zombie: Callable, stream_id: int, headers: headers.Headers, **kwargs):
  53         with self.lock:
  54             raise_zombie()
  55             self.send_headers(stream_id, headers.fields, **kwargs)
  56             self.conn.send(self.data_to_send())
  57 
  58     def safe_send_body(self, raise_zombie: Callable, stream_id: int, chunks: List[bytes], end_stream=True):
  59         for chunk in chunks:
  60             position = 0
  61             while position < len(chunk):
  62                 self.lock.acquire()
  63                 raise_zombie(self.lock.release)
  64                 max_outbound_frame_size = self.max_outbound_frame_size
  65                 frame_chunk = chunk[position:position + max_outbound_frame_size]
  66                 if self.local_flow_control_window(stream_id) < len(frame_chunk):  # pragma: no cover
  67                     self.lock.release()
  68                     time.sleep(0.1)
  69                     continue
  70                 self.send_data(stream_id, frame_chunk)
  71                 try:
  72                     self.conn.send(self.data_to_send())
  73                 except Exception as e:  # pragma: no cover
  74                     raise e
  75                 finally:
  76                     self.lock.release()
  77                 position += max_outbound_frame_size
  78         if end_stream:
  79             with self.lock:
  80                 raise_zombie()
  81                 self.end_stream(stream_id)
  82                 self.conn.send(self.data_to_send())

 141     def _handle_event(self, event, source_conn, other_conn, is_server):
 142         self.log(
 143             "HTTP2 Event from {}".format("server" if is_server else "client"),
 144             "debug",
 145             [repr(event)]
 146         )
 147 
 148         eid = None
 149         if hasattr(event, 'stream_id'):
 150             if is_server and event.stream_id % 2 == 1:
 151                 eid = self.server_to_client_stream_ids[event.stream_id]
 152             else:
 153                 eid = event.stream_id
 154 
 155         if isinstance(event, events.RequestReceived):
 156             return self._handle_request_received(eid, event)
 157         elif isinstance(event, events.ResponseReceived):
 158             return self._handle_response_received(eid, event)
 159         elif isinstance(event, events.DataReceived):
 160             return self._handle_data_received(eid, event, source_conn)
 161         elif isinstance(event, events.StreamEnded):
 162             return self._handle_stream_ended(eid)
 163         elif isinstance(event, events.StreamReset):
 164             return self._handle_stream_reset(eid, event, is_server, other_conn)
 165         elif isinstance(event, events.RemoteSettingsChanged):
 166             return self._handle_remote_settings_changed(event, other_conn)
 167         elif isinstance(event, events.ConnectionTerminated):
 168             return self._handle_connection_terminated(event, is_server)
 169         elif isinstance(event, events.PushedStreamReceived):
 170             return self._handle_pushed_stream_received(event)
 171         elif isinstance(event, events.PriorityUpdated):
 172             return self._handle_priority_updated(eid, event)
 173         elif isinstance(event, events.TrailersReceived):
 174             return self._handle_trailers(eid, event, is_server, other_conn)
 175 
 176         # fail-safe for unhandled events
 177         return True

 179     def _handle_request_received(self, eid, event):
 180         headers = mitmproxy.net.http.Headers([[k, v] for k, v in event.headers])
 181         self.streams[eid] = Http2SingleStreamLayer(self, self.connections[self.client_conn], eid, headers)
 182         self.streams[eid].timestamp_start = time.time()
 183         if event.priority_updated is not None:
 184             self.streams[eid].priority_exclusive = event.priority_updated.exclusive
 185             self.streams[eid].priority_depends_on = event.priority_updated.depends_on
 186             self.streams[eid].priority_weight = event.priority_updated.weight
 187             self.streams[eid].handled_priority_event = event.priority_updated
 188         self.streams[eid].start()
 189         self.streams[eid].request_message.arrived.set()
 190         return True
 191 
 192     def _handle_response_received(self, eid, event):
 193         headers = mitmproxy.net.http.Headers([[k, v] for k, v in event.headers])
 194         self.streams[eid].queued_data_length = 0
 195         self.streams[eid].timestamp_start = time.time()
 196         self.streams[eid].response_message.headers = headers
 197         self.streams[eid].response_message.arrived.set()
 198         return True
 199 
 200     def _handle_data_received(self, eid, event, source_conn):
 201         bsl = human.parse_size(self.config.options.body_size_limit)
 202         if bsl and self.streams[eid].queued_data_length > bsl:
 203             self.streams[eid].kill()
 204             self.connections[source_conn].safe_reset_stream(
 205                 event.stream_id,
 206                 h2.errors.ErrorCodes.REFUSED_STREAM
 207             )
 208             self.log("HTTP body too large. Limit is {}.".format(bsl), "info")
 209         else:
 210             self.streams[eid].data_queue.put(event.data)
 211             self.streams[eid].queued_data_length += len(event.data)
 212 
 213         # always acknowledge receved data with a WINDOW_UPDATE frame
 214         self.connections[source_conn].safe_acknowledge_received_data(
 215             event.flow_controlled_length,
 216             event.stream_id
 217         )
 218         return True
 219 
 220     def _handle_stream_ended(self, eid):
 221         self.streams[eid].timestamp_end = time.time()
 222         self.streams[eid].stream_ended.set()
 223         return True
 224 
 225     def _handle_stream_reset(self, eid, event, is_server, other_conn):
 226         if eid in self.streams:
 227             self.streams[eid].kill()
 228             if is_server:
 229                 other_stream_id = self.streams[eid].client_stream_id
 230             else:
 231                 other_stream_id = self.streams[eid].server_stream_id
 232             if other_stream_id is not None:
 233                 self.connections[other_conn].safe_reset_stream(other_stream_id, event.error_code)
 234         return True
 235 
 236     def _handle_trailers(self, eid, event, is_server, other_conn):
 237         trailers = mitmproxy.net.http.Headers([[k, v] for k, v in event.headers])
 238         self.streams[eid].trailers = trailers
 239         return True

 241     def _handle_remote_settings_changed(self, event, other_conn):
 242         new_settings = dict([(key, cs.new_value) for (key, cs) in event.changed_settings.items()])
 243         self.connections[other_conn].safe_update_settings(new_settings)
 244         return True

 246     def _handle_connection_terminated(self, event, is_server):
 247         self.log("HTTP/2 connection terminated by {}: error code: {}, last stream id: {}, additional data: {}".format(
 248             "server" if is_server else "client",
 249             event.error_code,
 250             event.last_stream_id,
 251             event.additional_data), "info")
 252 
 253         if event.error_code != h2.errors.ErrorCodes.NO_ERROR:
 254             # Something terrible has happened - kill everything!
 255             self.connections[self.client_conn].close_connection(
 256                 error_code=event.error_code,
 257                 last_stream_id=event.last_stream_id,
 258                 additional_data=event.additional_data
 259             )
 260             self.client_conn.send(self.connections[self.client_conn].data_to_send())
 261             self._kill_all_streams()
 262         else:
 263             """
 264             Do not immediately terminate the other connection.
 265             Some streams might be still sending data to the client.
 266             """
 267         return False

 269     def _handle_pushed_stream_received(self, event):
 270         # pushed stream ids should be unique and not dependent on race conditions
 271         # only the parent stream id must be looked up first
 272 
 273         parent_eid = self.server_to_client_stream_ids[event.parent_stream_id]
 274         with self.connections[self.client_conn].lock:
 275             self.connections[self.client_conn].push_stream(parent_eid, event.pushed_stream_id, event.headers)
 276             self.client_conn.send(self.connections[self.client_conn].data_to_send())
 277 
 278         headers = mitmproxy.net.http.Headers([[k, v] for k, v in event.headers])
 279         layer = Http2SingleStreamLayer(self, self.connections[self.client_conn], event.pushed_stream_id, headers)
 280         self.streams[event.pushed_stream_id] = layer
 281         self.streams[event.pushed_stream_id].timestamp_start = time.time()
 282         self.streams[event.pushed_stream_id].pushed = True
 283         self.streams[event.pushed_stream_id].parent_stream_id = parent_eid
 284         self.streams[event.pushed_stream_id].timestamp_end = time.time()
 285         self.streams[event.pushed_stream_id].request_message.arrived.set()
 286         self.streams[event.pushed_stream_id].request_message.stream_ended.set()
 287         self.streams[event.pushed_stream_id].start()
 288         return True

 290     def _handle_priority_updated(self, eid, event):
 291         if not self.config.options.http2_priority:
 292             self.log("HTTP/2 PRIORITY frame suppressed. Use --http2-priority to enable forwarding.", "debug")
 293             return True
 294 
 295         if eid in self.streams and self.streams[eid].handled_priority_event is event:
 296             # this event was already handled during stream creation
 297             # HeadersFrame + Priority information as RequestReceived
 298             return True
 299 
 300         with self.connections[self.server_conn].lock:
 301             mapped_stream_id = event.stream_id
 302             if mapped_stream_id in self.streams and self.streams[mapped_stream_id].server_stream_id:
 303                 # if the stream is already up and running and was sent to the server,
 304                 # use the mapped server stream id to update priority information
 305                 mapped_stream_id = self.streams[mapped_stream_id].server_stream_id
 306 
 307             if eid in self.streams:
 308                 self.streams[eid].priority_exclusive = event.exclusive
 309                 self.streams[eid].priority_depends_on = event.depends_on
 310                 self.streams[eid].priority_weight = event.weight
 311 
 312             self.connections[self.server_conn].prioritize(
 313                 mapped_stream_id,
 314                 weight=event.weight,
 315                 depends_on=self._map_depends_on_stream_id(mapped_stream_id, event.depends_on),
 316                 exclusive=event.exclusive
 317             )
 318             self.server_conn.send(self.connections[self.server_conn].data_to_send())
 319         return True
 320 
 321     def _map_depends_on_stream_id(self, stream_id, depends_on):
 322         mapped_depends_on = depends_on
 323         if mapped_depends_on in self.streams and self.streams[mapped_depends_on].server_stream_id:
 324             # if the depends-on-stream is already up and running and was sent to the server
 325             # use the mapped server stream id to update priority information
 326             mapped_depends_on = self.streams[mapped_depends_on].server_stream_id
 327         if stream_id == mapped_depends_on:
 328             # looks like one of the streams wasn't opened yet
 329             # prevent self-dependent streams which result in ProtocolError
 330             mapped_depends_on += 2
 331         return mapped_depends_on

 333     def _cleanup_streams(self):
 334         death_time = time.time() - 10
 335 
 336         zombie_streams = [(stream_id, stream) for stream_id, stream in list(self.streams.items()) if stream.zombie]
 337         outdated_streams = [stream_id for stream_id, stream in zombie_streams if stream.zombie <= death_time]
 338 
 339         for stream_id in outdated_streams:  # pragma: no cover
 340             self.streams.pop(stream_id, None)
 341 
 342     def _kill_all_streams(self):
 343         for stream in self.streams.values():
 344             stream.kill()

 346     def __call__(self):
 347         self._initiate_server_conn()
 348         self._complete_handshake()
 349 
 350         conns = [c.connection for c in self.connections.keys()]
 351 
 352         try:
 353             while True:
 354                 r = tcp.ssl_read_select(conns, 0.1)
 355                 for conn in r:
 356                     source_conn = self.client_conn if conn == self.client_conn.connection else self.server_conn
 357                     other_conn = self.server_conn if conn == self.client_conn.connection else self.client_conn
 358                     is_server = (source_conn == self.server_conn)
 359 
 360                     with self.connections[source_conn].lock:
 361                         try:
 362                             raw_frame = b''.join(http2.read_raw_frame(source_conn.rfile))
 363                         except:
 364                             # read frame failed: connection closed
 365                             self._kill_all_streams()
 366                             return
 367 
 368                         if self.connections[source_conn].state_machine.state == h2.connection.ConnectionState.CLOSED:
 369                             self.log("HTTP/2 connection entered closed state already", "debug")
 370                             return
 371 
 372                         incoming_events = self.connections[source_conn].receive_data(raw_frame)
 373                         source_conn.send(self.connections[source_conn].data_to_send())
 374 
 375                         for event in incoming_events:
 376                             if not self._handle_event(event, source_conn, other_conn, is_server):
 377                                 # connection terminated: GoAway
 378                                 self._kill_all_streams()
 379                                 return
 380 
 381                     self._cleanup_streams()
 382         except Exception as e:  # pragma: no cover
 383             self.log(repr(e), "info")
 384             self._kill_all_streams()

 387 def detect_zombie_stream(func):  # pragma: no cover
 388     @functools.wraps(func)
 389     def wrapper(self, *args, **kwargs):
 390         self.raise_zombie()
 391         result = func(self, *args, **kwargs)
 392         self.raise_zombie()
 393         return result
 394 
 395     return wrapper

 400     class Message:
 401         def __init__(self, headers=None):
 402             self.headers: Optional[mitmproxy.net.http.Headers] = headers  # headers are the first thing to be received on a new stream
 403             self.data_queue: queue.Queue[bytes] = queue.Queue()  # contains raw contents of DATA frames
 404             self.queued_data_length = 0  # used to enforce mitmproxy's config.options.body_size_limit
 405             self.trailers: Optional[mitmproxy.net.http.Headers] = None  # trailers are received after stream_ended is set
 406 
 407             self.arrived = threading.Event()  # indicates the HEADERS+CONTINUTATION frames have been received
 408             self.stream_ended = threading.Event()  # indicates the a frame with the END_STREAM flag has been received

 410     def __init__(self, ctx, h2_connection, stream_id: int, request_headers: mitmproxy.net.http.Headers) -> None:
 411         super().__init__(
 412             ctx, name="Http2SingleStreamLayer-{}".format(stream_id)

 431     def kill(self):
 432         if not self.zombie:
 433             self.zombie = time.time()
 434             self.request_message.stream_ended.set()
 435             self.request_message.arrived.set()
 436             self.response_message.arrived.set()
 437             self.response_message.stream_ended.set()

 494     def raise_zombie(self, pre_command=None):  # pragma: no cover
 495         connection_closed = self.h2_connection.state_machine.state == h2.connection.ConnectionState.CLOSED
 496         if self.zombie is not None or connection_closed:
 497             if pre_command is not None:
 498                 pre_command()
 499             raise exceptions.Http2ZombieException("Connection or stream already dead: {}, {}".format(self.zombie, connection_closed))

 501     @detect_zombie_stream
 502     def read_request_headers(self, flow):
 503         self.request_message.arrived.wait()
 504         self.raise_zombie()
 505 
 506         if self.pushed:
 507             flow.metadata['h2-pushed-stream'] = True
 508 
 509         first_line_format, method, scheme, host, port, path = http2.parse_headers(self.request_message.headers)
 510         return http.HTTPRequest(
 511             first_line_format,
 512             method,
 513             scheme,
 514             host,
 515             port,
 516             path,
 517             b"HTTP/2.0",
 518             self.request_message.headers,
 519             None,
 520             timestamp_start=self.timestamp_start,
 521             timestamp_end=self.timestamp_end,
 522         )
 523 
 524     @detect_zombie_stream
 525     def read_request_body(self, request):
 526         if not request.stream:
 527             self.request_message.stream_ended.wait()
 528 
 529         while True:
 530             try:
 531                 yield self.request_message.data_queue.get(timeout=0.1)
 532             except queue.Empty:  # pragma: no cover
 533                 pass
 534             if self.request_message.stream_ended.is_set():
 535                 self.raise_zombie()
 536                 while self.request_message.data_queue.qsize() > 0:
 537                     yield self.request_message.data_queue.get()
 538                 break
 539             self.raise_zombie()
 540 
 541     @detect_zombie_stream
 542     def read_request_trailers(self, request):
 543         return self.request_message.trailers
 544 
 545     @detect_zombie_stream
 546     def send_request_headers(self, request):
 547         if self.pushed:
 548             # nothing to do here
 549             return
 550 
 551         while True:
 552             self.raise_zombie()
 553             self.connections[self.server_conn].lock.acquire()
 554 
 555             max_streams = self.connections[self.server_conn].remote_settings.max_concurrent_streams
 556             if self.connections[self.server_conn].open_outbound_streams + 1 >= max_streams:
 557                 # wait until we get a free slot for a new outgoing stream
 558                 self.connections[self.server_conn].lock.release()
 559                 time.sleep(0.1)
 560                 continue
 561 
 562             # keep the lock
 563             break
 564 
 565         # We must not assign a stream id if we are already a zombie.
 566         self.raise_zombie()
 567 
 568         self.server_stream_id = self.connections[self.server_conn].get_next_available_stream_id()
 569         self.server_to_client_stream_ids[self.server_stream_id] = self.client_stream_id
 570 
 571         headers = request.headers.copy()
 572         headers.insert(0, ":path", request.path)
 573         headers.insert(0, ":method", request.method)
 574         headers.insert(0, ":scheme", request.scheme)
 575 
 576         priority_exclusive = None
 577         priority_depends_on = None
 578         priority_weight = None
 579         if self.handled_priority_event:
 580             # only send priority information if they actually came with the original HeadersFrame
 581             # and not if they got updated before/after with a PriorityFrame
 582             if not self.config.options.http2_priority:
 583                 self.log("HTTP/2 PRIORITY information in HEADERS frame suppressed. Use --http2-priority to enable forwarding.", "debug")
 584             else:
 585                 priority_exclusive = self.priority_exclusive
 586                 priority_depends_on = self._map_depends_on_stream_id(self.server_stream_id, self.priority_depends_on)
 587                 priority_weight = self.priority_weight
 588 
 589         try:
 590             self.connections[self.server_conn].safe_send_headers(
 591                 self.raise_zombie,
 592                 self.server_stream_id,
 593                 headers,
 594                 priority_exclusive=priority_exclusive,
 595                 priority_depends_on=priority_depends_on,
 596                 priority_weight=priority_weight,
 597             )
 598         except Exception as e:  # pragma: no cover
 599             raise e
 600         finally:
 601             self.raise_zombie()
 602             self.connections[self.server_conn].lock.release()
 603 
 604     @detect_zombie_stream
 605     def send_request_body(self, request, chunks):
 606         if self.pushed:
 607             # nothing to do here
 608             return
 609 
 610         self.connections[self.server_conn].safe_send_body(
 611             self.raise_zombie,
 612             self.server_stream_id,
 613             chunks,
 614             end_stream=(request.trailers is None),
 615         )
 616 
 617     @detect_zombie_stream
 618     def send_request_trailers(self, request):
 619         self._send_trailers(self.server_conn, request.trailers)
 620 
 621     @detect_zombie_stream
 622     def send_request(self, request):
 623         self.send_request_headers(request)
 624         self.send_request_body(request, [request.content])
 625         self.send_request_trailers(request)

 627     @detect_zombie_stream
 628     def read_response_headers(self):
 629         self.response_message.arrived.wait()
 630 
 631         self.raise_zombie()
 632 
 633         status_code = int(self.response_message.headers.get(':status', 502))
 634         headers = self.response_message.headers.copy()
 635         headers.pop(":status", None)
 636 
 637         return http.HTTPResponse(
 638             http_version=b"HTTP/2.0",
 639             status_code=status_code,
 640             reason=b'',
 641             headers=headers,
 642             content=None,
 643             timestamp_start=self.timestamp_start,
 644             timestamp_end=self.timestamp_end,
 645         )

 647     @detect_zombie_stream
 648     def read_response_body(self, request, response):
 649         while True:
 650             try:
 651                 yield self.response_message.data_queue.get(timeout=0.1)
 652             except queue.Empty:  # pragma: no cover
 653                 pass
 654             if self.response_message.stream_ended.is_set():
 655                 self.raise_zombie()
 656                 while self.response_message.data_queue.qsize() > 0:
 657                     yield self.response_message.data_queue.get()
 658                 break
 659             self.raise_zombie()
 660 
 661     @detect_zombie_stream
 662     def read_response_trailers(self, request, response):
 663         return self.response_message.trailers
 664 
 665     @detect_zombie_stream
 666     def send_response_headers(self, response):
 667         headers = response.headers.copy()
 668         headers.insert(0, ":status", str(response.status_code))
 669         with self.connections[self.client_conn].lock:
 670             self.connections[self.client_conn].safe_send_headers(
 671                 self.raise_zombie,
 672                 self.client_stream_id,
 673                 headers
 674             )

 676     @detect_zombie_stream
 677     def send_response_body(self, response, chunks):
 678         self.connections[self.client_conn].safe_send_body(
 679             self.raise_zombie,
 680             self.client_stream_id,
 681             chunks,
 682             end_stream=(response.trailers is None),
 683         )

 685     @detect_zombie_stream
 686     def send_response_trailers(self, response):
 687         self._send_trailers(self.client_conn, response.trailers)
 688 
 689     def _send_trailers(self, conn, trailers):
 690         if not trailers:
 691             return
 692         with self.connections[conn].lock:
 693             self.connections[conn].safe_send_headers(
 694                 self.raise_zombie,
 695                 self.client_stream_id,
 696                 trailers,
 697                 end_stream=True
 698             )

 700     def __call__(self):  # pragma: no cover
 701         raise EnvironmentError('Http2SingleStreamLayer must be run as thread')
 702 
 703     def run(self):
 704         layer = httpbase.HttpLayer(self, self.mode)
 705 
 706         try:
 707             layer()
 708         except exceptions.Http2ZombieException:  # pragma: no cover
 709             # zombies can be safely terminated - no need to kill them twice
 710             return
 711         except exceptions.ProtocolException as e:  # pragma: no cover
 712             self.log(repr(e), "info")
 713         except exceptions.SetServerNotAllowedException as e:  # pragma: no cover
 714             self.log("Changing the Host server for HTTP/2 connections not allowed: {}".format(e), "info")
 715         except exceptions.Kill:  # pragma: no cover
 716             self.log("Connection killed", "info")
 717 
 718         self.kill()
