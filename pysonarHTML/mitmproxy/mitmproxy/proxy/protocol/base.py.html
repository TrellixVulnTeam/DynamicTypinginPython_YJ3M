  25 class Layer(_LayerCodeCompletion):
  26 
  27     """
  28     Base class for all layers. All other protocol layers should inherit from this class.
  29     """
  30 
  31     def __init__(self, ctx, **mixin_args):
  32         """
  33         Each layer usually passes itself to its child layers as a context. Properties of the
  34         context are transparently mapped to the layer, so that the following works:
  35 
  36         .. code-block:: python
  37 
  38             root_layer = Layer(None)
  39             root_layer.client_conn = 42
  40             sub_layer = Layer(root_layer)
  41             print(sub_layer.client_conn) # 42
  42 
  43         The root layer is passed a :py:class:`mitmproxy.proxy.RootContext` object,
  44         which provides access to :py:attr:`.client_conn <mitmproxy.proxy.RootContext.client_conn>`,
  45         :py:attr:`.next_layer <mitmproxy.proxy.RootContext.next_layer>` and other basic attributes.
  46 
  47         Args:
  48             ctx: The (read-only) parent layer / context.
  49         """
  50         self.ctx = ctx
  51         """
  52         The parent layer.
  53 
  54         :type: :py:class:`Layer`
  55         """
  56         super().__init__(**mixin_args)
  57 
  58     def __call__(self):
  59         """Logic of the layer.
  60 
  61         Returns:
  62             Once the protocol has finished without exceptions.
  63 
  64         Raises:
  65             ~mitmproxy.exceptions.ProtocolException: if an exception occurs. No other exceptions must be raised.

  67         raise NotImplementedError()

  70         """
  71         Attributes not present on the current layer are looked up on the context.

  73         return getattr(self.ctx, name)

  76 class ServerConnectionMixin:
  77 
  78     """
  79     Mixin that provides a layer with the capabilities to manage a server connection.
  80     The server address can be passed in the constructor or set by calling :py:meth:`set_server`.
  81     Subclasses are responsible for calling :py:meth:`disconnect` before returning.
  82 
  83     Recommended Usage:
  84 
  85     .. code-block:: python
  86 
  87         class MyLayer(Layer, ServerConnectionMixin):
  88             def __call__(self):
  89                 try:
  90                     # Do something.
  91                 finally:
  92                     if self.server_conn.connected():
  93                         self.disconnect()
  94     """
  95 
  96     def __init__(self, server_address=None):
  97         super().__init__()
  98 
  99         self.server_conn = self.__make_server_conn(server_address)
 100 
 101         self.__check_self_connect()
 102 
 103     def __check_self_connect(self):
 104         """
 105         We try to protect the proxy from _accidentally_ connecting to itself,
 106         e.g. because of a failed transparent lookup or an invalid configuration.
 107         """
 108         address = self.server_conn.address
 109         if address:
 110             self_connect = (
 111                 address[1] == self.config.options.listen_port and
 112                 address[0] in ("localhost", "127.0.0.1", "::1")
 113             )
 114             if self_connect:
 115                 raise exceptions.ProtocolException(
 116                     "Invalid server address: {}\r\n"
 117                     "The proxy shall not connect to itself.".format(repr(address))
 118                 )
 119 
 120     def __make_server_conn(self, server_address):
 121         if self.config.options.spoof_source_address and self.config.options.upstream_bind_address == '':
 122             return connections.ServerConnection(
 123                 server_address, (self.ctx.client_conn.address[0], 0), True)
 124         else:
 125             return connections.ServerConnection(
 126                 server_address, (self.config.options.upstream_bind_address, 0),
 127                 self.config.options.spoof_source_address
 128             )
 129 
 130     def set_server(self, address):
 131         """
 132         Sets a new server address. If there is an existing connection, it will be closed.
 133         """
 134         if self.server_conn.connected():
 135             self.disconnect()
 136         self.log("Set new server address: {}:{}".format(address[0], address[1]), "debug")
 137         self.server_conn.address = address
 138         self.__check_self_connect()
 139 
 140     def disconnect(self):
 141         """
 142         Deletes (and closes) an existing server connection.
 143         Must not be called if there is no existing connection.
 144         """
 145         self.log("serverdisconnect", "debug", [repr(self.server_conn.address)])
 146         address = self.server_conn.address
 147         self.server_conn.finish()
 148         self.server_conn.close()
 149         self.channel.tell("serverdisconnect", self.server_conn)
 150 
 151         self.server_conn = self.__make_server_conn(address)
 152 
 153     def connect(self):
 154         """
 155         Establishes a server connection.
 156         Must not be called if there is an existing connection.
 157 
 158         Raises:
 159             ~mitmproxy.exceptions.ProtocolException: if the connection could not be established.
 160         """
 161         if not self.server_conn.address:
 162             raise exceptions.ProtocolException("Cannot connect to server, no server address given.")
 163         try:
 164             self.server_conn.connect()
 165             self.log("serverconnect", "debug", [repr(self.server_conn.address)])
 166             self.channel.ask("serverconnect", self.server_conn)
 167         except exceptions.TcpException as e:
 168             raise exceptions.ProtocolException(
 169                 "Server connection to {} failed: {}".format(
 170                     repr(self.server_conn.address), str(e)
 171                 )
 172             )
