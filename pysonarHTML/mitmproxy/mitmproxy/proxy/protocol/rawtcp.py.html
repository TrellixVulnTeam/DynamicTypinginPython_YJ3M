  42                     try:

  44                     except (SSL.WantReadError, SSL.WantWriteError):
  45                         continue

  56                         if len(conns) == 0:

  61                     if not self.ignore:

  63                         self.channel.ask("tcp_message", f)
  64                     dst.sendall(tcp_message.content)
