 102     def _send_connect_request(self):
 103         self.log("Sending CONNECT request", "debug", [
 104             "Proxy Server: {}".format(self.ctx.server_conn.address),
 105             "Connect to: {}:{}".format(self.connect_request.host, self.connect_request.port)
 106         ])
 107         self.send_request(self.connect_request)
 108         resp = self.read_response(self.connect_request)
 109         if resp.status_code != 200:
 110             raise exceptions.ProtocolException("Reconnect: Upstream server refuses CONNECT request")
 111 
 112     def connect(self):
 113         if not self.server_conn.connected():
 114             self.ctx.connect()
 115             self._send_connect_request()
 116         else:
 117             pass  # swallow the message
 118 
 119     def change_upstream_proxy_server(self, address):
 120         self.log("Changing upstream proxy to {} (CONNECTed)".format(repr(address)), "debug")
 121         if address != self.server_conn.via.address:
 122             self.ctx.set_server(address)
 123 
 124     def set_server(self, address):
 125         if self.ctx.server_conn.connected():
 126             self.ctx.disconnect()
 127         self.connect_request.host = address[0]
 128         self.connect_request.port = address[1]
 129         self.server_conn.address = address

 142 # At this point, we see only a subset of the proxy modes
 143 MODE_REQUEST_FORMS = {
 144     HTTPMode.regular: ("authority", "absolute"),
 145     HTTPMode.transparent: ("relative",),
 146     HTTPMode.upstream: ("authority", "absolute"),
 147 }

 150 def validate_request_form(mode, request):
 151     if request.first_line_format == "absolute" and request.scheme != "http":
 152         raise exceptions.HttpException(
 153             "Invalid request scheme: %s" % request.scheme
 154         )
 155     allowed_request_forms = MODE_REQUEST_FORMS[mode]
 156     if request.first_line_format not in allowed_request_forms:
 157         if mode == HTTPMode.transparent:
 158             err_message = textwrap.dedent((
 159                 """
 160                 Mitmproxy received an {} request even though it is not running
 161                 in regular mode. This usually indicates a misconfiguration,
 162                 please see the mitmproxy mode documentation for details.
 163                 """
 164             )).strip().format("HTTP CONNECT" if request.first_line_format == "authority" else "absolute-form")
 165         else:
 166             err_message = "Invalid HTTP request form (expected: %s, got: %s)" % (
 167                 " or ".join(allowed_request_forms), request.first_line_format
 168             )
 169         raise exceptions.HttpException(err_message)

 173

 178     def __init__(self, ctx, mode):
 179         super().__init__(ctx)
 180         self.mode = mode
 181         self.__initial_server_address: tuple = None
 182         "Contains the original destination in transparent mode, which needs to be restored"
 183         "if an inline script modified the target server for a single http request"
 184         # We cannot rely on server_conn.tls_established,
 185         # see https://github.com/mitmproxy/mitmproxy/issues/925
 186         self.__initial_server_tls = None
 187         # Requests happening after CONNECT do not need Proxy-Authorization headers.
 188         self.connect_request = False
 189 
 190     def __call__(self):
 191         if self.mode == HTTPMode.transparent:
 192             self.__initial_server_tls = self.server_tls
 193             self.__initial_server_address = self.server_conn.address
 194         while True:
 195             flow = http.HTTPFlow(
 196                 self.client_conn,
 197                 self.server_conn,
 198                 live=self,
 199                 mode=self.mode.name
 200             )
 201             if not self._process_flow(flow):
 202                 return
 203 
 204     def handle_regular_connect(self, f):
 205         self.connect_request = True
 206 
 207         try:
 208             self.set_server((f.request.host, f.request.port))
 209 
 210             if f.response:
 211                 resp = f.response
 212             else:
 213                 resp = http.make_connect_response(f.request.data.http_version)
 214 
 215             self.send_response(resp)
 216 
 217             if is_ok(resp.status_code):
 218                 layer = self.ctx.next_layer(self)
 219                 layer()
 220         except (
 221             exceptions.ProtocolException, exceptions.NetlibException
 222         ) as e:
 223             # HTTPS tasting means that ordinary errors like resolution
 224             # and connection errors can happen here.
 225             self.send_error_response(502, repr(e))
 226             f.error = flow.Error(str(e))
 227             self.channel.ask("error", f)
 228             return False
 229 
 230         return False
 231 
 232     def handle_upstream_connect(self, f):
 233         # if the user specifies a response in the http_connect hook, we do not connect upstream here.
 234         # https://github.com/mitmproxy/mitmproxy/pull/2473
 235         if not f.response:
 236             self.establish_server_connection(
 237                 f.request.host,
 238                 f.request.port,
 239                 f.request.scheme
 240             )
 241             self.send_request(f.request)
 242             f.response = self.read_response_headers()
 243             f.response.data.content = b"".join(
 244                 self.read_response_body(f.request, f.response)
 245             )
 246         self.send_response(f.response)
 247         if is_ok(f.response.status_code):
 248             layer = UpstreamConnectLayer(self, f.request)
 249             return layer()
 250         return False

 252     def _process_flow(self, f):
 253         try:
 254             try:
 255                 request = self.read_request_headers(f)
 256             except exceptions.HttpReadDisconnect:
 257                 # don't throw an error for disconnects that happen
 258                 # before/between requests.
 259                 return False
 260 
 261             f.request = request
 262 
 263             if request.first_line_format == "authority":
 264                 # The standards are silent on what we should do with a CONNECT
 265                 # request body, so although it's not common, it's allowed.
 266                 f.request.data.content = b"".join(
 267                     self.read_request_body(f.request)
 268                 )
 269                 f.request.data.trailers = self.read_request_trailers(f.request)
 270                 f.request.timestamp_end = time.time()
 271                 self.channel.ask("http_connect", f)
 272 
 273                 if self.mode is HTTPMode.regular:
 274                     return self.handle_regular_connect(f)
 275                 elif self.mode is HTTPMode.upstream:
 276                     return self.handle_upstream_connect(f)
 277                 else:
 278                     msg = "Unexpected CONNECT request."
 279                     self.send_error_response(400, msg)
 280                     return False
 281 
 282             validate_request_form(self.mode, request)
 283             self.channel.ask("requestheaders", f)
 284             # Re-validate request form in case the user has changed something.
 285             validate_request_form(self.mode, request)
 286 
 287             if request.headers.get("expect", "").lower() == "100-continue":
 288                 # TODO: We may have to use send_response_headers for HTTP2
 289                 # here.
 290                 self.send_response(http.expect_continue_response)
 291                 request.headers.pop("expect")
 292 
 293             if f.request.stream:
 294                 f.request.data.content = None
 295             else:
 296                 f.request.data.content = b"".join(self.read_request_body(request))
 297 
 298             f.request.data.trailers = self.read_request_trailers(f.request)
 299 
 300             request.timestamp_end = time.time()
 301         except exceptions.HttpException as e:
 302             # We optimistically guess there might be an HTTP client on the
 303             # other end
 304             self.send_error_response(400, repr(e))
 305             # Request may be malformed at this point, so we unset it.
 306             f.request = None
 307             f.error = flow.Error(str(e))
 308             self.channel.ask("error", f)
 309             self.log(
 310                 "request",
 311                 "warn",
 312                 ["HTTP protocol error in client request: {}".format(e)]
 313             )
 314             return False
 315 
 316         self.log("request", "debug", [repr(request)])
 317 
 318         # set first line format to relative in regular mode,
 319         # see https://github.com/mitmproxy/mitmproxy/issues/1759
 320         if self.mode is HTTPMode.regular and request.first_line_format == "absolute":
 321             request.first_line_format = "relative"
 322 
 323         # update host header in reverse proxy mode
 324         if self.config.options.mode.startswith("reverse:") and not self.config.options.keep_host_header:
 325             f.request.host_header = self.config.upstream_server.address[0]
 326 
 327         # Determine .scheme, .host and .port attributes for inline scripts. For
 328         # absolute-form requests, they are directly given in the request. For
 329         # authority-form requests, we only need to determine the request
 330         # scheme. For relative-form requests, we need to determine host and
 331         # port as well.
 332         if self.mode is HTTPMode.transparent:
 333             # Setting request.host also updates the host header, which we want
 334             # to preserve
 335             host_header = f.request.host_header
 336             f.request.host = self.__initial_server_address[0]
 337             f.request.port = self.__initial_server_address[1]
 338             f.request.host_header = host_header  # set again as .host overwrites this.
 339             f.request.scheme = "https" if self.__initial_server_tls else "http"
 340         self.channel.ask("request", f)
 341 
 342         try:
 343             if websockets.check_handshake(request.headers) and websockets.check_client_version(request.headers):
 344                 f.metadata['websocket'] = True
 345                 # We only support RFC6455 with WebSocket version 13
 346                 # allow inline scripts to manipulate the client handshake
 347                 self.channel.ask("websocket_handshake", f)
 348 
 349             if not f.response:
 350                 self.establish_server_connection(
 351                     f.request.host,
 352                     f.request.port,
 353                     f.request.scheme
 354                 )
 355 
 356                 def get_response():
 357                     self.send_request_headers(f.request)
 358                     if f.request.stream:
 359                         chunks = self.read_request_body(f.request)
 360                         if callable(f.request.stream):
 361                             chunks = f.request.stream(chunks)
 362                         self.send_request_body(f.request, chunks)
 363                     else:
 364                         self.send_request_body(f.request, [f.request.data.content])
 365 
 366                     self.send_request_trailers(f.request)
 367 
 368                     f.response = self.read_response_headers()
 369 
 370                 try:
 371                     get_response()
 372                 except exceptions.NetlibException as e:
 373                     self.log(
 374                         "server communication error: %s" % repr(e),
 375                         level="debug"
 376                     )
 377                     # In any case, we try to reconnect at least once. This is
 378                     # necessary because it might be possible that we already
 379                     # initiated an upstream connection after clientconnect that
 380                     # has already been expired, e.g consider the following event
 381                     # log:
 382                     # > clientconnect (transparent mode destination known)
 383                     # > serverconnect (required for client tls handshake)
 384                     # > read n% of large request
 385                     # > server detects timeout, disconnects
 386                     # > read (100-n)% of large request
 387                     # > send large request upstream
 388 
 389                     if isinstance(e, exceptions.Http2ProtocolException):
 390                         # do not try to reconnect for HTTP2
 391                         raise exceptions.ProtocolException(
 392                             "First and only attempt to get response via HTTP2 failed."
 393                         )
 394                     elif f.request.stream:
 395                         # We may have already consumed some request chunks already,
 396                         # so all we can do is signal downstream that upstream closed the connection.
 397                         self.send_error_response(408, "Request Timeout")
 398                         f.error = flow.Error(repr(e))
 399                         self.channel.ask("error", f)
 400                         return False
 401 
 402                     self.disconnect()
 403                     self.connect()
 404                     get_response()
 405 
 406                 # call the appropriate script hook - this is an opportunity for
 407                 # an inline script to set f.stream = True
 408                 self.channel.ask("responseheaders", f)
 409 
 410                 if f.response.stream:
 411                     f.response.data.content = None
 412                 else:
 413                     f.response.data.content = b"".join(
 414                         self.read_response_body(f.request, f.response)
 415                     )
 416                 f.response.timestamp_end = time.time()
 417 
 418                 # no further manipulation of self.server_conn beyond this point
 419                 # we can safely set it as the final attribute value here.
 420                 f.server_conn = self.server_conn
 421             else:
 422                 # response was set by an inline script.
 423                 # we now need to emulate the responseheaders hook.
 424                 self.channel.ask("responseheaders", f)
 425 
 426             f.response.data.trailers = self.read_response_trailers(f.request, f.response)
 427 
 428             self.log("response", "debug", [repr(f.response)])
 429             self.channel.ask("response", f)
 430 
 431             if not f.response.stream:
 432                 # no streaming:
 433                 # we already received the full response from the server and can
 434                 # send it to the client straight away.
 435                 self.send_response(f.response)
 436             else:
 437                 # streaming:
 438                 # First send the headers and then transfer the response incrementally
 439                 self.send_response_headers(f.response)
 440                 chunks = self.read_response_body(
 441                     f.request,
 442                     f.response
 443                 )
 444                 if callable(f.response.stream):
 445                     chunks = f.response.stream(chunks)
 446                 self.send_response_body(f.response, chunks)
 447                 f.response.timestamp_end = time.time()
 448 
 449             if self.check_close_connection(f):
 450                 return False
 451 
 452             # Handle 101 Switching Protocols
 453             if f.response.status_code == 101:
 454                 # Handle a successful HTTP 101 Switching Protocols Response,
 455                 # received after e.g. a WebSocket upgrade request.
 456                 # Check for WebSocket handshake
 457                 is_websocket = (
 458                     websockets.check_handshake(f.request.headers) and
 459                     websockets.check_handshake(f.response.headers)
 460                 )
 461                 if is_websocket and not self.config.options.websocket:
 462                     self.log(
 463                         "Client requested WebSocket connection, but the protocol is disabled.",
 464                         "info"
 465                     )
 466 
 467                 if is_websocket and self.config.options.websocket:
 468                     layer = WebSocketLayer(self, f)
 469                 else:
 470                     layer = self.ctx.next_layer(self)
 471                 layer()
 472                 return False  # should never be reached
 473 
 474         except (exceptions.ProtocolException, exceptions.NetlibException) as e:
 475             if not f.response:
 476                 self.send_error_response(502, repr(e))
 477                 f.error = flow.Error(str(e))
 478                 self.channel.ask("error", f)
 479                 return False
 480             else:
 481                 raise exceptions.ProtocolException(
 482                     "Error in HTTP connection: %s" % repr(e)
 483                 )
 484         finally:
 485             if f:
 486                 f.live = False
 487 
 488         return True
 489 
 490     def send_error_response(self, code, message, headers=None) -> None:
 491         try:
 492             response = http.make_error_response(code, message, headers)
 493             self.send_response(response)
 494         except (exceptions.NetlibException, h2.exceptions.H2Error, exceptions.Http2ProtocolException):
 495             self.log("Failed to send error response to client: {}".format(message), "debug")

 497     def change_upstream_proxy_server(self, address):
 498         # Make set_upstream_proxy_server always available,
 499         # even if there's no UpstreamConnectLayer
 500         if hasattr(self.ctx, "change_upstream_proxy_server"):
 501             self.ctx.change_upstream_proxy_server(address)
 502         elif address != self.server_conn.address:
 503             self.log("Changing upstream proxy to {} (not CONNECTed)".format(repr(address)), "debug")
 504             self.set_server(address)

 506     def establish_server_connection(self, host: str, port: int, scheme: str):
 507         tls = (scheme == "https")
 508 
 509         if self.mode is HTTPMode.regular or self.mode is HTTPMode.transparent:
 510             # If there's an existing connection that doesn't match our expectations, kill it.
 511             address = (host, port)
 512             if address != self.server_conn.address or tls != self.server_tls:
 513                 self.set_server(address)
 514                 self.set_server_tls(tls, address[0])
 515             # Establish connection is necessary.
 516             if not self.server_conn.connected():
 517                 self.connect()
 518         else:
 519             if not self.server_conn.connected():
 520                 self.connect()
 521             if tls:
 522                 raise exceptions.HttpProtocolException("Cannot change scheme in upstream proxy mode.")
