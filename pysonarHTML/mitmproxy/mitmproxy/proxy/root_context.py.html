  10 
  11     """
  12     The outermost context provided to the root layer.
  13     As a consequence, every layer has access to methods and attributes defined here.
  14 
  15     Attributes:
  16         client_conn:
  17             The :py:class:`client connection <mitmproxy.connections.ClientConnection>`.
  18         channel:
  19             A :py:class:`~mitmproxy.controller.Channel` to communicate with the FlowMaster.
  20             Provides :py:meth:`.ask() <mitmproxy.controller.Channel.ask>` and
  21             :py:meth:`.tell() <mitmproxy.controller.Channel.tell>` methods.
  22         config:
  23             The :py:class:`proxy server's configuration <mitmproxy.proxy.ProxyConfig>`
  24     """
  25 
  26     def __init__(self, client_conn, config, channel):
  27         self.client_conn = client_conn
  28         self.channel = channel
  29         self.config = config

  31     def next_layer(self, top_layer):
  32         """
  33         This function determines the next layer in the protocol stack.
  34 
  35         Arguments:
  36             top_layer: the current innermost layer.
  37 
  38         Returns:
  39             The next layer
  40         """
  41         layer = self._next_layer(top_layer)
  42         return self.channel.ask("next_layer", layer)
  43 
  44     def _next_layer(self, top_layer):
  45         try:
  46             d = top_layer.client_conn.rfile.peek(3)
  47         except exceptions.TcpException as e:
  48             raise exceptions.ProtocolException(str(e))
  49         client_tls = tls.is_tls_record_magic(d)
  50 
  51         # 1. check for filter
  52         if self.config.check_filter:
  53             is_filtered = self.config.check_filter(top_layer.server_conn.address)
  54             if not is_filtered and client_tls:
  55                 try:
  56                     client_hello = tls.ClientHello.from_file(self.client_conn.rfile)
  57                 except exceptions.TlsProtocolException as e:
  58                     self.log("Cannot parse Client Hello: %s" % repr(e), "error")
  59                 else:
  60                     sni_str = client_hello.sni and client_hello.sni.decode("idna")
  61                     is_filtered = self.config.check_filter((sni_str, 443))
  62             if is_filtered:
  63                 return protocol.RawTCPLayer(top_layer, ignore=True)
  64 
  65         # 2. Always insert a TLS layer, even if there's neither client nor server tls.
  66         # An inline script may upgrade from http to https,
  67         # in which case we need some form of TLS layer.
  68         if isinstance(top_layer, modes.ReverseProxy):
  69             return protocol.TlsLayer(
  70                 top_layer,
  71                 client_tls,
  72                 top_layer.server_tls,
  73                 top_layer.server_conn.address[0]
  74             )
  75         if isinstance(top_layer, protocol.ServerConnectionMixin):
  76             return protocol.TlsLayer(top_layer, client_tls, client_tls)
  77         if isinstance(top_layer, protocol.UpstreamConnectLayer):
  78             # if the user manually sets a scheme for connect requests, we use this to decide if we
  79             # want TLS or not.
  80             if top_layer.connect_request.scheme:
  81                 server_tls = top_layer.connect_request.scheme == "https"
  82             else:
  83                 server_tls = client_tls
  84             return protocol.TlsLayer(top_layer, client_tls, server_tls)
  85 
  86         # 3. In Http Proxy mode and Upstream Proxy mode, the next layer is fixed.
  87         if isinstance(top_layer, protocol.TlsLayer):
  88             if isinstance(top_layer.ctx, modes.HttpProxy):
  89                 return protocol.Http1Layer(top_layer, http.HTTPMode.regular)
  90             if isinstance(top_layer.ctx, modes.HttpUpstreamProxy):
  91                 return protocol.Http1Layer(top_layer, http.HTTPMode.upstream)
  92 
  93         # 4. Check for other TLS cases (e.g. after CONNECT).
  94         if client_tls:
  95             return protocol.TlsLayer(top_layer, True, True)
  96 
  97         # 4. Check for --tcp
  98         if self.config.check_tcp(top_layer.server_conn.address):
  99             return protocol.RawTCPLayer(top_layer)
 100 
 101         # 5. Check for TLS ALPN (HTTP1/HTTP2)
 102         if isinstance(top_layer, protocol.TlsLayer):
 103             alpn = top_layer.client_conn.get_alpn_proto_negotiated()
 104             if alpn == b'h2':
 105                 return protocol.Http2Layer(top_layer, http.HTTPMode.transparent)
 106             if alpn == b'http/1.1':
 107                 return protocol.Http1Layer(top_layer, http.HTTPMode.transparent)
 108 
 109         # 6. Check for raw tcp mode
 110         is_ascii = (
 111             len(d) == 3 and
 112             # expect A-Za-z
 113             all(65 <= x <= 90 or 97 <= x <= 122 for x in d)
 114         )
 115         if self.config.options.rawtcp and not is_ascii:
 116             return protocol.RawTCPLayer(top_layer)
 117 
 118         # 7. Assume HTTP1 by default
 119         return protocol.Http1Layer(top_layer, http.HTTPMode.transparent)

 121     def log(self, msg, level, subs=()):
 122         """
 123         Send a log message to the master.
 124         """
 125         full_msg = [
 126             "{}:{}: {}".format(self.client_conn.address[0], self.client_conn.address[1], msg)
 127         ]
 128         for i in subs:
 129             full_msg.append("  -> " + i)
 130         full_msg = "\n".join(full_msg)
 131         self.channel.tell("log", log.LogEntry(full_msg, level))
