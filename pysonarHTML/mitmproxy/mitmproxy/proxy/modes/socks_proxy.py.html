   6 class Socks5Proxy(protocol.Layer, protocol.ServerConnectionMixin):
   7 
   8     def __call__(self):
   9         try:
  10             # Parse Client Greeting
  11             client_greet = socks.ClientGreeting.from_file(self.client_conn.rfile, fail_early=True)
  12             client_greet.assert_socks5()
  13             if socks.METHOD.NO_AUTHENTICATION_REQUIRED not in client_greet.methods:
  14                 raise socks.SocksError(
  15                     socks.METHOD.NO_ACCEPTABLE_METHODS,
  16                     "mitmproxy only supports SOCKS without authentication"
  17                 )
  18 
  19             # Send Server Greeting
  20             server_greet = socks.ServerGreeting(
  21                 socks.VERSION.SOCKS5,
  22                 socks.METHOD.NO_AUTHENTICATION_REQUIRED
  23             )
  24             server_greet.to_file(self.client_conn.wfile)
  25             self.client_conn.wfile.flush()
  26 
  27             # Parse Connect Request
  28             connect_request = socks.Message.from_file(self.client_conn.rfile)
  29             connect_request.assert_socks5()
  30             if connect_request.msg != socks.CMD.CONNECT:
  31                 raise socks.SocksError(
  32                     socks.REP.COMMAND_NOT_SUPPORTED,
  33                     "mitmproxy only supports SOCKS5 CONNECT"
  34                 )
  35 
  36             # We always connect lazily, but we need to pretend to the client that we connected.
  37             connect_reply = socks.Message(
  38                 socks.VERSION.SOCKS5,
  39                 socks.REP.SUCCEEDED,
  40                 connect_request.atyp,
  41                 # dummy value, we don't have an upstream connection yet.
  42                 connect_request.addr
  43             )
  44             connect_reply.to_file(self.client_conn.wfile)
  45             self.client_conn.wfile.flush()
  46 
  47         except (socks.SocksError, exceptions.TcpException) as e:
  48             raise exceptions.Socks5ProtocolException("SOCKS5 mode failure: %s" % repr(e))
  49 
  50         self.server_conn.address = connect_request.addr
  51 
  52         layer = self.ctx.next_layer(self)
  53         try:
  54             layer()
  55         finally:
  56             if self.server_conn.connected():
  57                 self.disconnect()
