  13 class HostMatcher:
  14     def __init__(self, handle, patterns=tuple()):
  15         self.handle = handle
  16         self.patterns = list(patterns)
  17         self.regexes = [re.compile(p, re.IGNORECASE) for p in self.patterns]
  18 
  19     def __call__(self, address):
  20         if not address:
  21             return False
  22         host = "%s:%s" % address
  23         if self.handle in ["ignore", "tcp"]:
  24             return any(rex.search(host) for rex in self.regexes)
  25         else:  # self.handle == "allow"
  26             return not any(rex.search(host) for rex in self.regexes)
  27 
  28     def __bool__(self):
  29         return bool(self.patterns)

  32 class ProxyConfig:
  33 
  34     def __init__(self, options: moptions.Options) -> None:
  35         self.options = options
  36 
  37         self.certstore: certs.CertStore
  38         self.check_filter: typing.Optional[HostMatcher] = None
  39         self.check_tcp: typing.Optional[HostMatcher] = None
  40         self.upstream_server: typing.Optional[server_spec.ServerSpec] = None
  41         self.configure(options, set(options.keys()))
  42         options.changed.connect(self.configure)
  43 
  44     def configure(self, options: moptions.Options, updated: typing.Any) -> None:
  45         if options.allow_hosts and options.ignore_hosts:
  46             raise exceptions.OptionsError("--ignore-hosts and --allow-hosts are mutually "
  47                                           "exclusive; please choose one.")
  48 
  49         if options.ignore_hosts:
  50             self.check_filter = HostMatcher("ignore", options.ignore_hosts)
  51         elif options.allow_hosts:
  52             self.check_filter = HostMatcher("allow", options.allow_hosts)
  53         else:
  54             self.check_filter = HostMatcher(False)
  55         if "tcp_hosts" in updated:
  56             self.check_tcp = HostMatcher("tcp", options.tcp_hosts)
  57 
  58         certstore_path = os.path.expanduser(options.confdir)
  59         if not os.path.exists(os.path.dirname(certstore_path)):
  60             raise exceptions.OptionsError(
  61                 "Certificate Authority parent directory does not exist: %s" %
  62                 os.path.dirname(certstore_path)
  63             )
  64         key_size = options.key_size
  65         self.certstore = certs.CertStore.from_store(
  66             certstore_path,
  67             moptions.CONF_BASENAME,
  68             key_size
  69         )
  70 
  71         for c in options.certs:
  72             parts = c.split("=", 1)
  73             if len(parts) == 1:
  74                 parts = ["*", parts[0]]
  75 
  76             cert = os.path.expanduser(parts[1])
  77             if not os.path.exists(cert):
  78                 raise exceptions.OptionsError(
  79                     "Certificate file does not exist: %s" % cert
  80                 )
  81             try:
  82                 self.certstore.add_cert_file(parts[0], cert)
  83             except crypto.Error:
  84                 raise exceptions.OptionsError(
  85                     "Invalid certificate format: %s" % cert
  86                 )
  87         m = options.mode
  88         if m.startswith("upstream:") or m.startswith("reverse:"):
  89             _, spec = server_spec.parse_with_mode(options.mode)
  90             self.upstream_server = spec
