   8

  48 def only(*types):
  49     def decorator(fn):
  50         @functools.wraps(fn)
  51         def filter_types(self, flow):
  52             if isinstance(flow, types):
  53                 return fn(self, flow)
  54             return False
  55 
  56         return filter_types
  57 
  58     return decorator

 146     def __init__(self, expr):

 156 def _check_content_type(rex, message):
 157     return any(
 158         name.lower() == b"content-type" and
 159         rex.search(value)
 160         for name, value in message.headers.fields
 161     )

 254 class FBod(_Rex):
 255     code = "b"
 256     help = "Body"
 257     flags = re.DOTALL
 258 
 259     @only(http.HTTPFlow, websocket.WebSocketFlow, tcp.TCPFlow)
 260     def __call__(self, f):
 261         if isinstance(f, http.HTTPFlow):
 262             if f.request and f.request.raw_content:
 263                 if self.re.search(f.request.get_content(strict=False)):
 264                     return True
 265             if f.response and f.response.raw_content:
 266                 if self.re.search(f.response.get_content(strict=False)):
 267                     return True
 268         elif isinstance(f, websocket.WebSocketFlow) or isinstance(f, tcp.TCPFlow):
 269             for msg in f.messages:
 270                 if self.re.search(msg.content):
 271                     return True
 272         return False

 469 def _make():
 470     # Order is important - multi-char expressions need to come before narrow
 471     # ones.
 472     parts = []
 473     for cls in filter_unary:
 474         f = pp.Literal(f"~{cls.code}") + pp.WordEnd()
 475         f.setParseAction(cls.make)
 476         parts.append(f)
 477 
 478     # This is a bit of a hack to simulate Word(pyparsing_unicode.printables),
 479     # which has a horrible performance with len(pyparsing.pyparsing_unicode.printables) == 1114060
 480     unicode_words = pp.CharsNotIn("()~'\"" + pp.ParserElement.DEFAULT_WHITE_CHARS)
 481     unicode_words.skipWhitespace = True
 482     regex = (
 483             unicode_words
 484             | pp.QuotedString('"', escChar='\\')
 485             | pp.QuotedString("'", escChar='\\')
 486     )
 487     for cls in filter_rex:
 488         f = pp.Literal(f"~{cls.code}") + pp.WordEnd() + regex.copy()
 489         f.setParseAction(cls.make)
 490         parts.append(f)
 491 
 492     for cls in filter_int:
 493         f = pp.Literal(f"~{cls.code}") + pp.WordEnd() + pp.Word(pp.nums)
 494         f.setParseAction(cls.make)
 495         parts.append(f)
 496 
 497     # A naked rex is a URL rex:
 498     f = regex.copy()
 499     f.setParseAction(FUrl.make)
 500     parts.append(f)
 501 
 502     atom = pp.MatchFirst(parts)
 503     expr = pp.infixNotation(
 504         atom,
 505         [(pp.Literal("!").suppress(),
 506           1,
 507           pp.opAssoc.RIGHT,
 508           lambda x: FNot(*x)),
 509          (pp.Literal("&").suppress(),
 510           2,
 511           pp.opAssoc.LEFT,
 512           lambda x: FAnd(*x)),
 513          (pp.Literal("|").suppress(),
 514           2,
 515           pp.opAssoc.LEFT,
 516           lambda x: FOr(*x)),
 517          ])
 518     expr = pp.OneOrMore(expr)
 519     return expr.setParseAction(lambda x: FAnd(x) if len(x) != 1 else x)

 526 def parse(s: str) -> Optional[TFilter]:
 527     try:
 528         flt = bnf.parseString(s, parseAll=True)[0]
 529         flt.pattern = s
 530         return flt
 531     except pp.ParseException:
 532         return None
 533     except ValueError:
 534         return None

 537 def match(flt, flow):
 538     """
 539         Matches a flow against a compiled filter expression.
 540         Returns True if matched, False if not.
 541 
 542         If flt is a string, it will be compiled as a filter expression.
 543         If the expression is invalid, ValueError is raised.
 544     """
 545     if isinstance(flt, str):
 546         flt = parse(flt)
 547         if not flt:
 548             raise ValueError("Invalid filter expression.")
 549     if flt:
 550         return flt(flow)
 551     return True
