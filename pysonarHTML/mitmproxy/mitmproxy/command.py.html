  16 def verify_arg_signature(f: typing.Callable, args: typing.Iterable[typing.Any], kwargs: dict) -> None:
  17     sig = inspect.signature(f)
  18     try:
  19         sig.bind(*args, **kwargs)
  20     except TypeError as v:
  21         raise exceptions.CommandError("command argument mismatch: %s" % v.args[0])

  24 def typename(t: type) -> str:
  25     """
  26         Translates a type to an explanatory string.
  27     """
  28     if t == inspect._empty:  # type: ignore
  29         raise exceptions.CommandError("missing type annotation")
  30     to = mitmproxy.types.CommandTypes.get(t, None)
  31     if not to:
  32         raise exceptions.CommandError("unsupported type: %s" % getattr(t, "__name__", t))
  33     return to.display

  36 def _empty_as_none(x: typing.Any) -> typing.Any:
  37     if x == inspect.Signature.empty:
  38         return None
  39     return x

  54 class Command:
  55     name: str
  56     manager: "CommandManager"
  57     signature: inspect.Signature
  58     help: typing.Optional[str]
  59 
  60     def __init__(self, manager: "CommandManager", name: str, func: typing.Callable) -> None:
  61         self.name = name
  62         self.manager = manager
  63         self.func = func
  64         self.signature = inspect.signature(self.func)
  65 
  66         if func.__doc__:
  67             txt = func.__doc__.strip()
  68             self.help = "\n".join(textwrap.wrap(txt))
  69         else:
  70             self.help = None
  71 
  72         # This fails with a CommandException if types are invalid
  73         for name, parameter in self.signature.parameters.items():
  74             t = parameter.annotation
  75             if not mitmproxy.types.CommandTypes.get(parameter.annotation, None):
  76                 raise exceptions.CommandError(f"Argument {name} has an unknown type ({_empty_as_none(t)}) in {func}.")
  77         if self.return_type and not mitmproxy.types.CommandTypes.get(self.return_type, None):
  78             raise exceptions.CommandError(f"Return type has an unknown type ({self.return_type}) in {func}.")
  79 
  80     @property
  81     def return_type(self) -> typing.Optional[typing.Type]:
  82         return _empty_as_none(self.signature.return_annotation)
  83 
  84     @property
  85     def parameters(self) -> typing.List[CommandParameter]:
  86         """Returns a list of CommandParameters."""
  87         ret = []
  88         for name, param in self.signature.parameters.items():
  89             ret.append(CommandParameter(name, param.annotation, param.kind))
  90         return ret
  91 
  92     def signature_help(self) -> str:
  93         params = " ".join(str(param) for param in self.parameters)
  94         if self.return_type:
  95             ret = f" -> {typename(self.return_type)}"
  96         else:
  97             ret = ""
  98         return f"{self.name} {params}{ret}"
  99 
 100     def prepare_args(self, args: typing.Sequence[str]) -> inspect.BoundArguments:
 101         try:
 102             bound_arguments = self.signature.bind(*args)
 103         except TypeError:
 104             expected = f'Expected: {str(self.signature.parameters)}'
 105             received = f'Received: {str(args)}'
 106             raise exceptions.CommandError(f"Command argument mismatch: \n    {expected}\n    {received}")
 107 
 108         for name, value in bound_arguments.arguments.items():
 109             convert_to = self.signature.parameters[name].annotation
 110             bound_arguments.arguments[name] = parsearg(self.manager, value, convert_to)
 111 
 112         bound_arguments.apply_defaults()
 113 
 114         return bound_arguments
 115 
 116     def call(self, args: typing.Sequence[str]) -> typing.Any:
 117         """
 118         Call the command with a list of arguments. At this point, all
 119         arguments are strings.
 120         """
 121         bound_args = self.prepare_args(args)
 122         ret = self.func(*bound_args.args, **bound_args.kwargs)
 123         if ret is None and self.return_type is None:
 124             return
 125         typ = mitmproxy.types.CommandTypes.get(self.return_type)
 126         assert typ
 127         if not typ.is_valid(self.manager, typ, ret):
 128             raise exceptions.CommandError(
 129                 f"{self.name} returned unexpected data - expected {typ.display}"
 130             )
 131         return ret

 147     def collect_commands(self, addon):
 148         for i in dir(addon):
 149             if not i.startswith("__"):
 150                 o = getattr(addon, i)
 151                 try:
 152                     # hasattr is not enough, see https://github.com/mitmproxy/mitmproxy/issues/3794
 153                     is_command = isinstance(getattr(o, "command_name", None), str)
 154                 except Exception:
 155                     pass  # getattr may raise if o implements __getattr__.
 156                 else:
 157                     if is_command:
 158                         try:
 159                             self.add(o.command_name, o)
 160                         except exceptions.CommandError as e:
 161                             self.master.log.warn(
 162                                 "Could not load command %s: %s" % (o.command_name, e)
 163                             )
 164 
 165     def add(self, path: str, func: typing.Callable):
 166         self.commands[path] = Command(self, path, func)

 168     @functools.lru_cache(maxsize=128)
 169     def parse_partial(
 170             self,
 171             cmdstr: str
 172     ) -> typing.Tuple[typing.Sequence[ParseResult], typing.Sequence[CommandParameter]]:
 173         """
 174         Parse a possibly partial command. Return a sequence of ParseResults and a sequence of remainder type help items.
 175         """
 176 
 177         parts: typing.List[str] = command_lexer.expr.parseString(cmdstr, parseAll=True)
 178 
 179         parsed: typing.List[ParseResult] = []
 180         next_params: typing.List[CommandParameter] = [
 181             CommandParameter("", mitmproxy.types.Cmd),
 182             CommandParameter("", mitmproxy.types.CmdArgs),
 183         ]
 184         expected: typing.Optional[CommandParameter] = None
 185         for part in parts:
 186             if part.isspace():
 187                 parsed.append(
 188                     ParseResult(
 189                         value=part,
 190                         type=mitmproxy.types.Space,
 191                         valid=True,
 192                     )
 193                 )
 194                 continue
 195 
 196             if expected and expected.kind is inspect.Parameter.VAR_POSITIONAL:
 197                 assert not next_params
 198             elif next_params:
 199                 expected = next_params.pop(0)
 200             else:
 201                 expected = CommandParameter("", mitmproxy.types.Unknown)
 202 
 203             arg_is_known_command = (
 204                     expected.type == mitmproxy.types.Cmd and part in self.commands
 205             )
 206             arg_is_unknown_command = (
 207                     expected.type == mitmproxy.types.Cmd and part not in self.commands
 208             )
 209             command_args_following = (
 210                     next_params and next_params[0].type == mitmproxy.types.CmdArgs
 211             )
 212             if arg_is_known_command and command_args_following:
 213                 next_params = self.commands[part].parameters + next_params[1:]
 214             if arg_is_unknown_command and command_args_following:
 215                 next_params.pop(0)
 216 
 217             to = mitmproxy.types.CommandTypes.get(expected.type, None)
 218             valid = False
 219             if to:
 220                 try:
 221                     to.parse(self, expected.type, part)
 222                 except exceptions.TypeError:
 223                     valid = False
 224                 else:
 225                     valid = True
 226 
 227             parsed.append(
 228                 ParseResult(
 229                     value=part,
 230                     type=expected.type,
 231                     valid=valid,
 232                 )
 233             )
 234 
 235         return parsed, next_params

 237     def call(self, command_name: str, *args: typing.Sequence[typing.Any]) -> typing.Any:
 238         """
 239         Call a command with native arguments. May raise CommandError.
 240         """
 241         if command_name not in self.commands:
 242             raise exceptions.CommandError("Unknown command: %s" % command_name)
 243         return self.commands[command_name].func(*args)
 244 
 245     def call_strings(self, command_name: str, args: typing.Sequence[str]) -> typing.Any:
 246         """
 247         Call a command using a list of string arguments. May raise CommandError.
 248         """
 249         if command_name not in self.commands:
 250             raise exceptions.CommandError("Unknown command: %s" % command_name)
 251 
 252         return self.commands[command_name].call(args)
 253 
 254     def execute(self, cmdstr: str) -> typing.Any:
 255         """
 256         Execute a command string. May raise CommandError.
 257         """
 258         parts, _ = self.parse_partial(cmdstr)
 259         if not parts:
 260             raise exceptions.CommandError(f"Invalid command: {cmdstr!r}")
 261         command_name, *args = [
 262             unquote(part.value)
 263             for part in parts
 264             if part.type != mitmproxy.types.Space
 265         ]
 266         return self.call_strings(command_name, args)
 267 
 268     def dump(self, out=sys.stdout) -> None:
 269         cmds = list(self.commands.values())
 270         cmds.sort(key=lambda x: x.signature_help())
 271         for c in cmds:
 272             for hl in (c.help or "").splitlines():
 273                 print("# " + hl, file=out)
 274             print(c.signature_help(), file=out)
 275             print(file=out)

 278 def parsearg(manager: CommandManager, spec: str, argtype: type) -> typing.Any:
 279     """
 280         Convert a string to a argument to the appropriate type.
 281     """
 282     t = mitmproxy.types.CommandTypes.get(argtype, None)
 283     if not t:
 284         raise exceptions.CommandError(f"Unsupported argument type: {argtype}")
 285     try:
 286         return t.parse(manager, argtype, spec)
 287     except exceptions.TypeError as e:
 288         raise exceptions.CommandError(str(e)) from e

 291 def command(name: typing.Optional[str] = None):
 292     def decorator(function):
 293         @functools.wraps(function)
 294         def wrapper(*args, **kwargs):
 295             verify_arg_signature(function, args, kwargs)
 296             return function(*args, **kwargs)
 297 
 298         wrapper.__dict__["command_name"] = name or function.__name__.replace("_", ".")
 299         return wrapper
 300 
 301     return decorator

 304 def argument(name, type):
 305     """
 306         Set the type of a command argument at runtime. This is useful for more
 307         specific types such as mitmproxy.types.Choice, which we cannot annotate
 308         directly as mypy does not like that.
 309     """
 310 
 311     def decorator(f: types.FunctionType) -> types.FunctionType:
 312         assert name in f.__annotations__
 313         f.__annotations__[name] = type
 314         return f
 315 
 316     return decorator
