  11 class Error(stateobject.StateObject):
  12 
  13     """
  14         An Error.
  15 
  16         This is distinct from an protocol error response (say, a HTTP code 500),
  17         which is represented by a normal HTTPResponse object. This class is
  18         responsible for indicating errors that fall outside of normal protocol
  19         communications, like interrupted connections, timeouts, protocol errors.
  20 
  21         Exposes the following attributes:
  22 
  23             msg: Message describing the error
  24             timestamp: Seconds since the epoch
  25     """
  26 
  27     def __init__(self, msg: str, timestamp=None) -> None:
  28         """
  29         @type msg: str
  30         @type timestamp: float
  31         """
  32         self.msg = msg
  33         self.timestamp = timestamp or time.time()
  34 
  35     _stateobject_attributes = dict(
  36         msg=str,
  37         timestamp=float
  38     )
  39 
  40     def __str__(self):
  41         return self.msg
  42 
  43     def __repr__(self):
  44         return self.msg
  45 
  46     @classmethod
  47     def from_state(cls, state):
  48         # the default implementation assumes an empty constructor. Override
  49         # accordingly.
  50         f = cls(None)
  51         f.set_state(state)
  52         return f

  55 class Flow(stateobject.StateObject):
  56 
  57     """
  58     A Flow is a collection of objects representing a single transaction.
  59     This class is usually subclassed for each protocol, e.g. HTTPFlow.
  60     """
  61 
  62     def __init__(
  63             self,
  64             type: str,
  65             client_conn: connections.ClientConnection,
  66             server_conn: connections.ServerConnection,
  67             live: bool=None
  68     ) -> None:
  69         self.type = type
  70         self.id = str(uuid.uuid4())
  71         self.client_conn = client_conn
  72         self.server_conn = server_conn
  73         self.live = live
  74 
  75         self.error: typing.Optional[Error] = None
  76         self.intercepted: bool = False
  77         self._backup: typing.Optional[Flow] = None
  78         self.reply: typing.Optional[controller.Reply] = None
  79         self.marked: bool = False
  80         self.metadata: typing.Dict[str, typing.Any] = dict()
  81 
  82     _stateobject_attributes = dict(
  83         id=str,
  84         error=Error,
  85         client_conn=connections.ClientConnection,
  86         server_conn=connections.ServerConnection,
  87         type=str,
  88         intercepted=bool,
  89         marked=bool,
  90         metadata=typing.Dict[str, typing.Any],
  91     )
  92 
  93     def get_state(self):
  94         d = super().get_state()
  95         d.update(version=version.FLOW_FORMAT_VERSION)
  96         if self._backup and self._backup != d:
  97             d.update(backup=self._backup)
  98         return d
  99 
 100     def set_state(self, state):
 101         state = state.copy()
 102         state.pop("version")
 103         if "backup" in state:
 104             self._backup = state.pop("backup")
 105         super().set_state(state)
 106 
 107     @classmethod
 108     def from_state(cls, state):
 109         f = cls(None, None)
 110         f.set_state(state)
 111         return f
 112 
 113     def copy(self):
 114         f = super().copy()
 115         f.live = False
 116         if self.reply is not None:
 117             f.reply = controller.DummyReply()
 118         return f
 119 
 120     def modified(self):
 121         """
 122             Has this Flow been modified?
 123         """
 124         if self._backup:
 125             return self._backup != self.get_state()
 126         else:
 127             return False
 128 
 129     def backup(self, force=False):
 130         """
 131             Save a backup of this Flow, which can be reverted to using a
 132             call to .revert().
 133         """
 134         if not self._backup:
 135             self._backup = self.get_state()
 136 
 137     def revert(self):
 138         """
 139             Revert to the last backed up state.
 140         """
 141         if self._backup:
 142             self.set_state(self._backup)
 143             self._backup = None
 144 
 145     @property
 146     def killable(self):
 147         return (
 148             self.reply and
 149             self.reply.state in {"start", "taken"} and
 150             self.reply.value != exceptions.Kill
 151         )
 152 
 153     def kill(self):
 154         """
 155             Kill this request.
 156         """
 157         self.error = Error("Connection killed")
 158         self.intercepted = False
 159         self.reply.kill(force=True)
 160         self.live = False
 161 
 162     def intercept(self):
 163         """
 164             Intercept this Flow. Processing will stop until resume is
 165             called.
 166         """
 167         if self.intercepted:
 168             return
 169         self.intercepted = True
 170         self.reply.take()
 171 
 172     def resume(self):
 173         """
 174             Continue with the flow - called after an intercept().
 175         """
 176         if not self.intercepted:
 177             return
 178         self.intercepted = False
 179         # If a flow is intercepted and then duplicated, the duplicated one is not taken.
 180         if self.reply.state == "taken":
 181             self.reply.ack()
 182             self.reply.commit()
 183 
 184     @property
 185     def timestamp_start(self) -> float:
 186         """Start time of the flow."""
 187         return self.client_conn.timestamp_start
