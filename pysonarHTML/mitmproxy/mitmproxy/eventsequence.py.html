  42 TEventGenerator = typing.Iterator[typing.Tuple[str, typing.Any]]
  43 
  44 
  45 def _iterate_http(f: http.HTTPFlow) -> TEventGenerator:
  46     if f.request:
  47         yield "requestheaders", f
  48         yield "request", f
  49     if f.response:
  50         yield "responseheaders", f
  51         yield "response", f
  52     if f.error:
  53         yield "error", f
  54 
  55 
  56 def _iterate_websocket(f: websocket.WebSocketFlow) -> TEventGenerator:
  57     messages = f.messages
  58     f.messages = []
  59     f.reply = controller.DummyReply()
  60     yield "websocket_start", f
  61     while messages:
  62         f.messages.append(messages.pop(0))
  63         yield "websocket_message", f
  64     if f.error:
  65         yield "websocket_error", f
  66     yield "websocket_end", f
  67 
  68 
  69 def _iterate_tcp(f: tcp.TCPFlow) -> TEventGenerator:
  70     messages = f.messages
  71     f.messages = []
  72     f.reply = controller.DummyReply()
  73     yield "tcp_start", f
  74     while messages:
  75         f.messages.append(messages.pop(0))
  76         yield "tcp_message", f
  77     if f.error:
  78         yield "tcp_error", f
  79     yield "tcp_end", f

  82 _iterate_map: typing.Dict[typing.Type[flow.Flow], typing.Callable[[typing.Any], TEventGenerator]] = {
  83     http.HTTPFlow: _iterate_http,
  84     websocket.WebSocketFlow: _iterate_websocket,
  85     tcp.TCPFlow: _iterate_tcp,

  89 def iterate(f: flow.Flow) -> TEventGenerator:
  90     try:
  91         e = _iterate_map[type(f)]
  92     except KeyError as err:
  93         raise TypeError("Unknown flow type: {}".format(f)) from err
  94     else:
  95         yield from e(f)
