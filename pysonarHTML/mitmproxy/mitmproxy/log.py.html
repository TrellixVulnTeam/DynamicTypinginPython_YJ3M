   4 class LogEntry:
   5     def __init__(self, msg, level):
   6         self.msg = msg
   7         self.level = level
   8 
   9     def __eq__(self, other):
  10         if isinstance(other, LogEntry):
  11             return self.__dict__ == other.__dict__
  12         return False
  13 
  14     def __repr__(self):
  15         return "LogEntry({}, {})".format(self.msg, self.level)

  18 class Log:
  19     """
  20         The central logger, exposed to scripts as mitmproxy.ctx.log.
  21     """
  22     def __init__(self, master):
  23         self.master = master
  24 
  25     def debug(self, txt):
  26         """
  27             Log with level debug.
  28         """
  29         self(txt, "debug")
  30 
  31     def info(self, txt):
  32         """
  33             Log with level info.
  34         """
  35         self(txt, "info")
  36 
  37     def alert(self, txt):
  38         """
  39             Log with level alert. Alerts have the same urgency as info, but
  40             signals to interactive tools that the user's attention should be
  41             drawn to the output even if they're not currently looking at the
  42             event log.
  43         """
  44         self(txt, "alert")
  45 
  46     def warn(self, txt):
  47         """
  48             Log with level warn.
  49         """
  50         self(txt, "warn")
  51 
  52     def error(self, txt):
  53         """
  54             Log with level error.
  55         """
  56         self(txt, "error")
  57 
  58     def __call__(self, text, level="info"):
  59         asyncio.get_event_loop().call_soon(
  60             self.master.addons.trigger, "log", LogEntry(text, level)
  61         )
