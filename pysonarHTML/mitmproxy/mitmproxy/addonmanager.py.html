  19 def cut_traceback(tb, func_name):
  20     """
  21     Cut off a traceback at the function with the given name.
  22     The func_name's frame is excluded.
  23 
  24     Args:
  25         tb: traceback object, as returned by sys.exc_info()[2]
  26         func_name: function name
  27 
  28     Returns:
  29         Reduced traceback.
  30     """
  31     tb_orig = tb
  32     for _, _, fname, _ in traceback.extract_tb(tb):
  33         tb = tb.tb_next
  34         if fname == func_name:
  35             break
  36     return tb or tb_orig

  39 @contextlib.contextmanager
  40 def safecall():
  41     try:
  42         yield
  43     except (exceptions.AddonHalt, exceptions.OptionsError):
  44         raise
  45     except Exception:
  46         etype, value, tb = sys.exc_info()
  47         tb = cut_traceback(tb, "invoke_addon")
  48         ctx.log.error(
  49             "Addon error: %s" % "".join(
  50                 traceback.format_exception(etype, value, tb)
  51             )
  52         )

  62     def add_option(
  63         self,
  64         name: str,
  65         typespec: type,
  66         default: typing.Any,
  67         help: str,
  68         choices: typing.Optional[typing.Sequence[str]] = None
  69     ) -> None:
  70         """
  71             Add an option to mitmproxy.
  72 
  73             Help should be a single paragraph with no linebreaks - it will be
  74             reflowed by tools. Information on the data type should be omitted -
  75             it will be generated and added by tools as needed.
  76         """
  77         if name in self.master.options:
  78             existing = self.master.options._options[name]
  79             same_signature = (
  80                 existing.name == name and
  81                 existing.typespec == typespec and
  82                 existing.default == default and
  83                 existing.help == help and
  84                 existing.choices == choices
  85             )
  86             if same_signature:
  87                 return
  88             else:
  89                 ctx.log.warn("Over-riding existing option %s" % name)
  90         self.master.options.add_option(
  91             name,
  92             typespec,
  93             default,
  94             help,
  95             choices
  96         )

 102 def traverse(chain):
 103     """
 104         Recursively traverse an addon chain.
 105     """
 106     for a in chain:
 107         yield a
 108         if hasattr(a, "addons"):
 109             yield from traverse(a.addons)

 112 class AddonManager:
 113     def __init__(self, master):
 114         self.lookup = {}
 115         self.chain = []
 116         self.master = master
 117         master.options.changed.connect(self._configure_all)
 118 
 119     def _configure_all(self, options, updated):
 120         self.trigger("configure", updated)
 121 
 122     def clear(self):
 123         """
 124             Remove all addons.
 125         """
 126         for a in self.chain:
 127             self.invoke_addon(a, "done")
 128         self.lookup = {}
 129         self.chain = []
 130 
 131     def get(self, name):
 132         """
 133             Retrieve an addon by name. Addon names are equal to the .name
 134             attribute on the instance, or the lower case class name if that
 135             does not exist.
 136         """
 137         return self.lookup.get(name, None)
 138 
 139     def register(self, addon):
 140         """
 141             Register an addon, call its load event, and then register all its
 142             sub-addons. This should be used by addons that dynamically manage
 143             addons.
 144 
 145             If the calling addon is already running, it should follow with
 146             running and configure events. Must be called within a current
 147             context.
 148         """
 149         for a in traverse([addon]):
 150             name = _get_name(a)
 151             if name in self.lookup:
 152                 raise exceptions.AddonManagerError(
 153                     "An addon called '%s' already exists." % name
 154                 )
 155         l = Loader(self.master)
 156         self.invoke_addon(addon, "load", l)
 157         for a in traverse([addon]):
 158             name = _get_name(a)
 159             self.lookup[name] = a
 160         for a in traverse([addon]):
 161             self.master.commands.collect_commands(a)
 162         self.master.options.process_deferred()
 163         return addon
 164 
 165     def add(self, *addons):
 166         """
 167             Add addons to the end of the chain, and run their load event.
 168             If any addon has sub-addons, they are registered.
 169         """
 170         for i in addons:
 171             self.chain.append(self.register(i))
 172 
 173     def remove(self, addon):
 174         """
 175             Remove an addon and all its sub-addons.
 176 
 177             If the addon is not in the chain - that is, if it's managed by a
 178             parent addon - it's the parent's responsibility to remove it from
 179             its own addons attribute.
 180         """
 181         for a in traverse([addon]):
 182             n = _get_name(a)
 183             if n not in self.lookup:
 184                 raise exceptions.AddonManagerError("No such addon: %s" % n)
 185             self.chain = [i for i in self.chain if i is not a]
 186             del self.lookup[_get_name(a)]
 187         self.invoke_addon(addon, "done")
 188 
 189     def __len__(self):
 190         return len(self.chain)
 191 
 192     def __str__(self):
 193         return pprint.pformat([str(i) for i in self.chain])
 194 
 195     def __contains__(self, item):
 196         name = _get_name(item)
 197         return name in self.lookup

 199     async def handle_lifecycle(self, name, message):
 200         """
 201             Handle a lifecycle event.
 202         """
 203         if not hasattr(message, "reply"):  # pragma: no cover
 204             raise exceptions.ControlException(
 205                 "Message %s has no reply attribute" % message
 206             )
 207 
 208         # We can use DummyReply objects multiple times. We only clear them up on
 209         # the next handler so that we can access value and state in the
 210         # meantime.
 211         if isinstance(message.reply, controller.DummyReply):
 212             message.reply.reset()
 213 
 214         self.trigger(name, message)
 215 
 216         if message.reply.state == "start":
 217             message.reply.take()
 218             if not message.reply.has_message:
 219                 message.reply.ack()
 220             message.reply.commit()
 221 
 222             if isinstance(message.reply, controller.DummyReply):
 223                 message.reply.mark_reset()
 224 
 225         if isinstance(message, flow.Flow):
 226             self.trigger("update", [message])
 227 
 228     def invoke_addon(self, addon, name, *args, **kwargs):
 229         """
 230             Invoke an event on an addon and all its children.
 231         """
 232         if name not in eventsequence.Events:
 233             raise exceptions.AddonManagerError("Unknown event: %s" % name)
 234         for a in traverse([addon]):
 235             func = getattr(a, name, None)
 236             if func:
 237                 if callable(func):
 238                     func(*args, **kwargs)
 239                 elif isinstance(func, types.ModuleType):
 240                     # we gracefully exclude module imports with the same name as hooks.
 241                     # For example, a user may have "from mitmproxy import log" in an addon,
 242                     # which has the same name as the "log" hook. In this particular case,
 243                     # we end up in an error loop because we "log" this error.
 244                     pass
 245                 else:
 246                     raise exceptions.AddonManagerError(
 247                         "Addon handler {} ({}) not callable".format(name, a)
 248                     )
 249 
 250     def trigger(self, name, *args, **kwargs):
 251         """
 252             Trigger an event across all addons.
 253         """
 254         for i in self.chain:
 255             try:
 256                 with safecall():
 257                     self.invoke_addon(i, name, *args, **kwargs)
 258             except exceptions.AddonHalt:
 259                 return
