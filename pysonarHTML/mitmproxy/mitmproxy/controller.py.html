   6 class Channel:
   7     """
   8         The only way for the proxy server to communicate with the master
   9         is to use the channel it has been given.
  10     """
  11     def __init__(self, master, loop, should_exit):
  12         self.master = master
  13         self.loop = loop
  14         self.should_exit = should_exit
  15 
  16     def ask(self, mtype, m):
  17         """
  18         Decorate a message with a reply attribute, and send it to the master.
  19         Then wait for a response.
  20 
  21         Raises:
  22             exceptions.Kill: All connections should be closed immediately.
  23         """
  24         if not self.should_exit.is_set():
  25             m.reply = Reply(m)
  26             asyncio.run_coroutine_threadsafe(
  27                 self.master.addons.handle_lifecycle(mtype, m),
  28                 self.loop,
  29             )
  30             g = m.reply.q.get()
  31             if g == exceptions.Kill:
  32                 raise exceptions.Kill()
  33             return g
  34 
  35     def tell(self, mtype, m):
  36         """
  37         Decorate a message with a dummy reply attribute, send it to the master,
  38         then return immediately.
  39         """
  40         if not self.should_exit.is_set():
  41             m.reply = DummyReply()
  42             asyncio.run_coroutine_threadsafe(
  43                 self.master.addons.handle_lifecycle(mtype, m),
  44                 self.loop,
  45             )

  48 NO_REPLY = object()  # special object we can distinguish from a valid "None" reply.

  51 class Reply:
  52     """
  53     Messages sent through a channel are decorated with a "reply" attribute. This
  54     object is used to respond to the message through the return channel.
  55     """
  56     def __init__(self, obj):
  57         self.obj = obj
  58         # Spawn an event loop in the current thread
  59         self.q = queue.Queue()
  60 
  61         self._state = "start"  # "start" -> "taken" -> "committed"
  62 
  63         # Holds the reply value. May change before things are actually committed.
  64         self.value = NO_REPLY
  65 
  66     @property
  67     def state(self):
  68         """
  69         The state the reply is currently in. A normal reply object goes
  70         sequentially through the following lifecycle:
  71 
  72             1. start: Initial State.
  73             2. taken: The reply object has been taken to be committed.
  74             3. committed: The reply has been sent back to the requesting party.
  75 
  76         This attribute is read-only and can only be modified by calling one of
  77         state transition functions.
  78         """
  79         return self._state
  80 
  81     @property
  82     def has_message(self):
  83         return self.value != NO_REPLY
  84 
  85     def take(self):
  86         """
  87         Scripts or other parties make "take" a reply out of a normal flow.
  88         For example, intercepted flows are taken out so that the connection thread does not proceed.
  89         """
  90         if self.state != "start":
  91             raise exceptions.ControlException(
  92                 "Reply is {}, but expected it to be start.".format(self.state)
  93             )
  94         self._state = "taken"
  95 
  96     def commit(self):
  97         """
  98         Ultimately, messages are committed. This is done either automatically by
  99         the handler if the message is not taken or manually by the entity which
 100         called .take().
 101         """
 102         if self.state != "taken":
 103             raise exceptions.ControlException(
 104                 "Reply is {}, but expected it to be taken.".format(self.state)
 105             )
 106         if not self.has_message:
 107             raise exceptions.ControlException("There is no reply message.")
 108         self._state = "committed"
 109         self.q.put(self.value)
 110 
 111     def ack(self, force=False):
 112         self.send(self.obj, force)
 113 
 114     def kill(self, force=False):
 115         self.send(exceptions.Kill, force)
 116         if self._state == "taken":
 117             self.commit()
 118 
 119     def send(self, msg, force=False):
 120         if self.state not in {"start", "taken"}:
 121             raise exceptions.ControlException(
 122                 "Reply is {}, but expected it to be start or taken.".format(self.state)
 123             )
 124         if self.has_message and not force:
 125             raise exceptions.ControlException("There is already a reply message.")
 126         self.value = msg
 127 
 128     def __del__(self):
 129         if self.state != "committed":
 130             # This will be ignored by the interpreter, but emit a warning
 131             raise exceptions.ControlException("Uncommitted reply: %s" % self.obj)

 134 class DummyReply(Reply):
 135     """
 136     A reply object that is not connected to anything. In contrast to regular
 137     Reply objects, DummyReply objects are reset to "start" at the end of an
 138     handler so that they can be used multiple times. Useful when we need an
 139     object to seem like it has a channel, and during testing.
 140     """
 141     def __init__(self):
 142         super().__init__(None)
 143         self._should_reset = False
 144 
 145     def mark_reset(self):
 146         if self.state != "committed":
 147             raise exceptions.ControlException("Uncommitted reply: %s" % self.obj)
 148         self._should_reset = True
 149 
 150     def reset(self):
 151         if self._should_reset:
 152             self._state = "start"
 153             self.value = NO_REPLY
 154 
 155     def __del__(self):
 156         pass
