  11 def _move_attrs(s_obj, d_obj, attrs):
  12     for attr in attrs:
  13         if not isinstance(d_obj, dict):
  14             if hasattr(s_obj, attr) and getattr(s_obj, attr) is not None:
  15                 setattr(d_obj, attr, getattr(s_obj, attr))
  16         else:
  17             if hasattr(s_obj, attr) and getattr(s_obj, attr) is not None:
  18                 # ugly fix to set None in empty str or bytes fields
  19                 if getattr(s_obj, attr) == "" or getattr(s_obj, attr) == b"":
  20                     d_obj[attr] = None
  21                 else:
  22                     d_obj[attr] = getattr(s_obj, attr)

  25 def _dump_http_response(res: HTTPResponse) -> http_pb2.HTTPResponse:
  26     pres = http_pb2.HTTPResponse()
  27     _move_attrs(res, pres, ['http_version', 'status_code', 'reason',
  28                             'content', 'timestamp_start', 'timestamp_end', 'is_replay'])
  29     if res.headers:
  30         for h in res.headers.fields:
  31             header = pres.headers.add()
  32             header.name = h[0]
  33             header.value = h[1]
  34     return pres

  49 def _dump_http_client_conn(cc: ClientConnection) -> http_pb2.ClientConnection:
  50     pcc = http_pb2.ClientConnection()
  51     _move_attrs(cc, pcc, ['id', 'tls_established', 'timestamp_start', 'timestamp_tls_setup', 'timestamp_end', 'sni',
  52                           'cipher_name', 'alpn_proto_negotiated', 'tls_version'])
  53     for cert in ['clientcert', 'mitmcert']:
  54         if hasattr(cc, cert) and getattr(cc, cert) is not None:
  55             setattr(pcc, cert, getattr(cc, cert).to_pem())
  56     if cc.tls_extensions:
  57         for extension in cc.tls_extensions:
  58             ext = pcc.tls_extensions.add()
  59             ext.int = extension[0]
  60             ext.bytes = extension[1]
  61     if cc.address:
  62         pcc.address.host = cc.address[0]
  63         pcc.address.port = cc.address[1]
  64     return pcc

  67 def _dump_http_server_conn(sc: ServerConnection) -> http_pb2.ServerConnection:
  68     psc = http_pb2.ServerConnection()
  69     _move_attrs(sc, psc, ['id', 'tls_established', 'sni', 'alpn_proto_negotiated', 'tls_version',
  70                           'timestamp_start', 'timestamp_tcp_setup', 'timestamp_tls_setup', 'timestamp_end'])
  71     for addr in ['address', 'ip_address', 'source_address']:
  72         if hasattr(sc, addr) and getattr(sc, addr) is not None:
  73             getattr(psc, addr).host = getattr(sc, addr)[0]
  74             getattr(psc, addr).port = getattr(sc, addr)[1]
  75     if sc.cert:
  76         psc.cert = sc.cert.to_pem()
  77     if sc.via:
  78         psc.via.MergeFrom(_dump_http_server_conn(sc.via))
  79     return psc

  90 def dump_http(f: flow.Flow) -> http_pb2.HTTPFlow:
  91     pf = http_pb2.HTTPFlow()
  92     for p in ['request', 'response', 'client_conn', 'server_conn', 'error']:
  93         if hasattr(f, p) and getattr(f, p):
  94             getattr(pf, p).MergeFrom(eval(f"_dump_http_{p}")(getattr(f, p)))
  95     _move_attrs(f, pf, ['intercepted', 'marked', 'mode', 'id'])
  96     return pf

  99 def dumps(f: flow.Flow) -> bytes:
 100     if f.type != "http":
 101         raise exceptions.TypeError("Flow types different than HTTP not supported yet!")
 102     else:
 103         p = dump_http(f)
 104         return p.SerializeToString()

 107 def _load_http_request(o: http_pb2.HTTPRequest) -> HTTPRequest:
 108     d: dict = {}
 109     _move_attrs(o, d, ['first_line_format', 'method', 'scheme', 'host', 'port', 'path', 'http_version', 'content',
 110                        'timestamp_start', 'timestamp_end', 'is_replay'])
 111     if d['content'] is None:
 112         d['content'] = b""
 113     d["headers"] = []
 114     for header in o.headers:
 115         d["headers"].append((bytes(header.name, "utf-8"), bytes(header.value, "utf-8")))
 116 
 117     return HTTPRequest(**d)

 133 def _load_http_client_conn(o: http_pb2.ClientConnection) -> ClientConnection:
 134     d: dict = {}
 135     _move_attrs(o, d, ['id', 'tls_established', 'sni', 'cipher_name', 'alpn_proto_negotiated', 'tls_version',
 136                        'timestamp_start', 'timestamp_tcp_setup', 'timestamp_tls_setup', 'timestamp_end'])
 137     for cert in ['clientcert', 'mitmcert']:
 138         if hasattr(o, cert) and getattr(o, cert):
 139             d[cert] = Cert.from_pem(getattr(o, cert))
 140     if o.tls_extensions:
 141         d['tls_extensions'] = []
 142         for extension in o.tls_extensions:
 143             d['tls_extensions'].append((extension.int, extension.bytes))
 144     if o.address:
 145         d['address'] = (o.address.host, o.address.port)
 146     cc = ClientConnection(None, tuple(), None)
 147     for k, v in d.items():
 148         setattr(cc, k, v)
 149     return cc

 152 def _load_http_server_conn(o: http_pb2.ServerConnection) -> ServerConnection:
 153     d: dict = {}
 154     _move_attrs(o, d, ['id', 'tls_established', 'sni', 'alpn_proto_negotiated', 'tls_version',
 155                        'timestamp_start', 'timestamp_tcp_setup', 'timestamp_tls_setup', 'timestamp_end'])
 156     for addr in ['address', 'ip_address', 'source_address']:
 157         if hasattr(o, addr):
 158             d[addr] = (getattr(o, addr).host, getattr(o, addr).port)
 159     if o.cert:
 160         c = Cert.from_pem(o.cert)
 161         d['cert'] = c
 162     if o.HasField('via'):
 163         d['via'] = _load_http_server_conn(o.via)
 164     sc = ServerConnection(tuple())
 165     for k, v in d.items():
 166         setattr(sc, k, v)
 167     return sc

 170 def _load_http_error(o: http_pb2.HTTPError) -> typing.Optional[flow.Error]:
 171     d = {}
 172     for m in ['msg', 'timestamp']:
 173         if hasattr(o, m) and getattr(o, m):
 174             d[m] = getattr(o, m)
 175     return None if not d else flow.Error(**d)

 178 def load_http(hf: http_pb2.HTTPFlow) -> HTTPFlow:
 179     parts = {}
 180     for p in ['request', 'response', 'client_conn', 'server_conn', 'error']:
 181         if hf.HasField(p):
 182             parts[p] = eval(f"_load_http_{p}")(getattr(hf, p))
 183         else:
 184             parts[p] = None
 185     _move_attrs(hf, parts, ['intercepted', 'marked', 'mode', 'id'])
 186     f = HTTPFlow(ClientConnection(None, tuple(), None), ServerConnection(tuple()))
 187     for k, v in parts.items():
 188         setattr(f, k, v)
 189     return f

 192 def loads(b: bytes, typ="http") -> typing.Union[HTTPFlow]:
 193     if typ != 'http':
 194         raise exceptions.TypeError("Flow types different than HTTP not supported yet!")
 195     else:
 196         p = http_pb2.HTTPFlow()
 197         p.ParseFromString(b)
 198         return load_http(p)
