   1 """
   2 tnetstring:  data serialization using typed netstrings
   3 ======================================================
   4 
   5 This is a custom Python 3 implementation of tnetstrings.
   6 Compared to other implementations, the main difference
   7 is that this implementation supports a custom unicode datatype.
   8 
   9 An ordinary tnetstring is a blob of data prefixed with its length and postfixed
  10 with its type. Here are some examples:
  11 
  12     >>> tnetstring.dumps("hello world")
  13     11:hello world,
  14     >>> tnetstring.dumps(12345)
  15     5:12345#
  16     >>> tnetstring.dumps([12345, True, 0])
  17     19:5:12345#4:true!1:0#]
  18 
  19 This module gives you the following functions:
  20 
  21     :dump:    dump an object as a tnetstring to a file
  22     :dumps:   dump an object as a tnetstring to a string
  23     :load:    load a tnetstring-encoded object from a file
  24     :loads:   load a tnetstring-encoded object from a string
  25 
  26 Note that since parsing a tnetstring requires reading all the data into memory
  27 at once, there's no efficiency gain from using the file-based versions of these
  28 functions.  They're only here so you can use load() to read precisely one
  29 item from a file or socket without consuming any extra data.
  30 
  31 The tnetstrings specification explicitly states that strings are binary blobs
  32 and forbids the use of unicode at the protocol level.
  33 **This implementation decodes dictionary keys as surrogate-escaped ASCII**,
  34 all other strings are returned as plain bytes.
  35

  37 :Copyright: (c) 2014 by Carlo Pires <carlopires@gmail.com>.
  38 :Copyright: (c) 2016 by Maximilian Hils <tnetstring3@maximilianhils.com>.
  39 
  40 :License: MIT

  49 def dumps(value: TSerializable) -> bytes:
  50     """
  51     This function dumps a python object as a tnetstring.
  52     """
  53     #  This uses a deque to collect output fragments in reverse order,
  54     #  then joins them together at the end.  It's measurably faster
  55     #  than creating all the intermediate strings.
  56     q: collections.deque = collections.deque()
  57     _rdumpq(q, 0, value)
  58     return b''.join(q)
  59

  61 def dump(value: TSerializable, file_handle: typing.BinaryIO) -> None:
  62     """
  63     This function dumps a python object as a tnetstring and
  64     writes it to the given file.
  65     """
  66     file_handle.write(dumps(value))

  69 def _rdumpq(q: collections.deque, size: int, value: TSerializable) -> int:
  70     """
  71     Dump value as a tnetstring, to a deque instance, last chunks first.
  72 
  73     This function generates the tnetstring representation of the given value,
  74     pushing chunks of the output onto the given deque instance.  It pushes
  75     the last chunk first, then recursively generates more chunks.
  76 
  77     When passed in the current size of the string in the queue, it will return
  78     the new size of the string in the queue.
  79 
  80     Operating last-chunk-first makes it easy to calculate the size written
  81     for recursive structures without having to build their representation as
  82     a string.  This is measurably faster than generating the intermediate
  83     strings, especially on deeply nested structures.
  84     """
  85     write = q.appendleft
  86     if value is None:
  87         write(b'0:~')
  88         return size + 3
  89     elif value is True:
  90         write(b'4:true!')
  91         return size + 7
  92     elif value is False:
  93         write(b'5:false!')
  94         return size + 8
  95     elif isinstance(value, int):
  96         data = str(value).encode()
  97         ldata = len(data)
  98         span = str(ldata).encode()
  99         write(b'%s:%s#' % (span, data))
 100         return size + 2 + len(span) + ldata
 101     elif isinstance(value, float):
 102         #  Use repr() for float rather than str().
 103         #  It round-trips more accurately.
 104         #  Probably unnecessary in later python versions that
 105         #  use David Gay's ftoa routines.
 106         data = repr(value).encode()
 107         ldata = len(data)
 108         span = str(ldata).encode()
 109         write(b'%s:%s^' % (span, data))
 110         return size + 2 + len(span) + ldata
 111     elif isinstance(value, bytes):
 112         data = value
 113         ldata = len(data)
 114         span = str(ldata).encode()
 115         write(b',')
 116         write(data)
 117         write(b':')
 118         write(span)
 119         return size + 2 + len(span) + ldata
 120     elif isinstance(value, str):
 121         data = value.encode("utf8")
 122         ldata = len(data)
 123         span = str(ldata).encode()
 124         write(b';')
 125         write(data)
 126         write(b':')
 127         write(span)
 128         return size + 2 + len(span) + ldata
 129     elif isinstance(value, (list, tuple)):
 130         write(b']')
 131         init_size = size = size + 1
 132         for item in reversed(value):
 133             size = _rdumpq(q, size, item)
 134         span = str(size - init_size).encode()
 135         write(b':')
 136         write(span)
 137         return size + 1 + len(span)
 138     elif isinstance(value, dict):
 139         write(b'}')
 140         init_size = size = size + 1
 141         for (k, v) in value.items():
 142             size = _rdumpq(q, size, v)
 143             size = _rdumpq(q, size, k)
 144         span = str(size - init_size).encode()
 145         write(b':')
 146         write(span)
 147         return size + 1 + len(span)
 148     else:
 149         raise ValueError("unserializable object: {} ({})".format(value, type(value)))

 152 def loads(string: bytes) -> TSerializable:
 153     """
 154     This function parses a tnetstring into a python object.
 155     """
 156     return pop(string)[0]

 159 def load(file_handle: typing.BinaryIO) -> TSerializable:
 160     """load(file) -> object
 161 
 162     This function reads a tnetstring from a file and parses it into a
 163     python object.  The file must support the read() method, and this
 164     function promises not to read more data than necessary.
 165     """
 166     #  Read the length prefix one char at a time.
 167     #  Note that the netstring spec explicitly forbids padding zeros.
 168     c = file_handle.read(1)
 169     if c == b"":  # we want to detect this special case.
 170         raise ValueError("not a tnetstring: empty file")
 171     data_length = b""
 172     while c.isdigit():
 173         data_length += c
 174         if len(data_length) > 9:
 175             raise ValueError("not a tnetstring: absurdly large length prefix")
 176         c = file_handle.read(1)
 177     if c != b":":
 178         raise ValueError("not a tnetstring: missing or invalid length prefix")
 179 
 180     data = file_handle.read(int(data_length))
 181     data_type = file_handle.read(1)[0]
 182 
 183     return parse(data_type, data)

 186 def parse(data_type: int, data: bytes) -> TSerializable:
 187     if data_type == ord(b','):
 188         return data
 189     if data_type == ord(b';'):
 190         return data.decode("utf8")
 191     if data_type == ord(b'#'):
 192         try:
 193             return int(data)
 194         except ValueError:
 195             raise ValueError(f"not a tnetstring: invalid integer literal: {data!r}")
 196     if data_type == ord(b'^'):
 197         try:
 198             return float(data)
 199         except ValueError:
 200             raise ValueError(f"not a tnetstring: invalid float literal: {data!r}")
 201     if data_type == ord(b'!'):
 202         if data == b'true':
 203             return True
 204         elif data == b'false':
 205             return False
 206         else:
 207             raise ValueError(f"not a tnetstring: invalid boolean literal: {data!r}")
 208     if data_type == ord(b'~'):
 209         if data:
 210             raise ValueError(f"not a tnetstring: invalid null literal: {data!r}")
 211         return None
 212     if data_type == ord(b']'):
 213         l = []
 214         while data:
 215             item, data = pop(data)
 216             l.append(item)  # type: ignore
 217         return l
 218     if data_type == ord(b'}'):
 219         d = {}
 220         while data:
 221             key, data = pop(data)
 222             val, data = pop(data)
 223             d[key] = val  # type: ignore
 224         return d
 225     raise ValueError("unknown type tag: {}".format(data_type))

 228 def pop(data: bytes) -> typing.Tuple[TSerializable, bytes]:
 229     """
 230     This function parses a tnetstring into a python object.
 231     It returns a tuple giving the parsed object and a string
 232     containing any unparsed data from the end of the string.
 233     """
 234     #  Parse out data length, type and remaining string.
 235     try:
 236         blength, data = data.split(b':', 1)
 237         length = int(blength)
 238     except ValueError:
 239         raise ValueError(f"not a tnetstring: missing or invalid length prefix: {data!r}")
 240     try:
 241         data, data_type, remain = data[:length], data[length], data[length + 1:]
 242     except IndexError:
 243         #  This fires if len(data) < dlen, meaning we don't need
 244         #  to further validate that data is the right length.
 245         raise ValueError("not a tnetstring: invalid length prefix: {}".format(length))
 246     # Parse the data based on the type tag.
 247     return parse(data_type, data), remain
