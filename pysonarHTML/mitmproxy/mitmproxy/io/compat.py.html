 182 def _convert_dict_keys(o: Any) -> Any:
 183     if isinstance(o, dict):
 184         return {strutils.always_str(k): _convert_dict_keys(v) for k, v in o.items()}
 185     else:
 186         return o
 187 
 188 
 189 def _convert_dict_vals(o: dict, values_to_convert: dict) -> dict:
 190     for k, v in values_to_convert.items():
 191         if not o or k not in o:
 192             continue  # pragma: no cover
 193         if v is True:
 194             o[k] = strutils.always_str(o[k])
 195         else:
 196             _convert_dict_vals(o[k], v)
 197     return o

 200 def convert_unicode(data: dict) -> dict:
 201     """
 202     This method converts between Python 3 and Python 2 dumpfiles.
 203     """
 204     data = _convert_dict_keys(data)
 205     data = _convert_dict_vals(
 206         data, {
 207             "type": True,
 208             "id": True,
 209             "request": {
 210                 "first_line_format": True
 211             },
 212             "error": {
 213                 "msg": True
 214             }
 215         }
 216     )
 217     return data

 240 def migrate_flow(flow_data: Dict[Union[bytes, str], Any]) -> Dict[Union[bytes, str], Any]:
 241     while True:
 242         flow_version = flow_data.get(b"version", flow_data.get("version"))
 243 
 244         # Historically, we used the mitmproxy minor version tuple as the flow format version.
 245         if not isinstance(flow_version, int):
 246             flow_version = tuple(flow_version)[:2]
 247 
 248         if flow_version == version.FLOW_FORMAT_VERSION:
 249             break
 250         elif flow_version in converters:
 251             flow_data = converters[flow_version](flow_data)
 252         else:
 253             should_upgrade = (
 254                 isinstance(flow_version, int)
 255                 and flow_version > version.FLOW_FORMAT_VERSION
 256             )
 257             raise ValueError(
 258                 "{} cannot read files with flow format version {}{}.".format(
 259                     version.MITMPROXY,
 260                     flow_version,
 261                     ", please update mitmproxy" if should_upgrade else ""
 262                 )
 263             )
 264     return flow_data
