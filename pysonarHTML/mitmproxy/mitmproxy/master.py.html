  40 class Master:
  41     """
  42         The master handles mitmproxy's main event loop.
  43     """
  44     def __init__(self, opts):
  45         self.should_exit = threading.Event()
  46         self.channel = controller.Channel(
  47             self,
  48             asyncio.get_event_loop(),
  49             self.should_exit,
  50         )
  51 
  52         self.options: options.Options = opts or options.Options()
  53         self.commands = command.CommandManager(self)
  54         self.addons = addonmanager.AddonManager(self)
  55         self._server = None
  56         self.waiting_flows = []
  57         self.log = log.Log(self)
  58 
  59         mitmproxy_ctx.master = self
  60         mitmproxy_ctx.log = self.log
  61         mitmproxy_ctx.options = self.options

  63     @property

  65         return self._server

  68     def server(self, server):
  69         server.set_channel(self.channel)
  70         self._server = server

  72     def start(self):
  73         self.should_exit.clear()
  74         if self.server:
  75             ServerThread(self.server).start()

  77     async def running(self):

  80     def run_loop(self, loop):
  81         self.start()
  82         asyncio.ensure_future(self.running())
  83 
  84         exc = None
  85         try:
  86             loop()
  87         except Exception:  # pragma: no cover
  88             exc = traceback.format_exc()
  89         finally:
  90             if not self.should_exit.is_set():  # pragma: no cover
  91                 self.shutdown()
  92             loop = asyncio.get_event_loop()
  93             tasks = asyncio.all_tasks(loop) if sys.version_info >= (3, 7) else asyncio.Task.all_tasks(loop)
  94             for p in tasks:
  95                 p.cancel()
  96             loop.close()
  97 
  98         if exc:  # pragma: no cover
  99             print(exc, file=sys.stderr)
 100             print("mitmproxy has crashed!", file=sys.stderr)
 101             print("Please lodge a bug report at:", file=sys.stderr)
 102             print("\thttps://github.com/mitmproxy/mitmproxy", file=sys.stderr)
 103 
 104         self.addons.trigger("done")
 105 
 106     def run(self, func=None):
 107         loop = asyncio.get_event_loop()
 108         self.run_loop(loop.run_forever)
 109 
 110     async def _shutdown(self):
 111         self.should_exit.set()
 112         if self.server:
 113             self.server.shutdown()
 114         loop = asyncio.get_event_loop()
 115         loop.stop()
 116 
 117     def shutdown(self):
 118         """
 119             Shut down the proxy. This method is thread-safe.
 120         """
 121         if not self.should_exit.is_set():
 122             self.should_exit.set()
 123             ret = asyncio.run_coroutine_threadsafe(self._shutdown(), loop=self.channel.loop)
 124             # Weird band-aid to make sure that self._shutdown() is actually executed,
 125             # which otherwise hangs the process as the proxy server is threaded.
 126             # This all needs to be simplified when the proxy server runs on asyncio as well.
 127             if not self.channel.loop.is_running():  # pragma: no cover
 128                 try:
 129                     self.channel.loop.run_until_complete(asyncio.wrap_future(ret))
 130                 except RuntimeError:
 131                     pass  # Event loop stopped before Future completed.

 133     def _change_reverse_host(self, f):
 134         """
 135         When we load flows in reverse proxy mode, we adjust the target host to
 136         the reverse proxy destination for all flows we load. This makes it very
 137         easy to replay saved flows against a different host.
 138         """
 139         if self.options.mode.startswith("reverse:"):
 140             _, upstream_spec = server_spec.parse_with_mode(self.options.mode)
 141             f.request.host, f.request.port = upstream_spec.address
 142             f.request.scheme = upstream_spec.scheme
 143 
 144     async def load_flow(self, f):
 145         """
 146         Loads a flow and links websocket & handshake flows
 147         """
 148 
 149         if isinstance(f, http.HTTPFlow):
 150             self._change_reverse_host(f)
 151             if 'websocket' in f.metadata:
 152                 self.waiting_flows.append(f)
 153 
 154         if isinstance(f, websocket.WebSocketFlow):
 155             hf = [hf for hf in self.waiting_flows if hf.id == f.metadata['websocket_handshake']][0]
 156             f.handshake_flow = hf
 157             self.waiting_flows.remove(hf)
 158             self._change_reverse_host(f.handshake_flow)
 159 
 160         f.reply = controller.DummyReply()
 161         for e, o in eventsequence.iterate(f):
 162             await self.addons.handle_lifecycle(e, o)
