 142 class HTTPFlow(flow.Flow):
 143     """
 144     An HTTPFlow is a collection of objects representing a single HTTP
 145     transaction.
 146     """
 147     request: HTTPRequest
 148     response: Optional[HTTPResponse] = None
 149     error: Optional[flow.Error] = None
 150     """
 151     Note that it's possible for a Flow to have both a response and an error
 152     object. This might happen, for instance, when a response was received
 153     from the server, but there was an error sending it back to the client.
 154     """
 155     server_conn: connections.ServerConnection
 156     client_conn: connections.ClientConnection
 157     intercepted: bool = False
 158     """ Is this flow currently being intercepted? """
 159     mode: str
 160     """ What mode was the proxy layer in when receiving this request? """
 161 
 162     def __init__(self, client_conn, server_conn, live=None, mode="regular"):
 163         super().__init__("http", client_conn, server_conn, live)
 164         self.mode = mode
 165 
 166     _stateobject_attributes = flow.Flow._stateobject_attributes.copy()
 167     # mypy doesn't support update with kwargs
 168     _stateobject_attributes.update(dict(
 169         request=HTTPRequest,
 170         response=HTTPResponse,
 171         mode=str
 172     ))
 173 
 174     def __repr__(self):
 175         s = "<HTTPFlow"
 176         for a in ("request", "response", "error", "client_conn", "server_conn"):
 177             if getattr(self, a, False):
 178                 s += "\r\n  %s = {flow.%s}" % (a, a)
 179         s += ">"
 180         return s.format(flow=self)
 181 
 182     @property
 183     def timestamp_start(self) -> float:
 184         return self.request.timestamp_start
 185 
 186     def copy(self):
 187         f = super().copy()
 188         if self.request:
 189             f.request = self.request.copy()
 190         if self.response:
 191             f.response = self.response.copy()
 192         return f

 195 def make_error_response(
 196         status_code: int,
 197         message: str = "",
 198         headers: Optional[http.Headers] = None,
 199 ) -> HTTPResponse:
 200     reason = http.status_codes.RESPONSES.get(status_code, "Unknown")
 201     body = """
 202         <html>
 203             <head>
 204                 <title>{status_code} {reason}</title>
 205             </head>
 206             <body>
 207             <h1>{status_code} {reason}</h1>
 208             <p>{message}</p>
 209             </body>
 210         </html>
 211     """.strip().format(
 212         status_code=status_code,
 213         reason=reason,
 214         message=html.escape(message),
 215     ).encode("utf8", "replace")
 216 
 217     if not headers:
 218         headers = http.Headers(
 219             Server=version.MITMPROXY,
 220             Connection="close",
 221             Content_Length=str(len(body)),
 222             Content_Type="text/html"
 223         )
 224 
 225     return HTTPResponse(
 226         b"HTTP/1.1",
 227         status_code,
 228         reason,
 229         headers,
 230         body,
 231     )

 234 def make_connect_request(address):
 235     return HTTPRequest(
 236         "authority", b"CONNECT", None, address[0], address[1], None, b"HTTP/1.1",
 237         http.Headers(), b""

 241 def make_connect_response(http_version):
 242     # Do not send any response headers as it breaks proxying non-80 ports on
 243     # Android emulators using the -http-proxy option.
 244     return HTTPResponse(
 245         http_version,
 246         200,
 247         b"Connection established",
 248         http.Headers(),
 249         b"",
 250     )
