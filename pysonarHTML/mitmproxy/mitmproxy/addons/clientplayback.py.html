  23 class RequestReplayThread(basethread.BaseThread):
  24     daemon = True
  25 
  26     def __init__(
  27             self,
  28             opts: options.Options,
  29             channel: controller.Channel,
  30             queue: queue.Queue,
  31     ) -> None:
  32         self.options = opts
  33         self.channel = channel
  34         self.queue = queue
  35         self.inflight = threading.Event()
  36         super().__init__("RequestReplayThread")
  37 
  38     def run(self):
  39         while True:
  40             f = self.queue.get()
  41             self.inflight.set()
  42             self.replay(f)
  43             self.inflight.clear()
  44 
  45     def replay(self, f):  # pragma: no cover
  46         f.live = True
  47         r = f.request
  48         bsl = human.parse_size(self.options.body_size_limit)
  49         first_line_format_backup = r.first_line_format
  50         server = None
  51         try:
  52             f.response = None
  53 
  54             # If we have a channel, run script hooks.
  55             request_reply = self.channel.ask("request", f)
  56             if isinstance(request_reply, http.HTTPResponse):
  57                 f.response = request_reply
  58 
  59             if not f.response:
  60                 # In all modes, we directly connect to the server displayed
  61                 if self.options.mode.startswith("upstream:"):
  62                     server_address = server_spec.parse_with_mode(self.options.mode)[1].address
  63                     server = connections.ServerConnection(server_address)
  64                     server.connect()
  65                     if r.scheme == "https":
  66                         connect_request = http.make_connect_request((r.data.host, r.port))
  67                         server.wfile.write(http1.assemble_request(connect_request))
  68                         server.wfile.flush()
  69                         resp = http1.read_response(
  70                             server.rfile,
  71                             connect_request,
  72                             body_size_limit=bsl
  73                         )
  74                         if resp.status_code != 200:
  75                             raise exceptions.ReplayException(
  76                                 "Upstream server refuses CONNECT request"
  77                             )
  78                         server.establish_tls(
  79                             sni=f.server_conn.sni,
  80                             **tls.client_arguments_from_options(self.options)
  81                         )
  82                         r.first_line_format = "relative"
  83                     else:
  84                         r.first_line_format = "absolute"
  85                 else:
  86                     server_address = (r.host, r.port)
  87                     server = connections.ServerConnection(server_address)
  88                     server.connect()
  89                     if r.scheme == "https":
  90                         server.establish_tls(
  91                             sni=f.server_conn.sni,
  92                             **tls.client_arguments_from_options(self.options)
  93                         )
  94                     r.first_line_format = "relative"
  95 
  96                 server.wfile.write(http1.assemble_request(r))
  97                 server.wfile.flush()
  98                 r.timestamp_start = r.timestamp_end = time.time()
  99 
 100                 if f.server_conn:
 101                     f.server_conn.close()
 102                 f.server_conn = server
 103 
 104                 f.response = http.HTTPResponse.wrap(
 105                     http1.read_response(server.rfile, r, body_size_limit=bsl)
 106                 )
 107             response_reply = self.channel.ask("response", f)
 108             if response_reply == exceptions.Kill:
 109                 raise exceptions.Kill()
 110         except (exceptions.ReplayException, exceptions.NetlibException) as e:
 111             f.error = flow.Error(str(e))
 112             self.channel.ask("error", f)
 113         except exceptions.Kill:
 114             self.channel.tell("log", log.LogEntry("Connection killed", "info"))
 115         except Exception as e:
 116             self.channel.tell("log", log.LogEntry(repr(e), "error"))
 117         finally:
 118             r.first_line_format = first_line_format_backup
 119             f.live = False
 120             if server and server.connected():
 121                 server.finish()
 122                 server.close()

 125 class ClientPlayback:
 126     def __init__(self):
 127         self.q = queue.Queue()
 128         self.thread: RequestReplayThread = None
 129 
 130     def check(self, f: flow.Flow):
 131         if f.live:
 132             return "Can't replay live flow."
 133         if f.intercepted:
 134             return "Can't replay intercepted flow."
 135         if isinstance(f, http.HTTPFlow):
 136             if not f.request:
 137                 return "Can't replay flow with missing request."
 138             if f.request.raw_content is None:
 139                 return "Can't replay flow with missing content."
 140         else:
 141             return "Can only replay HTTP flows."
 142 
 143     def load(self, loader):
 144         loader.add_option(
 145             "client_replay", typing.Sequence[str], [],
 146             "Replay client requests from a saved file."
 147         )
 148 
 149     def running(self):
 150         self.thread = RequestReplayThread(
 151             ctx.options,
 152             ctx.master.channel,
 153             self.q,
 154         )
 155         self.thread.start()
 156 
 157     def configure(self, updated):
 158         if "client_replay" in updated and ctx.options.client_replay:
 159             try:
 160                 flows = io.read_flows_from_paths(ctx.options.client_replay)
 161             except exceptions.FlowReadException as e:
 162                 raise exceptions.OptionsError(str(e))
 163             self.start_replay(flows)
 164 
 165     @command.command("replay.client.count")
 166     def count(self) -> int:
 167         """
 168             Approximate number of flows queued for replay.
 169         """
 170         inflight = 1 if self.thread and self.thread.inflight.is_set() else 0
 171         return self.q.qsize() + inflight
 172 
 173     @command.command("replay.client.stop")
 174     def stop_replay(self) -> None:
 175         """
 176             Clear the replay queue.
 177         """
 178         with self.q.mutex:
 179             lst = list(self.q.queue)
 180             self.q.queue.clear()
 181             for f in lst:
 182                 f.revert()
 183         ctx.master.addons.trigger("update", lst)
 184         ctx.log.alert("Client replay queue cleared.")
 185 
 186     @command.command("replay.client")
 187     def start_replay(self, flows: typing.Sequence[flow.Flow]) -> None:
 188         """
 189             Add flows to the replay queue, skipping flows that can't be replayed.
 190         """
 191         lst = []
 192         for f in flows:
 193             hf = typing.cast(http.HTTPFlow, f)
 194 
 195             err = self.check(hf)
 196             if err:
 197                 ctx.log.warn(err)
 198                 continue
 199 
 200             lst.append(hf)
 201             # Prepare the flow for replay
 202             hf.backup()
 203             hf.request.is_replay = True
 204             hf.response = None
 205             hf.error = None
 206             # https://github.com/mitmproxy/mitmproxy/issues/2197
 207             if hf.request.http_version == "HTTP/2.0":
 208                 hf.request.http_version = "HTTP/1.1"
 209                 host = hf.request.headers.pop(":authority", None)
 210                 if host is not None:
 211                     hf.request.headers.insert(0, "host", host)
 212             self.q.put(hf)
 213         ctx.master.addons.trigger("update", lst)
 214 
 215     @command.command("replay.client.file")
 216     def load_file(self, path: mitmproxy.types.Path) -> None:
 217         """
 218             Load flows from file, and add them to the replay queue.
 219         """
 220         try:
 221             flows = io.read_flows_from_paths([path])
 222         except exceptions.FlowReadException as e:
 223             raise exceptions.CommandError(str(e))
 224         self.start_replay(flows)
