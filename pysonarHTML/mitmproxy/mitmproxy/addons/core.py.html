  38     def configure(self, updated):
  39         opts = ctx.options
  40         if opts.add_upstream_certs_to_client_chain and not opts.upstream_cert:
  41             raise exceptions.OptionsError(
  42                 "The no-upstream-cert and add-upstream-certs-to-client-chain "
  43                 "options are mutually exclusive. If no-upstream-cert is enabled "
  44                 "then the upstream certificate is not retrieved before generating "
  45                 "the client certificate chain."
  46             )
  47         if opts.add_upstream_certs_to_client_chain and not opts.ssl_insecure:
  48             raise exceptions.OptionsError(
  49                 "The verify-upstream-cert requires certificate verification to be disabled. "
  50                 "If upstream certificates are verified then extra upstream certificates are "
  51                 "not available for inclusion to the client chain."
  52             )
  53         if "body_size_limit" in updated:
  54             try:
  55                 human.parse_size(opts.body_size_limit)
  56             except ValueError:
  57                 raise exceptions.OptionsError(
  58                     "Invalid body size limit specification: %s" %
  59                     opts.body_size_limit
  60                 )
  61         if "mode" in updated:
  62             mode = opts.mode
  63             if mode.startswith("reverse:") or mode.startswith("upstream:"):
  64                 try:
  65                     server_spec.parse_with_mode(mode)
  66                 except ValueError as e:
  67                     raise exceptions.OptionsError(str(e)) from e
  68             elif mode == "transparent":
  69                 if not platform.original_addr:
  70                     raise exceptions.OptionsError(
  71                         "Transparent mode not supported on this platform."
  72                     )
  73             elif mode not in ["regular", "socks5"]:
  74                 raise exceptions.OptionsError(
  75                     "Invalid mode specification: %s" % mode
  76                 )
  77         if "client_certs" in updated:
  78             if opts.client_certs:
  79                 client_certs = os.path.expanduser(opts.client_certs)
  80                 if not os.path.exists(client_certs):
  81                     raise exceptions.OptionsError(
  82                         "Client certificate path does not exist: {}".format(opts.client_certs)
  83                     )

  85     @command.command("set")
  86     def set(self, option: str, value: str = "") -> None:
  87         """
  88             Set an option. When the value is omitted, booleans are set to true,
  89             strings and integers are set to None (if permitted), and sequences
  90             are emptied. Boolean values can be true, false or toggle.
  91             Multiple values are concatenated with a single space.
  92         """
  93         strspec = f"{option}={value}"
  94         try:
  95             ctx.options.set(strspec)
  96         except exceptions.OptionsError as e:
  97             raise exceptions.CommandError(e) from e

  99     @command.command("flow.resume")
 100     def resume(self, flows: typing.Sequence[flow.Flow]) -> None:
 101         """
 102             Resume flows if they are intercepted.
 103         """
 104         intercepted = [i for i in flows if i.intercepted]
 105         for f in intercepted:
 106             f.resume()
 107         ctx.master.addons.trigger("update", intercepted)

 170     @command.command("flow.set")
 171     @command.argument("attr", type=mitmproxy.types.Choice("flow.set.options"))
 172     def flow_set(
 173         self,
 174         flows: typing.Sequence[flow.Flow],
 175         attr: str,
 176         value: str
 177     ) -> None:
 178         """
 179             Quickly set a number of common values on flows.
 180         """
 181         val: typing.Union[int, str] = value
 182         if attr == "status_code":
 183             try:
 184                 val = int(val)  # type: ignore
 185             except ValueError as v:
 186                 raise exceptions.CommandError(
 187                     "Status code is not an integer: %s" % val
 188                 ) from v
 189 
 190         updated = []
 191         for f in flows:
 192             req = getattr(f, "request", None)
 193             rupdate = True
 194             if req:
 195                 if attr == "method":
 196                     req.method = val
 197                 elif attr == "host":
 198                     req.host = val
 199                 elif attr == "path":
 200                     req.path = val
 201                 elif attr == "url":
 202                     try:
 203                         req.url = val
 204                     except ValueError as e:
 205                         raise exceptions.CommandError(
 206                             "URL %s is invalid: %s" % (repr(val), e)
 207                         ) from e
 208                 else:
 209                     self.rupdate = False
 210 
 211             resp = getattr(f, "response", None)
 212             supdate = True
 213             if resp:
 214                 if attr == "status_code":
 215                     resp.status_code = val
 216                     if val in status_codes.RESPONSES:
 217                         resp.reason = status_codes.RESPONSES[val]  # type: ignore
 218                 elif attr == "reason":
 219                     resp.reason = val
 220                 else:
 221                     supdate = False
 222 
 223             if rupdate or supdate:
 224                 updated.append(f)
 225 
 226         ctx.master.addons.trigger("update", updated)
 227         ctx.log.alert("Set %s on  %s flows." % (attr, len(updated)))

 229     @command.command("flow.decode")
 230     def decode(self, flows: typing.Sequence[flow.Flow], part: str) -> None:
 231         """
 232             Decode flows.
 233         """
 234         updated = []
 235         for f in flows:
 236             p = getattr(f, part, None)
 237             if p:
 238                 f.backup()
 239                 p.decode()
 240                 updated.append(f)
 241         ctx.master.addons.trigger("update", updated)
 242         ctx.log.alert("Decoded %s flows." % len(updated))

 244     @command.command("flow.encode.toggle")
 245     def encode_toggle(self, flows: typing.Sequence[flow.Flow], part: str) -> None:
 246         """
 247             Toggle flow encoding on and off, using deflate for encoding.
 248         """
 249         updated = []
 250         for f in flows:
 251             p = getattr(f, part, None)
 252             if p:
 253                 f.backup()
 254                 current_enc = p.headers.get("content-encoding", "identity")
 255                 if current_enc == "identity":
 256                     p.encode("deflate")
 257                 else:
 258                     p.decode()
 259                 updated.append(f)
 260         ctx.master.addons.trigger("update", updated)
 261         ctx.log.alert("Toggled encoding on %s flows." % len(updated))

 263     @command.command("flow.encode")
 264     @command.argument("encoding", type=mitmproxy.types.Choice("flow.encode.options"))
 265     def encode(
 266         self,
 267         flows: typing.Sequence[flow.Flow],
 268         part: str,
 269         encoding: str,
 270     ) -> None:
 271         """
 272             Encode flows with a specified encoding.
 273         """
 274         updated = []
 275         for f in flows:
 276             p = getattr(f, part, None)
 277             if p:
 278                 current_enc = p.headers.get("content-encoding", "identity")
 279                 if current_enc == "identity":
 280                     f.backup()
 281                     p.encode(encoding)
 282                     updated.append(f)
 283         ctx.master.addons.trigger("update", updated)
 284         ctx.log.alert("Encoded %s flows." % len(updated))
