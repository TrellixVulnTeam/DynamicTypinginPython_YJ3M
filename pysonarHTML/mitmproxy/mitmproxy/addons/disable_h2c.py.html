   4 class DisableH2C:
   5 
   6     """
   7     We currently only support HTTP/2 over a TLS connection.
   8 
   9     Some clients try to upgrade a connection from HTTP/1.1 to h2c. We need to
  10     remove those headers to avoid protocol errors if one endpoints suddenly
  11     starts sending HTTP/2 frames.
  12 
  13     Some clients might use HTTP/2 Prior Knowledge to directly initiate a session
  14     by sending the connection preface. We just kill those flows.
  15     """
  16 
  17     def process_flow(self, f):
  18         if f.request.headers.get('upgrade', '') == 'h2c':
  19             mitmproxy.ctx.log.warn("HTTP/2 cleartext connections (h2c upgrade requests) are currently not supported.")
  20             del f.request.headers['upgrade']
  21             if 'connection' in f.request.headers:
  22                 del f.request.headers['connection']
  23             if 'http2-settings' in f.request.headers:
  24                 del f.request.headers['http2-settings']
  25 
  26         is_connection_preface = (
  27             f.request.method == 'PRI' and
  28             f.request.path == '*' and
  29             f.request.http_version == 'HTTP/2.0'
  30         )
  31         if is_connection_preface:
  32             f.kill()
  33             mitmproxy.ctx.log.warn("Initiating HTTP/2 connections with prior knowledge are currently not supported.")
  34 
  35     # Handlers
  36 
  37     def request(self, f):
  38         self.process_flow(f)
