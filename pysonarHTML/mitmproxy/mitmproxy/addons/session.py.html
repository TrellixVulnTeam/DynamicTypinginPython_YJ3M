  33 class SessionDB:
  34     """
  35     This class wraps connection to DB
  36     for Sessions and handles creation,
  37     retrieving and insertion in tables.
  38     """
  39     content_threshold = 1000
  40     type_mappings = {
  41         "body": {
  42             1: "request",
  43             2: "response"
  44         }
  45     }
  46 
  47     def __init__(self, db_path=None):
  48         """
  49         Connect to an already existing database,
  50         or create a new one with optional path.
  51         :param db_path:
  52         """
  53         self.live_components: typing.Dict[str, tuple] = {}
  54         self.tempdir: tempfile.TemporaryDirectory = None
  55         self.con: sqlite3.Connection = None
  56         # This is used for fast look-ups over bodies already dumped to database.
  57         # This permits to enforce one-to-one relationship between flow and body table.
  58         self.body_ledger: typing.Set[str] = set()
  59         self.id_ledger: typing.Set[str] = set()
  60         if db_path is not None and os.path.isfile(db_path):
  61             self._load_session(db_path)
  62         else:
  63             if db_path:
  64                 path = db_path
  65             else:
  66                 self.tempdir = tempfile.mkdtemp()
  67                 path = os.path.join(self.tempdir, 'tmp.sqlite')
  68             self.con = sqlite3.connect(path)
  69             self._create_session()
  70 
  71     def __del__(self):
  72         if self.con:
  73             self.con.close()
  74         if self.tempdir:
  75             shutil.rmtree(self.tempdir)
  76 
  77     def __contains__(self, fid):
  78         return fid in self.id_ledger
  79 
  80     def __len__(self):
  81         return len(self.id_ledger)
  82 
  83     def _load_session(self, path):
  84         if not self.is_session_db(path):
  85             raise SessionLoadException('Given path does not point to a valid Session')
  86         self.con = sqlite3.connect(path)
  87 
  88     def _create_session(self):
  89         script_path = pkg_data.path("io/sql/session_create.sql")
  90         with open(script_path, 'r') as qry:
  91             self.con.executescript(qry.read())
  92         self.con.commit()
  93 
  94     @staticmethod
  95     def is_session_db(path):
  96         """
  97         Check if database entered from user
  98         is a valid Session SQLite DB.
  99         :return: True if valid, False if invalid.
 100         """
 101         c = None
 102         try:
 103             c = sqlite3.connect(f'file:{path}?mode=rw', uri=True)
 104             cursor = c.cursor()
 105             cursor.execute("SELECT NAME FROM sqlite_master WHERE type='table';")
 106             rows = cursor.fetchall()
 107             tables = [('flow',), ('body',), ('annotation',)]
 108             if all(elem in rows for elem in tables):
 109                 c.close()
 110                 return True
 111         except sqlite3.Error:
 112             if c:
 113                 c.close()
 114         return False

 116     def _disassemble(self, flow):
 117         # Some live components of flows cannot be serialized, but they are needed to ensure correct functionality.
 118         # We solve this by keeping a list of tuples which "save" those components for each flow id, eventually
 119         # adding them back when needed.
 120         self.live_components[flow.id] = (
 121             flow.client_conn.wfile,
 122             flow.client_conn.rfile,
 123             flow.client_conn.reply,
 124             flow.server_conn.wfile,
 125             flow.server_conn.rfile,
 126             flow.server_conn.reply,
 127             (flow.server_conn.via.wfile, flow.server_conn.via.rfile,
 128              flow.server_conn.via.reply) if flow.server_conn.via else None,
 129             flow.reply
 130         )
 131 
 132     def _reassemble(self, flow):
 133         if flow.id in self.live_components:
 134             cwf, crf, crp, swf, srf, srp, via, rep = self.live_components[flow.id]
 135             flow.client_conn.wfile = cwf
 136             flow.client_conn.rfile = crf
 137             flow.client_conn.reply = crp
 138             flow.server_conn.wfile = swf
 139             flow.server_conn.rfile = srf
 140             flow.server_conn.reply = srp
 141             flow.reply = rep
 142             if via:
 143                 flow.server_conn.via.rfile, flow.server_conn.via.wfile, flow.server_conn.via.reply = via
 144         return flow
 145 
 146     def store_flows(self, flows):
 147         body_buf = []
 148         flow_buf = []
 149         for flow in flows:
 150             self.id_ledger.add(flow.id)
 151             self._disassemble(flow)
 152             f = copy.copy(flow)
 153             f.request = copy.deepcopy(flow.request)
 154             if flow.response:
 155                 f.response = copy.deepcopy(flow.response)
 156             f.id = flow.id
 157             if len(f.request.content) > self.content_threshold and f.id not in self.body_ledger:
 158                 body_buf.append((f.id, 1, f.request.content))
 159                 f.request.content = b""
 160                 self.body_ledger.add(f.id)
 161             if f.response and f.id not in self.body_ledger:
 162                 if len(f.response.content) > self.content_threshold:
 163                     body_buf.append((f.id, 2, f.response.content))
 164                     f.response.content = b""
 165             flow_buf.append((f.id, protobuf.dumps(f)))
 166         self.con.executemany("INSERT OR REPLACE INTO flow VALUES(?, ?);", flow_buf)
 167         if body_buf:
 168             self.con.executemany("INSERT INTO body (flow_id, type_id, content) VALUES(?, ?, ?);", body_buf)
 169         self.con.commit()
 170 
 171     def retrieve_flows(self, ids=None):
 172         flows = []
 173         with self.con as con:
 174             if not ids:
 175                 sql = "SELECT f.content, b.type_id, b.content " \
 176                       "FROM flow f " \
 177                       "LEFT OUTER JOIN body b ON f.id = b.flow_id;"
 178                 rows = con.execute(sql).fetchall()
 179             else:
 180                 sql = "SELECT f.content, b.type_id, b.content " \
 181                       "FROM flow f " \
 182                       "LEFT OUTER JOIN body b ON f.id = b.flow_id " \
 183                       f"AND f.id IN ({','.join(['?' for _ in range(len(ids))])});"
 184                 rows = con.execute(sql, ids).fetchall()
 185             for row in rows:
 186                 flow = protobuf.loads(row[0])
 187                 if row[1]:
 188                     typ = self.type_mappings["body"][row[1]]
 189                     if typ and row[2]:
 190                         setattr(getattr(flow, typ), "content", row[2])
 191                 flow = self._reassemble(flow)
 192                 flows.append(flow)
 193         return flows

 195     def clear(self):
 196         self.con.executescript("DELETE FROM body; DELETE FROM annotation; DELETE FROM flow;")

 209 class Session:
 210 
 211     _FP_RATE = 150
 212     _FP_DECREMENT = 0.9
 213     _FP_DEFAULT = 3.0
 214 
 215     def __init__(self):
 216         self.db_store: SessionDB = None
 217         self._hot_store: collections.OrderedDict = collections.OrderedDict()
 218         self._order_store: typing.Dict[str, typing.Dict[str, typing.Union[int, float, str, None]]] = {}
 219         self._view: typing.List[typing.Tuple[typing.Union[int, float, str, None], str]] = []
 220         self.order: str = orders[0]
 221         self.filter = matchall
 222         self._flush_period: float = self._FP_DEFAULT
 223         self._flush_rate: int = self._FP_RATE
 224         self.started: bool = False
 225 
 226     def load(self, loader):
 227         loader.add_option(
 228             "session_path", typing.Optional[types.Path], None,
 229             "Path of session to load or to create."
 230         )
 231         loader.add_option(
 232             "view_order", str, "time",
 233             "Flow sort order.",
 234             choices=list(map(lambda c: c[1], orders))
 235         )
 236         loader.add_option(
 237             "view_filter", typing.Optional[str], None,
 238             "Limit the view to matching flows."
 239         )
 240 
 241     def running(self):
 242         if not self.started:
 243             self.started = True
 244             self.db_store = SessionDB(ctx.options.session_path)
 245             loop = asyncio.get_event_loop()
 246             loop.create_task(self._writer())
 247 
 248     def configure(self, updated):
 249         if "view_order" in updated:
 250             self.set_order(ctx.options.view_order)
 251         if "view_filter" in updated:
 252             self.set_filter(ctx.options.view_filter)
 253 
 254     async def _writer(self):
 255         while True:
 256             await asyncio.sleep(self._flush_period)
 257             batches = -(-len(self._hot_store) // self._flush_rate)
 258             self._flush_period = self._flush_period * self._FP_DECREMENT if batches > 1 else self._FP_DEFAULT
 259             while batches:
 260                 tof = []
 261                 to_dump = min(len(self._hot_store), self._flush_rate)
 262                 for _ in range(to_dump):
 263                     tof.append(self._hot_store.popitem(last=False)[1])
 264                 self.db_store.store_flows(tof)
 265                 batches -= 1
 266                 await asyncio.sleep(0.01)

 268     def load_view(self) -> typing.Sequence[http.HTTPFlow]:
 269         ids = [fid for _, fid in self._view]
 270         flows = self.load_storage(ids)
 271         return sorted(flows, key=lambda f: self._generate_order(self.order, f))
 272 
 273     def load_storage(self, ids=None) -> typing.Sequence[http.HTTPFlow]:
 274         flows = []
 275         ids_from_store = []
 276         if ids is not None:
 277             for fid in ids:
 278                 # A same flow could be at the same time in hot and db storage. We want the most updated version.
 279                 if fid in self._hot_store:
 280                     flows.append(self._hot_store[fid])
 281                 elif fid in self.db_store:
 282                     ids_from_store.append(fid)
 283             flows += self.db_store.retrieve_flows(ids_from_store)
 284         else:
 285             for flow in self._hot_store.values():
 286                 flows.append(flow)
 287             for flow in self.db_store.retrieve_flows():
 288                 if flow.id not in self._hot_store:
 289                     flows.append(flow)
 290         return flows
 291 
 292     def clear_storage(self):
 293         self.db_store.clear()
 294         self._hot_store.clear()
 295         self._view = []
 296 
 297     def store_count(self) -> int:
 298         ln = 0
 299         for fid in self._hot_store.keys():
 300             if fid not in self.db_store:
 301                 ln += 1
 302         return ln + len(self.db_store)

 304     @staticmethod
 305     def _generate_order(o: str, f: http.HTTPFlow) -> typing.Optional[typing.Union[str, int, float]]:
 306         if o == "time":
 307             return f.request.timestamp_start or 0
 308         if o == "method":
 309             return f.request.method
 310         if o == "url":
 311             return f.request.url
 312         if o == "size":
 313             s = 0
 314             if f.request.raw_content:
 315                 s += len(f.request.raw_content)
 316             if f.response and f.response.raw_content:
 317                 s += len(f.response.raw_content)
 318             return s
 319         return None
 320 
 321     def _store_order(self, f: http.HTTPFlow):
 322         self._order_store[f.id] = {}
 323         for order in orders:
 324             self._order_store[f.id][order] = self._generate_order(order, f)
 325 
 326     def set_order(self, order: str) -> None:
 327         if order not in orders:
 328             raise CommandError(
 329                 "Unknown flow order: %s" % order
 330             )
 331         if order != self.order:
 332             self.order = order
 333             newview = [
 334                 (self._order_store[t[1]][order], t[1]) for t in self._view
 335             ]
 336             self._view = sorted(newview)
 337 
 338     def _refilter(self):
 339         self._view = []
 340         flows = self.load_storage()
 341         for f in flows:
 342             if self.filter(f):
 343                 self.update_view(f)
 344 
 345     def set_filter(self, input_filter: typing.Optional[str]) -> None:
 346         filt = matchall if not input_filter else flowfilter.parse(input_filter)
 347         if not filt:
 348             raise CommandError(
 349                 "Invalid interception filter: %s" % filt
 350             )
 351         self.filter = filt
 352         self._refilter()
 353 
 354     def update_view(self, f):
 355         if any([f.id == t[1] for t in self._view]):
 356             self._view = [(order, fid) for order, fid in self._view if fid != f.id]
 357         o = self._order_store[f.id][self.order]
 358         self._view.insert(bisect.bisect_left(KeyifyList(self._view, lambda x: x[0]), o), (o, f.id))
 359 
 360     def update(self, flows: typing.Sequence[http.HTTPFlow]) -> None:
 361         for f in flows:
 362             self._store_order(f)
 363             if f.id in self._hot_store:
 364                 self._hot_store.pop(f.id)
 365             self._hot_store[f.id] = f
 366             if self.filter(f):
 367                 self.update_view(f)
 368 
 369     def request(self, f):
 370         self.update([f])
 371 
 372     def error(self, f):
 373         self.update([f])

 376         self.update([f])

 378     def intercept(self, f):
 379         self.update([f])

 382         self.update([f])

 385         self.update([f])
