  15 def cleanup_request(f: flow.Flow) -> http.HTTPRequest:
  16     if not getattr(f, "request", None):
  17         raise exceptions.CommandError("Can't export flow with no request.")
  18     assert isinstance(f, http.HTTPFlow)
  19     request = f.request.copy()
  20     request.decode(strict=False)
  21     # a bit of clean-up - these headers should be automatically set by curl/httpie
  22     request.headers.pop('content-length')
  23     if request.headers.get("host", "") == request.host:
  24         request.headers.pop("host")
  25     if request.headers.get(":authority", "") == request.host:
  26         request.headers.pop(":authority")
  27     return request

  30 def cleanup_response(f: flow.Flow) -> http.HTTPResponse:
  31     if not getattr(f, "response", None):
  32         raise exceptions.CommandError("Can't export flow with no response.")
  33     assert isinstance(f, http.HTTPFlow)
  34     response = f.response.copy()  # type: ignore
  35     response.decode(strict=False)
  36     return response

  39 def request_content_for_console(request: http.HTTPRequest) -> str:
  40     try:
  41         text = request.get_text(strict=True)
  42         assert text
  43     except ValueError:
  44         # shlex.quote doesn't support a bytes object
  45         # see https://github.com/python/cpython/pull/10871
  46         raise exceptions.CommandError("Request content must be valid unicode")
  47     escape_control_chars = {chr(i): f"\\x{i:02x}" for i in range(32)}
  48     return "".join(
  49         escape_control_chars.get(x, x)
  50         for x in text
  51     )

  54 def curl_command(f: flow.Flow) -> str:
  55     request = cleanup_request(f)
  56     args = ["curl"]
  57     for k, v in request.headers.items(multi=True):
  58         if k.lower() == "accept-encoding":
  59             args.append("--compressed")
  60         else:
  61             args += ["-H", f"{k}: {v}"]
  62 
  63     if request.method != "GET":
  64         args += ["-X", request.method]
  65     args.append(request.url)
  66     if request.content:
  67         args += ["-d", request_content_for_console(request)]
  68     return ' '.join(shlex.quote(arg) for arg in args)

  71 def httpie_command(f: flow.Flow) -> str:
  72     request = cleanup_request(f)
  73     args = ["http", request.method, request.url]
  74     for k, v in request.headers.items(multi=True):
  75         args.append(f"{k}: {v}")
  76     cmd = ' '.join(shlex.quote(arg) for arg in args)
  77     if request.content:
  78         cmd += " <<< " + shlex.quote(request_content_for_console(request))
  79     return cmd

  90 def raw(f: flow.Flow, separator=b"\r\n\r\n") -> bytes:
  91     """Return either the request or response if only one exists, otherwise return both"""
  92     request_present = hasattr(f, "request") and f.request  # type: ignore
  93     response_present = hasattr(f, "response") and f.response  # type: ignore
  94 
  95     if not (request_present or response_present):
  96         raise exceptions.CommandError("Can't export flow with no request or response.")
  97 
  98     if request_present and response_present:
  99         return b"".join([raw_request(f), separator, raw_response(f)])
 100     elif not request_present:
 101         return raw_response(f)
 102     else:
 103         return raw_request(f)

 115 class Export():
 116     @command.command("export.formats")
 117     def formats(self) -> typing.Sequence[str]:
 118         """
 119             Return a list of the supported export formats.
 120         """
 121         return list(sorted(formats.keys()))
 122 
 123     @command.command("export.file")
 124     def file(self, format: str, flow: flow.Flow, path: mitmproxy.types.Path) -> None:
 125         """
 126             Export a flow to path.
 127         """
 128         if format not in formats:
 129             raise exceptions.CommandError("No such export format: %s" % format)
 130         func: typing.Any = formats[format]
 131         v = func(flow)
 132         try:
 133             with open(path, "wb") as fp:
 134                 if isinstance(v, bytes):
 135                     fp.write(v)
 136                 else:
 137                     fp.write(v.encode("utf-8"))
 138         except IOError as e:
 139             ctx.log.error(str(e))
 140 
 141     @command.command("export.clip")
 142     def clip(self, format: str, flow: flow.Flow) -> None:
 143         """
 144             Export a flow to the system clipboard.
 145         """
 146         if format not in formats:
 147             raise exceptions.CommandError("No such export format: %s" % format)
 148         func: typing.Any = formats[format]
 149         v = strutils.always_str(func(flow))
 150         try:
 151             pyperclip.copy(v)
 152         except pyperclip.PyperclipException as e:
 153             ctx.log.error(str(e))
