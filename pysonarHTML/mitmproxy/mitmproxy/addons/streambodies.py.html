   9 class StreamBodies:
  10     def __init__(self):
  11         self.max_size = None
  12 
  13     def load(self, loader):
  14         loader.add_option(
  15             "stream_large_bodies", typing.Optional[str], None,
  16             """
  17             Stream data to the client if response body exceeds the given
  18             threshold. If streamed, the body will not be stored in any way.
  19             Understands k/m/g suffixes, i.e. 3m for 3 megabytes.
  20             """
  21         )
  22         loader.add_option(
  23             "stream_websockets", bool, False,
  24             """
  25             Stream WebSocket messages between client and server.
  26             Messages are captured and cannot be modified.
  27             """
  28         )
  29 
  30     def configure(self, updated):
  31         if "stream_large_bodies" in updated and ctx.options.stream_large_bodies:
  32             try:
  33                 self.max_size = human.parse_size(ctx.options.stream_large_bodies)
  34             except ValueError as e:
  35                 raise exceptions.OptionsError(e)
  36 
  37     def run(self, f, is_request):
  38         if self.max_size:
  39             r = f.request if is_request else f.response
  40             try:
  41                 expected_size = http1.expected_http_body_size(
  42                     f.request, f.response if not is_request else None
  43                 )
  44             except exceptions.HttpException:
  45                 f.reply.kill()
  46                 return
  47             if expected_size and not r.raw_content and not (0 <= expected_size <= self.max_size):
  48                 # r.stream may already be a callable, which we want to preserve.
  49                 r.stream = r.stream or True
  50                 ctx.log.info("Streaming {} {}".format("response from" if not is_request else "request to", f.request.host))
  51 
  52     def requestheaders(self, f):
  53         self.run(f, True)
  54 
  55     def responseheaders(self, f):
  56         self.run(f, False)
  57 
  58     def websocket_start(self, f):
  59         if ctx.options.stream_websockets:
  60             f.stream = True
  61             ctx.log.info("Streaming WebSocket messages between {client} and {server}".format(
  62                 client=human.format_address(f.client_conn.address),
  63                 server=human.format_address(f.server_conn.address))
  64             )
