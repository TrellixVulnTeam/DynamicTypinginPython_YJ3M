  27 def read_cert(ext, content_type):
  28     filename = CONF_BASENAME + f"-ca-cert.{ext}"
  29     p = os.path.join(app.config["CONFDIR"], filename)
  30     p = os.path.expanduser(p)
  31     with open(p, "rb") as f:
  32         cert = f.read()
  33 
  34     return cert, {
  35         "Content-Type": content_type,
  36         "Content-Disposition": f"inline; filename={filename}",
  37     }
