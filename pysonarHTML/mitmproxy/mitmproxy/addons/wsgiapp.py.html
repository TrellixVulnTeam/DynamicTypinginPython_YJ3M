   8 class WSGIApp:
   9     """
  10         An addon that hosts a WSGI app within mitproxy, at a specified
  11         hostname and port.
  12     """
  13     def __init__(self, app, host, port):
  14         self.app, self.host, self.port = app, host, port
  15 
  16     @property
  17     def name(self):
  18         return "wsgiapp:%s:%s" % (self.host, self.port)
  19 
  20     def serve(self, app, flow):
  21         """
  22             Serves app on flow, and prevents further handling of the flow.
  23         """
  24         app = wsgi.WSGIAdaptor(
  25             app,
  26             flow.request.pretty_host,
  27             flow.request.port,
  28             version.MITMPROXY
  29         )
  30         err = app.serve(
  31             flow,
  32             flow.client_conn.wfile,
  33             **{"mitmproxy.master": ctx.master}
  34         )
  35         if err:
  36             ctx.log.error("Error in wsgi app. %s" % err)
  37             raise exceptions.AddonHalt()
  38         flow.reply.kill()
  39 
  40     def request(self, f):
  41         if (f.request.pretty_host, f.request.port) == (self.host, self.port):
  42             self.serve(self.app, f)
