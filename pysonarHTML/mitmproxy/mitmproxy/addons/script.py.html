  19 def load_script(path: str) -> typing.Optional[types.ModuleType]:
  20     fullname = "__mitmproxy_script__.{}".format(
  21         os.path.splitext(os.path.basename(path))[0]
  22     )
  23     # the fullname is not unique among scripts, so if there already is an existing script with said
  24     # fullname, remove it.
  25     sys.modules.pop(fullname, None)
  26     oldpath = sys.path
  27     sys.path.insert(0, os.path.dirname(path))
  28     m = None
  29     try:
  30         loader = importlib.machinery.SourceFileLoader(fullname, path)
  31         spec = importlib.util.spec_from_loader(fullname, loader=loader)
  32         m = importlib.util.module_from_spec(spec)
  33         loader.exec_module(m)
  34         if not getattr(m, "name", None):
  35             m.name = path  # type: ignore
  36     except Exception as e:
  37         script_error_handler(path, e, msg=str(e))
  38     finally:
  39         sys.path[:] = oldpath
  40         return m

  43 def script_error_handler(path, exc, msg="", tb=False):
  44     """
  45         Handles all the user's script errors with
  46         an optional traceback
  47     """
  48     exception = type(exc).__name__
  49     if msg:
  50         exception = msg
  51     lineno = ""
  52     if hasattr(exc, "lineno"):
  53         lineno = str(exc.lineno)
  54     log_msg = "in script {}:{} {}".format(path, lineno, exception)
  55     if tb:
  56         etype, value, tback = sys.exc_info()
  57         tback = addonmanager.cut_traceback(tback, "invoke_addon")
  58         log_msg = log_msg + "\n" + "".join(traceback.format_exception(etype, value, tback))
  59     ctx.log.error(log_msg)

  65 class Script:
  66     """
  67         An addon that manages a single script.
  68     """
  69 
  70     def __init__(self, path: str, reload: bool) -> None:
  71         self.name = "scriptmanager:" + path
  72         self.path = path
  73         self.fullpath = os.path.expanduser(
  74             path.strip("'\" ")
  75         )
  76         self.ns = None
  77 
  78         if not os.path.isfile(self.fullpath):
  79             raise exceptions.OptionsError('No such script')
  80 
  81         self.reloadtask = None
  82         if reload:
  83             self.reloadtask = asyncio.ensure_future(self.watcher())
  84         else:
  85             self.loadscript()
  86 
  87     def done(self):
  88         if self.reloadtask:
  89             self.reloadtask.cancel()
  90 
  91     @property
  92     def addons(self):
  93         return [self.ns] if self.ns else []
  94 
  95     def loadscript(self):
  96         ctx.log.info("Loading script %s" % self.path)
  97         if self.ns:
  98             ctx.master.addons.remove(self.ns)
  99         self.ns = None
 100         with addonmanager.safecall():
 101             ns = load_script(self.fullpath)
 102             ctx.master.addons.register(ns)
 103             self.ns = ns
 104         if self.ns:
 105             # We're already running, so we have to explicitly register and
 106             # configure the addon
 107             ctx.master.addons.invoke_addon(self.ns, "running")
 108             try:
 109                 ctx.master.addons.invoke_addon(
 110                     self.ns,
 111                     "configure",
 112                     ctx.options.keys()
 113                 )
 114             except exceptions.OptionsError as e:
 115                 script_error_handler(self.fullpath, e, msg=str(e))
 116 
 117     async def watcher(self):
 118         last_mtime = 0
 119         while True:
 120             try:
 121                 mtime = os.stat(self.fullpath).st_mtime
 122             except FileNotFoundError:
 123                 ctx.log.info("Removing script %s" % self.path)
 124                 scripts = list(ctx.options.scripts)
 125                 scripts.remove(self.path)
 126                 ctx.options.update(scripts=scripts)
 127                 return
 128             if mtime > last_mtime:
 129                 self.loadscript()
 130                 last_mtime = mtime
 131             await asyncio.sleep(ReloadInterval)

 134 class ScriptLoader:
 135     """
 136         An addon that manages loading scripts from options.
 137     """
 138     def __init__(self):
 139         self.is_running = False
 140         self.addons = []
 141 
 142     def load(self, loader):
 143         loader.add_option(
 144             "scripts", typing.Sequence[str], [],
 145             "Execute a script."
 146         )
 147 
 148     def running(self):
 149         self.is_running = True
 150 
 151     @command.command("script.run")
 152     def script_run(self, flows: typing.Sequence[flow.Flow], path: mtypes.Path) -> None:
 153         """
 154             Run a script on the specified flows. The script is configured with
 155             the current options and all lifecycle events for each flow are
 156             simulated. Note that the load event is not invoked.
 157         """
 158         if not os.path.isfile(path):
 159             ctx.log.error('No such script: %s' % path)
 160             return
 161         mod = load_script(path)
 162         if mod:
 163             with addonmanager.safecall():
 164                 ctx.master.addons.invoke_addon(mod, "running")
 165                 ctx.master.addons.invoke_addon(
 166                     mod,
 167                     "configure",
 168                     ctx.options.keys()
 169                 )
 170                 for f in flows:
 171                     for evt, arg in eventsequence.iterate(f):
 172                         ctx.master.addons.invoke_addon(mod, evt, arg)
 173 
 174     def configure(self, updated):
 175         if "scripts" in updated:
 176             for s in ctx.options.scripts:
 177                 if ctx.options.scripts.count(s) > 1:
 178                     raise exceptions.OptionsError("Duplicate script")
 179 
 180             for a in self.addons[:]:
 181                 if a.path not in ctx.options.scripts:
 182                     ctx.log.info("Un-loading script: %s" % a.path)
 183                     ctx.master.addons.remove(a)
 184                     self.addons.remove(a)
 185 
 186             # The machinations below are to ensure that:
 187             #   - Scripts remain in the same order
 188             #   - Scripts are not initialized un-necessarily. If only a
 189             #   script's order in the script list has changed, it is just
 190             #   moved.
 191 
 192             current = {}
 193             for a in self.addons:
 194                 current[a.path] = a
 195 
 196             ordered = []
 197             newscripts = []
 198             for s in ctx.options.scripts:
 199                 if s in current:
 200                     ordered.append(current[s])
 201                 else:
 202                     sc = Script(s, True)
 203                     ordered.append(sc)
 204                     newscripts.append(sc)
 205 
 206             self.addons = ordered
 207 
 208             for s in newscripts:
 209                 ctx.master.addons.register(s)
 210                 if self.is_running:
 211                     # If we're already running, we configure and tell the addon
 212                     # we're up and running.
 213                     ctx.master.addons.invoke_addon(s, "running")
