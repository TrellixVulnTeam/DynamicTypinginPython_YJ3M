  29 # The underlying sorted list implementation expects the sort key to be stable
  30 # for the lifetime of the object. However, if we sort by size, for instance,
  31 # the sort order changes as the flow progresses through its lifecycle. We
  32 # address this through two means:
  33 #
  34 # - Let order keys cache the sort value by flow ID.
  35 #
  36 # - Add a facility to refresh items in the list by removing and re-adding them
  37 # when they are updated.
  38 
  39 
  40 class _OrderKey:
  41     def __init__(self, view):
  42         self.view = view
  43 
  44     def generate(self, f: mitmproxy.flow.Flow) -> typing.Any:  # pragma: no cover
  45         pass
  46 
  47     def refresh(self, f):
  48         k = self._key()
  49         old = self.view.settings[f][k]
  50         new = self.generate(f)
  51         if old != new:
  52             self.view._view.remove(f)
  53             self.view.settings[f][k] = new
  54             self.view._view.add(f)
  55             self.view.sig_view_refresh.send(self.view)
  56 
  57     def _key(self):
  58         return "_order_%s" % id(self)
  59 
  60     def __call__(self, f):
  61         if f.id in self.view._store:
  62             k = self._key()
  63             s = self.view.settings[f]
  64             if k in s:
  65                 return s[k]
  66             val = self.generate(f)
  67             s[k] = val
  68             return val
  69         else:
  70             return self.generate(f)

  98 class OrderKeySize(_OrderKey):
  99     def generate(self, f: mitmproxy.flow.Flow) -> int:
 100         if isinstance(f, http.HTTPFlow):
 101             size = 0
 102             if f.request.raw_content:
 103                 size += len(f.request.raw_content)
 104             if f.response and f.response.raw_content:
 105                 size += len(f.response.raw_content)
 106             return size
 107         elif isinstance(f, tcp.TCPFlow):
 108             size = 0
 109             for message in f.messages:
 110                 size += len(message.content)
 111             return size
 112         else:
 113             raise NotImplementedError()

 126 class View(collections.abc.Sequence):
 127     def __init__(self):
 128         super().__init__()
 129         self._store = collections.OrderedDict()
 130         self.filter = matchall
 131         # Should we show only marked flows?
 132         self.show_marked = False
 133 
 134         self.default_order = OrderRequestStart(self)
 135         self.orders = dict(
 136             time = OrderRequestStart(self), method = OrderRequestMethod(self),
 137             url = OrderRequestURL(self), size = OrderKeySize(self),
 138         )
 139         self.order_key = self.default_order
 140         self.order_reversed = False
 141         self.focus_follow = False
 142 
 143         self._view = sortedcontainers.SortedListWithKey(
 144             key = self.order_key
 145         )
 146 
 147         # The sig_view* signals broadcast events that affect the view. That is,
 148         # an update to a flow in the store but not in the view does not trigger
 149         # a signal. All signals are called after the view has been updated.
 150         self.sig_view_update = blinker.Signal()
 151         self.sig_view_add = blinker.Signal()
 152         self.sig_view_remove = blinker.Signal()
 153         # Signals that the view should be refreshed completely
 154         self.sig_view_refresh = blinker.Signal()
 155 
 156         # The sig_store* signals broadcast events that affect the underlying
 157         # store. If a flow is removed from just the view, sig_view_remove is
 158         # triggered. If it is removed from the store while it is also in the
 159         # view, both sig_store_remove and sig_view_remove are triggered.
 160         self.sig_store_remove = blinker.Signal()
 161         # Signals that the store should be refreshed completely
 162         self.sig_store_refresh = blinker.Signal()
 163 
 164         self.focus = Focus(self)
 165         self.settings = Settings(self)

 189     def _rev(self, idx: int) -> int:
 190         """
 191             Reverses an index, if needed
 192         """
 193         if self.order_reversed:
 194             if idx < 0:
 195                 idx = -idx - 1
 196             else:
 197                 idx = len(self._view) - idx - 1
 198                 if idx < 0:
 199                     raise IndexError
 200         return idx
 201 
 202     def __len__(self):
 203         return len(self._view)
 204 
 205     def __getitem__(self, offset) -> typing.Any:
 206         return self._view[self._rev(offset)]

 208     # Reflect some methods to the efficient underlying implementation

 210     def _bisect(self, f: mitmproxy.flow.Flow) -> int:
 211         v = self._view.bisect_right(f)
 212         return self._rev(v - 1) + 1

 214     def index(self, f: mitmproxy.flow.Flow, start: int = 0, stop: typing.Optional[int] = None) -> int:
 215         return self._rev(self._view.index(f, start, stop))

 220     def _order_key_name(self):
 221         return "_order_%s" % id(self.order_key)

 223     def _base_add(self, f):
 224         self.settings[f][self._order_key_name()] = self.order_key(f)
 225         self._view.add(f)

 227     def _refilter(self):
 228         self._view.clear()
 229         for i in self._store.values():
 230             if self.show_marked and not i.marked:
 231                 continue
 232             if self.filter(i):
 233                 self._base_add(i)
 234         self.sig_view_refresh.send(self)

 239     @command.command("view.focus.go")
 240     def go(self, offset: int) -> None:
 241         """
 242             Go to a specified offset. Positive offests are from the beginning of
 243             the view, negative from the end of the view, so that 0 is the first
 244             flow, -1 is the last flow.
 245         """
 246         if len(self) == 0:
 247             return
 248         if offset < 0:
 249             offset = len(self) + offset
 250         if offset < 0:
 251             offset = 0
 252         if offset > len(self) - 1:
 253             offset = len(self) - 1
 254         self.focus.flow = self[offset]
 255

 258         """
 259             Set focus to the next flow.
 260         """
 261         if self.focus.index is not None:
 262             idx = self.focus.index + 1
 263             if self.inbounds(idx):
 264                 self.focus.flow = self[idx]
 265         else:
 266             pass

 414     @command.command("view.flows.remove")
 415     def remove(self, flows: typing.Sequence[mitmproxy.flow.Flow]) -> None:
 416         """
 417             Removes the flow from the underlying store and the view.
 418         """
 419         for f in flows:
 420             if f.id in self._store:
 421                 if f.killable:
 422                     f.kill()
 423                 if f in self._view:
 424                     # We manually pass the index here because multiple flows may have the same
 425                     # sorting key, and we cannot reconstruct the index from that.
 426                     idx = self._view.index(f)
 427                     self._view.remove(f)
 428                     self.sig_view_remove.send(self, flow=f, index=idx)
 429                 del self._store[f.id]
 430                 self.sig_store_remove.send(self, flow=f)
 431         if len(flows) > 1:
 432             ctx.log.alert("Removed %s flows" % len(flows))

 434     @command.command("view.flows.resolve")
 435     def resolve(self, flow_spec: str) -> typing.Sequence[mitmproxy.flow.Flow]:
 436         """
 437             Resolve a flow list specification to an actual list of flows.
 438         """
 439         if flow_spec == "@all":
 440             return [i for i in self._store.values()]
 441         if flow_spec == "@focus":
 442             return [self.focus.flow] if self.focus.flow else []
 443         elif flow_spec == "@shown":
 444             return [i for i in self]
 445         elif flow_spec == "@hidden":
 446             return [i for i in self._store.values() if i not in self._view]
 447         elif flow_spec == "@marked":
 448             return [i for i in self._store.values() if i.marked]
 449         elif flow_spec == "@unmarked":
 450             return [i for i in self._store.values() if not i.marked]
 451         else:
 452             filt = flowfilter.parse(flow_spec)
 453             if not filt:
 454                 raise exceptions.CommandError("Invalid flow filter: %s" % flow_spec)
 455             return [i for i in self._store.values() if filt(i)]

 487     def add(self, flows: typing.Sequence[mitmproxy.flow.Flow]) -> None:
 488         """
 489             Adds a flow to the state. If the flow already exists, it is
 490             ignored.
 491         """
 492         for f in flows:
 493             if f.id not in self._store:
 494                 self._store[f.id] = f
 495                 if self.filter(f):
 496                     self._base_add(f)
 497                     if self.focus_follow:
 498                         self.focus.flow = f
 499                     self.sig_view_add.send(self, flow=f)

 590     def update(self, flows: typing.Sequence[mitmproxy.flow.Flow]) -> None:
 591         """
 592             Updates a list of flows. If flow is not in the state, it's ignored.
 593         """
 594         for f in flows:
 595             if f.id in self._store:
 596                 if self.filter(f):
 597                     if f not in self._view:
 598                         self._base_add(f)
 599                         if self.focus_follow:
 600                             self.focus.flow = f
 601                         self.sig_view_add.send(self, flow=f)
 602                     else:
 603                         # This is a tad complicated. The sortedcontainers
 604                         # implementation assumes that the order key is stable. If
 605                         # it changes mid-way Very Bad Things happen. We detect when
 606                         # this happens, and re-fresh the item.
 607                         self.order_key.refresh(f)
 608                         self.sig_view_update.send(self, flow=f)
 609                 else:
 610                     try:
 611                         idx = self._view.index(f)
 612                     except ValueError:
 613                         pass  # The value was not in the view
 614                     else:
 615                         self._view.remove(f)
 616                         self.sig_view_remove.send(self, flow=f, index=idx)

 619 class Focus:
 620     """
 621         Tracks a focus element within a View.
 622     """
 623     def __init__(self, v: View) -> None:
 624         self.view = v
 625         self._flow: typing.Optional[mitmproxy.flow.Flow] = None
 626         self.sig_change = blinker.Signal()
 627         if len(self.view):
 628             self.flow = self.view[0]
 629         v.sig_view_add.connect(self._sig_view_add)
 630         v.sig_view_remove.connect(self._sig_view_remove)
 631         v.sig_view_refresh.connect(self._sig_view_refresh)
 632 
 633     @property
 634     def flow(self) -> typing.Optional[mitmproxy.flow.Flow]:
 635         return self._flow
 636 
 637     @flow.setter
 638     def flow(self, f: typing.Optional[mitmproxy.flow.Flow]):
 639         if f is not None and f not in self.view:
 640             raise ValueError("Attempt to set focus to flow not in view")
 641         self._flow = f
 642         self.sig_change.send(self)
 643 
 644     @property
 645     def index(self) -> typing.Optional[int]:
 646         if self.flow:
 647             return self.view.index(self.flow)
 648         return None
 649 
 650     @index.setter
 651     def index(self, idx):
 652         if idx < 0 or idx > len(self.view) - 1:
 653             raise ValueError("Index out of view bounds")
 654         self.flow = self.view[idx]
 655 
 656     def _nearest(self, f, v):
 657         return min(v._bisect(f), len(v) - 1)
 658 
 659     def _sig_view_remove(self, view, flow, index):
 660         if len(view) == 0:
 661             self.flow = None
 662         elif flow is self.flow:
 663             self.index = min(index, len(self.view) - 1)
 664 
 665     def _sig_view_refresh(self, view):
 666         if len(view) == 0:
 667             self.flow = None
 668         elif self.flow is None:
 669             self.flow = view[0]
 670         elif self.flow not in view:
 671             self.flow = view[self._nearest(self.flow, view)]

 674         # We only have to act if we don't have a focus element
 675         if not self.flow:

 679 class Settings(collections.abc.Mapping):
 680     def __init__(self, view: View) -> None:
 681         self.view = view
 682         self._values: typing.MutableMapping[str, typing.Dict] = {}
 683         view.sig_store_remove.connect(self._sig_store_remove)
 684         view.sig_store_refresh.connect(self._sig_store_refresh)
 685 
 686     def __iter__(self) -> typing.Iterator:
 687         return iter(self._values)
 688 
 689     def __len__(self) -> int:
 690         return len(self._values)
 691 
 692     def __getitem__(self, f: mitmproxy.flow.Flow) -> dict:
 693         if f.id not in self.view._store:
 694             raise KeyError
 695         return self._values.setdefault(f.id, {})
 696 
 697     def _sig_store_remove(self, view, flow):
 698         if flow.id in self._values:
 699             del self._values[flow.id]
 700 
 701     def _sig_store_refresh(self, view):
 702         for fid in list(self._values.keys()):
 703             if fid not in view._store:
 704                 del self._values[fid]
