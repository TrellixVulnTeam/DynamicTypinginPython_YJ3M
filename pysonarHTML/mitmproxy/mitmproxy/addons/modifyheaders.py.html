  12 class ModifySpec(typing.NamedTuple):
  13     matches: flowfilter.TFilter
  14     subject: bytes
  15     replacement_str: str

  17     def read_replacement(self) -> bytes:
  18         """
  19         Process the replacement str. This usually just involves converting it to bytes.
  20         However, if it starts with `@`, we interpret the rest as a file path to read from.
  21 
  22         Raises:
  23             - IOError if the file cannot be read.
  24         """
  25         if self.replacement_str.startswith("@"):
  26             return Path(self.replacement_str[1:]).expanduser().read_bytes()
  27         else:
  28             # We could cache this at some point, but unlikely to be a problem.
  29             return strutils.escaped_str_to_bytes(self.replacement_str)

  36 def parse_modify_spec(option, subject_is_regex: bool) -> ModifySpec:
  37     """
  38         The form for the modify_* options is as follows:
  39 
  40             * modify_headers: [/flow-filter]/header-name/[@]header-value
  41             * modify_body: [/flow-filter]/search-regex/[@]replace
  42 
  43         The @ allows to provide a file path that is used to read the respective option.
  44         Both ModifyHeaders and ModifyBody use ModifySpec to represent a single rule.
  45 
  46         The first character specifies the separator. Example:
  47 
  48             :~q:foo:bar
  49 
  50         If only two clauses are specified, the flow filter is set to
  51         match universally (i.e. ".*"). Example:
  52 
  53             /foo/bar
  54 
  55         Clauses are parsed from left to right. Extra separators are taken to be
  56         part of the final clause. For instance, the last parameter (header-value or
  57         replace) below is "foo/bar/":
  58 
  59             /one/two/foo/bar/
  60     """
  61     sep, rem = option[0], option[1:]
  62     parts = rem.split(sep, 2)
  63     if len(parts) == 2:
  64         flow_filter = _match_all
  65         subject, replacement = parts
  66     elif len(parts) == 3:
  67         flow_filter_pattern, subject, replacement = parts
  68         flow_filter = flowfilter.parse(flow_filter_pattern)  # type: ignore
  69         if not flow_filter:
  70             raise ValueError(f"Invalid filter pattern: {flow_filter_pattern}")
  71     else:
  72         raise ValueError("Invalid number of parameters (2 or 3 are expected)")
  73 
  74     subject = strutils.escaped_str_to_bytes(subject)
  75     if subject_is_regex:
  76         try:
  77             re.compile(subject)
  78         except re.error as e:
  79             raise ValueError(f"Invalid regular expression {subject!r} ({e})")
  80 
  81     spec = ModifySpec(flow_filter, subject, replacement)
  82 
  83     try:
  84         spec.read_replacement()
  85     except IOError as e:
  86         raise ValueError(f"Invalid file path: {replacement[1:]} ({e})")
  87 
  88     return spec

  91 class ModifyHeaders:
  92     def __init__(self):
  93         self.replacements: typing.List[ModifySpec] = []
  94 
  95     def load(self, loader):
  96         loader.add_option(
  97             "modify_headers", typing.Sequence[str], [],
  98             """
  99             Header modify pattern of the form "[/flow-filter]/header-name/[@]header-value", where the
 100             separator can be any character. The @ allows to provide a file path that is used to read
 101             the header value string. An empty header-value removes existing header-name headers.
 102             """
 103         )
 104 
 105     def configure(self, updated):
 106         self.replacements = []
 107         if "modify_headers" in updated:
 108             for option in ctx.options.modify_headers:
 109                 try:
 110                     spec = parse_modify_spec(option, False)
 111                 except ValueError as e:
 112                     raise exceptions.OptionsError(f"Cannot parse modify_headers option {option}: {e}") from e
 113                 self.replacements.append(spec)
 114 
 115     def request(self, flow):
 116         if not flow.reply.has_message:
 117             self.run(flow, flow.request.headers)
 118 
 119     def response(self, flow):
 120         if not flow.reply.has_message:
 121             self.run(flow, flow.response.headers)
 122 
 123     def run(self, flow: http.HTTPFlow, hdrs: Headers) -> None:
 124         # unset all specified headers
 125         for spec in self.replacements:
 126             if spec.matches(flow):
 127                 hdrs.pop(spec.subject, None)
 128 
 129         # set all specified headers if the replacement string is not empty
 130         for spec in self.replacements:
 131             if spec.matches(flow):
 132                 try:
 133                     replacement = spec.read_replacement()
 134                 except IOError as e:
 135                     ctx.log.warn(f"Could not read replacement file: {e}")
 136                     continue
 137                 else:
 138                     if replacement:
 139                         hdrs.add(spec.subject, replacement)
