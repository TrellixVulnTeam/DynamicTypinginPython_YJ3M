  17 def indent(n: int, text: str) -> str:
  18     l = str(text).strip().splitlines()
  19     pad = " " * n
  20     return "\n".join(pad + i for i in l)

  23 def colorful(line, styles):
  24     yield u"    "  # we can already indent here
  25     for (style, text) in line:
  26         yield click.style(text, **styles.get(style, {}))

  29 class Dumper:
  30     def __init__(self, outfile=sys.stdout, errfile=sys.stderr):
  31         self.filter: flowfilter.TFilter = None
  32         self.outfp: typing.io.TextIO = outfile
  33         self.errfp: typing.io.TextIO = errfile

  35     def load(self, loader):
  36         loader.add_option(
  37             "flow_detail", int, 1,
  38             """
  39             The display detail level for flows in mitmdump: 0 (almost quiet) to 3 (very verbose).
  40               0: shortened request URL, response status code, WebSocket and TCP message notifications.
  41               1: full request URL with response status code
  42               2: 1 + HTTP headers
  43               3: 2 + full response content, content of WebSocket and TCP messages.
  44             """
  45         )
  46         loader.add_option(
  47             "dumper_default_contentview", str, "auto",
  48             "The default content view mode.",
  49             choices = [i.name.lower() for i in contentviews.views]
  50         )
  51         loader.add_option(
  52             "dumper_filter", typing.Optional[str], None,
  53             "Limit which flows are dumped."
  54         )

  56     def configure(self, updated):
  57         if "dumper_filter" in updated:
  58             if ctx.options.dumper_filter:
  59                 self.filter = flowfilter.parse(ctx.options.dumper_filter)
  60                 if not self.filter:
  61                     raise exceptions.OptionsError(
  62                         "Invalid filter expression: %s" % ctx.options.dumper_filter
  63                     )
  64             else:
  65                 self.filter = None

  67     def echo(self, text, ident=None, **style):
  68         if ident:
  69             text = indent(ident, text)
  70         click.secho(text, file=self.outfp, **style)
  71         if self.outfp:
  72             self.outfp.flush()

  74     def echo_error(self, text, **style):

  76         if self.errfp:
  77             self.errfp.flush()

  79     def _echo_headers(self, headers):
  80         for k, v in headers.fields:
  81             k = strutils.bytes_to_escaped_str(k)
  82             v = strutils.bytes_to_escaped_str(v)
  83             out = "{}: {}".format(
  84                 click.style(k, fg="blue"),
  85                 click.style(v)
  86             )
  87             self.echo(out, ident=4)

  89     def _echo_message(self, message, flow):
  90         _, lines, error = contentviews.get_message_content_view(
  91             ctx.options.dumper_default_contentview,
  92             message,
  93             flow
  94         )
  95         if error:
  96             ctx.log.debug(error)
  97 
  98         if ctx.options.flow_detail == 3:
  99             lines_to_echo = itertools.islice(lines, 70)
 100         else:
 101             lines_to_echo = lines
 102 
 103         styles = dict(
 104             highlight=dict(bold=True),
 105             offset=dict(fg="blue"),
 106             header=dict(fg="green", bold=True),
 107             text=dict(fg="green")
 108         )
 109 
 110         content = u"\r\n".join(
 111             u"".join(colorful(line, styles)) for line in lines_to_echo
 112         )
 113         if content:
 114             self.echo("")
 115             self.echo(content)
 116 
 117         if next(lines, None):
 118             self.echo("(cut off)", ident=4, dim=True)
 119 
 120         if ctx.options.flow_detail >= 2:
 121             self.echo("")

 123     def _echo_request_line(self, flow):
 124         if flow.client_conn:
 125             client = click.style(
 126                 strutils.escape_control_characters(
 127                     human.format_address(flow.client_conn.address)
 128                 )
 129             )
 130         elif flow.request.is_replay:
 131             client = click.style("[replay]", fg="yellow", bold=True)
 132         else:
 133             client = ""
 134 
 135         pushed = ' PUSH_PROMISE' if 'h2-pushed-stream' in flow.metadata else ''
 136         method = flow.request.method + pushed
 137         method_color = dict(
 138             GET="green",
 139             DELETE="red"
 140         ).get(method.upper(), "magenta")
 141         method = click.style(
 142             strutils.escape_control_characters(method),
 143             fg=method_color,
 144             bold=True
 145         )
 146         if ctx.options.showhost:
 147             url = flow.request.pretty_url
 148         else:
 149             url = flow.request.url
 150         terminalWidthLimit = max(shutil.get_terminal_size()[0] - 25, 50)
 151         if ctx.options.flow_detail < 1 and len(url) > terminalWidthLimit:
 152             url = url[:terminalWidthLimit] + "…"
 153         url = click.style(strutils.escape_control_characters(url), bold=True)
 154 
 155         http_version = ""
 156         if flow.request.http_version not in ("HTTP/1.1", "HTTP/1.0"):
 157             # We hide "normal" HTTP 1.
 158             http_version = " " + flow.request.http_version
 159 
 160         line = "{client}: {method} {url}{http_version}".format(
 161             client=client,
 162             method=method,
 163             url=url,
 164             http_version=http_version
 165         )
 166         self.echo(line)

 168     def _echo_response_line(self, flow):
 169         if flow.response.is_replay:
 170             replay = click.style("[replay] ", fg="yellow", bold=True)
 171         else:
 172             replay = ""
 173 
 174         code = flow.response.status_code
 175         code_color = None
 176         if 200 <= code < 300:
 177             code_color = "green"
 178         elif 300 <= code < 400:
 179             code_color = "magenta"
 180         elif 400 <= code < 600:
 181             code_color = "red"
 182         code = click.style(
 183             str(code),
 184             fg=code_color,
 185             bold=True,
 186             blink=(code == 418)
 187         )
 188         reason = click.style(
 189             strutils.escape_control_characters(flow.response.reason),
 190             fg=code_color,
 191             bold=True
 192         )
 193 
 194         if flow.response.raw_content is None:
 195             size = "(content missing)"
 196         else:
 197             size = human.pretty_size(len(flow.response.raw_content))
 198         size = click.style(size, bold=True)
 199 
 200         arrows = click.style(" <<", bold=True)
 201         if ctx.options.flow_detail == 1:
 202             # This aligns the HTTP response code with the HTTP request method:
 203             # 127.0.0.1:59519: GET http://example.com/
 204             #               << 304 Not Modified 0b
 205             arrows = " " * (len(human.format_address(flow.client_conn.address)) - 2) + arrows
 206 
 207         line = "{replay}{arrows} {code} {reason} {size}".format(
 208             replay=replay,
 209             arrows=arrows,
 210             code=code,
 211             reason=reason,
 212             size=size
 213         )
 214         self.echo(line)

 216     def echo_flow(self, f):
 217         if f.request:
 218             self._echo_request_line(f)
 219             if ctx.options.flow_detail >= 2:
 220                 self._echo_headers(f.request.headers)
 221             if ctx.options.flow_detail >= 3:
 222                 self._echo_message(f.request, f)
 223 
 224         if f.response:
 225             self._echo_response_line(f)
 226             if ctx.options.flow_detail >= 2:
 227                 self._echo_headers(f.response.headers)
 228             if ctx.options.flow_detail >= 3:
 229                 self._echo_message(f.response, f)
 230 
 231         if f.error:
 232             msg = strutils.escape_control_characters(f.error.msg)
 233             self.echo(" << {}".format(msg), bold=True, fg="red")

 235     def match(self, f):
 236         if ctx.options.flow_detail == 0:
 237             return False
 238         if not self.filter:
 239             return True
 240         elif flowfilter.match(self.filter, f):
 241             return True
 242         return False
 243 
 244     def response(self, f):
 245         if self.match(f):
 246             self.echo_flow(f)
 247 
 248     def error(self, f):
 249         if self.match(f):
 250             self.echo_flow(f)

 252     def websocket_error(self, f):

 260     def websocket_message(self, f):
 261         if self.match(f):
 262             message = f.messages[-1]
 263             self.echo(f.message_info(message))
 264             if ctx.options.flow_detail >= 3:
 265                 message = message.from_state(message.get_state())
 266                 message.content = message.content.encode() if isinstance(message.content, str) else message.content
 267                 self._echo_message(message, f)
 268 
 269     def websocket_end(self, f):
 270         if self.match(f):
 271             self.echo("WebSocket connection closed by {}: {} {}, {}".format(
 272                 f.close_sender,
 273                 f.close_code,
 274                 f.close_message,
 275                 f.close_reason))

 277     def tcp_error(self, f):
 278         self.echo_error(
 279             "Error in TCP connection to {}: {}".format(
 280                 human.format_address(f.server_conn.address), f.error
 281             ),
 282             fg="red"
 283         )
 284 
 285     def tcp_message(self, f):
 286         if self.match(f):
 287             message = f.messages[-1]
 288             direction = "->" if message.from_client else "<-"
 289             self.echo("{client} {direction} tcp {direction} {server}".format(
 290                 client=human.format_address(f.client_conn.address),
 291                 server=human.format_address(f.server_conn.address),
 292                 direction=direction,
 293             ))
 294             if ctx.options.flow_detail >= 3:
 295                 self._echo_message(message, f)
