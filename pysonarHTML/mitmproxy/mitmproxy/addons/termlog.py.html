   1 import sys
   2 import click
   3 
   4 from mitmproxy import log
   5 from mitmproxy import ctx
   6 
   7 # These get over-ridden by the save execution context. Keep them around so we
   8 # can log directly.
   9 realstdout = sys.stdout
  10 realstderr = sys.stderr
  11 
  12 
  13 class TermLog:
  14     def __init__(self, outfile=None):
  15         self.outfile = outfile
  16 
  17     def load(self, loader):
  18         loader.add_option(
  19             "termlog_verbosity", str, 'info',
  20             "Log verbosity.",
  21             choices=log.LogTierOrder
  22         )
  23 
  24     def log(self, e):
  25         if log.log_tier(e.level) == log.log_tier("error"):
  26             outfile = self.outfile or realstderr
  27         else:
  28             outfile = self.outfile or realstdout
  29 
  30         if log.log_tier(ctx.options.termlog_verbosity) >= log.log_tier(e.level):
  31             click.secho(
  32                 e.msg,
  33                 file=outfile,
  34                 fg=dict(error="red", warn="yellow",
  35                         alert="magenta").get(e.level),
  36                 dim=(e.level == "debug"),
  37                 err=(e.level == "error")
  38             )
