  29     def open_file(self, path):
  30         if path.startswith("+"):
  31             path = path[1:]
  32             mode = "ab"
  33         else:
  34             mode = "wb"
  35         path = os.path.expanduser(path)
  36         return open(path, mode)

  38     def start_stream_to_path(self, path, flt):
  39         try:
  40             f = self.open_file(path)
  41         except IOError as v:
  42             raise exceptions.OptionsError(str(v))
  43         self.stream = io.FilteredFlowWriter(f, flt)
  44         self.active_flows = set()

  46     def configure(self, updated):
  47         # We're already streaming - stop the previous stream and restart
  48         if "save_stream_filter" in updated:
  49             if ctx.options.save_stream_filter:
  50                 self.filt = flowfilter.parse(ctx.options.save_stream_filter)
  51                 if not self.filt:
  52                     raise exceptions.OptionsError(
  53                         "Invalid filter specification: %s" % ctx.options.save_stream_filter
  54                     )
  55             else:
  56                 self.filt = None
  57         if "save_stream_file" in updated or "save_stream_filter" in updated:
  58             if self.stream:
  59                 self.done()
  60             if ctx.options.save_stream_file:
  61                 self.start_stream_to_path(ctx.options.save_stream_file, self.filt)

  63     @command.command("save.file")
  64     def save(self, flows: typing.Sequence[flow.Flow], path: mitmproxy.types.Path) -> None:
  65         """
  66             Save flows to a file. If the path starts with a +, flows are
  67             appended to the file, otherwise it is over-written.
  68         """
  69         try:
  70             f = self.open_file(path)
  71         except IOError as v:
  72             raise exceptions.CommandError(v) from v
  73         stream = io.FlowWriter(f)
  74         for i in flows:
  75             stream.add(i)
  76         f.close()
  77         ctx.log.alert("Saved %s flows." % len(flows))

 106     def done(self):
 107         if self.stream:
 108             for f in self.active_flows:
