  17 def headername(spec: str):
  18     if not (spec.startswith("header[") and spec.endswith("]")):
  19         raise exceptions.CommandError("Invalid header spec: %s" % spec)
  20     return spec[len("header["):-1].strip()

  27 def extract(cut: str, f: flow.Flow) -> typing.Union[str, bytes]:
  28     path = cut.split(".")
  29     current: typing.Any = f
  30     for i, spec in enumerate(path):
  31         if spec.startswith("_"):
  32             raise exceptions.CommandError("Can't access internal attribute %s" % spec)
  33 
  34         part = getattr(current, spec, None)
  35         if i == len(path) - 1:
  36             if spec == "port" and is_addr(current):
  37                 return str(current[1])
  38             if spec == "host" and is_addr(current):
  39                 return str(current[0])
  40             elif spec.startswith("header["):
  41                 if not current:
  42                     return ""
  43                 return current.headers.get(headername(spec), "")
  44             elif isinstance(part, bytes):
  45                 return part
  46             elif isinstance(part, bool):
  47                 return "true" if part else "false"
  48             elif isinstance(part, certs.Cert):
  49                 return part.to_pem().decode("ascii")
  50         current = part
  51     return str(current or "")

  54 class Cut:
  55     @command.command("cut")
  56     def cut(
  57         self,
  58         flows: typing.Sequence[flow.Flow],
  59         cuts: mitmproxy.types.CutSpec,
  60     ) -> mitmproxy.types.Data:
  61         """
  62             Cut data from a set of flows. Cut specifications are attribute paths
  63             from the base of the flow object, with a few conveniences - "port"
  64             and "host" retrieve parts of an address tuple, ".header[key]"
  65             retrieves a header value. Return values converted to strings or
  66             bytes: SSL certificates are converted to PEM format, bools are "true"
  67             or "false", "bytes" are preserved, and all other values are
  68             converted to strings.
  69         """
  70         ret: typing.List[typing.List[typing.Union[str, bytes]]] = []
  71         for f in flows:
  72             ret.append([extract(c, f) for c in cuts])
  73         return ret  # type: ignore

  75     @command.command("cut.save")
  76     def save(
  77         self,
  78         flows: typing.Sequence[flow.Flow],
  79         cuts: mitmproxy.types.CutSpec,
  80         path: mitmproxy.types.Path
  81     ) -> None:
  82         """
  83             Save cuts to file. If there are multiple flows or cuts, the format
  84             is UTF-8 encoded CSV. If there is exactly one row and one column,
  85             the data is written to file as-is, with raw bytes preserved. If the
  86             path is prefixed with a "+", values are appended if there is an
  87             existing file.
  88         """
  89         append = False
  90         if path.startswith("+"):
  91             append = True
  92             epath = os.path.expanduser(path[1:])
  93             path = mitmproxy.types.Path(epath)
  94         try:
  95             if len(cuts) == 1 and len(flows) == 1:
  96                 with open(path, "ab" if append else "wb") as fp:
  97                     if fp.tell() > 0:
  98                         # We're appending to a file that already exists and has content
  99                         fp.write(b"\n")
 100                     v = extract(cuts[0], flows[0])
 101                     if isinstance(v, bytes):
 102                         fp.write(v)
 103                     else:
 104                         fp.write(v.encode("utf8"))
 105                 ctx.log.alert("Saved single cut.")
 106             else:
 107                 with open(path, "a" if append else "w", newline='', encoding="utf8") as fp:
 108                     writer = csv.writer(fp)
 109                     for f in flows:
 110                         vals = [extract(c, f) for c in cuts]
 111                         writer.writerow(
 112                             [strutils.always_str(x) or "" for x in vals]  # type: ignore
 113                         )
 114                 ctx.log.alert("Saved %s cuts over %d flows as CSV." % (len(cuts), len(flows)))
 115         except IOError as e:
 116             ctx.log.error(str(e))

 118     @command.command("cut.clip")
 119     def clip(
 120         self,
 121         flows: typing.Sequence[flow.Flow],
 122         cuts: mitmproxy.types.CutSpec,
 123     ) -> None:
 124         """
 125             Send cuts to the clipboard. If there are multiple flows or cuts, the
 126             format is UTF-8 encoded CSV. If there is exactly one row and one
 127             column, the data is written to file as-is, with raw bytes preserved.
 128         """
 129         v: typing.Union[str, bytes]
 130         fp = io.StringIO(newline="")
 131         if len(cuts) == 1 and len(flows) == 1:
 132             v = extract(cuts[0], flows[0])
 133             fp.write(strutils.always_str(v))  # type: ignore
 134             ctx.log.alert("Clipped single cut.")
 135         else:
 136             writer = csv.writer(fp)
 137             for f in flows:
 138                 vals = [extract(c, f) for c in cuts]
 139                 writer.writerow(
 140                     [strutils.always_str(v) for v in vals]
 141                 )
 142             ctx.log.alert("Clipped %s cuts as CSV." % len(cuts))
 143         try:
 144             pyperclip.copy(fp.getvalue())
 145         except pyperclip.PyperclipException as e:
 146             ctx.log.error(str(e))
