   9 class CommandHistory:
  10     VACUUM_SIZE = 1024
  11 
  12     def __init__(self) -> None:
  13         self.history: typing.List[str] = []
  14         self.filtered_history: typing.List[str] = [""]
  15         self.current_index: int = 0
  16 
  17     def load(self, loader):
  18         loader.add_option(
  19             "command_history", bool, True,
  20             """Persist command history between mitmproxy invocations."""
  21         )
  22 
  23     @property
  24     def history_file(self) -> pathlib.Path:
  25         return pathlib.Path(os.path.expanduser(ctx.options.confdir)) / "command_history"
  26 
  27     def running(self):
  28         # FIXME: We have a weird bug where the contract for configure is not followed and it is never called with
  29         # confdir or command_history as updated.
  30         self.configure("command_history")  # pragma: no cover
  31 
  32     def configure(self, updated):
  33         if "command_history" in updated or "confdir" in updated:
  34             if ctx.options.command_history and self.history_file.is_file():
  35                 self.history = self.history_file.read_text().splitlines()
  36                 self.set_filter('')
  37 
  38     def done(self):
  39         if ctx.options.command_history and len(self.history) >= self.VACUUM_SIZE:
  40             # vacuum history so that it doesn't grow indefinitely.
  41             history_str = "\n".join(self.history[-self.VACUUM_SIZE // 2:]) + "\n"
  42             self.history_file.write_text(history_str)
  43 
  44     @command.command("commands.history.add")
  45     def add_command(self, command: str) -> None:
  46         if not command.strip():
  47             return
  48 
  49         self.history.append(command)
  50         if ctx.options.command_history:
  51             with self.history_file.open("a") as f:
  52                 f.write(f"{command}\n")
  53                 f.close()
  54 
  55         self.set_filter('')
  56 
  57     @command.command("commands.history.get")
  58     def get_history(self) -> typing.Sequence[str]:
  59         """Get the entire command history."""
  60         return self.history.copy()
  61 
  62     @command.command("commands.history.clear")
  63     def clear_history(self):
  64         if self.history_file.exists():
  65             self.history_file.unlink()
  66         self.history = []
  67         self.set_filter('')
  68 
  69     # Functionality to provide a filtered list that can be iterated through.
  70 
  71     @command.command("commands.history.filter")
  72     def set_filter(self, prefix: str) -> None:
  73         self.filtered_history = [
  74             cmd
  75             for cmd in self.history
  76             if cmd.startswith(prefix)
  77         ]
  78         self.filtered_history.append(prefix)
  79         self.current_index = len(self.filtered_history) - 1
  80 
  81     @command.command("commands.history.next")
  82     def get_next(self) -> str:
  83         self.current_index = min(self.current_index + 1, len(self.filtered_history) - 1)
  84         return self.filtered_history[self.current_index]
  85 
  86     @command.command("commands.history.prev")
  87     def get_prev(self) -> str:
  88         self.current_index = max(0, self.current_index - 1)
  89         return self.filtered_history[self.current_index]
