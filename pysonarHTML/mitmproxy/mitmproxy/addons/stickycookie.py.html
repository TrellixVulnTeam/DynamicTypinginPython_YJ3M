  11 def ckey(attrs: Dict[str, str], f: http.HTTPFlow) -> TOrigin:
  12     """
  13         Returns a (domain, port, path) tuple.
  14     """
  15     domain = f.request.host
  16     path = "/"
  17     if "domain" in attrs:
  18         domain = attrs["domain"]
  19     if "path" in attrs:
  20         path = attrs["path"]
  21     return (domain, f.request.port, path)

  24 def domain_match(a: str, b: str) -> bool:
  25     if cookiejar.domain_match(a, b):  # type: ignore
  26         return True
  27     elif cookiejar.domain_match(a, b.strip(".")):  # type: ignore
  28         return True
  29     return False

  32 class StickyCookie:
  33     def __init__(self):
  34         self.jar: Dict[TOrigin, Dict[str, str]] = collections.defaultdict(dict)
  35         self.flt: Optional[flowfilter.TFilter] = None
  36 
  37     def load(self, loader):
  38         loader.add_option(
  39             "stickycookie", Optional[str], None,
  40             "Set sticky cookie filter. Matched against requests."
  41         )
  42 
  43     def configure(self, updated):
  44         if "stickycookie" in updated:
  45             if ctx.options.stickycookie:
  46                 flt = flowfilter.parse(ctx.options.stickycookie)
  47                 if not flt:
  48                     raise exceptions.OptionsError(
  49                         "stickycookie: invalid filter expression: %s" % ctx.options.stickycookie
  50                     )
  51                 self.flt = flt
  52             else:
  53                 self.flt = None
  54 
  55     def response(self, flow: http.HTTPFlow):
  56         assert flow.response
  57         if self.flt:
  58             for name, (value, attrs) in flow.response.cookies.items(multi=True):
  59                 # FIXME: We now know that Cookie.py screws up some cookies with
  60                 # valid RFC 822/1123 datetime specifications for expiry. Sigh.
  61                 dom_port_path = ckey(attrs, flow)
  62 
  63                 if domain_match(flow.request.host, dom_port_path[0]):
  64                     if cookies.is_expired(attrs):
  65                         # Remove the cookie from jar
  66                         self.jar[dom_port_path].pop(name, None)
  67 
  68                         # If all cookies of a dom_port_path have been removed
  69                         # then remove it from the jar itself
  70                         if not self.jar[dom_port_path]:
  71                             self.jar.pop(dom_port_path, None)
  72                     else:
  73                         self.jar[dom_port_path][name] = value
  74 
  75     def request(self, flow: http.HTTPFlow):
  76         if self.flt:
  77             cookie_list: List[Tuple[str, str]] = []
  78             if flowfilter.match(self.flt, flow):
  79                 for (domain, port, path), c in self.jar.items():
  80                     match = [
  81                         domain_match(flow.request.host, domain),
  82                         flow.request.port == port,
  83                         flow.request.path.startswith(path)
  84                     ]
  85                     if all(match):
  86                         cookie_list.extend(c.items())
  87             if cookie_list:
  88                 # FIXME: we need to formalise this...
  89                 flow.metadata["stickycookie"] = True
  90                 flow.request.headers["cookie"] = cookies.format_cookie_header(cookie_list)
