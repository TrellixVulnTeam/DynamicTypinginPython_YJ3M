  82     @command.command("replay.server")
  83     def load_flows(self, flows: typing.Sequence[flow.Flow]) -> None:
  84         """
  85             Replay server responses from flows.
  86         """
  87         self.flowmap = {}
  88         for f in flows:
  89             if isinstance(f, http.HTTPFlow):
  90                 lst = self.flowmap.setdefault(self._hash(f), [])
  91                 lst.append(f)
  92         ctx.master.addons.trigger("update", [])

  94     @command.command("replay.server.file")
  95     def load_file(self, path: mitmproxy.types.Path) -> None:
  96         try:
  97             flows = io.read_flows_from_paths([path])
  98         except exceptions.FlowReadException as e:
  99             raise exceptions.CommandError(str(e))
 100         self.load_flows(flows)

 114     def _hash(self, flow: http.HTTPFlow) -> typing.Hashable:
 115         """
 116             Calculates a loose hash of the flow request.
 117         """
 118         r = flow.request
 119         _, _, path, _, query, _ = urllib.parse.urlparse(r.url)
 120         queriesArray = urllib.parse.parse_qsl(query, keep_blank_values=True)
 121 
 122         key: typing.List[typing.Any] = [str(r.scheme), str(r.method), str(path)]
 123         if not ctx.options.server_replay_ignore_content:
 124             if ctx.options.server_replay_ignore_payload_params and r.multipart_form:
 125                 key.extend(
 126                     (k, v)
 127                     for k, v in r.multipart_form.items(multi=True)
 128                     if k.decode(errors="replace") not in ctx.options.server_replay_ignore_payload_params
 129                 )
 130             elif ctx.options.server_replay_ignore_payload_params and r.urlencoded_form:
 131                 key.extend(
 132                     (k, v)
 133                     for k, v in r.urlencoded_form.items(multi=True)
 134                     if k not in ctx.options.server_replay_ignore_payload_params
 135                 )
 136             else:
 137                 key.append(str(r.raw_content))
 138 
 139         if not ctx.options.server_replay_ignore_host:
 140             key.append(r.pretty_host)
 141         if not ctx.options.server_replay_ignore_port:
 142             key.append(r.port)
 143 
 144         filtered = []
 145         ignore_params = ctx.options.server_replay_ignore_params or []
 146         for p in queriesArray:
 147             if p[0] not in ignore_params:
 148                 filtered.append(p)
 149         for p in filtered:
 150             key.append(p[0])
 151             key.append(p[1])
 152 
 153         if ctx.options.server_replay_use_headers:
 154             headers = []
 155             for i in ctx.options.server_replay_use_headers:
 156                 v = r.headers.get(i)
 157                 headers.append((i, v))
 158             key.append(headers)
 159         return hashlib.sha256(
 160             repr(key).encode("utf8", "surrogateescape")
 161         ).digest()

 163     def next_flow(self, flow: http.HTTPFlow) -> typing.Optional[http.HTTPFlow]:
 164         """
 165             Returns the next flow object, or None if no matching flow was
 166             found.
 167         """
 168         hash = self._hash(flow)
 169         if hash in self.flowmap:
 170             if ctx.options.server_replay_nopop:
 171                 return next((
 172                     flow
 173                     for flow in self.flowmap[hash]
 174                     if flow.response
 175                 ), None)
 176             else:
 177                 ret = self.flowmap[hash].pop(0)
 178                 while not ret.response:
 179                     if self.flowmap[hash]:
 180                         ret = self.flowmap[hash].pop(0)
 181                     else:
 182                         del self.flowmap[hash]
 183                         return None
 184                 if not self.flowmap[hash]:
 185                     del self.flowmap[hash]
 186                 return ret
 187         else:
 188             return None

 190     def configure(self, updated):
 191         if not self.configured and ctx.options.server_replay:
 192             self.configured = True
 193             try:
 194                 flows = io.read_flows_from_paths(ctx.options.server_replay)
 195             except exceptions.FlowReadException as e:
 196                 raise exceptions.OptionsError(str(e))
 197             self.load_flows(flows)

 199     def request(self, f: http.HTTPFlow) -> None:
 200         if self.flowmap:
 201             rflow = self.next_flow(f)
 202             if rflow:
 203                 assert rflow.response
 204                 response = rflow.response.copy()
 205                 response.is_replay = True
 206                 if ctx.options.server_replay_refresh:
 207                     response.refresh()
 208                 f.response = response
 209             elif ctx.options.server_replay_kill_extra:
 210                 ctx.log.warn(
 211                     "server_playback: killed non-replay request {}".format(
 212                         f.request.url
 213                     )
 214                 )
 215                 assert f.reply
 216                 f.reply.kill()
