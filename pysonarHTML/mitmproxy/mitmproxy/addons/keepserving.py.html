   5 class KeepServing:
   6     def load(self, loader):
   7         loader.add_option(
   8             "keepserving", bool, False,
   9             """
  10             Continue serving after client playback, server playback or file
  11             read. This option is ignored by interactive tools, which always keep
  12             serving.
  13             """
  14         )
  15 
  16     def keepgoing(self) -> bool:
  17         checks = [
  18             "readfile.reading",
  19             "replay.client.count",
  20             "replay.server.count",
  21         ]
  22         return any([ctx.master.commands.call(c) for c in checks])
  23 
  24     def shutdown(self):  # pragma: no cover
  25         ctx.master.shutdown()
  26 
  27     async def watch(self):
  28         while True:
  29             await asyncio.sleep(0.1)
  30             if not self.keepgoing():
  31                 self.shutdown()
  32 
  33     def running(self):
  34         opts = [
  35             ctx.options.client_replay,
  36             ctx.options.server_replay,
  37             ctx.options.rfile,
  38         ]
  39         if any(opts) and not ctx.options.keepserving:
  40             asyncio.get_event_loop().create_task(self.watch())
