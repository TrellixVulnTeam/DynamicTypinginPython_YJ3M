  20 def mkauth(username: str, password: str, scheme: str = "basic") -> str:
  21     """
  22     Craft a basic auth string
  23     """
  24     v = binascii.b2a_base64(
  25         (username + ":" + password).encode("utf8")
  26     ).decode("ascii")
  27     return scheme + " " + v

  30 def parse_http_basic_auth(s: str) -> Tuple[str, str, str]:
  31     """
  32     Parse a basic auth header.
  33     Raises a ValueError if the input is invalid.
  34     """
  35     scheme, authinfo = s.split()
  36     if scheme.lower() != "basic":
  37         raise ValueError("Unknown scheme")
  38     try:
  39         user, password = binascii.a2b_base64(authinfo.encode()).decode("utf8", "replace").split(":")
  40     except binascii.Error as e:
  41         raise ValueError(str(e))
  42     return scheme, user, password

  45 class ProxyAuth:
  46     def __init__(self):
  47         self.nonanonymous = False
  48         self.htpasswd = None
  49         self.singleuser = None
  50         self.ldapconn = None
  51         self.ldapserver = None
  52         self.authenticated: MutableMapping[connections.ClientConnection, Tuple[str, str]] = weakref.WeakKeyDictionary()
  53         """Contains all connections that are permanently authenticated after an HTTP CONNECT"""
  54 
  55     def load(self, loader):
  56         loader.add_option(
  57             "proxyauth", Optional[str], None,
  58             """
  59             Require proxy authentication. Format:
  60             "username:pass",
  61             "any" to accept any user/pass combination,
  62             "@path" to use an Apache htpasswd file,
  63             or "ldap[s]:url_server_ldap:dn_auth:password:dn_subtree" for LDAP authentication.
  64             """
  65         )
  66 
  67     def enabled(self) -> bool:
  68         return any([self.nonanonymous, self.htpasswd, self.singleuser, self.ldapconn, self.ldapserver])
  69 
  70     def is_proxy_auth(self) -> bool:
  71         """
  72         Returns:
  73             - True, if authentication is done as if mitmproxy is a proxy
  74             - False, if authentication is done as if mitmproxy is a HTTP server
  75         """
  76         return ctx.options.mode == "regular" or ctx.options.mode.startswith("upstream:")
  77 
  78     def which_auth_header(self) -> str:
  79         if self.is_proxy_auth():
  80             return 'Proxy-Authorization'
  81         else:
  82             return 'Authorization'
  83 
  84     def auth_required_response(self) -> http.HTTPResponse:
  85         if self.is_proxy_auth():
  86             return http.make_error_response(
  87                 status_codes.PROXY_AUTH_REQUIRED,
  88                 headers=mitmproxy.net.http.Headers(Proxy_Authenticate='Basic realm="{}"'.format(REALM)),
  89             )
  90         else:
  91             return http.make_error_response(
  92                 status_codes.UNAUTHORIZED,
  93                 headers=mitmproxy.net.http.Headers(WWW_Authenticate='Basic realm="{}"'.format(REALM)),
  94             )
  95 
  96     def check(self, f: http.HTTPFlow) -> Optional[Tuple[str, str]]:
  97         """
  98         Check if a request is correctly authenticated.
  99         Returns:
 100             - a (username, password) tuple if successful,
 101             - None, otherwise.
 102         """
 103         auth_value = f.request.headers.get(self.which_auth_header(), "")
 104         try:
 105             scheme, username, password = parse_http_basic_auth(auth_value)
 106         except ValueError:
 107             return None
 108 
 109         if self.nonanonymous:
 110             return username, password
 111         elif self.singleuser:
 112             if self.singleuser == [username, password]:
 113                 return username, password
 114         elif self.htpasswd:
 115             if self.htpasswd.check_password(username, password):
 116                 return username, password
 117         elif self.ldapconn:
 118             if not username or not password:
 119                 return None
 120             self.ldapconn.search(ctx.options.proxyauth.split(':')[4], '(cn=' + username + ')')
 121             if self.ldapconn.response:
 122                 conn = ldap3.Connection(
 123                     self.ldapserver,
 124                     self.ldapconn.response[0]['dn'],
 125                     password,
 126                     auto_bind=True)
 127                 if conn:
 128                     return username, password
 129         return None

 131     def authenticate(self, f: http.HTTPFlow) -> bool:
 132         valid_credentials = self.check(f)
 133         if valid_credentials:
 134             f.metadata["proxyauth"] = valid_credentials
 135             del f.request.headers[self.which_auth_header()]
 136             return True
 137         else:
 138             f.response = self.auth_required_response()
 139             return False

 142     def configure(self, updated):
 143         if "proxyauth" in updated:
 144             self.nonanonymous = False
 145             self.singleuser = None
 146             self.htpasswd = None
 147             self.ldapserver = None
 148             if ctx.options.proxyauth:
 149                 if ctx.options.proxyauth == "any":
 150                     self.nonanonymous = True
 151                 elif ctx.options.proxyauth.startswith("@"):
 152                     p = ctx.options.proxyauth[1:]
 153                     try:
 154                         self.htpasswd = passlib.apache.HtpasswdFile(p)
 155                     except (ValueError, OSError):
 156                         raise exceptions.OptionsError(
 157                             "Could not open htpasswd file: %s" % p
 158                         )
 159                 elif ctx.options.proxyauth.startswith("ldap"):
 160                     parts = ctx.options.proxyauth.split(':')
 161                     if len(parts) != 5:
 162                         raise exceptions.OptionsError(
 163                             "Invalid ldap specification"
 164                         )
 165                     security = parts[0]
 166                     ldap_server = parts[1]
 167                     dn_baseauth = parts[2]
 168                     password_baseauth = parts[3]
 169                     if security == "ldaps":
 170                         server = ldap3.Server(ldap_server, use_ssl=True)
 171                     elif security == "ldap":
 172                         server = ldap3.Server(ldap_server)
 173                     else:
 174                         raise exceptions.OptionsError(
 175                             "Invalid ldap specification on the first part"
 176                         )
 177                     conn = ldap3.Connection(
 178                         server,
 179                         dn_baseauth,
 180                         password_baseauth,
 181                         auto_bind=True)
 182                     self.ldapconn = conn
 183                     self.ldapserver = server
 184                 else:
 185                     parts = ctx.options.proxyauth.split(':')
 186                     if len(parts) != 2:
 187                         raise exceptions.OptionsError(
 188                             "Invalid single-user auth specification."
 189                         )
 190                     self.singleuser = parts
 191         if self.enabled():
 192             if ctx.options.mode == "transparent":
 193                 raise exceptions.OptionsError(
 194                     "Proxy Authentication not supported in transparent mode."
 195                 )
 196             if ctx.options.mode == "socks5":
 197                 raise exceptions.OptionsError(
 198                     "Proxy Authentication not supported in SOCKS mode. "
 199                     "https://github.com/mitmproxy/mitmproxy/issues/738"
 200                 )
 201                 # TODO: check for multiple auth options
 202 
 203     def http_connect(self, f: http.HTTPFlow) -> None:
 204         if self.enabled():
 205             if self.authenticate(f):
 206                 self.authenticated[f.client_conn] = f.metadata["proxyauth"]
 207 
 208     def requestheaders(self, f: http.HTTPFlow) -> None:
 209         if self.enabled():
 210             # Is this connection authenticated by a previous HTTP CONNECT?
 211             if f.client_conn in self.authenticated:
 212                 f.metadata["proxyauth"] = self.authenticated[f.client_conn]
 213                 return
 214             self.authenticate(f)
