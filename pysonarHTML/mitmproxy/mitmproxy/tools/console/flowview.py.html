 131     def view_tcp_stream(self) -> urwid.Widget:
 132         flow = self.flow
 133         assert isinstance(flow, tcp.TCPFlow)
 134 
 135         if not flow.messages:
 136             return searchable.Searchable([urwid.Text(("highlight", "No messages."))])
 137 
 138         viewmode = self.master.commands.call("console.flowview.mode")
 139 
 140         # Merge adjacent TCP "messages". For detailed explanation of this code block see:
 141         # https://github.com/mitmproxy/mitmproxy/pull/3970/files/469bd32582f764f9a29607efa4f5b04bd87961fb#r418670880
 142         from_client = None
 143         messages = []
 144         for message in flow.messages:
 145             if message.from_client is not from_client:
 146                 messages.append(message.content)
 147                 from_client = message.from_client
 148             else:
 149                 messages[-1] += message.content
 150 
 151         widget_lines = []
 152 
 153         from_client = flow.messages[0].from_client
 154         for m in messages:
 155             _, lines, _ = contentviews.get_tcp_content_view(viewmode, m)
 156 
 157             for line in lines:
 158                 if from_client:
 159                     line.insert(0, ("from_client", f"{common.SYMBOL_FROM_CLIENT} "))
 160                 else:
 161                     line.insert(0, ("to_client", f"{common.SYMBOL_TO_CLIENT} "))
 162 
 163                 widget_lines.append(urwid.Text(line))
 164 
 165             from_client = not from_client
 166 
 167         if flow.intercepted:
 168             markup = widget_lines[-1].get_text()[0]
 169             widget_lines[-1].set_text(("intercept", markup))
 170 
 171         widget_lines.insert(0, self._contentview_status_bar(viewmode.capitalize(), viewmode))
 172 
 173         return searchable.Searchable(widget_lines)

 178     def content_view(self, viewmode, message):
 179         if message.raw_content is None:
 180             msg, body = "", [urwid.Text([("error", "[content missing]")])]
 181             return msg, body
 182         else:
 183             full = self.master.commands.execute("view.settings.getval @focus fullcontents false")
 184             if full == "true":
 185                 limit = sys.maxsize
 186             else:
 187                 limit = ctx.options.content_view_lines_cutoff
 188 
 189             flow_modify_cache_invalidation = hash((
 190                 message.raw_content,
 191                 message.headers.fields,
 192                 getattr(message, "path", None),
 193             ))
 194             # we need to pass the message off-band because it's not hashable
 195             self._get_content_view_message = message
 196             return self._get_content_view(viewmode, limit, flow_modify_cache_invalidation)
 197 
 198     @lru_cache(maxsize=200)
 199     def _get_content_view(self, viewmode, max_lines, _):
 200         message = self._get_content_view_message
 201         self._get_content_view_message = None
 202         description, lines, error = contentviews.get_message_content_view(
 203             viewmode, message, self.flow
 204         )
 205         if error:
 206             self.master.log.debug(error)
 207         # Give hint that you have to tab for the response.
 208         if description == "No content" and isinstance(message, http.HTTPRequest):
 209             description = "No request content"
 210 
 211         # If the users has a wide terminal, he gets fewer lines; this should not be an issue.
 212         chars_per_line = 80
 213         max_chars = max_lines * chars_per_line
 214         total_chars = 0
 215         text_objects = []
 216         for line in lines:
 217             txt = []
 218             for (style, text) in line:
 219                 if total_chars + len(text) > max_chars:
 220                     text = text[:max_chars - total_chars]
 221                 txt.append((style, text))
 222                 total_chars += len(text)
 223                 if total_chars == max_chars:
 224                     break
 225 
 226             # round up to the next line.
 227             total_chars = int(math.ceil(total_chars / chars_per_line) * chars_per_line)
 228 
 229             text_objects.append(urwid.Text(txt))
 230             if total_chars == max_chars:
 231                 text_objects.append(urwid.Text([
 232                     ("highlight", "Stopped displaying data after %d lines. Press " % max_lines),
 233                     ("key", "f"),
 234                     ("highlight", " to load all data.")
 235                 ]))
 236                 break
 237 
 238         return description, text_objects

 240     def conn_text(self, conn):
 241         if conn:
 242             hdrs = []
 243             for k, v in conn.headers.fields:
 244                 # This will always force an ascii representation of headers. For example, if the server sends a
 245                 #
 246                 #     X-Authors: Made with ❤ in Hamburg
 247                 #
 248                 # header, mitmproxy will display the following:
 249                 #
 250                 #     X-Authors: Made with \xe2\x9d\xa4 in Hamburg.
 251                 #
 252                 # The alternative would be to just use the header's UTF-8 representation and maybe
 253                 # do `str.replace("\t", "\\t")` to exempt tabs from urwid's special characters escaping [1].
 254                 # That would in some terminals allow rendering UTF-8 characters, but the mapping
 255                 # wouldn't be bijective, i.e. a user couldn't distinguish "\\t" and "\t".
 256                 # Also, from a security perspective, a mitmproxy user couldn't be fooled by homoglyphs.
 257                 #
 258                 # 1) https://github.com/mitmproxy/mitmproxy/issues/1833
 259                 #    https://github.com/urwid/urwid/blob/6608ee2c9932d264abd1171468d833b7a4082e13/urwid/display_common.py#L35-L36,
 260 
 261                 k = strutils.bytes_to_escaped_str(k) + ":"
 262                 v = strutils.bytes_to_escaped_str(v)
 263                 hdrs.append((k, v))
 264             txt = common.format_keyvals(
 265                 hdrs,
 266                 key_format="header"
 267             )
 268             viewmode = self.master.commands.call("console.flowview.mode")
 269             msg, body = self.content_view(viewmode, conn)
 270 
 271             cols = [
 272                 urwid.Text(
 273                     [
 274                         ("heading", msg),
 275                     ]
 276                 ),
 277                 urwid.Text(
 278                     [
 279                         " ",
 280                         ('heading', "["),
 281                         ('heading_key', "m"),
 282                         ('heading', (":%s]" % viewmode)),
 283                     ],
 284                     align="right"
 285                 )
 286             ]
 287             title = urwid.AttrWrap(urwid.Columns(cols), "heading")
 288 
 289             txt.append(title)
 290             txt.extend(body)
 291         else:
 292             txt = [
 293                 urwid.Text(""),
 294                 urwid.Text(
 295                     [
 296                         ("highlight", "No response. Press "),
 297                         ("key", "e"),
 298                         ("highlight", " and edit any aspect to add one."),
 299                     ]
 300                 )
 301             ]
 302         return searchable.Searchable(txt)
