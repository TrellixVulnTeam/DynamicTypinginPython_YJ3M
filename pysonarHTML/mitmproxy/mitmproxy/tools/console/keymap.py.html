  41 class Binding:
  42     def __init__(self, key, command, contexts, help):
  43         self.key, self.command, self.contexts = key, command, sorted(contexts)
  44         self.help = help
  45 
  46     def keyspec(self):
  47         """
  48             Translate the key spec from a convenient user specification to one
  49             Urwid understands.
  50         """
  51         return self.key.replace("space", " ")
  52 
  53     def sortkey(self):
  54         return self.key + ",".join(self.contexts)

  57 class Keymap:
  58     def __init__(self, master):
  59         self.executor = commandexecutor.CommandExecutor(master)
  60         self.keys = {}
  61         for c in Contexts:
  62             self.keys[c] = {}
  63         self.bindings = []
  64 
  65     def _check_contexts(self, contexts):
  66         if not contexts:
  67             raise ValueError("Must specify at least one context.")
  68         for c in contexts:
  69             if c not in Contexts:
  70                 raise ValueError("Unsupported context: %s" % c)
  71 
  72     def add(
  73         self,
  74         key: str,
  75         command: str,
  76         contexts: typing.Sequence[str],
  77         help=""
  78     ) -> None:
  79         """
  80             Add a key to the key map.
  81         """
  82         self._check_contexts(contexts)
  83 
  84         for b in self.bindings:
  85             if b.key == key and b.command.strip() == command.strip():
  86                 b.contexts = sorted(list(set(b.contexts + contexts)))
  87                 if help:
  88                     b.help = help
  89                 self.bind(b)
  90                 break
  91         else:
  92             self.remove(key, contexts)
  93             b = Binding(key=key, command=command, contexts=contexts, help=help)
  94             self.bindings.append(b)
  95             self.bind(b)
  96         signals.keybindings_change.send(self)
  97 
  98     def remove(self, key: str, contexts: typing.Sequence[str]) -> None:
  99         """
 100             Remove a key from the key map.
 101         """
 102         self._check_contexts(contexts)
 103         for c in contexts:
 104             b = self.get(c, key)
 105             if b:
 106                 self.unbind(b)
 107                 b.contexts = [x for x in b.contexts if x != c]
 108                 if b.contexts:
 109                     self.bindings.append(b)
 110                     self.bind(b)
 111         signals.keybindings_change.send(self)
 112 
 113     def bind(self, binding: Binding) -> None:
 114         for c in binding.contexts:
 115             self.keys[c][binding.keyspec()] = binding
 116 
 117     def unbind(self, binding: Binding) -> None:
 118         """
 119             Unbind also removes the binding from the list.
 120         """
 121         for c in binding.contexts:
 122             del self.keys[c][binding.keyspec()]
 123             self.bindings = [b for b in self.bindings if b != binding]
 124 
 125     def get(self, context: str, key: str) -> typing.Optional[Binding]:
 126         if context in self.keys:
 127             return self.keys[context].get(key, None)
 128         return None
 129 
 130     def list(self, context: str) -> typing.Sequence[Binding]:
 131         b = [x for x in self.bindings if context in x.contexts or context == "all"]
 132         single = [x for x in b if len(x.key.split()) == 1]
 133         multi = [x for x in b if len(x.key.split()) != 1]
 134         single.sort(key=lambda x: x.sortkey())
 135         multi.sort(key=lambda x: x.sortkey())
 136         return single + multi
 137 
 138     def handle(self, context: str, key: str) -> typing.Optional[str]:
 139         """
 140             Returns the key if it has not been handled, or None.
 141         """
 142         b = self.get(context, key) or self.get("global", key)
 143         if b:
 144             return self.executor(b.command)
 145         return key
 146 
 147     def handle_only(self, context: str, key: str) -> typing.Optional[str]:
 148         """
 149             Like handle, but ignores global bindings. Returns the key if it has
 150             not been handled, or None.
 151         """
 152         b = self.get(context, key)
 153         if b:
 154             return self.executor(b.command)
 155         return key

 158 keyAttrs = {
 159     "key": lambda x: isinstance(x, str),
 160     "cmd": lambda x: isinstance(x, str),
 161     "ctx": lambda x: isinstance(x, list) and [isinstance(v, str) for v in x],
 162     "help": lambda x: isinstance(x, str),
 163 }
 164 requiredKeyAttrs = set(["key", "cmd"])

 170     @command.command("console.keymap.load")
 171     def keymap_load_path(self, path: mitmproxy.types.Path) -> None:
 172         try:
 173             self.load_path(ctx.master.keymap, path)  # type: ignore
 174         except (OSError, KeyBindingError) as e:
 175             raise exceptions.CommandError(
 176                 "Could not load key bindings - %s" % e
 177             ) from e
 178 
 179     def running(self):
 180         p = os.path.join(os.path.expanduser(ctx.options.confdir), self.defaultFile)
 181         if os.path.exists(p):
 182             try:
 183                 self.load_path(ctx.master.keymap, p)
 184             except KeyBindingError as e:
 185                 ctx.log.error(e)
 186 
 187     def load_path(self, km, p):
 188         if os.path.exists(p) and os.path.isfile(p):
 189             with open(p, "rt", encoding="utf8") as f:
 190                 try:
 191                     txt = f.read()
 192                 except UnicodeDecodeError as e:
 193                     raise KeyBindingError(
 194                         "Encoding error - expected UTF8: %s: %s" % (p, e)
 195                     )
 196             try:
 197                 vals = self.parse(txt)
 198             except KeyBindingError as e:
 199                 raise KeyBindingError(
 200                     "Error reading %s: %s" % (p, e)
 201                 ) from e
 202             for v in vals:
 203                 user_ctxs = v.get("ctx", ["global"])
 204                 try:
 205                     km._check_contexts(user_ctxs)
 206                     km.remove(v["key"], user_ctxs)
 207                     km.add(
 208                         key = v["key"],
 209                         command = v["cmd"],
 210                         contexts = user_ctxs,
 211                         help = v.get("help", None),
 212                     )
 213                 except ValueError as e:
 214                     raise KeyBindingError(
 215                         "Error reading %s: %s" % (p, e)
 216                     ) from e
 217 
 218     def parse(self, text):
 219         try:
 220             data = ruamel.yaml.safe_load(text)
 221         except ruamel.yaml.error.YAMLError as v:
 222             if hasattr(v, "problem_mark"):
 223                 snip = v.problem_mark.get_snippet()
 224                 raise KeyBindingError(
 225                     "Key binding config error at line %s:\n%s\n%s" %
 226                     (v.problem_mark.line + 1, snip, v.problem)
 227                 )
 228             else:
 229                 raise KeyBindingError("Could not parse key bindings.")
 230         if not data:
 231             return []
 232         if not isinstance(data, list):
 233             raise KeyBindingError("Inalid keybinding config - expected a list of keys")
 234 
 235         for k in data:
 236             unknown = k.keys() - keyAttrs.keys()
 237             if unknown:
 238                 raise KeyBindingError("Unknown key attributes: %s" % unknown)
 239             missing = requiredKeyAttrs - k.keys()
 240             if missing:
 241                 raise KeyBindingError("Missing required key attributes: %s" % unknown)
 242             for attr in k.keys():
 243                 if not keyAttrs[attr](k[attr]):
 244                     raise KeyBindingError("Invalid type for %s" % attr)
 245 
 246         return data
