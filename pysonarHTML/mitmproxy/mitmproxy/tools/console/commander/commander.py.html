  13 class Completer:
  14     @abc.abstractmethod
  15     def cycle(self, forward: bool = True) -> str:
  16         raise NotImplementedError()
  17

  19 class ListCompleter(Completer):
  20     def __init__(
  21             self,
  22             start: str,
  23             options: typing.Sequence[str],
  24     ) -> None:
  25         self.start = start
  26         self.options: typing.List[str] = []
  27         for o in options:
  28             if o.startswith(start):
  29                 self.options.append(o)
  30         self.options.sort()
  31         self.pos = -1
  32 
  33     def cycle(self, forward: bool = True) -> str:
  34         if not self.options:
  35             return self.start
  36         if self.pos == -1:
  37             self.pos = 0 if forward else len(self.options) - 1
  38         else:
  39             delta = 1 if forward else -1
  40             self.pos = (self.pos + delta) % len(self.options)
  41         return self.options[self.pos]

  49 class CommandBuffer:
  50     def __init__(self, master: mitmproxy.master.Master, start: str = "") -> None:
  51         self.master = master
  52         self.text = start
  53         # Cursor is always within the range [0:len(buffer)].
  54         self._cursor = len(self.text)
  55         self.completion: typing.Optional[CompletionState] = None
  56 
  57     @property
  58     def cursor(self) -> int:
  59         return self._cursor
  60 
  61     @cursor.setter
  62     def cursor(self, x) -> None:
  63         if x < 0:
  64             self._cursor = 0
  65         elif x > len(self.text):
  66             self._cursor = len(self.text)
  67         else:
  68             self._cursor = x
  69 
  70     def set_text(self, text: str) -> None:
  71         self.text = text
  72         self._cursor = len(self.text)
  73         self.render()
  74 
  75     def render(self):
  76         parts, remaining = self.master.commands.parse_partial(self.text)
  77         ret = []
  78         if not parts:
  79             # Means we just received the leader, so we need to give a blank
  80             # text to the widget to render or it crashes
  81             ret.append(("text", ""))
  82         else:
  83             for p in parts:
  84                 if p.valid:
  85                     if p.type == mitmproxy.types.Cmd:
  86                         ret.append(("commander_command", p.value))
  87                     else:
  88                         ret.append(("text", p.value))
  89                 elif p.value:
  90                     ret.append(("commander_invalid", p.value))
  91 
  92             if remaining:
  93                 if parts[-1].type != mitmproxy.types.Space:
  94                     ret.append(("text", " "))
  95                 for param in remaining:
  96                     ret.append(("commander_hint", f"{param} "))
  97 
  98         return ret
  99 
 100     def left(self) -> None:
 101         self.cursor = self.cursor - 1
 102 
 103     def right(self) -> None:
 104         self.cursor = self.cursor + 1
 105 
 106     def cycle_completion(self, forward: bool = True) -> None:
 107         if not self.completion:
 108             parts, remaining = self.master.commands.parse_partial(self.text[:self.cursor])
 109             if parts and parts[-1].type != mitmproxy.types.Space:
 110                 type_to_complete = parts[-1].type
 111                 cycle_prefix = parts[-1].value
 112                 parsed = parts[:-1]
 113             elif remaining:
 114                 type_to_complete = remaining[0].type
 115                 cycle_prefix = ""
 116                 parsed = parts
 117             else:
 118                 return
 119             ct = mitmproxy.types.CommandTypes.get(type_to_complete, None)
 120             if ct:
 121                 self.completion = CompletionState(
 122                     completer=ListCompleter(
 123                         cycle_prefix,
 124                         ct.completion(self.master.commands, type_to_complete, cycle_prefix)
 125                     ),
 126                     parsed=parsed,
 127                 )
 128         if self.completion:
 129             nxt = self.completion.completer.cycle(forward)
 130             buf = "".join([i.value for i in self.completion.parsed]) + nxt
 131             self.text = buf
 132             self.cursor = len(self.text)
 133 
 134     def backspace(self) -> None:
 135         if self.cursor == 0:
 136             return
 137         self.text = self.text[:self.cursor - 1] + self.text[self.cursor:]
 138         self.cursor = self.cursor - 1
 139         self.completion = None
 140 
 141     def delete(self) -> None:
 142         if self.cursor == len(self.text):
 143             return
 144         self.text = self.text[:self.cursor] + self.text[self.cursor + 1:]
 145         self.completion = None
 146 
 147     def insert(self, k: str) -> None:
 148         """
 149             Inserts text at the cursor.
 150         """
 151 
 152         # We don't want to insert a space before the command
 153         if k == ' ' and self.text[0:self.cursor].strip() == '':
 154             return
 155 
 156         self.text = self.text[:self.cursor] + k + self.text[self.cursor:]
 157         self.cursor += len(k)
 158         self.completion = None

 161 class CommandEdit(urwid.WidgetWrap):
 162     leader = ": "
 163 
 164     def __init__(self, master: mitmproxy.master.Master, text: str) -> None:
 165         super().__init__(urwid.Text(self.leader))
 166         self.master = master
 167         self.active_filter = False
 168         self.filter_str = ''
 169         self.cbuf = CommandBuffer(master, text)
 170         self.update()
 171 
 172     def keypress(self, size, key) -> None:
 173         if key == "delete":
 174             self.cbuf.delete()
 175         elif key == "ctrl a" or key == 'home':
 176             self.cbuf.cursor = 0
 177         elif key == "ctrl e" or key == 'end':
 178             self.cbuf.cursor = len(self.cbuf.text)
 179         elif key == "meta b":
 180             self.cbuf.cursor = self.cbuf.text.rfind(' ', 0, self.cbuf.cursor)
 181         elif key == "meta f":
 182             pos = self.cbuf.text.find(' ', self.cbuf.cursor + 1)
 183             if pos == -1:
 184                 pos = len(self.cbuf.text)
 185             self.cbuf.cursor = pos
 186         elif key == "ctrl w":
 187             prev_cursor = self.cbuf.cursor
 188             pos = self.cbuf.text.rfind(' ', 0, self.cbuf.cursor - 1)
 189             if pos == -1:
 190                 new_text = self.cbuf.text[self.cbuf.cursor:]
 191                 cursor_pos = 0
 192             else:
 193                 txt_after = self.cbuf.text[self.cbuf.cursor:]
 194                 txt_before = self.cbuf.text[0:pos]
 195                 new_text = f"{txt_before} {txt_after}"
 196                 cursor_pos = prev_cursor - (prev_cursor - pos) + 1
 197             self.cbuf.set_text(new_text)
 198             self.cbuf.cursor = cursor_pos
 199         elif key == "backspace":
 200             self.cbuf.backspace()
 201             if self.cbuf.text == '':
 202                 self.active_filter = False
 203                 self.master.commands.call("commands.history.filter", "")
 204                 self.filter_str = ''
 205         elif key == "left" or key == "ctrl b":
 206             self.cbuf.left()
 207         elif key == "right" or key == "ctrl f":
 208             self.cbuf.right()
 209         elif key == "up" or key == "ctrl p":
 210             if self.active_filter is False:
 211                 self.active_filter = True
 212                 self.filter_str = self.cbuf.text
 213                 self.master.commands.call("commands.history.filter", self.cbuf.text)
 214             cmd = self.master.commands.execute("commands.history.prev")
 215             self.cbuf = CommandBuffer(self.master, cmd)
 216         elif key == "down" or key == "ctrl n":
 217             prev_cmd = self.cbuf.text
 218             cmd = self.master.commands.execute("commands.history.next")
 219 
 220             if cmd == '':
 221                 if prev_cmd == self.filter_str:
 222                     self.cbuf = CommandBuffer(self.master, prev_cmd)
 223                 else:
 224                     self.active_filter = False
 225                     self.master.commands.call("commands.history.filter", "")
 226                     self.filter_str = ''
 227                     self.cbuf = CommandBuffer(self.master, '')
 228             else:
 229                 self.cbuf = CommandBuffer(self.master, cmd)
 230         elif key == "shift tab":
 231             self.cbuf.cycle_completion(False)
 232         elif key == "tab":
 233             self.cbuf.cycle_completion()
 234         elif len(key) == 1:
 235             self.cbuf.insert(key)
 236         self.update()
 237 
 238     def update(self) -> None:
 239         self._w.set_text([self.leader, self.cbuf.render()])
 240 
 241     def render(self, size, focus=False) -> urwid.Canvas:
 242         (maxcol,) = size
 243         canv = self._w.render((maxcol,))
 244         canv = urwid.CompositeCanvas(canv)
 245         canv.cursor = self.get_cursor_coords((maxcol,))
 246         return canv
 247 
 248     def get_cursor_coords(self, size) -> typing.Tuple[int, int]:
 249         p = self.cbuf.cursor + len(self.leader)
 250         trans = self._w.get_line_translation(size[0])
 251         x, y = calc_coords(self._w.get_text()[0], trans, p)
 252         return x, y
 253 
 254     def get_edit_text(self) -> str:
 255         return self.cbuf.text
