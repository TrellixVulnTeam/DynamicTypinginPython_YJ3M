  35     def set_focus(self, index):

  37             super().set_focus(index)

  39     def keypress(self, size, key):
  40         if key == "m_end":
  41             self.set_focus(len(self.walker) - 1)
  42         elif key == "m_start":
  43             self.set_focus(0)
  44         return super().keypress(size, key)

  46     def add_event(self, event_store, entry: log.LogEntry):
  47         if log.log_tier(self.master.options.console_eventlog_verbosity) < log.log_tier(entry.level):
  48             return
  49         txt = "%s: %s" % (entry.level, str(entry.msg))
  50         if entry.level in ("error", "warn", "alert"):
  51             e = urwid.Text((entry.level, txt))
  52         else:
  53             e = urwid.Text(txt)
  54         self.walker.append(e)
  55         if self.master.options.console_focus_follow:
  56             self.walker.set_focus(len(self.walker) - 1)
  57 
  58     def refresh_events(self, *_):
  59         self.walker.clear()
  60         for event in self.master.events.data:
  61             self.add_event(None, event)
