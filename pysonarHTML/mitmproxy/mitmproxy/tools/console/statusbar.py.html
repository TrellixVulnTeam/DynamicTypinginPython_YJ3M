  17     def __call__(self, pth):
  18         if not pth:
  19             return
  20         pth = os.path.expanduser(pth)
  21         try:
  22             return self.callback(pth, *self.args)
  23         except IOError as v:
  24             signals.status_message.send(message=v.strerror)

  50     def sig_message(self, sender, message, expire=1):
  51         if self.prompting:
  52             return
  53         cols, _ = self.master.ui.get_cols_rows()
  54         w = urwid.Text(self.shorten_message(message, cols))
  55         self._w = w
  56         if expire:
  57             def cb(*args):
  58                 if w == self._w:
  59                     self.clear()
  60 
  61             signals.call_in.send(seconds=expire, callback=cb)

  63     def prep_prompt(self, p):
  64         return p.strip() + ": "
  65 
  66     def shorten_message(self, msg, max_width):
  67         """
  68         Shorten message so that it fits into a single line in the statusbar.
  69         """
  70         if isinstance(msg, tuple):
  71             disp_attr, msg_text = msg
  72         elif isinstance(msg, str):
  73             disp_attr, msg_text = None, msg
  74         else:
  75             return msg
  76         msg_end = "\u2026"  # unicode ellipsis for the end of shortened message
  77         prompt = "(more in eventlog)"
  78 
  79         msg_lines = msg_text.split("\n")
  80         first_line = msg_lines[0]
  81         if len(msg_lines) > 1:
  82             # First line of messages with a few lines must end with prompt.
  83             line_length = len(first_line) + len(prompt)
  84         else:
  85             line_length = len(first_line)
  86 
  87         if line_length > max_width:
  88             shortening_index = max(0, max_width - len(prompt) - len(msg_end))
  89             first_line = first_line[:shortening_index] + msg_end
  90         else:
  91             if len(msg_lines) == 1:
  92                 prompt = ""
  93 
  94         return [(disp_attr, first_line), ("warn", prompt)]

 101     def sig_prompt_command(self, sender, partial: str = "", cursor: Optional[int] = None):
 102         signals.focus.send(self, section="footer")

 111     def execute_command(self, txt):
 112         if txt.strip():
 113             self.master.commands.call("commands.history.add", txt)
 114         execute = commandexecutor.CommandExecutor(self.master)

 117     def sig_prompt_onekey(self, sender, prompt, keys, callback, args=()):
 118         """
 119             Keys are a set of (word, key) tuples. The appropriate key in the
 120             word is highlighted.
 121         """
 122         signals.focus.send(self, section="footer")
 123         prompt = [prompt, " ("]
 124         mkup = []
 125         for i, e in enumerate(keys):
 126             mkup.extend(common.highlight_key(e[0], e[1]))
 127             if i < len(keys) - 1:
 128                 mkup.append(",")
 129         prompt.extend(mkup)
 130         prompt.append(")? ")
 131         self.onekey = set(i[1] for i in keys)
 132         self._w = urwid.Edit(prompt, "")
 133         self.prompting = PromptStub(callback, args)

 138     def keypress(self, size, k):
 139         if self.prompting:
 140             if k == "esc":
 141                 self.prompt_done()
 142             elif self.onekey:
 143                 if k == "enter":
 144                     self.prompt_done()
 145                 elif k in self.onekey:
 146                     self.prompt_execute(k)
 147             elif k == "enter":
 148                 text = self._w.get_edit_text()
 149                 self.prompt_execute(text)
 150             else:
 151                 if common.is_keypress(k):
 152                     self._w.keypress(size, k)
 153                 else:
 154                     return k

 156     def clear(self):
 157         self._w = urwid.Text("")

 160     def prompt_done(self):
 161         self.prompting = None
 162         self.onekey = False
 163         signals.status_message.send(message="")
 164         signals.focus.send(self, section="body")

 166     def prompt_execute(self, txt):
 167         p = self.prompting
 168         self.prompt_done()
 169         msg = p(txt)
 170         if msg:
 171             signals.status_message.send(message=msg, expire=1)

 291     def redraw(self):
 292         fc = self.master.commands.execute("view.properties.length")
 293         if self.master.view.focus.flow is None:
 294             offset = 0
 295         else:
 296             offset = self.master.view.focus.index + 1
 297 
 298         if self.master.options.view_order_reversed:
 299             arrow = common.SYMBOL_UP
 300         else:
 301             arrow = common.SYMBOL_DOWN
 302 
 303         marked = ""
 304         if self.master.commands.execute("view.properties.marked"):
 305             marked = "M"
 306 
 307         t = [
 308             ('heading', ("%s %s [%s/%s]" % (arrow, marked, offset, fc)).ljust(11)),
 309         ]
 310 
 311         if self.master.options.server:
 312             host = self.master.options.listen_host
 313             if host == "0.0.0.0" or host == "":
 314                 host = "*"
 315             boundaddr = "[%s:%s]" % (host, self.master.options.listen_port)
 316         else:
 317             boundaddr = ""
 318         t.extend(self.get_status())
 319         status = urwid.AttrWrap(urwid.Columns([
 320             urwid.Text(t),
 321             urwid.Text(boundaddr, align="right"),
 322         ]), "heading")
 323         self.ib._w = status
