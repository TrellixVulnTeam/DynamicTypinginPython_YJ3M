 120 class CookieAttributeEditor(base.FocusEditor):
 121     title = "Editing Set-Cookie attributes"
 122     columns = [
 123         col_text.Column("Name"),
 124         col_text.Column("Value"),
 125     ]
 126     grideditor: base.BaseGridEditor
 127 
 128     def data_in(self, data):
 129         return [(k, v or "") for k, v in data]
 130 
 131     def data_out(self, data):
 132         ret = []
 133         for i in data:
 134             if not i[1]:
 135                 ret.append([i[0], None])
 136             else:
 137                 ret.append(i)
 138         return ret
 139 
 140     def layout_pushed(self, prev):
 141         if self.grideditor.master.view.focus.flow:
 142             self._w = base.BaseGridEditor(
 143                 self.grideditor.master,
 144                 self.title,
 145                 self.columns,
 146                 self.grideditor.walker.get_current_value(),
 147                 self.grideditor.set_subeditor_value,
 148                 self.grideditor.walker.focus,
 149                 self.grideditor.walker.focus_col
 150             )
 151         else:
 152             self._w = urwid.Pile([])

 155 class SetCookieEditor(base.FocusEditor):
 156     title = "Edit SetCookie Header"
 157     columns = [
 158         col_text.Column("Name"),
 159         col_text.Column("Value"),
 160         col_subgrid.Column("Attributes", CookieAttributeEditor),
 161     ]
 162 
 163     def data_in(self, data):
 164         flattened = []
 165         for key, (value, attrs) in data:
 166             flattened.append([key, value, attrs.items(multi=True)])
 167         return flattened
 168 
 169     def data_out(self, data):
 170         vals = []
 171         for key, value, attrs in data:
 172             vals.append(
 173                 [
 174                     key,
 175                     (value, attrs)
 176                 ]
 177             )
 178         return vals
 179 
 180     def get_data(self, flow):
 181         return self.data_in(flow.response.cookies.items(multi=True))
 182 
 183     def set_data(self, vals, flow):
 184         flow.response.cookies = self.data_out(vals)

 187 class OptionsEditor(base.GridEditor, layoutwidget.LayoutWidget):
 188     title = ""
 189     columns = [
 190         col_text.Column("")
 191     ]
 192 
 193     def __init__(self, master, name, vals):
 194         self.name = name
 195         super().__init__(master, [[i] for i in vals], self.callback)
 196 
 197     def callback(self, vals):
 198         try:
 199             setattr(self.master.options, self.name, [i[0] for i in vals])
 200         except exceptions.OptionsError as v:
 201             signals.status_message.send(message=str(v))
 202 
 203     def is_error(self, col, val):
 204         pass

 207 class DataViewer(base.GridEditor, layoutwidget.LayoutWidget):
 208     title = ""
 209 
 210     def __init__(
 211             self,
 212             master,
 213             vals: typing.Union[
 214                 typing.List[typing.List[typing.Any]],
 215                 typing.List[typing.Any],
 216                 str,
 217             ]) -> None:
 218         if vals:
 219             # Whatever vals is, make it a list of rows containing lists of column values.
 220             if isinstance(vals, str):
 221                 vals = [vals]
 222             if not isinstance(vals[0], list):
 223                 vals = [[i] for i in vals]
 224 
 225             self.columns = [col_viewany.Column("")] * len(vals[0])
 226         super().__init__(master, vals, self.callback)
 227 
 228     def callback(self, vals):
 229         pass
 230 
 231     def is_error(self, col, val):
 232         pass
