  14 def read_file(filename: str, escaped: bool) -> typing.AnyStr:
  15     filename = os.path.expanduser(filename)
  16     try:
  17         with open(filename, "r" if escaped else "rb") as f:
  18             d = f.read()
  19     except IOError as v:
  20         raise exceptions.CommandError(v)
  21     if escaped:
  22         try:
  23             d = strutils.escaped_str_to_bytes(d)
  24         except ValueError:
  25             raise exceptions.CommandError("Invalid Python-style string encoding.")
  26     return d

  63 class GridRow(urwid.WidgetWrap):
  64 
  65     def __init__(
  66             self,
  67             focused: typing.Optional[int],
  68             editing: bool,
  69             editor: "GridEditor",
  70             values: typing.Tuple[typing.Iterable[bytes], typing.Container[int]]
  71     ) -> None:
  72         self.focused = focused
  73         self.editor = editor
  74         self.edit_col: typing.Optional[Cell] = None
  75 
  76         errors = values[1]
  77         self.fields: typing.Sequence[typing.Any] = []
  78         for i, v in enumerate(values[0]):
  79             if focused == i and editing:
  80                 self.edit_col = self.editor.columns[i].Edit(v)
  81                 self.fields.append(self.edit_col)
  82             else:
  83                 w = self.editor.columns[i].Display(v)
  84                 if focused == i:
  85                     if i in errors:
  86                         w = urwid.AttrWrap(w, "focusfield_error")
  87                     else:
  88                         w = urwid.AttrWrap(w, "focusfield")
  89                 elif i in errors:
  90                     w = urwid.AttrWrap(w, "field_error")
  91                 self.fields.append(w)
  92 
  93         fspecs = self.fields[:]
  94         if len(self.fields) > 1:
  95             fspecs[0] = ("fixed", self.editor.first_width + 2, fspecs[0])
  96         w = urwid.Columns(
  97             fspecs,
  98             dividechars=2
  99         )
 100         if focused is not None:
 101             w.set_focus_column(focused)
 102         super().__init__(w)
 103 
 104     def keypress(self, s, k):
 105         if self.edit_col:
 106             w = self._w.column_widths(s)[self.focused]
 107             k = self.edit_col.keypress((w,), k)
 108         return k

 114 class GridWalker(urwid.ListWalker):
 115     """
 116         Stores rows as a list of (rows, errors) tuples, where rows is a list
 117         and errors is a set with an entry of each offset in rows that is an
 118         error.
 119     """
 120 
 121     def __init__(
 122             self,
 123             lst: typing.Iterable[list],
 124             editor: "GridEditor"
 125     ) -> None:
 126         self.lst: typing.Sequence[typing.Tuple[typing.Any, typing.Set]] = [(i, set()) for i in lst]
 127         self.editor = editor
 128         self.focus = 0
 129         self.focus_col = 0
 130         self.edit_row: typing.Optional[GridRow] = None
 131 
 132     def _modified(self):
 133         self.editor.show_empty_msg()
 134         return super()._modified()
 135 
 136     def add_value(self, lst):
 137         self.lst.append(
 138             (lst[:], set())
 139         )
 140         self._modified()
 141 
 142     def get_current_value(self):
 143         if self.lst:
 144             return self.lst[self.focus][0][self.focus_col]
 145 
 146     def set_current_value(self, val):
 147         errors = self.lst[self.focus][1]
 148         emsg = self.editor.is_error(self.focus_col, val)
 149         if emsg:
 150             signals.status_message.send(message=emsg, expire=5)
 151             errors.add(self.focus_col)
 152         else:
 153             errors.discard(self.focus_col)
 154         self.set_value(val, self.focus, self.focus_col, errors)
 155 
 156     def set_value(self, val, focus, focus_col, errors=None):
 157         if not errors:
 158             errors = set([])
 159         row = list(self.lst[focus][0])
 160         row[focus_col] = val
 161         self.lst[focus] = [tuple(row), errors]
 162         self._modified()
 163 
 164     def delete_focus(self):
 165         if self.lst:
 166             del self.lst[self.focus]
 167             self.focus = min(len(self.lst) - 1, self.focus)
 168             self._modified()
 169 
 170     def _insert(self, pos):
 171         self.focus = pos
 172         self.lst.insert(
 173             self.focus,
 174             ([c.blank() for c in self.editor.columns], set([]))
 175         )
 176         self.focus_col = 0
 177         self.start_edit()
 178 
 179     def insert(self):
 180         return self._insert(self.focus)
 181 
 182     def add(self):
 183         return self._insert(min(self.focus + 1, len(self.lst)))
 184 
 185     def start_edit(self):
 186         col = self.editor.columns[self.focus_col]
 187         if self.lst and not col.subeditor:
 188             self.edit_row = GridRow(
 189                 self.focus_col, True, self.editor, self.lst[self.focus]
 190             )
 191             self._modified()
 192 
 193     def stop_edit(self):
 194         if self.edit_row:
 195             try:
 196                 val = self.edit_row.edit_col.get_data()
 197             except ValueError:
 198                 return
 199             self.edit_row = None
 200             self.set_current_value(val)
 201 
 202     def left(self):
 203         self.focus_col = max(self.focus_col - 1, 0)
 204         self._modified()
 205 
 206     def right(self):
 207         self.focus_col = min(self.focus_col + 1, len(self.editor.columns) - 1)
 208         self._modified()
 209 
 210     def tab_next(self):
 211         self.stop_edit()
 212         if self.focus_col < len(self.editor.columns) - 1:
 213             self.focus_col += 1
 214         elif self.focus != len(self.lst) - 1:
 215             self.focus_col = 0
 216             self.focus += 1
 217         self._modified()
 218 
 219     def get_focus(self):
 220         if self.edit_row:
 221             return self.edit_row, self.focus
 222         elif self.lst:
 223             return GridRow(
 224                 self.focus_col,
 225                 False,
 226                 self.editor,
 227                 self.lst[self.focus]
 228             ), self.focus
 229         else:
 230             return None, None
 231 
 232     def set_focus(self, focus):
 233         self.stop_edit()
 234         self.focus = focus
 235         self._modified()
 236 
 237     def get_next(self, pos):
 238         if pos + 1 >= len(self.lst):
 239             return None, None
 240         return GridRow(None, False, self.editor, self.lst[pos + 1]), pos + 1
 241 
 242     def get_prev(self, pos):
 243         if pos - 1 < 0:
 244             return None, None
 245         return GridRow(None, False, self.editor, self.lst[pos - 1]), pos - 1

 256 class BaseGridEditor(urwid.WidgetWrap):
 257     title: str = ""
 258     keyctx = "grideditor"
 259 
 260     def __init__(
 261             self,
 262             master: "mitmproxy.tools.console.master.ConsoleMaster",
 263             title,
 264             columns,
 265             value: typing.Any,
 266             callback: typing.Callable[..., None],
 267             *cb_args,
 268             **cb_kwargs
 269     ) -> None:
 270         value = self.data_in(copy.deepcopy(value))
 271         self.master = master
 272         self.title = title
 273         self.columns = columns
 274         self.value = value
 275         self.callback = callback
 276         self.cb_args = cb_args
 277         self.cb_kwargs = cb_kwargs
 278 
 279         first_width = 20
 280         if value:
 281             for r in value:
 282                 assert len(r) == len(self.columns)
 283                 first_width = max(len(r), first_width)
 284         self.first_width = min(first_width, FIRST_WIDTH_MAX)
 285 
 286         h = None
 287         if any(col.heading for col in self.columns):
 288             headings = []
 289             for i, col in enumerate(self.columns):
 290                 c = urwid.Text(col.heading)
 291                 if i == 0 and len(self.columns) > 1:
 292                     headings.append(("fixed", first_width + 2, c))
 293                 else:
 294                     headings.append(c)
 295             h = urwid.Columns(
 296                 headings,
 297                 dividechars=2
 298             )
 299             h = urwid.AttrWrap(h, "heading")
 300 
 301         self.walker = GridWalker(self.value, self)
 302         self.lb = GridListBox(self.walker)
 303         w = urwid.Frame(self.lb, header=h)
 304 
 305         super().__init__(w)
 306         self.show_empty_msg()
 307 
 308     def layout_popping(self):
 309         res = []
 310         for i in self.walker.lst:
 311             if not i[1] and any([x for x in i[0]]):
 312                 res.append(i[0])
 313         self.callback(self.data_out(res), *self.cb_args, **self.cb_kwargs)
 314 
 315     def show_empty_msg(self):
 316         if self.walker.lst:
 317             self._w.set_footer(None)
 318         else:
 319             self._w.set_footer(
 320                 urwid.Text(
 321                     [
 322                         ("highlight", "No values - you should add some. Press "),
 323                         ("key", "?"),
 324                         ("highlight", " for help."),
 325                     ]
 326                 )
 327             )
 328 
 329     def set_subeditor_value(self, val, focus, focus_col):
 330         self.walker.set_value(val, focus, focus_col)
 331 
 332     def keypress(self, size, key):
 333         if self.walker.edit_row:
 334             if key == "esc":
 335                 self.walker.stop_edit()
 336             elif key == "tab":
 337                 pf, pfc = self.walker.focus, self.walker.focus_col
 338                 self.walker.tab_next()
 339                 if self.walker.focus == pf and self.walker.focus_col != pfc:
 340                     self.walker.start_edit()
 341             else:
 342                 self._w.keypress(size, key)
 343             return None
 344 
 345         column = self.columns[self.walker.focus_col]
 346         if key == "m_start":
 347             self.walker.set_focus(0)
 348         elif key == "m_next":
 349             self.walker.tab_next()
 350         elif key == "m_end":
 351             self.walker.set_focus(len(self.walker.lst) - 1)
 352         elif key == "left":
 353             self.walker.left()
 354         elif key == "right":
 355             self.walker.right()
 356         elif column.keypress(key, self) and not self.handle_key(key):
 357             return self._w.keypress(size, key)
 358 
 359     def data_out(self, data: typing.Sequence[list]) -> typing.Any:
 360         """
 361             Called on raw list data, before data is returned through the
 362             callback.
 363         """
 364         return data
 365 
 366     def data_in(self, data: typing.Any) -> typing.Iterable[list]:
 367         """
 368             Called to prepare provided data.
 369         """
 370         return data
 371 
 372     def is_error(self, col: int, val: typing.Any) -> typing.Optional[str]:
 373         """
 374             Return None, or a string error message.
 375         """
 376         return None
 377 
 378     def handle_key(self, key):
 379         return False
 380 
 381     def cmd_add(self):
 382         self.walker.add()
 383 
 384     def cmd_insert(self):
 385         self.walker.insert()
 386 
 387     def cmd_delete(self):
 388         self.walker.delete_focus()
 389 
 390     def cmd_read_file(self, path):
 391         self.walker.set_current_value(read_file(path, False))
 392 
 393     def cmd_read_file_escaped(self, path):
 394         self.walker.set_current_value(read_file(path, True))
 395 
 396     def cmd_spawn_editor(self):
 397         o = self.walker.get_current_value()
 398         if o is not None:
 399             n = self.master.spawn_editor(o)
 400             n = strutils.clean_hanging_newline(n)
 401             self.walker.set_current_value(n)

 428 class FocusEditor(urwid.WidgetWrap, layoutwidget.LayoutWidget):
 429     """
 430         A specialised GridEditor that edits the current focused flow.
 431     """
 432     keyctx = "grideditor"
 433 
 434     def __init__(self, master):
 435         self.master = master
 436 
 437     def call(self, v, name, *args, **kwargs):
 438         f = getattr(v, name, None)
 439         if f:
 440             f(*args, **kwargs)
 441 
 442     def get_data(self, flow):
 443         """
 444             Retrieve the data to edit from the current flow.
 445         """
 446         raise NotImplementedError
 447 
 448     def set_data(self, vals, flow):
 449         """
 450             Set the current data on the flow.
 451         """
 452         raise NotImplementedError
 453 
 454     def set_data_update(self, vals, flow):
 455         self.set_data(vals, flow)
 456         signals.flow_change.send(self, flow = flow)
 457 
 458     def key_responder(self):
 459         return self._w
 460 
 461     def layout_popping(self):
 462         self.call(self._w, "layout_popping")
 463 
 464     def layout_pushed(self, prev):
 465         if self.master.view.focus.flow:
 466             self._w = BaseGridEditor(
 467                 self.master,
 468                 self.title,
 469                 self.columns,
 470                 self.get_data(self.master.view.focus.flow),
 471                 self.set_data_update,
 472                 self.master.view.focus.flow,
 473             )
 474         else:
 475             self._w = urwid.Pile([])
