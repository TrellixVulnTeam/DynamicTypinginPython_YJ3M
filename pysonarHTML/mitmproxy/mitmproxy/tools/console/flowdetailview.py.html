  11 def maybe_timestamp(base, attr):

  20 
  21     sc = flow.server_conn
  22     cc = flow.client_conn
  23     req: typing.Optional[http.HTTPRequest]

  27         resp = flow.response

  29         req = None
  30         resp = None
  31     metadata = flow.metadata
  32 
  33     if metadata is not None and len(metadata) > 0:
  34         parts = [(str(k), repr(v)) for k, v in metadata.items()]

  36         text.extend(common.format_keyvals(parts, indent=4))
  37 
  38     if sc is not None and sc.ip_address:
  39         text.append(urwid.Text([("head", "Server Connection:")]))
  40         parts = [
  41             ("Address", human.format_address(sc.address)),
  42         ]
  43         if sc.ip_address:
  44             parts.append(("Resolved Address", human.format_address(sc.ip_address)))
  45         if resp:
  46             parts.append(("HTTP Version", resp.http_version))
  47         if sc.alpn_proto_negotiated:
  48             parts.append(("ALPN", sc.alpn_proto_negotiated))
  49 
  50         text.extend(
  51             common.format_keyvals(parts, indent=4)
  52         )
  53 
  54         c = sc.cert
  55         if c:
  56             text.append(urwid.Text([("head", "Server Certificate:")]))
  57             parts = [
  58                 ("Type", "%s, %s bits" % c.keyinfo),
  59                 ("SHA1 digest", c.digest("sha1")),
  60                 ("Valid to", str(c.notafter)),
  61                 ("Valid from", str(c.notbefore)),
  62                 ("Serial", str(c.serial)),
  63                 (
  64                     "Subject",
  65                     urwid.BoxAdapter(
  66                         urwid.ListBox(
  67                             common.format_keyvals(
  68                                 c.subject,
  69                                 key_format="highlight"
  70                             )
  71                         ),
  72                         len(c.subject)
  73                     )
  74                 ),
  75                 (
  76                     "Issuer",
  77                     urwid.BoxAdapter(
  78                         urwid.ListBox(
  79                             common.format_keyvals(
  80                                 c.issuer,
  81                                 key_format="highlight"
  82                             )
  83                         ),
  84                         len(c.issuer)
  85                     )
  86                 )
  87             ]
  88 
  89             if c.altnames:
  90                 parts.append(
  91                     (
  92                         "Alt names",
  93                         ", ".join(strutils.bytes_to_escaped_str(x) for x in c.altnames)
  94                     )
  95                 )
  96             text.extend(
  97                 common.format_keyvals(parts, indent=4)
  98             )
  99 
 100     if cc is not None:
 101         text.append(urwid.Text([("head", "Client Connection:")]))
 102 
 103         parts = [
 104             ("Address", "{}:{}".format(cc.address[0], cc.address[1])),
 105         ]
 106         if req:
 107             parts.append(("HTTP Version", req.http_version))
 108         if cc.tls_version:
 109             parts.append(("TLS Version", cc.tls_version))
 110         if cc.sni:
 111             parts.append(("Server Name Indication", cc.sni))
 112         if cc.cipher_name:
 113             parts.append(("Cipher Name", cc.cipher_name))
 114         if cc.alpn_proto_negotiated:
 115             parts.append(("ALPN", cc.alpn_proto_negotiated))
 116 
 117         text.extend(
 118             common.format_keyvals(parts, indent=4)
 119         )
 120 
 121     parts = []
 122 
 123     if cc is not None and cc.timestamp_start:
 124         parts.append(
 125             (
 126                 "Client conn. established",
 127                 maybe_timestamp(cc, "timestamp_start")
 128             )
 129         )
 130         if cc.tls_established:
 131             parts.append(
 132                 (
 133                     "Client conn. TLS handshake",
 134                     maybe_timestamp(cc, "timestamp_tls_setup")
 135                 )
 136             )
 137         parts.append(
 138             (
 139                 "Client conn. closed",
 140                 maybe_timestamp(cc, "timestamp_end")
 141             )
 142         )
 143 
 144     if sc is not None and sc.timestamp_start:
 145         parts.append(
 146             (
 147                 "Server conn. initiated",
 148                 maybe_timestamp(sc, "timestamp_start")
 149             )
 150         )
 151         parts.append(
 152             (
 153                 "Server conn. TCP handshake",
 154                 maybe_timestamp(sc, "timestamp_tcp_setup")
 155             )
 156         )
 157         if sc.tls_established:
 158             parts.append(
 159                 (
 160                     "Server conn. TLS handshake",
 161                     maybe_timestamp(sc, "timestamp_tls_setup")
 162                 )
 163             )
 164         parts.append(
 165             (
 166                 "Server conn. closed",
 167                 maybe_timestamp(sc, "timestamp_end")
 168             )
 169         )
 170 
 171     if req is not None and req.timestamp_start:
 172         parts.append(
 173             (
 174                 "First request byte",
 175                 maybe_timestamp(req, "timestamp_start")
 176             )
 177         )
 178         parts.append(
 179             (
 180                 "Request complete",
 181                 maybe_timestamp(req, "timestamp_end")
 182             )
 183         )
 184 
 185     if resp is not None and resp.timestamp_start:
 186         parts.append(
 187             (
 188                 "First response byte",
 189                 maybe_timestamp(resp, "timestamp_start")
 190             )
 191         )
 192         parts.append(
 193             (
 194                 "Response complete",
 195                 maybe_timestamp(resp, "timestamp_end")
 196             )
 197         )
 198 
 199     if parts:
 200         # sort operations by timestamp
 201         parts = sorted(parts, key=lambda p: p[1])
 202 
 203         text.append(urwid.Text([("head", "Timing:")]))
 204         text.extend(common.format_keyvals(parts, indent=4))
 205 
 206     return searchable.Searchable(text)
