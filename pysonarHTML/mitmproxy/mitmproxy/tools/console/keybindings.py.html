  39 class KeyListWalker(urwid.ListWalker):
  40     def __init__(self, master):
  41         self.master = master
  42 
  43         self.index = 0
  44         self.focusobj = None
  45         self.bindings = list(master.keymap.list("all"))
  46         self.set_focus(0)
  47         signals.keybindings_change.connect(self.sig_modified)
  48 
  49     def sig_modified(self, sender):
  50         self.bindings = list(self.master.keymap.list("all"))
  51         self.set_focus(min(self.index, len(self.bindings) - 1))
  52         self._modified()
  53 
  54     def get_edit_text(self):
  55         return self.focus_obj.get_edit_text()
  56 
  57     def _get(self, pos):
  58         binding = self.bindings[pos]
  59         return KeyItem(self, binding, pos == self.index)
  60 
  61     def get_focus(self):
  62         return self.focus_obj, self.index
  63 
  64     def set_focus(self, index):
  65         binding = self.bindings[index]
  66         self.index = index
  67         self.focus_obj = self._get(self.index)
  68         keybinding_focus_change.send(binding.help or "")
  69 
  70     def get_next(self, pos):
  71         if pos >= len(self.bindings) - 1:
  72             return None, None
  73         pos = pos + 1
  74         return self._get(pos), pos
  75 
  76     def get_prev(self, pos):
  77         pos = pos - 1
  78         if pos < 0:
  79             return None, None
  80         return self._get(pos), pos

  83 class KeyList(urwid.ListBox):
  84     def __init__(self, master):
  85         self.master = master
  86         self.walker = KeyListWalker(master)
  87         super().__init__(self.walker)
  88 
  89     def keypress(self, size, key):
  90         if key == "m_select":
  91             foc, idx = self.get_focus()
  92             # Act here
  93         elif key == "m_start":
  94             self.set_focus(0)
  95             self.walker._modified()
  96         elif key == "m_end":
  97             self.set_focus(len(self.walker.bindings) - 1)
  98             self.walker._modified()
  99         return super().keypress(size, key)

 102 class KeyHelp(urwid.Frame):
 103     def __init__(self, master):
 104         self.master = master
 105         super().__init__(self.widget(""))
 106         self.set_active(False)
 107         keybinding_focus_change.connect(self.sig_mod)

 109     def set_active(self, val):
 110         h = urwid.Text("Key Binding Help")
 111         style = "heading" if val else "heading_inactive"
 112         self.header = urwid.AttrWrap(h, style)
 113 
 114     def widget(self, txt):
 115         cols, _ = self.master.ui.get_cols_rows()
 116         return urwid.ListBox(
 117             [urwid.Text(i) for i in textwrap.wrap(txt, cols)]
 118         )

 120     def sig_mod(self, txt):
 121         self.set_body(self.widget(txt))

 152         # This is essentially a copypasta from urwid.Pile's keypress handler.
