   9 class CListBox(urwid.ListBox):
  10     def __init__(self, contents):
  11         self.length = len(contents)
  12         contents = contents[:] + [urwid.Text(["\n"])] * 5
  13         super().__init__(contents)
  14 
  15     def keypress(self, size, key):
  16         if key == "m_end":
  17             self.set_focus(self.length - 1)
  18         elif key == "m_start":
  19             self.set_focus(0)
  20         else:
  21             return super().keypress(size, key)

  41     def format_keys(self, binds):
  42         kvs = []
  43         for b in binds:
  44             k = b.key
  45             if b.key == " ":
  46                 k = "space"
  47             kvs.append((k, b.help or b.command))
  48         return common.format_keyvals(kvs)

  50     def keybindings(self):

  63             urwid.Text(

  89     def filtexp(self):
  90         text = []
  91         text.extend(common.format_keyvals(flowfilter.help, indent=4))
  92         text.append(
  93             urwid.Text(
  94                 [
  95                     "\n",
  96                     ("text", "    Regexes are Python-style.\n"),
  97                     ("text", "    Regexes can be specified as quoted strings.\n"),
  98                     ("text", "    Header matching (~h, ~hq, ~hs) is against a string of the form \"name: value\".\n"),
  99                     ("text", "    Expressions with no operators are regex matches against URL.\n"),
 100                     ("text", "    Default binary operator is &.\n"),
 101                     ("head", "\n    Examples:\n"),
 102                 ]
 103             )
 104         )
 105         examples = [
 106             (r"google\.com", r"Url containing \"google.com"),
 107             ("~q ~b test", r"Requests where body contains \"test\""),
 108             (r"!(~q & ~t \"text/html\")", "Anything but requests with a text/html content type."),
 109         ]
 110         text.extend(
 111             common.format_keyvals(examples, indent=4)
 112         )
 113         return CListBox(text)
