  47     def positions(self, reverse=False):
  48         # The stub implementation of positions can go once this issue is resolved:
  49         # https://github.com/urwid/urwid/issues/294
  50         ret = range(self.master.commands.execute("view.properties.length"))
  51         if reverse:
  52             return reversed(ret)
  53         return ret
  54 
  55     def view_changed(self):
  56         self._modified()

  58     def get_focus(self):
  59         if not self.master.view.focus.flow:
  60             return None, 0
  61         f = FlowItem(self.master, self.master.view.focus.flow)
  62         return f, self.master.view.focus.index
  63 
  64     def set_focus(self, index):
  65         if self.master.commands.execute("view.properties.inbounds %d" % index):
  66             self.master.view.focus.index = index
  67 
  68     def get_next(self, pos):
  69         pos = pos + 1
  70         if not self.master.commands.execute("view.properties.inbounds %d" % pos):
  71             return None, None
  72         f = FlowItem(self.master, self.master.view[pos])
  73         return f, pos
  74 
  75     def get_prev(self, pos):
  76         pos = pos - 1
  77         if not self.master.commands.execute("view.properties.inbounds %d" % pos):
  78             return None, None
  79         f = FlowItem(self.master, self.master.view[pos])
  80         return f, pos
