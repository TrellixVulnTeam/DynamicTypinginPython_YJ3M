  52     def palette(self, transparent):
  53         l = []
  54         highback, lowback = None, None
  55         if not transparent:
  56             if self.high and self.high.get("background"):
  57                 highback = self.high["background"][1]
  58             lowback = self.low["background"][1]
  59 
  60         for i in self._fields:
  61             if transparent and i == "background":
  62                 l.append(["background", "default", "default"])
  63             else:
  64                 v = [i]
  65                 low = list(self.low[i])
  66                 if lowback and low[1] == "default":
  67                     low[1] = lowback
  68                 v.extend(low)
  69                 if self.high and i in self.high:
  70                     v.append(None)
  71                     high = list(self.high[i])
  72                     if highback and high[1] == "default":
  73                         high[1] = highback
  74                     v.extend(high)
  75                 elif highback and self.low[i][1] == "default":
  76                     high = [None, low[0], highback]
  77                     v.extend(high)
  78                 l.append(tuple(v))
  79         return l

  82 def gen_gradient(palette, cols):
  83     for i in range(100):
  84         palette['gradient_%02d' % i] = (cols[i * len(cols) // 100], 'default')

  87 def gen_rgb_gradient(palette, cols):
  88     parts = len(cols) - 1
  89     for i in range(100):
  90         p = i / 100
  91         idx = int(p * parts)
  92         t0 = cols[idx]
  93         t1 = cols[idx + 1]
  94         pp = p * parts % 1
  95         t = (
  96             round(t0[0] + (t1[0] - t0[0]) * pp),
  97             round(t0[1] + (t1[1] - t0[1]) * pp),
  98             round(t0[2] + (t1[2] - t0[2]) * pp),
  99         )
 100         palette['gradient_%02d' % i] = ("#%x%x%x" % t, 'default')
