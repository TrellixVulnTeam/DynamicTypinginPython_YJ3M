  42 def format_keyvals(
  43         entries: typing.Iterable[typing.Tuple[str, typing.Union[None, str, urwid.Widget]]],
  44         key_format: str = "key",
  45         value_format: str = "text",
  46         indent: int = 0
  47 ) -> typing.List[urwid.Columns]:
  48     """
  49     Format a list of (key, value) tuples.
  50 
  51     Args:
  52         entries: The list to format. keys must be strings, values can also be None or urwid widgets.
  53             The latter makes it possible to use the result of format_keyvals() as a value.
  54         key_format: The display attribute for the key.
  55         value_format: The display attribute for the value.
  56         indent: Additional indent to apply.
  57     """
  58     max_key_len = max((len(k) for k, v in entries if k is not None), default=0)
  59     max_key_len = min(max_key_len, KEY_MAX)
  60 
  61     if indent > 2:
  62         indent -= 2  # We use dividechars=2 below, which already adds two empty spaces
  63 
  64     ret = []
  65     for k, v in entries:
  66         if v is None:
  67             v = urwid.Text("")
  68         elif not isinstance(v, urwid.Widget):
  69             v = urwid.Text([(value_format, v)])
  70         ret.append(
  71             urwid.Columns(
  72                 [
  73                     ("fixed", indent, urwid.Text("")),
  74                     (
  75                         "fixed",
  76                         max_key_len,
  77                         urwid.Text([(key_format, k)])
  78                     ),
  79                     v
  80                 ],
  81                 dividechars=2
  82             )
  83         )
  84     return ret

 175     def render(self, size, focus=False):
 176         text = self.text
 177         attr = self.attr
 178         if self.align == 'right':
 179             text = text[::-1]
 180             attr = attr[::-1]

 182         text_len = len(text)  # TODO: unicode?
 183         if size is not None and len(size) > 0:

 216     """
 217     Append (a, r) (unpacked from *a_r*) to BEGINNING of rle.
 218     Merge with first run when possible

 233 def colorize_host(host):
 234     tld = get_tld(host)
 235     sld = get_sld(host)
 236 
 237     attr = []
 238 
 239     tld_size = len(tld)
 240     sld_size = len(sld) - tld_size
 241 
 242     for letter in reversed(range(len(host))):
 243         character = host[letter]
 244         if tld_size > 0:
 245             style = 'url_domain'
 246             tld_size -= 1
 247         elif tld_size == 0:
 248             style = 'text'
 249             tld_size -= 1
 250         elif sld_size > 0:
 251             sld_size -= 1
 252             style = 'url_extension'
 253         else:
 254             style = 'text'
 255         rle_append_beginning_modify(attr, (style, len(character.encode())))
 256     return attr

 259 def colorize_req(s):
 260     path = s.split('?', 2)[0]
 261     i_query = len(path)
 262     i_last_slash = path.rfind('/')
 263     i_ext = path[i_last_slash + 1:].rfind('.')
 264     i_ext = i_last_slash + i_ext if i_ext >= 0 else len(s)
 265     in_val = False
 266     attr = []
 267     for i in range(len(s)):
 268         c = s[i]
 269         if ((i < i_query and c == '/') or
 270                 (i < i_query and i > i_last_slash and c == '.') or
 271                 (i == i_query)):
 272             a = 'url_punctuation'
 273         elif i > i_query:
 274             if in_val:
 275                 if c == '&':
 276                     in_val = False
 277                     a = 'url_punctuation'
 278                 else:
 279                     a = 'url_query_value'
 280             else:
 281                 if c == '=':
 282                     in_val = True
 283                     a = 'url_punctuation'
 284                 else:
 285                     a = 'url_query_key'
 286         elif i > i_ext:
 287             a = 'url_extension'
 288         elif i > i_last_slash:
 289             a = 'url_filename'
 290         else:
 291             a = 'text'
 292         urwid.util.rle_append_modify(attr, (a, len(c.encode())))
 293     return attr

 296 def colorize_url(url):
 297     parts = url.split('/', 3)
 298     if len(parts) < 4 or len(parts[1]) > 0 or parts[0][-1:] != ':':
 299         return [('error', len(url))]  # bad URL
 300     schemes = {
 301         'http:': 'scheme_http',
 302         'https:': 'scheme_https',
 303     }
 304     return [
 305                (schemes.get(parts[0], "scheme_other"), len(parts[0]) - 1),
 306                ('url_punctuation', 3),  # ://
 307            ] + colorize_host(parts[2]) + colorize_req('/' + parts[3])

 310 def format_http_content_type(content_type: str) -> typing.Tuple[str, str]:
 311     content_type = content_type.split(";")[0]
 312     if content_type.endswith('/javascript'):
 313         style = 'content_script'
 314     elif content_type.startswith('text/'):
 315         style = 'content_text'
 316     elif (content_type.startswith('image/') or
 317           content_type.startswith('video/') or
 318           content_type.startswith('font/') or
 319           "/x-font-" in content_type):
 320         style = 'content_media'
 321     elif content_type.endswith('/json') or content_type.endswith('/xml'):
 322         style = 'content_data'
 323     elif content_type.startswith('application/'):
 324         style = 'content_raw'
 325     else:
 326         style = 'content_other'
 327     return content_type, style

 443             style = "intercept"
 444         else:
 445             style = ""
 446

 457             resp.append(fcol(ct, style or ct_style))

 461         elif response_content_length == 0:

 468 
 469         if duration:

 471             resp.append(fcol(dur, style or dur_style))
 472     elif error_message:
 473         resp.append(fcol(SYMBOL_RETURN, "error"))
