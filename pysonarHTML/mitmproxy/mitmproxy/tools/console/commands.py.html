  14 class CommandItem(urwid.WidgetWrap):
  15     def __init__(self, walker, cmd: command.Command, focused: bool):
  16         self.walker, self.cmd, self.focused = walker, cmd, focused
  17         super().__init__(None)
  18         self._w = self.get_widget()
  19 
  20     def get_widget(self):
  21         parts = [
  22             ("focus", ">> " if self.focused else "   "),
  23             ("title", self.cmd.name)
  24         ]
  25         if self.cmd.parameters:
  26             parts += [
  27                 ("text", " "),
  28                 ("text", " ".join(str(param) for param in self.cmd.parameters)),
  29             ]
  30         if self.cmd.return_type:
  31             parts += [
  32                 ("title", " -> "),
  33                 ("text", command.typename(self.cmd.return_type)),
  34             ]
  35 
  36         return urwid.AttrMap(
  37             urwid.Padding(urwid.Text(parts)),
  38             "text"
  39         )

  42         return self._w[1].get_edit_text()

  51 class CommandListWalker(urwid.ListWalker):
  52     def __init__(self, master):
  53         self.master = master
  54         self.index = 0
  55         self.refresh()
  56 
  57     def refresh(self):
  58         self.cmds = list(self.master.commands.commands.values())
  59         self.cmds.sort(key=lambda x: x.signature_help())
  60         self.set_focus(self.index)
  61 
  62     def get_edit_text(self):
  63         return self.focus_obj.get_edit_text()
  64 
  65     def _get(self, pos):
  66         cmd = self.cmds[pos]
  67         return CommandItem(self, cmd, pos == self.index)
  68 
  69     def get_focus(self):
  70         return self.focus_obj, self.index
  71 
  72     def set_focus(self, index):
  73         cmd = self.cmds[index]
  74         self.index = index
  75         self.focus_obj = self._get(self.index)
  76         command_focus_change.send(cmd.help or "")
  77 
  78     def get_next(self, pos):
  79         if pos >= len(self.cmds) - 1:
  80             return None, None
  81         pos = pos + 1
  82         return self._get(pos), pos
  83 
  84     def get_prev(self, pos):
  85         pos = pos - 1
  86         if pos < 0:
  87             return None, None
  88         return self._get(pos), pos

  91 class CommandsList(urwid.ListBox):
  92     def __init__(self, master):
  93         self.master = master
  94         self.walker = CommandListWalker(master)
  95         super().__init__(self.walker)
  96 
  97     def keypress(self, size, key):
  98         if key == "m_select":
  99             foc, idx = self.get_focus()
 100             signals.status_prompt_command.send(partial=foc.cmd.name + " ")
 101         elif key == "m_start":
 102             self.set_focus(0)
 103             self.walker._modified()
 104         elif key == "m_end":
 105             self.set_focus(len(self.walker.cmds) - 1)
 106             self.walker._modified()
 107         return super().keypress(size, key)

 110 class CommandHelp(urwid.Frame):
 111     def __init__(self, master):
 112         self.master = master
 113         super().__init__(self.widget(""))
 114         self.set_active(False)
 115         command_focus_change.connect(self.sig_mod)
 116 
 117     def set_active(self, val):
 118         h = urwid.Text("Command Help")
 119         style = "heading" if val else "heading_inactive"
 120         self.header = urwid.AttrWrap(h, style)
 121 
 122     def widget(self, txt):
 123         cols, _ = self.master.ui.get_cols_rows()
 124         return urwid.ListBox(
 125             [urwid.Text(i) for i in textwrap.wrap(txt, cols)]
 126         )
 127 
 128     def sig_mod(self, txt):
 129         self.set_body(self.widget(txt))

 132 class Commands(urwid.Pile, layoutwidget.LayoutWidget):

 136     def __init__(self, master):
 137         oh = CommandHelp(master)
 138         super().__init__(

 146     def layout_pushed(self, prev):
 147         self.widget_list[0].walker.refresh()
 148 
 149     def keypress(self, size, key):
 150         if key == "m_next":
 151             self.focus_position = (
 152                 self.focus_position + 1
 153             ) % len(self.widget_list)
 154             self.widget_list[1].set_active(self.focus_position == 1)
 155             key = None
 156 
 157         # This is essentially a copypasta from urwid.Pile's keypress handler.
 158         # So much for "closed for modification, but open for extension".
 159         item_rows = None
 160         if len(size) == 2:
 161             item_rows = self.get_item_rows(size, focus = True)
 162         i = self.widget_list.index(self.focus_item)
 163         tsize = self.get_item_size(size, i, True, item_rows)
 164         return self.focus_item.keypress(tsize, key)
