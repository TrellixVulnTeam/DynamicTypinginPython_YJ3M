  18 class StackWidget(urwid.Frame):
  19     def __init__(self, window, widget, title, focus):
  20         self.is_focused = focus
  21         self.window = window
  22 
  23         if title:
  24             header = urwid.AttrWrap(
  25                 urwid.Text(title),
  26                 "heading" if focus else "heading_inactive"
  27             )
  28         else:
  29             header = None
  30         super().__init__(
  31             widget,
  32             header=header
  33         )
  34 
  35     def mouse_event(self, size, event, button, col, row, focus):
  36         if event == "mouse press" and button == 1 and not self.is_focused:
  37             self.window.switch()
  38         return super().mouse_event(size, event, button, col, row, focus)
  39 
  40     def keypress(self, size, key):
  41         # Make sure that we don't propagate cursor events outside of the widget.
  42         # Otherwise, in a horizontal layout, urwid's Pile would change the focused widget
  43         # if we cannot scroll any further.
  44         ret = super().keypress(size, key)
  45         command = self._command_map[ret]  # awkward as they don't implement a full dict api
  46         if command and command.startswith("cursor"):
  47             return None
  48         return ret

  51 class WindowStack:
  52     def __init__(self, master, base):
  53         self.master = master
  54         self.windows = dict(
  55             flowlist = flowlist.FlowListBox(master),
  56             flowview = flowview.FlowView(master),
  57             commands = commands.Commands(master),
  58             keybindings = keybindings.KeyBindings(master),
  59             options = options.Options(master),
  60             help = help.HelpView(master),
  61             eventlog = eventlog.EventLog(master),
  62 
  63             edit_focus_query = grideditor.QueryEditor(master),
  64             edit_focus_cookies = grideditor.CookieEditor(master),
  65             edit_focus_setcookies = grideditor.SetCookieEditor(master),
  66             edit_focus_setcookie_attrs = grideditor.CookieAttributeEditor(master),
  67             edit_focus_multipart_form=grideditor.RequestMultipartEditor(master),
  68             edit_focus_urlencoded_form=grideditor.RequestUrlEncodedEditor(master),
  69             edit_focus_path = grideditor.PathEditor(master),
  70             edit_focus_request_headers = grideditor.RequestHeaderEditor(master),
  71             edit_focus_response_headers = grideditor.ResponseHeaderEditor(master),
  72         )
  73         self.stack = [base]
  74         self.overlay = None
  75 
  76     def set_overlay(self, o, **kwargs):
  77         self.overlay = overlay.SimpleOverlay(
  78             self, o, self.top_widget(), o.width, **kwargs,
  79         )
  80 
  81     def top_window(self):
  82         """
  83             The current top window, ignoring overlays.
  84         """
  85         return self.windows[self.stack[-1]]
  86 
  87     def top_widget(self):
  88         """
  89             The current top widget - either a window or the active overlay.
  90         """
  91         if self.overlay:
  92             return self.overlay
  93         return self.top_window()
  94 
  95     def push(self, wname):
  96         if self.stack[-1] == wname:
  97             return
  98         prev = self.top_window()
  99         self.stack.append(wname)
 100         self.call("layout_pushed", prev)
 101 
 102     def pop(self, *args, **kwargs):
 103         """
 104             Pop off the stack, return True if we're already at the top.
 105         """
 106         if not self.overlay and len(self.stack) == 1:
 107             return True
 108         self.call("layout_popping")
 109         if self.overlay:
 110             self.overlay = None
 111         else:
 112             self.stack.pop()
 113 
 114     def call(self, name, *args, **kwargs):
 115         """
 116             Call a function on both the top window, and the overlay if there is
 117             one. If the widget has a key_responder, we call the function on the
 118             responder instead.
 119         """
 120         getattr(self.top_window(), name)(*args, **kwargs)
 121         if self.overlay:
 122             getattr(self.overlay, name)(*args, **kwargs)

 125 class Window(urwid.Frame):
 126     def __init__(self, master):
 127         self.statusbar = statusbar.StatusBar(master)
 128         super().__init__(
 129             None,
 130             header = None,
 131             footer = urwid.AttrWrap(self.statusbar, "background")
 132         )
 133         self.master = master
 134         self.master.view.sig_view_refresh.connect(self.view_changed)
 135         self.master.view.sig_view_add.connect(self.view_changed)
 136         self.master.view.sig_view_remove.connect(self.view_changed)
 137         self.master.view.sig_view_update.connect(self.view_changed)
 138         self.master.view.focus.sig_change.connect(self.view_changed)
 139         self.master.view.focus.sig_change.connect(self.focus_changed)
 140 
 141         signals.focus.connect(self.sig_focus)
 142         signals.flow_change.connect(self.flow_changed)
 143         signals.pop_view_state.connect(self.pop)
 144         signals.push_view_state.connect(self.push)
 145 
 146         self.master.options.subscribe(self.configure, ["console_layout"])
 147         self.master.options.subscribe(self.configure, ["console_layout_headers"])
 148         self.pane = 0
 149         self.stacks = [
 150             WindowStack(master, "flowlist"),
 151             WindowStack(master, "eventlog")
 152         ]
 153 
 154     def focus_stack(self):
 155         return self.stacks[self.pane]
 156 
 157     def configure(self, otions, updated):
 158         self.refresh()
 159 
 160     def refresh(self):
 161         """
 162             Redraw the layout.
 163         """
 164         c = self.master.options.console_layout
 165         if c == "single":
 166             self.pane = 0
 167 
 168         def wrapped(idx):
 169             widget = self.stacks[idx].top_widget()
 170             if self.master.options.console_layout_headers:
 171                 title = self.stacks[idx].top_window().title
 172             else:
 173                 title = None
 174             return StackWidget(
 175                 self,
 176                 widget,
 177                 title,
 178                 self.pane == idx
 179             )
 180 
 181         w = None
 182         if c == "single":
 183             w = wrapped(0)
 184         elif c == "vertical":
 185             w = urwid.Pile(
 186                 [
 187                     wrapped(i) for i, s in enumerate(self.stacks)
 188                 ],
 189                 focus_item=self.pane
 190             )
 191         else:
 192             w = urwid.Columns(
 193                 [wrapped(i) for i, s in enumerate(self.stacks)],
 194                 dividechars=1,
 195                 focus_column=self.pane
 196             )
 197 
 198         self.body = urwid.AttrWrap(w, "background")
 199 
 200     def flow_changed(self, sender, flow):
 201         if self.master.view.focus.flow:
 202             if flow.id == self.master.view.focus.flow.id:
 203                 self.focus_changed()
 204 
 205     def focus_changed(self, *args, **kwargs):
 206         """
 207             Triggered when the focus changes - either when it's modified, or
 208             when it changes to a different flow altogether.
 209         """
 210         for i in self.stacks:
 211             i.call("focus_changed")
 212 
 213     def view_changed(self, *args, **kwargs):
 214         """
 215             Triggered when the view list has changed.
 216         """
 217         for i in self.stacks:
 218             i.call("view_changed")
 219 
 220     def set_overlay(self, o, **kwargs):
 221         """
 222             Set an overlay on the currently focused stack.
 223         """
 224         self.focus_stack().set_overlay(o, **kwargs)
 225         self.refresh()
 226 
 227     def push(self, wname):
 228         """
 229             Push a window onto the currently focused stack.
 230         """
 231         self.focus_stack().push(wname)
 232         self.refresh()
 233         self.view_changed()
 234         self.focus_changed()
 235 
 236     def pop(self, *args, **kwargs):
 237         """
 238             Pop a window from the currently focused stack. If there is only one
 239             window on the stack, this prompts for exit.
 240         """
 241         if self.focus_stack().pop():
 242             self.master.prompt_for_exit()
 243         else:
 244             self.refresh()
 245             self.view_changed()
 246             self.focus_changed()
 247 
 248     def stacks_sorted_by_focus(self):
 249         """
 250         Returns:
 251             self.stacks, with the focused stack first.
 252         """
 253         stacks = self.stacks.copy()
 254         stacks.insert(0, stacks.pop(self.pane))
 255         return stacks
 256 
 257     def current(self, keyctx):
 258         """
 259         Returns the active widget with a matching key context, including overlays.
 260         If multiple stacks have an active widget with a matching key context,
 261         the currently focused stack is preferred.
 262         """
 263         for s in self.stacks_sorted_by_focus():
 264             t = s.top_widget()
 265             if t.keyctx == keyctx:
 266                 return t
 267 
 268     def current_window(self, keyctx):
 269         """
 270         Returns the active window with a matching key context, ignoring overlays.
 271         If multiple stacks have an active widget with a matching key context,
 272         the currently focused stack is preferred.
 273         """
 274         for s in self.stacks_sorted_by_focus():
 275             t = s.top_window()
 276             if t.keyctx == keyctx:
 277                 return t
 278 
 279     def sig_focus(self, sender, section):
 280         self.focus_position = section
 281 
 282     def switch(self):
 283         """
 284             Switch between the two panes.
 285         """
 286         if self.master.options.console_layout == "single":
 287             self.pane = 0
 288         else:
 289             self.pane = (self.pane + 1) % len(self.stacks)
 290         self.refresh()
 291 
 292     def mouse_event(self, *args, **kwargs):
 293         # args: (size, event, button, col, row)
 294         k = super().mouse_event(*args, **kwargs)
 295         if not k:
 296             if args[1] == "mouse drag":
 297                 signals.status_message.send(
 298                     message = "Hold down fn, shift, alt or ctrl to select text or use the --set console_mouse=false parameter.",
 299                     expire = 1
 300                 )
 301             elif args[1] == "mouse press" and args[2] == 4:
 302                 self.keypress(args[0], "up")
 303             elif args[1] == "mouse press" and args[2] == 5:
 304                 self.keypress(args[0], "down")
 305             else:
 306                 return False
 307             return True
 308 
 309     def keypress(self, size, k):
 310         k = super().keypress(size, k)
 311         if k:
 312             return self.master.keymap.handle(
 313                 self.focus_stack().top_widget().keyctx,
 314                 k
 315             )

 320     def write(self, data):
 321         if common.IS_WSL:
 322             # replace urwid's SI/SO, which produce artifacts under WSL.
 323             # at some point we may figure out what they actually do.
 324             data = re.sub("[\x0e\x0f]", "", data)
 325         super().write(data)
