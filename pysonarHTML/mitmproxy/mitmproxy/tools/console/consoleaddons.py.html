  50     def websocket_message(self, f):
  51         message = f.messages[-1]
  52         ctx.log.info(f.message_info(message))
  53         ctx.log.debug(
  54             message.content if isinstance(message.content, str) else strutils.bytes_to_escaped_str(message.content))
  55 
  56     def websocket_end(self, f):
  57         ctx.log.info("WebSocket connection closed by {}: {} {}, {}".format(
  58             f.close_sender,
  59             f.close_code,
  60             f.close_message,
  61             f.close_reason))
  62 
  63     def tcp_message(self, f):
  64         message = f.messages[-1]
  65         direction = "->" if message.from_client else "<-"
  66         ctx.log.info("{client_host}:{client_port} {direction} tcp {direction} {server_host}:{server_port}".format(
  67             client_host=f.client_conn.address[0],
  68             client_port=f.client_conn.address[1],
  69             server_host=f.server_conn.address[0],
  70             server_port=f.server_conn.address[1],
  71             direction=direction,
  72         ))
  73         ctx.log.debug(strutils.bytes_to_escaped_str(message.content))

 236     @command.command("console.choose")
 237     def console_choose(
 238             self,
 239             prompt: str,
 240             choices: typing.Sequence[str],
 241             cmd: mitmproxy.types.Cmd,
 242             *args: mitmproxy.types.CmdArgs
 243     ) -> None:
 244         """
 245             Prompt the user to choose from a specified list of strings, then
 246             invoke another command with all occurrences of {choice} replaced by
 247             the choice the user made.
 248         """
 249 
 250         def callback(opt):
 251             # We're now outside of the call context...
 252             repl = [arg.replace("{choice}", opt) for arg in args]
 253             try:
 254                 self.master.commands.call_strings(cmd, repl)
 255             except exceptions.CommandError as e:
 256                 ctx.log.error(str(e))
 257 
 258         self.master.overlay(
 259             overlay.Chooser(self.master, prompt, choices, "", callback)
 260         )

 262     @command.command("console.choose.cmd")
 263     def console_choose_cmd(
 264             self,
 265             prompt: str,
 266             choicecmd: mitmproxy.types.Cmd,
 267             subcmd: mitmproxy.types.Cmd,
 268             *args: mitmproxy.types.CmdArgs
 269     ) -> None:
 270         """
 271             Prompt the user to choose from a list of strings returned by a
 272             command, then invoke another command with all occurrences of {choice}
 273             replaced by the choice the user made.
 274         """
 275         choices = ctx.master.commands.execute(choicecmd)
 276 
 277         def callback(opt):
 278             # We're now outside of the call context...
 279             repl = [arg.replace("{choice}", opt) for arg in args]
 280             try:
 281                 self.master.commands.call_strings(subcmd, repl)
 282             except exceptions.CommandError as e:
 283                 ctx.log.error(str(e))
 284 
 285         self.master.overlay(
 286             overlay.Chooser(self.master, prompt, choices, "", callback)
 287         )

 358     @command.command("console.bodyview")
 359     @command.argument("part", type=mitmproxy.types.Choice("console.bodyview.options"))
 360     def bodyview(self, flow: flow.Flow, part: str) -> None:
 361         """
 362             Spawn an external viewer for a flow request or response body based
 363             on the detected MIME type. We use the mailcap system to find the
 364             correct viewier, and fall back to the programs in $PAGER or $EDITOR
 365             if necessary.
 366         """
 367         fpart = getattr(flow, part, None)
 368         if not fpart:
 369             raise exceptions.CommandError("Part must be either request or response, not %s." % part)
 370         t = fpart.headers.get("content-type")
 371         content = fpart.get_content(strict=False)
 372         if not content:
 373             raise exceptions.CommandError("No content to view.")
 374         self.master.spawn_external_viewer(content, t)

 413     @command.command("console.edit.focus")
 414     @command.argument("flow_part", type=mitmproxy.types.Choice("console.edit.focus.options"))
 415     def edit_focus(self, flow_part: str) -> None:
 416         """
 417             Edit a component of the currently focused flow.
 418         """
 419         flow = self.master.view.focus.flow
 420         # This shouldn't be necessary once this command is "console.edit @focus",
 421         # but for now it is.
 422         if not flow:
 423             raise exceptions.CommandError("No flow selected.")
 424         flow.backup()
 425 
 426         require_dummy_response = (
 427                 flow_part in ("response-headers", "response-body", "set-cookies") and
 428                 flow.response is None
 429         )
 430         if require_dummy_response:
 431             flow.response = http.HTTPResponse.make()
 432         if flow_part == "cookies":
 433             self.master.switch_view("edit_focus_cookies")
 434         elif flow_part == "urlencoded form":
 435             self.master.switch_view("edit_focus_urlencoded_form")
 436         elif flow_part == "multipart form":
 437             self.master.switch_view("edit_focus_multipart_form")
 438         elif flow_part == "path":
 439             self.master.switch_view("edit_focus_path")
 440         elif flow_part == "query":
 441             self.master.switch_view("edit_focus_query")
 442         elif flow_part == "request-headers":
 443             self.master.switch_view("edit_focus_request_headers")
 444         elif flow_part == "response-headers":
 445             self.master.switch_view("edit_focus_response_headers")
 446         elif flow_part in ("request-body", "response-body"):
 447             if flow_part == "request-body":
 448                 message = flow.request
 449             else:
 450                 message = flow.response
 451             c = self.master.spawn_editor(message.get_content(strict=False) or b"")
 452             # Fix an issue caused by some editors when editing a
 453             # request/response body. Many editors make it hard to save a
 454             # file without a terminating newline on the last line. When
 455             # editing message bodies, this can cause problems. For now, I
 456             # just strip the newlines off the end of the body when we return
 457             # from an editor.
 458             message.content = c.rstrip(b"\n")
 459         elif flow_part == "set-cookies":
 460             self.master.switch_view("edit_focus_setcookies")
 461         elif flow_part == "url":
 462             url = flow.request.url.encode()
 463             edited_url = self.master.spawn_editor(url)
 464             url = edited_url.rstrip(b"\n")
 465             flow.request.url = url.decode()
 466         elif flow_part in ["method", "status_code", "reason"]:
 467             self.master.commands.call_strings(
 468                 "console.command",
 469                 ["flow.set", "@focus", flow_part]
 470             )
 471         elif flow_part == "tcp-message":
 472             message = flow.messages[-1]
 473             c = self.master.spawn_editor(message.content or b"")
 474             message.content = c.rstrip(b"\n")
