  21 def assert_utf8_env():
  22     spec = ""
  23     for i in ["LANG", "LC_CTYPE", "LC_ALL"]:
  24         spec += os.environ.get(i, "").lower()
  25     if "utf" not in spec:
  26         print(
  27             "Error: mitmproxy requires a UTF console environment.",
  28             file=sys.stderr
  29         )
  30         print(
  31             "Set your LANG environment variable to something like en_US.UTF-8",
  32             file=sys.stderr
  33         )
  34         sys.exit(1)

  37 def process_options(parser, opts, args):
  38     if args.version:
  39         print(debug.dump_system_info())
  40         sys.exit(0)
  41     if args.quiet or args.options or args.commands:
  42         # also reduce log verbosity if --options or --commands is passed,
  43         # we don't want log messages from regular startup then.
  44         args.termlog_verbosity = 'error'
  45         args.flow_detail = 0
  46     if args.verbose:
  47         args.termlog_verbosity = 'debug'
  48         args.flow_detail = 2
  49 
  50     adict = {}
  51     for n in dir(args):
  52         if n in opts:
  53             adict[n] = getattr(args, n)
  54     opts.merge(adict)
  55 
  56     return proxy.config.ProxyConfig(opts)

  59 def run(
  60         master_cls: typing.Type[master.Master],
  61         make_parser: typing.Callable[[options.Options], argparse.ArgumentParser],
  62         arguments: typing.Sequence[str],
  63         extra: typing.Callable[[typing.Any], dict] = None
  64 ) -> master.Master:  # pragma: no cover
  65     """
  66         extra: Extra argument processing callable which returns a dict of
  67         options.
  68     """
  69     debug.register_info_dumpers()
  70 
  71     opts = options.Options()
  72     master = master_cls(opts)
  73 
  74     parser = make_parser(opts)
  75 
  76     # To make migration from 2.x to 3.0 bearable.
  77     if "-R" in sys.argv and sys.argv[sys.argv.index("-R") + 1].startswith("http"):
  78         print("To use mitmproxy in reverse mode please use --mode reverse:SPEC instead")
  79 
  80     try:
  81         args = parser.parse_args(arguments)
  82     except SystemExit:
  83         arg_check.check()
  84         sys.exit(1)
  85 
  86     try:
  87         opts.set(*args.setoptions, defer=True)
  88         optmanager.load_paths(
  89             opts,
  90             os.path.join(opts.confdir, "config.yaml"),
  91             os.path.join(opts.confdir, "config.yml"),
  92         )
  93         pconf = process_options(parser, opts, args)
  94         server: typing.Any = None
  95         if pconf.options.server:
  96             try:
  97                 server = proxy.server.ProxyServer(pconf)
  98             except exceptions.ServerException as v:
  99                 print(str(v), file=sys.stderr)
 100                 sys.exit(1)
 101         else:
 102             server = proxy.server.DummyServer(pconf)
 103 
 104         master.server = server
 105         if args.options:
 106             print(optmanager.dump_defaults(opts))
 107             sys.exit(0)
 108         if args.commands:
 109             master.commands.dump()
 110             sys.exit(0)
 111         if extra:
 112             if(args.filter_args):
 113                 master.log.info(f"Only processing flows that match \"{' & '.join(args.filter_args)}\"")
 114             opts.update(**extra(args))
 115 
 116         loop = asyncio.get_event_loop()
 117         try:
 118             loop.add_signal_handler(signal.SIGINT, getattr(master, "prompt_for_exit", master.shutdown))
 119             loop.add_signal_handler(signal.SIGTERM, master.shutdown)
 120         except NotImplementedError:
 121             # Not supported on Windows
 122             pass
 123 
 124         # Make sure that we catch KeyboardInterrupts on Windows.
 125         # https://stackoverflow.com/a/36925722/934719
 126         if os.name == "nt":
 127             async def wakeup():
 128                 while True:
 129                     await asyncio.sleep(0.2)
 130             asyncio.ensure_future(wakeup())
 131 
 132         master.run()
 133     except exceptions.OptionsError as e:
 134         print("%s: %s" % (sys.argv[0], e), file=sys.stderr)
 135         sys.exit(1)
 136     except (KeyboardInterrupt, RuntimeError):
 137         pass
 138     return master

 141 def mitmproxy(args=None) -> typing.Optional[int]:  # pragma: no cover
 142     if os.name == "nt":
 143         print("Error: mitmproxy's console interface is not supported on Windows. "
 144               "You can run mitmdump or mitmweb instead.", file=sys.stderr)
 145         return 1
 146     assert_utf8_env()
 147     from mitmproxy.tools import console
 148     run(console.master.ConsoleMaster, cmdline.mitmproxy, args)
 149     return None

 152 def mitmdump(args=None) -> typing.Optional[int]:  # pragma: no cover
 153     from mitmproxy.tools import dump
 154 
 155     def extra(args):
 156         if args.filter_args:
 157             v = " ".join(args.filter_args)
 158             return dict(
 159                 save_stream_filter=v,
 160                 readfile_filter=v,
 161                 dumper_filter=v,
 162             )
 163         return {}
 164 
 165     m = run(dump.DumpMaster, cmdline.mitmdump, args, extra)
 166     if m and m.errorcheck.has_errored:  # type: ignore
 167         return 1
 168     return None
