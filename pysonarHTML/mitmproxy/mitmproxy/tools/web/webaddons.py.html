  25     def running(self):
  26         if hasattr(ctx.options, "web_open_browser") and ctx.options.web_open_browser:
  27             web_url = "http://{}:{}/".format(ctx.options.web_host, ctx.options.web_port)
  28             success = open_browser(web_url)
  29             if not success:
  30                 ctx.log.info(
  31                     "No web browser found. Please open a browser and point it to {}".format(web_url),
  32                 )

  35 def open_browser(url: str) -> bool:
  36     """
  37     Open a URL in a browser window.
  38     In contrast to webbrowser.open, we limit the list of suitable browsers.
  39     This gracefully degrades to a no-op on headless servers, where webbrowser.open
  40     would otherwise open lynx.
  41 
  42     Returns:
  43         True, if a browser has been opened
  44         False, if no suitable browser has been found.
  45     """
  46     browsers = (
  47         "windows-default", "macosx",
  48         "wslview %s",
  49         "x-www-browser %s", "gnome-open %s",
  50         "google-chrome", "chrome", "chromium", "chromium-browser",
  51         "firefox", "opera", "safari",
  52     )
  53     for browser in browsers:
  54         try:
  55             b = webbrowser.get(browser)
  56         except webbrowser.Error:
  57             pass
  58         else:
  59             if b.open(url):
  60                 return True
  61     return False
