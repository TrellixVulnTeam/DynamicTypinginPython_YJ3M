  18 def save_static(path: pathlib.Path) -> None:
  19     """
  20     Save the files for the static web view.
  21     """
  22     # We want to overwrite the static files to keep track of the update.
  23     if (path / "static").exists():
  24         shutil.rmtree(str(path / "static"))
  25     shutil.copytree(str(web_dir / "static"), str(path / "static"))
  26     shutil.copyfile(str(web_dir / 'templates' / 'index.html'), str(path / "index.html"))
  27 
  28     with open(str(path / "static" / "static.js"), "w") as f:
  29         f.write("MITMWEB_STATIC = true;")

  42 def save_flows(path: pathlib.Path, flows: typing.Iterable[flow.Flow]) -> None:
  43     with open(str(path / 'flows.json'), 'w') as f:
  44         json.dump(
  45             [flow_to_json(f) for f in flows],
  46             f
  47         )

  50 def save_flows_content(path: pathlib.Path, flows: typing.Iterable[flow.Flow]) -> None:
  51     for f in flows:
  52         for m in ('request', 'response'):
  53             message = getattr(f, m)
  54             message_path = path / "flows" / f.id / m
  55             os.makedirs(str(message_path / "content"), exist_ok=True)
  56 
  57             with open(str(message_path / 'content.data'), 'wb') as content_file:
  58                 # don't use raw_content here as this is served with a default content type
  59                 if message:
  60                     content_file.write(message.content)
  61                 else:
  62                     content_file.write(b'No content.')
  63 
  64             # content_view
  65             t = time.time()
  66             if message:
  67                 description, lines, error = contentviews.get_message_content_view(
  68                     'Auto', message, f
  69                 )
  70             else:
  71                 description, lines = 'No content.', []
  72             if time.time() - t > 0.1:
  73                 ctx.log(
  74                     "Slow content view: {} took {}s".format(
  75                         description.strip(),
  76                         round(time.time() - t, 1)
  77                     ),
  78                     "info"
  79                 )
  80             with open(str(message_path / "content" / "Auto.json"), "w") as content_view_file:
  81                 json.dump(
  82                     dict(lines=list(lines), description=description),
  83                     content_view_file
  84                 )

  87 class StaticViewer:
  88     # TODO: make this a command at some point.
  89     def load(self, loader):
  90         loader.add_option(
  91             "web_static_viewer", typing.Optional[str], "",
  92             "The path to output a static viewer."
  93         )
  94 
  95     def configure(self, updated):
  96         if "web_static_viewer" in updated and ctx.options.web_static_viewer:
  97             flows = io.read_flows_from_paths([ctx.options.rfile])
  98             p = pathlib.Path(ctx.options.web_static_viewer).expanduser()
  99             self.export(p, flows)
 100 
 101     def export(self, path: pathlib.Path, flows: typing.Iterable[flow.Flow]) -> None:
 102         save_static(path)
 103         save_filter_help(path)
 104         save_flows(path, flows)
 105         save_flows_content(path, flows)
