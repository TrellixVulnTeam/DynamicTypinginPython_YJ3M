 104     def run(self):  # pragma: no cover
 105         AsyncIOMainLoop().install()
 106         iol = tornado.ioloop.IOLoop.instance()
 107         http_server = tornado.httpserver.HTTPServer(self.app)
 108         http_server.listen(self.options.web_port, self.options.web_host)
 109         web_url = "http://{}:{}/".format(self.options.web_host, self.options.web_port)
