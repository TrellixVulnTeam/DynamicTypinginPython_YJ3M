  26 def flow_to_json(flow: mitmproxy.flow.Flow) -> dict:
  27     """
  28     Remove flow message content and cert to save transmission space.
  29 
  30     Args:
  31         flow: The original flow.
  32     """
  33     f = {
  34         "id": flow.id,
  35         "intercepted": flow.intercepted,
  36         "client_conn": flow.client_conn.get_state(),
  37         "server_conn": flow.server_conn.get_state(),
  38         "type": flow.type,
  39         "modified": flow.modified(),
  40         "marked": flow.marked,
  41     }
  42     # .alpn_proto_negotiated is bytes, we need to decode that.
  43     for conn in "client_conn", "server_conn":
  44         if f[conn]["alpn_proto_negotiated"] is None:
  45             continue
  46         f[conn]["alpn_proto_negotiated"] = \
  47             f[conn]["alpn_proto_negotiated"].decode(errors="backslashreplace")
  48     # There are some bytes in here as well, let's skip it until we have them in the UI.
  49     f["client_conn"].pop("tls_extensions", None)
  50     if flow.error:
  51         f["error"] = flow.error.get_state()
  52 
  53     if isinstance(flow, http.HTTPFlow):
  54         content_length: Optional[int]
  55         content_hash: Optional[str]
  56         if flow.request:
  57             if flow.request.raw_content:
  58                 content_length = len(flow.request.raw_content)
  59                 content_hash = hashlib.sha256(flow.request.raw_content).hexdigest()
  60             else:
  61                 content_length = None
  62                 content_hash = None
  63             f["request"] = {
  64                 "method": flow.request.method,
  65                 "scheme": flow.request.scheme,
  66                 "host": flow.request.host,
  67                 "port": flow.request.port,
  68                 "path": flow.request.path,
  69                 "http_version": flow.request.http_version,
  70                 "headers": tuple(flow.request.headers.items(True)),
  71                 "contentLength": content_length,
  72                 "contentHash": content_hash,
  73                 "timestamp_start": flow.request.timestamp_start,
  74                 "timestamp_end": flow.request.timestamp_end,
  75                 "is_replay": flow.request.is_replay,
  76                 "pretty_host": flow.request.pretty_host,
  77             }
  78         if flow.response:
  79             if flow.response.raw_content:
  80                 content_length = len(flow.response.raw_content)
  81                 content_hash = hashlib.sha256(flow.response.raw_content).hexdigest()
  82             else:
  83                 content_length = None
  84                 content_hash = None
  85             f["response"] = {
  86                 "http_version": flow.response.http_version,
  87                 "status_code": flow.response.status_code,
  88                 "reason": flow.response.reason,
  89                 "headers": tuple(flow.response.headers.items(True)),
  90                 "contentLength": content_length,
  91                 "contentHash": content_hash,
  92                 "timestamp_start": flow.response.timestamp_start,
  93                 "timestamp_end": flow.response.timestamp_end,
  94                 "is_replay": flow.response.is_replay,
  95             }
  96             if flow.response.data.trailers:
  97                 f["response"]["trailers"] = tuple(flow.response.data.trailers.items(True))
  98 
  99     f.get("server_conn", {}).pop("cert", None)
 100     f.get("client_conn", {}).pop("mitmcert", None)
 101 
 102     return f

 117 class RequestHandler(tornado.web.RequestHandler):
 118     application: "Application"
 119 
 120     def write(self, chunk):
 121         # Writing arrays on the top level is ok nowadays.
 122         # http://flask.pocoo.org/docs/0.11/security/#json-security
 123         if isinstance(chunk, list):
 124             chunk = tornado.escape.json_encode(chunk)
 125             self.set_header("Content-Type", "application/json; charset=UTF-8")
 126         super(RequestHandler, self).write(chunk)
 127 
 128     def set_default_headers(self):
 129         super().set_default_headers()
 130         self.set_header("Server", version.MITMPROXY)
 131         self.set_header("X-Frame-Options", "DENY")
 132         self.add_header("X-XSS-Protection", "1; mode=block")
 133         self.add_header("X-Content-Type-Options", "nosniff")
 134         self.add_header(
 135             "Content-Security-Policy",
 136             "default-src 'self'; "
 137             "connect-src 'self' ws:; "
 138             "style-src   'self' 'unsafe-inline'"
 139         )
 140 
 141     @property
 142     def json(self):
 143         if not self.request.headers.get("Content-Type", "").startswith("application/json"):
 144             raise APIError(400, "Invalid Content-Type, expected application/json.")
 145         try:
 146             return json.loads(self.request.body.decode())
 147         except Exception as e:
 148             raise APIError(400, "Malformed JSON: {}".format(str(e)))
 149 
 150     @property
 151     def filecontents(self):
 152         """
 153         Accept either a multipart/form file upload or just take the plain request body.
 154 
 155         """
 156         if self.request.files:
 157             return next(iter(self.request.files.values()))[0].body
 158         else:
 159             return self.request.body
 160 
 161     @property
 162     def view(self) -> "mitmproxy.addons.view.View":
 163         return self.application.master.view
 164 
 165     @property
 166     def master(self) -> "mitmproxy.tools.web.master.WebMaster":
 167         return self.application.master
 168 
 169     @property
 170     def flow(self) -> mitmproxy.flow.Flow:
 171         flow_id = str(self.path_kwargs["flow_id"])
 172         # FIXME: Add a facility to addon.view to safely access the store
 173         flow = self.view.get_by_id(flow_id)
 174         if flow:
 175             return flow
 176         else:
 177             raise APIError(404, "Flow not found.")
 178 
 179     def write_error(self, status_code: int, **kwargs):
 180         if "exc_info" in kwargs and isinstance(kwargs["exc_info"][1], APIError):
 181             self.finish(kwargs["exc_info"][1].log_message)
 182         else:
 183             super().write_error(status_code, **kwargs)

 210     @classmethod
 211     def broadcast(cls, **kwargs):
 212         message = json.dumps(kwargs, ensure_ascii=False).encode("utf8", "surrogateescape")
 213 
 214         for conn in cls.connections:
 215             try:
 216                 conn.write_message(message)

 285 class FlowHandler(RequestHandler):
 286     def delete(self, flow_id):
 287         if self.flow.killable:
 288             self.flow.kill()
 289         self.view.remove([self.flow])
 290 
 291     def put(self, flow_id):
 292         flow = self.flow
 293         flow.backup()
 294         try:
 295             for a, b in self.json.items():
 296                 if a == "request" and hasattr(flow, "request"):
 297                     request = flow.request
 298                     for k, v in b.items():
 299                         if k in ["method", "scheme", "host", "path", "http_version"]:
 300                             setattr(request, k, str(v))
 301                         elif k == "port":
 302                             request.port = int(v)
 303                         elif k == "headers":
 304                             request.headers.clear()
 305                             for header in v:
 306                                 request.headers.add(*header)
 307                         elif k == "content":
 308                             request.text = v
 309                         else:
 310                             raise APIError(400, "Unknown update request.{}: {}".format(k, v))
 311 
 312                 elif a == "response" and hasattr(flow, "response"):
 313                     response = flow.response
 314                     for k, v in b.items():
 315                         if k in ["msg", "http_version"]:
 316                             setattr(response, k, str(v))
 317                         elif k == "code":
 318                             response.status_code = int(v)
 319                         elif k == "headers":
 320                             response.headers.clear()
 321                             for header in v:
 322                                 response.headers.add(*header)
 323                         elif k == "content":
 324                             response.text = v
 325                         else:
 326                             raise APIError(400, "Unknown update response.{}: {}".format(k, v))
 327                 else:
 328                     raise APIError(400, "Unknown update {}: {}".format(a, b))
 329         except APIError:
 330             flow.revert()
 331             raise
 332         self.view.update([flow])

 361 class FlowContent(RequestHandler):
 362     def post(self, flow_id, message):
 363         self.flow.backup()
 364         message = getattr(self.flow, message)
 365         message.content = self.filecontents
 366         self.view.update([self.flow])
 367 
 368     def get(self, flow_id, message):
 369         message = getattr(self.flow, message)
 370 
 371         if not message.raw_content:
 372             raise APIError(400, "No content.")
 373 
 374         content_encoding = message.headers.get("Content-Encoding", None)
 375         if content_encoding:
 376             content_encoding = re.sub(r"[^\w]", "", content_encoding)
 377             self.set_header("Content-Encoding", content_encoding)
 378 
 379         original_cd = message.headers.get("Content-Disposition", None)
 380         filename = None
 381         if original_cd:
 382             filename = re.search(r'filename=([-\w" .()]+)', original_cd)
 383             if filename:
 384                 filename = filename.group(1)
 385         if not filename:
 386             filename = self.flow.request.path.split("?")[0].split("/")[-1]
 387 
 388         filename = re.sub(r'[^-\w" .()]', "", filename)
 389         cd = "attachment; filename={}".format(filename)
 390         self.set_header("Content-Disposition", cd)
 391         self.set_header("Content-Type", "application/text")
 392         self.set_header("X-Content-Type-Options", "nosniff")
 393         self.set_header("X-Frame-Options", "DENY")
 394         self.write(message.raw_content)

 418 class Settings(RequestHandler):
 419     def get(self):
 420         self.write(dict(
 421             version=version.VERSION,
 422             mode=str(self.master.options.mode),
 423             intercept_active=self.master.options.intercept_active,
 424             intercept=self.master.options.intercept,
 425             showhost=self.master.options.showhost,
 426             upstream_cert=self.master.options.upstream_cert,
 427             rawtcp=self.master.options.rawtcp,
 428             http2=self.master.options.http2,
 429             websocket=self.master.options.websocket,
 430             anticache=self.master.options.anticache,
 431             anticomp=self.master.options.anticomp,
 432             stickyauth=self.master.options.stickyauth,
 433             stickycookie=self.master.options.stickycookie,
 434             stream=self.master.options.stream_large_bodies,
 435             contentViews=[v.name.replace(' ', '_') for v in contentviews.views],
 436             listen_host=self.master.options.listen_host,
 437             listen_port=self.master.options.listen_port,
 438             server=self.master.options.server,
 439         ))
 440 
 441     def put(self):
 442         update = self.json
 443         allowed_options = {
 444             "intercept", "showhost", "upstream_cert", "ssl_insecure",
 445             "rawtcp", "http2", "websocket", "anticache", "anticomp",
 446             "stickycookie", "stickyauth", "stream_large_bodies"
 447         }
 448         for k in update:
 449             if k not in allowed_options:
 450                 raise APIError(400, "Unknown setting {}".format(k))
 451         self.master.options.update(**update)
