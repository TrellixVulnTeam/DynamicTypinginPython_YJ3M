 143 class _PathType(_BaseType):
 144     typ = Path
 145     display = "path"
 146 
 147     def completion(self, manager: "CommandManager", t: type, start: str) -> typing.Sequence[str]:
 148         if not start:
 149             start = "./"
 150         path = os.path.expanduser(start)
 151         ret = []
 152         if os.path.isdir(path):
 153             files = glob.glob(os.path.join(path, "*"))
 154             prefix = start
 155         else:
 156             files = glob.glob(path + "*")
 157             prefix = os.path.dirname(start)
 158         prefix = prefix or "./"
 159         for f in files:
 160             display = os.path.join(prefix, os.path.normpath(os.path.basename(f)))
 161             if os.path.isdir(f):
 162                 display += "/"
 163             ret.append(display)
 164         if not ret:
 165             ret = [start]
 166         ret.sort()
 167         return ret
 168 
 169     def parse(self, manager: "CommandManager", t: type, s: str) -> str:
 170         return os.path.expanduser(s)
 171 
 172     def is_valid(self, manager: "CommandManager", typ: typing.Any, val: typing.Any) -> bool:
 173         return isinstance(val, str)

 281     def is_valid(self, manager: "CommandManager", typ: typing.Any, val: typing.Any) -> bool:
 282         if not isinstance(val, str):
 283             return False
 284         parts = [x.strip() for x in val.split(",")]
 285         for p in parts:
 286             for pref in self.valid_prefixes:
 287                 if p.startswith(pref):
 288                     break
 289             else:
 290                 return False
 291         return True

 324     typ = flow.Flow
 325     display = "flow"

 327     def parse(self, manager: "CommandManager", t: type, s: str) -> flow.Flow:
 328         try:
 329             flows = manager.execute("view.flows.resolve %s" % (s))
 330         except exceptions.CommandError as e:
 331             raise exceptions.TypeError(str(e)) from e
 332         if len(flows) != 1:
 333             raise exceptions.TypeError(
 334                 "Command requires one flow, specification matched %s." % len(flows)
 335             )
 336         return flows[0]

 409 class TypeManager:
 410     def __init__(self, *types):
 411         self.typemap = {}
 412         for t in types:
 413             self.typemap[t.typ] = t()
 414 
 415     def get(self, t: typing.Optional[typing.Type], default=None) -> typing.Optional[_BaseType]:
 416         if type(t) in self.typemap:
 417             return self.typemap[type(t)]
 418         return self.typemap.get(t, default)
