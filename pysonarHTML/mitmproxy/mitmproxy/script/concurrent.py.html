   2 This module provides a @concurrent decorator primitive to
   3 offload computations from mitmproxy's main master thread.
   4 """

   6 from mitmproxy import eventsequence
   7 from mitmproxy.coretypes import basethread

   9 
  10 class ScriptThread(basethread.BaseThread):
  11     name = "ScriptThread"
  12 
  13 
  14 def concurrent(fn):
  15     if fn.__name__ not in eventsequence.Events - {"load", "configure"}:
  16         raise NotImplementedError(
  17             "Concurrent decorator not supported for '%s' method." % fn.__name__
  18         )
  19 
  20     def _concurrent(*args):
  21         # When annotating classmethods, "self" is passed as the first argument.
  22         # To support both class and static methods, we accept a variable number of arguments
  23         # and take the last one as our actual hook object.
  24         obj = args[-1]
  25 
  26         def run():
  27             fn(*args)
  28             if obj.reply.state == "taken":
  29                 if not obj.reply.has_message:
  30                     obj.reply.ack()
  31                 obj.reply.commit()
  32         obj.reply.take()
  33         ScriptThread(
  34             "script.concurrent (%s)" % fn.__name__,
  35             target=run
  36         ).start()
  37 
  38     return _concurrent
