   7 class _MultiDict(MutableMapping, metaclass=ABCMeta):
   8     def __repr__(self):
   9         fields = (
  10             repr(field)
  11             for field in self.fields
  12         )
  13         return "{cls}[{fields}]".format(
  14             cls=type(self).__name__,
  15             fields=", ".join(fields)
  16         )
  17 
  18     @staticmethod
  19     @abstractmethod
  20     def _reduce_values(values):
  21         """
  22         If a user accesses multidict["foo"], this method
  23         reduces all values for "foo" to a single value that is returned.
  24         For example, HTTP headers are folded, whereas we will just take
  25         the first cookie we found with that name.
  26         """
  27 
  28     @staticmethod
  29     @abstractmethod
  30     def _kconv(key):
  31         """
  32         This method converts a key to its canonical representation.
  33         For example, HTTP headers are case-insensitive, so this method returns key.lower().
  34         """
  35 
  36     def __getitem__(self, key):
  37         values = self.get_all(key)
  38         if not values:
  39             raise KeyError(key)
  40         return self._reduce_values(values)
  41 
  42     def __setitem__(self, key, value):
  43         self.set_all(key, [value])
  44 
  45     def __delitem__(self, key):
  46         if key not in self:
  47             raise KeyError(key)
  48         key = self._kconv(key)
  49         self.fields = tuple(
  50             field for field in self.fields
  51             if key != self._kconv(field[0])
  52         )
  53 
  54     def __iter__(self):
  55         seen = set()
  56         for key, _ in self.fields:
  57             key_kconv = self._kconv(key)
  58             if key_kconv not in seen:
  59                 seen.add(key_kconv)
  60                 yield key
  61 
  62     def __len__(self):
  63         return len(set(self._kconv(key) for key, _ in self.fields))
  64 
  65     def __eq__(self, other):
  66         if isinstance(other, MultiDict):
  67             return self.fields == other.fields
  68         return False
  69 
  70     def get_all(self, key):
  71         """
  72         Return the list of all values for a given key.
  73         If that key is not in the MultiDict, the return value will be an empty list.
  74         """
  75         key = self._kconv(key)
  76         return [
  77             value
  78             for k, value in self.fields
  79             if self._kconv(k) == key
  80         ]
  81 
  82     def set_all(self, key, values):
  83         """
  84         Remove the old values for a key and add new ones.
  85         """
  86         key_kconv = self._kconv(key)
  87 
  88         new_fields = []
  89         for field in self.fields:
  90             if self._kconv(field[0]) == key_kconv:
  91                 if values:
  92                     new_fields.append(
  93                         (field[0], values.pop(0))
  94                     )
  95             else:
  96                 new_fields.append(field)
  97         while values:
  98             new_fields.append(
  99                 (key, values.pop(0))
 100             )
 101         self.fields = tuple(new_fields)
 102 
 103     def add(self, key, value):
 104         """
 105         Add an additional value for the given key at the bottom.
 106         """
 107         self.insert(len(self.fields), key, value)
 108 
 109     def insert(self, index, key, value):
 110         """
 111         Insert an additional value for the given key at the specified position.
 112         """
 113         item = (key, value)
 114         self.fields = self.fields[:index] + (item,) + self.fields[index:]

 116     def keys(self, multi=False):
 117         """
 118         Get all keys.
 119 
 120         Args:
 121             multi(bool):
 122                 If True, one key per value will be returned.
 123                 If False, duplicate keys will only be returned once.
 124         """
 125         return (
 126             k
 127             for k, _ in self.items(multi)
 128         )
 129 
 130     def values(self, multi=False):
 131         """
 132         Get all values.
 133 
 134         Args:
 135             multi(bool):
 136                 If True, all values will be returned.
 137                 If False, only the first value per key will be returned.
 138         """
 139         return (
 140             v
 141             for _, v in self.items(multi)
 142         )
 143 
 144     def items(self, multi=False):
 145         """
 146         Get all (key, value) tuples.
 147 
 148         Args:
 149             multi(bool):
 150                 If True, all (key, value) pairs will be returned
 151                 If False, only the first (key, value) pair per unique key will be returned.
 152         """
 153         if multi:
 154             return self.fields
 155         else:
 156             return super().items()

 159 class MultiDict(_MultiDict, serializable.Serializable):
 160     def __init__(self, fields=()):
 161         super().__init__()
 162         self.fields = tuple(
 163             tuple(i) for i in fields
 164         )
 165 
 166     @staticmethod
 167     def _reduce_values(values):
 168         return values[0]
 169 
 170     @staticmethod
 171     def _kconv(key):
 172         return key
 173 
 174     def get_state(self):
 175         return self.fields
 176 
 177     def set_state(self, state):
 178         self.fields = tuple(tuple(x) for x in state)
 179 
 180     @classmethod
 181     def from_state(cls, state):
 182         return cls(state)

 185 class MultiDictView(_MultiDict):
 186     """
 187     The MultiDictView provides the MultiDict interface over calculated data.
 188     The view itself contains no state - data is retrieved from the parent on
 189     request, and stored back to the parent on change.
 190     """
 191     def __init__(self, getter, setter):
 192         self._getter = getter
 193         self._setter = setter
 194         super().__init__()
 195 
 196     @staticmethod
 197     def _kconv(key):
 198         # All request-attributes are case-sensitive.
 199         return key
 200 
 201     @staticmethod
 202     def _reduce_values(values):
 203         # We just return the first element if
 204         # multiple elements exist with the same key.
 205         return values[0]
 206 
 207     @property
 208     def fields(self):
 209         return self._getter()
 210 
 211     @fields.setter
 212     def fields(self, value):
 213         self._setter(value)
 214 
 215     def copy(self):
 216         return MultiDict(self.fields)
