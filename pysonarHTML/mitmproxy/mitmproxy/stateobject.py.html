   8 class StateObject(serializable.Serializable):
   9     """
  10     An object with serializable state.
  11 
  12     State attributes can either be serializable types(str, tuple, bool, ...)
  13     or StateObject instances themselves.
  14     """
  15 
  16     _stateobject_attributes: typing.ClassVar[typing.MutableMapping[str, typing.Any]]
  17     """
  18     An attribute-name -> class-or-type dict containing all attributes that
  19     should be serialized. If the attribute is a class, it must implement the
  20     Serializable protocol.
  21     """
  22 
  23     def get_state(self):
  24         """
  25         Retrieve object state.
  26         """
  27         state = {}
  28         for attr, cls in self._stateobject_attributes.items():
  29             val = getattr(self, attr)
  30             state[attr] = get_state(cls, val)
  31         return state
  32 
  33     def set_state(self, state):
  34         """
  35         Load object state from data returned by a get_state call.
  36         """
  37         state = state.copy()
  38         for attr, cls in self._stateobject_attributes.items():
  39             val = state.pop(attr)
  40             if val is None:
  41                 setattr(self, attr, val)
  42             else:
  43                 curr = getattr(self, attr, None)
  44                 if hasattr(curr, "set_state"):
  45                     curr.set_state(val)
  46                 else:
  47                     setattr(self, attr, make_object(cls, val))
  48         if state:
  49             raise RuntimeWarning("Unexpected State in __setstate__: {}".format(state))

  52 def _process(typeinfo: typecheck.Type, val: typing.Any, make: bool) -> typing.Any:
  53     if val is None:
  54         return None
  55     elif make and hasattr(typeinfo, "from_state"):
  56         return typeinfo.from_state(val)
  57     elif not make and hasattr(val, "get_state"):
  58         return val.get_state()
  59 
  60     typename = str(typeinfo)
  61 
  62     if typename.startswith("typing.List"):
  63         T = typecheck.sequence_type(typeinfo)
  64         return [_process(T, x, make) for x in val]
  65     elif typename.startswith("typing.Tuple"):
  66         Ts = typecheck.tuple_types(typeinfo)
  67         if len(Ts) != len(val):
  68             raise ValueError("Invalid data. Expected {}, got {}.".format(Ts, val))
  69         return tuple(
  70             _process(T, x, make) for T, x in zip(Ts, val)
  71         )
  72     elif typename.startswith("typing.Dict"):
  73         k_cls, v_cls = typecheck.mapping_types(typeinfo)
  74         return {
  75             _process(k_cls, k, make): _process(v_cls, v, make)
  76             for k, v in val.items()
  77         }
  78     elif typename.startswith("typing.Any"):
  79         # This requires a bit of explanation. We can't import our IO layer here,
  80         # because it causes a circular import. Rather than restructuring the
  81         # code for this, we use JSON serialization, which has similar primitive
  82         # type restrictions as tnetstring, to check for conformance.
  83         try:
  84             json.dumps(val)
  85         except TypeError:
  86             raise ValueError(f"Data not serializable: {val}")
  87         return val
  88     else:
  89         return typeinfo(val)
  90 
  91 
  92 def make_object(typeinfo: typecheck.Type, val: typing.Any) -> typing.Any:
  93     """Create an object based on the state given in val."""
  94     return _process(typeinfo, val, True)

  97 def get_state(typeinfo: typecheck.Type, val: typing.Any) -> typing.Any:
  98     """Get the state of the object given as val."""
  99     return _process(typeinfo, val, False)
