   7 def always_bytes(str_or_bytes: Union[str, bytes, None], *encode_args) -> Optional[bytes]:
   8     if isinstance(str_or_bytes, bytes) or str_or_bytes is None:
   9         return cast(Optional[bytes], str_or_bytes)
  10     elif isinstance(str_or_bytes, str):
  11         return str_or_bytes.encode(*encode_args)
  12     else:
  13         raise TypeError("Expected str or bytes, but got {}.".format(type(str_or_bytes).__name__))

  17     """
  18     Returns,

  31 # Translate control characters to "safe" characters. This implementation
  32 # initially replaced them with the matching control pictures
  33 # (http://unicode.org/charts/PDF/U2400.pdf), but that turned out to render badly
  34 # with monospace fonts. We are back to "." therefore.
  35 _control_char_trans = {
  36     x: ord(".")  # x + 0x2400 for unicode control group pictures
  37     for x in range(32)
  38 }
  39 _control_char_trans[127] = ord(".")  # 0x2421
  40 _control_char_trans_newline = _control_char_trans.copy()
  41 for x in ("\r", "\n", "\t"):
  42     del _control_char_trans_newline[ord(x)]
  43 
  44 _control_char_trans = str.maketrans(_control_char_trans)
  45 _control_char_trans_newline = str.maketrans(_control_char_trans_newline)

  48 def escape_control_characters(text: str, keep_spacing=True) -> str:
  49     """
  50     Replace all unicode C1 control characters from the given text with a single "."
  51 
  52     Args:
  53         keep_spacing: If True, tabs and newlines will not be replaced.
  54     """
  55     if not isinstance(text, str):
  56         raise ValueError("text type must be unicode but is {}".format(type(text).__name__))
  57 
  58     trans = _control_char_trans_newline if keep_spacing else _control_char_trans
  59     return text.translate(trans)

  62 def bytes_to_escaped_str(data, keep_spacing=False, escape_single_quotes=False):
  63     """
  64     Take bytes and return a safe string that can be displayed to the user.
  65 
  66     Single quotes are always escaped, double quotes are never escaped:
  67         "'" + bytes_to_escaped_str(...) + "'"
  68     gives a valid Python string.
  69 
  70     Args:
  71         keep_spacing: If True, tabs and newlines will not be escaped.
  72     """
  73 
  74     if not isinstance(data, bytes):
  75         raise ValueError("data must be bytes, but is {}".format(data.__class__.__name__))
  76     # We always insert a double-quote here so that we get a single-quoted string back
  77     # https://stackoverflow.com/questions/29019340/why-does-python-use-different-quotes-for-representing-strings-depending-on-their
  78     ret = repr(b'"' + data).lstrip("b")[2:-1]
  79     if not escape_single_quotes:
  80         ret = re.sub(r"(?<!\\)(\\\\)*\\'", lambda m: (m.group(1) or "") + "'", ret)
  81     if keep_spacing:
  82         ret = re.sub(
  83             r"(?<!\\)(\\\\)*\\([nrt])",
  84             lambda m: (m.group(1) or "") + dict(n="\n", r="\r", t="\t")[m.group(2)],
  85             ret
  86         )
  87     return ret

 105 def is_mostly_bin(s: bytes) -> bool:
 106     if not s or len(s) == 0:
 107         return False
 108 
 109     return sum(
 110         i < 9 or 13 < i < 32 or 126 < i
 111         for i in s[:100]

 119 def clean_hanging_newline(t):
 120     """
 121         Many editors will silently add a newline to the final line of a
 122         document (I'm looking at you, Vim). This function fixes this common
 123         problem at the risk of removing a hanging newline in the rare cases
 124         where the user actually intends it.
 125     """
 126     if t and t[-1] == "\n":
 127         return t[:-1]
 128     return t

 131 def hexdump(s):
 132     """
 133         Returns:
 134             A generator of (offset, hex, str) tuples
 135     """
 136     for i in range(0, len(s), 16):
 137         offset = "{:0=10x}".format(i)
 138         part = s[i:i + 16]
 139         x = " ".join("{:0=2x}".format(i) for i in part)
 140         x = x.ljust(47)  # 16*2 + 15
 141         part_repr = always_str(escape_control_characters(
 142             part.decode("ascii", "replace").replace(u"\ufffd", u"."),
 143             False
 144         ))
 145         yield (offset, x, part_repr)

 162 def split_special_areas(
 163         data: str,
 164         area_delimiter: Iterable[str],
 165 ):
 166     """
 167     Split a string of code into a [code, special area, code, special area, ..., code] list.
 168 
 169     For example,
 170 
 171     >>> split_special_areas(
 172     >>>     "test /* don't modify me */ foo",
 173     >>>     [r"/\\*[\\s\\S]*?\\*/"])  # (regex matching comments)
 174     ["test ", "/* don't modify me */", " foo"]
 175 
 176     "".join(split_special_areas(x, ...)) == x always holds true.
 177     """
 178     return re.split(
 179         "({})".format("|".join(area_delimiter)),
 180         data,
 181         flags=re.MULTILINE
 182     )

 185 def escape_special_areas(
 186         data: str,
 187         area_delimiter: Iterable[str],
 188         control_characters,
 189 ):
 190     """
 191     Escape all control characters present in special areas with UTF8 symbols
 192     in the private use plane (U+E000 t+ ord(char)).
 193     This is useful so that one can then use regex replacements on the resulting string without
 194     interfering with special areas.
 195 
 196     control_characters must be 0 < ord(x) < 256.
 197 
 198     Example:
 199 
 200     >>> print(x)
 201     if (true) { console.log('{}'); }
 202     >>> x = escape_special_areas(x, "{", ["'" + SINGLELINE_CONTENT + "'"])
 203     >>> print(x)
 204     if (true) { console.log('�}'); }
 205     >>> x = re.sub(r"\\s*{\\s*", " {\n    ", x)
 206     >>> x = unescape_special_areas(x)
 207     >>> print(x)
 208     if (true) {
 209         console.log('{}'); }
 210     """
 211     buf = io.StringIO()
 212     parts = split_special_areas(data, area_delimiter)
 213     rex = re.compile(r"[{}]".format(control_characters))
 214     for i, x in enumerate(parts):
 215         if i % 2:
 216             x = rex.sub(_move_to_private_code_plane, x)
 217         buf.write(x)
 218     return buf.getvalue()

 221 def unescape_special_areas(data: str):
 222     """
 223     Invert escape_special_areas.
 224 
 225     x == unescape_special_areas(escape_special_areas(x)) always holds true.
 226     """
 227     return re.sub(r"[\ue000-\ue0ff]", _restore_from_private_code_plane, data)
