 109 def check():
 110     args = sys.argv[1:]
 111     print()
 112     if "-U" in args:
 113         print("-U is deprecated, please use --mode upstream:SPEC instead")
 114 
 115     if "-T" in args:
 116         print("-T is deprecated, please use --mode transparent instead")
 117 
 118     for option in ("-e", "--eventlog", "--norefresh"):
 119         if option in args:
 120             print("{} has been removed.".format(option))
 121 
 122     for option in ("--nonanonymous", "--singleuser", "--htpasswd"):
 123         if option in args:
 124             print(
 125                 '{} is deprecated.\n'
 126                 'Please use `--proxyauth SPEC` instead.\n'
 127                 'SPEC Format: "username:pass", "any" to accept any user/pass combination,\n'
 128                 '"@path" to use an Apache htpasswd file, or\n'
 129                 '"ldap[s]:url_server_ldap:dn_auth:password:dn_subtree" '
 130                 'for LDAP authentication.'.format(option))
 131 
 132     for option in REPLACED.splitlines():
 133         if option in args:
 134             if isinstance(REPLACEMENTS.get(option), list):
 135                 new_options = REPLACEMENTS.get(option)
 136             else:
 137                 new_options = [REPLACEMENTS.get(option)]
 138             print(
 139                 "{} is deprecated.\n"
 140                 "Please use `{}` instead.".format(
 141                     option,
 142                     "` or `".join(new_options)
 143                 )
 144             )
 145 
 146     for option in DEPRECATED.splitlines():
 147         if option in args:
 148             print(
 149                 "{} is deprecated.\n"
 150                 "Please use `--set {}=value` instead.\n"
 151                 "To show all options and their default values use --options".format(
 152                     option,
 153                     REPLACEMENTS.get(option, None) or option.lstrip("-").replace("-", "_")
 154                 )
 155             )
