  39 def create_ca(organization, cn, exp, key_size):
  40     key = OpenSSL.crypto.PKey()
  41     key.generate_key(OpenSSL.crypto.TYPE_RSA, key_size)
  42     cert = OpenSSL.crypto.X509()
  43     cert.set_serial_number(int(time.time() * 10000))
  44     cert.set_version(2)
  45     cert.get_subject().CN = cn
  46     cert.get_subject().O = organization
  47     cert.gmtime_adj_notBefore(-3600 * 48)
  48     cert.gmtime_adj_notAfter(exp)
  49     cert.set_issuer(cert.get_subject())
  50     cert.set_pubkey(key)
  51     cert.add_extensions([
  52         OpenSSL.crypto.X509Extension(
  53             b"basicConstraints",
  54             True,
  55             b"CA:TRUE"
  56         ),
  57         OpenSSL.crypto.X509Extension(
  58             b"nsCertType",
  59             False,
  60             b"sslCA"
  61         ),
  62         OpenSSL.crypto.X509Extension(
  63             b"extendedKeyUsage",
  64             False,
  65             b"serverAuth,clientAuth,emailProtection,timeStamping,msCodeInd,msCodeCom,msCTLSign,msSGC,msEFS,nsSGC"
  66         ),
  67         OpenSSL.crypto.X509Extension(
  68             b"keyUsage",
  69             True,
  70             b"keyCertSign, cRLSign"
  71         ),
  72         OpenSSL.crypto.X509Extension(
  73             b"subjectKeyIdentifier",
  74             False,
  75             b"hash",
  76             subject=cert
  77         ),
  78     ])
  79     cert.sign(key, "sha256")
  80     return key, cert

  83 def dummy_cert(privkey, cacert, commonname, sans, organization):
  84     """
  85         Generates a dummy certificate.
  86 
  87         privkey: CA private key
  88         cacert: CA certificate
  89         commonname: Common name for the generated certificate.
  90         sans: A list of Subject Alternate Names.
  91         organization: Organization name for the generated certificate.
  92 
  93         Returns cert if operation succeeded, None if not.
  94     """
  95     ss = []
  96     for i in sans:
  97         try:
  98             ipaddress.ip_address(i.decode("ascii"))
  99         except ValueError:
 100             ss.append(b"DNS:%s" % i)
 101         else:
 102             ss.append(b"IP:%s" % i)
 103     ss = b", ".join(ss)
 104 
 105     cert = OpenSSL.crypto.X509()
 106     cert.gmtime_adj_notBefore(-3600 * 48)
 107     cert.gmtime_adj_notAfter(DEFAULT_EXP_DUMMY_CERT)
 108     cert.set_issuer(cacert.get_subject())
 109     is_valid_commonname = (
 110         commonname is not None and len(commonname) < 64
 111     )
 112     if is_valid_commonname:
 113         cert.get_subject().CN = commonname
 114     if organization is not None:
 115         cert.get_subject().O = organization
 116     cert.set_serial_number(int(time.time() * 10000))
 117     if ss:
 118         cert.set_version(2)
 119         cert.add_extensions(
 120             [OpenSSL.crypto.X509Extension(
 121                 b"subjectAltName",
 122                 # RFC 5280 §4.2.1.6: subjectAltName is critical if subject is empty.
 123                 not is_valid_commonname,
 124                 ss
 125             )]
 126         )
 127     cert.add_extensions([
 128         OpenSSL.crypto.X509Extension(
 129             b"extendedKeyUsage",
 130             False,
 131             b"serverAuth,clientAuth"
 132         )
 133     ])
 134     cert.set_pubkey(cacert.get_pubkey())
 135     cert.sign(privkey, "sha256")
 136     return Cert(cert)

 154     """
 155         Implements an in-memory certificate store.
 156     """
 157     STORE_CAP = 100
 158 
 159     def __init__(
 160             self,
 161             default_privatekey,
 162             default_ca,
 163             default_chain_file,
 164             dhparams):
 165         self.default_privatekey = default_privatekey
 166         self.default_ca = default_ca
 167         self.default_chain_file = default_chain_file
 168         self.dhparams = dhparams
 169         self.certs: typing.Dict[TCertId, CertStoreEntry] = {}
 170         self.expire_queue = []
 171 
 172     def expire(self, entry):
 173         self.expire_queue.append(entry)
 174         if len(self.expire_queue) > self.STORE_CAP:
 175             d = self.expire_queue.pop(0)
 176             self.certs = {k: v for k, v in self.certs.items() if v != d}

 178     @staticmethod
 179     def load_dhparam(path):
 180 
 181         # mitmproxy<=0.10 doesn't generate a dhparam file.
 182         # Create it now if necessary.
 183         if not os.path.exists(path):
 184             with open(path, "wb") as f:
 185                 f.write(DEFAULT_DHPARAM)
 186 
 187         bio = OpenSSL.SSL._lib.BIO_new_file(path.encode(sys.getfilesystemencoding()), b"r")
 188         if bio != OpenSSL.SSL._ffi.NULL:
 189             bio = OpenSSL.SSL._ffi.gc(bio, OpenSSL.SSL._lib.BIO_free)
 190             dh = OpenSSL.SSL._lib.PEM_read_bio_DHparams(
 191                 bio,
 192                 OpenSSL.SSL._ffi.NULL,
 193                 OpenSSL.SSL._ffi.NULL,
 194                 OpenSSL.SSL._ffi.NULL)
 195             dh = OpenSSL.SSL._ffi.gc(dh, OpenSSL.SSL._lib.DH_free)
 196             return dh

 198     @classmethod
 199     def from_store(cls, path, basename, key_size):
 200         ca_path = os.path.join(path, basename + "-ca.pem")
 201         if not os.path.exists(ca_path):
 202             key, ca = cls.create_store(path, basename, key_size)
 203         else:
 204             with open(ca_path, "rb") as f:
 205                 raw = f.read()
 206             ca = OpenSSL.crypto.load_certificate(
 207                 OpenSSL.crypto.FILETYPE_PEM,
 208                 raw)
 209             key = OpenSSL.crypto.load_privatekey(
 210                 OpenSSL.crypto.FILETYPE_PEM,
 211                 raw)
 212         dh_path = os.path.join(path, basename + "-dhparam.pem")
 213         dh = cls.load_dhparam(dh_path)
 214         return cls(key, ca, ca_path, dh)

 216     @staticmethod
 217     @contextlib.contextmanager
 218     def umask_secret():
 219         """
 220             Context to temporarily set umask to its original value bitor 0o77.
 221             Useful when writing private keys to disk so that only the owner
 222             will be able to read them.
 223         """
 224         original_umask = os.umask(0)
 225         os.umask(original_umask | 0o77)
 226         try:
 227             yield
 228         finally:
 229             os.umask(original_umask)
 230 
 231     @staticmethod
 232     def create_store(path, basename, key_size, organization=None, cn=None, expiry=DEFAULT_EXP):
 233         if not os.path.exists(path):
 234             os.makedirs(path)
 235 
 236         organization = organization or basename
 237         cn = cn or basename
 238 
 239         key, ca = create_ca(organization=organization, cn=cn, exp=expiry, key_size=key_size)
 240         # Dump the CA plus private key
 241         with CertStore.umask_secret(), open(os.path.join(path, basename + "-ca.pem"), "wb") as f:
 242             f.write(
 243                 OpenSSL.crypto.dump_privatekey(
 244                     OpenSSL.crypto.FILETYPE_PEM,
 245                     key))
 246             f.write(
 247                 OpenSSL.crypto.dump_certificate(
 248                     OpenSSL.crypto.FILETYPE_PEM,
 249                     ca))
 250 
 251         # Dump the certificate in PEM format
 252         with open(os.path.join(path, basename + "-ca-cert.pem"), "wb") as f:
 253             f.write(
 254                 OpenSSL.crypto.dump_certificate(
 255                     OpenSSL.crypto.FILETYPE_PEM,
 256                     ca))
 257 
 258         # Create a .cer file with the same contents for Android
 259         with open(os.path.join(path, basename + "-ca-cert.cer"), "wb") as f:
 260             f.write(
 261                 OpenSSL.crypto.dump_certificate(
 262                     OpenSSL.crypto.FILETYPE_PEM,
 263                     ca))
 264 
 265         # Dump the certificate in PKCS12 format for Windows devices
 266         with open(os.path.join(path, basename + "-ca-cert.p12"), "wb") as f:
 267             p12 = OpenSSL.crypto.PKCS12()
 268             p12.set_certificate(ca)
 269             f.write(p12.export())
 270 
 271         # Dump the certificate and key in a PKCS12 format for Windows devices
 272         with CertStore.umask_secret(), open(os.path.join(path, basename + "-ca.p12"), "wb") as f:
 273             p12 = OpenSSL.crypto.PKCS12()
 274             p12.set_certificate(ca)
 275             p12.set_privatekey(key)
 276             f.write(p12.export())
 277 
 278         with open(os.path.join(path, basename + "-dhparam.pem"), "wb") as f:
 279             f.write(DEFAULT_DHPARAM)
 280 
 281         return key, ca

 283     def add_cert_file(self, spec: str, path: str) -> None:
 284         with open(path, "rb") as f:
 285             raw = f.read()
 286         cert = Cert(
 287             OpenSSL.crypto.load_certificate(
 288                 OpenSSL.crypto.FILETYPE_PEM,
 289                 raw))
 290         try:
 291             privatekey = OpenSSL.crypto.load_privatekey(
 292                 OpenSSL.crypto.FILETYPE_PEM,
 293                 raw)
 294         except Exception:
 295             privatekey = self.default_privatekey
 296         self.add_cert(
 297             CertStoreEntry(cert, privatekey, path),
 298             spec.encode("idna")
 299         )

 301     def add_cert(self, entry: CertStoreEntry, *names: bytes):
 302         """
 303             Adds a cert to the certstore. We register the CN in the cert plus
 304             any SANs, and also the list of names provided as an argument.
 305         """
 306         if entry.cert.cn:
 307             self.certs[entry.cert.cn] = entry
 308         for i in entry.cert.altnames:
 309             self.certs[i] = entry
 310         for i in names:
 311             self.certs[i] = entry

 313     @staticmethod
 314     def asterisk_forms(dn: bytes) -> typing.List[bytes]:
 315         """
 316         Return all asterisk forms for a domain. For example, for www.example.com this will return
 317         [b"www.example.com", b"*.example.com", b"*.com"]. The single wildcard "*" is omitted.
 318         """
 319         parts = dn.split(b".")
 320         ret = [dn]
 321         for i in range(1, len(parts)):
 322             ret.append(b"*." + b".".join(parts[i:]))
 323         return ret

 325     def get_cert(
 326             self,
 327             commonname: typing.Optional[bytes],
 328             sans: typing.List[bytes],
 329             organization: typing.Optional[bytes] = None
 330     ) -> typing.Tuple["Cert", OpenSSL.SSL.PKey, str]:
 331         """
 332             Returns an (cert, privkey, cert_chain) tuple.
 333 
 334             commonname: Common name for the generated certificate. Must be a
 335             valid, plain-ASCII, IDNA-encoded domain name.
 336 
 337             sans: A list of Subject Alternate Names.
 338 
 339             organization: Organization name for the generated certificate.
 340         """
 341 
 342         potential_keys: typing.List[TCertId] = []
 343         if commonname:
 344             potential_keys.extend(self.asterisk_forms(commonname))
 345         for s in sans:
 346             potential_keys.extend(self.asterisk_forms(s))
 347         potential_keys.append(b"*")
 348         potential_keys.append((commonname, tuple(sans)))
 349 
 350         name = next(
 351             filter(lambda key: key in self.certs, potential_keys),
 352             None
 353         )
 354         if name:
 355             entry = self.certs[name]
 356         else:
 357             entry = CertStoreEntry(
 358                 cert=dummy_cert(
 359                     self.default_privatekey,
 360                     self.default_ca,
 361                     commonname,
 362                     sans,
 363                     organization),
 364                 privatekey=self.default_privatekey,
 365                 chain_file=self.default_chain_file)
 366             self.certs[(commonname, tuple(sans))] = entry
 367             self.expire(entry)
 368 
 369         return entry.cert, entry.privatekey, entry.chain_file

 372 class _GeneralName(univ.Choice):

 483     @property
 484     def altnames(self):
 485         """
 486         Returns:
 487             All DNS altnames.
 488         """
 489         # tcp.TCPClient.convert_to_tls assumes that this property only contains DNS altnames for hostname verification.
 490         altnames = []
 491         for i in range(self.x509.get_extension_count()):
 492             ext = self.x509.get_extension(i)
 493             if ext.get_short_name() == b"subjectAltName":
 494                 try:
 495                     dec = decode(ext.get_data(), asn1Spec=_GeneralNames())
 496                 except PyAsn1Error:
 497                     continue
 498                 for i in dec[0]:
 499                     if i[0].hasValue():
 500                         e = i[0].asOctets()
 501                         altnames.append(e)
 502 
 503         return altnames
