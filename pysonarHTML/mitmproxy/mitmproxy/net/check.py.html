   4 # Allow underscore in host name
   5 # Note: This could be a DNS label, a hostname, a FQDN, or an IP
   6 _label_valid = re.compile(br"[A-Z\d\-_]{1,63}$", re.IGNORECASE)

   9 def is_valid_host(host: bytes) -> bool:
  10     """
  11     Checks if the passed bytes are a valid DNS hostname or an IPv4/IPv6 address.
  12     """
  13     try:
  14         host.decode("idna")
  15     except ValueError:
  16         return False
  17     # RFC1035: 255 bytes or less.
  18     if len(host) > 255:
  19         return False
  20     if host and host[-1:] == b".":
  21         host = host[:-1]
  22     # DNS hostname
  23     if all(_label_valid.match(x) for x in host.split(b".")):
  24         return True
  25     # IPv4/IPv6 address
  26     try:
  27         ipaddress.ip_address(host.decode('idna'))
  28         return True
  29     except ValueError:
  30         return False
