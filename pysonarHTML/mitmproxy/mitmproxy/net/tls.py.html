  80 class MasterSecretLogger:
  81     def __init__(self, filename):
  82         self.filename = filename
  83         self.f = None
  84         self.lock = threading.Lock()
  85 
  86     # required for functools.wraps, which pyOpenSSL uses.
  87     __name__ = "MasterSecretLogger"
  88 
  89     def __call__(self, connection, where, ret):
  90         done_now = (
  91             where == SSL.SSL_CB_HANDSHAKE_DONE and ret == 1
  92         )
  93         # this is a horrendous workaround for https://github.com/mitmproxy/mitmproxy/pull/3692#issuecomment-608454530:
  94         # OpenSSL 1.1.1f decided to not make connection.master_key() fail in the SSL_CB_HANDSHAKE_DONE callback.
  95         # To support various OpenSSL versions and still log master secrets, we now mark connections where this has
  96         # happened and then try again on the next event. This is ugly and shouldn't be done, but eventually we
  97         # replace this with context.set_keylog_callback anyways.
  98         done_previously_but_not_logged_yet = (
  99             hasattr(connection, "_still_needs_masterkey")
 100         )
 101         if done_now or done_previously_but_not_logged_yet:
 102             with self.lock:
 103                 if not self.f:
 104                     d = os.path.dirname(self.filename)
 105                     if not os.path.isdir(d):
 106                         os.makedirs(d)
 107                     self.f = open(self.filename, "ab")
 108                     self.f.write(b"\r\n")
 109                 try:
 110                     client_random = binascii.hexlify(connection.client_random())
 111                     masterkey = binascii.hexlify(connection.master_key())
 112                 except (AssertionError, SSL.Error):  # careful: exception type changes between pyOpenSSL versions
 113                     connection._still_needs_masterkey = True
 114                 else:
 115                     self.f.write(b"CLIENT_RANDOM %s %s\r\n" % (client_random, masterkey))
 116                     self.f.flush()
 117                     if hasattr(connection, "_still_needs_masterkey"):
 118                         delattr(connection, "_still_needs_masterkey")
 119 
 120     def close(self):
 121         with self.lock:
 122             if self.f:
 123                 self.f.close()
 124 
 125     @staticmethod
 126     def create_logfun(filename):
 127         if filename:
 128             return MasterSecretLogger(filename)
 129         return None

 137 def _create_ssl_context(
 138         method: int = DEFAULT_METHOD,
 139         options: int = DEFAULT_OPTIONS,
 140         ca_path: str = None,
 141         ca_pemfile: str = None,
 142         cipher_list: str = None,
 143         alpn_protos: typing.Iterable[bytes] = None,
 144         alpn_select=None,
 145         alpn_select_callback: typing.Callable[[typing.Any, typing.Any], bytes] = None,
 146         verify: int = SSL.VERIFY_PEER,
 147         verify_callback: typing.Optional[
 148             typing.Callable[[SSL.Connection, SSL.X509, int, int, bool], bool]
 149         ] = None,
 150 ) -> SSL.Context:
 151     """
 152     Creates an SSL Context.
 153 
 154     :param method: One of SSLv2_METHOD, SSLv3_METHOD, SSLv23_METHOD, TLSv1_METHOD, TLSv1_1_METHOD, or TLSv1_2_METHOD
 155     :param options: A bit field consisting of OpenSSL.SSL.OP_* values
 156     :param verify: A bit field consisting of OpenSSL.SSL.VERIFY_* values
 157     :param ca_path: Path to a directory of trusted CA certificates prepared using the c_rehash tool
 158     :param ca_pemfile: Path to a PEM formatted trusted CA certificate
 159     :param cipher_list: A textual OpenSSL cipher list, see https://www.openssl.org/docs/apps/ciphers.html
 160     :rtype : SSL.Context
 161     """
 162     try:
 163         context = SSL.Context(method)
 164     except ValueError:
 165         method_name = METHOD_NAMES.get(method, "unknown")
 166         raise exceptions.TlsException(
 167             "SSL method \"%s\" is most likely not supported "
 168             "or disabled (for security reasons) in your libssl. "
 169             "Please refer to https://github.com/mitmproxy/mitmproxy/issues/1101 "
 170             "for more details." % method_name
 171         )
 172 
 173     # Options (NO_SSLv2/3)
 174     if options is not None:
 175         context.set_options(options)
 176 
 177     # Verify Options (NONE/PEER and trusted CAs)
 178     if verify is not None:
 179         context.set_verify(verify, verify_callback)
 180         if ca_path is None and ca_pemfile is None:
 181             ca_pemfile = certifi.where()
 182         try:
 183             context.load_verify_locations(ca_pemfile, ca_path)
 184         except SSL.Error:
 185             raise exceptions.TlsException(
 186                 "Cannot load trusted certificates ({}, {}).".format(
 187                     ca_pemfile, ca_path
 188                 )
 189             )
 190 
 191     # Workaround for
 192     # https://github.com/pyca/pyopenssl/issues/190
 193     # https://github.com/mitmproxy/mitmproxy/issues/472
 194     # Options already set before are not cleared.
 195     context.set_mode(SSL._lib.SSL_MODE_AUTO_RETRY)
 196 
 197     # Cipher List
 198     if cipher_list:
 199         try:
 200             context.set_cipher_list(cipher_list.encode())
 201         except SSL.Error as v:
 202             raise exceptions.TlsException("SSL cipher specification error: %s" % str(v))
 203 
 204     # SSLKEYLOGFILE
 205     if log_master_secret:
 206         context.set_info_callback(log_master_secret)
 207 
 208     if alpn_protos is not None:
 209         # advertise application layer protocols
 210         context.set_alpn_protos(alpn_protos)
 211     elif alpn_select is not None and alpn_select_callback is None:
 212         # select application layer protocol
 213         def alpn_select_callback(conn_, options):
 214             if alpn_select in options:
 215                 return bytes(alpn_select)
 216             else:  # pragma: no cover
 217                 return options[0]
 218 
 219         context.set_alpn_select_callback(alpn_select_callback)
 220     elif alpn_select_callback is not None and alpn_select is None:
 221         if not callable(alpn_select_callback):
 222             raise exceptions.TlsException("ALPN error: alpn_select_callback must be a function.")
 223         context.set_alpn_select_callback(alpn_select_callback)
 224     elif alpn_select_callback is not None and alpn_select is not None:
 225         raise exceptions.TlsException(
 226             "ALPN error: only define alpn_select (string) OR alpn_select_callback (function).")
 227 
 228     return context

 231 def create_client_context(
 232         cert: str = None,
 233         sni: str = None,
 234         address: str = None,
 235         verify: int = SSL.VERIFY_NONE,
 236         **sslctx_kwargs
 237 ) -> SSL.Context:
 238     """
 239     Args:
 240         cert: Path to a file containing both client cert and private key.
 241         sni: Server Name Indication. Required for VERIFY_PEER
 242         address: server address, used for expressive error messages only
 243         verify: A bit field consisting of OpenSSL.SSL.VERIFY_* values
 244     """
 245 
 246     if sni is None and verify != SSL.VERIFY_NONE:
 247         raise exceptions.TlsException("Cannot validate certificate hostname without SNI")
 248 
 249     def verify_callback(
 250             conn: SSL.Connection,
 251             x509: SSL.X509,
 252             errno: int,
 253             depth: int,
 254             is_cert_verified: bool
 255     ) -> bool:
 256         if is_cert_verified and depth == 0 and not sni:
 257             conn.cert_error = exceptions.InvalidCertificateException(
 258                 f"Certificate verification error for {address}: Cannot validate hostname, SNI missing."
 259             )
 260             is_cert_verified = False
 261         elif is_cert_verified:
 262             pass
 263         else:
 264             conn.cert_error = exceptions.InvalidCertificateException(
 265                 "Certificate verification error for {}: {} (errno: {}, depth: {})".format(
 266                     sni,
 267                     SSL._ffi.string(SSL._lib.X509_verify_cert_error_string(errno)).decode(),
 268                     errno,
 269                     depth
 270                 )
 271             )
 272 
 273         # SSL_VERIFY_NONE: The handshake will be continued regardless of the verification result.
 274         return is_cert_verified
 275 
 276     context = _create_ssl_context(
 277         verify=verify,
 278         verify_callback=verify_callback,
 279         **sslctx_kwargs,
 280     )
 281 
 282     if sni:
 283         # Manually enable hostname verification on the context object.
 284         # https://wiki.openssl.org/index.php/Hostname_validation
 285         param = SSL._lib.SSL_CTX_get0_param(context._context)
 286         # Matching on the CN is disabled in both Chrome and Firefox, so we disable it, too.
 287         # https://www.chromestatus.com/feature/4981025180483584
 288         SSL._lib.X509_VERIFY_PARAM_set_hostflags(
 289             param,
 290             SSL._lib.X509_CHECK_FLAG_NO_PARTIAL_WILDCARDS | SSL._lib.X509_CHECK_FLAG_NEVER_CHECK_SUBJECT
 291         )
 292         SSL._openssl_assert(
 293             SSL._lib.X509_VERIFY_PARAM_set1_host(param, sni.encode("idna"), 0) == 1
 294         )
 295 
 296     # Client Certs
 297     if cert:
 298         try:
 299             context.use_privatekey_file(cert)
 300             context.use_certificate_chain_file(cert)
 301         except SSL.Error as v:
 302             raise exceptions.TlsException("SSL client certificate error: %s" % str(v))
 303     return context

 317 def create_server_context(
 318         cert: typing.Union[certs.Cert, str],
 319         key: SSL.PKey,
 320         handle_sni: typing.Optional[typing.Callable[[SSL.Connection], None]] = None,
 321         request_client_cert: bool = False,
 322         chain_file=None,
 323         dhparams=None,
 324         extra_chain_certs: typing.Iterable[certs.Cert] = None,
 325         **sslctx_kwargs
 326 ) -> SSL.Context:
 327     """
 328         cert: A certs.Cert object or the path to a certificate
 329         chain file.
 330 
 331         handle_sni: SNI handler, should take a connection object. Server
 332         name can be retrieved like this:
 333 
 334                 connection.get_servername()
 335 
 336         The request_client_cert argument requires some explanation. We're
 337         supposed to be able to do this with no negative effects - if the
 338         client has no cert to present, we're notified and proceed as usual.
 339         Unfortunately, Android seems to have a bug (tested on 4.2.2) - when
 340         an Android client is asked to present a certificate it does not
 341         have, it hangs up, which is frankly bogus. Some time down the track
 342         we may be able to make the proper behaviour the default again, but
 343         until then we're conservative.
 344     """
 345 
 346     if request_client_cert:
 347         verify = SSL.VERIFY_PEER
 348     else:
 349         verify = SSL.VERIFY_NONE
 350 
 351     context = _create_ssl_context(
 352         ca_pemfile=chain_file,
 353         verify=verify,
 354         verify_callback=accept_all,
 355         **sslctx_kwargs,
 356     )
 357 
 358     context.use_privatekey(key)
 359     if isinstance(cert, certs.Cert):
 360         context.use_certificate(cert.x509)
 361     else:
 362         context.use_certificate_chain_file(cert)
 363 
 364     if extra_chain_certs:
 365         for i in extra_chain_certs:
 366             context.add_extra_chain_cert(i.x509)
 367 
 368     if handle_sni:
 369         # SNI callback happens during do_handshake()
 370         context.set_tlsext_servername_callback(handle_sni)
 371 
 372     if dhparams:
 373         SSL._lib.SSL_CTX_set_tmp_dh(context._context, dhparams)
 374 
 375     return context

 378 def is_tls_record_magic(d):
 379     """
 380     Returns:
 381         True, if the passed bytes start with the TLS record magic bytes.
 382         False, otherwise.
 383     """
 384     d = d[:3]
 385 
 386     # TLS ClientHello magic, works for SSLv3, TLSv1.0, TLSv1.1, TLSv1.2
 387     # http://www.moserware.com/2009/06/first-few-milliseconds-of-https.html#client-hello
 388     return (
 389         len(d) == 3 and
 390         d[0] == 0x16 and
 391         d[1] == 0x03 and
 392         0x0 <= d[2] <= 0x03
 393     )

 396 def get_client_hello(rfile):
 397     """
 398     Peek into the socket and read all records that contain the initial client hello message.
 399 
 400     client_conn:
 401         The :py:class:`client connection <mitmproxy.connections.ClientConnection>`.
 402 
 403     Returns:
 404         The raw handshake packet bytes, without TLS record header(s).
 405     """
 406     client_hello = b""
 407     client_hello_size = 1
 408     offset = 0
 409     while len(client_hello) < client_hello_size:
 410         record_header = rfile.peek(offset + 5)[offset:]
 411         if not is_tls_record_magic(record_header) or len(record_header) < 5:
 412             raise exceptions.TlsProtocolException(
 413                 'Expected TLS record, got "%s" instead.' % record_header)
 414         record_size = struct.unpack_from("!H", record_header, 3)[0] + 5
 415         record_body = rfile.peek(offset + record_size)[offset + 5:]
 416         if len(record_body) != record_size - 5:
 417             raise exceptions.TlsProtocolException(
 418                 "Unexpected EOF in TLS handshake: %s" % record_body)
 419         client_hello += record_body
 420         offset += record_size
 421         client_hello_size = struct.unpack("!I", b'\x00' + client_hello[1:4])[0] + 4
 422     return client_hello

 425 class ClientHello:
 426 
 427     def __init__(self, raw_client_hello):
 428         self._client_hello = tls_client_hello.TlsClientHello(
 429             KaitaiStream(io.BytesIO(raw_client_hello))
 430         )
 431 
 432     @property
 433     def cipher_suites(self):
 434         return self._client_hello.cipher_suites.cipher_suites
 435 
 436     @property
 437     def sni(self) -> typing.Optional[bytes]:
 438         if self._client_hello.extensions:
 439             for extension in self._client_hello.extensions.extensions:
 440                 is_valid_sni_extension = (
 441                     extension.type == 0x00 and
 442                     len(extension.body.server_names) == 1 and
 443                     extension.body.server_names[0].name_type == 0 and
 444                     check.is_valid_host(extension.body.server_names[0].host_name)
 445                 )
 446                 if is_valid_sni_extension:
 447                     return extension.body.server_names[0].host_name
 448         return None
 449 
 450     @property
 451     def alpn_protocols(self):
 452         if self._client_hello.extensions:
 453             for extension in self._client_hello.extensions.extensions:
 454                 if extension.type == 0x10:
 455                     return list(x.name for x in extension.body.alpn_protocols)
 456         return []
 457 
 458     @property
 459     def extensions(self) -> typing.List[typing.Tuple[int, bytes]]:
 460         ret = []
 461         if self._client_hello.extensions:
 462             for extension in self._client_hello.extensions.extensions:
 463                 body = getattr(extension, "_raw_body", extension.body)
 464                 ret.append((extension.type, body))
 465         return ret
 466 
 467     @classmethod
 468     def from_file(cls, client_conn) -> "ClientHello":
 469         """
 470         Peek into the connection, read the initial client hello and parse it to obtain ALPN values.
 471         client_conn:
 472             The :py:class:`client connection <mitmproxy.connections.ClientConnection>`.
 473         Returns:
 474             :py:class:`client hello <mitmproxy.net.tls.ClientHello>`.
 475         """
 476         try:
 477             raw_client_hello = get_client_hello(client_conn)[4:]  # exclude handshake header.
 478         except exceptions.ProtocolException as e:
 479             raise exceptions.TlsProtocolException('Cannot read raw Client Hello: %s' % repr(e))
 480 
 481         try:
 482             return cls(raw_client_hello)
 483         except EOFError as e:
 484             raise exceptions.TlsProtocolException(
 485                 f"Cannot parse Client Hello: {e!r}, Raw Client Hello: {binascii.hexlify(raw_client_hello)!r}"
 486             )
 487 
 488     def __repr__(self):
 489         return f"ClientHello(sni: {self.sni}, alpn_protocols: {self.alpn_protocols})"
