   4 def assemble_request(request):

  32 def assemble_body(headers, body_chunks):
  33     if "chunked" in headers.get("transfer-encoding", "").lower():
  34         for chunk in body_chunks:
  35             if chunk:
  36                 yield b"%x\r\n%s\r\n" % (len(chunk), chunk)
  37         yield b"0\r\n\r\n"
  38     else:
  39         for chunk in body_chunks:
  40             yield chunk

  43 def _assemble_request_line(request_data):
  44     """
  45     Args:
  46         request_data (mitmproxy.net.http.request.RequestData)
  47     """
  48     form = request_data.first_line_format
  49     if form == "relative":
  50         return b"%s %s %s" % (
  51             request_data.method,
  52             request_data.path,
  53             request_data.http_version
  54         )
  55     elif form == "authority":
  56         return b"%s %s:%d %s" % (
  57             request_data.method,
  58             request_data.host,
  59             request_data.port,
  60             request_data.http_version
  61         )
  62     elif form == "absolute":
  63         return b"%s %s://%s:%d%s %s" % (
  64             request_data.method,
  65             request_data.scheme,
  66             request_data.host,
  67             request_data.port,
  68             request_data.path,
  69             request_data.http_version
  70         )
  71     else:
  72         raise RuntimeError("Invalid request form")
