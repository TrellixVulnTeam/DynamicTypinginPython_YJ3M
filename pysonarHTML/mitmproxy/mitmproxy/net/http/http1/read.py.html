  15 def get_header_tokens(headers, key):
  16     """
  17         Retrieve all tokens for a header key. A number of different headers
  18         follow a pattern where each header line can containe comma-separated
  19         tokens, and headers can be set multiple times.
  20     """
  21     if key not in headers:
  22         return []
  23     tokens = headers[key].split(",")
  24     return [token.strip() for token in tokens]

  35 def read_request_head(rfile):
  36     """
  37     Parse an HTTP request head (request line + headers) from an input stream
  38 
  39     Args:
  40         rfile: The input stream
  41 
  42     Returns:
  43         The HTTP request object (without body)
  44 
  45     Raises:
  46         exceptions.HttpReadDisconnect: No bytes can be read from rfile.
  47         exceptions.HttpSyntaxException: The input is malformed HTTP.
  48         exceptions.HttpException: Any other error occurred.
  49     """
  50     timestamp_start = time.time()
  51     if hasattr(rfile, "reset_timestamps"):
  52         rfile.reset_timestamps()
  53 
  54     form, method, scheme, host, port, path, http_version = _read_request_line(rfile)
  55     headers = _read_headers(rfile)
  56 
  57     if hasattr(rfile, "first_byte_timestamp"):
  58         # more accurate timestamp_start
  59         timestamp_start = rfile.first_byte_timestamp
  60 
  61     return request.Request(
  62         form, method, scheme, host, port, path, http_version, headers, None, None, timestamp_start
  63     )

 104 def read_body(rfile, expected_size, limit=None, max_chunk_size=4096):
 105     """
 106         Read an HTTP message body
 107 
 108         Args:
 109             rfile: The input stream
 110             expected_size: The expected body size (see :py:meth:`expected_body_size`)
 111             limit: Maximum body size
 112             max_chunk_size: Maximium chunk size that gets yielded
 113 
 114         Returns:
 115             A generator that yields byte chunks of the content.
 116 
 117         Raises:
 118             exceptions.HttpException, if an error occurs
 119 
 120         Caveats:
 121             max_chunk_size is not considered if the transfer encoding is chunked.
 122     """
 123     if not limit or limit < 0:
 124         limit = sys.maxsize
 125     if not max_chunk_size:
 126         max_chunk_size = limit
 127 
 128     if expected_size is None:
 129         for x in _read_chunked(rfile, limit):
 130             yield x
 131     elif expected_size >= 0:
 132         if limit is not None and expected_size > limit:
 133             raise exceptions.HttpException(
 134                 "HTTP Body too large. "
 135                 "Limit is {}, content length was advertised as {}".format(limit, expected_size)
 136             )
 137         bytes_left = expected_size
 138         while bytes_left:
 139             chunk_size = min(bytes_left, max_chunk_size)
 140             content = rfile.read(chunk_size)
 141             if len(content) < chunk_size:
 142                 raise exceptions.HttpException("Unexpected EOF")
 143             yield content
 144             bytes_left -= chunk_size
 145     else:
 146         bytes_left = limit
 147         while bytes_left:
 148             chunk_size = min(bytes_left, max_chunk_size)
 149             content = rfile.read(chunk_size)
 150             if not content:
 151                 return
 152             yield content
 153             bytes_left -= chunk_size
 154         not_done = rfile.read(1)
 155         if not_done:
 156             raise exceptions.HttpException("HTTP body too large. Limit is {}.".format(limit))

 159 def connection_close(http_version, headers):
 160     """
 161         Checks the message to see if the client connection should be closed
 162         according to RFC 2616 Section 8.1.
 163         If we don't have a Connection header, HTTP 1.1 connections are assumed
 164         to be persistent.
 165     """
 166     if "connection" in headers:
 167         tokens = get_header_tokens(headers, "connection")
 168         if "close" in tokens:
 169             return True
 170         elif "keep-alive" in tokens:
 171             return False
 172 
 173     return http_version != "HTTP/1.1" and http_version != b"HTTP/1.1"

 176 def expected_http_body_size(
 177         request: request.Request,
 178         response: typing.Optional[response.Response] = None,
 179         expect_continue_as_0: bool = True
 180 ):
 181     """
 182         Args:
 183             - expect_continue_as_0: If true, incorrectly predict a body size of 0 for requests which are waiting
 184               for a 100 Continue response.
 185         Returns:
 186             The expected body length:
 187             - a positive integer, if the size is known in advance
 188             - None, if the size in unknown in advance (chunked encoding)
 189             - -1, if all data should be read until end of stream.
 190 
 191         Raises:
 192             exceptions.HttpSyntaxException, if the content length header is invalid
 193     """
 194     # Determine response size according to
 195     # http://tools.ietf.org/html/rfc7230#section-3.3
 196     if not response:
 197         headers = request.headers
 198         if expect_continue_as_0 and headers.get("expect", "").lower() == "100-continue":
 199             return 0
 200     else:
 201         headers = response.headers
 202         if request.method.upper() == "HEAD":
 203             return 0
 204         if 100 <= response.status_code <= 199:
 205             return 0
 206         if response.status_code == 200 and request.method.upper() == "CONNECT":
 207             return 0
 208         if response.status_code in (204, 304):
 209             return 0
 210 
 211     if "chunked" in headers.get("transfer-encoding", "").lower():
 212         return None
 213     if "content-length" in headers:
 214         try:
 215             sizes = headers.get_all("content-length")
 216             different_content_length_headers = any(x != sizes[0] for x in sizes)
 217             if different_content_length_headers:
 218                 raise exceptions.HttpSyntaxException("Conflicting Content Length Headers")
 219             size = int(sizes[0])
 220             if size < 0:
 221                 raise ValueError()
 222             return size
 223         except ValueError as e:
 224             raise exceptions.HttpSyntaxException("Unparseable Content Length") from e
 225     if not response:
 226         return 0
 227     return -1

 230 def _get_first_line(rfile):
 231     try:
 232         line = rfile.readline()
 233         if line == b"\r\n" or line == b"\n":
 234             # Possible leftover from previous message
 235             line = rfile.readline()
 236     except (exceptions.TcpDisconnect, exceptions.TlsException):
 237         raise exceptions.HttpReadDisconnect("Remote disconnected")
 238     if not line:
 239         raise exceptions.HttpReadDisconnect("Remote disconnected")
 240     return line.strip()

 243 def _read_request_line(rfile):
 244     try:
 245         line = _get_first_line(rfile)
 246     except exceptions.HttpReadDisconnect:
 247         # We want to provide a better error message.
 248         raise exceptions.HttpReadDisconnect("Client disconnected")
 249 
 250     try:
 251         method, path, http_version = line.split()
 252 
 253         if path == b"*" or path.startswith(b"/"):
 254             form = "relative"
 255             scheme, host, port = None, None, None
 256         elif method == b"CONNECT":
 257             form = "authority"
 258             host, port = _parse_authority_form(path)
 259             scheme, path = None, None
 260         else:
 261             form = "absolute"
 262             scheme, host, port, path = url.parse(path)
 263 
 264         _check_http_version(http_version)
 265     except ValueError:
 266         raise exceptions.HttpSyntaxException("Bad HTTP request line: {}".format(line))
 267 
 268     return form, method, scheme, host, port, path, http_version

 271 def _parse_authority_form(hostport):
 272     """
 273         Returns (host, port) if hostport is a valid authority-form host specification.
 274         http://tools.ietf.org/html/draft-luotonen-web-proxy-tunneling-01 section 3.1
 275 
 276         Raises:
 277             ValueError, if the input is malformed
 278     """
 279     try:
 280         host, port = hostport.rsplit(b":", 1)
 281         if host.startswith(b"[") and host.endswith(b"]"):
 282             host = host[1:-1]
 283         port = int(port)
 284         if not check.is_valid_host(host) or not check.is_valid_port(port):
 285             raise ValueError()
 286     except ValueError:
 287         raise exceptions.HttpSyntaxException("Invalid host specification: {}".format(hostport))
 288 
 289     return host, port

 292 def _read_response_line(rfile):
 293     try:
 294         line = _get_first_line(rfile)
 295     except exceptions.HttpReadDisconnect:
 296         # We want to provide a better error message.
 297         raise exceptions.HttpReadDisconnect("Server disconnected")
 298 
 299     try:
 300         parts = line.split(None, 2)
 301         if len(parts) == 2:  # handle missing message gracefully
 302             parts.append(b"")
 303 
 304         http_version, status_code, message = parts
 305         status_code = int(status_code)
 306         _check_http_version(http_version)
 307 
 308     except ValueError:
 309         raise exceptions.HttpSyntaxException("Bad HTTP response line: {}".format(line))
 310 
 311     return http_version, status_code, message

 319 def _read_headers(rfile):
 320     """
 321         Read a set of headers.
 322         Stop once a blank line is reached.
 323 
 324         Returns:
 325             A headers object
 326 
 327         Raises:
 328             exceptions.HttpSyntaxException
 329     """
 330     ret = []
 331     while True:
 332         line = rfile.readline()
 333         if not line or line == b"\r\n" or line == b"\n":
 334             # we do have coverage of this, but coverage.py does not detect it.
 335             break  # pragma: no cover
 336         if line[0] in b" \t":
 337             if not ret:
 338                 raise exceptions.HttpSyntaxException("Invalid headers")
 339             # continued header
 340             ret[-1] = (ret[-1][0], ret[-1][1] + b'\r\n ' + line.strip())
 341         else:
 342             try:
 343                 name, value = line.split(b":", 1)
 344                 value = value.strip()
 345                 if not name:
 346                     raise ValueError()
 347                 ret.append((name, value))
 348             except ValueError:
 349                 raise exceptions.HttpSyntaxException(
 350                     "Invalid header line: %s" % repr(line)
 351                 )
 352     return headers.Headers(ret)

 355 def _read_chunked(rfile, limit=sys.maxsize):
 356     """
 357     Read a HTTP body with chunked transfer encoding.
 358 
 359     Args:
 360         rfile: the input file
 361         limit: A positive integer
 362     """
 363     total = 0
 364     while True:
 365         line = rfile.readline(128)
 366         if line == b"":
 367             raise exceptions.HttpException("Connection closed prematurely")
 368         if line != b"\r\n" and line != b"\n":
 369             try:
 370                 length = int(line, 16)
 371             except ValueError:
 372                 raise exceptions.HttpSyntaxException("Invalid chunked encoding length: {}".format(line))
 373             total += length
 374             if total > limit:
 375                 raise exceptions.HttpException(
 376                     "HTTP Body too large. Limit is {}, "
 377                     "chunked content longer than {}".format(limit, total)
 378                 )
 379             chunk = rfile.read(length)
 380             suffix = rfile.readline(5)
 381             if suffix != b"\r\n":
 382                 raise exceptions.HttpSyntaxException("Malformed chunked body")
 383             if length == 0:
 384                 return
 385             yield chunk
