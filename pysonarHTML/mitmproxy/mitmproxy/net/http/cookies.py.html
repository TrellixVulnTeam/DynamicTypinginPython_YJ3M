   8 """
   9 A flexible module for cookie parsing and manipulation.
  10 
  11 This module differs from usual standards-compliant cookie modules in a number
  12 of ways. We try to be as permissive as possible, and to retain even mal-formed
  13 information. Duplicate cookies are preserved in parsing, and can be set in
  14 formatting. We do attempt to escape and quote values where needed, but will not
  15 reject data that violate the specs.
  16 
  17 Parsing accepts the formats in RFC6265 and partially RFC2109 and RFC2965. We
  18 also parse the comma-separated variant of Set-Cookie that allows multiple
  19 cookies to be set in a single header. Serialization follows RFC6265.
  20 
  21     http://tools.ietf.org/html/rfc6265
  22     http://tools.ietf.org/html/rfc2109
  23     http://tools.ietf.org/html/rfc2965
  24 """

  47 def _read_until(s, start, term):
  48     """
  49         Read until one of the characters in term is reached.
  50     """
  51     if start == len(s):
  52         return "", start + 1
  53     for i in range(start, len(s)):
  54         if s[i] in term:
  55             return s[start:i], i
  56     return s[start:i + 1], i + 1

  59 def _read_quoted_string(s, start):
  60     """
  61         start: offset to the first quote of the string to be read
  62 
  63         A sort of loose super-set of the various quoted string specifications.
  64 
  65         RFC6265 disallows backslashes or double quotes within quoted strings.
  66         Prior RFCs use backslashes to escape. This leaves us free to apply
  67         backslash escaping by default and be compatible with everything.
  68     """
  69     escaping = False
  70     ret = []
  71     # Skip the first quote
  72     i = start  # initialize in case the loop doesn't run.
  73     for i in range(start + 1, len(s)):
  74         if escaping:
  75             ret.append(s[i])
  76             escaping = False
  77         elif s[i] == '"':
  78             break
  79         elif s[i] == "\\":
  80             escaping = True
  81         else:
  82             ret.append(s[i])
  83     return "".join(ret), i + 1

  86 def _read_key(s, start, delims=";="):
  87     """
  88         Read a key - the LHS of a token/value pair in a cookie.
  89     """
  90     return _read_until(s, start, delims)
  91 
  92 
  93 def _read_value(s, start, delims):
  94     """
  95         Reads a value - the RHS of a token/value pair in a cookie.
  96     """
  97     if start >= len(s):
  98         return "", start
  99     elif s[start] == '"':
 100         return _read_quoted_string(s, start)
 101     else:
 102         return _read_until(s, start, delims)

 105 def _read_cookie_pairs(s, off=0):
 106     """
 107         Read pairs of lhs=rhs values from Cookie headers.
 108 
 109         off: start offset
 110     """
 111     pairs = []
 112 
 113     while True:
 114         lhs, off = _read_key(s, off)
 115         lhs = lhs.lstrip()
 116 
 117         rhs = ""
 118         if off < len(s) and s[off] == "=":
 119             rhs, off = _read_value(s, off + 1, ";")
 120         if rhs or lhs:
 121             pairs.append([lhs, rhs])
 122 
 123         off += 1
 124 
 125         if not off < len(s):
 126             break
 127 
 128     return pairs, off

 131 def _read_set_cookie_pairs(s: str, off=0) -> Tuple[List[TPairs], int]:
 132     """
 133         Read pairs of lhs=rhs values from SetCookie headers while handling multiple cookies.
 134 
 135         off: start offset
 136         specials: attributes that are treated specially
 137     """
 138     cookies: List[TPairs] = []
 139     pairs: TPairs = []
 140 
 141     while True:
 142         lhs, off = _read_key(s, off, ";=,")
 143         lhs = lhs.lstrip()
 144 
 145         rhs = ""
 146         if off < len(s) and s[off] == "=":
 147             rhs, off = _read_value(s, off + 1, ";,")
 148 
 149             # Special handling of attributes
 150             if lhs.lower() == "expires":
 151                 # 'expires' values can contain commas in them so they need to
 152                 # be handled separately.
 153 
 154                 # We actually bank on the fact that the expires value WILL
 155                 # contain a comma. Things will fail, if they don't.
 156 
 157                 # '3' is just a heuristic we use to determine whether we've
 158                 # only read a part of the expires value and we should read more.
 159                 if len(rhs) <= 3:
 160                     trail, off = _read_value(s, off + 1, ";,")
 161                     rhs = rhs + "," + trail
 162 
 163             # as long as there's a "=", we consider it a pair
 164             pairs.append([lhs, rhs])
 165 
 166         elif lhs:
 167             pairs.append([lhs, rhs])
 168 
 169         # comma marks the beginning of a new cookie
 170         if off < len(s) and s[off] == ",":
 171             cookies.append(pairs)
 172             pairs = []
 173 
 174         off += 1
 175 
 176         if not off < len(s):
 177             break
 178 
 179     if pairs or not cookies:
 180         cookies.append(pairs)
 181 
 182     return cookies, off

 185 def _has_special(s: str) -> bool:
 186     for i in s:
 187         if i in '",;\\':
 188             return True
 189         o = ord(i)
 190         if o < 0x21 or o > 0x7e:
 191             return True
 192     return False

 195 def _format_pairs(pairs, specials=(), sep="; "):
 196     """
 197         specials: A lower-cased list of keys that will not be quoted.
 198     """
 199     vals = []
 200     for k, v in pairs:
 201         if k.lower() not in specials and _has_special(v):
 202             v = ESCAPE.sub(r"\\\1", v)
 203             v = '"%s"' % v
 204         vals.append("%s=%s" % (k, v))
 205     return sep.join(vals)

 238 def parse_set_cookie_header(line: str) -> List[TSetCookie]:
 239     """
 240     Parse a Set-Cookie header value
 241 
 242     Returns:
 243         A list of (name, value, attrs) tuples, where attrs is a
 244         CookieAttrs dict of attributes. No attempt is made to parse attribute
 245         values - they are treated purely as strings.
 246     """
 247     cookie_pairs, off = _read_set_cookie_pairs(line)
 248     cookies = []
 249     for pairs in cookie_pairs:
 250         if pairs:
 251             cookie, *attrs = pairs
 252             cookies.append((
 253                 cookie[0],
 254                 cookie[1],
 255                 CookieAttrs(attrs)
 256             ))
 257     return cookies

 268 def format_set_cookie_header(set_cookies: List[TSetCookie]) -> str:
 269     """
 270         Formats a Set-Cookie header value.
 271     """
 272 
 273     rv = []
 274 
 275     for name, value, attrs in set_cookies:
 276 
 277         pairs = [(name, value)]
 278         pairs.extend(
 279             attrs.fields if hasattr(attrs, "fields") else attrs
 280         )
 281 
 282         rv.append(_format_set_cookie_pairs(pairs))
 283 
 284     return ", ".join(rv)

 287 def refresh_set_cookie_header(c: str, delta: int) -> str:
 288     """
 289     Args:
 290         c: A Set-Cookie string
 291         delta: Time delta in seconds
 292     Returns:
 293         A refreshed Set-Cookie string
 294     Raises:
 295         ValueError, if the cookie is invalid.
 296     """
 297     cookies = parse_set_cookie_header(c)
 298     for cookie in cookies:
 299         name, value, attrs = cookie
 300         if not name or not value:
 301             raise ValueError("Invalid Cookie")
 302 
 303         if "expires" in attrs:
 304             e = email.utils.parsedate_tz(attrs["expires"])
 305             if e:
 306                 f = email.utils.mktime_tz(e) + delta
 307                 attrs.set_all("expires", [email.utils.formatdate(f, usegmt=True)])
 308             else:
 309                 # This can happen when the expires tag is invalid.
 310                 # reddit.com sends a an expires tag like this: "Thu, 31 Dec
 311                 # 2037 23:59:59 GMT", which is valid RFC 1123, but not
 312                 # strictly correct according to the cookie spec. Browsers
 313                 # appear to parse this tolerantly - maybe we should too.
 314                 # For now, we just ignore this.
 315                 del attrs["expires"]
 316     return format_set_cookie_header(cookies)

 319 def get_expiration_ts(cookie_attrs):
 320     """
 321         Determines the time when the cookie will be expired.
 322 
 323         Considering both 'expires' and 'max-age' parameters.
 324 
 325         Returns: timestamp of when the cookie will expire.
 326                  None, if no expiration time is set.
 327     """
 328     if 'expires' in cookie_attrs:
 329         e = email.utils.parsedate_tz(cookie_attrs["expires"])
 330         if e:
 331             return email.utils.mktime_tz(e)
 332 
 333     elif 'max-age' in cookie_attrs:
 334         try:
 335             max_age = int(cookie_attrs['Max-Age'])
 336         except ValueError:
 337             pass
 338         else:
 339             now_ts = time.time()
 340             return now_ts + max_age
 341 
 342     return None

 345 def is_expired(cookie_attrs):
 346     """
 347         Determines whether a cookie has expired.
 348 
 349         Returns: boolean
 350     """
 351 
 352     exp_ts = get_expiration_ts(cookie_attrs)
 353     now_ts = time.time()
 354 
 355     # If no expiration information was provided with the cookie
 356     if exp_ts is None:
 357         return False
 358     else:
 359         return exp_ts <= now_ts

 362 def group_cookies(pairs):
 363     """
 364     Converts a list of pairs to a (name, value, attrs) for each cookie.
 365     """
 366 
 367     if not pairs:
 368         return []
 369 
 370     cookie_list = []
 371 
 372     # First pair is always a new cookie
 373     name, value = pairs[0]
 374     attrs = []
 375 
 376     for k, v in pairs[1:]:
 377         if k.lower() in _cookie_params:
 378             attrs.append((k, v))
 379         else:
 380             cookie_list.append((name, value, CookieAttrs(attrs)))
 381             name, value, attrs = k, v, []
 382 
 383     cookie_list.append((name, value, CookieAttrs(attrs)))
 384     return cookie_list
