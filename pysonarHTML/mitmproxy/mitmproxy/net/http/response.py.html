  60     def __repr__(self):
  61         if self.raw_content:
  62             details = "{}, {}".format(
  63                 self.headers.get("content-type", "unknown content type"),
  64                 human.pretty_size(len(self.raw_content))
  65             )
  66         else:
  67             details = "no content"
  68         return "Response({status_code} {reason}, {details})".format(
  69             status_code=self.status_code,
  70             reason=self.reason,
  71             details=details
  72         )

  74     @classmethod
  75     def make(
  76             cls,
  77             status_code: int=200,
  78             content: Union[bytes, str]=b"",
  79             headers: Union[Dict[str, AnyStr], Iterable[Tuple[bytes, bytes]]]=()
  80     ):
  81         """
  82         Simplified API for creating response objects.
  83         """
  84         resp = cls(
  85             b"HTTP/1.1",
  86             status_code,
  87             status_codes.RESPONSES.get(status_code, "").encode(),
  88             (),
  89             None
  90         )
  91 
  92         # Headers can be list or dict, we differentiate here.
  93         if isinstance(headers, dict):
  94             resp.headers = nheaders.Headers(**headers)
  95         elif isinstance(headers, Iterable):
  96             resp.headers = nheaders.Headers(headers)
  97         else:
  98             raise TypeError("Expected headers to be an iterable or dict, but is {}.".format(
  99                 type(headers).__name__
 100             ))
 101 
 102         # Assign this manually to update the content-length header.
 103         if isinstance(content, bytes):
 104             resp.content = content
 105         elif isinstance(content, str):
 106             resp.text = content
 107         else:
 108             raise TypeError("Expected content to be str or bytes, but is {}.".format(
 109                 type(content).__name__
 110             ))
 111 
 112         return resp

 125     @property
 126     def reason(self):
 127         """
 128         HTTP Reason Phrase, e.g. "Not Found".
 129         HTTP2 responses do not contain a reason phrase and self.data.reason will be :py:obj:`None`.
 130         When :py:obj:`None` return an empty reason phrase so that functions expecting a string work properly.
 131         """
 132         # Encoding: http://stackoverflow.com/a/16674906/934719
 133         if self.data.reason is not None:
 134             return self.data.reason.decode("ISO-8859-1", "surrogateescape")
 135         else:
 136             return ""
 137 
 138     @reason.setter
 139     def reason(self, reason):
 140         self.data.reason = strutils.always_bytes(reason, "ISO-8859-1", "surrogateescape")

 142     def _get_cookies(self):
 143         h = self.headers.get_all("set-cookie")
 144         all_cookies = cookies.parse_set_cookie_headers(h)
 145         return tuple(
 146             (name, (value, attrs))
 147             for name, value, attrs in all_cookies
 148         )
 149 
 150     def _set_cookies(self, value):
 151         cookie_headers = []
 152         for k, v in value:
 153             header = cookies.format_set_cookie_header([(k, v[0], v[1])])
 154             cookie_headers.append(header)
 155         self.headers.set_all("set-cookie", cookie_headers)
 156 
 157     @property
 158     def cookies(self) -> multidict.MultiDictView:
 159         """
 160         The response cookies. A possibly empty
 161         :py:class:`~mitmproxy.net.multidict.MultiDictView`, where the keys are cookie
 162         name strings, and values are (value, attr) tuples. Value is a string,
 163         and attr is an MultiDictView containing cookie attributes. Within
 164         attrs, unary attributes (e.g. HTTPOnly) are indicated by a Null value.
 165 
 166         Caveats:
 167             Updating the attr
 168         """
 169         return multidict.MultiDictView(
 170             self._get_cookies,
 171             self._set_cookies
 172         )
 173 
 174     @cookies.setter
 175     def cookies(self, value):
 176         self._set_cookies(value)

 178     def refresh(self, now=None):
 179         """
 180         This fairly complex and heuristic function refreshes a server
 181         response for replay.
 182 
 183             - It adjusts date, expires and last-modified headers.
 184             - It adjusts cookie expiration.
 185         """
 186         if not now:
 187             now = time.time()
 188         delta = now - self.timestamp_start
 189         refresh_headers = [
 190             "date",
 191             "expires",
 192             "last-modified",
 193         ]
 194         for i in refresh_headers:
 195             if i in self.headers:
 196                 d = parsedate_tz(self.headers[i])
 197                 if d:
 198                     new = mktime_tz(d) + delta
 199                     self.headers[i] = formatdate(new, usegmt=True)
 200         c = []
 201         for set_cookie_header in self.headers.get_all("set-cookie"):
 202             try:
 203                 refreshed = cookies.refresh_set_cookie_header(set_cookie_header, delta)
 204             except ValueError:
 205                 refreshed = set_cookie_header
 206             c.append(refreshed)
 207         if c:
 208             self.headers.set_all("set-cookie", c)
