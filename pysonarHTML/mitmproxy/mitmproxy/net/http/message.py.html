  89     def get_content(self, strict: bool=True) -> Optional[bytes]:
  90         """
  91         The uncompressed HTTP message body as bytes.
  92 
  93         Raises:
  94             ValueError, when the HTTP content-encoding is invalid and strict is True.
  95 
  96         See also: :py:class:`raw_content`, :py:attr:`text`
  97         """
  98         if self.raw_content is None:
  99             return None
 100         ce = self.headers.get("content-encoding")
 101         if ce:
 102             try:
 103                 content = encoding.decode(self.raw_content, ce)
 104                 # A client may illegally specify a byte -> str encoding here (e.g. utf8)
 105                 if isinstance(content, str):
 106                     raise ValueError("Invalid Content-Encoding: {}".format(ce))
 107                 return content
 108             except ValueError:
 109                 if strict:
 110                     raise
 111                 return self.raw_content
 112         else:
 113             return self.raw_content

 115     def set_content(self, value):
 116         if value is None:
 117             self.raw_content = None
 118             return
 119         if not isinstance(value, bytes):
 120             raise TypeError(
 121                 "Message content must be bytes, not {}. "
 122                 "Please use .text if you want to assign a str."
 123                 .format(type(value).__name__)
 124             )
 125         ce = self.headers.get("content-encoding")
 126         try:
 127             self.raw_content = encoding.encode(value, ce or "identity")
 128         except ValueError:
 129             # So we have an invalid content-encoding?
 130             # Let's remove it!
 131             del self.headers["content-encoding"]
 132             self.raw_content = value
 133         self.headers["content-length"] = str(len(self.raw_content))

 184     def _get_content_type_charset(self) -> Optional[str]:
 185         ct = mheaders.parse_content_type(self.headers.get("content-type", ""))
 186         if ct:
 187             return ct[2].get("charset")
 188         return None
 189 
 190     def _guess_encoding(self, content=b"") -> str:
 191         enc = self._get_content_type_charset()
 192         if not enc:
 193             if "json" in self.headers.get("content-type", ""):
 194                 enc = "utf8"
 195         if not enc:
 196             meta_charset = re.search(rb"""<meta[^>]+charset=['"]?([^'">]+)""", content)
 197             if meta_charset:
 198                 enc = meta_charset.group(1).decode("ascii", "ignore")
 199         if not enc:
 200             enc = "latin-1"
 201         # Use GB 18030 as the superset of GB2312 and GBK to fix common encoding problems on Chinese websites.
 202         if enc.lower() in ("gb2312", "gbk"):
 203             enc = "gb18030"
 204 
 205         return enc

 207     def get_text(self, strict: bool=True) -> Optional[str]:
 208         """
 209         The uncompressed and decoded HTTP message body as text.
 210 
 211         Raises:
 212             ValueError, when either content-encoding or charset is invalid and strict is True.
 213 
 214         See also: :py:attr:`content`, :py:class:`raw_content`
 215         """
 216         content = self.get_content(strict)
 217         if content is None:
 218             return None
 219         enc = self._guess_encoding(content)
 220         try:
 221             return encoding.decode(content, enc)
 222         except ValueError:
 223             if strict:
 224                 raise
 225             return content.decode("utf8", "surrogateescape")
 226 
 227     def set_text(self, text):
 228         if text is None:
 229             self.content = None
 230             return
 231         enc = self._guess_encoding()
 232 
 233         try:
 234             self.content = encoding.encode(text, enc)
 235         except ValueError:
 236             # Fall back to UTF-8 and update the content-type header.
 237             ct = mheaders.parse_content_type(self.headers.get("content-type", "")) or ("text", "plain", {})
 238             ct[2]["charset"] = "utf-8"
 239             self.headers["content-type"] = mheaders.assemble_content_type(*ct)
 240             enc = "utf8"
 241             self.content = text.encode(enc, "surrogateescape")

 245     def decode(self, strict=True):
 246         """
 247         Decodes body based on the current Content-Encoding header, then
 248         removes the header. If there is no Content-Encoding header, no
 249         action is taken.
 250 
 251         Raises:
 252             ValueError, when the content-encoding is invalid and strict is True.
 253         """
 254         decoded = self.get_content(strict)
 255         self.headers.pop("content-encoding", None)
 256         self.content = decoded
 257 
 258     def encode(self, e):
 259         """
 260         Encodes body with the encoding e, where e is "gzip", "deflate", "identity", "br", or "zstd".
 261         Any existing content-encodings are overwritten,
 262         the content is not decoded beforehand.
 263 
 264         Raises:
 265             ValueError, when the specified content-encoding is invalid.
 266         """
 267         self.headers["content-encoding"] = e
 268         self.content = self.raw_content
 269         if "content-encoding" not in self.headers:
 270             raise ValueError("Invalid content encoding {}".format(repr(e)))
