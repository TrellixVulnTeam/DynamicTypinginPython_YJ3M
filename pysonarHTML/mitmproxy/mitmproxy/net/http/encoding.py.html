  17 # We have a shared single-element cache for encoding and decoding.
  18 # This is quite useful in practice, e.g.
  19 # flow.request.content = flow.request.content.replace(b"foo", b"bar")
  20 # does not require an .encode() call if content does not contain b"foo"
  21 CachedDecode = collections.namedtuple(
  22     "CachedDecode", "encoded encoding errors decoded"
  23 )
  24 _cache = CachedDecode(None, None, None, None)
  25

  27 def decode(
  28     encoded: Optional[bytes], encoding: str, errors: str='strict'
  29 ) -> Optional[AnyStr]:
  30     """
  31     Decode the given input object
  32 
  33     Returns:
  34         The decoded value
  35 
  36     Raises:
  37         ValueError, if decoding fails.
  38     """
  39     if encoded is None:
  40         return None
  41 
  42     global _cache
  43     cached = (
  44         isinstance(encoded, bytes) and
  45         _cache.encoded == encoded and
  46         _cache.encoding == encoding and
  47         _cache.errors == errors
  48     )
  49     if cached:
  50         return _cache.decoded
  51     try:
  52         try:
  53             decoded = custom_decode[encoding](encoded)
  54         except KeyError:
  55             decoded = codecs.decode(encoded, encoding, errors)
  56         if encoding in ("gzip", "deflate", "br", "zstd"):
  57             _cache = CachedDecode(encoded, encoding, errors, decoded)
  58         return decoded
  59     except TypeError:
  60         raise
  61     except Exception as e:
  62         raise ValueError("{} when decoding {} with {}: {}".format(
  63             type(e).__name__,
  64             repr(encoded)[:10],
  65             repr(encoding),
  66             repr(e),
  67         ))

  70 def encode(decoded: Optional[str], encoding: str, errors: str='strict') -> Optional[AnyStr]:
  71     """
  72     Encode the given input object
  73 
  74     Returns:
  75         The encoded value
  76 
  77     Raises:
  78         ValueError, if encoding fails.
  79     """
  80     if decoded is None:
  81         return None
  82 
  83     global _cache
  84     cached = (
  85         isinstance(decoded, bytes) and
  86         _cache.decoded == decoded and
  87         _cache.encoding == encoding and
  88         _cache.errors == errors
  89     )
  90     if cached:
  91         return _cache.encoded
  92     try:
  93         try:
  94             encoded = custom_encode[encoding](decoded)
  95         except KeyError:
  96             encoded = codecs.encode(decoded, encoding, errors)
  97         if encoding in ("gzip", "deflate", "br", "zstd"):
  98             _cache = CachedDecode(encoded, encoding, errors, decoded)
  99         return encoded
 100     except TypeError:
 101         raise
 102     except Exception as e:
 103         raise ValueError("{} when encoding {} with {}: {}".format(
 104             type(e).__name__,
 105             repr(decoded)[:10],
 106             repr(encoding),
 107             repr(e),
 108         ))

 111 def identity(content):
 112     """
 113         Returns content unchanged. Identity is the default value of
 114         Accept-Encoding headers.
 115     """
 116     return content

 119 def decode_gzip(content: bytes) -> bytes:
 120     if not content:
 121         return b""
 122     gfile = gzip.GzipFile(fileobj=BytesIO(content))
 123     return gfile.read()
 124 
 125 
 126 def encode_gzip(content: bytes) -> bytes:
 127     s = BytesIO()
 128     gf = gzip.GzipFile(fileobj=s, mode='wb')
 129     gf.write(content)
 130     gf.close()
 131     return s.getvalue()

 134 def decode_brotli(content: bytes) -> bytes:
 135     if not content:
 136         return b""
 137     return brotli.decompress(content)
 138 
 139 
 140 def encode_brotli(content: bytes) -> bytes:
 141     return brotli.compress(content)

 144 def decode_zstd(content: bytes) -> bytes:
 145     if not content:
 146         return b""
 147     zstd_ctx = zstd.ZstdDecompressor()
 148     try:
 149         return zstd_ctx.decompress(content)
 150     except zstd.ZstdError:
 151         # If the zstd stream is streamed without a size header,
 152         # try decoding with a 10MiB output buffer
 153         return zstd_ctx.decompress(content, max_output_size=10 * 2**20)
 154 
 155 
 156 def encode_zstd(content: bytes) -> bytes:
 157     zstd_ctx = zstd.ZstdCompressor()
 158     return zstd_ctx.compress(content)

 161 def decode_deflate(content: bytes) -> bytes:
 162     """
 163         Returns decompressed data for DEFLATE. Some servers may respond with
 164         compressed data without a zlib header or checksum. An undocumented
 165         feature of zlib permits the lenient decompression of data missing both
 166         values.
 167 
 168         http://bugs.python.org/issue5784
 169     """
 170     if not content:
 171         return b""
 172     try:
 173         return zlib.decompress(content)
 174     except zlib.error:
 175         return zlib.decompress(content, -15)
 176 
 177 
 178 def encode_deflate(content: bytes) -> bytes:
 179     """
 180         Returns compressed content, always including zlib header and checksum.
 181     """
 182     return zlib.compress(content)
