  17 class Headers(multidict.MultiDict):
  18     """
  19     Header class which allows both convenient access to individual headers as well as
  20     direct access to the underlying raw data. Provides a full dictionary interface.
  21 
  22     Example:
  23 
  24     .. code-block:: python
  25 
  26         # Create headers with keyword arguments
  27         >>> h = Headers(host="example.com", content_type="application/xml")
  28 
  29         # Headers mostly behave like a normal dict.
  30         >>> h["Host"]
  31         "example.com"
  32 
  33         # HTTP Headers are case insensitive
  34         >>> h["host"]
  35         "example.com"
  36 
  37         # Headers can also be created from a list of raw (header_name, header_value) byte tuples
  38         >>> h = Headers([
  39             (b"Host",b"example.com"),
  40             (b"Accept",b"text/html"),
  41             (b"accept",b"application/xml")
  42         ])
  43 
  44         # Multiple headers are folded into a single header as per RFC7230
  45         >>> h["Accept"]
  46         "text/html, application/xml"
  47 
  48         # Setting a header removes all existing headers with the same name.
  49         >>> h["Accept"] = "application/text"
  50         >>> h["Accept"]
  51         "application/text"
  52 
  53         # bytes(h) returns a HTTP1 header block.
  54         >>> print(bytes(h))
  55         Host: example.com
  56         Accept: application/text
  57 
  58         # For full control, the raw header fields can be accessed
  59         >>> h.fields
  60 
  61     Caveats:
  62         For use with the "Set-Cookie" header, see :py:meth:`get_all`.
  63     """
  64 
  65     def __init__(self, fields=(), **headers):
  66         """
  67         Args:
  68             fields: (optional) list of ``(name, value)`` header byte tuples,
  69                 e.g. ``[(b"Host", b"example.com")]``. All names and values must be bytes.
  70             **headers: Additional headers to set. Will overwrite existing values from `fields`.
  71                 For convenience, underscores in header names will be transformed to dashes -
  72                 this behaviour does not extend to other methods.
  73                 If ``**headers`` contains multiple keys that have equal ``.lower()`` s,
  74                 the behavior is undefined.
  75         """
  76         super().__init__(fields)
  77 
  78         for key, value in self.fields:
  79             if not isinstance(key, bytes) or not isinstance(value, bytes):
  80                 raise TypeError("Header fields must be bytes.")
  81 
  82         # content_type -> content-type
  83         headers = {
  84             _always_bytes(name).replace(b"_", b"-"): _always_bytes(value)
  85             for name, value in headers.items()
  86         }
  87         self.update(headers)
  88 
  89     @staticmethod
  90     def _reduce_values(values):
  91         # Headers can be folded
  92         return ", ".join(values)
  93 
  94     @staticmethod
  95     def _kconv(key):
  96         # Headers are case-insensitive
  97         return key.lower()
  98 
  99     def __bytes__(self):
 100         if self.fields:
 101             return b"\r\n".join(b": ".join(field) for field in self.fields) + b"\r\n"
 102         else:
 103             return b""
 104 
 105     def __delitem__(self, key):
 106         key = _always_bytes(key)
 107         super().__delitem__(key)
 108 
 109     def __iter__(self):
 110         for x in super().__iter__():
 111             yield _native(x)
 112 
 113     def get_all(self, name):
 114         """
 115         Like :py:meth:`get`, but does not fold multiple headers into a single one.
 116         This is useful for Set-Cookie headers, which do not support folding.
 117         See also: https://tools.ietf.org/html/rfc7230#section-3.2.2
 118         """
 119         name = _always_bytes(name)
 120         return [
 121             _native(x) for x in
 122             super().get_all(name)
 123         ]
 124 
 125     def set_all(self, name, values):
 126         """
 127         Explicitly set multiple headers for the given key.
 128         See: :py:meth:`get_all`
 129         """
 130         name = _always_bytes(name)
 131         values = [_always_bytes(x) for x in values]
 132         return super().set_all(name, values)
 133 
 134     def insert(self, index, key, value):
 135         key = _always_bytes(key)
 136         value = _always_bytes(value)
 137         super().insert(index, key, value)
 138 
 139     def items(self, multi=False):
 140         if multi:
 141             return (
 142                 (_native(k), _native(v))
 143                 for k, v in self.fields
 144             )
 145         else:
 146             return super().items()

 149 def parse_content_type(c):
 150     """
 151         A simple parser for content-type values. Returns a (type, subtype,
 152         parameters) tuple, where type and subtype are strings, and parameters
 153         is a dict. If the string could not be parsed, return None.
 154 
 155         E.g. the following string:
 156 
 157             text/html; charset=UTF-8
 158 
 159         Returns:
 160 
 161             ("text", "html", {"charset": "UTF-8"})
 162     """
 163     parts = c.split(";", 1)
 164     ts = parts[0].split("/", 1)
 165     if len(ts) != 2:
 166         return None
 167     d = collections.OrderedDict()
 168     if len(parts) == 2:
 169         for i in parts[1].split(";"):
 170             clause = i.split("=", 1)
 171             if len(clause) == 2:
 172                 d[clause[0].strip()] = clause[1].strip()
 173     return ts[0].lower(), ts[1].lower(), d

 176 def assemble_content_type(type, subtype, parameters):
 177     if not parameters:
 178         return "{}/{}".format(type, subtype)
 179     params = "; ".join(
 180         "{}={}".format(k, v)
 181         for k, v in parameters.items()
 182     )
 183     return "{}/{}; {}".format(
 184         type, subtype, params
 185     )
