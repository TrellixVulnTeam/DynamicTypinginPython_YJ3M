  14 # This regex extracts & splits the host header into host and port.

  17 host_header_re = re.compile(r"^(?P<host>[^:]+|\[.+\])(?::(?P<port>\d+))?$")

  77     def __repr__(self):
  78         if self.host and self.port:
  79             hostport = "{}:{}".format(self.host, self.port)
  80         else:
  81             hostport = ""
  82         path = self.path or ""
  83         return "Request({} {}{})".format(
  84             self.method, hostport, path
  85         )

  87     @classmethod
  88     def make(
  89             cls,
  90             method: str,
  91             url: str,
  92             content: Union[bytes, str] = "",
  93             headers: Union[Dict[str, AnyStr], Iterable[Tuple[bytes, bytes]]] = ()
  94     ):
  95         """
  96         Simplified API for creating request objects.
  97         """
  98         req = cls(
  99             "absolute",
 100             method,
 101             "",
 102             "",
 103             "",
 104             "",
 105             "HTTP/1.1",
 106             (),
 107             b""
 108         )
 109 
 110         req.url = url
 111         req.timestamp_start = time.time()
 112 
 113         # Headers can be list or dict, we differentiate here.
 114         if isinstance(headers, dict):
 115             req.headers = nheaders.Headers(**headers)
 116         elif isinstance(headers, Iterable):
 117             req.headers = nheaders.Headers(headers)
 118         else:
 119             raise TypeError("Expected headers to be an iterable or dict, but is {}.".format(
 120                 type(headers).__name__
 121             ))
 122 
 123         # Assign this manually to update the content-length header.
 124         if isinstance(content, bytes):
 125             req.content = content
 126         elif isinstance(content, str):
 127             req.text = content
 128         else:
 129             raise TypeError("Expected content to be str or bytes, but is {}.".format(
 130                 type(content).__name__
 131             ))
 132 
 133         return req

 172     @property
 173     def host(self):
 174         """
 175         Target host. This may be parsed from the raw request
 176         (e.g. from a ``GET http://example.com/ HTTP/1.1`` request line)
 177         or inferred from the proxy mode (e.g. an IP in transparent mode).
 178 
 179         Setting the host attribute also updates the host header, if present.
 180         """
 181         if not self.data.host:
 182             return self.data.host
 183         try:
 184             return self.data.host.decode("idna")
 185         except UnicodeError:
 186             return self.data.host.decode("utf8", "surrogateescape")
 187 
 188     @host.setter
 189     def host(self, host):
 190         if isinstance(host, str):
 191             try:
 192                 # There's no non-strict mode for IDNA encoding.
 193                 # We don't want this operation to fail though, so we try
 194                 # utf8 as a last resort.
 195                 host = host.encode("idna", "strict")
 196             except UnicodeError:
 197                 host = host.encode("utf8", "surrogateescape")
 198 
 199         self.data.host = host
 200 
 201         # Update host header
 202         if self.host_header is not None:
 203             self.host_header = host
 204 
 205     @property
 206     def host_header(self) -> Optional[str]:
 207         """
 208         The request's host/authority header.
 209 
 210         This property maps to either ``request.headers["Host"]`` or
 211         ``request.headers[":authority"]``, depending on whether it's HTTP/1.x or HTTP/2.0.
 212         """
 213         if ":authority" in self.headers:
 214             return self.headers[":authority"]
 215         if "Host" in self.headers:
 216             return self.headers["Host"]
 217         return None
 218 
 219     @host_header.setter
 220     def host_header(self, val: Optional[str]) -> None:
 221         if val is None:
 222             self.headers.pop("Host", None)
 223             self.headers.pop(":authority", None)
 224         elif self.host_header is not None:
 225             # Update any existing headers.
 226             if ":authority" in self.headers:
 227                 self.headers[":authority"] = val
 228             if "Host" in self.headers:
 229                 self.headers["Host"] = val
 230         else:
 231             # Only add the correct new header.
 232             if self.http_version.upper().startswith("HTTP/2"):
 233                 self.headers[":authority"] = val
 234             else:
 235                 self.headers["Host"] = val
 236 
 237     @host_header.deleter
 238     def host_header(self):
 239         self.host_header = None

 267     @property

 269         """
 270         The URL string, constructed from the request's URL components
 271         """
 272         if self.first_line_format == "authority":
 273             return "%s:%d" % (self.host, self.port)
 274         return mitmproxy.net.http.url.unparse(self.scheme, self.host, self.port, self.path)
 275 
 276     @url.setter
 277     def url(self, url):
 278         self.scheme, self.host, self.port, self.path = mitmproxy.net.http.url.parse(url)

 280     def _parse_host_header(self):
 281         """Extract the host and port from Host header"""
 282         host = self.host_header
 283         if not host:
 284             return None, None
 285         port = None
 286         m = host_header_re.match(host)
 287         if m:
 288             host = m.group("host").strip("[]")
 289             if m.group("port"):
 290                 port = int(m.group("port"))
 291         return host, port
 292 
 293     @property
 294     def pretty_host(self):
 295         """
 296         Similar to :py:attr:`host`, but using the Host headers as an additional preferred data source.
 297         This is useful in transparent mode where :py:attr:`host` is only an IP address,
 298         but may not reflect the actual destination as the Host header could be spoofed.
 299         """
 300         host, port = self._parse_host_header()
 301         if not host:
 302             return self.host
 303         if not port:
 304             port = 443 if self.scheme == 'https' else 80
 305         # Prefer the original address if host header has an unexpected form
 306         return host if port == self.port else self.host

 308     @property
 309     def pretty_url(self):
 310         """
 311         Like :py:attr:`url`, but using :py:attr:`pretty_host` instead of :py:attr:`host`.
 312         """
 313         if self.first_line_format == "authority":
 314             return "%s:%d" % (self.pretty_host, self.port)
 315         return mitmproxy.net.http.url.unparse(self.scheme, self.pretty_host, self.port, self.path)

 317     def _get_query(self):

 363     @property
 364     def path_components(self):
 365         """
 366         The URL's path components as a tuple of strings.
 367         Components are unquoted.
 368         """
 369         path = urllib.parse.urlparse(self.url).path
 370         # This needs to be a tuple so that it's immutable.
 371         # Otherwise, this would fail silently:
 372         #   request.path_components.append("foo")
 373         return tuple(mitmproxy.net.http.url.unquote(i) for i in path.split("/") if i)
 374 
 375     @path_components.setter
 376     def path_components(self, components):
 377         components = map(lambda x: mitmproxy.net.http.url.quote(x, safe=""), components)
 378         path = "/" + "/".join(components)
 379         _, _, _, params, query, fragment = urllib.parse.urlparse(self.url)
 380         self.path = urllib.parse.urlunparse(["", "", path, params, query, fragment])

 382     def anticache(self):
 383         """
 384         Modifies this request to remove headers that might produce a cached
 385         response. That is, we remove ETags and If-Modified-Since headers.
 386         """
 387         delheaders = [
 388             "if-modified-since",
 389             "if-none-match",
 390         ]
 391         for i in delheaders:
 392             self.headers.pop(i, None)
 393 
 394     def anticomp(self):
 395         """
 396         Modifies this request to remove headers that will compress the
 397         resource's data.
 398         """
 399         self.headers["accept-encoding"] = "identity"

 401     def constrain_encoding(self):
 402         """
 403         Limits the permissible Accept-Encoding values, based on what we can
 404         decode appropriately.
 405         """
 406         accept_encoding = self.headers.get("accept-encoding")
 407         if accept_encoding:
 408             self.headers["accept-encoding"] = (
 409                 ', '.join(
 410                     e
 411                     for e in {"gzip", "identity", "deflate", "br", "zstd"}
 412                     if e in accept_encoding
 413                 )
 414             )

 416     def _get_urlencoded_form(self):
 417         is_valid_content_type = "application/x-www-form-urlencoded" in self.headers.get("content-type", "").lower()
 418         if is_valid_content_type:
 419             return tuple(mitmproxy.net.http.url.decode(self.get_text(strict=False)))
 420         return ()

 422     def _set_urlencoded_form(self, form_data):
 423         """
 424         Sets the body to the URL-encoded form data, and adds the appropriate content-type header.
 425         This will overwrite the existing content if there is one.
 426         """
 427         self.headers["content-type"] = "application/x-www-form-urlencoded"
 428         self.content = mitmproxy.net.http.url.encode(form_data, self.get_text(strict=False)).encode()

 448     def _get_multipart_form(self):

 450         if is_valid_content_type:
