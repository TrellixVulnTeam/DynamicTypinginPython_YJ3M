  56 class ClientGreeting:
  57     __slots__ = ("ver", "methods")
  58 
  59     def __init__(self, ver, methods):
  60         self.ver = ver
  61         self.methods = array.array("B")
  62         self.methods.extend(methods)
  63 
  64     def assert_socks5(self):
  65         if self.ver != VERSION.SOCKS5:
  66             if self.ver == ord("G") and len(self.methods) == ord("E"):
  67                 guess = "Probably not a SOCKS request but a regular HTTP request. "
  68             else:
  69                 guess = ""
  70 
  71             raise SocksError(
  72                 REP.GENERAL_SOCKS_SERVER_FAILURE,
  73                 guess + "Invalid SOCKS version. Expected 0x05, got 0x%x" % self.ver
  74             )
  75 
  76     @classmethod
  77     def from_file(cls, f, fail_early=False):
  78         """
  79         :param fail_early: If true, a SocksError will be raised if the first byte does not indicate socks5.
  80         """
  81         ver, nmethods = struct.unpack("!BB", f.safe_read(2))
  82         client_greeting = cls(ver, [])
  83         if fail_early:
  84             client_greeting.assert_socks5()
  85         client_greeting.methods.frombytes(f.safe_read(nmethods))
  86         return client_greeting
  87 
  88     def to_file(self, f):
  89         f.write(struct.pack("!BB", self.ver, len(self.methods)))
  90         f.write(self.methods.tobytes())

 121 class UsernamePasswordAuth:
 122     __slots__ = ("ver", "username", "password")
 123 
 124     def __init__(self, ver, username, password):
 125         self.ver = ver
 126         self.username = username
 127         self.password = password
 128 
 129     def assert_authver1(self):
 130         if self.ver != USERNAME_PASSWORD_VERSION.DEFAULT:
 131             raise SocksError(
 132                 0,
 133                 "Invalid auth version. Expected 0x01, got 0x%x" % self.ver
 134             )
 135 
 136     @classmethod
 137     def from_file(cls, f):
 138         ver, ulen = struct.unpack("!BB", f.safe_read(2))
 139         username = f.safe_read(ulen)
 140         plen, = struct.unpack("!B", f.safe_read(1))
 141         password = f.safe_read(plen)
 142         return cls(ver, username.decode(), password.decode())
 143 
 144     def to_file(self, f):
 145         f.write(struct.pack("!BB", self.ver, len(self.username)))
 146         f.write(self.username.encode())
 147         f.write(struct.pack("!B", len(self.password)))
 148         f.write(self.password.encode())

 174 class Message:
 175     __slots__ = ("ver", "msg", "atyp", "addr")
 176 
 177     def __init__(self, ver, msg, atyp, addr):
 178         self.ver = ver
 179         self.msg = msg
 180         self.atyp = atyp
 181         self.addr = addr
 182 
 183     def assert_socks5(self):
 184         if self.ver != VERSION.SOCKS5:
 185             raise SocksError(
 186                 REP.GENERAL_SOCKS_SERVER_FAILURE,
 187                 "Invalid SOCKS version. Expected 0x05, got 0x%x" % self.ver
 188             )
 189 
 190     @classmethod
 191     def from_file(cls, f):
 192         ver, msg, rsv, atyp = struct.unpack("!BBBB", f.safe_read(4))
 193         if rsv != 0x00:
 194             raise SocksError(
 195                 REP.GENERAL_SOCKS_SERVER_FAILURE,
 196                 "Socks Request: Invalid reserved byte: %s" % rsv
 197             )
 198         if atyp == ATYP.IPV4_ADDRESS:
 199             # We use tnoa here as ntop is not commonly available on Windows.
 200             host = ipaddress.IPv4Address(f.safe_read(4)).compressed
 201         elif atyp == ATYP.IPV6_ADDRESS:
 202             host = ipaddress.IPv6Address(f.safe_read(16)).compressed
 203         elif atyp == ATYP.DOMAINNAME:
 204             length, = struct.unpack("!B", f.safe_read(1))
 205             host = f.safe_read(length)
 206             if not check.is_valid_host(host):
 207                 raise SocksError(REP.GENERAL_SOCKS_SERVER_FAILURE, "Invalid hostname: %s" % host)
 208             host = host.decode("idna")
 209         else:
 210             raise SocksError(REP.ADDRESS_TYPE_NOT_SUPPORTED,
 211                              "Socks Request: Unknown ATYP: %s" % atyp)
 212 
 213         port, = struct.unpack("!H", f.safe_read(2))
 214         addr = (host, port)
 215         return cls(ver, msg, atyp, addr)
 216 
 217     def to_file(self, f):
 218         f.write(struct.pack("!BBBB", self.ver, self.msg, 0x00, self.atyp))
 219         if self.atyp == ATYP.IPV4_ADDRESS:
 220             f.write(ipaddress.IPv4Address(self.addr[0]).packed)
 221         elif self.atyp == ATYP.IPV6_ADDRESS:
 222             f.write(ipaddress.IPv6Address(self.addr[0]).packed)
 223         elif self.atyp == ATYP.DOMAINNAME:
 224             f.write(struct.pack("!B", len(self.addr[0])))
 225             f.write(self.addr[0].encode("idna"))
 226         else:
 227             raise SocksError(
 228                 REP.ADDRESS_TYPE_NOT_SUPPORTED,
 229                 "Unknown ATYP: %s" % self.atyp
 230             )
 231         f.write(struct.pack("!H", self.addr[1]))
