  91             self.first_byte_timestamp = self.first_byte_timestamp or time.time()

 106     def read(self, length):
 107         """
 108             If length is -1, we read until connection closes.
 109         """
 110         result = b''
 111         start = time.time()
 112         while length == -1 or length > 0:
 113             if length == -1 or length > self.BLOCKSIZE:
 114                 rlen = self.BLOCKSIZE
 115             else:
 116                 rlen = length
 117             try:
 118                 data = self.o.read(rlen)
 119             except SSL.ZeroReturnError:
 120                 # TLS connection was shut down cleanly
 121                 break
 122             except (SSL.WantWriteError, SSL.WantReadError):
 123                 # From the OpenSSL docs:
 124                 # If the underlying BIO is non-blocking, SSL_read() will also return when the
 125                 # underlying BIO could not satisfy the needs of SSL_read() to continue the
 126                 # operation. In this case a call to SSL_get_error with the return value of
 127                 # SSL_read() will yield SSL_ERROR_WANT_READ or SSL_ERROR_WANT_WRITE.
 128                 # 300 is OpenSSL default timeout
 129                 timeout = self.o.gettimeout() or 300
 130                 if (time.time() - start) < timeout:
 131                     time.sleep(0.1)
 132                     continue
 133                 else:
 134                     raise exceptions.TcpTimeout()
 135             except socket.timeout:
 136                 raise exceptions.TcpTimeout()
 137             except socket.error as e:
 138                 raise exceptions.TcpDisconnect(str(e))
 139             except SSL.SysCallError as e:
 140                 if e.args == (-1, 'Unexpected EOF'):
 141                     break
 142                 raise exceptions.TlsException(str(e))
 143             except SSL.Error as e:
 144                 raise exceptions.TlsException(str(e))
 145             self.first_byte_timestamp = self.first_byte_timestamp or time.time()
 146             if not data:
 147                 break
 148             result += data
 149             if length != -1:
 150                 length -= len(data)
 151         self.add_log(result)
 152         return result
 153 
 154     def readline(self, size=None):
 155         result = b''
 156         bytes_read = 0
 157         while True:
 158             if size is not None and bytes_read >= size:
 159                 break
 160             ch = self.read(1)
 161             bytes_read += 1
 162             if not ch:
 163                 break
 164             else:
 165                 result += ch
 166                 if ch == b'\n':
 167                     break
 168         return result

 170     def safe_read(self, length):
 171         """
 172             Like .read, but is guaranteed to either return length bytes, or
 173             raise an exception.
 174         """
 175         result = self.read(length)
 176         if length != -1 and len(result) != length:
 177             if not result:
 178                 raise exceptions.TcpDisconnect()
 179             else:
 180                 raise exceptions.TcpReadIncomplete(
 181                     "Expected %s bytes, got %s" % (length, len(result))
 182                 )
 183         return result

 185     def peek(self, length):
 186         """
 187         Tries to peek into the underlying file object.
 188 
 189         Returns:
 190             Up to the next N bytes if peeking is successful.
 191 
 192         Raises:
 193             exceptions.TcpException if there was an error with the socket
 194             TlsException if there was an error with pyOpenSSL.
 195             NotImplementedError if the underlying file object is not a [pyOpenSSL] socket
 196         """
 197         if isinstance(self.o, socket_fileobject):
 198             try:
 199                 return self.o._sock.recv(length, socket.MSG_PEEK)
 200             except socket.error as e:
 201                 raise exceptions.TcpException(repr(e))
 202         elif isinstance(self.o, SSL.Connection):
 203             try:
 204                 return self.o.recv(length, socket.MSG_PEEK)
 205             except SSL.Error as e:
 206                 raise exceptions.TlsException(str(e))
 207         else:
 208             raise NotImplementedError("Can only peek into (pyOpenSSL) sockets")

 211 def ssl_read_select(rlist, timeout):
 212     """
 213     This is a wrapper around select.select() which also works for SSL.Connections
 214     by taking ssl_connection.pending() into account.
 215 
 216     Caveats:
 217         If .pending() > 0 for any of the connections in rlist, we avoid the select syscall
 218         and **will not include any other connections which may or may not be ready**.
 219 
 220     Args:
 221         rlist: wait until ready for reading
 222 
 223     Returns:
 224         subset of rlist which is ready for reading.
 225     """
 226     return [
 227         conn for conn in rlist
 228         if isinstance(conn, SSL.Connection) and conn.pending() > 0
 229     ] or select.select(rlist, (), (), timeout)[0]

 232 def close_socket(sock):
 233     """
 234     Does a hard close of a socket, without emitting a RST.
 235     """
 236     try:
 237         # We already indicate that we close our end.
 238         # may raise "Transport endpoint is not connected" on Linux
 239         sock.shutdown(socket.SHUT_WR)
 240 
 241         # Section 4.2.2.13 of RFC 1122 tells us that a close() with any pending
 242         # readable data could lead to an immediate RST being sent (which is the
 243         # case on Windows).
 244         # http://ia600609.us.archive.org/22/items/TheUltimateSo_lingerPageOrWhyIsMyTcpNotReliable/the-ultimate-so_linger-page-or-why-is-my-tcp-not-reliable.html
 245         #
 246         # This in turn results in the following issue: If we send an error page
 247         # to the client and then close the socket, the RST may be received by
 248         # the client before the error page and the users sees a connection
 249         # error rather than the error page. Thus, we try to empty the read
 250         # buffer on Windows first. (see
 251         # https://github.com/mitmproxy/mitmproxy/issues/527#issuecomment-93782988)
 252         #
 253 
 254         if os.name == "nt":  # pragma: no cover
 255             # We cannot rely on the shutdown()-followed-by-read()-eof technique
 256             # proposed by the page above: Some remote machines just don't send
 257             # a TCP FIN, which would leave us in the unfortunate situation that
 258             # recv() would block infinitely. As a workaround, we set a timeout
 259             # here even if we are in blocking mode.
 260             sock.settimeout(sock.gettimeout() or 20)
 261 
 262             # limit at a megabyte so that we don't read infinitely
 263             for _ in range(1024 ** 3 // 4096):
 264                 # may raise a timeout/disconnect exception.
 265                 if not sock.recv(4096):
 266                     break
 267 
 268         # Now we can close the other half as well.
 269         sock.shutdown(socket.SHUT_RD)
 270 
 271     except socket.error:
 272         pass
 273 
 274     sock.close()

 282     def _makefile(self):
 283         """
 284         Set up .rfile and .wfile attributes from .connection
 285         """
 286         # Ideally, we would use the Buffered IO in Python 3 by default.
 287         # Unfortunately, the implementation of .peek() is broken for n>1 bytes,
 288         # as it may just return what's left in the buffer and not all the bytes we want.
 289         # As a workaround, we just use unbuffered sockets directly.
 290         # https://mail.python.org/pipermail/python-dev/2009-June/089986.html
 291         self.rfile = Reader(socket.SocketIO(self.connection, "rb"))
 292         self.wfile = Writer(socket.SocketIO(self.connection, "wb"))
 293 
 294     def __init__(self, connection):
 295         if connection:
 296             self.connection = connection
 297             self.ip_address = connection.getpeername()
 298             self._makefile()
 299         else:
 300             self.connection = None
 301             self.ip_address = None
 302             self.rfile = None
 303             self.wfile = None
 304 
 305         self.tls_established = False
 306         self.finished = False
 307 
 308     def get_current_cipher(self):
 309         if not self.tls_established:
 310             return None
 311 
 312         name = self.connection.get_cipher_name()
 313         bits = self.connection.get_cipher_bits()
 314         version = self.connection.get_cipher_version()
 315         return name, bits, version
 316 
 317     def finish(self):
 318         self.finished = True
 319         # If we have an SSL connection, wfile.close == connection.close
 320         # (We call _FileLike.set_descriptor(conn))
 321         # Closing the socket is not our task, therefore we don't call close
 322         # then.
 323         if not isinstance(self.connection, SSL.Connection):
 324             if not getattr(self.wfile, "closed", False):
 325                 try:
 326                     self.wfile.flush()
 327                     self.wfile.close()
 328                 except exceptions.TcpDisconnect:
 329                     pass
 330 
 331             self.rfile.close()
 332         else:
 333             try:
 334                 self.connection.shutdown()
 335             except SSL.Error:
 336                 pass

 417     def create_connection(self, timeout=None):
 418         # Based on the official socket.create_connection implementation of Python 3.6.
 419         # https://github.com/python/cpython/blob/3cc5817cfaf5663645f4ee447eaed603d2ad290a/Lib/socket.py
 420 
 421         err = None
 422         for res in socket.getaddrinfo(self.address[0], self.address[1], 0, socket.SOCK_STREAM):
 423             af, socktype, proto, canonname, sa = res
 424             sock = None
 425             try:
 426                 sock = self.makesocket(af, socktype, proto)
 427                 if timeout:
 428                     sock.settimeout(timeout)
 429                 if self.source_address:
 430                     sock.bind(self.source_address)
 431                 if self.spoof_source_address:
 432                     try:
 433                         if not sock.getsockopt(socket.SOL_IP, socket.IP_TRANSPARENT):
 434                             sock.setsockopt(socket.SOL_IP, socket.IP_TRANSPARENT, 1)  # pragma: windows no cover  pragma: osx no cover
 435                     except Exception as e:
 436                         # socket.IP_TRANSPARENT might not be available on every OS and Python version
 437                         if sock is not None:
 438                             sock.close()
 439                         raise exceptions.TcpException(
 440                             "Failed to spoof the source address: " + str(e)
 441                         )
 442                 sock.connect(sa)
 443                 return sock
 444 
 445             except socket.error as _:
 446                 err = _
 447                 if sock is not None:
 448                     sock.close()
 449 
 450         if err is not None:
 451             raise err
 452         else:
 453             raise socket.error("getaddrinfo returns an empty list")  # pragma: no cover

 549 class TCPServer:
 550 
 551     def __init__(self, address):
 552         self.address = address
 553         self.__is_shut_down = threading.Event()
 554         self.__is_shut_down.set()
 555         self.__shutdown_request = False
 556 
 557         if self.address[0] == 'localhost':
 558             raise socket.error("Binding to 'localhost' is prohibited. Please use '::1' or '127.0.0.1' directly.")
 559 
 560         self.socket = None
 561 
 562         try:
 563             # First try to bind an IPv6 socket, attempting to enable IPv4 support if the OS supports it.
 564             # This allows us to accept connections for ::1 and 127.0.0.1 on the same socket.
 565             # Only works if self.address == ""
 566             self.socket = socket.socket(socket.AF_INET6, socket.SOCK_STREAM)
 567             self.socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
 568             self.socket.setsockopt(socket.SOL_SOCKET, socket.SO_KEEPALIVE, 1)
 569             self.socket.setsockopt(IPPROTO_IPV6, socket.IPV6_V6ONLY, 0)
 570             self.socket.bind(self.address)
 571         except socket.error:
 572             if self.socket:
 573                 self.socket.close()
 574             self.socket = None
 575 
 576         if not self.socket:
 577             try:
 578                 # Binding to an IPv6 + IPv4 socket failed, lets fall back to IPv4 only.
 579                 self.socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
 580                 self.socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
 581                 self.socket.setsockopt(socket.SOL_SOCKET, socket.SO_KEEPALIVE, 1)
 582                 self.socket.bind(self.address)
 583             except socket.error:
 584                 if self.socket:
 585                     self.socket.close()
 586                 self.socket = None
 587 
 588         if not self.socket:
 589             # Binding to an IPv4 only socket failed, lets fall back to IPv6 only.
 590             self.socket = socket.socket(socket.AF_INET6, socket.SOCK_STREAM)
 591             self.socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
 592             self.socket.setsockopt(socket.SOL_SOCKET, socket.SO_KEEPALIVE, 1)
 593             self.socket.bind(self.address)
 594 
 595         self.address = self.socket.getsockname()
 596         self.socket.listen()
 597         self.handler_counter = Counter()
 598 
 599     def connection_thread(self, connection, client_address):
 600         with self.handler_counter:
 601             try:
 602                 self.handle_client_connection(connection, client_address)
 603             except OSError as e:  # pragma: no cover
 604                 # This catches situations where the underlying connection is
 605                 # closed beneath us. Syscalls on the connection object at this
 606                 # point returns EINVAL. If this happens, we close the socket and
 607                 # move on.
 608                 if not e.errno == errno.EINVAL:
 609                     raise
 610             except:
 611                 self.handle_error(connection, client_address)
 612             finally:
 613                 close_socket(connection)
 614 
 615     def serve_forever(self, poll_interval=0.1):
 616         self.__is_shut_down.clear()
 617         try:
 618             while not self.__shutdown_request:
 619                 r, w_, e_ = select.select([self.socket], [], [], poll_interval)
 620                 if self.socket in r:
 621                     connection, client_address = self.socket.accept()
 622                     t = basethread.BaseThread(
 623                         "TCPConnectionHandler (%s: %s:%s -> %s:%s)" % (
 624                             self.__class__.__name__,
 625                             client_address[0],
 626                             client_address[1],
 627                             self.address[0],
 628                             self.address[1],
 629                         ),
 630                         target=self.connection_thread,
 631                         args=(connection, client_address),
 632                     )
 633                     t.setDaemon(1)
 634                     try:
 635                         t.start()
 636                     except threading.ThreadError:
 637                         self.handle_error(connection, client_address)
 638                         connection.close()
 639         finally:
 640             self.__shutdown_request = False
 641             self.__is_shut_down.set()
 642 
 643     def shutdown(self):
 644         self.__shutdown_request = True
 645         self.__is_shut_down.wait()
 646         self.socket.close()
 647         self.handle_shutdown()
