   4 class Masker:
   5     """
   6     Data sent from the server must be masked to prevent malicious clients
   7     from sending data over the wire in predictable patterns.
   8 
   9     Servers do not have to mask data they send to the client.
  10     https://tools.ietf.org/html/rfc6455#section-5.3
  11     """
  12 
  13     def __init__(self, key):
  14         self.key = key
  15         self.offset = 0
  16 
  17     def mask(self, offset, data):
  18         datalen = len(data)
  19         offset_mod = offset % 4
  20         data = int.from_bytes(data, sys.byteorder)
  21         num_keys = (datalen + offset_mod + 3) // 4
  22         mask = int.from_bytes((self.key * num_keys)[offset_mod:datalen +
  23                                                     offset_mod], sys.byteorder)
  24         return (data ^ mask).to_bytes(datalen, sys.byteorder)
  25 
  26     def __call__(self, data):
  27         ret = self.mask(self.offset, data)
  28         self.offset += len(ret)
  29         return ret
