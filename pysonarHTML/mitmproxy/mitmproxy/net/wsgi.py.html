  31 def date_time_string():
  32     """Return the current date and time formatted for a message header."""
  33     WEEKS = ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun']
  34     MONTHS = [
  35         None,
  36         'Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun',
  37         'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'
  38     ]
  39     now = time.time()
  40     year, month, day, hh, mm, ss, wd, y_, z_ = time.gmtime(now)
  41     s = "%s, %02d %3s %4d %02d:%02d:%02d GMT" % (
  42         WEEKS[wd],
  43         day, MONTHS[month], year,
  44         hh, mm, ss
  45     )
  46     return s

  54     def make_environ(self, flow, errsoc, **extra):
  55         """
  56         Raises:
  57             ValueError, if the content-encoding is invalid.
  58         """
  59         path = strutils.always_str(flow.request.path, "latin-1")
  60         if '?' in path:
  61             path_info, query = strutils.always_str(path, "latin-1").split('?', 1)
  62         else:
  63             path_info = path
  64             query = ''
  65         environ = {
  66             'wsgi.version': (1, 0),
  67             'wsgi.url_scheme': strutils.always_str(flow.request.scheme, "latin-1"),
  68             'wsgi.input': io.BytesIO(flow.request.content or b""),
  69             'wsgi.errors': errsoc,
  70             'wsgi.multithread': True,
  71             'wsgi.multiprocess': False,
  72             'wsgi.run_once': False,
  73             'SERVER_SOFTWARE': self.sversion,
  74             'REQUEST_METHOD': strutils.always_str(flow.request.method, "latin-1"),
  75             'SCRIPT_NAME': '',
  76             'PATH_INFO': urllib.parse.unquote(path_info),
  77             'QUERY_STRING': query,
  78             'CONTENT_TYPE': strutils.always_str(flow.request.headers.get('Content-Type', ''), "latin-1"),
  79             'CONTENT_LENGTH': strutils.always_str(flow.request.headers.get('Content-Length', ''), "latin-1"),
  80             'SERVER_NAME': self.domain,
  81             'SERVER_PORT': str(self.port),
  82             'SERVER_PROTOCOL': strutils.always_str(flow.request.http_version, "latin-1"),
  83         }
  84         environ.update(extra)
  85         if flow.client_conn.address:
  86             environ["REMOTE_ADDR"] = strutils.always_str(flow.client_conn.address[0], "latin-1")
  87             environ["REMOTE_PORT"] = flow.client_conn.address[1]
  88 
  89         for key, value in flow.request.headers.items():
  90             key = 'HTTP_' + strutils.always_str(key, "latin-1").upper().replace('-', '_')
  91             if key not in ('HTTP_CONTENT_TYPE', 'HTTP_CONTENT_LENGTH'):
  92                 environ[key] = value
  93         return environ

  95     def error_page(self, soc, headers_sent, s):
  96         """
  97             Make a best-effort attempt to write an error page. If headers are
  98             already sent, we just bung the error into the page.
  99         """
 100         c = """
 101             <html>
 102                 <h1>Internal Server Error</h1>
 103                 <pre>{err}"</pre>
 104             </html>
 105         """.format(err=s).strip().encode()
 106 
 107         if not headers_sent:
 108             soc.write(b"HTTP/1.1 500 Internal Server Error\r\n")
 109             soc.write(b"Content-Type: text/html\r\n")
 110             soc.write("Content-Length: {length}\r\n".format(length=len(c)).encode())
 111             soc.write(b"\r\n")
 112         soc.write(c)

 114     def serve(self, request, soc, **env):
 115         state = dict(
 116             response_started=False,
 117             headers_sent=False,
 118             status=None,
 119             headers=None
 120         )
 121 
 122         def write(data):
 123             if not state["headers_sent"]:
 124                 soc.write("HTTP/1.1 {status}\r\n".format(status=state["status"]).encode())
 125                 headers = state["headers"]
 126                 if 'server' not in headers:
 127                     headers["Server"] = self.sversion
 128                 if 'date' not in headers:
 129                     headers["Date"] = date_time_string()
 130                 soc.write(bytes(headers))
 131                 soc.write(b"\r\n")
 132                 state["headers_sent"] = True
 133             if data:
 134                 soc.write(data)
 135             soc.flush()
 136 
 137         def start_response(status, headers, exc_info=None):
 138             if exc_info:
 139                 if state["headers_sent"]:
 140                     raise exc_info[1]
 141             elif state["status"]:
 142                 raise AssertionError('Response already started')
 143             state["status"] = status
 144             state["headers"] = http.Headers([[strutils.always_bytes(k), strutils.always_bytes(v)] for k, v in headers])
 145             if exc_info:
 146                 self.error_page(soc, state["headers_sent"], traceback.format_tb(exc_info[2]))
 147                 state["headers_sent"] = True
 148 
 149         errs = io.BytesIO()
 150         try:
 151             dataiter = self.app(
 152                 self.make_environ(request, errs, **env), start_response
 153             )
 154             for i in dataiter:
 155                 write(i)
 156             if not state["headers_sent"]:
 157                 write(b"")
 158         except Exception:
 159             try:
 160                 s = traceback.format_exc()
 161                 errs.write(s.encode("utf-8", "replace"))
 162                 self.error_page(soc, state["headers_sent"], s)
 163             except Exception:    # pragma: no cover
 164                 pass
 165         return errs.getvalue()
