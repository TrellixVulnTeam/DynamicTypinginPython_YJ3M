   5 """
   6     Doing this the "right" way by using DIOCNATLOOK on the pf device turns out
   7     to be a pain. Apple has made a number of modifications to the data
   8     structures returned, and compiling userspace tools to test and work with
   9     this turns out to be a pain in the ass. Parsing pfctl output is short,
  10     simple, and works.
  11 
  12     Note: Also Tested with FreeBSD 10 pkgng Python 2.7.x.
  13     Should work almost exactly as on Mac OS X and except with some changes to
  14     the output processing of pfctl (see pf.py).
  15 """

  17 STATECMD = ("sudo", "-n", "/sbin/pfctl", "-s", "state")

  20 def original_addr(csock):
  21     peer = csock.getpeername()
  22     try:
  23         stxt = subprocess.check_output(STATECMD, stderr=subprocess.STDOUT)
  24     except subprocess.CalledProcessError as e:
  25         if "sudo: a password is required" in e.output.decode(errors="replace"):
  26             insufficient_priv = True
  27         else:
  28             raise RuntimeError("Error getting pfctl state: " + repr(e))
  29     else:
  30         insufficient_priv = "sudo: a password is required" in stxt.decode(errors="replace")
  31 
  32     if insufficient_priv:
  33         raise RuntimeError(
  34             "Insufficient privileges to access pfctl. "
  35             "See https://mitmproxy.org/docs/latest/howto-transparent/#macos for details.")
  36     return pf.lookup(peer[0], peer[1], stxt)
