  37 class Resolver:
  38     sock: socket.socket
  39     lock: threading.RLock
  40 
  41     def __init__(self):
  42         self.sock = None
  43         self.lock = threading.RLock()
  44 
  45     def setup(self):
  46         with self.lock:
  47             TransparentProxy.setup()
  48             self._connect()
  49 
  50     def _connect(self):
  51         if self.sock:
  52             self.sock.close()
  53         self.sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
  54         self.sock.connect((REDIRECT_API_HOST, REDIRECT_API_PORT))
  55 
  56         self.wfile = self.sock.makefile('wb')
  57         self.rfile = self.sock.makefile('rb')
  58         write(os.getpid(), self.wfile)
  59 
  60     def original_addr(self, csock: socket.socket):
  61         ip, port = csock.getpeername()[:2]
  62         ip = re.sub(r"^::ffff:(?=\d+.\d+.\d+.\d+$)", "", ip)
  63         ip = ip.split("%", 1)[0]
  64         with self.lock:
  65             try:
  66                 write((ip, port), self.wfile)
  67                 addr = read(self.rfile)
  68                 if addr is None:
  69                     raise RuntimeError("Cannot resolve original destination.")
  70                 return tuple(addr)
  71             except (EOFError, socket.error):
  72                 self._connect()
  73                 return self.original_addr(csock)

  76 class APIRequestHandler(socketserver.StreamRequestHandler):
  77     """
  78     TransparentProxy API: Returns the pickled server address, port tuple
  79     for each received pickled client address, port tuple.
  80     """
  81 
  82     def handle(self):
  83         proxifier: TransparentProxy = self.server.proxifier
  84         try:
  85             pid: int = read(self.rfile)
  86             with proxifier.exempt(pid):
  87                 while True:
  88                     client = tuple(read(self.rfile))
  89                     try:
  90                         server = proxifier.client_server_map[client]
  91                     except KeyError:
  92                         server = None
  93                     write(server, self.wfile)
  94         except (EOFError, socket.error):
  95             pass

 175 class TcpConnectionTable(collections.abc.Mapping):
 176     DEFAULT_TABLE_SIZE = 4096
 177 
 178     def __init__(self):
 179         self._tcp = MIB_TCPTABLE_OWNER_PID(self.DEFAULT_TABLE_SIZE)
 180         self._tcp_size = ctypes.wintypes.DWORD(self.DEFAULT_TABLE_SIZE)
 181         self._tcp6 = MIB_TCP6TABLE_OWNER_PID(self.DEFAULT_TABLE_SIZE)
 182         self._tcp6_size = ctypes.wintypes.DWORD(self.DEFAULT_TABLE_SIZE)
 183         self._map = {}
 184 
 185     def __getitem__(self, item):
 186         return self._map[item]
 187 
 188     def __iter__(self):
 189         return self._map.__iter__()
 190 
 191     def __len__(self):
 192         return self._map.__len__()
 193 
 194     def refresh(self):
 195         self._map = {}
 196         self._refresh_ipv4()
 197         self._refresh_ipv6()
 198 
 199     def _refresh_ipv4(self):
 200         ret = ctypes.windll.iphlpapi.GetExtendedTcpTable(
 201             ctypes.byref(self._tcp),
 202             ctypes.byref(self._tcp_size),
 203             False,
 204             socket.AF_INET,
 205             TCP_TABLE_OWNER_PID_CONNECTIONS,
 206             0
 207         )
 208         if ret == 0:
 209             for row in self._tcp.table[:self._tcp.dwNumEntries]:
 210                 local_ip = socket.inet_ntop(socket.AF_INET, bytes(row.ucLocalAddr))
 211                 local_port = socket.htons(row.dwLocalPort)
 212                 self._map[(local_ip, local_port)] = row.dwOwningPid
 213         elif ret == ERROR_INSUFFICIENT_BUFFER:
 214             self._tcp = MIB_TCPTABLE_OWNER_PID(self._tcp_size.value)
 215             # no need to update size, that's already done.
 216             self._refresh_ipv4()
 217         else:
 218             raise RuntimeError("[IPv4] Unknown GetExtendedTcpTable return code: %s" % ret)
 219 
 220     def _refresh_ipv6(self):
 221         ret = ctypes.windll.iphlpapi.GetExtendedTcpTable(
 222             ctypes.byref(self._tcp6),
 223             ctypes.byref(self._tcp6_size),
 224             False,
 225             socket.AF_INET6,
 226             TCP_TABLE_OWNER_PID_CONNECTIONS,
 227             0
 228         )
 229         if ret == 0:
 230             for row in self._tcp6.table[:self._tcp6.dwNumEntries]:
 231                 local_ip = socket.inet_ntop(socket.AF_INET6, bytes(row.ucLocalAddr))
 232                 local_port = socket.htons(row.dwLocalPort)
 233                 self._map[(local_ip, local_port)] = row.dwOwningPid
 234         elif ret == ERROR_INSUFFICIENT_BUFFER:
 235             self._tcp6 = MIB_TCP6TABLE_OWNER_PID(self._tcp6_size.value)
 236             # no need to update size, that's already done.
 237             self._refresh_ipv6()
 238         else:
 239             raise RuntimeError("[IPv6] Unknown GetExtendedTcpTable return code: %s" % ret)

 242 def get_local_ip() -> typing.Optional[str]:
 243     # Auto-Detect local IP. This is required as re-injecting to 127.0.0.1 does not work.
 244     # https://stackoverflow.com/questions/166506/finding-local-ip-addresses-using-pythons-stdlib
 245     s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
 246     try:
 247         s.connect(("8.8.8.8", 80))
 248         return s.getsockname()[0]
 249     except OSError:
 250         return None
 251     finally:
 252         s.close()
 253

 255 def get_local_ip6(reachable: str) -> typing.Optional[str]:
 256     # The same goes for IPv6, with the added difficulty that .connect() fails if
 257     # the target network is not reachable.
 258     s = socket.socket(socket.AF_INET6, socket.SOCK_DGRAM)
 259     try:
 260         s.connect((reachable, 80))
 261         return s.getsockname()[0]
 262     except OSError:
 263         return None
 264     finally:
 265         s.close()

 268 class Redirect(threading.Thread):
 269     daemon = True
 270     windivert: pydivert.WinDivert
 271 
 272     def __init__(
 273         self,
 274         handle: typing.Callable[[pydivert.Packet], None],
 275         filter: str,
 276         layer: pydivert.Layer = pydivert.Layer.NETWORK,
 277         flags: pydivert.Flag = 0
 278     ) -> None:
 279         self.handle = handle
 280         self.windivert = pydivert.WinDivert(filter, layer, flags=flags)
 281         super().__init__()
 282 
 283     def start(self):
 284         self.windivert.open()
 285         super().start()
 286 
 287     def run(self):
 288         while True:
 289             try:
 290                 packet = self.windivert.recv()
 291             except WindowsError as e:
 292                 if e.winerror == 995:
 293                     return
 294                 else:
 295                     raise
 296             else:
 297                 self.handle(packet)
 298 
 299     def shutdown(self):
 300         self.windivert.close()
 301 
 302     def recv(self) -> typing.Optional[pydivert.Packet]:
 303         """
 304         Convenience function that receives a packet from the passed handler and handles error codes.
 305         If the process has been shut down, None is returned.
 306         """
 307         try:
 308             return self.windivert.recv()
 309         except WindowsError as e:
 310             if e.winerror == 995:
 311                 return None
 312             else:
 313                 raise

 316 class RedirectLocal(Redirect):
 317     trusted_pids: typing.Set[int]
 318 
 319     def __init__(
 320         self,
 321         redirect_request: typing.Callable[[pydivert.Packet], None],
 322         filter: str
 323     ) -> None:
 324         self.tcp_connections = TcpConnectionTable()
 325         self.trusted_pids = set()
 326         self.redirect_request = redirect_request
 327         super().__init__(self.handle, filter)
 328 
 329     def handle(self, packet):
 330         client = (packet.src_addr, packet.src_port)
 331 
 332         if client not in self.tcp_connections:
 333             self.tcp_connections.refresh()
 334 
 335         # If this fails, we most likely have a connection from an external client.
 336         # In this, case we always want to proxy the request.
 337         pid = self.tcp_connections.get(client, None)
 338 
 339         if pid not in self.trusted_pids:
 340             self.redirect_request(packet)
 341         else:
 342             # It's not really clear why we need to recalculate the checksum here,
 343             # but this was identified as necessary in https://github.com/mitmproxy/mitmproxy/pull/3174.
 344             self.windivert.send(packet, recalculate_checksum=True)

 350 class ClientServerMap:
 351     """A thread-safe LRU dict."""
 352     connection_cache_size: typing.ClassVar[int] = 65536
 353 
 354     def __init__(self):
 355         self._lock = threading.Lock()
 356         self._map = collections.OrderedDict()
 357 
 358     def __getitem__(self, item: TConnection) -> TConnection:
 359         with self._lock:
 360             return self._map[item]
 361 
 362     def __setitem__(self, key: TConnection, value: TConnection) -> None:
 363         with self._lock:
 364             self._map[key] = value
 365             self._map.move_to_end(key)
 366             while len(self._map) > self.connection_cache_size:
 367                 self._map.popitem(False)

 370 class TransparentProxy:
 371     """
 372     Transparent Windows Proxy for mitmproxy based on WinDivert/PyDivert. This module can be used to
 373     redirect both traffic that is forwarded by the host and traffic originating from the host itself.
 374 
 375     Requires elevated (admin) privileges. Can be started separately by manually running the file.
 376 
 377     How it works:
 378 
 379     (1) First, we intercept all packages that match our filter.
 380     We both consider traffic that is forwarded by the OS (WinDivert's NETWORK_FORWARD layer) as well
 381     as traffic sent from the local machine (WinDivert's NETWORK layer). In the case of traffic from
 382     the local machine, we need to exempt packets sent from the proxy to not create a redirect loop.
 383     To accomplish this, we use Windows' GetExtendedTcpTable syscall and determine the source
 384     application's PID.
 385 
 386     For each intercepted package, we
 387         1. Store the source -> destination mapping (address and port)
 388         2. Remove the package from the network (by not reinjecting it).
 389         3. Re-inject the package into the local network stack, but with the destination address
 390            changed to the proxy.
 391 
 392     (2) Next, the proxy receives the forwarded packet, but does not know the real destination yet
 393     (which we overwrote with the proxy's address). On Linux, we would now call
 394     getsockopt(SO_ORIGINAL_DST). We now access the redirect module's API (see APIRequestHandler),
 395     submit the source information and get the actual destination back (which we stored in 1.1).
 396 
 397     (3) The proxy now establishes the upstream connection as usual.
 398 
 399     (4) Finally, the proxy sends the response back to the client. To make it work, we need to change
 400     the packet's source address back to the original destination (using the mapping from 1.1),
 401     to which the client believes it is talking to.
 402 
 403     Limitations:
 404 
 405     - We assume that ephemeral TCP ports are not re-used for multiple connections at the same time.
 406     The proxy will fail if an application connects to example.com and example.org from
 407     192.168.0.42:4242 simultaneously. This could be mitigated by introducing unique "meta-addresses"
 408     which mitmproxy sees, but this would remove the correct client info from mitmproxy.
 409     """
 410     local: typing.Optional[RedirectLocal] = None
 411     # really weird linting error here.
 412     forward: typing.Optional[Redirect] = None  # noqa
 413     response: Redirect
 414     icmp: Redirect
 415 
 416     proxy_port: int
 417     filter: str
 418 
 419     client_server_map: ClientServerMap
 420 
 421     def __init__(
 422         self,
 423         local: bool = True,
 424         forward: bool = True,
 425         proxy_port: int = 8080,
 426         filter: typing.Optional[str] = "tcp.DstPort == 80 or tcp.DstPort == 443",
 427     ) -> None:
 428         self.proxy_port = proxy_port
 429         self.filter = (
 430             filter
 431             or
 432             f"tcp.DstPort != {proxy_port} and tcp.DstPort != {REDIRECT_API_PORT} and tcp.DstPort < 49152"
 433         )
 434 
 435         self.ipv4_address = get_local_ip()
 436         self.ipv6_address = get_local_ip6("2001:4860:4860::8888")
 437         # print(f"IPv4: {self.ipv4_address}, IPv6: {self.ipv6_address}")
 438         self.client_server_map = ClientServerMap()
 439 
 440         self.api = APIServer(self, (REDIRECT_API_HOST, REDIRECT_API_PORT), APIRequestHandler)
 441         self.api_thread = threading.Thread(target=self.api.serve_forever)
 442         self.api_thread.daemon = True
 443 
 444         if forward:
 445             self.forward = Redirect(
 446                 self.redirect_request,
 447                 self.filter,
 448                 pydivert.Layer.NETWORK_FORWARD
 449             )
 450         if local:
 451             self.local = RedirectLocal(
 452                 self.redirect_request,
 453                 self.filter
 454             )
 455 
 456         # The proxy server responds to the client. To the client,
 457         # this response should look like it has been sent by the real target
 458         self.response = Redirect(
 459             self.redirect_response,
 460             f"outbound and tcp.SrcPort == {proxy_port}",
 461         )
 462 
 463         # Block all ICMP requests (which are sent on Windows by default).
 464         # If we don't do this, our proxy machine may send an ICMP redirect to the client,
 465         # which instructs the client to directly connect to the real gateway
 466         # if they are on the same network.
 467         self.icmp = Redirect(
 468             lambda _: None,
 469             "icmp",
 470             flags=pydivert.Flag.DROP
 471         )
 472 
 473     @classmethod
 474     def setup(cls):
 475         # TODO: Make sure that server can be killed cleanly. That's a bit difficult as we don't have access to
 476         # controller.should_exit when this is called.
 477         s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
 478         server_unavailable = s.connect_ex((REDIRECT_API_HOST, REDIRECT_API_PORT))
 479         if server_unavailable:
 480             proxifier = TransparentProxy()
 481             proxifier.start()
 482 
 483     def start(self):
 484         self.api_thread.start()
 485         self.icmp.start()
 486         self.response.start()
 487         if self.forward:
 488             self.forward.start()
 489         if self.local:
 490             self.local.start()
 491 
 492     def shutdown(self):
 493         if self.local:
 494             self.local.shutdown()
 495         if self.forward:
 496             self.forward.shutdown()
 497         self.response.shutdown()
 498         self.icmp.shutdown()
 499         self.api.shutdown()
 500 
 501     def redirect_request(self, packet: pydivert.Packet):
 502         # print(" * Redirect client -> server to proxy")
 503         # print(f"{packet.src_addr}:{packet.src_port} -> {packet.dst_addr}:{packet.dst_port}")
 504         client = (packet.src_addr, packet.src_port)
 505 
 506         self.client_server_map[client] = (packet.dst_addr, packet.dst_port)
 507 
 508         # We do need to inject to an external IP here, 127.0.0.1 does not work.
 509         if packet.address_family == socket.AF_INET:
 510             assert self.ipv4_address
 511             packet.dst_addr = self.ipv4_address
 512         elif packet.address_family == socket.AF_INET6:
 513             if not self.ipv6_address:
 514                 self.ipv6_address = get_local_ip6(packet.src_addr)
 515             assert self.ipv6_address
 516             packet.dst_addr = self.ipv6_address
 517         else:
 518             raise RuntimeError("Unknown address family")
 519         packet.dst_port = self.proxy_port
 520         packet.direction = pydivert.consts.Direction.INBOUND
 521 
 522         # We need a handle on the NETWORK layer. the local handle is not guaranteed to exist,
 523         # so we use the response handle.
 524         self.response.windivert.send(packet)
 525 
 526     def redirect_response(self, packet: pydivert.Packet):
 527         """
 528         If the proxy responds to the client, let the client believe the target server sent the
 529         packets.
 530         """
 531         # print(" * Adjust proxy -> client")
 532         client = (packet.dst_addr, packet.dst_port)
 533         try:
 534             packet.src_addr, packet.src_port = self.client_server_map[client]
 535         except KeyError:
 536             print(f"Warning: Previously unseen connection from proxy to {client}")
 537         else:
 538             packet.recalculate_checksums()
 539 
 540         self.response.windivert.send(packet, recalculate_checksum=False)
 541 
 542     @contextlib.contextmanager
 543     def exempt(self, pid: int):
 544         if self.local:
 545             self.local.trusted_pids.add(pid)
 546         try:
 547             yield
 548         finally:
 549             if self.local:
 550                 self.local.trusted_pids.remove(pid)
