  97 def tflow(client_conn=True, server_conn=True, req=True, resp=None, err=None):
  98     """
  99     @type client_conn: bool | None | mitmproxy.proxy.connection.ClientConnection

 102     @type resp:        bool | None | mitmproxy.proxy.protocol.http.HTTPResponse
 103     @type err:         bool | None | mitmproxy.proxy.protocol.primitives.Error

 123     f.request = req
 124     f.response = resp
 125     f.error = err
