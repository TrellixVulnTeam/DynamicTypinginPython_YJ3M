  23 class RecordingMaster(mitmproxy.master.Master):
  24     def __init__(self, *args, **kwargs):
  25         super().__init__(*args, **kwargs)
  26         self.addons = TestAddons(self)
  27         self.logs = []
  28 
  29     def dump_log(self, outf=sys.stdout):
  30         for i in self.logs:
  31             print("%s: %s" % (i.level, i.msg), file=outf)
  32 
  33     def has_log(self, txt, level=None):
  34         for i in self.logs:
  35             if level and i.level != level:
  36                 continue
  37             if txt.lower() in i.msg.lower():
  38                 return True
  39         return False
  40 
  41     async def await_log(self, txt, level=None):
  42         for i in range(20):
  43             if self.has_log(txt, level):
  44                 return True
  45             else:
  46                 await asyncio.sleep(0.1)
  47         return False
  48 
  49     def clear(self):
  50         self.logs = []

  53 class context:
  54     """
  55         A context for testing addons, which sets up the mitmproxy.ctx module so
  56         handlers can run as they would within mitmproxy. The context also
  57         provides a number of helper methods for common testing scenarios.
  58     """
  59 
  60     def __init__(self, *addons, options=None, loadcore=True):
  61         options = options or mitmproxy.options.Options()
  62         self.master = RecordingMaster(
  63             options
  64         )
  65         self.options = self.master.options
  66 
  67         if loadcore:
  68             self.master.addons.add(core.Core())
  69 
  70         for a in addons:
  71             self.master.addons.add(a)
  72 
  73     def __enter__(self):
  74         return self
  75 
  76     def __exit__(self, exc_type, exc_value, traceback):
  77         return False
  78 
  79     @contextlib.contextmanager
  80     def cycle(self, addon, f):
  81         """
  82             Cycles the flow through the events for the flow. Stops if a reply
  83             is taken (as in flow interception).
  84         """
  85         f.reply._state = "start"
  86         for evt, arg in eventsequence.iterate(f):
  87             self.master.addons.invoke_addon(
  88                 addon,
  89                 evt,
  90                 arg
  91             )
  92             if f.reply.state == "taken":
  93                 return
  94 
  95     def configure(self, addon, **kwargs):
  96         """
  97             A helper for testing configure methods. Modifies the registered
  98             Options object with the given keyword arguments, then calls the
  99             configure method on the addon with the updated value.
 100         """
 101         if addon not in self.master.addons:
 102             self.master.addons.register(addon)
 103         with self.options.rollback(kwargs.keys(), reraise=True):
 104             if kwargs:
 105                 self.options.update(**kwargs)
 106             else:
 107                 self.master.addons.invoke_addon(addon, "configure", {})
 108 
 109     def script(self, path):
 110         """
 111             Loads a script from path, and returns the enclosed addon.
 112         """
 113         sc = script.Script(path, False)
 114         return sc.addons[0] if sc.addons else None
 115 
 116     def invoke(self, addon, event, *args, **kwargs):
 117         """
 118             Recursively invoke an event on an addon and all its children.
 119         """
 120         return self.master.addons.invoke_addon(addon, event, *args, **kwargs)
 121 
 122     def command(self, func, *args):
 123         """
 124             Invoke a command function with a list of string arguments within a command context, mimicking the actual command environment.
 125         """
 126         cmd = command.Command(self.master.commands, "test.command", func)
 127         return cmd.call(args)
