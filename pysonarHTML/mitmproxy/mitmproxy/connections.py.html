  15 class ClientConnection(tcp.BaseHandler, stateobject.StateObject):
  16     """
  17     A client connection
  18 
  19     Attributes:
  20         address: Remote address
  21         tls_established: True if TLS is established, False otherwise
  22         clientcert: The TLS client certificate
  23         mitmcert: The MITM'ed TLS server certificate presented to the client
  24         timestamp_start: Connection start timestamp
  25         timestamp_tls_setup: TLS established timestamp
  26         timestamp_end: Connection end timestamp
  27         sni: Server Name Indication sent by client during the TLS handshake
  28         cipher_name: The current used cipher
  29         alpn_proto_negotiated: The negotiated application protocol
  30         tls_version: TLS version
  31         tls_extensions: TLS ClientHello extensions
  32     """

  34     def __init__(self, client_connection, address, server):
  35         # Eventually, this object is restored from state. We don't have a
  36         # connection then.
  37         if client_connection:
  38             super().__init__(client_connection, address, server)
  39         else:
  40             self.connection = None
  41             self.server = None
  42             self.wfile = None
  43             self.rfile = None
  44             self.address = None
  45             self.clientcert = None
  46             self.tls_established = None
  47 
  48         self.id = str(uuid.uuid4())
  49         self.mitmcert = None
  50         self.timestamp_start = time.time()
  51         self.timestamp_end = None
  52         self.timestamp_tls_setup = None
  53         self.sni = None
  54         self.cipher_name = None
  55         self.alpn_proto_negotiated = None
  56         self.tls_version = None
  57         self.tls_extensions = None

  59     def connected(self):

  62     def __repr__(self):
  63         if self.tls_established:
  64             tls = "[{}] ".format(self.tls_version)
  65         else:
  66             tls = ""
  67 
  68         if self.alpn_proto_negotiated:

 135     def convert_to_tls(self, cert, *args, **kwargs):
 136         # Unfortunately OpenSSL provides no way to expose all TLS extensions, so we do this dance
 137         # here and use our Kaitai parser.
 138         try:
 139             client_hello = tls.ClientHello.from_file(self.rfile)
 140         except exceptions.TlsProtocolException:  # pragma: no cover
 141             pass  # if this fails, we don't want everything to go down.
 142         else:
 143             self.tls_extensions = client_hello.extensions
 144 
 145         super().convert_to_tls(cert, *args, **kwargs)
 146         self.timestamp_tls_setup = time.time()
 147         self.mitmcert = cert
 148         sni = self.connection.get_servername()
 149         if sni:
 150             self.sni = sni.decode("idna")
 151         else:
 152             self.sni = None
 153         self.cipher_name = self.connection.get_cipher_name()
 154         self.alpn_proto_negotiated = self.get_alpn_proto_negotiated()
 155         self.tls_version = self.connection.get_protocol_version_name()
 156 
 157     def finish(self):
 158         super().finish()
 159         self.timestamp_end = time.time()

 162 class ServerConnection(tcp.TCPClient, stateobject.StateObject):
 163     """
 164     A server connection
 165 
 166     Attributes:
 167         address: Remote address. Can be both a domain or an IP address.
 168         ip_address: Resolved remote IP address.
 169         source_address: Local IP address or client's source IP address.
 170         tls_established: True if TLS is established, False otherwise
 171         cert: The certificate presented by the remote during the TLS handshake
 172         sni: Server Name Indication sent by the proxy during the TLS handshake
 173         alpn_proto_negotiated: The negotiated application protocol
 174         tls_version: TLS version
 175         via: The underlying server connection (e.g. the connection to the upstream proxy in upstream proxy mode)
 176         timestamp_start: Connection start timestamp
 177         timestamp_tcp_setup: TCP ACK received timestamp
 178         timestamp_tls_setup: TLS established timestamp
 179         timestamp_end: Connection end timestamp
 180     """
 181 
 182     def __init__(self, address, source_address=None, spoof_source_address=None):
 183         tcp.TCPClient.__init__(self, address, source_address, spoof_source_address)
 184 
 185         self.id = str(uuid.uuid4())
 186         self.alpn_proto_negotiated = None
 187         self.tls_version = None
 188         self.via = None
 189         self.timestamp_start = None
 190         self.timestamp_end = None
 191         self.timestamp_tcp_setup = None
 192         self.timestamp_tls_setup = None
 193 
 194     def connected(self):
 195         return bool(self.connection) and not self.finished
 196 
 197     def __repr__(self):
 198         if self.tls_established and self.sni:
 199             tls = "[{}: {}] ".format(self.tls_version or "TLS", self.sni)
 200         elif self.tls_established:
 201             tls = "[{}] ".format(self.tls_version or "TLS")
 202         else:
 203             tls = ""
 204         if self.alpn_proto_negotiated:
 205             alpn = "[ALPN: {}] ".format(
 206                 strutils.bytes_to_escaped_str(self.alpn_proto_negotiated)
 207             )
 208         else:
 209             alpn = ""
 210         return "<ServerConnection: {tls}{alpn}{address}>".format(
 211             tls=tls,
 212             alpn=alpn,
 213             address=human.format_address(self.address),
 214         )
 215 
 216     def __eq__(self, other):
 217         if isinstance(other, ServerConnection):
 218             return self.id == other.id
 219         return False
 220 
 221     def __hash__(self):
 222         return hash(self.id)

 276     def establish_tls(self, *, sni=None, client_certs=None, **kwargs):
 277         if sni and not isinstance(sni, str):
 278             raise ValueError("sni must be str, not " + type(sni).__name__)
 279         client_cert = None
 280         if client_certs:
 281             client_certs = os.path.expanduser(client_certs)
 282             if os.path.isfile(client_certs):
 283                 client_cert = client_certs
 284             else:
 285                 path = os.path.join(
 286                     client_certs,
 287                     (sni or self.address[0].encode("idna").decode()) + ".pem"
 288                 )
 289                 if os.path.exists(path):
 290                     client_cert = path
 291 
 292         self.convert_to_tls(cert=client_cert, sni=sni, **kwargs)
 293         self.sni = sni
 294         self.alpn_proto_negotiated = self.get_alpn_proto_negotiated()
 295         self.tls_version = self.connection.get_protocol_version_name()
 296         self.timestamp_tls_setup = time.time()
 297 
 298     def finish(self):
 299         tcp.TCPClient.finish(self)
 300         self.timestamp_end = time.time()
