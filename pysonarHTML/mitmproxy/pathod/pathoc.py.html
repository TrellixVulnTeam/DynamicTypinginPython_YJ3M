  44     def __str__(self):
  45         parts = [
  46             "Application Layer Protocol: %s" % strutils.always_str(self.alp, "utf8"),
  47             "Cipher: %s, %s bit, %s" % self.cipher,
  48             "SSL certificate chain:"
  49         ]
  50         for n, i in enumerate(self.certchain):
  51             parts.append("  Certificate [%s]" % n)
  52             parts.append("\tSubject: ")
  53             for cn in i.get_subject().get_components():
  54                 parts.append("\t\t%s=%s" % (
  55                     strutils.always_str(cn[0], "utf8"),
  56                     strutils.always_str(cn[1], "utf8"))
  57                 )
  58             parts.append("\tIssuer: ")
  59             for cn in i.get_issuer().get_components():
  60                 parts.append("\t\t%s=%s" % (
  61                     strutils.always_str(cn[0], "utf8"),
  62                     strutils.always_str(cn[1], "utf8"))
  63                 )
  64             parts.extend(
  65                 [
  66                     "\tVersion: %s" % i.get_version(),
  67                     "\tValidity: %s - %s" % (
  68                         strutils.always_str(i.get_notBefore(), "utf8"),
  69                         strutils.always_str(i.get_notAfter(), "utf8")
  70                     ),
  71                     "\tSerial: %s" % i.get_serial_number(),
  72                     "\tAlgorithm: %s" % strutils.always_str(i.get_signature_algorithm(), "utf8")
  73                 ]
  74             )
  75             pk = i.get_pubkey()
  76             types = {
  77                 OpenSSL.crypto.TYPE_RSA: "RSA",
  78                 OpenSSL.crypto.TYPE_DSA: "DSA"
  79             }
  80             t = types.get(pk.type(), "Uknown")
  81             parts.append("\tPubkey: %s bit %s" % (pk.bits(), t))
  82             s = certs.Cert(i)
  83             if s.altnames:
  84                 parts.append("\tSANs: %s" % " ".join(strutils.always_str(n, "utf8") for n in s.altnames))
  85         return "\n".join(parts)

  88 class WebsocketFrameReader(basethread.BaseThread):
  89 
  90     def __init__(
  91             self,
  92             rfile,
  93             logfp,
  94             showresp,
  95             hexdump,
  96             ws_read_limit,
  97             timeout
  98     ):
  99         basethread.BaseThread.__init__(self, "WebsocketFrameReader")
 100         self.timeout = timeout
 101         self.ws_read_limit = ws_read_limit
 102         self.logfp = logfp
 103         self.showresp = showresp
 104         self.hexdump = hexdump
 105         self.rfile = rfile
 106         self.terminate = queue.Queue()
 107         self.frames_queue = queue.Queue()
 108         self.logger = log.ConnectionLogger(
 109             self.logfp,
 110             self.hexdump,
 111             False,
 112             rfile if showresp else None,
 113             None
 114         )
 115 
 116     @contextlib.contextmanager
 117     def terminator(self):
 118         yield
 119         self.frames_queue.put(None)
 120 
 121     def run(self):
 122         starttime = time.time()
 123         with self.terminator():
 124             while True:
 125                 if self.ws_read_limit == 0:
 126                     return
 127                 try:
 128                     r, _, _ = select.select([self.rfile], [], [], 0.05)
 129                 except OSError:  # pragma: no cover
 130                     return  # this is not reliably triggered due to its nature, so we exclude it from coverage.
 131                 delta = time.time() - starttime
 132                 if not r and self.timeout and delta > self.timeout:
 133                     return
 134                 try:
 135                     self.terminate.get_nowait()
 136                     return
 137                 except queue.Empty:
 138                     pass
 139                 for rfile in r:
 140                     with self.logger.ctx() as log:
 141                         try:
 142                             frm = websockets.Frame.from_file(self.rfile)
 143                         except exceptions.TcpDisconnect:
 144                             return
 145                         self.frames_queue.put(frm)
 146                         log("<< %s" % repr(frm.header))
 147                         if self.ws_read_limit is not None:
 148                             self.ws_read_limit -= 1
 149                         starttime = time.time()

 238     def http_connect(self, connect_to):
 239         req = net_http.Request(
 240             first_line_format='authority',
 241             method='CONNECT',
 242             scheme=None,
 243             host=connect_to[0].encode("idna"),
 244             port=connect_to[1],
 245             path=None,
 246             http_version='HTTP/1.1',
 247             headers=[(b"Host", connect_to[0].encode("idna"))],
 248             content=b'',
 249         )
 250         self.wfile.write(net_http.http1.assemble_request(req))
 251         self.wfile.flush()
 252         try:
 253             resp = self.protocol.read_response(self.rfile, req)
 254             if resp.status_code != 200:
 255                 raise exceptions.HttpException("Unexpected status code: %s" % resp.status_code)
 256         except exceptions.HttpException as e:
 257             raise PathocError(
 258                 "Proxy CONNECT failed: %s" % repr(e)
 259             )

 261     def socks_connect(self, connect_to):
 262         try:
 263             client_greet = socks.ClientGreeting(
 264                 socks.VERSION.SOCKS5,
 265                 [socks.METHOD.NO_AUTHENTICATION_REQUIRED]
 266             )
 267             client_greet.to_file(self.wfile)
 268             self.wfile.flush()
 269 
 270             server_greet = socks.ServerGreeting.from_file(self.rfile)
 271             server_greet.assert_socks5()
 272             if server_greet.method != socks.METHOD.NO_AUTHENTICATION_REQUIRED:
 273                 raise socks.SocksError(
 274                     socks.METHOD.NO_ACCEPTABLE_METHODS,
 275                     "pathoc only supports SOCKS without authentication"
 276                 )
 277 
 278             connect_request = socks.Message(
 279                 socks.VERSION.SOCKS5,
 280                 socks.CMD.CONNECT,
 281                 socks.ATYP.DOMAINNAME,
 282                 connect_to,
 283             )
 284             connect_request.to_file(self.wfile)
 285             self.wfile.flush()
 286 
 287             connect_reply = socks.Message.from_file(self.rfile)
 288             connect_reply.assert_socks5()
 289             if connect_reply.msg != socks.REP.SUCCEEDED:
 290                 raise socks.SocksError(
 291                     connect_reply.msg,
 292                     "SOCKS server error"
 293                 )
 294         except (socks.SocksError, exceptions.TcpDisconnect) as e:
 295             raise PathocError(str(e))

 297     def connect(self, connect_to=None, showssl=False, fp=sys.stdout):
 298         """
 299             connect_to: A (host, port) tuple, which will be connected to with
 300             an HTTP CONNECT request.
 301         """
 302         if self.use_http2 and not self.ssl:
 303             raise NotImplementedError("HTTP2 without SSL is not supported.")
 304 
 305         with tcp.TCPClient.connect(self) as closer:
 306             if connect_to:
 307                 self.http_connect(connect_to)
 308 
 309             self.sslinfo = None
 310             if self.ssl:
 311                 try:
 312                     alpn_protos = [b'http/1.1']
 313                     if self.use_http2:
 314                         alpn_protos.append(b'h2')
 315 
 316                     self.convert_to_tls(
 317                         sni=self.sni,
 318                         cert=self.clientcert,
 319                         method=self.ssl_version,
 320                         options=self.ssl_options,
 321                         cipher_list=self.ciphers,
 322                         alpn_protos=alpn_protos
 323                     )
 324                 except exceptions.TlsException as v:
 325                     raise PathocError(str(v))
 326 
 327                 self.sslinfo = SSLInfo(
 328                     self.connection.get_peer_cert_chain(),
 329                     self.get_current_cipher(),
 330                     self.get_alpn_proto_negotiated()
 331                 )
 332                 if showssl:
 333                     print(str(self.sslinfo), file=fp)
 334 
 335                 if self.use_http2:
 336                     self.protocol.check_alpn()
 337                     if not self.http2_skip_connection_preface:
 338                         self.protocol.perform_client_connection_preface()
 339 
 340             if self.timeout:
 341                 self.settimeout(self.timeout)
 342 
 343             return closer.pop()

 349     def wait(self, timeout=0.01, finish=True):
 350         """
 351             A generator that yields frames until Pathoc terminates.
 352 
 353             timeout: If specified None may be yielded instead if timeout is
 354             reached. If timeout is None, wait forever. If timeout is 0, return
 355             immediately if nothing is on the queue.
 356 
 357             finish: If true, consume messages until the reader shuts down.
 358             Otherwise, return None on timeout.
 359         """
 360         if self.ws_framereader:
 361             while True:
 362                 try:
 363                     frm = self.ws_framereader.frames_queue.get(
 364                         timeout=timeout,
 365                         block=True if timeout != 0 else False
 366                     )
 367                 except queue.Empty:
 368                     if finish:
 369                         continue
 370                     else:
 371                         return
 372                 if frm is None:
 373                     self.ws_framereader.join()
 374                     self.ws_framereader = None
 375                     return
 376                 yield frm

 378     def websocket_send_frame(self, r):
 379         """
 380             Sends a single websocket frame.
 381         """
 382         logger = log.ConnectionLogger(
 383             self.fp,
 384             self.hexdump,
 385             False,
 386             None,
 387             self.wfile if self.showreq else None,

 394     def websocket_start(self, r):
 395         """
 396             Performs an HTTP request, and attempts to drop into websocket
 397             connection.
 398         """
 399         resp = self.http(r)
 400         if resp.status_code == 101:
 401             self.ws_framereader = WebsocketFrameReader(
 402                 self.rfile,
 403                 self.fp,
 404                 self.showresp,
 405                 self.hexdump,
 406                 self.ws_read_limit,
 407                 self.timeout
 408             )
 409             self.ws_framereader.start()
 410         return resp

 412     def http(self, r):
 413         """
 414             Performs a single request.
 415 
 416             r: A language.http.Request object, or a string representing one
 417             request.
 418 
 419             Returns Response if we have a non-ignored response.
 420 
 421             May raise a exceptions.NetlibException
 422         """
 423         logger = log.ConnectionLogger(
 424             self.fp,
 425             self.hexdump,
 426             False,
 427             self.rfile if self.showresp else None,
 428             self.wfile if self.showreq else None,
 429         )
 430         with logger.ctx() as lg:
 431             lg(">> %s" % r)
 432             resp, req = None, None
 433             try:
 434                 req = language.serve(r, self.wfile, self.settings)
 435                 self.wfile.flush()
 436 
 437                 # build a dummy request to read the response
 438                 # ideally this would be returned directly from language.serve
 439                 dummy_req = net_http.Request(
 440                     first_line_format="relative",
 441                     method=req["method"],
 442                     scheme=b"http",
 443                     host=b"localhost",
 444                     port=80,
 445                     path=b"/",
 446                     http_version=b"HTTP/1.1",
 447                     content=b'',
 448                 )
 449 
 450                 resp = self.protocol.read_response(self.rfile, dummy_req)
 451                 resp.sslinfo = self.sslinfo
 452             except exceptions.HttpException as v:
 453                 lg("Invalid server response: %s" % v)
 454                 raise
 455             except exceptions.TcpTimeout:
 456                 if self.ignoretimeout:
 457                     lg("Timeout (ignored)")
 458                     return None
 459                 lg("Timeout")
 460                 raise
 461             finally:
 462                 if resp:
 463                     lg("<< %s %s: %s bytes" % (
 464                         resp.status_code, strutils.escape_control_characters(resp.reason) if resp.reason else "", len(resp.content)
 465                     ))
 466                     if resp.status_code in self.ignorecodes:
 467                         lg.suppress()
 468             return resp

 470     def request(self, r):
 471         """
 472             Performs a single request.
 473 
 474             r: A language.message.Message object, or a string representing
 475             one.
 476 
 477             Returns Response if we have a non-ignored response.
 478 
 479             May raise a exceptions.NetlibException
 480         """
 481         if isinstance(r, str):
 482             r = next(language.parse_pathoc(r, self.use_http2))
 483 
 484         if isinstance(r, language.http.Request):
 485             if r.ws:
 486                 return self.websocket_start(r)
 487             else:
 488                 return self.http(r)
 489         elif isinstance(r, language.websockets.WebsocketFrame):
 490             self.websocket_send_frame(r)
 491         elif isinstance(r, language.http2.Request):
 492             return self.http(r)

 494             # TODO: do something

 497 def main(args):  # pragma: no cover
 498     memo = set()
 499     p = None
 500 
 501     if args.repeat == 1:
 502         requests = args.requests
 503     else:
 504         # If we are replaying more than once, we must convert the request generators to lists
 505         # or they will be exhausted after the first run.
 506         # This is bad for the edge-case where get:/:x10000000 (see 0da3e51) is combined with -n 2,
 507         # but does not matter otherwise.
 508         requests = [list(x) for x in args.requests]
 509 
 510     try:
 511         requests_done = 0
 512         while True:
 513             if requests_done == args.repeat:
 514                 break
 515             if args.wait and requests_done > 0:
 516                 time.sleep(args.wait)
 517 
 518             requests_done += 1
 519             if args.random:
 520                 playlist = random.choice(requests)
 521             else:
 522                 playlist = itertools.chain.from_iterable(requests)
 523             p = Pathoc(
 524                 (args.host, args.port),
 525                 ssl=args.ssl,
 526                 sni=args.sni,
 527                 ssl_version=args.ssl_version,
 528                 ssl_options=args.ssl_options,
 529                 clientcert=args.clientcert,
 530                 ciphers=args.ciphers,
 531                 use_http2=args.use_http2,
 532                 http2_skip_connection_preface=args.http2_skip_connection_preface,
 533                 http2_framedump=args.http2_framedump,
 534                 showreq=args.showreq,
 535                 showresp=args.showresp,
 536                 explain=args.explain,
 537                 hexdump=args.hexdump,
 538                 ignorecodes=args.ignorecodes,
 539                 timeout=args.timeout,
 540                 ignoretimeout=args.ignoretimeout,
 541                 showsummary=True
 542             )
 543             trycount = 0
 544             try:
 545                 with p.connect(args.connect_to, args.showssl):
 546                     for spec in playlist:
 547                         if args.explain or args.memo:
 548                             spec = spec.freeze(p.settings)
 549                         if args.memo:
 550                             h = hashlib.sha256(spec.spec()).digest()
 551                             if h not in memo:
 552                                 trycount = 0
 553                                 memo.add(h)
 554                             else:
 555                                 trycount += 1
 556                                 if trycount > args.memolimit:
 557                                     print("Memo limit exceeded...", file=sys.stderr)
 558                                     return
 559                                 else:
 560                                     continue
 561                         try:
 562                             ret = p.request(spec)
 563                             if ret and args.oneshot:
 564                                 return
 565                             # We consume the queue when we can, so it doesn't build up.
 566                             for _ in p.wait(timeout=0, finish=False):
 567                                 pass
 568                         except exceptions.NetlibException:
 569                             break
 570                     for _ in p.wait(timeout=0.01, finish=True):
 571                         pass
 572             except exceptions.TcpException as v:
 573                 print(str(v), file=sys.stderr)
 574                 continue
 575             except PathocError as v:
 576                 print(str(v), file=sys.stderr)
 577                 sys.exit(1)
 578 
 579     except KeyboardInterrupt:
 580         pass
 581     if p:
 582         p.stop()
