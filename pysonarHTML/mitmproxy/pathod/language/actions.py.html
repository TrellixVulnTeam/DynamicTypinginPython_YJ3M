   9 @total_ordering  # type: ignore
  10 class _Action(base.Token):
  11 
  12     """
  13         An action that operates on the raw data stream of the message. All
  14         actions have one thing in common: an offset that specifies where the
  15         action should take place.
  16     """
  17 
  18     def __init__(self, offset):
  19         self.offset = offset
  20 
  21     def resolve(self, settings, msg):
  22         """
  23             Resolves offset specifications to a numeric offset. Returns a copy
  24             of the action object.
  25         """
  26         c = copy.copy(self)
  27         l = msg.length(settings)
  28         if c.offset == "r":
  29             c.offset = random.randrange(l)
  30         elif c.offset == "a":
  31             c.offset = l + 1
  32         return c
  33 
  34     def __lt__(self, other):
  35         return self.offset < other.offset
  36 
  37     def __eq__(self, other):
  38         return self.offset == other.offset
  39 
  40     def __repr__(self):
  41         return self.spec()
  42 
  43     @abc.abstractmethod
  44     def spec(self):  # pragma: no cover
  45         pass
  46 
  47     @abc.abstractmethod
  48     def intermediate(self, settings):  # pragma: no cover
  49         pass

  52 class PauseAt(_Action):
  53     unique_name = None
  54 
  55     def __init__(self, offset, seconds):
  56         _Action.__init__(self, offset)
  57         self.seconds = seconds
  58 
  59     @classmethod
  60     def expr(cls):
  61         e = pp.Literal("p").suppress()
  62         e += base.TokOffset
  63         e += pp.Literal(",").suppress()
  64         e += pp.MatchFirst(
  65             [
  66                 base.v_integer,
  67                 pp.Literal("f")
  68             ]
  69         )
  70         return e.setParseAction(lambda x: cls(*x))
  71 
  72     def spec(self):
  73         return "p%s,%s" % (self.offset, self.seconds)
  74 
  75     def intermediate(self, settings):
  76         return (self.offset, "pause", self.seconds)
  77 
  78     def freeze(self, settings_):
  79         return self

 103 class InjectAt(_Action):
 104     unique_name = None  # type: ignore
 105 
 106     def __init__(self, offset, value):
 107         _Action.__init__(self, offset)
 108         self.value = value
 109 
 110     @classmethod
 111     def expr(cls):
 112         e = pp.Literal("i").suppress()
 113         e += base.TokOffset
 114         e += pp.Literal(",").suppress()
 115         e += base.TokValue
 116         return e.setParseAction(lambda x: cls(*x))
 117 
 118     def spec(self):
 119         return "i%s,%s" % (self.offset, self.value.spec())
 120 
 121     def intermediate(self, settings):
 122         return (
 123             self.offset,
 124             "inject",
 125             self.value.get_generator(settings)
 126         )
 127 
 128     def freeze(self, settings):
 129         return InjectAt(self.offset, self.value.freeze(settings))
