  63 class Token:
  64 
  65     """
  66         A token in the specification language. Tokens are immutable. The token
  67         classes have no meaning in and of themselves, and are combined into
  68         Components and Actions to build the language.
  69     """
  70     __metaclass__ = abc.ABCMeta
  71 
  72     @classmethod
  73     def expr(cls):  # pragma: no cover
  74         """
  75             A parse expression.
  76         """
  77         return None
  78 
  79     @abc.abstractmethod
  80     def spec(self):  # pragma: no cover
  81         """
  82             A parseable specification for this token.
  83         """
  84         return None
  85 
  86     @property
  87     def unique_name(self) -> typing.Optional[str]:
  88         """
  89             Controls uniqueness constraints for tokens. No two tokens with the
  90             same name will be allowed. If no uniquness should be applied, this
  91             should be None.
  92         """
  93         return self.__class__.__name__.lower()
  94 
  95     def resolve(self, settings_, msg_):
  96         """
  97             Resolves this token to ready it for transmission. This means that
  98             the calculated offsets of actions are fixed.
  99 
 100             settings: a language.Settings instance
 101             msg: The containing message
 102         """
 103         return self
 104 
 105     def __repr__(self):
 106         return self.spec()

 153 class TokValueGenerate(Token):
 154 
 155     def __init__(self, usize, unit, datatype):
 156         if not unit:
 157             unit = "b"
 158         self.usize, self.unit, self.datatype = usize, unit, datatype
 159 
 160     def bytes(self):
 161         return self.usize * human.SIZE_UNITS[self.unit]
 162 
 163     def get_generator(self, settings_):
 164         return generators.RandomGenerator(self.datatype, self.bytes())
 165 
 166     def freeze(self, settings):
 167         g = self.get_generator(settings)
 168         return TokValueLiteral(strutils.bytes_to_escaped_str(g[:], escape_single_quotes=True))
 169 
 170     @classmethod
 171     def expr(cls):
 172         e = pp.Literal("@").suppress() + v_integer
 173 
 174         u = functools.reduce(
 175             operator.or_,
 176             [pp.Literal(i) for i in human.SIZE_UNITS.keys()]
 177         ).leaveWhitespace()
 178         e = e + pp.Optional(u, default=None)
 179 
 180         s = pp.Literal(",").suppress()
 181         s += functools.reduce(
 182             operator.or_,
 183             [pp.Literal(i) for i in generators.DATATYPES.keys()]
 184         )
 185         e += pp.Optional(s, default="bytes")
 186         return e.setParseAction(lambda x: cls(*x))
 187 
 188     def spec(self):
 189         s = "@%s" % self.usize
 190         if self.unit != "b":
 191             s += self.unit
 192         if self.datatype != "bytes":
 193             s += ",%s" % self.datatype
 194         return s

 197 class TokValueFile(Token):
 198 
 199     def __init__(self, path):
 200         self.path = str(path)
 201 
 202     @classmethod
 203     def expr(cls):
 204         e = pp.Literal("<").suppress()
 205         e = e + v_naked_literal
 206         return e.setParseAction(lambda x: cls(*x))
 207 
 208     def freeze(self, settings_):
 209         return self
 210 
 211     def get_generator(self, settings):
 212         if not settings.staticdir:
 213             raise exceptions.FileAccessDenied("File access disabled.")
 214         s = os.path.expanduser(self.path)
 215         s = os.path.normpath(
 216             os.path.abspath(os.path.join(settings.staticdir, s))
 217         )
 218         uf = settings.unconstrained_file_access
 219         if not uf and not s.startswith(os.path.normpath(settings.staticdir)):
 220             raise exceptions.FileAccessDenied(
 221                 "File access outside of configured directory"
 222             )
 223         if not os.path.isfile(s):
 224             raise exceptions.FileAccessDenied("File not readable")
 225         return generators.FileGenerator(s)
 226 
 227     def spec(self):
 228         return "<'%s'" % self.path

 331 class OptionsOrValue(_Component):
 332 
 333     """
 334         Can be any of a specified set of options, or a value specifier.
 335     """
 336     preamble = ""
 337     options: typing.List[str] = []
 338 
 339     def __init__(self, value):
 340         # If it's a string, we were passed one of the options, so we lower-case
 341         # it to be canonical. The user can specify a different case by using a
 342         # string value literal.
 343         self.option_used = False
 344         if isinstance(value, str):
 345             for i in self.options:
 346                 # Find the exact option value in a case-insensitive way
 347                 if i.lower() == value.lower():
 348                     self.option_used = True
 349                     value = TokValueLiteral(i)
 350                     break
 351         self.value = value
 352 
 353     @classmethod
 354     def expr(cls):
 355         parts = [pp.CaselessLiteral(i) for i in cls.options]
 356         m = pp.MatchFirst(parts)
 357         spec = m | TokValue.copy()
 358         spec = spec.setParseAction(lambda x: cls(*x))
 359         if cls.preamble:
 360             spec = pp.Literal(cls.preamble).suppress() + spec
 361         return spec
 362 
 363     def values(self, settings):
 364         return [
 365             self.value.get_generator(settings)
 366         ]
 367 
 368     def spec(self):
 369         s = self.value.spec()
 370         if s[1:-1].lower() in self.options:
 371             s = s[1:-1].lower()
 372         return "%s%s" % (self.preamble, s)
 373 
 374     def freeze(self, settings):
 375         return self.__class__(self.value.freeze(settings))

 378 class Integer(_Component):
 379     bounds: typing.Tuple[typing.Optional[int], typing.Optional[int]] = (None, None)
 380     preamble = ""
 381 
 382     def __init__(self, value):
 383         v = int(value)
 384         outofbounds = any([
 385             self.bounds[0] is not None and v < self.bounds[0],
 386             self.bounds[1] is not None and v > self.bounds[1]
 387         ])
 388         if outofbounds:
 389             raise exceptions.ParseException(
 390                 "Integer value must be between %s and %s." % self.bounds,
 391                 0, 0
 392             )
 393         self.value = str(value).encode()
 394 
 395     @classmethod
 396     def expr(cls):
 397         e = v_integer.copy()
 398         if cls.preamble:
 399             e = pp.Literal(cls.preamble).suppress() + e
 400         return e.setParseAction(lambda x: cls(*x))
 401 
 402     def values(self, settings):
 403         return [self.value]
 404 
 405     def spec(self):
 406         return "%s%s" % (self.preamble, self.value.decode())
 407 
 408     def freeze(self, settings_):
 409         return self

 439 class FixedLengthValue(Value):
 440 
 441     """
 442         A value component lead by an optional preamble.
 443     """
 444     preamble = ""
 445     length: typing.Optional[int] = None
 446 
 447     def __init__(self, value):
 448         Value.__init__(self, value)
 449         lenguess = None
 450         try:
 451             lenguess = len(value.get_generator(Settings()))
 452         except exceptions.RenderError:
 453             pass
 454         # This check will fail if we know the length upfront
 455         if lenguess is not None and lenguess != self.length:
 456             raise exceptions.RenderError(
 457                 "Invalid value length: '%s' is %s bytes, should be %s." % (
 458                     self.spec(),
 459                     lenguess,
 460                     self.length
 461                 )
 462             )
 463 
 464     def values(self, settings):
 465         ret = Value.values(self, settings)
 466         l = sum(len(i) for i in ret)
 467         # This check will fail if we don't know the length upfront - i.e. for
 468         # file inputs
 469         if l != self.length:
 470             raise exceptions.RenderError(
 471                 "Invalid value length: '%s' is %s bytes, should be %s." % (
 472                     self.spec(),
 473                     l,
 474                     self.length
 475                 )
 476             )
 477         return ret

 480 class Boolean(_Component):
 481 
 482     """
 483         A boolean flag.
 484             name  = true
 485             -name = false
 486     """
 487     name = ""
 488 
 489     def __init__(self, value):
 490         self.value = value
 491 
 492     @classmethod
 493     def expr(cls):
 494         e = pp.Optional(pp.Literal("-"), default=True)
 495         e += pp.Literal(cls.name).suppress()
 496 
 497         def parse(s_, loc_, toks):
 498             val = True
 499             if toks[0] == "-":
 500                 val = False
 501             return cls(val)
 502 
 503         return e.setParseAction(parse)
 504 
 505     def spec(self):
 506         return "%s%s" % ("-" if not self.value else "", self.name)

 509 class IntField(_Component):
 510 
 511     """
 512         An integer field, where values can optionally specified by name.
 513     """
 514     names: typing.Dict[str, int] = {}
 515     max = 16
 516     preamble = ""
 517 
 518     def __init__(self, value):
 519         self.origvalue = value
 520         self.value = self.names.get(value, value)
 521         if self.value > self.max:
 522             raise exceptions.ParseException(
 523                 "Value can't exceed %s" % self.max, 0, 0
 524             )
 525 
 526     @classmethod
 527     def expr(cls):
 528         parts = [pp.CaselessLiteral(i) for i in cls.names.keys()]
 529         m = pp.MatchFirst(parts)
 530         spec = m | v_integer.copy()
 531         spec = spec.setParseAction(lambda x: cls(*x))
 532         if cls.preamble:
 533             spec = pp.Literal(cls.preamble).suppress() + spec
 534         return spec
 535 
 536     def values(self, settings):
 537         return [str(self.value)]
 538 
 539     def spec(self):
 540         return "%s%s" % (self.preamble, self.origvalue)
