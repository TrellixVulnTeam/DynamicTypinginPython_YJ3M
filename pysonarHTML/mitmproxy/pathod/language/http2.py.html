   7 """
   8     Normal HTTP requests:
   9         <method>:<path>:<header>:<body>
  10     e.g.:
  11         GET:/
  12         GET:/:h"foo"="bar"

  31 def get_header(val, headers):
  32     """
  33         Header keys may be Values, so we have to "generate" them as we try the
  34         match.
  35     """
  36     for h in headers:
  37         k = h.key.get_generator({})
  38         if len(k) == len(val) and k[:].lower() == val.lower():
  39             return h
  40     return None

  60     @property
  61     def headers(self):
  62         headers = self.toks(_HeaderMixin)
  63 
  64         if not self.raw:
  65             if not get_header(b"content-length", headers):
  66                 if not self.body:
  67                     length = 0
  68                 else:
  69                     length = len(self.body.string())
  70                 headers.append(
  71                     Header(
  72                         base.TokValueLiteral("content-length"),
  73                         base.TokValueLiteral(str(length)),
  74                     )
  75                 )
  76         return headers

 150 class Response(_HTTP2Message):
 151     unique_name = None
 152     comps = (
 153         Header,
 154         Body,
 155         ShortcutContentType,
 156         ShortcutLocation,
 157         Raw,
 158     )
 159 
 160     def __init__(self, tokens):
 161         super(Response, self).__init__(tokens)
 162         self.rendered_values = None
 163         self.stream_id = 2
 164 
 165     @property
 166     def status_code(self):
 167         return self.tok(StatusCode)
 168 
 169     @classmethod
 170     def expr(cls):
 171         parts = [i.expr() for i in cls.comps]
 172         atom = pp.MatchFirst(parts)
 173         resp = pp.And(
 174             [
 175                 StatusCode.expr(),
 176                 pp.ZeroOrMore(base.Sep + atom)
 177             ]
 178         )
 179         resp = resp.setParseAction(cls)
 180         return resp
 181 
 182     def values(self, settings):
 183         if self.rendered_values:
 184             return self.rendered_values
 185         else:
 186             headers = Headers([header.values(settings) for header in self.headers])
 187 
 188             body = self.body
 189             if body:
 190                 body = body.string()
 191 
 192             resp = http.Response(
 193                 b'HTTP/2.0',
 194                 int(self.status_code.string()),
 195                 b'',
 196                 headers,
 197                 body,
 198             )
 199             resp.stream_id = self.stream_id
 200 
 201             self.rendered_values = settings.protocol.assemble(resp)
 202             return self.rendered_values
 203 
 204     def spec(self):
 205         return ":".join([i.spec() for i in self.tokens])

 213 class Request(_HTTP2Message):
 214     comps = (
 215         Header,
 216         ShortcutContentType,
 217         ShortcutUserAgent,
 218         Raw,
 219         NestedResponse,
 220         Body,
 221         Times,
 222     )
 223     logattrs = ["method", "path"]
 224 
 225     def __init__(self, tokens):
 226         super(Request, self).__init__(tokens)
 227         self.rendered_values = None
 228         self.stream_id = 1
 229 
 230     @property
 231     def method(self):
 232         return self.tok(Method)
 233 
 234     @property
 235     def path(self):
 236         return self.tok(Path)
 237 
 238     @property
 239     def nested_response(self):
 240         return self.tok(NestedResponse)
 241 
 242     @property
 243     def times(self):
 244         return self.tok(Times)
 245 
 246     @classmethod
 247     def expr(cls):
 248         parts = [i.expr() for i in cls.comps]
 249         atom = pp.MatchFirst(parts)
 250         resp = pp.And(
 251             [
 252                 Method.expr(),
 253                 base.Sep,
 254                 Path.expr(),
 255                 pp.ZeroOrMore(base.Sep + atom)
 256             ]
 257         )
 258         resp = resp.setParseAction(cls)
 259         return resp
 260 
 261     def values(self, settings):
 262         if self.rendered_values:
 263             return self.rendered_values
 264         else:
 265             path = self.path.string()
 266             if self.nested_response:
 267                 path += self.nested_response.parsed.spec().encode()
 268 
 269             headers = Headers([header.values(settings) for header in self.headers])
 270 
 271             body = self.body
 272             if body:
 273                 body = body.string()
 274 
 275             req = http.Request(
 276                 b'',
 277                 self.method.string(),
 278                 b'http',
 279                 b'',
 280                 b'',
 281                 path,
 282                 (2, 0),
 283                 headers,
 284                 body,
 285             )
 286             req.stream_id = self.stream_id
 287 
 288             self.rendered_values = settings.protocol.assemble(req)
 289             return self.rendered_values
 290 
 291     def spec(self):
 292         return ":".join([i.spec() for i in self.tokens])
