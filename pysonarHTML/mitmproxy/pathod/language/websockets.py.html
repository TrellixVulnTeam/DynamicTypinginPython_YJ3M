 157     @classmethod
 158     def expr(cls):
 159         parts = [i.expr() for i in cls.components]
 160         atom = pp.MatchFirst(parts)
 161         resp = pp.And(
 162             [
 163                 WF.expr(),
 164                 base.Sep,
 165                 pp.ZeroOrMore(base.Sep + atom)
 166             ]
 167         )
 168         resp = resp.setParseAction(cls)
 169         return resp

 171     @property
 172     def nested_frame(self):
 173         return self.tok(NestedFrame)

 175     def resolve(self, settings, msg=None):
 176         tokens = self.tokens[:]
 177         if not self.mask and settings.is_client:
 178             tokens.append(
 179                 Mask(True)
 180             )
 181         if not self.knone and self.mask and self.mask.value and not self.key:
 182             allowed_chars = string.ascii_letters + string.digits
 183             k = ''.join([allowed_chars[random.randrange(0, len(allowed_chars))] for i in range(4)])
 184             tokens.append(
 185                 Key(base.TokValueLiteral(k))
 186             )
 187         return self.__class__(
 188             [i.resolve(settings, self) for i in tokens]
 189         )
 190 
 191     def values(self, settings):
 192         if self.body:
 193             bodygen = self.body.value.get_generator(settings)
 194             length = len(self.body.value.get_generator(settings))
 195         elif self.rawbody:
 196             bodygen = self.rawbody.value.get_generator(settings)
 197             length = len(self.rawbody.value.get_generator(settings))
 198         elif self.nested_frame:
 199             bodygen = NESTED_LEADER + strutils.always_bytes(self.nested_frame.parsed.spec())
 200             length = len(bodygen)
 201         else:
 202             bodygen = None
 203             length = 0
 204         if self.toklength:
 205             length = int(self.toklength.value)
 206         frameparts = dict(
 207             payload_length=length
 208         )
 209         if self.mask and self.mask.value:
 210             frameparts["mask"] = True
 211         if self.knone:
 212             frameparts["masking_key"] = None
 213         elif self.key:
 214             key = self.key.values(settings)[0][:]
 215             frameparts["masking_key"] = key
 216         for i in ["opcode", "fin", "rsv1", "rsv2", "rsv3", "mask"]:
 217             v = getattr(self, i, None)
 218             if v is not None:
 219                 frameparts[i] = v.value
 220         frame = mitmproxy.net.websockets.FrameHeader(**frameparts)
 221         vals = [bytes(frame)]
 222         if bodygen:
 223             if frame.masking_key and not self.rawbody:
 224                 masker = mitmproxy.net.websockets.Masker(frame.masking_key)
 225                 vals.append(
 226                     generators.TransformGenerator(
 227                         bodygen,
 228                         masker.mask
 229                     )
 230                 )
 231             else:
 232                 vals.append(bodygen)
 233         return vals
 234 
 235     def spec(self):
 236         return ":".join([i.spec() for i in self.tokens])
