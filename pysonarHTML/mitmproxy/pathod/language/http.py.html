 101 def get_header(val, headers):
 102     """
 103         Header keys may be Values, so we have to "generate" them as we try the
 104         match.
 105     """
 106     for h in headers:
 107         k = h.key.get_generator({})
 108         if len(k) == len(val) and k[:].lower() == val.lower():
 109             return h
 110     return None

 113 class _HTTPMessage(message.Message):
 114     version = b"HTTP/1.1"
 115 
 116     @property
 117     def actions(self):
 118         return self.toks(actions._Action)
 119 
 120     @property
 121     def raw(self):
 122         return bool(self.tok(Raw))
 123 
 124     @property
 125     def body(self):
 126         return self.tok(Body)
 127 
 128     @abc.abstractmethod
 129     def preamble(self, settings):  # pragma: no cover
 130         pass
 131 
 132     @property
 133     def headers(self):
 134         return self.toks(_HeaderMixin)
 135 
 136     def values(self, settings):
 137         vals = self.preamble(settings)
 138         vals.append(b"\r\n")
 139         for h in self.headers:
 140             vals.extend(h.values(settings))
 141         vals.append(b"\r\n")
 142         if self.body:
 143             vals.extend(self.body.values(settings))
 144         return vals

 147 class Response(_HTTPMessage):
 148     unique_name = None  # type: ignore
 149     comps = (
 150         Header,
 151         ShortcutContentType,
 152         ShortcutLocation,
 153         Raw,
 154         Reason,
 155         Body,
 156 
 157         actions.PauseAt,
 158         actions.DisconnectAt,
 159         actions.InjectAt,
 160     )
 161     logattrs = ["status_code", "reason", "version", "body"]
 162 
 163     @property
 164     def ws(self):
 165         return self.tok(WS)
 166 
 167     @property
 168     def status_code(self):
 169         return self.tok(StatusCode)
 170 
 171     @property
 172     def reason(self):
 173         return self.tok(Reason)
 174 
 175     def preamble(self, settings):
 176         l = [self.version, b" "]
 177         l.extend(self.status_code.values(settings))
 178         status_code = int(self.status_code.value)
 179         l.append(b" ")
 180         if self.reason:
 181             l.extend(self.reason.values(settings))
 182         else:
 183             l.append(
 184                 status_codes.RESPONSES.get(
 185                     status_code,
 186                     "Unknown code"
 187                 ).encode()
 188             )
 189         return l
 190 
 191     def resolve(self, settings, msg=None):
 192         tokens = self.tokens[:]
 193         if self.ws:
 194             if not settings.websocket_key:
 195                 raise exceptions.RenderError(
 196                     "No websocket key - have we seen a client handshake?"
 197                 )
 198             if not self.status_code:
 199                 tokens.insert(
 200                     1,
 201                     StatusCode(101)
 202                 )
 203             headers = mitmproxy.net.websockets.server_handshake_headers(
 204                 settings.websocket_key
 205             )
 206             for i in headers.fields:
 207                 if not get_header(i[0], self.headers):
 208                     tokens.append(
 209                         Header(
 210                             base.TokValueLiteral(i[0].decode()),
 211                             base.TokValueLiteral(i[1].decode()))
 212                     )
 213         if not self.raw:
 214             if not get_header(b"Content-Length", self.headers):
 215                 if not self.body:
 216                     length = 0
 217                 else:
 218                     length = sum(
 219                         len(i) for i in self.body.values(settings)
 220                     )
 221                 tokens.append(
 222                     Header(
 223                         base.TokValueLiteral("Content-Length"),
 224                         base.TokValueLiteral(str(length)),
 225                     )
 226                 )
 227         intermediate = self.__class__(tokens)
 228         return self.__class__(
 229             [i.resolve(settings, intermediate) for i in tokens]
 230         )
 231 
 232     @classmethod
 233     def expr(cls):
 234         parts = [i.expr() for i in cls.comps]
 235         atom = pp.MatchFirst(parts)
 236         resp = pp.And(
 237             [
 238                 pp.MatchFirst(
 239                     [
 240                         WS.expr() + pp.Optional(
 241                             base.Sep + StatusCode.expr()
 242                         ),
 243                         StatusCode.expr(),
 244                     ]
 245                 ),
 246                 pp.ZeroOrMore(base.Sep + atom)
 247             ]
 248         )
 249         resp = resp.setParseAction(cls)
 250         return resp
 251 
 252     def spec(self):
 253         return ":".join([i.spec() for i in self.tokens])

 261 class Request(_HTTPMessage):
 262     comps = (
 263         Header,

 270

 272         actions.DisconnectAt,
 273         actions.InjectAt,
 274     )
 275     logattrs = ["method", "path", "body"]
 276 
 277     @property
 278     def ws(self):
 279         return self.tok(WS)
 280 
 281     @property
 282     def method(self):
 283         return self.tok(Method)
 284 
 285     @property
 286     def path(self):
 287         return self.tok(Path)
 288 
 289     @property
 290     def times(self):
 291         return self.tok(Times)
 292 
 293     @property
 294     def nested_response(self):
 295         return self.tok(NestedResponse)
 296 
 297     def preamble(self, settings):
 298         v = self.method.values(settings)
 299         v.append(b" ")
 300         v.extend(self.path.values(settings))
 301         if self.nested_response:
 302             v.append(self.nested_response.parsed.spec())
 303         v.append(b" ")
 304         v.append(self.version)
 305         return v
 306 
 307     def resolve(self, settings, msg=None):
 308         tokens = self.tokens[:]
 309         if self.ws:
 310             if not self.method:
 311                 tokens.insert(
 312                     1,
 313                     Method("get")
 314                 )
 315             for i in mitmproxy.net.websockets.client_handshake_headers().fields:
 316                 if not get_header(i[0], self.headers):
 317                     tokens.append(
 318                         Header(
 319                             base.TokValueLiteral(i[0].decode()),
 320                             base.TokValueLiteral(i[1].decode())
 321                         )
 322                     )
 323         if not self.raw:
 324             if not get_header(b"Content-Length", self.headers):
 325                 if self.body:
 326                     length = sum(
 327                         len(i) for i in self.body.values(settings)
 328                     )
 329                     tokens.append(
 330                         Header(
 331                             base.TokValueLiteral("Content-Length"),
 332                             base.TokValueLiteral(str(length)),
 333                         )
 334                     )
 335             if settings.request_host:
 336                 if not get_header(b"Host", self.headers):
 337                     h = settings.request_host
 338                     if self.path:
 339                         path = b"".join(self.path.values({})).decode(
 340                             "ascii", errors="ignore"
 341                         )
 342                         try:
 343                             _, h, _, _ = url.parse(path)
 344                             h = h.decode("ascii", errors="ignore")
 345                         except ValueError:
 346                             pass
 347                     tokens.append(
 348                         Header(
 349                             base.TokValueLiteral("Host"),
 350                             base.TokValueLiteral(h)
 351                         )
 352                     )
 353         intermediate = self.__class__(tokens)
 354         return self.__class__(
 355             [i.resolve(settings, intermediate) for i in tokens]
 356         )
 357 
 358     @classmethod
 359     def expr(cls):
 360         parts = [i.expr() for i in cls.comps]
 361         atom = pp.MatchFirst(parts)
 362         resp = pp.And(
 363             [
 364                 pp.MatchFirst(
 365                     [
 366                         WS.expr() + pp.Optional(
 367                             base.Sep + Method.expr()
 368                         ),
 369                         Method.expr(),
 370                     ]
 371                 ),
 372                 base.Sep,
 373                 Path.expr(),
 374                 pp.ZeroOrMore(base.Sep + atom)
 375             ]
 376         )
 377         resp = resp.setParseAction(cls)
 378         return resp
 379 
 380     def spec(self):
 381         return ":".join([i.spec() for i in self.tokens])

 384 def make_error_response(reason, body=None):
 385     tokens = [
 386         StatusCode("800"),
 387         Header(
 388             base.TokValueLiteral("Content-Type"),
 389             base.TokValueLiteral("text/plain")
 390         ),
 391         Reason(base.TokValueLiteral(reason)),
 392         Body(base.TokValueLiteral("pathod error: " + (body or reason))),
 393     ]
 394     return Response(tokens)
