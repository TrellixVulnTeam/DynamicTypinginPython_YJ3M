  12 class Message:
  13     __metaclass__ = abc.ABCMeta
  14     logattrs: typing.List[str] = []
  15 
  16     def __init__(self, tokens):
  17         track = set([])
  18         for i in tokens:
  19             if i.unique_name:
  20                 if i.unique_name in track:
  21                     raise exceptions.ParseException(
  22                         "Message has multiple %s clauses, "
  23                         "but should only have one." % i.unique_name,
  24                         0, 0
  25                     )
  26                 else:
  27                     track.add(i.unique_name)
  28         self.tokens = tokens
  29 
  30     def strike_token(self, name):
  31         toks = [i for i in self.tokens if i.unique_name != name]
  32         return self.__class__(toks)
  33 
  34     def toks(self, klass):
  35         """
  36             Fetch all tokens that are instances of klass
  37         """
  38         return [i for i in self.tokens if isinstance(i, klass)]
  39 
  40     def tok(self, klass):
  41         """
  42             Fetch first token that is an instance of klass
  43         """
  44         l = self.toks(klass)
  45         if l:
  46             return l[0]
  47 
  48     def length(self, settings):
  49         """
  50             Calculate the length of the base message without any applied
  51             actions.
  52         """
  53         return sum(len(x) for x in self.values(settings))
  54 
  55     def preview_safe(self):
  56         """
  57             Return a copy of this message that is safe for previews.
  58         """
  59         tokens = [i for i in self.tokens if not isinstance(i, actions.PauseAt)]
  60         return self.__class__(tokens)
  61 
  62     def maximum_length(self, settings):
  63         """
  64             Calculate the maximum length of the base message with all applied
  65             actions.
  66         """
  67         l = self.length(settings)
  68         for i in self.actions:
  69             if isinstance(i, actions.InjectAt):
  70                 l += len(i.value.get_generator(settings))
  71         return l
  72 
  73     @classmethod
  74     def expr(cls):  # pragma: no cover
  75         pass
  76 
  77     def log(self, settings):
  78         """
  79             A dictionary that should be logged if this message is served.
  80         """
  81         ret = {}
  82         for i in self.logattrs:
  83             v = getattr(self, i)
  84             # Careful not to log any VALUE specs without sanitizing them first.
  85             # We truncate at 1k.
  86             if hasattr(v, "values"):
  87                 v = [x[:LOG_TRUNCATE] for x in v.values(settings)]
  88                 v = strutils.bytes_to_escaped_str(b"".join(v))
  89             elif hasattr(v, "__len__"):
  90                 v = v[:LOG_TRUNCATE]
  91                 v = strutils.bytes_to_escaped_str(v)
  92             ret[i] = v
  93         ret["spec"] = self.spec()
  94         return ret
  95 
  96     def freeze(self, settings):
  97         r = self.resolve(settings)
  98         return self.__class__([i.freeze(settings) for i in r.tokens])
  99 
 100     def __repr__(self):
 101         return self.spec()

 104 class NestedMessage(base.Token):
 105     """
 106         A nested message, as an escaped string with a preamble.
 107     """
 108     preamble = ""
 109     nest_type: typing.Optional[typing.Type[Message]] = None
 110 
 111     def __init__(self, value):
 112         super().__init__()
 113         self.value = value
 114         try:
 115             self.parsed = self.nest_type(
 116                 self.nest_type.expr().parseString(
 117                     value.val.decode(),
 118                     parseAll=True
 119                 )
 120             )
 121         except pp.ParseException as v:
 122             raise exceptions.ParseException(v.msg, v.line, v.col)
 123 
 124     @classmethod
 125     def expr(cls):
 126         e = pp.Literal(cls.preamble).suppress()
 127         e = e + base.TokValueLiteral.expr()
 128         return e.setParseAction(lambda x: cls(*x))
 129 
 130     def values(self, settings):
 131         return [
 132             self.value.get_generator(settings),
 133         ]
 134 
 135     def spec(self):
 136         return "%s%s" % (self.preamble, self.value.spec())
 137 
 138     def freeze(self, settings):
 139         f = self.parsed.freeze(settings).spec()
 140         return self.__class__(
 141             base.TokValueLiteral(
 142                 strutils.bytes_to_escaped_str(f.encode(), escape_single_quotes=True)
 143             )
 144         )
