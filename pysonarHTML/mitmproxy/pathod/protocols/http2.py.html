 165     def assemble(self, message):
 166         if isinstance(message, mitmproxy.net.http.request.Request):
 167             return self.assemble_request(message)
 168         elif isinstance(message, mitmproxy.net.http.response.Response):
 169             return self.assemble_response(message)
 170         else:
 171             raise ValueError("HTTP message not supported.")
 172 
 173     def assemble_request(self, request):
 174         assert isinstance(request, mitmproxy.net.http.request.Request)
 175 
 176         authority = self.tcp_handler.sni if self.tcp_handler.sni else self.tcp_handler.address[0]
 177         if self.tcp_handler.address[1] != 443:
 178             authority += ":%d" % self.tcp_handler.address[1]
 179 
 180         headers = request.headers.copy()
 181 
 182         if ':authority' not in headers:
 183             headers.insert(0, ':authority', authority)
 184         headers.insert(0, ':scheme', request.scheme)
 185         headers.insert(0, ':path', request.path)
 186         headers.insert(0, ':method', request.method)
 187 
 188         if hasattr(request, 'stream_id'):
 189             stream_id = request.stream_id
 190         else:
 191             stream_id = self._next_stream_id()
 192 
 193         return list(itertools.chain(
 194             self._create_headers(headers, stream_id, end_stream=(request.content is None or len(request.content) == 0)),
 195             self._create_body(request.content, stream_id)))
 196 
 197     def assemble_response(self, response):
 198         assert isinstance(response, mitmproxy.net.http.response.Response)
 199 
 200         headers = response.headers.copy()
 201 
 202         if ':status' not in headers:
 203             headers.insert(0, b':status', str(response.status_code).encode())
 204 
 205         if hasattr(response, 'stream_id'):
 206             stream_id = response.stream_id
 207         else:
 208             stream_id = self._next_stream_id()
 209 
 210         return list(itertools.chain(
 211             self._create_headers(headers, stream_id, end_stream=(response.content is None or len(response.content) == 0)),
 212             self._create_body(response.content, stream_id),
 213         ))

 215     def perform_connection_preface(self, force=False):
 216         if force or not self.connection_preface_performed:
 217             if self.is_server:
 218                 self.perform_server_connection_preface(force)
 219             else:
 220                 self.perform_client_connection_preface(force)
 221 
 222     def perform_server_connection_preface(self, force=False):
 223         if force or not self.connection_preface_performed:
 224             self.connection_preface_performed = True
 225 
 226             magic_length = len(self.CLIENT_CONNECTION_PREFACE)
 227             magic = self.tcp_handler.rfile.safe_read(magic_length)
 228             assert magic == self.CLIENT_CONNECTION_PREFACE
 229 
 230             frm = hyperframe.frame.SettingsFrame(settings={
 231                 hyperframe.frame.SettingsFrame.ENABLE_PUSH: 0,
 232                 hyperframe.frame.SettingsFrame.MAX_CONCURRENT_STREAMS: 1,
 233             })
 234             self.send_frame(frm, hide=True)
 235             self._receive_settings(hide=True)
 236 
 237     def perform_client_connection_preface(self, force=False):
 238         if force or not self.connection_preface_performed:
 239             self.connection_preface_performed = True
 240 
 241             self.tcp_handler.wfile.write(self.CLIENT_CONNECTION_PREFACE)
 242 
 243             self.send_frame(hyperframe.frame.SettingsFrame(), hide=True)
 244             self._receive_settings(hide=True)  # server announces own settings
 245             self._receive_settings(hide=True)  # server acks my settings
 246 
 247     def send_frame(self, frm, hide=False):
 248         raw_bytes = frm.serialize()
 249         self.tcp_handler.wfile.write(raw_bytes)
 250         self.tcp_handler.wfile.flush()
 251         if not hide and self.dump_frames:  # pragma: no cover
 252             print(">> " + repr(frm))
 253 
 254     def read_frame(self, hide=False):
 255         while True:
 256             frm = http2.parse_frame(*http2.read_raw_frame(self.tcp_handler.rfile))
 257             if not hide and self.dump_frames:  # pragma: no cover
 258                 print("<< " + repr(frm))
 259 
 260             if isinstance(frm, hyperframe.frame.PingFrame):
 261                 raw_bytes = hyperframe.frame.PingFrame(flags=['ACK'], payload=frm.payload).serialize()
 262                 self.tcp_handler.wfile.write(raw_bytes)
 263                 self.tcp_handler.wfile.flush()
 264                 continue
 265             if isinstance(frm, hyperframe.frame.SettingsFrame) and 'ACK' not in frm.flags:
 266                 self._apply_settings(frm.settings, hide)
 267             if isinstance(frm, hyperframe.frame.DataFrame) and frm.flow_controlled_length > 0:
 268                 self._update_flow_control_window(frm.stream_id, frm.flow_controlled_length)
 269             return frm

 271     def check_alpn(self):
 272         alp = self.tcp_handler.get_alpn_proto_negotiated()
 273         if alp != b'h2':
 274             raise NotImplementedError(
 275                 "HTTP2Protocol can not handle unknown ALPN value: %s" % alp)
 276         return True

 278     def _handle_unexpected_frame(self, frm):
 279         if isinstance(frm, hyperframe.frame.SettingsFrame):
 280             return
 281         if self.unhandled_frame_cb:
 282             self.unhandled_frame_cb(frm)
 283 
 284     def _receive_settings(self, hide=False):
 285         while True:
 286             frm = self.read_frame(hide)
 287             if isinstance(frm, hyperframe.frame.SettingsFrame):
 288                 break
 289             else:
 290                 self._handle_unexpected_frame(frm)
 291 
 292     def _next_stream_id(self):
 293         if self.current_stream_id is None:
 294             if self.is_server:
 295                 # servers must use even stream ids
 296                 self.current_stream_id = 2
 297             else:
 298                 # clients must use odd stream ids
 299                 self.current_stream_id = 1
 300         else:
 301             self.current_stream_id += 2
 302         return self.current_stream_id
 303 
 304     def _apply_settings(self, settings, hide=False):
 305         for setting, value in settings.items():
 306             old_value = self.http2_settings[setting]
 307             if not old_value:
 308                 old_value = '-'
 309             self.http2_settings[setting] = value
 310 
 311         frm = hyperframe.frame.SettingsFrame(flags=['ACK'])
 312         self.send_frame(frm, hide)
 313 
 314     def _update_flow_control_window(self, stream_id, increment):
 315         frm = hyperframe.frame.WindowUpdateFrame(stream_id=0, window_increment=increment)
 316         self.send_frame(frm)
 317         frm = hyperframe.frame.WindowUpdateFrame(stream_id=stream_id, window_increment=increment)
 318         self.send_frame(frm)

 320     def _create_headers(self, headers, stream_id, end_stream=True):
 321         def frame_cls(chunks):
 322             for i in chunks:
 323                 if i == 0:
 324                     yield hyperframe.frame.HeadersFrame, i
 325                 else:
 326                     yield hyperframe.frame.ContinuationFrame, i
 327 
 328         header_block_fragment = self.encoder.encode(headers.fields)
 329 
 330         chunk_size = self.http2_settings[hyperframe.frame.SettingsFrame.MAX_FRAME_SIZE]
 331         chunks = range(0, len(header_block_fragment), chunk_size)
 332         frms = [frm_cls(
 333             flags=[],
 334             stream_id=stream_id,
 335             data=header_block_fragment[i:i + chunk_size]) for frm_cls, i in frame_cls(chunks)]
 336 
 337         frms[-1].flags.add('END_HEADERS')
 338         if end_stream:
 339             frms[0].flags.add('END_STREAM')
 340 
 341         if self.dump_frames:  # pragma: no cover
 342             for frm in frms:
 343                 print(">> ", repr(frm))
 344 
 345         return [frm.serialize() for frm in frms]
 346 
 347     def _create_body(self, body, stream_id):
 348         if body is None or len(body) == 0:
 349             return b''
 350 
 351         chunk_size = self.http2_settings[hyperframe.frame.SettingsFrame.MAX_FRAME_SIZE]
 352         chunks = range(0, len(body), chunk_size)
 353         frms = [hyperframe.frame.DataFrame(
 354             flags=[],
 355             stream_id=stream_id,
 356             data=body[i:i + chunk_size]) for i in chunks]
 357         frms[-1].flags.add('END_STREAM')
 358 
 359         if self.dump_frames:  # pragma: no cover
 360             for frm in frms:
 361                 print(">> ", repr(frm))
 362 
 363         return [frm.serialize() for frm in frms]
 364 
 365     def _receive_transmission(self, stream_id=None, include_body=True):
 366         if not include_body:
 367             raise NotImplementedError()
 368 
 369         body_expected = True
 370 
 371         header_blocks = b''
 372         body = b''
 373 
 374         while True:
 375             frm = self.read_frame()
 376             if (
 377                 (isinstance(frm, hyperframe.frame.HeadersFrame) or isinstance(frm, hyperframe.frame.ContinuationFrame)) and
 378                 (stream_id is None or frm.stream_id == stream_id)
 379             ):
 380                 stream_id = frm.stream_id
 381                 header_blocks += frm.data
 382                 if 'END_STREAM' in frm.flags:
 383                     body_expected = False
 384                 if 'END_HEADERS' in frm.flags:
 385                     break
 386             else:
 387                 self._handle_unexpected_frame(frm)
 388 
 389         while body_expected:
 390             frm = self.read_frame()
 391             if isinstance(frm, hyperframe.frame.DataFrame) and frm.stream_id == stream_id:
 392                 body += frm.data
 393                 if 'END_STREAM' in frm.flags:
 394                     break
 395             else:
 396                 self._handle_unexpected_frame(frm)
 397 
 398         headers = mitmproxy.net.http.headers.Headers(
 399             [[k, v] for k, v in self.decoder.decode(header_blocks, raw=True)]
 400         )
 401 
 402         return stream_id, headers, body
