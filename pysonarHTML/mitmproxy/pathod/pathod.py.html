  98     def http_serve_crafted(self, crafted, logctx):
  99         error, crafted = self.server.check_policy(
 100             crafted, self.settings
 101         )
 102         if error:
 103             err = self.make_http_error_response(error)
 104             language.serve(err, self.wfile, self.settings)
 105             return None, dict(
 106                 type="error",
 107                 msg=error
 108             )
 109 
 110         if self.server.explain and not hasattr(crafted, 'is_error_response'):
 111             crafted = crafted.freeze(self.settings)
 112             logctx(">> Spec: %s" % crafted.spec())
 113 
 114         response_log = language.serve(
 115             crafted,
 116             self.wfile,
 117             self.settings
 118         )
 119         if response_log["disconnect"]:
 120             return None, response_log
 121         return self.handle_http_request, response_log

 123     def handle_http_request(self, logger):
 124         """
 125             Returns a (handler, log) tuple.
 126 
 127             handler: Handler for the next request, or None to disconnect
 128             log: A dictionary, or None
 129         """
 130         with logger.ctx() as lg:
 131             try:
 132                 req = self.protocol.read_request(self.rfile)
 133             except exceptions.HttpReadDisconnect:
 134                 return None, None
 135             except exceptions.HttpException as s:
 136                 s = str(s)
 137                 lg(s)
 138                 return None, dict(type="error", msg=s)
 139 
 140             if req.method == 'CONNECT':
 141                 return self.protocol.handle_http_connect([req.host, req.port, req.http_version], lg)
 142 
 143             method = req.method
 144             path = req.path
 145             http_version = req.http_version
 146             headers = req.headers
 147             first_line_format = req.first_line_format
 148 
 149             clientcert = None
 150             if self.clientcert:
 151                 clientcert = dict(
 152                     cn=self.clientcert.cn,
 153                     subject=self.clientcert.subject,
 154                     serial=self.clientcert.serial,
 155                     notbefore=self.clientcert.notbefore.isoformat(),
 156                     notafter=self.clientcert.notafter.isoformat(),
 157                     keyinfo=self.clientcert.keyinfo,
 158                 )
 159 
 160             retlog = dict(
 161                 type="crafted",
 162                 protocol="http",
 163                 request=dict(
 164                     path=path,
 165                     method=method,
 166                     headers=headers.fields,
 167                     http_version=http_version,
 168                     sni=self.sni,
 169                     remote_address=self.address,
 170                     clientcert=clientcert,
 171                     first_line_format=first_line_format
 172                 ),
 173                 cipher=None,
 174             )
 175             if self.tls_established:
 176                 retlog["cipher"] = self.get_current_cipher()
 177 
 178             m = utils.MemBool()
 179 
 180             valid_websocket_handshake = websockets.check_handshake(headers)
 181             self.settings.websocket_key = websockets.get_client_key(headers)
 182 
 183             # If this is a websocket initiation, we respond with a proper
 184             # server response, unless over-ridden.
 185             if valid_websocket_handshake:
 186                 anchor_gen = language.parse_pathod("ws")
 187             else:
 188                 anchor_gen = None
 189 
 190             for regex, spec in self.server.anchors:
 191                 if regex.match(path):
 192                     anchor_gen = language.parse_pathod(spec, self.use_http2)
 193                     break
 194             else:
 195                 if m(path.startswith(self.server.craftanchor)):
 196                     spec = urllib.parse.unquote(path)[len(self.server.craftanchor):]
 197                     if spec:
 198                         try:
 199                             anchor_gen = language.parse_pathod(spec, self.use_http2)
 200                         except language.ParseException as v:
 201                             lg("Parse error: %s" % v.msg)
 202                             anchor_gen = iter([self.make_http_error_response(
 203                                 "Parse Error",
 204                                 "Error parsing response spec: %s\n" % (
 205                                     v.msg + v.marked()
 206                                 )
 207                             )])
 208                 else:
 209                     if self.use_http2:
 210                         anchor_gen = iter([self.make_http_error_response(
 211                             "Spec Error",
 212                             "HTTP/2 only supports request/response with the craft anchor point: %s" %
 213                             self.server.craftanchor
 214                         )])
 215 
 216             if not anchor_gen:
 217                 anchor_gen = iter([self.make_http_error_response(
 218                     "Not found",
 219                     "No valid craft request found"
 220                 )])
 221 
 222             spec = next(anchor_gen)
 223 
 224             if self.use_http2 and isinstance(spec, language.http2.Response):
 225                 spec.stream_id = req.stream_id
 226 
 227             lg("crafting spec: %s" % spec)
 228             nexthandler, retlog["response"] = self.http_serve_crafted(
 229                 spec,
 230                 lg
 231             )
 232             if nexthandler and valid_websocket_handshake:
 233                 self.protocol = protocols.websockets.WebsocketsProtocol(self)
 234                 return self.protocol.handle_websocket, retlog
 235             else:
 236                 return nexthandler, retlog
 237 
 238     def make_http_error_response(self, reason, body=None):
 239         resp = self.protocol.make_error_response(reason, body)
 240         resp.is_error_response = True
 241         return resp

 243     def handle(self):
 244         self.settimeout(self.server.timeout)
 245 
 246         if self.server.ssl:
 247             try:
 248                 cert, key, _ = self.server.ssloptions.get_cert(None)
 249                 self.convert_to_tls(
 250                     cert,
 251                     key,
 252                     handle_sni=self.handle_sni,
 253                     request_client_cert=self.server.ssloptions.request_client_cert,
 254                     cipher_list=self.server.ssloptions.ciphers,
 255                     method=self.server.ssloptions.ssl_version,
 256                     options=self.server.ssloptions.ssl_options,
 257                     alpn_select=self.server.ssloptions.alpn_select,
 258                 )
 259             except exceptions.TlsException as v:
 260                 s = str(v)
 261                 self.server.add_log(
 262                     dict(
 263                         type="error",
 264                         msg=s
 265                     )
 266                 )
 267                 log.write_raw(self.logfp, s)
 268                 return
 269 
 270             alp = self.get_alpn_proto_negotiated()
 271             if alp == b'h2':
 272                 self.protocol = protocols.http2.HTTP2Protocol(self)
 273                 self.use_http2 = True
 274 
 275         if not self.protocol:
 276             self.protocol = protocols.http.HTTPProtocol(self)
 277 
 278         lr = self.rfile if self.server.logreq else None
 279         lw = self.wfile if self.server.logresp else None
 280         logger = log.ConnectionLogger(self.logfp, self.server.hexdump, True, lr, lw)
 281 
 282         self.settings.protocol = self.protocol
 283 
 284         handler = self.handle_http_request
 285 
 286         while not self.finished:
 287             handler, l = handler(logger)
 288             if l:
 289                 self.addlog(l)
 290             if not handler:
 291                 return
 292 
 293     def addlog(self, log):
 294         if self.server.logreq:
 295             log["request_bytes"] = self.rfile.get_log()
 296         if self.server.logresp:
 297             log["response_bytes"] = self.wfile.get_log()
 298         self.server.add_log(log)

 301 class Pathod(tcp.TCPServer):
 302     LOGBUF = 500

 304     def __init__(
 305         self,
 306         addr,
 307         ssl=False,
 308         ssloptions=None,
 309         craftanchor=DEFAULT_CRAFT_ANCHOR,
 310         staticdir=None,
 311         anchors=(),
 312         sizelimit=None,
 313         nocraft=False,
 314         nohang=False,
 315         timeout=None,
 316         logreq=False,
 317         logresp=False,
 318         explain=False,
 319         hexdump=False,
 320         http2_framedump=False,
 321         webdebug=False,
 322         logfp=sys.stdout,
 323     ):
 324         """
 325             addr: (address, port) tuple. If port is 0, a free port will be
 326             automatically chosen.
 327             ssloptions: an SSLOptions object.
 328             craftanchor: URL prefix specifying the path under which to anchor
 329             response generation.
 330             staticdir: path to a directory of static resources, or None.
 331             anchors: List of (regex object, language.Request object) tuples, or
 332             None.
 333             sizelimit: Limit size of served data.
 334             nocraft: Disable response crafting.
 335             nohang: Disable pauses.
 336         """
 337         tcp.TCPServer.__init__(self, addr)
 338         self.ssl = ssl
 339         self.ssloptions = ssloptions or SSLOptions()
 340         self.staticdir = staticdir
 341         self.craftanchor = craftanchor
 342         self.sizelimit = sizelimit
 343         self.nocraft = nocraft
 344         self.nohang = nohang
 345         self.timeout, self.logreq = timeout, logreq
 346         self.logresp, self.hexdump = logresp, hexdump
 347         self.http2_framedump = http2_framedump
 348         self.explain = explain
 349         self.logfp = logfp
 350 
 351         self.log = []
 352         self.logid = 0
 353         self.anchors = anchors
 354 
 355         self.settings = language.Settings(
 356             staticdir=self.staticdir
 357         )
 358 
 359         self.loglock = threading.Lock()
 360 
 361     def check_policy(self, req, settings):
 362         """
 363             A policy check that verifies the request size is within limits.
 364         """
 365         if self.nocraft:
 366             return "Crafting disabled.", None
 367         try:
 368             req = req.resolve(settings)
 369             l = req.maximum_length(settings)
 370         except language.FileAccessDenied:
 371             return "File access denied.", None
 372         if self.sizelimit and l > self.sizelimit:
 373             return "Response too large.", None
 374         pauses = [isinstance(i, language.actions.PauseAt) for i in req.actions]
 375         if self.nohang and any(pauses):
 376             return "Pauses have been disabled.", None
 377         return None, req

 379     def handle_client_connection(self, request, client_address):
 380         h = PathodHandler(
 381             request,
 382             client_address,
 383             self,
 384             self.logfp,
 385             self.settings,
 386             self.http2_framedump,
 387         )
 388         try:
 389             h.handle()
 390             h.finish()
 391         except exceptions.TcpDisconnect:  # pragma: no cover
 392             log.write_raw(self.logfp, "Disconnect")
 393             self.add_log(
 394                 dict(
 395                     type="error",
 396                     msg="Disconnect"
 397                 )
 398             )
 399             return
 400         except exceptions.TcpTimeout:
 401             log.write_raw(self.logfp, "Timeout")
 402             self.add_log(
 403                 dict(
 404                     type="timeout",
 405                 )
 406             )
 407             return

 409     def add_log(self, d):
 410         with self.loglock:
 411             d["id"] = self.logid
 412             self.log.insert(0, d)
 413             if len(self.log) > self.LOGBUF:
 414                 self.log.pop()
 415             self.logid += 1
 416         return d["id"]
 417 
 418     def clear_log(self):
 419         with self.loglock:
 420             self.log = []
 421 
 422     def log_by_id(self, identifier):
 423         with self.loglock:
 424             for i in self.log:
 425                 if i["id"] == identifier:
 426                     return i
 427 
 428     def get_log(self):
 429         with self.loglock:
 430             return self.log
