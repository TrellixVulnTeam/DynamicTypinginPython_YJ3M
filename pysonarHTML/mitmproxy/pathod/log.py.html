   7 def write_raw(fp, lines, timestamp=True):
   8     if fp:
   9         if timestamp:
  10             fp.write(human.format_timestamp(time.time()))
  11         for i in lines:
  12             fp.write(i)
  13         fp.write("\n")
  14         fp.flush()

  17 class LogCtx:
  18 
  19     def __init__(self, fp, hex, timestamp, rfile, wfile):
  20         self.lines = []
  21         self.fp = fp
  22         self.suppressed = False
  23         self.hex = hex
  24         self.timestamp = timestamp
  25         self.rfile, self.wfile = rfile, wfile
  26 
  27     def __enter__(self):
  28         if self.wfile:
  29             self.wfile.start_log()
  30         if self.rfile:
  31             self.rfile.start_log()
  32         return self
  33 
  34     def __exit__(self, exc_type, exc_value, traceback):
  35         wlog = self.wfile.get_log() if self.wfile else None
  36         rlog = self.rfile.get_log() if self.rfile else None
  37         if self.suppressed or not self.fp:
  38             return
  39         if wlog:
  40             self("Bytes written:")
  41             self.dump(wlog, self.hex)
  42         if rlog:
  43             self("Bytes read:")
  44             self.dump(rlog, self.hex)
  45         if self.lines:
  46             write_raw(
  47                 self.fp,
  48                 [
  49                     "\n".join(self.lines),
  50                 ],
  51                 timestamp = self.timestamp
  52             )
  53         if exc_value:
  54             raise exc_value
  55 
  56     def suppress(self):
  57         self.suppressed = True
  58 
  59     def dump(self, data, hexdump):
  60         if hexdump:
  61             for line in strutils.hexdump(data):
  62                 self("\t%s %s %s" % line)
  63         else:
  64             data = strutils.always_str(
  65                 strutils.escape_control_characters(
  66                     data
  67                         .decode("ascii", "replace")
  68                         .replace(u"\ufffd", u".")
  69                 )
  70             )
  71             for i in data.split("\n"):
  72                 self("\t%s" % i)
  73 
  74     def __call__(self, line):
  75         self.lines.append(line)
