  25 class Train():
  26     """ The Faceswap Training Process.
  27 
  28     The training process is responsible for training a model on a set of source faces and a set of
  29     destination faces.
  30 
  31     The training process is self contained and should not be referenced by any other scripts, so it
  32     contains no public properties.
  33 
  34     Parameters
  35     ----------
  36     arguments: argparse.Namespace
  37         The arguments to be passed to the training process as generated from Faceswap's command
  38         line arguments
  39     """
  40     def __init__(self, arguments):
  41         logger.debug("Initializing %s: (args: %s", self.__class__.__name__, arguments)
  42         self._args = arguments
  43         self._timelapse = self._set_timelapse()
  44         self._images = self._get_images()
  45         self._stop = False
  46         self._save_now = False
  47         self._preview_buffer = dict()
  48         self._lock = Lock()
  49 
  50         self.trainer_name = self._args.trainer
  51         logger.debug("Initialized %s", self.__class__.__name__)
  52 
  53     @property
  54     def _image_size(self):
  55         """ int: The training image size. Reads the first image in the training folder and returns
  56         the size. """
  57         image = read_image(self._images["a"][0], raise_error=True)
  58         size = image.shape[0]
  59         logger.debug("Training image size: %s", size)
  60         return size
  61 
  62     @property
  63     def _alignments_paths(self):
  64         """ dict: The alignments paths for each of the source and destination faces. Key is the
  65             side, value is the path to the alignments file """
  66         alignments_paths = dict()
  67         for side in ("a", "b"):
  68             alignments_path = getattr(self._args, "alignments_path_{}".format(side))
  69             if not alignments_path:
  70                 image_path = getattr(self._args, "input_{}".format(side))
  71                 alignments_path = os.path.join(image_path, "alignments.fsa")
  72             alignments_paths[side] = alignments_path
  73         logger.debug("Alignments paths: %s", alignments_paths)
  74         return alignments_paths

  76     def _set_timelapse(self):
  77         """ Set time-lapse paths if requested.
  78 
  79         Returns
  80         -------
  81         dict
  82             The time-lapse keyword arguments for passing to the trainer
  83 
  84         """
  85         if (not self._args.timelapse_input_a and
  86                 not self._args.timelapse_input_b and
  87                 not self._args.timelapse_output):
  88             return None
  89         if (not self._args.timelapse_input_a or
  90                 not self._args.timelapse_input_b or
  91                 not self._args.timelapse_output):
  92             raise FaceswapError("To enable the timelapse, you have to supply all the parameters "
  93                                 "(--timelapse-input-A, --timelapse-input-B and "
  94                                 "--timelapse-output).")
  95 
  96         timelapse_output = str(get_folder(self._args.timelapse_output))
  97 
  98         for folder in (self._args.timelapse_input_a, self._args.timelapse_input_b):
  99             if folder is not None and not os.path.isdir(folder):
 100                 raise FaceswapError("The Timelapse path '{}' does not exist".format(folder))
 101             exts = [os.path.splitext(fname)[-1] for fname in os.listdir(folder)]
 102             if not any(ext in _image_extensions for ext in exts):
 103                 raise FaceswapError("The Timelapse path '{}' does not contain any valid "
 104                                     "images".format(folder))
 105         kwargs = {"input_a": self._args.timelapse_input_a,
 106                   "input_b": self._args.timelapse_input_b,
 107                   "output": timelapse_output}
 108         logger.debug("Timelapse enabled: %s", kwargs)
 109         return kwargs

 111     def _get_images(self):
 112         """ Check the image folders exist and contains images and obtain image paths.
 113 
 114         Returns
 115         -------
 116         dict
 117             The image paths for each side. The key is the side, the value is the list of paths
 118             for that side.
 119         """
 120         logger.debug("Getting image paths")
 121         images = dict()
 122         for side in ("a", "b"):
 123             image_dir = getattr(self._args, "input_{}".format(side))
 124             if not os.path.isdir(image_dir):
 125                 logger.error("Error: '%s' does not exist", image_dir)
 126                 sys.exit(1)
 127 
 128             images[side] = get_image_paths(image_dir)
 129             if not images[side]:
 130                 logger.error("Error: '%s' contains no images", image_dir)
 131                 sys.exit(1)
 132 
 133         logger.info("Model A Directory: %s", self._args.input_a)
 134         logger.info("Model B Directory: %s", self._args.input_b)
 135         logger.debug("Got image paths: %s", [(key, str(len(val)) + " images")
 136                                              for key, val in images.items()])
 137         return images

 139     def process(self):
 140         """ The entry point for triggering the Training Process.
 141 
 142         Should only be called from  :class:`lib.cli.launcher.ScriptExecutor`
 143         """
 144         logger.debug("Starting Training Process")
 145         logger.info("Training data directory: %s", self._args.model_dir)
 146 
 147         # TODO Move these args to config and remove these deprecation warnings
 148         if hasattr(self._args, "warp_to_landmarks") and self._args.warp_to_landmarks:
 149             deprecation_warning("`-wl`, ``--warp-to-landmarks``",
 150                                 additional_info="This option will be available within training "
 151                                                 "config settings (/config/train.ini).")
 152         if hasattr(self._args, "no_augment_color") and self._args.no_augment_color:
 153             deprecation_warning("`-nac`, ``--no-augment-color``",
 154                                 additional_info="This option will be available within training "
 155                                                 "config settings (/config/train.ini).")
 156         thread = self._start_thread()
 157         # from lib.queue_manager import queue_manager; queue_manager.debug_monitor(1)
 158 
 159         err = self._monitor(thread)
 160 
 161         self._end_thread(thread, err)
 162         logger.debug("Completed Training Process")

 164     def _start_thread(self):
 165         """ Put the :func:`_training` into a background thread so we can keep control.
 166 
 167         Returns
 168         -------
 169         :class:`lib.multithreading.MultiThread`
 170             The background thread for running training
 171         """
 172         logger.debug("Launching Trainer thread")
 173         thread = MultiThread(target=self._training)
 174         thread.start()
 175         logger.debug("Launched Trainer thread")
 176         return thread
 177 
 178     def _end_thread(self, thread, err):
 179         """ Output message and join thread back to main on termination.
 180 
 181         Parameters
 182         ----------
 183         thread: :class:`lib.multithreading.MultiThread`
 184             The background training thread
 185         err: bool
 186             Whether an error has been detected in :func:`_monitor`
 187         """
 188         logger.debug("Ending Training thread")
 189         if err:
 190             msg = "Error caught! Exiting..."
 191             log = logger.critical
 192         else:
 193             msg = ("Exit requested! The trainer will complete its current cycle, "
 194                    "save the models and quit (This can take a couple of minutes "
 195                    "depending on your training speed).")
 196             if not self._args.redirect_gui:
 197                 msg += " If you want to kill it now, press Ctrl + c"
 198             log = logger.info
 199         log(msg)
 200         self._stop = True
 201         thread.join()
 202         sys.stdout.flush()
 203         logger.debug("Ended training thread")

 205     def _training(self):
 206         """ The training process to be run inside a thread. """
 207         try:
 208             sleep(1)  # Let preview instructions flush out to logger
 209             logger.debug("Commencing Training")
 210             logger.info("Loading data, this may take a while...")
 211 
 212             if self._args.allow_growth:
 213                 self._set_tf_allow_growth()
 214             model = self._load_model()
 215             trainer = self._load_trainer(model)
 216             self._run_training_cycle(model, trainer)
 217         except KeyboardInterrupt:
 218             try:
 219                 logger.debug("Keyboard Interrupt Caught. Saving Weights and exiting")
 220                 model.save_models()
 221                 trainer.clear_tensorboard()
 222             except KeyboardInterrupt:
 223                 logger.info("Saving model weights has been cancelled!")
 224             sys.exit(0)
 225         except Exception as err:
 226             raise err

 228     def _load_model(self):
 229         """ Load the model requested for training.
 230 
 231         Returns
 232         -------
 233         :file:`plugins.train.model` plugin
 234             The requested model plugin
 235         """
 236         logger.debug("Loading Model")
 237         model_dir = get_folder(self._args.model_dir)
 238         configfile = self._args.configfile if hasattr(self._args, "configfile") else None
 239         augment_color = not self._args.no_augment_color
 240         model = PluginLoader.get_model(self.trainer_name)(
 241             model_dir,
 242             gpus=self._args.gpus,
 243             configfile=configfile,
 244             snapshot_interval=self._args.snapshot_interval,
 245             no_logs=self._args.no_logs,
 246             warp_to_landmarks=self._args.warp_to_landmarks,
 247             augment_color=augment_color,
 248             no_flip=self._args.no_flip,
 249             training_image_size=self._image_size,
 250             alignments_paths=self._alignments_paths,
 251             preview_scale=self._args.preview_scale,
 252             pingpong=self._args.pingpong,
 253             memory_saving_gradients=self._args.memory_saving_gradients,
 254             optimizer_savings=self._args.optimizer_savings,
 255             predict=False)
 256         logger.debug("Loaded Model")
 257         return model

 281     def _run_training_cycle(self, model, trainer):
 282         """ Perform the training cycle.
 283 
 284         Handles the background training, updating previews/time-lapse on each save interval,
 285         and saving the model.
 286 
 287         Parameters
 288         ----------
 289         model: :file:`plugins.train.model` plugin
 290             The requested model plugin
 291         trainer: :file:`plugins.train.trainer` plugin
 292             The requested model trainer plugin
 293         """
 294         logger.debug("Running Training Cycle")
 295         if self._args.write_image or self._args.redirect_gui or self._args.preview:
 296             display_func = self._show
 297         else:
 298             display_func = None
 299 
 300         for iteration in range(0, self._args.iterations):
 301             logger.trace("Training iteration: %s", iteration)
 302             save_iteration = iteration % self._args.save_interval == 0
 303             viewer = display_func if save_iteration or self._save_now else None
 304             timelapse = self._timelapse if save_iteration else None
 305             trainer.train_one_step(viewer, timelapse)
 306             if self._stop:
 307                 logger.debug("Stop received. Terminating")
 308                 break
 309             if save_iteration:
 310                 logger.trace("Save Iteration: (iteration: %s", iteration)
 311                 if self._args.pingpong:
 312                     model.save_models()
 313                     trainer.pingpong.switch()
 314                 else:
 315                     model.save_models()
 316             elif self._save_now:
 317                 logger.trace("Save Requested: (iteration: %s", iteration)
 318                 model.save_models()
 319                 self._save_now = False
 320         logger.debug("Training cycle complete")
 321         model.save_models()
 322         trainer.clear_tensorboard()
 323         self._stop = True

 325     def _monitor(self, thread):
 326         """ Monitor the background :func:`_training` thread for key presses and errors.
 327 
 328         Returns
 329         -------
 330         bool
 331             ``True`` if there has been an error in the background thread otherwise ``False``
 332         """
 333         is_preview = self._args.preview
 334         logger.debug("Launching Monitor")
 335         logger.info("===================================================")
 336         logger.info("  Starting")
 337         if is_preview:
 338             logger.info("  Using live preview")
 339         logger.info("  Press '%s' to save and quit",
 340                     "Stop" if self._args.redirect_gui or self._args.colab else "ENTER")
 341         if not self._args.redirect_gui and not self._args.colab:
 342             logger.info("  Press 'S' to save model weights immediately")
 343         logger.info("===================================================")
 344 
 345         keypress = KBHit(is_gui=self._args.redirect_gui)
 346         err = False
 347         while True:
 348             try:
 349                 if is_preview:
 350                     with self._lock:
 351                         for name, image in self._preview_buffer.items():
 352                             cv2.imshow(name, image)  # pylint: disable=no-member
 353                     cv_key = cv2.waitKey(1000)  # pylint: disable=no-member
 354                 else:
 355                     cv_key = None
 356 
 357                 if thread.has_error:
 358                     logger.debug("Thread error detected")
 359                     err = True
 360                     break
 361                 if self._stop:
 362                     logger.debug("Stop received")
 363                     break
 364 
 365                 # Preview Monitor
 366                 if is_preview and (cv_key == ord("\n") or cv_key == ord("\r")):
 367                     logger.debug("Exit requested")
 368                     break
 369                 if is_preview and cv_key == ord("s"):
 370                     logger.info("Save requested")
 371                     self._save_now = True
 372 
 373                 # Console Monitor
 374                 if keypress.kbhit():
 375                     console_key = keypress.getch()
 376                     if console_key in ("\n", "\r"):
 377                         logger.debug("Exit requested")
 378                         break
 379                     if console_key in ("s", "S"):
 380                         logger.info("Save requested")
 381                         self._save_now = True
 382 
 383                 sleep(1)
 384             except KeyboardInterrupt:
 385                 logger.debug("Keyboard Interrupt received")
 386                 break
 387         keypress.set_normal_term()
 388         logger.debug("Closed Monitor")
 389         return err

 391     @staticmethod
 392     def _set_tf_allow_growth():
 393         """ Allow TensorFlow to manage VRAM growth.
 394 
 395         Enables the Tensorflow allow_growth option if requested in the command line arguments
 396         """
 397         # pylint: disable=no-member
 398         logger.debug("Setting Tensorflow 'allow_growth' option")
 399         config = tf.ConfigProto()
 400         config.gpu_options.allow_growth = True
 401         config.gpu_options.visible_device_list = "0"
 402         set_session(tf.Session(config=config))
 403         logger.debug("Set Tensorflow 'allow_growth' option")

 405     def _show(self, image, name=""):
 406         """ Generate the preview and write preview file output.
 407 
 408         Handles the output and display of preview images.
 409 
 410         Parameters
 411         ----------
 412         image: :class:`numpy.ndarray`
 413             The preview image to be displayed and/or written out
 414         name: str, optional
 415             The name of the image for saving or display purposes. If an empty string is passed
 416             then it will automatically be names. Default: ""
 417         """
 418         logger.trace("Updating preview: (name: %s)", name)
 419         try:
 420             scriptpath = os.path.realpath(os.path.dirname(sys.argv[0]))
 421             if self._args.write_image:
 422                 logger.trace("Saving preview to disk")
 423                 img = "training_preview.jpg"
 424                 imgfile = os.path.join(scriptpath, img)
 425                 cv2.imwrite(imgfile, image)  # pylint: disable=no-member
 426                 logger.trace("Saved preview to: '%s'", img)
 427             if self._args.redirect_gui:
 428                 logger.trace("Generating preview for GUI")
 429                 img = ".gui_training_preview.jpg"
 430                 imgfile = os.path.join(scriptpath, "lib", "gui",
 431                                        ".cache", "preview", img)
 432                 cv2.imwrite(imgfile, image)  # pylint: disable=no-member
 433                 logger.trace("Generated preview for GUI: '%s'", img)
 434             if self._args.preview:
 435                 logger.trace("Generating preview for display: '%s'", name)
 436                 with self._lock:
 437                     self._preview_buffer[name] = image
 438                 logger.trace("Generated preview for display: '%s'", name)
 439         except Exception as err:
 440             logging.error("could not preview sample")
 441             raise err
 442         logger.trace("Updated preview: (name: %s)", name)
