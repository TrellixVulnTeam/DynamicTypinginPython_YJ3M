  47     def set_fonts(self):
  48         """ Set global default font """
  49         tk.font.nametofont("TkFixedFont").configure(size=self._config.default_font[1])
  50         for font in ("TkDefaultFont", "TkHeadingFont", "TkMenuFont"):
  51             tk.font.nametofont(font).configure(family=self._config.default_font[0],
  52                                                size=self._config.default_font[1])

  56         """ Set global custom styles """
  57         gui_style = ttk.Style()

  60     def build_gui(self, rebuild=False):
  61         """ Build the GUI """
  62         logger.debug("Building GUI")
  63         if not rebuild:
  64             self.tk.call('wm', 'iconphoto', self._w, get_images().icons["favicon"])
  65             self.configure(menu=MainMenuBar(self))
  66 
  67         if rebuild:
  68             objects = list(self.objects.keys())
  69             for obj in objects:
  70                 self.objects[obj].destroy()
  71                 del self.objects[obj]
  72 
  73         self.objects["taskbar"] = TaskBar(self)
  74         self.add_containers()
  75 
  76         self.objects["command"] = CommandNotebook(self.objects["container_top"])
  77         self.objects["display"] = DisplayNotebook(self.objects["container_top"])
  78         self.objects["console"] = ConsoleOut(self.objects["container_bottom"],
  79                                              self._init_args["debug"])
  80         self.set_initial_focus()
  81         self.set_layout()
  82         self._config.set_default_options()
  83         logger.debug("Built GUI")

 113     def set_initial_focus(self):
 114         """ Set the tab focus from settings """
 115         tab = self._config.user_config_dict["tab"]
 116         logger.debug("Setting focus for tab: %s", tab)
 117         self._config.set_active_tab_by_name(tab)
 118         logger.debug("Focus set to: %s", tab)
 119 
 120     def set_layout(self):
 121         """ Set initial layout """
 122         self.update_idletasks()
 123         config_opts = self._config.user_config_dict
 124         r_width = self.winfo_width()
 125         r_height = self.winfo_height()
 126         w_ratio = config_opts["options_panel_width"] / 100.0
 127         h_ratio = 1 - (config_opts["console_panel_height"] / 100.0)
 128         width = round(r_width * w_ratio)
 129         height = round(r_height * h_ratio)
 130         logger.debug("Setting Initial Layout: (root_width: %s, root_height: %s, width_ratio: %s, "
 131                      "height_ratio: %s, width: %s, height: %s", r_width, r_height, w_ratio,
 132                      h_ratio, width, height)
 133         self.objects["container_top"].sash_place(0, width, 1)
 134         self.objects["container_main"].sash_place(0, 1, height)
 135         self.update_idletasks()

 137     def rebuild(self):
 138         """ Rebuild the GUI on config change """
 139         logger.debug("Redrawing GUI")
 140         session_state = self._last_session.to_dict()
 141         self._config.refresh_config()
 142         get_images().__init__()
 143         self.set_fonts()
 144         self.build_gui(rebuild=True)
 145         if session_state is not None:
 146             self._last_session.from_dict(session_state)
 147         logger.debug("GUI Redrawn")

 149     def close_app(self, *args):  # pylint: disable=unused-argument
 150         """ Close Python. This is here because the graph
 151             animation function continues to run even when
 152             tkinter has gone away """
 153         logger.debug("Close Requested")
 154 
 155         if not self._confirm_close_on_running_task():
 156             return
 157         if not self._config.project.confirm_close():
 158             return
 159 
 160         if self._config.tk_vars["runningtask"].get():
 161             self.wrapper.task.terminate()
 162 
 163         self._last_session.save()
 164         get_images().delete_preview()
 165         self.quit()
 166         logger.debug("Closed GUI")
 167         sys.exit(0)
 168 
 169     def _confirm_close_on_running_task(self):
 170         """ Pop a confirmation box to close the GUI if a task is running
 171 
 172         Returns
 173         -------
 174         bool: ``True`` if user confirms close, ``False`` if user cancels close
 175         """
 176         if not self._config.tk_vars["runningtask"].get():
 177             logger.debug("No tasks currently running")
 178             return True
 179 
 180         confirmtxt = "Processes are still running.\n\nAre you sure you want to exit?"
 181         if not messagebox.askokcancel("Close", confirmtxt, default="cancel", icon="warning"):
 182             logger.debug("Close Cancelled")
 183             return True
 184         logger.debug("Close confirmed")
 185         return False
