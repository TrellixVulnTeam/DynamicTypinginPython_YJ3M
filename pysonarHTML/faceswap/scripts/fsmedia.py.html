  63         Default: False
  64     """
  65     def __init__(self, arguments, is_extract, input_is_video=False):
  66         logger.debug("Initializing %s: (is_extract: %s, input_is_video: %s)",
  67                      self.__class__.__name__, is_extract, input_is_video)
  68         self._args = arguments
  69         self._is_extract = is_extract
  70         folder, filename = self._set_folder_filename(input_is_video)
  71         super().__init__(folder, filename=filename)
  72         logger.debug("Initialized %s", self.__class__.__name__)

  74     def _set_folder_filename(self, input_is_video):
  75         """ Return the folder and the filename for the alignments file.
  76 
  77         If the input is a video, the alignments file will be stored in the same folder
  78         as the video, with filename `<videoname>_alignments`.
  79 
  80         If the input is a folder of images, the alignments file will be stored in folder with
  81         the images and just be called 'alignments'
  82 
  83         Parameters
  84         ----------
  85         input_is_video: bool, optional
  86             ``True`` if the input to the process is a video, ``False`` if it is a folder of images.
  87 
  88         Returns
  89         -------
  90         folder: str
  91             The folder where the alignments file will be stored
  92         filename: str
  93             The filename of the alignments file
  94         """
  95         if self._args.alignments_path:
  96             logger.debug("Alignments File provided: '%s'", self._args.alignments_path)
  97             folder, filename = os.path.split(str(self._args.alignments_path))
  98         elif input_is_video:
  99             logger.debug("Alignments from Video File: '%s'", self._args.input_dir)
 100             folder, filename = os.path.split(self._args.input_dir)
 101             filename = "{}_alignments".format(os.path.splitext(filename)[0])
 102         else:
 103             logger.debug("Alignments from Input Folder: '%s'", self._args.input_dir)
 104             folder = str(self._args.input_dir)
 105             filename = "alignments"
 106         logger.debug("Setting Alignments: (folder: '%s' filename: '%s')", folder, filename)
 107         return folder, filename

 109     def _load(self):
 110         """ Override the parent :func:`~lib.alignments.Alignments._load` to handle skip existing
 111         frames and faces on extract.
 112 
 113         If skip existing has been selected, existing alignments are loaded and returned to the
 114         calling script.
 115 
 116         Returns
 117         -------
 118         dict
 119             Any alignments that have already been extracted if skip existing has been selected
 120             otherwise an empty dictionary
 121         """
 122         data = dict()
 123         if not self._is_extract:
 124             if not self.have_alignments_file:
 125                 return data
 126             data = super()._load()
 127             return data
 128 
 129         skip_existing = hasattr(self._args, 'skip_existing') and self._args.skip_existing
 130         skip_faces = hasattr(self._args, 'skip_faces') and self._args.skip_faces
 131 
 132         if not skip_existing and not skip_faces:
 133             logger.debug("No skipping selected. Returning empty dictionary")
 134             return data
 135 
 136         if not self.have_alignments_file and (skip_existing or skip_faces):
 137             logger.warning("Skip Existing/Skip Faces selected, but no alignments file found!")
 138             return data
 139 
 140         data = self._serializer.load(self.file)
 141 
 142         if skip_faces:
 143             # Remove items from alignments that have no faces so they will
 144             # be re-detected
 145             del_keys = [key for key, val in data.items() if not val]
 146             logger.debug("Frames with no faces selected for redetection: %s", len(del_keys))
 147             for key in del_keys:
 148                 if key in data:
 149                     logger.trace("Selected for redetection: '%s'", key)
 150                     del data[key]
 151         return data

 187     def _count_images(self):
 188         """ Get the number of Frames from a video file or folder of images.
 189 
 190         Returns
 191         -------
 192         int
 193             The number of frames in the image source
 194         """
 195         if self._is_video:
 196             retval = int(count_frames(self._args.input_dir, fast=True))
 197         else:
 198             retval = len(self._input_images)
 199         return retval

 201     def _check_input_folder(self):
 202         """ Check whether the input is a folder or video.
 203 
 204         Returns
 205         -------
 206         bool
 207             ``True`` if the input is a video otherwise ``False``
 208         """
 209         if not os.path.exists(self._args.input_dir):
 210             logger.error("Input location %s not found.", self._args.input_dir)
 211             sys.exit(1)
 212         if (os.path.isfile(self._args.input_dir) and
 213                 os.path.splitext(self._args.input_dir)[1].lower() in _video_extensions):
 214             logger.info("Input Video: %s", self._args.input_dir)
 215             retval = True
 216         else:
 217             logger.info("Input Directory: %s", self._args.input_dir)
 218             retval = False
 219         return retval

 221     def _get_input_images(self):
 222         """ Return the list of images or path to video file that is to be processed.
 223 
 224         Returns
 225         -------
 226         str or list
 227             Path to the video file if the input is a video otherwise list of image paths.
 228         """
 229         if self._is_video:
 230             input_images = self._args.input_dir
 231         else:
 232             input_images = get_image_paths(self._args.input_dir)

 234         return input_images

 236     def load(self):
 237         """ Generator to load frames from a folder of images or from a video file.
 238 
 239         Yields
 240         ------
 241         filename: str
 242             The filename of the current frame
 243         image: :class:`numpy.ndarray`
 244             A single frame
 245         """
 246         iterator = self._load_video_frames if self._is_video else self._load_disk_frames
 247         for filename, image in iterator():
 248             yield filename, image
 249 
 250     def _load_disk_frames(self):
 251         """ Generator to load frames from a folder of images.
 252 
 253         Yields
 254         ------
 255         filename: str
 256             The filename of the current frame
 257         image: :class:`numpy.ndarray`
 258             A single frame
 259         """
 260         logger.debug("Input is separate Frames. Loading images")
 261         for filename in self._input_images:
 262             image = read_image(filename, raise_error=False)
 263             if image is None:
 264                 continue
 265             yield filename, image
 266 
 267     def _load_video_frames(self):
 268         """ Generator to load frames from a video file.
 269 
 270         Yields
 271         ------
 272         filename: str
 273             The filename of the current frame
 274         image: :class:`numpy.ndarray`
 275             A single frame
 276         """
 277         logger.debug("Input is video. Capturing frames")
 278         vidname = os.path.splitext(os.path.basename(self._args.input_dir))[0]
 279         reader = imageio.get_reader(self._args.input_dir, "ffmpeg")
 280         for i, frame in enumerate(reader):
 281             # Convert to BGR for cv2 compatibility
 282             frame = frame[:, :, ::-1]
 283             filename = "{}_{:06d}.png".format(vidname, i + 1)
 284             logger.trace("Loading video frame: '%s'", filename)
 285             yield filename, frame
 286         reader.close()
 287 
 288     def load_one_image(self, filename):
 289         """ Obtain a single image for the given filename.
 290 
 291         Parameters
 292         ----------
 293         filename: str
 294             The filename to return the image for
 295 
 296         Returns
 297         ------
 298         :class:`numpy.ndarray`
 299             The image for the requested filename,
 300 
 301         """
 302         logger.trace("Loading image: '%s'", filename)
 303         if self._is_video:
 304             if filename.isdigit():
 305                 frame_no = filename
 306             else:
 307                 frame_no = os.path.splitext(filename)[0][filename.rfind("_") + 1:]
 308                 logger.trace("Extracted frame_no %s from filename '%s'", frame_no, filename)
 309             retval = self._load_one_video_frame(int(frame_no))
 310         else:
 311             retval = read_image(filename, raise_error=True)
 312         return retval
 313 
 314     def _load_one_video_frame(self, frame_no):
 315         """ Obtain a single frame from a video file.
 316 
 317         Parameters
 318         ----------
 319         frame_no: int
 320             The frame index for the required frame
 321 
 322         Returns
 323         ------
 324         :class:`numpy.ndarray`
 325             The image for the requested frame index,
 326         """
 327         logger.trace("Loading video frame: %s", frame_no)
 328         reader = imageio.get_reader(self._args.input_dir, "ffmpeg")
 329         reader.set_image_index(frame_no - 1)
 330         frame = reader.get_next_data()[:, :, ::-1]
 331         reader.close()
 332         return frame

 335 class PostProcess():  # pylint:disable=too-few-public-methods
 336     """ Optional pre/post processing tasks for convert and extract.
 337 
 338     Builds a pipeline of actions that have optionally been requested to be performed
 339     in this session.
 340 
 341     Parameters
 342     ----------
 343     arguments: :class:`argparse.Namespace`
 344         The command line arguments that were passed to Faceswap
 345     """
 346     def __init__(self, arguments):
 347         logger.debug("Initializing %s", self.__class__.__name__)
 348         self._args = arguments
 349         self._actions = self._set_actions()
 350         logger.debug("Initialized %s", self.__class__.__name__)
 351 
 352     def _set_actions(self):
 353         """ Compile the requested actions to be performed into a list
 354 
 355         Returns
 356         -------
 357         list
 358             The list of :class:`PostProcessAction` to be performed
 359         """
 360         postprocess_items = self._get_items()
 361         actions = list()
 362         for action, options in postprocess_items.items():
 363             options = dict() if options is None else options
 364             args = options.get("args", tuple())
 365             kwargs = options.get("kwargs", dict())
 366             args = args if isinstance(args, tuple) else tuple()
 367             kwargs = kwargs if isinstance(kwargs, dict) else dict()
 368             task = globals()[action](*args, **kwargs)
 369             if task.valid:
 370                 logger.debug("Adding Postprocess action: '%s'", task)
 371                 actions.append(task)
 372 
 373         for action in actions:
 374             action_name = camel_case_split(action.__class__.__name__)
 375             logger.info("Adding post processing item: %s", " ".join(action_name))
 376 
 377         return actions
 378 
 379     def _get_items(self):
 380         """ Check the passed in command line arguments for requested actions,
 381 
 382         For any requested actions, add the item to the actions list along with
 383         any relevant arguments and keyword arguments.
 384 
 385         Returns
 386         -------
 387         dict
 388             The name of the action to be performed as the key. Any action specific
 389             arguments and keyword arguments as the value.
 390         """
 391         postprocess_items = dict()
 392         # Debug Landmarks
 393         if (hasattr(self._args, 'debug_landmarks') and self._args.debug_landmarks):
 394             postprocess_items["DebugLandmarks"] = None
 395 
 396         # Face Filter post processing
 397         if ((hasattr(self._args, "filter") and self._args.filter is not None) or
 398                 (hasattr(self._args, "nfilter") and
 399                  self._args.nfilter is not None)):
 400 
 401             if hasattr(self._args, "detector"):
 402                 detector = self._args.detector.replace("-", "_").lower()
 403             else:
 404                 detector = "cv2_dnn"
 405             if hasattr(self._args, "aligner"):
 406                 aligner = self._args.aligner.replace("-", "_").lower()
 407             else:
 408                 aligner = "cv2_dnn"
 409 
 410             face_filter = dict(detector=detector,
 411                                aligner=aligner,
 412                                multiprocess=not self._args.singleprocess)
 413             filter_lists = dict()
 414             if hasattr(self._args, "ref_threshold"):
 415                 face_filter["ref_threshold"] = self._args.ref_threshold
 416             for filter_type in ('filter', 'nfilter'):
 417                 filter_args = getattr(self._args, filter_type, None)
 418                 filter_args = None if not filter_args else filter_args
 419                 filter_lists[filter_type] = filter_args
 420             face_filter["filter_lists"] = filter_lists
 421             postprocess_items["FaceFilter"] = {"kwargs": face_filter}
 422 
 423         logger.debug("Postprocess Items: %s", postprocess_items)
 424         return postprocess_items
 425 
 426     def do_actions(self, extract_media):
 427         """ Perform the requested optional post-processing actions on the given image.
 428 
 429         Parameters
 430         ----------
 431         extract_media: :class:`~plugins.extract.pipeline.ExtractMedia`
 432             The :class:`~plugins.extract.pipeline.ExtractMedia` object to perform the
 433             action on.
 434 
 435         Returns
 436         -------
 437         :class:`~plugins.extract.pipeline.ExtractMedia`
 438             The original :class:`~plugins.extract.pipeline.ExtractMedia` with any actions applied
 439         """
 440         for action in self._actions:
 441             logger.debug("Performing postprocess action: '%s'", action.__class__.__name__)
 442             action.process(extract_media)

 508 class FaceFilter(PostProcessAction):
 509     """ Filter in or out faces based on input image(s). Extract or Convert
 510 
 511     Parameters
 512     -----------
 513     args: tuple
 514         Unused
 515     kwargs: dict
 516         Keyword arguments for face filter:
 517 
 518         * **detector** (`str`) - The detector to use
 519 
 520         * **aligner** (`str`) - The aligner to use
 521 
 522         * **multiprocess** (`bool`) - Whether to run the extraction pipeline in single process \
 523         mode or not
 524 
 525         * **ref_threshold** (`float`) - The reference threshold for a positive match
 526 
 527         * **filter_lists** (`dict`) - The filter and nfilter image paths
 528     """
 529 
 530     def __init__(self, *args, **kwargs):
 531         super().__init__(*args, **kwargs)
 532         logger.info("Extracting and aligning face for Face Filter...")
 533         self._filter = self._load_face_filter(**kwargs)
 534         logger.debug("Initialized %s", self.__class__.__name__)
 535 
 536     def _load_face_filter(self, filter_lists, ref_threshold, aligner, detector, multiprocess):
 537         """ Set up and load the :class:`~lib.face_filter.FaceFilter`.
 538 
 539         Parameters
 540         ----------
 541         filter_lists: dict
 542             The filter and nfilter image paths
 543         ref_threshold: float
 544             The reference threshold for a positive match
 545         aligner: str
 546             The aligner to use
 547         detector: str
 548             The detector to use
 549         multiprocess: bool
 550             Whether to run the extraction pipeline in single process mode or not
 551 
 552         Returns
 553         -------
 554         :class:`~lib.face_filter.FaceFilter`
 555             The face filter
 556         """
 557         if not any(val for val in filter_lists.values()):
 558             return None
 559 
 560         facefilter = None
 561         filter_files = [self._set_face_filter(f_type, filter_lists[f_type])
 562                         for f_type in ("filter", "nfilter")]
 563 
 564         if any(filters for filters in filter_files):
 565             facefilter = FilterFunc(filter_files[0],
 566                                     filter_files[1],
 567                                     detector,
 568                                     aligner,
 569                                     multiprocess,
 570                                     ref_threshold)
 571             logger.debug("Face filter: %s", facefilter)
 572         else:
 573             self.valid = False
 574         return facefilter
 575 
 576     @staticmethod
 577     def _set_face_filter(f_type, f_args):
 578         """ Check filter files exist and add the filter file paths to a list.
 579 
 580         Parameters
 581         ----------
 582         f_type: {"filter", "nfilter"}
 583             The type of filter to create this list for
 584         f_args: str or list
 585             The filter image(s) to use
 586 
 587         Returns
 588         -------
 589         list
 590             The confirmed existing paths to filter files to use
 591         """
 592         if not f_args:
 593             return list()
 594 
 595         logger.info("%s: %s", f_type.title(), f_args)
 596         filter_files = f_args if isinstance(f_args, list) else [f_args]
 597         filter_files = list(filter(lambda fpath: Path(fpath).exists(), filter_files))
 598         if not filter_files:
 599             logger.warning("Face %s files were requested, but no files could be found. This "
 600                            "filter will not be applied.", f_type)
 601         logger.debug("Face Filter files: %s", filter_files)
 602         return filter_files

 604     def process(self, extract_media):
 605         """ Filters in or out any wanted or unwanted faces based on command line arguments.
 606 
 607         Parameters
 608         ----------
 609         extract_media: :class:`~plugins.extract.pipeline.ExtractMedia`
 610             The :class:`~plugins.extract.pipeline.ExtractMedia` object to perform the
 611             face filtering on.
 612 
 613         Returns
 614         -------
 615         :class:`~plugins.extract.pipeline.ExtractMedia`
 616             The original :class:`~plugins.extract.pipeline.ExtractMedia` with any requested filters
 617             applied
 618         """
 619         if not self._filter:
 620             return
 621         ret_faces = list()
 622         for idx, detect_face in enumerate(extract_media.detected_faces):
 623             check_item = detect_face["face"] if isinstance(detect_face, dict) else detect_face
 624             check_item.load_aligned(extract_media.image)
 625             if not self._filter.check(check_item):
 626                 logger.verbose("Skipping not recognized face: (Frame: %s Face %s)",
 627                                extract_media.filename, idx)
 628                 continue
 629             logger.trace("Accepting recognised face. Frame: %s. Face: %s",
 630                          extract_media.filename, idx)
 631             ret_faces.append(detect_face)
 632         extract_media.detected_faces = ret_faces
