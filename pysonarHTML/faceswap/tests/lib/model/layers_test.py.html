  23 def layer_test(layer_cls, kwargs={}, input_shape=None, input_dtype=None,
  24                input_data=None, expected_output=None,
  25                expected_output_dtype=None, fixed_batch_size=False):
  26     """Test routine for a layer with a single input tensor
  27     and single output tensor.
  28     """
  29     # generate input data
  30     if input_data is None:
  31         assert input_shape
  32         if not input_dtype:
  33             input_dtype = K.floatx()
  34         input_data_shape = list(input_shape)
  35         for i, var_e in enumerate(input_data_shape):
  36             if var_e is None:
  37                 input_data_shape[i] = np.random.randint(1, 4)
  38         input_data = (10 * np.random.random(input_data_shape))
  39         input_data = input_data.astype(input_dtype)
  40     else:
  41         if input_shape is None:
  42             input_shape = input_data.shape
  43         if input_dtype is None:
  44             input_dtype = input_data.dtype
  45     if expected_output_dtype is None:
  46         expected_output_dtype = input_dtype
  47 
  48     # instantiation
  49     layer = layer_cls(**kwargs)
  50 
  51     # test get_weights , set_weights at layer level
  52     weights = layer.get_weights()
  53     layer.set_weights(weights)
  54 
  55     if isinstance(layer, layers.ReflectionPadding2D):
  56         layer.build(input_shape)
  57     expected_output_shape = layer.compute_output_shape(input_shape)
  58 
  59     # test in functional API
  60     if fixed_batch_size:
  61         inp = Input(batch_shape=input_shape, dtype=input_dtype)
  62     else:
  63         inp = Input(shape=input_shape[1:], dtype=input_dtype)
  64     outp = layer(inp)
  65     assert K.dtype(outp) == expected_output_dtype
  66 
  67     # check with the functional API
  68     model = Model(inp, outp)
  69 
  70     actual_output = model.predict(input_data)
  71     actual_output_shape = actual_output.shape
  72     for expected_dim, actual_dim in zip(expected_output_shape,
  73                                         actual_output_shape):
  74         if expected_dim is not None:
  75             assert expected_dim == actual_dim
  76 
  77     if expected_output is not None:
  78         assert_allclose(actual_output, expected_output, rtol=1e-3)
  79 
  80     # test serialization, weight setting at model level
  81     model_config = model.get_config()
  82     recovered_model = model.__class__.from_config(model_config)
  83     if model.weights:
  84         weights = model.get_weights()
  85         recovered_model.set_weights(weights)
  86         _output = recovered_model.predict(input_data)
  87         assert_allclose(_output, actual_output, rtol=1e-3)
  88 
  89     # test training mode (e.g. useful when the layer has a
  90     # different behavior at training and testing time).
  91     if has_arg(layer.call, 'training'):
  92         model.compile('rmsprop', 'mse')
  93         model.train_on_batch(input_data, actual_output)
  94 
  95     # test instantiation from layer config
  96     layer_config = layer.get_config()
  97     layer_config['batch_input_shape'] = input_shape
  98     layer = layer.__class__.from_config(layer_config)
  99 
 100     # for further checks in the caller function
 101     return actual_output
