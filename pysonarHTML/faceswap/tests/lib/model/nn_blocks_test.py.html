  24 def block_test(layer_func, kwargs={}, input_shape=None):
  25     """Test routine for faceswap neural network blocks.
  26 
  27     Tests are simple and are to ensure that the blocks compile on both tensorflow
  28     and plaidml backends
  29     """
  30     # generate input data
  31     assert input_shape
  32     input_dtype = K.floatx()
  33     input_data_shape = list(input_shape)
  34     for i, var_e in enumerate(input_data_shape):
  35         if var_e is None:
  36             input_data_shape[i] = np.random.randint(1, 4)
  37     input_data = (10 * np.random.random(input_data_shape))
  38     input_data = input_data.astype(input_dtype)
  39     expected_output_dtype = input_dtype
  40 
  41     # test in functional API
  42     inp = Input(shape=input_shape[1:], dtype=input_dtype)
  43     outp = layer_func(inp, **kwargs)
  44     assert K.dtype(outp) == expected_output_dtype
  45 
  46     # check with the functional API
  47     model = Model(inp, outp)
  48 
  49     actual_output = model.predict(input_data)
  50 
  51     # test serialization, weight setting at model level
  52     model_config = model.get_config()
  53     recovered_model = model.__class__.from_config(model_config)
  54     if model.weights:
  55         weights = model.get_weights()
  56         recovered_model.set_weights(weights)
  57         _output = recovered_model.predict(input_data)
  58         assert_allclose(_output, actual_output, rtol=1e-3)
  59 
  60     # for further checks in the caller function
  61     return actual_output

  65 def test_blocks(use_icnr_init, use_convaware_init, use_reflect_padding):
  66     """ Test for all blocks contained within the NNBlocks Class """
  67     cls_ = NNBlocks(use_icnr_init=use_icnr_init,
  68                     use_convaware_init=use_convaware_init,
  69                     use_reflect_padding=use_reflect_padding)
  70     block_test(cls_.conv2d, input_shape=(2, 5, 5, 128), kwargs=dict(filters=1024, kernel_size=3))
  71     block_test(cls_.conv, input_shape=(2, 8, 8, 32), kwargs=dict(filters=64))
  72     block_test(cls_.conv_sep, input_shape=(2, 8, 8, 32), kwargs=dict(filters=64))
  73     block_test(cls_.upscale, input_shape=(2, 4, 4, 128), kwargs=dict(filters=64))
  74     block_test(cls_.res_block, input_shape=(2, 2, 2, 64), kwargs=dict(filters=64))
