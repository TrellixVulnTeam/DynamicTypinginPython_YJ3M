  23 class Mask():  # pylint:disable=too-few-public-methods
  24     """ This tool is part of the Faceswap Tools suite and should be called from
  25     ``python tools.py mask`` command.
  26 
  27     Faceswap Masks tool. Generate masks from existing alignments files, and output masks
  28     for preview.
  29 
  30     Parameters
  31     ----------
  32     arguments: :class:`argparse.Namespace`
  33         The :mod:`argparse` arguments as passed in from :mod:`tools.py`
  34     """
  35     def __init__(self, arguments):
  36         logger.debug("Initializing %s: (arguments: %s", self.__class__.__name__, arguments)
  37         self._update_type = arguments.processing
  38         self._input_is_faces = arguments.input_type == "faces"
  39         self._mask_type = arguments.masker
  40         self._output = dict(opts=dict(blur_kernel=arguments.blur_kernel,
  41                                       threshold=arguments.threshold),
  42                             type=arguments.output_type,
  43                             full_frame=arguments.full_frame,
  44                             suffix=self._get_output_suffix(arguments))
  45         self._counts = dict(face=0, skip=0, update=0)
  46 
  47         self._check_input(arguments.input)
  48         self._saver = self._set_saver(arguments)
  49         loader = FacesLoader if self._input_is_faces else ImagesLoader
  50         self._loader = loader(arguments.input)
  51         self._alignments = Alignments(os.path.dirname(arguments.alignments),
  52                                       filename=os.path.basename(arguments.alignments))
  53 
  54         self._extractor = self._get_extractor()
  55         self._extractor_input_thread = self._feed_extractor()
  56 
  57         logger.debug("Initialized %s", self.__class__.__name__)

  59     def _check_input(self, mask_input):
  60         """ Check the input is valid. If it isn't exit with a logged error
  61 
  62         Parameters
  63         ----------
  64         mask_input: str
  65             Path to the input folder/video
  66         """
  67         if not os.path.exists(mask_input):
  68             logger.error("Location cannot be found: '%s'", mask_input)
  69             sys.exit(0)
  70         if os.path.isfile(mask_input) and self._input_is_faces:
  71             logger.error("Input type 'faces' was selected but input is not a folder: '%s'",
  72                          mask_input)
  73             sys.exit(0)
  74         logger.debug("input '%s' is valid", mask_input)

  76     def _set_saver(self, arguments):
  77         """ set the saver in a background thread
  78 
  79         Parameters
  80         ----------
  81         arguments: :class:`argparse.Namespace`
  82             The :mod:`argparse` arguments as passed in from :mod:`tools.py`
  83 
  84         Returns
  85         -------
  86         ``None`` or :class:`lib.image.ImagesSaver`:
  87             If output is requested, returns a :class:`lib.image.ImagesSaver` otherwise
  88             returns ``None``
  89         """
  90         if not hasattr(arguments, "output") or arguments.output is None or not arguments.output:
  91             if self._update_type == "output":
  92                 logger.error("Processing set as 'output' but no output folder provided.")
  93                 sys.exit(0)
  94             logger.debug("No output provided. Not creating saver")
  95             return None
  96         output_dir = str(get_folder(arguments.output, make_folder=True))
  97         logger.info("Saving preview masks to: '%s'", output_dir)
  98         saver = ImagesSaver(output_dir)
  99         logger.debug(saver)
 100         return saver

 102     def _get_extractor(self):
 103         """ Obtain a Mask extractor plugin and launch it
 104 
 105         Returns
 106         -------
 107         :class:`plugins.extract.pipeline.Extractor`:
 108             The launched Extractor
 109         """
 110         if self._update_type == "output":
 111             logger.debug("Update type `output` selected. Not launching extractor")
 112             return None
 113         logger.debug("masker: %s", self._mask_type)
 114         extractor = Extractor(None, None, self._mask_type,
 115                               image_is_aligned=self._input_is_faces)
 116         extractor.launch()
 117         logger.debug(extractor)
 118         return extractor

 120     def _feed_extractor(self):
 121         """ Feed the input queue to the Extractor from a faces folder or from source frames in a
 122         background thread
 123 
 124         Returns
 125         -------
 126         :class:`lib.multithreading.Multithread`:
 127             The thread that is feeding the extractor.
 128         """
 129         masker_input = getattr(self,
 130                                "_input_{}".format("faces" if self._input_is_faces else "frames"))
 131         logger.debug("masker_input: %s", masker_input)
 132 
 133         args = tuple() if self._update_type == "output" else (self._extractor.input_queue, )
 134         input_thread = MultiThread(masker_input, *args, thread_count=1)
 135         input_thread.start()
 136         logger.debug(input_thread)
 137         return input_thread

 139     def _input_faces(self, *args):
 140         """ Input pre-aligned faces to the Extractor plugin inside a thread
 141 
 142         Parameters
 143         ----------
 144         args: tuple
 145             The arguments that are to be loaded inside this thread. Contains the queue that the
 146             faces should be put to
 147         """
 148         logger.debug("args: %s", args)
 149         if self._update_type != "output":
 150             queue = args[0]
 151         for filename, image, hsh in tqdm(self._loader.load(), total=self._loader.count):
 152             if hsh not in self._alignments.hashes_to_frame:
 153                 self._counts["skip"] += 1
 154                 logger.warning("Skipping face not in alignments file: '%s'", filename)
 155                 continue
 156 
 157             frames = self._alignments.hashes_to_frame[hsh]
 158             if len(frames) > 1:
 159                 # Filter the output by filename in case of multiple frames with the same face
 160                 logger.debug("Filtering multiple hashes to current filename: (filename: '%s', "
 161                              "frames: %s", filename, frames)
 162                 lookup = os.path.splitext(os.path.basename(filename))[0]
 163                 frames = {k: v
 164                           for k, v in frames.items()
 165                           if lookup.startswith(os.path.splitext(k)[0])}
 166                 logger.debug("Filtered: (filename: '%s', frame: '%s')", filename, frames)
 167 
 168             for frame, idx in frames.items():
 169                 self._counts["face"] += 1
 170                 alignment = self._alignments.get_faces_in_frame(frame)[idx]
 171                 if self._check_for_missing(frame, idx, alignment):
 172                     continue
 173                 detected_face = self._get_detected_face(alignment)
 174                 if self._update_type == "output":
 175                     detected_face.image = image
 176                     self._save(frame, idx, detected_face)
 177                 else:
 178                     queue.put(ExtractMedia(filename, image, detected_faces=[detected_face]))
 179                     self._counts["update"] += 1
 180         if self._update_type != "output":
 181             queue.put("EOF")
 182 
 183     def _input_frames(self, *args):
 184         """ Input frames to the Extractor plugin inside a thread
 185 
 186         Parameters
 187         ----------
 188         args: tuple
 189             The arguments that are to be loaded inside this thread. Contains the queue that the
 190             faces should be put to
 191         """
 192         logger.debug("args: %s", args)
 193         if self._update_type != "output":
 194             queue = args[0]
 195         for filename, image in tqdm(self._loader.load(), total=self._loader.count):
 196             frame = os.path.basename(filename)
 197             if not self._alignments.frame_exists(frame):
 198                 self._counts["skip"] += 1
 199                 logger.warning("Skipping frame not in alignments file: '%s'", frame)
 200                 continue
 201             if not self._alignments.frame_has_faces(frame):
 202                 logger.debug("Skipping frame with no faces: '%s'", frame)
 203                 continue
 204 
 205             faces_in_frame = self._alignments.get_faces_in_frame(frame)
 206             self._counts["face"] += len(faces_in_frame)
 207 
 208             # To keep face indexes correct/cover off where only one face in an image is missing a
 209             # mask where there are multiple faces we process all faces again for any frames which
 210             # have missing masks.
 211             if all(self._check_for_missing(frame, idx, alignment)
 212                    for idx, alignment in enumerate(faces_in_frame)):
 213                 continue
 214 
 215             detected_faces = [self._get_detected_face(alignment) for alignment in faces_in_frame]
 216             if self._update_type == "output":
 217                 for idx, detected_face in enumerate(detected_faces):
 218                     detected_face.image = image
 219                     self._save(frame, idx, detected_face)
 220             else:
 221                 self._counts["update"] += len(detected_faces)
 222                 queue.put(ExtractMedia(filename, image, detected_faces=detected_faces))
 223         if self._update_type != "output":
 224             queue.put("EOF")

 226     def _check_for_missing(self, frame, idx, alignment):
 227         """ Check if the alignment is missing the requested mask_type
 228 
 229         Parameters
 230         ----------
 231         frame: str
 232             The frame name in the alignments file
 233         idx: int
 234             The index of the face for this frame in the alignments file
 235         alignment: dict
 236             The alignment for a face
 237 
 238         Returns
 239         -------
 240         bool:
 241             ``True`` if the update_type is "missing" and the mask does not exist in the alignments
 242             file otherwise ``False``
 243         """
 244         retval = (self._update_type == "missing" and
 245                   alignment.get("mask", None) is not None and
 246                   alignment["mask"].get(self._mask_type, None) is not None)
 247         if retval:
 248             logger.debug("Mask pre-exists for face: '%s' - %s", frame, idx)
 249         return retval

 251     def _get_output_suffix(self, arguments):
 252         """ The filename suffix, based on selected output options.
 253 
 254         Parameters
 255         ----------
 256         arguments: :class:`argparse.Namespace`
 257             The command line arguments for the mask tool
 258 
 259         Returns
 260         -------
 261         str:
 262             The suffix to be appended to the output filename
 263         """
 264         sfx = "{}_mask_preview_".format(self._mask_type)
 265         sfx += "face_" if not arguments.full_frame or self._input_is_faces else "frame_"
 266         sfx += "{}.png".format(arguments.output_type)
 267         return sfx

 271         """ Convert an alignment dict item to a detected_face object
 272 
 273         Parameters
 274         ----------
 275         alignment: dict
 276             The alignment dict for a face

 278         Returns
 279         -------

 287     def process(self):
 288         """ The entry point for the Mask tool from :file:`lib.tools.cli`. Runs the Mask process """
 289         logger.debug("Starting masker process")
 290         updater = getattr(self, "_update_{}".format("faces" if self._input_is_faces else "frames"))
 291         if self._update_type != "output":
 292             for extractor_output in self._extractor.detected_faces():
 293                 self._extractor_input_thread.check_and_raise_error()
 294                 updater(extractor_output)
 295             self._extractor_input_thread.join()
 296             if self._counts["update"] != 0:
 297                 self._alignments.backup()
 298                 self._alignments.save()
 299         else:
 300             self._extractor_input_thread.join()
 301             self._saver.close()
 302 
 303         if self._counts["skip"] != 0:
 304             logger.warning("%s face(s) skipped due to not existing in the alignments file",
 305                            self._counts["skip"])
 306         if self._update_type != "output":
 307             if self._counts["update"] == 0:
 308                 logger.warning("No masks were updated of the %s faces seen", self._counts["face"])
 309             else:
 310                 logger.info("Updated masks for %s faces of %s",
 311                             self._counts["update"], self._counts["face"])
 312         logger.debug("Completed masker process")

 314     def _update_faces(self, extractor_output):
 315         """ Update alignments for the mask if the input type is a faces folder
 316 
 317         If an output location has been indicated, then puts the mask preview to the save queue
 318 
 319         Parameters
 320         ----------
 321         extractor_output: dict
 322             The output from the :class:`plugins.extract.pipeline.Extractor` object
 323         """
 324         for face in extractor_output.detected_faces:
 325             for frame, idx in self._alignments.hashes_to_frame[face.hash].items():
 326                 self._alignments.update_face(frame, idx, face.to_alignment())
 327                 if self._saver is not None:
 328                     face.image = extractor_output.image
 329                     self._save(frame, idx, face)
 330 
 331     def _update_frames(self, extractor_output):
 332         """ Update alignments for the mask if the input type is a frames folder or video
 333 
 334         If an output location has been indicated, then puts the mask preview to the save queue
 335 
 336         Parameters
 337         ----------
 338         extractor_output: dict
 339             The output from the :class:`plugins.extract.pipeline.Extractor` object
 340         """
 341         frame = os.path.basename(extractor_output.filename)
 342         for idx, face in enumerate(extractor_output.detected_faces):
 343             self._alignments.update_face(frame, idx, face.to_alignment())
 344             if self._saver is not None:
 345                 self._save(frame, idx, face)

 347     def _save(self, frame, idx, detected_face):
 348         """ Build the mask preview image and save
 349 
 350         Parameters
 351         ----------
 352         frame: str
 353             The frame name in the alignments file
 354         idx: int
 355             The index of the face for this frame in the alignments file
 356         detected_face: `lib.FacesDetect.detected_face`
 357             A detected_face object for a face
 358         """
 359         filename = os.path.join(self._saver.location, "{}_{}_{}".format(
 360             os.path.splitext(frame)[0],
 361             idx,
 362             self._output["suffix"]))
 363 
 364         if detected_face.mask is None or detected_face.mask.get(self._mask_type, None) is None:
 365             logger.warning("Mask type '%s' does not exist for frame '%s' index %s. Skipping",
 366                            self._mask_type, frame, idx)
 367             return
 368         image = self._create_image(detected_face)
 369         logger.trace("filename: '%s', image_shape: %s", filename, image.shape)
 370         self._saver.save(filename, image)

 372     def _create_image(self, detected_face):
 373         """ Create a mask preview image for saving out to disk
 374 
 375         Parameters
 376         ----------
 377         detected_face: `lib.FacesDetect.detected_face`
 378             A detected_face object for a face
 379 
 380         Returns
 381         numpy.ndarray:
 382             A preview image depending on the output type in one of the following forms:
 383               - Containing 3 sub images: The original face, the masked face and the mask
 384               - The mask only
 385               - The masked face
 386         """
 387         mask = detected_face.mask[self._mask_type]
 388         mask.set_blur_and_threshold(**self._output["opts"])
 389         if not self._output["full_frame"] or self._input_is_faces:
 390             if self._input_is_faces:
 391                 face = detected_face.image
 392             else:
 393                 detected_face.load_aligned(detected_face.image)
 394                 face = detected_face.aligned_face
 395             mask = cv2.resize(detected_face.mask[self._mask_type].mask,
 396                               (face.shape[1], face.shape[0]),
 397                               interpolation=cv2.INTER_CUBIC)[..., None]
 398         else:
 399             face = detected_face.image
 400             mask = mask.get_full_frame_mask(face.shape[1], face.shape[0])
 401             mask = np.expand_dims(mask, -1)
 402 
 403         height, width = face.shape[:2]
 404         if self._output["type"] == "combined":
 405             masked = (face.astype("float32") * mask.astype("float32") / 255.).astype("uint8")
 406             mask = np.tile(mask, 3)
 407             for img in (face, masked, mask):
 408                 cv2.rectangle(img, (0, 0), (width - 1, height - 1), (255, 255, 255), 1)
 409                 out_image = np.concatenate((face, masked, mask), axis=1)
 410         elif self._output["type"] == "mask":
 411             out_image = mask
 412         elif self._output["type"] == "masked":
 413             out_image = np.concatenate([face, mask], axis=-1)
 414         return out_image
