  54         if name is None and self.path is not None:
  55             self.name = os.path.basename(self.path)
  56         elif name is not None and self.path is None:

  64     def set_type_ext(self, path=None):
  65         """ Set the extension """
  66         if path is not None:
  67             self.path = path
  68         if self.path is not None:
  69             item_ext = os.path.splitext(self.path)[1].lower()
  70             if item_ext in DataItem.vid_ext:
  71                 item_type = 'vid'
  72             elif item_ext in DataItem.audio_ext:
  73                 item_type = 'audio'
  74             else:
  75                 item_type = 'dir'
  76             self.type = item_type
  77             self.ext = item_ext
  78             logger.debug("path: '%s', type: '%s', ext: '%s'", self.path, self.type, self.ext)
  79         else:
  80             return

  94     def is_type(self, item_type=None):
  95         """ Get the type """
  96         if item_type == "media":
  97             chk_type = self.type in "vid audio"
  98         elif item_type == "dir":
  99             chk_type = self.type == "dir"
 100         elif item_type == "vid":
 101             chk_type = self.type == "vid"
 102         elif item_type == "audio":
 103             chk_type = self.type == "audio"
 104         elif item_type.lower() == "none":
 105             chk_type = self.type is None
 106         else:
 107             chk_type = False
 108         logger.debug("item_type: '%s', chk_type: '%s'", item_type, chk_type)
 109         return chk_type

 163     def process(self):
 164         """ EFFMPEG Process """
 165         logger.debug("Running Effmpeg")
 166         # Format action to match the method name
 167         self.args.action = self.args.action.replace('-', '_')
 168         logger.debug("action: '%s", self.args.action)
 169 
 170         # Instantiate input DataItem object
 171         self.input = DataItem(path=self.args.input)
 172 
 173         # Instantiate output DataItem object
 174         if self.args.action in self._actions_have_dir_output:
 175             self.output = DataItem(path=self.__get_default_output())
 176         elif self.args.action in self._actions_have_vid_output:
 177             if self.__check_have_fps(self.args.fps) > 0:
 178                 self.output = DataItem(path=self.__get_default_output(),
 179                                        fps=self.args.fps)
 180             else:
 181                 self.output = DataItem(path=self.__get_default_output())
 182 
 183         if self.args.ref_vid is None \
 184                 or self.args.ref_vid == '':
 185             self.args.ref_vid = None
 186 
 187         # Instantiate ref_vid DataItem object
 188         self.ref_vid = DataItem(path=self.args.ref_vid)
 189 
 190         # Check that correct input and output arguments were provided
 191         if self.args.action in self._actions_have_dir_input and not self.input.is_type("dir"):
 192             raise ValueError("The chosen action requires a directory as its "
 193                              "input, but you entered: "
 194                              "{}".format(self.input.path))
 195         if self.args.action in self._actions_have_vid_input and not self.input.is_type("vid"):
 196             raise ValueError("The chosen action requires a video as its "
 197                              "input, but you entered: "
 198                              "{}".format(self.input.path))
 199         if self.args.action in self._actions_have_dir_output and not self.output.is_type("dir"):
 200             raise ValueError("The chosen action requires a directory as its "
 201                              "output, but you entered: "
 202                              "{}".format(self.output.path))
 203         if self.args.action in self._actions_have_vid_output and not self.output.is_type("vid"):
 204             raise ValueError("The chosen action requires a video as its "
 205                              "output, but you entered: "
 206                              "{}".format(self.output.path))
 207 
 208         # Check that ref_vid is a video when it needs to be
 209         if self.args.action in self._actions_req_ref_video:
 210             if self.ref_vid.is_type("none"):
 211                 raise ValueError("The file chosen as the reference video is "
 212                                  "not a video, either leave the field blank "
 213                                  "or type 'None': "
 214                                  "{}".format(self.ref_vid.path))
 215         elif self.args.action in self._actions_can_use_ref_video:
 216             if self.ref_vid.is_type("none"):
 217                 logger.warning("Warning: no reference video was supplied, even though "
 218                                "one may be used with the chosen action. If this is "
 219                                "intentional then ignore this warning.")
 220 
 221         # Process start and duration arguments
 222         self.start = self.parse_time(self.args.start)
 223         self.end = self.parse_time(self.args.end)
 224         if not self.__check_equals_time(self.args.end, "00:00:00"):
 225             self.duration = self.__get_duration(self.start, self.end)
 226         else:
 227             self.duration = self.parse_time(str(self.args.duration))
 228         # If fps was left blank in gui, set it to default -1.0 value
 229         if self.args.fps == '':
 230             self.args.fps = str(-1.0)
 231 
 232         # Try to set fps automatically if needed and not supplied by user
 233         if self.args.action in self._actions_req_fps \
 234                 and self.__convert_fps(self.args.fps) <= 0:
 235             if self.__check_have_fps(['r', 'i']):
 236                 _error_str = "No fps, input or reference video was supplied, "
 237                 _error_str += "hence it's not possible to "
 238                 _error_str += "'{}'.".format(self.args.action)
 239                 raise ValueError(_error_str)
 240             if self.output.fps is not None and self.__check_have_fps(['r', 'i']):
 241                 self.args.fps = self.output.fps
 242             elif self.ref_vid.fps is not None and self.__check_have_fps(['i']):
 243                 self.args.fps = self.ref_vid.fps
 244             elif self.input.fps is not None and self.__check_have_fps(['r']):
 245                 self.args.fps = self.input.fps
 246 
 247         # Processing transpose
 248         if self.args.transpose is None or \
 249                 self.args.transpose.lower() == "none":
 250             self.args.transpose = None
 251         else:
 252             self.args.transpose = self.args.transpose[1]
 253 
 254         # Processing degrees
 255         if self.args.degrees is None \
 256                 or self.args.degrees.lower() == "none" \
 257                 or self.args.degrees == '':
 258             self.args.degrees = None
 259         elif self.args.transpose is None:
 260             try:
 261                 int(self.args.degrees)
 262             except ValueError:
 263                 logger.error("You have entered an invalid value for degrees: %s",
 264                              self.args.degrees)
 265                 sys.exit(1)
 266 
 267         # Set verbosity of output
 268         self.__set_verbosity(self.args.quiet, self.args.verbose)
 269 
 270         # Set self.print_ to True if output needs to be printed to stdout
 271         if self.args.action in self._actions_have_print_output:
 272             self.print_ = True
 273 
 274         self.effmpeg_process()
 275         logger.debug("Finished Effmpeg process")
 276 
 277     def effmpeg_process(self):
 278         """ The effmpeg process """
 279         kwargs = {"input_": self.input,
 280                   "output": self.output,
 281                   "ref_vid": self.ref_vid,
 282                   "fps": self.args.fps,
 283                   "extract_ext": self.args.extract_ext,
 284                   "start": self.start,
 285                   "duration": self.duration,
 286                   "mux_audio": self.args.mux_audio,
 287                   "degrees": self.args.degrees,

 293         action(**kwargs)

 296     def extract(input_=None, output=None, fps=None,  # pylint:disable=unused-argument
 297                 extract_ext=None, start=None, duration=None, **kwargs):
 298         """ Extract video to image frames """
 299         logger.debug("input_: %s, output: %s, fps: %s, extract_ext: '%s', start: %s, duration: %s",
 300                      input_, output, fps, extract_ext, start, duration)
 301         _input_opts = Effmpeg._common_ffmpeg_args[:]
 302         if start is not None and duration is not None:
 303             _input_opts += '-ss {} -t {}'.format(start, duration)
 304         _input = {input_.path: _input_opts}
 305         _output_opts = '-y -vf fps="' + str(fps) + '" -q:v 1'
 306         _output_path = output.path + "/" + input_.name + "_%05d" + extract_ext
 307         _output = {_output_path: _output_opts}
 308         os.makedirs(output.path, exist_ok=True)
 309         logger.debug("_input: %s, _output: %s", _input, _output)
 310         Effmpeg.__run_ffmpeg(inputs=_input, outputs=_output)

 313     def gen_vid(input_=None, output=None, fps=None,  # pylint:disable=unused-argument
 314                 mux_audio=False, ref_vid=None, exe=None, **kwargs):
 315         """ Generate Video """
 316         logger.debug("input: %s, output: %s, fps: %s, mux_audio: %s, ref_vid: '%s'exe: '%s'",
 317                      input, output, fps, mux_audio, ref_vid, exe)
 318         filename = Effmpeg.__get_extracted_filename(input_.path)
 319         _input_opts = Effmpeg._common_ffmpeg_args[:]
 320         _input_path = os.path.join(input_.path, filename)
 321         _fps_arg = '-r ' + str(fps) + ' '
 322         _input_opts += _fps_arg + "-f image2 "
 323         _output_opts = '-y ' + _fps_arg + ' -c:v libx264'
 324         if mux_audio:
 325             _ref_vid_opts = '-c copy -map 0:0 -map 1:1'
 326             _output_opts = _ref_vid_opts + ' ' + _output_opts
 327             _inputs = OrderedDict([(_input_path, _input_opts), (ref_vid.path, None)])
 328         else:
 329             _inputs = {_input_path: _input_opts}
 330         _outputs = {output.path: _output_opts}
 331         logger.debug("_inputs: %s, _outputs: %s", _inputs, _outputs)
 332         Effmpeg.__run_ffmpeg(exe=exe, inputs=_inputs, outputs=_outputs)

 334     @staticmethod
 335     def get_fps(input_=None, print_=False, **kwargs):
 336         """ Get Frames per Second """
 337         logger.debug("input_: %s, print_: %s, kwargs: %s", input_, print_, kwargs)
 338         input_ = input_ if isinstance(input_, str) else input_.path
 339         logger.debug("input: %s", input_)
 340         reader = imageio.get_reader(input_, "ffmpeg")

 342         logger.debug(_fps)
 343         reader.close()
 344         if print_:
 345             logger.info("Video fps: %s", _fps)
 346         return _fps

 431     def __get_default_output(self):
 432         """ Set output to the same directory as input
 433             if the user didn't specify it. """
 434         if self.args.output == "":
 435             if self.args.action in self._actions_have_dir_output:
 436                 retval = os.path.join(self.input.dirname, 'out')
 437             elif self.args.action in self._actions_have_vid_output:
 438                 if self.input.is_type("media"):
 439                     # Using the same extension as input leads to very poor
 440                     # output quality, hence the default is mkv for now
 441                     retval = os.path.join(self.input.dirname, "out.mkv")  # + self.input.ext)
 442                 else:  # case if input was a directory
 443                     retval = os.path.join(self.input.dirname, 'out.mkv')
 444         else:
 445             retval = self.args.output
 446         logger.debug(retval)
 447         return retval

 449     def __check_have_fps(self, items):
 450         items_to_check = list()
 451         for i in items:
 452             if i == 'r':
 453                 items_to_check.append('ref_vid')
 454             elif i == 'i':
 455                 items_to_check.append('input')
 456             elif i == 'o':
 457                 items_to_check.append('output')
 458 
 459         return all(getattr(self, i).fps is None for i in items_to_check)

 461     @staticmethod
 462     def __run_ffmpeg(exe=im_ffm.get_ffmpeg_exe(), inputs=None, outputs=None):
 463         """ Run ffmpeg """
 464         logger.debug("Running ffmpeg: (exe: '%s', inputs: %s, outputs: %s", exe, inputs, outputs)
 465         ffm = FFmpeg(executable=exe, inputs=inputs, outputs=outputs)
 466         try:
 467             ffm.run(stderr=subprocess.STDOUT)
 468         except FFRuntimeError as ffe:
 469             # After receiving SIGINT ffmpeg has a 255 exit code
 470             if ffe.exit_code == 255:
 471                 pass
 472             else:
 473                 raise ValueError("An unexpected FFRuntimeError occurred: "
 474                                  "{}".format(ffe))
 475         except KeyboardInterrupt:
 476             pass  # Do nothing if voluntary interruption
 477         logger.debug("ffmpeg finished")

 479     @staticmethod
 480     def __convert_fps(fps):
 481         """ Convert to Frames per Second """
 482         if '/' in fps:
 483             _fps = fps.split('/')
 484             retval = float(_fps[0]) / float(_fps[1])
 485         else:
 486             retval = float(fps)
 487         logger.debug(retval)
 488         return retval

 490     @staticmethod
 491     def __get_duration(start_time, end_time):
 492         """ Get the duration """
 493         start = [int(i) for i in start_time.split(':')]
 494         end = [int(i) for i in end_time.split(':')]
 495         start = datetime.timedelta(hours=start[0], minutes=start[1], seconds=start[2])
 496         end = datetime.timedelta(hours=end[0], minutes=end[1], seconds=end[2])
 497         delta = end - start
 498         secs = delta.total_seconds()
 499         retval = '{:02}:{:02}:{:02}'.format(int(secs // 3600),
 500                                             int(secs % 3600 // 60),
 501                                             int(secs % 60))
 502         logger.debug(retval)
 503         return retval

 505     @staticmethod
 506     def __get_extracted_filename(path):
 507         """ Get the extracted filename """
 508         logger.debug("path: '%s'", path)
 509         filename = ''
 510         for file in os.listdir(path):
 511             if any(i in file for i in DataItem.img_ext):
 512                 filename = file
 513                 break
 514         logger.debug("sample filename: '%s'", filename)
 515         filename, img_ext = os.path.splitext(filename)
 516         zero_pad = Effmpeg.__get_zero_pad(filename)
 517         name = filename[:-zero_pad]
 518         retval = "{}%{}d{}".format(name, zero_pad, img_ext)
 519         logger.debug("filename: %s, img_ext: '%s', zero_pad: %s, name: '%s'",
 520                      filename, img_ext, zero_pad, name)
 521         logger.debug(retval)
 522         return retval

 524     @staticmethod
 525     def __get_zero_pad(filename):
 526         """ Return the starting position of zero padding from a filename """
 527         chkstring = filename[::-1]
 528         logger.trace("filename: %s, chkstring: %s", filename, chkstring)
 529         pos = 0
 530         for pos in range(len(chkstring)):
 531             if not chkstring[pos].isdigit():
 532                 break
 533         logger.debug("filename: '%s', pos: %s", filename, pos)
 534         return pos

 553     @staticmethod
 554     def parse_time(txt):
 555         """ Parse Time """
 556         clean_txt = txt.replace(':', '')
 557         hours = clean_txt[0:2]
 558         minutes = clean_txt[2:4]
 559         seconds = clean_txt[4:6]
 560         retval = hours + ':' + minutes + ':' + seconds
 561         logger.debug("txt: '%s', retval: %s", txt, retval)
 562         return retval
