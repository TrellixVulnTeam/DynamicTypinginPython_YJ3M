  18     @staticmethod
  19     def __parse_transpose(value):
  20         index = 0
  21         opts = ["(0, 90CounterClockwise&VerticalFlip)",
  22                 "(1, 90Clockwise)",
  23                 "(2, 90CounterClockwise)",
  24                 "(3, 90Clockwise&VerticalFlip)"]
  25         if len(value) == 1:
  26             index = int(value)
  27         else:
  28             for i in range(5):
  29                 if value in opts[i]:
  30                     index = i
  31                     break
  32         return opts[index]
