  38         """ Main processing function of the sort tool """
  39

  45             logger.verbose("No output directory provided. Using input dir as output dir.")
  46             self.args.output_dir = self.args.input_dir

 129     def sort_process(self):
 130         """
 131         This method dynamically assigns the functions that will be used to run
 132         the core process of sorting, optionally grouping, renaming/moving into
 133         folders. After the functions are assigned they are executed.
 134         """
 135         sort_method = self.args.sort_method.lower()
 136         group_method = self.args.group_method.lower()
 137         final_method = self.args.final_process.lower()
 138 
 139         img_list = getattr(self, sort_method)()
 140         if "folders" in final_method:
 141             # Check if non-dissim sort method and group method are not the same
 142             if group_method.replace('group_', '') not in sort_method:
 143                 img_list = self.reload_images(group_method, img_list)
 144                 img_list = getattr(self, group_method)(img_list)
 145             else:
 146                 img_list = getattr(self, group_method)(img_list)
 147 
 148         getattr(self, final_method)(img_list)
 149 
 150         logger.info("Done.")

 187     def sort_face_cnn(self):
 188         """ Sort by landmark similarity """
 189         logger.info("Sorting by landmark similarity...")
 190         filename_list, _, landmarks = self._get_landmarks()
 191         img_list = list(zip(filename_list, landmarks))
 192 
 193         logger.info("Comparing landmarks and sorting...")
 194         img_list_len = len(img_list)
 195         for i in tqdm(range(0, img_list_len - 1), desc="Comparing...", file=sys.stdout):
 196             min_score = float("inf")
 197             j_min_score = i + 1
 198             for j in range(i + 1, img_list_len):
 199                 fl1 = img_list[i][1]
 200                 fl2 = img_list[j][1]
 201                 score = np.sum(np.absolute((fl2 - fl1).flatten()))
 202                 if score < min_score:
 203                     min_score = score
 204                     j_min_score = j
 205             (img_list[i + 1], img_list[j_min_score]) = (img_list[j_min_score], img_list[i + 1])
 206         return img_list

 292     def sort_color(self):
 293         """ Score by channel average intensity """
 294         logger.info("Sorting by channel average intensity...")
 295         desired_channel = {'gray': 0, 'luma': 0, 'orange': 1, 'green': 2}
 296         method = self.args.color_method
 297         channel_to_sort = next(v for (k, v) in desired_channel.items() if method.endswith(k))
 298         filename_list, image_list = self._get_images()
 299 
 300         logger.info("Converting to appropriate colorspace...")
 301         same_size = all(img.size == image_list[0].size for img in image_list)
 302         images = np.array(image_list, dtype='float32')[None, ...] if same_size else image_list
 303         converted_images = self._convert_color(images, same_size, method)
 304 
 305         logger.info("Scoring each image...")
 306         if same_size:
 307             scores = np.average(converted_images[0], axis=(1, 2))
 308         else:
 309             progress_bar = tqdm(converted_images, desc="Scoring", file=sys.stdout)
 310             scores = np.array([np.average(image, axis=(0, 1)) for image in progress_bar])
 311 
 312         logger.info("Sorting...")
 313         matched_list = list(zip(filename_list, scores[:, channel_to_sort]))
 314         sorted_file_img_list = sorted(matched_list, key=operator.itemgetter(1), reverse=True)
 315         return sorted_file_img_list

 318     def group_blur(self, img_list):
 319         """ Group into bins by blur """
 320         # Starting the binning process
 321         num_bins = self.args.num_bins
 322 
 323         # The last bin will get all extra images if it's
 324         # not possible to distribute them evenly
 325         num_per_bin = len(img_list) // num_bins
 326         remainder = len(img_list) % num_bins
 327 
 328         logger.info("Grouping by blur...")
 329         bins = [[] for _ in range(num_bins)]
 330         idx = 0
 331         for i in range(num_bins):
 332             for _ in range(num_per_bin):
 333                 bins[i].append(img_list[idx][0])
 334                 idx += 1
 335 
 336         # If remainder is 0, nothing gets added to the last bin.
 337         for i in range(1, remainder + 1):
 338             bins[-1].append(img_list[-i][0])
 339 
 340         return bins

 342     def group_face_cnn(self, img_list):
 343         """ Group into bins by CNN face similarity """
 344         logger.info("Grouping by face-cnn similarity...")
 345 
 346         # Groups are of the form: group_num -> reference faces
 347         reference_groups = dict()
 348 
 349         # Bins array, where index is the group number and value is
 350         # an array containing the file paths to the images in that group.
 351         bins = []
 352 
 353         # Comparison threshold used to decide how similar
 354         # faces have to be to be grouped together.
 355         # It is multiplied by 1000 here to allow the cli option to use smaller
 356         # numbers.
 357         min_threshold = self.args.min_threshold * 1000
 358 
 359         img_list_len = len(img_list)
 360 
 361         for i in tqdm(range(0, img_list_len - 1),
 362                       desc="Grouping",
 363                       file=sys.stdout):
 364             fl1 = img_list[i][1]
 365 
 366             current_best = [-1, float("inf")]
 367 
 368             for key, references in reference_groups.items():
 369                 try:
 370                     score = self.get_avg_score_faces_cnn(fl1, references)
 371                 except TypeError:
 372                     score = float("inf")
 373                 except ZeroDivisionError:
 374                     score = float("inf")
 375                 if score < current_best[1]:
 376                     current_best[0], current_best[1] = key, score
 377 
 378             if current_best[1] < min_threshold:
 379                 reference_groups[current_best[0]].append(fl1[0])
 380                 bins[current_best[0]].append(img_list[i][0])
 381             else:
 382                 reference_groups[len(reference_groups)] = [img_list[i][1]]
 383                 bins.append([img_list[i][0]])
 384 
 385         return bins

 447     def final_process_rename(self, img_list):
 448         """ Rename the files """
 449         output_dir = self.args.output_dir
 450 
 451         process_file = self.set_process_file_method(self.args.log_changes,
 452                                                     self.args.keep_original)
 453 
 454         # Make sure output directory exists
 455         if not os.path.exists(output_dir):
 456             os.makedirs(output_dir)
 457 
 458         description = (
 459             "Copying and Renaming" if self.args.keep_original
 460             else "Moving and Renaming"
 461         )
 462 
 463         for i in tqdm(range(0, len(img_list)),
 464                       desc=description,
 465                       leave=False,
 466                       file=sys.stdout):
 467             src = img_list[i] if isinstance(img_list[i], str) else img_list[i][0]
 468             src_basename = os.path.basename(src)
 469 
 470             dst = os.path.join(output_dir, '{:05d}_{}'.format(i, src_basename))
 471             try:
 472                 process_file(src, dst, self.changes)
 473             except FileNotFoundError as err:
 474                 logger.error(err)
 475                 logger.error('fail to rename %s', src)
 476 
 477         for i in tqdm(range(0, len(img_list)),
 478                       desc=description,
 479                       file=sys.stdout):
 480             renaming = self.set_renaming_method(self.args.log_changes)
 481             fname = img_list[i] if isinstance(img_list[i], str) else img_list[i][0]
 482             src, dst = renaming(fname, output_dir, i, self.changes)
 483 
 484             try:
 485                 os.rename(src, dst)
 486             except FileNotFoundError as err:
 487                 logger.error(err)
 488                 logger.error('fail to rename %s', format(src))
 489 
 490         if self.args.log_changes:
 491             self.write_to_log(self.changes)

 535     def reload_images(self, group_method, img_list):
 536         """
 537         Reloads the image list by replacing the comparative values with those
 538         that the chosen grouping method expects.
 539         :param group_method: str name of the grouping method that will be used.
 540         :param img_list: image list that has been sorted by one of the sort
 541         methods.
 542         :return: img_list but with the comparative values that the chosen
 543         grouping method expects.
 544         """
 545         logger.info("Preparing to group...")
 546         if group_method == 'group_blur':
 547             filename_list, image_list = self._get_images()
 548             blurs = [self.estimate_blur(img) for img in image_list]
 549             temp_list = list(zip(filename_list, blurs))
 550         elif group_method == 'group_face_cnn':
 551             filename_list, image_list, landmarks = self._get_landmarks()
 552             temp_list = list(zip(filename_list, landmarks))
 553         elif group_method == 'group_face_yaw':
 554             filename_list, image_list, landmarks = self._get_landmarks()
 555             yaws = [self.calc_landmarks_face_yaw(mark) for mark in landmarks]
 556             temp_list = list(zip(filename_list, yaws))
 557         elif group_method == 'group_hist':
 558             filename_list, image_list = self._get_images()
 559             histograms = [cv2.calcHist([img], [0], None, [256], [0, 256]) for img in image_list]
 560             temp_list = list(zip(filename_list, histograms))
 561         else:
 562             raise ValueError("{} group_method not found.".format(group_method))
 563 
 564         return self.splice_lists(img_list, temp_list)

 571             conversion = np.array([[0.0722], [0.7152], [0.2126]])
 572         else:
 573             conversion = np.array([[0.25, 0.5, 0.25], [-0.5, 0.0, 0.5], [-0.25, 0.5, -0.25]])
 574 
 575         if same_size:
 576             path = 'greedy'
 577             operation = 'bijk, kl -> bijl' if method.endswith('gray') else 'bijl, kl -> bijk'
 578         else:
 579             operation = 'ijk, kl -> ijl' if method.endswith('gray') else 'ijl, kl -> ijk'
 580             path = np.einsum_path(operation, imgs[0][..., :3], conversion, optimize='optimal')[0]
 581 
 582         progress_bar = tqdm(imgs, desc="Converting", file=sys.stdout)
 583         images = [np.einsum(operation, img[..., :3], conversion, optimize=path).astype('float32')
 584                   for img in progress_bar]
 585         return images

 587     @staticmethod
 588     def splice_lists(sorted_list, new_vals_list):
 589         """
 590         This method replaces the value at index 1 in each sub-list in the
 591         sorted_list with the value that is calculated for the same img_path,
 592         but found in new_vals_list.
 593 
 594         Format of lists: [[img_path, value], [img_path2, value2], ...]
 595 
 596         :param sorted_list: list that has been sorted by one of the sort
 597         methods.
 598         :param new_vals_list: list that has been loaded by a different method
 599         than the sorted_list.
 600         :return: list that is sorted in the same way as the input sorted list
 601         but the values corresponding to each image are from new_vals_list.
 602         """
 603         new_list = []
 604         # Make new list of just image paths to serve as an index
 605         val_index_list = [i[0] for i in new_vals_list]
 606         for i in tqdm(range(len(sorted_list)), desc="Splicing", file=sys.stdout):
 607             current_img = sorted_list[i] if isinstance(sorted_list[i], str) else sorted_list[i][0]
 608             new_val_index = val_index_list.index(current_img)
 609             new_list.append([current_img, new_vals_list[new_val_index][1]])
 610 
 611         return new_list

 625     @staticmethod
 626     def estimate_blur(image):
 627         """
 628         Estimate the amount of blur an image has with the variance of the Laplacian.
 629         Normalize by pixel number to offset the effect of image size on pixel gradients & variance
 630         """
 631         if image.ndim == 3:
 632             image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
 633         blur_map = cv2.Laplacian(image, cv2.CV_32F)
 634         score = np.var(blur_map) / np.sqrt(image.shape[0] * image.shape[1])
 635         return score

 721     @staticmethod
 722     def get_avg_score_hist(img1, references):
 723         """ Return the average histogram score between a face and
 724             reference image """
 725         scores = []
 726         for img2 in references:
 727             score = cv2.compareHist(img1, img2, cv2.HISTCMP_BHATTACHARYYA)
 728             scores.append(score)
 729         return sum(scores) / len(scores)

 731     @staticmethod
 732     def get_avg_score_faces_cnn(fl1, references):
 733         """ Return the average CNN similarity score
 734             between a face and reference image """
 735         scores = []
 736         for fl2 in references:
 737             score = np.sum(np.absolute((fl2 - fl1).flatten()))
 738             scores.append(score)
 739         return sum(scores) / len(scores)
