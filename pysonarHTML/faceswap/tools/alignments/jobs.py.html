  39     def get_source_dir(self, arguments):
  40         """ Set the correct source folder """
  41         if (hasattr(arguments, "faces_dir") and arguments.faces_dir and
  42                 hasattr(arguments, "frames_dir") and arguments.frames_dir):
  43             logger.error("Only select a source frames (-fr) or source faces (-fc) folder")
  44             sys.exit(1)
  45         elif hasattr(arguments, "faces_dir") and arguments.faces_dir:
  46             self.type = "faces"
  47             source_dir = arguments.faces_dir
  48         elif hasattr(arguments, "frames_dir") and arguments.frames_dir:
  49             self.type = "frames"
  50             source_dir = arguments.frames_dir
  51         else:
  52             logger.error("No source folder (-fr or -fc) was provided")
  53             sys.exit(1)
  54         logger.debug("type: '%s', source_dir: '%s'", self.type, source_dir)
  55         return source_dir

  57     def get_items(self):
  58         """ Set the correct items to process """
  59         items = globals()[self.type.title()](self.source_dir)
  60         self.is_video = items.is_video
  61         return items.file_list_sorted

  63     def process(self):
  64         """ Process the frames check against the alignments file """
  65         logger.info("[CHECK %s]", self.type.upper())
  66         items_output = self.compile_output()
  67         self.output_results(items_output)

  69     def validate(self):
  70         """ Check that the selected type is valid for
  71             selected task and job """
  72         if self.job == "missing-frames" and self.output == "move":
  73             logger.warning("Missing_frames was selected with move output, but there will "
  74                            "be nothing to move. Defaulting to output: console")
  75             self.output = "console"
  76         if self.type == "faces" and self.job not in ("multi-faces", "leftover-faces"):
  77             logger.warning("The selected folder is not valid. Faces folder (-fc) is only "
  78                            "supported for 'multi-faces' and 'leftover-faces'")
  79             sys.exit(1)

 105     def get_multi_faces_frames(self):
 106         """ Return Frames that contain multiple faces """
 107         self.output_message = "Frames with multiple faces"
 108         for item in tqdm(self.items, desc=self.output_message):

 115     def get_multi_faces_faces(self):
 116         """ Return Faces when there are multiple faces in a frame """
 117         self.output_message = "Multiple faces in frame"
 118         seen_hash_dupes = set()
 119         for item in tqdm(self.items, desc=self.output_message):
 120             filename = item["face_fullname"]
 121             f_hash = item["face_hash"]
 122             frame_idx = [(frame, idx)
 123                          for frame, idx in self.alignments.hashes_to_frame[f_hash].items()]
 124 
 125             if len(frame_idx) > 1:
 126                 # If the same hash exists in multiple frames, select arbitrary frame
 127                 # and add to seen_hash_dupes so it is not selected again
 128                 logger.trace("Dupe hashes: %s", frame_idx)
 129                 frame_idx = [f_i for f_i in frame_idx if f_i not in seen_hash_dupes][0]
 130                 seen_hash_dupes.add(frame_idx)
 131                 frame_idx = [frame_idx]
 132 
 133             frame_name, idx = frame_idx[0]
 134             if not self.alignments.frame_has_multiple_faces(frame_name):
 135                 continue
 136             retval = (filename, idx)
 137             logger.trace("Returning: '%s'", retval)
 138             yield retval

 140     def get_missing_alignments(self):
 141         """ yield each frame that does not exist in alignments file """

 148                 logger.debug("Returning: '%s'", frame_name)

 151     def get_missing_frames(self):
 152         """ yield each frame in alignments that does
 153             not have a matching file """
 154         self.output_message = "Missing frames that are in alignments file"
 155         frames = set(item["frame_fullname"] for item in self.items)
 156         for frame in tqdm(self.alignments.data.keys(), desc=self.output_message):
 157             if frame not in frames:
 158                 logger.debug("Returning: '%s'", frame)

 162         """yield each face that isn't in the alignments file."""

 170     def output_results(self, items_output):
 171         """ Output the results in the requested format """
 172         logger.trace("items_output: %s", items_output)
 173         if self.output == "move" and self.is_video and self.type == "frames":
 174             logger.warning("Move was selected with an input video. This is not possible so "
 175                            "falling back to console output")
 176             self.output = "console"
 177         if not items_output:
 178             logger.info("No %s were found meeting the criteria", self.type)
 179             return
 180         if self.output == "move":
 181             self.move_file(items_output)
 182             return
 183         if self.job in ("multi-faces", "leftover-faces") and self.type == "faces":
 184             # Strip the index for printed/file output
 185             items_output = [item[0] for item in items_output]
 186         output_message = "-----------------------------------------------\r\n"
 187         output_message += " {} ({})\r\n".format(self.output_message,
 188                                                 len(items_output))
 189         output_message += "-----------------------------------------------\r\n"
 190         output_message += "\r\n".join(items_output)
 191         if self.output == "console":
 192             for line in output_message.splitlines():
 193                 logger.info(line)
 194         if self.output == "file":
 195             self.output_file(output_message, len(items_output))

 260 class Dfl():
 261     """ Reformat Alignment file """
 262     def __init__(self, alignments, arguments):
 263         logger.debug("Initializing %s: (arguments: %s)", self.__class__.__name__, arguments)
 264         self.alignments = alignments
 265         if self.alignments.file != "dfl.fsa":
 266             logger.error("Alignments file must be specified as 'dfl' to reformat dfl alignmnets")
 267             sys.exit(1)
 268         logger.debug("Loading DFL faces")
 269         self.faces = Faces(arguments.faces_dir)
 270         logger.debug("Initialized %s", self.__class__.__name__)
 271 
 272     def process(self):
 273         """ Run reformat """
 274         logger.info("[REFORMAT DFL ALIGNMENTS]")  # Tidy up cli output
 275         self.alignments.data_from_dfl(self.load_dfl(), self.faces.folder)
 276         self.alignments.save()
 277 
 278     def load_dfl(self):
 279         """ Load alignments from DeepFaceLab and format for Faceswap """
 280         alignments = dict()
 281         for face in tqdm(self.faces.file_list_sorted, desc="Converting DFL Faces"):
 282             if face["face_extension"] not in (".png", ".jpg"):
 283                 logger.verbose("'%s' is not a png or jpeg. Skipping", face["face_fullname"])
 284                 continue
 285             f_hash = face["face_hash"]
 286             fullpath = os.path.join(self.faces.folder, face["face_fullname"])
 287             dfl = self.get_dfl_alignment(fullpath)
 288 
 289             if not dfl:
 290                 continue
 291 
 292             self.convert_dfl_alignment(dfl, f_hash, alignments)
 293         return alignments
 294 
 295     @staticmethod
 296     def get_dfl_alignment(filename):
 297         """ Process the alignment of one face """
 298         ext = os.path.splitext(filename)[1]
 299 
 300         if ext.lower() in (".jpg", ".jpeg"):
 301             img = Image.open(filename)
 302             try:
 303                 dfl_alignments = pickle.loads(img.app["APP15"])
 304                 dfl_alignments["source_rect"] = [n.item()  # comes as non-JSONable np.int32
 305                                                  for n in dfl_alignments["source_rect"]]
 306                 return dfl_alignments
 307             except pickle.UnpicklingError:
 308                 return None
 309 
 310         with open(filename, "rb") as dfl:
 311             header = dfl.read(8)
 312             if header != b"\x89PNG\r\n\x1a\n":
 313                 logger.error("No Valid PNG header: %s", filename)
 314                 return None
 315             while True:
 316                 chunk_start = dfl.tell()
 317                 chunk_hdr = dfl.read(8)
 318                 if not chunk_hdr:
 319                     break
 320                 chunk_length, chunk_name = struct.unpack("!I4s", chunk_hdr)
 321                 dfl.seek(chunk_start, os.SEEK_SET)
 322                 if chunk_name == b"fcWp":
 323                     chunk = dfl.read(chunk_length + 12)
 324                     retval = pickle.loads(chunk[8:-4])
 325                     logger.trace("Loaded DFL Alignment: (filename: '%s', alignment: %s",
 326                                  filename, retval)
 327                     return retval
 328                 dfl.seek(chunk_length+12, os.SEEK_CUR)
 329             logger.error("Couldn't find DFL alignments: %s", filename)
 330 
 331     @staticmethod
 332     def convert_dfl_alignment(dfl_alignments, f_hash, alignments):
 333         """ Add Deep Face Lab Alignments to alignments in Faceswap format """
 334         sourcefile = dfl_alignments["source_filename"]
 335         left, top, right, bottom = dfl_alignments["source_rect"]
 336         alignment = {"x": left,
 337                      "w": right - left,
 338                      "y": top,
 339                      "h": bottom - top,
 340                      "hash": f_hash,
 341                      "landmarks_xy": np.array(dfl_alignments["source_landmarks"], dtype="float32")}
 342         logger.trace("Adding alignment: (frame: '%s', alignment: %s", sourcefile, alignment)
 343         alignments.setdefault(sourcefile, dict()).setdefault("faces", []).append(alignment)

 346 class Draw():
 347     """ Draw Alignments on passed in images """
 348     def __init__(self, alignments, arguments):
 349         logger.debug("Initializing %s: (arguments: %s)", self.__class__.__name__, arguments)
 350         self.arguments = arguments
 351         self.alignments = alignments
 352         self.frames = Frames(arguments.frames_dir)
 353         self.output_folder = self.set_output()
 354         self.extracted_faces = None
 355         logger.debug("Initialized %s", self.__class__.__name__)

 357     def set_output(self):
 358         """ Set the output folder path """
 359         now = datetime.now().strftime("%Y%m%d_%H%M%S")
 360         folder_name = "drawn_landmarks_{}".format(now)
 361         if self.frames.is_video:
 362             dest_folder = os.path.dirname(self.frames.folder)
 363         else:
 364             dest_folder = self.frames.folder
 365         output_folder = os.path.join(dest_folder, folder_name)
 366         logger.debug("Creating folder: '%s'", output_folder)
 367         os.makedirs(output_folder)
 368         return output_folder
 369 
 370     def process(self):
 371         """ Run the draw alignments process """
 372         logger.info("[DRAW LANDMARKS]")  # Tidy up cli output
 373         self.extracted_faces = ExtractedFaces(self.frames, self.alignments, size=256)
 374         frames_drawn = 0
 375         for frame in tqdm(self.frames.file_list_sorted, desc="Drawing landmarks"):
 376             frame_name = frame["frame_fullname"]
 377 
 378             if not self.alignments.frame_exists(frame_name):
 379                 logger.verbose("Skipping '%s' - Alignments not found", frame_name)
 380                 continue
 381 
 382             self.annotate_image(frame_name)
 383             frames_drawn += 1
 384         logger.info("%s Frame(s) output", frames_drawn)
 385 
 386     def annotate_image(self, frame):
 387         """ Draw the alignments """
 388         logger.trace("Annotating frame: '%s'", frame)
 389         alignments = self.alignments.get_faces_in_frame(frame)
 390         image = self.frames.load_image(frame)
 391         self.extracted_faces.get_faces_in_frame(frame)
 392         original_roi = [face.original_roi
 393                         for face in self.extracted_faces.faces]
 394         annotate = Annotate(image, alignments, original_roi)
 395         annotate.draw_bounding_box(1, 1)
 396         annotate.draw_extract_box(2, 1)
 397         annotate.draw_landmarks(3, 1)
 398         annotate.draw_landmarks_mesh(4, 1)
 399 
 400         image = annotate.image
 401         self.frames.save_image(self.output_folder, frame, image)

 404 class Extract():  # pylint:disable=too-few-public-methods
 405     """ Re-extract faces from source frames based on Alignment data
 406 
 407     Parameters
 408     ----------
 409     alignments: :class:`tools.lib_alignments.media.AlignmentData`
 410         The alignments data loaded from an alignments file for this rename job
 411     arguments: :class:`argparse.Namespace`
 412         The :mod:`argparse` arguments as passed in from :mod:`tools.py`
 413     """
 414     def __init__(self, alignments, arguments):
 415         logger.debug("Initializing %s: (arguments: %s)", self.__class__.__name__, arguments)
 416         self._arguments = arguments
 417         self._alignments = alignments
 418         self._faces_dir = arguments.faces_dir
 419         self._frames = Frames(arguments.frames_dir)
 420         self._extracted_faces = ExtractedFaces(self._frames,
 421                                                self._alignments,
 422                                                size=arguments.size,
 423                                                align_eyes=arguments.align_eyes)
 424         logger.debug("Initialized %s", self.__class__.__name__)
 425 
 426     def process(self):
 427         """ Run the re-extraction from Alignments file process"""
 428         logger.info("[EXTRACT FACES]")  # Tidy up cli output
 429         self._check_folder()
 430         self._export_faces()
 431 
 432     def _check_folder(self):
 433         """ Check that the faces folder doesn't pre-exist and create. """
 434         err = None
 435         if not self._faces_dir:
 436             err = "ERROR: Output faces folder not provided."
 437         elif not os.path.isdir(self._faces_dir):
 438             logger.debug("Creating folder: '%s'", self._faces_dir)
 439             os.makedirs(self._faces_dir)
 440         elif os.listdir(self._faces_dir):
 441             err = "ERROR: Output faces folder should be empty: '{}'".format(self._faces_dir)
 442         if err:
 443             logger.error(err)
 444             sys.exit(0)
 445         logger.verbose("Creating output folder at '%s'", self._faces_dir)
 446 
 447     def _export_faces(self):
 448         """ Export the faces to the output folder and update the alignments file with
 449         new hashes. """
 450         extracted_faces = 0
 451         skip_list = self._set_skip_list()
 452         count = self._frames.count if skip_list is None else self._frames.count - len(skip_list)
 453         for filename, image in tqdm(self._frames.stream(skip_list=skip_list),
 454                                     total=count, desc="Saving extracted faces"):
 455             frame_name = os.path.basename(filename)
 456             if not self._alignments.frame_exists(frame_name):
 457                 logger.verbose("Skipping '%s' - Alignments not found", frame_name)
 458                 continue
 459             extracted_faces += self._output_faces(frame_name, image)
 460         if extracted_faces != 0 and not self._arguments.large:
 461             self._alignments.save()
 462         logger.info("%s face(s) extracted", extracted_faces)
 463 
 464     def _set_skip_list(self):
 465         """ Set the indices for frames that should be skipped based on the `extract_every_n`
 466         command line option.
 467 
 468         Returns
 469         -------
 470         list or ``None``
 471             A list of indices to be skipped if extract_every_n is not `1` otherwise
 472             returns ``None``
 473         """
 474         skip_num = self._arguments.extract_every_n
 475         if skip_num == 1:
 476             logger.debug("Not skipping any frames")
 477             return None
 478         skip_list = []
 479         for idx, item in enumerate(self._frames.file_list_sorted):
 480             if idx % skip_num != 0:
 481                 logger.trace("Adding image '%s' to skip list due to extract_every_n = %s",
 482                              item["frame_fullname"], skip_num)
 483                 skip_list.append(idx)
 484         logger.debug("Adding skip list: %s", skip_list)
 485         return skip_list
 486 
 487     def _output_faces(self, filename, image):
 488         """ For each frame save out the faces and update the face hash back to alignments
 489 
 490         Parameters
 491         ----------
 492         filename: str
 493             The filename (without the full path) of the current frame
 494         image: :class:`numpy.ndarray`
 495             The full frame that faces are to be extracted from
 496 
 497         Returns
 498         -------
 499         int
 500             The total number of faces that have been extracted
 501         """
 502         logger.trace("Outputting frame: %s", filename)
 503         face_count = 0
 504         frame_name, extension = os.path.splitext(filename)
 505         faces = self._select_valid_faces(filename, image)
 506 
 507         for idx, face in enumerate(faces):
 508             output = "{}_{}{}".format(frame_name, str(idx), extension)
 509             if self._arguments.large:
 510                 self._frames.save_image(self._faces_dir, output, face.aligned_face)
 511             else:
 512                 output = os.path.join(self._faces_dir, output)
 513                 f_hash = self._extracted_faces.save_face_with_hash(output,
 514                                                                    extension,
 515                                                                    face.aligned_face)
 516                 self._alignments.data[filename]["faces"][idx]["hash"] = f_hash
 517             face_count += 1
 518         return face_count

 520     def _select_valid_faces(self, frame, image):
 521         """ Return the aligned faces from a frame that meet the selection criteria,
 522 
 523         Parameters
 524         ----------
 525         frame: str
 526             The filename (without the full path) of the current frame
 527         image: :class:`numpy.ndarray`
 528             The full frame that faces are to be extracted from
 529 
 530         Returns
 531         -------
 532         list:
 533             List of valid :class:`lib,faces_detect.DetectedFace` objects
 534         """
 535         faces = self._extracted_faces.get_faces_in_frame(frame, image=image)
 536         if not self._arguments.large:
 537             valid_faces = faces
 538         else:
 539             sizes = self._extracted_faces.get_roi_size_for_frame(frame)
 540             valid_faces = [faces[idx] for idx, size in enumerate(sizes)
 541                            if size >= self._extracted_faces.size]
 542         logger.trace("frame: '%s', total_faces: %s, valid_faces: %s",
 543                      frame, len(faces), len(valid_faces))
 544         return valid_faces

 547 class Fix():
 548     """ Fix alignments that were impacted by the 'out by one' bug when extracting from video

 566     def _check_file_needs_fixing(self):
 567         """ Check that these alignments are in video format and that the first frame in the "
 568         "alignments file does not already start with 1 """
 569         retval = True
 570         min_frame = min(key for key in self.alignments.data.keys())
 571         logger.debug("First frame: '%s'", min_frame)
 572         fname = os.path.splitext(min_frame)[0]
 573         frame_id = fname.split("_")[-1]
 574         if ("_") not in fname or not frame_id.isdigit():
 575             logger.info("Alignments file not generated from a video. Nothing to do.")
 576             retval = False
 577         elif int(frame_id) == 1:
 578             logger.info("Alignments file does not require fixing. First frame: '%s'", fname)
 579             retval = False
 580         logger.debug(retval)
 581         return retval

 583     def _fix(self):
 584         """ Renumber frame names, reducing each one by 1 """
 585         frame_names = sorted(key for key in self.alignments.data.keys())
 586         for old_name in tqdm(frame_names, desc="Fixing Alignments file"):
 587             fname, ext = os.path.splitext(old_name)
 588             vid_name, new_frame_id = ("_".join(fname.split("_")[:-1]),
 589                                       int(fname.split("_")[-1]) - 1)
 590             new_name = "{}_{:06d}{}".format(vid_name, new_frame_id, ext)
 591             logger.debug("Re-assigning: '%s' > '%s'", old_name, new_name)
 592             self.alignments.data[new_name] = self.alignments.data[old_name]
 593             del self.alignments.data[old_name]

 596 class Merge():
 597     """ Merge two alignments files into one """
 598     def __init__(self, alignments, arguments):
 599         self.alignments = alignments
 600         self.faces = self.get_faces(arguments)
 601         self.final_alignments = alignments[0]
 602         self.process_alignments = alignments[1:]
 603         self._hashes_to_frame = None
 604 
 605     @staticmethod
 606     def get_faces(arguments):
 607         """ If faces argument is specified, load faces_dir
 608             otherwise return None """
 609         if not hasattr(arguments, "faces_dir") or not arguments.faces_dir:
 610             return None
 611         return Faces(arguments.faces_dir)
 612 
 613     def process(self):
 614         """Process the alignments file merge """
 615         logger.info("[MERGE ALIGNMENTS]")  # Tidy up cli output
 616         if self.faces is not None:
 617             self.remove_faces()
 618         self._hashes_to_frame = self.final_alignments.hashes_to_frame
 619         skip_count = 0
 620         merge_count = 0
 621         total_count = sum([alignments.frames_count for alignments in self.process_alignments])
 622 
 623         with tqdm(desc="Merging Alignments", total=total_count) as pbar:
 624             for alignments in self.process_alignments:
 625                 for _, src_alignments, _, frame in alignments.yield_faces():
 626                     for idx, alignment in enumerate(src_alignments):
 627                         if not alignment.get("hash", None):
 628                             logger.warning("Alignment '%s':%s has no Hash! Skipping", frame, idx)
 629                             skip_count += 1
 630                             continue
 631                         if self.check_exists(frame, alignment, idx):
 632                             skip_count += 1
 633                             continue
 634                         self.merge_alignment(frame, alignment, idx)
 635                         merge_count += 1
 636                     pbar.update(1)
 637         logger.info("Alignments Merged: %s", merge_count)
 638         logger.info("Alignments Skipped: %s", skip_count)
 639         if merge_count != 0:
 640             self.set_destination_filename()
 641             self.final_alignments.save()
 642 
 643     def remove_faces(self):
 644         """ Process to remove faces from an alignments file """
 645         face_hashes = list(self.faces.items.keys())
 646         del_faces_count = 0
 647         del_frames_count = 0
 648         if not face_hashes:
 649             logger.error("No face hashes. This would remove all faces from your alignments file.")
 650             return
 651         for alignments in tqdm(self.alignments, desc="Filtering out faces"):
 652             pre_face_count = alignments.faces_count
 653             pre_frames_count = alignments.frames_count
 654             alignments.filter_hashes(face_hashes, filter_out=False)
 655             # Remove frames with no faces
 656             frames = list(alignments.data.keys())
 657             for frame in frames:
 658                 if not alignments.frame_has_faces(frame):
 659                     del alignments.data[frame]
 660             post_face_count = alignments.faces_count
 661             post_frames_count = alignments.frames_count
 662             removed_faces = pre_face_count - post_face_count
 663             removed_frames = pre_frames_count - post_frames_count
 664             del_faces_count += removed_faces
 665             del_frames_count += removed_frames
 666             logger.verbose("Removed %s faces and %s frames from %s",
 667                            removed_faces, removed_frames, os.path.basename(alignments.file))
 668         logger.info("Total removed - faces: %s, frames: %s", del_faces_count, del_frames_count)
 669 
 670     def check_exists(self, frame, alignment, idx):
 671         """ Check whether this face already exists """
 672         existing_frame = self._hashes_to_frame.get(alignment["hash"], None)
 673         if not existing_frame:
 674             return False
 675         if frame in existing_frame.keys():
 676             logger.verbose("Face '%s': %s already exists in destination at position %s. "
 677                            "Skipping", frame, idx, existing_frame[frame])
 678         elif frame not in existing_frame.keys():
 679             logger.verbose("Face '%s': %s exists in destination as: %s. "
 680                            "Skipping", frame, idx, existing_frame)
 681         return True
 682 
 683     def merge_alignment(self, frame, alignment, idx):
 684         """ Merge the source alignment into the destination """
 685         logger.debug("Merging alignment: (frame: %s, src_idx: %s, hash: %s)",
 686                      frame, idx, alignment["hash"])
 687         self._hashes_to_frame.setdefault(alignment["hash"], dict())[frame] = idx
 688         self.final_alignments.data.setdefault(frame,
 689                                               dict()).setdefault("faces", []).append(alignment)
 690 
 691     def set_destination_filename(self):
 692         """ Set the destination filename """
 693         folder = os.path.split(self.final_alignments.file)[0]
 694         ext = os.path.splitext(self.final_alignments.file)[1]
 695         now = datetime.now().strftime("%Y%m%d_%H%M%S")
 696         filename = os.path.join(folder, "alignments_merged_{}{}".format(now, ext))
 697         logger.debug("Output set to: '%s'", filename)
 698         self.final_alignments.set_filename(filename)

 701 class RemoveAlignments():
 702     """ Remove items from alignments file """
 703     def __init__(self, alignments, arguments):
 704         logger.debug("Initializing %s: (arguments: %s)", self.__class__.__name__, arguments)
 705         self.alignments = alignments
 706         self.type = arguments.job.replace("remove-", "")
 707         self.items = self.get_items(arguments)
 708         self.removed = set()
 709         logger.debug("Initialized %s", self.__class__.__name__)

 711     def get_items(self, arguments):
 712         """ Set the correct items to process """
 713         retval = None
 714         if self.type == "frames":
 715             retval = Frames(arguments.frames_dir).items
 716         elif self.type == "faces":
 717             retval = Faces(arguments.faces_dir)
 718         return retval

 720     def process(self):
 721         """ run removal """
 722         logger.info("[REMOVE ALIGNMENTS DATA]")  # Tidy up cli output
 723         del_count = 0
 724         task = getattr(self, "remove_{}".format(self.type))
 725 
 726         if self.type == "frames":
 727             logger.debug("Removing Frames")
 728             for frame in tqdm(list(item[3] for item in self.alignments.yield_faces()),
 729                               desc="Removing Frames",
 730                               total=self.alignments.frames_count):
 731                 del_count += task(frame)
 732         else:
 733             logger.debug("Removing Faces")
 734             del_count = task()
 735 
 736         if del_count == 0:
 737             logger.info("No changes made to alignments file. Exiting")
 738             return
 739 
 740         logger.info("%s alignment(s) were removed from alignments file", del_count)
 741         self.alignments.save()
 742 
 743         if self.type == "faces":
 744             rename = Rename(self.alignments, None, self.items)
 745             rename.process()
 746 
 747     def remove_frames(self, frame):
 748         """ Process to remove frames from an alignments file """
 749         if frame in self.items:
 750             logger.trace("Not deleting frame: '%s'", frame)
 751             return 0
 752         logger.debug("Deleting frame: '%s'", frame)
 753         del self.alignments.data[frame]
 754         return 1
 755 
 756     def remove_faces(self):
 757         """ Process to remove faces from an alignments file """
 758         face_hashes = self.items.items
 759         if not face_hashes:
 760             logger.error("No face hashes. This would remove all faces from your alignments file.")
 761             return 0
 762         pre_face_count = self.alignments.faces_count
 763         self.alignments.filter_hashes(face_hashes, filter_out=False)
 764         post_face_count = self.alignments.faces_count
 765         return pre_face_count - post_face_count

 768 class Rename():
 769     """ Rename faces in a folder to match their filename as stored in an alignments file.
 770 
 771     Parameters
 772     ----------
 773     alignments: :class:`tools.lib_alignments.media.AlignmentData`
 774         The alignments data loaded from an alignments file for this rename job
 775     arguments: :class:`argparse.Namespace`
 776         The :mod:`argparse` arguments as passed in from :mod:`tools.py`
 777     faces: :class:`tools.lib_alignments.media.Faces`, Optional
 778         An optional faces object, if the rename task is being called by another job.
 779         Default: ``None``
 780     """
 781     def __init__(self, alignments, arguments, faces=None):
 782         logger.debug("Initializing %s: (arguments: %s, faces: %s)",
 783                      self.__class__.__name__, arguments, faces)
 784         self.alignments = alignments
 785         self.faces = faces if faces else Faces(arguments.faces_dir)
 786         logger.debug("Initialized %s", self.__class__.__name__)
 787 
 788     def process(self):
 789         """ Process the face renaming """
 790         logger.info("[RENAME FACES]")  # Tidy up cli output
 791         rename_mappings = self._build_rename_list()
 792         rename_count = self._rename_faces(rename_mappings)
 793         logger.info("%s faces renamed", rename_count)
 794 
 795     def _build_rename_list(self):
 796         """ Build a list of source and destination filenames for renaming.
 797 
 798         Validates that all files in the faces folder have a corresponding match in the alignments
 799         file. Orders the rename list by destination filename to avoid potential for filename clash.
 800 
 801         Returns
 802         -------
 803         list
 804             List of tuples of (`source filename`, `destination filename`) ordered by destination
 805             filename
 806         """
 807         source_filenames = []
 808         dest_filenames = []
 809         errors = []
 810         pbar = tqdm(desc="Building Rename Lists", total=self.faces.count)
 811         for disk_hash, disk_faces in self.faces.items.items():
 812             align_faces = self.alignments.hashes_to_frame.get(disk_hash, None)
 813             face_error = self._validate_hash_match(disk_faces, align_faces)
 814             if face_error is not None:
 815                 errors.extend(face_error)
 816                 pbar.update(len(disk_faces))
 817                 continue
 818             src_faces, dst_faces = self._get_filename_mapping(disk_faces, align_faces)
 819             source_filenames.extend(src_faces)
 820             dest_filenames.extend(dst_faces)
 821             pbar.update(len(src_faces))
 822         pbar.close()
 823         if errors:
 824             logger.error("There are faces in the given folder that do not correspond to entries "
 825                          "in the alignments file. Please check your data, and if neccesarry run "
 826                          "the `remove-faces` job. To get a list of faces missing alignments "
 827                          "entries, run with VERBOSE logging")
 828             logger.verbose("Files in faces folder not in alignments file: %s", errors)
 829             sys.exit(1)
 830         return self._sort_mappings(source_filenames, dest_filenames)
 831 
 832     @staticmethod
 833     def _validate_hash_match(disk_faces, align_faces):
 834         """ Validate that the hash has returned corresponding faces from disk and alignments file.
 835 
 836         Parameters
 837         ----------
 838         disk_faces: list
 839             List of tuples of (`file name`, `file extension`) for all faces that exist for the
 840             current hash
 841         align_faces: dict
 842             `frame filename`: `index` for all faces that exist in the alignments file for the
 843             current hash
 844 
 845         Returns
 846         -------
 847         list
 848             List of disk_faces that do not correspond to a matching entry in the alignments file.
 849             Returns `None` if there is a valid match
 850         """
 851         if align_faces is None:
 852             logger.debug("No matching hash found for faces: %s", disk_faces)
 853             return [face[0] + face[1] for face in disk_faces]
 854         if len(disk_faces) != len(align_faces):
 855             logger.debug("Number of faces mismatch for hash: (disk_faces: %s, align_faces: %s)",
 856                          disk_faces, align_faces)
 857             return [face[0] + face[1] for face in disk_faces[: len(align_faces)]]
 858         return None
 859 
 860     @staticmethod
 861     def _get_filename_mapping(disk_faces, align_faces):
 862         """ Map the source filenames for this hash to the destination filenames.
 863 
 864         Parameters
 865         ----------
 866         disk_faces: list
 867             List of tuples of (`file name`, `file extension`) for all faces that exist for the
 868             current hash
 869         align_faces: dict
 870             `frame filename`: `index` for all faces that exist in the alignments file for the
 871             current hash
 872 
 873         Returns
 874         -------
 875         source_filenames: list
 876             List of source filenames to be renamed for this hash
 877         dest_filenames: list
 878             List of destination filenames that faces for this hash are to be renamed to
 879             List of disk_faces that do not correspond to a matching entry in the alignments file.
 880             Returns `None` if there is a valid match
 881         """
 882         source_filenames = []
 883         dest_filenames = []
 884         # Force deterministic order on alignments dict for multi hash faces
 885         sorted_aligned = sorted([(frame, idx) for frame, idx in align_faces.items()])
 886         for disk_face, align_face in zip(disk_faces, sorted_aligned):
 887             extension = disk_face[1]
 888             src_fname = disk_face[0] + extension
 889 
 890             dst_frame = os.path.splitext(align_face[0])[0]
 891             dst_fname = "{}_{}{}".format(dst_frame, align_face[1], extension)
 892             logger.debug("Mapping rename from '%s' to '%s'", src_fname, dst_fname)
 893             source_filenames.append(src_fname)
 894             dest_filenames.append(dst_fname)
 895         return source_filenames, dest_filenames
 896 
 897     @staticmethod
 898     def _sort_mappings(sources, destinations):
 899         """ Sort the mapping lists by destinations to avoid filename clash.
 900 
 901         Parameters
 902         ----------
 903         sources: list
 904             List of source filenames in the same order as :attr:`destinations`
 905         destinations: dict
 906             List of destination filenames in the same order as :attr:`sources`
 907 
 908         Returns
 909         -------
 910         list
 911             List of tuples of (`source filename`, `destination filename`) ordered by destination
 912             filename
 913         """
 914         sorted_indices = [idx for idx, _ in sorted(enumerate(destinations), key=lambda x: x[1])]
 915         mappings = [(sources[idx], destinations[idx]) for idx in sorted_indices]
 916         logger.trace("filename mappings: %s", mappings)
 917         return mappings
 918 
 919     def _rename_faces(self, filename_mappings):
 920         """ Rename faces back to their original name as exists in the alignments file.
 921 
 922         If the source and destination filename are the same then skip that file.
 923 
 924         Parameters
 925         ----------
 926         filename_mappings: list
 927             List of tuples of (`source filename`, `destination filename`) ordered by destination
 928             filename
 929 
 930         Returns
 931         -------
 932         int
 933             The number of faces that have been renamed
 934         """
 935         rename_count = 0
 936         for src, dst in tqdm(filename_mappings, desc="Renaming Faces"):
 937             if src == dst:
 938                 logger.debug("Skipping rename of '%s' as destination name is same as souce", src)
 939                 continue
 940             old = os.path.join(self.faces.folder, src)
 941             new = os.path.join(self.faces.folder, dst)
 942             if os.path.exists(new):
 943                 # This should never happen, but is a safety measure to prevent deletion of faces
 944                 # when multiple files have the same hash.
 945                 logger.debug("Skipping renaming to an existing file: (src: '%s', dst: '%s'",
 946                              src, dst)
 947                 continue
 948             logger.verbose("Renaming '%s' to '%s'", old, new)
 949             os.rename(old, new)
 950             rename_count += 1
 951         return rename_count

 982         reindexed = 0
 983         for alignment in tqdm(self.alignments.yield_faces(),
 984                               desc="Sort alignment indexes", total=self.alignments.frames_count):
 985             frame, alignments, count, key = alignment

 995             reindexed += 1
 996         logger.info("%s Frames had their faces reindexed", reindexed)

1000 class Spatial():
1001     """ Apply spatial temporal filtering to landmarks
1002         Adapted from:
1003         https://www.kaggle.com/selfishgene/animating-and-smoothing-3d-facial-keypoints/notebook """
1004 
1005     def __init__(self, alignments, arguments):
1006         logger.debug("Initializing %s: (arguments: %s)", self.__class__.__name__, arguments)
1007         self.arguments = arguments
1008         self.alignments = alignments
1009         self.mappings = dict()
1010         self.normalized = dict()
1011         self.shapes_model = None
1012         logger.debug("Initialized %s", self.__class__.__name__)
1013 
1014     def process(self):
1015         """ Perform spatial filtering """
1016         logger.info("[SPATIO-TEMPORAL FILTERING]")  # Tidy up cli output
1017         logger.info("NB: The process only processes the alignments for the first "
1018                     "face it finds for any given frame. For best results only run this when "
1019                     "there is only a single face in the alignments file and all false positives "
1020                     "have been removed")
1021 
1022         self.normalize()
1023         self.shape_model()
1024         landmarks = self.spatially_filter()
1025         landmarks = self.temporally_smooth(landmarks)
1026         self.update_alignments(landmarks)
1027         self.alignments.save()
1028 
1029         logger.info("Done! To re-extract faces run: python tools.py "
1030                     "alignments -j extract -a %s -fr <path_to_frames_dir> -fc "
1031                     "<output_folder>", self.arguments.alignments_file)
1032 
1033     # Define shape normalization utility functions
1034     @staticmethod
1035     def normalize_shapes(shapes_im_coords):
1036         """ Normalize a 2D or 3D shape """
1037         logger.debug("Normalize shapes")
1038         (num_pts, num_dims, _) = shapes_im_coords.shape
1039 
1040         # Calculate mean coordinates and subtract from shapes
1041         mean_coords = shapes_im_coords.mean(axis=0)
1042         shapes_centered = np.zeros(shapes_im_coords.shape)
1043         shapes_centered = shapes_im_coords - np.tile(mean_coords, [num_pts, 1, 1])
1044 
1045         # Calculate scale factors and divide shapes
1046         scale_factors = np.sqrt((shapes_centered**2).sum(axis=1)).mean(axis=0)
1047         shapes_normalized = np.zeros(shapes_centered.shape)
1048         shapes_normalized = shapes_centered / np.tile(scale_factors, [num_pts, num_dims, 1])
1049 
1050         logger.debug("Normalized shapes: (shapes_normalized: %s, scale_factors: %s, mean_coords: "
1051                      "%s", shapes_normalized, scale_factors, mean_coords)
1052         return shapes_normalized, scale_factors, mean_coords
1053 
1054     @staticmethod
1055     def normalized_to_original(shapes_normalized, scale_factors, mean_coords):
1056         """ Transform a normalized shape back to original image coordinates """
1057         logger.debug("Normalize to original")
1058         (num_pts, num_dims, _) = shapes_normalized.shape
1059 
1060         # move back to the correct scale
1061         shapes_centered = shapes_normalized * np.tile(scale_factors, [num_pts, num_dims, 1])
1062         # move back to the correct location
1063         shapes_im_coords = shapes_centered + np.tile(mean_coords, [num_pts, 1, 1])
1064 
1065         logger.debug("Normalized to original: %s", shapes_im_coords)
1066         return shapes_im_coords
1067 
1068     def normalize(self):
1069         """ Compile all original and normalized alignments """
1070         logger.debug("Normalize")
1071         count = sum(1 for val in self.alignments.data.values() if val["faces"])
1072         landmarks_all = np.zeros((68, 2, int(count)))
1073 
1074         end = 0
1075         for key in tqdm(sorted(self.alignments.data.keys()), desc="Compiling"):
1076             val = self.alignments.data[key]["faces"]
1077             if not val:
1078                 continue
1079             # We should only be normalizing a single face, so just take
1080             # the first landmarks found
1081             landmarks = np.array(val[0]["landmarks_xy"]).reshape(68, 2, 1)
1082             start = end
1083             end = start + landmarks.shape[2]
1084             # Store in one big array
1085             landmarks_all[:, :, start:end] = landmarks
1086             # Make sure we keep track of the mapping to the original frame
1087             self.mappings[start] = key
1088 
1089         # Normalize shapes
1090         normalized_shape = self.normalize_shapes(landmarks_all)
1091         self.normalized["landmarks"] = normalized_shape[0]
1092         self.normalized["scale_factors"] = normalized_shape[1]
1093         self.normalized["mean_coords"] = normalized_shape[2]
1094         logger.debug("Normalized: %s", self.normalized)

1096     def shape_model(self):
1097         """ build 2D shape model """
1098         logger.debug("Shape model")
1099         landmarks_norm = self.normalized["landmarks"]
1100         num_components = 20
1101         normalized_shapes_tbl = np.reshape(landmarks_norm, [68*2, landmarks_norm.shape[2]]).T
1102         self.shapes_model = decomposition.PCA(n_components=num_components,
1103                                               whiten=True,
1104                                               random_state=1).fit(normalized_shapes_tbl)
1105         explained = self.shapes_model.explained_variance_ratio_.sum()
1106         logger.info("Total explained percent by PCA model with %s components is %s%%",
1107                     num_components, round(100 * explained, 1))
1108         logger.debug("Shaped model")

1110     def spatially_filter(self):
1111         """ interpret the shapes using our shape model
1112             (project and reconstruct) """
1113         logger.debug("Spatially Filter")
1114         landmarks_norm = self.normalized["landmarks"]
1115         # Convert to matrix form
1116         landmarks_norm_table = np.reshape(landmarks_norm, [68 * 2, landmarks_norm.shape[2]]).T
1117         # Project onto shapes model and reconstruct
1118         landmarks_norm_table_rec = self.shapes_model.inverse_transform(
1119             self.shapes_model.transform(landmarks_norm_table))
1120         # Convert back to shapes (numKeypoint, num_dims, numFrames)
1121         landmarks_norm_rec = np.reshape(landmarks_norm_table_rec.T,
1122                                         [68, 2, landmarks_norm.shape[2]])
1123         # Transform back to image co-ordinates
1124         retval = self.normalized_to_original(landmarks_norm_rec,
1125                                              self.normalized["scale_factors"],
1126                                              self.normalized["mean_coords"])
1127 
1128         logger.debug("Spatially Filtered: %s", retval)
1129         return retval

1131     @staticmethod
1132     def temporally_smooth(landmarks):
1133         """ apply temporal filtering on the 2D points """
1134         logger.debug("Temporally Smooth")
1135         filter_half_length = 2
1136         temporal_filter = np.ones((1, 1, 2 * filter_half_length + 1))
1137         temporal_filter = temporal_filter / temporal_filter.sum()
1138 
1139         start_tileblock = np.tile(landmarks[:, :, 0][:, :, np.newaxis], [1, 1, filter_half_length])
1140         end_tileblock = np.tile(landmarks[:, :, -1][:, :, np.newaxis], [1, 1, filter_half_length])
1141         landmarks_padded = np.dstack((start_tileblock, landmarks, end_tileblock))
1142 
1143         retval = signal.convolve(landmarks_padded, temporal_filter, mode='valid', method='fft')
1144         logger.debug("Temporally Smoothed: %s", retval)
1145         return retval

1159 class UpdateHashes():
1160     """ Update hashes in an alignments file """
1161     def __init__(self, alignments, arguments):
1162         logger.debug("Initializing %s: (arguments: %s)", self.__class__.__name__, arguments)
1163         self.alignments = alignments
1164         self.faces = Faces(arguments.faces_dir).file_list_sorted
1165         self.face_hashes = dict()
1166         logger.debug("Initialized %s", self.__class__.__name__)
1167 
1168     def process(self):
1169         """ Update Face Hashes to the alignments file """
1170         logger.info("[UPDATE FACE HASHES]")  # Tidy up cli output
1171         self.get_hashes()
1172         updated = self.update_hashes()
1173         if updated == 0:
1174             logger.info("No hashes were updated. Exiting")
1175             return
1176         self.alignments.save()
1177         logger.info("%s frame(s) had their face hashes updated.", updated)
1178 
1179     def get_hashes(self):
1180         """ Read the face hashes from the faces """
1181         logger.info("Getting original filenames, indexes and hashes...")
1182         for face in self.faces:
1183             filename = face["face_name"]
1184             extension = face["face_extension"]
1185             if "_" not in face["face_name"]:
1186                 logger.warning("Unable to determine index of file. Skipping: '%s'", filename)
1187                 continue
1188             index = filename[filename.rfind("_") + 1:]
1189             if not index.isdigit():
1190                 logger.warning("Unable to determine index of file. Skipping: '%s'", filename)
1191                 continue
1192             orig_frame = filename[:filename.rfind("_")] + extension
1193             self.face_hashes.setdefault(orig_frame, dict())[int(index)] = face["face_hash"]
1194 
1195     def update_hashes(self):
1196         """ Update hashes to alignments """
1197         logger.info("Updating hashes to alignments...")
1198         updated = 0
1199         for frame, hashes in self.face_hashes.items():
1200             if not self.alignments.frame_exists(frame):
1201                 logger.warning("Frame not found in alignments file. Skipping: '%s'", frame)
1202                 continue
1203             if not self.alignments.frame_has_faces(frame):
1204                 logger.warning("Frame does not have faces. Skipping: '%s'", frame)
1205                 continue
1206             existing = [face.get("hash", None)
1207                         for face in self.alignments.get_faces_in_frame(frame)]
1208             if any(hsh not in existing for hsh in list(hashes.values())):
1209                 self.alignments.add_face_hashes(frame, hashes)
1210                 updated += 1
1211         return updated
