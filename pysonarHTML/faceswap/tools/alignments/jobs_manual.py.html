 175     def delete_alignment(self, *args):  # pylint: disable=unused-argument
 176         """ Save alignments """
 177         logger.debug("Deleting Alignments")
 178         selected_face = self.get_selected_face_id()
 179         if self.get_edit_mode() == "View" or selected_face is None:
 180             logger.debug("Delete received, but edit mode is 'View'. Not deleting")
 181             return
 182         frame = self.get_frame_name()
 183         if self.alignments.delete_face_at_index(frame, selected_face):
 184             self.state["edit"]["selected"] = None
 185             self.state["edit"]["updated"] = True
 186             self.state["edit"]["update_faces"] = True
 187             self.set_redraw(True)

 189     def copy_alignments(self, *args):
 190         """ Copy the alignments from the previous or next frame
 191             to the current frame """
 192         logger.debug("Copying Alignments")
 193         if self.get_edit_mode() != "Edit":
 194             logger.debug("Copy received, but edit mode is not 'Edit'. Not copying")
 195             return
 196         frame_id = self.get_next_face_idx(args[1])
 197         if not 0 <= frame_id <= self.state["navigation"]["max_frame"]:
 198             return
 199         current_frame = self.get_frame_name()
 200         get_frame = self.frames.file_list_sorted[frame_id]["frame_fullname"]
 201         alignments = self.alignments.get_faces_in_frame(get_frame)
 202         for alignment in alignments:
 203             self.alignments. add_face(current_frame, alignment)
 204         self.state["edit"]["updated"] = True
 205         self.state["edit"]["update_faces"] = True
 206         self.set_redraw(True)

 209         """ Toggle state of requested item """

 216     def iterate_state(self, item, category):
 217         """ Cycle through options (6 possible or 3 currently supported) """
 218         logger.debug("Cycling state: (item: %s, category: %s)", item, category)
 219         if category == "color":
 220             max_val = 7
 221         elif category == "frame-size":
 222             max_val = 6
 223         elif category == "skip-mode":
 224             max_val = 4
 225         else:
 226             max_val = 3
 227         val = self.state[item][category]
 228         val = val + 1 if val != max_val else 1
 229         self.state[item][category] = val
 230         logger.debug("Cycled state: (item: %s, category: %s, value: %s)",
 231                      item, category, self.state[item][category])
 232         self.set_redraw(True)
 233 
 234     def set_state_value(self, item, category, value):
 235         """ Set state of requested item or toggle off """
 236         logger.debug("Setting state value: (item: %s, category: %s, value: %s)",
 237                      item, category, value)
 238         state = self.state[item][category]
 239         value = str(value) if value is not None else value
 240         if state == value:
 241             self.state[item][category] = None
 242         else:
 243             self.state[item][category] = value
 244         logger.debug("Setting state value: (item: %s, category: %s, value: %s)",
 245                      item, category, self.state[item][category])
 246         self.set_redraw(True)
 247 
 248     def iterate_frame(self, *args):
 249         """ Iterate frame up or down, stopping at either end """
 250         logger.debug("Iterating frame: (args: %s)", args)
 251         iteration = args[1]
 252         max_frame = self.state["navigation"]["max_frame"]
 253         if iteration in ("first", "last"):
 254             next_frame = 0 if iteration == "first" else max_frame
 255             self.state["navigation"]["frame_idx"] = next_frame
 256             self.state["navigation"]["last_request"] = 0
 257             self.set_redraw(True)
 258             return
 259 
 260         current_frame = self.state["navigation"]["frame_idx"]
 261         next_frame = current_frame + iteration
 262         end = 0 if iteration < 0 else max_frame
 263         if (max_frame == 0 or
 264                 (end > 0 and next_frame >= end) or
 265                 (end == 0 and next_frame <= end)):
 266             next_frame = end
 267         self.state["navigation"]["frame_idx"] = next_frame
 268         self.state["navigation"]["last_request"] = iteration
 269         self.set_state_value("edit", "selected", None)

 326     def get_next_face_idx(self, increment):
 327         """Get the index of the previous or next frame which has a face"""
 328         navigation = self.state["navigation"]
 329         frame_list = self.frames.file_list_sorted
 330         frame_idx = navigation["frame_idx"] + increment
 331         while True:
 332             if not 0 <= frame_idx <= navigation["max_frame"]:
 333                 break
 334             frame = frame_list[frame_idx]["frame_fullname"]
 335             if not self.alignments.frame_has_faces(frame):
 336                 frame_idx += increment
 337             else:
 338                 break
 339         return frame_idx

 343     """ Generate and display help in cli and in window """

 345         logger.debug("Initializing %s: (interface: %s)", self.__class__.__name__, interface)

 350     def generate(self):
 351         """ Generate help output """
 352         logger.debug("Generating help")
 353         sections = ("navigation", "display", "color", "size", "edit")
 354         helpout = {section: list() for section in sections}
 355         helptext = ""
 356         for key, val in self.interface.controls.items():
 357             logger.trace("Generating help for:(key: '%s', val: '%s'", key, val)
 358             help_section = val["args"][0]
 359             if help_section not in ("navigation", "edit"):
 360                 help_section = val["args"][1]
 361             key_text = val.get("key_text", None)
 362             key_text = key_text if key_text else key
 363             logger.trace("Adding help for:(section: '%s', val: '%s', text: '%s'",
 364                          help_section, val["help"], key_text)
 365             helpout[help_section].append((val["help"], key_text))
 366 
 367         helpout["edit"].append(("Bounding Box - Move", "Left Click"))
 368         helpout["edit"].append(("Bounding Box - Resize", "Middle Click"))
 369 
 370         for section in sections:
 371             spacer = "=" * int((40 - len(section)) / 2)
 372             display = "\n{0} {1} {0}\n".format(spacer, section.upper())
 373             helpsection = sorted(helpout[section])
 374             if section == "navigation":
 375                 helpsection = sorted(helpout[section], reverse=True)
 376             display += "\n".join("  - '{}': {}".format(item[1], item[0])
 377                                  for item in helpsection)
 378 
 379             helptext += display
 380         logger.debug("Added helptext: '%s'", helptext)
 381         return helptext

 406     def compile_status(self):
 407         """ Render the status text """
 408         logger.trace("Compiling Status text")
 409         status = "\n=== STATUS\n"
 410         navigation = self.interface.state["navigation"]
 411         frame_scale = int(self.interface.get_frame_scaling() * 100)
 412         status += "  File: {}\n".format(self.interface.get_frame_name())
 413         status += "  Frame: {} / {}\n".format(
 414             navigation["frame_idx"] + 1, navigation["max_frame"] + 1)
 415         status += "  Frame Size: {}%\n".format(frame_scale)
 416         status += "  Skip Mode: {}\n".format(self.interface.get_skip_mode())
 417         status += "  View Mode: {}\n".format(self.interface.get_edit_mode())
 418         if self.interface.get_selected_face_id() is not None:
 419             status += "  Selected Face Index: {}\n".format(self.interface.get_selected_face_id())
 420         if self.interface.state["edit"]["updated"]:
 421             status += "  Warning: There are unsaved changes\n"

 426     @staticmethod
 427     def text_to_image(image, display_text):
 428         """ Write out and format help text to image """
 429         # pylint: disable=no-member
 430         logger.trace("Converting help text to image")
 431         pos_y = 0
 432         for line in display_text.split("\n"):
 433             if line.startswith("==="):
 434                 pos_y += 10
 435                 line = line.replace("=", "").strip()
 436             line = line.replace("- '", "[ ").replace("':", " ]")
 437             cv2.putText(image, line, (20, pos_y),
 438                         cv2.FONT_HERSHEY_SIMPLEX, 0.43, (255, 255, 255), 1)
 439             pos_y += 20
 440         logger.trace("Converted help text to image")

 443 class Manual():
 444     """ Manually adjust or create landmarks data """
 445     def __init__(self, alignments, arguments):
 446         logger.debug("Initializing %s: (alignments: %s, arguments: %s)",
 447                      self.__class__.__name__, alignments, arguments)
 448         self.arguments = arguments
 449         self.alignments = alignments
 450         self.frames = Frames(arguments.frames_dir)
 451         self.extracted_faces = None
 452         self.interface = None
 453         self.help = None
 454         self.mouse_handler = None
 455         logger.debug("Initialized %s", self.__class__.__name__)
 456 
 457     def process(self):
 458         """ Process manual extraction """
 459         logger.info("[MANUAL PROCESSING]")  # Tidy up cli output
 460         self.extracted_faces = ExtractedFaces(self.frames, self.alignments, size=256)
 461         self.interface = Interface(self.alignments, self.frames)
 462         self.help = Help(self.interface)
 463         self.mouse_handler = MouseHandler(self.interface, self.arguments.loglevel)
 464 
 465         print(self.help.helptext)
 466         max_idx = self.frames.count - 1
 467         self.interface.state["navigation"]["max_frame"] = max_idx
 468         self.display_frames()
 469 
 470     def display_frames(self):
 471         """ Iterate through frames """
 472         # pylint: disable=no-member
 473         logger.debug("Display frames")
 474         is_windows = platform.system() == "Windows"
 475         is_conda = "conda" in sys.version.lower()
 476         logger.debug("is_windows: %s, is_conda: %s", is_windows, is_conda)
 477         cv2.namedWindow("Frame")
 478         cv2.namedWindow("Faces")
 479         cv2.setMouseCallback('Frame', self.mouse_handler.on_event)
 480 
 481         frame, faces = self.get_frame()
 482         press = self.get_keys()
 483         self.interface.set_redraw(True)
 484 
 485         while True:
 486             if self.interface.redraw():
 487                 self.help.render()
 488                 cv2.imshow("Frame", frame)
 489                 cv2.imshow("Faces", faces)
 490                 self.interface.set_redraw(False)
 491             key = cv2.waitKey(1000)
 492 
 493             if self.window_closed(is_windows, is_conda, key):
 494                 queue_manager.terminate_queues()
 495                 break
 496 
 497             if key and key != -1:
 498                 logger.trace("Keypress received: '%s'", key)
 499             if key in press.keys():
 500                 action = press[key]["action"]
 501                 logger.debug("Keypress action: key: ('%s', action: '%s')", key, action)
 502                 if action == "quit":
 503                     break
 504 
 505                 if press[key].get("key_type") == range:
 506                     args = press[key]["args"] + [chr(key)]
 507                 else:
 508                     args = press[key]["args"]
 509                 action(*args)
 510 
 511             if not self.interface.redraw():
 512                 continue
 513 
 514             logger.trace("Redraw requested")
 515             frame, faces = self.get_frame()
 516 
 517         cv2.destroyAllWindows()
 518 
 519     def window_closed(self, is_windows, is_conda, key):
 520         """ Check whether the window has been closed
 521 
 522         MS Windows doesn't appear to read the window state property
 523         properly, so we check for a negative key press.
 524 
 525         Conda (tested on Windows) doesn't appear to read the window
 526         state property or negative key press properly, so we arbitrarily
 527         use another property """
 528         # pylint: disable=no-member
 529         logger.trace("Commencing closed window check")
 530         closed = False
 531         prop_autosize = cv2.getWindowProperty('Frame', cv2.WND_PROP_AUTOSIZE)
 532         prop_visible = cv2.getWindowProperty('Frame', cv2.WND_PROP_VISIBLE)
 533         if self.arguments.disable_monitor:
 534             closed = False
 535         elif is_conda and prop_autosize < 1:
 536             closed = True
 537         elif is_windows and not is_conda and key == -1:
 538             closed = True
 539         elif not is_windows and not is_conda and prop_visible < 1:
 540             closed = True
 541         logger.trace("Completed closed window check. Closed is %s", closed)
 542         if closed:
 543             logger.debug("Window closed detected")
 544         return closed

 546     def get_keys(self):
 547         """ Convert keys dict into something useful
 548             for OpenCV """
 549         keys = dict()
 550         for key, val in self.interface.controls.items():
 551             if val.get("key_type", str) == range:
 552                 for range_key in key:
 553                     keys[ord(str(range_key))] = val
 554             elif val.get("key_type", str) == ord:
 555                 keys[key] = val
 556             else:
 557                 keys[ord(key)] = val
 558 
 559         return keys

 563         image = self.frame_selector()

 566         alignments = self.alignments.get_faces_in_frame(frame_name)
 567         faces_updated = self.interface.state["edit"]["update_faces"]
 568         logger.debug("Faces Updated: %s", faces_updated)
 569         self.extracted_faces.get_faces(frame_name)
 570         roi = [face.original_roi for face in self.extracted_faces.faces]
 571 
 572         if faces_updated:
 573             self.interface.state["edit"]["update_faces"] = False
 574 
 575         frame = FrameDisplay(image, alignments, roi, self.interface).image
 576         faces = self.set_faces(frame_name).image
 577         return frame, faces
 578 
 579     def frame_selector(self):
 580         """ Return frame at given index """
 581         navigation = self.interface.state["navigation"]
 582         frame_list = self.frames.file_list_sorted
 583         frame = frame_list[navigation["frame_idx"]]["frame_fullname"]
 584         skip_mode = self.interface.get_skip_mode().lower()
 585         logger.debug("navigation: %s, frame: '%s', skip_mode: '%s'", navigation, frame, skip_mode)
 586 
 587         while True:
 588             if navigation["last_request"] == 0:
 589                 break
 590             if navigation["frame_idx"] in (0, navigation["max_frame"]):
 591                 break
 592             if skip_mode == "standard":
 593                 break
 594             if skip_mode == "no faces" and not self.alignments.frame_has_faces(frame):
 595                 break
 596             if skip_mode == "multi-faces" and self.alignments.frame_has_multiple_faces(frame):
 597                 break
 598             if skip_mode == "has faces" and self.alignments.frame_has_faces(frame):
 599                 break
 600             self.interface.iterate_frame("navigation", navigation["last_request"])
 601             frame = frame_list[navigation["frame_idx"]]["frame_fullname"]
 602 
 603         image = self.frames.load_image(frame)
 604         navigation["last_request"] = 0
 605         navigation["frame_name"] = frame
 606         return image

 608     def set_faces(self, frame):
 609         """ Pass the current frame faces to faces window """
 610         faces = self.extracted_faces.get_faces_in_frame(frame)
 611         landmarks = [{"landmarks_xy": face.aligned_landmarks}
 612                      for face in self.extracted_faces.faces]
 613         return FacesDisplay(faces, landmarks, self.extracted_faces.size, self.interface)

 616 class FrameDisplay():
 617     """" Window that holds the frame """
 618     def __init__(self, image, alignments, roi, interface):
 619         logger.trace("Initializing %s: (alignments: %s, roi: %s, interface: %s)",
 620                      self.__class__.__name__, alignments, roi, interface)
 621         self.image = image
 622         self.roi = roi
 623         self.alignments = alignments
 624         self.interface = interface
 625         self.annotate_frame()
 626         logger.trace("Initialized %s", self.__class__.__name__)

 628     def annotate_frame(self):
 629         """ Annotate the frame """
 630         state = self.interface.state
 631         logger.trace("State: %s", state)
 632         annotate = Annotate(self.image, self.alignments, self.roi)
 633         if not state["image"]["display"]:
 634             annotate.draw_black_image()
 635 
 636         for item in ("bounding_box", "extract_box", "landmarks", "landmarks_mesh"):
 637             color = self.interface.get_color(item)
 638             size = self.interface.get_size(item)
 639             state[item]["display"] = color != 7
 640             if not state[item]["display"]:
 641                 continue
 642             logger.trace("Annotating: '%s'", item)
 643             annotation = getattr(annotate, "draw_{}".format(item))
 644             annotation(color, size)
 645 
 646         selected_face = self.interface.get_selected_face_id()
 647         if (selected_face is not None and
 648                 int(selected_face) < len(self.alignments)):
 649             annotate.draw_grey_out_faces(selected_face)
 650 
 651         self.image = self.resize_frame(annotate.image)
 652 
 653     def resize_frame(self, image):
 654         """ Set the displayed frame size and add state border"""
 655         # pylint: disable=no-member
 656         logger.trace("Resizing frame")
 657         height, width = image.shape[:2]
 658         color = self.interface.get_state_color()
 659         cv2.rectangle(image, (0, 0), (width - 1, height - 1), color, 1)
 660         scaling = self.interface.get_frame_scaling()
 661         image = cv2.resize(image, (0, 0), fx=scaling, fy=scaling)
 662         logger.trace("Resized frame")
 663         return image

 666 class FacesDisplay():
 667     """ Window that holds faces thumbnail """
 668     def __init__(self, extracted_faces, landmarks, size, interface):
 669         logger.trace("Initializing %s: (extracted_faces: %s, landmarks: %s, size: %s, "
 670                      "interface: %s)", self.__class__.__name__, extracted_faces,
 671                      landmarks, size, interface)
 672         self.row_length = 4
 673         self.faces = self.copy_faces(extracted_faces)
 674         self.roi = self.set_full_roi(size)
 675         self.landmarks = landmarks
 676         self.interface = interface
 677 
 678         self.annotate_faces()
 679 
 680         self.image = self.build_faces_image(size)
 681         logger.trace("Initialized %s", self.__class__.__name__)
 682 
 683     @staticmethod
 684     def copy_faces(faces):
 685         """ Copy the extracted faces so as not to save the annotations back """
 686         return [face.aligned_face.copy() for face in faces]
 687 
 688     @staticmethod
 689     def set_full_roi(size):
 690         """ ROI is the full frame for faces, so set based on size """
 691         return [np.array([[(0, 0), (0, size - 1), (size - 1, size - 1), (size - 1, 0)]], np.int32)]
 692 
 693     def annotate_faces(self):
 694         """ Annotate each of the faces """
 695         state = self.interface.state
 696         selected_face = self.interface.get_selected_face_id()
 697         logger.trace("State: %s, Selected Face ID: %s", state, selected_face)
 698         for idx, face in enumerate(self.faces):
 699             annotate = Annotate(face, [self.landmarks[idx]], self.roi)
 700             if not state["image"]["display"]:
 701                 annotate.draw_black_image()
 702 
 703             for item in ("landmarks", "landmarks_mesh"):
 704                 if not state[item]["display"]:
 705                     continue
 706                 logger.trace("Annotating: '%s'", item)
 707                 color = self.interface.get_color(item)
 708                 size = self.interface.get_size(item)
 709                 annotation = getattr(annotate, "draw_{}".format(item))
 710                 annotation(color, size)
 711 
 712             if (selected_face is not None
 713                     and int(selected_face) < len(self.faces)
 714                     and int(selected_face) != idx):
 715                 annotate.draw_grey_out_faces(1)
 716 
 717             self.faces[idx] = annotate.image

 719     def build_faces_image(self, size):
 720         """ Display associated faces """
 721         total_faces = len(self.faces)
 722         logger.trace("Building faces panel. (total_faces: %s)", total_faces)
 723         if not total_faces:
 724             logger.trace("Returning empty row")
 725             image = self.build_faces_row(list(), size)
 726             return image
 727         total_rows = int(total_faces / self.row_length) + 1
 728         for idx in range(total_rows):
 729             logger.trace("Building row %s", idx)
 730             face_idx = idx * self.row_length
 731             row_faces = self.faces[face_idx:face_idx + self.row_length]
 732             if not row_faces:
 733                 break
 734             row = self.build_faces_row(row_faces, size)
 735             image = row if idx == 0 else np.concatenate((image, row), axis=0)
 736         return image
 737 
 738     def build_faces_row(self, faces, size):
 739         """ Build a row of 4 faces """
 740         # pylint: disable=no-member
 741         logger.trace("Building row for %s faces", len(faces))
 742         if len(faces) != 4:
 743             remainder = 4 - (len(faces) % self.row_length)
 744             for _ in range(remainder):
 745                 faces.append(np.zeros((size, size, 3), np.uint8))
 746         for idx, face in enumerate(faces):
 747             color = self.interface.get_state_color()
 748             cv2.rectangle(face, (0, 0), (size - 1, size - 1),
 749                           color, 1)
 750             if idx == 0:
 751                 row = face
 752             else:
 753                 row = np.concatenate((row, face), axis=1)
 754         return row

 791     def on_event(self, event, x, y, flags, param):  # pylint: disable=unused-argument,invalid-name
 792         """ Handle the mouse events """
 793         # pylint: disable=no-member
 794         if self.interface.get_edit_mode() != "Edit":
 795             return
 796         logger.trace("Mouse event: (event: %s, x: %s, y: %s, flags: %s, param: %s",
 797                      event, x, y, flags, param)
 798         if not self.mouse_state and event not in (cv2.EVENT_LBUTTONDOWN, cv2.EVENT_MBUTTONDOWN):
 799             return
 800 
 801         self.initialize()
 802 
 803         if event in (cv2.EVENT_LBUTTONUP, cv2.EVENT_MBUTTONUP):
 804             self.mouse_state = None
 805             self.last_move = None
 806         elif event == cv2.EVENT_LBUTTONDOWN:
 807             self.mouse_state = "left"
 808             self.set_bounding_box(x, y)
 809         elif event == cv2.EVENT_MBUTTONDOWN:
 810             self.mouse_state = "middle"
 811             self.set_bounding_box(x, y)
 812         elif event == cv2.EVENT_MOUSEMOVE:
 813             if self.mouse_state == "left":
 814                 self.move_bounding_box(x, y)
 815             elif self.mouse_state == "middle":
 816                 self.resize_bounding_box(x, y)

 818     def initialize(self):

 824         self.media["frame_id"] = frame
 825         self.media["image"] = self.frames.load_image(frame)
 826         self.dims = None
 827         self.center = None
 828         self.last_move = None
 829         self.mouse_state = None
 830         self.media["bounding_box"] = DetectedFace()
 831         self.media["bounding_box_orig"] = None

 833     def set_bounding_box(self, pt_x, pt_y):
 834         """ Select or create bounding box """
 835         if self.interface.get_selected_face_id() is None:
 836             self.check_click_location(pt_x, pt_y)
 837 
 838         if self.interface.get_selected_face_id() is not None:
 839             self.dims_from_alignment()
 840         else:
 841             self.dims_from_image()
 842 
 843         self.move_bounding_box(pt_x, pt_y)

 845     def check_click_location(self, pt_x, pt_y):
 846         """ Check whether the point clicked is within an existing
 847             bounding box and set face_id """
 848         frame = self.media["frame_id"]
 849         alignments = self.alignments.get_faces_in_frame(frame)
 850         scale = self.interface.get_frame_scaling()
 851         pt_x = int(pt_x / scale)
 852         pt_y = int(pt_y / scale)
 853 
 854         for idx, alignment in enumerate(alignments):
 855             left = alignment["x"]
 856             right = alignment["x"] + alignment["w"]
 857             top = alignment["y"]
 858             bottom = alignment["y"] + alignment["h"]
 859 
 860             if left <= pt_x <= right and top <= pt_y <= bottom:
 861                 self.interface.set_state_value("edit", "selected", idx)
 862                 break

 864     def dims_from_alignment(self):
 865         """ Set the height and width of bounding box from alignment """
 866         frame = self.media["frame_id"]
 867         face_id = self.interface.get_selected_face_id()
 868         alignment = self.alignments.get_faces_in_frame(frame)[face_id]
 869         self.dims = (alignment["w"], alignment["h"])

 871     def dims_from_image(self):
 872         """ Set the height and width of bounding
 873             box at 10% of longest axis """
 874         size = max(self.media["image"].shape[:2])
 875         dim = int(size / 10.00)
 876         self.dims = (dim, dim)
 877 
 878     def bounding_from_center(self):
 879         """ Get bounding X Y from center """
 880         pt_x, pt_y = self.center
 881         width, height = self.dims
 882         scale = self.interface.get_frame_scaling()
 883         self.media["bounding_box"].x = int((pt_x / scale) - width / 2)
 884         self.media["bounding_box"].y = int((pt_y / scale) - height / 2)
 885         self.media["bounding_box"].w = width
 886         self.media["bounding_box"].h = height
 887 
 888     def move_bounding_box(self, pt_x, pt_y):
 889         """ Move the bounding box """
 890         self.center = (pt_x, pt_y)
 891         self.bounding_from_center()
 892         self.update_landmarks()
 893 
 894     def resize_bounding_box(self, pt_x, pt_y):
 895         """ Resize the bounding box """
 896         scale = self.interface.get_frame_scaling()
 897         if not self.last_move:
 898             self.last_move = (pt_x, pt_y)
 899             self.media["bounding_box_orig"] = self.media["bounding_box"]
 900 
 901         move_x = int(pt_x - self.last_move[0])
 902         move_y = int(self.last_move[1] - pt_y)
 903 
 904         original = self.media["bounding_box_orig"]
 905         updated = self.media["bounding_box"]
 906 
 907         minsize = int(20 / scale)
 908         center = (int(self.center[0] / scale), int(self.center[1] / scale))
 909         updated.x = min(center[0] - (minsize // 2), original.x - move_x)
 910         updated.y = min(center[1] - (minsize // 2), original.y - move_y)
 911         updated.w = max(minsize, original.w + move_x)
 912         updated.h = max(minsize, original.h + move_y)
 913         self.update_landmarks()
 914         self.last_move = (pt_x, pt_y)
 915 
 916     def update_landmarks(self):
 917         """ Update the landmarks """
 918         feed = ExtractMedia(self.media["frame_id"],
 919                             self.media["image"],
 920                             detected_faces=[self.media["bounding_box"]])
 921         self.queues["in"].put(feed)
 922         detected_face = next(self.extractor.detected_faces()).detected_faces[0]
 923         alignment = detected_face.to_alignment()
 924         # Mask will now be incorrect for updated landmarks so delete
 925         alignment["mask"] = dict()
 926 
 927         frame = self.media["frame_id"]
 928 
 929         if self.interface.get_selected_face_id() is None:
 930             idx = self.alignments.add_face(frame, alignment)
 931             self.interface.set_state_value("edit", "selected", idx)
 932         else:
 933             self.alignments.update_face(frame,
 934                                         self.interface.get_selected_face_id(),
 935                                         alignment)
 936             self.interface.set_redraw(True)
 937 
 938         self.interface.state["edit"]["updated"] = True
 939         self.interface.state["edit"]["update_faces"] = True
