  14 class Annotate():
  15     """ Annotate an input image """

  17     def __init__(self, image, alignments, original_roi=None):
  18         logger.debug("Initializing %s: (alignments: %s, original_roi: %s)",
  19                      self.__class__.__name__, alignments, original_roi)
  20         self.image = image
  21         self.alignments = alignments
  22         self.roi = original_roi
  23         self.colors = {1: (255, 0, 0),
  24                        2: (0, 255, 0),
  25                        3: (0, 0, 255),
  26                        4: (255, 255, 0),
  27                        5: (255, 0, 255),
  28                        6: (0, 255, 255)}
  29         logger.debug("Initialized %s", self.__class__.__name__)
  30 
  31     def draw_black_image(self):
  32         """ Change image to black at correct dimensions """
  33         logger.trace("Drawing black image")
  34         height, width = self.image.shape[:2]
  35         self.image = np.zeros((height, width, 3), dtype="uint8")
  36 
  37     def draw_bounding_box(self, color_id=1, thickness=1):
  38         """ Draw the bounding box around faces """
  39         color = self.colors[color_id]
  40         for alignment in self.alignments:
  41             top_left = (alignment["x"], alignment["y"])
  42             bottom_right = (alignment["x"] + alignment["w"], alignment["y"] + alignment["h"])
  43             logger.trace("Drawing bounding box: (top_left: %s, bottom_right: %s, color: %s, "
  44                          "thickness: %s)", top_left, bottom_right, color, thickness)
  45             cv2.rectangle(self.image, top_left, bottom_right, color, thickness)
  46 
  47     def draw_extract_box(self, color_id=2, thickness=1):
  48         """ Draw the extracted face box """
  49         if not self.roi:
  50             return
  51         color = self.colors[color_id]
  52         for idx, roi in enumerate(self.roi):
  53             logger.trace("Drawing Extract Box: (idx: %s, roi: %s)", idx, roi)
  54             top_left = [point for point in roi.squeeze()[0]]
  55             top_left = (top_left[0], top_left[1] - 10)
  56             cv2.putText(self.image,
  57                         str(idx),
  58                         top_left,
  59                         cv2.FONT_HERSHEY_DUPLEX,
  60                         1.0,
  61                         color,
  62                         thickness)
  63             cv2.polylines(self.image, [roi], True, color, thickness)
  64 
  65     def draw_landmarks(self, color_id=3, radius=1):
  66         """ Draw the facial landmarks """
  67         color = self.colors[color_id]
  68         for alignment in self.alignments:
  69             landmarks = alignment["landmarks_xy"].astype("int32")
  70             logger.trace("Drawing Landmarks: (landmarks: %s, color: %s, radius: %s)",
  71                          landmarks, color, radius)
  72             for (pos_x, pos_y) in landmarks:
  73                 cv2.circle(self.image, (pos_x, pos_y), radius, color, -1)
  74 
  75     def draw_landmarks_mesh(self, color_id=4, thickness=1):
  76         """ Draw the facial landmarks """
  77         color = self.colors[color_id]
  78         facial_landmarks_idxs = OrderedDict([("mouth", (48, 68)),
  79                                              ("right_eyebrow", (17, 22)),
  80                                              ("left_eyebrow", (22, 27)),
  81                                              ("right_eye", (36, 42)),
  82                                              ("left_eye", (42, 48)),
  83                                              ("nose", (27, 36)),
  84                                              ("jaw", (0, 17)),
  85                                              ("chin", (8, 11))])
  86         for alignment in self.alignments:
  87             landmarks = alignment["landmarks_xy"]
  88             logger.trace("Drawing Landmarks Mesh: (landmarks: %s, color: %s, thickness: %s)",
  89                          landmarks, color, thickness)
  90             for key, val in facial_landmarks_idxs.items():
  91                 points = np.array([landmarks[val[0]:val[1]]], np.int32)
  92                 fill_poly = bool(key in ("right_eye", "left_eye", "mouth"))
  93                 cv2.polylines(self.image, points, fill_poly, color, thickness)
  94 
  95     def draw_grey_out_faces(self, live_face):
  96         """ Grey out all faces except target """
  97         if not self.roi:
  98             return
  99         alpha = 0.6
 100         overlay = self.image.copy()
 101         for idx, roi in enumerate(self.roi):
 102             if idx != int(live_face):
 103                 logger.trace("Greying out face: (idx: %s, roi: %s)", idx, roi)
 104                 cv2.fillPoly(overlay, roi, (0, 0, 0))
 105 
 106         cv2.addWeighted(overlay, alpha, self.image, 1. - alpha, 0., self.image)
