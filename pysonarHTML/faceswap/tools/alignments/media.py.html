  44         """ Check the alignments file exists"""
  45         folder, filename = os.path.split(alignments_file)
  46         if filename.lower() == "dfl":
  47             folder = None
  48             filename = "dfl"

  50         elif not os.path.isfile(alignments_file):

  52             sys.exit(0)
  53         if folder:
  54             logger.verbose("Alignments file exists at '%s'", alignments_file)

  68     def add_face_hashes(self, frame_name, hashes):
  69         """ Recalculate face hashes """
  70         logger.trace("Adding face hash: (frame: '%s', hashes: %s)", frame_name, hashes)
  71         faces = self.get_faces_in_frame(frame_name)
  72         count_match = len(faces) - len(hashes)
  73         if count_match != 0:
  74             msg = "more" if count_match > 0 else "fewer"
  75             logger.warning("There are %s %s face(s) in the alignments file than exist in the "
  76                            "faces folder. Check your sources for frame '%s'.",
  77                            abs(count_match), msg, frame_name)
  78         for idx, i_hash in hashes.items():
  79             faces[idx]["hash"] = i_hash

 106 class MediaLoader():
 107     """ Class to load filenames from folder """
 108     def __init__(self, folder):
 109         logger.debug("Initializing %s: (folder: '%s')", self.__class__.__name__, folder)
 110         logger.info("[%s DATA]", self.__class__.__name__.upper())
 111         self._count = None
 112         self.folder = folder
 113         self.vid_reader = self.check_input_folder()
 114         self.file_list_sorted = self.sorted_items()
 115         self.items = self.load_items()
 116         logger.verbose("%s items loaded", self.count)
 117         logger.debug("Initialized %s", self.__class__.__name__)
 118 
 119     @property
 120     def is_video(self):
 121         """ Return whether source is a video or not """
 122         return self.vid_reader is not None
 123 
 124     @property
 125     def count(self):
 126         """ Number of faces or frames """
 127         if self._count is not None:
 128             return self._count
 129         if self.is_video:
 130             self._count = int(count_frames(self.folder))
 131         else:
 132             self._count = len(self.file_list_sorted)
 133         return self._count
 134 
 135     def check_input_folder(self):
 136         """ makes sure that the frames or faces folder exists
 137             If frames folder contains a video file return imageio reader object """
 138         err = None
 139         loadtype = self.__class__.__name__
 140         if not self.folder:
 141             err = "ERROR: A {} folder must be specified".format(loadtype)
 142         elif not os.path.exists(self.folder):
 143             err = ("ERROR: The {} location {} could not be "
 144                    "found".format(loadtype, self.folder))
 145         if err:
 146             logger.error(err)
 147             sys.exit(0)
 148 
 149         if (loadtype == "Frames" and
 150                 os.path.isfile(self.folder) and
 151                 os.path.splitext(self.folder)[1].lower() in _video_extensions):
 152             logger.verbose("Video exists at: '%s'", self.folder)
 153             retval = cv2.VideoCapture(self.folder)  # pylint: disable=no-member
 154             # TODO ImageIO single frame seek seems slow. Look into this
 155             # retval = imageio.get_reader(self.folder, "ffmpeg")
 156         else:
 157             logger.verbose("Folder exists at '%s'", self.folder)
 158             retval = None
 159         return retval
 160 
 161     @staticmethod
 162     def valid_extension(filename):
 163         """ Check whether passed in file has a valid extension """
 164         extension = os.path.splitext(filename)[1]
 165         retval = extension.lower() in _image_extensions
 166         logger.trace("Filename has valid extension: '%s': %s", filename, retval)
 167         return retval

 184     def load_image(self, filename):
 185         """ Load an image """
 186         if self.is_video:
 187             image = self.load_video_frame(filename)
 188         else:
 189             src = os.path.join(self.folder, filename)
 190             logger.trace("Loading image: '%s'", src)
 191             image = read_image(src, raise_error=True)
 192         return image

 194     def load_video_frame(self, filename):
 195         """ Load a requested frame from video """
 196         frame = os.path.splitext(filename)[0]
 197         logger.trace("Loading video frame: '%s'", frame)
 198         frame_no = int(frame[frame.rfind("_") + 1:]) - 1
 199         self.vid_reader.set(cv2.CAP_PROP_POS_FRAMES, frame_no)  # pylint: disable=no-member
 200         _, image = self.vid_reader.read()
 201         # TODO imageio single frame seek seems slow. Look into this
 202         # self.vid_reader.set_image_index(frame_no)
 203         # image = self.vid_reader.get_next_data()[:, :, ::-1]
 204         return image

 206     def stream(self, skip_list=None):
 207         """ Load the images in :attr:`folder` in the order they are received from
 208         :class:`lib.image.ImagesLoader` in a background thread.
 209 
 210         Parameters
 211         ----------
 212         skip_list: list, optional
 213             A list of frame indices that should not be loaded. Pass ``None`` if all images should
 214             be loaded. Default: ``None``
 215 
 216         Yields
 217         ------
 218         str
 219             The filename of the image that is being returned
 220         numpy.ndarray
 221             The image that has been loaded from disk
 222         """
 223         loader = ImagesLoader(self.folder, queue_size=32)
 224         if skip_list is not None:
 225             loader.add_skip_list(skip_list)
 226         for filename, image in loader.load():
 227             yield filename, image
 228 
 229     @staticmethod
 230     def save_image(output_folder, filename, image):
 231         """ Save an image """
 232         output_file = os.path.join(output_folder, filename)
 233         output_file = os.path.splitext(output_file)[0]+'.png'
 234         logger.trace("Saving image: '%s'", output_file)
 235         cv2.imwrite(output_file, image)  # pylint: disable=no-member

 238 class Faces(MediaLoader):
 239     """ Object to hold the faces that are to be swapped out """
 240 
 241     def process_folder(self):
 242         """ Iterate through the faces folder pulling out various information """
 243         logger.info("Loading file list from %s", self.folder)
 244 
 245         filelist = [os.path.join(self.folder, face)
 246                     for face in os.listdir(self.folder)
 247                     if self.valid_extension(face)]
 248         for fullpath, face_hash in tqdm(read_image_hash_batch(filelist),
 249                                         total=len(filelist),
 250                                         desc="Reading Face Hashes"):
 251             filename = os.path.basename(fullpath)
 252             face_name, extension = os.path.splitext(filename)
 253             retval = {"face_fullname": filename,
 254                       "face_name": face_name,
 255                       "face_extension": extension,
 256                       "face_hash": face_hash}
 257             logger.trace(retval)
 258             yield retval
 259 
 260     def load_items(self):
 261         """ Load the face names into dictionary """
 262         faces = dict()
 263         for face in self.file_list_sorted:
 264             faces.setdefault(face["face_hash"], list()).append((face["face_name"],
 265                                                                 face["face_extension"]))
 266         logger.trace(faces)
 267         return faces
 268 
 269     def sorted_items(self):
 270         """ Return the items sorted by face name """
 271         items = sorted([item for item in self.process_folder()],
 272                        key=lambda x: (x["face_name"]))
 273         logger.trace(items)
 274         return items

 277 class Frames(MediaLoader):
 278     """ Object to hold the frames that are to be checked against """
 279 
 280     def process_folder(self):
 281         """ Iterate through the frames folder pulling the base filename """
 282         iterator = self.process_video if self.is_video else self.process_frames
 283         for item in iterator():
 284             yield item
 285 
 286     def process_frames(self):
 287         """ Process exported Frames """
 288         logger.info("Loading file list from %s", self.folder)
 289         for frame in os.listdir(self.folder):
 290             if not self.valid_extension(frame):
 291                 continue
 292             filename = os.path.splitext(frame)[0]
 293             file_extension = os.path.splitext(frame)[1]
 294 
 295             retval = {"frame_fullname": frame,
 296                       "frame_name": filename,
 297                       "frame_extension": file_extension}
 298             logger.trace(retval)
 299             yield retval
 300 
 301     def process_video(self):
 302         """Dummy in frames for video """
 303         logger.info("Loading video frames from %s", self.folder)
 304         vidname = os.path.splitext(os.path.basename(self.folder))[0]
 305         for i in range(self.count):
 306             idx = i + 1
 307             # Keep filename format for outputted face
 308             filename = "{}_{:06d}".format(vidname, idx)
 309             retval = {"frame_fullname": "{}.png".format(filename),
 310                       "frame_name": filename,
 311                       "frame_extension": ".png"}
 312             logger.trace(retval)
 313             yield retval
 314 
 315     def load_items(self):
 316         """ Load the frame info into dictionary """
 317         frames = dict()
 318         for frame in self.file_list_sorted:
 319             frames[frame["frame_fullname"]] = (frame["frame_name"],
 320                                                frame["frame_extension"])
 321         logger.trace(frames)
 322         return frames
 323 
 324     def sorted_items(self):
 325         """ Return the items sorted by filename """
 326         items = sorted([item for item in self.process_folder()],
 327                        key=lambda x: (x["frame_name"]))
 328         logger.trace(items)
 329         return items

 332 class ExtractedFaces():
 333     """ Holds the extracted faces and matrix for
 334         alignments """
 335     def __init__(self, frames, alignments, size=256, align_eyes=False):
 336         logger.trace("Initializing %s: size: %s", self.__class__.__name__, size)
 337         self.size = size
 338         self.padding = int(size * 0.1875)
 339         self.align_eyes_bool = align_eyes
 340         self.alignments = alignments
 341         self.frames = frames
 342         self.current_frame = None
 343         self.faces = list()
 344         logger.trace("Initialized %s", self.__class__.__name__)
 345 
 346     def get_faces(self, frame, image=None):
 347         """ Return faces and transformed landmarks
 348             for each face in a given frame with it's alignments"""
 349         logger.trace("Getting faces for frame: '%s'", frame)
 350         self.current_frame = None
 351         alignments = self.alignments.get_faces_in_frame(frame)
 352         logger.trace("Alignments for frame: (frame: '%s', alignments: %s)", frame, alignments)
 353         if not alignments:
 354             self.faces = list()
 355             return
 356         image = self.frames.load_image(frame) if image is None else image
 357         self.faces = [self.extract_one_face(alignment, image) for alignment in alignments]
 358         self.current_frame = frame
 359 
 360     def extract_one_face(self, alignment, image):
 361         """ Extract one face from image """
 362         logger.trace("Extracting one face: (frame: '%s', alignment: %s)",
 363                      self.current_frame, alignment)
 364         face = DetectedFace()
 365         face.from_alignment(alignment, image=image)
 366         face.load_aligned(image, size=self.size)
 367         face = self.align_eyes(face, image) if self.align_eyes_bool else face
 368         return face
 369 
 370     def get_faces_in_frame(self, frame, update=False, image=None):
 371         """ Return the faces for the selected frame """
 372         logger.trace("frame: '%s', update: %s", frame, update)
 373         if self.current_frame != frame or update:
 374             self.get_faces(frame, image=image)
 375         return self.faces
 376 
 377     def get_roi_size_for_frame(self, frame):
 378         """ Return the size of the original extract box for
 379             the selected frame """
 380         logger.trace("frame: '%s'", frame)
 381         if self.current_frame != frame:
 382             self.get_faces(frame)
 383         sizes = list()
 384         for face in self.faces:
 385             roi = face.original_roi.squeeze()
 386             top_left, top_right = roi[0], roi[3]
 387             len_x = top_right[0] - top_left[0]
 388             len_y = top_right[1] - top_left[1]
 389             if top_left[1] == top_right[1]:
 390                 length = len_y
 391             else:
 392                 length = int(((len_x ** 2) + (len_y ** 2)) ** 0.5)
 393             sizes.append(length)
 394         logger.trace("sizes: '%s'", sizes)
 395         return sizes

 406     @staticmethod
 407     def align_eyes(face, image):
 408         """ Re-extract a face with the pupils forced to be absolutely horizontally aligned """
 409         umeyama_landmarks = face.aligned_landmarks
 410         left_eye_center = umeyama_landmarks[42:48].mean(axis=0)
 411         right_eye_center = umeyama_landmarks[36:42].mean(axis=0)
 412         d_y = right_eye_center[1] - left_eye_center[1]
 413         d_x = right_eye_center[0] - left_eye_center[0]
 414         theta = np.pi - np.arctan2(d_y, d_x)
 415         rot_cos = np.cos(theta)
 416         rot_sin = np.sin(theta)
 417         rotation_matrix = np.array([[rot_cos, -rot_sin, 0.],
 418                                     [rot_sin, rot_cos, 0.],
 419                                     [0., 0., 1.]])
 420 
 421         mat_umeyama = np.concatenate((face.aligned["matrix"], np.array([[0., 0., 1.]])), axis=0)
 422         corrected_mat = np.dot(rotation_matrix, mat_umeyama)
 423         face.aligned["matrix"] = corrected_mat[:2]
 424         face.aligned["face"] = AlignerExtract().transform(image,
 425                                                           face.aligned["matrix"],
 426                                                           face.aligned["size"],
 427                                                           int(face.aligned["size"] * 0.375) // 2)
 428         logger.trace("Adjusted matrix: %s", face.aligned["matrix"])
 429         return face
