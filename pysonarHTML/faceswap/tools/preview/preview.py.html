  37 class Preview(tk.Tk):  # pylint:disable=too-few-public-methods
  38     """ This tool is part of the Faceswap Tools suite and should be called from
  39     ``python tools.py preview`` command.
  40 
  41     Loads up 5 semi-random face swaps and displays them, cropped, in place in the final frame.
  42     Allows user to live tweak settings, before saving the final config to
  43     :file:`./config/convert.ini`
  44 
  45     Parameters
  46     ----------
  47     arguments: :class:`argparse.Namespace`
  48         The :mod:`argparse` arguments as passed in from :mod:`tools.py`
  49     """

  52         logger.debug("Initializing %s: (arguments: '%s'", self.__class__.__name__, arguments)

  55         self._lock = Lock()

 155 class Samples():
 156     """ The display samples.
 157 
 158     Obtains and holds :attr:`sample_size` semi random test faces for displaying in the
 159     preview GUI.
 160 
 161     The file list is split into evenly sized groups of :attr:`sample_size`. When a display set is
 162     generated, a random image from each of the groups is selected to provide an array of images
 163     across the length of the video.
 164 
 165     Parameters
 166     ----------
 167     arguments: :class:`argparse.Namespace`
 168         The :mod:`argparse` arguments as passed in from :mod:`tools.py`
 169     sample_size: int
 170         The number of samples to take from the input video/images
 171     display: :class:`FacesDisplay`
 172         The display section of the Preview GUI.
 173     lock: :class:`threading.Lock`
 174         A threading lock to prevent multiple GUI updates at the same time.
 175     trigger_patch:  :class:`threading.Event`
 176         An event to indicate that a converter patch should be run
 177     """
 178 
 179     def __init__(self, arguments, sample_size, display, lock, trigger_patch):
 180         logger.debug("Initializing %s: (arguments: '%s', sample_size: %s, display: %s, lock: %s, "
 181                      "trigger_patch: %s)", self.__class__.__name__, arguments, sample_size,
 182                      display, lock, trigger_patch)
 183         self._sample_size = sample_size
 184         self._display = display
 185         self._lock = lock
 186         self._trigger_patch = trigger_patch
 187         self._input_images = list()
 188         self._predicted_images = list()
 189 
 190         self._images = Images(arguments)
 191         self._alignments = Alignments(arguments,
 192                                       is_extract=False,
 193                                       input_is_video=self._images.is_video)
 194         if not self._alignments.have_alignments_file:
 195             logger.error("Alignments file not found at: '%s'", self._alignments.file)
 196             sys.exit(1)
 197         self._filelist = self._get_filelist()
 198         self._indices = self._get_indices()
 199 
 200         self._predictor = Predict(queue_manager.get_queue("preview_predict_in"),
 201                                   sample_size,
 202                                   arguments)
 203         self.generate()
 204 
 205         logger.debug("Initialized %s", self.__class__.__name__)
 206 
 207     @property
 208     def sample_size(self):
 209         """ int: The number of samples to take from the input video/images """
 210         return self._sample_size

 227     @property
 228     def _random_choice(self):
 229         """ list: Random indices from the :attr:`_indices` group """
 230         retval = [random.choice(indices) for indices in self._indices]
 231         logger.debug(retval)
 232         return retval

 234     def _get_filelist(self):
 235         """ Get a list of files for the input, filtering out those frames which do
 236         not contain faces.
 237 
 238         Returns
 239         -------
 240         list
 241             A list of filenames of frames that contain faces.
 242         """
 243         logger.debug("Filtering file list to frames with faces")
 244         if self._images.is_video:
 245             filelist = ["{}_{:06d}.png".format(os.path.splitext(self._images.input_images)[0],
 246                                                frame_no)
 247                         for frame_no in range(1, self._images.images_found + 1)]
 248         else:
 249             filelist = self._images.input_images
 250 
 251         retval = [filename for filename in filelist
 252                   if self._alignments.frame_has_faces(os.path.basename(filename))]
 253         logger.debug("Filtered out frames: %s", self._images.images_found - len(retval))
 254         try:
 255             assert retval
 256         except AssertionError as err:
 257             msg = ("No faces were found in any of the frames passed in. Make sure you are passing "
 258                    "in a frames source rather than extracted faces, and that you have provided "
 259                    "the correct alignments file.")
 260             raise FaceswapError(msg) from err
 261         return retval
 262 
 263     def _get_indices(self):
 264         """ Get indices for each sample group.
 265 
 266         Obtain :attr:`self.sample_size` evenly sized groups of indices
 267         pertaining to the filtered :attr:`self._file_list`
 268 
 269         Returns
 270         -------
 271         list
 272             list of indices relating to the filtered file list, split into groups
 273         """
 274         # Remove start and end values to get a list divisible by self.sample_size
 275         no_files = len(self._filelist)
 276         crop = no_files % self._sample_size
 277         top_tail = list(range(no_files))[
 278             crop // 2:no_files - (crop - (crop // 2))]
 279         # Partition the indices
 280         size = len(top_tail)
 281         retval = [top_tail[start:start + size // self._sample_size]
 282                   for start in range(0, size, size // self._sample_size)]
 283         logger.debug("Indices pools: %s", ["{}: (start: {}, end: {}, size: {})".format(idx,
 284                                                                                        min(pool),
 285                                                                                        max(pool),
 286                                                                                        len(pool))
 287                                            for idx, pool in enumerate(retval)])
 288         return retval

 290     def generate(self):
 291         """ Generate a sample set.
 292 
 293         Selects :attr:`sample_size` random faces. Runs them through prediction to obtain the
 294         swap, then trigger the patch event to run the faces through patching.
 295         """
 296         self._load_frames()
 297         self._predict()
 298         self._trigger_patch.set()
 299 
 300     def _load_frames(self):
 301         """ Load a sample of random frames.
 302 
 303         * Picks a random face from each indices group.
 304 
 305         * Takes the first face from the image (if there) are multiple faces. Adds the images to \
 306             :attr:`self._input_images`.
 307 
 308         * Sets :attr:`_display.source` to the input images and flags that the display should \
 309             be updated
 310         """
 311         self._input_images = list()
 312         for selection in self._random_choice:
 313             filename = os.path.basename(self._filelist[selection])
 314             image = self._images.load_one_image(self._filelist[selection])
 315             # Get first face only
 316             face = self._alignments.get_faces_in_frame(filename)[0]
 317             detected_face = DetectedFace()
 318             detected_face.from_alignment(face, image=image)
 319             self._input_images.append({"filename": filename,
 320                                        "image": image,
 321                                        "detected_faces": [detected_face]})
 322         self._display.source = self._input_images
 323         self._display.update_source = True
 324         logger.debug("Selected frames: %s", [frame["filename"] for frame in self._input_images])
 325 
 326     def _predict(self):
 327         """ Predict from the loaded frames.
 328 
 329         With a threading lock (to prevent stacking), run the selected faces through the Faceswap
 330         model predict function and add the output to :attr:`predicted`
 331         """
 332         with self._lock:
 333             self._predicted_images = list()
 334             for frame in self._input_images:
 335                 self._predictor.in_queue.put(frame)
 336             idx = 0
 337             while idx < self._sample_size:
 338                 logger.debug("Predicting face %s of %s", idx + 1, self._sample_size)
 339                 items = self._predictor.out_queue.get()
 340                 if items == "EOF":
 341                     logger.debug("Received EOF")
 342                     break
 343                 for item in items:
 344                     self._predicted_images.append(item)
 345                     logger.debug("Predicted face %s of %s", idx + 1, self._sample_size)
 346                     idx += 1
 347         logger.debug("Predicted faces")

 350 class Patch():
 351     """ The Patch pipeline
 352 
 353     Runs in it's own thread. Takes the output from the Faceswap model predictor and runs the faces
 354     through the convert pipeline using the currently selected options.
 355 
 356     Parameters
 357     ----------
 358     arguments: :class:`argparse.Namespace`
 359         The :mod:`argparse` arguments as passed in from :mod:`tools.py`
 360     available_masks: list
 361         The masks that are available for convert
 362     samples: :class:`Samples`
 363         The Samples for display.
 364     display: :class:`FacesDisplay`
 365         The display section of the Preview GUI.
 366     lock: :class:`threading.Lock`
 367         A threading lock to prevent multiple GUI updates at the same time.
 368     trigger:  :class:`threading.Event`
 369         An event to indicate that a converter patch should be run
 370     config_tools: :class:`ConfigTools`
 371         Tools for loading and saving configuration files
 372     tk_vars: dict
 373         Global tkinter variables. `Refresh` and `Busy` :class:`tkinter.BooleanVar`
 374 
 375     Attributes
 376     ----------
 377     converter_arguments: dict
 378         The currently selected converter command line arguments for the patch queue
 379     current_config::class:`lib.config.FaceswapConfig`
 380         The currently set configuration for the patch queue
 381     """
 382     def __init__(self, arguments, available_masks, samples,
 383                  display, lock, trigger, config_tools, tk_vars):
 384         logger.debug("Initializing %s: (arguments: '%s', available_masks: %s, samples: %s, "
 385                      "display: %s, lock: %s, trigger: %s, config_tools: %s, tk_vars %s)",
 386                      self.__class__.__name__, arguments, available_masks, samples, display, lock,
 387                      trigger, config_tools, tk_vars)
 388         self._samples = samples
 389         self._queue_patch_in = queue_manager.get_queue("preview_patch_in")
 390         self._display = display
 391         self._lock = lock
 392         self._trigger = trigger
 393         self.current_config = config_tools.config
 394         self.converter_arguments = None  # Updated converter arguments dict
 395 
 396         configfile = arguments.configfile if hasattr(arguments, "configfile") else None
 397         self._converter = Converter(output_size=self._samples.predictor.output_size,
 398                                     coverage_ratio=self._samples.predictor.coverage_ratio,
 399                                     draw_transparent=False,
 400                                     pre_encode=None,
 401                                     arguments=self._generate_converter_arguments(arguments,
 402                                                                                  available_masks),
 403                                     configfile=configfile)
 404         self._shutdown = Event()
 405 
 406         self._thread = MultiThread(self._process,
 407                                    self._trigger,
 408                                    self._shutdown,
 409                                    self._queue_patch_in,
 410                                    self._samples,
 411                                    tk_vars,
 412                                    thread_count=1,
 413                                    name="patch_thread")
 414         self._thread.start()
 415 
 416     @property
 417     def trigger(self):
 418         """ :class:`threading.Event`: The trigger to indicate that a patching run should
 419         commence. """
 420         return self._trigger

 427     @staticmethod
 428     def _generate_converter_arguments(arguments, available_masks):
 429         """ Add the default converter arguments to the initial arguments. Ensure the mask selection
 430         is available.
 431 
 432         Parameters
 433         ----------
 434         arguments: :class:`argparse.Namespace`
 435             The :mod:`argparse` arguments as passed in from :mod:`tools.py`
 436         available_masks: list
 437             The masks that are available for convert
 438         Returns
 439         ----------
 440         arguments: :class:`argparse.Namespace`
 441             The :mod:`argparse` arguments as passed in with converter default
 442             arguments added
 443         """
 444         valid_masks = available_masks + ["none"]
 445         converter_arguments = ConvertArgs(None, "convert").get_optional_arguments()
 446         for item in converter_arguments:
 447             value = item.get("default", None)
 448             # Skip options without a default value
 449             if value is None:
 450                 continue
 451             option = item.get("dest", item["opts"][1].replace("--", ""))
 452             if option == "mask_type" and value not in valid_masks:
 453                 logger.debug("Amending default mask from '%s' to '%s'", value, valid_masks[0])
 454                 value = valid_masks[0]
 455             # Skip options already in arguments
 456             if hasattr(arguments, option):
 457                 continue
 458             # Add option to arguments
 459             setattr(arguments, option, value)
 460         logger.debug(arguments)
 461         return arguments

 463     def _process(self, trigger_event, shutdown_event, patch_queue_in, samples, tk_vars):
 464         """ The face patching process.
 465 
 466         Runs in a thread, and waits for an event to be set. Once triggered, runs a patching
 467         cycle and sets the :class:`Display` destination images.
 468 
 469         Parameters
 470         ----------
 471         trigger_event: :class:`threading.Event`
 472             Set by parent process when a patching run should be executed
 473         shutdown_event :class:`threading.Event`
 474             Set by parent process if a shutdown has been requested
 475         patch_queue_in: :class:`queue.Queue`
 476             The input queue for the patching process
 477         samples: :class:`Samples`
 478             The Samples for display.
 479         tk_vars: dict
 480             Global tkinter variables. `Refresh` and `Busy` :class:`tkinter.BooleanVar`
 481         """
 482         patch_queue_out = queue_manager.get_queue("preview_patch_out")
 483         while True:
 484             trigger = trigger_event.wait(1)
 485             if shutdown_event.is_set():
 486                 logger.debug("Shutdown received")
 487                 break
 488             if not trigger:
 489                 continue
 490             # Clear trigger so calling process can set it during this run
 491             trigger_event.clear()
 492             tk_vars["busy"].set(True)
 493             queue_manager.flush_queue("preview_patch_in")
 494             self._feed_swapped_faces(patch_queue_in, samples)
 495             with self._lock:
 496                 self._update_converter_arguments()
 497                 self._converter.reinitialize(config=self.current_config)
 498             swapped = self._patch_faces(patch_queue_in, patch_queue_out, samples.sample_size)
 499             with self._lock:
 500                 self._display.destination = swapped
 501             tk_vars["refresh"].set(True)
 502             tk_vars["busy"].set(False)
 503 
 504     def _update_converter_arguments(self):
 505         """ Update the converter arguments to the currently selected values. """
 506         logger.debug("Updating Converter cli arguments")
 507         if self.converter_arguments is None:
 508             logger.debug("No arguments to update")
 509             return
 510         for key, val in self.converter_arguments.items():
 511             logger.debug("Updating %s to %s", key, val)
 512             setattr(self._converter.cli_arguments, key, val)
 513         logger.debug("Updated Converter cli arguments")
 514 
 515     @staticmethod
 516     def _feed_swapped_faces(patch_queue_in, samples):
 517         """ Feed swapped faces to the converter's in-queue.
 518 
 519         Parameters
 520         ----------
 521         patch_queue_in: :class:`queue.Queue`
 522             The input queue for the patching process
 523         samples: :class:`Samples`
 524             The Samples for display.
 525         """
 526         logger.trace("feeding swapped faces to converter")
 527         for item in samples.predicted_images:
 528             patch_queue_in.put(item)
 529         logger.trace("fed %s swapped faces to converter", len(samples.predicted_images))
 530         logger.trace("Putting EOF to converter")
 531         patch_queue_in.put("EOF")
 532 
 533     def _patch_faces(self, queue_in, queue_out, sample_size):
 534         """ Patch faces.
 535 
 536         Run the convert process on the swapped faces and return the patched faces.
 537 
 538         patch_queue_in: :class:`queue.Queue`
 539             The input queue for the patching process
 540         queue_out: :class:`queue.Queue`
 541             The output queue from the patching process
 542         sample_size: int
 543             The number of samples to be displayed
 544 
 545         Returns
 546         -------
 547         list
 548             The swapped faces patched with the selected convert settings
 549         """
 550         logger.trace("Patching faces")
 551         self._converter.process(queue_in, queue_out)
 552         swapped = list()
 553         idx = 0
 554         while idx < sample_size:
 555             logger.trace("Patching image %s of %s", idx + 1, sample_size)
 556             item = queue_out.get()
 557             swapped.append(item[1])
 558             logger.trace("Patched image %s of %s", idx + 1, sample_size)
 559             idx += 1
 560         logger.trace("Patched faces")
 561         return swapped

 564 class FacesDisplay():
 565     """ Compiles the 2 rows of sample faces (original and swapped) into a single image
 566 
 567     Parameters
 568     ----------
 569     size: int
 570         The size of each individual face sample in pixels
 571     padding: int
 572         The amount of extra padding to apply to the outside of the face
 573     tk_vars: dict
 574         Global tkinter variables. `Refresh` and `Busy` :class:`tkinter.BooleanVar`
 575 
 576     Attributes
 577     ----------
 578     update_source: bool
 579         Flag to indicate that the source images for the preview have been updated, so the preview
 580         should be recompiled.
 581     source: list
 582         The list of :class:`numpy.ndarray` source preview images for top row of display
 583     destination: list
 584         The list of :class:`numpy.ndarray` swapped and patched preview images for bottom row of
 585         display
 586     """
 587     def __init__(self, size, padding, tk_vars):
 588         logger.trace("Initializing %s: (size: %s, padding: %s, tk_vars: %s)",
 589                      self.__class__.__name__, size, padding, tk_vars)
 590         self._size = size
 591         self._display_dims = (1, 1)
 592         self._tk_vars = tk_vars
 593         self._padding = padding
 594 
 595         self._faces = dict()
 596         self._faces_source = None
 597         self._faces_dest = None
 598         self._tk_image = None
 599 
 600         # Set from Samples
 601         self.update_source = False
 602         self.source = list()  # Source images, filenames + detected faces

 629     def update_tk_image(self):
 630         """ Build the full preview images and compile :attr:`tk_image` for display. """
 631         logger.trace("Updating tk image")
 632         self._build_faces_image()
 633         img = np.vstack((self._faces_source, self._faces_dest))
 634         size = self._get_scale_size(img)
 635         img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
 636         img = Image.fromarray(img)
 637         img = img.resize(size, Image.ANTIALIAS)
 638         self._tk_image = ImageTk.PhotoImage(img)
 639         self._tk_vars["refresh"].set(False)

 642     def _get_scale_size(self, image):
 643         """ Get the size that the full preview image should be resized to fit in the
 644         display window.
 645 
 646         Parameters
 647         ----------
 648         image: :class:`numpy.ndarray`
 649             The full sized compiled preview image
 650 
 651         Returns
 652         -------
 653         tuple
 654             The (`width`, `height`) that the display image should be sized to fit in the display
 655             window
 656         """
 657         frameratio = float(self._display_dims[0]) / float(self._display_dims[1])
 658         imgratio = float(image.shape[1]) / float(image.shape[0])
 659 
 660         if frameratio <= imgratio:
 661             scale = self._display_dims[0] / float(image.shape[1])
 662             size = (self._display_dims[0], max(1, int(image.shape[0] * scale)))
 663         else:
 664             scale = self._display_dims[1] / float(image.shape[0])
 665             size = (max(1, int(image.shape[1] * scale)), self._display_dims[1])
 666         logger.trace("scale: %s, size: %s", scale, size)
 667         return size

 669     def _build_faces_image(self):
 670         """ Compile the source and destination rows of the preview image. """
 671         logger.trace("Building Faces Image")
 672         update_all = self.update_source
 673         self._faces_from_frames()
 674         if update_all:
 675             header = self._header_text()
 676             source = np.hstack([self._draw_rect(face) for face in self._faces["src"]])
 677             self._faces_source = np.vstack((header, source))
 678         self._faces_dest = np.hstack([self._draw_rect(face) for face in self._faces["dst"]])
 679         logger.debug("source row shape: %s, swapped row shape: %s",
 680                      self._faces_dest.shape, self._faces_source.shape)
 681 
 682     def _faces_from_frames(self):
 683         """ Extract the preview faces from the source frames and apply the requisite padding. """
 684         logger.debug("Extracting faces from frames: Number images: %s", len(self.source))
 685         if self.update_source:
 686             self._crop_source_faces()
 687         self._crop_destination_faces()
 688         logger.debug("Extracted faces from frames: %s",
 689                      {k: len(v) for k, v in self._faces.items()})
 690 
 691     def _crop_source_faces(self):
 692         """ Extract the source faces from the source frames, along with their filenames and the
 693         transformation matrix used to extract the faces. """
 694         logger.debug("Updating source faces")
 695         self._faces = dict()
 696         for image in self.source:
 697             detected_face = image["detected_faces"][0]
 698             src_img = image["image"]
 699             detected_face.load_aligned(src_img, self._size)
 700             matrix = detected_face.aligned["matrix"]
 701             self._faces.setdefault("filenames",
 702                                    list()).append(os.path.splitext(image["filename"])[0])
 703             self._faces.setdefault("matrix", list()).append(matrix)
 704             self._faces.setdefault("src", list()).append(AlignerExtract().transform(
 705                 src_img,
 706                 matrix,
 707                 self._size,
 708                 self._padding))
 709         self.update_source = False
 710         logger.debug("Updated source faces")
 711 
 712     def _crop_destination_faces(self):
 713         """ Extract the swapped faces from the swapped frames using the source face destination
 714         matrices. """
 715         logger.debug("Updating destination faces")
 716         self._faces["dst"] = list()
 717         destination = self.destination if self.destination else [np.ones_like(src["image"])
 718                                                                  for src in self.source]
 719         for idx, image in enumerate(destination):
 720             self._faces["dst"].append(AlignerExtract().transform(
 721                 image,
 722                 self._faces["matrix"][idx],
 723                 self._size,
 724                 self._padding))
 725         logger.debug("Updated destination faces")
 726 
 727     def _header_text(self):
 728         """ Create the header text displaying the frame name for each preview column.
 729 
 730         Returns
 731         -------
 732         :class:`numpy.ndarray`
 733             The header row of the preview image containing the frame names for each column
 734         """
 735         font_scale = self._size / 640
 736         height = self._size // 8
 737         font = cv2.FONT_HERSHEY_SIMPLEX
 738         # Get size of placed text for positioning
 739         text_sizes = [cv2.getTextSize(self._faces["filenames"][idx],
 740                                       font,
 741                                       font_scale,
 742                                       1)[0]
 743                       for idx in range(self._total_columns)]
 744         # Get X and Y co-ordinates for each text item
 745         text_y = int((height + text_sizes[0][1]) / 2)
 746         text_x = [int((self._size - text_sizes[idx][0]) / 2) + self._size * idx
 747                   for idx in range(self._total_columns)]
 748         logger.debug("filenames: %s, text_sizes: %s, text_x: %s, text_y: %s",
 749                      self._faces["filenames"], text_sizes, text_x, text_y)
 750         header_box = np.ones((height, self._size * self._total_columns, 3), np.uint8) * 255
 751         for idx, text in enumerate(self._faces["filenames"]):
 752             cv2.putText(header_box,
 753                         text,
 754                         (text_x[idx], text_y),
 755                         font,
 756                         font_scale,
 757                         (0, 0, 0),
 758                         1,
 759                         lineType=cv2.LINE_AA)
 760         logger.debug("header_box.shape: %s", header_box.shape)
 761         return header_box
 762 
 763     def _draw_rect(self, image):
 764         """ Place a white border around a given image.
 765 
 766         Parameters
 767         ----------
 768         image: :class:`numpy.ndarray`
 769             The image to place a border on to
 770         Returns
 771         -------
 772         :class:`numpy.ndarray`
 773             The given image with a border drawn around the outside
 774         """
 775         cv2.rectangle(image, (0, 0), (self._size - 1, self._size - 1), (255, 255, 255), 1)
 776         image = np.clip(image, 0.0, 255.0)
 777         return image.astype("uint8")

 861                                                helptext=val["helptext"])
 862                 self.tk_vars.setdefault(section, dict())[key] = cp_option.tk_var
 863                 config_dicts.setdefault(section, dict())[key] = cp_option

 910     def save_config(self, section=None):
 911         """ Save the configuration ``.ini`` file with the currently stored values.
 912 
 913         Parameters
 914         ----------
 915         section: str, optional
 916             The configuration section to save, If ``None`` provided then all sections are saved.
 917             Default: ``None``
 918         """
 919         logger.debug("Saving %s config", section)
 920         new_config = ConfigParser(allow_no_value=True)
 921         for config_section, items in self._config_dicts.items():
 922             logger.debug("Adding section: '%s')", config_section)
 923             self._config.insert_config_section(config_section,
 924                                                items["helptext"],
 925                                                config=new_config)
 926             for item, options in items.items():
 927                 if item == "helptext":
 928                     continue
 929                 if ((section is not None and config_section != section)
 930                         or config_section not in self.tk_vars):
 931                     new_opt = options.value  # Keep saved item for other sections
 932                     logger.debug("Retaining option: (item: '%s', value: '%s')", item, new_opt)
 933                 else:
 934                     new_opt = self.tk_vars[config_section][item].get()
 935                     logger.debug("Setting option: (item: '%s', value: '%s')", item, new_opt)
 936                     # Set config_dicts value to new saved value
 937                     options.set_initial_value(new_opt)
 938                 helptext = self._config.format_help(options.helptext, is_section=False)
 939                 new_config.set(config_section, helptext)
 940                 new_config.set(config_section, item, str(new_opt))
 941         self._config.config = new_config
 942         self._config.save_config()
 943         logger.info("Saved config: '%s'", self._config.configfile)
