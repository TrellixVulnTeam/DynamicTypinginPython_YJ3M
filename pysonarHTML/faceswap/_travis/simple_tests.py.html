  28 def print_colored(text, color="OK", bold=False):
  29     """ Print colored text
  30     This might not work on windows,
  31     although travis runs windows stuff in git bash, so it might ?
  32     """
  33     color = _COLORS.get(color, color)
  34     print("%s%s%s%s" % (
  35         color, "" if not bold else _COLORS["BOLD"], text, _COLORS["ENDC"]
  36     ))

  54 def run_test(name, cmd):
  55     """ run a test """
  56     global FAIL_COUNT, TEST_COUNT  # pylint:disable=global-statement
  57     print_status("[?] running %s" % name)
  58     print("Cmd: %s" % " ".join(cmd))
  59     TEST_COUNT += 1
  60     try:
  61         check_call(cmd)
  62         print_ok("[+] Test success")
  63         return True
  64     except CalledProcessError as err:
  65         print_fail("[-] Test failed with %s" % err)
  66         FAIL_COUNT += 1
  67         return False

  70 def download_file(url, filename):  # TODO: retry
  71     """ Download a file from given url """
  72     if os.path.isfile(filename):
  73         print_status("[?] '%s' already cached as '%s'" % (url, filename))
  74         return filename
  75     try:
  76         print_status("[?] Downloading '%s' to '%s'" % (url, filename))
  77         video, _ = urlretrieve(url, filename)
  78         return video
  79     except urllib.error.URLError as err:
  80         print_fail("[-] Failed downloading: %s" % err)
  81         return None

  84 def extract_args(detector, aligner, in_path, out_path, args=None):
  85     """ Extraction command """
  86     py_exe = sys.executable
  87     _extract_args = "%s faceswap.py extract -i %s -o %s -D %s -A %s" % (
  88         py_exe, in_path, out_path, detector, aligner
  89     )
  90     if args:
  91         _extract_args += " %s" % args
  92     return _extract_args.split()

  95 def train_args(model, model_path, faces, alignments, iterations=5, batchsize=8, extra_args=""):
  96     """ Train command """

  98     args = "%s faceswap.py train -A %s -ala %s -B %s -alb %s -m %s -t %s -bs %i -it %s %s" % (
  99         py_exe, faces, alignments, faces,
 100         alignments, model_path, model, batchsize, iterations, extra_args
 101     )
 102     return args.split()

 125 def main():
 126     """ Main testing script """
 127     vid_src = "https://faceswap.dev/data/test.mp4"
 128     img_src = "https://archive.org/download/GPN-2003-00070/GPN-2003-00070.jpg"
 129     base_dir = pathjoin(expanduser("~"), "cache", "tests")
 130 
 131     vid_base = pathjoin(base_dir, "vid")
 132     img_base = pathjoin(base_dir, "imgs")
 133     os.makedirs(vid_base, exist_ok=True)
 134     os.makedirs(img_base, exist_ok=True)
 135     py_exe = sys.executable
 136     was_trained = False
 137 
 138     vid_path = download_file(vid_src, pathjoin(vid_base, "test.mp4"))
 139     if not vid_path:
 140         print_fail("[-] Aborting")
 141         sys.exit(1)
 142     vid_extract = run_test(
 143         "Extraction video with cv2-dnn detector and cv2-dnn aligner.",
 144         extract_args("Cv2-Dnn", "Cv2-Dnn", vid_path, pathjoin(vid_base, "faces"))
 145     )
 146 
 147     img_path = download_file(img_src, pathjoin(img_base, "test_img.jpg"))
 148     if not img_path:
 149         print_fail("[-] Aborting")
 150         sys.exit(1)
 151     run_test(
 152         "Extraction images with cv2-dnn detector and cv2-dnn aligner.",
 153         extract_args("Cv2-Dnn", "Cv2-Dnn", img_base, pathjoin(img_base, "faces"))
 154     )
 155 
 156     if vid_extract:
 157         run_test(
 158             "Sort faces.",
 159             sort_args(
 160                 pathjoin(vid_base, "faces"), pathjoin(vid_base, "faces_sorted"),
 161                 sortby="face", method="rename"
 162             )
 163         )
 164 
 165         run_test(
 166             "Rename sorted faces.",
 167             (
 168                 py_exe, "tools.py", "alignments", "-j", "rename",
 169                 "-a", pathjoin(vid_base, "test_alignments.fsa"),
 170                 "-fc", pathjoin(vid_base, "faces_sorted"),
 171             )
 172         )
 173 
 174         run_test(
 175             "Train lightweight model for 1 iteration with WTL.",
 176             train_args(
 177                 "lightweight", pathjoin(vid_base, "model"),
 178                 pathjoin(vid_base, "faces"), pathjoin(vid_base, "test_alignments.fsa"),
 179                 iterations=1, extra_args="-wl"
 180             )
 181         )
 182 
 183         was_trained = run_test(
 184             "Train lightweight model for 5 iterations WITHOUT WTL.",
 185             train_args(
 186                 "lightweight", pathjoin(vid_base, "model"),
 187                 pathjoin(vid_base, "faces"), pathjoin(vid_base, "test_alignments.fsa")
 188             )
 189         )
 190 
 191     if was_trained:
 192         run_test(
 193             "Convert video.",
 194             convert_args(
 195                 vid_path, pathjoin(vid_base, "conv"),
 196                 pathjoin(vid_base, "model"), "ffmpeg"
 197             )
 198         )
 199 
 200         run_test(
 201             "Convert images.",
 202             convert_args(
 203                 img_base, pathjoin(img_base, "conv"),
 204                 pathjoin(vid_base, "model"), "opencv"
 205             )
 206         )
 207 
 208     if FAIL_COUNT == 0:
 209         print_ok("[+] Failed %i/%i tests." % (FAIL_COUNT, TEST_COUNT))
 210         sys.exit(0)
 211     else:
 212         print_fail("[-] Failed %i/%i tests." % (FAIL_COUNT, TEST_COUNT))
 213         sys.exit(1)
