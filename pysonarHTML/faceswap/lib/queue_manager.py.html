   2 """ Queue Manager for faceswap

   7 import logging
   8 import threading

  10 from queue import Queue, Empty as QueueEmpty  # pylint: disable=unused-import; # noqa

  13 logger = logging.getLogger(__name__)  # pylint: disable=invalid-name

  15 
  16 class QueueManager():
  17     """ Manage queues for availabilty across processes
  18         Don't import this class directly, instead
  19         import the variable: queue_manager """
  20     def __init__(self):
  21         logger.debug("Initializing %s", self.__class__.__name__)
  22 
  23         self.shutdown = threading.Event()
  24         self.queues = dict()
  25         logger.debug("Initialized %s", self.__class__.__name__)
  26 
  27     def add_queue(self, name, maxsize=0):
  28         """ Add a queue to the manager
  29 
  30             Adds an event "shutdown" to the queue that can be used to indicate
  31             to a process that any activity on the queue should cease """
  32 
  33         logger.debug("QueueManager adding: (name: '%s', maxsize: %s)", name, maxsize)
  34         if name in self.queues.keys():
  35             raise ValueError("Queue '{}' already exists.".format(name))
  36 
  37         queue = Queue(maxsize=maxsize)
  38 
  39         setattr(queue, "shutdown", self.shutdown)
  40         self.queues[name] = queue
  41         logger.debug("QueueManager added: (name: '%s')", name)
  42 
  43     def del_queue(self, name):
  44         """ remove a queue from the manager """
  45         logger.debug("QueueManager deleting: '%s'", name)
  46         del self.queues[name]
  47         logger.debug("QueueManager deleted: '%s'", name)
  48 
  49     def get_queue(self, name, maxsize=0):
  50         """ Return a queue from the manager
  51             If it doesn't exist, create it """
  52         logger.debug("QueueManager getting: '%s'", name)
  53         queue = self.queues.get(name, None)
  54         if not queue:
  55             self.add_queue(name, maxsize)
  56             queue = self.queues[name]
  57         logger.debug("QueueManager got: '%s'", name)
  58         return queue
  59 
  60     def terminate_queues(self):
  61         """ Set shutdown event, clear and send EOF to all queues
  62             To be called if there is an error """
  63         logger.debug("QueueManager terminating all queues")
  64         self.shutdown.set()
  65         self.flush_queues()
  66         for q_name, queue in self.queues.items():
  67             logger.debug("QueueManager terminating: '%s'", q_name)
  68             queue.put("EOF")
  69         logger.debug("QueueManager terminated all queues")
  70 
  71     def flush_queues(self):
  72         """ Empty out all queues """
  73         for q_name in self.queues.keys():
  74             self.flush_queue(q_name)
  75         logger.debug("QueueManager flushed all queues")
  76 
  77     def flush_queue(self, q_name):
  78         """ Empty out a specific queue """
  79         logger.debug("QueueManager flushing: '%s'", q_name)
  80         queue = self.queues[q_name]
  81         while not queue.empty():
  82             queue.get(True, 1)
  83 
  84     def debug_monitor(self, update_secs=2):
  85         """ Debug tool for monitoring queues """
  86         thread = threading.Thread(target=self.debug_queue_sizes,
  87                                   args=(update_secs, ))
  88         thread.daemon = True
  89         thread.start()
  90 
  91     def debug_queue_sizes(self, update_secs):
  92         """ Output the queue sizes
  93             logged to INFO so it also displays in console
  94         """
  95         while True:
  96             logger.info("====================================================")
  97             for name in sorted(self.queues.keys()):
  98                 logger.info("%s: %s", name, self.queues[name].qsize())
  99             sleep(update_secs)
