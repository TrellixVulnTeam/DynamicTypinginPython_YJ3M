  19 class TrainingDataGenerator():
  20     """ A Training Data Generator for compiling data for feeding to a model.
  21 
  22     This class is called from :mod:`plugins.train.trainer._base` and launches a background
  23     iterator that compiles augmented data, target data and sample data.
  24 
  25     Parameters
  26     ----------
  27     model_input_size: int
  28         The expected input size for the model. It is assumed that the input to the model is always
  29         a square image. This is the size, in pixels, of the `width` and the `height` of the input
  30         to the model.
  31     model_output_shapes: list
  32         A list of tuples defining the output shapes from the model, in the order that the outputs
  33         are returned. The tuples should be in (`height`, `width`, `channels`) format.
  34     training_opts: dict
  35         This is a dictionary of model training options as defined in
  36         :mod:`plugins.train.model._base`. These options will be defined by the user from the
  37         provided cli options or from the model ``config.ini``. At a minimum this ``dict`` should
  38         contain the following keys:
  39 
  40         * **coverage_ratio** (`float`) - The ratio of the training image to be trained on. \
  41         Dictates how much of the image will be cropped out. E.G: a coverage ratio of 0.625 \
  42         will result in cropping a 160px box from a 256px image (256 * 0.625 = 160).
  43 
  44         * **augment_color** (`bool`) - ``True`` if color is to be augmented, otherwise ``False`` \
  45 
  46         * **no_flip** (`bool`) - ``True`` if the image shouldn't be randomly flipped as part of \
  47         augmentation, otherwise ``False``
  48 
  49         * **warp_to_landmarks** (`bool`) - ``True`` if the random warp method should warp to \
  50         similar landmarks from the other side, ``False`` if the standard random warp method \
  51         should be used. If ``True`` then the additional key ``landmarks`` must be provided.
  52 
  53         * **landmarks** (`dict`, `optional`). Required if :attr:`warp_to_landmarks` is \
  54         ``True``. Returning dictionary has a key of **side** (`str`) the value of which is a \
  55         `dict` of {**filename** (`str`): **68 point landmarks** (`numpy.ndarray`)}.
  56 
  57         * **masks** (`dict`, `optional`). Required if :attr:`penalized_mask_loss` or \
  58         :attr:`learn_mask` is ``True``. Returning dictionary has a key of **side** (`str`) the \
  59         value of which is a `dict` of {**filename** (`str`): :class:`lib.faces_detect.Mask`}.
  60 
  61     config: dict
  62         The configuration ``dict`` generated from :file:`config.train.ini` containing the trainer \
  63         plugin configuration options.
  64     """

  67                      "training_opts: %s, landmarks: %s, masks: %s, config: %s)",
  68                      self.__class__.__name__, model_input_size, model_output_shapes,
  69                      {key: val
  70                       for key, val in training_opts.items() if key not in ("landmarks", "masks")},
  71                      {key: len(val)
  72                       for key, val in training_opts.get("landmarks", dict()).items()},
  73                      {key: len(val) for key, val in training_opts.get("masks", dict()).items()},
  74                      config)
  75         self._config = config
  76         self._model_input_size = model_input_size
  77         self._model_output_shapes = model_output_shapes
  78         self._training_opts = training_opts
  79         self._landmarks = self._training_opts.get("landmarks", None)

  89     def minibatch_ab(self, images, batchsize, side,
  90                      do_shuffle=True, is_preview=False, is_timelapse=False):
  91         """ A Background iterator to return augmented images, samples and targets.
  92 
  93         The exit point from this class and the sole attribute that should be referenced. Called
  94         from :mod:`plugins.train.trainer._base`. Returns an iterator that yields images for
  95         training, preview and time-lapses.
  96 
  97         Parameters
  98         ----------
  99         images: list
 100             A list of image paths that will be used to compile the final augmented data from.
 101         batchsize: int
 102             The batchsize for this iterator. Images will be returned in ``numpy.ndarray`` s of
 103             this size from the iterator.
 104         side: {'a' or 'b'}
 105             The side of the model that this iterator is for.
 106         do_shuffle: bool, optional
 107             Whether data should be shuffled prior to loading from disk. If true, each time the full
 108             list of filenames are processed, the data will be reshuffled to make sure they are not
 109             returned in the same order. Default: ``True``
 110         is_preview: bool, optional
 111             Indicates whether this iterator is generating preview images. If ``True`` then certain
 112             augmentations will not be performed. Default: ``False``
 113         is_timelapse: bool optional
 114             Indicates whether this iterator is generating time-lapse images. If ``True``, then
 115             certain augmentations will not be performed. Default: ``False``
 116 
 117         Yields
 118         ------
 119         dict
 120             The following items are contained in each ``dict`` yielded from this iterator:
 121 
 122             * **feed** (`numpy.ndarray`) - The feed for the model. The array returned is in the \
 123             format (`batchsize`, `height`, `width`, `channels`). This is the :attr:`x` parameter \
 124             for :func:`keras.models.model.train_on_batch`.
 125 
 126             * **targets** (`list`) - A list of 4-dimensional ``numpy.ndarray`` s in the order \
 127             and size of each output of the model as defined in :attr:`model_output_shapes`. the \
 128             format of these arrays will be (`batchsize`, `height`, `width`, `3`). This is \
 129             the :attr:`y` parameter for :func:`keras.models.model.train_on_batch` **NB:** \
 130             masks are not included in the ``targets`` list. If required for feeding into the \
 131             Keras model, they will need to be added to this list in \
 132             :mod:`plugins.train.trainer._base` from the ``masks`` key.
 133 
 134             * **masks** (`numpy.ndarray`) - A 4-dimensional array containing the target masks in \
 135             the format (`batchsize`, `height`, `width`, `1`).
 136 
 137             * **samples** (`numpy.ndarray`) - A 4-dimensional array containing the samples for \
 138             feeding to the model's predict function for generating preview and time-lapse \
 139             samples. The array will be in the format (`batchsize`, `height`, `width`, \
 140             `channels`). **NB:** This item will only exist in the ``dict`` if :attr:`is_preview` \
 141             or :attr:`is_timelapse` is ``True``
 142         """
 143         logger.debug("Queue batches: (image_count: %s, batchsize: %s, side: '%s', do_shuffle: %s, "
 144                      "is_preview, %s, is_timelapse: %s)", len(images), batchsize, side, do_shuffle,
 145                      is_preview, is_timelapse)
 146         self._batchsize = batchsize
 147         self._processing = ImageAugmentation(batchsize,
 148                                              is_preview or is_timelapse,
 149                                              self._model_input_size,
 150                                              self._model_output_shapes,
 151                                              self._training_opts.get("coverage_ratio", 0.625),
 152                                              self._config)
 153         args = (images, side, do_shuffle, batchsize)
 154         batcher = BackgroundGenerator(self._minibatch, thread_count=2, args=args)
 155         return batcher.iterator()

 158     def _validate_samples(self, data):
 159         """ Ensures that the total number of images within :attr:`images` is greater or equal to
 160         the selected :attr:`batchsize`. Raises an exception if this is not the case. """
 161         length = len(data)
 162         msg = ("Number of images is lower than batch-size (Note that too few "
 163                "images may lead to bad training). # images: {}, "
 164                "batch-size: {}".format(length, self._batchsize))
 165         try:
 166             assert length >= self._batchsize, msg
 167         except AssertionError as err:
 168             msg += ("\nYou should increase the number of images in your training set or lower "
 169                     "your batch-size.")
 170             raise FaceswapError(msg) from err
 171 
 172     def _minibatch(self, images, side, do_shuffle, batchsize):
 173         """ A generator function that yields the augmented, target and sample images.
 174         see :func:`minibatch_ab` for more details on the output. """
 175         logger.debug("Loading minibatch generator: (image_count: %s, side: '%s', do_shuffle: %s)",
 176                      len(images), side, do_shuffle)
 177         self._validate_samples(images)
 178 
 179         def _img_iter(imgs):
 180             while True:
 181                 if do_shuffle:
 182                     shuffle(imgs)
 183                 for img in imgs:
 184                     yield img
 185 
 186         img_iter = _img_iter(images)
 187         while True:
 188             img_paths = [next(img_iter) for _ in range(batchsize)]
 189             yield self._process_batch(img_paths, side)
 190 
 191         logger.debug("Finished minibatch generator: (side: '%s')", side)
 192 
 193     def _process_batch(self, filenames, side):
 194         """ Performs the augmentation and compiles target images and samples. See
 195         :func:`minibatch_ab` for more details on the output. """
 196         logger.trace("Process batch: (filenames: '%s', side: '%s')", filenames, side)
 197         batch = read_image_batch(filenames)
 198         batch = self._apply_mask(filenames, batch, side)
 199         processed = dict()
 200 
 201         # Initialize processing training size on first image
 202         if not self._processing.initialized:
 203             self._processing.initialize(batch.shape[1])
 204 
 205         # Get Landmarks prior to manipulating the image
 206         if self._training_opts["warp_to_landmarks"]:
 207             batch_src_pts = self._get_landmarks(filenames, side)
 208             batch_dst_pts = self._get_closest_match(filenames, side, batch_src_pts)
 209             warp_kwargs = dict(batch_src_points=batch_src_pts,
 210                                batch_dst_points=batch_dst_pts)
 211         else:
 212             warp_kwargs = dict()
 213 
 214         # Color Augmentation of the image only
 215         if self._training_opts["augment_color"]:
 216             batch[..., :3] = self._processing.color_adjust(batch[..., :3])
 217 
 218         # Random Transform and flip
 219         batch = self._processing.transform(batch)
 220         if not self._training_opts["no_flip"]:
 221             batch = self._processing.random_flip(batch)
 222 
 223         # Add samples to output if this is for display
 224         if self._processing.is_display:
 225             processed["samples"] = batch[..., :3].astype("float32") / 255.0
 226 
 227         # Get Targets
 228         processed.update(self._processing.get_targets(batch))
 229 
 230         # Random Warp # TODO change masks to have a input mask and a warped target mask
 231         processed["feed"] = [self._processing.warp(batch[..., :3],
 232                                                    self._training_opts["warp_to_landmarks"],
 233                                                    **warp_kwargs)]
 234 
 235         logger.trace("Processed batch: (filenames: %s, side: '%s', processed: %s)",
 236                      filenames,
 237                      side,
 238                      {k: v.shape if isinstance(v, np.ndarray) else[i.shape for i in v]
 239                       for k, v in processed.items()})
 240 
 241         return processed

 243     def _apply_mask(self, filenames, batch, side):
 244         """ Applies the mask to the 4th channel of the image. If masks are not being used
 245         applies a dummy all ones mask """
 246         logger.trace("Input batch shape: %s, side: %s", batch.shape, side)
 247         if self._masks is None:
 248             logger.trace("Creating dummy masks. side: %s", side)
 249             masks = np.ones_like(batch[..., :1], dtype=batch.dtype)
 250         else:
 251             logger.trace("Obtaining masks for batch. side: %s", side)
 252             masks = np.array([self._masks[side][filename].mask
 253                               for filename, face in zip(filenames, batch)], dtype=batch.dtype)
 254             masks = self._resize_masks(batch.shape[1], masks)
 255 
 256         logger.trace("masks shape: %s", masks.shape)
 257         batch = np.concatenate((batch, masks), axis=-1)
 258         logger.trace("Output batch shape: %s, side: %s", batch.shape, side)
 259         return batch

 261     @staticmethod
 262     def _resize_masks(target_size, masks):
 263         """ Resize the masks to the target size """
 264         logger.trace("target size: %s, masks shape: %s", target_size, masks.shape)
 265         mask_size = masks.shape[1]
 266         if target_size == mask_size:
 267             logger.trace("Mask and targets the same size. Not resizing")
 268             return masks
 269         interpolator = cv2.INTER_CUBIC if mask_size < target_size else cv2.INTER_AREA
 270         masks = np.array([cv2.resize(mask,
 271                                      (target_size, target_size),
 272                                      interpolation=interpolator)[..., None]
 273                           for mask in masks])
 274         logger.trace("Resized masks: %s", masks.shape)
 275         return masks

 277     def _get_landmarks(self, filenames, side):
 278         """ Obtains the 68 Point Landmarks for the images in this batch. This is only called if
 279         config item ``warp_to_landmarks`` is ``True``. If the landmarks for an image cannot be
 280         found, then an error is raised. """
 281         logger.trace("Retrieving landmarks: (filenames: %s, side: '%s')", filenames, side)
 282         src_points = [self._landmarks[side].get(filename, None) for filename in filenames]
 283         # Raise error on missing alignments
 284         if not all(isinstance(pts, np.ndarray) for pts in src_points):
 285             missing = [filenames[idx] for idx, pts in enumerate(src_points) if pts is None]
 286             msg = ("Files missing alignments for this batch: {}"
 287                    "\nAt least one of your images does not have a matching entry in your "
 288                    "alignments file."
 289                    "\nIf you are using 'warp to landmarks' then every "
 290                    "face you intend to train on must exist within the alignments file."
 291                    "\nThe specific files that caused this failure are listed above."
 292                    "\nMost likely there will be more than just these files missing from the "
 293                    "alignments file. You can use the Alignments Tool to help identify missing "
 294                    "alignments".format(missing))
 295             raise FaceswapError(msg)
 296 
 297         logger.trace("Returning: (src_points: %s)", [str(src) for src in src_points])
 298         return np.array(src_points)

 300     def _get_closest_match(self, filenames, side, batch_src_points):
 301         """ Only called if the config item ``warp_to_landmarks`` is ``True``. Gets the closest
 302         matched 68 point landmarks from the opposite training set. """
 303         logger.trace("Retrieving closest matched landmarks: (filenames: '%s', src_points: '%s'",
 304                      filenames, batch_src_points)
 305         landmarks = self._landmarks["a"] if side == "b" else self._landmarks["b"]
 306         closest_hashes = [self._nearest_landmarks.get(filename) for filename in filenames]
 307         if None in closest_hashes:
 308             closest_hashes = self._cache_closest_hashes(filenames, batch_src_points, landmarks)
 309 
 310         batch_dst_points = np.array([landmarks[choice(hsh)] for hsh in closest_hashes])
 311         logger.trace("Returning: (batch_dst_points: %s)", batch_dst_points.shape)
 312         return batch_dst_points
 313 
 314     def _cache_closest_hashes(self, filenames, batch_src_points, landmarks):
 315         """ Cache the nearest landmarks for this batch """
 316         logger.trace("Caching closest hashes")
 317         dst_landmarks = list(landmarks.items())
 318         dst_points = np.array([lm[1] for lm in dst_landmarks])
 319         batch_closest_hashes = list()
 320 
 321         for filename, src_points in zip(filenames, batch_src_points):
 322             closest = (np.mean(np.square(src_points - dst_points), axis=(1, 2))).argsort()[:10]
 323             closest_hashes = tuple(dst_landmarks[i][0] for i in closest)
 324             self._nearest_landmarks[filename] = closest_hashes
 325             batch_closest_hashes.append(closest_hashes)
 326         logger.trace("Cached closest hashes")
 327         return batch_closest_hashes

 330 class ImageAugmentation():
 331     """ Performs augmentation on batches of training images.
 332 
 333     Parameters
 334     ----------
 335     batchsize: int
 336         The number of images that will be fed through the augmentation functions at once.
 337     is_display: bool
 338         Whether the images being fed through will be used for Preview or Time-lapse. Disables
 339         the "warp" augmentation for these images.
 340     input_size: int
 341         The expected input size for the model. It is assumed that the input to the model is always
 342         a square image. This is the size, in pixels, of the `width` and the `height` of the input
 343         to the model.
 344     output_shapes: list
 345         A list of tuples defining the output shapes from the model, in the order that the outputs
 346         are returned. The tuples should be in (`height`, `width`, `channels`) format.
 347     coverage_ratio: float
 348         The ratio of the training image to be trained on. Dictates how much of the image will be
 349         cropped out. E.G: a coverage ratio of 0.625 will result in cropping a 160px box from a "
 350         "256px image (256 * 0.625 = 160).
 351     config: dict
 352         The configuration ``dict`` generated from :file:`config.train.ini` containing the trainer \
 353         plugin configuration options.
 354 
 355     Attributes
 356     ----------
 357     initialized: bool
 358         Flag to indicate whether :class:`ImageAugmentation` has been initialized with the training
 359         image size in order to cache certain augmentation operations (see :func:`initialize`)
 360     is_display: bool
 361         Flag to indicate whether these augmentations are for time-lapses/preview images (``True``)
 362         or standard training data (``False)``
 363     """
 364     def __init__(self, batchsize, is_display, input_size, output_shapes, coverage_ratio, config):
 365         logger.debug("Initializing %s: (batchsize: %s, is_display: %s, input_size: %s, "
 366                      "output_shapes: %s, coverage_ratio: %s, config: %s)",
 367                      self.__class__.__name__, batchsize, is_display, input_size, output_shapes,
 368                      coverage_ratio, config)
 369 
 370         self.initialized = False
 371         self.is_display = is_display
 372 
 373         # Set on first image load from initialize
 374         self._training_size = 0
 375         self._constants = None
 376 
 377         self._batchsize = batchsize
 378         self._config = config
 379         # Transform and Warp args
 380         self._input_size = input_size
 381         self._output_sizes = [shape[1] for shape in output_shapes if shape[2] == 3]
 382         logger.debug("Output sizes: %s", self._output_sizes)
 383         # Warp args
 384         self._coverage_ratio = coverage_ratio
 385         self._scale = 5  # Normal random variable scale
 386 
 387         logger.debug("Initialized %s", self.__class__.__name__)
 388 
 389     def initialize(self, training_size):
 390         """ Initializes the caching of constants for use in various image augmentations.
 391 
 392         The training image size is not known prior to loading the images from disk and commencing
 393         training, so it cannot be set in the ``__init__`` method. When the first training batch is
 394         loaded this function should be called to initialize the class and perform various
 395         calculations based on this input size to cache certain constants for image augmentation
 396         calculations.
 397 
 398         Parameters
 399         ----------
 400         training_size: int
 401              The size of the training images stored on disk that are to be fed into
 402              :class:`ImageAugmentation`. The training images should always be square and of the
 403              same size. This is the size, in pixels, of the `width` and the `height` of the
 404              training images.
 405          """
 406         logger.debug("Initializing constants. training_size: %s", training_size)
 407         self._training_size = training_size
 408         coverage = int(self._training_size * self._coverage_ratio)
 409 
 410         # Color Aug
 411         clahe_base_contrast = training_size // 128
 412         # Target Images
 413         tgt_slices = slice(self._training_size // 2 - coverage // 2,
 414                            self._training_size // 2 + coverage // 2)
 415 
 416         # Random Warp
 417         warp_range_ = np.linspace(self._training_size // 2 - coverage // 2,
 418                                   self._training_size // 2 + coverage // 2, 5, dtype='float32')
 419         warp_mapx = np.broadcast_to(warp_range_, (self._batchsize, 5, 5)).astype("float32")
 420         warp_mapy = np.broadcast_to(warp_mapx[0].T, (self._batchsize, 5, 5)).astype("float32")
 421 
 422         warp_pad = int(1.25 * self._input_size)
 423         warp_slices = slice(warp_pad // 10, -warp_pad // 10)
 424 
 425         # Random Warp Landmarks
 426         p_mx = self._training_size - 1
 427         p_hf = (self._training_size // 2) - 1
 428         edge_anchors = np.array([(0, 0), (0, p_mx), (p_mx, p_mx), (p_mx, 0),
 429                                  (p_hf, 0), (p_hf, p_mx), (p_mx, p_hf), (0, p_hf)]).astype("int32")
 430         edge_anchors = np.broadcast_to(edge_anchors, (self._batchsize, 8, 2))
 431         grids = np.mgrid[0:p_mx:complex(self._training_size), 0:p_mx:complex(self._training_size)]
 432 
 433         self._constants = dict(clahe_base_contrast=clahe_base_contrast,
 434                                tgt_slices=tgt_slices,
 435                                warp_mapx=warp_mapx,
 436                                warp_mapy=warp_mapy,
 437                                warp_pad=warp_pad,
 438                                warp_slices=warp_slices,
 439                                warp_lm_edge_anchors=edge_anchors,
 440                                warp_lm_grids=grids)
 441         self.initialized = True
 442         logger.debug("Initialized constants: %s", {k: str(v) if isinstance(v, np.ndarray) else v
 443                                                    for k, v in self._constants.items()})
 444 
 445     # <<< TARGET IMAGES >>> #
 446     def get_targets(self, batch):
 447         """ Returns the target images, and masks, if required.
 448 
 449         Parameters
 450         ----------
 451         batch: numpy.ndarray
 452             This should be a 4-dimensional array of training images in the format (`batchsize`,
 453             `height`, `width`, `channels`). Targets should be requested after performing image
 454             transformations but prior to performing warps.
 455 
 456         Returns
 457         -------
 458         dict
 459             The following keys will be within the returned dictionary:
 460 
 461             * **targets** (`list`) - A list of 4-dimensional ``numpy.ndarray`` s in the order \
 462             and size of each output of the model as defined in :attr:`output_shapes`. The \
 463             format of these arrays will be (`batchsize`, `height`, `width`, `3`). **NB:** \
 464             masks are not included in the ``targets`` list. If masks are to be included in the \
 465             output they will be returned as their own item from the ``masks`` key.
 466 
 467             * **masks** (`numpy.ndarray`) - A 4-dimensional array containing the target masks in \
 468             the format (`batchsize`, `height`, `width`, `1`).
 469         """
 470         logger.trace("Compiling targets")
 471         slices = self._constants["tgt_slices"]
 472         target_batch = [np.array([cv2.resize(image[slices, slices, :],
 473                                              (size, size),
 474                                              cv2.INTER_AREA)
 475                                   for image in batch], dtype='float32') / 255.
 476                         for size in self._output_sizes]
 477         logger.trace("Target image shapes: %s",
 478                      [tgt_images.shape[1:] for tgt_images in target_batch])
 479 
 480         retval = self._separate_target_mask(target_batch)
 481         logger.trace("Final targets: %s",
 482                      {k: v.shape if isinstance(v, np.ndarray) else [img.shape for img in v]
 483                       for k, v in retval.items()})
 484         return retval
 485 
 486     @staticmethod
 487     def _separate_target_mask(target_batch):
 488         """ Return the batch and the batch of final masks
 489 
 490         Returns the targets as a list of 4-dimensional ``numpy.ndarray`` s of shape (`batchsize`,
 491         `height`, `width`, 3).
 492 
 493         The target masks are returned as its own item and is the 4th channel of the final target
 494         output.
 495         """
 496         logger.trace("target_batch shapes: %s", [tgt.shape for tgt in target_batch])
 497         retval = dict(targets=[batch[..., :3] for batch in target_batch],
 498                       masks=[target_batch[-1][..., 3:]])
 499         logger.trace("returning: %s", {k: [tgt.shape for tgt in v] for k, v in retval.items()})
 500         return retval

 503     def color_adjust(self, batch):
 504         """ Perform color augmentation on the passed in batch.
 505 
 506         The color adjustment parameters are set in :file:`config.train.ini`
 507 
 508         Parameters
 509         ----------
 510         batch: numpy.ndarray
 511             The batch should be a 4-dimensional array of shape (`batchsize`, `height`, `width`,
 512             `3`) and in `BGR` format.
 513 
 514         Returns
 515         ----------
 516         numpy.ndarray
 517             A 4-dimensional array of the same shape as :attr:`batch` with color augmentation
 518             applied.
 519         """
 520         if not self.is_display:
 521             logger.trace("Augmenting color")
 522             batch = batch_convert_color(batch, "BGR2LAB")
 523             batch = self._random_clahe(batch)
 524             batch = self._random_lab(batch)
 525             batch = batch_convert_color(batch, "LAB2BGR")
 526         return batch
 527 
 528     def _random_clahe(self, batch):
 529         """ Randomly perform Contrast Limited Adaptive Histogram Equalization on
 530         a batch of images """
 531         base_contrast = self._constants["clahe_base_contrast"]
 532 
 533         batch_random = np.random.rand(self._batchsize)
 534         indices = np.where(batch_random > self._config.get("color_clahe_chance", 50) / 100)[0]
 535 
 536         grid_bases = np.rint(np.random.uniform(0,
 537                                                self._config.get("color_clahe_max_size", 4),
 538                                                size=indices.shape[0])).astype("uint8")
 539         contrast_adjustment = (grid_bases * (base_contrast // 2))
 540         grid_sizes = contrast_adjustment + base_contrast
 541         logger.trace("Adjusting Contrast. Grid Sizes: %s", grid_sizes)
 542 
 543         clahes = [cv2.createCLAHE(clipLimit=2.0,  # pylint: disable=no-member
 544                                   tileGridSize=(grid_size, grid_size))
 545                   for grid_size in grid_sizes]
 546 
 547         for idx, clahe in zip(indices, clahes):
 548             batch[idx, :, :, 0] = clahe.apply(batch[idx, :, :, 0])
 549         return batch
 550 
 551     def _random_lab(self, batch):
 552         """ Perform random color/lightness adjustment in L*a*b* color space on a batch of
 553         images """
 554         amount_l = self._config.get("color_lightness", 30) / 100
 555         amount_ab = self._config.get("color_ab", 8) / 100
 556         adjust = np.array([amount_l, amount_ab, amount_ab], dtype="float32")
 557         randoms = (
 558             (np.random.rand(self._batchsize, 1, 1, 3).astype("float32") * (adjust * 2)) - adjust)
 559         logger.trace("Random LAB adjustments: %s", randoms)
 560 
 561         for image, rand in zip(batch, randoms):
 562             for idx in range(rand.shape[-1]):
 563                 adjustment = rand[:, :, idx]
 564                 if adjustment >= 0:
 565                     image[:, :, idx] = ((255 - image[:, :, idx]) * adjustment) + image[:, :, idx]
 566                 else:
 567                     image[:, :, idx] = image[:, :, idx] * (1 + adjustment)
 568         return batch

 571     def transform(self, batch):
 572         """ Perform random transformation on the passed in batch.
 573 
 574         The transformation parameters are set in :file:`config.train.ini`
 575 
 576         Parameters
 577         ----------
 578         batch: numpy.ndarray
 579             The batch should be a 4-dimensional array of shape (`batchsize`, `height`, `width`,
 580             `channels`) and in `BGR` format.
 581 
 582         Returns
 583         ----------
 584         numpy.ndarray
 585             A 4-dimensional array of the same shape as :attr:`batch` with transformation applied.
 586         """
 587         if self.is_display:
 588             return batch
 589         logger.trace("Randomly transforming image")
 590         rotation_range = self._config.get("rotation_range", 10)
 591         zoom_range = self._config.get("zoom_range", 5) / 100
 592         shift_range = self._config.get("shift_range", 5) / 100
 593 
 594         rotation = np.random.uniform(-rotation_range,
 595                                      rotation_range,
 596                                      size=self._batchsize).astype("float32")
 597         scale = np.random.uniform(1 - zoom_range,
 598                                   1 + zoom_range,
 599                                   size=self._batchsize).astype("float32")
 600         tform = np.random.uniform(
 601             -shift_range,
 602             shift_range,
 603             size=(self._batchsize, 2)).astype("float32") * self._training_size
 604 
 605         mats = np.array(
 606             [cv2.getRotationMatrix2D((self._training_size // 2, self._training_size // 2),
 607                                      rot,
 608                                      scl)
 609              for rot, scl in zip(rotation, scale)]).astype("float32")
 610         mats[..., 2] += tform
 611 
 612         batch = np.array([cv2.warpAffine(image,
 613                                          mat,
 614                                          (self._training_size, self._training_size),
 615                                          borderMode=cv2.BORDER_REPLICATE)
 616                           for image, mat in zip(batch, mats)])
 617 
 618         logger.trace("Randomly transformed image")
 619         return batch

 621     def random_flip(self, batch):
 622         """ Perform random horizontal flipping on the passed in batch.
 623 
 624         The probability of flipping an image is set in :file:`config.train.ini`
 625 
 626         Parameters
 627         ----------
 628         batch: numpy.ndarray
 629             The batch should be a 4-dimensional array of shape (`batchsize`, `height`, `width`,
 630             `channels`) and in `BGR` format.
 631 
 632         Returns
 633         ----------
 634         numpy.ndarray
 635             A 4-dimensional array of the same shape as :attr:`batch` with transformation applied.
 636         """
 637         if not self.is_display:
 638             logger.trace("Randomly flipping image")
 639             randoms = np.random.rand(self._batchsize)
 640             indices = np.where(randoms > self._config.get("random_flip", 50) / 100)[0]
 641             batch[indices] = batch[indices, :, ::-1]
 642             logger.trace("Randomly flipped %s images of %s", len(indices), self._batchsize)
 643         return batch

 645     def warp(self, batch, to_landmarks=False, **kwargs):
 646         """ Perform random warping on the passed in batch by one of two methods.
 647 
 648         Parameters
 649         ----------
 650         batch: numpy.ndarray
 651             The batch should be a 4-dimensional array of shape (`batchsize`, `height`, `width`,
 652             `3`) and in `BGR` format.
 653         to_landmarks: bool, optional
 654             If ``False`` perform standard random warping of the input image. If ``True`` perform
 655             warping to semi-random similar corresponding landmarks from the other side. Default:
 656             ``False``
 657         kwargs: dict
 658             If :attr:`to_landmarks` is ``True`` the following additional kwargs must be passed in:
 659 
 660             * **batch_src_points** (`numpy.ndarray`) - A batch of 68 point landmarks for the \
 661             source faces. This is a 3-dimensional array in the shape (`batchsize`, `68`, `2`).
 662 
 663             * **batch_dst_points** (`numpy.ndarray`) - A batch of randomly chosen closest match \
 664             destination faces landmarks. This is a 3-dimensional array in the shape (`batchsize`, \
 665              `68`, `2`).
 666         Returns
 667         ----------
 668         numpy.ndarray
 669             A 4-dimensional array of the same shape as :attr:`batch` with warping applied.
 670         """
 671         if to_landmarks:
 672             return self._random_warp_landmarks(batch, **kwargs).astype("float32") / 255.0
 673         return self._random_warp(batch).astype("float32") / 255.0
 674 
 675     def _random_warp(self, batch):
 676         """ Randomly warp the input batch """
 677         logger.trace("Randomly warping batch")
 678         mapx = self._constants["warp_mapx"]
 679         mapy = self._constants["warp_mapy"]
 680         pad = self._constants["warp_pad"]
 681         slices = self._constants["warp_slices"]
 682 
 683         rands = np.random.normal(size=(self._batchsize, 2, 5, 5),
 684                                  scale=self._scale).astype("float32")
 685         batch_maps = np.stack((mapx, mapy), axis=1) + rands
 686         batch_interp = np.array([[cv2.resize(map_, (pad, pad))[slices, slices] for map_ in maps]
 687                                  for maps in batch_maps])
 688         warped_batch = np.array([cv2.remap(image, interp[0], interp[1], cv2.INTER_LINEAR)
 689                                  for image, interp in zip(batch, batch_interp)])
 690 
 691         logger.trace("Warped image shape: %s", warped_batch.shape)
 692         return warped_batch
 693 
 694     def _random_warp_landmarks(self, batch, batch_src_points, batch_dst_points):
 695         """ From dfaker. Warp the image to a similar set of landmarks from the opposite side """
 696         logger.trace("Randomly warping landmarks")
 697         edge_anchors = self._constants["warp_lm_edge_anchors"]
 698         grids = self._constants["warp_lm_grids"]
 699         slices = self._constants["tgt_slices"]
 700 
 701         batch_dst = (batch_dst_points + np.random.normal(size=batch_dst_points.shape,
 702                                                          scale=2.0))
 703 
 704         face_cores = [cv2.convexHull(np.concatenate([src[17:], dst[17:]], axis=0))
 705                       for src, dst in zip(batch_src_points.astype("int32"),
 706                                           batch_dst.astype("int32"))]
 707 
 708         batch_src = np.append(batch_src_points, edge_anchors, axis=1)
 709         batch_dst = np.append(batch_dst, edge_anchors, axis=1)
 710 
 711         rem_indices = [list(set(idx for fpl in (src, dst)
 712                                 for idx, (pty, ptx) in enumerate(fpl)
 713                                 if cv2.pointPolygonTest(face_core, (pty, ptx), False) >= 0))
 714                        for src, dst, face_core in zip(batch_src[:, :18, :],
 715                                                       batch_dst[:, :18, :],
 716                                                       face_cores)]
 717         batch_src = [np.delete(src, idxs, axis=0) for idxs, src in zip(rem_indices, batch_src)]
 718         batch_dst = [np.delete(dst, idxs, axis=0) for idxs, dst in zip(rem_indices, batch_dst)]
 719 
 720         grid_z = np.array([griddata(dst, src, (grids[0], grids[1]), method="linear")
 721                            for src, dst in zip(batch_src, batch_dst)])
 722         maps = grid_z.reshape(self._batchsize,
 723                               self._training_size,
 724                               self._training_size,
 725                               2).astype("float32")
 726         warped_batch = np.array([cv2.remap(image,
 727                                            map_[..., 1],
 728                                            map_[..., 0],
 729                                            cv2.INTER_LINEAR,
 730                                            cv2.BORDER_TRANSPARENT)
 731                                  for image, map_ in zip(batch, maps)])
 732         warped_batch = np.array([cv2.resize(image[slices, slices, :],
 733                                             (self._input_size, self._input_size),
 734                                             cv2.INTER_AREA)
 735                                  for image in warped_batch])
 736         logger.trace("Warped batch shape: %s", warped_batch.shape)
 737         return warped_batch
