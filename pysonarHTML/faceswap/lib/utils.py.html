  27 class Backend():
  28     """ Return the backend from config/.faceswap of from the `FACESWAP_BACKEND` Environment
  29     Variable.
  30 
  31     If file doesn't exist and a variable hasn't been set, create the config file. """
  32     def __init__(self):
  33         self.backends = {"1": "amd", "2": "cpu", "3": "nvidia"}
  34         self.config_file = self.get_config_file()
  35         self.backend = self.get_backend()
  36 
  37     @staticmethod
  38     def get_config_file():
  39         """ Return location of config file """
  40         pypath = os.path.dirname(os.path.realpath(sys.argv[0]))
  41         config_file = os.path.join(pypath, "config", ".faceswap")
  42         return config_file
  43 
  44     def get_backend(self):
  45         """ Return the backend from either the `FACESWAP_BACKEND` Environment Variable or from
  46         the :loc:`config/.faceswap` configuration file. """
  47         # Check if environment variable is set, if so use that
  48         if "FACESWAP_BACKEND" in os.environ:
  49             fs_backend = os.environ["FACESWAP_BACKEND"].lower()
  50             print("Setting Faceswap backend from environment variable to "
  51                   "{}".format(fs_backend.upper()))
  52             return fs_backend
  53         # Intercept for sphinx docs build
  54         if sys.argv[0].endswith("sphinx-build"):
  55             return "nvidia"
  56         if not os.path.isfile(self.config_file):
  57             self.configure_backend()
  58         while True:
  59             try:
  60                 with open(self.config_file, "r") as cnf:
  61                     config = json.load(cnf)
  62                 break
  63             except json.decoder.JSONDecodeError:
  64                 self.configure_backend()
  65                 continue
  66         fs_backend = config.get("backend", None)
  67         if fs_backend is None or fs_backend.lower() not in self.backends.values():
  68             fs_backend = self.configure_backend()
  69         if current_process().name == "MainProcess":
  70             print("Setting Faceswap backend to {}".format(fs_backend.upper()))
  71         return fs_backend.lower()
  72 
  73     def configure_backend(self):
  74         """ Configure the backend if config file doesn't exist or there is a
  75             problem with the file """
  76         print("First time configuration. Please select the required backend")
  77         while True:
  78             selection = input("1: AMD, 2: CPU, 3: NVIDIA: ")
  79             if selection not in ("1", "2", "3"):
  80                 print("'{}' is not a valid selection. Please try again".format(selection))
  81                 continue
  82             break
  83         fs_backend = self.backends[selection].lower()
  84         config = {"backend": fs_backend}
  85         with open(self.config_file, "w") as cnf:
  86             json.dump(config, cnf)
  87         print("Faceswap config written to: {}".format(self.config_file))
  88         return fs_backend

  99 def get_folder(path, make_folder=True):
 100     """ Return a path to a folder, creating it if it doesn't exist """
 101     logger = logging.getLogger(__name__)  # pylint:disable=invalid-name
 102     logger.debug("Requested path: '%s'", path)
 103     output_dir = Path(path)
 104     if not make_folder and not output_dir.exists():
 105         logger.debug("%s does not exist", path)
 106         return None
 107     output_dir.mkdir(parents=True, exist_ok=True)
 108     logger.debug("Returning: '%s'", output_dir)
 109     return output_dir

 112 def get_image_paths(directory):
 113     """ Return a list of images that reside in a folder """
 114     logger = logging.getLogger(__name__)  # pylint:disable=invalid-name
 115     image_extensions = _image_extensions
 116     dir_contents = list()
 117 
 118     if not os.path.exists(directory):
 119         logger.debug("Creating folder: '%s'", directory)
 120         directory = get_folder(directory)
 121 
 122     dir_scanned = sorted(os.scandir(directory), key=lambda x: x.name)
 123     logger.debug("Scanned Folder contains %s files", len(dir_scanned))
 124     logger.trace("Scanned Folder Contents: %s", dir_scanned)
 125 
 126     for chkfile in dir_scanned:
 127         if any([chkfile.name.lower().endswith(ext)
 128                 for ext in image_extensions]):
 129             logger.trace("Adding '%s' to image list", chkfile.path)
 130             dir_contents.append(chkfile.path)
 131 
 132     logger.debug("Returning %s images", len(dir_contents))
 133     return dir_contents

 136 def convert_to_secs(*args):
 137     """ converts a time to second. Either convert_to_secs(min, secs) or
 138         convert_to_secs(hours, minutes, secs). """
 139     logger = logging.getLogger(__name__)  # pylint:disable=invalid-name
 140     logger.debug("from time: %s", args)
 141     retval = 0.0
 142     if len(args) == 1:
 143         retval = float(args[0])
 144     elif len(args) == 2:
 145         retval = 60 * float(args[0]) + float(args[1])
 146     elif len(args) == 3:
 147         retval = 3600 * float(args[0]) + 60 * float(args[1]) + float(args[2])
 148     logger.debug("to secs: %s", retval)
 149     return retval

 152 def full_path_split(path):
 153     """ Split a given path into all of it's separate components """
 154     logger = logging.getLogger(__name__)  # pylint:disable=invalid-name
 155     allparts = list()
 156     while True:
 157         parts = os.path.split(path)
 158         if parts[0] == path:   # sentinel for absolute paths
 159             allparts.insert(0, parts[0])
 160             break
 161         if parts[1] == path:  # sentinel for relative paths
 162             allparts.insert(0, parts[1])
 163             break
 164         path = parts[0]
 165         allparts.insert(0, parts[1])
 166     logger.trace("path: %s, allparts: %s", path, allparts)
 167     return allparts

 184 def set_system_verbosity(loglevel):
 185     """ Set the verbosity level of tensorflow and suppresses
 186         future and deprecation warnings from any modules
 187         From:
 188         https://stackoverflow.com/questions/35911252/disable-tensorflow-debugging-information
 189         Can be set to:
 190         0 - all logs shown
 191         1 - filter out INFO logs
 192         2 - filter out WARNING logs
 193         3 - filter out ERROR logs  """
 194 
 195     logger = logging.getLogger(__name__)  # pylint:disable=invalid-name
 196     from lib.logger import get_loglevel  # pylint:disable=import-outside-toplevel
 197     numeric_level = get_loglevel(loglevel)
 198     loglevel = "2" if numeric_level > 15 else "0"
 199     logger.debug("System Verbosity level: %s", loglevel)
 200     os.environ['TF_CPP_MIN_LOG_LEVEL'] = loglevel
 201     if loglevel != '0':
 202         for warncat in (FutureWarning, DeprecationWarning, UserWarning):
 203             warnings.simplefilter(action='ignore', category=warncat)

 216 def camel_case_split(identifier):
 217     """ Split a camel case name
 218         from: https://stackoverflow.com/questions/29916065 """
 219     matches = finditer(
 220         ".+?(?:(?<=[a-z])(?=[A-Z])|(?<=[A-Z])(?=[A-Z][a-z])|$)",
 221         identifier)
 222     return [m.group(0) for m in matches]

 240 class GetModel():
 241     """ Check for models in their cache path
 242         If available, return the path, if not available, get, unzip and install model
 243 
 244         model_filename: The name of the model to be loaded (see notes below)
 245         cache_dir:      The model cache folder of the current plugin calling this class
 246                         IE: The folder that holds the model to be loaded.
 247         git_model_id:   The second digit in the github tag that identifies this model.
 248                         See https://github.com/deepfakes-models/faceswap-models for more
 249                         information
 250 
 251         NB: Models must have a certain naming convention:
 252             IE: <model_name>_v<version_number>.<extension>
 253             EG: s3fd_v1.pb
 254 
 255             Multiple models can exist within the model_filename. They should be passed as a list
 256             and follow the same naming convention as above. Any differences in filename should
 257             occur AFTER the version number.
 258             IE: [<model_name>_v<version_number><differentiating_information>.<extension>]
 259             EG: [mtcnn_det_v1.1.py, mtcnn_det_v1.2.py, mtcnn_det_v1.3.py]
 260                 [resnet_ssd_v1.caffemodel, resnet_ssd_v1.prototext]
 261         """
 262 
 263     def __init__(self, model_filename, cache_dir, git_model_id):
 264         self.logger = logging.getLogger(__name__)  # pylint:disable=invalid-name
 265         if not isinstance(model_filename, list):
 266             model_filename = [model_filename]
 267         self.model_filename = model_filename
 268         self.cache_dir = cache_dir
 269         self.git_model_id = git_model_id
 270         self.url_base = "https://github.com/deepfakes-models/faceswap-models/releases/download"
 271         self.chunk_size = 1024  # Chunk size for downloading and unzipping
 272         self.retries = 6
 273         self.get()
 274         self.model_path = self._model_path
 275 
 276     @property
 277     def _model_full_name(self):
 278         """ Return the model full name from the filename(s) """
 279         common_prefix = os.path.commonprefix(self.model_filename)
 280         retval = os.path.splitext(common_prefix)[0]
 281         self.logger.trace(retval)
 282         return retval
 283 
 284     @property
 285     def _model_name(self):
 286         """ Return the model name from the model full name """
 287         retval = self._model_full_name[:self._model_full_name.rfind("_")]
 288         self.logger.trace(retval)
 289         return retval
 290 
 291     @property
 292     def _model_version(self):
 293         """ Return the model version from the model full name """
 294         retval = int(self._model_full_name[self._model_full_name.rfind("_") + 2:])
 295         self.logger.trace(retval)
 296         return retval
 297 
 298     @property
 299     def _model_path(self):
 300         """ Return the model path(s) in the cache folder """
 301         retval = [os.path.join(self.cache_dir, fname) for fname in self.model_filename]
 302         retval = retval[0] if len(retval) == 1 else retval
 303         self.logger.trace(retval)
 304         return retval
 305 
 306     @property
 307     def _model_zip_path(self):
 308         """ Full path to downloaded zip file """
 309         retval = os.path.join(self.cache_dir, "{}.zip".format(self._model_full_name))
 310         self.logger.trace(retval)
 311         return retval
 312 
 313     @property
 314     def _model_exists(self):
 315         """ Check model(s) exist """
 316         if isinstance(self._model_path, list):
 317             retval = all(os.path.exists(pth) for pth in self._model_path)
 318         else:
 319             retval = os.path.exists(self._model_path)
 320         self.logger.trace(retval)
 321         return retval

 323     @property
 324     def _plugin_section(self):
 325         """ Get the plugin section from the config_dir """
 326         path = os.path.normpath(self.cache_dir)
 327         split = path.split(os.sep)
 328         retval = split[split.index("plugins") + 1]
 329         self.logger.trace(retval)
 330         return retval
 331 
 332     @property
 333     def _url_section(self):
 334         """ Return the section ID in github for this plugin type """
 335         sections = dict(extract=1, train=2, convert=3)
 336         retval = sections[self._plugin_section]
 337         self.logger.trace(retval)
 338         return retval

 340     @property
 341     def _url_download(self):
 342         """ Base URL for models """
 343         tag = "v{}.{}.{}".format(self._url_section, self.git_model_id, self._model_version)
 344         retval = "{}/{}/{}.zip".format(self.url_base, tag, self._model_full_name)
 345         self.logger.trace("Download url: %s", retval)
 346         return retval
 347 
 348     @property
 349     def _url_partial_size(self):
 350         """ Return how many bytes have already been downloaded """
 351         zip_file = self._model_zip_path
 352         retval = os.path.getsize(zip_file) if os.path.exists(zip_file) else 0
 353         self.logger.trace(retval)
 354         return retval

 356     def get(self):
 357         """ Check the model exists, if not, download and unzip into location """
 358         if self._model_exists:
 359             self.logger.debug("Model exists: %s", self._model_path)
 360             return
 361         self.download_model()
 362         self.unzip_model()
 363         os.remove(self._model_zip_path)
 364 
 365     def download_model(self):
 366         """ Download model zip to cache folder """
 367         self.logger.info("Downloading model: '%s' from: %s", self._model_name, self._url_download)
 368         for attempt in range(self.retries):
 369             try:
 370                 downloaded_size = self._url_partial_size
 371                 req = urllib.request.Request(self._url_download)
 372                 if downloaded_size != 0:
 373                     req.add_header("Range", "bytes={}-".format(downloaded_size))
 374                 response = urllib.request.urlopen(req, timeout=10)
 375                 self.logger.debug("header info: {%s}", response.info())
 376                 self.logger.debug("Return Code: %s", response.getcode())
 377                 self.write_zipfile(response, downloaded_size)
 378                 break
 379             except (socket_error, socket_timeout,
 380                     urllib.error.HTTPError, urllib.error.URLError) as err:
 381                 if attempt + 1 < self.retries:
 382                     self.logger.warning("Error downloading model (%s). Retrying %s of %s...",
 383                                         str(err), attempt + 2, self.retries)
 384                 else:
 385                     self.logger.error("Failed to download model. Exiting. (Error: '%s', URL: "
 386                                       "'%s')", str(err), self._url_download)
 387                     self.logger.info("You can try running again to resume the download.")
 388                     self.logger.info("Alternatively, you can manually download the model from: %s "
 389                                      "and unzip the contents to: %s",
 390                                      self._url_download, self.cache_dir)
 391                     sys.exit(1)

 393     def write_zipfile(self, response, downloaded_size):
 394         """ Write the model zip file to disk """
 395         length = int(response.getheader("content-length")) + downloaded_size
 396         if length == downloaded_size:
 397             self.logger.info("Zip already exists. Skipping download")
 398             return
 399         write_type = "wb" if downloaded_size == 0 else "ab"
 400         with open(self._model_zip_path, write_type) as out_file:
 401             pbar = tqdm(desc="Downloading",
 402                         unit="B",
 403                         total=length,
 404                         unit_scale=True,
 405                         unit_divisor=1024)
 406             if downloaded_size != 0:
 407                 pbar.update(downloaded_size)
 408             while True:
 409                 buffer = response.read(self.chunk_size)
 410                 if not buffer:
 411                     break
 412                 pbar.update(len(buffer))
 413                 out_file.write(buffer)
 414             pbar.close()
 415 
 416     def unzip_model(self):
 417         """ Unzip the model file to the cache folder """
 418         self.logger.info("Extracting: '%s'", self._model_name)
 419         try:
 420             zip_file = zipfile.ZipFile(self._model_zip_path, "r")
 421             self.write_model(zip_file)
 422         except Exception as err:  # pylint:disable=broad-except
 423             self.logger.error("Unable to extract model file: %s", str(err))
 424             sys.exit(1)
 425 
 426     def write_model(self, zip_file):
 427         """ Extract files from zip file and write, with progress bar """
 428         length = sum(f.file_size for f in zip_file.infolist())
 429         fnames = zip_file.namelist()
 430         self.logger.debug("Zipfile: Filenames: %s, Total Size: %s", fnames, length)
 431         pbar = tqdm(desc="Decompressing",
 432                     unit="B",
 433                     total=length,
 434                     unit_scale=True,
 435                     unit_divisor=1024)
 436         for fname in fnames:
 437             out_fname = os.path.join(self.cache_dir, fname)
 438             self.logger.debug("Extracting from: '%s' to '%s'", self._model_zip_path, out_fname)
 439             zipped = zip_file.open(fname)
 440             with open(out_fname, "wb") as out_file:
 441                 while True:
 442                     buffer = zipped.read(self.chunk_size)
 443                     if not buffer:
 444                         break
 445                     pbar.update(len(buffer))
 446                     out_file.write(buffer)
 447         zip_file.close()
 448         pbar.close()
