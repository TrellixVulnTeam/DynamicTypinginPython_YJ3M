  17 class Alignments():
  18     """ The alignments file is a custom serialized ``.fsa`` file that holds information for each
  19     frame for a video or series of images.
  20 
  21     Specifically, it holds a list of faces that appear in each frame. Each face contains
  22     information detailing their detected bounding box location within the frame, the 68 point
  23     facial landmarks and any masks that have been extracted.
  24 
  25     Additionally it can also hold video meta information (timestamp and whether a frame is a
  26     key frame.)
  27 
  28     Parameters
  29     ----------
  30     folder: str
  31         The folder that contains the alignments ``.fsa`` file
  32     filename: str, optional
  33         The filename of the ``.fsa`` alignments file. If not provided then the given folder will be
  34         checked for a default alignments file filename. Default: "alignments"
  35     """
  36     def __init__(self, folder, filename="alignments"):
  37         logger.debug("Initializing %s: (folder: '%s', filename: '%s')",
  38                      self.__class__.__name__, folder, filename)
  39         self._serializer = get_serializer("compressed")
  40         self._file = self._get_location(folder, filename)
  41         self._data = self._load()
  42         self._update_legacy()
  43         self._hashes_to_frame = dict()
  44         logger.debug("Initialized %s", self.__class__.__name__)

  80     @property
  81     def hashes_to_frame(self):
  82         """ dict: The SHA1 hash of the face mapped to the frame(s) and face index within the frame
  83         that the hash corresponds to. The structure of the dictionary is:
  84 
  85         {**SHA1_hash** (`str`): {**filename** (`str`): **face_index** (`int`)}}.
  86 
  87         Notes
  88         -----
  89         The first time this property is referenced, the dictionary will be created and cached.
  90         Subsequent references will be made to this cached dictionary.
  91         """
  92         if not self._hashes_to_frame:
  93             logger.debug("Generating hashes to frame")
  94             for frame_name, val in self._data.items():
  95                 for idx, face in enumerate(val["faces"]):
  96                     self._hashes_to_frame.setdefault(face["hash"], dict())[frame_name] = idx
  97         return self._hashes_to_frame

  99     @property
 100     def mask_summary(self):
 101         """ dict: The mask type names stored in the alignments :attr:`data` as key with the number
 102         of faces which possess the mask type as value. """
 103         masks = dict()
 104         for val in self._data.values():
 105             for face in val["faces"]:
 106                 if face.get("mask", None) is None:
 107                     masks["none"] = masks.get("none", 0) + 1
 108                 for key in face.get("mask", dict()):
 109                     masks[key] = masks.get(key, 0) + 1
 110         return masks

 115         alignments file then ``None`` is returned for each Key """
 116         retval = dict(pts_time=None, keyframes=None)
 117         pts_time = []

 131     def _get_location(self, folder, filename):
 132         """ Obtains the location of an alignments file.
 133 
 134         If a legacy alignments file is provided/discovered, then the alignments file will be
 135         updated to the custom ``.fsa`` format and saved.
 136 
 137         Parameters
 138         ----------
 139         folder: str
 140             The folder that the alignments file is located in
 141         filename: str
 142             The filename of the alignments file
 143 
 144         Returns
 145         -------
 146         str
 147             The full path to the alignments file
 148         """
 149         logger.debug("Getting location: (folder: '%s', filename: '%s')", folder, filename)
 150         noext_name, extension = os.path.splitext(filename)
 151         if extension in (".json", ".p", ".pickle", ".yaml", ".yml"):
 152             # Reformat legacy alignments file
 153             filename = self._update_file_format(folder, filename)
 154             logger.debug("Updated legacy alignments. New filename: '%s'", filename)
 155         if extension[1:] == self._serializer.file_extension:
 156             logger.debug("Valid Alignments filename provided: '%s'", filename)
 157         else:
 158             filename = "{}.{}".format(noext_name, self._serializer.file_extension)
 159             logger.debug("File extension set from serializer: '%s'",
 160                          self._serializer.file_extension)
 161         location = os.path.join(str(folder), filename)
 162         if not os.path.exists(location):
 163             # Test for old format alignments files and reformat if they exist. This will be
 164             # executed if an alignments file has not been explicitly provided therefore it will not
 165             # have been picked up in the extension test
 166             self._test_for_legacy(location)
 167         logger.verbose("Alignments filepath: '%s'", location)
 168         return location

 198     def backup(self):
 199         """ Create a backup copy of the alignments :attr:`file`.
 200 
 201         Creates a copy of the serialized alignments :attr:`file` appending a
 202         timestamp onto the end of the file name and storing in the same folder as
 203         the original :attr:`file`.
 204         """
 205         logger.debug("Backing up alignments")
 206         if not os.path.isfile(self._file):
 207             logger.debug("No alignments to back up")
 208             return
 209         now = datetime.now().strftime("%Y%m%d_%H%M%S")
 210         src = self._file
 211         split = os.path.splitext(src)
 212         dst = split[0] + "_" + now + split[1]
 213         logger.info("Backing up original alignments to '%s'", dst)
 214         os.rename(src, dst)
 215         logger.debug("Backed up alignments")

 217     def save_video_meta_data(self, pts_time, keyframes):
 218         """ Save video meta data to the alignments file.
 219 
 220         If the alignments file does not have an entry for every frame (e.g. if Extract Every N
 221         was used) then the frame is added to the alignments file with no faces, so that they video
 222         meta data can be stored.
 223 
 224         Parameters
 225         ----------
 226         pts_time: list
 227             A list of presentation timestamps (`float`) in frame index order for every frame in
 228             the input video
 229         keyframes: list
 230             A list of frame indices corresponding to the key frames in the input video
 231         """
 232         sample_filename = next(fname for fname in self.data)
 233         basename = sample_filename[:sample_filename.rfind("_")]
 234         logger.debug("sample filename: %s, base filename: %s", sample_filename, basename)
 235         logger.info("Saving video meta information to Alignments file")
 236         for idx, pts in enumerate(pts_time):
 237             meta = dict(pts_time=pts, keyframe=idx in keyframes)
 238             key = "{}_{:06d}.png".format(basename, idx + 1)
 239             if key not in self.data:
 240                 self.data[key] = dict(video_meta=meta, faces=[])
 241             else:
 242                 self.data[key]["video_meta"] = meta
 243         logger.debug("Alignments count: %s, timestamp count: %s", len(self.data), len(pts_time))
 244         if len(self.data) != len(pts_time):
 245             raise FaceswapError(
 246                 "There is a mismatch between the number of frames found in the video file ({}) "
 247                 "and the number of frames found in the alignments file ({})."
 248                 "\nThis can be caused by a number of issues:"
 249                 "\n  - The video has a Variable Frame Rate and FFMPEG is having a hard time "
 250                 "calculating the correct number of frames."
 251                 "\n  - The video was not cut on a key frame and FFMPEG has dummied in some extra "
 252                 "frames to fill the gap."
 253                 "\n  - You are working with a Merged Alignments file. This is not supported for "
 254                 "your current use case."
 255                 "\nYou should either extract the video to individual frames, re-encode the "
 256                 "video at a constant frame rate and re-run extraction or work with a dedicated "
 257                 "alignments file for your requested video.".format(len(pts_time), len(self.data)))
 258         self.save()

 453     def filter_hashes(self, hash_list, filter_out=False):
 454         """ Remove faces from :attr:`data` based on a given hash list.
 455 
 456         Parameters
 457         ----------
 458         hash_list: list
 459             List of SHA1 hashes in `str` format to use as a filter against :attr:`data`
 460         filter_out: bool, optional
 461             ``True`` if faces should be removed from :attr:`data` when there is a corresponding
 462             match in the given hash_list. ``False`` if faces should be kept in :attr:`data` when
 463             there is a corresponding match in the given hash_list, but removed if there is no
 464             match. Default: ``False``
 465         """
 466         hashset = set(hash_list)
 467         for filename, val in self._data.items():
 468             for idx, face in reversed(list(enumerate(val["faces"]))):
 469                 if ((filter_out and face.get("hash", None) in hashset) or
 470                         (not filter_out and face.get("hash", None) not in hashset)):
 471                     logger.verbose("Filtering out face: (filename: %s, index: %s)", filename, idx)
 472                     del val["faces"][idx]
 473                 else:
 474                     logger.trace("Not filtering out face: (filename: %s, index: %s)",
 475                                  filename, idx)

 479     def yield_faces(self):
 480         """ Generator to obtain all faces with meta information from :attr:`data`. The results
 481         are yielded by frame.
 482 
 483         Notes
 484         -----
 485         The yielded order is non-deterministic.
 486 
 487         Yields
 488         ------
 489         frame_name: str
 490             The frame name that the face belongs to. This is the base name of the frame, as it
 491             appears in :attr:`data`, not the full path
 492         faces: list
 493             The list of face `dict` objects that exist for this frame
 494         face_count: int
 495             The number of faces that exist within :attr:`data` for this frame
 496         frame_fullname: str
 497             The full path (folder and filename) for the yielded frame
 498         """
 499         for frame_fullname, val in self._data.items():
 500             frame_name = os.path.splitext(frame_fullname)[0]
 501             face_count = len(val["faces"])
 502             logger.trace("Yielding: (frame: '%s', faces: %s, frame_fullname: '%s')",
 503                          frame_name, face_count, frame_fullname)
 504             yield frame_name, val["faces"], face_count, frame_fullname

 531         serialization formats and update to current ``.fsa`` format if any are found.

 548     def _update_file_format(self, folder, filename):
 549         """ Convert old style serialized alignments to new ``.fsa`` format.
 550 
 551         Parameters
 552         ----------
 553         folder: str
 554             The folder that the legacy alignments exist in
 555         filename: str
 556             The file name of the legacy alignments
 557 
 558         Returns
 559         -------
 560         str
 561             The full path to the newly created ``.fsa`` alignments file
 562         """
 563         logger.info("Reformatting legacy alignments file...")
 564         old_location = os.path.join(str(folder), filename)
 565         new_location = "{}.{}".format(os.path.splitext(old_location)[0],
 566                                       self._serializer.file_extension)
 567         if os.path.exists(old_location):
 568             if os.path.exists(new_location):
 569                 logger.info("Using existing updated alignments file found at '%s'. If you do not "
 570                             "wish to use this existing file then you should delete or rename it.",
 571                             new_location)
 572             else:
 573                 logger.info("Old location: '%s', New location: '%s'", old_location, new_location)
 574                 load_serializer = get_serializer_from_filename(old_location)
 575                 data = load_serializer.load(old_location)
 576                 self._serializer.save(new_location, data)
 577         return os.path.basename(new_location)

 593         return retval

 595     def _update_legacy_structure(self):
 596         """ Update legacy alignments files from the format `{frame_name: [faces}` to the
 597         format `{frame_name: {faces: [faces]}`."""
 598         for key, val in self._data.items():
 599             self._data[key] = dict(faces=val)

 620     def _update_legacy_landmarksxy(self):
 621         """ Update legacy `landmarksXY` keys to PEP compliant `landmarks_xy` keys. """
 622         update_count = 0
 623         for val in self._data.values():
