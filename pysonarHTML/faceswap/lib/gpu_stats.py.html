   1 #!/usr/bin python3
   2 """ Collects and returns Information on available GPUs.
   3 
   4 The information returned from this module provides information for both Nvidia and AMD GPUs.
   5 However, the information available for Nvidia is far more thorough than what is available for
   6 AMD, where we need to plug into plaidML to pull stats. The quality of this data will vary
   7 depending on the OS' particular OpenCL implementation.
   8 """

  30 class GPUStats():
  31     """ Holds information and statistics about the GPU(s) available on the currently
  32     running system.
  33 
  34     Parameters
  35     ----------
  36     log: bool, optional
  37         Whether the class should output information to the logger. There may be occasions where the
  38         logger has not yet been set up when this class is queried. Attempting to log in these
  39         instances will raise an error. If GPU stats are being queried prior to the logger being
  40         available then this parameter should be set to ``False``. Otherwise set to ``True``.
  41         Default: ``True``
  42     """
  43     def __init__(self, log=True):
  44         # Logger is held internally, as we don't want to log when obtaining system stats on crash
  45         self._logger = logging.getLogger(__name__) if log else None
  46         self._log("debug", "Initializing {}".format(self.__class__.__name__))
  47 
  48         self._plaid = None
  49         self._initialized = False
  50         self._device_count = 0
  51         self._active_devices = list()
  52         self._handles = list()
  53         self._driver = None
  54         self._devices = list()
  55         self._vram = None
  56 
  57         self._initialize(log)
  58 
  59         self._driver = self._get_driver()
  60         self._devices = self._get_devices()
  61         self._vram = self._get_vram()
  62         if not self._active_devices:
  63             self._log("warning", "No GPU detected. Switching to CPU mode")
  64             return
  65 
  66         self._shutdown()
  67         self._log("debug", "Initialized {}".format(self.__class__.__name__))
  68 
  69     @property
  70     def device_count(self):
  71         """int: The number of GPU devices discovered on the system. """
  72         return self._device_count

  74     @property
  75     def _is_plaidml(self):
  76         """ bool: ``True`` if the backend is plaidML otherwise ``False``. """
  77         return self._plaid is not None

  79     @property
  80     def sys_info(self):
  81         """ dict: GPU Stats that are required for system information logging.
  82 
  83         The dictionary contains the following data:
  84 
  85             **vram** (`list`): the total amount of VRAM in Megabytes for each GPU as pertaining to
  86             :attr:`_handles`
  87 
  88             **driver** (`str`): The GPU driver version that is installed on the OS
  89 
  90             **devices** (`list`): The device name of each GPU on the system as pertaining
  91             to :attr:`_handles`
  92 
  93             **devices_active** (`list`): The device name of each active GPU on the system as
  94             pertaining to :attr:`_handles`
  95         """
  96         return dict(vram=self._vram,
  97                     driver=self._driver,
  98                     devices=self._devices,
  99                     devices_active=self._active_devices)

 101     def _log(self, level, message):
 102         """ If the class has been initialized with :attr:`log` as `True` then log the message
 103         otherwise skip logging.
 104 
 105         Parameters
 106         ----------
 107         level: str
 108             The log level to log at
 109         message: str
 110             The message to log
 111         """
 112         if self._logger is None:
 113             return
 114         logger = getattr(self._logger, level.lower())
 115         logger(message)

 117     def _initialize(self, log=False):
 118         """ Initialize the library that will be returning stats for the system's GPU(s).
 119         For Nvidia (on Linux and Windows) the library is `pynvml`. For Nvidia (on macOS) the
 120         library is `pynvx`. For AMD `plaidML` is used.
 121 
 122         Parameters
 123         ----------
 124         log: bool, optional
 125             Whether the class should output information to the logger. There may be occasions where
 126             the logger has not yet been set up when this class is queried. Attempting to log in
 127             these instances will raise an error. If GPU stats are being queried prior to the
 128             logger being available then this parameter should be set to ``False``. Otherwise set
 129             to ``True``. Default: ``False``
 130         """
 131         if not self._initialized:
 132             if get_backend() == "amd":
 133                 self._log("debug", "AMD Detected. Using plaidMLStats")
 134                 loglevel = "INFO" if self._logger is None else self._logger.getEffectiveLevel()
 135                 self._plaid = plaidlib(loglevel=loglevel, log=log)
 136             elif IS_MACOS:
 137                 self._log("debug", "macOS Detected. Using pynvx")
 138                 try:
 139                     pynvx.cudaInit()
 140                 except RuntimeError:
 141                     self._initialized = True
 142                     return
 143             else:
 144                 try:
 145                     self._log("debug", "OS is not macOS. Trying pynvml")
 146                     pynvml.nvmlInit()
 147                 except (pynvml.NVMLError_LibraryNotFound,  # pylint: disable=no-member
 148                         pynvml.NVMLError_DriverNotLoaded,  # pylint: disable=no-member
 149                         pynvml.NVMLError_NoPermission) as err:  # pylint: disable=no-member
 150                     if plaidlib is not None:
 151                         self._log("debug", "pynvml errored. Trying plaidML")
 152                         self._plaid = plaidlib(log=log)
 153                     else:
 154                         msg = ("There was an error reading from the Nvidia Machine Learning "
 155                                "Library. Either you do not have an Nvidia GPU (in which case "
 156                                "this warning can be ignored) or the most likely cause is "
 157                                "incorrectly installed drivers. If this is the case, Please remove "
 158                                "and reinstall your Nvidia drivers before reporting."
 159                                "Original Error: {}".format(str(err)))
 160                         self._log("warning", msg)
 161                         self._initialized = True
 162                         return
 163                 except Exception as err:  # pylint: disable=broad-except
 164                     msg = ("An unhandled exception occured loading pynvml. "
 165                            "Original error: {}".format(str(err)))
 166                     if self._logger:
 167                         self._logger.error(msg)
 168                     else:
 169                         print(msg)
 170                     self._initialized = True
 171                     return
 172             self._initialized = True
 173             self._get_device_count()
 174             self._get_active_devices()
 175             self._get_handles()
 176 
 177     def _shutdown(self):
 178         """ Shutdown pynvml if it was the library used for obtaining stats and set
 179         :attr:`_initialized` back to ``False``. """
 180         if self._initialized:
 181             self._handles = list()
 182             if not IS_MACOS and not self._is_plaidml:
 183                 pynvml.nvmlShutdown()
 184             self._initialized = False
 185 
 186     def _get_device_count(self):
 187         """ Detect the number of GPUs attached to the system and allocate to
 188         :attr:`_device_count`. """
 189         if self._is_plaidml:
 190             self._device_count = self._plaid.device_count
 191         elif IS_MACOS:
 192             self._device_count = pynvx.cudaDeviceGetCount(ignore=True)
 193         else:
 194             try:
 195                 self._device_count = pynvml.nvmlDeviceGetCount()
 196             except pynvml.NVMLError:
 197                 self._device_count = 0
 198         self._log("debug", "GPU Device count: {}".format(self._device_count))
 199 
 200     def _get_active_devices(self):
 201         """ Obtain the indices of active GPUs (those that have not been explicitly excluded by
 202         CUDA_VISIBLE_DEVICES or plaidML) and allocate to :attr:`_active_devices`. """
 203         if self._is_plaidml:
 204             self._active_devices = self._plaid.active_devices
 205         else:
 206             devices = os.environ.get("CUDA_VISIBLE_DEVICES", None)
 207             if self._device_count == 0:
 208                 self._active_devices = list()
 209             elif devices is not None:
 210                 self._active_devices = [int(i) for i in devices.split(",") if devices]
 211             else:
 212                 self._active_devices = list(range(self._device_count))
 213             self._log("debug", "Active GPU Devices: {}".format(self._active_devices))
 214 
 215     def _get_handles(self):
 216         """ Obtain the internal handle identifiers for the system GPUs and allocate to
 217         :attr:`_handles`. """
 218         if self._is_plaidml:
 219             self._handles = self._plaid.devices
 220         elif IS_MACOS:
 221             self._handles = pynvx.cudaDeviceGetHandles(ignore=True)
 222         else:
 223             self._handles = [pynvml.nvmlDeviceGetHandleByIndex(i)
 224                              for i in range(self._device_count)]
 225         self._log("debug", "GPU Handles found: {}".format(len(self._handles)))
 226 
 227     def _get_driver(self):
 228         """ Obtain and return the installed driver version for the system's GPUs.
 229 
 230         Returns
 231         -------
 232         str
 233             The currently installed GPU driver version
 234         """
 235         if self._is_plaidml:
 236             driver = self._plaid.drivers
 237         elif IS_MACOS:
 238             driver = pynvx.cudaSystemGetDriverVersion(ignore=True)
 239         else:
 240             try:
 241                 driver = pynvml.nvmlSystemGetDriverVersion().decode("utf-8")
 242             except pynvml.NVMLError:
 243                 driver = "No Nvidia driver found"
 244         self._log("debug", "GPU Driver: {}".format(driver))
 245         return driver
 246 
 247     def _get_devices(self):
 248         """ Obtain the name of the installed devices. The quality of this information depends on
 249         the backend and OS being used, but it should be sufficient for identifying cards.
 250 
 251         Returns
 252         -------
 253         list
 254             List of device names for connected GPUs as corresponding to the values in
 255             :attr:`_handles`
 256         """
 257         self._initialize()
 258         if self._device_count == 0:
 259             names = list()
 260         if self._is_plaidml:
 261             names = self._plaid.names
 262         elif IS_MACOS:
 263             names = [pynvx.cudaGetName(handle, ignore=True)
 264                      for handle in self._handles]
 265         else:
 266             names = [pynvml.nvmlDeviceGetName(handle).decode("utf-8")
 267                      for handle in self._handles]
 268         self._log("debug", "GPU Devices: {}".format(names))
 269         return names
 270 
 271     def _get_vram(self):
 272         """ Obtain the total VRAM in Megabytes for each connected GPU.
 273 
 274         Returns
 275         -------
 276         list
 277              List of floats containing the total amount of VRAM in Megabytes for each connected GPU
 278              as corresponding to the values in :attr:`_handles
 279         """
 280         self._initialize()
 281         if self._device_count == 0:
 282             vram = list()
 283         elif self._is_plaidml:
 284             vram = self._plaid.vram
 285         elif IS_MACOS:
 286             vram = [pynvx.cudaGetMemTotal(handle, ignore=True) / (1024 * 1024)
 287                     for handle in self._handles]
 288         else:
 289             vram = [pynvml.nvmlDeviceGetMemoryInfo(handle).total /
 290                     (1024 * 1024)
 291                     for handle in self._handles]
 292         self._log("debug", "GPU VRAM: {}".format(vram))
 293         return vram
 294 
 295     def _get_free_vram(self):
 296         """ Obtain the amount of VRAM that is available, in Megabytes, for each connected GPU.
 297 
 298         Returns
 299         -------
 300         list
 301              List of floats containing the amount of VRAM available, in Megabytes, for each
 302              connected GPU as corresponding to the values in :attr:`_handles
 303 
 304         Notes
 305         -----
 306         There is no useful way to get free VRAM on PlaidML. OpenCL loads and unloads VRAM as
 307         required, so this returns the total memory available per card for AMD cards, which us
 308         not particularly useful.
 309 
 310         """
 311         self._initialize()
 312         if self._is_plaidml:
 313             vram = self._plaid.vram
 314         elif IS_MACOS:
 315             vram = [pynvx.cudaGetMemFree(handle, ignore=True) / (1024 * 1024)
 316                     for handle in self._handles]
 317         else:
 318             vram = [pynvml.nvmlDeviceGetMemoryInfo(handle).free / (1024 * 1024)
 319                     for handle in self._handles]
 320         self._shutdown()
 321         self._log("debug", "GPU VRAM free: {}".format(vram))
 322         return vram

 324     def get_card_most_free(self):
 325         """ Obtain statistics for the GPU with the most available free VRAM.
 326 
 327         Returns
 328         -------
 329         dict
 330             The dictionary contains the following data:
 331 
 332                 **card_id** (`int`):  The index of the card as pertaining to :attr:`_handles`
 333 
 334                 **device** (`str`): The name of the device
 335 
 336                 **free** (`float`): The amount of available VRAM on the GPU
 337 
 338                 **total** (`float`): the total amount of VRAM on the GPU
 339 
 340             If a GPU is not detected then the **card_id** is returned as ``-1`` and the amount
 341             of free and total RAM available is fixed to 2048 Megabytes.
 342         """
 343         if self._device_count == 0:
 344             return {"card_id": -1,
 345                     "device": "No GPU devices found",
 346                     "free": 2048,
 347                     "total": 2048}
 348         free_vram = [self._get_free_vram()[i] for i in self._active_devices]
 349         vram_free = max(free_vram)
 350         card_id = self._active_devices[free_vram.index(vram_free)]
 351         retval = {"card_id": card_id,
 352                   "device": self._devices[card_id],
 353                   "free": vram_free,
 354                   "total": self._vram[card_id]}
 355         self._log("debug", "Active GPU Card with most free VRAM: {}".format(retval))
 356         return retval
