  24 def mask_loss_wrapper(loss_func, preprocessing_func=None):
  25     """ A wrapper for mask loss that can perform pre-processing on the input prior to calling the
  26     loss function.
  27 
  28     Parameters
  29     ----------
  30     loss_func: class or function
  31         The actual loss function to use
  32     preprocessing_func: function
  33         The pre-processing function to use. Should take a Keras Input as it's only argument
  34     """
  35     def func(y_true, y_pred):
  36         """ Process input if a processing function has been passed, otherwise just return loss """
  37         if preprocessing_func is not None:
  38             y_true = K.reshape(y_true, [-1] + list(K.int_shape(y_pred)[1:]))
  39             y_true = preprocessing_func(y_true)
  40         return loss_func(y_true, y_pred)
  41     return func

  44 class DSSIMObjective():
  45     """ DSSIM Loss Function
  46 
  47     Difference of Structural Similarity (DSSIM loss function). Clipped between 0 and 0.5
  48 
  49     Parameters
  50     ----------
  51     k_1: float, optional
  52         Parameter of the SSIM. Default: `0.01`
  53     k_2: float, optional
  54         Parameter of the SSIM. Default: `0.03`
  55     kernel_size: int, optional
  56         Size of the sliding window Default: `3`
  57     max_value: float, optional
  58         Max value of the output. Default: `1.0`
  59 
  60     Notes
  61     ------
  62     You should add a regularization term like a l2 loss in addition to this one.

  90     def __init__(self, k_1=0.01, k_2=0.03, kernel_size=3, max_value=1.0):
  91         self.__name__ = 'DSSIMObjective'
  92         self.kernel_size = kernel_size
  93         self.k_1 = k_1
  94         self.k_2 = k_2
  95         self.max_value = max_value
  96         self.c_1 = (self.k_1 * self.max_value) ** 2
  97         self.c_2 = (self.k_2 * self.max_value) ** 2
  98         self.dim_ordering = K.image_data_format()
  99         self.backend = K.backend()

 101     @staticmethod
 102     def __int_shape(input_tensor):
 103         """ Returns the shape of tensor or variable as a tuple of int or None entries.
 104 
 105         Parameters
 106         ----------
 107         input_tensor: tensor or variable
 108             The input to return the shape for
 109 
 110         Returns
 111         -------
 112         tuple
 113             A tuple of integers (or None entries)
 114         """
 115         return K.int_shape(input_tensor)

 117     def __call__(self, y_true, y_pred):
 118         """ Call the DSSIM Loss Function.
 119 
 120         Parameters
 121         ----------
 122         y_true: tensor or variable
 123             The ground truth value
 124         y_pred: tensor or variable
 125             The predicted value
 126 
 127         Returns
 128         -------
 129         tensor
 130             The DSSIM Loss value
 131 
 132         Notes
 133         -----
 134         There are additional parameters for this function. some of the 'modes' for edge behavior
 135         do not yet have a gradient definition in the Theano tree and cannot be used for learning
 136         """
 137 
 138         kernel = [self.kernel_size, self.kernel_size]
 139         y_true = K.reshape(y_true, [-1] + list(self.__int_shape(y_pred)[1:]))
 140         y_pred = K.reshape(y_pred, [-1] + list(self.__int_shape(y_pred)[1:]))
 141         patches_pred = self.extract_image_patches(y_pred,
 142                                                   kernel,
 143                                                   kernel,
 144                                                   'valid',
 145                                                   self.dim_ordering)
 146         patches_true = self.extract_image_patches(y_true,
 147                                                   kernel,
 148                                                   kernel,
 149                                                   'valid',
 150                                                   self.dim_ordering)
 151 
 152         # Get mean
 153         u_true = K.mean(patches_true, axis=-1)
 154         u_pred = K.mean(patches_pred, axis=-1)
 155         # Get variance
 156         var_true = K.var(patches_true, axis=-1)
 157         var_pred = K.var(patches_pred, axis=-1)
 158         # Get standard deviation
 159         covar_true_pred = K.mean(
 160             patches_true * patches_pred, axis=-1) - u_true * u_pred
 161 
 162         ssim = (2 * u_true * u_pred + self.c_1) * (
 163             2 * covar_true_pred + self.c_2)
 164         denom = (K.square(u_true) + K.square(u_pred) + self.c_1) * (
 165             var_pred + var_true + self.c_2)
 166         ssim /= denom  # no need for clipping, c_1 + c_2 make the denorm non-zero
 167         return K.mean((1.0 - ssim) / 2.0)

 196     def extract_image_patches(self, input_tensor, k_sizes, s_sizes,
 197                               padding='same', data_format='channels_last'):
 198         """ Extract the patches from an image.
 199 
 200         Parameters
 201         ----------
 202         input_tensor: tensor
 203             The input image
 204         k_sizes: tuple
 205             2-d tuple with the kernel size
 206         s_sizes: tuple
 207             2-d tuple with the strides size
 208         padding: str, optional
 209             `"same"` or `"valid"`. Default: `"same"`
 210         data_format: str, optional.
 211             `"channels_last"` or `"channels_first"`. Default: `"channels_last"`
 212 
 213         Returns
 214         -------
 215         The (k_w, k_h) patches extracted
 216             Tensorflow ==> (batch_size, w, h, k_w, k_h, c)
 217             Theano ==> (batch_size, w, h, c, k_w, k_h)
 218         """
 219         kernel = [1, k_sizes[0], k_sizes[1], 1]
 220         strides = [1, s_sizes[0], s_sizes[1], 1]
 221         padding = self._preprocess_padding(padding)
 222         if data_format == 'channels_first':
 223             input_tensor = K.permute_dimensions(input_tensor, (0, 2, 3, 1))
 224         patches = extract_image_patches(input_tensor, kernel, strides, [1, 1, 1, 1], padding)
 225         return patches

 229 def PenalizedLoss(mask, loss_func,  # pylint: disable=invalid-name
 230                   mask_prop=1.0, mask_scaling=1.0, preprocessing_func=None):
 231     """ Plaidml and Tensorflow Penalized Loss function.
 232 
 233     Applies the given loss function just to the masked area of the image.
 234 
 235     Parameters
 236     ----------
 237     mask: input tensor
 238         The mask for the current image
 239     loss_func: function
 240         The actual loss function to use
 241     mask_prop: float, optional
 242         The amount of mask propagation. Default: `1.0`
 243     mask_scaling: float, optional
 244         For multi-decoder output the target mask will likely be at full size scaling, so this is
 245         the scaling factor to reduce the mask by. Default: `1.0`
 246     preprocessing_func: function, optional
 247         If preprocessing is required on the input mask, then this should be the function to use.
 248         The function should take a Keras Input as it's only argument. Set to ``None`` if no
 249         preprocessing is to be performed. Default: ``None``
 250     """
 251     def _scale_mask(mask, scaling):
 252         """ Scale the input mask to be the same size as the input face
 253 
 254         Parameters
 255         ----------
 256         mask: input tensor
 257             The mask for the current image
 258         scaling: float
 259             The amount to scale the input mask by
 260 
 261         Returns
 262         -------
 263         tensor
 264             The resized input mask
 265         """
 266         if scaling != 1.0:
 267             size = round(1 / scaling)
 268             mask = K.pool2d(mask,
 269                             pool_size=(size, size),
 270                             strides=(size, size),
 271                             padding="valid",
 272                             data_format=K.image_data_format(),
 273                             pool_mode="avg")
 274         logger.debug("resized tensor: %s", mask)
 275         return mask
 276 
 277     mask = _scale_mask(mask, mask_scaling)
 278     if preprocessing_func is not None:
 279         mask = preprocessing_func(mask)
 280     mask_as_k_inv_prop = 1 - mask_prop
 281     mask = (mask * mask_prop) + mask_as_k_inv_prop
 282 
 283     def _inner_loss(y_true, y_pred):
 284         """ Apply the loss function to the masked area of the image.
 285 
 286          Parameters
 287         ----------
 288         y_true: tensor or variable
 289             The ground truth value
 290         y_pred: tensor or variable
 291             The predicted value
 292 
 293         Returns
 294         -------
 295         tensor
 296             The Loss value
 297 
 298         Notes
 299         -----
 300         Branching because TensorFlow's broadcasting is wonky and plaidML's concatenate is
 301         implemented inefficiently.
 302         """
 303         if K.backend() == "plaidml.keras.backend":
 304             n_true = y_true * mask
 305             n_pred = y_pred * mask
 306         else:
 307             n_true = K.concatenate([y_true[:, :, :, i:i+1] * mask for i in range(3)], axis=-1)
 308             n_pred = K.concatenate([y_pred[:, :, :, i:i+1] * mask for i in range(3)], axis=-1)
 309         return loss_func(n_true, n_pred)
 310     return _inner_loss

 314 def generalized_loss(y_true, y_pred, alpha=1.0, beta=1.0/255.0):
 315     """ Generalized function used to return a large variety of mathematical loss functions.
 316 
 317     The primary benefit is a smooth, differentiable version of L1 loss.
 318 
 319     Parameters
 320     ----------
 321     y_true: tensor or variable
 322         The ground truth value
 323     y_pred: tensor or variable
 324         The predicted value
 325     alpha: float, optional
 326         Penalty factor. Larger number give larger weight to large deviations. Default: `1.0`
 327     beta: float, optional
 328         Scale factor used to adjust to the input scale (i.e. inputs of mean `1e-4` or `256`).
 329         Default: `1.0/255.0`
 330 
 331     Returns
 332     -------
 333     tensor
 334         The loss value from the results of function(y_pred - y_true)
 335 
 336     References
 337     ----------
 338     Barron, J. A More General Robust Loss Function - https://arxiv.org/pdf/1701.03077.pdf
 339 
 340     Example
 341     -------
 342     >>> a=1.0, x>>c , c=1.0/255.0  # will give a smoothly differentiable version of L1 / MAE loss
 343     >>> a=1.999999 (limit as a->2), beta=1.0/255.0 # will give L2 / RMSE loss
 344     """
 345     diff = y_pred - y_true
 346     second = (K.pow(K.pow(diff/beta, 2.) / K.abs(2.-alpha) + 1., (alpha/2.)) - 1.)
 347     loss = (K.abs(2.-alpha)/alpha) * second
 348     loss = K.mean(loss, axis=-1) * beta
 349     return loss

 352 def l_inf_norm(y_true, y_pred):
 353     """ Calculate the L-inf norm as a loss function.
 354 
 355     Parameters
 356     ----------
 357     y_true: tensor or variable
 358         The ground truth value
 359     y_pred: tensor or variable
 360         The predicted value
 361 
 362     Returns
 363     -------
 364     tensor
 365         The loss value
 366     """
 367     diff = K.abs(y_true - y_pred)
 368     max_loss = K.max(diff, axis=(1, 2), keepdims=True)
 369     loss = K.mean(max_loss, axis=-1)
 370     return loss

 373 def gradient_loss(y_true, y_pred):
 374     """ Gradient Loss Function.
 375 
 376     Calculates the first and second order gradient difference between pixels of an image in the x
 377     and y dimensions. These gradients are then compared between the ground truth and the predicted
 378     image and the difference is taken. When used as a loss, its minimization will result in
 379     predicted images approaching the same level of sharpness / blurriness as the ground truth.
 380 
 381     Parameters
 382     ----------
 383     y_true: tensor or variable
 384         The ground truth value
 385     y_pred: tensor or variable
 386         The predicted value
 387 
 388     Returns
 389     -------
 390     tensor
 391         The loss value
 392 
 393     References
 394     ----------
 395     TV+TV2 Regularization with Non-Convex Sparseness-Inducing Penalty for Image Restoration,
 396     Chengwu Lu & Hua Huang, 2014 - http://downloads.hindawi.com/journals/mpe/2014/790547.pdf
 397     """
 398 
 399     def _diff_x(img):
 400         """ X Difference """
 401         x_left = img[:, :, 1:2, :] - img[:, :, 0:1, :]
 402         x_inner = img[:, :, 2:, :] - img[:, :, :-2, :]
 403         x_right = img[:, :, -1:, :] - img[:, :, -2:-1, :]
 404         x_out = K.concatenate([x_left, x_inner, x_right], axis=2)
 405         return x_out * 0.5
 406 
 407     def _diff_y(img):
 408         """ Y Difference """
 409         y_top = img[:, 1:2, :, :] - img[:, 0:1, :, :]
 410         y_inner = img[:, 2:, :, :] - img[:, :-2, :, :]
 411         y_bot = img[:, -1:, :, :] - img[:, -2:-1, :, :]
 412         y_out = K.concatenate([y_top, y_inner, y_bot], axis=1)
 413         return y_out * 0.5
 414 
 415     def _diff_xx(img):
 416         """ X-X Difference """
 417         x_left = img[:, :, 1:2, :] + img[:, :, 0:1, :]
 418         x_inner = img[:, :, 2:, :] + img[:, :, :-2, :]
 419         x_right = img[:, :, -1:, :] + img[:, :, -2:-1, :]
 420         x_out = K.concatenate([x_left, x_inner, x_right], axis=2)
 421         return x_out - 2.0 * img
 422 
 423     def _diff_yy(img):
 424         """ Y-Y Difference """
 425         y_top = img[:, 1:2, :, :] + img[:, 0:1, :, :]
 426         y_inner = img[:, 2:, :, :] + img[:, :-2, :, :]
 427         y_bot = img[:, -1:, :, :] + img[:, -2:-1, :, :]
 428         y_out = K.concatenate([y_top, y_inner, y_bot], axis=1)
 429         return y_out - 2.0 * img
 430 
 431     def _diff_xy(img):
 432         """ X-Y Difference """
 433         # xout1
 434         top_left = img[:, 1:2, 1:2, :] + img[:, 0:1, 0:1, :]
 435         inner_left = img[:, 2:, 1:2, :] + img[:, :-2, 0:1, :]
 436         bot_left = img[:, -1:, 1:2, :] + img[:, -2:-1, 0:1, :]
 437         xy_left = K.concatenate([top_left, inner_left, bot_left], axis=1)
 438 
 439         top_mid = img[:, 1:2, 2:, :] + img[:, 0:1, :-2, :]
 440         mid_mid = img[:, 2:, 2:, :] + img[:, :-2, :-2, :]
 441         bot_mid = img[:, -1:, 2:, :] + img[:, -2:-1, :-2, :]
 442         xy_mid = K.concatenate([top_mid, mid_mid, bot_mid], axis=1)
 443 
 444         top_right = img[:, 1:2, -1:, :] + img[:, 0:1, -2:-1, :]
 445         inner_right = img[:, 2:, -1:, :] + img[:, :-2, -2:-1, :]
 446         bot_right = img[:, -1:, -1:, :] + img[:, -2:-1, -2:-1, :]
 447         xy_right = K.concatenate([top_right, inner_right, bot_right], axis=1)
 448 
 449         # Xout2
 450         top_left = img[:, 0:1, 1:2, :] + img[:, 1:2, 0:1, :]
 451         inner_left = img[:, :-2, 1:2, :] + img[:, 2:, 0:1, :]
 452         bot_left = img[:, -2:-1, 1:2, :] + img[:, -1:, 0:1, :]
 453         xy_left = K.concatenate([top_left, inner_left, bot_left], axis=1)
 454 
 455         top_mid = img[:, 0:1, 2:, :] + img[:, 1:2, :-2, :]
 456         mid_mid = img[:, :-2, 2:, :] + img[:, 2:, :-2, :]
 457         bot_mid = img[:, -2:-1, 2:, :] + img[:, -1:, :-2, :]
 458         xy_mid = K.concatenate([top_mid, mid_mid, bot_mid], axis=1)
 459 
 460         top_right = img[:, 0:1, -1:, :] + img[:, 1:2, -2:-1, :]
 461         inner_right = img[:, :-2, -1:, :] + img[:, 2:, -2:-1, :]
 462         bot_right = img[:, -2:-1, -1:, :] + img[:, -1:, -2:-1, :]
 463         xy_right = K.concatenate([top_right, inner_right, bot_right], axis=1)
 464 
 465         xy_out1 = K.concatenate([xy_left, xy_mid, xy_right], axis=2)
 466         xy_out2 = K.concatenate([xy_left, xy_mid, xy_right], axis=2)
 467         return (xy_out1 - xy_out2) * 0.25
 468 
 469     tv_weight = 1.0
 470     tv2_weight = 1.0
 471     loss = 0.0
 472     loss += tv_weight * (generalized_loss(_diff_x(y_true), _diff_x(y_pred), alpha=1.9999) +
 473                          generalized_loss(_diff_y(y_true), _diff_y(y_pred), alpha=1.9999))
 474     loss += tv2_weight * (generalized_loss(_diff_xx(y_true), _diff_xx(y_pred), alpha=1.9999) +
 475                           generalized_loss(_diff_yy(y_true), _diff_yy(y_pred), alpha=1.9999) +
 476                           generalized_loss(_diff_xy(y_true), _diff_xy(y_pred), alpha=1.9999) * 2.)
 477     loss = loss / (tv_weight + tv2_weight)
 478     # TODO simplify to use MSE instead
 479     return loss

 482 def scharr_edges(image, magnitude):
 483     """ Returns a tensor holding modified Scharr edge maps.
 484 
 485     Parameters
 486     ----------
 487     image: tensor
 488         Image tensor with shape [batch_size, h, w, d] and type float32. The image(s) must be 2x2
 489         or larger.
 490     magnitude: bool
 491         Boolean to determine if the edge magnitude or edge direction is returned
 492 
 493     Returns
 494     -------
 495     tensor
 496         Tensor holding edge maps for each channel. Returns a tensor with shape `[batch_size, h, w,
 497         d, 2]` where the last two dimensions hold `[[dy[0], dx[0]], [dy[1], dx[1]], ..., [dy[d-1],
 498         dx[d-1]]]` calculated using the Scharr filter.
 499     """
 500 
 501     # Define vertical and horizontal Scharr filters.
 502     static_image_shape = image.shape.dims if get_backend() == "amd" else image.get_shape()
 503     image_shape = K.shape(image)
 504 
 505     # 5x5 modified Scharr kernel ( reshape to (5,5,1,2) )
 506     matrix = np.array([[[[0.00070, 0.00070]],
 507                         [[0.00520, 0.00370]],
 508                         [[0.03700, 0.00000]],
 509                         [[0.00520, -0.0037]],
 510                         [[0.00070, -0.0007]]],
 511                        [[[0.00370, 0.00520]],
 512                         [[0.11870, 0.11870]],
 513                         [[0.25890, 0.00000]],
 514                         [[0.11870, -0.1187]],
 515                         [[0.00370, -0.0052]]],
 516                        [[[0.00000, 0.03700]],
 517                         [[0.00000, 0.25890]],
 518                         [[0.00000, 0.00000]],
 519                         [[0.00000, -0.2589]],
 520                         [[0.00000, -0.0370]]],
 521                        [[[-0.0037, 0.00520]],
 522                         [[-0.1187, 0.11870]],
 523                         [[-0.2589, 0.00000]],
 524                         [[-0.1187, -0.1187]],
 525                         [[-0.0037, -0.0052]]],
 526                        [[[-0.0007, 0.00070]],
 527                         [[-0.0052, 0.00370]],
 528                         [[-0.0370, 0.00000]],
 529                         [[-0.0052, -0.0037]],
 530                         [[-0.0007, -0.0007]]]])
 531     num_kernels = [2]
 532     kernels = K.constant(matrix, dtype='float32')
 533     kernels = K.tile(kernels, [1, 1, image_shape[-1], 1])
 534 
 535     # Use depth-wise convolution to calculate edge maps per channel.
 536     # Output tensor has shape [batch_size, h, w, d * num_kernels].
 537     pad_sizes = [[0, 0], [2, 2], [2, 2], [0, 0]]
 538     padded = pad(image, pad_sizes, mode='REFLECT')
 539     output = K.depthwise_conv2d(padded, kernels)
 540 
 541     if not magnitude:  # direction of edges
 542         # Reshape to [batch_size, h, w, d, num_kernels].
 543         shape = K.concatenate([image_shape, num_kernels], axis=0)
 544         output = K.reshape(output, shape=shape)
 545         output.set_shape(static_image_shape.concatenate(num_kernels))
 546         output = tf.atan(K.squeeze(output[:, :, :, :, 0] / output[:, :, :, :, 1], axis=None))
 547     # magnitude of edges -- unified x & y edges don't work well with Neural Networks
 548     return output

 551 def gmsd_loss(y_true, y_pred):
 552     """ Gradient Magnitude Similarity Deviation Loss.
 553 
 554     Improved image quality metric over MS-SSIM with easier calculations
 555 
 556     Parameters
 557     ----------
 558     y_true: tensor or variable
 559         The ground truth value
 560     y_pred: tensor or variable
 561         The predicted value
 562 
 563     Returns
 564     -------
 565     tensor
 566         The loss value
 567 
 568     References
 569     ----------
 570     http://www4.comp.polyu.edu.hk/~cslzhang/IQA/GMSD/GMSD.htm
 571     https://arxiv.org/ftp/arxiv/papers/1308/1308.3052.pdf
 572 
 573     """
 574     true_edge = scharr_edges(y_true, True)
 575     pred_edge = scharr_edges(y_pred, True)
 576     ephsilon = 0.0025
 577     upper = 2.0 * true_edge * pred_edge
 578     lower = K.square(true_edge) + K.square(pred_edge)
 579     gms = (upper + ephsilon) / (lower + ephsilon)
 580     gmsd = K.std(gms, axis=(1, 2, 3), keepdims=True)
 581     gmsd = K.squeeze(gmsd, axis=-1)
 582     return gmsd

 588 def gaussian_blur(radius=2.0):
 589     """ Apply gaussian blur to an input.
 590 
 591     Used for blurring mask in training.
 592 
 593     Parameters
 594     ----------
 595     radius: float, optional
 596         The kernel radius for applying gaussian blur. Default: `2.0`
 597 
 598     Returns
 599     -------
 600     tensor
 601         The input tensor with gaussian blurring applied
 602 
 603     References
 604     ----------
 605     https://github.com/iperov/DeepFaceLab
 606     """
 607     def _gaussian(var_x, radius, sigma):
 608         """ Obtain the gaussian kernel. """
 609         return np.exp(-(float(var_x) - float(radius)) ** 2 / (2 * sigma ** 2))
 610 
 611     def _make_kernel(sigma):
 612         """ Make the gaussian kernel. """
 613         kernel_size = max(3, int(2 * 2 * sigma + 1))
 614         mean = np.floor(0.5 * kernel_size)
 615         kernel_1d = np.array([_gaussian(x, mean, sigma) for x in range(kernel_size)])
 616         np_kernel = np.outer(kernel_1d, kernel_1d).astype(dtype=K.floatx())
 617         kernel = np_kernel / np.sum(np_kernel)
 618         return kernel
 619 
 620     gauss_kernel = _make_kernel(radius)
 621     gauss_kernel = gauss_kernel[:, :, np.newaxis, np.newaxis]
 622 
 623     def func(input_tensor):
 624         """ Apply gaussian blurring to the input tensor
 625 
 626         Parameters
 627         ----------
 628         input_tensor: tensor
 629             The input to have gaussian blurring applied.
 630 
 631         Returns
 632         -------
 633         tensor
 634             The input with gaussian blurring applied
 635         """
 636         inputs = [input_tensor[:, :, :, i:i + 1] for i in range(K.int_shape(input_tensor)[-1])]
 637         outputs = [K.conv2d(inp, K.constant(gauss_kernel), strides=(1, 1), padding="same")
 638                    for inp in inputs]
 639         return K.concatenate(outputs, axis=-1)
 640     return func
