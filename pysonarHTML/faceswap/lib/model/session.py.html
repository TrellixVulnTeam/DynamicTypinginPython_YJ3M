  17 class KSession():
  18     """ Handles the settings of backend sessions.
  19 
  20     This class acts as a wrapper for various :class:`keras.Model()` functions, ensuring that
  21     actions performed on a model are handled consistently within the correct graph.
  22 
  23     This is an early implementation of this class, and should be expanded out over time
  24     with relevant `AMD`, `CPU` and `NVIDIA` backend methods.
  25 
  26     Parameters
  27     ----------
  28     name: str
  29         The name of the model that is to be loaded
  30     model_path: str
  31         The path to the keras model file
  32     model_kwargs: dict, optional
  33         Any kwargs that need to be passed to :func:`keras.models.load_models()`. Default: None
  34     allow_growth: bool, optional
  35         Enable the Tensorflow GPU allow_growth configuration option. This option prevents "
  36         Tensorflow from allocating all of the GPU VRAM, but can lead to higher fragmentation and "
  37         slower performance. Default: False
  38     """
  39     def __init__(self, name, model_path, model_kwargs=None, allow_growth=False):
  40         logger.trace("Initializing: %s (name: %s, model_path: %s, model_kwargs: %s, "
  41                      "allow_growth: %s)",
  42                      self.__class__.__name__, name, model_path, model_kwargs, allow_growth)
  43         self._name = name
  44         self._session = self._set_session(allow_growth)
  45         self._model_path = model_path
  46         self._model_kwargs = model_kwargs
  47         self._model = None
  48         logger.trace("Initialized: %s", self.__class__.__name__,)
  49 
  50     def predict(self, feed, batch_size=None):
  51         """ Get predictions from the model in the correct session.
  52 
  53         This method is a wrapper for :func:`keras.predict()` function.
  54 
  55         Parameters
  56         ----------
  57         feed: numpy.ndarray or list
  58             The feed to be provided to the model as input. This should be a ``numpy.ndarray``
  59             for single inputs or a ``list`` of ``numpy.ndarrays`` for multiple inputs.
  60         """
  61         if self._session is None:
  62             if batch_size is None:
  63                 return self._model.predict(feed)
  64             return self._amd_predict_with_optimized_batchsizes(feed, batch_size)
  65 
  66         with self._session.as_default():  # pylint: disable=not-context-manager
  67             with self._session.graph.as_default():
  68                 return self._model.predict(feed, batch_size=batch_size)
  69 
  70     def _amd_predict_with_optimized_batchsizes(self, feed, batch_size):
  71         """ Minimizes the amount of kernels to be compiled when using
  72         the ``Amd`` backend with varying batchsizes while trying to keep
  73         the batchsize as high as possible.
  74 
  75         Parameters
  76         ----------
  77         feed: numpy.ndarray or list
  78             The feed to be provided to the model as input. This should be a ``numpy.ndarray``
  79             for single inputs or a ``list`` of ``numpy.ndarrays`` for multiple inputs.
  80         batch_size: int
  81             The upper batchsize to use.
  82         """
  83         if isinstance(feed, np.ndarray):
  84             feed = [feed]
  85         items = feed[0].shape[0]
  86         done_items = 0
  87         results = list()
  88         while done_items < items:
  89             if batch_size < 4:  # Not much difference in BS < 4
  90                 batch_size = 1
  91             batch_items = ((items - done_items) // batch_size) * batch_size
  92             if batch_items:
  93                 pred_data = [x[done_items:done_items + batch_items] for x in feed]
  94                 pred = self._model.predict(pred_data, batch_size=batch_size)
  95                 done_items += batch_items
  96                 results.append(pred)
  97             batch_size //= 2
  98         if isinstance(results[0], np.ndarray):
  99             return np.concatenate(results)
 100         return [np.concatenate(x) for x in zip(*results)]

 102     def _set_session(self, allow_growth):
 103         """ Sets the session and graph.
 104 
 105         If the backend is AMD then this does nothing and the global ``Keras`` ``Session``
 106         is used
 107         """
 108         if get_backend() == "amd":
 109             return None
 110 
 111         self.graph = tf.Graph()
 112         config = tf.ConfigProto()
 113         if allow_growth and get_backend() == "nvidia":
 114             config.gpu_options.allow_growth = True
 115         try:
 116             session = tf.Session(graph=tf.Graph(), config=config)
 117         except tf_error.InternalError as err:
 118             if "driver version is insufficient" in str(err):
 119                 msg = ("Your Nvidia Graphics Driver is insufficient for running Faceswap. "
 120                        "Please upgrade to the latest version.")
 121                 raise FaceswapError(msg) from err
 122             raise err
 123         logger.debug("Created tf.session: (graph: %s, session: %s, config: %s)",
 124                      session.graph, session, config)
 125         return session

 127     def load_model(self):
 128         """ Loads a model within the correct session.
 129 
 130         This method is a wrapper for :func:`keras.models.load_model()`. Loads a model and its
 131         weights from :attr:`model_path`. Any additional ``kwargs`` to be passed to
 132         :func:`keras.models.load_model()` should also be defined during initialization of the
 133         class.
 134         """
 135         logger.verbose("Initializing plugin model: %s", self._name)
 136         if self._session is None:
 137             self._model = k_load_model(self._model_path, **self._model_kwargs)
 138         else:
 139             with self._session.as_default():  # pylint: disable=not-context-manager
 140                 with self._session.graph.as_default():
 141                     self._model = k_load_model(self._model_path, **self._model_kwargs)
 142 
 143     def define_model(self, function):
 144         """ Defines a given model in the correct session.
 145 
 146         This method acts as a wrapper for :class:`keras.models.Model()` to ensure that the model
 147         is defined within it's own graph.
 148 
 149         Parameters
 150         ----------
 151         function: function
 152             A function that defines a :class:`keras.Model` and returns it's ``inputs`` and
 153             ``outputs``. The function that generates these results should be passed in, NOT the
 154             results themselves, as the function needs to be executed within the correct context.
 155         """
 156         if self._session is None:
 157             self._model = Model(*function())
 158         else:
 159             with self._session.as_default():  # pylint: disable=not-context-manager
 160                 with self._session.graph.as_default():
 161                     self._model = Model(*function())
 162 
 163     def load_model_weights(self):
 164         """ Load model weights for a defined model inside the correct session.
 165 
 166         This method is a wrapper for :class:`keras.load_weights()`. Once a model has been defined
 167         in :func:`define_model()` this method can be called to load its weights in the correct
 168         graph from the :attr:`model_path` defined during initialization of this class.
 169         """
 170         logger.verbose("Initializing plugin model: %s", self._name)
 171         if self._session is None:
 172             self._model.load_weights(self._model_path)
 173         else:
 174             with self._session.as_default():  # pylint: disable=not-context-manager
 175                 with self._session.graph.as_default():
 176                     self._model.load_weights(self._model_path)

 178     def append_softmax_activation(self, layer_index=-1):
 179         """ Append a softmax activation layer to a model
 180 
 181         Occasionally a softmax activation layer needs to be added to a model's output.
 182         This is a convenience fuction to append this layer to the loaded model.
 183 
 184         Parameters
 185         ----------
 186         layer_index: int, optional
 187             The layer index of the model to select the output from to use as an input to the
 188             softmax activation layer. Default: -1 (The final layer of the model)
 189         """
 190         logger.debug("Appending Softmax Activation to model: (layer_index: %s)", layer_index)
 191         softmax = Activation("softmax", name="softmax")(self._model.layers[layer_index].output)
 192         self._model = Model(inputs=self._model.input, outputs=[softmax])
