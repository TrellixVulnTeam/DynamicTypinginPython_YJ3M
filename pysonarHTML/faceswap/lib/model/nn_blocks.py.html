  18 class NNBlocks():
  19     """ Blocks that are often used for multiple models are stored here for easy access.
  20 
  21     This class is always brought in as ``self.blocks`` in all model plugins so that all models
  22     have access to them.
  23 
  24     The parameters passed into this class should ultimately originate from the user's training
  25     configuration file, rather than being hard-coded at the plugin level.
  26 
  27     Parameters
  28     ----------
  29     use_icnr_init: bool, Optional
  30         ``True`` if ICNR initialization should be used rather than the default. Default: ``False``
  31     use_convaware_init: bool, Optional
  32         ``True`` if Convolutional Aware initialization should be used rather than the default.
  33         Default: ``False``
  34     use_reflect_padding: bool, Optional
  35         ``True`` if Reflect Padding initialization should be used rather than the padding.
  36         Default: ``False``
  37     first_run: bool, Optional
  38         ``True`` if a model is being created for the first time, ``False`` if a model is being
  39         resumed. Used to prevent Convolutional Aware weights from being calculated when a model
  40         is being reloaded. Default: ``True``
  41     """
  42     def __init__(self,
  43                  use_icnr_init=False,
  44                  use_convaware_init=False,
  45                  use_reflect_padding=False,
  46                  first_run=True):
  47         logger.debug("Initializing %s: (use_icnr_init: %s, use_convaware_init: %s, "
  48                      "use_reflect_padding: %s, first_run: %s)",
  49                      self.__class__.__name__, use_icnr_init, use_convaware_init,
  50                      use_reflect_padding, first_run)
  51         self.names = dict()
  52         self.first_run = first_run
  53         self.use_icnr_init = use_icnr_init
  54         self.use_convaware_init = use_convaware_init
  55         self.use_reflect_padding = use_reflect_padding
  56         if self.use_convaware_init and self.first_run:
  57             logger.info("Using Convolutional Aware Initialization. Model generation will take a "
  58                         "few minutes...")
  59         logger.debug("Initialized %s", self.__class__.__name__)

  61     def _get_name(self, name):
  62         """ Return unique layer name for requested block.
  63 
  64         As blocks can be used multiple times, auto appends an integer to the end of the requested
  65         name to keep all block names unique
  66 
  67         Parameters
  68         ----------
  69         name: str
  70             The requested name for the layer
  71 
  72         Returns
  73         -------
  74         str
  75             The unique name for this layer
  76         """
  77         self.names[name] = self.names.setdefault(name, -1) + 1
  78         name = "{}_{}".format(name, self.names[name])
  79         logger.debug("Generating block name: %s", name)
  80         return name

  82     def _set_default_initializer(self, kwargs):
  83         """ Sets the default initializer for convolution 2D and Seperable convolution 2D layers
  84             to Convolutional Aware or he_uniform.
  85 
  86             if a specific initializer has been passed in from the model plugin, then the specified
  87             initializer will be used rather than the default.
  88 
  89             Parameters
  90             ----------
  91             kwargs: dict
  92                 The keyword arguments for the current layer
  93 
  94             Returns
  95             -------
  96             dict
  97                 The keyword arguments for the current layer with the initializer updated to
  98                 the select default value
  99             """
 100         if "kernel_initializer" in kwargs:
 101             logger.debug("Using model specified initializer: %s", kwargs["kernel_initializer"])
 102             return kwargs
 103         if self.use_convaware_init:
 104             default = ConvolutionAware()
 105             if self.first_run:
 106                 # Indicate the Convolutional Aware should be calculated on first run
 107                 default._init = True  # pylint:disable=protected-access
 108         else:
 109             default = he_uniform()
 110         if kwargs.get("kernel_initializer", None) != default:
 111             kwargs["kernel_initializer"] = default
 112             logger.debug("Set default kernel_initializer to: %s", kwargs["kernel_initializer"])
 113         return kwargs

 115     @staticmethod
 116     def _switch_kernel_initializer(kwargs, initializer):
 117         """ Switch the initializer in the given kwargs to the given initializer and return the
 118         previous initializer to caller.
 119 
 120         For residual blocks and up-scaling, user selected initializer methods should replace those
 121         set by the model. This method updates the initializer for the layer, and returns the
 122         original initializer so that it can be set back to the layer's key word arguments for
 123         subsequent layers where the initializer should not be switched.
 124 
 125         Parameters
 126         ----------
 127         kwargs: dict
 128             The keyword arguments for the current layer
 129         initializer: keras or faceswap initializer class
 130             The initializer that should replace the current initializer that exists in keyword
 131             arguments
 132 
 133         Returns
 134         -------
 135         keras or faceswap initializer class
 136             The original initializer that existed in the given keyword arguments
 137         """
 138         original = kwargs.get("kernel_initializer", None)
 139         kwargs["kernel_initializer"] = initializer
 140         logger.debug("Switched kernel_initializer from %s to %s", original, initializer)
 141         return original

 143     def conv2d(self, input_tensor, filters, kernel_size, strides=(1, 1), padding="same", **kwargs):
 144         """ A standard Convolution 2D layer with correct initialization.
 145 
 146         This layer creates a convolution kernel that is convolved with the layer input to produce
 147         a tensor of outputs.
 148 
 149         Parameters
 150         ----------
 151         input_tensor: tensor
 152             The input tensor to the layer
 153         filters: int
 154             The dimensionality of the output space (i.e. the number of output filters in the
 155             convolution)
 156         kernel_size: int
 157             An integer or tuple/list of 2 integers, specifying the height and width of the 2D
 158             convolution window. Can be a single integer to specify the same value for all spatial
 159             dimensions
 160         strides: tuple, optional
 161             An integer or tuple/list of 2 integers, specifying the strides of the convolution along
 162             the height and width. Can be a single integer to specify the same value for all spatial
 163             dimensions. Default: `(1, 1)`
 164         padding: ["valid", "same"], optional
 165             The padding to use. Default: `"same"`
 166         kwargs: dict
 167             Any additional Keras standard layer keyword arguments
 168 
 169         Returns
 170         -------
 171         tensor
 172             The output tensor from the Convolution 2D Layer
 173         """
 174         logger.debug("input_tensor: %s, filters: %s, kernel_size: %s, strides: %s, padding: %s, "
 175                      "kwargs: %s)", input_tensor, filters, kernel_size, strides, padding, kwargs)
 176         if kwargs.get("name", None) is None:
 177             kwargs["name"] = self._get_name("conv2d_{}".format(input_tensor.shape[1]))
 178         kwargs = self._set_default_initializer(kwargs)
 179         var_x = Conv2D(filters, kernel_size,
 180                        strides=strides,
 181                        padding=padding,
 182                        **kwargs)(input_tensor)
 183         return var_x

 186     def conv(self, input_tensor, filters, kernel_size=5, strides=2, padding="same",
 187              use_instance_norm=False, res_block_follows=False, **kwargs):
 188         """ A standard Convolution 2D layer which applies user specified configuration to the
 189         layer.
 190 
 191         Adds reflection padding if it has been selected by the user, and other post-processing
 192         if requested by the plugin.
 193 
 194         Parameters
 195         ----------
 196         input_tensor: tensor
 197             The input tensor to the layer
 198         filters: int
 199             The dimensionality of the output space (i.e. the number of output filters in the
 200             convolution)
 201         kernel_size: int, optional
 202             An integer or tuple/list of 2 integers, specifying the height and width of the 2D
 203             convolution window. Can be a single integer to specify the same value for all spatial
 204             dimensions. Default: 5
 205         strides: tuple or int, optional
 206             An integer or tuple/list of 2 integers, specifying the strides of the convolution along
 207             the height and width. Can be a single integer to specify the same value for all spatial
 208             dimensions. Default: `2`
 209         padding: ["valid", "same"], optional
 210             The padding to use. Default: `"same"`
 211         use_instance_norm: bool, optional
 212             ``True`` if instance normalization should be applied after the convolutional layer.
 213             Default: ``False``
 214         res_block_follows: bool, optional
 215             If a residual block will follow this layer, then this should be set to `True` to add
 216             a leaky ReLu after the convolutional layer. Default: ``False``
 217         kwargs: dict
 218             Any additional Keras standard layer keyword arguments
 219 
 220         Returns
 221         -------
 222         tensor
 223             The output tensor from the Convolution 2D Layer
 224         """
 225         logger.debug("input_tensor: %s, filters: %s, kernel_size: %s, strides: %s, "
 226                      "use_instance_norm: %s, kwargs: %s)", input_tensor, filters, kernel_size,
 227                      strides, use_instance_norm, kwargs)
 228         name = self._get_name("conv_{}".format(input_tensor.shape[1]))
 229         if self.use_reflect_padding:
 230             input_tensor = ReflectionPadding2D(
 231                 stride=strides,
 232                 kernel_size=kernel_size,
 233                 name="{}_reflectionpadding2d".format(name))(input_tensor)
 234             padding = "valid"
 235         var_x = self.conv2d(input_tensor, filters,
 236                             kernel_size=kernel_size,
 237                             strides=strides,
 238                             padding=padding,
 239                             name="{}_conv2d".format(name),
 240                             **kwargs)
 241         if use_instance_norm:
 242             var_x = InstanceNormalization(name="{}_instancenorm".format(name))(var_x)
 243         if not res_block_follows:
 244             var_x = LeakyReLU(0.1, name="{}_leakyrelu".format(name))(var_x)
 245         return var_x

 247     def upscale(self, input_tensor, filters, kernel_size=3, padding="same",
 248                 use_instance_norm=False, res_block_follows=False, scale_factor=2, **kwargs):
 249         """ An upscale layer for sub-pixel up-scaling.
 250 
 251         Adds reflection padding if it has been selected by the user, and other post-processing
 252         if requested by the plugin.
 253 
 254         Parameters
 255         ----------
 256         input_tensor: tensor
 257             The input tensor to the layer
 258         filters: int
 259             The dimensionality of the output space (i.e. the number of output filters in the
 260             convolution)
 261         kernel_size: int, optional
 262             An integer or tuple/list of 2 integers, specifying the height and width of the 2D
 263             convolution window. Can be a single integer to specify the same value for all spatial
 264             dimensions. Default: 3
 265         padding: ["valid", "same"], optional
 266             The padding to use. Default: `"same"`
 267         use_instance_norm: bool, optional
 268             ``True`` if instance normalization should be applied after the convolutional layer.
 269             Default: ``False``
 270         res_block_follows: bool, optional
 271             If a residual block will follow this layer, then this should be set to `True` to add
 272             a leaky ReLu after the convolutional layer. Default: ``False``
 273         scale_factor: int, optional
 274             The amount to upscale the image. Default: `2`
 275         kwargs: dict
 276             Any additional Keras standard layer keyword arguments
 277 
 278         Returns
 279         -------
 280         tensor
 281             The output tensor from the Upscale layer
 282         """
 283         logger.debug("input_tensor: %s, filters: %s, kernel_size: %s, use_instance_norm: %s, "
 284                      "kwargs: %s)", input_tensor, filters, kernel_size, use_instance_norm, kwargs)
 285         name = self._get_name("upscale_{}".format(input_tensor.shape[1]))
 286         if self.use_reflect_padding:
 287             input_tensor = ReflectionPadding2D(
 288                 stride=1,
 289                 kernel_size=kernel_size,
 290                 name="{}_reflectionpadding2d".format(name))(input_tensor)
 291             padding = "valid"
 292         kwargs = self._set_default_initializer(kwargs)
 293         if self.use_icnr_init:
 294             original_init = self._switch_kernel_initializer(
 295                 kwargs,
 296                 ICNR(initializer=kwargs["kernel_initializer"]))
 297         var_x = self.conv2d(input_tensor, filters * scale_factor * scale_factor,
 298                             kernel_size=kernel_size,
 299                             padding=padding,
 300                             name="{}_conv2d".format(name),
 301                             **kwargs)
 302         if self.use_icnr_init:
 303             self._switch_kernel_initializer(kwargs, original_init)
 304         if use_instance_norm:
 305             var_x = InstanceNormalization(name="{}_instancenorm".format(name))(var_x)
 306         if not res_block_follows:
 307             var_x = LeakyReLU(0.1, name="{}_leakyrelu".format(name))(var_x)
 308         var_x = PixelShuffler(name="{}_pixelshuffler".format(name), size=scale_factor)(var_x)
 309         return var_x

 312     def res_block(self, input_tensor, filters, kernel_size=3, padding="same", **kwargs):
 313         """ Residual block.
 314 
 315         Parameters
 316         ----------
 317         input_tensor: tensor
 318             The input tensor to the layer
 319         filters: int
 320             The dimensionality of the output space (i.e. the number of output filters in the
 321             convolution)
 322         kernel_size: int, optional
 323             An integer or tuple/list of 2 integers, specifying the height and width of the 2D
 324             convolution window. Can be a single integer to specify the same value for all spatial
 325             dimensions. Default: 3
 326         padding: ["valid", "same"], optional
 327             The padding to use. Default: `"same"`
 328         kwargs: dict
 329             Any additional Keras standard layer keyword arguments
 330 
 331         Returns
 332         -------
 333         tensor
 334             The output tensor from the Upscale layer
 335         """
 336         logger.debug("input_tensor: %s, filters: %s, kernel_size: %s, kwargs: %s)",
 337                      input_tensor, filters, kernel_size, kwargs)
 338         name = self._get_name("residual_{}".format(input_tensor.shape[1]))
 339         var_x = LeakyReLU(alpha=0.2, name="{}_leakyrelu_0".format(name))(input_tensor)
 340         if self.use_reflect_padding:
 341             var_x = ReflectionPadding2D(stride=1,
 342                                         kernel_size=kernel_size,
 343                                         name="{}_reflectionpadding2d_0".format(name))(var_x)
 344             padding = "valid"
 345         var_x = self.conv2d(var_x, filters,
 346                             kernel_size=kernel_size,
 347                             padding=padding,
 348                             name="{}_conv2d_0".format(name),
 349                             **kwargs)
 350         var_x = LeakyReLU(alpha=0.2, name="{}_leakyrelu_1".format(name))(var_x)
 351         if self.use_reflect_padding:
 352             var_x = ReflectionPadding2D(stride=1,
 353                                         kernel_size=kernel_size,
 354                                         name="{}_reflectionpadding2d_1".format(name))(var_x)
 355             padding = "valid"
 356         if not self.use_convaware_init:
 357             original_init = self._switch_kernel_initializer(kwargs, VarianceScaling(
 358                 scale=0.2,
 359                 mode="fan_in",
 360                 distribution="uniform"))
 361         var_x = self.conv2d(var_x, filters,
 362                             kernel_size=kernel_size,
 363                             padding=padding,
 364                             **kwargs)
 365         if not self.use_convaware_init:
 366             self._switch_kernel_initializer(kwargs, original_init)
 367         var_x = Add()([var_x, input_tensor])
 368         var_x = LeakyReLU(alpha=0.2, name="{}_leakyrelu_3".format(name))(var_x)
 369         return var_x
