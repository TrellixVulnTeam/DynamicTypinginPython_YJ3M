  14 class Adam(KerasAdam):
  15     """Adapted Keras Adam Optimizer to allow support of calculations on CPU for Tensorflow.
  16 
  17     Default parameters follow those provided in the original paper. Adapted from
  18     https://github.com/iperov/DeepFaceLab
  19 
  20     Parameters
  21     ----------
  22     lr: float, optional
  23         >= `0`. Learning rate. Default: `0.001`
  24     beta_1: float, optional
  25         `0` < beta < `1` Generally close to `1`. Default: `0.9`
  26     beta_2: float, optional
  27         `0` < beta < `1`. Generally close to `1`. Default: `0.999`
  28     epsilon: float, optional
  29         >= `0`. Fuzz factor. If ``None``, defaults to `K.epsilon()`. Default: ``None``
  30     decay: float, optional
  31         >= 0. Learning rate decay over each update. Default: `0`
  32     amsgrad: bool, optional
  33         ``True`` to apply the AMSGrad variant of this algorithm from the paper "On the Convergence
  34         of Adam and Beyond" otherwise ``False``. Default: ``False``
  35     cpu_mode: bool, optional
  36         Set to ``True`` to perform some of the calculations on CPU for Nvidia backends, otherwise
  37         ``False``. Default: ``False``
  38     kwargs: dict
  39         Any additional standard Keras optimizer keyword arguments
  40 
  41     References
  42     ----------
  43     - Adam - A Method for Stochastic Optimization - https://arxiv.org/abs/1412.6980v8
  44 
  45     - On the Convergence of Adam and Beyond - https://openreview.net/forum?id=ryQu7f-RZ
  46     """
  47 
  48     def __init__(self,
  49                  lr=0.001,
  50                  beta_1=0.9,
  51                  beta_2=0.999,
  52                  epsilon=None,
  53                  decay=0.,
  54                  amsgrad=False,
  55                  cpu_mode=False,
  56                  **kwargs):
  57         super().__init__(lr, beta_1, beta_2, epsilon, decay, **kwargs)
  58         self.cpu_mode = self._set_cpu_mode(cpu_mode)
  59 
  60     @staticmethod
  61     def _set_cpu_mode(cpu_mode):
  62         """ Sets the CPU mode to False if not using Tensorflow, otherwise the given value.
  63 
  64         Parameters
  65         ----------
  66         cpu_mode: bool
  67             Set to ``True`` to perform some of the calculations on CPU for Nvidia backends,
  68             otherwise ``False``.
  69 
  70         Returns
  71         -------
  72         bool
  73             ``True`` if some calculations should be performed on CPU otherwise ``False``
  74         """
  75         retval = False if K.backend() != "tensorflow" else cpu_mode
  76         logger.debug("Optimizer CPU Mode set to %s", retval)
  77         return retval
  78 
  79     def get_updates(self, loss, params):
  80         """ Obtain the optimizer loss updates.
  81 
  82         Parameters
  83         ----------
  84         loss: list
  85             List of tensors
  86 
  87         params: list
  88             List of tensors
  89 
  90         Returns
  91         -------
  92         list
  93             List of tensors
  94         """
  95         grads = self.get_gradients(loss, params)
  96         self.updates = [K.update_add(self.iterations, 1)]
  97 
  98         lr = self.lr
  99         if self.initial_decay > 0:
 100             lr = lr * (1. / (1. + self.decay * K.cast(self.iterations,
 101                                                       K.dtype(self.decay))))
 102 
 103         t = K.cast(self.iterations, K.floatx()) + 1
 104         lr_t = lr * (K.sqrt(1. - K.pow(self.beta_2, t)) /
 105                      (1. - K.pow(self.beta_1, t)))
 106 
 107         # Pass off to CPU if requested
 108         if self.cpu_mode:
 109             with K.tf.device("/cpu:0"):
 110                 ms, vs, vhats = self._update_1(params)
 111         else:
 112             ms, vs, vhats = self._update_1(params)
 113 
 114         self.weights = [self.iterations] + ms + vs + vhats
 115 
 116         for p, g, m, v, vhat in zip(params, grads, ms, vs, vhats):
 117             m_t = (self.beta_1 * m) + (1. - self.beta_1) * g
 118             v_t = (self.beta_2 * v) + (1. - self.beta_2) * K.square(g)
 119             if self.amsgrad:
 120                 vhat_t = K.maximum(vhat, v_t)
 121                 p_t = p - lr_t * m_t / (K.sqrt(vhat_t) + self.epsilon)
 122                 self.updates.append(K.update(vhat, vhat_t))
 123             else:
 124                 p_t = p - lr_t * m_t / (K.sqrt(v_t) + self.epsilon)
 125 
 126             self.updates.append(K.update(m, m_t))
 127             self.updates.append(K.update(v, v_t))
 128             new_p = p_t
 129 
 130             # Apply constraints.
 131             if getattr(p, 'constraint', None) is not None:
 132                 new_p = p.constraint(new_p)
 133 
 134             self.updates.append(K.update(p, new_p))
 135         return self.updates
 136 
 137     def _update_1(self, params):
 138         """ Perform the first update. Run under CPU context if running on Tensorflow and CPU mode
 139         is enabled, otherwise run on the default device. """
 140         ms = [K.zeros(K.int_shape(p), dtype=K.dtype(p)) for p in params]
 141         vs = [K.zeros(K.int_shape(p), dtype=K.dtype(p)) for p in params]
 142         if self.amsgrad:
 143             vhats = [K.zeros(K.int_shape(p), dtype=K.dtype(p)) for p in params]
 144         else:
 145             vhats = [K.zeros(1) for _ in params]
 146         return ms, vs, vhats
