  16 class Backup():
  17     """ Holds information about model location and functions for backing up
  18         Restoring and Snapshotting models """
  19     def __init__(self, model_dir, model_name):
  20         logger.debug("Initializing %s: (model_dir: '%s', model_name: '%s')",
  21                      self.__class__.__name__, model_dir, model_name)
  22         self.model_dir = str(model_dir)
  23         self.model_name = model_name
  24         logger.debug("Initialized %s", self.__class__.__name__)
  25 
  26     def check_valid(self, filename, for_restore=False):
  27         """ Check if the passed in filename is valid for a backup operation """
  28         fullpath = os.path.join(self.model_dir, filename)
  29         if not filename.startswith(self.model_name):
  30             # Any filename that does not start with the model name are invalid
  31             # for all operations
  32             retval = False
  33         elif for_restore and filename.endswith(".bk"):
  34             # Only filenames ending in .bk are valid for restoring
  35             retval = True
  36         elif not for_restore and ((os.path.isfile(fullpath) and not filename.endswith(".bk")) or
  37                                   (os.path.isdir(fullpath) and
  38                                    filename == "{}_logs".format(self.model_name))):
  39             # Only filenames that do not end with .bk or folders that are the logs folder
  40             # are valid for backup
  41             retval = True
  42         else:
  43             retval = False
  44         logger.debug("'%s' valid for backup operation: %s", filename, retval)
  45         return retval

  47     @staticmethod
  48     def backup_model(fullpath):
  49         """ Backup Model File
  50             Fullpath should be the path to an h5.py file or a state.json file """
  51         backupfile = fullpath + ".bk"
  52         logger.verbose("Backing up: '%s' to '%s'", fullpath, backupfile)
  53         if os.path.exists(backupfile):
  54             os.remove(backupfile)
  55         if os.path.exists(fullpath):
  56             os.rename(fullpath, backupfile)
  57 
  58     def snapshot_models(self, iterations):
  59         """ Take a snapshot of the model at current state and back up """
  60         logger.info("Saving snapshot")
  61         snapshot_dir = "{}_snapshot_{}_iters".format(self.model_dir, iterations)
  62 
  63         if os.path.isdir(snapshot_dir):
  64             logger.debug("Removing previously existing snapshot folder: '%s'", snapshot_dir)
  65             rmtree(snapshot_dir)
  66 
  67         dst = str(get_folder(snapshot_dir))
  68         for filename in os.listdir(self.model_dir):
  69             if not self.check_valid(filename, for_restore=False):
  70                 logger.debug("Not snapshotting file: '%s'", filename)
  71                 continue
  72             srcfile = os.path.join(self.model_dir, filename)
  73             dstfile = os.path.join(dst, filename)
  74             copyfunc = copytree if os.path.isdir(srcfile) else copyfile
  75             logger.debug("Saving snapshot: '%s' > '%s'", srcfile, dstfile)
  76             copyfunc(srcfile, dstfile)
  77         logger.info("Saved snapshot")

  79     def restore(self):
  80         """ Restores a model from backup.
  81             This will place all existing models/logs into a folder named:
  82                 - "<model_name>_archived_<timestamp>"
  83             Copy all .bk files to replace original files
  84             Remove logs from after the restore session_id from the logs folder """
  85         archive_dir = self.move_archived()
  86         self.restore_files()
  87         self.restore_logs(archive_dir)
  88 
  89     def move_archived(self):
  90         """ Move archived files to archived folder and return archived folder name """
  91         logger.info("Archiving existing model files...")
  92         now = datetime.now().strftime("%Y%m%d_%H%M%S")
  93         archive_dir = os.path.join(self.model_dir, "{}_archived_{}".format(self.model_name, now))
  94         os.mkdir(archive_dir)
  95         for filename in os.listdir(self.model_dir):
  96             if not self.check_valid(filename, for_restore=False):
  97                 logger.debug("Not moving file to archived: '%s'", filename)
  98                 continue
  99             logger.verbose("Moving '%s' to archived model folder: '%s'", filename, archive_dir)
 100             src = os.path.join(self.model_dir, filename)
 101             dst = os.path.join(archive_dir, filename)
 102             os.rename(src, dst)
 103         logger.verbose("Archived existing model files")
 104         return archive_dir

 106     def restore_files(self):
 107         """ Restore files from .bk """
 108         logger.info("Restoring models from backup...")
 109         for filename in os.listdir(self.model_dir):
 110             if not self.check_valid(filename, for_restore=True):
 111                 logger.debug("Not restoring file: '%s'", filename)
 112                 continue
 113             dstfile = os.path.splitext(filename)[0]
 114             logger.verbose("Restoring '%s' to '%s'", filename, dstfile)
 115             src = os.path.join(self.model_dir, filename)
 116             dst = os.path.join(self.model_dir, dstfile)
 117             copyfile(src, dst)
 118         logger.verbose("Restored models from backup")
 119 
 120     def restore_logs(self, archive_dir):
 121         """ Restore the log files since before archive """
 122         logger.info("Restoring Logs...")
 123         session_names = self.get_session_names()
 124         log_dirs = self.get_log_dirs(archive_dir, session_names)
 125         for log_dir in log_dirs:
 126             src = os.path.join(archive_dir, log_dir)
 127             dst = os.path.join(self.model_dir, log_dir)
 128             logger.verbose("Restoring logfile: %s", dst)
 129             copytree(src, dst)
 130         logger.verbose("Restored Logs")
 131 
 132     def get_session_names(self):
 133         """ Get the existing session names from state file """
 134         serializer = get_serializer("json")
 135         state_file = os.path.join(self.model_dir,
 136                                   "{}_state.{}".format(self.model_name, serializer.file_extension))
 137         state = serializer.load(state_file)
 138         session_names = ["session_{}".format(key)
 139                          for key in state["sessions"].keys()]
 140         logger.debug("Session to restore: %s", session_names)
 141         return session_names
 142 
 143     def get_log_dirs(self, archive_dir, session_names):
 144         """ Get the session logdir paths in the archive folder """
 145         archive_logs = os.path.join(archive_dir, "{}_logs".format(self.model_name))
 146         paths = [os.path.join(dirpath.replace(archive_dir, "")[1:], folder)
 147                  for dirpath, dirnames, _ in os.walk(archive_logs)
 148                  for folder in dirnames
 149                  if folder in session_names]
 150         logger.debug("log folders to restore: %s", paths)
 151         return paths
