  25 class PixelShuffler(Layer):
  26     """ PixelShuffler layer for Keras.
  27 
  28     This layer requires a Convolution2D prior to it, having output filters computed according to
  29     the formula :math:`filters = k * (scale_factor * scale_factor)` where `k` is a user defined
  30     number of filters (generally larger than 32) and `scale_factor` is the up-scaling factor
  31     (generally 2).
  32 
  33     This layer performs the depth to space operation on the convolution filters, and returns a
  34     tensor with the size as defined below.
  35 
  36     Notes
  37     -----
  38     In practice, it is useful to have a second convolution layer after the
  39     :class:`PixelShuffler` layer to speed up the learning process. However, if you are stacking
  40     multiple :class:`PixelShuffler` blocks, it may increase the number of parameters greatly,
  41     so the Convolution layer after :class:`PixelShuffler` layer can be removed.
  42 
  43     Example
  44     -------
  45     >>> # A standard sub-pixel up-scaling block
  46     >>> x = Convolution2D(256, 3, 3, padding="same", activation="relu")(...)
  47     >>> u = PixelShuffler(size=(2, 2))(x)
  48     [Optional]
  49     >>> x = Convolution2D(256, 3, 3, padding="same", activation="relu")(u)
  50 
  51     Parameters
  52     ----------
  53     size: tuple, optional
  54         The (`h`, `w`) scaling factor for up-scaling. Default: `(2, 2)`
  55     data_format: ["channels_first", "channels_last", ``None``], optional
  56         The data format for the input. Default: ``None``
  57     kwargs: dict
  58         The standard Keras Layer keyword arguments (if any)
  59 
  60     References
  61     ----------
  62     https://gist.github.com/t-ae/6e1016cc188104d123676ccef3264981
  63     """
  64     def __init__(self, size=(2, 2), data_format=None, **kwargs):
  65         super().__init__(**kwargs)
  66         self.data_format = K.normalize_data_format(data_format)
  67         self.size = conv_utils.normalize_tuple(size, 2, "size")
  68 
  69     def call(self, inputs, **kwargs):
  70         """This is where the layer's logic lives.
  71 
  72         Parameters
  73         ----------
  74         inputs: tensor
  75             Input tensor, or list/tuple of input tensors
  76         kwargs: dict
  77             Additional keyword arguments
  78 
  79         Returns
  80         -------
  81         tensor
  82             A tensor or list/tuple of tensors
  83         """
  84         input_shape = K.int_shape(inputs)
  85         if len(input_shape) != 4:
  86             raise ValueError('Inputs should have rank ' +
  87                              str(4) +
  88                              '; Received input shape:', str(input_shape))
  89 
  90         if self.data_format == 'channels_first':
  91             batch_size, channels, height, width = input_shape
  92             if batch_size is None:
  93                 batch_size = -1
  94             r_height, r_width = self.size
  95             o_height, o_width = height * r_height, width * r_width
  96             o_channels = channels // (r_height * r_width)
  97 
  98             out = K.reshape(inputs, (batch_size, r_height, r_width, o_channels, height, width))
  99             out = K.permute_dimensions(out, (0, 3, 4, 1, 5, 2))
 100             out = K.reshape(out, (batch_size, o_channels, o_height, o_width))
 101         elif self.data_format == 'channels_last':
 102             batch_size, height, width, channels = input_shape
 103             if batch_size is None:
 104                 batch_size = -1
 105             r_height, r_width = self.size
 106             o_height, o_width = height * r_height, width * r_width
 107             o_channels = channels // (r_height * r_width)
 108 
 109             out = K.reshape(inputs, (batch_size, height, width, r_height, r_width, o_channels))
 110             out = K.permute_dimensions(out, (0, 1, 3, 2, 4, 5))
 111             out = K.reshape(out, (batch_size, o_height, o_width, o_channels))
 112         return out

 114     def compute_output_shape(self, input_shape):
 115         """Computes the output shape of the layer.
 116 
 117         Assumes that the layer will be built to match that input shape provided.
 118 
 119         Parameters
 120         ----------
 121         input_shape: tuple or list of tuples
 122             Shape tuple (tuple of integers) or list of shape tuples (one per output tensor of the
 123             layer).  Shape tuples can include None for free dimensions, instead of an integer.
 124 
 125         Returns
 126         -------
 127         tuple
 128             An input shape tuple
 129         """
 130         if len(input_shape) != 4:
 131             raise ValueError('Inputs should have rank ' +
 132                              str(4) +
 133                              '; Received input shape:', str(input_shape))
 134 
 135         if self.data_format == 'channels_first':
 136             height = None
 137             width = None
 138             if input_shape[2] is not None:
 139                 height = input_shape[2] * self.size[0]
 140             if input_shape[3] is not None:
 141                 width = input_shape[3] * self.size[1]
 142             channels = input_shape[1] // self.size[0] // self.size[1]
 143 
 144             if channels * self.size[0] * self.size[1] != input_shape[1]:
 145                 raise ValueError('channels of input and size are incompatible')
 146 
 147             retval = (input_shape[0],
 148                       channels,
 149                       height,
 150                       width)
 151         elif self.data_format == 'channels_last':
 152             height = None
 153             width = None
 154             if input_shape[1] is not None:
 155                 height = input_shape[1] * self.size[0]
 156             if input_shape[2] is not None:
 157                 width = input_shape[2] * self.size[1]
 158             channels = input_shape[3] // self.size[0] // self.size[1]
 159 
 160             if channels * self.size[0] * self.size[1] != input_shape[3]:
 161                 raise ValueError('channels of input and size are incompatible')
 162 
 163             retval = (input_shape[0],
 164                       height,
 165                       width,
 166                       channels)
 167         return retval

 191 class SubPixelUpscaling(Layer):
 192     """ Sub-pixel convolutional up-scaling layer.
 193 
 194     This layer requires a Convolution2D prior to it, having output filters computed according to
 195     the formula :math:`filters = k * (scale_factor * scale_factor)` where `k` is a user defined
 196     number of filters (generally larger than 32) and `scale_factor` is the up-scaling factor
 197     (generally 2).
 198 
 199     This layer performs the depth to space operation on the convolution filters, and returns a
 200     tensor with the size as defined below.
 201 
 202     Notes
 203     -----
 204     This method is deprecated as it just performs the same as :class:`PixelShuffler`
 205     using explicit Tensorflow ops. The method is kept in the repository to support legacy
 206     models that have been created with this layer.
 207 
 208     In practice, it is useful to have a second convolution layer after the
 209     :class:`SubPixelUpscaling` layer to speed up the learning process. However, if you are stacking
 210     multiple :class:`SubPixelUpscaling` blocks, it may increase the number of parameters greatly,
 211     so the Convolution layer after :class:`SubPixelUpscaling` layer can be removed.
 212 
 213     Example
 214     -------
 215     >>> # A standard sub-pixel up-scaling block
 216     >>> x = Convolution2D(256, 3, 3, padding="same", activation="relu")(...)
 217     >>> u = SubPixelUpscaling(scale_factor=2)(x)
 218     [Optional]
 219     >>> x = Convolution2D(256, 3, 3, padding="same", activation="relu")(u)
 220 
 221     Parameters
 222     ----------
 223     size: int, optional
 224         The up-scaling factor. Default: `2`
 225     data_format: ["channels_first", "channels_last", ``None``], optional
 226         The data format for the input. Default: ``None``
 227     kwargs: dict
 228         The standard Keras Layer keyword arguments (if any)
 229 
 230     References
 231     ----------
 232     based on the paper "Real-Time Single Image and Video Super-Resolution Using an Efficient
 233     Sub-Pixel Convolutional Neural Network" (https://arxiv.org/abs/1609.05158).
 234     """
 235 
 236     def __init__(self, scale_factor=2, data_format=None, **kwargs):
 237         super(SubPixelUpscaling, self).__init__(**kwargs)
 238 
 239         self.scale_factor = scale_factor
 240         self.data_format = K.normalize_data_format(data_format)
 241 
 242     def build(self, input_shape):
 243         """Creates the layer weights.
 244 
 245         Must be implemented on all layers that have weights.
 246 
 247         Parameters
 248         ----------
 249         input_shape: tensor
 250             Keras tensor (future input to layer) or ``list``/``tuple`` of Keras tensors to
 251             reference for weight shape computations.
 252         """
 253         pass

 255     def call(self, input_tensor, mask=None):  # pylint:disable=unused-argument,arguments-differ
 256         """This is where the layer's logic lives.
 257 
 258         Parameters
 259         ----------
 260         inputs: tensor
 261             Input tensor, or list/tuple of input tensors
 262         kwargs: dict
 263             Additional keyword arguments
 264 
 265         Returns
 266         -------
 267         tensor
 268             A tensor or list/tuple of tensors
 269         """
 270         retval = self._depth_to_space(input_tensor, self.scale_factor, self.data_format)
 271         return retval

 273     def compute_output_shape(self, input_shape):
 274         """Computes the output shape of the layer.
 275 
 276         Assumes that the layer will be built to match that input shape provided.
 277 
 278         Parameters
 279         ----------
 280         input_shape: tuple or list of tuples
 281             Shape tuple (tuple of integers) or list of shape tuples (one per output tensor of the
 282             layer).  Shape tuples can include None for free dimensions, instead of an integer.
 283 
 284         Returns
 285         -------
 286         tuple
 287             An input shape tuple
 288         """
 289         if self.data_format == "channels_first":
 290             batch, channels, rows, columns = input_shape
 291             return (batch,
 292                     channels // (self.scale_factor ** 2),
 293                     rows * self.scale_factor,
 294                     columns * self.scale_factor)
 295         batch, rows, columns, channels = input_shape
 296         return (batch,
 297                 rows * self.scale_factor,
 298                 columns * self.scale_factor,
 299                 channels // (self.scale_factor ** 2))

 301     @classmethod
 302     def _depth_to_space(cls, ipt, scale, data_format=None):
 303         """ Uses phase shift algorithm to convert channels/depth for spatial resolution """
 304         if data_format is None:
 305             data_format = K.image_data_format()
 306         data_format = data_format.lower()
 307         ipt = cls._preprocess_conv2d_input(ipt, data_format)
 308         out = tf.depth_to_space(ipt, scale)
 309         out = cls._postprocess_conv2d_output(out, data_format)
 310         return out

 312     @staticmethod
 313     def _postprocess_conv2d_output(input_tensor, data_format):
 314         """Transpose and cast the output from conv2d if needed.
 315 
 316         Parameters
 317         ----------
 318         input_tensor: tensor
 319             The input that requires transposing and casting
 320         data_format: str
 321             `"channels_last"` or `"channels_first"`
 322 
 323         Returns
 324         -------
 325         tensor
 326             The transposed and cast input tensor
 327         """
 328 
 329         if data_format == "channels_first":
 330             input_tensor = tf.transpose(input_tensor, (0, 3, 1, 2))
 331 
 332         if K.floatx() == "float64":
 333             input_tensor = tf.cast(input_tensor, "float64")
 334         return input_tensor
 335 
 336     @staticmethod
 337     def _preprocess_conv2d_input(input_tensor, data_format):
 338         """Transpose and cast the input before the conv2d.
 339 
 340         Parameters
 341         ----------
 342         input_tensor: tensor
 343             The input that requires transposing and casting
 344         data_format: str
 345             `"channels_last"` or `"channels_first"`
 346 
 347         Returns
 348         -------
 349         tensor
 350             The transposed and cast input tensor
 351         """
 352         if K.dtype(input_tensor) == "float64":
 353             input_tensor = tf.cast(input_tensor, "float32")
 354         if data_format == "channels_first":
 355             # Tensorflow uses the last dimension as channel dimension, instead of the 2nd one.
 356             # Theano input shape: (samples, input_depth, rows, cols)
 357             # Tensorflow input shape: (samples, rows, cols, input_depth)
 358             input_tensor = tf.transpose(input_tensor, (0, 2, 3, 1))
 359         return input_tensor

 382 class ReflectionPadding2D(Layer):
 383     """Reflection-padding layer for 2D input (e.g. picture).
 384 
 385     This layer can add rows and columns at the top, bottom, left and right side of an image tensor.
 386 
 387     Parameters
 388     ----------
 389     stride: int, optional
 390         The stride of the following convolution. Default: `2`
 391     kernel_size: int, optional
 392         The kernel size of the following convolution. Default: `5`
 393     kwargs: dict
 394         The standard Keras Layer keyword arguments (if any)
 395     """
 396     def __init__(self, stride=2, kernel_size=5, **kwargs):
 397         self.stride = stride
 398         self.kernel_size = kernel_size
 399         super().__init__(**kwargs)
 400 
 401     def build(self, input_shape):
 402         """Creates the layer weights.
 403 
 404         Must be implemented on all layers that have weights.
 405 
 406         Parameters
 407         ----------
 408         input_shape: tensor
 409             Keras tensor (future input to layer) or ``list``/``tuple`` of Keras tensors to
 410             reference for weight shape computations.
 411         """
 412         self.input_spec = [InputSpec(shape=input_shape)]
 413         super().build(input_shape)
 414 
 415     def compute_output_shape(self, input_shape):
 416         """Computes the output shape of the layer.
 417 
 418         Assumes that the layer will be built to match that input shape provided.
 419 
 420         Parameters
 421         ----------
 422         input_shape: tuple or list of tuples
 423             Shape tuple (tuple of integers) or list of shape tuples (one per output tensor of the
 424             layer).  Shape tuples can include None for free dimensions, instead of an integer.
 425 
 426         Returns
 427         -------
 428         tuple
 429             An input shape tuple
 430         """
 431         input_shape = self.input_spec[0].shape
 432         in_width, in_height = input_shape[2], input_shape[1]
 433         kernel_width, kernel_height = self.kernel_size, self.kernel_size
 434 
 435         if (in_height % self.stride) == 0:
 436             padding_height = max(kernel_height - self.stride, 0)
 437         else:
 438             padding_height = max(kernel_height - (in_height % self.stride), 0)
 439         if (in_width % self.stride) == 0:
 440             padding_width = max(kernel_width - self.stride, 0)
 441         else:
 442             padding_width = max(kernel_width - (in_width % self.stride), 0)
 443 
 444         return (input_shape[0],
 445                 input_shape[1] + padding_height,
 446                 input_shape[2] + padding_width,
 447                 input_shape[3])
 448 
 449     def call(self, x, mask=None):  # pylint:disable=unused-argument,arguments-differ
 450         """This is where the layer's logic lives.
 451 
 452         Parameters
 453         ----------
 454         inputs: tensor
 455             Input tensor, or list/tuple of input tensors
 456         kwargs: dict
 457             Additional keyword arguments
 458 
 459         Returns
 460         -------
 461         tensor
 462             A tensor or list/tuple of tensors
 463         """
 464         input_shape = self.input_spec[0].shape
 465         in_width, in_height = input_shape[2], input_shape[1]
 466         kernel_width, kernel_height = self.kernel_size, self.kernel_size
 467 
 468         if (in_height % self.stride) == 0:
 469             padding_height = max(kernel_height - self.stride, 0)
 470         else:
 471             padding_height = max(kernel_height - (in_height % self.stride), 0)
 472         if (in_width % self.stride) == 0:
 473             padding_width = max(kernel_width - self.stride, 0)
 474         else:
 475             padding_width = max(kernel_width - (in_width % self.stride), 0)
 476 
 477         padding_top = padding_height // 2
 478         padding_bot = padding_height - padding_top
 479         padding_left = padding_width // 2
 480         padding_right = padding_width - padding_left
 481 
 482         return pad(x,
 483                    [[0, 0],
 484                     [padding_top, padding_bot],
 485                     [padding_left, padding_right],
 486                     [0, 0]],
 487                    'REFLECT')
