  19 class ICNR(initializers.Initializer):  # pylint: disable=invalid-name
  20     """ ICNR initializer for checkerboard artifact free sub pixel convolution
  21 
  22     Parameters
  23     ----------
  24     initializer: :class:`keras.initializers.Initializer`
  25         The initializer used for sub kernels (orthogonal, glorot uniform, etc.)
  26     scale: int
  27         scaling factor of sub pixel convolution (up sampling from 8x8 to 16x16 is scale 2)
  28 
  29     Returns
  30     -------
  31     tensor
  32         The modified kernel weights
  33 
  34     Example
  35     -------
  36     >>> x = conv2d(... weights_initializer=ICNR(initializer=he_uniform(), scale=2))
  37 
  38     References
  39     ----------
  40     Andrew Aitken et al. Checkerboard artifact free sub-pixel convolution
  41     https://arxiv.org/pdf/1707.02937.pdf,  https://distill.pub/2016/deconv-checkerboard/
  42     """
  43 
  44     def __init__(self, initializer, scale=2):
  45         self.scale = scale
  46         self.initializer = initializer
  47 
  48     def __call__(self, shape, dtype="float32"):
  49         """ Call function for the ICNR initializer.
  50 
  51         Parameters
  52         ----------
  53         shape: tuple or list
  54             The required resized shape for the output tensor
  55         dtype: str
  56             The data type for the tensor
  57 
  58         Returns
  59         -------
  60         tensor
  61             The modified kernel weights
  62         """
  63         shape = list(shape)
  64         if self.scale == 1:
  65             return self.initializer(shape)
  66         new_shape = shape[:3] + [shape[3] // (self.scale ** 2)]
  67         if isinstance(self.initializer, dict):
  68             self.initializer = initializers.deserialize(self.initializer)
  69         var_x = self.initializer(new_shape, dtype)
  70         var_x = K.permute_dimensions(var_x, [2, 0, 1, 3])
  71         var_x = self._resize_nearest_neighbour(var_x,
  72                                                (shape[0] * self.scale, shape[1] * self.scale))
  73         var_x = self._space_to_depth(var_x)
  74         var_x = K.permute_dimensions(var_x, [1, 2, 0, 3])
  75         logger.debug("Output: %s", var_x)
  76         return var_x
  77 
  78     def _resize_nearest_neighbour(self, input_tensor, size):
  79         """ Resize a tensor using nearest neighbor interpolation.
  80 
  81         Notes
  82         -----
  83         Tensorflow has a bug that resizes the image incorrectly if :attr:`align_corners` is not set
  84         to ``True``. Keras Backend does not set this flag, so we explicitly call the Tensorflow
  85         operation for non-amd backends.
  86 
  87         Parameters
  88         ----------
  89         input_tensor: tensor
  90             The tensor to be resized
  91         tuple: int
  92             The (`h`, `w`) that the tensor should be resized to (used for non-amd backends only)
  93 
  94         Returns
  95         -------
  96         tensor
  97             The input tensor resized to the given size
  98         """
  99         if get_backend() == "amd":
 100             retval = K.resize_images(input_tensor, self.scale, self.scale, "channels_last",
 101                                      interpolation="nearest")
 102         else:
 103             retval = tf.image.resize_nearest_neighbor(input_tensor, size=size, align_corners=True)
 104         logger.debug("Input Tensor: %s, Output Tensor: %s", input_tensor, retval)
 105         return retval
 106 
 107     def _space_to_depth(self, input_tensor):
 108         """ Space to depth implementation.
 109 
 110         PlaidML does not have a space to depth operation, so calculate if backend is amd
 111         otherwise returns the :func:`tensorflow.space_to_depth` operation.
 112 
 113         Parameters
 114         ----------
 115         input_tensor: tensor
 116             The tensor to be manipulated
 117 
 118         Returns
 119         -------
 120         tensor
 121             The manipulated input tensor
 122         """
 123         if get_backend() == "amd":
 124             batch, height, width, depth = input_tensor.shape.dims
 125             new_height = height // self.scale
 126             new_width = width // self.scale
 127             reshaped = K.reshape(input_tensor,
 128                                  (batch, new_height, self.scale, new_width, self.scale, depth))
 129             retval = K.reshape(K.permute_dimensions(reshaped, [0, 1, 3, 2, 4, 5]),
 130                                (batch, new_height, new_width, -1))
 131         else:
 132             retval = tf.space_to_depth(input_tensor, block_size=self.scale, data_format="NHWC")
 133         logger.debug("Input Tensor: %s, Output Tensor: %s", input_tensor, retval)
 134         return retval
 135 
 136     def get_config(self):
 137         """ Return the ICNR Initializer configuration.
 138 
 139         Returns
 140         -------
 141         dict

 148         return dict(list(base_config.items()) + list(config.items()))

 151 class ConvolutionAware(initializers.Initializer):
 152     """
 153     Initializer that generates orthogonal convolution filters in the Fourier space. If this
 154     initializer is passed a shape that is not 3D or 4D, orthogonal initialization will be used.
 155 
 156     Adapted, fixed and optimized from:
 157     https://github.com/keras-team/keras-contrib/blob/master/keras_contrib/initializers/convaware.py
 158 
 159     Parameters
 160     ----------
 161     eps_std: float
 162         The Standard deviation for the random normal noise used to break symmetry in the inverse
 163         Fourier transform.
 164     seed: int, optional
 165         Used to seed the random generator. Default: ``None``
 166 
 167     Returns
 168     -------
 169     tensor
 170         The modified kernel weights
 171 
 172     References
 173     ----------
 174     Armen Aghajanyan, https://arxiv.org/abs/1702.06295
 175 
 176     Notes
 177     -----
 178     Convolutional Aware Initialization takes a long time. Keras model loading loads a model,
 179     performs initialization and then loads weights, which is an unnecessary waste of time.
 180     init defaults to False so that this is bypassed when loading a saved model passing zeros.
 181     """
 182 
 183     def __init__(self, eps_std=0.05, seed=None, init=False):
 184         self._init = init
 185         self.eps_std = eps_std
 186         self.seed = seed
 187         self.orthogonal = initializers.Orthogonal()
 188         self.he_uniform = initializers.he_uniform()
 189 
 190     def __call__(self, shape, dtype=None):
 191         """ Call function for the ICNR initializer.
 192 
 193         Parameters
 194         ----------
 195         shape: tuple or list
 196             The required shape for the output tensor
 197         dtype: str
 198             The data type for the tensor
 199 
 200         Returns
 201         -------
 202         tensor
 203             The modified kernel weights
 204         """
 205         dtype = K.floatx() if dtype is None else dtype
 206         if self._init:
 207             logger.info("Calculating Convolution Aware Initializer for shape: %s", shape)
 208         else:
 209             logger.debug("Bypassing Convolutional Aware Initializer for saved model")
 210             # Dummy in he_uniform just in case there aren't any weighs being loaded
 211             # and it needs some kind of initialization
 212             return self.he_uniform(shape, dtype=dtype)
 213 
 214         rank = len(shape)
 215         if self.seed is not None:
 216             np.random.seed(self.seed)
 217 
 218         fan_in, _ = initializers._compute_fans(shape)  # pylint:disable=protected-access
 219         variance = 2 / fan_in
 220 
 221         if rank == 3:
 222             row, stack_size, filters_size = shape
 223 
 224             transpose_dimensions = (2, 1, 0)
 225             kernel_shape = (row,)
 226             correct_ifft = lambda shape, s=[None]: np.fft.irfft(shape, s[0])  # noqa
 227             correct_fft = np.fft.rfft
 228 
 229         elif rank == 4:
 230             row, column, stack_size, filters_size = shape
 231 
 232             transpose_dimensions = (2, 3, 1, 0)
 233             kernel_shape = (row, column)
 234             correct_ifft = np.fft.irfft2
 235             correct_fft = np.fft.rfft2
 236 
 237         elif rank == 5:
 238             var_x, var_y, var_z, stack_size, filters_size = shape
 239 
 240             transpose_dimensions = (3, 4, 0, 1, 2)
 241             kernel_shape = (var_x, var_y, var_z)
 242             correct_fft = np.fft.rfftn
 243             correct_ifft = np.fft.irfftn
 244 
 245         else:
 246             return K.variable(self.orthogonal(shape), dtype=dtype)
 247 
 248         kernel_fourier_shape = correct_fft(np.zeros(kernel_shape)).shape
 249 
 250         basis = self._create_basis(filters_size, stack_size, np.prod(kernel_fourier_shape), dtype)
 251         basis = basis.reshape((filters_size, stack_size,) + kernel_fourier_shape)
 252         randoms = np.random.normal(0, self.eps_std, basis.shape[:-2] + kernel_shape)
 253         init = correct_ifft(basis, kernel_shape) + randoms
 254         init = self._scale_filters(init, variance)
 255         return K.variable(init.transpose(transpose_dimensions), dtype=dtype, name="conv_aware")
 256 
 257     def _create_basis(self, filters_size, filters, size, dtype):
 258         """ Create the basis for convolutional aware initialization """
 259         if size == 1:
 260             return np.random.normal(0.0, self.eps_std, (filters_size, filters, size))
 261         nbb = filters // size + 1
 262         var_a = np.random.normal(0.0, 1.0, (filters_size, nbb, size, size))
 263         var_a = self._symmetrize(var_a)
 264         var_u = np.linalg.svd(var_a)[0].transpose(0, 1, 3, 2)
 265         var_p = np.reshape(var_u, (filters_size, nbb * size, size))[:, :filters, :].astype(dtype)
 266         return var_p
 267 
 268     @staticmethod
 269     def _symmetrize(var_a):
 270         """ Make the given tensor symmetrical. """
 271         var_b = np.transpose(var_a, axes=(0, 1, 3, 2))
 272         diag = var_a.diagonal(axis1=2, axis2=3)
 273         var_c = np.array([[np.diag(arr) for arr in batch] for batch in diag])
 274         return var_a + var_b - var_c
 275 
 276     @staticmethod
 277     def _scale_filters(filters, variance):
 278         """ Scale the given filters. """
 279         c_var = np.var(filters)
 280         var_p = np.sqrt(variance / c_var)
 281         return filters * var_p
 282 
 283     def get_config(self):
 284         """ Return the Convolutional Aware Initializer configuration.
 285 
 286         Returns
 287         -------
 288         dict
 289             The configuration for ICNR Initialization
 290         """
 291         return {
 292             "eps_std": self.eps_std,
 293             "seed": self.seed
 294         }
