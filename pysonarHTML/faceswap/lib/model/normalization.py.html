  13 class InstanceNormalization(Layer):
  14     """Instance normalization layer (Lei Ba et al, 2016, Ulyanov et al., 2016).
  15 
  16     Normalize the activations of the previous layer at each step, i.e. applies a transformation
  17     that maintains the mean activation close to 0 and the activation standard deviation close to 1.
  18 
  19     Parameters
  20     ----------
  21     axis: int, optional
  22         The axis that should be normalized (typically the features axis). For instance, after a
  23         `Conv2D` layer with `data_format="channels_first"`, set `axis=1` in
  24         :class:`InstanceNormalization`. Setting `axis=None` will normalize all values in each
  25         instance of the batch. Axis 0 is the batch dimension. `axis` cannot be set to 0 to avoid
  26         errors. Default: ``None``
  27     epsilon: float, optional
  28         Small float added to variance to avoid dividing by zero. Default: `1e-3`
  29     center: bool, optional
  30         If ``True``, add offset of `beta` to normalized tensor. If ``False``, `beta` is ignored.
  31         Default: ``True``
  32     scale: bool, optional
  33         If ``True``, multiply by `gamma`. If ``False``, `gamma` is not used. When the next layer
  34         is linear (also e.g. `relu`), this can be disabled since the scaling will be done by
  35         the next layer. Default: ``True``
  36     beta_initializer: str, optional
  37         Initializer for the beta weight. Default: `"zeros"`
  38     gamma_initializer: str, optional
  39         Initializer for the gamma weight. Default: `"ones"`
  40     beta_regularizer: str, optional
  41         Optional regularizer for the beta weight. Default: ``None``
  42     gamma_regularizer: str, optional
  43         Optional regularizer for the gamma weight. Default: ``None``
  44     beta_constraint: float, optional
  45         Optional constraint for the beta weight. Default: ``None``
  46     gamma_constraint: float, optional
  47         Optional constraint for the gamma weight. Default: ``None``
  48 
  49     References
  50     ----------
  51         - Layer Normalization - https://arxiv.org/abs/1607.06450
  52 
  53         - Instance Normalization: The Missing Ingredient for Fast Stylization -
  54     https://arxiv.org/abs/1607.08022
  55 
  56     """
  57     def __init__(self,
  58                  axis=None,
  59                  epsilon=1e-3,
  60                  center=True,
  61                  scale=True,
  62                  beta_initializer="zeros",
  63                  gamma_initializer="ones",
  64                  beta_regularizer=None,
  65                  gamma_regularizer=None,
  66                  beta_constraint=None,
  67                  gamma_constraint=None,
  68                  **kwargs):
  69         self.beta = None
  70         self.gamma = None
  71         super().__init__(**kwargs)
  72         self.supports_masking = True
  73         self.axis = axis
  74         self.epsilon = epsilon
  75         self.center = center
  76         self.scale = scale
  77         self.beta_initializer = initializers.get(beta_initializer)
  78         self.gamma_initializer = initializers.get(gamma_initializer)
  79         self.beta_regularizer = regularizers.get(beta_regularizer)
  80         self.gamma_regularizer = regularizers.get(gamma_regularizer)
  81         self.beta_constraint = constraints.get(beta_constraint)
  82         self.gamma_constraint = constraints.get(gamma_constraint)
  83 
  84     def build(self, input_shape):
  85         """Creates the layer weights.
  86 
  87         Must be implemented on all layers that have weights.
  88 
  89         Parameters
  90         ----------
  91         input_shape: tensor
  92             Keras tensor (future input to layer) or ``list``/``tuple`` of Keras tensors to
  93             reference for weight shape computations.
  94         """
  95         ndim = len(input_shape)
  96         if self.axis == 0:
  97             raise ValueError("Axis cannot be zero")
  98 
  99         if (self.axis is not None) and (ndim == 2):
 100             raise ValueError("Cannot specify axis for rank 1 tensor")
 101 
 102         self.input_spec = InputSpec(ndim=ndim)
 103 
 104         if self.axis is None:
 105             shape = (1,)
 106         else:
 107             shape = (input_shape[self.axis],)
 108 
 109         if self.scale:
 110             self.gamma = self.add_weight(shape=shape,
 111                                          name="gamma",
 112                                          initializer=self.gamma_initializer,
 113                                          regularizer=self.gamma_regularizer,
 114                                          constraint=self.gamma_constraint)
 115         else:
 116             self.gamma = None
 117         if self.center:
 118             self.beta = self.add_weight(shape=shape,
 119                                         name="beta",
 120                                         initializer=self.beta_initializer,
 121                                         regularizer=self.beta_regularizer,
 122                                         constraint=self.beta_constraint)
 123         else:
 124             self.beta = None
 125         self.built = True
 126 
 127     def call(self, inputs, training=None):  # pylint:disable=arguments-differ,unused-argument
 128         """This is where the layer's logic lives.
 129 
 130         Parameters
 131         ----------
 132         inputs: tensor
 133             Input tensor, or list/tuple of input tensors
 134 
 135         Returns
 136         -------
 137         tensor
 138             A tensor or list/tuple of tensors
 139         """
 140         input_shape = K.int_shape(inputs)
 141         reduction_axes = list(range(0, len(input_shape)))
 142 
 143         if self.axis is not None:
 144             del reduction_axes[self.axis]
 145 
 146         del reduction_axes[0]
 147 
 148         mean = K.mean(inputs, reduction_axes, keepdims=True)
 149         stddev = K.std(inputs, reduction_axes, keepdims=True) + self.epsilon
 150         normed = (inputs - mean) / stddev
 151 
 152         broadcast_shape = [1] * len(input_shape)
 153         if self.axis is not None:
 154             broadcast_shape[self.axis] = input_shape[self.axis]
 155 
 156         if self.scale:
 157             broadcast_gamma = K.reshape(self.gamma, broadcast_shape)
 158             normed = normed * broadcast_gamma
 159         if self.center:
 160             broadcast_beta = K.reshape(self.beta, broadcast_shape)
 161             normed = normed + broadcast_beta
 162         return normed
 163 
 164     def get_config(self):
 165         config = {
 166             "axis": self.axis,
 167             "epsilon": self.epsilon,
 168             "center": self.center,
 169             "scale": self.scale,
 170             "beta_initializer": initializers.serialize(self.beta_initializer),
 171             "gamma_initializer": initializers.serialize(self.gamma_initializer),
 172             "beta_regularizer": regularizers.serialize(self.beta_regularizer),
 173             "gamma_regularizer": regularizers.serialize(self.gamma_regularizer),
 174             "beta_constraint": constraints.serialize(self.beta_constraint),
 175             "gamma_constraint": constraints.serialize(self.gamma_constraint)
 176         }
 177         base_config = super(InstanceNormalization, self).get_config()
 178         return dict(list(base_config.items()) + list(config.items()))
