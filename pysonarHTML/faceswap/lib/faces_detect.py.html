  15 class DetectedFace():
  16     """ Detected face and landmark information
  17 
  18     Holds information about a detected face, it's location in a source image
  19     and the face's 68 point landmarks.
  20 
  21     Methods for aligning a face are also callable from here.
  22 
  23     Parameters
  24     ----------
  25     image: numpy.ndarray, optional
  26         Original frame that holds this face. Optional (not required if just storing coordinates)
  27     x: int
  28         The left most point (in pixels) of the face's bounding box as discovered in
  29         :mod:`plugins.extract.detect`
  30     w: int
  31         The width (in pixels) of the face's bounding box as discovered in
  32         :mod:`plugins.extract.detect`
  33     y: int
  34         The top most point (in pixels) of the face's bounding box as discovered in
  35         :mod:`plugins.extract.detect`
  36     h: int
  37         The height (in pixels) of the face's bounding box as discovered in
  38         :mod:`plugins.extract.detect`
  39     landmarks_xy: list
  40         The 68 point landmarks as discovered in :mod:`plugins.extract.align`. Should be a ``list``
  41         of 68 `(x, y)` ``tuples`` with each of the landmark co-ordinates.
  42     mask: dict
  43         The generated mask(s) for the face as generated in :mod:`plugins.extract.mask`. Must be a
  44         dict of {**name** (`str`): :class:`Mask`}.
  45 
  46     Attributes
  47     ----------
  48     image: numpy.ndarray, optional
  49         This is a generic image placeholder that should not be relied on to be holding a particular
  50         image. It may hold the source frame that holds the face, a cropped face or a scaled image
  51         depending on the method using this object.
  52     x: int
  53         The left most point (in pixels) of the face's bounding box as discovered in
  54         :mod:`plugins.extract.detect`
  55     w: int
  56         The width (in pixels) of the face's bounding box as discovered in
  57         :mod:`plugins.extract.detect`
  58     y: int
  59         The top most point (in pixels) of the face's bounding box as discovered in
  60         :mod:`plugins.extract.detect`
  61     h: int
  62         The height (in pixels) of the face's bounding box as discovered in
  63         :mod:`plugins.extract.detect`
  64     landmarks_xy: list

  70         The hash of the face. This cannot be set until the file is saved due to image compression,

 121     def add_mask(self, name, mask, affine_matrix, interpolator, storage_size=128):
 122         """ Add a :class:`Mask` to this detected face
 123 
 124         The mask should be the original output from  :mod:`plugins.extract.mask`
 125         If a mask with this name already exists it will be overwritten by the given
 126         mask.
 127 
 128         Parameters
 129         ----------
 130         name: str
 131             The name of the mask as defined by the :attr:`plugins.extract.mask._base.name`
 132             parameter.
 133         mask: numpy.ndarray
 134             The mask that is to be added as output from :mod:`plugins.extract.mask`
 135             It should be in the range 0.0 - 1.0 ideally with a ``dtype`` of ``float32``
 136         affine_matrix: numpy.ndarray
 137             The transformation matrix required to transform the mask to the original frame.
 138         interpolator, int:
 139             The CV2 interpolator required to transform this mask to it's original frame.
 140         storage_size, int (optional):
 141             The size the mask is to be stored at. Default: 128
 142         """
 143         logger.trace("name: '%s', mask shape: %s, affine_matrix: %s, interpolator: %s)",
 144                      name, mask.shape, affine_matrix, interpolator)
 145         fsmask = Mask(storage_size=storage_size)
 146         fsmask.add(mask, affine_matrix, interpolator)
 147         self.mask[name] = fsmask

 149     def to_alignment(self):
 150         """  Return the detected face formatted for an alignments file
 151 
 152         returns
 153         -------
 154         alignment: dict
 155             The alignment dict will be returned with the keys ``x``, ``w``, ``y``, ``h``,
 156             ``landmarks_xy``, ``mask``, ``hash``.
 157         """
 158 
 159         alignment = dict()
 160         alignment["x"] = self.x
 161         alignment["w"] = self.w
 162         alignment["y"] = self.y
 163         alignment["h"] = self.h
 164         alignment["landmarks_xy"] = self.landmarks_xy
 165         alignment["hash"] = self.hash
 166         alignment["mask"] = {name: mask.to_dict() for name, mask in self.mask.items()}
 167         logger.trace("Returning: %s", alignment)
 168         return alignment

 170     def from_alignment(self, alignment, image=None):
 171         """ Set the attributes of this class from an alignments file and optionally load the face
 172         into the ``image`` attribute.
 173 
 174         Parameters
 175         ----------
 176         alignment: dict
 177             A dictionary entry for a face from an alignments file containing the keys
 178             ``x``, ``w``, ``y``, ``h``, ``landmarks_xy``.
 179             Optionally the key ``hash`` will be provided, but not all use cases will know the
 180             face hash at this time.
 181             Optionally the key ``mask`` will be provided, but legacy alignments will not have
 182             this key.
 183         image: numpy.ndarray, optional
 184             If an image is passed in, then the ``image`` attribute will
 185             be set to the cropped face based on the passed in bounding box co-ordinates
 186         """
 187 
 188         logger.trace("Creating from alignment: (alignment: %s, has_image: %s)",
 189                      alignment, bool(image is not None))
 190         self.x = alignment["x"]
 191         self.w = alignment["w"]
 192         self.y = alignment["y"]
 193         self.h = alignment["h"]
 194         landmarks = alignment["landmarks_xy"]
 195         if not isinstance(landmarks, np.ndarray):
 196             landmarks = np.array(landmarks, dtype="float32")
 197         self.landmarks_xy = landmarks
 198         # Manual tool does not know the final hash so default to None
 199         self.hash = alignment.get("hash", None)
 200         # Manual tool and legacy alignments will not have a mask
 201         if alignment.get("mask", None) is not None:
 202             self.mask = dict()
 203             for name, mask_dict in alignment["mask"].items():
 204                 self.mask[name] = Mask()
 205                 self.mask[name].from_dict(mask_dict)
 206         if image is not None and image.any():
 207             self._image_to_face(image)
 208         logger.trace("Created from alignment: (x: %s, w: %s, y: %s. h: %s, "
 209                      "landmarks: %s, mask: %s)",
 210                      self.x, self.w, self.y, self.h, self.landmarks_xy, self.mask)

 212     def _image_to_face(self, image):
 213         """ set self.image to be the cropped face from detected bounding box """
 214         logger.trace("Cropping face from image")
 215         self.image = image[self.top: self.bottom,
 216                            self.left: self.right]

 219     def load_aligned(self, image, size=256, dtype=None, force=False):
 220         """ Align a face from a given image.
 221 
 222         Aligning a face is a relatively expensive task and is not required for all uses of
 223         the :class:`~lib.faces_detect.DetectedFace` object, so call this function explicitly to
 224         load an aligned face.
 225 
 226         This method plugs into :mod:`lib.aligner` to perform face alignment based on this face's
 227         ``landmarks_xy``. If the face has already been aligned, then this function will return
 228         having performed no action.
 229 
 230         Parameters
 231         ----------
 232         image: numpy.ndarray
 233             The image that contains the face to be aligned
 234         size: int
 235             The size of the output face in pixels
 236         dtype: str, optional
 237             Optionally set a ``dtype`` for the final face to be formatted in. Default: ``None``
 238         force: bool, optional
 239             Force an update of the aligned face, even if it is already loaded. Default: ``False``
 240 
 241         Notes
 242         -----
 243         This method must be executed to get access to the following `properties`:
 244             - :func:`original_roi`
 245             - :func:`aligned_landmarks`
 246             - :func:`aligned_face`
 247             - :func:`adjusted_interpolators`
 248         """
 249         if self.aligned and not force:
 250             # Don't reload an already aligned face
 251             logger.trace("Skipping alignment calculation for already aligned face")
 252         else:
 253             logger.trace("Loading aligned face: (size: %s, dtype: %s)", size, dtype)
 254             padding = int(size * self._extract_ratio) // 2
 255             self.aligned["size"] = size
 256             self.aligned["padding"] = padding
 257             self.aligned["matrix"] = get_align_mat(self)
 258             self.aligned["face"] = None
 259         if image is not None and (self.aligned["face"] is None or force):
 260             logger.trace("Getting aligned face")
 261             face = AlignerExtract().transform(image, self.aligned["matrix"], size, padding)
 262             self.aligned["face"] = face if dtype is None else face.astype(dtype)
 263 
 264         logger.trace("Loaded aligned face: %s", {k: str(v) if isinstance(v, np.ndarray) else v
 265                                                  for k, v in self.aligned.items()
 266                                                  if k != "face"})

 268     def _padding_from_coverage(self, size, coverage_ratio):
 269         """ Return the image padding for a face from coverage_ratio set against a
 270             pre-padded training image """
 271         adjusted_ratio = coverage_ratio - (1 - self._extract_ratio)
 272         padding = round((size * adjusted_ratio) / 2)
 273         logger.trace(padding)
 274         return padding

 276     def load_feed_face(self, image, size=64, coverage_ratio=0.625, dtype=None,
 277                        is_aligned_face=False):
 278         """ Align a face in the correct dimensions for feeding into a model.
 279 
 280         Parameters
 281         ----------
 282         image: numpy.ndarray
 283             The image that contains the face to be aligned
 284         size: int
 285             The size of the face in pixels to be fed into the model
 286         coverage_ratio: float, optional
 287             the ratio of the extracted image that was used for training. Default: `0.625`
 288         dtype: str, optional
 289             Optionally set a ``dtype`` for the final face to be formatted in. Default: ``None``
 290         is_aligned_face: bool, optional
 291             Indicates that the :attr:`image` is an aligned face rather than a frame.
 292             Default: ``False``
 293 
 294         Notes
 295         -----
 296         This method must be executed to get access to the following `properties`:
 297             - :func:`feed_face`
 298             - :func:`feed_interpolators`
 299         """
 300         logger.trace("Loading feed face: (size: %s, coverage_ratio: %s, dtype: %s, "
 301                      "is_aligned_face: %s)", size, coverage_ratio, dtype, is_aligned_face)
 302 
 303         self.feed["size"] = size
 304         self.feed["padding"] = self._padding_from_coverage(size, coverage_ratio)
 305         self.feed["matrix"] = get_align_mat(self)
 306         if is_aligned_face:
 307             original_size = image.shape[0]
 308             interp = cv2.INTER_CUBIC if original_size < size else cv2.INTER_AREA
 309             face = cv2.resize(image, (size, size), interpolation=interp)
 310         else:
 311             face = AlignerExtract().transform(image,
 312                                               self.feed["matrix"],
 313                                               size,
 314                                               self.feed["padding"])
 315         self.feed["face"] = face if dtype is None else face.astype(dtype)
 316 
 317         logger.trace("Loaded feed face. (face_shape: %s, matrix: %s)",
 318                      self.feed_face.shape, self.feed_matrix)

 498 class Mask():
 499     """ Face Mask information and convenience methods
 500 
 501     Holds a Faceswap mask as generated from :mod:`plugins.extract.mask` and the information
 502     required to transform it to its original frame.
 503 
 504     Holds convenience methods to handle the warping, storing and retrieval of the mask.
 505 
 506     Parameters
 507     ----------
 508     storage_size: int, optional
 509         The size (in pixels) that the mask should be stored at. Default: 128.
 510 
 511     Attributes
 512     ----------
 513     stored_size: int
 514         The size, in pixels, of the stored mask across its height and width.
 515     """
 516 
 517     def __init__(self, storage_size=128):
 518         self.stored_size = storage_size
 519 
 520         self._mask = None
 521         self._affine_matrix = None
 522         self._interpolator = None
 523 
 524         self._blur = dict()
 525         self._blur_kernel = 0
 526         self._threshold = 0.0
 527         self.set_blur_and_threshold()
 528 
 529     @property
 530     def mask(self):
 531         """ numpy.ndarray: The mask at the size of :attr:`stored_size` with any requested blurring
 532         and threshold amount applied."""
 533         dims = (self.stored_size, self.stored_size, 1)
 534         mask = np.frombuffer(decompress(self._mask), dtype="uint8").reshape(dims)
 535         if self._threshold != 0.0 or self._blur["kernel"] != 0:
 536             mask = mask.copy()
 537         if self._threshold != 0.0:
 538             mask[mask < self._threshold] = 0.0
 539             mask[mask > 255.0 - self._threshold] = 255.0
 540         if self._blur["kernel"] != 0:
 541             mask = BlurMask(self._blur["type"],
 542                             mask,
 543                             self._blur["kernel"],
 544                             passes=self._blur["passes"]).blurred
 545         logger.trace("mask shape: %s", mask.shape)
 546         return mask
 547 
 548     @property
 549     def original_roi(self):
 550         """ :class: `numpy.ndarray`: The original region of interest of the mask in the
 551         source frame. """
 552         points = np.array([[0, 0],
 553                            [0, self.stored_size - 1],
 554                            [self.stored_size - 1, self.stored_size - 1],
 555                            [self.stored_size - 1, 0]], np.int32).reshape((-1, 1, 2))
 556         matrix = cv2.invertAffineTransform(self._affine_matrix)
 557         roi = cv2.transform(points, matrix).reshape((4, 2))
 558         logger.trace("Returning: %s", roi)
 559         return roi
 560 
 561     @property
 562     def affine_matrix(self):
 563         """ :class: `numpy.ndarray`: The affine matrix to transpose the mask to a full frame. """
 564         return self._affine_matrix
 565 
 566     @property
 567     def interpolator(self):
 568         """ int: The cv2 interpolator required to transpose the mask to a full frame. """
 569         return self._interpolator
 570 
 571     def get_full_frame_mask(self, width, height):
 572         """ Return the stored mask in a full size frame of the given dimensions
 573 
 574         Parameters
 575         ----------
 576         width: int
 577             The width of the original frame that the mask was extracted from
 578         height: int
 579             The height of the original frame that the mask was extracted from
 580 
 581         Returns
 582         -------
 583         numpy.ndarray: The mask affined to the original full frame of the given dimensions
 584         """
 585         frame = np.zeros((width, height, 1), dtype="uint8")
 586         mask = cv2.warpAffine(self.mask,
 587                               self._affine_matrix,
 588                               (width, height),
 589                               frame,
 590                               flags=cv2.WARP_INVERSE_MAP | self._interpolator,
 591                               borderMode=cv2.BORDER_CONSTANT)
 592         logger.trace("mask shape: %s, mask dtype: %s, mask min: %s, mask max: %s",
 593                      mask.shape, mask.dtype, mask.min(), mask.max())
 594         return mask

 596     def add(self, mask, affine_matrix, interpolator):
 597         """ Add a Faceswap mask to this :class:`Mask`.
 598 
 599         The mask should be the original output from  :mod:`plugins.extract.mask`
 600 
 601         Parameters
 602         ----------
 603         mask: numpy.ndarray
 604             The mask that is to be added as output from :mod:`plugins.extract.mask`
 605             It should be in the range 0.0 - 1.0 ideally with a ``dtype`` of ``float32``
 606         affine_matrix: numpy.ndarray
 607             The transformation matrix required to transform the mask to the original frame.
 608         interpolator, int:
 609             The CV2 interpolator required to transform this mask to it's original frame
 610         """
 611         logger.trace("mask shape: %s, mask dtype: %s, mask min: %s, mask max: %s, "
 612                      "affine_matrix: %s, interpolator: %s)", mask.shape, mask.dtype, mask.min(),
 613                      affine_matrix, mask.max(), interpolator)
 614         self._affine_matrix = self._adjust_affine_matrix(mask.shape[0], affine_matrix)
 615         self._interpolator = interpolator
 616         self.replace_mask(mask)

 618     def replace_mask(self, mask):
 619         """ Replace the existing :attr:`_mask` with the given mask.
 620 
 621         Parameters
 622         ----------
 623         mask: numpy.ndarray
 624             The mask that is to be added as output from :mod:`plugins.extract.mask`.
 625             It should be in the range 0.0 - 1.0 ideally with a ``dtype`` of ``float32``
 626         """
 627         mask = (cv2.resize(mask,
 628                            (self.stored_size, self.stored_size),
 629                            interpolation=cv2.INTER_AREA) * 255.0).astype("uint8")
 630         self._mask = compress(mask)

 632     def set_blur_and_threshold(self,
 633                                blur_kernel=0, blur_type="gaussian", blur_passes=1, threshold=0):
 634         """ Set the internal blur kernel and threshold amount for returned masks
 635 
 636         Parameters
 637         ----------
 638         blur_kernel: int, optional
 639             The kernel size, in pixels to apply gaussian blurring to the mask. Set to 0 for no
 640             blurring. Should be odd, if an even number is passed in (outside of 0) then it is
 641             rounded up to the next odd number. Default: 0
 642         blur_type: ["gaussian", "normalized"], optional
 643             The blur type to use. ``gaussian`` or ``normalized`` box filter. Default: ``gaussian``
 644         blur_passes: int, optional
 645             The number of passed to perform when blurring. Default: 1
 646         threshold: int, optional
 647             The threshold amount to minimize/maximize mask values to 0 and 100. Percentage value.
 648             Default: 0
 649         """
 650         logger.trace("blur_kernel: %s, threshold: %s", blur_kernel, threshold)
 651         if blur_type is not None:
 652             blur_kernel += 0 if blur_kernel == 0 or blur_kernel % 2 == 1 else 1
 653             self._blur["kernel"] = blur_kernel
 654             self._blur["type"] = blur_type
 655             self._blur["passes"] = blur_passes
 656         self._threshold = (threshold / 100.0) * 255.0
 657 
 658     def _adjust_affine_matrix(self, mask_size, affine_matrix):
 659         """ Adjust the affine matrix for the mask's storage size
 660 
 661         Parameters
 662         ----------
 663         mask_size: int
 664             The original size of the mask.
 665         affine_matrix: numpy.ndarray
 666             The affine matrix to transform the mask at original size to the parent frame.
 667 
 668         Returns
 669         -------
 670         affine_matrix: numpy,ndarray
 671             The affine matrix adjusted for the mask at its stored dimensions.
 672         """
 673         zoom = self.stored_size / mask_size
 674         zoom_mat = np.array([[zoom, 0, 0.], [0, zoom, 0.]])
 675         adjust_mat = np.dot(zoom_mat, np.concatenate((affine_matrix, np.array([[0., 0., 1.]]))))
 676         logger.trace("storage_size: %s, mask_size: %s, zoom: %s, original matrix: %s, "
 677                      "adjusted_matrix: %s", self.stored_size, mask_size, zoom, affine_matrix.shape,
 678                      adjust_mat.shape)
 679         return adjust_mat

 728 class BlurMask():
 729     """ Factory class to return the correct blur object for requested blur type.
 730 
 731     Works for square images only. Currently supports Gaussian and Normalized Box Filters.
 732 
 733     Parameters
 734     ----------
 735     blur_type: ["gaussian", "normalized"]
 736         The type of blur to use
 737     mask: :class:`numpy.ndarray`
 738         The mask to apply the blur to
 739     kernel: int or float
 740         Either the kernel size (in pixels) or the size of the kernel as a ratio of mask size
 741     is_ratio: bool, optional
 742         Whether the given :attr:`kernel` parameter is a ratio or not. If ``True`` then the
 743         actual kernel size will be calculated from the given ratio and the mask size. If
 744         ``False`` then the kernel size will be set directly from the :attr:`kernel` parameter.
 745         Default: ``False``
 746     passes: int, optional
 747         The number of passes to perform when blurring. Default: ``1``
 748 
 749     Example
 750     -------
 751     >>> print(mask.shape)
 752     (128, 128, 1)
 753     >>> new_mask = BlurMask("gaussian", mask, 3, is_ratio=False, passes=1).blurred
 754     >>> print(new_mask.shape)
 755     (128, 128, 1)
 756     """
 757     def __init__(self, blur_type, mask, kernel, is_ratio=False, passes=1):
 758         logger.trace("Initializing %s: (blur_type: '%s', mask_shape: %s, kernel: %s, "
 759                      "is_ratio: %s, passes: %s)", self.__class__.__name__, blur_type, mask.shape,
 760                      kernel, is_ratio, passes)
 761         self._blur_type = blur_type.lower()
 762         self._mask = mask
 763         self._passes = passes
 764         kernel_size = self._get_kernel_size(kernel, is_ratio)
 765         self._kernel_size = self._get_kernel_tuple(kernel_size)
 766         logger.trace("Initialized %s", self.__class__.__name__)
 767 
 768     @property
 769     def blurred(self):
 770         """ :class:`numpy.ndarray`: The final mask with blurring applied. """
 771         func = self._func_mapping[self._blur_type]
 772         kwargs = self._get_kwargs()
 773         blurred = self._mask
 774         for i in range(self._passes):
 775             ksize = int(kwargs["ksize"][0])
 776             logger.trace("Pass: %s, kernel_size: %s", i + 1, (ksize, ksize))
 777             blurred = func(blurred, **kwargs)
 778             ksize = int(round(ksize * self._multipass_factor))
 779             kwargs["ksize"] = self._get_kernel_tuple(ksize)
 780         blurred = blurred[..., None]
 781         logger.trace("Returning blurred mask. Shape: %s", blurred.shape)
 782         return blurred
 783 
 784     @property
 785     def _multipass_factor(self):
 786         """ For multiple passes the kernel must be scaled down. This value is
 787             different for box filter and gaussian """
 788         factor = dict(gaussian=0.8, normalized=0.5)
 789         return factor[self._blur_type]
 790 
 791     @property
 792     def _sigma(self):
 793         """ int: The Sigma for Gaussian Blur. Returns 0 to force calculation from kernel size. """
 794         return 0
 795 
 796     @property
 797     def _func_mapping(self):
 798         """ dict: :attr:`_blur_type` mapped to cv2 Function name. """
 799         return dict(gaussian=cv2.GaussianBlur,  # pylint: disable = no-member
 800                     normalized=cv2.blur)  # pylint: disable = no-member
 801 
 802     @property
 803     def _kwarg_requirements(self):
 804         """ dict: :attr:`_blur_type` mapped to cv2 Function required keyword arguments. """
 805         return dict(gaussian=["ksize", "sigmaX"],
 806                     normalized=["ksize"])
 807 
 808     @property
 809     def _kwarg_mapping(self):
 810         """ dict: cv2 function keyword arguments mapped to their parameters. """
 811         return dict(ksize=self._kernel_size,
 812                     sigmaX=self._sigma)
 813 
 814     def _get_kernel_size(self, kernel, is_ratio):
 815         """ Set the kernel size to absolute value.
 816 
 817         If :attr:`is_ratio` is ``True`` then the kernel size is calculated from the given ratio and
 818         the :attr:`_mask` size, otherwise the given kernel size is just returned.
 819 
 820         Parameters
 821         ----------
 822         kernel: int or float
 823             Either the kernel size (in pixels) or the size of the kernel as a ratio of mask size
 824         is_ratio: bool, optional
 825             Whether the given :attr:`kernel` parameter is a ratio or not. If ``True`` then the
 826             actual kernel size will be calculated from the given ratio and the mask size. If
 827             ``False`` then the kernel size will be set directly from the :attr:`kernel` parameter.
 828 
 829         Returns
 830         -------
 831         int
 832             The size (in pixels) of the blur kernel
 833         """
 834         if not is_ratio:
 835             return kernel
 836 
 837         mask_diameter = np.sqrt(np.sum(self._mask))
 838         radius = round(max(1., mask_diameter * kernel / 100.))
 839         kernel_size = int(radius * 2 + 1)
 840         logger.trace("kernel_size: %s", kernel_size)
 841         return kernel_size
 842 
 843     @staticmethod
 844     def _get_kernel_tuple(kernel_size):
 845         """ Make sure kernel_size is odd and return it as a tuple.
 846 
 847         Parameters
 848         ----------
 849         kernel_size: int
 850             The size in pixels of the blur kernel
 851 
 852         Returns
 853         -------
 854         tuple
 855             The kernel size as a tuple of ('int', 'int')
 856         """
 857         kernel_size += 1 if kernel_size % 2 == 0 else 0
 858         retval = (kernel_size, kernel_size)
 859         logger.trace(retval)
 860         return retval
 861 
 862     def _get_kwargs(self):
 863         """ dict: the valid keyword arguments for the requested :attr:`_blur_type` """
 864         retval = {kword: self._kwarg_mapping[kword]
 865                   for kword in self._kwarg_requirements[self._blur_type]}
 866         logger.trace("BlurMask kwargs: %s", retval)
 867         return retval
