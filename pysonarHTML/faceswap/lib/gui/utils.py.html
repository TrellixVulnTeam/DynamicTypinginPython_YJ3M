  84 class FileHandler():  # pylint:disable=too-few-public-methods
  85     """ Handles all GUI File Dialog actions and tasks.
  86 
  87     Parameters
  88     ----------
  89     handletype: ['open', 'save', 'filename', 'filename_multi', 'savefilename', 'context']
  90         The type of file dialog to return. `open` and `save` will perform the open and save actions
  91         and return the file. `filename` returns the filename from an `open` dialog.
  92         `filename_multi` allows for multi-selection of files and returns a list of files selected.
  93         `savefilename` returns the filename from a `save as` dialog. `context` is a context
  94         sensitive parameter that returns a certain dialog based on the current options
  95     filetype: ['default', 'alignments', 'config_project', 'config_task', 'config_all', 'csv', \
  96                'image', 'ini', 'state', 'log', 'video']
  97         The type of file that this dialog is for. `default` allows selection of any files. Other
  98         options limit the file type selection
  99     title: str, optional
 100         The title to display on the file dialog. If `None` then the default title will be used.
 101         Default: ``None``
 102     initialdir: str, optional
 103         The folder to initially open with the file dialog. If `None` then tkinter will decide.
 104         Default: ``None``
 105     command: str, optional
 106         Required for context handling file dialog, otherwise unused. Default: ``None``
 107     action: str, optional
 108         Required for context handling file dialog, otherwise unused. Default: ``None``
 109     variable: :class:`tkinter.StringVar`, optional
 110         Required for context handling file dialog, otherwise unused. The variable to associate
 111         with this file dialog. Default: ``None``
 112 
 113     Attributes
 114     ----------
 115     retfile: str or object
 116         The return value from the file dialog
 117 
 118     Example
 119     -------
 120     >>> handler = FileHandler('filename', 'video', title='Select a video...')
 121     >>> video_file = handler.retfile
 122     >>> print(video_file)
 123     '/path/to/selected/video.mp4'
 124     """
 125

 129                      "initialdir: '%s, 'command: '%s', action: '%s', variable: %s)",
 130                      self.__class__.__name__, handletype, filetype, title, initialdir, command,
 131                      action, variable)
 132         self._handletype = handletype
 133         self._defaults = self._set_defaults()
 134         self._kwargs = self._set_kwargs(title, initialdir, filetype, command, action, variable)
 135         self.retfile = getattr(self, "_{}".format(self._handletype.lower()))()
 136         logger.debug("Initialized %s", self.__class__.__name__)

 202     def _set_defaults(self):
 203         """ Set the default file type for the file dialog. Generally the first found file type
 204         will be used, but this is overridden if it is not appropriate.
 205

 219     def _set_kwargs(self, title, initialdir, filetype, command, action, variable=None):
 220         """ Generate the required kwargs for the requested file dialog browser.
 221 
 222         Returns
 223         -------
 224         dict:
 225             The key word arguments for the file dialog to be launched
 226         """
 227         logger.debug("Setting Kwargs: (title: %s, initialdir: %s, filetype: '%s', "
 228                      "command: '%s': action: '%s', variable: '%s')",
 229                      title, initialdir, filetype, command, action, variable)
 230         kwargs = dict()
 231         if self._handletype.lower() == "context":
 232             self._set_context_handletype(command, action, variable)
 233 
 234         if title is not None:
 235             kwargs["title"] = title
 236 
 237         if initialdir is not None:
 238             kwargs["initialdir"] = initialdir
 239 
 240         if self._handletype.lower() in (
 241                 "open", "save", "filename", "filename_multi", "savefilename"):
 242             kwargs["filetypes"] = self._filetypes[filetype]
 243             if self._defaults.get(filetype, None):
 244                 kwargs['defaultextension'] = self._defaults[filetype]
 245         if self._handletype.lower() == "save":
 246             kwargs["mode"] = "w"
 247         if self._handletype.lower() == "open":
 248             kwargs["mode"] = "r"
 249         logger.debug("Set Kwargs: %s", kwargs)
 250         return kwargs

 252     def _set_context_handletype(self, command, action, variable):
 253         """ Sets the correct handle type  based on context.
 254 
 255         Parameters
 256         ----------
 257         command: str
 258             The command that is being executed. Used to look up the context actions
 259         action: str
 260             The action that is being performed. Used to look up the correct file dialog
 261         variable: :class:`tkinter.StringVar`
 262             The variable associated with this file dialog
 263         """
 264         if self._contexts[command].get(variable, None) is not None:
 265             handletype = self._contexts[command][variable][action]
 266         else:
 267             handletype = self._contexts[command][action]
 268         logger.debug(handletype)
 269         self._handletype = handletype

 408     def delete_preview(self):
 409         """ Delete the preview files in the cache folder and reset the image cache.
 410 
 411         Should be called when terminating tasks, or when Faceswap starts up or shuts down.
 412         """
 413         logger.debug("Deleting previews")
 414         for item in os.listdir(self._pathpreview):
 415             if item.startswith(".gui_training_preview") and item.endswith(".jpg"):
 416                 fullitem = os.path.join(self._pathpreview, item)
 417                 logger.debug("Deleting: '%s'", fullitem)
 418                 os.remove(fullitem)
 419         for fname in self._previewcache["filenames"]:
 420             if os.path.basename(fname) == ".gui_preview.jpg":
 421                 logger.debug("Deleting: '%s'", fname)
 422                 try:
 423                     os.remove(fname)
 424                 except FileNotFoundError:
 425                     logger.debug("File does not exist: %s", fname)
 426         self._clear_image_cache()

 463     def load_latest_preview(self, thumbnail_size, frame_dims):
 464         """ Load the latest preview image for extract and convert.
 465 
 466         Retrieves the latest preview images from the faceswap output folder, resizes to thumbnails
 467         and lays out for display. Places the images into :attr:`previewoutput` for loading into
 468         the display panel.
 469 
 470         Parameters
 471         ----------
 472         thumbnail_size: int
 473             The size of each thumbnail that should be created
 474         frame_dims: tuple
 475             The (width (`int`), height (`int`)) of the display panel that will display the preview
 476         """
 477         logger.debug("Loading preview image: (thumbnail_size: %s, frame_dims: %s)",
 478                      thumbnail_size, frame_dims)
 479         image_files = self._get_images(self._pathoutput)
 480         gui_preview = os.path.join(self._pathoutput, ".gui_preview.jpg")
 481         if not image_files or (len(image_files) == 1 and gui_preview not in image_files):
 482             logger.debug("No preview to display")
 483             self._previewoutput = None
 484             return
 485         # Filter to just the gui_preview if it exists in folder output
 486         image_files = [gui_preview] if gui_preview in image_files else image_files
 487         logger.debug("Image Files: %s", len(image_files))
 488 
 489         image_files = self._get_newest_filenames(image_files)
 490         if not image_files:
 491             return
 492 
 493         self._load_images_to_cache(image_files, frame_dims, thumbnail_size)
 494         if image_files == [gui_preview]:
 495             # Delete the preview image so that the main scripts know to output another
 496             logger.debug("Deleting preview image")
 497             os.remove(image_files[0])
 498         show_image = self._place_previews(frame_dims)
 499         if not show_image:
 500             self._previewoutput = None
 501             return
 502         logger.debug("Displaying preview: %s", self._previewcache["filenames"])
 503         self._previewoutput = (show_image, ImageTk.PhotoImage(show_image))
 504 
 505     def _get_newest_filenames(self, image_files):
 506         """ Return image filenames that have been modified since the last check.
 507 
 508         Parameters
 509         ----------
 510         image_files: list
 511             The list of image files to check the modification date for
 512 
 513         Returns
 514         -------
 515         list:
 516             A list of images that have been modified since the last check
 517         """
 518         if self._previewcache["modified"] is None:
 519             retval = image_files
 520         else:
 521             retval = [fname for fname in image_files
 522                       if os.path.getmtime(fname) > self._previewcache["modified"]]
 523         if not retval:
 524             logger.debug("No new images in output folder")
 525         else:
 526             self._previewcache["modified"] = max([os.path.getmtime(img) for img in retval])
 527             logger.debug("Number new images: %s, Last Modified: %s",
 528                          len(retval), self._previewcache["modified"])
 529         return retval

 531     def _load_images_to_cache(self, image_files, frame_dims, thumbnail_size):
 532         """ Load preview images to the image cache.
 533 
 534         Load new images and append to cache, filtering the cache the number of thumbnails that will
 535         fit  inside the display panel.
 536 
 537         Parameters
 538         ----------
 539         image_files: list
 540             A list of new image files that have been modified since the last check
 541         frame_dims: tuple
 542             The (width (`int`), height (`int`)) of the display panel that will display the preview
 543         thumbnail_size: int
 544             The size of each thumbnail that should be created
 545         """
 546         logger.debug("Number image_files: %s, frame_dims: %s, thumbnail_size: %s",
 547                      len(image_files), frame_dims, thumbnail_size)
 548         num_images = (frame_dims[0] // thumbnail_size) * (frame_dims[1] // thumbnail_size)
 549         logger.debug("num_images: %s", num_images)
 550         if num_images == 0:
 551             return
 552         samples = list()
 553         start_idx = len(image_files) - num_images if len(image_files) > num_images else 0
 554         show_files = sorted(image_files, key=os.path.getctime)[start_idx:]
 555         for fname in show_files:
 556             img = Image.open(fname)
 557             width, height = img.size
 558             scaling = thumbnail_size / max(width, height)
 559             logger.debug("image width: %s, height: %s, scaling: %s", width, height, scaling)
 560             img = img.resize((int(width * scaling), int(height * scaling)))
 561             if img.size[0] != img.size[1]:
 562                 # Pad to square
 563                 new_img = Image.new("RGB", (thumbnail_size, thumbnail_size))
 564                 new_img.paste(img, ((thumbnail_size - img.size[0])//2,
 565                                     (thumbnail_size - img.size[1])//2))
 566                 img = new_img
 567             draw = ImageDraw.Draw(img)
 568             draw.rectangle(((0, 0), (thumbnail_size, thumbnail_size)), outline="#E5E5E5", width=1)
 569             samples.append(np.array(img))
 570         samples = np.array(samples)
 571         self._previewcache["filenames"] = (self._previewcache["filenames"] +
 572                                            show_files)[-num_images:]
 573         cache = self._previewcache["images"]
 574         if cache is None:
 575             logger.debug("Creating new cache")
 576             cache = samples[-num_images:]
 577         else:
 578             logger.debug("Appending to existing cache")
 579             cache = np.concatenate((cache, samples))[-num_images:]
 580         self._previewcache["images"] = cache
 581         logger.debug("Cache shape: %s", self._previewcache["images"].shape)
 582 
 583     def _place_previews(self, frame_dims):
 584         """ Format the preview thumbnails stored in the cache into a grid fitting the display
 585         panel.
 586 
 587         Parameters
 588         ----------
 589         frame_dims: tuple
 590             The (width (`int`), height (`int`)) of the display panel that will display the preview
 591 
 592         Returns
 593         :class:`PIL.Image`:
 594             The final preview display image
 595         """
 596         if self._previewcache.get("images", None) is None:
 597             logger.debug("No images in cache. Returning None")
 598             return None
 599         samples = self._previewcache["images"].copy()
 600         num_images, thumbnail_size = samples.shape[:2]
 601         if self._previewcache["placeholder"] is None:
 602             self._create_placeholder(thumbnail_size)
 603 
 604         logger.debug("num_images: %s, thumbnail_size: %s", num_images, thumbnail_size)
 605         cols, rows = frame_dims[0] // thumbnail_size, frame_dims[1] // thumbnail_size
 606         logger.debug("cols: %s, rows: %s", cols, rows)
 607         if cols == 0 or rows == 0:
 608             logger.debug("Cols or Rows is zero. No items to display")
 609             return None
 610         remainder = (cols * rows) - num_images
 611         if remainder != 0:
 612             logger.debug("Padding sample display. Remainder: %s", remainder)
 613             placeholder = np.concatenate([np.expand_dims(self._previewcache["placeholder"],
 614                                                          0)] * remainder)
 615             samples = np.concatenate((samples, placeholder))
 616 
 617         display = np.vstack([np.hstack(samples[row * cols: (row + 1) * cols])
 618                              for row in range(rows)])
 619         logger.debug("display shape: %s", display.shape)
 620         return Image.fromarray(display)
 621 
 622     def _create_placeholder(self, thumbnail_size):
 623         """ Create a placeholder image for when there are fewer thumbnails available
 624         than columns to display them.
 625 
 626         Parameters
 627         ----------
 628         thumbnail_size: int
 629             The size of the thumbnail that the placeholder should replicate
 630         """
 631         logger.debug("Creating placeholder. thumbnail_size: %s", thumbnail_size)
 632         placeholder = Image.new("RGB", (thumbnail_size, thumbnail_size))
 633         draw = ImageDraw.Draw(placeholder)
 634         draw.rectangle(((0, 0), (thumbnail_size, thumbnail_size)), outline="#E5E5E5", width=1)
 635         placeholder = np.array(placeholder)
 636         self._previewcache["placeholder"] = placeholder
 637         logger.debug("Created placeholder. shape: %s", placeholder.shape)

 639     def load_training_preview(self):
 640         """ Load the training preview images.
 641 
 642         Reads the training image currently stored in the cache folder and loads them to
 643         :attr:`previewtrain` for retrieval in the GUI.
 644         """
 645         logger.debug("Loading Training preview images")
 646         image_files = self._get_images(self._pathpreview)
 647         modified = None
 648         if not image_files:
 649             logger.debug("No preview to display")
 650             self._previewtrain = dict()
 651             return
 652         for img in image_files:
 653             modified = os.path.getmtime(img) if modified is None else modified
 654             name = os.path.basename(img)
 655             name = os.path.splitext(name)[0]
 656             name = name[name.rfind("_") + 1:].title()
 657             try:
 658                 logger.debug("Displaying preview: '%s'", img)
 659                 size = self._get_current_size(name)
 660                 self._previewtrain[name] = [Image.open(img), None, modified]
 661                 self.resize_image(name, size)
 662                 self._errcount = 0
 663             except ValueError:
 664                 # This is probably an error reading the file whilst it's
 665                 # being saved  so ignore it for now and only pick up if
 666                 # there have been multiple consecutive fails
 667                 logger.warning("Unable to display preview: (image: '%s', attempt: %s)",
 668                                img, self._errcount)
 669                 if self._errcount < 10:
 670                     self._errcount += 1
 671                 else:
 672                     logger.error("Error reading the preview file for '%s'", img)
 673                     print("Error reading the preview file for {}".format(name))
 674                     self._previewtrain[name] = None

 676     def _get_current_size(self, name):
 677         """ Return the size of the currently displayed training preview image.
 678 
 679         Parameters
 680         ----------
 681         name: str
 682             The name of the training image to get the size for
 683 
 684         Returns
 685         -------
 686         width: int
 687             The width of the training image
 688         height: int
 689             The height of the training image
 690         """
 691         logger.debug("Getting size: '%s'", name)
 692         if not self._previewtrain.get(name, None):
 693             return None
 694         img = self._previewtrain[name][1]
 695         if not img:
 696             return None
 697         logger.debug("Got size: (name: '%s', width: '%s', height: '%s')",
 698                      name, img.width(), img.height())
 699         return img.width(), img.height()

 701     def resize_image(self, name, frame_dims):
 702         """ Resize the training preview image based on the passed in frame size.
 703 
 704         If the canvas that holds the preview image changes, update the image size
 705         to fit the new canvas and refresh :attr:`previewtrain`.
 706 
 707         Parameters
 708         ----------
 709         name: str
 710             The name of the training image to be resized
 711         frame_dims: tuple
 712             The (width (`int`), height (`int`)) of the display panel that will display the preview
 713         """
 714         logger.debug("Resizing image: (name: '%s', frame_dims: %s", name, frame_dims)
 715         displayimg = self._previewtrain[name][0]
 716         if frame_dims:
 717             frameratio = float(frame_dims[0]) / float(frame_dims[1])
 718             imgratio = float(displayimg.size[0]) / float(displayimg.size[1])
 719 
 720             if frameratio <= imgratio:
 721                 scale = frame_dims[0] / float(displayimg.size[0])
 722                 size = (frame_dims[0], int(displayimg.size[1] * scale))
 723             else:
 724                 scale = frame_dims[1] / float(displayimg.size[1])
 725                 size = (int(displayimg.size[0] * scale), frame_dims[1])
 726             logger.debug("Scaling: (scale: %s, size: %s", scale, size)
 727 
 728             # Hacky fix to force a reload if it happens to find corrupted
 729             # data, probably due to reading the image whilst it is partially
 730             # saved. If it continues to fail, then eventually raise.
 731             for i in range(0, 1000):
 732                 try:
 733                     displayimg = displayimg.resize(size, Image.ANTIALIAS)
 734                 except OSError:
 735                     if i == 999:
 736                         raise
 737                     continue
 738                 break
 739         self._previewtrain[name][1] = ImageTk.PhotoImage(displayimg)

1077 class LongRunningTask(Thread):
1078     """ Runs long running tasks in a background thread to prevent the GUI from becoming
1079     unresponsive.
1080 
1081     This is sub-classed from :class:`Threading.Thread` so check documentation there for base
1082     parameters. Additional parameters listed below.
1083 
1084     Parameters
1085     ----------
1086     widget: tkinter object, optional
1087         The widget that this :class:`LongRunningTask` is associated with. Used for setting the busy
1088         cursor in the correct location. Default: ``None``.
1089     """
1090     def __init__(self, group=None, target=None, name=None, args=(), kwargs=None, *, daemon=True,
1091                  widget=None):
1092         logger.debug("Initializing %s: (group: %s, target: %s, name: %s, args: %s, kwargs: %s, "
1093                      "daemon: %s)", self.__class__.__name__, group, target, name, args, kwargs,
1094                      daemon)
1095         super().__init__(group=group, target=target, name=name, args=args, kwargs=kwargs,
1096                          daemon=daemon)
1097         self.err = None
1098         self._widget = widget
1099         self._config = get_config()
1100         self._config.set_cursor_busy(widget=self._widget)
1101         self._complete = Event()
1102         self._queue = Queue()
1103         logger.debug("Initialized %s", self.__class__.__name__,)
1104 
1105     @property
1106     def complete(self):
1107         """ :class:`threading.Event`:  Event is set if the thread has completed its task,
1108         otherwise it is unset.
1109         """
1110         return self._complete
1111 
1112     def run(self):
1113         """ Commence the given task in a background thread. """
1114         try:
1115             if self._target:
1116                 retval = self._target(*self._args, **self._kwargs)
1117                 self._queue.put(retval)
1118         except Exception:  # pylint: disable=broad-except
1119             self.err = sys.exc_info()
1120             logger.debug("Error in thread (%s): %s", self._name,
1121                          self.err[1].with_traceback(self.err[2]))
1122         finally:
1123             self._complete.set()
1124             # Avoid a ref-cycle if the thread is running a function with
1125             # an argument that has a member that points to the thread.
1126             del self._target, self._args, self._kwargs
1127 
1128     def get_result(self):
1129         """ Return the result from the given task.
1130 
1131         Returns
1132         -------
1133         varies:
1134             The result of the thread will depend on the given task. If a call is made to
1135             :func:`get_result` prior to the thread completing its task then ``None`` will be
1136             returned
1137         """
1138         if not self._complete.is_set():
1139             logger.warning("Aborting attempt to retrieve result from a LongRunningTask that is "
1140                            "still running")
1141             return None
1142         if self.err:
1143             logger.debug("Error caught in thread")
1144             self._config.set_cursor_default(widget=self._widget)
1145             raise self.err[1].with_traceback(self.err[2])
1146 
1147         logger.debug("Getting result from thread")
1148         retval = self._queue.get()
1149         logger.debug("Got result from thread")
1150         self._config.set_cursor_default(widget=self._widget)
1151         return retval
