  14 class DisplayPage(ttk.Frame):  # pylint: disable=too-many-ancestors
  15     """ Parent frame holder for each tab.
  16         Defines uniform structure for each tab to inherit from """
  17     def __init__(self, parent, tabname, helptext):
  18         logger.debug("Initializing %s: (tabname: '%s', helptext: %s)",
  19                      self.__class__.__name__, tabname, helptext)
  20         ttk.Frame.__init__(self, parent)
  21         self.pack(fill=tk.BOTH, side=tk.TOP, anchor=tk.NW)
  22 
  23         self.runningtask = parent.runningtask
  24         self.helptext = helptext
  25         self.tabname = tabname
  26 
  27         self.vars = {"info": tk.StringVar()}
  28         self.add_optional_vars(self.set_vars())
  29 
  30         self.subnotebook = self.add_subnotebook()
  31         self.optsframe = self.add_options_frame()
  32         self.add_options_info()
  33 
  34         self.add_frame_separator()
  35         self.set_mainframe_single_tab_style()
  36         parent.add(self, text=self.tabname.title())
  37         logger.debug("Initialized %s", self.__class__.__name__,)
  38 
  39     def add_optional_vars(self, varsdict):
  40         """ Add page specific variables """
  41         if isinstance(varsdict, dict):
  42             for key, val in varsdict.items():
  43                 logger.debug("Adding: (%s: %s)", key, val)
  44                 self.vars[key] = val
  45 
  46     @staticmethod
  47     def set_vars():
  48         """ Override to return a dict of page specific variables """
  49         return dict()
  50 
  51     def add_subnotebook(self):
  52         """ Add the main frame notebook """
  53         logger.debug("Adding subnotebook")
  54         notebook = ttk.Notebook(self)
  55         notebook.pack(side=tk.TOP, anchor=tk.NW, fill=tk.BOTH, expand=True)
  56         return notebook
  57 
  58     def add_options_frame(self):
  59         """ Add the display tab options """
  60         logger.debug("Adding options frame")
  61         optsframe = ttk.Frame(self)
  62         optsframe.pack(side=tk.BOTTOM, padx=5, pady=5, fill=tk.X)
  63         return optsframe
  64 
  65     def add_options_info(self):
  66         """ Add the info bar """
  67         logger.debug("Adding options info")
  68         lblinfo = ttk.Label(self.optsframe,
  69                             textvariable=self.vars["info"],
  70                             anchor=tk.W,
  71                             width=70)
  72         lblinfo.pack(side=tk.LEFT, padx=5, pady=5, anchor=tk.W)
  73 
  74     def set_info(self, msg):
  75         """ Set the info message """
  76         logger.debug("Setting info: %s", msg)
  77         self.vars["info"].set(msg)
  78 
  79     def add_frame_separator(self):
  80         """ Add a separator between top and bottom frames """
  81         logger.debug("Adding frame seperator")
  82         sep = ttk.Frame(self, height=2, relief=tk.RIDGE)
  83         sep.pack(fill=tk.X, pady=(5, 0), side=tk.BOTTOM)

  85     @staticmethod
  86     def set_mainframe_single_tab_style():
  87         """ Configure ttk notebook style to represent a single frame """
  88         logger.debug("Setting main frame single tab style")
  89         nbstyle = ttk.Style()
  90         nbstyle.configure("single.TNotebook", borderwidth=0)
  91         nbstyle.layout("single.TNotebook.Tab", [])

  93     def subnotebook_add_page(self, tabtitle, widget=None):
  94         """ Add a page to the sub notebook """
  95         logger.debug("Adding subnotebook page: %s", tabtitle)
  96         frame = widget if widget else ttk.Frame(self.subnotebook)
  97         frame.pack(padx=5, pady=5, fill=tk.BOTH, expand=True)
  98         self.subnotebook.add(frame, text=tabtitle)
  99         self.subnotebook_configure()
 100         return frame

 102     def subnotebook_configure(self):
 103         """ Configure notebook to display or hide tabs """
 104         if len(self.subnotebook.children) == 1:
 105             logger.debug("Setting single page style")
 106             self.subnotebook.configure(style="single.TNotebook")
 107         else:
 108             logger.debug("Setting multi page style")
 109             self.subnotebook.configure(style="TNotebook")
 110 
 111     def subnotebook_hide(self):
 112         """ Hide the subnotebook. Used for hiding
 113             Optional displays """
 114         if self.subnotebook and self.subnotebook.winfo_ismapped():
 115             logger.debug("Hiding subnotebook")
 116             self.subnotebook.pack_forget()
 117             self.subnotebook.destroy()
 118             self.subnotebook = None
 119 
 120     def subnotebook_show(self):
 121         """ Show subnotebook. Used for displaying
 122             Optional displays  """
 123         if not self.subnotebook:
 124             logger.debug("Showing subnotebook")
 125             self.subnotebook = self.add_subnotebook()
 126 
 127     def subnotebook_get_widgets(self):
 128         """ Return each widget that sits within each
 129             subnotebook frame """
 130         logger.debug("Getting subnotebook widgets")
 131         for child in self.subnotebook.winfo_children():
 132             for widget in child.winfo_children():
 133                 yield widget
 134 
 135     def subnotebook_get_titles_ids(self):
 136         """ Return tabs ids and titles """
 137         tabs = dict()
 138         for tab_id in range(0, self.subnotebook.index("end")):
 139             tabs[self.subnotebook.tab(tab_id, "text")] = tab_id
 140         logger.debug(tabs)
 141         return tabs
 142 
 143     def subnotebook_page_from_id(self, tab_id):
 144         """ Return subnotebook tab widget from it's ID """
 145         tab_name = self.subnotebook.tabs()[tab_id].split(".")[-1]
 146         logger.debug(tab_name)
 147         return self.subnotebook.children[tab_name]

 168     @staticmethod
 169     def set_vars():
 170         """ Analysis specific vars """
 171         enabled = tk.BooleanVar()
 172         enabled.set(True)
 173 
 174         ready = tk.BooleanVar()
 175         ready.set(False)
 176 
 177         modified = tk.DoubleVar()
 178         modified.set(None)
 179 
 180         tk_vars = {"enabled": enabled,
 181                    "ready": ready,
 182                    "modified": modified}
 183         logger.debug(tk_vars)
 184         return tk_vars
 185 
 186     # INFO LABEL
 187     def set_info_text(self):
 188         """ Set waiting for display text """
 189         if not self.vars["enabled"].get():
 190             msg = "{} disabled".format(self.tabname.title())
 191         elif self.vars["enabled"].get() and not self.vars["ready"].get():
 192             msg = "Waiting for {}...".format(self.tabname)
 193         else:
 194             msg = "Displaying {}".format(self.tabname)
 195         logger.debug(msg)
 196         self.set_info(msg)

 204     def add_option_save(self):
 205         """ Add save button to save page output to file """
 206         logger.debug("Adding save option")

 215     def add_option_enable(self):

 228         """ Save items. Override for display specific saving """

 231     def on_chkenable_change(self):
 232         """ Update the display immediately on a checkbutton change """
 233         logger.debug("Enabled checkbox changed")
 234         if self.vars["enabled"].get():
 235             self.subnotebook_show()
 236         else:
 237             self.subnotebook_hide()
 238         self.set_info_text()
 239 
 240     def update_page(self, waittime):
 241         """ Update the latest preview item """
 242         if not self.runningtask.get():
 243             return
 244         if self.vars["enabled"].get():
 245             logger.trace("Updating page")
 246             self.display_item_set()
 247             self.load_display()
 248         self.after(waittime, lambda t=waittime: self.update_page(t))
 249 
 250     def display_item_set(self):
 251         """ Override for display specific loading """
 252         raise NotImplementedError()
 253 
 254     def load_display(self):
 255         """ Load the display """
 256         if not self.display_item:
 257             return
 258         logger.debug("Loading display")
 259         self.display_item_process()
 260         self.vars["ready"].set(True)
 261         self.set_info_text()
 262 
 263     def display_item_process(self):
 264         """ Override for display specific loading """
 265         raise NotImplementedError()

 267     def close(self):
 268         """ Called when the parent notebook is shutting down
 269             Children must be destroyed as forget only hides display
 270             Override for page specific shutdown """
 271         for child in self.winfo_children():
 272             logger.debug("Destroying child: %s", child)
 273             child.destroy()
