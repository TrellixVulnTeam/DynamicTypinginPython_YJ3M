  15 class CommandNotebook(ttk.Notebook):  # pylint:disable=too-many-ancestors
  16     """ Frame to hold each individual tab of the command notebook """
  17 
  18     def __init__(self, parent):
  19         logger.debug("Initializing %s: (parent: %s)", self.__class__.__name__, parent)
  20         self.actionbtns = dict()
  21         super().__init__(parent)
  22         parent.add(self)
  23 
  24         self.tools_notebook = ToolsNotebook(self)
  25         self.set_running_task_trace()
  26         self.build_tabs()
  27         self.modified_vars = self._set_modified_vars()
  28         get_config().set_command_notebook(self)
  29         logger.debug("Initialized %s", self.__class__.__name__)
  30 
  31     @property
  32     def tab_names(self):
  33         """ dict: Command tab titles with their IDs """
  34         return {self.tab(tab_id, "text").lower(): tab_id
  35                 for tab_id in range(0, self.index("end"))}
  36 
  37     @property
  38     def tools_tab_names(self):
  39         """ dict: Tools tab titles with their IDs """
  40         return {self.tools_notebook.tab(tab_id, "text").lower(): tab_id
  41                 for tab_id in range(0, self.tools_notebook.index("end"))}
  42 
  43     def set_running_task_trace(self):
  44         """ Set trigger action for the running task
  45             to change the action buttons text and command """
  46         logger.debug("Set running trace")
  47         tk_vars = get_config().tk_vars
  48         tk_vars["runningtask"].trace("w", self.change_action_button)
  49 
  50     def build_tabs(self):
  51         """ Build the tabs for the relevant command """
  52         logger.debug("Build Tabs")
  53         cli_opts = get_config().cli_opts
  54         for category in cli_opts.categories:
  55             book = self.tools_notebook if category == "tools" else self
  56             cmdlist = cli_opts.commands[category]
  57             for command in cmdlist:
  58                 title = command.title()
  59                 commandtab = CommandTab(book, category, command)
  60                 book.add(commandtab, text=title)
  61         self.add(self.tools_notebook, text="Tools")
  62         logger.debug("Built Tabs")
  63 
  64     def change_action_button(self, *args):
  65         """ Change the action button to relevant control """
  66         logger.debug("Update Action Buttons: (args: %s", args)
  67         tk_vars = get_config().tk_vars
  68 
  69         for cmd, action in self.actionbtns.items():
  70             btnact = action
  71             if tk_vars["runningtask"].get():
  72                 ttl = " Stop"
  73                 img = get_images().icons["stop"]
  74                 hlp = "Exit the running process"
  75             else:
  76                 ttl = " {}".format(cmd.title())
  77                 img = get_images().icons["start"]
  78                 hlp = "Run the {} script".format(cmd.title())
  79             logger.debug("Updated Action Button: '%s'", ttl)
  80             btnact.config(text=ttl, image=img)
  81             Tooltip(btnact, text=hlp, wraplength=200)
  82 
  83     def _set_modified_vars(self):
  84         """ Set the tkinter variable for each tab to indicate whether contents
  85         have been modified """
  86         tkvars = dict()
  87         for tab in self.tab_names:
  88             if tab == "tools":
  89                 for ttab in self.tools_tab_names:
  90                     var = tk.BooleanVar()
  91                     var.set(False)
  92                     tkvars[ttab] = var
  93                 continue
  94             var = tk.BooleanVar()
  95             var.set(False)
  96             tkvars[tab] = var
  97         logger.debug("Set modified vars: %s", tkvars)
  98         return tkvars

 108 class CommandTab(ttk.Frame):  # pylint:disable=too-many-ancestors
 109     """ Frame to hold each individual tab of the command notebook """
 110 
 111     def __init__(self, parent, category, command):
 112         logger.debug("Initializing %s: (category: '%s', command: '%s')",
 113                      self.__class__.__name__, category, command)
 114         super().__init__(parent, name="tab_{}".format(command.lower()))
 115 
 116         self.category = category
 117         self.actionbtns = parent.actionbtns
 118         self.command = command
 119 
 120         self.build_tab()
 121         logger.debug("Initialized %s", self.__class__.__name__)
 122 
 123     def build_tab(self):
 124         """ Build the tab """
 125         logger.debug("Build Tab: '%s'", self.command)
 126         options = get_config().cli_opts.opts[self.command]
 127         cp_opts = [val["cpanel_option"] for key, val in options.items() if key != "helptext"]
 128         ControlPanel(self,
 129                      cp_opts,
 130                      label_width=16,
 131                      option_columns=3,
 132                      columns=1,
 133                      header_text=options.get("helptext", None))
 134         self.add_frame_separator()
 135         ActionFrame(self)
 136         logger.debug("Built Tab: '%s'", self.command)
 137 
 138     def add_frame_separator(self):
 139         """ Add a separator between top and bottom frames """
 140         logger.debug("Add frame seperator")
 141         sep = ttk.Frame(self, height=2, relief=tk.RIDGE)
 142         sep.pack(fill=tk.X, pady=(5, 0), side=tk.TOP)
 143         logger.debug("Added frame seperator")

 146 class ActionFrame(ttk.Frame):  # pylint:disable=too-many-ancestors
 147     """Action Frame - Displays action controls for the command tab """
 148 
 149     def __init__(self, parent):
 150         logger.debug("Initializing %s: (command: '%s')", self.__class__.__name__, parent.command)
 151         super().__init__(parent)
 152         self.pack(fill=tk.BOTH, padx=5, pady=5, side=tk.BOTTOM, anchor=tk.N)
 153 
 154         self.command = parent.command
 155         self.title = self.command.title()
 156 
 157         self.add_action_button(parent.category,
 158                                parent.actionbtns)
 159         logger.debug("Initialized %s", self.__class__.__name__)
 160 
 161     def add_action_button(self, category, actionbtns):
 162         """ Add the action buttons for page """
 163         logger.debug("Add action buttons: '%s'", self.title)
 164         actframe = ttk.Frame(self)
 165         actframe.pack(fill=tk.X, side=tk.RIGHT)
 166 
 167         tk_vars = get_config().tk_vars
 168         var_value = "{},{}".format(category, self.command)
 169 
 170         btngen = ttk.Button(actframe,
 171                             image=get_images().icons["generate"],
 172                             text=" Generate",
 173                             compound=tk.LEFT,
 174                             width=14,
 175                             command=lambda: tk_vars["generate"].set(var_value))
 176         btngen.pack(side=tk.LEFT, padx=5)
 177         Tooltip(btngen,
 178                 text="Output command line options to the console",
 179                 wraplength=200)
 180 
 181         btnact = ttk.Button(actframe,
 182                             image=get_images().icons["start"],
 183                             text=" {}".format(self.title),
 184                             compound=tk.LEFT,
 185                             width=14,
 186                             command=lambda: tk_vars["action"].set(var_value))
 187         btnact.pack(side=tk.LEFT, fill=tk.X, expand=True)
 188         Tooltip(btnact,
 189                 text="Run the {} script".format(self.title),
 190                 wraplength=200)
 191         actionbtns[self.command] = btnact
 192 
 193         logger.debug("Added action buttons: '%s'", self.title)
