  49 def set_slider_rounding(value, var, d_type, round_to, min_max):
  50     """ Set the value of sliders underlying variable based on their datatype,
  51     rounding value and min/max.
  52 
  53     Parameters
  54     ----------
  55     var: tkinter.Var
  56         The variable to set the value for
  57     d_type: [:class:`int`, :class:`float`]
  58         The type of value that is stored in :attr:`var`
  59     round_to: int
  60         If :attr:`dtype` is :class:`float` then this is the decimal place rounding for :attr:`var`.
  61         If :attr:`dtype` is :class:`int` then this is the number of steps between each increment
  62         for :attr:`var`
  63     min_max: tuple (`int`, `int`)
  64         The (``min``, ``max``) values that this slider accepts
  65     """
  66     if d_type == float:
  67         var.set(round(float(value), round_to))
  68     else:
  69         steps = range(min_max[0], min_max[1] + round_to, round_to)
  70         value = min(steps, key=lambda x: abs(x - int(float(value))))
  71         var.set(value)

 214     def get(self):
 215         """ Return the value from the tk_var
 216 
 217         Notes
 218         -----
 219         tk variables don't like empty values if it's not a stringVar. This seems to be pretty
 220         much the only reason that a get() call would fail, so replace any numerical variable
 221         with it's numerical zero equivalent on a TCL Error. Only impacts variables linked
 222         to Entry widgets.
 223         """
 224         try:
 225             val = self.tk_var.get()
 226         except TclError:
 227             if isinstance(self.tk_var, tk.IntVar):
 228                 val = 0
 229             elif isinstance(self.tk_var, tk.DoubleVar):
 230                 val = 0.0
 231             else:
 232                 raise
 233         return val

 250     def get_control(self):
 251         """ Set the correct control type based on the datatype or for this option """
 252         if self.choices and self.is_radio:
 253             control = "radio"
 254         elif self.choices and self.is_multi_option:
 255             control = "multi"
 256         elif self.choices:
 257             control = ttk.Combobox
 258         elif self.dtype == bool:
 259             control = ttk.Checkbutton
 260         elif self.dtype in (int, float):
 261             control = "scale"
 262         else:
 263             control = ttk.Entry
 264         logger.debug("Setting control '%s' to %s", self.title, control)
 265         return control

 267     def get_tk_var(self, track_modified):
 268         """ Correct variable type for control """
 269         if self.dtype == bool:
 270             var = tk.BooleanVar()
 271         elif self.dtype == int:
 272             var = tk.IntVar()
 273         elif self.dtype == float:
 274             var = tk.DoubleVar()
 275         else:
 276             var = tk.StringVar()
 277         logger.debug("Setting tk variable: (name: '%s', dtype: %s, tk_var: %s)",
 278                      self.name, self.dtype, var)
 279         if track_modified and self._command is not None:
 280             logger.debug("Tracking variable modification: %s", self.name)
 281             var.trace("w",
 282                       lambda name, index, mode, cmd=self._command: self._modified_callback(cmd))
 283 
 284         if track_modified and self._command in ("train", "convert") and self.title == "Model Dir":
 285             var.trace("w", lambda name, index, mode, v=var: self._model_callback(v))
 286 
 287         return var

 503 class AutoFillContainer():
 504     """ A container object that auto-fills columns """
 505     def __init__(self, parent, initial_columns, max_columns):
 506         logger.debug("Initializing: %s: (parent: %s, initial_columns: %s, max_columns: %s)",
 507                      self.__class__.__name__, parent, initial_columns, max_columns)
 508         self.max_columns = max_columns
 509         self.columns = initial_columns
 510         self.parent = parent
 511 #        self.columns = min(columns, self.max_columns)
 512         self.single_column_width = self.scale_column_width(288, 9)
 513         self.max_width = self.max_columns * self.single_column_width
 514         self._items = 0
 515         self._idx = 0
 516         self._widget_config = []  # Master list of all children in order
 517         self.subframes = self.set_subframes()
 518         logger.debug("Initialized: %s", self.__class__.__name__)
 519 
 520     @staticmethod
 521     def scale_column_width(original_size, original_fontsize):
 522         """ Scale the column width based on selected font size """
 523         font_size = get_config().user_config_dict["font_size"]
 524         if font_size == original_fontsize:
 525             return original_size
 526         scale = 1 + (((font_size / original_fontsize) - 1) / 2)
 527         retval = round(original_size * scale)
 528         logger.debug("scaled column width: (old_width: %s, scale: %s, new_width:%s)",
 529                      original_size, scale, retval)
 530         return retval
 531 
 532     @property
 533     def items(self):
 534         """ Returns the number of items held in this container """
 535         return self._items
 536 
 537     @property
 538     def subframe(self):
 539         """ Returns the next sub-frame to be populated """
 540         frame = self.subframes[self._idx]
 541         next_idx = self._idx + 1 if self._idx + 1 < self.columns else 0
 542         logger.debug("current_idx: %s, next_idx: %s", self._idx, next_idx)
 543         self._idx = next_idx
 544         self._items += 1
 545         return frame
 546 
 547     def set_subframes(self):
 548         """ Set a sub-frame for each possible column """
 549         subframes = []
 550         for idx in range(self.max_columns):
 551             name = "af_subframe_{}".format(idx)
 552             subframe = ttk.Frame(self.parent, name=name)
 553             if idx < self.columns:
 554                 # Only pack visible columns
 555                 subframe.pack(padx=5, pady=5, side=tk.LEFT, anchor=tk.N, expand=True, fill=tk.X)
 556             subframes.append(subframe)
 557             logger.debug("Added subframe: %s", name)
 558         return subframes

 563             return
 564 
 565         new_columns = min(self.max_columns, max(1, width // self.single_column_width))
 566         logger.debug("Rearranging columns: (width: %s, old_columns: %s, new_columns: %s)",
 567                      width, self.columns, new_columns)
 568         self.columns = new_columns
 569         if not self._widget_config:
 570             self.compile_widget_config()
 571         self.destroy_children()
 572         self.repack_columns()
 573         # Reset counters
 574         self._items = 0
 575         self._idx = 0
 576         self.pack_widget_clones(self._widget_config)
 577 
 578     def validate(self, width):
 579         """ Validate that passed in width should trigger column re-arranging """
 580         if ((width < self.single_column_width and self.columns == 1) or
 581                 (width > self.max_width and self.columns == self.max_columns)):
 582             logger.debug("width outside min/max thresholds: (min: %s, width: %s, max: %s)",
 583                          self.single_column_width, width, self.max_width)
 584             return False
 585         range_min = self.columns * self.single_column_width
 586         range_max = (self.columns + 1) * self.single_column_width
 587         if range_min < width < range_max:
 588             logger.debug("width outside next step refresh threshold: (step down: %s, width: %s,"
 589                          "step up: %s)", range_min, width, range_max)
 590             return False
 591         return True

 593     def compile_widget_config(self):
 594         """ Compile all children recursively in correct order if not already compiled """
 595         zipped = zip_longest(*(subframe.winfo_children() for subframe in self.subframes))
 596         children = [child for group in zipped for child in group if child is not None]
 597         self._widget_config = [{"class": child.__class__,
 598                                 "id": str(child),
 599                                 "tooltip": _RECREATE_OBJECTS["tooltips"].get(str(child), None),
 600                                 "rc_menu": _RECREATE_OBJECTS["contextmenus"].get(str(child), None),
 601                                 "pack_info": self.pack_config_cleaner(child),
 602                                 "name": child.winfo_name(),
 603                                 "config": self.config_cleaner(child),
 604                                 "children": self.get_all_children_config(child, []),
 605                                 # Some children have custom kwargs, so keep dicts in sync
 606                                 "custom_kwargs": dict()}
 607                                for idx, child in enumerate(children)]
 608         logger.debug("Compiled AutoFillContainer children: %s", self._widget_config)

 610     def get_all_children_config(self, widget, child_list):
 611         """ Return all children, recursively, of given widget """
 612         for child in widget.winfo_children():
 613             if child.winfo_ismapped():
 614                 id_ = str(child)
 615                 if child.__class__.__name__ == "MultiOption":
 616                     # MultiOption checkbox groups are a custom object with additional parameter
 617                     # requirements.
 618                     custom_kwargs = dict(
 619                         value=child._value,  # pylint:disable=protected-access
 620                         variable=child._master_variable)  # pylint:disable=protected-access
 621                 else:
 622                     custom_kwargs = dict()
 623 
 624                 child_list.append({
 625                     "class": child.__class__,
 626                     "id": id_,
 627                     "tooltip": _RECREATE_OBJECTS["tooltips"].get(id_, None),
 628                     "rc_menu": _RECREATE_OBJECTS["contextmenus"].get(str(id_), None),
 629                     "pack_info": self.pack_config_cleaner(child),
 630                     "name": child.winfo_name(),
 631                     "config": self.config_cleaner(child),
 632                     "parent": child.winfo_parent(),
 633                     "custom_kwargs": custom_kwargs})
 634             self.get_all_children_config(child, child_list)
 635         return child_list

 637     @staticmethod
 638     def config_cleaner(widget):
 639         """ Some options don't like to be copied, so this returns a cleaned
 640             configuration from a widget
 641             We use config() instead of configure() because some items (ttk Scale) do
 642             not populate configure()"""
 643         new_config = dict()
 644         for key in widget.config():
 645             if key == "class":
 646                 continue
 647             val = widget.cget(key)
 648             # Some keys default to "" but tkinter doesn't like to set config to this value
 649             # so skip them to use default value.
 650             if key in ("anchor", "justify", "compound") and val == "":
 651                 continue
 652             val = str(val) if isinstance(val, Tcl_Obj) else val
 653             # Return correct command from master command dict
 654             val = _RECREATE_OBJECTS["commands"][val] if key == "command" and val != "" else val
 655             new_config[key] = val
 656         return new_config

 659     def pack_config_cleaner(widget):
 660         """ Some options don't like to be copied, so this returns a cleaned
 661             configuration from a widget """
 662         return {key: val for key, val in widget.pack_info().items() if key != "in"}

 666         for subframe in self.subframes:
 667             for child in subframe.winfo_children():

 670     def repack_columns(self):
 671         """ Repack or unpack columns based on display columns """
 672         for idx, subframe in enumerate(self.subframes):
 673             logger.trace("Processing subframe: %s", subframe)
 674             if idx < self.columns and not subframe.winfo_ismapped():
 675                 logger.trace("Packing subframe: %s", subframe)
 676                 subframe.pack(padx=5, pady=5, side=tk.LEFT, anchor=tk.N, expand=True, fill=tk.X)
 677             elif idx >= self.columns and subframe.winfo_ismapped():
 678                 logger.trace("Forgetting subframe: %s", subframe)
 679                 subframe.pack_forget()

 681     def pack_widget_clones(self, widget_dicts, old_children=None, new_children=None):
 682         """ Widgets cannot be given a new parent so we need to clone
 683             them and then pack the new widget """
 684         for widget_dict in widget_dicts:
 685             logger.debug("Cloning widget: %s", widget_dict)
 686             old_children = [] if old_children is None else old_children
 687             new_children = [] if new_children is None else new_children
 688             if widget_dict.get("parent", None) is not None:
 689                 parent = new_children[old_children.index(widget_dict["parent"])]
 690                 logger.trace("old parent: '%s', new_parent: '%s'", widget_dict["parent"], parent)
 691             else:
 692                 # Get the next sub-frame if this doesn't have a logged parent
 693                 parent = self.subframe
 694             clone = widget_dict["class"](parent,
 695                                          name=widget_dict["name"],
 696                                          **widget_dict["custom_kwargs"])
 697             if widget_dict["config"] is not None:
 698                 clone.configure(**widget_dict["config"])
 699             if widget_dict["tooltip"] is not None:
 700                 Tooltip(clone, **widget_dict["tooltip"])
 701             rc_menu = widget_dict["rc_menu"]
 702             if rc_menu is not None:
 703                 # Re-initialize for new widget and bind
 704                 rc_menu.__init__(widget=clone)
 705                 rc_menu.cm_bind()
 706             clone.pack(**widget_dict["pack_info"])
 707             old_children.append(widget_dict["id"])
 708             new_children.append(clone)
 709             if widget_dict.get("children", None) is not None:
 710                 self.pack_widget_clones(widget_dict["children"], old_children, new_children)

 807     def _multi_option_control(self, option_type):
 808         """ Create a group of buttons for single or multi-select
 809 
 810         Parameters
 811         ----------
 812         option_type: {"radio", "multi"}
 813             The type of boxes that this control should hold. "radio" for single item select,
 814             "multi" for multi item select.
 815 
 816         """
 817         logger.debug("Adding %s group: %s", option_type, self.option.name)
 818         help_intro, help_items = self._get_multi_help_items(self.option.helptext)
 819         ctl = ttk.LabelFrame(self.frame,
 820                              text=self.option.title,
 821                              name="{}_labelframe".format(option_type))
 822         holder = AutoFillContainer(ctl, self.option_columns, self.option_columns)
 823         for choice in self.option.choices:
 824             ctl = ttk.Radiobutton if option_type == "radio" else MultiOption
 825             ctl = ctl(holder.subframe,
 826                       text=choice.replace("_", " ").title(),
 827                       value=choice,
 828                       variable=self.option.tk_var)
 829             if choice.lower() in help_items:
 830                 self.helpset = True
 831                 helptext = help_items[choice.lower()].capitalize()
 832                 helptext = "{}\n\n - {}".format(
 833                     '. '.join(item.capitalize() for item in helptext.split('. ')),
 834                     help_intro)
 835                 _get_tooltip(ctl, text=helptext, wraplength=600)
 836             ctl.pack(anchor=tk.W)
 837             logger.debug("Added %s option %s", option_type, choice)
 838         return holder.parent

 840     @staticmethod
 841     def _get_multi_help_items(helptext):
 842         """ Split the help text up, for formatted help text, into the individual options
 843         for multi/radio buttons.
 844 
 845         Parameters
 846         ----------
 847         helptext: str
 848             The raw help text for this cli. option
 849 
 850         Returns
 851         -------
 852         tuple (`str`, `dict`)
 853             The help text intro and a dictionary containing the help text split into separate
 854             entries for each option choice
 855         """
 856         logger.debug("raw help: %s", helptext)
 857         all_help = helptext.splitlines()
 858         intro = ""
 859         if any(line.startswith(" - ") for line in all_help):
 860             intro = all_help[0]
 861         retval = (intro, {re.sub(r'[^A-Za-z0-9\-]+', '',
 862                                  line.split()[1].lower()): " ".join(line.split()[1:])
 863                           for line in all_help if line.startswith(" - ")})
 864         logger.debug("help items: %s", retval)
 865         return retval
