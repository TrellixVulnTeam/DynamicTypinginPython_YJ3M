  16 class ContextMenu(tk.Menu):  # pylint: disable=too-many-ancestors
  17     """ A Pop up menu to be triggered when right clicking on widgets that this menu has been
  18     applied to.
  19 
  20     This widget provides a simple right click pop up menu to the widget passed in with `Cut`,
  21     `Copy`, `Paste` and `Select all` menu items.
  22 
  23     Parameters
  24     ----------
  25     widget: tkinter object
  26         The widget to apply the :class:`ContextMenu` to
  27 
  28     Example
  29     -------
  30     >>> text_box = ttk.Entry(parent)
  31     >>> text_box.pack()
  32     >>> right_click_menu = ContextMenu(text_box)
  33     >>> right_click_menu.cm_bind()
  34     """
  35     def __init__(self, widget):
  36         logger.debug("Initializing %s: (widget_class: '%s')",
  37                      self.__class__.__name__, widget.winfo_class())
  38         super().__init__(tearoff=0)
  39         self._widget = widget
  40         self._standard_actions()
  41         logger.debug("Initialized %s", self.__class__.__name__)
  42 
  43     def _standard_actions(self):
  44         """ Standard menu actions """
  45         self.add_command(label="Cut", command=lambda: self._widget.event_generate("<<Cut>>"))
  46         self.add_command(label="Copy", command=lambda: self._widget.event_generate("<<Copy>>"))
  47         self.add_command(label="Paste", command=lambda: self._widget.event_generate("<<Paste>>"))
  48         self.add_separator()
  49         self.add_command(label="Select all", command=self._select_all)
  50 
  51     def cm_bind(self):
  52         """ Bind the menu to the given widgets Right Click event
  53 
  54         After associating a widget with this :class:`ContextMenu` this function should be called
  55         to bind it to the right click button
  56         """
  57         button = "<Button-2>" if platform.system() == "Darwin" else "<Button-3>"
  58         logger.debug("Binding '%s' to '%s'", button, self._widget.winfo_class())
  59         self._widget.bind(button, lambda event: self.tk_popup(event.x_root, event.y_root))
  60 
  61     def _select_all(self):
  62         """ Select all for Text or Entry widgets """
  63         logger.debug("Selecting all for '%s'", self._widget.winfo_class())
  64         if self._widget.winfo_class() == "Text":
  65             self._widget.focus_force()
  66             self._widget.tag_add("sel", "1.0", "end")
  67         else:
  68             self._widget.focus_force()
  69             self._widget.select_range(0, tk.END)

  72 class RightClickMenu(tk.Menu):  # pylint: disable=too-many-ancestors
  73     """ A Pop up menu that can be bound to a right click mouse event to bring up a context menu
  74 
  75     Parameters
  76     ----------
  77     labels: list
  78         A list of label titles that will appear in the right click menu
  79     actions: list
  80         A list of python functions that are called when the corresponding label is clicked on
  81     hotkeys: list, optional
  82         The hotkeys corresponding to the labels. If using hotkeys, then there must be an entry in
  83         the list for every label even if they don't all use hotkeys. Labels without a hotkey can be
  84         an empty string or ``None``. Passing ``None`` instead of a list means that no actions will
  85         be given hotkeys. NB: The hotkey is not bound by this class, that needs to be done in code.
  86         Giving hotkeys here means that they will be displayed in the menu though. Default: ``None``
  87     """
  88     # TODO This should probably be merged with Context Menu
  89     def __init__(self, labels, actions, hotkeys=None):
  90         logger.debug("Initializing %s: (labels: %s, actions: %s)", self.__class__.__name__, labels,
  91                      actions)
  92         super().__init__(tearoff=0)
  93         self._labels = labels
  94         self._actions = actions
  95         self._hotkeys = hotkeys
  96         self._create_menu()
  97         logger.debug("Initialized %s", self.__class__.__name__)
  98 
  99     def _create_menu(self):
 100         """ Create the menu based on :attr:`_labels` and :attr:`_actions`. """
 101         for idx, (label, action) in enumerate(zip(self._labels, self._actions)):
 102             kwargs = dict(label=label, command=action)
 103             if isinstance(self._hotkeys, (list, tuple)) and self._hotkeys[idx]:
 104                 kwargs["accelerator"] = self._hotkeys[idx]
 105             self.add_command(**kwargs)
 106 
 107     def popup(self, event):
 108         """ Pop up the right click menu.
 109 
 110         Parameters
 111         ----------
 112         event: class:`tkinter.Event`
 113             The tkinter mouse event calling this popup
 114         """
 115         self.tk_popup(event.x_root, event.y_root)

 118 class ConsoleOut(ttk.Frame):  # pylint: disable=too-many-ancestors
 119     """ The Console out section of the GUI.
 120 
 121     A Read only text box for displaying the output from stdout/stderr.
 122 
 123     All handling is internal to this method. To clear the console, the stored tkinter variable in
 124     :attr:`~lib.gui.Config.tk_vars` ``consoleclear`` should be triggered.
 125 
 126     Parameters
 127     ----------
 128     parent: tkinter object
 129         The Console's parent widget
 130     debug: bool
 131         ``True`` if console output should not be directed to this widget otherwise ``False``
 132 
 133     """
 134 
 135     def __init__(self, parent, debug):
 136         logger.debug("Initializing %s: (parent: %s, debug: %s)",
 137                      self.__class__.__name__, parent, debug)
 138         super().__init__(parent)
 139         self.pack(side=tk.TOP, anchor=tk.W, padx=10, pady=(2, 0),
 140                   fill=tk.BOTH, expand=True)
 141         self._console = _ReadOnlyText(self)
 142         rc_menu = ContextMenu(self._console)
 143         rc_menu.cm_bind()
 144         self._console_clear = get_config().tk_vars['consoleclear']
 145         self._set_console_clear_var_trace()
 146         self._debug = debug
 147         self._build_console()
 148         self._add_tags()
 149         logger.debug("Initialized %s", self.__class__.__name__)

 151     def _set_console_clear_var_trace(self):
 152         """ Set a trace on the consoleclear tkinter variable to trigger :func:`_clear` """
 153         logger.debug("Set clear trace")
 154         self._console_clear.trace("w", self._clear)
 155 
 156     def _build_console(self):
 157         """ Build and place the console  and add stdout/stderr redirection """
 158         logger.debug("Build console")
 159         self._console.config(width=100, height=6, bg="gray90", fg="black")
 160         self._console.pack(side=tk.LEFT, anchor=tk.N, fill=tk.BOTH, expand=True)
 161 
 162         scrollbar = ttk.Scrollbar(self, command=self._console.yview)
 163         scrollbar.pack(side=tk.LEFT, fill="y")
 164         self._console.configure(yscrollcommand=scrollbar.set)
 165 
 166         self._redirect_console()
 167         logger.debug("Built console")
 168 
 169     def _add_tags(self):
 170         """ Add tags to text widget to color based on output """
 171         logger.debug("Adding text color tags")
 172         self._console.tag_config("default", foreground="#1E1E1E")
 173         self._console.tag_config("stderr", foreground="#E25056")
 174         self._console.tag_config("info", foreground="#2B445E")
 175         self._console.tag_config("verbose", foreground="#008140")
 176         self._console.tag_config("warning", foreground="#F77B00")
 177         self._console.tag_config("critical", foreground="red")
 178         self._console.tag_config("error", foreground="red")

 180     def _redirect_console(self):
 181         """ Redirect stdout/stderr to console Text Box """
 182         logger.debug("Redirect console")
 183         if self._debug:
 184             logger.info("Console debug activated. Outputting to main terminal")
 185         else:
 186             sys.stdout = _SysOutRouter(self._console, "stdout")
 187             sys.stderr = _SysOutRouter(self._console, "stderr")
 188         logger.debug("Redirected console")
 189 
 190     def _clear(self, *args):  # pylint: disable=unused-argument
 191         """ Clear the console output screen """
 192         logger.debug("Clear console")
 193         if not self._console_clear.get():
 194             logger.debug("Console not set for clearing. Skipping")
 195             return
 196         self._console.delete(1.0, tk.END)
 197         self._console_clear.set(False)
 198         logger.debug("Cleared console")

 201 class _ReadOnlyText(tk.Text):  # pylint: disable=too-many-ancestors
 202     """ A read only text widget.
 203 
 204     Standard tkinter Text widgets are read/write by default. As we want to make the console
 205     display writable by the Faceswap process but not the user, we need to redirect its insert and
 206     delete attributes.
 207 
 208     Source: https://stackoverflow.com/questions/3842155
 209     """
 210     def __init__(self, *args, **kwargs):
 211         super().__init__(*args, **kwargs)
 212         self.redirector = _WidgetRedirector(self)
 213         self.insert = self.redirector.register("insert", lambda *args, **kw: "break")
 214         self.delete = self.redirector.register("delete", lambda *args, **kw: "break")

 217 class _SysOutRouter():
 218     """ Route stdout/stderr to the given text box.
 219 
 220     Parameters
 221     ----------
 222     console: tkinter Object
 223         The widget that will receive the output from stderr/stdout
 224     out_type: ['stdout', 'stderr']
 225         The output type to redirect
 226     """
 227 
 228     def __init__(self, console, out_type):
 229         logger.debug("Initializing %s: (console: %s, out_type: '%s')",
 230                      self.__class__.__name__, console, out_type)
 231         self._console = console
 232         self._out_type = out_type
 233         self._recolor = re.compile(r".+?(\s\d+:\d+:\d+\s)(?P<lvl>[A-Z]+)\s")
 234         logger.debug("Initialized %s", self.__class__.__name__)
 235 
 236     def _get_tag(self, string):
 237         """ Set the tag based on regex of log output """
 238         if self._out_type == "stderr":
 239             # Output all stderr in red
 240             return self._out_type
 241 
 242         output = self._recolor.match(string)
 243         if not output:
 244             return "default"
 245         tag = output.groupdict()["lvl"].strip().lower()
 246         return tag
 247 
 248     def write(self, string):
 249         """ Capture stdout/stderr """
 250         self._console.insert(tk.END, string, self._get_tag(string))
 251         self._console.see(tk.END)
 252 
 253     @staticmethod
 254     def flush():
 255         """ If flush is forced, send it to normal terminal """
 256         sys.__stdout__.flush()

 259 class _WidgetRedirector:
 260     """Support for redirecting arbitrary widget sub-commands.
 261 
 262     Some Tk operations don't normally pass through tkinter.  For example, if a
 263     character is inserted into a Text widget by pressing a key, a default Tk
 264     binding to the widget's 'insert' operation is activated, and the Tk library
 265     processes the insert without calling back into tkinter.
 266 
 267     Although a binding to <Key> could be made via tkinter, what we really want
 268     to do is to hook the Tk 'insert' operation itself.  For one thing, we want
 269     a text.insert call in idle code to have the same effect as a key press.
 270 
 271     When a widget is instantiated, a Tcl command is created whose name is the
 272     same as the path name widget._w.  This command is used to invoke the various
 273     widget operations, e.g. insert (for a Text widget). We are going to hook
 274     this command and provide a facility ('register') to intercept the widget
 275     operation.  We will also intercept method calls on the tkinter class
 276     instance that represents the tk widget.
 277 
 278     In IDLE, WidgetRedirector is used in Percolator to intercept Text
 279     commands.  The function being registered provides access to the top
 280     of a Percolator chain.  At the bottom of the chain is a call to the
 281     original Tk widget operation.
 282 
 283     Attributes
 284     -----------
 285     _operations: dict
 286         Dictionary mapping operation name to new function. widget: the widget whose tcl command
 287         is to be intercepted.
 288     tk: widget.tk
 289         A convenience attribute, probably not needed.
 290     orig: str
 291         new name of the original tcl command.
 292 
 293     Notes
 294     -----
 295     Since renaming to orig fails with TclError when orig already exists, only one
 296     WidgetDirector can exist for a given widget.
 297     """
 298     def __init__(self, widget):
 299         self._operations = {}
 300         self.widget = widget                                # widget instance
 301         self.tk_ = tk_ = widget.tk                          # widget's root
 302         wgt = widget._w  # pylint:disable=protected-access  # widget's (full) Tk pathname
 303         self.orig = wgt + "_orig"
 304         # Rename the Tcl command within Tcl:
 305         tk_.call("rename", wgt, self.orig)
 306         # Create a new Tcl command whose name is the widget's path name, and
 307         # whose action is to dispatch on the operation passed to the widget:
 308         tk_.createcommand(wgt, self.dispatch)
 309 
 310     def __repr__(self):
 311         return "%s(%s<%s>)" % (self.__class__.__name__,
 312                                self.widget.__class__.__name__,
 313                                self.widget._w)  # pylint:disable=protected-access
 314 
 315     def close(self):
 316         "Unregister operations and revert redirection created by .__init__."
 317         for operation in list(self._operations):
 318             self.unregister(operation)
 319         widget = self.widget
 320         tk_ = widget.tk
 321         wgt = widget._w  # pylint:disable=protected-access
 322         # Restore the original widget Tcl command.
 323         tk_.deletecommand(wgt)
 324         tk_.call("rename", self.orig, wgt)
 325         del self.widget, self.tk_  # Should not be needed
 326         # if instance is deleted after close, as in Percolator.
 327 
 328     def register(self, operation, function):
 329         """Return _OriginalCommand(operation) after registering function.
 330 
 331         Registration adds an operation: function pair to ._operations.
 332         It also adds a widget function attribute that masks the tkinter
 333         class instance method.  Method masking operates independently
 334         from command dispatch.
 335 
 336         If a second function is registered for the same operation, the
 337         first function is replaced in both places.
 338         """
 339         self._operations[operation] = function
 340         setattr(self.widget, operation, function)
 341         return _OriginalCommand(self, operation)
 342 
 343     def unregister(self, operation):
 344         """Return the function for the operation, or None.
 345 
 346         Deleting the instance attribute unmasks the class attribute.
 347         """
 348         if operation in self._operations:
 349             function = self._operations[operation]
 350             del self._operations[operation]
 351             try:
 352                 delattr(self.widget, operation)
 353             except AttributeError:
 354                 pass
 355             return function
 356         return None
 357 
 358     def dispatch(self, operation, *args):
 359         """Callback from Tcl which runs when the widget is referenced.
 360 
 361         If an operation has been registered in self._operations, apply the
 362         associated function to the args passed into Tcl. Otherwise, pass the
 363         operation through to Tk via the original Tcl function.
 364 
 365         Note that if a registered function is called, the operation is not
 366         passed through to Tk.  Apply the function returned by self.register()
 367         to *args to accomplish that.  For an example, see colorizer.py.
 368 
 369         """
 370         op_ = self._operations.get(operation)
 371         try:
 372             if op_:
 373                 return op_(*args)
 374             return self.tk_.call((self.orig, operation) + args)
 375         except TclError:
 376             return ""
 377

 379 class _OriginalCommand:
 380     """Callable for original tk command that has been redirected.
 381 
 382     Returned by .register; can be used in the function registered.
 383     redir = WidgetRedirector(text)
 384     def my_insert(*args):
 385         print("insert", args)
 386         original_insert(*args)
 387     original_insert = redir.register("insert", my_insert)
 388     """
 389 
 390     def __init__(self, redir, operation):
 391         """Create .tk_call and .orig_and_operation for .__call__ method.
 392 
 393         .redir and .operation store the input args for __repr__.
 394         .tk and .orig copy attributes of .redir (probably not needed).
 395         """
 396         self.redir = redir
 397         self.operation = operation
 398         self.tk_ = redir.tk_  # redundant with self.redir
 399         self.orig = redir.orig  # redundant with self.redir
 400         # These two could be deleted after checking recipient code.
 401         self.tk_call = redir.tk_.call
 402         self.orig_and_operation = (redir.orig, operation)
 403 
 404     def __repr__(self):
 405         return "%s(%r, %r)" % (self.__class__.__name__,
 406                                self.redir, self.operation)
 407 
 408     def __call__(self, *args):
 409         return self.tk_call(self.orig_and_operation + args)

 530 class Tooltip:
 531     """
 532     Create a tooltip for a given widget as the mouse goes on it.
 533 
 534     Parameters
 535     ----------
 536     widget: tkinter object
 537         The widget to apply the tool-tip to
 538     background: str, optional
 539         The hex code for the background color. Default:'#FFFFEA'
 540     pad: tuple, optional
 541         (left, top, right, bottom) padding for the tool-tip. Default: (5, 3, 5, 3)
 542     text: str, optional
 543         The text to be displayed in the tool-tip. Default: 'widget info'
 544     waittime: int, optional
 545         The time in milliseconds to wait before showing the tool-tip. Default: 400
 546     wraplength: int, optional
 547         The text length for each line before wrapping. Default: 250
 548 
 549     Example
 550     -------
 551     >>> button = ttk.Button(parent, text="Exit")
 552     >>> Tooltip(button, text="Click to exit")
 553     >>> button.pack()
 554 
 555     Notes
 556     -----
 557     Adapted from StackOverflow: http://stackoverflow.com/questions/3221956 and
 558     http://www.daniweb.com/programming/software-development/code/484591/a-tooltip-class-for-tkinter
 559     """
 560     def __init__(self, widget, *, background="#FFFFEA", pad=(5, 3, 5, 3), text="widget info",
 561                  waittime=400, wraplength=250):
 562 
 563         self._waittime = waittime  # in milliseconds, originally 500
 564         self._wraplength = wraplength  # in pixels, originally 180
 565         self._widget = widget
 566         self._text = text
 567         self._widget.bind("<Enter>", self._on_enter)
 568         self._widget.bind("<Leave>", self._on_leave)
 569         self._widget.bind("<ButtonPress>", self._on_leave)
 570         self._background = background
 571         self._pad = pad
 572         self._ident = None
 573         self._topwidget = None
 574 
 575     def _on_enter(self, event=None):  # pylint:disable=unused-argument
 576         """ Schedule on an enter event """
 577         self._schedule()
 578 
 579     def _on_leave(self, event=None):  # pylint:disable=unused-argument
 580         """ remove schedule on a leave event """
 581         self._unschedule()
 582         self._hide()
 583 
 584     def _schedule(self):
 585         """ Show the tooltip after wait period """
 586         self._unschedule()
 587         self._ident = self._widget.after(self._waittime, self._show)
 588 
 589     def _unschedule(self):
 590         """ Hide the tooltip """
 591         id_ = self._ident
 592         self._ident = None
 593         if id_:
 594             self._widget.after_cancel(id_)
 595 
 596     def _show(self):
 597         """ Show the tooltip """
 598         def tip_pos_calculator(widget, label,
 599                                *,
 600                                tip_delta=(10, 5), pad=(5, 3, 5, 3)):
 601             """ Calculate the tooltip position """
 602 
 603             s_width, s_height = widget.winfo_screenwidth(), widget.winfo_screenheight()
 604 
 605             width, height = (pad[0] + label.winfo_reqwidth() + pad[2],
 606                              pad[1] + label.winfo_reqheight() + pad[3])
 607 
 608             mouse_x, mouse_y = widget.winfo_pointerxy()
 609 
 610             x_1, y_1 = mouse_x + tip_delta[0], mouse_y + tip_delta[1]
 611             x_2, y_2 = x_1 + width, y_1 + height
 612 
 613             x_delta = x_2 - s_width
 614             if x_delta < 0:
 615                 x_delta = 0
 616             y_delta = y_2 - s_height
 617             if y_delta < 0:
 618                 y_delta = 0
 619 
 620             offscreen = (x_delta, y_delta) != (0, 0)
 621 
 622             if offscreen:
 623 
 624                 if x_delta:
 625                     x_1 = mouse_x - tip_delta[0] - width
 626 
 627                 if y_delta:
 628                     y_1 = mouse_y - tip_delta[1] - height
 629 
 630             offscreen_again = y_1 < 0  # out on the top
 631 
 632             if offscreen_again:
 633                 # No further checks will be done.
 634                 # TIP:
 635                 # A further mod might auto-magically augment the wrap length when the tooltip is
 636                 # too high to be kept inside the screen.
 637                 y_1 = 0
 638 
 639             return x_1, y_1
 640 
 641         background = self._background
 642         pad = self._pad
 643         widget = self._widget
 644 
 645         # Creates a top level window
 646         self._topwidget = tk.Toplevel(widget)
 647         if platform.system() == "Darwin":
 648             # For Mac OS
 649             self._topwidget.tk.call("::tk::unsupported::MacWindowStyle",
 650                                     "style", self._topwidget._w,  # pylint:disable=protected-access
 651                                     "help", "none")
 652 
 653         # Leaves only the label and removes the app window
 654         self._topwidget.wm_overrideredirect(True)
 655 
 656         win = tk.Frame(self._topwidget,
 657                        background=background,
 658                        borderwidth=0)
 659         label = tk.Label(win,
 660                          text=self._text,
 661                          justify=tk.LEFT,
 662                          background=background,
 663                          relief=tk.SOLID,
 664                          borderwidth=0,
 665                          wraplength=self._wraplength)

 667         label.grid(padx=(pad[0], pad[2]),
 668                    pady=(pad[1], pad[3]),
 669                    sticky=tk.NSEW)
 670         win.grid()
 671 
 672         xpos, ypos = tip_pos_calculator(widget, label)
 673 
 674         self._topwidget.wm_geometry("+%d+%d" % (xpos, ypos))

 684 class MultiOption(ttk.Checkbutton):  # pylint: disable=too-many-ancestors
 685     """ Similar to the standard :class:`ttk.Radio` widget, but with the ability to select
 686     multiple pre-defined options. Selected options are generated as `nargs` for the argument
 687     parser to consume.
 688 
 689     Parameters
 690     ----------
 691     parent: :class:`ttk.Frame`
 692         The tkinter parent widget for the check button
 693     value: str
 694         The raw option value for this check button
 695     variable: :class:`tkinter.StingVar`
 696         The master variable for the group of check buttons that this check button will belong to.
 697         The output of this variable will be a string containing a space separated list of the
 698         selected check button options
 699     """
 700     def __init__(self, parent, value, variable, **kwargs):
 701         self._tk_var = tk.BooleanVar()
 702         self._tk_var.set(False)
 703         super().__init__(parent, variable=self._tk_var, **kwargs)
 704         self._value = value
 705         self._master_variable = variable
 706         self._tk_var.trace("w", self._on_update)
 707         self._master_variable.trace("w", self._on_master_update)
 708 
 709     @property
 710     def _master_list(self):
 711         """ list: The contents of the check box group's :attr:`_master_variable` in list form.
 712         Selected check boxes will appear in this list. """
 713         retval = self._master_variable.get().split()
 714         logger.trace(retval)
 715         return retval
 716 
 717     @property
 718     def _master_needs_update(self):
 719         """ bool: ``True`` if :attr:`_master_variable` requires updating otherwise ``False``. """
 720         active = self._tk_var.get()
 721         retval = ((active and self._value not in self._master_list) or
 722                   (not active and self._value in self._master_list))
 723         logger.trace(retval)
 724         return retval
 725 
 726     def _on_update(self, *args):  # pylint: disable=unused-argument
 727         """ Update the master variable on a check button change.
 728 
 729         The value for this checked option is added or removed from the :attr:`_master_variable`
 730         on a ``True``, ``False`` change for this check button.
 731 
 732         Parameters
 733         ----------
 734         args: tuple
 735             Required for variable callback, but unused
 736         """
 737         if not self._master_needs_update:
 738             return
 739         new_vals = self._master_list + [self._value] if self._tk_var.get() else [
 740             val
 741             for val in self._master_list
 742             if val != self._value]
 743         val = " ".join(new_vals)
 744         logger.trace("Setting master variable to: %s", val)
 745         self._master_variable.set(val)
 746 
 747     def _on_master_update(self, *args):  # pylint: disable=unused-argument
 748         """ Update the check button on a master variable change (e.g. load .fsw file in the GUI).
 749 
 750         The value for this option is set to ``True`` or ``False`` depending on it's existence in
 751         the :attr:`_master_variable`
 752 
 753         Parameters
 754         ----------
 755         args: tuple
 756             Required for variable callback, but unused
 757         """
 758         if not self._master_needs_update:
 759             return
 760         state = self._value in self._master_list
 761         logger.trace("Setting '%s' to %s", self._value, state)
 762         self._tk_var.set(state)
