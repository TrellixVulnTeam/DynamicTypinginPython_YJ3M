  26 class NavigationToolbar(NavigationToolbar2Tk):  # pylint: disable=too-many-ancestors
  27     """ Same as default, but only including buttons we need
  28         with custom icons and layout
  29         From: https://stackoverflow.com/questions/12695678 """
  30     toolitems = [t for t in NavigationToolbar2Tk.toolitems if
  31                  t[0] in ("Home", "Pan", "Zoom", "Save")]

  33     @staticmethod
  34     def _Button(frame, text, file, command, extension=".gif"):  # pylint: disable=arguments-differ
  35         """ Map Buttons to their own frame.
  36             Use custom button icons, Use ttk buttons pack to the right """
  37         iconmapping = {"home": "reload",
  38                        "filesave": "save",
  39                        "zoom_to_rect": "zoom"}
  40         icon = iconmapping[file] if iconmapping.get(file, None) else file
  41         img = get_images().icons[icon]
  42         btn = ttk.Button(frame, text=text, image=img, command=command)
  43         btn.pack(side=tk.RIGHT, padx=2)
  44         return btn

  46     def _init_toolbar(self):
  47         """ Same as original but ttk widgets and standard tool-tips used. Separator added and
  48             message label packed to the left """
  49         xmin, xmax = self.canvas.figure.bbox.intervalx
  50         height, width = 50, xmax-xmin
  51         ttk.Frame.__init__(self, master=self.window, width=int(width), height=int(height))
  52 
  53         sep = ttk.Frame(self, height=2, relief=tk.RIDGE)
  54         sep.pack(fill=tk.X, pady=(5, 0), side=tk.TOP)
  55 
  56         self.update()  # Make axes menu
  57 
  58         btnframe = ttk.Frame(self)
  59         btnframe.pack(fill=tk.X, padx=5, pady=5, side=tk.RIGHT)
  60 
  61         for text, tooltip_text, image_file, callback in self.toolitems:
  62             if text is None:
  63                 # Add a spacer; return value is unused.
  64                 self._Spacer()
  65             else:
  66                 button = self._Button(btnframe, text=text, file=image_file,
  67                                       command=getattr(self, callback))
  68                 if tooltip_text is not None:
  69                     Tooltip(button, text=tooltip_text, wraplength=200)
  70 
  71         self.message = tk.StringVar(master=self)
  72         self._message_label = ttk.Label(master=self, textvariable=self.message)
  73         self._message_label.pack(side=tk.LEFT, padx=5)
  74         self.pack(side=tk.BOTTOM, fill=tk.X)

 111     def update_plot(self, initiate=True):
 112         """ Update the plot with incoming data """
 113         logger.trace("Updating plot")
 114         if initiate:
 115             logger.debug("Initializing plot")
 116             self.lines = list()
 117             self.ax1.clear()
 118             self.axes_labels_set()
 119             logger.debug("Initialized plot")
 120 
 121         fulldata = [item for item in self.calcs.stats.values()]
 122         self.axes_limits_set(fulldata)
 123 
 124         xrng = [x for x in range(self.calcs.iterations)]
 125         keys = list(self.calcs.stats.keys())
 126         for idx, item in enumerate(self.lines_sort(keys)):
 127             if initiate:
 128                 self.lines.extend(self.ax1.plot(xrng, self.calcs.stats[item[0]],
 129                                                 label=item[1], linewidth=item[2], color=item[3]))
 130             else:
 131                 self.lines[idx].set_data(xrng, self.calcs.stats[item[0]])
 132 
 133         if initiate:
 134             self.legend_place()
 135         logger.trace("Updated plot")

 149     def axes_limits_set(self, data):
 150         """ Set the axes limits """
 151         xmax = self.calcs.iterations - 1 if self.calcs.iterations > 1 else 1
 152         if data:
 153             ymin, ymax = self.axes_data_get_min_max(data)
 154             self.ax1.set_ylim(ymin, ymax)
 155             self.ax1.set_xlim(0, xmax)
 156             logger.trace("axes ranges: (y: (%s, %s), x:(0, %s)", ymin, ymax, xmax)
 157         else:
 158             self.axes_limits_set_default()

 160     @staticmethod
 161     def axes_data_get_min_max(data):
 162         """ Return the minimum and maximum values from list of lists """
 163         ymin, ymax = list(), list()
 164         for item in data:
 165             dataset = list(filter(lambda x: x is not None, item))
 166             if not dataset:
 167                 continue
 168             ymin.append(min(dataset) * 1000)
 169             ymax.append(max(dataset) * 1000)
 170         ymin = floor(min(ymin)) / 1000
 171         ymax = ceil(max(ymax)) / 1000
 172         logger.trace("ymin: %s, ymax: %s", ymin, ymax)
 173         return ymin, ymax

 180     def lines_sort(self, keys):
 181         """ Sort the data keys into consistent order
 182             and set line color map and line width """
 183         logger.trace("Sorting lines")
 184         raw_lines = list()
 185         sorted_lines = list()
 186         for key in sorted(keys):
 187             title = key.replace("_", " ").title()
 188             if key.startswith("raw"):
 189                 raw_lines.append([key, title])
 190             else:
 191                 sorted_lines.append([key, title])
 192 
 193         groupsize = self.lines_groupsize(raw_lines, sorted_lines)
 194         sorted_lines = raw_lines + sorted_lines
 195         lines = self.lines_style(sorted_lines, groupsize)
 196         return lines
 197 
 198     @staticmethod
 199     def lines_groupsize(raw_lines, sorted_lines):
 200         """ Get the number of items in each group.
 201             If raw data isn't selected, then check the length of
 202             remaining groups until something is found """
 203         groupsize = 1
 204         if raw_lines:
 205             groupsize = len(raw_lines)
 206         elif sorted_lines:
 207             keys = [key[0][:key[0].find("_")] for key in sorted_lines]
 208             distinct_keys = set(keys)
 209             groupsize = len(keys) // len(distinct_keys)
 210         logger.trace(groupsize)
 211         return groupsize
 212 
 213     def lines_style(self, lines, groupsize):
 214         """ Set the color map and line width for each group """
 215         logger.trace("Setting lines style")
 216         groups = int(len(lines) / groupsize)
 217         colours = self.lines_create_colors(groupsize, groups)
 218         for idx, item in enumerate(lines):
 219             linewidth = ceil((idx + 1) / groupsize)
 220             item.extend((linewidth, colours[idx]))
 221         return lines
 222 
 223     def lines_create_colors(self, groupsize, groups):
 224         """ Create the colors """
 225         colours = list()
 226         for i in range(1, groups + 1):
 227             for colour in self.colourmaps[0:groupsize]:
 228                 cmap = matplotlib.cm.get_cmap(colour)
 229                 cpoint = 1 - (i / 5)
 230                 colours.append(cmap(cpoint))
 231         logger.trace(colours)
 232         return colours

 261     def add_callback(self):
 262         """ Add the variable trace to update graph on recent button or save iteration """
 263         get_config().tk_vars["refreshgraph"].trace("w", self.refresh)
 264 
 265     def build(self):
 266         """ Update the plot area with loss values """
 267         logger.debug("Building training graph")
 268         self.plotcanvas.draw()
 269         logger.debug("Built training graph")
 270 
 271     def refresh(self, *args):  # pylint: disable=unused-argument
 272         """ Read loss data and apply to graph """
 273         refresh_var = get_config().tk_vars["refreshgraph"]
 274         if not refresh_var.get() and self.thread is None:
 275             return
 276 
 277         if self.thread is None:
 278             logger.debug("Updating plot data")
 279             self.thread = LongRunningTask(target=self.calcs.refresh)
 280             self.thread.start()
 281             self.after(1000, self.refresh)
 282         elif not self.thread.complete.is_set():
 283             logger.debug("Graph Data not yet available")
 284             self.after(1000, self.refresh)
 285         else:
 286             logger.debug("Updating plot with data from background thread")
 287             self.calcs = self.thread.get_result()  # Terminate the LongRunningTask object
 288             self.thread = None
 289             self.update_plot(initiate=False)
 290             self.plotcanvas.draw()
 291             refresh_var.set(False)
 292 
 293     def save_fig(self, location):
 294         """ Save the figure to file """
 295         logger.debug("Saving graph: '%s'", location)
 296         keys = sorted([key.replace("raw_", "") for key in self.calcs.stats.keys()
 297                        if key.startswith("raw_")])
 298         filename = " - ".join(keys)
 299         now = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
 300         filename = os.path.join(location, "{}_{}.{}".format(filename, now, "png"))
 301         self.fig.set_size_inches(16, 9)
 302         self.fig.savefig(filename, bbox_inches="tight", dpi=120)
 303         print("Saved graph to {}".format(filename))
 304         logger.debug("Saved graph: '%s'", filename)
 305         self.resize_fig()
 306 
 307     def resize_fig(self):
 308         """ Resize the figure back to the canvas """
 309         class Event():  # pylint: disable=too-few-public-methods
 310             """ Event class that needs to be passed to plotcanvas.resize """
 311             pass  # pylint: disable=unnecessary-pass
 312         Event.width = self.winfo_width()
 313         Event.height = self.winfo_height()
 314         self.plotcanvas.resize(Event)  # pylint: disable=no-value-for-parameter
