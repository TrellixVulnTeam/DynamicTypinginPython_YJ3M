  43     def action_command(self, *args):
  44         """ The action to perform when the action button is pressed """
  45         if not self.tk_vars["action"].get():
  46             return
  47         category, command = self.tk_vars["action"].get().split(",")
  48 
  49         if self.tk_vars["runningtask"].get():
  50             self.task.terminate()
  51         else:
  52             self.command = command
  53             args = self.prepare(category)
  54             self.task.execute_script(command, args)
  55         self.tk_vars["action"].set(None)

  57     def generate_command(self, *args):
  58         """ Generate the command line arguments and output """

  60             return
  61         category, command = self.tk_vars["generate"].get().split(",")
  62         args = self.build_args(category, command=command, generate=True)
  63         self.tk_vars["consoleclear"].set(True)
  64         logger.debug(" ".join(args))
  65         print(" ".join(args))

  68     def prepare(self, category):
  69         """ Prepare the environment for execution """
  70         logger.debug("Preparing for execution")
  71         self.tk_vars["runningtask"].set(True)
  72         self.tk_vars["consoleclear"].set(True)
  73         if self.command == "train":
  74             self.tk_vars["istraining"].set(True)
  75         print("Loading...")
  76 
  77         self.statusbar.message.set("Executing - {}.py".format(self.command))
  78         mode = "indeterminate" if self.command in ("effmpeg", "train") else "determinate"
  79         self.statusbar.start(mode)
  80 
  81         args = self.build_args(category)
  82         self.tk_vars["display"].set(self.command)
  83         logger.debug("Prepared for execution")
  84         return args

  86     def build_args(self, category, command=None, generate=False):
  87         """ Build the faceswap command and arguments list """
  88         logger.debug("Build cli arguments: (category: %s, command: %s, generate: %s)",
  89                      category, command, generate)
  90         command = self.command if not command else command
  91         script = "{}.{}".format(category, "py")
  92         pathexecscript = os.path.join(self.pathscript, script)
  93 
  94         args = [sys.executable] if generate else [sys.executable, "-u"]
  95         args.extend([pathexecscript, command])
  96 
  97         cli_opts = get_config().cli_opts
  98         for cliopt in cli_opts.gen_cli_arguments(command):
  99             args.extend(cliopt)
 100             if command == "train" and not generate:
 101                 self.init_training_session(cliopt)
 102         if not generate:
 103             args.append("-gui")  # Indicate to Faceswap that we are running the GUI
 104         if generate:
 105             # Delimit args with spaces
 106             args = ['"{}"'.format(arg) if " " in arg and not arg.startswith(("[", "("))
 107                     and not arg.endswith(("]", ")")) else arg
 108                     for arg in args]
 109         logger.debug("Built cli arguments: (%s)", args)
 110         return args

 124         """ Finalize wrapper when process has exited """
 125         logger.debug("Terminating Faceswap processes")
 126         self.tk_vars["runningtask"].set(False)
 127         if self.task.command == "train":

 139 class FaceswapControl():
 140     """ Control the underlying Faceswap tasks """
 141     def __init__(self, wrapper):
 142         logger.debug("Initializing %s", self.__class__.__name__)
 143         self.wrapper = wrapper
 144         self.config = get_config()
 145         self.statusbar = self.config.statusbar
 146         self.command = None
 147         self.args = None
 148         self.process = None
 149         self.thread = None  # Thread for LongRunningTask termination
 150         self.train_stats = {"iterations": 0, "timestamp": None}
 151         self.consoleregex = {
 152             "loss": re.compile(r"[\W]+(\d+)?[\W]+([a-zA-Z\s]*)[\W]+?(\d+\.\d+)"),
 153             "tqdm": re.compile(r"(?P<dsc>.*?)(?P<pct>\d+%).*?(?P<itm>\S+/\S+)\W\["
 154                                r"(?P<tme>[\d+:]+<.*),\W(?P<rte>.*)[a-zA-Z/]*\]"),
 155             "ffmpeg": re.compile(r"([a-zA-Z]+)=\s*(-?[\d|N/A]\S+)")}
 156         logger.debug("Initialized %s", self.__class__.__name__)

 158     def execute_script(self, command, args):
 159         """ Execute the requested Faceswap Script """
 160         logger.debug("Executing Faceswap: (command: '%s', args: %s)", command, args)
 161         self.thread = None
 162         self.command = command
 163         kwargs = {"stdout": PIPE,
 164                   "stderr": PIPE,
 165                   "bufsize": 1,
 166                   "universal_newlines": True}
 167 
 168         self.process = Popen(args, **kwargs, stdin=PIPE)
 169         self.thread_stdout()
 170         self.thread_stderr()
 171         logger.debug("Executed Faceswap")

 173     def read_stdout(self):
 174         """ Read stdout from the subprocess. If training, pass the loss
 175         values to Queue """
 176         logger.debug("Opening stdout reader")
 177         while True:
 178             try:
 179                 output = self.process.stdout.readline()
 180             except ValueError as err:
 181                 if str(err).lower().startswith("i/o operation on closed file"):
 182                     break
 183                 raise
 184             if output == "" and self.process.poll() is not None:
 185                 break
 186             if output:
 187                 if ((self.command == "train" and self.capture_loss(output)) or
 188                         (self.command == "effmpeg" and self.capture_ffmpeg(output)) or
 189                         (self.command not in ("train", "effmpeg") and self.capture_tqdm(output))):
 190                     continue
 191                 if (self.command == "train" and
 192                         self.wrapper.tk_vars["istraining"].get() and
 193                         "[saved models]" in output.strip().lower()):
 194                     logger.debug("Trigger GUI Training update")
 195                     logger.trace("tk_vars: %s", {itm: var.get()
 196                                                  for itm, var in self.wrapper.tk_vars.items()})
 197                     if not self.config.session.initialized:
 198                         # Don't initialize session until after the first save as state
 199                         # file must exist first
 200                         logger.debug("Initializing curret training session")
 201                         self.config.session.initialize_session(is_training=True)
 202                     self.wrapper.tk_vars["updatepreview"].set(True)
 203                     self.wrapper.tk_vars["refreshgraph"].set(True)
 204                 print(output.strip())
 205         returncode = self.process.poll()
 206         message = self.set_final_status(returncode)
 207         self.wrapper.terminate(message)
 208         logger.debug("Terminated stdout reader. returncode: %s", returncode)
 209 
 210     def read_stderr(self):
 211         """ Read stdout from the subprocess. If training, pass the loss
 212         values to Queue """
 213         logger.debug("Opening stderr reader")
 214         while True:
 215             try:
 216                 output = self.process.stderr.readline()
 217             except ValueError as err:
 218                 if str(err).lower().startswith("i/o operation on closed file"):
 219                     break
 220                 raise
 221             if output == "" and self.process.poll() is not None:
 222                 break
 223             if output:
 224                 if self.command != "train" and self.capture_tqdm(output):
 225                     continue
 226                 print(output.strip(), file=sys.stderr)
 227         logger.debug("Terminated stderr reader")

 247     def capture_loss(self, string):
 248         """ Capture loss values from stdout """
 249         logger.trace("Capturing loss")
 250         if not str.startswith(string, "["):
 251             logger.trace("Not loss message. Returning False")
 252             return False
 253 
 254         loss = self.consoleregex["loss"].findall(string)
 255         if len(loss) != 2 or not all(len(itm) == 3 for itm in loss):
 256             logger.trace("Not loss message. Returning False")
 257             return False
 258 
 259         message = "Total Iterations: {} | ".format(int(loss[0][0]))
 260         message += "  ".join(["{}: {}".format(itm[1], itm[2]) for itm in loss])
 261         if not message:
 262             logger.trace("Error creating loss message. Returning False")
 263             return False
 264 
 265         iterations = self.train_stats["iterations"]
 266 
 267         if iterations == 0:
 268             # Set initial timestamp
 269             self.train_stats["timestamp"] = time()
 270 
 271         iterations += 1
 272         self.train_stats["iterations"] = iterations
 273 
 274         elapsed = self.calc_elapsed()
 275         message = "Elapsed: {} | Session Iterations: {}  {}".format(
 276             elapsed,
 277             self.train_stats["iterations"], message)
 278         self.statusbar.progress_update(message, 0, False)
 279         logger.trace("Succesfully captured loss: %s", message)
 280         return True

 282     def calc_elapsed(self):
 283         """ Calculate and format time since training started """
 284         now = time()
 285         elapsed_time = now - self.train_stats["timestamp"]
 286         try:
 287             hrs = int(elapsed_time // 3600)
 288             if hrs < 10:
 289                 hrs = "{0:02d}".format(hrs)
 290             mins = "{0:02d}".format((int(elapsed_time % 3600) // 60))
 291             secs = "{0:02d}".format((int(elapsed_time % 3600) % 60))
 292         except ZeroDivisionError:
 293             hrs = "00"
 294             mins = "00"
 295             secs = "00"
 296         return "{}:{}:{}".format(hrs, mins, secs)

 298     def capture_tqdm(self, string):
 299         """ Capture tqdm output for progress bar """
 300         logger.trace("Capturing tqdm")
 301         tqdm = self.consoleregex["tqdm"].match(string)
 302         if not tqdm:
 303             return False
 304         tqdm = tqdm.groupdict()
 305         if any("?" in val for val in tqdm.values()):
 306             logger.trace("tqdm initializing. Skipping")
 307             return True
 308         description = tqdm["dsc"].strip()
 309         description = description if description == "" else "{}  |  ".format(description[:-1])
 310         processtime = "Elapsed: {}  Remaining: {}".format(tqdm["tme"].split("<")[0],
 311                                                           tqdm["tme"].split("<")[1])
 312         message = "{}{}  |  {}  |  {}  |  {}".format(description,
 313                                                      processtime,
 314                                                      tqdm["rte"],
 315                                                      tqdm["itm"],
 316                                                      tqdm["pct"])
 317 
 318         position = tqdm["pct"].replace("%", "")
 319         position = int(position) if position.isdigit() else 0
 320 
 321         self.statusbar.progress_update(message, position, True)
 322         logger.trace("Succesfully captured tqdm message: %s", message)
 323         return True

 325     def capture_ffmpeg(self, string):
 326         """ Capture tqdm output for progress bar """
 327         logger.trace("Capturing ffmpeg")
 328         ffmpeg = self.consoleregex["ffmpeg"].findall(string)
 329         if len(ffmpeg) < 7:
 330             logger.trace("Not ffmpeg message. Returning False")
 331             return False
 332 
 333         message = ""
 334         for item in ffmpeg:
 335             message += "{}: {}  ".format(item[0], item[1])
 336         if not message:
 337             logger.trace("Error creating ffmpeg message. Returning False")
 338             return False
 339 
 340         self.statusbar.progress_update(message, 0, False)
 341         logger.trace("Succesfully captured ffmpeg message: %s", message)
 342         return True

 344     def terminate(self):
 345         """ Terminate the running process in a LongRunningTask so we can still
 346             output to console """
 347         if self.thread is None:
 348             logger.debug("Terminating wrapper in LongRunningTask")
 349             self.thread = LongRunningTask(target=self.terminate_in_thread,
 350                                           args=(self.command, self.process))
 351             if self.command == "train":
 352                 self.wrapper.tk_vars["istraining"].set(False)
 353             self.thread.start()
 354             self.config.root.after(1000, self.terminate)
 355         elif not self.thread.complete.is_set():
 356             logger.debug("Not finished terminating")
 357             self.config.root.after(1000, self.terminate)
 358         else:
 359             logger.debug("Termination Complete. Cleaning up")
 360             _ = self.thread.get_result()  # Terminate the LongRunningTask object
 361             self.thread = None

 363     def terminate_in_thread(self, command, process):
 364         """ Terminate the subprocess """
 365         logger.debug("Terminating wrapper")
 366         if command == "train":
 367             timeout = self.config.user_config_dict.get("timeout", 120)
 368             logger.debug("Sending Exit Signal")
 369             print("Sending Exit Signal", flush=True)
 370             now = time()
 371             if os.name == "nt":
 372                 logger.debug("Sending carriage return to process")
 373                 con_in = win32console.GetStdHandle(  # pylint:disable=c-extension-no-member
 374                     win32console.STD_INPUT_HANDLE)  # pylint:disable=c-extension-no-member
 375                 keypress = self.generate_windows_keypress("\n")
 376                 con_in.WriteConsoleInput([keypress])
 377             else:
 378                 logger.debug("Sending SIGINT to process")
 379                 process.send_signal(signal.SIGINT)
 380             while True:
 381                 timeelapsed = time() - now
 382                 if process.poll() is not None:
 383                     break
 384                 if timeelapsed > timeout:
 385                     logger.error("Timeout reached sending Exit Signal")
 386                     self.terminate_all_children()
 387         else:
 388             self.terminate_all_children()
 389         return True

 391     @staticmethod
 392     def generate_windows_keypress(character):
 393         """ Generate an 'Enter' key press to terminate Windows training """
 394         buf = win32console.PyINPUT_RECORDType(  # pylint:disable=c-extension-no-member
 395             win32console.KEY_EVENT)  # pylint:disable=c-extension-no-member
 396         buf.KeyDown = 1
 397         buf.RepeatCount = 1
 398         buf.Char = character
 399         return buf

 401     @staticmethod
 402     def terminate_all_children():
 403         """ Terminates all children """
 404         logger.debug("Terminating Process...")
 405         print("Terminating Process...", flush=True)
 406         children = psutil.Process().children(recursive=True)
 407         for child in children:
 408             child.terminate()
 409         _, alive = psutil.wait_procs(children, timeout=10)
 410         if not alive:
 411             logger.debug("Terminated")
 412             print("Terminated")
 413             return
 414 
 415         logger.debug("Termination timed out. Killing Process...")
 416         print("Termination timed out. Killing Process...", flush=True)
 417         for child in alive:
 418             child.kill()
 419         _, alive = psutil.wait_procs(alive, timeout=10)
 420         if not alive:
 421             logger.debug("Killed")
 422             print("Killed")
 423         else:
 424             for child in alive:
 425                 msg = "Process {} survived SIGKILL. Giving up".format(child)
 426                 logger.debug(msg)
 427                 print(msg)

 429     def set_final_status(self, returncode):
 430         """ Set the status bar output based on subprocess return code
 431             and reset training stats """
 432         logger.debug("Setting final status. returncode: %s", returncode)
 433         self.train_stats = {"iterations": 0, "timestamp": None}
 434         if returncode in (0, 3221225786):
 435             status = "Ready"
 436         elif returncode == -15:
 437             status = "Terminated - {}.py".format(self.command)
 438         elif returncode == -9:
 439             status = "Killed - {}.py".format(self.command)
 440         elif returncode == -6:
 441             status = "Aborted - {}.py".format(self.command)
 442         else:
 443             status = "Failed - {}.py. Return Code: {}".format(self.command, returncode)
 444         logger.debug("Set final status: %s", status)
 445         return status
