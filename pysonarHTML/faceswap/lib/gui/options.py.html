  21     def __init__(self):

  23         self.categories = ("faceswap", "tools")

  25         self.opts = dict()

  29     def build_options(self):
  30         """ Get the commands that belong to each category """
  31         for category in self.categories:
  32             logger.debug("Building '%s'", category)
  33             if category == "tools":
  34                 mod_classes = self._get_tools_cli_classes()
  35                 self.commands[category] = self.sort_commands(category, mod_classes)
  36                 for tool in sorted(mod_classes):
  37                     self.opts.update(self.extract_options(mod_classes[tool], [tool]))
  38             else:
  39                 mod_classes = self.get_cli_classes(cli)
  40                 self.commands[category] = self.sort_commands(category, mod_classes)
  41                 self.opts.update(self.extract_options(cli, mod_classes))
  42             logger.debug("Built '%s'", category)

  44     @staticmethod
  45     def get_cli_classes(cli_source):
  46         """ Parse the cli scripts for the argument classes """
  47         mod_classes = []
  48         for name, obj in inspect.getmembers(cli_source):
  49             if inspect.isclass(obj) and name.lower().endswith("args") \
  50                     and name.lower() not in (("faceswapargs",
  51                                               "extractconvertargs",
  52                                               "guiargs")):
  53                 mod_classes.append(name)
  54         logger.debug(mod_classes)
  55         return mod_classes

  57     @staticmethod
  58     def _get_tools_cli_classes():
  59         """ Parse the tools cli scripts for the argument classes """
  60         base_path = os.path.realpath(os.path.dirname(sys.argv[0]))
  61         tools_dir = os.path.join(base_path, "tools")
  62         mod_classes = dict()
  63         for tool_name in sorted(os.listdir(tools_dir)):
  64             cli_file = os.path.join(tools_dir, tool_name, "cli.py")
  65             if os.path.exists(cli_file):
  66                 mod = ".".join(("tools", tool_name, "cli"))
  67                 mod_classes["{}Args".format(tool_name.title())] = import_module(mod)
  68         return mod_classes

  70     def sort_commands(self, category, classes):
  71         """ Format classes into command names and sort:
  72             Specific workflow order for faceswap.
  73             Alphabetical for all others """
  74         commands = sorted(self.format_command_name(command)
  75                           for command in classes)
  76         if category == "faceswap":
  77             ordered = ["extract", "train", "convert"]
  78             commands = ordered + [command for command in commands
  79                                   if command not in ordered]
  80         logger.debug(commands)
  81         return commands

  83     @staticmethod
  84     def format_command_name(classname):
  85         """ Format args class name to command """
  86         return classname.lower()[:-4]

  88     def extract_options(self, cli_source, mod_classes):
  89         """ Extract the existing ArgParse Options
  90             into master options Dictionary """
  91         subopts = dict()
  92         for classname in mod_classes:
  93             logger.debug("Processing: (classname: '%s')", classname)
  94             command = self.format_command_name(classname)
  95             info, options = self.get_cli_arguments(cli_source, classname, command)
  96             options = self.process_options(options, command)
  97             options["helptext"] = info
  98             logger.debug("Processed: (classname: '%s', command: '%s', options: %s)",
  99                          classname, command, options)
 100             subopts[command] = options
 101         return subopts

 103     @staticmethod
 104     def get_cli_arguments(cli_source, classname, command):
 105         """ Extract the options from the main and tools cli files """
 106         meth = getattr(cli_source, classname)(None, command)
 107         return meth.info, meth.argument_list + meth.optional_arguments + meth.global_arguments

 109     def process_options(self, command_options, command):
 110         """ Process the options for a single command """
 111         gui_options = OrderedDict()
 112         for opt in command_options:
 113             logger.trace("Processing: %s", opt)
 114             if opt.get("help", "") == SUPPRESS:
 115                 logger.trace("Skipping suppressed option: %s", opt)
 116                 continue
 117             title = self.set_control_title(opt["opts"])
 118             cpanel_option = ControlPanelOption(
 119                 title,
 120                 self.get_data_type(opt),
 121                 group=opt.get("group", None),
 122                 default=opt.get("default", None),
 123                 choices=opt.get("choices", None),
 124                 is_radio=opt.get("action", "") == actions.Radio,
 125                 is_multi_option=opt.get("action", "") == actions.MultiOption,
 126                 rounding=self.get_rounding(opt),
 127                 min_max=opt.get("min_max", None),
 128                 sysbrowser=self.get_sysbrowser(opt, command_options, command),
 129                 helptext=opt["help"],
 130                 track_modified=True,
 131                 command=command)
 132             gui_options[title] = dict(cpanel_option=cpanel_option,
 133                                       opts=opt["opts"],
 134                                       nargs=opt.get("nargs", None))
 135             logger.trace("Processed: %s", gui_options)
 136         return gui_options

 138     @staticmethod
 139     def set_control_title(opts):
 140         """ Take the option switch and format it nicely """
 141         ctltitle = opts[1] if len(opts) == 2 else opts[0]
 142         ctltitle = ctltitle.replace("-", " ").replace("_", " ").strip().title()
 143         return ctltitle

 145     @staticmethod
 146     def get_data_type(opt):
 147         """ Return a datatype for passing into control_helper.py to get the correct control """
 148         if opt.get("type", None) is not None and isinstance(opt["type"], type):
 149             retval = opt["type"]
 150         elif opt.get("action", "") in ("store_true", "store_false"):
 151             retval = bool
 152         else:
 153             retval = str
 154         return retval

 156     @staticmethod
 157     def get_rounding(opt):
 158         """ Return rounding if correct data type, else None """
 159         dtype = opt.get("type", None)
 160         if dtype == float:
 161             retval = opt.get("rounding", 2)
 162         elif dtype == int:
 163             retval = opt.get("rounding", 1)
 164         else:
 165             retval = None
 166         return retval

 168     def get_sysbrowser(self, option, options, command):
 169         """ Return the system file browser and file types if required else None """
 170         action = option.get("action", None)
 171         if action not in (actions.DirFullPaths,
 172                           actions.FileFullPaths,
 173                           actions.FilesFullPaths,
 174                           actions.DirOrFileFullPaths,
 175                           actions.SaveFileFullPaths,
 176                           actions.ContextFullPaths):
 177             return None
 178 
 179         retval = dict()
 180         action_option = None
 181         if option.get("action_option", None) is not None:
 182             self.expand_action_option(option, options)
 183             action_option = option["action_option"]
 184         retval["filetypes"] = option.get("filetypes", "default")
 185         if action == actions.FileFullPaths:
 186             retval["browser"] = ["load"]
 187         elif action == actions.FilesFullPaths:
 188             retval["browser"] = ["multi_load"]
 189         elif action == actions.SaveFileFullPaths:
 190             retval["browser"] = ["save"]
 191         elif action == actions.DirOrFileFullPaths:
 192             retval["browser"] = ["folder", "load"]
 193         elif action == actions.ContextFullPaths and action_option:
 194             retval["browser"] = ["context"]
 195             retval["command"] = command
 196             retval["action_option"] = action_option
 197             retval["destination"] = option.get("dest", option["opts"][1].replace("--", ""))
 198         else:
 199             retval["browser"] = ["folder"]
 200         logger.debug(retval)
 201         return retval

 203     @staticmethod
 204     def expand_action_option(option, options):
 205         """ Expand the action option to the full command name """
 206         opts = {opt["opts"][0]: opt["opts"][-1]
 207                 for opt in options}
 208         old_val = option["action_option"]
 209         new_val = opts[old_val]
 210         logger.debug("Updating action option from '%s' to '%s'", old_val, new_val)
 211         option["action_option"] = new_val

 213     def gen_command_options(self, command):
 214         """ Yield each option for specified command """
 215         for key, val in self.opts[command].items():
 216             if not isinstance(val, dict):
 217                 continue
 218             yield key, val

 220     def options_to_process(self, command=None):
 221         """ Return a consistent object for processing regardless of whether processing all commands
 222             or just one command for reset and clear. Removes helptext from return value """
 223         if command is None:
 224             options = [opt for opts in self.opts.values()
 225                        for opt in opts.values() if isinstance(opt, dict)]
 226         else:
 227             options = [opt for opt in self.opts[command].values() if isinstance(opt, dict)]
 228         return options
 229 
 230     def reset(self, command=None):
 231         """ Reset the options for all or passed command
 232             back to default value """
 233         logger.debug("Resetting options to default. (command: '%s'", command)
 234         for option in self.options_to_process(command):
 235             cp_opt = option["cpanel_option"]
 236             default = "" if cp_opt.default is None else cp_opt.default
 237             if (option.get("nargs", None)
 238                     and isinstance(default, (list, tuple))):
 239                 default = ' '.join(str(val) for val in default)
 240             cp_opt.set(default)
 241 
 242     def clear(self, command=None):
 243         """ Clear the options values for all or passed commands """
 244         logger.debug("Clearing options. (command: '%s'", command)
 245         for option in self.options_to_process(command):
 246             cp_opt = option["cpanel_option"]
 247             if isinstance(cp_opt.get(), bool):
 248                 cp_opt.set(False)
 249             elif isinstance(cp_opt.get(), (int, float)):
 250                 cp_opt.set(0)
 251             else:
 252                 cp_opt.set("")

 254     def get_option_values(self, command=None):
 255         """ Return all or single command control titles with the associated tk_var value """
 256         ctl_dict = dict()
 257         for cmd, opts in self.opts.items():
 258             if command and command != cmd:
 259                 continue
 260             cmd_dict = dict()
 261             for key, val in opts.items():
 262                 if not isinstance(val, dict):
 263                     continue
 264                 cmd_dict[key] = val["cpanel_option"].get()
 265             ctl_dict[cmd] = cmd_dict
 266         logger.debug("command: '%s', ctl_dict: %s", command, ctl_dict)
 267         return ctl_dict

 269     def get_one_option_variable(self, command, title):
 270         """ Return a single tk_var for the specified
 271             command and control_title """
 272         for opt_title, option in self.gen_command_options(command):
 273             if opt_title == title:
 274                 return option["cpanel_option"].tk_var
 275         return None

 277     def gen_cli_arguments(self, command):
 278         """ Return the generated cli arguments for the selected command """
 279         for _, option in self.gen_command_options(command):
 280             optval = str(option["cpanel_option"].get())
 281             opt = option["opts"][0]
 282             if command in ("extract", "convert") and opt == "-o":
 283                 get_images().set_faceswap_output_path(optval)
 284             if optval in ("False", ""):
 285                 continue
 286             if optval == "True":
 287                 yield (opt, )
 288             else:
 289                 if option.get("nargs", None):
 290                     if "\"" in optval:
 291                         optval = [arg[1:-1] for arg in re.findall(r"\".+?\"", optval)]
 292                     else:
 293                         optval = optval.split(" ")
 294                     opt = [opt] + optval
 295                 else:
 296                     opt = (opt, optval)
 297                 yield opt
