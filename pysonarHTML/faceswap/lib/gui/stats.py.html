  19 def convert_time(timestamp):
  20     """ Convert time stamp to total hours, minutes and seconds """
  21     hrs = int(timestamp // 3600)
  22     if hrs < 10:
  23         hrs = "{0:02d}".format(hrs)
  24     mins = "{0:02d}".format((int(timestamp % 3600) // 60))
  25     secs = "{0:02d}".format((int(timestamp % 3600) % 60))
  26     return hrs, mins, secs

  29 class TensorBoardLogs():
  30     """ Parse and return data from TensorBoard logs """
  31     def __init__(self, logs_folder):
  32         self.folder_base = logs_folder
  33         self.log_filenames = self.set_log_filenames()
  34 
  35     def set_log_filenames(self):
  36         """ Set the TensorBoard log filenames for all existing sessions """
  37         logger.debug("Loading log filenames. base_dir: '%s'", self.folder_base)
  38         log_filenames = dict()
  39         for dirpath, _, filenames in os.walk(self.folder_base):
  40             if not any(filename.startswith("events.out.tfevents") for filename in filenames):
  41                 continue
  42             logfiles = [filename for filename in filenames
  43                         if filename.startswith("events.out.tfevents")]
  44             # Take the last logfile, in case of previous crash
  45             logfile = os.path.join(dirpath, sorted(logfiles)[-1])
  46             side, session = os.path.split(dirpath)
  47             side = os.path.split(side)[1]
  48             session = int(session[session.rfind("_") + 1:])
  49             log_filenames.setdefault(session, dict())[side] = logfile
  50         logger.debug("logfiles: %s", log_filenames)
  51         return log_filenames
  52 
  53     def get_loss(self, side=None, session=None):
  54         """ Read the loss from the TensorBoard logs
  55             Specify a side or a session or leave at None for all
  56         """
  57         logger.debug("Getting loss: (side: %s, session: %s)", side, session)
  58         all_loss = dict()
  59         for sess, sides in self.log_filenames.items():
  60             if session is not None and sess != session:
  61                 logger.debug("Skipping session: %s", sess)
  62                 continue
  63             loss = dict()
  64             for sde, logfile in sides.items():
  65                 if side is not None and sde != side:
  66                     logger.debug("Skipping side: %s", sde)
  67                     continue
  68                 for event in tf.train.summary_iterator(logfile):
  69                     for summary in event.summary.value:
  70                         if "loss" not in summary.tag:
  71                             continue
  72                         tag = summary.tag.replace("batch_", "")
  73                         loss.setdefault(tag,
  74                                         dict()).setdefault(sde,
  75                                                            list()).append(summary.simple_value)
  76             all_loss[sess] = loss
  77         return all_loss
  78 
  79     def get_timestamps(self, session=None):
  80         """ Read the timestamps from the TensorBoard logs
  81             Specify a session or leave at None for all
  82             NB: For all intents and purposes timestamps are the same for
  83                 both sides, so just read from one side """
  84         logger.debug("Getting timestamps")
  85         all_timestamps = dict()
  86         for sess, sides in self.log_filenames.items():
  87             if session is not None and sess != session:
  88                 logger.debug("Skipping sessions: %s", sess)
  89                 continue
  90             try:
  91                 for logfile in sides.values():
  92                     timestamps = [event.wall_time
  93                                   for event in tf.train.summary_iterator(logfile)
  94                                   if event.summary.value]
  95                     logger.debug("Total timestamps for session %s: %s", sess, len(timestamps))
  96                     all_timestamps[sess] = timestamps
  97                     break  # break after first file read
  98             except tf_errors.DataLossError as err:
  99                 logger.warning("The logs for Session %s are corrupted and cannot be displayed. "
 100                                "The totals do not include this session. Original error message: "
 101                                "'%s'", sess, str(err))
 102         return all_timestamps

 252 class SessionsSummary():
 253     """ Calculations for analysis summary stats """
 254 
 255     def __init__(self, session):
 256         logger.debug("Initializing %s: (session: %s)", self.__class__.__name__, session)
 257         self.session = session
 258         logger.debug("Initialized %s", self.__class__.__name__)
 259 
 260     @property
 261     def time_stats(self):
 262         """ Return session time stats """
 263         ts_data = self.session.tb_logs.get_timestamps()
 264         time_stats = {sess_id: {"start_time": min(timestamps) if timestamps else 0,
 265                                 "end_time": max(timestamps) if timestamps else 0,
 266                                 "datapoints": len(timestamps) if timestamps else 0}
 267                       for sess_id, timestamps in ts_data.items()}
 268         return time_stats
 269 
 270     @property
 271     def sessions_stats(self):
 272         """ Return compiled stats """
 273         compiled = list()
 274         for sess_idx, ts_data in self.time_stats.items():
 275             logger.debug("Compiling session ID: %s", sess_idx)
 276             if self.session.state is None:
 277                 logger.debug("Session state dict doesn't exist. Most likely task has been "
 278                              "terminated during compilation")
 279                 return None
 280             iterations = self.session.get_iterations_for_session(sess_idx)
 281             elapsed = ts_data["end_time"] - ts_data["start_time"]
 282             batchsize = self.session.total_batchsize.get(sess_idx, 0)
 283             compiled.append({"session": sess_idx,
 284                              "start": ts_data["start_time"],
 285                              "end": ts_data["end_time"],
 286                              "elapsed": elapsed,
 287                              "rate": (batchsize * iterations) / elapsed if elapsed != 0 else 0,
 288                              "batch": batchsize,
 289                              "iterations": iterations})
 290         compiled = sorted(compiled, key=lambda k: k["session"])
 291         return compiled
 292 
 293     def compile_stats(self):
 294         """ Compile sessions stats with totals, format and return """
 295         logger.debug("Compiling sessions summary data")
 296         compiled_stats = self.sessions_stats
 297         if compiled_stats is None:
 298             return compiled_stats
 299         logger.debug("sessions_stats: %s", compiled_stats)
 300         total_stats = self.total_stats(compiled_stats)
 301         compiled_stats.append(total_stats)
 302         compiled_stats = self.format_stats(compiled_stats)
 303         logger.debug("Final stats: %s", compiled_stats)
 304         return compiled_stats
 305 
 306     @staticmethod
 307     def total_stats(sessions_stats):
 308         """ Return total stats """
 309         logger.debug("Compiling Totals")
 310         elapsed = 0
 311         examples = 0
 312         iterations = 0
 313         batchset = set()
 314         total_summaries = len(sessions_stats)
 315         for idx, summary in enumerate(sessions_stats):
 316             if idx == 0:
 317                 starttime = summary["start"]
 318             if idx == total_summaries - 1:
 319                 endtime = summary["end"]
 320             elapsed += summary["elapsed"]
 321             examples += (summary["batch"] * summary["iterations"])
 322             batchset.add(summary["batch"])
 323             iterations += summary["iterations"]
 324         batch = ",".join(str(bs) for bs in batchset)
 325         totals = {"session": "Total",
 326                   "start": starttime,
 327                   "end": endtime,
 328                   "elapsed": elapsed,
 329                   "rate": examples / elapsed if elapsed != 0 else 0,
 330                   "batch": batch,
 331                   "iterations": iterations}
 332         logger.debug(totals)
 333         return totals
 334 
 335     @staticmethod
 336     def format_stats(compiled_stats):
 337         """ Format for display """
 338         logger.debug("Formatting stats")
 339         for summary in compiled_stats:
 340             hrs, mins, secs = convert_time(summary["elapsed"])
 341             summary["start"] = time.strftime("%x %X", time.localtime(summary["start"]))
 342             summary["end"] = time.strftime("%x %X", time.localtime(summary["end"]))
 343             summary["elapsed"] = "{}:{}:{}".format(hrs, mins, secs)
 344             summary["rate"] = "{0:.1f}".format(summary["rate"])
 345         return compiled_stats

 348 class Calculations():
 349     """ Class to pull raw data for given session(s) and perform calculations """
 350     def __init__(self, session, display="loss", loss_keys=["loss"], selections=["raw"],
 351                  avg_samples=500, smooth_amount=0.90, flatten_outliers=False, is_totals=False):
 352         logger.debug("Initializing %s: (session: %s, display: %s, loss_keys: %s, selections: %s, "
 353                      "avg_samples: %s, smooth_amount: %s, flatten_outliers: %s, is_totals: %s",
 354                      self.__class__.__name__, session, display, loss_keys, selections, avg_samples,
 355                      smooth_amount, flatten_outliers, is_totals)
 356 
 357         warnings.simplefilter("ignore", np.RankWarning)
 358 
 359         self.session = session
 360         self.display = display
 361         self.loss_keys = loss_keys
 362         self.selections = selections
 363         self.is_totals = is_totals
 364         self.args = {"avg_samples": avg_samples,
 365                      "smooth_amount": smooth_amount,
 366                      "flatten_outliers": flatten_outliers}
 367         self.iterations = 0
 368         self.stats = None
 369         self.refresh()
 370         logger.debug("Initialized %s", self.__class__.__name__)
 371 
 372     def refresh(self):
 373         """ Refresh the stats """
 374         logger.debug("Refreshing")
 375         if not self.session.initialized:
 376             logger.warning("Session data is not initialized. Not refreshing")
 377             return None
 378         self.iterations = 0
 379         self.stats = self.get_raw()
 380         self.get_calculations()
 381         self.remove_raw()
 382         logger.debug("Refreshed")
 383         return self

 385     def get_raw(self):
 386         """ Add raw data to stats dict """
 387         logger.debug("Getting Raw Data")
 388 
 389         raw = dict()
 390         iterations = set()
 391         if self.display.lower() == "loss":
 392             loss_dict = self.session.total_loss if self.is_totals else self.session.loss
 393             for loss_name, side_loss in loss_dict.items():
 394                 if loss_name not in self.loss_keys:
 395                     continue
 396                 for side, loss in side_loss.items():
 397                     if self.args["flatten_outliers"]:
 398                         loss = self.flatten_outliers(loss)
 399                     iterations.add(len(loss))
 400                     raw["raw_{}_{}".format(loss_name, side)] = loss
 401 
 402             self.iterations = 0 if not iterations else min(iterations)
 403             if len(iterations) > 1:
 404                 # Crop all losses to the same number of items
 405                 if self.iterations == 0:
 406                     raw = {lossname: list() for lossname in raw}
 407                 else:
 408                     raw = {lossname: loss[:self.iterations] for lossname, loss in raw.items()}
 409 
 410         else:  # Rate calulation
 411             data = self.calc_rate_total() if self.is_totals else self.calc_rate()
 412             if self.args["flatten_outliers"]:
 413                 data = self.flatten_outliers(data)
 414             self.iterations = len(data)
 415             raw = {"raw_rate": data}
 416 
 417         logger.debug("Got Raw Data")
 418         return raw

 420     def remove_raw(self):
 421         """ Remove raw values from stats if not requested """
 422         if "raw" in self.selections:
 423             return
 424         logger.debug("Removing Raw Data from output")
 425         for key in list(self.stats.keys()):
 426             if key.startswith("raw"):
 427                 del self.stats[key]
 428         logger.debug("Removed Raw Data from output")

 430     def calc_rate(self):
 431         """ Calculate rate per iteration """
 432         logger.debug("Calculating rate")
 433         batchsize = self.session.batchsize
 434         timestamps = self.session.timestamps
 435         iterations = range(len(timestamps) - 1)
 436         rate = [batchsize / (timestamps[i + 1] - timestamps[i]) for i in iterations]
 437         logger.debug("Calculated rate: Item_count: %s", len(rate))
 438         return rate

 456             rate.extend([batchsize / (timestamps[i + 1] - timestamps[i]) for i in iterations])
 457         logger.debug("Calculated totals rate: Item_count: %s", len(rate))

 460     @staticmethod
 461     def flatten_outliers(data):
 462         """ Remove the outliers from a provided list """
 463         logger.debug("Flattening outliers")
 464         retdata = list()
 465         samples = len(data)
 466         mean = (sum(data) / samples)
 467         limit = sqrt(sum([(item - mean)**2 for item in data]) / samples)
 468         logger.debug("samples: %s, mean: %s, limit: %s", samples, mean, limit)
 469 
 470         for idx, item in enumerate(data):
 471             if (mean - limit) <= item <= (mean + limit):
 472                 retdata.append(item)
 473             else:
 474                 logger.trace("Item idx: %s, value: %s flattened to %s", idx, item, mean)
 475                 retdata.append(mean)
 476         logger.debug("Flattened outliers")
 477         return retdata

 479     def get_calculations(self):
 480         """ Perform the required calculations """
 481         for selection in self.selections:
 482             if selection == "raw":
 483                 continue
 484             logger.debug("Calculating: %s", selection)
 485             method = getattr(self, "calc_{}".format(selection))
 486             raw_keys = [key for key in self.stats.keys() if key.startswith("raw_")]
 487             for key in raw_keys:
 488                 selected_key = "{}_{}".format(selection, key.replace("raw_", ""))
 489                 self.stats[selected_key] = method(self.stats[key])
 490 
 491     def calc_avg(self, data):
 492         """ Calculate rolling average """
 493         logger.debug("Calculating Average")
 494         avgs = list()
 495         presample = ceil(self.args["avg_samples"] / 2)
 496         postsample = self.args["avg_samples"] - presample
 497         datapoints = len(data)
 498 
 499         if datapoints <= (self.args["avg_samples"] * 2):
 500             logger.info("Not enough data to compile rolling average")
 501             return avgs
 502 
 503         for idx in range(0, datapoints):
 504             if idx < presample or idx >= datapoints - postsample:
 505                 avgs.append(None)
 506                 continue
 507             avg = sum(data[idx - presample:idx + postsample]) / self.args["avg_samples"]
 508             avgs.append(avg)
 509         logger.debug("Calculated Average")
 510         return avgs
 511 
 512     def calc_smoothed(self, data):
 513         """ Smooth the data """
 514         last = data[0]  # First value in the plot (first timestep)
 515         weight = self.args["smooth_amount"]
 516         smoothed = list()
 517         for point in data:
 518             smoothed_val = last * weight + (1 - weight) * point  # Calculate smoothed value
 519             smoothed.append(smoothed_val)                        # Save it
 520             last = smoothed_val                                  # Anchor the last smoothed value
 521 
 522         return smoothed

 524     @staticmethod
 525     def calc_trend(data):
 526         """ Compile trend data """
 527         logger.debug("Calculating Trend")
 528         points = len(data)
 529         if points < 10:
 530             dummy = [None for i in range(points)]
 531             return dummy
 532         x_range = range(points)
 533         fit = np.polyfit(x_range, data, 3)
 534         poly = np.poly1d(fit)
 535         trend = poly(x_range)
 536         logger.debug("Calculated Trend")
 537         return trend
