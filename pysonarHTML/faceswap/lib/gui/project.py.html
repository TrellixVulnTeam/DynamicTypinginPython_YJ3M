  89     @property
  90     def _selected_to_choices(self):
  91         """ dict: The selected value and valid choices for multi-option, radio or combo options.
  92         """
  93         valid_choices = {cmd: {opt: dict(choices=val["cpanel_option"].choices,
  94                                          is_multi=val["cpanel_option"].is_multi_option)
  95                                for opt, val in data.items()
  96                                if isinstance(val, dict) and "cpanel_option" in val
  97                                and val["cpanel_option"].choices is not None}
  98                          for cmd, data in self._config.cli_opts.opts.items()}
  99         logger.trace("valid_choices: %s", valid_choices)
 100         retval = {command: {option: {"value": value,
 101                                      "is_multi": valid_choices[command][option]["is_multi"],
 102                                      "choices":  valid_choices[command][option]["choices"]}
 103                             for option, value in options.items()
 104                             if value and command in valid_choices
 105                             and option in valid_choices[command]}
 106                   for command, options in self._options.items()
 107                   if isinstance(options, dict)}

 126     def _set_filename(self, filename=None, sess_type="project"):
 127         """ Set the :attr:`_filename` attribute.
 128 
 129         :attr:`_filename` is set either from a given filename or the result from
 130         a :attr:`_file_handler`.
 131 
 132         Parameters
 133         ----------
 134         filename: str, optional
 135             An optional filename. If given then this filename will be used otherwise it will be
 136             collected by a :attr:`_file_handler`
 137 
 138         sess_type: {all, project, task}, optional
 139             The session type that the filename is being set for. Dictates the type of file handler
 140             that is opened.
 141 
 142         Returns
 143         -------
 144         bool: `True` if filename has been successfully set otherwise ``False``
 145         """
 146         logger.debug("filename: '%s', sess_type: '%s'", filename, sess_type)
 147         handler = "config_{}".format(sess_type)
 148 
 149         if filename is None:
 150             logger.debug("Popping file handler")
 151             cfgfile = self._file_handler("open", handler).retfile
 152             if not cfgfile:
 153                 logger.debug("No filename given")
 154                 return False
 155             filename = cfgfile.name
 156             cfgfile.close()
 157 
 158         if not os.path.isfile(filename):
 159             msg = "File does not exist: '{}'".format(filename)
 160             logger.error(msg)
 161             return False
 162         ext = os.path.splitext(filename)[1]
 163         if (sess_type == "project" and ext != ".fsw") or (sess_type == "task" and ext != ".fst"):
 164             logger.debug("Invalid file extension for session type: (sess_type: '%s', "
 165                          "extension: '%s')", sess_type, ext)
 166             return False
 167         logger.debug("Setting filename: '%s'", filename)
 168         self._filename = filename
 169         return True

 172     def _set_options(self, command=None):
 173         """ Set the GUI options based on the currently stored properties of :attr:`_options`
 174         and sets the active tab.
 175 
 176         Parameters
 177         ----------
 178         command: str, optional
 179             The tab to set the options for. If None then sets options for all tabs.
 180             Default: ``None``
 181         """
 182         opts = self._get_options_for_command(command) if command else self._cli_options
 183         logger.debug("command: %s, opts: %s", command, opts)
 184         if opts is None:
 185             logger.debug("No options found. Returning")
 186             return

 189         tab_name = self._options.get("tab_name", None) if command is None else command
 190         tab_name = tab_name if tab_name is not None else "extract"

 192         self._config.set_active_tab_by_name(tab_name)

 194     def _get_options_for_command(self, command):
 195         """ Return a single command's options from :attr:`_options` formatted consistently with
 196         an all options dict.
 197 
 198         Parameters
 199         ----------
 200         command: str
 201             The command to return the options for
 202 
 203         Returns
 204         -------
 205         dict: The options for a single command in the format {command: options}. If the command
 206         is not found then returns ``None``
 207         """
 208         logger.debug(command)
 209         opts = self._options.get(command, None)
 210         retval = {command: opts}
 211         if not opts:
 212             self._config.tk_vars["consoleclear"].set(True)
 213             logger.info("No  %s section found in file", command)
 214             retval = None
 215         logger.debug(retval)
 216         return retval

 218     def _set_gui_state_for_command(self, command, options):
 219         """ Set the GUI state for the given command.
 220 
 221         Parameters
 222         ----------
 223         command: str
 224             The tab to set the options for
 225         options: dict
 226             The option values to set the GUI to
 227         """
 228         logger.debug("command: %s: options: %s", command, options)
 229         if not options:
 230             logger.debug("No options provided, not updating GUI")
 231             return
 232         for srcopt, srcval in options.items():
 233             optvar = self._config.cli_opts.get_one_option_variable(command, srcopt)
 234             if not optvar:
 235                 continue
 236             logger.trace("setting option: (srcopt: %s, optvar: %s, srcval: %s)",
 237                          srcopt, optvar, srcval)
 238             optvar.set(srcval)

 255     # RECENT FILE HANDLING
 256     def _add_to_recent(self, command=None):
 257         """ Add the file for this session to the recent files list.
 258 
 259         Parameters
 260         ----------
 261         command: str, optional
 262             The command that this session relates to. If `None` then the whole project is added.
 263             Default: ``None``
 264         """
 265         logger.debug(command)
 266         if self._filename is None:
 267             logger.debug("No filename for selected file. Not adding to recent.")
 268             return
 269         recent_filename = os.path.join(self._config.pathcache, ".recent.json")
 270         logger.debug("Adding to recent files '%s': (%s, %s)",
 271                      recent_filename, self._filename, command)
 272         if not os.path.exists(recent_filename) or os.path.getsize(recent_filename) == 0:
 273             logger.debug("Starting with empty recent_files list")
 274             recent_files = []
 275         else:
 276             logger.debug("loading recent_files list: %s", recent_filename)
 277             recent_files = self._serializer.load(recent_filename)
 278         logger.debug("Initial recent files: %s", recent_files)
 279         recent_files = self._del_from_recent(self._filename, recent_files)
 280         ftype = "project" if command is None else command
 281         recent_files.insert(0, (self._filename, ftype))
 282         recent_files = recent_files[:20]
 283         logger.debug("Final recent files: %s", recent_files)
 284         self._serializer.save(recent_filename, recent_files)
 285 
 286     def _del_from_recent(self, filename, recent_files=None, save=False):
 287         """ Remove an item from the recent files list.
 288 
 289         Parameters
 290         ----------
 291         filename: str
 292             The filename to be removed from the recent files list
 293         recent_files: list, optional
 294             If the recent files list has already been loaded, it can be passed in to avoid
 295             loading again. If ``None`` then load the recent files list from disk. Default: ``None``
 296         save: bool, optional
 297             Whether the recent files list should be saved after removing the file. ``True`` saves
 298             the file, ``False`` does not. Default: ``False``
 299         """
 300         recent_filename = os.path.join(self._config.pathcache, ".recent.json")
 301         if recent_files is None:
 302             logger.debug("Loading file list from disk: %s", recent_filename)
 303             if not os.path.exists(recent_filename) or os.path.getsize(recent_filename) == 0:
 304                 logger.debug("No recent file list")
 305                 return None
 306             recent_files = self._serializer.load(recent_filename)
 307         filenames = [recent[0] for recent in recent_files]
 308         if filename in filenames:
 309             idx = filenames.index(filename)
 310             logger.debug("Removing from recent file list: %s", filename)
 311             del recent_files[idx]
 312             if save:
 313                 logger.debug("Saving recent files list: %s", recent_filename)
 314                 self._serializer.save(recent_filename, recent_files)
 315         else:
 316             logger.debug("Filename '%s' does not appear in recent file list", filename)
 317         return recent_files

 335         """ Load GUI options from :attr:`_filename` location and set to :attr:`_options`.
 336

 341         if self._file_exists:

 343             self._options = self._serializer.load(self._filename)

 351     def _check_valid_choices(self):
 352         """ Check whether the loaded file has any selected combo/radio/multi-option values that are
 353         no longer valid and remove them so that they are not passed into faceswap. """
 354         for command, options in self._selected_to_choices.items():
 355             for option, data in options.items():
 356                 if ((data["is_multi"] and all(v in data["choices"] for v in data["value"].split()))
 357                         or not data["is_multi"] and data["value"] in data["choices"]):
 358                     continue
 359                 if data["is_multi"]:
 360                     val = " ".join([v for v in data["value"].split() if v in data["choices"]])
 361                 else:
 362                     val = ""
 363                 val = self._default_options[command][option] if not val else val
 364                 logger.debug("Updating invalid value to default: (command: '%s', option: '%s', "
 365                              "original value: '%s', new value: '%s')", command, option,
 366                              self._options[command][option], val)
 367                 self._options[command][option] = val

 369     def _save_as_to_filename(self, session_type):
 370         """ Set :attr:`_filename` from a save as dialog.
 371 
 372         Parameters
 373         ----------
 374         session_type: ['all', 'task', 'project']
 375             The type of session to pop the save as dialog for. Limits the allowed filetypes

 377         Returns
 378         -------
 379         bool:

 382         logger.debug("Popping save as file handler. session_type: '%s'", session_type)

 387                                      title=title,

 391             return False
 392         self._filename = cfgfile.name
 393         logger.debug("Set filename: (session_type: '%s', filename: '%s'",
 394                      session_type, self._filename)
 395         cfgfile.close()
 396         return True

 398     def _save(self, command=None):
 399         """ Collect the options in the current GUI state and save.
 400 
 401         Obtains the current options set in the GUI with the selected tab and applies them to
 402         :attr:`_options`. Saves :attr:`_options` to :attr:`_filename`. Resets :attr:_modified_vars
 403         for either the given command or all commands,
 404 
 405         Parameters
 406         ----------
 407         command: str, optional
 408             The tab to collect the current state for. If ``None`` then collects the current
 409             state for all tabs. Default: ``None``
 410         """
 411         self._options = self._current_gui_state(command)
 412         self._options["tab_name"] = self._active_tab
 413         logger.debug("Saving options: (filename: %s, options: %s", self._filename, self._options)
 414         self._serializer.save(self._filename, self._options)
 415         self._reset_modified_var(command)
 416         self._add_to_recent(command)

 419 class Tasks(_GuiSession):
 420     """ Faceswap ``.fst`` Task File handling.
 421 
 422     Faceswap tasks handle the management of each individual task tab in the GUI. Unlike
 423     :class:`Projects`, Tasks contains all the active tasks currently running, rather than an
 424     individual task.
 425 
 426     Parameters
 427     ----------
 428     config: :class:`lib.gui.utils.Config`
 429         The master GUI config
 430     file_handler: :class:`lib.gui.utils.FileHandler`
 431         A file handler object
 432     """
 433     def __init__(self, config, file_handler):
 434         super().__init__(config, file_handler)
 435         self._tasks = dict()
 436 
 437     @property
 438     def _is_project(self):
 439         """ str: ``True`` if all tasks are from an overarching session project else ``False``."""
 440         retval = False if not self._tasks else all(v.get("is_project", False)
 441                                                    for v in self._tasks.values())
 442         return retval
 443 
 444     @property
 445     def _project_filename(self):
 446         """ str: The overarching session project filename."""
 447         fname = None
 448         if not self._is_project:
 449             return fname
 450 
 451         for val in self._tasks.values():
 452             fname = val["filename"]
 453             break
 454         return fname
 455 
 456     def load(self, *args,  # pylint:disable=unused-argument
 457              filename=None, current_tab=True):
 458         """ Load a task into this :class:`Tasks` class.
 459 
 460         Tasks can be loaded from project ``.fsw`` files or task ``.fst`` files, depending on where
 461         this function is being called from.
 462 
 463         Parameters
 464         ----------
 465         *args: tuple
 466             Unused, but needs to be present for arguments passed by tkinter event handling
 467         filename: str, optional
 468             If a filename is passed in, This will be used, otherwise a file handler will be
 469             launched to select the relevant file.
 470         current_tab: bool, optional
 471             ``True`` if the task to be loaded must be for the currently selected tab. ``False``
 472             if loading a task into any tab. If current_tab is `True` then tasks can be loaded from
 473             ``.fsw`` and ``.fst`` files, otherwise they can only be loaded from ``.fst`` files.
 474             Default: ``True``
 475         """
 476         logger.debug("Loading task config: (filename: '%s', current_tab: '%s')",
 477                      filename, current_tab)
 478 
 479         # Option to load specific task from project files:
 480         sess_type = "all" if current_tab else "task"
 481 
 482         is_legacy = (not self._is_project and
 483                      filename is not None and sess_type == "task" and
 484                      os.path.splitext(filename)[1] == ".fsw")
 485         if is_legacy:
 486             logger.debug("Legacy task found: '%s'", filename)
 487             filename = self._update_legacy_task(filename)
 488 
 489         filename_set = self._set_filename(filename, sess_type=sess_type)
 490         if not filename_set:
 491             return
 492         loaded = self._load()
 493         if not loaded:
 494             return
 495 
 496         command = self._active_tab if current_tab else self._stored_tab_name
 497         command = self._get_lone_task() if command is None else command
 498         if command is None:
 499             logger.error("Unable to determine task from the given file: '%s'", filename)
 500             return
 501         if command not in self._options:
 502             logger.error("No '%s' task in '%s'", command, self._filename)
 503             return
 504 
 505         self._set_options(command)
 506         self._add_to_recent(command)
 507 
 508         if self._is_project:
 509             self._filename = self._project_filename
 510         elif self._filename.endswith(".fsw"):
 511             self._filename = None
 512 
 513         self._add_task(command)
 514         if is_legacy:
 515             self.save()
 516 
 517         logger.debug("Loaded task config: (command: '%s', filename: '%s')", command, filename)
 518 
 519     def _update_legacy_task(self, filename):
 520         """ Update legacy ``.fsw`` tasks to ``.fst`` tasks.
 521 
 522         Tasks loaded from the recent files menu may be passed in with a ``.fsw`` extension.
 523         This renames the file and removes it from the recent file list.
 524 
 525         Parameters
 526         ----------
 527         filename: str
 528             The filename of the `.fsw` file that needs converting
 529 
 530         Returns
 531         -------
 532         str:
 533             The new filename of the updated tasks file
 534         """
 535         # TODO remove this code after a period of time. Implemented November 2019
 536         logger.debug("original filename: '%s'", filename)
 537         fname, ext = os.path.splitext(filename)
 538         if ext != ".fsw":
 539             logger.debug("Not a .fsw file: '%s'", filename)
 540             return filename
 541 
 542         new_filename = "{}.fst".format(fname)
 543         logger.debug("Renaming '%s' to '%s'", filename, new_filename)
 544         os.rename(filename, new_filename)
 545         self._del_from_recent(filename, save=True)
 546         logger.debug("new filename: '%s'", new_filename)
 547         return new_filename
 548 
 549     def save(self, save_as=False):
 550         """ Save the current GUI state for the active tab to a ``.fst`` faceswap task file.
 551 
 552         Parameters
 553         ----------
 554         save_as: bool, optional
 555             Whether to save to the stored filename, or pop open a file handler to ask for a
 556             location. If there is no stored filename, then a file handler will automatically be
 557             popped.
 558         """
 559         logger.debug("Saving config...")
 560         self._set_active_task()
 561         save_as = save_as or self._is_project or self._filename is None
 562 
 563         if save_as and not self._save_as_to_filename("task"):
 564             return
 565 
 566         command = self._active_tab
 567         self._save(command=command)
 568         self._add_task(command)
 569         if not save_as:
 570             logger.info("Saved project to: '%s'", self._filename)
 571         else:
 572             logger.debug("Saved project to: '%s'", self._filename)

 590     def _add_task(self, command):

 592

 596         stays with the project.
 597 
 598         Parameters
 599         ----------
 600         command: str
 601             The tab that pertains to the currently active task
 602 
 603         """
 604         self._tasks[command] = dict(filename=self._filename,
 605                                     options=self._options,
 606                                     is_project=self._is_project)

 608     def clear_tasks(self):
 609         """ Clears all of the stored tasks.
 610 
 611         This is required when loading a task stored in a legacy project file, and is only to be
 612         called by :class:`Project` when a project has been loaded which is in fact a task.
 613         """
 614         logger.debug("Clearing stored tasks")
 615         self._tasks = dict()
 616 
 617     def add_project_task(self, filename, command, options):
 618         """ Add an individual task from a loaded :class:`Project` to the internal :attr:`_tasks`
 619         dict.
 620 
 621         Project tasks take priority over any other tasks, so the individual tasks from a new
 622         project must be placed in the _tasks dict.
 623 
 624         Parameters
 625         ----------
 626         filename: str
 627             The filename of the session project file
 628         command: str
 629             The tab that this task's options belong to
 630         options: dict
 631             The options for this task loaded from the project
 632         """
 633         self._tasks[command] = dict(filename=filename, options=options, is_project=True)
 634 
 635     def _set_active_task(self, command=None):
 636         """ Set the active :attr:`_filename` and :attr:`_options` to currently selected tab's
 637         options.
 638 
 639         Parameters
 640         ----------
 641         command: str, optional
 642             If a command is passed in then set the given tab to active, If this is none set the tab
 643             which currently has focus to active. Default: ``None``
 644         """
 645         logger.debug(command)
 646         command = self._active_tab if command is None else command
 647         task = self._tasks.get(command, None)
 648         if task is None:
 649             self._filename, self._options = (None, None)
 650         else:
 651             self._filename, self._options = (task.get("filename", None), task.get("options", None))
 652         logger.debug("tab: %s, filename: %s, options: %s",
 653                      self._active_tab, self._filename, self._options)

 656 class Project(_GuiSession):
 657     """ Faceswap ``.fsw`` Project File handling.
 658 
 659     Faceswap projects handle the management of all task tabs in the GUI and updates
 660     the main Faceswap title bar with the project name and modified state.
 661 
 662     Parameters
 663     ----------
 664     config: :class:`lib.gui.utils.Config`
 665         The master GUI config
 666     file_handler: :class:`lib.gui.utils.FileHandler`
 667         A file handler object
 668     """
 669 
 670     def __init__(self, config, file_handler):
 671         super().__init__(config, file_handler)
 672         self._update_root_title()
 673 
 674     @property
 675     def filename(self):
 676         """ str: The currently active project filename. """
 677         return self._filename
 678 
 679     @property
 680     def cli_options(self):
 681         """ dict: the raw cli options from :attr:`_options` with project fields removed. """
 682         return self._cli_options
 683 
 684     @property
 685     def _project_modified(self):
 686         """bool: ``True`` if the project has been modified otherwise ``False``. """
 687         return any([var.get() for var in self._modified_vars.values()])
 688 
 689     @property
 690     def _tasks(self):
 691         """ :class:`Tasks`: The current session's :class:``Tasks``. """
 692         return self._config.tasks
 693 
 694     def set_default_options(self):
 695         """ Set the default options. The Default GUI options are stored on Faceswap startup.
 696 
 697         Exposed as the :attr:`_default_options` for a project cannot be set until after the main
 698         Command Tabs have been loaded.
 699         """
 700         logger.debug("Setting options to default")
 701         self._options = self._default_options
 702 
 703     # MODIFIED STATE CALLBACK
 704     def set_modified_callback(self):
 705         """ Adds a callback to each of the :attr:`_modified_vars` tkinter variables
 706         When one of these variables is changed, triggers :func:`_modified_callback`
 707         with the command that was changed.
 708 
 709         This is exposed as the callback can only be added after the main Command Tabs have
 710         been drawn, and their options' initial values have been set.
 711 
 712         """
 713         for key, tkvar in self._modified_vars.items():
 714             logger.debug("Adding callback for tab: %s", key)
 715             tkvar.trace("w", self._modified_callback)
 716 
 717     def _modified_callback(self, *args):  # pylint:disable=unused-argument
 718         """ Update the project modified state on a GUI modification change and
 719         update the Faceswap title bar. """
 720         if self._project_modified and self._current_gui_state() == self._cli_options:
 721             logger.debug("Project is same as stored. Setting modified to False")
 722             self._reset_modified_var()
 723 
 724         if self._modified != self._project_modified:
 725             logger.debug("Updating project state from variable: (modified: %s)",
 726                          self._project_modified)
 727             self._modified = self._project_modified
 728             self._update_root_title()
 729 
 730     def load(self, *args, filename=None):  # pylint:disable=unused-argument
 731         """ Load a project from a saved ``.fsw`` project file.
 732 
 733         Parameters
 734         ----------
 735         *args: tuple
 736             Unused, but needs to be present for arguments passed by tkinter event handling
 737         filename: str, optional
 738             If a filename is passed in, This will be used, otherwise a file handler will be
 739             launched to select the relevant file.
 740         """
 741         logger.debug("Loading project config: (filename: '%s')", filename)
 742         filename_set = self._set_filename(filename, sess_type="project")
 743 
 744         if not filename_set:
 745             logger.debug("No filename set")
 746             return
 747         loaded = self._load()
 748         if not loaded:
 749             logger.debug("Options not loaded")
 750             return
 751 
 752         # Legacy .fsw files could store projects or tasks. Check if this is a legacy file
 753         # and hand off file to Tasks if necessary
 754         command = self._get_lone_task()
 755         legacy = command is not None
 756         if legacy:
 757             self._handoff_legacy_task()
 758             return
 759 
 760         self._set_options()
 761         self._update_tasks()
 762         self._add_to_recent()
 763         self._reset_modified_var()
 764         self._update_root_title()
 765         logger.debug("Loaded project config: (command: '%s', filename: '%s')", command, filename)
 766 
 767     def _handoff_legacy_task(self):
 768         """ Update legacy tasks saved with the old file extension ``.fsw`` to tasks ``.fst``.
 769 
 770         Hands off file handling to :class:`Tasks` and resets project to default.
 771         """
 772         logger.debug("Updating legacy task '%s", self._filename)
 773         filename = self._filename
 774         self._filename = None
 775         self.set_default_options()
 776         self._tasks.clear_tasks()
 777         self._tasks.load(filename=filename, current_tab=False)
 778         logger.debug("Updated legacy task and reset project")
 779 
 780     def _update_tasks(self):
 781         """ Add the tasks from the loaded project to the :class:`Tasks` class. """
 782         for key, val in self._cli_options.items():
 783             opts = {key: val}
 784             opts["tab_name"] = key
 785             self._tasks.add_project_task(self._filename, key, opts)
 786 
 787     def reload(self, *args):  # pylint:disable=unused-argument
 788         """ Reset all GUI's option tabs to their last saved state.
 789 
 790         Parameters
 791         ----------
 792         *args: tuple
 793             Unused, but needs to be present for arguments passed by tkinter event handling
 794         """
 795         if self._options is None:
 796             logger.info("No active project to reload")
 797             return
 798         logger.debug("Reloading project")
 799         self._set_options()
 800         self._update_tasks()
 801         self._reset_modified_var()
 802         self._update_root_title()
 803 
 804     def _update_root_title(self):
 805         """ Update the root Window title with the project name. Add a asterisk
 806         if the file is modified. """
 807         text = "<untitled project>" if self._basename is None else self._basename
 808         text += "*" if self._modified else ""
 809         self._config.set_root_title(text=text)
 810 
 811     def save(self, *args, save_as=False):  # pylint:disable=unused-argument
 812         """ Save the current GUI state to a ``.fsw`` project file.
 813 
 814         Parameters
 815         ----------
 816         *args: tuple
 817             Unused, but needs to be present for arguments passed by tkinter event handling
 818         save_as: bool, optional
 819             Whether to save to the stored filename, or pop open a file handler to ask for a
 820             location. If there is no stored filename, then a file handler will automatically be
 821             popped.
 822         """
 823         logger.debug("Saving config as...")
 824 
 825         save_as = save_as or self._filename is None
 826         if save_as and not self._save_as_to_filename("project"):
 827             return
 828         self._save()
 829         self._update_tasks()
 830         self._update_root_title()
 831         if not save_as:
 832             logger.info("Saved project to: '%s'", self._filename)
 833         else:
 834             logger.debug("Saved project to: '%s'", self._filename)
 835 
 836     def new(self, *args):  # pylint:disable=unused-argument
 837         """ Create a new project with default options.
 838 
 839         Pops a file handler to select location.
 840 
 841         Parameters
 842         ----------
 843         *args: tuple
 844             Unused, but needs to be present for arguments passed by tkinter event handling
 845         """
 846         logger.debug("Creating new project")
 847         if not self.confirm_close():
 848             logger.debug("Creating new project cancelled")
 849             return
 850 
 851         cfgfile = self._file_handler("save",
 852                                      "config_project",
 853                                      title="New Project...",
 854                                      initialdir=self._basename).retfile
 855         if not cfgfile:
 856             logger.debug("No filename selected")
 857             return
 858         self._filename = cfgfile.name
 859         cfgfile.close()
 860 
 861         self.set_default_options()
 862         self._config.cli_opts.reset()
 863         self._save()
 864         self._update_root_title()
 865 
 866     def close(self, *args):  # pylint:disable=unused-argument
 867         """ Clear the current project and set all options to default.
 868 
 869         Parameters
 870         ----------
 871         *args: tuple
 872             Unused, but needs to be present for arguments passed by tkinter event handling
 873         """
 874         logger.debug("Close requested")
 875         if not self.confirm_close():
 876             logger.debug("Close cancelled")
 877             return
 878         self._config.cli_opts.reset()
 879         self._filename = None
 880         self.set_default_options()
 881         self._reset_modified_var()
 882         self._update_root_title()
 883         self._config.set_active_tab_by_name(self._config.user_config_dict["tab"])
 884 
 885     def confirm_close(self):
 886         """ Pop a message box to get confirmation that an unsaved project should be closed
 887 
 888         Returns
 889         -------
 890         bool: ``True`` if user confirms close, ``False`` if user cancels close
 891         """
 892         if not self._modified:
 893             logger.debug("Project is not modified")
 894             return True
 895         confirmtxt = "You have unsaved changes.\n\nAre you sure you want to close the project?"
 896         if messagebox.askokcancel("Close", confirmtxt, default="cancel", icon="warning"):
 897             logger.debug("Close Cancelled")
 898             return True
 899         logger.debug("Close confirmed")
 900         return False

 903 class LastSession(_GuiSession):
 904     """ Faceswap Last Session handling.
 905 
 906     Faceswap :class:`LastSession` handles saving the state of the Faceswap GUI at close and
 907     reloading the state  at launch.
 908 
 909     Last Session behavior can be configured in :file:`config.gui.ini`.
 910 
 911     Parameters
 912     ----------
 913     config: :class:`lib.gui.utils.Config`
 914         The master GUI config
 915     """
 916 
 917     def __init__(self, config):
 918         super().__init__(config)
 919         self._filename = os.path.join(self._config.pathcache, ".last_session.json")
 920         if not self._enabled:
 921             return
 922 
 923         if self._save_option == "prompt":
 924             self.ask_load()
 925         elif self._save_option == "always":
 926             self.load()
 927 
 928     @property
 929     def _save_option(self):
 930         """ str: The user config autosave option. """
 931         return self._config.user_config_dict.get("autosave_last_session", "never")
 932 
 933     @property
 934     def _enabled(self):
 935         """ bool: ``True`` if autosave is enabled otherwise ``False``. """
 936         return self._save_option != "never"
 937 
 938     def from_dict(self, options):
 939         """ Set the :attr:`_options` property based on the given options dictionary
 940         and update the GUI to use these values.
 941 
 942         This function is required for reloading the GUI state when the GUI has been force
 943         refreshed on a config change.
 944 
 945         Parameters
 946         ----------
 947         options: dict
 948             The options to set. Should be the output of :func:`to_dict`
 949         """
 950         logger.debug("Setting options from dict: %s", options)
 951         self._options = options
 952         self._set_options()
 953 
 954     def to_dict(self):
 955         """ Collect the current GUI options and place them in a dict for retrieval or storage.
 956 
 957         This function is required for reloading the GUI state when the GUI has been force
 958         refreshed on a config change.
 959 
 960         Returns
 961         -------
 962         dict: The current cli options ready for saving or retrieval by :func:`from_dict`
 963         """
 964         opts = self._current_gui_state()
 965         logger.debug("Collected opts: %s", opts)
 966         if not opts or opts == self._default_options:
 967             logger.debug("Default session, or no opts found. Not saving last session.")
 968             return None
 969         opts["tab_name"] = self._active_tab
 970         opts["project"] = self._config.project.filename
 971         logger.debug("Added project items: %s", {k: v for k, v in opts.items()
 972                                                  if k in ("tab_name", "project")})
 973         return opts
 974 
 975     def ask_load(self):
 976         """ Pop a message box to ask the user if they wish to load their last session.  """
 977         if not self._file_exists:
 978             logger.debug("No last session file found")
 979         elif tk.messagebox.askyesno("Last Session", "Load last session?"):
 980             logger.debug("Loading last session at user request")
 981             self.load()
 982         else:
 983             logger.debug("Not loading last session at user request")
 984             logger.debug("Deleting LastSession file")
 985             os.remove(self._filename)
 986 
 987     def load(self):
 988         """ Load the last session.
 989 
 990         Loads the last saved session options. Checks if a previous project was loaded
 991         and whether there have been changes since the last saved version of the project.
 992         Sets the display and :class:`Project` and :class:`Task` objects accordingly.
 993         """
 994         loaded = self._load()
 995         if not loaded:
 996             return
 997         needs_update = self._set_project()
 998         if needs_update:
 999             self._set_options()
1000 
1001     def _set_project(self):
1002         """ Set the :class:`Project` if session is resuming from one.
1003 
1004         Returns
1005         -------
1006         bool:
1007             ``True`` If the GUI still needs to be updated from the last session, ``False`` if
1008             the returned GUI state is the last session
1009         """
1010         if self._options.get("project", None) is None:
1011             logger.debug("No project stored")
1012             retval = True
1013         else:
1014             logger.debug("Loading stored project")
1015             self._config.project.load(filename=self._options["project"])
1016             retval = self._cli_options != self._config.project.cli_options
1017 
1018         logger.debug("Needs update: %s", retval)
1019         return retval
1020 
1021     def save(self):
1022         """ Save a snapshot of currently set GUI config options.
1023 
1024         Called on Faceswap shutdown.
1025         """
1026         if not self._enabled:
1027             logger.debug("LastSession not enabled")
1028             if os.path.exists(self._filename):
1029                 logger.debug("Deleting existing LastSession file")
1030                 os.remove(self._filename)
1031             return
1032 
1033         opts = self.to_dict()
1034         if opts is None and os.path.exists(self._filename):
1035             logger.debug("Last session default or blank. Clearing saved last session.")
1036             os.remove(self._filename)
1037         if opts is not None:
1038             self._serializer.save(self._filename, opts)
1039             logger.debug("Saved last session. (filename: '%s', opts: %s", self._filename, opts)
