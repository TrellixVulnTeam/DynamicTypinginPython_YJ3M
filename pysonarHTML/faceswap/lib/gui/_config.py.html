  77 def get_commands():
  78     """ Return commands formatted for GUI """
  79     root_path = os.path.abspath(os.path.dirname(sys.argv[0]))
  80     command_path = os.path.join(root_path, "scripts")
  81     tools_path = os.path.join(root_path, "tools")
  82     commands = [os.path.splitext(item)[0] for item in os.listdir(command_path)
  83                 if os.path.splitext(item)[1] == ".py"
  84                 and os.path.splitext(item)[0] not in ("gui", "fsmedia")
  85                 and not os.path.splitext(item)[0].startswith("_")]
  86     tools = [os.path.splitext(item)[0] for item in os.listdir(tools_path)
  87              if os.path.splitext(item)[1] == ".py"
  88              and os.path.splitext(item)[0] not in ("gui", "cli")
  89              and not os.path.splitext(item)[0].startswith("_")]
  90     return commands + tools

  93 def get_clean_fonts():
  94     """ Return the font list with any @prefixed or non-Unicode characters stripped
  95         and default prefixed """
  96     cleaned_fonts = sorted([fnt for fnt in font.families()
  97                             if not fnt.startswith("@") and not any([ord(c) > 127 for c in fnt])])
  98     return ["default"] + cleaned_fonts
