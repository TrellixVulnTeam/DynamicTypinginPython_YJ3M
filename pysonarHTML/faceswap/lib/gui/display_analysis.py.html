 109     def load_session(self, fullpath=None):
 110         """ Load previously saved sessions """
 111         logger.debug("Loading session")
 112         if fullpath is None:
 113             fullpath = FileHandler("filename", "state").retfile
 114             if not fullpath:
 115                 return
 116         self.clear_session()
 117         logger.debug("state_file: '%s'", fullpath)
 118         model_dir, state_file = os.path.split(fullpath)
 119         logger.debug("model_dir: '%s'", model_dir)
 120         model_name = self.get_model_name(model_dir, state_file)
 121         if not model_name:
 122             return
 123         self.session = Session(model_dir=model_dir, model_name=model_name)
 124         self.session.initialize_session(is_training=False)
 125         msg = fullpath
 126         if len(msg) > 70:
 127             msg = "...{}".format(msg[-70:])
 128         self.set_session_summary(msg)

 142     def reset_session(self):
 143         """ Reset currently training sessions """
 144         logger.debug("Reset current training session")
 145         self.clear_session()
 146         session = get_config().session
 147         if not session.initialized:
 148             logger.debug("Training not running")
 149             return
 150         if session.logging_disabled:
 151             logger.trace("Logging disabled. Not triggering analysis update")
 152             return
 153         msg = "Currently running training session"
 154         self.session = session
 155         # Reload the state file to get approx currently training iterations
 156         self.session.load_state_file()
 157         self.set_session_summary(msg)
 158 
 159     def set_session_summary(self, message):
 160         """ Set the summary data and info message """
 161         if self.thread is None:
 162             logger.debug("Setting session summary. (message: '%s')", message)
 163             self.thread = LongRunningTask(target=self.summarise_data,
 164                                           args=(self.session, ),
 165                                           widget=self)
 166             self.thread.start()
 167             self.after(1000, lambda msg=message: self.set_session_summary(msg))
 168         elif not self.thread.complete.is_set():
 169             logger.debug("Data not yet available")
 170             self.after(1000, lambda msg=message: self.set_session_summary(msg))
 171         else:
 172             logger.debug("Retrieving data from thread")
 173             result = self.thread.get_result()
 174             if result is None:
 175                 logger.debug("No result from session summary. Clearing analysis view")
 176                 self.clear_session()
 177                 return
 178             self.summary = result
 179             self.thread = None
 180             self.set_info("Session: {}".format(message))
 181             self.stats.session = self.session
 182             self.stats.tree_insert_data(self.summary)

 191         logger.debug("Clearing session")
 192         if self.session is None:
 193             logger.trace("No session loaded. Returning")
 194             return
 195         self.summary = None

 201     def save_session(self):
 202         """ Save sessions stats to csv """
 203         logger.debug("Saving session")
 204         if not self.summary:
 205             logger.debug("No summary data loaded. Nothing to save")
 206             print("No summary data loaded. Nothing to save")
 207             return
 208         savefile = FileHandler("save", "csv").retfile
 209         if not savefile:
 210             logger.debug("No save file. Returning")
 211             return
 212 
 213         logger.debug("Saving to: '%s'", savefile)
 214         fieldnames = sorted(key for key in self.summary[0].keys())
 215         with savefile as outfile:
 216             csvout = csv.DictWriter(outfile, fieldnames)
 217             csvout.writeheader()
 218             for row in self.summary:
 219                 csvout.writerow(row)

 362     def select_item(self, event):

 364             the selected item or launch graph """
 365         region = self.tree.identify("region", event.x, event.y)
 366         selection = self.tree.focus()
 367         values = self.tree.item(selection, "values")
 368         if values:
 369             logger.debug("Selected values: %s", values)
 370             self.selected_id.set(values[0])

 372                 datapoints = int(values[self.columns.index("iterations")])

 375     def check_valid_data(self, values):
 376         """ Check there is valid data available for popping up a graph """
 377         col_indices = [self.columns.index("batch"), self.columns.index("iterations")]
 378         for idx in col_indices:
 379             if (isinstance(values[idx], int) or values[idx].isdigit()) and int(values[idx]) == 0:
 380                 logger.warning("No data to graph for selected session")
 381                 return False
 382         return True

 421     def data_popup_get_position(self):
 422         """ Get the position of the next window """
 423         logger.debug("getting poup position")
 424         init_pos = [120, 120]
 425         pos = init_pos
 426         while True:
 427             if pos not in self.popup_positions:
 428                 self.popup_positions.append(pos)
 429                 break
 430             pos = [item + 200 for item in pos]
 431             init_pos, pos = self.data_popup_check_boundaries(init_pos, pos)
 432         logger.debug("Position: %s", pos)
 433         return pos

 435     def data_popup_check_boundaries(self, initial_position, position):
 436         """ Check that the popup remains within the screen boundaries """
 437         logger.debug("Checking poup boundaries: (initial_position: %s, position: %s)",
 438                      initial_position, position)
 439         boundary_x = self.winfo_screenwidth() - 120
 440         boundary_y = self.winfo_screenheight() - 120
 441         if position[0] >= boundary_x or position[1] >= boundary_y:
 442             initial_position = [initial_position[0] + 50, initial_position[1]]
 443             position = initial_position
 444         logger.debug("Returning poup boundaries: (initial_position: %s, position: %s)",
 445                      initial_position, position)
 446         return initial_position, position

 740     def compile_display_data(self):
 741         """ Compile the data to be displayed """
 742         if self.thread is None:
 743             logger.debug("Compiling Display Data in background thread")
 744             loss_keys = [key for key, val in self.vars["loss_keys"].items()
 745                          if val.get()]
 746             logger.debug("Selected loss_keys: %s", loss_keys)
 747 
 748             selections = self.selections_to_list()
 749 
 750             if not self.check_valid_selection(loss_keys, selections):
 751                 logger.warning("No data to display. Not refreshing")
 752                 return False
 753             self.vars["status"].set("Loading Data...")
 754             kwargs = dict(session=self.session,
 755                           display=self.vars["display"].get(),
 756                           loss_keys=loss_keys,
 757                           selections=selections,
 758                           avg_samples=self.vars["avgiterations"].get(),
 759                           smooth_amount=self.vars["smoothamount"].get(),
 760                           flatten_outliers=self.vars["outliers"].get(),
 761                           is_totals=self.is_totals)
 762             self.thread = LongRunningTask(target=self.get_display_data, kwargs=kwargs, widget=self)
 763             self.thread.start()
 764             self.after(1000, self.compile_display_data)
 765             return True
 766         if not self.thread.complete.is_set():
 767             logger.debug("Popup Data not yet available")
 768             self.after(1000, self.compile_display_data)
 769             return True
 770 
 771         logger.debug("Getting Popup from background Thread")
 772         self.display_data = self.thread.get_result()
 773         self.thread = None
 774         if not self.check_valid_data():
 775             logger.warning("No valid data to display. Not refreshing")
 776             self.vars["status"].set("")
 777             return False
 778         logger.debug("Compiled Display Data")
 779         self.vars["buildgraph"].set(True)
 780         return True

 783     def get_display_data(**kwargs):
 784         """ Get the display data in a LongRunningTask """
 785         return Calculations(**kwargs)

 787     def check_valid_selection(self, loss_keys, selections):
 788         """ Check that there will be data to display """
 789         display = self.vars["display"].get().lower()
 790         logger.debug("Validating selection. (loss_keys: %s, selections: %s, display: %s)",
 791                      loss_keys, selections, display)
 792         if not selections or (display == "loss" and not loss_keys):
 793             return False
 794         return True

 796     def check_valid_data(self):
 797         """ Check that the selections holds valid data to display
 798             NB: len-as-condition is used as data could be a list or a numpy array
 799         """
 800         logger.debug("Validating data. %s",
 801                      {key: len(val) for key, val in self.display_data.stats.items()})
 802         if any(len(val) == 0  # pylint:disable=len-as-condition
 803                for val in self.display_data.stats.values()):
 804             return False
 805         return True

 807     def selections_to_list(self):
 808         """ Compile checkbox selections to list """
 809         logger.debug("Compiling selections to list")
 810         selections = list()
 811         for key, val in self.vars.items():
 812             if (isinstance(val, tk.BooleanVar)
 813                     and key != "outliers"
 814                     and val.get()):
 815                 selections.append(key)
 816         logger.debug("Compiling selections to list: %s", selections)
 817         return selections

 819     def graph_build(self, *args):  # pylint:disable=unused-argument
 820         """ Build the graph in the top right paned window """
 821         if not self.vars["buildgraph"].get():
 822             return
 823         self.vars["status"].set("Loading Data...")
 824         logger.debug("Building Graph")
 825         if self.graph is None:
 826             self.graph = SessionGraph(self.graph_frame,
 827                                       self.display_data,
 828                                       self.vars["display"].get(),
 829                                       self.vars["scale"].get())
 830             self.graph.pack(expand=True, fill=tk.BOTH)
 831             self.graph.build()
 832             self.graph_initialised = True
 833         else:
 834             self.graph.refresh(self.display_data,
 835                                self.vars["display"].get(),
 836                                self.vars["scale"].get())
 837         self.vars["status"].set("")
 838         self.vars["buildgraph"].set(False)
