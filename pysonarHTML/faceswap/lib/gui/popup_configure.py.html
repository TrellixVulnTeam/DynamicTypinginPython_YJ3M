  19 def popup_config(config, root):
  20     """ Close any open popup and open requested popup """
  21     if POPUP:
  22         p_key = list(POPUP.keys())[0]
  23         logger.debug("Closing open popup: '%s'", p_key)
  24         POPUP[p_key].destroy()
  25         del POPUP[p_key]
  26     window = ConfigurePlugins(config, root)
  27     POPUP[config[0]] = window

  30 class ConfigurePlugins(tk.Toplevel):
  31     """ Pop up for detailed graph/stats for selected session """
  32     def __init__(self, config, root):

  35         self._name, self.config = config

  51     def set_geometry(self):
  52         """ Set pop-up geometry """
  53         scaling_factor = get_config().scaling_factor
  54         pos_x = self._root.winfo_x() + 80
  55         pos_y = self._root.winfo_y() + 80
  56         width = int(600 * scaling_factor)
  57         height = int(400 * scaling_factor)
  58         logger.debug("Pop up Geometry: %sx%s, %s+%s", width, height, pos_x, pos_y)
  59         self.geometry("{}x{}+{}+{}".format(width, height, pos_x, pos_y))

  61     def get_config(self):
  62         """ Format config into a dict of ControlPanelOptions """
  63         logger.debug("Formatting Config for GUI")
  64         conf = dict()
  65         for section in self.config.config.sections():
  66             self.config.section = section
  67             category = section.split(".")[0]
  68             options = self.config.defaults[section]
  69             section = section.split(".")[-1]
  70             conf.setdefault(category, dict())[section] = OrderedDict()
  71             for key, val in options.items():
  72                 if key == "helptext":
  73                     self.plugin_info[section] = val
  74                     continue
  75                 conf[category][section][key] = ControlPanelOption(
  76                     title=key,
  77                     dtype=val["type"],
  78                     group=val["group"],
  79                     default=val["default"],
  80                     initial_value=self.config.config_dict.get(key, val["default"]),
  81                     choices=val["choices"],
  82                     is_radio=val["gui_radio"],
  83                     rounding=val["rounding"],
  84                     min_max=val["min_max"],
  85                     helptext=val["helptext"])
  86         logger.debug("Formatted Config for GUI: %s", conf)
  87         return conf

  89     def build(self):
  90         """ Build the config popup """
  91         logger.debug("Building plugin config popup")
  92         container = ttk.Notebook(self.page_frame)
  93         container.pack(fill=tk.BOTH, expand=True)
  94         categories = sorted(list(self.config_cpanel_dict.keys()))
  95         if "global" in categories:  # Move global to first item
  96             categories.insert(0, categories.pop(categories.index("global")))
  97         for category in categories:
  98             page = self.build_page(container, category)
  99             container.add(page, text=category.title())
 100 
 101         self.add_frame_separator()
 102         self.add_actions()
 103         logger.debug("Built plugin config popup")
 104 
 105     def build_page(self, container, category):
 106         """ Build a plugin config page """
 107         logger.debug("Building plugin config page: '%s'", category)
 108         plugins = sorted(list(key for key in self.config_cpanel_dict[category].keys()))
 109         panel_kwargs = dict(columns=2, max_columns=2, option_columns=2, blank_nones=False)
 110         if any(plugin != category for plugin in plugins):
 111             page = ttk.Notebook(container)
 112             page.pack(side=tk.TOP, fill=tk.BOTH, expand=True)
 113             for plugin in plugins:
 114                 cp_options = list(self.config_cpanel_dict[category][plugin].values())
 115                 frame = ControlPanel(page,
 116                                      cp_options,
 117                                      header_text=self.plugin_info[plugin],
 118                                      **panel_kwargs)
 119                 title = plugin[plugin.rfind(".") + 1:]
 120                 title = title.replace("_", " ").title()
 121                 page.add(frame, text=title)
 122         else:
 123             cp_options = list(self.config_cpanel_dict[category][plugins[0]].values())
 124             page = ControlPanel(container,
 125                                 cp_options,
 126                                 header_text=self.plugin_info[plugins[0]],
 127                                 **panel_kwargs)
 128 
 129         logger.debug("Built plugin config page: '%s'", category)
 130 
 131         return page

 140     def add_actions(self):
 141         """ Add Action buttons """
 142         logger.debug("Add action buttons")
 143         frame = ttk.Frame(self.page_frame)
 144         frame.pack(fill=tk.BOTH, padx=5, pady=5, side=tk.BOTTOM)
 145         btn_cls = ttk.Button(frame, text="Cancel", width=10, command=self.destroy)
 146         btn_cls.pack(padx=2, side=tk.RIGHT)
 147         Tooltip(btn_cls, text="Close without saving", wraplength=720)
 148         btn_ok = ttk.Button(frame, text="OK", width=10, command=self.save_config)
 149         btn_ok.pack(padx=2, side=tk.RIGHT)
 150         Tooltip(btn_ok, text="Close and save config", wraplength=720)
 151         btn_rst = ttk.Button(frame, text="Reset", width=10, command=self.reset)
 152         btn_rst.pack(padx=2, side=tk.RIGHT)
 153         Tooltip(btn_rst, text="Reset all plugins to default values", wraplength=720)

 156     def reset(self):
 157         """ Reset all config options to default """
 158         logger.debug("Resetting config")
 159         for section, items in self.config.defaults.items():
 160             logger.debug("Resetting section: '%s'", section)
 161             lookup = [section.split(".")[0], section.split(".")[-1]]
 162             for item, def_opt in items.items():
 163                 if item == "helptext":
 164                     continue
 165                 default = def_opt["default"]
 166                 logger.debug("Resetting: '%s' to '%s'", item, default)
 167                 self.config_cpanel_dict[lookup[0]][lookup[1]][item].set(default)
 168 
 169     def save_config(self):
 170         """ Save the config file """
 171         logger.debug("Saving config")
 172         options = {".".join((key, sect)) if sect != key else key: opts
 173                    for key, value in self.config_cpanel_dict.items()
 174                    for sect, opts in value.items()}
 175         new_config = ConfigParser(allow_no_value=True)
 176         for section, items in self.config.defaults.items():
 177             logger.debug("Adding section: '%s')", section)
 178             self.config.insert_config_section(section, items["helptext"], config=new_config)
 179             for item, def_opt in items.items():
 180                 if item == "helptext":
 181                     continue
 182                 new_opt = options[section][item].get()
 183                 logger.debug("Adding option: (item: '%s', default: %s new: '%s'",
 184                              item, def_opt, new_opt)
 185                 helptext = def_opt["helptext"]
 186                 helptext = self.config.format_help(helptext, is_section=False)
 187                 new_config.set(section, helptext)
 188                 new_config.set(section, item, str(new_opt))
 189         self.config.config = new_config
 190         self.config.save_config()
 191         logger.info("Saved config: '%s'", self.config.configfile)
 192         self.destroy()
 193 
 194         running_task = get_config().tk_vars["runningtask"].get()
 195         if self._name.lower() == "gui" and not running_task:
 196             self._root.rebuild()
 197         elif self._name.lower() == "gui" and running_task:
 198             logger.info("Can't redraw GUI whilst a task is running. GUI Settings will be applied "
 199                         "at the next restart.")
 200         logger.debug("Saved config")
