  63     def scan_for_plugin_configs(self):
  64         """ Scan for config.ini file locations """
  65         global _CONFIGS, _CONFIG_FILES  # pylint:disable=global-statement
  66         root_path = os.path.abspath(os.path.dirname(sys.argv[0]))
  67         plugins_path = os.path.join(root_path, "plugins")
  68         logger.debug("Scanning path: '%s'", plugins_path)
  69         configs = dict()
  70         for dirpath, _, filenames in os.walk(plugins_path):
  71             if "_config.py" in filenames:
  72                 plugin_type = os.path.split(dirpath)[-1]
  73                 config = self.load_config(plugin_type)
  74                 configs[plugin_type] = config
  75         logger.debug("Configs loaded: %s", sorted(list(configs.keys())))
  76         keys = list(configs.keys())
  77         for key in ("extract", "train", "convert"):
  78             if key in keys:
  79                 _CONFIG_FILES.append(keys.pop(keys.index(key)))
  80         _CONFIG_FILES.extend([key for key in sorted(keys)])
  81         _CONFIGS = configs
  82         return configs
  83 
  84     @staticmethod
  85     def load_config(plugin_type):
  86         """ Load the config to generate config file if it doesn't exist and get filename """
  87         # Load config to generate default if doesn't exist
  88         mod = ".".join(("plugins", plugin_type, "_config"))
  89         module = import_module(mod)
  90         config = module.Config(None)
  91         logger.debug("Found '%s' config at '%s'", plugin_type, config.configfile)
  92         return config

 175     def build_recent_menu(self):
 176         """ Load recent files into menu bar """
 177         logger.debug("Building Recent Files menu")
 178         serializer = get_serializer("json")
 179         menu_file = os.path.join(self._config.pathcache, ".recent.json")
 180         if not os.path.isfile(menu_file) or os.path.getsize(menu_file) == 0:
 181             self.clear_recent_files(serializer, menu_file)
 182         recent_files = serializer.load(menu_file)
 183         logger.debug("Loaded recent files: %s", recent_files)
 184         removed_files = []
 185         for recent_item in recent_files:
 186             filename, command = recent_item
 187             if not os.path.isfile(filename):
 188                 logger.debug("File does not exist. Flagging for removal: '%s'", filename)
 189                 removed_files.append(recent_item)
 190                 continue
 191             # Legacy project files didn't have a command stored
 192             command = command if command else "project"
 193             logger.debug("processing: ('%s', %s)", filename, command)
 194             if command.lower() == "project":
 195                 load_func = self._config.project.load
 196                 lbl = command
 197                 kwargs = dict(filename=filename)
 198             else:
 199                 load_func = self._config.tasks.load
 200                 lbl = "{} Task".format(command)
 201                 kwargs = dict(filename=filename, current_tab=False)
 202             self.recent_menu.add_command(
 203                 label="{} ({})".format(filename, lbl.title()),
 204                 command=lambda kw=kwargs, fn=load_func: fn(**kw))
 205         if removed_files:
 206             for recent_item in removed_files:
 207                 logger.debug("Removing from recent files: `%s`", recent_item[0])
 208                 recent_files.remove(recent_item)
 209             serializer.save(menu_file, recent_files)
 210         self.recent_menu.add_separator()
 211         self.recent_menu.add_command(
 212             label="Clear recent files",
 213             underline=0,
 214             command=lambda srl=serializer, mnu=menu_file: self.clear_recent_files(srl, mnu))
 215 
 216         logger.debug("Built Recent Files menu")

 262     def _build_branches_menu(self):
 263         """ Build branch selection menu.
 264 
 265         Queries git for available branches and builds a menu based on output.
 266 
 267         Returns
 268         -------
 269         bool
 270             ``True`` if menu was successfully built otherwise ``False``
 271         """
 272         stdout = self._get_branches()
 273         if stdout is None:
 274             return False
 275 
 276         branches = self._filter_branches(stdout)
 277         if not branches:
 278             return False
 279 
 280         for branch in branches:
 281             self._branches_menu.add_command(
 282                 label=branch,
 283                 command=lambda b=branch: self._switch_branch(b))
 284         return True

 286     @staticmethod
 287     def _get_branches():
 288         """ Get the available github branches
 289 
 290         Returns
 291         -------
 292         str
 293             The list of branches available. If no branches were found or there was an
 294             error then `None` is returned
 295         """
 296         gitcmd = "git branch -a"
 297         cmd = Popen(gitcmd, shell=True, stdout=PIPE, stderr=STDOUT, cwd=_WORKING_DIR)
 298         stdout, _ = cmd.communicate()
 299         retcode = cmd.poll()
 300         if retcode != 0:
 301             logger.debug("Unable to list git branches. return code: %s, message: %s",
 302                          retcode, stdout.decode().strip().replace("\n", " - "))
 303             return None
 304         return stdout.decode(locale.getpreferredencoding())

 306     @staticmethod
 307     def _filter_branches(stdout):
 308         """ Filter the branches, remove duplicates and the current branch and return a sorted
 309         list.
 310 
 311         Parameters
 312         ----------
 313         stdout: str
 314             The output from the git branch query converted to a string
 315 
 316         Returns
 317         -------
 318         list
 319             Unique list of available branches sorted in alphabetical order
 320         """
 321         current = None
 322         branches = set()
 323         for line in stdout.splitlines():
 324             branch = line[line.rfind("/") + 1:] if "/" in line else line.strip()
 325             if branch.startswith("*"):
 326                 branch = branch.replace("*", "").strip()
 327                 current = branch
 328                 continue
 329             branches.add(branch)
 330         logger.debug("Found branches: %s", branches)
 331         if current in branches:
 332             logger.debug("Removing current branch from output: %s", current)
 333             branches.remove(current)
 334 
 335         branches = sorted(list(branches), key=str.casefold)
 336         logger.debug("Final branches: %s", branches)
 337         return branches

 339     @staticmethod
 340     def _switch_branch(branch):
 341         """ Change the currently checked out branch, and return a notification.
 342 
 343         Parameters
 344         ----------
 345         str
 346             The branch to switch to
 347         """
 348         logger.info("Switching branch to '%s'...", branch)
 349         gitcmd = "git checkout {}".format(branch)
 350         cmd = Popen(gitcmd, shell=True, stdout=PIPE, stderr=STDOUT, cwd=_WORKING_DIR)
 351         stdout, _ = cmd.communicate()
 352         retcode = cmd.poll()
 353         if retcode != 0:
 354             logger.error("Unable to switch branch. return code: %s, message: %s",
 355                          retcode, stdout.decode().strip().replace("\n", " - "))
 356             return
 357         logger.info("Succesfully switched to '%s'. You may want to check for updates to make sure "
 358                     "that you have the latest code.", branch)
 359         logger.info("Please restart Faceswap to complete the switch.")

 407     def update(self):
 408         """ Check for updates and clone repository """
 409         logger.debug("Updating Faceswap...")
 410         self.root.config(cursor="watch")
 411         encoding = locale.getpreferredencoding()
 412         logger.debug("Encoding: %s", encoding)
 413         success = False
 414         if self.check_for_updates(encoding):
 415             success = self.do_update(encoding)
 416         update_deps.main(logger=logger)
 417         if success:
 418             logger.info("Please restart Faceswap to complete the update.")
 419         self.root.config(cursor="")

 421     @staticmethod
 422     def check_for_updates(encoding, check=False):
 423         """ Check whether an update is required """
 424         # Do the check
 425         logger.info("Checking for updates...")
 426         update = False
 427         msg = ""
 428         gitcmd = "git remote update && git status -uno"
 429         cmd = Popen(gitcmd, shell=True, stdout=PIPE, stderr=STDOUT, cwd=_WORKING_DIR)
 430         stdout, _ = cmd.communicate()
 431         retcode = cmd.poll()
 432         if retcode != 0:
 433             msg = ("Git is not installed or you are not running a cloned repo. "
 434                    "Unable to check for updates")
 435         else:
 436             chk = stdout.decode(encoding).splitlines()
 437             for line in chk:
 438                 if line.lower().startswith("your branch is ahead"):
 439                     msg = "Your branch is ahead of the remote repo. Not updating"
 440                     break
 441                 if line.lower().startswith("your branch is up to date"):
 442                     msg = "Faceswap is up to date."
 443                     break
 444                 if line.lower().startswith("your branch is behind"):
 445                     msg = "There are updates available"
 446                     update = True
 447                     break
 448                 if "have diverged" in line.lower():
 449                     msg = "Your branch has diverged from the remote repo. Not updating"
 450                     break
 451         if not update or check:
 452             logger.info(msg)
 453         logger.debug("Checked for update. Update required: %s", update)
 454         return update

 456     @staticmethod
 457     def do_update(encoding):
 458         """ Update Faceswap """
 459         logger.info("A new version is available. Updating...")
 460         gitcmd = "git pull"
 461         cmd = Popen(gitcmd, shell=True, stdout=PIPE, stderr=STDOUT, bufsize=1, cwd=_WORKING_DIR)
 462         while True:
 463             output = cmd.stdout.readline().decode(encoding)
 464             if output == "" and cmd.poll() is not None:
 465                 break
 466             if output:
 467                 logger.debug("'%s' output: '%s'", gitcmd, output.strip())
 468                 print(output.strip())
 469         retcode = cmd.poll()
 470         logger.debug("'%s' returncode: %s", gitcmd, retcode)
 471         if retcode != 0:
 472             logger.info("An error occurred during update. return code: %s", retcode)
 473             retval = False
 474         else:
 475             retval = True
 476         return retval
