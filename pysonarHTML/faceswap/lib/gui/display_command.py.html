  85     def display_item_process(self):
  86         """ Display the preview(s) resized as appropriate """
  87         logger.trace("Displaying preview")
  88         sortednames = sorted(list(get_images().previewtrain.keys()))
  89         existing = self.subnotebook_get_titles_ids()
  90         should_update = self.update_preview.get()
  91 
  92         for name in sortednames:
  93             if name not in existing.keys():
  94                 self.add_child(name)
  95             elif should_update:
  96                 tab_id = existing[name]
  97                 self.update_child(tab_id, name)
  98 
  99         if should_update:
 100             self.update_preview.set(False)

 127 class PreviewTrainCanvas(ttk.Frame):  # pylint: disable=too-many-ancestors
 128     """ Canvas to hold a training preview image """
 129     def __init__(self, parent, previewname):
 130         logger.debug("Initializing %s: (previewname: '%s')", self.__class__.__name__, previewname)
 131         ttk.Frame.__init__(self, parent)
 132 
 133         self.name = previewname
 134         get_images().resize_image(self.name, None)
 135         self.previewimage = get_images().previewtrain[self.name][1]
 136 
 137         self.canvas = tk.Canvas(self, bd=0, highlightthickness=0)
 138         self.canvas.pack(side=tk.TOP, fill=tk.BOTH, expand=True)
 139         self.imgcanvas = self.canvas.create_image(0,
 140                                                   0,
 141                                                   image=self.previewimage,
 142                                                   anchor=tk.NW)
 143         self.bind("<Configure>", self.resize)
 144         logger.debug("Initialized %s:", self.__class__.__name__)
 145 
 146     def resize(self, event):
 147         """  Resize the image to fit the frame, maintaining aspect ratio """
 148         logger.trace("Resizing preview image")
 149         framesize = (event.width, event.height)
 150         # Sometimes image is resized before frame is drawn
 151         framesize = None if framesize == (1, 1) else framesize
 152         get_images().resize_image(self.name, framesize)
 153         self.reload()
 154 
 155     def reload(self):
 156         """ Reload the preview image """
 157         logger.trace("Reloading preview image")
 158         self.previewimage = get_images().previewtrain[self.name][1]
 159         self.canvas.itemconfig(self.imgcanvas, image=self.previewimage)
 160 
 161     def save_preview(self, location):
 162         """ Save the figure to file """
 163         filename = self.name
 164         now = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
 165         filename = os.path.join(location,
 166                                 "{}_{}.{}".format(filename,
 167                                                   now,
 168                                                   "png"))
 169         get_images().previewtrain[self.name][0].save(filename)
 170         logger.debug("Saved preview to %s", filename)
 171         print("Saved preview to {}".format(filename))

 229     def display_item_set(self):
 230         """ Load the graph(s) if available """
 231         session = get_config().session
 232         smooth_amount_var = get_config().tk_vars["smoothgraph"]
 233         if session.initialized and session.logging_disabled:
 234             logger.trace("Logs disabled. Hiding graph")
 235             self.set_info("Graph is disabled as 'no-logs' or 'pingpong' has been selected")
 236             self.display_item = None
 237             if self.trace_var is not None:
 238                 smooth_amount_var.trace_vdelete("w", self.trace_var)
 239                 self.trace_var = None
 240         elif session.initialized:
 241             logger.trace("Loading graph")
 242             self.display_item = session
 243             if self.trace_var is None:
 244                 self.trace_var = smooth_amount_var.trace("w", self.smooth_amount_callback)
 245         else:
 246             self.display_item = None
 247             if self.trace_var is not None:
 248                 smooth_amount_var.trace_vdelete("w", self.trace_var)
 249                 self.trace_var = None
 250 
 251     def display_item_process(self):
 252         """ Add a single graph to the graph window """
 253         logger.trace("Adding graph")
 254         existing = list(self.subnotebook_get_titles_ids().keys())
 255         display_tabs = sorted(self.display_item.loss_keys)
 256         if any(key.startswith("total") for key in display_tabs):
 257             total_idx = [idx for idx, key in enumerate(display_tabs) if key.startswith("total")][0]
 258             display_tabs.insert(0, display_tabs.pop(total_idx))
 259         for loss_key in display_tabs:
 260             tabname = loss_key.replace("_", " ").title()
 261             if tabname in existing:
 262                 continue
 263 
 264             data = Calculations(session=get_config().session,
 265                                 display="loss",
 266                                 loss_keys=[loss_key],
 267                                 selections=["raw", "smoothed"],
 268                                 smooth_amount=get_config().tk_vars["smoothgraph"].get())
 269             self.add_child(tabname, data)
 270 
 271     def smooth_amount_callback(self, *args):
 272         """ Update each graph's smooth amount on variable change """
 273         smooth_amount = get_config().tk_vars["smoothgraph"].get()
 274         logger.debug("Updating graph smooth_amount: (new_value: %s, args: %s)",
 275                      smooth_amount, args)
 276         for graph in self.subnotebook.children.values():
 277             graph.calcs.args["smooth_amount"] = smooth_amount
 278

 295     def close(self):
 296         """ Clear the plots from RAM """
 297         if self.trace_var is not None:
 298             get_config().tk_vars["smoothgraph"].trace_vdelete("w", self.trace_var)
 299             self.trace_var = None
 300         if self.subnotebook is None:
 301             logger.debug("No graphs to clear. Returning")
 302             return
 303         for name, graph in self.subnotebook.children.items():
 304             logger.debug("Clearing: %s", name)
 305             graph.clear()
 306         super().close()
