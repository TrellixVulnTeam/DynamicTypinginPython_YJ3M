  13 def avg(arr):
  14     """ Return an average """
  15     return sum(arr) * 1.0 / len(arr)

  45     @staticmethod
  46     def load_images(reference_file_paths, nreference_file_paths):
  47         """ Load the images """
  48         retval = dict()
  49         for fpath in reference_file_paths:
  50             retval[fpath] = {"image": read_image(fpath, raise_error=True),
  51                              "type": "filter"}
  52         for fpath in nreference_file_paths:
  53             retval[fpath] = {"image": read_image(fpath, raise_error=True),
  54                              "type": "nfilter"}
  55         logger.debug("Loaded filter images: %s", {k: v["type"] for k, v in retval.items()})
  56         return retval

  58     # Extraction pipeline
  59     def align_faces(self, detector_name, aligner_name, masker_name, multiprocess):
  60         """ Use the requested detectors to retrieve landmarks for filter images """
  61         extractor = Extractor(detector_name,
  62                               aligner_name,
  63                               masker_name,
  64                               multiprocess=multiprocess)
  65         self.run_extractor(extractor)
  66         del extractor
  67         self.load_aligned_face()
  68 
  69     def run_extractor(self, extractor):
  70         """ Run extractor to get faces """
  71         for _ in range(extractor.passes):
  72             self.queue_images(extractor)
  73             extractor.launch()
  74             for faces in extractor.detected_faces():
  75                 filename = faces["filename"]
  76                 detected_faces = faces["detected_faces"]
  77                 if len(detected_faces) > 1:
  78                     logger.warning("Multiple faces found in %s file: '%s'. Using first detected "
  79                                    "face.", self.filters[filename]["type"], filename)
  80                 self.filters[filename]["detected_face"] = detected_faces[0]
  81 
  82     def queue_images(self, extractor):
  83         """ queue images for detection and alignment """
  84         in_queue = extractor.input_queue
  85         for fname, img in self.filters.items():
  86             logger.debug("Adding to filter queue: '%s' (%s)", fname, img["type"])
  87             feed_dict = dict(filename=fname, image=img["image"])
  88             if img.get("detected_faces", None):
  89                 feed_dict["detected_faces"] = img["detected_faces"]
  90             logger.debug("Queueing filename: '%s' items: %s",
  91                          fname, list(feed_dict.keys()))
  92             in_queue.put(feed_dict)
  93         logger.debug("Sending EOF to filter queue")
  94         in_queue.put("EOF")
  95 
  96     def load_aligned_face(self):
  97         """ Align the faces for vgg_face input """
  98         for filename, face in self.filters.items():
  99             logger.debug("Loading aligned face: '%s'", filename)
 100             image = face["image"]
 101             detected_face = face["detected_face"]
 102             detected_face.load_aligned(image, size=224)
 103             face["face"] = detected_face.aligned_face
 104             del face["image"]
 105             logger.debug("Loaded aligned face: ('%s', shape: %s)",
 106                          filename, face["face"].shape)
 107 
 108     def get_filter_encodings(self):
 109         """ Return filter face encodings from Keras VGG Face """
 110         for filename, face in self.filters.items():
 111             logger.debug("Getting encodings for: '%s'", filename)
 112             encodings = self.vgg_face.predict(face["face"])
 113             logger.debug("Filter Filename: %s, encoding shape: %s", filename, encodings.shape)
 114             face["encoding"] = encodings
 115             del face["face"]
 116 
 117     def check(self, detected_face):
 118         """ Check the extracted Face """
 119         logger.trace("Checking face with FaceFilter")
 120         distances = {"filter": list(), "nfilter": list()}
 121         encodings = self.vgg_face.predict(detected_face.aligned_face)
 122         for filt in self.filters.values():
 123             similarity = self.vgg_face.find_cosine_similiarity(filt["encoding"], encodings)
 124             distances[filt["type"]].append(similarity)
 125 
 126         avgs = {key: avg(val) if val else None for key, val in distances.items()}
 127         mins = {key: min(val) if val else None for key, val in distances.items()}
 128         # Filter
 129         if distances["filter"] and avgs["filter"] > self.threshold:
 130             msg = "Rejecting filter face: {} > {}".format(round(avgs["filter"], 2), self.threshold)
 131             retval = False
 132         # nFilter no Filter
 133         elif not distances["filter"] and avgs["nfilter"] < self.threshold:
 134             msg = "Rejecting nFilter face: {} < {}".format(round(avgs["nfilter"], 2),
 135                                                            self.threshold)
 136             retval = False
 137         # Filter with nFilter
 138         elif distances["filter"] and distances["nfilter"] and mins["filter"] > mins["nfilter"]:
 139             msg = ("Rejecting face as distance from nfilter sample is smaller: (filter: {}, "
 140                    "nfilter: {})".format(round(mins["filter"], 2), round(mins["nfilter"], 2)))
 141             retval = False
 142         elif distances["filter"] and distances["nfilter"] and avgs["filter"] > avgs["nfilter"]:
 143             msg = ("Rejecting face as average distance from nfilter sample is smaller: (filter: "
 144                    "{}, nfilter: {})".format(round(mins["filter"], 2), round(mins["nfilter"], 2)))
 145             retval = False
 146         elif distances["filter"] and distances["nfilter"]:
 147             # k-nn classifier
 148             var_k = min(5, min(len(distances["filter"]), len(distances["nfilter"])) + 1)
 149             var_n = sum(list(map(lambda x: x[0],
 150                                  list(sorted([(1, d) for d in distances["filter"]] +
 151                                              [(0, d) for d in distances["nfilter"]],
 152                                              key=lambda x: x[1]))[:var_k])))
 153             ratio = var_n/var_k
 154             if ratio < 0.5:
 155                 msg = ("Rejecting face as k-nearest neighbors classification is less than "
 156                        "0.5: {}".format(round(ratio, 2)))
 157                 retval = False
 158             else:
 159                 msg = None
 160                 retval = True
 161         else:
 162             msg = None
 163             retval = True
 164         if msg:
 165             logger.verbose(msg)
 166         else:
 167             logger.trace("Accepted face: (similarity: %s, threshold: %s)",
 168                          distances, self.threshold)
 169         return retval
