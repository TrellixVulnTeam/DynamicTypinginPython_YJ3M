  19 class FSThread(threading.Thread):
  20     """ Subclass of thread that passes errors back to parent """
  21     def __init__(self, group=None, target=None, name=None,  # pylint: disable=too-many-arguments
  22                  args=(), kwargs=None, *, daemon=None):
  23         super().__init__(group=group, target=target, name=name,
  24                          args=args, kwargs=kwargs, daemon=daemon)
  25         self.err = None
  26 
  27     def check_and_raise_error(self):
  28         """ Checks for errors in thread and raises them in caller """
  29         if not self.err:
  30             return
  31         logger.debug("Thread error caught: %s", self.err)
  32         raise self.err[1].with_traceback(self.err[2])
  33 
  34     def run(self):
  35         try:
  36             if self._target:
  37                 self._target(*self._args, **self._kwargs)
  38         except Exception as err:  # pylint: disable=broad-except
  39             self.err = sys.exc_info()
  40             logger.debug("Error in thread (%s): %s", self._name, str(err))
  41         finally:
  42             # Avoid a refcycle if the thread is running a function with
  43             # an argument that has a member that points to the thread.
  44             del self._target, self._args, self._kwargs

  47 class MultiThread():
  48     """ Threading for IO heavy ops
  49         Catches errors in thread and rethrows to parent """
  50     def __init__(self, target, *args, thread_count=1, name=None, **kwargs):
  51         self._name = name if name else target.__name__
  52         logger.debug("Initializing %s: (target: '%s', thread_count: %s)",
  53                      self.__class__.__name__, self._name, thread_count)
  54         logger.trace("args: %s, kwargs: %s", args, kwargs)
  55         self.daemon = True
  56         self._thread_count = thread_count
  57         self._threads = list()
  58         self._target = target
  59         self._args = args
  60         self._kwargs = kwargs
  61         logger.debug("Initialized %s: '%s'", self.__class__.__name__, self._name)
  62 
  63     @property
  64     def has_error(self):
  65         """ Return true if a thread has errored, otherwise false """
  66         return any(thread.err for thread in self._threads)
  67 
  68     @property
  69     def errors(self):
  70         """ Return a list of thread errors """
  71         return [thread.err for thread in self._threads if thread.err]
  72 
  73     @property
  74     def name(self):
  75         """ Return thread name """
  76         return self._name
  77 
  78     def check_and_raise_error(self):
  79         """ Checks for errors in thread and raises them in caller """
  80         if not self.has_error:
  81             return
  82         logger.debug("Thread error caught: %s", self.errors)
  83         error = self.errors[0]
  84         raise error[1].with_traceback(error[2])
  85 
  86     def is_alive(self):
  87         """ Return true if any thread is alive else false """
  88         return any(thread.is_alive() for thread in self._threads)
  89 
  90     def start(self):
  91         """ Start a thread with the given method and args """
  92         logger.debug("Starting thread(s): '%s'", self._name)
  93         for idx in range(self._thread_count):
  94             name = "{}_{}".format(self._name, idx)
  95             logger.debug("Starting thread %s of %s: '%s'",
  96                          idx + 1, self._thread_count, name)
  97             thread = FSThread(name=name,
  98                               target=self._target,
  99                               args=self._args,
 100                               kwargs=self._kwargs)
 101             thread.daemon = self.daemon
 102             thread.start()
 103             self._threads.append(thread)
 104         logger.debug("Started all threads '%s': %s", self._name, len(self._threads))
 105 
 106     def completed(self):
 107         """ Return False if there are any alive threads else True """
 108         retval = all(not thread.is_alive() for thread in self._threads)
 109         logger.debug(retval)
 110         return retval
 111 
 112     def join(self):
 113         """ Join the running threads, catching and re-raising any errors """
 114         logger.debug("Joining Threads: '%s'", self._name)
 115         for thread in self._threads:
 116             logger.debug("Joining Thread: '%s'", thread._name)  # pylint: disable=protected-access
 117             thread.join()
 118             if thread.err:
 119                 logger.error("Caught exception in thread: '%s'",
 120                              thread._name)  # pylint: disable=protected-access
 121                 raise thread.err[1].with_traceback(thread.err[2])
 122         logger.debug("Joined all Threads: '%s'", self._name)

 125 class BackgroundGenerator(MultiThread):
 126     """ Run a queue in the background. From:
 127         https://stackoverflow.com/questions/7323664/ """
 128     # See below why prefetch count is flawed
 129     def __init__(self, generator, prefetch=1, thread_count=2,
 130                  queue=None, args=None, kwargs=None):
 131         # pylint:disable=too-many-arguments
 132         super().__init__(target=self._run, thread_count=thread_count)
 133         self.queue = queue or Queue.Queue(prefetch)
 134         self.generator = generator
 135         self._gen_args = args or tuple()
 136         self._gen_kwargs = kwargs or dict()
 137         self.start()
 138 
 139     def _run(self):
 140         """ Put until queue size is reached.
 141             Note: put blocks only if put is called while queue has already
 142             reached max size => this makes prefetch + thread_count prefetched items!
 143             N in the the queue, one waiting for insertion per thread! """
 144         try:
 145             for item in self.generator(*self._gen_args, **self._gen_kwargs):
 146                 self.queue.put(item)
 147             self.queue.put(None)
 148         except Exception:
 149             self.queue.put(None)
 150             raise
 151 
 152     def iterator(self):
 153         """ Iterate items out of the queue """
 154         while True:
 155             next_item = self.queue.get()
 156             self.check_and_raise_error()
 157             if next_item is None or next_item == "EOF":
 158                 logger.debug("Got EOF OR NONE in BackgroundGenerator")
 159                 break
 160             yield next_item
