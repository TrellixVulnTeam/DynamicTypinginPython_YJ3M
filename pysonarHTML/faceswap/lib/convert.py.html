  14 class Converter():
  15     """ The converter is responsible for swapping the original face(s) in a frame with the output
  16     of a trained Faceswap model.
  17 
  18     Parameters
  19     ----------
  20     output_size: int
  21         The size of the face, in pixels, that is output from the Faceswap model
  22     coverage_ratio: float
  23         The ratio of the training image that was used for training the Faceswap model
  24     draw_transparent: bool
  25         Whether the final output should be drawn onto a transparent layer rather than the original
  26         frame. Only available with certain writer plugins.
  27     pre_encode: python function
  28         Some writer plugins support the pre-encoding of images prior to saving out. As patching is
  29         done in multiple threads, but writing is done in a single thread, it can speed up the
  30         process to do any pre-encoding as part of the converter process.
  31     arguments: :class:`argparse.Namespace`
  32         The arguments that were passed to the convert process as generated from Faceswap's command
  33         line arguments
  34     configfile: str, optional
  35         Optional location of custom configuration ``ini`` file. If ``None`` then use the default
  36         config location. Default: ``None``
  37     """
  38     def __init__(self, output_size, coverage_ratio, draw_transparent, pre_encode,
  39                  arguments, configfile=None):
  40         logger.debug("Initializing %s: (output_size: %s,  coverage_ratio: %s, draw_transparent: "
  41                      "%s, pre_encode: %s, arguments: %s, configfile: %s)", self.__class__.__name__,
  42                      output_size, coverage_ratio, draw_transparent, pre_encode, arguments,
  43                      configfile)
  44         self._output_size = output_size
  45         self._coverage_ratio = coverage_ratio
  46         self._draw_transparent = draw_transparent
  47         self._writer_pre_encode = pre_encode
  48         self._args = arguments
  49         self._configfile = configfile
  50 
  51         self._scale = arguments.output_scale / 100
  52         self._adjustments = dict(box=None, mask=None, color=None, seamless=None, scaling=None)
  53 
  54         self._load_plugins()
  55         logger.debug("Initialized %s", self.__class__.__name__)
  56 
  57     @property
  58     def cli_arguments(self):
  59         """:class:`argparse.Namespace`: The command line arguments passed to the convert
  60         process """
  61         return self._args

  63     def reinitialize(self, config):
  64         """ Reinitialize this :class:`Converter`.
  65 
  66         Called as part of the :mod:`~tools.preview` tool. Resets all adjustments then loads the
  67         plugins as specified in the given config.
  68 
  69         Parameters
  70         ----------
  71         config: :class:`lib.config.FaceswapConfig`
  72             Pre-loaded :class:`lib.config.FaceswapConfig`. used over any configuration on disk.
  73         """
  74         logger.debug("Reinitializing converter")
  75         self._adjustments = dict(box=None, mask=None, color=None, seamless=None, scaling=None)
  76         self._load_plugins(config=config, disable_logging=True)
  77         logger.debug("Reinitialized converter")

  79     def _load_plugins(self, config=None, disable_logging=False):
  80         """ Load the requested adjustment plugins.
  81 
  82         Loads the :mod:`plugins.converter` plugins that have been requested for this conversion
  83         session.
  84 
  85         Parameters
  86         ----------
  87         config: :class:`lib.config.FaceswapConfig`, optional
  88             Optional pre-loaded :class:`lib.config.FaceswapConfig`. If passed, then this will be
  89             used over any configuration on disk. If ``None`` then it is ignored. Default: ``None``
  90         disable_logging: bool, optional
  91             Plugin loader outputs logging info every time a plugin is loaded. Set to ``True`` to
  92             suppress these messages otherwise ``False``. Default: ``False``
  93         """
  94         logger.debug("Loading plugins. config: %s", config)
  95         self._adjustments["box"] = PluginLoader.get_converter(
  96             "mask",
  97             "box_blend",
  98             disable_logging=disable_logging)(self._output_size,
  99                                              configfile=self._configfile,
 100                                              config=config)
 101 
 102         self._adjustments["mask"] = PluginLoader.get_converter(
 103             "mask",
 104             "mask_blend",
 105             disable_logging=disable_logging)(self._args.mask_type,
 106                                              self._output_size,
 107                                              self._coverage_ratio,
 108                                              configfile=self._configfile,
 109                                              config=config)
 110 
 111         if self._args.color_adjustment != "none" and self._args.color_adjustment is not None:
 112             self._adjustments["color"] = PluginLoader.get_converter(
 113                 "color",
 114                 self._args.color_adjustment,
 115                 disable_logging=disable_logging)(configfile=self._configfile, config=config)
 116 
 117         if self._args.scaling != "none" and self._args.scaling is not None:
 118             self._adjustments["scaling"] = PluginLoader.get_converter(
 119                 "scaling",
 120                 self._args.scaling,
 121                 disable_logging=disable_logging)(configfile=self._configfile, config=config)
 122         logger.debug("Loaded plugins: %s", self._adjustments)

 124     def process(self, in_queue, out_queue):
 125         """ Main convert process.
 126 
 127         Takes items from the in queue, runs the relevant adjustments, patches faces to final frame
 128         and outputs patched frame to the out queue.
 129 
 130         Parameters
 131         ----------
 132         in_queue: :class:`queue.Queue`
 133             The output from :class:`scripts.convert.Predictor`. Contains detected faces from the
 134             Faceswap model as well as the frame to be patched.
 135         out_queue: :class:`queue.Queue`
 136             The queue to place patched frames into for writing by one of Faceswap's
 137             :mod:`plugins.convert.writer` plugins.
 138         """
 139         logger.debug("Starting convert process. (in_queue: %s, out_queue: %s)",
 140                      in_queue, out_queue)
 141         while True:
 142             items = in_queue.get()
 143             if items == "EOF":
 144                 logger.debug("EOF Received")
 145                 logger.debug("Patch queue finished")
 146                 # Signal EOF to other processes in pool
 147                 logger.debug("Putting EOF back to in_queue")
 148                 in_queue.put(items)
 149                 break
 150 
 151             if isinstance(items, dict):
 152                 items = [items]
 153             for item in items:
 154                 logger.trace("Patch queue got: '%s'", item["filename"])
 155                 try:
 156                     image = self._patch_image(item)
 157                 except Exception as err:  # pylint: disable=broad-except
 158                     # Log error and output original frame
 159                     logger.error("Failed to convert image: '%s'. Reason: %s",
 160                                  item["filename"], str(err))
 161                     image = item["image"]
 162                     # UNCOMMENT THIS CODE BLOCK TO PRINT TRACEBACK ERRORS
 163                     # import sys ; import traceback
 164                     # exc_info = sys.exc_info() ; traceback.print_exception(*exc_info)
 165                 logger.trace("Out queue put: %s", item["filename"])
 166                 out_queue.put((item["filename"], image))
 167         logger.debug("Completed convert process")

 169     def _patch_image(self, predicted):
 170         """ Patch a swapped face onto a frame.
 171 
 172         Run selected adjustments and swap the faces in a frame.
 173 
 174         Parameters
 175         ----------
 176         predicted: dict
 177             The output from :class:`scripts.convert.Predictor`.
 178 
 179         Returns
 180         -------
 181         :class: `numpy.ndarray` or pre-encoded image output
 182             The final frame ready for writing by a :mod:`plugins.convert.writer` plugin.
 183             Frame is either an array, or the pre-encoded output from the writer's pre-encode
 184             function (if it has one)
 185 
 186         """
 187         logger.trace("Patching image: '%s'", predicted["filename"])
 188         frame_size = (predicted["image"].shape[1], predicted["image"].shape[0])
 189         new_image, background = self._get_new_image(predicted, frame_size)
 190         patched_face = self._post_warp_adjustments(background, new_image)
 191         patched_face = self._scale_image(patched_face)
 192         patched_face *= 255.0
 193         patched_face = np.rint(patched_face,
 194                                out=np.empty(patched_face.shape, dtype="uint8"),
 195                                casting='unsafe')
 196         if self._writer_pre_encode is not None:
 197             patched_face = self._writer_pre_encode(patched_face)
 198         logger.trace("Patched image: '%s'", predicted["filename"])
 199         return patched_face

 201     def _get_new_image(self, predicted, frame_size):
 202         """ Get the new face from the predictor and apply pre-warp manipulations.
 203 
 204         Applies any requested adjustments to the raw output of the Faceswap model
 205         before transforming the image into the target frame.
 206 
 207         Parameters
 208         ----------
 209         predicted: dict
 210             The output from :class:`scripts.convert.Predictor`.
 211         frame_size: tuple
 212             The (`width`, `height`) of the final frame in pixels
 213 
 214         Returns
 215         -------
 216         placeholder:  :class: `numpy.ndarray`
 217             The original frame with the swapped faces patched onto it
 218         background:  :class: `numpy.ndarray`
 219             The original frame
 220         """
 221         logger.trace("Getting: (filename: '%s', faces: %s)",
 222                      predicted["filename"], len(predicted["swapped_faces"]))
 223 
 224         placeholder = np.zeros((frame_size[1], frame_size[0], 4), dtype="float32")
 225         background = predicted["image"] / np.array(255.0, dtype="float32")
 226         placeholder[:, :, :3] = background
 227 
 228         for new_face, detected_face in zip(predicted["swapped_faces"],
 229                                            predicted["detected_faces"]):
 230             predicted_mask = new_face[:, :, -1] if new_face.shape[2] == 4 else None
 231             new_face = new_face[:, :, :3]
 232             interpolator = detected_face.reference_interpolators[1]
 233 
 234             new_face = self._pre_warp_adjustments(new_face, detected_face, predicted_mask)
 235 
 236             # Warp face with the mask
 237             cv2.warpAffine(new_face,
 238                            detected_face.reference_matrix,
 239                            frame_size,
 240                            placeholder,
 241                            flags=cv2.WARP_INVERSE_MAP | interpolator,
 242                            borderMode=cv2.BORDER_TRANSPARENT)
 243 
 244         np.clip(placeholder, 0.0, 1.0, out=placeholder)
 245         logger.trace("Got filename: '%s'. (placeholders: %s)",
 246                      predicted["filename"], placeholder.shape)
 247 
 248         return placeholder, background

 250     def _pre_warp_adjustments(self, new_face, detected_face, predicted_mask):
 251         """ Run any requested adjustments that can be performed on the raw output from the Faceswap
 252         model.
 253 
 254         Any adjustments that can be performed before warping the face into the final frame are
 255         performed here.
 256 
 257         Parameters
 258         ----------
 259         new_face: :class:`numpy.ndarray`
 260             The swapped face received from the faceswap model.
 261         detected_face: :class:`~lib.faces_detect.DetectedFace`
 262             The detected_face object as defined in :class:`scripts.convert.Predictor`
 263         predicted_mask: :class:`numpy.ndarray` or ``None``
 264             The predicted mask output from the Faceswap model. ``None`` if the model
 265             did not learn a mask
 266 
 267         Returns
 268         -------
 269         :class:`numpy.ndarray`
 270             The face output from the Faceswap Model with any requested pre-warp adjustments
 271             performed.
 272         """
 273         logger.trace("new_face shape: %s, predicted_mask shape: %s", new_face.shape,
 274                      predicted_mask.shape if predicted_mask is not None else None)
 275         old_face = detected_face.reference_face[..., :3] / 255.0
 276         new_face = self._adjustments["box"].run(new_face)
 277         new_face, raw_mask = self._get_image_mask(new_face, detected_face, predicted_mask)
 278         if self._adjustments["color"] is not None:
 279             new_face = self._adjustments["color"].run(old_face, new_face, raw_mask)
 280         if self._adjustments["seamless"] is not None:
 281             new_face = self._adjustments["seamless"].run(old_face, new_face, raw_mask)
 282         logger.trace("returning: new_face shape %s", new_face.shape)
 283         return new_face

 285     def _get_image_mask(self, new_face, detected_face, predicted_mask):
 286         """ Return any selected image mask and intersect with any box mask.
 287 
 288         Places the requested mask into the new face's Alpha channel, intersecting with any box
 289         mask that has already been applied.
 290 
 291         Parameters
 292         ----------
 293         new_face: :class:`numpy.ndarray`
 294             The swapped face received from the faceswap model, with any box mask applied
 295         detected_face: :class:`~lib.faces_detect.DetectedFace`
 296             The detected_face object as defined in :class:`scripts.convert.Predictor`
 297         predicted_mask: :class:`numpy.ndarray` or ``None``
 298             The predicted mask output from the Faceswap model. ``None`` if the model
 299             did not learn a mask
 300 
 301         Returns
 302         :class:`numpy.ndarray`
 303             The swapped face with the requested mask added to the Alpha channel
 304         """
 305         logger.trace("Getting mask. Image shape: %s", new_face.shape)
 306         mask, raw_mask = self._adjustments["mask"].run(detected_face, predicted_mask)
 307         if new_face.shape[2] == 4:
 308             logger.trace("Combining mask with alpha channel box mask")
 309             new_face[:, :, -1] = np.minimum(new_face[:, :, -1], mask.squeeze())
 310         else:
 311             logger.trace("Adding mask to alpha channel")
 312             new_face = np.concatenate((new_face, mask), -1)
 313         logger.trace("Got mask. Image shape: %s", new_face.shape)
 314         return new_face, raw_mask

 316     def _post_warp_adjustments(self, background, new_image):
 317         """ Perform any requested adjustments to the swapped faces after they have been transformed
 318         into the final frame.
 319 
 320         Parameters
 321         ----------
 322         background: :class:`numpy.ndarray`
 323             The original frame
 324         new_image: :class:`numpy.ndarray`
 325             A blank frame of original frame size with the faces warped onto it
 326 
 327         Returns
 328         -------
 329         :class:`numpy.ndarray`
 330             The final merged and swapped frame with any requested post-warp adjustments applied
 331         """
 332         if self._adjustments["scaling"] is not None:
 333             new_image = self._adjustments["scaling"].run(new_image)
 334 
 335         if self._draw_transparent:
 336             frame = new_image
 337         else:
 338             foreground, mask = np.split(new_image,  # pylint:disable=unbalanced-tuple-unpacking
 339                                         (3, ),
 340                                         axis=-1)
 341             foreground *= mask
 342             background *= (1.0 - mask)
 343             background += foreground
 344             frame = background
 345         np.clip(frame, 0.0, 1.0, out=frame)
 346         return frame

 348     def _scale_image(self, frame):
 349         """ Scale the final image if requested.
 350 
 351         If output scale has been requested in command line arguments, scale the output
 352         otherwise return the final frame.
 353 
 354         Parameters
 355         ----------
 356         frame: :class:`numpy.ndarray`
 357             The final frame with faces swapped
 358 
 359         Returns
 360         -------
 361         :class:`numpy.ndarray`
 362             The final frame scaled by the requested scaling factor
 363         """
 364         if self._scale == 1:
 365             return frame
 366         logger.trace("source frame: %s", frame.shape)
 367         interp = cv2.INTER_CUBIC if self._scale > 1 else cv2.INTER_AREA
 368         dims = (round((frame.shape[1] / 2 * self._scale) * 2),
 369                 round((frame.shape[0] / 2 * self._scale) * 2))
 370         frame = cv2.resize(frame, dims, interpolation=interp)
 371         logger.trace("resized frame: %s", frame.shape)
 372         np.clip(frame, 0.0, 1.0, out=frame)
 373         return frame
