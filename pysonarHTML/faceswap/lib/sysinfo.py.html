 153     @property
 154     def _cuda_version(self):
 155         """ str: The installed CUDA version. """
 156         # TODO Handle multiple CUDA installs
 157         chk = Popen("nvcc -V", shell=True, stdout=PIPE, stderr=PIPE)
 158         stdout, stderr = chk.communicate()
 159         if not stderr:
 160             version = re.search(r".*release (?P<cuda>\d+\.\d+)", stdout.decode(self._encoding))
 161             version = version.groupdict().get("cuda", None)
 162             if version:
 163                 return version
 164         # Failed to load nvcc
 165         if self._is_linux:
 166             version = self._cuda_version_linux()
 167         elif self._is_windows:
 168             version = self._cuda_version_windows()
 169         else:
 170             version = "Unsupported OS"
 171             if self._is_conda:
 172                 version += ". Check Conda packages for Conda Cuda"
 173         return version

 175     @property
 176     def _cudnn_version(self):
 177         """ str: The installed cuDNN version. """
 178         if self._is_linux:
 179             cudnn_checkfiles = self._cudnn_checkfiles_linux()
 180         elif self._is_windows:
 181             cudnn_checkfiles = self._cudnn_checkfiles_windows()
 182         else:
 183             retval = "Unsupported OS"
 184             if self._is_conda:
 185                 retval += ". Check Conda packages for Conda cuDNN"
 186             return retval
 187 
 188         cudnn_checkfile = None
 189         for checkfile in cudnn_checkfiles:
 190             if os.path.isfile(checkfile):
 191                 cudnn_checkfile = checkfile
 192                 break
 193 
 194         if not cudnn_checkfile:
 195             retval = "No global version found"
 196             if self._is_conda:
 197                 retval += ". Check Conda packages for Conda cuDNN"
 198             return retval
 199 
 200         found = 0
 201         with open(cudnn_checkfile, "r") as ofile:
 202             for line in ofile:
 203                 if line.lower().startswith("#define cudnn_major"):
 204                     major = line[line.rfind(" ") + 1:].strip()
 205                     found += 1
 206                 elif line.lower().startswith("#define cudnn_minor"):
 207                     minor = line[line.rfind(" ") + 1:].strip()
 208                     found += 1
 209                 elif line.lower().startswith("#define cudnn_patchlevel"):
 210                     patchlevel = line[line.rfind(" ") + 1:].strip()
 211                     found += 1
 212                 if found == 3:
 213                     break
 214         if found != 3:
 215             retval = "No global version found"
 216             if self._is_conda:
 217                 retval += ". Check Conda packages for Conda cuDNN"
 218             return retval
 219         return "{}.{}.{}".format(major, minor, patchlevel)

 221     @staticmethod
 222     def _cudnn_checkfiles_linux():
 223         """ Obtain the location of the files to check for cuDNN location in Linux.
 224 
 225         Returns
 226         str:
 227             The location of the header files for cuDNN
 228         """
 229         chk = os.popen("ldconfig -p | grep -P \"libcudnn.so.\\d+\" | head -n 1").read()
 230         if "libcudnn.so." not in chk:
 231             return list()
 232         chk = chk.strip().replace("libcudnn.so.", "")
 233         cudnn_vers = chk[0]
 234         cudnn_path = chk[chk.find("=>") + 3:chk.find("libcudnn") - 1]
 235         cudnn_path = cudnn_path.replace("lib", "include")
 236         cudnn_checkfiles = [os.path.join(cudnn_path, "cudnn_v{}.h".format(cudnn_vers)),
 237                             os.path.join(cudnn_path, "cudnn.h")]
 238         return cudnn_checkfiles

 272     @staticmethod
 273     def _cuda_path_linux():
 274         """ Obtain the path to Cuda install location on Linux.
 275 
 276         Returns
 277         -------
 278         str
 279             The path to the install location of Cuda on a Linux system
 280         """
 281         ld_library_path = os.environ.get("LD_LIBRARY_PATH", None)
 282         chk = os.popen("ldconfig -p | grep -P \"libcudart.so.\\d+.\\d+\" | head -n 1").read()
 283         if ld_library_path and not chk:
 284             paths = ld_library_path.split(":")
 285             for path in paths:
 286                 chk = os.popen("ls {} | grep -P -o \"libcudart.so.\\d+.\\d+\" | "
 287                                "head -n 1".format(path)).read()
 288                 if chk:
 289                     break
 290         if not chk:
 291             return None
 292         return chk[chk.find("=>") + 3:chk.find("targets") - 1]

 346     def full_info(self):
 347         """ Obtain extensive system information stats, formatted into a human readable format.
 348 
 349         Returns
 350         -------
 351         str
 352             The system information for the currently running system, formatted for output to
 353             console or a log file.
 354         """
 355         retval = "\n============ System Information ============\n"
 356         sys_info = {"os_platform": self._system["platform"],
 357                     "os_machine": self._system["machine"],
 358                     "os_release": self._system["release"],
 359                     "py_conda_version": self._conda_version,
 360                     "py_implementation": self._python["implementation"],
 361                     "py_version": self._python["version"],
 362                     "py_command": self._fs_command,
 363                     "py_virtual_env": self._is_virtual_env,
 364                     "sys_cores": self._system["cpu_count"],
 365                     "sys_processor": self._system["processor"],
 366                     "sys_ram": self._format_ram(),
 367                     "encoding": self._encoding,
 368                     "git_branch": self._git_branch,
 369                     "git_commits": self._git_commits,
 370                     "gpu_cuda": self._cuda_version,
 371                     "gpu_cudnn": self._cudnn_version,
 372                     "gpu_driver": self._gpu["driver"],
 373                     "gpu_devices": ", ".join(["GPU_{}: {}".format(idx, device)
 374                                               for idx, device in enumerate(self._gpu["devices"])]),
 375                     "gpu_vram": ", ".join(["GPU_{}: {}MB".format(idx, int(vram))
 376                                            for idx, vram in enumerate(self._gpu["vram"])]),
 377                     "gpu_devices_active": ", ".join(["GPU_{}".format(idx)
 378                                                      for idx in self._gpu["devices_active"]])}
 379         for key in sorted(sys_info.keys()):
 380             retval += ("{0: <20} {1}\n".format(key + ":", sys_info[key]))
 381         retval += "\n=============== Pip Packages ===============\n"
 382         retval += self._installed_pip
 383         if self._is_conda:
 384             retval += "\n\n============== Conda Packages ==============\n"
 385             retval += self._installed_conda
 386         retval += self._state_file
 387         retval += "\n\n================= Configs =================="
 388         retval += self._configs
 389         return retval

 391     def _format_ram(self):
 392         """ Format the RAM stats into Megabytes to make it more readable.
 393 
 394         Returns
 395         -------
 396         str
 397             The total, available, used and free RAM displayed in Megabytes
 398         """
 399         retval = list()
 400         for name in ("total", "available", "used", "free"):
 401             value = getattr(self, "_ram_{}".format(name))
 402             value = int(value / (1024 * 1024))
 403             retval.append("{}: {}MB".format(name.capitalize(), value))
 404         return ", ".join(retval)

 425 class _Configs():
 426     """ Parses the config files in /faceswap/config and outputs the information stored within them
 427     in a human readable format. """
 428 
 429     def __init__(self):
 430         self.config_dir = os.path.join(os.path.abspath(os.path.dirname(sys.argv[0])), "config")
 431         self.configs = self._get_configs()
 432 
 433     def _get_configs(self):
 434         """ Obtain the formatted configurations from the config folder.
 435 
 436         Returns
 437         -------
 438         str
 439             The current configuration in the config files formatted in a human readable format
 440         """
 441         config_files = [os.path.join(self.config_dir, cfile)
 442                         for cfile in os.listdir(self.config_dir)
 443                         if os.path.basename(cfile) == ".faceswap"
 444                         or os.path.splitext(cfile)[1] == ".ini"]
 445         return self._parse_configs(config_files)
 446 
 447     def _parse_configs(self, config_files):
 448         """ Parse the given list of config files into a human readable format.
 449 
 450         Parameters
 451         ----------
 452         config_files: list
 453             A list of paths to the faceswap config files
 454 
 455         Returns
 456         -------
 457         str
 458             The current configuration in the config files formatted in a human readable format
 459         """
 460         formatted = ""
 461         for cfile in config_files:
 462             fname = os.path.basename(cfile)
 463             ext = os.path.splitext(cfile)[1]
 464             formatted += "\n--------- {} ---------\n".format(fname)
 465             if ext == ".ini":
 466                 formatted += self._parse_ini(cfile)
 467             elif fname == ".faceswap":
 468                 formatted += self._parse_json(cfile)
 469         return formatted
 470 
 471     def _parse_ini(self, config_file):
 472         """ Parse an ``.ini`` formatted config file into a human readable format.
 473 
 474         Parameters
 475         ----------
 476         config_file: str
 477             The path to the config.ini file
 478 
 479         Returns
 480         -------
 481         str
 482             The current configuration in the config file formatted in a human readable format
 483         """
 484         formatted = ""
 485         with open(config_file, "r") as cfile:
 486             for line in cfile.readlines():
 487                 line = line.strip()
 488                 if line.startswith("#") or not line:
 489                     continue
 490                 item = line.split("=")
 491                 if len(item) == 1:
 492                     formatted += "\n{}\n".format(item[0].strip())
 493                 else:
 494                     formatted += self._format_text(item[0], item[1])
 495         return formatted
 496 
 497     def _parse_json(self, config_file):
 498         """ Parse an ``.json`` formatted config file into a python dictionary.
 499 
 500         Parameters
 501         ----------
 502         config_file: str
 503             The path to the config.json file
 504 
 505         Returns
 506         -------
 507         dict
 508             The current configuration in the config file formatted as a python dictionary
 509         """
 510         formatted = ""
 511         with open(config_file, "r") as cfile:
 512             conf_dict = json.load(cfile)
 513             for key in sorted(conf_dict.keys()):
 514                 formatted += self._format_text(key, conf_dict[key])
 515         return formatted
 516 
 517     @staticmethod
 518     def _format_text(key, value):
 519         """Format a key value pair into a consistently spaced string output for display.
 520 
 521         Parameters
 522         ----------
 523         key: str
 524             The label for this display item
 525         value: str
 526             The value for this display item
 527 
 528         Returns
 529         -------
 530         str
 531             The formatted key value pair for display
 532         """
 533         return "{0: <25} {1}\n".format(key.strip() + ":", value.strip())

 536 class _State():
 537     """ Parses the state file in the current model directory, if the model is training, and
 538     formats the content into a human readable format. """
 539     def __init__(self):
 540         self._model_dir = self._get_arg("-m", "--model-dir")
 541         self._trainer = self._get_arg("-t", "--trainer")
 542         self.state_file = self._get_state_file()
 543 
 544     @property
 545     def _is_training(self):
 546         """ bool: ``True`` if this function has been called during a training session
 547         otherwise ``False``. """
 548         return len(sys.argv) > 1 and sys.argv[1].lower() == "train"
 549 
 550     @staticmethod
 551     def _get_arg(*args):
 552         """ Obtain the value for a given command line option from sys.argv.
 553 
 554         Returns
 555         -------
 556         str or ``None``
 557             The value of the given command line option, if it exists, otherwise ``None``
 558         """
 559         cmd = sys.argv
 560         for opt in args:
 561             if opt in cmd:
 562                 return cmd[cmd.index(opt) + 1]
 563         return None
 564 
 565     def _get_state_file(self):
 566         """ Parses the model's state file and compiles the contents into a human readable string.
 567 
 568         Returns
 569         -------
 570         str
 571             The state file formatted into a human readable format
 572         """
 573         if not self._is_training or self._model_dir is None or self._trainer is None:
 574             return ""
 575         fname = os.path.join(self._model_dir, "{}_state.json".format(self._trainer))
 576         if not os.path.isfile(fname):
 577             return ""
 578 
 579         retval = "\n\n=============== State File =================\n"
 580         with open(fname, "r") as sfile:
 581             retval += sfile.read()
 582         return retval
