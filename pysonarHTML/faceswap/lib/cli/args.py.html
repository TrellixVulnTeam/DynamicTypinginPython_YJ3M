  26 class SmartFormatter(argparse.HelpFormatter):
  27     """ Extends the class :class:`argparse.HelpFormatter` to allow custom formatting in help text.
  28 
  29     Adapted from: https://stackoverflow.com/questions/3853722
  30 
  31     Notes
  32     -----
  33     Prefix help text with "R|" to override default formatting and use explicitly defined formatting
  34     within the help text.
  35     Prefixing a new line within the help text with "L|" will turn that line into a list item in
  36     both the cli help text and the GUI.
  37     """
  38     def __init__(self, prog, indent_increment=2, max_help_position=24, width=None):
  39         super().__init__(prog, indent_increment, max_help_position, width)
  40         self._whitespace_matcher_limited = re.compile(r'[ \r\f\v]+', re.ASCII)
  41 
  42     def _split_lines(self, text, width):
  43         """ Split the given text by the given display width.
  44 
  45         If the text is not prefixed with "R|" then the standard
  46         :func:`argparse.HelpFormatter._split_lines` function is used, otherwise raw
  47         formatting is processed,
  48 
  49         Parameters
  50         ----------
  51         text: str
  52             The help text that is to be formatted for display
  53         width: int
  54             The display width, in characters, for the help text
  55         """
  56         if text.startswith("R|"):
  57             text = self._whitespace_matcher_limited.sub(' ', text).strip()[2:]
  58             output = list()
  59             for txt in text.splitlines():
  60                 indent = ""
  61                 if txt.startswith("L|"):
  62                     indent = "    "
  63                     txt = "  - {}".format(txt[2:])
  64                 output.extend(textwrap.wrap(txt, width, subsequent_indent=indent))
  65             return output
  66         return argparse.HelpFormatter._split_lines(self, text, width)

  69 class FaceSwapArgs():
  70     """ Faceswap argument parser functions that are universal to all commands.
  71 
  72     This is the parent class to all subsequent argparsers which holds global arguments that pertain
  73     to all commands.
  74 
  75     Process the incoming command line arguments, validates then launches the relevant faceswap
  76     script with the given arguments.
  77 
  78     Parameters
  79     ----------
  80     subparser: :class:`argparse._SubParsersAction`
  81         The subparser for the given command
  82     command: str
  83         The faceswap command that is to be executed
  84     description: str, optional
  85         The description for the given command. Default: "default"
  86     """
  87     def __init__(self, subparser, command, description="default"):
  88         self.global_arguments = self._get_global_arguments()
  89         self.info = self.get_info()
  90         self.argument_list = self.get_argument_list()
  91         self.optional_arguments = self.get_optional_arguments()
  92         self._process_suppressions()
  93         if not subparser:
  94             return
  95         self.parser = self._create_parser(subparser, command, description)
  96         self._add_arguments()
  97         script = ScriptExecutor(command)
  98         self.parser.set_defaults(func=script.execute_script)
  99

 199     @staticmethod
 200     def _create_parser(subparser, command, description):
 201         """ Create the parser for the selected command.
 202 
 203         Parameters
 204         ----------
 205         command: str
 206             The faceswap command that is to be executed
 207         description: str
 208             The description for the given command
 209 
 210         Returns
 211         -------
 212         :class:`~lib.cli.args.FullHelpArgumentParser`
 213             The parser for the given command
 214         """
 215         parser = subparser.add_parser(command,
 216                                       help=description,
 217                                       description=description,
 218                                       epilog="Questions and feedback: https://faceswap.dev/forum",
 219                                       formatter_class=SmartFormatter)
 220         return parser
 221 
 222     def _add_arguments(self):
 223         """ Parse the list of dictionaries containing the command line arguments and convert to
 224         argparse parser arguments. """
 225         options = self.global_arguments + self.argument_list + self.optional_arguments
 226         for option in options:
 227             args = option["opts"]
 228             kwargs = {key: option[key] for key in option.keys() if key not in ("opts", "group")}
 229             self.parser.add_argument(*args, **kwargs)
 230 
 231     def _process_suppressions(self):
 232         """ Certain options are only available for certain backends.
 233 
 234         Suppresses command line options that are not available for the running backend.
 235         """
 236         fs_backend = get_backend()
 237         for opt_list in [self.global_arguments, self.argument_list, self.optional_arguments]:
 238             for opts in opt_list:
 239                 if opts.get("backend", None) is None:
 240                     continue
 241                 opt_backend = opts.pop("backend")
 242                 if isinstance(opt_backend, (list, tuple)):
 243                     opt_backend = [backend.lower() for backend in opt_backend]
 244                 else:
 245                     opt_backend = [opt_backend.lower()]
 246                 if fs_backend not in opt_backend:
 247                     opts["help"] = argparse.SUPPRESS
