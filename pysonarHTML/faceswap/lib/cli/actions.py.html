  14 class _FullPaths(argparse.Action):  # pylint: disable=too-few-public-methods
  15     """ Parent class for various file type and file path handling classes.
  16 
  17     Expands out given paths to their full absolute paths. This class should not be
  18     called directly. It is the base class for the various different file handling
  19     methods.
  20     """
  21     def __call__(self, parser, namespace, values, option_string=None):
  22         if isinstance(values, (list, tuple)):
  23             vals = [os.path.abspath(os.path.expanduser(val)) for val in values]
  24         else:
  25             vals = os.path.abspath(os.path.expanduser(values))
  26         setattr(namespace, self.dest, vals)

  48 class FileFullPaths(_FullPaths):
  49     """ Adds support for a File browser to select a single file in the GUI.
  50 
  51     This extends the standard :class:`argparse.Action` and adds an additional parameter
  52     :attr:`filetypes`, indicating to the GUI that it should pop a file browser for opening a file
  53     and limit the results to the file types listed. As well as the standard parameters, the
  54     following parameter is required:
  55 
  56     Parameters
  57     ----------
  58     filetypes: str
  59         The accepted file types for this option. This is the key for the GUIs lookup table which
  60         can be found in :class:`lib.gui.utils.FileHandler`
  61 
  62     Example
  63     -------
  64     >>> argument_list = []
  65     >>> argument_list.append(dict(
  66     >>>        opts=("-f", "--video_location"),
  67     >>>        action=FileFullPaths,
  68     >>>        filetypes="video))"
  69     """
  70     # pylint: disable=too-few-public-methods
  71     def __init__(self, *args, filetypes=None, **kwargs):
  72         super().__init__(*args, **kwargs)
  73         self.filetypes = filetypes
  74 
  75     def _get_kwargs(self):
  76         names = ["option_strings",
  77                  "dest",
  78                  "nargs",
  79                  "const",
  80                  "default",
  81                  "type",
  82                  "choices",
  83                  "help",
  84                  "metavar",
  85                  "filetypes"]
  86         return [(name, getattr(self, name)) for name in names]

  89 class FilesFullPaths(FileFullPaths):  # pylint: disable=too-few-public-methods
  90     """ Adds support for a File browser to select multiple files in the GUI.
  91 
  92     This extends the standard :class:`argparse.Action` and adds an additional parameter
  93     :attr:`filetypes`, indicating to the GUI that it should pop a file browser, and limit
  94     the results to the file types listed. Multiple files can be selected for opening, so the
  95     :attr:`nargs` parameter must be set. As well as the standard parameters, the following
  96     parameter is required:
  97 
  98     Parameters
  99     ----------
 100     filetypes: str
 101         The accepted file types for this option. This is the key for the GUIs lookup table which
 102         can be found in :class:`lib.gui.utils.FileHandler`
 103 
 104     Example
 105     -------
 106     >>> argument_list = []
 107     >>> argument_list.append(dict(
 108     >>>        opts=("-f", "--images"),
 109     >>>        action=FilesFullPaths,
 110     >>>        filetypes="image",
 111     >>>        nargs="+"))
 112     """
 113     def __init__(self, *args, filetypes=None, **kwargs):
 114         if kwargs.get("nargs", None) is None:
 115             opt = kwargs["option_strings"]
 116             raise ValueError("nargs must be provided for FilesFullPaths: {}".format(opt))
 117         super().__init__(*args, **kwargs)

 176 class ContextFullPaths(FileFullPaths):
 177     """ Adds support for context sensitive browser dialog opening in the GUI.
 178 
 179     For some tasks, the type of action (file load, folder open, file save etc.) can vary
 180     depending on the task to be performed (a good example of this is the effmpeg tool).
 181     Using this action indicates to the GUI that the type of dialog to be launched can change
 182     depending on another option. As well as the standard parameters, the below parameters are
 183     required. NB: :attr:`nargs` are explicitly disallowed.
 184 
 185     Parameters
 186     ----------
 187     filetypes: str
 188         The accepted file types for this option. This is the key for the GUIs lookup table which
 189         can be found in :class:`lib.gui.utils.FileHandler`
 190     action_option: str
 191         The command line option that dictates the context of the file dialog to be opened.
 192         Bespoke actions are set in :class:`lib.gui.utils.FileHandler`
 193 
 194     Example
 195     -------
 196     Assuming an argument has already been set with option string `-a` indicating the action to be
 197     performed, the following will pop a different type of dialog depending on the action selected:
 198 
 199     >>> argument_list = []
 200     >>> argument_list.append(dict(
 201     >>>        opts=("-f", "--input_video"),
 202     >>>        action=ContextFullPaths,
 203     >>>        filetypes="video",
 204     >>>        action_option="-a"))
 205     """
 206     # pylint: disable=too-few-public-methods, too-many-arguments
 207     def __init__(self, *args, filetypes=None, action_option=None, **kwargs):
 208         opt = kwargs["option_strings"]
 209         if kwargs.get("nargs", None) is not None:
 210             raise ValueError("nargs not allowed for ContextFullPaths: {}".format(opt))
 211         if filetypes is None:
 212             raise ValueError("filetypes is required for ContextFullPaths: {}".format(opt))
 213         if action_option is None:
 214             raise ValueError("action_option is required for ContextFullPaths: {}".format(opt))
 215         super().__init__(*args, filetypes=filetypes, **kwargs)
 216         self.action_option = action_option
 217 
 218     def _get_kwargs(self):
 219         names = ["option_strings",
 220                  "dest",
 221                  "nargs",
 222                  "const",
 223                  "default",
 224                  "type",
 225                  "choices",
 226                  "help",
 227                  "metavar",
 228                  "filetypes",
 229                  "action_option"]
 230         return [(name, getattr(self, name)) for name in names]

 235 class Radio(argparse.Action):  # pylint: disable=too-few-public-methods
 236     """ Adds support for a GUI Radio options box.
 237 
 238     This is a standard :class:`argparse.Action` (with stock parameters) which indicates to the GUI
 239     that the options passed should be rendered as a group of Radio Buttons rather than a combo box.
 240 
 241     No additional parameters are required, but the :attr:`choices` parameter must be provided as
 242     these will be the Radio Box options. :attr:`nargs` are explicitly disallowed.
 243 
 244     Example
 245     -------
 246     >>> argument_list = []
 247     >>> argument_list.append(dict(
 248     >>>        opts=("-f", "--foobar"),
 249     >>>        action=Radio,
 250     >>>        choices=["foo", "bar"))
 251     """
 252     def __init__(self, *args, **kwargs):
 253         opt = kwargs["option_strings"]
 254         if kwargs.get("nargs", None) is not None:
 255             raise ValueError("nargs not allowed for Radio buttons: {}".format(opt))
 256         if not kwargs.get("choices", []):
 257             raise ValueError("Choices must be provided for Radio buttons: {}".format(opt))
 258         super().__init__(*args, **kwargs)
 259 
 260     def __call__(self, parser, namespace, values, option_string=None):
 261         setattr(namespace, self.dest, values)

 292 class Slider(argparse.Action):  # pylint: disable=too-few-public-methods
 293     """ Adds support for a slider in the GUI.
 294 
 295     The standard :class:`argparse.Action` is extended with the additional parameters listed below.
 296     The :attr:`default` value must be supplied and the :attr:`type` must be either :class:`int` or
 297     :class:`float`. :attr:`nargs` are explicitly disallowed.
 298 
 299     Parameters
 300     ----------
 301     min_max: tuple
 302         The (`min`, `max`) values that the slider's range should be set to. The values should be a
 303         pair of `float` or `int` data types, depending on the data type of the slider. NB: These
 304         min/max values are not enforced, they are purely for setting the slider range. Values
 305         outside of this range can still be explicitly passed in from the cli.
 306     rounding: int
 307         If the underlying data type for the option is a `float` then this value is the number of
 308         decimal places to round the slider values to. If the underlying data type for the option is
 309         an `int` then this is the step interval between each value for the slider.
 310 
 311     Examples
 312     --------
 313     For integer values:
 314 
 315     >>> argument_list = []
 316     >>> argument_list.append(dict(
 317     >>>        opts=("-f", "--foobar"),
 318     >>>        action=Slider,
 319     >>>        min_max=(0, 10)
 320     >>>        rounding=1
 321     >>>        type=int,
 322     >>>        default=5))
 323 
 324     For floating point values:
 325 
 326     >>> argument_list = []
 327     >>> argument_list.append(dict(
 328     >>>        opts=("-f", "--foobar"),
 329     >>>        action=Slider,
 330     >>>        min_max=(0.00, 1.00)
 331     >>>        rounding=2
 332     >>>        type=float,
 333     >>>        default=5.00))
 334     """
 335     def __init__(self, *args, min_max=None, rounding=None, **kwargs):
 336         opt = kwargs["option_strings"]
 337         if kwargs.get("nargs", None) is not None:
 338             raise ValueError("nargs not allowed for Slider: {}".format(opt))
 339         if kwargs.get("default", None) is None:
 340             raise ValueError("A default value must be supplied for Slider: {}".format(opt))
 341         if kwargs.get("type", None) not in (int, float):
 342             raise ValueError("Sliders only accept int and float data types: {}".format(opt))
 343         if min_max is None:
 344             raise ValueError("min_max must be provided for Sliders: {}".format(opt))
 345         if rounding is None:
 346             raise ValueError("rounding must be provided for Sliders: {}".format(opt))
 347 
 348         super().__init__(*args, **kwargs)
 349         self.min_max = min_max
 350         self.rounding = rounding
 351 
 352     def _get_kwargs(self):
 353         names = ["option_strings",
 354                  "dest",
 355                  "nargs",
 356                  "const",
 357                  "default",
 358                  "type",
 359                  "choices",
 360                  "help",
 361                  "metavar",
 362                  "min_max",  # Tuple containing min and max values of scale
 363                  "rounding"]  # Decimal places to round floats to or step interval for ints
 364         return [(name, getattr(self, name)) for name in names]
 365 
 366     def __call__(self, parser, namespace, values, option_string=None):
 367         setattr(namespace, self.dest, values)
