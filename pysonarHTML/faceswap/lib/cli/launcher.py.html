  15 class ScriptExecutor():  # pylint:disable=too-few-public-methods
  16     """ Loads the relevant script modules and executes the script.
  17 
  18         This class is initialized in each of the argparsers for the relevant
  19         command, then execute script is called within their set_default
  20         function.
  21 
  22         Parameters
  23         ----------
  24         command: str
  25             The faceswap command that is being executed
  26         """
  27     def __init__(self, command):
  28         self._command = command.lower()
  29 
  30     def _import_script(self):
  31         """ Imports the relevant script as indicated by :attr:`_command` from the scripts folder.
  32 
  33         Returns
  34         -------
  35         class: Faceswap Script
  36             The uninitialized script from the faceswap scripts folder.
  37         """
  38         self._test_for_tf_version()
  39         self._test_for_gui()
  40         cmd = os.path.basename(sys.argv[0])
  41         src = "tools.{}".format(self._command.lower()) if cmd == "tools.py" else "scripts"
  42         mod = ".".join((src, self._command.lower()))
  43         module = import_module(mod)
  44         script = getattr(module, self._command.title())
  45         return script

  47     @staticmethod
  48     def _test_for_tf_version():
  49         """ Check that the required Tensorflow version is installed.
  50 
  51         Raises
  52         ------
  53         FaceswapError
  54             If Tensorflow is not found, or is not between versions 1.12 and 1.15
  55         """
  56         min_ver = 1.12
  57         max_ver = 1.15
  58         try:
  59             # Ensure tensorflow doesn't pin all threads to one core when using Math Kernel Library
  60             os.environ["KMP_AFFINITY"] = "disabled"
  61             import tensorflow as tf  # pylint:disable=import-outside-toplevel
  62         except ImportError as err:
  63             raise FaceswapError("There was an error importing Tensorflow. This is most likely "
  64                                 "because you do not have TensorFlow installed, or you are trying "
  65                                 "to run tensorflow-gpu on a system without an Nvidia graphics "
  66                                 "card. Original import error: {}".format(str(err)))
  67         tf_ver = float(".".join(tf.__version__.split(".")[:2]))  # pylint:disable=no-member
  68         if tf_ver < min_ver:
  69             raise FaceswapError("The minimum supported Tensorflow is version {} but you have "
  70                                 "version {} installed. Please upgrade Tensorflow.".format(
  71                                     min_ver, tf_ver))
  72         if tf_ver > max_ver:
  73             raise FaceswapError("The maximumum supported Tensorflow is version {} but you have "
  74                                 "version {} installed. Please downgrade Tensorflow.".format(
  75                                     max_ver, tf_ver))
  76         logger.debug("Installed Tensorflow Version: %s", tf_ver)
  77 
  78     def _test_for_gui(self):
  79         """ If running the gui, performs check to ensure necessary prerequisites are present. """
  80         if self._command != "gui":
  81             return
  82         self._test_tkinter()
  83         self._check_display()

  85     @staticmethod
  86     def _test_tkinter():
  87         """ If the user is running the GUI, test whether the tkinter app is available on their
  88         machine. If not exit gracefully.
  89 
  90         This avoids having to import every tkinter function within the GUI in a wrapper and
  91         potentially spamming traceback errors to console.
  92 
  93         Raises
  94         ------
  95         FaceswapError
  96             If tkinter cannot be imported
  97         """
  98         try:
  99             # pylint: disable=unused-variable
 100             import tkinter  # noqa pylint: disable=unused-import,import-outside-toplevel
 101         except ImportError:
 102             logger.error("It looks like TkInter isn't installed for your OS, so the GUI has been "
 103                          "disabled. To enable the GUI please install the TkInter application. You "
 104                          "can try:")
 105             logger.info("Anaconda: conda install tk")
 106             logger.info("Windows/macOS: Install ActiveTcl Community Edition from "
 107                         "http://www.activestate.com")
 108             logger.info("Ubuntu/Mint/Debian: sudo apt install python3-tk")
 109             logger.info("Arch: sudo pacman -S tk")
 110             logger.info("CentOS/Redhat: sudo yum install tkinter")
 111             logger.info("Fedora: sudo dnf install python3-tkinter")
 112             raise FaceswapError("TkInter not found")

 114     @staticmethod
 115     def _check_display():
 116         """ Check whether there is a display to output the GUI to.
 117 
 118         If running on Windows then it is assumed that we are not running in headless mode
 119 
 120         Raises
 121         ------
 122         FaceswapError
 123             If a DISPLAY environmental cannot be found
 124         """
 125         if not os.environ.get("DISPLAY", None) and os.name != "nt":
 126             if platform.system() == "Darwin":
 127                 logger.info("macOS users need to install XQuartz. "
 128                             "See https://support.apple.com/en-gb/HT201341")
 129             raise FaceswapError("No display detected. GUI mode has been disabled.")

 131     def execute_script(self, arguments):
 132         """ Performs final set up and launches the requested :attr:`_command` with the given
 133         command line arguments.
 134 
 135         Monitors for errors and attempts to shut down the process cleanly on exit.
 136 
 137         Parameters
 138         ----------
 139         arguments: :class:`argparse.Namespace`
 140             The command line arguments to be passed to the executing script.
 141         """
 142         set_system_verbosity(arguments.loglevel)
 143         is_gui = hasattr(arguments, "redirect_gui") and arguments.redirect_gui
 144         log_setup(arguments.loglevel, arguments.logfile, self._command, is_gui)
 145         logger.debug("Executing: %s. PID: %s", self._command, os.getpid())
 146         success = False
 147         if get_backend() == "amd":
 148             plaidml_found = self._setup_amd(arguments.loglevel)
 149             if not plaidml_found:
 150                 safe_shutdown(got_error=True)
 151                 return
 152         try:
 153             script = self._import_script()
 154             process = script(arguments)
 155             process.process()
 156             success = True
 157         except FaceswapError as err:
 158             for line in str(err).splitlines():
 159                 logger.error(line)
 160         except KeyboardInterrupt:  # pylint: disable=try-except-raise
 161             raise
 162         except SystemExit:
 163             pass
 164         except Exception:  # pylint: disable=broad-except
 165             crash_file = crash_log()
 166             logger.exception("Got Exception on main handler:")
 167             logger.critical("An unexpected crash has occurred. Crash report written to '%s'. "
 168                             "You MUST provide this file if seeking assistance. Please verify you "
 169                             "are running the latest version of faceswap before reporting",
 170                             crash_file)
 171 
 172         finally:
 173             safe_shutdown(got_error=not success)
 174 
 175     @staticmethod
 176     def _setup_amd(log_level):
 177         """ Test for plaidml and perform setup for AMD.
 178 
 179         Parameters
 180         ----------
 181         log_level: str
 182             The requested log level to run at
 183         """
 184         logger.debug("Setting up for AMD")
 185         try:
 186             import plaidml  # noqa pylint:disable=unused-import,import-outside-toplevel
 187         except ImportError:
 188             logger.error("PlaidML not found. Run `pip install plaidml-keras` for AMD support")
 189             return False
 190         from lib.plaidml_tools import setup_plaidml  # pylint:disable=import-outside-toplevel
 191         setup_plaidml(log_level)
 192         logger.debug("setup up for PlaidML")
 193         return True
