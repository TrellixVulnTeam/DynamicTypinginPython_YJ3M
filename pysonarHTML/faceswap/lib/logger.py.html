  14 class FaceswapLogger(logging.Logger):
  15     """ Create custom logger  with custom levels """
  16     def __init__(self, name):
  17         for new_level in (("VERBOSE", 15), ("TRACE", 5)):
  18             level_name, level_num = new_level
  19             if hasattr(logging, level_name):
  20                 continue
  21             logging.addLevelName(level_num, level_name)
  22             setattr(logging, level_name, level_num)
  23         super().__init__(name)
  24 
  25     def verbose(self, msg, *args, **kwargs):
  26         """
  27         Log 'msg % args' with severity 'VERBOSE'.
  28         """
  29         if self.isEnabledFor(15):
  30             self._log(15, msg, args, **kwargs)
  31 
  32     def trace(self, msg, *args, **kwargs):
  33         """
  34         Log 'msg % args' with severity 'VERBOSE'.
  35         """
  36         if self.isEnabledFor(5):
  37             self._log(5, msg, args, **kwargs)

  40 class FaceswapFormatter(logging.Formatter):
  41     """ Override formatter to strip newlines the final message """
  42 
  43     def format(self, record):
  44         record.message = record.getMessage()
  45         record = self.rewrite_tf_deprecation(record)
  46         # strip newlines
  47         if "\n" in record.message or "\r" in record.message:
  48             record.message = record.message.replace("\n", "\\n").replace("\r", "\\r")
  49 
  50         if self.usesTime():
  51             record.asctime = self.formatTime(record, self.datefmt)
  52         msg = self.formatMessage(record)
  53         if record.exc_info:
  54             # Cache the traceback text to avoid converting it multiple times
  55             # (it's constant anyway)
  56             if not record.exc_text:
  57                 record.exc_text = self.formatException(record.exc_info)
  58         if record.exc_text:
  59             if msg[-1:] != "\n":
  60                 msg = msg + "\n"
  61             msg = msg + record.exc_text
  62         if record.stack_info:
  63             if msg[-1:] != "\n":
  64                 msg = msg + "\n"
  65             msg = msg + self.formatStack(record.stack_info)
  66         return msg
  67 
  68     @staticmethod
  69     def rewrite_tf_deprecation(record):
  70         """ Change TF deprecation messages from WARNING to DEBUG """
  71         if record.levelno == 30 and (record.funcName == "_tfmw_add_deprecation_warning" or
  72                                      record.module in("deprecation", "deprecation_wrapper")):
  73             record.levelno = 10
  74             record.levelname = "DEBUG"
  75         return record

  78 class RollingBuffer(collections.deque):
  79     """File-like that keeps a certain number of lines of text in memory."""
  80     def write(self, buffer):
  81         """ Write line to buffer """
  82         for line in buffer.rstrip().splitlines():
  83             self.append(line + "\n")

  86 class TqdmHandler(logging.StreamHandler):
  87     """ Use TQDM Write for outputting to console """
  88     def emit(self, record):
  89         msg = self.format(record)
  90         tqdm.write(msg)

 100 def log_setup(loglevel, logfile, command, is_gui=False):
 101     """ initial log set up. """
 102     numeric_loglevel = get_loglevel(loglevel)
 103     root_loglevel = min(logging.DEBUG, numeric_loglevel)
 104     rootlogger = set_root_logger(loglevel=root_loglevel)
 105     log_format = FaceswapFormatter("%(asctime)s %(processName)-15s %(threadName)-15s "
 106                                    "%(module)-15s %(funcName)-25s %(levelname)-8s %(message)s",
 107                                    datefmt="%m/%d/%Y %H:%M:%S")
 108     f_handler = file_handler(numeric_loglevel, logfile, log_format, command)
 109     s_handler = stream_handler(numeric_loglevel, is_gui)
 110     c_handler = crash_handler(log_format)
 111     rootlogger.addHandler(f_handler)
 112     rootlogger.addHandler(s_handler)
 113     rootlogger.addHandler(c_handler)
 114     logging.info("Log level set to: %s", loglevel.upper())

 117 def file_handler(loglevel, logfile, log_format, command):
 118     """ Add a logging rotating file handler """
 119     if logfile is not None:
 120         filename = logfile
 121     else:
 122         filename = os.path.join(os.path.dirname(os.path.realpath(sys.argv[0])), "faceswap")
 123         # Windows has issues sharing the log file with subprocesses, so log GUI separately
 124         filename += "_gui.log" if command == "gui" else ".log"
 125 
 126     should_rotate = os.path.isfile(filename)
 127     log_file = RotatingFileHandler(filename, backupCount=1)
 128     if should_rotate:
 129         log_file.doRollover()
 130     log_file.setFormatter(log_format)
 131     log_file.setLevel(loglevel)
 132     return log_file

 135 def stream_handler(loglevel, is_gui):
 136     """ Add a logging cli handler """
 137     # Don't set stdout to lower than verbose
 138     loglevel = max(loglevel, 15)
 139     log_format = FaceswapFormatter("%(asctime)s %(levelname)-8s %(message)s",
 140                                    datefmt="%m/%d/%Y %H:%M:%S")
 141 
 142     if is_gui:
 143         # tqdm.write inserts extra lines in the GUI, so use standard output as
 144         # it is not needed there.
 145         log_console = logging.StreamHandler(sys.stdout)
 146     else:
 147         log_console = TqdmHandler(sys.stdout)
 148     log_console.setFormatter(log_format)
 149     log_console.setLevel(loglevel)
 150     return log_console

 153 def crash_handler(log_format):
 154     """ Add a handler that sores the last 100 debug lines to 'debug_buffer'
 155         for use in crash reports """
 156     log_crash = logging.StreamHandler(debug_buffer)
 157     log_crash.setFormatter(log_format)
 158     log_crash.setLevel(logging.DEBUG)
 159     return log_crash

 162 def get_loglevel(loglevel):
 163     """ Check valid log level supplied and return numeric log level """
 164     numeric_level = getattr(logging, loglevel.upper(), None)
 165     if not isinstance(numeric_level, int):
 166         raise ValueError("Invalid log level: %s" % loglevel)
 167 
 168     return numeric_level

 171 def crash_log():
 172     """ Write debug_buffer to a crash log on crash """
 173     original_traceback = traceback.format_exc()
 174     path = os.path.dirname(os.path.realpath(sys.argv[0]))
 175     filename = os.path.join(path, datetime.now().strftime("crash_report.%Y.%m.%d.%H%M%S%f.log"))
 176     freeze_log = list(debug_buffer)
 177     try:
 178         from lib.sysinfo import sysinfo  # pylint:disable=import-outside-toplevel
 179     except Exception:  # pylint:disable=broad-except
 180         sysinfo = ("\n\nThere was an error importing System Information from lib.sysinfo. This is "
 181                    "probably a bug which should be fixed:\n{}".format(traceback.format_exc()))
 182     with open(filename, "w") as outfile:
 183         outfile.writelines(freeze_log)
 184         outfile.write(original_traceback)
 185         outfile.write(sysinfo)
 186     return filename
