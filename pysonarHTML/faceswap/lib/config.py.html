  16 class FaceswapConfig():
  17     """ Config Items """
  18     def __init__(self, section, configfile=None):
  19         """ Init Configuration  """
  20         logger.debug("Initializing: %s", self.__class__.__name__)
  21         self.configfile = self.get_config_file(configfile)
  22         self.config = ConfigParser(allow_no_value=True)
  23         self.defaults = OrderedDict()
  24         self.config.optionxform = str
  25         self.section = section
  26 
  27         self.set_defaults()
  28         self.handle_config()
  29         logger.debug("Initialized: %s", self.__class__.__name__)

  31     @property
  32     def changeable_items(self):
  33         """ Training only.
  34             Return a dict of config items with their set values for items
  35             that can be altered after the model has been created """
  36         retval = dict()
  37         for sect in ("global", self.section):
  38             if sect not in self.defaults:
  39                 continue
  40             for key, val in self.defaults[sect].items():
  41                 if key == "helptext" or val["fixed"]:
  42                     continue
  43                 retval[key] = self.get(sect, key)
  44         logger.debug("Alterable for existing models: %s", retval)
  45         return retval

  47     def set_defaults(self):
  48         """ Override for plugin specific config defaults
  49 
  50             Should be a series of self.add_section() and self.add_item() calls
  51 
  52             e.g:
  53 
  54             section = "sect_1"
  55             self.add_section(title=section,
  56                          info="Section 1 Information")
  57 
  58             self.add_item(section=section,
  59                           title="option_1",
  60                           datatype=bool,
  61                           default=False,
  62                           info="sect_1 option_1 information")
  63         """
  64         raise NotImplementedError

  66     @property
  67     def config_dict(self):
  68         """ Collate global options and requested section into a dictionary
  69             with the correct datatypes """
  70         conf = dict()
  71         for sect in ("global", self.section):
  72             if sect not in self.config.sections():
  73                 continue
  74             for key in self.config[sect]:
  75                 if key.startswith(("#", "\n")):  # Skip comments
  76                     continue
  77                 conf[key] = self.get(sect, key)
  78         return conf
  79 
  80     def get(self, section, option):
  81         """ Return a config item in it's correct format """
  82         logger.debug("Getting config item: (section: '%s', option: '%s')", section, option)
  83         datatype = self.defaults[section][option]["type"]
  84         if datatype == bool:
  85             func = self.config.getboolean
  86         elif datatype == int:
  87             func = self.config.getint
  88         elif datatype == float:
  89             func = self.config.getfloat
  90         else:
  91             func = self.config.get
  92         retval = func(section, option)
  93         if isinstance(retval, str) and retval.lower() == "none":
  94             retval = None
  95         logger.debug("Returning item: (type: %s, value: %s)", datatype, retval)
  96         return retval

  98     def get_config_file(self, configfile):
  99         """ Return the config file from the calling folder or the provided file """
 100         if configfile is not None:
 101             if not os.path.isfile(configfile):
 102                 err = "Config file does not exist at: {}".format(configfile)
 103                 logger.error(err)
 104                 raise ValueError(err)
 105             return configfile
 106         dirname = os.path.dirname(sys.modules[self.__module__].__file__)
 107         folder, fname = os.path.split(dirname)
 108         retval = os.path.join(os.path.dirname(folder), "config", "{}.ini".format(fname))
 109         logger.debug("Config File location: '%s'", retval)
 110         return retval

 112     def add_section(self, title=None, info=None):
 113         """ Add a default section to config file """
 114         logger.debug("Add section: (title: '%s', info: '%s')", title, info)
 115         if None in (title, info):
 116             raise ValueError("Default config sections must have a title and "
 117                              "information text")
 118         self.defaults[title] = OrderedDict()
 119         self.defaults[title]["helptext"] = info

 121     def add_item(self, section=None, title=None, datatype=str, default=None, info=None,
 122                  rounding=None, min_max=None, choices=None, gui_radio=False, fixed=True,
 123                  group=None):
 124         """ Add a default item to a config section
 125 
 126             For int or float values, rounding and min_max must be set
 127             This is for the slider in the GUI. The min/max values are not enforced:
 128             rounding:   sets the decimal places for floats or the step interval for ints.
 129             min_max:    tuple of min and max accepted values
 130 
 131             For str values choices can be set to validate input and create a combo box
 132             in the GUI
 133 
 134             is_radio is to indicate to the GUI that it should display Radio Buttons rather than
 135             combo boxes for multiple choice options.
 136 
 137             The 'fixed' parameter is only for training configs. Training configurations
 138             are set when the model is created, and then reloaded from the state file.
 139             Marking an item as fixed=False indicates that this value can be changed for
 140             existing models, and will overide the value saved in the state file with the
 141             updated value in config.
 142 
 143             The 'Group' parameter allows you to assign the config item to a group in the GUI
 144 
 145         """
 146         logger.debug("Add item: (section: '%s', title: '%s', datatype: '%s', default: '%s', "
 147                      "info: '%s', rounding: '%s', min_max: %s, choices: %s, gui_radio: %s, "
 148                      "fixed: %s, group: %s)", section, title, datatype, default, info, rounding,
 149                      min_max, choices, gui_radio, fixed, group)
 150 
 151         choices = list() if not choices else choices
 152 
 153         if None in (section, title, default, info):
 154             raise ValueError("Default config items must have a section, "
 155                              "title, defult and  "
 156                              "information text")
 157         if not self.defaults.get(section, None):
 158             raise ValueError("Section does not exist: {}".format(section))
 159         if datatype not in (str, bool, float, int):
 160             raise ValueError("'datatype' must be one of str, bool, float or "
 161                              "int: {} - {}".format(section, title))
 162         if datatype in (float, int) and (rounding is None or min_max is None):
 163             raise ValueError("'rounding' and 'min_max' must be set for numerical options")
 164         if not isinstance(choices, (list, tuple)):
 165             raise ValueError("'choices' must be a list or tuple")
 166 
 167         info = self.expand_helptext(info, choices, default, datatype, min_max, fixed)
 168         self.defaults[section][title] = {"default": default,
 169                                          "helptext": info,
 170                                          "type": datatype,
 171                                          "rounding": rounding,
 172                                          "min_max": min_max,
 173                                          "choices": choices,
 174                                          "gui_radio": gui_radio,
 175                                          "fixed": fixed,
 176                                          "group": group}
 177 
 178     @staticmethod
 179     def expand_helptext(helptext, choices, default, datatype, min_max, fixed):
 180         """ Add extra helptext info from parameters """
 181         if not fixed:
 182             helptext += "\nThis option can be updated for existing models."
 183         if choices:
 184             helptext += "\nChoose from: {}".format(choices)
 185         elif datatype == bool:
 186             helptext += "\nChoose from: True, False"
 187         elif datatype == int:
 188             cmin, cmax = min_max
 189             helptext += "\nSelect an integer between {} and {}".format(cmin, cmax)
 190         elif datatype == float:
 191             cmin, cmax = min_max
 192             helptext += "\nSelect a decimal number between {} and {}".format(cmin, cmax)
 193         helptext += "\n[Default: {}]".format(default)
 194         return helptext

 196     def check_exists(self):
 197         """ Check that a config file exists """
 198         if not os.path.isfile(self.configfile):
 199             logger.debug("Config file does not exist: '%s'", self.configfile)
 200             return False
 201         logger.debug("Config file exists: '%s'", self.configfile)
 202         return True

 204     def create_default(self):
 205         """ Generate a default config if it does not exist """
 206         logger.debug("Creating default Config")
 207         for section, items in self.defaults.items():
 208             logger.debug("Adding section: '%s')", section)
 209             self.insert_config_section(section, items["helptext"])
 210             for item, opt in items.items():
 211                 logger.debug("Adding option: (item: '%s', opt: '%s'", item, opt)
 212                 if item == "helptext":
 213                     continue
 214                 self.insert_config_item(section,
 215                                         item,
 216                                         opt["default"],
 217                                         opt)
 218         self.save_config()

 220     def insert_config_section(self, section, helptext, config=None):
 221         """ Insert a section into the config """
 222         logger.debug("Inserting section: (section: '%s', helptext: '%s', config: '%s')",
 223                      section, helptext, config)
 224         config = self.config if config is None else config
 225         helptext = self.format_help(helptext, is_section=True)
 226         config.add_section(section)
 227         config.set(section, helptext)
 228         logger.debug("Inserted section: '%s'", section)
 229 
 230     def insert_config_item(self, section, item, default, option,
 231                            config=None):
 232         """ Insert an item into a config section """
 233         logger.debug("Inserting item: (section: '%s', item: '%s', default: '%s', helptext: '%s', "
 234                      "config: '%s')", section, item, default, option["helptext"], config)
 235         config = self.config if config is None else config
 236         helptext = option["helptext"]
 237         helptext = self.format_help(helptext, is_section=False)
 238         config.set(section, helptext)
 239         config.set(section, item, str(default))
 240         logger.debug("Inserted item: '%s'", item)

 242     @staticmethod
 243     def format_help(helptext, is_section=False):
 244         """ Format comments for default ini file """
 245         logger.debug("Formatting help: (helptext: '%s', is_section: '%s')", helptext, is_section)
 246         helptext = '# {}'.format(helptext.replace("\n", "\n# "))
 247         if is_section:
 248             helptext = helptext.upper()
 249         else:
 250             helptext = "\n{}".format(helptext)
 251         logger.debug("formatted help: '%s'", helptext)
 252         return helptext

 254     def load_config(self):
 255         """ Load values from config """
 256         logger.verbose("Loading config: '%s'", self.configfile)
 257         self.config.read(self.configfile)

 259     def save_config(self):
 260         """ Save a config file """
 261         logger.info("Updating config at: '%s'", self.configfile)
 262         f_cfgfile = open(self.configfile, "w")
 263         self.config.write(f_cfgfile)
 264         f_cfgfile.close()
 265         logger.debug("Updated config at: '%s'", self.configfile)

 267     def validate_config(self):
 268         """ Check for options in default config against saved config
 269             and add/remove as appropriate """
 270         logger.debug("Validating config")
 271         if self.check_config_change():
 272             self.add_new_config_items()
 273         self.check_config_choices()
 274         logger.debug("Validated config")
 275 
 276     def add_new_config_items(self):
 277         """ Add new items to the config file """
 278         logger.debug("Updating config")
 279         new_config = ConfigParser(allow_no_value=True)
 280         for section, items in self.defaults.items():
 281             self.insert_config_section(section, items["helptext"], new_config)
 282             for item, opt in items.items():
 283                 if item == "helptext":
 284                     continue
 285                 if section not in self.config.sections():
 286                     logger.debug("Adding new config section: '%s'", section)
 287                     opt_value = opt["default"]
 288                 else:
 289                     opt_value = self.config[section].get(item, opt["default"])
 290                 self.insert_config_item(section,
 291                                         item,
 292                                         opt_value,
 293                                         opt,
 294                                         new_config)
 295         self.config = new_config
 296         self.config.optionxform = str
 297         self.save_config()
 298         logger.debug("Updated config")
 299 
 300     def check_config_choices(self):
 301         """ Check that config items are valid choices """
 302         logger.debug("Checking config choices")
 303         for section, items in self.defaults.items():
 304             for item, opt in items.items():
 305                 if item == "helptext" or not opt["choices"]:
 306                     continue
 307                 opt_value = self.config.get(section, item)
 308                 if opt_value.lower() == "none" and any(choice.lower() == "none"
 309                                                        for choice in opt["choices"]):
 310                     continue
 311                 if opt_value not in opt["choices"]:
 312                     default = str(opt["default"])
 313                     logger.warning("'%s' is not a valid config choice for '%s': '%s'. Defaulting "
 314                                    "to: '%s'", opt_value, section, item, default)
 315                     self.config.set(section, item, default)
 316         logger.debug("Checked config choices")
 317 
 318     def check_config_change(self):
 319         """ Check whether new default items have been added or removed
 320             from the config file compared to saved version """
 321         if set(self.config.sections()) != set(self.defaults.keys()):
 322             logger.debug("Default config has new section(s)")
 323             return True
 324 
 325         for section, items in self.defaults.items():
 326             opts = [opt for opt in items.keys() if opt != "helptext"]
 327             exists = [opt for opt in self.config[section].keys()
 328                       if not opt.startswith(("# ", "\n# "))]
 329             if set(exists) != set(opts):
 330                 logger.debug("Default config has new item(s)")
 331                 return True
 332         logger.debug("Default config has not changed")
 333         return False
 334 
 335     def handle_config(self):
 336         """ Handle the config """
 337         logger.debug("Handling config")
 338         if not self.check_exists():
 339             self.create_default()
 340         self.load_config()
 341         self.validate_config()
 342         logger.debug("Handled config")

 345 def generate_configs():
 346     """ Generate config files if they don't exist.
 347 
 348     This script is run prior to anything being set up, so don't use logging
 349     Generates the default config files for plugins in the faceswap config folder
 350     """
 351 
 352     base_path = os.path.realpath(os.path.dirname(sys.argv[0]))
 353     plugins_path = os.path.join(base_path, "plugins")
 354     configs_path = os.path.join(base_path, "config")
 355     for dirpath, _, filenames in os.walk(plugins_path):
 356         if "_config.py" in filenames:
 357             section = os.path.split(dirpath)[-1]
 358             config_file = os.path.join(configs_path, "{}.ini".format(section))
 359             if not os.path.exists(config_file):
 360                 mod = import_module("plugins.{}.{}".format(section, "_config"))
 361                 mod.Config(None)
