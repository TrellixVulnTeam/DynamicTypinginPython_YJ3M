  14 _INIT = False
  15 _LOGGER = None

  18 class PlaidMLStats():
  19     """ Stats for plaidML """
  20     def __init__(self, loglevel="INFO", log=True):
  21         if not _INIT and log:
  22             # Logger is held internally, as we don't want to log
  23             # when obtaining system stats on crash
  24             global _LOGGER  # pylint:disable=global-statement
  25             _LOGGER = logging.getLogger(__name__)  # pylint:disable=invalid-name
  26             _LOGGER.debug("Initializing: %s: (loglevel: %s, log: %s)",
  27                           self.__class__.__name__, loglevel, log)
  28         self.initialize(loglevel)
  29         self.ctx = plaidml.Context()
  30         self.supported_devices = self.get_supported_devices()
  31         self.devices = self.get_all_devices()
  32 
  33         self.device_details = [json.loads(device.details.decode()) for device in self.devices]
  34         if _LOGGER:
  35             _LOGGER.debug("Initialized: %s", self.__class__.__name__)
  36 
  37     # PROPERTIES
  38     @property
  39     def active_devices(self):
  40         """ Return the active device IDs """
  41         return [idx for idx, d_id in enumerate(self.ids) if d_id in plaidml.settings.device_ids]
  42 
  43     @property
  44     def device_count(self):
  45         """ Return count of PlaidML Devices """
  46         return len(self.devices)
  47 
  48     @property
  49     def drivers(self):
  50         """ Return all PlaidML device drivers """
  51         return [device.get("driverVersion", "No Driver Found") for device in self.device_details]
  52 
  53     @property
  54     def vram(self):
  55         """ Return Total VRAM for all PlaidML Devices """
  56         return [int(device.get("globalMemSize", 0)) / (1024 * 1024)
  57                 for device in self.device_details]
  58 
  59     @property
  60     def max_alloc(self):
  61         """ Return Maximum allowed VRAM allocation for all PlaidML Devices """
  62         return [int(device.get("maxMemAllocSize", 0)) / (1024 * 1024)
  63                 for device in self.device_details]
  64 
  65     @property
  66     def ids(self):
  67         """ Return all PlaidML Device IDs """
  68         return [device.id.decode() for device in self.devices]
  69 
  70     @property
  71     def names(self):
  72         """ Return all PlaidML Device Names """
  73         return ["{} - {} ({})".format(
  74             device.get("vendor", "unknown"),
  75             device.get("name", "unknown"),
  76             "supported" if idx in self.supported_indices else "experimental")
  77                 for idx, device in enumerate(self.device_details)]
  78 
  79     @property
  80     def supported_indices(self):
  81         """ Return the indices from self.devices of GPUs categorized as supported """
  82         retval = [idx for idx, device in enumerate(self.devices)
  83                   if device in self.supported_devices]
  84         if _LOGGER:
  85             _LOGGER.debug(retval)
  86         return retval
  87 
  88     @property
  89     def experimental_indices(self):
  90         """ Return the indices from self.devices of GPUs categorized as experimental """
  91         retval = [idx for idx, device in enumerate(self.devices)
  92                   if device not in self.supported_devices]
  93         if _LOGGER:
  94             _LOGGER.debug(retval)
  95         return retval
  96 
  97     # INITIALIZATION
  98     def initialize(self, loglevel):
  99         """ Initialize PlaidML """
 100         global _INIT  # pylint:disable=global-statement
 101         if _INIT:
 102             if _LOGGER:
 103                 _LOGGER.debug("PlaidML already initialized")
 104             return
 105         if _LOGGER:
 106             _LOGGER.debug("Initializing PlaidML")
 107         self.set_plaidml_logger()
 108         self.set_verbosity(loglevel)
 109         _INIT = True
 110         if _LOGGER:
 111             _LOGGER.debug("Initialized PlaidML")
 112 
 113     @staticmethod
 114     def set_plaidml_logger():
 115         """ Set PlaidMLs default logger to Faceswap Logger and prevent propagation """
 116         if _LOGGER:
 117             _LOGGER.debug("Setting PlaidML Default Logger")
 118         plaidml.DEFAULT_LOG_HANDLER = logging.getLogger("plaidml_root")
 119         plaidml.DEFAULT_LOG_HANDLER.propagate = 0
 120         if _LOGGER:
 121             _LOGGER.debug("Set PlaidML Default Logger")
 122 
 123     @staticmethod
 124     def set_verbosity(loglevel):
 125         """ Set the PlaidML Verbosity """
 126         if _LOGGER:
 127             _LOGGER.debug("Setting PlaidML Loglevel: %s", loglevel)
 128         if isinstance(loglevel, int):
 129             numeric_level = loglevel
 130         else:
 131             numeric_level = getattr(logging, loglevel.upper(), None)
 132         if numeric_level < 10:
 133             # DEBUG Logging
 134             plaidml._internal_set_vlog(1)  # pylint:disable=protected-access
 135         elif numeric_level < 20:
 136             # INFO Logging
 137             plaidml._internal_set_vlog(0)  # pylint:disable=protected-access
 138         else:
 139             # WARNING Logging
 140             plaidml.quiet()
 141 
 142     def get_supported_devices(self):
 143         """ Return a list of supported devices """
 144         experimental_setting = plaidml.settings.experimental
 145         plaidml.settings.experimental = False
 146         devices, _ = plaidml.devices(self.ctx, limit=100, return_all=True)
 147         plaidml.settings.experimental = experimental_setting
 148 
 149         supported = [device for device in devices
 150                      if json.loads(device.details.decode()).get("type", "cpu").lower() == "gpu"]
 151         if _LOGGER:
 152             _LOGGER.debug(supported)
 153         return supported
 154 
 155     def get_all_devices(self):
 156         """ Return list of supported and experimental devices """
 157         experimental_setting = plaidml.settings.experimental
 158         plaidml.settings.experimental = True
 159         devices, _ = plaidml.devices(self.ctx, limit=100, return_all=True)
 160         plaidml.settings.experimental = experimental_setting
 161 
 162         experimental = [device for device in devices
 163                         if json.loads(device.details.decode()).get("type", "cpu").lower() == "gpu"]
 164         if _LOGGER:
 165             _LOGGER.debug("Experimental Devices: %s", experimental)
 166         all_devices = experimental + self.supported_devices
 167         if _LOGGER:
 168             _LOGGER.debug(all_devices)
 169         return all_devices
 170 
 171     def load_active_devices(self):
 172         """ Load settings from PlaidML.settings.usersettings or select biggest gpu """
 173         if not os.path.exists(plaidml.settings.user_settings):  # pylint:disable=no-member
 174             if _LOGGER:
 175                 _LOGGER.debug("Setting largest PlaidML device")
 176             self.set_largest_gpu()
 177         else:
 178             if _LOGGER:
 179                 _LOGGER.debug("Setting PlaidML devices from user_settings")
 180 
 181     def set_largest_gpu(self):
 182         """ Get a supported GPU with largest VRAM. If no supported, get largest experimental """
 183         category = "supported" if self.supported_devices else "experimental"
 184         if _LOGGER:
 185             _LOGGER.debug("Obtaining largest %s device", category)
 186         indices = getattr(self, "{}_indices".format(category))
 187         if not indices:
 188             _LOGGER.error("Failed to automatically detect your GPU.")
 189             _LOGGER.error("Please run `plaidml-setup` to set up your GPU.")
 190             exit()
 191         max_vram = max([self.vram[idx] for idx in indices])
 192         if _LOGGER:
 193             _LOGGER.debug("Max VRAM: %s", max_vram)
 194         gpu_idx = min([idx for idx, vram in enumerate(self.vram)
 195                        if vram == max_vram and idx in indices])
 196         if _LOGGER:
 197             _LOGGER.debug("GPU IDX: %s", gpu_idx)
 198 
 199         selected_gpu = self.ids[gpu_idx]
 200         if _LOGGER:
 201             _LOGGER.info("Setting GPU to largest available %s device. If you want to override "
 202                          "this selection, run `plaidml-setup` from the command line.", category)
 203 
 204         plaidml.settings.experimental = category == "experimental"
 205         plaidml.settings.device_ids = [selected_gpu]

 208 def setup_plaidml(loglevel):
 209     """ Setup plaidml for AMD Cards """
 210     logger = logging.getLogger(__name__)  # pylint:disable=invalid-name
 211     logger.info("Setting up for PlaidML")
 212     logger.verbose("Setting Keras Backend to PlaidML")
 213     os.environ["KERAS_BACKEND"] = "plaidml.keras.backend"
 214     plaid = PlaidMLStats(loglevel)
 215     plaid.load_active_devices()
 216     logger.info("Using GPU: %s", [plaid.ids[i] for i in plaid.active_devices])
 217     logger.info("Successfully set up for PlaidML")
