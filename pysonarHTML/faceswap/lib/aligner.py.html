  18     def extract(self, image, face, size):
  19         """ Extract a face from an image """
  20         logger.trace("size: %s", size)
  21         padding = int(size * 0.1875)
  22         alignment = get_align_mat(face)
  23         extracted = self.transform(image, alignment, size, padding)
  24         logger.trace("Returning face and alignment matrix: (alignment_matrix: %s)", alignment)
  25         return extracted, alignment

  27     @staticmethod
  28     def transform_matrix(mat, size, padding):
  29         """ Transform the matrix for current size and padding """
  30         logger.trace("size: %s. padding: %s", size, padding)
  31         matrix = mat * (size - 2 * padding)
  32         matrix[:, 2] += padding
  33         logger.trace("Returning: %s", matrix)
  34         return matrix
  35 
  36     def transform(self, image, mat, size, padding=0):
  37         """ Transform Image """
  38         logger.trace("matrix: %s, size: %s. padding: %s", mat, size, padding)
  39         matrix = self.transform_matrix(mat, size, padding)
  40         interpolators = get_matrix_scaling(matrix)
  41         retval = cv2.warpAffine(image, matrix, (size, size), flags=interpolators[0])
  42         return retval

  44     def transform_points(self, points, mat, size, padding=0):
  45         """ Transform points along matrix """
  46         logger.trace("points: %s, matrix: %s, size: %s. padding: %s", points, mat, size, padding)
  47         matrix = self.transform_matrix(mat, size, padding)
  48         points = np.expand_dims(points, axis=1)
  49         points = cv2.transform(points, matrix, points.shape)
  50         retval = np.squeeze(points)
  51         logger.trace("Returning: %s", retval)
  52         return retval

  54     def get_original_roi(self, mat, size, padding=0):
  55         """ Return the square aligned box location on the original image """
  56         logger.trace("matrix: %s, size: %s. padding: %s", mat, size, padding)
  57         matrix = self.transform_matrix(mat, size, padding)
  58         points = np.array([[0, 0], [0, size - 1], [size - 1, size - 1], [size - 1, 0]], np.int32)
  59         points = points.reshape((-1, 1, 2))
  60         matrix = cv2.invertAffineTransform(matrix)
  61         logger.trace("Returning: (points: %s, matrix: %s", points, matrix)
  62         return cv2.transform(points, matrix)

  64     @staticmethod
  65     def get_feature_mask(aligned_landmarks_68, size, padding=0, dilation=30):
  66         """ Return the face feature mask """
  67         logger.trace("aligned_landmarks_68: %s, size: %s, padding: %s, dilation: %s",
  68                      aligned_landmarks_68, size, padding, dilation)
  69         scale = size - 2 * padding
  70         translation = padding
  71         pad_mat = np.matrix([[scale, 0.0, translation], [0.0, scale, translation]])
  72         aligned_landmarks_68 = np.expand_dims(aligned_landmarks_68, axis=1)
  73         aligned_landmarks_68 = cv2.transform(aligned_landmarks_68,
  74                                              pad_mat,
  75                                              aligned_landmarks_68.shape)
  76         aligned_landmarks_68 = np.squeeze(aligned_landmarks_68)
  77         l_eye_points = aligned_landmarks_68[42:48].tolist()
  78         l_brow_points = aligned_landmarks_68[22:27].tolist()
  79         r_eye_points = aligned_landmarks_68[36:42].tolist()
  80         r_brow_points = aligned_landmarks_68[17:22].tolist()
  81         nose_points = aligned_landmarks_68[27:36].tolist()
  82         chin_points = aligned_landmarks_68[8:11].tolist()
  83         mouth_points = aligned_landmarks_68[48:68].tolist()
  84         # TODO remove excessive reshapes and flattens
  85 
  86         l_eye = np.array(l_eye_points + l_brow_points).reshape((-1, 2)).astype('int32').flatten()
  87         r_eye = np.array(r_eye_points + r_brow_points).reshape((-1, 2)).astype('int32').flatten()
  88         mouth = np.array(mouth_points + nose_points + chin_points)
  89         mouth = mouth.reshape((-1, 2)).astype('int32').flatten()
  90         l_eye_hull = cv2.convexHull(l_eye.reshape((-1, 2)))
  91         r_eye_hull = cv2.convexHull(r_eye.reshape((-1, 2)))
  92         mouth_hull = cv2.convexHull(mouth.reshape((-1, 2)))
  93 
  94         mask = np.zeros((size, size, 3), dtype=float)
  95         cv2.fillConvexPoly(mask, l_eye_hull, (1, 1, 1))
  96         cv2.fillConvexPoly(mask, r_eye_hull, (1, 1, 1))
  97         cv2.fillConvexPoly(mask, mouth_hull, (1, 1, 1))
  98 
  99         if dilation > 0:
 100             kernel = np.ones((dilation, dilation), np.uint8)
 101             mask = cv2.dilate(mask, kernel, iterations=1)
 102 
 103         logger.trace("Returning: %s", mask)
 104         return mask

 107 def get_matrix_scaling(mat):
 108     """ Get the correct interpolator """
 109     x_scale = np.sqrt(mat[0, 0] * mat[0, 0] + mat[0, 1] * mat[0, 1])
 110     y_scale = (mat[0, 0] * mat[1, 1] - mat[0, 1] * mat[1, 0]) / x_scale
 111     avg_scale = (x_scale + y_scale) * 0.5
 112     if avg_scale >= 1.:
 113         interpolators = cv2.INTER_CUBIC, cv2.INTER_AREA
 114     else:
 115         interpolators = cv2.INTER_AREA, cv2.INTER_CUBIC
 116     logger.trace("interpolator: %s, inverse interpolator: %s", interpolators[0], interpolators[1])
 117     return interpolators
