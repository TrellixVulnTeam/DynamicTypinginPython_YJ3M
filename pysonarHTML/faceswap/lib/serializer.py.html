  26 class Serializer():
  27     """ A convenience class for various serializers.
  28 
  29     This class should not be called directly as it acts as the parent for various serializers.
  30     All serializers should be called from :func:`get_serializer` or
  31     :func:`get_serializer_from_filename`
  32 
  33     Example
  34     -------
  35     >>> from lib.serializer import get_serializer
  36     >>> serializer = get_serializer('json')
  37     >>> json_file = '/path/to/json/file.json'
  38     >>> data = serializer.load(json_file)
  39     >>> serializer.save(json_file, data)
  40 
  41     """
  42     def __init__(self):
  43         self._file_extension = None
  44         self._write_option = "wb"
  45         self._read_option = "rb"
  46 
  47     @property
  48     def file_extension(self):
  49         """ str: The file extension of the serializer """
  50         return self._file_extension
  51 
  52     def save(self, filename, data):
  53         """ Serialize data and save to a file
  54 
  55         Parameters
  56         ----------
  57         filename: str
  58             The path to where the serialized file should be saved
  59         data: varies
  60             The data that is to be serialized to file
  61 
  62         Example
  63         ------
  64         >>> serializer = get_serializer('json')
  65         >>> data ['foo', 'bar']
  66         >>> json_file = '/path/to/json/file.json'
  67         >>> serializer.save(json_file, data)
  68         """
  69         logger.debug("filename: %s, data type: %s", filename, type(data))
  70         filename = self._check_extension(filename)
  71         try:
  72             with open(filename, self._write_option) as s_file:
  73                 s_file.write(self.marshal(data))
  74         except IOError as err:
  75             msg = "Error writing to '{}': {}".format(filename, err.strerror)
  76             raise FaceswapError(msg) from err
  77 
  78     def _check_extension(self, filename):
  79         """ Check the filename has an extension. If not add the correct one for the serializer """
  80         extension = os.path.splitext(filename)[1]
  81         retval = filename if extension else "{}.{}".format(filename, self.file_extension)
  82         logger.debug("Original filename: '%s', final filename: '%s'", filename, retval)
  83         return retval
  84 
  85     def load(self, filename):
  86         """ Load data from an existing serialized file
  87 
  88         Parameters
  89         ----------
  90         filename: str
  91             The path to the serialized file
  92 
  93         Returns
  94         ----------
  95         data: varies
  96             The data in a python object format
  97 
  98         Example
  99         ------
 100         >>> serializer = get_serializer('json')
 101         >>> json_file = '/path/to/json/file.json'
 102         >>> data = serializer.load(json_file)
 103         """
 104         logger.debug("filename: %s", filename)
 105         try:
 106             with open(filename, self._read_option) as s_file:
 107                 data = s_file.read()
 108                 logger.debug("stored data type: %s", type(data))
 109                 retval = self.unmarshal(data)
 110 
 111         except IOError as err:
 112             msg = "Error reading from '{}': {}".format(filename, err.strerror)
 113             raise FaceswapError(msg) from err
 114         logger.debug("data type: %s", type(retval))
 115         return retval

 117     def marshal(self, data):
 118         """ Serialize an object
 119 
 120         Parameters
 121         ----------
 122         data: varies
 123             The data that is to be serialized
 124 
 125         Returns
 126         -------
 127         data: varies
 128             The data in a the serialized data format
 129 
 130         Example
 131         ------
 132         >>> serializer = get_serializer('json')
 133         >>> data ['foo', 'bar']
 134         >>> json_data = serializer.marshal(data)
 135         """
 136         logger.debug("data type: %s", type(data))
 137         try:
 138             retval = self._marshal(data)
 139         except Exception as err:
 140             msg = "Error serializing data for type {}: {}".format(type(data), str(err))
 141             raise FaceswapError(msg) from err
 142         logger.debug("returned data type: %s", type(retval))
 143         return retval
 144 
 145     def unmarshal(self, serialized_data):
 146         """ Unserialize data to its original object type
 147 
 148         Parameters
 149         ----------
 150         serialized_data: varies
 151             Data in serializer format that is to be unmarshalled to its original object
 152 
 153         Returns
 154         -------
 155         data: varies
 156             The data in a python object format
 157 
 158         Example
 159         ------
 160         >>> serializer = get_serializer('json')
 161         >>> json_data = <json object>
 162         >>> data = serializer.unmarshal(json_data)
 163         """
 164         logger.debug("data type: %s", type(serialized_data))
 165         try:
 166             retval = self._unmarshal(serialized_data)
 167         except Exception as err:
 168             msg = "Error unserializing data for type {}: {}".format(type(serialized_data),
 169                                                                     str(err))
 170             raise FaceswapError(msg) from err
 171         logger.debug("returned data type: %s", type(retval))
 172         return retval

 185 class _YAMLSerializer(Serializer):
 186     """ YAML Serializer """
 187     def __init__(self):
 188         super().__init__()
 189         self._file_extension = "yml"
 190 
 191     @classmethod
 192     def _marshal(cls, data):
 193         return yaml.dump(data, default_flow_style=False).encode("utf-8")
 194 
 195     @classmethod
 196     def _unmarshal(cls, data):
 197         return yaml.load(data.decode("utf-8"), Loader=yaml.FullLoader)

 230 class _NPYSerializer(Serializer):
 231     """ NPY Serializer """
 232     def __init__(self):
 233         super().__init__()
 234         self._file_extension = "npy"
 235         self._bytes = BytesIO()
 236 
 237     def _marshal(self, data):
 238         """ NPY Marshal to bytesIO so standard bytes writer can write out """
 239         b_handler = BytesIO()
 240         np.save(b_handler, data)
 241         b_handler.seek(0)
 242         return b_handler.read()
 243 
 244     def _unmarshal(self, data):
 245         """ NPY Unmarshal to bytesIO so we can use numpy loader """
 246         b_handler = BytesIO(data)
 247         retval = np.load(b_handler)
 248         del b_handler
 249         if retval.dtype == "object":
 250             retval = retval[()]
 251         return retval

 254 class _CompressedSerializer(Serializer):
 255     """ A compressed pickle serializer for Faceswap """
 256     def __init__(self):
 257         super().__init__()
 258         self._file_extension = "fsa"
 259         self._child = get_serializer("pickle")
 260 
 261     def _marshal(self, data):
 262         """ Pickle and compress data """
 263         data = self._child._marshal(data)  # pylint: disable=protected-access
 264         return zlib.compress(data)
 265 
 266     def _unmarshal(self, data):
 267         """ Decompress and unpicke data """
 268         data = zlib.decompress(data)
 269         return self._child._unmarshal(data)  # pylint: disable=protected-access

 272 def get_serializer(serializer):
 273     """ Obtain a serializer object
 274 
 275     Parameters
 276     ----------
 277     serializer: {'json', 'pickle', yaml', 'npy', 'compressed'}
 278         The required serializer format
 279 
 280     Returns
 281     -------
 282     serializer: :class:`Serializer`
 283         A serializer object for handling the requested data format
 284 
 285     Example
 286     -------
 287     >>> serializer = get_serializer('json')
 288     """
 289     if serializer.lower() == "npy":
 290         retval = _NPYSerializer()
 291     elif serializer.lower() == "compressed":
 292         retval = _CompressedSerializer()
 293     elif serializer.lower() == "json":
 294         retval = _JSONSerializer()
 295     elif serializer.lower() == "pickle":
 296         retval = _PickleSerializer()
 297     elif serializer.lower() == "yaml" and yaml is not None:
 298         retval = _YAMLSerializer()
 299     elif serializer.lower() == "yaml" and yaml is None:
 300         logger.warning("You must have PyYAML installed to use YAML as the serializer."
 301                        "Switching to JSON as the serializer.")
 302         retval = _JSONSerializer
 303     else:
 304         logger.warning("Unrecognized serializer: '%s'. Returning json serializer", serializer)
 305     logger.debug(retval)
 306     return retval

 309 def get_serializer_from_filename(filename):
 310     """ Obtain a serializer object from a filename
 311 
 312     Parameters
 313     ----------
 314     filename: str
 315         Filename to determine the serializer type from
 316 
 317     Returns
 318     -------
 319     serializer: :class:`Serializer`
 320         A serializer object for handling the requested data format
 321 
 322     Example
 323     -------
 324     >>> filename = '/path/to/json/file.json'
 325     >>> serializer = get_serializer_from_filename(filename)
 326     """
 327     logger.debug("filename: '%s'", filename)
 328     extension = os.path.splitext(filename)[1].lower()
 329     logger.debug("extension: '%s'", extension)
 330 
 331     if extension == ".json":
 332         retval = _JSONSerializer()
 333     elif extension in (".p", ".pickle"):
 334         retval = _PickleSerializer()
 335     elif extension == ".npy":
 336         retval = _NPYSerializer()
 337     elif extension == ".fsa":
 338         retval = _CompressedSerializer()
 339     elif extension in (".yaml", ".yml") and yaml is not None:
 340         retval = _YAMLSerializer()
 341     elif extension in (".yaml", ".yml") and yaml is None:
 342         logger.warning("You must have PyYAML installed to use YAML as the serializer.\n"
 343                        "Switching to JSON as the serializer.")
 344         retval = _JSONSerializer()
 345     else:
 346         logger.warning("Unrecognized extension: '%s'. Returning json serializer", extension)
 347         retval = _JSONSerializer()
 348     logger.debug(retval)
 349     return retval
