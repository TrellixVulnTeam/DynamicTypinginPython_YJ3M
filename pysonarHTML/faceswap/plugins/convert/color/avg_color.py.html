   8 class Color(Adjustment):
   9     """ Adjust the mean of the color channels to be the same for the swap and old frame """
  10 
  11     @staticmethod
  12     def process(old_face, new_face, raw_mask):
  13         for _ in [0, 1]:
  14             diff = old_face - new_face
  15             avg_diff = np.sum(diff * raw_mask, axis=(0, 1))
  16             adjustment = avg_diff / np.sum(raw_mask, axis=(0, 1))
  17             new_face += adjustment
  18         return new_face
