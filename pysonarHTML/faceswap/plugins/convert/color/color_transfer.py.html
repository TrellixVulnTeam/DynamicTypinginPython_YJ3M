  31 class Color(Adjustment):
  32     """
  33     Transfers the color distribution from the source to the target
  34     image using the mean and standard deviations of the L*a*b*
  35     color space.
  36 
  37     This implementation is (loosely) based on to the "Color Transfer
  38     between Images" paper by Reinhard et al., 2001.
  39     """
  40 
  41     def process(self, old_face, new_face, raw_mask):
  42         """
  43         Parameters:
  44         -------
  45         source: NumPy array
  46             OpenCV image in BGR color space (the source image)
  47         target: NumPy array
  48             OpenCV image in BGR color space (the target image)
  49         clip: Should components of L*a*b* image be scaled by np.clip before
  50             converting back to BGR color space?
  51             If False then components will be min-max scaled appropriately.
  52             Clipping will keep target image brightness truer to the input.
  53             Scaling will adjust image brightness to avoid washed out portions
  54             in the resulting color transfer that can be caused by clipping.
  55         preserve_paper: Should color transfer strictly follow methodology
  56             layed out in original paper? The method does not always produce
  57             aesthetically pleasing results.
  58             If False then L*a*b* components will scaled using the reciprocal of
  59             the scaling factor proposed in the paper.  This method seems to produce
  60             more consistently aesthetically pleasing results
  61 
  62         Returns:
  63         -------
  64         transfer: NumPy array
  65             OpenCV image (w, h, 3) NumPy array (uint8)
  66         """
  67         clip = self.config.get("clip", True)
  68         preserve_paper = self.config.get("preserve_paper", True)
  69 
  70         # convert the images from the RGB to L*ab* color space, being
  71         # sure to utilizing the floating point data type (note: OpenCV
  72         # expects floats to be 32-bit, so use that instead of 64-bit)
  73         source = cv2.cvtColor(  # pylint: disable=no-member
  74             np.rint(old_face * raw_mask * 255.0).astype("uint8"),
  75             cv2.COLOR_BGR2LAB).astype("float32")  # pylint: disable=no-member
  76         target = cv2.cvtColor(  # pylint: disable=no-member
  77             np.rint(new_face * raw_mask * 255.0).astype("uint8"),
  78             cv2.COLOR_BGR2LAB).astype("float32")  # pylint: disable=no-member
  79         # compute color statistics for the source and target images
  80         (l_mean_src, l_std_src,
  81          a_mean_src, a_std_src,
  82          b_mean_src, b_std_src) = self.image_stats(source)
  83         (l_mean_tar, l_std_tar,
  84          a_mean_tar, a_std_tar,
  85          b_mean_tar, b_std_tar) = self.image_stats(target)
  86 
  87         # subtract the means from the target image
  88         (light, col_a, col_b) = cv2.split(target)  # pylint: disable=no-member
  89         light -= l_mean_tar
  90         col_a -= a_mean_tar
  91         col_b -= b_mean_tar
  92 
  93         if preserve_paper:
  94             # scale by the standard deviations using paper proposed factor
  95             light = (l_std_tar / l_std_src) * light
  96             col_a = (a_std_tar / a_std_src) * col_a
  97             col_b = (b_std_tar / b_std_src) * col_b
  98         else:
  99             # scale by the standard deviations using reciprocal of paper proposed factor
 100             light = (l_std_src / l_std_tar) * light
 101             col_a = (a_std_src / a_std_tar) * col_a
 102             col_b = (b_std_src / b_std_tar) * col_b
 103 
 104         # add in the source mean
 105         light += l_mean_src
 106         col_a += a_mean_src
 107         col_b += b_mean_src
 108 
 109         # clip/scale the pixel intensities to [0, 255] if they fall
 110         # outside this range
 111         light = self._scale_array(light, clip=clip)
 112         col_a = self._scale_array(col_a, clip=clip)
 113         col_b = self._scale_array(col_b, clip=clip)
 114 
 115         # merge the channels together and convert back to the RGB color
 116         # space, being sure to utilize the 8-bit unsigned integer data
 117         # type
 118         transfer = cv2.merge([light, col_a, col_b])  # pylint: disable=no-member
 119         transfer = cv2.cvtColor(  # pylint: disable=no-member
 120             transfer.astype("uint8"),
 121             cv2.COLOR_LAB2BGR).astype("float32") / 255.0  # pylint: disable=no-member
 122         background = new_face * (1 - raw_mask)
 123         merged = transfer + background
 124         # return the color transferred image
 125         return merged

 127     @staticmethod
 128     def image_stats(image):
 129         """
 130         Parameters:
 131         -------
 132         image: NumPy array
 133             OpenCV image in L*a*b* color space
 134 
 135         Returns:
 136         -------
 137         Tuple of mean and standard deviations for the L*, a*, and b*
 138         channels, respectively
 139         """
 140         # compute the mean and standard deviation of each channel
 141         (light, col_a, col_b) = cv2.split(image)  # pylint: disable=no-member
 142         (l_mean, l_std) = (light.mean(), light.std())
 143         (a_mean, a_std) = (col_a.mean(), col_a.std())
 144         (b_mean, b_std) = (col_b.mean(), col_b.std())
 145 
 146         # return the color statistics
 147         return (l_mean, l_std, a_mean, a_std, b_mean, b_std)
 148 
 149     @staticmethod
 150     def _min_max_scale(arr, new_range=(0, 255)):
 151         """
 152         Perform min-max scaling to a NumPy array
 153 
 154         Parameters:
 155         -------
 156         arr: NumPy array to be scaled to [new_min, new_max] range
 157         new_range: tuple of form (min, max) specifying range of
 158             transformed array
 159 
 160         Returns:
 161         -------
 162         NumPy array that has been scaled to be in
 163         [new_range[0], new_range[1]] range
 164         """
 165         # get array's current min and max
 166         arr_min = arr.min()
 167         arr_max = arr.max()
 168 
 169         # check if scaling needs to be done to be in new_range
 170         if arr_min < new_range[0] or arr_max > new_range[1]:
 171             # perform min-max scaling
 172             scaled = (new_range[1] - new_range[0]) * (arr - arr_min) / (arr_max -
 173                                                                         arr_min) + new_range[0]
 174         else:
 175             # return array if already in range
 176             scaled = arr
 177 
 178         return scaled
 179 
 180     def _scale_array(self, arr, clip=True):
 181         """
 182         Trim NumPy array values to be in [0, 255] range with option of
 183         clipping or scaling.
 184 
 185         Parameters:
 186         -------
 187         arr: array to be trimmed to [0, 255] range
 188         clip: should array be scaled by np.clip? if False then input
 189             array will be min-max scaled to range
 190             [max([arr.min(), 0]), min([arr.max(), 255])]
 191 
 192         Returns:
 193         -------
 194         NumPy array that has been scaled to be in [0, 255] range
 195         """
 196         if clip:
 197             scaled = np.clip(arr, 0, 255)
 198         else:
 199             scale_range = (max([arr.min(), 0]), min([arr.max(), 255]))
 200             scaled = self._min_max_scale(arr, new_range=scale_range)
 201 
 202         return scaled
