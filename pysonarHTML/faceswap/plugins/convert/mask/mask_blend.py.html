  10 class Mask(Adjustment):
  11     """ Manipulations to perform to the mask that is to be applied to the output of the Faceswap
  12     model.
  13 
  14     Parameters
  15     ----------
  16     mask_type: str
  17         The mask type to use for this plugin
  18     output_size: int
  19         The size of the output from the Faceswap model.
  20     coverage_ratio: float
  21         The coverage ratio that the Faceswap model was trained at.
  22     **kwargs: dict, optional
  23         See the parent :class:`~plugins.convert.mask._base` for additional keyword arguments.
  24     """
  25     def __init__(self, mask_type, output_size, coverage_ratio, **kwargs):
  26         super().__init__(mask_type, output_size, **kwargs)
  27         self._do_erode = self.config.get("erosion", 0) != 0
  28         self._coverage_ratio = coverage_ratio
  29 
  30     def process(self, detected_face, predicted_mask=None):  # pylint:disable=arguments-differ
  31         """ Obtain the requested mask type and perform any defined mask manipulations.
  32 
  33         Parameters
  34         ----------
  35         detected_face: :class:`lib.faces_detect.DetectedFace`
  36             The DetectedFace object as returned from :class:`scripts.convert.Predictor`.
  37         predicted_mask: :class:`numpy.ndarray`, optional
  38             The predicted mask as output from the Faceswap Model, if the model was trained
  39             with a mask, otherwise ``None``. Default: ``None``.
  40 
  41         Returns
  42         -------
  43         mask: :class:`numpy.ndarray`
  44             The mask with all requested manipulations applied
  45         raw_mask: :class:`numpy.ndarray`
  46             The mask with no erosion/dilation applied
  47         """
  48         mask = self._get_mask(detected_face, predicted_mask)
  49         raw_mask = mask.copy()
  50         if not self.skip and self._do_erode:
  51             mask = self._erode(mask)
  52         logger.trace("mask shape: %s, raw_mask shape: %s", mask.shape, raw_mask.shape)
  53         return mask, raw_mask
  54 
  55     def _get_mask(self, detected_face, predicted_mask):
  56         """ Return the requested mask with any requested blurring applied.
  57 
  58         Parameters
  59         ----------
  60         detected_face: :class:`lib.faces_detect.DetectedFace`
  61             The DetectedFace object as returned from :class:`scripts.convert.Predictor`.
  62         predicted_mask: :class:`numpy.ndarray`
  63             The predicted mask as output from the Faceswap Model if the model was trained
  64             with a mask, otherwise ``None``
  65 
  66         Returns
  67         -------
  68         :class:`numpy.ndarray`
  69             The mask sized to Faceswap model output with any requested blurring applied.
  70         """
  71         if self.mask_type == "none":
  72             # Return a dummy mask if not using a mask
  73             mask = np.ones_like(self.dummy[:, :, 1], dtype="float32")[..., None]
  74         elif self.mask_type == "predicted":
  75             mask = predicted_mask[..., None]
  76         else:
  77             mask = detected_face.mask[self.mask_type]
  78             mask.set_blur_and_threshold(blur_kernel=self.config["kernel_size"],
  79                                         blur_type=self.config["type"],
  80                                         blur_passes=self.config["passes"],
  81                                         threshold=self.config["threshold"])
  82             mask = self._crop_to_coverage(mask.mask)
  83             mask_size = mask.shape[0]
  84             face_size = self.dummy.shape[0]
  85             if mask_size != face_size:
  86                 interp = cv2.INTER_CUBIC if mask_size < face_size else cv2.INTER_AREA
  87                 mask = cv2.resize(mask,
  88                                   self.dummy.shape[:2],
  89                                   interpolation=interp)[..., None]
  90             mask = mask.astype("float32") / 255.0
  91         logger.trace(mask.shape)
  92         return mask
  93 
  94     def _crop_to_coverage(self, mask):
  95         """ Crop the mask to the correct dimensions based on coverage ratio.
  96 
  97         Parameters
  98         ----------
  99         mask: :class:`numpy.ndarray`
 100             The original mask to be cropped
 101 
 102         Returns
 103         -------
 104         :class:`numpy.ndarray`
 105             The cropped mask
 106         """
 107         if self._coverage_ratio == 1.0:
 108             return mask
 109         mask_size = mask.shape[0]
 110         padding = round((mask_size * (1 - self._coverage_ratio)) / 2)
 111         mask_slice = slice(padding, mask_size - padding)
 112         mask = mask[mask_slice, mask_slice, :]
 113         logger.trace("mask_size: %s, coverage: %s, padding: %s, final shape: %s",
 114                      mask_size, self._coverage_ratio, padding, mask.shape)
 115         return mask

 117     # MASK MANIPULATIONS
 118     def _erode(self, mask):
 119         """ Erode or dilate mask the mask based on configuration options.
 120 
 121         Parameters
 122         ----------
 123         mask: :class:`numpy.ndarray`
 124             The mask to be eroded or dilated
 125 
 126         Returns
 127         -------
 128         :class:`numpy.ndarray`
 129             The mask with erosion/dilation applied
 130         """
 131         kernel = self._get_erosion_kernel(mask)
 132         if self.config["erosion"] > 0:
 133             logger.trace("Eroding mask")
 134             mask = cv2.erode(mask, kernel, iterations=1)
 135         else:
 136             logger.trace("Dilating mask")
 137             mask = cv2.dilate(mask, kernel, iterations=1)
 138         return mask
 139 
 140     def _get_erosion_kernel(self, mask):
 141         """ Get the erosion kernel.
 142 
 143         Parameters
 144         ----------
 145         mask: :class:`numpy.ndarray`
 146             The mask to be eroded or dilated
 147 
 148         Returns
 149         -------
 150         :class:`numpy.ndarray`
 151             The erosion kernel to be used for erosion/dilation
 152         """
 153         erosion_ratio = self.config["erosion"] / 100
 154         mask_radius = np.sqrt(np.sum(mask)) / 2
 155         kernel_size = max(1, int(abs(erosion_ratio * mask_radius)))
 156         erosion_kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (kernel_size, kernel_size))
 157         logger.trace("erosion_kernel shape: %s", erosion_kernel.shape)
 158         return erosion_kernel
