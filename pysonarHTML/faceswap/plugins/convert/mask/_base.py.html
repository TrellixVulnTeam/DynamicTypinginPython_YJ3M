  13 def _get_config(plugin_name, configfile=None):
  14     """ Return the :attr:`lib.config.FaceswapConfig.config_dict` for the requested plugin.
  15 
  16     Parameters
  17     ----------

  19         The name of the plugin to retrieve the config for

  33 class Adjustment():
  34     """ Parent class for Mask Adjustment Plugins.
  35 
  36     All mask plugins must inherit from this class.
  37 
  38     Parameters
  39     ----------
  40     mask_type: str
  41         The type of mask that this plugin is being used for
  42     output_size: int
  43         The size, in pixels, of the output from the Faceswap model.
  44     configfile: str, Optional
  45         Optional location of custom configuration ``ini`` file. If ``None`` then use the default
  46         config location. Default: ``None``
  47     config: :class:`lib.config.FaceswapConfig`, Optional
  48         Optional pre-loaded :class:`lib.config.FaceswapConfig`. If passed, then this will be used
  49         over any configuration on disk. If ``None`` then it is ignored. Default: ``None``
  50 
  51 
  52     Attributes
  53     ----------
  54     config: dict
  55         The configuration dictionary for this plugin.
  56     mask_type: str
  57         The type of mask that this plugin is being used for.
  58     """
  59     def __init__(self, mask_type, output_size, configfile=None, config=None):
  60         logger.debug("Initializing %s: (arguments: '%s', output_size: %s, "
  61                      "configfile: %s, config: %s)", self.__class__.__name__, mask_type,
  62                      output_size, configfile, config)
  63         self.config = self._set_config(configfile, config)
  64         logger.debug("config: %s", self.config)
  65         self.mask_type = mask_type
  66         self._dummy = np.zeros((output_size, output_size, 3), dtype='float32')
  67         logger.debug("Initialized %s", self.__class__.__name__)
  68 
  69     @property
  70     def dummy(self):
  71         """:class:`numpy.ndarray`: A dummy mask of all zeros of the shape:
  72         (:attr:`output_size`, :attr:`output_size`, `3`)
  73         """
  74         return self._dummy
  75 
  76     @property
  77     def skip(self):
  78         """bool: ``True`` if the blur type config attribute is ``None`` otherwise ``False`` """
  79         return self.config.get("type", None) is None

  81     def _set_config(self, configfile, config):
  82         """ Set the correct configuration for the plugin based on whether a config file
  83         or pre-loaded config has been passed in.
  84 
  85         Parameters
  86         ----------
  87         configfile: str
  88             Location of custom configuration ``ini`` file. If ``None`` then use the
  89             default config location
  90         config: :class:`lib.config.FaceswapConfig`
  91             Pre-loaded :class:`lib.config.FaceswapConfig`. If passed, then this will be
  92             used over any configuration on disk. If ``None`` then it is ignored.
  93 
  94         Returns
  95         -------
  96         dict
  97             The configuration in dictionary form for the given from
  98             :attr:`lib.config.FaceswapConfig.config_dict`
  99         """
 100         section = ".".join(self.__module__.split(".")[-2:])
 101         if config is None:
 102             retval = _get_config(section, configfile=configfile)
 103         else:
 104             config.section = section
 105             retval = config.config_dict
 106             config.section = None
 107         logger.debug("Config: %s", retval)
 108         return retval

 119     def run(self, *args, **kwargs):
 120         """ Perform selected adjustment on face """
 121         logger.trace("Performing mask adjustment: (plugin: %s, args: %s, kwargs: %s",
 122                      self.__module__, args, kwargs)
 123         retval = self.process(*args, **kwargs)
 124         return retval
