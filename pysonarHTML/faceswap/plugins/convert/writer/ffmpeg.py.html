  14 class Writer(Output):
  15     """ Video output writer using imageio """
  16     def __init__(self, output_folder, total_count, frame_ranges, source_video, **kwargs):
  17         super().__init__(output_folder, **kwargs)
  18         logger.debug("total_count: %s, frame_ranges: %s, source_video: '%s'",
  19                      total_count, frame_ranges, source_video)
  20         self.source_video = source_video
  21         self.frame_ranges = frame_ranges
  22         self.frame_order = self.set_frame_order(total_count)
  23         self.output_dimensions = None  # Fix dims of 1st frame in case of different sized images
  24         self.writer = None  # Need to know dimensions of first frame, so set writer then
  25 
  26     @property
  27     def video_file(self):
  28         """ Return full path to video output """
  29         filename = os.path.basename(self.source_video)
  30         filename = os.path.splitext(filename)[0]
  31         filename = "{}_converted.{}".format(filename, self.config["container"])
  32         retval = os.path.join(self.output_folder, filename)
  33         logger.debug(retval)
  34         return retval
  35 
  36     @property
  37     def video_tmp_file(self):
  38         """ Temporary video file, prior to muxing final audio """
  39         path, filename = os.path.split(self.video_file)
  40         retval = os.path.join(path, "__tmp_{}".format(filename))
  41         logger.debug(retval)
  42         return retval
  43 
  44     @property
  45     def valid_tune(self):
  46         """ Return whether selected tune is valid for selected codec """
  47         return {"libx264": ["film", "animation", "grain", "stillimage", "fastdecode",
  48                             "zerolatency"],
  49                 "libx265": ["grain", "fastdecode", "zerolatency"]}
  50 
  51     @property
  52     def video_fps(self):
  53         """ Return the fps of source video """
  54         reader = imageio.get_reader(self.source_video, "ffmpeg")
  55         retval = reader.get_meta_data()["fps"]
  56         reader.close()
  57         logger.debug(retval)
  58         return retval
  59 
  60     @property
  61     def output_params(self):
  62         """ FFMPEG Output parameters """
  63         codec = self.config["codec"]
  64         tune = self.config["tune"]
  65         # Force all frames to the same size
  66         output_args = ["-vf", "scale={}".format(self.output_dimensions)]
  67 
  68         output_args.extend(["-c:v", codec])
  69         output_args.extend(["-crf", str(self.config["crf"])])
  70         output_args.extend(["-preset", self.config["preset"]])
  71 
  72         if tune is not None and tune in self.valid_tune[codec]:
  73             output_args.extend(["-tune", tune])
  74 
  75         if codec == "libx264" and self.config["profile"] != "auto":
  76             output_args.extend(["-profile:v", self.config["profile"]])
  77 
  78         if codec == "libx264" and self.config["level"] != "auto":
  79             output_args.extend(["-level", self.config["level"]])
  80 
  81         logger.debug(output_args)
  82         return output_args
  83 
  84     def set_frame_order(self, total_count):
  85         """ Return the full list of frames to be converted in order """
  86         if self.frame_ranges is None:
  87             retval = list(range(1, total_count + 1))
  88         else:
  89             retval = list()
  90             for rng in self.frame_ranges:
  91                 retval.extend(list(range(rng[0], rng[1] + 1)))
  92         logger.debug("frame_order: %s", retval)
  93         return retval
  94 
  95     def get_writer(self):
  96         """ Add the requested encoding options and return the writer """
  97         logger.debug("writer config: %s", self.config)
  98         return imageio.get_writer(self.video_tmp_file,
  99                                   fps=self.video_fps,
 100                                   ffmpeg_log_level="error",
 101                                   quality=None,
 102                                   macro_block_size=8,
 103                                   output_params=self.output_params)
 104 
 105     def write(self, filename, image):
 106         """ Frames come from the pool in arbitrary order, so cache frames
 107             for writing out in correct order """
 108         logger.trace("Received frame: (filename: '%s', shape: %s", filename, image.shape)
 109         if not self.output_dimensions:
 110             logger.info("Outputting to: '%s'", self.video_file)
 111             self.set_dimensions(image.shape[:2])
 112             self.writer = self.get_writer()
 113         self.cache_frame(filename, image)
 114         self.save_from_cache()
 115 
 116     def set_dimensions(self, frame_dims):
 117         """ Set the dimensions based on a given frame frame. This protects against different
 118             sized images coming in and ensure all images go out at the same size for writers
 119             that require it and mapped to a macro block size 16"""
 120         logger.debug("input dimensions: %s", frame_dims)
 121         self.output_dimensions = "{}:{}".format(
 122             int(ceil(frame_dims[1] / 16) * 16),
 123             int(ceil(frame_dims[0] / 16) * 16))
 124         logger.debug("Set dimensions: %s", self.output_dimensions)
 125 
 126     def save_from_cache(self):
 127         """ Save all the frames that are ready to be output from cache """
 128         while self.frame_order:
 129             if self.frame_order[0] not in self.cache:
 130                 logger.trace("Next frame not ready. Continuing")
 131                 break
 132             save_no = self.frame_order.pop(0)
 133             save_image = self.cache.pop(save_no)
 134             logger.trace("Rendering from cache. Frame no: %s", save_no)
 135             self.writer.append_data(save_image[:, :, ::-1])
 136         logger.trace("Current cache size: %s", len(self.cache))
 137 
 138     def close(self):
 139         """ Close the ffmpeg writer and mux the audio """
 140         self.writer.close()
 141         self.mux_audio()
 142 
 143     def mux_audio(self):
 144         """ Mux audio
 145             ImageIO is a useful lib for frames > video as it also packages the ffmpeg binary
 146             however muxing audio is non-trivial, so this is done afterwards with ffmpy.
 147             A future fix could be implemented to mux audio with the frames """
 148         logger.info("Muxing Audio...")
 149         if self.frame_ranges is not None:
 150             logger.warning("Muxing audio is not currently supported for limited frame ranges."
 151                            "The output video has been created but you will need to mux audio "
 152                            "yourself")
 153             os.rename(self.video_tmp_file, self.video_file)
 154             logger.debug("Removing temp file")
 155             if os.path.isfile(self.video_tmp_file):
 156                 os.remove(self.video_tmp_file)
 157             return
 158 
 159         exe = im_ffm.get_ffmpeg_exe()
 160         inputs = OrderedDict([(self.video_tmp_file, None), (self.source_video, None)])
 161         outputs = {self.video_file: "-map 0:v:0 -map 1:a:0 -c: copy"}
 162         ffm = FFmpeg(executable=exe,
 163                      global_options="-hide_banner -nostats -v 0 -y",
 164                      inputs=inputs,
 165                      outputs=outputs)
 166         logger.debug("Executing: %s", ffm.cmd)
 167         # Sometimes ffmpy exits for no discernible reason, but then works on a later attempt,
 168         # so take 5 shots at this
 169         attempts = 5
 170         for attempt in range(attempts):
 171             logger.debug("Muxing attempt: %s", attempt + 1)
 172             try:
 173                 ffm.run()
 174             except FFRuntimeError as err:
 175                 logger.debug("ffmpy runtime error: %s", str(err))
 176                 if attempt != attempts - 1:
 177                     continue
 178                 logger.error("There was a problem muxing audio. The output video has been "
 179                              "created but you will need to mux audio yourself either with the "
 180                              "EFFMpeg tool or an external application.")
 181                 os.rename(self.video_tmp_file, self.video_file)
 182             break
 183         logger.debug("Removing temp file")
 184         if os.path.isfile(self.video_tmp_file):
 185             os.remove(self.video_tmp_file)
