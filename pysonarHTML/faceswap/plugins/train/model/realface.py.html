  18 class Model(ModelBase):
  19     """ RealFace(tm) Faceswap Model """
  20     def __init__(self, *args, **kwargs):
  21         logger.debug("Initializing %s: (args: %s, kwargs: %s",
  22                      self.__class__.__name__, args, kwargs)
  23 
  24         self.configfile = kwargs.get("configfile", None)
  25         self.check_input_output()
  26         self.dense_width, self.upscalers_no = self.get_dense_width_upscalers_numbers()
  27         kwargs["input_shape"] = (self.config["input_size"], self.config["input_size"], 3)
  28         self.kernel_initializer = RandomNormal(0, 0.02)
  29 
  30         super().__init__(*args, **kwargs)
  31         logger.debug("Initialized %s", self.__class__.__name__)

  33     @property
  34     def downscalers_no(self):
  35         """ Number of downscalers. Don't change! """
  36         return 4
  37 
  38     @property
  39     def _downscale_ratio(self):
  40         """ Downscale Ratio """
  41         return 2**self.downscalers_no
  42 
  43     @property
  44     def dense_filters(self):
  45         """ Dense Filters. Don't change! """
  46         return (int(1024 - (self.dense_width - 4) * 64) // 16) * 16
  47 
  48     def check_input_output(self):
  49         """ Confirm valid input and output sized have been provided """
  50         if not 64 <= self.config["input_size"] <= 128 or self.config["input_size"] % 16 != 0:
  51             logger.error("Config error: input_size must be between 64 and 128 and be divisible by "
  52                          "16.")
  53             exit(1)
  54         if not 64 <= self.config["output_size"] <= 256 or self.config["output_size"] % 32 != 0:
  55             logger.error("Config error: output_size must be between 64 and 256 and be divisible "
  56                          "by 32.")
  57             exit(1)
  58         logger.debug("Input and output sizes are valid")
  59 
  60     def get_dense_width_upscalers_numbers(self):
  61         """ Return the dense width and number of upscalers """
  62         output_size = self.config["output_size"]
  63         sides = [(output_size // 2**n, n) for n in [4, 5] if (output_size // 2**n) < 10]
  64         closest = min([x * self._downscale_ratio for x, _ in sides],
  65                       key=lambda x: abs(x - self.config["input_size"]))
  66         dense_width, upscalers_no = [(s, n) for s, n in sides
  67                                      if s * self._downscale_ratio == closest][0]
  68         logger.debug("dense_width: %s, upscalers_no: %s", dense_width, upscalers_no)
  69         return dense_width, upscalers_no

  71     def add_networks(self):
  72         """ Add the realface model weights """
  73         logger.debug("Adding networks")
  74         self.add_network("decoder", "a", self.decoder_a(), is_output=True)
  75         self.add_network("decoder", "b", self.decoder_b(), is_output=True)
  76         self.add_network("encoder", None, self.encoder())
  77         logger.debug("Added networks")
  78 
  79     def build_autoencoders(self, inputs):
  80         """ Initialize realface model """
  81         logger.debug("Initializing model")
  82         for side in "a", "b":
  83             logger.debug("Adding Autoencoder. Side: %s", side)
  84             decoder = self.networks["decoder_{}".format(side)].network
  85             output = decoder(self.networks["encoder"].network(inputs[0]))
  86             autoencoder = KerasModel(inputs, output)
  87             self.add_predictor(side, autoencoder)
  88         logger.debug("Initialized model")

  90     def encoder(self):
  91         """ RealFace Encoder Network """
  92         input_ = Input(shape=self.input_shape)
  93         var_x = input_
  94 
  95         encoder_complexity = self.config["complexity_encoder"]
  96 
  97         for idx in range(self.downscalers_no - 1):
  98             var_x = self.blocks.conv(var_x, encoder_complexity * 2**idx)
  99             var_x = self.blocks.res_block(var_x, encoder_complexity * 2**idx, use_bias=True)
 100             var_x = self.blocks.res_block(var_x, encoder_complexity * 2**idx, use_bias=True)
 101 
 102         var_x = self.blocks.conv(var_x, encoder_complexity * 2**(idx + 1))
 103 
 104         return KerasModel(input_, var_x)

 106     def decoder_b(self):
 107         """ RealFace Decoder Network """
 108         input_filters = self.config["complexity_encoder"] * 2**(self.downscalers_no-1)
 109         input_width = self.config["input_size"] // self._downscale_ratio
 110         input_ = Input(shape=(input_width, input_width, input_filters))
 111 
 112         var_xy = input_
 113 
 114         var_xy = Dense(self.config["dense_nodes"])(Flatten()(var_xy))
 115         var_xy = Dense(self.dense_width * self.dense_width * self.dense_filters)(var_xy)
 116         var_xy = Reshape((self.dense_width, self.dense_width, self.dense_filters))(var_xy)
 117         var_xy = self.blocks.upscale(var_xy, self.dense_filters)
 118 
 119         var_x = var_xy
 120         var_x = self.blocks.res_block(var_x, self.dense_filters, use_bias=False)
 121 
 122         decoder_b_complexity = self.config["complexity_decoder"]
 123         for idx in range(self.upscalers_no - 2):
 124             var_x = self.blocks.upscale(var_x, decoder_b_complexity // 2**idx)
 125             var_x = self.blocks.res_block(var_x, decoder_b_complexity // 2**idx, use_bias=False)
 126             var_x = self.blocks.res_block(var_x, decoder_b_complexity // 2**idx, use_bias=True)
 127         var_x = self.blocks.upscale(var_x, decoder_b_complexity // 2**(idx + 1))
 128 
 129         var_x = self.blocks.conv2d(var_x, 3,
 130                                    kernel_size=5,
 131                                    padding="same",
 132                                    activation="sigmoid",
 133                                    name="face_out")
 134 
 135         outputs = [var_x]
 136 
 137         if self.config.get("learn_mask", False):
 138             var_y = var_xy
 139             mask_b_complexity = 384
 140             for idx in range(self.upscalers_no-2):
 141                 var_y = self.blocks.upscale(var_y, mask_b_complexity // 2**idx)
 142             var_y = self.blocks.upscale(var_y, mask_b_complexity // 2**(idx + 1))
 143 
 144             var_y = self.blocks.conv2d(var_y, 1,
 145                                        kernel_size=5,
 146                                        padding="same",
 147                                        activation="sigmoid",
 148                                        name="mask_out")
 149 
 150             outputs += [var_y]
 151 
 152         return KerasModel(input_, outputs=outputs)

 154     def decoder_a(self):
 155         """ RealFace Decoder (A) Network """
 156         input_filters = self.config["complexity_encoder"] * 2**(self.downscalers_no-1)
 157         input_width = self.config["input_size"] // self._downscale_ratio
 158         input_ = Input(shape=(input_width, input_width, input_filters))
 159 
 160         var_xy = input_
 161 
 162         dense_nodes = int(self.config["dense_nodes"]/1.5)
 163         dense_filters = int(self.dense_filters/1.5)
 164 
 165         var_xy = Dense(dense_nodes)(Flatten()(var_xy))
 166         var_xy = Dense(self.dense_width * self.dense_width * dense_filters)(var_xy)
 167         var_xy = Reshape((self.dense_width, self.dense_width, dense_filters))(var_xy)
 168 
 169         var_xy = self.blocks.upscale(var_xy, dense_filters)
 170 
 171         var_x = var_xy
 172         var_x = self.blocks.res_block(var_x, dense_filters, use_bias=False)
 173 
 174         decoder_a_complexity = int(self.config["complexity_decoder"] / 1.5)
 175         for idx in range(self.upscalers_no-2):
 176             var_x = self.blocks.upscale(var_x, decoder_a_complexity // 2**idx)
 177         var_x = self.blocks.upscale(var_x, decoder_a_complexity // 2**(idx + 1))
 178 
 179         var_x = self.blocks.conv2d(var_x, 3,
 180                                    kernel_size=5,
 181                                    padding="same",
 182                                    activation="sigmoid",
 183                                    name="face_out")
 184 
 185         outputs = [var_x]
 186 
 187         if self.config.get("learn_mask", False):
 188             var_y = var_xy
 189             mask_a_complexity = 384
 190             for idx in range(self.upscalers_no-2):
 191                 var_y = self.blocks.upscale(var_y, mask_a_complexity // 2**idx)
 192             var_y = self.blocks.upscale(var_y, mask_a_complexity // 2**(idx + 1))
 193 
 194             var_y = self.blocks.conv2d(var_y, 1,
 195                                        kernel_size=5,
 196                                        padding="same",
 197                                        activation="sigmoid",
 198                                        name="mask_out")
 199 
 200             outputs += [var_y]
