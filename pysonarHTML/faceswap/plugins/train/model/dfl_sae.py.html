  14 class Model(ModelBase):
  15     """ Low Memory version of Original Faceswap Model """
  16     def __init__(self, *args, **kwargs):
  17         logger.debug("Initializing %s: (args: %s, kwargs: %s",
  18                      self.__class__.__name__, args, kwargs)
  19 
  20         self.configfile = kwargs.get("configfile", None)
  21         kwargs["input_shape"] = (self.config["input_size"], self.config["input_size"], 3)
  22 
  23         super().__init__(*args, **kwargs)
  24         logger.debug("Initialized %s", self.__class__.__name__)
  25 
  26     @property
  27     def architecture(self):
  28         """ Return the architecture used from config """
  29         return self.config["architecture"].lower()
  30 
  31     @property
  32     def use_mask(self):
  33         """ Return True if a mask has been set else false """
  34         return self.config.get("learn_mask", False)
  35 
  36     @property
  37     def ae_dims(self):
  38         """ Set the Autoencoder Dimensions or set to default """
  39         retval = self.config["autoencoder_dims"]
  40         if retval == 0:
  41             retval = 256 if self.architecture == "liae" else 512
  42         return retval
  43 
  44     @property
  45     def multiscale_count(self):
  46         """ Return 3 if multiscale decoder is set else 1 """
  47         retval = 3 if self.config["multiscale_decoder"] else 1
  48         return retval
  49 
  50     def add_networks(self):
  51         """ Add the DFL SAE Networks """
  52         logger.debug("Adding networks")
  53         # Encoder
  54         self.add_network("encoder", None, getattr(self, "encoder_{}".format(self.architecture))())
  55 
  56         # Intermediate
  57         if self.architecture == "liae":
  58             self.add_network("intermediate", "b", self.inter_liae())
  59             self.add_network("intermediate", None, self.inter_liae())
  60 
  61         # Decoder
  62         decoder_sides = [None] if self.architecture == "liae" else ["a", "b"]
  63         for side in decoder_sides:
  64             self.add_network("decoder", side, self.decoder(), is_output=True)
  65         logger.debug("Added networks")
  66 
  67     def build_autoencoders(self, inputs):
  68         """ Initialize DFL SAE model """
  69         logger.debug("Initializing model")
  70         getattr(self, "build_{}_autoencoder".format(self.architecture))(inputs)
  71         logger.debug("Initialized model")
  72 
  73     def build_liae_autoencoder(self, inputs):
  74         """ Build the LIAE Autoencoder """
  75         for side in ("a", "b"):
  76             encoder = self.networks["encoder"].network(inputs[0])
  77             if side == "a":
  78                 intermediate = Concatenate()([self.networks["intermediate"].network(encoder),
  79                                               self.networks["intermediate"].network(encoder)])
  80             else:
  81                 intermediate = Concatenate()([self.networks["intermediate_b"].network(encoder),
  82                                               self.networks["intermediate"].network(encoder)])
  83             output = self.networks["decoder"].network(intermediate)
  84             autoencoder = KerasModel(inputs, output)
  85             self.add_predictor(side, autoencoder)
  86 
  87     def build_df_autoencoder(self, inputs):
  88         """ Build the DF Autoencoder """
  89         for side in ("a", "b"):
  90             logger.debug("Adding Autoencoder. Side: %s", side)
  91             decoder = self.networks["decoder_{}".format(side)].network
  92             output = decoder(self.networks["encoder"].network(inputs[0]))
  93             autoencoder = KerasModel(inputs, output)
  94             self.add_predictor(side, autoencoder)
  95 
  96     def encoder_df(self):
  97         """ DFL SAE DF Encoder Network"""
  98         input_ = Input(shape=self.input_shape)
  99         dims = self.input_shape[-1] * self.config["encoder_dims"]
 100         lowest_dense_res = self.input_shape[0] // 16
 101         var_x = input_
 102         var_x = self.blocks.conv(var_x, dims)
 103         var_x = self.blocks.conv(var_x, dims * 2)
 104         var_x = self.blocks.conv(var_x, dims * 4)
 105         var_x = self.blocks.conv(var_x, dims * 8)
 106         var_x = Dense(self.ae_dims)(Flatten()(var_x))
 107         var_x = Dense(lowest_dense_res * lowest_dense_res * self.ae_dims)(var_x)
 108         var_x = Reshape((lowest_dense_res, lowest_dense_res, self.ae_dims))(var_x)
 109         var_x = self.blocks.upscale(var_x, self.ae_dims)
 110         return KerasModel(input_, var_x)
 111 
 112     def encoder_liae(self):
 113         """ DFL SAE LIAE Encoder Network """
 114         input_ = Input(shape=self.input_shape)
 115         dims = self.input_shape[-1] * self.config["encoder_dims"]
 116         var_x = input_
 117         var_x = self.blocks.conv(var_x, dims)
 118         var_x = self.blocks.conv(var_x, dims * 2)
 119         var_x = self.blocks.conv(var_x, dims * 4)
 120         var_x = self.blocks.conv(var_x, dims * 8)
 121         var_x = Flatten()(var_x)
 122         return KerasModel(input_, var_x)
 123 
 124     def inter_liae(self):
 125         """ DFL SAE LIAE Intermediate Network """
 126         input_ = Input(shape=self.networks["encoder"].output_shapes[0][1:])
 127         lowest_dense_res = self.input_shape[0] // 16
 128         var_x = input_
 129         var_x = Dense(self.ae_dims)(var_x)
 130         var_x = Dense(lowest_dense_res * lowest_dense_res * self.ae_dims * 2)(var_x)
 131         var_x = Reshape((lowest_dense_res, lowest_dense_res, self.ae_dims * 2))(var_x)
 132         var_x = self.blocks.upscale(var_x, self.ae_dims * 2)
 133         return KerasModel(input_, var_x)
 134 
 135     def decoder(self):
 136         """ DFL SAE Decoder Network"""
 137         if self.architecture == "liae":
 138             input_shape = np.array(self.networks["intermediate"].output_shapes[0][1:]) * (1, 1, 2)
 139         else:
 140             input_shape = self.networks["encoder"].output_shapes[0][1:]
 141         input_ = Input(shape=input_shape)
 142         outputs = list()
 143 
 144         dims = self.input_shape[-1] * self.config["decoder_dims"]
 145         var_x = input_
 146 
 147         var_x1 = self.blocks.upscale(var_x, dims * 8, res_block_follows=True)
 148         var_x1 = self.blocks.res_block(var_x1, dims * 8)
 149         var_x1 = self.blocks.res_block(var_x1, dims * 8)
 150         if self.multiscale_count >= 3:
 151             outputs.append(self.blocks.conv2d(var_x1, 3,
 152                                               kernel_size=5,
 153                                               padding="same",
 154                                               activation="sigmoid",
 155                                               name="face_out_32"))
 156 
 157         var_x2 = self.blocks.upscale(var_x1, dims * 4, res_block_follows=True)
 158         var_x2 = self.blocks.res_block(var_x2, dims * 4)
 159         var_x2 = self.blocks.res_block(var_x2, dims * 4)
 160         if self.multiscale_count >= 2:
 161             outputs.append(self.blocks.conv2d(var_x2, 3,
 162                                               kernel_size=5,
 163                                               padding="same",
 164                                               activation="sigmoid",
 165                                               name="face_out_64"))
 166 
 167         var_x3 = self.blocks.upscale(var_x2, dims * 2, res_block_follows=True)
 168         var_x3 = self.blocks.res_block(var_x3, dims * 2)
 169         var_x3 = self.blocks.res_block(var_x3, dims * 2)
 170 
 171         outputs.append(self.blocks.conv2d(var_x3, 3,
 172                                           kernel_size=5,
 173                                           padding="same",
 174                                           activation="sigmoid",
 175                                           name="face_out_128"))
 176 
 177         if self.use_mask:
 178             var_y = input_
 179             var_y = self.blocks.upscale(var_y, self.config["decoder_dims"] * 8)
 180             var_y = self.blocks.upscale(var_y, self.config["decoder_dims"] * 4)
 181             var_y = self.blocks.upscale(var_y, self.config["decoder_dims"] * 2)
 182             var_y = self.blocks.conv2d(var_y, 1,
 183                                        kernel_size=5,
 184                                        padding="same",
 185                                        activation="sigmoid",
 186                                        name="mask_out")
 187             outputs.append(var_y)
 188         return KerasModel(input_, outputs=outputs)
