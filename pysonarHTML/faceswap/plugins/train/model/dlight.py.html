  30 def upscale2x_hyb(self, inp, filters, kernel_size=3, padding='same',
  31                   sr_ratio=0.5, scale_factor=2, interpolation='bilinear',
  32                   res_block_follows=False, **kwargs):
  33     """Hybrid Upscale Layer"""
  34     name = self._get_name("upscale2x_hyb")
  35     var_x = inp
  36 
  37     sr_filters = int(filters * sr_ratio)
  38     upscale_filters = filters - sr_filters
  39 
  40     var_x_sr = self.upscale(var_x, upscale_filters, kernel_size=kernel_size,
  41                             padding=padding, scale_factor=scale_factor,
  42                             res_block_follows=res_block_follows, **kwargs)
  43     if upscale_filters > 0:
  44         var_x_us = self.conv2d(var_x, upscale_filters,  kernel_size=3, padding=padding,
  45                                name="{}_conv2d".format(name), **kwargs)
  46         var_x_us = UpSampling2D(size=(scale_factor, scale_factor), interpolation=interpolation,
  47                                 name="{}_upsampling2D".format(name))(var_x_us)
  48         var_x = Concatenate(name="{}_concatenate".format(name))([var_x_sr, var_x_us])
  49     else:
  50         var_x = var_x_sr
  51 
  52     return var_x

  55 def upscale2x_fast(self, inp, filters, kernel_size=3, padding='same',
  56                    sr_ratio=0.5, scale_factor=2, interpolation='bilinear',
  57                    res_block_follows=False, **kwargs):
  58     """Fast Upscale Layer"""
  59     name = self._get_name("upscale2x_fast")
  60     var_x = inp
  61 
  62     var_x2 = self.conv2d(var_x, filters,  kernel_size=3, padding=padding,
  63                          name="{}_conv2d".format(name), **kwargs)
  64     var_x2 = UpSampling2D(size=(scale_factor, scale_factor), interpolation=interpolation,
  65                           name="{}_upsampling2D".format(name))(var_x2)
  66 
  67     var_x1 = self.upscale(var_x, filters, kernel_size=kernel_size,
  68                           padding=padding, scale_factor=scale_factor,
  69                           res_block_follows=res_block_follows, **kwargs)
  70     var_x = Add()([var_x2, var_x1])
  71     return var_x

  89     def _detail_level_setup(self):
  90         logger.debug('self.config[output_size]: %d', self.config["output_size"])
  91 
  92         self.features = {
  93             'lowmem': 0,
  94             'fair':  1,
  95             'best':  2,
  96             }[self.config["features"]]
  97         logger.debug('self.features: %d', self.features)
  98 
  99         self.encoder_filters = 64 if self.features > 0 else 48
 100         logger.debug('self.encoder_filters: %d', self.encoder_filters)
 101         bonum_fortunam = 128
 102         self.encoder_dim = {
 103             0: 512 + bonum_fortunam,
 104             1: 1024 + bonum_fortunam,
 105             2: 1536 + bonum_fortunam,
 106             }[self.features]
 107         logger.debug('self.encoder_dim: %d', self.encoder_dim)
 108 
 109         self.details = {
 110             'fast': 0,
 111             'good':  1,
 112             }[self.config["details"]]
 113         logger.debug('self.details: %d', self.details)
 114 
 115         try:
 116             self.upscale_ratio = {
 117                 128: 2,
 118                 256: 4,
 119                 384: 6
 120                 }[self.config["output_size"]]
 121         except KeyError:
 122             logger.error("Config error: output_size must be one of: 128, 256, or 384.")
 123             raise FaceswapError("Config error: output_size must be one of: 128, 256, or 384.")
 124         logger.debug('output_size: %r', self.config["output_size"])
 125         logger.debug('self.upscale_ratio: %r', self.upscale_ratio)

 165     def encoder(self):
 166         """ DeLight Encoder Network """
 167         input_ = Input(shape=self.input_shape)
 168         var_x = input_
 169 
 170         var_x1 = self.blocks.conv(var_x, self.encoder_filters // 2)
 171         var_x2 = AveragePooling2D()(var_x)
 172         var_x2 = LeakyReLU(0.1)(var_x2)
 173         var_x = Concatenate()([var_x1, var_x2])
 174 
 175         var_x1 = self.blocks.conv(var_x, self.encoder_filters)
 176         var_x2 = AveragePooling2D()(var_x)
 177         var_x2 = LeakyReLU(0.1)(var_x2)
 178         var_x = Concatenate()([var_x1, var_x2])
 179 
 180         var_x1 = self.blocks.conv(var_x, self.encoder_filters * 2)
 181         var_x2 = AveragePooling2D()(var_x)
 182         var_x2 = LeakyReLU(0.1)(var_x2)
 183         var_x = Concatenate()([var_x1, var_x2])
 184 
 185         var_x1 = self.blocks.conv(var_x, self.encoder_filters * 4)
 186         var_x2 = AveragePooling2D()(var_x)
 187         var_x2 = LeakyReLU(0.1)(var_x2)
 188         var_x = Concatenate()([var_x1, var_x2])
 189 
 190         var_x1 = self.blocks.conv(var_x, self.encoder_filters * 8)
 191         var_x2 = AveragePooling2D()(var_x)
 192         var_x2 = LeakyReLU(0.1)(var_x2)
 193         var_x = Concatenate()([var_x1, var_x2])
 194 
 195         var_x = Dense(self.encoder_dim)(Flatten()(var_x))
 196         var_x = Dropout(0.05)(var_x)
 197         var_x = Dense(4 * 4 * 1024)(var_x)
 198         var_x = Dropout(0.05)(var_x)
 199         var_x = Reshape((4, 4, 1024))(var_x)
 200 
 201         return KerasModel(input_, var_x)

 203     def decoder_a(self):
 204         """ DeLight Decoder A(old face) Network """
 205         input_ = Input(shape=(4, 4, 1024))
 206         decoder_a_complexity = 256
 207         mask_complexity = 128
 208 
 209         var_xy = input_
 210         var_xy = UpSampling2D(self.upscale_ratio, interpolation='bilinear')(var_xy)
 211 
 212         var_x = var_xy
 213         var_x = self.blocks.upscale2x_hyb(var_x, decoder_a_complexity)
 214         var_x = self.blocks.upscale2x_hyb(var_x, decoder_a_complexity // 2)
 215         var_x = self.blocks.upscale2x_hyb(var_x, decoder_a_complexity // 4)
 216         var_x = self.blocks.upscale2x_hyb(var_x, decoder_a_complexity // 8)
 217 
 218         var_x = self.blocks.conv2d(var_x, 3, kernel_size=5, padding="same",
 219                                    activation="sigmoid", name="face_out")
 220 
 221         outputs = [var_x]
 222 
 223         if self.config.get("learn_mask", False):
 224             var_y = var_xy  # mask decoder
 225             var_y = self.blocks.upscale2x_hyb(var_y, mask_complexity)
 226             var_y = self.blocks.upscale2x_hyb(var_y, mask_complexity // 2)
 227             var_y = self.blocks.upscale2x_hyb(var_y, mask_complexity // 4)
 228             var_y = self.blocks.upscale2x_hyb(var_y, mask_complexity // 8)
 229 
 230             var_y = self.blocks.conv2d(var_y, 1, kernel_size=5, padding="same",
 231                                        activation="sigmoid", name="mask_out")
 232 
 233             outputs.append(var_y)
 234 
 235         return KerasModel([input_], outputs=outputs)
