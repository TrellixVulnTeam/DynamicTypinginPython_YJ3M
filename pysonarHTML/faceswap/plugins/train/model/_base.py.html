 188     @property
 189     def largest_face_index(self):
 190         """ Return the index from model.outputs of the largest face
 191             Required for multi-output model prediction. The largest face
 192             is assumed to be the final output
 193         """
 194         sizes = [shape[1] for shape in self.output_shapes if shape[2] == 3]
 195         if not sizes:
 196             return None
 197         max_face = max(sizes)
 198         retval = [idx for idx, shape in enumerate(self.output_shapes)
 199                   if shape[1] == max_face and shape[2] == 3][0]
 200         logger.debug(retval)
 201         return retval

 232     def calculate_coverage_ratio(self):
 233         """ Coverage must be a ratio, leading to a cropped shape divisible by 2 """
 234         coverage_ratio = self.config.get("coverage", 62.5) / 100
 235         logger.debug("Requested coverage_ratio: %s", coverage_ratio)
 236         cropped_size = (self.state.training_size * coverage_ratio) // 2 * 2
 237         coverage_ratio = cropped_size / self.state.training_size
 238         logger.debug("Final coverage_ratio: %s", coverage_ratio)
 239         return coverage_ratio

 241     def build(self):
 242         """ Build the model. Override for custom build methods """
 243         self.add_networks()
 244         self.load_models(swapped=False)
 245         inputs = self.get_inputs()
 246         try:
 247             self.build_autoencoders(inputs)
 248         except ValueError as err:
 249             if "must be from the same graph" in str(err).lower():
 250                 msg = ("There was an error loading saved weights. This is most likely due to "
 251                        "model corruption during a previous save."
 252                        "\nYou should restore weights from a snapshot or from backup files. "
 253                        "You can use the 'Restore' Tool to restore from backup.")
 254                 raise FaceswapError(msg) from err
 255             if "multi_gpu_model" in str(err).lower():
 256                 raise FaceswapError(str(err)) from err
 257             raise err
 258         self.log_summary()
 259         self.compile_predictors(initialize=True)

 261     def get_inputs(self):
 262         """ Return the inputs for the model """
 263         logger.debug("Getting inputs")
 264         inputs = [Input(shape=self.input_shape, name="face_in")]
 265         output_network = [network for network in self.networks.values() if network.is_output][0]
 266         if self.feed_mask:
 267             # TODO penalized mask doesn't have a mask output, so we can't use output shapes
 268             # mask should always be last output..this needs to be a rule
 269             mask_shape = output_network.output_shapes[-1]
 270             inputs.append(Input(shape=(mask_shape[1:-1] + (1,)), name="mask_in"))
 271         logger.debug("Got inputs: %s", inputs)
 272         return inputs

 320     def add_predictor(self, side, model):

 330     def store_input_shapes(self, model):
 331         """ Store the input and output shapes to state """
 332         logger.debug("Adding input shapes to state for model")
 333         inputs = {tensor.name: K.int_shape(tensor)[-3:] for tensor in model.inputs}
 334         if not any(inp for inp in inputs.keys() if inp.startswith("face")):
 335             raise ValueError("No input named 'face' was found. Check your input naming. "
 336                              "Current input names: {}".format(inputs))
 337         # Make sure they are all ints so that it can be json serialized
 338         inputs = {key: tuple(int(i) for i in val) for key, val in inputs.items()}
 339         self.state.inputs = inputs
 340         logger.debug("Added input shapes: %s", self.state.inputs)

 342     def reset_pingpong(self):

 344         logger.debug("Resetting models")
 345

 347         self.predictors = dict()

 360     def compile_predictors(self, initialize=True):
 361         """ Compile the predictors """
 362         logger.debug("Compiling Predictors")
 363         learning_rate = self.config.get("learning_rate", 5e-5)
 364         optimizer = self.get_optimizer(lr=learning_rate, beta_1=0.5, beta_2=0.999)
 365 
 366         for side, model in self.predictors.items():
 367             loss = Loss(model.inputs, model.outputs)
 368             model.compile(optimizer=optimizer, loss=loss.funcs)
 369             if initialize:
 370                 self.state.add_session_loss_names(side, loss.names)
 371                 self.history[side] = list()
 372         logger.debug("Compiled Predictors. Losses: %s", loss.names)

 374     def get_optimizer(self, lr=5e-5, beta_1=0.5, beta_2=0.999):  # pylint: disable=invalid-name
 375         """ Build and return Optimizer """
 376         opt_kwargs = dict(lr=lr, beta_1=beta_1, beta_2=beta_2)
 377         if (self.config.get("clipnorm", False) and
 378                 keras.backend.backend() != "plaidml.keras.backend"):
 379             # NB: Clip-norm is ballooning VRAM usage, which is not expected behavior
 380             # and may be a bug in Keras/Tensorflow.
 381             # PlaidML has a bug regarding the clip-norm parameter
 382             # See: https://github.com/plaidml/plaidml/issues/228
 383             # Workaround by simply removing it.
 384             # TODO: Remove this as soon it is fixed in PlaidML.
 385             opt_kwargs["clipnorm"] = 1.0
 386         logger.debug("Optimizer kwargs: %s", opt_kwargs)
 387         return Adam(**opt_kwargs, cpu_mode=self.vram_savings.optimizer_savings)

 438     def load_models(self, swapped):
 439         """ Load models from file """
 440         logger.debug("Load model: (swapped: %s)", swapped)
 441 
 442         if not self.models_exist and not self.predict:
 443             logger.info("Creating new '%s' model in folder: '%s'", self.name, self.model_dir)
 444             return None
 445         if not self.models_exist and self.predict:
 446             logger.error("Model could not be found in folder '%s'. Exiting", self.model_dir)
 447             exit(0)
 448 
 449         if not self.is_legacy or not self.predict:
 450             K.clear_session()
 451         model_mapping = self.map_models(swapped)
 452         for network in self.networks.values():
 453             if not network.side:
 454                 is_loaded = network.load()
 455             else:
 456                 is_loaded = network.load(fullpath=model_mapping[network.side][network.type])
 457             if not is_loaded:
 458                 break
 459         if is_loaded:
 460             logger.info("Loaded model from disk: '%s'", self.model_dir)
 461         return is_loaded

 463     def save_models(self):
 464         """ Backup and save the models """
 465         logger.debug("Backing up and saving models")
 466         # Insert a new line to avoid spamming the same row as loss output
 467         print("")
 468         save_averages = self.get_save_averages()
 469         backup_func = self.backup.backup_model if self.should_backup(save_averages) else None
 470         if backup_func:
 471             logger.info("Backing up models...")
 472         executor = futures.ThreadPoolExecutor()
 473         save_threads = [executor.submit(network.save, backup_func=backup_func)
 474                         for network in self.networks.values()]
 475         save_threads.append(executor.submit(self.state.save, backup_func=backup_func))
 476         futures.wait(save_threads)
 477         # call result() to capture errors
 478         _ = [thread.result() for thread in save_threads]
 479         msg = "[Saved models]"
 480         if save_averages:
 481             lossmsg = ["{}_{}: {:.5f}".format(self.state.loss_names[side][0],
 482                                               side.capitalize(),
 483                                               save_averages[side])
 484                        for side in sorted(list(save_averages.keys()))]
 485             msg += " - Average since last save: {}".format(", ".join(lossmsg))
 486         logger.info(msg)

 488     def get_save_averages(self):
 489         """ Return the average loss since the last save iteration and reset historical loss """
 490         logger.debug("Getting save averages")
 491         avgs = dict()
 492         for side, loss in self.history.items():
 493             if not loss:
 494                 logger.debug("No loss in self.history: %s", side)
 495                 break
 496             avgs[side] = sum(loss) / len(loss)
 497             self.history[side] = list()  # Reset historical loss
 498         logger.debug("Average losses since last save: %s", avgs)
 499         return avgs

 501     def should_backup(self, save_averages):
 502         """ Check whether the loss averages for all losses is the lowest that has been seen.
 503 
 504             This protects against model corruption by only backing up the model
 505             if any of the loss values have fallen.
 506             TODO This is not a perfect system. If the model corrupts on save_iteration - 1
 507             then model may still backup
 508         """
 509         backup = True
 510 
 511         if not save_averages:
 512             logger.debug("No save averages. Not backing up")
 513             return False
 514 
 515         for side, loss in save_averages.items():
 516             if not self.state.lowest_avg_loss.get(side, None):
 517                 logger.debug("Setting initial save iteration loss average for '%s': %s",
 518                              side, loss)
 519                 self.state.lowest_avg_loss[side] = loss
 520                 continue
 521             if backup:
 522                 # Only run this if backup is true. All losses must have dropped for a valid backup
 523                 backup = self.check_loss_drop(side, loss)
 524 
 525         logger.debug("Lowest historical save iteration loss average: %s",
 526                      self.state.lowest_avg_loss)
 527 
 528         if backup:  # Update lowest loss values to the state
 529             for side, avg_loss in save_averages.items():
 530                 logger.debug("Updating lowest save iteration average for '%s': %s", side, avg_loss)
 531                 self.state.lowest_avg_loss[side] = avg_loss
 532 
 533         logger.debug("Backing up: %s", backup)
 534         return backup

 546             Rename them if they are found and update """

 549                                   ("IAE_inter_A.h5", "iae_intermediate_A.h5"),

 571                     set_lowmem = True

 575             logger.debug("No legacy files to rename")

 588         self.state.config["lowmem"] = False
 589         self.encoder_dim = 1024
 590

 593             self.encoder_dim = 512

 708     @property
 709     def mask_input(self):
 710         """ Return the mask input or None """
 711         mask_inputs = [inp for inp in self.inputs if inp.name.startswith("mask")]
 712         if not mask_inputs:
 713             return None
 714         return mask_inputs[0]

 717     def mask_shape(self):
 718         """ Return the mask shape """
 719         if self.mask_input is None:
 720             return None
 721         return K.int_shape(self.mask_input)[1:]

 723     def get_loss_names(self):
 724         """ Return the loss names based on model output """
 725         output_names = [output.name for output in self.outputs]
 726         logger.debug("Model output names: %s", output_names)
 727         loss_names = [name[name.find("/") + 1:name.rfind("/")].replace("_out", "")
 728                       for name in output_names]
 729         if not all(name.startswith("face") or name.startswith("mask") for name in loss_names):
 730             # Handle incorrectly named/legacy outputs
 731             logger.debug("Renaming loss names from: %s", loss_names)
 732             loss_names = self.update_loss_names()
 733         loss_names = ["{}_loss".format(name) for name in loss_names]
 734         logger.debug(loss_names)
 735         return loss_names
 736 
 737     def update_loss_names(self):
 738         """ Update loss names if named incorrectly or legacy model """
 739         output_types = ["mask" if shape[-1] == 1 else "face" for shape in self.output_shapes]
 740         loss_names = ["{}{}".format(name,
 741                                     "" if output_types.count(name) == 1 else "_{}".format(idx))
 742                       for idx, name in enumerate(output_types)]
 743         logger.debug("Renamed loss names to: %s", loss_names)
 744         return loss_names
 745 
 746     def get_loss_functions(self):
 747         """ Set the loss function """
 748         loss_funcs = []
 749         for idx, loss_name in enumerate(self.names):
 750             if loss_name.startswith("mask"):
 751                 loss_funcs.append(self.selected_mask_loss)
 752             elif self.config["penalized_mask_loss"] and self.config["mask_type"] is not None:
 753                 face_size = self.output_shapes[idx][1]
 754                 mask_size = self.mask_shape[1]
 755                 scaling = face_size / mask_size
 756                 logger.debug("face_size: %s mask_size: %s, mask_scaling: %s",
 757                              face_size, mask_size, scaling)
 758                 loss_funcs.append(PenalizedLoss(self.mask_input, self.selected_loss,
 759                                                 mask_scaling=scaling,
 760                                                 preprocessing_func=self.mask_preprocessing_func))
 761             else:
 762                 loss_funcs.append(self.selected_loss)
 763             logger.debug("%s: %s", loss_name, loss_funcs[-1])
 764         logger.debug(loss_funcs)
 765         return loss_funcs

1003     def _update_legacy_config(self):
1004         """ Legacy updates for new config additions.
1005 
1006         When new config items are added to the Faceswap code, existing model state files need to be
1007         updated to handle these new items.
1008 
1009         Current existing legacy update items:
1010 
1011             * loss - If old `dssim_loss` is ``true`` set new `loss_function` to `ssim` otherwise
1012             set it to `mae`. Remove old `dssim_loss` item
1013 
1014             * masks - If `learn_mask` does not exist then it is set to ``True`` if `mask_type` is
1015             not ``None`` otherwise it is set to ``False``.
1016 
1017             * masks type - Replace removed masks 'dfl_full' and 'facehull' with `components` mask
1018 
1019         Returns
1020         -------
1021         bool
1022             ``True`` if legacy items exist and state file has been updated, otherwise ``False``
1023         """
1024         logger.debug("Checking for legacy state file update")
1025         priors = ["dssim_loss", "mask_type", "mask_type"]
1026         new_items = ["loss_function", "learn_mask", "mask_type"]
1027         updated = False
1028         for old, new in zip(priors, new_items):
1029             if old not in self.config:
1030                 logger.debug("Legacy item '%s' not in config. Skipping update", old)
1031                 continue
1032 
1033             # dssim_loss > loss_function
1034             if old == "dssim_loss":
1035                 self.config[new] = "ssim" if self.config[old] else "mae"
1036                 del self.config[old]
1037                 updated = True
1038                 logger.info("Updated config from legacy dssim format. New config loss "
1039                             "function: '%s'", self.config[new])
1040                 continue
1041 
1042             # Add learn mask option and set to True if model has "penalized_mask_loss" specified
1043             if old == "mask_type" and new == "learn_mask" and new not in self.config:
1044                 self.config[new] = self.config["mask_type"] is not None
1045                 updated = True
1046                 logger.info("Added new 'learn_mask' config item for this model. Value set to: %s",
1047                             self.config[new])
1048                 continue
1049 
1050             # Replace removed masks with most similar equivalent
1051             if old == "mask_type" and new == "mask_type" and self.config[old] in ("facehull",
1052                                                                                   "dfl_full"):
1053                 old_mask = self.config[old]
1054                 self.config[new] = "components"
1055                 updated = True
1056                 logger.info("Updated 'mask_type' from '%s' to '%s' for this model",
1057                             old_mask, self.config[new])
1058 
1059         logger.debug("State file updated for legacy config: %s", updated)
1060         return updated
1061 
1062     def update_changed_config_items(self, config_changeable_items):
1063         """ Update any parameters which are not fixed and have been changed """
1064         if not config_changeable_items:
1065             logger.debug("No changeable parameters have been updated")
1066             return
1067         for key, val in config_changeable_items.items():
1068             old_val = self.config[key]
1069             if old_val == val:
1070                 continue
1071             self.config[key] = val
1072             logger.info("Config item: '%s' has been updated from '%s' to '%s'", key, old_val, val)
