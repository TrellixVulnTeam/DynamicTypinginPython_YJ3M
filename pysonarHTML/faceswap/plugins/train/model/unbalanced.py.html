  36     def encoder(self):
  37         """ Unbalanced Encoder """
  38         kwargs = dict(kernel_initializer=self.kernel_initializer)
  39         encoder_complexity = 128 if self.lowmem else self.config["complexity_encoder"]
  40         dense_dim = 384 if self.lowmem else 512
  41         dense_shape = self.input_shape[0] // 16
  42         input_ = Input(shape=self.input_shape)
  43 
  44         var_x = input_
  45         var_x = self.blocks.conv(var_x, encoder_complexity, use_instance_norm=True, **kwargs)
  46         var_x = self.blocks.conv(var_x, encoder_complexity * 2, use_instance_norm=True, **kwargs)
  47         var_x = self.blocks.conv(var_x, encoder_complexity * 4, **kwargs)
  48         var_x = self.blocks.conv(var_x, encoder_complexity * 6, **kwargs)
  49         var_x = self.blocks.conv(var_x, encoder_complexity * 8, **kwargs)
  50         var_x = Dense(self.encoder_dim,
  51                       kernel_initializer=self.kernel_initializer)(Flatten()(var_x))
  52         var_x = Dense(dense_shape * dense_shape * dense_dim,
  53                       kernel_initializer=self.kernel_initializer)(var_x)
  54         var_x = Reshape((dense_shape, dense_shape, dense_dim))(var_x)
  55         return KerasModel(input_, var_x)

  57     def decoder_a(self):
  58         """ Decoder for side A """
  59         kwargs = dict(kernel_size=5, kernel_initializer=self.kernel_initializer)
  60         decoder_complexity = 320 if self.lowmem else self.config["complexity_decoder_a"]
  61         dense_dim = 384 if self.lowmem else 512
  62         decoder_shape = self.input_shape[0] // 16
  63         input_ = Input(shape=(decoder_shape, decoder_shape, dense_dim))
  64 
  65         var_x = input_
  66 
  67         var_x = self.blocks.upscale(var_x, decoder_complexity, **kwargs)
  68         var_x = SpatialDropout2D(0.25)(var_x)
  69         var_x = self.blocks.upscale(var_x, decoder_complexity, **kwargs)
  70         if self.lowmem:
  71             var_x = SpatialDropout2D(0.15)(var_x)
  72         else:
  73             var_x = SpatialDropout2D(0.25)(var_x)
  74         var_x = self.blocks.upscale(var_x, decoder_complexity // 2, **kwargs)
  75         var_x = self.blocks.upscale(var_x, decoder_complexity // 4, **kwargs)
  76         var_x = self.blocks.conv2d(var_x, 3,
  77                                    kernel_size=5,
  78                                    padding="same",
  79                                    activation="sigmoid",
  80                                    name="face_out")
  81         outputs = [var_x]
  82 
  83         if self.config.get("learn_mask", False):
  84             var_y = input_
  85             var_y = self.blocks.upscale(var_y, decoder_complexity)
  86             var_y = self.blocks.upscale(var_y, decoder_complexity)
  87             var_y = self.blocks.upscale(var_y, decoder_complexity // 2)
  88             var_y = self.blocks.upscale(var_y, decoder_complexity // 4)
  89             var_y = self.blocks.conv2d(var_y, 1,
  90                                        kernel_size=5,
  91                                        padding="same",
  92                                        activation="sigmoid",
  93                                        name="mask_out")
  94             outputs.append(var_y)
  95         return KerasModel(input_, outputs=outputs)

  97     def decoder_b(self):
  98         """ Decoder for side B """
  99         kwargs = dict(kernel_size=5, kernel_initializer=self.kernel_initializer)
 100         dense_dim = 384 if self.lowmem else self.config["complexity_decoder_b"]
 101         decoder_complexity = 384 if self.lowmem else 512
 102         decoder_shape = self.input_shape[0] // 16
 103         input_ = Input(shape=(decoder_shape, decoder_shape, dense_dim))
 104 
 105         var_x = input_
 106         if self.lowmem:
 107             var_x = self.blocks.upscale(var_x, decoder_complexity, **kwargs)
 108             var_x = self.blocks.upscale(var_x, decoder_complexity // 2, **kwargs)
 109             var_x = self.blocks.upscale(var_x, decoder_complexity // 4, **kwargs)
 110             var_x = self.blocks.upscale(var_x, decoder_complexity // 8, **kwargs)
 111         else:
 112             var_x = self.blocks.upscale(var_x, decoder_complexity,
 113                                         res_block_follows=True, **kwargs)
 114             var_x = self.blocks.res_block(var_x, decoder_complexity,
 115                                           kernel_initializer=self.kernel_initializer)
 116             var_x = self.blocks.upscale(var_x, decoder_complexity,
 117                                         res_block_follows=True, **kwargs)
 118             var_x = self.blocks.res_block(var_x, decoder_complexity,
 119                                           kernel_initializer=self.kernel_initializer)
 120             var_x = self.blocks.upscale(var_x, decoder_complexity // 2,
 121                                         res_block_follows=True, **kwargs)
 122             var_x = self.blocks.res_block(var_x, decoder_complexity // 2,
 123                                           kernel_initializer=self.kernel_initializer)
 124             var_x = self.blocks.upscale(var_x, decoder_complexity // 4, **kwargs)
 125         var_x = self.blocks.conv2d(var_x, 3,
 126                                    kernel_size=5,
 127                                    padding="same",
 128                                    activation="sigmoid",
 129                                    name="face_out")
 130         outputs = [var_x]
 131 
 132         if self.config.get("learn_mask", False):
 133             var_y = input_
 134             var_y = self.blocks.upscale(var_y, decoder_complexity)
 135             if not self.lowmem:
 136                 var_y = self.blocks.upscale(var_y, decoder_complexity)
 137             var_y = self.blocks.upscale(var_y, decoder_complexity // 2)
 138             var_y = self.blocks.upscale(var_y, decoder_complexity // 4)
 139             if self.lowmem:
 140                 var_y = self.blocks.upscale(var_y, decoder_complexity // 8)
 141             var_y = self.blocks.conv2d(var_y, 1,
 142                                        kernel_size=5,
 143                                        padding="same",
 144                                        activation="sigmoid",
 145                                        name="mask_out")
 146             outputs.append(var_y)
 147         return KerasModel(input_, outputs=outputs)
