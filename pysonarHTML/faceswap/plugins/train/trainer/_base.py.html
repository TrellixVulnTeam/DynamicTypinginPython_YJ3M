   1 #!/usr/bin/env python3
   2 """ Base Class for Faceswap Trainer plugins. All Trainer plugins should be inherited from
   3 this class.
   4 
   5 At present there is only the :class:`~plugins.train.trainer.original` plugin, so that entirely
   6 inherits from this class.
   7 
   8 This class heavily references the :attr:`plugins.train.model._base.ModelBase.training_opts`
   9 ``dict``. The following keys are expected from this ``dict``:
  10 
  11     * **alignments** (`dict`, `optional`) - If training with a mask or the warp to landmarks \
  12     command line option is selected then this is required, otherwise it can be ``None``. The \
  13     dictionary should contain 2 keys ("a" and "b") with the values being the path to the \
  14     alignments file for the corresponding side.
  15 
  16     * **preview_scaling** (`int`) - How much to scale displayed preview image by.
  17 
  18     * **training_size** ('int') - Size of the training images in pixels.
  19 
  20     * **coverage_ratio** ('float') - Ratio of face to be cropped out of the training image.
  21 
  22     * **mask_type** ('str') - The type of mask to select from the alignments file.
  23 
  24     * **mask_blur_kernel** ('int') - The size of the kernel to use for gaussian blurring the mask.
  25 
  26     * **mask_threshold** ('int') - The threshold for min/maxing mask to 0/100.
  27 
  28     * **learn_mask** ('bool') - Whether the mask should be trained in the model.
  29 
  30     * **penalized_mask_loss** ('bool') - Whether the mask should be penalized from loss.
  31 
  32     * **no_logs** ('bool') - Whether Tensorboard logging should be disabled.
  33 
  34     * **snapshot_interval** ('int') - How many iterations between model snapshot saves.
  35 
  36     * **warp_to_landmarks** ('bool') - Whether to use random_warp_landmarks instead of random_warp.
  37 
  38     * **augment_color** ('bool') - Whether to use color augmentation.
  39 
  40     * **no_flip** ('bool') - Whether to turn off random horizontal flipping.
  41 
  42     * **pingpong** ('bool') - Train each side separately per save iteration rather than together.
  43 """

 104     def __init__(self, model, images, batch_size, configfile):
 105         logger.debug("Initializing %s: (model: '%s', batch_size: %s)",
 106                      self.__class__.__name__, model, batch_size)
 107         self._config = _get_config(".".join(self.__module__.split(".")[-2:]),
 108                                    configfile=configfile)
 109         self._model = model
 110         self._model.state.add_session_batchsize(batch_size)
 111         self._images = images
 112         self._sides = sorted(key for key in self._images.keys())
 113 
 114         self._process_training_opts()
 115         self._pingpong = PingPong(model, self._sides)
 116 
 117         self._batchers = {side: Batcher(side,
 118                                         images[side],
 119                                         self._model,
 120                                         self._use_mask,

 122                                         self._config)
 123                           for side in self._sides}

 125         self._tensorboard = self._set_tensorboard()
 126         self._samples = Samples(self._model,
 127                                 self._use_mask,
 128                                 self._model.training_opts["coverage_ratio"],
 129                                 self._model.training_opts["preview_scaling"])

 132                                     self._model.training_opts["coverage_ratio"],
 133                                     self._config.get("preview_images", 14),
 134                                     self._batchers)

 162     def _process_training_opts(self):
 163         """ Extrapolate alignments and masks from the alignments file into
 164         :attr:`_model.training_opts`."""
 165         logger.debug(self._model.training_opts)
 166         if not self._landmarks_required and not self._use_mask:
 167             return
 168 
 169         alignments = TrainingAlignments(self._model.training_opts, self._images)
 170         if self._landmarks_required:
 171             logger.debug("Adding landmarks to training opts dict")
 172             self._model.training_opts["landmarks"] = alignments.landmarks
 173 
 174         if self._use_mask:
 175             logger.debug("Adding masks to training opts dict")
 176             self._model.training_opts["masks"] = alignments.masks

 178     def _set_tensorboard(self):
 179         """ Set up Tensorboard callback for logging loss.
 180 
 181         Bypassed if command line option "no-logs" has been selected.
 182 
 183         Returns
 184         -------
 185         dict:
 186             2 Dictionary keys of "a" and "b" the values of which are the
 187         :class:`tf.keras.callbacks.TensorBoard` objects for the respective sides.
 188         """
 189         if self._model.training_opts["no_logs"]:
 190             logger.verbose("TensorBoard logging disabled")
 191             return None
 192         if self._pingpong.active:
 193             # Currently TensorBoard uses the tf.session, meaning that VRAM does not
 194             # get cleared when model switching
 195             # TODO find a fix for this
 196             logger.warning("Currently TensorBoard logging is not supported for Ping-Pong "
 197                            "training. Session stats and graphing will not be available for this "
 198                            "training session.")
 199             return None
 200 
 201         logger.debug("Enabling TensorBoard Logging")
 202         tensorboard = dict()
 203 
 204         for side in self._sides:
 205             logger.debug("Setting up TensorBoard Logging. Side: %s", side)
 206             log_dir = os.path.join(str(self._model.model_dir),
 207                                    "{}_logs".format(self._model.name),
 208                                    side,
 209                                    "session_{}".format(self._model.state.session_id))
 210             tbs = tf.keras.callbacks.TensorBoard(log_dir=log_dir, **self._tensorboard_kwargs)
 211             tbs.set_model(self._model.predictors[side])
 212             tensorboard[side] = tbs
 213         logger.info("Enabled TensorBoard Logging")
 214         return tensorboard
 215 
 216     @property
 217     def _tensorboard_kwargs(self):
 218         """ dict: The keyword arguments to be passed to :class:`tf.keras.callbacks.TensorBoard`.
 219         NB: Tensorflow 1.13 + needs an additional keyword argument which is not valid for earlier
 220         versions """
 221         kwargs = dict(histogram_freq=0,  # Must be 0 or hangs
 222                       batch_size=64,
 223                       write_graph=True,
 224                       write_grads=True)
 225         tf_version = [int(ver) for ver in tf.__version__.split(".") if ver.isdigit()]
 226         logger.debug("Tensorflow version: %s", tf_version)
 227         if tf_version[0] > 1 or (tf_version[0] == 1 and tf_version[1] > 12):
 228             kwargs["update_freq"] = "batch"
 229         if tf_version[0] > 1 or (tf_version[0] == 1 and tf_version[1] > 13):
 230             kwargs["profile_batch"] = 0
 231         logger.debug(kwargs)
 232         return kwargs

 234     def __print_loss(self, loss):
 235         """ Outputs the loss for the current iteration to the console.
 236 
 237         Parameters
 238         ----------
 239         loss: dict
 240             The loss for each side. The dictionary should contain 2 keys ("a" and "b") with the
 241             values being a list of loss values for the current iteration corresponding to
 242             each side.
 243          """
 244         logger.trace(loss)
 245         output = ["Loss {}: {:.5f}".format(side.capitalize(), loss[side][0])
 246                   for side in sorted(loss.keys())]
 247         output = ", ".join(output)
 248         output = "[{}] [#{:05d}] {}".format(self._timestamp, self._model.iterations, output)
 249         print("\r{}".format(output), end="")

 251     def train_one_step(self, viewer, timelapse_kwargs):
 252         """ Running training on a batch of images for each side.
 253 
 254         Triggered from the training cycle in :class:`scripts.train.Train`.
 255 
 256         Notes
 257         -----
 258         As every iteration is called explicitly, the Parameters defined should always be ``None``
 259         except on save iterations.
 260 
 261         Parameters
 262         ----------
 263         viewer: :func:`scripts.train.Train._show`
 264             The function that will display the preview image
 265         timelapse_kwargs: dict
 266             The keyword arguments for generating time-lapse previews. If a time-lapse preview is
 267             not required then this should be ``None``. Otherwise all values should be full paths
 268             the keys being `input_a`, `input_b`, `output`.
 269         """
 270         logger.trace("Training one step: (iteration: %s)", self._model.iterations)
 271         do_preview = viewer is not None
 272         do_timelapse = timelapse_kwargs is not None
 273         snapshot_interval = self._model.training_opts.get("snapshot_interval", 0)
 274         do_snapshot = (snapshot_interval != 0 and
 275                        self._model.iterations >= snapshot_interval and
 276                        self._model.iterations % snapshot_interval == 0)
 277 
 278         loss = dict()
 279         try:
 280             for side, batcher in self._batchers.items():
 281                 if self._pingpong.active and side != self._pingpong.side:
 282                     continue
 283                 loss[side] = batcher.train_one_batch()
 284                 if not do_preview and not do_timelapse:
 285                     continue
 286                 if do_preview:
 287                     batcher.generate_preview(do_preview)
 288                     self._samples.images[side] = batcher.compile_sample(None)
 289                 if do_timelapse:
 290                     self._timelapse.get_sample(side, timelapse_kwargs)
 291 
 292             self._model.state.increment_iterations()
 293 
 294             for side, side_loss in loss.items():
 295                 self._store_history(side, side_loss)
 296                 self._log_tensorboard(side, side_loss)
 297 
 298             if not self._pingpong.active:
 299                 self.__print_loss(loss)
 300             else:
 301                 for key, val in loss.items():
 302                     self._pingpong.loss[key] = val
 303                 self.__print_loss(self._pingpong.loss)
 304 
 305             if do_preview:
 306                 samples = self._samples.show_sample()
 307                 if samples is not None:
 308                     viewer(samples, "Training - 'S': Save Now. 'ENTER': Save and Quit")
 309 
 310             if do_timelapse:
 311                 self._timelapse.output_timelapse()
 312 
 313             if do_snapshot:
 314                 self._model.do_snapshot()
 315         except Exception as err:
 316             raise err
 317 
 318     def _store_history(self, side, loss):
 319         """ Store the loss for this step into :attr:`model.history`.
 320 
 321         Parameters
 322         ----------
 323         side: {"a", "b"}
 324             The side to store the loss for
 325         loss: list
 326             The list of loss ``floats`` for this side
 327         """
 328         logger.trace("Updating loss history: '%s'", side)
 329         self._model.history[side].append(loss[0])  # Either only loss or total loss
 330         logger.trace("Updated loss history: '%s'", side)
 331 
 332     def _log_tensorboard(self, side, loss):
 333         """ Log current loss to Tensorboard log files
 334 
 335         Parameters
 336         ----------
 337         side: {"a", "b"}
 338             The side to store the loss for
 339         loss: list
 340             The list of loss ``floats`` for this side
 341         """
 342         if not self._tensorboard:
 343             return
 344         logger.trace("Updating TensorBoard log: '%s'", side)
 345         logs = {log[0]: log[1]
 346                 for log in zip(self._model.state.loss_names[side], loss)}
 347         self._tensorboard[side].on_batch_end(self._model.state.iterations, logs)
 348         logger.trace("Updated TensorBoard log: '%s'", side)
 349 
 350     def clear_tensorboard(self):
 351         """ Stop Tensorboard logging.
 352 
 353         Tensorboard logging needs to be explicitly shutdown on training termination. Called from
 354         :class:`scripts.train.Train` when training is stopped.
 355          """
 356         if not self._tensorboard:
 357             return
 358         for side, tensorboard in self._tensorboard.items():
 359             logger.debug("Ending Tensorboard. Side: '%s'", side)
 360             tensorboard.on_train_end(None)

 363 class Batcher():
 364     """ Handles the processing of a Batch for a single side.
 365 
 366     Parameters
 367     ----------
 368     side: {"a" or "b"}
 369         The side that this :class:`Batcher` belongs to
 370     images: list
 371         The list of full paths to the training images for this :class:`Batcher`
 372     model: plugin from :mod:`plugins.train.model`
 373         The selected model that will be running this trainer
 374     use_mask: bool
 375         ``True`` if a mask is required for training otherwise ``False``
 376     batch_size: int
 377         The size of the batch to be processed at each iteration
 378     config: :class:`lib.config.FaceswapConfig`
 379         The configuration for this trainer
 380     """
 381     def __init__(self, side, images, model, use_mask, batch_size, config):
 382         logger.debug("Initializing %s: side: '%s', num_images: %s, use_mask: %s, batch_size: %s, "
 383                      "config: %s)",
 384                      self.__class__.__name__, side, len(images), use_mask, batch_size, config)
 385         self._model = model
 386         self._use_mask = use_mask
 387         self._side = side
 388         self._images = images
 389         self._config = config
 390         self._target = None
 391         self._samples = None
 392         self._masks = None
 393 
 394         generator = self._load_generator()
 395         self._feed = generator.minibatch_ab(images, batch_size, self._side)
 396 
 397         self._preview_feed = None
 398         self._timelapse_feed = None
 399         self._set_preview_feed()
 400 
 401     def _load_generator(self):
 402         """ Load the :class:`lib.training_data.TrainingDataGenerator` for this batcher """
 403         logger.debug("Loading generator: %s", self._side)
 404         input_size = self._model.input_shape[0]
 405         output_shapes = self._model.output_shapes
 406         logger.debug("input_size: %s, output_shapes: %s", input_size, output_shapes)
 407         generator = TrainingDataGenerator(input_size,
 408                                           output_shapes,
 409                                           self._model.training_opts,
 410                                           self._config)
 411         return generator
 412 
 413     def train_one_batch(self):
 414         """ Train on a single batch of images for this :class:`Batcher`
 415 
 416         Returns
 417         -------
 418         list
 419             The list of loss values (as ``float``) for this batch
 420         """
 421         logger.trace("Training one step: (side: %s)", self._side)
 422         model_inputs, model_targets = self._get_next()
 423         try:
 424             loss = self._model.predictors[self._side].train_on_batch(model_inputs, model_targets)
 425         except tf_errors.ResourceExhaustedError as err:
 426             msg = ("You do not have enough GPU memory available to train the selected model at "
 427                    "the selected settings. You can try a number of things:"
 428                    "\n1) Close any other application that is using your GPU (web browsers are "
 429                    "particularly bad for this)."
 430                    "\n2) Lower the batchsize (the amount of images fed into the model each "
 431                    "iteration)."
 432                    "\n3) Try 'Memory Saving Gradients' and/or 'Optimizer Savings' and/or 'Ping "
 433                    "Pong Training'."
 434                    "\n4) Use a more lightweight model, or select the model's 'LowMem' option "
 435                    "(in config) if it has one.")
 436             raise FaceswapError(msg) from err
 437         loss = loss if isinstance(loss, list) else [loss]
 438         return loss
 439 
 440     def _get_next(self):
 441         """ Return the next batch from the :class:`lib.training_data.TrainingDataGenerator` for
 442         this batcher ready for feeding into the model.
 443 
 444         Returns
 445         -------
 446         model_inputs: list
 447             A list of :class:`numpy.ndarray` for feeding into the model
 448         model_targets: list
 449             A list of :class:`numpy.ndarray` for comparing the output of the model
 450         """
 451         logger.trace("Generating targets")
 452         batch = next(self._feed)
 453         targets_use_mask = self._model.training_opts["learn_mask"]
 454         model_inputs = batch["feed"] + batch["masks"] if self._use_mask else batch["feed"]
 455         model_targets = batch["targets"] + batch["masks"] if targets_use_mask else batch["targets"]
 456         return model_inputs, model_targets

 458     def generate_preview(self, do_preview):
 459         """ Generate the preview images.
 460 
 461         Parameters
 462         ----------
 463         do_preview: bool
 464             Whether the previews should be generated. ``True`` if they should ``False`` if they
 465             should not be generated, in which case currently stored previews should be deleted.
 466         """
 467         if not do_preview:
 468             self._samples = None
 469             self._target = None
 470             self._masks = None
 471             return
 472         logger.debug("Generating preview")
 473         batch = next(self._preview_feed)
 474         self._samples = batch["samples"]
 475         self._target = batch["targets"][self._model.largest_face_index]
 476         self._masks = batch["masks"][0]

 478     def _set_preview_feed(self):
 479         """ Set the preview feed for this batcher.
 480 
 481         Creates a generator from :class:`lib.training_data.TrainingDataGenerator` specifically
 482         for previews for the batcher.
 483         """
 484         logger.debug("Setting preview feed: (side: '%s')", self._side)
 485         preview_images = self._config.get("preview_images", 14)
 486         preview_images = min(max(preview_images, 2), 16)
 487         batchsize = min(len(self._images), preview_images)
 488         self._preview_feed = self._load_generator().minibatch_ab(self._images,
 489                                                                  batchsize,
 490                                                                  self._side,
 491                                                                  do_shuffle=True,
 492                                                                  is_preview=True)
 493         logger.debug("Set preview feed. Batchsize: %s", batchsize)

 495     def compile_sample(self, batch_size, samples=None, images=None, masks=None):
 496         """ Compile the preview samples for display.
 497 
 498         Parameters
 499         ----------
 500         batch_size: int
 501             The requested batch size for each training iterations
 502         samples: :class:`numpy.ndarray`, optional
 503             The sample images that should be used for creating the preview. If ``None`` then the
 504             samples will be generated from the internal random image generator.
 505             Default: ``None``
 506         images:  :class:`numpy.ndarray`, optional
 507             The target images that should be used for creating the preview. If ``None`` then the
 508             targets will be generated from the internal random image generator.
 509             Default: ``None``
 510         masks:  :class:`numpy.ndarray`, optional
 511             The masks that should be used for creating the preview. If ``None`` then the
 512             masks will be generated from the internal random image generator.
 513             Default: ``None``
 514 
 515         Returns
 516         -------
 517         list
 518             The list of samples, targets and masks as :class:`numpy.ndarrays` for creating a
 519             preview image
 520          """
 521         num_images = self._config.get("preview_images", 14)
 522         num_images = min(batch_size, num_images) if batch_size is not None else num_images
 523         logger.debug("Compiling samples: (side: '%s', samples: %s)", self._side, num_images)
 524         images = images if images is not None else self._target
 525         masks = masks if masks is not None else self._masks
 526         samples = samples if samples is not None else self._samples
 527         retval = [samples[0:num_images], images[0:num_images], masks[0:num_images]]
 528         return retval
 529 
 530     def compile_timelapse_sample(self):
 531         """ Compile the sample images for creating a time-lapse frame.
 532 
 533         Returns
 534         -------
 535         list
 536             The list of samples, targets and masks as :class:`numpy.ndarrays` for creating a
 537             time-lapse frame
 538         """
 539         batch = next(self._timelapse_feed)
 540         batchsize = len(batch["samples"])
 541         images = batch["targets"][self._model.largest_face_index]
 542         masks = batch["masks"][0]
 543         sample = self.compile_sample(batchsize,
 544                                      samples=batch["samples"],
 545                                      images=images,
 546                                      masks=masks)
 547         return sample

 572 class Samples():
 573     """ Compile samples for display for preview and time-lapse
 574 
 575     Parameters
 576     ----------
 577     model: plugin from :mod:`plugins.train.model`
 578         The selected model that will be running this trainer
 579     use_mask: bool
 580         ``True`` if a mask should be displayed otherwise ``False``
 581     coverage_ratio: float
 582         Ratio of face to be cropped out of the training image.
 583     scaling: float, optional
 584         The amount to scale the final preview image by. Default: `1.0`
 585 
 586     Attributes
 587     ----------
 588     images: dict
 589         The :class:`numpy.ndarray` training images for generating previews on each side. The
 590         dictionary should contain 2 keys ("a" and "b") with the values being the training images
 591         for generating samples corresponding to each side.

 593     def __init__(self, model, use_mask, coverage_ratio, scaling=1.0):

 603     def show_sample(self):
 604         """ Compile a preview image.
 605 
 606         Returns
 607         -------
 608         :class:`numpy.ndarry`
 609             A compiled preview image ready for display or saving
 610         """
 611         if len(self.images) != 2:
 612             logger.debug("Ping Pong training - Only one side trained. Aborting preview")
 613             return None
 614         logger.debug("Showing sample")
 615         feeds = dict()
 616         figures = dict()
 617         headers = dict()
 618         for side, samples in self.images.items():
 619             faces = samples[1]
 620             if self._model.input_shape[0] / faces.shape[1] != 1.0:
 621                 feeds[side] = self._resize_sample(side, faces, self._model.input_shape[0])
 622                 feeds[side] = feeds[side].reshape((-1, ) + self._model.input_shape)
 623             else:
 624                 feeds[side] = faces
 625             if self._use_mask:
 626                 mask = samples[-1]
 627                 feeds[side] = [feeds[side], mask]
 628 
 629         preds = self._get_predictions(feeds["a"], feeds["b"])
 630 
 631         for side, samples in self.images.items():
 632             other_side = "a" if side == "b" else "b"
 633             predictions = [preds["{0}_{0}".format(side)],
 634                            preds["{}_{}".format(other_side, side)]]
 635             display = self._to_full_frame(side, samples, predictions)
 636             headers[side] = self._get_headers(side, display[0].shape[1])
 637             figures[side] = np.stack([display[0], display[1], display[2], ], axis=1)
 638             if self.images[side][0].shape[0] % 2 == 1:
 639                 figures[side] = np.concatenate([figures[side],
 640                                                 np.expand_dims(figures[side][0], 0)])
 641 
 642         width = 4
 643         side_cols = width // 2
 644         if side_cols != 1:
 645             headers = self._duplicate_headers(headers, side_cols)
 646 
 647         header = np.concatenate([headers["a"], headers["b"]], axis=1)
 648         figure = np.concatenate([figures["a"], figures["b"]], axis=0)
 649         height = int(figure.shape[0] / width)
 650         figure = figure.reshape((width, height) + figure.shape[1:])
 651         figure = _stack_images(figure)
 652         figure = np.concatenate((header, figure), axis=0)
 653 
 654         logger.debug("Compiled sample")
 655         return np.clip(figure * 255, 0, 255).astype('uint8')

 657     @staticmethod
 658     def _resize_sample(side, sample, target_size):
 659         """ Resize a given image to the target size.
 660 
 661         Parameters
 662         ----------
 663         sample: :class:`numpy.ndarray`
 664             The sample to be resized
 665         target_size: int
 666             The size that the sample should be resized to
 667 
 668         Returns
 669         -------
 670         :class:`numpy.ndarray`
 671             The sample resized to the target size
 672         """
 673         scale = target_size / sample.shape[1]
 674         if scale == 1.0:
 675             return sample
 676         logger.debug("Resizing sample: (side: '%s', sample.shape: %s, target_size: %s, scale: %s)",
 677                      side, sample.shape, target_size, scale)
 678         interpn = cv2.INTER_CUBIC if scale > 1.0 else cv2.INTER_AREA
 679         retval = np.array([cv2.resize(img, (target_size, target_size), interpn)
 680                            for img in sample])
 681         logger.debug("Resized sample: (side: '%s' shape: %s)", side, retval.shape)
 682         return retval

 684     def _get_predictions(self, feed_a, feed_b):
 685         """ Feed the samples to the model and return predictions
 686 
 687         Parameters
 688         ----------
 689         feed_a: list
 690             List of :class:`numpy.ndarray` of feed images for the "a" side
 691         feed_a: list
 692             List of :class:`numpy.ndarray` of feed images for the "b" side
 693 
 694         Returns
 695         list:
 696             List of :class:`numpy.ndarray` of predictions received from the model
 697         """
 698         logger.debug("Getting Predictions")
 699         preds = dict()
 700         preds["a_a"] = self._model.predictors["a"].predict(feed_a)
 701         preds["b_a"] = self._model.predictors["b"].predict(feed_a)
 702         preds["a_b"] = self._model.predictors["a"].predict(feed_b)
 703         preds["b_b"] = self._model.predictors["b"].predict(feed_b)
 704         # Get the returned largest image from predictors that emit multiple items
 705         if not isinstance(preds["a_a"], np.ndarray):
 706             for key, val in preds.items():
 707                 preds[key] = val[self._model.largest_face_index]
 708         logger.debug("Returning predictions: %s", {key: val.shape for key, val in preds.items()})
 709         return preds

 711     def _to_full_frame(self, side, samples, predictions):
 712         """ Patch targets and prediction images into images of training image size.
 713 
 714         Parameters
 715         ----------
 716         side: {"a" or "b"}
 717             The side that these samples are for
 718         samples: list
 719             List of :class:`numpy.ndarray` of target images and feed images
 720         predictions: list
 721             List of :class: `numpy.ndarray` of predictions from the model
 722         """
 723         logger.debug("side: '%s', number of sample arrays: %s, prediction.shapes: %s)",
 724                      side, len(samples), [pred.shape for pred in predictions])
 725         full, faces = samples[:2]
 726         images = [faces] + predictions
 727         full_size = full.shape[1]
 728         target_size = int(full_size * self._coverage_ratio)
 729         if target_size != full_size:
 730             frame = self._frame_overlay(full, target_size, (0, 0, 255))
 731 
 732         if self._use_mask:
 733             images = self._compile_masked(images, samples[-1])
 734         images = [self._resize_sample(side, image, target_size) for image in images]
 735         if target_size != full_size:
 736             images = [self._overlay_foreground(frame, image) for image in images]
 737         if self._scaling != 1.0:
 738             new_size = int(full_size * self._scaling)
 739             images = [self._resize_sample(side, image, new_size) for image in images]
 740         return images

 742     @staticmethod
 743     def _frame_overlay(images, target_size, color):
 744         """ Add a frame overlay to preview images indicating the region of interest.
 745 
 746         This is the red border that appears in the preview images.
 747 
 748         Parameters
 749         ----------
 750         images: :class:`numpy.ndarray`
 751             The samples to apply the frame to
 752         target_size: int
 753             The size of the sample within the full size frame
 754         color: tuple
 755             The (Blue, Green, Red) color to use for the frame
 756 
 757         Returns
 758         -------
 759         :class:`numpy,ndarray`
 760             The samples with the frame overlay applied
 761         """
 762         logger.debug("full_size: %s, target_size: %s, color: %s",
 763                      images.shape[1], target_size, color)
 764         new_images = list()
 765         full_size = images.shape[1]
 766         padding = (full_size - target_size) // 2
 767         length = target_size // 4
 768         t_l, b_r = (padding, full_size - padding)
 769         for img in images:
 770             cv2.rectangle(img, (t_l, t_l), (t_l + length, t_l + length), color, 3)
 771             cv2.rectangle(img, (b_r, t_l), (b_r - length, t_l + length), color, 3)
 772             cv2.rectangle(img, (b_r, b_r), (b_r - length, b_r - length), color, 3)
 773             cv2.rectangle(img, (t_l, b_r), (t_l + length, b_r - length), color, 3)
 774             new_images.append(img)
 775         retval = np.array(new_images)
 776         logger.debug("Overlayed background. Shape: %s", retval.shape)
 777         return retval

 779     @staticmethod
 780     def _compile_masked(faces, masks):
 781         """ Add the mask to the faces for masked preview.
 782 
 783         Places an opaque red layer over areas of the face that are masked out.
 784 
 785         Parameters
 786         ----------
 787         faces: :class:`numpy.ndarray`
 788             The sample faces that are to have the mask applied
 789         masks: :class:`numpy.ndarray`
 790             The masks that are to be applied to the faces
 791 
 792         Returns
 793         -------
 794         list
 795             List of :class:`numpy.ndarray` faces with the opaque mask layer applied
 796         """
 797         retval = list()
 798         masks3 = np.tile(1 - np.rint(masks), 3)
 799         for mask in masks3:
 800             mask[np.where((mask == [1., 1., 1.]).all(axis=2))] = [0., 0., 1.]
 801         for previews in faces:
 802             images = np.array([cv2.addWeighted(img, 1.0, masks3[idx], 0.3, 0)
 803                                for idx, img in enumerate(previews)])
 804             retval.append(images)
 805         logger.debug("masked shapes: %s", [faces.shape for faces in retval])
 806         return retval

 808     @staticmethod
 809     def _overlay_foreground(backgrounds, foregrounds):
 810         """ Overlay the preview images into the center of the background images
 811 
 812         Parameters
 813         ----------
 814         backgrounds: list
 815             List of :class:`numpy.ndarray` background images for placing the preview images onto
 816         backgrounds: list
 817             List of :class:`numpy.ndarray` preview images for placing onto the background images
 818 
 819         Returns
 820         -------
 821         :class:`numpy.ndarray`
 822             The preview images compiled into the full frame size for each preview
 823         """
 824         offset = (backgrounds.shape[1] - foregrounds.shape[1]) // 2
 825         new_images = list()
 826         for idx, img in enumerate(backgrounds):
 827             img[offset:offset + foregrounds[idx].shape[0],
 828                 offset:offset + foregrounds[idx].shape[1], :3] = foregrounds[idx]
 829             new_images.append(img)
 830         retval = np.array(new_images)
 831         logger.debug("Overlayed foreground. Shape: %s", retval.shape)
 832         return retval

 834     def _get_headers(self, side, width):
 835         """ Set header row for the final preview frame
 836 
 837         Parameters
 838         ----------
 839         side: {"a" or "b"}
 840             The side that the headers should be generated for
 841         width: int
 842             The width of each column in the preview frame
 843 
 844         Returns
 845         -------
 846         :class:`numpy.ndarray`
 847             The column headings for the given side
 848         """
 849         logger.debug("side: '%s', width: %s",
 850                      side, width)
 851         titles = ("Original", "Swap") if side == "a" else ("Swap", "Original")
 852         side = side.upper()
 853         height = int(64 * self._scaling)
 854         total_width = width * 3
 855         logger.debug("height: %s, total_width: %s", height, total_width)
 856         font = cv2.FONT_HERSHEY_SIMPLEX
 857         texts = ["{} ({})".format(titles[0], side),
 858                  "{0} > {0}".format(titles[0]),
 859                  "{} > {}".format(titles[0], titles[1])]
 860         text_sizes = [cv2.getTextSize(texts[idx], font, self._scaling * 0.8, 1)[0]
 861                       for idx in range(len(texts))]
 862         text_y = int((height + text_sizes[0][1]) / 2)
 863         text_x = [int((width - text_sizes[idx][0]) / 2) + width * idx
 864                   for idx in range(len(texts))]
 865         logger.debug("texts: %s, text_sizes: %s, text_x: %s, text_y: %s",
 866                      texts, text_sizes, text_x, text_y)
 867         header_box = np.ones((height, total_width, 3), np.float32)
 868         for idx, text in enumerate(texts):
 869             cv2.putText(header_box,
 870                         text,
 871                         (text_x[idx], text_y),
 872                         font,
 873                         self._scaling * 0.8,
 874                         (0, 0, 0),
 875                         1,
 876                         lineType=cv2.LINE_AA)
 877         logger.debug("header_box.shape: %s", header_box.shape)
 878         return header_box

 880     @staticmethod
 881     def _duplicate_headers(headers, columns):
 882         """ Duplicate headers for the number of columns displayed for each side.
 883 
 884         Parameters
 885         ----------
 886         headers: :class:`numpy.ndarray`
 887             The header to be duplicated
 888         columns: int
 889             The number of columns that the header needs to be duplicated for
 890 
 891         Returns
 892         -------
 893         :class:`numpy.ndarray`
 894             The original headers duplicated by the number of columns
 895         """
 896         for side, header in headers.items():
 897             duped = tuple([header for _ in range(columns)])
 898             headers[side] = np.concatenate(duped, axis=1)
 899             logger.debug("side: %s header.shape: %s", side, header.shape)
 900         return headers

 907     ----------

 910     use_mask: bool
 911         ``True`` if a mask should be displayed otherwise ``False``

 913         Ratio of face to be cropped out of the training image.
 914     scaling: float, optional
 915         The amount to scale the final preview image by. Default: `1.0`
 916     image_count: int
 917         The number of preview images to be displayed in the time-lapse

 923         logger.debug("Initializing %s: model: %s, use_mask: %s, coverage_ratio: %s, "
 924                      "image_count: %s, batchers: '%s')", self.__class__.__name__, model,
 925                      use_mask, coverage_ratio, image_count, batchers)

 927         self._samples = Samples(model, use_mask, coverage_ratio)

 931         logger.debug("Initialized %s", self.__class__.__name__)

 935

 937         ----------

 939             The side that the time-lapse is being generated for
 940         timelapse_kwargs: dict
 941             The keyword arguments for setting up the time-lapse. All values should be full paths
 942             the keys being `input_a`, `input_b`, `output`
 943         """
 944         logger.debug("Getting time-lapse samples: '%s'", side)
 945         if not self._output_file:
 946             self._setup(**timelapse_kwargs)

 948         logger.debug("Got time-lapse samples: '%s' - %s", side, len(self._samples.images[side]))

 950     def _setup(self, input_a=None, input_b=None, output=None):
 951         """ Setup the time-lapse folder locations and the time-lapse feed.
 952 
 953         Parameters
 954         ----------
 955         input_a: str
 956             The full path to the time-lapse input folder containing faces for the "a" side
 957         input_b: str
 958             The full path to the time-lapse input folder containing faces for the "b" side
 959         output: str, optional
 960             The full path to the time-lapse output folder. If ``None`` is provided this will
 961             default to the model folder
 962         """
 963         logger.debug("Setting up time-lapse")
 964         if output is None:
 965             output = str(get_folder(os.path.join(str(self._model.model_dir),
 966                                                  "{}_timelapse".format(self._model.name))))

 971         batchsize = min(len(images["a"]),
 972                         len(images["b"]),
 973                         self._num_images)

1175         -------
1176         dict
1177             For keys "a" and "b" values are a dict with the key being the sha1 hash of the face
1178             and the value being the corresponding :class:`lib.faces_detect.DetectedFace` object.
1179         """
1180         logger.debug("Loading alignments")
1181         retval = dict()
1182         for side, fullpath in self._training_opts["alignments"].items():
1183             logger.debug("side: '%s', path: '%s'", side, fullpath)
1184             path, filename = os.path.split(fullpath)
1185             alignments = Alignments(path, filename=filename)
1186             retval[side] = self._to_detected_faces(alignments, side)
1187         logger.debug("Returning: %s", {k: len(v) for k, v in retval.items()})
1188         return retval
1189 
1190     def _to_detected_faces(self, alignments, side):
1191         """ Convert alignments to DetectedFace objects.
1192 
1193         Filter the detected faces to only those that exist in the training folders.
1194 
1195         Parameters
1196         ----------
1197         alignments: :class:`lib.alignments.Alignments`
1198             The alignments for the current faces
1199         side: {"a" or "b"}
1200             The side being processed
1201 
1202         Returns
1203         -------
1204         dict
1205             key is sha1 hash of face, value is the corresponding
1206             :class:`lib.faces_detect.DetectedFace` object
1207         """
1208         skip_count = 0
1209         dupe_count = 0
1210         side_hashes = set(self._hashes[side])
1211         detected_faces = dict()
1212         for _, faces, _, filename in alignments.yield_faces():
1213             for idx, face in enumerate(faces):
1214                 if face["hash"] in detected_faces:
1215                     dupe_count += 1
1216                     logger.debug("Face already exists, skipping: '%s'", filename)
1217                 if not self._validate_face(face, filename, idx, side, side_hashes):
1218                     skip_count += 1
1219                     continue
1220                 detected_face = DetectedFace()
1221                 detected_face.from_alignment(face)
1222                 detected_faces[face["hash"]] = detected_face
1223         logger.debug("Detected Faces count: %s, Skipped faces count: %s, duplicate faces "
1224                      "count: %s", len(detected_faces), skip_count, dupe_count)
1225         if skip_count != 0:
1226             logger.warning("%s alignments have been removed as their corresponding faces do not "
1227                            "exist in the input folder for side %s. Run in verbose mode if you "
1228                            "wish to see which alignments have been excluded.",
1229                            skip_count, side.upper())
1230         return detected_faces

1232     # Validation
1233     def _validate_face(self, face, filename, idx, side, side_hashes):
1234         """ Validate that the currently processing face has a corresponding hash entry and the
1235         requested mask exists
1236 
1237         Parameters
1238         ----------
1239         face: dict
1240             A face retrieved from an alignments file
1241         filename: str
1242             The original frame filename that the given face comes from
1243         idx: int
1244             The index of the face in the frame
1245         side: {'A', 'B'}
1246             The side that this face belongs to
1247         side_hashes: set
1248             A set of hashes that exist in the alignments folder for these faces
1249 
1250         Returns
1251         -------
1252         bool
1253             ``True`` if the face is valid otherwise ``False``
1254 
1255         Raises
1256         ------
1257         FaceswapError
1258             If the current face doesn't pass validation
1259         """
1260         mask_type = self._training_opts["mask_type"]
1261         if mask_type is not None and "mask" not in face:
1262             msg = ("You have selected a Mask Type in your training configuration options but at "
1263                    "least one face has no mask stored for it.\nYou should generate the required "
1264                    "masks with the Mask Tool or set the Mask Type configuration option to `none`."
1265                    "\nThe face that caused this failure was side: `{}`, frame: `{}`, index: {}. "
1266                    "However there are probably more faces without masks".format(
1267                        side.upper(), filename, idx))
1268             raise FaceswapError(msg)
1269 
1270         if mask_type is not None and mask_type not in face["mask"]:
1271             msg = ("At least one of your faces does not have the mask `{}` stored for it.\nYou "
1272                    "should run the Mask Tool to generate this mask for your faceset or "
1273                    "select a different mask in the training configuration options.\n"
1274                    "The face that caused this failure was [side: `{}`, frame: `{}`, index: {}]. "
1275                    "The masks that exist for this face are: {}.\nBe aware that there are probably "
1276                    "more faces without this Mask Type".format(
1277                        mask_type, side.upper(), filename, idx, list(face["mask"].keys())))
1278             raise FaceswapError(msg)
1279 
1280         if face["hash"] not in side_hashes:
1281             logger.verbose("Skipping alignment for non-existant face in frame '%s' index: %s",
1282                            filename, idx)
1283             return False
1284         return True

1286     def _check_all_faces(self):
1287         """ Ensure that all faces in the training folder exist in the alignments file.
1288         If not, output missing filenames
1289 
1290         Raises
1291         ------
1292         FaceswapError
1293             If there are faces in the training folder which do not exist in the alignments file
1294         """
1295         logger.debug("Checking faces exist in alignments")
1296         missing_alignments = dict()
1297         for side, train_hashes in self._hashes.items():
1298             align_hashes = set(self._detected_faces[side])
1299             if not align_hashes.issuperset(set(train_hashes)):
1300                 missing_alignments[side] = [
1301                     os.path.basename(filename)
1302                     for hsh, filenames in train_hashes.items()
1303                     for filename in filenames
1304                     if hsh not in align_hashes]
1305         if missing_alignments:
1306             msg = ("There are faces in your training folder(s) which do not exist in your "
1307                    "alignments file. Training cannot continue. See above for a full list of "
1308                    "files missing alignments.")
1309             for side, filelist in missing_alignments.items():
1310                 logger.error("Faces missing alignments for side %s: %s",
1311                              side.capitalize(), filelist)
1312             raise FaceswapError(msg)

1315     def _hash_to_filenames(self, side, face_hash):
1316         """ For a given hash return all the filenames that match for the given side.
1317 
1318         Notes
1319         -----
1320         Multiple faces can have the same hash, so this makes sure that all filenames are updated
1321         for all instances of a hash.
1322 
1323         Parameters
1324         ----------
1325         side: {"a" or "b"}
1326             The side currently being processed
1327         face_hash: str
1328             The sha1 hash of the face to obtain the filename for
1329 
1330         Returns
1331         -------
1332         list
1333             The filenames that exist for the given hash
1334         """
1335         retval = self._hashes[side][face_hash]

1337         return retval

1341     """ Stack images evenly for preview.
1342 
1343     Parameters
1344     ----------
1345     images: :class:`numpy.ndarray`
1346         The preview images to be stacked
1347 
1348     Returns
1349     -------
1350     :class:`numpy.ndarray`
1351         The stacked preview images
1352     """
1353     logger.debug("Stack images")
1354 
1355     def get_transpose_axes(num):
1356         if num % 2 == 0:
1357             logger.debug("Even number of images to stack")
1358             y_axes = list(range(1, num - 1, 2))
1359             x_axes = list(range(0, num - 1, 2))
1360         else:
1361             logger.debug("Odd number of images to stack")
1362             y_axes = list(range(0, num - 1, 2))
1363             x_axes = list(range(1, num - 1, 2))
1364         return y_axes, x_axes, [num - 1]
1365 
1366     images_shape = np.array(images.shape)
1367     new_axes = get_transpose_axes(len(images_shape))
1368     new_shape = [np.prod(images_shape[x]) for x in new_axes]
1369     logger.debug("Stacked images")
1370     return np.transpose(images, axes=np.concatenate(new_axes)).reshape(new_shape)
