  23     def set_defaults(self):
  24         """ Set the default values for config """
  25         logger.debug("Setting defaults")
  26         self.set_globals()
  27         current_dir = os.path.dirname(__file__)
  28         for dirpath, _, filenames in os.walk(current_dir):
  29             default_files = [fname for fname in filenames if fname.endswith("_defaults.py")]
  30             if not default_files:
  31                 continue
  32             base_path = os.path.dirname(os.path.realpath(sys.argv[0]))
  33             import_path = ".".join(full_path_split(dirpath.replace(base_path, ""))[1:])
  34             plugin_type = import_path.split(".")[-1]
  35             for filename in default_files:
  36                 self.load_module(filename, import_path, plugin_type)

 193     def load_module(self, filename, module_path, plugin_type):
 194         """ Load the defaults module and add defaults """
 195         logger.debug("Adding defaults: (filename: %s, module_path: %s, plugin_type: %s",
 196                      filename, module_path, plugin_type)
 197         module = os.path.splitext(filename)[0]
 198         section = ".".join((plugin_type, module.replace("_defaults", "")))
 199         logger.debug("Importing defaults module: %s.%s", module_path, module)
 200         mod = import_module("{}.{}".format(module_path, module))
 201         helptext = mod._HELPTEXT  # pylint:disable=protected-access
 202         helptext += ADDITIONAL_INFO if module_path.endswith("model") else ""
 203         self.add_section(title=section, info=helptext)
 204         for key, val in mod._DEFAULTS.items():  # pylint:disable=protected-access
 205             self.add_item(section=section, title=key, **val)
 206         logger.debug("Added defaults: %s", section)
