  81     def get_batch(self, queue):
  82         """ Get items for inputting into the aligner from the queue in batches
  83 
  84         Items are returned from the ``queue`` in batches of
  85         :attr:`~plugins.extract._base.Extractor.batchsize`
  86 
  87         Items are received as :class:`~plugins.extract.pipeline.ExtractMedia` objects and converted
  88         to ``dict`` for internal processing.
  89 
  90         To ensure consistent batch sizes for aligner the items are split into separate items for
  91         each :class:`~lib.faces_detect.DetectedFace` object.
  92 
  93         Remember to put ``'EOF'`` to the out queue after processing
  94         the final batch
  95 
  96         Outputs items in the following format. All lists are of length
  97         :attr:`~plugins.extract._base.Extractor.batchsize`:
  98 
  99         >>> {'filename': [<filenames of source frames>],
 100         >>>  'image': [<source images>],
 101         >>>  'detected_faces': [[<lib.faces_detect.DetectedFace objects]]}
 102 
 103         Parameters
 104         ----------
 105         queue : queue.Queue()
 106             The ``queue`` that the plugin will be fed from.
 107 
 108         Returns
 109         -------
 110         exhausted, bool
 111             ``True`` if queue is exhausted, ``False`` if not
 112         batch, dict
 113             A dictionary of lists of :attr:`~plugins.extract._base.Extractor.batchsize`:
 114         """
 115         exhausted = False
 116         batch = dict()
 117         idx = 0
 118         while idx < self.batchsize:
 119             item = self._collect_item(queue)
 120             if item == "EOF":
 121                 logger.trace("EOF received")
 122                 exhausted = True
 123                 break
 124             # Put frames with no faces into the out queue to keep TQDM consistent
 125             if not item.detected_faces:
 126                 self._queues["out"].put(item)
 127                 continue
 128 
 129             converted_image = item.get_image_copy(self.color_format)
 130             for f_idx, face in enumerate(item.detected_faces):
 131                 batch.setdefault("image", []).append(converted_image)
 132                 batch.setdefault("detected_faces", []).append(face)
 133                 batch.setdefault("filename", []).append(item.filename)
 134                 idx += 1
 135                 if idx == self.batchsize:
 136                     frame_faces = len(item.detected_faces)
 137                     if f_idx + 1 != frame_faces:
 138                         self._rollover = ExtractMedia(
 139                             item.filename,
 140                             item.image,
 141                             detected_faces=item.detected_faces[f_idx + 1:])
 142                         logger.trace("Rolled over %s faces of %s to next batch for '%s'",
 143                                      len(self._rollover.detected_faces), frame_faces,
 144                                      item.filename)
 145                     break
 146         if batch:
 147             logger.trace("Returning batch: %s", {k: v.shape if isinstance(v, np.ndarray) else v
 148                                                  for k, v in batch.items()})
 149         else:
 150             logger.trace(item)
 151         return exhausted, batch
 152 
 153     def _collect_item(self, queue):
 154         """ Collect the item from the :attr:`_rollover` dict or from the queue
 155             Add face count per frame to self._faces_per_filename for joining
 156             batches back up in finalize """
 157         if self._rollover is not None:
 158             logger.trace("Getting from _rollover: (filename: `%s`, faces: %s)",
 159                          self._rollover.filename, len(self._rollover.detected_faces))
 160             item = self._rollover
 161             self._rollover = None
 162         else:
 163             item = self._get_item(queue)
 164             if item != "EOF":
 165                 logger.trace("Getting from queue: (filename: %s, faces: %s)",
 166                              item.filename, len(item.detected_faces))
 167                 self._faces_per_filename[item.filename] = len(item.detected_faces)
 168         return item

 170     # <<< FINALIZE METHODS >>> #
 171     def finalize(self, batch):
 172         """ Finalize the output from Aligner
 173 
 174         This should be called as the final task of each `plugin`.
 175 
 176         Pairs the detected faces back up with their original frame before yielding each frame.
 177 
 178         Parameters
 179         ----------
 180         batch : dict
 181             The final ``dict`` from the `plugin` process. It must contain the `keys`:
 182             ``detected_faces``, ``landmarks``, ``filename``
 183 
 184         Yields
 185         ------
 186         :class:`~plugins.extract.pipeline.ExtractMedia`
 187             The :attr:`DetectedFaces` list will be populated for this class with the bounding boxes
 188             and landmarks for the detected faces found in the frame.
 189         """
 190 
 191         for face, landmarks in zip(batch["detected_faces"], batch["landmarks"]):
 192             if not isinstance(landmarks, np.ndarray):
 193                 landmarks = np.array(landmarks)
 194             face.landmarks_xy = landmarks
 195 
 196         logger.trace("Item out: %s", {key: val.shape if isinstance(val, np.ndarray) else val
 197                                       for key, val in batch.items()})
 198 
 199         for filename, face in zip(batch["filename"], batch["detected_faces"]):
 200             self._output_faces.append(face)
 201             if len(self._output_faces) != self._faces_per_filename[filename]:
 202                 continue
 203 
 204             output = self._extract_media.pop(filename)
 205             output.add_detected_faces(self._output_faces)
 206             self._output_faces = []
 207 
 208             logger.trace("Final Output: (filename: '%s', image shape: %s, detected_faces: %s, "
 209                          "item: %s)",
 210                          output.filename, output.image_shape, output.detected_faces, output)
 211             yield output

 219     # <<< FACE NORMALIZATION METHODS >>> #
 220     def _normalize_faces(self, faces):
 221         """ Normalizes the face for feeding into model
 222 
 223         The normalization method is dictated by the normalization command line argument
 224         """
 225         if self._normalize_method is None:
 226             return faces
 227         logger.trace("Normalizing faces")
 228         meth = getattr(self, "_normalize_{}".format(self._normalize_method.lower()))
 229         faces = [meth(face) for face in faces]
 230         logger.trace("Normalized faces")
 231         return faces

 233     @staticmethod
 234     def _normalize_mean(face):
 235         """ Normalize Face to the Mean """
 236         face = face / 255.0
 237         for chan in range(3):
 238             layer = face[:, :, chan]
 239             layer = (layer - layer.min()) / (layer.max() - layer.min())
 240             face[:, :, chan] = layer
 241         return face * 255.0
 242 
 243     @staticmethod
 244     def _normalize_hist(face):
 245         """ Equalize the RGB histogram channels """
 246         for chan in range(3):
 247             face[:, :, chan] = cv2.equalizeHist(face[:, :, chan])
 248         return face
 249 
 250     @staticmethod
 251     def _normalize_clahe(face):
 252         """ Perform Contrast Limited Adaptive Histogram Equalization """
 253         clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(4, 4))
 254         for chan in range(3):
 255             face[:, :, chan] = clahe.apply(face[:, :, chan])
 256         return face
