  27 import cv2
  28 import numpy as np

  33 class Align(Aligner):
  34     """ Perform transformation to align and get landmarks """
  35     def __init__(self, **kwargs):
  36         git_model_id = 1
  37         model_filename = "cnn-facial-landmark_v1.pb"
  38         super().__init__(git_model_id=git_model_id, model_filename=model_filename, **kwargs)
  39 
  40         self.name = "cv2-DNN Aligner"
  41         self.input_size = 128
  42         self.color_format = "RGB"
  43         self.vram = 0  # Doesn't use GPU
  44         self.vram_per_batch = 0
  45         self.batchsize = 1
  46 
  47     def init_model(self):
  48         """ Initialize CV2 DNN Detector Model"""
  49         self.model = cv2.dnn.readNetFromTensorflow(self.model_path)  # pylint: disable=no-member
  50         self.model.setPreferableTarget(cv2.dnn.DNN_TARGET_CPU)  # pylint: disable=no-member
  51 
  52     def process_input(self, batch):
  53         """ Compile the detected faces for prediction """
  54         faces, batch["roi"], batch["offsets"] = self.align_image(batch)
  55         faces = self._normalize_faces(faces)
  56         batch["feed"] = np.array(faces, dtype="float32")[..., :3].transpose((0, 3, 1, 2))
  57         return batch
  58 
  59     def align_image(self, batch):
  60         """ Align the incoming image for prediction """
  61         logger.trace("Aligning image around center")
  62         sizes = (self.input_size, self.input_size)
  63         rois = []
  64         faces = []
  65         offsets = []
  66         for det_face, image in zip(batch["detected_faces"], batch["image"]):
  67             box = (det_face.left,
  68                    det_face.top,
  69                    det_face.right,
  70                    det_face.bottom)
  71             diff_height_width = det_face.h - det_face.w
  72             offset_y = int(abs(diff_height_width / 2))
  73             box_moved = self.move_box(box, [0, offset_y])
  74             # Make box square.
  75             roi = self.get_square_box(box_moved)
  76 
  77             # Pad the image and adjust roi if face is outside of boundaries
  78             image, offset = self.pad_image(roi, image)
  79             face = image[roi[1] + abs(offset[1]): roi[3] + abs(offset[1]),
  80                          roi[0] + abs(offset[0]): roi[2] + abs(offset[0])]
  81             interpolation = cv2.INTER_CUBIC if face.shape[0] < self.input_size else cv2.INTER_AREA
  82             face = cv2.resize(face, dsize=sizes, interpolation=interpolation)
  83             faces.append(face)
  84             rois.append(roi)
  85             offsets.append(offset)
  86         return faces, rois, offsets
  87 
  88     @staticmethod
  89     def move_box(box, offset):
  90         """Move the box to direction specified by vector offset"""
  91         left = box[0] + offset[0]
  92         top = box[1] + offset[1]
  93         right = box[2] + offset[0]
  94         bottom = box[3] + offset[1]
  95         return [left, top, right, bottom]
  96 
  97     @staticmethod
  98     def get_square_box(box):
  99         """Get a square box out of the given box, by expanding it."""
 100         left = box[0]
 101         top = box[1]
 102         right = box[2]
 103         bottom = box[3]
 104 
 105         box_width = right - left
 106         box_height = bottom - top
 107 
 108         # Check if box is already a square. If not, make it a square.
 109         diff = box_height - box_width
 110         delta = int(abs(diff) / 2)
 111 
 112         if diff == 0:                   # Already a square.
 113             return box
 114         if diff > 0:                    # Height > width, a slim box.
 115             left -= delta
 116             right += delta
 117             if diff % 2 == 1:
 118                 right += 1
 119         else:                           # Width > height, a short box.
 120             top -= delta
 121             bottom += delta
 122             if diff % 2 == 1:
 123                 bottom += 1
 124 
 125         # Make sure box is always square.
 126         assert ((right - left) == (bottom - top)), 'Box is not square.'
 127 
 128         return [left, top, right, bottom]
 129 
 130     @staticmethod
 131     def pad_image(box, image):
 132         """Pad image if face-box falls outside of boundaries """
 133         height, width = image.shape[:2]
 134         pad_l = 1 - box[0] if box[0] < 0 else 0
 135         pad_t = 1 - box[1] if box[1] < 0 else 0
 136         pad_r = box[2] - width if box[2] > width else 0
 137         pad_b = box[3] - height if box[3] > height else 0
 138         logger.trace("Padding: (l: %s, t: %s, r: %s, b: %s)", pad_l, pad_t, pad_r, pad_b)
 139         padded_image = cv2.copyMakeBorder(image.copy(),
 140                                           pad_t,
 141                                           pad_b,
 142                                           pad_l,
 143                                           pad_r,
 144                                           cv2.BORDER_CONSTANT,
 145                                           value=(0, 0, 0))
 146         offsets = (pad_l - pad_r, pad_t - pad_b)
 147         logger.trace("image_shape: %s, Padded shape: %s, box: %s, offsets: %s",
 148                      image.shape, padded_image.shape, box, offsets)
 149         return padded_image, offsets
 150 
 151     def predict(self, batch):
 152         """ Predict the 68 point landmarks """
 153         logger.trace("Predicting Landmarks")
 154         self.model.setInput(batch["feed"])
 155         batch["prediction"] = self.model.forward()
 156         return batch
 157 
 158     def process_output(self, batch):
 159         """ Process the output from the model """
 160         self.get_pts_from_predict(batch)
 161         return batch
 162 
 163     @staticmethod
 164     def get_pts_from_predict(batch):
 165         """ Get points from predictor """
 166         for prediction, roi, offset in zip(batch["prediction"], batch["roi"], batch["offsets"]):
 167             points = np.reshape(prediction, (-1, 2))
 168             points *= (roi[2] - roi[0])
 169             points[:, 0] += (roi[0] - offset[0])
 170             points[:, 1] += (roi[1] - offset[1])
 171             batch.setdefault("landmarks", []).append(points)
 172         logger.trace("Predicted Landmarks: %s", batch["landmarks"])
