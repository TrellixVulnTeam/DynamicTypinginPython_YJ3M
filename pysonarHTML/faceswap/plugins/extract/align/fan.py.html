  15 class Align(Aligner):
  16     """ Perform transformation to align and get landmarks """
  17     def __init__(self, **kwargs):
  18         git_model_id = 9
  19         model_filename = "face-alignment-network_2d4_keras_v1.h5"
  20         super().__init__(git_model_id=git_model_id, model_filename=model_filename, **kwargs)
  21         self.name = "FAN"
  22         self.input_size = 256
  23         self.color_format = "RGB"
  24         self.vram = 2240
  25         self.vram_warnings = 512  # Will run at this with warnings
  26         self.vram_per_batch = 64
  27         self.batchsize = self.config["batch-size"]
  28         self.reference_scale = 200. / 195.

  30     def init_model(self):
  31         """ Initialize FAN model """
  32         model_kwargs = dict(custom_objects={'TorchBatchNorm2D': TorchBatchNorm2D})

  34                               self.model_path,

  43     def process_input(self, batch):
  44         """ Compile the detected faces for prediction """
  45         logger.debug("Aligning faces around center")
  46         batch["center_scale"] = self.get_center_scale(batch["detected_faces"])
  47         faces = self.crop(batch)
  48         logger.trace("Aligned image around center")
  49         faces = self._normalize_faces(faces)
  50         batch["feed"] = np.array(faces, dtype="float32")[..., :3].transpose((0, 3, 1, 2)) / 255.0
  51         return batch

  53     def get_center_scale(self, detected_faces):
  54         """ Get the center and set scale of bounding box """
  55         logger.debug("Calculating center and scale")
  56         center_scale = np.empty((len(detected_faces), 68, 3), dtype='float32')
  57         for index, face in enumerate(detected_faces):
  58             x_center = (face.left + face.right) / 2.0
  59             y_center = (face.top + face.bottom) / 2.0 - face.h * 0.12
  60             scale = (face.w + face.h) * self.reference_scale
  61             center_scale[index, :, 0] = np.full(68, x_center, dtype='float32')
  62             center_scale[index, :, 1] = np.full(68, y_center, dtype='float32')
  63             center_scale[index, :, 2] = np.full(68, scale, dtype='float32')
  64         logger.trace("Calculated center and scale: %s", center_scale)
  65         return center_scale
  66 
  67     def crop(self, batch):  # pylint:disable=too-many-locals
  68         """ Crop image around the center point """
  69         logger.debug("Cropping images")
  70         sizes = (self.input_size, self.input_size)
  71         batch_shape = batch["center_scale"].shape[:2]
  72         resolutions = np.full(batch_shape, self.input_size, dtype='float32')
  73         matrix_ones = np.ones(batch_shape + (3,), dtype='float32')
  74         matrix_size = np.full(batch_shape + (3,), self.input_size, dtype='float32')
  75         matrix_size[..., 2] = 1.0
  76         upper_left = self.transform(matrix_ones, batch["center_scale"], resolutions)
  77         bot_right = self.transform(matrix_size, batch["center_scale"], resolutions)
  78 
  79         # TODO second pass .. convert to matrix
  80         new_images = []
  81         for image, top_left, bottom_right in zip(batch["image"], upper_left, bot_right):
  82             height, width = image.shape[:2]
  83             channels = 3 if image.ndim > 2 else 1
  84             bottom_right_width, bottom_right_height = bottom_right[0].astype('int32')
  85             top_left_width, top_left_height = top_left[0].astype('int32')
  86             new_dim = (bottom_right_height - top_left_height,
  87                        bottom_right_width - top_left_width,
  88                        channels)
  89             new_img = np.empty(new_dim, dtype=np.uint8)
  90 
  91             new_x = slice(max(0, -top_left_width), min(bottom_right_width, width) - top_left_width)
  92             new_y = slice(max(0, -top_left_height),
  93                           min(bottom_right_height, height) - top_left_height)
  94             old_x = slice(max(0, top_left_width), min(bottom_right_width, width))
  95             old_y = slice(max(0, top_left_height), min(bottom_right_height, height))
  96             new_img[new_y, new_x] = image[old_y, old_x]
  97 
  98             interp = cv2.INTER_CUBIC if new_dim[0] < self.input_size else cv2.INTER_AREA
  99             new_images.append(cv2.resize(new_img, dsize=sizes, interpolation=interp))
 100         logger.trace("Cropped images")
 101         return new_images

 103     @staticmethod
 104     def transform(points, center_scales, resolutions):
 105         """ Transform Image """
 106         logger.debug("Transforming Points")
 107         num_images, num_landmarks = points.shape[:2]
 108         transform_matrix = np.eye(3, dtype='float32')
 109         transform_matrix = np.repeat(transform_matrix[None, :], num_landmarks, axis=0)
 110         transform_matrix = np.repeat(transform_matrix[None, :, :], num_images, axis=0)
 111         scales = center_scales[:, :, 2] / resolutions
 112         translations = center_scales[..., 2:3] * -0.5 + center_scales[..., :2]
 113         transform_matrix[:, :, 0, 0] = scales  # x scale
 114         transform_matrix[:, :, 1, 1] = scales  # y scale
 115         transform_matrix[:, :, 0, 2] = translations[:, :, 0]  # x translation
 116         transform_matrix[:, :, 1, 2] = translations[:, :, 1]  # y translation
 117         new_points = np.einsum('abij, abj -> abi', transform_matrix, points, optimize='greedy')
 118         retval = new_points[:, :, :2].astype('float32')
 119         logger.trace("Transformed Points: %s", retval)
 120         return retval

 134     def get_pts_from_predict(self, batch):
 135         """ Get points from predictor """
 136         logger.debug("Obtain points from prediction")
 137         num_images, num_landmarks, height, width = batch["prediction"].shape
 138         image_slice = np.repeat(np.arange(num_images)[:, None], num_landmarks, axis=1)
 139         landmark_slice = np.repeat(np.arange(num_landmarks)[None, :], num_images, axis=0)
 140         resolution = np.full((num_images, num_landmarks), 64, dtype='int32')
 141         subpixel_landmarks = np.ones((num_images, num_landmarks, 3), dtype='float32')
 142 
 143         flat_indices = batch["prediction"].reshape(num_images, num_landmarks, -1).argmax(-1)
 144         indices = np.array(np.unravel_index(flat_indices, (height, width)))
 145         min_clipped = np.minimum(indices + 1, height - 1)
 146         max_clipped = np.maximum(indices - 1, 0)
 147         offsets = [(image_slice, landmark_slice, indices[0], min_clipped[1]),
 148                    (image_slice, landmark_slice, indices[0], max_clipped[1]),
 149                    (image_slice, landmark_slice, min_clipped[0], indices[1]),
 150                    (image_slice, landmark_slice, max_clipped[0], indices[1])]
 151         x_subpixel_shift = batch["prediction"][offsets[0]] - batch["prediction"][offsets[1]]
 152         y_subpixel_shift = batch["prediction"][offsets[2]] - batch["prediction"][offsets[3]]
 153         # TODO improve rudimentary sub-pixel logic to centroid of 3x3 window algorithm
 154         subpixel_landmarks[:, :, 0] = indices[1] + np.sign(x_subpixel_shift) * 0.25 + 0.5
 155         subpixel_landmarks[:, :, 1] = indices[0] + np.sign(y_subpixel_shift) * 0.25 + 0.5
 156 
 157         batch["landmarks"] = self.transform(subpixel_landmarks, batch["center_scale"], resolution)
 158         logger.trace("Obtained points from prediction: %s", batch["landmarks"])

 161 class TorchBatchNorm2D(keras.engine.base_layer.Layer):
 162     # pylint:disable=too-many-instance-attributes
 163     """" Required for FAN_keras model """
 164     def __init__(self, axis=-1, momentum=0.99, epsilon=1e-3, **kwargs):
 165         super(TorchBatchNorm2D, self).__init__(**kwargs)
 166         self.supports_masking = True
 167         self.axis = axis
 168         self.momentum = momentum
 169         self.epsilon = epsilon
 170         self._epsilon_const = K.constant(self.epsilon, dtype='float32')
 171 
 172         self.built = False
 173         self.gamma = None
 174         self.beta = None
 175         self.moving_mean = None
 176         self.moving_variance = None
 177 
 178     def build(self, input_shape):
 179         dim = input_shape[self.axis]
 180         if dim is None:
 181             raise ValueError("Axis {} of input tensor should have a "
 182                              "defined dimension but the layer received "
 183                              "an input with  shape {}."
 184                              .format(str(self.axis), str(input_shape)))
 185         shape = (dim,)
 186         self.gamma = self.add_weight(shape=shape,
 187                                      name='gamma',
 188                                      initializer='ones',
 189                                      regularizer=None,
 190                                      constraint=None)
 191         self.beta = self.add_weight(shape=shape,
 192                                     name='beta',
 193                                     initializer='zeros',
 194                                     regularizer=None,
 195                                     constraint=None)
 196         self.moving_mean = self.add_weight(shape=shape,
 197                                            name='moving_mean',
 198                                            initializer='zeros',
 199                                            trainable=False)
 200         self.moving_variance = self.add_weight(shape=shape,
 201                                                name='moving_variance',
 202                                                initializer='ones',
 203                                                trainable=False)
 204         self.built = True
 205 
 206     def call(self, inputs, **kwargs):
 207         input_shape = K.int_shape(inputs)
 208 
 209         broadcast_shape = [1] * len(input_shape)
 210         broadcast_shape[self.axis] = input_shape[self.axis]
 211 
 212         broadcast_moving_mean = K.reshape(self.moving_mean, broadcast_shape)
 213         broadcast_moving_variance = K.reshape(self.moving_variance,
 214                                               broadcast_shape)
 215         broadcast_gamma = K.reshape(self.gamma, broadcast_shape)
 216         broadcast_beta = K.reshape(self.beta, broadcast_shape)
 217         invstd = (
 218             K.ones(shape=broadcast_shape, dtype='float32')
 219             / K.sqrt(broadcast_moving_variance + self._epsilon_const)
 220         )
 221 
 222         return((inputs - broadcast_moving_mean)
 223                * invstd
 224                * broadcast_gamma
 225                + broadcast_beta)
 226 
 227     def get_config(self):
 228         config = {'axis': self.axis,
 229                   'momentum': self.momentum,
 230                   'epsilon': self.epsilon}
 231         base_config = super(TorchBatchNorm2D, self).get_config()
 232         return dict(list(base_config.items()) + list(config.items()))
