   9 class Mask(Masker):
  10     """ Perform transformation to align and get landmarks """
  11     def __init__(self, **kwargs):
  12         git_model_id = None
  13         model_filename = None
  14         super().__init__(git_model_id=git_model_id, model_filename=model_filename, **kwargs)
  15         self.input_size = 256
  16         self.name = "Components"

  18         self.vram_per_batch = 0

  22         logger.debug("No mask model to initialize")

  24     def process_input(self, batch):
  25         """ Compile the detected faces for prediction """
  26         batch["feed"] = np.zeros((self.batchsize, self.input_size, self.input_size, 1),
  27                                  dtype="float32")
  28         return batch
  29 
  30     def predict(self, batch):
  31         """ Run model to get predictions """
  32         for mask, face in zip(batch["feed"], batch["detected_faces"]):
  33             parts = self.parse_parts(np.array(face.feed_landmarks))
  34             for item in parts:
  35                 item = np.rint(np.concatenate(item)).astype("int32")
  36                 hull = cv2.convexHull(item)
  37                 cv2.fillConvexPoly(mask, hull, 1.0, lineType=cv2.LINE_AA)
  38         batch["prediction"] = batch["feed"]
  39         return batch
  40 
  41     def process_output(self, batch):
  42         """ Compile found faces for output """
  43         return batch
  44 
  45     @staticmethod
  46     def parse_parts(landmarks):
  47         """ Component face hull mask """
  48         r_jaw = (landmarks[0:9], landmarks[17:18])
  49         l_jaw = (landmarks[8:17], landmarks[26:27])
  50         r_cheek = (landmarks[17:20], landmarks[8:9])
  51         l_cheek = (landmarks[24:27], landmarks[8:9])
  52         nose_ridge = (landmarks[19:25], landmarks[8:9],)
  53         r_eye = (landmarks[17:22],
  54                  landmarks[27:28],
  55                  landmarks[31:36],
  56                  landmarks[8:9])
  57         l_eye = (landmarks[22:27],
  58                  landmarks[27:28],
  59                  landmarks[31:36],
  60                  landmarks[8:9])
  61         nose = (landmarks[27:31], landmarks[31:36])
  62         parts = [r_jaw, l_jaw, r_cheek, l_cheek, nose_ridge, r_eye, l_eye, nose]
  63         return parts
