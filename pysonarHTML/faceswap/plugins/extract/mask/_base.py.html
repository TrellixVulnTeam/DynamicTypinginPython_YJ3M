  71     def get_batch(self, queue):
  72         """ Get items for inputting into the masker from the queue in batches
  73 
  74         Items are returned from the ``queue`` in batches of
  75         :attr:`~plugins.extract._base.Extractor.batchsize`
  76 
  77         Items are received as :class:`~plugins.extract.pipeline.ExtractMedia` objects and converted
  78         to ``dict`` for internal processing.
  79 
  80         To ensure consistent batch sizes for masker the items are split into separate items for
  81         each :class:`~lib.faces_detect.DetectedFace` object.
  82 
  83         Remember to put ``'EOF'`` to the out queue after processing
  84         the final batch
  85 
  86         Outputs items in the following format. All lists are of length
  87         :attr:`~plugins.extract._base.Extractor.batchsize`:
  88 
  89         >>> {'filename': [<filenames of source frames>],
  90         >>>  'detected_faces': [[<lib.faces_detect.DetectedFace objects]]}
  91 
  92         Parameters
  93         ----------
  94         queue : queue.Queue()
  95             The ``queue`` that the plugin will be fed from.
  96 
  97         Returns
  98         -------
  99         exhausted, bool
 100             ``True`` if queue is exhausted, ``False`` if not
 101         batch, dict
 102             A dictionary of lists of :attr:`~plugins.extract._base.Extractor.batchsize`:
 103         """
 104         exhausted = False
 105         batch = dict()
 106         idx = 0
 107         while idx < self.batchsize:
 108             item = self._collect_item(queue)
 109             if item == "EOF":
 110                 logger.trace("EOF received")
 111                 exhausted = True
 112                 break
 113             # Put frames with no faces into the out queue to keep TQDM consistent
 114             if not item.detected_faces:
 115                 self._queues["out"].put(item)
 116                 continue
 117             for f_idx, face in enumerate(item.detected_faces):
 118                 face.load_feed_face(item.get_image_copy(self.color_format),
 119                                     size=self.input_size,
 120                                     coverage_ratio=1.0,
 121                                     dtype="float32",
 122                                     is_aligned_face=self._image_is_aligned)
 123 
 124                 batch.setdefault("detected_faces", []).append(face)
 125                 batch.setdefault("filename", []).append(item.filename)
 126                 idx += 1
 127                 if idx == self.batchsize:
 128                     frame_faces = len(item.detected_faces)
 129                     if f_idx + 1 != frame_faces:
 130                         self._rollover = ExtractMedia(
 131                             item.filename,
 132                             item.image,
 133                             detected_faces=item.detected_faces[f_idx + 1:])
 134                         logger.trace("Rolled over %s faces of %s to next batch for '%s'",
 135                                      len(self._rollover.detected_faces), frame_faces,
 136                                      item.filename)
 137                     break
 138         if batch:
 139             logger.trace("Returning batch: %s", {k: v.shape if isinstance(v, np.ndarray) else v
 140                                                  for k, v in batch.items()})
 141         else:
 142             logger.trace(item)
 143         return exhausted, batch

 145     def _collect_item(self, queue):
 146         """ Collect the item from the _rollover dict or from the queue
 147             Add face count per frame to self._faces_per_filename for joining
 148             batches back up in finalize """
 149         if self._rollover is not None:
 150             logger.trace("Getting from _rollover: (filename: `%s`, faces: %s)",
 151                          self._rollover.filename, len(self._rollover.detected_faces))
 152             item = self._rollover
 153             self._rollover = None
 154         else:
 155             item = self._get_item(queue)
 156             if item != "EOF":
 157                 logger.trace("Getting from queue: (filename: %s, faces: %s)",
 158                              item.filename, len(item.detected_faces))
 159                 self._faces_per_filename[item.filename] = len(item.detected_faces)
 160         return item

 166     def finalize(self, batch):
 167         """ Finalize the output from Masker
 168 
 169         This should be called as the final task of each `plugin`.
 170 
 171         Pairs the detected faces back up with their original frame before yielding each frame.
 172 
 173         Parameters
 174         ----------
 175         batch : dict
 176             The final ``dict`` from the `plugin` process. It must contain the `keys`:
 177             ``detected_faces``, ``filename``
 178 
 179         Yields
 180         ------
 181         :class:`~plugins.extract.pipeline.ExtractMedia`
 182             The :attr:`DetectedFaces` list will be populated for this class with the bounding
 183             boxes, landmarks and masks for the detected faces found in the frame.
 184         """
 185         for mask, face in zip(batch["prediction"], batch["detected_faces"]):
 186             face.add_mask(self._storage_name,
 187                           mask,
 188                           face.feed_matrix,
 189                           face.feed_interpolators[1],
 190                           storage_size=self._storage_size)
 191             face.feed = dict()
 192 
 193         logger.trace("Item out: %s", {key: val.shape if isinstance(val, np.ndarray) else val
 194                                       for key, val in batch.items()})
 195         for filename, face in zip(batch["filename"], batch["detected_faces"]):
 196             self._output_faces.append(face)
 197             if len(self._output_faces) != self._faces_per_filename[filename]:
 198                 continue
 199 
 200             output = self._extract_media.pop(filename)
 201             output.add_detected_faces(self._output_faces)
 202             self._output_faces = []
 203             logger.trace("Yielding: (filename: '%s', image: %s, detected_faces: %s)",
 204                          output.filename, output.image_shape, len(output.detected_faces))
 205             yield output

 208     @staticmethod
 209     def _resize(image, target_size):
 210         """ resize input and output of mask models appropriately """
 211         height, width, channels = image.shape
 212         image_size = max(height, width)
 213         scale = target_size / image_size
 214         if scale == 1.:
 215             return image
 216         method = cv2.INTER_CUBIC if scale > 1. else cv2.INTER_AREA  # pylint: disable=no-member
 217         resized = cv2.resize(image, (0, 0), fx=scale, fy=scale, interpolation=method)
 218         resized = resized if channels > 1 else resized[..., None]
 219         return resized
