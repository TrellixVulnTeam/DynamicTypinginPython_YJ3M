  24     def process_input(self, batch):
  25         """ Compile the detected faces for prediction """
  26         batch["feed"] = np.zeros((self.batchsize, self.input_size, self.input_size, 1),
  27                                  dtype="float32")
  28         return batch
  29 
  30     def predict(self, batch):
  31         """ Run model to get predictions """
  32         for mask, face in zip(batch["feed"], batch["detected_faces"]):
  33             parts = self.parse_parts(np.array(face.feed_landmarks))
  34             for item in parts:
  35                 item = np.rint(np.concatenate(item)).astype("int32")
  36                 hull = cv2.convexHull(item)
  37                 cv2.fillConvexPoly(mask, hull, 1.0, lineType=cv2.LINE_AA)
  38         batch["prediction"] = batch["feed"]
  39         return batch
  40 
  41     def process_output(self, batch):
  42         """ Compile found faces for output """
  43         return batch

  45     @staticmethod
  46     def parse_parts(landmarks):
  47         """ Extended face hull mask """
  48         # mid points between the side of face and eye point
  49         ml_pnt = (landmarks[36] + landmarks[0]) // 2
  50         mr_pnt = (landmarks[16] + landmarks[45]) // 2
  51 
  52         # mid points between the mid points and eye
  53         ql_pnt = (landmarks[36] + ml_pnt) // 2
  54         qr_pnt = (landmarks[45] + mr_pnt) // 2
  55 
  56         # Top of the eye arrays
  57         bot_l = np.array((ql_pnt, landmarks[36], landmarks[37], landmarks[38], landmarks[39]))
  58         bot_r = np.array((landmarks[42], landmarks[43], landmarks[44], landmarks[45], qr_pnt))
  59 
  60         # Eyebrow arrays
  61         top_l = landmarks[17:22]
  62         top_r = landmarks[22:27]
  63 
  64         # Adjust eyebrow arrays
  65         landmarks[17:22] = top_l + ((top_l - bot_l) // 2)
  66         landmarks[22:27] = top_r + ((top_r - bot_r) // 2)
  67 
  68         r_jaw = (landmarks[0:9], landmarks[17:18])
  69         l_jaw = (landmarks[8:17], landmarks[26:27])
  70         r_cheek = (landmarks[17:20], landmarks[8:9])
  71         l_cheek = (landmarks[24:27], landmarks[8:9])
  72         nose_ridge = (landmarks[19:25], landmarks[8:9],)
  73         r_eye = (landmarks[17:22],
  74                  landmarks[27:28],
  75                  landmarks[31:36],
  76                  landmarks[8:9])
  77         l_eye = (landmarks[22:27],
  78                  landmarks[27:28],
  79                  landmarks[31:36],
  80                  landmarks[8:9])
  81         nose = (landmarks[27:31], landmarks[31:36])
  82         parts = [r_jaw, l_jaw, r_cheek, l_cheek, nose_ridge, r_eye, l_eye, nose]
  83         return parts
