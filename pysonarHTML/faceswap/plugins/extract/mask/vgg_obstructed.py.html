  22 class Mask(Masker):

  24     def __init__(self, **kwargs):
  25         git_model_id = 5

  35     def init_model(self):
  36         self.model = KSession(self.name, self.model_path,
  37                               model_kwargs=dict(), allow_growth=self.config["allow_growth"])
  38         self.model.load_model()
  39         self.model.append_softmax_activation(layer_index=-1)
  40         placeholder = np.zeros((self.batchsize, self.input_size, self.input_size, 3),
  41                                dtype="float32")
  42         self.model.predict(placeholder)
  43 
  44     def process_input(self, batch):
  45         """ Compile the detected faces for prediction """
  46         input_ = [face.feed_face[..., :3] for face in batch["detected_faces"]]
  47         batch["feed"] = input_ - np.mean(input_, axis=(1, 2))[:, None, None, :]
  48         logger.trace("feed shape: %s", batch["feed"].shape)
  49         return batch
  50 
  51     def predict(self, batch):
  52         """ Run model to get predictions """
  53         predictions = self.model.predict(batch["feed"])
  54         batch["prediction"] = predictions[..., 0] * -1.0 + 1.0
  55         return batch
