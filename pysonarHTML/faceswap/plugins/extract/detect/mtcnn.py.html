  15 class Detect(Detector):
  16     """ MTCNN detector for face recognition """
  17     def __init__(self, **kwargs):
  18         git_model_id = 2
  19         model_filename = ["mtcnn_det_v2.1.h5", "mtcnn_det_v2.2.h5", "mtcnn_det_v2.3.h5"]
  20         super().__init__(git_model_id=git_model_id, model_filename=model_filename, **kwargs)
  21         self.name = "MTCNN"
  22         self.input_size = 640
  23         self.vram = 320
  24         self.vram_warnings = 64  # Will run at this with warnings
  25         self.vram_per_batch = 32
  26         self.batchsize = self.config["batch-size"]
  27         self.kwargs = self.validate_kwargs()
  28         self.color_format = "RGB"
  29 
  30     def validate_kwargs(self):
  31         """ Validate that config options are correct. If not reset to default """
  32         valid = True
  33         threshold = [self.config["threshold_1"],
  34                      self.config["threshold_2"],
  35                      self.config["threshold_3"]]
  36         kwargs = {"minsize": self.config["minsize"],
  37                   "threshold": threshold,
  38                   "factor": self.config["scalefactor"]}
  39 
  40         if kwargs["minsize"] < 10:
  41             valid = False
  42         elif not all(0.0 < threshold <= 1.0 for threshold in kwargs['threshold']):
  43             valid = False
  44         elif not 0.0 < kwargs['factor'] < 1.0:
  45             valid = False
  46 
  47         if not valid:
  48             kwargs = {"minsize": 20,  # minimum size of face
  49                       "threshold": [0.6, 0.7, 0.7],  # three steps threshold
  50                       "factor": 0.709}               # scale factor
  51             logger.warning("Invalid MTCNN options in config. Running with defaults")
  52         logger.debug("Using mtcnn kwargs: %s", kwargs)
  53         return kwargs
  54 
  55     def init_model(self):
  56         """ Initialize S3FD Model"""
  57         self.model = MTCNN(self.model_path, self.config["allow_growth"], **self.kwargs)
  58 
  59     def process_input(self, batch):
  60         """ Compile the detection image(s) for prediction """
  61         batch["feed"] = (batch["image"] - 127.5) / 127.5
  62         return batch
  63 
  64     def predict(self, batch):
  65         """ Run model to get predictions """
  66         prediction, points = self.model.detect_faces(batch["feed"])
  67         logger.trace("filename: %s, prediction: %s, mtcnn_points: %s",
  68                      batch["filename"], prediction, points)
  69         batch["prediction"], batch["mtcnn_points"] = prediction, points
  70         return batch
  71 
  72     def process_output(self, batch):
  73         """ Post process the detected faces """
  74         return batch

 107 class PNet(KSession):
 108     """ Keras PNet model for MTCNN """
 109     def __init__(self, model_path, allow_growth):
 110         super().__init__("MTCNN-PNet", model_path, allow_growth=allow_growth)
 111         self.define_model(self.model_definition)
 112         self.load_model_weights()
 113 
 114     @staticmethod
 115     def model_definition():
 116         """ Keras PNetwork for MTCNN """
 117         input_ = Input(shape=(None, None, 3))
 118         var_x = Conv2D(10, (3, 3), strides=1, padding='valid', name='conv1')(input_)
 119         var_x = PReLU(shared_axes=[1, 2], name='PReLU1')(var_x)
 120         var_x = MaxPool2D(pool_size=2)(var_x)
 121         var_x = Conv2D(16, (3, 3), strides=1, padding='valid', name='conv2')(var_x)
 122         var_x = PReLU(shared_axes=[1, 2], name='PReLU2')(var_x)
 123         var_x = Conv2D(32, (3, 3), strides=1, padding='valid', name='conv3')(var_x)
 124         var_x = PReLU(shared_axes=[1, 2], name='PReLU3')(var_x)
 125         classifier = Conv2D(2, (1, 1), activation='softmax', name='conv4-1')(var_x)
 126         bbox_regress = Conv2D(4, (1, 1), name='conv4-2')(var_x)
 127         return [input_], [classifier, bbox_regress]

 130 class RNet(KSession):
 131     """ Keras RNet model for MTCNN """
 132     def __init__(self, model_path, allow_growth):
 133         super().__init__("MTCNN-RNet", model_path, allow_growth=allow_growth)
 134         self.define_model(self.model_definition)

 136 
 137     @staticmethod
 138     def model_definition():
 139         """ Keras RNetwork for MTCNN """
 140         input_ = Input(shape=(24, 24, 3))
 141         var_x = Conv2D(28, (3, 3), strides=1, padding='valid', name='conv1')(input_)
 142         var_x = PReLU(shared_axes=[1, 2], name='prelu1')(var_x)
 143         var_x = MaxPool2D(pool_size=3, strides=2, padding='same')(var_x)
 144 
 145         var_x = Conv2D(48, (3, 3), strides=1, padding='valid', name='conv2')(var_x)
 146         var_x = PReLU(shared_axes=[1, 2], name='prelu2')(var_x)
 147         var_x = MaxPool2D(pool_size=3, strides=2)(var_x)
 148 
 149         var_x = Conv2D(64, (2, 2), strides=1, padding='valid', name='conv3')(var_x)
 150         var_x = PReLU(shared_axes=[1, 2], name='prelu3')(var_x)
 151         var_x = Permute((3, 2, 1))(var_x)
 152         var_x = Flatten()(var_x)
 153         var_x = Dense(128, name='conv4')(var_x)
 154         var_x = PReLU(name='prelu4')(var_x)
 155         classifier = Dense(2, activation='softmax', name='conv5-1')(var_x)
 156         bbox_regress = Dense(4, name='conv5-2')(var_x)
 157         return [input_], [classifier, bbox_regress]

 160 class ONet(KSession):
 161     """ Keras ONet model for MTCNN """
 162     def __init__(self, model_path, allow_growth):
 163         super().__init__("MTCNN-ONet", model_path, allow_growth=allow_growth)
 164         self.define_model(self.model_definition)
 165         self.load_model_weights()
 166 
 167     @staticmethod
 168     def model_definition():
 169         """ Keras ONetwork for MTCNN """
 170         input_ = Input(shape=(48, 48, 3))
 171         var_x = Conv2D(32, (3, 3), strides=1, padding='valid', name='conv1')(input_)
 172         var_x = PReLU(shared_axes=[1, 2], name='prelu1')(var_x)
 173         var_x = MaxPool2D(pool_size=3, strides=2, padding='same')(var_x)
 174         var_x = Conv2D(64, (3, 3), strides=1, padding='valid', name='conv2')(var_x)
 175         var_x = PReLU(shared_axes=[1, 2], name='prelu2')(var_x)
 176         var_x = MaxPool2D(pool_size=3, strides=2)(var_x)
 177         var_x = Conv2D(64, (3, 3), strides=1, padding='valid', name='conv3')(var_x)
 178         var_x = PReLU(shared_axes=[1, 2], name='prelu3')(var_x)
 179         var_x = MaxPool2D(pool_size=2)(var_x)
 180         var_x = Conv2D(128, (2, 2), strides=1, padding='valid', name='conv4')(var_x)
 181         var_x = PReLU(shared_axes=[1, 2], name='prelu4')(var_x)
 182         var_x = Permute((3, 2, 1))(var_x)
 183         var_x = Flatten()(var_x)
 184         var_x = Dense(256, name='conv5')(var_x)
 185         var_x = PReLU(name='prelu5')(var_x)
 186 
 187         classifier = Dense(2, activation='softmax', name='conv6-1')(var_x)
 188         bbox_regress = Dense(4, name='conv6-2')(var_x)
 189         landmark_regress = Dense(10, name='conv6-3')(var_x)
 190         return [input_], [classifier, bbox_regress, landmark_regress]

 197     def __init__(self, model_path, allow_growth, minsize, threshold, factor):
 198         """
 199         minsize: minimum faces' size
 200         threshold: threshold=[th1, th2, th3], th1-3 are three steps threshold
 201         factor: the factor used to create a scaling pyramid of face sizes to
 202                 detect in the image.
 203         pnet, rnet, onet: caffemodel
 204         """
 205         logger.debug("Initializing: %s: (model_path: '%s', allow_growth: %s, minsize: %s, "
 206                      "threshold: %s, factor: %s)", self.__class__.__name__, model_path,
 207                      allow_growth, minsize, threshold, factor)
 208         self.minsize = minsize
 209         self.threshold = threshold
 210         self.factor = factor
 211 
 212         self.pnet = PNet(model_path[0], allow_growth)
 213         self.rnet = RNet(model_path[1], allow_growth)
 214         self.onet = ONet(model_path[2], allow_growth)
 215         self._pnet_scales = None
 216         logger.debug("Initialized: %s", self.__class__.__name__)

 218     def detect_faces(self, batch):
 219         """Detects faces in an image, and returns bounding boxes and points for them.
 220         batch: input batch
 221         """
 222         origin_h, origin_w = batch.shape[1:3]
 223         rectangles = self.detect_pnet(batch, origin_h, origin_w)
 224         rectangles = self.detect_rnet(batch, rectangles, origin_h, origin_w)
 225         rectangles = self.detect_onet(batch, rectangles, origin_h, origin_w)
 226         ret_boxes = list()
 227         ret_points = list()
 228         for rects in rectangles:
 229             if rects:
 230                 total_boxes = np.array([result[:5] for result in rects])
 231                 points = np.array([result[5:] for result in rects]).T
 232             else:
 233                 total_boxes = np.empty((0, 9))
 234                 points = np.empty(0)
 235             ret_boxes.append(total_boxes)
 236             ret_points.append(points)
 237         return ret_boxes, ret_points

 239     def detect_pnet(self, images, height, width):
 240         # pylint: disable=too-many-locals
 241         """ first stage - fast proposal network (pnet) to obtain face candidates """
 242         if self._pnet_scales is None:
 243             self._pnet_scales = calculate_scales(height, width, self.minsize, self.factor)
 244         rectangles = [[] for _ in range(images.shape[0])]
 245         batch_items = images.shape[0]
 246         for scale in self._pnet_scales:
 247             rwidth, rheight = int(width * scale), int(height * scale)
 248             batch = np.empty((batch_items, rheight, rwidth, 3), dtype="float32")
 249             for idx in range(batch_items):
 250                 batch[idx, ...] = cv2.resize(images[idx, ...], (rwidth, rheight))
 251             output = self.pnet.predict(batch)
 252             cls_prob = output[0][..., 1]
 253             roi = output[1]
 254             out_h, out_w = cls_prob.shape[1:3]
 255             out_side = max(out_h, out_w)
 256             cls_prob = np.swapaxes(cls_prob, 1, 2)
 257             roi = np.swapaxes(roi, 1, 3)
 258             for idx in range(batch_items):
 259                 # first index 0 = class score, 1 = one hot repr
 260                 rectangle = detect_face_12net(cls_prob[idx, ...],
 261                                               roi[idx, ...],
 262                                               out_side,
 263                                               1 / scale,
 264                                               width,
 265                                               height,
 266                                               self.threshold[0])
 267                 rectangles[idx].extend(rectangle)
 268         return [nms(x, 0.7, 'iou') for x in rectangles]

 270     def detect_rnet(self, images, rectangle_batch, height, width):
 271         """ second stage - refinement of face candidates with rnet """
 272         ret = []
 273         # TODO: batching
 274         for idx, rectangles in enumerate(rectangle_batch):
 275             if not rectangles:
 276                 ret.append(list())
 277                 continue
 278             image = images[idx]
 279             crop_number = 0
 280             predict_24_batch = []
 281             for rect in rectangles:
 282                 crop_img = image[int(rect[1]):int(rect[3]), int(rect[0]):int(rect[2])]
 283                 scale_img = cv2.resize(crop_img, (24, 24))
 284                 predict_24_batch.append(scale_img)
 285                 crop_number += 1
 286             predict_24_batch = np.array(predict_24_batch)
 287             output = self.rnet.predict(predict_24_batch, batch_size=128)
 288             cls_prob = output[0]
 289             cls_prob = np.array(cls_prob)
 290             roi_prob = output[1]
 291             roi_prob = np.array(roi_prob)
 292             ret.append(filter_face_24net(
 293                 cls_prob, roi_prob, rectangles, width, height, self.threshold[1]
 294             ))
 295         return ret

 297     def detect_onet(self, images, rectangle_batch, height, width):
 298         """ third stage - further refinement and facial landmarks positions with onet """
 299         ret = list()
 300         # TODO: batching
 301         for idx, rectangles in enumerate(rectangle_batch):
 302             if not rectangles:
 303                 ret.append(list())
 304                 continue
 305             image = images[idx]
 306             crop_number = 0
 307             predict_batch = []
 308             for rect in rectangles:
 309                 crop_img = image[int(rect[1]):int(rect[3]), int(rect[0]):int(rect[2])]
 310                 scale_img = cv2.resize(crop_img, (48, 48))
 311                 predict_batch.append(scale_img)
 312                 crop_number += 1
 313             predict_batch = np.array(predict_batch)
 314             output = self.onet.predict(predict_batch, batch_size=128)
 315             cls_prob = output[0]
 316             roi_prob = output[1]
 317             pts_prob = output[2]  # index
 318             ret.append(filter_face_48net(
 319                 cls_prob,
 320                 roi_prob,
 321                 pts_prob,
 322                 rectangles,
 323                 width,
 324                 height,
 325                 self.threshold[2]
 326             ))
 327         return ret

 330 def detect_face_12net(cls_prob, roi, out_side, scale, width, height, threshold):
 331     # pylint: disable=too-many-locals, too-many-arguments
 332     """ Detect face position and calibrate bounding box on 12net feature map(matrix version)
 333     Input:
 334         cls_prob : softmax feature map for face classify
 335         roi      : feature map for regression
 336         out_side : feature map's largest size
 337         scale    : current input image scale in multi-scales
 338         width    : image's origin width
 339         height   : image's origin height
 340         threshold: 0.6 can have 99% recall rate
 341     """
 342     in_side = 2*out_side+11
 343     stride = 0
 344     if out_side != 1:
 345         stride = float(in_side-12)/(out_side-1)
 346     (var_x, var_y) = np.where(cls_prob >= threshold)
 347     boundingbox = np.array([var_x, var_y]).T
 348     bb1 = np.fix((stride * (boundingbox) + 0) * scale)
 349     bb2 = np.fix((stride * (boundingbox) + 11) * scale)
 350     boundingbox = np.concatenate((bb1, bb2), axis=1)
 351     dx_1 = roi[0][var_x, var_y]
 352     dx_2 = roi[1][var_x, var_y]
 353     dx3 = roi[2][var_x, var_y]
 354     dx4 = roi[3][var_x, var_y]
 355     score = np.array([cls_prob[var_x, var_y]]).T
 356     offset = np.array([dx_1, dx_2, dx3, dx4]).T
 357     boundingbox = boundingbox + offset*12.0*scale
 358     rectangles = np.concatenate((boundingbox, score), axis=1)
 359     rectangles = rect2square(rectangles)
 360     pick = []
 361     for rect in rectangles:
 362         x_1 = int(max(0, rect[0]))
 363         y_1 = int(max(0, rect[1]))
 364         x_2 = int(min(width, rect[2]))
 365         y_2 = int(min(height, rect[3]))
 366         sc_ = rect[4]
 367         if x_2 > x_1 and y_2 > y_1:
 368             pick.append([x_1, y_1, x_2, y_2, sc_])
 369     return nms(pick, 0.3, "iou")

 372 def filter_face_24net(cls_prob, roi, rectangles, width, height, threshold):
 373     # pylint: disable=too-many-locals, too-many-arguments
 374     """ Filter face position and calibrate bounding box on 12net's output
 375     Input:
 376         cls_prob  : softmax feature map for face classify
 377         roi_prob  : feature map for regression
 378         rectangles: 12net's predict
 379         width     : image's origin width
 380         height    : image's origin height
 381         threshold : 0.6 can have 97% recall rate
 382     Output:
 383         rectangles: possible face positions
 384     """
 385     prob = cls_prob[:, 1]
 386     pick = np.where(prob >= threshold)
 387     rectangles = np.array(rectangles)
 388     x_1 = rectangles[pick, 0]
 389     y_1 = rectangles[pick, 1]
 390     x_2 = rectangles[pick, 2]
 391     y_2 = rectangles[pick, 3]
 392     sc_ = np.array([prob[pick]]).T
 393     dx_1 = roi[pick, 0]
 394     dx_2 = roi[pick, 1]
 395     dx3 = roi[pick, 2]
 396     dx4 = roi[pick, 3]
 397     r_width = x_2-x_1
 398     r_height = y_2-y_1
 399     x_1 = np.array([(x_1 + dx_1 * r_width)[0]]).T
 400     y_1 = np.array([(y_1 + dx_2 * r_height)[0]]).T
 401     x_2 = np.array([(x_2 + dx3 * r_width)[0]]).T
 402     y_2 = np.array([(y_2 + dx4 * r_height)[0]]).T
 403     rectangles = np.concatenate((x_1, y_1, x_2, y_2, sc_), axis=1)
 404     rectangles = rect2square(rectangles)
 405     pick = []
 406     for rect in rectangles:
 407         x_1 = int(max(0, rect[0]))
 408         y_1 = int(max(0, rect[1]))
 409         x_2 = int(min(width, rect[2]))
 410         y_2 = int(min(height, rect[3]))
 411         sc_ = rect[4]
 412         if x_2 > x_1 and y_2 > y_1:
 413             pick.append([x_1, y_1, x_2, y_2, sc_])
 414     return nms(pick, 0.3, 'iou')

 417 def filter_face_48net(cls_prob, roi, pts, rectangles, width, height, threshold):
 418     # pylint: disable=too-many-locals, too-many-arguments
 419     """ Filter face position and calibrate bounding box on 12net's output
 420     Input:
 421         cls_prob  : cls_prob[1] is face possibility
 422         roi       : roi offset
 423         pts       : 5 landmark
 424         rectangles: 12net's predict, rectangles[i][0:3] is the position, rectangles[i][4] is score
 425         width     : image's origin width
 426         height    : image's origin height
 427         threshold : 0.7 can have 94% recall rate on CelebA-database
 428     Output:
 429         rectangles: face positions and landmarks
 430     """
 431     prob = cls_prob[:, 1]
 432     pick = np.where(prob >= threshold)
 433     rectangles = np.array(rectangles)
 434     x_1 = rectangles[pick, 0]
 435     y_1 = rectangles[pick, 1]
 436     x_2 = rectangles[pick, 2]
 437     y_2 = rectangles[pick, 3]
 438     sc_ = np.array([prob[pick]]).T
 439     dx_1 = roi[pick, 0]
 440     dx_2 = roi[pick, 1]
 441     dx3 = roi[pick, 2]
 442     dx4 = roi[pick, 3]
 443     r_width = x_2-x_1
 444     r_height = y_2-y_1
 445     pts0 = np.array([(r_width * pts[pick, 0] + x_1)[0]]).T
 446     pts1 = np.array([(r_height * pts[pick, 5] + y_1)[0]]).T
 447     pts2 = np.array([(r_width * pts[pick, 1] + x_1)[0]]).T
 448     pts3 = np.array([(r_height * pts[pick, 6] + y_1)[0]]).T
 449     pts4 = np.array([(r_width * pts[pick, 2] + x_1)[0]]).T
 450     pts5 = np.array([(r_height * pts[pick, 7] + y_1)[0]]).T
 451     pts6 = np.array([(r_width * pts[pick, 3] + x_1)[0]]).T
 452     pts7 = np.array([(r_height * pts[pick, 8] + y_1)[0]]).T
 453     pts8 = np.array([(r_width * pts[pick, 4] + x_1)[0]]).T
 454     pts9 = np.array([(r_height * pts[pick, 9] + y_1)[0]]).T
 455     x_1 = np.array([(x_1 + dx_1 * r_width)[0]]).T
 456     y_1 = np.array([(y_1 + dx_2 * r_height)[0]]).T
 457     x_2 = np.array([(x_2 + dx3 * r_width)[0]]).T
 458     y_2 = np.array([(y_2 + dx4 * r_height)[0]]).T
 459     rectangles = np.concatenate((x_1, y_1, x_2, y_2, sc_,
 460                                  pts0, pts1, pts2, pts3, pts4, pts5, pts6, pts7, pts8, pts9),
 461                                 axis=1)
 462     pick = []
 463     for rect in rectangles:
 464         x_1 = int(max(0, rect[0]))
 465         y_1 = int(max(0, rect[1]))
 466         x_2 = int(min(width, rect[2]))
 467         y_2 = int(min(height, rect[3]))
 468         if x_2 > x_1 and y_2 > y_1:
 469             pick.append([x_1, y_1, x_2, y_2,
 470                          rect[4], rect[5], rect[6], rect[7], rect[8], rect[9],
 471                          rect[10], rect[11], rect[12], rect[13], rect[14]])
 472     return nms(pick, 0.3, 'iom')

 475 def nms(rectangles, threshold, method):
 476     # pylint:disable=too-many-locals
 477     """ apply NMS(non-maximum suppression) on ROIs in same scale(matrix version)
 478     Input:
 479         rectangles: rectangles[i][0:3] is the position, rectangles[i][4] is score
 480     Output:
 481         rectangles: same as input
 482     """
 483     if not rectangles:
 484         return rectangles
 485     boxes = np.array(rectangles)
 486     x_1 = boxes[:, 0]
 487     y_1 = boxes[:, 1]
 488     x_2 = boxes[:, 2]
 489     y_2 = boxes[:, 3]
 490     var_s = boxes[:, 4]
 491     area = np.multiply(x_2-x_1+1, y_2-y_1+1)
 492     s_sort = np.array(var_s.argsort())
 493     pick = []
 494     while len(s_sort) > 0:
 495         # s_sort[-1] have highest prob score, s_sort[0:-1]->others
 496         xx_1 = np.maximum(x_1[s_sort[-1]], x_1[s_sort[0:-1]])
 497         yy_1 = np.maximum(y_1[s_sort[-1]], y_1[s_sort[0:-1]])
 498         xx_2 = np.minimum(x_2[s_sort[-1]], x_2[s_sort[0:-1]])
 499         yy_2 = np.minimum(y_2[s_sort[-1]], y_2[s_sort[0:-1]])
 500         width = np.maximum(0.0, xx_2 - xx_1 + 1)
 501         height = np.maximum(0.0, yy_2 - yy_1 + 1)
 502         inter = width * height
 503         if method == 'iom':
 504             var_o = inter / np.minimum(area[s_sort[-1]], area[s_sort[0:-1]])
 505         else:
 506             var_o = inter / (area[s_sort[-1]] + area[s_sort[0:-1]] - inter)
 507         pick.append(s_sort[-1])
 508         s_sort = s_sort[np.where(var_o <= threshold)[0]]
 509     result_rectangle = boxes[pick].tolist()
 510     return result_rectangle

 513 def calculate_scales(height, width, minsize, factor):
 514     """ Calculate multi-scale
 515         Input:
 516             height: Original image height
 517             width: Original image width
 518             minsize: Minimum size for a face to be accepted
 519             factor: Scaling factor
 520         Output:
 521             scales  : Multi-scale
 522     """
 523     factor_count = 0
 524     minl = np.amin([height, width])
 525     var_m = 12.0 / minsize
 526     minl = minl * var_m
 527     # create scale pyramid
 528     scales = []
 529     while minl >= 12:
 530         scales += [var_m * np.power(factor, factor_count)]
 531         minl = minl * factor
 532         factor_count += 1
 533     logger.trace(scales)
 534     return scales

 537 def rect2square(rectangles):
 538     """ change rectangles into squares (matrix version)
 539     Input:
 540         rectangles: rectangles[i][0:3] is the position, rectangles[i][4] is score
 541     Output:
 542         squares: same as input
 543     """
 544     width = rectangles[:, 2] - rectangles[:, 0]
 545     height = rectangles[:, 3] - rectangles[:, 1]
 546     length = np.maximum(width, height).T
 547     rectangles[:, 0] = rectangles[:, 0] + width * 0.5 - length * 0.5
 548     rectangles[:, 1] = rectangles[:, 1] + height * 0.5 - length * 0.5
 549     rectangles[:, 2:4] = rectangles[:, 0:2] + np.repeat([length], 2, axis=0).T
 550     return rectangles
