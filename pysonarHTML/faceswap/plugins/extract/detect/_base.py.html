  75     # <<< QUEUE METHODS >>> #
  76     def get_batch(self, queue):
  77         """ Get items for inputting to the detector plugin in batches
  78 
  79         Items are received as :class:`~plugins.extract.pipeline.ExtractMedia` objects and converted
  80         to ``dict`` for internal processing.
  81 
  82         Items are returned from the ``queue`` in batches of
  83         :attr:`~plugins.extract._base.Extractor.batchsize`
  84 
  85         Remember to put ``'EOF'`` to the out queue after processing
  86         the final batch
  87 
  88         Outputs items in the following format. All lists are of length
  89         :attr:`~plugins.extract._base.Extractor.batchsize`:
  90 
  91         >>> {'filename': [<filenames of source frames>],
  92         >>>  'image': <numpy.ndarray of images standardized for prediction>,
  93         >>>  'scale': [<scaling factors for each image>],
  94         >>>  'pad': [<padding for each image>],
  95         >>>  'detected_faces': [[<lib.faces_detect.DetectedFace objects]]}
  96 
  97         Parameters
  98         ----------
  99         queue : queue.Queue()
 100             The ``queue`` that the batch will be fed from. This will be a queue that loads
 101             images.
 102 
 103         Returns
 104         -------
 105         exhausted, bool
 106             ``True`` if queue is exhausted, ``False`` if not.
 107         batch, dict
 108             A dictionary of lists of :attr:`~plugins.extract._base.Extractor.batchsize`.
 109         """
 110         exhausted = False
 111         batch = dict()
 112         for _ in range(self.batchsize):
 113             item = self._get_item(queue)
 114             if item == "EOF":
 115                 exhausted = True
 116                 break
 117             batch.setdefault("filename", []).append(item.filename)
 118             image, scale, pad = self._compile_detection_image(item)
 119             batch.setdefault("image", []).append(image)
 120             batch.setdefault("scale", []).append(scale)
 121             batch.setdefault("pad", []).append(pad)
 122 
 123         if batch:
 124             batch["image"] = np.array(batch["image"], dtype="float32")
 125             logger.trace("Returning batch: %s", {k: v.shape if isinstance(v, np.ndarray) else v
 126                                                  for k, v in batch.items()})
 127         else:
 128             logger.trace(item)
 129         return exhausted, batch

 131     # <<< FINALIZE METHODS>>> #
 132     def finalize(self, batch):
 133         """ Finalize the output from Detector
 134 
 135         This should be called as the final task of each ``plugin``.
 136 
 137         Parameters
 138         ----------
 139         batch : dict
 140             The final ``dict`` from the `plugin` process. It must contain the keys  ``filename``,
 141             ``faces``
 142 
 143         Yields
 144         ------
 145         :class:`~plugins.extract.pipeline.ExtractMedia`
 146             The :attr:`DetectedFaces` list will be populated for this class with the bounding boxes
 147             for the detected faces found in the frame.
 148         """
 149         if not isinstance(batch, dict):
 150             logger.trace("Item out: %s", batch)
 151             return batch
 152 
 153         logger.trace("Item out: %s", {k: v.shape if isinstance(v, np.ndarray) else v
 154                                       for k, v in batch.items()})
 155 
 156         batch_faces = [[self.to_detected_face(face[0], face[1], face[2], face[3])
 157                         for face in faces]
 158                        for faces in batch["prediction"]]
 159         # Rotations
 160         if any(m.any() for m in batch["rotmat"]) and any(batch_faces):
 161             batch_faces = [[self._rotate_face(face, rotmat) if rotmat.any() else face
 162                             for face in faces]
 163                            for faces, rotmat in zip(batch_faces, batch["rotmat"])]
 164 
 165         # Remove zero sized faces
 166         batch_faces = self._remove_zero_sized_faces(batch_faces)
 167 
 168         # Scale back out to original frame
 169         batch["detected_faces"] = [[self.to_detected_face((face.left - pad[0]) / scale,
 170                                                           (face.top - pad[1]) / scale,
 171                                                           (face.right - pad[0]) / scale,
 172                                                           (face.bottom - pad[1]) / scale)
 173                                     for face in faces]
 174                                    for scale, pad, faces in zip(batch["scale"],
 175                                                                 batch["pad"],
 176                                                                 batch_faces)]
 177 
 178         if self.min_size > 0 and batch.get("detected_faces", None):
 179             batch["detected_faces"] = self._filter_small_faces(batch["detected_faces"])
 180 
 181         batch = self._dict_lists_to_list_dicts(batch)
 182         for item in batch:
 183             output = self._extract_media.pop(item["filename"])
 184             output.add_detected_faces(item["detected_faces"])
 185             logger.trace("final output: (filename: '%s', image shape: %s, detected_faces: %s, "
 186                          "item: %s", output.filename, output.image_shape, output.detected_faces,
 187                          output)
 188             yield output

 190     @staticmethod
 191     def to_detected_face(left, top, right, bottom):
 192         """ Return a :class:`~lib.faces_detect.DetectedFace` object for the bounding box """
 193         return DetectedFace(x=int(round(left)),
 194                             w=int(round(right - left)),
 195                             y=int(round(top)),
 196                             h=int(round(bottom - top)))

 199     # <<< PREDICT WRAPPER >>> #
 200     def _predict(self, batch):
 201         """ Wrap models predict function in rotations """
 202         batch["rotmat"] = [np.array([]) for _ in range(len(batch["feed"]))]
 203         found_faces = [np.array([]) for _ in range(len(batch["feed"]))]
 204         for angle in self.rotation:
 205             # Rotate the batch and insert placeholders for already found faces
 206             self._rotate_batch(batch, angle)
 207             batch = self.predict(batch)
 208 
 209             if angle != 0 and any([face.any() for face in batch["prediction"]]):
 210                 logger.verbose("found face(s) by rotating image %s degrees", angle)
 211 
 212             found_faces = [face if not found.any() else found
 213                            for face, found in zip(batch["prediction"], found_faces)]
 214 
 215             if all([face.any() for face in found_faces]):
 216                 logger.trace("Faces found for all images")
 217                 break
 218 
 219         batch["prediction"] = found_faces
 220         logger.trace("detect_prediction output: (filenames: %s, prediction: %s, rotmat: %s)",
 221                      batch["filename"], batch["prediction"], batch["rotmat"])
 222         return batch

 224     # <<< DETECTION IMAGE COMPILATION METHODS >>> #
 225     def _compile_detection_image(self, item):
 226         """ Compile the detection image for feeding into the model
 227 
 228         Parameters
 229         ----------
 230         item: :class:`plugins.extract.pipeline.ExtractMedia`
 231             The input item from the pipeline
 232         """
 233         image = item.get_image_copy(self.color_format)
 234         scale = self._set_scale(item.image_size)
 235         pad = self._set_padding(item.image_size, scale)
 236 
 237         image = self._scale_image(image, item.image_size, scale)
 238         image = self._pad_image(image)
 239         logger.trace("compiled: (images shape: %s, scale: %s, pad: %s)", image.shape, scale, pad)
 240         return image, scale, pad
 241 
 242     def _set_scale(self, image_size):
 243         """ Set the scale factor for incoming image """
 244         scale = self.input_size / max(image_size)
 245         logger.trace("Detector scale: %s", scale)
 246         return scale
 247 
 248     def _set_padding(self, image_size, scale):
 249         """ Set the image padding for non-square images """
 250         pad_left = int(self.input_size - int(image_size[1] * scale)) // 2
 251         pad_top = int(self.input_size - int(image_size[0] * scale)) // 2
 252         return pad_left, pad_top
 253 
 254     @staticmethod
 255     def _scale_image(image, image_size, scale):
 256         """ Scale the image and optional pad to given size """
 257         interpln = cv2.INTER_CUBIC if scale > 1.0 else cv2.INTER_AREA
 258         if scale != 1.0:
 259             dims = (int(image_size[1] * scale), int(image_size[0] * scale))
 260             logger.trace("Resizing detection image from %s to %s. Scale=%s",
 261                          "x".join(str(i) for i in reversed(image_size)),
 262                          "x".join(str(i) for i in dims), scale)
 263             image = cv2.resize(image, dims, interpolation=interpln)
 264         logger.trace("Resized image shape: %s", image.shape)
 265         return image
 266 
 267     def _pad_image(self, image):
 268         """ Pad a resized image to input size """
 269         height, width = image.shape[:2]
 270         if width < self.input_size or height < self.input_size:
 271             pad_l = (self.input_size - width) // 2
 272             pad_r = (self.input_size - width) - pad_l
 273             pad_t = (self.input_size - height) // 2
 274             pad_b = (self.input_size - height) - pad_t
 275             image = cv2.copyMakeBorder(image,
 276                                        pad_t,
 277                                        pad_b,
 278                                        pad_l,
 279                                        pad_r,
 280                                        cv2.BORDER_CONSTANT)
 281         logger.trace("Padded image shape: %s", image.shape)
 282         return image

 285     def _remove_zero_sized_faces(self, batch_faces):
 286         """ Remove items from batch_faces where detected face is of zero size
 287             or face falls entirely outside of image """
 288         logger.trace("Input sizes: %s", [len(face) for face in batch_faces])
 289         retval = [[face
 290                    for face in faces
 291                    if face.right > 0 and face.left < self.input_size
 292                    and face.bottom > 0 and face.top < self.input_size]
 293                   for faces in batch_faces]
 294         logger.trace("Output sizes: %s", [len(face) for face in retval])
 295         return retval

 297     def _filter_small_faces(self, detected_faces):
 298         """ Filter out any faces smaller than the min size threshold """
 299         retval = []
 300         for faces in detected_faces:
 301             this_image = []
 302             for face in faces:
 303                 face_size = (face.w ** 2 + face.h ** 2) ** 0.5
 304                 if face_size < self.min_size:
 305                     logger.debug("Removing detected face: (face_size: %s, min_size: %s",
 306                                  face_size, self.min_size)
 307                     continue
 308                 this_image.append(face)
 309             retval.append(this_image)
 310         return retval

 312     # <<< IMAGE ROTATION METHODS >>> #
 313     @staticmethod
 314     def _get_rotation_angles(rotation):
 315         """ Set the rotation angles. Includes backwards compatibility for the
 316             'on' and 'off' options:
 317                 - 'on' - increment 90 degrees
 318                 - 'off' - disable
 319                 - 0 is prepended to the list, as whatever happens, we want to
 320                   scan the image in it's upright state """
 321         rotation_angles = [0]
 322 
 323         if not rotation or rotation.lower() == "off":
 324             logger.debug("Not setting rotation angles")
 325             return rotation_angles
 326 
 327         if rotation.lower() == "on":
 328             rotation_angles.extend(range(90, 360, 90))
 329         else:
 330             passed_angles = [int(angle)
 331                              for angle in rotation.split(",")
 332                              if int(angle) != 0]
 333             if len(passed_angles) == 1:
 334                 rotation_step_size = passed_angles[0]
 335                 rotation_angles.extend(range(rotation_step_size,
 336                                              360,
 337                                              rotation_step_size))
 338             elif len(passed_angles) > 1:
 339                 rotation_angles.extend(passed_angles)
 340 
 341         logger.debug("Rotation Angles: %s", rotation_angles)
 342         return rotation_angles

 344     def _rotate_batch(self, batch, angle):
 345         """ Rotate images in a batch by given angle
 346             if any faces have already been detected for a batch, store the existing rotation
 347             matrix and replace the feed image with a placeholder """
 348         if angle == 0:
 349             # Set the initial batch so we always rotate from zero
 350             batch["initial_feed"] = batch["feed"].copy()
 351             return
 352 
 353         retval = dict()
 354         for img, faces, rotmat in zip(batch["initial_feed"], batch["prediction"], batch["rotmat"]):
 355             if faces.any():
 356                 image = np.zeros_like(img)
 357                 matrix = rotmat
 358             else:
 359                 image, matrix = self._rotate_image_by_angle(img, angle)
 360             retval.setdefault("feed", []).append(image)
 361             retval.setdefault("rotmat", []).append(matrix)
 362         batch["feed"] = np.array(retval["feed"], dtype="float32")
 363         batch["rotmat"] = retval["rotmat"]

 365     @staticmethod
 366     def _rotate_face(face, rotation_matrix):
 367         """ Rotates the detection bounding box around the given rotation matrix.
 368 
 369         Parameters
 370         ----------
 371         face: :class:`DetectedFace`
 372             A :class:`DetectedFace` containing the `x`, `w`, `y`, `h` detection bounding box
 373             points.
 374         rotation_matrix: numpy.ndarray
 375             The rotation matrix to rotate the given object by.
 376 
 377         Returns
 378         -------
 379         :class:`DetectedFace`
 380             The same class with the detection bounding box points rotated by the given matrix.
 381         """
 382         logger.trace("Rotating face: (face: %s, rotation_matrix: %s)", face, rotation_matrix)
 383         bounding_box = [[face.left, face.top],
 384                         [face.right, face.top],
 385                         [face.right, face.bottom],
 386                         [face.left, face.bottom]]
 387         rotation_matrix = cv2.invertAffineTransform(rotation_matrix)
 388 
 389         points = np.array(bounding_box, "int32")
 390         points = np.expand_dims(points, axis=0)
 391         transformed = cv2.transform(points, rotation_matrix).astype("int32")
 392         rotated = transformed.squeeze()
 393 
 394         # Bounding box should follow x, y planes, so get min/max for non-90 degree rotations
 395         pt_x = min([pnt[0] for pnt in rotated])
 396         pt_y = min([pnt[1] for pnt in rotated])
 397         pt_x1 = max([pnt[0] for pnt in rotated])
 398         pt_y1 = max([pnt[1] for pnt in rotated])
 399         width = pt_x1 - pt_x
 400         height = pt_y1 - pt_y
 401 
 402         face.x = int(pt_x)
 403         face.y = int(pt_y)
 404         face.w = int(width)
 405         face.h = int(height)
 406         return face

 408     def _rotate_image_by_angle(self, image, angle):
 409         """ Rotate an image by a given angle.
 410             From: https://stackoverflow.com/questions/22041699 """
 411 
 412         logger.trace("Rotating image: (image: %s, angle: %s)", image.shape, angle)
 413         channels_first = image.shape[0] <= 4
 414         if channels_first:
 415             image = np.moveaxis(image, 0, 2)
 416 
 417         height, width = image.shape[:2]
 418         image_center = (width/2, height/2)
 419         rotation_matrix = cv2.getRotationMatrix2D(image_center, -1.*angle, 1.)
 420         rotation_matrix[0, 2] += self.input_size / 2 - image_center[0]
 421         rotation_matrix[1, 2] += self.input_size / 2 - image_center[1]
 422         logger.trace("Rotated image: (rotation_matrix: %s", rotation_matrix)
 423         image = cv2.warpAffine(image, rotation_matrix, (self.input_size, self.input_size))
 424         if channels_first:
 425             image = np.moveaxis(image, 2, 0)
 426 
 427         return image, rotation_matrix
