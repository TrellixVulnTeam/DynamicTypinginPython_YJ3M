   9 class Detect(Detector):
  10     """ CV2 DNN detector for face recognition """
  11     def __init__(self, **kwargs):
  12         git_model_id = 4
  13         model_filename = ["resnet_ssd_v1.caffemodel", "resnet_ssd_v1.prototxt"]
  14         super().__init__(git_model_id=git_model_id, model_filename=model_filename, **kwargs)
  15         self.name = "cv2-DNN Detector"
  16         self.input_size = 300
  17         self.vram = 0  # CPU Only. Doesn't use VRAM
  18         self.vram_per_batch = 0
  19         self.batchsize = 1
  20         self.confidence = self.config["confidence"] / 100
  21 
  22     def init_model(self):
  23         """ Initialize CV2 DNN Detector Model"""
  24         self.model = cv2.dnn.readNetFromCaffe(self.model_path[1],  # pylint: disable=no-member
  25                                               self.model_path[0])
  26         self.model.setPreferableTarget(cv2.dnn.DNN_TARGET_CPU)  # pylint: disable=no-member
  27 
  28     def process_input(self, batch):
  29         """ Compile the detection image(s) for prediction """
  30         batch["feed"] = cv2.dnn.blobFromImages(batch["image"],  # pylint: disable=no-member
  31                                                scalefactor=1.0,
  32                                                size=(self.input_size, self.input_size),
  33                                                mean=[104, 117, 123],
  34                                                swapRB=False,
  35                                                crop=False)
  36         return batch
  37 
  38     def predict(self, batch):
  39         """ Run model to get predictions """
  40         self.model.setInput(batch["feed"])
  41         predictions = self.model.forward()
  42         batch["prediction"] = self.finalize_predictions(predictions)
  43         return batch
  44 
  45     def finalize_predictions(self, predictions):
  46         """ Filter faces based on confidence level """
  47         faces = list()
  48         for i in range(predictions.shape[2]):
  49             confidence = predictions[0, 0, i, 2]
  50             if confidence >= self.confidence:
  51                 logger.trace("Accepting due to confidence %s >= %s",
  52                              confidence, self.confidence)
  53                 faces.append([(predictions[0, 0, i, 3] * self.input_size),
  54                               (predictions[0, 0, i, 4] * self.input_size),
  55                               (predictions[0, 0, i, 5] * self.input_size),
  56                               (predictions[0, 0, i, 6] * self.input_size)])
  57         logger.trace("faces: %s", faces)
  58         return [np.array(faces)]
  59

  61         """ Compile found faces for output """
