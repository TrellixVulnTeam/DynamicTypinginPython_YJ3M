  19     """ S3FD detector for face recognition """
  20     def __init__(self, **kwargs):
  21         git_model_id = 11

  24         self.name = "S3FD"

  31     def init_model(self):
  32         """ Initialize S3FD Model"""
  33         confidence = self.config["confidence"] / 100
  34         model_kwargs = dict(custom_objects=dict(O2K_Add=O2K_Add,
  35                                                 O2K_Slice=O2K_Slice,

  38                                                 O2K_Pow=O2K_Pow,
  39                                                 O2K_ConstantLayer=O2K_ConstantLayer,

  42 
  43     def process_input(self, batch):
  44         """ Compile the detection image(s) for prediction """
  45         batch["feed"] = self.model.prepare_batch(batch["image"])

  47 
  48     def predict(self, batch):
  49         """ Run model to get predictions """
  50         predictions = self.model.predict(batch["feed"])
  51         batch["prediction"] = self.model.finalize_predictions(predictions)

  89 class O2K_Slice(keras.engine.Layer):
  90     def __init__(self, starts, ends, axes=None, steps=None, **kwargs):
  91         self._starts = starts
  92         self._ends = ends
  93         self._axes = axes
  94         self._steps = steps
  95         super(O2K_Slice, self).__init__(**kwargs)
  96 
  97     def get_config(self):
  98         config = super(O2K_Slice, self).get_config()
  99         config.update({
 100             'starts': self._starts, 'ends': self._ends,
 101             'axes': self._axes, 'steps': self._steps
 102         })
 103         return config
 104 
 105     def get_slices(self, ndims):
 106         axes = self._axes
 107         steps = self._steps
 108         if axes is None:
 109             axes = tuple(range(ndims))
 110         if steps is None:
 111             steps = (1,) * len(axes)
 112         assert len(axes) == len(steps) == len(self._starts) == len(self._ends)
 113         return list(zip(axes, self._starts, self._ends, steps))
 114 
 115     def compute_output_shape(self, input_shape):
 116         input_shape = list(input_shape)
 117         for ax, start, end, steps in self.get_slices(len(input_shape)):
 118             size = input_shape[ax]
 119             if ax == 0:
 120                 raise AttributeError("Can not slice batch axis.")
 121             if size is None:
 122                 if start < 0 or end < 0:
 123                     raise AttributeError("Negative slices not supported on symbolic axes")
 124                 logger.warning("Slicing symbolic axis might lead to problems.")
 125                 input_shape[ax] = (end - start) // steps
 126                 continue
 127             if start < 0:
 128                 start = size - start
 129             if end < 0:
 130                 end = size - end
 131             input_shape[ax] = (min(size, end) - start) // steps
 132         return tuple(input_shape)
 133 
 134     def call(self, x, *args):
 135         ax_map = dict((x[0], slice(*x[1:])) for x in self.get_slices(K.ndim(x)))
 136         shape = K.int_shape(x)
 137         slices = [(ax_map[a] if a in ax_map else slice(None)) for a in range(len(shape))]
 138         x = x[tuple(slices)]
 139         return x

 143     def __init__(self, axes=None, keepdims=True, **kwargs):
 144         self._axes = [axes] if isinstance(axes, int) else axes
 145         self._keepdims = bool(keepdims)
 146         super(O2K_ReduceLayer, self).__init__(**kwargs)

 148     def get_config(self):
 149         config = super(O2K_ReduceLayer, self).get_config()

 151             'axes': self._axes,

 153         })
 154         return config
 155 
 156     def compute_output_shape(self, input_shape):
 157         if self._axes is None:
 158             return (1,)*len(input_shape) if self._keepdims else tuple()
 159         ret = list(input_shape)
 160         for i in sorted(self._axes, reverse=True):
 161             if self._keepdims:
 162                 ret[i] = 1
 163             else:
 164                 ret.pop(i)
 165         return tuple(ret)

 215 class S3fd(KSession):
 216     """ Keras Network """
 217     def __init__(self, model_path, model_kwargs, allow_growth, confidence):
 218         logger.debug("Initializing: %s: (model_path: '%s', allow_growth: %s)",
 219                      self.__class__.__name__, model_path, allow_growth)
 220         super().__init__("S3FD", model_path, model_kwargs=model_kwargs, allow_growth=allow_growth)
 221         self.load_model()
 222         self.confidence = confidence
 223         self.average_img = np.array([104.0, 117.0, 123.0])
 224         logger.debug("Initialized: %s", self.__class__.__name__)
 225 
 226     def prepare_batch(self, batch):
 227         """ Prepare a batch for prediction """
 228         batch = batch - self.average_img
 229         batch = batch.transpose(0, 3, 1, 2)
 230         return batch
 231 
 232     def finalize_predictions(self, bounding_boxes_scales):
 233         """ Detect faces """
 234         ret = list()
 235         batch_size = range(bounding_boxes_scales[0].shape[0])
 236         for img in batch_size:
 237             bboxlist = [scale[img:img+1] for scale in bounding_boxes_scales]
 238             boxes = self._post_process(bboxlist)
 239             bboxlist = self._nms(boxes, 0.5)
 240             ret.append(bboxlist)
 241         return ret
 242 
 243     def _post_process(self, bboxlist):
 244         """ Perform post processing on output
 245             TODO: do this on the batch.
 246         """
 247         retval = list()
 248         for i in range(len(bboxlist) // 2):
 249             bboxlist[i * 2] = self.softmax(bboxlist[i * 2], axis=1)
 250         for i in range(len(bboxlist) // 2):
 251             ocls, oreg = bboxlist[i * 2], bboxlist[i * 2 + 1]
 252             stride = 2 ** (i + 2)    # 4,8,16,32,64,128
 253             poss = zip(*np.where(ocls[:, 1, :, :] > 0.05))
 254             for _, hindex, windex in poss:
 255                 axc, ayc = stride / 2 + windex * stride, stride / 2 + hindex * stride
 256                 score = ocls[0, 1, hindex, windex]
 257                 if score >= self.confidence:
 258                     loc = np.ascontiguousarray(oreg[0, :, hindex, windex]).reshape((1, 4))
 259                     priors = np.array([[axc / 1.0, ayc / 1.0, stride * 4 / 1.0, stride * 4 / 1.0]])
 260                     box = self.decode(loc, priors)
 261                     x_1, y_1, x_2, y_2 = box[0] * 1.0
 262                     retval.append([x_1, y_1, x_2, y_2, score])
 263         return_numpy = np.array(retval) if len(retval) != 0 else np.zeros((1, 5))
 264         return return_numpy
 265 
 266     @staticmethod
 267     def softmax(inp, axis):
 268         """Compute softmax values for each sets of scores in x."""
 269         return np.exp(inp - logsumexp(inp, axis=axis, keepdims=True))
 270 
 271     @staticmethod
 272     def decode(loc, priors):
 273         """Decode locations from predictions using priors to undo
 274         the encoding we did for offset regression at train time.
 275         Args:
 276             loc (tensor): location predictions for loc layers,
 277                 Shape: [num_priors,4]
 278             priors (tensor): Prior boxes in center-offset form.
 279                 Shape: [num_priors,4].
 280             variances: (list[float]) Variances of prior boxes
 281         Return:
 282             decoded bounding box predictions
 283         """
 284         variances = [0.1, 0.2]
 285         boxes = np.concatenate((priors[:, :2] + loc[:, :2] * variances[0] * priors[:, 2:],
 286                                 priors[:, 2:] * np.exp(loc[:, 2:] * variances[1])), axis=1)
 287         boxes[:, :2] -= boxes[:, 2:] / 2
 288         boxes[:, 2:] += boxes[:, :2]
 289         return boxes
 290 
 291     @staticmethod
 292     def _nms(boxes, threshold):
 293         """ Perform Non-Maximum Suppression """
 294         retained_box_indices = list()
 295 
 296         areas = (boxes[:, 2] - boxes[:, 0] + 1) * (boxes[:, 3] - boxes[:, 1] + 1)
 297         ranked_indices = boxes[:, 4].argsort()[::-1]
 298         while ranked_indices.size > 0:
 299             best = ranked_indices[0]
 300             rest = ranked_indices[1:]
 301 
 302             max_of_xy = np.maximum(boxes[best, :2], boxes[rest, :2])
 303             min_of_xy = np.minimum(boxes[best, 2:4], boxes[rest, 2:4])
 304             width_height = np.maximum(0, min_of_xy - max_of_xy + 1)
 305             intersection_areas = width_height[:, 0] * width_height[:, 1]
 306             iou = intersection_areas / (areas[best] + areas[rest] - intersection_areas)
 307 
 308             overlapping_boxes = (iou > threshold).nonzero()[0]
 309             if len(overlapping_boxes) != 0:
 310                 overlap_set = ranked_indices[overlapping_boxes + 1]
 311                 vote = np.average(boxes[overlap_set, :4], axis=0, weights=boxes[overlap_set, 4])
 312                 boxes[best, :4] = vote
 313             retained_box_indices.append(best)
 314 
 315             non_overlapping_boxes = (iou <= threshold).nonzero()[0]
 316             ranked_indices = ranked_indices[non_overlapping_boxes + 1]
 317         return boxes[retained_box_indices]
