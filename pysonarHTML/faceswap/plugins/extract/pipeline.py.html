  33 class Extractor():
  34     """ Creates a :mod:`~plugins.extract.detect`/:mod:`~plugins.extract.align``/\
  35     :mod:`~plugins.extract.mask` pipeline and yields results frame by frame from the
  36     :attr:`detected_faces` generator
  37 
  38     :attr:`input_queue` is dynamically set depending on the current :attr:`phase` of extraction
  39 
  40     Parameters
  41     ----------
  42     detector: str
  43         The name of a detector plugin as exists in :mod:`plugins.extract.detect`
  44     aligner: str
  45         The name of an aligner plugin as exists in :mod:`plugins.extract.align`
  46     masker: str or list
  47         The name of a masker plugin(s) as exists in :mod:`plugins.extract.mask`.
  48         This can be a single masker or a list of multiple maskers
  49     configfile: str, optional
  50         The path to a custom ``extract.ini`` configfile. If ``None`` then the system
  51         :file:`config/extract.ini` file will be used.
  52     multiprocess: bool, optional
  53         Whether to attempt processing the plugins in parallel. This may get overridden
  54         internally depending on the plugin combination. Default: ``False``
  55     rotate_images: str, optional
  56         Used to set the :attr:`plugins.extract.detect.rotation` attribute. Pass in a single number
  57         to use increments of that size up to 360, or pass in a ``list`` of ``ints`` to enumerate
  58         exactly what angles to check. Can also pass in ``'on'`` to increment at 90 degree
  59         intervals. Default: ``None``
  60     min_size: int, optional
  61         Used to set the :attr:`plugins.extract.detect.min_size` attribute Filters out faces
  62         detected below this size. Length, in pixels across the diagonal of the bounding box. Set
  63         to ``0`` for off. Default: ``0``
  64     normalize_method: {`None`, 'clahe', 'hist', 'mean'}, optional
  65         Used to set the :attr:`plugins.extract.align.normalize_method` attribute. Normalize the
  66         images fed to the aligner.Default: ``None``
  67     image_is_aligned: bool, optional
  68         Used to set the :attr:`plugins.extract.mask.image_is_aligned` attribute. Indicates to the
  69         masker that the fed in image is an aligned face rather than a frame. Default: ``False``
  70 
  71     Attributes
  72     ----------
  73     phase: str
  74         The current phase that the pipeline is running. Used in conjunction with :attr:`passes` and
  75         :attr:`final_pass` to indicate to the caller which phase is being processed
  76     """
  77     def __init__(self, detector, aligner, masker, configfile=None,
  78                  multiprocess=False, rotate_images=None, min_size=20,
  79                  normalize_method=None, image_is_aligned=False):
  80         logger.debug("Initializing %s: (detector: %s, aligner: %s, masker: %s, "
  81                      "configfile: %s, multiprocess: %s, rotate_images: %s, min_size: %s, "
  82                      "normalize_method: %s, image_is_aligned: %s)",
  83                      self.__class__.__name__, detector, aligner, masker, configfile,
  84                      multiprocess, rotate_images, min_size, normalize_method, image_is_aligned)
  85         self._instance = _get_instance()
  86         masker = [masker] if not isinstance(masker, list) else masker
  87         self._flow = self._set_flow(detector, aligner, masker)
  88         # We only ever need 1 item in each queue. This is 2 items cached (1 in queue 1 waiting
  89         # for queue) at each point. Adding more just stacks RAM with no speed benefit.
  90         self._queue_size = 1
  91         self._vram_stats = self._get_vram_stats()
  92         self._detect = self._load_detect(detector, rotate_images, min_size, configfile)
  93         self._align = self._load_align(aligner, configfile, normalize_method)
  94         self._mask = [self._load_mask(mask, image_is_aligned, configfile) for mask in masker]
  95         self._is_parallel = self._set_parallel_processing(multiprocess)
  96         self._phases = self._set_phases(multiprocess)
  97         self._phase_index = 0
  98         self._set_extractor_batchsize()
  99         self._queues = self._add_queues()
 100         logger.debug("Initialized %s", self.__class__.__name__)

 103     def input_queue(self):
 104         """ queue: Return the correct input queue depending on the current phase
 105 
 106         The input queue is the entry point into the extraction pipeline. An :class:`ExtractMedia`
 107         object should be put to the queue.
 108 
 109         For detect/single phase operations the :attr:`ExtractMedia.filename` and
 110         :attr:`~ExtractMedia.image` attributes should be populated.
 111 
 112         For align/mask (2nd/3rd pass operations) the :attr:`ExtractMedia.detected_faces` should
 113         also be populated by calling :func:`ExtractMedia.set_detected_faces`.
 114         """
 115         qname = "extract{}_{}_in".format(self._instance, self._current_phase[0])
 116         retval = self._queues[qname]
 117         logger.trace("%s: %s", qname, retval)
 118         return retval

 122         """ int: Returns the total number of passes the extractor needs to make.
 123 
 124         This is calculated on several factors (vram available, plugin choice,
 125         :attr:`multiprocess` etc.). It is useful for iterating over the pipeline

 131         >>>     if phase == 1:

 189     def launch(self):
 190         """ Launches the plugin(s)
 191 
 192         This launches the plugins held in the pipeline, and should be called at the beginning
 193         of each :attr:`phase`. To ensure VRAM is conserved, It will only launch the plugin(s)
 194         required for the currently running phase
 195 
 196         Example
 197         -------
 198         >>> for phase in extractor.passes:
 199         >>>     extractor.launch():
 200         >>>         <do processing>
 201         """
 202         for phase in self._current_phase:
 203             self._launch_plugin(phase)

 205     def detected_faces(self):
 206         """ Generator that returns results, frame by frame from the extraction pipeline
 207 
 208         This is the exit point for the extraction pipeline and is used to obtain the output
 209         of any pipeline :attr:`phase`
 210 
 211         Yields
 212         ------
 213         faces: :class:`ExtractMedia`
 214             The populated extracted media object.
 215 
 216         Example
 217         -------
 218         >>> for extract_media in extractor.detected_faces():
 219         >>>     filename = extract_media.filename
 220         >>>     image = extract_media.image
 221         >>>     detected_faces = extract_media.detected_faces
 222         """
 223         logger.debug("Running Detection. Phase: '%s'", self._current_phase)
 224         # If not multiprocessing, intercept the align in queue for
 225         # detection phase
 226         out_queue = self._output_queue
 227         while True:
 228             try:
 229                 if self._check_and_raise_error():
 230                     break
 231                 faces = out_queue.get(True, 1)
 232                 if faces == "EOF":
 233                     break
 234             except QueueEmpty:
 235                 continue
 236             yield faces
 237 
 238         self._join_threads()
 239         if self.final_pass:
 240             # Cleanup queues
 241             for q_name in self._queues:
 242                 queue_manager.del_queue(q_name)
 243             logger.debug("Detection Complete")
 244         else:
 245             self._phase_index += 1
 246             logger.debug("Switching to phase: %s", self._current_phase)

 249     @property
 250     def _parallel_scaling(self):
 251         """ dict: key is number of parallel plugins being loaded, value is the scaling factor that
 252         the total base vram for those plugins should be scaled by
 253 
 254         Notes
 255         -----
 256         VRAM for parallel plugins does not stack in a linear manner. Calculating the precise
 257         scaling for any given plugin combination is non trivial, however the following are
 258         calculations based on running 2-5 plugins in parallel using s3fd, fan, unet, vgg-clear
 259         and vgg-obstructed. The worst ratio is selected for each combination, plus a little extra
 260         to ensure that vram is not used up.
 261 
 262         If OOM errors are being reported, then these ratios should be relaxed some more
 263         """
 264         retval = {0: 1.0,
 265                   1: 1.0,
 266                   2: 0.7,
 267                   3: 0.55,
 268                   4: 0.5,
 269                   5: 0.4}
 270         logger.trace(retval)
 271         return retval

 273     @property
 274     def _vram_per_phase(self):
 275         """ dict: The amount of vram required for each phase in :attr:`_flow`. """
 276         retval = dict()
 277         for phase in self._flow:
 278             plugin_type, idx = self._get_plugin_type_and_index(phase)
 279             attr = getattr(self, "_{}".format(plugin_type))
 280             attr = attr[idx] if idx is not None else attr
 281             retval[phase] = attr.vram
 282         logger.trace(retval)
 283         return retval

 285     @property
 286     def _total_vram_required(self):
 287         """ Return vram required for all phases plus the buffer """
 288         vrams = self._vram_per_phase
 289         vram_required_count = sum(1 for p in vrams.values() if p > 0)
 290         logger.debug("VRAM requirements: %s. Plugins requiring VRAM: %s",
 291                      vrams, vram_required_count)
 292         retval = (sum(vrams.values()) *
 293                   self._parallel_scaling[vram_required_count])
 294         logger.debug("Total VRAM required: %s", retval)
 295         return retval

 361     @staticmethod
 362     def _get_plugin_type_and_index(flow_phase):
 363         """ Obtain the plugin type and index for the plugin for the given flow phase.
 364 
 365         When multiple plugins for the same phase are allowed (e.g. Mask) this will return
 366         the plugin type and the index of the plugin required. If only one plugin is allowed
 367         then the plugin type will be returned and the index will be ``None``.
 368 
 369         Parameters
 370         ----------
 371         flow_phase: str
 372             The phase within :attr:`_flow` that is to have the plugin type and index returned
 373 
 374         Returns
 375         -------
 376         plugin_type: str
 377             The plugin type for the given flow phase
 378         index: int
 379             The index of this plugin type within the flow, if there are multiple plugins in use
 380             otherwise ``None`` if there is only 1 plugin in use for the given phase
 381         """
 382         idx = flow_phase.split("_")[-1]
 383         if idx.isdigit():
 384             idx = int(idx)
 385             plugin_type = "_".join(flow_phase.split("_")[:-1])
 386         else:
 387             plugin_type = flow_phase
 388             idx = None
 389         return plugin_type, idx

 391     def _add_queues(self):
 392         """ Add the required processing queues to Queue Manager """
 393         queues = dict()
 394         tasks = ["extract{}_{}_in".format(self._instance, phase) for phase in self._flow]
 395         tasks.append("extract{}_{}_out".format(self._instance, self._final_phase))
 396         for task in tasks:
 397             # Limit queue size to avoid stacking ram
 398             queue_manager.add_queue(task, maxsize=self._queue_size)
 399             queues[task] = queue_manager.get_queue(task)
 400         logger.debug("Queues: %s", queues)
 401         return queues

 403     @staticmethod
 404     def _get_vram_stats():
 405         """ Obtain statistics on available VRAM and subtract a constant buffer from available vram.
 406 
 407         Returns
 408         -------
 409         dict
 410             Statistics on available VRAM
 411         """
 412         vram_buffer = 256  # Leave a buffer for VRAM allocation
 413         gpu_stats = GPUStats()
 414         stats = gpu_stats.get_card_most_free()
 415         retval = dict(count=gpu_stats.device_count,
 416                       device=stats["device"],
 417                       vram_free=int(stats["free"] - vram_buffer),
 418                       vram_total=int(stats["total"]))
 419         logger.debug(retval)
 420         return retval

 422     def _set_parallel_processing(self, multiprocess):
 423         """ Set whether to run detect, align, and mask together or separately.
 424 
 425         Parameters
 426         ----------
 427         multiprocess: bool
 428             ``True`` if the single-process command line flag has not been set otherwise ``False``
 429         """
 430         if not multiprocess:
 431             logger.debug("Parallel processing disabled by cli.")
 432             return False
 433 
 434         if self._vram_stats["count"] == 0:
 435             logger.debug("No GPU detected. Enabling parallel processing.")
 436             return True
 437 
 438         if get_backend() == "amd":
 439             logger.debug("Parallel processing disabled by amd")
 440             return False
 441 
 442         logger.verbose("%s - %sMB free of %sMB",
 443                        self._vram_stats["device"],
 444                        self._vram_stats["vram_free"],
 445                        self._vram_stats["vram_total"])
 446         if self._vram_stats["vram_free"] <= self._total_vram_required:
 447             logger.warning("Not enough free VRAM for parallel processing. "
 448                            "Switching to serial")
 449             return False
 450         return True

 452     def _set_phases(self, multiprocess):
 453         """ If not enough VRAM is available, then chunk :attr:`_flow` up into phases that will fit
 454         into VRAM, otherwise return the single flow.
 455 
 456         Parameters
 457         ----------
 458         multiprocess: bool
 459             ``True`` if the single-process command line flag has not been set otherwise ``False``
 460 
 461         Returns
 462         -------
 463         list:
 464             The jobs to be undertaken split into phases that fit into GPU RAM
 465         """
 466         force_single_process = not multiprocess or get_backend() == "amd"
 467         phases = []
 468         current_phase = []
 469         available = self._vram_stats["vram_free"]
 470         for phase in self._flow:
 471             num_plugins = len([p for p in current_phase if self._vram_per_phase[p] > 0])
 472             num_plugins += 1 if self._vram_per_phase[phase] > 0 else 0
 473             scaling = self._parallel_scaling[num_plugins]
 474             required = sum(self._vram_per_phase[p] for p in current_phase + [phase]) * scaling
 475             logger.debug("Num plugins for phase: %s, scaling: %s, vram required: %s",
 476                          num_plugins, scaling, required)
 477             if required <= available and not force_single_process:
 478                 logger.debug("Required: %s, available: %s. Adding phase '%s' to current phase: %s",
 479                              required, available, phase, current_phase)
 480                 current_phase.append(phase)
 481             elif len(current_phase) == 0 or force_single_process:
 482                 # Amount of VRAM required to run a single plugin is greater than available. We add
 483                 # it anyway, and hope it will run with warnings, as the alternative is to not run
 484                 # at all.
 485                 # This will also run if forcing single process
 486                 logger.debug("Required: %s, available: %s. Single plugin has higher requirements "
 487                              "than available or forcing single process: '%s'",
 488                              required, available, phase)
 489                 phases.append([phase])
 490             else:
 491                 logger.debug("Required: %s, available: %s. Adding phase to flow: %s",
 492                              required, available, current_phase)
 493                 phases.append(current_phase)
 494                 current_phase = [phase]
 495         if current_phase:
 496             phases.append(current_phase)
 497         logger.debug("Total phases: %s, Phases: %s", len(phases), phases)
 498         return phases

 538     def _launch_plugin(self, phase):
 539         """ Launch an extraction plugin """
 540         logger.debug("Launching %s plugin", phase)
 541         in_qname = "extract{}_{}_in".format(self._instance, phase)
 542         if phase == self._final_phase:
 543             out_qname = "extract{}_{}_out".format(self._instance, self._final_phase)
 544         else:
 545             next_phase = self._flow[self._flow.index(phase) + 1]
 546             out_qname = "extract{}_{}_in".format(self._instance, next_phase)
 547         logger.debug("in_qname: %s, out_qname: %s", in_qname, out_qname)
 548         kwargs = dict(in_queue=self._queues[in_qname], out_queue=self._queues[out_qname])
 549 
 550         plugin_type, idx = self._get_plugin_type_and_index(phase)
 551         plugin = getattr(self, "_{}".format(plugin_type))
 552         plugin = plugin[idx] if idx is not None else plugin
 553         plugin.initialize(**kwargs)
 554         plugin.start()
 555         logger.debug("Launched %s plugin", phase)

 557     def _set_extractor_batchsize(self):
 558         """
 559         Sets the batch size of the requested plugins based on their vram, their
 560         vram_per_batch_requirements and the number of plugins being loaded in the current phase.
 561         Only adjusts if the the configured batch size requires more vram than is available. Nvidia
 562         only.
 563         """
 564         if get_backend() != "nvidia":
 565             logger.debug("Backend is not Nvidia. Not updating batchsize requirements")
 566             return
 567         if sum([plugin.vram for plugin in self._active_plugins]) == 0:
 568             logger.debug("No plugins use VRAM. Not updating batchsize requirements.")
 569             return
 570 
 571         batch_required = sum([plugin.vram_per_batch * plugin.batchsize
 572                               for plugin in self._active_plugins])
 573         gpu_plugins = [p for p in self._current_phase if self._vram_per_phase[p] > 0]
 574         plugins_required = sum([self._vram_per_phase[p]
 575                                 for p in gpu_plugins]) * self._parallel_scaling[len(gpu_plugins)]
 576         if plugins_required + batch_required <= self._vram_stats["vram_free"]:
 577             logger.debug("Plugin requirements within threshold: (plugins_required: %sMB, "
 578                          "vram_free: %sMB)", plugins_required, self._vram_stats["vram_free"])
 579             return
 580         # Hacky split across plugins that use vram
 581         available_vram = (self._vram_stats["vram_free"] - plugins_required) // len(gpu_plugins)
 582         self._set_plugin_batchsize(gpu_plugins, available_vram)
 583 
 584     def set_aligner_normalization_method(self, method):
 585         """ Change the normalization method for faces fed into the aligner.
 586 
 587         Parameters
 588         ----------
 589         method: {"none", "clahe", "hist", "mean"}
 590             The normalization method to apply to faces prior to feeding into the aligner's model
 591         """
 592         logger.debug("Setting to: '%s'", method)
 593         self._align.set_normalize_method(method)
 594 
 595     def _set_plugin_batchsize(self, gpu_plugins, available_vram):
 596         """ Set the batch size for the given plugin based on given available vram.
 597         Do not update plugins which have a vram_per_batch of 0 (CPU plugins) due to
 598         zero division error.
 599         """
 600         plugins = [self._active_plugins[idx]
 601                    for idx, plugin in enumerate(self._current_phase)
 602                    if plugin in gpu_plugins]
 603         vram_per_batch = [plugin.vram_per_batch for plugin in plugins]
 604         ratios = [vram / sum(vram_per_batch) for vram in vram_per_batch]
 605         requested_batchsizes = [plugin.batchsize for plugin in plugins]
 606         batchsizes = [min(requested, max(1, int((available_vram * ratio) / plugin.vram_per_batch)))
 607                       for ratio, plugin, requested in zip(ratios, plugins, requested_batchsizes)]
 608         remaining = available_vram - sum(batchsize * plugin.vram_per_batch
 609                                          for batchsize, plugin in zip(batchsizes, plugins))
 610         sorted_indices = [i[0] for i in sorted(enumerate(plugins),
 611                                                key=lambda x: x[1].vram_per_batch, reverse=True)]
 612 
 613         logger.debug("requested_batchsizes: %s, batchsizes: %s, remaining vram: %s",
 614                      requested_batchsizes, batchsizes, remaining)
 615 
 616         while remaining > min(plugin.vram_per_batch
 617                               for plugin in plugins) and requested_batchsizes != batchsizes:
 618             for idx in sorted_indices:
 619                 plugin = plugins[idx]
 620                 if plugin.vram_per_batch > remaining:
 621                     logger.debug("Not enough VRAM to increase batch size of %s. Required: %sMB, "
 622                                  "Available: %sMB", plugin, plugin.vram_per_batch, remaining)
 623                     continue
 624                 if plugin.batchsize == batchsizes[idx]:
 625                     logger.debug("Threshold reached for %s. Batch size: %s",
 626                                  plugin, plugin.batchsize)
 627                     continue
 628                 logger.debug("Incrementing batch size of %s to %s", plugin, batchsizes[idx] + 1)
 629                 batchsizes[idx] += 1
 630                 remaining -= plugin.vram_per_batch
 631                 logger.debug("Remaining VRAM to allocate: %sMB", remaining)
 632 
 633         if batchsizes != requested_batchsizes:
 634             text = ", ".join(["{}: {}".format(plugin.__class__.__name__, batchsize)
 635                               for plugin, batchsize in zip(plugins, batchsizes)])
 636             for plugin, batchsize in zip(plugins, batchsizes):
 637                 plugin.batchsize = batchsize
 638             logger.info("Reset batch sizes due to available VRAM: %s", text)
 639 
 640     def _join_threads(self):
 641         """ Join threads for current pass """
 642         for plugin in self._active_plugins:
 643             plugin.join()
 644 
 645     def _check_and_raise_error(self):
 646         """ Check all threads for errors and raise if one occurs """
 647         for plugin in self._active_plugins:
 648             if plugin.check_and_raise_error():
 649                 return True
 650         return False

 667     def __init__(self, filename, image, detected_faces=None):
 668         logger.trace("Initializing %s: (filename: '%s', image shape: %s, detected_faces: %s)",
 669                      self.__class__.__name__, filename, image.shape, detected_faces)
 670         self._filename = filename
 671         self._image = image
 672         self._detected_faces = detected_faces

 687         return self._image.shape

 700     def get_image_copy(self, color_format):
 701         """ Get a copy of the image in the requested color format.
 702 
 703         Parameters

 706             The requested color format of :attr:`image`

 708         Returns

 710         :class:`numpy.ndarray`:
 711             A copy of :attr:`image` in the requested :attr:`color_format`
 712         """

 714         image = getattr(self, "_image_as_{}".format(color_format.lower()))()

 717     def add_detected_faces(self, faces):
 718         """ Add detected faces to the object. Called at the end of each extraction phase.

 720         Parameters

 725         logger.trace("Adding detected faces for filename: '%s'. (faces: %s, lrtb: %s)",
 726                      self._filename, faces,
 727                      [(face.left, face.right, face.top, face.bottom) for face in faces])

 737         self._image = None
