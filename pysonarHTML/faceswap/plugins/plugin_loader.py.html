 118     @staticmethod
 119     def get_converter(category, name, disable_logging=False):
 120         """ Return requested converter plugin
 121 
 122         Converters work slightly differently to other faceswap plugins. They are created to do a
 123         specific task (e.g. color adjustment, mask blending etc.), so multiple plugins will be
 124         loaded in the convert phase, rather than just one plugin for the other phases.
 125 
 126         Parameters
 127         ----------
 128         name: str
 129             The name of the requested converter plugin
 130         disable_logging: bool, optional
 131             Whether to disable the INFO log message that the plugin is being imported.
 132             Default: `False`
 133 
 134         Returns
 135         -------
 136         :class:`plugins.convert` object:
 137             A converter sub plugin
 138         """
 139         return PluginLoader._import("convert.{}".format(category), name, disable_logging)

 141     @staticmethod
 142     def _import(attr, name, disable_logging):
 143         """ Import the plugin's module
 144 
 145         Parameters
 146         ----------
 147         name: str
 148             The name of the requested converter plugin
 149         disable_logging: bool
 150             Whether to disable the INFO log message that the plugin is being imported.
 151 
 152         Returns
 153         -------
 154         :class:`plugin` object:
 155             A plugin
 156         """
 157         name = name.replace("-", "_")
 158         ttl = attr.split(".")[-1].title()
 159         if not disable_logging:
 160             logger.info("Loading %s from %s plugin...", ttl, name.title())
 161         attr = "model" if attr == "Trainer" else attr.lower()
 162         mod = ".".join(("plugins", attr, name))
 163         module = import_module(mod)
 164         return getattr(module, ttl)

 166     @staticmethod
 167     def get_available_extractors(extractor_type, add_none=False):
 168         """ Return a list of available extractors of the given type
 169 
 170         Parameters
 171         ----------
 172         extractor_type: {'aligner', 'detector', 'masker'}
 173             The type of extractor to return the plugins for
 174         add_none: bool, optional
 175             Append "none" to the list of returned plugins. Default: False
 176         Returns
 177         -------
 178         list:
 179             A list of the available extractor plugin names for the given type
 180         """
 181         extractpath = os.path.join(os.path.dirname(__file__),
 182                                    "extract",
 183                                    extractor_type)
 184         extractors = sorted(item.name.replace(".py", "").replace("_", "-")
 185                             for item in os.scandir(extractpath)
 186                             if not item.name.startswith("_")
 187                             and not item.name.endswith("defaults.py")
 188                             and item.name.endswith(".py"))
 189         if add_none:
 190             extractors.insert(0, "none")
 191         return extractors

 210     @staticmethod
 211     def get_default_model():
 212         """ Return the default training model plugin name
 213 
 214         Returns
 215         -------
 216         str:
 217             The default faceswap training model
 218 
 219         """
 220         models = PluginLoader.get_available_models()
 221         return 'original' if 'original' in models else models[0]
