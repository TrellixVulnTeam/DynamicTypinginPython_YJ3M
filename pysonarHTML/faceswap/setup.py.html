 117     def get_required_packages(self):
 118         """ Load requirements list """
 119         if self.enable_amd:
 120             suffix = "amd.txt"
 121         elif self.enable_cuda:
 122             suffix = "nvidia.txt"
 123         else:
 124             suffix = "cpu.txt"
 125         req_files = ["_requirements_base.txt", f"requirements_{suffix}"]
 126         pypath = os.path.dirname(os.path.realpath(__file__))
 127         requirements = list()
 128         git_requirements = list()
 129         for req_file in req_files:
 130             requirements_file = os.path.join(pypath, req_file)
 131             with open(requirements_file) as req:
 132                 for package in req.readlines():
 133                     package = package.strip()
 134                     # parse_requirements can't handle git dependencies, so extract and then
 135                     # manually add to final list
 136                     if package and package.startswith("git+"):
 137                         git_requirements.append((package, []))
 138                         continue
 139                     if package and (not package.startswith(("#", "-r"))):
 140                         requirements.append(package)
 141         self.required_packages = [(pkg.name, pkg.specs)
 142                                   for pkg in parse_requirements(requirements)
 143                                   if pkg.marker is None or pkg.marker.evaluate()]
 144         self.required_packages.extend(git_requirements)

 194     def upgrade_pip(self):
 195         """ Upgrade pip to latest version """
 196         if not self.is_conda:
 197             # Don't do this with Conda, as we must use conda's pip
 198             self.output.info("Upgrading pip...")
 199             pipexe = [sys.executable, "-m", "pip"]
 200             pipexe.extend(["install", "--no-cache-dir", "-qq", "--upgrade"])
 201             if not self.is_admin and not self.is_virtualenv:
 202                 pipexe.append("--user")
 203             pipexe.append("pip")
 204             run(pipexe)
 205         import pip
 206         pip_version = pip.__version__
 207         self.output.info("Installed pip: {}".format(pip_version))

 209     def get_installed_packages(self):
 210         """ Get currently installed packages """
 211         installed_packages = dict()
 212         chk = Popen("\"{}\" -m pip freeze".format(sys.executable),
 213                     shell=True, stdout=PIPE)
 214         installed = chk.communicate()[0].decode(self.encoding).splitlines()
 215 
 216         for pkg in installed:
 217             if "==" not in pkg:
 218                 continue
 219             item = pkg.split("==")
 220             installed_packages[item[0]] = item[1]
 221         return installed_packages

 234     def update_tf_dep(self):
 235         """ Update Tensorflow Dependency """
 236         if self.is_conda or not self.enable_cuda:
 237             # CPU/AMD doesn't need Cuda and Conda handles Cuda and cuDNN so nothing to do here
 238             return
 239 
 240         tf_ver = None
 241         cudnn_inst = self.cudnn_version.split(".")
 242         for key, val in TENSORFLOW_REQUIREMENTS.items():
 243             cuda_req = val[0]
 244             cudnn_req = val[1].split(".")
 245             if cuda_req == self.cuda_version and (cudnn_req[0] == cudnn_inst[0] and
 246                                                   cudnn_req[1] <= cudnn_inst[1]):
 247                 tf_ver = key
 248                 break
 249         if tf_ver:
 250             # Remove the version of tensorflow in requirements.txt and add the correct version that
 251             # corresponds to the installed Cuda/cuDNN versions
 252             self.required_packages = [pkg for pkg in self.required_packages
 253                                       if not pkg.startswith("tensorflow-gpu")]
 254             tf_ver = "tensorflow-gpu{}".format(tf_ver)
 255             self.required_packages.append(tf_ver)
 256             return
 257 
 258         self.output.warning(
 259             "The minimum Tensorflow requirement is 1.12. \n"
 260             "Tensorflow currently has no official prebuild for your CUDA, cuDNN "
 261             "combination.\nEither install a combination that Tensorflow supports or "
 262             "build and install your own tensorflow-gpu.\r\n"
 263             "CUDA Version: {}\r\n"
 264             "cuDNN Version: {}\r\n"
 265             "Help:\n"
 266             "Building Tensorflow: https://www.tensorflow.org/install/install_sources\r\n"
 267             "Tensorflow supported versions: "
 268             "https://www.tensorflow.org/install/source#tested_build_configurations".format(
 269                 self.cuda_version, self.cudnn_version))
 270 
 271         custom_tf = input("Location of custom tensorflow-gpu wheel (leave "
 272                           "blank to manually install): ")
 273         if not custom_tf:
 274             return
 275 
 276         custom_tf = os.path.realpath(os.path.expanduser(custom_tf))
 277         if not os.path.isfile(custom_tf):
 278             self.output.error("{} not found".format(custom_tf))
 279         elif os.path.splitext(custom_tf)[1] != ".whl":
 280             self.output.error("{} is not a valid pip wheel".format(custom_tf))
 281         elif custom_tf:
 282             self.required_packages.append(custom_tf)

 300 class Output():
 301     """ Format and display output """
 302     def __init__(self):
 303         self.red = "\033[31m"
 304         self.green = "\033[32m"
 305         self.yellow = "\033[33m"
 306         self.default_color = "\033[0m"
 307         self.term_support_color = platform.system() in ("Linux", "Darwin")
 308 
 309     @staticmethod
 310     def __indent_text_block(text):
 311         """ Indent a text block """
 312         lines = text.splitlines()
 313         if len(lines) > 1:
 314             out = lines[0] + "\r\n"
 315             for i in range(1, len(lines)-1):
 316                 out = out + "        " + lines[i] + "\r\n"
 317             out = out + "        " + lines[-1]
 318             return out
 319         return text
 320 
 321     def info(self, text):
 322         """ Format INFO Text """
 323         trm = "INFO    "
 324         if self.term_support_color:
 325             trm = "{}INFO   {} ".format(self.green, self.default_color)
 326         print(trm + self.__indent_text_block(text))
 327 
 328     def warning(self, text):
 329         """ Format WARNING Text """
 330         trm = "WARNING "
 331         if self.term_support_color:
 332             trm = "{}WARNING{} ".format(self.yellow, self.default_color)
 333         print(trm + self.__indent_text_block(text))
 334 
 335     def error(self, text):
 336         """ Format ERROR Text """
 337         global INSTALL_FAILED  # pylint:disable=global-statement
 338         trm = "ERROR   "
 339         if self.term_support_color:
 340             trm = "{}ERROR  {} ".format(self.red, self.default_color)
 341         print(trm + self.__indent_text_block(text))
 342         INSTALL_FAILED = True

 436     def cuda_check(self):
 437         """ Check Cuda for Linux or Windows """
 438         chk = Popen("nvcc -V", shell=True, stdout=PIPE, stderr=PIPE)
 439         stdout, stderr = chk.communicate()
 440         if not stderr:
 441             version = re.search(r".*release (?P<cuda>\d+\.\d+)", stdout.decode(self.env.encoding))
 442             self.env.cuda_version = version.groupdict().get("cuda", None)
 443             if self.env.cuda_version:
 444                 self.output.info("CUDA version: " + self.env.cuda_version)
 445                 return
 446         # Failed to load nvcc
 447         if self.env.os_version[0] == "Linux":
 448             self.cuda_check_linux()
 449         elif self.env.os_version[0] == "Windows":
 450             self.cuda_check_windows()

 452     def cuda_check_linux(self):
 453         """ Check Linux CUDA Version """
 454         chk = os.popen("ldconfig -p | grep -P \"libcudart.so.\\d+.\\d+\" | head -n 1").read()
 455         if self.env.ld_library_path and not chk:
 456             paths = self.env.ld_library_path.split(":")
 457             for path in paths:
 458                 chk = os.popen("ls {} | grep -P -o \"libcudart.so.\\d+.\\d+\" | "
 459                                "head -n 1".format(path)).read()
 460                 if chk:
 461                     break
 462         if not chk:
 463             self.output.error("CUDA not found. Install and try again.\n"
 464                               "Recommended version:      CUDA 9.0     cuDNN 7.1.3\n"
 465                               "CUDA: https://developer.nvidia.com/cuda-downloads\n"
 466                               "cuDNN: https://developer.nvidia.com/rdp/cudnn-download")
 467             return
 468         cudavers = chk.strip().replace("libcudart.so.", "")
 469         self.env.cuda_version = cudavers[:cudavers.find(" ")]
 470         if self.env.cuda_version:
 471             self.output.info("CUDA version: " + self.env.cuda_version)
 472             self.env.cuda_path = chk[chk.find("=>") + 3:chk.find("targets") - 1]

 474     def cuda_check_windows(self):
 475         """ Check Windows CUDA Version """
 476         cuda_keys = self.cuda_keys_windows
 477         if not cuda_keys:
 478             self.output.error("CUDA not found. See "
 479                               "https://github.com/deepfakes/faceswap/blob/master/INSTALL.md#cuda "
 480                               "for instructions")
 481             return
 482 
 483         self.env.cuda_version = cuda_keys[0].lower().replace("cuda_path_v", "").replace("_", ".")
 484         self.env.cuda_path = os.environ[cuda_keys[0]]
 485         self.output.info("CUDA version: " + self.env.cuda_version)

 487     def cudnn_check(self):
 488         """ Check Linux or Windows cuDNN Version from cudnn.h """
 489         if self.env.os_version[0] == "Linux":
 490             cudnn_checkfiles = self.cudnn_checkfiles_linux()
 491         elif self.env.os_version[0] == "Windows":
 492             if not self.env.cuda_path and not self.cuda_keys_windows:
 493                 self.output.error(
 494                     "CUDA not found. See "
 495                     "https://github.com/deepfakes/faceswap/blob/master/INSTALL.md#cuda "
 496                     "for instructions")
 497                 return
 498             if not self.env.cuda_path:
 499                 self.env.cuda_path = os.environ[self.cuda_keys_windows[0]]
 500             cudnn_checkfiles = self.cudnn_checkfiles_windows()
 501 
 502         cudnn_checkfile = None
 503         for checkfile in cudnn_checkfiles:
 504             if os.path.isfile(checkfile):
 505                 cudnn_checkfile = checkfile
 506                 break
 507         if not cudnn_checkfile:
 508             self.output.error("cuDNN not found. See "
 509                               "https://github.com/deepfakes/faceswap/blob/master/INSTALL.md#cudnn "
 510                               "for instructions")
 511             return
 512         found = 0
 513         with open(cudnn_checkfile, "r") as ofile:
 514             for line in ofile:
 515                 if line.lower().startswith("#define cudnn_major"):
 516                     major = line[line.rfind(" ") + 1:].strip()
 517                     found += 1
 518                 elif line.lower().startswith("#define cudnn_minor"):
 519                     minor = line[line.rfind(" ") + 1:].strip()
 520                     found += 1
 521                 elif line.lower().startswith("#define cudnn_patchlevel"):
 522                     patchlevel = line[line.rfind(" ") + 1:].strip()
 523                     found += 1
 524                 if found == 3:
 525                     break
 526         if found != 3:
 527             self.output.error("cuDNN version could not be determined. See "
 528                               "https://github.com/deepfakes/faceswap/blob/master/INSTALL.md#cudnn "
 529                               "for instructions")
 530             return
 531 
 532         self.env.cudnn_version = "{}.{}".format(major, minor)
 533         self.output.info("cuDNN version: {}.{}".format(self.env.cudnn_version, patchlevel))
 534 
 535     @staticmethod
 536     def cudnn_checkfiles_linux():
 537         """ Return the checkfile locations for linux """
 538         chk = os.popen("ldconfig -p | grep -P \"libcudnn.so.\\d+\" | head -n 1").read()
 539         chk = chk.strip().replace("libcudnn.so.", "")
 540         if not chk:
 541             return list()
 542         cudnn_vers = chk[0]
 543         cudnn_path = chk[chk.find("=>") + 3:chk.find("libcudnn") - 1]
 544         cudnn_path = cudnn_path.replace("lib", "include")
 545         cudnn_checkfiles = [os.path.join(cudnn_path, "cudnn_v{}.h".format(cudnn_vers)),
 546                             os.path.join(cudnn_path, "cudnn.h")]
 547         return cudnn_checkfiles

 552         if not self.env.cuda_path:
 553             return list()
 554         cudnn_checkfile = os.path.join(self.env.cuda_path, "include", "cudnn.h")

 558 class Install():
 559     """ Install the requirements """
 560     def __init__(self, environment):
 561         self._operators = {"==": operator.eq,
 562                            ">=": operator.ge,
 563                            "<=": operator.le,
 564                            ">": operator.gt,
 565                            "<": operator.lt}
 566         self.output = environment.output
 567         self.env = environment
 568 
 569         if not self.env.is_installer and not self.env.updater:
 570             self.ask_continue()
 571         self.env.get_required_packages()
 572         self.check_missing_dep()
 573         self.check_conda_missing_dep()
 574         if (self.env.updater and
 575                 not self.env.missing_packages and not self.env.conda_missing_packages):
 576             self.output.info("All Dependencies are up to date")
 577             return
 578         self.install_missing_dep()
 579         if self.env.updater:
 580             return
 581         self.output.info("All python3 dependencies are met.\r\nYou are good to go.\r\n\r\n"
 582                          "Enter:  'python faceswap.py -h' to see the options\r\n"
 583                          "        'python faceswap.py gui' to launch the GUI")
 584 
 585     def ask_continue(self):
 586         """ Ask Continue with Install """
 587         inp = input("Please ensure your System Dependencies are met. Continue? [y/N] ")
 588         if inp in ("", "N", "n"):
 589             self.output.error("Please install system dependencies to continue")
 590             sys.exit(1)
 591 
 592     def check_missing_dep(self):
 593         """ Check for missing dependencies """
 594         for key, specs in self.env.required_packages:
 595             if self.env.is_conda:
 596                 # Get Conda alias for Key
 597                 key = CONDA_MAPPING.get(key, (key, None))[0]
 598             if (key == "git+https://github.com/deepfakes/nvidia-ml-py3.git" and
 599                     self.env.installed_packages.get("nvidia-ml-py3", "") == "7.352.1"):
 600                 # Annoying explicit hack to get around our custom version of nvidia-ml=py3 being
 601                 # constantly re-downloaded
 602                 continue
 603             if key not in self.env.installed_packages:
 604                 self.env.missing_packages.append((key, specs))
 605                 continue
 606             installed_vers = self.env.installed_packages.get(key, "")
 607             if specs and not all(self._operators[spec[0]](installed_vers, spec[1])
 608                                  for spec in specs):
 609                 self.env.missing_packages.append((key, specs))
 610 
 611     def check_conda_missing_dep(self):
 612         """ Check for conda missing dependencies """
 613         if not self.env.is_conda:
 614             return
 615         for pkg in self.env.conda_required_packages:
 616             key = pkg[0].split("==")[0]
 617             if key not in self.env.installed_packages:
 618                 self.env.conda_missing_packages.append(pkg)
 619                 continue
 620             if len(pkg[0].split("==")) > 1:
 621                 if pkg[0].split("==")[1] != self.env.installed_conda_packages.get(key):
 622                     self.env.conda_missing_packages.append(pkg)
 623                     continue
 624 
 625     def install_missing_dep(self):
 626         """ Install missing dependencies """
 627         # Install conda packages first
 628         if self.env.conda_missing_packages:
 629             self.install_conda_packages()
 630         if self.env.missing_packages:
 631             self.install_python_packages()
 632 
 633     def install_python_packages(self):
 634         """ Install required pip packages """
 635         self.output.info("Installing Required Python Packages. This may take some time...")
 636         for pkg, version in self.env.missing_packages:
 637             if version:
 638                 pkg = "{}{}".format(pkg, ",".join("".join(spec) for spec in version))
 639             if self.env.is_conda and not pkg.startswith("git"):
 640                 verbose = pkg.startswith("tensorflow") or self.env.updater
 641                 pkg = CONDA_MAPPING.get(pkg, (pkg, None))
 642                 channel = None if len(pkg) != 2 else pkg[1]
 643                 pkg = pkg[0]
 644                 if self.conda_installer(pkg, verbose=verbose, channel=channel, conda_only=False):
 645                     continue
 646             self.pip_installer(pkg)
 647 
 648     def install_conda_packages(self):
 649         """ Install required conda packages """
 650         self.output.info("Installing Required Conda Packages. This may take some time...")
 651         for pkg in self.env.conda_missing_packages:
 652             channel = None if len(pkg) != 2 else pkg[1]
 653             self.conda_installer(pkg[0], channel=channel, conda_only=True)
 654 
 655     def conda_installer(self, package, channel=None, verbose=False, conda_only=False):
 656         """ Install a conda package """
 657         #  Packages with special characters need to be enclosed in double quotes
 658         if any(char in package for char in (" ", "<", ">", "*", "|")):
 659             package = "\"{}\"".format(package)
 660         success = True
 661         condaexe = ["conda", "install", "-y"]
 662         if not verbose or self.env.updater:
 663             condaexe.append("-q")
 664         if channel:
 665             condaexe.extend(["-c", channel])
 666         condaexe.append(package)
 667         self.output.info("Installing {}".format(package))
 668         shell = self.env.os_version[0] == "Windows"
 669         try:
 670             if verbose:
 671                 run(condaexe, check=True, shell=shell)
 672             else:
 673                 with open(os.devnull, "w") as devnull:
 674                     run(condaexe, stdout=devnull, stderr=devnull, check=True, shell=shell)
 675         except CalledProcessError:
 676             if not conda_only:
 677                 self.output.info("{} not available in Conda. Installing with pip".format(package))
 678             else:
 679                 self.output.warning("Couldn't install {} with Conda. "
 680                                     "Please install this package manually".format(package))
 681             success = False
 682         return success
 683 
 684     def pip_installer(self, package):
 685         """ Install a pip package """
 686         pipexe = [sys.executable, "-m", "pip"]
 687         # hide info/warning and fix cache hang
 688         pipexe.extend(["install", "--no-cache-dir"])
 689         if not self.env.updater:
 690             pipexe.append("-qq")
 691         # install as user to solve perm restriction
 692         if not self.env.is_admin and not self.env.is_virtualenv:
 693             pipexe.append("--user")
 694         msg = "Installing {}".format(package)
 695         self.output.info(msg)
 696         pipexe.append(package)
 697         try:
 698             run(pipexe, check=True)
 699         except CalledProcessError:
 700             self.output.warning("Couldn't install {} with pip. "
 701                                 "Please install this package manually".format(package))
