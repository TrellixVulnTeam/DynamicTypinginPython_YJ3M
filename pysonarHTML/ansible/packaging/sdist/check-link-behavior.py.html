   1 #!/usr/bin/env python
   2 """Checks for link behavior required for sdist to retain symlinks."""
   3 
   4 from __future__ import (absolute_import, division, print_function)

   6 __metaclass__ = type

   8 import os
   9 import platform
  10 import shutil
  11 import sys
  12 import tempfile
  13 
  14 
  15 def main():
  16     """Main program entry point."""
  17     temp_dir = tempfile.mkdtemp()
  18 
  19     target_path = os.path.join(temp_dir, 'file.txt')
  20     symlink_path = os.path.join(temp_dir, 'symlink.txt')
  21     hardlink_path = os.path.join(temp_dir, 'hardlink.txt')
  22 
  23     try:
  24         with open(target_path, 'w'):
  25             pass
  26 
  27         os.symlink(target_path, symlink_path)
  28         os.link(symlink_path, hardlink_path)
  29 
  30         if not os.path.islink(symlink_path):
  31             abort('Symbolic link not created.')
  32 
  33         if not os.path.islink(hardlink_path):
  34             # known issue on MacOS (Darwin)
  35             abort('Hard link of symbolic link created as a regular file.')
  36     finally:
  37         shutil.rmtree(temp_dir)
  38 
  39 
  40 def abort(reason):
  41     """
  42     :type reason: str
  43     """
  44     sys.exit('ERROR: %s\n'
  45              'This will prevent symbolic links from being preserved in the resulting tarball.\n'
  46              'Aborting creation of sdist on platform: %s'
  47              % (reason, platform.system()))
  48 
  49 
  50 if __name__ == '__main__':
  51     main()
