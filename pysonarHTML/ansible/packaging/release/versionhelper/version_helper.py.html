  12 class AnsibleVersionMunger(object):
  13     tag_offsets = dict(
  14         dev=0,
  15         a=100,
  16         b=200,
  17         rc=1000
  18     )

  20     # TODO: allow overrides here for packaging bump etc
  21     def __init__(self, raw_version, revision=None, codename=None):
  22         self._raw_version = raw_version
  23         self._revision = revision
  24         self._parsed_version = Version(raw_version)
  25         self._codename = codename
  26         self._parsed_regex_match = re.match(VERSION_PATTERN, raw_version, re.VERBOSE | re.IGNORECASE)
  27 
  28     @property
  29     def deb_version(self):
  30         v = self._parsed_version
  31 
  32         match = self._parsed_regex_match
  33 
  34         # treat dev/post as prerelease for now; treat dev/post as equivalent and disallow together
  35         if v.is_prerelease or match.group('dev') or match.group('post'):
  36             if match.group('dev') and match.group('post'):
  37                 raise Exception("dev and post may not currently be used together")
  38             if match.group('pre'):
  39                 tag_value = match.group('pre')
  40                 tag_type = match.group('pre_l')
  41                 if match.group('dev'):
  42                     tag_value += ('~%s' % match.group('dev').strip('.'))
  43                 if match.group('post'):
  44                     tag_value += ('~%s' % match.group('post').strip('.'))
  45             elif match.group('dev'):
  46                 tag_type = "dev"
  47                 tag_value = match.group('dev').strip('.')
  48             elif match.group('post'):
  49                 tag_type = "dev"
  50                 tag_value = match.group('post').strip('.')
  51             else:
  52                 raise Exception("unknown prerelease type for version {0}".format(self._raw_version))
  53         else:
  54             tag_type = None
  55             tag_value = ''
  56 
  57         # not a pre/post/dev release, just return base version
  58         if not tag_type:
  59             return '{base_version}'.format(base_version=self.base_version)
  60 
  61         # it is a pre/dev release, include the tag value with a ~
  62         return '{base_version}~{tag_value}'.format(base_version=self.base_version, tag_value=tag_value)
  63 
  64     @property
  65     def deb_release(self):
  66         return '1' if self._revision is None else str(self._revision)
  67 
  68     @property
  69     def rpm_release(self):
  70         v = self._parsed_version
  71         match = self._parsed_regex_match
  72 
  73         # treat presence of dev/post as prerelease for now; treat dev/post the same and disallow together
  74         if v.is_prerelease or match.group('dev') or match.group('post'):
  75             if match.group('dev') and match.group('post'):
  76                 raise Exception("dev and post may not currently be used together")
  77             if match.group('pre'):
  78                 tag_value = match.group('pre')
  79                 tag_type = match.group('pre_l')
  80                 tag_ver = match.group('pre_n')
  81                 if match.group('dev'):
  82                     tag_value += match.group('dev')
  83                 if match.group('post'):
  84                     tag_value += match.group('post')
  85             elif match.group('dev'):
  86                 tag_type = "dev"
  87                 tag_value = match.group('dev')
  88                 tag_ver = match.group('dev_n')
  89             elif match.group('post'):
  90                 tag_type = "dev"
  91                 tag_value = match.group('post')
  92                 tag_ver = match.group('post_n')
  93             else:
  94                 raise Exception("unknown prerelease type for version {0}".format(self._raw_version))
  95         else:
  96             tag_type = None
  97             tag_value = ''
  98             tag_ver = 0
  99 
 100         # not a pre/post/dev release, just append revision (default 1)
 101         if not tag_type:
 102             if self._revision is None:
 103                 self._revision = 1
 104             return '{revision}'.format(revision=self._revision)
 105 
 106         # cleanse tag value in case it starts with .
 107         tag_value = tag_value.strip('.')
 108 
 109         # coerce to int and None == 0
 110         tag_ver = int(tag_ver if tag_ver else 0)
 111 
 112         if self._revision is None:
 113             tag_offset = self.tag_offsets.get(tag_type)
 114             if tag_offset is None:
 115                 raise Exception('no tag offset defined for tag {0}'.format(tag_type))
 116             pkgrel = '0.{0}'.format(tag_offset + tag_ver)
 117         else:
 118             pkgrel = self._revision
 119 
 120         return '{pkgrel}.{tag_value}'.format(pkgrel=pkgrel, tag_value=tag_value)
