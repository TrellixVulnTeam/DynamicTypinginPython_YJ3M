  16 def _ensure_default_collection(collection_list=None):
  17     default_collection = AnsibleCollectionConfig.default_collection
  18 
  19     # Will be None when used as the default
  20     if collection_list is None:
  21         collection_list = []
  22 
  23     # FIXME: exclude role tasks?
  24     if default_collection and default_collection not in collection_list:
  25         collection_list.insert(0, default_collection)
  26 
  27     # if there's something in the list, ensure that builtin or legacy is always there too
  28     if collection_list and 'ansible.builtin' not in collection_list and 'ansible.legacy' not in collection_list:
  29         collection_list.append('ansible.legacy')
  30 
  31     return collection_list

  34 class CollectionSearch:
  35 
  36     # this needs to be populated before we can resolve tasks/roles/etc
  37     _collections = FieldAttribute(isa='list', listof=string_types, priority=100, default=_ensure_default_collection,
  38                                   always_post_validate=True, static=True)
  39 
  40     def _load_collections(self, attr, ds):
  41         # We are always a mixin with Base, so we can validate this untemplated
  42         # field early on to guarantee we are dealing with a list.
  43         ds = self.get_validated_value('collections', self._collections, ds, None)
  44 
  45         # this will only be called if someone specified a value; call the shared value
  46         _ensure_default_collection(collection_list=ds)
  47 
  48         if not ds:  # don't return an empty collection list, just return None
  49             return None
  50 
  51         # This duplicates static attr checking logic from post_validate()
  52         # because if the user attempts to template a collection name, it may
  53         # error before it ever gets to the post_validate() warning (e.g. trying
  54         # to import a role from the collection).
  55         env = Environment()
  56         for collection_name in ds:
  57             if is_template(collection_name, env):
  58                 display.warning('"collections" is not templatable, but we found: %s, '
  59                                 'it will not be templated and will be used "as is".' % (collection_name))
  60 
  61         return ds
