  45     """
  46     A play is a language feature that represents a list of roles and/or
  47     task/handler blocks to execute on a given set of hosts.

 100     def get_name(self):
 101         ''' return the name of the Play '''

 104     @staticmethod
 105     def load(data, variable_manager=None, loader=None, vars=None):
 106         if ('name' not in data or data['name'] is None) and 'hosts' in data:
 107             if data['hosts'] is None or all(host is None for host in data['hosts']):
 108                 raise AnsibleParserError("Hosts list cannot be empty - please check your playbook")
 109             if isinstance(data['hosts'], list):
 110                 data['name'] = ','.join(data['hosts'])
 111             else:
 112                 data['name'] = data['hosts']
 113         p = Play()
 114         if vars:
 115             p.vars = vars.copy()
 116         return p.load_data(data, variable_manager=variable_manager, loader=loader)

 118     def preprocess_data(self, ds):
 119         '''
 120         Adjusts play datastructure to cleanup old/legacy items
 121         '''
 122 
 123         if not isinstance(ds, dict):
 124             raise AnsibleAssertionError('while preprocessing data (%s), ds should be a dict but was a %s' % (ds, type(ds)))
 125 
 126         # The use of 'user' in the Play datastructure was deprecated to
 127         # line up with the same change for Tasks, due to the fact that
 128         # 'user' conflicted with the user module.
 129         if 'user' in ds:
 130             # this should never happen, but error out with a helpful message
 131             # to the user if it does...
 132             if 'remote_user' in ds:
 133                 raise AnsibleParserError("both 'user' and 'remote_user' are set for %s. "
 134                                          "The use of 'user' is deprecated, and should be removed" % self.get_name(), obj=ds)
 135 
 136             ds['remote_user'] = ds['user']
 137             del ds['user']
 138 
 139         return super(Play, self).preprocess_data(ds)

 141     def _load_tasks(self, attr, ds):
 142         '''
 143         Loads a list of blocks from a list which may be mixed tasks/blocks.
 144         Bare tasks outside of a block are given an implicit block.
 145         '''
 146         try:
 147             return load_list_of_blocks(ds=ds, play=self, variable_manager=self._variable_manager, loader=self._loader)
 148         except AssertionError as e:
 149             raise AnsibleParserError("A malformed block was encountered while loading tasks: %s" % to_native(e), obj=self._ds, orig_exc=e)
 150

 171     def _load_handlers(self, attr, ds):
 172         '''
 173         Loads a list of blocks from a list which may be mixed handlers/blocks.
 174         Bare handlers outside of a block are given an implicit block.
 175         '''
 176         try:
 177             return self._extend_value(
 178                 self.handlers,
 179                 load_list_of_blocks(ds=ds, play=self, use_handlers=True, variable_manager=self._variable_manager, loader=self._loader),
 180                 prepend=True
 181             )
 182         except AssertionError as e:
 183             raise AnsibleParserError("A malformed block was encountered while loading handlers", obj=self._ds, orig_exc=e)
 184 
 185     def _load_roles(self, attr, ds):
 186         '''
 187         Loads and returns a list of RoleInclude objects from the datastructure
 188         list of role definitions and creates the Role from those objects
 189         '''
 190 
 191         if ds is None:
 192             ds = []
 193 
 194         try:
 195             role_includes = load_list_of_roles(ds, play=self, variable_manager=self._variable_manager,
 196                                                loader=self._loader, collection_search_list=self.collections)
 197         except AssertionError as e:
 198             raise AnsibleParserError("A malformed role declaration was encountered.", obj=self._ds, orig_exc=e)
 199 
 200         roles = []
 201         for ri in role_includes:
 202             roles.append(Role.load(ri, play=self))
 203 
 204         self.roles[:0] = roles
 205 
 206         return self.roles
 207 
 208     def _load_vars_prompt(self, attr, ds):
 209         new_ds = preprocess_vars(ds)
 210         vars_prompts = []
 211         if new_ds is not None:
 212             for prompt_data in new_ds:
 213                 if 'name' not in prompt_data:
 214                     raise AnsibleParserError("Invalid vars_prompt data structure, missing 'name' key", obj=ds)
 215                 for key in prompt_data:
 216                     if key not in ('name', 'prompt', 'default', 'private', 'confirm', 'encrypt', 'salt_size', 'salt', 'unsafe'):
 217                         raise AnsibleParserError("Invalid vars_prompt data structure, found unsupported key '%s'" % key, obj=ds)
 218                 vars_prompts.append(prompt_data)
 219         return vars_prompts

 221     def _compile_roles(self):
 222         '''
 223         Handles the role compilation step, returning a flat list of tasks
 224         with the lowest level dependencies first. For example, if a role R
 225         has a dependency D1, which also has a dependency D2, the tasks from
 226         D2 are merged first, followed by D1, and lastly by the tasks from
 227         the parent role R last. This is done for all roles in the Play.
 228         '''
 229 
 230         block_list = []
 231 
 232         if len(self.roles) > 0:
 233             for r in self.roles:
 234                 # Don't insert tasks from ``import/include_role``, preventing
 235                 # duplicate execution at the wrong time
 236                 if r.from_include:
 237                     continue
 238                 block_list.extend(r.compile(play=self))
 239 
 240         return block_list
 241 
 242     def compile_roles_handlers(self):
 243         '''
 244         Handles the role handler compilation step, returning a flat list of Handlers
 245         This is done for all roles in the Play.
 246         '''
 247 
 248         block_list = []
 249 
 250         if len(self.roles) > 0:
 251             for r in self.roles:
 252                 if r.from_include:
 253                     continue
 254                 block_list.extend(r.get_handler_blocks(play=self))
 255 
 256         return block_list
 257 
 258     def compile(self):
 259         '''
 260         Compiles and returns the task list for this play, compiled from the
 261         roles (which are themselves compiled recursively) and/or the list of
 262         tasks specified in the play.
 263         '''
 264 
 265         # create a block containing a single flush handlers meta
 266         # task, so we can be sure to run handlers at certain points
 267         # of the playbook execution
 268         flush_block = Block.load(
 269             data={'meta': 'flush_handlers'},
 270             play=self,
 271             variable_manager=self._variable_manager,
 272             loader=self._loader
 273         )
 274 
 275         block_list = []
 276 
 277         block_list.extend(self.pre_tasks)
 278         block_list.append(flush_block)
 279         block_list.extend(self._compile_roles())
 280         block_list.extend(self.tasks)
 281         block_list.append(flush_block)
 282         block_list.extend(self.post_tasks)
 283         block_list.append(flush_block)
 284 
 285         return block_list

 303     def get_tasks(self):
 304         tasklist = []
 305         for task in self.pre_tasks + self.tasks + self.post_tasks:
 306             if isinstance(task, Block):
 307                 tasklist.append(task.block + task.rescue + task.always)
 308             else:
 309                 tasklist.append(task)
 310         return tasklist

 312     def serialize(self):
 313         data = super(Play, self).serialize()
 314 
 315         roles = []
 316         for role in self.get_roles():
 317             roles.append(role.serialize())
 318         data['roles'] = roles
 319         data['included_path'] = self._included_path
 320 
 321         return data
 322 
 323     def deserialize(self, data):
 324         super(Play, self).deserialize(data)
 325 
 326         self._included_path = data.get('included_path', None)
 327         if 'roles' in data:
 328             role_data = data.get('roles', [])
 329             roles = []
 330             for role in role_data:
 331                 r = Role()
 332                 r.deserialize(role)
 333                 roles.append(r)
 334 
 335             setattr(self, 'roles', roles)
 336             del data['roles']

 338     def copy(self):
 339         new_me = super(Play, self).copy()
 340         new_me.ROLE_CACHE = self.ROLE_CACHE.copy()
 341         new_me._included_conditional = self._included_conditional
 342         new_me._included_path = self._included_path
 343         return new_me
