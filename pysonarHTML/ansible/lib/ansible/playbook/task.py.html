  91     def __init__(self, block=None, role=None, task_include=None):
  92         ''' constructors a task, without the Task.load classmethod, it will be pretty blank '''
  93 
  94         # This is a reference of all the candidate action names for transparent execution of module_defaults with redirected content
  95         # This isn't a FieldAttribute to prevent it from being set via the playbook
  96         self._ansible_internal_redirect_list = []
  97 
  98         self._role = role
  99         self._parent = None

 102             self._parent = task_include

 104             self._parent = block

 108     def get_path(self):
 109         ''' return the absolute path of the task with its line number '''
 110 
 111         path = ""
 112         if hasattr(self, '_ds') and hasattr(self._ds, '_data_source') and hasattr(self._ds, '_line_number'):
 113             path = "%s:%s" % (self._ds._data_source, self._ds._line_number)
 114         elif hasattr(self._parent._play, '_ds') and hasattr(self._parent._play._ds, '_data_source') and hasattr(self._parent._play._ds, '_line_number'):
 115             path = "%s:%s" % (self._parent._play._ds._data_source, self._parent._play._ds._line_number)
 116         return path

 118     def get_name(self, include_role_fqcn=True):
 119         ''' return the name of the task '''
 120 
 121         if self._role:
 122             role_name = self._role.get_name(include_role_fqcn=include_role_fqcn)
 123 
 124         if self._role and self.name and role_name not in self.name:
 125             return "%s : %s" % (role_name, self.name)
 126         elif self.name:
 127             return self.name
 128         else:
 129             if self._role:
 130                 return "%s : %s" % (role_name, self.action)
 131             else:
 132                 return "%s" % (self.action,)

 134     def _merge_kv(self, ds):
 135         if ds is None:
 136             return ""
 137         elif isinstance(ds, string_types):
 138             return ds
 139         elif isinstance(ds, dict):
 140             buf = ""
 141             for (k, v) in iteritems(ds):
 142                 if k.startswith('_'):
 143                     continue
 144                 buf = buf + "%s=%s " % (k, v)
 145             buf = buf.strip()
 146             return buf

 154         ''' returns a human readable representation of the task '''
 155         if self.get_name() == 'meta':

 160     def _preprocess_with_loop(self, ds, new_ds, k, v):
 161         ''' take a lookup plugin name and store it correctly '''
 162 
 163         loop_name = k.replace("with_", "")
 164         if new_ds.get('loop') is not None or new_ds.get('loop_with') is not None:
 165             raise AnsibleError("duplicate loop in task: %s" % loop_name, obj=ds)
 166         if v is None:
 167             raise AnsibleError("you must specify a value when using %s" % k, obj=ds)
 168         new_ds['loop_with'] = loop_name
 169         new_ds['loop'] = v
 170         # display.deprecated("with_ type loops are being phased out, use the 'loop' keyword instead",
 171         #                    version="2.10", collection_name='ansible.builtin')
 172 
 173     def preprocess_data(self, ds):
 174         '''
 175         tasks are especially complex arguments so need pre-processing.
 176         keep it short.
 177         '''
 178 
 179         if not isinstance(ds, dict):
 180             raise AnsibleAssertionError('ds (%s) should be a dict but was a %s' % (ds, type(ds)))
 181 
 182         # the new, cleaned datastructure, which will have legacy
 183         # items reduced to a standard structure suitable for the
 184         # attributes of the task class
 185         new_ds = AnsibleMapping()
 186         if isinstance(ds, AnsibleBaseYAMLObject):
 187             new_ds.ansible_pos = ds.ansible_pos
 188 
 189         # since this affects the task action parsing, we have to resolve in preprocess instead of in typical validator
 190         default_collection = AnsibleCollectionConfig.default_collection
 191 
 192         collections_list = ds.get('collections')
 193         if collections_list is None:
 194             # use the parent value if our ds doesn't define it
 195             collections_list = self.collections
 196         else:
 197             # Validate this untemplated field early on to guarantee we are dealing with a list.
 198             # This is also done in CollectionSearch._load_collections() but this runs before that call.
 199             collections_list = self.get_validated_value('collections', self._collections, collections_list, None)
 200 
 201         if default_collection and not self._role:  # FIXME: and not a collections role
 202             if collections_list:
 203                 if default_collection not in collections_list:
 204                     collections_list.insert(0, default_collection)
 205             else:
 206                 collections_list = [default_collection]
 207 
 208         if collections_list and 'ansible.builtin' not in collections_list and 'ansible.legacy' not in collections_list:
 209             collections_list.append('ansible.legacy')
 210 
 211         if collections_list:
 212             ds['collections'] = collections_list
 213 
 214         # use the args parsing class to determine the action, args,
 215         # and the delegate_to value from the various possible forms
 216         # supported as legacy
 217         args_parser = ModuleArgsParser(task_ds=ds, collection_list=collections_list)
 218         try:
 219             (action, args, delegate_to) = args_parser.parse()
 220         except AnsibleParserError as e:
 221             # if the raises exception was created with obj=ds args, then it includes the detail
 222             # so we dont need to add it so we can just re raise.
 223             if e._obj:
 224                 raise
 225             # But if it wasn't, we can add the yaml object now to get more detail
 226             raise AnsibleParserError(to_native(e), obj=ds, orig_exc=e)
 227         else:
 228             self._ansible_internal_redirect_list = args_parser.internal_redirect_list[:]
 229 
 230         # the command/shell/script modules used to support the `cmd` arg,
 231         # which corresponds to what we now call _raw_params, so move that
 232         # value over to _raw_params (assuming it is empty)
 233         if action in ('command', 'shell', 'script'):
 234             if 'cmd' in args:
 235                 if args.get('_raw_params', '') != '':
 236                     raise AnsibleError("The 'cmd' argument cannot be used when other raw parameters are specified."
 237                                        " Please put everything in one or the other place.", obj=ds)
 238                 args['_raw_params'] = args.pop('cmd')
 239 
 240         new_ds['action'] = action
 241         new_ds['args'] = args
 242         new_ds['delegate_to'] = delegate_to
 243 
 244         # we handle any 'vars' specified in the ds here, as we may
 245         # be adding things to them below (special handling for includes).
 246         # When that deprecated feature is removed, this can be too.
 247         if 'vars' in ds:
 248             # _load_vars is defined in Base, and is used to load a dictionary
 249             # or list of dictionaries in a standard way
 250             new_ds['vars'] = self._load_vars(None, ds.get('vars'))
 251         else:
 252             new_ds['vars'] = dict()
 253 
 254         for (k, v) in iteritems(ds):
 255             if k in ('action', 'local_action', 'args', 'delegate_to') or k == action or k == 'shell':
 256                 # we don't want to re-assign these values, which were determined by the ModuleArgsParser() above
 257                 continue
 258             elif k.startswith('with_') and k.replace("with_", "") in lookup_loader:
 259                 # transform into loop property
 260                 self._preprocess_with_loop(ds, new_ds, k, v)
 261             else:
 262                 # pre-2.0 syntax allowed variables for include statements at the top level of the task,
 263                 # so we move those into the 'vars' dictionary here, and show a deprecation message
 264                 # as we will remove this at some point in the future.
 265                 if action in ('include',) and k not in self._valid_attrs and k not in self.DEPRECATED_ATTRIBUTES:
 266                     display.deprecated("Specifying include variables at the top-level of the task is deprecated."
 267                                        " Please see:\nhttps://docs.ansible.com/ansible/playbooks_roles.html#task-include-files-and-encouraging-reuse\n\n"
 268                                        " for currently supported syntax regarding included files and variables",
 269                                        version="2.12", collection_name='ansible.builtin')
 270                     new_ds['vars'][k] = v
 271                 elif C.INVALID_TASK_ATTRIBUTE_FAILED or k in self._valid_attrs:
 272                     new_ds[k] = v
 273                 else:
 274                     display.warning("Ignoring invalid attribute: %s" % k)
 275 
 276         return super(Task, self).preprocess_data(new_ds)

 278     def _load_loop_control(self, attr, ds):
 279         if not isinstance(ds, dict):

 316     def _post_validate_environment(self, attr, value, templar):
 317         '''
 318         Override post validation of vars on the play, as we don't want to
 319         template these too early.
 320         '''
 321         env = {}
 322         if value is not None:
 323 
 324             def _parse_env_kv(k, v):
 325                 try:
 326                     env[k] = templar.template(v, convert_bare=False)
 327                 except AnsibleUndefinedVariable as e:
 328                     error = to_native(e)
 329                     if self.action in ('setup', 'gather_facts') and 'ansible_facts.env' in error or 'ansible_env' in error:
 330                         # ignore as fact gathering is required for 'env' facts
 331                         return
 332                     raise
 333 
 334             if isinstance(value, list):
 335                 for env_item in value:
 336                     if isinstance(env_item, dict):
 337                         for k in env_item:
 338                             _parse_env_kv(k, env_item[k])
 339                     else:
 340                         isdict = templar.template(env_item, convert_bare=False)
 341                         if isinstance(isdict, dict):
 342                             env.update(isdict)
 343                         else:
 344                             display.warning("could not parse environment value, skipping: %s" % value)
 345 
 346             elif isinstance(value, dict):
 347                 # should not really happen
 348                 env = dict()
 349                 for env_item in value:
 350                     _parse_env_kv(env_item, value[env_item])
 351             else:
 352                 # at this point it should be a simple string, also should not happen
 353                 env = templar.template(value, convert_bare=False)
 354 
 355         return env

 379         all_vars = dict()

 400     def copy(self, exclude_parent=False, exclude_tasks=False):
 401         new_me = super(Task, self).copy()
 402 
 403         # if the task has an associated list of candidate names, copy it to the new object too
 404         new_me._ansible_internal_redirect_list = self._ansible_internal_redirect_list[:]
 405 
 406         new_me._parent = None
 407         if self._parent and not exclude_parent:
 408             new_me._parent = self._parent.copy(exclude_tasks=exclude_tasks)
 409 
 410         new_me._role = None
 411         if self._role:
 412             new_me._role = self._role
 413 
 414         return new_me

 432     def deserialize(self, data):
 433 
 434         # import is here to avoid import loops
 435         from ansible.playbook.task_include import TaskInclude
 436         from ansible.playbook.handler_task_include import HandlerTaskInclude
 437 
 438         parent_data = data.get('parent', None)
 439         if parent_data:
 440             parent_type = data.get('parent_type')
 441             if parent_type == 'Block':
 442                 p = Block()
 443             elif parent_type == 'TaskInclude':
 444                 p = TaskInclude()
 445             elif parent_type == 'HandlerTaskInclude':
 446                 p = HandlerTaskInclude()
 447             p.deserialize(parent_data)
 448             self._parent = p
 449             del data['parent']
 450 
 451         role_data = data.get('role')
 452         if role_data:
 453             r = Role()
 454             r.deserialize(role_data)
 455             self._role = r
 456             del data['role']
 457 
 458         self._ansible_internal_redirect_list = data.get('_ansible_internal_redirect_list', [])
 459 
 460         super(Task, self).deserialize(data)

 474     def _get_parent_attribute(self, attr, extend=False, prepend=False):
 475         '''
 476         Generic logic to get the attribute or parent attribute for a task value.
 477         '''
 478 
 479         extend = self._valid_attrs[attr].extend
 480         prepend = self._valid_attrs[attr].prepend
 481         try:
 482             value = self._attributes[attr]
 483             # If parent is static, we can grab attrs from the parent
 484             # otherwise, defer to the grandparent
 485             if getattr(self._parent, 'statically_loaded', True):
 486                 _parent = self._parent
 487             else:
 488                 _parent = self._parent._parent
 489 
 490             if _parent and (value is Sentinel or extend):
 491                 if getattr(_parent, 'statically_loaded', True):
 492                     # vars are always inheritable, other attributes might not be for the parent but still should be for other ancestors
 493                     if attr != 'vars' and hasattr(_parent, '_get_parent_attribute'):
 494                         parent_value = _parent._get_parent_attribute(attr)
 495                     else:
 496                         parent_value = _parent._attributes.get(attr, Sentinel)
 497 
 498                     if extend:
 499                         value = self._extend_value(value, parent_value, prepend)
 500                     else:
 501                         value = parent_value
 502         except KeyError:
 503             pass
 504 
 505         return value

 513     def get_search_path(self):
 514         '''
 515         Return the list of paths you should search for files, in order.
 516         This follows role/playbook dependency chain.
 517         '''
 518         path_stack = []
 519 
 520         dep_chain = self.get_dep_chain()
 521         # inside role: add the dependency chain from current to dependent
 522         if dep_chain:
 523             path_stack.extend(reversed([x._role_path for x in dep_chain]))
 524 
 525         # add path of task itself, unless it is already in the list
 526         task_dir = os.path.dirname(self.get_path())
 527         if task_dir not in path_stack:
 528             path_stack.append(task_dir)
 529 
 530         return path_stack
