  38 class PlaybookInclude(Base, Conditional, Taggable):
  39 
  40     _import_playbook = FieldAttribute(isa='string')
  41     _vars = FieldAttribute(isa='dict', default=dict)

  44     def load(data, basedir, variable_manager=None, loader=None):
  45         return PlaybookInclude().load_data(ds=data, basedir=basedir, variable_manager=variable_manager, loader=loader)

  47     def load_data(self, ds, basedir, variable_manager=None, loader=None):
  48         '''
  49         Overrides the base load_data(), as we're actually going to return a new
  50         Playbook() object rather than a PlaybookInclude object
  51         '''
  52 
  53         # import here to avoid a dependency loop
  54         from ansible.playbook import Playbook
  55         from ansible.playbook.play import Play
  56 
  57         # first, we use the original parent method to correctly load the object
  58         # via the load_data/preprocess_data system we normally use for other
  59         # playbook objects
  60         new_obj = super(PlaybookInclude, self).load_data(ds, variable_manager, loader)
  61 
  62         all_vars = self.vars.copy()
  63         if variable_manager:
  64             all_vars.update(variable_manager.get_vars())
  65 
  66         templar = Templar(loader=loader, variables=all_vars)
  67 
  68         # then we use the object to load a Playbook
  69         pb = Playbook(loader=loader)
  70 
  71         file_name = templar.template(new_obj.import_playbook)
  72         if not os.path.isabs(file_name):
  73             file_name = os.path.join(basedir, file_name)
  74 
  75         pb._load_playbook_data(file_name=file_name, variable_manager=variable_manager, vars=self.vars.copy())
  76 
  77         # finally, update each loaded playbook entry with any variables specified
  78         # on the included playbook and/or any tags which may have been set
  79         for entry in pb._entries:
  80 
  81             # conditional includes on a playbook need a marker to skip gathering
  82             if new_obj.when and isinstance(entry, Play):
  83                 entry._included_conditional = new_obj.when[:]
  84 
  85             temp_vars = entry.vars.copy()
  86             temp_vars.update(new_obj.vars)
  87             param_tags = temp_vars.pop('tags', None)
  88             if param_tags is not None:
  89                 entry.tags.extend(param_tags.split(','))
  90             entry.vars = temp_vars
  91             entry.tags = list(set(entry.tags).union(new_obj.tags))
  92             if entry._included_path is None:
  93                 entry._included_path = os.path.dirname(file_name)
  94 
  95             # Check to see if we need to forward the conditionals on to the included
  96             # plays. If so, we can take a shortcut here and simply prepend them to
  97             # those attached to each block (if any)
  98             if new_obj.when:
  99                 for task_block in (entry.pre_tasks + entry.roles + entry.tasks + entry.post_tasks):
 100                     task_block._attributes['when'] = new_obj.when[:] + task_block.when[:]
 101 
 102         return pb

 104     def preprocess_data(self, ds):
 105         '''
 106         Regorganizes the data for a PlaybookInclude datastructure to line
 107         up with what we expect the proper attributes to be
 108         '''
 109 
 110         if not isinstance(ds, dict):
 111             raise AnsibleAssertionError('ds (%s) should be a dict but was a %s' % (ds, type(ds)))
 112 
 113         # the new, cleaned datastructure, which will have legacy
 114         # items reduced to a standard structure
 115         new_ds = AnsibleMapping()
 116         if isinstance(ds, AnsibleBaseYAMLObject):
 117             new_ds.ansible_pos = ds.ansible_pos
 118 
 119         for (k, v) in iteritems(ds):
 120             if k in ('include', 'import_playbook'):
 121                 self._preprocess_import(ds, new_ds, k, v)
 122             else:
 123                 # some basic error checking, to make sure vars are properly
 124                 # formatted and do not conflict with k=v parameters
 125                 if k == 'vars':
 126                     if 'vars' in new_ds:
 127                         raise AnsibleParserError("import_playbook parameters cannot be mixed with 'vars' entries for import statements", obj=ds)
 128                     elif not isinstance(v, dict):
 129                         raise AnsibleParserError("vars for import_playbook statements must be specified as a dictionary", obj=ds)
 130                 new_ds[k] = v
 131 
 132         return super(PlaybookInclude, self).preprocess_data(new_ds)
 133 
 134     def _preprocess_import(self, ds, new_ds, k, v):
 135         '''
 136         Splits the playbook import line up into filename and parameters
 137         '''
 138 
 139         if v is None:
 140             raise AnsibleParserError("playbook import parameter is missing", obj=ds)
 141         elif not isinstance(v, string_types):
 142             raise AnsibleParserError("playbook import parameter must be a string indicating a file path, got %s instead" % type(v), obj=ds)
 143 
 144         # The import_playbook line must include at least one item, which is the filename
 145         # to import. Anything after that should be regarded as a parameter to the import
 146         items = split_args(v)
 147         if len(items) == 0:
 148             raise AnsibleParserError("import_playbook statements must specify the file name to import", obj=ds)
 149         else:
 150             new_ds['import_playbook'] = items[0].strip()
 151             if len(items) > 1:
 152                 display.warning('Additional parameters in import_playbook statements are not supported. This will be an error in version 2.14')
 153                 # rejoin the parameter portion of the arguments and
 154                 # then use parse_kv() to get a dict of params back
 155                 params = parse_kv(" ".join(items[1:]))
 156                 if 'tags' in params:
 157                     new_ds['tags'] = params.pop('tags')
 158                 if 'vars' in new_ds:
 159                     raise AnsibleParserError("import_playbook parameters cannot be mixed with 'vars' entries for import statements", obj=ds)
 160                 new_ds['vars'] = params
