  38 class Playbook:
  39 
  40     def __init__(self, loader):
  41         # Entries in the datastructure of a playbook may
  42         # be either a play or an include statement
  43         self._entries = []
  44         self._basedir = to_text(os.getcwd(), errors='surrogate_or_strict')
  45         self._loader = loader
  46         self._file_name = None
  47 
  48     @staticmethod
  49     def load(file_name, variable_manager=None, loader=None):
  50         pb = Playbook(loader=loader)
  51         pb._load_playbook_data(file_name=file_name, variable_manager=variable_manager)
  52         return pb
  53 
  54     def _load_playbook_data(self, file_name, variable_manager, vars=None):
  55 
  56         if os.path.isabs(file_name):
  57             self._basedir = os.path.dirname(file_name)
  58         else:
  59             self._basedir = os.path.normpath(os.path.join(self._basedir, os.path.dirname(file_name)))
  60 
  61         # set the loaders basedir
  62         cur_basedir = self._loader.get_basedir()
  63         self._loader.set_basedir(self._basedir)
  64 
  65         add_all_plugin_dirs(self._basedir)
  66 
  67         self._file_name = file_name
  68 
  69         try:
  70             ds = self._loader.load_from_file(os.path.basename(file_name))
  71         except UnicodeDecodeError as e:
  72             raise AnsibleParserError("Could not read playbook (%s) due to encoding issues: %s" % (file_name, to_native(e)))
  73 
  74         # check for errors and restore the basedir in case this error is caught and handled
  75         if ds is None:
  76             self._loader.set_basedir(cur_basedir)
  77             raise AnsibleParserError("Empty playbook, nothing to do", obj=ds)
  78         elif not isinstance(ds, list):
  79             self._loader.set_basedir(cur_basedir)
  80             raise AnsibleParserError("A playbook must be a list of plays, got a %s instead" % type(ds), obj=ds)
  81         elif not ds:
  82             display.deprecated("Empty plays will currently be skipped, in the future they will cause a syntax error",
  83                                version='2.12', collection_name='ansible.builtin')
  84 
  85         # Parse the playbook entries. For plays, we simply parse them
  86         # using the Play() object, and includes are parsed using the
  87         # PlaybookInclude() object
  88         for entry in ds:
  89             if not isinstance(entry, dict):
  90                 # restore the basedir in case this error is caught and handled
  91                 self._loader.set_basedir(cur_basedir)
  92                 raise AnsibleParserError("playbook entries must be either a valid play or an include statement", obj=entry)
  93 
  94             if any(action in entry for action in ('import_playbook', 'include')):
  95                 if 'include' in entry:
  96                     display.deprecated("'include' for playbook includes. You should use 'import_playbook' instead",
  97                                        version="2.12", collection_name='ansible.builtin')
  98                 pb = PlaybookInclude.load(entry, basedir=self._basedir, variable_manager=variable_manager, loader=self._loader)
  99                 if pb is not None:
 100                     self._entries.extend(pb._entries)
 101                 else:
 102                     which = entry.get('import_playbook', entry.get('include', entry))
 103                     display.display("skipping playbook '%s' due to conditional test failure" % which, color=C.COLOR_SKIP)
 104             else:
 105                 entry_obj = Play.load(entry, variable_manager=variable_manager, loader=self._loader, vars=vars)
 106                 self._entries.append(entry_obj)
 107 
 108         # we're done, so restore the old basedir in the loader
 109         self._loader.set_basedir(cur_basedir)
 110 
 111     def get_loader(self):
 112         return self._loader
 113 
 114     def get_plays(self):
 115         return self._entries[:]
