  33 def load_list_of_blocks(ds, play, parent_block=None, role=None, task_include=None, use_handlers=False, variable_manager=None, loader=None):
  34     '''
  35     Given a list of mixed task/block data (parsed from YAML),
  36     return a list of Block() objects, where implicit blocks
  37     are created for each bare Task.
  38     '''
  39 
  40     # we import here to prevent a circular dependency with imports
  41     from ansible.playbook.block import Block
  42 
  43     if not isinstance(ds, (list, type(None))):
  44         raise AnsibleAssertionError('%s should be a list or None but is %s' % (ds, type(ds)))
  45 
  46     block_list = []
  47     if ds:
  48         count = iter(range(len(ds)))
  49         for i in count:
  50             block_ds = ds[i]
  51             # Implicit blocks are created by bare tasks listed in a play without
  52             # an explicit block statement. If we have two implicit blocks in a row,
  53             # squash them down to a single block to save processing time later.
  54             implicit_blocks = []
  55             while block_ds is not None and not Block.is_block(block_ds):
  56                 implicit_blocks.append(block_ds)
  57                 i += 1
  58                 # Advance the iterator, so we don't repeat
  59                 next(count, None)
  60                 try:
  61                     block_ds = ds[i]
  62                 except IndexError:
  63                     block_ds = None
  64 
  65             # Loop both implicit blocks and block_ds as block_ds is the next in the list
  66             for b in (implicit_blocks, block_ds):
  67                 if b:
  68                     block_list.append(
  69                         Block.load(
  70                             b,
  71                             play=play,
  72                             parent_block=parent_block,
  73                             role=role,
  74                             task_include=task_include,
  75                             use_handlers=use_handlers,
  76                             variable_manager=variable_manager,
  77                             loader=loader,
  78                         )
  79                     )
  80 
  81     return block_list

  84 def load_list_of_tasks(ds, play, block=None, role=None, task_include=None, use_handlers=False, variable_manager=None, loader=None):
  85     '''
  86     Given a list of task datastructures (parsed from YAML),
  87     return a list of Task() or TaskInclude() objects.
  88     '''
  89 
  90     # we import here to prevent a circular dependency with imports
  91     from ansible.playbook.block import Block
  92     from ansible.playbook.handler import Handler
  93     from ansible.playbook.task import Task
  94     from ansible.playbook.task_include import TaskInclude
  95     from ansible.playbook.role_include import IncludeRole
  96     from ansible.playbook.handler_task_include import HandlerTaskInclude
  97     from ansible.template import Templar
  98 
  99     if not isinstance(ds, list):
 100         raise AnsibleAssertionError('The ds (%s) should be a list but was a %s' % (ds, type(ds)))
 101 
 102     task_list = []
 103     for task_ds in ds:
 104         if not isinstance(task_ds, dict):
 105             raise AnsibleAssertionError('The ds (%s) should be a dict but was a %s' % (ds, type(ds)))
 106 
 107         if 'block' in task_ds:
 108             t = Block.load(
 109                 task_ds,
 110                 play=play,
 111                 parent_block=block,
 112                 role=role,
 113                 task_include=task_include,
 114                 use_handlers=use_handlers,
 115                 variable_manager=variable_manager,
 116                 loader=loader,
 117             )
 118             task_list.append(t)
 119         else:
 120             args_parser = ModuleArgsParser(task_ds)
 121             try:
 122                 (action, args, delegate_to) = args_parser.parse(skip_action_validation=True)
 123             except AnsibleParserError as e:
 124                 # if the raises exception was created with obj=ds args, then it includes the detail
 125                 # so we dont need to add it so we can just re raise.
 126                 if e._obj:
 127                     raise
 128                 # But if it wasn't, we can add the yaml object now to get more detail
 129                 raise AnsibleParserError(to_native(e), obj=task_ds, orig_exc=e)
 130 
 131             if action in ('include', 'import_tasks', 'include_tasks'):
 132 
 133                 if use_handlers:
 134                     include_class = HandlerTaskInclude
 135                 else:
 136                     include_class = TaskInclude
 137 
 138                 t = include_class.load(
 139                     task_ds,
 140                     block=block,
 141                     role=role,
 142                     task_include=None,
 143                     variable_manager=variable_manager,
 144                     loader=loader
 145                 )
 146 
 147                 all_vars = variable_manager.get_vars(play=play, task=t)
 148                 templar = Templar(loader=loader, variables=all_vars)
 149 
 150                 # check to see if this include is dynamic or static:
 151                 # 1. the user has set the 'static' option to false or true
 152                 # 2. one of the appropriate config options was set
 153                 if action == 'include_tasks':
 154                     is_static = False
 155                 elif action == 'import_tasks':
 156                     is_static = True
 157                 elif t.static is not None:
 158                     display.deprecated("The use of 'static' has been deprecated. "
 159                                        "Use 'import_tasks' for static inclusion, or 'include_tasks' for dynamic inclusion",
 160                                        version='2.12', collection_name='ansible.builtin')
 161                     is_static = t.static
 162                 else:
 163                     is_static = C.DEFAULT_TASK_INCLUDES_STATIC or \
 164                         (use_handlers and C.DEFAULT_HANDLER_INCLUDES_STATIC) or \
 165                         (not templar.is_template(t.args['_raw_params']) and t.all_parents_static() and not t.loop)
 166 
 167                 if is_static:
 168                     if t.loop is not None:
 169                         if action == 'import_tasks':
 170                             raise AnsibleParserError("You cannot use loops on 'import_tasks' statements. You should use 'include_tasks' instead.", obj=task_ds)
 171                         else:
 172                             raise AnsibleParserError("You cannot use 'static' on an include with a loop", obj=task_ds)
 173 
 174                     # we set a flag to indicate this include was static
 175                     t.statically_loaded = True
 176 
 177                     # handle relative includes by walking up the list of parent include
 178                     # tasks and checking the relative result to see if it exists
 179                     parent_include = block
 180                     cumulative_path = None
 181 
 182                     found = False
 183                     subdir = 'tasks'
 184                     if use_handlers:
 185                         subdir = 'handlers'
 186                     while parent_include is not None:
 187                         if not isinstance(parent_include, TaskInclude):
 188                             parent_include = parent_include._parent
 189                             continue
 190                         try:
 191                             parent_include_dir = os.path.dirname(templar.template(parent_include.args.get('_raw_params')))
 192                         except AnsibleUndefinedVariable as e:
 193                             if not parent_include.statically_loaded:
 194                                 raise AnsibleParserError(
 195                                     "Error when evaluating variable in dynamic parent include path: %s. "
 196                                     "When using static imports, the parent dynamic include cannot utilize host facts "
 197                                     "or variables from inventory" % parent_include.args.get('_raw_params'),
 198                                     obj=task_ds,
 199                                     suppress_extended_error=True,
 200                                     orig_exc=e
 201                                 )
 202                             raise
 203                         if cumulative_path is None:
 204                             cumulative_path = parent_include_dir
 205                         elif not os.path.isabs(cumulative_path):
 206                             cumulative_path = os.path.join(parent_include_dir, cumulative_path)
 207                         include_target = templar.template(t.args['_raw_params'])
 208                         if t._role:
 209                             new_basedir = os.path.join(t._role._role_path, subdir, cumulative_path)
 210                             include_file = loader.path_dwim_relative(new_basedir, subdir, include_target)
 211                         else:
 212                             include_file = loader.path_dwim_relative(loader.get_basedir(), cumulative_path, include_target)
 213 
 214                         if os.path.exists(include_file):
 215                             found = True
 216                             break
 217                         else:
 218                             parent_include = parent_include._parent
 219 
 220                     if not found:
 221                         try:
 222                             include_target = templar.template(t.args['_raw_params'])
 223                         except AnsibleUndefinedVariable as e:
 224                             raise AnsibleParserError(
 225                                 "Error when evaluating variable in import path: %s.\n\n"
 226                                 "When using static imports, ensure that any variables used in their names are defined in vars/vars_files\n"
 227                                 "or extra-vars passed in from the command line. Static imports cannot use variables from facts or inventory\n"
 228                                 "sources like group or host vars." % t.args['_raw_params'],
 229                                 obj=task_ds,
 230                                 suppress_extended_error=True,
 231                                 orig_exc=e)
 232                         if t._role:
 233                             include_file = loader.path_dwim_relative(t._role._role_path, subdir, include_target)
 234                         else:
 235                             include_file = loader.path_dwim(include_target)
 236 
 237                     try:
 238                         data = loader.load_from_file(include_file)
 239                         if data is None:
 240                             display.warning('file %s is empty and had no tasks to include' % include_file)
 241                             continue
 242                         elif not isinstance(data, list):
 243                             raise AnsibleParserError("included task files must contain a list of tasks", obj=data)
 244 
 245                         # since we can't send callbacks here, we display a message directly in
 246                         # the same fashion used by the on_include callback. We also do it here,
 247                         # because the recursive nature of helper methods means we may be loading
 248                         # nested includes, and we want the include order printed correctly
 249                         display.vv("statically imported: %s" % include_file)
 250                     except AnsibleFileNotFound:
 251                         if action != 'include' or t.static or \
 252                            C.DEFAULT_TASK_INCLUDES_STATIC or \
 253                            C.DEFAULT_HANDLER_INCLUDES_STATIC and use_handlers:
 254                             raise
 255                         display.deprecated(
 256                             "Included file '%s' not found, however since this include is not "
 257                             "explicitly marked as 'static: yes', we will try and include it dynamically "
 258                             "later. In the future, this will be an error unless 'static: no' is used "
 259                             "on the include task. If you do not want missing includes to be considered "
 260                             "dynamic, use 'static: yes' on the include or set the global ansible.cfg "
 261                             "options to make all includes static for tasks and/or handlers" % include_file,
 262                             version="2.12", collection_name='ansible.builtin'
 263                         )
 264                         task_list.append(t)
 265                         continue
 266 
 267                     ti_copy = t.copy(exclude_parent=True)
 268                     ti_copy._parent = block
 269                     included_blocks = load_list_of_blocks(
 270                         data,
 271                         play=play,
 272                         parent_block=None,
 273                         task_include=ti_copy,
 274                         role=role,
 275                         use_handlers=use_handlers,
 276                         loader=loader,
 277                         variable_manager=variable_manager,
 278                     )
 279 
 280                     # FIXME: remove once 'include' is removed
 281                     # pop tags out of the include args, if they were specified there, and assign
 282                     # them to the include. If the include already had tags specified, we raise an
 283                     # error so that users know not to specify them both ways
 284                     tags = ti_copy.vars.pop('tags', [])
 285                     if isinstance(tags, string_types):
 286                         tags = tags.split(',')
 287 
 288                     if len(tags) > 0:
 289                         if action in ('include_tasks', 'import_tasks'):
 290                             raise AnsibleParserError('You cannot specify "tags" inline to the task, it is a task keyword')
 291                         if len(ti_copy.tags) > 0:
 292                             raise AnsibleParserError(
 293                                 "Include tasks should not specify tags in more than one way (both via args and directly on the task). "
 294                                 "Mixing styles in which tags are specified is prohibited for whole import hierarchy, not only for single import statement",
 295                                 obj=task_ds,
 296                                 suppress_extended_error=True,
 297                             )
 298                         display.deprecated("You should not specify tags in the include parameters. All tags should be specified using the task-level option",
 299                                            version="2.12", collection_name='ansible.builtin')
 300                     else:
 301                         tags = ti_copy.tags[:]
 302 
 303                     # now we extend the tags on each of the included blocks
 304                     for b in included_blocks:
 305                         b.tags = list(set(b.tags).union(tags))
 306                     # END FIXME
 307 
 308                     # FIXME: handlers shouldn't need this special handling, but do
 309                     #        right now because they don't iterate blocks correctly
 310                     if use_handlers:
 311                         for b in included_blocks:
 312                             task_list.extend(b.block)
 313                     else:
 314                         task_list.extend(included_blocks)
 315                 else:
 316                     t.is_static = False
 317                     task_list.append(t)
 318 
 319             elif action in ('include_role', 'import_role'):
 320                 ir = IncludeRole.load(
 321                     task_ds,
 322                     block=block,
 323                     role=role,
 324                     task_include=None,
 325                     variable_manager=variable_manager,
 326                     loader=loader,
 327                 )
 328 
 329                 #   1. the user has set the 'static' option to false or true
 330                 #   2. one of the appropriate config options was set
 331                 is_static = False
 332                 if action == 'import_role':
 333                     is_static = True
 334 
 335                 elif ir.static is not None:
 336                     display.deprecated("The use of 'static' for 'include_role' has been deprecated. "
 337                                        "Use 'import_role' for static inclusion, or 'include_role' for dynamic inclusion",
 338                                        version='2.12', collection_name='ansible.builtin')
 339                     is_static = ir.static
 340 
 341                 if is_static:
 342                     if ir.loop is not None:
 343                         if action == 'import_role':
 344                             raise AnsibleParserError("You cannot use loops on 'import_role' statements. You should use 'include_role' instead.", obj=task_ds)
 345                         else:
 346                             raise AnsibleParserError("You cannot use 'static' on an include_role with a loop", obj=task_ds)
 347 
 348                     # we set a flag to indicate this include was static
 349                     ir.statically_loaded = True
 350 
 351                     # template the role name now, if needed
 352                     all_vars = variable_manager.get_vars(play=play, task=ir)
 353                     templar = Templar(loader=loader, variables=all_vars)
 354                     ir._role_name = templar.template(ir._role_name)
 355 
 356                     # uses compiled list from object
 357                     blocks, _ = ir.get_block_list(variable_manager=variable_manager, loader=loader)
 358                     task_list.extend(blocks)
 359                 else:
 360                     # passes task object itself for latter generation of list
 361                     task_list.append(ir)
 362             else:
 363                 if use_handlers:
 364                     t = Handler.load(task_ds, block=block, role=role, task_include=task_include, variable_manager=variable_manager, loader=loader)
 365                 else:
 366                     t = Task.load(task_ds, block=block, role=role, task_include=task_include, variable_manager=variable_manager, loader=loader)
 367 
 368                 task_list.append(t)
 369 
 370     return task_list

 373 def load_list_of_roles(ds, play, current_role_path=None, variable_manager=None, loader=None, collection_search_list=None):
 374     """
 375     Loads and returns a list of RoleInclude objects from the ds list of role definitions
 376     :param ds: list of roles to load
 377     :param play: calling Play object
 378     :param current_role_path: path of the owning role, if any
 379     :param variable_manager: varmgr to use for templating
 380     :param loader: loader to use for DS parsing/services
 381     :param collection_search_list: list of collections to search for unqualified role names
 382     :return:
 383     """
 384     # we import here to prevent a circular dependency with imports
 385     from ansible.playbook.role.include import RoleInclude
 386 
 387     if not isinstance(ds, list):
 388         raise AnsibleAssertionError('ds (%s) should be a list but was a %s' % (ds, type(ds)))
 389 
 390     roles = []
 391     for role_def in ds:
 392         i = RoleInclude.load(role_def, play=play, current_role_path=current_role_path, variable_manager=variable_manager,
 393                              loader=loader, collection_list=collection_search_list)
 394         roles.append(i)
 395 
 396     return roles
