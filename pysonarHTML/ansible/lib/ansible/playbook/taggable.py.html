  22 from ansible.errors import AnsibleError
  23 from ansible.module_utils.six import string_types
  24 from ansible.playbook.attribute import FieldAttribute
  25 from ansible.template import Templar
  26 
  27 
  28 class Taggable:
  29 
  30     untagged = frozenset(['untagged'])
  31     _tags = FieldAttribute(isa='list', default=list, listof=(string_types, int), extend=True)
  32 
  33     def _load_tags(self, attr, ds):
  34         if isinstance(ds, list):
  35             return ds
  36         elif isinstance(ds, string_types):
  37             value = ds.split(',')
  38             if isinstance(value, list):
  39                 return [x.strip() for x in value]
  40             else:
  41                 return [ds]
  42         else:
  43             raise AnsibleError('tags must be specified as a list', obj=ds)
  44 
  45     def evaluate_tags(self, only_tags, skip_tags, all_vars):
  46         ''' this checks if the current item should be executed depending on tag options '''
  47 
  48         if self.tags:
  49             templar = Templar(loader=self._loader, variables=all_vars)
  50             tags = templar.template(self.tags)
  51 
  52             _temp_tags = set()
  53             for tag in tags:
  54                 if isinstance(tag, list):
  55                     _temp_tags.update(tag)
  56                 else:
  57                     _temp_tags.add(tag)
  58             tags = _temp_tags
  59             self.tags = list(tags)
  60         else:
  61             # this makes isdisjoint work for untagged
  62             tags = self.untagged
  63 
  64         should_run = True  # default, tasks to run
  65 
  66         if only_tags:
  67             if 'always' in tags:
  68                 should_run = True
  69             elif ('all' in only_tags and 'never' not in tags):
  70                 should_run = True
  71             elif not tags.isdisjoint(only_tags):
  72                 should_run = True
  73             elif 'tagged' in only_tags and tags != self.untagged and 'never' not in tags:
  74                 should_run = True
  75             else:
  76                 should_run = False
  77 
  78         if should_run and skip_tags:
  79 
  80             # Check for tags that we need to skip
  81             if 'all' in skip_tags:
  82                 if 'always' not in tags or 'always' in skip_tags:
  83                     should_run = False
  84             elif not tags.isdisjoint(skip_tags):
  85                 should_run = False
  86             elif 'tagged' in skip_tags and tags != self.untagged:
  87                 should_run = False
  88 
  89         return should_run
