  37 DEFINED_REGEX = re.compile(r'(hostvars\[.+\]|[\w_]+)\s+(not\s+is|is|is\s+not)\s+(defined|undefined)')
  38 LOOKUP_REGEX = re.compile(r'lookup\s*\(')
  39 VALID_VAR_REGEX = re.compile("^[_A-Za-z][_a-zA-Z0-9]*$")

  42 class Conditional:
  43 
  44     '''
  45     This is a mix-in class, to be used with Base to allow the object
  46     to be run conditionally when a condition is met or skipped.
  47     '''
  48 
  49     _when = FieldAttribute(isa='list', default=list, extend=True, prepend=True)
  50 
  51     def __init__(self, loader=None):
  52         # when used directly, this class needs a loader, but we want to
  53         # make sure we don't trample on the existing one if this class
  54         # is used as a mix-in with a playbook base class
  55         if not hasattr(self, '_loader'):
  56             if loader is None:
  57                 raise AnsibleError("a loader must be specified when using Conditional() directly")
  58             else:
  59                 self._loader = loader
  60         super(Conditional, self).__init__()
  61 
  62     def _validate_when(self, attr, name, value):
  63         if not isinstance(value, list):
  64             setattr(self, name, [value])
  65 
  66     def extract_defined_undefined(self, conditional):
  67         results = []
  68 
  69         cond = conditional
  70         m = DEFINED_REGEX.search(cond)
  71         while m:
  72             results.append(m.groups())
  73             cond = cond[m.end():]
  74             m = DEFINED_REGEX.search(cond)
  75 
  76         return results
  77 
  78     def evaluate_conditional(self, templar, all_vars):
  79         '''
  80         Loops through the conditionals set on this object, returning
  81         False if any of them evaluate as such.
  82         '''
  83 
  84         # since this is a mix-in, it may not have an underlying datastructure
  85         # associated with it, so we pull it out now in case we need it for
  86         # error reporting below
  87         ds = None
  88         if hasattr(self, '_ds'):
  89             ds = getattr(self, '_ds')
  90 
  91         try:
  92             for conditional in self.when:
  93                 if not self._check_conditional(conditional, templar, all_vars):
  94                     return False
  95         except Exception as e:
  96             raise AnsibleError(
  97                 "The conditional check '%s' failed. The error was: %s" % (to_native(conditional), to_native(e)), obj=ds
  98             )
  99 
 100         return True
 101 
 102     def _check_conditional(self, conditional, templar, all_vars):
 103         '''
 104         This method does the low-level evaluation of each conditional
 105         set on this object, using jinja2 to wrap the conditionals for
 106         evaluation.
 107         '''
 108 
 109         original = conditional
 110         if conditional is None or conditional == '':
 111             return True
 112 
 113         # this allows for direct boolean assignments to conditionals "when: False"
 114         if isinstance(conditional, bool):
 115             return conditional
 116 
 117         if templar.is_template(conditional):
 118             display.warning('conditional statements should not include jinja2 '
 119                             'templating delimiters such as {{ }} or {%% %%}. '
 120                             'Found: %s' % conditional)
 121 
 122         bare_vars_warning = False
 123         if C.CONDITIONAL_BARE_VARS:
 124             if conditional in all_vars and VALID_VAR_REGEX.match(conditional):
 125                 conditional = all_vars[conditional]
 126                 bare_vars_warning = True
 127 
 128         # make sure the templar is using the variables specified with this method
 129         templar.available_variables = all_vars
 130 
 131         try:
 132             # if the conditional is "unsafe", disable lookups
 133             disable_lookups = hasattr(conditional, '__UNSAFE__')
 134             conditional = templar.template(conditional, disable_lookups=disable_lookups)
 135             if bare_vars_warning and not isinstance(conditional, bool):
 136                 display.deprecated('evaluating %r as a bare variable, this behaviour will go away and you might need to add |bool'
 137                                    ' to the expression in the future. Also see CONDITIONAL_BARE_VARS configuration toggle' % original,
 138                                    version="2.12", collection_name='ansible.builtin')
 139             if not isinstance(conditional, text_type) or conditional == "":
 140                 return conditional
 141 
 142             # update the lookups flag, as the string returned above may now be unsafe
 143             # and we don't want future templating calls to do unsafe things
 144             disable_lookups |= hasattr(conditional, '__UNSAFE__')
 145 
 146             # First, we do some low-level jinja2 parsing involving the AST format of the
 147             # statement to ensure we don't do anything unsafe (using the disable_lookup flag above)
 148             class CleansingNodeVisitor(ast.NodeVisitor):
 149                 def generic_visit(self, node, inside_call=False, inside_yield=False):
 150                     if isinstance(node, ast.Call):
 151                         inside_call = True
 152                     elif isinstance(node, ast.Yield):
 153                         inside_yield = True
 154                     elif isinstance(node, ast.Str):
 155                         if disable_lookups:
 156                             if inside_call and node.s.startswith("__"):
 157                                 # calling things with a dunder is generally bad at this point...
 158                                 raise AnsibleError(
 159                                     "Invalid access found in the conditional: '%s'" % conditional
 160                                 )
 161                             elif inside_yield:
 162                                 # we're inside a yield, so recursively parse and traverse the AST
 163                                 # of the result to catch forbidden syntax from executing
 164                                 parsed = ast.parse(node.s, mode='exec')
 165                                 cnv = CleansingNodeVisitor()
 166                                 cnv.visit(parsed)
 167                     # iterate over all child nodes
 168                     for child_node in ast.iter_child_nodes(node):
 169                         self.generic_visit(
 170                             child_node,
 171                             inside_call=inside_call,
 172                             inside_yield=inside_yield
 173                         )
 174             try:
 175                 e = templar.environment.overlay()
 176                 e.filters.update(templar.environment.filters)
 177                 e.tests.update(templar.environment.tests)
 178 
 179                 res = e._parse(conditional, None, None)
 180                 res = generate(res, e, None, None)
 181                 parsed = ast.parse(res, mode='exec')
 182 
 183                 cnv = CleansingNodeVisitor()
 184                 cnv.visit(parsed)
 185             except Exception as e:
 186                 raise AnsibleError("Invalid conditional detected: %s" % to_native(e))
 187 
 188             # and finally we generate and template the presented string and look at the resulting string
 189             presented = "{%% if %s %%} True {%% else %%} False {%% endif %%}" % conditional
 190             val = templar.template(presented, disable_lookups=disable_lookups).strip()
 191             if val == "True":
 192                 return True
 193             elif val == "False":
 194                 return False
 195             else:
 196                 raise AnsibleError("unable to evaluate conditional: %s" % original)
 197         except (AnsibleUndefinedVariable, UndefinedError) as e:
 198             # the templating failed, meaning most likely a variable was undefined. If we happened
 199             # to be looking for an undefined variable, return True, otherwise fail
 200             try:
 201                 # first we extract the variable name from the error message
 202                 var_name = re.compile(r"'(hostvars\[.+\]|[\w_]+)' is undefined").search(str(e)).groups()[0]
 203                 # next we extract all defined/undefined tests from the conditional string
 204                 def_undef = self.extract_defined_undefined(conditional)
 205                 # then we loop through these, comparing the error variable name against
 206                 # each def/undef test we found above. If there is a match, we determine
 207                 # whether the logic/state mean the variable should exist or not and return
 208                 # the corresponding True/False
 209                 for (du_var, logic, state) in def_undef:
 210                     # when we compare the var names, normalize quotes because something
 211                     # like hostvars['foo'] may be tested against hostvars["foo"]
 212                     if var_name.replace("'", '"') == du_var.replace("'", '"'):
 213                         # the should exist is a xor test between a negation in the logic portion
 214                         # against the state (defined or undefined)
 215                         should_exist = ('not' in logic) != (state == 'defined')
 216                         if should_exist:
 217                             return False
 218                         else:
 219                             return True
 220                 # as nothing above matched the failed var name, re-raise here to
 221                 # trigger the AnsibleUndefinedVariable exception again below
 222                 raise
 223             except Exception:
 224                 raise AnsibleUndefinedVariable("error while evaluating conditional (%s): %s" % (original, e))
