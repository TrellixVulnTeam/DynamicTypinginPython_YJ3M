  34 class IncludedFile:
  35 
  36     def __init__(self, filename, args, vars, task, is_role=False):
  37         self._filename = filename
  38         self._args = args
  39         self._vars = vars
  40         self._task = task
  41         self._hosts = []
  42         self._is_role = is_role
  43 
  44     def add_host(self, host):
  45         if host not in self._hosts:
  46             self._hosts.append(host)
  47             return
  48         raise ValueError()
  49 
  50     def __eq__(self, other):
  51         return (other._filename == self._filename and
  52                 other._args == self._args and
  53                 other._vars == self._vars and
  54                 other._task._uuid == self._task._uuid and
  55                 other._task._parent._uuid == self._task._parent._uuid)
  56 
  57     def __repr__(self):
  58         return "%s (args=%s vars=%s): %s" % (self._filename, self._args, self._vars, self._hosts)
  59 
  60     @staticmethod
  61     def process_include_results(results, iterator, loader, variable_manager):
  62         included_files = []
  63         task_vars_cache = {}
  64 
  65         for res in results:
  66 
  67             original_host = res._host
  68             original_task = res._task
  69 
  70             if original_task.action in ('include', 'include_tasks', 'include_role'):
  71                 if original_task.loop:
  72                     if 'results' not in res._result:
  73                         continue
  74                     include_results = res._result['results']
  75                 else:
  76                     include_results = [res._result]
  77 
  78                 for include_result in include_results:
  79                     # if the task result was skipped or failed, continue
  80                     if 'skipped' in include_result and include_result['skipped'] or 'failed' in include_result and include_result['failed']:
  81                         continue
  82 
  83                     cache_key = (iterator._play, original_host, original_task)
  84                     try:
  85                         task_vars = task_vars_cache[cache_key]
  86                     except KeyError:
  87                         task_vars = task_vars_cache[cache_key] = variable_manager.get_vars(play=iterator._play, host=original_host, task=original_task)
  88 
  89                     include_args = include_result.get('include_args', dict())
  90                     special_vars = {}
  91                     loop_var = include_result.get('ansible_loop_var', 'item')
  92                     index_var = include_result.get('ansible_index_var')
  93                     if loop_var in include_result:
  94                         task_vars[loop_var] = special_vars[loop_var] = include_result[loop_var]
  95                     if index_var and index_var in include_result:
  96                         task_vars[index_var] = special_vars[index_var] = include_result[index_var]
  97                     if '_ansible_item_label' in include_result:
  98                         task_vars['_ansible_item_label'] = special_vars['_ansible_item_label'] = include_result['_ansible_item_label']
  99                     if 'ansible_loop' in include_result:
 100                         task_vars['ansible_loop'] = special_vars['ansible_loop'] = include_result['ansible_loop']
 101                     if original_task.no_log and '_ansible_no_log' not in include_args:
 102                         task_vars['_ansible_no_log'] = special_vars['_ansible_no_log'] = original_task.no_log
 103 
 104                     # get search path for this task to pass to lookup plugins that may be used in pathing to
 105                     # the included file
 106                     task_vars['ansible_search_path'] = original_task.get_search_path()
 107 
 108                     # ensure basedir is always in (dwim already searches here but we need to display it)
 109                     if loader.get_basedir() not in task_vars['ansible_search_path']:
 110                         task_vars['ansible_search_path'].append(loader.get_basedir())
 111 
 112                     templar = Templar(loader=loader, variables=task_vars)
 113 
 114                     if original_task.action in ('include', 'include_tasks'):
 115                         include_file = None
 116                         if original_task:
 117                             if original_task.static:
 118                                 continue
 119 
 120                             if original_task._parent:
 121                                 # handle relative includes by walking up the list of parent include
 122                                 # tasks and checking the relative result to see if it exists
 123                                 parent_include = original_task._parent
 124                                 cumulative_path = None
 125                                 while parent_include is not None:
 126                                     if not isinstance(parent_include, TaskInclude):
 127                                         parent_include = parent_include._parent
 128                                         continue
 129                                     if isinstance(parent_include, IncludeRole):
 130                                         parent_include_dir = parent_include._role_path
 131                                     else:
 132                                         try:
 133                                             parent_include_dir = os.path.dirname(templar.template(parent_include.args.get('_raw_params')))
 134                                         except AnsibleError as e:
 135                                             parent_include_dir = ''
 136                                             display.warning(
 137                                                 'Templating the path of the parent %s failed. The path to the '
 138                                                 'included file may not be found. '
 139                                                 'The error was: %s.' % (original_task.action, to_text(e))
 140                                             )
 141                                     if cumulative_path is not None and not os.path.isabs(cumulative_path):
 142                                         cumulative_path = os.path.join(parent_include_dir, cumulative_path)
 143                                     else:
 144                                         cumulative_path = parent_include_dir
 145                                     include_target = templar.template(include_result['include'])
 146                                     if original_task._role:
 147                                         new_basedir = os.path.join(original_task._role._role_path, 'tasks', cumulative_path)
 148                                         candidates = [loader.path_dwim_relative(original_task._role._role_path, 'tasks', include_target),
 149                                                       loader.path_dwim_relative(new_basedir, 'tasks', include_target)]
 150                                         for include_file in candidates:
 151                                             try:
 152                                                 # may throw OSError
 153                                                 os.stat(include_file)
 154                                                 # or select the task file if it exists
 155                                                 break
 156                                             except OSError:
 157                                                 pass
 158                                     else:
 159                                         include_file = loader.path_dwim_relative(loader.get_basedir(), cumulative_path, include_target)
 160 
 161                                     if os.path.exists(include_file):
 162                                         break
 163                                     else:
 164                                         parent_include = parent_include._parent
 165 
 166                         if include_file is None:
 167                             if original_task._role:
 168                                 include_target = templar.template(include_result['include'])
 169                                 include_file = loader.path_dwim_relative(original_task._role._role_path, 'tasks', include_target)
 170                             else:
 171                                 include_file = loader.path_dwim(include_result['include'])
 172 
 173                         include_file = templar.template(include_file)
 174                         inc_file = IncludedFile(include_file, include_args, special_vars, original_task)
 175                     else:
 176                         # template the included role's name here
 177                         role_name = include_args.pop('name', include_args.pop('role', None))
 178                         if role_name is not None:
 179                             role_name = templar.template(role_name)
 180 
 181                         new_task = original_task.copy()
 182                         new_task._role_name = role_name
 183                         for from_arg in new_task.FROM_ARGS:
 184                             if from_arg in include_args:
 185                                 from_key = from_arg.replace('_from', '')
 186                                 new_task._from_files[from_key] = templar.template(include_args.pop(from_arg))
 187 
 188                         inc_file = IncludedFile(role_name, include_args, special_vars, new_task, is_role=True)
 189 
 190                     idx = 0
 191                     orig_inc_file = inc_file
 192                     while 1:
 193                         try:
 194                             pos = included_files[idx:].index(orig_inc_file)
 195                             # pos is relative to idx since we are slicing
 196                             # use idx + pos due to relative indexing
 197                             inc_file = included_files[idx + pos]
 198                         except ValueError:
 199                             included_files.append(orig_inc_file)
 200                             inc_file = orig_inc_file
 201 
 202                         try:
 203                             inc_file.add_host(original_host)
 204                         except ValueError:
 205                             # The host already exists for this include, advance forward, this is a new include
 206                             idx += pos + 1
 207                         else:
 208                             break
 209 
 210         return included_files
