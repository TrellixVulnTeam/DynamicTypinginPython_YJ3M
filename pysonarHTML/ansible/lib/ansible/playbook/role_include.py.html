  37 class IncludeRole(TaskInclude):
  38 
  39     """
  40     A Role include is derived from a regular role to handle the special
  41     circumstances related to the `- include_role: ...`
  42     """
  43 
  44     BASE = ('name', 'role')  # directly assigned
  45     FROM_ARGS = ('tasks_from', 'vars_from', 'defaults_from', 'handlers_from')  # used to populate from dict in role
  46     OTHER_ARGS = ('apply', 'public', 'allow_duplicates')  # assigned to matching property
  47     VALID_ARGS = tuple(frozenset(BASE + FROM_ARGS + OTHER_ARGS))  # all valid args
  48 
  49     # =================================================================================
  50     # ATTRIBUTES
  51 
  52     # private as this is a 'module options' vs a task property
  53     _allow_duplicates = FieldAttribute(isa='bool', default=True, private=True)
  54     _public = FieldAttribute(isa='bool', default=False, private=True)
  55 
  56     def __init__(self, block=None, role=None, task_include=None):
  57 
  58         super(IncludeRole, self).__init__(block=block, role=role, task_include=task_include)
  59 
  60         self._from_files = {}
  61         self._parent_role = role
  62         self._role_name = None
  63         self._role_path = None
  64 
  65     def get_name(self):
  66         ''' return the name of the task '''
  67         return self.name or "%s : %s" % (self.action, self._role_name)
  68 
  69     def get_block_list(self, play=None, variable_manager=None, loader=None):
  70 
  71         # only need play passed in when dynamic
  72         if play is None:
  73             myplay = self._parent._play
  74         else:
  75             myplay = play
  76 
  77         ri = RoleInclude.load(self._role_name, play=myplay, variable_manager=variable_manager, loader=loader, collection_list=self.collections)
  78         ri.vars.update(self.vars)
  79 
  80         # build role
  81         actual_role = Role.load(ri, myplay, parent_role=self._parent_role, from_files=self._from_files,
  82                                 from_include=True)
  83         actual_role._metadata.allow_duplicates = self.allow_duplicates
  84 
  85         if self.statically_loaded or self.public:
  86             myplay.roles.append(actual_role)
  87 
  88         # save this for later use
  89         self._role_path = actual_role._role_path
  90 
  91         # compile role with parent roles as dependencies to ensure they inherit
  92         # variables
  93         if not self._parent_role:
  94             dep_chain = []
  95         else:
  96             dep_chain = list(self._parent_role._parents)
  97             dep_chain.append(self._parent_role)
  98 
  99         p_block = self.build_parent_block()
 100 
 101         # collections value is not inherited; override with the value we calculated during role setup
 102         p_block.collections = actual_role.collections
 103 
 104         blocks = actual_role.compile(play=myplay, dep_chain=dep_chain)
 105         for b in blocks:
 106             b._parent = p_block
 107             # HACK: parent inheritance doesn't seem to have a way to handle this intermediate override until squashed/finalized
 108             b.collections = actual_role.collections
 109 
 110         # updated available handlers in play
 111         handlers = actual_role.get_handler_blocks(play=myplay)
 112         for h in handlers:
 113             h._parent = p_block
 114         myplay.handlers = myplay.handlers + handlers
 115         return blocks, handlers

 117     @staticmethod
 118     def load(data, block=None, role=None, task_include=None, variable_manager=None, loader=None):
 119 
 120         ir = IncludeRole(block, role, task_include=task_include).load_data(data, variable_manager=variable_manager, loader=loader)
 121 
 122         # Validate options
 123         my_arg_names = frozenset(ir.args.keys())
 124 
 125         # name is needed, or use role as alias
 126         ir._role_name = ir.args.get('name', ir.args.get('role'))
 127         if ir._role_name is None:
 128             raise AnsibleParserError("'name' is a required field for %s." % ir.action, obj=data)
 129 
 130         if 'public' in ir.args and ir.action != 'include_role':
 131             raise AnsibleParserError('Invalid options for %s: public' % ir.action, obj=data)
 132 
 133         # validate bad args, otherwise we silently ignore
 134         bad_opts = my_arg_names.difference(IncludeRole.VALID_ARGS)
 135         if bad_opts:
 136             raise AnsibleParserError('Invalid options for %s: %s' % (ir.action, ','.join(list(bad_opts))), obj=data)
 137 
 138         # build options for role includes
 139         for key in my_arg_names.intersection(IncludeRole.FROM_ARGS):
 140             from_key = key.replace('_from', '')
 141             args_value = ir.args.get(key)
 142             if not isinstance(args_value, string_types):
 143                 raise AnsibleParserError('Expected a string for %s but got %s instead' % (key, type(args_value)))
 144             ir._from_files[from_key] = basename(args_value)
 145 
 146         apply_attrs = ir.args.get('apply', {})
 147         if apply_attrs and ir.action != 'include_role':
 148             raise AnsibleParserError('Invalid options for %s: apply' % ir.action, obj=data)
 149         elif not isinstance(apply_attrs, dict):
 150             raise AnsibleParserError('Expected a dict for apply but got %s instead' % type(apply_attrs), obj=data)
 151 
 152         # manual list as otherwise the options would set other task parameters we don't want.
 153         for option in my_arg_names.intersection(IncludeRole.OTHER_ARGS):
 154             setattr(ir, option, ir.args.get(option))
 155 
 156         return ir
 157 
 158     def copy(self, exclude_parent=False, exclude_tasks=False):
 159 
 160         new_me = super(IncludeRole, self).copy(exclude_parent=exclude_parent, exclude_tasks=exclude_tasks)
 161         new_me.statically_loaded = self.statically_loaded
 162         new_me._from_files = self._from_files.copy()
 163         new_me._parent_role = self._parent_role
 164         new_me._role_name = self._role_name
 165         new_me._role_path = self._role_path
 166 
 167         return new_me

 169     def get_include_params(self):
 170         v = super(IncludeRole, self).get_include_params()
 171         if self._parent_role:
 172             v.update(self._parent_role.get_role_params())
 173             v.setdefault('ansible_parent_role_names', []).insert(0, self._parent_role.get_name())
 174             v.setdefault('ansible_parent_role_paths', []).insert(0, self._parent_role._role_path)
 175         return v
