  35 class TaskInclude(Task):
  36 
  37     """
  38     A task include is derived from a regular task to handle the special
  39     circumstances related to the `- include: ...` task.
  40     """
  41 
  42     BASE = frozenset(('file', '_raw_params'))  # directly assigned
  43     OTHER_ARGS = frozenset(('apply',))  # assigned to matching property
  44     VALID_ARGS = BASE.union(OTHER_ARGS)  # all valid args
  45     VALID_INCLUDE_KEYWORDS = frozenset(('action', 'args', 'collections', 'debugger', 'ignore_errors', 'loop', 'loop_control',
  46                                         'loop_with', 'name', 'no_log', 'register', 'run_once', 'tags', 'vars',
  47                                         'when'))
  48 
  49     # =================================================================================
  50     # ATTRIBUTES
  51 
  52     _static = FieldAttribute(isa='bool', default=None)
  53 
  54     def __init__(self, block=None, role=None, task_include=None):
  55         super(TaskInclude, self).__init__(block=block, role=role, task_include=task_include)
  56         self.statically_loaded = False
  57 
  58     @staticmethod
  59     def load(data, block=None, role=None, task_include=None, variable_manager=None, loader=None):
  60         ti = TaskInclude(block=block, role=role, task_include=task_include)
  61         task = ti.check_options(
  62             ti.load_data(data, variable_manager=variable_manager, loader=loader),
  63             data
  64         )
  65 
  66         return task
  67 
  68     def check_options(self, task, data):
  69         '''
  70         Method for options validation to use in 'load_data' for TaskInclude and HandlerTaskInclude
  71         since they share the same validations. It is not named 'validate_options' on purpose
  72         to prevent confusion with '_validate_*" methods. Note that the task passed might be changed
  73         as a side-effect of this method.
  74         '''
  75         my_arg_names = frozenset(task.args.keys())
  76 
  77         # validate bad args, otherwise we silently ignore
  78         bad_opts = my_arg_names.difference(self.VALID_ARGS)
  79         if bad_opts and task.action in ('include_tasks', 'import_tasks'):
  80             raise AnsibleParserError('Invalid options for %s: %s' % (task.action, ','.join(list(bad_opts))), obj=data)
  81 
  82         if not task.args.get('_raw_params'):
  83             task.args['_raw_params'] = task.args.pop('file', None)
  84             if not task.args['_raw_params']:
  85                 raise AnsibleParserError('No file specified for %s' % task.action)
  86 
  87         apply_attrs = task.args.get('apply', {})
  88         if apply_attrs and task.action != 'include_tasks':
  89             raise AnsibleParserError('Invalid options for %s: apply' % task.action, obj=data)
  90         elif not isinstance(apply_attrs, dict):
  91             raise AnsibleParserError('Expected a dict for apply but got %s instead' % type(apply_attrs), obj=data)
  92 
  93         return task
  94 
  95     def preprocess_data(self, ds):
  96         ds = super(TaskInclude, self).preprocess_data(ds)
  97 
  98         diff = set(ds.keys()).difference(self.VALID_INCLUDE_KEYWORDS)
  99         for k in diff:
 100             # This check doesn't handle ``include`` as we have no idea at this point if it is static or not
 101             if ds[k] is not Sentinel and ds['action'] in ('include_tasks', 'include_role'):
 102                 if C.INVALID_TASK_ATTRIBUTE_FAILED:
 103                     raise AnsibleParserError("'%s' is not a valid attribute for a %s" % (k, self.__class__.__name__), obj=ds)
 104                 else:
 105                     display.warning("Ignoring invalid attribute: %s" % k)
 106 
 107         return ds
 108 
 109     def copy(self, exclude_parent=False, exclude_tasks=False):
 110         new_me = super(TaskInclude, self).copy(exclude_parent=exclude_parent, exclude_tasks=exclude_tasks)
 111         new_me.statically_loaded = self.statically_loaded
 112         return new_me
 113 
 114     def get_vars(self):
 115         '''
 116         We override the parent Task() classes get_vars here because
 117         we need to include the args of the include into the vars as
 118         they are params to the included tasks. But ONLY for 'include'
 119         '''
 120         if self.action != 'include':
 121             all_vars = super(TaskInclude, self).get_vars()
 122         else:
 123             all_vars = dict()
 124             if self._parent:
 125                 all_vars.update(self._parent.get_vars())
 126 
 127             all_vars.update(self.vars)
 128             all_vars.update(self.args)
 129 
 130             if 'tags' in all_vars:
 131                 del all_vars['tags']
 132             if 'when' in all_vars:
 133                 del all_vars['when']
 134 
 135         return all_vars
 136 
 137     def build_parent_block(self):
 138         '''
 139         This method is used to create the parent block for the included tasks
 140         when ``apply`` is specified
 141         '''
 142         apply_attrs = self.args.pop('apply', {})
 143         if apply_attrs:
 144             apply_attrs['block'] = []
 145             p_block = Block.load(
 146                 apply_attrs,
 147                 play=self._parent._play,
 148                 task_include=self,
 149                 role=self._role,
 150                 variable_manager=self._variable_manager,
 151                 loader=self._loader,
 152             )
 153         else:
 154             p_block = self
 155 
 156         return p_block
