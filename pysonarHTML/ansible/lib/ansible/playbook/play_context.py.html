 138     def __init__(self, play=None, passwords=None, connection_lockfd=None):
 139         # Note: play is really not optional.  The only time it could be omitted is when we create
 140         # a PlayContext just so we can invoke its deserialize method to load it from a serialized
 141         # data source.
 142 
 143         super(PlayContext, self).__init__()
 144 
 145         if passwords is None:
 146             passwords = {}
 147 
 148         self.password = passwords.get('conn_pass', '')
 149         self.become_pass = passwords.get('become_pass', '')
 150 
 151         self._become_plugin = None
 152 
 153         self.prompt = ''
 154         self.success_key = ''
 155 
 156         # a file descriptor to be used during locking operations
 157         self.connection_lockfd = connection_lockfd
 158 
 159         # set options before play to allow play to override them
 160         if context.CLIARGS:
 161             self.set_attributes_from_cli()
 162 
 163         if play:
 164             self.set_attributes_from_play(play)

 166     def set_attributes_from_plugin(self, plugin):
 167         # generic derived from connection plugin, temporary for backwards compat, in the end we should not set play_context properties
 168 
 169         # get options for plugins
 170         options = C.config.get_configuration_definitions(get_plugin_class(plugin), plugin._load_name)
 171         for option in options:
 172             if option:
 173                 flag = options[option].get('name')
 174                 if flag:
 175                     setattr(self, flag, self.connection.get_option(flag))

 180     def set_attributes_from_cli(self):
 181         '''
 182         Configures this connection information instance with data from
 183         options specified by the user on the command line. These have a
 184         lower precedence than those set on the play or host.
 185         '''
 186         if context.CLIARGS.get('timeout', False):
 187             self.timeout = int(context.CLIARGS['timeout'])
 188 
 189         # From the command line.  These should probably be used directly by plugins instead
 190         # For now, they are likely to be moved to FieldAttribute defaults
 191         self.private_key_file = context.CLIARGS.get('private_key_file')  # Else default
 192         self.verbosity = context.CLIARGS.get('verbosity')  # Else default
 193         self.ssh_common_args = context.CLIARGS.get('ssh_common_args')  # Else default
 194         self.ssh_extra_args = context.CLIARGS.get('ssh_extra_args')  # Else default
 195         self.sftp_extra_args = context.CLIARGS.get('sftp_extra_args')  # Else default
 196         self.scp_extra_args = context.CLIARGS.get('scp_extra_args')  # Else default
 197 
 198         # Not every cli that uses PlayContext has these command line args so have a default
 199         self.start_at_task = context.CLIARGS.get('start_at_task', None)  # Else default

 201     def set_task_and_variable_override(self, task, variables, templar):
 202         '''
 203         Sets attributes from the task if they are set, which will override
 204         those from the play.
 205 
 206         :arg task: the task object with the parameters that were set on it
 207         :arg variables: variables from inventory
 208         :arg templar: templar instance if templating variables is needed
 209         '''
 210 
 211         new_info = self.copy()
 212 
 213         # loop through a subset of attributes on the task object and set
 214         # connection fields based on their values
 215         for attr in TASK_ATTRIBUTE_OVERRIDES:
 216             if hasattr(task, attr):
 217                 attr_val = getattr(task, attr)
 218                 if attr_val is not None:
 219                     setattr(new_info, attr, attr_val)
 220 
 221         # next, use the MAGIC_VARIABLE_MAPPING dictionary to update this
 222         # connection info object with 'magic' variables from the variable list.
 223         # If the value 'ansible_delegated_vars' is in the variables, it means
 224         # we have a delegated-to host, so we check there first before looking
 225         # at the variables in general
 226         if task.delegate_to is not None:
 227             # In the case of a loop, the delegated_to host may have been
 228             # templated based on the loop variable, so we try and locate
 229             # the host name in the delegated variable dictionary here
 230             delegated_host_name = templar.template(task.delegate_to)
 231             delegated_vars = variables.get('ansible_delegated_vars', dict()).get(delegated_host_name, dict())
 232 
 233             delegated_transport = C.DEFAULT_TRANSPORT
 234             for transport_var in C.MAGIC_VARIABLE_MAPPING.get('connection'):
 235                 if transport_var in delegated_vars:
 236                     delegated_transport = delegated_vars[transport_var]
 237                     break
 238 
 239             # make sure this delegated_to host has something set for its remote
 240             # address, otherwise we default to connecting to it by name. This
 241             # may happen when users put an IP entry into their inventory, or if
 242             # they rely on DNS for a non-inventory hostname
 243             for address_var in ('ansible_%s_host' % delegated_transport,) + C.MAGIC_VARIABLE_MAPPING.get('remote_addr'):
 244                 if address_var in delegated_vars:
 245                     break
 246             else:
 247                 display.debug("no remote address found for delegated host %s\nusing its name, so success depends on DNS resolution" % delegated_host_name)
 248                 delegated_vars['ansible_host'] = delegated_host_name
 249 
 250             # reset the port back to the default if none was specified, to prevent
 251             # the delegated host from inheriting the original host's setting
 252             for port_var in ('ansible_%s_port' % delegated_transport,) + C.MAGIC_VARIABLE_MAPPING.get('port'):
 253                 if port_var in delegated_vars:
 254                     break
 255             else:
 256                 if delegated_transport == 'winrm':
 257                     delegated_vars['ansible_port'] = 5986
 258                 else:
 259                     delegated_vars['ansible_port'] = C.DEFAULT_REMOTE_PORT
 260 
 261             # and likewise for the remote user
 262             for user_var in ('ansible_%s_user' % delegated_transport,) + C.MAGIC_VARIABLE_MAPPING.get('remote_user'):
 263                 if user_var in delegated_vars and delegated_vars[user_var]:
 264                     break
 265             else:
 266                 delegated_vars['ansible_user'] = task.remote_user or self.remote_user
 267         else:
 268             delegated_vars = dict()
 269 
 270             # setup shell
 271             for exe_var in C.MAGIC_VARIABLE_MAPPING.get('executable'):
 272                 if exe_var in variables:
 273                     setattr(new_info, 'executable', variables.get(exe_var))
 274 
 275         attrs_considered = []
 276         for (attr, variable_names) in iteritems(C.MAGIC_VARIABLE_MAPPING):
 277             for variable_name in variable_names:
 278                 if attr in attrs_considered:
 279                     continue
 280                 # if delegation task ONLY use delegated host vars, avoid delegated FOR host vars
 281                 if task.delegate_to is not None:
 282                     if isinstance(delegated_vars, dict) and variable_name in delegated_vars:
 283                         setattr(new_info, attr, delegated_vars[variable_name])
 284                         attrs_considered.append(attr)
 285                 elif variable_name in variables:
 286                     setattr(new_info, attr, variables[variable_name])
 287                     attrs_considered.append(attr)
 288                 # no else, as no other vars should be considered
 289 
 290         # become legacy updates -- from inventory file (inventory overrides
 291         # commandline)
 292         for become_pass_name in C.MAGIC_VARIABLE_MAPPING.get('become_pass'):
 293             if become_pass_name in variables:
 294                 break
 295 
 296         # make sure we get port defaults if needed
 297         if new_info.port is None and C.DEFAULT_REMOTE_PORT is not None:
 298             new_info.port = int(C.DEFAULT_REMOTE_PORT)
 299 
 300         # special overrides for the connection setting
 301         if len(delegated_vars) > 0:
 302             # in the event that we were using local before make sure to reset the
 303             # connection type to the default transport for the delegated-to host,
 304             # if not otherwise specified
 305             for connection_type in C.MAGIC_VARIABLE_MAPPING.get('connection'):
 306                 if connection_type in delegated_vars:
 307                     break
 308             else:
 309                 remote_addr_local = new_info.remote_addr in C.LOCALHOST
 310                 inv_hostname_local = delegated_vars.get('inventory_hostname') in C.LOCALHOST
 311                 if remote_addr_local and inv_hostname_local:
 312                     setattr(new_info, 'connection', 'local')
 313                 elif getattr(new_info, 'connection', None) == 'local' and (not remote_addr_local or not inv_hostname_local):
 314                     setattr(new_info, 'connection', C.DEFAULT_TRANSPORT)
 315 
 316         # if the final connection type is local, reset the remote_user value to that of the currently logged in user
 317         # this ensures any become settings are obeyed correctly
 318         # we store original in 'connection_user' for use of network/other modules that fallback to it as login user
 319         # connection_user to be deprecated once connection=local is removed for
 320         # network modules
 321         if new_info.connection == 'local':
 322             if not new_info.connection_user:
 323                 new_info.connection_user = new_info.remote_user
 324             new_info.remote_user = pwd.getpwuid(os.getuid()).pw_name
 325 
 326         # set no_log to default if it was not previously set
 327         if new_info.no_log is None:
 328             new_info.no_log = C.DEFAULT_NO_LOG
 329 
 330         if task.check_mode is not None:
 331             new_info.check_mode = task.check_mode
 332 
 333         if task.diff is not None:
 334             new_info.diff = task.diff
 335 
 336         return new_info

 341     def make_become_cmd(self, cmd, executable=None):
 342         """ helper function to create privilege escalation commands """
 343         display.deprecated(
 344             "PlayContext.make_become_cmd should not be used, the calling code should be using become plugins instead",
 345             version="2.12", collection_name='ansible.builtin'
 346         )
 347 
 348         if not cmd or not self.become:
 349             return cmd
 350 
 351         become_method = self.become_method
 352 
 353         # load/call become plugins here
 354         plugin = self._become_plugin
 355 
 356         if plugin:
 357             options = {
 358                 'become_exe': self.become_exe or become_method,
 359                 'become_flags': self.become_flags or '',
 360                 'become_user': self.become_user,
 361                 'become_pass': self.become_pass
 362             }
 363             plugin.set_options(direct=options)
 364 
 365             if not executable:
 366                 executable = self.executable
 367 
 368             shell = get_shell_plugin(executable=executable)
 369             cmd = plugin.build_become_command(cmd, shell)
 370             # for backwards compat:
 371             if self.become_pass:
 372                 self.prompt = plugin.prompt
 373         else:
 374             raise AnsibleError("Privilege escalation method not found: %s" % become_method)
 375 
 376         return cmd

 378     def update_vars(self, variables):
 379         '''
 380         Adds 'magic' variables relating to connections to the variable dictionary provided.
 381         In case users need to access from the play, this is a legacy from runner.
 382         '''
 383 
 384         for prop, var_list in C.MAGIC_VARIABLE_MAPPING.items():
 385             try:
 386                 if 'become' in prop:
 387                     continue
 388 
 389                 var_val = getattr(self, prop)
 390                 for var_opt in var_list:
 391                     if var_opt not in variables and var_val is not None:
 392                         variables[var_opt] = var_val
 393             except AttributeError:
 394                 continue

 396     def _get_attr_connection(self):
 397         ''' connections are special, this takes care of responding correctly '''
 398         conn_type = None
 399         if self._attributes['connection'] == 'smart':
 400             conn_type = 'ssh'
 401             # see if SSH can support ControlPersist if not use paramiko
 402             if not check_for_controlpersist(self.ssh_executable) and paramiko is not None:
 403                 conn_type = "paramiko"
 404 
 405         # if someone did `connection: persistent`, default it to using a persistent paramiko connection to avoid problems
 406         elif self._attributes['connection'] == 'persistent' and paramiko is not None:
 407             conn_type = 'paramiko'
 408 
 409         if conn_type:
 410             self.connection = conn_type
 411 
 412         return self._attributes['connection']
