  69     def preprocess_data(self, ds):
  70         # role names that are simply numbers can be parsed by PyYAML
  71         # as integers even when quoted, so turn it into a string type
  72         if isinstance(ds, int):
  73             ds = "%s" % ds
  74 
  75         if not isinstance(ds, dict) and not isinstance(ds, string_types) and not isinstance(ds, AnsibleBaseYAMLObject):
  76             raise AnsibleAssertionError()
  77 
  78         if isinstance(ds, dict):
  79             ds = super(RoleDefinition, self).preprocess_data(ds)
  80 
  81         # save the original ds for use later
  82         self._ds = ds
  83 
  84         # we create a new data structure here, using the same
  85         # object used internally by the YAML parsing code so we
  86         # can preserve file:line:column information if it exists
  87         new_ds = AnsibleMapping()
  88         if isinstance(ds, AnsibleBaseYAMLObject):
  89             new_ds.ansible_pos = ds.ansible_pos
  90 
  91         # first we pull the role name out of the data structure,
  92         # and then use that to determine the role path (which may
  93         # result in a new role name, if it was a file path)
  94         role_name = self._load_role_name(ds)
  95         (role_name, role_path) = self._load_role_path(role_name)
  96 
  97         # next, we split the role params out from the valid role
  98         # attributes and update the new datastructure with that
  99         # result and the role name
 100         if isinstance(ds, dict):
 101             (new_role_def, role_params) = self._split_role_params(ds)
 102             new_ds.update(new_role_def)
 103             self._role_params = role_params
 104 
 105         # set the role name in the new ds
 106         new_ds['role'] = role_name
 107 
 108         # we store the role path internally
 109         self._role_path = role_path
 110 
 111         # and return the cleaned-up data structure
 112         return new_ds

 114     def _load_role_name(self, ds):
 115         '''
 116         Returns the role name (either the role: or name: field) from
 117         the role definition, or (when the role definition is a simple
 118         string), just that string
 119         '''
 120 
 121         if isinstance(ds, string_types):
 122             return ds
 123 
 124         role_name = ds.get('role', ds.get('name'))
 125         if not role_name or not isinstance(role_name, string_types):
 126             raise AnsibleError('role definitions must contain a role name', obj=ds)
 127 
 128         # if we have the required datastructures, and if the role_name
 129         # contains a variable, try and template it now
 130         if self._variable_manager:
 131             all_vars = self._variable_manager.get_vars(play=self._play)
 132             templar = Templar(loader=self._loader, variables=all_vars)
 133             role_name = templar.template(role_name)
 134 
 135         return role_name
 136 
 137     def _load_role_path(self, role_name):
 138         '''
 139         the 'role', as specified in the ds (or as a bare string), can either
 140         be a simple name or a full path. If it is a full path, we use the
 141         basename as the role name, otherwise we take the name as-given and
 142         append it to the default role path
 143         '''
 144 
 145         # create a templar class to template the dependency names, in
 146         # case they contain variables
 147         if self._variable_manager is not None:
 148             all_vars = self._variable_manager.get_vars(play=self._play)
 149         else:
 150             all_vars = dict()
 151 
 152         templar = Templar(loader=self._loader, variables=all_vars)
 153         role_name = templar.template(role_name)
 154 
 155         role_tuple = None
 156 
 157         # try to load as a collection-based role first
 158         if self._collection_list or AnsibleCollectionRef.is_valid_fqcr(role_name):
 159             role_tuple = _get_collection_role_path(role_name, self._collection_list)
 160 
 161         if role_tuple:
 162             # we found it, stash collection data and return the name/path tuple
 163             self._role_collection = role_tuple[2]
 164             return role_tuple[0:2]
 165 
 166         # We didn't find a collection role, look in defined role paths
 167         # FUTURE: refactor this to be callable from internal so we can properly order
 168         # ansible.legacy searches with the collections keyword
 169 
 170         # we always start the search for roles in the base directory of the playbook
 171         role_search_paths = [
 172             os.path.join(self._loader.get_basedir(), u'roles'),
 173         ]
 174 
 175         # also search in the configured roles path
 176         if C.DEFAULT_ROLES_PATH:
 177             role_search_paths.extend(C.DEFAULT_ROLES_PATH)
 178 
 179         # next, append the roles basedir, if it was set, so we can
 180         # search relative to that directory for dependent roles
 181         if self._role_basedir:
 182             role_search_paths.append(self._role_basedir)
 183 
 184         # finally as a last resort we look in the current basedir as set
 185         # in the loader (which should be the playbook dir itself) but without
 186         # the roles/ dir appended
 187         role_search_paths.append(self._loader.get_basedir())
 188 
 189         # now iterate through the possible paths and return the first one we find
 190         for path in role_search_paths:
 191             path = templar.template(path)
 192             role_path = unfrackpath(os.path.join(path, role_name))
 193             if self._loader.path_exists(role_path):
 194                 return (role_name, role_path)
 195 
 196         # if not found elsewhere try to extract path from name
 197         role_path = unfrackpath(role_name)
 198         if self._loader.path_exists(role_path):
 199             role_name = os.path.basename(role_name)
 200             return (role_name, role_path)
 201 
 202         searches = (self._collection_list or []) + role_search_paths
 203         raise AnsibleError("the role '%s' was not found in %s" % (role_name, ":".join(searches)), obj=self._ds)

 205     def _split_role_params(self, ds):
 206         '''
 207         Splits any random role params off from the role spec and store
 208         them in a dictionary of params for parsing later
 209         '''
 210 
 211         role_def = dict()
 212         role_params = dict()
 213         base_attribute_names = frozenset(self._valid_attrs.keys())
 214         for (key, value) in iteritems(ds):
 215             # use the list of FieldAttribute values to determine what is and is not
 216             # an extra parameter for this role (or sub-class of this role)
 217             # FIXME: hard-coded list of exception key names here corresponds to the
 218             #        connection fields in the Base class. There may need to be some
 219             #        other mechanism where we exclude certain kinds of field attributes,
 220             #        or make this list more automatic in some way so we don't have to
 221             #        remember to update it manually.
 222             if key not in base_attribute_names:
 223                 # this key does not match a field attribute, so it must be a role param
 224                 role_params[key] = value
 225             else:
 226                 # this is a field attribute, so copy it over directly
 227                 role_def[key] = value
 228 
 229         return (role_def, role_params)

 237     def get_name(self, include_role_fqcn=True):
 238         if include_role_fqcn:
 239             return '.'.join(x for x in (self._role_collection, self.role) if x)
 240         return self.role
