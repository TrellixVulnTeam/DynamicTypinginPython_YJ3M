  41 class RoleRequirement(RoleDefinition):
  42 
  43     """
  44     Helper class for Galaxy, which is used to parse both dependencies
  45     specified in meta/main.yml and requirements.yml files.
  46     """
  47 
  48     def __init__(self):
  49         pass
  50 
  51     @staticmethod
  52     def repo_url_to_role_name(repo_url):
  53         # gets the role name out of a repo like
  54         # http://git.example.com/repos/repo.git" => "repo"
  55 
  56         if '://' not in repo_url and '@' not in repo_url:
  57             return repo_url
  58         trailing_path = repo_url.split('/')[-1]
  59         if trailing_path.endswith('.git'):
  60             trailing_path = trailing_path[:-4]
  61         if trailing_path.endswith('.tar.gz'):
  62             trailing_path = trailing_path[:-7]
  63         if ',' in trailing_path:
  64             trailing_path = trailing_path.split(',')[0]
  65         return trailing_path
  66 
  67     @staticmethod
  68     def role_yaml_parse(role):
  69 
  70         if isinstance(role, string_types):
  71             name = None
  72             scm = None
  73             src = None
  74             version = None
  75             if ',' in role:
  76                 if role.count(',') == 1:
  77                     (src, version) = role.strip().split(',', 1)
  78                 elif role.count(',') == 2:
  79                     (src, version, name) = role.strip().split(',', 2)
  80                 else:
  81                     raise AnsibleError("Invalid role line (%s). Proper format is 'role_name[,version[,name]]'" % role)
  82             else:
  83                 src = role
  84 
  85             if name is None:
  86                 name = RoleRequirement.repo_url_to_role_name(src)
  87             if '+' in src:
  88                 (scm, src) = src.split('+', 1)
  89 
  90             return dict(name=name, src=src, scm=scm, version=version)
  91 
  92         if 'role' in role:
  93             name = role['role']
  94             if ',' in name:
  95                 raise AnsibleError("Invalid old style role requirement: %s" % name)
  96             else:
  97                 del role['role']
  98                 role['name'] = name
  99         else:
 100             role = role.copy()
 101 
 102             if 'src' in role:
 103                 # New style: { src: 'galaxy.role,version,name', other_vars: "here" }
 104                 if 'github.com' in role["src"] and 'http' in role["src"] and '+' not in role["src"] and not role["src"].endswith('.tar.gz'):
 105                     role["src"] = "git+" + role["src"]
 106 
 107                 if '+' in role["src"]:
 108                     (scm, src) = role["src"].split('+')
 109                     role["scm"] = scm
 110                     role["src"] = src
 111 
 112                 if 'name' not in role:
 113                     role["name"] = RoleRequirement.repo_url_to_role_name(role["src"])
 114 
 115             if 'version' not in role:
 116                 role['version'] = ''
 117 
 118             if 'scm' not in role:
 119                 role['scm'] = None
 120 
 121         for key in list(role.keys()):
 122             if key not in VALID_SPEC_KEYS:
 123                 role.pop(key)
 124 
 125         return role
 126 
 127     @staticmethod
 128     def scm_archive_role(src, scm='git', name=None, version='HEAD', keep_scm_meta=False):
 129 
 130         return scm_archive_resource(src, scm=scm, name=name, version=version, keep_scm_meta=keep_scm_meta)
