  47 def hash_params(params):
  48     """
  49     Construct a data structure of parameters that is hashable.
  50 
  51     This requires changing any mutable data structures into immutable ones.
  52     We chose a frozenset because role parameters have to be unique.
  53 
  54     .. warning::  this does not handle unhashable scalars.  Two things
  55         mitigate that limitation:
  56 
  57         1) There shouldn't be any unhashable scalars specified in the yaml
  58         2) Our only choice would be to return an error anyway.
  59     """
  60     # Any container is unhashable if it contains unhashable items (for
  61     # instance, tuple() is a Hashable subclass but if it contains a dict, it
  62     # cannot be hashed)
  63     if isinstance(params, Container) and not isinstance(params, (text_type, binary_type)):
  64         if isinstance(params, Mapping):
  65             try:
  66                 # Optimistically hope the contents are all hashable
  67                 new_params = frozenset(params.items())
  68             except TypeError:
  69                 new_params = set()
  70                 for k, v in params.items():
  71                     # Hash each entry individually
  72                     new_params.add((k, hash_params(v)))
  73                 new_params = frozenset(new_params)
  74 
  75         elif isinstance(params, (Set, Sequence)):
  76             try:
  77                 # Optimistically hope the contents are all hashable
  78                 new_params = frozenset(params)
  79             except TypeError:
  80                 new_params = set()
  81                 for v in params:
  82                     # Hash each entry individually
  83                     new_params.add(hash_params(v))
  84                 new_params = frozenset(new_params)
  85         else:
  86             # This is just a guess.
  87             new_params = frozenset(params)
  88         return new_params
  89 
  90     # Note: We do not handle unhashable scalars but our only choice would be
  91     # to raise an error there anyway.
  92     return frozenset((params,))

 131     def get_name(self, include_role_fqcn=True):
 132         if include_role_fqcn:

 136     @staticmethod
 137     def load(role_include, play, parent_role=None, from_files=None, from_include=False):
 138 
 139         if from_files is None:
 140             from_files = {}
 141         try:
 142             # The ROLE_CACHE is a dictionary of role names, with each entry
 143             # containing another dictionary corresponding to a set of parameters
 144             # specified for a role as the key and the Role() object itself.
 145             # We use frozenset to make the dictionary hashable.
 146 
 147             params = role_include.get_role_params()
 148             if role_include.when is not None:
 149                 params['when'] = role_include.when
 150             if role_include.tags is not None:
 151                 params['tags'] = role_include.tags
 152             if from_files is not None:
 153                 params['from_files'] = from_files
 154             if role_include.vars:
 155                 params['vars'] = role_include.vars
 156 
 157             params['from_include'] = from_include
 158 
 159             hashed_params = hash_params(params)
 160             if role_include.get_name() in play.ROLE_CACHE:
 161                 for (entry, role_obj) in iteritems(play.ROLE_CACHE[role_include.get_name()]):
 162                     if hashed_params == entry:
 163                         if parent_role:
 164                             role_obj.add_parent(parent_role)
 165                         return role_obj
 166 
 167             # TODO: need to fix cycle detection in role load (maybe use an empty dict
 168             #  for the in-flight in role cache as a sentinel that we're already trying to load
 169             #  that role?)
 170             # see https://github.com/ansible/ansible/issues/61527
 171             r = Role(play=play, from_files=from_files, from_include=from_include)
 172             r._load_role_data(role_include, parent_role=parent_role)
 173 
 174             if role_include.get_name() not in play.ROLE_CACHE:
 175                 play.ROLE_CACHE[role_include.get_name()] = dict()
 176 
 177             # FIXME: how to handle cache keys for collection-based roles, since they're technically adjustable per task?
 178             play.ROLE_CACHE[role_include.get_name()][hashed_params] = r
 179             return r
 180 
 181         except RuntimeError:
 182             raise AnsibleError("A recursion loop was detected with the roles specified. Make sure child roles do not have dependencies on parent roles",
 183                                obj=role_include._ds)

 185     def _load_role_data(self, role_include, parent_role=None):
 186         self._role_name = role_include.role
 187         self._role_path = role_include.get_role_path()
 188         self._role_collection = role_include._role_collection
 189         self._role_params = role_include.get_role_params()
 190         self._variable_manager = role_include.get_variable_manager()
 191         self._loader = role_include.get_loader()
 192 
 193         if parent_role:
 194             self.add_parent(parent_role)
 195 
 196         # copy over all field attributes from the RoleInclude
 197         # update self._attributes directly, to avoid squashing
 198         for (attr_name, _) in iteritems(self._valid_attrs):
 199             if attr_name in ('when', 'tags'):
 200                 self._attributes[attr_name] = self._extend_value(
 201                     self._attributes[attr_name],
 202                     role_include._attributes[attr_name],
 203                 )
 204             else:
 205                 self._attributes[attr_name] = role_include._attributes[attr_name]
 206 
 207         # vars and default vars are regular dictionaries
 208         self._role_vars = self._load_role_yaml('vars', main=self._from_files.get('vars'), allow_dir=True)
 209         if self._role_vars is None:
 210             self._role_vars = dict()
 211         elif not isinstance(self._role_vars, dict):
 212             raise AnsibleParserError("The vars/main.yml file for role '%s' must contain a dictionary of variables" % self._role_name)
 213 
 214         self._default_vars = self._load_role_yaml('defaults', main=self._from_files.get('defaults'), allow_dir=True)
 215         if self._default_vars is None:
 216             self._default_vars = dict()
 217         elif not isinstance(self._default_vars, dict):
 218             raise AnsibleParserError("The defaults/main.yml file for role '%s' must contain a dictionary of variables" % self._role_name)
 219 
 220         # load the role's other files, if they exist
 221         metadata = self._load_role_yaml('meta')
 222         if metadata:
 223             self._metadata = RoleMetadata.load(metadata, owner=self, variable_manager=self._variable_manager, loader=self._loader)
 224             self._dependencies = self._load_dependencies()
 225         else:
 226             self._metadata = RoleMetadata()
 227 
 228         # reset collections list; roles do not inherit collections from parents, just use the defaults
 229         # FUTURE: use a private config default for this so we can allow it to be overridden later
 230         self.collections = []
 231 
 232         # configure plugin/collection loading; either prepend the current role's collection or configure legacy plugin loading
 233         # FIXME: need exception for explicit ansible.legacy?
 234         if self._role_collection:  # this is a collection-hosted role
 235             self.collections.insert(0, self._role_collection)
 236         else:  # this is a legacy role, but set the default collection if there is one
 237             default_collection = AnsibleCollectionConfig.default_collection
 238             if default_collection:
 239                 self.collections.insert(0, default_collection)
 240             # legacy role, ensure all plugin dirs under the role are added to plugin search path
 241             add_all_plugin_dirs(self._role_path)
 242 
 243         # collections can be specified in metadata for legacy or collection-hosted roles
 244         if self._metadata.collections:
 245             self.collections.extend((c for c in self._metadata.collections if c not in self.collections))
 246 
 247         # if any collections were specified, ensure that core or legacy synthetic collections are always included
 248         if self.collections:
 249             # default append collection is core for collection-hosted roles, legacy for others
 250             default_append_collection = 'ansible.builtin' if self._role_collection else 'ansible.legacy'
 251             if 'ansible.builtin' not in self.collections and 'ansible.legacy' not in self.collections:
 252                 self.collections.append(default_append_collection)
 253 
 254         task_data = self._load_role_yaml('tasks', main=self._from_files.get('tasks'))
 255         if task_data:
 256             try:
 257                 self._task_blocks = load_list_of_blocks(task_data, play=self._play, role=self, loader=self._loader, variable_manager=self._variable_manager)
 258             except AssertionError as e:
 259                 raise AnsibleParserError("The tasks/main.yml file for role '%s' must contain a list of tasks" % self._role_name,
 260                                          obj=task_data, orig_exc=e)
 261 
 262         handler_data = self._load_role_yaml('handlers', main=self._from_files.get('handlers'))
 263         if handler_data:
 264             try:
 265                 self._handler_blocks = load_list_of_blocks(handler_data, play=self._play, role=self, use_handlers=True, loader=self._loader,
 266                                                            variable_manager=self._variable_manager)
 267             except AssertionError as e:
 268                 raise AnsibleParserError("The handlers/main.yml file for role '%s' must contain a list of tasks" % self._role_name,
 269                                          obj=handler_data, orig_exc=e)
 270 
 271     def _load_role_yaml(self, subdir, main=None, allow_dir=False):
 272         file_path = os.path.join(self._role_path, subdir)
 273         if self._loader.path_exists(file_path) and self._loader.is_directory(file_path):
 274             # Valid extensions and ordering for roles is hard-coded to maintain
 275             # role portability
 276             extensions = ['.yml', '.yaml', '.json']
 277             # If no <main> is specified by the user, look for files with
 278             # extensions before bare name. Otherwise, look for bare name first.
 279             if main is None:
 280                 _main = 'main'
 281                 extensions.append('')
 282             else:
 283                 _main = main
 284                 extensions.insert(0, '')
 285             found_files = self._loader.find_vars_files(file_path, _main, extensions, allow_dir)
 286             if found_files:
 287                 data = {}
 288                 for found in found_files:
 289                     new_data = self._loader.load_from_file(found)
 290                     if new_data and allow_dir:
 291                         data = combine_vars(data, new_data)
 292                     else:
 293                         data = new_data
 294                 return data
 295             elif main is not None:
 296                 raise AnsibleParserError("Could not find specified file in role: %s/%s" % (subdir, main))
 297         return None

 299     def _load_dependencies(self):
 300         '''
 301         Recursively loads role dependencies from the metadata list of
 302         dependencies, if it exists
 303         '''
 304 
 305         deps = []
 306         if self._metadata:
 307             for role_include in self._metadata.dependencies:
 308                 r = Role.load(role_include, play=self._play, parent_role=self)
 309                 deps.append(r)
 310 
 311         return deps

 315     def add_parent(self, parent_role):
 316         ''' adds a role to the list of this roles parents '''
 317         if not isinstance(parent_role, Role):
 318             raise AnsibleAssertionError()
 319 
 320         if parent_role not in self._parents:
 321             self._parents.append(parent_role)
 322 
 323     def get_parents(self):
 324         return self._parents

 326     def get_default_vars(self, dep_chain=None):
 327         dep_chain = [] if dep_chain is None else dep_chain
 328 
 329         default_vars = dict()
 330         for dep in self.get_all_dependencies():
 331             default_vars = combine_vars(default_vars, dep.get_default_vars())
 332         if dep_chain:
 333             for parent in dep_chain:
 334                 default_vars = combine_vars(default_vars, parent._default_vars)
 335         default_vars = combine_vars(default_vars, self._default_vars)
 336         return default_vars

 338     def get_inherited_vars(self, dep_chain=None):
 339         dep_chain = [] if dep_chain is None else dep_chain
 340 
 341         inherited_vars = dict()
 342 
 343         if dep_chain:
 344             for parent in dep_chain:
 345                 inherited_vars = combine_vars(inherited_vars, parent._role_vars)
 346         return inherited_vars

 348     def get_role_params(self, dep_chain=None):
 349         dep_chain = [] if dep_chain is None else dep_chain
 350 
 351         params = {}
 352         if dep_chain:
 353             for parent in dep_chain:
 354                 params = combine_vars(params, parent._role_params)
 355         params = combine_vars(params, self._role_params)
 356         return params

 358     def get_vars(self, dep_chain=None, include_params=True):
 359         dep_chain = [] if dep_chain is None else dep_chain
 360 
 361         all_vars = self.get_inherited_vars(dep_chain)
 362 
 363         for dep in self.get_all_dependencies():
 364             all_vars = combine_vars(all_vars, dep.get_vars(include_params=include_params))
 365 
 366         all_vars = combine_vars(all_vars, self.vars)
 367         all_vars = combine_vars(all_vars, self._role_vars)
 368         if include_params:
 369             all_vars = combine_vars(all_vars, self.get_role_params(dep_chain=dep_chain))
 370 
 371         return all_vars

 376     def get_all_dependencies(self):
 377         '''
 378         Returns a list of all deps, built recursively from all child dependencies,
 379         in the proper order in which they should be executed or evaluated.
 380         '''
 381 
 382         child_deps = []
 383 
 384         for dep in self.get_direct_dependencies():
 385             for child_dep in dep.get_all_dependencies():
 386                 child_deps.append(child_dep)
 387             child_deps.append(dep)
 388 
 389         return child_deps

 394     def get_handler_blocks(self, play, dep_chain=None):
 395         # Do not recreate this list each time ``get_handler_blocks`` is called.
 396         # Cache the results so that we don't potentially overwrite with copied duplicates
 397         #
 398         # ``get_handler_blocks`` may be called when handling ``import_role`` during parsing
 399         # as well as with ``Play.compile_roles_handlers`` from ``TaskExecutor``
 400         if self._compiled_handler_blocks:
 401             return self._compiled_handler_blocks
 402 
 403         self._compiled_handler_blocks = block_list = []
 404 
 405         # update the dependency chain here
 406         if dep_chain is None:
 407             dep_chain = []
 408         new_dep_chain = dep_chain + [self]
 409 
 410         for dep in self.get_direct_dependencies():
 411             dep_blocks = dep.get_handler_blocks(play=play, dep_chain=new_dep_chain)
 412             block_list.extend(dep_blocks)
 413 
 414         for task_block in self._handler_blocks:
 415             new_task_block = task_block.copy()
 416             new_task_block._dep_chain = new_dep_chain
 417             new_task_block._play = play
 418             block_list.append(new_task_block)
 419 
 420         return block_list

 422     def has_run(self, host):
 423         '''
 424         Returns true if this role has been iterated over completely and
 425         at least one task was run
 426         '''
 427 
 428         return host.name in self._completed and not self._metadata.allow_duplicates

 430     def compile(self, play, dep_chain=None):
 431         '''
 432         Returns the task list for this role, which is created by first
 433         recursively compiling the tasks for all direct dependencies, and
 434         then adding on the tasks for this role.
 435 
 436         The role compile() also remembers and saves the dependency chain
 437         with each task, so tasks know by which route they were found, and
 438         can correctly take their parent's tags/conditionals into account.
 439         '''
 440 
 441         block_list = []
 442 
 443         # update the dependency chain here
 444         if dep_chain is None:
 445             dep_chain = []
 446         new_dep_chain = dep_chain + [self]
 447 
 448         deps = self.get_direct_dependencies()
 449         for dep in deps:
 450             dep_blocks = dep.compile(play=play, dep_chain=new_dep_chain)
 451             block_list.extend(dep_blocks)
 452 
 453         for idx, task_block in enumerate(self._task_blocks):
 454             new_task_block = task_block.copy()
 455             new_task_block._dep_chain = new_dep_chain
 456             new_task_block._play = play
 457             if idx == len(self._task_blocks) - 1:
 458                 new_task_block._eor = True
 459             block_list.append(new_task_block)
 460 
 461         return block_list

 463     def serialize(self, include_deps=True):
 464         res = super(Role, self).serialize()
 465 
 466         res['_role_name'] = self._role_name
 467         res['_role_path'] = self._role_path
 468         res['_role_vars'] = self._role_vars
 469         res['_role_params'] = self._role_params
 470         res['_default_vars'] = self._default_vars
 471         res['_had_task_run'] = self._had_task_run.copy()
 472         res['_completed'] = self._completed.copy()
 473 
 474         if self._metadata:
 475             res['_metadata'] = self._metadata.serialize()
 476 
 477         if include_deps:
 478             deps = []
 479             for role in self.get_direct_dependencies():
 480                 deps.append(role.serialize())
 481             res['_dependencies'] = deps
 482 
 483         parents = []
 484         for parent in self._parents:
 485             parents.append(parent.serialize(include_deps=False))
 486         res['_parents'] = parents
 487 
 488         return res

 490     def deserialize(self, data, include_deps=True):
 491         self._role_name = data.get('_role_name', '')
 492         self._role_path = data.get('_role_path', '')
 493         self._role_vars = data.get('_role_vars', dict())
 494         self._role_params = data.get('_role_params', dict())
 495         self._default_vars = data.get('_default_vars', dict())
 496         self._had_task_run = data.get('_had_task_run', dict())
 497         self._completed = data.get('_completed', dict())
 498 
 499         if include_deps:
 500             deps = []
 501             for dep in data.get('_dependencies', []):
 502                 r = Role()
 503                 r.deserialize(dep)
 504                 deps.append(r)
 505             setattr(self, '_dependencies', deps)
 506 
 507         parent_data = data.get('_parents', [])
 508         parents = []
 509         for parent in parent_data:
 510             r = Role()
 511             r.deserialize(parent, include_deps=False)
 512             parents.append(r)
 513         setattr(self, '_parents', parents)
 514 
 515         metadata_data = data.get('_metadata')
 516         if metadata_data:
 517             m = RoleMetadata()
 518             m.deserialize(metadata_data)
 519             self._metadata = m
 520 
 521         super(Role, self).deserialize(data)

 523     def set_loader(self, loader):
 524         self._loader = loader
 525         for parent in self._parents:
 526             parent.set_loader(loader)
 527         for dep in self.get_direct_dependencies():
 528             dep.set_loader(loader)
