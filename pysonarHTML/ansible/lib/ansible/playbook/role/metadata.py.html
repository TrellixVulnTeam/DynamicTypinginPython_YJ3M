  36 class RoleMetadata(Base, CollectionSearch):
  37     '''
  38     This class wraps the parsing and validation of the optional metadata
  39     within each Role (meta/main.yml).
  40     '''
  41 
  42     _allow_duplicates = FieldAttribute(isa='bool', default=False)
  43     _dependencies = FieldAttribute(isa='list', default=list)
  44     _galaxy_info = FieldAttribute(isa='GalaxyInfo')
  45 
  46     def __init__(self, owner=None):
  47         self._owner = owner
  48         super(RoleMetadata, self).__init__()
  49 
  50     @staticmethod
  51     def load(data, owner, variable_manager=None, loader=None):
  52         '''
  53         Returns a new RoleMetadata object based on the datastructure passed in.
  54         '''
  55 
  56         if not isinstance(data, dict):
  57             raise AnsibleParserError("the 'meta/main.yml' for role %s is not a dictionary" % owner.get_name())
  58 
  59         m = RoleMetadata(owner=owner).load_data(data, variable_manager=variable_manager, loader=loader)
  60         return m
  61 
  62     def _load_dependencies(self, attr, ds):
  63         '''
  64         This is a helper loading function for the dependencies list,
  65         which returns a list of RoleInclude objects
  66         '''
  67 
  68         roles = []
  69         if ds:
  70             if not isinstance(ds, list):
  71                 raise AnsibleParserError("Expected role dependencies to be a list.", obj=self._ds)
  72 
  73             for role_def in ds:
  74                 if isinstance(role_def, string_types) or 'role' in role_def or 'name' in role_def:
  75                     roles.append(role_def)
  76                     continue
  77                 try:
  78                     # role_def is new style: { src: 'galaxy.role,version,name', other_vars: "here" }
  79                     def_parsed = RoleRequirement.role_yaml_parse(role_def)
  80                     if def_parsed.get('name'):
  81                         role_def['name'] = def_parsed['name']
  82                     roles.append(role_def)
  83                 except AnsibleError as exc:
  84                     raise AnsibleParserError(to_native(exc), obj=role_def, orig_exc=exc)
  85 
  86         current_role_path = None
  87         collection_search_list = None
  88 
  89         if self._owner:
  90             current_role_path = os.path.dirname(self._owner._role_path)
  91 
  92             # if the calling role has a collections search path defined, consult it
  93             collection_search_list = self._owner.collections[:] or []
  94 
  95             # if the calling role is a collection role, ensure that its containing collection is searched first
  96             owner_collection = self._owner._role_collection
  97             if owner_collection:
  98                 collection_search_list = [c for c in collection_search_list if c != owner_collection]
  99                 collection_search_list.insert(0, owner_collection)
 100             # ensure fallback role search works
 101             if 'ansible.legacy' not in collection_search_list:
 102                 collection_search_list.append('ansible.legacy')
 103 
 104         try:
 105             return load_list_of_roles(roles, play=self._owner._play, current_role_path=current_role_path,
 106                                       variable_manager=self._variable_manager, loader=self._loader,
 107                                       collection_search_list=collection_search_list)
 108         except AssertionError as e:
 109             raise AnsibleParserError("A malformed list of role dependencies was encountered.", obj=self._ds, orig_exc=e)
 110 
 111     def _load_galaxy_info(self, attr, ds):
 112         '''
 113         This is a helper loading function for the galaxy info entry
 114         in the metadata, which returns a GalaxyInfo object rather than
 115         a simple dictionary.
 116         '''
 117 
 118         return ds
 119 
 120     def serialize(self):
 121         return dict(
 122             allow_duplicates=self._allow_duplicates,
 123             dependencies=self._dependencies
 124         )
 125 
 126     def deserialize(self, data):
 127         setattr(self, 'allow_duplicates', data.get('allow_duplicates', False))
 128         setattr(self, 'dependencies', data.get('dependencies', []))
