  25 _CONTAINERS = frozenset(('list', 'dict', 'set'))

  28 class Attribute:
  29 
  30     def __init__(
  31         self,
  32         isa=None,
  33         private=False,
  34         default=None,
  35         required=False,
  36         listof=None,
  37         priority=0,
  38         class_type=None,
  39         always_post_validate=False,
  40         inherit=True,
  41         alias=None,
  42         extend=False,
  43         prepend=False,
  44         static=False,
  45     ):
  46 
  47         """
  48         :class:`Attribute` specifies constraints for attributes of objects which
  49         derive from playbook data.  The attributes of the object are basically
  50         a schema for the yaml playbook.
  51 
  52         :kwarg isa: The type of the attribute.  Allowable values are a string
  53             representation of any yaml basic datatype, python class, or percent.
  54             (Enforced at post-validation time).
  55         :kwarg private: Not used at runtime.  The docs playbook keyword dumper uses it to determine
  56             that a keyword should not be documented.  mpdehaan had plans to remove attributes marked
  57             private from the ds so they would not have been available at all.
  58         :kwarg default: Default value if unspecified in the YAML document.
  59         :kwarg required: Whether or not the YAML document must contain this field.
  60             If the attribute is None when post-validated, an error will be raised.
  61         :kwarg listof: If isa is set to "list", this can optionally be set to
  62             ensure that all elements in the list are of the given type. Valid
  63             values here are the same as those for isa.
  64         :kwarg priority: The order in which the fields should be parsed. Generally
  65             this does not need to be set, it is for rare situations where another
  66             field depends on the fact that another field was parsed first.
  67         :kwarg class_type: If isa is set to "class", this can be optionally set to
  68             a class (not a string name). The YAML data for this field will be
  69             passed to the __init__ method of that class during post validation and
  70             the field will be an instance of that class.
  71         :kwarg always_post_validate: Controls whether a field should be post
  72             validated or not (default: False).
  73         :kwarg inherit: A boolean value, which controls whether the object
  74             containing this field should attempt to inherit the value from its
  75             parent object if the local value is None.
  76         :kwarg alias: An alias to use for the attribute name, for situations where
  77             the attribute name may conflict with a Python reserved word.
  78         """
  79 
  80         self.isa = isa
  81         self.private = private
  82         self.default = default
  83         self.required = required
  84         self.listof = listof
  85         self.priority = priority
  86         self.class_type = class_type
  87         self.always_post_validate = always_post_validate
  88         self.inherit = inherit
  89         self.alias = alias
  90         self.extend = extend
  91         self.prepend = prepend
  92         self.static = static
  93 
  94         if default is not None and self.isa in _CONTAINERS and not callable(default):
  95             raise TypeError('defaults for FieldAttribute may not be mutable, please provide a callable instead')
  96 
  97     def __eq__(self, other):
  98         return other.priority == self.priority
  99 
 100     def __ne__(self, other):
 101         return other.priority != self.priority
 102 
 103     # NB: higher priority numbers sort first
 104 
 105     def __lt__(self, other):
 106         return other.priority < self.priority
 107 
 108     def __gt__(self, other):
 109         return other.priority > self.priority
 110 
 111     def __le__(self, other):
 112         return other.priority <= self.priority
 113 
 114     def __ge__(self, other):
 115         return other.priority >= self.priority
 116

 118 class FieldAttribute(Attribute):
 119     pass
