  43 def _get_entry(plugin_type, plugin_name, config):
  44     ''' construct entry for requested config '''
  45     entry = ''
  46     if plugin_type:
  47         entry += 'plugin_type: %s ' % plugin_type
  48         if plugin_name:
  49             entry += 'plugin: %s ' % plugin_name
  50     entry += 'setting: %s ' % config
  51     return entry

  55 def ensure_type(value, value_type, origin=None):
  56     ''' return a configuration variable with casting
  57     :arg value: The value to ensure correct typing of
  58     :kwarg value_type: The type of the value.  This can be any of the following strings:
  59         :boolean: sets the value to a True or False value
  60         :bool: Same as 'boolean'
  61         :integer: Sets the value to an integer or raises a ValueType error
  62         :int: Same as 'integer'
  63         :float: Sets the value to a float or raises a ValueType error
  64         :list: Treats the value as a comma separated list.  Split the value
  65             and return it as a python list.
  66         :none: Sets the value to None
  67         :path: Expands any environment variables and tilde's in the value.
  68         :tmppath: Create a unique temporary directory inside of the directory
  69             specified by value and return its path.
  70         :temppath: Same as 'tmppath'
  71         :tmp: Same as 'tmppath'
  72         :pathlist: Treat the value as a typical PATH string.  (On POSIX, this
  73             means colon separated strings.)  Split the value and then expand
  74             each part for environment variables and tildes.
  75         :pathspec: Treat the value as a PATH string. Expands any environment variables
  76             tildes's in the value.
  77         :str: Sets the value to string types.
  78         :string: Same as 'str'
  79     '''
  80 
  81     errmsg = ''
  82     basedir = None
  83     if origin and os.path.isabs(origin) and os.path.exists(to_bytes(origin)):
  84         basedir = origin
  85 
  86     if value_type:
  87         value_type = value_type.lower()
  88 
  89     if value is not None:
  90         if value_type in ('boolean', 'bool'):
  91             value = boolean(value, strict=False)
  92 
  93         elif value_type in ('integer', 'int'):
  94             value = int(value)
  95 
  96         elif value_type == 'float':
  97             value = float(value)
  98 
  99         elif value_type == 'list':
 100             if isinstance(value, string_types):
 101                 value = [x.strip() for x in value.split(',')]
 102             elif not isinstance(value, Sequence):
 103                 errmsg = 'list'
 104 
 105         elif value_type == 'none':
 106             if value == "None":
 107                 value = None
 108 
 109             if value is not None:
 110                 errmsg = 'None'
 111 
 112         elif value_type == 'path':
 113             if isinstance(value, string_types):
 114                 value = resolve_path(value, basedir=basedir)
 115             else:
 116                 errmsg = 'path'
 117 
 118         elif value_type in ('tmp', 'temppath', 'tmppath'):
 119             if isinstance(value, string_types):
 120                 value = resolve_path(value, basedir=basedir)
 121                 if not os.path.exists(value):
 122                     makedirs_safe(value, 0o700)
 123                 prefix = 'ansible-local-%s' % os.getpid()
 124                 value = tempfile.mkdtemp(prefix=prefix, dir=value)
 125                 atexit.register(cleanup_tmp_file, value, warn=True)
 126             else:
 127                 errmsg = 'temppath'
 128 
 129         elif value_type == 'pathspec':
 130             if isinstance(value, string_types):
 131                 value = value.split(os.pathsep)
 132 
 133             if isinstance(value, Sequence):
 134                 value = [resolve_path(x, basedir=basedir) for x in value]
 135             else:
 136                 errmsg = 'pathspec'
 137 
 138         elif value_type == 'pathlist':
 139             if isinstance(value, string_types):
 140                 value = [x.strip() for x in value.split(',')]
 141 
 142             if isinstance(value, Sequence):
 143                 value = [resolve_path(x, basedir=basedir) for x in value]
 144             else:
 145                 errmsg = 'pathlist'
 146 
 147         elif value_type in ('str', 'string'):
 148             if isinstance(value, (string_types, AnsibleVaultEncryptedUnicode)):
 149                 value = unquote(to_text(value, errors='surrogate_or_strict'))
 150             else:
 151                 errmsg = 'string'
 152 
 153         # defaults to string type
 154         elif isinstance(value, (string_types, AnsibleVaultEncryptedUnicode)):
 155             value = unquote(to_text(value, errors='surrogate_or_strict'))
 156 
 157         if errmsg:
 158             raise ValueError('Invalid type provided for "%s": %s' % (errmsg, to_native(value)))
 159 
 160     return to_text(value, errors='surrogate_or_strict', nonstring='passthru')

 164 def resolve_path(path, basedir=None):
 165     ''' resolve relative or 'variable' paths '''
 166     if '{{CWD}}' in path:  # allow users to force CWD using 'magic' {{CWD}}
 167         path = path.replace('{{CWD}}', os.getcwd())
 168 
 169     return unfrackpath(path, follow=False, basedir=basedir)

 173 def get_config_type(cfile):
 174 
 175     ftype = None
 176     if cfile is not None:
 177         ext = os.path.splitext(cfile)[-1]
 178         if ext in ('.ini', '.cfg'):
 179             ftype = 'ini'
 180         elif ext in ('.yaml', '.yml'):
 181             ftype = 'yaml'
 182         else:
 183             raise AnsibleOptionsError("Unsupported configuration file extension for %s: %s" % (cfile, to_native(ext)))
 184 
 185     return ftype

 200 def find_ini_config_file(warnings=None):
 201     ''' Load INI Config File order(first found is used): ENV, CWD, HOME, /etc/ansible '''
 202     # FIXME: eventually deprecate ini configs
 203 
 204     if warnings is None:
 205         # Note: In this case, warnings does nothing
 206         warnings = set()
 207 
 208     # A value that can never be a valid path so that we can tell if ANSIBLE_CONFIG was set later
 209     # We can't use None because we could set path to None.
 210     SENTINEL = object
 211 
 212     potential_paths = []
 213 
 214     # Environment setting
 215     path_from_env = os.getenv("ANSIBLE_CONFIG", SENTINEL)
 216     if path_from_env is not SENTINEL:
 217         path_from_env = unfrackpath(path_from_env, follow=False)
 218         if os.path.isdir(to_bytes(path_from_env)):
 219             path_from_env = os.path.join(path_from_env, "ansible.cfg")
 220         potential_paths.append(path_from_env)
 221 
 222     # Current working directory
 223     warn_cmd_public = False
 224     try:
 225         cwd = os.getcwd()
 226         perms = os.stat(cwd)
 227         cwd_cfg = os.path.join(cwd, "ansible.cfg")
 228         if perms.st_mode & stat.S_IWOTH:
 229             # Working directory is world writable so we'll skip it.
 230             # Still have to look for a file here, though, so that we know if we have to warn
 231             if os.path.exists(cwd_cfg):
 232                 warn_cmd_public = True
 233         else:
 234             potential_paths.append(to_text(cwd_cfg, errors='surrogate_or_strict'))
 235     except OSError:
 236         # If we can't access cwd, we'll simply skip it as a possible config source
 237         pass
 238 
 239     # Per user location
 240     potential_paths.append(unfrackpath("~/.ansible.cfg", follow=False))
 241 
 242     # System location
 243     potential_paths.append("/etc/ansible/ansible.cfg")
 244 
 245     for path in potential_paths:
 246         b_path = to_bytes(path)
 247         if os.path.exists(b_path) and os.access(b_path, os.R_OK):
 248             break
 249     else:
 250         path = None
 251 
 252     # Emit a warning if all the following are true:
 253     # * We did not use a config from ANSIBLE_CONFIG
 254     # * There's an ansible.cfg in the current working directory that we skipped
 255     if path_from_env != path and warn_cmd_public:
 256         warnings.add(u"Ansible is being run in a world writable directory (%s),"
 257                      u" ignoring it as an ansible.cfg source."
 258                      u" For more information see"
 259                      u" https://docs.ansible.com/ansible/devel/reference_appendices/config.html#cfg-in-world-writable-dir"
 260                      % to_text(cwd))
 261 
 262     return path

 293     def _read_config_yaml_file(self, yml_file):
 294         # TODO: handle relative paths as relative to the directory containing the current playbook instead of CWD
 295         # Currently this is only used with absolute paths to the `ansible/config` directory
 296         yml_file = to_bytes(yml_file)
 297         if os.path.exists(yml_file):
 298             with open(yml_file, 'rb') as config_def:
 299                 return yaml_load(config_def, Loader=SafeLoader) or {}
 300         raise AnsibleError(
 301             "Missing base YAML definition file (bad install?): %s" % to_native(yml_file))

 303     def _parse_config_file(self, cfile=None):
 304         ''' return flat configuration settings from file(s) '''
 305         # TODO: take list of files with merge/nomerge
 306 
 307         if cfile is None:
 308             cfile = self._config_file
 309 
 310         ftype = get_config_type(cfile)
 311         if cfile is not None:
 312             if ftype == 'ini':
 313                 self._parsers[cfile] = configparser.ConfigParser()
 314                 with open(to_bytes(cfile), 'rb') as f:
 315                     try:
 316                         cfg_text = to_text(f.read(), errors='surrogate_or_strict')
 317                     except UnicodeError as e:
 318                         raise AnsibleOptionsError("Error reading config file(%s) because the config file was not utf8 encoded: %s" % (cfile, to_native(e)))
 319                 try:
 320                     if PY3:
 321                         self._parsers[cfile].read_string(cfg_text)
 322                     else:
 323                         cfg_file = io.StringIO(cfg_text)
 324                         self._parsers[cfile].readfp(cfg_file)
 325                 except configparser.Error as e:
 326                     raise AnsibleOptionsError("Error reading config file (%s): %s" % (cfile, to_native(e)))
 327             # FIXME: this should eventually handle yaml config files
 328             # elif ftype == 'yaml':
 329             #     with open(cfile, 'rb') as config_stream:
 330             #         self._parsers[cfile] = yaml.safe_load(config_stream)
 331             else:
 332                 raise AnsibleOptionsError("Unsupported configuration file type: %s" % to_native(ftype))

 338     def get_plugin_options(self, plugin_type, name, keys=None, variables=None, direct=None):
 339 
 340         options = {}
 341         defs = self.get_configuration_definitions(plugin_type, name)
 342         for option in defs:
 343             options[option] = self.get_config_value(option, plugin_type=plugin_type, plugin_name=name, keys=keys, variables=variables, direct=direct)
 344 
 345         return options

 347     def get_plugin_vars(self, plugin_type, name):
 348 
 349         pvars = []
 350         for pdef in self.get_configuration_definitions(plugin_type, name).values():
 351             if 'vars' in pdef and pdef['vars']:
 352                 for var_entry in pdef['vars']:
 353                     pvars.append(var_entry['name'])
 354         return pvars

 381     def _loop_entries(self, container, entry_list):
 382         ''' repeat code for value entry assignment '''
 383 
 384         value = None
 385         origin = None
 386         for entry in entry_list:
 387             name = entry.get('name')
 388             try:
 389                 temp_value = container.get(name, None)
 390             except UnicodeEncodeError:
 391                 self.WARNINGS.add(u'value for config entry {0} contains invalid characters, ignoring...'.format(to_text(name)))
 392                 continue
 393             if temp_value is not None:  # only set if entry is defined in container
 394                 # inline vault variables should be converted to a text string
 395                 if isinstance(temp_value, AnsibleVaultEncryptedUnicode):
 396                     temp_value = to_text(temp_value, errors='surrogate_or_strict')
 397 
 398                 value = temp_value
 399                 origin = name
 400 
 401                 # deal with deprecation of setting source, if used
 402                 if 'deprecated' in entry:
 403                     self.DEPRECATED.append((entry['name'], entry['deprecated']))
 404 
 405         return value, origin

 407     def get_config_value(self, config, cfile=None, plugin_type=None, plugin_name=None, keys=None, variables=None, direct=None):
 408         ''' wrapper '''
 409 
 410         try:
 411             value, _drop = self.get_config_value_and_origin(config, cfile=cfile, plugin_type=plugin_type, plugin_name=plugin_name,
 412                                                             keys=keys, variables=variables, direct=direct)
 413         except AnsibleError:
 414             raise
 415         except Exception as e:
 416             raise AnsibleError("Unhandled exception when retrieving %s:\n%s" % (config, to_native(e)), orig_exc=e)
 417         return value

 419     def get_config_value_and_origin(self, config, cfile=None, plugin_type=None, plugin_name=None, keys=None, variables=None, direct=None):
 420         ''' Given a config key figure out the actual value and report on the origin of the settings '''
 421         if cfile is None:
 422             # use default config
 423             cfile = self._config_file
 424 
 425         # Note: sources that are lists listed in low to high precedence (last one wins)
 426         value = None
 427         origin = None
 428 
 429         defs = self.get_configuration_definitions(plugin_type, plugin_name)
 430         if config in defs:
 431 
 432             # direct setting via plugin arguments, can set to None so we bypass rest of processing/defaults
 433             direct_aliases = []
 434             if direct:
 435                 direct_aliases = [direct[alias] for alias in defs[config].get('aliases', []) if alias in direct]
 436             if direct and config in direct:
 437                 value = direct[config]
 438                 origin = 'Direct'
 439             elif direct and direct_aliases:
 440                 value = direct_aliases[0]
 441                 origin = 'Direct'
 442 
 443             else:
 444                 # Use 'variable overrides' if present, highest precedence, but only present when querying running play
 445                 if variables and defs[config].get('vars'):
 446                     value, origin = self._loop_entries(variables, defs[config]['vars'])
 447                     origin = 'var: %s' % origin
 448 
 449                 # use playbook keywords if you have em
 450                 if value is None and keys and config in keys:
 451                     value, origin = keys[config], 'keyword'
 452                     origin = 'keyword: %s' % origin
 453 
 454                 # env vars are next precedence
 455                 if value is None and defs[config].get('env'):
 456                     value, origin = self._loop_entries(py3compat.environ, defs[config]['env'])
 457                     origin = 'env: %s' % origin
 458 
 459                 # try config file entries next, if we have one
 460                 if self._parsers.get(cfile, None) is None:
 461                     self._parse_config_file(cfile)
 462 
 463                 if value is None and cfile is not None:
 464                     ftype = get_config_type(cfile)
 465                     if ftype and defs[config].get(ftype):
 466                         if ftype == 'ini':
 467                             # load from ini config
 468                             try:  # FIXME: generalize _loop_entries to allow for files also, most of this code is dupe
 469                                 for ini_entry in defs[config]['ini']:
 470                                     temp_value = get_ini_config_value(self._parsers[cfile], ini_entry)
 471                                     if temp_value is not None:
 472                                         value = temp_value
 473                                         origin = cfile
 474                                         if 'deprecated' in ini_entry:
 475                                             self.DEPRECATED.append(('[%s]%s' % (ini_entry['section'], ini_entry['key']), ini_entry['deprecated']))
 476                             except Exception as e:
 477                                 sys.stderr.write("Error while loading ini config %s: %s" % (cfile, to_native(e)))
 478                         elif ftype == 'yaml':
 479                             # FIXME: implement, also , break down key from defs (. notation???)
 480                             origin = cfile
 481 
 482                 # set default if we got here w/o a value
 483                 if value is None:
 484                     if defs[config].get('required', False):
 485                         if not plugin_type or config not in INTERNAL_DEFS.get(plugin_type, {}):
 486                             raise AnsibleError("No setting was provided for required configuration %s" %
 487                                                to_native(_get_entry(plugin_type, plugin_name, config)))
 488                     else:
 489                         value = defs[config].get('default')
 490                         origin = 'default'
 491                         # skip typing as this is a templated default that will be resolved later in constants, which has needed vars
 492                         if plugin_type is None and isinstance(value, string_types) and (value.startswith('{{') and value.endswith('}}')):
 493                             return value, origin
 494 
 495             # ensure correct type, can raise exceptions on mismatched types
 496             try:
 497                 value = ensure_type(value, defs[config].get('type'), origin=origin)
 498             except ValueError as e:
 499                 if origin.startswith('env:') and value == '':
 500                     # this is empty env var for non string so we can set to default
 501                     origin = 'default'
 502                     value = ensure_type(defs[config].get('default'), defs[config].get('type'), origin=origin)
 503                 else:
 504                     raise AnsibleOptionsError('Invalid type for configuration option %s: %s' %
 505                                               (to_native(_get_entry(plugin_type, plugin_name, config)), to_native(e)))
 506 
 507             # deal with deprecation of the setting
 508             if 'deprecated' in defs[config] and origin != 'default':
 509                 self.DEPRECATED.append((config, defs[config].get('deprecated')))
 510         else:
 511             raise AnsibleError('Requested entry (%s) was not defined in configuration.' % to_native(_get_entry(plugin_type, plugin_name, config)))
 512 
 513         return value, origin

 522     def update_config_data(self, defs=None, configfile=None):
 523         ''' really: update constants '''
 524 
 525         if defs is None:
 526             defs = self._base_defs
 527 
 528         if configfile is None:
 529             configfile = self._config_file
 530 
 531         if not isinstance(defs, dict):
 532             raise AnsibleOptionsError("Invalid configuration definition type: %s for %s" % (type(defs), defs))
 533 
 534         # update the constant for config file
 535         self.data.update_setting(Setting('CONFIG_FILE', configfile, '', 'string'))
 536 
 537         origin = None
 538         # env and config defs can have several entries, ordered in list from lowest to highest precedence
 539         for config in defs:
 540             if not isinstance(defs[config], dict):
 541                 raise AnsibleOptionsError("Invalid configuration definition '%s': type is %s" % (to_native(config), type(defs[config])))
 542 
 543             # get value and origin
 544             try:
 545                 value, origin = self.get_config_value_and_origin(config, configfile)
 546             except Exception as e:
 547                 # Printing the problem here because, in the current code:
 548                 # (1) we can't reach the error handler for AnsibleError before we
 549                 #     hit a different error due to lack of working config.
 550                 # (2) We don't have access to display yet because display depends on config
 551                 #     being properly loaded.
 552                 #
 553                 # If we start getting double errors printed from this section of code, then the
 554                 # above problem #1 has been fixed.  Revamp this to be more like the try: except
 555                 # in get_config_value() at that time.
 556                 sys.stderr.write("Unhandled error:\n %s\n\n" % traceback.format_exc())
 557                 raise AnsibleError("Invalid settings supplied for %s: %s\n" % (config, to_native(e)), orig_exc=e)
 558 
 559             # set the constant
 560             self.data.update_setting(Setting(config, value, origin, defs[config].get('type', 'string')))
