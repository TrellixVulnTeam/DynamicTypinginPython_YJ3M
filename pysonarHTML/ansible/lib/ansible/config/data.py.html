  15

  17         if plugin is None:
  18             setting = self._global_settings.get(name)
  19         elif plugin.type in self._plugins and plugin.name in self._plugins[plugin.type]:
  20             setting = self._plugins[plugin.type][plugin.name].get(name)
  21 
  22         return setting

  24     def get_settings(self, plugin=None):
  25 
  26         settings = []
  27         if plugin is None:
  28             settings = [self._global_settings[k] for k in self._global_settings]
  29         elif plugin.type in self._plugins and plugin.name in self._plugins[plugin.type]:
  30             settings = [self._plugins[plugin.type][plugin.name][k] for k in self._plugins[plugin.type][plugin.name]]
  31 
  32         return settings
  33 
  34     def update_setting(self, setting, plugin=None):
  35 
  36         if plugin is None:
  37             self._global_settings[setting.name] = setting
  38         else:
  39             if plugin.type not in self._plugins:
  40                 self._plugins[plugin.type] = {}
  41             if plugin.name not in self._plugins[plugin.type]:
  42                 self._plugins[plugin.type][plugin.name] = {}
  43             self._plugins[plugin.type][plugin.name][setting.name] = setting
