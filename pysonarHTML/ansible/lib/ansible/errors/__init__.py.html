  38 class AnsibleError(Exception):
  39     '''
  40     This is the base class for all errors raised from Ansible code,
  41     and can be instantiated with two optional parameters beyond the
  42     error message to control whether detailed information is displayed
  43     when the error occurred while parsing a data file of some kind.
  44 
  45     Usage:
  46 
  47         raise AnsibleError('some message here', obj=obj, show_content=True)
  48 
  49     Where "obj" is some subclass of ansible.parsing.yaml.objects.AnsibleBaseYAMLObject,
  50     which should be returned by the DataLoader() class.
  51     '''
  52 
  53     def __init__(self, message="", obj=None, show_content=True, suppress_extended_error=False, orig_exc=None):
  54         super(AnsibleError, self).__init__(message)
  55 
  56         # we import this here to prevent an import loop problem,
  57         # since the objects code also imports ansible.errors
  58         from ansible.parsing.yaml.objects import AnsibleBaseYAMLObject
  59 
  60         self._obj = obj
  61         self._show_content = show_content
  62         if obj and isinstance(obj, AnsibleBaseYAMLObject):
  63             extended_error = self._get_extended_error()
  64             if extended_error and not suppress_extended_error:
  65                 self.message = '%s\n\n%s' % (to_native(message), to_native(extended_error))
  66             else:
  67                 self.message = '%s' % to_native(message)
  68         else:
  69             self.message = '%s' % to_native(message)
  70         if orig_exc:
  71             self.orig_exc = orig_exc
  72 
  73     def __str__(self):
  74         return self.message
  75 
  76     def __repr__(self):
  77         return self.message
  78 
  79     def _get_error_lines_from_file(self, file_name, line_number):
  80         '''
  81         Returns the line in the file which corresponds to the reported error
  82         location, as well as the line preceding it (if the error did not
  83         occur on the first line), to provide context to the error.
  84         '''
  85 
  86         target_line = ''
  87         prev_line = ''
  88 
  89         with open(file_name, 'r') as f:
  90             lines = f.readlines()
  91 
  92             target_line = lines[line_number]
  93             if line_number > 0:
  94                 prev_line = lines[line_number - 1]
  95 
  96         return (target_line, prev_line)
  97 
  98     def _get_extended_error(self):
  99         '''
 100         Given an object reporting the location of the exception in a file, return
 101         detailed information regarding it including:
 102 
 103           * the line which caused the error as well as the one preceding it
 104           * causes and suggested remedies for common syntax errors
 105 
 106         If this error was created with show_content=False, the reporting of content
 107         is suppressed, as the file contents may be sensitive (ie. vault data).
 108         '''
 109 
 110         error_message = ''
 111 
 112         try:
 113             (src_file, line_number, col_number) = self._obj.ansible_pos
 114             error_message += YAML_POSITION_DETAILS % (src_file, line_number, col_number)
 115             if src_file not in ('<string>', '<unicode>') and self._show_content:
 116                 (target_line, prev_line) = self._get_error_lines_from_file(src_file, line_number - 1)
 117                 target_line = to_text(target_line)
 118                 prev_line = to_text(prev_line)
 119                 if target_line:
 120                     stripped_line = target_line.replace(" ", "")
 121 
 122                     # Check for k=v syntax in addition to YAML syntax and set the appropriate error position,
 123                     # arrow index
 124                     if re.search(r'\w+(\s+)?=(\s+)?[\w/-]+', prev_line):
 125                         error_position = prev_line.rstrip().find('=')
 126                         arrow_line = (" " * error_position) + "^ here"
 127                         error_message = YAML_POSITION_DETAILS % (src_file, line_number - 1, error_position + 1)
 128                         error_message += "\nThe offending line appears to be:\n\n%s\n%s\n\n" % (prev_line.rstrip(), arrow_line)
 129                         error_message += YAML_AND_SHORTHAND_ERROR
 130                     else:
 131                         arrow_line = (" " * (col_number - 1)) + "^ here"
 132                         error_message += "\nThe offending line appears to be:\n\n%s\n%s\n%s\n" % (prev_line.rstrip(), target_line.rstrip(), arrow_line)
 133 
 134                     # TODO: There may be cases where there is a valid tab in a line that has other errors.
 135                     if '\t' in target_line:
 136                         error_message += YAML_COMMON_LEADING_TAB_ERROR
 137                     # common error/remediation checking here:
 138                     # check for unquoted vars starting lines
 139                     if ('{{' in target_line and '}}' in target_line) and ('"{{' not in target_line or "'{{" not in target_line):
 140                         error_message += YAML_COMMON_UNQUOTED_VARIABLE_ERROR
 141                     # check for common dictionary mistakes
 142                     elif ":{{" in stripped_line and "}}" in stripped_line:
 143                         error_message += YAML_COMMON_DICT_ERROR
 144                     # check for common unquoted colon mistakes
 145                     elif (len(target_line) and
 146                             len(target_line) > 1 and
 147                             len(target_line) > col_number and
 148                             target_line[col_number] == ":" and
 149                             target_line.count(':') > 1):
 150                         error_message += YAML_COMMON_UNQUOTED_COLON_ERROR
 151                     # otherwise, check for some common quoting mistakes
 152                     else:
 153                         # FIXME: This needs to split on the first ':' to account for modules like lineinfile
 154                         # that may have lines that contain legitimate colons, e.g., line: 'i ALL= (ALL) NOPASSWD: ALL'
 155                         # and throw off the quote matching logic.
 156                         parts = target_line.split(":")
 157                         if len(parts) > 1:
 158                             middle = parts[1].strip()
 159                             match = False
 160                             unbalanced = False
 161 
 162                             if middle.startswith("'") and not middle.endswith("'"):
 163                                 match = True
 164                             elif middle.startswith('"') and not middle.endswith('"'):
 165                                 match = True
 166 
 167                             if (len(middle) > 0 and
 168                                     middle[0] in ['"', "'"] and
 169                                     middle[-1] in ['"', "'"] and
 170                                     target_line.count("'") > 2 or
 171                                     target_line.count('"') > 2):
 172                                 unbalanced = True
 173 
 174                             if match:
 175                                 error_message += YAML_COMMON_PARTIALLY_QUOTED_LINE_ERROR
 176                             if unbalanced:
 177                                 error_message += YAML_COMMON_UNBALANCED_QUOTES_ERROR
 178 
 179         except (IOError, TypeError):
 180             error_message += '\n(could not open file to display line)'
 181         except IndexError:
 182             error_message += '\n(specified line no longer in file, maybe it changed?)'
 183 
 184         return error_message

 252 class AnsibleFileNotFound(AnsibleRuntimeError):
 253     ''' a file missing failure '''
 254 
 255     def __init__(self, message="", obj=None, show_content=True, suppress_extended_error=False, orig_exc=None, paths=None, file_name=None):
 256 
 257         self.file_name = file_name
 258         self.paths = paths
 259 
 260         if message:
 261             message += "\n"
 262         if self.file_name:
 263             message += "Could not find or access '%s'" % to_text(self.file_name)
 264         else:
 265             message += "Could not find file"
 266 
 267         if self.paths and isinstance(self.paths, Sequence):
 268             searched = to_text('\n\t'.join(self.paths))
 269             if message:
 270                 message += "\n"
 271             message += "Searched in:\n\t%s" % searched
 272 
 273         message += " on the Ansible Controller.\nIf you are using a module and expect the file to exist on the remote, see the remote_src option"
 274 
 275         super(AnsibleFileNotFound, self).__init__(message=message, obj=obj, show_content=show_content,
 276                                                   suppress_extended_error=suppress_extended_error, orig_exc=orig_exc)

 279 # These Exceptions are temporary, using them as flow control until we can get a better solution.
 280 # DO NOT USE as they will probably be removed soon.
 281 # We will port the action modules in our tree to use a context manager instead.
 282 class AnsibleAction(AnsibleRuntimeError):
 283     ''' Base Exception for Action plugin flow control '''
 284 
 285     def __init__(self, message="", obj=None, show_content=True, suppress_extended_error=False, orig_exc=None, result=None):
 286 
 287         super(AnsibleAction, self).__init__(message=message, obj=obj, show_content=show_content,
 288                                             suppress_extended_error=suppress_extended_error, orig_exc=orig_exc)
 289         if result is None:
 290             self.result = {}
 291         else:
 292             self.result = result
 293
