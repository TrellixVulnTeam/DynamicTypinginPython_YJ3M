  25 class PSModuleDepFinder(object):
  26 
  27     def __init__(self):
  28         # This is also used by validate-modules to get a module's required utils in base and a collection.
  29         self.ps_modules = dict()
  30         self.exec_scripts = dict()
  31 
  32         # by defining an explicit dict of cs utils and where they are used, we
  33         # can potentially save time by not adding the type multiple times if it
  34         # isn't needed
  35         self.cs_utils_wrapper = dict()
  36         self.cs_utils_module = dict()
  37 
  38         self.ps_version = None
  39         self.os_version = None
  40         self.become = False
  41 
  42         self._re_cs_module = [
  43             # Reference C# module_util in another C# util, this must always be the fully qualified name.
  44             # 'using ansible_collections.{namespace}.{collection}.plugins.module_utils.{name}'
  45             re.compile(to_bytes(r'(?i)^using\s((Ansible\..+)|'
  46                                 r'(ansible_collections\.\w+\.\w+\.plugins\.module_utils\.[\w\.]+));\s*$')),
  47         ]
  48 
  49         self._re_cs_in_ps_module = [
  50             # Reference C# module_util in a PowerShell module
  51             # '#AnsibleRequires -CSharpUtil Ansible.{name}'
  52             # '#AnsibleRequires -CSharpUtil ansible_collections.{namespace}.{collection}.plugins.module_utils.{name}'
  53             # '#AnsibleRequires -CSharpUtil ..module_utils.{name}'
  54             re.compile(to_bytes(r'(?i)^#\s*ansiblerequires\s+-csharputil\s+((Ansible\..+)|'
  55                                 r'(ansible_collections\.\w+\.\w+\.plugins\.module_utils\.[\w\.]+)|'
  56                                 r'(\.[\w\.]+))')),
  57         ]
  58 
  59         self._re_ps_module = [
  60             # Original way of referencing a builtin module_util
  61             # '#Requires -Module Ansible.ModuleUtils.{name}
  62             re.compile(to_bytes(r'(?i)^#\s*requires\s+\-module(?:s?)\s*(Ansible\.ModuleUtils\..+)')),
  63             # New way of referencing a builtin and collection module_util
  64             # '#AnsibleRequires -PowerShell Ansible.ModuleUtils.{name}'
  65             # '#AnsibleRequires -PowerShell ansible_collections.{namespace}.{collection}.plugins.module_utils.{name}'
  66             # '#AnsibleRequires -PowerShell ..module_utils.{name}'
  67             re.compile(to_bytes(r'(?i)^#\s*ansiblerequires\s+-powershell\s+((Ansible\.ModuleUtils\..+)|'
  68                                 r'(ansible_collections\.\w+\.\w+\.plugins\.module_utils\.[\w\.]+)|'
  69                                 r'(\.[\w\.]+))')),
  70         ]
  71 
  72         self._re_wrapper = re.compile(to_bytes(r'(?i)^#\s*ansiblerequires\s+-wrapper\s+(\w*)'))
  73         self._re_ps_version = re.compile(to_bytes(r'(?i)^#requires\s+\-version\s+([0-9]+(\.[0-9]+){0,3})$'))
  74         self._re_os_version = re.compile(to_bytes(r'(?i)^#ansiblerequires\s+\-osversion\s+([0-9]+(\.[0-9]+){0,3})$'))
  75         self._re_become = re.compile(to_bytes(r'(?i)^#ansiblerequires\s+\-become$'))
  76 
  77     def scan_module(self, module_data, fqn=None, wrapper=False, powershell=True):
  78         lines = module_data.split(b'\n')
  79         module_utils = set()
  80         if wrapper:
  81             cs_utils = self.cs_utils_wrapper
  82         else:
  83             cs_utils = self.cs_utils_module
  84 
  85         if powershell:
  86             checks = [
  87                 # PS module contains '#Requires -Module Ansible.ModuleUtils.*'
  88                 # PS module contains '#AnsibleRequires -Powershell Ansible.*' (or collections module_utils ref)
  89                 (self._re_ps_module, self.ps_modules, ".psm1"),
  90                 # PS module contains '#AnsibleRequires -CSharpUtil Ansible.*' (or collections module_utils ref)
  91                 (self._re_cs_in_ps_module, cs_utils, ".cs"),
  92             ]
  93         else:
  94             checks = [
  95                 # CS module contains 'using Ansible.*;' or 'using ansible_collections.ns.coll.plugins.module_utils.*;'
  96                 (self._re_cs_module, cs_utils, ".cs"),
  97             ]
  98 
  99         for line in lines:
 100             for check in checks:
 101                 for pattern in check[0]:
 102                     match = pattern.match(line)
 103                     if match:
 104                         # tolerate windows line endings by stripping any remaining
 105                         # newline chars
 106                         module_util_name = to_text(match.group(1).rstrip())
 107 
 108                         if module_util_name not in check[1].keys():
 109                             module_utils.add((module_util_name, check[2], fqn))
 110 
 111                         break
 112 
 113             if powershell:
 114                 ps_version_match = self._re_ps_version.match(line)
 115                 if ps_version_match:
 116                     self._parse_version_match(ps_version_match, "ps_version")
 117 
 118                 os_version_match = self._re_os_version.match(line)
 119                 if os_version_match:
 120                     self._parse_version_match(os_version_match, "os_version")
 121 
 122                 # once become is set, no need to keep on checking recursively
 123                 if not self.become:
 124                     become_match = self._re_become.match(line)
 125                     if become_match:
 126                         self.become = True
 127 
 128             if wrapper:
 129                 wrapper_match = self._re_wrapper.match(line)
 130                 if wrapper_match:
 131                     self.scan_exec_script(wrapper_match.group(1).rstrip())
 132 
 133         # recursively drill into each Requires to see if there are any more
 134         # requirements
 135         for m in set(module_utils):
 136             self._add_module(m, wrapper=wrapper)
 137 
 138     def scan_exec_script(self, name):
 139         # scans lib/ansible/executor/powershell for scripts used in the module
 140         # exec side. It also scans these scripts for any dependencies
 141         name = to_text(name)
 142         if name in self.exec_scripts.keys():
 143             return
 144 
 145         data = pkgutil.get_data("ansible.executor.powershell", to_native(name + ".ps1"))
 146         if data is None:
 147             raise AnsibleError("Could not find executor powershell script "
 148                                "for '%s'" % name)
 149 
 150         b_data = to_bytes(data)
 151 
 152         # remove comments to reduce the payload size in the exec wrappers
 153         if C.DEFAULT_DEBUG:
 154             exec_script = b_data
 155         else:
 156             exec_script = _strip_comments(b_data)
 157         self.exec_scripts[name] = to_bytes(exec_script)
 158         self.scan_module(b_data, wrapper=True, powershell=True)

 160     def _add_module(self, name, wrapper=False):
 161         m, ext, fqn = name
 162         m = to_text(m)
 163 
 164         util_fqn = None
 165 
 166         if m.startswith("Ansible."):
 167             # Builtin util, use plugin loader to get the data
 168             mu_path = ps_module_utils_loader.find_plugin(m, ext)
 169 
 170             if not mu_path:
 171                 raise AnsibleError('Could not find imported module support code '
 172                                    'for \'%s\'' % m)
 173 
 174             module_util_data = to_bytes(_slurp(mu_path))
 175         else:
 176             # Collection util, load the package data based on the util import.
 177 
 178             submodules = m.split(".")
 179             if m.startswith('.'):
 180                 fqn_submodules = fqn.split('.')
 181                 for submodule in submodules:
 182                     if submodule:
 183                         break
 184                     del fqn_submodules[-1]
 185 
 186                 submodules = fqn_submodules + [s for s in submodules if s]
 187 
 188             n_package_name = to_native('.'.join(submodules[:-1]), errors='surrogate_or_strict')
 189             n_resource_name = to_native(submodules[-1] + ext, errors='surrogate_or_strict')
 190 
 191             try:
 192                 module_util = import_module(n_package_name)
 193                 module_util_data = to_bytes(pkgutil.get_data(n_package_name, n_resource_name),
 194                                             errors='surrogate_or_strict')
 195                 util_fqn = to_text("%s.%s " % (n_package_name, submodules[-1]), errors='surrogate_or_strict')
 196 
 197                 # Get the path of the util which is required for coverage collection.
 198                 resource_paths = list(module_util.__path__)
 199                 if len(resource_paths) != 1:
 200                     # This should never happen with a collection but we are just being defensive about it.
 201                     raise AnsibleError("Internal error: Referenced module_util package '%s' contains 0 or multiple "
 202                                        "import locations when we only expect 1." % n_package_name)
 203                 mu_path = os.path.join(resource_paths[0], n_resource_name)
 204             except OSError as err:
 205                 if err.errno == errno.ENOENT:
 206                     raise AnsibleError('Could not find collection imported module support code for \'%s\''
 207                                        % to_native(m))
 208                 else:
 209                     raise
 210 
 211         util_info = {
 212             'data': module_util_data,
 213             'path': to_text(mu_path),
 214         }
 215         if ext == ".psm1":
 216             self.ps_modules[m] = util_info
 217         else:
 218             if wrapper:
 219                 self.cs_utils_wrapper[m] = util_info
 220             else:
 221                 self.cs_utils_module[m] = util_info
 222         self.scan_module(module_util_data, fqn=util_fqn, wrapper=wrapper, powershell=(ext == ".psm1"))
 223 
 224     def _parse_version_match(self, match, attribute):
 225         new_version = to_text(match.group(1)).rstrip()
 226 
 227         # PowerShell cannot cast a string of "1" to Version, it must have at
 228         # least the major.minor for it to be valid so we append 0
 229         if match.group(2) is None:
 230             new_version = "%s.0" % new_version
 231 
 232         existing_version = getattr(self, attribute, None)
 233         if existing_version is None:
 234             setattr(self, attribute, new_version)
 235         else:
 236             # determine which is the latest version and set that
 237             if LooseVersion(new_version) > LooseVersion(existing_version):
 238                 setattr(self, attribute, new_version)

 241 def _slurp(path):
 242     if not os.path.exists(path):
 243         raise AnsibleError("imported module support code does not exist at %s"
 244                            % os.path.abspath(path))
 245     fd = open(path, 'rb')
 246     data = fd.read()
 247     fd.close()
 248     return data

 251 def _strip_comments(source):
 252     # Strip comments and blank lines from the wrapper
 253     buf = []
 254     start_block = False
 255     for line in source.splitlines():
 256         l = line.strip()
 257 
 258         if start_block and l.endswith(b'#>'):
 259             start_block = False
 260             continue
 261         elif start_block:
 262             continue
 263         elif l.startswith(b'<#'):
 264             start_block = True
 265             continue
 266         elif not l or l.startswith(b'#'):
 267             continue
 268 
 269         buf.append(line)
 270     return b'\n'.join(buf)

 273 def _create_powershell_wrapper(b_module_data, module_path, module_args,
 274                                environment, async_timeout, become,
 275                                become_method, become_user, become_password,
 276                                become_flags, substyle, task_vars, module_fqn):
 277     # creates the manifest/wrapper used in PowerShell/C# modules to enable
 278     # things like become and async - this is also called in action/script.py
 279 
 280     # FUTURE: add process_wrapper.ps1 to run module_wrapper in a new process
 281     # if running under a persistent connection and substyle is C# so we
 282     # don't have type conflicts
 283     finder = PSModuleDepFinder()
 284     if substyle != 'script':
 285         # don't scan the module for util dependencies and other Ansible related
 286         # flags if the substyle is 'script' which is set by action/script
 287         finder.scan_module(b_module_data, fqn=module_fqn, powershell=(substyle == "powershell"))
 288 
 289     module_wrapper = "module_%s_wrapper" % substyle
 290     exec_manifest = dict(
 291         module_entry=to_text(base64.b64encode(b_module_data)),
 292         powershell_modules=dict(),
 293         csharp_utils=dict(),
 294         csharp_utils_module=list(),  # csharp_utils only required by a module
 295         module_args=module_args,
 296         actions=[module_wrapper],
 297         environment=environment,
 298         encoded_output=False,
 299     )
 300     finder.scan_exec_script(module_wrapper)
 301 
 302     if async_timeout > 0:
 303         finder.scan_exec_script('exec_wrapper')
 304         finder.scan_exec_script('async_watchdog')
 305         finder.scan_exec_script('async_wrapper')
 306 
 307         exec_manifest["actions"].insert(0, 'async_watchdog')
 308         exec_manifest["actions"].insert(0, 'async_wrapper')
 309         exec_manifest["async_jid"] = str(random.randint(0, 999999999999))
 310         exec_manifest["async_timeout_sec"] = async_timeout
 311         exec_manifest["async_startup_timeout"] = C.config.get_config_value("WIN_ASYNC_STARTUP_TIMEOUT", variables=task_vars)
 312 
 313     if become and resource_from_fqcr(become_method) == 'runas':  # runas and namespace.collection.runas
 314         finder.scan_exec_script('exec_wrapper')
 315         finder.scan_exec_script('become_wrapper')
 316 
 317         exec_manifest["actions"].insert(0, 'become_wrapper')
 318         exec_manifest["become_user"] = become_user
 319         exec_manifest["become_password"] = become_password
 320         exec_manifest['become_flags'] = become_flags
 321 
 322     exec_manifest['min_ps_version'] = finder.ps_version
 323     exec_manifest['min_os_version'] = finder.os_version
 324     if finder.become and 'become_wrapper' not in exec_manifest['actions']:
 325         finder.scan_exec_script('exec_wrapper')
 326         finder.scan_exec_script('become_wrapper')
 327 
 328         exec_manifest['actions'].insert(0, 'become_wrapper')
 329         exec_manifest['become_user'] = 'SYSTEM'
 330         exec_manifest['become_password'] = None
 331         exec_manifest['become_flags'] = None
 332 
 333     coverage_manifest = dict(
 334         module_path=module_path,
 335         module_util_paths=dict(),
 336         output=None,
 337     )
 338     coverage_output = C.config.get_config_value('COVERAGE_REMOTE_OUTPUT', variables=task_vars)
 339     if coverage_output and substyle == 'powershell':
 340         finder.scan_exec_script('coverage_wrapper')
 341         coverage_manifest['output'] = coverage_output
 342 
 343         coverage_path_filter = C.config.get_config_value('COVERAGE_REMOTE_PATH_FILTER', variables=task_vars)
 344         coverage_manifest['path_filter'] = coverage_path_filter
 345 
 346     # make sure Ansible.ModuleUtils.AddType is added if any C# utils are used
 347     if len(finder.cs_utils_wrapper) > 0 or len(finder.cs_utils_module) > 0:
 348         finder._add_module((b"Ansible.ModuleUtils.AddType", ".psm1", None),
 349                            wrapper=False)
 350 
 351     # exec_wrapper is only required to be part of the payload if using
 352     # become or async, to save on payload space we check if exec_wrapper has
 353     # already been added, and remove it manually if it hasn't later
 354     exec_required = "exec_wrapper" in finder.exec_scripts.keys()
 355     finder.scan_exec_script("exec_wrapper")
 356     # must contain an empty newline so it runs the begin/process/end block
 357     finder.exec_scripts["exec_wrapper"] += b"\n\n"
 358 
 359     exec_wrapper = finder.exec_scripts["exec_wrapper"]
 360     if not exec_required:
 361         finder.exec_scripts.pop("exec_wrapper")
 362 
 363     for name, data in finder.exec_scripts.items():
 364         b64_data = to_text(base64.b64encode(data))
 365         exec_manifest[name] = b64_data
 366 
 367     for name, data in finder.ps_modules.items():
 368         b64_data = to_text(base64.b64encode(data['data']))
 369         exec_manifest['powershell_modules'][name] = b64_data
 370         coverage_manifest['module_util_paths'][name] = data['path']
 371 
 372     cs_utils = {}
 373     for cs_util in [finder.cs_utils_wrapper, finder.cs_utils_module]:
 374         for name, data in cs_util.items():
 375             cs_utils[name] = data['data']
 376 
 377     for name, data in cs_utils.items():
 378         b64_data = to_text(base64.b64encode(data))
 379         exec_manifest['csharp_utils'][name] = b64_data
 380     exec_manifest['csharp_utils_module'] = list(finder.cs_utils_module.keys())
 381 
 382     # To save on the data we are sending across we only add the coverage info if coverage is being run
 383     if 'coverage_wrapper' in exec_manifest:
 384         exec_manifest['coverage'] = coverage_manifest
 385 
 386     b_json = to_bytes(json.dumps(exec_manifest))
 387     # delimit the payload JSON from the wrapper to keep sensitive contents out of scriptblocks (which can be logged)
 388     b_data = exec_wrapper + b'\0\0\0\0' + b_json
 389     return b_data
