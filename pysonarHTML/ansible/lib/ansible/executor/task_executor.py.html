  52 def remove_omit(task_args, omit_token):
  53     '''
  54     Remove args with a value equal to the ``omit_token`` recursively
  55     to align with now having suboptions in the argument_spec
  56     '''
  57 
  58     if not isinstance(task_args, dict):
  59         return task_args
  60 
  61     new_args = {}
  62     for i in iteritems(task_args):
  63         if i[1] == omit_token:
  64             continue
  65         elif isinstance(i[1], dict):
  66             new_args[i[0]] = remove_omit(i[1], omit_token)
  67         elif isinstance(i[1], list):
  68             new_args[i[0]] = [remove_omit(v, omit_token) for v in i[1]]
  69         else:
  70             new_args[i[0]] = i[1]
  71 
  72     return new_args

  98     def run(self):
  99         '''
 100         The main executor entrypoint, where we determine if the specified
 101         task requires looping and either runs the task with self._run_loop()
 102         or self._execute(). After that, the returned results are parsed and
 103         returned as a dict.
 104         '''
 105 
 106         display.debug("in run() - task %s" % self._task._uuid)
 107 
 108         try:
 109             try:
 110                 items = self._get_loop_items()
 111             except AnsibleUndefinedVariable as e:
 112                 # save the error raised here for use later
 113                 items = None
 114                 self._loop_eval_error = e
 115 
 116             if items is not None:
 117                 if len(items) > 0:
 118                     item_results = self._run_loop(items)
 119 
 120                     # create the overall result item
 121                     res = dict(results=item_results)
 122 
 123                     # loop through the item results, and set the global changed/failed result flags based on any item.
 124                     for item in item_results:
 125                         if 'changed' in item and item['changed'] and not res.get('changed'):
 126                             res['changed'] = True
 127                         if 'failed' in item and item['failed']:
 128                             item_ignore = item.pop('_ansible_ignore_errors')
 129                             if not res.get('failed'):
 130                                 res['failed'] = True
 131                                 res['msg'] = 'One or more items failed'
 132                                 self._task.ignore_errors = item_ignore
 133                             elif self._task.ignore_errors and not item_ignore:
 134                                 self._task.ignore_errors = item_ignore
 135 
 136                         # ensure to accumulate these
 137                         for array in ['warnings', 'deprecations']:
 138                             if array in item and item[array]:
 139                                 if array not in res:
 140                                     res[array] = []
 141                                 if not isinstance(item[array], list):
 142                                     item[array] = [item[array]]
 143                                 res[array] = res[array] + item[array]
 144                                 del item[array]
 145 
 146                     if not res.get('Failed', False):
 147                         res['msg'] = 'All items completed'
 148                 else:
 149                     res = dict(changed=False, skipped=True, skipped_reason='No items in the list', results=[])
 150             else:
 151                 display.debug("calling self._execute()")
 152                 res = self._execute()
 153                 display.debug("_execute() done")
 154 
 155             # make sure changed is set in the result, if it's not present
 156             if 'changed' not in res:
 157                 res['changed'] = False
 158 
 159             def _clean_res(res, errors='surrogate_or_strict'):
 160                 if isinstance(res, binary_type):
 161                     return to_unsafe_text(res, errors=errors)
 162                 elif isinstance(res, dict):
 163                     for k in res:
 164                         try:
 165                             res[k] = _clean_res(res[k], errors=errors)
 166                         except UnicodeError:
 167                             if k == 'diff':
 168                                 # If this is a diff, substitute a replacement character if the value
 169                                 # is undecodable as utf8.  (Fix #21804)
 170                                 display.warning("We were unable to decode all characters in the module return data."
 171                                                 " Replaced some in an effort to return as much as possible")
 172                                 res[k] = _clean_res(res[k], errors='surrogate_then_replace')
 173                             else:
 174                                 raise
 175                 elif isinstance(res, list):
 176                     for idx, item in enumerate(res):
 177                         res[idx] = _clean_res(item, errors=errors)
 178                 return res
 179 
 180             display.debug("dumping result to json")
 181             res = _clean_res(res)
 182             display.debug("done dumping result, returning")
 183             return res
 184         except AnsibleError as e:
 185             return dict(failed=True, msg=wrap_var(to_text(e, nonstring='simplerepr')), _ansible_no_log=self._play_context.no_log)
 186         except Exception as e:
 187             return dict(failed=True, msg='Unexpected failure during module execution.', exception=to_text(traceback.format_exc()),
 188                         stdout='', _ansible_no_log=self._play_context.no_log)
 189         finally:
 190             try:
 191                 self._connection.close()
 192             except AttributeError:
 193                 pass
 194             except Exception as e:
 195                 display.debug(u"error closing connection: %s" % to_text(e))

 197     def _get_loop_items(self):
 198         '''
 199         Loads a lookup plugin to handle the with_* portion of a task (if specified),
 200         and returns the items result.
 201         '''
 202 
 203         # get search path for this task to pass to lookup plugins
 204         self._job_vars['ansible_search_path'] = self._task.get_search_path()
 205 
 206         # ensure basedir is always in (dwim already searches here but we need to display it)
 207         if self._loader.get_basedir() not in self._job_vars['ansible_search_path']:
 208             self._job_vars['ansible_search_path'].append(self._loader.get_basedir())
 209 
 210         templar = Templar(loader=self._loader, shared_loader_obj=self._shared_loader_obj, variables=self._job_vars)
 211         items = None
 212         loop_cache = self._job_vars.get('_ansible_loop_cache')
 213         if loop_cache is not None:
 214             # _ansible_loop_cache may be set in `get_vars` when calculating `delegate_to`
 215             # to avoid reprocessing the loop
 216             items = loop_cache
 217         elif self._task.loop_with:
 218             if self._task.loop_with in self._shared_loader_obj.lookup_loader:
 219                 fail = True
 220                 if self._task.loop_with == 'first_found':
 221                     # first_found loops are special. If the item is undefined then we want to fall through to the next value rather than failing.
 222                     fail = False
 223 
 224                 loop_terms = listify_lookup_plugin_terms(terms=self._task.loop, templar=templar, loader=self._loader, fail_on_undefined=fail,
 225                                                          convert_bare=False)
 226                 if not fail:
 227                     loop_terms = [t for t in loop_terms if not templar.is_template(t)]
 228 
 229                 # get lookup
 230                 mylookup = self._shared_loader_obj.lookup_loader.get(self._task.loop_with, loader=self._loader, templar=templar)
 231 
 232                 # give lookup task 'context' for subdir (mostly needed for first_found)
 233                 for subdir in ['template', 'var', 'file']:  # TODO: move this to constants?
 234                     if subdir in self._task.action:
 235                         break
 236                 setattr(mylookup, '_subdir', subdir + 's')
 237 
 238                 # run lookup
 239                 items = wrap_var(mylookup.run(terms=loop_terms, variables=self._job_vars, wantlist=True))
 240             else:
 241                 raise AnsibleError("Unexpected failure in finding the lookup named '%s' in the available lookup plugins" % self._task.loop_with)
 242 
 243         elif self._task.loop is not None:
 244             items = templar.template(self._task.loop)
 245             if not isinstance(items, list):
 246                 raise AnsibleError(
 247                     "Invalid data passed to 'loop', it requires a list, got this instead: %s."
 248                     " Hint: If you passed a list/dict of just one element,"
 249                     " try adding wantlist=True to your lookup invocation or use q/query instead of lookup." % items
 250                 )
 251 
 252         return items

 254     def _run_loop(self, items):
 255         '''
 256         Runs the task with the loop items specified and collates the result
 257         into an array named 'results' which is inserted into the final result
 258         along with the item for which the loop ran.
 259         '''
 260 
 261         results = []
 262 
 263         # make copies of the job vars and task so we can add the item to
 264         # the variables and re-validate the task with the item variable
 265         # task_vars = self._job_vars.copy()
 266         task_vars = self._job_vars
 267 
 268         loop_var = 'item'
 269         index_var = None
 270         label = None
 271         loop_pause = 0
 272         extended = False
 273         templar = Templar(loader=self._loader, shared_loader_obj=self._shared_loader_obj, variables=self._job_vars)
 274 
 275         # FIXME: move this to the object itself to allow post_validate to take care of templating (loop_control.post_validate)
 276         if self._task.loop_control:
 277             loop_var = templar.template(self._task.loop_control.loop_var)
 278             index_var = templar.template(self._task.loop_control.index_var)
 279             loop_pause = templar.template(self._task.loop_control.pause)
 280             extended = templar.template(self._task.loop_control.extended)
 281 
 282             # This may be 'None',so it is templated below after we ensure a value and an item is assigned
 283             label = self._task.loop_control.label
 284 
 285         # ensure we always have a label
 286         if label is None:
 287             label = '{{' + loop_var + '}}'
 288 
 289         if loop_var in task_vars:
 290             display.warning(u"The loop variable '%s' is already in use. "
 291                             u"You should set the `loop_var` value in the `loop_control` option for the task"
 292                             u" to something else to avoid variable collisions and unexpected behavior." % loop_var)
 293 
 294         ran_once = False
 295 
 296         no_log = False
 297         items_len = len(items)
 298         for item_index, item in enumerate(items):
 299             task_vars['ansible_loop_var'] = loop_var
 300 
 301             task_vars[loop_var] = item
 302             if index_var:
 303                 task_vars['ansible_index_var'] = index_var
 304                 task_vars[index_var] = item_index
 305 
 306             if extended:
 307                 task_vars['ansible_loop'] = {
 308                     'allitems': items,
 309                     'index': item_index + 1,
 310                     'index0': item_index,
 311                     'first': item_index == 0,
 312                     'last': item_index + 1 == items_len,
 313                     'length': items_len,
 314                     'revindex': items_len - item_index,
 315                     'revindex0': items_len - item_index - 1,
 316                 }
 317                 try:
 318                     task_vars['ansible_loop']['nextitem'] = items[item_index + 1]
 319                 except IndexError:
 320                     pass
 321                 if item_index - 1 >= 0:
 322                     task_vars['ansible_loop']['previtem'] = items[item_index - 1]
 323 
 324             # Update template vars to reflect current loop iteration
 325             templar.available_variables = task_vars
 326 
 327             # pause between loop iterations
 328             if loop_pause and ran_once:
 329                 try:
 330                     time.sleep(float(loop_pause))
 331                 except ValueError as e:
 332                     raise AnsibleError('Invalid pause value: %s, produced error: %s' % (loop_pause, to_native(e)))
 333             else:
 334                 ran_once = True
 335 
 336             try:
 337                 tmp_task = self._task.copy(exclude_parent=True, exclude_tasks=True)
 338                 tmp_task._parent = self._task._parent
 339                 tmp_play_context = self._play_context.copy()
 340             except AnsibleParserError as e:
 341                 results.append(dict(failed=True, msg=to_text(e)))
 342                 continue
 343 
 344             # now we swap the internal task and play context with their copies,
 345             # execute, and swap them back so we can do the next iteration cleanly
 346             (self._task, tmp_task) = (tmp_task, self._task)
 347             (self._play_context, tmp_play_context) = (tmp_play_context, self._play_context)
 348             res = self._execute(variables=task_vars)
 349             task_fields = self._task.dump_attrs()
 350             (self._task, tmp_task) = (tmp_task, self._task)
 351             (self._play_context, tmp_play_context) = (tmp_play_context, self._play_context)
 352 
 353             # update 'general no_log' based on specific no_log
 354             no_log = no_log or tmp_task.no_log
 355 
 356             # now update the result with the item info, and append the result
 357             # to the list of results
 358             res[loop_var] = item
 359             res['ansible_loop_var'] = loop_var
 360             if index_var:
 361                 res[index_var] = item_index
 362                 res['ansible_index_var'] = index_var
 363             if extended:
 364                 res['ansible_loop'] = task_vars['ansible_loop']
 365 
 366             res['_ansible_item_result'] = True
 367             res['_ansible_ignore_errors'] = task_fields.get('ignore_errors')
 368 
 369             # gets templated here unlike rest of loop_control fields, depends on loop_var above
 370             try:
 371                 res['_ansible_item_label'] = templar.template(label, cache=False)
 372             except AnsibleUndefinedVariable as e:
 373                 res.update({
 374                     'failed': True,
 375                     'msg': 'Failed to template loop_control.label: %s' % to_text(e)
 376                 })
 377 
 378             self._final_q.put(
 379                 TaskResult(
 380                     self._host.name,
 381                     self._task._uuid,
 382                     res,
 383                     task_fields=task_fields,
 384                 ),
 385                 block=False,
 386             )
 387             results.append(res)
 388             del task_vars[loop_var]
 389 
 390             # clear 'connection related' plugin variables for next iteration
 391             if self._connection:
 392                 clear_plugins = {
 393                     'connection': self._connection._load_name,
 394                     'shell': self._connection._shell._load_name
 395                 }
 396                 if self._connection.become:
 397                     clear_plugins['become'] = self._connection.become._load_name
 398 
 399                 for plugin_type, plugin_name in iteritems(clear_plugins):
 400                     for var in C.config.get_plugin_vars(plugin_type, plugin_name):
 401                         if var in task_vars and var not in self._job_vars:
 402                             del task_vars[var]
 403 
 404         self._task.no_log = no_log
 405 
 406         return results

 408     def _execute(self, variables=None):
 409         '''
 410         The primary workhorse of the executor system, this runs the task
 411         on the specified host (which may be the delegated_to host) and handles
 412         the retry/until and block rescue/always execution
 413         '''
 414 
 415         if variables is None:
 416             variables = self._job_vars
 417 
 418         templar = Templar(loader=self._loader, shared_loader_obj=self._shared_loader_obj, variables=variables)
 419 
 420         context_validation_error = None
 421         try:
 422             # apply the given task's information to the connection info,
 423             # which may override some fields already set by the play or
 424             # the options specified on the command line
 425             self._play_context = self._play_context.set_task_and_variable_override(task=self._task, variables=variables, templar=templar)
 426 
 427             # fields set from the play/task may be based on variables, so we have to
 428             # do the same kind of post validation step on it here before we use it.
 429             self._play_context.post_validate(templar=templar)
 430 
 431             # now that the play context is finalized, if the remote_addr is not set
 432             # default to using the host's address field as the remote address
 433             if not self._play_context.remote_addr:
 434                 self._play_context.remote_addr = self._host.address
 435 
 436             # We also add "magic" variables back into the variables dict to make sure
 437             # a certain subset of variables exist.
 438             self._play_context.update_vars(variables)
 439 
 440             # FIXME: update connection/shell plugin options
 441         except AnsibleError as e:
 442             # save the error, which we'll raise later if we don't end up
 443             # skipping this task during the conditional evaluation step
 444             context_validation_error = e
 445 
 446         # Evaluate the conditional (if any) for this task, which we do before running
 447         # the final task post-validation. We do this before the post validation due to
 448         # the fact that the conditional may specify that the task be skipped due to a
 449         # variable not being present which would otherwise cause validation to fail
 450         try:
 451             if not self._task.evaluate_conditional(templar, variables):
 452                 display.debug("when evaluation is False, skipping this task")
 453                 return dict(changed=False, skipped=True, skip_reason='Conditional result was False', _ansible_no_log=self._play_context.no_log)
 454         except AnsibleError as e:
 455             # loop error takes precedence
 456             if self._loop_eval_error is not None:
 457                 # Display the error from the conditional as well to prevent
 458                 # losing information useful for debugging.
 459                 display.v(to_text(e))
 460                 raise self._loop_eval_error  # pylint: disable=raising-bad-type
 461             raise
 462 
 463         # Not skipping, if we had loop error raised earlier we need to raise it now to halt the execution of this task
 464         if self._loop_eval_error is not None:
 465             raise self._loop_eval_error  # pylint: disable=raising-bad-type
 466 
 467         # if we ran into an error while setting up the PlayContext, raise it now
 468         if context_validation_error is not None:
 469             raise context_validation_error  # pylint: disable=raising-bad-type
 470 
 471         # if this task is a TaskInclude, we just return now with a success code so the
 472         # main thread can expand the task list for the given host
 473         if self._task.action in ('include', 'include_tasks'):
 474             include_args = self._task.args.copy()
 475             include_file = include_args.pop('_raw_params', None)
 476             if not include_file:
 477                 return dict(failed=True, msg="No include file was specified to the include")
 478 
 479             include_file = templar.template(include_file)
 480             return dict(include=include_file, include_args=include_args)
 481 
 482         # if this task is a IncludeRole, we just return now with a success code so the main thread can expand the task list for the given host
 483         elif self._task.action == 'include_role':
 484             include_args = self._task.args.copy()
 485             return dict(include_args=include_args)
 486 
 487         # Now we do final validation on the task, which sets all fields to their final values.
 488         try:
 489             self._task.post_validate(templar=templar)
 490         except AnsibleError:
 491             raise
 492         except Exception:
 493             return dict(changed=False, failed=True, _ansible_no_log=self._play_context.no_log, exception=to_text(traceback.format_exc()))
 494         if '_variable_params' in self._task.args:
 495             variable_params = self._task.args.pop('_variable_params')
 496             if isinstance(variable_params, dict):
 497                 if C.INJECT_FACTS_AS_VARS:
 498                     display.warning("Using a variable for a task's 'args' is unsafe in some situations "
 499                                     "(see https://docs.ansible.com/ansible/devel/reference_appendices/faq.html#argsplat-unsafe)")
 500                 variable_params.update(self._task.args)
 501                 self._task.args = variable_params
 502 
 503         # get the connection and the handler for this execution
 504         if (not self._connection or
 505                 not getattr(self._connection, 'connected', False) or
 506                 self._play_context.remote_addr != self._connection._play_context.remote_addr):
 507             self._connection = self._get_connection(variables=variables, templar=templar)
 508         else:
 509             # if connection is reused, its _play_context is no longer valid and needs
 510             # to be replaced with the one templated above, in case other data changed
 511             self._connection._play_context = self._play_context
 512 
 513         if self._task.delegate_to:
 514             # use vars from delegated host (which already include task vars) instead of original host
 515             delegated_vars = variables.get('ansible_delegated_vars', {}).get(self._task.delegate_to, {})
 516             orig_vars = templar.available_variables
 517             templar.available_variables = delegated_vars
 518             plugin_vars = self._set_connection_options(delegated_vars, templar)
 519             templar.available_variables = orig_vars
 520         else:
 521             # just use normal host vars
 522             plugin_vars = self._set_connection_options(variables, templar)
 523 
 524         # get handler
 525         self._handler = self._get_action_handler(connection=self._connection, templar=templar)
 526 
 527         # Apply default params for action/module, if present
 528         self._task.args = get_action_args_with_defaults(
 529             self._task.action, self._task.args, self._task.module_defaults, templar, self._task._ansible_internal_redirect_list
 530         )
 531 
 532         # And filter out any fields which were set to default(omit), and got the omit token value
 533         omit_token = variables.get('omit')
 534         if omit_token is not None:
 535             self._task.args = remove_omit(self._task.args, omit_token)
 536 
 537         # Read some values from the task, so that we can modify them if need be
 538         if self._task.until:
 539             retries = self._task.retries
 540             if retries is None:
 541                 retries = 3
 542             elif retries <= 0:
 543                 retries = 1
 544             else:
 545                 retries += 1
 546         else:
 547             retries = 1
 548 
 549         delay = self._task.delay
 550         if delay < 0:
 551             delay = 1
 552 
 553         # make a copy of the job vars here, in case we need to update them
 554         # with the registered variable value later on when testing conditions
 555         vars_copy = variables.copy()
 556 
 557         display.debug("starting attempt loop")
 558         result = None
 559         for attempt in xrange(1, retries + 1):
 560             display.debug("running the handler")
 561             try:
 562                 if self._task.timeout:
 563                     old_sig = signal.signal(signal.SIGALRM, task_timeout)
 564                     signal.alarm(self._task.timeout)
 565                 result = self._handler.run(task_vars=variables)
 566             except AnsibleActionSkip as e:
 567                 return dict(skipped=True, msg=to_text(e))
 568             except AnsibleActionFail as e:
 569                 return dict(failed=True, msg=to_text(e))
 570             except AnsibleConnectionFailure as e:
 571                 return dict(unreachable=True, msg=to_text(e))
 572             except TaskTimeoutError as e:
 573                 msg = 'The %s action failed to execute in the expected time frame (%d) and was terminated' % (self._task.action, self._task.timeout)
 574                 return dict(failed=True, msg=msg)
 575             finally:
 576                 if self._task.timeout:
 577                     signal.alarm(0)
 578                     old_sig = signal.signal(signal.SIGALRM, old_sig)
 579                 self._handler.cleanup()
 580             display.debug("handler run complete")
 581 
 582             # preserve no log
 583             result["_ansible_no_log"] = self._play_context.no_log
 584 
 585             # update the local copy of vars with the registered value, if specified,
 586             # or any facts which may have been generated by the module execution
 587             if self._task.register:
 588                 if not isidentifier(self._task.register):
 589                     raise AnsibleError("Invalid variable name in 'register' specified: '%s'" % self._task.register)
 590 
 591                 vars_copy[self._task.register] = result = wrap_var(result)
 592 
 593             if self._task.async_val > 0:
 594                 if self._task.poll > 0 and not result.get('skipped') and not result.get('failed'):
 595                     result = self._poll_async_result(result=result, templar=templar, task_vars=vars_copy)
 596                     # FIXME callback 'v2_runner_on_async_poll' here
 597 
 598                 # ensure no log is preserved
 599                 result["_ansible_no_log"] = self._play_context.no_log
 600 
 601             # helper methods for use below in evaluating changed/failed_when
 602             def _evaluate_changed_when_result(result):
 603                 if self._task.changed_when is not None and self._task.changed_when:
 604                     cond = Conditional(loader=self._loader)
 605                     cond.when = self._task.changed_when
 606                     result['changed'] = cond.evaluate_conditional(templar, vars_copy)
 607 
 608             def _evaluate_failed_when_result(result):
 609                 if self._task.failed_when:
 610                     cond = Conditional(loader=self._loader)
 611                     cond.when = self._task.failed_when
 612                     failed_when_result = cond.evaluate_conditional(templar, vars_copy)
 613                     result['failed_when_result'] = result['failed'] = failed_when_result
 614                 else:
 615                     failed_when_result = False
 616                 return failed_when_result
 617 
 618             if 'ansible_facts' in result:
 619                 if self._task.action in ('set_fact', 'include_vars'):
 620                     vars_copy.update(result['ansible_facts'])
 621                 else:
 622                     # TODO: cleaning of facts should eventually become part of taskresults instead of vars
 623                     af = wrap_var(result['ansible_facts'])
 624                     vars_copy.update(namespace_facts(af))
 625                     if C.INJECT_FACTS_AS_VARS:
 626                         vars_copy.update(clean_facts(af))
 627 
 628             # set the failed property if it was missing.
 629             if 'failed' not in result:
 630                 # rc is here for backwards compatibility and modules that use it instead of 'failed'
 631                 if 'rc' in result and result['rc'] not in [0, "0"]:
 632                     result['failed'] = True
 633                 else:
 634                     result['failed'] = False
 635 
 636             # Make attempts and retries available early to allow their use in changed/failed_when
 637             if self._task.until:
 638                 result['attempts'] = attempt
 639 
 640             # set the changed property if it was missing.
 641             if 'changed' not in result:
 642                 result['changed'] = False
 643 
 644             # re-update the local copy of vars with the registered value, if specified,
 645             # or any facts which may have been generated by the module execution
 646             # This gives changed/failed_when access to additional recently modified
 647             # attributes of result
 648             if self._task.register:
 649                 vars_copy[self._task.register] = result = wrap_var(result)
 650 
 651             # if we didn't skip this task, use the helpers to evaluate the changed/
 652             # failed_when properties
 653             if 'skipped' not in result:
 654                 _evaluate_changed_when_result(result)
 655                 _evaluate_failed_when_result(result)
 656 
 657             if retries > 1:
 658                 cond = Conditional(loader=self._loader)
 659                 cond.when = self._task.until
 660                 if cond.evaluate_conditional(templar, vars_copy):
 661                     break
 662                 else:
 663                     # no conditional check, or it failed, so sleep for the specified time
 664                     if attempt < retries:
 665                         result['_ansible_retry'] = True
 666                         result['retries'] = retries
 667                         display.debug('Retrying task, attempt %d of %d' % (attempt, retries))
 668                         self._final_q.put(TaskResult(self._host.name, self._task._uuid, result, task_fields=self._task.dump_attrs()), block=False)
 669                         time.sleep(delay)
 670                         self._handler = self._get_action_handler(connection=self._connection, templar=templar)
 671         else:
 672             if retries > 1:
 673                 # we ran out of attempts, so mark the result as failed
 674                 result['attempts'] = retries - 1
 675                 result['failed'] = True
 676 
 677         # do the final update of the local variables here, for both registered
 678         # values and any facts which may have been created
 679         if self._task.register:
 680             variables[self._task.register] = result = wrap_var(result)
 681 
 682         if 'ansible_facts' in result:
 683             if self._task.action in ('set_fact', 'include_vars'):
 684                 variables.update(result['ansible_facts'])
 685             else:
 686                 # TODO: cleaning of facts should eventually become part of taskresults instead of vars
 687                 af = wrap_var(result['ansible_facts'])
 688                 variables.update(namespace_facts(af))
 689                 if C.INJECT_FACTS_AS_VARS:
 690                     variables.update(clean_facts(af))
 691 
 692         # save the notification target in the result, if it was specified, as
 693         # this task may be running in a loop in which case the notification
 694         # may be item-specific, ie. "notify: service {{item}}"
 695         if self._task.notify is not None:
 696             result['_ansible_notify'] = self._task.notify
 697 
 698         # add the delegated vars to the result, so we can reference them
 699         # on the results side without having to do any further templating
 700         if self._task.delegate_to:
 701             result["_ansible_delegated_vars"] = {'ansible_delegated_host': self._task.delegate_to}
 702             for k in plugin_vars:
 703                 result["_ansible_delegated_vars"][k] = delegated_vars.get(k)
 704         # and return
 705         display.debug("attempt loop complete, returning result")
 706         return result

 708     def _poll_async_result(self, result, templar, task_vars=None):
 709         '''
 710         Polls for the specified JID to be complete
 711         '''
 712 
 713         if task_vars is None:
 714             task_vars = self._job_vars
 715 
 716         async_jid = result.get('ansible_job_id')
 717         if async_jid is None:
 718             return dict(failed=True, msg="No job id was returned by the async task")
 719 
 720         # Create a new pseudo-task to run the async_status module, and run
 721         # that (with a sleep for "poll" seconds between each retry) until the
 722         # async time limit is exceeded.
 723 
 724         async_task = Task().load(dict(action='async_status jid=%s' % async_jid, environment=self._task.environment))
 725 
 726         # FIXME: this is no longer the case, normal takes care of all, see if this can just be generalized
 727         # Because this is an async task, the action handler is async. However,
 728         # we need the 'normal' action handler for the status check, so get it
 729         # now via the action_loader
 730         async_handler = self._shared_loader_obj.action_loader.get(
 731             'async_status',
 732             task=async_task,
 733             connection=self._connection,
 734             play_context=self._play_context,
 735             loader=self._loader,
 736             templar=templar,
 737             shared_loader_obj=self._shared_loader_obj,
 738         )
 739 
 740         time_left = self._task.async_val
 741         while time_left > 0:
 742             time.sleep(self._task.poll)
 743 
 744             try:
 745                 async_result = async_handler.run(task_vars=task_vars)
 746                 # We do not bail out of the loop in cases where the failure
 747                 # is associated with a parsing error. The async_runner can
 748                 # have issues which result in a half-written/unparseable result
 749                 # file on disk, which manifests to the user as a timeout happening
 750                 # before it's time to timeout.
 751                 if (int(async_result.get('finished', 0)) == 1 or
 752                         ('failed' in async_result and async_result.get('_ansible_parsed', False)) or
 753                         'skipped' in async_result):
 754                     break
 755             except Exception as e:
 756                 # Connections can raise exceptions during polling (eg, network bounce, reboot); these should be non-fatal.
 757                 # On an exception, call the connection's reset method if it has one
 758                 # (eg, drop/recreate WinRM connection; some reused connections are in a broken state)
 759                 display.vvvv("Exception during async poll, retrying... (%s)" % to_text(e))
 760                 display.debug("Async poll exception was:\n%s" % to_text(traceback.format_exc()))
 761                 try:
 762                     async_handler._connection.reset()
 763                 except AttributeError:
 764                     pass
 765 
 766                 # Little hack to raise the exception if we've exhausted the timeout period
 767                 time_left -= self._task.poll
 768                 if time_left <= 0:
 769                     raise
 770             else:
 771                 time_left -= self._task.poll
 772 
 773         if int(async_result.get('finished', 0)) != 1:
 774             if async_result.get('_ansible_parsed'):
 775                 return dict(failed=True, msg="async task did not complete within the requested time - %ss" % self._task.async_val)
 776             else:
 777                 return dict(failed=True, msg="async task produced unparseable results", async_result=async_result)
 778         else:
 779             async_handler.cleanup(force=True)
 780             return async_result

 789     def _get_connection(self, variables, templar):
 790         '''
 791         Reads the connection property for the host, and returns the
 792         correct connection object from the list of connection plugins
 793         '''
 794 
 795         if self._task.delegate_to is not None:
 796             cvars = variables.get('ansible_delegated_vars', {}).get(self._task.delegate_to, {})
 797         else:
 798             cvars = variables
 799 
 800         # use magic var if it exists, if not, let task inheritance do it's thing.
 801         self._play_context.connection = cvars.get('ansible_connection', self._task.connection)
 802 
 803         # TODO: play context has logic to update the conneciton for 'smart'
 804         # (default value, will chose between ssh and paramiko) and 'persistent'
 805         # (really paramiko), evnentually this should move to task object itself.
 806         connection_name = self._play_context.connection
 807 
 808         # load connection
 809         conn_type = connection_name
 810         connection, plugin_load_context = self._shared_loader_obj.connection_loader.get_with_context(
 811             conn_type,
 812             self._play_context,
 813             self._new_stdin,
 814             task_uuid=self._task._uuid,
 815             ansible_playbook_pid=to_text(os.getppid())
 816         )
 817 
 818         if not connection:
 819             raise AnsibleError("the connection plugin '%s' was not found" % conn_type)
 820 
 821         # load become plugin if needed
 822         if boolean(cvars.get('ansible_become', self._task.become)):
 823             become_plugin = self._get_become(cvars.get('ansible_become_method', self._task.become_method))
 824 
 825             try:
 826                 connection.set_become_plugin(become_plugin)
 827             except AttributeError:
 828                 # Older connection plugin that does not support set_become_plugin
 829                 pass
 830 
 831             if getattr(connection.become, 'require_tty', False) and not getattr(connection, 'has_tty', False):
 832                 raise AnsibleError(
 833                     "The '%s' connection does not provide a TTY which is required for the selected "
 834                     "become plugin: %s." % (conn_type, become_plugin.name)
 835                 )
 836 
 837             # Backwards compat for connection plugins that don't support become plugins
 838             # Just do this unconditionally for now, we could move it inside of the
 839             # AttributeError above later
 840             self._play_context.set_become_plugin(become_plugin.name)
 841 
 842         # Also backwards compat call for those still using play_context
 843         self._play_context.set_attributes_from_plugin(connection)
 844 
 845         if any(((connection.supports_persistence and C.USE_PERSISTENT_CONNECTIONS), connection.force_persistence)):
 846             self._play_context.timeout = connection.get_option('persistent_command_timeout')
 847             display.vvvv('attempting to start connection', host=self._play_context.remote_addr)
 848             display.vvvv('using connection plugin %s' % connection.transport, host=self._play_context.remote_addr)
 849 
 850             options = self._get_persistent_connection_options(connection, variables, templar)
 851             socket_path = start_connection(self._play_context, options, self._task._uuid)
 852             display.vvvv('local domain socket path is %s' % socket_path, host=self._play_context.remote_addr)
 853             setattr(connection, '_socket_path', socket_path)
 854 
 855         return connection

 889     def _set_connection_options(self, variables, templar):
 890 
 891         # keep list of variable names possibly consumed
 892         varnames = []
 893 
 894         # grab list of usable vars for this plugin
 895         option_vars = C.config.get_plugin_vars('connection', self._connection._load_name)
 896         varnames.extend(option_vars)
 897 
 898         # create dict of 'templated vars'
 899         options = {'_extras': {}}
 900         for k in option_vars:
 901             if k in variables:
 902                 options[k] = templar.template(variables[k])
 903 
 904         # add extras if plugin supports them
 905         if getattr(self._connection, 'allow_extras', False):
 906             for k in variables:
 907                 if k.startswith('ansible_%s_' % self._connection._load_name) and k not in options:
 908                     options['_extras'][k] = templar.template(variables[k])
 909 
 910         task_keys = self._task.dump_attrs()
 911 
 912         if self._play_context.password:
 913             # The connection password is threaded through the play_context for
 914             # now. This is something we ultimately want to avoid, but the first
 915             # step is to get connection plugins pulling the password through the
 916             # config system instead of directly accessing play_context.
 917             task_keys['password'] = self._play_context.password
 918 
 919         # set options with 'templated vars' specific to this plugin and dependent ones
 920         self._connection.set_options(task_keys=task_keys, var_options=options)
 921         varnames.extend(self._set_plugin_options('shell', variables, templar, task_keys))
 922 
 923         if self._connection.become is not None:
 924             if self._play_context.become_pass:
 925                 # FIXME: eventually remove from task and play_context, here for backwards compat
 926                 # keep out of play objects to avoid accidental disclosure, only become plugin should have
 927                 # The become pass is already in the play_context if given on
 928                 # the CLI (-K). Make the plugin aware of it in this case.
 929                 task_keys['become_pass'] = self._play_context.become_pass
 930 
 931             varnames.extend(self._set_plugin_options('become', variables, templar, task_keys))
 932 
 933             # FOR BACKWARDS COMPAT:
 934             for option in ('become_user', 'become_flags', 'become_exe', 'become_pass'):
 935                 try:
 936                     setattr(self._play_context, option, self._connection.become.get_option(option))
 937                 except KeyError:
 938                     pass  # some plugins don't support all base flags
 939             self._play_context.prompt = self._connection.become.prompt
 940 
 941         return varnames

 943     def _get_action_handler(self, connection, templar):
 944         '''
 945         Returns the correct action plugin to handle the requestion task action
 946         '''
 947 
 948         module_collection, separator, module_name = self._task.action.rpartition(".")
 949         module_prefix = module_name.split('_')[0]
 950         if module_collection:
 951             # For network modules, which look for one action plugin per platform, look for the
 952             # action plugin in the same collection as the module by prefixing the action plugin
 953             # with the same collection.
 954             network_action = "{0}.{1}".format(module_collection, module_prefix)
 955         else:
 956             network_action = module_prefix
 957 
 958         collections = self._task.collections
 959 
 960         # let action plugin override module, fallback to 'normal' action plugin otherwise
 961         if self._shared_loader_obj.action_loader.has_plugin(self._task.action, collection_list=collections):
 962             handler_name = self._task.action
 963         elif all((module_prefix in C.NETWORK_GROUP_MODULES, self._shared_loader_obj.action_loader.has_plugin(network_action, collection_list=collections))):
 964             handler_name = network_action
 965         else:
 966             # FUTURE: once we're comfortable with collections impl, preface this action with ansible.builtin so it can't be hijacked
 967             handler_name = 'normal'
 968             collections = None  # until then, we don't want the task's collection list to be consulted; use the builtin
 969 
 970         handler = self._shared_loader_obj.action_loader.get(
 971             handler_name,
 972             task=self._task,
 973             connection=connection,
 974             play_context=self._play_context,
 975             loader=self._loader,
 976             templar=templar,
 977             shared_loader_obj=self._shared_loader_obj,
 978             collection_list=collections
 979         )
 980 
 981         if not handler:
 982             raise AnsibleError("the handler '%s' was not found" % handler_name)
 983 
 984         return handler

 987 def start_connection(play_context, variables, task_uuid):
 988     '''
 989     Starts the persistent connection
 990     '''
 991     candidate_paths = [C.ANSIBLE_CONNECTION_PATH or os.path.dirname(sys.argv[0])]
 992     candidate_paths.extend(os.environ['PATH'].split(os.pathsep))
 993     for dirname in candidate_paths:
 994         ansible_connection = os.path.join(dirname, 'ansible-connection')
 995         if os.path.isfile(ansible_connection):
 996             display.vvvv("Found ansible-connection at path {0}".format(ansible_connection))
 997             break
 998     else:
 999         raise AnsibleError("Unable to find location of 'ansible-connection'. "
1000                            "Please set or check the value of ANSIBLE_CONNECTION_PATH")
1001 
1002     env = os.environ.copy()
1003     env.update({
1004         # HACK; most of these paths may change during the controller's lifetime
1005         # (eg, due to late dynamic role includes, multi-playbook execution), without a way
1006         # to invalidate/update, ansible-connection won't always see the same plugins the controller
1007         # can.
1008         'ANSIBLE_BECOME_PLUGINS': become_loader.print_paths(),
1009         'ANSIBLE_CLICONF_PLUGINS': cliconf_loader.print_paths(),
1010         'ANSIBLE_COLLECTIONS_PATH': to_native(os.pathsep.join(AnsibleCollectionConfig.collection_paths)),
1011         'ANSIBLE_CONNECTION_PLUGINS': connection_loader.print_paths(),
1012         'ANSIBLE_HTTPAPI_PLUGINS': httpapi_loader.print_paths(),
1013         'ANSIBLE_NETCONF_PLUGINS': netconf_loader.print_paths(),
1014         'ANSIBLE_TERMINAL_PLUGINS': terminal_loader.print_paths(),
1015     })
1016     python = sys.executable
1017     master, slave = pty.openpty()
1018     p = subprocess.Popen(
1019         [python, ansible_connection, to_text(os.getppid()), to_text(task_uuid)],
1020         stdin=slave, stdout=subprocess.PIPE, stderr=subprocess.PIPE, env=env
1021     )
1022     os.close(slave)
1023 
1024     # We need to set the pty into noncanonical mode. This ensures that we
1025     # can receive lines longer than 4095 characters (plus newline) without
1026     # truncating.
1027     old = termios.tcgetattr(master)
1028     new = termios.tcgetattr(master)
1029     new[3] = new[3] & ~termios.ICANON
1030 
1031     try:
1032         termios.tcsetattr(master, termios.TCSANOW, new)
1033         write_to_file_descriptor(master, variables)
1034         write_to_file_descriptor(master, play_context.serialize())
1035 
1036         (stdout, stderr) = p.communicate()
1037     finally:
1038         termios.tcsetattr(master, termios.TCSANOW, old)
1039     os.close(master)
1040 
1041     if p.returncode == 0:
1042         result = json.loads(to_text(stdout, errors='surrogate_then_replace'))
1043     else:
1044         try:
1045             result = json.loads(to_text(stderr, errors='surrogate_then_replace'))
1046         except getattr(json.decoder, 'JSONDecodeError', ValueError):
1047             # JSONDecodeError only available on Python 3.5+
1048             result = {'error': to_text(stderr, errors='surrogate_then_replace')}
1049 
1050     if 'messages' in result:
1051         for level, message in result['messages']:
1052             if level == 'log':
1053                 display.display(message, log_only=True)
1054             elif level in ('debug', 'v', 'vv', 'vvv', 'vvvv', 'vvvvv', 'vvvvvv'):
1055                 getattr(display, level)(message, host=play_context.remote_addr)
1056             else:
1057                 if hasattr(display, level):
1058                     getattr(display, level)(message)
1059                 else:
1060                     display.vvvv(message, host=play_context.remote_addr)
1061 
1062     if 'error' in result:
1063         if play_context.verbosity > 2:
1064             if result.get('exception'):
1065                 msg = "The full traceback is:\n" + result['exception']
1066                 display.display(msg, color=C.COLOR_ERROR)
1067         raise AnsibleError(result['error'])
1068 
1069     return result['socket_path']
