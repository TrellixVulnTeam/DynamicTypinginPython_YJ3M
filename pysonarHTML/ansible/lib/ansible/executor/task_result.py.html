  12 _IGNORE = ('failed', 'skipped')
  13 _PRESERVE = ('attempts', 'changed', 'retries')
  14 _SUB_PRESERVE = {'_ansible_delegated_vars': ('ansible_host', 'ansible_port', 'ansible_user', 'ansible_connection')}
  15 
  16 # stuff callbacks need
  17 CLEAN_EXCEPTIONS = (
  18     '_ansible_verbose_always',  # for debug and other actions, to always expand data (pretty jsonification)
  19     '_ansible_item_label',  # to know actual 'item' variable
  20     '_ansible_no_log',  # jic we didnt clean up well enough, DON'T LOG
  21     '_ansible_verbose_override',  # controls display of ansible_facts, gathering would be very noise with -v otherwise
  22 )

  25 class TaskResult:
  26     '''
  27     This class is responsible for interpreting the resulting data
  28     from an executed task, and provides helper methods for determining
  29     the result of a given task.
  30     '''
  31 
  32     def __init__(self, host, task, return_data, task_fields=None):
  33         self._host = host
  34         self._task = task
  35 
  36         if isinstance(return_data, dict):
  37             self._result = return_data.copy()
  38         else:
  39             self._result = DataLoader().load(return_data)
  40 
  41         if task_fields is None:
  42             self._task_fields = dict()
  43         else:
  44             self._task_fields = task_fields
  45 
  46     @property
  47     def task_name(self):
  48         return self._task_fields.get('name', None) or self._task.get_name()
  49 
  50     def is_changed(self):
  51         return self._check_key('changed')
  52 
  53     def is_skipped(self):
  54         # loop results
  55         if 'results' in self._result:
  56             results = self._result['results']
  57             # Loop tasks are only considered skipped if all items were skipped.
  58             # some squashed results (eg, yum) are not dicts and can't be skipped individually
  59             if results and all(isinstance(res, dict) and res.get('skipped', False) for res in results):
  60                 return True
  61 
  62         # regular tasks and squashed non-dict results
  63         return self._result.get('skipped', False)
  64 
  65     def is_failed(self):
  66         if 'failed_when_result' in self._result or \
  67            'results' in self._result and True in [True for x in self._result['results'] if 'failed_when_result' in x]:
  68             return self._check_key('failed_when_result')
  69         else:
  70             return self._check_key('failed')
  71 
  72     def is_unreachable(self):
  73         return self._check_key('unreachable')
  74 
  75     def needs_debugger(self, globally_enabled=False):
  76         _debugger = self._task_fields.get('debugger')
  77         _ignore_errors = C.TASK_DEBUGGER_IGNORE_ERRORS and self._task_fields.get('ignore_errors')
  78 
  79         ret = False
  80         if globally_enabled and ((self.is_failed() and not _ignore_errors) or self.is_unreachable()):
  81             ret = True
  82 
  83         if _debugger in ('always',):
  84             ret = True
  85         elif _debugger in ('never',):
  86             ret = False
  87         elif _debugger in ('on_failed',) and self.is_failed() and not _ignore_errors:
  88             ret = True
  89         elif _debugger in ('on_unreachable',) and self.is_unreachable():
  90             ret = True
  91         elif _debugger in ('on_skipped',) and self.is_skipped():
  92             ret = True
  93 
  94         return ret
  95 
  96     def _check_key(self, key):
  97         '''get a specific key from the result or its items'''
  98 
  99         if isinstance(self._result, dict) and key in self._result:
 100             return self._result.get(key, False)
 101         else:
 102             flag = False
 103             for res in self._result.get('results', []):
 104                 if isinstance(res, dict):
 105                     flag |= res.get(key, False)
 106             return flag
 107 
 108     def clean_copy(self):
 109 
 110         ''' returns 'clean' taskresult object '''
 111 
 112         # FIXME: clean task_fields, _task and _host copies
 113         result = TaskResult(self._host, self._task, {}, self._task_fields)
 114 
 115         # statuses are already reflected on the event type
 116         if result._task and result._task.action in ['debug']:
 117             # debug is verbose by default to display vars, no need to add invocation
 118             ignore = _IGNORE + ('invocation',)
 119         else:
 120             ignore = _IGNORE
 121 
 122         subset = {}
 123         # preserve subset for later
 124         for sub in _SUB_PRESERVE:
 125             if sub in self._result:
 126                 subset[sub] = {}
 127                 for key in _SUB_PRESERVE[sub]:
 128                     if key in self._result[sub]:
 129                         subset[sub][key] = self._result[sub][key]
 130 
 131         if isinstance(self._task.no_log, bool) and self._task.no_log or self._result.get('_ansible_no_log', False):
 132             x = {"censored": "the output has been hidden due to the fact that 'no_log: true' was specified for this result"}
 133 
 134             # preserve full
 135             for preserve in _PRESERVE:
 136                 if preserve in self._result:
 137                     x[preserve] = self._result[preserve]
 138 
 139             result._result = x
 140         elif self._result:
 141             result._result = module_response_deepcopy(self._result)
 142 
 143             # actualy remove
 144             for remove_key in ignore:
 145                 if remove_key in result._result:
 146                     del result._result[remove_key]
 147 
 148             # remove almost ALL internal keys, keep ones relevant to callback
 149             strip_internal_keys(result._result, exceptions=CLEAN_EXCEPTIONS)
 150 
 151         # keep subset
 152         result._result.update(subset)
 153 
 154         return result
