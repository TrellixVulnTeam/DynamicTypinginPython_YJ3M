  16 def read_utf8_file(path, encoding='utf-8'):
  17     if not os.access(path, os.R_OK):
  18         return None
  19     with io.open(path, 'r', encoding=encoding) as fd:
  20         content = fd.read()
  21 
  22     return content
  23 
  24 
  25 def get_platform_info():
  26     result = dict(platform_dist_result=[])
  27 
  28     if hasattr(platform, 'dist'):
  29         result['platform_dist_result'] = platform.dist()
  30 
  31     osrelease_content = read_utf8_file('/etc/os-release')
  32     # try to fall back to /usr/lib/os-release
  33     if not osrelease_content:
  34         osrelease_content = read_utf8_file('/usr/lib/os-release')
  35 
  36     result['osrelease_content'] = osrelease_content
  37 
  38     return result
  39

  41 def main():
  42     info = get_platform_info()
  43 
  44     print(json.dumps(info))
  45 
  46 
  47 if __name__ == '__main__':
  48     main()
