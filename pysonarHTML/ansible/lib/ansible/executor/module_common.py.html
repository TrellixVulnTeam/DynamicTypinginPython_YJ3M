 442 class ModuleDepFinder(ast.NodeVisitor):
 443 
 444     def __init__(self, module_fqn, *args, **kwargs):
 445         """
 446         Walk the ast tree for the python module.
 447         :arg module_fqn: The fully qualified name to reach this module in dotted notation.
 448             example: ansible.module_utils.basic
 449 
 450         Save submodule[.submoduleN][.identifier] into self.submodules
 451         when they are from ansible.module_utils or ansible_collections packages
 452 
 453         self.submodules will end up with tuples like:
 454           - ('ansible', 'module_utils', 'basic',)
 455           - ('ansible', 'module_utils', 'urls', 'fetch_url')
 456           - ('ansible', 'module_utils', 'database', 'postgres')
 457           - ('ansible', 'module_utils', 'database', 'postgres', 'quote')
 458           - ('ansible', 'module_utils', 'database', 'postgres', 'quote')
 459           - ('ansible_collections', 'my_ns', 'my_col', 'plugins', 'module_utils', 'foo')
 460 
 461         It's up to calling code to determine whether the final element of the
 462         tuple are module names or something else (function, class, or variable names)
 463         .. seealso:: :python3:class:`ast.NodeVisitor`
 464         """
 465         super(ModuleDepFinder, self).__init__(*args, **kwargs)
 466         self.submodules = set()
 467         self.module_fqn = module_fqn
 468 
 469         self._visit_map = {
 470             Import: self.visit_Import,
 471             ImportFrom: self.visit_ImportFrom,
 472         }
 473 
 474     def generic_visit(self, node):
 475         """Overridden ``generic_visit`` that makes some assumptions about our
 476         use case, and improves performance by calling visitors directly instead
 477         of calling ``visit`` to offload calling visitors.
 478         """
 479         visit_map = self._visit_map
 480         generic_visit = self.generic_visit
 481         for field, value in ast.iter_fields(node):
 482             if isinstance(value, list):
 483                 for item in value:
 484                     if isinstance(item, (Import, ImportFrom)):
 485                         visit_map[item.__class__](item)
 486                     elif isinstance(item, AST):
 487                         generic_visit(item)
 488 
 489     visit = generic_visit
 490 
 491     def visit_Import(self, node):
 492         """
 493         Handle import ansible.module_utils.MODLIB[.MODLIBn] [as asname]
 494 
 495         We save these as interesting submodules when the imported library is in ansible.module_utils
 496         or ansible.collections
 497         """
 498         for alias in node.names:
 499             if (alias.name.startswith('ansible.module_utils.') or
 500                     alias.name.startswith('ansible_collections.')):
 501                 py_mod = tuple(alias.name.split('.'))
 502                 self.submodules.add(py_mod)
 503         self.generic_visit(node)
 504 
 505     def visit_ImportFrom(self, node):
 506         """
 507         Handle from ansible.module_utils.MODLIB import [.MODLIBn] [as asname]
 508 
 509         Also has to handle relative imports
 510 
 511         We save these as interesting submodules when the imported library is in ansible.module_utils
 512         or ansible.collections
 513         """
 514 
 515         # FIXME: These should all get skipped:
 516         # from ansible.executor import module_common
 517         # from ...executor import module_common
 518         # from ... import executor (Currently it gives a non-helpful error)
 519         if node.level > 0:
 520             if self.module_fqn:
 521                 parts = tuple(self.module_fqn.split('.'))
 522                 if node.module:
 523                     # relative import: from .module import x
 524                     node_module = '.'.join(parts[:-node.level] + (node.module,))
 525                 else:
 526                     # relative import: from . import x
 527                     node_module = '.'.join(parts[:-node.level])
 528             else:
 529                 # fall back to an absolute import
 530                 node_module = node.module
 531         else:
 532             # absolute import: from module import x
 533             node_module = node.module
 534 
 535         # Specialcase: six is a special case because of its
 536         # import logic
 537         py_mod = None
 538         if node.names[0].name == '_six':
 539             self.submodules.add(('_six',))
 540         elif node_module.startswith('ansible.module_utils'):
 541             # from ansible.module_utils.MODULE1[.MODULEn] import IDENTIFIER [as asname]
 542             # from ansible.module_utils.MODULE1[.MODULEn] import MODULEn+1 [as asname]
 543             # from ansible.module_utils.MODULE1[.MODULEn] import MODULEn+1 [,IDENTIFIER] [as asname]
 544             # from ansible.module_utils import MODULE1 [,MODULEn] [as asname]
 545             py_mod = tuple(node_module.split('.'))
 546 
 547         elif node_module.startswith('ansible_collections.'):
 548             if node_module.endswith('plugins.module_utils') or '.plugins.module_utils.' in node_module:
 549                 # from ansible_collections.ns.coll.plugins.module_utils import MODULE [as aname] [,MODULE2] [as aname]
 550                 # from ansible_collections.ns.coll.plugins.module_utils.MODULE import IDENTIFIER [as aname]
 551                 # FIXME: Unhandled cornercase (needs to be ignored):
 552                 # from ansible_collections.ns.coll.plugins.[!module_utils].[FOO].plugins.module_utils import IDENTIFIER
 553                 py_mod = tuple(node_module.split('.'))
 554             else:
 555                 # Not from module_utils so ignore.  for instance:
 556                 # from ansible_collections.ns.coll.plugins.lookup import IDENTIFIER
 557                 pass
 558 
 559         if py_mod:
 560             for alias in node.names:
 561                 self.submodules.add(py_mod + (alias.name,))
 562 
 563         self.generic_visit(node)

 574 def _get_shebang(interpreter, task_vars, templar, args=tuple()):
 575     """
 576     Note not stellar API:
 577        Returns None instead of always returning a shebang line.  Doing it this
 578        way allows the caller to decide to use the shebang it read from the
 579        file rather than trust that we reformatted what they already have
 580        correctly.
 581     """
 582     interpreter_name = os.path.basename(interpreter).strip()
 583 
 584     # FUTURE: add logical equivalence for python3 in the case of py3-only modules
 585 
 586     # check for first-class interpreter config
 587     interpreter_config_key = "INTERPRETER_%s" % interpreter_name.upper()
 588 
 589     if C.config.get_configuration_definitions().get(interpreter_config_key):
 590         # a config def exists for this interpreter type; consult config for the value
 591         interpreter_out = C.config.get_config_value(interpreter_config_key, variables=task_vars)
 592         discovered_interpreter_config = u'discovered_interpreter_%s' % interpreter_name
 593 
 594         interpreter_out = templar.template(interpreter_out.strip())
 595 
 596         facts_from_task_vars = task_vars.get('ansible_facts', {})
 597 
 598         # handle interpreter discovery if requested
 599         if interpreter_out in ['auto', 'auto_legacy', 'auto_silent', 'auto_legacy_silent']:
 600             if discovered_interpreter_config not in facts_from_task_vars:
 601                 # interpreter discovery is desired, but has not been run for this host
 602                 raise InterpreterDiscoveryRequiredError("interpreter discovery needed",
 603                                                         interpreter_name=interpreter_name,
 604                                                         discovery_mode=interpreter_out)
 605             else:
 606                 interpreter_out = facts_from_task_vars[discovered_interpreter_config]
 607     else:
 608         # a config def does not exist for this interpreter type; consult vars for a possible direct override
 609         interpreter_config = u'ansible_%s_interpreter' % interpreter_name
 610 
 611         if interpreter_config not in task_vars:
 612             return None, interpreter
 613 
 614         interpreter_out = templar.template(task_vars[interpreter_config].strip())
 615 
 616     shebang = u'#!' + interpreter_out
 617 
 618     if args:
 619         shebang = shebang + u' ' + u' '.join(args)
 620 
 621     return shebang, interpreter_out

 624 class ModuleInfo:
 625     def __init__(self, name, paths):
 626         self.py_src = False
 627         self.pkg_dir = False
 628         path = None
 629 
 630         if imp is None:
 631             # don't pretend this is a top-level module, prefix the rest of the namespace
 632             self._info = info = importlib.machinery.PathFinder.find_spec('ansible.module_utils.' + name, paths)
 633             if info is not None:
 634                 self.py_src = os.path.splitext(info.origin)[1] in importlib.machinery.SOURCE_SUFFIXES
 635                 self.pkg_dir = info.origin.endswith('/__init__.py')
 636                 path = info.origin
 637             else:
 638                 raise ImportError("No module named '%s'" % name)
 639         else:
 640             self._info = info = imp.find_module(name, paths)
 641             self.py_src = info[2][2] == imp.PY_SOURCE
 642             self.pkg_dir = info[2][2] == imp.PKG_DIRECTORY
 643             if self.pkg_dir:
 644                 path = os.path.join(info[1], '__init__.py')
 645             else:
 646                 path = info[1]
 647 
 648         self.path = path
 649 
 650     def get_source(self):
 651         if imp and self.py_src:
 652             try:
 653                 return self._info[0].read()
 654             finally:
 655                 self._info[0].close()
 656         return _slurp(self.path)
 657 
 658     def __repr__(self):
 659         return 'ModuleInfo: py_src=%s, pkg_dir=%s, path=%s' % (self.py_src, self.pkg_dir, self.path)

 720 def recursive_finder(name, module_fqn, data, py_module_names, py_module_cache, zf):
 721     """
 722     Using ModuleDepFinder, make sure we have all of the module_utils files that
 723     the module and its module_utils files needs.
 724     :arg name: Name of the python module we're examining
 725     :arg module_fqn: Fully qualified name of the python module we're scanning
 726     :arg py_module_names: set of the fully qualified module names represented as a tuple of their
 727         FQN with __init__ appended if the module is also a python package).  Presence of a FQN in
 728         this set means that we've already examined it for module_util deps.
 729     :arg py_module_cache: map python module names (represented as a tuple of their FQN with __init__
 730         appended if the module is also a python package) to a tuple of the code in the module and
 731         the pathname the module would have inside of a Python toplevel (like site-packages)
 732     :arg zf: An open :python:class:`zipfile.ZipFile` object that holds the Ansible module payload
 733         which we're assembling
 734     """
 735     # Parse the module and find the imports of ansible.module_utils
 736     try:
 737         tree = compile(data, '<unknown>', 'exec', ast.PyCF_ONLY_AST)
 738     except (SyntaxError, IndentationError) as e:
 739         raise AnsibleError("Unable to import %s due to %s" % (name, e.msg))
 740 
 741     finder = ModuleDepFinder(module_fqn)
 742     finder.visit(tree)
 743 
 744     #
 745     # Determine what imports that we've found are modules (vs class, function.
 746     # variable names) for packages
 747     #
 748     module_utils_paths = [p for p in module_utils_loader._get_paths(subdirs=False) if os.path.isdir(p)]
 749     # FIXME: Do we still need this?  It feels like module-utils_loader should include
 750     # _MODULE_UTILS_PATH
 751     module_utils_paths.append(_MODULE_UTILS_PATH)
 752 
 753     normalized_modules = set()
 754     # Loop through the imports that we've found to normalize them
 755     # Exclude paths that match with paths we've already processed
 756     # (Have to exclude them a second time once the paths are processed)
 757 
 758     for py_module_name in finder.submodules.difference(py_module_names):
 759         module_info = None
 760 
 761         if py_module_name[0:3] == ('ansible', 'module_utils', 'six'):
 762             # Special case the python six library because it messes with the
 763             # import process in an incompatible way
 764             module_info = ModuleInfo('six', module_utils_paths)
 765             py_module_name = ('ansible', 'module_utils', 'six')
 766             idx = 0
 767         elif py_module_name[0:3] == ('ansible', 'module_utils', '_six'):
 768             # Special case the python six library because it messes with the
 769             # import process in an incompatible way
 770             module_info = ModuleInfo('_six', [os.path.join(p, 'six') for p in module_utils_paths])
 771             py_module_name = ('ansible', 'module_utils', 'six', '_six')
 772             idx = 0
 773         elif py_module_name[0] == 'ansible_collections':
 774             # FIXME (nitz): replicate module name resolution like below for granular imports
 775             for idx in (1, 2):
 776                 if len(py_module_name) < idx:
 777                     break
 778                 try:
 779                     # this is a collection-hosted MU; look it up with pkgutil.get_data()
 780                     module_info = CollectionModuleInfo(py_module_name[-idx], '.'.join(py_module_name[:-idx]))
 781                     break
 782                 except ImportError:
 783                     continue
 784         elif py_module_name[0:2] == ('ansible', 'module_utils'):
 785             # Need to remove ansible.module_utils because PluginLoader may find different paths
 786             # for us to look in
 787             relative_module_utils_dir = py_module_name[2:]
 788             # Check whether either the last or the second to last identifier is
 789             # a module name
 790             for idx in (1, 2):
 791                 if len(relative_module_utils_dir) < idx:
 792                     break
 793                 try:
 794                     module_info = ModuleInfo(py_module_name[-idx],
 795                                              [os.path.join(p, *relative_module_utils_dir[:-idx]) for p in module_utils_paths])
 796                     break
 797                 except ImportError:
 798                     # check metadata for redirect, generate stub if present
 799                     try:
 800                         module_info = InternalRedirectModuleInfo(py_module_name[-idx],
 801                                                                  '.'.join(py_module_name[:(None if idx == 1 else -1)]))
 802                         break
 803                     except ImportError:
 804                         continue
 805         else:
 806             # If we get here, it's because of a bug in ModuleDepFinder.  If we get a reproducer we
 807             # should then fix ModuleDepFinder
 808             display.warning('ModuleDepFinder improperly found a non-module_utils import %s'
 809                             % [py_module_name])
 810             continue
 811 
 812         # Could not find the module.  Construct a helpful error message.
 813         if module_info is None:
 814             msg = ['Could not find imported module support code for %s.  Looked for' % (name,)]
 815             if idx == 2:
 816                 msg.append('either %s.py or %s.py' % (py_module_name[-1], py_module_name[-2]))
 817             else:
 818                 msg.append(py_module_name[-1])
 819             raise AnsibleError(' '.join(msg))
 820 
 821         if isinstance(module_info, CollectionModuleInfo):
 822             if idx == 2:
 823                 # We've determined that the last portion was an identifier and
 824                 # thus, not part of the module name
 825                 py_module_name = py_module_name[:-1]
 826 
 827             # HACK: maybe surface collection dirs in here and use existing find_module code?
 828             normalized_name = py_module_name
 829             normalized_data = module_info.get_source()
 830             normalized_path = os.path.join(*py_module_name)
 831             py_module_cache[normalized_name] = (normalized_data, normalized_path)
 832             normalized_modules.add(normalized_name)
 833 
 834             # HACK: walk back up the package hierarchy to pick up package inits; this won't do the right thing
 835             # for actual packages yet...
 836             accumulated_pkg_name = []
 837             for pkg in py_module_name[:-1]:
 838                 accumulated_pkg_name.append(pkg)  # we're accumulating this across iterations
 839                 normalized_name = tuple(accumulated_pkg_name[:] + ['__init__'])  # extra machinations to get a hashable type (list is not)
 840                 if normalized_name not in py_module_cache:
 841                     normalized_path = os.path.join(*accumulated_pkg_name)
 842                     # HACK: possibly preserve some of the actual package file contents; problematic for extend_paths and others though?
 843                     normalized_data = ''
 844                     py_module_cache[normalized_name] = (normalized_data, normalized_path)
 845                     normalized_modules.add(normalized_name)
 846 
 847         else:
 848             # Found a byte compiled file rather than source.  We cannot send byte
 849             # compiled over the wire as the python version might be different.
 850             # imp.find_module seems to prefer to return source packages so we just
 851             # error out if imp.find_module returns byte compiled files (This is
 852             # fragile as it depends on undocumented imp.find_module behaviour)
 853             if not module_info.pkg_dir and not module_info.py_src:
 854                 msg = ['Could not find python source for imported module support code for %s.  Looked for' % name]
 855                 if idx == 2:
 856                     msg.append('either %s.py or %s.py' % (py_module_name[-1], py_module_name[-2]))
 857                 else:
 858                     msg.append(py_module_name[-1])
 859                 raise AnsibleError(' '.join(msg))
 860 
 861             if idx == 2:
 862                 # We've determined that the last portion was an identifier and
 863                 # thus, not part of the module name
 864                 py_module_name = py_module_name[:-1]
 865 
 866             # If not already processed then we've got work to do
 867             # If not in the cache, then read the file into the cache
 868             # We already have a file handle for the module open so it makes
 869             # sense to read it now
 870             if py_module_name not in py_module_cache:
 871                 if module_info.pkg_dir:
 872                     # Read the __init__.py instead of the module file as this is
 873                     # a python package
 874                     normalized_name = py_module_name + ('__init__',)
 875                     if normalized_name not in py_module_names:
 876                         normalized_data = module_info.get_source()
 877                         py_module_cache[normalized_name] = (normalized_data, module_info.path)
 878                         normalized_modules.add(normalized_name)
 879                 else:
 880                     normalized_name = py_module_name
 881                     if normalized_name not in py_module_names:
 882                         normalized_data = module_info.get_source()
 883                         py_module_cache[normalized_name] = (normalized_data, module_info.path)
 884                         normalized_modules.add(normalized_name)
 885 
 886                 #
 887                 # Make sure that all the packages that this module is a part of
 888                 # are also added
 889                 #
 890                 for i in range(1, len(py_module_name)):
 891                     py_pkg_name = py_module_name[:-i] + ('__init__',)
 892                     if py_pkg_name not in py_module_names:
 893                         # Need to remove ansible.module_utils because PluginLoader may find
 894                         # different paths for us to look in
 895                         relative_module_utils = py_pkg_name[2:]
 896                         pkg_dir_info = ModuleInfo(relative_module_utils[-1],
 897                                                   [os.path.join(p, *relative_module_utils[:-1]) for p in module_utils_paths])
 898                         normalized_modules.add(py_pkg_name)
 899                         py_module_cache[py_pkg_name] = (pkg_dir_info.get_source(), pkg_dir_info.path)
 900 
 901     # FIXME: Currently the AnsiBallZ wrapper monkeypatches module args into a global
 902     # variable in basic.py.  If a module doesn't import basic.py, then the AnsiBallZ wrapper will
 903     # traceback when it tries to monkypatch.  So, for now, we have to unconditionally include
 904     # basic.py.
 905     #
 906     # In the future we need to change the wrapper to monkeypatch the args into a global variable in
 907     # their own, separate python module.  That way we won't require basic.py.  Modules which don't
 908     # want basic.py can import that instead.  AnsibleModule will need to change to import the vars
 909     # from the separate python module and mirror the args into its global variable for backwards
 910     # compatibility.
 911     if ('ansible', 'module_utils', 'basic',) not in py_module_names:
 912         pkg_dir_info = ModuleInfo('basic', module_utils_paths)
 913         normalized_modules.add(('ansible', 'module_utils', 'basic',))
 914         py_module_cache[('ansible', 'module_utils', 'basic',)] = (pkg_dir_info.get_source(), pkg_dir_info.path)
 915     # End of AnsiballZ hack
 916 
 917     #
 918     # iterate through all of the ansible.module_utils* imports that we haven't
 919     # already checked for new imports
 920     #
 921 
 922     # set of modules that we haven't added to the zipfile
 923     unprocessed_py_module_names = normalized_modules.difference(py_module_names)
 924 
 925     for py_module_name in unprocessed_py_module_names:
 926 
 927         py_module_path = os.path.join(*py_module_name)
 928         py_module_file_name = '%s.py' % py_module_path
 929 
 930         zf.writestr(py_module_file_name, py_module_cache[py_module_name][0])
 931         mu_file = to_text(py_module_cache[py_module_name][1], errors='surrogate_or_strict')
 932         display.vvvvv("Using module_utils file %s" % mu_file)
 933 
 934     # Add the names of the files we're scheduling to examine in the loop to
 935     # py_module_names so that we don't re-examine them in the next pass
 936     # through recursive_finder()
 937     py_module_names.update(unprocessed_py_module_names)
 938 
 939     for py_module_file in unprocessed_py_module_names:
 940         next_fqn = '.'.join(py_module_file)
 941         recursive_finder(py_module_file[-1], next_fqn, py_module_cache[py_module_file][0],
 942                          py_module_names, py_module_cache, zf)
 943         # Save memory; the file won't have to be read again for this ansible module.
 944         del py_module_cache[py_module_file]

 947 def _is_binary(b_module_data):
 948     textchars = bytearray(set([7, 8, 9, 10, 12, 13, 27]) | set(range(0x20, 0x100)) - set([0x7f]))
 949     start = b_module_data[:1024]
 950     return bool(start.translate(None, textchars))

 953 def _get_ansible_module_fqn(module_path):
 954     """
 955     Get the fully qualified name for an ansible module based on its pathname
 956 
 957     remote_module_fqn is the fully qualified name.  Like ansible.modules.system.ping
 958     Or ansible_collections.Namespace.Collection_name.plugins.modules.ping
 959     .. warning:: This function is for ansible modules only.  It won't work for other things
 960         (non-module plugins, etc)
 961     """
 962     remote_module_fqn = None
 963 
 964     # Is this a core module?
 965     match = CORE_LIBRARY_PATH_RE.search(module_path)
 966     if not match:
 967         # Is this a module in a collection?
 968         match = COLLECTION_PATH_RE.search(module_path)
 969 
 970     # We can tell the FQN for core modules and collection modules
 971     if match:
 972         path = match.group('path')
 973         if '.' in path:
 974             # FQNs must be valid as python identifiers.  This sanity check has failed.
 975             # we could check other things as well
 976             raise ValueError('Module name (or path) was not a valid python identifier')
 977 
 978         remote_module_fqn = '.'.join(path.split('/'))
 979     else:
 980         # Currently we do not handle modules in roles so we can end up here for that reason
 981         raise ValueError("Unable to determine module's fully qualified name")
 982 
 983     return remote_module_fqn

1014 def _find_module_utils(module_name, b_module_data, module_path, module_args, task_vars, templar, module_compression, async_timeout, become,
1015                        become_method, become_user, become_password, become_flags, environment):
1016     """
1017     Given the source of the module, convert it to a Jinja2 template to insert
1018     module code and return whether it's a new or old style module.
1019     """
1020     module_substyle = module_style = 'old'
1021 
1022     # module_style is something important to calling code (ActionBase).  It
1023     # determines how arguments are formatted (json vs k=v) and whether
1024     # a separate arguments file needs to be sent over the wire.
1025     # module_substyle is extra information that's useful internally.  It tells
1026     # us what we have to look to substitute in the module files and whether
1027     # we're using module replacer or ansiballz to format the module itself.
1028     if _is_binary(b_module_data):
1029         module_substyle = module_style = 'binary'
1030     elif REPLACER in b_module_data:
1031         # Do REPLACER before from ansible.module_utils because we need make sure
1032         # we substitute "from ansible.module_utils basic" for REPLACER
1033         module_style = 'new'
1034         module_substyle = 'python'
1035         b_module_data = b_module_data.replace(REPLACER, b'from ansible.module_utils.basic import *')
1036     elif NEW_STYLE_PYTHON_MODULE_RE.search(b_module_data):
1037         module_style = 'new'
1038         module_substyle = 'python'
1039     elif REPLACER_WINDOWS in b_module_data:
1040         module_style = 'new'
1041         module_substyle = 'powershell'
1042         b_module_data = b_module_data.replace(REPLACER_WINDOWS, b'#Requires -Module Ansible.ModuleUtils.Legacy')

1281 def modify_module(module_name, module_path, module_args, templar, task_vars=None, module_compression='ZIP_STORED', async_timeout=0, become=False,
1282                   become_method=None, become_user=None, become_password=None, become_flags=None, environment=None):
1283     """
1284     Used to insert chunks of code into modules before transfer rather than
1285     doing regular python imports.  This allows for more efficient transfer in
1286     a non-bootstrapping scenario by not moving extra files over the wire and
1287     also takes care of embedding arguments in the transferred modules.
1288 
1289     This version is done in such a way that local imports can still be
1290     used in the module code, so IDEs don't have to be aware of what is going on.
1291 
1292     Example:
1293 
1294     from ansible.module_utils.basic import *
1295 
1296        ... will result in the insertion of basic.py into the module
1297        from the module_utils/ directory in the source tree.
1298 
1299     For powershell, this code effectively no-ops, as the exec wrapper requires access to a number of
1300     properties not available here.
1301 
1302     """
1303     task_vars = {} if task_vars is None else task_vars
1304     environment = {} if environment is None else environment
1305 
1306     with open(module_path, 'rb') as f:
1307 
1308         # read in the module source
1309         b_module_data = f.read()
1310 
1311     (b_module_data, module_style, shebang) = _find_module_utils(module_name, b_module_data, module_path, module_args, task_vars, templar, module_compression,
1312                                                                 async_timeout=async_timeout, become=become, become_method=become_method,
1313                                                                 become_user=become_user, become_password=become_password, become_flags=become_flags,
1314                                                                 environment=environment)
1315 
1316     if module_style == 'binary':
1317         return (b_module_data, module_style, to_text(shebang, nonstring='passthru'))
1318     elif shebang is None:
1319         b_lines = b_module_data.split(b"\n", 1)
1320         if b_lines[0].startswith(b"#!"):
1321             b_shebang = b_lines[0].strip()
1322             # shlex.split on python-2.6 needs bytes.  On python-3.x it needs text
1323             args = shlex.split(to_native(b_shebang[2:], errors='surrogate_or_strict'))
1324 
1325             # _get_shebang() takes text strings
1326             args = [to_text(a, errors='surrogate_or_strict') for a in args]
1327             interpreter = args[0]
1328             b_new_shebang = to_bytes(_get_shebang(interpreter, task_vars, templar, args[1:])[0],
1329                                      errors='surrogate_or_strict', nonstring='passthru')
1330 
1331             if b_new_shebang:
1332                 b_lines[0] = b_shebang = b_new_shebang
1333 
1334             if os.path.basename(interpreter).startswith(u'python'):
1335                 b_lines.insert(1, b_ENCODING_STRING)
1336 
1337             shebang = to_text(b_shebang, nonstring='passthru', errors='surrogate_or_strict')
1338         else:

1340             pass

1343
