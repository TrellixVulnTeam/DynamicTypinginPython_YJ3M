  52 class TaskQueueManager:
  53 
  54     '''
  55     This class handles the multiprocessing requirements of Ansible by
  56     creating a pool of worker forks, a result handler fork, and a
  57     manager object with shared datastructures/queues for coordinating
  58     work between all processes.
  59 
  60     The queue manager is responsible for loading the play strategy plugin,
  61     which dispatches the Play's tasks to hosts.
  62     '''
  63 
  64     RUN_OK = 0
  65     RUN_ERROR = 1
  66     RUN_FAILED_HOSTS = 2
  67     RUN_UNREACHABLE_HOSTS = 4
  68     RUN_FAILED_BREAK_PLAY = 8
  69     RUN_UNKNOWN_ERROR = 255

  71     def __init__(self, inventory, variable_manager, loader, passwords, stdout_callback=None, run_additional_callbacks=True, run_tree=False, forks=None):
  72 
  73         self._inventory = inventory
  74         self._variable_manager = variable_manager
  75         self._loader = loader
  76         self._stats = AggregateStats()
  77         self.passwords = passwords
  78         self._stdout_callback = stdout_callback
  79         self._run_additional_callbacks = run_additional_callbacks
  80         self._run_tree = run_tree
  81         self._forks = forks or 5
  82 
  83         self._callbacks_loaded = False
  84         self._callback_plugins = []
  85         self._start_at_done = False
  86 
  87         # make sure any module paths (if specified) are added to the module_loader
  88         if context.CLIARGS.get('module_path', False):
  89             for path in context.CLIARGS['module_path']:
  90                 if path:
  91                     module_loader.add_directory(path)
  92 
  93         # a special flag to help us exit cleanly
  94         self._terminated = False
  95 
  96         # dictionaries to keep track of failed/unreachable hosts
  97         self._failed_hosts = dict()
  98         self._unreachable_hosts = dict()
  99 
 100         try:
 101             self._final_q = multiprocessing_context.Queue()
 102         except OSError as e:
 103             raise AnsibleError("Unable to use multiprocessing, this is normally caused by lack of access to /dev/shm: %s" % to_native(e))
 104 
 105         # A temporary file (opened pre-fork) used by connection
 106         # plugins for inter-process locking.
 107         self._connection_lockfile = tempfile.TemporaryFile()
 108 
 109     def _initialize_processes(self, num):
 110         self._workers = []
 111 
 112         for i in range(num):
 113             self._workers.append(None)

 115     def load_callbacks(self):
 116         '''
 117         Loads all available callbacks, with the exception of those which
 118         utilize the CALLBACK_TYPE option. When CALLBACK_TYPE is set to 'stdout',
 119         only one such callback plugin will be loaded.
 120         '''
 121 
 122         if self._callbacks_loaded:
 123             return
 124 
 125         stdout_callback_loaded = False
 126         if self._stdout_callback is None:
 127             self._stdout_callback = C.DEFAULT_STDOUT_CALLBACK
 128 
 129         if isinstance(self._stdout_callback, CallbackBase):
 130             stdout_callback_loaded = True
 131         elif isinstance(self._stdout_callback, string_types):
 132             if self._stdout_callback not in callback_loader:
 133                 raise AnsibleError("Invalid callback for stdout specified: %s" % self._stdout_callback)
 134             else:
 135                 self._stdout_callback = callback_loader.get(self._stdout_callback)
 136                 self._stdout_callback.set_options()
 137                 stdout_callback_loaded = True
 138         else:
 139             raise AnsibleError("callback must be an instance of CallbackBase or the name of a callback plugin")
 140 
 141         loaded_callbacks = set()
 142 
 143         # first, load callbacks in the core distribution and configured callback paths
 144         for callback_plugin in callback_loader.all(class_only=True):
 145             callback_type = getattr(callback_plugin, 'CALLBACK_TYPE', '')
 146             callback_needs_whitelist = getattr(callback_plugin, 'CALLBACK_NEEDS_WHITELIST', False)
 147             (callback_name, _) = os.path.splitext(os.path.basename(callback_plugin._original_path))
 148             if callback_type == 'stdout':
 149                 # we only allow one callback of type 'stdout' to be loaded,
 150                 if callback_name != self._stdout_callback or stdout_callback_loaded:
 151                     continue
 152                 stdout_callback_loaded = True
 153             elif callback_name == 'tree' and self._run_tree:
 154                 # special case for ansible cli option
 155                 pass
 156             # eg, ad-hoc doesn't allow non-default callbacks
 157             elif not self._run_additional_callbacks or (callback_needs_whitelist and (
 158                     C.DEFAULT_CALLBACK_WHITELIST is None or callback_name not in C.DEFAULT_CALLBACK_WHITELIST)):
 159                 # 2.x plugins shipped with ansible should require whitelisting, older or non shipped should load automatically
 160                 continue
 161 
 162             callback_obj = callback_plugin()
 163             loaded_callbacks.add(callback_name)  # mark as loaded so we skip in second pass
 164             callback_obj.set_options()
 165             self._callback_plugins.append(callback_obj)
 166 
 167         # eg, ad-hoc doesn't allow non-default callbacks
 168         if self._run_additional_callbacks:
 169             # Second pass over everything in the whitelist we haven't already loaded, try to explicitly load. This will catch
 170             # collection-hosted callbacks, as well as formerly-core callbacks that have been redirected to collections.
 171             for callback_plugin_name in (c for c in C.DEFAULT_CALLBACK_WHITELIST if c not in loaded_callbacks):
 172                 # TODO: need to extend/duplicate the stdout callback check here (and possible move this ahead of the old way
 173                 callback_obj, plugin_load_context = callback_loader.get_with_context(callback_plugin_name)
 174                 if callback_obj:
 175                     loaded_as_name = callback_obj._redirected_names[-1]
 176                     if loaded_as_name in loaded_callbacks:
 177                         display.warning("Skipping callback '%s', already loaded as '%s'." % (callback_plugin_name, loaded_as_name))
 178                         continue
 179                     loaded_callbacks.add(loaded_as_name)
 180                     callback_obj.set_options()
 181                     self._callback_plugins.append(callback_obj)
 182                 else:
 183                     display.warning("Skipping '%s', unable to load or use as a callback" % callback_plugin_name)
 184 
 185         self._callbacks_loaded = True
 186 
 187     def run(self, play):
 188         '''
 189         Iterates over the roles/tasks in a play, using the given (or default)
 190         strategy for queueing tasks. The default is the linear strategy, which
 191         operates like classic Ansible by keeping all hosts in lock-step with
 192         a given task (meaning no hosts move on to the next task until all hosts
 193         are done with the current task).
 194         '''
 195 
 196         if not self._callbacks_loaded:
 197             self.load_callbacks()
 198 
 199         all_vars = self._variable_manager.get_vars(play=play)
 200         warn_if_reserved(all_vars)
 201         templar = Templar(loader=self._loader, variables=all_vars)
 202 
 203         new_play = play.copy()
 204         new_play.post_validate(templar)
 205         new_play.handlers = new_play.compile_roles_handlers() + new_play.handlers
 206 
 207         self.hostvars = HostVars(
 208             inventory=self._inventory,
 209             variable_manager=self._variable_manager,
 210             loader=self._loader,
 211         )
 212 
 213         play_context = PlayContext(new_play, self.passwords, self._connection_lockfile.fileno())
 214         if (self._stdout_callback and
 215                 hasattr(self._stdout_callback, 'set_play_context')):
 216             self._stdout_callback.set_play_context(play_context)
 217 
 218         for callback_plugin in self._callback_plugins:
 219             if hasattr(callback_plugin, 'set_play_context'):
 220                 callback_plugin.set_play_context(play_context)
 221 
 222         self.send_callback('v2_playbook_on_play_start', new_play)
 223 
 224         # build the iterator
 225         iterator = PlayIterator(
 226             inventory=self._inventory,
 227             play=new_play,
 228             play_context=play_context,
 229             variable_manager=self._variable_manager,
 230             all_vars=all_vars,
 231             start_at_done=self._start_at_done,
 232         )
 233 
 234         # adjust to # of workers to configured forks or size of batch, whatever is lower
 235         self._initialize_processes(min(self._forks, iterator.batch_size))
 236 
 237         # load the specified strategy (or the default linear one)
 238         strategy = strategy_loader.get(new_play.strategy, self)
 239         if strategy is None:
 240             raise AnsibleError("Invalid play strategy specified: %s" % new_play.strategy, obj=play._ds)
 241 
 242         # Because the TQM may survive multiple play runs, we start by marking
 243         # any hosts as failed in the iterator here which may have been marked
 244         # as failed in previous runs. Then we clear the internal list of failed
 245         # hosts so we know what failed this round.
 246         for host_name in self._failed_hosts.keys():
 247             host = self._inventory.get_host(host_name)
 248             iterator.mark_host_failed(host)
 249 
 250         self.clear_failed_hosts()
 251 
 252         # during initialization, the PlayContext will clear the start_at_task
 253         # field to signal that a matching task was found, so check that here
 254         # and remember it so we don't try to skip tasks on future plays
 255         if context.CLIARGS.get('start_at_task') is not None and play_context.start_at_task is None:
 256             self._start_at_done = True
 257 
 258         # and run the play using the strategy and cleanup on way out
 259         play_return = strategy.run(iterator, play_context)
 260 
 261         # now re-save the hosts that failed from the iterator to our internal list
 262         for host_name in iterator.get_failed_hosts():
 263             self._failed_hosts[host_name] = True
 264 
 265         strategy.cleanup()
 266         self._cleanup_processes()
 267         return play_return
 268 
 269     def cleanup(self):
 270         display.debug("RUNNING CLEANUP")
 271         self.terminate()
 272         self._final_q.close()
 273         self._cleanup_processes()
 274 
 275     def _cleanup_processes(self):
 276         if hasattr(self, '_workers'):
 277             for attempts_remaining in range(C.WORKER_SHUTDOWN_POLL_COUNT - 1, -1, -1):
 278                 if not any(worker_prc and worker_prc.is_alive() for worker_prc in self._workers):
 279                     break
 280 
 281                 if attempts_remaining:
 282                     time.sleep(C.WORKER_SHUTDOWN_POLL_DELAY)
 283                 else:
 284                     display.warning('One or more worker processes are still running and will be terminated.')
 285 
 286             for worker_prc in self._workers:
 287                 if worker_prc and worker_prc.is_alive():
 288                     try:
 289                         worker_prc.terminate()
 290                     except AttributeError:
 291                         pass

 311     def has_dead_workers(self):
 312 
 313         # [<WorkerProcess(WorkerProcess-2, stopped[SIGKILL])>,
 314         # <WorkerProcess(WorkerProcess-2, stopped[SIGTERM])>
 315 
 316         defunct = False
 317         for x in self._workers:
 318             if getattr(x, 'exitcode', None):
 319                 defunct = True
 320         return defunct

 322     def send_callback(self, method_name, *args, **kwargs):
 323         for callback_plugin in [self._stdout_callback] + self._callback_plugins:
 324             # a plugin that set self.disabled to True will not be called
 325             # see osx_say.py example for such a plugin
 326             if getattr(callback_plugin, 'disabled', False):
 327                 continue
 328 
 329             # try to find v2 method, fallback to v1 method, ignore callback if no method found
 330             methods = []
 331             for possible in [method_name, 'v2_on_any']:
 332                 gotit = getattr(callback_plugin, possible, None)
 333                 if gotit is None:
 334                     gotit = getattr(callback_plugin, possible.replace('v2_', ''), None)
 335                 if gotit is not None:
 336                     methods.append(gotit)
 337 
 338             # send clean copies
 339             new_args = []
 340             for arg in args:
 341                 # FIXME: add play/task cleaners
 342                 if isinstance(arg, TaskResult):
 343                     new_args.append(arg.clean_copy())
 344                 # elif isinstance(arg, Play):
 345                 # elif isinstance(arg, Task):
 346                 else:
 347                     new_args.append(arg)
 348 
 349             for method in methods:
 350                 try:
 351                     method(*new_args, **kwargs)
 352                 except Exception as e:
 353                     # TODO: add config toggle to make this fatal or not?
 354                     display.warning(u"Failure using method (%s) in callback plugin (%s): %s" % (to_text(method_name), to_text(callback_plugin), to_text(e)))
 355                     from traceback import format_tb
 356                     from sys import exc_info
 357                     display.vvv('Callback Exception: \n' + ' '.join(format_tb(exc_info()[2])))
