  50 class WorkerProcess(multiprocessing_context.Process):
  51     '''
  52     The worker thread class, which uses TaskExecutor to run tasks
  53     read from a job queue and pushes results into a results queue
  54     for reading later.
  55     '''
  56 
  57     def __init__(self, final_q, task_vars, host, task, play_context, loader, variable_manager, shared_loader_obj):
  58 
  59         super(WorkerProcess, self).__init__()
  60         # takes a task queue manager as the sole param:
  61         self._final_q = final_q
  62         self._task_vars = task_vars
  63         self._host = host
  64         self._task = task
  65         self._play_context = play_context
  66         self._loader = loader
  67         self._variable_manager = variable_manager
  68         self._shared_loader_obj = shared_loader_obj
  69 
  70         # NOTE: this works due to fork, if switching to threads this should change to per thread storage of temp files
  71         # clear var to ensure we only delete files for this child
  72         self._loader._tempfiles = set()
  73 
  74     def _save_stdin(self):
  75         self._new_stdin = os.devnull
  76         try:
  77             if sys.stdin.isatty() and sys.stdin.fileno() is not None:
  78                 try:
  79                     self._new_stdin = os.fdopen(os.dup(sys.stdin.fileno()))
  80                 except OSError:
  81                     # couldn't dupe stdin, most likely because it's
  82                     # not a valid file descriptor, so we just rely on
  83                     # using the one that was passed in
  84                     pass
  85         except (AttributeError, ValueError):
  86             # couldn't get stdin's fileno, so we just carry on
  87             pass
  88 
  89     def start(self):
  90         '''
  91         multiprocessing.Process replaces the worker's stdin with a new file
  92         opened on os.devnull, but we wish to preserve it if it is connected to
  93         a terminal. Therefore dup a copy prior to calling the real start(),
  94         ensuring the descriptor is preserved somewhere in the new child, and
  95         make sure it is closed in the parent when start() completes.
  96         '''
  97 
  98         self._save_stdin()
  99         try:
 100             return super(WorkerProcess, self).start()
 101         finally:
 102             if self._new_stdin != os.devnull:
 103                 self._new_stdin.close()
 104 
 105     def _hard_exit(self, e):
 106         '''
 107         There is no safe exception to return to higher level code that does not
 108         risk an innocent try/except finding itself executing in the wrong
 109         process. All code executing above WorkerProcess.run() on the stack
 110         conceptually belongs to another program.
 111         '''
 112 
 113         try:
 114             display.debug(u"WORKER HARD EXIT: %s" % to_text(e))
 115         except BaseException:
 116             # If the cause of the fault is IOError being generated by stdio,
 117             # attempting to log a debug message may trigger another IOError.
 118             # Try printing once then give up.
 119             pass
 120 
 121         os._exit(1)
 122 
 123     def run(self):
 124         '''
 125         Wrap _run() to ensure no possibility an errant exception can cause
 126         control to return to the StrategyBase task loop, or any other code
 127         higher in the stack.
 128 
 129         As multiprocessing in Python 2.x provides no protection, it is possible
 130         a try/except added in far-away code can cause a crashed child process
 131         to suddenly assume the role and prior state of its parent.
 132         '''
 133         try:
 134             return self._run()
 135         except BaseException as e:
 136             self._hard_exit(e)
 137 
 138     def _run(self):
 139         '''
 140         Called when the process is started.  Pushes the result onto the
 141         results queue. We also remove the host from the blocked hosts list, to
 142         signify that they are ready for their next task.
 143         '''
 144 
 145         # import cProfile, pstats, StringIO
 146         # pr = cProfile.Profile()
 147         # pr.enable()
 148 
 149         if HAS_PYCRYPTO_ATFORK:
 150             atfork()
 151 
 152         try:
 153             # execute the task and build a TaskResult from the result
 154             display.debug("running TaskExecutor() for %s/%s" % (self._host, self._task))
 155             executor_result = TaskExecutor(
 156                 self._host,
 157                 self._task,
 158                 self._task_vars,
 159                 self._play_context,
 160                 self._new_stdin,
 161                 self._loader,
 162                 self._shared_loader_obj,
 163                 self._final_q
 164             ).run()
 165 
 166             display.debug("done running TaskExecutor() for %s/%s [%s]" % (self._host, self._task, self._task._uuid))
 167             self._host.vars = dict()
 168             self._host.groups = []
 169             task_result = TaskResult(
 170                 self._host.name,
 171                 self._task._uuid,
 172                 executor_result,
 173                 task_fields=self._task.dump_attrs(),
 174             )
 175 
 176             # put the result on the result queue
 177             display.debug("sending task result for task %s" % self._task._uuid)
 178             self._final_q.put(task_result)
 179             display.debug("done sending task result for task %s" % self._task._uuid)
 180 
 181         except AnsibleConnectionFailure:
 182             self._host.vars = dict()
 183             self._host.groups = []
 184             task_result = TaskResult(
 185                 self._host.name,
 186                 self._task._uuid,
 187                 dict(unreachable=True),
 188                 task_fields=self._task.dump_attrs(),
 189             )
 190             self._final_q.put(task_result, block=False)
 191 
 192         except Exception as e:
 193             if not isinstance(e, (IOError, EOFError, KeyboardInterrupt, SystemExit)) or isinstance(e, TemplateNotFound):
 194                 try:
 195                     self._host.vars = dict()
 196                     self._host.groups = []
 197                     task_result = TaskResult(
 198                         self._host.name,
 199                         self._task._uuid,
 200                         dict(failed=True, exception=to_text(traceback.format_exc()), stdout=''),
 201                         task_fields=self._task.dump_attrs(),
 202                     )
 203                     self._final_q.put(task_result, block=False)
 204                 except Exception:
 205                     display.debug(u"WORKER EXCEPTION: %s" % to_text(e))
 206                     display.debug(u"WORKER TRACEBACK: %s" % to_text(traceback.format_exc()))
 207                 finally:
 208                     self._clean_up()
 209 
 210         display.debug("WORKER PROCESS EXITING")
 211 
 212         # pr.disable()
 213         # s = StringIO.StringIO()
 214         # sortby = 'time'
 215         # ps = pstats.Stats(pr, stream=s).sort_stats(sortby)
 216         # ps.print_stats()
 217         # with open('worker_%06d.stats' % os.getpid(), 'w') as f:
 218         #     f.write(s.getvalue())

 220     def _clean_up(self):
 221         # NOTE: see note in init about forks
 222         # ensure we cleanup all temp files for this worker
 223         self._loader.cleanup_all_tmp_files()
