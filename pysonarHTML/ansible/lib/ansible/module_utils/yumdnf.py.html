  60 class YumDnf(with_metaclass(ABCMeta, object)):
  61     """
  62     Abstract class that handles the population of instance variables that should
  63     be identical between both YUM and DNF modules because of the feature parity
  64     and shared argument spec
  65     """
  66 
  67     def __init__(self, module):
  68 
  69         self.module = module
  70 
  71         self.allow_downgrade = self.module.params['allow_downgrade']
  72         self.autoremove = self.module.params['autoremove']
  73         self.bugfix = self.module.params['bugfix']
  74         self.conf_file = self.module.params['conf_file']
  75         self.disable_excludes = self.module.params['disable_excludes']
  76         self.disable_gpg_check = self.module.params['disable_gpg_check']
  77         self.disable_plugin = self.module.params['disable_plugin']
  78         self.disablerepo = self.module.params.get('disablerepo', [])
  79         self.download_only = self.module.params['download_only']
  80         self.download_dir = self.module.params['download_dir']
  81         self.enable_plugin = self.module.params['enable_plugin']
  82         self.enablerepo = self.module.params.get('enablerepo', [])
  83         self.exclude = self.module.params['exclude']
  84         self.installroot = self.module.params['installroot']
  85         self.install_repoquery = self.module.params['install_repoquery']
  86         self.install_weak_deps = self.module.params['install_weak_deps']
  87         self.list = self.module.params['list']
  88         self.names = [p.strip() for p in self.module.params['name']]
  89         self.releasever = self.module.params['releasever']
  90         self.security = self.module.params['security']
  91         self.skip_broken = self.module.params['skip_broken']
  92         self.state = self.module.params['state']
  93         self.update_only = self.module.params['update_only']
  94         self.update_cache = self.module.params['update_cache']
  95         self.validate_certs = self.module.params['validate_certs']
  96         self.lock_timeout = self.module.params['lock_timeout']
  97 
  98         # It's possible someone passed a comma separated string since it used
  99         # to be a string type, so we should handle that
 100         self.names = self.listify_comma_sep_strings_in_list(self.names)
 101         self.disablerepo = self.listify_comma_sep_strings_in_list(self.disablerepo)
 102         self.enablerepo = self.listify_comma_sep_strings_in_list(self.enablerepo)
 103         self.exclude = self.listify_comma_sep_strings_in_list(self.exclude)
 104 
 105         # Fail if someone passed a space separated string
 106         # https://github.com/ansible/ansible/issues/46301
 107         for name in self.names:
 108             if ' ' in name and not any(spec in name for spec in ['@', '>', '<', '=']):
 109                 module.fail_json(
 110                     msg='It appears that a space separated string of packages was passed in '
 111                         'as an argument. To operate on several packages, pass a comma separated '
 112                         'string of packages or a list of packages.'
 113                 )
 114 
 115         # Sanity checking for autoremove
 116         if self.state is None:
 117             if self.autoremove:
 118                 self.state = "absent"
 119             else:
 120                 self.state = "present"
 121 
 122         if self.autoremove and (self.state != "absent"):
 123             self.module.fail_json(
 124                 msg="Autoremove should be used alone or with state=absent",
 125                 results=[],
 126             )
 127 
 128         # This should really be redefined by both the yum and dnf module but a
 129         # default isn't a bad idea
 130         self.lockfile = '/var/run/yum.pid'
 131 
 132     @abstractmethod
 133     def is_lockfile_pid_valid(self):
 134         return
 135 
 136     def _is_lockfile_present(self):
 137         return (os.path.isfile(self.lockfile) or glob.glob(self.lockfile)) and self.is_lockfile_pid_valid()
 138 
 139     def wait_for_lock(self):
 140         '''Poll until the lock is removed if timeout is a positive number'''
 141 
 142         if not self._is_lockfile_present():
 143             return
 144 
 145         if self.lock_timeout > 0:
 146             for iteration in range(0, self.lock_timeout):
 147                 time.sleep(1)
 148                 if not self._is_lockfile_present():
 149                     return
 150 
 151         self.module.fail_json(msg='{0} lockfile is held by another process'.format(self.pkg_mgr_name))
 152 
 153     def listify_comma_sep_strings_in_list(self, some_list):
 154         """
 155         method to accept a list of strings as the parameter, find any strings
 156         in that list that are comma separated, remove them from the list and add
 157         their comma separated elements to the original list
 158         """
 159         new_list = []
 160         remove_from_original_list = []
 161         for element in some_list:
 162             if ',' in element:
 163                 remove_from_original_list.append(element)
 164                 new_list.extend([e.strip() for e in element.split(',')])
 165 
 166         for element in remove_from_original_list:
 167             some_list.remove(element)
 168 
 169         some_list.extend(new_list)
 170 
 171         if some_list == [""]:
 172             return []
 173 
 174         return some_list
