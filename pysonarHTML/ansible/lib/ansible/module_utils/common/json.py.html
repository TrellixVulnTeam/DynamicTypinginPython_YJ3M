  18 def _preprocess_unsafe_encode(value):
  19     """Recursively preprocess a data structure converting instances of ``AnsibleUnsafe``
  20     into their JSON dict representations
  21 
  22     Used in ``AnsibleJSONEncoder.iterencode``
  23     """
  24     if getattr(value, '__UNSAFE__', False) and not getattr(value, '__ENCRYPTED__', False):
  25         value = {'__ansible_unsafe': to_text(value, errors='surrogate_or_strict', nonstring='strict')}
  26     elif is_sequence(value):
  27         value = [_preprocess_unsafe_encode(v) for v in value]
  28     elif isinstance(value, Mapping):
  29         value = dict((k, _preprocess_unsafe_encode(v)) for k, v in value.items())
  30 
  31     return value

  34 class AnsibleJSONEncoder(json.JSONEncoder):
  35     '''
  36     Simple encoder class to deal with JSON encoding of Ansible internal types
  37     '''
  38 
  39     def __init__(self, preprocess_unsafe=False, vault_to_text=False, **kwargs):
  40         self._preprocess_unsafe = preprocess_unsafe
  41         self._vault_to_text = vault_to_text
  42         super(AnsibleJSONEncoder, self).__init__(**kwargs)
  43 
  44     # NOTE: ALWAYS inform AWS/Tower when new items get added as they consume them downstream via a callback
  45     def default(self, o):
  46         if getattr(o, '__ENCRYPTED__', False):
  47             # vault object
  48             if self._vault_to_text:
  49                 value = to_text(o, errors='surrogate_or_strict')
  50             else:
  51                 value = {'__ansible_vault': to_text(o._ciphertext, errors='surrogate_or_strict', nonstring='strict')}
  52         elif getattr(o, '__UNSAFE__', False):
  53             # unsafe object, this will never be triggered, see ``AnsibleJSONEncoder.iterencode``
  54             value = {'__ansible_unsafe': to_text(o, errors='surrogate_or_strict', nonstring='strict')}
  55         elif isinstance(o, Mapping):
  56             # hostvars and other objects
  57             value = dict(o)
  58         elif isinstance(o, (datetime.date, datetime.datetime)):
  59             # date object
  60             value = o.isoformat()
  61         else:
  62             # use default encoder
  63             value = super(AnsibleJSONEncoder, self).default(o)
  64         return value
  65 
  66     def iterencode(self, o, **kwargs):
  67         """Custom iterencode, primarily design to handle encoding ``AnsibleUnsafe``
  68         as the ``AnsibleUnsafe`` subclasses inherit from string types and
  69         ``json.JSONEncoder`` does not support custom encoders for string types
  70         """
  71         if self._preprocess_unsafe:
  72             o = _preprocess_unsafe_encode(o)
  73 
  74         return super(AnsibleJSONEncoder, self).iterencode(o, **kwargs)
