  14 class ImmutableDict(Hashable, Mapping):
  15     """Dictionary that cannot be updated"""
  16     def __init__(self, *args, **kwargs):
  17         self._store = dict(*args, **kwargs)
  18 
  19     def __getitem__(self, key):
  20         return self._store[key]
  21 
  22     def __iter__(self):
  23         return self._store.__iter__()
  24 
  25     def __len__(self):
  26         return self._store.__len__()
  27 
  28     def __hash__(self):
  29         return hash(frozenset(self.items()))
  30 
  31     def __eq__(self, other):
  32         try:
  33             if self.__hash__() == hash(other):
  34                 return True
  35         except TypeError:
  36             pass
  37 
  38         return False
  39 
  40     def __repr__(self):
  41         return 'ImmutableDict({0})'.format(repr(self._store))
  42 
  43     def union(self, overriding_mapping):
  44         """
  45         Create an ImmutableDict as a combination of the original and overriding_mapping
  46 
  47         :arg overriding_mapping: A Mapping of replacement and additional items
  48         :return: A copy of the ImmutableDict with key-value pairs from the overriding_mapping added
  49 
  50         If any of the keys in overriding_mapping are already present in the original ImmutableDict,
  51         the overriding_mapping item replaces the one in the original ImmutableDict.
  52         """
  53         return ImmutableDict(self._store, **overriding_mapping)
  54 
  55     def difference(self, subtractive_iterable):
  56         """
  57         Create an ImmutableDict as a combination of the original minus keys in subtractive_iterable
  58 
  59         :arg subtractive_iterable: Any iterable containing keys that should not be present in the
  60             new ImmutableDict
  61         :return: A copy of the ImmutableDict with keys from the subtractive_iterable removed
  62         """
  63         remove_keys = frozenset(subtractive_iterable)
  64         keys = (k for k in self._store.keys() if k not in remove_keys)
  65         return ImmutableDict((k, self._store[k]) for k in keys)

  68 def is_string(seq):
  69     """Identify whether the input has a string-like type (inclding bytes)."""
  70     return isinstance(seq, (text_type, binary_type))
  71

  73 def is_iterable(seq, include_strings=False):
  74     """Identify whether the input is an iterable."""
  75     if not include_strings and is_string(seq):
  76         return False
  77 
  78     try:
  79         iter(seq)
  80         return True
  81     except TypeError:
  82         return False
  83 
  84 
  85 def is_sequence(seq, include_strings=False):
  86     """Identify whether the input is a sequence.
  87 
  88     Strings and bytes are not sequences here,
  89     unless ``include_string`` is ``True``.
  90 
  91     Non-indexable things are never of a sequence type.
  92     """
  93     if not include_strings and is_string(seq):
  94         return False
  95 
  96     return isinstance(seq, Sequence)

  99 def count(seq):
 100     """Returns a dictionary with the number of appearances of each element of the iterable.
 101 
 102     Resembles the collections.Counter class functionality. It is meant to be used when the
 103     code is run on Python 2.6.* where collections.Counter is not available. It should be
 104     deprecated and replaced when support for Python < 2.7 is dropped.
 105     """
 106     if not is_iterable(seq):
 107         raise Exception('Argument provided  is not an iterable')
 108     counters = dict()
 109     for elem in seq:
 110         counters[elem] = counters.get(elem, 0) + 1
 111     return counters
