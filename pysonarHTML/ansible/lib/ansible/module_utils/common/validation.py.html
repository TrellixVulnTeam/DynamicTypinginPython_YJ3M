  26 def count_terms(terms, module_parameters):
  27     """Count the number of occurrences of a key in a given dictionary
  28 
  29     :arg terms: String or iterable of values to check
  30     :arg module_parameters: Dictionary of module parameters
  31 
  32     :returns: An integer that is the number of occurrences of the terms values
  33         in the provided dictionary.
  34     """
  35 
  36     if not is_iterable(terms):
  37         terms = [terms]
  38 
  39     return len(set(terms).intersection(module_parameters))

  42 def check_mutually_exclusive(terms, module_parameters):
  43     """Check mutually exclusive terms against argument parameters
  44 
  45     Accepts a single list or list of lists that are groups of terms that should be
  46     mutually exclusive with one another
  47 
  48     :arg terms: List of mutually exclusive module parameters
  49     :arg module_parameters: Dictionary of module parameters
  50 
  51     :returns: Empty list or raises TypeError if the check fails.
  52     """
  53 
  54     results = []
  55     if terms is None:
  56         return results
  57 
  58     for check in terms:
  59         count = count_terms(check, module_parameters)
  60         if count > 1:
  61             results.append(check)
  62 
  63     if results:
  64         full_list = ['|'.join(check) for check in results]
  65         msg = "parameters are mutually exclusive: %s" % ', '.join(full_list)
  66         raise TypeError(to_native(msg))
  67 
  68     return results

  71 def check_required_one_of(terms, module_parameters):
  72     """Check each list of terms to ensure at least one exists in the given module
  73     parameters
  74 
  75     Accepts a list of lists or tuples
  76 
  77     :arg terms: List of lists of terms to check. For each list of terms, at
  78         least one is required.
  79     :arg module_parameters: Dictionary of module parameters
  80 
  81     :returns: Empty list or raises TypeError if the check fails.
  82     """
  83 
  84     results = []
  85     if terms is None:
  86         return results
  87 
  88     for term in terms:
  89         count = count_terms(term, module_parameters)
  90         if count == 0:
  91             results.append(term)
  92 
  93     if results:
  94         for term in results:
  95             msg = "one of the following is required: %s" % ', '.join(term)
  96             raise TypeError(to_native(msg))
  97 
  98     return results

 101 def check_required_together(terms, module_parameters):
 102     """Check each list of terms to ensure every parameter in each list exists
 103     in the given module parameters
 104 
 105     Accepts a list of lists or tuples
 106 
 107     :arg terms: List of lists of terms to check. Each list should include
 108         parameters that are all required when at least one is specified
 109         in the module_parameters.
 110     :arg module_parameters: Dictionary of module parameters
 111 
 112     :returns: Empty list or raises TypeError if the check fails.
 113     """
 114 
 115     results = []
 116     if terms is None:
 117         return results
 118 
 119     for term in terms:
 120         counts = [count_terms(field, module_parameters) for field in term]
 121         non_zero = [c for c in counts if c > 0]
 122         if len(non_zero) > 0:
 123             if 0 in counts:
 124                 results.append(term)
 125     if results:
 126         for term in results:
 127             msg = "parameters are required together: %s" % ', '.join(term)
 128             raise TypeError(to_native(msg))
 129 
 130     return results

 133 def check_required_by(requirements, module_parameters):
 134     """For each key in requirements, check the corresponding list to see if they
 135     exist in module_parameters
 136 
 137     Accepts a single string or list of values for each key
 138 
 139     :arg requirements: Dictionary of requirements
 140     :arg module_parameters: Dictionary of module parameters
 141 
 142     :returns: Empty dictionary or raises TypeError if the
 143     """
 144 
 145     result = {}
 146     if requirements is None:
 147         return result
 148 
 149     for (key, value) in requirements.items():
 150         if key not in module_parameters or module_parameters[key] is None:
 151             continue
 152         result[key] = []
 153         # Support strings (single-item lists)
 154         if isinstance(value, string_types):
 155             value = [value]
 156         for required in value:
 157             if required not in module_parameters or module_parameters[required] is None:
 158                 result[key].append(required)
 159 
 160     if result:
 161         for key, missing in result.items():
 162             if len(missing) > 0:
 163                 msg = "missing parameter(s) required by '%s': %s" % (key, ', '.join(missing))
 164                 raise TypeError(to_native(msg))
 165 
 166     return result

 169 def check_required_arguments(argument_spec, module_parameters):
 170     """Check all paramaters in argument_spec and return a list of parameters
 171     that are required but not present in module_parameters
 172 
 173     Raises TypeError if the check fails
 174 
 175     :arg argument_spec: Argument spec dicitionary containing all parameters
 176         and their specification
 177     :arg module_paramaters: Dictionary of module parameters
 178 
 179     :returns: Empty list or raises TypeError if the check fails.
 180     """
 181 
 182     missing = []
 183     if argument_spec is None:
 184         return missing
 185 
 186     for (k, v) in argument_spec.items():
 187         required = v.get('required', False)
 188         if required and k not in module_parameters:
 189             missing.append(k)
 190 
 191     if missing:
 192         msg = "missing required arguments: %s" % ", ".join(sorted(missing))
 193         raise TypeError(to_native(msg))
 194 
 195     return missing

 198 def check_required_if(requirements, module_parameters):
 199     """Check parameters that are conditionally required
 200 
 201     Raises TypeError if the check fails
 202 
 203     :arg requirements: List of lists specifying a parameter, value, parameters
 204         required when the given parameter is the specified value, and optionally
 205         a boolean indicating any or all parameters are required.
 206 
 207         Example:
 208             required_if=[
 209                 ['state', 'present', ('path',), True],
 210                 ['someint', 99, ('bool_param', 'string_param')],
 211             ]
 212 
 213     :arg module_paramaters: Dictionary of module parameters
 214 
 215     :returns: Empty list or raises TypeError if the check fails.
 216         The results attribute of the exception contains a list of dictionaries.
 217         Each dictionary is the result of evaluting each item in requirements.
 218         Each return dictionary contains the following keys:
 219 
 220             :key missing: List of parameters that are required but missing
 221             :key requires: 'any' or 'all'
 222             :key paramater: Parameter name that has the requirement
 223             :key value: Original value of the paramater
 224             :key requirements: Original required parameters
 225 
 226         Example:
 227             [
 228                 {
 229                     'parameter': 'someint',
 230                     'value': 99
 231                     'requirements': ('bool_param', 'string_param'),
 232                     'missing': ['string_param'],
 233                     'requires': 'all',
 234                 }
 235             ]
 236 
 237     """
 238     results = []
 239     if requirements is None:
 240         return results
 241 
 242     for req in requirements:
 243         missing = {}
 244         missing['missing'] = []
 245         max_missing_count = 0
 246         is_one_of = False
 247         if len(req) == 4:
 248             key, val, requirements, is_one_of = req
 249         else:
 250             key, val, requirements = req
 251 
 252         # is_one_of is True at least one requirement should be
 253         # present, else all requirements should be present.
 254         if is_one_of:
 255             max_missing_count = len(requirements)
 256             missing['requires'] = 'any'
 257         else:
 258             missing['requires'] = 'all'
 259 
 260         if key in module_parameters and module_parameters[key] == val:
 261             for check in requirements:
 262                 count = count_terms(check, module_parameters)
 263                 if count == 0:
 264                     missing['missing'].append(check)
 265         if len(missing['missing']) and len(missing['missing']) >= max_missing_count:
 266             missing['parameter'] = key
 267             missing['value'] = val
 268             missing['requirements'] = requirements
 269             results.append(missing)
 270 
 271     if results:
 272         for missing in results:
 273             msg = "%s is %s but %s of the following are missing: %s" % (
 274                 missing['parameter'], missing['value'], missing['requires'], ', '.join(missing['missing']))
 275             raise TypeError(to_native(msg))
 276 
 277     return results

 280 def check_missing_parameters(module_parameters, required_parameters=None):
 281     """This is for checking for required params when we can not check via
 282     argspec because we need more information than is simply given in the argspec.
 283 
 284     Raises TypeError if any required parameters are missing
 285 
 286     :arg module_paramaters: Dictionary of module parameters
 287     :arg required_parameters: List of parameters to look for in the given module
 288         parameters
 289 
 290     :returns: Empty list or raises TypeError if the check fails.
 291     """
 292     missing_params = []
 293     if required_parameters is None:
 294         return missing_params
 295 
 296     for param in required_parameters:
 297         if not module_parameters.get(param):
 298             missing_params.append(param)
 299 
 300     if missing_params:
 301         msg = "missing required arguments: %s" % ', '.join(missing_params)
 302         raise TypeError(to_native(msg))
 303 
 304     return missing_params

 307 def safe_eval(value, locals=None, include_exceptions=False):
 308     # do not allow method calls to modules
 309     if not isinstance(value, string_types):
 310         # already templated to a datavaluestructure, perhaps?
 311         if include_exceptions:
 312             return (value, None)
 313         return value
 314     if re.search(r'\w\.\w+\(', value):
 315         if include_exceptions:
 316             return (value, None)
 317         return value
 318     # do not allow imports
 319     if re.search(r'import \w+', value):
 320         if include_exceptions:
 321             return (value, None)
 322         return value
 323     try:
 324         result = literal_eval(value)
 325         if include_exceptions:
 326             return (result, None)
 327         else:
 328             return result
 329     except Exception as e:
 330         if include_exceptions:
 331             return (value, e)
 332         return value

 335 def check_type_str(value, allow_conversion=True):
 336     """Verify that the value is a string or convert to a string.
 337 
 338     Since unexpected changes can sometimes happen when converting to a string,
 339     ``allow_conversion`` controls whether or not the value will be converted or a
 340     TypeError will be raised if the value is not a string and would be converted
 341 
 342     :arg value: Value to validate or convert to a string
 343     :arg allow_conversion: Whether to convert the string and return it or raise
 344         a TypeError
 345 
 346     :returns: Original value if it is a string, the value converted to a string
 347         if allow_conversion=True, or raises a TypeError if allow_conversion=False.
 348     """
 349     if isinstance(value, string_types):
 350         return value
 351 
 352     if allow_conversion:
 353         return to_native(value, errors='surrogate_or_strict')
 354 
 355     msg = "'{0!r}' is not a string and conversion is not allowed".format(value)
 356     raise TypeError(to_native(msg))

 359 def check_type_list(value):
 360     """Verify that the value is a list or convert to a list
 361 
 362     A comma separated string will be split into a list. Rases a TypeError if
 363     unable to convert to a list.
 364 
 365     :arg value: Value to validate or convert to a list
 366 
 367     :returns: Original value if it is already a list, single item list if a
 368         float, int or string without commas, or a multi-item list if a
 369         comma-delimited string.
 370     """
 371     if isinstance(value, list):
 372         return value
 373 
 374     if isinstance(value, string_types):
 375         return value.split(",")
 376     elif isinstance(value, int) or isinstance(value, float):
 377         return [str(value)]
 378 
 379     raise TypeError('%s cannot be converted to a list' % type(value))

 382 def check_type_dict(value):
 383     """Verify that value is a dict or convert it to a dict and return it.
 384 
 385     Raises TypeError if unable to convert to a dict
 386 
 387     :arg value: Dict or string to convert to a dict. Accepts 'k1=v2, k2=v2'.
 388 
 389     :returns: value converted to a dictionary
 390     """
 391     if isinstance(value, dict):
 392         return value
 393 
 394     if isinstance(value, string_types):
 395         if value.startswith("{"):
 396             try:
 397                 return json.loads(value)
 398             except Exception:
 399                 (result, exc) = safe_eval(value, dict(), include_exceptions=True)
 400                 if exc is not None:
 401                     raise TypeError('unable to evaluate string as dictionary')
 402                 return result
 403         elif '=' in value:
 404             fields = []
 405             field_buffer = []
 406             in_quote = False
 407             in_escape = False
 408             for c in value.strip():
 409                 if in_escape:
 410                     field_buffer.append(c)
 411                     in_escape = False
 412                 elif c == '\\':
 413                     in_escape = True
 414                 elif not in_quote and c in ('\'', '"'):
 415                     in_quote = c
 416                 elif in_quote and in_quote == c:
 417                     in_quote = False
 418                 elif not in_quote and c in (',', ' '):
 419                     field = ''.join(field_buffer)
 420                     if field:
 421                         fields.append(field)
 422                     field_buffer = []
 423                 else:
 424                     field_buffer.append(c)
 425 
 426             field = ''.join(field_buffer)
 427             if field:
 428                 fields.append(field)
 429             return dict(x.split("=", 1) for x in fields)
 430         else:
 431             raise TypeError("dictionary requested, could not parse JSON or key=value")
 432 
 433     raise TypeError('%s cannot be converted to a dict' % type(value))

 436 def check_type_bool(value):
 437     """Verify that the value is a bool or convert it to a bool and return it.
 438 
 439     Raises TypeError if unable to convert to a bool
 440 
 441     :arg value: String, int, or float to convert to bool. Valid booleans include:
 442          '1', 'on', 1, '0', 0, 'n', 'f', 'false', 'true', 'y', 't', 'yes', 'no', 'off'
 443 
 444     :returns: Boolean True or False
 445     """
 446     if isinstance(value, bool):
 447         return value
 448 
 449     if isinstance(value, string_types) or isinstance(value, (int, float)):
 450         return boolean(value)
 451 
 452     raise TypeError('%s cannot be converted to a bool' % type(value))

 455 def check_type_int(value):
 456     """Verify that the value is an integer and return it or convert the value
 457     to an integer and return it
 458 
 459     Raises TypeError if unable to convert to an int
 460 
 461     :arg value: String or int to convert of verify
 462 
 463     :return: Int of given value
 464     """
 465     if isinstance(value, integer_types):
 466         return value
 467 
 468     if isinstance(value, string_types):
 469         try:
 470             return int(value)
 471         except ValueError:
 472             pass
 473 
 474     raise TypeError('%s cannot be converted to an int' % type(value))

 536 def check_type_jsonarg(value):
 537     """Return a jsonified string. Sometimes the controller turns a json string
 538     into a dict/list so transform it back into json here
 539 
 540     Raises TypeError if unable to covert the value
 541 
 542     """
 543     if isinstance(value, (text_type, binary_type)):
 544         return value.strip()
 545     elif isinstance(value, (list, tuple, dict)):
 546         return jsonify(value)
 547     raise TypeError('%s cannot be converted to a json string' % type(value))
