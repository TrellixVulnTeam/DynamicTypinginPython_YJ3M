   5 from __future__ import absolute_import, division, print_function
   6 __metaclass__ = type

   8 from ansible.module_utils.six import string_types
   9 
  10 _global_warnings = []
  11 _global_deprecations = []
  12 
  13 
  14 def warn(warning):
  15     if isinstance(warning, string_types):
  16         _global_warnings.append(warning)
  17     else:
  18         raise TypeError("warn requires a string not a %s" % type(warning))
  19 
  20 
  21 def deprecate(msg, version=None, date=None, collection_name=None):
  22     if isinstance(msg, string_types):
  23         # For compatibility, we accept that neither version nor date is set,
  24         # and treat that the same as if version would haven been set
  25         if date is not None:
  26             _global_deprecations.append({'msg': msg, 'date': date, 'collection_name': collection_name})
  27         else:
  28             _global_deprecations.append({'msg': msg, 'version': version, 'collection_name': collection_name})
  29     else:
  30         raise TypeError("deprecate requires a string not a %s" % type(msg))
  31 
  32 
  33 def get_warning_messages():
  34     """Return a tuple of warning messages accumulated over this run"""
  35     return tuple(_global_warnings)
  36 
  37 
  38 def get_deprecation_messages():
  39     """Return a tuple of deprecations accumulated over this run"""
  40     return tuple(_global_deprecations)
