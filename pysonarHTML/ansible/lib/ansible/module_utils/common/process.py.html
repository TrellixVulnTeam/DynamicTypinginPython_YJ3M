  12 def get_bin_path(arg, opt_dirs=None, required=None):
  13     '''
  14     Find system executable in PATH. Raises ValueError if executable is not found.
  15     Optional arguments:
  16        - required:  [Deprecated] Prior to 2.10, if executable is not found and required is true it raises an Exception.
  17                     In 2.10 and later, an Exception is always raised. This parameter will be removed in 2.14.
  18        - opt_dirs:  optional list of directories to search in addition to PATH
  19     If found return full path, otherwise raise ValueError.
  20     '''
  21     opt_dirs = [] if opt_dirs is None else opt_dirs
  22 
  23     sbin_paths = ['/sbin', '/usr/sbin', '/usr/local/sbin']
  24     paths = []
  25     for d in opt_dirs:
  26         if d is not None and os.path.exists(d):
  27             paths.append(d)
  28     paths += os.environ.get('PATH', '').split(os.pathsep)
  29     bin_path = None
  30     # mangle PATH to include /sbin dirs
  31     for p in sbin_paths:
  32         if p not in paths and os.path.exists(p):
  33             paths.append(p)
  34     for d in paths:
  35         if not d:
  36             continue
  37         path = os.path.join(d, arg)
  38         if os.path.exists(path) and not os.path.isdir(path) and is_executable(path):
  39             bin_path = path
  40             break
  41     if bin_path is None:
  42         raise ValueError('Failed to find required executable %s in paths: %s' % (arg, os.pathsep.join(paths)))
  43 
  44     return bin_path
