  14 def removed_module(removed_in, msg='This module has been removed. The module documentation for'
  15                                    ' Ansible-%(version)s may contain hints for porting'):
  16     """
  17     This function is deprecated and should not be used. Instead the module should just be
  18     actually removed. This function is scheduled for removal for the 2.12 release.
  19 
  20     Returns module failure along with a message about the module being removed
  21 
  22     :arg removed_in: The version that the module was removed in
  23     :kwarg msg: Message to use in the module's failure message. The default says that the module
  24         has been removed and what version of the Ansible documentation to search for porting help.
  25 
  26     Remove the actual code and instead have boilerplate like this::
  27 
  28         from ansible.module_utils.common.removed import removed_module
  29 
  30         if __name__ == '__main__':
  31             removed_module("2.4")
  32     """
  33     results = {
  34         'failed': True,
  35         'deprecations': [
  36             {
  37                 'msg': 'The removed_module function is deprecated',
  38                 'version': '2.12',
  39             },
  40         ]
  41     }
  42 
  43     # Convert numbers into strings
  44     removed_in = to_native(removed_in)
  45 
  46     version = removed_in.split('.')
  47     try:
  48         numeric_minor = int(version[-1])
  49     except Exception:
  50         last_version = None
  51     else:
  52         version = version[:-1]
  53         version.append(to_native(numeric_minor - 1))
  54         last_version = '.'.join(version)
  55 
  56     if last_version is None:
  57         results['warnings'] = ['removed modules should specify the version they were removed in']
  58         results['msg'] = 'This module has been removed'
  59     else:
  60         results['msg'] = msg % {'version': last_version}
  61 
  62     print('\n{0}\n'.format(json.dumps(results)))
  63     sys.exit(1)
