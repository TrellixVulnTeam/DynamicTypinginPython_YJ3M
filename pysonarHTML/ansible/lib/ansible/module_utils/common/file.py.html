  65 def is_executable(path):
  66     # This function's signature needs to be repeated
  67     # as the first line of its docstring.
  68     # This method is reused by the basic module,
  69     # the repetion helps the basic module's html documentation come out right.
  70     # http://www.sphinx-doc.org/en/master/usage/extensions/autodoc.html#confval-autodoc_docstring_signature
  71     '''is_executable(path)
  72 
  73     is the given path executable?
  74 
  75     :arg path: The path of the file to check.
  76 
  77     Limitations:
  78 
  79     * Does not account for FSACLs.
  80     * Most times we really want to know "Can the current user execute this
  81       file".  This function does not tell us that, only if any execute bit is set.
  82     '''
  83     # These are all bitfields so first bitwise-or all the permissions we're
  84     # looking for, then bitwise-and with the file's mode to determine if any
  85     # execute bits are set.
  86     return ((stat.S_IXUSR | stat.S_IXGRP | stat.S_IXOTH) & os.stat(path)[stat.ST_MODE])

 117 class FileLock:
 118     '''
 119     Currently FileLock is implemented via fcntl.flock on a lock file, however this
 120     behaviour may change in the future. Avoid mixing lock types fcntl.flock,
 121     fcntl.lockf and module_utils.common.file.FileLock as it will certainly cause
 122     unwanted and/or unexpected behaviour
 123     '''
 124     def __init__(self):
 125         self.lockfd = None
 126 
 127     @contextmanager
 128     def lock_file(self, path, tmpdir, lock_timeout=None):
 129         '''
 130         Context for lock acquisition
 131         '''
 132         try:
 133             self.set_lock(path, tmpdir, lock_timeout)
 134             yield
 135         finally:
 136             self.unlock()
 137 
 138     def set_lock(self, path, tmpdir, lock_timeout=None):
 139         '''
 140         Create a lock file based on path with flock to prevent other processes
 141         using given path.
 142         Please note that currently file locking only works when it's executed by
 143         the same user, I.E single user scenarios
 144 
 145         :kw path: Path (file) to lock
 146         :kw tmpdir: Path where to place the temporary .lock file
 147         :kw lock_timeout:
 148             Wait n seconds for lock acquisition, fail if timeout is reached.
 149             0 = Do not wait, fail if lock cannot be acquired immediately,
 150             Default is None, wait indefinitely until lock is released.
 151         :returns: True
 152         '''
 153         lock_path = os.path.join(tmpdir, 'ansible-{0}.lock'.format(os.path.basename(path)))
 154         l_wait = 0.1
 155         r_exception = IOError
 156         if sys.version_info[0] == 3:
 157             r_exception = BlockingIOError
 158 
 159         self.lockfd = open(lock_path, 'w')
 160 
 161         if lock_timeout <= 0:
 162             fcntl.flock(self.lockfd, fcntl.LOCK_EX | fcntl.LOCK_NB)
 163             os.chmod(lock_path, stat.S_IWRITE | stat.S_IREAD)
 164             return True
 165 
 166         if lock_timeout:
 167             e_secs = 0
 168             while e_secs < lock_timeout:
 169                 try:
 170                     fcntl.flock(self.lockfd, fcntl.LOCK_EX | fcntl.LOCK_NB)
 171                     os.chmod(lock_path, stat.S_IWRITE | stat.S_IREAD)
 172                     return True
 173                 except r_exception:
 174                     time.sleep(l_wait)
 175                     e_secs += l_wait
 176                     continue
 177 
 178             self.lockfd.close()
 179             raise LockTimeout('{0} sec'.format(lock_timeout))
 180 
 181         fcntl.flock(self.lockfd, fcntl.LOCK_EX)
 182         os.chmod(lock_path, stat.S_IWRITE | stat.S_IREAD)
 183 
 184         return True
 185 
 186     def unlock(self):
 187         '''
 188         Make sure lock file is available for everyone and Unlock the file descriptor
 189         locked by set_lock
 190 
 191         :returns: True
 192         '''
 193         if not self.lockfd:
 194             return True
 195 
 196         try:
 197             fcntl.flock(self.lockfd, fcntl.LOCK_UN)
 198             self.lockfd.close()
 199         except ValueError:  # file wasn't opened, let context manager fail gracefully
 200             pass
 201 
 202         return True
