  17 def get_distribution():
  18     '''
  19     Return the name of the distribution the module is running on
  20 
  21     :rtype: NativeString or None
  22     :returns: Name of the distribution the module is running on
  23 
  24     This function attempts to determine what Linux distribution the code is running on and return
  25     a string representing that value.  If the distribution cannot be determined, it returns
  26     ``OtherLinux``.  If not run on Linux it returns None.
  27     '''
  28     distribution = None
  29 
  30     if platform.system() == 'Linux':
  31         distribution = distro.id().capitalize()
  32 
  33         if distribution == 'Amzn':
  34             distribution = 'Amazon'
  35         elif distribution == 'Rhel':
  36             distribution = 'Redhat'
  37         elif not distribution:
  38             distribution = 'OtherLinux'
  39 
  40     return distribution

  43 def get_distribution_version():
  44     '''
  45     Get the version of the Linux distribution the code is running on
  46 
  47     :rtype: NativeString or None
  48     :returns: A string representation of the version of the distribution. If it cannot determine
  49         the version, it returns empty string. If this is not run on a Linux machine it returns None
  50     '''
  51     version = None
  52 
  53     needs_best_version = frozenset((
  54         u'centos',
  55         u'debian',
  56     ))
  57 
  58     if platform.system() == 'Linux':
  59         version = distro.version()
  60         distro_id = distro.id()
  61 
  62         if version is not None:
  63             if distro_id in needs_best_version:
  64                 version_best = distro.version(best=True)
  65 
  66                 # CentoOS maintainers believe only the major version is appropriate
  67                 # but Ansible users desire minor version information, e.g., 7.5.
  68                 # https://github.com/ansible/ansible/issues/50141#issuecomment-449452781
  69                 if distro_id == u'centos':
  70                     version = u'.'.join(version_best.split(u'.')[:2])
  71 
  72                 # Debian does not include minor version in /etc/os-release.
  73                 # Bug report filed upstream requesting this be added to /etc/os-release
  74                 # https://bugs.debian.org/cgi-bin/bugreport.cgi?bug=931197
  75                 if distro_id == u'debian':
  76                     version = version_best
  77 
  78         else:
  79             version = u''
  80 
  81     return version

  84 def get_distribution_codename():
  85     '''
  86     Return the code name for this Linux Distribution
  87 
  88     :rtype: NativeString or None
  89     :returns: A string representation of the distribution's codename or None if not a Linux distro
  90     '''
  91     codename = None
  92     if platform.system() == 'Linux':
  93         # Until this gets merged and we update our bundled copy of distro:
  94         # https://github.com/nir0s/distro/pull/230
  95         # Fixes Fedora 28+ not having a code name and Ubuntu Xenial Xerus needing to be "xenial"
  96         os_release_info = distro.os_release_info()
  97         codename = os_release_info.get('version_codename')
  98 
  99         if codename is None:
 100             codename = os_release_info.get('ubuntu_codename')
 101 
 102         if codename is None and distro.id() == 'ubuntu':
 103             lsb_release_info = distro.lsb_release_info()
 104             codename = lsb_release_info.get('codename')
 105 
 106         if codename is None:
 107             codename = distro.codename()
 108             if codename == u'':
 109                 codename = None
 110 
 111     return codename

 114 def get_platform_subclass(cls):
 115     '''
 116     Finds a subclass implementing desired functionality on the platform the code is running on
 117 
 118     :arg cls: Class to find an appropriate subclass for
 119     :returns: A class that implements the functionality on this platform
 120 
 121     Some Ansible modules have different implementations depending on the platform they run on.  This
 122     function is used to select between the various implementations and choose one.  You can look at
 123     the implementation of the Ansible :ref:`User module<user_module>` module for an example of how to use this.
 124 
 125     This function replaces ``basic.load_platform_subclass()``.  When you port code, you need to
 126     change the callers to be explicit about instantiating the class.  For instance, code in the
 127     Ansible User module changed from::
 128 
 129     .. code-block:: python
 130 
 131         # Old
 132         class User:
 133             def __new__(cls, args, kwargs):
 134                 return load_platform_subclass(User, args, kwargs)
 135 
 136         # New
 137         class User:
 138             def __new__(cls, *args, **kwargs):
 139                 new_cls = get_platform_subclass(User)
 140                 return super(cls, new_cls).__new__(new_cls)
 141     '''
 142 
 143     this_platform = platform.system()
 144     distribution = get_distribution()
 145     subclass = None
 146 
 147     # get the most specific superclass for this platform
 148     if distribution is not None:
 149         for sc in get_all_subclasses(cls):
 150             if sc.distribution is not None and sc.distribution == distribution and sc.platform == this_platform:
 151                 subclass = sc
 152     if subclass is None:
 153         for sc in get_all_subclasses(cls):
 154             if sc.platform == this_platform and sc.distribution is None:
 155                 subclass = sc
 156     if subclass is None:
 157         subclass = cls
 158 
 159     return subclass
