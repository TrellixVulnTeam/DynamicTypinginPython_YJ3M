  14 def get_all_subclasses(cls):
  15     '''
  16     Recursively search and find all subclasses of a given class
  17 
  18     :arg cls: A python class
  19     :rtype: set
  20     :returns: The set of python classes which are the subclasses of `cls`.
  21 
  22     In python, you can use a class's :py:meth:`__subclasses__` method to determine what subclasses
  23     of a class exist.  However, `__subclasses__` only goes one level deep.  This function searches
  24     each child class's `__subclasses__` method to find all of the descendent classes.  It then
  25     returns an iterable of the descendent classes.
  26     '''
  27     # Retrieve direct subclasses
  28     subclasses = set(cls.__subclasses__())
  29     to_visit = list(subclasses)
  30     # Then visit all subclasses
  31     while to_visit:
  32         for sc in to_visit:
  33             # The current class is now visited, so remove it from list
  34             to_visit.remove(sc)
  35             # Appending all subclasses to visit and keep a reference of available class
  36             for ssc in sc.__subclasses__():
  37                 if ssc not in subclasses:
  38                     to_visit.append(ssc)
  39                     subclasses.add(ssc)
  40     return subclasses
