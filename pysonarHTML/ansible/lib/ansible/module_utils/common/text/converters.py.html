  33 def to_bytes(obj, encoding='utf-8', errors=None, nonstring='simplerepr'):
  34     """Make sure that a string is a byte string
  35 
  36     :arg obj: An object to make sure is a byte string.  In most cases this
  37         will be either a text string or a byte string.  However, with
  38         ``nonstring='simplerepr'``, this can be used as a traceback-free
  39         version of ``str(obj)``.
  40     :kwarg encoding: The encoding to use to transform from a text string to
  41         a byte string.  Defaults to using 'utf-8'.
  42     :kwarg errors: The error handler to use if the text string is not
  43         encodable using the specified encoding.  Any valid `codecs error
  44         handler <https://docs.python.org/2/library/codecs.html#codec-base-classes>`_
  45         may be specified. There are three additional error strategies
  46         specifically aimed at helping people to port code.  The first two are:
  47 
  48             :surrogate_or_strict: Will use ``surrogateescape`` if it is a valid
  49                 handler, otherwise it will use ``strict``
  50             :surrogate_or_replace: Will use ``surrogateescape`` if it is a valid
  51                 handler, otherwise it will use ``replace``.
  52 
  53         Because ``surrogateescape`` was added in Python3 this usually means that
  54         Python3 will use ``surrogateescape`` and Python2 will use the fallback
  55         error handler. Note that the code checks for ``surrogateescape`` when the
  56         module is imported.  If you have a backport of ``surrogateescape`` for
  57         Python2, be sure to register the error handler prior to importing this
  58         module.
  59 
  60         The last error handler is:
  61 
  62             :surrogate_then_replace: Will use ``surrogateescape`` if it is a valid
  63                 handler.  If encoding with ``surrogateescape`` would traceback,
  64                 surrogates are first replaced with a replacement characters
  65                 and then the string is encoded using ``replace`` (which replaces
  66                 the rest of the nonencodable bytes).  If ``surrogateescape`` is
  67                 not present it will simply use ``replace``.  (Added in Ansible 2.3)
  68                 This strategy is designed to never traceback when it attempts
  69                 to encode a string.
  70 
  71         The default until Ansible-2.2 was ``surrogate_or_replace``
  72         From Ansible-2.3 onwards, the default is ``surrogate_then_replace``.
  73 
  74     :kwarg nonstring: The strategy to use if a nonstring is specified in
  75         ``obj``.  Default is 'simplerepr'.  Valid values are:
  76 
  77         :simplerepr: The default.  This takes the ``str`` of the object and
  78             then returns the bytes version of that string.
  79         :empty: Return an empty byte string
  80         :passthru: Return the object passed in
  81         :strict: Raise a :exc:`TypeError`
  82 
  83     :returns: Typically this returns a byte string.  If a nonstring object is
  84         passed in this may be a different type depending on the strategy
  85         specified by nonstring.  This will never return a text string.
  86 
  87     .. note:: If passed a byte string, this function does not check that the
  88         string is valid in the specified encoding.  If it's important that the
  89         byte string is in the specified encoding do::
  90 
  91             encoded_string = to_bytes(to_text(input_string, 'latin-1'), 'utf-8')
  92 
  93     .. version_changed:: 2.3
  94 
  95         Added the ``surrogate_then_replace`` error handler and made it the default error handler.
  96     """
  97     if isinstance(obj, binary_type):
  98         return obj
  99 
 100     # We're given a text string
 101     # If it has surrogates, we know because it will decode
 102     original_errors = errors
 103     if errors in _COMPOSED_ERROR_HANDLERS:
 104         if HAS_SURROGATEESCAPE:
 105             errors = 'surrogateescape'
 106         elif errors == 'surrogate_or_strict':
 107             errors = 'strict'
 108         else:
 109             errors = 'replace'
 110 
 111     if isinstance(obj, text_type):
 112         try:
 113             # Try this first as it's the fastest
 114             return obj.encode(encoding, errors)
 115         except UnicodeEncodeError:
 116             if original_errors in (None, 'surrogate_then_replace'):
 117                 # We should only reach this if encoding was non-utf8 original_errors was
 118                 # surrogate_then_escape and errors was surrogateescape
 119 
 120                 # Slow but works
 121                 return_string = obj.encode('utf-8', 'surrogateescape')
 122                 return_string = return_string.decode('utf-8', 'replace')
 123                 return return_string.encode(encoding, 'replace')
 124             raise
 125 
 126     # Note: We do these last even though we have to call to_bytes again on the
 127     # value because we're optimizing the common case
 128     if nonstring == 'simplerepr':
 129         try:
 130             value = str(obj)
 131         except UnicodeError:
 132             try:
 133                 value = repr(obj)
 134             except UnicodeError:
 135                 # Giving up
 136                 return to_bytes('')
 137     elif nonstring == 'passthru':
 138         return obj
 139     elif nonstring == 'empty':
 140         # python2.4 doesn't have b''
 141         return to_bytes('')
 142     elif nonstring == 'strict':
 143         raise TypeError('obj must be a string type')
 144     else:
 145         raise TypeError('Invalid value %s for to_bytes\' nonstring parameter' % nonstring)
 146 
 147     return to_bytes(value, encoding, errors)

 150 def to_text(obj, encoding='utf-8', errors=None, nonstring='simplerepr'):
 151     """Make sure that a string is a text string
 152 
 153     :arg obj: An object to make sure is a text string.  In most cases this
 154         will be either a text string or a byte string.  However, with
 155         ``nonstring='simplerepr'``, this can be used as a traceback-free
 156         version of ``str(obj)``.
 157     :kwarg encoding: The encoding to use to transform from a byte string to
 158         a text string.  Defaults to using 'utf-8'.
 159     :kwarg errors: The error handler to use if the byte string is not
 160         decodable using the specified encoding.  Any valid `codecs error
 161         handler <https://docs.python.org/2/library/codecs.html#codec-base-classes>`_
 162         may be specified.   We support three additional error strategies
 163         specifically aimed at helping people to port code:
 164 
 165             :surrogate_or_strict: Will use surrogateescape if it is a valid
 166                 handler, otherwise it will use strict
 167             :surrogate_or_replace: Will use surrogateescape if it is a valid
 168                 handler, otherwise it will use replace.
 169             :surrogate_then_replace: Does the same as surrogate_or_replace but
 170                 `was added for symmetry with the error handlers in
 171                 :func:`ansible.module_utils._text.to_bytes` (Added in Ansible 2.3)
 172 
 173         Because surrogateescape was added in Python3 this usually means that
 174         Python3 will use `surrogateescape` and Python2 will use the fallback
 175         error handler. Note that the code checks for surrogateescape when the
 176         module is imported.  If you have a backport of `surrogateescape` for
 177         python2, be sure to register the error handler prior to importing this
 178         module.
 179 
 180         The default until Ansible-2.2 was `surrogate_or_replace`
 181         In Ansible-2.3 this defaults to `surrogate_then_replace` for symmetry
 182         with :func:`ansible.module_utils._text.to_bytes` .
 183     :kwarg nonstring: The strategy to use if a nonstring is specified in
 184         ``obj``.  Default is 'simplerepr'.  Valid values are:
 185 
 186         :simplerepr: The default.  This takes the ``str`` of the object and
 187             then returns the text version of that string.
 188         :empty: Return an empty text string
 189         :passthru: Return the object passed in
 190         :strict: Raise a :exc:`TypeError`
 191 
 192     :returns: Typically this returns a text string.  If a nonstring object is
 193         passed in this may be a different type depending on the strategy
 194         specified by nonstring.  This will never return a byte string.
 195         From Ansible-2.3 onwards, the default is `surrogate_then_replace`.
 196 
 197     .. version_changed:: 2.3
 198 
 199         Added the surrogate_then_replace error handler and made it the default error handler.
 200     """
 201     if isinstance(obj, text_type):
 202         return obj
 203 
 204     if errors in _COMPOSED_ERROR_HANDLERS:
 205         if HAS_SURROGATEESCAPE:
 206             errors = 'surrogateescape'
 207         elif errors == 'surrogate_or_strict':
 208             errors = 'strict'
 209         else:
 210             errors = 'replace'
 211 
 212     if isinstance(obj, binary_type):
 213         # Note: We don't need special handling for surrogate_then_replace
 214         # because all bytes will either be made into surrogates or are valid
 215         # to decode.
 216         return obj.decode(encoding, errors)
 217 
 218     # Note: We do these last even though we have to call to_text again on the
 219     # value because we're optimizing the common case
 220     if nonstring == 'simplerepr':
 221         try:
 222             value = str(obj)
 223         except UnicodeError:
 224             try:
 225                 value = repr(obj)
 226             except UnicodeError:
 227                 # Giving up
 228                 return u''
 229     elif nonstring == 'passthru':
 230         return obj
 231     elif nonstring == 'empty':
 232         return u''
 233     elif nonstring == 'strict':
 234         raise TypeError('obj must be a string type')
 235     else:
 236         raise TypeError('Invalid value %s for to_text\'s nonstring parameter' % nonstring)
 237 
 238     return to_text(value, encoding, errors)

 241 #: :py:func:`to_native`
 242 #:      Transform a variable into the native str type for the python version
 243 #:
 244 #:      On Python2, this is an alias for
 245 #:      :func:`~ansible.module_utils.to_bytes`.  On Python3 it is an alias for
 246 #:      :func:`~ansible.module_utils.to_text`.  It makes it easier to
 247 #:      transform a variable into the native str type for the python version
 248 #:      the code is running on.  Use this when constructing the message to
 249 #:      send to exceptions or when dealing with an API that needs to take
 250 #:      a native string.  Example::
 251 #:
 252 #:          try:
 253 #:              1//0
 254 #:          except ZeroDivisionError as e:
 255 #:              raise MyException('Encountered and error: %s' % to_native(e))
 256 if PY3:
 257     to_native = to_text
 258 else:
 259     to_native = to_bytes

 262 def _json_encode_fallback(obj):
 263     if isinstance(obj, Set):
 264         return list(obj)
 265     elif isinstance(obj, datetime.datetime):
 266         return obj.isoformat()
 267     raise TypeError("Cannot json serialize %s" % to_native(obj))

 270 def jsonify(data, **kwargs):
 271     for encoding in ("utf-8", "latin-1"):
 272         try:
 273             return json.dumps(data, encoding=encoding, default=_json_encode_fallback, **kwargs)
 274         # Old systems using old simplejson module does not support encoding keyword.
 275         except TypeError:
 276             try:
 277                 new_data = container_to_text(data, encoding=encoding)
 278             except UnicodeDecodeError:
 279                 continue
 280             return json.dumps(new_data, default=_json_encode_fallback, **kwargs)
 281         except UnicodeDecodeError:
 282             continue
 283     raise UnicodeError('Invalid unicode encoding encountered')

 286 def container_to_bytes(d, encoding='utf-8', errors='surrogate_or_strict'):
 287     ''' Recursively convert dict keys and values to byte str
 288 
 289         Specialized for json return because this only handles, lists, tuples,
 290         and dict container types (the containers that the json module returns)
 291     '''
 292 
 293     if isinstance(d, text_type):
 294         return to_bytes(d, encoding=encoding, errors=errors)
 295     elif isinstance(d, dict):
 296         return dict(container_to_bytes(o, encoding, errors) for o in iteritems(d))
 297     elif isinstance(d, list):
 298         return [container_to_bytes(o, encoding, errors) for o in d]
 299     elif isinstance(d, tuple):
 300         return tuple(container_to_bytes(o, encoding, errors) for o in d)
 301     else:
 302         return d

 305 def container_to_text(d, encoding='utf-8', errors='surrogate_or_strict'):
 306     """Recursively convert dict keys and values to text str
 307 
 308     Specialized for json return because this only handles, lists, tuples,
 309     and dict container types (the containers that the json module returns)
 310     """
 311 
 312     if isinstance(d, binary_type):
 313         # Warning, can traceback
 314         return to_text(d, encoding=encoding, errors=errors)
 315     elif isinstance(d, dict):
 316         return dict(container_to_text(o, encoding, errors) for o in iteritems(d))
 317     elif isinstance(d, list):
 318         return [container_to_text(o, encoding, errors) for o in d]
 319     elif isinstance(d, tuple):
 320         return tuple(container_to_text(o, encoding, errors) for o in d)
 321     else:
 322         return d
