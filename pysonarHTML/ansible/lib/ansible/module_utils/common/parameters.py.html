  46 def _return_datastructure_name(obj):
  47     """ Return native stringified values from datastructures.
  48 
  49     For use with removing sensitive values pre-jsonification."""
  50     if isinstance(obj, (text_type, binary_type)):
  51         if obj:
  52             yield to_native(obj, errors='surrogate_or_strict')
  53         return
  54     elif isinstance(obj, Mapping):
  55         for element in obj.items():
  56             for subelement in _return_datastructure_name(element[1]):
  57                 yield subelement
  58     elif is_iterable(obj):
  59         for element in obj:
  60             for subelement in _return_datastructure_name(element):
  61                 yield subelement
  62     elif isinstance(obj, (bool, NoneType)):
  63         # This must come before int because bools are also ints
  64         return
  65     elif isinstance(obj, tuple(list(integer_types) + [float])):
  66         yield to_native(obj, nonstring='simplerepr')
  67     else:
  68         raise TypeError('Unknown parameter type: %s' % (type(obj)))

  71 def list_no_log_values(argument_spec, params):
  72     """Return set of no log values
  73 
  74     :arg argument_spec: An argument spec dictionary from a module
  75     :arg params: Dictionary of all module parameters
  76 
  77     :returns: Set of strings that should be hidden from output::
  78 
  79         {'secret_dict_value', 'secret_list_item_one', 'secret_list_item_two', 'secret_string'}
  80     """
  81 
  82     no_log_values = set()
  83     for arg_name, arg_opts in argument_spec.items():
  84         if arg_opts.get('no_log', False):
  85             # Find the value for the no_log'd param
  86             no_log_object = params.get(arg_name, None)
  87 
  88             if no_log_object:
  89                 try:
  90                     no_log_values.update(_return_datastructure_name(no_log_object))
  91                 except TypeError as e:
  92                     raise TypeError('Failed to convert "%s": %s' % (arg_name, to_native(e)))
  93 
  94         # Get no_log values from suboptions
  95         sub_argument_spec = arg_opts.get('options')
  96         if sub_argument_spec is not None:
  97             wanted_type = arg_opts.get('type')
  98             sub_parameters = params.get(arg_name)
  99 
 100             if sub_parameters is not None:
 101                 if wanted_type == 'dict' or (wanted_type == 'list' and arg_opts.get('elements', '') == 'dict'):
 102                     # Sub parameters can be a dict or list of dicts. Ensure parameters are always a list.
 103                     if not isinstance(sub_parameters, list):
 104                         sub_parameters = [sub_parameters]
 105 
 106                     for sub_param in sub_parameters:
 107                         # Validate dict fields in case they came in as strings
 108 
 109                         if isinstance(sub_param, string_types):
 110                             sub_param = check_type_dict(sub_param)
 111 
 112                         if not isinstance(sub_param, Mapping):
 113                             raise TypeError("Value '{1}' in the sub parameter field '{0}' must by a {2}, "
 114                                             "not '{1.__class__.__name__}'".format(arg_name, sub_param, wanted_type))
 115 
 116                         no_log_values.update(list_no_log_values(sub_argument_spec, sub_param))
 117 
 118     return no_log_values

 121 def list_deprecations(argument_spec, params, prefix=''):
 122     """Return a list of deprecations
 123 
 124     :arg argument_spec: An argument spec dictionary from a module
 125     :arg params: Dictionary of all module parameters
 126 
 127     :returns: List of dictionaries containing a message and version in which
 128         the deprecated parameter will be removed, or an empty list::
 129 
 130             [{'msg': "Param 'deptest' is deprecated. See the module docs for more information", 'version': '2.9'}]
 131     """
 132 
 133     deprecations = []
 134     for arg_name, arg_opts in argument_spec.items():
 135         if arg_name in params:
 136             if prefix:
 137                 sub_prefix = '%s["%s"]' % (prefix, arg_name)
 138             else:
 139                 sub_prefix = arg_name
 140             if arg_opts.get('removed_at_date') is not None:
 141                 deprecations.append({
 142                     'msg': "Param '%s' is deprecated. See the module docs for more information" % sub_prefix,
 143                     'date': arg_opts.get('removed_at_date'),
 144                     'collection_name': arg_opts.get('removed_from_collection'),
 145                 })
 146             elif arg_opts.get('removed_in_version') is not None:
 147                 deprecations.append({
 148                     'msg': "Param '%s' is deprecated. See the module docs for more information" % sub_prefix,
 149                     'version': arg_opts.get('removed_in_version'),
 150                     'collection_name': arg_opts.get('removed_from_collection'),
 151                 })
 152             # Check sub-argument spec
 153             sub_argument_spec = arg_opts.get('options')
 154             if sub_argument_spec is not None:
 155                 sub_arguments = params[arg_name]
 156                 if isinstance(sub_arguments, Mapping):
 157                     sub_arguments = [sub_arguments]
 158                 if isinstance(sub_arguments, list):
 159                     for sub_params in sub_arguments:
 160                         if isinstance(sub_params, Mapping):
 161                             deprecations.extend(list_deprecations(sub_argument_spec, sub_params, prefix=sub_prefix))
 162 
 163     return deprecations

 166 def handle_aliases(argument_spec, params, alias_warnings=None):
 167     """Return a two item tuple. The first is a dictionary of aliases, the second is
 168     a list of legal inputs.
 169 
 170     If a list is provided to the alias_warnings parameter, it will be filled with tuples
 171     (option, alias) in every case where both an option and its alias are specified.
 172     """
 173 
 174     legal_inputs = ['_ansible_%s' % k for k in PASS_VARS]
 175     aliases_results = {}  # alias:canon
 176 
 177     for (k, v) in argument_spec.items():
 178         legal_inputs.append(k)
 179         aliases = v.get('aliases', None)
 180         default = v.get('default', None)
 181         required = v.get('required', False)
 182         if default is not None and required:
 183             # not alias specific but this is a good place to check this
 184             raise ValueError("internal error: required and default are mutually exclusive for %s" % k)
 185         if aliases is None:
 186             continue
 187         if not is_iterable(aliases) or isinstance(aliases, (binary_type, text_type)):
 188             raise TypeError('internal error: aliases must be a list or tuple')
 189         for alias in aliases:
 190             legal_inputs.append(alias)
 191             aliases_results[alias] = k
 192             if alias in params:
 193                 if k in params and alias_warnings is not None:
 194                     alias_warnings.append((k, alias))
 195                 params[k] = params[alias]
 196 
 197     return aliases_results, legal_inputs
