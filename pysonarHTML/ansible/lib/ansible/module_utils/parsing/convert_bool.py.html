   4 from __future__ import (absolute_import, division, print_function)
   5 __metaclass__ = type

   7 from ansible.module_utils.six import binary_type, text_type
   8 from ansible.module_utils._text import to_text
   9 
  10 
  11 BOOLEANS_TRUE = frozenset(('y', 'yes', 'on', '1', 'true', 't', 1, 1.0, True))
  12 BOOLEANS_FALSE = frozenset(('n', 'no', 'off', '0', 'false', 'f', 0, 0.0, False))
  13 BOOLEANS = BOOLEANS_TRUE.union(BOOLEANS_FALSE)
  14 
  15 
  16 def boolean(value, strict=True):
  17     if isinstance(value, bool):
  18         return value
  19 
  20     normalized_value = value
  21     if isinstance(value, (text_type, binary_type)):
  22         normalized_value = to_text(value, errors='surrogate_or_strict').lower().strip()
  23 
  24     if normalized_value in BOOLEANS_TRUE:
  25         return True
  26     elif normalized_value in BOOLEANS_FALSE or not strict:
  27         return False
  28 
  29     raise TypeError("The value '%s' is not a valid boolean.  Valid booleans include: %s" % (to_text(value), ', '.join(repr(i) for i in BOOLEANS)))
