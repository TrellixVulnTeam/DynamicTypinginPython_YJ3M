 195     def _dnsname_match(dn, hostname):
 196         """Matching according to RFC 6125, section 6.4.3
 197 
 198         - Hostnames are compared lower case.
 199         - For IDNA, both dn and hostname must be encoded as IDN A-label (ACE).
 200         - Partial wildcards like 'www*.example.org', multiple wildcards, sole
 201           wildcard or wildcards in labels other then the left-most label are not
 202           supported and a CertificateError is raised.
 203         - A wildcard must match at least one character.
 204         """
 205         if not dn:
 206             return False
 207 
 208         wildcards = dn.count('*')
 209         # speed up common case w/o wildcards
 210         if not wildcards:
 211             return dn.lower() == hostname.lower()
 212 
 213         if wildcards > 1:
 214             # Divergence .format() to percent formatting for Python < 2.6
 215             raise CertificateError(
 216                 "too many wildcards in certificate DNS name: %s" % repr(dn))
 217 
 218         dn_leftmost, sep, dn_remainder = dn.partition('.')
 219 
 220         if '*' in dn_remainder:
 221             # Only match wildcard in leftmost segment.
 222             # Divergence .format() to percent formatting for Python < 2.6
 223             raise CertificateError(
 224                 "wildcard can only be present in the leftmost label: "
 225                 "%s." % repr(dn))
 226 
 227         if not sep:
 228             # no right side
 229             # Divergence .format() to percent formatting for Python < 2.6
 230             raise CertificateError(
 231                 "sole wildcard without additional labels are not support: "
 232                 "%s." % repr(dn))
 233 
 234         if dn_leftmost != '*':
 235             # no partial wildcard matching
 236             # Divergence .format() to percent formatting for Python < 2.6
 237             raise CertificateError(
 238                 "partial wildcards in leftmost label are not supported: "
 239                 "%s." % repr(dn))
 240 
 241         hostname_leftmost, sep, hostname_remainder = hostname.partition('.')
 242         if not hostname_leftmost or not sep:
 243             # wildcard must match at least one char
 244             return False
 245         return dn_remainder.lower() == hostname_remainder.lower()

 247     def _inet_paton(ipname):
 248         """Try to convert an IP address to packed binary form
 249 
 250         Supports IPv4 addresses on all platforms and IPv6 on platforms with IPv6
 251         support.
 252         """
 253         # inet_aton() also accepts strings like '1'
 254         # Divergence: We make sure we have native string type for all python versions
 255         try:
 256             b_ipname = to_bytes(ipname, errors='strict')
 257         except UnicodeError:
 258             raise ValueError("%s must be an all-ascii string." % repr(ipname))
 259 
 260         # Set ipname in native string format
 261         if sys.version_info < (3,):
 262             n_ipname = b_ipname
 263         else:
 264             n_ipname = ipname
 265 
 266         if n_ipname.count('.') == 3:
 267             try:
 268                 return socket.inet_aton(n_ipname)
 269             # Divergence: OSError on late python3.  socket.error earlier.
 270             # Null bytes generate ValueError on python3(we want to raise
 271             # ValueError anyway), TypeError # earlier
 272             except (OSError, socket.error, TypeError):
 273                 pass
 274 
 275         try:
 276             return socket.inet_pton(socket.AF_INET6, n_ipname)
 277         # Divergence: OSError on late python3.  socket.error earlier.
 278         # Null bytes generate ValueError on python3(we want to raise
 279         # ValueError anyway), TypeError # earlier
 280         except (OSError, socket.error, TypeError):
 281             # Divergence .format() to percent formatting for Python < 2.6
 282             raise ValueError("%s is neither an IPv4 nor an IP6 "
 283                              "address." % repr(ipname))
 284         except AttributeError:
 285             # AF_INET6 not available
 286             pass
 287 
 288         # Divergence .format() to percent formatting for Python < 2.6
 289         raise ValueError("%s is not an IPv4 address." % repr(ipname))

 291     def _ipaddress_match(ipname, host_ip):
 292         """Exact matching of IP addresses.
 293 
 294         RFC 6125 explicitly doesn't define an algorithm for this
 295         (section 1.7.2 - "Out of Scope").
 296         """
 297         # OpenSSL may add a trailing newline to a subjectAltName's IP address
 298         ip = _inet_paton(ipname.rstrip())
 299         return ip == host_ip

 301     def match_hostname(cert, hostname):
 302         """Verify that *cert* (in decoded format as returned by
 303         SSLSocket.getpeercert()) matches the *hostname*.  RFC 2818 and RFC 6125
 304         rules are followed.
 305 
 306         The function matches IP addresses rather than dNSNames if hostname is a
 307         valid ipaddress string. IPv4 addresses are supported on all platforms.
 308         IPv6 addresses are supported on platforms with IPv6 support (AF_INET6
 309         and inet_pton).
 310 
 311         CertificateError is raised on failure. On success, the function
 312         returns nothing.
 313         """
 314         if not cert:
 315             raise ValueError("empty or no certificate, match_hostname needs a "
 316                              "SSL socket or SSL context with either "
 317                              "CERT_OPTIONAL or CERT_REQUIRED")
 318         try:
 319             # Divergence: Deal with hostname as bytes
 320             host_ip = _inet_paton(to_text(hostname, errors='strict'))
 321         except UnicodeError:
 322             # Divergence: Deal with hostname as byte strings.
 323             # IP addresses should be all ascii, so we consider it not
 324             # an IP address if this fails
 325             host_ip = None
 326         except ValueError:
 327             # Not an IP address (common case)
 328             host_ip = None
 329         dnsnames = []
 330         san = cert.get('subjectAltName', ())
 331         for key, value in san:
 332             if key == 'DNS':
 333                 if host_ip is None and _dnsname_match(value, hostname):
 334                     return
 335                 dnsnames.append(value)
 336             elif key == 'IP Address':
 337                 if host_ip is not None and _ipaddress_match(value, host_ip):
 338                     return
 339                 dnsnames.append(value)
 340         if not dnsnames:
 341             # The subject is only checked when there is no dNSName entry
 342             # in subjectAltName
 343             for sub in cert.get('subject', ()):
 344                 for key, value in sub:
 345                     # XXX according to RFC 2818, the most specific Common Name
 346                     # must be used.
 347                     if key == 'commonName':
 348                         if _dnsname_match(value, hostname):
 349                             return
 350                         dnsnames.append(value)
 351         if len(dnsnames) > 1:
 352             raise CertificateError("hostname %r doesn't match either of %s" % (hostname, ', '.join(map(repr, dnsnames))))
 353         elif len(dnsnames) == 1:
 354             raise CertificateError("hostname %r doesn't match %r" % (hostname, dnsnames[0]))
 355         else:
 356             raise CertificateError("no appropriate commonName or subjectAltName fields were found")

 574 def generic_urlparse(parts):
 575     '''
 576     Returns a dictionary of url parts as parsed by urlparse,
 577     but accounts for the fact that older versions of that
 578     library do not support named attributes (ie. .netloc)
 579     '''
 580     generic_parts = ParseResultDottedDict()
 581     if hasattr(parts, 'netloc'):
 582         # urlparse is newer, just read the fields straight
 583         # from the parts object
 584         generic_parts['scheme'] = parts.scheme
 585         generic_parts['netloc'] = parts.netloc
 586         generic_parts['path'] = parts.path
 587         generic_parts['params'] = parts.params
 588         generic_parts['query'] = parts.query
 589         generic_parts['fragment'] = parts.fragment
 590         generic_parts['username'] = parts.username
 591         generic_parts['password'] = parts.password
 592         hostname = parts.hostname
 593         if hostname and hostname[0] == '[' and '[' in parts.netloc and ']' in parts.netloc:
 594             # Py2.6 doesn't parse IPv6 addresses correctly
 595             hostname = parts.netloc.split(']')[0][1:].lower()
 596         generic_parts['hostname'] = hostname
 597 
 598         try:
 599             port = parts.port
 600         except ValueError:
 601             # Py2.6 doesn't parse IPv6 addresses correctly
 602             netloc = parts.netloc.split('@')[-1].split(']')[-1]
 603             if ':' in netloc:
 604                 port = netloc.split(':')[1]
 605                 if port:
 606                     port = int(port)
 607             else:
 608                 port = None
 609         generic_parts['port'] = port
 610     else:
 611         # we have to use indexes, and then parse out
 612         # the other parts not supported by indexing
 613         generic_parts['scheme'] = parts[0]
 614         generic_parts['netloc'] = parts[1]
 615         generic_parts['path'] = parts[2]
 616         generic_parts['params'] = parts[3]
 617         generic_parts['query'] = parts[4]
 618         generic_parts['fragment'] = parts[5]
 619         # get the username, password, etc.
 620         try:
 621             netloc_re = re.compile(r'^((?:\w)+(?::(?:\w)+)?@)?([A-Za-z0-9.-]+)(:\d+)?$')
 622             match = netloc_re.match(parts[1])
 623             auth = match.group(1)
 624             hostname = match.group(2)
 625             port = match.group(3)
 626             if port:
 627                 # the capture group for the port will include the ':',
 628                 # so remove it and convert the port to an integer
 629                 port = int(port[1:])
 630             if auth:
 631                 # the capture group above includes the @, so remove it
 632                 # and then split it up based on the first ':' found
 633                 auth = auth[:-1]
 634                 username, password = auth.split(':', 1)
 635             else:
 636                 username = password = None
 637             generic_parts['username'] = username
 638             generic_parts['password'] = password
 639             generic_parts['hostname'] = hostname
 640             generic_parts['port'] = port
 641         except Exception:
 642             generic_parts['username'] = None
 643             generic_parts['password'] = None
 644             generic_parts['hostname'] = parts[1]
 645             generic_parts['port'] = None
 646     return generic_parts

 668 def RedirectHandlerFactory(follow_redirects=None, validate_certs=True, ca_path=None):
 669     """This is a class factory that closes over the value of
 670     ``follow_redirects`` so that the RedirectHandler class has access to
 671     that value without having to use globals, and potentially cause problems
 672     where ``open_url`` or ``fetch_url`` are used multiple times in a module.
 673     """
 674 
 675     class RedirectHandler(urllib_request.HTTPRedirectHandler):
 676         """This is an implementation of a RedirectHandler to match the
 677         functionality provided by httplib2. It will utilize the value of
 678         ``follow_redirects`` that is passed into ``RedirectHandlerFactory``
 679         to determine how redirects should be handled in urllib2.
 680         """
 681 
 682         def redirect_request(self, req, fp, code, msg, hdrs, newurl):
 683             if not HAS_SSLCONTEXT:
 684                 handler = maybe_add_ssl_handler(newurl, validate_certs, ca_path=ca_path)
 685                 if handler:
 686                     urllib_request._opener.add_handler(handler)
 687 
 688             # Preserve urllib2 compatibility
 689             if follow_redirects == 'urllib2':
 690                 return urllib_request.HTTPRedirectHandler.redirect_request(self, req, fp, code, msg, hdrs, newurl)
 691 
 692             # Handle disabled redirects
 693             elif follow_redirects in ['no', 'none', False]:
 694                 raise urllib_error.HTTPError(newurl, code, msg, hdrs, fp)
 695 
 696             method = req.get_method()
 697 
 698             # Handle non-redirect HTTP status or invalid follow_redirects
 699             if follow_redirects in ['all', 'yes', True]:
 700                 if code < 300 or code >= 400:
 701                     raise urllib_error.HTTPError(req.get_full_url(), code, msg, hdrs, fp)
 702             elif follow_redirects == 'safe':
 703                 if code < 300 or code >= 400 or method not in ('GET', 'HEAD'):
 704                     raise urllib_error.HTTPError(req.get_full_url(), code, msg, hdrs, fp)
 705             else:
 706                 raise urllib_error.HTTPError(req.get_full_url(), code, msg, hdrs, fp)
 707 
 708             try:
 709                 # Python 2-3.3
 710                 data = req.get_data()
 711                 origin_req_host = req.get_origin_req_host()
 712             except AttributeError:
 713                 # Python 3.4+
 714                 data = req.data
 715                 origin_req_host = req.origin_req_host
 716 
 717             # Be conciliant with URIs containing a space
 718             newurl = newurl.replace(' ', '%20')
 719 
 720             # Suport redirect with payload and original headers
 721             if code in (307, 308):
 722                 # Preserve payload and headers
 723                 headers = req.headers
 724             else:
 725                 # Do not preserve payload and filter headers
 726                 data = None
 727                 headers = dict((k, v) for k, v in req.headers.items()
 728                                if k.lower() not in ("content-length", "content-type", "transfer-encoding"))
 729 
 730                 # http://tools.ietf.org/html/rfc7231#section-6.4.4
 731                 if code == 303 and method != 'HEAD':
 732                     method = 'GET'
 733 
 734                 # Do what the browsers do, despite standards...
 735                 # First, turn 302s into GETs.
 736                 if code == 302 and method != 'HEAD':
 737                     method = 'GET'
 738 
 739                 # Second, if a POST is responded to with a 301, turn it into a GET.
 740                 if code == 301 and method == 'POST':
 741                     method = 'GET'
 742 
 743             return RequestWithMethod(newurl,
 744                                      method=method,
 745                                      headers=headers,
 746                                      data=data,
 747                                      origin_req_host=origin_req_host,
 748                                      unverifiable=True,
 749                                      )
 750 
 751     return RedirectHandler

 795 class SSLValidationHandler(urllib_request.BaseHandler):
 796     '''
 797     A custom handler class for SSL validation.
 798 
 799     Based on:
 800     http://stackoverflow.com/questions/1087227/validate-ssl-certificates-with-python
 801     http://techknack.net/python-urllib2-handlers/
 802     '''
 803     CONNECT_COMMAND = "CONNECT %s:%s HTTP/1.0\r\n"
 804 
 805     def __init__(self, hostname, port, ca_path=None):
 806         self.hostname = hostname
 807         self.port = port
 808         self.ca_path = ca_path

 810     def get_ca_certs(self):
 811         # tries to find a valid CA cert in one of the
 812         # standard locations for the current distribution
 813 
 814         ca_certs = []
 815         cadata = bytearray()
 816         paths_checked = []
 817 
 818         if self.ca_path:
 819             paths_checked = [self.ca_path]
 820             with open(to_bytes(self.ca_path, errors='surrogate_or_strict'), 'rb') as f:
 821                 if HAS_SSLCONTEXT:
 822                     cadata.extend(
 823                         ssl.PEM_cert_to_DER_cert(
 824                             to_native(f.read(), errors='surrogate_or_strict')
 825                         )
 826                     )
 827                 else:
 828                     ca_certs.append(f.read())
 829             return ca_certs, cadata, paths_checked
 830 
 831         if not HAS_SSLCONTEXT:
 832             paths_checked.append('/etc/ssl/certs')
 833 
 834         system = to_text(platform.system(), errors='surrogate_or_strict')
 835         # build a list of paths to check for .crt/.pem files
 836         # based on the platform type
 837         if system == u'Linux':
 838             paths_checked.append('/etc/pki/ca-trust/extracted/pem')
 839             paths_checked.append('/etc/pki/tls/certs')
 840             paths_checked.append('/usr/share/ca-certificates/cacert.org')
 841         elif system == u'FreeBSD':
 842             paths_checked.append('/usr/local/share/certs')
 843         elif system == u'OpenBSD':
 844             paths_checked.append('/etc/ssl')
 845         elif system == u'NetBSD':
 846             ca_certs.append('/etc/openssl/certs')
 847         elif system == u'SunOS':
 848             paths_checked.append('/opt/local/etc/openssl/certs')
 849 
 850         # fall back to a user-deployed cert in a standard
 851         # location if the OS platform one is not available
 852         paths_checked.append('/etc/ansible')
 853 
 854         tmp_path = None
 855         if not HAS_SSLCONTEXT:
 856             tmp_fd, tmp_path = tempfile.mkstemp()
 857             atexit.register(atexit_remove_file, tmp_path)
 858 
 859         # Write the dummy ca cert if we are running on macOS
 860         if system == u'Darwin':
 861             if HAS_SSLCONTEXT:
 862                 cadata.extend(
 863                     ssl.PEM_cert_to_DER_cert(
 864                         to_native(b_DUMMY_CA_CERT, errors='surrogate_or_strict')
 865                     )
 866                 )
 867             else:
 868                 os.write(tmp_fd, b_DUMMY_CA_CERT)
 869             # Default Homebrew path for OpenSSL certs
 870             paths_checked.append('/usr/local/etc/openssl')
 871 
 872         # for all of the paths, find any  .crt or .pem files
 873         # and compile them into single temp file for use
 874         # in the ssl check to speed up the test
 875         for path in paths_checked:
 876             if os.path.exists(path) and os.path.isdir(path):
 877                 dir_contents = os.listdir(path)
 878                 for f in dir_contents:
 879                     full_path = os.path.join(path, f)
 880                     if os.path.isfile(full_path) and os.path.splitext(f)[1] in ('.crt', '.pem'):
 881                         try:
 882                             if full_path not in LOADED_VERIFY_LOCATIONS:
 883                                 with open(full_path, 'rb') as cert_file:
 884                                     b_cert = cert_file.read()
 885                                 if HAS_SSLCONTEXT:
 886                                     try:
 887                                         cadata.extend(
 888                                             ssl.PEM_cert_to_DER_cert(
 889                                                 to_native(b_cert, errors='surrogate_or_strict')
 890                                             )
 891                                         )
 892                                     except Exception:
 893                                         continue
 894                                 else:
 895                                     os.write(tmp_fd, b_cert)
 896                                     os.write(tmp_fd, b'\n')
 897                         except (OSError, IOError):
 898                             pass
 899 
 900         if HAS_SSLCONTEXT:
 901             default_verify_paths = ssl.get_default_verify_paths()
 902             paths_checked[:0] = [default_verify_paths.capath]
 903 
 904         return (tmp_path, cadata, paths_checked)

 922         '''

 953     def http_request(self, req):
 954         tmp_ca_cert_path, cadata, paths_checked = self.get_ca_certs()
 955 
 956         # Detect if 'no_proxy' environment variable is set and if our URL is included
 957         use_proxy = self.detect_no_proxy(req.get_full_url())
 958         https_proxy = os.environ.get('https_proxy')
 959 
 960         context = None
 961         try:
 962             context = self.make_context(tmp_ca_cert_path, cadata)
 963         except NotImplementedError:
 964             # We'll make do with no context below
 965             pass
 966 
 967         try:
 968             if use_proxy and https_proxy:
 969                 proxy_parts = generic_urlparse(urlparse(https_proxy))
 970                 port = proxy_parts.get('port') or 443
 971                 proxy_hostname = proxy_parts.get('hostname', None)
 972                 if proxy_hostname is None or proxy_parts.get('scheme') == '':
 973                     raise ProxyError("Failed to parse https_proxy environment variable."
 974                                      " Please make sure you export https proxy as 'https_proxy=<SCHEME>://<IP_ADDRESS>:<PORT>'")
 975 
 976                 s = socket.create_connection((proxy_hostname, port))
 977                 if proxy_parts.get('scheme') == 'http':
 978                     s.sendall(to_bytes(self.CONNECT_COMMAND % (self.hostname, self.port), errors='surrogate_or_strict'))
 979                     if proxy_parts.get('username'):
 980                         credentials = "%s:%s" % (proxy_parts.get('username', ''), proxy_parts.get('password', ''))
 981                         s.sendall(b'Proxy-Authorization: Basic %s\r\n' % base64.b64encode(to_bytes(credentials, errors='surrogate_or_strict')).strip())
 982                     s.sendall(b'\r\n')
 983                     connect_result = b""
 984                     while connect_result.find(b"\r\n\r\n") <= 0:
 985                         connect_result += s.recv(4096)
 986                         # 128 kilobytes of headers should be enough for everyone.
 987                         if len(connect_result) > 131072:
 988                             raise ProxyError('Proxy sent too verbose headers. Only 128KiB allowed.')
 989                     self.validate_proxy_response(connect_result)
 990                     if context:
 991                         ssl_s = context.wrap_socket(s, server_hostname=self.hostname)
 992                     elif HAS_URLLIB3_SSL_WRAP_SOCKET:
 993                         ssl_s = ssl_wrap_socket(s, ca_certs=tmp_ca_cert_path, cert_reqs=ssl.CERT_REQUIRED, ssl_version=PROTOCOL, server_hostname=self.hostname)
 994                     else:
 995                         ssl_s = ssl.wrap_socket(s, ca_certs=tmp_ca_cert_path, cert_reqs=ssl.CERT_REQUIRED, ssl_version=PROTOCOL)
 996                         match_hostname(ssl_s.getpeercert(), self.hostname)
 997                 else:
 998                     raise ProxyError('Unsupported proxy scheme: %s. Currently ansible only supports HTTP proxies.' % proxy_parts.get('scheme'))
 999             else:
1000                 s = socket.create_connection((self.hostname, self.port))
1001                 if context:
1002                     ssl_s = context.wrap_socket(s, server_hostname=self.hostname)
1003                 elif HAS_URLLIB3_SSL_WRAP_SOCKET:
1004                     ssl_s = ssl_wrap_socket(s, ca_certs=tmp_ca_cert_path, cert_reqs=ssl.CERT_REQUIRED, ssl_version=PROTOCOL, server_hostname=self.hostname)
1005                 else:
1006                     ssl_s = ssl.wrap_socket(s, ca_certs=tmp_ca_cert_path, cert_reqs=ssl.CERT_REQUIRED, ssl_version=PROTOCOL)
1007                     match_hostname(ssl_s.getpeercert(), self.hostname)
1008             # close the ssl connection
1009             # ssl_s.unwrap()
1010             s.close()
1011         except (ssl.SSLError, CertificateError) as e:
1012             build_ssl_validation_error(self.hostname, self.port, paths_checked, e)
1013         except socket.error as e:
1014             raise ConnectionError('Failed to connect to %s at port %s: %s' % (self.hostname, self.port, to_native(e)))
1015 
1016         return req

1402 def prepare_multipart(fields):
1403     """Takes a mapping, and prepares a multipart/form-data body
1404 
1405     :arg fields: Mapping
1406     :returns: tuple of (content_type, body) where ``content_type`` is
1407         the ``multipart/form-data`` ``Content-Type`` header including
1408         ``boundary`` and ``body`` is the prepared bytestring body
1409 
1410     Payload content from a file will be base64 encoded and will include
1411     the appropriate ``Content-Transfer-Encoding`` and ``Content-Type``
1412     headers.
1413 
1414     Example:
1415         {
1416             "file1": {
1417                 "filename": "/bin/true",
1418                 "mime_type": "application/octet-stream"
1419             },
1420             "file2": {
1421                 "content": "text based file content",
1422                 "filename": "fake.txt",
1423                 "mime_type": "text/plain",
1424             },
1425             "text_form_field": "value"
1426         }
1427     """
1428 
1429     if not isinstance(fields, Mapping):
1430         raise TypeError(
1431             'Mapping is required, cannot be type %s' % fields.__class__.__name__
1432         )
1433 
1434     m = email.mime.multipart.MIMEMultipart('form-data')
1435     for field, value in sorted(fields.items()):
1436         if isinstance(value, string_types):
1437             main_type = 'text'
1438             sub_type = 'plain'
1439             content = value
1440             filename = None
1441         elif isinstance(value, Mapping):
1442             filename = value.get('filename')
1443             content = value.get('content')
1444             if not any((filename, content)):
1445                 raise ValueError('at least one of filename or content must be provided')
1446 
1447             mime = value.get('mime_type')
1448             if not mime:
1449                 try:
1450                     mime = mimetypes.guess_type(filename or '', strict=False)[0] or 'application/octet-stream'
1451                 except Exception:
1452                     mime = 'application/octet-stream'
1453             main_type, sep, sub_type = mime.partition('/')
1454         else:
1455             raise TypeError(
1456                 'value must be a string, or mapping, cannot be type %s' % value.__class__.__name__
1457             )
1458 
1459         if not content and filename:
1460             with open(to_bytes(filename, errors='surrogate_or_strict'), 'rb') as f:
1461                 part = email.mime.application.MIMEApplication(f.read())
1462                 del part['Content-Type']
1463                 part.add_header('Content-Type', '%s/%s' % (main_type, sub_type))
1464         else:
1465             part = email.mime.nonmultipart.MIMENonMultipart(main_type, sub_type)
1466             part.set_payload(to_bytes(content))
1467 
1468         part.add_header('Content-Disposition', 'form-data')
1469         del part['MIME-Version']
1470         part.set_param(
1471             'name',
1472             field,
1473             header='Content-Disposition'
1474         )
1475         if filename:
1476             part.set_param(
1477                 'filename',
1478                 to_native(os.path.basename(filename)),
1479                 header='Content-Disposition'
1480             )
1481 
1482         m.attach(part)
1483 
1484     if PY3:
1485         # Ensure headers are not split over multiple lines
1486         # The HTTP policy also uses CRLF by default
1487         b_data = m.as_bytes(policy=email.policy.HTTP)
1488     else:
1489         # Py2
1490         # We cannot just call ``as_string`` since it provides no way
1491         # to specify ``maxheaderlen``
1492         fp = cStringIO()  # cStringIO seems to be required here
1493         # Ensure headers are not split over multiple lines
1494         g = email.generator.Generator(fp, maxheaderlen=0)
1495         g.flatten(m)
1496         # ``fix_eols`` switches from ``\n`` to ``\r\n``
1497         b_data = email.utils.fix_eols(fp.getvalue())
1498     del m
1499 
1500     headers, sep, b_content = b_data.partition(b'\r\n\r\n')
1501     del b_data
1502 
1503     if PY3:
1504         parser = email.parser.BytesHeaderParser().parsebytes
1505     else:
1506         # Py2
1507         parser = email.parser.HeaderParser().parsestr
1508 
1509     return (
1510         parser(headers)['content-type'],  # Message converts to native strings
1511         b_content
1512     )

1547 def fetch_url(module, url, data=None, headers=None, method=None,
1548               use_proxy=True, force=False, last_mod_time=None, timeout=10,
1549               use_gssapi=False, unix_socket=None, ca_path=None, cookies=None):
1550     """Sends a request via HTTP(S) or FTP (needs the module as parameter)
1551 
1552     :arg module: The AnsibleModule (used to get username, password etc. (s.b.).
1553     :arg url:             The url to use.
1554 
1555     :kwarg data:          The data to be sent (in case of POST/PUT).
1556     :kwarg headers:       A dict with the request headers.
1557     :kwarg method:        "POST", "PUT", etc.
1558     :kwarg boolean use_proxy:     Default: True
1559     :kwarg boolean force: If True: Do not get a cached copy (Default: False)
1560     :kwarg last_mod_time: Default: None
1561     :kwarg int timeout:   Default: 10
1562     :kwarg boolean use_gssapi:   Default: False
1563     :kwarg unix_socket: (optional) String of file system path to unix socket file to use when establishing
1564         connection to the provided url
1565     :kwarg ca_path: (optional) String of file system path to CA cert bundle to use
1566 
1567     :returns: A tuple of (**response**, **info**). Use ``response.read()`` to read the data.
1568         The **info** contains the 'status' and other meta data. When a HttpError (status >= 400)
1569         occurred then ``info['body']`` contains the error response data::
1570 
1571     Example::
1572 
1573         data={...}
1574         resp, info = fetch_url(module,
1575                                "http://example.com",
1576                                data=module.jsonify(data),
1577                                headers={'Content-type': 'application/json'},
1578                                method="POST")
1579         status_code = info["status"]
1580         body = resp.read()
1581         if status_code >= 400 :
1582             body = info['body']
1583     """
1584 
1585     if not HAS_URLPARSE:
1586         module.fail_json(msg='urlparse is not installed')
1587 
1588     # ensure we use proper tempdir
1589     old_tempdir = tempfile.tempdir
1590     tempfile.tempdir = module.tmpdir
1591 
1592     # Get validate_certs from the module params
1593     validate_certs = module.params.get('validate_certs', True)
1594 
1595     username = module.params.get('url_username', '')
1596     password = module.params.get('url_password', '')
1597     http_agent = module.params.get('http_agent', 'ansible-httpget')
1598     force_basic_auth = module.params.get('force_basic_auth', '')
1599 
1600     follow_redirects = module.params.get('follow_redirects', 'urllib2')
1601 
1602     client_cert = module.params.get('client_cert')
1603     client_key = module.params.get('client_key')
1604 
1605     if not isinstance(cookies, cookiejar.CookieJar):
1606         cookies = cookiejar.LWPCookieJar()
1607 
1608     r = None
1609     info = dict(url=url, status=-1)
1610     try:
1611         r = open_url(url, data=data, headers=headers, method=method,
1612                      use_proxy=use_proxy, force=force, last_mod_time=last_mod_time, timeout=timeout,
1613                      validate_certs=validate_certs, url_username=username,
1614                      url_password=password, http_agent=http_agent, force_basic_auth=force_basic_auth,
1615                      follow_redirects=follow_redirects, client_cert=client_cert,
1616                      client_key=client_key, cookies=cookies, use_gssapi=use_gssapi,
1617                      unix_socket=unix_socket, ca_path=ca_path)
1618         # Lowercase keys, to conform to py2 behavior, so that py3 and py2 are predictable
1619         info.update(dict((k.lower(), v) for k, v in r.info().items()))
1620 
1621         # Don't be lossy, append header values for duplicate headers
1622         # In Py2 there is nothing that needs done, py2 does this for us
1623         if PY3:
1624             temp_headers = {}
1625             for name, value in r.headers.items():
1626                 # The same as above, lower case keys to match py2 behavior, and create more consistent results
1627                 name = name.lower()
1628                 if name in temp_headers:
1629                     temp_headers[name] = ', '.join((temp_headers[name], value))
1630                 else:
1631                     temp_headers[name] = value
1632             info.update(temp_headers)
1633 
1634         # parse the cookies into a nice dictionary
1635         cookie_list = []
1636         cookie_dict = dict()
1637         # Python sorts cookies in order of most specific (ie. longest) path first. See ``CookieJar._cookie_attrs``
1638         # Cookies with the same path are reversed from response order.
1639         # This code makes no assumptions about that, and accepts the order given by python
1640         for cookie in cookies:
1641             cookie_dict[cookie.name] = cookie.value
1642             cookie_list.append((cookie.name, cookie.value))
1643         info['cookies_string'] = '; '.join('%s=%s' % c for c in cookie_list)
1644 
1645         info['cookies'] = cookie_dict
1646         # finally update the result with a message about the fetch
1647         info.update(dict(msg="OK (%s bytes)" % r.headers.get('Content-Length', 'unknown'), url=r.geturl(), status=r.code))
1648     except NoSSLError as e:
1649         distribution = get_distribution()
1650         if distribution is not None and distribution.lower() == 'redhat':
1651             module.fail_json(msg='%s. You can also install python-ssl from EPEL' % to_native(e), **info)
1652         else:
1653             module.fail_json(msg='%s' % to_native(e), **info)
1654     except (ConnectionError, ValueError) as e:
1655         module.fail_json(msg=to_native(e), **info)
1656     except urllib_error.HTTPError as e:
1657         try:
1658             body = e.read()
1659         except AttributeError:
1660             body = ''

1688                use_proxy=True, force=False, last_mod_time=None, timeout=10):
1689     '''Download and save a file via HTTP(S) or FTP (needs the module as parameter).
1690     This is basically a wrapper around fetch_url().
1691 
1692     :arg module: The AnsibleModule (used to get username, password etc. (s.b.).
1693     :arg url:             The url to use.
1694

1696     :kwarg headers:       A dict with the request headers.
1697     :kwarg method:        "POST", "PUT", etc.
1698     :kwarg boolean use_proxy:     Default: True
1699     :kwarg boolean force: If True: Do not get a cached copy (Default: False)
1700     :kwarg last_mod_time: Default: None
1701     :kwarg int timeout:   Default: 10
1702 
1703     :returns: A string, the path to the downloaded file.
1704     '''
1705     # download file
1706     bufsize = 65536
1707     file_name, file_ext = os.path.splitext(str(url.rsplit('/', 1)[1]))
1708     fetch_temp_file = tempfile.NamedTemporaryFile(dir=module.tmpdir, prefix=file_name, suffix=file_ext, delete=False)
1709     module.add_cleanup_file(fetch_temp_file.name)
1710     try:
1711         rsp, info = fetch_url(module, url, data, headers, method, use_proxy, force, last_mod_time, timeout)
1712         if not rsp:

1714         data = rsp.read(bufsize)
1715         while data:
1716             fetch_temp_file.write(data)
1717             data = rsp.read(bufsize)
1718         fetch_temp_file.close()
1719     except Exception as e:
1720         module.fail_json(msg="Failure downloading %s, %s" % (url, to_native(e)))
1721     return fetch_temp_file.name
