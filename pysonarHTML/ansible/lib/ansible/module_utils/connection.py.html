  47 def write_to_file_descriptor(fd, obj):
  48     """Handles making sure all data is properly written to file descriptor fd.
  49 
  50     In particular, that data is encoded in a character stream-friendly way and
  51     that all data gets written before returning.
  52     """
  53     # Need to force a protocol that is compatible with both py2 and py3.
  54     # That would be protocol=2 or less.
  55     # Also need to force a protocol that excludes certain control chars as
  56     # stdin in this case is a pty and control chars will cause problems.
  57     # that means only protocol=0 will work.
  58     src = cPickle.dumps(obj, protocol=0)
  59 
  60     # raw \r characters will not survive pty round-trip
  61     # They should be rehydrated on the receiving end
  62     src = src.replace(b'\r', br'\r')
  63     data_hash = to_bytes(hashlib.sha1(src).hexdigest())
  64 
  65     os.write(fd, b'%d\n' % len(src))
  66     os.write(fd, src)
  67     os.write(fd, b'%s\n' % data_hash)

  70 def send_data(s, data):
  71     packed_len = struct.pack('!Q', len(data))
  72     return s.sendall(packed_len + data)
  73 
  74 
  75 def recv_data(s):
  76     header_len = 8  # size of a packed unsigned long long
  77     data = to_bytes("")
  78     while len(data) < header_len:
  79         d = s.recv(header_len - len(data))
  80         if not d:
  81             return None
  82         data += d
  83     data_len = struct.unpack('!Q', data[:header_len])[0]
  84     data = data[header_len:]
  85     while len(data) < data_len:
  86         d = s.recv(data_len - len(data))
  87         if not d:
  88             return None
  89         data += d
  90     return data

  93 def exec_command(module, command):
  94     connection = Connection(module._socket_path)
  95     try:
  96         out = connection.exec_command(command)
  97     except ConnectionError as exc:
  98         code = getattr(exc, 'code', 1)
  99         message = getattr(exc, 'err', exc)
 100         return code, '', to_text(message, errors='surrogate_then_replace')
 101     return 0, out, ''

 104 def request_builder(method_, *args, **kwargs):
 105     reqid = str(uuid.uuid4())
 106     req = {'jsonrpc': '2.0', 'method': method_, 'id': reqid}
 107     req['params'] = (args, kwargs)
 108 
 109     return req

 112 class ConnectionError(Exception):
 113 
 114     def __init__(self, message, *args, **kwargs):
 115         super(ConnectionError, self).__init__(message)
 116         for k, v in iteritems(kwargs):
 117             setattr(self, k, v)

 120 class Connection(object):
 121 
 122     def __init__(self, socket_path):
 123         if socket_path is None:
 124             raise AssertionError('socket_path must be a value')
 125         self.socket_path = socket_path
 126 
 127     def __getattr__(self, name):
 128         try:
 129             return self.__dict__[name]
 130         except KeyError:
 131             if name.startswith('_'):
 132                 raise AttributeError("'%s' object has no attribute '%s'" % (self.__class__.__name__, name))
 133             return partial(self.__rpc__, name)
 134 
 135     def _exec_jsonrpc(self, name, *args, **kwargs):
 136 
 137         req = request_builder(name, *args, **kwargs)
 138         reqid = req['id']
 139 
 140         if not os.path.exists(self.socket_path):
 141             raise ConnectionError(
 142                 'socket path %s does not exist or cannot be found. See Troubleshooting socket '
 143                 'path issues in the Network Debug and Troubleshooting Guide' % self.socket_path
 144             )
 145 
 146         try:
 147             data = json.dumps(req, cls=AnsibleJSONEncoder)
 148         except TypeError as exc:
 149             raise ConnectionError(
 150                 "Failed to encode some variables as JSON for communication with ansible-connection. "
 151                 "The original exception was: %s" % to_text(exc)
 152             )
 153 
 154         try:
 155             out = self.send(data)
 156         except socket.error as e:
 157             raise ConnectionError(
 158                 'unable to connect to socket %s. See Troubleshooting socket path issues '
 159                 'in the Network Debug and Troubleshooting Guide' % self.socket_path,
 160                 err=to_text(e, errors='surrogate_then_replace'), exception=traceback.format_exc()
 161             )
 162 
 163         try:
 164             response = json.loads(out)
 165         except ValueError:
 166             params = [repr(arg) for arg in args] + ['{0}={1!r}'.format(k, v) for k, v in iteritems(kwargs)]
 167             params = ', '.join(params)
 168             raise ConnectionError(
 169                 "Unable to decode JSON from response to {0}({1}). Received '{2}'.".format(name, params, out)
 170             )
 171 
 172         if response['id'] != reqid:
 173             raise ConnectionError('invalid json-rpc id received')
 174         if "result_type" in response:
 175             response["result"] = cPickle.loads(to_bytes(response["result"]))
 176 
 177         return response
 178 
 179     def __rpc__(self, name, *args, **kwargs):
 180         """Executes the json-rpc and returns the output received
 181            from remote device.
 182            :name: rpc method to be executed over connection plugin that implements jsonrpc 2.0
 183            :args: Ordered list of params passed as arguments to rpc method
 184            :kwargs: Dict of valid key, value pairs passed as arguments to rpc method
 185 
 186            For usage refer the respective connection plugin docs.
 187         """
 188 
 189         response = self._exec_jsonrpc(name, *args, **kwargs)
 190 
 191         if 'error' in response:
 192             err = response.get('error')
 193             msg = err.get('data') or err['message']
 194             code = err['code']
 195             raise ConnectionError(to_text(msg, errors='surrogate_then_replace'), code=code)
 196 
 197         return response['result']
 198 
 199     def send(self, data):
 200         try:
 201             sf = socket.socket(socket.AF_UNIX, socket.SOCK_STREAM)
 202             sf.connect(self.socket_path)
 203 
 204             send_data(sf, to_bytes(data))
 205             response = recv_data(sf)
 206 
 207         except socket.error as e:
 208             sf.close()
 209             raise ConnectionError(
 210                 'unable to connect to socket %s. See the socket path issue category in '
 211                 'Network Debug and Troubleshooting Guide' % self.socket_path,
 212                 err=to_text(e, errors='surrogate_then_replace'), exception=traceback.format_exc()
 213             )
 214 
 215         sf.close()
 216 
 217         return to_text(response, errors='surrogate_or_strict')
