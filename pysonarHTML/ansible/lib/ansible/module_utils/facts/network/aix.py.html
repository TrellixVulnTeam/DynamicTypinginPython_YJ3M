  25 class AIXNetwork(GenericBsdIfconfigNetwork):
  26     """
  27     This is the AIX Network Class.
  28     It uses the GenericBsdIfconfigNetwork unchanged.
  29     """
  30     platform = 'AIX'

  32     def get_default_interfaces(self, route_path):
  33         netstat_path = self.module.get_bin_path('netstat')
  34 
  35         rc, out, err = self.module.run_command([netstat_path, '-nr'])
  36 
  37         interface = dict(v4={}, v6={})
  38 
  39         lines = out.splitlines()
  40         for line in lines:
  41             words = line.split()
  42             if len(words) > 1 and words[0] == 'default':
  43                 if '.' in words[1]:
  44                     interface['v4']['gateway'] = words[1]
  45                     interface['v4']['interface'] = words[5]
  46                 elif ':' in words[1]:
  47                     interface['v6']['gateway'] = words[1]
  48                     interface['v6']['interface'] = words[5]
  49 
  50         return interface['v4'], interface['v6']

  52     # AIX 'ifconfig -a' does not have three words in the interface line
  53     def get_interfaces_info(self, ifconfig_path, ifconfig_options='-a'):
  54         interfaces = {}
  55         current_if = {}
  56         ips = dict(
  57             all_ipv4_addresses=[],
  58             all_ipv6_addresses=[],
  59         )
  60 
  61         uname_rc = None
  62         uname_out = None
  63         uname_err = None
  64         uname_path = self.module.get_bin_path('uname')
  65         if uname_path:
  66             uname_rc, uname_out, uname_err = self.module.run_command([uname_path, '-W'])
  67 
  68         rc, out, err = self.module.run_command([ifconfig_path, ifconfig_options])
  69 
  70         for line in out.splitlines():
  71 
  72             if line:
  73                 words = line.split()
  74 
  75                 # only this condition differs from GenericBsdIfconfigNetwork
  76                 if re.match(r'^\w*\d*:', line):
  77                     current_if = self.parse_interface_line(words)
  78                     interfaces[current_if['device']] = current_if
  79                 elif words[0].startswith('options='):
  80                     self.parse_options_line(words, current_if, ips)
  81                 elif words[0] == 'nd6':
  82                     self.parse_nd6_line(words, current_if, ips)
  83                 elif words[0] == 'ether':
  84                     self.parse_ether_line(words, current_if, ips)
  85                 elif words[0] == 'media:':
  86                     self.parse_media_line(words, current_if, ips)
  87                 elif words[0] == 'status:':
  88                     self.parse_status_line(words, current_if, ips)
  89                 elif words[0] == 'lladdr':
  90                     self.parse_lladdr_line(words, current_if, ips)
  91                 elif words[0] == 'inet':
  92                     self.parse_inet_line(words, current_if, ips)
  93                 elif words[0] == 'inet6':
  94                     self.parse_inet6_line(words, current_if, ips)
  95                 else:
  96                     self.parse_unknown_line(words, current_if, ips)
  97 
  98             # don't bother with wpars it does not work
  99             # zero means not in wpar
 100             if not uname_rc and uname_out.split()[0] == '0':
 101 
 102                 if current_if['macaddress'] == 'unknown' and re.match('^en', current_if['device']):
 103                     entstat_path = self.module.get_bin_path('entstat')
 104                     if entstat_path:
 105                         rc, out, err = self.module.run_command([entstat_path, current_if['device']])
 106                         if rc != 0:
 107                             break
 108                         for line in out.splitlines():
 109                             if not line:
 110                                 pass
 111                             buff = re.match('^Hardware Address: (.*)', line)
 112                             if buff:
 113                                 current_if['macaddress'] = buff.group(1)
 114 
 115                             buff = re.match('^Device Type:', line)
 116                             if buff and re.match('.*Ethernet', line):
 117                                 current_if['type'] = 'ether'
 118 
 119                 # device must have mtu attribute in ODM
 120                 if 'mtu' not in current_if:
 121                     lsattr_path = self.module.get_bin_path('lsattr')
 122                     if lsattr_path:
 123                         rc, out, err = self.module.run_command([lsattr_path, '-El', current_if['device']])
 124                         if rc != 0:
 125                             break
 126                         for line in out.splitlines():
 127                             if line:
 128                                 words = line.split()
 129                                 if words[0] == 'mtu':
 130                                     current_if['mtu'] = words[1]
 131         return interfaces, ips
 132 
 133     # AIX 'ifconfig -a' does not inform about MTU, so remove current_if['mtu'] here
 134     def parse_interface_line(self, words):
 135         device = words[0][0:-1]
 136         current_if = {'device': device, 'ipv4': [], 'ipv6': [], 'type': 'unknown'}
 137         current_if['flags'] = self.get_options(words[1])
 138         current_if['macaddress'] = 'unknown'    # will be overwritten later
 139         return current_if
