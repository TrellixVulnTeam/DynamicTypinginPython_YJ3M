  23 class NetBSDNetwork(GenericBsdIfconfigNetwork):
  24     """
  25     This is the NetBSD Network Class.
  26     It uses the GenericBsdIfconfigNetwork
  27     """
  28     platform = 'NetBSD'
  29 
  30     def parse_media_line(self, words, current_if, ips):
  31         # example of line:
  32         # $ ifconfig
  33         # ne0: flags=8863<UP,BROADCAST,NOTRAILERS,RUNNING,SIMPLEX,MULTICAST> mtu 1500
  34         #    ec_capabilities=1<VLAN_MTU>
  35         #    ec_enabled=0
  36         #    address: 00:20:91:45:00:78
  37         #    media: Ethernet 10baseT full-duplex
  38         #    inet 192.168.156.29 netmask 0xffffff00 broadcast 192.168.156.255
  39         current_if['media'] = words[1]
  40         if len(words) > 2:
  41             current_if['media_type'] = words[2]
  42         if len(words) > 3:
  43             current_if['media_options'] = words[3].split(',')
  44
