  26 class GenericBsdIfconfigNetwork(Network):
  27     """
  28     This is a generic BSD subclass of Network using the ifconfig command.
  29     It defines
  30     - interfaces (a list of interface names)
  31     - interface_<name> dictionary of ipv4, ipv6, and mac address information.
  32     - all_ipv4_addresses and all_ipv6_addresses: lists of all configured addresses.
  33     """
  34     platform = 'Generic_BSD_Ifconfig'
  35 
  36     def populate(self, collected_facts=None):
  37         network_facts = {}
  38         ifconfig_path = self.module.get_bin_path('ifconfig')
  39 
  40         if ifconfig_path is None:
  41             return network_facts
  42 
  43         route_path = self.module.get_bin_path('route')
  44 
  45         if route_path is None:
  46             return network_facts
  47 
  48         default_ipv4, default_ipv6 = self.get_default_interfaces(route_path)
  49         interfaces, ips = self.get_interfaces_info(ifconfig_path)
  50         interfaces = self.detect_type_media(interfaces)
  51 
  52         self.merge_default_interface(default_ipv4, interfaces, 'ipv4')
  53         self.merge_default_interface(default_ipv6, interfaces, 'ipv6')
  54         network_facts['interfaces'] = sorted(list(interfaces.keys()))
  55 
  56         for iface in interfaces:
  57             network_facts[iface] = interfaces[iface]
  58 
  59         network_facts['default_ipv4'] = default_ipv4
  60         network_facts['default_ipv6'] = default_ipv6
  61         network_facts['all_ipv4_addresses'] = ips['all_ipv4_addresses']
  62         network_facts['all_ipv6_addresses'] = ips['all_ipv6_addresses']
  63 
  64         return network_facts
  65 
  66     def detect_type_media(self, interfaces):
  67         for iface in interfaces:
  68             if 'media' in interfaces[iface]:
  69                 if 'ether' in interfaces[iface]['media'].lower():
  70                     interfaces[iface]['type'] = 'ether'
  71         return interfaces

  73     def get_default_interfaces(self, route_path):
  74 
  75         # Use the commands:
  76         #     route -n get default
  77         #     route -n get -inet6 default
  78         # to find out the default outgoing interface, address, and gateway
  79 
  80         command = dict(v4=[route_path, '-n', 'get', 'default'],
  81                        v6=[route_path, '-n', 'get', '-inet6', 'default'])
  82 
  83         interface = dict(v4={}, v6={})
  84 
  85         for v in 'v4', 'v6':
  86 
  87             if v == 'v6' and not socket.has_ipv6:
  88                 continue
  89             rc, out, err = self.module.run_command(command[v])
  90             if not out:
  91                 # v6 routing may result in
  92                 #   RTNETLINK answers: Invalid argument
  93                 continue
  94             for line in out.splitlines():
  95                 words = line.strip().split(': ')
  96                 # Collect output from route command
  97                 if len(words) > 1:
  98                     if words[0] == 'interface':
  99                         interface[v]['interface'] = words[1]
 100                     if words[0] == 'gateway':
 101                         interface[v]['gateway'] = words[1]
 102                     # help pick the right interface address on OpenBSD
 103                     if words[0] == 'if address':
 104                         interface[v]['address'] = words[1]
 105                     # help pick the right interface address on NetBSD
 106                     if words[0] == 'local addr':
 107                         interface[v]['address'] = words[1]
 108 
 109         return interface['v4'], interface['v6']

 111     def get_interfaces_info(self, ifconfig_path, ifconfig_options='-a'):
 112         interfaces = {}
 113         current_if = {}
 114         ips = dict(
 115             all_ipv4_addresses=[],
 116             all_ipv6_addresses=[],
 117         )
 118         # FreeBSD, DragonflyBSD, NetBSD, OpenBSD and macOS all implicitly add '-a'
 119         # when running the command 'ifconfig'.
 120         # Solaris must explicitly run the command 'ifconfig -a'.
 121         rc, out, err = self.module.run_command([ifconfig_path, ifconfig_options])
 122 
 123         for line in out.splitlines():
 124 
 125             if line:
 126                 words = line.split()
 127 
 128                 if words[0] == 'pass':
 129                     continue
 130                 elif re.match(r'^\S', line) and len(words) > 3:
 131                     current_if = self.parse_interface_line(words)
 132                     interfaces[current_if['device']] = current_if
 133                 elif words[0].startswith('options='):
 134                     self.parse_options_line(words, current_if, ips)
 135                 elif words[0] == 'nd6':
 136                     self.parse_nd6_line(words, current_if, ips)
 137                 elif words[0] == 'ether':
 138                     self.parse_ether_line(words, current_if, ips)
 139                 elif words[0] == 'media:':
 140                     self.parse_media_line(words, current_if, ips)
 141                 elif words[0] == 'status:':
 142                     self.parse_status_line(words, current_if, ips)
 143                 elif words[0] == 'lladdr':
 144                     self.parse_lladdr_line(words, current_if, ips)
 145                 elif words[0] == 'inet':
 146                     self.parse_inet_line(words, current_if, ips)
 147                 elif words[0] == 'inet6':
 148                     self.parse_inet6_line(words, current_if, ips)
 149                 elif words[0] == 'tunnel':
 150                     self.parse_tunnel_line(words, current_if, ips)
 151                 else:
 152                     self.parse_unknown_line(words, current_if, ips)
 153 
 154         return interfaces, ips

 156     def parse_interface_line(self, words):
 157         device = words[0][0:-1]
 158         current_if = {'device': device, 'ipv4': [], 'ipv6': [], 'type': 'unknown'}
 159         current_if['flags'] = self.get_options(words[1])
 160         if 'LOOPBACK' in current_if['flags']:
 161             current_if['type'] = 'loopback'
 162         current_if['macaddress'] = 'unknown'    # will be overwritten later
 163 
 164         if len(words) >= 5:  # Newer FreeBSD versions
 165             current_if['metric'] = words[3]
 166             current_if['mtu'] = words[5]
 167         else:
 168             current_if['mtu'] = words[3]
 169 
 170         return current_if

 184     def parse_media_line(self, words, current_if, ips):
 185         # not sure if this is useful - we also drop information
 186         current_if['media'] = words[1]
 187         if len(words) > 2:
 188             current_if['media_select'] = words[2]
 189         if len(words) > 3:
 190             current_if['media_type'] = words[3][1:]
 191         if len(words) > 4:
 192             current_if['media_options'] = self.get_options(words[4])

 200     def parse_inet_line(self, words, current_if, ips):
 201         # netbsd show aliases like this
 202         #  lo0: flags=8049<UP,LOOPBACK,RUNNING,MULTICAST> mtu 33184
 203         #         inet 127.0.0.1 netmask 0xff000000
 204         #         inet alias 127.1.1.1 netmask 0xff000000
 205         if words[1] == 'alias':
 206             del words[1]
 207 
 208         address = {'address': words[1]}
 209         # cidr style ip address (eg, 127.0.0.1/24) in inet line
 210         # used in netbsd ifconfig -e output after 7.1
 211         if '/' in address['address']:
 212             ip_address, cidr_mask = address['address'].split('/')
 213 
 214             address['address'] = ip_address
 215 
 216             netmask_length = int(cidr_mask)
 217             netmask_bin = (1 << 32) - (1 << 32 >> int(netmask_length))
 218             address['netmask'] = socket.inet_ntoa(struct.pack('!L', netmask_bin))
 219 
 220             if len(words) > 5:
 221                 address['broadcast'] = words[3]
 222 
 223         else:
 224             # deal with hex netmask
 225             if re.match('([0-9a-f]){8}', words[3]) and len(words[3]) == 8:
 226                 words[3] = '0x' + words[3]
 227             if words[3].startswith('0x'):
 228                 address['netmask'] = socket.inet_ntoa(struct.pack('!L', int(words[3], base=16)))
 229             else:
 230                 # otherwise assume this is a dotted quad
 231                 address['netmask'] = words[3]
 232         # calculate the network
 233         address_bin = struct.unpack('!L', socket.inet_aton(address['address']))[0]
 234         netmask_bin = struct.unpack('!L', socket.inet_aton(address['netmask']))[0]
 235         address['network'] = socket.inet_ntoa(struct.pack('!L', address_bin & netmask_bin))
 236         if 'broadcast' not in address:
 237             # broadcast may be given or we need to calculate
 238             if len(words) > 5:
 239                 address['broadcast'] = words[5]
 240             else:
 241                 address['broadcast'] = socket.inet_ntoa(struct.pack('!L', address_bin | (~netmask_bin & 0xffffffff)))
 242 
 243         # add to our list of addresses
 244         if not words[1].startswith('127.'):
 245             ips['all_ipv4_addresses'].append(address['address'])
 246         current_if['ipv4'].append(address)
 247 
 248     def parse_inet6_line(self, words, current_if, ips):
 249         address = {'address': words[1]}
 250 
 251         # using cidr style addresses, ala NetBSD ifconfig post 7.1
 252         if '/' in address['address']:
 253             ip_address, cidr_mask = address['address'].split('/')
 254 
 255             address['address'] = ip_address
 256             address['prefix'] = cidr_mask
 257 
 258             if len(words) > 5:
 259                 address['scope'] = words[5]
 260         else:
 261             if (len(words) >= 4) and (words[2] == 'prefixlen'):
 262                 address['prefix'] = words[3]
 263             if (len(words) >= 6) and (words[4] == 'scopeid'):
 264                 address['scope'] = words[5]
 265 
 266         localhost6 = ['::1', '::1/128', 'fe80::1%lo0']
 267         if address['address'] not in localhost6:
 268             ips['all_ipv6_addresses'].append(address['address'])
 269         current_if['ipv6'].append(address)

 281     def get_options(self, option_string):
 282         start = option_string.find('<') + 1
 283         end = option_string.rfind('>')
 284         if (start > 0) and (end > 0) and (end > start + 1):
 285             option_csv = option_string[start:end]
 286             return option_csv.split(',')
 287         else:
 288             return []

 290     def merge_default_interface(self, defaults, interfaces, ip_type):
 291         if 'interface' not in defaults:
 292             return
 293         if not defaults['interface'] in interfaces:
 294             return
 295         ifinfo = interfaces[defaults['interface']]
 296         # copy all the interface values across except addresses
 297         for item in ifinfo:
 298             if item != 'ipv4' and item != 'ipv6':
 299                 defaults[item] = ifinfo[item]
 300 
 301         ipinfo = []
 302         if 'address' in defaults:
 303             ipinfo = [x for x in ifinfo[ip_type] if x['address'] == defaults['address']]
 304 
 305         if len(ipinfo) == 0:
 306             ipinfo = ifinfo[ip_type]
 307 
 308         if len(ipinfo) > 0:
 309             for item in ipinfo[0]:
 310                 defaults[item] = ipinfo[0][item]
