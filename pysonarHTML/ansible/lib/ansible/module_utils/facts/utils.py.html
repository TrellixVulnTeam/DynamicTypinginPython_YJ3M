  22 def get_file_content(path, default=None, strip=True):
  23     data = default
  24     if os.path.exists(path) and os.access(path, os.R_OK):
  25         try:
  26             try:
  27                 datafile = open(path)
  28                 data = datafile.read()
  29                 if strip:
  30                     data = data.strip()
  31                 if len(data) == 0:
  32                     data = default
  33             finally:
  34                 datafile.close()
  35         except Exception:
  36             # ignore errors as some jails/containers might have readable permissions but not allow reads to proc
  37             # done in 2 blocks for 2.4 compat
  38             pass
  39     return data

  42 def get_file_lines(path, strip=True, line_sep=None):
  43     '''get list of lines from file'''
  44     data = get_file_content(path, strip=strip)
  45     if data:
  46         if line_sep is None:
  47             ret = data.splitlines()
  48         else:
  49             if len(line_sep) == 1:
  50                 ret = data.rstrip(line_sep).split(line_sep)
  51             else:
  52                 ret = data.split(line_sep)
  53     else:
  54         ret = []
  55     return ret

  58 def get_mount_size(mountpoint):
  59     mount_size = {}
  60 
  61     try:
  62         statvfs_result = os.statvfs(mountpoint)
  63         mount_size['size_total'] = statvfs_result.f_frsize * statvfs_result.f_blocks
  64         mount_size['size_available'] = statvfs_result.f_frsize * (statvfs_result.f_bavail)
  65 
  66         # Block total/available/used
  67         mount_size['block_size'] = statvfs_result.f_bsize
  68         mount_size['block_total'] = statvfs_result.f_blocks
  69         mount_size['block_available'] = statvfs_result.f_bavail
  70         mount_size['block_used'] = mount_size['block_total'] - mount_size['block_available']
  71 
  72         # Inode total/available/used
  73         mount_size['inode_total'] = statvfs_result.f_files
  74         mount_size['inode_available'] = statvfs_result.f_favail
  75         mount_size['inode_used'] = mount_size['inode_total'] - mount_size['inode_available']
  76     except OSError:
  77         pass
  78 
  79     return mount_size
