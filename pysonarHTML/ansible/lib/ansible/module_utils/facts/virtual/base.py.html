  25 class Virtual:
  26     """
  27     This is a generic Virtual subclass of Facts.  This should be further
  28     subclassed to implement per platform.  If you subclass this,
  29     you should define:
  30     - virtualization_type
  31     - virtualization_role
  32     - container (e.g. solaris zones, freebsd jails, linux containers)
  33 
  34     All subclasses MUST define platform.
  35     """
  36     platform = 'Generic'
  37 
  38     # FIXME: remove load_on_init if we can
  39     def __init__(self, module, load_on_init=False):
  40         self.module = module
  41 
  42     # FIXME: just here for existing tests cases till they are updated
  43     def populate(self, collected_facts=None):
  44         virtual_facts = self.get_virtual_facts()
  45 
  46         return virtual_facts
  47 
  48     def get_virtual_facts(self):
  49         virtual_facts = {'virtualization_type': '',
  50                          'virtualization_role': ''}
  51         return virtual_facts
  52

  54 class VirtualCollector(BaseFactCollector):
  55     name = 'virtual'
  56     _fact_class = Virtual
  57     _fact_ids = set(['virtualization_type',
  58                      'virtualization_role'])
  59 
  60     def collect(self, module=None, collected_facts=None):
  61         collected_facts = collected_facts or {}
  62         if not module:
  63             return {}
  64 
  65         # Network munges cached_facts by side effect, so give it a copy
  66         facts_obj = self._fact_class(module)
  67 
  68         facts_dict = facts_obj.populate(collected_facts=collected_facts)
  69 
  70         return facts_dict
