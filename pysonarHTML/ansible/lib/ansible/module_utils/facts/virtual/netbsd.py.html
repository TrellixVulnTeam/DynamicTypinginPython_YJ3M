  25 class NetBSDVirtual(Virtual, VirtualSysctlDetectionMixin):
  26     platform = 'NetBSD'
  27 
  28     def get_virtual_facts(self):
  29         virtual_facts = {}
  30         # Set empty values as default
  31         virtual_facts['virtualization_type'] = ''
  32         virtual_facts['virtualization_role'] = ''
  33 
  34         virtual_product_facts = self.detect_virt_product('machdep.dmi.system-product')
  35         virtual_facts.update(virtual_product_facts)
  36 
  37         if virtual_facts['virtualization_type'] == '':
  38             virtual_vendor_facts = self.detect_virt_vendor('machdep.dmi.system-vendor')
  39             virtual_facts.update(virtual_vendor_facts)
  40 
  41         # The above logic is tried first for backwards compatibility. If
  42         # something above matches, use it. Otherwise if the result is still
  43         # empty, try machdep.hypervisor.
  44         if virtual_facts['virtualization_type'] == '':
  45             virtual_vendor_facts = self.detect_virt_vendor('machdep.hypervisor')
  46             virtual_facts.update(virtual_vendor_facts)
  47 
  48         if (virtual_facts['virtualization_type'] == '' and
  49                 os.path.exists('/dev/xencons')):
  50             virtual_facts['virtualization_type'] = 'xen'
  51             virtual_facts['virtualization_role'] = 'guest'
  52 
  53         return virtual_facts
  54 
  55 
  56 class NetBSDVirtualCollector(VirtualCollector):
  57     _fact_class = NetBSDVirtual
  58     _platform = 'NetBSD'
