  24 class SunOSVirtual(Virtual):
  25     """
  26     This is a SunOS-specific subclass of Virtual.  It defines
  27     - virtualization_type
  28     - virtualization_role
  29     - container
  30     """
  31     platform = 'SunOS'
  32 
  33     def get_virtual_facts(self):
  34         virtual_facts = {}
  35         # Check if it's a zone
  36 
  37         zonename = self.module.get_bin_path('zonename')
  38         if zonename:
  39             rc, out, err = self.module.run_command(zonename)
  40             if rc == 0 and out.rstrip() != "global":
  41                 virtual_facts['container'] = 'zone'
  42         # Check if it's a branded zone (i.e. Solaris 8/9 zone)
  43         if os.path.isdir('/.SUNWnative'):
  44             virtual_facts['container'] = 'zone'
  45         # If it's a zone check if we can detect if our global zone is itself virtualized.
  46         # Relies on the "guest tools" (e.g. vmware tools) to be installed
  47 
  48         if 'container' in virtual_facts and virtual_facts['container'] == 'zone':
  49             modinfo = self.module.get_bin_path('modinfo')
  50             if modinfo:
  51                 rc, out, err = self.module.run_command(modinfo)
  52                 if rc == 0:
  53                     for line in out.splitlines():
  54                         if 'VMware' in line:
  55                             virtual_facts['virtualization_type'] = 'vmware'
  56                             virtual_facts['virtualization_role'] = 'guest'
  57                         if 'VirtualBox' in line:
  58                             virtual_facts['virtualization_type'] = 'virtualbox'
  59                             virtual_facts['virtualization_role'] = 'guest'
  60 
  61         if os.path.exists('/proc/vz'):
  62             virtual_facts['virtualization_type'] = 'virtuozzo'
  63             virtual_facts['virtualization_role'] = 'guest'
  64 
  65         # Detect domaining on Sparc hardware
  66         virtinfo = self.module.get_bin_path('virtinfo')
  67         if virtinfo:
  68             # The output of virtinfo is different whether we are on a machine with logical
  69             # domains ('LDoms') on a T-series or domains ('Domains') on a M-series. Try LDoms first.
  70             rc, out, err = self.module.run_command("/usr/sbin/virtinfo -p")
  71             # The output contains multiple lines with different keys like this:
  72             #   DOMAINROLE|impl=LDoms|control=false|io=false|service=false|root=false
  73             # The output may also be not formatted and the returncode is set to 0 regardless of the error condition:
  74             #   virtinfo can only be run from the global zone
  75             if rc == 0:
  76                 try:
  77                     for line in out.splitlines():
  78                         fields = line.split('|')
  79                         if fields[0] == 'DOMAINROLE' and fields[1] == 'impl=LDoms':
  80                             virtual_facts['virtualization_type'] = 'ldom'
  81                             virtual_facts['virtualization_role'] = 'guest'
  82                             hostfeatures = []
  83                             for field in fields[2:]:
  84                                 arg = field.split('=')
  85                                 if arg[1] == 'true':
  86                                     hostfeatures.append(arg[0])
  87                             if len(hostfeatures) > 0:
  88                                 virtual_facts['virtualization_role'] = 'host (' + ','.join(hostfeatures) + ')'
  89                 except ValueError:
  90                     pass
  91 
  92         else:
  93             smbios = self.module.get_bin_path('smbios')
  94             if not smbios:
  95                 return
  96             rc, out, err = self.module.run_command(smbios)
  97             if rc == 0:
  98                 for line in out.splitlines():
  99                     if 'VMware' in line:
 100                         virtual_facts['virtualization_type'] = 'vmware'
 101                         virtual_facts['virtualization_role'] = 'guest'
 102                     elif 'Parallels' in line:
 103                         virtual_facts['virtualization_type'] = 'parallels'
 104                         virtual_facts['virtualization_role'] = 'guest'
 105                     elif 'VirtualBox' in line:
 106                         virtual_facts['virtualization_type'] = 'virtualbox'
 107                         virtual_facts['virtualization_role'] = 'guest'
 108                     elif 'HVM domU' in line:
 109                         virtual_facts['virtualization_type'] = 'xen'
 110                         virtual_facts['virtualization_role'] = 'guest'
 111                     elif 'KVM' in line:
 112                         virtual_facts['virtualization_type'] = 'kvm'
 113                         virtual_facts['virtualization_role'] = 'guest'
 114 
 115         return virtual_facts
