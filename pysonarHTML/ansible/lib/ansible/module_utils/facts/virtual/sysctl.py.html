  22 class VirtualSysctlDetectionMixin(object):
  23     def detect_sysctl(self):
  24         self.sysctl_path = self.module.get_bin_path('sysctl')
  25 
  26     def detect_virt_product(self, key):
  27         virtual_product_facts = {}
  28         self.detect_sysctl()
  29         if self.sysctl_path:
  30             rc, out, err = self.module.run_command("%s -n %s" % (self.sysctl_path, key))
  31             if rc == 0:
  32                 if re.match('(KVM|kvm|Bochs|SmartDC).*', out):
  33                     virtual_product_facts['virtualization_type'] = 'kvm'
  34                     virtual_product_facts['virtualization_role'] = 'guest'
  35                 elif re.match('.*VMware.*', out):
  36                     virtual_product_facts['virtualization_type'] = 'VMware'
  37                     virtual_product_facts['virtualization_role'] = 'guest'
  38                 elif out.rstrip() == 'VirtualBox':
  39                     virtual_product_facts['virtualization_type'] = 'virtualbox'
  40                     virtual_product_facts['virtualization_role'] = 'guest'
  41                 elif re.match('(HVM domU|XenPVH|XenPV|XenPVHVM).*', out):
  42                     virtual_product_facts['virtualization_type'] = 'xen'
  43                     virtual_product_facts['virtualization_role'] = 'guest'
  44                 elif out.rstrip() == 'Hyper-V':
  45                     virtual_product_facts['virtualization_type'] = 'Hyper-V'
  46                     virtual_product_facts['virtualization_role'] = 'guest'
  47                 elif out.rstrip() == 'Parallels':
  48                     virtual_product_facts['virtualization_type'] = 'parallels'
  49                     virtual_product_facts['virtualization_role'] = 'guest'
  50                 elif out.rstrip() == 'RHEV Hypervisor':
  51                     virtual_product_facts['virtualization_type'] = 'RHEV'
  52                     virtual_product_facts['virtualization_role'] = 'guest'
  53                 elif (key == 'security.jail.jailed') and (out.rstrip() == '1'):
  54                     virtual_product_facts['virtualization_type'] = 'jails'
  55                     virtual_product_facts['virtualization_role'] = 'guest'
  56 
  57         return virtual_product_facts
  58 
  59     def detect_virt_vendor(self, key):
  60         virtual_vendor_facts = {}
  61         self.detect_sysctl()
  62         if self.sysctl_path:
  63             rc, out, err = self.module.run_command("%s -n %s" % (self.sysctl_path, key))
  64             if rc == 0:
  65                 if out.rstrip() == 'QEMU':
  66                     virtual_vendor_facts['virtualization_type'] = 'kvm'
  67                     virtual_vendor_facts['virtualization_role'] = 'guest'
  68                 if out.rstrip() == 'OpenBSD':
  69                     virtual_vendor_facts['virtualization_type'] = 'vmm'
  70                     virtual_vendor_facts['virtualization_role'] = 'guest'
  71 
  72         return virtual_vendor_facts
