  27 class LinuxVirtual(Virtual):
  28     """
  29     This is a Linux-specific subclass of Virtual.  It defines
  30     - virtualization_type
  31     - virtualization_role
  32     """
  33     platform = 'Linux'
  34 
  35     # For more information, check: http://people.redhat.com/~rjones/virt-what/
  36     def get_virtual_facts(self):
  37         virtual_facts = {}
  38         # lxc/docker
  39         if os.path.exists('/proc/1/cgroup'):
  40             for line in get_file_lines('/proc/1/cgroup'):
  41                 if re.search(r'/docker(/|-[0-9a-f]+\.scope)', line):
  42                     virtual_facts['virtualization_type'] = 'docker'
  43                     virtual_facts['virtualization_role'] = 'guest'
  44                     return virtual_facts
  45                 if re.search('/lxc/', line) or re.search('/machine.slice/machine-lxc', line):
  46                     virtual_facts['virtualization_type'] = 'lxc'
  47                     virtual_facts['virtualization_role'] = 'guest'
  48                     return virtual_facts
  49 
  50         # lxc does not always appear in cgroups anymore but sets 'container=lxc' environment var, requires root privs
  51         if os.path.exists('/proc/1/environ'):
  52             for line in get_file_lines('/proc/1/environ', line_sep='\x00'):
  53                 if re.search('container=lxc', line):
  54                     virtual_facts['virtualization_type'] = 'lxc'
  55                     virtual_facts['virtualization_role'] = 'guest'
  56                     return virtual_facts
  57                 if re.search('container=podman', line):
  58                     virtual_facts['virtualization_type'] = 'podman'
  59                     virtual_facts['virtualization_role'] = 'guest'
  60                     return virtual_facts
  61                 if re.search('^container=.', line):
  62                     virtual_facts['virtualization_type'] = 'container'
  63                     virtual_facts['virtualization_role'] = 'guest'
  64                     return virtual_facts
  65 
  66         if os.path.exists('/proc/vz') and not os.path.exists('/proc/lve'):
  67             virtual_facts['virtualization_type'] = 'openvz'
  68             if os.path.exists('/proc/bc'):
  69                 virtual_facts['virtualization_role'] = 'host'
  70             else:
  71                 virtual_facts['virtualization_role'] = 'guest'
  72             return virtual_facts
  73 
  74         systemd_container = get_file_content('/run/systemd/container')
  75         if systemd_container:
  76             virtual_facts['virtualization_type'] = systemd_container
  77             virtual_facts['virtualization_role'] = 'guest'
  78             return virtual_facts
  79 
  80         if os.path.exists("/proc/xen"):
  81             virtual_facts['virtualization_type'] = 'xen'
  82             virtual_facts['virtualization_role'] = 'guest'
  83             try:
  84                 for line in get_file_lines('/proc/xen/capabilities'):
  85                     if "control_d" in line:
  86                         virtual_facts['virtualization_role'] = 'host'
  87             except IOError:
  88                 pass
  89             return virtual_facts
  90 
  91         # assume guest for this block
  92         virtual_facts['virtualization_role'] = 'guest'
  93 
  94         product_name = get_file_content('/sys/devices/virtual/dmi/id/product_name')
  95 
  96         if product_name in ('KVM', 'KVM Server', 'Bochs', 'AHV'):
  97             virtual_facts['virtualization_type'] = 'kvm'
  98             return virtual_facts
  99 
 100         if product_name == 'RHEV Hypervisor':
 101             virtual_facts['virtualization_type'] = 'RHEV'
 102             return virtual_facts
 103 
 104         if product_name in ('VMware Virtual Platform', 'VMware7,1'):
 105             virtual_facts['virtualization_type'] = 'VMware'
 106             return virtual_facts
 107 
 108         if product_name in ('OpenStack Compute', 'OpenStack Nova'):
 109             virtual_facts['virtualization_type'] = 'openstack'
 110             return virtual_facts
 111 
 112         bios_vendor = get_file_content('/sys/devices/virtual/dmi/id/bios_vendor')
 113 
 114         if bios_vendor == 'Xen':
 115             virtual_facts['virtualization_type'] = 'xen'
 116             return virtual_facts
 117 
 118         if bios_vendor == 'innotek GmbH':
 119             virtual_facts['virtualization_type'] = 'virtualbox'
 120             return virtual_facts
 121 
 122         if bios_vendor in ('Amazon EC2', 'DigitalOcean', 'Hetzner'):
 123             virtual_facts['virtualization_type'] = 'kvm'
 124             return virtual_facts
 125 
 126         sys_vendor = get_file_content('/sys/devices/virtual/dmi/id/sys_vendor')
 127 
 128         KVM_SYS_VENDORS = ('QEMU', 'oVirt', 'Amazon EC2', 'DigitalOcean', 'Google', 'Scaleway', 'Nutanix')
 129         if sys_vendor in KVM_SYS_VENDORS:
 130             virtual_facts['virtualization_type'] = 'kvm'
 131             return virtual_facts
 132 
 133         # FIXME: This does also match hyperv
 134         if sys_vendor == 'Microsoft Corporation':
 135             virtual_facts['virtualization_type'] = 'VirtualPC'
 136             return virtual_facts
 137 
 138         if sys_vendor == 'Parallels Software International Inc.':
 139             virtual_facts['virtualization_type'] = 'parallels'
 140             return virtual_facts
 141 
 142         if sys_vendor == 'OpenStack Foundation':
 143             virtual_facts['virtualization_type'] = 'openstack'
 144             return virtual_facts
 145 
 146         # unassume guest
 147         del virtual_facts['virtualization_role']
 148 
 149         if os.path.exists('/proc/self/status'):
 150             for line in get_file_lines('/proc/self/status'):
 151                 if re.match(r'^VxID:\s+\d+', line):
 152                     virtual_facts['virtualization_type'] = 'linux_vserver'
 153                     if re.match(r'^VxID:\s+0', line):
 154                         virtual_facts['virtualization_role'] = 'host'
 155                     else:
 156                         virtual_facts['virtualization_role'] = 'guest'
 157                     return virtual_facts
 158 
 159         if os.path.exists('/proc/cpuinfo'):
 160             for line in get_file_lines('/proc/cpuinfo'):
 161                 if re.match('^model name.*QEMU Virtual CPU', line):
 162                     virtual_facts['virtualization_type'] = 'kvm'
 163                 elif re.match('^vendor_id.*User Mode Linux', line):
 164                     virtual_facts['virtualization_type'] = 'uml'
 165                 elif re.match('^model name.*UML', line):
 166                     virtual_facts['virtualization_type'] = 'uml'
 167                 elif re.match('^machine.*CHRP IBM pSeries .emulated by qemu.', line):
 168                     virtual_facts['virtualization_type'] = 'kvm'
 169                 elif re.match('^vendor_id.*PowerVM Lx86', line):
 170                     virtual_facts['virtualization_type'] = 'powervm_lx86'
 171                 elif re.match('^vendor_id.*IBM/S390', line):
 172                     virtual_facts['virtualization_type'] = 'PR/SM'
 173                     lscpu = self.module.get_bin_path('lscpu')
 174                     if lscpu:
 175                         rc, out, err = self.module.run_command(["lscpu"])
 176                         if rc == 0:
 177                             for line in out.splitlines():
 178                                 data = line.split(":", 1)
 179                                 key = data[0].strip()
 180                                 if key == 'Hypervisor':
 181                                     virtual_facts['virtualization_type'] = data[1].strip()
 182                     else:
 183                         virtual_facts['virtualization_type'] = 'ibm_systemz'
 184                 else:
 185                     continue
 186                 if virtual_facts['virtualization_type'] == 'PR/SM':
 187                     virtual_facts['virtualization_role'] = 'LPAR'
 188                 else:
 189                     virtual_facts['virtualization_role'] = 'guest'
 190                 return virtual_facts
 191 
 192         # Beware that we can have both kvm and virtualbox running on a single system
 193         if os.path.exists("/proc/modules") and os.access('/proc/modules', os.R_OK):
 194             modules = []
 195             for line in get_file_lines("/proc/modules"):
 196                 data = line.split(" ", 1)
 197                 modules.append(data[0])
 198 
 199             if 'kvm' in modules:
 200                 virtual_facts['virtualization_type'] = 'kvm'
 201                 virtual_facts['virtualization_role'] = 'host'
 202 
 203                 if os.path.isdir('/rhev/'):
 204                     # Check whether this is a RHEV hypervisor (is vdsm running ?)
 205                     for f in glob.glob('/proc/[0-9]*/comm'):
 206                         try:
 207                             with open(f) as virt_fh:
 208                                 comm_content = virt_fh.read().rstrip()
 209                             if comm_content == 'vdsm':
 210                                 virtual_facts['virtualization_type'] = 'RHEV'
 211                                 break
 212                         except Exception:
 213                             pass
 214 
 215                 return virtual_facts
 216 
 217             if 'vboxdrv' in modules:
 218                 virtual_facts['virtualization_type'] = 'virtualbox'
 219                 virtual_facts['virtualization_role'] = 'host'
 220                 return virtual_facts
 221 
 222             if 'virtio' in modules:
 223                 virtual_facts['virtualization_type'] = 'kvm'
 224                 virtual_facts['virtualization_role'] = 'guest'
 225                 return virtual_facts
 226 
 227         # In older Linux Kernel versions, /sys filesystem is not available
 228         # dmidecode is the safest option to parse virtualization related values
 229         dmi_bin = self.module.get_bin_path('dmidecode')
 230         # We still want to continue even if dmidecode is not available
 231         if dmi_bin is not None:
 232             (rc, out, err) = self.module.run_command('%s -s system-product-name' % dmi_bin)
 233             if rc == 0:
 234                 # Strip out commented lines (specific dmidecode output)
 235                 vendor_name = ''.join([line.strip() for line in out.splitlines() if not line.startswith('#')])
 236                 if vendor_name.startswith('VMware'):
 237                     virtual_facts['virtualization_type'] = 'VMware'
 238                     virtual_facts['virtualization_role'] = 'guest'
 239                     return virtual_facts
 240 
 241         # If none of the above matches, return 'NA' for virtualization_type
 242         # and virtualization_role. This allows for proper grouping.
 243         virtual_facts['virtualization_type'] = 'NA'
 244         virtual_facts['virtualization_role'] = 'NA'
 245 
 246         return virtual_facts
 247

 249 class LinuxVirtualCollector(VirtualCollector):
 250     _fact_class = LinuxVirtual
 251     _platform = 'Linux'
