  27 class OpenBSDVirtual(Virtual, VirtualSysctlDetectionMixin):
  28     """
  29     This is a OpenBSD-specific subclass of Virtual.  It defines
  30     - virtualization_type
  31     - virtualization_role
  32     """
  33     platform = 'OpenBSD'
  34     DMESG_BOOT = '/var/run/dmesg.boot'
  35 
  36     def get_virtual_facts(self):
  37         virtual_facts = {}
  38 
  39         # Set empty values as default
  40         virtual_facts['virtualization_type'] = ''
  41         virtual_facts['virtualization_role'] = ''
  42 
  43         virtual_product_facts = self.detect_virt_product('hw.product')
  44         virtual_facts.update(virtual_product_facts)
  45 
  46         if virtual_facts['virtualization_type'] == '':
  47             virtual_vendor_facts = self.detect_virt_vendor('hw.vendor')
  48             virtual_facts.update(virtual_vendor_facts)
  49 
  50         # Check the dmesg if vmm(4) attached, indicating the host is
  51         # capable of virtualization.
  52         dmesg_boot = get_file_content(OpenBSDVirtual.DMESG_BOOT)
  53         for line in dmesg_boot.splitlines():
  54             match = re.match('^vmm0 at mainbus0: (SVM/RVI|VMX/EPT)$', line)
  55             if match:
  56                 virtual_facts['virtualization_type'] = 'vmm'
  57                 virtual_facts['virtualization_role'] = 'host'
  58 
  59         return virtual_facts
  60 
  61 
  62 class OpenBSDVirtualCollector(VirtualCollector):
  63     _fact_class = OpenBSDVirtual
  64     _platform = 'OpenBSD'
