  37 def get_all_facts(module):
  38     '''compat api for ansible 2.2/2.3 module_utils.facts.get_all_facts method
  39 
  40     Expects module to be an instance of AnsibleModule, with a 'gather_subset' param.
  41 
  42     returns a dict mapping the bare fact name ('default_ipv4' with no 'ansible_' namespace) to
  43     the fact value.'''
  44 
  45     gather_subset = module.params['gather_subset']
  46     return ansible_facts(module, gather_subset=gather_subset)

  49 def ansible_facts(module, gather_subset=None):
  50     '''Compat api for ansible 2.0/2.2/2.3 module_utils.facts.ansible_facts method
  51 
  52     2.3/2.3 expects a gather_subset arg.
  53     2.0/2.1 does not except a gather_subset arg
  54 
  55     So make gather_subsets an optional arg, defaulting to configured DEFAULT_GATHER_TIMEOUT
  56 
  57     'module' should be an instance of an AnsibleModule.
  58 
  59     returns a dict mapping the bare fact name ('default_ipv4' with no 'ansible_' namespace) to
  60     the fact value.
  61     '''
  62 
  63     gather_subset = gather_subset or module.params.get('gather_subset', ['all'])
  64     gather_timeout = module.params.get('gather_timeout', 10)
  65     filter_spec = module.params.get('filter', '*')
  66 
  67     minimal_gather_subset = frozenset(['apparmor', 'caps', 'cmdline', 'date_time',
  68                                        'distribution', 'dns', 'env', 'fips', 'local',
  69                                        'lsb', 'pkg_mgr', 'platform', 'python', 'selinux',
  70                                        'service_mgr', 'ssh_pub_keys', 'user'])
  71 
  72     all_collector_classes = default_collectors.collectors
  73 
  74     # don't add a prefix
  75     namespace = PrefixFactNamespace(namespace_name='ansible', prefix='')
  76 
  77     fact_collector = \
  78         ansible_collector.get_ansible_collector(all_collector_classes=all_collector_classes,
  79                                                 namespace=namespace,
  80                                                 filter_spec=filter_spec,
  81                                                 gather_subset=gather_subset,
  82                                                 gather_timeout=gather_timeout,
  83                                                 minimal_gather_subset=minimal_gather_subset)
  84 
  85     facts_dict = fact_collector.collect(module=module)
  86 
  87     return facts_dict
