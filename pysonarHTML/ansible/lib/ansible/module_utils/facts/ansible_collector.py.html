  39 class AnsibleFactCollector(collector.BaseFactCollector):
  40     '''A FactCollector that returns results under 'ansible_facts' top level key.
  41 
  42        If a namespace if provided, facts will be collected under that namespace.
  43        For ex, a ansible.module_utils.facts.namespace.PrefixFactNamespace(prefix='ansible_')
  44 
  45        Has a 'from_gather_subset() constructor that populates collectors based on a
  46        gather_subset specifier.'''
  47 
  48     def __init__(self, collectors=None, namespace=None, filter_spec=None):
  49 
  50         super(AnsibleFactCollector, self).__init__(collectors=collectors,
  51                                                    namespace=namespace)
  52 
  53         self.filter_spec = filter_spec
  54 
  55     def _filter(self, facts_dict, filter_spec):
  56         # assume a filter_spec='' is equilv to filter_spec='*'
  57         if not filter_spec or filter_spec == '*':
  58             return facts_dict
  59 
  60         return [(x, y) for x, y in facts_dict.items() if fnmatch.fnmatch(x, filter_spec)]
  61 
  62     def collect(self, module=None, collected_facts=None):
  63         collected_facts = collected_facts or {}
  64 
  65         facts_dict = {}
  66 
  67         for collector_obj in self.collectors:
  68             info_dict = {}
  69 
  70             try:
  71 
  72                 # Note: this collects with namespaces, so collected_facts also includes namespaces
  73                 info_dict = collector_obj.collect_with_namespace(module=module,
  74                                                                  collected_facts=collected_facts)
  75             except Exception as e:
  76                 sys.stderr.write(repr(e))
  77                 sys.stderr.write('\n')
  78 
  79             # shallow copy of the new facts to pass to each collector in collected_facts so facts
  80             # can reference other facts they depend on.
  81             collected_facts.update(info_dict.copy())
  82 
  83             # NOTE: If we want complicated fact dict merging, this is where it would hook in
  84             facts_dict.update(self._filter(info_dict, self.filter_spec))
  85 
  86         return facts_dict

  89 class CollectorMetaDataCollector(collector.BaseFactCollector):
  90     '''Collector that provides a facts with the gather_subset metadata.'''
  91 
  92     name = 'gather_subset'
  93     _fact_ids = set([])
  94 
  95     def __init__(self, collectors=None, namespace=None, gather_subset=None, module_setup=None):
  96         super(CollectorMetaDataCollector, self).__init__(collectors, namespace)
  97         self.gather_subset = gather_subset
  98         self.module_setup = module_setup
  99 
 100     def collect(self, module=None, collected_facts=None):
 101         meta_facts = {'gather_subset': self.gather_subset}
 102         if self.module_setup:
 103             meta_facts['module_setup'] = self.module_setup
 104         return meta_facts

 107 def get_ansible_collector(all_collector_classes,
 108                           namespace=None,
 109                           filter_spec=None,
 110                           gather_subset=None,
 111                           gather_timeout=None,
 112                           minimal_gather_subset=None):
 113 
 114     filter_spec = filter_spec or '*'
 115     gather_subset = gather_subset or ['all']
 116     gather_timeout = gather_timeout or timeout.DEFAULT_GATHER_TIMEOUT
 117     minimal_gather_subset = minimal_gather_subset or frozenset()
 118 
 119     collector_classes = \
 120         collector.collector_classes_from_gather_subset(
 121             all_collector_classes=all_collector_classes,
 122             minimal_gather_subset=minimal_gather_subset,
 123             gather_subset=gather_subset,
 124             gather_timeout=gather_timeout)
 125 
 126     collectors = []
 127     for collector_class in collector_classes:
 128         collector_obj = collector_class(namespace=namespace)
 129         collectors.append(collector_obj)
 130 
 131     # Add a collector that knows what gather_subset we used so it it can provide a fact
 132     collector_meta_data_collector = \
 133         CollectorMetaDataCollector(gather_subset=gather_subset,
 134                                    module_setup=True)
 135     collectors.append(collector_meta_data_collector)
 136 
 137     fact_collector = \
 138         AnsibleFactCollector(collectors=collectors,
 139                              filter_spec=filter_spec,
 140                              namespace=namespace)
 141 
 142     return fact_collector
