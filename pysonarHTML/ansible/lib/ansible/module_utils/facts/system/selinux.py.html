  21 from ansible.module_utils.facts.collector import BaseFactCollector
  22 
  23 try:
  24     import selinux
  25     HAVE_SELINUX = True
  26 except ImportError:
  27     HAVE_SELINUX = False
  28 
  29 SELINUX_MODE_DICT = {
  30     1: 'enforcing',
  31     0: 'permissive',
  32     -1: 'disabled'
  33 }
  34 
  35 
  36 class SelinuxFactCollector(BaseFactCollector):
  37     name = 'selinux'
  38     _fact_ids = set()
  39 
  40     def collect(self, module=None, collected_facts=None):
  41         facts_dict = {}
  42         selinux_facts = {}
  43 
  44         # If selinux library is missing, only set the status and selinux_python_present since
  45         # there is no way to tell if SELinux is enabled or disabled on the system
  46         # without the library.
  47         if not HAVE_SELINUX:
  48             selinux_facts['status'] = 'Missing selinux Python library'
  49             facts_dict['selinux'] = selinux_facts
  50             facts_dict['selinux_python_present'] = False
  51             return facts_dict
  52 
  53         # Set a boolean for testing whether the Python library is present
  54         facts_dict['selinux_python_present'] = True
  55 
  56         if not selinux.is_selinux_enabled():
  57             selinux_facts['status'] = 'disabled'
  58         else:
  59             selinux_facts['status'] = 'enabled'
  60 
  61             try:
  62                 selinux_facts['policyvers'] = selinux.security_policyvers()
  63             except (AttributeError, OSError):
  64                 selinux_facts['policyvers'] = 'unknown'
  65 
  66             try:
  67                 (rc, configmode) = selinux.selinux_getenforcemode()
  68                 if rc == 0:
  69                     selinux_facts['config_mode'] = SELINUX_MODE_DICT.get(configmode, 'unknown')
  70                 else:
  71                     selinux_facts['config_mode'] = 'unknown'
  72             except (AttributeError, OSError):
  73                 selinux_facts['config_mode'] = 'unknown'
  74 
  75             try:
  76                 mode = selinux.security_getenforce()
  77                 selinux_facts['mode'] = SELINUX_MODE_DICT.get(mode, 'unknown')
  78             except (AttributeError, OSError):
  79                 selinux_facts['mode'] = 'unknown'
  80 
  81             try:
  82                 (rc, policytype) = selinux.selinux_getpolicytype()
  83                 if rc == 0:
  84                     selinux_facts['type'] = policytype
  85                 else:
  86                     selinux_facts['type'] = 'unknown'
  87             except (AttributeError, OSError):
  88                 selinux_facts['type'] = 'unknown'
  89 
  90         facts_dict['selinux'] = selinux_facts
  91         return facts_dict
