  56 # the fact ends up being 'pkg_mgr' so stick with that naming/spelling
  57 class PkgMgrFactCollector(BaseFactCollector):
  58     name = 'pkg_mgr'
  59     _fact_ids = set()
  60     _platform = 'Generic'
  61     required_facts = set(['distribution'])
  62 
  63     def _check_rh_versions(self, pkg_mgr_name, collected_facts):
  64         if collected_facts['ansible_distribution'] == 'Fedora':
  65             if os.path.exists('/run/ostree-booted'):
  66                 return "atomic_container"
  67             try:
  68                 if int(collected_facts['ansible_distribution_major_version']) < 23:
  69                     for yum in [pkg_mgr for pkg_mgr in PKG_MGRS if pkg_mgr['name'] == 'yum']:
  70                         if os.path.exists(yum['path']):
  71                             pkg_mgr_name = 'yum'
  72                             break
  73                 else:
  74                     for dnf in [pkg_mgr for pkg_mgr in PKG_MGRS if pkg_mgr['name'] == 'dnf']:
  75                         if os.path.exists(dnf['path']):
  76                             pkg_mgr_name = 'dnf'
  77                             break
  78             except ValueError:
  79                 # If there's some new magical Fedora version in the future,
  80                 # just default to dnf
  81                 pkg_mgr_name = 'dnf'
  82         elif collected_facts['ansible_distribution'] == 'Amazon':
  83             pkg_mgr_name = 'yum'
  84         else:
  85             # If it's not one of the above and it's Red Hat family of distros, assume
  86             # RHEL or a clone. For versions of RHEL < 8 that Ansible supports, the
  87             # vendor supported official package manager is 'yum' and in RHEL 8+
  88             # (as far as we know at the time of this writing) it is 'dnf'.
  89             # If anyone wants to force a non-official package manager then they
  90             # can define a provider to either the package or yum action plugins.
  91             if int(collected_facts['ansible_distribution_major_version']) < 8:
  92                 pkg_mgr_name = 'yum'
  93             else:
  94                 pkg_mgr_name = 'dnf'
  95         return pkg_mgr_name

  97     def _check_apt_flavor(self, pkg_mgr_name):
  98         # Check if '/usr/bin/apt' is APT-RPM or an ordinary (dpkg-based) APT.
  99         # There's rpm package on Debian, so checking if /usr/bin/rpm exists
 100         # is not enough. Instead ask RPM if /usr/bin/apt-get belongs to some
 101         # RPM package.
 102         rpm_query = '/usr/bin/rpm -q --whatprovides /usr/bin/apt-get'.split()
 103         if os.path.exists('/usr/bin/rpm'):
 104             with open(os.devnull, 'w') as null:
 105                 try:
 106                     subprocess.check_call(rpm_query, stdout=null, stderr=null)
 107                     pkg_mgr_name = 'apt_rpm'
 108                 except subprocess.CalledProcessError:
 109                     # No apt-get in RPM database. Looks like Debian/Ubuntu
 110                     # with rpm package installed
 111                     pkg_mgr_name = 'apt'
 112         return pkg_mgr_name
 113 
 114     def collect(self, module=None, collected_facts=None):
 115         facts_dict = {}
 116         collected_facts = collected_facts or {}
 117 
 118         pkg_mgr_name = 'unknown'
 119         for pkg in PKG_MGRS:
 120             if os.path.exists(pkg['path']):
 121                 pkg_mgr_name = pkg['name']
 122 
 123         # Handle distro family defaults when more than one package manager is
 124         # installed or available to the distro, the ansible_fact entry should be
 125         # the default package manager officially supported by the distro.
 126         if collected_facts['ansible_os_family'] == "RedHat":
 127             pkg_mgr_name = self._check_rh_versions(pkg_mgr_name, collected_facts)
 128         elif collected_facts['ansible_os_family'] == 'Debian' and pkg_mgr_name != 'apt':
 129             # It's possible to install yum, dnf, zypper, rpm, etc inside of
 130             # Debian. Doing so does not mean the system wants to use them.
 131             pkg_mgr_name = 'apt'
 132         elif collected_facts['ansible_os_family'] == 'Altlinux':
 133             if pkg_mgr_name == 'apt':
 134                 pkg_mgr_name = 'apt_rpm'
 135 
 136         # Check if /usr/bin/apt-get is ordinary (dpkg-based) APT or APT-RPM
 137         if pkg_mgr_name == 'apt':
 138             pkg_mgr_name = self._check_apt_flavor(pkg_mgr_name)
 139 
 140         facts_dict['pkg_mgr'] = pkg_mgr_name
 141         return facts_dict
