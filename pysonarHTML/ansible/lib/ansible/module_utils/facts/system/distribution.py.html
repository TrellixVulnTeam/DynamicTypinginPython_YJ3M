  19 def get_uname(module, flags=('-v')):
  20     if isinstance(flags, str):
  21         flags = flags.split()
  22     command = ['uname']
  23     command.extend(flags)
  24     rc, out, err = module.run_command(command)
  25     if rc == 0:
  26         return out
  27     return None

  30 def _file_exists(path, allow_empty=False):
  31     # not finding the file, exit early
  32     if not os.path.exists(path):
  33         return False
  34 
  35     # if just the path needs to exists (ie, it can be empty) we are done
  36     if allow_empty:
  37         return True
  38 
  39     # file exists but is empty and we dont allow_empty
  40     if os.path.getsize(path) == 0:
  41         return False
  42 
  43     # file exists with some content
  44     return True

 108     def _parse_dist_file(self, name, dist_file_content, path, collected_facts):
 109         dist_file_dict = {}
 110         dist_file_content = dist_file_content.strip(DistributionFiles.STRIP_QUOTES)
 111         if name in self.SEARCH_STRING:
 112             # look for the distribution string in the data and replace according to RELEASE_NAME_MAP
 113             # only the distribution name is set, the version is assumed to be correct from distro.linux_distribution()
 114             if self.SEARCH_STRING[name] in dist_file_content:
 115                 # this sets distribution=RedHat if 'Red Hat' shows up in data
 116                 dist_file_dict['distribution'] = name
 117                 dist_file_dict['distribution_file_search_string'] = self.SEARCH_STRING[name]
 118             else:
 119                 # this sets distribution to what's in the data, e.g. CentOS, Scientific, ...
 120                 dist_file_dict['distribution'] = dist_file_content.split()[0]
 121 
 122             return True, dist_file_dict
 123 
 124         if name in self.OS_RELEASE_ALIAS:
 125             if self.OS_RELEASE_ALIAS[name] in dist_file_content:
 126                 dist_file_dict['distribution'] = name
 127                 return True, dist_file_dict
 128             return False, dist_file_dict
 129 
 130         # call a dedicated function for parsing the file content
 131         # TODO: replace with a map or a class
 132         try:
 133             # FIXME: most of these dont actually look at the dist file contents, but random other stuff
 134             distfunc_name = 'parse_distribution_file_' + name
 135             distfunc = getattr(self, distfunc_name)
 136             parsed, dist_file_dict = distfunc(name, dist_file_content, path, collected_facts)
 137             return parsed, dist_file_dict
 138         except AttributeError as exc:
 139             self.module.debug('exc: %s' % exc)
 140             # this should never happen, but if it does fail quietly and not with a traceback
 141             return False, dist_file_dict
 142 
 143         return True, dist_file_dict
 144         # to debug multiple matching release files, one can use:
 145         # self.facts['distribution_debug'].append({path + ' ' + name:
 146         #         (parsed,
 147         #          self.facts['distribution'],
 148         #          self.facts['distribution_version'],
 149         #          self.facts['distribution_release'],
 150         #          )})

 152     def _guess_distribution(self):
 153         # try to find out which linux distribution this is
 154         dist = (get_distribution(), get_distribution_version(), get_distribution_codename())
 155         distribution_guess = {
 156             'distribution': dist[0] or 'NA',
 157             'distribution_version': dist[1] or 'NA',
 158             # distribution_release can be the empty string
 159             'distribution_release': 'NA' if dist[2] is None else dist[2]
 160         }
 161 
 162         distribution_guess['distribution_major_version'] = distribution_guess['distribution_version'].split('.')[0] or 'NA'
 163         return distribution_guess

 165     def process_dist_files(self):
 166         # Try to handle the exceptions now ...
 167         # self.facts['distribution_debug'] = []
 168         dist_file_facts = {}
 169 
 170         dist_guess = self._guess_distribution()
 171         dist_file_facts.update(dist_guess)
 172 
 173         for ddict in self.OSDIST_LIST:
 174             name = ddict['name']
 175             path = ddict['path']
 176             allow_empty = ddict.get('allowempty', False)
 177 
 178             has_dist_file, dist_file_content = self._get_dist_file_content(path, allow_empty=allow_empty)
 179 
 180             # but we allow_empty. For example, ArchLinux with an empty /etc/arch-release and a
 181             # /etc/os-release with a different name
 182             if has_dist_file and allow_empty:
 183                 dist_file_facts['distribution'] = name
 184                 dist_file_facts['distribution_file_path'] = path
 185                 dist_file_facts['distribution_file_variety'] = name
 186                 break
 187 
 188             if not has_dist_file:
 189                 # keep looking
 190                 continue
 191 
 192             parsed_dist_file, parsed_dist_file_facts = self._parse_dist_file(name, dist_file_content, path, dist_file_facts)
 193 
 194             # finally found the right os dist file and were able to parse it
 195             if parsed_dist_file:
 196                 dist_file_facts['distribution'] = name
 197                 dist_file_facts['distribution_file_path'] = path
 198                 # distribution and file_variety are the same here, but distribution
 199                 # will be changed/mapped to a more specific name.
 200                 # ie, dist=Fedora, file_variety=RedHat
 201                 dist_file_facts['distribution_file_variety'] = name
 202                 dist_file_facts['distribution_file_parsed'] = parsed_dist_file
 203                 dist_file_facts.update(parsed_dist_file_facts)
 204                 break
 205 
 206         return dist_file_facts

 248     def parse_distribution_file_SUSE(self, name, data, path, collected_facts):
 249         suse_facts = {}
 250         if 'suse' not in data.lower():
 251             return False, suse_facts  # TODO: remove if tested without this
 252         if path == '/etc/os-release':
 253             for line in data.splitlines():
 254                 distribution = re.search("^NAME=(.*)", line)
 255                 if distribution:
 256                     suse_facts['distribution'] = distribution.group(1).strip('"')
 257                 # example pattern are 13.04 13.0 13
 258                 distribution_version = re.search(r'^VERSION_ID="?([0-9]+\.?[0-9]*)"?', line)
 259                 if distribution_version:
 260                     suse_facts['distribution_version'] = distribution_version.group(1)
 261                     suse_facts['distribution_major_version'] = distribution_version.group(1).split('.')[0]
 262                 if 'open' in data.lower():
 263                     release = re.search(r'^VERSION_ID="?[0-9]+\.?([0-9]*)"?', line)
 264                     if release:
 265                         suse_facts['distribution_release'] = release.groups()[0]
 266                 elif 'enterprise' in data.lower() and 'VERSION_ID' in line:
 267                     # SLES doesn't got funny release names
 268                     release = re.search(r'^VERSION_ID="?[0-9]+\.?([0-9]*)"?', line)
 269                     if release.group(1):
 270                         release = release.group(1)
 271                     else:
 272                         release = "0"  # no minor number, so it is the first release
 273                     suse_facts['distribution_release'] = release
 274                 # Starting with SLES4SAP12 SP3 NAME reports 'SLES' instead of 'SLES_SAP'
 275                 # According to SuSe Support (SR101182877871) we should use the CPE_NAME to detect SLES4SAP
 276                 if re.search("^CPE_NAME=.*sles_sap.*$", line):
 277                     suse_facts['distribution'] = 'SLES_SAP'
 278         elif path == '/etc/SuSE-release':
 279             if 'open' in data.lower():
 280                 data = data.splitlines()
 281                 distdata = get_file_content(path).splitlines()[0]
 282                 suse_facts['distribution'] = distdata.split()[0]
 283                 for line in data:
 284                     release = re.search('CODENAME *= *([^\n]+)', line)
 285                     if release:
 286                         suse_facts['distribution_release'] = release.groups()[0].strip()
 287             elif 'enterprise' in data.lower():
 288                 lines = data.splitlines()
 289                 distribution = lines[0].split()[0]
 290                 if "Server" in data:
 291                     suse_facts['distribution'] = "SLES"
 292                 elif "Desktop" in data:
 293                     suse_facts['distribution'] = "SLED"
 294                 for line in lines:
 295                     release = re.search('PATCHLEVEL = ([0-9]+)', line)  # SLES doesn't got funny release names
 296                     if release:
 297                         suse_facts['distribution_release'] = release.group(1)
 298                         suse_facts['distribution_version'] = collected_facts['distribution_version'] + '.' + release.group(1)
 299 
 300         return True, suse_facts

 302     def parse_distribution_file_Debian(self, name, data, path, collected_facts):
 303         debian_facts = {}
 304         if 'Debian' in data or 'Raspbian' in data:
 305             debian_facts['distribution'] = 'Debian'
 306             release = re.search(r"PRETTY_NAME=[^(]+ \(?([^)]+?)\)", data)
 307             if release:
 308                 debian_facts['distribution_release'] = release.groups()[0]
 309 
 310             # Last resort: try to find release from tzdata as either lsb is missing or this is very old debian
 311             if collected_facts['distribution_release'] == 'NA' and 'Debian' in data:
 312                 dpkg_cmd = self.module.get_bin_path('dpkg')
 313                 if dpkg_cmd:
 314                     cmd = "%s --status tzdata|grep Provides|cut -f2 -d'-'" % dpkg_cmd
 315                     rc, out, err = self.module.run_command(cmd)
 316                     if rc == 0:
 317                         debian_facts['distribution_release'] = out.strip()
 318         elif 'Ubuntu' in data:
 319             debian_facts['distribution'] = 'Ubuntu'
 320             # nothing else to do, Ubuntu gets correct info from python functions
 321         elif 'SteamOS' in data:
 322             debian_facts['distribution'] = 'SteamOS'
 323             # nothing else to do, SteamOS gets correct info from python functions
 324         elif path in ('/etc/lsb-release', '/etc/os-release') and ('Kali' in data or 'Parrot' in data):
 325             if 'Kali' in data:
 326                 # Kali does not provide /etc/lsb-release anymore
 327                 debian_facts['distribution'] = 'Kali'
 328             elif 'Parrot' in data:
 329                 debian_facts['distribution'] = 'Parrot'
 330             release = re.search('DISTRIB_RELEASE=(.*)', data)
 331             if release:
 332                 debian_facts['distribution_release'] = release.groups()[0]
 333         elif 'Devuan' in data:
 334             debian_facts['distribution'] = 'Devuan'
 335             release = re.search(r"PRETTY_NAME=\"?[^(\"]+ \(?([^) \"]+)\)?", data)
 336             if release:
 337                 debian_facts['distribution_release'] = release.groups()[0]
 338             version = re.search(r"VERSION_ID=\"(.*)\"", data)
 339             if version:
 340                 debian_facts['distribution_version'] = version.group(1)
 341                 debian_facts['distribution_major_version'] = version.group(1)
 342         elif 'Cumulus' in data:
 343             debian_facts['distribution'] = 'Cumulus Linux'
 344             version = re.search(r"VERSION_ID=(.*)", data)
 345             if version:
 346                 major, _minor, _dummy_ver = version.group(1).split(".")
 347                 debian_facts['distribution_version'] = version.group(1)
 348                 debian_facts['distribution_major_version'] = major
 349 
 350             release = re.search(r'VERSION="(.*)"', data)
 351             if release:
 352                 debian_facts['distribution_release'] = release.groups()[0]
 353         elif "Mint" in data:
 354             debian_facts['distribution'] = 'Linux Mint'
 355             version = re.search(r"VERSION_ID=\"(.*)\"", data)
 356             if version:
 357                 debian_facts['distribution_version'] = version.group(1)
 358                 debian_facts['distribution_major_version'] = version.group(1).split('.')[0]
 359         else:
 360             return False, debian_facts
 361 
 362         return True, debian_facts

 519     def get_distribution_facts(self):
 520         distribution_facts = {}
 521 
 522         # The platform module provides information about the running
 523         # system/distribution. Use this as a baseline and fix buggy systems
 524         # afterwards
 525         system = platform.system()
 526         distribution_facts['distribution'] = system
 527         distribution_facts['distribution_release'] = platform.release()
 528         distribution_facts['distribution_version'] = platform.version()
 529 
 530         systems_implemented = ('AIX', 'HP-UX', 'Darwin', 'FreeBSD', 'OpenBSD', 'SunOS', 'DragonFly', 'NetBSD')
 531 
 532         if system in systems_implemented:
 533             cleanedname = system.replace('-', '')
 534             distfunc = getattr(self, 'get_distribution_' + cleanedname)
 535             dist_func_facts = distfunc()
 536             distribution_facts.update(dist_func_facts)
 537         elif system == 'Linux':
 538 
 539             distribution_files = DistributionFiles(module=self.module)
 540 
 541             # linux_distribution_facts = LinuxDistribution(module).get_distribution_facts()
 542             dist_file_facts = distribution_files.process_dist_files()
 543 
 544             distribution_facts.update(dist_file_facts)
 545 
 546         distro = distribution_facts['distribution']
 547 
 548         # look for a os family alias for the 'distribution', if there isnt one, use 'distribution'
 549         distribution_facts['os_family'] = self.OS_FAMILY.get(distro, None) or distro
 550 
 551         return distribution_facts

 621     def get_distribution_SunOS(self):
 622         sunos_facts = {}
 623 
 624         data = get_file_content('/etc/release').splitlines()[0]
 625 
 626         if 'Solaris' in data:
 627             # for solaris 10 uname_r will contain 5.10, for solaris 11 it will have 5.11
 628             uname_r = get_uname(self.module, flags=['-r'])
 629             ora_prefix = ''
 630             if 'Oracle Solaris' in data:
 631                 data = data.replace('Oracle ', '')
 632                 ora_prefix = 'Oracle '
 633             sunos_facts['distribution'] = data.split()[0]
 634             sunos_facts['distribution_version'] = data.split()[1]
 635             sunos_facts['distribution_release'] = ora_prefix + data
 636             sunos_facts['distribution_major_version'] = uname_r.split('.')[1].rstrip()
 637             return sunos_facts
 638 
 639         uname_v = get_uname(self.module, flags=['-v'])
 640         distribution_version = None
 641 
 642         if 'SmartOS' in data:
 643             sunos_facts['distribution'] = 'SmartOS'
 644             if _file_exists('/etc/product'):
 645                 product_data = dict([l.split(': ', 1) for l in get_file_content('/etc/product').splitlines() if ': ' in l])
 646                 if 'Image' in product_data:
 647                     distribution_version = product_data.get('Image').split()[-1]
 648         elif 'OpenIndiana' in data:
 649             sunos_facts['distribution'] = 'OpenIndiana'
 650         elif 'OmniOS' in data:
 651             sunos_facts['distribution'] = 'OmniOS'
 652             distribution_version = data.split()[-1]
 653         elif uname_v is not None and 'NexentaOS_' in uname_v:
 654             sunos_facts['distribution'] = 'Nexenta'
 655             distribution_version = data.split()[-1].lstrip('v')
 656 
 657         if sunos_facts.get('distribution', '') in ('SmartOS', 'OpenIndiana', 'OmniOS', 'Nexenta'):
 658             sunos_facts['distribution_release'] = data.strip()
 659             if distribution_version is not None:
 660                 sunos_facts['distribution_version'] = distribution_version
 661             elif uname_v is not None:
 662                 sunos_facts['distribution_version'] = uname_v.splitlines()[0].strip()
 663             return sunos_facts
 664 
 665         return sunos_facts
