  24 class SshPubKeyFactCollector(BaseFactCollector):
  25     name = 'ssh_pub_keys'
  26     _fact_ids = set(['ssh_host_pub_keys',
  27                      'ssh_host_key_dsa_public',
  28                      'ssh_host_key_rsa_public',
  29                      'ssh_host_key_ecdsa_public',
  30                      'ssh_host_key_ed25519_public'])
  31 
  32     def collect(self, module=None, collected_facts=None):
  33         ssh_pub_key_facts = {}
  34         algos = ('dsa', 'rsa', 'ecdsa', 'ed25519')
  35 
  36         # list of directories to check for ssh keys
  37         # used in the order listed here, the first one with keys is used
  38         keydirs = ['/etc/ssh', '/etc/openssh', '/etc']
  39 
  40         for keydir in keydirs:
  41             for algo in algos:
  42                 factname = 'ssh_host_key_%s_public' % algo
  43                 if factname in ssh_pub_key_facts:
  44                     # a previous keydir was already successful, stop looking
  45                     # for keys
  46                     return ssh_pub_key_facts
  47                 key_filename = '%s/ssh_host_%s_key.pub' % (keydir, algo)
  48                 keydata = get_file_content(key_filename)
  49                 if keydata is not None:
  50                     (keytype, key) = keydata.split()[0:2]
  51                     ssh_pub_key_facts[factname] = key
  52                     ssh_pub_key_facts[factname + '_keytype'] = keytype
  53 
  54         return ssh_pub_key_facts
