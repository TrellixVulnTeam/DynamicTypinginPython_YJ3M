  38 class ServiceMgrFactCollector(BaseFactCollector):
  39     name = 'service_mgr'
  40     _fact_ids = set()
  41     required_facts = set(['platform', 'distribution'])
  42 
  43     @staticmethod
  44     def is_systemd_managed(module):
  45         # tools must be installed
  46         if module.get_bin_path('systemctl'):
  47 
  48             # this should show if systemd is the boot init system, if checking init faild to mark as systemd
  49             # these mirror systemd's own sd_boot test http://www.freedesktop.org/software/systemd/man/sd_booted.html
  50             for canary in ["/run/systemd/system/", "/dev/.run/systemd/", "/dev/.systemd/"]:
  51                 if os.path.exists(canary):
  52                     return True
  53         return False
  54 
  55     @staticmethod
  56     def is_systemd_managed_offline(module):
  57         # tools must be installed
  58         if module.get_bin_path('systemctl'):
  59             # check if /sbin/init is a symlink to systemd
  60             # on SUSE, /sbin/init may be missing if systemd-sysvinit package is not installed.
  61             if os.path.islink('/sbin/init') and os.path.basename(os.readlink('/sbin/init')) == 'systemd':
  62                 return True
  63         return False
  64 
  65     def collect(self, module=None, collected_facts=None):
  66         facts_dict = {}
  67 
  68         if not module:
  69             return facts_dict
  70 
  71         collected_facts = collected_facts or {}
  72         service_mgr_name = None
  73 
  74         # TODO: detect more custom init setups like bootscripts, dmd, s6, Epoch, etc
  75         # also other OSs other than linux might need to check across several possible candidates
  76 
  77         # Mapping of proc_1 values to more useful names
  78         proc_1_map = {
  79             'procd': 'openwrt_init',
  80             'runit-init': 'runit',
  81             'svscan': 'svc',
  82             'openrc-init': 'openrc',
  83         }
  84 
  85         # try various forms of querying pid 1
  86         proc_1 = get_file_content('/proc/1/comm')
  87         if proc_1 is None:
  88             # FIXME: return code isnt checked
  89             # FIXME: if stdout is empty string, odd things
  90             # FIXME: other code seems to think we could get proc_1 == None past this point
  91             rc, proc_1, err = module.run_command("ps -p 1 -o comm|tail -n 1", use_unsafe_shell=True)
  92             # If the output of the command starts with what looks like a PID, then the 'ps' command
  93             # probably didn't work the way we wanted, probably because it's busybox
  94             if re.match(r' *[0-9]+ ', proc_1):
  95                 proc_1 = None
  96 
  97         # The ps command above may return "COMMAND" if the user cannot read /proc, e.g. with grsecurity
  98         if proc_1 == "COMMAND\n":
  99             proc_1 = None
 100 
 101         # FIXME: empty string proc_1 staus empty string
 102         if proc_1 is not None:
 103             proc_1 = os.path.basename(proc_1)
 104             proc_1 = to_native(proc_1)
 105             proc_1 = proc_1.strip()
 106 
 107         if proc_1 is not None and (proc_1 == 'init' or proc_1.endswith('sh')):
 108             # many systems return init, so this cannot be trusted, if it ends in 'sh' it probalby is a shell in a container
 109             proc_1 = None
 110 
 111         # if not init/None it should be an identifiable or custom init, so we are done!
 112         if proc_1 is not None:
 113             # Lookup proc_1 value in map and use proc_1 value itself if no match
 114             # FIXME: empty string still falls through
 115             service_mgr_name = proc_1_map.get(proc_1, proc_1)
 116 
 117         # FIXME: replace with a system->service_mgr_name map?
 118         # start with the easy ones
 119         elif collected_facts.get('ansible_distribution', None) == 'MacOSX':
 120             # FIXME: find way to query executable, version matching is not ideal
 121             if LooseVersion(platform.mac_ver()[0]) >= LooseVersion('10.4'):
 122                 service_mgr_name = 'launchd'
 123             else:
 124                 service_mgr_name = 'systemstarter'
 125         elif 'BSD' in collected_facts.get('ansible_system', '') or collected_facts.get('ansible_system') in ['Bitrig', 'DragonFly']:
 126             # FIXME: we might want to break out to individual BSDs or 'rc'
 127             service_mgr_name = 'bsdinit'
 128         elif collected_facts.get('ansible_system') == 'AIX':
 129             service_mgr_name = 'src'
 130         elif collected_facts.get('ansible_system') == 'SunOS':
 131             service_mgr_name = 'smf'
 132         elif collected_facts.get('ansible_distribution') == 'OpenWrt':
 133             service_mgr_name = 'openwrt_init'
 134         elif collected_facts.get('ansible_system') == 'Linux':
 135             # FIXME: mv is_systemd_managed
 136             if self.is_systemd_managed(module=module):
 137                 service_mgr_name = 'systemd'
 138             elif module.get_bin_path('initctl') and os.path.exists("/etc/init/"):
 139                 service_mgr_name = 'upstart'
 140             elif os.path.exists('/sbin/openrc'):
 141                 service_mgr_name = 'openrc'
 142             elif self.is_systemd_managed_offline(module=module):
 143                 service_mgr_name = 'systemd'
 144             elif os.path.exists('/etc/init.d/'):
 145                 service_mgr_name = 'sysvinit'
 146 
 147         if not service_mgr_name:
 148             # if we cannot detect, fallback to generic 'service'
 149             service_mgr_name = 'service'
 150 
 151         facts_dict['service_mgr'] = service_mgr_name
 152         return facts_dict
