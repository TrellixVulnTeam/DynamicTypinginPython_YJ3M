  19 import os

  21 from ansible.module_utils.six import iteritems
  22 
  23 from ansible.module_utils.facts.collector import BaseFactCollector
  24 
  25 
  26 class EnvFactCollector(BaseFactCollector):
  27     name = 'env'
  28     _fact_ids = set()
  29 
  30     def collect(self, module=None, collected_facts=None):
  31         env_facts = {}
  32         env_facts['env'] = {}
  33 
  34         for k, v in iteritems(os.environ):
  35             env_facts['env'][k] = v
  36 
  37         return env_facts
