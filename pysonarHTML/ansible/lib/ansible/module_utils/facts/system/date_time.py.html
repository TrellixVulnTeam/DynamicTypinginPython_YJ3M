  21 import datetime
  22 import time

  24 from ansible.module_utils.facts.collector import BaseFactCollector
  25 
  26 
  27 class DateTimeFactCollector(BaseFactCollector):
  28     name = 'date_time'
  29     _fact_ids = set()
  30 
  31     def collect(self, module=None, collected_facts=None):
  32         facts_dict = {}
  33         date_time_facts = {}
  34 
  35         now = datetime.datetime.now()
  36         date_time_facts['year'] = now.strftime('%Y')
  37         date_time_facts['month'] = now.strftime('%m')
  38         date_time_facts['weekday'] = now.strftime('%A')
  39         date_time_facts['weekday_number'] = now.strftime('%w')
  40         date_time_facts['weeknumber'] = now.strftime('%W')
  41         date_time_facts['day'] = now.strftime('%d')
  42         date_time_facts['hour'] = now.strftime('%H')
  43         date_time_facts['minute'] = now.strftime('%M')
  44         date_time_facts['second'] = now.strftime('%S')
  45         date_time_facts['epoch'] = now.strftime('%s')
  46         if date_time_facts['epoch'] == '' or date_time_facts['epoch'][0] == '%':
  47             # NOTE: in this case, the epoch wont match the rest of the date_time facts? ie, it's a few milliseconds later..? -akl
  48             date_time_facts['epoch'] = str(int(time.time()))
  49         date_time_facts['date'] = now.strftime('%Y-%m-%d')
  50         date_time_facts['time'] = now.strftime('%H:%M:%S')
  51         date_time_facts['iso8601_micro'] = now.utcnow().strftime("%Y-%m-%dT%H:%M:%S.%fZ")
  52         date_time_facts['iso8601'] = now.utcnow().strftime("%Y-%m-%dT%H:%M:%SZ")
  53         date_time_facts['iso8601_basic'] = now.strftime("%Y%m%dT%H%M%S%f")
  54         date_time_facts['iso8601_basic_short'] = now.strftime("%Y%m%dT%H%M%S")
  55         date_time_facts['tz'] = time.strftime("%Z")
  56         date_time_facts['tz_dst'] = time.tzname[1]
  57         date_time_facts['tz_offset'] = time.strftime("%z")
  58 
  59         facts_dict['date_time'] = date_time_facts
  60         return facts_dict
