  25 class DnsFactCollector(BaseFactCollector):
  26     name = 'dns'
  27     _fact_ids = set()
  28 
  29     def collect(self, module=None, collected_facts=None):
  30         dns_facts = {}
  31 
  32         # TODO: flatten
  33         dns_facts['dns'] = {}
  34 
  35         for line in get_file_content('/etc/resolv.conf', '').splitlines():
  36             if line.startswith('#') or line.startswith(';') or line.strip() == '':
  37                 continue
  38             tokens = line.split()
  39             if len(tokens) == 0:
  40                 continue
  41             if tokens[0] == 'nameserver':
  42                 if 'nameservers' not in dns_facts['dns']:
  43                     dns_facts['dns']['nameservers'] = []
  44                 for nameserver in tokens[1:]:
  45                     dns_facts['dns']['nameservers'].append(nameserver)
  46             elif tokens[0] == 'domain':
  47                 if len(tokens) > 1:
  48                     dns_facts['dns']['domain'] = tokens[1]
  49             elif tokens[0] == 'search':
  50                 dns_facts['dns']['search'] = []
  51                 for suffix in tokens[1:]:
  52                     dns_facts['dns']['search'].append(suffix)
  53             elif tokens[0] == 'sortlist':
  54                 dns_facts['dns']['sortlist'] = []
  55                 for address in tokens[1:]:
  56                     dns_facts['dns']['sortlist'].append(address)
  57             elif tokens[0] == 'options':
  58                 dns_facts['dns']['options'] = {}
  59                 if len(tokens) > 1:
  60                     for option in tokens[1:]:
  61                         option_tokens = option.split(':', 1)
  62                         if len(option_tokens) == 0:
  63                             continue
  64                         val = len(option_tokens) == 2 and option_tokens[1] or True
  65                         dns_facts['dns']['options'][option_tokens[0]] = val
  66 
  67         return dns_facts
