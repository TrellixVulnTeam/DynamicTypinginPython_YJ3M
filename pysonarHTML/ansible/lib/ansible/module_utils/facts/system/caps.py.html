  25 class SystemCapabilitiesFactCollector(BaseFactCollector):
  26     name = 'caps'
  27     _fact_ids = set(['system_capabilities',
  28                      'system_capabilities_enforced'])
  29 
  30     def collect(self, module=None, collected_facts=None):
  31         facts_dict = {}
  32         if not module:
  33             return facts_dict
  34 
  35         capsh_path = module.get_bin_path('capsh')
  36         # NOTE: early exit 'if not crash_path' and unindent rest of method -akl
  37         if capsh_path:
  38             # NOTE: -> get_caps_data()/parse_caps_data() for easier mocking -akl
  39             rc, out, err = module.run_command([capsh_path, "--print"], errors='surrogate_then_replace')
  40             enforced_caps = []
  41             enforced = 'NA'
  42             for line in out.splitlines():
  43                 if len(line) < 1:
  44                     continue
  45                 if line.startswith('Current:'):
  46                     if line.split(':')[1].strip() == '=ep':
  47                         enforced = 'False'
  48                     else:
  49                         enforced = 'True'
  50                         enforced_caps = [i.strip() for i in line.split('=')[1].split(',')]
  51 
  52             facts_dict['system_capabilities_enforced'] = enforced
  53             facts_dict['system_capabilities'] = enforced_caps
  54 
  55         return facts_dict
