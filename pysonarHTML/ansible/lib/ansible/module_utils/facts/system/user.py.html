  19 import getpass
  20 import os
  21 import pwd
  22 
  23 from ansible.module_utils.facts.collector import BaseFactCollector
  24 
  25 
  26 class UserFactCollector(BaseFactCollector):
  27     name = 'user'
  28     _fact_ids = set(['user_id', 'user_uid', 'user_gid',
  29                      'user_gecos', 'user_dir', 'user_shell',
  30                      'real_user_id', 'effective_user_id',
  31                      'effective_group_ids'])
  32 
  33     def collect(self, module=None, collected_facts=None):
  34         user_facts = {}
  35 
  36         user_facts['user_id'] = getpass.getuser()
  37 
  38         pwent = pwd.getpwnam(getpass.getuser())
  39 
  40         user_facts['user_uid'] = pwent.pw_uid
  41         user_facts['user_gid'] = pwent.pw_gid
  42         user_facts['user_gecos'] = pwent.pw_gecos
  43         user_facts['user_dir'] = pwent.pw_dir
  44         user_facts['user_shell'] = pwent.pw_shell
  45         user_facts['real_user_id'] = os.getuid()
  46         user_facts['effective_user_id'] = os.geteuid()
  47         user_facts['real_group_id'] = os.getgid()
  48         user_facts['effective_group_id'] = os.getgid()
  49 
  50         return user_facts
