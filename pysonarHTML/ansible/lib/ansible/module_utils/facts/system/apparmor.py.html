  26 class ApparmorFactCollector(BaseFactCollector):
  27     name = 'apparmor'
  28     _fact_ids = set()
  29 
  30     def collect(self, module=None, collected_facts=None):
  31         facts_dict = {}
  32         apparmor_facts = {}
  33         if os.path.exists('/sys/kernel/security/apparmor'):
  34             apparmor_facts['status'] = 'enabled'
  35         else:
  36             apparmor_facts['status'] = 'disabled'
  37 
  38         facts_dict['apparmor'] = apparmor_facts
  39         return facts_dict
