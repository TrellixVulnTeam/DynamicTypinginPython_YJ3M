  19 class PkgMgr(with_metaclass(ABCMeta, object)):
  20 
  21     @abstractmethod
  22     def is_available(self):
  23         # This method is supposed to return True/False if the package manager is currently installed/usable
  24         # It can also 'prep' the required systems in the process of detecting availability
  25         pass
  26 
  27     @abstractmethod
  28     def list_installed(self):
  29         # This method should return a list of installed packages, each list item will be passed to get_package_details
  30         pass
  31 
  32     @abstractmethod
  33     def get_package_details(self, package):
  34         # This takes a 'package' item and returns a dictionary with the package information, name and version are minimal requirements
  35         pass
  36 
  37     def get_packages(self):
  38         # Take all of the above and return a dictionary of lists of dictionaries (package = list of installed versions)
  39 
  40         installed_packages = {}
  41         for package in self.list_installed():
  42             package_details = self.get_package_details(package)
  43             if 'source' not in package_details:
  44                 package_details['source'] = self.__class__.__name__.lower()
  45             name = package_details['name']
  46             if name not in installed_packages:
  47                 installed_packages[name] = [package_details]
  48             else:
  49                 installed_packages[name].append(package_details)
  50         return installed_packages

  53 class LibMgr(PkgMgr):
  54 
  55     LIB = None
  56 
  57     def __init__(self):
  58 
  59         self._lib = None
  60         super(LibMgr, self).__init__()
  61 
  62     def is_available(self):
  63         found = False
  64         try:
  65             self._lib = __import__(self.LIB)
  66             found = True
  67         except ImportError:
  68             pass
  69         return found
