  25 class HPUXHardware(Hardware):
  26     """
  27     HP-UX-specific subclass of Hardware. Defines memory and CPU facts:
  28     - memfree_mb
  29     - memtotal_mb
  30     - swapfree_mb
  31     - swaptotal_mb
  32     - processor
  33     - processor_cores
  34     - processor_count
  35     - model
  36     - firmware
  37     """
  38 
  39     platform = 'HP-UX'
  40 
  41     def populate(self, collected_facts=None):
  42         hardware_facts = {}
  43 
  44         cpu_facts = self.get_cpu_facts(collected_facts=collected_facts)
  45         memory_facts = self.get_memory_facts()
  46         hw_facts = self.get_hw_facts()
  47 
  48         hardware_facts.update(cpu_facts)
  49         hardware_facts.update(memory_facts)
  50         hardware_facts.update(hw_facts)
  51 
  52         return hardware_facts
  53 
  54     def get_cpu_facts(self, collected_facts=None):
  55         cpu_facts = {}
  56         collected_facts = collected_facts or {}
  57 
  58         if collected_facts.get('ansible_architecture') in ['9000/800', '9000/785']:
  59             rc, out, err = self.module.run_command("ioscan -FkCprocessor | wc -l", use_unsafe_shell=True)
  60             cpu_facts['processor_count'] = int(out.strip())
  61         # Working with machinfo mess
  62         elif collected_facts.get('ansible_architecture') == 'ia64':
  63             if collected_facts.get('ansible_distribution_version') == "B.11.23":
  64                 rc, out, err = self.module.run_command("/usr/contrib/bin/machinfo | grep 'Number of CPUs'", use_unsafe_shell=True)
  65                 if out:
  66                     cpu_facts['processor_count'] = int(out.strip().split('=')[1])
  67                 rc, out, err = self.module.run_command("/usr/contrib/bin/machinfo | grep 'processor family'", use_unsafe_shell=True)
  68                 if out:
  69                     cpu_facts['processor'] = re.search('.*(Intel.*)', out).groups()[0].strip()
  70                 rc, out, err = self.module.run_command("ioscan -FkCprocessor | wc -l", use_unsafe_shell=True)
  71                 cpu_facts['processor_cores'] = int(out.strip())
  72             if collected_facts.get('ansible_distribution_version') == "B.11.31":
  73                 # if machinfo return cores strings release B.11.31 > 1204
  74                 rc, out, err = self.module.run_command("/usr/contrib/bin/machinfo | grep core | wc -l", use_unsafe_shell=True)
  75                 if out.strip() == '0':
  76                     rc, out, err = self.module.run_command("/usr/contrib/bin/machinfo | grep Intel", use_unsafe_shell=True)
  77                     cpu_facts['processor_count'] = int(out.strip().split(" ")[0])
  78                     # If hyperthreading is active divide cores by 2
  79                     rc, out, err = self.module.run_command("/usr/sbin/psrset | grep LCPU", use_unsafe_shell=True)
  80                     data = re.sub(' +', ' ', out).strip().split(' ')
  81                     if len(data) == 1:
  82                         hyperthreading = 'OFF'
  83                     else:
  84                         hyperthreading = data[1]
  85                     rc, out, err = self.module.run_command("/usr/contrib/bin/machinfo | grep logical", use_unsafe_shell=True)
  86                     data = out.strip().split(" ")
  87                     if hyperthreading == 'ON':
  88                         cpu_facts['processor_cores'] = int(data[0]) / 2
  89                     else:
  90                         if len(data) == 1:
  91                             cpu_facts['processor_cores'] = cpu_facts['processor_count']
  92                         else:
  93                             cpu_facts['processor_cores'] = int(data[0])
  94                     rc, out, err = self.module.run_command("/usr/contrib/bin/machinfo | grep Intel |cut -d' ' -f4-", use_unsafe_shell=True)
  95                     cpu_facts['processor'] = out.strip()
  96                 else:
  97                     rc, out, err = self.module.run_command("/usr/contrib/bin/machinfo | egrep 'socket[s]?$' | tail -1", use_unsafe_shell=True)
  98                     cpu_facts['processor_count'] = int(out.strip().split(" ")[0])
  99                     rc, out, err = self.module.run_command("/usr/contrib/bin/machinfo | grep -e '[0-9] core' | tail -1", use_unsafe_shell=True)
 100                     cpu_facts['processor_cores'] = int(out.strip().split(" ")[0])
 101                     rc, out, err = self.module.run_command("/usr/contrib/bin/machinfo | grep Intel", use_unsafe_shell=True)
 102                     cpu_facts['processor'] = out.strip()
 103 
 104         return cpu_facts

 106     def get_memory_facts(self, collected_facts=None):
 107         memory_facts = {}
 108         collected_facts = collected_facts or {}
 109 
 110         pagesize = 4096
 111         rc, out, err = self.module.run_command("/usr/bin/vmstat | tail -1", use_unsafe_shell=True)
 112         data = int(re.sub(' +', ' ', out).split(' ')[5].strip())
 113         memory_facts['memfree_mb'] = pagesize * data // 1024 // 1024
 114         if collected_facts.get('ansible_architecture') in ['9000/800', '9000/785']:
 115             try:
 116                 rc, out, err = self.module.run_command("grep Physical /var/adm/syslog/syslog.log")
 117                 data = re.search('.*Physical: ([0-9]*) Kbytes.*', out).groups()[0].strip()
 118                 memory_facts['memtotal_mb'] = int(data) // 1024
 119             except AttributeError:
 120                 # For systems where memory details aren't sent to syslog or the log has rotated, use parsed
 121                 # adb output. Unfortunately /dev/kmem doesn't have world-read, so this only works as root.
 122                 if os.access("/dev/kmem", os.R_OK):
 123                     rc, out, err = self.module.run_command("echo 'phys_mem_pages/D' | adb -k /stand/vmunix /dev/kmem | tail -1 | awk '{print $2}'",
 124                                                            use_unsafe_shell=True)
 125                     if not err:
 126                         data = out
 127                         memory_facts['memtotal_mb'] = int(data) / 256
 128         else:
 129             rc, out, err = self.module.run_command("/usr/contrib/bin/machinfo | grep Memory", use_unsafe_shell=True)
 130             data = re.search(r'Memory[\ :=]*([0-9]*).*MB.*', out).groups()[0].strip()
 131             memory_facts['memtotal_mb'] = int(data)
 132         rc, out, err = self.module.run_command("/usr/sbin/swapinfo -m -d -f -q")
 133         memory_facts['swaptotal_mb'] = int(out.strip())
 134         rc, out, err = self.module.run_command("/usr/sbin/swapinfo -m -d -f | egrep '^dev|^fs'", use_unsafe_shell=True)
 135         swap = 0
 136         for line in out.strip().splitlines():
 137             swap += int(re.sub(' +', ' ', line).split(' ')[3].strip())
 138         memory_facts['swapfree_mb'] = swap
 139 
 140         return memory_facts
 141 
 142     def get_hw_facts(self, collected_facts=None):
 143         hw_facts = {}
 144         collected_facts = collected_facts or {}
 145 
 146         rc, out, err = self.module.run_command("model")
 147         hw_facts['model'] = out.strip()
 148         if collected_facts.get('ansible_architecture') == 'ia64':
 149             separator = ':'
 150             if collected_facts.get('ansible_distribution_version') == "B.11.23":
 151                 separator = '='
 152             rc, out, err = self.module.run_command("/usr/contrib/bin/machinfo |grep -i 'Firmware revision' | grep -v BMC", use_unsafe_shell=True)
 153             hw_facts['firmware_version'] = out.split(separator)[1].strip()
 154             rc, out, err = self.module.run_command("/usr/contrib/bin/machinfo |grep -i 'Machine serial number' ", use_unsafe_shell=True)
 155             if rc == 0 and out:
 156                 hw_facts['product_serial'] = out.split(separator)[1].strip()
 157 
 158         return hw_facts
