  30 class OpenBSDHardware(Hardware):
  31     """
  32     OpenBSD-specific subclass of Hardware. Defines memory, CPU and device facts:
  33     - memfree_mb
  34     - memtotal_mb
  35     - swapfree_mb
  36     - swaptotal_mb
  37     - processor (a list)
  38     - processor_cores
  39     - processor_count
  40     - processor_speed
  41 
  42     In addition, it also defines number of DMI facts and device facts.
  43     """
  44     platform = 'OpenBSD'
  45 
  46     def populate(self, collected_facts=None):
  47         hardware_facts = {}
  48         self.sysctl = get_sysctl(self.module, ['hw'])
  49 
  50         # TODO: change name
  51         cpu_facts = self.get_processor_facts()
  52         memory_facts = self.get_memory_facts()
  53         device_facts = self.get_device_facts()
  54         dmi_facts = self.get_dmi_facts()
  55 
  56         mount_facts = {}
  57         try:
  58             mount_facts = self.get_mount_facts()
  59         except timeout.TimeoutError:
  60             pass
  61 
  62         hardware_facts.update(cpu_facts)
  63         hardware_facts.update(memory_facts)
  64         hardware_facts.update(dmi_facts)
  65         hardware_facts.update(device_facts)
  66         hardware_facts.update(mount_facts)
  67 
  68         return hardware_facts
  69 
  70     @timeout.timeout()
  71     def get_mount_facts(self):
  72         mount_facts = {}
  73 
  74         mount_facts['mounts'] = []
  75         fstab = get_file_content('/etc/fstab')
  76         if fstab:
  77             for line in fstab.splitlines():
  78                 if line.startswith('#') or line.strip() == '':
  79                     continue
  80                 fields = re.sub(r'\s+', ' ', line).split()
  81                 if fields[1] == 'none' or fields[3] == 'xx':
  82                     continue
  83                 mount_statvfs_info = get_mount_size(fields[1])
  84                 mount_info = {'mount': fields[1],
  85                               'device': fields[0],
  86                               'fstype': fields[2],
  87                               'options': fields[3]}
  88                 mount_info.update(mount_statvfs_info)
  89                 mount_facts['mounts'].append(mount_info)
  90         return mount_facts
  91 
  92     def get_memory_facts(self):
  93         memory_facts = {}
  94         # Get free memory. vmstat output looks like:
  95         #  procs    memory       page                    disks    traps          cpu
  96         #  r b w    avm     fre  flt  re  pi  po  fr  sr wd0 fd0  int   sys   cs us sy id
  97         #  0 0 0  47512   28160   51   0   0   0   0   0   1   0  116    89   17  0  1 99
  98         rc, out, err = self.module.run_command("/usr/bin/vmstat")
  99         if rc == 0:
 100             memory_facts['memfree_mb'] = int(out.splitlines()[-1].split()[4]) // 1024
 101             memory_facts['memtotal_mb'] = int(self.sysctl['hw.usermem']) // 1024 // 1024
 102 
 103         # Get swapctl info. swapctl output looks like:
 104         # total: 69268 1K-blocks allocated, 0 used, 69268 available
 105         # And for older OpenBSD:
 106         # total: 69268k bytes allocated = 0k used, 69268k available
 107         rc, out, err = self.module.run_command("/sbin/swapctl -sk")
 108         if rc == 0:
 109             swaptrans = {ord(u'k'): None,
 110                          ord(u'm'): None,
 111                          ord(u'g'): None}
 112             data = to_text(out, errors='surrogate_or_strict').split()
 113             memory_facts['swapfree_mb'] = int(data[-2].translate(swaptrans)) // 1024
 114             memory_facts['swaptotal_mb'] = int(data[1].translate(swaptrans)) // 1024
 115 
 116         return memory_facts
 117 
 118     def get_processor_facts(self):
 119         cpu_facts = {}
 120         processor = []
 121         for i in range(int(self.sysctl['hw.ncpu'])):
 122             processor.append(self.sysctl['hw.model'])
 123 
 124         cpu_facts['processor'] = processor
 125         # The following is partly a lie because there is no reliable way to
 126         # determine the number of physical CPUs in the system. We can only
 127         # query the number of logical CPUs, which hides the number of cores.
 128         # On amd64/i386 we could try to inspect the smt/core/package lines in
 129         # dmesg, however even those have proven to be unreliable.
 130         # So take a shortcut and report the logical number of processors in
 131         # 'processor_count' and 'processor_cores' and leave it at that.
 132         cpu_facts['processor_count'] = self.sysctl['hw.ncpu']
 133         cpu_facts['processor_cores'] = self.sysctl['hw.ncpu']
 134 
 135         return cpu_facts
 136 
 137     def get_device_facts(self):
 138         device_facts = {}
 139         devices = []
 140         devices.extend(self.sysctl['hw.disknames'].split(','))
 141         device_facts['devices'] = devices
 142 
 143         return device_facts
 144 
 145     def get_dmi_facts(self):
 146         dmi_facts = {}
 147         # We don't use dmidecode(8) here because:
 148         # - it would add dependency on an external package
 149         # - dmidecode(8) can only be ran as root
 150         # So instead we rely on sysctl(8) to provide us the information on a
 151         # best-effort basis. As a bonus we also get facts on non-amd64/i386
 152         # platforms this way.
 153         sysctl_to_dmi = {
 154             'hw.product': 'product_name',
 155             'hw.version': 'product_version',
 156             'hw.uuid': 'product_uuid',
 157             'hw.serialno': 'product_serial',
 158             'hw.vendor': 'system_vendor',
 159         }
 160 
 161         for mib in sysctl_to_dmi:
 162             if mib in self.sysctl:
 163                 dmi_facts[sysctl_to_dmi[mib]] = self.sysctl[mib]
 164 
 165         return dmi_facts
