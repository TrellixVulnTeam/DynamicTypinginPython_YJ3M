  29 class FreeBSDHardware(Hardware):
  30     """
  31     FreeBSD-specific subclass of Hardware.  Defines memory and CPU facts:
  32     - memfree_mb
  33     - memtotal_mb
  34     - swapfree_mb
  35     - swaptotal_mb
  36     - processor (a list)
  37     - processor_cores
  38     - processor_count
  39     - devices
  40     """
  41     platform = 'FreeBSD'
  42     DMESG_BOOT = '/var/run/dmesg.boot'
  43 
  44     def populate(self, collected_facts=None):
  45         hardware_facts = {}
  46 
  47         cpu_facts = self.get_cpu_facts()
  48         memory_facts = self.get_memory_facts()
  49         dmi_facts = self.get_dmi_facts()
  50         device_facts = self.get_device_facts()
  51 
  52         mount_facts = {}
  53         try:
  54             mount_facts = self.get_mount_facts()
  55         except TimeoutError:
  56             pass
  57 
  58         hardware_facts.update(cpu_facts)
  59         hardware_facts.update(memory_facts)
  60         hardware_facts.update(dmi_facts)
  61         hardware_facts.update(device_facts)
  62         hardware_facts.update(mount_facts)
  63 
  64         return hardware_facts
  65 
  66     def get_cpu_facts(self):
  67         cpu_facts = {}
  68         cpu_facts['processor'] = []
  69         sysctl = self.module.get_bin_path('sysctl')
  70         if sysctl:
  71             rc, out, err = self.module.run_command("%s -n hw.ncpu" % sysctl, check_rc=False)
  72             cpu_facts['processor_count'] = out.strip()
  73 
  74         dmesg_boot = get_file_content(FreeBSDHardware.DMESG_BOOT)
  75         if not dmesg_boot:
  76             try:
  77                 rc, dmesg_boot, err = self.module.run_command(self.module.get_bin_path("dmesg"), check_rc=False)
  78             except Exception:
  79                 dmesg_boot = ''
  80 
  81         for line in dmesg_boot.splitlines():
  82             if 'CPU:' in line:
  83                 cpu = re.sub(r'CPU:\s+', r"", line)
  84                 cpu_facts['processor'].append(cpu.strip())
  85             if 'Logical CPUs per core' in line:
  86                 cpu_facts['processor_cores'] = line.split()[4]
  87 
  88         return cpu_facts
  89 
  90     def get_memory_facts(self):
  91         memory_facts = {}
  92 
  93         sysctl = self.module.get_bin_path('sysctl')
  94         if sysctl:
  95             rc, out, err = self.module.run_command("%s vm.stats" % sysctl, check_rc=False)
  96             for line in out.splitlines():
  97                 data = line.split()
  98                 if 'vm.stats.vm.v_page_size' in line:
  99                     pagesize = int(data[1])
 100                 if 'vm.stats.vm.v_page_count' in line:
 101                     pagecount = int(data[1])
 102                 if 'vm.stats.vm.v_free_count' in line:
 103                     freecount = int(data[1])
 104             memory_facts['memtotal_mb'] = pagesize * pagecount // 1024 // 1024
 105             memory_facts['memfree_mb'] = pagesize * freecount // 1024 // 1024
 106 
 107         swapinfo = self.module.get_bin_path('swapinfo')
 108         if swapinfo:
 109             # Get swapinfo.  swapinfo output looks like:
 110             # Device          1M-blocks     Used    Avail Capacity
 111             # /dev/ada0p3        314368        0   314368     0%
 112             #
 113             rc, out, err = self.module.run_command("%s -k" % swapinfo)
 114             lines = out.splitlines()
 115             if len(lines[-1]) == 0:
 116                 lines.pop()
 117             data = lines[-1].split()
 118             if data[0] != 'Device':
 119                 memory_facts['swaptotal_mb'] = int(data[1]) // 1024
 120                 memory_facts['swapfree_mb'] = int(data[3]) // 1024
 121 
 122         return memory_facts

 124     @timeout()
 125     def get_mount_facts(self):
 126         mount_facts = {}
 127 
 128         mount_facts['mounts'] = []
 129         fstab = get_file_content('/etc/fstab')
 130         if fstab:
 131             for line in fstab.splitlines():
 132                 if line.startswith('#') or line.strip() == '':
 133                     continue
 134                 fields = re.sub(r'\s+', ' ', line).split()
 135                 mount_statvfs_info = get_mount_size(fields[1])
 136                 mount_info = {'mount': fields[1],
 137                               'device': fields[0],
 138                               'fstype': fields[2],
 139                               'options': fields[3]}
 140                 mount_info.update(mount_statvfs_info)
 141                 mount_facts['mounts'].append(mount_info)
 142 
 143         return mount_facts
 144 
 145     def get_device_facts(self):
 146         device_facts = {}
 147 
 148         sysdir = '/dev'
 149         device_facts['devices'] = {}
 150         drives = re.compile(r'(ada?\d+|da\d+|a?cd\d+)')  # TODO: rc, disks, err = self.module.run_command("/sbin/sysctl kern.disks")
 151         slices = re.compile(r'(ada?\d+s\d+\w*|da\d+s\d+\w*)')
 152         if os.path.isdir(sysdir):
 153             dirlist = sorted(os.listdir(sysdir))
 154             for device in dirlist:
 155                 d = drives.match(device)
 156                 if d:
 157                     device_facts['devices'][d.group(1)] = []
 158                 s = slices.match(device)
 159                 if s:
 160                     device_facts['devices'][d.group(1)].append(s.group(1))
 161 
 162         return device_facts

 164     def get_dmi_facts(self):
 165         ''' learn dmi facts from system
 166 
 167         Use dmidecode executable if available'''
 168 
 169         dmi_facts = {}
 170 
 171         # Fall back to using dmidecode, if available
 172         dmi_bin = self.module.get_bin_path('dmidecode')
 173         DMI_DICT = {
 174             'bios_date': 'bios-release-date',
 175             'bios_vendor': 'bios-vendor',
 176             'bios_version': 'bios-version',
 177             'board_asset_tag': 'baseboard-asset-tag',
 178             'board_name': 'baseboard-product-name',
 179             'board_serial': 'baseboard-serial-number',
 180             'board_vendor': 'baseboard-manufacturer',
 181             'board_version': 'baseboard-version',
 182             'chassis_asset_tag': 'chassis-asset-tag',
 183             'chassis_serial': 'chassis-serial-number',
 184             'chassis_vendor': 'chassis-manufacturer',
 185             'chassis_version': 'chassis-version',
 186             'form_factor': 'chassis-type',
 187             'product_name': 'system-product-name',
 188             'product_serial': 'system-serial-number',
 189             'product_uuid': 'system-uuid',
 190             'product_version': 'system-version',
 191             'system_vendor': 'system-manufacturer',
 192         }
 193         for (k, v) in DMI_DICT.items():
 194             if dmi_bin is not None:
 195                 (rc, out, err) = self.module.run_command('%s -s %s' % (dmi_bin, v))
 196                 if rc == 0:
 197                     # Strip out commented lines (specific dmidecode output)
 198                     # FIXME: why add the fact and then test if it is json?
 199                     dmi_facts[k] = ''.join([line for line in out.splitlines() if not line.startswith('#')])
 200                     try:
 201                         json.dumps(dmi_facts[k])
 202                     except UnicodeDecodeError:
 203                         dmi_facts[k] = 'NA'
 204                 else:
 205                     dmi_facts[k] = 'NA'
 206             else:
 207                 dmi_facts[k] = 'NA'
 208 
 209         return dmi_facts
