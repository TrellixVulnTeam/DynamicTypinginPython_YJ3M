  25 class DarwinHardware(Hardware):
  26     """
  27     Darwin-specific subclass of Hardware.  Defines memory and CPU facts:
  28     - processor
  29     - processor_cores
  30     - memtotal_mb
  31     - memfree_mb
  32     - model
  33     - osversion
  34     - osrevision
  35     """
  36     platform = 'Darwin'
  37 
  38     def populate(self, collected_facts=None):
  39         hardware_facts = {}
  40 
  41         self.sysctl = get_sysctl(self.module, ['hw', 'machdep', 'kern'])
  42         mac_facts = self.get_mac_facts()
  43         cpu_facts = self.get_cpu_facts()
  44         memory_facts = self.get_memory_facts()
  45 
  46         hardware_facts.update(mac_facts)
  47         hardware_facts.update(cpu_facts)
  48         hardware_facts.update(memory_facts)
  49 
  50         return hardware_facts
  51 
  52     def get_system_profile(self):
  53         rc, out, err = self.module.run_command(["/usr/sbin/system_profiler", "SPHardwareDataType"])
  54         if rc != 0:
  55             return dict()
  56         system_profile = dict()
  57         for line in out.splitlines():
  58             if ': ' in line:
  59                 (key, value) = line.split(': ', 1)
  60                 system_profile[key.strip()] = ' '.join(value.strip().split())
  61         return system_profile
  62 
  63     def get_mac_facts(self):
  64         mac_facts = {}
  65         rc, out, err = self.module.run_command("sysctl hw.model")
  66         if rc == 0:
  67             mac_facts['model'] = mac_facts['product_name'] = out.splitlines()[-1].split()[1]
  68         mac_facts['osversion'] = self.sysctl['kern.osversion']
  69         mac_facts['osrevision'] = self.sysctl['kern.osrevision']
  70 
  71         return mac_facts
  72 
  73     def get_cpu_facts(self):
  74         cpu_facts = {}
  75         if 'machdep.cpu.brand_string' in self.sysctl:  # Intel
  76             cpu_facts['processor'] = self.sysctl['machdep.cpu.brand_string']
  77             cpu_facts['processor_cores'] = self.sysctl['machdep.cpu.core_count']
  78         else:  # PowerPC
  79             system_profile = self.get_system_profile()
  80             cpu_facts['processor'] = '%s @ %s' % (system_profile['Processor Name'], system_profile['Processor Speed'])
  81             cpu_facts['processor_cores'] = self.sysctl['hw.physicalcpu']
  82         cpu_facts['processor_vcpus'] = self.sysctl.get('hw.logicalcpu') or self.sysctl.get('hw.ncpu') or ''
  83 
  84         return cpu_facts
  85 
  86     def get_memory_facts(self):
  87         memory_facts = {
  88             'memtotal_mb': int(self.sysctl['hw.memsize']) // 1024 // 1024,
  89             'memfree_mb': 0,
  90         }
  91 
  92         total_used = 0
  93         page_size = 4096
  94         try:
  95             vm_stat_command = get_bin_path('vm_stat')
  96         except ValueError:
  97             return memory_facts
  98 
  99         rc, out, err = self.module.run_command(vm_stat_command)
 100         if rc == 0:
 101             # Free = Total - (Wired + active + inactive)
 102             # Get a generator of tuples from the command output so we can later
 103             # turn it into a dictionary
 104             memory_stats = (line.rstrip('.').split(':', 1) for line in out.splitlines())
 105 
 106             # Strip extra left spaces from the value
 107             memory_stats = dict((k, v.lstrip()) for k, v in memory_stats)
 108 
 109             for k, v in memory_stats.items():
 110                 try:
 111                     memory_stats[k] = int(v)
 112                 except ValueError:
 113                     # Most values convert cleanly to integer values but if the field does
 114                     # not convert to an integer, just leave it alone.
 115                     pass
 116 
 117             if memory_stats.get('Pages wired down'):
 118                 total_used += memory_stats['Pages wired down'] * page_size
 119             if memory_stats.get('Pages active'):
 120                 total_used += memory_stats['Pages active'] * page_size
 121             if memory_stats.get('Pages inactive'):
 122                 total_used += memory_stats['Pages inactive'] * page_size
 123 
 124             memory_facts['memfree_mb'] = memory_facts['memtotal_mb'] - (total_used // 1024 // 1024)
 125 
 126         return memory_facts
