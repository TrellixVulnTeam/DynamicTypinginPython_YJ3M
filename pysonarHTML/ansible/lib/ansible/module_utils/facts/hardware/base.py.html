  46 class HardwareCollector(BaseFactCollector):
  47     name = 'hardware'
  48     _fact_ids = set(['processor',
  49                      'processor_cores',
  50                      'processor_count',
  51                      # TODO: mounts isnt exactly hardware
  52                      'mounts',
  53                      'devices'])
  54     _fact_class = Hardware
  55 
  56     def collect(self, module=None, collected_facts=None):
  57         collected_facts = collected_facts or {}
  58         if not module:
  59             return {}
  60 
  61         # Network munges cached_facts by side effect, so give it a copy
  62         facts_obj = self._fact_class(module)
  63 
  64         facts_dict = facts_obj.populate(collected_facts=collected_facts)
  65 
  66         return facts_dict
