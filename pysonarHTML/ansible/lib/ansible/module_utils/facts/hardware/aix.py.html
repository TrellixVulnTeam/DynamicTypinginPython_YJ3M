  25 class AIXHardware(Hardware):
  26     """
  27     AIX-specific subclass of Hardware.  Defines memory and CPU facts:
  28     - memfree_mb
  29     - memtotal_mb
  30     - swapfree_mb
  31     - swaptotal_mb
  32     - processor (a list)
  33     - processor_cores
  34     - processor_count
  35     """
  36     platform = 'AIX'
  37 
  38     def populate(self, collected_facts=None):
  39         hardware_facts = {}
  40 
  41         cpu_facts = self.get_cpu_facts()
  42         memory_facts = self.get_memory_facts()
  43         dmi_facts = self.get_dmi_facts()
  44         vgs_facts = self.get_vgs_facts()
  45         mount_facts = self.get_mount_facts()
  46         devices_facts = self.get_device_facts()
  47 
  48         hardware_facts.update(cpu_facts)
  49         hardware_facts.update(memory_facts)
  50         hardware_facts.update(dmi_facts)
  51         hardware_facts.update(vgs_facts)
  52         hardware_facts.update(mount_facts)
  53         hardware_facts.update(devices_facts)
  54 
  55         return hardware_facts
  56 
  57     def get_cpu_facts(self):
  58         cpu_facts = {}
  59         cpu_facts['processor'] = []
  60 
  61         rc, out, err = self.module.run_command("/usr/sbin/lsdev -Cc processor")
  62         if out:
  63             i = 0
  64             for line in out.splitlines():
  65 
  66                 if 'Available' in line:
  67                     if i == 0:
  68                         data = line.split(' ')
  69                         cpudev = data[0]
  70 
  71                     i += 1
  72             cpu_facts['processor_count'] = int(i)
  73 
  74             rc, out, err = self.module.run_command("/usr/sbin/lsattr -El " + cpudev + " -a type")
  75 
  76             data = out.split(' ')
  77             cpu_facts['processor'] = data[1]
  78 
  79             rc, out, err = self.module.run_command("/usr/sbin/lsattr -El " + cpudev + " -a smt_threads")
  80             if out:
  81                 data = out.split(' ')
  82                 cpu_facts['processor_cores'] = int(data[1])
  83 
  84         return cpu_facts
  85 
  86     def get_memory_facts(self):
  87         memory_facts = {}
  88         pagesize = 4096
  89         rc, out, err = self.module.run_command("/usr/bin/vmstat -v")
  90         for line in out.splitlines():
  91             data = line.split()
  92             if 'memory pages' in line:
  93                 pagecount = int(data[0])
  94             if 'free pages' in line:
  95                 freecount = int(data[0])
  96         memory_facts['memtotal_mb'] = pagesize * pagecount // 1024 // 1024
  97         memory_facts['memfree_mb'] = pagesize * freecount // 1024 // 1024
  98         # Get swapinfo.  swapinfo output looks like:
  99         # Device          1M-blocks     Used    Avail Capacity
 100         # /dev/ada0p3        314368        0   314368     0%
 101         #
 102         rc, out, err = self.module.run_command("/usr/sbin/lsps -s")
 103         if out:
 104             lines = out.splitlines()
 105             data = lines[1].split()
 106             swaptotal_mb = int(data[0].rstrip('MB'))
 107             percused = int(data[1].rstrip('%'))
 108             memory_facts['swaptotal_mb'] = swaptotal_mb
 109             memory_facts['swapfree_mb'] = int(swaptotal_mb * (100 - percused) / 100)
 110 
 111         return memory_facts

 113     def get_dmi_facts(self):
 114         dmi_facts = {}
 115 
 116         rc, out, err = self.module.run_command("/usr/sbin/lsattr -El sys0 -a fwversion")
 117         data = out.split()
 118         dmi_facts['firmware_version'] = data[1].strip('IBM,')
 119         lsconf_path = self.module.get_bin_path("lsconf")
 120         if lsconf_path:
 121             rc, out, err = self.module.run_command(lsconf_path)
 122             if rc == 0 and out:
 123                 for line in out.splitlines():
 124                     data = line.split(':')
 125                     if 'Machine Serial Number' in line:
 126                         dmi_facts['product_serial'] = data[1].strip()
 127                     if 'LPAR Info' in line:
 128                         dmi_facts['lpar_info'] = data[1].strip()
 129                     if 'System Model' in line:
 130                         dmi_facts['product_name'] = data[1].strip()
 131         return dmi_facts

 133     def get_vgs_facts(self):
 134         """
 135         Get vg and pv Facts
 136         rootvg:
 137         PV_NAME           PV STATE          TOTAL PPs   FREE PPs    FREE DISTRIBUTION
 138         hdisk0            active            546         0           00..00..00..00..00
 139         hdisk1            active            546         113         00..00..00..21..92
 140         realsyncvg:
 141         PV_NAME           PV STATE          TOTAL PPs   FREE PPs    FREE DISTRIBUTION
 142         hdisk74           active            1999        6           00..00..00..00..06
 143         testvg:
 144         PV_NAME           PV STATE          TOTAL PPs   FREE PPs    FREE DISTRIBUTION
 145         hdisk105          active            999         838         200..39..199..200..200
 146         hdisk106          active            999         599         200..00..00..199..200
 147         """
 148 
 149         vgs_facts = {}
 150         lsvg_path = self.module.get_bin_path("lsvg")
 151         xargs_path = self.module.get_bin_path("xargs")
 152         cmd = "%s -o | %s %s -p" % (lsvg_path, xargs_path, lsvg_path)
 153         if lsvg_path and xargs_path:
 154             rc, out, err = self.module.run_command(cmd, use_unsafe_shell=True)
 155             if rc == 0 and out:
 156                 vgs_facts['vgs'] = {}
 157                 for m in re.finditer(r'(\S+):\n.*FREE DISTRIBUTION(\n(\S+)\s+(\w+)\s+(\d+)\s+(\d+).*)+', out):
 158                     vgs_facts['vgs'][m.group(1)] = []
 159                     pp_size = 0
 160                     cmd = "%s %s" % (lsvg_path, m.group(1))
 161                     rc, out, err = self.module.run_command(cmd)
 162                     if rc == 0 and out:
 163                         pp_size = re.search(r'PP SIZE:\s+(\d+\s+\S+)', out).group(1)
 164                         for n in re.finditer(r'(\S+)\s+(\w+)\s+(\d+)\s+(\d+).*', m.group(0)):
 165                             pv_info = {'pv_name': n.group(1),
 166                                        'pv_state': n.group(2),
 167                                        'total_pps': n.group(3),
 168                                        'free_pps': n.group(4),
 169                                        'pp_size': pp_size
 170                                        }
 171                             vgs_facts['vgs'][m.group(1)].append(pv_info)
 172 
 173         return vgs_facts

 175     def get_mount_facts(self):
 176         mount_facts = {}
 177 
 178         mount_facts['mounts'] = []
 179 
 180         mounts = []
 181 
 182         # AIX does not have mtab but mount command is only source of info (or to use
 183         # api calls to get same info)
 184         mount_path = self.module.get_bin_path('mount')
 185         rc, mount_out, err = self.module.run_command(mount_path)
 186         if mount_out:
 187             for line in mount_out.split('\n'):
 188                 fields = line.split()
 189                 if len(fields) != 0 and fields[0] != 'node' and fields[0][0] != '-' and re.match('^/.*|^[a-zA-Z].*|^[0-9].*', fields[0]):
 190                     if re.match('^/', fields[0]):
 191                         # normal mount
 192                         mount = fields[1]
 193                         mount_info = {'mount': mount,
 194                                       'device': fields[0],
 195                                       'fstype': fields[2],
 196                                       'options': fields[6],
 197                                       'time': '%s %s %s' % (fields[3], fields[4], fields[5])}
 198                         mount_info.update(get_mount_size(mount))
 199                     else:
 200                         # nfs or cifs based mount
 201                         # in case of nfs if no mount options are provided on command line
 202                         # add into fields empty string...
 203                         if len(fields) < 8:
 204                             fields.append("")
 205 
 206                         mount_info = {'mount': fields[2],
 207                                       'device': '%s:%s' % (fields[0], fields[1]),
 208                                       'fstype': fields[3],
 209                                       'options': fields[7],
 210                                       'time': '%s %s %s' % (fields[4], fields[5], fields[6])}
 211 
 212                     mounts.append(mount_info)
 213 
 214         mount_facts['mounts'] = mounts
 215 
 216         return mount_facts

 218     def get_device_facts(self):
 219         device_facts = {}
 220         device_facts['devices'] = {}
 221 
 222         lsdev_cmd = self.module.get_bin_path('lsdev', True)
 223         lsattr_cmd = self.module.get_bin_path('lsattr', True)
 224         rc, out_lsdev, err = self.module.run_command(lsdev_cmd)
 225 
 226         for line in out_lsdev.splitlines():
 227             field = line.split()
 228 
 229             device_attrs = {}
 230             device_name = field[0]
 231             device_state = field[1]
 232             device_type = field[2:]
 233             lsattr_cmd_args = [lsattr_cmd, '-E', '-l', device_name]
 234             rc, out_lsattr, err = self.module.run_command(lsattr_cmd_args)
 235             for attr in out_lsattr.splitlines():
 236                 attr_fields = attr.split()
 237                 attr_name = attr_fields[0]
 238                 attr_parameter = attr_fields[1]
 239                 device_attrs[attr_name] = attr_parameter
 240 
 241             device_facts['devices'][device_name] = {
 242                 'state': device_state,
 243                 'type': ' '.join(device_type),
 244                 'attributes': device_attrs
 245             }
 246 
 247         return device_facts
