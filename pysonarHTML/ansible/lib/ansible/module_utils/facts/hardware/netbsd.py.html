  31 class NetBSDHardware(Hardware):
  32     """
  33     NetBSD-specific subclass of Hardware.  Defines memory and CPU facts:
  34     - memfree_mb
  35     - memtotal_mb
  36     - swapfree_mb
  37     - swaptotal_mb
  38     - processor (a list)
  39     - processor_cores
  40     - processor_count
  41     - devices
  42     """
  43     platform = 'NetBSD'
  44     MEMORY_FACTS = ['MemTotal', 'SwapTotal', 'MemFree', 'SwapFree']
  45 
  46     def populate(self, collected_facts=None):
  47         hardware_facts = {}
  48         self.sysctl = get_sysctl(self.module, ['machdep'])
  49         cpu_facts = self.get_cpu_facts()
  50         memory_facts = self.get_memory_facts()
  51 
  52         mount_facts = {}
  53         try:
  54             mount_facts = self.get_mount_facts()
  55         except TimeoutError:
  56             pass
  57 
  58         dmi_facts = self.get_dmi_facts()
  59 
  60         hardware_facts.update(cpu_facts)
  61         hardware_facts.update(memory_facts)
  62         hardware_facts.update(mount_facts)
  63         hardware_facts.update(dmi_facts)
  64 
  65         return hardware_facts
  66 
  67     def get_cpu_facts(self):
  68         cpu_facts = {}
  69 
  70         i = 0
  71         physid = 0
  72         sockets = {}
  73         if not os.access("/proc/cpuinfo", os.R_OK):
  74             return cpu_facts
  75         cpu_facts['processor'] = []
  76         for line in get_file_lines("/proc/cpuinfo"):
  77             data = line.split(":", 1)
  78             key = data[0].strip()
  79             # model name is for Intel arch, Processor (mind the uppercase P)
  80             # works for some ARM devices, like the Sheevaplug.
  81             if key == 'model name' or key == 'Processor':
  82                 if 'processor' not in cpu_facts:
  83                     cpu_facts['processor'] = []
  84                 cpu_facts['processor'].append(data[1].strip())
  85                 i += 1
  86             elif key == 'physical id':
  87                 physid = data[1].strip()
  88                 if physid not in sockets:
  89                     sockets[physid] = 1
  90             elif key == 'cpu cores':
  91                 sockets[physid] = int(data[1].strip())
  92         if len(sockets) > 0:
  93             cpu_facts['processor_count'] = len(sockets)
  94             cpu_facts['processor_cores'] = reduce(lambda x, y: x + y, sockets.values())
  95         else:
  96             cpu_facts['processor_count'] = i
  97             cpu_facts['processor_cores'] = 'NA'
  98 
  99         return cpu_facts
 100 
 101     def get_memory_facts(self):
 102         memory_facts = {}
 103         if not os.access("/proc/meminfo", os.R_OK):
 104             return memory_facts
 105         for line in get_file_lines("/proc/meminfo"):
 106             data = line.split(":", 1)
 107             key = data[0]
 108             if key in NetBSDHardware.MEMORY_FACTS:
 109                 val = data[1].strip().split(' ')[0]
 110                 memory_facts["%s_mb" % key.lower()] = int(val) // 1024
 111 
 112         return memory_facts
 113 
 114     @timeout()
 115     def get_mount_facts(self):
 116         mount_facts = {}
 117 
 118         mount_facts['mounts'] = []
 119         fstab = get_file_content('/etc/fstab')
 120 
 121         if not fstab:
 122             return mount_facts
 123 
 124         for line in fstab.splitlines():
 125             if line.startswith('#') or line.strip() == '':
 126                 continue
 127             fields = re.sub(r'\s+', ' ', line).split()
 128             mount_statvfs_info = get_mount_size(fields[1])
 129             mount_info = {'mount': fields[1],
 130                           'device': fields[0],
 131                           'fstype': fields[2],
 132                           'options': fields[3]}
 133             mount_info.update(mount_statvfs_info)
 134             mount_facts['mounts'].append(mount_info)
 135         return mount_facts
 136 
 137     def get_dmi_facts(self):
 138         dmi_facts = {}
 139         # We don't use dmidecode(8) here because:
 140         # - it would add dependency on an external package
 141         # - dmidecode(8) can only be ran as root
 142         # So instead we rely on sysctl(8) to provide us the information on a
 143         # best-effort basis. As a bonus we also get facts on non-amd64/i386
 144         # platforms this way.
 145         sysctl_to_dmi = {
 146             'machdep.dmi.system-product': 'product_name',
 147             'machdep.dmi.system-version': 'product_version',
 148             'machdep.dmi.system-uuid': 'product_uuid',
 149             'machdep.dmi.system-serial': 'product_serial',
 150             'machdep.dmi.system-vendor': 'system_vendor',
 151         }
 152 
 153         for mib in sysctl_to_dmi:
 154             if mib in self.sysctl:
 155                 dmi_facts[sysctl_to_dmi[mib]] = self.sysctl[mib]
 156 
 157         return dmi_facts

 162     _platform = 'NetBSD'
