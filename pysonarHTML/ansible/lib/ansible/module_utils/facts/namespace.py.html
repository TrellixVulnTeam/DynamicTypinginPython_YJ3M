  32 class FactNamespace:
  33     def __init__(self, namespace_name):
  34         self.namespace_name = namespace_name
  35 
  36     def transform(self, name):
  37         '''Take a text name, and transforms it as needed (add a namespace prefix, etc)'''
  38         return name
  39 
  40     def _underscore(self, name):
  41         return name.replace('-', '_')
  42 
  43 
  44 class PrefixFactNamespace(FactNamespace):
  45     def __init__(self, namespace_name, prefix=None):
  46         super(PrefixFactNamespace, self).__init__(namespace_name)
  47         self.prefix = prefix
  48 
  49     def transform(self, name):
  50         new_name = self._underscore(name)
  51         return '%s%s' % (self.prefix, new_name)
