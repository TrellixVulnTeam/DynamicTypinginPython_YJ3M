  26 class OhaiFactCollector(BaseFactCollector):
  27     '''This is a subclass of Facts for including information gathered from Ohai.'''
  28     name = 'ohai'
  29     _fact_ids = set()
  30 
  31     def __init__(self, collectors=None, namespace=None):
  32         namespace = PrefixFactNamespace(namespace_name='ohai',
  33                                         prefix='ohai_')
  34         super(OhaiFactCollector, self).__init__(collectors=collectors,
  35                                                 namespace=namespace)
  36 
  37     def find_ohai(self, module):
  38         ohai_path = module.get_bin_path('ohai')
  39         return ohai_path
  40 
  41     def run_ohai(self, module, ohai_path,):
  42         rc, out, err = module.run_command(ohai_path)
  43         return rc, out, err
  44 
  45     def get_ohai_output(self, module):
  46         ohai_path = self.find_ohai(module)
  47         if not ohai_path:
  48             return None
  49 
  50         rc, out, err = self.run_ohai(module, ohai_path)
  51         if rc != 0:
  52             return None
  53 
  54         return out
  55 
  56     def collect(self, module=None, collected_facts=None):
  57         ohai_facts = {}
  58         if not module:
  59             return ohai_facts
  60 
  61         ohai_output = self.get_ohai_output(module)
  62 
  63         if ohai_output is None:
  64             return ohai_facts
  65 
  66         try:
  67             ohai_facts = json.loads(ohai_output)
  68         except Exception:
  69             # FIXME: useful error, logging, something...
  70             pass
  71 
  72         return ohai_facts
