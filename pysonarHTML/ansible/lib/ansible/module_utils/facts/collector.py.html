  39 class CycleFoundInFactDeps(Exception):
  40     '''Indicates there is a cycle in fact collector deps
  41 
  42     If collector-B requires collector-A, and collector-A requires
  43     collector-B, that is a cycle. In that case, there is no ordering
  44     that will satisfy B before A and A and before B. That will cause this
  45     error to be raised.
  46     '''
  47     pass

  58 class BaseFactCollector:
  59     _fact_ids = set()
  60 
  61     _platform = 'Generic'
  62     name = None
  63     required_facts = set()
  64 
  65     def __init__(self, collectors=None, namespace=None):
  66         '''Base class for things that collect facts.
  67 
  68         'collectors' is an optional list of other FactCollectors for composing.'''
  69         self.collectors = collectors or []
  70 
  71         # self.namespace is a object with a 'transform' method that transforms
  72         # the name to indicate the namespace (ie, adds a prefix or suffix).
  73         self.namespace = namespace
  74 
  75         self.fact_ids = set([self.name])
  76         self.fact_ids.update(self._fact_ids)
  77 
  78     @classmethod
  79     def platform_match(cls, platform_info):
  80         if platform_info.get('system', None) == cls._platform:
  81             return cls
  82         return None
  83 
  84     def _transform_name(self, key_name):
  85         if self.namespace:
  86             return self.namespace.transform(key_name)
  87         return key_name
  88 
  89     def _transform_dict_keys(self, fact_dict):
  90         '''update a dicts keys to use new names as transformed by self._transform_name'''
  91 
  92         for old_key in list(fact_dict.keys()):
  93             new_key = self._transform_name(old_key)
  94             # pop the item by old_key and replace it using new_key
  95             fact_dict[new_key] = fact_dict.pop(old_key)
  96         return fact_dict
  97 
  98     # TODO/MAYBE: rename to 'collect' and add 'collect_without_namespace'
  99     def collect_with_namespace(self, module=None, collected_facts=None):
 100         # collect, then transform the key names if needed
 101         facts_dict = self.collect(module=module, collected_facts=collected_facts)
 102         if self.namespace:
 103             facts_dict = self._transform_dict_keys(facts_dict)
 104         return facts_dict
 105 
 106     def collect(self, module=None, collected_facts=None):
 107         '''do the fact collection
 108 
 109         'collected_facts' is a object (a dict, likely) that holds all previously
 110           facts. This is intended to be used if a FactCollector needs to reference
 111           another fact (for ex, the system arch) and should not be modified (usually).
 112 
 113           Returns a dict of facts.
 114 
 115           '''
 116         facts_dict = {}
 117         return facts_dict

 120 def get_collector_names(valid_subsets=None,
 121                         minimal_gather_subset=None,
 122                         gather_subset=None,
 123                         aliases_map=None,
 124                         platform_info=None):
 125     '''return a set of FactCollector names based on gather_subset spec.
 126 
 127     gather_subset is a spec describing which facts to gather.
 128     valid_subsets is a frozenset of potential matches for gather_subset ('all', 'network') etc
 129     minimal_gather_subsets is a frozenset of matches to always use, even for gather_subset='!all'
 130     '''
 131 
 132     # Retrieve module parameters
 133     gather_subset = gather_subset or ['all']
 134 
 135     # the list of everything that 'all' expands to
 136     valid_subsets = valid_subsets or frozenset()
 137 
 138     # if provided, minimal_gather_subset is always added, even after all negations
 139     minimal_gather_subset = minimal_gather_subset or frozenset()
 140 
 141     aliases_map = aliases_map or defaultdict(set)
 142 
 143     # Retrieve all facts elements
 144     additional_subsets = set()
 145     exclude_subsets = set()
 146 
 147     # total always starts with the min set, then
 148     # adds of the additions in gather_subset, then
 149     # excludes all of the excludes, then add any explicitly
 150     # requested subsets.
 151     gather_subset_with_min = ['min']
 152     gather_subset_with_min.extend(gather_subset)
 153 
 154     # subsets we mention in gather_subset explicitly, except for 'all'/'min'
 155     explicitly_added = set()
 156 
 157     for subset in gather_subset_with_min:
 158         subset_id = subset
 159         if subset_id == 'min':
 160             additional_subsets.update(minimal_gather_subset)
 161             continue
 162         if subset_id == 'all':
 163             additional_subsets.update(valid_subsets)
 164             continue
 165         if subset_id.startswith('!'):
 166             subset = subset[1:]
 167             if subset == 'min':
 168                 exclude_subsets.update(minimal_gather_subset)
 169                 continue
 170             if subset == 'all':
 171                 exclude_subsets.update(valid_subsets - minimal_gather_subset)
 172                 continue
 173             exclude = True
 174         else:
 175             exclude = False
 176 
 177         if exclude:
 178             # include 'devices', 'dmi' etc for '!hardware'
 179             exclude_subsets.update(aliases_map.get(subset, set()))
 180             exclude_subsets.add(subset)
 181         else:
 182             # NOTE: this only considers adding an unknown gather subsetup an error. Asking to
 183             #       exclude an unknown gather subset is ignored.
 184             if subset_id not in valid_subsets:
 185                 raise TypeError("Bad subset '%s' given to Ansible. gather_subset options allowed: all, %s" %
 186                                 (subset, ", ".join(sorted(valid_subsets))))
 187 
 188             explicitly_added.add(subset)
 189             additional_subsets.add(subset)
 190 
 191     if not additional_subsets:
 192         additional_subsets.update(valid_subsets)
 193 
 194     additional_subsets.difference_update(exclude_subsets - explicitly_added)
 195 
 196     return additional_subsets

 199 def find_collectors_for_platform(all_collector_classes, compat_platforms):
 200     found_collectors = set()
 201     found_collectors_names = set()
 202 
 203     # start from specific platform, then try generic
 204     for compat_platform in compat_platforms:
 205         platform_match = None
 206         for all_collector_class in all_collector_classes:
 207 
 208             # ask the class if it is compatible with the platform info
 209             platform_match = all_collector_class.platform_match(compat_platform)
 210 
 211             if not platform_match:
 212                 continue
 213 
 214             primary_name = all_collector_class.name
 215 
 216             if primary_name not in found_collectors_names:
 217                 found_collectors.add(all_collector_class)
 218                 found_collectors_names.add(all_collector_class.name)
 219 
 220     return found_collectors

 223 def build_fact_id_to_collector_map(collectors_for_platform):
 224     fact_id_to_collector_map = defaultdict(list)
 225     aliases_map = defaultdict(set)
 226 
 227     for collector_class in collectors_for_platform:
 228         primary_name = collector_class.name
 229 
 230         fact_id_to_collector_map[primary_name].append(collector_class)
 231 
 232         for fact_id in collector_class._fact_ids:
 233             fact_id_to_collector_map[fact_id].append(collector_class)
 234             aliases_map[primary_name].add(fact_id)
 235 
 236     return fact_id_to_collector_map, aliases_map

 239 def select_collector_classes(collector_names, all_fact_subsets):
 240     seen_collector_classes = set()
 241 
 242     selected_collector_classes = []
 243 
 244     for collector_name in collector_names:
 245         collector_classes = all_fact_subsets.get(collector_name, [])
 246         for collector_class in collector_classes:
 247             if collector_class not in seen_collector_classes:
 248                 selected_collector_classes.append(collector_class)
 249                 seen_collector_classes.add(collector_class)
 250 
 251     return selected_collector_classes

 254 def _get_requires_by_collector_name(collector_name, all_fact_subsets):
 255     required_facts = set()
 256 
 257     try:
 258         collector_classes = all_fact_subsets[collector_name]
 259     except KeyError:
 260         raise CollectorNotFoundError('Fact collector "%s" not found' % collector_name)
 261     for collector_class in collector_classes:
 262         required_facts.update(collector_class.required_facts)
 263     return required_facts

 266 def find_unresolved_requires(collector_names, all_fact_subsets):
 267     '''Find any collector names that have unresolved requires
 268 
 269     Returns a list of collector names that correspond to collector
 270     classes whose .requires_facts() are not in collector_names.
 271     '''
 272     unresolved = set()
 273 
 274     for collector_name in collector_names:
 275         required_facts = _get_requires_by_collector_name(collector_name, all_fact_subsets)
 276         for required_fact in required_facts:
 277             if required_fact not in collector_names:
 278                 unresolved.add(required_fact)
 279 
 280     return unresolved

 283 def resolve_requires(unresolved_requires, all_fact_subsets):
 284     new_names = set()
 285     failed = []
 286     for unresolved in unresolved_requires:
 287         if unresolved in all_fact_subsets:
 288             new_names.add(unresolved)
 289         else:
 290             failed.append(unresolved)
 291 
 292     if failed:
 293         raise UnresolvedFactDep('unresolved fact dep %s' % ','.join(failed))
 294     return new_names

 297 def build_dep_data(collector_names, all_fact_subsets):
 298     dep_map = defaultdict(set)
 299     for collector_name in collector_names:
 300         collector_deps = set()
 301         for collector in all_fact_subsets[collector_name]:
 302             for dep in collector.required_facts:
 303                 collector_deps.add(dep)
 304         dep_map[collector_name] = collector_deps
 305     return dep_map

 308 def tsort(dep_map):
 309     sorted_list = []
 310 
 311     unsorted_map = dep_map.copy()
 312 
 313     while unsorted_map:
 314         acyclic = False
 315         for node, edges in list(unsorted_map.items()):
 316             for edge in edges:
 317                 if edge in unsorted_map:
 318                     break
 319             else:
 320                 acyclic = True
 321                 del unsorted_map[node]
 322                 sorted_list.append((node, edges))
 323 
 324         if not acyclic:
 325             raise CycleFoundInFactDeps('Unable to tsort deps, there was a cycle in the graph. sorted=%s' % sorted_list)
 326 
 327     return sorted_list

 330 def _solve_deps(collector_names, all_fact_subsets):
 331     unresolved = collector_names.copy()
 332     solutions = collector_names.copy()
 333 
 334     while True:
 335         unresolved = find_unresolved_requires(solutions, all_fact_subsets)
 336         if unresolved == set():
 337             break
 338 
 339         new_names = resolve_requires(unresolved, all_fact_subsets)
 340         solutions.update(new_names)
 341 
 342     return solutions

 345 def collector_classes_from_gather_subset(all_collector_classes=None,
 346                                          valid_subsets=None,
 347                                          minimal_gather_subset=None,
 348                                          gather_subset=None,
 349                                          gather_timeout=None,
 350                                          platform_info=None):
 351     '''return a list of collector classes that match the args'''
 352 
 353     # use gather_name etc to get the list of collectors
 354 
 355     all_collector_classes = all_collector_classes or []
 356 
 357     minimal_gather_subset = minimal_gather_subset or frozenset()
 358 
 359     platform_info = platform_info or {'system': platform.system()}
 360 
 361     gather_timeout = gather_timeout or timeout.DEFAULT_GATHER_TIMEOUT
 362 
 363     # tweak the modules GATHER_TIMEOUT
 364     timeout.GATHER_TIMEOUT = gather_timeout
 365 
 366     valid_subsets = valid_subsets or frozenset()
 367 
 368     # maps alias names like 'hardware' to the list of names that are part of hardware
 369     # like 'devices' and 'dmi'
 370     aliases_map = defaultdict(set)
 371 
 372     compat_platforms = [platform_info, {'system': 'Generic'}]
 373 
 374     collectors_for_platform = find_collectors_for_platform(all_collector_classes, compat_platforms)
 375 
 376     # all_facts_subsets maps the subset name ('hardware') to the class that provides it.
 377 
 378     # TODO: name collisions here? are there facts with the same name as a gather_subset (all, network, hardware, virtual, ohai, facter)
 379     all_fact_subsets, aliases_map = build_fact_id_to_collector_map(collectors_for_platform)
 380 
 381     all_valid_subsets = frozenset(all_fact_subsets.keys())
 382 
 383     # expand any fact_id/collectorname/gather_subset term ('all', 'env', etc) to the list of names that represents
 384     collector_names = get_collector_names(valid_subsets=all_valid_subsets,
 385                                           minimal_gather_subset=minimal_gather_subset,
 386                                           gather_subset=gather_subset,
 387                                           aliases_map=aliases_map,
 388                                           platform_info=platform_info)
 389 
 390     complete_collector_names = _solve_deps(collector_names, all_fact_subsets)
 391 
 392     dep_map = build_dep_data(complete_collector_names, all_fact_subsets)
 393 
 394     ordered_deps = tsort(dep_map)
 395     ordered_collector_names = [x[0] for x in ordered_deps]
 396 
 397     selected_collector_classes = select_collector_classes(ordered_collector_names,
 398                                                           all_fact_subsets)
 399 
 400     return selected_collector_classes
