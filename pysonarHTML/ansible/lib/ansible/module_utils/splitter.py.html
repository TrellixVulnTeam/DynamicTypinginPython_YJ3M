  33 def _get_quote_state(token, quote_char):
  34     '''
  35     the goal of this block is to determine if the quoted string
  36     is unterminated in which case it needs to be put back together
  37     '''
  38     # the char before the current one, used to see if
  39     # the current character is escaped
  40     prev_char = None
  41     for idx, cur_char in enumerate(token):
  42         if idx > 0:
  43             prev_char = token[idx - 1]
  44         if cur_char in '"\'' and prev_char != '\\':
  45             if quote_char:
  46                 if cur_char == quote_char:
  47                     quote_char = None
  48             else:
  49                 quote_char = cur_char
  50     return quote_char

  53 def _count_jinja2_blocks(token, cur_depth, open_token, close_token):
  54     '''
  55     this function counts the number of opening/closing blocks for a
  56     given opening/closing type and adjusts the current depth for that
  57     block based on the difference
  58     '''
  59     num_open = token.count(open_token)
  60     num_close = token.count(close_token)
  61     if num_open != num_close:
  62         cur_depth += (num_open - num_close)
  63         if cur_depth < 0:
  64             cur_depth = 0
  65     return cur_depth

  68 def split_args(args):
  69     '''
  70     Splits args on whitespace, but intelligently reassembles
  71     those that may have been split over a jinja2 block or quotes.
  72 
  73     When used in a remote module, we won't ever have to be concerned about
  74     jinja2 blocks, however this function is/will be used in the
  75     core portions as well before the args are templated.
  76 
  77     example input: a=b c="foo bar"
  78     example output: ['a=b', 'c="foo bar"']
  79 
  80     Basically this is a variation shlex that has some more intelligence for
  81     how Ansible needs to use it.
  82     '''
  83 
  84     # the list of params parsed out of the arg string
  85     # this is going to be the result value when we are donei
  86     params = []
  87 
  88     # here we encode the args, so we have a uniform charset to
  89     # work with, and split on white space
  90     args = args.strip()
  91     try:
  92         args = args.encode('utf-8')
  93         do_decode = True
  94     except UnicodeDecodeError:
  95         do_decode = False
  96     items = args.split('\n')
  97 
  98     # iterate over the tokens, and reassemble any that may have been
  99     # split on a space inside a jinja2 block.
 100     # ex if tokens are "{{", "foo", "}}" these go together
 101 
 102     # These variables are used
 103     # to keep track of the state of the parsing, since blocks and quotes
 104     # may be nested within each other.
 105 
 106     quote_char = None
 107     inside_quotes = False
 108     print_depth = 0  # used to count nested jinja2 {{ }} blocks
 109     block_depth = 0  # used to count nested jinja2 {% %} blocks
 110     comment_depth = 0  # used to count nested jinja2 {# #} blocks
 111 
 112     # now we loop over each split chunk, coalescing tokens if the white space
 113     # split occurred within quotes or a jinja2 block of some kind
 114     for itemidx, item in enumerate(items):
 115 
 116         # we split on spaces and newlines separately, so that we
 117         # can tell which character we split on for reassembly
 118         # inside quotation characters
 119         tokens = item.strip().split(' ')
 120 
 121         line_continuation = False
 122         for idx, token in enumerate(tokens):
 123 
 124             # if we hit a line continuation character, but
 125             # we're not inside quotes, ignore it and continue
 126             # on to the next token while setting a flag
 127             if token == '\\' and not inside_quotes:
 128                 line_continuation = True
 129                 continue
 130 
 131             # store the previous quoting state for checking later
 132             was_inside_quotes = inside_quotes
 133             quote_char = _get_quote_state(token, quote_char)
 134             inside_quotes = quote_char is not None
 135 
 136             # multiple conditions may append a token to the list of params,
 137             # so we keep track with this flag to make sure it only happens once
 138             # append means add to the end of the list, don't append means concatenate
 139             # it to the end of the last token
 140             appended = False
 141 
 142             # if we're inside quotes now, but weren't before, append the token
 143             # to the end of the list, since we'll tack on more to it later
 144             # otherwise, if we're inside any jinja2 block, inside quotes, or we were
 145             # inside quotes (but aren't now) concat this token to the last param
 146             if inside_quotes and not was_inside_quotes:
 147                 params.append(token)
 148                 appended = True
 149             elif print_depth or block_depth or comment_depth or inside_quotes or was_inside_quotes:
 150                 if idx == 0 and not inside_quotes and was_inside_quotes:
 151                     params[-1] = "%s%s" % (params[-1], token)
 152                 elif len(tokens) > 1:
 153                     spacer = ''
 154                     if idx > 0:
 155                         spacer = ' '
 156                     params[-1] = "%s%s%s" % (params[-1], spacer, token)
 157                 else:
 158                     spacer = ''
 159                     if not params[-1].endswith('\n') and idx == 0:
 160                         spacer = '\n'
 161                     params[-1] = "%s%s%s" % (params[-1], spacer, token)
 162                 appended = True
 163 
 164             # if the number of paired block tags is not the same, the depth has changed, so we calculate that here
 165             # and may append the current token to the params (if we haven't previously done so)
 166             prev_print_depth = print_depth
 167             print_depth = _count_jinja2_blocks(token, print_depth, "{{", "}}")
 168             if print_depth != prev_print_depth and not appended:
 169                 params.append(token)
 170                 appended = True
 171 
 172             prev_block_depth = block_depth
 173             block_depth = _count_jinja2_blocks(token, block_depth, "{%", "%}")
 174             if block_depth != prev_block_depth and not appended:
 175                 params.append(token)
 176                 appended = True
 177 
 178             prev_comment_depth = comment_depth
 179             comment_depth = _count_jinja2_blocks(token, comment_depth, "{#", "#}")
 180             if comment_depth != prev_comment_depth and not appended:
 181                 params.append(token)
 182                 appended = True
 183 
 184             # finally, if we're at zero depth for all blocks and not inside quotes, and have not
 185             # yet appended anything to the list of params, we do so now
 186             if not (print_depth or block_depth or comment_depth) and not inside_quotes and not appended and token != '':
 187                 params.append(token)
 188 
 189         # if this was the last token in the list, and we have more than
 190         # one item (meaning we split on newlines), add a newline back here
 191         # to preserve the original structure
 192         if len(items) > 1 and itemidx != len(items) - 1 and not line_continuation:
 193             if not params[-1].endswith('\n') or item == '':
 194                 params[-1] += '\n'
 195 
 196         # always clear the line continuation flag
 197         line_continuation = False
 198 
 199     # If we're done and things are not at zero depth or we're still inside quotes,
 200     # raise an error to indicate that the args were unbalanced
 201     if print_depth or block_depth or comment_depth or inside_quotes:
 202         raise Exception("error while splitting arguments, either an unbalanced jinja2 block or quotes")
 203 
 204     # finally, we decode each param back to the unicode it was in the arg string
 205     if do_decode:
 206         params = [x.decode('utf-8') for x in params]
 207 
 208     return params

 211 def is_quoted(data):
 212     return len(data) > 0 and (data[0] == '"' and data[-1] == '"' or data[0] == "'" and data[-1] == "'")

 215 def unquote(data):
 216     ''' removes first and last quotes from a string, if the string starts and ends with the same quotes '''
 217     if is_quoted(data):
 218         return data[1:-1]
 219     return data
