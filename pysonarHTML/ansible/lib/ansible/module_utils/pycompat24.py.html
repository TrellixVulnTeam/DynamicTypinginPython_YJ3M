  35 def get_exception():
  36     """Get the current exception.
  37 
  38     This code needs to work on Python 2.4 through 3.x, so we cannot use
  39     "except Exception, e:" (SyntaxError on Python 3.x) nor
  40     "except Exception as e:" (SyntaxError on Python 2.4-2.5).
  41     Instead we must use ::
  42 
  43         except Exception:
  44             e = get_exception()
  45 
  46     """
  47     return sys.exc_info()[1]

  50 try:
  51     # Python 2.6+
  52     from ast import literal_eval
  53 except ImportError:
  54     # a replacement for literal_eval that works with python 2.4. from:
  55     # https://mail.python.org/pipermail/python-list/2009-September/551880.html
  56     # which is essentially a cut/paste from an earlier (2.6) version of python's
  57     # ast.py
  58     from compiler import ast, parse
  59     from ansible.module_utils.six import binary_type, integer_types, string_types, text_type
  60 
  61     def literal_eval(node_or_string):
  62         """
  63         Safely evaluate an expression node or a string containing a Python
  64         expression.  The string or node provided may only consist of the  following
  65         Python literal structures: strings, numbers, tuples, lists, dicts,  booleans,
  66         and None.
  67         """
  68         _safe_names = {'None': None, 'True': True, 'False': False}
  69         if isinstance(node_or_string, string_types):
  70             node_or_string = parse(node_or_string, mode='eval')
  71         if isinstance(node_or_string, ast.Expression):
  72             node_or_string = node_or_string.node
  73 
  74         def _convert(node):
  75             if isinstance(node, ast.Const) and isinstance(node.value, (text_type, binary_type, float, complex) + integer_types):
  76                 return node.value
  77             elif isinstance(node, ast.Tuple):
  78                 return tuple(map(_convert, node.nodes))
  79             elif isinstance(node, ast.List):
  80                 return list(map(_convert, node.nodes))
  81             elif isinstance(node, ast.Dict):
  82                 return dict((_convert(k), _convert(v)) for k, v in node.items())
  83             elif isinstance(node, ast.Name):
  84                 if node.name in _safe_names:
  85                     return _safe_names[node.name]
  86             elif isinstance(node, ast.UnarySub):
  87                 return -_convert(node.expr)  # pylint: disable=invalid-unary-operand-type
  88             raise ValueError('malformed string')
  89         return _convert(node_or_string)
