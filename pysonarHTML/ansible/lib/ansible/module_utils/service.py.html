  45 def sysv_is_enabled(name, runlevel=None):
  46     '''
  47     This function will check if the service name supplied
  48     is enabled in any of the sysv runlevels
  49 
  50     :arg name: name of the service to test for
  51     :kw runlevel: runlevel to check (default: None)
  52     '''
  53     if runlevel:
  54         if not os.path.isdir('/etc/rc0.d/'):
  55             return bool(glob.glob('/etc/init.d/rc%s.d/S??%s' % (runlevel, name)))
  56         return bool(glob.glob('/etc/rc%s.d/S??%s' % (runlevel, name)))
  57     else:
  58         if not os.path.isdir('/etc/rc0.d/'):
  59             return bool(glob.glob('/etc/init.d/rc?.d/S??%s' % name))
  60         return bool(glob.glob('/etc/rc?.d/S??%s' % name))

  63 def get_sysv_script(name):
  64     '''
  65     This function will return the expected path for an init script
  66     corresponding to the service name supplied.
  67 
  68     :arg name: name or path of the service to test for
  69     '''
  70     if name.startswith('/'):
  71         result = name
  72     else:
  73         result = '/etc/init.d/%s' % name
  74 
  75     return result

  88 def get_ps(module, pattern):
  89     '''
  90     Last resort to find a service by trying to match pattern to programs in memory
  91     '''
  92     found = False
  93     if platform.system() == 'SunOS':
  94         flags = '-ef'
  95     else:
  96         flags = 'auxww'
  97     psbin = module.get_bin_path('ps', True)
  98 
  99     (rc, psout, pserr) = module.run_command([psbin, flags])
 100     if rc == 0:
 101         for line in psout.splitlines():
 102             if pattern in line:
 103                 # FIXME: should add logic to prevent matching 'self', though that should be extreemly rare
 104                 found = True
 105                 break
 106     return found

 109 def fail_if_missing(module, found, service, msg=''):
 110     '''
 111     This function will return an error or exit gracefully depending on check mode status
 112     and if the service is missing or not.

 123 def fork_process():
 124     '''
 125     This function performs the double fork process to detach from the
 126     parent process and execute.
 127     '''
 128     pid = os.fork()
 129 
 130     if pid == 0:
 131         # Set stdin/stdout/stderr to /dev/null
 132         fd = os.open(os.devnull, os.O_RDWR)
 133 
 134         # clone stdin/out/err
 135         for num in range(3):
 136             if fd != num:
 137                 os.dup2(fd, num)
 138 
 139         # close otherwise
 140         if fd not in range(3):
 141             os.close(fd)
 142 
 143         # Make us a daemon
 144         pid = os.fork()
 145 
 146         # end if not in child
 147         if pid > 0:
 148             os._exit(0)
 149 
 150         # get new process session and detach
 151         sid = os.setsid()
 152         if sid == -1:
 153             raise Exception("Unable to detach session while daemonizing")
 154 
 155         # avoid possible problems with cwd being removed
 156         os.chdir("/")
 157 
 158         pid = os.fork()
 159         if pid > 0:
 160             os._exit(0)
 161 
 162     return pid

 165 def daemonize(module, cmd):
 166     '''
 167     Execute a command while detaching as a daemon, returns rc, stdout, and stderr.
 168 
 169     :arg module: is an  AnsibleModule object, used for it's utility methods
 170     :arg cmd: is a list or string representing the command and options to run
 171 
 172     This is complex because daemonization is hard for people.
 173     What we do is daemonize a part of this module, the daemon runs the command,
 174     picks up the return code and output, and returns it to the main process.
 175     '''
 176 
 177     # init some vars
 178     chunk = 4096  # FIXME: pass in as arg?
 179     errors = 'surrogate_or_strict'
 180 
 181     # start it!
 182     try:
 183         pipe = os.pipe()
 184         pid = fork_process()
 185     except OSError:
 186         module.fail_json(msg="Error while attempting to fork: %s", exception=traceback.format_exc())
 187     except Exception as exc:
 188         module.fail_json(msg=to_text(exc), exception=traceback.format_exc())
 189 
 190     # we don't do any locking as this should be a unique module/process
 191     if pid == 0:
 192         os.close(pipe[0])
 193 
 194         # if command is string deal with  py2 vs py3 conversions for shlex
 195         if not isinstance(cmd, list):
 196             if PY2:
 197                 cmd = shlex.split(to_bytes(cmd, errors=errors))
 198             else:
 199                 cmd = shlex.split(to_text(cmd, errors=errors))
 200 
 201         # make sure we always use byte strings
 202         run_cmd = []
 203         for c in cmd:
 204             run_cmd.append(to_bytes(c, errors=errors))
 205 
 206         # execute the command in forked process
 207         p = subprocess.Popen(run_cmd, shell=False, stdout=subprocess.PIPE, stderr=subprocess.PIPE, preexec_fn=lambda: os.close(pipe[1]))
 208         fds = [p.stdout, p.stderr]
 209 
 210         # loop reading output till its done
 211         output = {p.stdout: b(""), p.stderr: b("")}
 212         while fds:
 213             rfd, wfd, efd = select.select(fds, [], fds, 1)
 214             if (rfd + wfd + efd) or p.poll():
 215                 for out in fds:
 216                     if out in rfd:
 217                         data = os.read(out.fileno(), chunk)
 218                         if not data:
 219                             fds.remove(out)
 220                     output[out] += b(data)
 221 
 222         # even after fds close, we might want to wait for pid to die
 223         p.wait()
 224 
 225         # Return a pickled data of parent
 226         return_data = pickle.dumps([p.returncode, to_text(output[p.stdout]), to_text(output[p.stderr])], protocol=pickle.HIGHEST_PROTOCOL)
 227         os.write(pipe[1], to_bytes(return_data, errors=errors))
 228 
 229         # clean up
 230         os.close(pipe[1])
 231         os._exit(0)
 232 
 233     elif pid == -1:
 234         module.fail_json(msg="Unable to fork, no exception thrown, probably due to lack of resources, check logs.")
 235 
 236     else:
 237         # in parent
 238         os.close(pipe[1])
 239         os.waitpid(pid, 0)
 240 
 241         # Grab response data after child finishes
 242         return_data = b("")
 243         while True:
 244             rfd, wfd, efd = select.select([pipe[0]], [], [pipe[0]])
 245             if pipe[0] in rfd:
 246                 data = os.read(pipe[0], chunk)
 247                 if not data:
 248                     break
 249                 return_data += b(data)
 250 
 251         # Note: no need to specify encoding on py3 as this module sends the
 252         # pickle to itself (thus same python interpreter so we aren't mixing
 253         # py2 and py3)
 254         return pickle.loads(to_bytes(return_data, errors=errors))

 257 def check_ps(module, pattern):
 258 
 259     # Set ps flags
 260     if platform.system() == 'SunOS':
 261         psflags = '-ef'
 262     else:
 263         psflags = 'auxww'
 264 
 265     # Find ps binary
 266     psbin = module.get_bin_path('ps', True)
 267 
 268     (rc, out, err) = module.run_command('%s %s' % (psbin, psflags))
 269     # If rc is 0, set running as appropriate
 270     if rc == 0:
 271         for line in out.split('\n'):
 272             if pattern in line:
 273                 return True
 274     return False
