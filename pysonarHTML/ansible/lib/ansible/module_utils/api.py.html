  28 """
  29 This module adds shared support for generic api modules
  30 
  31 In order to use this module, include it as part of a custom
  32 module as shown below.
  33 
  34 The 'api' module provides the following common argument specs:
  35 
  36     * rate limit spec
  37         - rate: number of requests per time unit (int)
  38         - rate_limit: time window in which the limit is applied in seconds
  39 
  40     * retry spec
  41         - retries: number of attempts
  42         - retry_pause: delay between attempts in seconds
  43 """
  44 from __future__ import (absolute_import, division, print_function)
  45 __metaclass__ = type
  46 
  47 import time

  50 def rate_limit_argument_spec(spec=None):
  51     """Creates an argument spec for working with rate limiting"""
  52     arg_spec = (dict(
  53         rate=dict(type='int'),
  54         rate_limit=dict(type='int'),
  55     ))
  56     if spec:
  57         arg_spec.update(spec)
  58     return arg_spec
  59 
  60 
  61 def retry_argument_spec(spec=None):
  62     """Creates an argument spec for working with retrying"""
  63     arg_spec = (dict(
  64         retries=dict(type='int'),
  65         retry_pause=dict(type='float', default=1),
  66     ))
  67     if spec:
  68         arg_spec.update(spec)
  69     return arg_spec

  84 def rate_limit(rate=None, rate_limit=None):
  85     """rate limiting decorator"""
  86     minrate = None
  87     if rate is not None and rate_limit is not None:
  88         minrate = float(rate_limit) / float(rate)
  89 
  90     def wrapper(f):
  91         last = [0.0]
  92 
  93         def ratelimited(*args, **kwargs):
  94             if minrate is not None:
  95                 elapsed = time.clock() - last[0]
  96                 left = minrate - elapsed
  97                 if left > 0:
  98                     time.sleep(left)
  99                 last[0] = time.clock()
 100             ret = f(*args, **kwargs)
 101             return ret
 102 
 103         return ratelimited
 104     return wrapper
 105

 107 def retry(retries=None, retry_pause=1):
 108     """Retry decorator"""
 109     def wrapper(f):
 110         retry_count = 0
 111 
 112         def retried(*args, **kwargs):
 113             if retries is not None:
 114                 ret = None
 115                 while True:
 116                     # pylint doesn't understand this is a closure
 117                     retry_count += 1  # pylint: disable=undefined-variable
 118                     if retry_count >= retries:
 119                         raise Exception("Retry limit exceeded: %d" % retries)
 120                     try:
 121                         ret = f(*args, **kwargs)
 122                     except Exception:
 123                         pass
 124                     if ret:
 125                         break
 126                     time.sleep(retry_pause)
 127                 return ret
 128 
 129         return retried
 130     return wrapper
