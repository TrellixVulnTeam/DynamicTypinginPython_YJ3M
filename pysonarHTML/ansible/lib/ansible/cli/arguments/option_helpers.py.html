  26 class SortingHelpFormatter(argparse.HelpFormatter):
  27     def add_arguments(self, actions):
  28         actions = sorted(actions, key=operator.attrgetter('option_strings'))
  29         super(SortingHelpFormatter, self).add_arguments(actions)

  32 class AnsibleVersion(argparse.Action):
  33     def __call__(self, parser, namespace, values, option_string=None):
  34         ansible_version = to_native(version(getattr(parser, 'prog')))
  35         print(ansible_version)
  36         parser.exit()

  48 class PrependListAction(argparse.Action):
  49     """A near clone of ``argparse._AppendAction``, but designed to prepend list values
  50     instead of appending.
  51     """
  52     def __init__(self, option_strings, dest, nargs=None, const=None, default=None, type=None,
  53                  choices=None, required=False, help=None, metavar=None):
  54         if nargs == 0:
  55             raise ValueError('nargs for append actions must be > 0; if arg '
  56                              'strings are not supplying the value to append, '
  57                              'the append const action may be more appropriate')
  58         if const is not None and nargs != argparse.OPTIONAL:
  59             raise ValueError('nargs must be %r to supply const' % argparse.OPTIONAL)
  60         super(PrependListAction, self).__init__(
  61             option_strings=option_strings,
  62             dest=dest,
  63             nargs=nargs,
  64             const=const,
  65             default=default,
  66             type=type,
  67             choices=choices,
  68             required=required,
  69             help=help,
  70             metavar=metavar
  71         )
  72 
  73     def __call__(self, parser, namespace, values, option_string=None):
  74         items = copy.copy(ensure_value(namespace, self.dest, []))
  75         items[0:0] = values
  76         setattr(namespace, self.dest, items)

  79 def ensure_value(namespace, name, value):
  80     if getattr(namespace, name, None) is None:
  81         setattr(namespace, name, value)
  82     return getattr(namespace, name)

  88 def unfrack_path(pathsep=False):
  89     """Turn an Option's data into a single path in Ansible locations"""
  90     def inner(value):
  91         if pathsep:
  92             return [unfrackpath(x) for x in value.split(os.pathsep) if x]
  93 
  94         if value == '-':
  95             return value
  96 
  97         return unfrackpath(value)
  98     return inner

 101 def _git_repo_info(repo_path):
 102     """ returns a string containing git branch, commit id and commit date """
 103     result = None
 104     if os.path.exists(repo_path):
 105         # Check if the .git is a file. If it is a file, it means that we are in a submodule structure.
 106         if os.path.isfile(repo_path):
 107             try:
 108                 gitdir = yaml.safe_load(open(repo_path)).get('gitdir')
 109                 # There is a possibility the .git file to have an absolute path.
 110                 if os.path.isabs(gitdir):
 111                     repo_path = gitdir
 112                 else:
 113                     repo_path = os.path.join(repo_path[:-4], gitdir)
 114             except (IOError, AttributeError):
 115                 return ''
 116         with open(os.path.join(repo_path, "HEAD")) as f:
 117             line = f.readline().rstrip("\n")
 118             if line.startswith("ref:"):
 119                 branch_path = os.path.join(repo_path, line[5:])
 120             else:
 121                 branch_path = None
 122         if branch_path and os.path.exists(branch_path):
 123             branch = '/'.join(line.split('/')[2:])
 124             with open(branch_path) as f:
 125                 commit = f.readline()[:10]
 126         else:
 127             # detached HEAD
 128             commit = line[:10]
 129             branch = 'detached HEAD'
 130             branch_path = os.path.join(repo_path, "HEAD")
 131 
 132         date = time.localtime(os.stat(branch_path).st_mtime)
 133         if time.daylight == 0:
 134             offset = time.timezone
 135         else:
 136             offset = time.altzone
 137         result = "({0} {1}) last updated {2} (GMT {3:+04d})".format(branch, commit, time.strftime("%Y/%m/%d %H:%M:%S", date), int(offset / -36))
 138     else:
 139         result = ''
 140     return result

 143 def _gitinfo():
 144     basedir = os.path.join(os.path.dirname(__file__), '..', '..', '..')
 145     repo_path = os.path.join(basedir, '.git')
 146     result = _git_repo_info(repo_path)
 147     submodules = os.path.join(basedir, '.gitmodules')
 148 
 149     if not os.path.exists(submodules):
 150         return result
 151 
 152     with open(submodules) as f:
 153         for line in f:
 154             tokens = line.strip().split(' ')
 155             if tokens[0] == 'path':
 156                 submodule_path = tokens[2]
 157                 submodule_info = _git_repo_info(os.path.join(basedir, submodule_path, '.git'))
 158                 if not submodule_info:
 159                     submodule_info = ' not found - use git submodule update --init ' + submodule_path
 160                 result += "\n  {0}: {1}".format(submodule_path, submodule_info)
 161     return result

 164 def version(prog=None):
 165     """ return ansible version """
 166     if prog:
 167         result = " ".join((prog, __version__))
 168     else:
 169         result = __version__
 170 
 171     gitinfo = _gitinfo()
 172     if gitinfo:
 173         result = result + " {0}".format(gitinfo)
 174     result += "\n  config file = %s" % C.CONFIG_FILE
 175     if C.DEFAULT_MODULE_PATH is None:
 176         cpath = "Default w/o overrides"
 177     else:
 178         cpath = C.DEFAULT_MODULE_PATH
 179     result = result + "\n  configured module search path = %s" % cpath
 180     result = result + "\n  ansible python module location = %s" % ':'.join(ansible.__path__)
 181     result = result + "\n  ansible collection location = %s" % ':'.join(C.COLLECTIONS_PATHS)
 182     result = result + "\n  executable location = %s" % sys.argv[0]
 183     result = result + "\n  python version = %s" % ''.join(sys.version.splitlines())
 184     return result

 191 def create_base_parser(prog, usage="", desc=None, epilog=None):
 192     """
 193     Create an options parser for all ansible scripts
 194     """
 195     # base opts
 196     parser = argparse.ArgumentParser(
 197         prog=prog,
 198         formatter_class=SortingHelpFormatter,
 199         epilog=epilog,
 200         description=desc,
 201         conflict_handler='resolve',
 202     )
 203     version_help = "show program's version number, config file location, configured module search path," \
 204                    " module location, executable location and exit"
 205 
 206     parser.add_argument('--version', action=AnsibleVersion, nargs=0, help=version_help)
 207     add_verbosity_options(parser)
 208     return parser

 309 def add_runas_options(parser):
 310     """
 311     Add options for commands which can run tasks as another user
 312 
 313     Note that this includes the options from add_runas_prompt_options().  Only one of these

 325                              help='run operations as this user (default=%s)' % C.DEFAULT_BECOME_USER)
