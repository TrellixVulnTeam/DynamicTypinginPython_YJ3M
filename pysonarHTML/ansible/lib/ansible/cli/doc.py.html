 115     def display_plugin_list(self, results):
 116 
 117         # format for user
 118         displace = max(len(x) for x in self.plugin_list)
 119         linelimit = display.columns - displace - 5
 120         text = []
 121 
 122         # format display per option
 123         if context.CLIARGS['list_files']:
 124             # list plugin file names
 125             for plugin in results.keys():
 126                 filename = results[plugin]
 127                 text.append("%-*s %-*.*s" % (displace, plugin, linelimit, len(filename), filename))
 128         else:
 129             # list plugin names and short desc
 130             deprecated = []
 131             for plugin in results.keys():
 132                 desc = DocCLI.tty_ify(results[plugin])
 133 
 134                 if len(desc) > linelimit:
 135                     desc = desc[:linelimit] + '...'
 136 
 137                 if plugin.startswith('_'):  # Handle deprecated # TODO: add mark for deprecated collection plugins
 138                     deprecated.append("%-*s %-*.*s" % (displace, plugin[1:], linelimit, len(desc), desc))
 139                 else:
 140                     text.append("%-*s %-*.*s" % (displace, plugin, linelimit, len(desc), desc))
 141 
 142                 if len(deprecated) > 0:
 143                     text.append("\nDEPRECATED:")
 144                     text.extend(deprecated)
 145 
 146         # display results
 147         DocCLI.pager("\n".join(text))

 149     def run(self):
 150 
 151         super(DocCLI, self).run()
 152 
 153         plugin_type = context.CLIARGS['type']
 154         do_json = context.CLIARGS['json_format']
 155 
 156         if plugin_type in C.DOCUMENTABLE_PLUGINS:
 157             loader = getattr(plugin_loader, '%s_loader' % plugin_type)
 158         else:
 159             raise AnsibleOptionsError("Unknown or undocumentable plugin type: %s" % plugin_type)
 160 
 161         # add to plugin paths from command line
 162         basedir = context.CLIARGS['basedir']
 163         if basedir:
 164             AnsibleCollectionConfig.playbook_paths = basedir
 165             loader.add_directory(basedir, with_subdir=True)
 166 
 167         if context.CLIARGS['module_path']:
 168             for path in context.CLIARGS['module_path']:
 169                 if path:
 170                     loader.add_directory(path)
 171 
 172         # save only top level paths for errors
 173         search_paths = DocCLI.print_paths(loader)
 174         loader._paths = None  # reset so we can use subdirs below
 175 
 176         # list plugins names or filepath for type, both options share most code
 177         if context.CLIARGS['list_files'] or context.CLIARGS['list_dir']:
 178 
 179             coll_filter = None
 180             if len(context.CLIARGS['args']) == 1:
 181                 coll_filter = context.CLIARGS['args'][0]
 182 
 183             if coll_filter in ('', None):
 184                 paths = loader._get_paths_with_context()
 185                 for path_context in paths:
 186                     self.plugin_list.update(
 187                         DocCLI.find_plugins(path_context.path, path_context.internal, plugin_type))
 188 
 189             add_collection_plugins(self.plugin_list, plugin_type, coll_filter=coll_filter)
 190 
 191             # get appropriate content depending on option
 192             if context.CLIARGS['list_dir']:
 193                 results = self._get_plugin_list_descriptions(loader)
 194             elif context.CLIARGS['list_files']:
 195                 results = self._get_plugin_list_filenames(loader)
 196 
 197             if do_json:
 198                 jdump(results)
 199             elif self.plugin_list:
 200                 self.display_plugin_list(results)
 201             else:
 202                 display.warning("No plugins found.")
 203         # dump plugin desc/data as JSON
 204         elif context.CLIARGS['dump']:
 205             plugin_data = {}
 206             plugin_names = DocCLI.get_all_plugins_of_type(plugin_type)
 207             for plugin_name in plugin_names:
 208                 plugin_info = DocCLI.get_plugin_metadata(plugin_type, plugin_name)
 209                 if plugin_info is not None:
 210                     plugin_data[plugin_name] = plugin_info
 211 
 212             jdump(plugin_data)
 213         else:
 214             # display specific plugin docs
 215             if len(context.CLIARGS['args']) == 0:
 216                 raise AnsibleOptionsError("Incorrect options passed")
 217 
 218             # get the docs for plugins in the command line list
 219             plugin_docs = {}
 220             for plugin in context.CLIARGS['args']:
 221                 try:
 222                     doc, plainexamples, returndocs, metadata = DocCLI._get_plugin_doc(plugin, plugin_type, loader, search_paths)
 223                 except PluginNotFound:
 224                     display.warning("%s %s not found in:\n%s\n" % (plugin_type, plugin, search_paths))
 225                     continue
 226                 except Exception as e:
 227                     display.vvv(traceback.format_exc())
 228                     raise AnsibleError("%s %s missing documentation (or could not parse"
 229                                        " documentation): %s\n" %
 230                                        (plugin_type, plugin, to_native(e)))
 231 
 232                 if not doc:
 233                     # The doc section existed but was empty
 234                     continue
 235 
 236                 plugin_docs[plugin] = {'doc': doc, 'examples': plainexamples, 'return': returndocs, 'metadata': metadata}
 237 
 238             if do_json:
 239                 jdump(plugin_docs)
 240 
 241             else:
 242                 # Some changes to how plain text docs are formatted
 243                 text = []
 244                 for plugin, doc_data in plugin_docs.items():
 245                     textret = DocCLI.format_plugin_doc(plugin, plugin_type,
 246                                                        doc_data['doc'], doc_data['examples'],
 247                                                        doc_data['return'], doc_data['metadata'])
 248                     if textret:
 249                         text.append(textret)
 250                     else:
 251                         display.warning("No valid documentation was retrieved from '%s'" % plugin)
 252 
 253                 if text:
 254                     DocCLI.pager(''.join(text))
 255 
 256         return 0

 258     @staticmethod
 259     def get_all_plugins_of_type(plugin_type):
 260         loader = getattr(plugin_loader, '%s_loader' % plugin_type)
 261         plugin_list = set()
 262         paths = loader._get_paths_with_context()
 263         for path_context in paths:
 264             plugins_to_add = DocCLI.find_plugins(path_context.path, path_context.internal, plugin_type)
 265             plugin_list.update(plugins_to_add)
 266         return sorted(set(plugin_list))

 268     @staticmethod
 269     def get_plugin_metadata(plugin_type, plugin_name):
 270         # if the plugin lives in a non-python file (eg, win_X.ps1), require the corresponding python file for docs
 271         loader = getattr(plugin_loader, '%s_loader' % plugin_type)
 272         result = loader.find_plugin_with_context(plugin_name, mod_type='.py', ignore_deprecated=True, check_aliases=True)
 273         if not result.resolved:
 274             raise AnsibleError("unable to load {0} plugin named {1} ".format(plugin_type, plugin_name))
 275         filename = result.plugin_resolved_path
 276         collection_name = result.plugin_resolved_collection
 277 
 278         try:
 279             doc, __, __, __ = get_docstring(filename, fragment_loader, verbose=(context.CLIARGS['verbosity'] > 0),
 280                                             collection_name=collection_name, is_module=(plugin_type == 'module'))
 281         except Exception:
 282             display.vvv(traceback.format_exc())
 283             raise AnsibleError("%s %s at %s has a documentation formatting error or is missing documentation." % (plugin_type, plugin_name, filename))
 284 
 285         if doc is None:
 286             # Removed plugins don't have any documentation
 287             return None
 288 
 289         return dict(
 290             name=plugin_name,
 291             namespace=DocCLI.namespace_from_plugin_filepath(filename, plugin_name, loader.package_path),
 292             description=doc.get('short_description', "UNKNOWN"),
 293             version_added=doc.get('version_added', "UNKNOWN")
 294         )

 296     @staticmethod
 297     def namespace_from_plugin_filepath(filepath, plugin_name, basedir):
 298         if not basedir.endswith('/'):
 299             basedir += '/'
 300         rel_path = filepath.replace(basedir, '')
 301         extension_free = os.path.splitext(rel_path)[0]
 302         namespace_only = extension_free.rsplit(plugin_name, 1)[0].strip('/_')
 303         clean_ns = namespace_only.replace('/', '.')
 304         if clean_ns == '':
 305             clean_ns = None
 306 
 307         return clean_ns

 309     @staticmethod
 310     def _get_plugin_doc(plugin, plugin_type, loader, search_paths):
 311         # if the plugin lives in a non-python file (eg, win_X.ps1), require the corresponding python file for docs
 312         result = loader.find_plugin_with_context(plugin, mod_type='.py', ignore_deprecated=True, check_aliases=True)
 313         if not result.resolved:
 314             raise PluginNotFound('%s was not found in %s' % (plugin, search_paths))
 315         plugin_name = result.plugin_resolved_name
 316         filename = result.plugin_resolved_path
 317         collection_name = result.plugin_resolved_collection
 318 
 319         doc, plainexamples, returndocs, metadata = get_docstring(
 320             filename, fragment_loader, verbose=(context.CLIARGS['verbosity'] > 0),
 321             collection_name=collection_name, is_module=(plugin_type == 'module'))
 322 
 323         # If the plugin existed but did not have a DOCUMENTATION element and was not removed, it's an error
 324         if doc is None:
 325             raise ValueError('%s did not contain a DOCUMENTATION attribute' % plugin)
 326 
 327         doc['filename'] = filename
 328         doc['collection'] = collection_name
 329         return doc, plainexamples, returndocs, metadata

 331     @staticmethod
 332     def format_plugin_doc(plugin, plugin_type, doc, plainexamples, returndocs, metadata):
 333         collection_name = doc['collection']
 334 
 335         # TODO: do we really want this?
 336         # add_collection_to_versions_and_dates(doc, '(unknown)', is_module=(plugin_type == 'module'))
 337         # remove_current_collection_from_versions_and_dates(doc, collection_name, is_module=(plugin_type == 'module'))
 338         # remove_current_collection_from_versions_and_dates(
 339         #     returndocs, collection_name, is_module=(plugin_type == 'module'), return_docs=True)
 340 
 341         # assign from other sections
 342         doc['plainexamples'] = plainexamples
 343         doc['returndocs'] = returndocs
 344         doc['metadata'] = metadata
 345 
 346         # generate extra data
 347         if plugin_type == 'module':
 348             # is there corresponding action plugin?
 349             if plugin in action_loader:
 350                 doc['action'] = True
 351             else:
 352                 doc['action'] = False
 353 
 354         doc['now_date'] = datetime.date.today().strftime('%Y-%m-%d')
 355         if 'docuri' in doc:
 356             doc['docuri'] = doc[plugin_type].replace('_', '-')
 357 
 358         if context.CLIARGS['show_snippet'] and plugin_type == 'module':
 359             text = DocCLI.get_snippet_text(doc)
 360         else:
 361             try:
 362                 text = DocCLI.get_man_text(doc, collection_name)
 363             except Exception as e:
 364                 raise AnsibleError("Unable to retrieve documentation from '%s' due to: %s" % (plugin, to_native(e)))
 365 
 366         return text

 368     @staticmethod
 369     def find_plugins(path, internal, ptype, collection=None):
 370         # if internal, collection could be set to `ansible.builtin`
 371 
 372         display.vvvv("Searching %s for plugins" % path)
 373 
 374         plugin_list = set()
 375 
 376         if not os.path.exists(path):
 377             display.vvvv("%s does not exist" % path)
 378             return plugin_list
 379 
 380         if not os.path.isdir(path):
 381             display.vvvv("%s is not a directory" % path)
 382             return plugin_list
 383 
 384         bkey = ptype.upper()
 385         for plugin in os.listdir(path):
 386             display.vvvv("Found %s" % plugin)
 387             full_path = '/'.join([path, plugin])
 388 
 389             if plugin.startswith('.'):
 390                 continue
 391             elif os.path.isdir(full_path):
 392                 continue
 393             elif any(plugin.endswith(x) for x in C.BLACKLIST_EXTS):
 394                 continue
 395             elif plugin.startswith('__'):
 396                 continue
 397             elif plugin in C.IGNORE_FILES:
 398                 continue
 399             elif plugin .startswith('_'):
 400                 if os.path.islink(full_path):  # avoids aliases
 401                     continue
 402 
 403             plugin = os.path.splitext(plugin)[0]  # removes the extension
 404             plugin = plugin.lstrip('_')  # remove underscore from deprecated plugins
 405 
 406             if plugin not in BLACKLIST.get(bkey, ()):
 407 
 408                 if collection:
 409                     plugin = '%s.%s' % (collection, plugin)
 410 
 411                 plugin_list.add(plugin)
 412                 display.vvvv("Added %s" % plugin)
 413 
 414         return plugin_list

 416     def _get_plugin_list_descriptions(self, loader):
 417 
 418         descs = {}
 419         plugins = self._get_plugin_list_filenames(loader)
 420         for plugin in plugins.keys():
 421 
 422             filename = plugins[plugin]
 423 
 424             doc = None
 425             try:
 426                 doc = read_docstub(filename)
 427             except Exception:
 428                 display.warning("%s has a documentation formatting error" % plugin)
 429                 continue
 430 
 431             if not doc or not isinstance(doc, dict):
 432                 desc = 'UNDOCUMENTED'
 433             else:
 434                 desc = doc.get('short_description', 'INVALID SHORT DESCRIPTION').strip()
 435 
 436             descs[plugin] = desc
 437 
 438         return descs

 440     def _get_plugin_list_filenames(self, loader):
 441         pfiles = {}
 442         for plugin in sorted(self.plugin_list):
 443 
 444             try:
 445                 # if the module lives in a non-python file (eg, win_X.ps1), require the corresponding python file for docs
 446                 filename = loader.find_plugin(plugin, mod_type='.py', ignore_deprecated=True, check_aliases=True)
 447 
 448                 if filename is None:
 449                     continue
 450                 if filename.endswith(".ps1"):
 451                     continue
 452                 if os.path.isdir(filename):
 453                     continue
 454 
 455                 pfiles[plugin] = filename
 456 
 457             except Exception as e:
 458                 raise AnsibleError("Failed reading docs at %s: %s" % (plugin, to_native(e)), orig_exc=e)
 459 
 460         return pfiles

 474     @staticmethod
 475     def get_snippet_text(doc):
 476 
 477         text = []
 478         desc = DocCLI.tty_ify(doc['short_description'])
 479         text.append("- name: %s" % (desc))
 480         text.append("  %s:" % (doc['module']))
 481         pad = 31
 482         subdent = " " * pad
 483         limit = display.columns - pad
 484 
 485         for o in sorted(doc['options'].keys()):
 486             opt = doc['options'][o]
 487             if isinstance(opt['description'], string_types):
 488                 desc = DocCLI.tty_ify(opt['description'])
 489             else:
 490                 desc = DocCLI.tty_ify(" ".join(opt['description']))
 491 
 492             required = opt.get('required', False)
 493             if not isinstance(required, bool):
 494                 raise("Incorrect value for 'Required', a boolean is needed.: %s" % required)
 495             if required:
 496                 desc = "(required) %s" % desc
 497             o = '%s:' % o
 498             text.append("      %-20s   # %s" % (o, textwrap.fill(desc, limit, subsequent_indent=subdent)))
 499         text.append('')
 500 
 501         return "\n".join(text)

 509     @staticmethod
 510     def add_fields(text, fields, limit, opt_indent, return_values=False, base_indent=''):
 511 
 512         for o in sorted(fields):
 513             # Create a copy so we don't modify the original (in case YAML anchors have been used)
 514             opt = dict(fields[o])
 515 
 516             required = opt.pop('required', False)
 517             if not isinstance(required, bool):
 518                 raise AnsibleError("Incorrect value for 'Required', a boolean is needed.: %s" % required)
 519             if required:
 520                 opt_leadin = "="
 521             else:
 522                 opt_leadin = "-"
 523 
 524             text.append("%s%s %s" % (base_indent, opt_leadin, o))
 525 
 526             if 'description' not in opt:
 527                 raise AnsibleError("All (sub-)options and return values must have a 'description' field")
 528             if isinstance(opt['description'], list):
 529                 for entry_idx, entry in enumerate(opt['description'], 1):
 530                     if not isinstance(entry, string_types):
 531                         raise AnsibleError("Expected string in description of %s at index %s, got %s" % (o, entry_idx, type(entry)))
 532                     text.append(textwrap.fill(DocCLI.tty_ify(entry), limit, initial_indent=opt_indent, subsequent_indent=opt_indent))
 533             else:
 534                 if not isinstance(opt['description'], string_types):
 535                     raise AnsibleError("Expected string in description of %s, got %s" % (o, type(opt['description'])))
 536                 text.append(textwrap.fill(DocCLI.tty_ify(opt['description']), limit, initial_indent=opt_indent, subsequent_indent=opt_indent))
 537             del opt['description']
 538 
 539             aliases = ''
 540             if 'aliases' in opt:
 541                 if len(opt['aliases']) > 0:
 542                     aliases = "(Aliases: " + ", ".join(to_text(i) for i in opt['aliases']) + ")"
 543                 del opt['aliases']
 544             choices = ''
 545             if 'choices' in opt:
 546                 if len(opt['choices']) > 0:
 547                     choices = "(Choices: " + ", ".join(to_text(i) for i in opt['choices']) + ")"
 548                 del opt['choices']
 549             default = ''
 550             if not return_values:
 551                 if 'default' in opt or not required:
 552                     default = "[Default: %s" % to_text(opt.pop('default', '(null)')) + "]"
 553 
 554             text.append(textwrap.fill(DocCLI.tty_ify(aliases + choices + default), limit,
 555                                       initial_indent=opt_indent, subsequent_indent=opt_indent))
 556 
 557             suboptions = []
 558             for subkey in ('options', 'suboptions', 'contains', 'spec'):
 559                 if subkey in opt:
 560                     suboptions.append((subkey, opt.pop(subkey)))
 561 
 562             conf = {}
 563             for config in ('env', 'ini', 'yaml', 'vars', 'keywords'):
 564                 if config in opt and opt[config]:
 565                     # Create a copy so we don't modify the original (in case YAML anchors have been used)
 566                     conf[config] = [dict(item) for item in opt.pop(config)]
 567                     for ignore in DocCLI.IGNORE:
 568                         for item in conf[config]:
 569                             if ignore in item:
 570                                 del item[ignore]
 571 
 572             if conf:
 573                 text.append(DocCLI._dump_yaml({'set_via': conf}, opt_indent))
 574 
 575             for k in sorted(opt):
 576                 if k.startswith('_'):
 577                     continue
 578                 if isinstance(opt[k], string_types):
 579                     text.append('%s%s: %s' % (opt_indent, k,
 580                                               textwrap.fill(DocCLI.tty_ify(opt[k]),
 581                                                             limit - (len(k) + 2),
 582                                                             subsequent_indent=opt_indent)))
 583                 elif isinstance(opt[k], (Sequence)) and all(isinstance(x, string_types) for x in opt[k]):
 584                     text.append(DocCLI.tty_ify('%s%s: %s' % (opt_indent, k, ', '.join(opt[k]))))
 585                 else:
 586                     text.append(DocCLI._dump_yaml({k: opt[k]}, opt_indent))
 587 
 588             for subkey, subdata in suboptions:
 589                 text.append('')
 590                 text.append("%s%s:\n" % (opt_indent, subkey.upper()))
 591                 DocCLI.add_fields(text, subdata, limit, opt_indent + '    ', return_values, opt_indent)
 592             if not suboptions:
 593                 text.append('')

 595     @staticmethod
 596     def get_man_text(doc, collection_name=''):
 597         # Create a copy so we don't modify the original
 598         doc = dict(doc)
 599 
 600         DocCLI.IGNORE = DocCLI.IGNORE + (context.CLIARGS['type'],)
 601         opt_indent = "        "
 602         text = []
 603         pad = display.columns * 0.20
 604         limit = max(display.columns - int(pad), 70)
 605 
 606         plugin_name = doc.get(context.CLIARGS['type'], doc.get('plugin_type'))
 607         if collection_name:
 608             plugin_name = '%s.%s' % (collection_name, plugin_name)
 609 
 610         text.append("> %s    (%s)\n" % (plugin_name.upper(), doc.pop('filename')))
 611 
 612         if isinstance(doc['description'], list):
 613             desc = " ".join(doc.pop('description'))
 614         else:
 615             desc = doc.pop('description')
 616 
 617         text.append("%s\n" % textwrap.fill(DocCLI.tty_ify(desc), limit, initial_indent=opt_indent,
 618                                            subsequent_indent=opt_indent))
 619 
 620         if doc.get('deprecated', False):
 621             text.append("DEPRECATED: \n")
 622             if isinstance(doc['deprecated'], dict):
 623                 if 'removed_at_date' in doc['deprecated']:
 624                     text.append(
 625                         "\tReason: %(why)s\n\tWill be removed in a release after %(removed_at_date)s\n\tAlternatives: %(alternative)s" % doc.pop('deprecated')
 626                     )
 627                 else:
 628                     if 'version' in doc['deprecated'] and 'removed_in' not in doc['deprecated']:
 629                         doc['deprecated']['removed_in'] = doc['deprecated']['version']
 630                     text.append("\tReason: %(why)s\n\tWill be removed in: Ansible %(removed_in)s\n\tAlternatives: %(alternative)s" % doc.pop('deprecated'))
 631             else:
 632                 text.append("%s" % doc.pop('deprecated'))
 633             text.append("\n")
 634 
 635         if doc.pop('action', False):
 636             text.append("  * note: %s\n" % "This module has a corresponding action plugin.")
 637 
 638         if doc.get('options', False):
 639             text.append("OPTIONS (= is mandatory):\n")
 640             DocCLI.add_fields(text, doc.pop('options'), limit, opt_indent)
 641             text.append('')
 642 
 643         if doc.get('notes', False):
 644             text.append("NOTES:")
 645             for note in doc['notes']:
 646                 text.append(textwrap.fill(DocCLI.tty_ify(note), limit - 6,
 647                                           initial_indent=opt_indent[:-2] + "* ", subsequent_indent=opt_indent))
 648             text.append('')
 649             text.append('')
 650             del doc['notes']
 651 
 652         if doc.get('seealso', False):
 653             text.append("SEE ALSO:")
 654             for item in doc['seealso']:
 655                 if 'module' in item:
 656                     text.append(textwrap.fill(DocCLI.tty_ify('Module %s' % item['module']),
 657                                 limit - 6, initial_indent=opt_indent[:-2] + "* ", subsequent_indent=opt_indent))
 658                     description = item.get('description', 'The official documentation on the %s module.' % item['module'])
 659                     text.append(textwrap.fill(DocCLI.tty_ify(description), limit - 6, initial_indent=opt_indent + '   ', subsequent_indent=opt_indent + '   '))
 660                     text.append(textwrap.fill(DocCLI.tty_ify(get_versioned_doclink('modules/%s_module.html' % item['module'])),
 661                                 limit - 6, initial_indent=opt_indent + '   ', subsequent_indent=opt_indent))
 662                 elif 'name' in item and 'link' in item and 'description' in item:
 663                     text.append(textwrap.fill(DocCLI.tty_ify(item['name']),
 664                                 limit - 6, initial_indent=opt_indent[:-2] + "* ", subsequent_indent=opt_indent))
 665                     text.append(textwrap.fill(DocCLI.tty_ify(item['description']),
 666                                 limit - 6, initial_indent=opt_indent + '   ', subsequent_indent=opt_indent + '   '))
 667                     text.append(textwrap.fill(DocCLI.tty_ify(item['link']),
 668                                 limit - 6, initial_indent=opt_indent + '   ', subsequent_indent=opt_indent + '   '))
 669                 elif 'ref' in item and 'description' in item:
 670                     text.append(textwrap.fill(DocCLI.tty_ify('Ansible documentation [%s]' % item['ref']),
 671                                 limit - 6, initial_indent=opt_indent[:-2] + "* ", subsequent_indent=opt_indent))
 672                     text.append(textwrap.fill(DocCLI.tty_ify(item['description']),
 673                                 limit - 6, initial_indent=opt_indent + '   ', subsequent_indent=opt_indent + '   '))
 674                     text.append(textwrap.fill(DocCLI.tty_ify(get_versioned_doclink('/#stq=%s&stp=1' % item['ref'])),
 675                                 limit - 6, initial_indent=opt_indent + '   ', subsequent_indent=opt_indent + '   '))
 676 
 677             text.append('')
 678             text.append('')
 679             del doc['seealso']
 680 
 681         if doc.get('requirements', False):
 682             req = ", ".join(doc.pop('requirements'))
 683             text.append("REQUIREMENTS:%s\n" % textwrap.fill(DocCLI.tty_ify(req), limit - 16, initial_indent="  ", subsequent_indent=opt_indent))
 684 
 685         # Generic handler
 686         for k in sorted(doc):
 687             if k in DocCLI.IGNORE or not doc[k]:
 688                 continue
 689             if isinstance(doc[k], string_types):
 690                 text.append('%s: %s' % (k.upper(), textwrap.fill(DocCLI.tty_ify(doc[k]), limit - (len(k) + 2), subsequent_indent=opt_indent)))
 691             elif isinstance(doc[k], (list, tuple)):
 692                 text.append('%s: %s' % (k.upper(), ', '.join(doc[k])))
 693             else:
 694                 # use empty indent since this affects the start of the yaml doc, not it's keys
 695                 text.append(DocCLI._dump_yaml({k.upper(): doc[k]}, ''))
 696             del doc[k]
 697             text.append('')
 698 
 699         if doc.get('plainexamples', False):
 700             text.append("EXAMPLES:")
 701             text.append('')
 702             if isinstance(doc['plainexamples'], string_types):
 703                 text.append(doc.pop('plainexamples').strip())
 704             else:
 705                 text.append(yaml.dump(doc.pop('plainexamples'), indent=2, default_flow_style=False))
 706             text.append('')
 707             text.append('')
 708 
 709         if doc.get('returndocs', False):
 710             text.append("RETURN VALUES:")
 711             DocCLI.add_fields(text, doc.pop('returndocs'), limit, opt_indent, return_values=True)
 712 
 713         return "\n".join(text)
