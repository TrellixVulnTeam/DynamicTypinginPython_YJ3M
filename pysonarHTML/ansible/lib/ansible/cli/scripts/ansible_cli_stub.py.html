  39 # Used for determining if the system is running a new enough python version
  40 # and should only restrict on our documented minimum versions
  41 _PY3_MIN = sys.version_info[:2] >= (3, 5)
  42 _PY2_MIN = (2, 6) <= sys.version_info[:2] < (3,)
  43 _PY_MIN = _PY3_MIN or _PY2_MIN
  44 if not _PY_MIN:
  45     raise SystemExit('ERROR: Ansible requires a minimum of Python2 version 2.6 or Python3 version 3.5. Current version: %s' % ''.join(sys.version.splitlines()))

  50     def display(self, msg, log_only=None):

  61     try:  # bad ANSIBLE_CONFIG or config options can force ugly stacktrace
  62         import ansible.constants as C
  63         from ansible.utils.display import Display, initialize_locale
  64     except AnsibleOptionsError as e:
  65         display.error(to_text(e), wrap_text=False)

  69 
  70     cli = None
  71     me = os.path.basename(sys.argv[0])
  72 
  73     try:
  74         display = Display()
  75         display.debug("starting run")
  76 
  77         sub = None
  78         target = me.split('-')
  79         if target[-1][0].isdigit():
  80             # Remove any version or python version info as downstreams
  81             # sometimes add that
  82             target = target[:-1]
  83 
  84         if len(target) > 1:
  85             sub = target[1]
  86             myclass = "%sCLI" % sub.capitalize()
  87         elif target[0] == 'ansible':
  88             sub = 'adhoc'
  89             myclass = 'AdHocCLI'
  90         else:
  91             raise AnsibleError("Unknown Ansible alias: %s" % me)
  92 
  93         try:
  94             mycli = getattr(__import__("ansible.cli.%s" % sub, fromlist=[myclass]), myclass)
  95         except ImportError as e:
  96             # ImportError members have changed in py3
  97             if 'msg' in dir(e):
  98                 msg = e.msg
  99             else:
 100                 msg = e.message
 101             if msg.endswith(' %s' % sub):
 102                 raise AnsibleError("Ansible sub-program not implemented: %s" % me)
 103             else:
 104                 raise
 105 
 106         b_ansible_dir = os.path.expanduser(os.path.expandvars(b"~/.ansible"))
 107         try:
 108             os.mkdir(b_ansible_dir, 0o700)
 109         except OSError as exc:
 110             if exc.errno != errno.EEXIST:
 111                 display.warning("Failed to create the directory '%s': %s"
 112                                 % (to_text(b_ansible_dir, errors='surrogate_or_replace'),
 113                                    to_text(exc, errors='surrogate_or_replace')))
 114         else:
 115             display.debug("Created the '%s' directory" % to_text(b_ansible_dir, errors='surrogate_or_replace'))
 116 
 117         try:
 118             args = [to_text(a, errors='surrogate_or_strict') for a in sys.argv]
 119         except UnicodeError:
 120             display.error('Command line args are not in utf-8, unable to continue.  Ansible currently only understands utf-8')
 121             display.display(u"The full traceback was:\n\n%s" % to_text(traceback.format_exc()))
 122             exit_code = 6
 123         else:
 124             cli = mycli(args)
 125             exit_code = cli.run()
 126 
 127     except AnsibleOptionsError as e:
 128         cli.parser.print_help()
 129         display.error(to_text(e), wrap_text=False)
 130         exit_code = 5
 131     except AnsibleParserError as e:
 132         display.error(to_text(e), wrap_text=False)
 133         exit_code = 4
 134 # TQM takes care of these, but leaving comment to reserve the exit codes
 135 #    except AnsibleHostUnreachable as e:
 136 #        display.error(str(e))
 137 #        exit_code = 3
 138 #    except AnsibleHostFailed as e:
 139 #        display.error(str(e))
 140 #        exit_code = 2
 141     except AnsibleError as e:
 142         display.error(to_text(e), wrap_text=False)
 143         exit_code = 1
 144     except KeyboardInterrupt:
 145         display.error("User interrupted execution")
 146         exit_code = 99
 147     except Exception as e:
 148         if C.DEFAULT_DEBUG:
 149             # Show raw stacktraces in debug mode, It also allow pdb to
 150             # enter post mortem mode.
 151             raise
 152         have_cli_options = bool(context.CLIARGS)
 153         display.error("Unexpected Exception, this is probably a bug: %s" % to_text(e), wrap_text=False)
 154         if not have_cli_options or have_cli_options and context.CLIARGS['verbosity'] > 2:
 155             log_only = False
 156             if hasattr(e, 'orig_exc'):
 157                 display.vvv('\nexception type: %s' % to_text(type(e.orig_exc)))
 158                 why = to_text(e.orig_exc)
 159                 if to_text(e) != why:
 160                     display.vvv('\noriginal msg: %s' % why)
 161         else:
 162             display.display("to see the full traceback, use -vvv")
 163             log_only = True
 164         display.display(u"the full traceback was:\n\n%s" % to_text(traceback.format_exc()), log_only=log_only)
 165         exit_code = 250
 166 
 167     sys.exit(exit_code)
