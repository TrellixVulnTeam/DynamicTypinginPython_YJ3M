  37 def read_stream(byte_stream):
  38     size = int(byte_stream.readline().strip())
  39 
  40     data = byte_stream.read(size)
  41     if len(data) < size:
  42         raise Exception("EOF found before data was complete")
  43 
  44     data_hash = to_text(byte_stream.readline().strip())
  45     if data_hash != hashlib.sha1(data).hexdigest():
  46         raise Exception("Read {0} bytes, but data did not match checksum".format(size))
  47 
  48     # restore escaped loose \r characters
  49     data = data.replace(br'\r', b'\r')
  50 
  51     return data

  54 @contextmanager
  55 def file_lock(lock_path):
  56     """
  57     Uses contextmanager to create and release a file lock based on the
  58     given path. This allows us to create locks using `with file_lock()`
  59     to prevent deadlocks related to failure to unlock properly.
  60     """
  61 
  62     lock_fd = os.open(lock_path, os.O_RDWR | os.O_CREAT, 0o600)
  63     fcntl.lockf(lock_fd, fcntl.LOCK_EX)
  64     yield
  65     fcntl.lockf(lock_fd, fcntl.LOCK_UN)
  66     os.close(lock_fd)

  69 class ConnectionProcess(object):
  70     '''
  71     The connection process wraps around a Connection object that manages
  72     the connection to a remote device that persists over the playbook
  73     '''
  74     def __init__(self, fd, play_context, socket_path, original_path, task_uuid=None, ansible_playbook_pid=None):
  75         self.play_context = play_context
  76         self.socket_path = socket_path
  77         self.original_path = original_path
  78         self._task_uuid = task_uuid
  79 
  80         self.fd = fd
  81         self.exception = None
  82 
  83         self.srv = JsonRpcServer()
  84         self.sock = None
  85 
  86         self.connection = None
  87         self._ansible_playbook_pid = ansible_playbook_pid
  88 
  89     def start(self, variables):
  90         try:
  91             messages = list()
  92             result = {}
  93 
  94             messages.append(('vvvv', 'control socket path is %s' % self.socket_path))
  95 
  96             # If this is a relative path (~ gets expanded later) then plug the
  97             # key's path on to the directory we originally came from, so we can
  98             # find it now that our cwd is /
  99             if self.play_context.private_key_file and self.play_context.private_key_file[0] not in '~/':
 100                 self.play_context.private_key_file = os.path.join(self.original_path, self.play_context.private_key_file)
 101             self.connection = connection_loader.get(self.play_context.connection, self.play_context, '/dev/null',
 102                                                     task_uuid=self._task_uuid, ansible_playbook_pid=self._ansible_playbook_pid)
 103             self.connection.set_options(var_options=variables)
 104 
 105             self.connection._socket_path = self.socket_path
 106             self.srv.register(self.connection)
 107             messages.extend([('vvvv', msg) for msg in sys.stdout.getvalue().splitlines()])
 108 
 109             self.sock = socket.socket(socket.AF_UNIX, socket.SOCK_STREAM)
 110             self.sock.bind(self.socket_path)
 111             self.sock.listen(1)
 112             messages.append(('vvvv', 'local domain socket listeners started successfully'))
 113         except Exception as exc:
 114             messages.extend(self.connection.pop_messages())
 115             result['error'] = to_text(exc)
 116             result['exception'] = traceback.format_exc()
 117         finally:
 118             result['messages'] = messages
 119             self.fd.write(json.dumps(result, cls=AnsibleJSONEncoder))
 120             self.fd.close()
 121 
 122     def run(self):
 123         try:
 124             while not self.connection._conn_closed:
 125                 signal.signal(signal.SIGALRM, self.connect_timeout)
 126                 signal.signal(signal.SIGTERM, self.handler)
 127                 signal.alarm(self.connection.get_option('persistent_connect_timeout'))
 128 
 129                 self.exception = None
 130                 (s, addr) = self.sock.accept()
 131                 signal.alarm(0)
 132                 signal.signal(signal.SIGALRM, self.command_timeout)
 133                 while True:
 134                     data = recv_data(s)
 135                     if not data:
 136                         break
 137                     log_messages = self.connection.get_option('persistent_log_messages')
 138 
 139                     if log_messages:
 140                         display.display("jsonrpc request: %s" % data, log_only=True)
 141 
 142                     request = json.loads(to_text(data, errors='surrogate_or_strict'))
 143                     if request.get('method') == "exec_command" and not self.connection.connected:
 144                         self.connection._connect()
 145 
 146                     signal.alarm(self.connection.get_option('persistent_command_timeout'))
 147 
 148                     resp = self.srv.handle_request(data)
 149                     signal.alarm(0)
 150 
 151                     if log_messages:
 152                         display.display("jsonrpc response: %s" % resp, log_only=True)
 153 
 154                     send_data(s, to_bytes(resp))
 155 
 156                 s.close()
 157 
 158         except Exception as e:
 159             # socket.accept() will raise EINTR if the socket.close() is called
 160             if hasattr(e, 'errno'):
 161                 if e.errno != errno.EINTR:
 162                     self.exception = traceback.format_exc()
 163             else:
 164                 self.exception = traceback.format_exc()
 165 
 166         finally:
 167             # allow time for any exception msg send over socket to receive at other end before shutting down
 168             time.sleep(0.1)
 169 
 170             # when done, close the connection properly and cleanup the socket file so it can be recreated
 171             self.shutdown()
 172 
 173     def connect_timeout(self, signum, frame):
 174         msg = 'persistent connection idle timeout triggered, timeout value is %s secs.\nSee the timeout setting options in the Network Debug and ' \
 175               'Troubleshooting Guide.' % self.connection.get_option('persistent_connect_timeout')
 176         display.display(msg, log_only=True)
 177         raise Exception(msg)
 178 
 179     def command_timeout(self, signum, frame):
 180         msg = 'command timeout triggered, timeout value is %s secs.\nSee the timeout setting options in the Network Debug and Troubleshooting Guide.'\
 181               % self.connection.get_option('persistent_command_timeout')
 182         display.display(msg, log_only=True)
 183         raise Exception(msg)
 184 
 185     def handler(self, signum, frame):
 186         msg = 'signal handler called with signal %s.' % signum
 187         display.display(msg, log_only=True)
 188         raise Exception(msg)
 189 
 190     def shutdown(self):
 191         """ Shuts down the local domain socket
 192         """
 193         lock_path = unfrackpath("%s/.ansible_pc_lock_%s" % os.path.split(self.socket_path))
 194         if os.path.exists(self.socket_path):
 195             try:
 196                 if self.sock:
 197                     self.sock.close()
 198                 if self.connection:
 199                     self.connection.close()
 200                     if self.connection.get_option("persistent_log_messages"):
 201                         for _level, message in self.connection.pop_messages():
 202                             display.display(message, log_only=True)
 203             except Exception:
 204                 pass
 205             finally:
 206                 if os.path.exists(self.socket_path):
 207                     os.remove(self.socket_path)
 208                     setattr(self.connection, '_socket_path', None)
 209                     setattr(self.connection, '_connected', False)
 210 
 211         if os.path.exists(lock_path):
 212             os.remove(lock_path)
 213 
 214         display.display('shutdown complete', log_only=True)

 217 def main():
 218     """ Called to initiate the connect to the remote device
 219     """
 220     rc = 0
 221     result = {}
 222     messages = list()
 223     socket_path = None
 224 
 225     # Need stdin as a byte stream
 226     if PY3:
 227         stdin = sys.stdin.buffer
 228     else:
 229         stdin = sys.stdin
 230 
 231     # Note: update the below log capture code after Display.display() is refactored.
 232     saved_stdout = sys.stdout
 233     sys.stdout = StringIO()
 234 
 235     try:
 236         # read the play context data via stdin, which means depickling it
 237         vars_data = read_stream(stdin)
 238         init_data = read_stream(stdin)
 239 
 240         if PY3:
 241             pc_data = cPickle.loads(init_data, encoding='bytes')
 242             variables = cPickle.loads(vars_data, encoding='bytes')
 243         else:
 244             pc_data = cPickle.loads(init_data)
 245             variables = cPickle.loads(vars_data)
 246 
 247         play_context = PlayContext()
 248         play_context.deserialize(pc_data)
 249         display.verbosity = play_context.verbosity
 250 
 251     except Exception as e:
 252         rc = 1
 253         result.update({
 254             'error': to_text(e),
 255             'exception': traceback.format_exc()
 256         })
 257 
 258     if rc == 0:
 259         ssh = connection_loader.get('ssh', class_only=True)
 260         ansible_playbook_pid = sys.argv[1]
 261         task_uuid = sys.argv[2]
 262         cp = ssh._create_control_path(play_context.remote_addr, play_context.port, play_context.remote_user, play_context.connection, ansible_playbook_pid)
 263         # create the persistent connection dir if need be and create the paths
 264         # which we will be using later
 265         tmp_path = unfrackpath(C.PERSISTENT_CONTROL_PATH_DIR)
 266         makedirs_safe(tmp_path)
 267 
 268         socket_path = unfrackpath(cp % dict(directory=tmp_path))
 269         lock_path = unfrackpath("%s/.ansible_pc_lock_%s" % os.path.split(socket_path))
 270 
 271         with file_lock(lock_path):
 272             if not os.path.exists(socket_path):
 273                 messages.append(('vvvv', 'local domain socket does not exist, starting it'))
 274                 original_path = os.getcwd()
 275                 r, w = os.pipe()
 276                 pid = fork_process()
 277 
 278                 if pid == 0:
 279                     try:
 280                         os.close(r)
 281                         wfd = os.fdopen(w, 'w')
 282                         process = ConnectionProcess(wfd, play_context, socket_path, original_path, task_uuid, ansible_playbook_pid)
 283                         process.start(variables)
 284                     except Exception:
 285                         messages.append(('error', traceback.format_exc()))
 286                         rc = 1
 287 
 288                     if rc == 0:
 289                         process.run()
 290                     else:
 291                         process.shutdown()
 292 
 293                     sys.exit(rc)
 294 
 295                 else:
 296                     os.close(w)
 297                     rfd = os.fdopen(r, 'r')
 298                     data = json.loads(rfd.read(), cls=AnsibleJSONDecoder)
 299                     messages.extend(data.pop('messages'))
 300                     result.update(data)
 301 
 302             else:
 303                 messages.append(('vvvv', 'found existing local domain socket, using it!'))
 304                 conn = Connection(socket_path)
 305                 conn.set_options(var_options=variables)
 306                 pc_data = to_text(init_data)
 307                 try:
 308                     conn.update_play_context(pc_data)
 309                     conn.set_check_prompt(task_uuid)
 310                 except Exception as exc:
 311                     # Only network_cli has update_play context and set_check_prompt, so missing this is
 312                     # not fatal e.g. netconf
 313                     if isinstance(exc, ConnectionError) and getattr(exc, 'code', None) == -32601:
 314                         pass
 315                     else:
 316                         result.update({
 317                             'error': to_text(exc),
 318                             'exception': traceback.format_exc()
 319                         })
 320 
 321     if os.path.exists(socket_path):
 322         messages.extend(Connection(socket_path).pop_messages())
 323     messages.append(('vvvv', sys.stdout.getvalue()))
 324     result.update({
 325         'messages': messages,
 326         'socket_path': socket_path
 327     })
 328 
 329     sys.stdout = saved_stdout
 330     if 'exception' in result:
 331         rc = 1
 332         sys.stderr.write(json.dumps(result, cls=AnsibleJSONEncoder))
 333     else:
 334         rc = 0
 335         sys.stdout.write(json.dumps(result, cls=AnsibleJSONEncoder))
 336 
 337     sys.exit(rc)
 338 
 339 
 340 if __name__ == '__main__':
 341     display = Display()
 342     main()
