  81     @abstractmethod
  82     def run(self):
  83         """Run the ansible command
  84 
  85         Subclasses must implement this method.  It does the actual work of
  86         running an Ansible command.
  87         """
  88         self.parse()
  89 
  90         display.vv(to_text(opt_help.version(self.parser.prog)))
  91 
  92         if C.CONFIG_FILE:
  93             display.v(u"Using %s as config file" % to_text(C.CONFIG_FILE))
  94         else:
  95             display.v(u"No config file found; using defaults")
  96 
  97         # warn about deprecated config options
  98         for deprecated in C.config.DEPRECATED:
  99             name = deprecated[0]
 100             why = deprecated[1]['why']
 101             if 'alternatives' in deprecated[1]:
 102                 alt = ', use %s instead' % deprecated[1]['alternatives']
 103             else:
 104                 alt = ''
 105             ver = deprecated[1].get('version')
 106             date = deprecated[1].get('date')
 107             collection_name = deprecated[1].get('collection_name')
 108             display.deprecated("%s option, %s %s" % (name, why, alt),
 109                                version=ver, date=date, collection_name=collection_name)

 111     @staticmethod
 112     def split_vault_id(vault_id):
 113         # return (before_@, after_@)
 114         # if no @, return whole string as after_
 115         if '@' not in vault_id:
 116             return (None, vault_id)
 117 
 118         parts = vault_id.split('@', 1)
 119         ret = tuple(parts)
 120         return ret

 122     @staticmethod
 123     def build_vault_ids(vault_ids, vault_password_files=None,
 124                         ask_vault_pass=None, create_new_password=None,
 125                         auto_prompt=True):
 126         vault_password_files = vault_password_files or []
 127         vault_ids = vault_ids or []
 128 
 129         # convert vault_password_files into vault_ids slugs
 130         for password_file in vault_password_files:
 131             id_slug = u'%s@%s' % (C.DEFAULT_VAULT_IDENTITY, password_file)
 132 
 133             # note this makes --vault-id higher precedence than --vault-password-file
 134             # if we want to intertwingle them in order probably need a cli callback to populate vault_ids
 135             # used by --vault-id and --vault-password-file
 136             vault_ids.append(id_slug)
 137 
 138         # if an action needs an encrypt password (create_new_password=True) and we dont
 139         # have other secrets setup, then automatically add a password prompt as well.
 140         # prompts cant/shouldnt work without a tty, so dont add prompt secrets
 141         if ask_vault_pass or (not vault_ids and auto_prompt):
 142 
 143             id_slug = u'%s@%s' % (C.DEFAULT_VAULT_IDENTITY, u'prompt_ask_vault_pass')
 144             vault_ids.append(id_slug)
 145 
 146         return vault_ids

 148     # TODO: remove the now unused args
 149     @staticmethod
 150     def setup_vault_secrets(loader, vault_ids, vault_password_files=None,
 151                             ask_vault_pass=None, create_new_password=False,
 152                             auto_prompt=True):
 153         # list of tuples
 154         vault_secrets = []
 155 
 156         # Depending on the vault_id value (including how --ask-vault-pass / --vault-password-file create a vault_id)
 157         # we need to show different prompts. This is for compat with older Towers that expect a
 158         # certain vault password prompt format, so 'promp_ask_vault_pass' vault_id gets the old format.
 159         prompt_formats = {}
 160 
 161         # If there are configured default vault identities, they are considered 'first'
 162         # so we prepend them to vault_ids (from cli) here
 163 
 164         vault_password_files = vault_password_files or []
 165         if C.DEFAULT_VAULT_PASSWORD_FILE:
 166             vault_password_files.append(C.DEFAULT_VAULT_PASSWORD_FILE)
 167 
 168         if create_new_password:
 169             prompt_formats['prompt'] = ['New vault password (%(vault_id)s): ',
 170                                         'Confirm new vault password (%(vault_id)s): ']
 171             # 2.3 format prompts for --ask-vault-pass
 172             prompt_formats['prompt_ask_vault_pass'] = ['New Vault password: ',
 173                                                        'Confirm New Vault password: ']
 174         else:
 175             prompt_formats['prompt'] = ['Vault password (%(vault_id)s): ']
 176             # The format when we use just --ask-vault-pass needs to match 'Vault password:\s*?$'
 177             prompt_formats['prompt_ask_vault_pass'] = ['Vault password: ']
 178 
 179         vault_ids = CLI.build_vault_ids(vault_ids,
 180                                         vault_password_files,
 181                                         ask_vault_pass,
 182                                         create_new_password,
 183                                         auto_prompt=auto_prompt)
 184 
 185         for vault_id_slug in vault_ids:
 186             vault_id_name, vault_id_value = CLI.split_vault_id(vault_id_slug)
 187             if vault_id_value in ['prompt', 'prompt_ask_vault_pass']:
 188 
 189                 # --vault-id some_name@prompt_ask_vault_pass --vault-id other_name@prompt_ask_vault_pass will be a little
 190                 # confusing since it will use the old format without the vault id in the prompt
 191                 built_vault_id = vault_id_name or C.DEFAULT_VAULT_IDENTITY
 192 
 193                 # choose the prompt based on --vault-id=prompt or --ask-vault-pass. --ask-vault-pass
 194                 # always gets the old format for Tower compatibility.
 195                 # ie, we used --ask-vault-pass, so we need to use the old vault password prompt
 196                 # format since Tower needs to match on that format.
 197                 prompted_vault_secret = PromptVaultSecret(prompt_formats=prompt_formats[vault_id_value],
 198                                                           vault_id=built_vault_id)
 199 
 200                 # a empty or invalid password from the prompt will warn and continue to the next
 201                 # without erroring globally
 202                 try:
 203                     prompted_vault_secret.load()
 204                 except AnsibleError as exc:
 205                     display.warning('Error in vault password prompt (%s): %s' % (vault_id_name, exc))
 206                     raise
 207 
 208                 vault_secrets.append((built_vault_id, prompted_vault_secret))
 209 
 210                 # update loader with new secrets incrementally, so we can load a vault password
 211                 # that is encrypted with a vault secret provided earlier
 212                 loader.set_vault_secrets(vault_secrets)
 213                 continue
 214 
 215             # assuming anything else is a password file
 216             display.vvvvv('Reading vault password file: %s' % vault_id_value)
 217             # read vault_pass from a file
 218             file_vault_secret = get_file_vault_secret(filename=vault_id_value,
 219                                                       vault_id=vault_id_name,
 220                                                       loader=loader)
 221 
 222             # an invalid password file will error globally
 223             try:
 224                 file_vault_secret.load()
 225             except AnsibleError as exc:
 226                 display.warning('Error in vault password file loading (%s): %s' % (vault_id_name, to_text(exc)))
 227                 raise
 228 
 229             if vault_id_name:
 230                 vault_secrets.append((vault_id_name, file_vault_secret))
 231             else:
 232                 vault_secrets.append((C.DEFAULT_VAULT_IDENTITY, file_vault_secret))
 233 
 234             # update loader with as-yet-known vault secrets
 235             loader.set_vault_secrets(vault_secrets)
 236 
 237         return vault_secrets

 239     @staticmethod
 240     def ask_passwords():
 241         ''' prompt for connection and become passwords if needed '''
 242 
 243         op = context.CLIARGS
 244         sshpass = None
 245         becomepass = None
 246         become_prompt = ''
 247 
 248         become_prompt_method = "BECOME" if C.AGNOSTIC_BECOME_PROMPT else op['become_method'].upper()
 249 
 250         try:
 251             if op['ask_pass']:
 252                 sshpass = getpass.getpass(prompt="SSH password: ")
 253                 become_prompt = "%s password[defaults to SSH password]: " % become_prompt_method
 254             else:
 255                 become_prompt = "%s password: " % become_prompt_method
 256 
 257             if op['become_ask_pass']:
 258                 becomepass = getpass.getpass(prompt=become_prompt)
 259                 if op['ask_pass'] and becomepass == '':
 260                     becomepass = sshpass
 261         except EOFError:
 262             pass
 263 
 264         # we 'wrap' the passwords to prevent templating as
 265         # they can contain special chars and trigger it incorrectly
 266         if sshpass:
 267             sshpass = to_unsafe_text(sshpass)
 268         if becomepass:
 269             becomepass = to_unsafe_text(becomepass)
 270 
 271         return (sshpass, becomepass)

 273     def validate_conflicts(self, op, runas_opts=False, fork_opts=False):
 274         ''' check for conflicting options '''
 275 
 276         if fork_opts:
 277             if op.forks < 1:
 278                 self.parser.error("The number of processes (--forks) must be >= 1")
 279 
 280         return op

 282     @abstractmethod
 283     def init_parser(self, usage="", desc=None, epilog=None):
 284         """
 285         Create an options parser for most ansible scripts
 286 
 287         Subclasses need to implement this method.  They will usually call the base class's
 288         init_parser to create a basic version and then add their own options on top of that.
 289 
 290         An implementation will look something like this::
 291 
 292             def init_parser(self):
 293                 super(MyCLI, self).init_parser(usage="My Ansible CLI", inventory_opts=True)
 294                 ansible.arguments.option_helpers.add_runas_options(self.parser)
 295                 self.parser.add_option('--my-option', dest='my_option', action='store')
 296         """
 297         self.parser = opt_help.create_base_parser(os.path.basename(self.args[0]), usage=usage, desc=desc, epilog=epilog, )

 299     @abstractmethod
 300     def post_process_args(self, options):
 301         """Process the command line args
 302 
 303         Subclasses need to implement this method.  This method validates and transforms the command
 304         line arguments.  It can be used to check whether conflicting values were given, whether filenames
 305         exist, etc.
 306 
 307         An implementation will look something like this::
 308 
 309             def post_process_args(self, options):
 310                 options = super(MyCLI, self).post_process_args(options)
 311                 if options.addition and options.subtraction:
 312                     raise AnsibleOptionsError('Only one of --addition and --subtraction can be specified')
 313                 if isinstance(options.listofhosts, string_types):
 314                     options.listofhosts = string_types.split(',')
 315                 return options
 316         """
 317 
 318         # process tags
 319         if hasattr(options, 'tags') and not options.tags:
 320             # optparse defaults does not do what's expected
 321             options.tags = ['all']
 322         if hasattr(options, 'tags') and options.tags:
 323             tags = set()
 324             for tag_set in options.tags:
 325                 for tag in tag_set.split(u','):
 326                     tags.add(tag.strip())
 327             options.tags = list(tags)
 328 
 329         # process skip_tags
 330         if hasattr(options, 'skip_tags') and options.skip_tags:
 331             skip_tags = set()
 332             for tag_set in options.skip_tags:
 333                 for tag in tag_set.split(u','):
 334                     skip_tags.add(tag.strip())
 335             options.skip_tags = list(skip_tags)
 336 
 337         # process inventory options except for CLIs that require their own processing
 338         if hasattr(options, 'inventory') and not self.SKIP_INVENTORY_DEFAULTS:
 339 
 340             if options.inventory:
 341 
 342                 # should always be list
 343                 if isinstance(options.inventory, string_types):
 344                     options.inventory = [options.inventory]
 345 
 346                 # Ensure full paths when needed
 347                 options.inventory = [unfrackpath(opt, follow=False) if ',' not in opt else opt for opt in options.inventory]
 348             else:
 349                 options.inventory = C.DEFAULT_HOST_LIST
 350 
 351         # Dup args set on the root parser and sub parsers results in the root parser ignoring the args. e.g. doing
 352         # 'ansible-galaxy -vvv init' has no verbosity set but 'ansible-galaxy init -vvv' sets a level of 3. To preserve
 353         # back compat with pre-argparse changes we manually scan and set verbosity based on the argv values.
 354         if self.parser.prog in ['ansible-galaxy', 'ansible-vault'] and not options.verbosity:
 355             verbosity_arg = next(iter([arg for arg in self.args if arg.startswith('-v')]), None)
 356             if verbosity_arg:
 357                 display.deprecated("Setting verbosity before the arg sub command is deprecated, set the verbosity "
 358                                    "after the sub command", "2.13", collection_name='ansible.builtin')
 359                 options.verbosity = verbosity_arg.count('v')
 360 
 361         return options
 362 
 363     def parse(self):
 364         """Parse the command line args
 365 
 366         This method parses the command line arguments.  It uses the parser
 367         stored in the self.parser attribute and saves the args and options in
 368         context.CLIARGS.
 369 
 370         Subclasses need to implement two helper methods, init_parser() and post_process_args() which
 371         are called from this function before and after parsing the arguments.
 372         """
 373         self.init_parser()
 374 
 375         if HAS_ARGCOMPLETE:
 376             argcomplete.autocomplete(self.parser)
 377 
 378         try:
 379             options = self.parser.parse_args(self.args[1:])
 380         except SystemExit as e:
 381             if(e.code != 0):
 382                 self.parser.exit(status=2, message=" \n%s " % self.parser.format_help())
 383             raise
 384         options = self.post_process_args(options)
 385         context._init_global_context(options)

 387     @staticmethod
 388     def version_info(gitinfo=False):
 389         ''' return full ansible version info '''
 390         if gitinfo:
 391             # expensive call, user with care
 392             ansible_version_string = opt_help.version()
 393         else:
 394             ansible_version_string = __version__
 395         ansible_version = ansible_version_string.split()[0]
 396         ansible_versions = ansible_version.split('.')
 397         for counter in range(len(ansible_versions)):
 398             if ansible_versions[counter] == "":
 399                 ansible_versions[counter] = 0
 400             try:
 401                 ansible_versions[counter] = int(ansible_versions[counter])
 402             except Exception:
 403                 pass
 404         if len(ansible_versions) < 3:
 405             for counter in range(len(ansible_versions), 3):
 406                 ansible_versions.append(0)
 407         return {'string': ansible_version_string.strip(),
 408                 'full': ansible_version,
 409                 'major': ansible_versions[0],
 410                 'minor': ansible_versions[1],
 411                 'revision': ansible_versions[2]}

 413     @staticmethod
 414     def pager(text):
 415         ''' find reasonable way to display text '''
 416         # this is a much simpler form of what is in pydoc.py
 417         if not sys.stdout.isatty():
 418             display.display(text, screen_only=True)
 419         elif 'PAGER' in os.environ:
 420             if sys.platform == 'win32':
 421                 display.display(text, screen_only=True)
 422             else:
 423                 CLI.pager_pipe(text, os.environ['PAGER'])
 424         else:
 425             p = subprocess.Popen('less --version', shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
 426             p.communicate()
 427             if p.returncode == 0:
 428                 CLI.pager_pipe(text, 'less')
 429             else:
 430                 display.display(text, screen_only=True)

 432     @staticmethod
 433     def pager_pipe(text, cmd):
 434         ''' pipe text through a pager '''
 435         if 'LESS' not in os.environ:
 436             os.environ['LESS'] = CLI.LESS_OPTS
 437         try:
 438             cmd = subprocess.Popen(cmd, shell=True, stdin=subprocess.PIPE, stdout=sys.stdout)
 439             cmd.communicate(input=to_bytes(text))
 440         except IOError:
 441             pass
 442         except KeyboardInterrupt:
 443             pass

 445     @classmethod
 446     def tty_ify(cls, text):
 447 
 448         t = cls._ITALIC.sub("`" + r"\1" + "'", text)    # I(word) => `word'
 449         t = cls._BOLD.sub("*" + r"\1" + "*", t)         # B(word) => *word*
 450         t = cls._MODULE.sub("[" + r"\1" + "]", t)       # M(word) => [word]
 451         t = cls._URL.sub(r"\1", t)                      # U(word) => word
 452         t = cls._CONST.sub("`" + r"\1" + "'", t)        # C(word) => `word'
 453 
 454         return t

 456     @staticmethod
 457     def _play_prereqs():
 458         options = context.CLIARGS
 459 
 460         # all needs loader
 461         loader = DataLoader()
 462 
 463         basedir = options.get('basedir', False)
 464         if basedir:
 465             loader.set_basedir(basedir)
 466             add_all_plugin_dirs(basedir)
 467             AnsibleCollectionConfig.playbook_paths = basedir
 468             default_collection = _get_collection_name_from_path(basedir)
 469             if default_collection:
 470                 display.warning(u'running with default collection {0}'.format(default_collection))
 471                 AnsibleCollectionConfig.default_collection = default_collection
 472 
 473         vault_ids = list(options['vault_ids'])
 474         default_vault_ids = C.DEFAULT_VAULT_IDENTITY_LIST
 475         vault_ids = default_vault_ids + vault_ids
 476 
 477         vault_secrets = CLI.setup_vault_secrets(loader,
 478                                                 vault_ids=vault_ids,
 479                                                 vault_password_files=list(options['vault_password_files']),
 480                                                 ask_vault_pass=options['ask_vault_pass'],
 481                                                 auto_prompt=False)
 482         loader.set_vault_secrets(vault_secrets)
 483 
 484         # create the inventory, and filter it based on the subset specified (if any)
 485         inventory = InventoryManager(loader=loader, sources=options['inventory'])
 486 
 487         # create the variable manager, which will be shared throughout
 488         # the code, ensuring a consistent view of global variables
 489         variable_manager = VariableManager(loader=loader, inventory=inventory, version_info=CLI.version_info(gitinfo=False))
 490 
 491         return loader, inventory, variable_manager

 493     @staticmethod
 494     def get_host_list(inventory, subset, pattern='all'):
 495 
 496         no_hosts = False
 497         if len(inventory.list_hosts()) == 0:
 498             # Empty inventory
 499             if C.LOCALHOST_WARNING and pattern not in C.LOCALHOST:
 500                 display.warning("provided hosts list is empty, only localhost is available. Note that the implicit localhost does not match 'all'")
 501             no_hosts = True
 502 
 503         inventory.subset(subset)
 504 
 505         hosts = inventory.list_hosts(pattern)
 506         if not hosts and no_hosts is False:
 507             raise AnsibleError("Specified hosts and/or --limit does not match any hosts")
 508 
 509         return hosts
