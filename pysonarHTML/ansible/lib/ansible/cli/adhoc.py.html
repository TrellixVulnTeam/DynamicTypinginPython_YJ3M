  23 class AdHocCLI(CLI):
  24     ''' is an extra-simple tool/framework/API for doing 'remote things'.
  25         this command allows you to define and run a single task 'playbook' against a set of hosts
  26     '''
  27 
  28     def init_parser(self):
  29         ''' create an options parser for bin/ansible '''
  30         super(AdHocCLI, self).init_parser(usage='%prog <host-pattern> [options]',
  31                                           desc="Define and run a single task 'playbook' against"
  32                                           " a set of hosts",
  33                                           epilog="Some modules do not make sense in Ad-Hoc (include,"
  34                                           " meta, etc)")
  35 
  36         opt_help.add_runas_options(self.parser)
  37         opt_help.add_inventory_options(self.parser)
  38         opt_help.add_async_options(self.parser)
  39         opt_help.add_output_options(self.parser)
  40         opt_help.add_connect_options(self.parser)
  41         opt_help.add_check_options(self.parser)
  42         opt_help.add_runtask_options(self.parser)
  43         opt_help.add_vault_options(self.parser)
  44         opt_help.add_fork_options(self.parser)
  45         opt_help.add_module_options(self.parser)
  46         opt_help.add_basedir_options(self.parser)
  47 
  48         # options unique to ansible ad-hoc
  49         self.parser.add_argument('-a', '--args', dest='module_args',
  50                                  help="module arguments", default=C.DEFAULT_MODULE_ARGS)
  51         self.parser.add_argument('-m', '--module-name', dest='module_name',
  52                                  help="module name to execute (default=%s)" % C.DEFAULT_MODULE_NAME,
  53                                  default=C.DEFAULT_MODULE_NAME)
  54         self.parser.add_argument('args', metavar='pattern', help='host pattern')
  55 
  56     def post_process_args(self, options):
  57         '''Post process and validate options for bin/ansible '''
  58 
  59         options = super(AdHocCLI, self).post_process_args(options)
  60 
  61         display.verbosity = options.verbosity
  62         self.validate_conflicts(options, runas_opts=True, fork_opts=True)
  63 
  64         return options
  65 
  66     def _play_ds(self, pattern, async_val, poll):
  67         check_raw = context.CLIARGS['module_name'] in C.MODULE_REQUIRE_ARGS
  68 
  69         mytask = {'action': {'module': context.CLIARGS['module_name'], 'args': parse_kv(context.CLIARGS['module_args'], check_raw=check_raw)}}
  70 
  71         # avoid adding to tasks that don't support it, unless set, then give user an error
  72         if context.CLIARGS['module_name'] not in ('include_role', 'include_tasks') and any(frozenset((async_val, poll))):
  73             mytask['async_val'] = async_val
  74             mytask['poll'] = poll
  75 
  76         return dict(
  77             name="Ansible Ad-Hoc",
  78             hosts=pattern,
  79             gather_facts='no',
  80             tasks=[mytask])
  81 
  82     def run(self):
  83         ''' create and execute the single task playbook '''
  84 
  85         super(AdHocCLI, self).run()
  86 
  87         # only thing left should be host pattern
  88         pattern = to_text(context.CLIARGS['args'], errors='surrogate_or_strict')
  89 
  90         sshpass = None
  91         becomepass = None
  92 
  93         (sshpass, becomepass) = self.ask_passwords()
  94         passwords = {'conn_pass': sshpass, 'become_pass': becomepass}
  95 
  96         # get basic objects
  97         loader, inventory, variable_manager = self._play_prereqs()
  98 
  99         try:
 100             hosts = self.get_host_list(inventory, context.CLIARGS['subset'], pattern)
 101         except AnsibleError:
 102             if context.CLIARGS['subset']:
 103                 raise
 104             else:
 105                 hosts = []
 106                 display.warning("No hosts matched, nothing to do")
 107 
 108         if context.CLIARGS['listhosts']:
 109             display.display('  hosts (%d):' % len(hosts))
 110             for host in hosts:
 111                 display.display('    %s' % host)
 112             return 0
 113 
 114         if context.CLIARGS['module_name'] in C.MODULE_REQUIRE_ARGS and not context.CLIARGS['module_args']:
 115             err = "No argument passed to %s module" % context.CLIARGS['module_name']
 116             if pattern.endswith(".yml"):
 117                 err = err + ' (did you mean to run ansible-playbook?)'
 118             raise AnsibleOptionsError(err)
 119 
 120         # Avoid modules that don't work with ad-hoc
 121         if context.CLIARGS['module_name'] in ('import_playbook',):
 122             raise AnsibleOptionsError("'%s' is not a valid action for ad-hoc commands"
 123                                       % context.CLIARGS['module_name'])
 124 
 125         play_ds = self._play_ds(pattern, context.CLIARGS['seconds'], context.CLIARGS['poll_interval'])
 126         play = Play().load(play_ds, variable_manager=variable_manager, loader=loader)
 127 
 128         # used in start callback
 129         playbook = Playbook(loader)
 130         playbook._entries.append(play)
 131         playbook._file_name = '__adhoc_playbook__'
 132 
 133         if self.callback:
 134             cb = self.callback
 135         elif context.CLIARGS['one_line']:
 136             cb = 'oneline'
 137         # Respect custom 'stdout_callback' only with enabled 'bin_ansible_callbacks'
 138         elif C.DEFAULT_LOAD_CALLBACK_PLUGINS and C.DEFAULT_STDOUT_CALLBACK != 'default':
 139             cb = C.DEFAULT_STDOUT_CALLBACK
 140         else:
 141             cb = 'minimal'
 142 
 143         run_tree = False
 144         if context.CLIARGS['tree']:
 145             C.DEFAULT_CALLBACK_WHITELIST.append('tree')
 146             C.TREE_DIR = context.CLIARGS['tree']
 147             run_tree = True
 148 
 149         # now create a task queue manager to execute the play
 150         self._tqm = None
 151         try:
 152             self._tqm = TaskQueueManager(
 153                 inventory=inventory,
 154                 variable_manager=variable_manager,
 155                 loader=loader,
 156                 passwords=passwords,
 157                 stdout_callback=cb,
 158                 run_additional_callbacks=C.DEFAULT_LOAD_CALLBACK_PLUGINS,
 159                 run_tree=run_tree,
 160                 forks=context.CLIARGS['forks'],
 161             )
 162 
 163             self._tqm.load_callbacks()
 164             self._tqm.send_callback('v2_playbook_on_start', playbook)
 165 
 166             result = self._tqm.run(play)
 167 
 168             self._tqm.send_callback('v2_playbook_on_stats', self._tqm._stats)
 169         finally:
 170             if self._tqm:
 171                 self._tqm.cleanup()
 172             if loader:
 173                 loader.cleanup_all_tmp_files()
 174 
 175         return result
