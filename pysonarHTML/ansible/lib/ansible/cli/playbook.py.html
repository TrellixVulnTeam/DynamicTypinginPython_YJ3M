  27 class PlaybookCLI(CLI):
  28     ''' the tool to run *Ansible playbooks*, which are a configuration and multinode deployment system.
  29         See the project home page (https://docs.ansible.com) for more information. '''

  31     def init_parser(self):
  32 
  33         # create parser for CLI options
  34         super(PlaybookCLI, self).init_parser(
  35             usage="%prog [options] playbook.yml [playbook2 ...]",
  36             desc="Runs Ansible playbooks, executing the defined tasks on the targeted hosts.")
  37 
  38         opt_help.add_connect_options(self.parser)
  39         opt_help.add_meta_options(self.parser)
  40         opt_help.add_runas_options(self.parser)
  41         opt_help.add_subset_options(self.parser)
  42         opt_help.add_check_options(self.parser)
  43         opt_help.add_inventory_options(self.parser)
  44         opt_help.add_runtask_options(self.parser)
  45         opt_help.add_vault_options(self.parser)
  46         opt_help.add_fork_options(self.parser)
  47         opt_help.add_module_options(self.parser)
  48 
  49         # ansible playbook specific opts
  50         self.parser.add_argument('--list-tasks', dest='listtasks', action='store_true',
  51                                  help="list all tasks that would be executed")
  52         self.parser.add_argument('--list-tags', dest='listtags', action='store_true',
  53                                  help="list all available tags")
  54         self.parser.add_argument('--step', dest='step', action='store_true',
  55                                  help="one-step-at-a-time: confirm each task before running")
  56         self.parser.add_argument('--start-at-task', dest='start_at_task',
  57                                  help="start the playbook at the task matching this name")
  58         self.parser.add_argument('args', help='Playbook(s)', metavar='playbook', nargs='+')
  59 
  60     def post_process_args(self, options):
  61         options = super(PlaybookCLI, self).post_process_args(options)
  62 
  63         display.verbosity = options.verbosity
  64         self.validate_conflicts(options, runas_opts=True, fork_opts=True)
  65 
  66         return options

  68     def run(self):
  69 
  70         super(PlaybookCLI, self).run()
  71 
  72         # Note: slightly wrong, this is written so that implicit localhost
  73         # manages passwords
  74         sshpass = None
  75         becomepass = None
  76         passwords = {}
  77 
  78         # initial error check, to make sure all specified playbooks are accessible
  79         # before we start running anything through the playbook executor
  80 
  81         b_playbook_dirs = []
  82         for playbook in context.CLIARGS['args']:
  83             if not os.path.exists(playbook):
  84                 raise AnsibleError("the playbook: %s could not be found" % playbook)
  85             if not (os.path.isfile(playbook) or stat.S_ISFIFO(os.stat(playbook).st_mode)):
  86                 raise AnsibleError("the playbook: %s does not appear to be a file" % playbook)
  87 
  88             b_playbook_dir = os.path.dirname(os.path.abspath(to_bytes(playbook, errors='surrogate_or_strict')))
  89             # load plugins from all playbooks in case they add callbacks/inventory/etc
  90             add_all_plugin_dirs(b_playbook_dir)
  91 
  92             b_playbook_dirs.append(b_playbook_dir)
  93 
  94         AnsibleCollectionConfig.playbook_paths = b_playbook_dirs
  95 
  96         playbook_collection = _get_collection_name_from_path(b_playbook_dirs[0])
  97 
  98         if playbook_collection:
  99             display.warning("running playbook inside collection {0}".format(playbook_collection))
 100             AnsibleCollectionConfig.default_collection = playbook_collection
 101 
 102         # don't deal with privilege escalation or passwords when we don't need to
 103         if not (context.CLIARGS['listhosts'] or context.CLIARGS['listtasks'] or
 104                 context.CLIARGS['listtags'] or context.CLIARGS['syntax']):
 105             (sshpass, becomepass) = self.ask_passwords()
 106             passwords = {'conn_pass': sshpass, 'become_pass': becomepass}
 107 
 108         # create base objects
 109         loader, inventory, variable_manager = self._play_prereqs()
 110 
 111         # (which is not returned in list_hosts()) is taken into account for
 112         # warning if inventory is empty.  But it can't be taken into account for
 113         # checking if limit doesn't match any hosts.  Instead we don't worry about
 114         # limit if only implicit localhost was in inventory to start with.
 115         #
 116         # Fix this when we rewrite inventory by making localhost a real host (and thus show up in list_hosts())
 117         CLI.get_host_list(inventory, context.CLIARGS['subset'])
 118 
 119         # flush fact cache if requested
 120         if context.CLIARGS['flush_cache']:
 121             self._flush_cache(inventory, variable_manager)
 122 
 123         # create the playbook executor, which manages running the plays via a task queue manager
 124         pbex = PlaybookExecutor(playbooks=context.CLIARGS['args'], inventory=inventory,
 125                                 variable_manager=variable_manager, loader=loader,
 126                                 passwords=passwords)
 127 
 128         results = pbex.run()
 129 
 130         if isinstance(results, list):
 131             for p in results:
 132 
 133                 display.display('\nplaybook: %s' % p['playbook'])
 134                 for idx, play in enumerate(p['plays']):
 135                     if play._included_path is not None:
 136                         loader.set_basedir(play._included_path)
 137                     else:
 138                         pb_dir = os.path.realpath(os.path.dirname(p['playbook']))
 139                         loader.set_basedir(pb_dir)
 140 
 141                     msg = "\n  play #%d (%s): %s" % (idx + 1, ','.join(play.hosts), play.name)
 142                     mytags = set(play.tags)
 143                     msg += '\tTAGS: [%s]' % (','.join(mytags))
 144 
 145                     if context.CLIARGS['listhosts']:
 146                         playhosts = set(inventory.get_hosts(play.hosts))
 147                         msg += "\n    pattern: %s\n    hosts (%d):" % (play.hosts, len(playhosts))
 148                         for host in playhosts:
 149                             msg += "\n      %s" % host
 150 
 151                     display.display(msg)
 152 
 153                     all_tags = set()
 154                     if context.CLIARGS['listtags'] or context.CLIARGS['listtasks']:
 155                         taskmsg = ''
 156                         if context.CLIARGS['listtasks']:
 157                             taskmsg = '    tasks:\n'
 158 
 159                         def _process_block(b):
 160                             taskmsg = ''
 161                             for task in b.block:
 162                                 if isinstance(task, Block):
 163                                     taskmsg += _process_block(task)
 164                                 else:
 165                                     if task.action == 'meta':
 166                                         continue
 167 
 168                                     all_tags.update(task.tags)
 169                                     if context.CLIARGS['listtasks']:
 170                                         cur_tags = list(mytags.union(set(task.tags)))
 171                                         cur_tags.sort()
 172                                         if task.name:
 173                                             taskmsg += "      %s" % task.get_name()
 174                                         else:
 175                                             taskmsg += "      %s" % task.action
 176                                         taskmsg += "\tTAGS: [%s]\n" % ', '.join(cur_tags)
 177 
 178                             return taskmsg
 179 
 180                         all_vars = variable_manager.get_vars(play=play)
 181                         for block in play.compile():
 182                             block = block.filter_tagged_tasks(all_vars)
 183                             if not block.has_tasks():
 184                                 continue
 185                             taskmsg += _process_block(block)
 186 
 187                         if context.CLIARGS['listtags']:
 188                             cur_tags = list(mytags.union(all_tags))
 189                             cur_tags.sort()
 190                             taskmsg += "      TASK TAGS: [%s]\n" % ', '.join(cur_tags)
 191 
 192                         display.display(taskmsg)
 193 
 194             return 0
 195         else:
 196             return results

 198     @staticmethod
 199     def _flush_cache(inventory, variable_manager):
 200         for host in inventory.list_hosts():
 201             hostname = host.get_name()
 202             variable_manager.clear_facts(hostname)
