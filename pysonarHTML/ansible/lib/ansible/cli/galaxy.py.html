  53 def _display_header(path, h1, h2, w1=10, w2=7):

  84 def _get_collection_widths(collections):
  85     if is_iterable(collections):
  86         fqcn_set = set(to_text(c) for c in collections)
  87         version_set = set(to_text(c.latest_version) for c in collections)
  88     else:
  89         fqcn_set = set([to_text(collections)])
  90         version_set = set([collections.latest_version])
  91 
  92     fqcn_length = len(max(fqcn_set, key=len))
  93     version_length = len(max(version_set, key=len))
  94 
  95     return fqcn_length, version_length

 409     def run(self):
 410 
 411         super(GalaxyCLI, self).run()
 412 
 413         self.galaxy = Galaxy()
 414 
 415         def server_config_def(section, key, required):
 416             return {
 417                 'description': 'The %s of the %s Galaxy server' % (key, section),
 418                 'ini': [
 419                     {
 420                         'section': 'galaxy_server.%s' % section,
 421                         'key': key,
 422                     }
 423                 ],
 424                 'env': [
 425                     {'name': 'ANSIBLE_GALAXY_SERVER_%s_%s' % (section.upper(), key.upper())},
 426                 ],
 427                 'required': required,
 428             }
 429         server_def = [('url', True), ('username', False), ('password', False), ('token', False),
 430                       ('auth_url', False), ('v3', False)]
 431 
 432         validate_certs = not context.CLIARGS['ignore_certs']
 433 
 434         config_servers = []
 435 
 436         # Need to filter out empty strings or non truthy values as an empty server list env var is equal to [''].
 437         server_list = [s for s in C.GALAXY_SERVER_LIST or [] if s]
 438         for server_key in server_list:
 439             # Config definitions are looked up dynamically based on the C.GALAXY_SERVER_LIST entry. We look up the
 440             # section [galaxy_server.<server>] for the values url, username, password, and token.
 441             config_dict = dict((k, server_config_def(server_key, k, req)) for k, req in server_def)
 442             defs = AnsibleLoader(yaml.safe_dump(config_dict)).get_single_data()
 443             C.config.initialize_plugin_configuration_definitions('galaxy_server', server_key, defs)
 444 
 445             server_options = C.config.get_plugin_options('galaxy_server', server_key)
 446             # auth_url is used to create the token, but not directly by GalaxyAPI, so
 447             # it doesn't need to be passed as kwarg to GalaxyApi
 448             auth_url = server_options.pop('auth_url', None)
 449             token_val = server_options['token'] or NoTokenSentinel

 522     def _parse_requirements_file(self, requirements_file, allow_old_format=True):
 523         """
 524         Parses an Ansible requirement.yml file and returns all the roles and/or collections defined in it. There are 2
 525         requirements file format:
 526 
 527             # v1 (roles only)
 528             - src: The source of the role, required if include is not set. Can be Galaxy role name, URL to a SCM repo or tarball.
 529               name: Downloads the role to the specified name, defaults to Galaxy name from Galaxy or name of repo if src is a URL.
 530               scm: If src is a URL, specify the SCM. Only git or hd are supported and defaults ot git.
 531               version: The version of the role to download. Can also be tag, commit, or branch name and defaults to master.
 532               include: Path to additional requirements.yml files.
 533 
 534             # v2 (roles and collections)
 535             ---
 536             roles:
 537             # Same as v1 format just under the roles key
 538 
 539             collections:
 540             - namespace.collection
 541             - name: namespace.collection
 542               version: version identifier, multiple identifiers are separated by ','
 543               source: the URL or a predefined source name that relates to C.GALAXY_SERVER_LIST
 544               type: git|file|url|galaxy
 545 
 546         :param requirements_file: The path to the requirements file.
 547         :param allow_old_format: Will fail if a v1 requirements file is found and this is set to False.
 548         :return: a dict containing roles and collections to found in the requirements file.
 549         """
 550         requirements = {
 551             'roles': [],
 552             'collections': [],
 553         }
 554 
 555         b_requirements_file = to_bytes(requirements_file, errors='surrogate_or_strict')
 556         if not os.path.exists(b_requirements_file):
 557             raise AnsibleError("The requirements file '%s' does not exist." % to_native(requirements_file))
 558 
 559         display.vvv("Reading requirement file at '%s'" % requirements_file)
 560         with open(b_requirements_file, 'rb') as req_obj:
 561             try:
 562                 file_requirements = yaml.safe_load(req_obj)
 563             except YAMLError as err:
 564                 raise AnsibleError(
 565                     "Failed to parse the requirements yml at '%s' with the following error:\n%s"
 566                     % (to_native(requirements_file), to_native(err)))
 567 
 568         if file_requirements is None:
 569             raise AnsibleError("No requirements found in file '%s'" % to_native(requirements_file))
 570 
 571         def parse_role_req(requirement):
 572             if "include" not in requirement:
 573                 role = RoleRequirement.role_yaml_parse(requirement)
 574                 display.vvv("found role %s in yaml file" % to_text(role))
 575                 if "name" not in role and "src" not in role:
 576                     raise AnsibleError("Must specify name or src for role")
 577                 return [GalaxyRole(self.galaxy, self.api, **role)]
 578             else:
 579                 b_include_path = to_bytes(requirement["include"], errors="surrogate_or_strict")
 580                 if not os.path.isfile(b_include_path):
 581                     raise AnsibleError("Failed to find include requirements file '%s' in '%s'"
 582                                        % (to_native(b_include_path), to_native(requirements_file)))
 583 
 584                 with open(b_include_path, 'rb') as f_include:
 585                     try:
 586                         return [GalaxyRole(self.galaxy, self.api, **r) for r in
 587                                 (RoleRequirement.role_yaml_parse(i) for i in yaml.safe_load(f_include))]
 588                     except Exception as e:
 589                         raise AnsibleError("Unable to load data from include requirements file: %s %s"
 590                                            % (to_native(requirements_file), to_native(e)))
 591 
 592         if isinstance(file_requirements, list):
 593             # Older format that contains only roles
 594             if not allow_old_format:
 595                 raise AnsibleError("Expecting requirements file to be a dict with the key 'collections' that contains "
 596                                    "a list of collections to install")
 597 
 598             for role_req in file_requirements:
 599                 requirements['roles'] += parse_role_req(role_req)
 600 
 601         else:
 602             # Newer format with a collections and/or roles key
 603             extra_keys = set(file_requirements.keys()).difference(set(['roles', 'collections']))
 604             if extra_keys:
 605                 raise AnsibleError("Expecting only 'roles' and/or 'collections' as base keys in the requirements "
 606                                    "file. Found: %s" % (to_native(", ".join(extra_keys))))
 607 
 608             for role_req in file_requirements.get('roles') or []:
 609                 requirements['roles'] += parse_role_req(role_req)
 610 
 611             for collection_req in file_requirements.get('collections') or []:
 612                 if isinstance(collection_req, dict):
 613                     req_name = collection_req.get('name', None)
 614                     if req_name is None:
 615                         raise AnsibleError("Collections requirement entry should contain the key name.")
 616 
 617                     req_type = collection_req.get('type')
 618                     if req_type not in ('file', 'galaxy', 'git', 'url', None):
 619                         raise AnsibleError("The collection requirement entry key 'type' must be one of file, galaxy, git, or url.")
 620 
 621                     req_version = collection_req.get('version', '*')
 622                     req_source = collection_req.get('source', None)
 623                     if req_source:
 624                         # Try and match up the requirement source with our list of Galaxy API servers defined in the
 625                         # config, otherwise create a server with that URL without any auth.
 626                         req_source = next(iter([a for a in self.api_servers if req_source in [a.name, a.api_server]]),
 627                                           GalaxyAPI(self.galaxy,
 628                                                     "explicit_requirement_%s" % req_name,
 629                                                     req_source,
 630                                                     validate_certs=not context.CLIARGS['ignore_certs']))
 631 
 632                     requirements['collections'].append((req_name, req_version, req_source, req_type))
 633                 else:
 634                     requirements['collections'].append((collection_req, '*', None, None))
 635 
 636         return requirements

 680     def _get_skeleton_galaxy_yml(template_path, inject_data):
 681         with open(to_bytes(template_path, errors='surrogate_or_strict'), 'rb') as template_obj:
 682             meta_template = to_text(template_obj.read(), errors='surrogate_or_strict')
 683

 685 
 686         required_config = []

 690

 692             if not value:
 693                 meta_type = meta_entry.get('type', 'str')
 694 
 695                 if meta_type == 'str':
 696                     value = ''
 697                 elif meta_type == 'list':
 698                     value = []
 699                 elif meta_type == 'dict':
 700                     value = {}
 701 
 702             meta_entry['value'] = value
 703             config_list.append(meta_entry)
 704 
 705         link_pattern = re.compile(r"L\(([^)]+),\s+([^)]+)\)")
 706         const_pattern = re.compile(r"C\(([^)]+)\)")
 707 
 708         def comment_ify(v):
 709             if isinstance(v, list):
 710                 v = ". ".join([l.rstrip('.') for l in v])

 712             v = link_pattern.sub(r"\1 <\2>", v)
 713             v = const_pattern.sub(r"'\1'", v)
 714 
 715             return textwrap.fill(v, width=117, initial_indent="# ", subsequent_indent="# ", break_on_hyphens=False)
 716 
 717         loader = DataLoader()
 718         templar = Templar(loader, variables={'required_config': required_config, 'optional_config': optional_config})
 719         templar.environment.filters['comment_ify'] = comment_ify
 720 
 721         meta_value = templar.template(meta_template)
 722 
 723         return meta_value

 725     def _require_one_of_collections_requirements(self, collections, requirements_file):
 726         if collections and requirements_file:
 727             raise AnsibleError("The positional collection_name arg and --requirements-file are mutually exclusive.")

 734             requirements = {'collections': [], 'roles': []}

 743                     name, dummy, requirement = collection_input.partition(':')
 744                 requirements['collections'].append((name, requirement or '*', None, None))
