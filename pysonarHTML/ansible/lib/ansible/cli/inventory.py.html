  46 class InventoryCLI(CLI):
  47     ''' used to display or dump the configured inventory as Ansible sees it '''

  59     def init_parser(self):
  60         super(InventoryCLI, self).init_parser(
  61             usage='usage: %prog [options] [host|group]',

  64         opt_help.add_inventory_options(self.parser)

  99     def post_process_args(self, options):
 100         options = super(InventoryCLI, self).post_process_args(options)
 101 
 102         display.verbosity = options.verbosity
 103         self.validate_conflicts(options)
 104 
 105         # there can be only one! and, at least, one!
 106         used = 0
 107         for opt in (options.list, options.host, options.graph):
 108             if opt:
 109                 used += 1
 110         if used == 0:
 111             raise AnsibleOptionsError("No action selected, at least one of --host, --graph or --list needs to be specified.")
 112         elif used > 1:
 113             raise AnsibleOptionsError("Conflicting options used, only one of --host, --graph or --list can be used at the same time.")
 114 
 115         # set host pattern to default if not supplied
 116         if options.args:
 117             options.pattern = options.args
 118         else:
 119             options.pattern = 'all'
 120 
 121         return options

 123     def run(self):
 124 
 125         super(InventoryCLI, self).run()
 126 
 127         # Initialize needed objects
 128         self.loader, self.inventory, self.vm = self._play_prereqs()
 129 
 130         results = None
 131         if context.CLIARGS['host']:
 132             hosts = self.inventory.get_hosts(context.CLIARGS['host'])
 133             if len(hosts) != 1:
 134                 raise AnsibleOptionsError("You must pass a single valid host to --host parameter")
 135 
 136             myvars = self._get_host_variables(host=hosts[0])
 137 
 138             # FIXME: should we template first?
 139             results = self.dump(myvars)
 140 
 141         elif context.CLIARGS['graph']:
 142             results = self.inventory_graph()
 143         elif context.CLIARGS['list']:
 144             top = self._get_group('all')
 145             if context.CLIARGS['yaml']:
 146                 results = self.yaml_inventory(top)
 147             elif context.CLIARGS['toml']:
 148                 results = self.toml_inventory(top)
 149             else:
 150                 results = self.json_inventory(top)
 151             results = self.dump(results)
 152 
 153         if results:
 154             outfile = context.CLIARGS['output_file']
 155             if outfile is None:
 156                 # FIXME: pager?
 157                 display.display(results)
 158             else:
 159                 try:
 160                     with open(to_bytes(outfile), 'wt') as f:
 161                         f.write(results)
 162                 except (OSError, IOError) as e:
 163                     raise AnsibleError('Unable to write to destination file (%s): %s' % (to_native(outfile), to_native(e)))
 164             sys.exit(0)
 165 
 166         sys.exit(1)

 168     @staticmethod
 169     def dump(stuff):
 170 
 171         if context.CLIARGS['yaml']:
 172             import yaml
 173             from ansible.parsing.yaml.dumper import AnsibleDumper
 174             results = yaml.dump(stuff, Dumper=AnsibleDumper, default_flow_style=False)
 175         elif context.CLIARGS['toml']:
 176             from ansible.plugins.inventory.toml import toml_dumps, HAS_TOML
 177             if not HAS_TOML:
 178                 raise AnsibleError(
 179                     'The python "toml" library is required when using the TOML output format'
 180                 )
 181             results = toml_dumps(stuff)
 182         else:
 183             import json
 184             from ansible.parsing.ajson import AnsibleJSONEncoder
 185             results = json.dumps(stuff, cls=AnsibleJSONEncoder, sort_keys=True, indent=4, preprocess_unsafe=True)
 186 
 187         return results

 189     def _get_group_variables(self, group):
 190 
 191         # get info from inventory source
 192         res = group.get_vars()
 193 
 194         # Always load vars plugins
 195         res = combine_vars(res, get_vars_from_inventory_sources(self.loader, self.inventory._sources, [group], 'all'))
 196         if context.CLIARGS['basedir']:
 197             res = combine_vars(res, get_vars_from_path(self.loader, context.CLIARGS['basedir'], [group], 'all'))
 198 
 199         if group.priority != 1:
 200             res['ansible_group_priority'] = group.priority
 201 
 202         return self._remove_internal(res)
 203 
 204     def _get_host_variables(self, host):
 205 
 206         if context.CLIARGS['export']:
 207             # only get vars defined directly host
 208             hostvars = host.get_vars()
 209 
 210             # Always load vars plugins
 211             hostvars = combine_vars(hostvars, get_vars_from_inventory_sources(self.loader, self.inventory._sources, [host], 'all'))
 212             if context.CLIARGS['basedir']:
 213                 hostvars = combine_vars(hostvars, get_vars_from_path(self.loader, context.CLIARGS['basedir'], [host], 'all'))
 214         else:
 215             # get all vars flattened by host, but skip magic hostvars
 216             hostvars = self.vm.get_vars(host=host, include_hostvars=False, stage='all')
 217 
 218         return self._remove_internal(hostvars)

 220     def _get_group(self, gname):
 221         group = self.inventory.groups.get(gname)
 222         return group

 224     @staticmethod
 225     def _remove_internal(dump):
 226 
 227         for internal in INTERNAL_VARS:
 228             if internal in dump:
 229                 del dump[internal]
 230 
 231         return dump
 232 
 233     @staticmethod
 234     def _remove_empty(dump):
 235         # remove empty keys
 236         for x in ('hosts', 'vars', 'children'):
 237             if x in dump and not dump[x]:
 238                 del dump[x]
 239 
 240     @staticmethod
 241     def _show_vars(dump, depth):
 242         result = []
 243         for (name, val) in sorted(dump.items()):
 244             result.append(InventoryCLI._graph_name('{%s = %s}' % (name, val), depth))
 245         return result
 246 
 247     @staticmethod
 248     def _graph_name(name, depth=0):
 249         if depth:
 250             name = "  |" * (depth) + "--%s" % name
 251         return name
 252 
 253     def _graph_group(self, group, depth=0):
 254 
 255         result = [self._graph_name('@%s:' % group.name, depth)]
 256         depth = depth + 1
 257         for kid in sorted(group.child_groups, key=attrgetter('name')):
 258             result.extend(self._graph_group(kid, depth))
 259 
 260         if group.name != 'all':
 261             for host in sorted(group.hosts, key=attrgetter('name')):
 262                 result.append(self._graph_name(host.name, depth))
 263                 if context.CLIARGS['show_vars']:
 264                     result.extend(self._show_vars(self._get_host_variables(host), depth + 1))
 265 
 266         if context.CLIARGS['show_vars']:
 267             result.extend(self._show_vars(self._get_group_variables(group), depth))
 268 
 269         return result

 271     def inventory_graph(self):
 272 
 273         start_at = self._get_group(context.CLIARGS['pattern'])
 274         if start_at:
 275             return '\n'.join(self._graph_group(start_at))
 276         else:
 277             raise AnsibleOptionsError("Pattern must be valid group name when using --graph")

 279     def json_inventory(self, top):
 280 
 281         seen = set()
 282 
 283         def format_group(group):
 284             results = {}
 285             results[group.name] = {}
 286             if group.name != 'all':
 287                 results[group.name]['hosts'] = [h.name for h in sorted(group.hosts, key=attrgetter('name'))]
 288             results[group.name]['children'] = []
 289             for subgroup in sorted(group.child_groups, key=attrgetter('name')):
 290                 results[group.name]['children'].append(subgroup.name)
 291                 if subgroup.name not in seen:
 292                     results.update(format_group(subgroup))
 293                     seen.add(subgroup.name)
 294             if context.CLIARGS['export']:
 295                 results[group.name]['vars'] = self._get_group_variables(group)
 296 
 297             self._remove_empty(results[group.name])
 298             if not results[group.name]:
 299                 del results[group.name]
 300 
 301             return results
 302 
 303         results = format_group(top)
 304 
 305         # populate meta
 306         results['_meta'] = {'hostvars': {}}
 307         hosts = self.inventory.get_hosts()
 308         for host in hosts:
 309             hvars = self._get_host_variables(host)
 310             if hvars:
 311                 results['_meta']['hostvars'][host.name] = hvars
 312 
 313         return results

 315     def yaml_inventory(self, top):
 316 
 317         seen = []
 318 
 319         def format_group(group):
 320             results = {}
 321 
 322             # initialize group + vars
 323             results[group.name] = {}
 324 
 325             # subgroups
 326             results[group.name]['children'] = {}
 327             for subgroup in sorted(group.child_groups, key=attrgetter('name')):
 328                 if subgroup.name != 'all':
 329                     results[group.name]['children'].update(format_group(subgroup))
 330 
 331             # hosts for group
 332             results[group.name]['hosts'] = {}
 333             if group.name != 'all':
 334                 for h in sorted(group.hosts, key=attrgetter('name')):
 335                     myvars = {}
 336                     if h.name not in seen:  # avoid defining host vars more than once
 337                         seen.append(h.name)
 338                         myvars = self._get_host_variables(host=h)
 339                     results[group.name]['hosts'][h.name] = myvars
 340 
 341             if context.CLIARGS['export']:
 342                 gvars = self._get_group_variables(group)
 343                 if gvars:
 344                     results[group.name]['vars'] = gvars
 345 
 346             self._remove_empty(results[group.name])
 347 
 348             return results
 349 
 350         return format_group(top)

 352     def toml_inventory(self, top):
 353         seen = set()
 354         has_ungrouped = bool(next(g.hosts for g in top.child_groups if g.name == 'ungrouped'))
 355 
 356         def format_group(group):
 357             results = {}
 358             results[group.name] = {}
 359 
 360             results[group.name]['children'] = []
 361             for subgroup in sorted(group.child_groups, key=attrgetter('name')):
 362                 if subgroup.name == 'ungrouped' and not has_ungrouped:
 363                     continue
 364                 if group.name != 'all':
 365                     results[group.name]['children'].append(subgroup.name)
 366                 results.update(format_group(subgroup))
 367 
 368             if group.name != 'all':
 369                 for host in sorted(group.hosts, key=attrgetter('name')):
 370                     if host.name not in seen:
 371                         seen.add(host.name)
 372                         host_vars = self._get_host_variables(host=host)
 373                     else:
 374                         host_vars = {}
 375                     try:
 376                         results[group.name]['hosts'][host.name] = host_vars
 377                     except KeyError:
 378                         results[group.name]['hosts'] = {host.name: host_vars}
 379 
 380             if context.CLIARGS['export']:
 381                 results[group.name]['vars'] = self._get_group_variables(group)
 382 
 383             self._remove_empty(results[group.name])
 384             if not results[group.name]:
 385                 del results[group.name]
 386 
 387             return results
 388 
 389         results = format_group(top)
 390 
 391         return results
