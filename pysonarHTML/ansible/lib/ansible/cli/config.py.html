  76     def run(self):
  77 
  78         super(ConfigCLI, self).run()
  79 
  80         if context.CLIARGS['config_file']:
  81             self.config_file = unfrackpath(context.CLIARGS['config_file'], follow=False)
  82             b_config = to_bytes(self.config_file)
  83             if os.path.exists(b_config) and os.access(b_config, os.R_OK):
  84                 self.config = ConfigManager(self.config_file)
  85             else:
  86                 raise AnsibleOptionsError('The provided configuration file is missing or not accessible: %s' % to_native(self.config_file))
  87         else:
  88             self.config = ConfigManager()
  89             self.config_file = find_ini_config_file()
  90 
  91         if self.config_file:
  92             try:
  93                 if not os.path.exists(self.config_file):
  94                     raise AnsibleOptionsError("%s does not exist or is not accessible" % (self.config_file))
  95                 elif not os.path.isfile(self.config_file):
  96                     raise AnsibleOptionsError("%s is not a valid file" % (self.config_file))
  97 
  98                 os.environ['ANSIBLE_CONFIG'] = to_native(self.config_file)
  99             except Exception:
 100                 if context.CLIARGS['action'] in ['view']:
 101                     raise
 102                 elif context.CLIARGS['action'] in ['edit', 'update']:
 103                     display.warning("File does not exist, used empty file: %s" % self.config_file)
 104 
 105         elif context.CLIARGS['action'] == 'view':
 106             raise AnsibleError('Invalid or no config file was supplied')
 107 
 108         context.CLIARGS['func']()

 110     def execute_update(self):
 111         '''
 112         Updates a single setting in the specified ansible.cfg
 113         '''
 114         raise AnsibleError("Option not implemented yet")
 115 
 116         # pylint: disable=unreachable
 117         if context.CLIARGS['setting'] is None:
 118             raise AnsibleOptionsError("update option requires a setting to update")
 119 
 120         (entry, value) = context.CLIARGS['setting'].split('=')
 121         if '.' in entry:
 122             (section, option) = entry.split('.')
 123         else:
 124             section = 'defaults'
 125             option = entry
 126         subprocess.call([
 127             'ansible',
 128             '-m', 'ini_file',
 129             'localhost',
 130             '-c', 'local',
 131             '-a', '"dest=%s section=%s option=%s value=%s backup=yes"' % (self.config_file, section, option, value)
 132         ])
 133 
 134     def execute_view(self):
 135         '''
 136         Displays the current config file
 137         '''
 138         try:
 139             with open(self.config_file, 'rb') as f:
 140                 self.pager(to_text(f.read(), errors='surrogate_or_strict'))
 141         except Exception as e:
 142             raise AnsibleError("Failed to open config file: %s" % to_native(e))
 143 
 144     def execute_edit(self):
 145         '''
 146         Opens ansible.cfg in the default EDITOR
 147         '''
 148         raise AnsibleError("Option not implemented yet")
 149 
 150         # pylint: disable=unreachable
 151         try:
 152             editor = shlex.split(os.environ.get('EDITOR', 'vi'))
 153             editor.append(self.config_file)
 154             subprocess.call(editor)
 155         except Exception as e:
 156             raise AnsibleError("Failed to open editor: %s" % to_native(e))
 157 
 158     def execute_list(self):
 159         '''
 160         list all current configs reading lib/constants.py and shows env and config file setting names
 161         '''
 162         self.pager(to_text(yaml.dump(self.config.get_configuration_definitions(), Dumper=AnsibleDumper), errors='surrogate_or_strict'))
 163 
 164     def execute_dump(self):
 165         '''
 166         Shows the current settings, merges ansible.cfg if specified
 167         '''
 168         # FIXME: deal with plugins, not just base config
 169         text = []
 170         defaults = self.config.get_configuration_definitions().copy()
 171         for setting in self.config.data.get_settings():
 172             if setting.name in defaults:
 173                 defaults[setting.name] = setting
 174 
 175         for setting in sorted(defaults):
 176             if isinstance(defaults[setting], Setting):
 177                 if defaults[setting].origin == 'default':
 178                     color = 'green'
 179                 else:
 180                     color = 'yellow'
 181                 msg = "%s(%s) = %s" % (setting, defaults[setting].origin, defaults[setting].value)
 182             else:
 183                 color = 'green'
 184                 msg = "%s(%s) = %s" % (setting, 'default', defaults[setting].get('default'))
 185             if not context.CLIARGS['only_changed'] or color == 'yellow':
 186                 text.append(stringc(msg, color))
 187 
 188         self.pager(to_text('\n'.join(text), errors='surrogate_or_strict'))
