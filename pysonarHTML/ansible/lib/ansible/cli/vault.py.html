  24 class VaultCLI(CLI):
  25     ''' can encrypt any structured data file used by Ansible.
  26     This can include *group_vars/* or *host_vars/* inventory variables,
  27     variables loaded by *include_vars* or *vars_files*, or variable files
  28     passed on the ansible-playbook command line with *-e @file.yml* or *-e @file.json*.
  29     Role variables and defaults are also included!
  30 
  31     Because Ansible tasks, handlers, and other objects are data, these can also be encrypted with vault.
  32     If you'd like to not expose what variables you are using, you can keep an individual task file entirely encrypted.
  33     '''

 118     def post_process_args(self, options):
 119         options = super(VaultCLI, self).post_process_args(options)
 120 
 121         display.verbosity = options.verbosity
 122 
 123         if options.vault_ids:
 124             for vault_id in options.vault_ids:
 125                 if u';' in vault_id:
 126                     raise AnsibleOptionsError("'%s' is not a valid vault id. The character ';' is not allowed in vault ids" % vault_id)
 127 
 128         if getattr(options, 'output_file', None) and len(options.args) > 1:
 129             raise AnsibleOptionsError("At most one input file may be used with the --output option")
 130 
 131         if options.action == 'encrypt_string':
 132             if '-' in options.args or not options.args or options.encrypt_string_stdin_name:
 133                 self.encrypt_string_read_stdin = True
 134 
 135             # TODO: prompting from stdin and reading from stdin seem mutually exclusive, but verify that.
 136             if options.encrypt_string_prompt and self.encrypt_string_read_stdin:
 137                 raise AnsibleOptionsError('The --prompt option is not supported if also reading input from stdin')
 138 
 139         return options

 141     def run(self):
 142         super(VaultCLI, self).run()
 143         loader = DataLoader()
 144 
 145         # set default restrictive umask
 146         old_umask = os.umask(0o077)
 147 
 148         vault_ids = list(context.CLIARGS['vault_ids'])
 149 
 150         # there are 3 types of actions, those that just 'read' (decrypt, view) and only
 151         # need to ask for a password once, and those that 'write' (create, encrypt) that
 152         # ask for a new password and confirm it, and 'read/write (rekey) that asks for the
 153         # old password, then asks for a new one and confirms it.
 154 
 155         default_vault_ids = C.DEFAULT_VAULT_IDENTITY_LIST
 156         vault_ids = default_vault_ids + vault_ids
 157 
 158         action = context.CLIARGS['action']
 159 
 160         # TODO: instead of prompting for these before, we could let VaultEditor
 161         #       call a callback when it needs it.
 162         if action in ['decrypt', 'view', 'rekey', 'edit']:
 163             vault_secrets = self.setup_vault_secrets(loader, vault_ids=vault_ids,
 164                                                      vault_password_files=list(context.CLIARGS['vault_password_files']),
 165                                                      ask_vault_pass=context.CLIARGS['ask_vault_pass'])
 166             if not vault_secrets:
 167                 raise AnsibleOptionsError("A vault password is required to use Ansible's Vault")
 168 
 169         if action in ['encrypt', 'encrypt_string', 'create']:
 170 
 171             encrypt_vault_id = None
 172             # no --encrypt-vault-id context.CLIARGS['encrypt_vault_id'] for 'edit'
 173             if action not in ['edit']:
 174                 encrypt_vault_id = context.CLIARGS['encrypt_vault_id'] or C.DEFAULT_VAULT_ENCRYPT_IDENTITY
 175 
 176             vault_secrets = None
 177             vault_secrets = \
 178                 self.setup_vault_secrets(loader,
 179                                          vault_ids=vault_ids,
 180                                          vault_password_files=list(context.CLIARGS['vault_password_files']),
 181                                          ask_vault_pass=context.CLIARGS['ask_vault_pass'],
 182                                          create_new_password=True)
 183 
 184             if len(vault_secrets) > 1 and not encrypt_vault_id:
 185                 raise AnsibleOptionsError("The vault-ids %s are available to encrypt. Specify the vault-id to encrypt with --encrypt-vault-id" %
 186                                           ','.join([x[0] for x in vault_secrets]))
 187 
 188             if not vault_secrets:
 189                 raise AnsibleOptionsError("A vault password is required to use Ansible's Vault")
 190 
 191             encrypt_secret = match_encrypt_secret(vault_secrets,
 192                                                   encrypt_vault_id=encrypt_vault_id)
 193 
 194             # only one secret for encrypt for now, use the first vault_id and use its first secret
 195             # TODO: exception if more than one?
 196             self.encrypt_vault_id = encrypt_secret[0]
 197             self.encrypt_secret = encrypt_secret[1]
 198 
 199         if action in ['rekey']:
 200             encrypt_vault_id = context.CLIARGS['encrypt_vault_id'] or C.DEFAULT_VAULT_ENCRYPT_IDENTITY
 201             # print('encrypt_vault_id: %s' % encrypt_vault_id)
 202             # print('default_encrypt_vault_id: %s' % default_encrypt_vault_id)
 203 
 204             # new_vault_ids should only ever be one item, from
 205             # load the default vault ids if we are using encrypt-vault-id
 206             new_vault_ids = []
 207             if encrypt_vault_id:
 208                 new_vault_ids = default_vault_ids
 209             if context.CLIARGS['new_vault_id']:
 210                 new_vault_ids.append(context.CLIARGS['new_vault_id'])
 211 
 212             new_vault_password_files = []
 213             if context.CLIARGS['new_vault_password_file']:
 214                 new_vault_password_files.append(context.CLIARGS['new_vault_password_file'])
 215 
 216             new_vault_secrets = \
 217                 self.setup_vault_secrets(loader,
 218                                          vault_ids=new_vault_ids,
 219                                          vault_password_files=new_vault_password_files,
 220                                          ask_vault_pass=context.CLIARGS['ask_vault_pass'],
 221                                          create_new_password=True)
 222 
 223             if not new_vault_secrets:
 224                 raise AnsibleOptionsError("A new vault password is required to use Ansible's Vault rekey")
 225 
 226             # There is only one new_vault_id currently and one new_vault_secret, or we
 227             # use the id specified in --encrypt-vault-id
 228             new_encrypt_secret = match_encrypt_secret(new_vault_secrets,
 229                                                       encrypt_vault_id=encrypt_vault_id)
 230 
 231             self.new_encrypt_vault_id = new_encrypt_secret[0]
 232             self.new_encrypt_secret = new_encrypt_secret[1]
 233 
 234         loader.set_vault_secrets(vault_secrets)
 235 
 236         # FIXME: do we need to create VaultEditor here? its not reused
 237         vault = VaultLib(vault_secrets)
 238         self.editor = VaultEditor(vault)
 239 
 240         context.CLIARGS['func']()
 241 
 242         # and restore umask
 243         os.umask(old_umask)
 244 
 245     def execute_encrypt(self):
 246         ''' encrypt the supplied file using the provided vault secret '''
 247 
 248         if not context.CLIARGS['args'] and sys.stdin.isatty():
 249             display.display("Reading plaintext input from stdin", stderr=True)
 250 
 251         for f in context.CLIARGS['args'] or ['-']:
 252             # Fixme: use the correct vau
 253             self.editor.encrypt_file(f, self.encrypt_secret,
 254                                      vault_id=self.encrypt_vault_id,
 255                                      output_file=context.CLIARGS['output_file'])
 256 
 257         if sys.stdout.isatty():
 258             display.display("Encryption successful", stderr=True)

 260     @staticmethod
 261     def format_ciphertext_yaml(b_ciphertext, indent=None, name=None):
 262         indent = indent or 10
 263 
 264         block_format_var_name = ""
 265         if name:
 266             block_format_var_name = "%s: " % name
 267 
 268         block_format_header = "%s!vault |" % block_format_var_name
 269         lines = []
 270         vault_ciphertext = to_text(b_ciphertext)
 271 
 272         lines.append(block_format_header)
 273         for line in vault_ciphertext.splitlines():
 274             lines.append('%s%s' % (' ' * indent, line))
 275 
 276         yaml_ciphertext = '\n'.join(lines)
 277         return yaml_ciphertext

 279     def execute_encrypt_string(self):
 280         ''' encrypt the supplied string using the provided vault secret '''
 281         b_plaintext = None
 282 
 283         # Holds tuples (the_text, the_source_of_the_string, the variable name if its provided).
 284         b_plaintext_list = []
 285 
 286         # remove the non-option '-' arg (used to indicate 'read from stdin') from the candidate args so
 287         # we don't add it to the plaintext list
 288         args = [x for x in context.CLIARGS['args'] if x != '-']
 289 
 290         # We can prompt and read input, or read from stdin, but not both.
 291         if context.CLIARGS['encrypt_string_prompt']:
 292             msg = "String to encrypt: "
 293 
 294             name = None
 295             name_prompt_response = display.prompt('Variable name (enter for no name): ')
 296 
 297             # TODO: enforce var naming rules?
 298             if name_prompt_response != "":
 299                 name = name_prompt_response
 300 
 301             # TODO: could prompt for which vault_id to use for each plaintext string
 302             #       currently, it will just be the default
 303             # could use private=True for shadowed input if useful
 304             prompt_response = display.prompt(msg)
 305 
 306             if prompt_response == '':
 307                 raise AnsibleOptionsError('The plaintext provided from the prompt was empty, not encrypting')
 308 
 309             b_plaintext = to_bytes(prompt_response)
 310             b_plaintext_list.append((b_plaintext, self.FROM_PROMPT, name))
 311 
 312         # read from stdin
 313         if self.encrypt_string_read_stdin:
 314             if sys.stdout.isatty():
 315                 display.display("Reading plaintext input from stdin. (ctrl-d to end input)", stderr=True)
 316 
 317             stdin_text = sys.stdin.read()
 318             if stdin_text == '':
 319                 raise AnsibleOptionsError('stdin was empty, not encrypting')
 320 
 321             if sys.stdout.isatty() and not stdin_text.endswith("\n"):
 322                 display.display("\n")
 323 
 324             b_plaintext = to_bytes(stdin_text)
 325 
 326             # defaults to None
 327             name = context.CLIARGS['encrypt_string_stdin_name']
 328             b_plaintext_list.append((b_plaintext, self.FROM_STDIN, name))
 329 
 330         # use any leftover args as strings to encrypt
 331         # Try to match args up to --name options
 332         if context.CLIARGS.get('encrypt_string_names', False):
 333             name_and_text_list = list(zip(context.CLIARGS['encrypt_string_names'], args))
 334 
 335             # Some but not enough --name's to name each var
 336             if len(args) > len(name_and_text_list):
 337                 # Trying to avoid ever showing the plaintext in the output, so this warning is vague to avoid that.
 338                 display.display('The number of --name options do not match the number of args.',
 339                                 stderr=True)
 340                 display.display('The last named variable will be "%s". The rest will not have'
 341                                 ' names.' % context.CLIARGS['encrypt_string_names'][-1],
 342                                 stderr=True)
 343 
 344             # Add the rest of the args without specifying a name
 345             for extra_arg in args[len(name_and_text_list):]:
 346                 name_and_text_list.append((None, extra_arg))
 347 
 348         # if no --names are provided, just use the args without a name.
 349         else:
 350             name_and_text_list = [(None, x) for x in args]
 351 
 352         # Convert the plaintext text objects to bytestrings and collect
 353         for name_and_text in name_and_text_list:
 354             name, plaintext = name_and_text
 355 
 356             if plaintext == '':
 357                 raise AnsibleOptionsError('The plaintext provided from the command line args was empty, not encrypting')
 358 
 359             b_plaintext = to_bytes(plaintext)
 360             b_plaintext_list.append((b_plaintext, self.FROM_ARGS, name))
 361 
 362         # TODO: specify vault_id per string?
 363         # Format the encrypted strings and any corresponding stderr output
 364         outputs = self._format_output_vault_strings(b_plaintext_list, vault_id=self.encrypt_vault_id)
 365 
 366         for output in outputs:
 367             err = output.get('err', None)
 368             out = output.get('out', '')
 369             if err:
 370                 sys.stderr.write(err)
 371             print(out)
 372 
 373         if sys.stdout.isatty():
 374             display.display("Encryption successful", stderr=True)
 375

 378     def _format_output_vault_strings(self, b_plaintext_list, vault_id=None):
 379         # If we are only showing one item in the output, we don't need to included commented
 380         # delimiters in the text
 381         show_delimiter = False
 382         if len(b_plaintext_list) > 1:
 383             show_delimiter = True
 384 
 385         # list of dicts {'out': '', 'err': ''}
 386         output = []
 387 
 388         # Encrypt the plaintext, and format it into a yaml block that can be pasted into a playbook.
 389         # For more than one input, show some differentiating info in the stderr output so we can tell them
 390         # apart. If we have a var name, we include that in the yaml
 391         for index, b_plaintext_info in enumerate(b_plaintext_list):
 392             # (the text itself, which input it came from, its name)
 393             b_plaintext, src, name = b_plaintext_info
 394 
 395             b_ciphertext = self.editor.encrypt_bytes(b_plaintext, self.encrypt_secret,
 396                                                      vault_id=vault_id)
 397 
 398             # block formatting
 399             yaml_text = self.format_ciphertext_yaml(b_ciphertext, name=name)
 400 
 401             err_msg = None
 402             if show_delimiter:
 403                 human_index = index + 1
 404                 if name:
 405                     err_msg = '# The encrypted version of variable ("%s", the string #%d from %s).\n' % (name, human_index, src)
 406                 else:
 407                     err_msg = '# The encrypted version of the string #%d from %s.)\n' % (human_index, src)
 408             output.append({'out': yaml_text, 'err': err_msg})
 409 
 410         return output

 412     def execute_decrypt(self):
 413         ''' decrypt the supplied file using the provided vault secret '''
 414 
 415         if not context.CLIARGS['args'] and sys.stdin.isatty():
 416             display.display("Reading ciphertext input from stdin", stderr=True)
 417 
 418         for f in context.CLIARGS['args'] or ['-']:
 419             self.editor.decrypt_file(f, output_file=context.CLIARGS['output_file'])
 420 
 421         if sys.stdout.isatty():
 422             display.display("Decryption successful", stderr=True)
 423 
 424     def execute_create(self):
 425         ''' create and open a file in an editor that will be encrypted with the provided vault secret when closed'''
 426 
 427         if len(context.CLIARGS['args']) != 1:
 428             raise AnsibleOptionsError("ansible-vault create can take only one filename argument")
 429 
 430         self.editor.create_file(context.CLIARGS['args'][0], self.encrypt_secret,
 431                                 vault_id=self.encrypt_vault_id)
 432 
 433     def execute_edit(self):
 434         ''' open and decrypt an existing vaulted file in an editor, that will be encrypted again when closed'''
 435         for f in context.CLIARGS['args']:
 436             self.editor.edit_file(f)
 437 
 438     def execute_view(self):
 439         ''' open, decrypt and view an existing vaulted file using a pager using the supplied vault secret '''
 440 
 441         for f in context.CLIARGS['args']:
 442             # Note: vault should return byte strings because it could encrypt
 443             # and decrypt binary files.  We are responsible for changing it to
 444             # unicode here because we are displaying it and therefore can make
 445             # the decision that the display doesn't have to be precisely what
 446             # the input was (leave that to decrypt instead)
 447             plaintext = self.editor.plaintext(f)
 448             self.pager(to_text(plaintext))
 449 
 450     def execute_rekey(self):
 451         ''' re-encrypt a vaulted file with a new secret, the previous secret is required '''
 452         for f in context.CLIARGS['args']:
 453             # FIXME: plumb in vault_id, use the default new_vault_secret for now
 454             self.editor.rekey_file(f, self.new_encrypt_secret,
 455                                    self.new_encrypt_vault_id)
 456 
 457         display.display("Rekey successful", stderr=True)
