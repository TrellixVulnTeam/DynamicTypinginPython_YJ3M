  62 def add_all_plugin_dirs(path):
  63     ''' add any existing plugin dirs in the path provided '''
  64     b_path = os.path.expanduser(to_bytes(path, errors='surrogate_or_strict'))
  65     if os.path.isdir(b_path):
  66         for name, obj in get_all_plugin_loaders():
  67             if obj.subdir:
  68                 plugin_path = os.path.join(b_path, to_bytes(obj.subdir))
  69                 if os.path.isdir(plugin_path):
  70                     obj.add_directory(to_text(plugin_path))
  71     else:
  72         display.warning("Ignoring invalid path provided to plugin path: '%s' is not a directory" % to_text(path))

  75 def get_shell_plugin(shell_type=None, executable=None):
  76 
  77     if not shell_type:
  78         # default to sh
  79         shell_type = 'sh'
  80 
  81         # mostly for backwards compat
  82         if executable:
  83             if isinstance(executable, string_types):
  84                 shell_filename = os.path.basename(executable)
  85                 try:
  86                     shell = shell_loader.get(shell_filename)
  87                 except Exception:
  88                     shell = None
  89 
  90                 if shell is None:
  91                     for shell in shell_loader.all():
  92                         if shell_filename in shell.COMPATIBLE_SHELLS:
  93                             shell_type = shell.SHELL_FAMILY
  94                             break
  95         else:
  96             raise AnsibleError("Either a shell type or a shell executable must be provided ")
  97 
  98     shell = shell_loader.get(shell_type)
  99     if not shell:
 100         raise AnsibleError("Could not find the shell plugin required (%s)." % shell_type)
 101 
 102     if executable:
 103         setattr(shell, 'executable', executable)
 104 
 105     return shell

 121 class PluginLoadContext(object):
 122     def __init__(self):
 123         self.original_name = None
 124         self.redirect_list = []
 125         self.error_list = []
 126         self.import_error_list = []
 127         self.load_attempts = []
 128         self.pending_redirect = None
 129         self.exit_reason = None
 130         self.plugin_resolved_path = None
 131         self.plugin_resolved_name = None
 132         self.plugin_resolved_collection = None  # empty string for resolved plugins from user-supplied paths
 133         self.deprecated = False
 134         self.removal_date = None
 135         self.removal_version = None
 136         self.deprecation_warnings = []
 137         self.resolved = False
 138 
 139     def record_deprecation(self, name, deprecation, collection_name):
 140         if not deprecation:
 141             return self
 142 
 143         warning_text = deprecation.get('warning_text', None)
 144         removal_date = deprecation.get('removal_date', None)
 145         removal_version = deprecation.get('removal_version', None)
 146         # If both removal_date and removal_version are specified, use removal_date
 147         if removal_date is not None:
 148             removal_version = None
 149         if not warning_text:
 150             warning_text = '{0} has been deprecated'.format(name)
 151 
 152         display.deprecated(warning_text, date=removal_date, version=removal_version, collection_name=collection_name)
 153 
 154         self.deprecated = True
 155         if removal_date:
 156             self.removal_date = removal_date
 157         if removal_version:
 158             self.removal_version = removal_version
 159         self.deprecation_warnings.append(warning_text)
 160         return self
 161 
 162     def resolve(self, resolved_name, resolved_path, resolved_collection, exit_reason):
 163         self.pending_redirect = None
 164         self.plugin_resolved_name = resolved_name
 165         self.plugin_resolved_path = resolved_path
 166         self.plugin_resolved_collection = resolved_collection
 167         self.exit_reason = exit_reason
 168         self.resolved = True
 169         return self
 170 
 171     def redirect(self, redirect_name):
 172         self.pending_redirect = redirect_name
 173         self.exit_reason = 'pending redirect resolution from {0} to {1}'.format(self.original_name, redirect_name)
 174         self.resolved = False
 175         return self
 176 
 177     def nope(self, exit_reason):
 178         self.pending_redirect = None
 179         self.exit_reason = exit_reason
 180         self.resolved = False
 181         return self

 184 class PluginLoader:
 185     '''
 186     PluginLoader loads plugins from the configured plugin directories.
 187 
 188     It searches for plugins by iterating through the combined list of play basedirs, configured
 189     paths, and the python path.  The first match is used.
 190     '''
 191 
 192     def __init__(self, class_name, package, config, subdir, aliases=None, required_base_class=None):
 193         aliases = {} if aliases is None else aliases
 194 
 195         self.class_name = class_name
 196         self.base_class = required_base_class
 197         self.package = package
 198         self.subdir = subdir
 199 
 200         # FIXME: remove alias dict in favor of alias by symlink?
 201         self.aliases = aliases
 202 
 203         if config and not isinstance(config, list):
 204             config = [config]
 205         elif not config:
 206             config = []
 207 
 208         self.config = config
 209 
 210         if class_name not in MODULE_CACHE:
 211             MODULE_CACHE[class_name] = {}
 212         if class_name not in PATH_CACHE:
 213             PATH_CACHE[class_name] = None
 214         if class_name not in PLUGIN_PATH_CACHE:
 215             PLUGIN_PATH_CACHE[class_name] = defaultdict(dict)
 216 
 217         # hold dirs added at runtime outside of config
 218         self._extra_dirs = []
 219 
 220         # caches
 221         self._module_cache = MODULE_CACHE[class_name]
 222         self._paths = PATH_CACHE[class_name]
 223         self._plugin_path_cache = PLUGIN_PATH_CACHE[class_name]
 224 
 225         self._searched_paths = set()
 226 
 227     def __repr__(self):
 228         return 'PluginLoader(type={0})'.format(AnsibleCollectionRef.legacy_plugin_dir_to_plugin_type(self.subdir))
 229 
 230     def _clear_caches(self):
 231 
 232         if C.OLD_PLUGIN_CACHE_CLEARING:
 233             self._paths = None
 234         else:
 235             # reset global caches
 236             MODULE_CACHE[self.class_name] = {}
 237             PATH_CACHE[self.class_name] = None
 238             PLUGIN_PATH_CACHE[self.class_name] = defaultdict(dict)
 239 
 240             # reset internal caches
 241             self._module_cache = MODULE_CACHE[self.class_name]
 242             self._paths = PATH_CACHE[self.class_name]
 243             self._plugin_path_cache = PLUGIN_PATH_CACHE[self.class_name]
 244             self._searched_paths = set()
 245 
 246     def __setstate__(self, data):
 247         '''
 248         Deserializer.
 249         '''
 250 
 251         class_name = data.get('class_name')
 252         package = data.get('package')
 253         config = data.get('config')
 254         subdir = data.get('subdir')
 255         aliases = data.get('aliases')
 256         base_class = data.get('base_class')
 257 
 258         PATH_CACHE[class_name] = data.get('PATH_CACHE')
 259         PLUGIN_PATH_CACHE[class_name] = data.get('PLUGIN_PATH_CACHE')
 260 
 261         self.__init__(class_name, package, config, subdir, aliases, base_class)
 262         self._extra_dirs = data.get('_extra_dirs', [])
 263         self._searched_paths = data.get('_searched_paths', set())
 264 
 265     def __getstate__(self):
 266         '''
 267         Serializer.
 268         '''
 269 
 270         return dict(
 271             class_name=self.class_name,
 272             base_class=self.base_class,
 273             package=self.package,
 274             config=self.config,
 275             subdir=self.subdir,
 276             aliases=self.aliases,
 277             _extra_dirs=self._extra_dirs,
 278             _searched_paths=self._searched_paths,
 279             PATH_CACHE=PATH_CACHE[self.class_name],
 280             PLUGIN_PATH_CACHE=PLUGIN_PATH_CACHE[self.class_name],
 281         )

 283     def format_paths(self, paths):
 284         ''' Returns a string suitable for printing of the search path '''
 285 
 286         # Uses a list to get the order right
 287         ret = []
 288         for i in paths:
 289             if i not in ret:
 290                 ret.append(i)
 291         return os.pathsep.join(ret)

 293     def print_paths(self):
 294         return self.format_paths(self._get_paths(subdirs=False))

 296     def _all_directories(self, dir):
 297         results = []
 298         results.append(dir)
 299         for root, subdirs, files in os.walk(dir, followlinks=True):
 300             if '__init__.py' in files:
 301                 for x in subdirs:
 302                     results.append(os.path.join(root, x))
 303         return results

 305     def _get_package_paths(self, subdirs=True):
 306         ''' Gets the path of a Python package '''
 307 
 308         if not self.package:
 309             return []
 310         if not hasattr(self, 'package_path'):
 311             m = __import__(self.package)
 312             parts = self.package.split('.')[1:]
 313             for parent_mod in parts:
 314                 m = getattr(m, parent_mod)
 315             self.package_path = os.path.dirname(m.__file__)
 316         if subdirs:
 317             return self._all_directories(self.package_path)
 318         return [self.package_path]

 320     def _get_paths_with_context(self, subdirs=True):
 321         ''' Return a list of PluginPathContext objects to search for plugins in '''
 322 
 323         # FIXME: This is potentially buggy if subdirs is sometimes True and sometimes False.
 324         # In current usage, everything calls this with subdirs=True except for module_utils_loader and ansible-doc
 325         # which always calls it with subdirs=False. So there currently isn't a problem with this caching.
 326         if self._paths is not None:
 327             return self._paths
 328 
 329         ret = [PluginPathContext(p, False) for p in self._extra_dirs]
 330 
 331         # look in any configured plugin paths, allow one level deep for subcategories
 332         if self.config is not None:
 333             for path in self.config:
 334                 path = os.path.realpath(os.path.expanduser(path))
 335                 if subdirs:
 336                     contents = glob.glob("%s/*" % path) + glob.glob("%s/*/*" % path)
 337                     for c in contents:
 338                         if os.path.isdir(c) and c not in ret:
 339                             ret.append(PluginPathContext(c, False))
 340                 if path not in ret:
 341                     ret.append(PluginPathContext(path, False))
 342 
 343         # look for any plugins installed in the package subtree
 344         # Note package path always gets added last so that every other type of
 345         # path is searched before it.
 346         ret.extend([PluginPathContext(p, True) for p in self._get_package_paths(subdirs=subdirs)])
 347 
 348         # HACK: because powershell modules are in the same directory
 349         # hierarchy as other modules we have to process them last.  This is
 350         # because powershell only works on windows but the other modules work
 351         # anywhere (possibly including windows if the correct language
 352         # interpreter is installed).  the non-powershell modules can have any
 353         # file extension and thus powershell modules are picked up in that.
 354         # The non-hack way to fix this is to have powershell modules be
 355         # a different PluginLoader/ModuleLoader.  But that requires changing
 356         # other things too (known thing to change would be PATHS_CACHE,
 357         # PLUGIN_PATHS_CACHE, and MODULE_CACHE.  Since those three dicts key
 358         # on the class_name and neither regular modules nor powershell modules
 359         # would have class_names, they would not work as written.
 360         #
 361         # The expected sort order is paths in the order in 'ret' with paths ending in '/windows' at the end,
 362         # also in the original order they were found in 'ret'.
 363         # The .sort() method is guaranteed to be stable, so original order is preserved.
 364         ret.sort(key=lambda p: p.path.endswith('/windows'))
 365 
 366         # cache and return the result
 367         self._paths = ret
 368         return ret
 369 
 370     def _get_paths(self, subdirs=True):
 371         ''' Return a list of paths to search for plugins in '''
 372 
 373         paths_with_context = self._get_paths_with_context(subdirs=subdirs)
 374         return [path_with_context.path for path_with_context in paths_with_context]

 376     def _load_config_defs(self, name, module, path):
 377         ''' Reads plugin docs to find configuration setting definitions, to push to config manager for later use '''
 378 
 379         # plugins w/o class name don't support config
 380         if self.class_name:
 381             type_name = get_plugin_class(self.class_name)
 382 
 383             # if type name != 'module_doc_fragment':
 384             if type_name in C.CONFIGURABLE_PLUGINS:
 385                 dstring = AnsibleLoader(getattr(module, 'DOCUMENTATION', ''), file_name=path).get_single_data()
 386                 if dstring:
 387                     add_fragments(dstring, path, fragment_loader=fragment_loader, is_module=(type_name == 'module'))
 388 
 389                 if dstring and 'options' in dstring and isinstance(dstring['options'], dict):
 390                     C.config.initialize_plugin_configuration_definitions(type_name, name, dstring['options'])
 391                     display.debug('Loaded config def from plugin (%s/%s)' % (type_name, name))

 393     def add_directory(self, directory, with_subdir=False):
 394         ''' Adds an additional directory to the search path '''
 395 
 396         directory = os.path.realpath(directory)
 397 
 398         if directory is not None:
 399             if with_subdir:
 400                 directory = os.path.join(directory, self.subdir)
 401             if directory not in self._extra_dirs:
 402                 # append the directory and invalidate the path cache
 403                 self._extra_dirs.append(directory)
 404                 self._clear_caches()
 405                 display.debug('Added %s to loader search path' % (directory))

 407     def _query_collection_routing_meta(self, acr, plugin_type, extension=None):
 408         collection_pkg = import_module(acr.n_python_collection_package_name)
 409         if not collection_pkg:
 410             return None
 411 
 412         # FIXME: shouldn't need this...
 413         try:
 414             # force any type-specific metadata postprocessing to occur
 415             import_module(acr.n_python_collection_package_name + '.plugins.{0}'.format(plugin_type))
 416         except ImportError:
 417             pass
 418 
 419         # this will be created by the collection PEP302 loader
 420         collection_meta = getattr(collection_pkg, '_collection_meta', None)
 421 
 422         if not collection_meta:
 423             return None
 424 
 425         # TODO: add subdirs support
 426         # check for extension-specific entry first (eg 'setup.ps1')
 427         # TODO: str/bytes on extension/name munging
 428         if acr.subdirs:
 429             subdir_qualified_resource = '.'.join([acr.subdirs, acr.resource])
 430         else:
 431             subdir_qualified_resource = acr.resource
 432         entry = collection_meta.get('plugin_routing', {}).get(plugin_type, {}).get(subdir_qualified_resource + extension, None)
 433         if not entry:
 434             # try for extension-agnostic entry
 435             entry = collection_meta.get('plugin_routing', {}).get(plugin_type, {}).get(subdir_qualified_resource, None)
 436         return entry

 438     def _find_fq_plugin(self, fq_name, extension, plugin_load_context):
 439         """Search builtin paths to find a plugin. No external paths are searched,
 440         meaning plugins inside roles inside collections will be ignored.
 441         """
 442 
 443         plugin_load_context.resolved = False
 444 
 445         plugin_type = AnsibleCollectionRef.legacy_plugin_dir_to_plugin_type(self.subdir)
 446 
 447         acr = AnsibleCollectionRef.from_fqcr(fq_name, plugin_type)
 448 
 449         # check collection metadata to see if any special handling is required for this plugin
 450         routing_metadata = self._query_collection_routing_meta(acr, plugin_type, extension=extension)
 451 
 452         # TODO: factor this into a wrapper method
 453         if routing_metadata:
 454             deprecation = routing_metadata.get('deprecation', None)
 455 
 456             # this will no-op if there's no deprecation metadata for this plugin
 457             plugin_load_context.record_deprecation(fq_name, deprecation, acr.collection)
 458 
 459             tombstone = routing_metadata.get('tombstone', None)
 460 
 461             # FIXME: clean up text gen
 462             if tombstone:
 463                 removal_date = tombstone.get('removal_date')
 464                 removal_version = tombstone.get('removal_version')
 465                 warning_text = tombstone.get('warning_text') or '{0} has been removed.'.format(fq_name)
 466                 removed_msg = display.get_deprecation_message(msg=warning_text, version=removal_version,
 467                                                               date=removal_date, removed=True,
 468                                                               collection_name=acr.collection)
 469                 plugin_load_context.removal_date = removal_date
 470                 plugin_load_context.removal_version = removal_version
 471                 plugin_load_context.resolved = True
 472                 plugin_load_context.exit_reason = removed_msg
 473                 raise AnsiblePluginRemovedError(removed_msg, plugin_load_context=plugin_load_context)
 474 
 475             redirect = routing_metadata.get('redirect', None)
 476 
 477             if redirect:
 478                 # FIXME: remove once this is covered in debug or whatever
 479                 display.vv("redirecting (type: {0}) {1} to {2}".format(plugin_type, fq_name, redirect))
 480                 return plugin_load_context.redirect(redirect)
 481                 # TODO: non-FQCN case, do we support `.` prefix for current collection, assume it with no dots, require it for subdirs in current, or ?
 482 
 483         n_resource = to_native(acr.resource, errors='strict')
 484         # we want this before the extension is added
 485         full_name = '{0}.{1}'.format(acr.n_python_package_name, n_resource)
 486 
 487         if extension:
 488             n_resource += extension
 489 
 490         pkg = sys.modules.get(acr.n_python_package_name)
 491         if not pkg:
 492             # FIXME: there must be cheaper/safer way to do this
 493             try:
 494                 pkg = import_module(acr.n_python_package_name)
 495             except ImportError:
 496                 return plugin_load_context.nope('Python package {0} not found'.format(acr.n_python_package_name))
 497 
 498         pkg_path = os.path.dirname(pkg.__file__)
 499 
 500         n_resource_path = os.path.join(pkg_path, n_resource)
 501 
 502         # FIXME: and is file or file link or ...
 503         if os.path.exists(n_resource_path):
 504             return plugin_load_context.resolve(
 505                 full_name, to_text(n_resource_path), acr.collection, 'found exact match for {0} in {1}'.format(full_name, acr.collection))
 506 
 507         if extension:
 508             # the request was extension-specific, don't try for an extensionless match
 509             return plugin_load_context.nope('no match for {0} in {1}'.format(to_text(n_resource), acr.collection))
 510 
 511         # look for any matching extension in the package location (sans filter)
 512         found_files = [f
 513                        for f in glob.iglob(os.path.join(pkg_path, n_resource) + '.*')
 514                        if os.path.isfile(f) and not f.endswith(C.MODULE_IGNORE_EXTS)]
 515 
 516         if not found_files:
 517             return plugin_load_context.nope('failed fuzzy extension match for {0} in {1}'.format(full_name, acr.collection))
 518 
 519         if len(found_files) > 1:
 520             # TODO: warn?
 521             pass
 522 
 523         return plugin_load_context.resolve(
 524             full_name, to_text(found_files[0]), acr.collection, 'found fuzzy extension match for {0} in {1}'.format(full_name, acr.collection))
 525 
 526     def find_plugin(self, name, mod_type='', ignore_deprecated=False, check_aliases=False, collection_list=None):
 527         ''' Find a plugin named name '''
 528         result = self.find_plugin_with_context(name, mod_type, ignore_deprecated, check_aliases, collection_list)
 529         if result.resolved and result.plugin_resolved_path:
 530             return result.plugin_resolved_path
 531 
 532         return None

 534     def find_plugin_with_context(self, name, mod_type='', ignore_deprecated=False, check_aliases=False, collection_list=None):
 535         ''' Find a plugin named name, returning contextual info about the load, recursively resolving redirection '''
 536         plugin_load_context = PluginLoadContext()
 537         plugin_load_context.original_name = name
 538         while True:
 539             result = self._resolve_plugin_step(name, mod_type, ignore_deprecated, check_aliases, collection_list, plugin_load_context=plugin_load_context)
 540             if result.pending_redirect:
 541                 if result.pending_redirect in result.redirect_list:
 542                     raise AnsiblePluginCircularRedirect('plugin redirect loop resolving {0} (path: {1})'.format(result.original_name, result.redirect_list))
 543                 name = result.pending_redirect
 544                 result.pending_redirect = None
 545                 plugin_load_context = result
 546             else:
 547                 break
 548 
 549         # TODO: smuggle these to the controller when we're in a worker, reduce noise from normal things like missing plugin packages during collection search
 550         if plugin_load_context.error_list:
 551             display.warning("errors were encountered during the plugin load for {0}:\n{1}".format(name, plugin_load_context.error_list))
 552 
 553         # TODO: display/return import_error_list? Only useful for forensics...
 554 
 555         # FIXME: store structured deprecation data in PluginLoadContext and use display.deprecate

 563     # FIXME: name bikeshed
 564     def _resolve_plugin_step(self, name, mod_type='', ignore_deprecated=False,
 565                              check_aliases=False, collection_list=None, plugin_load_context=PluginLoadContext()):
 566         if not plugin_load_context:
 567             raise ValueError('A PluginLoadContext is required')
 568 
 569         plugin_load_context.redirect_list.append(name)
 570         plugin_load_context.resolved = False
 571 
 572         global _PLUGIN_FILTERS
 573         if name in _PLUGIN_FILTERS[self.package]:
 574             plugin_load_context.exit_reason = '{0} matched a defined plugin filter'.format(name)
 575             return plugin_load_context
 576 
 577         if mod_type:
 578             suffix = mod_type
 579         elif self.class_name:
 580             # Ansible plugins that run in the controller process (most plugins)
 581             suffix = '.py'
 582         else:
 583             # Only Ansible Modules.  Ansible modules can be any executable so
 584             # they can have any suffix
 585             suffix = ''
 586 
 587         # FIXME: need this right now so we can still load shipped PS module_utils- come up with a more robust solution
 588         if (AnsibleCollectionRef.is_valid_fqcr(name) or collection_list) and not name.startswith('Ansible'):
 589             if '.' in name or not collection_list:
 590                 candidates = [name]
 591             else:
 592                 candidates = ['{0}.{1}'.format(c, name) for c in collection_list]
 593 
 594             for candidate_name in candidates:
 595                 try:
 596                     plugin_load_context.load_attempts.append(candidate_name)
 597                     # HACK: refactor this properly
 598                     if candidate_name.startswith('ansible.legacy'):
 599                         # 'ansible.legacy' refers to the plugin finding behavior used before collections existed.
 600                         # They need to search 'library' and the various '*_plugins' directories in order to find the file.
 601                         plugin_load_context = self._find_plugin_legacy(name.replace('ansible.legacy.', '', 1),
 602                                                                        plugin_load_context, ignore_deprecated, check_aliases, suffix)
 603                     else:
 604                         # 'ansible.builtin' should be handled here. This means only internal, or builtin, paths are searched.
 605                         plugin_load_context = self._find_fq_plugin(candidate_name, suffix, plugin_load_context=plugin_load_context)
 606 
 607                         if candidate_name != plugin_load_context.original_name and candidate_name not in plugin_load_context.redirect_list:
 608                             plugin_load_context.redirect_list.append(candidate_name)
 609 
 610                     if plugin_load_context.resolved or plugin_load_context.pending_redirect:  # if we got an answer or need to chase down a redirect, return
 611                         return plugin_load_context
 612                 except (AnsiblePluginRemovedError, AnsiblePluginCircularRedirect, AnsibleCollectionUnsupportedVersionError):
 613                     # these are generally fatal, let them fly
 614                     raise
 615                 except ImportError as ie:
 616                     plugin_load_context.import_error_list.append(ie)
 617                 except Exception as ex:
 618                     # FIXME: keep actual errors, not just assembled messages
 619                     plugin_load_context.error_list.append(to_native(ex))
 620 
 621             if plugin_load_context.error_list:
 622                 display.debug(msg='plugin lookup for {0} failed; errors: {1}'.format(name, '; '.join(plugin_load_context.error_list)))
 623 
 624             plugin_load_context.exit_reason = 'no matches found for {0}'.format(name)
 625 
 626             return plugin_load_context
 627 
 628         # if we got here, there's no collection list and it's not an FQ name, so do legacy lookup
 629 
 630         return self._find_plugin_legacy(name, plugin_load_context, ignore_deprecated, check_aliases, suffix)

 632     def _find_plugin_legacy(self, name, plugin_load_context, ignore_deprecated=False, check_aliases=False, suffix=None):
 633         """Search library and various *_plugins paths in order to find the file.
 634         This was behavior prior to the existence of collections.
 635         """
 636         plugin_load_context.resolved = False
 637 
 638         if check_aliases:
 639             name = self.aliases.get(name, name)
 640 
 641         # The particular cache to look for modules within.  This matches the
 642         # requested mod_type
 643         pull_cache = self._plugin_path_cache[suffix]
 644         try:
 645             path_with_context = pull_cache[name]
 646             plugin_load_context.plugin_resolved_path = path_with_context.path
 647             plugin_load_context.plugin_resolved_name = name
 648             plugin_load_context.plugin_resolved_collection = 'ansible.builtin' if path_with_context.internal else ''
 649             plugin_load_context.resolved = True
 650             return plugin_load_context
 651         except KeyError:
 652             # Cache miss.  Now let's find the plugin
 653             pass
 654 
 655         # TODO: Instead of using the self._paths cache (PATH_CACHE) and
 656         #       self._searched_paths we could use an iterator.  Before enabling that
 657         #       we need to make sure we don't want to add additional directories
 658         #       (add_directory()) once we start using the iterator.
 659         #       We can use _get_paths_with_context() since add_directory() forces a cache refresh.
 660         for path_context in (p for p in self._get_paths_with_context() if p.path not in self._searched_paths and os.path.isdir(p.path)):
 661             path = path_context.path
 662             display.debug('trying %s' % path)
 663             plugin_load_context.load_attempts.append(path)
 664             try:
 665                 full_paths = (os.path.join(path, f) for f in os.listdir(path))
 666             except OSError as e:
 667                 display.warning("Error accessing plugin paths: %s" % to_text(e))
 668 
 669             for full_path in (f for f in full_paths if os.path.isfile(f) and not f.endswith('__init__.py')):
 670                 full_name = os.path.basename(full_path)
 671 
 672                 # HACK: We have no way of executing python byte compiled files as ansible modules so specifically exclude them
 673                 # FIXME: I believe this is only correct for modules and module_utils.
 674                 # For all other plugins we want .pyc and .pyo should be valid
 675                 if any(full_path.endswith(x) for x in C.MODULE_IGNORE_EXTS):
 676                     continue
 677 
 678                 splitname = os.path.splitext(full_name)
 679                 base_name = splitname[0]
 680                 internal = path_context.internal
 681                 try:
 682                     extension = splitname[1]
 683                 except IndexError:
 684                     extension = ''
 685 
 686                 # Module found, now enter it into the caches that match this file
 687                 if base_name not in self._plugin_path_cache['']:
 688                     self._plugin_path_cache[''][base_name] = PluginPathContext(full_path, internal)
 689 
 690                 if full_name not in self._plugin_path_cache['']:
 691                     self._plugin_path_cache[''][full_name] = PluginPathContext(full_path, internal)
 692 
 693                 if base_name not in self._plugin_path_cache[extension]:
 694                     self._plugin_path_cache[extension][base_name] = PluginPathContext(full_path, internal)
 695 
 696                 if full_name not in self._plugin_path_cache[extension]:
 697                     self._plugin_path_cache[extension][full_name] = PluginPathContext(full_path, internal)
 698 
 699             self._searched_paths.add(path)
 700             try:
 701                 path_with_context = pull_cache[name]
 702                 plugin_load_context.plugin_resolved_path = path_with_context.path
 703                 plugin_load_context.plugin_resolved_name = name
 704                 plugin_load_context.plugin_resolved_collection = 'ansible.builtin' if path_with_context.internal else ''
 705                 plugin_load_context.resolved = True
 706                 return plugin_load_context
 707             except KeyError:
 708                 # Didn't find the plugin in this directory. Load modules from the next one
 709                 pass
 710 
 711         # if nothing is found, try finding alias/deprecated
 712         if not name.startswith('_'):
 713             alias_name = '_' + name
 714             # We've already cached all the paths at this point
 715             if alias_name in pull_cache:
 716                 path_with_context = pull_cache[alias_name]
 717                 if not ignore_deprecated and not os.path.islink(path_with_context.path):
 718                     # FIXME: this is not always the case, some are just aliases
 719                     display.deprecated('%s is kept for backwards compatibility but usage is discouraged. '  # pylint: disable=ansible-deprecated-no-version
 720                                        'The module documentation details page may explain more about this rationale.' % name.lstrip('_'))
 721                 plugin_load_context.plugin_resolved_path = path_with_context.path
 722                 plugin_load_context.plugin_resolved_name = alias_name
 723                 plugin_load_context.plugin_resolved_collection = 'ansible.builtin' if path_with_context.internal else ''
 724                 plugin_load_context.resolved = True
 725                 return plugin_load_context
 726 
 727         # last ditch, if it's something that can be redirected, look for a builtin redirect before giving up
 728         candidate_fqcr = 'ansible.builtin.{0}'.format(name)
 729         if '.' not in name and AnsibleCollectionRef.is_valid_fqcr(candidate_fqcr):
 730             return self._find_fq_plugin(fq_name=candidate_fqcr, extension=suffix, plugin_load_context=plugin_load_context)
 731 
 732         return plugin_load_context.nope('{0} is not eligible for last-chance resolution'.format(name))
 733

 747     def _load_module_source(self, name, path):
 748 
 749         # avoid collisions across plugins
 750         if name.startswith('ansible_collections.'):
 751             full_name = name
 752         else:
 753             full_name = '.'.join([self.package, name])
 754 
 755         if full_name in sys.modules:
 756             # Avoids double loading, See https://github.com/ansible/ansible/issues/13110
 757             return sys.modules[full_name]
 758 
 759         with warnings.catch_warnings():
 760             warnings.simplefilter("ignore", RuntimeWarning)
 761             if imp is None:
 762                 spec = importlib.util.spec_from_file_location(to_native(full_name), to_native(path))
 763                 module = importlib.util.module_from_spec(spec)
 764                 spec.loader.exec_module(module)
 765                 sys.modules[full_name] = module
 766             else:
 767                 with open(to_bytes(path), 'rb') as module_file:
 768                     # to_native is used here because imp.load_source's path is for tracebacks and python's traceback formatting uses native strings
 769                     module = imp.load_source(to_native(full_name), to_native(path), module_file)
 770         return module

 782     def get_with_context(self, name, *args, **kwargs):
 783         ''' instantiates a plugin of the given name using arguments '''
 784 
 785         found_in_cache = True
 786         class_only = kwargs.pop('class_only', False)
 787         collection_list = kwargs.pop('collection_list', None)
 788         if name in self.aliases:
 789             name = self.aliases[name]
 790         plugin_load_context = self.find_plugin_with_context(name, collection_list=collection_list)
 791         if not plugin_load_context.resolved or not plugin_load_context.plugin_resolved_path:
 792             # FIXME: this is probably an error (eg removed plugin)
 793             return get_with_context_result(None, plugin_load_context)
 794 
 795         name = plugin_load_context.plugin_resolved_name
 796         path = plugin_load_context.plugin_resolved_path
 797         redirected_names = plugin_load_context.redirect_list or []
 798 
 799         if path not in self._module_cache:
 800             self._module_cache[path] = self._load_module_source(name, path)
 801             self._load_config_defs(name, self._module_cache[path], path)
 802             found_in_cache = False
 803 
 804         obj = getattr(self._module_cache[path], self.class_name)
 805         if self.base_class:
 806             # The import path is hardcoded and should be the right place,
 807             # so we are not expecting an ImportError.
 808             module = __import__(self.package, fromlist=[self.base_class])
 809             # Check whether this obj has the required base class.
 810             try:
 811                 plugin_class = getattr(module, self.base_class)
 812             except AttributeError:
 813                 return get_with_context_result(None, plugin_load_context)
 814             if not issubclass(obj, plugin_class):
 815                 return get_with_context_result(None, plugin_load_context)
 816 
 817         # FIXME: update this to use the load context
 818         self._display_plugin_load(self.class_name, name, self._searched_paths, path, found_in_cache=found_in_cache, class_only=class_only)
 819 
 820         if not class_only:
 821             try:
 822                 # A plugin may need to use its _load_name in __init__ (for example, to set
 823                 # or get options from config), so update the object before using the constructor
 824                 instance = object.__new__(obj)
 825                 self._update_object(instance, name, path, redirected_names)
 826                 obj.__init__(instance, *args, **kwargs)
 827                 obj = instance
 828             except TypeError as e:
 829                 if "abstract" in e.args[0]:
 830                     # Abstract Base Class.  The found plugin file does not
 831                     # fully implement the defined interface.
 832                     return get_with_context_result(None, plugin_load_context)
 833                 raise
 834 
 835         self._update_object(obj, name, path, redirected_names)
 836         return get_with_context_result(obj, plugin_load_context)

 838     def _display_plugin_load(self, class_name, name, searched_paths, path, found_in_cache=None, class_only=None):
 839         ''' formats data to display debug info for plugin loading, also avoids processing unless really needed '''
 840         if C.DEFAULT_DEBUG:
 841             msg = 'Loading %s \'%s\' from %s' % (class_name, os.path.basename(name), path)
 842 
 843             if len(searched_paths) > 1:
 844                 msg = '%s (searched paths: %s)' % (msg, self.format_paths(searched_paths))
 845 
 846             if found_in_cache or class_only:
 847                 msg = '%s (found_in_cache=%s, class_only=%s)' % (msg, found_in_cache, class_only)
 848 
 849             display.debug(msg)

 851     def all(self, *args, **kwargs):
 852         '''
 853         Iterate through all plugins of this type
 854 
 855         A plugin loader is initialized with a specific type.  This function is an iterator returning
 856         all of the plugins of that type to the caller.
 857 
 858         :kwarg path_only: If this is set to True, then we return the paths to where the plugins reside
 859             instead of an instance of the plugin.  This conflicts with class_only and both should
 860             not be set.
 861         :kwarg class_only: If this is set to True then we return the python class which implements
 862             a plugin rather than an instance of the plugin.  This conflicts with path_only and both
 863             should not be set.
 864         :kwarg _dedupe: By default, we only return one plugin per plugin name.  Deduplication happens
 865             in the same way as the :meth:`get` and :meth:`find_plugin` methods resolve which plugin
 866             should take precedence.  If this is set to False, then we return all of the plugins
 867             found, including those with duplicate names.  In the case of duplicates, the order in
 868             which they are returned is the one that would take precedence first, followed by the
 869             others  in decreasing precedence order.  This should only be used by subclasses which
 870             want to manage their own deduplication of the plugins.
 871         :*args: Any extra arguments are passed to each plugin when it is instantiated.
 872         :**kwargs: Any extra keyword arguments are passed to each plugin when it is instantiated.
 873         '''
 874         # TODO: Change the signature of this method to:
 875         # def all(return_type='instance', args=None, kwargs=None):
 876         #     if args is None: args = []
 877         #     if kwargs is None: kwargs = {}
 878         #     return_type can be instance, class, or path.
 879         #     These changes will mean that plugin parameters won't conflict with our params and
 880         #     will also make it impossible to request both a path and a class at the same time.
 881         #
 882         #     Move _dedupe to be a class attribute, CUSTOM_DEDUPE, with subclasses for filters and
 883         #     tests setting it to True
 884 
 885         global _PLUGIN_FILTERS
 886 
 887         dedupe = kwargs.pop('_dedupe', True)
 888         path_only = kwargs.pop('path_only', False)
 889         class_only = kwargs.pop('class_only', False)
 890         # Having both path_only and class_only is a coding bug
 891         if path_only and class_only:
 892             raise AnsibleError('Do not set both path_only and class_only when calling PluginLoader.all()')
 893 
 894         all_matches = []
 895         found_in_cache = True
 896 
 897         for i in self._get_paths():
 898             all_matches.extend(glob.glob(os.path.join(i, "*.py")))
 899 
 900         loaded_modules = set()
 901         for path in sorted(all_matches, key=os.path.basename):
 902             name = os.path.splitext(path)[0]
 903             basename = os.path.basename(name)
 904 
 905             if basename == '__init__' or basename in _PLUGIN_FILTERS[self.package]:
 906                 continue
 907 
 908             if dedupe and basename in loaded_modules:
 909                 continue
 910             loaded_modules.add(basename)
 911 
 912             if path_only:
 913                 yield path
 914                 continue
 915 
 916             if path not in self._module_cache:
 917                 try:
 918                     if self.subdir in ('filter_plugins', 'test_plugins'):
 919                         # filter and test plugin files can contain multiple plugins
 920                         # they must have a unique python module name to prevent them from shadowing each other
 921                         full_name = '{0}_{1}'.format(abs(hash(path)), basename)
 922                     else:
 923                         full_name = basename
 924                     module = self._load_module_source(full_name, path)
 925                     self._load_config_defs(basename, module, path)
 926                 except Exception as e:
 927                     display.warning("Skipping plugin (%s) as it seems to be invalid: %s" % (path, to_text(e)))
 928                     continue
 929                 self._module_cache[path] = module
 930                 found_in_cache = False
 931 
 932             try:
 933                 obj = getattr(self._module_cache[path], self.class_name)
 934             except AttributeError as e:
 935                 display.warning("Skipping plugin (%s) as it seems to be invalid: %s" % (path, to_text(e)))
 936                 continue
 937 
 938             if self.base_class:
 939                 # The import path is hardcoded and should be the right place,
 940                 # so we are not expecting an ImportError.
 941                 module = __import__(self.package, fromlist=[self.base_class])
 942                 # Check whether this obj has the required base class.
 943                 try:
 944                     plugin_class = getattr(module, self.base_class)
 945                 except AttributeError:
 946                     continue
 947                 if not issubclass(obj, plugin_class):
 948                     continue
 949 
 950             self._display_plugin_load(self.class_name, basename, self._searched_paths, path, found_in_cache=found_in_cache, class_only=class_only)
 951 
 952             if not class_only:
 953                 try:
 954                     obj = obj(*args, **kwargs)
 955                 except TypeError as e:
 956                     display.warning("Skipping plugin (%s) as it seems to be incomplete: %s" % (path, to_text(e)))
 957 
 958             self._update_object(obj, basename, path)
 959             yield obj

 962 class Jinja2Loader(PluginLoader):
 963     """
 964     PluginLoader optimized for Jinja2 plugins
 965 
 966     The filter and test plugins are Jinja2 plugins encapsulated inside of our plugin format.
 967     The way the calling code is setup, we need to do a few things differently in the all() method
 968     """
 969     def find_plugin(self, name, collection_list=None):
 970         # Nothing using Jinja2Loader use this method.  We can't use the base class version because
 971         # we deduplicate differently than the base class
 972         if '.' in name:
 973             return super(Jinja2Loader, self).find_plugin(name, collection_list=collection_list)
 974 
 975         raise AnsibleError('No code should call find_plugin for Jinja2Loaders (Not implemented)')
 976 
 977     def get(self, name, *args, **kwargs):
 978         # Nothing using Jinja2Loader use this method.  We can't use the base class version because
 979         # we deduplicate differently than the base class
 980         if '.' in name:
 981             return super(Jinja2Loader, self).get(name, *args, **kwargs)
 982 
 983         raise AnsibleError('No code should call find_plugin for Jinja2Loaders (Not implemented)')
 984 
 985     def all(self, *args, **kwargs):
 986         """
 987         Differences with :meth:`PluginLoader.all`:
 988 
 989         * We do not deduplicate ansible plugin names.  This is because we don't care about our
 990           plugin names, here.  We care about the names of the actual jinja2 plugins which are inside
 991           of our plugins.
 992         * We reverse the order of the list of plugins compared to other PluginLoaders.  This is
 993           because of how calling code chooses to sync the plugins from the list.  It adds all the
 994           Jinja2 plugins from one of our Ansible plugins into a dict.  Then it adds the Jinja2
 995           plugins from the next Ansible plugin, overwriting any Jinja2 plugins that had the same
 996           name.  This is an encapsulation violation (the PluginLoader should not know about what
 997           calling code does with the data) but we're pushing the common code here.  We'll fix
 998           this in the future by moving more of the common code into this PluginLoader.
 999         * We return a list.  We could iterate the list instead but that's extra work for no gain because
1000           the API receiving this doesn't care.  It just needs an iterable
1001         """
1002         # We don't deduplicate ansible plugin names.  Instead, calling code deduplicates jinja2
1003         # plugin names.
1004         kwargs['_dedupe'] = False
1005 
1006         # We have to instantiate a list of all plugins so that we can reverse it.  We reverse it so
1007         # that calling code will deduplicate this correctly.
1008         plugins = list(super(Jinja2Loader, self).all(*args, **kwargs))
1009         plugins.reverse()
1010 
1011         return plugins

1014 def _load_plugin_filter():
1015     filters = defaultdict(frozenset)
1016     user_set = False
1017     if C.PLUGIN_FILTERS_CFG is None:
1018         filter_cfg = '/etc/ansible/plugin_filters.yml'
1019     else:
1020         filter_cfg = C.PLUGIN_FILTERS_CFG
1021         user_set = True
1022 
1023     if os.path.exists(filter_cfg):
1024         with open(filter_cfg, 'rb') as f:
1025             try:
1026                 filter_data = from_yaml(f.read())
1027             except Exception as e:
1028                 display.warning(u'The plugin filter file, {0} was not parsable.'
1029                                 u' Skipping: {1}'.format(filter_cfg, to_text(e)))
1030                 return filters
1031 
1032         try:
1033             version = filter_data['filter_version']
1034         except KeyError:
1035             display.warning(u'The plugin filter file, {0} was invalid.'
1036                             u' Skipping.'.format(filter_cfg))
1037             return filters
1038 
1039         # Try to convert for people specifying version as a float instead of string
1040         version = to_text(version)
1041         version = version.strip()
1042 
1043         if version == u'1.0':
1044             # Modules and action plugins share the same blacklist since the difference between the
1045             # two isn't visible to the users
1046             try:
1047                 filters['ansible.modules'] = frozenset(filter_data['module_blacklist'])
1048             except TypeError:
1049                 display.warning(u'Unable to parse the plugin filter file {0} as'
1050                                 u' module_blacklist is not a list.'
1051                                 u' Skipping.'.format(filter_cfg))
1052                 return filters
1053             filters['ansible.plugins.action'] = filters['ansible.modules']
1054         else:
1055             display.warning(u'The plugin filter file, {0} was a version not recognized by this'
1056                             u' version of Ansible. Skipping.'.format(filter_cfg))
1057     else:
1058         if user_set:
1059             display.warning(u'The plugin filter file, {0} does not exist.'
1060                             u' Skipping.'.format(filter_cfg))
1061 
1062     # Specialcase the stat module as Ansible can run very few things if stat is blacklisted.
1063     if 'stat' in filters['ansible.modules']:
1064         raise AnsibleError('The stat module was specified in the module blacklist file, {0}, but'
1065                            ' Ansible will not function without the stat module.  Please remove stat'
1066                            ' from the blacklist.'.format(to_native(filter_cfg)))
1067     return filters

1070 # since we don't want the actual collection loader understanding metadata, we'll do it in an event handler
1071 def _on_collection_load_handler(collection_name, collection_path):
1072     display.vvvv(to_text('Loading collection {0} from {1}'.format(collection_name, collection_path)))
1073 
1074     collection_meta = _get_collection_metadata(collection_name)
1075 
1076     try:
1077         if not _does_collection_support_ansible_version(collection_meta.get('requires_ansible', ''), ansible_version):
1078             mismatch_behavior = C.config.get_config_value('COLLECTIONS_ON_ANSIBLE_VERSION_MISMATCH')
1079             message = 'Collection {0} does not support Ansible version {1}'.format(collection_name, ansible_version)
1080             if mismatch_behavior == 'warning':
1081                 display.warning(message)
1082             elif mismatch_behavior == 'error':
1083                 raise AnsibleCollectionUnsupportedVersionError(message)
1084     except AnsibleError:
1085         raise
1086     except Exception as ex:
1087         display.warning('Error parsing collection metadata requires_ansible value from collection {0}: {1}'.format(collection_name, ex))

1090 def _does_collection_support_ansible_version(requirement_string, ansible_version):
1091     if not requirement_string:
1092         return True
1093 
1094     if not SpecifierSet:
1095         display.warning('packaging Python module unavailable; unable to validate collection Ansible version requirements')
1096         return True
1097 
1098     ss = SpecifierSet(requirement_string)
1099 
1100     # ignore prerelease/postrelease/beta/dev flags for simplicity
1101     base_ansible_version = Version(ansible_version).base_version
1102 
1103     return ss.contains(base_ansible_version)

1108         display.warning('AnsibleCollectionFinder has already been configured')
1109         return

1111     finder = _AnsibleCollectionFinder(C.config.get_config_value('COLLECTIONS_PATHS'), C.config.get_config_value('COLLECTIONS_SCAN_SYS_PATH'))
1112     finder._install()
1113 
1114     # this should succeed now
1115     AnsibleCollectionConfig.on_collection_load += _on_collection_load_handler
