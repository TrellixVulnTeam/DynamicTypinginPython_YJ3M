  45 def ensure_ncclient(func):
  46     @wraps(func)
  47     def wrapped(self, *args, **kwargs):
  48         if not HAS_NCCLIENT:
  49             raise AnsibleError("%s: %s" % (missing_required_lib('ncclient'), to_native(NCCLIENT_IMP_ERR)))
  50         return func(self, *args, **kwargs)
  51     return wrapped

  54 class NetconfBase(AnsiblePlugin):
  55     """
  56     A base class for implementing Netconf connections
  57 
  58     .. note:: Unlike most of Ansible, nearly all strings in
  59         :class:`TerminalBase` plugins are byte strings.  This is because of
  60         how close to the underlying platform these plugins operate.  Remember
  61         to mark literal strings as byte string (``b"string"``) and to use
  62         :func:`~ansible.module_utils._text.to_bytes` and
  63         :func:`~ansible.module_utils._text.to_text` to avoid unexpected
  64         problems.
  65 
  66         List of supported rpc's:
  67             :get: Retrieves running configuration and device state information
  68             :get_config: Retrieves the specified configuration from the device
  69             :edit_config: Loads the specified commands into the remote device
  70             :commit: Load configuration from candidate to running
  71             :discard_changes: Discard changes to candidate datastore
  72             :validate: Validate the contents of the specified configuration.
  73             :lock: Allows the client to lock the configuration system of a device.
  74             :unlock: Release a configuration lock, previously obtained with the lock operation.
  75             :copy_config: create or replace an entire configuration datastore with the contents of another complete
  76                           configuration datastore.
  77             :get-schema: Retrieves the required schema from the device
  78             :get_capabilities: Retrieves device information and supported rpc methods
  79 
  80             For JUNOS:
  81             :execute_rpc: RPC to be execute on remote device
  82             :load_configuration: Loads given configuration on device
  83 
  84         Note: rpc support depends on the capabilites of remote device.
  85 
  86         :returns: Returns output received from remote device as byte string
  87         Note: the 'result' or 'error' from response should to be converted to object
  88               of ElementTree using 'fromstring' to parse output as xml doc
  89 
  90               'get_capabilities()' returns 'result' as a json string.
  91 
  92             Usage:
  93             from ansible.module_utils.connection import Connection
  94 
  95             conn = Connection()
  96             data = conn.execute_rpc(rpc)
  97             reply = fromstring(reply)
  98 
  99             data = conn.get_capabilities()
 100             json.loads(data)
 101 
 102             conn.load_configuration(config=[''set system ntp server 1.1.1.1''], action='set', format='text')
 103     """
 104

 108     def __init__(self, connection):
 109         super(NetconfBase, self).__init__()

 116     def rpc(self, name):
 117         """
 118         RPC to be execute on remote device
 119         :param name: Name of rpc in string format
 120         :return: Received rpc response from remote host
 121         """
 122         try:
 123             obj = to_ele(name)
 124             resp = self.m.rpc(obj)
 125             return resp.data_xml if hasattr(resp, 'data_xml') else resp.xml
 126         except RPCError as exc:
 127             msg = exc.xml
 128             raise Exception(to_xml(msg))
 129 
 130     def get_config(self, source=None, filter=None):
 131         """
 132         Retrieve all or part of a specified configuration
 133         (by default entire configuration is retrieved).
 134         :param source: Name of the configuration datastore being queried, defaults to running datastore
 135         :param filter: This argument specifies the portion of the configuration data to retrieve
 136         :return: Returns xml string containing the RPC response received from remote host
 137         """
 138         if isinstance(filter, list):
 139             filter = tuple(filter)
 140 
 141         if not source:
 142             source = 'running'
 143         resp = self.m.get_config(source=source, filter=filter)
 144         return resp.data_xml if hasattr(resp, 'data_xml') else resp.xml
 145 
 146     def get(self, filter=None, with_defaults=None):
 147         """
 148         Retrieve device configuration and state information.
 149         :param filter: This argument specifies the portion of the state data to retrieve
 150                        (by default entire state data is retrieved)
 151         :param with_defaults: defines an explicit method of retrieving default values
 152                               from the configuration
 153         :return: Returns xml string containing the RPC response received from remote host
 154         """
 155         if isinstance(filter, list):
 156             filter = tuple(filter)
 157         resp = self.m.get(filter=filter, with_defaults=with_defaults)
 158         response = resp.data_xml if hasattr(resp, 'data_xml') else resp.xml
 159         return response

 161     def edit_config(self, config=None, format='xml', target='candidate', default_operation=None, test_option=None, error_option=None):
 162         """
 163         Loads all or part of the specified *config* to the *target* configuration datastore.
 164         :param config: Is the configuration, which must be rooted in the `config` element.
 165                        It can be specified either as a string or an :class:`~xml.etree.ElementTree.Element`.
 166         :param format: The format of configuration eg. xml, text
 167         :param target: Is the name of the configuration datastore being edited
 168         :param default_operation: If specified must be one of { `"merge"`, `"replace"`, or `"none"` }
 169         :param test_option: If specified must be one of { `"test_then_set"`, `"set"` }
 170         :param error_option: If specified must be one of { `"stop-on-error"`, `"continue-on-error"`, `"rollback-on-error"` }
 171                              The `"rollback-on-error"` *error_option* depends on the `:rollback-on-error` capability.
 172         :return: Returns xml string containing the RPC response received from remote host
 173         """
 174         if config is None:
 175             raise ValueError('config value must be provided')
 176         resp = self.m.edit_config(config, format=format, target=target, default_operation=default_operation, test_option=test_option,
 177                                   error_option=error_option)
 178         return resp.data_xml if hasattr(resp, 'data_xml') else resp.xml

 201     def dispatch(self, rpc_command=None, source=None, filter=None):
 202         """
 203         Execute rpc on the remote device eg. dispatch('clear-arp-table')
 204         :param rpc_command: specifies rpc command to be dispatched either in plain text or in xml element format (depending on command)
 205         :param source: name of the configuration datastore being queried
 206         :param filter: specifies the portion of the configuration to retrieve (by default entire configuration is retrieved)
 207         :return: Returns xml string containing the RPC response received from remote host
 208         """
 209         if rpc_command is None:
 210             raise ValueError('rpc_command value must be provided')
 211 
 212         resp = self.m.dispatch(fromstring(rpc_command), source=source, filter=filter)
 213 
 214         if isinstance(resp, NCElement):
 215             # In case xml reply is transformed or namespace is removed in
 216             # ncclient device specific handler return modified xml response
 217             result = resp.data_xml
 218         elif hasattr(resp, 'data_ele') and resp.data_ele:
 219             # if data node is present in xml response return the xml string
 220             # with data node as root
 221             result = resp.data_xml
 222         else:
 223             # return raw xml string received from host with rpc-reply as the root node
 224             result = resp.xml
 225 
 226         return result

 257     def commit(self, confirmed=False, timeout=None, persist=None):
 258         """
 259         Commit the candidate configuration as the device's new current configuration.
 260         Depends on the `:candidate` capability.
 261         A confirmed commit (i.e. if *confirmed* is `True`) is reverted if there is no
 262         followup commit within the *timeout* interval. If no timeout is specified the
 263         confirm timeout defaults to 600 seconds (10 minutes).
 264         A confirming commit may have the *confirmed* parameter but this is not required.
 265         Depends on the `:confirmed-commit` capability.
 266         :param confirmed: whether this is a confirmed commit
 267         :param timeout: specifies the confirm timeout in seconds
 268         :param persist: make the confirmed commit survive a session termination,
 269                         and set a token on the ongoing confirmed commit
 270         :return: Returns xml string containing the RPC response received from remote host
 271         """
 272         resp = self.m.commit(confirmed=confirmed, timeout=timeout, persist=persist)
 273         return resp.data_xml if hasattr(resp, 'data_xml') else resp.xml

 342     def get_device_operations(self, server_capabilities):
 343         """
 344         Retrieve remote host capability from Netconf server hello message.
 345         :param server_capabilities: Server capabilities received during Netconf session initialization
 346         :return: Remote host capabilities in dictionary format
 347         """
 348         operations = {}
 349         capabilities = '\n'.join(server_capabilities)
 350         operations['supports_commit'] = ':candidate' in capabilities
 351         operations['supports_defaults'] = ':with-defaults' in capabilities
 352         operations['supports_confirm_commit'] = ':confirmed-commit' in capabilities
 353         operations['supports_startup'] = ':startup' in capabilities
 354         operations['supports_xpath'] = ':xpath' in capabilities
 355         operations['supports_writable_running'] = ':writable-running' in capabilities
 356         operations['supports_validate'] = ':writable-validate' in capabilities
 357 
 358         operations['lock_datastore'] = []
 359         if operations['supports_writable_running']:
 360             operations['lock_datastore'].append('running')
 361 
 362         if operations['supports_commit']:
 363             operations['lock_datastore'].append('candidate')
 364 
 365         if operations['supports_startup']:
 366             operations['lock_datastore'].append('startup')
 367 
 368         operations['supports_lock'] = bool(operations['lock_datastore'])
 369 
 370         return operations
