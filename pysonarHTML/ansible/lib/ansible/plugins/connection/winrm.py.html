 221     def _build_winrm_kwargs(self):
 222         # this used to be in set_options, as win_reboot needs to be able to
 223         # override the conn timeout, we need to be able to build the args
 224         # after setting individual options. This is called by _connect before
 225         # starting the WinRM connection
 226         self._winrm_host = self.get_option('remote_addr')
 227         self._winrm_user = self.get_option('remote_user')
 228         self._winrm_pass = self.get_option('remote_password')
 229 
 230         self._winrm_port = self.get_option('port')
 231 
 232         self._winrm_scheme = self.get_option('scheme')
 233         # old behaviour, scheme should default to http if not set and the port
 234         # is 5985 otherwise https
 235         if self._winrm_scheme is None:
 236             self._winrm_scheme = 'http' if self._winrm_port == 5985 else 'https'
 237 
 238         self._winrm_path = self.get_option('path')
 239         self._kinit_cmd = self.get_option('kerberos_command')
 240         self._winrm_transport = self.get_option('transport')
 241         self._winrm_connection_timeout = self.get_option('connection_timeout')
 242 
 243         if hasattr(winrm, 'FEATURE_SUPPORTED_AUTHTYPES'):
 244             self._winrm_supported_authtypes = set(winrm.FEATURE_SUPPORTED_AUTHTYPES)
 245         else:
 246             # for legacy versions of pywinrm, use the values we know are supported
 247             self._winrm_supported_authtypes = set(['plaintext', 'ssl', 'kerberos'])
 248 
 249         # calculate transport if needed
 250         if self._winrm_transport is None or self._winrm_transport[0] is None:
 251             # TODO: figure out what we want to do with auto-transport selection in the face of NTLM/Kerb/CredSSP/Cert/Basic
 252             transport_selector = ['ssl'] if self._winrm_scheme == 'https' else ['plaintext']
 253 
 254             if HAVE_KERBEROS and ((self._winrm_user and '@' in self._winrm_user)):
 255                 self._winrm_transport = ['kerberos'] + transport_selector
 256             else:
 257                 self._winrm_transport = transport_selector
 258 
 259         unsupported_transports = set(self._winrm_transport).difference(self._winrm_supported_authtypes)
 260 
 261         if unsupported_transports:
 262             raise AnsibleError('The installed version of WinRM does not support transport(s) %s' %
 263                                to_native(list(unsupported_transports), nonstring='simplerepr'))
 264 
 265         # if kerberos is among our transports and there's a password specified, we're managing the tickets
 266         kinit_mode = self.get_option('kerberos_mode')
 267         if kinit_mode is None:
 268             # HACK: ideally, remove multi-transport stuff
 269             self._kerb_managed = "kerberos" in self._winrm_transport and (self._winrm_pass is not None and self._winrm_pass != "")
 270         elif kinit_mode == "managed":
 271             self._kerb_managed = True
 272         elif kinit_mode == "manual":
 273             self._kerb_managed = False
 274 
 275         # arg names we're going passing directly
 276         internal_kwarg_mask = set(['self', 'endpoint', 'transport', 'username', 'password', 'scheme', 'path', 'kinit_mode', 'kinit_cmd'])
 277 
 278         self._winrm_kwargs = dict(username=self._winrm_user, password=self._winrm_pass)
 279         argspec = getargspec(Protocol.__init__)
 280         supported_winrm_args = set(argspec.args)
 281         supported_winrm_args.update(internal_kwarg_mask)
 282         passed_winrm_args = set([v.replace('ansible_winrm_', '') for v in self.get_option('_extras')])
 283         unsupported_args = passed_winrm_args.difference(supported_winrm_args)
 284 
 285         # warn for kwargs unsupported by the installed version of pywinrm
 286         for arg in unsupported_args:
 287             display.warning("ansible_winrm_{0} unsupported by pywinrm (is an up-to-date version of pywinrm installed?)".format(arg))
 288 
 289         # pass through matching extras, excluding the list we want to treat specially
 290         for arg in passed_winrm_args.difference(internal_kwarg_mask).intersection(supported_winrm_args):
 291             self._winrm_kwargs[arg] = self.get_option('_extras')['ansible_winrm_%s' % arg]

 293     # Until pykerberos has enough goodies to implement a rudimentary kinit/klist, simplest way is to let each connection
 294     # auth itself with a private CCACHE.
 295     def _kerb_auth(self, principal, password):
 296         if password is None:
 297             password = ""
 298 
 299         self._kerb_ccache = tempfile.NamedTemporaryFile()
 300         display.vvvvv("creating Kerberos CC at %s" % self._kerb_ccache.name)
 301         krb5ccname = "FILE:%s" % self._kerb_ccache.name
 302         os.environ["KRB5CCNAME"] = krb5ccname
 303         krb5env = dict(KRB5CCNAME=krb5ccname)
 304 
 305         # Stores various flags to call with kinit, these could be explicit args set by 'ansible_winrm_kinit_args' OR
 306         # '-f' if kerberos delegation is requested (ansible_winrm_kerberos_delegation).
 307         kinit_cmdline = [self._kinit_cmd]
 308         kinit_args = self.get_option('kinit_args')
 309         if kinit_args:
 310             kinit_args = [to_text(a) for a in shlex.split(kinit_args) if a.strip()]
 311             kinit_cmdline.extend(kinit_args)
 312 
 313         elif boolean(self.get_option('_extras').get('ansible_winrm_kerberos_delegation', False)):
 314             kinit_cmdline.append('-f')
 315 
 316         kinit_cmdline.append(principal)
 317 
 318         # pexpect runs the process in its own pty so it can correctly send
 319         # the password as input even on MacOS which blocks subprocess from
 320         # doing so. Unfortunately it is not available on the built in Python
 321         # so we can only use it if someone has installed it
 322         if HAS_PEXPECT:
 323             proc_mechanism = "pexpect"
 324             command = kinit_cmdline.pop(0)
 325             password = to_text(password, encoding='utf-8',
 326                                errors='surrogate_or_strict')
 327 
 328             display.vvvv("calling kinit with pexpect for principal %s"
 329                          % principal)
 330             try:
 331                 child = pexpect.spawn(command, kinit_cmdline, timeout=60,
 332                                       env=krb5env, echo=False)
 333             except pexpect.ExceptionPexpect as err:
 334                 err_msg = "Kerberos auth failure when calling kinit cmd " \
 335                           "'%s': %s" % (command, to_native(err))
 336                 raise AnsibleConnectionFailure(err_msg)
 337 
 338             try:
 339                 child.expect(".*:")
 340                 child.sendline(password)
 341             except OSError as err:
 342                 # child exited before the pass was sent, Ansible will raise
 343                 # error based on the rc below, just display the error here
 344                 display.vvvv("kinit with pexpect raised OSError: %s"
 345                              % to_native(err))
 346 
 347             # technically this is the stdout + stderr but to match the
 348             # subprocess error checking behaviour, we will call it stderr
 349             stderr = child.read()
 350             child.wait()
 351             rc = child.exitstatus
 352         else:
 353             proc_mechanism = "subprocess"
 354             password = to_bytes(password, encoding='utf-8',
 355                                 errors='surrogate_or_strict')
 356 
 357             display.vvvv("calling kinit with subprocess for principal %s"
 358                          % principal)
 359             try:
 360                 p = subprocess.Popen(kinit_cmdline, stdin=subprocess.PIPE,
 361                                      stdout=subprocess.PIPE,
 362                                      stderr=subprocess.PIPE,
 363                                      env=krb5env)
 364 
 365             except OSError as err:
 366                 err_msg = "Kerberos auth failure when calling kinit cmd " \
 367                           "'%s': %s" % (self._kinit_cmd, to_native(err))
 368                 raise AnsibleConnectionFailure(err_msg)
 369 
 370             stdout, stderr = p.communicate(password + b'\n')
 371             rc = p.returncode != 0
 372 
 373         if rc != 0:
 374             # one last attempt at making sure the password does not exist
 375             # in the output
 376             exp_msg = to_native(stderr.strip())
 377             exp_msg = exp_msg.replace(to_native(password), "<redacted>")
 378 
 379             err_msg = "Kerberos auth failure for principal %s with %s: %s" \
 380                       % (principal, proc_mechanism, exp_msg)
 381             raise AnsibleConnectionFailure(err_msg)
 382 
 383         display.vvvvv("kinit succeeded for principal %s" % principal)

 385     def _winrm_connect(self):
 386         '''
 387         Establish a WinRM connection over HTTP/HTTPS.
 388         '''
 389         display.vvv("ESTABLISH WINRM CONNECTION FOR USER: %s on PORT %s TO %s" %
 390                     (self._winrm_user, self._winrm_port, self._winrm_host), host=self._winrm_host)
 391 
 392         winrm_host = self._winrm_host
 393         if HAS_IPADDRESS:
 394             display.debug("checking if winrm_host %s is an IPv6 address" % winrm_host)
 395             try:
 396                 ipaddress.IPv6Address(winrm_host)
 397             except ipaddress.AddressValueError:
 398                 pass
 399             else:
 400                 winrm_host = "[%s]" % winrm_host
 401 
 402         netloc = '%s:%d' % (winrm_host, self._winrm_port)
 403         endpoint = urlunsplit((self._winrm_scheme, netloc, self._winrm_path, '', ''))
 404         errors = []
 405         for transport in self._winrm_transport:
 406             if transport == 'kerberos':
 407                 if not HAVE_KERBEROS:
 408                     errors.append('kerberos: the python kerberos library is not installed')
 409                     continue
 410                 if self._kerb_managed:
 411                     self._kerb_auth(self._winrm_user, self._winrm_pass)
 412             display.vvvvv('WINRM CONNECT: transport=%s endpoint=%s' % (transport, endpoint), host=self._winrm_host)
 413             try:
 414                 winrm_kwargs = self._winrm_kwargs.copy()
 415                 if self._winrm_connection_timeout:
 416                     winrm_kwargs['operation_timeout_sec'] = self._winrm_connection_timeout
 417                     winrm_kwargs['read_timeout_sec'] = self._winrm_connection_timeout + 1
 418                 protocol = Protocol(endpoint, transport=transport, **winrm_kwargs)
 419 
 420                 # open the shell from connect so we know we're able to talk to the server
 421                 if not self.shell_id:
 422                     self.shell_id = protocol.open_shell(codepage=65001)  # UTF-8
 423                     display.vvvvv('WINRM OPEN SHELL: %s' % self.shell_id, host=self._winrm_host)
 424 
 425                 return protocol
 426             except Exception as e:
 427                 err_msg = to_text(e).strip()
 428                 if re.search(to_text(r'Operation\s+?timed\s+?out'), err_msg, re.I):
 429                     raise AnsibleError('the connection attempt timed out')
 430                 m = re.search(to_text(r'Code\s+?(\d{3})'), err_msg)
 431                 if m:
 432                     code = int(m.groups()[0])
 433                     if code == 401:
 434                         err_msg = 'the specified credentials were rejected by the server'
 435                     elif code == 411:
 436                         return protocol
 437                 errors.append(u'%s: %s' % (transport, err_msg))
 438                 display.vvvvv(u'WINRM CONNECTION ERROR: %s\n%s' % (err_msg, to_text(traceback.format_exc())), host=self._winrm_host)
 439         if errors:
 440             raise AnsibleConnectionFailure(', '.join(map(to_native, errors)))
 441         else:
 442             raise AnsibleError('No transport found for WinRM connection')

 444     def _winrm_send_input(self, protocol, shell_id, command_id, stdin, eof=False):
 445         rq = {'env:Envelope': protocol._get_soap_header(
 446             resource_uri='http://schemas.microsoft.com/wbem/wsman/1/windows/shell/cmd',
 447             action='http://schemas.microsoft.com/wbem/wsman/1/windows/shell/Send',
 448             shell_id=shell_id)}
 449         stream = rq['env:Envelope'].setdefault('env:Body', {}).setdefault('rsp:Send', {})\
 450             .setdefault('rsp:Stream', {})
 451         stream['@Name'] = 'stdin'
 452         stream['@CommandId'] = command_id
 453         stream['#text'] = base64.b64encode(to_bytes(stdin))
 454         if eof:
 455             stream['@End'] = 'true'
 456         protocol.send_message(xmltodict.unparse(rq))

 458     def _winrm_exec(self, command, args=(), from_exec=False, stdin_iterator=None):
 459         if not self.protocol:
 460             self.protocol = self._winrm_connect()
 461             self._connected = True
 462         if from_exec:
 463             display.vvvvv("WINRM EXEC %r %r" % (command, args), host=self._winrm_host)
 464         else:
 465             display.vvvvvv("WINRM EXEC %r %r" % (command, args), host=self._winrm_host)
 466         command_id = None
 467         try:
 468             stdin_push_failed = False
 469             command_id = self.protocol.run_command(self.shell_id, to_bytes(command), map(to_bytes, args), console_mode_stdin=(stdin_iterator is None))
 470 
 471             try:
 472                 if stdin_iterator:
 473                     for (data, is_last) in stdin_iterator:
 474                         self._winrm_send_input(self.protocol, self.shell_id, command_id, data, eof=is_last)
 475 
 476             except Exception as ex:
 477                 display.warning("ERROR DURING WINRM SEND INPUT - attempting to recover: %s %s"
 478                                 % (type(ex).__name__, to_text(ex)))
 479                 display.debug(traceback.format_exc())
 480                 stdin_push_failed = True
 481 
 482             # NB: this can hang if the receiver is still running (eg, network failed a Send request but the server's still happy).
 483             # FUTURE: Consider adding pywinrm status check/abort operations to see if the target is still running after a failure.
 484             resptuple = self.protocol.get_command_output(self.shell_id, command_id)
 485             # ensure stdout/stderr are text for py3
 486             # FUTURE: this should probably be done internally by pywinrm
 487             response = Response(tuple(to_text(v) if isinstance(v, binary_type) else v for v in resptuple))
 488 
 489             # TODO: check result from response and set stdin_push_failed if we have nonzero
 490             if from_exec:
 491                 display.vvvvv('WINRM RESULT %r' % to_text(response), host=self._winrm_host)
 492             else:
 493                 display.vvvvvv('WINRM RESULT %r' % to_text(response), host=self._winrm_host)
 494 
 495             display.vvvvvv('WINRM STDOUT %s' % to_text(response.std_out), host=self._winrm_host)
 496             display.vvvvvv('WINRM STDERR %s' % to_text(response.std_err), host=self._winrm_host)
 497 
 498             if stdin_push_failed:
 499                 # There are cases where the stdin input failed but the WinRM service still processed it. We attempt to
 500                 # see if stdout contains a valid json return value so we can ignore this error
 501                 try:
 502                     filtered_output, dummy = _filter_non_json_lines(response.std_out)
 503                     json.loads(filtered_output)
 504                 except ValueError:
 505                     # stdout does not contain a return response, stdin input was a fatal error
 506                     stderr = to_bytes(response.std_err, encoding='utf-8')
 507                     if stderr.startswith(b"#< CLIXML"):
 508                         stderr = _parse_clixml(stderr)
 509 
 510                     raise AnsibleError('winrm send_input failed; \nstdout: %s\nstderr %s'
 511                                        % (to_native(response.std_out), to_native(stderr)))
 512 
 513             return response
 514         except requests.exceptions.Timeout as exc:
 515             raise AnsibleConnectionFailure('winrm connection error: %s' % to_native(exc))
 516         finally:
 517             if command_id:
 518                 self.protocol.cleanup_command(self.shell_id, command_id)

 520     def _connect(self):
 521 
 522         if not HAS_WINRM:
 523             raise AnsibleError("winrm or requests is not installed: %s" % to_native(WINRM_IMPORT_ERR))
 524         elif not HAS_XMLTODICT:
 525             raise AnsibleError("xmltodict is not installed: %s" % to_native(XMLTODICT_IMPORT_ERR))
 526 
 527         super(Connection, self)._connect()
 528         if not self.protocol:
 529             self._build_winrm_kwargs()  # build the kwargs from the options set
 530             self.protocol = self._winrm_connect()
 531             self._connected = True
 532         return self

 534     def reset(self):
 535         self.protocol = None
 536         self.shell_id = None
 537         self._connect()

 539     def _wrapper_payload_stream(self, payload, buffer_size=200000):
 540         payload_bytes = to_bytes(payload)
 541         byte_count = len(payload_bytes)
 542         for i in range(0, byte_count, buffer_size):
 543             yield payload_bytes[i:i + buffer_size], i + buffer_size >= byte_count

 547         cmd_parts = self._shell._encode_script(cmd, as_list=True, strict_mode=False, preserve_rc=False)
 548 
 549         # TODO: display something meaningful here
 550         display.vvv("EXEC (via pipeline wrapper)")
 551 
 552         stdin_iterator = None
 553 
 554         if in_data:
 555             stdin_iterator = self._wrapper_payload_stream(in_data)
 556 
 557         result = self._winrm_exec(cmd_parts[0], cmd_parts[1:], from_exec=True, stdin_iterator=stdin_iterator)
 558 
 559         result.std_out = to_bytes(result.std_out)
 560         result.std_err = to_bytes(result.std_err)
 561 
 562         # parse just stderr from CLIXML output
 563         if result.std_err.startswith(b"#< CLIXML"):
 564             try:
 565                 result.std_err = _parse_clixml(result.std_err)
 566             except Exception:
 567                 # unsure if we're guaranteed a valid xml doc- use raw output in case of error
 568                 pass
 569 
 570         return (result.status_code, result.std_out, result.std_err)

 572     # FUTURE: determine buffer size at runtime via remote winrm config?
 573     def _put_file_stdin_iterator(self, in_path, out_path, buffer_size=250000):
 574         in_size = os.path.getsize(to_bytes(in_path, errors='surrogate_or_strict'))
 575         offset = 0
 576         with open(to_bytes(in_path, errors='surrogate_or_strict'), 'rb') as in_file:
 577             for out_data in iter((lambda: in_file.read(buffer_size)), b''):
 578                 offset += len(out_data)
 579                 self._display.vvvvv('WINRM PUT "%s" to "%s" (offset=%d size=%d)' % (in_path, out_path, offset, len(out_data)), host=self._winrm_host)
 580                 # yes, we're double-encoding over the wire in this case- we want to ensure that the data shipped to the end PS pipeline is still b64-encoded
 581                 b64_data = base64.b64encode(out_data) + b'\r\n'
 582                 # cough up the data, as well as an indicator if this is the last chunk so winrm_send knows to set the End signal
 583                 yield b64_data, (in_file.tell() == in_size)
 584 
 585             if offset == 0:  # empty file, return an empty buffer + eof to close it
 586                 yield "", True

 588     def put_file(self, in_path, out_path):
 589         super(Connection, self).put_file(in_path, out_path)
 590         out_path = self._shell._unquote(out_path)
 591         display.vvv('PUT "%s" TO "%s"' % (in_path, out_path), host=self._winrm_host)
 592         if not os.path.exists(to_bytes(in_path, errors='surrogate_or_strict')):
 593             raise AnsibleFileNotFound('file or module does not exist: "%s"' % to_native(in_path))
 594 
 595         script_template = u'''
 596             begin {{
 597                 $path = '{0}'
 598 
 599                 $DebugPreference = "Continue"
 600                 $ErrorActionPreference = "Stop"
 601                 Set-StrictMode -Version 2
 602 
 603                 $fd = [System.IO.File]::Create($path)
 604 
 605                 $sha1 = [System.Security.Cryptography.SHA1CryptoServiceProvider]::Create()
 606 
 607                 $bytes = @() #initialize for empty file case
 608             }}
 609             process {{
 610                $bytes = [System.Convert]::FromBase64String($input)
 611                $sha1.TransformBlock($bytes, 0, $bytes.Length, $bytes, 0) | Out-Null
 612                $fd.Write($bytes, 0, $bytes.Length)
 613             }}
 614             end {{
 615                 $sha1.TransformFinalBlock($bytes, 0, 0) | Out-Null
 616 
 617                 $hash = [System.BitConverter]::ToString($sha1.Hash).Replace("-", "").ToLowerInvariant()
 618 
 619                 $fd.Close()
 620 
 621                 Write-Output "{{""sha1"":""$hash""}}"
 622             }}
 623         '''
 624 
 625         script = script_template.format(self._shell._escape(out_path))
 626         cmd_parts = self._shell._encode_script(script, as_list=True, strict_mode=False, preserve_rc=False)
 627 
 628         result = self._winrm_exec(cmd_parts[0], cmd_parts[1:], stdin_iterator=self._put_file_stdin_iterator(in_path, out_path))
 629         # TODO: improve error handling
 630         if result.status_code != 0:
 631             raise AnsibleError(to_native(result.std_err))
 632 
 633         try:
 634             put_output = json.loads(result.std_out)
 635         except ValueError:
 636             # stdout does not contain a valid response
 637             stderr = to_bytes(result.std_err, encoding='utf-8')
 638             if stderr.startswith(b"#< CLIXML"):
 639                 stderr = _parse_clixml(stderr)
 640             raise AnsibleError('winrm put_file failed; \nstdout: %s\nstderr %s' % (to_native(result.std_out), to_native(stderr)))
 641 
 642         remote_sha1 = put_output.get("sha1")
 643         if not remote_sha1:
 644             raise AnsibleError("Remote sha1 was not returned")
 645 
 646         local_sha1 = secure_hash(in_path)
 647 
 648         if not remote_sha1 == local_sha1:
 649             raise AnsibleError("Remote sha1 hash {0} does not match local hash {1}".format(to_native(remote_sha1), to_native(local_sha1)))

 651     def fetch_file(self, in_path, out_path):
 652         super(Connection, self).fetch_file(in_path, out_path)
 653         in_path = self._shell._unquote(in_path)
 654         out_path = out_path.replace('\\', '/')
 655         # consistent with other connection plugins, we assume the caller has created the target dir
 656         display.vvv('FETCH "%s" TO "%s"' % (in_path, out_path), host=self._winrm_host)
 657         buffer_size = 2**19  # 0.5MB chunks
 658         out_file = None
 659         try:
 660             offset = 0
 661             while True:
 662                 try:
 663                     script = '''
 664                         $path = "%(path)s"
 665                         If (Test-Path -Path $path -PathType Leaf)
 666                         {
 667                             $buffer_size = %(buffer_size)d
 668                             $offset = %(offset)d
 669 
 670                             $stream = New-Object -TypeName IO.FileStream($path, [IO.FileMode]::Open, [IO.FileAccess]::Read, [IO.FileShare]::ReadWrite)
 671                             $stream.Seek($offset, [System.IO.SeekOrigin]::Begin) > $null
 672                             $buffer = New-Object -TypeName byte[] $buffer_size
 673                             $bytes_read = $stream.Read($buffer, 0, $buffer_size)
 674                             if ($bytes_read -gt 0) {
 675                                 $bytes = $buffer[0..($bytes_read - 1)]
 676                                 [System.Convert]::ToBase64String($bytes)
 677                             }
 678                             $stream.Close() > $null
 679                         }
 680                         ElseIf (Test-Path -Path $path -PathType Container)
 681                         {
 682                             Write-Host "[DIR]";
 683                         }
 684                         Else
 685                         {
 686                             Write-Error "$path does not exist";
 687                             Exit 1;
 688                         }
 689                     ''' % dict(buffer_size=buffer_size, path=self._shell._escape(in_path), offset=offset)
 690                     display.vvvvv('WINRM FETCH "%s" to "%s" (offset=%d)' % (in_path, out_path, offset), host=self._winrm_host)
 691                     cmd_parts = self._shell._encode_script(script, as_list=True, preserve_rc=False)
 692                     result = self._winrm_exec(cmd_parts[0], cmd_parts[1:])
 693                     if result.status_code != 0:
 694                         raise IOError(to_native(result.std_err))
 695                     if result.std_out.strip() == '[DIR]':
 696                         data = None
 697                     else:
 698                         data = base64.b64decode(result.std_out.strip())
 699                     if data is None:
 700                         break
 701                     else:
 702                         if not out_file:
 703                             # If out_path is a directory and we're expecting a file, bail out now.
 704                             if os.path.isdir(to_bytes(out_path, errors='surrogate_or_strict')):
 705                                 break
 706                             out_file = open(to_bytes(out_path, errors='surrogate_or_strict'), 'wb')
 707                         out_file.write(data)
 708                         if len(data) < buffer_size:
 709                             break
 710                         offset += len(data)
 711                 except Exception:
 712                     traceback.print_exc()
 713                     raise AnsibleError('failed to transfer file to "%s"' % to_native(out_path))
 714         finally:
 715             if out_file:
 716                 out_file.close()

 718     def close(self):
 719         if self.protocol and self.shell_id:
 720             display.vvvvv('WINRM CLOSE SHELL: %s' % self.shell_id, host=self._winrm_host)
 721             self.protocol.close_shell(self.shell_id)
 722         self.shell_id = None
 723         self.protocol = None
 724         self._connected = False
