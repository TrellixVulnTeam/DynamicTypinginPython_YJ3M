 329 def _handle_error(remaining_retries, command, return_tuple, no_log, host, display=display):
 330 
 331     # sshpass errors
 332     if command == b'sshpass':
 333         # Error 5 is invalid/incorrect password. Raise an exception to prevent retries from locking the account.
 334         if return_tuple[0] == 5:
 335             msg = 'Invalid/incorrect username/password. Skipping remaining {0} retries to prevent account lockout:'.format(remaining_retries)
 336             if remaining_retries <= 0:
 337                 msg = 'Invalid/incorrect password:'
 338             if no_log:
 339                 msg = '{0} <error censored due to no log>'.format(msg)
 340             else:
 341                 msg = '{0} {1}'.format(msg, to_native(return_tuple[2]).rstrip())
 342             raise AnsibleAuthenticationFailure(msg)
 343 
 344         # sshpass returns codes are 1-6. We handle 5 previously, so this catches other scenarios.
 345         # No exception is raised, so the connection is retried - except when attempting to use
 346         # sshpass_prompt with an sshpass that won't let us pass -P, in which case we fail loudly.
 347         elif return_tuple[0] in [1, 2, 3, 4, 6]:
 348             msg = 'sshpass error:'
 349             if no_log:
 350                 msg = '{0} <error censored due to no log>'.format(msg)
 351             else:
 352                 details = to_native(return_tuple[2]).rstrip()
 353                 if "sshpass: invalid option -- 'P'" in details:
 354                     details = 'Installed sshpass version does not support customized password prompts. ' \
 355                               'Upgrade sshpass to use sshpass_prompt, or otherwise switch to ssh keys.'
 356                     raise AnsibleError('{0} {1}'.format(msg, details))
 357                 msg = '{0} {1}'.format(msg, details)
 358 
 359     if return_tuple[0] == 255:
 360         SSH_ERROR = True
 361         for signature in b_NOT_SSH_ERRORS:
 362             if signature in return_tuple[1]:
 363                 SSH_ERROR = False
 364                 break
 365 
 366         if SSH_ERROR:
 367             msg = "Failed to connect to the host via ssh:"
 368             if no_log:
 369                 msg = '{0} <error censored due to no log>'.format(msg)
 370             else:
 371                 msg = '{0} {1}'.format(msg, to_native(return_tuple[2]).rstrip())
 372             raise AnsibleConnectionFailure(msg)
 373 
 374     # For other errors, no exception is raised so the connection is retried and we only log the messages
 375     if 1 <= return_tuple[0] <= 254:
 376         msg = u"Failed to connect to the host via ssh:"
 377         if no_log:
 378             msg = u'{0} <error censored due to no log>'.format(msg)
 379         else:
 380             msg = u'{0} {1}'.format(msg, to_text(return_tuple[2]).rstrip())
 381         display.vvv(msg, host=host)

 384 def _ssh_retry(func):
 385     """
 386     Decorator to retry ssh/scp/sftp in the case of a connection failure
 387 
 388     Will retry if:
 389     * an exception is caught
 390     * ssh returns 255
 391     Will not retry if
 392     * sshpass returns 5 (invalid password, to prevent account lockouts)
 393     * remaining_tries is < 2
 394     * retries limit reached
 395     """
 396     @wraps(func)
 397     def wrapped(self, *args, **kwargs):
 398         remaining_tries = int(C.ANSIBLE_SSH_RETRIES) + 1
 399         cmd_summary = u"%s..." % to_text(args[0])
 400         conn_password = self.get_option('password') or self._play_context.password
 401         for attempt in range(remaining_tries):
 402             cmd = args[0]
 403             if attempt != 0 and conn_password and isinstance(cmd, list):
 404                 # If this is a retry, the fd/pipe for sshpass is closed, and we need a new one
 405                 self.sshpass_pipe = os.pipe()
 406                 cmd[1] = b'-d' + to_bytes(self.sshpass_pipe[0], nonstring='simplerepr', errors='surrogate_or_strict')
 407 
 408             try:
 409                 try:
 410                     return_tuple = func(self, *args, **kwargs)
 411                     if self._play_context.no_log:
 412                         display.vvv(u'rc=%s, stdout and stderr censored due to no log' % return_tuple[0], host=self.host)
 413                     else:
 414                         display.vvv(return_tuple, host=self.host)
 415                     # 0 = success
 416                     # 1-254 = remote command return code
 417                     # 255 could be a failure from the ssh command itself
 418                 except (AnsibleControlPersistBrokenPipeError):
 419                     # Retry one more time because of the ControlPersist broken pipe (see #16731)
 420                     cmd = args[0]
 421                     if conn_password and isinstance(cmd, list):
 422                         # This is a retry, so the fd/pipe for sshpass is closed, and we need a new one
 423                         self.sshpass_pipe = os.pipe()
 424                         cmd[1] = b'-d' + to_bytes(self.sshpass_pipe[0], nonstring='simplerepr', errors='surrogate_or_strict')
 425                     display.vvv(u"RETRYING BECAUSE OF CONTROLPERSIST BROKEN PIPE")
 426                     return_tuple = func(self, *args, **kwargs)
 427 
 428                 remaining_retries = remaining_tries - attempt - 1
 429                 _handle_error(remaining_retries, cmd[0], return_tuple, self._play_context.no_log, self.host)
 430 
 431                 break
 432 
 433             # 5 = Invalid/incorrect password from sshpass
 434             except AnsibleAuthenticationFailure:
 435                 # Raising this exception, which is subclassed from AnsibleConnectionFailure, prevents further retries
 436                 raise
 437 
 438             except (AnsibleConnectionFailure, Exception) as e:
 439 
 440                 if attempt == remaining_tries - 1:
 441                     raise
 442                 else:
 443                     pause = 2 ** attempt - 1
 444                     if pause > 30:
 445                         pause = 30
 446 
 447                     if isinstance(e, AnsibleConnectionFailure):
 448                         msg = u"ssh_retry: attempt: %d, ssh return code is 255. cmd (%s), pausing for %d seconds" % (attempt + 1, cmd_summary, pause)
 449                     else:
 450                         msg = (u"ssh_retry: attempt: %d, caught exception(%s) from cmd (%s), "
 451                                u"pausing for %d seconds" % (attempt + 1, to_text(e), cmd_summary, pause))
 452 
 453                     display.vv(msg, host=self.host)
 454 
 455                     time.sleep(pause)
 456                     continue
 457 
 458         return return_tuple
 459     return wrapped

 493     @staticmethod
 494     def _create_control_path(host, port, user, connection=None, pid=None):
 495         '''Make a hash for the controlpath based on con attributes'''
 496         pstring = '%s-%s-%s' % (host, port, user)
 497         if connection:
 498             pstring += '-%s' % connection
 499         if pid:
 500             pstring += '-%s' % to_text(pid)
 501         m = hashlib.sha1()
 502         m.update(to_bytes(pstring))
 503         digest = m.hexdigest()
 504         cpath = '%(directory)s/' + digest[:10]
 505         return cpath

 507     @staticmethod
 508     def _sshpass_available():
 509         global SSHPASS_AVAILABLE
 510 
 511         # We test once if sshpass is available, and remember the result. It
 512         # would be nice to use distutils.spawn.find_executable for this, but
 513         # distutils isn't always available; shutils.which() is Python3-only.
 514 
 515         if SSHPASS_AVAILABLE is None:
 516             try:
 517                 p = subprocess.Popen(["sshpass"], stdin=subprocess.PIPE, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
 518                 p.communicate()
 519                 SSHPASS_AVAILABLE = True
 520             except OSError:
 521                 SSHPASS_AVAILABLE = False
 522 
 523         return SSHPASS_AVAILABLE

 525     @staticmethod
 526     def _persistence_controls(b_command):
 527         '''
 528         Takes a command array and scans it for ControlPersist and ControlPath
 529         settings and returns two booleans indicating whether either was found.
 530         This could be smarter, e.g. returning false if ControlPersist is 'no',
 531         but for now we do it simple way.
 532         '''
 533 
 534         controlpersist = False
 535         controlpath = False
 536 
 537         for b_arg in (a.lower() for a in b_command):
 538             if b'controlpersist' in b_arg:
 539                 controlpersist = True
 540             elif b'controlpath' in b_arg:
 541                 controlpath = True
 542 
 543         return controlpersist, controlpath

 545     def _add_args(self, b_command, b_args, explanation):
 546         """
 547         Adds arguments to the ssh command and displays a caller-supplied explanation of why.
 548 
 549         :arg b_command: A list containing the command to add the new arguments to.
 550             This list will be modified by this method.
 551         :arg b_args: An iterable of new arguments to add.  This iterable is used
 552             more than once so it must be persistent (ie: a list is okay but a
 553             StringIO would not)
 554         :arg explanation: A text string containing explaining why the arguments
 555             were added.  It will be displayed with a high enough verbosity.
 556         .. note:: This function does its work via side-effect.  The b_command list has the new arguments appended.
 557         """
 558         display.vvvvv(u'SSH: %s: (%s)' % (explanation, ')('.join(to_text(a) for a in b_args)), host=self._play_context.remote_addr)
 559         b_command += b_args

 561     def _build_command(self, binary, *other_args):
 562         '''
 563         Takes a binary (ssh, scp, sftp) and optional extra arguments and returns
 564         a command line as an array that can be passed to subprocess.Popen.
 565         '''
 566 
 567         b_command = []
 568         conn_password = self.get_option('password') or self._play_context.password
 569 
 570         #
 571         # First, the command to invoke
 572         #
 573 
 574         # If we want to use password authentication, we have to set up a pipe to
 575         # write the password to sshpass.
 576 
 577         if conn_password:
 578             if not self._sshpass_available():
 579                 raise AnsibleError("to use the 'ssh' connection type with passwords, you must install the sshpass program")
 580 
 581             self.sshpass_pipe = os.pipe()
 582             b_command += [b'sshpass', b'-d' + to_bytes(self.sshpass_pipe[0], nonstring='simplerepr', errors='surrogate_or_strict')]
 583 
 584             password_prompt = self.get_option('sshpass_prompt')
 585             if password_prompt:
 586                 b_command += [b'-P', to_bytes(password_prompt, errors='surrogate_or_strict')]
 587 
 588         if binary == 'ssh':
 589             b_command += [to_bytes(self._play_context.ssh_executable, errors='surrogate_or_strict')]
 590         else:
 591             b_command += [to_bytes(binary, errors='surrogate_or_strict')]
 592 
 593         #
 594         # Next, additional arguments based on the configuration.
 595         #
 596 
 597         # sftp batch mode allows us to correctly catch failed transfers, but can
 598         # be disabled if the client side doesn't support the option. However,
 599         # sftp batch mode does not prompt for passwords so it must be disabled
 600         # if not using controlpersist and using sshpass
 601         if binary == 'sftp' and C.DEFAULT_SFTP_BATCH_MODE:
 602             if conn_password:
 603                 b_args = [b'-o', b'BatchMode=no']
 604                 self._add_args(b_command, b_args, u'disable batch mode for sshpass')
 605             b_command += [b'-b', b'-']
 606 
 607         if self._play_context.verbosity > 3:
 608             b_command.append(b'-vvv')
 609 
 610         #
 611         # Next, we add [ssh_connection]ssh_args from ansible.cfg.
 612         #
 613 
 614         ssh_args = self.get_option('ssh_args')
 615         if ssh_args:
 616             b_args = [to_bytes(a, errors='surrogate_or_strict') for a in
 617                       self._split_ssh_args(ssh_args)]
 618             self._add_args(b_command, b_args, u"ansible.cfg set ssh_args")
 619 
 620         # Now we add various arguments controlled by configuration file settings
 621         # (e.g. host_key_checking) or inventory variables (ansible_ssh_port) or
 622         # a combination thereof.
 623 
 624         if not C.HOST_KEY_CHECKING:
 625             b_args = (b"-o", b"StrictHostKeyChecking=no")
 626             self._add_args(b_command, b_args, u"ANSIBLE_HOST_KEY_CHECKING/host_key_checking disabled")
 627 
 628         if self._play_context.port is not None:
 629             b_args = (b"-o", b"Port=" + to_bytes(self._play_context.port, nonstring='simplerepr', errors='surrogate_or_strict'))
 630             self._add_args(b_command, b_args, u"ANSIBLE_REMOTE_PORT/remote_port/ansible_port set")
 631 
 632         key = self._play_context.private_key_file
 633         if key:
 634             b_args = (b"-o", b'IdentityFile="' + to_bytes(os.path.expanduser(key), errors='surrogate_or_strict') + b'"')
 635             self._add_args(b_command, b_args, u"ANSIBLE_PRIVATE_KEY_FILE/private_key_file/ansible_ssh_private_key_file set")
 636 
 637         if not conn_password:
 638             self._add_args(
 639                 b_command, (
 640                     b"-o", b"KbdInteractiveAuthentication=no",
 641                     b"-o", b"PreferredAuthentications=gssapi-with-mic,gssapi-keyex,hostbased,publickey",
 642                     b"-o", b"PasswordAuthentication=no"
 643                 ),
 644                 u"ansible_password/ansible_ssh_password not set"
 645             )
 646 
 647         user = self._play_context.remote_user
 648         if user:
 649             self._add_args(
 650                 b_command,
 651                 (b"-o", b'User="%s"' % to_bytes(self._play_context.remote_user, errors='surrogate_or_strict')),
 652                 u"ANSIBLE_REMOTE_USER/remote_user/ansible_user/user/-u set"
 653             )
 654 
 655         self._add_args(
 656             b_command,
 657             (b"-o", b"ConnectTimeout=" + to_bytes(self._play_context.timeout, errors='surrogate_or_strict', nonstring='simplerepr')),
 658             u"ANSIBLE_TIMEOUT/timeout set"
 659         )
 660 
 661         # Add in any common or binary-specific arguments from the PlayContext
 662         # (i.e. inventory or task settings or overrides on the command line).
 663 
 664         for opt in (u'ssh_common_args', u'{0}_extra_args'.format(binary)):
 665             attr = getattr(self._play_context, opt, None)
 666             if attr is not None:
 667                 b_args = [to_bytes(a, errors='surrogate_or_strict') for a in self._split_ssh_args(attr)]
 668                 self._add_args(b_command, b_args, u"PlayContext set %s" % opt)
 669 
 670         # Check if ControlPersist is enabled and add a ControlPath if one hasn't
 671         # already been set.
 672 
 673         controlpersist, controlpath = self._persistence_controls(b_command)
 674 
 675         if controlpersist:
 676             self._persistent = True
 677 
 678             if not controlpath:
 679                 cpdir = unfrackpath(self.control_path_dir)
 680                 b_cpdir = to_bytes(cpdir, errors='surrogate_or_strict')
 681 
 682                 # The directory must exist and be writable.
 683                 makedirs_safe(b_cpdir, 0o700)
 684                 if not os.access(b_cpdir, os.W_OK):
 685                     raise AnsibleError("Cannot write to ControlPath %s" % to_native(cpdir))
 686 
 687                 if not self.control_path:
 688                     self.control_path = self._create_control_path(
 689                         self.host,
 690                         self.port,
 691                         self.user
 692                     )
 693                 b_args = (b"-o", b"ControlPath=" + to_bytes(self.control_path % dict(directory=cpdir), errors='surrogate_or_strict'))
 694                 self._add_args(b_command, b_args, u"found only ControlPersist; added ControlPath")
 695 
 696         # Finally, we add any caller-supplied extras.
 697         if other_args:
 698             b_command += [to_bytes(a) for a in other_args]
 699 
 700         return b_command

 702     def _send_initial_data(self, fh, in_data, ssh_process):
 703         '''
 704         Writes initial data to the stdin filehandle of the subprocess and closes
 705         it. (The handle must be closed; otherwise, for example, "sftp -b -" will
 706         just hang forever waiting for more commands.)
 707         '''
 708 
 709         display.debug(u'Sending initial data')
 710 
 711         try:
 712             fh.write(to_bytes(in_data))
 713             fh.close()
 714         except (OSError, IOError) as e:
 715             # The ssh connection may have already terminated at this point, with a more useful error
 716             # Only raise AnsibleConnectionFailure if the ssh process is still alive
 717             time.sleep(0.001)
 718             ssh_process.poll()
 719             if getattr(ssh_process, 'returncode', None) is None:
 720                 raise AnsibleConnectionFailure(
 721                     'Data could not be sent to remote host "%s". Make sure this host can be reached '
 722                     'over ssh: %s' % (self.host, to_native(e)), orig_exc=e
 723                 )
 724 
 725         display.debug(u'Sent initial data (%d bytes)' % len(in_data))

 730         """ Terminate a process, ignoring errors """
 731         try:
 732             p.terminate()
 733         except (OSError, IOError):
 734             pass

 736     # This is separate from _run() because we need to do the same thing for stdout
 737     # and stderr.
 738     def _examine_output(self, source, state, b_chunk, sudoable):
 739         '''
 740         Takes a string, extracts complete lines from it, tests to see if they
 741         are a prompt, error message, etc., and sets appropriate flags in self.
 742         Prompt and success lines are removed.
 743 
 744         Returns the processed (i.e. possibly-edited) output and the unprocessed
 745         remainder (to be processed with the next chunk) as strings.
 746         '''
 747 
 748         output = []
 749         for b_line in b_chunk.splitlines(True):
 750             display_line = to_text(b_line).rstrip('\r\n')
 751             suppress_output = False
 752 
 753             # display.debug("Examining line (source=%s, state=%s): '%s'" % (source, state, display_line))
 754             if self.become.expect_prompt() and self.become.check_password_prompt(b_line):
 755                 display.debug(u"become_prompt: (source=%s, state=%s): '%s'" % (source, state, display_line))
 756                 self._flags['become_prompt'] = True
 757                 suppress_output = True
 758             elif self.become.success and self.become.check_success(b_line):
 759                 display.debug(u"become_success: (source=%s, state=%s): '%s'" % (source, state, display_line))
 760                 self._flags['become_success'] = True
 761                 suppress_output = True
 762             elif sudoable and self.become.check_incorrect_password(b_line):
 763                 display.debug(u"become_error: (source=%s, state=%s): '%s'" % (source, state, display_line))
 764                 self._flags['become_error'] = True
 765             elif sudoable and self.become.check_missing_password(b_line):
 766                 display.debug(u"become_nopasswd_error: (source=%s, state=%s): '%s'" % (source, state, display_line))
 767                 self._flags['become_nopasswd_error'] = True
 768 
 769             if not suppress_output:
 770                 output.append(b_line)
 771 
 772         # The chunk we read was most likely a series of complete lines, but just
 773         # in case the last line was incomplete (and not a prompt, which we would
 774         # have removed from the output), we retain it to be processed with the
 775         # next chunk.
 776 
 777         remainder = b''
 778         if output and not output[-1].endswith(b'\n'):
 779             remainder = output[-1]
 780             output = output[:-1]
 781 
 782         return b''.join(output), remainder

 784     def _bare_run(self, cmd, in_data, sudoable=True, checkrc=True):
 785         '''
 786         Starts the command and communicates with it until it ends.
 787         '''
 788 
 789         # We don't use _shell.quote as this is run on the controller and independent from the shell plugin chosen
 790         display_cmd = u' '.join(shlex_quote(to_text(c)) for c in cmd)
 791         display.vvv(u'SSH: EXEC {0}'.format(display_cmd), host=self.host)
 792 
 793         # Start the given command. If we don't need to pipeline data, we can try
 794         # to use a pseudo-tty (ssh will have been invoked with -tt). If we are
 795         # pipelining data, or can't create a pty, we fall back to using plain
 796         # old pipes.
 797 
 798         p = None
 799 
 800         if isinstance(cmd, (text_type, binary_type)):
 801             cmd = to_bytes(cmd)
 802         else:
 803             cmd = list(map(to_bytes, cmd))
 804 
 805         conn_password = self.get_option('password') or self._play_context.password
 806 
 807         if not in_data:
 808             try:
 809                 # Make sure stdin is a proper pty to avoid tcgetattr errors
 810                 master, slave = pty.openpty()
 811                 if PY3 and conn_password:
 812                     # pylint: disable=unexpected-keyword-arg
 813                     p = subprocess.Popen(cmd, stdin=slave, stdout=subprocess.PIPE, stderr=subprocess.PIPE, pass_fds=self.sshpass_pipe)
 814                 else:
 815                     p = subprocess.Popen(cmd, stdin=slave, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
 816                 stdin = os.fdopen(master, 'wb', 0)
 817                 os.close(slave)
 818             except (OSError, IOError):
 819                 p = None
 820 
 821         if not p:
 822             try:
 823                 if PY3 and conn_password:
 824                     # pylint: disable=unexpected-keyword-arg
 825                     p = subprocess.Popen(cmd, stdin=subprocess.PIPE, stdout=subprocess.PIPE,
 826                                          stderr=subprocess.PIPE, pass_fds=self.sshpass_pipe)
 827                 else:
 828                     p = subprocess.Popen(cmd, stdin=subprocess.PIPE, stdout=subprocess.PIPE,
 829                                          stderr=subprocess.PIPE)
 830                 stdin = p.stdin
 831             except (OSError, IOError) as e:
 832                 raise AnsibleError('Unable to execute ssh command line on a controller due to: %s' % to_native(e))
 833 
 834         # If we are using SSH password authentication, write the password into
 835         # the pipe we opened in _build_command.
 836 
 837         if conn_password:
 838             os.close(self.sshpass_pipe[0])
 839             try:
 840                 os.write(self.sshpass_pipe[1], to_bytes(conn_password) + b'\n')
 841             except OSError as e:
 842                 # Ignore broken pipe errors if the sshpass process has exited.
 843                 if e.errno != errno.EPIPE or p.poll() is None:
 844                     raise
 845             os.close(self.sshpass_pipe[1])
 846 
 847         #
 848         # SSH state machine
 849         #
 850 
 851         # Now we read and accumulate output from the running process until it
 852         # exits. Depending on the circumstances, we may also need to write an
 853         # escalation password and/or pipelined input to the process.
 854 
 855         states = [
 856             'awaiting_prompt', 'awaiting_escalation', 'ready_to_send', 'awaiting_exit'
 857         ]
 858 
 859         # Are we requesting privilege escalation? Right now, we may be invoked
 860         # to execute sftp/scp with sudoable=True, but we can request escalation
 861         # only when using ssh. Otherwise we can send initial data straightaway.
 862 
 863         state = states.index('ready_to_send')
 864         if to_bytes(self.get_option('ssh_executable')) in cmd and sudoable:
 865             prompt = getattr(self.become, 'prompt', None)
 866             if prompt:
 867                 # We're requesting escalation with a password, so we have to
 868                 # wait for a password prompt.
 869                 state = states.index('awaiting_prompt')
 870                 display.debug(u'Initial state: %s: %s' % (states[state], to_text(prompt)))
 871             elif self.become and self.become.success:
 872                 # We're requesting escalation without a password, so we have to
 873                 # detect success/failure before sending any initial data.
 874                 state = states.index('awaiting_escalation')
 875                 display.debug(u'Initial state: %s: %s' % (states[state], to_text(self.become.success)))
 876 
 877         # We store accumulated stdout and stderr output from the process here,
 878         # but strip any privilege escalation prompt/confirmation lines first.
 879         # Output is accumulated into tmp_*, complete lines are extracted into
 880         # an array, then checked and removed or copied to stdout or stderr. We
 881         # set any flags based on examining the output in self._flags.
 882 
 883         b_stdout = b_stderr = b''
 884         b_tmp_stdout = b_tmp_stderr = b''
 885 
 886         self._flags = dict(
 887             become_prompt=False, become_success=False,
 888             become_error=False, become_nopasswd_error=False
 889         )
 890 
 891         # select timeout should be longer than the connect timeout, otherwise
 892         # they will race each other when we can't connect, and the connect
 893         # timeout usually fails
 894         timeout = 2 + self._play_context.timeout
 895         for fd in (p.stdout, p.stderr):
 896             fcntl.fcntl(fd, fcntl.F_SETFL, fcntl.fcntl(fd, fcntl.F_GETFL) | os.O_NONBLOCK)
 897 
 898         # TODO: bcoca would like to use SelectSelector() when open
 899         # filehandles is low, then switch to more efficient ones when higher.
 900         # select is faster when filehandles is low.
 901         selector = selectors.DefaultSelector()
 902         selector.register(p.stdout, selectors.EVENT_READ)
 903         selector.register(p.stderr, selectors.EVENT_READ)
 904 
 905         # If we can send initial data without waiting for anything, we do so
 906         # before we start polling
 907         if states[state] == 'ready_to_send' and in_data:
 908             self._send_initial_data(stdin, in_data, p)
 909             state += 1
 910 
 911         try:
 912             while True:
 913                 poll = p.poll()
 914                 events = selector.select(timeout)
 915 
 916                 # We pay attention to timeouts only while negotiating a prompt.
 917 
 918                 if not events:
 919                     # We timed out
 920                     if state <= states.index('awaiting_escalation'):
 921                         # If the process has already exited, then it's not really a
 922                         # timeout; we'll let the normal error handling deal with it.
 923                         if poll is not None:
 924                             break
 925                         self._terminate_process(p)
 926                         raise AnsibleError('Timeout (%ds) waiting for privilege escalation prompt: %s' % (timeout, to_native(b_stdout)))
 927 
 928                 # Read whatever output is available on stdout and stderr, and stop
 929                 # listening to the pipe if it's been closed.
 930 
 931                 for key, event in events:
 932                     if key.fileobj == p.stdout:
 933                         b_chunk = p.stdout.read()
 934                         if b_chunk == b'':
 935                             # stdout has been closed, stop watching it
 936                             selector.unregister(p.stdout)
 937                             # When ssh has ControlMaster (+ControlPath/Persist) enabled, the
 938                             # first connection goes into the background and we never see EOF
 939                             # on stderr. If we see EOF on stdout, lower the select timeout
 940                             # to reduce the time wasted selecting on stderr if we observe
 941                             # that the process has not yet existed after this EOF. Otherwise
 942                             # we may spend a long timeout period waiting for an EOF that is
 943                             # not going to arrive until the persisted connection closes.
 944                             timeout = 1
 945                         b_tmp_stdout += b_chunk
 946                         display.debug(u"stdout chunk (state=%s):\n>>>%s<<<\n" % (state, to_text(b_chunk)))
 947                     elif key.fileobj == p.stderr:
 948                         b_chunk = p.stderr.read()
 949                         if b_chunk == b'':
 950                             # stderr has been closed, stop watching it
 951                             selector.unregister(p.stderr)
 952                         b_tmp_stderr += b_chunk
 953                         display.debug("stderr chunk (state=%s):\n>>>%s<<<\n" % (state, to_text(b_chunk)))
 954 
 955                 # We examine the output line-by-line until we have negotiated any
 956                 # privilege escalation prompt and subsequent success/error message.
 957                 # Afterwards, we can accumulate output without looking at it.
 958 
 959                 if state < states.index('ready_to_send'):
 960                     if b_tmp_stdout:
 961                         b_output, b_unprocessed = self._examine_output('stdout', states[state], b_tmp_stdout, sudoable)
 962                         b_stdout += b_output
 963                         b_tmp_stdout = b_unprocessed
 964 
 965                     if b_tmp_stderr:
 966                         b_output, b_unprocessed = self._examine_output('stderr', states[state], b_tmp_stderr, sudoable)
 967                         b_stderr += b_output
 968                         b_tmp_stderr = b_unprocessed
 969                 else:
 970                     b_stdout += b_tmp_stdout
 971                     b_stderr += b_tmp_stderr
 972                     b_tmp_stdout = b_tmp_stderr = b''
 973 
 974                 # If we see a privilege escalation prompt, we send the password.
 975                 # (If we're expecting a prompt but the escalation succeeds, we
 976                 # didn't need the password and can carry on regardless.)
 977 
 978                 if states[state] == 'awaiting_prompt':
 979                     if self._flags['become_prompt']:
 980                         display.debug(u'Sending become_password in response to prompt')
 981                         become_pass = self.become.get_option('become_pass', playcontext=self._play_context)
 982                         stdin.write(to_bytes(become_pass, errors='surrogate_or_strict') + b'\n')
 983                         # On python3 stdin is a BufferedWriter, and we don't have a guarantee
 984                         # that the write will happen without a flush
 985                         stdin.flush()
 986                         self._flags['become_prompt'] = False
 987                         state += 1
 988                     elif self._flags['become_success']:
 989                         state += 1
 990 
 991                 # We've requested escalation (with or without a password), now we
 992                 # wait for an error message or a successful escalation.
 993 
 994                 if states[state] == 'awaiting_escalation':
 995                     if self._flags['become_success']:
 996                         display.vvv(u'Escalation succeeded')
 997                         self._flags['become_success'] = False
 998                         state += 1
 999                     elif self._flags['become_error']:
1000                         display.vvv(u'Escalation failed')
1001                         self._terminate_process(p)
1002                         self._flags['become_error'] = False
1003                         raise AnsibleError('Incorrect %s password' % self.become.name)
1004                     elif self._flags['become_nopasswd_error']:
1005                         display.vvv(u'Escalation requires password')
1006                         self._terminate_process(p)
1007                         self._flags['become_nopasswd_error'] = False
1008                         raise AnsibleError('Missing %s password' % self.become.name)
1009                     elif self._flags['become_prompt']:
1010                         # This shouldn't happen, because we should see the "Sorry,
1011                         # try again" message first.
1012                         display.vvv(u'Escalation prompt repeated')
1013                         self._terminate_process(p)
1014                         self._flags['become_prompt'] = False
1015                         raise AnsibleError('Incorrect %s password' % self.become.name)
1016 
1017                 # Once we're sure that the privilege escalation prompt, if any, has
1018                 # been dealt with, we can send any initial data and start waiting
1019                 # for output.
1020 
1021                 if states[state] == 'ready_to_send':
1022                     if in_data:
1023                         self._send_initial_data(stdin, in_data, p)
1024                     state += 1
1025 
1026                 # Now we're awaiting_exit: has the child process exited? If it has,
1027                 # and we've read all available output from it, we're done.
1028 
1029                 if poll is not None:
1030                     if not selector.get_map() or not events:
1031                         break
1032                     # We should not see further writes to the stdout/stderr file
1033                     # descriptors after the process has closed, set the select
1034                     # timeout to gather any last writes we may have missed.
1035                     timeout = 0
1036                     continue
1037 
1038                 # If the process has not yet exited, but we've already read EOF from
1039                 # its stdout and stderr (and thus no longer watching any file
1040                 # descriptors), we can just wait for it to exit.
1041 
1042                 elif not selector.get_map():
1043                     p.wait()
1044                     break
1045 
1046                 # Otherwise there may still be outstanding data to read.
1047         finally:
1048             selector.close()
1049             # close stdin, stdout, and stderr after process is terminated and
1050             # stdout/stderr are read completely (see also issues #848, #64768).
1051             stdin.close()
1052             p.stdout.close()
1053             p.stderr.close()
1054 
1055         if C.HOST_KEY_CHECKING:
1056             if cmd[0] == b"sshpass" and p.returncode == 6:
1057                 raise AnsibleError('Using a SSH password instead of a key is not possible because Host Key checking is enabled and sshpass does not support '
1058                                    'this.  Please add this host\'s fingerprint to your known_hosts file to manage this host.')
1059 
1060         controlpersisterror = b'Bad configuration option: ControlPersist' in b_stderr or b'unknown configuration option: ControlPersist' in b_stderr
1061         if p.returncode != 0 and controlpersisterror:
1062             raise AnsibleError('using -c ssh on certain older ssh versions may not support ControlPersist, set ANSIBLE_SSH_ARGS="" '
1063                                '(or ssh_args in [ssh_connection] section of the config file) before running again')
1064 
1065         # If we find a broken pipe because of ControlPersist timeout expiring (see #16731),
1066         # we raise a special exception so that we can retry a connection.
1067         controlpersist_broken_pipe = b'mux_client_hello_exchange: write packet: Broken pipe' in b_stderr
1068         if p.returncode == 255:
1069 
1070             additional = to_native(b_stderr)
1071             if controlpersist_broken_pipe:
1072                 raise AnsibleControlPersistBrokenPipeError('Data could not be sent because of ControlPersist broken pipe: %s' % additional)
1073 
1074             elif in_data and checkrc:
1075                 raise AnsibleConnectionFailure('Data could not be sent to remote host "%s". Make sure this host can be reached over ssh: %s'
1076                                                % (self.host, additional))
1077 
1078         return (p.returncode, b_stdout, b_stderr)

1086     @_ssh_retry
1087     def _file_transport_command(self, in_path, out_path, sftp_action):
1088         # scp and sftp require square brackets for IPv6 addresses, but
1089         # accept them for hostnames and IPv4 addresses too.
1090         host = '[%s]' % self.host
1091 
1092         smart_methods = ['sftp', 'scp', 'piped']
1093 
1094         # Windows does not support dd so we cannot use the piped method
1095         if getattr(self._shell, "_IS_WINDOWS", False):
1096             smart_methods.remove('piped')
1097 
1098         # Transfer methods to try
1099         methods = []
1100 
1101         # Use the transfer_method option if set, otherwise use scp_if_ssh
1102         ssh_transfer_method = self._play_context.ssh_transfer_method
1103         if ssh_transfer_method is not None:
1104             if not (ssh_transfer_method in ('smart', 'sftp', 'scp', 'piped')):
1105                 raise AnsibleOptionsError('transfer_method needs to be one of [smart|sftp|scp|piped]')
1106             if ssh_transfer_method == 'smart':
1107                 methods = smart_methods
1108             else:
1109                 methods = [ssh_transfer_method]
1110         else:
1111             # since this can be a non-bool now, we need to handle it correctly
1112             scp_if_ssh = C.DEFAULT_SCP_IF_SSH
1113             if not isinstance(scp_if_ssh, bool):
1114                 scp_if_ssh = scp_if_ssh.lower()
1115                 if scp_if_ssh in BOOLEANS:
1116                     scp_if_ssh = boolean(scp_if_ssh, strict=False)
1117                 elif scp_if_ssh != 'smart':
1118                     raise AnsibleOptionsError('scp_if_ssh needs to be one of [smart|True|False]')
1119             if scp_if_ssh == 'smart':
1120                 methods = smart_methods
1121             elif scp_if_ssh is True:
1122                 methods = ['scp']
1123             else:
1124                 methods = ['sftp']
1125 
1126         for method in methods:
1127             returncode = stdout = stderr = None
1128             if method == 'sftp':
1129                 cmd = self._build_command(self.get_option('sftp_executable'), to_bytes(host))
1130                 in_data = u"{0} {1} {2}\n".format(sftp_action, shlex_quote(in_path), shlex_quote(out_path))
1131                 in_data = to_bytes(in_data, nonstring='passthru')
1132                 (returncode, stdout, stderr) = self._bare_run(cmd, in_data, checkrc=False)
1133             elif method == 'scp':
1134                 scp = self.get_option('scp_executable')
1135 
1136                 if sftp_action == 'get':
1137                     cmd = self._build_command(scp, u'{0}:{1}'.format(host, self._shell.quote(in_path)), out_path)
1138                 else:
1139                     cmd = self._build_command(scp, in_path, u'{0}:{1}'.format(host, self._shell.quote(out_path)))
1140                 in_data = None
1141                 (returncode, stdout, stderr) = self._bare_run(cmd, in_data, checkrc=False)
1142             elif method == 'piped':
1143                 if sftp_action == 'get':
1144                     # we pass sudoable=False to disable pty allocation, which
1145                     # would end up mixing stdout/stderr and screwing with newlines
1146                     (returncode, stdout, stderr) = self.exec_command('dd if=%s bs=%s' % (in_path, BUFSIZE), sudoable=False)
1147                     with open(to_bytes(out_path, errors='surrogate_or_strict'), 'wb+') as out_file:
1148                         out_file.write(stdout)
1149                 else:
1150                     with open(to_bytes(in_path, errors='surrogate_or_strict'), 'rb') as f:
1151                         in_data = to_bytes(f.read(), nonstring='passthru')
1152                     if not in_data:
1153                         count = ' count=0'
1154                     else:
1155                         count = ''
1156                     (returncode, stdout, stderr) = self.exec_command('dd of=%s bs=%s%s' % (out_path, BUFSIZE, count), in_data=in_data, sudoable=False)
1157 
1158             # Check the return code and rollover to next method if failed
1159             if returncode == 0:
1160                 return (returncode, stdout, stderr)
1161             else:
1162                 # If not in smart mode, the data will be printed by the raise below
1163                 if len(methods) > 1:
1164                     display.warning(u'%s transfer mechanism failed on %s. Use ANSIBLE_DEBUG=1 to see detailed information' % (method, host))
1165                     display.debug(u'%s' % to_text(stdout))
1166                     display.debug(u'%s' % to_text(stderr))
1167 
1168         if returncode == 255:
1169             raise AnsibleConnectionFailure("Failed to connect to the host via %s: %s" % (method, to_native(stderr)))
1170         else:
1171             raise AnsibleError("failed to transfer file to %s %s:\n%s\n%s" %
1172                                (to_native(in_path), to_native(out_path), to_native(stdout), to_native(stderr)))

1174     def _escape_win_path(self, path):
1175         """ converts a Windows path to one that's supported by SFTP and SCP """
1176         # If using a root path then we need to start with /
1177         prefix = ""
1178         if re.match(r'^\w{1}:', path):
1179             prefix = "/"
1180 
1181         # Convert all '\' to '/'
1182         return "%s%s" % (prefix, path.replace("\\", "/"))

1187     def exec_command(self, cmd, in_data=None, sudoable=True):
1188         ''' run a command on the remote host '''
1189 
1190         super(Connection, self).exec_command(cmd, in_data=in_data, sudoable=sudoable)
1191 
1192         display.vvv(u"ESTABLISH SSH CONNECTION FOR USER: {0}".format(self._play_context.remote_user), host=self._play_context.remote_addr)
1193 
1194         if getattr(self._shell, "_IS_WINDOWS", False):
1195             # Become method 'runas' is done in the wrapper that is executed,
1196             # need to disable sudoable so the bare_run is not waiting for a
1197             # prompt that will not occur
1198             sudoable = False
1199 
1200             # Make sure our first command is to set the console encoding to
1201             # utf-8, this must be done via chcp to get utf-8 (65001)
1202             cmd_parts = ["chcp.com", "65001", self._shell._SHELL_REDIRECT_ALLNULL, self._shell._SHELL_AND]
1203             cmd_parts.extend(self._shell._encode_script(cmd, as_list=True, strict_mode=False, preserve_rc=False))
1204             cmd = ' '.join(cmd_parts)
1205 
1206         # we can only use tty when we are not pipelining the modules. piping
1207         # data into /usr/bin/python inside a tty automatically invokes the
1208         # python interactive-mode but the modules are not compatible with the
1209         # interactive-mode ("unexpected indent" mainly because of empty lines)
1210 
1211         ssh_executable = self._play_context.ssh_executable
1212 
1213         # -tt can cause various issues in some environments so allow the user
1214         # to disable it as a troubleshooting method.
1215         use_tty = self.get_option('use_tty')
1216 
1217         if not in_data and sudoable and use_tty:
1218             args = (ssh_executable, '-tt', self.host, cmd)
1219         else:
1220             args = (ssh_executable, self.host, cmd)
1221 
1222         cmd = self._build_command(*args)
1223         (returncode, stdout, stderr) = self._run(cmd, in_data, sudoable=sudoable)
1224 
1225         # When running on Windows, stderr may contain CLIXML encoded output
1226         if getattr(self._shell, "_IS_WINDOWS", False) and stderr.startswith(b"#< CLIXML"):
1227             stderr = _parse_clixml(stderr)
1228 
1229         return (returncode, stdout, stderr)

1258     def reset(self):
1259         # If we have a persistent ssh connection (ControlPersist), we can ask it to stop listening.
1260         cmd = self._build_command(self._play_context.ssh_executable, '-O', 'stop', self.host)
1261         controlpersist, controlpath = self._persistence_controls(cmd)
1262         cp_arg = [a for a in cmd if a.startswith(b"ControlPath=")]
1263 
1264         # only run the reset if the ControlPath already exists or if it isn't
1265         # configured and ControlPersist is set
1266         run_reset = False
1267         if controlpersist and len(cp_arg) > 0:
1268             cp_path = cp_arg[0].split(b"=", 1)[-1]
1269             if os.path.exists(cp_path):
1270                 run_reset = True
1271         elif controlpersist:
1272             run_reset = True
1273 
1274         if run_reset:
1275             display.vvv(u'sending stop: %s' % to_text(cmd))
1276             p = subprocess.Popen(cmd, stdin=subprocess.PIPE, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
1277             stdout, stderr = p.communicate()
1278             status_code = p.wait()
1279             if status_code != 0:
1280                 display.warning(u"Failed to reset connection:%s" % to_text(stderr))
1281 
1282         self.close()
