 356     def _connect(self):
 357         if not HAS_PYPSRP:
 358             raise AnsibleError("pypsrp or dependencies are not installed: %s"
 359                                % to_native(PYPSRP_IMP_ERR))
 360         super(Connection, self)._connect()
 361         self._build_kwargs()
 362         display.vvv("ESTABLISH PSRP CONNECTION FOR USER: %s ON PORT %s TO %s" %
 363                     (self._psrp_user, self._psrp_port, self._psrp_host),
 364                     host=self._psrp_host)
 365 
 366         if not self.runspace:
 367             connection = WSMan(**self._psrp_conn_kwargs)
 368 
 369             # create our psuedo host to capture the exit code and host output
 370             host_ui = PSHostUserInterface()
 371             self.host = PSHost(None, None, False, "Ansible PSRP Host", None,
 372                                host_ui, None)
 373 
 374             self.runspace = RunspacePool(
 375                 connection, host=self.host,
 376                 configuration_name=self._psrp_configuration_name
 377             )
 378             display.vvvvv(
 379                 "PSRP OPEN RUNSPACE: auth=%s configuration=%s endpoint=%s" %
 380                 (self._psrp_auth, self._psrp_configuration_name,
 381                  connection.transport.endpoint), host=self._psrp_host
 382             )
 383             try:
 384                 self.runspace.open()
 385             except AuthenticationError as e:
 386                 raise AnsibleConnectionFailure("failed to authenticate with "
 387                                                "the server: %s" % to_native(e))
 388             except WinRMError as e:
 389                 raise AnsibleConnectionFailure(
 390                     "psrp connection failure during runspace open: %s"
 391                     % to_native(e)
 392                 )
 393             except (ConnectionError, ConnectTimeout) as e:
 394                 raise AnsibleConnectionFailure(
 395                     "Failed to connect to the host via PSRP: %s"
 396                     % to_native(e)
 397                 )
 398 
 399             self._connected = True
 400         return self

 402     def reset(self):
 403         display.vvvvv("PSRP: Reset Connection", host=self._psrp_host)
 404         self.runspace = None
 405         self._connect()

 407     def exec_command(self, cmd, in_data=None, sudoable=True):
 408         super(Connection, self).exec_command(cmd, in_data=in_data,
 409                                              sudoable=sudoable)
 410 
 411         if cmd.startswith(" ".join(_common_args) + " -EncodedCommand"):
 412             # This is a PowerShell script encoded by the shell plugin, we will
 413             # decode the script and execute it in the runspace instead of
 414             # starting a new interpreter to save on time
 415             b_command = base64.b64decode(cmd.split(" ")[-1])
 416             script = to_text(b_command, 'utf-16-le')
 417             in_data = to_text(in_data, errors="surrogate_or_strict", nonstring="passthru")
 418 
 419             if in_data and in_data.startswith(u"#!"):
 420                 # ANSIBALLZ wrapper, we need to get the interpreter and execute
 421                 # that as the script - note this won't work as basic.py relies
 422                 # on packages not available on Windows, once fixed we can enable
 423                 # this path
 424                 interpreter = to_native(in_data.splitlines()[0][2:])
 425                 # script = "$input | &'%s' -" % interpreter
 426                 # in_data = to_text(in_data)
 427                 raise AnsibleError("cannot run the interpreter '%s' on the psrp "
 428                                    "connection plugin" % interpreter)
 429 
 430             # call build_module_command to get the bootstrap wrapper text
 431             bootstrap_wrapper = self._shell.build_module_command('', '', '')
 432             if bootstrap_wrapper == cmd:
 433                 # Do not display to the user each invocation of the bootstrap wrapper
 434                 display.vvv("PSRP: EXEC (via pipeline wrapper)")
 435             else:
 436                 display.vvv("PSRP: EXEC %s" % script, host=self._psrp_host)
 437         else:
 438             # In other cases we want to execute the cmd as the script. We add on the 'exit $LASTEXITCODE' to ensure the
 439             # rc is propagated back to the connection plugin.
 440             script = to_text(u"%s\nexit $LASTEXITCODE" % cmd)
 441             display.vvv(u"PSRP: EXEC %s" % script, host=self._psrp_host)
 442 
 443         rc, stdout, stderr = self._exec_psrp_script(script, in_data)
 444         return rc, stdout, stderr

 446     def put_file(self, in_path, out_path):
 447         super(Connection, self).put_file(in_path, out_path)
 448         display.vvv("PUT %s TO %s" % (in_path, out_path), host=self._psrp_host)
 449 
 450         out_path = self._shell._unquote(out_path)
 451         script = u'''begin {
 452     $ErrorActionPreference = "Stop"
 453 
 454     $path = '%s'
 455     $fd = [System.IO.File]::Create($path)
 456     $algo = [System.Security.Cryptography.SHA1CryptoServiceProvider]::Create()
 457     $bytes = @()
 458 } process {
 459     $bytes = [System.Convert]::FromBase64String($input)
 460     $algo.TransformBlock($bytes, 0, $bytes.Length, $bytes, 0) > $null
 461     $fd.Write($bytes, 0, $bytes.Length)
 462 } end {
 463     $fd.Close()
 464     $algo.TransformFinalBlock($bytes, 0, 0) > $null
 465     $hash = [System.BitConverter]::ToString($algo.Hash)
 466     $hash = $hash.Replace("-", "").ToLowerInvariant()
 467 
 468     Write-Output -InputObject "{`"sha1`":`"$hash`"}"
 469 }''' % self._shell._escape(out_path)
 470 
 471         cmd_parts = self._shell._encode_script(script, as_list=True,
 472                                                strict_mode=False,
 473                                                preserve_rc=False)
 474         b_in_path = to_bytes(in_path, errors='surrogate_or_strict')
 475         if not os.path.exists(b_in_path):
 476             raise AnsibleFileNotFound('file or module does not exist: "%s"'
 477                                       % to_native(in_path))
 478 
 479         in_size = os.path.getsize(b_in_path)
 480         buffer_size = int(self.runspace.connection.max_payload_size / 4 * 3)
 481 
 482         # copying files is faster when using the raw WinRM shell and not PSRP
 483         # we will create a WinRS shell just for this process
 484         # TODO: speed this up as there is overhead creating a shell for this
 485         with WinRS(self.runspace.connection, codepage=65001) as shell:
 486             process = Process(shell, cmd_parts[0], cmd_parts[1:])
 487             process.begin_invoke()
 488 
 489             offset = 0
 490             with open(b_in_path, 'rb') as src_file:
 491                 for data in iter((lambda: src_file.read(buffer_size)), b""):
 492                     offset += len(data)
 493                     display.vvvvv("PSRP PUT %s to %s (offset=%d, size=%d" %
 494                                   (in_path, out_path, offset, len(data)),
 495                                   host=self._psrp_host)
 496                     b64_data = base64.b64encode(data) + b"\r\n"
 497                     process.send(b64_data, end=(src_file.tell() == in_size))
 498 
 499                 # the file was empty, return empty buffer
 500                 if offset == 0:
 501                     process.send(b"", end=True)
 502 
 503             process.end_invoke()
 504             process.signal(SignalCode.CTRL_C)
 505 
 506         if process.rc != 0:
 507             raise AnsibleError(to_native(process.stderr))
 508 
 509         put_output = json.loads(process.stdout)
 510         remote_sha1 = put_output.get("sha1")
 511 
 512         if not remote_sha1:
 513             raise AnsibleError("Remote sha1 was not returned, stdout: '%s', "
 514                                "stderr: '%s'" % (to_native(process.stdout),
 515                                                  to_native(process.stderr)))
 516 
 517         local_sha1 = secure_hash(in_path)
 518         if not remote_sha1 == local_sha1:
 519             raise AnsibleError("Remote sha1 hash %s does not match local hash "
 520                                "%s" % (to_native(remote_sha1),
 521                                        to_native(local_sha1)))
 522 
 523     def fetch_file(self, in_path, out_path):
 524         super(Connection, self).fetch_file(in_path, out_path)
 525         display.vvv("FETCH %s TO %s" % (in_path, out_path),
 526                     host=self._psrp_host)
 527 
 528         in_path = self._shell._unquote(in_path)
 529         out_path = out_path.replace('\\', '/')
 530 
 531         # because we are dealing with base64 data we need to get the max size
 532         # of the bytes that the base64 size would equal
 533         max_b64_size = int(self.runspace.connection.max_payload_size -
 534                            (self.runspace.connection.max_payload_size / 4 * 3))
 535         buffer_size = max_b64_size - (max_b64_size % 1024)
 536 
 537         # setup the file stream with read only mode
 538         setup_script = '''$ErrorActionPreference = "Stop"
 539 $path = '%s'
 540 
 541 if (Test-Path -Path $path -PathType Leaf) {
 542     $fs = New-Object -TypeName System.IO.FileStream -ArgumentList @(
 543         $path,
 544         [System.IO.FileMode]::Open,
 545         [System.IO.FileAccess]::Read,
 546         [System.IO.FileShare]::Read
 547     )
 548     $buffer_size = %d
 549 } elseif (Test-Path -Path $path -PathType Container) {
 550     Write-Output -InputObject "[DIR]"
 551 } else {
 552     Write-Error -Message "$path does not exist"
 553     $host.SetShouldExit(1)
 554 }''' % (self._shell._escape(in_path), buffer_size)
 555 
 556         # read the file stream at the offset and return the b64 string
 557         read_script = '''$ErrorActionPreference = "Stop"
 558 $fs.Seek(%d, [System.IO.SeekOrigin]::Begin) > $null
 559 $buffer = New-Object -TypeName byte[] -ArgumentList $buffer_size
 560 $bytes_read = $fs.Read($buffer, 0, $buffer_size)
 561 
 562 if ($bytes_read -gt 0) {
 563     $bytes = $buffer[0..($bytes_read - 1)]
 564     Write-Output -InputObject ([System.Convert]::ToBase64String($bytes))
 565 }'''
 566 
 567         # need to run the setup script outside of the local scope so the
 568         # file stream stays active between fetch operations
 569         rc, stdout, stderr = self._exec_psrp_script(setup_script,
 570                                                     use_local_scope=False,
 571                                                     force_stop=True)
 572         if rc != 0:
 573             raise AnsibleError("failed to setup file stream for fetch '%s': %s"
 574                                % (out_path, to_native(stderr)))
 575         elif stdout.strip() == '[DIR]':
 576             # to be consistent with other connection plugins, we assume the caller has created the target dir
 577             return
 578 
 579         b_out_path = to_bytes(out_path, errors='surrogate_or_strict')
 580         # to be consistent with other connection plugins, we assume the caller has created the target dir
 581         offset = 0
 582         with open(b_out_path, 'wb') as out_file:
 583             while True:
 584                 display.vvvvv("PSRP FETCH %s to %s (offset=%d" %
 585                               (in_path, out_path, offset), host=self._psrp_host)
 586                 rc, stdout, stderr = self._exec_psrp_script(read_script % offset, force_stop=True)
 587                 if rc != 0:
 588                     raise AnsibleError("failed to transfer file to '%s': %s"
 589                                        % (out_path, to_native(stderr)))
 590 
 591                 data = base64.b64decode(stdout.strip())
 592                 out_file.write(data)
 593                 if len(data) < buffer_size:
 594                     break
 595                 offset += len(data)
 596 
 597             rc, stdout, stderr = self._exec_psrp_script("$fs.Close()", force_stop=True)
 598             if rc != 0:
 599                 display.warning("failed to close remote file stream of file "
 600                                 "'%s': %s" % (in_path, to_native(stderr)))

 602     def close(self):
 603         if self.runspace and self.runspace.state == RunspacePoolState.OPENED:
 604             display.vvvvv("PSRP CLOSE RUNSPACE: %s" % (self.runspace.id),
 605                           host=self._psrp_host)
 606             self.runspace.close()
 607         self.runspace = None
 608         self._connected = False

 610     def _build_kwargs(self):

 612         self._psrp_user = self.get_option('remote_user')

 614 
 615         protocol = self.get_option('protocol')
 616         port = self.get_option('port')
 617         if protocol is None and port is None:
 618             protocol = 'https'
 619             port = 5986
 620         elif protocol is None:
 621             protocol = 'https' if int(port) != 5985 else 'http'
 622         elif port is None:
 623             port = 5986 if protocol == 'https' else 5985
 624 
 625         self._psrp_protocol = protocol
 626         self._psrp_port = int(port)
 627 
 628         self._psrp_path = self.get_option('path')
 629         self._psrp_auth = self.get_option('auth')
 630         # cert validation can either be a bool or a path to the cert
 631         cert_validation = self.get_option('cert_validation')
 632         cert_trust_path = self.get_option('ca_cert')
 633         if cert_validation == 'ignore':
 634             self._psrp_cert_validation = False
 635         elif cert_trust_path is not None:
 636             self._psrp_cert_validation = cert_trust_path
 637         else:
 638             self._psrp_cert_validation = True
 639 
 640         self._psrp_connection_timeout = self.get_option('connection_timeout')  # Can be None
 641         self._psrp_read_timeout = self.get_option('read_timeout')  # Can be None
 642         self._psrp_message_encryption = self.get_option('message_encryption')
 643         self._psrp_proxy = self.get_option('proxy')
 644         self._psrp_ignore_proxy = boolean(self.get_option('ignore_proxy'))
 645         self._psrp_operation_timeout = int(self.get_option('operation_timeout'))
 646         self._psrp_max_envelope_size = int(self.get_option('max_envelope_size'))
 647         self._psrp_configuration_name = self.get_option('configuration_name')
 648         self._psrp_reconnection_retries = int(self.get_option('reconnection_retries'))
 649         self._psrp_reconnection_backoff = float(self.get_option('reconnection_backoff'))
 650 
 651         self._psrp_certificate_key_pem = self.get_option('certificate_key_pem')
 652         self._psrp_certificate_pem = self.get_option('certificate_pem')
 653         self._psrp_credssp_auth_mechanism = self.get_option('credssp_auth_mechanism')
 654         self._psrp_credssp_disable_tlsv1_2 = self.get_option('credssp_disable_tlsv1_2')
 655         self._psrp_credssp_minimum_version = self.get_option('credssp_minimum_version')
 656         self._psrp_negotiate_send_cbt = self.get_option('negotiate_send_cbt')
 657         self._psrp_negotiate_delegate = self.get_option('negotiate_delegate')
 658         self._psrp_negotiate_hostname_override = self.get_option('negotiate_hostname_override')
 659         self._psrp_negotiate_service = self.get_option('negotiate_service')
 660 
 661         supported_args = []
 662         for auth_kwarg in AUTH_KWARGS.values():
 663             supported_args.extend(auth_kwarg)
 664         extra_args = set([v.replace('ansible_psrp_', '') for v in
 665                           self.get_option('_extras')])
 666         unsupported_args = extra_args.difference(supported_args)
 667 
 668         for arg in unsupported_args:
 669             display.warning("ansible_psrp_%s is unsupported by the current "
 670                             "psrp version installed" % arg)
 671 
 672         self._psrp_conn_kwargs = dict(
 673             server=self._psrp_host, port=self._psrp_port,
 674             username=self._psrp_user, password=self._psrp_pass,
 675             ssl=self._psrp_protocol == 'https', path=self._psrp_path,
 676             auth=self._psrp_auth, cert_validation=self._psrp_cert_validation,
 677             connection_timeout=self._psrp_connection_timeout,
 678             encryption=self._psrp_message_encryption, proxy=self._psrp_proxy,
 679             no_proxy=self._psrp_ignore_proxy,
 680             max_envelope_size=self._psrp_max_envelope_size,
 681             operation_timeout=self._psrp_operation_timeout,
 682             certificate_key_pem=self._psrp_certificate_key_pem,
 683             certificate_pem=self._psrp_certificate_pem,
 684             credssp_auth_mechanism=self._psrp_credssp_auth_mechanism,
 685             credssp_disable_tlsv1_2=self._psrp_credssp_disable_tlsv1_2,
 686             credssp_minimum_version=self._psrp_credssp_minimum_version,
 687             negotiate_send_cbt=self._psrp_negotiate_send_cbt,
 688             negotiate_delegate=self._psrp_negotiate_delegate,
 689             negotiate_hostname_override=self._psrp_negotiate_hostname_override,
 690             negotiate_service=self._psrp_negotiate_service,
 691         )
 692 
 693         # Check if PSRP version supports newer read_timeout argument (needs pypsrp 0.3.0+)
 694         if hasattr(pypsrp, 'FEATURES') and 'wsman_read_timeout' in pypsrp.FEATURES:
 695             self._psrp_conn_kwargs['read_timeout'] = self._psrp_read_timeout
 696         elif self._psrp_read_timeout is not None:
 697             display.warning("ansible_psrp_read_timeout is unsupported by the current psrp version installed, "
 698                             "using ansible_psrp_connection_timeout value for read_timeout instead.")
 699 
 700         # Check if PSRP version supports newer reconnection_retries argument (needs pypsrp 0.3.0+)
 701         if hasattr(pypsrp, 'FEATURES') and 'wsman_reconnections' in pypsrp.FEATURES:
 702             self._psrp_conn_kwargs['reconnection_retries'] = self._psrp_reconnection_retries
 703             self._psrp_conn_kwargs['reconnection_backoff'] = self._psrp_reconnection_backoff
 704         else:
 705             if self._psrp_reconnection_retries is not None:
 706                 display.warning("ansible_psrp_reconnection_retries is unsupported by the current psrp version installed.")
 707             if self._psrp_reconnection_backoff is not None:
 708                 display.warning("ansible_psrp_reconnection_backoff is unsupported by the current psrp version installed.")
 709 
 710         # add in the extra args that were set
 711         for arg in extra_args.intersection(supported_args):
 712             option = self.get_option('_extras')['ansible_psrp_%s' % arg]
 713             self._psrp_conn_kwargs[arg] = option
 714 
 715     def _exec_psrp_script(self, script, input_data=None, use_local_scope=True, force_stop=False):
 716         ps = PowerShell(self.runspace)
 717         ps.add_script(script, use_local_scope=use_local_scope)
 718         ps.invoke(input=input_data)
 719 
 720         rc, stdout, stderr = self._parse_pipeline_result(ps)
 721 
 722         if force_stop:
 723             # This is usually not needed because we close the Runspace after our exec and we skip the call to close the
 724             # pipeline manually to save on some time. Set to True when running multiple exec calls in the same runspace.
 725 
 726             # Current pypsrp versions raise an exception if the current state was not RUNNING. We manually set it so we
 727             # can call stop without any issues.
 728             ps.state = PSInvocationState.RUNNING
 729             ps.stop()
 730 
 731         return rc, stdout, stderr
 732 
 733     def _parse_pipeline_result(self, pipeline):
 734         """
 735         PSRP doesn't have the same concept as other protocols with its output.
 736         We need some extra logic to convert the pipeline streams and host
 737         output into the format that Ansible understands.
 738 
 739         :param pipeline: The finished PowerShell pipeline that invoked our
 740             commands
 741         :return: rc, stdout, stderr based on the pipeline output
 742         """
 743         # we try and get the rc from our host implementation, this is set if
 744         # exit or $host.SetShouldExit() is called in our pipeline, if not we
 745         # set to 0 if the pipeline had not errors and 1 if it did
 746         rc = self.host.rc or (1 if pipeline.had_errors else 0)
 747 
 748         # TODO: figure out a better way of merging this with the host output
 749         stdout_list = []
 750         for output in pipeline.output:
 751             # Not all pipeline outputs are a string or contain a __str__ value,
 752             # we will create our own output based on the properties of the
 753             # complex object if that is the case.
 754             if isinstance(output, GenericComplexObject) and output.to_string is None:
 755                 obj_lines = output.property_sets
 756                 for key, value in output.adapted_properties.items():
 757                     obj_lines.append(u"%s: %s" % (key, value))
 758                 for key, value in output.extended_properties.items():
 759                     obj_lines.append(u"%s: %s" % (key, value))
 760                 output_msg = u"\n".join(obj_lines)
 761             else:
 762                 output_msg = to_text(output, nonstring='simplerepr')
 763 
 764             stdout_list.append(output_msg)
 765 
 766         if len(self.host.ui.stdout) > 0:
 767             stdout_list += self.host.ui.stdout
 768         stdout = u"\r\n".join(stdout_list)
 769 
 770         stderr_list = []
 771         for error in pipeline.streams.error:
 772             # the error record is not as fully fleshed out like we usually get
 773             # in PS, we will manually create it here
 774             command_name = "%s : " % error.command_name if error.command_name else ''
 775             position = "%s\r\n" % error.invocation_position_message if error.invocation_position_message else ''
 776             error_msg = "%s%s\r\n%s" \
 777                         "    + CategoryInfo          : %s\r\n" \
 778                         "    + FullyQualifiedErrorId : %s" \
 779                         % (command_name, str(error), position,
 780                            error.message, error.fq_error)
 781             stacktrace = error.script_stacktrace
 782             if self._play_context.verbosity >= 3 and stacktrace is not None:
 783                 error_msg += "\r\nStackTrace:\r\n%s" % stacktrace
 784             stderr_list.append(error_msg)
 785 
 786         if len(self.host.ui.stderr) > 0:
 787             stderr_list += self.host.ui.stderr
 788         stderr = u"\r\n".join([to_text(o) for o in stderr_list])
 789 
 790         display.vvvvv("PSRP RC: %d" % rc, host=self._psrp_host)
 791         display.vvvvv("PSRP STDOUT: %s" % stdout, host=self._psrp_host)
 792         display.vvvvv("PSRP STDERR: %s" % stderr, host=self._psrp_host)
 793 
 794         # reset the host back output back to defaults, needed if running
 795         # multiple pipelines on the same RunspacePool
 796         self.host.rc = 0
 797         self.host.ui.stdout = []
 798         self.host.ui.stderr = []
 799 
 800         return rc, to_bytes(stdout, encoding='utf-8'), to_bytes(stderr, encoding='utf-8')
