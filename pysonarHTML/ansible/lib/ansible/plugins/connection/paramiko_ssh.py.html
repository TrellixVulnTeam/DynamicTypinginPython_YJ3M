 162 paramiko: The authenticity of host '%s' can't be established.
 163 The %s key fingerprint is %s.
 164 Are you sure you want to continue connecting (yes/no)?

 168 SETTINGS_REGEX = re.compile(r'(\w+)(?:\s*=\s*|\s+)(.+)')

 171 class MyAddPolicy(object):
 172     """
 173     Based on AutoAddPolicy in paramiko so we can determine when keys are added
 174 
 175     and also prompt for input.
 176 
 177     Policy for automatically adding the hostname and new host key to the
 178     local L{HostKeys} object, and saving it.  This is used by L{SSHClient}.
 179     """

 181     def __init__(self, new_stdin, connection):
 182         self._new_stdin = new_stdin
 183         self.connection = connection
 184         self._options = connection._options

 186     def missing_host_key(self, client, hostname, key):
 187 
 188         if all((self._options['host_key_checking'], not self._options['host_key_auto_add'])):
 189 
 190             fingerprint = hexlify(key.get_fingerprint())
 191             ktype = key.get_name()
 192 
 193             if self.connection.get_option('use_persistent_connections') or self.connection.force_persistence:
 194                 # don't print the prompt string since the user cannot respond
 195                 # to the question anyway
 196                 raise AnsibleError(AUTHENTICITY_MSG[1:92] % (hostname, ktype, fingerprint))
 197 
 198             self.connection.connection_lock()
 199 
 200             old_stdin = sys.stdin
 201             sys.stdin = self._new_stdin
 202 
 203             # clear out any premature input on sys.stdin
 204             tcflush(sys.stdin, TCIFLUSH)
 205 
 206             inp = input(AUTHENTICITY_MSG % (hostname, ktype, fingerprint))
 207             sys.stdin = old_stdin
 208 
 209             self.connection.connection_unlock()
 210 
 211             if inp not in ['yes', 'y', '']:
 212                 raise AnsibleError("host connection rejected by user")
 213 
 214         key._added_by_ansible_this_time = True
 215 
 216         # existing implementation below:
 217         client._host_keys.add(hostname, key.get_name(), key)
 218 
 219         # host keys are actually saved in close() function below
 220         # in order to control ordering.

 238     def _connect(self):
 239         cache_key = self._cache_key()
 240         if cache_key in SSH_CONNECTION_CACHE:
 241             self.ssh = SSH_CONNECTION_CACHE[cache_key]
 242         else:
 243             self.ssh = SSH_CONNECTION_CACHE[cache_key] = self._connect_uncached()
 244         return self

 250     def _parse_proxy_command(self, port=22):
 251         proxy_command = None
 252         # Parse ansible_ssh_common_args, specifically looking for ProxyCommand
 253         ssh_args = [
 254             getattr(self._play_context, 'ssh_extra_args', '') or '',
 255             getattr(self._play_context, 'ssh_common_args', '') or '',
 256             getattr(self._play_context, 'ssh_args', '') or '',
 257         ]
 258 
 259         args = self._split_ssh_args(' '.join(ssh_args))
 260         for i, arg in enumerate(args):
 261             if arg.lower() == 'proxycommand':
 262                 # _split_ssh_args split ProxyCommand from the command itself
 263                 proxy_command = args[i + 1]
 264             else:
 265                 # ProxyCommand and the command itself are a single string
 266                 match = SETTINGS_REGEX.match(arg)
 267                 if match:
 268                     if match.group(1).lower() == 'proxycommand':
 269                         proxy_command = match.group(2)
 270 
 271             if proxy_command:
 272                 break
 273 
 274         proxy_command = proxy_command or self.get_option('proxy_command')
 275 
 276         sock_kwarg = {}
 277         if proxy_command:
 278             replacers = {
 279                 '%h': self._play_context.remote_addr,
 280                 '%p': port,
 281                 '%r': self._play_context.remote_user
 282             }
 283             for find, replace in replacers.items():
 284                 proxy_command = proxy_command.replace(find, str(replace))
 285             try:
 286                 sock_kwarg = {'sock': paramiko.ProxyCommand(proxy_command)}
 287                 display.vvv("CONFIGURE PROXY COMMAND FOR CONNECTION: %s" % proxy_command, host=self._play_context.remote_addr)
 288             except AttributeError:
 289                 display.warning('Paramiko ProxyCommand support unavailable. '
 290                                 'Please upgrade to Paramiko 1.9.0 or newer. '
 291                                 'Not using configured ProxyCommand')
 292 
 293         return sock_kwarg

 295     def _connect_uncached(self):
 296         ''' activates the connection object '''
 297 
 298         if paramiko is None:
 299             raise AnsibleError("paramiko is not installed: %s" % to_native(PARAMIKO_IMPORT_ERR))
 300 
 301         port = self._play_context.port or 22
 302         display.vvv("ESTABLISH PARAMIKO SSH CONNECTION FOR USER: %s on PORT %s TO %s" % (self._play_context.remote_user, port, self._play_context.remote_addr),
 303                     host=self._play_context.remote_addr)
 304 
 305         ssh = paramiko.SSHClient()
 306 
 307         # override paramiko's default logger name
 308         if self._log_channel is not None:
 309             ssh.set_log_channel(self._log_channel)
 310 
 311         self.keyfile = os.path.expanduser("~/.ssh/known_hosts")
 312 
 313         if self.get_option('host_key_checking'):
 314             for ssh_known_hosts in ("/etc/ssh/ssh_known_hosts", "/etc/openssh/ssh_known_hosts"):
 315                 try:
 316                     # TODO: check if we need to look at several possible locations, possible for loop
 317                     ssh.load_system_host_keys(ssh_known_hosts)
 318                     break
 319                 except IOError:
 320                     pass  # file was not found, but not required to function
 321             ssh.load_system_host_keys()
 322 
 323         ssh_connect_kwargs = self._parse_proxy_command(port)
 324 
 325         ssh.set_missing_host_key_policy(MyAddPolicy(self._new_stdin, self))
 326 
 327         conn_password = self.get_option('password') or self._play_context.password
 328 
 329         allow_agent = True
 330 
 331         if conn_password is not None:
 332             allow_agent = False
 333 
 334         try:
 335             key_filename = None
 336             if self._play_context.private_key_file:
 337                 key_filename = os.path.expanduser(self._play_context.private_key_file)
 338 
 339             # paramiko 2.2 introduced auth_timeout parameter
 340             if LooseVersion(paramiko.__version__) >= LooseVersion('2.2.0'):
 341                 ssh_connect_kwargs['auth_timeout'] = self._play_context.timeout
 342 
 343             ssh.connect(
 344                 self._play_context.remote_addr.lower(),
 345                 username=self._play_context.remote_user,
 346                 allow_agent=allow_agent,
 347                 look_for_keys=self.get_option('look_for_keys'),
 348                 key_filename=key_filename,
 349                 password=conn_password,
 350                 timeout=self._play_context.timeout,
 351                 port=port,
 352                 **ssh_connect_kwargs
 353             )
 354         except paramiko.ssh_exception.BadHostKeyException as e:
 355             raise AnsibleConnectionFailure('host key mismatch for %s' % e.hostname)
 356         except paramiko.ssh_exception.AuthenticationException as e:
 357             msg = 'Failed to authenticate: {0}'.format(to_text(e))
 358             raise AnsibleAuthenticationFailure(msg)
 359         except Exception as e:
 360             msg = to_text(e)
 361             if u"PID check failed" in msg:
 362                 raise AnsibleError("paramiko version issue, please upgrade paramiko on the machine running ansible")
 363             elif u"Private key file is encrypted" in msg:
 364                 msg = 'ssh %s@%s:%s : %s\nTo connect as a different user, use -u <username>.' % (
 365                     self._play_context.remote_user, self._play_context.remote_addr, port, msg)
 366                 raise AnsibleConnectionFailure(msg)
 367             else:
 368                 raise AnsibleConnectionFailure(msg)
 369 
 370         return ssh

 372     def exec_command(self, cmd, in_data=None, sudoable=True):
 373         ''' run a command on the remote host '''
 374 
 375         super(Connection, self).exec_command(cmd, in_data=in_data, sudoable=sudoable)
 376 
 377         if in_data:
 378             raise AnsibleError("Internal Error: this module does not support optimized module pipelining")
 379 
 380         bufsize = 4096
 381 
 382         try:
 383             self.ssh.get_transport().set_keepalive(5)
 384             chan = self.ssh.get_transport().open_session()
 385         except Exception as e:
 386             text_e = to_text(e)
 387             msg = u"Failed to open session"
 388             if text_e:
 389                 msg += u": %s" % text_e
 390             raise AnsibleConnectionFailure(to_native(msg))
 391 
 392         # sudo usually requires a PTY (cf. requiretty option), therefore
 393         # we give it one by default (pty=True in ansible.cfg), and we try
 394         # to initialise from the calling environment when sudoable is enabled
 395         if self.get_option('pty') and sudoable:
 396             chan.get_pty(term=os.getenv('TERM', 'vt100'), width=int(os.getenv('COLUMNS', 0)), height=int(os.getenv('LINES', 0)))
 397 
 398         display.vvv("EXEC %s" % cmd, host=self._play_context.remote_addr)
 399 
 400         cmd = to_bytes(cmd, errors='surrogate_or_strict')
 401 
 402         no_prompt_out = b''
 403         no_prompt_err = b''
 404         become_output = b''
 405 
 406         try:
 407             chan.exec_command(cmd)
 408             if self.become and self.become.expect_prompt():
 409                 passprompt = False
 410                 become_sucess = False
 411                 while not (become_sucess or passprompt):
 412                     display.debug('Waiting for Privilege Escalation input')
 413 
 414                     chunk = chan.recv(bufsize)
 415                     display.debug("chunk is: %s" % chunk)
 416                     if not chunk:
 417                         if b'unknown user' in become_output:
 418                             n_become_user = to_native(self.become.get_option('become_user',
 419                                                                              playcontext=self._play_context))
 420                             raise AnsibleError('user %s does not exist' % n_become_user)
 421                         else:
 422                             break
 423                             # raise AnsibleError('ssh connection closed waiting for password prompt')
 424                     become_output += chunk
 425 
 426                     # need to check every line because we might get lectured
 427                     # and we might get the middle of a line in a chunk
 428                     for l in become_output.splitlines(True):
 429                         if self.become.check_success(l):
 430                             become_sucess = True
 431                             break
 432                         elif self.become.check_password_prompt(l):
 433                             passprompt = True
 434                             break
 435 
 436                 if passprompt:
 437                     if self.become:
 438                         become_pass = self.become.get_option('become_pass', playcontext=self._play_context)
 439                         chan.sendall(to_bytes(become_pass, errors='surrogate_or_strict') + b'\n')
 440                     else:
 441                         raise AnsibleError("A password is required but none was supplied")
 442                 else:
 443                     no_prompt_out += become_output
 444                     no_prompt_err += become_output
 445         except socket.timeout:
 446             raise AnsibleError('ssh timed out waiting for privilege escalation.\n' + become_output)
 447 
 448         stdout = b''.join(chan.makefile('rb', bufsize))
 449         stderr = b''.join(chan.makefile_stderr('rb', bufsize))
 450 
 451         return (chan.recv_exit_status(), no_prompt_out + stdout, no_prompt_out + stderr)

 453     def put_file(self, in_path, out_path):
 454         ''' transfer a file from local to remote '''
 455 
 456         super(Connection, self).put_file(in_path, out_path)
 457 
 458         display.vvv("PUT %s TO %s" % (in_path, out_path), host=self._play_context.remote_addr)
 459 
 460         if not os.path.exists(to_bytes(in_path, errors='surrogate_or_strict')):
 461             raise AnsibleFileNotFound("file or module does not exist: %s" % in_path)
 462 
 463         try:
 464             self.sftp = self.ssh.open_sftp()
 465         except Exception as e:
 466             raise AnsibleError("failed to open a SFTP connection (%s)" % e)
 467 
 468         try:
 469             self.sftp.put(to_bytes(in_path, errors='surrogate_or_strict'), to_bytes(out_path, errors='surrogate_or_strict'))
 470         except IOError:
 471             raise AnsibleError("failed to transfer file to %s" % out_path)
 472 
 473     def _connect_sftp(self):
 474 
 475         cache_key = "%s__%s__" % (self._play_context.remote_addr, self._play_context.remote_user)
 476         if cache_key in SFTP_CONNECTION_CACHE:
 477             return SFTP_CONNECTION_CACHE[cache_key]
 478         else:
 479             result = SFTP_CONNECTION_CACHE[cache_key] = self._connect().ssh.open_sftp()
 480             return result

 482     def fetch_file(self, in_path, out_path):
 483         ''' save a remote file to the specified path '''
 484 
 485         super(Connection, self).fetch_file(in_path, out_path)
 486 
 487         display.vvv("FETCH %s TO %s" % (in_path, out_path), host=self._play_context.remote_addr)
 488 
 489         try:
 490             self.sftp = self._connect_sftp()
 491         except Exception as e:
 492             raise AnsibleError("failed to open a SFTP connection (%s)" % to_native(e))
 493 
 494         try:
 495             self.sftp.get(to_bytes(in_path, errors='surrogate_or_strict'), to_bytes(out_path, errors='surrogate_or_strict'))
 496         except IOError:
 497             raise AnsibleError("failed to transfer file from %s" % in_path)

 499     def _any_keys_added(self):
 500 
 501         for hostname, keys in iteritems(self.ssh._host_keys):
 502             for keytype, key in iteritems(keys):
 503                 added_this_time = getattr(key, '_added_by_ansible_this_time', False)
 504                 if added_this_time:
 505                     return True
 506         return False

 508     def _save_ssh_host_keys(self, filename):
 509         '''
 510         not using the paramiko save_ssh_host_keys function as we want to add new SSH keys at the bottom so folks
 511         don't complain about it :)
 512         '''
 513 
 514         if not self._any_keys_added():
 515             return False
 516 
 517         path = os.path.expanduser("~/.ssh")
 518         makedirs_safe(path)
 519 
 520         with open(filename, 'w') as f:
 521 
 522             for hostname, keys in iteritems(self.ssh._host_keys):
 523 
 524                 for keytype, key in iteritems(keys):
 525 
 526                     # was f.write
 527                     added_this_time = getattr(key, '_added_by_ansible_this_time', False)
 528                     if not added_this_time:
 529                         f.write("%s %s %s\n" % (hostname, keytype, key.get_base64()))
 530 
 531             for hostname, keys in iteritems(self.ssh._host_keys):
 532 
 533                 for keytype, key in iteritems(keys):
 534                     added_this_time = getattr(key, '_added_by_ansible_this_time', False)
 535                     if added_this_time:
 536                         f.write("%s %s %s\n" % (hostname, keytype, key.get_base64()))

 542     def close(self):
 543         ''' terminate the connection '''
 544 
 545         cache_key = self._cache_key()
 546         SSH_CONNECTION_CACHE.pop(cache_key, None)
 547         SFTP_CONNECTION_CACHE.pop(cache_key, None)
 548 
 549         if hasattr(self, 'sftp'):
 550             if self.sftp is not None:
 551                 self.sftp.close()
 552 
 553         if self.get_option('host_key_checking') and self.get_option('record_host_keys') and self._any_keys_added():
 554 
 555             # add any new SSH host keys -- warning -- this could be slow
 556             # (This doesn't acquire the connection lock because it needs
 557             # to exclude only other known_hosts writers, not connections
 558             # that are starting up.)
 559             lockfile = self.keyfile.replace("known_hosts", ".known_hosts.lock")
 560             dirname = os.path.dirname(self.keyfile)
 561             makedirs_safe(dirname)
 562 
 563             KEY_LOCK = open(lockfile, 'w')
 564             fcntl.lockf(KEY_LOCK, fcntl.LOCK_EX)
 565 
 566             try:
 567                 # just in case any were added recently
 568 
 569                 self.ssh.load_system_host_keys()
 570                 self.ssh._host_keys.update(self.ssh._system_host_keys)
 571 
 572                 # gather information about the current key file, so
 573                 # we can ensure the new file has the correct mode/owner
 574 
 575                 key_dir = os.path.dirname(self.keyfile)
 576                 if os.path.exists(self.keyfile):
 577                     key_stat = os.stat(self.keyfile)
 578                     mode = key_stat.st_mode
 579                     uid = key_stat.st_uid
 580                     gid = key_stat.st_gid
 581                 else:
 582                     mode = 33188
 583                     uid = os.getuid()
 584                     gid = os.getgid()
 585 
 586                 # Save the new keys to a temporary file and move it into place
 587                 # rather than rewriting the file. We set delete=False because
 588                 # the file will be moved into place rather than cleaned up.
 589 
 590                 tmp_keyfile = tempfile.NamedTemporaryFile(dir=key_dir, delete=False)
 591                 os.chmod(tmp_keyfile.name, mode & 0o7777)
 592                 os.chown(tmp_keyfile.name, uid, gid)
 593 
 594                 self._save_ssh_host_keys(tmp_keyfile.name)
 595                 tmp_keyfile.close()
 596 
 597                 os.rename(tmp_keyfile.name, self.keyfile)
 598 
 599             except Exception:
 600 
 601                 # unable to save keys, including scenario when key was invalid
 602                 # and caught earlier
 603                 traceback.print_exc()
 604             fcntl.lockf(KEY_LOCK, fcntl.LOCK_UN)
 605 
 606         self.ssh.close()
 607         self._connected = False
