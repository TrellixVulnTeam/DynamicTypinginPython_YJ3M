  30 def ensure_connect(func):
  31     @wraps(func)
  32     def wrapped(self, *args, **kwargs):
  33         if not self._connected:
  34             self._connect()
  35         return func(self, *args, **kwargs)
  36     return wrapped

 105     @staticmethod
 106     def _split_ssh_args(argstring):
 107         """
 108         Takes a string like '-o Foo=1 -o Bar="foo bar"' and returns a
 109         list ['-o', 'Foo=1', '-o', 'Bar=foo bar'] that can be added to
 110         the argument list. The list will not contain any empty elements.
 111         """
 112         try:
 113             # Python 2.6.x shlex doesn't handle unicode type so we have to
 114             # convert args to byte string for that case.  More efficient to
 115             # try without conversion first but python2.6 doesn't throw an
 116             # exception, it merely mangles the output:
 117             # >>> shlex.split(u't e')
 118             # ['t\x00\x00\x00', '\x00\x00\x00e\x00\x00\x00']
 119             return [to_text(x.strip()) for x in shlex.split(to_bytes(argstring)) if x.strip()]
 120         except AttributeError:
 121             # In Python3, shlex.split doesn't work on a byte string.
 122             return [to_text(x.strip()) for x in shlex.split(argstring) if x.strip()]

 133     @ensure_connect
 134     @abstractmethod
 135     def exec_command(self, cmd, in_data=None, sudoable=True):
 136         """Run a command on the remote host.
 137 
 138         :arg cmd: byte string containing the command
 139         :kwarg in_data: If set, this data is passed to the command's stdin.
 140             This is used to implement pipelining.  Currently not all
 141             connection plugins implement pipelining.
 142         :kwarg sudoable: Tell the connection plugin if we're executing
 143             a command via a privilege escalation mechanism.  This may affect
 144             how the connection plugin returns data.  Note that not all
 145             connections can handle privilege escalation.
 146         :returns: a tuple of (return code, stdout, stderr)  The return code is
 147             an int while stdout and stderr are both byte strings.
 148 
 149         When a command is executed, it goes through multiple commands to get
 150         there.  It looks approximately like this::
 151 
 152             [LocalShell] ConnectionCommand [UsersLoginShell (*)] ANSIBLE_SHELL_EXECUTABLE [(BecomeCommand ANSIBLE_SHELL_EXECUTABLE)] Command
 153         :LocalShell: Is optional.  It is run locally to invoke the
 154             ``Connection Command``.  In most instances, the
 155             ``ConnectionCommand`` can be invoked directly instead.  The ssh
 156             connection plugin which can have values that need expanding
 157             locally specified via ssh_args is the sole known exception to
 158             this.  Shell metacharacters in the command itself should be
 159             processed on the remote machine, not on the local machine so no
 160             shell is needed on the local machine.  (Example, ``/bin/sh``)
 161         :ConnectionCommand: This is the command that connects us to the remote
 162             machine to run the rest of the command.  ``ansible_user``,
 163             ``ansible_ssh_host`` and so forth are fed to this piece of the
 164             command to connect to the correct host (Examples ``ssh``,
 165             ``chroot``)
 166         :UsersLoginShell: This shell may or may not be created depending on
 167             the ConnectionCommand used by the connection plugin.  This is the
 168             shell that the ``ansible_user`` has configured as their login
 169             shell.  In traditional UNIX parlance, this is the last field of
 170             a user's ``/etc/passwd`` entry   We do not specifically try to run
 171             the ``UsersLoginShell`` when we connect.  Instead it is implicit
 172             in the actions that the ``ConnectionCommand`` takes when it
 173             connects to a remote machine.  ``ansible_shell_type`` may be set
 174             to inform ansible of differences in how the ``UsersLoginShell``
 175             handles things like quoting if a shell has different semantics
 176             than the Bourne shell.
 177         :ANSIBLE_SHELL_EXECUTABLE: This is the shell set via the inventory var
 178             ``ansible_shell_executable`` or via
 179             ``constants.DEFAULT_EXECUTABLE`` if the inventory var is not set.
 180             We explicitly invoke this shell so that we have predictable
 181             quoting rules at this point.  ``ANSIBLE_SHELL_EXECUTABLE`` is only
 182             settable by the user because some sudo setups may only allow
 183             invoking a specific shell.  (For instance, ``/bin/bash`` may be
 184             allowed but ``/bin/sh``, our default, may not).  We invoke this
 185             twice, once after the ``ConnectionCommand`` and once after the
 186             ``BecomeCommand``.  After the ConnectionCommand, this is run by
 187             the ``UsersLoginShell``.  After the ``BecomeCommand`` we specify
 188             that the ``ANSIBLE_SHELL_EXECUTABLE`` is being invoked directly.
 189         :BecomeComand ANSIBLE_SHELL_EXECUTABLE: Is the command that performs
 190             privilege escalation.  Setting this up is performed by the action
 191             plugin prior to running ``exec_command``. So we just get passed
 192             :param:`cmd` which has the BecomeCommand already added.
 193             (Examples: sudo, su)  If we have a BecomeCommand then we will
 194             invoke a ANSIBLE_SHELL_EXECUTABLE shell inside of it so that we
 195             have a consistent view of quoting.
 196         :Command: Is the command we're actually trying to run remotely.
 197             (Examples: mkdir -p $HOME/.ansible, python $HOME/.ansible/tmp-script-file)
 198         """
 199         pass

 218     def connection_lock(self):
 219         f = self._play_context.connection_lockfd
 220         display.vvvv('CONNECTION: pid %d waiting for lock on %d' % (os.getpid(), f), host=self._play_context.remote_addr)
 221         fcntl.lockf(f, fcntl.LOCK_EX)
 222         display.vvvv('CONNECTION: pid %d acquired lock on %d' % (os.getpid(), f), host=self._play_context.remote_addr)
 223 
 224     def connection_unlock(self):
 225         f = self._play_context.connection_lockfd
 226         fcntl.lockf(f, fcntl.LOCK_UN)
 227         display.vvvv('CONNECTION: pid %d released lock on %d' % (os.getpid(), f), host=self._play_context.remote_addr)

 266 class NetworkConnectionBase(ConnectionBase):
 267     """
 268     A base class for network-style connections.
 269     """
 270 
 271     force_persistence = True
 272     # Do not use _remote_is_local in other connections
 273     _remote_is_local = True
 274 
 275     def __init__(self, play_context, new_stdin, *args, **kwargs):
 276         super(NetworkConnectionBase, self).__init__(play_context, new_stdin, *args, **kwargs)
 277         self._messages = []
 278         self._conn_closed = False
 279 
 280         self._network_os = self._play_context.network_os
 281 
 282         self._local = connection_loader.get('local', play_context, '/dev/null')
 283         self._local.set_options()
 284 
 285         self._sub_plugin = {}
 286         self._cached_variables = (None, None, None)
 287 
 288         # reconstruct the socket_path and set instance values accordingly
 289         self._ansible_playbook_pid = kwargs.get('ansible_playbook_pid')
 290         self._update_connection_state()
 291 
 292     def __getattr__(self, name):
 293         try:
 294             return self.__dict__[name]
 295         except KeyError:
 296             if not name.startswith('_'):
 297                 plugin = self._sub_plugin.get('obj')
 298                 if plugin:
 299                     method = getattr(plugin, name, None)
 300                     if method is not None:
 301                         return method
 302             raise AttributeError("'%s' object has no attribute '%s'" % (self.__class__.__name__, name))
 303 
 304     def exec_command(self, cmd, in_data=None, sudoable=True):
 305         return self._local.exec_command(cmd, in_data, sudoable)
 306 
 307     def queue_message(self, level, message):
 308         """
 309         Adds a message to the queue of messages waiting to be pushed back to the controller process.
 310 
 311         :arg level: A string which can either be the name of a method in display, or 'log'. When
 312             the messages are returned to task_executor, a value of log will correspond to
 313             ``display.display(message, log_only=True)``, while another value will call ``display.[level](message)``
 314         """
 315         self._messages.append((level, message))
 316 
 317     def pop_messages(self):
 318         messages, self._messages = self._messages, []
 319         return messages
 320 
 321     def put_file(self, in_path, out_path):
 322         """Transfer a file from local to remote"""
 323         return self._local.put_file(in_path, out_path)
 324 
 325     def fetch_file(self, in_path, out_path):
 326         """Fetch a file from remote to local"""
 327         return self._local.fetch_file(in_path, out_path)

 329     def reset(self):
 330         '''
 331         Reset the connection
 332         '''
 333         if self._socket_path:
 334             self.queue_message('vvvv', 'resetting persistent connection for socket_path %s' % self._socket_path)
 335             self.close()
 336         self.queue_message('vvvv', 'reset call on connection instance')
 337 
 338     def close(self):
 339         self._conn_closed = True
 340         if self._connected:
 341             self._connected = False
 342 
 343     def set_options(self, task_keys=None, var_options=None, direct=None):
 344         super(NetworkConnectionBase, self).set_options(task_keys=task_keys, var_options=var_options, direct=direct)
 345         if self.get_option('persistent_log_messages'):
 346             warning = "Persistent connection logging is enabled for %s. This will log ALL interactions" % self._play_context.remote_addr
 347             logpath = getattr(C, 'DEFAULT_LOG_PATH')
 348             if logpath is not None:
 349                 warning += " to %s" % logpath
 350             self.queue_message('warning', "%s and WILL NOT redact sensitive configuration like passwords. USE WITH CAUTION!" % warning)
 351 
 352         if self._sub_plugin.get('obj') and self._sub_plugin.get('type') != 'external':
 353             try:
 354                 self._sub_plugin['obj'].set_options(task_keys=task_keys, var_options=var_options, direct=direct)
 355             except AttributeError:
 356                 pass

 358     def _update_connection_state(self):
 359         '''
 360         Reconstruct the connection socket_path and check if it exists
 361 
 362         If the socket path exists then the connection is active and set
 363         both the _socket_path value to the path and the _connected value
 364         to True.  If the socket path doesn't exist, leave the socket path
 365         value to None and the _connected value to False
 366         '''
 367         ssh = connection_loader.get('ssh', class_only=True)
 368         control_path = ssh._create_control_path(
 369             self._play_context.remote_addr, self._play_context.port,
 370             self._play_context.remote_user, self._play_context.connection,
 371             self._ansible_playbook_pid
 372         )
 373 
 374         tmp_path = unfrackpath(C.PERSISTENT_CONTROL_PATH_DIR)
 375         socket_path = unfrackpath(control_path % dict(directory=tmp_path))
 376 
 377         if os.path.exists(socket_path):
 378             self._connected = True
 379             self._socket_path = socket_path
 380 
 381     def _log_messages(self, message):
 382         if self.get_option('persistent_log_messages'):
 383             self.queue_message('log', message)
