  37 class Connection(ConnectionBase):
  38     ''' Local based connections '''
  39 
  40     transport = 'local'
  41     has_pipelining = True
  42 
  43     def __init__(self, *args, **kwargs):
  44 
  45         super(Connection, self).__init__(*args, **kwargs)
  46         self.cwd = None
  47 
  48     def _connect(self):
  49         ''' connect to the local host; nothing to do here '''
  50 
  51         # Because we haven't made any remote connection we're running as
  52         # the local user, rather than as whatever is configured in
  53         # remote_user.
  54         self._play_context.remote_user = getpass.getuser()
  55 
  56         if not self._connected:
  57             display.vvv(u"ESTABLISH LOCAL CONNECTION FOR USER: {0}".format(self._play_context.remote_user), host=self._play_context.remote_addr)
  58             self._connected = True
  59         return self
  60 
  61     def exec_command(self, cmd, in_data=None, sudoable=True):
  62         ''' run a command on the local host '''
  63 
  64         super(Connection, self).exec_command(cmd, in_data=in_data, sudoable=sudoable)
  65 
  66         display.debug("in local.exec_command()")
  67 
  68         executable = C.DEFAULT_EXECUTABLE.split()[0] if C.DEFAULT_EXECUTABLE else None
  69 
  70         if not os.path.exists(to_bytes(executable, errors='surrogate_or_strict')):
  71             raise AnsibleError("failed to find the executable specified %s."
  72                                " Please verify if the executable exists and re-try." % executable)
  73 
  74         display.vvv(u"EXEC {0}".format(to_text(cmd)), host=self._play_context.remote_addr)
  75         display.debug("opening command with Popen()")
  76 
  77         if isinstance(cmd, (text_type, binary_type)):
  78             cmd = to_bytes(cmd)
  79         else:
  80             cmd = map(to_bytes, cmd)
  81 
  82         p = subprocess.Popen(
  83             cmd,
  84             shell=isinstance(cmd, (text_type, binary_type)),
  85             executable=executable,
  86             cwd=self.cwd,
  87             stdin=subprocess.PIPE,
  88             stdout=subprocess.PIPE,
  89             stderr=subprocess.PIPE,
  90         )
  91         display.debug("done running command with Popen()")
  92 
  93         if self.become and self.become.expect_prompt() and sudoable:
  94             fcntl.fcntl(p.stdout, fcntl.F_SETFL, fcntl.fcntl(p.stdout, fcntl.F_GETFL) | os.O_NONBLOCK)
  95             fcntl.fcntl(p.stderr, fcntl.F_SETFL, fcntl.fcntl(p.stderr, fcntl.F_GETFL) | os.O_NONBLOCK)
  96             selector = selectors.DefaultSelector()
  97             selector.register(p.stdout, selectors.EVENT_READ)
  98             selector.register(p.stderr, selectors.EVENT_READ)
  99 
 100             become_output = b''
 101             try:
 102                 while not self.become.check_success(become_output) and not self.become.check_password_prompt(become_output):
 103                     events = selector.select(self._play_context.timeout)
 104                     if not events:
 105                         stdout, stderr = p.communicate()
 106                         raise AnsibleError('timeout waiting for privilege escalation password prompt:\n' + to_native(become_output))
 107 
 108                     for key, event in events:
 109                         if key.fileobj == p.stdout:
 110                             chunk = p.stdout.read()
 111                         elif key.fileobj == p.stderr:
 112                             chunk = p.stderr.read()
 113 
 114                     if not chunk:
 115                         stdout, stderr = p.communicate()
 116                         raise AnsibleError('privilege output closed while waiting for password prompt:\n' + to_native(become_output))
 117                     become_output += chunk
 118             finally:
 119                 selector.close()
 120 
 121             if not self.become.check_success(become_output):
 122                 become_pass = self.become.get_option('become_pass', playcontext=self._play_context)
 123                 p.stdin.write(to_bytes(become_pass, errors='surrogate_or_strict') + b'\n')
 124             fcntl.fcntl(p.stdout, fcntl.F_SETFL, fcntl.fcntl(p.stdout, fcntl.F_GETFL) & ~os.O_NONBLOCK)
 125             fcntl.fcntl(p.stderr, fcntl.F_SETFL, fcntl.fcntl(p.stderr, fcntl.F_GETFL) & ~os.O_NONBLOCK)
 126 
 127         display.debug("getting output with communicate()")
 128         stdout, stderr = p.communicate(in_data)
 129         display.debug("done communicating")
 130 
 131         display.debug("done with local.exec_command()")
 132         return (p.returncode, stdout, stderr)

 134     def put_file(self, in_path, out_path):
 135         ''' transfer a file from local to local '''
 136 
 137         super(Connection, self).put_file(in_path, out_path)
 138 
 139         in_path = unfrackpath(in_path, basedir=self.cwd)
 140         out_path = unfrackpath(out_path, basedir=self.cwd)
 141 
 142         display.vvv(u"PUT {0} TO {1}".format(in_path, out_path), host=self._play_context.remote_addr)
 143         if not os.path.exists(to_bytes(in_path, errors='surrogate_or_strict')):
 144             raise AnsibleFileNotFound("file or module does not exist: {0}".format(to_native(in_path)))
 145         try:
 146             shutil.copyfile(to_bytes(in_path, errors='surrogate_or_strict'), to_bytes(out_path, errors='surrogate_or_strict'))
 147         except shutil.Error:
 148             raise AnsibleError("failed to copy: {0} and {1} are the same".format(to_native(in_path), to_native(out_path)))
 149         except IOError as e:
 150             raise AnsibleError("failed to transfer file to {0}: {1}".format(to_native(out_path), to_native(e)))
 151 
 152     def fetch_file(self, in_path, out_path):
 153         ''' fetch a file from local to local -- for compatibility '''
 154 
 155         super(Connection, self).fetch_file(in_path, out_path)
 156 
 157         display.vvv(u"FETCH {0} TO {1}".format(in_path, out_path), host=self._play_context.remote_addr)
 158         self.put_file(in_path, out_path)
 159 
 160     def close(self):
 161         ''' terminate the connection; nothing to do here '''
 162         self._connected = False
