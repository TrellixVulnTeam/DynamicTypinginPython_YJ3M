  22 class ShellModule(ShellBase):
  23 
  24     # Common shell filenames that this plugin handles.
  25     # Note: sh is the default shell plugin so this plugin may also be selected
  26     # This code needs to be SH-compliant. BASH-isms will not work if /bin/sh points to a non-BASH shell.
  27 
  28     # if the filename is not listed in any Shell plugin.
  29     COMPATIBLE_SHELLS = frozenset(('sh', 'zsh', 'bash', 'dash', 'ksh'))
  30     # Family of shells this has.  Must match the filename without extension
  31     SHELL_FAMILY = 'sh'
  32 
  33     # commonly used
  34     ECHO = 'echo'
  35     COMMAND_SEP = ';'
  36 
  37     # How to end lines in a python script one-liner
  38     _SHELL_EMBEDDED_PY_EOL = '\n'
  39     _SHELL_REDIRECT_ALLNULL = '> /dev/null 2>&1'
  40     _SHELL_AND = '&&'
  41     _SHELL_OR = '||'
  42     _SHELL_SUB_LEFT = '"`'
  43     _SHELL_SUB_RIGHT = '`"'
  44     _SHELL_GROUP_LEFT = '('
  45     _SHELL_GROUP_RIGHT = ')'
  46 
  47     def checksum(self, path, python_interp):
  48         # In the following test, each condition is a check and logical
  49         # comparison (|| or &&) that sets the rc value.  Every check is run so
  50         # the last check in the series to fail will be the rc that is returned.
  51         #
  52         # If a check fails we error before invoking the hash functions because
  53         # hash functions may successfully take the hash of a directory on BSDs
  54         # (UFS filesystem?) which is not what the rest of the ansible code expects
  55         #
  56         # If all of the available hashing methods fail we fail with an rc of 0.
  57         # This logic is added to the end of the cmd at the bottom of this function.
  58 
  59         # Return codes:
  60         # checksum: success!
  61         # 0: Unknown error
  62         # 1: Remote file does not exist
  63         # 2: No read permissions on the file
  64         # 3: File is a directory
  65         # 4: No python interpreter
  66 
  67         # Quoting gets complex here.  We're writing a python string that's
  68         # used by a variety of shells on the remote host to invoke a python
  69         # "one-liner".
  70         shell_escaped_path = shlex_quote(path)
  71         test = "rc=flag; [ -r %(p)s ] %(shell_or)s rc=2; [ -f %(p)s ] %(shell_or)s rc=1; [ -d %(p)s ] %(shell_and)s rc=3; %(i)s -V 2>/dev/null %(shell_or)s rc=4; [ x\"$rc\" != \"xflag\" ] %(shell_and)s echo \"${rc}  \"%(p)s %(shell_and)s exit 0" % dict(p=shell_escaped_path, i=python_interp, shell_and=self._SHELL_AND, shell_or=self._SHELL_OR)  # NOQA
  72         csums = [
  73             u"({0} -c 'import hashlib; BLOCKSIZE = 65536; hasher = hashlib.sha1();{2}afile = open(\"'{1}'\", \"rb\"){2}buf = afile.read(BLOCKSIZE){2}while len(buf) > 0:{2}\thasher.update(buf){2}\tbuf = afile.read(BLOCKSIZE){2}afile.close(){2}print(hasher.hexdigest())' 2>/dev/null)".format(python_interp, shell_escaped_path, self._SHELL_EMBEDDED_PY_EOL),  # NOQA  Python > 2.4 (including python3)
  74             u"({0} -c 'import sha; BLOCKSIZE = 65536; hasher = sha.sha();{2}afile = open(\"'{1}'\", \"rb\"){2}buf = afile.read(BLOCKSIZE){2}while len(buf) > 0:{2}\thasher.update(buf){2}\tbuf = afile.read(BLOCKSIZE){2}afile.close(){2}print(hasher.hexdigest())' 2>/dev/null)".format(python_interp, shell_escaped_path, self._SHELL_EMBEDDED_PY_EOL),  # NOQA  Python == 2.4
  75         ]
  76 
  77         cmd = (" %s " % self._SHELL_OR).join(csums)
  78         cmd = "%s; %s %s (echo \'0  \'%s)" % (test, cmd, self._SHELL_OR, shell_escaped_path)
  79         return cmd
