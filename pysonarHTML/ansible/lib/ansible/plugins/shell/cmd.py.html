  17 import re
  18 
  19 from ansible.plugins.shell.powershell import ShellModule as PSShellModule
  20 
  21 # these are the metachars that have a special meaning in cmd that we want to escape when quoting
  22 _find_unsafe = re.compile(r'[\s\(\)\%\!^\"\<\>\&\|]').search
  23 
  24 
  25 class ShellModule(PSShellModule):
  26 
  27     # Common shell filenames that this plugin handles
  28     COMPATIBLE_SHELLS = frozenset()
  29     # Family of shells this has.  Must match the filename without extension
  30     SHELL_FAMILY = 'cmd'
  31 
  32     _SHELL_REDIRECT_ALLNULL = '>nul 2>&1'
  33     _SHELL_AND = '&&'
  34 
  35     # Used by various parts of Ansible to do Windows specific changes
  36     _IS_WINDOWS = True
  37 
  38     def quote(self, s):
  39         # cmd does not support single quotes that the shlex_quote uses. We need to override the quoting behaviour to
  40         # better match cmd.exe.
  41         # https://blogs.msdn.microsoft.com/twistylittlepassagesallalike/2011/04/23/everyone-quotes-command-line-arguments-the-wrong-way/
  42 
  43         # Return an empty argument
  44         if not s:
  45             return '""'
  46 
  47         if _find_unsafe(s) is None:
  48             return s
  49 
  50         # Escape the metachars as we are quoting the string to stop cmd from interpreting that metachar. For example
  51         # 'file &whoami.exe' would result in 'file $(whoami.exe)' instead of the literal string
  52         # https://stackoverflow.com/questions/3411771/multiple-character-replace-with-python
  53         for c in '^()%!"<>&|':  # '^' must be the first char that we scan and replace
  54             if c in s:
  55                 # I can't find any docs that explicitly say this but to escape ", it needs to be prefixed with \^.
  56                 s = s.replace(c, ("\\^" if c == '"' else "^") + c)
  57 
  58         return '^"' + s + '^"'
