  35 class ShellBase(AnsiblePlugin):
  36     def __init__(self):
  37 
  38         super(ShellBase, self).__init__()

  40         self.env = {}
  41         self.tmpdir = None

  44     def _normalize_system_tmpdirs(self):
  45         # Normalize the tmp directory strings. We don't use expanduser/expandvars because those
  46         # can vary between remote user and become user.  Therefore the safest practice will be for
  47         # this to always be specified as full paths)
  48         normalized_paths = [d.rstrip('/') for d in self.get_option('system_tmpdirs')]
  49 
  50         # Make sure all system_tmpdirs are absolute otherwise they'd be relative to the login dir
  51         # which is almost certainly going to fail in a cornercase.
  52         if not all(os.path.isabs(d) for d in normalized_paths):
  53             raise AnsibleError('The configured system_tmpdirs contains a relative path: {0}. All'
  54                                ' system_tmpdirs must be absolute'.format(to_native(normalized_paths)))
  55 
  56         self.set_option('system_tmpdirs', normalized_paths)

  58     def set_options(self, task_keys=None, var_options=None, direct=None):
  59 
  60         super(ShellBase, self).set_options(task_keys=task_keys, var_options=var_options, direct=direct)
  61 
  62         # set env if needed, deal with environment's 'dual nature' list of dicts or dict
  63         env = self.get_option('environment')
  64         if isinstance(env, list):
  65             for env_dict in env:
  66                 self.env.update(env_dict)
  67         else:
  68             self.env.update(env)
  69 
  70         # We can remove the try: except in the future when we make ShellBase a proper subset of
  71         # *all* shells.  Right now powershell and third party shells which do not use the
  72         # shell_common documentation fragment (and so do not have system_tmpdirs) will fail
  73         try:
  74             self._normalize_system_tmpdirs()
  75         except KeyError:
  76             pass

  78     @staticmethod
  79     def _generate_temp_dir_name():
  80         return 'ansible-tmp-%s-%s-%s' % (time.time(), os.getpid(), random.randint(0, 2**48))

  82     def env_prefix(self, **kwargs):
  83         return ' '.join(['%s=%s' % (k, shlex_quote(text_type(v))) for k, v in kwargs.items()])

 117     def set_user_facl(self, paths, user, mode):
 118         """Only sets acls for users as that's really all we need"""
 119         cmd = ['setfacl', '-m', 'u:%s:%s' % (user, mode)]
 120         cmd.extend(paths)
 121         cmd = [shlex_quote(c) for c in cmd]

 123         return ' '.join(cmd)

 125     def remove(self, path, recurse=False):
 126         path = shlex_quote(path)
 127         cmd = 'rm -f '
 128         if recurse:
 129             cmd += '-r '
 130         return cmd + "%s %s" % (path, self._SHELL_REDIRECT_ALLNULL)
 131 
 132     def exists(self, path):
 133         cmd = ['test', '-e', shlex_quote(path)]
 134         return ' '.join(cmd)

 136     def mkdtemp(self, basefile=None, system=False, mode=0o700, tmpdir=None):
 137         if not basefile:
 138             basefile = self.__class__._generate_temp_dir_name()
 139 
 140         # When system is specified we have to create this in a directory where
 141         # other users can read and access the tmp directory.
 142         # This is because we use system to create tmp dirs for unprivileged users who are
 143         # sudo'ing to a second unprivileged user.
 144         # The 'system_tmpdirs' setting defines dirctories we can use for this purpose
 145         # the default are, /tmp and /var/tmp.
 146         # So we only allow one of those locations if system=True, using the
 147         # passed in tmpdir if it is valid or the first one from the setting if not.
 148 
 149         if system:
 150             if tmpdir:
 151                 tmpdir = tmpdir.rstrip('/')
 152 
 153             if tmpdir in self.get_option('system_tmpdirs'):
 154                 basetmpdir = tmpdir
 155             else:
 156                 basetmpdir = self.get_option('system_tmpdirs')[0]
 157         else:
 158             if tmpdir is None:
 159                 basetmpdir = self.get_option('remote_tmp')
 160             else:
 161                 basetmpdir = tmpdir
 162 
 163         basetmp = self.join_path(basetmpdir, basefile)
 164 
 165         # use mkdir -p to ensure parents exist, but mkdir fullpath to ensure last one is created by us
 166         cmd = 'mkdir -p %s echo %s %s' % (self._SHELL_SUB_LEFT, basetmpdir, self._SHELL_SUB_RIGHT)
 167         cmd += '%s mkdir %s' % (self._SHELL_AND, basetmp)
 168         cmd += ' %s echo %s=%s echo %s %s' % (self._SHELL_AND, basefile, self._SHELL_SUB_LEFT, basetmp, self._SHELL_SUB_RIGHT)
 169 
 170         # change the umask in a subshell to achieve the desired mode
 171         # also for directories created with `mkdir -p`
 172         if mode:
 173             tmp_umask = 0o777 & ~mode
 174             cmd = '%s umask %o %s %s %s' % (self._SHELL_GROUP_LEFT, tmp_umask, self._SHELL_AND, cmd, self._SHELL_GROUP_RIGHT)
 175 
 176         return cmd

 178     def expand_user(self, user_home_path, username=''):
 179         ''' Return a command to expand tildes in a path
 180 
 181         It can be either "~" or "~username". We just ignore $HOME
 182         We use the POSIX definition of a username:
 183             http://pubs.opengroup.org/onlinepubs/000095399/basedefs/xbd_chap03.html#tag_03_426
 184             http://pubs.opengroup.org/onlinepubs/000095399/basedefs/xbd_chap03.html#tag_03_276
 185 
 186             Falls back to 'current working directory' as we assume 'home is where the remote user ends up'
 187         '''
 188 
 189         # Check that the user_path to expand is safe
 190         if user_home_path != '~':
 191             if not _USER_HOME_PATH_RE.match(user_home_path):
 192                 # shlex_quote will make the shell return the string verbatim
 193                 user_home_path = shlex_quote(user_home_path)
 194         elif username:
 195             # if present the user name is appended to resolve "that user's home"
 196             user_home_path += username
 197 
 198         return 'echo %s' % user_home_path

 200     def pwd(self):
 201         """Return the working directory after connecting"""
 202         return 'echo %spwd%s' % (self._SHELL_SUB_LEFT, self._SHELL_SUB_RIGHT)
 203 
 204     def build_module_command(self, env_string, shebang, cmd, arg_path=None):
 205         # don't quote the cmd if it's an empty string, because this will break pipelining mode
 206         if cmd.strip() != '':
 207             cmd = shlex_quote(cmd)
 208 
 209         cmd_parts = []
 210         if shebang:
 211             shebang = shebang.replace("#!", "").strip()
 212         else:
 213             shebang = ""
 214         cmd_parts.extend([env_string.strip(), shebang, cmd])
 215         if arg_path is not None:
 216             cmd_parts.append(arg_path)
 217         new_cmd = " ".join(cmd_parts)
 218         return new_cmd

 220     def append_command(self, cmd, cmd_to_append):
 221         """Append an additional command if supported by the shell"""
 222 
 223         if self._SHELL_AND:
 224             cmd += ' %s %s' % (self._SHELL_AND, cmd_to_append)
 225 
 226         return cmd
 227 
 228     def wrap_for_exec(self, cmd):
 229         """wrap script execution with any necessary decoration (eg '&' for quoted powershell script paths)"""
 230         return cmd
 231 
 232     def quote(self, cmd):
 233         """Returns a shell-escaped string that can be safely used as one token in a shell command line"""
 234         return shlex_quote(cmd)
