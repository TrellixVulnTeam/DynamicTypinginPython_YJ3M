  40 def _parse_clixml(data, stream="Error"):
  41     """
  42     Takes a byte string like '#< CLIXML\r\n<Objs...' and extracts the stream
  43     message encoded in the XML data. CLIXML is used by PowerShell to encode
  44     multiple objects in stderr.
  45     """
  46     clixml = ET.fromstring(data.split(b"\r\n", 1)[-1])
  47     namespace_match = re.match(r'{(.*)}', clixml.tag)
  48     namespace = "{%s}" % namespace_match.group(1) if namespace_match else ""
  49 
  50     strings = clixml.findall("./%sS" % namespace)
  51     lines = [e.text.replace('_x000D__x000A_', '') for e in strings if e.attrib.get('S') == stream]
  52     return to_bytes('\r\n'.join(lines))

  76     def join_path(self, *args):
  77         # use normpath() to remove doubled slashed and convert forward to backslashes
  78         parts = [ntpath.normpath(self._unquote(arg)) for arg in args]
  79 
  80         # Becuase ntpath.join treats any component that begins with a backslash as an absolute path,
  81         # we have to strip slashes from at least the beginning, otherwise join will ignore all previous
  82         # path components except for the drive.
  83         return ntpath.join(parts[0], *[part.strip('\\') for part in parts[1:]])

  85     def get_remote_filename(self, pathname):
  86         # powershell requires that script files end with .ps1
  87         base_name = os.path.basename(pathname.strip())
  88         name, ext = os.path.splitext(base_name.strip())
  89         if ext.lower() not in ['.ps1', '.exe']:
  90             return name + '.ps1'
  91 
  92         return base_name.strip()

  94     def path_has_trailing_slash(self, path):
  95         # Allow Windows paths to be specified using either slash.
  96         path = self._unquote(path)
  97         return path.endswith('/') or path.endswith('\\')

 108     def remove(self, path, recurse=False):
 109         path = self._escape(self._unquote(path))
 110         if recurse:
 111             return self._encode_script('''Remove-Item "%s" -Force -Recurse;''' % path)
 112         else:
 113             return self._encode_script('''Remove-Item "%s" -Force;''' % path)

 115     def mkdtemp(self, basefile=None, system=False, mode=None, tmpdir=None):
 116         # Windows does not have an equivalent for the system temp files, so
 117         # the param is ignored
 118         if not basefile:
 119             basefile = self.__class__._generate_temp_dir_name()
 120         basefile = self._escape(self._unquote(basefile))
 121         basetmpdir = tmpdir if tmpdir else self.get_option('remote_tmp')
 122 
 123         script = '''
 124         $tmp_path = [System.Environment]::ExpandEnvironmentVariables('%s')
 125         $tmp = New-Item -Type Directory -Path $tmp_path -Name '%s'
 126         Write-Output -InputObject $tmp.FullName
 127         ''' % (basetmpdir, basefile)
 128         return self._encode_script(script.strip())

 130     def expand_user(self, user_home_path, username=''):
 131         # PowerShell only supports "~" (not "~username").  Resolve-Path ~ does
 132         # not seem to work remotely, though by default we are always starting
 133         # in the user's home directory.
 134         user_home_path = self._unquote(user_home_path)
 135         if user_home_path == '~':
 136             script = 'Write-Output (Get-Location).Path'
 137         elif user_home_path.startswith('~\\'):
 138             script = 'Write-Output ((Get-Location).Path + "%s")' % self._escape(user_home_path[1:])
 139         else:
 140             script = 'Write-Output "%s"' % self._escape(user_home_path)
 141         return self._encode_script(script)

 143     def exists(self, path):
 144         path = self._escape(self._unquote(path))
 145         script = '''
 146             If (Test-Path "%s")
 147             {
 148                 $res = 0;
 149             }
 150             Else
 151             {
 152                 $res = 1;
 153             }
 154             Write-Output "$res";
 155             Exit $res;
 156          ''' % path
 157         return self._encode_script(script)

 159     def checksum(self, path, *args, **kwargs):
 160         path = self._escape(self._unquote(path))
 161         script = '''
 162             If (Test-Path -PathType Leaf "%(path)s")
 163             {
 164                 $sp = new-object -TypeName System.Security.Cryptography.SHA1CryptoServiceProvider;
 165                 $fp = [System.IO.File]::Open("%(path)s", [System.IO.Filemode]::Open, [System.IO.FileAccess]::Read);
 166                 [System.BitConverter]::ToString($sp.ComputeHash($fp)).Replace("-", "").ToLower();
 167                 $fp.Dispose();
 168             }
 169             ElseIf (Test-Path -PathType Container "%(path)s")
 170             {
 171                 Write-Output "3";
 172             }
 173             Else
 174             {
 175                 Write-Output "1";
 176             }
 177         ''' % dict(path=path)
 178         return self._encode_script(script)

 180     def build_module_command(self, env_string, shebang, cmd, arg_path=None):
 181         bootstrap_wrapper = pkgutil.get_data("ansible.executor.powershell", "bootstrap_wrapper.ps1")
 182 
 183         # pipelining bypass
 184         if cmd == '':
 185             return self._encode_script(script=bootstrap_wrapper, strict_mode=False, preserve_rc=False)
 186 
 187         # non-pipelining
 188 
 189         cmd_parts = shlex.split(cmd, posix=False)
 190         cmd_parts = list(map(to_text, cmd_parts))
 191         if shebang and shebang.lower() == '#!powershell':
 192             if not self._unquote(cmd_parts[0]).lower().endswith('.ps1'):
 193                 # we're running a module via the bootstrap wrapper
 194                 cmd_parts[0] = '"%s.ps1"' % self._unquote(cmd_parts[0])
 195             wrapper_cmd = "type " + cmd_parts[0] + " | " + self._encode_script(script=bootstrap_wrapper, strict_mode=False, preserve_rc=False)
 196             return wrapper_cmd
 197         elif shebang and shebang.startswith('#!'):
 198             cmd_parts.insert(0, shebang[2:])
 199         elif not shebang:
 200             # The module is assumed to be a binary
 201             cmd_parts[0] = self._unquote(cmd_parts[0])
 202             cmd_parts.append(arg_path)
 203         script = '''
 204             Try
 205             {
 206                 %s
 207                 %s
 208             }
 209             Catch
 210             {
 211                 $_obj = @{ failed = $true }
 212                 If ($_.Exception.GetType)
 213                 {
 214                     $_obj.Add('msg', $_.Exception.Message)
 215                 }
 216                 Else
 217                 {
 218                     $_obj.Add('msg', $_.ToString())
 219                 }
 220                 If ($_.InvocationInfo.PositionMessage)
 221                 {
 222                     $_obj.Add('exception', $_.InvocationInfo.PositionMessage)
 223                 }
 224                 ElseIf ($_.ScriptStackTrace)
 225                 {
 226                     $_obj.Add('exception', $_.ScriptStackTrace)
 227                 }
 228                 Try
 229                 {
 230                     $_obj.Add('error_record', ($_ | ConvertTo-Json | ConvertFrom-Json))
 231                 }
 232                 Catch
 233                 {
 234                 }
 235                 Echo $_obj | ConvertTo-Json -Compress -Depth 99
 236                 Exit 1
 237             }
 238         ''' % (env_string, ' '.join(cmd_parts))
 239         return self._encode_script(script, preserve_rc=False)

 241     def wrap_for_exec(self, cmd):
 242         return '& %s; exit $LASTEXITCODE' % cmd

 244     def _unquote(self, value):
 245         '''Remove any matching quotes that wrap the given value.'''
 246         value = to_text(value or '')
 247         m = re.match(r'^\s*?\'(.*?)\'\s*?$', value)
 248         if m:
 249             return m.group(1)
 250         m = re.match(r'^\s*?"(.*?)"\s*?$', value)
 251         if m:
 252             return m.group(1)
 253         return value

 255     def _escape(self, value, include_vars=False):
 256         '''Return value escaped for use in PowerShell command.'''
 257         # http://www.techotopia.com/index.php/Windows_PowerShell_1.0_String_Quoting_and_Escape_Sequences
 258         # http://stackoverflow.com/questions/764360/a-list-of-string-replacements-in-python
 259         subs = [('\n', '`n'), ('\r', '`r'), ('\t', '`t'), ('\a', '`a'),
 260                 ('\b', '`b'), ('\f', '`f'), ('\v', '`v'), ('"', '`"'),
 261                 ('\'', '`\''), ('`', '``'), ('\x00', '`0')]
 262         if include_vars:
 263             subs.append(('$', '`$'))
 264         pattern = '|'.join('(%s)' % re.escape(p) for p, s in subs)
 265         substs = [s for p, s in subs]
 266 
 267         def replace(m):
 268             return substs[m.lastindex - 1]
 269 
 270         return re.sub(pattern, replace, value)

 272     def _encode_script(self, script, as_list=False, strict_mode=True, preserve_rc=True):
 273         '''Convert a PowerShell script to a single base64-encoded command.'''
 274         script = to_text(script)
 275 
 276         if script == u'-':
 277             cmd_parts = _common_args + ['-Command', '-']
 278 
 279         else:
 280             if strict_mode:
 281                 script = u'Set-StrictMode -Version Latest\r\n%s' % script
 282             # try to propagate exit code if present- won't work with begin/process/end-style scripts (ala put_file)
 283             # NB: the exit code returned may be incorrect in the case of a successful command followed by an invalid command
 284             if preserve_rc:
 285                 script = u'%s\r\nIf (-not $?) { If (Get-Variable LASTEXITCODE -ErrorAction SilentlyContinue) { exit $LASTEXITCODE } Else { exit 1 } }\r\n'\
 286                     % script
 287             script = '\n'.join([x.strip() for x in script.splitlines() if x.strip()])
 288             encoded_script = to_text(base64.b64encode(script.encode('utf-16-le')), 'utf-8')
 289             cmd_parts = _common_args + ['-EncodedCommand', encoded_script]
 290 
 291         if as_list:
 292             return cmd_parts
 293         return ' '.join(cmd_parts)
