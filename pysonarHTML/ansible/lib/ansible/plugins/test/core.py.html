  35 def failed(result):
  36     ''' Test if task result yields failed '''
  37     if not isinstance(result, MutableMapping):
  38         raise errors.AnsibleFilterError("The 'failed' test expects a dictionary")
  39     return result.get('failed', False)

  59 def changed(result):
  60     ''' Test if task result yields changed '''
  61     if not isinstance(result, MutableMapping):
  62         raise errors.AnsibleFilterError("The 'changed' test expects a dictionary")
  63     if 'changed' not in result:
  64         changed = False
  65         if (
  66             'results' in result and   # some modules return a 'results' key
  67             isinstance(result['results'], MutableSequence) and
  68             isinstance(result['results'][0], MutableMapping)
  69         ):
  70             for res in result['results']:
  71                 if res.get('changed', False):
  72                     changed = True
  73                     break
  74     else:
  75         changed = result.get('changed', False)
  76     return changed

  86 def started(result):
  87     ''' Test if async task has started '''
  88     if not isinstance(result, MutableMapping):
  89         raise errors.AnsibleFilterError("The 'started' test expects a dictionary")
  90     if 'started' in result:
  91         # For async tasks, return status
  92         # NOTE: The value of started is 0 or 1, not False or True :-/
  93         return result.get('started', 0) == 1
  94     else:
  95         # For non-async tasks, warn user, but return as if started
  96         display.warning("The 'started' test expects an async task, but a non-async task was tested")
  97         return True

 114 def regex(value='', pattern='', ignorecase=False, multiline=False, match_type='search'):
 115     ''' Expose `re` as a boolean filter using the `search` method by default.
 116         This is likely only useful for `search` and `match` which already
 117         have their own filters.
 118     '''
 119     # In addition to ensuring the correct type, to_text here will ensure
 120     # _fail_with_undefined_error happens if the value is Undefined
 121     value = to_text(value, errors='surrogate_or_strict')
 122     flags = 0
 123     if ignorecase:
 124         flags |= re.I
 125     if multiline:
 126         flags |= re.M
 127     _re = re.compile(pattern, flags=flags)
 128     return bool(getattr(_re, match_type, 'search')(value))

 131 def vault_encrypted(value):
 132     """Evaulate whether a variable is a single vault encrypted value
 133 
 134     .. versionadded:: 2.10
 135     """
 136     return getattr(value, '__ENCRYPTED__', False) and value.is_encrypted()

 149 def version_compare(value, version, operator='eq', strict=False):
 150     ''' Perform a version comparison on a value '''
 151     op_map = {
 152         '==': 'eq', '=': 'eq', 'eq': 'eq',
 153         '<': 'lt', 'lt': 'lt',
 154         '<=': 'le', 'le': 'le',
 155         '>': 'gt', 'gt': 'gt',
 156         '>=': 'ge', 'ge': 'ge',
 157         '!=': 'ne', '<>': 'ne', 'ne': 'ne'
 158     }
 159 
 160     if strict:
 161         Version = StrictVersion
 162     else:
 163         Version = LooseVersion
 164 
 165     if operator in op_map:
 166         operator = op_map[operator]
 167     else:
 168         raise errors.AnsibleFilterError('Invalid operator type')
 169 
 170     try:
 171         method = getattr(py_operator, operator)
 172         return method(Version(str(value)), Version(str(version)))
 173     except Exception as e:
 174         raise errors.AnsibleFilterError('Version comparison: %s' % e)

 177 def truthy(value, convert_bool=False):
 178     """Evaluate as value for truthiness using python ``bool``
 179 
 180     Optionally, attempt to do a conversion to bool from boolean like values
 181     such as ``"false"``, ``"true"``, ``"yes"``, ``"no"``, ``"on"``, ``"off"``, etc.
 182 
 183     .. versionadded:: 2.10
 184     """
 185     if convert_bool:
 186         try:
 187             value = boolean(value)
 188         except TypeError:
 189             pass
 190 
 191     return bool(value)
 192 
 193 
 194 def falsy(value, convert_bool=False):
 195     """Evaluate as value for falsiness using python ``bool``
 196 
 197     Optionally, attempt to do a conversion to bool from boolean like values
 198     such as ``"false"``, ``"true"``, ``"yes"``, ``"no"``, ``"on"``, ``"off"``, etc.
 199 
 200     .. versionadded:: 2.10
 201     """
 202     return not truthy(value, convert_bool=convert_bool)
