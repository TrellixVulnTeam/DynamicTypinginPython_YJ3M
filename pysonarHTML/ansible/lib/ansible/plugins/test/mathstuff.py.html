  24 def issubset(a, b):
  25     return set(a) <= set(b)
  26 
  27 
  28 def issuperset(a, b):
  29     return set(a) >= set(b)

  32 def isnotanumber(x):
  33     try:
  34         return math.isnan(x)
  35     except TypeError:
  36         return False

  40     '''Opposite of the ``in`` test, allowing use as a test in filters like ``selectattr``
