   9 from ansible.errors import AnsibleFilterError
  10 from ansible.module_utils.six.moves.urllib.parse import urlsplit

  14 def split_url(value, query='', alias='urlsplit'):
  15 
  16     results = helpers.object_to_dict(urlsplit(value), exclude=['count', 'index', 'geturl', 'encode'])
  17 
  18     # If a query is supplied, make sure it's valid then return the results.
  19     # If no option is supplied, return the entire dictionary.
  20     if query:
  21         if query not in results:
  22             raise AnsibleFilterError(alias + ': unknown URL component: %s' % query)
  23         return results[query]
  24     else:
  25         return results

  29 class FilterModule(object):
  30     ''' URI filter '''
  31 
  32     def filters(self):
  33         return {
  34             'urlsplit': split_url
  35         }
