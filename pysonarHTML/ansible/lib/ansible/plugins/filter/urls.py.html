  20 def unicode_urldecode(string):
  21     if PY3:
  22         return unquote_plus(string)
  23     return to_text(unquote_plus(to_bytes(string)))

  30 # NOTE: We implement urlencode when Jinja2 is older than v2.7
  31 def unicode_urlencode(string, for_qs=False):
  32     safe = b'' if for_qs else b'/'
  33     if for_qs:
  34         quote_func = quote_plus
  35     else:
  36         quote_func = quote
  37     if PY3:
  38         return quote_func(string, safe)
  39     return to_text(quote_func(to_bytes(string), safe))

  42 def do_urlencode(value):
  43     itemiter = None
  44     if isinstance(value, dict):
  45         itemiter = iteritems(value)
  46     elif not isinstance(value, string_types):
  47         try:
  48             itemiter = iter(value)
  49         except TypeError:
  50             pass
  51     if itemiter is None:
  52         return unicode_urlencode(value)
  53     return u'&'.join(unicode_urlencode(k) + '=' +
  54                      unicode_urlencode(v, for_qs=True)
  55                      for k, v in itemiter)
