  86 def to_bool(a):
  87     ''' return a bool for the arg '''
  88     if a is None or isinstance(a, bool):
  89         return a
  90     if isinstance(a, string_types):
  91         a = a.lower()
  92     if a in ('yes', 'on', '1', 'true', 1):
  93         return True
  94     return False

 121 def regex_replace(value='', pattern='', replacement='', ignorecase=False, multiline=False):
 122     ''' Perform a `re.sub` returning a string '''
 123 
 124     value = to_text(value, errors='surrogate_or_strict', nonstring='simplerepr')
 125 
 126     flags = 0
 127     if ignorecase:
 128         flags |= re.I
 129     if multiline:
 130         flags |= re.M
 131     _re = re.compile(pattern, flags=flags)
 132     return _re.sub(replacement, value)

 148 def regex_search(value, regex, *args, **kwargs):
 149     ''' Perform re.search and return the list of matches or a backref '''
 150 
 151     value = to_text(value, errors='surrogate_or_strict', nonstring='simplerepr')
 152 
 153     groups = list()
 154     for arg in args:
 155         if arg.startswith('\\g'):
 156             match = re.match(r'\\g<(\S+)>', arg).group(1)
 157             groups.append(match)
 158         elif arg.startswith('\\'):
 159             match = int(re.match(r'\\(\d+)', arg).group(1))
 160             groups.append(match)
 161         else:
 162             raise AnsibleFilterError('Unknown argument')
 163 
 164     flags = 0
 165     if kwargs.get('ignorecase'):
 166         flags |= re.I
 167     if kwargs.get('multiline'):
 168         flags |= re.M
 169 
 170     match = re.search(regex, value, flags)
 171     if match:
 172         if not groups:
 173             return match.group()
 174         else:
 175             items = list()
 176             for item in groups:
 177                 items.append(match.group(item))
 178             return items

 181 def ternary(value, true_val, false_val, none_val=None):
 182     '''  value ? true_val : false_val '''
 183     if value is None and none_val is not None:
 184         return none_val
 185     elif bool(value):
 186         return true_val
 187     else:
 188         return false_val

 191 def regex_escape(string, re_type='python'):
 192     string = to_text(string, errors='surrogate_or_strict', nonstring='simplerepr')
 193     '''Escape all regular expressions special characters from STRING.'''
 194     if re_type == 'python':
 195         return re.escape(string)
 196     elif re_type == 'posix_basic':
 197         # list of BRE special chars:
 198         # https://en.wikibooks.org/wiki/Regular_Expressions/POSIX_Basic_Regular_Expressions
 199         return regex_replace(string, r'([].[^$*\\])', r'\\\1')
 200     # TODO: implement posix_extended
 201     # It's similar to, but different from python regex, which is similar to,
 202     # but different from PCRE.  It's possible that re.escape would work here.
 203     # https://remram44.github.io/regex-cheatsheet/regex.html#programs
 204     elif re_type == 'posix_extended':
 205         raise AnsibleFilterError('Regex type (%s) not yet implemented' % re_type)
 206     else:
 207         raise AnsibleFilterError('Invalid regex type (%s)' % re_type)

 222 @environmentfilter
 223 def rand(environment, end, start=None, step=None, seed=None):
 224     if seed is None:
 225         r = SystemRandom()
 226     else:
 227         r = Random(seed)
 228     if isinstance(end, integer_types):
 229         if not start:
 230             start = 0
 231         if not step:
 232             step = 1
 233         return r.randrange(start, end, step)
 234     elif hasattr(end, '__iter__'):
 235         if start or step:
 236             raise AnsibleFilterError('start and step can only be used with integer values')
 237         return r.choice(end)
 238     else:
 239         raise AnsibleFilterError('random can only be used on sequences and integers')

 243     try:
 244         mylist = list(mylist)

 255 def get_hash(data, hashtype='sha1'):
 256     try:
 257         h = hashlib.new(hashtype)
 258     except Exception as e:
 259         # hash is not supported?
 260         raise AnsibleFilterError(e)
 261 
 262     h.update(to_bytes(data, errors='surrogate_or_strict'))
 263     return h.hexdigest()

 266 def get_encrypted_password(password, hashtype='sha512', salt=None, salt_size=None, rounds=None):
 267     passlib_mapping = {
 268         'md5': 'md5_crypt',
 269         'blowfish': 'bcrypt',
 270         'sha256': 'sha256_crypt',
 271         'sha512': 'sha512_crypt',
 272     }
 273 
 274     hashtype = passlib_mapping.get(hashtype, hashtype)
 275     try:
 276         return passlib_or_crypt(password, hashtype, salt=salt, salt_size=salt_size, rounds=rounds)
 277     except AnsibleError as e:
 278         reraise(AnsibleFilterError, AnsibleFilterError(to_native(e), orig_exc=e), sys.exc_info()[2])

 281 def to_uuid(string, namespace=UUID_NAMESPACE_ANSIBLE):
 282     uuid_namespace = namespace
 283     if not isinstance(uuid_namespace, uuid.UUID):
 284         try:
 285             uuid_namespace = uuid.UUID(namespace)
 286         except (AttributeError, ValueError) as e:
 287             raise AnsibleFilterError("Invalid value '%s' for 'namespace': %s" % (to_native(namespace), to_native(e)))
 288     # uuid.uuid5() requires bytes on Python 2 and bytes or text or Python 3
 289     return to_text(uuid.uuid5(uuid_namespace, to_native(string, errors='surrogate_or_strict')))

 310 def combine(*terms, **kwargs):
 311     recursive = kwargs.pop('recursive', False)
 312     list_merge = kwargs.pop('list_merge', 'replace')
 313     if kwargs:
 314         raise AnsibleFilterError("'recursive' and 'list_merge' are the only valid keyword arguments")
 315 
 316     # allow the user to do `[dict1, dict2, ...] | combine`
 317     dictionaries = flatten(terms, levels=1)
 318 
 319     # recursively check that every elements are defined (for jinja2)
 320     recursive_check_defined(dictionaries)
 321 
 322     if not dictionaries:
 323         return {}
 324 
 325     if len(dictionaries) == 1:
 326         return dictionaries[0]
 327 
 328     # merge all the dicts so that the dict at the end of the array have precedence
 329     # over the dict at the beginning.
 330     # we merge the dicts from the highest to the lowest priority because there is
 331     # a huge probability that the lowest priority dict will be the biggest in size
 332     # (as the low prio dict will hold the "default" values and the others will be "patches")
 333     # and merge_hash create a copy of it's first argument.
 334     # so high/right -> low/left is more efficient than low/left -> high/right
 335     high_to_low_prio_dict_iterator = reversed(dictionaries)
 336     result = next(high_to_low_prio_dict_iterator)
 337     for dictionary in high_to_low_prio_dict_iterator:
 338         result = merge_hash(dictionary, result, recursive, list_merge)
 339 
 340     return result

 343 def comment(text, style='plain', **kw):
 344     # Predefined comment types
 345     comment_styles = {
 346         'plain': {
 347             'decoration': '# '
 348         },
 349         'erlang': {
 350             'decoration': '% '
 351         },
 352         'c': {
 353             'decoration': '// '
 354         },
 355         'cblock': {
 356             'beginning': '/*',
 357             'decoration': ' * ',
 358             'end': ' */'
 359         },
 360         'xml': {
 361             'beginning': '<!--',
 362             'decoration': ' - ',
 363             'end': '-->'
 364         }
 365     }
 366 
 367     # Pointer to the right comment type
 368     style_params = comment_styles[style]
 369 
 370     if 'decoration' in kw:
 371         prepostfix = kw['decoration']
 372     else:
 373         prepostfix = style_params['decoration']
 374 
 375     # Default params
 376     p = {
 377         'newline': '\n',
 378         'beginning': '',
 379         'prefix': (prepostfix).rstrip(),
 380         'prefix_count': 1,
 381         'decoration': '',
 382         'postfix': (prepostfix).rstrip(),
 383         'postfix_count': 1,
 384         'end': ''
 385     }
 386 
 387     # Update default params
 388     p.update(style_params)
 389     p.update(kw)
 390 
 391     # Compose substrings for the final string
 392     str_beginning = ''
 393     if p['beginning']:
 394         str_beginning = "%s%s" % (p['beginning'], p['newline'])
 395     str_prefix = ''
 396     if p['prefix']:
 397         if p['prefix'] != p['newline']:
 398             str_prefix = str(
 399                 "%s%s" % (p['prefix'], p['newline'])) * int(p['prefix_count'])
 400         else:
 401             str_prefix = str(
 402                 "%s" % (p['newline'])) * int(p['prefix_count'])
 403     str_text = ("%s%s" % (
 404         p['decoration'],
 405         # Prepend each line of the text with the decorator
 406         text.replace(
 407             p['newline'], "%s%s" % (p['newline'], p['decoration'])))).replace(
 408                 # Remove trailing spaces when only decorator is on the line
 409                 "%s%s" % (p['decoration'], p['newline']),
 410                 "%s%s" % (p['decoration'].rstrip(), p['newline']))
 411     str_postfix = p['newline'].join(
 412         [''] + [p['postfix'] for x in range(p['postfix_count'])])
 413     str_end = ''
 414     if p['end']:
 415         str_end = "%s%s" % (p['newline'], p['end'])
 416 
 417     # Return the final string
 418     return "%s%s%s%s%s" % (
 419         str_beginning,
 420         str_prefix,
 421         str_text,
 422         str_postfix,
 423         str_end)

 426 @environmentfilter
 427 def extract(environment, item, container, morekeys=None):
 428     if morekeys is None:
 429         keys = [item]
 430     elif isinstance(morekeys, list):
 431         keys = [item] + morekeys
 432     else:
 433         keys = [item, morekeys]
 434 
 435     value = container
 436     for key in keys:
 437         value = environment.getitem(value, key)
 438 
 439     return value

 442 @environmentfilter
 443 def do_groupby(environment, value, attribute):
 444     """Overridden groupby filter for jinja2, to address an issue with
 445     jinja2>=2.9.0,<2.9.5 where a namedtuple was returned which
 446     has repr that prevents ansible.template.safe_eval.safe_eval from being
 447     able to parse and eval the data.
 448 
 449     jinja2<2.9.0,>=2.9.5 is not affected, as <2.9.0 uses a tuple, and
 450     >=2.9.5 uses a standard tuple repr on the namedtuple.
 451 
 452     The adaptation here, is to run the jinja2 `do_groupby` function, and
 453     cast all of the namedtuples to a regular tuple.
 454 
 455     See https://github.com/ansible/ansible/issues/20098
 456 
 457     We may be able to remove this in the future.
 458     """
 459     return [tuple(t) for t in _do_groupby(environment, value, attribute)]

 470 def flatten(mylist, levels=None, skip_nulls=True):
 471 
 472     ret = []
 473     for element in mylist:
 474         if skip_nulls and element in (None, 'None', 'null'):
 475             # ignore null items
 476             continue
 477         elif is_sequence(element):
 478             if levels is None:
 479                 ret.extend(flatten(element, skip_nulls=skip_nulls))
 480             elif levels >= 1:
 481                 # decrement as we go down the stack
 482                 ret.extend(flatten(element, levels=(int(levels) - 1), skip_nulls=skip_nulls))
 483             else:
 484                 ret.append(element)
 485         else:
 486             ret.append(element)
 487 
 488     return ret

 491 def subelements(obj, subelements, skip_missing=False):
 492     '''Accepts a dict or list of dicts, and a dotted accessor and produces a product
 493     of the element and the results of the dotted accessor
 494 
 495     >>> obj = [{"name": "alice", "groups": ["wheel"], "authorized": ["/tmp/alice/onekey.pub"]}]
 496     >>> subelements(obj, 'groups')
 497     [({'name': 'alice', 'groups': ['wheel'], 'authorized': ['/tmp/alice/onekey.pub']}, 'wheel')]
 498 
 499     '''
 500     if isinstance(obj, dict):
 501         element_list = list(obj.values())
 502     elif isinstance(obj, list):
 503         element_list = obj[:]
 504     else:
 505         raise AnsibleFilterError('obj must be a list of dicts or a nested dict')
 506 
 507     if isinstance(subelements, list):
 508         subelement_list = subelements[:]
 509     elif isinstance(subelements, string_types):
 510         subelement_list = subelements.split('.')
 511     else:
 512         raise AnsibleFilterTypeError('subelements must be a list or a string')
 513 
 514     results = []
 515 
 516     for element in element_list:
 517         values = element
 518         for subelement in subelement_list:
 519             try:
 520                 values = values[subelement]
 521             except KeyError:
 522                 if skip_missing:
 523                     values = []
 524                     break
 525                 raise AnsibleFilterError("could not find %r key in iterated item %r" % (subelement, values))
 526             except TypeError:
 527                 raise AnsibleFilterTypeError("the key %s should point to a dictionary, got '%s'" % (subelement, values))
 528         if not isinstance(values, list):
 529             raise AnsibleFilterTypeError("the key %r should point to a list, got %r" % (subelement, values))
 530 
 531         for value in values:
 532             results.append((element, value))
 533 
 534     return results

 537 def dict_to_list_of_dict_key_value_elements(mydict, key_name='key', value_name='value'):
 538     ''' takes a dictionary and transforms it into a list of dictionaries,
 539         with each having a 'key' and 'value' keys that correspond to the keys and values of the original '''
 540 
 541     if not isinstance(mydict, Mapping):
 542         raise AnsibleFilterTypeError("dict2items requires a dictionary, got %s instead." % type(mydict))
 543 
 544     ret = []
 545     for key in mydict:
 546         ret.append({key_name: key, value_name: mydict[key]})
 547     return ret

 550 def list_of_dict_key_value_elements_to_dict(mylist, key_name='key', value_name='value'):
 551     ''' takes a list of dicts with each having a 'key' and 'value' keys, and transforms the list into a dictionary,

 553 
 554     if not is_sequence(mylist):

 557     return dict((item[key_name], item[value_name]) for item in mylist)

 560 def path_join(paths):
