  48 @environmentfilter
  49 def unique(environment, a, case_sensitive=False, attribute=None):
  50 
  51     def _do_fail(e):
  52         if case_sensitive or attribute:
  53             raise AnsibleFilterError("Jinja2's unique filter failed and we cannot fall back to Ansible's version "
  54                                      "as it does not support the parameters supplied", orig_exc=e)
  55 
  56     error = e = None
  57     try:
  58         if HAS_UNIQUE:
  59             c = do_unique(environment, a, case_sensitive=case_sensitive, attribute=attribute)
  60             if isinstance(a, Hashable):
  61                 c = set(c)
  62             else:
  63                 c = list(c)
  64     except TypeError as e:
  65         error = e
  66         _do_fail(e)
  67     except Exception as e:
  68         error = e
  69         _do_fail(e)
  70         display.warning('Falling back to Ansible unique filter as Jinja2 one failed: %s' % to_text(e))
  71 
  72     if not HAS_UNIQUE or error:
  73 
  74         # handle Jinja2 specific attributes when using Ansible's version
  75         if case_sensitive or attribute:
  76             raise AnsibleFilterError("Ansible's unique filter does not support case_sensitive nor attribute parameters, "
  77                                      "you need a newer version of Jinja2 that provides their version of the filter.")
  78 
  79         if isinstance(a, Hashable):
  80             c = set(a)
  81         else:
  82             c = []
  83             for x in a:
  84                 if x not in c:
  85                     c.append(x)
  86     return c

  89 @environmentfilter
  90 def intersect(environment, a, b):
  91     if isinstance(a, Hashable) and isinstance(b, Hashable):
  92         c = set(a) & set(b)
  93     else:
  94         c = unique(environment, [x for x in a if x in b])
  95     return c

  98 @environmentfilter
  99 def difference(environment, a, b):
 100     if isinstance(a, Hashable) and isinstance(b, Hashable):
 101         c = set(a) - set(b)
 102     else:
 103         c = unique(environment, [x for x in a if x not in b])
 104     return c

 107 @environmentfilter
 108 def symmetric_difference(environment, a, b):
 109     if isinstance(a, Hashable) and isinstance(b, Hashable):
 110         c = set(a) ^ set(b)
 111     else:
 112         isect = intersect(environment, a, b)
 113         c = [x for x in union(environment, a, b) if x not in isect]
 114     return c

 117 @environmentfilter
 118 def union(environment, a, b):
 119     if isinstance(a, Hashable) and isinstance(b, Hashable):
 120         c = set(a) | set(b)
 121     else:
 122         c = unique(environment, a + b)
 123     return c

 136 def logarithm(x, base=math.e):
 137     try:
 138         if base == 10:
 139             return math.log10(x)
 140         else:
 141             return math.log(x, base)
 142     except TypeError as e:
 143         raise AnsibleFilterTypeError('log() can only be used on numbers: %s' % to_native(e))
 144

 146 def power(x, y):
 147     try:
 148         return math.pow(x, y)
 149     except TypeError as e:
 150         raise AnsibleFilterTypeError('pow() can only be used on numbers: %s' % to_native(e))
 151 
 152 
 153 def inversepower(x, base=2):
 154     try:
 155         if base == 2:
 156             return math.sqrt(x)
 157         else:
 158             return math.pow(x, 1.0 / float(base))
 159     except (ValueError, TypeError) as e:
 160         raise AnsibleFilterTypeError('root() can only be used on numbers: %s' % to_native(e))

 163 def human_readable(size, isbits=False, unit=None):
 164     ''' Return a human readable string '''
 165     try:
 166         return formatters.bytes_to_human(size, isbits, unit)
 167     except TypeError as e:
 168         raise AnsibleFilterTypeError("human_readable() failed on bad input: %s" % to_native(e))
 169     except Exception:
 170         raise AnsibleFilterError("human_readable() can't interpret following string: %s" % size)

 183 def rekey_on_member(data, key, duplicates='error'):
 184     """
 185     Rekey a dict of dicts on another member
 186 
 187     May also create a dict from a list of dicts.
 188 
 189     duplicates can be one of ``error`` or ``overwrite`` to specify whether to error out if the key
 190     value would be duplicated or to overwrite previous entries if that's the case.
 191     """
 192     if duplicates not in ('error', 'overwrite'):
 193         raise AnsibleFilterError("duplicates parameter to rekey_on_member has unknown value: {0}".format(duplicates))
 194 
 195     new_obj = {}
 196 
 197     if isinstance(data, Mapping):
 198         iterate_over = data.values()
 199     elif isinstance(data, Iterable) and not isinstance(data, (text_type, binary_type)):
 200         iterate_over = data
 201     else:
 202         raise AnsibleFilterTypeError("Type is not a valid list, set, or dict")
 203 
 204     for item in iterate_over:
 205         if not isinstance(item, Mapping):
 206             raise AnsibleFilterTypeError("List item is not a valid dict")
 207 
 208         try:
 209             key_elem = item[key]
 210         except KeyError:
 211             raise AnsibleFilterError("Key {0} was not found".format(key))
 212         except TypeError as e:
 213             raise AnsibleFilterTypeError(to_native(e))
 214         except Exception as e:
 215             raise AnsibleFilterError(to_native(e))
 216 
 217         # Note: if new_obj[key_elem] exists it will always be a non-empty dict (it will at
 218         # minimum contain {key: key_elem}
 219         if new_obj.get(key_elem, None):
 220             if duplicates == 'error':
 221                 raise AnsibleFilterError("Key {0} is not unique, cannot correctly turn into dict".format(key_elem))
 222             elif duplicates == 'overwrite':
 223                 new_obj[key_elem] = item
 224         else:
 225             new_obj[key_elem] = item
 226 
 227     return new_obj

 230 class FilterModule(object):
 231     ''' Ansible math jinja2 filters '''
 232 
 233     def filters(self):
 234         filters = {
 235             # general math
 236             'min': min,
 237             'max': max,
 238 
 239             # exponents and logarithms
 240             'log': logarithm,
 241             'pow': power,
 242             'root': inversepower,
 243 
 244             # set theory
 245             'unique': unique,
 246             'intersect': intersect,
 247             'difference': difference,
 248             'symmetric_difference': symmetric_difference,
 249             'union': union,
 250 
 251             # combinatorial
 252             'product': itertools.product,
 253             'permutations': itertools.permutations,
 254             'combinations': itertools.combinations,
 255 
 256             # computer theory
 257             'human_readable': human_readable,
 258             'human_to_bytes': human_to_bytes,
 259             'rekey_on_member': rekey_on_member,
 260 
 261             # zip
 262             'zip': zip,
 263             'zip_longest': zip_longest,
 264 
 265         }
 266 
 267         return filters
