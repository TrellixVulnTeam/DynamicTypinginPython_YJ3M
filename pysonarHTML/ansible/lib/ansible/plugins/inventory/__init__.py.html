  48 def detect_range(line=None):
  49     '''
  50     A helper function that checks a given host line to see if it contains
  51     a range pattern described in the docstring above.
  52 
  53     Returns True if the given line contains a pattern, else False.
  54     '''
  55     return '[' in line

  58 def expand_hostname_range(line=None):
  59     '''
  60     A helper function that expands a given line that contains a pattern
  61     specified in top docstring, and returns a list that consists of the
  62     expanded version.
  63 
  64     The '[' and ']' characters are used to maintain the pseudo-code
  65     appearance. They are replaced in this function with '|' to ease
  66     string splitting.
  67 
  68     References: https://docs.ansible.com/ansible/latest/user_guide/intro_inventory.html#hosts-and-groups
  69     '''
  70     all_hosts = []
  71     if line:
  72         # A hostname such as db[1:6]-node is considered to consists
  73         # three parts:
  74         # head: 'db'
  75         # nrange: [1:6]; range() is a built-in. Can't use the name
  76         # tail: '-node'
  77 
  78         # Add support for multiple ranges in a host so:
  79         # db[01:10:3]node-[01:10]
  80         # - to do this we split off at the first [...] set, getting the list
  81         #   of hosts and then repeat until none left.
  82         # - also add an optional third parameter which contains the step. (Default: 1)
  83         #   so range can be [01:10:2] -> 01 03 05 07 09
  84 
  85         (head, nrange, tail) = line.replace('[', '|', 1).replace(']', '|', 1).split('|')
  86         bounds = nrange.split(":")
  87         if len(bounds) != 2 and len(bounds) != 3:
  88             raise AnsibleError("host range must be begin:end or begin:end:step")
  89         beg = bounds[0]
  90         end = bounds[1]
  91         if len(bounds) == 2:
  92             step = 1
  93         else:
  94             step = bounds[2]
  95         if not beg:
  96             beg = "0"
  97         if not end:
  98             raise AnsibleError("host range must specify end value")
  99         if beg[0] == '0' and len(beg) > 1:
 100             rlen = len(beg)  # range length formatting hint
 101             if rlen != len(end):
 102                 raise AnsibleError("host range must specify equal-length begin and end formats")
 103 
 104             def fill(x):
 105                 return str(x).zfill(rlen)  # range sequence
 106 
 107         else:
 108             fill = str
 109 
 110         try:
 111             i_beg = string.ascii_letters.index(beg)
 112             i_end = string.ascii_letters.index(end)
 113             if i_beg > i_end:
 114                 raise AnsibleError("host range must have begin <= end")
 115             seq = list(string.ascii_letters[i_beg:i_end + 1:int(step)])
 116         except ValueError:  # not an alpha range
 117             seq = range(int(beg), int(end) + 1, int(step))
 118 
 119         for rseq in seq:
 120             hname = ''.join((head, fill(rseq), tail))
 121 
 122             if detect_range(hname):
 123                 all_hosts.extend(expand_hostname_range(hname))
 124             else:
 125                 all_hosts.append(hname)
 126 
 127         return all_hosts

 130 def get_cache_plugin(plugin_name, **kwargs):
 131     try:
 132         cache = CacheObject(plugin_name, **kwargs)
 133     except AnsibleError as e:
 134         if 'fact_caching_connection' in to_native(e):
 135             raise AnsibleError("error, '%s' inventory cache plugin requires the one of the following to be set "
 136                                "to a writeable directory path:\nansible.cfg:\n[default]: fact_caching_connection,\n"
 137                                "[inventory]: cache_connection;\nEnvironment:\nANSIBLE_INVENTORY_CACHE_CONNECTION,\n"
 138                                "ANSIBLE_CACHE_PLUGIN_CONNECTION." % plugin_name)
 139         else:
 140             raise e
 141 
 142     if plugin_name != 'memory' and kwargs and not getattr(cache._plugin, '_options', None):
 143         raise AnsibleError('Unable to use cache plugin {0} for inventory. Cache options were provided but may not reconcile '
 144                            'correctly unless set via set_options. Refer to the porting guide if the plugin derives user settings '
 145                            'from ansible.constants.'.format(plugin_name))
 146     return cache

 149 class BaseInventoryPlugin(AnsiblePlugin):
 150     """ Parses an Inventory Source"""
 151 
 152     TYPE = 'generator'
 153     _sanitize_group_name = staticmethod(to_safe_group_name)
 154 
 155     def __init__(self):
 156 
 157         super(BaseInventoryPlugin, self).__init__()
 158 
 159         self._options = {}
 160         self.inventory = None
 161         self.display = display
 162 
 163     def parse(self, inventory, loader, path, cache=True):
 164         ''' Populates inventory from the given data. Raises an error on any parse failure
 165             :arg inventory: a copy of the previously accumulated inventory data,
 166                  to be updated with any new data this plugin provides.
 167                  The inventory can be empty if no other source/plugin ran successfully.
 168             :arg loader: a reference to the DataLoader, which can read in YAML and JSON files,
 169                  it also has Vault support to automatically decrypt files.
 170             :arg path: the string that represents the 'inventory source',
 171                  normally a path to a configuration file for this inventory,
 172                  but it can also be a raw string for this plugin to consume
 173             :arg cache: a boolean that indicates if the plugin should use the cache or not
 174                  you can ignore if this plugin does not implement caching.
 175         '''
 176 
 177         self.loader = loader
 178         self.inventory = inventory
 179         self.templar = Templar(loader=loader)
 180 
 181     def verify_file(self, path):
 182         ''' Verify if file is usable by this plugin, base does minimal accessibility check
 183             :arg path: a string that was passed as an inventory source,
 184                  it normally is a path to a config file, but this is not a requirement,
 185                  it can also be parsed itself as the inventory data to process.
 186                  So only call this base class if you expect it to be a file.
 187         '''
 188 
 189         valid = False
 190         b_path = to_bytes(path, errors='surrogate_or_strict')
 191         if (os.path.exists(b_path) and os.access(b_path, os.R_OK)):
 192             valid = True
 193         else:
 194             self.display.vvv('Skipping due to inventory source not existing or not being readable by the current user')
 195         return valid
 196 
 197     def _populate_host_vars(self, hosts, variables, group=None, port=None):
 198         if not isinstance(variables, Mapping):
 199             raise AnsibleParserError("Invalid data from file, expected dictionary and got:\n\n%s" % to_native(variables))
 200 
 201         for host in hosts:
 202             self.inventory.add_host(host, group=group, port=port)
 203             for k in variables:
 204                 self.inventory.set_variable(host, k, variables[k])
 205 
 206     def _read_config_data(self, path):
 207         ''' validate config and set options as appropriate
 208             :arg path: path to common yaml format config file for this plugin
 209         '''
 210 
 211         config = {}
 212         try:
 213             # avoid loader cache so meta: refresh_inventory can pick up config changes
 214             # if we read more than once, fs cache should be good enough
 215             config = self.loader.load_from_file(path, cache=False)
 216         except Exception as e:
 217             raise AnsibleParserError(to_native(e))
 218 
 219         # a plugin can be loaded via many different names with redirection- if so, we want to accept any of those names
 220         valid_names = getattr(self, '_redirected_names') or [self.NAME]
 221 
 222         if not config:
 223             # no data
 224             raise AnsibleParserError("%s is empty" % (to_native(path)))
 225         elif config.get('plugin') not in valid_names:
 226             # this is not my config file
 227             raise AnsibleParserError("Incorrect plugin name in file: %s" % config.get('plugin', 'none found'))
 228         elif not isinstance(config, Mapping):
 229             # configs are dictionaries
 230             raise AnsibleParserError('inventory source has invalid structure, it should be a dictionary, got: %s' % type(config))
 231 
 232         self.set_options(direct=config)
 233         if 'cache' in self._options and self.get_option('cache'):
 234             cache_option_keys = [('_uri', 'cache_connection'), ('_timeout', 'cache_timeout'), ('_prefix', 'cache_prefix')]
 235             cache_options = dict((opt[0], self.get_option(opt[1])) for opt in cache_option_keys if self.get_option(opt[1]))
 236             self._cache = get_cache_plugin(self.get_option('cache_plugin'), **cache_options)
 237 
 238         return config
 239 
 240     def _consume_options(self, data):
 241         ''' update existing options from alternate configuration sources not normally used by Ansible.
 242             Many API libraries already have existing configuration sources, this allows plugin author to leverage them.
 243             :arg data: key/value pairs that correspond to configuration options for this plugin
 244         '''
 245 
 246         for k in self._options:
 247             if k in data:
 248                 self._options[k] = data.pop(k)
 249 
 250     def _expand_hostpattern(self, hostpattern):
 251         '''
 252         Takes a single host pattern and returns a list of hostnames and an
 253         optional port number that applies to all of them.
 254         '''
 255         # Can the given hostpattern be parsed as a host with an optional port
 256         # specification?
 257 
 258         try:
 259             (pattern, port) = parse_address(hostpattern, allow_ranges=True)
 260         except Exception:
 261             # not a recognizable host pattern
 262             pattern = hostpattern
 263             port = None
 264 
 265         # Once we have separated the pattern, we expand it into list of one or
 266         # more hostnames, depending on whether it contains any [x:y] ranges.
 267 
 268         if detect_range(pattern):
 269             hostnames = expand_hostname_range(pattern)
 270         else:
 271             hostnames = [pattern]
 272 
 273         return (hostnames, port)

 286 class DeprecatedCache(object):
 287     def __init__(self, real_cacheable):
 288         self.real_cacheable = real_cacheable
 289 
 290     def get(self, key):
 291         display.deprecated('InventoryModule should utilize self._cache as a dict instead of self.cache. '
 292                            'When expecting a KeyError, use self._cache[key] instead of using self.cache.get(key). '
 293                            'self._cache is a dictionary and will return a default value instead of raising a KeyError '
 294                            'when the key does not exist', version='2.12', collection_name='ansible.builtin')
 295         return self.real_cacheable._cache[key]
 296 
 297     def set(self, key, value):
 298         display.deprecated('InventoryModule should utilize self._cache as a dict instead of self.cache. '
 299                            'To set the self._cache dictionary, use self._cache[key] = value instead of self.cache.set(key, value). '
 300                            'To force update the underlying cache plugin with the contents of self._cache before parse() is complete, '
 301                            'call self.set_cache_plugin and it will use the self._cache dictionary to update the cache plugin',
 302                            version='2.12', collection_name='ansible.builtin')
 303         self.real_cacheable._cache[key] = value
 304         self.real_cacheable.set_cache_plugin()
 305 
 306     def __getattr__(self, name):
 307         display.deprecated('InventoryModule should utilize self._cache instead of self.cache',
 308                            version='2.12', collection_name='ansible.builtin')
 309         return self.real_cacheable._cache.__getattribute__(name)

 312 class Cacheable(object):
 313 
 314     _cache = CacheObject()
 315 
 316     @property
 317     def cache(self):
 318         return DeprecatedCache(self)
 319 
 320     def load_cache_plugin(self):
 321         plugin_name = self.get_option('cache_plugin')
 322         cache_option_keys = [('_uri', 'cache_connection'), ('_timeout', 'cache_timeout'), ('_prefix', 'cache_prefix')]
 323         cache_options = dict((opt[0], self.get_option(opt[1])) for opt in cache_option_keys if self.get_option(opt[1]))
 324         self._cache = get_cache_plugin(plugin_name, **cache_options)
 325 
 326     def get_cache_key(self, path):
 327         return "{0}_{1}".format(self.NAME, self._get_cache_prefix(path))
 328 
 329     def _get_cache_prefix(self, path):
 330         ''' create predictable unique prefix for plugin/inventory '''
 331 
 332         m = hashlib.sha1()
 333         m.update(to_bytes(self.NAME, errors='surrogate_or_strict'))
 334         d1 = m.hexdigest()
 335 
 336         n = hashlib.sha1()
 337         n.update(to_bytes(path, errors='surrogate_or_strict'))
 338         d2 = n.hexdigest()
 339 
 340         return 's_'.join([d1[:5], d2[:5]])
 341 
 342     def clear_cache(self):
 343         self._cache.flush()
 344 
 345     def update_cache_if_changed(self):
 346         self._cache.update_cache_if_changed()
 347 
 348     def set_cache_plugin(self):
 349         self._cache.set_cache()

 352 class Constructable(object):
 353 
 354     def _compose(self, template, variables):
 355         ''' helper method for plugins to compose variables for Ansible based on jinja2 expression and inventory vars'''
 356         t = self.templar
 357         t.available_variables = variables
 358         return t.template('%s%s%s' % (t.environment.variable_start_string, template, t.environment.variable_end_string), disable_lookups=True)
 359 
 360     def _set_composite_vars(self, compose, variables, host, strict=False):
 361         ''' loops over compose entries to create vars for hosts '''
 362         if compose and isinstance(compose, dict):
 363             for varname in compose:
 364                 try:
 365                     composite = self._compose(compose[varname], variables)
 366                 except Exception as e:
 367                     if strict:
 368                         raise AnsibleError("Could not set %s for host %s: %s" % (varname, host, to_native(e)))
 369                     continue
 370                 self.inventory.set_variable(host, varname, composite)
 371 
 372     def _add_host_to_composed_groups(self, groups, variables, host, strict=False):
 373         ''' helper to create complex groups for plugins based on jinja2 conditionals, hosts that meet the conditional are added to group'''
 374         # process each 'group entry'
 375         if groups and isinstance(groups, dict):
 376             variables = combine_vars(variables, self.inventory.get_host(host).get_vars())
 377             self.templar.available_variables = variables
 378             for group_name in groups:
 379                 conditional = "{%% if %s %%} True {%% else %%} False {%% endif %%}" % groups[group_name]
 380                 group_name = original_safe(group_name, force=True)
 381                 try:
 382                     result = boolean(self.templar.template(conditional))
 383                 except Exception as e:
 384                     if strict:
 385                         raise AnsibleParserError("Could not add host %s to group %s: %s" % (host, group_name, to_native(e)))
 386                     continue
 387 
 388                 if result:
 389                     # ensure group exists, use sanitized name
 390                     group_name = self.inventory.add_group(group_name)
 391                     # add host to group
 392                     self.inventory.add_child(group_name, host)
 393 
 394     def _add_host_to_keyed_groups(self, keys, variables, host, strict=False):
 395         ''' helper to create groups for plugins based on variable values and add the corresponding hosts to it'''
 396         if keys and isinstance(keys, list):
 397             for keyed in keys:
 398                 if keyed and isinstance(keyed, dict):
 399 
 400                     variables = combine_vars(variables, self.inventory.get_host(host).get_vars())
 401                     try:
 402                         key = self._compose(keyed.get('key'), variables)
 403                     except Exception as e:
 404                         if strict:
 405                             raise AnsibleParserError("Could not generate group for host %s from %s entry: %s" % (host, keyed.get('key'), to_native(e)))
 406                         continue
 407 
 408                     if key:
 409                         prefix = keyed.get('prefix', '')
 410                         sep = keyed.get('separator', '_')
 411                         raw_parent_name = keyed.get('parent_group', None)
 412                         if raw_parent_name:
 413                             try:
 414                                 raw_parent_name = self.templar.template(raw_parent_name)
 415                             except AnsibleError as e:
 416                                 if strict:
 417                                     raise AnsibleParserError("Could not generate parent group %s for group %s: %s" % (raw_parent_name, key, to_native(e)))
 418                                 continue
 419 
 420                         new_raw_group_names = []
 421                         if isinstance(key, string_types):
 422                             new_raw_group_names.append(key)
 423                         elif isinstance(key, list):
 424                             for name in key:
 425                                 new_raw_group_names.append(name)
 426                         elif isinstance(key, Mapping):
 427                             for (gname, gval) in key.items():
 428                                 name = '%s%s%s' % (gname, sep, gval)
 429                                 new_raw_group_names.append(name)
 430                         else:
 431                             raise AnsibleParserError("Invalid group name format, expected a string or a list of them or dictionary, got: %s" % type(key))
 432 
 433                         for bare_name in new_raw_group_names:
 434                             gname = self._sanitize_group_name('%s%s%s' % (prefix, sep, bare_name))
 435                             result_gname = self.inventory.add_group(gname)
 436                             self.inventory.add_host(host, result_gname)
 437 
 438                             if raw_parent_name:
 439                                 parent_name = self._sanitize_group_name(raw_parent_name)
 440                                 self.inventory.add_group(parent_name)
 441                                 self.inventory.add_child(parent_name, result_gname)
 442 
 443                     else:
 444                         # exclude case of empty list and dictionary, because these are valid constructions
 445                         # simply no groups need to be constructed, but are still falsy
 446                         if strict and key not in ([], {}):
 447                             raise AnsibleParserError("No key or key resulted empty for %s in host %s, invalid entry" % (keyed.get('key'), host))
 448                 else:
 449                     raise AnsibleParserError("Invalid keyed group entry, it must be a dictionary: %s " % keyed)
