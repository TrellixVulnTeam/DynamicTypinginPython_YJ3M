  70     def verify_file(self, path):
  71         ''' Verify if file is usable by this plugin, base does minimal accessibility check '''
  72 
  73         valid = super(InventoryModule, self).verify_file(path)
  74 
  75         if valid:
  76             # not only accessible, file must be executable and/or have shebang
  77             shebang_present = False
  78             try:
  79                 with open(path, 'rb') as inv_file:
  80                     initial_chars = inv_file.read(2)
  81                     if initial_chars.startswith(b'#!'):
  82                         shebang_present = True
  83             except Exception:
  84                 pass
  85 
  86             if not os.access(path, os.X_OK) and not shebang_present:
  87                 valid = False
  88 
  89         return valid

  91     def parse(self, inventory, loader, path, cache=None):
  92 
  93         super(InventoryModule, self).parse(inventory, loader, path)
  94         self.set_options()
  95 
  96         if self.get_option('cache') is not None:
  97             display.deprecated(
  98                 msg="The 'cache' option is deprecated for the script inventory plugin. "
  99                 "External scripts implement their own caching and this option has never been used",
 100                 version="2.12", collection_name='ansible.builtin'
 101             )
 102 
 103         # Support inventory scripts that are not prefixed with some
 104         # path information but happen to be in the current working
 105         # directory when '.' is not in PATH.
 106         cmd = [path, "--list"]
 107 
 108         try:
 109             try:
 110                 sp = subprocess.Popen(cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
 111             except OSError as e:
 112                 raise AnsibleParserError("problem running %s (%s)" % (' '.join(cmd), to_native(e)))
 113             (stdout, stderr) = sp.communicate()
 114 
 115             path = to_native(path)
 116             err = to_native(stderr or "")
 117 
 118             if err and not err.endswith('\n'):
 119                 err += '\n'
 120 
 121             if sp.returncode != 0:
 122                 raise AnsibleError("Inventory script (%s) had an execution error: %s " % (path, err))
 123 
 124             # make sure script output is unicode so that json loader will output unicode strings itself
 125             try:
 126                 data = to_text(stdout, errors="strict")
 127             except Exception as e:
 128                 raise AnsibleError("Inventory {0} contained characters that cannot be interpreted as UTF-8: {1}".format(path, to_native(e)))
 129 
 130             try:
 131                 processed = self.loader.load(data, json_only=True)
 132             except Exception as e:
 133                 raise AnsibleError("failed to parse executable inventory script results from {0}: {1}\n{2}".format(path, to_native(e), err))
 134 
 135             # if no other errors happened and you want to force displaying stderr, do so now
 136             if stderr and self.get_option('always_show_stderr'):
 137                 self.display.error(msg=to_text(err))
 138 
 139             if not isinstance(processed, Mapping):
 140                 raise AnsibleError("failed to parse executable inventory script results from {0}: needs to be a json dict\n{1}".format(path, err))
 141 
 142             group = None
 143             data_from_meta = None
 144 
 145             # A "_meta" subelement may contain a variable "hostvars" which contains a hash for each host
 146             # if this "hostvars" exists at all then do not call --host for each # host.
 147             # This is for efficiency and scripts should still return data
 148             # if called with --host for backwards compat with 1.2 and earlier.
 149             for (group, gdata) in processed.items():
 150                 if group == '_meta':
 151                     if 'hostvars' in gdata:
 152                         data_from_meta = gdata['hostvars']
 153                 else:
 154                     self._parse_group(group, gdata)
 155 
 156             for host in self._hosts:
 157                 got = {}
 158                 if data_from_meta is None:
 159                     got = self.get_host_variables(path, host)
 160                 else:
 161                     try:
 162                         got = data_from_meta.get(host, {})
 163                     except AttributeError as e:
 164                         raise AnsibleError("Improperly formatted host information for %s: %s" % (host, to_native(e)), orig_exc=e)
 165 
 166                 self._populate_host_vars([host], got)
 167 
 168         except Exception as e:
 169             raise AnsibleParserError(to_native(e))
 170 
 171     def _parse_group(self, group, data):
 172 
 173         group = self.inventory.add_group(group)
 174 
 175         if not isinstance(data, dict):
 176             data = {'hosts': data}
 177         # is not those subkeys, then simplified syntax, host with vars
 178         elif not any(k in data for k in ('hosts', 'vars', 'children')):
 179             data = {'hosts': [group], 'vars': data}
 180 
 181         if 'hosts' in data:
 182             if not isinstance(data['hosts'], list):
 183                 raise AnsibleError("You defined a group '%s' with bad data for the host list:\n %s" % (group, data))
 184 
 185             for hostname in data['hosts']:
 186                 self._hosts.add(hostname)
 187                 self.inventory.add_host(hostname, group)
 188 
 189         if 'vars' in data:
 190             if not isinstance(data['vars'], dict):
 191                 raise AnsibleError("You defined a group '%s' with bad data for variables:\n %s" % (group, data))
 192 
 193             for k, v in iteritems(data['vars']):
 194                 self.inventory.set_variable(group, k, v)
 195 
 196         if group != '_meta' and isinstance(data, dict) and 'children' in data:
 197             for child_name in data['children']:
 198                 child_name = self.inventory.add_group(child_name)
 199                 self.inventory.add_child(group, child_name)
 200 
 201     def get_host_variables(self, path, host):
 202         """ Runs <script> --host <hostname>, to determine additional host variables """
 203 
 204         cmd = [path, "--host", host]
 205         try:
 206             sp = subprocess.Popen(cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
 207         except OSError as e:
 208             raise AnsibleError("problem running %s (%s)" % (' '.join(cmd), e))
 209         (out, err) = sp.communicate()
 210         if out.strip() == '':
 211             return {}
 212         try:
 213             return json_dict_bytes_to_unicode(self.loader.load(out, file_name=path))
 214         except ValueError:
 215             raise AnsibleError("could not parse post variable response: %s, %s" % (cmd, out))
