 115 if HAS_TOML and hasattr(toml, 'TomlEncoder'):
 116     class AnsibleTomlEncoder(toml.TomlEncoder):
 117         def __init__(self, *args, **kwargs):
 118             super(AnsibleTomlEncoder, self).__init__(*args, **kwargs)
 119             # Map our custom YAML object types to dump_funcs from ``toml``
 120             self.dump_funcs.update({
 121                 AnsibleSequence: self.dump_funcs.get(list),
 122                 AnsibleUnicode: self.dump_funcs.get(str),
 123             })
 124     toml_dumps = partial(toml.dumps, encoder=AnsibleTomlEncoder())

 130 def convert_yaml_objects_to_native(obj):
 131     """Older versions of the ``toml`` python library, don't have a pluggable
 132     way to tell the encoder about custom types, so we need to ensure objects
 133     that we pass are native types.
 134 
 135     Only used on ``toml<0.10.0`` where ``toml.TomlEncoder`` is missing.
 136 
 137     This function recurses an object and ensures we cast any of the types from
 138     ``ansible.parsing.yaml.objects`` into their native types, effectively cleansing
 139     the data before we hand it over to ``toml``
 140 
 141     This function doesn't directly check for the types from ``ansible.parsing.yaml.objects``
 142     but instead checks for the types those objects inherit from, to offer more flexibility.
 143     """
 144     if isinstance(obj, dict):
 145         return dict((k, convert_yaml_objects_to_native(v)) for k, v in obj.items())
 146     elif isinstance(obj, list):
 147         return [convert_yaml_objects_to_native(v) for v in obj]
 148     elif isinstance(obj, text_type):
 149         return text_type(obj)
 150     else:
 151         return obj

 154 class InventoryModule(BaseFileInventoryPlugin):
 155     NAME = 'toml'
 156 
 157     def _parse_group(self, group, group_data):
 158         if not isinstance(group_data, (MutableMapping, type(None))):
 159             self.display.warning("Skipping '%s' as this is not a valid group definition" % group)
 160             return
 161 
 162         group = self.inventory.add_group(group)
 163         if group_data is None:
 164             return
 165 
 166         for key, data in group_data.items():
 167             if key == 'vars':
 168                 if not isinstance(data, MutableMapping):
 169                     raise AnsibleParserError(
 170                         'Invalid "vars" entry for "%s" group, requires a dict, found "%s" instead.' %
 171                         (group, type(data))
 172                     )
 173                 for var, value in data.items():
 174                     self.inventory.set_variable(group, var, value)
 175 
 176             elif key == 'children':
 177                 if not isinstance(data, MutableSequence):
 178                     raise AnsibleParserError(
 179                         'Invalid "children" entry for "%s" group, requires a list, found "%s" instead.' %
 180                         (group, type(data))
 181                     )
 182                 for subgroup in data:
 183                     self._parse_group(subgroup, {})
 184                     self.inventory.add_child(group, subgroup)
 185 
 186             elif key == 'hosts':
 187                 if not isinstance(data, MutableMapping):
 188                     raise AnsibleParserError(
 189                         'Invalid "hosts" entry for "%s" group, requires a dict, found "%s" instead.' %
 190                         (group, type(data))
 191                     )
 192                 for host_pattern, value in data.items():
 193                     hosts, port = self._expand_hostpattern(host_pattern)
 194                     self._populate_host_vars(hosts, value, group, port)
 195             else:
 196                 self.display.warning(
 197                     'Skipping unexpected key "%s" in group "%s", only "vars", "children" and "hosts" are valid' %
 198                     (key, group)
 199                 )

 201     def _load_file(self, file_name):
 202         if not file_name or not isinstance(file_name, string_types):
 203             raise AnsibleParserError("Invalid filename: '%s'" % to_native(file_name))
 204 
 205         b_file_name = to_bytes(self.loader.path_dwim(file_name))
 206         if not self.loader.path_exists(b_file_name):
 207             raise AnsibleFileNotFound("Unable to retrieve file contents", file_name=file_name)
 208 
 209         try:
 210             (b_data, private) = self.loader._get_file_contents(file_name)
 211             return toml.loads(to_text(b_data, errors='surrogate_or_strict'))
 212         except toml.TomlDecodeError as e:
 213             raise AnsibleParserError(
 214                 'TOML file (%s) is invalid: %s' % (file_name, to_native(e)),
 215                 orig_exc=e
 216             )
 217         except (IOError, OSError) as e:
 218             raise AnsibleParserError(
 219                 "An error occurred while trying to read the file '%s': %s" % (file_name, to_native(e)),
 220                 orig_exc=e
 221             )
 222         except Exception as e:
 223             raise AnsibleParserError(
 224                 "An unexpected error occurred while parsing the file '%s': %s" % (file_name, to_native(e)),
 225                 orig_exc=e
 226             )

 228     def parse(self, inventory, loader, path, cache=True):
 229         ''' parses the inventory file '''
 230         if not HAS_TOML:
 231             raise AnsibleParserError(
 232                 'The TOML inventory plugin requires the python "toml" library'
 233             )
 234 
 235         display.warning(WARNING_MSG)
 236 
 237         super(InventoryModule, self).parse(inventory, loader, path)
 238         self.set_options()
 239 
 240         try:
 241             data = self._load_file(path)
 242         except Exception as e:
 243             raise AnsibleParserError(e)
 244 
 245         if not data:
 246             raise AnsibleParserError('Parsed empty TOML file')
 247         elif data.get('plugin'):
 248             raise AnsibleParserError('Plugin configuration TOML file, not TOML inventory')
 249 
 250         for group_name in data:
 251             self._parse_group(group_name, data[group_name])
 252 
 253     def verify_file(self, path):
 254         if super(InventoryModule, self).verify_file(path):
 255             file_name, ext = os.path.splitext(path)
 256             if ext == '.toml':
 257                 return True
 258         return False
