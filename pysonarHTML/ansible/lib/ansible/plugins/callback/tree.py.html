  20 import os
  21 
  22 from ansible.constants import TREE_DIR
  23 from ansible.module_utils._text import to_bytes, to_text
  24 from ansible.plugins.callback import CallbackBase
  25 from ansible.utils.path import makedirs_safe
  26 
  27 
  28 class CallbackModule(CallbackBase):
  29     '''
  30     This callback puts results into a host specific file in a directory in json format.
  31     '''
  32 
  33     CALLBACK_VERSION = 2.0
  34     CALLBACK_TYPE = 'aggregate'
  35     CALLBACK_NAME = 'tree'
  36     CALLBACK_NEEDS_WHITELIST = True
  37 
  38     def __init__(self):
  39         super(CallbackModule, self).__init__()
  40 
  41         self.tree = TREE_DIR
  42         if not self.tree:
  43             self.tree = os.path.expanduser("~/.ansible/tree")
  44             self._display.warning("The tree callback is defaulting to ~/.ansible/tree, as an invalid directory was provided: %s" % self.tree)
  45 
  46     def write_tree_file(self, hostname, buf):
  47         ''' write something into treedir/hostname '''
  48 
  49         buf = to_bytes(buf)
  50         try:
  51             makedirs_safe(self.tree)
  52             path = os.path.join(self.tree, hostname)
  53             with open(path, 'wb+') as fd:
  54                 fd.write(buf)
  55         except (OSError, IOError) as e:
  56             self._display.warning(u"Unable to write to %s's file: %s" % (hostname, to_text(e)))
  57 
  58     def result_to_tree(self, result):
  59         if self.tree:
  60             self.write_tree_file(result._host.get_name(), self._dump_results(result._result))
  61 
  62     def v2_runner_on_ok(self, result):
  63         self.result_to_tree(result)
  64 
  65     def v2_runner_on_failed(self, result, ignore_errors=False):
  66         self.result_to_tree(result)
  67 
  68     def v2_runner_on_unreachable(self, result):
  69         self.result_to_tree(result)
