  28 # These values use ansible.constants for historical reasons, mostly to allow
  29 # unmodified derivative plugins to work. However, newer options added to the
  30 # plugin are not also added to ansible.constants, so authors of derivative
  31 # callback plugins will eventually need to add a reference to the common docs
  32 # fragment for the 'default' callback plugin
  33 
  34 # these are used to provide backwards compat with old plugins that subclass from default
  35 # but still don't use the new config system and/or fail to document the options
  36 # TODO: Change the default of check_mode_markers to True in a future release (2.13)

  38                   ('display_ok_hosts', True),
  39                   ('show_custom_stats', C.SHOW_CUSTOM_STATS),
  40                   ('display_failed_stderr', False),
  41                   ('check_mode_markers', False),
  42                   ('show_per_host_start', False))

  48     This is the default callback interface, which simply prints messages
  49     to stdout when new callback events are received.

  56     def __init__(self):
  57 
  58         self._play = None
  59         self._last_task_banner = None
  60         self._last_task_name = None
  61         self._task_type_cache = {}

  64     def set_options(self, task_keys=None, var_options=None, direct=None):
  65 
  66         super(CallbackModule, self).set_options(task_keys=task_keys, var_options=var_options, direct=direct)
  67 
  68         # for backwards compat with plugins subclassing default, fallback to constants
  69         for option, constant in COMPAT_OPTIONS:
  70             try:
  71                 value = self.get_option(option)
  72             except (AttributeError, KeyError):
  73                 self._display.deprecated("'%s' is subclassing DefaultCallback without the corresponding doc_fragment." % self._load_name,
  74                                          version='2.14', collection_name='ansible.builtin')
  75                 value = constant
  76             setattr(self, option, value)

  78     def v2_runner_on_failed(self, result, ignore_errors=False):
  79 
  80         delegated_vars = result._result.get('_ansible_delegated_vars', None)
  81         self._clean_results(result._result, result._task.action)
  82 
  83         if self._last_task_banner != result._task._uuid:
  84             self._print_task_banner(result._task)
  85 
  86         self._handle_exception(result._result, use_stderr=self.display_failed_stderr)
  87         self._handle_warnings(result._result)
  88 
  89         if result._task.loop and 'results' in result._result:
  90             self._process_items(result)
  91 
  92         else:
  93             if delegated_vars:
  94                 self._display.display("fatal: [%s -> %s]: FAILED! => %s" % (result._host.get_name(), delegated_vars['ansible_host'],
  95                                                                             self._dump_results(result._result)),
  96                                       color=C.COLOR_ERROR, stderr=self.display_failed_stderr)
  97             else:
  98                 self._display.display("fatal: [%s]: FAILED! => %s" % (result._host.get_name(), self._dump_results(result._result)),
  99                                       color=C.COLOR_ERROR, stderr=self.display_failed_stderr)
 100 
 101         if ignore_errors:
 102             self._display.display("...ignoring", color=C.COLOR_SKIP)

 104     def v2_runner_on_ok(self, result):
 105 
 106         delegated_vars = result._result.get('_ansible_delegated_vars', None)
 107 
 108         if isinstance(result._task, TaskInclude):
 109             return
 110         elif result._result.get('changed', False):
 111             if self._last_task_banner != result._task._uuid:
 112                 self._print_task_banner(result._task)
 113 
 114             if delegated_vars:
 115                 msg = "changed: [%s -> %s]" % (result._host.get_name(), delegated_vars['ansible_host'])
 116             else:
 117                 msg = "changed: [%s]" % result._host.get_name()
 118             color = C.COLOR_CHANGED
 119         else:
 120             if not self.display_ok_hosts:
 121                 return
 122 
 123             if self._last_task_banner != result._task._uuid:
 124                 self._print_task_banner(result._task)
 125 
 126             if delegated_vars:
 127                 msg = "ok: [%s -> %s]" % (result._host.get_name(), delegated_vars['ansible_host'])
 128             else:
 129                 msg = "ok: [%s]" % result._host.get_name()
 130             color = C.COLOR_OK
 131 
 132         self._handle_warnings(result._result)
 133 
 134         if result._task.loop and 'results' in result._result:
 135             self._process_items(result)
 136         else:
 137             self._clean_results(result._result, result._task.action)
 138 
 139             if self._run_is_verbose(result):
 140                 msg += " => %s" % (self._dump_results(result._result),)
 141             self._display.display(msg, color=color)
 142 
 143     def v2_runner_on_skipped(self, result):
 144 
 145         if self.display_skipped_hosts:
 146 
 147             self._clean_results(result._result, result._task.action)
 148 
 149             if self._last_task_banner != result._task._uuid:
 150                 self._print_task_banner(result._task)
 151 
 152             if result._task.loop and 'results' in result._result:
 153                 self._process_items(result)
 154             else:
 155                 msg = "skipping: [%s]" % result._host.get_name()
 156                 if self._run_is_verbose(result):
 157                     msg += " => %s" % self._dump_results(result._result)
 158                 self._display.display(msg, color=C.COLOR_SKIP)

 160     def v2_runner_on_unreachable(self, result):
 161         if self._last_task_banner != result._task._uuid:
 162             self._print_task_banner(result._task)
 163 
 164         delegated_vars = result._result.get('_ansible_delegated_vars', None)
 165         if delegated_vars:
 166             msg = "fatal: [%s -> %s]: UNREACHABLE! => %s" % (result._host.get_name(), delegated_vars['ansible_host'], self._dump_results(result._result))
 167         else:
 168             msg = "fatal: [%s]: UNREACHABLE! => %s" % (result._host.get_name(), self._dump_results(result._result))
 169         self._display.display(msg, color=C.COLOR_UNREACHABLE, stderr=self.display_failed_stderr)

 180     def _task_start(self, task, prefix=None):
 181         # Cache output prefix for task if provided
 182         # This is needed to properly display 'RUNNING HANDLER' and similar
 183         # when hiding skipped/ok task results
 184         if prefix is not None:
 185             self._task_type_cache[task._uuid] = prefix
 186 
 187         # Preserve task name, as all vars may not be available for templating
 188         # when we need it later
 189         if self._play.strategy == 'free':
 190             # Explicitly set to None for strategy 'free' to account for any cached
 191             # task title from a previous non-free play
 192             self._last_task_name = None
 193         else:
 194             self._last_task_name = task.get_name().strip()
 195 
 196             # Display the task banner immediately if we're not doing any filtering based on task result
 197             if self.display_skipped_hosts and self.display_ok_hosts:
 198                 self._print_task_banner(task)
 199 
 200     def _print_task_banner(self, task):
 201         # args can be specified as no_log in several places: in the task or in
 202         # the argument spec.  We can check whether the task is no_log but the
 203         # argument spec can't be because that is only run on the target
 204         # machine and we haven't run it thereyet at this time.
 205         #
 206         # So we give people a config option to affect display of the args so
 207         # that they can secure this if they feel that their stdout is insecure
 208         # (shoulder surfing, logging stdout straight to a file, etc).
 209         args = ''
 210         if not task.no_log and C.DISPLAY_ARGS_TO_STDOUT:
 211             args = u', '.join(u'%s=%s' % a for a in task.args.items())
 212             args = u' %s' % args
 213 
 214         prefix = self._task_type_cache.get(task._uuid, 'TASK')
 215 
 216         # Use cached task name
 217         task_name = self._last_task_name
 218         if task_name is None:
 219             task_name = task.get_name().strip()
 220 
 221         if task.check_mode and self.check_mode_markers:
 222             checkmsg = " [CHECK MODE]"
 223         else:
 224             checkmsg = ""
 225         self._display.banner(u"%s [%s%s]%s" % (prefix, task_name, args, checkmsg))
 226         if self._display.verbosity >= 2:
 227             path = task.get_path()
 228             if path:
 229                 self._display.display(u"task path: %s" % path, color=C.COLOR_DEBUG)
 230 
 231         self._last_task_banner = task._uuid
 232

 243     def v2_playbook_on_play_start(self, play):
 244         name = play.get_name().strip()
 245         if play.check_mode and self.check_mode_markers:
 246             checkmsg = " [CHECK MODE]"
 247         else:
 248             checkmsg = ""
 249         if not name:
 250             msg = u"PLAY%s" % checkmsg
 251         else:
 252             msg = u"PLAY [%s]%s" % (name, checkmsg)
 253 
 254         self._play = play
 255 
 256         self._display.banner(msg)

 258     def v2_on_file_diff(self, result):
 259         if result._task.loop and 'results' in result._result:
 260             for res in result._result['results']:
 261                 if 'diff' in res and res['diff'] and res.get('changed', False):
 262                     diff = self._get_diff(res['diff'])
 263                     if diff:
 264                         if self._last_task_banner != result._task._uuid:
 265                             self._print_task_banner(result._task)
 266                         self._display.display(diff)
 267         elif 'diff' in result._result and result._result['diff'] and result._result.get('changed', False):
 268             diff = self._get_diff(result._result['diff'])
 269             if diff:
 270                 if self._last_task_banner != result._task._uuid:
 271                     self._print_task_banner(result._task)
 272                 self._display.display(diff)

 274     def v2_runner_item_on_ok(self, result):
 275 
 276         delegated_vars = result._result.get('_ansible_delegated_vars', None)
 277         if isinstance(result._task, TaskInclude):
 278             return
 279         elif result._result.get('changed', False):
 280             if self._last_task_banner != result._task._uuid:
 281                 self._print_task_banner(result._task)
 282 
 283             msg = 'changed'
 284             color = C.COLOR_CHANGED
 285         else:
 286             if not self.display_ok_hosts:
 287                 return
 288 
 289             if self._last_task_banner != result._task._uuid:
 290                 self._print_task_banner(result._task)
 291 
 292             msg = 'ok'
 293             color = C.COLOR_OK
 294 
 295         if delegated_vars:
 296             msg += ": [%s -> %s]" % (result._host.get_name(), delegated_vars['ansible_host'])
 297         else:
 298             msg += ": [%s]" % result._host.get_name()
 299 
 300         msg += " => (item=%s)" % (self._get_item_label(result._result),)
 301 
 302         self._clean_results(result._result, result._task.action)
 303         if self._run_is_verbose(result):
 304             msg += " => %s" % self._dump_results(result._result)
 305         self._display.display(msg, color=color)
 306 
 307     def v2_runner_item_on_failed(self, result):
 308         if self._last_task_banner != result._task._uuid:
 309             self._print_task_banner(result._task)
 310 
 311         delegated_vars = result._result.get('_ansible_delegated_vars', None)
 312         self._clean_results(result._result, result._task.action)
 313         self._handle_exception(result._result)
 314 
 315         msg = "failed: "
 316         if delegated_vars:
 317             msg += "[%s -> %s]" % (result._host.get_name(), delegated_vars['ansible_host'])
 318         else:
 319             msg += "[%s]" % (result._host.get_name())
 320 
 321         self._handle_warnings(result._result)
 322         self._display.display(msg + " (item=%s) => %s" % (self._get_item_label(result._result), self._dump_results(result._result)), color=C.COLOR_ERROR)
 323 
 324     def v2_runner_item_on_skipped(self, result):
 325         if self.display_skipped_hosts:
 326             if self._last_task_banner != result._task._uuid:
 327                 self._print_task_banner(result._task)
 328 
 329             self._clean_results(result._result, result._task.action)
 330             msg = "skipping: [%s] => (item=%s) " % (result._host.get_name(), self._get_item_label(result._result))
 331             if self._run_is_verbose(result):
 332                 msg += " => %s" % self._dump_results(result._result)
 333             self._display.display(msg, color=C.COLOR_SKIP)

 335     def v2_playbook_on_include(self, included_file):
 336         msg = 'included: %s for %s' % (included_file._filename, ", ".join([h.name for h in included_file._hosts]))
 337         label = self._get_item_label(included_file._vars)
 338         if label:
 339             msg += " => (item=%s)" % label
 340         self._display.display(msg, color=C.COLOR_SKIP)

 342     def v2_playbook_on_stats(self, stats):
 343         self._display.banner("PLAY RECAP")
 344 
 345         hosts = sorted(stats.processed.keys())
 346         for h in hosts:
 347             t = stats.summarize(h)
 348 
 349             self._display.display(
 350                 u"%s : %s %s %s %s %s %s %s" % (
 351                     hostcolor(h, t),
 352                     colorize(u'ok', t['ok'], C.COLOR_OK),
 353                     colorize(u'changed', t['changed'], C.COLOR_CHANGED),
 354                     colorize(u'unreachable', t['unreachable'], C.COLOR_UNREACHABLE),
 355                     colorize(u'failed', t['failures'], C.COLOR_ERROR),
 356                     colorize(u'skipped', t['skipped'], C.COLOR_SKIP),
 357                     colorize(u'rescued', t['rescued'], C.COLOR_OK),
 358                     colorize(u'ignored', t['ignored'], C.COLOR_WARN),
 359                 ),
 360                 screen_only=True
 361             )
 362 
 363             self._display.display(
 364                 u"%s : %s %s %s %s %s %s %s" % (
 365                     hostcolor(h, t, False),
 366                     colorize(u'ok', t['ok'], None),
 367                     colorize(u'changed', t['changed'], None),
 368                     colorize(u'unreachable', t['unreachable'], None),
 369                     colorize(u'failed', t['failures'], None),
 370                     colorize(u'skipped', t['skipped'], None),
 371                     colorize(u'rescued', t['rescued'], None),
 372                     colorize(u'ignored', t['ignored'], None),
 373                 ),
 374                 log_only=True
 375             )
 376 
 377         self._display.display("", screen_only=True)
 378 
 379         # print custom stats if required
 380         if stats.custom and self.show_custom_stats:
 381             self._display.banner("CUSTOM STATS: ")
 382             # per host
 383             # TODO: come up with 'pretty format'
 384             for k in sorted(stats.custom.keys()):
 385                 if k == '_run':
 386                     continue
 387                 self._display.display('\t%s: %s' % (k, self._dump_results(stats.custom[k], indent=1).replace('\n', '')))
 388 
 389             # print per run custom stats
 390             if '_run' in stats.custom:
 391                 self._display.display("", screen_only=True)
 392                 self._display.display('\tRUN: %s' % self._dump_results(stats.custom['_run'], indent=1).replace('\n', ''))
 393             self._display.display("", screen_only=True)
 394 
 395         if context.CLIARGS['check'] and self.check_mode_markers:
 396             self._display.banner("DRY RUN")
 397 
 398     def v2_playbook_on_start(self, playbook):
 399         if self._display.verbosity > 1:
 400             from os.path import basename
 401             self._display.banner("PLAYBOOK: %s" % basename(playbook._file_name))
 402 
 403         # show CLI arguments
 404         if self._display.verbosity > 3:
 405             if context.CLIARGS.get('args'):
 406                 self._display.display('Positional arguments: %s' % ' '.join(context.CLIARGS['args']),
 407                                       color=C.COLOR_VERBOSE, screen_only=True)
 408 
 409             for argument in (a for a in context.CLIARGS if a != 'args'):
 410                 val = context.CLIARGS[argument]
 411                 if val:
 412                     self._display.display('%s: %s' % (argument, val), color=C.COLOR_VERBOSE, screen_only=True)
 413 
 414         if context.CLIARGS['check'] and self.check_mode_markers:
 415             self._display.banner("DRY RUN")
 416 
 417     def v2_runner_retry(self, result):
 418         task_name = result.task_name or result._task
 419         msg = "FAILED - RETRYING: %s (%d retries left)." % (task_name, result._result['retries'] - result._result['attempts'])
 420         if self._run_is_verbose(result, verbosity=2):
 421             msg += "Result was: %s" % self._dump_results(result._result)
 422         self._display.display(msg, color=C.COLOR_DEBUG)
