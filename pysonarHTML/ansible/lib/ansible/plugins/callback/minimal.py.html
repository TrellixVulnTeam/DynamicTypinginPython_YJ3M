  18 from ansible.plugins.callback import CallbackBase
  19 from ansible import constants as C
  20 
  21 
  22 class CallbackModule(CallbackBase):
  23 
  24     '''
  25     This is the default callback interface, which simply prints messages
  26     to stdout when new callback events are received.
  27     '''
  28 
  29     CALLBACK_VERSION = 2.0
  30     CALLBACK_TYPE = 'stdout'
  31     CALLBACK_NAME = 'minimal'
  32 
  33     def _command_generic_msg(self, host, result, caption):
  34         ''' output the result of a command run '''
  35 
  36         buf = "%s | %s | rc=%s >>\n" % (host, caption, result.get('rc', -1))
  37         buf += result.get('stdout', '')
  38         buf += result.get('stderr', '')
  39         buf += result.get('msg', '')
  40 
  41         return buf + "\n"
  42 
  43     def v2_runner_on_failed(self, result, ignore_errors=False):
  44 
  45         self._handle_exception(result._result)
  46         self._handle_warnings(result._result)
  47 
  48         if result._task.action in C.MODULE_NO_JSON and 'module_stderr' not in result._result:
  49             self._display.display(self._command_generic_msg(result._host.get_name(), result._result, "FAILED"), color=C.COLOR_ERROR)
  50         else:
  51             self._display.display("%s | FAILED! => %s" % (result._host.get_name(), self._dump_results(result._result, indent=4)), color=C.COLOR_ERROR)
  52 
  53     def v2_runner_on_ok(self, result):
  54         self._clean_results(result._result, result._task.action)
  55 
  56         self._handle_warnings(result._result)
  57 
  58         if result._result.get('changed', False):
  59             color = C.COLOR_CHANGED
  60             state = 'CHANGED'
  61         else:
  62             color = C.COLOR_OK
  63             state = 'SUCCESS'
  64 
  65         if result._task.action in C.MODULE_NO_JSON and 'ansible_job_id' not in result._result:
  66             self._display.display(self._command_generic_msg(result._host.get_name(), result._result, state), color=color)
  67         else:
  68             self._display.display("%s | %s => %s" % (result._host.get_name(), state, self._dump_results(result._result, indent=4)), color=color)
  69 
  70     def v2_runner_on_skipped(self, result):
  71         self._display.display("%s | SKIPPED" % (result._host.get_name()), color=C.COLOR_SKIP)
  72 
  73     def v2_runner_on_unreachable(self, result):
  74         self._display.display("%s | UNREACHABLE! => %s" % (result._host.get_name(), self._dump_results(result._result, indent=4)), color=C.COLOR_UNREACHABLE)
  75 
  76     def v2_on_file_diff(self, result):
  77         if 'diff' in result._result and result._result['diff']:
  78             self._display.display(self._get_diff(result._result['diff']))
