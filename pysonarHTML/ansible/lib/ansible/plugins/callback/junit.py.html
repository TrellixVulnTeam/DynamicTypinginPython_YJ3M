 113 class CallbackModule(CallbackBase):
 114     """
 115     This callback writes playbook output to a JUnit formatted XML file.
 116 
 117     Tasks show up in the report as follows:
 118         'ok': pass
 119         'failed' with 'EXPECTED FAILURE' in the task name: pass
 120         'failed' with 'TOGGLE RESULT' in the task name: pass
 121         'ok' with 'TOGGLE RESULT' in the task name: failure
 122         'failed' due to an exception: error
 123         'failed' for other reasons: failure
 124         'skipped': skipped
 125 
 126     This plugin makes use of the following environment variables:
 127         JUNIT_OUTPUT_DIR (optional): Directory to write XML files to.
 128                                      Default: ~/.ansible.log
 129         JUNIT_TASK_CLASS (optional): Configure the output to be one class per yaml file
 130                                      Default: False
 131         JUNIT_TASK_RELATIVE_PATH (optional): Configure the output to use relative paths to given directory
 132                                      Default: none
 133         JUNIT_FAIL_ON_CHANGE (optional): Consider any tasks reporting "changed" as a junit test failure
 134                                      Default: False
 135         JUNIT_FAIL_ON_IGNORE (optional): Consider failed tasks as a junit test failure even if ignore_on_error is set
 136                                      Default: False
 137         JUNIT_INCLUDE_SETUP_TASKS_IN_REPORT (optional): Should the setup tasks be included in the final report
 138                                      Default: True
 139         JUNIT_HIDE_TASK_ARGUMENTS (optional): Hide the arguments for a task
 140                                      Default: False
 141         JUNIT_TEST_CASE_PREFIX (optional): Consider a task only as test case if it has this value as prefix. Additionaly failing tasks are recorded as failed
 142                                      test cases.
 143                                      Default: <empty>
 144 
 145     Requires:
 146         junit_xml
 147 
 148     """
 149 
 150     CALLBACK_VERSION = 2.0
 151     CALLBACK_TYPE = 'aggregate'
 152     CALLBACK_NAME = 'junit'
 153     CALLBACK_NEEDS_WHITELIST = True
 154 
 155     def __init__(self):
 156         super(CallbackModule, self).__init__()
 157 
 158         self._output_dir = os.getenv('JUNIT_OUTPUT_DIR', os.path.expanduser('~/.ansible.log'))
 159         self._task_class = os.getenv('JUNIT_TASK_CLASS', 'False').lower()
 160         self._task_relative_path = os.getenv('JUNIT_TASK_RELATIVE_PATH', '')
 161         self._fail_on_change = os.getenv('JUNIT_FAIL_ON_CHANGE', 'False').lower()
 162         self._fail_on_ignore = os.getenv('JUNIT_FAIL_ON_IGNORE', 'False').lower()
 163         self._include_setup_tasks_in_report = os.getenv('JUNIT_INCLUDE_SETUP_TASKS_IN_REPORT', 'True').lower()
 164         self._hide_task_arguments = os.getenv('JUNIT_HIDE_TASK_ARGUMENTS', 'False').lower()
 165         self._test_case_prefix = os.getenv('JUNIT_TEST_CASE_PREFIX', '')
 166         self._playbook_path = None
 167         self._playbook_name = None
 168         self._play_name = None
 169         self._task_data = None
 170 
 171         self.disabled = False
 172 
 173         if not HAS_JUNIT_XML:
 174             self.disabled = True
 175             self._display.warning('The `junit_xml` python module is not installed. '
 176                                   'Disabling the `junit` callback plugin.')
 177 
 178         if HAS_ORDERED_DICT:
 179             self._task_data = OrderedDict()
 180         else:
 181             self.disabled = True
 182             self._display.warning('The `ordereddict` python module is not installed. '
 183                                   'Disabling the `junit` callback plugin.')
 184 
 185         if not os.path.exists(self._output_dir):
 186             os.makedirs(self._output_dir)
 187 
 188     def _start_task(self, task):
 189         """ record the start of a task for one or more hosts """
 190 
 191         uuid = task._uuid
 192 
 193         if uuid in self._task_data:
 194             return
 195 
 196         play = self._play_name
 197         name = task.get_name().strip()
 198         path = task.get_path()
 199         action = task.action
 200 
 201         if not task.no_log and self._hide_task_arguments == 'false':
 202             args = ', '.join(('%s=%s' % a for a in task.args.items()))
 203             if args:
 204                 name += ' ' + args
 205 
 206         self._task_data[uuid] = TaskData(uuid, name, path, play, action)
 207 
 208     def _finish_task(self, status, result):
 209         """ record the results of a task for a single host """
 210 
 211         task_uuid = result._task._uuid
 212 
 213         if hasattr(result, '_host'):
 214             host_uuid = result._host._uuid
 215             host_name = result._host.name
 216         else:
 217             host_uuid = 'include'
 218             host_name = 'include'
 219 
 220         task_data = self._task_data[task_uuid]
 221 
 222         if self._fail_on_change == 'true' and status == 'ok' and result._result.get('changed', False):
 223             status = 'failed'
 224 
 225         # ignore failure if expected and toggle result if asked for
 226         if status == 'failed' and 'EXPECTED FAILURE' in task_data.name:
 227             status = 'ok'
 228         elif 'TOGGLE RESULT' in task_data.name:
 229             if status == 'failed':
 230                 status = 'ok'
 231             elif status == 'ok':
 232                 status = 'failed'
 233 
 234         if task_data.name.startswith(self._test_case_prefix) or status == 'failed':
 235             task_data.add_host(HostData(host_uuid, host_name, status, result))
 236 
 237     def _build_test_case(self, task_data, host_data):
 238         """ build a TestCase from the given TaskData and HostData """
 239 
 240         name = '[%s] %s: %s' % (host_data.name, task_data.play, task_data.name)
 241         duration = host_data.finish - task_data.start
 242 
 243         if self._task_relative_path:
 244             junit_classname = os.path.relpath(task_data.path, self._task_relative_path)
 245         else:
 246             junit_classname = task_data.path
 247 
 248         if self._task_class == 'true':
 249             junit_classname = re.sub(r'\.yml:[0-9]+$', '', junit_classname)
 250 
 251         if host_data.status == 'included':
 252             return TestCase(name, junit_classname, duration, host_data.result)
 253 
 254         res = host_data.result._result
 255         rc = res.get('rc', 0)
 256         dump = self._dump_results(res, indent=0)
 257         dump = self._cleanse_string(dump)
 258 
 259         if host_data.status == 'ok':
 260             return TestCase(name, junit_classname, duration, dump)
 261 
 262         test_case = TestCase(name, junit_classname, duration)
 263 
 264         if host_data.status == 'failed':
 265             if 'exception' in res:
 266                 message = res['exception'].strip().split('\n')[-1]
 267                 output = res['exception']
 268                 test_case.add_error_info(message, output)
 269             elif 'msg' in res:
 270                 message = res['msg']
 271                 test_case.add_failure_info(message, dump)
 272             else:
 273                 test_case.add_failure_info('rc=%s' % rc, dump)
 274         elif host_data.status == 'skipped':
 275             if 'skip_reason' in res:
 276                 message = res['skip_reason']
 277             else:
 278                 message = 'skipped'
 279             test_case.add_skipped_info(message)
 280 
 281         return test_case
 282 
 283     def _cleanse_string(self, value):
 284         """ convert surrogate escapes to the unicode replacement character to avoid XML encoding errors """
 285         return to_text(to_bytes(value, errors='surrogateescape'), errors='replace')
 286 
 287     def _generate_report(self):
 288         """ generate a TestSuite report from the collected TaskData and HostData """
 289 
 290         test_cases = []
 291 
 292         for task_uuid, task_data in self._task_data.items():
 293             if task_data.action == 'setup' and self._include_setup_tasks_in_report == 'false':
 294                 continue
 295 
 296             for host_uuid, host_data in task_data.host_data.items():
 297                 test_cases.append(self._build_test_case(task_data, host_data))
 298 
 299         test_suite = TestSuite(self._playbook_name, test_cases)
 300         report = to_xml_report_string([test_suite])
 301 
 302         output_file = os.path.join(self._output_dir, '%s-%s.xml' % (self._playbook_name, time.time()))
 303 
 304         with open(output_file, 'wb') as xml:
 305             xml.write(to_bytes(report, errors='surrogate_or_strict'))

 307     def v2_playbook_on_start(self, playbook):
 308         self._playbook_path = playbook._file_name
 309         self._playbook_name = os.path.splitext(os.path.basename(self._playbook_path))[0]

 311     def v2_playbook_on_play_start(self, play):
 312         self._play_name = play.get_name()

 314     def v2_runner_on_no_hosts(self, task):
 315         self._start_task(task)
 316 
 317     def v2_playbook_on_task_start(self, task, is_conditional):
 318         self._start_task(task)
 319 
 320     def v2_playbook_on_cleanup_task_start(self, task):
 321         self._start_task(task)
 322 
 323     def v2_playbook_on_handler_task_start(self, task):
 324         self._start_task(task)

 326     def v2_runner_on_failed(self, result, ignore_errors=False):
 327         if ignore_errors and self._fail_on_ignore != 'true':
 328             self._finish_task('ok', result)
 329         else:
 330             self._finish_task('failed', result)
 331 
 332     def v2_runner_on_ok(self, result):
 333         self._finish_task('ok', result)
 334 
 335     def v2_runner_on_skipped(self, result):
 336         self._finish_task('skipped', result)
 337 
 338     def v2_playbook_on_include(self, included_file):
 339         self._finish_task('included', included_file)
 340 
 341     def v2_playbook_on_stats(self, stats):
 342         self._generate_report()

 345 class TaskData:
 346     """
 347     Data about an individual task.
 348     """
 349 
 350     def __init__(self, uuid, name, path, play, action):
 351         self.uuid = uuid
 352         self.name = name
 353         self.path = path
 354         self.play = play
 355         self.start = None
 356         self.host_data = OrderedDict()
 357         self.start = time.time()
 358         self.action = action
 359 
 360     def add_host(self, host):
 361         if host.uuid in self.host_data:
 362             if host.status == 'included':
 363                 # concatenate task include output from multiple items
 364                 host.result = '%s\n%s' % (self.host_data[host.uuid].result, host.result)
 365             else:
 366                 raise Exception('%s: %s: %s: duplicate host callback: %s' % (self.path, self.play, self.name, host.name))
 367 
 368         self.host_data[host.uuid] = host
