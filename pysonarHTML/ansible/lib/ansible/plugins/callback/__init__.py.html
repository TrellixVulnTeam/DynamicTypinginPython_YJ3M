  64     def __init__(self, display=None, options=None):
  65         if display:
  66             self._display = display
  67         else:
  68             self._display = global_display
  69 
  70         if self._display.verbosity >= 4:
  71             name = getattr(self, 'CALLBACK_NAME', 'unnamed')
  72             ctype = getattr(self, 'CALLBACK_TYPE', 'old')
  73             version = getattr(self, 'CALLBACK_VERSION', '1.0')
  74             self._display.vvvv('Loading callback plugin %s of type %s, v%s from %s' % (name, ctype, version, sys.modules[self.__module__].__file__))
  75 
  76         self.disabled = False
  77 
  78         self._plugin_options = {}
  79         if options is not None:
  80             self.set_options(options)
  81 
  82         self._hide_in_debug = ('changed', 'failed', 'skipped', 'invocation', 'skip_reason')

 105     def _dump_results(self, result, indent=None, sort_keys=True, keep_invocation=False):
 106 
 107         if not indent and (result.get('_ansible_verbose_always') or self._display.verbosity > 2):
 108             indent = 4
 109 
 110         # All result keys stating with _ansible_ are internal, so remove them from the result before we output anything.
 111         abridged_result = strip_internal_keys(module_response_deepcopy(result))
 112 
 113         # remove invocation unless specifically wanting it
 114         if not keep_invocation and self._display.verbosity < 3 and 'invocation' in result:
 115             del abridged_result['invocation']
 116 
 117         # remove diff information from screen output
 118         if self._display.verbosity < 3 and 'diff' in result:
 119             del abridged_result['diff']
 120 
 121         # remove exception from screen output
 122         if 'exception' in abridged_result:
 123             del abridged_result['exception']
 124 
 125         try:
 126             jsonified_results = json.dumps(abridged_result, cls=AnsibleJSONEncoder, indent=indent, ensure_ascii=False, sort_keys=sort_keys)
 127         except TypeError:
 128             # Python3 bug: throws an exception when keys are non-homogenous types:
 129             # https://bugs.python.org/issue25457
 130             # sort into an OrderedDict and then json.dumps() that instead
 131             if not OrderedDict:
 132                 raise
 133             jsonified_results = json.dumps(OrderedDict(sorted(abridged_result.items(), key=to_text)),
 134                                            cls=AnsibleJSONEncoder, indent=indent,
 135                                            ensure_ascii=False, sort_keys=False)
 136         return jsonified_results

 138     def _handle_warnings(self, res):
 139         ''' display warnings, if enabled and any exist in the result '''
 140         if C.ACTION_WARNINGS:
 141             if 'warnings' in res and res['warnings']:
 142                 for warning in res['warnings']:
 143                     self._display.warning(warning)
 144                 del res['warnings']
 145             if 'deprecations' in res and res['deprecations']:
 146                 for warning in res['deprecations']:
 147                     self._display.deprecated(**warning)
 148                 del res['deprecations']
 149 
 150     def _handle_exception(self, result, use_stderr=False):
 151 
 152         if 'exception' in result:
 153             msg = "An exception occurred during task execution. "
 154             if self._display.verbosity < 3:
 155                 # extract just the actual error message from the exception text
 156                 error = result['exception'].strip().split('\n')[-1]
 157                 msg += "To see the full traceback, use -vvv. The error was: %s" % error
 158             else:
 159                 msg = "The full traceback is:\n" + result['exception']
 160                 del result['exception']
 161 
 162             self._display.display(msg, color=C.COLOR_ERROR, stderr=use_stderr)

 164     def _serialize_diff(self, diff):
 165         return json.dumps(diff, sort_keys=True, indent=4, separators=(u',', u': ')) + u'\n'

 167     def _get_diff(self, difflist):
 168 
 169         if not isinstance(difflist, list):
 170             difflist = [difflist]
 171 
 172         ret = []
 173         for diff in difflist:
 174             if 'dst_binary' in diff:
 175                 ret.append(u"diff skipped: destination file appears to be binary\n")
 176             if 'src_binary' in diff:
 177                 ret.append(u"diff skipped: source file appears to be binary\n")
 178             if 'dst_larger' in diff:
 179                 ret.append(u"diff skipped: destination file size is greater than %d\n" % diff['dst_larger'])
 180             if 'src_larger' in diff:
 181                 ret.append(u"diff skipped: source file size is greater than %d\n" % diff['src_larger'])
 182             if 'before' in diff and 'after' in diff:
 183                 # format complex structures into 'files'
 184                 for x in ['before', 'after']:
 185                     if isinstance(diff[x], MutableMapping):
 186                         diff[x] = self._serialize_diff(diff[x])
 187                     elif diff[x] is None:
 188                         diff[x] = ''
 189                 if 'before_header' in diff:
 190                     before_header = u"before: %s" % diff['before_header']
 191                 else:
 192                     before_header = u'before'
 193                 if 'after_header' in diff:
 194                     after_header = u"after: %s" % diff['after_header']
 195                 else:
 196                     after_header = u'after'
 197                 before_lines = diff['before'].splitlines(True)
 198                 after_lines = diff['after'].splitlines(True)
 199                 if before_lines and not before_lines[-1].endswith(u'\n'):
 200                     before_lines[-1] += u'\n\\ No newline at end of file\n'
 201                 if after_lines and not after_lines[-1].endswith('\n'):
 202                     after_lines[-1] += u'\n\\ No newline at end of file\n'
 203                 differ = difflib.unified_diff(before_lines,
 204                                               after_lines,
 205                                               fromfile=before_header,
 206                                               tofile=after_header,
 207                                               fromfiledate=u'',
 208                                               tofiledate=u'',
 209                                               n=C.DIFF_CONTEXT)
 210                 difflines = list(differ)
 211                 if len(difflines) >= 3 and sys.version_info[:2] == (2, 6):
 212                     # difflib in Python 2.6 adds trailing spaces after
 213                     # filenames in the -- before/++ after headers.
 214                     difflines[0] = difflines[0].replace(u' \n', u'\n')
 215                     difflines[1] = difflines[1].replace(u' \n', u'\n')
 216                     # it also treats empty files differently
 217                     difflines[2] = difflines[2].replace(u'-1,0', u'-0,0').replace(u'+1,0', u'+0,0')
 218                 has_diff = False
 219                 for line in difflines:
 220                     has_diff = True
 221                     if line.startswith(u'+'):
 222                         line = stringc(line, C.COLOR_DIFF_ADD)
 223                     elif line.startswith(u'-'):
 224                         line = stringc(line, C.COLOR_DIFF_REMOVE)
 225                     elif line.startswith(u'@@'):
 226                         line = stringc(line, C.COLOR_DIFF_LINES)
 227                     ret.append(line)
 228                 if has_diff:
 229                     ret.append('\n')
 230             if 'prepared' in diff:
 231                 ret.append(diff['prepared'])
 232         return u''.join(ret)

 234     def _get_item_label(self, result):
 235         ''' retrieves the value to be displayed as a label for an item entry from a result object'''
 236         if result.get('_ansible_no_log', False):
 237             item = "(censored due to no_log)"
 238         else:
 239             item = result.get('_ansible_item_label', result.get('item'))
 240         return item

 246     def _clean_results(self, result, task_name):
 247         ''' removes data from results for display '''
 248 
 249         # mostly controls that debug only outputs what it was meant to
 250         if task_name == 'debug':
 251             if 'msg' in result:
 252                 # msg should be alone
 253                 for key in list(result.keys()):
 254                     if key not in _DEBUG_ALLOWED_KEYS and not key.startswith('_'):
 255                         result.pop(key)
 256             else:
 257                 # 'var' value as field, so eliminate others and what is left should be varname
 258                 for hidme in self._hide_in_debug:
 259                     result.pop(hidme, None)
