  18 from ansible.plugins.callback import CallbackBase
  19 from ansible import constants as C
  20 
  21 
  22 class CallbackModule(CallbackBase):
  23 
  24     '''
  25     This is the default callback interface, which simply prints messages
  26     to stdout when new callback events are received.
  27     '''
  28 
  29     CALLBACK_VERSION = 2.0
  30     CALLBACK_TYPE = 'stdout'
  31     CALLBACK_NAME = 'oneline'
  32 
  33     def _command_generic_msg(self, hostname, result, caption):
  34         stdout = result.get('stdout', '').replace('\n', '\\n').replace('\r', '\\r')
  35         if 'stderr' in result and result['stderr']:
  36             stderr = result.get('stderr', '').replace('\n', '\\n').replace('\r', '\\r')
  37             return "%s | %s | rc=%s | (stdout) %s (stderr) %s" % (hostname, caption, result.get('rc', -1), stdout, stderr)
  38         else:
  39             return "%s | %s | rc=%s | (stdout) %s" % (hostname, caption, result.get('rc', -1), stdout)
  40 
  41     def v2_runner_on_failed(self, result, ignore_errors=False):
  42         if 'exception' in result._result:
  43             if self._display.verbosity < 3:
  44                 # extract just the actual error message from the exception text
  45                 error = result._result['exception'].strip().split('\n')[-1]
  46                 msg = "An exception occurred during task execution. To see the full traceback, use -vvv. The error was: %s" % error
  47             else:
  48                 msg = "An exception occurred during task execution. The full traceback is:\n" + result._result['exception'].replace('\n', '')
  49 
  50             if result._task.action in C.MODULE_NO_JSON and 'module_stderr' not in result._result:
  51                 self._display.display(self._command_generic_msg(result._host.get_name(), result._result, 'FAILED'), color=C.COLOR_ERROR)
  52             else:
  53                 self._display.display(msg, color=C.COLOR_ERROR)
  54 
  55         self._display.display("%s | FAILED! => %s" % (result._host.get_name(), self._dump_results(result._result, indent=0).replace('\n', '')),
  56                               color=C.COLOR_ERROR)
  57 
  58     def v2_runner_on_ok(self, result):
  59 
  60         if result._result.get('changed', False):
  61             color = C.COLOR_CHANGED
  62             state = 'CHANGED'
  63         else:
  64             color = C.COLOR_OK
  65             state = 'SUCCESS'
  66 
  67         if result._task.action in C.MODULE_NO_JSON and 'ansible_job_id' not in result._result:
  68             self._display.display(self._command_generic_msg(result._host.get_name(), result._result, state), color=color)
  69         else:
  70             self._display.display("%s | %s => %s" % (result._host.get_name(), state, self._dump_results(result._result, indent=0).replace('\n', '')),
  71                                   color=color)
  72 
  73     def v2_runner_on_unreachable(self, result):
  74         self._display.display("%s | UNREACHABLE!: %s" % (result._host.get_name(), result._result.get('msg', '')), color=C.COLOR_UNREACHABLE)
  75 
  76     def v2_runner_on_skipped(self, result):
  77         self._display.display("%s | SKIPPED" % (result._host.get_name()), color=C.COLOR_SKIP)
