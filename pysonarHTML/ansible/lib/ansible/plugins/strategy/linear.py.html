  21 DOCUMENTATION = '''
  22     strategy: linear
  23     short_description: Executes tasks in a linear fashion
  24     description:
  25         - Task execution is in lockstep per host batch as defined by C(serial) (default all).
  26           Up to the fork limit of hosts will execute each task at the same time and then
  27           the next series of hosts until the batch is done, before going on to the next task.
  28     version_added: "2.0"
  29     notes:
  30      - This was the default Ansible behaviour before 'strategy plugins' were introduced in 2.0.
  31     author: Ansible Core Team
  32 '''

  51     noop_task = None

  53     def _replace_with_noop(self, target):
  54         if self.noop_task is None:
  55             raise AnsibleAssertionError('strategy.linear.StrategyModule.noop_task is None, need Task()')
  56 
  57         result = []
  58         for el in target:
  59             if isinstance(el, Task):
  60                 result.append(self.noop_task)
  61             elif isinstance(el, Block):
  62                 result.append(self._create_noop_block_from(el, el._parent))
  63         return result
  64 
  65     def _create_noop_block_from(self, original_block, parent):
  66         noop_block = Block(parent_block=parent)
  67         noop_block.block = self._replace_with_noop(original_block.block)
  68         noop_block.always = self._replace_with_noop(original_block.always)
  69         noop_block.rescue = self._replace_with_noop(original_block.rescue)
  70 
  71         return noop_block
  72 
  73     def _prepare_and_create_noop_block_from(self, original_block, parent, iterator):
  74         self.noop_task = Task()
  75         self.noop_task.action = 'meta'
  76         self.noop_task.args['_raw_params'] = 'noop'
  77         self.noop_task.set_loader(iterator._play._loader)
  78 
  79         return self._create_noop_block_from(original_block, parent)

  81     def _get_next_task_lockstep(self, hosts, iterator):
  82         '''
  83         Returns a list of (host, task) tuples, where the task may
  84         be a noop task to keep the iterator in lock step across
  85         all hosts.
  86         '''
  87 
  88         noop_task = Task()
  89         noop_task.action = 'meta'
  90         noop_task.args['_raw_params'] = 'noop'
  91         noop_task.set_loader(iterator._play._loader)
  92 
  93         host_tasks = {}
  94         display.debug("building list of next tasks for hosts")
  95         for host in hosts:
  96             host_tasks[host.name] = iterator.get_next_task_for_host(host, peek=True)
  97         display.debug("done building task lists")
  98 
  99         num_setups = 0
 100         num_tasks = 0
 101         num_rescue = 0
 102         num_always = 0
 103 
 104         display.debug("counting tasks in each state of execution")
 105         host_tasks_to_run = [(host, state_task)
 106                              for host, state_task in iteritems(host_tasks)
 107                              if state_task and state_task[1]]
 108 
 109         if host_tasks_to_run:
 110             try:
 111                 lowest_cur_block = min(
 112                     (iterator.get_active_state(s).cur_block for h, (s, t) in host_tasks_to_run
 113                      if s.run_state != PlayIterator.ITERATING_COMPLETE))
 114             except ValueError:
 115                 lowest_cur_block = None
 116         else:
 117             # empty host_tasks_to_run will just run till the end of the function
 118             # without ever touching lowest_cur_block
 119             lowest_cur_block = None
 120 
 121         for (k, v) in host_tasks_to_run:
 122             (s, t) = v
 123 
 124             s = iterator.get_active_state(s)
 125             if s.cur_block > lowest_cur_block:
 126                 # Not the current block, ignore it
 127                 continue
 128 
 129             if s.run_state == PlayIterator.ITERATING_SETUP:
 130                 num_setups += 1
 131             elif s.run_state == PlayIterator.ITERATING_TASKS:
 132                 num_tasks += 1
 133             elif s.run_state == PlayIterator.ITERATING_RESCUE:
 134                 num_rescue += 1
 135             elif s.run_state == PlayIterator.ITERATING_ALWAYS:
 136                 num_always += 1
 137         display.debug("done counting tasks in each state of execution:\n\tnum_setups: %s\n\tnum_tasks: %s\n\tnum_rescue: %s\n\tnum_always: %s" % (num_setups,
 138                                                                                                                                                   num_tasks,
 139                                                                                                                                                   num_rescue,
 140                                                                                                                                                   num_always))

 142         def _advance_selected_hosts(hosts, cur_block, cur_state):
 143             '''
 144             This helper returns the task for all hosts in the requested
 145             state, otherwise they get a noop dummy task. This also advances
 146             the state of the host, since the given states are determined
 147             while using peek=True.
 148             '''
 149             # we return the values in the order they were originally
 150             # specified in the given hosts array
 151             rvals = []
 152             display.debug("starting to advance hosts")
 153             for host in hosts:
 154                 host_state_task = host_tasks.get(host.name)
 155                 if host_state_task is None:
 156                     continue
 157                 (s, t) = host_state_task
 158                 s = iterator.get_active_state(s)
 159                 if t is None:
 160                     continue
 161                 if s.run_state == cur_state and s.cur_block == cur_block:
 162                     new_t = iterator.get_next_task_for_host(host)
 163                     rvals.append((host, t))
 164                 else:
 165                     rvals.append((host, noop_task))
 166             display.debug("done advancing hosts to next task")
 167             return rvals
 168 
 169         # if any hosts are in ITERATING_SETUP, return the setup task
 170         # while all other hosts get a noop
 171         if num_setups:
 172             display.debug("advancing hosts in ITERATING_SETUP")
 173             return _advance_selected_hosts(hosts, lowest_cur_block, PlayIterator.ITERATING_SETUP)
 174 
 175         # if any hosts are in ITERATING_TASKS, return the next normal
 176         # task for these hosts, while all other hosts get a noop
 177         if num_tasks:
 178             display.debug("advancing hosts in ITERATING_TASKS")
 179             return _advance_selected_hosts(hosts, lowest_cur_block, PlayIterator.ITERATING_TASKS)
 180 
 181         # if any hosts are in ITERATING_RESCUE, return the next rescue
 182         # task for these hosts, while all other hosts get a noop
 183         if num_rescue:
 184             display.debug("advancing hosts in ITERATING_RESCUE")
 185             return _advance_selected_hosts(hosts, lowest_cur_block, PlayIterator.ITERATING_RESCUE)
 186 
 187         # if any hosts are in ITERATING_ALWAYS, return the next always
 188         # task for these hosts, while all other hosts get a noop
 189         if num_always:
 190             display.debug("advancing hosts in ITERATING_ALWAYS")
 191             return _advance_selected_hosts(hosts, lowest_cur_block, PlayIterator.ITERATING_ALWAYS)
 192 
 193         # at this point, everything must be ITERATING_COMPLETE, so we
 194         # return None for all hosts in the list
 195         display.debug("all hosts are done, so returning None's for all hosts")
 196         return [(host, None) for host in hosts]
 197 
 198     def run(self, iterator, play_context):
 199         '''
 200         The linear strategy is simple - get the next task and queue
 201         it for all hosts, then wait for the queue to drain before
 202         moving on to the next task
 203         '''
 204 
 205         # iterate over each task, while there is one left to run
 206         result = self._tqm.RUN_OK
 207         work_to_do = True
 208 
 209         self._set_hosts_cache(iterator._play)
 210 
 211         while work_to_do and not self._tqm._terminated:
 212 
 213             try:
 214                 display.debug("getting the remaining hosts for this loop")
 215                 hosts_left = self.get_hosts_left(iterator)
 216                 display.debug("done getting the remaining hosts for this loop")
 217 
 218                 # queue up this task for each host in the inventory
 219                 callback_sent = False
 220                 work_to_do = False
 221 
 222                 host_results = []
 223                 host_tasks = self._get_next_task_lockstep(hosts_left, iterator)
 224 
 225                 # skip control
 226                 skip_rest = False
 227                 choose_step = True
 228 
 229                 # flag set if task is set to any_errors_fatal
 230                 any_errors_fatal = False
 231 
 232                 results = []
 233                 for (host, task) in host_tasks:
 234                     if not task:
 235                         continue
 236 
 237                     if self._tqm._terminated:
 238                         break
 239 
 240                     run_once = False
 241                     work_to_do = True
 242 
 243                     # check to see if this task should be skipped, due to it being a member of a
 244                     # role which has already run (and whether that role allows duplicate execution)
 245                     if task._role and task._role.has_run(host):
 246                         # If there is no metadata, the default behavior is to not allow duplicates,
 247                         # if there is metadata, check to see if the allow_duplicates flag was set to true
 248                         if task._role._metadata is None or task._role._metadata and not task._role._metadata.allow_duplicates:
 249                             display.debug("'%s' skipped because role has already run" % task)
 250                             continue
 251 
 252                     display.debug("getting variables")
 253                     task_vars = self._variable_manager.get_vars(play=iterator._play, host=host, task=task,
 254                                                                 _hosts=self._hosts_cache, _hosts_all=self._hosts_cache_all)
 255                     self.add_tqm_variables(task_vars, play=iterator._play)
 256                     templar = Templar(loader=self._loader, variables=task_vars)
 257                     display.debug("done getting variables")
 258 
 259                     # test to see if the task across all hosts points to an action plugin which
 260                     # sets BYPASS_HOST_LOOP to true, or if it has run_once enabled. If so, we
 261                     # will only send this task to the first host in the list.
 262 
 263                     task.action = templar.template(task.action)
 264 
 265                     try:
 266                         action = action_loader.get(task.action, class_only=True)
 267                     except KeyError:
 268                         # we don't care here, because the action may simply not have a
 269                         # corresponding action plugin
 270                         action = None
 271 
 272                     if task.action == 'meta':
 273                         # for the linear strategy, we run meta tasks just once and for
 274                         # all hosts currently being iterated over rather than one host
 275                         results.extend(self._execute_meta(task, play_context, iterator, host))
 276                         if task.args.get('_raw_params', None) not in ('noop', 'reset_connection', 'end_host'):
 277                             run_once = True
 278                         if (task.any_errors_fatal or run_once) and not task.ignore_errors:
 279                             any_errors_fatal = True
 280                     else:
 281                         # handle step if needed, skip meta actions as they are used internally
 282                         if self._step and choose_step:
 283                             if self._take_step(task):
 284                                 choose_step = False
 285                             else:
 286                                 skip_rest = True
 287                                 break
 288 
 289                         run_once = templar.template(task.run_once) or action and getattr(action, 'BYPASS_HOST_LOOP', False)
 290 
 291                         if (task.any_errors_fatal or run_once) and not task.ignore_errors:
 292                             any_errors_fatal = True
 293 
 294                         if not callback_sent:
 295                             display.debug("sending task start callback, copying the task so we can template it temporarily")
 296                             saved_name = task.name
 297                             display.debug("done copying, going to template now")
 298                             try:
 299                                 task.name = to_text(templar.template(task.name, fail_on_undefined=False), nonstring='empty')
 300                                 display.debug("done templating")
 301                             except Exception:
 302                                 # just ignore any errors during task name templating,
 303                                 # we don't care if it just shows the raw name
 304                                 display.debug("templating failed for some reason")
 305                             display.debug("here goes the callback...")
 306                             self._tqm.send_callback('v2_playbook_on_task_start', task, is_conditional=False)
 307                             task.name = saved_name
 308                             callback_sent = True
 309                             display.debug("sending task start callback")
 310 
 311                         self._blocked_hosts[host.get_name()] = True
 312                         self._queue_task(host, task, task_vars, play_context)
 313                         del task_vars
 314 
 315                     # if we're bypassing the host loop, break out now
 316                     if run_once:
 317                         break
 318 
 319                     results += self._process_pending_results(iterator, max_passes=max(1, int(len(self._tqm._workers) * 0.1)))
 320 
 321                 # go to next host/task group
 322                 if skip_rest:
 323                     continue
 324 
 325                 display.debug("done queuing things up, now waiting for results queue to drain")
 326                 if self._pending_results > 0:
 327                     results += self._wait_on_pending_results(iterator)
 328 
 329                 host_results.extend(results)
 330 
 331                 self.update_active_connections(results)
 332 
 333                 included_files = IncludedFile.process_include_results(
 334                     host_results,
 335                     iterator=iterator,
 336                     loader=self._loader,
 337                     variable_manager=self._variable_manager
 338                 )
 339 
 340                 include_failure = False
 341                 if len(included_files) > 0:
 342                     display.debug("we have included files to process")
 343 
 344                     display.debug("generating all_blocks data")
 345                     all_blocks = dict((host, []) for host in hosts_left)
 346                     display.debug("done generating all_blocks data")
 347                     for included_file in included_files:
 348                         display.debug("processing included file: %s" % included_file._filename)
 349                         # included hosts get the task list while those excluded get an equal-length
 350                         # list of noop tasks, to make sure that they continue running in lock-step
 351                         try:
 352                             if included_file._is_role:
 353                                 new_ir = self._copy_included_file(included_file)
 354 
 355                                 new_blocks, handler_blocks = new_ir.get_block_list(
 356                                     play=iterator._play,
 357                                     variable_manager=self._variable_manager,
 358                                     loader=self._loader,
 359                                 )
 360                             else:
 361                                 new_blocks = self._load_included_file(included_file, iterator=iterator)
 362 
 363                             display.debug("iterating over new_blocks loaded from include file")
 364                             for new_block in new_blocks:
 365                                 task_vars = self._variable_manager.get_vars(
 366                                     play=iterator._play,
 367                                     task=new_block._parent,
 368                                     _hosts=self._hosts_cache,
 369                                     _hosts_all=self._hosts_cache_all,
 370                                 )
 371                                 display.debug("filtering new block on tags")
 372                                 final_block = new_block.filter_tagged_tasks(task_vars)
 373                                 display.debug("done filtering new block on tags")
 374 
 375                                 noop_block = self._prepare_and_create_noop_block_from(final_block, task._parent, iterator)
 376 
 377                                 for host in hosts_left:
 378                                     if host in included_file._hosts:
 379                                         all_blocks[host].append(final_block)
 380                                     else:
 381                                         all_blocks[host].append(noop_block)
 382                             display.debug("done iterating over new_blocks loaded from include file")
 383 
 384                         except AnsibleError as e:
 385                             for host in included_file._hosts:
 386                                 self._tqm._failed_hosts[host.name] = True
 387                                 iterator.mark_host_failed(host)
 388                             display.error(to_text(e), wrap_text=False)
 389                             include_failure = True
 390                             continue
 391 
 392                     # finally go through all of the hosts and append the
 393                     # accumulated blocks to their list of tasks
 394                     display.debug("extending task lists for all hosts with included blocks")
 395 
 396                     for host in hosts_left:
 397                         iterator.add_tasks(host, all_blocks[host])
 398 
 399                     display.debug("done extending task lists")
 400                     display.debug("done processing included files")
 401 
 402                 display.debug("results queue empty")
 403 
 404                 display.debug("checking for any_errors_fatal")
 405                 failed_hosts = []
 406                 unreachable_hosts = []
 407                 for res in results:
 408                     # execute_meta() does not set 'failed' in the TaskResult
 409                     # so we skip checking it with the meta tasks and look just at the iterator
 410                     if (res.is_failed() or res._task.action == 'meta') and iterator.is_failed(res._host):
 411                         failed_hosts.append(res._host.name)
 412                     elif res.is_unreachable():
 413                         unreachable_hosts.append(res._host.name)
 414 
 415                 # if any_errors_fatal and we had an error, mark all hosts as failed
 416                 if any_errors_fatal and (len(failed_hosts) > 0 or len(unreachable_hosts) > 0):
 417                     dont_fail_states = frozenset([iterator.ITERATING_RESCUE, iterator.ITERATING_ALWAYS])
 418                     for host in hosts_left:
 419                         (s, _) = iterator.get_next_task_for_host(host, peek=True)
 420                         # the state may actually be in a child state, use the get_active_state()
 421                         # method in the iterator to figure out the true active state
 422                         s = iterator.get_active_state(s)
 423                         if s.run_state not in dont_fail_states or \
 424                            s.run_state == iterator.ITERATING_RESCUE and s.fail_state & iterator.FAILED_RESCUE != 0:
 425                             self._tqm._failed_hosts[host.name] = True
 426                             result |= self._tqm.RUN_FAILED_BREAK_PLAY
 427                 display.debug("done checking for any_errors_fatal")
 428 
 429                 display.debug("checking for max_fail_percentage")
 430                 if iterator._play.max_fail_percentage is not None and len(results) > 0:
 431                     percentage = iterator._play.max_fail_percentage / 100.0
 432 
 433                     if (len(self._tqm._failed_hosts) / iterator.batch_size) > percentage:
 434                         for host in hosts_left:
 435                             # don't double-mark hosts, or the iterator will potentially
 436                             # fail them out of the rescue/always states
 437                             if host.name not in failed_hosts:
 438                                 self._tqm._failed_hosts[host.name] = True
 439                                 iterator.mark_host_failed(host)
 440                         self._tqm.send_callback('v2_playbook_on_no_hosts_remaining')
 441                         result |= self._tqm.RUN_FAILED_BREAK_PLAY
 442                     display.debug('(%s failed / %s total )> %s max fail' % (len(self._tqm._failed_hosts), iterator.batch_size, percentage))
 443                 display.debug("done checking for max_fail_percentage")
 444 
 445                 display.debug("checking to see if all hosts have failed and the running result is not ok")
 446                 if result != self._tqm.RUN_OK and len(self._tqm._failed_hosts) >= len(hosts_left):
 447                     display.debug("^ not ok, so returning result now")
 448                     self._tqm.send_callback('v2_playbook_on_no_hosts_remaining')
 449                     return result
 450                 display.debug("done checking to see if all hosts have failed")
 451 
 452             except (IOError, EOFError) as e:
 453                 display.debug("got IOError/EOFError in task loop: %s" % e)
 454                 # most likely an abort, return failed
 455                 return self._tqm.RUN_UNKNOWN_ERROR
 456 
 457         # run the base class run() method, which executes the cleanup function
 458         # and runs any outstanding handlers which have been triggered
 459 
 460         return super(StrategyModule, self).run(iterator, play_context, result)
