  21 DOCUMENTATION = '''
  22     strategy: free
  23     short_description: Executes tasks without waiting for all hosts
  24     description:
  25         - Task execution is as fast as possible per batch as defined by C(serial) (default all).
  26           Ansible will not wait for other hosts to finish the current task before queuing more tasks for other hosts.
  27           All hosts are still attempted for the current task, but it prevents blocking new tasks for hosts that have already finished.
  28         - With the free strategy, unlike the default linear strategy, a host that is slow or stuck on a specific task
  29           won't hold up the rest of the hosts and tasks.
  30     version_added: "2.0"
  31     author: Ansible Core Team
  32 '''

  48 class StrategyModule(StrategyBase):
  49 
  50     # This strategy manages throttling on its own, so we don't want it done in queue_task
  51     ALLOW_BASE_THROTTLING = False
  52 
  53     def _filter_notified_failed_hosts(self, iterator, notified_hosts):
  54 
  55         # If --force-handlers is used we may act on hosts that have failed
  56         return [host for host in notified_hosts if iterator.is_failed(host)]
  57 
  58     def _filter_notified_hosts(self, notified_hosts):
  59         '''
  60         Filter notified hosts accordingly to strategy
  61         '''
  62 
  63         # We act only on hosts that are ready to flush handlers
  64         return [host for host in notified_hosts
  65                 if host in self._flushed_hosts and self._flushed_hosts[host]]
  66 
  67     def __init__(self, tqm):
  68         super(StrategyModule, self).__init__(tqm)
  69         self._host_pinned = False
  70 
  71     def run(self, iterator, play_context):
  72         '''
  73         The "free" strategy is a bit more complex, in that it allows tasks to
  74         be sent to hosts as quickly as they can be processed. This means that
  75         some hosts may finish very quickly if run tasks result in little or no
  76         work being done versus other systems.
  77 
  78         The algorithm used here also tries to be more "fair" when iterating
  79         through hosts by remembering the last host in the list to be given a task
  80         and starting the search from there as opposed to the top of the hosts
  81         list again, which would end up favoring hosts near the beginning of the
  82         list.
  83         '''
  84 
  85         # the last host to be given a task
  86         last_host = 0
  87 
  88         result = self._tqm.RUN_OK
  89 
  90         # start with all workers being counted as being free
  91         workers_free = len(self._workers)
  92 
  93         self._set_hosts_cache(iterator._play)
  94 
  95         work_to_do = True
  96         while work_to_do and not self._tqm._terminated:
  97 
  98             hosts_left = self.get_hosts_left(iterator)
  99 
 100             if len(hosts_left) == 0:
 101                 self._tqm.send_callback('v2_playbook_on_no_hosts_remaining')
 102                 result = False
 103                 break
 104 
 105             work_to_do = False        # assume we have no more work to do
 106             starting_host = last_host  # save current position so we know when we've looped back around and need to break
 107 
 108             # try and find an unblocked host with a task to run
 109             host_results = []
 110             while True:
 111                 host = hosts_left[last_host]
 112                 display.debug("next free host: %s" % host)
 113                 host_name = host.get_name()
 114 
 115                 # peek at the next task for the host, to see if there's
 116                 # anything to do do for this host
 117                 (state, task) = iterator.get_next_task_for_host(host, peek=True)
 118                 display.debug("free host state: %s" % state, host=host_name)
 119                 display.debug("free host task: %s" % task, host=host_name)
 120                 if host_name not in self._tqm._unreachable_hosts and task:
 121 
 122                     # set the flag so the outer loop knows we've still found
 123                     # some work which needs to be done
 124                     work_to_do = True
 125 
 126                     display.debug("this host has work to do", host=host_name)
 127 
 128                     # check to see if this host is blocked (still executing a previous task)
 129                     if (host_name not in self._blocked_hosts or not self._blocked_hosts[host_name]):
 130 
 131                         display.debug("getting variables", host=host_name)
 132                         task_vars = self._variable_manager.get_vars(play=iterator._play, host=host, task=task,
 133                                                                     _hosts=self._hosts_cache,
 134                                                                     _hosts_all=self._hosts_cache_all)
 135                         self.add_tqm_variables(task_vars, play=iterator._play)
 136                         templar = Templar(loader=self._loader, variables=task_vars)
 137                         display.debug("done getting variables", host=host_name)
 138 
 139                         try:
 140                             throttle = int(templar.template(task.throttle))
 141                         except Exception as e:
 142                             raise AnsibleError("Failed to convert the throttle value to an integer.", obj=task._ds, orig_exc=e)
 143 
 144                         if throttle > 0:
 145                             same_tasks = 0
 146                             for worker in self._workers:
 147                                 if worker and worker.is_alive() and worker._task._uuid == task._uuid:
 148                                     same_tasks += 1
 149 
 150                             display.debug("task: %s, same_tasks: %d" % (task.get_name(), same_tasks))
 151                             if same_tasks >= throttle:
 152                                 break
 153 
 154                         # pop the task, mark the host blocked, and queue it
 155                         self._blocked_hosts[host_name] = True
 156                         (state, task) = iterator.get_next_task_for_host(host)
 157 
 158                         try:
 159                             action = action_loader.get(task.action, class_only=True)
 160                         except KeyError:
 161                             # we don't care here, because the action may simply not have a
 162                             # corresponding action plugin
 163                             action = None
 164 
 165                         try:
 166                             task.name = to_text(templar.template(task.name, fail_on_undefined=False), nonstring='empty')
 167                             display.debug("done templating", host=host_name)
 168                         except Exception:
 169                             # just ignore any errors during task name templating,
 170                             # we don't care if it just shows the raw name
 171                             display.debug("templating failed for some reason", host=host_name)
 172 
 173                         run_once = templar.template(task.run_once) or action and getattr(action, 'BYPASS_HOST_LOOP', False)
 174                         if run_once:
 175                             if action and getattr(action, 'BYPASS_HOST_LOOP', False):
 176                                 raise AnsibleError("The '%s' module bypasses the host loop, which is currently not supported in the free strategy "
 177                                                    "and would instead execute for every host in the inventory list." % task.action, obj=task._ds)
 178                             else:
 179                                 display.warning("Using run_once with the free strategy is not currently supported. This task will still be "
 180                                                 "executed for every host in the inventory list.")
 181 
 182                         # check to see if this task should be skipped, due to it being a member of a
 183                         # role which has already run (and whether that role allows duplicate execution)
 184                         if task._role and task._role.has_run(host):
 185                             # If there is no metadata, the default behavior is to not allow duplicates,
 186                             # if there is metadata, check to see if the allow_duplicates flag was set to true
 187                             if task._role._metadata is None or task._role._metadata and not task._role._metadata.allow_duplicates:
 188                                 display.debug("'%s' skipped because role has already run" % task, host=host_name)
 189                                 del self._blocked_hosts[host_name]
 190                                 continue
 191 
 192                         if task.action == 'meta':
 193                             self._execute_meta(task, play_context, iterator, target_host=host)
 194                             self._blocked_hosts[host_name] = False
 195                         else:
 196                             # handle step if needed, skip meta actions as they are used internally
 197                             if not self._step or self._take_step(task, host_name):
 198                                 if task.any_errors_fatal:
 199                                     display.warning("Using any_errors_fatal with the free strategy is not supported, "
 200                                                     "as tasks are executed independently on each host")
 201                                 self._tqm.send_callback('v2_playbook_on_task_start', task, is_conditional=False)
 202                                 self._queue_task(host, task, task_vars, play_context)
 203                                 # each task is counted as a worker being busy
 204                                 workers_free -= 1
 205                                 del task_vars
 206                     else:
 207                         display.debug("%s is blocked, skipping for now" % host_name)
 208 
 209                 # all workers have tasks to do (and the current host isn't done with the play).
 210                 # loop back to starting host and break out
 211                 if self._host_pinned and workers_free == 0 and work_to_do:
 212                     last_host = starting_host
 213                     break
 214 
 215                 # move on to the next host and make sure we
 216                 # haven't gone past the end of our hosts list
 217                 last_host += 1
 218                 if last_host > len(hosts_left) - 1:
 219                     last_host = 0
 220 
 221                 # if we've looped around back to the start, break out
 222                 if last_host == starting_host:
 223                     break
 224 
 225             results = self._process_pending_results(iterator)
 226             host_results.extend(results)
 227 
 228             # each result is counted as a worker being free again
 229             workers_free += len(results)
 230 
 231             self.update_active_connections(results)
 232 
 233             included_files = IncludedFile.process_include_results(
 234                 host_results,
 235                 iterator=iterator,
 236                 loader=self._loader,
 237                 variable_manager=self._variable_manager
 238             )
 239 
 240             if len(included_files) > 0:
 241                 all_blocks = dict((host, []) for host in hosts_left)
 242                 for included_file in included_files:
 243                     display.debug("collecting new blocks for %s" % included_file)
 244                     try:
 245                         if included_file._is_role:
 246                             new_ir = self._copy_included_file(included_file)
 247 
 248                             new_blocks, handler_blocks = new_ir.get_block_list(
 249                                 play=iterator._play,
 250                                 variable_manager=self._variable_manager,
 251                                 loader=self._loader,
 252                             )
 253                         else:
 254                             new_blocks = self._load_included_file(included_file, iterator=iterator)
 255                     except AnsibleError as e:
 256                         for host in included_file._hosts:
 257                             iterator.mark_host_failed(host)
 258                         display.warning(to_text(e))
 259                         continue
 260 
 261                     for new_block in new_blocks:
 262                         task_vars = self._variable_manager.get_vars(play=iterator._play, task=new_block._parent,
 263                                                                     _hosts=self._hosts_cache,
 264                                                                     _hosts_all=self._hosts_cache_all)
 265                         final_block = new_block.filter_tagged_tasks(task_vars)
 266                         for host in hosts_left:
 267                             if host in included_file._hosts:
 268                                 all_blocks[host].append(final_block)
 269                     display.debug("done collecting new blocks for %s" % included_file)
 270 
 271                 display.debug("adding all collected blocks from %d included file(s) to iterator" % len(included_files))
 272                 for host in hosts_left:
 273                     iterator.add_tasks(host, all_blocks[host])
 274                 display.debug("done adding collected blocks to iterator")
 275 
 276             # pause briefly so we don't spin lock
 277             time.sleep(C.DEFAULT_INTERNAL_POLL_INTERVAL)
 278 
 279         # collect all the final results
 280         results = self._wait_on_pending_results(iterator)
 281 
 282         # run the base class run() method, which executes the cleanup function
 283         # and runs any outstanding handlers which have been triggered
 284         return super(StrategyModule, self).run(iterator, play_context, result)
