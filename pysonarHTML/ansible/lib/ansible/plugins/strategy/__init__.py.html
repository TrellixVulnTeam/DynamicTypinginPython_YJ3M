  73 def results_thread_main(strategy):
  74     while True:
  75         try:
  76             result = strategy._final_q.get()
  77             if isinstance(result, StrategySentinel):
  78                 break
  79             else:
  80                 strategy._results_lock.acquire()
  81                 # only handlers have the listen attr, so this must be a handler
  82                 # we split up the results into two queues here to make sure
  83                 # handler and regular result processing don't cross wires
  84                 if 'listen' in result._task_fields:
  85                     strategy._handler_results.append(result)
  86                 else:
  87                     strategy._results.append(result)
  88                 strategy._results_lock.release()
  89         except (IOError, EOFError):
  90             break
  91         except Queue.Empty:
  92             pass

  95 def debug_closure(func):
  96     """Closure to wrap ``StrategyBase._process_pending_results`` and invoke the task debugger"""
  97     @functools.wraps(func)
  98     def inner(self, iterator, one_pass=False, max_passes=None, do_handlers=False):
  99         status_to_stats_map = (
 100             ('is_failed', 'failures'),
 101             ('is_unreachable', 'dark'),
 102             ('is_changed', 'changed'),
 103             ('is_skipped', 'skipped'),
 104         )
 105 
 106         # We don't know the host yet, copy the previous states, for lookup after we process new results
 107         prev_host_states = iterator._host_states.copy()
 108 
 109         results = func(self, iterator, one_pass=one_pass, max_passes=max_passes, do_handlers=do_handlers)
 110         _processed_results = []
 111 
 112         for result in results:
 113             task = result._task
 114             host = result._host
 115             _queued_task_args = self._queued_task_cache.pop((host.name, task._uuid), None)
 116             task_vars = _queued_task_args['task_vars']
 117             play_context = _queued_task_args['play_context']
 118             # Try to grab the previous host state, if it doesn't exist use get_host_state to generate an empty state
 119             try:
 120                 prev_host_state = prev_host_states[host.name]
 121             except KeyError:
 122                 prev_host_state = iterator.get_host_state(host)
 123 
 124             while result.needs_debugger(globally_enabled=self.debugger_active):
 125                 next_action = NextAction()
 126                 dbg = Debugger(task, host, task_vars, play_context, result, next_action)
 127                 dbg.cmdloop()
 128 
 129                 if next_action.result == NextAction.REDO:
 130                     # rollback host state
 131                     self._tqm.clear_failed_hosts()
 132                     iterator._host_states[host.name] = prev_host_state
 133                     for method, what in status_to_stats_map:
 134                         if getattr(result, method)():
 135                             self._tqm._stats.decrement(what, host.name)
 136                     self._tqm._stats.decrement('ok', host.name)
 137 
 138                     # redo
 139                     self._queue_task(host, task, task_vars, play_context)
 140 
 141                     _processed_results.extend(debug_closure(func)(self, iterator, one_pass))
 142                     break
 143                 elif next_action.result == NextAction.CONTINUE:
 144                     _processed_results.append(result)
 145                     break
 146                 elif next_action.result == NextAction.EXIT:
 147                     # Matches KeyboardInterrupt from bin/ansible
 148                     sys.exit(99)
 149             else:
 150                 _processed_results.append(result)
 151 
 152         return _processed_results
 153     return inner

 156 class StrategyBase:
 157 
 158     '''
 159     This is the base class for strategy plugins, which contains some common
 160     code useful to all strategies like running handlers, cleanup actions, etc.
 161     '''
 162 
 163     # by default, strategies should support throttling but we allow individual
 164     # strategies to disable this and either forego supporting it or managing
 165     # the throttling internally (as `free` does)
 166     ALLOW_BASE_THROTTLING = True
 167 
 168     def __init__(self, tqm):
 169         self._tqm = tqm
 170         self._inventory = tqm.get_inventory()
 171         self._workers = tqm._workers
 172         self._variable_manager = tqm.get_variable_manager()
 173         self._loader = tqm.get_loader()
 174         self._final_q = tqm._final_q
 175         self._step = context.CLIARGS.get('step', False)
 176         self._diff = context.CLIARGS.get('diff', False)
 177         self.flush_cache = context.CLIARGS.get('flush_cache', False)
 178 
 179         # the task cache is a dictionary of tuples of (host.name, task._uuid)
 180         # used to find the original task object of in-flight tasks and to store
 181         # the task args/vars and play context info used to queue the task.
 182         self._queued_task_cache = {}
 183 
 184         # Backwards compat: self._display isn't really needed, just import the global display and use that.
 185         self._display = display
 186 
 187         # internal counters
 188         self._pending_results = 0
 189         self._pending_handler_results = 0
 190         self._cur_worker = 0
 191 
 192         # this dictionary is used to keep track of hosts that have
 193         # outstanding tasks still in queue
 194         self._blocked_hosts = dict()
 195 
 196         # this dictionary is used to keep track of hosts that have
 197         # flushed handlers
 198         self._flushed_hosts = dict()
 199 
 200         self._results = deque()
 201         self._handler_results = deque()
 202         self._results_lock = threading.Condition(threading.Lock())
 203 
 204         # create the result processing thread for reading results in the background
 205         self._results_thread = threading.Thread(target=results_thread_main, args=(self,))
 206         self._results_thread.daemon = True
 207         self._results_thread.start()
 208 
 209         # holds the list of active (persistent) connections to be shutdown at
 210         # play completion
 211         self._active_connections = dict()
 212 
 213         # Caches for get_host calls, to avoid calling excessively
 214         # These values should be set at the top of the ``run`` method of each
 215         # strategy plugin. Use ``_set_hosts_cache`` to set these values
 216         self._hosts_cache = []
 217         self._hosts_cache_all = []
 218 
 219         self.debugger_active = C.ENABLE_TASK_DEBUGGER
 220 
 221     def _set_hosts_cache(self, play, refresh=True):
 222         """Responsible for setting _hosts_cache and _hosts_cache_all
 223 
 224         See comment in ``__init__`` for the purpose of these caches
 225         """
 226         if not refresh and all((self._hosts_cache, self._hosts_cache_all)):
 227             return
 228 
 229         if Templar(None).is_template(play.hosts):
 230             _pattern = 'all'
 231         else:
 232             _pattern = play.hosts or 'all'
 233         self._hosts_cache_all = [h.name for h in self._inventory.get_hosts(pattern=_pattern, ignore_restrictions=True)]
 234         self._hosts_cache = [h.name for h in self._inventory.get_hosts(play.hosts, order=play.order)]
 235 
 236     def cleanup(self):
 237         # close active persistent connections
 238         for sock in itervalues(self._active_connections):
 239             try:
 240                 conn = Connection(sock)
 241                 conn.reset()
 242             except ConnectionError as e:
 243                 # most likely socket is already closed
 244                 display.debug("got an error while closing persistent connection: %s" % e)
 245         self._final_q.put(_sentinel)
 246         self._results_thread.join()
 247 
 248     def run(self, iterator, play_context, result=0):
 249         # execute one more pass through the iterator without peeking, to
 250         # make sure that all of the hosts are advanced to their final task.
 251         # This should be safe, as everything should be ITERATING_COMPLETE by
 252         # this point, though the strategy may not advance the hosts itself.
 253 
 254         for host in self._hosts_cache:
 255             if host not in self._tqm._unreachable_hosts:
 256                 try:
 257                     iterator.get_next_task_for_host(self._inventory.hosts[host])
 258                 except KeyError:
 259                     iterator.get_next_task_for_host(self._inventory.get_host(host))
 260 
 261         # save the failed/unreachable hosts, as the run_handlers()
 262         # method will clear that information during its execution
 263         failed_hosts = iterator.get_failed_hosts()
 264         unreachable_hosts = self._tqm._unreachable_hosts.keys()
 265 
 266         display.debug("running handlers")
 267         handler_result = self.run_handlers(iterator, play_context)
 268         if isinstance(handler_result, bool) and not handler_result:
 269             result |= self._tqm.RUN_ERROR
 270         elif not handler_result:
 271             result |= handler_result
 272 
 273         # now update with the hosts (if any) that failed or were
 274         # unreachable during the handler execution phase
 275         failed_hosts = set(failed_hosts).union(iterator.get_failed_hosts())
 276         unreachable_hosts = set(unreachable_hosts).union(self._tqm._unreachable_hosts.keys())
 277 
 278         # return the appropriate code, depending on the status hosts after the run
 279         if not isinstance(result, bool) and result != self._tqm.RUN_OK:
 280             return result
 281         elif len(unreachable_hosts) > 0:
 282             return self._tqm.RUN_UNREACHABLE_HOSTS
 283         elif len(failed_hosts) > 0:
 284             return self._tqm.RUN_FAILED_HOSTS
 285         else:
 286             return self._tqm.RUN_OK
 287 
 288     def get_hosts_remaining(self, play):
 289         self._set_hosts_cache(play, refresh=False)
 290         ignore = set(self._tqm._failed_hosts).union(self._tqm._unreachable_hosts)
 291         return [host for host in self._hosts_cache if host not in ignore]
 292 
 293     def get_failed_hosts(self, play):
 294         self._set_hosts_cache(play, refresh=False)
 295         return [host for host in self._hosts_cache if host in self._tqm._failed_hosts]

 305     def _queue_task(self, host, task, task_vars, play_context):
 306         ''' handles queueing the task up to be sent to a worker '''
 307 
 308         display.debug("entering _queue_task() for %s/%s" % (host.name, task.action))
 309 
 310         # Add a write lock for tasks.
 311         # Maybe this should be added somewhere further up the call stack but
 312         # this is the earliest in the code where we have task (1) extracted
 313         # into its own variable and (2) there's only a single code path
 314         # leading to the module being run.  This is called by three
 315         # functions: __init__.py::_do_handler_run(), linear.py::run(), and
 316         # free.py::run() so we'd have to add to all three to do it there.
 317         # The next common higher level is __init__.py::run() and that has
 318         # tasks inside of play_iterator so we'd have to extract them to do it
 319         # there.
 320 
 321         if task.action not in action_write_locks.action_write_locks:
 322             display.debug('Creating lock for %s' % task.action)
 323             action_write_locks.action_write_locks[task.action] = Lock()
 324 
 325         # create a templar and template things we need later for the queuing process
 326         templar = Templar(loader=self._loader, variables=task_vars)
 327 
 328         try:
 329             throttle = int(templar.template(task.throttle))
 330         except Exception as e:
 331             raise AnsibleError("Failed to convert the throttle value to an integer.", obj=task._ds, orig_exc=e)
 332 
 333         # and then queue the new task
 334         try:
 335             # Determine the "rewind point" of the worker list. This means we start
 336             # iterating over the list of workers until the end of the list is found.
 337             # Normally, that is simply the length of the workers list (as determined
 338             # by the forks or serial setting), however a task/block/play may "throttle"
 339             # that limit down.
 340             rewind_point = len(self._workers)
 341             if throttle > 0 and self.ALLOW_BASE_THROTTLING:
 342                 if task.run_once:
 343                     display.debug("Ignoring 'throttle' as 'run_once' is also set for '%s'" % task.get_name())
 344                 else:
 345                     if throttle <= rewind_point:
 346                         display.debug("task: %s, throttle: %d" % (task.get_name(), throttle))
 347                         rewind_point = throttle
 348 
 349             queued = False
 350             starting_worker = self._cur_worker
 351             while True:
 352                 if self._cur_worker >= rewind_point:
 353                     self._cur_worker = 0
 354 
 355                 worker_prc = self._workers[self._cur_worker]
 356                 if worker_prc is None or not worker_prc.is_alive():
 357                     self._queued_task_cache[(host.name, task._uuid)] = {
 358                         'host': host,
 359                         'task': task,
 360                         'task_vars': task_vars,
 361                         'play_context': play_context
 362                     }
 363 
 364                     worker_prc = WorkerProcess(self._final_q, task_vars, host, task, play_context, self._loader, self._variable_manager, plugin_loader)
 365                     self._workers[self._cur_worker] = worker_prc
 366                     self._tqm.send_callback('v2_runner_on_start', host, task)
 367                     worker_prc.start()
 368                     display.debug("worker is %d (out of %d available)" % (self._cur_worker + 1, len(self._workers)))
 369                     queued = True
 370 
 371                 self._cur_worker += 1
 372 
 373                 if self._cur_worker >= rewind_point:
 374                     self._cur_worker = 0
 375 
 376                 if queued:
 377                     break
 378                 elif self._cur_worker == starting_worker:
 379                     time.sleep(0.0001)
 380 
 381             if isinstance(task, Handler):
 382                 self._pending_handler_results += 1
 383             else:
 384                 self._pending_results += 1
 385         except (EOFError, IOError, AssertionError) as e:
 386             # most likely an abort
 387             display.debug("got an error while queuing: %s" % e)
 388             return
 389         display.debug("exiting _queue_task() for %s/%s" % (host.name, task.action))

 402     def _set_always_delegated_facts(self, result, task):
 403         """Sets host facts for ``delegate_to`` hosts for facts that should
 404         always be delegated
 405 
 406         This operation mutates ``result`` to remove the always delegated facts
 407 
 408         See ``ALWAYS_DELEGATE_FACT_PREFIXES``
 409         """
 410         if task.delegate_to is None:
 411             return
 412 
 413         facts = result['ansible_facts']
 414         always_keys = set()
 415         _add = always_keys.add
 416         for fact_key in facts:
 417             for always_key in ALWAYS_DELEGATE_FACT_PREFIXES:
 418                 if fact_key.startswith(always_key):
 419                     _add(fact_key)
 420         if always_keys:
 421             _pop = facts.pop
 422             always_facts = {
 423                 'ansible_facts': dict((k, _pop(k)) for k in list(facts) if k in always_keys)
 424             }
 425             host_list = self.get_delegated_hosts(result, task)
 426             _set_host_facts = self._variable_manager.set_host_facts
 427             for target_host in host_list:
 428                 _set_host_facts(target_host, always_facts)

 430     @debug_closure
 431     def _process_pending_results(self, iterator, one_pass=False, max_passes=None, do_handlers=False):
 432         '''
 433         Reads results off the final queue and takes appropriate action
 434         based on the result (executing callbacks, updating state, etc.).
 435         '''
 436 
 437         ret_results = []
 438         handler_templar = Templar(self._loader)
 439 
 440         def get_original_host(host_name):
 441             # FIXME: this should not need x2 _inventory
 442             host_name = to_text(host_name)
 443             if host_name in self._inventory.hosts:
 444                 return self._inventory.hosts[host_name]
 445             else:
 446                 return self._inventory.get_host(host_name)
 447 
 448         def search_handler_blocks_by_name(handler_name, handler_blocks):
 449             # iterate in reversed order since last handler loaded with the same name wins
 450             for handler_block in reversed(handler_blocks):
 451                 for handler_task in handler_block.block:
 452                     if handler_task.name:
 453                         if not handler_task.cached_name:
 454                             if handler_templar.is_template(handler_task.name):
 455                                 handler_templar.available_variables = self._variable_manager.get_vars(play=iterator._play,
 456                                                                                                       task=handler_task,
 457                                                                                                       _hosts=self._hosts_cache,
 458                                                                                                       _hosts_all=self._hosts_cache_all)
 459                                 handler_task.name = handler_templar.template(handler_task.name)
 460                             handler_task.cached_name = True
 461 
 462                         try:
 463                             # first we check with the full result of get_name(), which may
 464                             # include the role name (if the handler is from a role). If that
 465                             # is not found, we resort to the simple name field, which doesn't
 466                             # have anything extra added to it.
 467                             candidates = (
 468                                 handler_task.name,
 469                                 handler_task.get_name(include_role_fqcn=False),
 470                                 handler_task.get_name(include_role_fqcn=True),
 471                             )
 472 
 473                             if handler_name in candidates:
 474                                 return handler_task
 475                         except (UndefinedError, AnsibleUndefinedVariable):
 476                             # We skip this handler due to the fact that it may be using
 477                             # a variable in the name that was conditionally included via
 478                             # set_fact or some other method, and we don't want to error
 479                             # out unnecessarily
 480                             continue
 481             return None
 482 
 483         cur_pass = 0
 484         while True:
 485             try:
 486                 self._results_lock.acquire()
 487                 if do_handlers:
 488                     task_result = self._handler_results.popleft()
 489                 else:
 490                     task_result = self._results.popleft()
 491             except IndexError:
 492                 break
 493             finally:
 494                 self._results_lock.release()
 495 
 496             # get the original host and task. We then assign them to the TaskResult for use in callbacks/etc.
 497             original_host = get_original_host(task_result._host)
 498             queue_cache_entry = (original_host.name, task_result._task)
 499             found_task = self._queued_task_cache.get(queue_cache_entry)['task']
 500             original_task = found_task.copy(exclude_parent=True, exclude_tasks=True)
 501             original_task._parent = found_task._parent
 502             original_task.from_attrs(task_result._task_fields)
 503 
 504             task_result._host = original_host
 505             task_result._task = original_task
 506 
 507             # send callbacks for 'non final' results
 508             if '_ansible_retry' in task_result._result:
 509                 self._tqm.send_callback('v2_runner_retry', task_result)
 510                 continue
 511             elif '_ansible_item_result' in task_result._result:
 512                 if task_result.is_failed() or task_result.is_unreachable():
 513                     self._tqm.send_callback('v2_runner_item_on_failed', task_result)
 514                 elif task_result.is_skipped():
 515                     self._tqm.send_callback('v2_runner_item_on_skipped', task_result)
 516                 else:
 517                     if 'diff' in task_result._result:
 518                         if self._diff or getattr(original_task, 'diff', False):
 519                             self._tqm.send_callback('v2_on_file_diff', task_result)
 520                     self._tqm.send_callback('v2_runner_item_on_ok', task_result)
 521                 continue
 522 
 523             # all host status messages contain 2 entries: (msg, task_result)
 524             role_ran = False
 525             if task_result.is_failed():
 526                 role_ran = True
 527                 ignore_errors = original_task.ignore_errors
 528                 if not ignore_errors:
 529                     display.debug("marking %s as failed" % original_host.name)
 530                     if original_task.run_once:
 531                         # if we're using run_once, we have to fail every host here
 532                         for h in self._inventory.get_hosts(iterator._play.hosts):
 533                             if h.name not in self._tqm._unreachable_hosts:
 534                                 state, _ = iterator.get_next_task_for_host(h, peek=True)
 535                                 iterator.mark_host_failed(h)
 536                                 state, new_task = iterator.get_next_task_for_host(h, peek=True)
 537                     else:
 538                         iterator.mark_host_failed(original_host)
 539 
 540                     # grab the current state and if we're iterating on the rescue portion
 541                     # of a block then we save the failed task in a special var for use
 542                     # within the rescue/always
 543                     state, _ = iterator.get_next_task_for_host(original_host, peek=True)
 544 
 545                     if iterator.is_failed(original_host) and state and state.run_state == iterator.ITERATING_COMPLETE:
 546                         self._tqm._failed_hosts[original_host.name] = True
 547 
 548                     if state and iterator.get_active_state(state).run_state == iterator.ITERATING_RESCUE:
 549                         self._tqm._stats.increment('rescued', original_host.name)
 550                         self._variable_manager.set_nonpersistent_facts(
 551                             original_host.name,
 552                             dict(
 553                                 ansible_failed_task=original_task.serialize(),
 554                                 ansible_failed_result=task_result._result,
 555                             ),
 556                         )
 557                     else:
 558                         self._tqm._stats.increment('failures', original_host.name)
 559                 else:
 560                     self._tqm._stats.increment('ok', original_host.name)
 561                     self._tqm._stats.increment('ignored', original_host.name)
 562                     if 'changed' in task_result._result and task_result._result['changed']:
 563                         self._tqm._stats.increment('changed', original_host.name)
 564                 self._tqm.send_callback('v2_runner_on_failed', task_result, ignore_errors=ignore_errors)
 565             elif task_result.is_unreachable():
 566                 ignore_unreachable = original_task.ignore_unreachable
 567                 if not ignore_unreachable:
 568                     self._tqm._unreachable_hosts[original_host.name] = True
 569                     iterator._play._removed_hosts.append(original_host.name)
 570                 else:
 571                     self._tqm._stats.increment('skipped', original_host.name)
 572                     task_result._result['skip_reason'] = 'Host %s is unreachable' % original_host.name
 573                 self._tqm._stats.increment('dark', original_host.name)
 574                 self._tqm.send_callback('v2_runner_on_unreachable', task_result)
 575             elif task_result.is_skipped():
 576                 self._tqm._stats.increment('skipped', original_host.name)
 577                 self._tqm.send_callback('v2_runner_on_skipped', task_result)
 578             else:
 579                 role_ran = True
 580 
 581                 if original_task.loop:
 582                     # this task had a loop, and has more than one result, so
 583                     # loop over all of them instead of a single result
 584                     result_items = task_result._result.get('results', [])
 585                 else:
 586                     result_items = [task_result._result]
 587 
 588                 for result_item in result_items:
 589                     if '_ansible_notify' in result_item:
 590                         if task_result.is_changed():
 591                             # The shared dictionary for notified handlers is a proxy, which
 592                             # does not detect when sub-objects within the proxy are modified.
 593                             # So, per the docs, we reassign the list so the proxy picks up and
 594                             # notifies all other threads
 595                             for handler_name in result_item['_ansible_notify']:
 596                                 found = False
 597                                 # Find the handler using the above helper.  First we look up the
 598                                 # dependency chain of the current task (if it's from a role), otherwise
 599                                 # we just look through the list of handlers in the current play/all
 600                                 # roles and use the first one that matches the notify name
 601                                 target_handler = search_handler_blocks_by_name(handler_name, iterator._play.handlers)
 602                                 if target_handler is not None:
 603                                     found = True
 604                                     if target_handler.notify_host(original_host):
 605                                         self._tqm.send_callback('v2_playbook_on_notify', target_handler, original_host)
 606 
 607                                 for listening_handler_block in iterator._play.handlers:
 608                                     for listening_handler in listening_handler_block.block:
 609                                         listeners = getattr(listening_handler, 'listen', []) or []
 610                                         if not listeners:
 611                                             continue
 612 
 613                                         listeners = listening_handler.get_validated_value(
 614                                             'listen', listening_handler._valid_attrs['listen'], listeners, handler_templar
 615                                         )
 616                                         if handler_name not in listeners:
 617                                             continue
 618                                         else:
 619                                             found = True
 620 
 621                                         if listening_handler.notify_host(original_host):
 622                                             self._tqm.send_callback('v2_playbook_on_notify', listening_handler, original_host)
 623 
 624                                 # and if none were found, then we raise an error
 625                                 if not found:
 626                                     msg = ("The requested handler '%s' was not found in either the main handlers list nor in the listening "
 627                                            "handlers list" % handler_name)
 628                                     if C.ERROR_ON_MISSING_HANDLER:
 629                                         raise AnsibleError(msg)
 630                                     else:
 631                                         display.warning(msg)
 632 
 633                     if 'add_host' in result_item:
 634                         # this task added a new host (add_host module)
 635                         new_host_info = result_item.get('add_host', dict())
 636                         self._add_host(new_host_info, result_item)
 637 
 638                     elif 'add_group' in result_item:
 639                         # this task added a new group (group_by module)
 640                         self._add_group(original_host, result_item)
 641 
 642                     if 'ansible_facts' in result_item:
 643                         # if delegated fact and we are delegating facts, we need to change target host for them
 644                         if original_task.delegate_to is not None and original_task.delegate_facts:
 645                             host_list = self.get_delegated_hosts(result_item, original_task)
 646                         else:
 647                             # Set facts that should always be on the delegated hosts
 648                             self._set_always_delegated_facts(result_item, original_task)
 649 
 650                             host_list = self.get_task_hosts(iterator, original_host, original_task)
 651 
 652                         if original_task.action == 'include_vars':
 653                             for (var_name, var_value) in iteritems(result_item['ansible_facts']):
 654                                 # find the host we're actually referring too here, which may
 655                                 # be a host that is not really in inventory at all
 656                                 for target_host in host_list:
 657                                     self._variable_manager.set_host_variable(target_host, var_name, var_value)
 658                         else:
 659                             cacheable = result_item.pop('_ansible_facts_cacheable', False)
 660                             for target_host in host_list:
 661                                 # so set_fact is a misnomer but 'cacheable = true' was meant to create an 'actual fact'
 662                                 # to avoid issues with precedence and confusion with set_fact normal operation,
 663                                 # we set BOTH fact and nonpersistent_facts (aka hostvar)
 664                                 # when fact is retrieved from cache in subsequent operations it will have the lower precedence,
 665                                 # but for playbook setting it the 'higher' precedence is kept
 666                                 if original_task.action != 'set_fact' or cacheable:
 667                                     self._variable_manager.set_host_facts(target_host, result_item['ansible_facts'].copy())
 668                                 if original_task.action == 'set_fact':
 669                                     self._variable_manager.set_nonpersistent_facts(target_host, result_item['ansible_facts'].copy())
 670 
 671                     if 'ansible_stats' in result_item and 'data' in result_item['ansible_stats'] and result_item['ansible_stats']['data']:
 672 
 673                         if 'per_host' not in result_item['ansible_stats'] or result_item['ansible_stats']['per_host']:
 674                             host_list = self.get_task_hosts(iterator, original_host, original_task)
 675                         else:
 676                             host_list = [None]
 677 
 678                         data = result_item['ansible_stats']['data']
 679                         aggregate = 'aggregate' in result_item['ansible_stats'] and result_item['ansible_stats']['aggregate']
 680                         for myhost in host_list:
 681                             for k in data.keys():
 682                                 if aggregate:
 683                                     self._tqm._stats.update_custom_stats(k, data[k], myhost)
 684                                 else:
 685                                     self._tqm._stats.set_custom_stats(k, data[k], myhost)
 686 
 687                 if 'diff' in task_result._result:
 688                     if self._diff or getattr(original_task, 'diff', False):
 689                         self._tqm.send_callback('v2_on_file_diff', task_result)
 690 
 691                 if not isinstance(original_task, TaskInclude):
 692                     self._tqm._stats.increment('ok', original_host.name)
 693                     if 'changed' in task_result._result and task_result._result['changed']:
 694                         self._tqm._stats.increment('changed', original_host.name)
 695 
 696                 # finally, send the ok for this task
 697                 self._tqm.send_callback('v2_runner_on_ok', task_result)
 698 
 699             # register final results
 700             if original_task.register:
 701                 host_list = self.get_task_hosts(iterator, original_host, original_task)
 702 
 703                 clean_copy = strip_internal_keys(module_response_deepcopy(task_result._result))
 704                 if 'invocation' in clean_copy:
 705                     del clean_copy['invocation']
 706 
 707                 for target_host in host_list:
 708                     self._variable_manager.set_nonpersistent_facts(target_host, {original_task.register: clean_copy})
 709 
 710             if do_handlers:
 711                 self._pending_handler_results -= 1
 712             else:
 713                 self._pending_results -= 1
 714             if original_host.name in self._blocked_hosts:
 715                 del self._blocked_hosts[original_host.name]
 716 
 717             # If this is a role task, mark the parent role as being run (if
 718             # the task was ok or failed, but not skipped or unreachable)
 719             if original_task._role is not None and role_ran:  # TODO:  and original_task.action != 'include_role':?
 720                 # lookup the role in the ROLE_CACHE to make sure we're dealing
 721                 # with the correct object and mark it as executed
 722                 for (entry, role_obj) in iteritems(iterator._play.ROLE_CACHE[original_task._role.get_name()]):
 723                     if role_obj._uuid == original_task._role._uuid:
 724                         role_obj._had_task_run[original_host.name] = True
 725 
 726             ret_results.append(task_result)
 727 
 728             if one_pass or max_passes is not None and (cur_pass + 1) >= max_passes:
 729                 break
 730 
 731             cur_pass += 1
 732 
 733         return ret_results

 735     def _wait_on_handler_results(self, iterator, handler, notified_hosts):
 736         '''
 737         Wait for the handler tasks to complete, using a short sleep
 738         between checks to ensure we don't spin lock
 739         '''
 740 
 741         ret_results = []
 742         handler_results = 0
 743 
 744         display.debug("waiting for handler results...")
 745         while (self._pending_handler_results > 0 and
 746                handler_results < len(notified_hosts) and
 747                not self._tqm._terminated):
 748 
 749             if self._tqm.has_dead_workers():
 750                 raise AnsibleError("A worker was found in a dead state")
 751 
 752             results = self._process_pending_results(iterator, do_handlers=True)
 753             ret_results.extend(results)
 754             handler_results += len([
 755                 r._host for r in results if r._host in notified_hosts and
 756                 r.task_name == handler.name])
 757             if self._pending_handler_results > 0:
 758                 time.sleep(C.DEFAULT_INTERNAL_POLL_INTERVAL)
 759 
 760         display.debug("no more pending handlers, returning what we have")
 761 
 762         return ret_results

 765         '''
 766         Wait for the shared counter to drop to zero, using a short sleep
 767         between checks to ensure we don't spin lock
 768         '''
 769 
 770         ret_results = []
 771 
 772         display.debug("waiting for pending results...")
 773         while self._pending_results > 0 and not self._tqm._terminated:
 774 
 775             if self._tqm.has_dead_workers():

 777 
 778             results = self._process_pending_results(iterator)

 780             if self._pending_results > 0:
 781                 time.sleep(C.DEFAULT_INTERNAL_POLL_INTERVAL)
 782 
 783         display.debug("no more pending results, returning what we have")
 784 
 785         return ret_results

 787     def _add_host(self, host_info, result_item):
 788         '''
 789         Helper function to add a new host to inventory based on a task result.
 790         '''
 791 
 792         changed = False
 793 
 794         if host_info:
 795             host_name = host_info.get('host_name')
 796 
 797             # Check if host in inventory, add if not
 798             if host_name not in self._inventory.hosts:
 799                 self._inventory.add_host(host_name, 'all')
 800                 self._hosts_cache_all.append(host_name)
 801                 changed = True
 802             new_host = self._inventory.hosts.get(host_name)
 803 
 804             # Set/update the vars for this host
 805             new_host_vars = new_host.get_vars()
 806             new_host_combined_vars = combine_vars(new_host_vars, host_info.get('host_vars', dict()))
 807             if new_host_vars != new_host_combined_vars:
 808                 new_host.vars = new_host_combined_vars
 809                 changed = True
 810 
 811             new_groups = host_info.get('groups', [])
 812             for group_name in new_groups:
 813                 if group_name not in self._inventory.groups:
 814                     group_name = self._inventory.add_group(group_name)
 815                     changed = True
 816                 new_group = self._inventory.groups[group_name]
 817                 if new_group.add_host(self._inventory.hosts[host_name]):
 818                     changed = True
 819 
 820             # reconcile inventory, ensures inventory rules are followed
 821             if changed:
 822                 self._inventory.reconcile_inventory()
 823 
 824             result_item['changed'] = changed

 826     def _add_group(self, host, result_item):
 827         '''
 828         Helper function to add a group (if it does not exist), and to assign the
 829         specified host to that group.
 830         '''
 831 
 832         changed = False
 833 
 834         # the host here is from the executor side, which means it was a
 835         # serialized/cloned copy and we'll need to look up the proper
 836         # host object from the master inventory
 837         real_host = self._inventory.hosts.get(host.name)
 838         if real_host is None:
 839             if host.name == self._inventory.localhost.name:
 840                 real_host = self._inventory.localhost
 841             else:
 842                 raise AnsibleError('%s cannot be matched in inventory' % host.name)
 843         group_name = result_item.get('add_group')
 844         parent_group_names = result_item.get('parent_groups', [])
 845 
 846         if group_name not in self._inventory.groups:
 847             group_name = self._inventory.add_group(group_name)
 848 
 849         for name in parent_group_names:
 850             if name not in self._inventory.groups:
 851                 # create the new group and add it to inventory
 852                 self._inventory.add_group(name)
 853                 changed = True
 854 
 855         group = self._inventory.groups[group_name]
 856         for parent_group_name in parent_group_names:
 857             parent_group = self._inventory.groups[parent_group_name]
 858             new = parent_group.add_child_group(group)
 859             if new and not changed:
 860                 changed = True
 861 
 862         if real_host not in group.get_hosts():
 863             changed = group.add_host(real_host)
 864 
 865         if group not in real_host.get_groups():
 866             changed = real_host.add_group(group)
 867 
 868         if changed:
 869             self._inventory.reconcile_inventory()
 870 
 871         result_item['changed'] = changed

 887     def _load_included_file(self, included_file, iterator, is_handler=False):
 888         '''
 889         Loads an included YAML file of tasks, applying the optional set of variables.
 890         '''
 891 
 892         display.debug("loading included file: %s" % included_file._filename)
 893         try:
 894             data = self._loader.load_from_file(included_file._filename)
 895             if data is None:
 896                 return []
 897             elif not isinstance(data, list):
 898                 raise AnsibleError("included task files must contain a list of tasks")
 899 
 900             ti_copy = self._copy_included_file(included_file)
 901             # pop tags out of the include args, if they were specified there, and assign
 902             # them to the include. If the include already had tags specified, we raise an
 903             # error so that users know not to specify them both ways
 904             tags = included_file._task.vars.pop('tags', [])
 905             if isinstance(tags, string_types):
 906                 tags = tags.split(',')
 907             if len(tags) > 0:
 908                 if len(included_file._task.tags) > 0:
 909                     raise AnsibleParserError("Include tasks should not specify tags in more than one way (both via args and directly on the task). "
 910                                              "Mixing tag specify styles is prohibited for whole import hierarchy, not only for single import statement",
 911                                              obj=included_file._task._ds)
 912                 display.deprecated("You should not specify tags in the include parameters. All tags should be specified using the task-level option",
 913                                    version='2.12', collection_name='ansible.builtin')
 914                 included_file._task.tags = tags
 915 
 916             block_list = load_list_of_blocks(
 917                 data,
 918                 play=iterator._play,
 919                 parent_block=ti_copy.build_parent_block(),
 920                 role=included_file._task._role,
 921                 use_handlers=is_handler,
 922                 loader=self._loader,
 923                 variable_manager=self._variable_manager,
 924             )
 925 
 926             # since we skip incrementing the stats when the task result is
 927             # first processed, we do so now for each host in the list
 928             for host in included_file._hosts:
 929                 self._tqm._stats.increment('ok', host.name)
 930 
 931         except AnsibleError as e:
 932             if isinstance(e, AnsibleFileNotFound):
 933                 reason = "Could not find or access '%s' on the Ansible Controller." % to_text(e.file_name)
 934             else:
 935                 reason = to_text(e)
 936 
 937             # mark all of the hosts including this file as failed, send callbacks,
 938             # and increment the stats for this host
 939             for host in included_file._hosts:
 940                 tr = TaskResult(host=host, task=included_file._task, return_data=dict(failed=True, reason=reason))
 941                 iterator.mark_host_failed(host)
 942                 self._tqm._failed_hosts[host.name] = True
 943                 self._tqm._stats.increment('failures', host.name)
 944                 self._tqm.send_callback('v2_runner_on_failed', tr)
 945             return []
 946 
 947         # finally, send the callback and return the list of blocks loaded
 948         self._tqm.send_callback('v2_playbook_on_include', included_file)
 949         display.debug("done processing included file")
 950         return block_list

 952     def run_handlers(self, iterator, play_context):
 953         '''
 954         Runs handlers on those hosts which have been notified.
 955         '''
 956 
 957         result = self._tqm.RUN_OK
 958 
 959         for handler_block in iterator._play.handlers:
 960             # FIXME: handlers need to support the rescue/always portions of blocks too,
 961             #        but this may take some work in the iterator and gets tricky when
 962             #        we consider the ability of meta tasks to flush handlers
 963             for handler in handler_block.block:
 964                 if handler.notified_hosts:
 965                     result = self._do_handler_run(handler, handler.get_name(), iterator=iterator, play_context=play_context)
 966                     if not result:
 967                         break
 968         return result

 970     def _do_handler_run(self, handler, handler_name, iterator, play_context, notified_hosts=None):
 971 
 972         # FIXME: need to use iterator.get_failed_hosts() instead?
 973         # if not len(self.get_hosts_remaining(iterator._play)):
 974         #     self._tqm.send_callback('v2_playbook_on_no_hosts_remaining')
 975         #     result = False
 976         #     break
 977         if notified_hosts is None:
 978             notified_hosts = handler.notified_hosts[:]
 979 
 980         # strategy plugins that filter hosts need access to the iterator to identify failed hosts
 981         failed_hosts = self._filter_notified_failed_hosts(iterator, notified_hosts)
 982         notified_hosts = self._filter_notified_hosts(notified_hosts)
 983         notified_hosts += failed_hosts
 984 
 985         if len(notified_hosts) > 0:
 986             saved_name = handler.name
 987             handler.name = handler_name
 988             self._tqm.send_callback('v2_playbook_on_handler_task_start', handler)
 989             handler.name = saved_name
 990 
 991         bypass_host_loop = False
 992         try:
 993             action = plugin_loader.action_loader.get(handler.action, class_only=True)
 994             if getattr(action, 'BYPASS_HOST_LOOP', False):
 995                 bypass_host_loop = True
 996         except KeyError:
 997             # we don't care here, because the action may simply not have a
 998             # corresponding action plugin
 999             pass
1000 
1001         host_results = []
1002         for host in notified_hosts:
1003             if not iterator.is_failed(host) or iterator._play.force_handlers:
1004                 task_vars = self._variable_manager.get_vars(play=iterator._play, host=host, task=handler,
1005                                                             _hosts=self._hosts_cache, _hosts_all=self._hosts_cache_all)
1006                 self.add_tqm_variables(task_vars, play=iterator._play)
1007                 templar = Templar(loader=self._loader, variables=task_vars)
1008                 if not handler.cached_name:
1009                     handler.name = templar.template(handler.name)
1010                     handler.cached_name = True
1011 
1012                 self._queue_task(host, handler, task_vars, play_context)
1013 
1014                 if templar.template(handler.run_once) or bypass_host_loop:
1015                     break
1016 
1017         # collect the results from the handler run
1018         host_results = self._wait_on_handler_results(iterator, handler, notified_hosts)
1019 
1020         included_files = IncludedFile.process_include_results(
1021             host_results,
1022             iterator=iterator,
1023             loader=self._loader,
1024             variable_manager=self._variable_manager
1025         )
1026 
1027         result = True
1028         if len(included_files) > 0:
1029             for included_file in included_files:
1030                 try:
1031                     new_blocks = self._load_included_file(included_file, iterator=iterator, is_handler=True)
1032                     # for every task in each block brought in by the include, add the list
1033                     # of hosts which included the file to the notified_handlers dict
1034                     for block in new_blocks:
1035                         iterator._play.handlers.append(block)
1036                         for task in block.block:
1037                             task_name = task.get_name()
1038                             display.debug("adding task '%s' included in handler '%s'" % (task_name, handler_name))
1039                             task.notified_hosts = included_file._hosts[:]
1040                             result = self._do_handler_run(
1041                                 handler=task,
1042                                 handler_name=task_name,
1043                                 iterator=iterator,
1044                                 play_context=play_context,
1045                                 notified_hosts=included_file._hosts[:],
1046                             )
1047                             if not result:
1048                                 break
1049                 except AnsibleError as e:
1050                     for host in included_file._hosts:
1051                         iterator.mark_host_failed(host)
1052                         self._tqm._failed_hosts[host.name] = True
1053                     display.warning(to_text(e))
1054                     continue
1055 
1056         # remove hosts from notification list
1057         handler.notified_hosts = [
1058             h for h in handler.notified_hosts
1059             if h not in notified_hosts]
1060         display.debug("done running handlers, result is: %s" % result)
1061         return result

1101     def _execute_meta(self, task, play_context, iterator, target_host):
1102
