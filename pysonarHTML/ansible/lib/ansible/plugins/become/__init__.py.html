  18 def _gen_id(length=32):
  19     ''' return random string used to identify the current privilege escalation '''
  20     return ''.join(choice(ascii_lowercase) for x in range(length))

  23 class BecomeBase(AnsiblePlugin):
  24 
  25     name = None
  26 
  27     # messages for detecting prompted password issues
  28     fail = tuple()
  29     missing = tuple()
  30 
  31     # many connection plugins cannot provide tty, set to True if your become
  32     # plugin requires a tty, i.e su
  33     require_tty = False
  34 
  35     # prompt to match
  36     prompt = ''
  37 
  38     def __init__(self):
  39         super(BecomeBase, self).__init__()
  40         self._id = ''
  41         self.success = ''
  42 
  43     def get_option(self, option, hostvars=None, playcontext=None):
  44         """ Overrides the base get_option to provide a fallback to playcontext vars in case a 3rd party plugin did not
  45         implement the base become options required in Ansible. """
  46         # TODO: add deprecation warning for ValueError in devel that removes the playcontext fallback
  47         try:
  48             return super(BecomeBase, self).get_option(option, hostvars=hostvars)
  49         except KeyError:
  50             pc_fallback = ['become_user', 'become_pass', 'become_flags', 'become_exe']
  51             if option not in pc_fallback:
  52                 raise
  53 
  54             return getattr(playcontext, option, None)
  55 
  56     def expect_prompt(self):
  57         """This function assists connection plugins in determining if they need to wait for
  58         a prompt. Both a prompt and a password are required.
  59         """
  60         return self.prompt and self.get_option('become_pass')
  61 
  62     def _build_success_command(self, cmd, shell, noexe=False):
  63         if not all((cmd, shell, self.success)):
  64             return cmd
  65 
  66         try:
  67             cmd = shlex_quote('%s %s %s %s' % (shell.ECHO, self.success, shell.COMMAND_SEP, cmd))
  68         except AttributeError:
  69             # TODO: This should probably become some more robust functionlity used to detect incompat
  70             raise AnsibleError('The %s shell family is incompatible with the %s become plugin' % (shell.SHELL_FAMILY, self.name))
  71         exe = getattr(shell, 'executable', None)
  72         if exe and not noexe:
  73             cmd = '%s -c %s' % (exe, cmd)
  74         return cmd
  75 
  76     @abstractmethod
  77     def build_become_command(self, cmd, shell):
  78         self._id = _gen_id()
  79         self.success = 'BECOME-SUCCESS-%s' % self._id
  80 
  81     def check_success(self, b_output):
  82         b_success = to_bytes(self.success)
  83         return any(b_success in l.rstrip() for l in b_output.splitlines(True))
  84 
  85     def check_password_prompt(self, b_output):
  86         ''' checks if the expected password prompt exists in b_output '''
  87         if self.prompt:
  88             b_prompt = to_bytes(self.prompt).strip()
  89             return any(l.strip().startswith(b_prompt) for l in b_output.splitlines())
  90         return False
  91 
  92     def _check_password_error(self, b_out, msg):
  93         ''' returns True/False if domain specific i18n version of msg is found in b_out '''
  94         b_fail = to_bytes(dgettext(self.name, msg))
  95         return b_fail and b_fail in b_out
  96 
  97     def check_incorrect_password(self, b_output):
  98         for errstring in self.fail:
  99             if self._check_password_error(b_output, errstring):
 100                 return True
 101         return False
 102 
 103     def check_missing_password(self, b_output):
 104         for errstring in self.missing:
 105             if self._check_password_error(b_output, errstring):
 106                 return True
 107         return False
