  17 class ActionModule(ActionBase):
  18 
  19     def _get_module_args(self, fact_module, task_vars):
  20 
  21         mod_args = self._task.args.copy()
  22 
  23         # deal with 'setup specific arguments'
  24         if fact_module != 'setup':
  25 
  26             # network facts modules must support gather_subset
  27             if self._connection._load_name not in ('network_cli', 'httpapi', 'netconf'):
  28                 subset = mod_args.pop('gather_subset', None)
  29                 if subset not in ('all', ['all']):
  30                     self._display.warning('Ignoring subset(%s) for %s' % (subset, fact_module))
  31 
  32             timeout = mod_args.pop('gather_timeout', None)
  33             if timeout is not None:
  34                 self._display.warning('Ignoring timeout(%s) for %s' % (timeout, fact_module))
  35 
  36             fact_filter = mod_args.pop('filter', None)
  37             if fact_filter is not None:
  38                 self._display.warning('Ignoring filter(%s) for %s' % (fact_filter, fact_module))
  39 
  40         # Strip out keys with ``None`` values, effectively mimicking ``omit`` behavior
  41         # This ensures we don't pass a ``None`` value as an argument expecting a specific type
  42         mod_args = dict((k, v) for k, v in mod_args.items() if v is not None)
  43 
  44         # handle module defaults
  45         mod_args = get_action_args_with_defaults(fact_module, mod_args, self._task.module_defaults, self._templar, self._task._ansible_internal_redirect_list)
  46 
  47         return mod_args
  48 
  49     def _combine_task_result(self, result, task_result):
  50         filtered_res = {
  51             'ansible_facts': task_result.get('ansible_facts', {}),
  52             'warnings': task_result.get('warnings', []),
  53             'deprecations': task_result.get('deprecations', []),
  54         }
  55 
  56         # on conflict the last plugin processed wins, but try to do deep merge and append to lists.
  57         return merge_hash(result, filtered_res, list_merge='append_rp')
  58 
  59     def run(self, tmp=None, task_vars=None):
  60 
  61         self._supports_check_mode = True
  62 
  63         result = super(ActionModule, self).run(tmp, task_vars)
  64         result['ansible_facts'] = {}
  65 
  66         modules = C.config.get_config_value('FACTS_MODULES', variables=task_vars)
  67         parallel = task_vars.pop('ansible_facts_parallel', self._task.args.pop('parallel', None))
  68         if 'smart' in modules:
  69             connection_map = C.config.get_config_value('CONNECTION_FACTS_MODULES', variables=task_vars)
  70             network_os = self._task.args.get('network_os', task_vars.get('ansible_network_os', task_vars.get('ansible_facts', {}).get('network_os')))
  71             modules.extend([connection_map.get(network_os or self._connection._load_name, 'setup')])
  72             modules.pop(modules.index('smart'))
  73 
  74         failed = {}
  75         skipped = {}
  76 
  77         if parallel is None and len(modules) >= 1:
  78             parallel = True
  79         else:
  80             parallel = boolean(parallel)
  81 
  82         if parallel:
  83             # serially execute each module
  84             for fact_module in modules:
  85                 # just one module, no need for fancy async
  86                 mod_args = self._get_module_args(fact_module, task_vars)
  87                 res = self._execute_module(module_name=fact_module, module_args=mod_args, task_vars=task_vars, wrap_async=False)
  88                 if res.get('failed', False):
  89                     failed[fact_module] = res
  90                 elif res.get('skipped', False):
  91                     skipped[fact_module] = res
  92                 else:
  93                     result = self._combine_task_result(result, res)
  94 
  95             self._remove_tmp_path(self._connection._shell.tmpdir)
  96         else:
  97             # do it async
  98             jobs = {}
  99             for fact_module in modules:
 100                 mod_args = self._get_module_args(fact_module, task_vars)
 101                 self._display.vvvv("Running %s" % fact_module)
 102                 jobs[fact_module] = (self._execute_module(module_name=fact_module, module_args=mod_args, task_vars=task_vars, wrap_async=True))
 103 
 104             while jobs:
 105                 for module in jobs:
 106                     poll_args = {'jid': jobs[module]['ansible_job_id'], '_async_dir': os.path.dirname(jobs[module]['results_file'])}
 107                     res = self._execute_module(module_name='async_status', module_args=poll_args, task_vars=task_vars, wrap_async=False)
 108                     if res.get('finished', 0) == 1:
 109                         if res.get('failed', False):
 110                             failed[module] = res
 111                         elif res.get('skipped', False):
 112                             skipped[module] = res
 113                         else:
 114                             result = self._combine_task_result(result, res)
 115                         del jobs[module]
 116                         break
 117                     else:
 118                         time.sleep(0.1)
 119                 else:
 120                     time.sleep(0.5)
 121 
 122         if skipped:
 123             result['msg'] = "The following modules were skipped: %s\n" % (', '.join(skipped.keys()))
 124             result['skipped_modules'] = skipped
 125             if len(skipped) == len(modules):
 126                 result['skipped'] = True
 127 
 128         if failed:
 129             result['failed'] = True
 130             result['msg'] = "The following modules failed to execute: %s\n" % (', '.join(failed.keys()))
 131             result['failed_modules'] = failed
 132 
 133         # tell executor facts were gathered
 134         result['ansible_facts']['_ansible_facts_gathered'] = True
 135 
 136         # hack to keep --verbose from showing all the setup module result
 137         result['_ansible_verbose_override'] = True
 138 
 139         return result
