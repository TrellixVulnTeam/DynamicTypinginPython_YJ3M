  16 class ActionModule(ActionBase):
  17 
  18     TRANSFERS_FILES = False
  19 
  20     VALID_FILE_EXTENSIONS = ['yaml', 'yml', 'json']
  21     VALID_DIR_ARGUMENTS = ['dir', 'depth', 'files_matching', 'ignore_files', 'extensions', 'ignore_unknown_extensions']
  22     VALID_FILE_ARGUMENTS = ['file', '_raw_params']
  23     VALID_ALL = ['name']

  25     def _set_dir_defaults(self):
  26         if not self.depth:
  27             self.depth = 0
  28 
  29         if self.files_matching:
  30             self.matcher = re.compile(r'{0}'.format(self.files_matching))
  31         else:
  32             self.matcher = None
  33 
  34         if not self.ignore_files:
  35             self.ignore_files = list()
  36 
  37         if isinstance(self.ignore_files, string_types):
  38             self.ignore_files = self.ignore_files.split()
  39 
  40         elif isinstance(self.ignore_files, dict):
  41             return {
  42                 'failed': True,
  43                 'message': '{0} must be a list'.format(self.ignore_files)
  44             }

  46     def _set_args(self):
  47         """ Set instance variables based on the arguments that were passed """
  48 
  49         self.return_results_as_name = self._task.args.get('name', None)
  50         self.source_dir = self._task.args.get('dir', None)
  51         self.source_file = self._task.args.get('file', None)
  52         if not self.source_dir and not self.source_file:
  53             self.source_file = self._task.args.get('_raw_params')
  54             if self.source_file:
  55                 self.source_file = self.source_file.rstrip('\n')
  56 
  57         self.depth = self._task.args.get('depth', None)
  58         self.files_matching = self._task.args.get('files_matching', None)
  59         self.ignore_unknown_extensions = self._task.args.get('ignore_unknown_extensions', False)
  60         self.ignore_files = self._task.args.get('ignore_files', None)
  61         self.valid_extensions = self._task.args.get('extensions', self.VALID_FILE_EXTENSIONS)
  62 
  63         # convert/validate extensions list
  64         if isinstance(self.valid_extensions, string_types):
  65             self.valid_extensions = list(self.valid_extensions)
  66         if not isinstance(self.valid_extensions, list):
  67             raise AnsibleError('Invalid type for "extensions" option, it must be a list')

  69     def run(self, tmp=None, task_vars=None):
  70         """ Load yml files recursively from a directory.
  71         """
  72         del tmp  # tmp no longer has any effect
  73 
  74         if task_vars is None:
  75             task_vars = dict()
  76 
  77         self.show_content = True
  78         self.included_files = []
  79 
  80         # Validate arguments
  81         dirs = 0
  82         files = 0
  83         for arg in self._task.args:
  84             if arg in self.VALID_DIR_ARGUMENTS:
  85                 dirs += 1
  86             elif arg in self.VALID_FILE_ARGUMENTS:
  87                 files += 1
  88             elif arg in self.VALID_ALL:
  89                 pass
  90             else:
  91                 raise AnsibleError('{0} is not a valid option in include_vars'.format(to_native(arg)))
  92 
  93         if dirs and files:
  94             raise AnsibleError("You are mixing file only and dir only arguments, these are incompatible")
  95 
  96         # set internal vars from args
  97         self._set_args()
  98 
  99         results = dict()
 100         if self.source_dir:
 101             self._set_dir_defaults()
 102             self._set_root_dir()
 103             if not path.exists(self.source_dir):
 104                 failed = True
 105                 err_msg = ('{0} directory does not exist'.format(to_native(self.source_dir)))
 106             elif not path.isdir(self.source_dir):
 107                 failed = True
 108                 err_msg = ('{0} is not a directory'.format(to_native(self.source_dir)))
 109             else:
 110                 for root_dir, filenames in self._traverse_dir_depth():
 111                     failed, err_msg, updated_results = (self._load_files_in_dir(root_dir, filenames))
 112                     if failed:
 113                         break
 114                     results.update(updated_results)
 115         else:
 116             try:
 117                 self.source_file = self._find_needle('vars', self.source_file)
 118                 failed, err_msg, updated_results = (
 119                     self._load_files(self.source_file)
 120                 )
 121                 if not failed:
 122                     results.update(updated_results)
 123 
 124             except AnsibleError as e:
 125                 failed = True
 126                 err_msg = to_native(e)
 127 
 128         if self.return_results_as_name:
 129             scope = dict()
 130             scope[self.return_results_as_name] = results
 131             results = scope
 132 
 133         result = super(ActionModule, self).run(task_vars=task_vars)
 134 
 135         if failed:
 136             result['failed'] = failed
 137             result['message'] = err_msg
 138 
 139         result['ansible_included_var_files'] = self.included_files
 140         result['ansible_facts'] = results
 141         result['_ansible_no_log'] = not self.show_content
 142 
 143         return result

 145     def _set_root_dir(self):
 146         if self._task._role:
 147             if self.source_dir.split('/')[0] == 'vars':
 148                 path_to_use = (
 149                     path.join(self._task._role._role_path, self.source_dir)
 150                 )
 151                 if path.exists(path_to_use):
 152                     self.source_dir = path_to_use
 153             else:
 154                 path_to_use = (
 155                     path.join(
 156                         self._task._role._role_path, 'vars', self.source_dir
 157                     )
 158                 )
 159                 self.source_dir = path_to_use
 160         else:
 161             if hasattr(self._task._ds, '_data_source'):
 162                 current_dir = (
 163                     "/".join(self._task._ds._data_source.split('/')[:-1])
 164                 )
 165                 self.source_dir = path.join(current_dir, self.source_dir)

 167     def _traverse_dir_depth(self):
 168         """ Recursively iterate over a directory and sort the files in
 169             alphabetical order. Do not iterate pass the set depth.
 170             The default depth is unlimited.
 171         """
 172         current_depth = 0
 173         sorted_walk = list(walk(self.source_dir))
 174         sorted_walk.sort(key=lambda x: x[0])
 175         for current_root, current_dir, current_files in sorted_walk:
 176             current_depth += 1
 177             if current_depth <= self.depth or self.depth == 0:
 178                 current_files.sort()
 179                 yield (current_root, current_files)
 180             else:
 181                 break

 183     def _ignore_file(self, filename):
 184         """ Return True if a file matches the list of ignore_files.
 185         Args:
 186             filename (str): The filename that is being matched against.
 187 
 188         Returns:
 189             Boolean
 190         """
 191         for file_type in self.ignore_files:
 192             try:
 193                 if re.search(r'{0}$'.format(file_type), filename):
 194                     return True
 195             except Exception:
 196                 err_msg = 'Invalid regular expression: {0}'.format(file_type)
 197                 raise AnsibleError(err_msg)
 198         return False

 200     def _is_valid_file_ext(self, source_file):
 201         """ Verify if source file has a valid extension
 202         Args:
 203             source_file (str): The full path of source file or source file.
 204         Returns:
 205             Bool
 206         """
 207         file_ext = path.splitext(source_file)
 208         return bool(len(file_ext) > 1 and file_ext[-1][1:] in self.valid_extensions)

 210     def _load_files(self, filename, validate_extensions=False):
 211         """ Loads a file and converts the output into a valid Python dict.
 212         Args:
 213             filename (str): The source file.
 214 
 215         Returns:
 216             Tuple (bool, str, dict)
 217         """
 218         results = dict()
 219         failed = False
 220         err_msg = ''
 221         if validate_extensions and not self._is_valid_file_ext(filename):
 222             failed = True
 223             err_msg = ('{0} does not have a valid extension: {1}'.format(to_native(filename), ', '.join(self.valid_extensions)))
 224         else:
 225             b_data, show_content = self._loader._get_file_contents(filename)
 226             data = to_text(b_data, errors='surrogate_or_strict')
 227 
 228             self.show_content = show_content
 229             data = self._loader.load(data, file_name=filename, show_content=show_content)
 230             if not data:
 231                 data = dict()
 232             if not isinstance(data, dict):
 233                 failed = True
 234                 err_msg = ('{0} must be stored as a dictionary/hash'.format(to_native(filename)))
 235             else:
 236                 self.included_files.append(filename)
 237                 results.update(data)
 238 
 239         return failed, err_msg, results

 241     def _load_files_in_dir(self, root_dir, var_files):
 242         """ Load the found yml files and update/overwrite the dictionary.
 243         Args:
 244             root_dir (str): The base directory of the list of files that is being passed.
 245             var_files: (list): List of files to iterate over and load into a dictionary.
 246 
 247         Returns:
 248             Tuple (bool, str, dict)
 249         """
 250         results = dict()
 251         failed = False
 252         err_msg = ''
 253         for filename in var_files:
 254             stop_iter = False
 255             # Never include main.yml from a role, as that is the default included by the role
 256             if self._task._role:
 257                 if path.join(self._task._role._role_path, filename) == path.join(root_dir, 'vars', 'main.yml'):
 258                     stop_iter = True
 259                     continue
 260 
 261             filepath = path.join(root_dir, filename)
 262             if self.files_matching:
 263                 if not self.matcher.search(filename):
 264                     stop_iter = True
 265 
 266             if not stop_iter and not failed:
 267                 if self.ignore_unknown_extensions:
 268                     if path.exists(filepath) and not self._ignore_file(filename) and self._is_valid_file_ext(filename):
 269                         failed, err_msg, loaded_data = self._load_files(filepath, validate_extensions=True)
 270                         if not failed:
 271                             results.update(loaded_data)
 272                 else:
 273                     if path.exists(filepath) and not self._ignore_file(filename):
 274                         failed, err_msg, loaded_data = self._load_files(filepath, validate_extensions=True)
 275                         if not failed:
 276                             results.update(loaded_data)
 277 
 278         return failed, err_msg, results
