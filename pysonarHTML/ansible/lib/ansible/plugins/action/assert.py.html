  20 from ansible.errors import AnsibleError
  21 from ansible.playbook.conditional import Conditional
  22 from ansible.plugins.action import ActionBase
  23 from ansible.module_utils.six import string_types
  24 from ansible.module_utils.parsing.convert_bool import boolean
  25 
  26 
  27 class ActionModule(ActionBase):
  28     ''' Fail with custom message '''
  29 
  30     TRANSFERS_FILES = False
  31     _VALID_ARGS = frozenset(('fail_msg', 'msg', 'quiet', 'success_msg', 'that'))
  32 
  33     def run(self, tmp=None, task_vars=None):
  34         if task_vars is None:
  35             task_vars = dict()
  36 
  37         result = super(ActionModule, self).run(tmp, task_vars)
  38         del tmp  # tmp no longer has any effect
  39 
  40         if 'that' not in self._task.args:
  41             raise AnsibleError('conditional required in "that" string')
  42 
  43         fail_msg = None
  44         success_msg = None
  45 
  46         fail_msg = self._task.args.get('fail_msg', self._task.args.get('msg'))
  47         if fail_msg is None:
  48             fail_msg = 'Assertion failed'
  49         elif isinstance(fail_msg, list):
  50             if not all(isinstance(x, string_types) for x in fail_msg):
  51                 raise AnsibleError('Type of one of the elements in fail_msg or msg list is not string type')
  52         elif not isinstance(fail_msg, (string_types, list)):
  53             raise AnsibleError('Incorrect type for fail_msg or msg, expected a string or list and got %s' % type(fail_msg))
  54 
  55         success_msg = self._task.args.get('success_msg')
  56         if success_msg is None:
  57             success_msg = 'All assertions passed'
  58         elif isinstance(success_msg, list):
  59             if not all(isinstance(x, string_types) for x in success_msg):
  60                 raise AnsibleError('Type of one of the elements in success_msg list is not string type')
  61         elif not isinstance(success_msg, (string_types, list)):
  62             raise AnsibleError('Incorrect type for success_msg, expected a string or list and got %s' % type(success_msg))
  63 
  64         quiet = boolean(self._task.args.get('quiet', False), strict=False)
  65 
  66         # make sure the 'that' items are a list
  67         thats = self._task.args['that']
  68         if not isinstance(thats, list):
  69             thats = [thats]
  70 
  71         # Now we iterate over the that items, temporarily assigning them
  72         # to the task's when value so we can evaluate the conditional using
  73         # the built in evaluate function. The when has already been evaluated
  74         # by this point, and is not used again, so we don't care about mangling
  75         # that value now
  76         cond = Conditional(loader=self._loader)
  77         if not quiet:
  78             result['_ansible_verbose_always'] = True
  79 
  80         for that in thats:
  81             cond.when = [that]
  82             test_result = cond.evaluate_conditional(templar=self._templar, all_vars=task_vars)
  83             if not test_result:
  84                 result['failed'] = True
  85                 result['evaluated_to'] = test_result
  86                 result['assertion'] = that
  87 
  88                 result['msg'] = fail_msg
  89 
  90                 return result
  91 
  92         result['changed'] = False
  93         result['msg'] = success_msg
  94         return result
