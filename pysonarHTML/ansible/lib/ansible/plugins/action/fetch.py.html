  35 class ActionModule(ActionBase):
  36 
  37     def run(self, tmp=None, task_vars=None):
  38         ''' handler for fetch operations '''
  39         if task_vars is None:
  40             task_vars = dict()
  41 
  42         result = super(ActionModule, self).run(tmp, task_vars)
  43         del tmp  # tmp no longer has any effect
  44 
  45         try:
  46             if self._play_context.check_mode:
  47                 raise AnsibleActionSkip('check mode not (yet) supported for this module')
  48 
  49             source = self._task.args.get('src', None)
  50             original_dest = dest = self._task.args.get('dest', None)
  51             flat = boolean(self._task.args.get('flat'), strict=False)
  52             fail_on_missing = boolean(self._task.args.get('fail_on_missing', True), strict=False)
  53             validate_checksum = boolean(self._task.args.get('validate_checksum', True), strict=False)
  54 
  55             msg = ''
  56             # validate source and dest are strings FIXME: use basic.py and module specs
  57             if not isinstance(source, string_types):
  58                 msg = "Invalid type supplied for source option, it must be a string"
  59 
  60             if not isinstance(dest, string_types):
  61                 msg = "Invalid type supplied for dest option, it must be a string"
  62 
  63             if source is None or dest is None:
  64                 msg = "src and dest are required"
  65 
  66             if msg:
  67                 raise AnsibleActionFail(msg)
  68 
  69             source = self._connection._shell.join_path(source)
  70             source = self._remote_expand_user(source)
  71 
  72             remote_checksum = None
  73             if not self._connection.become:
  74                 # calculate checksum for the remote file, don't bother if using become as slurp will be used
  75                 # Force remote_checksum to follow symlinks because fetch always follows symlinks
  76                 remote_checksum = self._remote_checksum(source, all_vars=task_vars, follow=True)
  77 
  78             # use slurp if permissions are lacking or privilege escalation is needed
  79             remote_data = None
  80             if remote_checksum in ('1', '2', None):
  81                 slurpres = self._execute_module(module_name='slurp', module_args=dict(src=source), task_vars=task_vars)
  82                 if slurpres.get('failed'):
  83                     if not fail_on_missing and (slurpres.get('msg').startswith('file not found') or remote_checksum == '1'):
  84                         result['msg'] = "the remote file does not exist, not transferring, ignored"
  85                         result['file'] = source
  86                         result['changed'] = False
  87                     else:
  88                         result.update(slurpres)
  89                     return result
  90                 else:
  91                     if slurpres['encoding'] == 'base64':
  92                         remote_data = base64.b64decode(slurpres['content'])
  93                     if remote_data is not None:
  94                         remote_checksum = checksum_s(remote_data)
  95 
  96             # calculate the destination name
  97             if os.path.sep not in self._connection._shell.join_path('a', ''):
  98                 source = self._connection._shell._unquote(source)
  99                 source_local = source.replace('\\', '/')
 100             else:
 101                 source_local = source
 102 
 103             # ensure we only use file name, avoid relative paths
 104             if not is_subpath(dest, original_dest):
 105                 # TODO: ? dest = os.path.expanduser(dest.replace(('../','')))
 106                 raise AnsibleActionFail("Detected directory traversal, expected to be contained in '%s' but got '%s'" % (original_dest, dest))
 107 
 108             if flat:
 109                 if os.path.isdir(to_bytes(dest, errors='surrogate_or_strict')) and not dest.endswith(os.sep):
 110                     raise AnsibleActionFail("dest is an existing directory, use a trailing slash if you want to fetch src into that directory")
 111                 if dest.endswith(os.sep):
 112                     # if the path ends with "/", we'll use the source filename as the
 113                     # destination filename
 114                     base = os.path.basename(source_local)
 115                     dest = os.path.join(dest, base)
 116                 if not dest.startswith("/"):
 117                     # if dest does not start with "/", we'll assume a relative path
 118                     dest = self._loader.path_dwim(dest)
 119             else:
 120                 # files are saved in dest dir, with a subdir for each host, then the filename
 121                 if 'inventory_hostname' in task_vars:
 122                     target_name = task_vars['inventory_hostname']
 123                 else:
 124                     target_name = self._play_context.remote_addr
 125                 dest = "%s/%s/%s" % (self._loader.path_dwim(dest), target_name, source_local)
 126 
 127             if remote_checksum in ('0', '1', '2', '3', '4', '5'):
 128                 result['changed'] = False
 129                 result['file'] = source
 130                 if remote_checksum == '0':
 131                     result['msg'] = "unable to calculate the checksum of the remote file"
 132                 elif remote_checksum == '1':
 133                     result['msg'] = "the remote file does not exist"
 134                 elif remote_checksum == '2':
 135                     result['msg'] = "no read permission on remote file"
 136                 elif remote_checksum == '3':
 137                     result['msg'] = "remote file is a directory, fetch cannot work on directories"
 138                 elif remote_checksum == '4':
 139                     result['msg'] = "python isn't present on the system.  Unable to compute checksum"
 140                 elif remote_checksum == '5':
 141                     result['msg'] = "stdlib json was not found on the remote machine. Only the raw module can work without those installed"
 142                 # Historically, these don't fail because you may want to transfer
 143                 # a log file that possibly MAY exist but keep going to fetch other
 144                 # log files. Today, this is better achieved by adding
 145                 # ignore_errors or failed_when to the task.  Control the behaviour
 146                 # via fail_when_missing
 147                 if fail_on_missing:
 148                     result['failed'] = True
 149                     del result['changed']
 150                 else:
 151                     result['msg'] += ", not transferring, ignored"
 152                 return result
 153 
 154             dest = os.path.normpath(dest)
 155 
 156             # calculate checksum for the local file
 157             local_checksum = checksum(dest)
 158 
 159             if remote_checksum != local_checksum:
 160                 # create the containing directories, if needed
 161                 makedirs_safe(os.path.dirname(dest))
 162 
 163                 # fetch the file and check for changes
 164                 if remote_data is None:
 165                     self._connection.fetch_file(source, dest)
 166                 else:
 167                     try:
 168                         f = open(to_bytes(dest, errors='surrogate_or_strict'), 'wb')
 169                         f.write(remote_data)
 170                         f.close()
 171                     except (IOError, OSError) as e:
 172                         raise AnsibleActionFail("Failed to fetch the file: %s" % e)
 173                 new_checksum = secure_hash(dest)
 174                 # For backwards compatibility. We'll return None on FIPS enabled systems
 175                 try:
 176                     new_md5 = md5(dest)
 177                 except ValueError:
 178                     new_md5 = None
 179 
 180                 if validate_checksum and new_checksum != remote_checksum:
 181                     result.update(dict(failed=True, md5sum=new_md5,
 182                                        msg="checksum mismatch", file=source, dest=dest, remote_md5sum=None,
 183                                        checksum=new_checksum, remote_checksum=remote_checksum))
 184                 else:
 185                     result.update({'changed': True, 'md5sum': new_md5, 'dest': dest,
 186                                    'remote_md5sum': None, 'checksum': new_checksum,
 187                                    'remote_checksum': remote_checksum})
 188             else:
 189                 # For backwards compatibility. We'll return None on FIPS enabled systems
 190                 try:
 191                     local_md5 = md5(dest)
 192                 except ValueError:
 193                     local_md5 = None
 194                 result.update(dict(changed=False, md5sum=local_md5, file=source, dest=dest, checksum=local_checksum))
 195 
 196         finally:
 197             self._remove_tmp_path(self._connection._shell.tmpdir)
 198 
 199         return result
