  20 from ansible.plugins.action import ActionBase
  21 from ansible.module_utils.six import string_types
  22 
  23 
  24 class ActionModule(ActionBase):
  25     ''' Create inventory groups based on variables '''
  26 
  27     # We need to be able to modify the inventory
  28     TRANSFERS_FILES = False
  29     _VALID_ARGS = frozenset(('key', 'parents'))
  30 
  31     def run(self, tmp=None, task_vars=None):
  32         if task_vars is None:
  33             task_vars = dict()
  34 
  35         result = super(ActionModule, self).run(tmp, task_vars)
  36         del tmp  # tmp no longer has any effect
  37 
  38         if 'key' not in self._task.args:
  39             result['failed'] = True
  40             result['msg'] = "the 'key' param is required when using group_by"
  41             return result
  42 
  43         group_name = self._task.args.get('key')
  44         parent_groups = self._task.args.get('parents', ['all'])
  45         if isinstance(parent_groups, string_types):
  46             parent_groups = [parent_groups]
  47 
  48         result['changed'] = False
  49         result['add_group'] = group_name.replace(' ', '-')
  50         result['parent_groups'] = [name.replace(' ', '-') for name in parent_groups]
  51         return result
