  96     def _check_delay(self, key, default):
  97         """Ensure that the value is positive or zero"""
  98         value = int(self._task.args.get(key, self._task.args.get(key + '_sec', default)))
  99         if value < 0:
 100             value = 0
 101         return value

 103     def _get_value_from_facts(self, variable_name, distribution, default_value):
 104         """Get dist+version specific args first, then distribution, then family, lastly use default"""
 105         attr = getattr(self, variable_name)
 106         value = attr.get(
 107             distribution['name'] + distribution['version'],
 108             attr.get(
 109                 distribution['name'],
 110                 attr.get(
 111                     distribution['family'],
 112                     getattr(self, default_value))))
 113         return value

 115     def get_shutdown_command_args(self, distribution):
 116         args = self._get_value_from_facts('SHUTDOWN_COMMAND_ARGS', distribution, 'DEFAULT_SHUTDOWN_COMMAND_ARGS')
 117         # Convert seconds to minutes. If less that 60, set it to 0.
 118         delay_min = self.pre_reboot_delay // 60
 119         reboot_message = self._task.args.get('msg', self.DEFAULT_REBOOT_MESSAGE)
 120         return args.format(delay_sec=self.pre_reboot_delay, delay_min=delay_min, message=reboot_message)

 122     def get_distribution(self, task_vars):
 123         distribution = {}
 124         display.debug('{action}: running setup module to get distribution'.format(action=self._task.action))
 125         module_output = self._execute_module(
 126             task_vars=task_vars,
 127             module_name='setup',
 128             module_args={'gather_subset': 'min'})
 129         try:
 130             if module_output.get('failed', False):
 131                 raise AnsibleError('Failed to determine system distribution. {0}, {1}'.format(
 132                     to_native(module_output['module_stdout']).strip(),
 133                     to_native(module_output['module_stderr']).strip()))
 134             distribution['name'] = module_output['ansible_facts']['ansible_distribution'].lower()
 135             distribution['version'] = to_text(module_output['ansible_facts']['ansible_distribution_version'].split('.')[0])
 136             distribution['family'] = to_text(module_output['ansible_facts']['ansible_os_family'].lower())
 137             display.debug("{action}: distribution: {dist}".format(action=self._task.action, dist=distribution))
 138             return distribution
 139         except KeyError as ke:
 140             raise AnsibleError('Failed to get distribution information. Missing "{0}" in output.'.format(ke.args[0]))

 142     def get_shutdown_command(self, task_vars, distribution):
 143         shutdown_bin = self._get_value_from_facts('SHUTDOWN_COMMANDS', distribution, 'DEFAULT_SHUTDOWN_COMMAND')
 144         default_search_paths = ['/sbin', '/usr/sbin', '/usr/local/sbin']
 145         search_paths = self._task.args.get('search_paths', default_search_paths)
 146 
 147         # FIXME: switch all this to user arg spec validation methods when they are available
 148         # Convert bare strings to a list
 149         if is_string(search_paths):
 150             search_paths = [search_paths]
 151 
 152         # Error if we didn't get a list
 153         err_msg = "'search_paths' must be a string or flat list of strings, got {0}"
 154         try:
 155             incorrect_type = any(not is_string(x) for x in search_paths)
 156             if not isinstance(search_paths, list) or incorrect_type:
 157                 raise TypeError
 158         except TypeError:
 159             raise AnsibleError(err_msg.format(search_paths))
 160 
 161         display.debug('{action}: running find module looking in {paths} to get path for "{command}"'.format(
 162             action=self._task.action,
 163             command=shutdown_bin,
 164             paths=search_paths))
 165         find_result = self._execute_module(
 166             task_vars=task_vars,
 167             module_name='find',
 168             module_args={
 169                 'paths': search_paths,
 170                 'patterns': [shutdown_bin],
 171                 'file_type': 'any'
 172             }
 173         )
 174 
 175         full_path = [x['path'] for x in find_result['files']]
 176         if not full_path:
 177             raise AnsibleError('Unable to find command "{0}" in search paths: {1}'.format(shutdown_bin, search_paths))
 178         self._shutdown_command = full_path[0]
 179         return self._shutdown_command

 181     def deprecated_args(self):
 182         for arg, version in self.DEPRECATED_ARGS.items():
 183             if self._task.args.get(arg) is not None:
 184                 display.warning("Since Ansible {version}, {arg} is no longer a valid option for {action}".format(
 185                     version=version,
 186                     arg=arg,
 187                     action=self._task.action))

 189     def get_system_boot_time(self, distribution):
 190         boot_time_command = self._get_value_from_facts('BOOT_TIME_COMMANDS', distribution, 'DEFAULT_BOOT_TIME_COMMAND')
 191         if self._task.args.get('boot_time_command'):
 192             boot_time_command = self._task.args.get('boot_time_command')
 193 
 194             try:
 195                 check_type_str(boot_time_command, allow_conversion=False)
 196             except TypeError as e:
 197                 raise AnsibleError("Invalid value given for 'boot_time_command': %s." % to_native(e))
 198 
 199         display.debug("{action}: getting boot time with command: '{command}'".format(action=self._task.action, command=boot_time_command))
 200         command_result = self._low_level_execute_command(boot_time_command, sudoable=self.DEFAULT_SUDOABLE)
 201 
 202         if command_result['rc'] != 0:
 203             stdout = command_result['stdout']
 204             stderr = command_result['stderr']
 205             raise AnsibleError("{action}: failed to get host boot time info, rc: {rc}, stdout: {out}, stderr: {err}".format(
 206                                action=self._task.action,
 207                                rc=command_result['rc'],
 208                                out=to_native(stdout),
 209                                err=to_native(stderr)))
 210         display.debug("{action}: last boot time: {boot}".format(action=self._task.action, boot=command_result['stdout'].strip()))
 211         return command_result['stdout'].strip()

 213     def check_boot_time(self, distribution, previous_boot_time):
 214         display.vvv("{action}: attempting to get system boot time".format(action=self._task.action))
 215         connect_timeout = self._task.args.get('connect_timeout', self._task.args.get('connect_timeout_sec', self.DEFAULT_CONNECT_TIMEOUT))
 216 
 217         # override connection timeout from defaults to custom value
 218         if connect_timeout:
 219             try:
 220                 display.debug("{action}: setting connect_timeout to {value}".format(action=self._task.action, value=connect_timeout))
 221                 self._connection.set_option("connection_timeout", connect_timeout)
 222                 self._connection.reset()
 223             except AttributeError:
 224                 display.warning("Connection plugin does not allow the connection timeout to be overridden")
 225 
 226         # try and get boot time
 227         try:
 228             current_boot_time = self.get_system_boot_time(distribution)
 229         except Exception as e:
 230             raise e
 231 
 232         # FreeBSD returns an empty string immediately before reboot so adding a length
 233         # check to prevent prematurely assuming system has rebooted
 234         if len(current_boot_time) == 0 or current_boot_time == previous_boot_time:
 235             raise ValueError("boot time has not changed")

 237     def run_test_command(self, distribution, **kwargs):
 238         test_command = self._task.args.get('test_command', self._get_value_from_facts('TEST_COMMANDS', distribution, 'DEFAULT_TEST_COMMAND'))
 239         display.vvv("{action}: attempting post-reboot test command".format(action=self._task.action))
 240         display.debug("{action}: attempting post-reboot test command '{command}'".format(action=self._task.action, command=test_command))
 241         try:
 242             command_result = self._low_level_execute_command(test_command, sudoable=self.DEFAULT_SUDOABLE)
 243         except Exception:
 244             # may need to reset the connection in case another reboot occurred
 245             # which has invalidated our connection
 246             try:
 247                 self._connection.reset()
 248             except AttributeError:
 249                 pass
 250             raise
 251 
 252         if command_result['rc'] != 0:
 253             msg = 'Test command failed: {err} {out}'.format(
 254                 err=to_native(command_result['stderr']),
 255                 out=to_native(command_result['stdout']))
 256             raise RuntimeError(msg)
 257 
 258         display.vvv("{action}: system successfully rebooted".format(action=self._task.action))

 260     def do_until_success_or_timeout(self, action, reboot_timeout, action_desc, distribution, action_kwargs=None):
 261         max_end_time = datetime.utcnow() + timedelta(seconds=reboot_timeout)
 262         if action_kwargs is None:
 263             action_kwargs = {}
 264 
 265         fail_count = 0
 266         max_fail_sleep = 12
 267 
 268         while datetime.utcnow() < max_end_time:
 269             try:
 270                 action(distribution=distribution, **action_kwargs)
 271                 if action_desc:
 272                     display.debug('{action}: {desc} success'.format(action=self._task.action, desc=action_desc))
 273                 return
 274             except Exception as e:
 275                 if isinstance(e, AnsibleConnectionFailure):
 276                     try:
 277                         self._connection.reset()
 278                     except AnsibleConnectionFailure:
 279                         pass
 280                 # Use exponential backoff with a max timout, plus a little bit of randomness
 281                 random_int = random.randint(0, 1000) / 1000
 282                 fail_sleep = 2 ** fail_count + random_int
 283                 if fail_sleep > max_fail_sleep:
 284 
 285                     fail_sleep = max_fail_sleep + random_int
 286                 if action_desc:
 287                     try:
 288                         error = to_text(e).splitlines()[-1]
 289                     except IndexError as e:
 290                         error = to_text(e)
 291                     display.debug("{action}: {desc} fail '{err}', retrying in {sleep:.4} seconds...".format(
 292                         action=self._task.action,
 293                         desc=action_desc,
 294                         err=error,
 295                         sleep=fail_sleep))
 296                 fail_count += 1
 297                 time.sleep(fail_sleep)
 298 
 299         raise TimedOutException('Timed out waiting for {desc} (timeout={timeout})'.format(desc=action_desc, timeout=reboot_timeout))

 301     def perform_reboot(self, task_vars, distribution):
 302         result = {}
 303         reboot_result = {}
 304         shutdown_command = self.get_shutdown_command(task_vars, distribution)
 305         shutdown_command_args = self.get_shutdown_command_args(distribution)
 306         reboot_command = '{0} {1}'.format(shutdown_command, shutdown_command_args)
 307 
 308         try:
 309             display.vvv("{action}: rebooting server...".format(action=self._task.action))
 310             display.debug("{action}: rebooting server with command '{command}'".format(action=self._task.action, command=reboot_command))
 311             reboot_result = self._low_level_execute_command(reboot_command, sudoable=self.DEFAULT_SUDOABLE)
 312         except AnsibleConnectionFailure as e:
 313             # If the connection is closed too quickly due to the system being shutdown, carry on
 314             display.debug('{action}: AnsibleConnectionFailure caught and handled: {error}'.format(action=self._task.action, error=to_text(e)))
 315             reboot_result['rc'] = 0
 316 
 317         result['start'] = datetime.utcnow()
 318 
 319         if reboot_result['rc'] != 0:
 320             result['failed'] = True
 321             result['rebooted'] = False
 322             result['msg'] = "Reboot command failed. Error was {stdout}, {stderr}".format(
 323                 stdout=to_native(reboot_result['stdout'].strip()),
 324                 stderr=to_native(reboot_result['stderr'].strip()))
 325             return result
 326 
 327         result['failed'] = False
 328         return result

 330     def validate_reboot(self, distribution, original_connection_timeout=None, action_kwargs=None):
 331         display.vvv('{action}: validating reboot'.format(action=self._task.action))
 332         result = {}
 333 
 334         try:
 335             # keep on checking system boot_time with short connection responses
 336             reboot_timeout = int(self._task.args.get('reboot_timeout', self._task.args.get('reboot_timeout_sec', self.DEFAULT_REBOOT_TIMEOUT)))
 337 
 338             self.do_until_success_or_timeout(
 339                 action=self.check_boot_time,
 340                 action_desc="last boot time check",
 341                 reboot_timeout=reboot_timeout,
 342                 distribution=distribution,
 343                 action_kwargs=action_kwargs)
 344 
 345             # Get the connect_timeout set on the connection to compare to the original
 346             try:
 347                 connect_timeout = self._connection.get_option('connection_timeout')
 348             except KeyError:
 349                 pass
 350             else:
 351                 if original_connection_timeout != connect_timeout:
 352                     try:
 353                         display.debug("{action}: setting connect_timeout back to original value of {value}".format(
 354                             action=self._task.action,
 355                             value=original_connection_timeout))
 356                         self._connection.set_option("connection_timeout", original_connection_timeout)
 357                         self._connection.reset()
 358                     except (AnsibleError, AttributeError) as e:
 359                         # reset the connection to clear the custom connection timeout
 360                         display.debug("{action}: failed to reset connection_timeout back to default: {error}".format(action=self._task.action,
 361                                                                                                                      error=to_text(e)))
 362 
 363             # finally run test command to ensure everything is working
 364             # FUTURE: add a stability check (system must remain up for N seconds) to deal with self-multi-reboot updates
 365             self.do_until_success_or_timeout(
 366                 action=self.run_test_command,
 367                 action_desc="post-reboot test command",
 368                 reboot_timeout=reboot_timeout,
 369                 distribution=distribution,
 370                 action_kwargs=action_kwargs)
 371 
 372             result['rebooted'] = True
 373             result['changed'] = True
 374 
 375         except TimedOutException as toex:
 376             result['failed'] = True
 377             result['rebooted'] = True
 378             result['msg'] = to_text(toex)
 379             return result
 380 
 381         return result

 383     def run(self, tmp=None, task_vars=None):
 384         self._supports_check_mode = True
 385         self._supports_async = True
 386 
 387         # If running with local connection, fail so we don't reboot ourself
 388         if self._connection.transport == 'local':
 389             msg = 'Running {0} with local connection would reboot the control node.'.format(self._task.action)
 390             return {'changed': False, 'elapsed': 0, 'rebooted': False, 'failed': True, 'msg': msg}
 391 
 392         if self._play_context.check_mode:
 393             return {'changed': True, 'elapsed': 0, 'rebooted': True}
 394 
 395         if task_vars is None:
 396             task_vars = {}
 397 
 398         self.deprecated_args()
 399 
 400         result = super(ActionModule, self).run(tmp, task_vars)
 401 
 402         if result.get('skipped', False) or result.get('failed', False):
 403             return result
 404 
 405         distribution = self.get_distribution(task_vars)
 406 
 407         # Get current boot time
 408         try:
 409             previous_boot_time = self.get_system_boot_time(distribution)
 410         except Exception as e:
 411             result['failed'] = True
 412             result['reboot'] = False
 413             result['msg'] = to_text(e)
 414             return result
 415 
 416         # Get the original connection_timeout option var so it can be reset after
 417         original_connection_timeout = None
 418         try:
 419             original_connection_timeout = self._connection.get_option('connection_timeout')
 420             display.debug("{action}: saving original connect_timeout of {timeout}".format(action=self._task.action, timeout=original_connection_timeout))
 421         except KeyError:
 422             display.debug("{action}: connect_timeout connection option has not been set".format(action=self._task.action))
 423         # Initiate reboot
 424         reboot_result = self.perform_reboot(task_vars, distribution)
 425 
 426         if reboot_result['failed']:
 427             result = reboot_result
 428             elapsed = datetime.utcnow() - reboot_result['start']
 429             result['elapsed'] = elapsed.seconds
 430             return result
 431 
 432         if self.post_reboot_delay != 0:
 433             display.debug("{action}: waiting an additional {delay} seconds".format(action=self._task.action, delay=self.post_reboot_delay))
 434             display.vvv("{action}: waiting an additional {delay} seconds".format(action=self._task.action, delay=self.post_reboot_delay))
 435             time.sleep(self.post_reboot_delay)
 436 
 437         # Make sure reboot was successful
 438         result = self.validate_reboot(distribution, original_connection_timeout, action_kwargs={'previous_boot_time': previous_boot_time})
 439 
 440         elapsed = datetime.utcnow() - reboot_result['start']
 441         result['elapsed'] = elapsed.seconds
 442 
 443         return result
