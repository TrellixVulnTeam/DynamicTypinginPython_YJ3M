  18 from __future__ import (absolute_import, division, print_function)
  19 __metaclass__ = type

  21 from ansible.errors import AnsibleUndefinedVariable
  22 from ansible.module_utils.six import string_types
  23 from ansible.module_utils._text import to_text
  24 from ansible.plugins.action import ActionBase
  25 
  26 
  27 class ActionModule(ActionBase):
  28     ''' Print statements during execution '''
  29 
  30     TRANSFERS_FILES = False
  31     _VALID_ARGS = frozenset(('msg', 'var', 'verbosity'))
  32 
  33     def run(self, tmp=None, task_vars=None):
  34         if task_vars is None:
  35             task_vars = dict()
  36 
  37         if 'msg' in self._task.args and 'var' in self._task.args:
  38             return {"failed": True, "msg": "'msg' and 'var' are incompatible options"}
  39 
  40         result = super(ActionModule, self).run(tmp, task_vars)
  41         del tmp  # tmp no longer has any effect
  42 
  43         # get task verbosity
  44         verbosity = int(self._task.args.get('verbosity', 0))
  45 
  46         if verbosity <= self._display.verbosity:
  47             if 'msg' in self._task.args:
  48                 result['msg'] = self._task.args['msg']
  49 
  50             elif 'var' in self._task.args:
  51                 try:
  52                     results = self._templar.template(self._task.args['var'], convert_bare=True, fail_on_undefined=True)
  53                     if results == self._task.args['var']:
  54                         # if results is not str/unicode type, raise an exception
  55                         if not isinstance(results, string_types):
  56                             raise AnsibleUndefinedVariable
  57                         # If var name is same as result, try to template it
  58                         results = self._templar.template("{{" + results + "}}", convert_bare=True, fail_on_undefined=True)
  59                 except AnsibleUndefinedVariable as e:
  60                     results = u"VARIABLE IS NOT DEFINED!"
  61                     if self._display.verbosity > 0:
  62                         results += u": %s" % to_text(e)
  63 
  64                 if isinstance(self._task.args['var'], (list, dict)):
  65                     # If var is a list or dict, use the type as key to display
  66                     result[to_text(type(self._task.args['var']))] = results
  67                 else:
  68                     result[self._task.args['var']] = results
  69             else:
  70                 result['msg'] = 'Hello world!'
  71 
  72             # force flag to make debug output module always verbose
  73             result['_ansible_verbose_always'] = True
  74         else:
  75             result['skipped_reason'] = "Verbosity threshold not met."
  76             result['skipped'] = True
  77 
  78         result['failed'] = False
  79 
  80         return result
