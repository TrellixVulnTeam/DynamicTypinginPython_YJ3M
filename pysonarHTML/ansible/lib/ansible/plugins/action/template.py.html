  23 class ActionModule(ActionBase):
  24 
  25     TRANSFERS_FILES = True
  26     DEFAULT_NEWLINE_SEQUENCE = "\n"
  27 
  28     def run(self, tmp=None, task_vars=None):
  29         ''' handler for template operations '''
  30 
  31         if task_vars is None:
  32             task_vars = dict()
  33 
  34         result = super(ActionModule, self).run(tmp, task_vars)
  35         del tmp  # tmp no longer has any effect
  36 
  37         # Options type validation
  38         # stings
  39         for s_type in ('src', 'dest', 'state', 'newline_sequence', 'variable_start_string', 'variable_end_string', 'block_start_string',
  40                        'block_end_string'):
  41             if s_type in self._task.args:
  42                 value = ensure_type(self._task.args[s_type], 'string')
  43                 if value is not None and not isinstance(value, string_types):
  44                     raise AnsibleActionFail("%s is expected to be a string, but got %s instead" % (s_type, type(value)))
  45                 self._task.args[s_type] = value
  46 
  47         # booleans
  48         try:
  49             follow = boolean(self._task.args.get('follow', False), strict=False)
  50             trim_blocks = boolean(self._task.args.get('trim_blocks', True), strict=False)
  51             lstrip_blocks = boolean(self._task.args.get('lstrip_blocks', False), strict=False)
  52         except TypeError as e:
  53             raise AnsibleActionFail(to_native(e))
  54 
  55         # assign to local vars for ease of use
  56         source = self._task.args.get('src', None)
  57         dest = self._task.args.get('dest', None)
  58         state = self._task.args.get('state', None)
  59         newline_sequence = self._task.args.get('newline_sequence', self.DEFAULT_NEWLINE_SEQUENCE)
  60         variable_start_string = self._task.args.get('variable_start_string', None)
  61         variable_end_string = self._task.args.get('variable_end_string', None)
  62         block_start_string = self._task.args.get('block_start_string', None)
  63         block_end_string = self._task.args.get('block_end_string', None)
  64         output_encoding = self._task.args.get('output_encoding', 'utf-8') or 'utf-8'
  65 
  66         # Option `lstrip_blocks' was added in Jinja2 version 2.7.
  67         if lstrip_blocks:
  68             try:
  69                 import jinja2.defaults
  70             except ImportError:
  71                 raise AnsibleError('Unable to import Jinja2 defaults for determining Jinja2 features.')
  72 
  73             try:
  74                 jinja2.defaults.LSTRIP_BLOCKS
  75             except AttributeError:
  76                 raise AnsibleError("Option `lstrip_blocks' is only available in Jinja2 versions >=2.7")
  77 
  78         wrong_sequences = ["\\n", "\\r", "\\r\\n"]
  79         allowed_sequences = ["\n", "\r", "\r\n"]
  80 
  81         # We need to convert unescaped sequences to proper escaped sequences for Jinja2
  82         if newline_sequence in wrong_sequences:
  83             newline_sequence = allowed_sequences[wrong_sequences.index(newline_sequence)]
  84 
  85         try:
  86             # logical validation
  87             if state is not None:
  88                 raise AnsibleActionFail("'state' cannot be specified on a template")
  89             elif source is None or dest is None:
  90                 raise AnsibleActionFail("src and dest are required")
  91             elif newline_sequence not in allowed_sequences:
  92                 raise AnsibleActionFail("newline_sequence needs to be one of: \n, \r or \r\n")
  93             else:
  94                 try:
  95                     source = self._find_needle('templates', source)
  96                 except AnsibleError as e:
  97                     raise AnsibleActionFail(to_text(e))
  98 
  99             mode = self._task.args.get('mode', None)
 100             if mode == 'preserve':
 101                 mode = '0%03o' % stat.S_IMODE(os.stat(source).st_mode)
 102 
 103             # Get vault decrypted tmp file
 104             try:
 105                 tmp_source = self._loader.get_real_file(source)
 106             except AnsibleFileNotFound as e:
 107                 raise AnsibleActionFail("could not find src=%s, %s" % (source, to_text(e)))
 108             b_tmp_source = to_bytes(tmp_source, errors='surrogate_or_strict')
 109 
 110             # template the source data locally & get ready to transfer
 111             try:
 112                 with open(b_tmp_source, 'rb') as f:
 113                     try:
 114                         template_data = to_text(f.read(), errors='surrogate_or_strict')
 115                     except UnicodeError:
 116                         raise AnsibleActionFail("Template source files must be utf-8 encoded")
 117 
 118                 # set jinja2 internal search path for includes
 119                 searchpath = task_vars.get('ansible_search_path', [])
 120                 searchpath.extend([self._loader._basedir, os.path.dirname(source)])
 121 
 122                 # We want to search into the 'templates' subdir of each search path in
 123                 # addition to our original search paths.
 124                 newsearchpath = []
 125                 for p in searchpath:
 126                     newsearchpath.append(os.path.join(p, 'templates'))
 127                     newsearchpath.append(p)
 128                 searchpath = newsearchpath
 129 
 130                 # add ansible 'template' vars
 131                 temp_vars = task_vars.copy()
 132                 temp_vars.update(generate_ansible_template_vars(source, dest))
 133 
 134                 with self._templar.set_temporary_context(searchpath=searchpath, newline_sequence=newline_sequence,
 135                                                          block_start_string=block_start_string, block_end_string=block_end_string,
 136                                                          variable_start_string=variable_start_string, variable_end_string=variable_end_string,
 137                                                          trim_blocks=trim_blocks, lstrip_blocks=lstrip_blocks,
 138                                                          available_variables=temp_vars):
 139                     resultant = self._templar.do_template(template_data, preserve_trailing_newlines=True, escape_backslashes=False)
 140             except AnsibleAction:
 141                 raise
 142             except Exception as e:
 143                 raise AnsibleActionFail("%s: %s" % (type(e).__name__, to_text(e)))
 144             finally:
 145                 self._loader.cleanup_tmp_file(b_tmp_source)
 146 
 147             new_task = self._task.copy()
 148             # mode is either the mode from task.args or the mode of the source file if the task.args
 149             # mode == 'preserve'
 150             new_task.args['mode'] = mode
 151 
 152             # remove 'template only' options:
 153             for remove in ('newline_sequence', 'block_start_string', 'block_end_string', 'variable_start_string', 'variable_end_string',
 154                            'trim_blocks', 'lstrip_blocks', 'output_encoding'):
 155                 new_task.args.pop(remove, None)
 156 
 157             local_tempdir = tempfile.mkdtemp(dir=C.DEFAULT_LOCAL_TMP)
 158 
 159             try:
 160                 result_file = os.path.join(local_tempdir, os.path.basename(source))
 161                 with open(to_bytes(result_file, errors='surrogate_or_strict'), 'wb') as f:
 162                     f.write(to_bytes(resultant, encoding=output_encoding, errors='surrogate_or_strict'))
 163 
 164                 new_task.args.update(
 165                     dict(
 166                         src=result_file,
 167                         dest=dest,
 168                         follow=follow,
 169                     ),
 170                 )
 171                 copy_action = self._shared_loader_obj.action_loader.get('copy',
 172                                                                         task=new_task,
 173                                                                         connection=self._connection,
 174                                                                         play_context=self._play_context,
 175                                                                         loader=self._loader,
 176                                                                         templar=self._templar,
 177                                                                         shared_loader_obj=self._shared_loader_obj)
 178                 result.update(copy_action.run(task_vars=task_vars))
 179             finally:
 180                 shutil.rmtree(to_bytes(local_tempdir, errors='surrogate_or_strict'))
 181 
 182         except AnsibleAction as e:
 183             result.update(e.result)
 184         finally:
 185             self._remove_tmp_path(self._connection._shell.tmpdir)
 186 
 187         return result
