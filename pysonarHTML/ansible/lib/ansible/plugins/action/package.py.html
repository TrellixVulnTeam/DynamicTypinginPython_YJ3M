  20 from ansible.errors import AnsibleAction, AnsibleActionFail
  21 from ansible.executor.module_common import get_action_args_with_defaults
  22 from ansible.plugins.action import ActionBase
  23 from ansible.utils.display import Display
  24 
  25 display = Display()
  26 
  27 
  28 class ActionModule(ActionBase):
  29 
  30     TRANSFERS_FILES = False
  31 
  32     def run(self, tmp=None, task_vars=None):
  33         ''' handler for package operations '''
  34 
  35         self._supports_check_mode = True
  36         self._supports_async = True
  37 
  38         result = super(ActionModule, self).run(tmp, task_vars)
  39         del tmp  # tmp no longer has any effect
  40 
  41         module = self._task.args.get('use', 'auto')
  42 
  43         if module == 'auto':
  44             try:
  45                 if self._task.delegate_to:  # if we delegate, we should use delegated host's facts
  46                     module = self._templar.template("{{hostvars['%s']['ansible_facts']['pkg_mgr']}}" % self._task.delegate_to)
  47                 else:
  48                     module = self._templar.template('{{ansible_facts.pkg_mgr}}')
  49             except Exception:
  50                 pass  # could not get it from template!
  51 
  52         try:
  53             if module == 'auto':
  54                 facts = self._execute_module(module_name='setup', module_args=dict(filter='ansible_pkg_mgr', gather_subset='!all'), task_vars=task_vars)
  55                 display.debug("Facts %s" % facts)
  56                 module = facts.get('ansible_facts', {}).get('ansible_pkg_mgr', 'auto')
  57 
  58             if module != 'auto':
  59 
  60                 if module not in self._shared_loader_obj.module_loader:
  61                     raise AnsibleActionFail('Could not find a module for %s.' % module)
  62                 else:
  63                     # run the 'package' module
  64                     new_module_args = self._task.args.copy()
  65                     if 'use' in new_module_args:
  66                         del new_module_args['use']
  67 
  68                     # get defaults for specific module
  69                     new_module_args = get_action_args_with_defaults(
  70                         module, new_module_args, self._task.module_defaults, self._templar, self._task._ansible_internal_redirect_list
  71                     )
  72 
  73                     display.vvvv("Running %s" % module)
  74                     result.update(self._execute_module(module_name=module, module_args=new_module_args, task_vars=task_vars, wrap_async=self._task.async_val))
  75             else:
  76                 raise AnsibleActionFail('Could not detect which package manager to use. Try gathering facts or setting the "use" option.')
  77 
  78         except AnsibleAction as e:
  79             result.update(e.result)
  80         finally:
  81             if not self._task.async_val:
  82                 # remove a temporary path we created
  83                 self._remove_tmp_path(self._connection._shell.tmpdir)
  84 
  85         return result
