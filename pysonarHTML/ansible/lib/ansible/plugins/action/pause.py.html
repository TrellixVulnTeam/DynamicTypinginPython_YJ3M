  70 class ActionModule(ActionBase):
  71     ''' pauses execution for a length or time, or until input is received '''
  72 
  73     BYPASS_HOST_LOOP = True
  74     _VALID_ARGS = frozenset(('echo', 'minutes', 'prompt', 'seconds'))
  75 
  76     def run(self, tmp=None, task_vars=None):
  77         ''' run the pause action module '''
  78         if task_vars is None:
  79             task_vars = dict()
  80 
  81         result = super(ActionModule, self).run(tmp, task_vars)
  82         del tmp  # tmp no longer has any effect
  83 
  84         duration_unit = 'minutes'
  85         prompt = None
  86         seconds = None
  87         echo = True
  88         echo_prompt = ''
  89         result.update(dict(
  90             changed=False,
  91             rc=0,
  92             stderr='',
  93             stdout='',
  94             start=None,
  95             stop=None,
  96             delta=None,
  97             echo=echo
  98         ))
  99 
 100         # Should keystrokes be echoed to stdout?
 101         if 'echo' in self._task.args:
 102             try:
 103                 echo = boolean(self._task.args['echo'])
 104             except TypeError as e:
 105                 result['failed'] = True
 106                 result['msg'] = to_native(e)
 107                 return result
 108 
 109             # Add a note saying the output is hidden if echo is disabled
 110             if not echo:
 111                 echo_prompt = ' (output is hidden)'
 112 
 113         # Is 'prompt' a key in 'args'?
 114         if 'prompt' in self._task.args:
 115             prompt = "[%s]\n%s%s:" % (self._task.get_name().strip(), self._task.args['prompt'], echo_prompt)
 116         else:
 117             # If no custom prompt is specified, set a default prompt
 118             prompt = "[%s]\n%s%s:" % (self._task.get_name().strip(), 'Press enter to continue, Ctrl+C to interrupt', echo_prompt)
 119 
 120         # Are 'minutes' or 'seconds' keys that exist in 'args'?
 121         if 'minutes' in self._task.args or 'seconds' in self._task.args:
 122             try:
 123                 if 'minutes' in self._task.args:
 124                     # The time() command operates in seconds so we need to
 125                     # recalculate for minutes=X values.
 126                     seconds = int(self._task.args['minutes']) * 60
 127                 else:
 128                     seconds = int(self._task.args['seconds'])
 129                     duration_unit = 'seconds'
 130 
 131             except ValueError as e:
 132                 result['failed'] = True
 133                 result['msg'] = u"non-integer value given for prompt duration:\n%s" % to_text(e)
 134                 return result
 135 
 136         ########################################################################
 137         # Begin the hard work!
 138 
 139         start = time.time()
 140         result['start'] = to_text(datetime.datetime.now())
 141         result['user_input'] = b''
 142 
 143         stdin_fd = None
 144         old_settings = None
 145         try:
 146             if seconds is not None:
 147                 if seconds < 1:
 148                     seconds = 1
 149 
 150                 # setup the alarm handler
 151                 signal.signal(signal.SIGALRM, timeout_handler)
 152                 signal.alarm(seconds)
 153 
 154                 # show the timer and control prompts
 155                 display.display("Pausing for %d seconds%s" % (seconds, echo_prompt))
 156                 display.display("(ctrl+C then 'C' = continue early, ctrl+C then 'A' = abort)\r"),
 157 
 158                 # show the prompt specified in the task
 159                 if 'prompt' in self._task.args:
 160                     display.display(prompt)
 161 
 162             else:
 163                 display.display(prompt)
 164 
 165             # save the attributes on the existing (duped) stdin so
 166             # that we can restore them later after we set raw mode
 167             stdin_fd = None
 168             stdout_fd = None
 169             try:
 170                 if PY3:
 171                     stdin = self._connection._new_stdin.buffer
 172                     stdout = sys.stdout.buffer
 173                 else:
 174                     stdin = self._connection._new_stdin
 175                     stdout = sys.stdout
 176                 stdin_fd = stdin.fileno()
 177                 stdout_fd = stdout.fileno()
 178             except (ValueError, AttributeError):
 179                 # ValueError: someone is using a closed file descriptor as stdin
 180                 # AttributeError: someone is using a null file descriptor as stdin on windoez
 181                 stdin = None
 182 
 183             if stdin_fd is not None:
 184                 if isatty(stdin_fd):
 185                     # grab actual Ctrl+C sequence
 186                     try:
 187                         intr = termios.tcgetattr(stdin_fd)[6][termios.VINTR]
 188                     except Exception:
 189                         # unsupported/not present, use default
 190                         intr = b'\x03'  # value for Ctrl+C
 191 
 192                     # get backspace sequences
 193                     try:
 194                         backspace = termios.tcgetattr(stdin_fd)[6][termios.VERASE]
 195                     except Exception:
 196                         backspace = [b'\x7f', b'\x08']
 197 
 198                     old_settings = termios.tcgetattr(stdin_fd)
 199                     tty.setraw(stdin_fd)
 200 
 201                     # Only set stdout to raw mode if it is a TTY. This is needed when redirecting
 202                     # stdout to a file since a file cannot be set to raw mode.
 203                     if isatty(stdout_fd):
 204                         tty.setraw(stdout_fd)
 205 
 206                     # Only echo input if no timeout is specified
 207                     if not seconds and echo:
 208                         new_settings = termios.tcgetattr(stdin_fd)
 209                         new_settings[3] = new_settings[3] | termios.ECHO
 210                         termios.tcsetattr(stdin_fd, termios.TCSANOW, new_settings)
 211 
 212                     # flush the buffer to make sure no previous key presses
 213                     # are read in below
 214                     termios.tcflush(stdin, termios.TCIFLUSH)
 215 
 216             while True:
 217 
 218                 try:
 219                     if stdin_fd is not None:
 220 
 221                         key_pressed = stdin.read(1)
 222 
 223                         if key_pressed == intr:  # value for Ctrl+C
 224                             clear_line(stdout)
 225                             raise KeyboardInterrupt
 226 
 227                     if not seconds:
 228                         if stdin_fd is None or not isatty(stdin_fd):
 229                             display.warning("Not waiting for response to prompt as stdin is not interactive")
 230                             break
 231 
 232                         # read key presses and act accordingly
 233                         if key_pressed in (b'\r', b'\n'):
 234                             clear_line(stdout)
 235                             break
 236                         elif key_pressed in backspace:
 237                             # delete a character if backspace is pressed
 238                             result['user_input'] = result['user_input'][:-1]
 239                             clear_line(stdout)
 240                             if echo:
 241                                 stdout.write(result['user_input'])
 242                             stdout.flush()
 243                         else:
 244                             result['user_input'] += key_pressed
 245 
 246                 except KeyboardInterrupt:
 247                     signal.alarm(0)
 248                     display.display("Press 'C' to continue the play or 'A' to abort \r"),
 249                     if self._c_or_a(stdin):
 250                         clear_line(stdout)
 251                         break
 252 
 253                     clear_line(stdout)
 254 
 255                     raise AnsibleError('user requested abort!')
 256 
 257         except AnsibleTimeoutExceeded:
 258             # this is the exception we expect when the alarm signal
 259             # fires, so we simply ignore it to move into the cleanup
 260             pass
 261         finally:
 262             # cleanup and save some information
 263             # restore the old settings for the duped stdin stdin_fd
 264             if not(None in (stdin_fd, old_settings)) and isatty(stdin_fd):
 265                 termios.tcsetattr(stdin_fd, termios.TCSADRAIN, old_settings)
 266 
 267             duration = time.time() - start
 268             result['stop'] = to_text(datetime.datetime.now())
 269             result['delta'] = int(duration)
 270 
 271             if duration_unit == 'minutes':
 272                 duration = round(duration / 60.0, 2)
 273             else:
 274                 duration = round(duration, 2)
 275             result['stdout'] = "Paused for %s %s" % (duration, duration_unit)
 276 
 277         result['user_input'] = to_text(result['user_input'], errors='surrogate_or_strict')
 278         return result
 279 
 280     def _c_or_a(self, stdin):
 281         while True:
 282             key_pressed = stdin.read(1)
 283             if key_pressed.lower() == b'a':
 284                 return False
 285             elif key_pressed.lower() == b'c':
 286                 return True
