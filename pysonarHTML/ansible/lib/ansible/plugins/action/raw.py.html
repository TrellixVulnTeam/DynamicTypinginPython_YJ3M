  20 from ansible.plugins.action import ActionBase

  23 class ActionModule(ActionBase):
  24     TRANSFERS_FILES = False
  25 
  26     def run(self, tmp=None, task_vars=None):
  27         if task_vars is None:
  28             task_vars = dict()
  29 
  30         if self._task.environment and any(self._task.environment):
  31             self._display.warning('raw module does not support the environment keyword')
  32 
  33         result = super(ActionModule, self).run(tmp, task_vars)
  34         del tmp  # tmp no longer has any effect
  35 
  36         if self._play_context.check_mode:
  37             # in --check mode, always skip this module execution
  38             result['skipped'] = True
  39             return result
  40 
  41         executable = self._task.args.get('executable', False)
  42         result.update(self._low_level_execute_command(self._task.args.get('_raw_params'), executable=executable))
  43 
  44         result['changed'] = True
  45 
  46         if 'rc' in result and result['rc'] != 0:
  47             result['failed'] = True
  48             result['msg'] = 'non-zero return code'
  49 
  50         return result
