  21 from ansible.errors import AnsibleAction, AnsibleActionFail
  22 from ansible.executor.module_common import get_action_args_with_defaults
  23 from ansible.plugins.action import ActionBase
  24 
  25 
  26 class ActionModule(ActionBase):
  27 
  28     TRANSFERS_FILES = False
  29 
  30     UNUSED_PARAMS = {
  31         'systemd': ['pattern', 'runlevel', 'sleep', 'arguments', 'args'],
  32     }
  33 
  34     def run(self, tmp=None, task_vars=None):
  35         ''' handler for package operations '''
  36 
  37         self._supports_check_mode = True
  38         self._supports_async = True
  39 
  40         result = super(ActionModule, self).run(tmp, task_vars)
  41         del tmp  # tmp no longer has any effect
  42 
  43         module = self._task.args.get('use', 'auto').lower()
  44 
  45         if module == 'auto':
  46             try:
  47                 if self._task.delegate_to:  # if we delegate, we should use delegated host's facts
  48                     module = self._templar.template("{{hostvars['%s']['ansible_facts']['service_mgr']}}" % self._task.delegate_to)
  49                 else:
  50                     module = self._templar.template('{{ansible_facts.service_mgr}}')
  51             except Exception:
  52                 pass  # could not get it from template!
  53 
  54         try:
  55             if module == 'auto':
  56                 facts = self._execute_module(module_name='setup', module_args=dict(gather_subset='!all', filter='ansible_service_mgr'), task_vars=task_vars)
  57                 self._display.debug("Facts %s" % facts)
  58                 module = facts.get('ansible_facts', {}).get('ansible_service_mgr', 'auto')
  59 
  60             if not module or module == 'auto' or module not in self._shared_loader_obj.module_loader:
  61                 module = 'service'
  62 
  63             if module != 'auto':
  64                 # run the 'service' module
  65                 new_module_args = self._task.args.copy()
  66                 if 'use' in new_module_args:
  67                     del new_module_args['use']
  68 
  69                 if module in self.UNUSED_PARAMS:
  70                     for unused in self.UNUSED_PARAMS[module]:
  71                         if unused in new_module_args:
  72                             del new_module_args[unused]
  73                             self._display.warning('Ignoring "%s" as it is not used in "%s"' % (unused, module))
  74 
  75                 # get defaults for specific module
  76                 new_module_args = get_action_args_with_defaults(
  77                     module, new_module_args, self._task.module_defaults, self._templar, self._task._ansible_internal_redirect_list
  78                 )
  79 
  80                 self._display.vvvv("Running %s" % module)
  81                 result.update(self._execute_module(module_name=module, module_args=new_module_args, task_vars=task_vars, wrap_async=self._task.async_val))
  82             else:
  83                 raise AnsibleActionFail('Could not detect which service manager to use. Try gathering facts or setting the "use" option.')
  84 
  85         except AnsibleAction as e:
  86             result.update(e.result)
  87         finally:
  88             if not self._task.async_val:
  89                 self._remove_tmp_path(self._connection._shell.tmpdir)
  90 
  91         return result
