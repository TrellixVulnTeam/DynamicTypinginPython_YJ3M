  23 from datetime import datetime, timedelta

  29 display = Display()

  32 class TimedOutException(Exception):
  33     pass
  34 
  35 
  36 class ActionModule(ActionBase):
  37     TRANSFERS_FILES = False
  38     _VALID_ARGS = frozenset(('connect_timeout', 'delay', 'sleep', 'timeout'))
  39 
  40     DEFAULT_CONNECT_TIMEOUT = 5
  41     DEFAULT_DELAY = 0
  42     DEFAULT_SLEEP = 1
  43     DEFAULT_TIMEOUT = 600
  44 
  45     def do_until_success_or_timeout(self, what, timeout, connect_timeout, what_desc, sleep=1):
  46         max_end_time = datetime.utcnow() + timedelta(seconds=timeout)
  47 
  48         e = None
  49         while datetime.utcnow() < max_end_time:
  50             try:
  51                 what(connect_timeout)
  52                 if what_desc:
  53                     display.debug("wait_for_connection: %s success" % what_desc)
  54                 return
  55             except Exception as e:
  56                 error = e  # PY3 compatibility to store exception for use outside of this block
  57                 if what_desc:
  58                     display.debug("wait_for_connection: %s fail (expected), retrying in %d seconds..." % (what_desc, sleep))
  59                 time.sleep(sleep)
  60 
  61         raise TimedOutException("timed out waiting for %s: %s" % (what_desc, error))
  62 
  63     def run(self, tmp=None, task_vars=None):
  64         if task_vars is None:
  65             task_vars = dict()
  66 
  67         connect_timeout = int(self._task.args.get('connect_timeout', self.DEFAULT_CONNECT_TIMEOUT))
  68         delay = int(self._task.args.get('delay', self.DEFAULT_DELAY))
  69         sleep = int(self._task.args.get('sleep', self.DEFAULT_SLEEP))
  70         timeout = int(self._task.args.get('timeout', self.DEFAULT_TIMEOUT))
  71 
  72         if self._play_context.check_mode:
  73             display.vvv("wait_for_connection: skipping for check_mode")
  74             return dict(skipped=True)
  75 
  76         result = super(ActionModule, self).run(tmp, task_vars)
  77         del tmp  # tmp no longer has any effect
  78 
  79         def ping_module_test(connect_timeout):
  80             ''' Test ping module, if available '''
  81             display.vvv("wait_for_connection: attempting ping module test")
  82             # re-run interpreter discovery if we ran it in the first iteration
  83             if self._discovered_interpreter_key:
  84                 task_vars['ansible_facts'].pop(self._discovered_interpreter_key, None)
  85             # call connection reset between runs if it's there
  86             try:
  87                 self._connection.reset()
  88             except AttributeError:
  89                 pass
  90 
  91             ping_result = self._execute_module(module_name='ping', module_args=dict(), task_vars=task_vars)
  92 
  93             # Test module output
  94             if ping_result['ping'] != 'pong':
  95                 raise Exception('ping test failed')
  96 
  97         start = datetime.now()
  98 
  99         if delay:
 100             time.sleep(delay)
 101 
 102         try:
 103             # If the connection has a transport_test method, use it first
 104             if hasattr(self._connection, 'transport_test'):
 105                 self.do_until_success_or_timeout(self._connection.transport_test, timeout, connect_timeout, what_desc="connection port up", sleep=sleep)
 106 
 107             # Use the ping module test to determine end-to-end connectivity
 108             self.do_until_success_or_timeout(ping_module_test, timeout, connect_timeout, what_desc="ping module test", sleep=sleep)
 109 
 110         except TimedOutException as e:
 111             result['failed'] = True
 112             result['msg'] = to_text(e)
 113 
 114         elapsed = datetime.now() - start
 115         result['elapsed'] = elapsed.seconds
 116 
 117         # remove a temporary path we created
 118         self._remove_tmp_path(self._connection._shell.tmpdir)
 119 
 120         return result
