  21 from ansible.module_utils.six import iteritems, string_types
  22 from ansible.module_utils.parsing.convert_bool import boolean
  23 from ansible.plugins.action import ActionBase
  24 from ansible.utils.vars import isidentifier

  27 class ActionModule(ActionBase):
  28 
  29     TRANSFERS_FILES = False
  30     _VALID_ARGS = frozenset(('aggregate', 'data', 'per_host'))
  31 
  32     # TODO: document this in non-empty set_stats.py module
  33     def run(self, tmp=None, task_vars=None):
  34         if task_vars is None:
  35             task_vars = dict()
  36 
  37         result = super(ActionModule, self).run(tmp, task_vars)
  38         del tmp  # tmp no longer has any effect
  39 
  40         stats = {'data': {}, 'per_host': False, 'aggregate': True}
  41 
  42         if self._task.args:
  43             data = self._task.args.get('data', {})
  44 
  45             if not isinstance(data, dict):
  46                 data = self._templar.template(data, convert_bare=False, fail_on_undefined=True)
  47 
  48             if not isinstance(data, dict):
  49                 result['failed'] = True
  50                 result['msg'] = "The 'data' option needs to be a dictionary/hash"
  51                 return result
  52 
  53             # set boolean options, defaults are set above in stats init
  54             for opt in ['per_host', 'aggregate']:
  55                 val = self._task.args.get(opt, None)
  56                 if val is not None:
  57                     if not isinstance(val, bool):
  58                         stats[opt] = boolean(self._templar.template(val), strict=False)
  59                     else:
  60                         stats[opt] = val
  61 
  62             for (k, v) in iteritems(data):
  63 
  64                 k = self._templar.template(k)
  65 
  66                 if not isidentifier(k):
  67                     result['failed'] = True
  68                     result['msg'] = ("The variable name '%s' is not valid. Variables must start with a letter or underscore character, and contain only "
  69                                      "letters, numbers and underscores." % k)
  70                     return result
  71 
  72                 stats['data'][k] = self._templar.template(v)
  73 
  74         result['changed'] = False
  75         result['ansible_stats'] = stats
  76 
  77         return result
