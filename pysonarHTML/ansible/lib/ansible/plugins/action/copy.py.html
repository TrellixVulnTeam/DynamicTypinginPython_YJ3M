  55 def _walk_dirs(topdir, base_path=None, local_follow=False, trailing_slash_detector=None):
  56     """
  57     Walk a filesystem tree returning enough information to copy the files
  58 
  59     :arg topdir: The directory that the filesystem tree is rooted at
  60     :kwarg base_path: The initial directory structure to strip off of the
  61         files for the destination directory.  If this is None (the default),
  62         the base_path is set to ``top_dir``.
  63     :kwarg local_follow: Whether to follow symlinks on the source.  When set
  64         to False, no symlinks are dereferenced.  When set to True (the
  65         default), the code will dereference most symlinks.  However, symlinks
  66         can still be present if needed to break a circular link.
  67     :kwarg trailing_slash_detector: Function to determine if a path has
  68         a trailing directory separator. Only needed when dealing with paths on
  69         a remote machine (in which case, pass in a function that is aware of the
  70         directory separator conventions on the remote machine).
  71     :returns: dictionary of tuples.  All of the path elements in the structure are text strings.
  72             This separates all the files, directories, and symlinks along with
  73             important information about each::
  74 
  75                 { 'files': [('/absolute/path/to/copy/from', 'relative/path/to/copy/to'), ...],
  76                   'directories': [('/absolute/path/to/copy/from', 'relative/path/to/copy/to'), ...],
  77                   'symlinks': [('/symlink/target/path', 'relative/path/to/copy/to'), ...],
  78                 }
  79 
  80         The ``symlinks`` field is only populated if ``local_follow`` is set to False
  81         *or* a circular symlink cannot be dereferenced.
  82 
  83     """
  84     # Convert the path segments into byte strings
  85 
  86     r_files = {'files': [], 'directories': [], 'symlinks': []}
  87 
  88     def _recurse(topdir, rel_offset, parent_dirs, rel_base=u''):
  89         """
  90         This is a closure (function utilizing variables from it's parent
  91         function's scope) so that we only need one copy of all the containers.
  92         Note that this function uses side effects (See the Variables used from
  93         outer scope).
  94 
  95         :arg topdir: The directory we are walking for files
  96         :arg rel_offset: Integer defining how many characters to strip off of
  97             the beginning of a path
  98         :arg parent_dirs: Directories that we're copying that this directory is in.
  99         :kwarg rel_base: String to prepend to the path after ``rel_offset`` is
 100             applied to form the relative path.
 101 
 102         Variables used from the outer scope
 103         ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
 104 
 105         :r_files: Dictionary of files in the hierarchy.  See the return value
 106             for :func:`walk` for the structure of this dictionary.
 107         :local_follow: Read-only inside of :func:`_recurse`. Whether to follow symlinks
 108         """
 109         for base_path, sub_folders, files in os.walk(topdir):
 110             for filename in files:
 111                 filepath = os.path.join(base_path, filename)
 112                 dest_filepath = os.path.join(rel_base, filepath[rel_offset:])
 113 
 114                 if os.path.islink(filepath):
 115                     # Dereference the symlnk
 116                     real_file = os.path.realpath(filepath)
 117                     if local_follow and os.path.isfile(real_file):
 118                         # Add the file pointed to by the symlink
 119                         r_files['files'].append((real_file, dest_filepath))
 120                     else:
 121                         # Mark this file as a symlink to copy
 122                         r_files['symlinks'].append((os.readlink(filepath), dest_filepath))
 123                 else:
 124                     # Just a normal file
 125                     r_files['files'].append((filepath, dest_filepath))
 126 
 127             for dirname in sub_folders:
 128                 dirpath = os.path.join(base_path, dirname)
 129                 dest_dirpath = os.path.join(rel_base, dirpath[rel_offset:])
 130                 real_dir = os.path.realpath(dirpath)
 131                 dir_stats = os.stat(real_dir)
 132 
 133                 if os.path.islink(dirpath):
 134                     if local_follow:
 135                         if (dir_stats.st_dev, dir_stats.st_ino) in parent_dirs:
 136                             # Just insert the symlink if the target directory
 137                             # exists inside of the copy already
 138                             r_files['symlinks'].append((os.readlink(dirpath), dest_dirpath))
 139                         else:
 140                             # Walk the dirpath to find all parent directories.
 141                             new_parents = set()
 142                             parent_dir_list = os.path.dirname(dirpath).split(os.path.sep)
 143                             for parent in range(len(parent_dir_list), 0, -1):
 144                                 parent_stat = os.stat(u'/'.join(parent_dir_list[:parent]))
 145                                 if (parent_stat.st_dev, parent_stat.st_ino) in parent_dirs:
 146                                     # Reached the point at which the directory
 147                                     # tree is already known.  Don't add any
 148                                     # more or we might go to an ancestor that
 149                                     # isn't being copied.
 150                                     break
 151                                 new_parents.add((parent_stat.st_dev, parent_stat.st_ino))
 152 
 153                             if (dir_stats.st_dev, dir_stats.st_ino) in new_parents:
 154                                 # This was a a circular symlink.  So add it as
 155                                 # a symlink
 156                                 r_files['symlinks'].append((os.readlink(dirpath), dest_dirpath))
 157                             else:
 158                                 # Walk the directory pointed to by the symlink
 159                                 r_files['directories'].append((real_dir, dest_dirpath))
 160                                 offset = len(real_dir) + 1
 161                                 _recurse(real_dir, offset, parent_dirs.union(new_parents), rel_base=dest_dirpath)
 162                     else:
 163                         # Add the symlink to the destination
 164                         r_files['symlinks'].append((os.readlink(dirpath), dest_dirpath))
 165                 else:
 166                     # Just a normal directory
 167                     r_files['directories'].append((dirpath, dest_dirpath))
 168 
 169     # Check if the source ends with a "/" so that we know which directory
 170     # level to work at (similar to rsync)
 171     source_trailing_slash = False
 172     if trailing_slash_detector:
 173         source_trailing_slash = trailing_slash_detector(topdir)
 174     else:
 175         source_trailing_slash = topdir.endswith(os.path.sep)
 176 
 177     # Calculate the offset needed to strip the base_path to make relative
 178     # paths
 179     if base_path is None:
 180         base_path = topdir
 181     if not source_trailing_slash:
 182         base_path = os.path.dirname(base_path)
 183     if topdir.startswith(base_path):
 184         offset = len(base_path)
 185 
 186     # Make sure we're making the new paths relative
 187     if trailing_slash_detector and not trailing_slash_detector(base_path):
 188         offset += 1
 189     elif not base_path.endswith(os.path.sep):
 190         offset += 1
 191 
 192     if os.path.islink(topdir) and not local_follow:
 193         r_files['symlinks'] = (os.readlink(topdir), os.path.basename(topdir))
 194         return r_files
 195 
 196     dir_stats = os.stat(topdir)
 197     parents = frozenset(((dir_stats.st_dev, dir_stats.st_ino),))
 198     # Actually walk the directory hierarchy
 199     _recurse(topdir, offset, parents)
 200 
 201     return r_files

 208     def _ensure_invocation(self, result):
 209         # NOTE: adding invocation arguments here needs to be kept in sync with
 210         # any no_log specified in the argument_spec in the module.
 211         # This is not automatic.
 212         if 'invocation' not in result:
 213             if self._play_context.no_log:
 214                 result['invocation'] = "CENSORED: no_log is set"
 215             else:
 216                 # NOTE: Should be removed in the future. For now keep this broken
 217                 # behaviour, have a look in the PR 51582
 218                 result['invocation'] = self._task.args.copy()
 219                 result['invocation']['module_args'] = self._task.args.copy()
 220 
 221         if isinstance(result['invocation'], dict) and 'content' in result['invocation']:
 222             result['invocation']['content'] = 'CENSORED: content is a no_log parameter'
 223 
 224         return result

 226     def _copy_file(self, source_full, source_rel, content, content_tempfile,
 227                    dest, task_vars, follow):
 228         decrypt = boolean(self._task.args.get('decrypt', True), strict=False)
 229         force = boolean(self._task.args.get('force', 'yes'), strict=False)
 230         raw = boolean(self._task.args.get('raw', 'no'), strict=False)
 231 
 232         result = {}
 233         result['diff'] = []
 234 
 235         # If the local file does not exist, get_real_file() raises AnsibleFileNotFound
 236         try:
 237             source_full = self._loader.get_real_file(source_full, decrypt=decrypt)
 238         except AnsibleFileNotFound as e:
 239             result['failed'] = True
 240             result['msg'] = "could not find src=%s, %s" % (source_full, to_text(e))
 241             return result
 242 
 243         # Get the local mode and set if user wanted it preserved
 244         # https://github.com/ansible/ansible-modules-core/issues/1124
 245         lmode = None
 246         if self._task.args.get('mode', None) == 'preserve':
 247             lmode = '0%03o' % stat.S_IMODE(os.stat(source_full).st_mode)
 248 
 249         # This is kind of optimization - if user told us destination is
 250         # dir, do path manipulation right away, otherwise we still check
 251         # for dest being a dir via remote call below.
 252         if self._connection._shell.path_has_trailing_slash(dest):
 253             dest_file = self._connection._shell.join_path(dest, source_rel)
 254         else:
 255             dest_file = dest
 256 
 257         # Attempt to get remote file info
 258         dest_status = self._execute_remote_stat(dest_file, all_vars=task_vars, follow=follow, checksum=force)
 259 
 260         if dest_status['exists'] and dest_status['isdir']:
 261             # The dest is a directory.
 262             if content is not None:
 263                 # If source was defined as content remove the temporary file and fail out.
 264                 self._remove_tempfile_if_content_defined(content, content_tempfile)
 265                 result['failed'] = True
 266                 result['msg'] = "can not use content with a dir as dest"
 267                 return result
 268             else:
 269                 # Append the relative source location to the destination and get remote stats again
 270                 dest_file = self._connection._shell.join_path(dest, source_rel)
 271                 dest_status = self._execute_remote_stat(dest_file, all_vars=task_vars, follow=follow, checksum=force)
 272 
 273         if dest_status['exists'] and not force:
 274             # remote_file exists so continue to next iteration.
 275             return None
 276 
 277         # Generate a hash of the local file.
 278         local_checksum = checksum(source_full)
 279 
 280         if local_checksum != dest_status['checksum']:
 281             # The checksums don't match and we will change or error out.
 282 
 283             if self._play_context.diff and not raw:
 284                 result['diff'].append(self._get_diff_data(dest_file, source_full, task_vars))
 285 
 286             if self._play_context.check_mode:
 287                 self._remove_tempfile_if_content_defined(content, content_tempfile)
 288                 result['changed'] = True
 289                 return result
 290 
 291             # Define a remote directory that we will copy the file to.
 292             tmp_src = self._connection._shell.join_path(self._connection._shell.tmpdir, 'source')
 293 
 294             remote_path = None
 295 
 296             if not raw:
 297                 remote_path = self._transfer_file(source_full, tmp_src)
 298             else:
 299                 self._transfer_file(source_full, dest_file)
 300 
 301             # We have copied the file remotely and no longer require our content_tempfile
 302             self._remove_tempfile_if_content_defined(content, content_tempfile)
 303             self._loader.cleanup_tmp_file(source_full)
 304 
 305             # FIXME: I don't think this is needed when PIPELINING=0 because the source is created
 306             # world readable.  Access to the directory itself is controlled via fixup_perms2() as
 307             # part of executing the module. Check that umask with scp/sftp/piped doesn't cause
 308             # a problem before acting on this idea. (This idea would save a round-trip)
 309             # fix file permissions when the copy is done as a different user
 310             if remote_path:
 311                 self._fixup_perms2((self._connection._shell.tmpdir, remote_path))
 312 
 313             if raw:
 314                 # Continue to next iteration if raw is defined.
 315                 return None
 316 
 317             # Run the copy module
 318 
 319             # src and dest here come after original and override them
 320             # we pass dest only to make sure it includes trailing slash in case of recursive copy
 321             new_module_args = _create_remote_copy_args(self._task.args)
 322             new_module_args.update(
 323                 dict(
 324                     src=tmp_src,
 325                     dest=dest,
 326                     _original_basename=source_rel,
 327                     follow=follow
 328                 )
 329             )
 330             if not self._task.args.get('checksum'):
 331                 new_module_args['checksum'] = local_checksum
 332 
 333             if lmode:
 334                 new_module_args['mode'] = lmode
 335 
 336             module_return = self._execute_module(module_name='copy', module_args=new_module_args, task_vars=task_vars)
 337 
 338         else:
 339             # no need to transfer the file, already correct hash, but still need to call
 340             # the file module in case we want to change attributes
 341             self._remove_tempfile_if_content_defined(content, content_tempfile)
 342             self._loader.cleanup_tmp_file(source_full)
 343 
 344             if raw:
 345                 return None
 346 
 347             # Fix for https://github.com/ansible/ansible-modules-core/issues/1568.
 348             # If checksums match, and follow = True, find out if 'dest' is a link. If so,
 349             # change it to point to the source of the link.
 350             if follow:
 351                 dest_status_nofollow = self._execute_remote_stat(dest_file, all_vars=task_vars, follow=False)
 352                 if dest_status_nofollow['islnk'] and 'lnk_source' in dest_status_nofollow.keys():
 353                     dest = dest_status_nofollow['lnk_source']
 354 
 355             # Build temporary module_args.
 356             new_module_args = _create_remote_file_args(self._task.args)
 357             new_module_args.update(
 358                 dict(
 359                     dest=dest,
 360                     _original_basename=source_rel,
 361                     recurse=False,
 362                     state='file',
 363                 )
 364             )
 365             # src is sent to the file module in _original_basename, not in src
 366             try:
 367                 del new_module_args['src']
 368             except KeyError:
 369                 pass
 370 
 371             if lmode:
 372                 new_module_args['mode'] = lmode
 373 
 374             # Execute the file module.
 375             module_return = self._execute_module(module_name='file', module_args=new_module_args, task_vars=task_vars)
 376 
 377         if not module_return.get('checksum'):
 378             module_return['checksum'] = local_checksum
 379 
 380         result.update(module_return)
 381         return result

 383     def _create_content_tempfile(self, content):
 384         ''' Create a tempfile containing defined content '''
 385         fd, content_tempfile = tempfile.mkstemp(dir=C.DEFAULT_LOCAL_TMP)
 386         f = os.fdopen(fd, 'wb')
 387         content = to_bytes(content)
 388         try:
 389             f.write(content)
 390         except Exception as err:
 391             os.remove(content_tempfile)
 392             raise Exception(err)
 393         finally:
 394             f.close()
 395         return content_tempfile
 396 
 397     def _remove_tempfile_if_content_defined(self, content, content_tempfile):
 398         if content is not None:
 399             os.remove(content_tempfile)

 401     def run(self, tmp=None, task_vars=None):
 402         ''' handler for file transfer operations '''
 403         if task_vars is None:
 404             task_vars = dict()
 405 
 406         result = super(ActionModule, self).run(tmp, task_vars)
 407         del tmp  # tmp no longer has any effect
 408 
 409         source = self._task.args.get('src', None)
 410         content = self._task.args.get('content', None)
 411         dest = self._task.args.get('dest', None)
 412         remote_src = boolean(self._task.args.get('remote_src', False), strict=False)
 413         local_follow = boolean(self._task.args.get('local_follow', True), strict=False)
 414 
 415         result['failed'] = True
 416         if not source and content is None:
 417             result['msg'] = 'src (or content) is required'
 418         elif not dest:
 419             result['msg'] = 'dest is required'
 420         elif source and content is not None:
 421             result['msg'] = 'src and content are mutually exclusive'
 422         elif content is not None and dest is not None and dest.endswith("/"):
 423             result['msg'] = "can not use content with a dir as dest"
 424         else:
 425             del result['failed']
 426 
 427         if result.get('failed'):
 428             return self._ensure_invocation(result)
 429 
 430         # Define content_tempfile in case we set it after finding content populated.
 431         content_tempfile = None
 432 
 433         # If content is defined make a tmp file and write the content into it.
 434         if content is not None:
 435             try:
 436                 # If content comes to us as a dict it should be decoded json.
 437                 # We need to encode it back into a string to write it out.
 438                 if isinstance(content, dict) or isinstance(content, list):
 439                     content_tempfile = self._create_content_tempfile(json.dumps(content))
 440                 else:
 441                     content_tempfile = self._create_content_tempfile(content)
 442                 source = content_tempfile
 443             except Exception as err:
 444                 result['failed'] = True
 445                 result['msg'] = "could not write content temp file: %s" % to_native(err)
 446                 return self._ensure_invocation(result)
 447 
 448         # if we have first_available_file in our vars
 449         # look up the files and use the first one we find as src
 450         elif remote_src:
 451             result.update(self._execute_module(module_name='copy', task_vars=task_vars))
 452             return self._ensure_invocation(result)
 453         else:
 454             # find_needle returns a path that may not have a trailing slash on
 455             # a directory so we need to determine that now (we use it just
 456             # like rsync does to figure out whether to include the directory
 457             # or only the files inside the directory
 458             trailing_slash = source.endswith(os.path.sep)
 459             try:
 460                 # find in expected paths
 461                 source = self._find_needle('files', source)
 462             except AnsibleError as e:
 463                 result['failed'] = True
 464                 result['msg'] = to_text(e)
 465                 result['exception'] = traceback.format_exc()
 466                 return self._ensure_invocation(result)
 467 
 468             if trailing_slash != source.endswith(os.path.sep):
 469                 if source[-1] == os.path.sep:
 470                     source = source[:-1]
 471                 else:
 472                     source = source + os.path.sep
 473 
 474         # A list of source file tuples (full_path, relative_path) which will try to copy to the destination
 475         source_files = {'files': [], 'directories': [], 'symlinks': []}
 476 
 477         # If source is a directory populate our list else source is a file and translate it to a tuple.
 478         if os.path.isdir(to_bytes(source, errors='surrogate_or_strict')):
 479             # Get a list of the files we want to replicate on the remote side
 480             source_files = _walk_dirs(source, local_follow=local_follow,
 481                                       trailing_slash_detector=self._connection._shell.path_has_trailing_slash)
 482 
 483             # If it's recursive copy, destination is always a dir,
 484             # explicitly mark it so (note - copy module relies on this).
 485             if not self._connection._shell.path_has_trailing_slash(dest):
 486                 dest = self._connection._shell.join_path(dest, '')
 487             # FIXME: Can we optimize cases where there's only one file, no
 488             # symlinks and any number of directories?  In the original code,
 489             # empty directories are not copied....
 490         else:
 491             source_files['files'] = [(source, os.path.basename(source))]
 492 
 493         changed = False
 494         module_return = dict(changed=False)
 495 
 496         # A register for if we executed a module.
 497         # Used to cut down on command calls when not recursive.
 498         module_executed = False
 499 
 500         # expand any user home dir specifier
 501         dest = self._remote_expand_user(dest)
 502 
 503         implicit_directories = set()
 504         for source_full, source_rel in source_files['files']:
 505             # copy files over.  This happens first as directories that have
 506             # a file do not need to be created later
 507 
 508             # We only follow symlinks for files in the non-recursive case
 509             if source_files['directories']:
 510                 follow = False
 511             else:
 512                 follow = boolean(self._task.args.get('follow', False), strict=False)
 513 
 514             module_return = self._copy_file(source_full, source_rel, content, content_tempfile, dest, task_vars, follow)
 515             if module_return is None:
 516                 continue
 517 
 518             if module_return.get('failed'):
 519                 result.update(module_return)
 520                 return self._ensure_invocation(result)
 521 
 522             paths = os.path.split(source_rel)
 523             dir_path = ''
 524             for dir_component in paths:
 525                 os.path.join(dir_path, dir_component)
 526                 implicit_directories.add(dir_path)
 527             if 'diff' in result and not result['diff']:
 528                 del result['diff']
 529             module_executed = True
 530             changed = changed or module_return.get('changed', False)
 531 
 532         for src, dest_path in source_files['directories']:
 533             # Find directories that are leaves as they might not have been
 534             # created yet.
 535             if dest_path in implicit_directories:
 536                 continue
 537 
 538             # Use file module to create these
 539             new_module_args = _create_remote_file_args(self._task.args)
 540             new_module_args['path'] = os.path.join(dest, dest_path)
 541             new_module_args['state'] = 'directory'
 542             new_module_args['mode'] = self._task.args.get('directory_mode', None)
 543             new_module_args['recurse'] = False
 544             del new_module_args['src']
 545 
 546             module_return = self._execute_module(module_name='file', module_args=new_module_args, task_vars=task_vars)
 547 
 548             if module_return.get('failed'):
 549                 result.update(module_return)
 550                 return self._ensure_invocation(result)
 551 
 552             module_executed = True
 553             changed = changed or module_return.get('changed', False)
 554 
 555         for target_path, dest_path in source_files['symlinks']:
 556             # Copy symlinks over
 557             new_module_args = _create_remote_file_args(self._task.args)
 558             new_module_args['path'] = os.path.join(dest, dest_path)
 559             new_module_args['src'] = target_path
 560             new_module_args['state'] = 'link'
 561             new_module_args['force'] = True
 562 
 563             # Only follow remote symlinks in the non-recursive case
 564             if source_files['directories']:
 565                 new_module_args['follow'] = False
 566 
 567             # file module cannot deal with 'preserve' mode and is meaningless
 568             # for symlinks anyway, so just don't pass it.
 569             if new_module_args.get('mode', None) == 'preserve':
 570                 new_module_args.pop('mode')
 571 
 572             module_return = self._execute_module(module_name='file', module_args=new_module_args, task_vars=task_vars)
 573             module_executed = True
 574 
 575             if module_return.get('failed'):
 576                 result.update(module_return)
 577                 return self._ensure_invocation(result)
 578 
 579             changed = changed or module_return.get('changed', False)
 580 
 581         if module_executed and len(source_files['files']) == 1:
 582             result.update(module_return)
 583 
 584             # the file module returns the file path as 'path', but
 585             # the copy module uses 'dest', so add it if it's not there
 586             if 'path' in result and 'dest' not in result:
 587                 result['dest'] = result['path']
 588         else:
 589             result.update(dict(dest=dest, src=source, changed=changed))
 590 
 591         # Delete tmp path
 592         self._remove_tmp_path(self._connection._shell.tmpdir)
 593 
 594         return self._ensure_invocation(result)
