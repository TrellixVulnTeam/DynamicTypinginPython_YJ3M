  21 from ansible.plugins.action import ActionBase

  24 class ActionModule(ActionBase):
  25     ''' Fail with custom message '''
  26 
  27     TRANSFERS_FILES = False
  28     _VALID_ARGS = frozenset(('msg',))
  29 
  30     def run(self, tmp=None, task_vars=None):
  31         if task_vars is None:
  32             task_vars = dict()
  33 
  34         result = super(ActionModule, self).run(tmp, task_vars)
  35         del tmp  # tmp no longer has any effect
  36 
  37         msg = 'Failed as requested from task'
  38         if self._task.args and 'msg' in self._task.args:
  39             msg = self._task.args.get('msg')
  40 
  41         result['failed'] = True
  42         result['msg'] = msg
  43         return result
