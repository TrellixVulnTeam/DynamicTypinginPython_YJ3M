   7 from ansible import constants as C
   8 from ansible.plugins.action import ActionBase
   9 from ansible.utils.vars import merge_hash

  12 class ActionModule(ActionBase):
  13 
  14     def run(self, tmp=None, task_vars=None):
  15         self._supports_async = True
  16         results = super(ActionModule, self).run(tmp, task_vars)
  17         del tmp  # tmp no longer has any effect
  18 
  19         # Command module has a special config option to turn off the command nanny warnings
  20         if 'warn' not in self._task.args and C.COMMAND_WARNINGS:
  21             self._task.args['warn'] = C.COMMAND_WARNINGS
  22 
  23         wrap_async = self._task.async_val and not self._connection.has_native_async
  24         results = merge_hash(results, self._execute_module(task_vars=task_vars, wrap_async=wrap_async))
  25 
  26         if not wrap_async:
  27             # remove a temporary path we created
  28             self._remove_tmp_path(self._connection._shell.tmpdir)
  29 
  30         return results
