   7 from ansible.errors import AnsibleError
   8 from ansible.plugins.action import ActionBase
   9 from ansible.utils.vars import merge_hash
  10 
  11 
  12 class ActionModule(ActionBase):
  13 
  14     _VALID_ARGS = frozenset(('jid', 'mode'))
  15 
  16     def run(self, tmp=None, task_vars=None):
  17         results = super(ActionModule, self).run(tmp, task_vars)
  18         del tmp  # tmp no longer has any effect
  19 
  20         if "jid" not in self._task.args:
  21             raise AnsibleError("jid is required")
  22         jid = self._task.args["jid"]
  23         mode = self._task.args.get("mode", "status")
  24 
  25         env_async_dir = [e for e in self._task.environment if
  26                          "ANSIBLE_ASYNC_DIR" in e]
  27         if len(env_async_dir) > 0:
  28             # for backwards compatibility we need to get the dir from
  29             # ANSIBLE_ASYNC_DIR that is defined in the environment. This is
  30             # deprecated and will be removed in favour of shell options
  31             async_dir = env_async_dir[0]['ANSIBLE_ASYNC_DIR']
  32 
  33             msg = "Setting the async dir from the environment keyword " \
  34                   "ANSIBLE_ASYNC_DIR is deprecated. Set the async_dir " \
  35                   "shell option instead"
  36             self._display.deprecated(msg, "2.12", collection_name='ansible.builtin')
  37         else:
  38             # inject the async directory based on the shell option into the
  39             # module args
  40             async_dir = self.get_shell_option('async_dir', default="~/.ansible_async")
  41 
  42         module_args = dict(jid=jid, mode=mode, _async_dir=async_dir)
  43         status = self._execute_module(task_vars=task_vars,
  44                                       module_args=module_args)
  45         results = merge_hash(results, status)
  46         return results
