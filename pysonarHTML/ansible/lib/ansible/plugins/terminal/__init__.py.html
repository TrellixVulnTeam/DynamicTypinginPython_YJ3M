  30 class TerminalBase(with_metaclass(ABCMeta, object)):
  31     '''
  32     A base class for implementing cli connections
  33 
  34     .. note:: Unlike most of Ansible, nearly all strings in
  35         :class:`TerminalBase` plugins are byte strings.  This is because of
  36         how close to the underlying platform these plugins operate.  Remember
  37         to mark literal strings as byte string (``b"string"``) and to use
  38         :func:`~ansible.module_utils._text.to_bytes` and
  39         :func:`~ansible.module_utils._text.to_text` to avoid unexpected
  40         problems.
  41     '''
  42 
  43     #: compiled bytes regular expressions as stdout
  44     terminal_stdout_re = []
  45 
  46     #: compiled bytes regular expressions as stderr
  47     terminal_stderr_re = []
  48 
  49     #: compiled bytes regular expressions to remove ANSI codes
  50     ansi_re = [
  51         re.compile(br'(\x1b\[\?1h\x1b=)'),
  52         re.compile(br'\x08.')
  53     ]
  54 
  55     #: terminal initial prompt
  56     terminal_initial_prompt = None
  57 
  58     #: terminal initial answer
  59     terminal_initial_answer = None
  60 
  61     #: Send newline after prompt match
  62     terminal_inital_prompt_newline = True
  63 
  64     def __init__(self, connection):
  65         self._connection = connection
  66 
  67     def _exec_cli_command(self, cmd, check_rc=True):
  68         '''
  69         Executes the CLI command on the remote device and returns the output
  70 
  71         :arg cmd: Byte string command to be executed
  72         '''
  73         return self._connection.exec_command(cmd)
  74 
  75     def _get_prompt(self):
  76         """
  77         Returns the current prompt from the device
  78 
  79         :returns: A byte string of the prompt
  80         """
  81         return self._connection.get_prompt()
  82 
  83     def on_open_shell(self):
  84         """Called after the SSH session is established
  85 
  86         This method is called right after the invoke_shell() is called from
  87         the Paramiko SSHClient instance.  It provides an opportunity to setup
  88         terminal parameters such as disbling paging for instance.
  89         """
  90         pass
  91 
  92     def on_close_shell(self):
  93         """Called before the connection is closed
  94 
  95         This method gets called once the connection close has been requested
  96         but before the connection is actually closed.  It provides an
  97         opportunity to clean up any terminal resources before the shell is
  98         actually closed
  99         """
 100         pass
 101 
 102     def on_become(self, passwd=None):
 103         """Called when privilege escalation is requested
 104 
 105         :kwarg passwd: String containing the password
 106 
 107         This method is called when the privilege is requested to be elevated
 108         in the play context by setting become to True.  It is the responsibility
 109         of the terminal plugin to actually do the privilege escalation such
 110         as entering `enable` mode for instance
 111         """
 112         pass
 113 
 114     def on_unbecome(self):
 115         """Called when privilege deescalation is requested
 116 
 117         This method is called when the privilege changed from escalated
 118         (become=True) to non escalated (become=False).  It is the responsibility
 119         of this method to actually perform the deauthorization procedure
 120         """
 121         pass
 122 
 123     def on_authorize(self, passwd=None):
 124         """Deprecated method for privilege escalation
 125 
 126         :kwarg passwd: String containing the password
 127         """
 128         return self.on_become(passwd)
 129 
 130     def on_deauthorize(self):
 131         """Deprecated method for privilege deescalation
 132         """
 133         return self.on_unbecome()
