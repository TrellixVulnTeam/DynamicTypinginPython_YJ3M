  12     # Common options for Ansible.ModuleUtils.WebRequest
  13     DOCUMENTATION = r'''
  14 options:
  15   method:
  16     description:
  17     - The HTTP Method of the request.
  18     type: str
  19   follow_redirects:
  20     description:
  21     - Whether or the module should follow redirects.
  22     - C(all) will follow all redirect.
  23     - C(none) will not follow any redirect.
  24     - C(safe) will follow only "safe" redirects, where "safe" means that the
  25       client is only doing a C(GET) or C(HEAD) on the URI to which it is being
  26       redirected.
  27     - When following a redirected URL, the C(Authorization) header and any
  28       credentials set will be dropped and not redirected.
  29     choices:
  30     - all
  31     - none
  32     - safe
  33     default: safe
  34     type: str
  35   headers:
  36     description:
  37     - Extra headers to set on the request.
  38     - This should be a dictionary where the key is the header name and the
  39       value is the value for that header.
  40     type: dict
  41   http_agent:
  42     description:
  43     - Header to identify as, generally appears in web server logs.
  44     - This is set to the C(User-Agent) header on a HTTP request.
  45     default: ansible-httpget
  46     type: str
  47   maximum_redirection:
  48     description:
  49     - Specify how many times the module will redirect a connection to an
  50       alternative URI before the connection fails.
  51     - If set to C(0) or I(follow_redirects) is set to C(none), or C(safe) when
  52       not doing a C(GET) or C(HEAD) it prevents all redirection.
  53     default: 50
  54     type: int
  55   timeout:
  56     description:
  57     - Specifies how long the request can be pending before it times out (in
  58       seconds).
  59     - Set to C(0) to specify an infinite timeout.
  60     default: 30
  61     type: int
  62   validate_certs:
  63     description:
  64     - If C(no), SSL certificates will not be validated.
  65     - This should only be used on personally controlled sites using self-signed
  66       certificates.
  67     default: yes
  68     type: bool
  69   client_cert:
  70     description:
  71     - The path to the client certificate (.pfx) that is used for X509
  72       authentication. This path can either be the path to the C(pfx) on the
  73       filesystem or the PowerShell certificate path
  74       C(Cert:\CurrentUser\My\<thumbprint>).
  75     - The WinRM connection must be authenticated with C(CredSSP) or C(become)
  76       is used on the task if the certificate file is not password protected.
  77     - Other authentication types can set I(client_cert_password) when the cert
  78       is password protected.
  79     type: str
  80   client_cert_password:
  81     description:
  82     - The password for I(client_cert) if the cert is password protected.
  83     type: str
  84   force_basic_auth:
  85     description:
  86     - By default the authentication header is only sent when a webservice
  87       responses to an initial request with a 401 status. Since some basic auth
  88       services do not properly send a 401, logins will fail.
  89     - This option forces the sending of the Basic authentication header upon
  90       the original request.
  91     default: no
  92     type: bool
  93   url_username:
  94     description:
  95     - The username to use for authentication.
  96     type: str
  97   url_password:
  98     description:
  99     - The password for I(url_username).
 100     type: str
 101   use_default_credential:
 102     description:
 103     - Uses the current user's credentials when authenticating with a server
 104       protected with C(NTLM), C(Kerberos), or C(Negotiate) authentication.
 105     - Sites that use C(Basic) auth will still require explicit credentials
 106       through the I(url_username) and I(url_password) options.
 107     - The module will only have access to the user's credentials if using
 108       C(become) with a password, you are connecting with SSH using a password,
 109       or connecting with WinRM using C(CredSSP) or C(Kerberos with delegation).
 110     - If not using C(become) or a different auth method to the ones stated
 111       above, there will be no default credentials available and no
 112       authentication will occur.
 113     default: no
 114     type: bool
 115   use_proxy:
 116     description:
 117     - If C(no), it will not use the proxy defined in IE for the current user.
 118     default: yes
 119     type: bool
 120   proxy_url:
 121     description:
 122     - An explicit proxy to use for the request.
 123     - By default, the request will use the IE defined proxy unless I(use_proxy)
 124       is set to C(no).
 125     type: str
 126   proxy_username:
 127     description:
 128     - The username to use for proxy authentication.
 129     type: str
 130   proxy_password:
 131     description:
 132     - The password for I(proxy_username).
 133     type: str
 134   proxy_use_default_credential:
 135     description:
 136     - Uses the current user's credentials when authenticating with a proxy host
 137       protected with C(NTLM), C(Kerberos), or C(Negotiate) authentication.
 138     - Proxies that use C(Basic) auth will still require explicit credentials
 139       through the I(proxy_username) and I(proxy_password) options.
 140     - The module will only have access to the user's credentials if using
 141       C(become) with a password, you are connecting with SSH using a password,
 142       or connecting with WinRM using C(CredSSP) or C(Kerberos with delegation).
 143     - If not using C(become) or a different auth method to the ones stated
 144       above, there will be no default credentials available and no proxy
 145       authentication will occur.
 146     default: no
 147     type: bool
 148 seealso:
 149 - module: community.windows.win_inet_proxy
