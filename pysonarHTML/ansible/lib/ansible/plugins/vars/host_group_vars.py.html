  55 import os
  56 from ansible import constants as C
  57 from ansible.errors import AnsibleParserError
  58 from ansible.module_utils._text import to_bytes, to_native, to_text
  59 from ansible.plugins.vars import BaseVarsPlugin
  60 from ansible.inventory.host import Host
  61 from ansible.inventory.group import Group
  62 from ansible.utils.vars import combine_vars
  63 
  64 FOUND = {}
  65 
  66 
  67 class VarsModule(BaseVarsPlugin):
  68 
  69     REQUIRES_WHITELIST = True
  70 
  71     def get_vars(self, loader, path, entities, cache=True):
  72         ''' parses the inventory file '''
  73 
  74         if not isinstance(entities, list):
  75             entities = [entities]
  76 
  77         super(VarsModule, self).get_vars(loader, path, entities)
  78 
  79         data = {}
  80         for entity in entities:
  81             if isinstance(entity, Host):
  82                 subdir = 'host_vars'
  83             elif isinstance(entity, Group):
  84                 subdir = 'group_vars'
  85             else:
  86                 raise AnsibleParserError("Supplied entity must be Host or Group, got %s instead" % (type(entity)))
  87 
  88             # avoid 'chroot' type inventory hostnames /path/to/chroot
  89             if not entity.name.startswith(os.path.sep):
  90                 try:
  91                     found_files = []
  92                     # load vars
  93                     b_opath = os.path.realpath(to_bytes(os.path.join(self._basedir, subdir)))
  94                     opath = to_text(b_opath)
  95                     key = '%s.%s' % (entity.name, opath)
  96                     if cache and key in FOUND:
  97                         found_files = FOUND[key]
  98                     else:
  99                         # no need to do much if path does not exist for basedir
 100                         if os.path.exists(b_opath):
 101                             if os.path.isdir(b_opath):
 102                                 self._display.debug("\tprocessing dir %s" % opath)
 103                                 found_files = loader.find_vars_files(opath, entity.name)
 104                                 FOUND[key] = found_files
 105                             else:
 106                                 self._display.warning("Found %s that is not a directory, skipping: %s" % (subdir, opath))
 107 
 108                     for found in found_files:
 109                         new_data = loader.load_from_file(found, cache=True, unsafe=True)
 110                         if new_data:  # ignore empty files
 111                             data = combine_vars(data, new_data)
 112 
 113                 except Exception as e:
 114                     raise AnsibleParserError(to_native(e))
 115         return data
