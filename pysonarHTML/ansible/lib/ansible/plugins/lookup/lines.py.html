  42 import subprocess
  43 from ansible.errors import AnsibleError
  44 from ansible.plugins.lookup import LookupBase
  45 from ansible.module_utils._text import to_text
  46 
  47 
  48 class LookupModule(LookupBase):
  49 
  50     def run(self, terms, variables, **kwargs):
  51 
  52         ret = []
  53         for term in terms:
  54             p = subprocess.Popen(term, cwd=self._loader.get_basedir(), shell=True, stdin=subprocess.PIPE, stdout=subprocess.PIPE)
  55             (stdout, stderr) = p.communicate()
  56             if p.returncode == 0:
  57                 ret.extend([to_text(l) for l in stdout.splitlines()])
  58             else:
  59                 raise AnsibleError("lookup_plugin.lines(%s) returned %d" % (term, p.returncode))
  60         return ret
