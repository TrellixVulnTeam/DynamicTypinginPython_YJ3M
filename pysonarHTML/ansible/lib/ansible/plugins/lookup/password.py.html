 109 DEFAULT_LENGTH = 20
 110 VALID_PARAMS = frozenset(('length', 'encrypt', 'chars'))

 113 def _parse_parameters(term):
 114     """Hacky parsing of params
 115 
 116     See https://github.com/ansible/ansible-modules-core/issues/1968#issuecomment-136842156
 117     and the first_found lookup For how we want to fix this later
 118     """
 119     first_split = term.split(' ', 1)
 120     if len(first_split) <= 1:
 121         # Only a single argument given, therefore it's a path
 122         relpath = term
 123         params = dict()
 124     else:
 125         relpath = first_split[0]
 126         params = parse_kv(first_split[1])
 127         if '_raw_params' in params:
 128             # Spaces in the path?
 129             relpath = u' '.join((relpath, params['_raw_params']))
 130             del params['_raw_params']
 131 
 132             # Check that we parsed the params correctly
 133             if not term.startswith(relpath):
 134                 # Likely, the user had a non parameter following a parameter.
 135                 # Reject this as a user typo
 136                 raise AnsibleError('Unrecognized value after key=value parameters given to password lookup')
 137         # No _raw_params means we already found the complete path when
 138         # we split it initially
 139 
 140     # Check for invalid parameters.  Probably a user typo
 141     invalid_params = frozenset(params.keys()).difference(VALID_PARAMS)
 142     if invalid_params:
 143         raise AnsibleError('Unrecognized parameter(s) given to password lookup: %s' % ', '.join(invalid_params))
 144 
 145     # Set defaults
 146     params['length'] = int(params.get('length', DEFAULT_LENGTH))
 147     params['encrypt'] = params.get('encrypt', None)
 148 
 149     params['chars'] = params.get('chars', None)
 150     if params['chars']:
 151         tmp_chars = []
 152         if u',,' in params['chars']:
 153             tmp_chars.append(u',')
 154         tmp_chars.extend(c for c in params['chars'].replace(u',,', u',').split(u',') if c)
 155         params['chars'] = tmp_chars
 156     else:
 157         # Default chars for password
 158         params['chars'] = [u'ascii_letters', u'digits', u".,:-_"]
 159 
 160     return relpath, params

 163 def _read_password_file(b_path):
 164     """Read the contents of a password file and return it
 165     :arg b_path: A byte string containing the path to the password file
 166     :returns: a text string containing the contents of the password file or
 167         None if no password file was present.
 168     """
 169     content = None
 170 
 171     if os.path.exists(b_path):
 172         with open(b_path, 'rb') as f:
 173             b_content = f.read().rstrip()
 174         content = to_text(b_content, errors='surrogate_or_strict')
 175 
 176     return content

 179 def _gen_candidate_chars(characters):
 180     '''Generate a string containing all valid chars as defined by ``characters``
 181 
 182     :arg characters: A list of character specs. The character specs are
 183         shorthand names for sets of characters like 'digits', 'ascii_letters',
 184         or 'punctuation' or a string to be included verbatim.
 185 
 186     The values of each char spec can be:
 187 
 188     * a name of an attribute in the 'strings' module ('digits' for example).
 189       The value of the attribute will be added to the candidate chars.
 190     * a string of characters. If the string isn't an attribute in 'string'
 191       module, the string will be directly added to the candidate chars.
 192 
 193     For example::
 194 
 195         characters=['digits', '?|']``
 196 
 197     will match ``string.digits`` and add all ascii digits.  ``'?|'`` will add
 198     the question mark and pipe characters directly. Return will be the string::
 199 
 200         u'0123456789?|'
 201     '''
 202     chars = []
 203     for chars_spec in characters:
 204         # getattr from string expands things like "ascii_letters" and "digits"
 205         # into a set of characters.
 206         chars.append(to_text(getattr(string, to_native(chars_spec), chars_spec),
 207                      errors='strict'))
 208     chars = u''.join(chars).replace(u'"', u'').replace(u"'", u'')
 209     return chars

 212 def _parse_content(content):
 213     '''parse our password data format into password and salt
 214 
 215     :arg content: The data read from the file
 216     :returns: password and salt
 217     '''
 218     password = content
 219     salt = None
 220 
 221     salt_slug = u' salt='
 222     try:
 223         sep = content.rindex(salt_slug)
 224     except ValueError:
 225         # No salt
 226         pass
 227     else:
 228         salt = password[sep + len(salt_slug):]
 229         password = content[:sep]
 230 
 231     return password, salt

 234 def _format_content(password, salt, encrypt=None):
 235     """Format the password and salt for saving
 236     :arg password: the plaintext password to save
 237     :arg salt: the salt to use when encrypting a password
 238     :arg encrypt: Which method the user requests that this password is encrypted.
 239         Note that the password is saved in clear.  Encrypt just tells us if we
 240         must save the salt value for idempotence.  Defaults to None.
 241     :returns: a text string containing the formatted information
 242 
 243     .. warning:: Passwords are saved in clear.  This is because the playbooks
 244         expect to get cleartext passwords from this lookup.
 245     """
 246     if not encrypt and not salt:
 247         return password
 248 
 249     # At this point, the calling code should have assured us that there is a salt value.
 250     if not salt:
 251         raise AnsibleAssertionError('_format_content was called with encryption requested but no salt value')
 252 
 253     return u'%s salt=%s' % (password, salt)

 256 def _write_password_file(b_path, content):
 257     b_pathdir = os.path.dirname(b_path)
 258     makedirs_safe(b_pathdir, mode=0o700)
 259 
 260     with open(b_path, 'wb') as f:
 261         os.chmod(b_path, 0o600)
 262         b_content = to_bytes(content, errors='surrogate_or_strict') + b'\n'
 263         f.write(b_content)

 266 def _get_lock(b_path):
 267     """Get the lock for writing password file."""
 268     first_process = False
 269     b_pathdir = os.path.dirname(b_path)
 270     lockfile_name = to_bytes("%s.ansible_lockfile" % hashlib.sha1(b_path).hexdigest())
 271     lockfile = os.path.join(b_pathdir, lockfile_name)
 272     if not os.path.exists(lockfile) and b_path != to_bytes('/dev/null'):
 273         try:
 274             makedirs_safe(b_pathdir, mode=0o700)
 275             fd = os.open(lockfile, os.O_CREAT | os.O_EXCL)
 276             os.close(fd)
 277             first_process = True
 278         except OSError as e:
 279             if e.strerror != 'File exists':
 280                 raise
 281 
 282     counter = 0
 283     # if the lock is got by other process, wait until it's released
 284     while os.path.exists(lockfile) and not first_process:
 285         time.sleep(2 ** counter)
 286         if counter >= 2:
 287             raise AnsibleError("Password lookup cannot get the lock in 7 seconds, abort..."
 288                                "This may caused by un-removed lockfile"
 289                                "you can manually remove it from controller machine at %s and try again" % lockfile)
 290         counter += 1
 291     return first_process, lockfile
 292

 294 def _release_lock(lockfile):
 295     """Release the lock so other processes can read the password file."""
 296     if os.path.exists(lockfile):
 297         os.remove(lockfile)

 300 class LookupModule(LookupBase):
 301     def run(self, terms, variables, **kwargs):
 302         ret = []
 303 
 304         for term in terms:
 305             relpath, params = _parse_parameters(term)
 306             path = self._loader.path_dwim(relpath)
 307             b_path = to_bytes(path, errors='surrogate_or_strict')
 308             chars = _gen_candidate_chars(params['chars'])
 309 
 310             changed = None
 311             # make sure only one process finishes all the job first
 312             first_process, lockfile = _get_lock(b_path)
 313 
 314             content = _read_password_file(b_path)
 315 
 316             if content is None or b_path == to_bytes('/dev/null'):
 317                 plaintext_password = random_password(params['length'], chars)
 318                 salt = None
 319                 changed = True
 320             else:
 321                 plaintext_password, salt = _parse_content(content)
 322 
 323             if params['encrypt'] and not salt:
 324                 changed = True
 325                 salt = random_salt()
 326 
 327             if changed and b_path != to_bytes('/dev/null'):
 328                 content = _format_content(plaintext_password, salt, encrypt=params['encrypt'])
 329                 _write_password_file(b_path, content)
 330 
 331             if first_process:
 332                 # let other processes continue
 333                 _release_lock(lockfile)
 334 
 335             if params['encrypt']:
 336                 password = do_encrypt(plaintext_password, params['encrypt'], salt=salt)
 337                 ret.append(password)
 338             else:
 339                 ret.append(plaintext_password)
 340 
 341         return ret
