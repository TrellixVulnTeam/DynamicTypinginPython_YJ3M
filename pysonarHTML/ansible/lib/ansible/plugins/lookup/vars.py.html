  68 from ansible.errors import AnsibleError, AnsibleUndefinedVariable
  69 from ansible.module_utils.six import string_types
  70 from ansible.plugins.lookup import LookupBase
  71 
  72 
  73 class LookupModule(LookupBase):
  74 
  75     def run(self, terms, variables=None, **kwargs):
  76         if variables is not None:
  77             self._templar.available_variables = variables
  78         myvars = getattr(self._templar, '_available_variables', {})
  79 
  80         self.set_options(direct=kwargs)
  81         default = self.get_option('default')
  82 
  83         ret = []
  84         for term in terms:
  85             if not isinstance(term, string_types):
  86                 raise AnsibleError('Invalid setting identifier, "%s" is not a string, its a %s' % (term, type(term)))
  87 
  88             try:
  89                 try:
  90                     value = myvars[term]
  91                 except KeyError:
  92                     try:
  93                         value = myvars['hostvars'][myvars['inventory_hostname']][term]
  94                     except KeyError:
  95                         raise AnsibleUndefinedVariable('No variable found with this name: %s' % term)
  96 
  97                 ret.append(self._templar.template(value, fail_on_undefined=True))
  98             except AnsibleUndefinedVariable:
  99                 if default is not None:
 100                     ret.append(default)
 101                 else:
 102                     raise
 103 
 104         return ret
