  58 from ansible import constants as C
  59 from ansible.errors import AnsibleError
  60 from ansible.module_utils.six import string_types
  61 from ansible.plugins.lookup import LookupBase
  62 
  63 
  64 class LookupModule(LookupBase):
  65 
  66     def run(self, terms, variables=None, **kwargs):
  67 
  68         missing = kwargs.get('on_missing', 'error').lower()
  69         if not isinstance(missing, string_types) or missing not in ['error', 'warn', 'skip']:
  70             raise AnsibleError('"on_missing" must be a string and one of "error", "warn" or "skip", not %s' % missing)
  71 
  72         ret = []
  73         for term in terms:
  74             if not isinstance(term, string_types):
  75                 raise AnsibleError('Invalid setting identifier, "%s" is not a string, its a %s' % (term, type(term)))
  76             try:
  77                 result = getattr(C, term)
  78                 if callable(result):
  79                     raise AnsibleError('Invalid setting "%s" attempted' % term)
  80                 ret.append(result)
  81             except AttributeError:
  82                 if missing == 'error':
  83                     raise AnsibleError('Unable to find setting %s' % term)
  84                 elif missing == 'warn':
  85                     self._display.warning('Skipping, did not find setting %s' % term)
  86         return ret
