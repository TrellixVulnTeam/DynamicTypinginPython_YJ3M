  77 class CSVRecoder:
  78     """
  79     Iterator that reads an encoded stream and reencodes the input to UTF-8
  80     """
  81     def __init__(self, f, encoding='utf-8'):
  82         self.reader = codecs.getreader(encoding)(f)
  83 
  84     def __iter__(self):
  85         return self
  86 
  87     def __next__(self):
  88         return next(self.reader).encode("utf-8")
  89 
  90     next = __next__   # For Python 2
  91 
  92 
  93 class CSVReader:
  94     """
  95     A CSV reader which will iterate over lines in the CSV file "f",
  96     which is encoded in the given encoding.
  97     """
  98 
  99     def __init__(self, f, dialect=csv.excel, encoding='utf-8', **kwds):
 100         if PY2:
 101             f = CSVRecoder(f, encoding)
 102         else:
 103             f = codecs.getreader(encoding)(f)
 104 
 105         self.reader = csv.reader(f, dialect=dialect, **kwds)
 106 
 107     def __next__(self):
 108         row = next(self.reader)
 109         return [to_text(s) for s in row]
 110 
 111     next = __next__  # For Python 2
 112 
 113     def __iter__(self):
 114         return self

 117 class LookupModule(LookupBase):
 118 
 119     def read_csv(self, filename, key, delimiter, encoding='utf-8', dflt=None, col=1):
 120 
 121         try:
 122             f = open(filename, 'rb')
 123             creader = CSVReader(f, delimiter=to_native(delimiter), encoding=encoding)
 124 
 125             for row in creader:
 126                 if len(row) and row[0] == key:
 127                     return row[int(col)]
 128         except Exception as e:
 129             raise AnsibleError("csvfile: %s" % to_native(e))
 130 
 131         return dflt
 132 
 133     def run(self, terms, variables=None, **kwargs):
 134 
 135         ret = []
 136 
 137         for term in terms:
 138             kv = parse_kv(term)
 139 
 140             if '_raw_params' not in kv:
 141                 raise AnsibleError('Search key is required but was not found')
 142 
 143             key = kv['_raw_params']
 144 
 145             paramvals = {
 146                 'col': "1",          # column to return
 147                 'default': None,
 148                 'delimiter': "TAB",
 149                 'file': 'ansible.csv',
 150                 'encoding': 'utf-8',
 151             }
 152 
 153             # parameters specified?
 154             try:
 155                 for name, value in kv.items():
 156                     if name == '_raw_params':
 157                         continue
 158                     if name not in paramvals:
 159                         raise AnsibleAssertionError('%s not in paramvals' % name)
 160                     paramvals[name] = value
 161             except (ValueError, AssertionError) as e:
 162                 raise AnsibleError(e)
 163 
 164             if paramvals['delimiter'] == 'TAB':
 165                 paramvals['delimiter'] = "\t"
 166 
 167             lookupfile = self.find_file_in_search_path(variables, 'files', paramvals['file'])
 168             var = self.read_csv(lookupfile, key, paramvals['delimiter'], paramvals['encoding'], paramvals['default'], paramvals['col'])
 169             if var is not None:
 170                 if isinstance(var, MutableSequence):
 171                     for v in var:
 172                         ret.append(v)
 173                 else:
 174                     ret.append(var)
 175         return ret
