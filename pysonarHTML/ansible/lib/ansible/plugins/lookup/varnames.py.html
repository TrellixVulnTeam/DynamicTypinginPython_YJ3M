  48 from ansible.errors import AnsibleError
  49 from ansible.module_utils._text import to_native

  54 class LookupModule(LookupBase):
  55 
  56     def run(self, terms, variables=None, **kwargs):
  57 
  58         if variables is None:
  59             raise AnsibleError('No variables available to search')
  60 
  61         # no options, yet
  62         # self.set_options(direct=kwargs)
  63 
  64         ret = []
  65         variable_names = list(variables.keys())
  66         for term in terms:
  67 
  68             if not isinstance(term, string_types):
  69                 raise AnsibleError('Invalid setting identifier, "%s" is not a string, it is a %s' % (term, type(term)))
  70 
  71             try:
  72                 name = re.compile(term)
  73             except Exception as e:
  74                 raise AnsibleError('Unable to use "%s" as a search parameter: %s' % (term, to_native(e)))
  75 
  76             for varname in variable_names:
  77                 if name.search(varname):
  78                     ret.append(varname)
  79 
  80         return ret
