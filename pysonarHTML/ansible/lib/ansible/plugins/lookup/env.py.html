  48 from ansible.plugins.lookup import LookupBase
  49 from ansible.utils import py3compat
  50 
  51 
  52 class LookupModule(LookupBase):
  53     def run(self, terms, variables, **kwargs):
  54         ret = []
  55 
  56         for term in terms:
  57             var = term.split()[0]
  58             ret.append(py3compat.environ.get(var, ''))
  59 
  60         return ret
