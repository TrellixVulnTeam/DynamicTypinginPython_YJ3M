  87 # shortcut format
  88 NUM = "(0?x?[0-9a-f]+)"
  89 SHORTCUT = re_compile(
  90     "^(" +        # Group 0
  91     NUM +         # Group 1: Start
  92     "-)?" +
  93     NUM +         # Group 2: End
  94     "(/" +        # Group 3
  95     NUM +         # Group 4: Stride
  96     ")?" +
  97     "(:(.+))?$",  # Group 5, Group 6: Format String
  98     IGNORECASE
  99 )

 102 class LookupModule(LookupBase):
 103     """
 104     sequence lookup module
 105 
 106     Used to generate some sequence of items. Takes arguments in two forms.
 107 
 108     The simple / shortcut form is:
 109 
 110       [start-]end[/stride][:format]
 111 
 112     As indicated by the brackets: start, stride, and format string are all
 113     optional.  The format string is in the style of printf.  This can be used
 114     to pad with zeros, format in hexadecimal, etc.  All of the numerical values
 115     can be specified in octal (i.e. 0664) or hexadecimal (i.e. 0x3f8).
 116     Negative numbers are not supported.
 117 
 118     Some examples:
 119 
 120       5 -> ["1","2","3","4","5"]
 121       5-8 -> ["5", "6", "7", "8"]
 122       2-10/2 -> ["2", "4", "6", "8", "10"]
 123       4:host%02d -> ["host01","host02","host03","host04"]
 124 
 125     The standard Ansible key-value form is accepted as well.  For example:
 126 
 127       start=5 end=11 stride=2 format=0x%02x -> ["0x05","0x07","0x09","0x0a"]
 128 
 129     This format takes an alternate form of "end" called "count", which counts
 130     some number from the starting value.  For example:
 131 
 132       count=5 -> ["1", "2", "3", "4", "5"]
 133       start=0x0f00 count=4 format=%04x -> ["0f00", "0f01", "0f02", "0f03"]
 134       start=0 count=5 stride=2 -> ["0", "2", "4", "6", "8"]
 135       start=1 count=5 stride=2 -> ["1", "3", "5", "7", "9"]
 136 
 137     The count option is mostly useful for avoiding off-by-one errors and errors
 138     calculating the number of entries in a sequence when a stride is specified.
 139     """
 140 
 141     def reset(self):
 142         """set sensible defaults"""
 143         self.start = 1
 144         self.count = None
 145         self.end = None
 146         self.stride = 1
 147         self.format = "%d"
 148 
 149     def parse_kv_args(self, args):
 150         """parse key-value style arguments"""
 151         for arg in ["start", "end", "count", "stride"]:
 152             try:
 153                 arg_raw = args.pop(arg, None)
 154                 if arg_raw is None:
 155                     continue
 156                 arg_cooked = int(arg_raw, 0)
 157                 setattr(self, arg, arg_cooked)
 158             except ValueError:
 159                 raise AnsibleError(
 160                     "can't parse arg %s=%r as integer"
 161                     % (arg, arg_raw)
 162                 )
 163         if 'format' in args:
 164             self.format = args.pop("format")
 165         if args:
 166             raise AnsibleError(
 167                 "unrecognized arguments to with_sequence: %r"
 168                 % args.keys()
 169             )
 170 
 171     def parse_simple_args(self, term):
 172         """parse the shortcut forms, return True/False"""
 173         match = SHORTCUT.match(term)
 174         if not match:
 175             return False
 176 
 177         _, start, end, _, stride, _, format = match.groups()
 178 
 179         if start is not None:
 180             try:
 181                 start = int(start, 0)
 182             except ValueError:
 183                 raise AnsibleError("can't parse start=%s as integer" % start)
 184         if end is not None:
 185             try:
 186                 end = int(end, 0)
 187             except ValueError:
 188                 raise AnsibleError("can't parse end=%s as integer" % end)
 189         if stride is not None:
 190             try:
 191                 stride = int(stride, 0)
 192             except ValueError:
 193                 raise AnsibleError("can't parse stride=%s as integer" % stride)
 194 
 195         if start is not None:
 196             self.start = start
 197         if end is not None:
 198             self.end = end
 199         if stride is not None:
 200             self.stride = stride
 201         if format is not None:
 202             self.format = format
 203 
 204         return True
 205 
 206     def sanity_check(self):
 207         if self.count is None and self.end is None:
 208             raise AnsibleError("must specify count or end in with_sequence")
 209         elif self.count is not None and self.end is not None:
 210             raise AnsibleError("can't specify both count and end in with_sequence")
 211         elif self.count is not None:
 212             # convert count to end
 213             if self.count != 0:
 214                 self.end = self.start + self.count * self.stride - 1
 215             else:
 216                 self.start = 0
 217                 self.end = 0
 218                 self.stride = 0
 219             del self.count
 220         if self.stride > 0 and self.end < self.start:
 221             raise AnsibleError("to count backwards make stride negative")
 222         if self.stride < 0 and self.end > self.start:
 223             raise AnsibleError("to count forward don't make stride negative")
 224         if self.format.count('%') != 1:
 225             raise AnsibleError("bad formatting string: %s" % self.format)
 226 
 227     def generate_sequence(self):
 228         if self.stride >= 0:
 229             adjust = 1
 230         else:
 231             adjust = -1
 232         numbers = xrange(self.start, self.end + adjust, self.stride)
 233 
 234         for i in numbers:
 235             try:
 236                 formatted = self.format % i
 237                 yield formatted
 238             except (ValueError, TypeError):
 239                 raise AnsibleError(
 240                     "problem formatting %r with %r" % (i, self.format)
 241                 )
 242 
 243     def run(self, terms, variables, **kwargs):
 244         results = []
 245 
 246         for term in terms:
 247             try:
 248                 self.reset()  # clear out things for this iteration
 249                 try:
 250                     if not self.parse_simple_args(term):
 251                         self.parse_kv_args(parse_kv(term))
 252                 except AnsibleError:
 253                     raise
 254                 except Exception as e:
 255                     raise AnsibleError("unknown error parsing with_sequence arguments: %r. Error was: %s" % (term, e))
 256 
 257                 self.sanity_check()
 258                 if self.stride != 0:
 259                     results.extend(self.generate_sequence())
 260             except AnsibleError:
 261                 raise
 262             except Exception as e:
 263                 raise AnsibleError(
 264                     "unknown error generating sequence: %s" % e
 265                 )
 266 
 267         return results
