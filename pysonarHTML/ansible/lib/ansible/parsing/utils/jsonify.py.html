  25 def jsonify(result, format=False):
  26     ''' format JSON output (uncompressed or uncompressed) '''
  27 
  28     if result is None:
  29         return "{}"
  30 
  31     indent = None
  32     if format:
  33         indent = 4
  34 
  35     try:
  36         return json.dumps(result, sort_keys=True, indent=indent, ensure_ascii=False)
  37     except UnicodeDecodeError:
  38         return json.dumps(result, sort_keys=True, indent=indent)
