  34 class DataLoader:
  35 
  36     '''
  37     The DataLoader class is used to load and parse YAML or JSON content,
  38     either from a given file name or from a string that was previously
  39     read in through other means. A Vault password can be specified, and
  40     any vault-encrypted files will be decrypted.
  41 
  42     Data read from files will also be cached, so the file will never be
  43     read from disk more than once.
  44 
  45     Usage:
  46 
  47         dl = DataLoader()
  48         # optionally: dl.set_vault_password('foo')
  49         ds = dl.load('...')
  50         ds = dl.load_from_file('/path/to/file')
  51     '''
  52 
  53     def __init__(self):
  54 
  55         self._basedir = '.'
  56 
  57         # NOTE: not effective with forks as the main copy does not get updated.
  58         # avoids rereading files
  59         self._FILE_CACHE = dict()
  60 
  61         # NOTE: not thread safe, also issues with forks not returning data to main proc
  62         #       so they need to be cleaned independently. See WorkerProcess for example.
  63         # used to keep track of temp files for cleaning
  64         self._tempfiles = set()
  65 
  66         # initialize the vault stuff with an empty password
  67         # TODO: replace with a ref to something that can get the password
  68         #       a creds/auth provider
  69         # self.set_vault_password(None)
  70         self._vaults = {}
  71         self._vault = VaultLib()
  72         self.set_vault_secrets(None)

  78     def load(self, data, file_name='<string>', show_content=True, json_only=False):
  79         '''Backwards compat for now'''
  80         return from_yaml(data, file_name, show_content, self._vault.secrets, json_only=json_only)

  82     def load_from_file(self, file_name, cache=True, unsafe=False, json_only=False):
  83         ''' Loads data from a file, which can contain either JSON or YAML.  '''
  84 
  85         file_name = self.path_dwim(file_name)
  86         display.debug("Loading data from %s" % file_name)
  87 
  88         # if the file has already been read in and cached, we'll
  89         # return those results to avoid more file/vault operations
  90         if cache and file_name in self._FILE_CACHE:
  91             parsed_data = self._FILE_CACHE[file_name]
  92         else:
  93             # read the file contents and load the data structure from them
  94             (b_file_data, show_content) = self._get_file_contents(file_name)
  95 
  96             file_data = to_text(b_file_data, errors='surrogate_or_strict')
  97             parsed_data = self.load(data=file_data, file_name=file_name, show_content=show_content, json_only=json_only)
  98 
  99             # cache the file contents for next time
 100             self._FILE_CACHE[file_name] = parsed_data
 101 
 102         if unsafe:
 103             return parsed_data
 104         else:
 105             # return a deep copy here, so the cache is not affected
 106             return copy.deepcopy(parsed_data)

 129     def _decrypt_if_vault_data(self, b_vault_data, b_file_name=None):
 130         '''Decrypt b_vault_data if encrypted and return b_data and the show_content flag'''
 131 
 132         if not is_encrypted(b_vault_data):
 133             show_content = True
 134             return b_vault_data, show_content
 135 
 136         b_ciphertext, b_version, cipher_name, vault_id = parse_vaulttext_envelope(b_vault_data)
 137         b_data = self._vault.decrypt(b_vault_data, filename=b_file_name)
 138 
 139         show_content = False
 140         return b_data, show_content

 142     def _get_file_contents(self, file_name):
 143         '''
 144         Reads the file contents from the given file name
 145 
 146         If the contents are vault-encrypted, it will decrypt them and return
 147         the decrypted data
 148 
 149         :arg file_name: The name of the file to read.  If this is a relative
 150             path, it will be expanded relative to the basedir
 151         :raises AnsibleFileNotFound: if the file_name does not refer to a file
 152         :raises AnsibleParserError: if we were unable to read the file
 153         :return: Returns a byte string of the file contents
 154         '''
 155         if not file_name or not isinstance(file_name, (binary_type, text_type)):
 156             raise AnsibleParserError("Invalid filename: '%s'" % to_native(file_name))
 157 
 158         b_file_name = to_bytes(self.path_dwim(file_name))
 159         # This is what we really want but have to fix unittests to make it pass
 160         # if not os.path.exists(b_file_name) or not os.path.isfile(b_file_name):
 161         if not self.path_exists(b_file_name):
 162             raise AnsibleFileNotFound("Unable to retrieve file contents", file_name=file_name)
 163 
 164         try:
 165             with open(b_file_name, 'rb') as f:
 166                 data = f.read()
 167                 return self._decrypt_if_vault_data(data, b_file_name)
 168         except (IOError, OSError) as e:
 169             raise AnsibleParserError("an error occurred while trying to read the file '%s': %s" % (file_name, to_native(e)), orig_exc=e)

 181     def path_dwim(self, given):
 182         '''
 183         make relative paths work like folks expect.
 184         '''
 185 
 186         given = unquote(given)
 187         given = to_text(given, errors='surrogate_or_strict')
 188 
 189         if given.startswith(to_text(os.path.sep)) or given.startswith(u'~'):
 190             path = given
 191         else:
 192             basedir = to_text(self._basedir, errors='surrogate_or_strict')
 193             path = os.path.join(basedir, given)
 194 
 195         return unfrackpath(path, follow=False)

 197     def _is_role(self, path):
 198         ''' imperfect role detection, roles are still valid w/o tasks|meta/main.yml|yaml|etc '''
 199 
 200         b_path = to_bytes(path, errors='surrogate_or_strict')
 201         b_upath = to_bytes(unfrackpath(path, follow=False), errors='surrogate_or_strict')
 202 
 203         for b_finddir in (b'meta', b'tasks'):
 204             for b_suffix in (b'.yml', b'.yaml', b''):
 205                 b_main = b'main%s' % (b_suffix)
 206                 b_tasked = os.path.join(b_finddir, b_main)
 207 
 208                 if (
 209                     RE_TASKS.search(path) and
 210                     os.path.exists(os.path.join(b_path, b_main)) or
 211                     os.path.exists(os.path.join(b_upath, b_tasked)) or
 212                     os.path.exists(os.path.join(os.path.dirname(b_path), b_tasked))
 213                 ):
 214                     return True
 215         return False

 217     def path_dwim_relative(self, path, dirname, source, is_role=False):
 218         '''
 219         find one file in either a role or playbook dir with or without
 220         explicitly named dirname subdirs
 221 
 222         Used in action plugins and lookups to find supplemental files that
 223         could be in either place.
 224         '''
 225 
 226         search = []
 227         source = to_text(source, errors='surrogate_or_strict')
 228 
 229         # I have full path, nothing else needs to be looked at
 230         if source.startswith(to_text(os.path.sep)) or source.startswith(u'~'):
 231             search.append(unfrackpath(source, follow=False))
 232         else:
 233             # base role/play path + templates/files/vars + relative filename
 234             search.append(os.path.join(path, dirname, source))
 235             basedir = unfrackpath(path, follow=False)
 236 
 237             # not told if role, but detect if it is a role and if so make sure you get correct base path
 238             if not is_role:
 239                 is_role = self._is_role(path)
 240 
 241             if is_role and RE_TASKS.search(path):
 242                 basedir = unfrackpath(os.path.dirname(path), follow=False)
 243 
 244             cur_basedir = self._basedir
 245             self.set_basedir(basedir)
 246             # resolved base role/play path + templates/files/vars + relative filename
 247             search.append(unfrackpath(os.path.join(basedir, dirname, source), follow=False))
 248             self.set_basedir(cur_basedir)
 249 
 250             if is_role and not source.endswith(dirname):
 251                 # look in role's tasks dir w/o dirname
 252                 search.append(unfrackpath(os.path.join(basedir, 'tasks', source), follow=False))
 253 
 254             # try to create absolute path for loader basedir + templates/files/vars + filename
 255             search.append(unfrackpath(os.path.join(dirname, source), follow=False))
 256 
 257             # try to create absolute path for loader basedir
 258             search.append(unfrackpath(os.path.join(basedir, source), follow=False))
 259 
 260             # try to create absolute path for  dirname + filename
 261             search.append(self.path_dwim(os.path.join(dirname, source)))
 262 
 263             # try to create absolute path for filename
 264             search.append(self.path_dwim(source))
 265 
 266         for candidate in search:
 267             if os.path.exists(to_bytes(candidate, errors='surrogate_or_strict')):
 268                 break
 269 
 270         return candidate

 272     def path_dwim_relative_stack(self, paths, dirname, source, is_role=False):
 273         '''
 274         find one file in first path in stack taking roles into account and adding play basedir as fallback
 275 
 276         :arg paths: A list of text strings which are the paths to look for the filename in.
 277         :arg dirname: A text string representing a directory.  The directory
 278             is prepended to the source to form the path to search for.
 279         :arg source: A text string which is the filename to search for
 280         :rtype: A text string
 281         :returns: An absolute path to the filename ``source`` if found
 282         :raises: An AnsibleFileNotFound Exception if the file is found to exist in the search paths
 283         '''
 284         b_dirname = to_bytes(dirname, errors='surrogate_or_strict')
 285         b_source = to_bytes(source, errors='surrogate_or_strict')
 286 
 287         result = None
 288         search = []
 289         if source is None:
 290             display.warning('Invalid request to find a file that matches a "null" value')
 291         elif source and (source.startswith('~') or source.startswith(os.path.sep)):
 292             # path is absolute, no relative needed, check existence and return source
 293             test_path = unfrackpath(b_source, follow=False)
 294             if os.path.exists(to_bytes(test_path, errors='surrogate_or_strict')):
 295                 result = test_path
 296         else:
 297             display.debug(u'evaluation_path:\n\t%s' % '\n\t'.join(paths))
 298             for path in paths:
 299                 upath = unfrackpath(path, follow=False)
 300                 b_upath = to_bytes(upath, errors='surrogate_or_strict')
 301                 b_pb_base_dir = os.path.dirname(b_upath)
 302 
 303                 # if path is in role and 'tasks' not there already, add it into the search
 304                 if (is_role or self._is_role(path)) and b_pb_base_dir.endswith(b'/tasks'):
 305                     search.append(os.path.join(os.path.dirname(b_pb_base_dir), b_dirname, b_source))
 306                     search.append(os.path.join(b_pb_base_dir, b_source))
 307                 else:
 308                     # don't add dirname if user already is using it in source
 309                     if b_source.split(b'/')[0] != dirname:
 310                         search.append(os.path.join(b_upath, b_dirname, b_source))
 311                     search.append(os.path.join(b_upath, b_source))
 312 
 313             # always append basedir as last resort
 314             # don't add dirname if user already is using it in source
 315             if b_source.split(b'/')[0] != dirname:
 316                 search.append(os.path.join(to_bytes(self.get_basedir(), errors='surrogate_or_strict'), b_dirname, b_source))
 317             search.append(os.path.join(to_bytes(self.get_basedir(), errors='surrogate_or_strict'), b_source))
 318 
 319             display.debug(u'search_path:\n\t%s' % to_text(b'\n\t'.join(search)))
 320             for b_candidate in search:
 321                 display.vvvvv(u'looking for "%s" at "%s"' % (source, to_text(b_candidate)))
 322                 if os.path.exists(b_candidate):
 323                     result = to_text(b_candidate)
 324                     break
 325 
 326         if result is None:
 327             raise AnsibleFileNotFound(file_name=source, paths=[to_native(p) for p in search])
 328 
 329         return result

 331     def _create_content_tempfile(self, content):
 332         ''' Create a tempfile containing defined content '''
 333         fd, content_tempfile = tempfile.mkstemp(dir=C.DEFAULT_LOCAL_TMP)
 334         f = os.fdopen(fd, 'wb')
 335         content = to_bytes(content)
 336         try:
 337             f.write(content)
 338         except Exception as err:
 339             os.remove(content_tempfile)
 340             raise Exception(err)
 341         finally:
 342             f.close()
 343         return content_tempfile

 345     def get_real_file(self, file_path, decrypt=True):
 346         """
 347         If the file is vault encrypted return a path to a temporary decrypted file
 348         If the file is not encrypted then the path is returned
 349         Temporary files are cleanup in the destructor
 350         """
 351 
 352         if not file_path or not isinstance(file_path, (binary_type, text_type)):
 353             raise AnsibleParserError("Invalid filename: '%s'" % to_native(file_path))
 354 
 355         b_file_path = to_bytes(file_path, errors='surrogate_or_strict')
 356         if not self.path_exists(b_file_path) or not self.is_file(b_file_path):
 357             raise AnsibleFileNotFound(file_name=file_path)
 358 
 359         real_path = self.path_dwim(file_path)
 360 
 361         try:
 362             if decrypt:
 363                 with open(to_bytes(real_path), 'rb') as f:
 364                     # Limit how much of the file is read since we do not know
 365                     # whether this is a vault file and therefore it could be very
 366                     # large.
 367                     if is_encrypted_file(f, count=len(b_HEADER)):
 368                         # if the file is encrypted and no password was specified,
 369                         # the decrypt call would throw an error, but we check first
 370                         # since the decrypt function doesn't know the file name
 371                         data = f.read()
 372                         if not self._vault.secrets:
 373                             raise AnsibleParserError("A vault password or secret must be specified to decrypt %s" % to_native(file_path))
 374 
 375                         data = self._vault.decrypt(data, filename=real_path)
 376                         # Make a temp file
 377                         real_path = self._create_content_tempfile(data)
 378                         self._tempfiles.add(real_path)
 379 
 380             return real_path
 381 
 382         except (IOError, OSError) as e:
 383             raise AnsibleParserError("an error occurred while trying to read the file '%s': %s" % (to_native(real_path), to_native(e)), orig_exc=e)

 385     def cleanup_tmp_file(self, file_path):
 386         """
 387         Removes any temporary files created from a previous call to
 388         get_real_file. file_path must be the path returned from a
 389         previous call to get_real_file.
 390         """
 391         if file_path in self._tempfiles:
 392             os.unlink(file_path)
 393             self._tempfiles.remove(file_path)

 395     def cleanup_all_tmp_files(self):
 396         """
 397         Removes all temporary files that DataLoader has created
 398         NOTE: not thread safe, forks also need special handling see __init__ for details.
 399         """
 400         for f in self._tempfiles:
 401             try:
 402                 self.cleanup_tmp_file(f)
 403             except Exception as e:
 404                 display.warning("Unable to cleanup temp files: %s" % to_text(e))

 406     def find_vars_files(self, path, name, extensions=None, allow_dir=True):
 407         """
 408         Find vars files in a given path with specified name. This will find
 409         files in a dir named <name>/ or a file called <name> ending in known
 410         extensions.
 411         """
 412 
 413         b_path = to_bytes(os.path.join(path, name))
 414         found = []
 415 
 416         if extensions is None:
 417             # Look for file with no extension first to find dir before file
 418             extensions = [''] + C.YAML_FILENAME_EXTENSIONS
 419         # add valid extensions to name
 420         for ext in extensions:
 421 
 422             if '.' in ext:
 423                 full_path = b_path + to_bytes(ext)
 424             elif ext:
 425                 full_path = b'.'.join([b_path, to_bytes(ext)])
 426             else:
 427                 full_path = b_path
 428 
 429             if self.path_exists(full_path):
 430                 if self.is_directory(full_path):
 431                     if allow_dir:
 432                         found.extend(self._get_dir_vars_files(to_text(full_path), extensions))
 433                     else:
 434                         continue
 435                 else:
 436                     found.append(full_path)
 437                 break
 438         return found

 440     def _get_dir_vars_files(self, path, extensions):
 441         found = []
 442         for spath in sorted(self.list_directory(path)):
 443             if not spath.startswith(u'.') and not spath.endswith(u'~'):  # skip hidden and backups
 444 
 445                 ext = os.path.splitext(spath)[-1]
 446                 full_spath = os.path.join(path, spath)
 447 
 448                 if self.is_directory(full_spath) and not ext:  # recursive search if dir
 449                     found.extend(self._get_dir_vars_files(full_spath, extensions))
 450                 elif self.is_file(full_spath) and (not ext or to_text(ext) in extensions):
 451                     # only consider files with valid extensions or no extension
 452                     found.append(full_spath)
 453 
 454         return found
