  30 # Decode escapes adapted from rspeer's answer here:
  31 # http://stackoverflow.com/questions/4020539/process-escape-sequences-in-a-string-in-python
  32 _HEXCHAR = '[a-fA-F0-9]'
  33 _ESCAPE_SEQUENCE_RE = re.compile(r'''
  34     ( \\U{0}           # 8-digit hex escapes
  35     | \\u{1}           # 4-digit hex escapes
  36     | \\x{2}           # 2-digit hex escapes
  37     | \\N\{{[^}}]+\}}  # Unicode characters by name
  38     | \\[\\'"abfnrtv]  # Single-character escapes
  39     )'''.format(_HEXCHAR * 8, _HEXCHAR * 4, _HEXCHAR * 2), re.UNICODE | re.VERBOSE)
  40

  42 def _decode_escapes(s):
  43     def decode_match(match):
  44         return codecs.decode(match.group(0), 'unicode-escape')
  45 
  46     return _ESCAPE_SEQUENCE_RE.sub(decode_match, s)

  49 def parse_kv(args, check_raw=False):
  50     '''
  51     Convert a string of key/value items to a dict. If any free-form params
  52     are found and the check_raw option is set to True, they will be added
  53     to a new parameter called '_raw_params'. If check_raw is not enabled,
  54     they will simply be ignored.
  55     '''
  56 
  57     args = to_text(args, nonstring='passthru')
  58 
  59     options = {}
  60     if args is not None:
  61         try:
  62             vargs = split_args(args)
  63         except IndexError as e:
  64             raise AnsibleParserError("Unable to parse argument string", orig_exc=e)
  65         except ValueError as ve:
  66             if 'no closing quotation' in str(ve).lower():
  67                 raise AnsibleParserError("error parsing argument string, try quoting the entire line.", orig_exc=ve)
  68             else:
  69                 raise
  70 
  71         raw_params = []
  72         for orig_x in vargs:
  73             x = _decode_escapes(orig_x)
  74             if "=" in x:
  75                 pos = 0
  76                 try:
  77                     while True:
  78                         pos = x.index('=', pos + 1)
  79                         if pos > 0 and x[pos - 1] != '\\':
  80                             break
  81                 except ValueError:
  82                     # ran out of string, but we must have some escaped equals,
  83                     # so replace those and append this to the list of raw params
  84                     raw_params.append(x.replace('\\=', '='))
  85                     continue
  86 
  87                 k = x[:pos]
  88                 v = x[pos + 1:]
  89 
  90                 # FIXME: make the retrieval of this list of shell/command
  91                 #        options a function, so the list is centralized
  92                 if check_raw and k not in ('creates', 'removes', 'chdir', 'executable', 'warn'):
  93                     raw_params.append(orig_x)
  94                 else:
  95                     options[k.strip()] = unquote(v.strip())
  96             else:
  97                 raw_params.append(orig_x)
  98 
  99         # recombine the free-form params, if any were found, and assign
 100         # them to a special option for use later by the shell/command module
 101         if len(raw_params) > 0:
 102             options[u'_raw_params'] = join_args(raw_params)
 103 
 104     return options

 107 def _get_quote_state(token, quote_char):
 108     '''
 109     the goal of this block is to determine if the quoted string
 110     is unterminated in which case it needs to be put back together
 111     '''
 112     # the char before the current one, used to see if
 113     # the current character is escaped
 114     prev_char = None
 115     for idx, cur_char in enumerate(token):
 116         if idx > 0:
 117             prev_char = token[idx - 1]
 118         if cur_char in '"\'' and prev_char != '\\':
 119             if quote_char:
 120                 if cur_char == quote_char:
 121                     quote_char = None
 122             else:
 123                 quote_char = cur_char
 124     return quote_char

 127 def _count_jinja2_blocks(token, cur_depth, open_token, close_token):
 128     '''
 129     this function counts the number of opening/closing blocks for a
 130     given opening/closing type and adjusts the current depth for that
 131     block based on the difference
 132     '''
 133     num_open = token.count(open_token)
 134     num_close = token.count(close_token)
 135     if num_open != num_close:
 136         cur_depth += (num_open - num_close)
 137         if cur_depth < 0:
 138             cur_depth = 0
 139     return cur_depth

 142 def join_args(s):
 143     '''
 144     Join the original cmd based on manipulations by split_args().
 145     This retains the original newlines and whitespaces.
 146     '''
 147     result = ''
 148     for p in s:
 149         if len(result) == 0 or result.endswith('\n'):
 150             result += p
 151         else:
 152             result += ' ' + p
 153     return result

 156 def split_args(args):
 157     '''
 158     Splits args on whitespace, but intelligently reassembles
 159     those that may have been split over a jinja2 block or quotes.
 160 
 161     When used in a remote module, we won't ever have to be concerned about
 162     jinja2 blocks, however this function is/will be used in the
 163     core portions as well before the args are templated.
 164 
 165     example input: a=b c="foo bar"
 166     example output: ['a=b', 'c="foo bar"']
 167 
 168     Basically this is a variation shlex that has some more intelligence for
 169     how Ansible needs to use it.
 170     '''
 171 
 172     # the list of params parsed out of the arg string
 173     # this is going to be the result value when we are done
 174     params = []
 175 
 176     # Initial split on newlines
 177     items = args.split('\n')
 178 
 179     # iterate over the tokens, and reassemble any that may have been
 180     # split on a space inside a jinja2 block.
 181     # ex if tokens are "{{", "foo", "}}" these go together
 182 
 183     # These variables are used
 184     # to keep track of the state of the parsing, since blocks and quotes
 185     # may be nested within each other.
 186 
 187     quote_char = None
 188     inside_quotes = False
 189     print_depth = 0  # used to count nested jinja2 {{ }} blocks
 190     block_depth = 0  # used to count nested jinja2 {% %} blocks
 191     comment_depth = 0  # used to count nested jinja2 {# #} blocks
 192 
 193     # now we loop over each split chunk, coalescing tokens if the white space
 194     # split occurred within quotes or a jinja2 block of some kind
 195     for (itemidx, item) in enumerate(items):
 196 
 197         # we split on spaces and newlines separately, so that we
 198         # can tell which character we split on for reassembly
 199         # inside quotation characters
 200         tokens = item.split(' ')
 201 
 202         line_continuation = False
 203         for (idx, token) in enumerate(tokens):
 204 
 205             # Empty entries means we have subsequent spaces
 206             # We want to hold onto them so we can reconstruct them later
 207             if len(token) == 0 and idx != 0:
 208                 params[-1] += ' '
 209                 continue
 210 
 211             # if we hit a line continuation character, but
 212             # we're not inside quotes, ignore it and continue
 213             # on to the next token while setting a flag
 214             if token == '\\' and not inside_quotes:
 215                 line_continuation = True
 216                 continue
 217 
 218             # store the previous quoting state for checking later
 219             was_inside_quotes = inside_quotes
 220             quote_char = _get_quote_state(token, quote_char)
 221             inside_quotes = quote_char is not None
 222 
 223             # multiple conditions may append a token to the list of params,
 224             # so we keep track with this flag to make sure it only happens once
 225             # append means add to the end of the list, don't append means concatenate
 226             # it to the end of the last token
 227             appended = False
 228 
 229             # if we're inside quotes now, but weren't before, append the token
 230             # to the end of the list, since we'll tack on more to it later
 231             # otherwise, if we're inside any jinja2 block, inside quotes, or we were
 232             # inside quotes (but aren't now) concat this token to the last param
 233             if inside_quotes and not was_inside_quotes and not(print_depth or block_depth or comment_depth):
 234                 params.append(token)
 235                 appended = True
 236             elif print_depth or block_depth or comment_depth or inside_quotes or was_inside_quotes:
 237                 if idx == 0 and was_inside_quotes:
 238                     params[-1] = "%s%s" % (params[-1], token)
 239                 elif len(tokens) > 1:
 240                     spacer = ''
 241                     if idx > 0:
 242                         spacer = ' '
 243                     params[-1] = "%s%s%s" % (params[-1], spacer, token)
 244                 else:
 245                     params[-1] = "%s\n%s" % (params[-1], token)
 246                 appended = True
 247 
 248             # if the number of paired block tags is not the same, the depth has changed, so we calculate that here
 249             # and may append the current token to the params (if we haven't previously done so)
 250             prev_print_depth = print_depth
 251             print_depth = _count_jinja2_blocks(token, print_depth, "{{", "}}")
 252             if print_depth != prev_print_depth and not appended:
 253                 params.append(token)
 254                 appended = True
 255 
 256             prev_block_depth = block_depth
 257             block_depth = _count_jinja2_blocks(token, block_depth, "{%", "%}")
 258             if block_depth != prev_block_depth and not appended:
 259                 params.append(token)
 260                 appended = True
 261 
 262             prev_comment_depth = comment_depth
 263             comment_depth = _count_jinja2_blocks(token, comment_depth, "{#", "#}")
 264             if comment_depth != prev_comment_depth and not appended:
 265                 params.append(token)
 266                 appended = True
 267 
 268             # finally, if we're at zero depth for all blocks and not inside quotes, and have not
 269             # yet appended anything to the list of params, we do so now
 270             if not (print_depth or block_depth or comment_depth) and not inside_quotes and not appended and token != '':
 271                 params.append(token)
 272 
 273         # if this was the last token in the list, and we have more than
 274         # one item (meaning we split on newlines), add a newline back here
 275         # to preserve the original structure
 276         if len(items) > 1 and itemidx != len(items) - 1 and not line_continuation:
 277             params[-1] += '\n'
 278 
 279         # always clear the line continuation flag
 280         line_continuation = False
 281 
 282     # If we're done and things are not at zero depth or we're still inside quotes,
 283     # raise an error to indicate that the args were unbalanced
 284     if print_depth or block_depth or comment_depth or inside_quotes:
 285         raise AnsibleParserError(u"failed at splitting arguments, either an unbalanced jinja2 block or quotes: {0}".format(args))
 286 
 287     return params
