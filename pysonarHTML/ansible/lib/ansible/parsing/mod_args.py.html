  58 class ModuleArgsParser:
  59 
  60     """
  61     There are several ways a module and argument set can be expressed:
  62 
  63     # legacy form (for a shell command)
  64     - action: shell echo hi
  65 
  66     # common shorthand for local actions vs delegate_to
  67     - local_action: shell echo hi
  68 
  69     # most commonly:
  70     - copy: src=a dest=b
  71 
  72     # legacy form
  73     - action: copy src=a dest=b
  74 
  75     # complex args form, for passing structured data
  76     - copy:
  77         src: a
  78         dest: b
  79 
  80     # gross, but technically legal
  81     - action:
  82         module: copy
  83         args:
  84           src: a
  85           dest: b
  86 
  87     # Standard YAML form for command-type modules. In this case, the args specified
  88     # will act as 'defaults' and will be overridden by any args specified
  89     # in one of the other formats (complex args under the action, or
  90     # parsed from the k=v string
  91     - command: 'pwd'
  92       args:
  93         chdir: '/tmp'
  94 
  95 
  96     This class has some of the logic to canonicalize these into the form
  97 
  98     - module: <module_name>
  99       delegate_to: <optional>
 100       args: <args>
 101 
 102     Args may also be munged for certain shell command parameters.
 103     """
 104 
 105     def __init__(self, task_ds=None, collection_list=None):
 106         task_ds = {} if task_ds is None else task_ds
 107 
 108         if not isinstance(task_ds, dict):
 109             raise AnsibleAssertionError("the type of 'task_ds' should be a dict, but is a %s" % type(task_ds))
 110         self._task_ds = task_ds
 111         self._collection_list = collection_list
 112         # delayed local imports to prevent circular import
 113         from ansible.playbook.task import Task
 114         from ansible.playbook.handler import Handler
 115         # store the valid Task/Handler attrs for quick access
 116         self._task_attrs = set(Task._valid_attrs.keys())
 117         self._task_attrs.update(set(Handler._valid_attrs.keys()))
 118         # HACK: why are these not FieldAttributes on task with a post-validate to check usage?
 119         self._task_attrs.update(['local_action', 'static'])
 120         self._task_attrs = frozenset(self._task_attrs)
 121 
 122         self.internal_redirect_list = []
 123 
 124     def _split_module_string(self, module_string):
 125         '''
 126         when module names are expressed like:
 127         action: copy src=a dest=b
 128         the first part of the string is the name of the module
 129         and the rest are strings pertaining to the arguments.
 130         '''
 131 
 132         tokens = split_args(module_string)
 133         if len(tokens) > 1:
 134             return (tokens[0], " ".join(tokens[1:]))
 135         else:
 136             return (tokens[0], "")
 137 
 138     def _normalize_parameters(self, thing, action=None, additional_args=None):
 139         '''
 140         arguments can be fuzzy.  Deal with all the forms.
 141         '''
 142 
 143         additional_args = {} if additional_args is None else additional_args
 144 
 145         # final args are the ones we'll eventually return, so first update
 146         # them with any additional args specified, which have lower priority
 147         # than those which may be parsed/normalized next
 148         final_args = dict()
 149         if additional_args:
 150             if isinstance(additional_args, string_types):
 151                 templar = Templar(loader=None)
 152                 if templar.is_template(additional_args):
 153                     final_args['_variable_params'] = additional_args
 154                 else:
 155                     raise AnsibleParserError("Complex args containing variables cannot use bare variables (without Jinja2 delimiters), "
 156                                              "and must use the full variable style ('{{var_name}}')")
 157             elif isinstance(additional_args, dict):
 158                 final_args.update(additional_args)
 159             else:
 160                 raise AnsibleParserError('Complex args must be a dictionary or variable string ("{{var}}").')
 161 
 162         # how we normalize depends if we figured out what the module name is
 163         # yet.  If we have already figured it out, it's a 'new style' invocation.
 164         # otherwise, it's not
 165 
 166         if action is not None:
 167             args = self._normalize_new_style_args(thing, action)
 168         else:
 169             (action, args) = self._normalize_old_style_args(thing)
 170 
 171             # this can occasionally happen, simplify
 172             if args and 'args' in args:
 173                 tmp_args = args.pop('args')
 174                 if isinstance(tmp_args, string_types):
 175                     tmp_args = parse_kv(tmp_args)
 176                 args.update(tmp_args)
 177 
 178         # only internal variables can start with an underscore, so
 179         # we don't allow users to set them directly in arguments
 180         if args and action not in FREEFORM_ACTIONS:
 181             for arg in args:
 182                 arg = to_text(arg)
 183                 if arg.startswith('_ansible_'):
 184                     raise AnsibleError("invalid parameter specified for action '%s': '%s'" % (action, arg))
 185 
 186         # finally, update the args we're going to return with the ones
 187         # which were normalized above
 188         if args:
 189             final_args.update(args)
 190 
 191         return (action, final_args)
 192 
 193     def _normalize_new_style_args(self, thing, action):
 194         '''
 195         deals with fuzziness in new style module invocations
 196         accepting key=value pairs and dictionaries, and returns
 197         a dictionary of arguments
 198 
 199         possible example inputs:
 200             'echo hi', 'shell'
 201             {'region': 'xyz'}, 'ec2'
 202         standardized outputs like:
 203             { _raw_params: 'echo hi', _uses_shell: True }
 204         '''
 205 
 206         if isinstance(thing, dict):
 207             # form is like: { xyz: { x: 2, y: 3 } }
 208             args = thing
 209         elif isinstance(thing, string_types):
 210             # form is like: copy: src=a dest=b
 211             check_raw = action in FREEFORM_ACTIONS
 212             args = parse_kv(thing, check_raw=check_raw)
 213         elif thing is None:
 214             # this can happen with modules which take no params, like ping:
 215             args = None
 216         else:
 217             raise AnsibleParserError("unexpected parameter type in action: %s" % type(thing), obj=self._task_ds)
 218         return args
 219 
 220     def _normalize_old_style_args(self, thing):
 221         '''
 222         deals with fuzziness in old-style (action/local_action) module invocations
 223         returns tuple of (module_name, dictionary_args)
 224 
 225         possible example inputs:
 226            { 'shell' : 'echo hi' }
 227            'shell echo hi'
 228            {'module': 'ec2', 'x': 1 }
 229         standardized outputs like:
 230            ('ec2', { 'x': 1} )
 231         '''
 232 
 233         action = None
 234         args = None
 235 
 236         if isinstance(thing, dict):
 237             # form is like:  action: { module: 'copy', src: 'a', dest: 'b' }
 238             thing = thing.copy()
 239             if 'module' in thing:
 240                 action, module_args = self._split_module_string(thing['module'])
 241                 args = thing.copy()
 242                 check_raw = action in FREEFORM_ACTIONS
 243                 args.update(parse_kv(module_args, check_raw=check_raw))
 244                 del args['module']
 245 
 246         elif isinstance(thing, string_types):
 247             # form is like:  action: copy src=a dest=b
 248             (action, args) = self._split_module_string(thing)
 249             check_raw = action in FREEFORM_ACTIONS
 250             args = parse_kv(args, check_raw=check_raw)
 251 
 252         else:
 253             # need a dict or a string, so giving up
 254             raise AnsibleParserError("unexpected parameter type in action: %s" % type(thing), obj=self._task_ds)
 255 
 256         return (action, args)
 257 
 258     def parse(self, skip_action_validation=False):
 259         '''
 260         Given a task in one of the supported forms, parses and returns
 261         returns the action, arguments, and delegate_to values for the
 262         task, dealing with all sorts of levels of fuzziness.
 263         '''
 264 
 265         thing = None
 266 
 267         action = None
 268         delegate_to = self._task_ds.get('delegate_to', Sentinel)
 269         args = dict()
 270 
 271         self.internal_redirect_list = []
 272 
 273         # This is the standard YAML form for command-type modules. We grab
 274         # the args and pass them in as additional arguments, which can/will
 275         # be overwritten via dict updates from the other arg sources below
 276         additional_args = self._task_ds.get('args', dict())
 277 
 278         # We can have one of action, local_action, or module specified
 279         # action
 280         if 'action' in self._task_ds:
 281             # an old school 'action' statement
 282             thing = self._task_ds['action']
 283             action, args = self._normalize_parameters(thing, action=action, additional_args=additional_args)
 284 
 285         # local_action
 286         if 'local_action' in self._task_ds:
 287             # local_action is similar but also implies a delegate_to
 288             if action is not None:
 289                 raise AnsibleParserError("action and local_action are mutually exclusive", obj=self._task_ds)
 290             thing = self._task_ds.get('local_action', '')
 291             delegate_to = 'localhost'
 292             action, args = self._normalize_parameters(thing, action=action, additional_args=additional_args)
 293 
 294         # module: <stuff> is the more new-style invocation
 295 
 296         # filter out task attributes so we're only querying unrecognized keys as actions/modules
 297         non_task_ds = dict((k, v) for k, v in iteritems(self._task_ds) if (k not in self._task_attrs) and (not k.startswith('with_')))
 298 
 299         # walk the filtered input dictionary to see if we recognize a module name
 300         for item, value in iteritems(non_task_ds):
 301             is_action_candidate = False
 302             if item in BUILTIN_TASKS:
 303                 is_action_candidate = True
 304             elif skip_action_validation:
 305                 is_action_candidate = True
 306             else:
 307                 # If the plugin is resolved and redirected smuggle the list of candidate names via the task attribute 'internal_redirect_list'
 308                 context = action_loader.find_plugin_with_context(item, collection_list=self._collection_list)
 309                 if not context.resolved:
 310                     context = module_loader.find_plugin_with_context(item, collection_list=self._collection_list)
 311                     if context.resolved and context.redirect_list:
 312                         self.internal_redirect_list = context.redirect_list
 313                 elif context.redirect_list:
 314                     self.internal_redirect_list = context.redirect_list
 315 
 316                 is_action_candidate = bool(self.internal_redirect_list)
 317 
 318             if is_action_candidate:
 319                 # finding more than one module name is a problem
 320                 if action is not None:
 321                     raise AnsibleParserError("conflicting action statements: %s, %s" % (action, item), obj=self._task_ds)
 322                 action = item
 323                 thing = value
 324                 action, args = self._normalize_parameters(thing, action=action, additional_args=additional_args)
 325 
 326         # if we didn't see any module in the task at all, it's not a task really
 327         if action is None:
 328             if non_task_ds:  # there was one non-task action, but we couldn't find it
 329                 bad_action = list(non_task_ds.keys())[0]
 330                 raise AnsibleParserError("couldn't resolve module/action '{0}'. This often indicates a "
 331                                          "misspelling, missing collection, or incorrect module path.".format(bad_action),
 332                                          obj=self._task_ds)
 333             else:
 334                 raise AnsibleParserError("no module/action detected in task.",
 335                                          obj=self._task_ds)
 336         elif args.get('_raw_params', '') != '' and action not in RAW_PARAM_MODULES:
 337             templar = Templar(loader=None)
 338             raw_params = args.pop('_raw_params')
 339             if templar.is_template(raw_params):
 340                 args['_variable_params'] = raw_params
 341             else:
 342                 raise AnsibleParserError("this task '%s' has extra params, which is only allowed in the following modules: %s" % (action,
 343                                                                                                                                   ", ".join(RAW_PARAM_MODULES)),
 344                                          obj=self._task_ds)
 345 
 346         return (action, args, delegate_to)
