  36 class AnsibleConstructor(SafeConstructor):
  37     def __init__(self, file_name=None, vault_secrets=None):
  38         self._ansible_file_name = file_name
  39         super(AnsibleConstructor, self).__init__()
  40         self._vaults = {}
  41         self.vault_secrets = vault_secrets or []
  42         self._vaults['default'] = VaultLib(secrets=self.vault_secrets)
  43 
  44     def construct_yaml_map(self, node):
  45         data = AnsibleMapping()
  46         yield data
  47         value = self.construct_mapping(node)
  48         data.update(value)
  49         data.ansible_pos = self._node_position_info(node)
  50 
  51     def construct_mapping(self, node, deep=False):
  52         # Most of this is from yaml.constructor.SafeConstructor.  We replicate
  53         # it here so that we can warn users when they have duplicate dict keys
  54         # (pyyaml silently allows overwriting keys)
  55         if not isinstance(node, MappingNode):
  56             raise ConstructorError(None, None,
  57                                    "expected a mapping node, but found %s" % node.id,
  58                                    node.start_mark)
  59         self.flatten_mapping(node)
  60         mapping = AnsibleMapping()
  61 
  62         # Add our extra information to the returned value
  63         mapping.ansible_pos = self._node_position_info(node)
  64 
  65         for key_node, value_node in node.value:
  66             key = self.construct_object(key_node, deep=deep)
  67             try:
  68                 hash(key)
  69             except TypeError as exc:
  70                 raise ConstructorError("while constructing a mapping", node.start_mark,
  71                                        "found unacceptable key (%s)" % exc, key_node.start_mark)
  72 
  73             if key in mapping:
  74                 msg = (u'While constructing a mapping from {1}, line {2}, column {3}, found a duplicate dict key ({0}).'
  75                        u' Using last defined value only.'.format(key, *mapping.ansible_pos))
  76                 if C.DUPLICATE_YAML_DICT_KEY == 'warn':
  77                     display.warning(msg)
  78                 elif C.DUPLICATE_YAML_DICT_KEY == 'error':
  79                     raise ConstructorError(context=None, context_mark=None,
  80                                            problem=to_native(msg),
  81                                            problem_mark=node.start_mark,
  82                                            note=None)
  83                 else:
  84                     # when 'ignore'
  85                     display.debug(msg)
  86 
  87             value = self.construct_object(value_node, deep=deep)
  88             mapping[key] = value
  89 
  90         return mapping
  91 
  92     def construct_yaml_str(self, node):
  93         # Override the default string handling function
  94         # to always return unicode objects
  95         value = self.construct_scalar(node)
  96         ret = AnsibleUnicode(value)
  97 
  98         ret.ansible_pos = self._node_position_info(node)
  99 
 100         return ret
 101 
 102     def construct_vault_encrypted_unicode(self, node):
 103         value = self.construct_scalar(node)
 104         b_ciphertext_data = to_bytes(value)
 105         # could pass in a key id here to choose the vault to associate with
 106         # TODO/FIXME: plugin vault selector
 107         vault = self._vaults['default']
 108         if vault.secrets is None:
 109             raise ConstructorError(context=None, context_mark=None,
 110                                    problem="found !vault but no vault password provided",
 111                                    problem_mark=node.start_mark,
 112                                    note=None)
 113         ret = AnsibleVaultEncryptedUnicode(b_ciphertext_data)
 114         ret.vault = vault
 115         return ret
 116 
 117     def construct_yaml_seq(self, node):
 118         data = AnsibleSequence()
 119         yield data
 120         data.extend(self.construct_sequence(node))
 121         data.ansible_pos = self._node_position_info(node)
 122 
 123     def construct_yaml_unsafe(self, node):
 124         return wrap_var(self.construct_yaml_str(node))
 125 
 126     def _node_position_info(self, node):
 127         # the line number where the previous token has ended (plus empty lines)
 128         # Add one so that the first line is line 1 rather than line 0
 129         column = node.start_mark.column + 1
 130         line = node.start_mark.line + 1
 131 
 132         # in some cases, we may have pre-read the data and then
 133         # passed it to the load() call for YAML, in which case we
 134         # want to override the default datasource (which would be
 135         # '<string>') to the actual filename we read in
 136         datasource = self._ansible_file_name or node.start_mark.name
 137 
 138         return (datasource, line, column)
