  33 class AnsibleBaseYAMLObject(object):
  34     '''
  35     the base class used to sub-class python built-in objects
  36     so that we can add attributes to them during yaml parsing
  37 
  38     '''
  39     _data_source = None
  40     _line_number = 0
  41     _column_number = 0
  42 
  43     def _get_ansible_position(self):
  44         return (self._data_source, self._line_number, self._column_number)
  45 
  46     def _set_ansible_position(self, obj):
  47         try:
  48             (src, line, col) = obj
  49         except (TypeError, ValueError):
  50             raise AssertionError(
  51                 'ansible_pos can only be set with a tuple/list '
  52                 'of three values: source, line number, column number'
  53             )
  54         self._data_source = src
  55         self._line_number = line
  56         self._column_number = col
  57 
  58     ansible_pos = property(_get_ansible_position, _set_ansible_position)

  86 class AnsibleVaultEncryptedUnicode(Sequence, AnsibleBaseYAMLObject):
  87     '''Unicode like object that is not evaluated (decrypted) until it needs to be'''
  88     __UNSAFE__ = True
  89     __ENCRYPTED__ = True
  90     yaml_tag = u'!vault'
  91 
  92     @classmethod
  93     def from_plaintext(cls, seq, vault, secret):
  94         if not vault:
  95             raise vault.AnsibleVaultError('Error creating AnsibleVaultEncryptedUnicode, invalid vault (%s) provided' % vault)
  96 
  97         ciphertext = vault.encrypt(seq, secret)
  98         avu = cls(ciphertext)
  99         avu.vault = vault
 100         return avu
 101 
 102     def __init__(self, ciphertext):
 103         '''A AnsibleUnicode with a Vault attribute that can decrypt it.
 104 
 105         ciphertext is a byte string (str on PY2, bytestring on PY3).
 106 
 107         The .data attribute is a property that returns the decrypted plaintext
 108         of the ciphertext as a PY2 unicode or PY3 string object.
 109         '''
 110         super(AnsibleVaultEncryptedUnicode, self).__init__()
 111 
 112         # after construction, calling code has to set the .vault attribute to a vaultlib object
 113         self.vault = None
 114         self._ciphertext = to_bytes(ciphertext)
 115 
 116     @property
 117     def data(self):
 118         if not self.vault:
 119             return to_text(self._ciphertext)
 120         return to_text(self.vault.decrypt(self._ciphertext))
 121 
 122     @data.setter
 123     def data(self, value):
 124         self._ciphertext = to_bytes(value)
 125 
 126     def is_encrypted(self):
 127         return self.vault and self.vault.is_encrypted(self._ciphertext)
 128 
 129     def __eq__(self, other):
 130         if self.vault:
 131             return other == self.data
 132         return False
 133 
 134     def __ne__(self, other):
 135         if self.vault:
 136             return other != self.data
 137         return True
 138 
 139     def __reversed__(self):
 140         # This gets inerhited from ``collections.Sequence`` which returns a generator
 141         # make this act more like the string implementation
 142         return to_text(self[::-1], errors='surrogate_or_strict')
 143 
 144     def __str__(self):
 145         return to_native(self.data, errors='surrogate_or_strict')
 146 
 147     def __unicode__(self):
 148         return to_text(self.data, errors='surrogate_or_strict')
 149 
 150     def encode(self, encoding=None, errors=None):

 212         if isinstance(other, AnsibleVaultEncryptedUnicode):
