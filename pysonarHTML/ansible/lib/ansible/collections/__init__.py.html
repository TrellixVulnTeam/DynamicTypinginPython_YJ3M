   9 from ansible.module_utils._text import to_bytes
  10 
  11 B_FLAG_FILES = frozenset([b'MANIFEST.json', b'galaxy.yml'])
  12 
  13 
  14 def is_collection_path(path):
  15     """
  16     Verify that a path meets min requirements to be a collection
  17     :param path: byte-string path to evaluate for collection containment
  18     :return: boolean signifying 'collectionness'
  19     """
  20 
  21     is_coll = False
  22     b_path = to_bytes(path)
  23     if os.path.isdir(b_path):
  24         for b_flag in B_FLAG_FILES:
  25             if os.path.exists(os.path.join(b_path, b_flag)):
  26                 is_coll = True
  27                 break
  28 
  29     return is_coll
