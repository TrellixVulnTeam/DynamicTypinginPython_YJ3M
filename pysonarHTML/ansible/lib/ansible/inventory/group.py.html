  32 def to_safe_group_name(name, replacer="_", force=False, silent=False):
  33     # Converts 'bad' characters in a string to underscores (or provided replacer) so they can be used as Ansible hosts or groups
  34 
  35     warn = ''
  36     if name:  # when deserializing we might not have name yet
  37         invalid_chars = C.INVALID_VARIABLE_NAMES.findall(name)
  38         if invalid_chars:
  39             msg = 'invalid character(s) "%s" in group name (%s)' % (to_text(set(invalid_chars)), to_text(name))
  40             if C.TRANSFORM_INVALID_GROUP_CHARS not in ('never', 'ignore') or force:
  41                 name = C.INVALID_VARIABLE_NAMES.sub(replacer, name)
  42                 if not (silent or C.TRANSFORM_INVALID_GROUP_CHARS == 'silently'):
  43                     display.vvvv('Replacing ' + msg)
  44                     warn = 'Invalid characters were found in group names and automatically replaced, use -vvvv to see details'
  45             else:
  46                 if C.TRANSFORM_INVALID_GROUP_CHARS == 'never':
  47                     display.vvvv('Not replacing %s' % msg)
  48                     warn = True
  49                     warn = 'Invalid characters were found in group names but not replaced, use -vvvv to see details'
  50 
  51     if warn:
  52         display.warning(warn)
  53 
  54     return name

  57 class Group:
  58     ''' a group of ansible hosts '''

  62     def __init__(self, name=None):
  63 
  64         self.depth = 0
  65         self.name = to_safe_group_name(name)
  66         self.hosts = []
  67         self._hosts = None
  68         self.vars = {}
  69         self.child_groups = []
  70         self.parent_groups = []

  72         self.priority = 1

  86     def serialize(self):
  87         parent_groups = []
  88         for parent in self.parent_groups:
  89             parent_groups.append(parent.serialize())
  90 
  91         self._hosts = None
  92 
  93         result = dict(
  94             name=self.name,
  95             vars=self.vars.copy(),
  96             parent_groups=parent_groups,
  97             depth=self.depth,
  98             hosts=self.hosts,
  99         )
 100 
 101         return result
 102 
 103     def deserialize(self, data):
 104         self.__init__()
 105         self.name = data.get('name')
 106         self.vars = data.get('vars', dict())
 107         self.depth = data.get('depth', 0)
 108         self.hosts = data.get('hosts', [])
 109         self._hosts = None
 110 
 111         parent_groups = data.get('parent_groups', [])
 112         for parent_data in parent_groups:
 113             g = Group()
 114             g.deserialize(parent_data)
 115             self.parent_groups.append(g)

 117     def _walk_relationship(self, rel, include_self=False, preserve_ordering=False):
 118         '''
 119         Given `rel` that is an iterable property of Group,
 120         consitituting a directed acyclic graph among all groups,
 121         Returns a set of all groups in full tree
 122         A   B    C
 123         |  / |  /
 124         | /  | /
 125         D -> E
 126         |  /    vertical connections
 127         | /     are directed upward
 128         F
 129         Called on F, returns set of (A, B, C, D, E)
 130         '''
 131         seen = set([])
 132         unprocessed = set(getattr(self, rel))
 133         if include_self:
 134             unprocessed.add(self)
 135         if preserve_ordering:
 136             ordered = [self] if include_self else []
 137             ordered.extend(getattr(self, rel))
 138 
 139         while unprocessed:
 140             seen.update(unprocessed)
 141             new_unprocessed = set([])
 142 
 143             for new_item in chain.from_iterable(getattr(g, rel) for g in unprocessed):
 144                 new_unprocessed.add(new_item)
 145                 if preserve_ordering:
 146                     if new_item not in seen:
 147                         ordered.append(new_item)
 148 
 149             new_unprocessed.difference_update(seen)
 150             unprocessed = new_unprocessed
 151 
 152         if preserve_ordering:
 153             return ordered
 154         return seen

 156     def get_ancestors(self):
 157         return self._walk_relationship('parent_groups')

 160         return self._walk_relationship('child_groups', **kwargs)

 171     def add_child_group(self, group):
 172         added = False
 173         if self == group:
 174             raise Exception("can't add group to itself")
 175 
 176         # don't add if it's already there
 177         if group not in self.child_groups:
 178 
 179             # prepare list of group's new ancestors this edge creates
 180             start_ancestors = group.get_ancestors()
 181             new_ancestors = self.get_ancestors()
 182             if group in new_ancestors:
 183                 raise AnsibleError("Adding group '%s' as child to '%s' creates a recursive dependency loop." % (to_native(group.name), to_native(self.name)))
 184             new_ancestors.add(self)
 185             new_ancestors.difference_update(start_ancestors)
 186 
 187             added = True
 188             self.child_groups.append(group)
 189 
 190             # update the depth of the child
 191             group.depth = max([self.depth + 1, group.depth])
 192 
 193             # update the depth of the grandchildren
 194             group._check_children_depth()
 195 
 196             # now add self to child's parent_groups list, but only if there
 197             # isn't already a group with the same name
 198             if self.name not in [g.name for g in group.parent_groups]:
 199                 group.parent_groups.append(self)
 200                 for h in group.get_hosts():
 201                     h.populate_ancestors(additions=new_ancestors)
 202 
 203             self.clear_hosts_cache()
 204         return added
 205 
 206     def _check_children_depth(self):
 207 
 208         depth = self.depth
 209         start_depth = self.depth  # self.depth could change over loop
 210         seen = set([])
 211         unprocessed = set(self.child_groups)
 212 
 213         while unprocessed:
 214             seen.update(unprocessed)
 215             depth += 1
 216             to_process = unprocessed.copy()
 217             unprocessed = set([])
 218             for g in to_process:
 219                 if g.depth < depth:
 220                     g.depth = depth
 221                     unprocessed.update(g.child_groups)
 222             if depth - start_depth > len(seen):
 223                 raise AnsibleError("The group named '%s' has a recursive dependency loop." % to_native(self.name))
 224 
 225     def add_host(self, host):
 226         added = False
 227         if host.name not in self.host_names:
 228             self.hosts.append(host)
 229             self._hosts.add(host.name)
 230             host.add_group(self)
 231             self.clear_hosts_cache()
 232             added = True
 233         return added
 234 
 235     def remove_host(self, host):
 236         removed = False
 237         if host.name in self.host_names:
 238             self.hosts.remove(host)
 239             self._hosts.remove(host.name)
 240             host.remove_group(self)
 241             self.clear_hosts_cache()
 242             removed = True
 243         return removed

 245     def set_variable(self, key, value):
 246 
 247         if key == 'ansible_group_priority':
 248             self.set_priority(int(value))
 249         else:
 250             if key in self.vars and isinstance(self.vars[key], MutableMapping) and isinstance(value, Mapping):
 251                 self.vars[key] = combine_vars(self.vars[key], value)
 252             else:
 253                 self.vars[key] = value
 254 
 255     def clear_hosts_cache(self):
 256 
 257         self._hosts_cache = None
 258         for g in self.get_ancestors():
 259             g._hosts_cache = None
 260 
 261     def get_hosts(self):
 262 
 263         if self._hosts_cache is None:
 264             self._hosts_cache = self._get_hosts()
 265         return self._hosts_cache
 266 
 267     def _get_hosts(self):
 268 
 269         hosts = []
 270         seen = {}
 271         for kid in self.get_descendants(include_self=True, preserve_ordering=True):
 272             kid_hosts = kid.hosts
 273             for kk in kid_hosts:
 274                 if kk not in seen:
 275                     seen[kk] = 1
 276                     if self.name == 'all' and kk.implicit:
 277                         continue
 278                     hosts.append(kk)
 279         return hosts
