  53 PATTERN_WITH_SUBSCRIPT = re.compile(
  54     r'''^
  55         (.+)                    # A pattern expression ending with...
  56         \[(?:                   # A [subscript] expression comprising:
  57             (-?[0-9]+)|         # A single positive or negative number
  58             ([0-9]+)([:-])      # Or an x:y or x: range.
  59             ([0-9]*)
  60         )\]

  66 def order_patterns(patterns):
  67     ''' takes a list of patterns and reorders them by modifier to apply them consistently '''
  68 
  69     # FIXME: this goes away if we apply patterns incrementally or by groups
  70     pattern_regular = []
  71     pattern_intersection = []
  72     pattern_exclude = []
  73     for p in patterns:
  74         if not p:
  75             continue
  76 
  77         if p[0] == "!":
  78             pattern_exclude.append(p)
  79         elif p[0] == "&":
  80             pattern_intersection.append(p)
  81         else:
  82             pattern_regular.append(p)
  83 
  84     # if no regular pattern was given, hence only exclude and/or intersection
  85     # make that magically work
  86     if pattern_regular == []:
  87         pattern_regular = ['all']
  88 
  89     # when applying the host selectors, run those without the "&" or "!"
  90     # first, then the &s, then the !s.
  91     return pattern_regular + pattern_intersection + pattern_exclude

  94 def split_host_pattern(pattern):
  95     """
  96     Takes a string containing host patterns separated by commas (or a list
  97     thereof) and returns a list of single patterns (which may not contain
  98     commas). Whitespace is ignored.
  99 
 100     Also accepts ':' as a separator for backwards compatibility, but it is
 101     not recommended due to the conflict with IPv6 addresses and host ranges.
 102 
 103     Example: 'a,b[1], c[2:3] , d' -> ['a', 'b[1]', 'c[2:3]', 'd']
 104     """
 105 
 106     if isinstance(pattern, list):
 107         results = (split_host_pattern(p) for p in pattern)
 108         # flatten the results
 109         return list(itertools.chain.from_iterable(results))
 110     elif not isinstance(pattern, string_types):
 111         pattern = to_text(pattern, errors='surrogate_or_strict')
 112 
 113     # If it's got commas in it, we'll treat it as a straightforward
 114     # comma-separated list of patterns.
 115     if u',' in pattern:
 116         patterns = pattern.split(u',')
 117 
 118     # If it doesn't, it could still be a single pattern. This accounts for
 119     # non-separator uses of colons: IPv6 addresses and [x:y] host ranges.
 120     else:
 121         try:
 122             (base, port) = parse_address(pattern, allow_ranges=True)
 123             patterns = [pattern]
 124         except Exception:
 125             # The only other case we accept is a ':'-separated list of patterns.
 126             # This mishandles IPv6 addresses, and is retained only for backwards
 127             # compatibility.
 128             patterns = re.findall(
 129                 to_text(r'''(?:     # We want to match something comprising:
 130                         [^\s:\[\]]  # (anything other than whitespace or ':[]'
 131                         |           # ...or...
 132                         \[[^\]]*\]  # a single complete bracketed expression)
 133                     )+              # occurring once or more
 134                 '''), pattern, re.X
 135             )
 136 
 137     return [p.strip() for p in patterns if p.strip()]

 196     def _fetch_inventory_plugins(self):
 197         ''' sets up loaded inventory plugins for usage '''
 198 
 199         display.vvvv('setting up inventory plugins')
 200 
 201         plugins = []
 202         for name in C.INVENTORY_ENABLED:
 203             plugin = inventory_loader.get(name)
 204             if plugin:
 205                 plugins.append(plugin)
 206             else:
 207                 display.warning('Failed to load inventory plugin, skipping %s' % name)
 208 
 209         if not plugins:
 210             raise AnsibleError("No inventory plugins available to generate inventory, make sure you have at least one whitelisted.")
 211 
 212         return plugins

 214     def parse_sources(self, cache=False):
 215         ''' iterate over inventory sources and parse each one to populate it'''
 216 
 217         parsed = False
 218         # allow for multiple inventory parsing
 219         for source in self._sources:
 220 
 221             if source:
 222                 if ',' not in source:
 223                     source = unfrackpath(source, follow=False)
 224                 parse = self.parse_source(source, cache=cache)
 225                 if parse and not parsed:
 226                     parsed = True
 227 
 228         if parsed:
 229             # do post processing
 230             self._inventory.reconcile_inventory()
 231         else:
 232             if C.INVENTORY_UNPARSED_IS_FAILED:
 233                 raise AnsibleError("No inventory was parsed, please check your configuration and options.")
 234             else:
 235                 display.warning("No inventory was parsed, only implicit localhost is available")
 236 
 237         for group in self.groups.values():
 238             group.vars = combine_vars(group.vars, get_vars_from_inventory_sources(self._loader, self._sources, [group], 'inventory'))
 239         for host in self.hosts.values():
 240             host.vars = combine_vars(host.vars, get_vars_from_inventory_sources(self._loader, self._sources, [host], 'inventory'))

 242     def parse_source(self, source, cache=False):
 243         ''' Generate or update inventory for the source provided '''
 244 
 245         parsed = False
 246         display.debug(u'Examining possible inventory source: %s' % source)
 247 
 248         # use binary for path functions
 249         b_source = to_bytes(source)
 250 
 251         # process directories as a collection of inventories
 252         if os.path.isdir(b_source):
 253             display.debug(u'Searching for inventory files in directory: %s' % source)
 254             for i in sorted(os.listdir(b_source)):
 255 
 256                 display.debug(u'Considering %s' % i)
 257                 # Skip hidden files and stuff we explicitly ignore
 258                 if IGNORED.search(i):
 259                     continue
 260 
 261                 # recursively deal with directory entries
 262                 fullpath = to_text(os.path.join(b_source, i), errors='surrogate_or_strict')
 263                 parsed_this_one = self.parse_source(fullpath, cache=cache)
 264                 display.debug(u'parsed %s as %s' % (fullpath, parsed_this_one))
 265                 if not parsed:
 266                     parsed = parsed_this_one
 267         else:
 268             # left with strings or files, let plugins figure it out
 269 
 270             # set so new hosts can use for inventory_file/dir vars
 271             self._inventory.current_source = source
 272 
 273             # try source with each plugin
 274             failures = []
 275             for plugin in self._fetch_inventory_plugins():
 276 
 277                 plugin_name = to_text(getattr(plugin, '_load_name', getattr(plugin, '_original_path', '')))
 278                 display.debug(u'Attempting to use plugin %s (%s)' % (plugin_name, plugin._original_path))
 279 
 280                 # initialize and figure out if plugin wants to attempt parsing this file
 281                 try:
 282                     plugin_wants = bool(plugin.verify_file(source))
 283                 except Exception:
 284                     plugin_wants = False
 285 
 286                 if plugin_wants:
 287                     try:
 288                         # FIXME in case plugin fails 1/2 way we have partial inventory
 289                         plugin.parse(self._inventory, self._loader, source, cache=cache)
 290                         try:
 291                             plugin.update_cache_if_changed()
 292                         except AttributeError:
 293                             # some plugins might not implement caching
 294                             pass
 295                         parsed = True
 296                         display.vvv('Parsed %s inventory source with %s plugin' % (source, plugin_name))
 297                         break
 298                     except AnsibleParserError as e:
 299                         display.debug('%s was not parsable by %s' % (source, plugin_name))
 300                         tb = ''.join(traceback.format_tb(sys.exc_info()[2]))
 301                         failures.append({'src': source, 'plugin': plugin_name, 'exc': e, 'tb': tb})
 302                     except Exception as e:
 303                         display.debug('%s failed while attempting to parse %s' % (plugin_name, source))
 304                         tb = ''.join(traceback.format_tb(sys.exc_info()[2]))
 305                         failures.append({'src': source, 'plugin': plugin_name, 'exc': AnsibleError(e), 'tb': tb})
 306                 else:
 307                     display.vvv("%s declined parsing %s as it did not pass its verify_file() method" % (plugin_name, source))
 308             else:
 309                 if not parsed and failures:
 310                     # only if no plugin processed files should we show errors.
 311                     for fail in failures:
 312                         display.warning(u'\n* Failed to parse %s with %s plugin: %s' % (to_text(fail['src']), fail['plugin'], to_text(fail['exc'])))
 313                         if 'tb' in fail:
 314                             display.vvv(to_text(fail['tb']))
 315                     if C.INVENTORY_ANY_UNPARSED_IS_FAILED:
 316                         raise AnsibleError(u'Completely failed to parse inventory source %s' % (source))
 317         if not parsed:
 318             if source != '/etc/ansible/hosts' or os.path.exists(source):
 319                 # only warn if NOT using the default and if using it, only if the file is present
 320                 display.warning("Unable to parse %s as an inventory source" % source)
 321 
 322         # clear up, jic
 323         self._inventory.current_source = None
 324 
 325         return parsed

 340     def _match_list(self, items, pattern_str):
 341         # compile patterns
 342         try:
 343             if not pattern_str[0] == '~':
 344                 pattern = re.compile(fnmatch.translate(pattern_str))
 345             else:
 346                 pattern = re.compile(pattern_str[1:])
 347         except Exception:
 348             raise AnsibleError('Invalid host list pattern: %s' % pattern_str)
 349 
 350         # apply patterns
 351         results = []
 352         for item in items:
 353             if pattern.match(item):
 354                 results.append(item)
 355         return results

 357     def get_hosts(self, pattern="all", ignore_limits=False, ignore_restrictions=False, order=None):
 358         """
 359         Takes a pattern or list of patterns and returns a list of matching
 360         inventory host names, taking into account any active restrictions
 361         or applied subsets
 362         """
 363 
 364         hosts = []
 365 
 366         # Check if pattern already computed
 367         if isinstance(pattern, list):
 368             pattern_list = pattern[:]
 369         else:
 370             pattern_list = [pattern]
 371 
 372         if pattern_list:
 373             if not ignore_limits and self._subset:
 374                 pattern_list.extend(self._subset)
 375 
 376             if not ignore_restrictions and self._restriction:
 377                 pattern_list.extend(self._restriction)
 378 
 379             # This is only used as a hash key in the self._hosts_patterns_cache dict
 380             # a tuple is faster than stringifying
 381             pattern_hash = tuple(pattern_list)
 382 
 383             if pattern_hash not in self._hosts_patterns_cache:
 384 
 385                 patterns = split_host_pattern(pattern)
 386                 hosts = self._evaluate_patterns(patterns)
 387 
 388                 # mainly useful for hostvars[host] access
 389                 if not ignore_limits and self._subset:
 390                     # exclude hosts not in a subset, if defined
 391                     subset_uuids = set(s._uuid for s in self._evaluate_patterns(self._subset))
 392                     hosts = [h for h in hosts if h._uuid in subset_uuids]
 393 
 394                 if not ignore_restrictions and self._restriction:
 395                     # exclude hosts mentioned in any restriction (ex: failed hosts)
 396                     hosts = [h for h in hosts if h.name in self._restriction]
 397 
 398                 self._hosts_patterns_cache[pattern_hash] = deduplicate_list(hosts)
 399 
 400             # sort hosts list if needed (should only happen when called from strategy)
 401             if order in ['sorted', 'reverse_sorted']:
 402                 hosts = sorted(self._hosts_patterns_cache[pattern_hash][:], key=attrgetter('name'), reverse=(order == 'reverse_sorted'))
 403             elif order == 'reverse_inventory':
 404                 hosts = self._hosts_patterns_cache[pattern_hash][::-1]
 405             else:
 406                 hosts = self._hosts_patterns_cache[pattern_hash][:]
 407                 if order == 'shuffle':
 408                     shuffle(hosts)
 409                 elif order not in [None, 'inventory']:
 410                     raise AnsibleOptionsError("Invalid 'order' specified for inventory hosts: %s" % order)
 411 
 412         return hosts

 414     def _evaluate_patterns(self, patterns):
 415         """
 416         Takes a list of patterns and returns a list of matching host names,
 417         taking into account any negative and intersection patterns.
 418         """
 419 
 420         patterns = order_patterns(patterns)
 421         hosts = []
 422 
 423         for p in patterns:
 424             # avoid resolving a pattern that is a plain host
 425             if p in self._inventory.hosts:
 426                 hosts.append(self._inventory.get_host(p))
 427             else:
 428                 that = self._match_one_pattern(p)
 429                 if p[0] == "!":
 430                     that = set(that)
 431                     hosts = [h for h in hosts if h not in that]
 432                 elif p[0] == "&":
 433                     that = set(that)
 434                     hosts = [h for h in hosts if h in that]
 435                 else:
 436                     existing_hosts = set(y.name for y in hosts)
 437                     hosts.extend([h for h in that if h.name not in existing_hosts])
 438         return hosts

 440     def _match_one_pattern(self, pattern):
 441         """
 442         Takes a single pattern and returns a list of matching host names.
 443         Ignores intersection (&) and exclusion (!) specifiers.
 444 
 445         The pattern may be:
 446 
 447             1. A regex starting with ~, e.g. '~[abc]*'
 448             2. A shell glob pattern with ?/*/[chars]/[!chars], e.g. 'foo*'
 449             3. An ordinary word that matches itself only, e.g. 'foo'
 450 
 451         The pattern is matched using the following rules:
 452 
 453             1. If it's 'all', it matches all hosts in all groups.
 454             2. Otherwise, for each known group name:
 455                 (a) if it matches the group name, the results include all hosts
 456                     in the group or any of its children.
 457                 (b) otherwise, if it matches any hosts in the group, the results
 458                     include the matching hosts.
 459 
 460         This means that 'foo*' may match one or more groups (thus including all
 461         hosts therein) but also hosts in other groups.
 462 
 463         The built-in groups 'all' and 'ungrouped' are special. No pattern can
 464         match these group names (though 'all' behaves as though it matches, as
 465         described above). The word 'ungrouped' can match a host of that name,
 466         and patterns like 'ungr*' and 'al*' can match either hosts or groups
 467         other than all and ungrouped.
 468 
 469         If the pattern matches one or more group names according to these rules,
 470         it may have an optional range suffix to select a subset of the results.
 471         This is allowed only if the pattern is not a regex, i.e. '~foo[1]' does
 472         not work (the [1] is interpreted as part of the regex), but 'foo*[1]'
 473         would work if 'foo*' matched the name of one or more groups.
 474 
 475         Duplicate matches are always eliminated from the results.
 476         """
 477 
 478         if pattern[0] in ("&", "!"):
 479             pattern = pattern[1:]
 480 
 481         if pattern not in self._pattern_cache:
 482             (expr, slice) = self._split_subscript(pattern)
 483             hosts = self._enumerate_matches(expr)
 484             try:
 485                 hosts = self._apply_subscript(hosts, slice)
 486             except IndexError:
 487                 raise AnsibleError("No hosts matched the subscripted pattern '%s'" % pattern)
 488             self._pattern_cache[pattern] = hosts
 489 
 490         return self._pattern_cache[pattern]
 491 
 492     def _split_subscript(self, pattern):
 493         """
 494         Takes a pattern, checks if it has a subscript, and returns the pattern
 495         without the subscript and a (start,end) tuple representing the given
 496         subscript (or None if there is no subscript).
 497 
 498         Validates that the subscript is in the right syntax, but doesn't make
 499         sure the actual indices make sense in context.
 500         """
 501 
 502         # Do not parse regexes for enumeration info
 503         if pattern[0] == '~':
 504             return (pattern, None)
 505 
 506         # We want a pattern followed by an integer or range subscript.
 507         # (We can't be more restrictive about the expression because the
 508         # fnmatch semantics permit [\[:\]] to occur.)
 509 
 510         subscript = None
 511         m = PATTERN_WITH_SUBSCRIPT.match(pattern)
 512         if m:
 513             (pattern, idx, start, sep, end) = m.groups()
 514             if idx:
 515                 subscript = (int(idx), None)
 516             else:
 517                 if not end:
 518                     end = -1
 519                 subscript = (int(start), int(end))
 520                 if sep == '-':
 521                     display.warning("Use [x:y] inclusive subscripts instead of [x-y] which has been removed")
 522 
 523         return (pattern, subscript)
 524 
 525     def _apply_subscript(self, hosts, subscript):
 526         """
 527         Takes a list of hosts and a (start,end) tuple and returns the subset of
 528         hosts based on the subscript (which may be None to return all hosts).
 529         """
 530 
 531         if not hosts or not subscript:
 532             return hosts
 533 
 534         (start, end) = subscript
 535 
 536         if end:
 537             if end == -1:
 538                 end = len(hosts) - 1
 539             return hosts[start:end + 1]
 540         else:
 541             return [hosts[start]]
 542 
 543     def _enumerate_matches(self, pattern):
 544         """
 545         Returns a list of host names matching the given pattern according to the
 546         rules explained above in _match_one_pattern.
 547         """
 548 
 549         results = []
 550         # check if pattern matches group
 551         matching_groups = self._match_list(self._inventory.groups, pattern)
 552         if matching_groups:
 553             for groupname in matching_groups:
 554                 results.extend(self._inventory.groups[groupname].get_hosts())
 555 
 556         # check hosts if no groups matched or it is a regex/glob pattern
 557         if not matching_groups or pattern[0] == '~' or any(special in pattern for special in ('.', '?', '*', '[')):
 558             # pattern might match host
 559             matching_hosts = self._match_list(self._inventory.hosts, pattern)
 560             if matching_hosts:
 561                 for hostname in matching_hosts:
 562                     results.append(self._inventory.hosts[hostname])
 563 
 564         if not results and pattern in C.LOCALHOST:
 565             # get_host autocreates implicit when needed
 566             implicit = self._inventory.get_host(pattern)
 567             if implicit:
 568                 results.append(implicit)
 569 
 570         # Display warning if specified host pattern did not match any groups or hosts
 571         if not results and not matching_groups and pattern != 'all':
 572             msg = "Could not match supplied host pattern, ignoring: %s" % pattern
 573             display.debug(msg)
 574             if C.HOST_PATTERN_MISMATCH == 'warning':
 575                 display.warning(msg)
 576             elif C.HOST_PATTERN_MISMATCH == 'error':
 577                 raise AnsibleError(msg)
 578             # no need to write 'ignore' state
 579 
 580         return results

 597     def restrict_to_hosts(self, restriction):
 598         """
 599         Restrict list operations to the hosts given in restriction.  This is used
 600         to batch serial operations in main playbook code, don't use this for other
 601         reasons.
 602         """
 603         if restriction is None:
 604             return
 605         elif not isinstance(restriction, list):
 606             restriction = [restriction]
 607         self._restriction = set(to_text(h.name) for h in restriction)

 609     def subset(self, subset_pattern):
 610         """
 611         Limits inventory results to a subset of inventory that matches a given
 612         pattern, such as to select a given geographic of numeric slice amongst
 613         a previous 'hosts' selection that only select roles, or vice versa.
 614         Corresponds to --limit parameter to ansible-playbook
 615         """
 616         if subset_pattern is None:
 617             self._subset = None
 618         else:
 619             subset_patterns = split_host_pattern(subset_pattern)
 620             results = []
 621             # allow Unix style @filename data
 622             for x in subset_patterns:
 623                 if not x:
 624                     continue
 625 
 626                 if x[0] == "@":
 627                     b_limit_file = to_bytes(x[1:])
 628                     if not os.path.exists(b_limit_file):
 629                         raise AnsibleError(u'Unable to find limit file %s' % b_limit_file)
 630                     with open(b_limit_file) as fd:
 631                         results.extend([to_text(l.strip()) for l in fd.read().split("\n")])
 632                 else:
 633                     results.append(to_text(x))
 634             self._subset = results

 636     def remove_restriction(self):
