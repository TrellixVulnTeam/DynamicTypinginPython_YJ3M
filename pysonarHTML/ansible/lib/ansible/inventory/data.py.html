  36 class InventoryData(object):
  37     """
  38     Holds inventory data (host and group objects).
  39     Using it's methods should guarantee expected relationships and data.
  40     """
  41 
  42     def __init__(self):
  43 
  44         self.groups = {}
  45         self.hosts = {}
  46 
  47         # provides 'groups' magic var, host object has group_names
  48         self._groups_dict_cache = {}
  49 
  50         # current localhost, implicit or explicit
  51         self.localhost = None
  52 
  53         self.current_source = None
  54 
  55         # Always create the 'all' and 'ungrouped' groups,
  56         for group in ('all', 'ungrouped'):
  57             self.add_group(group)
  58         self.add_child('all', 'ungrouped')
  59 
  60     def serialize(self):
  61         self._groups_dict_cache = None
  62         data = {
  63             'groups': self.groups,
  64             'hosts': self.hosts,
  65             'local': self.localhost,
  66             'source': self.current_source,
  67         }
  68         return data
  69 
  70     def deserialize(self, data):
  71         self._groups_dict_cache = {}
  72         self.hosts = data.get('hosts')
  73         self.groups = data.get('groups')
  74         self.localhost = data.get('local')
  75         self.current_source = data.get('source')
  76 
  77     def _create_implicit_localhost(self, pattern):
  78 
  79         if self.localhost:
  80             new_host = self.localhost
  81         else:
  82             new_host = Host(pattern)
  83 
  84             new_host.address = "127.0.0.1"
  85             new_host.implicit = True
  86 
  87             # set localhost defaults
  88             py_interp = sys.executable
  89             if not py_interp:
  90                 # sys.executable is not set in some cornercases. see issue #13585
  91                 py_interp = '/usr/bin/python'
  92                 display.warning('Unable to determine python interpreter from sys.executable. Using /usr/bin/python default. '
  93                                 'You can correct this by setting ansible_python_interpreter for localhost')
  94             new_host.set_variable("ansible_python_interpreter", py_interp)
  95             new_host.set_variable("ansible_connection", 'local')
  96 
  97             self.localhost = new_host
  98 
  99         return new_host

 101     def reconcile_inventory(self):
 102         ''' Ensure inventory basic rules, run after updates '''
 103 
 104         display.debug('Reconcile groups and hosts in inventory.')
 105         self.current_source = None
 106 
 107         group_names = set()
 108         # set group vars from group_vars/ files and vars plugins
 109         for g in self.groups:
 110             group = self.groups[g]
 111             group_names.add(group.name)
 112 
 113             # ensure all groups inherit from 'all'
 114             if group.name != 'all' and not group.get_ancestors():
 115                 self.add_child('all', group.name)
 116 
 117         host_names = set()
 118         # get host vars from host_vars/ files and vars plugins
 119         for host in self.hosts.values():
 120             host_names.add(host.name)
 121 
 122             mygroups = host.get_groups()
 123 
 124             if self.groups['ungrouped'] in mygroups:
 125                 # clear ungrouped of any incorrectly stored by parser
 126                 if set(mygroups).difference(set([self.groups['all'], self.groups['ungrouped']])):
 127                     self.groups['ungrouped'].remove_host(host)
 128 
 129             elif not host.implicit:
 130                 # add ungrouped hosts to ungrouped, except implicit
 131                 length = len(mygroups)
 132                 if length == 0 or (length == 1 and self.groups['all'] in mygroups):
 133                     self.add_child('ungrouped', host.name)
 134 
 135             # special case for implicit hosts
 136             if host.implicit:
 137                 host.vars = combine_vars(self.groups['all'].get_vars(), host.vars)
 138 
 139         # warn if overloading identifier as both group and host
 140         for conflict in group_names.intersection(host_names):
 141             display.warning("Found both group and host with same name: %s" % conflict)
 142 
 143         self._groups_dict_cache = {}
 144 
 145     def get_host(self, hostname):
 146         ''' fetch host object using name deal with implicit localhost '''
 147 
 148         matching_host = self.hosts.get(hostname, None)
 149 
 150         # if host is not in hosts dict
 151         if matching_host is None and hostname in C.LOCALHOST:
 152             # might need to create implicit localhost
 153             matching_host = self._create_implicit_localhost(hostname)
 154 
 155         return matching_host
 156 
 157     def add_group(self, group):
 158         ''' adds a group to inventory if not there already, returns named actually used '''
 159 
 160         if group:
 161             if not isinstance(group, string_types):
 162                 raise AnsibleError("Invalid group name supplied, expected a string but got %s for %s" % (type(group), group))
 163             if group not in self.groups:
 164                 g = Group(group)
 165                 if g.name not in self.groups:
 166                     self.groups[g.name] = g
 167                     self._groups_dict_cache = {}
 168                     display.debug("Added group %s to inventory" % group)
 169                 group = g.name
 170             else:
 171                 display.debug("group %s already in inventory" % group)
 172         else:
 173             raise AnsibleError("Invalid empty/false group name provided: %s" % group)
 174 
 175         return group
 176 
 177     def remove_group(self, group):
 178 
 179         if group in self.groups:
 180             del self.groups[group]
 181             display.debug("Removed group %s from inventory" % group)
 182             self._groups_dict_cache = {}
 183 
 184         for host in self.hosts:
 185             h = self.hosts[host]
 186             h.remove_group(group)
 187 
 188     def add_host(self, host, group=None, port=None):
 189         ''' adds a host to inventory and possibly a group if not there already '''
 190 
 191         if host:
 192             if not isinstance(host, string_types):
 193                 raise AnsibleError("Invalid host name supplied, expected a string but got %s for %s" % (type(host), host))
 194 
 195             # TODO: add to_safe_host_name
 196             g = None
 197             if group:
 198                 if group in self.groups:
 199                     g = self.groups[group]
 200                 else:
 201                     raise AnsibleError("Could not find group %s in inventory" % group)
 202 
 203             if host not in self.hosts:
 204                 h = Host(host, port)
 205                 self.hosts[host] = h
 206                 if self.current_source:  # set to 'first source' in which host was encountered
 207                     self.set_variable(host, 'inventory_file', self.current_source)
 208                     self.set_variable(host, 'inventory_dir', basedir(self.current_source))
 209                 else:
 210                     self.set_variable(host, 'inventory_file', None)
 211                     self.set_variable(host, 'inventory_dir', None)
 212                 display.debug("Added host %s to inventory" % (host))
 213 
 214                 # set default localhost from inventory to avoid creating an implicit one. Last localhost defined 'wins'.
 215                 if host in C.LOCALHOST:
 216                     if self.localhost is None:
 217                         self.localhost = self.hosts[host]
 218                         display.vvvv("Set default localhost to %s" % h)
 219                     else:
 220                         display.warning("A duplicate localhost-like entry was found (%s). First found localhost was %s" % (h, self.localhost.name))
 221             else:
 222                 h = self.hosts[host]
 223 
 224             if g:
 225                 g.add_host(h)
 226                 self._groups_dict_cache = {}
 227                 display.debug("Added host %s to group %s" % (host, group))
 228         else:
 229             raise AnsibleError("Invalid empty host name provided: %s" % host)
 230 
 231         return host
 232 
 233     def remove_host(self, host):
 234 
 235         if host.name in self.hosts:
 236             del self.hosts[host.name]
 237 
 238         for group in self.groups:
 239             g = self.groups[group]
 240             g.remove_host(host)
 241 
 242     def set_variable(self, entity, varname, value):
 243         ''' sets a variable for an inventory object '''
 244 
 245         if entity in self.groups:
 246             inv_object = self.groups[entity]
 247         elif entity in self.hosts:
 248             inv_object = self.hosts[entity]
 249         else:
 250             raise AnsibleError("Could not identify group or host named %s" % entity)
 251 
 252         inv_object.set_variable(varname, value)
 253         display.debug('set %s for %s' % (varname, entity))
 254 
 255     def add_child(self, group, child):
 256         ''' Add host or group to group '''
 257         added = False
 258         if group in self.groups:
 259             g = self.groups[group]
 260             if child in self.groups:
 261                 added = g.add_child_group(self.groups[child])
 262             elif child in self.hosts:
 263                 added = g.add_host(self.hosts[child])
 264             else:
 265                 raise AnsibleError("%s is not a known host nor group" % child)
 266             self._groups_dict_cache = {}
 267             display.debug('Group %s now contains %s' % (group, child))
 268         else:
 269             raise AnsibleError("%s is not a known group" % group)
 270         return added
 271 
 272     def get_groups_dict(self):
 273         """
 274         We merge a 'magic' var 'groups' with group name keys and hostname list values into every host variable set. Cache for speed.
 275         """
 276         if not self._groups_dict_cache:
 277             for (group_name, group) in iteritems(self.groups):
 278                 self._groups_dict_cache[group_name] = [h.name for h in group.get_hosts()]
 279 
 280         return self._groups_dict_cache
