  22 from ansible.inventory.group import Group

  24 from ansible.utils.vars import combine_vars, get_unique_id

  29 class Host:
  30     ''' a single ansible host '''
  31 
  32     # __slots__ = [ 'name', 'vars', 'groups' ]
  33 
  34     def __getstate__(self):
  35         return self.serialize()
  36 
  37     def __setstate__(self, data):
  38         return self.deserialize(data)
  39 
  40     def __eq__(self, other):
  41         if not isinstance(other, Host):
  42             return False
  43         return self._uuid == other._uuid
  44 
  45     def __ne__(self, other):
  46         return not self.__eq__(other)
  47 
  48     def __hash__(self):
  49         return hash(self.name)
  50 
  51     def __str__(self):
  52         return self.get_name()
  53 
  54     def __repr__(self):
  55         return self.get_name()
  56 
  57     def serialize(self):
  58         groups = []
  59         for group in self.groups:
  60             groups.append(group.serialize())
  61 
  62         return dict(
  63             name=self.name,
  64             vars=self.vars.copy(),
  65             address=self.address,
  66             uuid=self._uuid,
  67             groups=groups,
  68             implicit=self.implicit,
  69         )
  70 
  71     def deserialize(self, data):
  72         self.__init__(gen_uuid=False)
  73 
  74         self.name = data.get('name')
  75         self.vars = data.get('vars', dict())
  76         self.address = data.get('address', '')
  77         self._uuid = data.get('uuid', None)
  78         self.implicit = data.get('implicit', False)
  79 
  80         groups = data.get('groups', [])
  81         for group_data in groups:
  82             g = Group()
  83             g.deserialize(group_data)
  84             self.groups.append(g)
  85 
  86     def __init__(self, name=None, port=None, gen_uuid=True):
  87 
  88         self.vars = {}
  89         self.groups = []
  90         self._uuid = None
  91 
  92         self.name = name
  93         self.address = name
  94 
  95         if port:
  96             self.set_variable('ansible_port', int(port))
  97 
  98         if gen_uuid:
  99             self._uuid = get_unique_id()
 100         self.implicit = False
 101 
 102     def get_name(self):
 103         return self.name
 104 
 105     def populate_ancestors(self, additions=None):
 106         # populate ancestors
 107         if additions is None:
 108             for group in self.groups:
 109                 self.add_group(group)
 110         else:
 111             for group in additions:
 112                 if group not in self.groups:
 113                     self.groups.append(group)
 114 
 115     def add_group(self, group):
 116         added = False
 117         # populate ancestors first
 118         for oldg in group.get_ancestors():
 119             if oldg not in self.groups:
 120                 self.groups.append(oldg)
 121 
 122         # actually add group
 123         if group not in self.groups:
 124             self.groups.append(group)
 125             added = True
 126         return added
 127 
 128     def remove_group(self, group):
 129         removed = False
 130         if group in self.groups:
 131             self.groups.remove(group)
 132             removed = True
 133 
 134             # remove exclusive ancestors, xcept all!
 135             for oldg in group.get_ancestors():
 136                 if oldg.name != 'all':
 137                     for childg in self.groups:
 138                         if oldg in childg.get_ancestors():
 139                             break
 140                     else:
 141                         self.remove_group(oldg)
 142         return removed
 143 
 144     def set_variable(self, key, value):
 145         if key in self.vars and isinstance(self.vars[key], MutableMapping) and isinstance(value, Mapping):
 146             self.vars[key] = combine_vars(self.vars[key], value)
 147         else:
 148             self.vars[key] = value
 149 
 150     def get_groups(self):
 151         return self.groups
 152 
 153     def get_magic_vars(self):
 154         results = {}
 155         results['inventory_hostname'] = self.name
 156         results['inventory_hostname_short'] = self.name.split('.')[0]
 157         results['group_names'] = sorted([g.name for g in self.get_groups() if g.name != 'all'])
 158 
 159         return results
 160 
 161     def get_vars(self):
 162         return combine_vars(self.vars, self.get_magic_vars())
