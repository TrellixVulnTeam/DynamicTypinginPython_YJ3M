  31 def get_reserved_names(include_private=True):
  32     ''' this function returns the list of reserved names associated with play objects'''
  33 
  34     public = set()
  35     private = set()
  36     result = set()
  37 
  38     # FIXME: find a way to 'not hardcode', possibly need role deps/includes
  39     class_list = [Play, Role, Block, Task]
  40 
  41     for aclass in class_list:
  42         aobj = aclass()
  43 
  44         # build ordered list to loop over and dict with attributes
  45         for attribute in aobj.__dict__['_attributes']:
  46             if 'private' in attribute:
  47                 private.add(attribute)
  48             else:
  49                 public.add(attribute)
  50 
  51     # local_action is implicit with action
  52     if 'action' in public:
  53         public.add('local_action')
  54 
  55     # loop implies with_
  56     # FIXME: remove after with_ is not only deprecated but removed
  57     if 'loop' in private or 'loop' in public:
  58         public.add('with_')
  59 
  60     if include_private:
  61         result = public.union(private)
  62     else:
  63         result = public
  64 
  65     return result
  66

  68 def warn_if_reserved(myvars):
  69     ''' this function warns if any variable passed conflicts with internally reserved names '''
  70 
  71     varnames = set(myvars)
  72     varnames.discard('vars')  # we add this one internally, so safe to ignore
  73     for varname in varnames.intersection(_RESERVED_NAMES):
  74         display.warning('Found variable using reserved name: %s' % varname)
  75 
  76 
  77 def is_reserved_name(name):
  78     return name in _RESERVED_NAMES
  79 
  80 
  81 _RESERVED_NAMES = frozenset(get_reserved_names())
