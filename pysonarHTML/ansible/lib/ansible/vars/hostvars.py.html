  47 # Note -- this is a Mapping, not a MutableMapping
  48 class HostVars(Mapping):
  49     ''' A special view of vars_cache that adds values from the inventory when needed. '''
  50 
  51     def __init__(self, inventory, variable_manager, loader):
  52         self._inventory = inventory
  53         self._loader = loader
  54         self._variable_manager = variable_manager
  55         variable_manager._hostvars = self
  56 
  57     def set_variable_manager(self, variable_manager):
  58         self._variable_manager = variable_manager
  59         variable_manager._hostvars = self
  60 
  61     def set_inventory(self, inventory):
  62         self._inventory = inventory
  63 
  64     def _find_host(self, host_name):
  65         # does not use inventory.hosts so it can create localhost on demand
  66         return self._inventory.get_host(host_name)
  67 
  68     def raw_get(self, host_name):
  69         '''
  70         Similar to __getitem__, however the returned data is not run through
  71         the templating engine to expand variables in the hostvars.
  72         '''
  73         host = self._find_host(host_name)
  74         if host is None:
  75             return AnsibleUndefined(name="hostvars['%s']" % host_name)
  76 
  77         return self._variable_manager.get_vars(host=host, include_hostvars=False)
  78 
  79     def __setstate__(self, state):
  80         self.__dict__.update(state)
  81 
  82         # Methods __getstate__ and __setstate__ of VariableManager do not
  83         # preserve _loader and _hostvars attributes to improve pickle
  84         # performance and memory utilization. Since HostVars holds values
  85         # of those attributes already, assign them if needed.
  86         if self._variable_manager._loader is None:
  87             self._variable_manager._loader = self._loader
  88 
  89         if self._variable_manager._hostvars is None:
  90             self._variable_manager._hostvars = self
  91 
  92     def __getitem__(self, host_name):
  93         data = self.raw_get(host_name)
  94         if isinstance(data, AnsibleUndefined):
  95             return data
  96         return HostVarsVars(data, loader=self._loader)
  97 
  98     def set_host_variable(self, host, varname, value):
  99         self._variable_manager.set_host_variable(host, varname, value)
 100 
 101     def set_nonpersistent_facts(self, host, facts):
 102         self._variable_manager.set_nonpersistent_facts(host, facts)
 103 
 104     def set_host_facts(self, host, facts):
 105         self._variable_manager.set_host_facts(host, facts)
 106 
 107     def __contains__(self, host_name):
 108         # does not use inventory.hosts so it can create localhost on demand
 109         return self._find_host(host_name) is not None
 110 
 111     def __iter__(self):
 112         for host in self._inventory.hosts:
 113             yield host
 114 
 115     def __len__(self):
 116         return len(self._inventory.hosts)
 117 
 118     def __repr__(self):
 119         out = {}
 120         for host in self._inventory.hosts:
 121             out[host] = self.get(host)
 122         return repr(out)
 123 
 124     def __deepcopy__(self, memo):
 125         # We do not need to deepcopy because HostVars is immutable,
 126         # however we have to implement the method so we can deepcopy
 127         # variables' dicts that contain HostVars.
 128         return self
 129 
 130 
 131 class HostVarsVars(Mapping):
 132 
 133     def __init__(self, variables, loader):
 134         self._vars = variables
 135         self._loader = loader
 136 
 137     def __getitem__(self, var):
 138         templar = Templar(variables=self._vars, loader=self._loader)
 139         foo = templar.template(self._vars[var], fail_on_undefined=False, static_vars=STATIC_VARS)
 140         return foo
 141 
 142     def __contains__(self, var):
 143         return (var in self._vars)
 144 
 145     def __iter__(self):
 146         for var in self._vars.keys():
 147             yield var
 148 
 149     def __len__(self):
 150         return len(self._vars.keys())
 151 
 152     def __repr__(self):
 153         templar = Templar(variables=self._vars, loader=self._loader)
 154         return repr(templar.template(self._vars, fail_on_undefined=False, static_vars=STATIC_VARS))
