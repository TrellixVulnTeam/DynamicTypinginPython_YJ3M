  22 def module_response_deepcopy(v):
  23     """Function to create a deep copy of module response data
  24 
  25     Designed to be used within the Ansible "engine" to improve performance
  26     issues where ``copy.deepcopy`` was used previously, largely with CPU
  27     and memory contention.
  28 
  29     This only supports the following data types, and was designed to only
  30     handle specific workloads:
  31 
  32     * ``dict``
  33     * ``list``
  34 
  35     The data we pass here will come from a serialization such
  36     as JSON, so we shouldn't have need for other data types such as
  37     ``set`` or ``tuple``.
  38 
  39     Take note that this function should not be used extensively as a
  40     replacement for ``deepcopy`` due to the naive way in which this
  41     handles other data types.
  42 
  43     Do not expect uses outside of those listed below to maintain
  44     backwards compatibility, in case we need to extend this function
  45     to handle our specific needs:
  46 
  47     * ``ansible.executor.task_result.TaskResult.clean_copy``
  48     * ``ansible.vars.clean.clean_facts``
  49     * ``ansible.vars.namespace_facts``
  50     """
  51     if isinstance(v, dict):
  52         ret = v.copy()
  53         items = six.iteritems(ret)
  54     elif isinstance(v, list):
  55         ret = v[:]
  56         items = enumerate(ret)
  57     else:
  58         return v
  59 
  60     for key, value in items:
  61         if isinstance(value, (dict, list)):
  62             ret[key] = module_response_deepcopy(value)
  63         else:
  64             ret[key] = value
  65 
  66     return ret

  69 def strip_internal_keys(dirty, exceptions=None):
  70     # All keys starting with _ansible_ are internal, so change the 'dirty' mapping and remove them.
  71 
  72     if exceptions is None:
  73         exceptions = tuple()
  74 
  75     if isinstance(dirty, MutableSequence):
  76 
  77         for element in dirty:
  78             if isinstance(element, (MutableMapping, MutableSequence)):
  79                 strip_internal_keys(element, exceptions=exceptions)
  80 
  81     elif isinstance(dirty, MutableMapping):
  82 
  83         # listify to avoid updating dict while iterating over it
  84         for k in list(dirty.keys()):
  85             if isinstance(k, six.string_types):
  86                 if k.startswith('_ansible_') and k not in exceptions:
  87                     del dirty[k]
  88                     continue
  89 
  90             if isinstance(dirty[k], (MutableMapping, MutableSequence)):
  91                 strip_internal_keys(dirty[k], exceptions=exceptions)
  92     else:
  93         raise AnsibleError("Cannot strip invalid keys from %s" % type(dirty))
  94 
  95     return dirty

  98 def remove_internal_keys(data):
  99     '''
 100     More nuanced version of strip_internal_keys
 101     '''
 102     for key in list(data.keys()):
 103         if (key.startswith('_ansible_') and key != '_ansible_parsed') or key in C.INTERNAL_RESULT_KEYS:
 104             display.warning("Removed unexpected internal key in module return: %s = %s" % (key, data[key]))
 105             del data[key]
 106 
 107     # remove bad/empty internal keys
 108     for key in ['warnings', 'deprecations']:
 109         if key in data and not data[key]:
 110             del data[key]
 111 
 112     # cleanse fact values that are allowed from actions but not modules
 113     for key in list(data.get('ansible_facts', {}).keys()):
 114         if key.startswith('discovered_interpreter_') or key.startswith('ansible_discovered_interpreter_'):
 115             del data['ansible_facts'][key]

 118 def clean_facts(facts):
 119     ''' remove facts that can override internal keys or otherwise deemed unsafe '''
 120     data = module_response_deepcopy(facts)
 121 
 122     remove_keys = set()
 123     fact_keys = set(data.keys())
 124     # first we add all of our magic variable names to the set of
 125     # keys we want to remove from facts
 126     # NOTE: these will eventually disappear in favor of others below
 127     for magic_var in C.MAGIC_VARIABLE_MAPPING:
 128         remove_keys.update(fact_keys.intersection(C.MAGIC_VARIABLE_MAPPING[magic_var]))
 129 
 130     # remove common connection vars
 131     remove_keys.update(fact_keys.intersection(C.COMMON_CONNECTION_VARS))
 132 
 133     # next we remove any connection plugin specific vars
 134     for conn_path in connection_loader.all(path_only=True):
 135         conn_name = os.path.splitext(os.path.basename(conn_path))[0]
 136         re_key = re.compile('^ansible_%s_' % conn_name)
 137         for fact_key in fact_keys:
 138             # most lightweight VM or container tech creates devices with this pattern, this avoids filtering them out
 139             if (re_key.match(fact_key) and not fact_key.endswith(('_bridge', '_gwbridge'))) or fact_key.startswith('ansible_become_'):
 140                 remove_keys.add(fact_key)
 141 
 142     # remove some KNOWN keys
 143     for hard in C.RESTRICTED_RESULT_KEYS + C.INTERNAL_RESULT_KEYS:
 144         if hard in fact_keys:
 145             remove_keys.add(hard)
 146 
 147     # finally, we search for interpreter keys to remove
 148     re_interp = re.compile('^ansible_.*_interpreter$')
 149     for fact_key in fact_keys:
 150         if re_interp.match(fact_key):
 151             remove_keys.add(fact_key)
 152     # then we remove them (except for ssh host keys)
 153     for r_key in remove_keys:
 154         if not r_key.startswith('ansible_ssh_host_key_'):
 155             try:
 156                 r_val = to_text(data[r_key])
 157                 if len(r_val) > 24:
 158                     r_val = '%s ... %s' % (r_val[:13], r_val[-6:])
 159             except Exception:
 160                 r_val = ' <failed to convert value to a string> '
 161             display.warning("Removed restricted key from module data: %s = %s" % (r_key, r_val))
 162             del data[r_key]
 163 
 164     return strip_internal_keys(data)

 167 def namespace_facts(facts):
 168     ''' return all facts inside 'ansible_facts' w/o an ansible_ prefix '''
 169     deprefixed = {}
 170     for k in facts:
 171         if k.startswith('ansible_') and k not in ('ansible_local',):
 172             deprefixed[k[8:]] = module_response_deepcopy(facts[k])
 173         else:
 174             deprefixed[k] = module_response_deepcopy(facts[k])
 175 
 176     return {'ansible_facts': deprefixed}
