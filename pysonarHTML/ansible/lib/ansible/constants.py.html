  22 def _warning(msg):
  23     ''' display is not guaranteed here, nor it being the full class, but try anyways, fallback to sys.stderr.write '''
  24     try:
  25         from ansible.utils.display import Display
  26         Display().warning(msg)
  27     except Exception:
  28         import sys
  29         sys.stderr.write(' [WARNING] %s\n' % (msg))

  32 def _deprecated(msg, version='2.8'):
  33     ''' display is not guaranteed here, nor it being the full class, but try anyways, fallback to sys.stderr.write '''
  34     try:
  35         from ansible.utils.display import Display
  36         Display().deprecated(msg, version=version)
  37     except Exception:
  38         import sys
  39         sys.stderr.write(' [DEPRECATED] %s, to be removed in %s\n' % (msg, version))

  47 class _DeprecatedSequenceConstant(Sequence):
  48     def __init__(self, value, msg, version):
  49         self._value = value
  50         self._msg = msg
  51         self._version = version
  52 
  53     def __len__(self):
  54         _deprecated(self._msg, version=self._version)
  55         return len(self._value)
  56 
  57     def __getitem__(self, y):
  58         _deprecated(self._msg, version=self._version)
  59         return self._value[y]

 141 # POPULATE SETTINGS FROM CONFIG ###
 142 config = ConfigManager()
 143 
 144 # Generate constants from config
 145 for setting in config.data.get_settings():
 146 
 147     value = setting.value
 148     if setting.origin == 'default' and \
 149        isinstance(setting.value, string_types) and \
 150        (setting.value.startswith('{{') and setting.value.endswith('}}')):
 151         try:
 152             t = Template(setting.value)
 153             value = t.render(vars())
 154             try:
 155                 value = literal_eval(value)
 156             except ValueError:
 157                 pass  # not a python data structure
 158         except Exception:
 159             pass  # not templatable
 160 
 161         value = ensure_type(value, setting.type)
 162 
 163     set_constant(setting.name, value)
 164 
 165 for warn in config.WARNINGS:
 166     _warning(warn)
