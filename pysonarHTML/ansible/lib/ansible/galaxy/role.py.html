  47     SUPPORTED_SCMS = set(['git', 'hg'])
  48     META_MAIN = (os.path.join('meta', 'main.yml'), os.path.join('meta', 'main.yaml'))
  49     META_INSTALL = os.path.join('meta', '.galaxy_install_info')
  50     META_REQUIREMENTS = (os.path.join('meta', 'requirements.yml'), os.path.join('meta', 'requirements.yaml'))

  53     def __init__(self, galaxy, api, name, src=None, version=None, scm=None, path=None):
  54 
  55         self._metadata = None
  56         self._requirements = None
  57         self._install_info = None
  58         self._validate_certs = not context.CLIARGS['ignore_certs']
  59 
  60         display.debug('Validate TLS certificates: %s' % self._validate_certs)
  61 
  62         self.galaxy = galaxy
  63         self.api = api
  64 
  65         self.name = name
  66         self.version = version
  67         self.src = src or name
  68         self.scm = scm
  69         self.paths = [os.path.join(x, self.name) for x in galaxy.roles_paths]
  70 
  71         if path is not None:
  72             if not path.endswith(os.path.join(os.path.sep, self.name)):
  73                 path = os.path.join(path, self.name)
  74             else:
  75                 # Look for a meta/main.ya?ml inside the potential role dir in case
  76                 #  the role name is the same as parent directory of the role.
  77                 #
  78                 # Example:
  79                 #   ./roles/testing/testing/meta/main.yml
  80                 for meta_main in self.META_MAIN:
  81                     if os.path.exists(os.path.join(path, name, meta_main)):
  82                         path = os.path.join(path, self.name)
  83                         break
  84             self.path = path
  85         else:
  86             # use the first path by default
  87             self.path = os.path.join(galaxy.roles_paths[0], self.name)

  89     def __repr__(self):
  90         """
  91         Returns "rolename (version)" if version is not null
  92         Returns "rolename" otherwise
  93         """
  94         if self.version:
  95             return "%s (%s)" % (self.name, self.version)
  96         else:
  97             return self.name
  98 
  99     def __eq__(self, other):
 100         return self.name == other.name

 102     @property
 103     def metadata(self):
 104         """
 105         Returns role metadata
 106         """
 107         if self._metadata is None:
 108             for path in self.paths:
 109                 for meta_main in self.META_MAIN:
 110                     meta_path = os.path.join(path, meta_main)
 111                     if os.path.isfile(meta_path):
 112                         try:
 113                             with open(meta_path, 'r') as f:
 114                                 self._metadata = yaml.safe_load(f)
 115                         except Exception:
 116                             display.vvvvv("Unable to load metadata for %s" % self.name)
 117                             return False
 118                         break
 119 
 120         return self._metadata
 121 
 122     @property
 123     def install_info(self):
 124         """
 125         Returns role install info
 126         """
 127         if self._install_info is None:
 128 
 129             info_path = os.path.join(self.path, self.META_INSTALL)
 130             if os.path.isfile(info_path):
 131                 try:
 132                     f = open(info_path, 'r')
 133                     self._install_info = yaml.safe_load(f)
 134                 except Exception:
 135                     display.vvvvv("Unable to load Galaxy install info for %s" % self.name)
 136                     return False
 137                 finally:
 138                     f.close()
 139         return self._install_info
 140 
 141     @property
 142     def _exists(self):
 143         for path in self.paths:
 144             if os.path.isdir(path):
 145                 return True
 146 
 147         return False

 149     def _write_galaxy_install_info(self):
 150         """
 151         Writes a YAML-formatted file to the role's meta/ directory
 152         (named .galaxy_install_info) which contains some information
 153         we can use later for commands like 'list' and 'info'.
 154         """
 155 
 156         info = dict(
 157             version=self.version,
 158             install_date=datetime.datetime.utcnow().strftime("%c"),
 159         )
 160         if not os.path.exists(os.path.join(self.path, 'meta')):
 161             os.makedirs(os.path.join(self.path, 'meta'))
 162         info_path = os.path.join(self.path, self.META_INSTALL)
 163         with open(info_path, 'w+') as f:
 164             try:
 165                 self._install_info = yaml.safe_dump(info, f)
 166             except Exception:
 167                 return False
 168 
 169         return True

 171     def remove(self):
 172         """
 173         Removes the specified role from the roles path.
 174         There is a sanity check to make sure there's a meta/main.yml file at this
 175         path so the user doesn't blow away random directories.
 176         """
 177         if self.metadata:
 178             try:
 179                 rmtree(self.path)
 180                 return True
 181             except Exception:
 182                 pass
 183 
 184         return False

 186     def fetch(self, role_data):
 187         """
 188         Downloads the archived role to a temp location based on role data
 189         """
 190         if role_data:
 191 
 192             # first grab the file and save it to a temp location
 193             if "github_user" in role_data and "github_repo" in role_data:
 194                 archive_url = 'https://github.com/%s/%s/archive/%s.tar.gz' % (role_data["github_user"], role_data["github_repo"], self.version)
 195             else:
 196                 archive_url = self.src
 197 
 198             display.display("- downloading role from %s" % archive_url)
 199 
 200             try:
 201                 url_file = open_url(archive_url, validate_certs=self._validate_certs, http_agent=user_agent())
 202                 temp_file = tempfile.NamedTemporaryFile(delete=False)
 203                 data = url_file.read()
 204                 while data:
 205                     temp_file.write(data)
 206                     data = url_file.read()
 207                 temp_file.close()
 208                 return temp_file.name
 209             except Exception as e:
 210                 display.error(u"failed to download the file: %s" % to_text(e))
 211 
 212         return False

 214     def install(self):
 215 
 216         if self.scm:
 217             # create tar file from scm url
 218             tmp_file = RoleRequirement.scm_archive_role(keep_scm_meta=context.CLIARGS['keep_scm_meta'], **self.spec)
 219         elif self.src:
 220             if os.path.isfile(self.src):
 221                 tmp_file = self.src
 222             elif '://' in self.src:
 223                 role_data = self.src
 224                 tmp_file = self.fetch(role_data)
 225             else:
 226                 role_data = self.api.lookup_role_by_name(self.src)
 227                 if not role_data:
 228                     raise AnsibleError("- sorry, %s was not found on %s." % (self.src, self.api.api_server))
 229 
 230                 if role_data.get('role_type') == 'APP':
 231                     # Container Role
 232                     display.warning("%s is a Container App role, and should only be installed using Ansible "
 233                                     "Container" % self.name)
 234 
 235                 role_versions = self.api.fetch_role_related('versions', role_data['id'])
 236                 if not self.version:
 237                     # convert the version names to LooseVersion objects
 238                     # and sort them to get the latest version. If there
 239                     # are no versions in the list, we'll grab the head
 240                     # of the master branch
 241                     if len(role_versions) > 0:
 242                         loose_versions = [LooseVersion(a.get('name', None)) for a in role_versions]
 243                         try:
 244                             loose_versions.sort()
 245                         except TypeError:
 246                             raise AnsibleError(
 247                                 'Unable to compare role versions (%s) to determine the most recent version due to incompatible version formats. '
 248                                 'Please contact the role author to resolve versioning conflicts, or specify an explicit role version to '
 249                                 'install.' % ', '.join([v.vstring for v in loose_versions])
 250                             )
 251                         self.version = to_text(loose_versions[-1])
 252                     elif role_data.get('github_branch', None):
 253                         self.version = role_data['github_branch']
 254                     else:
 255                         self.version = 'master'
 256                 elif self.version != 'master':
 257                     if role_versions and to_text(self.version) not in [a.get('name', None) for a in role_versions]:
 258                         raise AnsibleError("- the specified version (%s) of %s was not found in the list of available versions (%s)." % (self.version,
 259                                                                                                                                          self.name,
 260                                                                                                                                          role_versions))
 261 
 262                 # check if there's a source link for our role_version
 263                 for role_version in role_versions:
 264                     if role_version['name'] == self.version and 'source' in role_version:
 265                         self.src = role_version['source']
 266 
 267                 tmp_file = self.fetch(role_data)
 268 
 269         else:
 270             raise AnsibleError("No valid role data found")
 271 
 272         if tmp_file:
 273 
 274             display.debug("installing from %s" % tmp_file)
 275 
 276             if not tarfile.is_tarfile(tmp_file):
 277                 raise AnsibleError("the downloaded file does not appear to be a valid tar archive.")
 278             else:
 279                 role_tar_file = tarfile.open(tmp_file, "r")
 280                 # verify the role's meta file
 281                 meta_file = None
 282                 members = role_tar_file.getmembers()
 283                 # next find the metadata file
 284                 for member in members:
 285                     for meta_main in self.META_MAIN:
 286                         if meta_main in member.name:
 287                             # Look for parent of meta/main.yml
 288                             # Due to possibility of sub roles each containing meta/main.yml
 289                             # look for shortest length parent
 290                             meta_parent_dir = os.path.dirname(os.path.dirname(member.name))
 291                             if not meta_file:
 292                                 archive_parent_dir = meta_parent_dir
 293                                 meta_file = member
 294                             else:
 295                                 if len(meta_parent_dir) < len(archive_parent_dir):
 296                                     archive_parent_dir = meta_parent_dir
 297                                     meta_file = member
 298                 if not meta_file:
 299                     raise AnsibleError("this role does not appear to have a meta/main.yml file.")
 300                 else:
 301                     try:
 302                         self._metadata = yaml.safe_load(role_tar_file.extractfile(meta_file))
 303                     except Exception:
 304                         raise AnsibleError("this role does not appear to have a valid meta/main.yml file.")
 305 
 306                 # we strip off any higher-level directories for all of the files contained within
 307                 # the tar file here. The default is 'github_repo-target'. Gerrit instances, on the other
 308                 # hand, does not have a parent directory at all.
 309                 installed = False
 310                 while not installed:
 311                     display.display("- extracting %s to %s" % (self.name, self.path))
 312                     try:
 313                         if os.path.exists(self.path):
 314                             if not os.path.isdir(self.path):
 315                                 raise AnsibleError("the specified roles path exists and is not a directory.")
 316                             elif not context.CLIARGS.get("force", False):
 317                                 raise AnsibleError("the specified role %s appears to already exist. Use --force to replace it." % self.name)
 318                             else:
 319                                 # using --force, remove the old path
 320                                 if not self.remove():
 321                                     raise AnsibleError("%s doesn't appear to contain a role.\n  please remove this directory manually if you really "
 322                                                        "want to put the role here." % self.path)
 323                         else:
 324                             os.makedirs(self.path)
 325 
 326                         # now we do the actual extraction to the path
 327                         for member in members:
 328                             # we only extract files, and remove any relative path
 329                             # bits that might be in the file for security purposes
 330                             # and drop any containing directory, as mentioned above
 331                             if member.isreg() or member.issym():
 332                                 n_member_name = to_native(member.name)
 333                                 n_archive_parent_dir = to_native(archive_parent_dir)
 334                                 n_parts = n_member_name.replace(n_archive_parent_dir, "", 1).split(os.sep)
 335                                 n_final_parts = []
 336                                 for n_part in n_parts:
 337                                     if n_part != '..' and '~' not in n_part and '$' not in n_part:
 338                                         n_final_parts.append(n_part)
 339                                 member.name = os.path.join(*n_final_parts)
 340                                 role_tar_file.extract(member, to_native(self.path))
 341 
 342                         # write out the install info file for later use
 343                         self._write_galaxy_install_info()
 344                         installed = True
 345                     except OSError as e:
 346                         error = True
 347                         if e.errno == errno.EACCES and len(self.paths) > 1:
 348                             current = self.paths.index(self.path)
 349                             if len(self.paths) > current:
 350                                 self.path = self.paths[current + 1]
 351                                 error = False
 352                         if error:
 353                             raise AnsibleError("Could not update files in %s: %s" % (self.path, to_native(e)))
 354 
 355                 # return the parsed yaml metadata
 356                 display.display("- %s was installed successfully" % str(self))
 357                 if not (self.src and os.path.isfile(self.src)):
 358                     try:
 359                         os.unlink(tmp_file)
 360                     except (OSError, IOError) as e:
 361                         display.warning(u"Unable to remove tmp file (%s): %s" % (tmp_file, to_text(e)))
 362                 return True
 363 
 364         return False

 366     @property
 367     def spec(self):
 368         """
 369         Returns role spec info
 370         {
 371            'scm': 'git',
 372            'src': 'http://git.example.com/repos/repo.git',
 373            'version': 'v1.0',
 374            'name': 'repo'
 375         }
 376         """
 377         return dict(scm=self.scm, src=self.src, version=self.version, name=self.name)

 379     @property
 380     def requirements(self):
 381         """
 382         Returns role requirements
 383         """
 384         if self._requirements is None:
 385             self._requirements = []
 386             for meta_requirements in self.META_REQUIREMENTS:
 387                 meta_path = os.path.join(self.path, meta_requirements)
 388                 if os.path.isfile(meta_path):
 389                     try:
 390                         f = open(meta_path, 'r')
 391                         self._requirements = yaml.safe_load(f)
 392                     except Exception:
 393                         display.vvvvv("Unable to load requirements for %s" % self.name)
 394                     finally:
 395                         f.close()
 396 
 397                     break
 398 
 399         return self._requirements
