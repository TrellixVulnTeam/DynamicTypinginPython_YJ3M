  34 def g_connect(versions):
  35     """
  36     Wrapper to lazily initialize connection info to Galaxy and verify the API versions required are available on the
  37     endpoint.
  38 
  39     :param versions: A list of API versions that the function supports.
  40     """
  41     def decorator(method):
  42         def wrapped(self, *args, **kwargs):
  43             if not self._available_api_versions:
  44                 display.vvvv("Initial connection to galaxy_server: %s" % self.api_server)
  45 
  46                 # Determine the type of Galaxy server we are talking to. First try it unauthenticated then with Bearer
  47                 # auth for Automation Hub.
  48                 n_url = self.api_server
  49                 error_context_msg = 'Error when finding available api versions from %s (%s)' % (self.name, n_url)
  50 
  51                 if self.api_server == 'https://galaxy.ansible.com' or self.api_server == 'https://galaxy.ansible.com/':
  52                     n_url = 'https://galaxy.ansible.com/api/'
  53 
  54                 try:
  55                     data = self._call_galaxy(n_url, method='GET', error_context_msg=error_context_msg)
  56                 except (AnsibleError, GalaxyError, ValueError, KeyError) as err:
  57                     # Either the URL doesnt exist, or other error. Or the URL exists, but isn't a galaxy API
  58                     # root (not JSON, no 'available_versions') so try appending '/api/'
  59                     if n_url.endswith('/api') or n_url.endswith('/api/'):
  60                         raise
  61 
  62                     # Let exceptions here bubble up but raise the original if this returns a 404 (/api/ wasn't found).
  63                     n_url = _urljoin(n_url, '/api/')
  64                     try:
  65                         data = self._call_galaxy(n_url, method='GET', error_context_msg=error_context_msg)
  66                     except GalaxyError as new_err:
  67                         if new_err.http_code == 404:
  68                             raise err
  69                         raise
  70 
  71                 if 'available_versions' not in data:
  72                     raise AnsibleError("Tried to find galaxy API root at %s but no 'available_versions' are available "
  73                                        "on %s" % (n_url, self.api_server))
  74 
  75                 # Update api_server to point to the "real" API root, which in this case could have been the configured
  76                 # url + '/api/' appended.
  77                 self.api_server = n_url
  78 
  79                 # Default to only supporting v1, if only v1 is returned we also assume that v2 is available even though
  80                 # it isn't returned in the available_versions dict.
  81                 available_versions = data.get('available_versions', {u'v1': u'v1/'})
  82                 if list(available_versions.keys()) == [u'v1']:
  83                     available_versions[u'v2'] = u'v2/'
  84 
  85                 self._available_api_versions = available_versions
  86                 display.vvvv("Found API version '%s' with Galaxy server %s (%s)"
  87                              % (', '.join(available_versions.keys()), self.name, self.api_server))
  88 
  89             # Verify that the API versions the function works with are available on the server specified.
  90             available_versions = set(self._available_api_versions.keys())
  91             common_versions = set(versions).intersection(available_versions)
  92             if not common_versions:
  93                 raise AnsibleError("Galaxy action %s requires API versions '%s' but only '%s' are available on %s %s"
  94                                    % (method.__name__, ", ".join(versions), ", ".join(available_versions),
  95                                       self.name, self.api_server))
  96 
  97             return method(self, *args, **kwargs)
  98         return wrapped
  99     return decorator

 102 def _urljoin(*args):
 103     return '/'.join(to_native(a, errors='surrogate_or_strict').strip('/') for a in args + ('',) if a)

 106 class GalaxyError(AnsibleError):
 107     """ Error for bad Galaxy server responses. """
 108 
 109     def __init__(self, http_error, message):
 110         super(GalaxyError, self).__init__(message)
 111         self.http_code = http_error.code
 112         self.url = http_error.geturl()
 113 
 114         try:
 115             http_msg = to_text(http_error.read())
 116             err_info = json.loads(http_msg)
 117         except (AttributeError, ValueError):
 118             err_info = {}
 119 
 120         url_split = self.url.split('/')
 121         if 'v2' in url_split:
 122             galaxy_msg = err_info.get('message', http_error.reason)
 123             code = err_info.get('code', 'Unknown')
 124             full_error_msg = u"%s (HTTP Code: %d, Message: %s Code: %s)" % (message, self.http_code, galaxy_msg, code)
 125         elif 'v3' in url_split:
 126             errors = err_info.get('errors', [])
 127             if not errors:
 128                 errors = [{}]  # Defaults are set below, we just need to make sure 1 error is present.
 129 
 130             message_lines = []
 131             for error in errors:
 132                 error_msg = error.get('detail') or error.get('title') or http_error.reason
 133                 error_code = error.get('code') or 'Unknown'
 134                 message_line = u"(HTTP Code: %d, Message: %s Code: %s)" % (self.http_code, error_msg, error_code)
 135                 message_lines.append(message_line)
 136 
 137             full_error_msg = "%s %s" % (message, ', '.join(message_lines))
 138         else:
 139             # v1 and unknown API endpoints
 140             galaxy_msg = err_info.get('default', http_error.reason)
 141             full_error_msg = u"%s (HTTP Code: %d, Message: %s)" % (message, self.http_code, galaxy_msg)
 142 
 143         self.message = to_native(full_error_msg)

 190     def _call_galaxy(self, url, args=None, headers=None, method=None, auth_required=False, error_context_msg=None):
 191         headers = headers or {}
 192         self._add_auth_token(headers, url, required=auth_required)
 193 
 194         try:
 195             display.vvvv("Calling Galaxy at %s" % url)
 196             resp = open_url(to_native(url), data=args, validate_certs=self.validate_certs, headers=headers,
 197                             method=method, timeout=20, http_agent=user_agent(), follow_redirects='safe')
 198         except HTTPError as e:
 199             raise GalaxyError(e, error_context_msg)
 200         except Exception as e:
 201             raise AnsibleError("Unknown error when attempting to call Galaxy at '%s': %s" % (url, to_native(e)))
 202 
 203         resp_data = to_text(resp.read(), errors='surrogate_or_strict')
 204         try:
 205             data = json.loads(resp_data)
 206         except ValueError:
 207             raise AnsibleError("Failed to parse Galaxy response from '%s' as JSON:\n%s"
 208                                % (resp.url, to_native(resp_data)))
 209 
 210         return data

 212     def _add_auth_token(self, headers, url, token_type=None, required=False):
 213         # Don't add the auth token if one is already present
 214         if 'Authorization' in headers:
 215             return
 216 
 217         if not self.token and required:
 218             raise AnsibleError("No access token or username set. A token can be set with --api-key, with "
 219                                "'ansible-galaxy login', or set in ansible.cfg.")
 220 
 221         if self.token:
 222             headers.update(self.token.headers())

 224     @g_connect(['v1'])
 225     def authenticate(self, github_token):
 226         """
 227         Retrieve an authentication token
 228         """
 229         url = _urljoin(self.api_server, self.available_api_versions['v1'], "tokens") + '/'
 230         args = urlencode({"github_token": github_token})
 231         resp = open_url(url, data=args, validate_certs=self.validate_certs, method="POST", http_agent=user_agent())
 232         data = json.loads(to_text(resp.read(), errors='surrogate_or_strict'))
 233         return data

 235     @g_connect(['v1'])
 236     def create_import_task(self, github_user, github_repo, reference=None, role_name=None):
 237         """
 238         Post an import request
 239         """
 240         url = _urljoin(self.api_server, self.available_api_versions['v1'], "imports") + '/'
 241         args = {
 242             "github_user": github_user,
 243             "github_repo": github_repo,
 244             "github_reference": reference if reference else ""
 245         }
 246         if role_name:
 247             args['alternate_role_name'] = role_name
 248         elif github_repo.startswith('ansible-role'):
 249             args['alternate_role_name'] = github_repo[len('ansible-role') + 1:]
 250         data = self._call_galaxy(url, args=urlencode(args), method="POST")
 251         if data.get('results', None):
 252             return data['results']
 253         return data

 271     @g_connect(['v1'])
 272     def lookup_role_by_name(self, role_name, notify=True):
 273         """
 274         Find a role by name.
 275         """
 276         role_name = to_text(urlquote(to_bytes(role_name)))
 277 
 278         try:
 279             parts = role_name.split(".")
 280             user_name = ".".join(parts[0:-1])
 281             role_name = parts[-1]
 282             if notify:
 283                 display.display("- downloading role '%s', owned by %s" % (role_name, user_name))
 284         except Exception:
 285             raise AnsibleError("Invalid role name (%s). Specify role as format: username.rolename" % role_name)
 286 
 287         url = _urljoin(self.api_server, self.available_api_versions['v1'], "roles",
 288                        "?owner__username=%s&name=%s" % (user_name, role_name))
 289         data = self._call_galaxy(url)
 290         if len(data["results"]) != 0:
 291             return data["results"][0]
 292         return None

 294     @g_connect(['v1'])
 295     def fetch_role_related(self, related, role_id):
 296         """
 297         Fetch the list of related items for the given role.
 298         The url comes from the 'related' field of the role.
 299         """
 300 
 301         results = []
 302         try:
 303             url = _urljoin(self.api_server, self.available_api_versions['v1'], "roles", role_id, related,
 304                            "?page_size=50")
 305             data = self._call_galaxy(url)
 306             results = data['results']
 307             done = (data.get('next_link', None) is None)
 308 
 309             # https://github.com/ansible/ansible/issues/64355
 310             # api_server contains part of the API path but next_link includes the /api part so strip it out.
 311             url_info = urlparse(self.api_server)
 312             base_url = "%s://%s/" % (url_info.scheme, url_info.netloc)
 313 
 314             while not done:
 315                 url = _urljoin(base_url, data['next_link'])
 316                 data = self._call_galaxy(url)
 317                 results += data['results']
 318                 done = (data.get('next_link', None) is None)
 319         except Exception as e:
 320             display.warning("Unable to retrieve role (id=%s) data (%s), but this is not fatal so we continue: %s"
 321                             % (role_id, related, to_text(e)))
 322         return results

 324     @g_connect(['v1'])
 325     def get_list(self, what):
 326         """
 327         Fetch the list of items specified.
 328         """
 329         try:
 330             url = _urljoin(self.api_server, self.available_api_versions['v1'], what, "?page_size")
 331             data = self._call_galaxy(url)
 332             if "results" in data:
 333                 results = data['results']
 334             else:
 335                 results = data
 336             done = True
 337             if "next" in data:
 338                 done = (data.get('next_link', None) is None)
 339             while not done:
 340                 url = _urljoin(self.api_server, data['next_link'])
 341                 data = self._call_galaxy(url)
 342                 results += data['results']
 343                 done = (data.get('next_link', None) is None)
 344             return results
 345         except Exception as error:
 346             raise AnsibleError("Failed to download the %s list: %s" % (what, to_native(error)))

 350 
 351         search_url = _urljoin(self.api_server, self.available_api_versions['v1'], "search", "roles", "?")

 354             search_url += '&autocomplete=' + to_text(urlquote(to_bytes(search)))

 359         author = kwargs.get('author', None)
 360 
 361         if tags and isinstance(tags, string_types):

 363             search_url += '&tags_autocomplete=' + '+'.join(tags)
 364 
 365         if platforms and isinstance(platforms, string_types):
 366             platforms = platforms.split(',')
 367             search_url += '&platforms_autocomplete=' + '+'.join(platforms)
 368 
 369         if page_size:
 370             search_url += '&page_size=%s' % page_size
 371 
 372         if author:
 373             search_url += '&username_autocomplete=%s' % author
 374 
 375         data = self._call_galaxy(search_url)
 376         return data

 411     @g_connect(['v2', 'v3'])
 412     def publish_collection(self, collection_path):
 413         """
 414         Publishes a collection to a Galaxy server and returns the import task URI.
 415 
 416         :param collection_path: The path to the collection tarball to publish.
 417         :return: The import task URI that contains the import results.
 418         """
 419         display.display("Publishing collection artifact '%s' to %s %s" % (collection_path, self.name, self.api_server))
 420 
 421         b_collection_path = to_bytes(collection_path, errors='surrogate_or_strict')
 422         if not os.path.exists(b_collection_path):
 423             raise AnsibleError("The collection path specified '%s' does not exist." % to_native(collection_path))
 424         elif not tarfile.is_tarfile(b_collection_path):
 425             raise AnsibleError("The collection path specified '%s' is not a tarball, use 'ansible-galaxy collection "
 426                                "build' to create a proper release artifact." % to_native(collection_path))
 427 
 428         with open(b_collection_path, 'rb') as collection_tar:
 429             sha256 = secure_hash_s(collection_tar.read(), hash_func=hashlib.sha256)
 430 
 431         content_type, b_form_data = prepare_multipart(
 432             {
 433                 'sha256': sha256,
 434                 'file': {
 435                     'filename': b_collection_path,
 436                     'mime_type': 'application/octet-stream',
 437                 },
 438             }
 439         )
 440 
 441         headers = {
 442             'Content-type': content_type,
 443             'Content-length': len(b_form_data),
 444         }
 445 
 446         if 'v3' in self.available_api_versions:
 447             n_url = _urljoin(self.api_server, self.available_api_versions['v3'], 'artifacts', 'collections') + '/'
 448         else:
 449             n_url = _urljoin(self.api_server, self.available_api_versions['v2'], 'collections') + '/'
 450 
 451         resp = self._call_galaxy(n_url, args=b_form_data, headers=headers, method='POST', auth_required=True,
 452                                  error_context_msg='Error when publishing collection to %s (%s)'
 453                                                    % (self.name, self.api_server))
 454 
 455         return resp['task']

 457     @g_connect(['v2', 'v3'])
 458     def wait_import_task(self, task_id, timeout=0):
 459         """
 460         Waits until the import process on the Galaxy server has completed or the timeout is reached.
 461 
 462         :param task_id: The id of the import task to wait for. This can be parsed out of the return
 463             value for GalaxyAPI.publish_collection.
 464         :param timeout: The timeout in seconds, 0 is no timeout.
 465         """
 466         state = 'waiting'
 467         data = None
 468 
 469         # Construct the appropriate URL per version
 470         if 'v3' in self.available_api_versions:
 471             full_url = _urljoin(self.api_server, self.available_api_versions['v3'],
 472                                 'imports/collections', task_id, '/')
 473         else:
 474             full_url = _urljoin(self.api_server, self.available_api_versions['v2'],
 475                                 'collection-imports', task_id, '/')
 476 
 477         display.display("Waiting until Galaxy import task %s has completed" % full_url)
 478         start = time.time()
 479         wait = 2
 480 
 481         while timeout == 0 or (time.time() - start) < timeout:
 482             try:
 483                 data = self._call_galaxy(full_url, method='GET', auth_required=True,
 484                                          error_context_msg='Error when getting import task results at %s' % full_url)
 485             except GalaxyError as e:
 486                 if e.http_code != 404:
 487                     raise
 488                 # The import job may not have started, and as such, the task url may not yet exist
 489                 display.vvv('Galaxy import process has not started, wait %s seconds before trying again' % wait)
 490                 time.sleep(wait)
 491                 continue
 492 
 493             state = data.get('state', 'waiting')
 494 
 495             if data.get('finished_at', None):
 496                 break
 497 
 498             display.vvv('Galaxy import process has a status of %s, wait %d seconds before trying again'
 499                         % (state, wait))
 500             time.sleep(wait)
 501 
 502             # poor man's exponential backoff algo so we don't flood the Galaxy API, cap at 30 seconds.
 503             wait = min(30, wait * 1.5)
 504         if state == 'waiting':
 505             raise AnsibleError("Timeout while waiting for the Galaxy import process to finish, check progress at '%s'"
 506                                % to_native(full_url))
 507 
 508         for message in data.get('messages', []):
 509             level = message['level']
 510             if level == 'error':
 511                 display.error("Galaxy import error message: %s" % message['message'])
 512             elif level == 'warning':
 513                 display.warning("Galaxy import warning message: %s" % message['message'])
 514             else:
 515                 display.vvv("Galaxy import message: %s - %s" % (level, message['message']))
 516 
 517         if state == 'failed':
 518             code = to_native(data['error'].get('code', 'UNKNOWN'))
 519             description = to_native(
 520                 data['error'].get('description', "Unknown error, see %s for more details" % full_url))
 521             raise AnsibleError("Galaxy import process failed: %s (Code: %s)" % (description, code))

 523     @g_connect(['v2', 'v3'])
 524     def get_collection_version_metadata(self, namespace, name, version):
 525         """
 526         Gets the collection information from the Galaxy server about a specific Collection version.
 527 
 528         :param namespace: The collection namespace.
 529         :param name: The collection name.
 530         :param version: Version of the collection to get the information for.
 531         :return: CollectionVersionMetadata about the collection at the version requested.
 532         """
 533         api_path = self.available_api_versions.get('v3', self.available_api_versions.get('v2'))
 534         url_paths = [self.api_server, api_path, 'collections', namespace, name, 'versions', version, '/']
 535 
 536         n_collection_url = _urljoin(*url_paths)
 537         error_context_msg = 'Error when getting collection version metadata for %s.%s:%s from %s (%s)' \
 538                             % (namespace, name, version, self.name, self.api_server)
 539         data = self._call_galaxy(n_collection_url, error_context_msg=error_context_msg)
 540 
 541         return CollectionVersionMetadata(data['namespace']['name'], data['collection']['name'], data['version'],
 542                                          data['download_url'], data['artifact']['sha256'],
 543                                          data['metadata']['dependencies'])

 545     @g_connect(['v2', 'v3'])
 546     def get_collection_versions(self, namespace, name):
 547         """
 548         Gets a list of available versions for a collection on a Galaxy server.
 549 
 550         :param namespace: The collection namespace.
 551         :param name: The collection name.
 552         :return: A list of versions that are available.
 553         """
 554         relative_link = False
 555         if 'v3' in self.available_api_versions:
 556             api_path = self.available_api_versions['v3']
 557             pagination_path = ['links', 'next']
 558             relative_link = True  # AH pagination results are relative an not an absolute URI.
 559         else:
 560             api_path = self.available_api_versions['v2']
 561             pagination_path = ['next']
 562 
 563         n_url = _urljoin(self.api_server, api_path, 'collections', namespace, name, 'versions', '/')
 564 
 565         error_context_msg = 'Error when getting available collection versions for %s.%s from %s (%s)' \
 566                             % (namespace, name, self.name, self.api_server)
 567         data = self._call_galaxy(n_url, error_context_msg=error_context_msg)
 568 
 569         if 'data' in data:
 570             # v3 automation-hub is the only known API that uses `data`
 571             # since v3 pulp_ansible does not, we cannot rely on version
 572             # to indicate which key to use
 573             results_key = 'data'
 574         else:
 575             results_key = 'results'
 576 
 577         versions = []
 578         while True:
 579             versions += [v['version'] for v in data[results_key]]
 580 
 581             next_link = data
 582             for path in pagination_path:
 583                 next_link = next_link.get(path, {})
 584 
 585             if not next_link:
 586                 break
 587             elif relative_link:
 588                 # TODO: This assumes the pagination result is relative to the root server. Will need to be verified
 589                 # with someone who knows the AH API.
 590                 next_link = n_url.replace(urlparse(n_url).path, next_link)
 591 
 592             data = self._call_galaxy(to_native(next_link, errors='surrogate_or_strict'),
 593                                      error_context_msg=error_context_msg)
 594 
 595         return versions
