  63     def __init__(self, namespace, name, b_path, api, versions, requirement, force, parent=None, metadata=None,
  64                  files=None, skip=False, allow_pre_releases=False):
  65         """Represents a collection requirement, the versions that are available to be installed as well as any
  66         dependencies the collection has.
  67 
  68         :param namespace: The collection namespace.
  69         :param name: The collection name.
  70         :param b_path: Byte str of the path to the collection tarball if it has already been downloaded.
  71         :param api: The GalaxyAPI to use if the collection is from Galaxy.
  72         :param versions: A list of versions of the collection that are available.
  73         :param requirement: The version requirement string used to verify the list of versions fit the requirements.
  74         :param force: Whether the force flag applied to the collection.
  75         :param parent: The name of the parent the collection is a dependency of.
  76         :param metadata: The galaxy.api.CollectionVersionMetadata that has already been retrieved from the Galaxy
  77             server.
  78         :param files: The files that exist inside the collection. This is based on the FILES.json file inside the
  79             collection artifact.

  81             and force is not set.
  82         :param allow_pre_releases: Whether to skip pre-release versions of collections.

  85         self.name = name

  90         self.skip = skip

  92         self.allow_pre_releases = allow_pre_releases

  95         self._files = files

 184     def add_requirement(self, parent, requirement):
 185         self.required_by.append((parent, requirement))
 186         new_versions = set(v for v in self.versions if self._meets_requirements(v, requirement, parent))
 187         if len(new_versions) == 0:
 188             if self.skip:
 189                 force_flag = '--force-with-deps' if parent else '--force'
 190                 version = self.latest_version if self.latest_version != '*' else 'unknown'
 191                 msg = "Cannot meet requirement %s:%s as it is already installed at version '%s'. Use %s to overwrite" \
 192                       % (to_text(self), requirement, version, force_flag)
 193                 raise AnsibleError(msg)
 194             elif parent is None:
 195                 msg = "Cannot meet requirement %s for dependency %s" % (requirement, to_text(self))
 196             else:
 197                 msg = "Cannot meet dependency requirement '%s:%s' for collection %s" \
 198                       % (to_text(self), requirement, parent)
 199 
 200             collection_source = to_text(self.b_path, nonstring='passthru') or self.api.api_server
 201             req_by = "\n".join(
 202                 "\t%s - '%s:%s'" % (to_text(p) if p else 'base', to_text(self), r)
 203                 for p, r in self.required_by
 204             )
 205 
 206             versions = ", ".join(sorted(self.versions, key=SemanticVersion))
 207             if not self.versions and self.pre_releases:
 208                 pre_release_msg = (
 209                     '\nThis collection only contains pre-releases. Utilize `--pre` to install pre-releases, or '
 210                     'explicitly provide the pre-release version.'
 211                 )
 212             else:
 213                 pre_release_msg = ''
 214 
 215             raise AnsibleError(
 216                 "%s from source '%s'. Available versions before last requirement added: %s\nRequirements from:\n%s%s"
 217                 % (msg, collection_source, versions, req_by, pre_release_msg)
 218             )
 219 
 220         self.versions = new_versions

 233     def install(self, path, b_temp_path):
 234         if self.skip:
 235             display.display("Skipping '%s' as it is already installed" % to_text(self))
 236             return
 237 
 238         # Install if it is not
 239         collection_path = os.path.join(path, self.namespace, self.name)
 240         b_collection_path = to_bytes(collection_path, errors='surrogate_or_strict')
 241         display.display("Installing '%s:%s' to '%s'" % (to_text(self), self.latest_version, collection_path))
 242 
 243         if self.b_path is None:
 244             self.b_path = self.download(b_temp_path)
 245 
 246         if os.path.exists(b_collection_path):
 247             shutil.rmtree(b_collection_path)
 248 
 249         if os.path.isfile(self.b_path):
 250             self.install_artifact(b_collection_path, b_temp_path)
 251         else:
 252             self.install_scm(b_collection_path)
 253 
 254         display.display("%s (%s) was installed successfully" % (to_text(self), self.latest_version))

 256     def install_artifact(self, b_collection_path, b_temp_path):
 257 
 258         try:
 259             with tarfile.open(self.b_path, mode='r') as collection_tar:
 260                 files_member_obj = collection_tar.getmember('FILES.json')
 261                 with _tarfile_extract(collection_tar, files_member_obj) as (dummy, files_obj):
 262                     files = json.loads(to_text(files_obj.read(), errors='surrogate_or_strict'))
 263 
 264                 _extract_tar_file(collection_tar, 'MANIFEST.json', b_collection_path, b_temp_path)
 265                 _extract_tar_file(collection_tar, 'FILES.json', b_collection_path, b_temp_path)
 266 
 267                 for file_info in files['files']:
 268                     file_name = file_info['name']
 269                     if file_name == '.':
 270                         continue
 271 
 272                     if file_info['ftype'] == 'file':
 273                         _extract_tar_file(collection_tar, file_name, b_collection_path, b_temp_path,
 274                                           expected_hash=file_info['chksum_sha256'])
 275 
 276                     else:
 277                         _extract_tar_dir(collection_tar, file_name, b_collection_path)
 278 
 279         except Exception:
 280             # Ensure we don't leave the dir behind in case of a failure.
 281             shutil.rmtree(b_collection_path)
 282 
 283             b_namespace_path = os.path.dirname(b_collection_path)
 284             if not os.listdir(b_namespace_path):
 285                 os.rmdir(b_namespace_path)
 286 
 287             raise

 319     def verify(self, remote_collection, path, b_temp_tar_path):
 320         if not self.skip:
 321             display.display("'%s' has not been installed, nothing to verify" % (to_text(self)))
 322             return
 323 
 324         collection_path = os.path.join(path, self.namespace, self.name)
 325         b_collection_path = to_bytes(collection_path, errors='surrogate_or_strict')
 326 
 327         display.vvv("Verifying '%s:%s'." % (to_text(self), self.latest_version))
 328         display.vvv("Installed collection found at '%s'" % collection_path)
 329         display.vvv("Remote collection found at '%s'" % remote_collection.metadata.download_url)
 330 
 331         # Compare installed version versus requirement version
 332         if self.latest_version != remote_collection.latest_version:
 333             err = "%s has the version '%s' but is being compared to '%s'" % (to_text(self), self.latest_version, remote_collection.latest_version)
 334             display.display(err)
 335             return
 336 
 337         modified_content = []
 338 
 339         # Verify the manifest hash matches before verifying the file manifest
 340         expected_hash = _get_tar_file_hash(b_temp_tar_path, 'MANIFEST.json')
 341         self._verify_file_hash(b_collection_path, 'MANIFEST.json', expected_hash, modified_content)
 342         manifest = _get_json_from_tar_file(b_temp_tar_path, 'MANIFEST.json')
 343 
 344         # Use the manifest to verify the file manifest checksum
 345         file_manifest_data = manifest['file_manifest_file']
 346         file_manifest_filename = file_manifest_data['name']
 347         expected_hash = file_manifest_data['chksum_%s' % file_manifest_data['chksum_type']]
 348 
 349         # Verify the file manifest before using it to verify individual files
 350         self._verify_file_hash(b_collection_path, file_manifest_filename, expected_hash, modified_content)
 351         file_manifest = _get_json_from_tar_file(b_temp_tar_path, file_manifest_filename)
 352 
 353         # Use the file manifest to verify individual file checksums
 354         for manifest_data in file_manifest['files']:
 355             if manifest_data['ftype'] == 'file':
 356                 expected_hash = manifest_data['chksum_%s' % manifest_data['chksum_type']]
 357                 self._verify_file_hash(b_collection_path, manifest_data['name'], expected_hash, modified_content)
 358 
 359         if modified_content:
 360             display.display("Collection %s contains modified content in the following files:" % to_text(self))
 361             display.display(to_text(self))
 362             display.vvv(to_text(self.b_path))
 363             for content_change in modified_content:
 364                 display.display('    %s' % content_change.filename)
 365                 display.vvv("    Expected: %s\n    Found: %s" % (content_change.expected, content_change.installed))
 366         else:
 367             display.vvv("Successfully verified that checksums for '%s:%s' match the remote collection" % (to_text(self), self.latest_version))

 370         b_file_path = to_bytes(os.path.join(to_text(b_path), filename), errors='surrogate_or_strict')

 386     def _meets_requirements(self, version, requirements, parent):
 387         """
 388         Supports version identifiers can be '==', '!=', '>', '>=', '<', '<=', '*'. Each requirement is delimited by ','
 389         """
 390         op_map = {
 391             '!=': operator.ne,
 392             '==': operator.eq,
 393             '=': operator.eq,
 394             '>=': operator.ge,
 395             '>': operator.gt,
 396             '<=': operator.le,
 397             '<': operator.lt,
 398         }
 399 
 400         for req in list(requirements.split(',')):
 401             op_pos = 2 if len(req) > 1 and req[1] == '=' else 1
 402             op = op_map.get(req[:op_pos])
 403 
 404             requirement = req[op_pos:]
 405             if not op:
 406                 requirement = req
 407                 op = operator.eq
 408 
 409             # In the case we are checking a new requirement on a base requirement (parent != None) we can't accept
 410             # version as '*' (unknown version) unless the requirement is also '*'.
 411             if parent and version == '*' and requirement != '*':
 412                 display.warning("Failed to validate the collection requirement '%s:%s' for %s when the existing "
 413                                 "install does not have a version set, the collection may not work."
 414                                 % (to_text(self), req, parent))
 415                 continue
 416             elif requirement == '*' or version == '*':
 417                 continue
 418 
 419             if not op(SemanticVersion(version), SemanticVersion.from_loose_version(LooseVersion(requirement))):
 420                 break
 421         else:
 422             return True
 423 
 424         # The loop was broken early, it does not meet all the requirements
 425         return False

 427     @staticmethod
 428     def from_tar(b_path, force, parent=None):
 429         if not tarfile.is_tarfile(b_path):
 430             raise AnsibleError("Collection artifact at '%s' is not a valid tar file." % to_native(b_path))
 431 
 432         info = {}
 433         with tarfile.open(b_path, mode='r') as collection_tar:
 434             for b_member_name, property_name in CollectionRequirement._FILE_MAPPING:
 435                 n_member_name = to_native(b_member_name)
 436                 try:
 437                     member = collection_tar.getmember(n_member_name)
 438                 except KeyError:
 439                     raise AnsibleError("Collection at '%s' does not contain the required file %s."
 440                                        % (to_native(b_path), n_member_name))
 441 
 442                 with _tarfile_extract(collection_tar, member) as (dummy, member_obj):
 443                     try:
 444                         info[property_name] = json.loads(to_text(member_obj.read(), errors='surrogate_or_strict'))
 445                     except ValueError:
 446                         raise AnsibleError("Collection tar file member %s does not contain a valid json string."
 447                                            % n_member_name)
 448 
 449         meta = info['manifest_file']['collection_info']
 450         files = info['files_file']['files']
 451 
 452         namespace = meta['namespace']
 453         name = meta['name']
 454         version = meta['version']
 455         meta = CollectionVersionMetadata(namespace, name, version, None, None, meta['dependencies'])
 456 
 457         if SemanticVersion(version).is_prerelease:
 458             allow_pre_release = True
 459         else:
 460             allow_pre_release = False
 461 
 462         return CollectionRequirement(namespace, name, b_path, None, [version], version, force, parent=parent,
 463                                      metadata=meta, files=files, allow_pre_releases=allow_pre_release)

 465     @staticmethod
 466     def from_path(b_path, force, parent=None, fallback_metadata=False, skip=True):
 467         info = CollectionRequirement.collection_info(b_path, fallback_metadata)
 468 
 469         allow_pre_release = False
 470         if 'manifest_file' in info:
 471             manifest = info['manifest_file']['collection_info']
 472             namespace = manifest['namespace']
 473             name = manifest['name']
 474             version = to_text(manifest['version'], errors='surrogate_or_strict')
 475 
 476             try:
 477                 _v = SemanticVersion()
 478                 _v.parse(version)
 479                 if _v.is_prerelease:
 480                     allow_pre_release = True
 481             except ValueError:
 482                 display.warning("Collection at '%s' does not have a valid version set, falling back to '*'. Found "
 483                                 "version: '%s'" % (to_text(b_path), version))
 484                 version = '*'
 485 
 486             dependencies = manifest['dependencies']
 487         else:
 488             if fallback_metadata:
 489                 warning = "Collection at '%s' does not have a galaxy.yml or a MANIFEST.json file, cannot detect version."
 490             else:
 491                 warning = "Collection at '%s' does not have a MANIFEST.json file, cannot detect version."
 492             display.warning(warning % to_text(b_path))
 493             parent_dir, name = os.path.split(to_text(b_path, errors='surrogate_or_strict'))
 494             namespace = os.path.split(parent_dir)[1]
 495 
 496             version = '*'
 497             dependencies = {}
 498 
 499         meta = CollectionVersionMetadata(namespace, name, version, None, None, dependencies)
 500 
 501         files = info.get('files_file', {}).get('files', {})
 502 
 503         return CollectionRequirement(namespace, name, b_path, None, [version], version, force, parent=parent,
 504                                      metadata=meta, files=files, skip=skip, allow_pre_releases=allow_pre_release)

 506     @staticmethod
 507     def from_name(collection, apis, requirement, force, parent=None, allow_pre_release=False):
 508         namespace, name = collection.split('.', 1)
 509         galaxy_meta = None
 510 
 511         for api in apis:
 512             try:
 513                 if not (requirement == '*' or requirement.startswith('<') or requirement.startswith('>') or
 514                         requirement.startswith('!=')):
 515                     # Exact requirement
 516                     allow_pre_release = True
 517 
 518                     if requirement.startswith('='):
 519                         requirement = requirement.lstrip('=')
 520 
 521                     resp = api.get_collection_version_metadata(namespace, name, requirement)
 522 
 523                     galaxy_meta = resp
 524                     versions = [resp.version]
 525                 else:
 526                     versions = api.get_collection_versions(namespace, name)
 527             except GalaxyError as err:
 528                 if err.http_code == 404:
 529                     display.vvv("Collection '%s' is not available from server %s %s"
 530                                 % (collection, api.name, api.api_server))
 531                     continue
 532                 raise
 533 
 534             display.vvv("Collection '%s' obtained from server %s %s" % (collection, api.name, api.api_server))
 535             break
 536         else:
 537             raise AnsibleError("Failed to find collection %s:%s" % (collection, requirement))
 538 
 539         req = CollectionRequirement(namespace, name, None, api, versions, requirement, force, parent=parent,
 540                                     metadata=galaxy_meta, allow_pre_releases=allow_pre_release)
 541         return req

 544 def build_collection(collection_path, output_path, force):
 545     """Creates the Ansible collection artifact in a .tar.gz file.
 546 
 547     :param collection_path: The path to the collection to build. This should be the directory that contains the
 548         galaxy.yml file.
 549     :param output_path: The path to create the collection build artifact. This should be a directory.
 550     :param force: Whether to overwrite an existing collection build artifact or fail.

 617 def publish_collection(collection_path, api, wait, timeout):
 618     """Publish an Ansible collection tarball into an Ansible Galaxy server.
 619 
 620     :param collection_path: The path to the collection tarball to publish.
 621     :param api: A GalaxyAPI to publish the collection to.
 622     :param wait: Whether to wait until the import process is complete.
 623     :param timeout: The time in seconds to wait for the import process to finish, 0 is indefinite.
 624     """
 625     import_uri = api.publish_collection(collection_path)
 626 
 627     if wait:
 628         # Galaxy returns a url fragment which differs between v2 and v3.  The second to last entry is
 629         # always the task_id, though.
 630         # v2: {"task": "https://galaxy-dev.ansible.com/api/v2/collection-imports/35573/"}
 631         # v3: {"task": "/api/automation-hub/v3/imports/collections/838d1308-a8f4-402c-95cb-7823f3806cd8/"}
 632         task_id = None
 633         for path_segment in reversed(import_uri.split('/')):
 634             if path_segment:
 635                 task_id = path_segment
 636                 break
 637 
 638         if not task_id:
 639             raise AnsibleError("Publishing the collection did not return valid task info. Cannot wait for task status. Returned task info: '%s'" % import_uri)
 640 
 641         display.display("Collection has been published to the Galaxy server %s %s" % (api.name, api.api_server))
 642         with _display_progress():
 643             api.wait_import_task(task_id, timeout)
 644         display.display("Collection has been successfully published and imported to the Galaxy server %s %s"
 645                         % (api.name, api.api_server))
 646     else:
 647         display.display("Collection has been pushed to the Galaxy server %s %s, not waiting until import has "
 648                         "completed due to --no-wait being set. Import task results can be found at %s"
 649                         % (api.name, api.api_server, import_uri))

 786 def _display_progress():
 787     config_display = C.GALAXY_DISPLAY_PROGRESS
 788     display_wheel = sys.stdout.isatty() if config_display is None else config_display

 794     def progress(display_queue, actual_display):
 795         actual_display.debug("Starting display_progress display thread")
 796         t = threading.current_thread()
 797 
 798         while True:
 799             for c in "|/-\\":
 800                 actual_display.display(c + "\b", newline=False)
 801                 time.sleep(0.1)
 802 
 803                 # Display a message from the main thread
 804                 while True:
 805                     try:
 806                         method, args, kwargs = display_queue.get(block=False, timeout=0.1)
 807                     except queue.Empty:
 808                         break
 809                     else:
 810                         func = getattr(actual_display, method)
 811                         func(*args, **kwargs)
 812 
 813                 if getattr(t, "finish", False):
 814                     actual_display.debug("Received end signal for display_progress display thread")
 815                     return
 816 
 817     class DisplayThread(object):
 818 
 819         def __init__(self, display_queue):
 820             self.display_queue = display_queue
 821 
 822         def __getattr__(self, attr):
 823             def call_display(*args, **kwargs):
 824                 self.display_queue.put((attr, args, kwargs))
 825 
 826             return call_display
 827 
 828     # Temporary override the global display class with our own which add the calls to a queue for the thread to call.
 829     global display
 830     old_display = display
 831     try:
 832         display_queue = queue.Queue()
 833         display = DisplayThread(display_queue)
 834         t = threading.Thread(target=progress, args=(display_queue, old_display))
 835         t.daemon = True
 836         t.start()
 837 
 838         try:
 839             yield
 840         finally:
 841             t.finish = True
 842             t.join()
 843     except Exception:
 844         # The exception is re-raised so we can sure the thread is finished and not using the display anymore
 845         raise
 846     finally:
 847         display = old_display

 850 def _get_galaxy_yml(b_galaxy_yml_path):
 851     meta_info = get_collections_galaxy_meta_info()
 852 
 853     mandatory_keys = set()
 854     string_keys = set()
 855     list_keys = set()
 856     dict_keys = set()
 857 
 858     for info in meta_info:
 859         if info.get('required', False):
 860             mandatory_keys.add(info['key'])
 861 
 862         key_list_type = {
 863             'str': string_keys,
 864             'list': list_keys,
 865             'dict': dict_keys,
 866         }[info.get('type', 'str')]
 867         key_list_type.add(info['key'])
 868 
 869     all_keys = frozenset(list(mandatory_keys) + list(string_keys) + list(list_keys) + list(dict_keys))
 870 
 871     try:
 872         with open(b_galaxy_yml_path, 'rb') as g_yaml:
 873             galaxy_yml = yaml.safe_load(g_yaml)
 874     except YAMLError as err:
 875         raise AnsibleError("Failed to parse the galaxy.yml at '%s' with the following error:\n%s"
 876                            % (to_native(b_galaxy_yml_path), to_native(err)))
 877 
 878     set_keys = set(galaxy_yml.keys())
 879     missing_keys = mandatory_keys.difference(set_keys)
 880     if missing_keys:
 881         raise AnsibleError("The collection galaxy.yml at '%s' is missing the following mandatory keys: %s"
 882                            % (to_native(b_galaxy_yml_path), ", ".join(sorted(missing_keys))))
 883 
 884     extra_keys = set_keys.difference(all_keys)
 885     if len(extra_keys) > 0:
 886         display.warning("Found unknown keys in collection galaxy.yml at '%s': %s"
 887                         % (to_text(b_galaxy_yml_path), ", ".join(extra_keys)))
 888 
 889     # Add the defaults if they have not been set
 890     for optional_string in string_keys:
 891         if optional_string not in galaxy_yml:
 892             galaxy_yml[optional_string] = None
 893 
 894     for optional_list in list_keys:
 895         list_val = galaxy_yml.get(optional_list, None)
 896 
 897         if list_val is None:
 898             galaxy_yml[optional_list] = []
 899         elif not isinstance(list_val, list):
 900             galaxy_yml[optional_list] = [list_val]
 901 
 902     for optional_dict in dict_keys:
 903         if optional_dict not in galaxy_yml:
 904             galaxy_yml[optional_dict] = {}
 905 
 906     # license is a builtin var in Python, to avoid confusion we just rename it to license_ids
 907     galaxy_yml['license_ids'] = galaxy_yml['license']
 908     del galaxy_yml['license']
 909 
 910     return galaxy_yml

 913 def _build_files_manifest(b_collection_path, namespace, name, ignore_patterns):
 914     # We always ignore .pyc and .retry files as well as some well known version control directories. The ignore
 915     # patterns can be extended by the build_ignore key in galaxy.yml
 916     b_ignore_patterns = [
 917         b'galaxy.yml',
 918         b'galaxy.yaml',
 919         b'.git',
 920         b'*.pyc',
 921         b'*.retry',
 922         b'tests/output',  # Ignore ansible-test result output directory.
 923         to_bytes('{0}-{1}-*.tar.gz'.format(namespace, name)),  # Ignores previously built artifacts in the root dir.
 924     ]
 925     b_ignore_patterns += [to_bytes(p) for p in ignore_patterns]
 926     b_ignore_dirs = frozenset([b'CVS', b'.bzr', b'.hg', b'.git', b'.svn', b'__pycache__', b'.tox'])
 927 
 928     entry_template = {
 929         'name': None,
 930         'ftype': None,
 931         'chksum_type': None,
 932         'chksum_sha256': None,
 933         'format': MANIFEST_FORMAT
 934     }
 935     manifest = {
 936         'files': [
 937             {
 938                 'name': '.',
 939                 'ftype': 'dir',
 940                 'chksum_type': None,
 941                 'chksum_sha256': None,
 942                 'format': MANIFEST_FORMAT,
 943             },
 944         ],
 945         'format': MANIFEST_FORMAT,
 946     }
 947 
 948     def _walk(b_path, b_top_level_dir):
 949         for b_item in os.listdir(b_path):
 950             b_abs_path = os.path.join(b_path, b_item)
 951             b_rel_base_dir = b'' if b_path == b_top_level_dir else b_path[len(b_top_level_dir) + 1:]
 952             b_rel_path = os.path.join(b_rel_base_dir, b_item)
 953             rel_path = to_text(b_rel_path, errors='surrogate_or_strict')
 954 
 955             if os.path.isdir(b_abs_path):
 956                 if any(b_item == b_path for b_path in b_ignore_dirs) or \
 957                         any(fnmatch.fnmatch(b_rel_path, b_pattern) for b_pattern in b_ignore_patterns):
 958                     display.vvv("Skipping '%s' for collection build" % to_text(b_abs_path))
 959                     continue
 960 
 961                 if os.path.islink(b_abs_path):
 962                     b_link_target = os.path.realpath(b_abs_path)
 963 
 964                     if not _is_child_path(b_link_target, b_top_level_dir):
 965                         display.warning("Skipping '%s' as it is a symbolic link to a directory outside the collection"
 966                                         % to_text(b_abs_path))
 967                         continue
 968 
 969                 manifest_entry = entry_template.copy()
 970                 manifest_entry['name'] = rel_path
 971                 manifest_entry['ftype'] = 'dir'
 972 
 973                 manifest['files'].append(manifest_entry)
 974 
 975                 if not os.path.islink(b_abs_path):
 976                     _walk(b_abs_path, b_top_level_dir)
 977             else:
 978                 if any(fnmatch.fnmatch(b_rel_path, b_pattern) for b_pattern in b_ignore_patterns):
 979                     display.vvv("Skipping '%s' for collection build" % to_text(b_abs_path))
 980                     continue
 981 
 982                 # Handling of file symlinks occur in _build_collection_tar, the manifest for a symlink is the same for
 983                 # a normal file.
 984                 manifest_entry = entry_template.copy()
 985                 manifest_entry['name'] = rel_path
 986                 manifest_entry['ftype'] = 'file'
 987                 manifest_entry['chksum_type'] = 'sha256'
 988                 manifest_entry['chksum_sha256'] = secure_hash(b_abs_path, hash_func=sha256)
 989 
 990                 manifest['files'].append(manifest_entry)
 991 
 992     _walk(b_collection_path, b_collection_path)
 993 
 994     return manifest

1030 def _build_collection_tar(b_collection_path, b_tar_path, collection_manifest, file_manifest):
1031     """Build a tar.gz collection artifact from the manifest data."""
1032     files_manifest_json = to_bytes(json.dumps(file_manifest, indent=True), errors='surrogate_or_strict')
1033     collection_manifest['file_manifest_file']['chksum_sha256'] = secure_hash_s(files_manifest_json, hash_func=sha256)
1034     collection_manifest_json = to_bytes(json.dumps(collection_manifest, indent=True), errors='surrogate_or_strict')
1035 
1036     with _tempdir() as b_temp_path:
1037         b_tar_filepath = os.path.join(b_temp_path, os.path.basename(b_tar_path))
1038 
1039         with tarfile.open(b_tar_filepath, mode='w:gz') as tar_file:
1040             # Add the MANIFEST.json and FILES.json file to the archive
1041             for name, b in [('MANIFEST.json', collection_manifest_json), ('FILES.json', files_manifest_json)]:
1042                 b_io = BytesIO(b)
1043                 tar_info = tarfile.TarInfo(name)
1044                 tar_info.size = len(b)
1045                 tar_info.mtime = time.time()
1046                 tar_info.mode = 0o0644
1047                 tar_file.addfile(tarinfo=tar_info, fileobj=b_io)
1048 
1049             for file_info in file_manifest['files']:
1050                 if file_info['name'] == '.':
1051                     continue
1052 
1053                 # arcname expects a native string, cannot be bytes
1054                 filename = to_native(file_info['name'], errors='surrogate_or_strict')
1055                 b_src_path = os.path.join(b_collection_path, to_bytes(filename, errors='surrogate_or_strict'))
1056 
1057                 def reset_stat(tarinfo):
1058                     if tarinfo.type != tarfile.SYMTYPE:
1059                         existing_is_exec = tarinfo.mode & stat.S_IXUSR
1060                         tarinfo.mode = 0o0755 if existing_is_exec or tarinfo.isdir() else 0o0644
1061                     tarinfo.uid = tarinfo.gid = 0
1062                     tarinfo.uname = tarinfo.gname = ''
1063 
1064                     return tarinfo
1065 
1066                 if os.path.islink(b_src_path):
1067                     b_link_target = os.path.realpath(b_src_path)
1068                     if _is_child_path(b_link_target, b_collection_path):
1069                         b_rel_path = os.path.relpath(b_link_target, start=os.path.dirname(b_src_path))
1070 
1071                         tar_info = tarfile.TarInfo(filename)
1072                         tar_info.type = tarfile.SYMTYPE
1073                         tar_info.linkname = to_native(b_rel_path, errors='surrogate_or_strict')
1074                         tar_info = reset_stat(tar_info)
1075                         tar_file.addfile(tarinfo=tar_info)
1076 
1077                         continue
1078 
1079                 # Dealing with a normal file, just add it by name.
1080                 tar_file.add(os.path.realpath(b_src_path), arcname=filename, recursive=False, filter=reset_stat)
1081

1151 def _build_dependency_map(collections, existing_collections, b_temp_path, apis, validate_certs, force, force_deps,
1152                           no_deps, allow_pre_release=False):
1153     dependency_map = {}
1154 
1155     # First build the dependency map on the actual requirements
1156     for name, version, source, req_type in collections:
1157         _get_collection_info(dependency_map, existing_collections, name, version, source, b_temp_path, apis,
1158                              validate_certs, (force or force_deps), allow_pre_release=allow_pre_release, req_type=req_type)
1159 
1160     checked_parents = set([to_text(c) for c in dependency_map.values() if c.skip])
1161     while len(dependency_map) != len(checked_parents):
1162         while not no_deps:  # Only parse dependencies if no_deps was not set
1163             parents_to_check = set(dependency_map.keys()).difference(checked_parents)
1164 
1165             deps_exhausted = True
1166             for parent in parents_to_check:
1167                 parent_info = dependency_map[parent]
1168 
1169                 if parent_info.dependencies:
1170                     deps_exhausted = False
1171                     for dep_name, dep_requirement in parent_info.dependencies.items():
1172                         _get_collection_info(dependency_map, existing_collections, dep_name, dep_requirement,
1173                                              parent_info.api, b_temp_path, apis, validate_certs, force_deps,
1174                                              parent=parent, allow_pre_release=allow_pre_release)
1175 
1176                     checked_parents.add(parent)
1177 
1178             # No extra dependencies were resolved, exit loop
1179             if deps_exhausted:
1180                 break
1181 
1182         # Now we have resolved the deps to our best extent, now select the latest version for collections with
1183         # multiple versions found and go from there
1184         deps_not_checked = set(dependency_map.keys()).difference(checked_parents)
1185         for collection in deps_not_checked:
1186             dependency_map[collection].set_latest_version()
1187             if no_deps or len(dependency_map[collection].dependencies) == 0:
1188                 checked_parents.add(collection)
1189 
1190     return dependency_map

1193 def _collections_from_scm(collection, requirement, b_temp_path, force, parent=None):
1194     """Returns a list of collections found in the repo. If there is a galaxy.yml in the collection then just return
1195     the specific collection. Otherwise, check each top-level directory for a galaxy.yml.
1196 
1197     :param collection: URI to a git repo
1198     :param requirement: The version of the artifact
1199     :param b_temp_path: The temporary path to the archive of a collection
1200     :param force: Whether to overwrite an existing collection or fail
1201     :param parent: The name of the parent collection
1202     :raises AnsibleError: if nothing found
1203     :return: List of CollectionRequirement objects
1204     :rtype: list
1205     """
1206 
1207     reqs = []
1208     name, version, path, fragment = parse_scm(collection, requirement)
1209     b_repo_root = to_bytes(name, errors='surrogate_or_strict')
1210 
1211     b_collection_path = os.path.join(b_temp_path, b_repo_root)
1212     if fragment:
1213         b_fragment = to_bytes(fragment, errors='surrogate_or_strict')
1214         b_collection_path = os.path.join(b_collection_path, b_fragment)
1215 
1216     b_galaxy_path = get_galaxy_metadata_path(b_collection_path)
1217 
1218     err = ("%s appears to be an SCM collection source, but the required galaxy.yml was not found. "
1219            "Append #path/to/collection/ to your URI (before the comma separated version, if one is specified) "
1220            "to point to a directory containing the galaxy.yml or directories of collections" % collection)
1221 
1222     display.vvvvv("Considering %s as a possible path to a collection's galaxy.yml" % b_galaxy_path)
1223     if os.path.exists(b_galaxy_path):
1224         return [CollectionRequirement.from_path(b_collection_path, force, parent, fallback_metadata=True, skip=False)]
1225 
1226     if not os.path.isdir(b_collection_path) or not os.listdir(b_collection_path):
1227         raise AnsibleError(err)
1228 
1229     for b_possible_collection in os.listdir(b_collection_path):
1230         b_collection = os.path.join(b_collection_path, b_possible_collection)
1231         if not os.path.isdir(b_collection):
1232             continue
1233         b_galaxy = get_galaxy_metadata_path(b_collection)
1234         display.vvvvv("Considering %s as a possible path to a collection's galaxy.yml" % b_galaxy)
1235         if os.path.exists(b_galaxy):
1236             reqs.append(CollectionRequirement.from_path(b_collection, force, parent, fallback_metadata=True, skip=False))
1237     if not reqs:
1238         raise AnsibleError(err)
1239 
1240     return reqs

1243 def _get_collection_info(dep_map, existing_collections, collection, requirement, source, b_temp_path, apis,
1244                          validate_certs, force, parent=None, allow_pre_release=False, req_type=None):
1245     dep_msg = ""
1246     if parent:
1247         dep_msg = " - as dependency of %s" % parent
1248     display.vvv("Processing requirement collection '%s'%s" % (to_text(collection), dep_msg))
1249 
1250     b_tar_path = None
1251 
1252     is_file = (
1253         req_type == 'file' or
1254         (not req_type and os.path.isfile(to_bytes(collection, errors='surrogate_or_strict')))
1255     )
1256 
1257     is_url = (
1258         req_type == 'url' or
1259         (not req_type and urlparse(collection).scheme.lower() in ['http', 'https'])
1260     )
1261 
1262     is_scm = (
1263         req_type == 'git' or
1264         (not req_type and not b_tar_path and collection.startswith(('git+', 'git@')))
1265     )
1266 
1267     if is_file:
1268         display.vvvv("Collection requirement '%s' is a tar artifact" % to_text(collection))
1269         b_tar_path = to_bytes(collection, errors='surrogate_or_strict')
1270     elif is_url:
1271         display.vvvv("Collection requirement '%s' is a URL to a tar artifact" % collection)
1272         try:
1273             b_tar_path = _download_file(collection, b_temp_path, None, validate_certs)
1274         except urllib_error.URLError as err:
1275             raise AnsibleError("Failed to download collection tar from '%s': %s"
1276                                % (to_native(collection), to_native(err)))
1277 
1278     if is_scm:
1279         if not collection.startswith('git'):
1280             collection = 'git+' + collection
1281 
1282         name, version, path, fragment = parse_scm(collection, requirement)
1283         b_tar_path = scm_archive_collection(path, name=name, version=version)
1284 
1285         with tarfile.open(b_tar_path, mode='r') as collection_tar:
1286             collection_tar.extractall(path=to_text(b_temp_path))
1287 
1288         # Ignore requirement if it is set (it must follow semantic versioning, unlike a git version, which is any tree-ish)
1289         # If the requirement was the only place version was set, requirement == version at this point
1290         if requirement not in {"*", ""} and requirement != version:
1291             display.warning(
1292                 "The collection {0} appears to be a git repository and two versions were provided: '{1}', and '{2}'. "
1293                 "The version {2} is being disregarded.".format(collection, version, requirement)
1294             )
1295         requirement = "*"
1296 
1297         reqs = _collections_from_scm(collection, requirement, b_temp_path, force, parent)
1298         for req in reqs:
1299             collection_info = get_collection_info_from_req(dep_map, req)
1300             update_dep_map_collection_info(dep_map, existing_collections, collection_info, parent, requirement)
1301     else:
1302         if b_tar_path:
1303             req = CollectionRequirement.from_tar(b_tar_path, force, parent=parent)
1304             collection_info = get_collection_info_from_req(dep_map, req)
1305         else:
1306             validate_collection_name(collection)
1307 
1308             display.vvvv("Collection requirement '%s' is the name of a collection" % collection)
1309             if collection in dep_map:
1310                 collection_info = dep_map[collection]
1311                 collection_info.add_requirement(parent, requirement)
1312             else:
1313                 apis = [source] if source else apis
1314                 collection_info = CollectionRequirement.from_name(collection, apis, requirement, force, parent=parent,
1315                                                                   allow_pre_release=allow_pre_release)
1316 
1317         update_dep_map_collection_info(dep_map, existing_collections, collection_info, parent, requirement)

1340 def parse_scm(collection, version):
1341     if ',' in collection:
1342         collection, version = collection.split(',', 1)
1343     elif version == '*' or not version:
1344         version = 'HEAD'
1345 
1346     if collection.startswith('git+'):
1347         path = collection[4:]
1348     else:
1349         path = collection
1350 
1351     path, fragment = urldefrag(path)
1352     fragment = fragment.strip(os.path.sep)
1353 
1354     if path.endswith(os.path.sep + '.git'):
1355         name = path.split(os.path.sep)[-2]
1356     elif '://' not in path and '@' not in path:
1357         name = path
1358     else:
1359         name = path.split('/')[-1]
1360         if name.endswith('.git'):
1361             name = name[:-4]
1362 
1363     return name, version, path, fragment

1429 def _extract_tar_file(tar, filename, b_dest, b_temp_path, expected_hash=None):
1430     """ Extracts a file from a collection tar. """
1431     with _get_tar_file_member(tar, filename) as (tar_member, tar_obj):
1432         if tar_member.type == tarfile.SYMTYPE:
1433             actual_hash = _consume_file(tar_obj)
1434 
1435         else:
1436             with tempfile.NamedTemporaryFile(dir=b_temp_path, delete=False) as tmpfile_obj:
1437                 actual_hash = _consume_file(tar_obj, tmpfile_obj)
1438 
1439         if expected_hash and actual_hash != expected_hash:

1519 def _consume_file(read_from, write_to=None):
1520     bufsize = 65536
1521     sha256_digest = sha256()
1522     data = read_from.read(bufsize)
1523     while data:
1524         if write_to is not None:
1525             write_to.write(data)
1526             write_to.flush()
1527         sha256_digest.update(data)
1528         data = read_from.read(bufsize)
1529 
1530     return sha256_digest.hexdigest()
