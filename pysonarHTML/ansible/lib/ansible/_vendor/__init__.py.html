   4 from __future__ import (absolute_import, division, print_function)

   7 import os
   8 import pkgutil
   9 import sys
  10 import warnings
  11 
  12 # This package exists to host vendored top-level Python packages for downstream packaging. Any Python packages
  13 # installed beneath this one will be masked from the Ansible loader, and available from the front of sys.path.
  14 # It is expected that the vendored packages will be loaded very early, so a warning will be fired on import of
  15 # the top-level ansible package if any packages beneath this are already loaded at that point.
  16 #
  17 # Python packages may be installed here during downstream packaging using something like:
  18 # pip install --upgrade -t (path to this dir) cryptography pyyaml packaging jinja2
  19 
  20 # mask vendored content below this package from being accessed as an ansible subpackage
  21 __path__ = []
  22 
  23 
  24 def _ensure_vendored_path_entry():
  25     """
  26     Ensure that any downstream-bundled content beneath this package is available at the top of sys.path
  27     """
  28     # patch our vendored dir onto sys.path
  29     vendored_path_entry = os.path.dirname(__file__)
  30     vendored_module_names = set(m[1] for m in pkgutil.iter_modules([vendored_path_entry], ''))  # m[1] == m.name
  31 
  32     if vendored_module_names:
  33         # patch us early to load vendored deps transparently
  34         if vendored_path_entry in sys.path:
  35             # handle reload case by removing the existing entry, wherever it might be
  36             sys.path.remove(vendored_path_entry)
  37         sys.path.insert(0, vendored_path_entry)
  38 
  39         already_loaded_vendored_modules = set(sys.modules.keys()).intersection(vendored_module_names)
  40 
  41         if already_loaded_vendored_modules:
  42             warnings.warn('One or more Python packages bundled by this ansible-base distribution were already '
  43                           'loaded ({0}). This may result in undefined behavior.'.format(', '.join(sorted(already_loaded_vendored_modules))))
  44
