 291 def clear_facls(path):
 292     setfacl = get_bin_path('setfacl')
 293     # FIXME "setfacl -b" is available on Linux and FreeBSD. There is "setfacl -D e" on z/OS. Others?
 294     acl_command = [setfacl, '-b', path]
 295     b_acl_command = [to_bytes(x) for x in acl_command]
 296     rc, out, err = module.run_command(b_acl_command, environ_update=dict(LANG='C', LC_ALL='C', LC_MESSAGES='C'))
 297     if rc != 0:
 298         raise RuntimeError('Error running "{0}": stdout: "{1}"; stderr: "{2}"'.format(' '.join(b_acl_command), out, err))

 301 def split_pre_existing_dir(dirname):
 302     '''
 303     Return the first pre-existing directory and a list of the new directories that will be created.
 304     '''
 305     head, tail = os.path.split(dirname)
 306     b_head = to_bytes(head, errors='surrogate_or_strict')
 307     if head == '':
 308         return ('.', [tail])
 309     if not os.path.exists(b_head):
 310         if head == '/':
 311             raise AnsibleModuleError(results={'msg': "The '/' directory doesn't exist on this machine."})
 312         (pre_existing_dir, new_directory_list) = split_pre_existing_dir(head)
 313     else:
 314         return (head, [tail])
 315     new_directory_list.append(tail)
 316     return (pre_existing_dir, new_directory_list)

 332 def chown_recursive(path, module):
 333     changed = False
 334     owner = module.params['owner']
 335     group = module.params['group']
 336 
 337     if owner is not None:
 338         if not module.check_mode:
 339             for dirpath, dirnames, filenames in os.walk(path):
 340                 owner_changed = module.set_owner_if_different(dirpath, owner, False)
 341                 if owner_changed is True:
 342                     changed = owner_changed
 343                 for dir in [os.path.join(dirpath, d) for d in dirnames]:
 344                     owner_changed = module.set_owner_if_different(dir, owner, False)
 345                     if owner_changed is True:
 346                         changed = owner_changed
 347                 for file in [os.path.join(dirpath, f) for f in filenames]:
 348                     owner_changed = module.set_owner_if_different(file, owner, False)
 349                     if owner_changed is True:
 350                         changed = owner_changed
 351         else:
 352             uid = pwd.getpwnam(owner).pw_uid
 353             for dirpath, dirnames, filenames in os.walk(path):
 354                 owner_changed = (os.stat(dirpath).st_uid != uid)
 355                 if owner_changed is True:
 356                     changed = owner_changed
 357                 for dir in [os.path.join(dirpath, d) for d in dirnames]:
 358                     owner_changed = (os.stat(dir).st_uid != uid)
 359                     if owner_changed is True:
 360                         changed = owner_changed
 361                 for file in [os.path.join(dirpath, f) for f in filenames]:
 362                     owner_changed = (os.stat(file).st_uid != uid)
 363                     if owner_changed is True:
 364                         changed = owner_changed
 365     if group is not None:
 366         if not module.check_mode:
 367             for dirpath, dirnames, filenames in os.walk(path):
 368                 group_changed = module.set_group_if_different(dirpath, group, False)
 369                 if group_changed is True:
 370                     changed = group_changed
 371                 for dir in [os.path.join(dirpath, d) for d in dirnames]:
 372                     group_changed = module.set_group_if_different(dir, group, False)
 373                     if group_changed is True:
 374                         changed = group_changed
 375                 for file in [os.path.join(dirpath, f) for f in filenames]:
 376                     group_changed = module.set_group_if_different(file, group, False)
 377                     if group_changed is True:
 378                         changed = group_changed
 379         else:
 380             gid = grp.getgrnam(group).gr_gid
 381             for dirpath, dirnames, filenames in os.walk(path):
 382                 group_changed = (os.stat(dirpath).st_gid != gid)
 383                 if group_changed is True:
 384                     changed = group_changed
 385                 for dir in [os.path.join(dirpath, d) for d in dirnames]:
 386                     group_changed = (os.stat(dir).st_gid != gid)
 387                     if group_changed is True:
 388                         changed = group_changed
 389                 for file in [os.path.join(dirpath, f) for f in filenames]:
 390                     group_changed = (os.stat(file).st_gid != gid)
 391                     if group_changed is True:
 392                         changed = group_changed
 393 
 394     return changed

 397 def copy_diff_files(src, dest, module):
 398     """Copy files that are different between `src` directory and `dest` directory."""
 399 
 400     changed = False
 401     owner = module.params['owner']
 402     group = module.params['group']
 403     local_follow = module.params['local_follow']
 404     diff_files = filecmp.dircmp(src, dest).diff_files
 405     if len(diff_files):
 406         changed = True
 407     if not module.check_mode:
 408         for item in diff_files:
 409             src_item_path = os.path.join(src, item)
 410             dest_item_path = os.path.join(dest, item)
 411             b_src_item_path = to_bytes(src_item_path, errors='surrogate_or_strict')
 412             b_dest_item_path = to_bytes(dest_item_path, errors='surrogate_or_strict')
 413             if os.path.islink(b_src_item_path) and local_follow is False:
 414                 linkto = os.readlink(b_src_item_path)
 415                 os.symlink(linkto, b_dest_item_path)
 416             else:
 417                 shutil.copyfile(b_src_item_path, b_dest_item_path)
 418                 shutil.copymode(b_src_item_path, b_dest_item_path)
 419 
 420             if owner is not None:
 421                 module.set_owner_if_different(b_dest_item_path, owner, False)
 422             if group is not None:
 423                 module.set_group_if_different(b_dest_item_path, group, False)
 424             changed = True
 425     return changed

 428 def copy_left_only(src, dest, module):
 429     """Copy files that exist in `src` directory only to the `dest` directory."""
 430 
 431     changed = False
 432     owner = module.params['owner']
 433     group = module.params['group']
 434     local_follow = module.params['local_follow']
 435     left_only = filecmp.dircmp(src, dest).left_only
 436     if len(left_only):
 437         changed = True
 438     if not module.check_mode:
 439         for item in left_only:
 440             src_item_path = os.path.join(src, item)
 441             dest_item_path = os.path.join(dest, item)
 442             b_src_item_path = to_bytes(src_item_path, errors='surrogate_or_strict')
 443             b_dest_item_path = to_bytes(dest_item_path, errors='surrogate_or_strict')
 444 
 445             if os.path.islink(b_src_item_path) and os.path.isdir(b_src_item_path) and local_follow is True:
 446                 shutil.copytree(b_src_item_path, b_dest_item_path, symlinks=not(local_follow))
 447                 chown_recursive(b_dest_item_path, module)
 448 
 449             if os.path.islink(b_src_item_path) and os.path.isdir(b_src_item_path) and local_follow is False:
 450                 linkto = os.readlink(b_src_item_path)
 451                 os.symlink(linkto, b_dest_item_path)
 452 
 453             if os.path.islink(b_src_item_path) and os.path.isfile(b_src_item_path) and local_follow is True:
 454                 shutil.copyfile(b_src_item_path, b_dest_item_path)
 455                 if owner is not None:
 456                     module.set_owner_if_different(b_dest_item_path, owner, False)
 457                 if group is not None:
 458                     module.set_group_if_different(b_dest_item_path, group, False)
 459 
 460             if os.path.islink(b_src_item_path) and os.path.isfile(b_src_item_path) and local_follow is False:
 461                 linkto = os.readlink(b_src_item_path)
 462                 os.symlink(linkto, b_dest_item_path)
 463 
 464             if not os.path.islink(b_src_item_path) and os.path.isfile(b_src_item_path):
 465                 shutil.copyfile(b_src_item_path, b_dest_item_path)
 466                 shutil.copymode(b_src_item_path, b_dest_item_path)
 467 
 468                 if owner is not None:
 469                     module.set_owner_if_different(b_dest_item_path, owner, False)
 470                 if group is not None:
 471                     module.set_group_if_different(b_dest_item_path, group, False)
 472 
 473             if not os.path.islink(b_src_item_path) and os.path.isdir(b_src_item_path):
 474                 shutil.copytree(b_src_item_path, b_dest_item_path, symlinks=not(local_follow))
 475                 chown_recursive(b_dest_item_path, module)
 476 
 477             changed = True
 478     return changed

 481 def copy_common_dirs(src, dest, module):
 482     changed = False
 483     common_dirs = filecmp.dircmp(src, dest).common_dirs
 484     for item in common_dirs:
 485         src_item_path = os.path.join(src, item)
 486         dest_item_path = os.path.join(dest, item)
 487         b_src_item_path = to_bytes(src_item_path, errors='surrogate_or_strict')
 488         b_dest_item_path = to_bytes(dest_item_path, errors='surrogate_or_strict')
 489         diff_files_changed = copy_diff_files(b_src_item_path, b_dest_item_path, module)
 490         left_only_changed = copy_left_only(b_src_item_path, b_dest_item_path, module)
 491         if diff_files_changed or left_only_changed:
 492             changed = True
 493 
 494         # recurse into subdirectory
 495         changed = changed or copy_common_dirs(os.path.join(src, item), os.path.join(dest, item), module)
 496     return changed

 499 def main():
 500 
 501     global module
 502 
 503     module = AnsibleModule(
 504         # not checking because of daisy chain to file module
 505         argument_spec=dict(
 506             src=dict(type='path'),
 507             _original_basename=dict(type='str'),  # used to handle 'dest is a directory' via template, a slight hack
 508             content=dict(type='str', no_log=True),
 509             dest=dict(type='path', required=True),
 510             backup=dict(type='bool', default=False),
 511             force=dict(type='bool', default=True, aliases=['thirsty']),
 512             validate=dict(type='str'),
 513             directory_mode=dict(type='raw'),
 514             remote_src=dict(type='bool'),
 515             local_follow=dict(type='bool'),
 516             checksum=dict(type='str'),
 517             follow=dict(type='bool', default=False),
 518         ),
 519         add_file_common_args=True,
 520         supports_check_mode=True,
 521     )
 522 
 523     if module.params.get('thirsty'):
 524         module.deprecate('The alias "thirsty" has been deprecated and will be removed, use "force" instead',
 525                          version='2.13', collection_name='ansible.builtin')
 526 
 527     src = module.params['src']
 528     b_src = to_bytes(src, errors='surrogate_or_strict')
 529     dest = module.params['dest']
 530     # Make sure we always have a directory component for later processing
 531     if os.path.sep not in dest:
 532         dest = '.{0}{1}'.format(os.path.sep, dest)
 533     b_dest = to_bytes(dest, errors='surrogate_or_strict')
 534     backup = module.params['backup']
 535     force = module.params['force']
 536     _original_basename = module.params.get('_original_basename', None)
 537     validate = module.params.get('validate', None)
 538     follow = module.params['follow']
 539     local_follow = module.params['local_follow']
 540     mode = module.params['mode']
 541     owner = module.params['owner']
 542     group = module.params['group']
 543     remote_src = module.params['remote_src']
 544     checksum = module.params['checksum']
 545 
 546     if not os.path.exists(b_src):
 547         module.fail_json(msg="Source %s not found" % (src))
 548     if not os.access(b_src, os.R_OK):
 549         module.fail_json(msg="Source %s not readable" % (src))
 550 
 551     # Preserve is usually handled in the action plugin but mode + remote_src has to be done on the
 552     # remote host
 553     if module.params['mode'] == 'preserve':
 554         module.params['mode'] = '0%03o' % stat.S_IMODE(os.stat(b_src).st_mode)
 555     mode = module.params['mode']
 556 
 557     checksum_dest = None
 558 
 559     if os.path.isfile(src):
 560         checksum_src = module.sha1(src)
 561     else:
 562         checksum_src = None
 563 
 564     # Backwards compat only.  This will be None in FIPS mode
 565     try:
 566         if os.path.isfile(src):
 567             md5sum_src = module.md5(src)
 568         else:
 569             md5sum_src = None
 570     except ValueError:
 571         md5sum_src = None
 572 
 573     changed = False
 574 
 575     if checksum and checksum_src != checksum:
 576         module.fail_json(
 577             msg='Copied file does not match the expected checksum. Transfer failed.',
 578             checksum=checksum_src,
 579             expected_checksum=checksum
 580         )
 581 
 582     # Special handling for recursive copy - create intermediate dirs
 583     if dest.endswith(os.sep):
 584         if _original_basename:
 585             dest = os.path.join(dest, _original_basename)
 586         b_dest = to_bytes(dest, errors='surrogate_or_strict')
