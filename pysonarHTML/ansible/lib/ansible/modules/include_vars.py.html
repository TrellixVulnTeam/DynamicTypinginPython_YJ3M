  10 DOCUMENTATION = r'''
  11 ---
  12 author: Allen Sanabria (@linuxdynasty)
  13 module: include_vars
  14 short_description: Load variables from files, dynamically within a task
  15 description:
  16   - Loads YAML/JSON variables dynamically from a file or directory, recursively, during task runtime.
  17   - If loading a directory, the files are sorted alphabetically before being loaded.
  18   - This module is also supported for Windows targets.
  19   - To assign included variables to a different host than C(inventory_hostname),
  20     use C(delegate_to) and set C(delegate_facts=yes).
  21 version_added: "1.4"
  22 options:
  23   file:
  24     description:
  25       - The file name from which variables should be loaded.
  26       - If the path is relative, it will look for the file in vars/ subdirectory of a role or relative to playbook.
  27     type: path
  28     version_added: "2.2"
  29   dir:
  30     description:
  31       - The directory name from which the variables should be loaded.
  32       - If the path is relative and the task is inside a role, it will look inside the role's vars/ subdirectory.
  33       - If the path is relative and not inside a role, it will be parsed relative to the playbook.
  34     type: path
  35     version_added: "2.2"
  36   name:
  37     description:
  38       - The name of a variable into which assign the included vars.
  39       - If omitted (null) they will be made top level vars.
  40     type: str
  41     version_added: "2.2"
  42   depth:
  43     description:
  44       - When using C(dir), this module will, by default, recursively go through each sub directory and load up the
  45         variables. By explicitly setting the depth, this module will only go as deep as the depth.
  46     type: int
  47     default: 0
  48     version_added: "2.2"
  49   files_matching:
  50     description:
  51       - Limit the files that are loaded within any directory to this regular expression.
  52     type: str
  53     version_added: "2.2"
  54   ignore_files:
  55     description:
  56       - List of file names to ignore.
  57     type: list
  58     version_added: "2.2"
  59   extensions:
  60     description:
  61       - List of file extensions to read when using C(dir).
  62     type: list
  63     default: [ json, yaml, yml ]
  64     version_added: "2.3"
  65   ignore_unknown_extensions:
  66     description:
  67       - Ignore unknown file extensions within the directory.
  68       - This allows users to specify a directory containing vars files that are intermingled with non-vars files extension types
  69         (e.g. a directory with a README in it and vars files).
  70     type: bool
  71     default: no
  72     version_added: "2.7"
  73   free-form:
  74     description:
  75       - This module allows you to specify the 'file' option directly without any other options.
  76       - There is no 'free-form' option, this is just an indicator, see example below.
  77 notes:
  78   - This module is also supported for Windows targets.
  79 seealso:
  80 - module: set_fact
  81 - ref: playbooks_delegation
  82   description: More information related to task delegation.
  83 '''

  85 EXAMPLES = r'''
  86 - name: Include vars of stuff.yaml into the 'stuff' variable (2.2).
  87   include_vars:
  88     file: stuff.yaml
  89     name: stuff
  90 
  91 - name: Conditionally decide to load in variables into 'plans' when x is 0, otherwise do not. (2.2)
  92   include_vars:
  93     file: contingency_plan.yaml
  94     name: plans
  95   when: x == 0
  96 
  97 - name: Load a variable file based on the OS type, or a default if not found. Using free-form to specify the file.
  98   include_vars: "{{ lookup('first_found', params) }}"
  99   vars:
 100     params:
 101       files:
 102         - '{{ansible_distribution}}.yaml'
 103         - '{{ansible_os_family}}.yaml'
 104         - default.yaml
 105       paths:
 106         - 'vars'
 107 
 108 - name: Bare include (free-form)
 109   include_vars: myvars.yaml
 110 
 111 - name: Include all .json and .jsn files in vars/all and all nested directories (2.3)
 112   include_vars:
 113     dir: vars/all
 114     extensions:
 115       - 'json'
 116       - 'jsn'
 117 
 118 - name: Include all default extension files in vars/all and all nested directories and save the output in test. (2.2)
 119   include_vars:
 120     dir: vars/all
 121     name: test
 122 
 123 - name: Include default extension files in vars/services (2.2)
 124   include_vars:
 125     dir: vars/services
 126     depth: 1
 127 
 128 - name: Include only files matching bastion.yaml (2.2)
 129   include_vars:
 130     dir: vars
 131     files_matching: bastion.yaml
 132 
 133 - name: Include all .yaml files except bastion.yaml (2.3)
 134   include_vars:
 135     dir: vars
 136     ignore_files:
 137       - 'bastion.yaml'
 138     extensions:
 139       - 'yaml'
 140 
 141 - name: Ignore warnings raised for files with unknown extensions while loading (2.7)
 142   include_vars:
 143     dir: vars
 144     ignore_unknown_extensions: True
 145     extensions:
 146       - ''
 147       - 'yaml'
 148       - 'yml'
 149       - 'json'
 150 '''
