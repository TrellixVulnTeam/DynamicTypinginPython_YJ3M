 112 class Group(object):
 113     """
 114     This is a generic Group manipulation class that is subclassed
 115     based on platform.
 116 
 117     A subclass may wish to override the following action methods:-
 118       - group_del()
 119       - group_add()
 120       - group_mod()
 121 
 122     All subclasses MUST define platform and distribution (which may be None).
 123     """
 124 
 125     platform = 'Generic'
 126     distribution = None
 127     GROUPFILE = '/etc/group'
 128 
 129     def __new__(cls, *args, **kwargs):
 130         new_cls = get_platform_subclass(Group)
 131         return super(cls, new_cls).__new__(new_cls)
 132 
 133     def __init__(self, module):
 134         self.module = module
 135         self.state = module.params['state']
 136         self.name = module.params['name']
 137         self.gid = module.params['gid']
 138         self.system = module.params['system']
 139         self.local = module.params['local']
 140         self.non_unique = module.params['non_unique']
 141 
 142     def execute_command(self, cmd):
 143         return self.module.run_command(cmd)
 144 
 145     def group_del(self):
 146         if self.local:
 147             command_name = 'lgroupdel'
 148         else:
 149             command_name = 'groupdel'
 150         cmd = [self.module.get_bin_path(command_name, True), self.name]
 151         return self.execute_command(cmd)
 152 
 153     def _local_check_gid_exists(self):
 154         if self.gid:
 155             for gr in grp.getgrall():
 156                 if self.gid == gr.gr_gid and self.name != gr.gr_name:
 157                     self.module.fail_json(msg="GID '{0}' already exists with group '{1}'".format(self.gid, gr.gr_name))
 158 
 159     def group_add(self, **kwargs):
 160         if self.local:
 161             command_name = 'lgroupadd'
 162             self._local_check_gid_exists()
 163         else:
 164             command_name = 'groupadd'
 165         cmd = [self.module.get_bin_path(command_name, True)]
 166         for key in kwargs:
 167             if key == 'gid' and kwargs[key] is not None:
 168                 cmd.append('-g')
 169                 cmd.append(str(kwargs[key]))
 170                 if self.non_unique:
 171                     cmd.append('-o')
 172             elif key == 'system' and kwargs[key] is True:
 173                 cmd.append('-r')
 174         cmd.append(self.name)
 175         return self.execute_command(cmd)
 176 
 177     def group_mod(self, **kwargs):
 178         if self.local:
 179             command_name = 'lgroupmod'
 180             self._local_check_gid_exists()
 181         else:
 182             command_name = 'groupmod'
 183         cmd = [self.module.get_bin_path(command_name, True)]
 184         info = self.group_info()
 185         for key in kwargs:
 186             if key == 'gid':
 187                 if kwargs[key] is not None and info[2] != int(kwargs[key]):
 188                     cmd.append('-g')
 189                     cmd.append(str(kwargs[key]))
 190                     if self.non_unique:
 191                         cmd.append('-o')
 192         if len(cmd) == 1:
 193             return (None, '', '')
 194         if self.module.check_mode:
 195             return (0, '', '')
 196         cmd.append(self.name)
 197         return self.execute_command(cmd)
 198 
 199     def group_exists(self):
 200         # The grp module does not distinguish between local and directory accounts.
 201         # It's output cannot be used to determine whether or not a group exists locally.
 202         # It returns True if the group exists locally or in the directory, so instead
 203         # look in the local GROUP file for an existing account.
 204         if self.local:
 205             if not os.path.exists(self.GROUPFILE):
 206                 self.module.fail_json(msg="'local: true' specified but unable to find local group file {0} to parse.".format(self.GROUPFILE))
 207 
 208             exists = False
 209             name_test = '{0}:'.format(self.name)
 210             with open(self.GROUPFILE, 'rb') as f:
 211                 reversed_lines = f.readlines()[::-1]
 212                 for line in reversed_lines:
 213                     if line.startswith(to_bytes(name_test)):
 214                         exists = True
 215                         break
 216 
 217             if not exists:
 218                 self.module.warn(
 219                     "'local: true' specified and group was not found in {file}. "
 220                     "The local group may already exist if the local group database exists somewhere other than {file}.".format(file=self.GROUPFILE))
 221 
 222             return exists
 223 
 224         else:
 225             try:
 226                 if grp.getgrnam(self.name):
 227                     return True
 228             except KeyError:
 229                 return False
 230 
 231     def group_info(self):
 232         if not self.group_exists():
 233             return False
 234         try:
 235             info = list(grp.getgrnam(self.name))
 236         except KeyError:
 237             return False
 238         return info

 270 class AIX(Group):
 271     """
 272     This is a AIX Group manipulation class.
 273 
 274     This overrides the following methods from the generic class:-
 275       - group_del()
 276       - group_add()
 277       - group_mod()
 278     """
 279 
 280     platform = 'AIX'
 281     distribution = None
 282     GROUPFILE = '/etc/group'
 283 
 284     def group_del(self):
 285         cmd = [self.module.get_bin_path('rmgroup', True), self.name]
 286         return self.execute_command(cmd)
 287 
 288     def group_add(self, **kwargs):
 289         cmd = [self.module.get_bin_path('mkgroup', True)]
 290         for key in kwargs:
 291             if key == 'gid' and kwargs[key] is not None:
 292                 cmd.append('id=' + str(kwargs[key]))
 293             elif key == 'system' and kwargs[key] is True:
 294                 cmd.append('-a')
 295         cmd.append(self.name)
 296         return self.execute_command(cmd)
 297 
 298     def group_mod(self, **kwargs):
 299         cmd = [self.module.get_bin_path('chgroup', True)]
 300         info = self.group_info()
 301         for key in kwargs:
 302             if key == 'gid':
 303                 if kwargs[key] is not None and info[2] != int(kwargs[key]):
 304                     cmd.append('id=' + str(kwargs[key]))
 305         if len(cmd) == 1:
 306             return (None, '', '')
 307         if self.module.check_mode:
 308             return (0, '', '')
 309         cmd.append(self.name)
 310         return self.execute_command(cmd)

 316     """
 317     This is a FreeBSD Group manipulation class.

 319     This overrides the following methods from the generic class:-
 320       - group_del()
 321       - group_add()
 322       - group_mod()
 323     """
 324 
 325     platform = 'FreeBSD'
 326     distribution = None
 327     GROUPFILE = '/etc/group'
 328 
 329     def group_del(self):
 330         cmd = [self.module.get_bin_path('pw', True), 'groupdel', self.name]
 331         return self.execute_command(cmd)
 332 
 333     def group_add(self, **kwargs):
 334         cmd = [self.module.get_bin_path('pw', True), 'groupadd', self.name]
 335         if self.gid is not None:
 336             cmd.append('-g')
 337             cmd.append(str(self.gid))
 338             if self.non_unique:
 339                 cmd.append('-o')
 340         return self.execute_command(cmd)
 341 
 342     def group_mod(self, **kwargs):
 343         cmd = [self.module.get_bin_path('pw', True), 'groupmod', self.name]
 344         info = self.group_info()
 345         cmd_len = len(cmd)
 346         if self.gid is not None and int(self.gid) != info[2]:
 347             cmd.append('-g')
 348             cmd.append(str(self.gid))
 349             if self.non_unique:
 350                 cmd.append('-o')
 351         # modify the group if cmd will do anything
 352         if cmd_len != len(cmd):
 353             if self.module.check_mode:
 354                 return (0, '', '')
 355             return self.execute_command(cmd)
 356         return (None, '', '')

 370 class DarwinGroup(Group):
 371     """
 372     This is a Mac macOS Darwin Group manipulation class.
 373 
 374     This overrides the following methods from the generic class:-
 375       - group_del()
 376       - group_add()
 377       - group_mod()
 378 
 379     group manipulation are done using dseditgroup(1).
 380     """
 381 
 382     platform = 'Darwin'
 383     distribution = None
 384 
 385     def group_add(self, **kwargs):
 386         cmd = [self.module.get_bin_path('dseditgroup', True)]
 387         cmd += ['-o', 'create']
 388         if self.gid is not None:
 389             cmd += ['-i', str(self.gid)]
 390         elif 'system' in kwargs and kwargs['system'] is True:
 391             gid = self.get_lowest_available_system_gid()
 392             if gid is not False:
 393                 self.gid = str(gid)
 394                 cmd += ['-i', str(self.gid)]
 395         cmd += ['-L', self.name]
 396         (rc, out, err) = self.execute_command(cmd)
 397         return (rc, out, err)
 398 
 399     def group_del(self):
 400         cmd = [self.module.get_bin_path('dseditgroup', True)]
 401         cmd += ['-o', 'delete']
 402         cmd += ['-L', self.name]
 403         (rc, out, err) = self.execute_command(cmd)
 404         return (rc, out, err)
 405 
 406     def group_mod(self, gid=None):
 407         info = self.group_info()
 408         if self.gid is not None and int(self.gid) != info[2]:
 409             cmd = [self.module.get_bin_path('dseditgroup', True)]
 410             cmd += ['-o', 'edit']
 411             if gid is not None:
 412                 cmd += ['-i', str(gid)]
 413             cmd += ['-L', self.name]
 414             (rc, out, err) = self.execute_command(cmd)
 415             return (rc, out, err)
 416         return (None, '', '')
 417 
 418     def get_lowest_available_system_gid(self):
 419         # check for lowest available system gid (< 500)
 420         try:
 421             cmd = [self.module.get_bin_path('dscl', True)]
 422             cmd += ['/Local/Default', '-list', '/Groups', 'PrimaryGroupID']
 423             (rc, out, err) = self.execute_command(cmd)
 424             lines = out.splitlines()
 425             highest = 0
 426             for group_info in lines:
 427                 parts = group_info.split(' ')
 428                 if len(parts) > 1:
 429                     gid = int(parts[-1])
 430                     if gid > highest and gid < 500:
 431                         highest = gid
 432             if highest == 0 or highest == 499:
 433                 return False
 434             return (highest + 1)
 435         except Exception:
 436             return False

 532 class BusyBoxGroup(Group):
 533     """
 534     BusyBox group manipulation class for systems that have addgroup and delgroup.
 535 
 536     It overrides the following methods:
 537         - group_add()
 538         - group_del()
 539         - group_mod()
 540     """
 541 
 542     def group_add(self, **kwargs):
 543         cmd = [self.module.get_bin_path('addgroup', True)]
 544         if self.gid is not None:
 545             cmd.extend(['-g', str(self.gid)])
 546 
 547         if self.system:
 548             cmd.append('-S')
 549 
 550         cmd.append(self.name)
 551 
 552         return self.execute_command(cmd)
 553 
 554     def group_del(self):
 555         cmd = [self.module.get_bin_path('delgroup', True), self.name]
 556         return self.execute_command(cmd)
 557 
 558     def group_mod(self, **kwargs):
 559         # Since there is no groupmod command, modify /etc/group directly
 560         info = self.group_info()
 561         if self.gid is not None and self.gid != info[2]:
 562             with open('/etc/group', 'rb') as f:
 563                 b_groups = f.read()
 564 
 565             b_name = to_bytes(self.name)
 566             b_current_group_string = b'%s:x:%d:' % (b_name, info[2])
 567             b_new_group_string = b'%s:x:%d:' % (b_name, self.gid)
 568 
 569             if b':%d:' % self.gid in b_groups:
 570                 self.module.fail_json(msg="gid '{gid}' in use".format(gid=self.gid))
 571 
 572             if self.module.check_mode:
 573                 return 0, '', ''
 574             b_new_groups = b_groups.replace(b_current_group_string, b_new_group_string)
 575             with open('/etc/group', 'wb') as f:
 576                 f.write(b_new_groups)
 577             return 0, '', ''
 578 
 579         return None, '', ''

 588 def main():
 589     module = AnsibleModule(
 590         argument_spec=dict(
 591             state=dict(type='str', default='present', choices=['absent', 'present']),
 592             name=dict(type='str', required=True),
 593             gid=dict(type='int'),
 594             system=dict(type='bool', default=False),
 595             local=dict(type='bool', default=False),
 596             non_unique=dict(type='bool', default=False),
 597         ),
 598         supports_check_mode=True,
 599         required_if=[
 600             ['non_unique', True, ['gid']],
 601         ],
 602     )
 603 
 604     group = Group(module)
 605 
 606     module.debug('Group instantiated - platform %s' % group.platform)
 607     if group.distribution:
 608         module.debug('Group instantiated - distribution %s' % group.distribution)
 609 
 610     rc = None
 611     out = ''
 612     err = ''
 613     result = {}
 614     result['name'] = group.name
 615     result['state'] = group.state
 616 
 617     if group.state == 'absent':
 618 
 619         if group.group_exists():
 620             if module.check_mode:
 621                 module.exit_json(changed=True)
 622             (rc, out, err) = group.group_del()
 623             if rc != 0:
 624                 module.fail_json(name=group.name, msg=err)
 625 
 626     elif group.state == 'present':
 627 
 628         if not group.group_exists():
 629             if module.check_mode:
 630                 module.exit_json(changed=True)
 631             (rc, out, err) = group.group_add(gid=group.gid, system=group.system)
 632         else:
 633             (rc, out, err) = group.group_mod(gid=group.gid)
 634 
 635         if rc is not None and rc != 0:
 636             module.fail_json(name=group.name, msg=err)
 637 
 638     if rc is None:
 639         result['changed'] = False
 640     else:
 641         result['changed'] = True
 642     if out:
 643         result['stdout'] = out
 644     if err:
 645         result['stderr'] = err
 646 
 647     if group.group_exists():
 648         info = group.group_info()
 649         result['system'] = group.system
 650         result['gid'] = info[2]
 651 
 652     module.exit_json(**result)
 653 
 654 
 655 if __name__ == '__main__':
 656     main()
