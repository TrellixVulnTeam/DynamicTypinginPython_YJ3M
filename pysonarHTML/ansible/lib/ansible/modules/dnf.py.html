 340     def _sanitize_dnf_error_msg_install(self, spec, error):
 341         """
 342         For unhandled dnf.exceptions.Error scenarios, there are certain error
 343         messages we want to filter in an install scenario. Do that here.
 344         """
 345         if (
 346             to_text("no package matched") in to_text(error) or
 347             to_text("No match for argument:") in to_text(error)
 348         ):
 349             return "No package {0} available.".format(spec)
 350 
 351         return error
 352 
 353     def _sanitize_dnf_error_msg_remove(self, spec, error):
 354         """
 355         For unhandled dnf.exceptions.Error scenarios, there are certain error
 356         messages we want to ignore in a removal scenario as known benign
 357         failures. Do that here.
 358         """
 359         if (
 360             'no package matched' in to_native(error) or
 361             'No match for argument:' in to_native(error)
 362         ):
 363             return (False, "{0} is not installed".format(spec))
 364 
 365         # Return value is tuple of:
 366         #   ("Is this actually a failure?", "Error Message")
 367         return (True, error)

 369     def _package_dict(self, package):
 370         """Return a dictionary of information for the package."""
 371         # NOTE: This no longer contains the 'dnfstate' field because it is
 372         # already known based on the query type.
 373         result = {
 374             'name': package.name,
 375             'arch': package.arch,
 376             'epoch': str(package.epoch),
 377             'release': package.release,
 378             'version': package.version,
 379             'repo': package.repoid}
 380         result['nevra'] = '{epoch}:{name}-{version}-{release}.{arch}'.format(
 381             **result)
 382 
 383         if package.installtime == 0:
 384             result['yumstate'] = 'available'
 385         else:
 386             result['yumstate'] = 'installed'
 387 
 388         return result

 390     def _packagename_dict(self, packagename):
 391         """
 392         Return a dictionary of information for a package name string or None
 393         if the package name doesn't contain at least all NVR elements
 394         """
 395 
 396         if packagename[-4:] == '.rpm':
 397             packagename = packagename[:-4]
 398 
 399         # This list was auto generated on a Fedora 28 system with the following one-liner
 400         #   printf '[ '; for arch in $(ls /usr/lib/rpm/platform); do  printf '"%s", ' ${arch%-linux}; done; printf ']\n'
 401         redhat_rpm_arches = [
 402             "aarch64", "alphaev56", "alphaev5", "alphaev67", "alphaev6", "alpha",
 403             "alphapca56", "amd64", "armv3l", "armv4b", "armv4l", "armv5tejl", "armv5tel",
 404             "armv5tl", "armv6hl", "armv6l", "armv7hl", "armv7hnl", "armv7l", "athlon",
 405             "geode", "i386", "i486", "i586", "i686", "ia32e", "ia64", "m68k", "mips64el",
 406             "mips64", "mips64r6el", "mips64r6", "mipsel", "mips", "mipsr6el", "mipsr6",
 407             "noarch", "pentium3", "pentium4", "ppc32dy4", "ppc64iseries", "ppc64le", "ppc64",
 408             "ppc64p7", "ppc64pseries", "ppc8260", "ppc8560", "ppciseries", "ppc", "ppcpseries",
 409             "riscv64", "s390", "s390x", "sh3", "sh4a", "sh4", "sh", "sparc64", "sparc64v",
 410             "sparc", "sparcv8", "sparcv9", "sparcv9v", "x86_64"
 411         ]
 412 
 413         rpm_arch_re = re.compile(r'(.*)\.(.*)')
 414         rpm_nevr_re = re.compile(r'(\S+)-(?:(\d*):)?(.*)-(~?\w+[\w.+]*)')
 415         try:
 416             arch = None
 417             rpm_arch_match = rpm_arch_re.match(packagename)
 418             if rpm_arch_match:
 419                 nevr, arch = rpm_arch_match.groups()
 420                 if arch in redhat_rpm_arches:
 421                     packagename = nevr
 422             rpm_nevr_match = rpm_nevr_re.match(packagename)
 423             if rpm_nevr_match:
 424                 name, epoch, version, release = rpm_nevr_re.match(packagename).groups()
 425                 if not version or not version.split('.')[0].isdigit():
 426                     return None
 427             else:
 428                 return None
 429         except AttributeError as e:
 430             self.module.fail_json(
 431                 msg='Error attempting to parse package: %s, %s' % (packagename, to_native(e)),
 432                 rc=1,
 433                 results=[]
 434             )
 435 
 436         if not epoch:
 437             epoch = "0"
 438 
 439         if ':' in name:
 440             epoch_name = name.split(":")
 441 
 442             epoch = epoch_name[0]
 443             name = ''.join(epoch_name[1:])
 444 
 445         result = {
 446             'name': name,
 447             'epoch': epoch,
 448             'release': release,
 449             'version': version,
 450         }
 451 
 452         return result

 454     # Original implementation from yum.rpmUtils.miscutils (GPLv2+)
 455     #   http://yum.baseurl.org/gitweb?p=yum.git;a=blob;f=rpmUtils/miscutils.py
 456     def _compare_evr(self, e1, v1, r1, e2, v2, r2):
 457         # return 1: a is newer than b
 458         # 0: a and b are the same version
 459         # -1: b is newer than a
 460         if e1 is None:
 461             e1 = '0'
 462         else:
 463             e1 = str(e1)
 464         v1 = str(v1)
 465         r1 = str(r1)
 466         if e2 is None:
 467             e2 = '0'
 468         else:
 469             e2 = str(e2)
 470         v2 = str(v2)
 471         r2 = str(r2)
 472         # print '%s, %s, %s vs %s, %s, %s' % (e1, v1, r1, e2, v2, r2)
 473         rc = dnf.rpm.rpm.labelCompare((e1, v1, r1), (e2, v2, r2))
 474         # print '%s, %s, %s vs %s, %s, %s = %s' % (e1, v1, r1, e2, v2, r2, rc)
 475         return rc

 477     def _ensure_dnf(self):
 478         if not HAS_DNF:
 479             if PY2:

 482                 package = 'python3-dnf'

 486                     msg="`{0}` is not installed, but it is required"
 487                     "for the Ansible dnf module.".format(package),
 488                     results=[],
 489                 )
 490 
 491             rc, stdout, stderr = self.module.run_command(['dnf', 'install', '-y', package])
 492             global dnf
 493             try:
 494                 import dnf
 495                 import dnf.cli
 496                 import dnf.const
 497                 import dnf.exceptions
 498                 import dnf.subject
 499                 import dnf.util
 500             except ImportError:

 503                         "Please install `{2}` package or ensure you have specified the "

 507                     cmd='dnf install -y {0}'.format(package),

 513     def _configure_base(self, base, conf_file, disable_gpg_check, installroot='/'):
 514         """Configure the dnf Base object."""
 515 
 516         conf = base.conf
 517 
 518         # Change the configuration file path if provided, this must be done before conf.read() is called
 519         if conf_file:
 520             # Fail if we can't read the configuration file.
 521             if not os.access(conf_file, os.R_OK):
 522                 self.module.fail_json(
 523                     msg="cannot read configuration file", conf_file=conf_file,
 524                     results=[],
 525                 )
 526             else:
 527                 conf.config_file_path = conf_file
 528 
 529         # Read the configuration file
 530         conf.read()
 531 
 532         # Turn off debug messages in the output
 533         conf.debuglevel = 0
 534 
 535         # Set whether to check gpg signatures
 536         conf.gpgcheck = not disable_gpg_check
 537         conf.localpkg_gpgcheck = not disable_gpg_check
 538 
 539         # Don't prompt for user confirmations
 540         conf.assumeyes = True
 541 
 542         # Set installroot
 543         conf.installroot = installroot
 544 
 545         # Load substitutions from the filesystem
 546         conf.substitutions.update_from_etc(installroot)
 547 
 548         # Handle different DNF versions immutable mutable datatypes and
 549         # dnf v1/v2/v3
 550         #
 551         # In DNF < 3.0 are lists, and modifying them works
 552         # In DNF >= 3.0 < 3.6 are lists, but modifying them doesn't work
 553         # In DNF >= 3.6 have been turned into tuples, to communicate that modifying them doesn't work
 554         #
 555         # https://www.happyassassin.net/2018/06/27/adams-debugging-adventures-the-immutable-mutable-object/
 556         #
 557         # Set excludes
 558         if self.exclude:
 559             _excludes = list(conf.exclude)
 560             _excludes.extend(self.exclude)
 561             conf.exclude = _excludes
 562         # Set disable_excludes
 563         if self.disable_excludes:
 564             _disable_excludes = list(conf.disable_excludes)
 565             if self.disable_excludes not in _disable_excludes:
 566                 _disable_excludes.append(self.disable_excludes)
 567                 conf.disable_excludes = _disable_excludes
 568 
 569         # Set releasever
 570         if self.releasever is not None:
 571             conf.substitutions['releasever'] = self.releasever
 572 
 573         # Set skip_broken (in dnf this is strict=0)
 574         if self.skip_broken:
 575             conf.strict = 0
 576 
 577         if self.download_only:
 578             conf.downloadonly = True
 579             if self.download_dir:
 580                 conf.destdir = self.download_dir
 581 
 582         # Default in dnf upstream is true
 583         conf.clean_requirements_on_remove = self.autoremove
 584 
 585         # Default in dnf (and module default) is True
 586         conf.install_weak_deps = self.install_weak_deps
 587 
 588     def _specify_repositories(self, base, disablerepo, enablerepo):
 589         """Enable and disable repositories matching the provided patterns."""
 590         base.read_all_repos()
 591         repos = base.repos
 592 
 593         # Disable repositories
 594         for repo_pattern in disablerepo:
 595             if repo_pattern:
 596                 for repo in repos.get_matching(repo_pattern):
 597                     repo.disable()
 598 
 599         # Enable repositories
 600         for repo_pattern in enablerepo:
 601             if repo_pattern:
 602                 for repo in repos.get_matching(repo_pattern):
 603                     repo.enable()
 604 
 605     def _base(self, conf_file, disable_gpg_check, disablerepo, enablerepo, installroot):
 606         """Return a fully configured dnf Base object."""
 607         base = dnf.Base()
 608         self._configure_base(base, conf_file, disable_gpg_check, installroot)
 609         try:
 610             # this method has been supported in dnf-4.2.17-6 or later
 611             # https://bugzilla.redhat.com/show_bug.cgi?id=1788212
 612             base.setup_loggers()
 613         except AttributeError:
 614             pass
 615         try:
 616             base.init_plugins(set(self.disable_plugin), set(self.enable_plugin))
 617             base.pre_configure_plugins()
 618         except AttributeError:
 619             pass  # older versions of dnf didn't require this and don't have these methods
 620         self._specify_repositories(base, disablerepo, enablerepo)
 621         try:
 622             base.configure_plugins()
 623         except AttributeError:
 624             pass  # older versions of dnf didn't require this and don't have these methods
 625 
 626         try:
 627             if self.update_cache:
 628                 try:
 629                     base.update_cache()
 630                 except dnf.exceptions.RepoError as e:
 631                     self.module.fail_json(
 632                         msg="{0}".format(to_text(e)),
 633                         results=[],
 634                         rc=1
 635                     )
 636             base.fill_sack(load_system_repo='auto')
 637         except dnf.exceptions.RepoError as e:
 638             self.module.fail_json(
 639                 msg="{0}".format(to_text(e)),
 640                 results=[],
 641                 rc=1
 642             )
 643         if self.bugfix:
 644             key = {'advisory_type__eq': 'bugfix'}
 645             base._update_security_filters = [base.sack.query().filter(**key)]
 646         if self.security:
 647             key = {'advisory_type__eq': 'security'}
 648             base._update_security_filters = [base.sack.query().filter(**key)]
 649 
 650         return base

 652     def list_items(self, command):
 653         """List package info based on the command."""
 654         # Rename updates to upgrades
 655         if command == 'updates':
 656             command = 'upgrades'
 657 
 658         # Return the corresponding packages
 659         if command in ['installed', 'upgrades', 'available']:
 660             results = [
 661                 self._package_dict(package)
 662                 for package in getattr(self.base.sack.query(), command)()]
 663         # Return the enabled repository ids
 664         elif command in ['repos', 'repositories']:
 665             results = [
 666                 {'repoid': repo.id, 'state': 'enabled'}
 667                 for repo in self.base.repos.iter_enabled()]
 668         # Return any matching packages
 669         else:
 670             packages = dnf.subject.Subject(command).get_best_query(self.base.sack)
 671             results = [self._package_dict(package) for package in packages]
 672 
 673         self.module.exit_json(msg="", results=results)

 682     def _is_newer_version_installed(self, pkg_name):
 683         candidate_pkg = self._packagename_dict(pkg_name)
 684         if not candidate_pkg:
 685             # The user didn't provide a versioned rpm, so version checking is
 686             # not required
 687             return False
 688 
 689         installed = self.base.sack.query().installed()
 690         installed_pkg = installed.filter(name=candidate_pkg['name']).run()
 691         if installed_pkg:
 692             installed_pkg = installed_pkg[0]
 693 
 694             # this looks weird but one is a dict and the other is a dnf.Package
 695             evr_cmp = self._compare_evr(
 696                 installed_pkg.epoch, installed_pkg.version, installed_pkg.release,
 697                 candidate_pkg['epoch'], candidate_pkg['version'], candidate_pkg['release'],
 698             )
 699 
 700             if evr_cmp == 1:
 701                 return True
 702             else:
 703                 return False
 704 
 705         else:
 706 
 707             return False

 709     def _mark_package_install(self, pkg_spec, upgrade=False):
 710         """Mark the package for install."""
 711         is_newer_version_installed = self._is_newer_version_installed(pkg_spec)
 712         is_installed = self._is_installed(pkg_spec)
 713         try:
 714             if is_newer_version_installed:
 715                 if self.allow_downgrade:
 716                     # dnf only does allow_downgrade, we have to handle this ourselves
 717                     # because it allows a possibility for non-idempotent transactions
 718                     # on a system's package set (pending the yum repo has many old
 719                     # NVRs indexed)
 720                     if upgrade:
 721                         if is_installed:
 722                             self.base.upgrade(pkg_spec)
 723                         else:
 724                             self.base.install(pkg_spec)
 725                     else:
 726                         self.base.install(pkg_spec)
 727                 else:  # Nothing to do, report back
 728                     pass
 729             elif is_installed:  # An potentially older (or same) version is installed
 730                 if upgrade:
 731                     self.base.upgrade(pkg_spec)
 732                 else:  # Nothing to do, report back
 733                     pass
 734             else:  # The package is not installed, simply install it
 735                 self.base.install(pkg_spec)
 736 
 737             return {'failed': False, 'msg': '', 'failure': '', 'rc': 0}
 738 
 739         except dnf.exceptions.MarkingError as e:
 740             return {
 741                 'failed': True,
 742                 'msg': "No package {0} available.".format(pkg_spec),
 743                 'failure': " ".join((pkg_spec, to_native(e))),
 744                 'rc': 1,
 745                 "results": []
 746             }
 747 
 748         except dnf.exceptions.DepsolveError as e:
 749             return {
 750                 'failed': True,
 751                 'msg': "Depsolve Error occured for package {0}.".format(pkg_spec),
 752                 'failure': " ".join((pkg_spec, to_native(e))),
 753                 'rc': 1,
 754                 "results": []
 755             }
 756 
 757         except dnf.exceptions.Error as e:
 758             if to_text("already installed") in to_text(e):
 759                 return {'failed': False, 'msg': '', 'failure': ''}
 760             else:
 761                 return {
 762                     'failed': True,
 763                     'msg': "Unknown Error occured for package {0}.".format(pkg_spec),
 764                     'failure': " ".join((pkg_spec, to_native(e))),
 765                     'rc': 1,
 766                     "results": []
 767                 }

 776     def _parse_spec_group_file(self):
 777         pkg_specs, grp_specs, module_specs, filenames = [], [], [], []
 778         already_loaded_comps = False  # Only load this if necessary, it's slow
 779 
 780         for name in self.names:
 781             if '://' in name:
 782                 name = fetch_file(self.module, name)
 783                 filenames.append(name)
 784             elif name.endswith(".rpm"):
 785                 filenames.append(name)
 786             elif name.startswith("@") or ('/' in name):
 787                 # like "dnf install /usr/bin/vi"
 788                 if '/' in name:
 789                     pkg_spec = self._whatprovides(name)
 790                     if pkg_spec:
 791                         pkg_specs.append(pkg_spec)
 792                         continue
 793 
 794                 if not already_loaded_comps:
 795                     self.base.read_comps()
 796                     already_loaded_comps = True
 797 
 798                 grp_env_mdl_candidate = name[1:].strip()
 799 
 800                 if self.with_modules:
 801                     mdl = self.module_base._get_modules(grp_env_mdl_candidate)
 802                     if mdl[0]:
 803                         module_specs.append(grp_env_mdl_candidate)
 804                     else:
 805                         grp_specs.append(grp_env_mdl_candidate)
 806                 else:
 807                     grp_specs.append(grp_env_mdl_candidate)
 808             else:
 809                 pkg_specs.append(name)
 810         return pkg_specs, grp_specs, module_specs, filenames

 812     def _update_only(self, pkgs):
 813         not_installed = []
 814         for pkg in pkgs:
 815             if self._is_installed(pkg):
 816                 try:
 817                     if isinstance(to_text(pkg), text_type):
 818                         self.base.upgrade(pkg)
 819                     else:
 820                         self.base.package_upgrade(pkg)
 821                 except Exception as e:
 822                     self.module.fail_json(
 823                         msg="Error occured attempting update_only operation: {0}".format(to_native(e)),
 824                         results=[],
 825                         rc=1,
 826                     )
 827             else:
 828                 not_installed.append(pkg)
 829 
 830         return not_installed

 832     def _install_remote_rpms(self, filenames):
 833         if int(dnf.__version__.split(".")[0]) >= 2:
 834             pkgs = list(sorted(self.base.add_remote_rpms(list(filenames)), reverse=True))
 835         else:
 836             pkgs = []
 837             try:
 838                 for filename in filenames:
 839                     pkgs.append(self.base.add_remote_rpm(filename))
 840             except IOError as e:
 841                 if to_text("Can not load RPM file") in to_text(e):
 842                     self.module.fail_json(
 843                         msg="Error occured attempting remote rpm install of package: {0}. {1}".format(filename, to_native(e)),
 844                         results=[],
 845                         rc=1,
 846                     )
 847         if self.update_only:
 848             self._update_only(pkgs)
 849         else:
 850             for pkg in pkgs:
 851                 try:
 852                     if self._is_newer_version_installed(self._package_dict(pkg)['nevra']):
 853                         if self.allow_downgrade:
 854                             self.base.package_install(pkg)
 855                     else:
 856                         self.base.package_install(pkg)
 857                 except Exception as e:
 858                     self.module.fail_json(
 859                         msg="Error occured attempting remote rpm operation: {0}".format(to_native(e)),
 860                         results=[],
 861                         rc=1,
 862                     )

 864     def _is_module_installed(self, module_spec):
 865         if self.with_modules:
 866             module_spec = module_spec.strip()
 867             module_list, nsv = self.module_base._get_modules(module_spec)
 868             enabled_streams = self.base._moduleContainer.getEnabledStream(nsv.name)
 869 
 870             if enabled_streams:
 871                 if nsv.stream:
 872                     if nsv.stream in enabled_streams:
 873                         return True     # The provided stream was found
 874                     else:
 875                         return False    # The provided stream was not found
 876                 else:
 877                     return True         # No stream provided, but module found
 878 
 879         return False  # seems like a sane default

 881     def ensure(self):
 882 
 883         response = {
 884             'msg': "",
 885             'changed': False,
 886             'results': [],
 887             'rc': 0
 888         }
 889 
 890         # Accumulate failures.  Package management modules install what they can
 891         # and fail with a message about what they can't.
 892         failure_response = {
 893             'msg': "",
 894             'failures': [],
 895             'results': [],
 896             'rc': 1
 897         }
 898 
 899         # Autoremove is called alone
 900         # Jump to remove path where base.autoremove() is run
 901         if not self.names and self.autoremove:
 902             self.names = []
 903             self.state = 'absent'
 904 
 905         if self.names == ['*'] and self.state == 'latest':
 906             try:
 907                 self.base.upgrade_all()
 908             except dnf.exceptions.DepsolveError as e:
 909                 failure_response['msg'] = "Depsolve Error occured attempting to upgrade all packages"
 910                 self.module.fail_json(**failure_response)
 911         else:
 912             pkg_specs, group_specs, module_specs, filenames = self._parse_spec_group_file()
 913 
 914             pkg_specs = [p.strip() for p in pkg_specs]
 915             filenames = [f.strip() for f in filenames]
 916             groups = []
 917             environments = []
 918             for group_spec in (g.strip() for g in group_specs):
 919                 group = self.base.comps.group_by_pattern(group_spec)
 920                 if group:
 921                     groups.append(group.id)
 922                 else:
 923                     environment = self.base.comps.environment_by_pattern(group_spec)
 924                     if environment:
 925                         environments.append(environment.id)
 926                     else:
 927                         self.module.fail_json(
 928                             msg="No group {0} available.".format(group_spec),
 929                             results=[],
 930                         )
 931 
 932             if self.state in ['installed', 'present']:
 933                 # Install files.
 934                 self._install_remote_rpms(filenames)
 935                 for filename in filenames:
 936                     response['results'].append("Installed {0}".format(filename))
 937 
 938                 # Install modules
 939                 if module_specs and self.with_modules:
 940                     for module in module_specs:
 941                         try:
 942                             if not self._is_module_installed(module):
 943                                 response['results'].append("Module {0} installed.".format(module))
 944                             self.module_base.install([module])
 945                             self.module_base.enable([module])
 946                         except dnf.exceptions.MarkingErrors as e:
 947                             failure_response['failures'].append(' '.join((module, to_native(e))))
 948 
 949                 # Install groups.
 950                 for group in groups:
 951                     try:
 952                         group_pkg_count_installed = self.base.group_install(group, dnf.const.GROUP_PACKAGE_TYPES)
 953                         if group_pkg_count_installed == 0:
 954                             response['results'].append("Group {0} already installed.".format(group))
 955                         else:
 956                             response['results'].append("Group {0} installed.".format(group))
 957                     except dnf.exceptions.DepsolveError as e:
 958                         failure_response['msg'] = "Depsolve Error occured attempting to install group: {0}".format(group)
 959                         self.module.fail_json(**failure_response)
 960                     except dnf.exceptions.Error as e:
 961                         # In dnf 2.0 if all the mandatory packages in a group do
 962                         # not install, an error is raised.  We want to capture
 963                         # this but still install as much as possible.
 964                         failure_response['failures'].append(" ".join((group, to_native(e))))
 965 
 966                 for environment in environments:
 967                     try:
 968                         self.base.environment_install(environment, dnf.const.GROUP_PACKAGE_TYPES)
 969                     except dnf.exceptions.DepsolveError as e:
 970                         failure_response['msg'] = "Depsolve Error occured attempting to install environment: {0}".format(environment)
 971                         self.module.fail_json(**failure_response)
 972                     except dnf.exceptions.Error as e:
 973                         failure_response['failures'].append(" ".join((environment, to_native(e))))
 974 
 975                 if module_specs and not self.with_modules:
 976                     # This means that the group or env wasn't found in comps
 977                     self.module.fail_json(
 978                         msg="No group {0} available.".format(module_specs[0]),
 979                         results=[],
 980                     )
 981 
 982                 # Install packages.
 983                 if self.update_only:
 984                     not_installed = self._update_only(pkg_specs)
 985                     for spec in not_installed:
 986                         response['results'].append("Packages providing %s not installed due to update_only specified" % spec)
 987                 else:
 988                     for pkg_spec in pkg_specs:
 989                         install_result = self._mark_package_install(pkg_spec)
 990                         if install_result['failed']:
 991                             if install_result['msg']:
 992                                 failure_response['msg'] += install_result['msg']
 993                             failure_response['failures'].append(self._sanitize_dnf_error_msg_install(pkg_spec, install_result['failure']))
 994                         else:
 995                             if install_result['msg']:
 996                                 response['results'].append(install_result['msg'])
 997 
 998             elif self.state == 'latest':
 999                 # "latest" is same as "installed" for filenames.
1000                 self._install_remote_rpms(filenames)
1001                 for filename in filenames:
1002                     response['results'].append("Installed {0}".format(filename))
1003 
1004                 # Upgrade modules
1005                 if module_specs and self.with_modules:
1006                     for module in module_specs:
1007                         try:
1008                             if self._is_module_installed(module):
1009                                 response['results'].append("Module {0} upgraded.".format(module))
1010                             self.module_base.upgrade([module])
1011                         except dnf.exceptions.MarkingErrors as e:
1012                             failure_response['failures'].append(' '.join((module, to_native(e))))
1013 
1014                 for group in groups:
1015                     try:
1016                         try:
1017                             self.base.group_upgrade(group)
1018                             response['results'].append("Group {0} upgraded.".format(group))
1019                         except dnf.exceptions.CompsError:
1020                             if not self.update_only:
1021                                 # If not already installed, try to install.
1022                                 group_pkg_count_installed = self.base.group_install(group, dnf.const.GROUP_PACKAGE_TYPES)
1023                                 if group_pkg_count_installed == 0:
1024                                     response['results'].append("Group {0} already installed.".format(group))
1025                                 else:
1026                                     response['results'].append("Group {0} installed.".format(group))
1027                     except dnf.exceptions.Error as e:
1028                         failure_response['failures'].append(" ".join((group, to_native(e))))
1029 
1030                 for environment in environments:
1031                     try:
1032                         try:
1033                             self.base.environment_upgrade(environment)
1034                         except dnf.exceptions.CompsError:
1035                             # If not already installed, try to install.
1036                             self.base.environment_install(environment, dnf.const.GROUP_PACKAGE_TYPES)
1037                     except dnf.exceptions.DepsolveError as e:
1038                         failure_response['msg'] = "Depsolve Error occured attempting to install environment: {0}".format(environment)
1039                     except dnf.exceptions.Error as e:
1040                         failure_response['failures'].append(" ".join((environment, to_native(e))))
1041 
1042                 if self.update_only:
1043                     not_installed = self._update_only(pkg_specs)
1044                     for spec in not_installed:
1045                         response['results'].append("Packages providing %s not installed due to update_only specified" % spec)
1046                 else:
1047                     for pkg_spec in pkg_specs:
1048                         # best effort causes to install the latest package
1049                         # even if not previously installed
1050                         self.base.conf.best = True
1051                         install_result = self._mark_package_install(pkg_spec, upgrade=True)
1052                         if install_result['failed']:
1053                             if install_result['msg']:
1054                                 failure_response['msg'] += install_result['msg']
1055                             failure_response['failures'].append(self._sanitize_dnf_error_msg_install(pkg_spec, install_result['failure']))
1056                         else:
1057                             if install_result['msg']:
1058                                 response['results'].append(install_result['msg'])
1059 
1060             else:
1061                 # state == absent
1062                 if filenames:
1063                     self.module.fail_json(
1064                         msg="Cannot remove paths -- please specify package name.",
1065                         results=[],
1066                     )
1067 
1068                 # Remove modules
1069                 if module_specs and self.with_modules:
1070                     for module in module_specs:
1071                         try:
1072                             if self._is_module_installed(module):
1073                                 response['results'].append("Module {0} removed.".format(module))
1074                             self.module_base.remove([module])
1075                             self.module_base.disable([module])
1076                             self.module_base.reset([module])
1077                         except dnf.exceptions.MarkingErrors as e:
1078                             failure_response['failures'].append(' '.join((module, to_native(e))))
1079 
1080                 for group in groups:
1081                     try:
1082                         self.base.group_remove(group)
1083                     except dnf.exceptions.CompsError:
1084                         # Group is already uninstalled.
1085                         pass
1086                     except AttributeError:
1087                         # Group either isn't installed or wasn't marked installed at install time
1088                         # because of DNF bug
1089                         #
1090                         # This is necessary until the upstream dnf API bug is fixed where installing
1091                         # a group via the dnf API doesn't actually mark the group as installed
1092                         #   https://bugzilla.redhat.com/show_bug.cgi?id=1620324
1093                         pass
1094 
1095                 for environment in environments:
1096                     try:
1097                         self.base.environment_remove(environment)
1098                     except dnf.exceptions.CompsError:
1099                         # Environment is already uninstalled.
1100                         pass
1101 
1102                 installed = self.base.sack.query().installed()
1103                 for pkg_spec in pkg_specs:
1104                     # short-circuit installed check for wildcard matching
1105                     if '*' in pkg_spec:
1106                         try:
1107                             self.base.remove(pkg_spec)
1108                         except dnf.exceptions.MarkingError as e:
1109                             is_failure, handled_remove_error = self._sanitize_dnf_error_msg_remove(pkg_spec, to_native(e))
1110                             if is_failure:
1111                                 failure_response['failures'].append('{0} - {1}'.format(pkg_spec, to_native(e)))
1112                             else:
1113                                 response['results'].append(handled_remove_error)
1114                         continue
1115 
1116                     installed_pkg = list(map(str, installed.filter(name=pkg_spec).run()))
1117                     if installed_pkg:
1118                         candidate_pkg = self._packagename_dict(installed_pkg[0])
1119                         installed_pkg = installed.filter(name=candidate_pkg['name']).run()
1120                     else:
1121                         candidate_pkg = self._packagename_dict(pkg_spec)
1122                         installed_pkg = installed.filter(nevra=pkg_spec).run()
1123                     if installed_pkg:
1124                         installed_pkg = installed_pkg[0]
1125                         evr_cmp = self._compare_evr(
1126                             installed_pkg.epoch, installed_pkg.version, installed_pkg.release,
1127                             candidate_pkg['epoch'], candidate_pkg['version'], candidate_pkg['release'],
1128                         )
1129                         if evr_cmp == 0:
1130                             self.base.remove(pkg_spec)
1131 
1132                 # Like the dnf CLI we want to allow recursive removal of dependent
1133                 # packages
1134                 self.allowerasing = True
1135 
1136                 if self.autoremove:
1137                     self.base.autoremove()
1138 
1139         try:
1140             if not self.base.resolve(allow_erasing=self.allowerasing):
1141                 if failure_response['failures']:
1142                     failure_response['msg'] = 'Failed to install some of the specified packages'
1143                     self.module.fail_json(**failure_response)
1144 
1145                 response['msg'] = "Nothing to do"
1146                 self.module.exit_json(**response)
1147             else:
1148                 response['changed'] = True
1149                 if failure_response['failures']:
1150                     failure_response['msg'] = 'Failed to install some of the specified packages'
1151                     self.module.fail_json(**failure_response)
1152                 if self.module.check_mode:
1153                     response['msg'] = "Check mode: No changes made, but would have if not in check mode"
1154                     self.module.exit_json(**response)
1155 
1156                 try:
1157                     if self.download_only and self.download_dir and self.base.conf.destdir:
1158                         dnf.util.ensure_dir(self.base.conf.destdir)
1159                         self.base.repos.all().pkgdir = self.base.conf.destdir
1160 
1161                     self.base.download_packages(self.base.transaction.install_set)
1162                 except dnf.exceptions.DownloadError as e:
1163                     self.module.fail_json(
1164                         msg="Failed to download packages: {0}".format(to_text(e)),
1165                         results=[],
1166                     )
1167 
1168                 if self.download_only:
1169                     for package in self.base.transaction.install_set:
1170                         response['results'].append("Downloaded: {0}".format(package))
1171                     self.module.exit_json(**response)
1172                 else:
1173                     self.base.do_transaction()
1174                     for package in self.base.transaction.install_set:
1175                         response['results'].append("Installed: {0}".format(package))
1176                     for package in self.base.transaction.remove_set:
1177                         response['results'].append("Removed: {0}".format(package))
1178 
1179                 if failure_response['failures']:
1180                     failure_response['msg'] = 'Failed to install some of the specified packages'
1181                     self.module.exit_json(**response)
1182                 self.module.exit_json(**response)
1183         except dnf.exceptions.DepsolveError as e:
1184             failure_response['msg'] = "Depsolve Error occured: {0}".format(to_native(e))
1185             self.module.fail_json(**failure_response)
1186         except dnf.exceptions.Error as e:
1187             if to_text("already installed") in to_text(e):
1188                 response['changed'] = False
1189                 response['results'].append("Package already installed: {0}".format(to_native(e)))
1190                 self.module.exit_json(**response)
1191             else:
1192                 failure_response['msg'] = "Unknown Error occured: {0}".format(to_native(e))
1193                 self.module.fail_json(**failure_response)
