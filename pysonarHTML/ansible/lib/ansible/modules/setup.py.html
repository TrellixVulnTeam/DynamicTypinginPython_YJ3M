 131 from ..module_utils.basic import AnsibleModule
 132 
 133 from ansible.module_utils.facts.namespace import PrefixFactNamespace
 134 from ansible.module_utils.facts import ansible_collector
 135 
 136 from ansible.module_utils.facts import default_collectors
 137 
 138 
 139 def main():
 140     module = AnsibleModule(
 141         argument_spec=dict(
 142             gather_subset=dict(default=["all"], required=False, type='list'),
 143             gather_timeout=dict(default=10, required=False, type='int'),
 144             filter=dict(default="*", required=False),
 145             fact_path=dict(default='/etc/ansible/facts.d', required=False, type='path'),
 146         ),
 147         supports_check_mode=True,
 148     )
 149 
 150     gather_subset = module.params['gather_subset']
 151     gather_timeout = module.params['gather_timeout']
 152     filter_spec = module.params['filter']
 153 
 154     # TODO: this mimics existing behavior where gather_subset=["!all"] actually means
 155     #       to collect nothing except for the below list
 156     # TODO: decide what '!all' means, I lean towards making it mean none, but likely needs
 157     #       some tweaking on how gather_subset operations are performed
 158     minimal_gather_subset = frozenset(['apparmor', 'caps', 'cmdline', 'date_time',
 159                                        'distribution', 'dns', 'env', 'fips', 'local',
 160                                        'lsb', 'pkg_mgr', 'platform', 'python', 'selinux',
 161                                        'service_mgr', 'ssh_pub_keys', 'user'])
 162 
 163     all_collector_classes = default_collectors.collectors
 164 
 165     # rename namespace_name to root_key?
 166     namespace = PrefixFactNamespace(namespace_name='ansible',
 167                                     prefix='ansible_')
 168 
 169     fact_collector = \
 170         ansible_collector.get_ansible_collector(all_collector_classes=all_collector_classes,
 171                                                 namespace=namespace,
 172                                                 filter_spec=filter_spec,
 173                                                 gather_subset=gather_subset,
 174                                                 gather_timeout=gather_timeout,
 175                                                 minimal_gather_subset=minimal_gather_subset)
 176 
 177     facts_dict = fact_collector.collect(module=module)
 178 
 179     module.exit_json(ansible_facts=facts_dict)
 180 
 181 
 182 if __name__ == '__main__':
 183     main()
