 174 # Simple version of aptsources.sourceslist.SourcesList.
 175 # No advanced logic and no backups inside.
 176 class SourcesList(object):
 177     def __init__(self, module):
 178         self.module = module
 179         self.files = {}  # group sources by file
 180         # Repositories that we're adding -- used to implement mode param
 181         self.new_repos = set()
 182         self.default_file = self._apt_cfg_file('Dir::Etc::sourcelist')
 183 
 184         # read sources.list if it exists
 185         if os.path.isfile(self.default_file):
 186             self.load(self.default_file)
 187 
 188         # read sources.list.d
 189         for file in glob.iglob('%s/*.list' % self._apt_cfg_dir('Dir::Etc::sourceparts')):
 190             self.load(file)
 191 
 192     def __iter__(self):
 193         '''Simple iterator to go over all sources. Empty, non-source, and other not valid lines will be skipped.'''
 194         for file, sources in self.files.items():
 195             for n, valid, enabled, source, comment in sources:
 196                 if valid:
 197                     yield file, n, enabled, source, comment
 198 
 199     def _expand_path(self, filename):
 200         if '/' in filename:
 201             return filename
 202         else:
 203             return os.path.abspath(os.path.join(self._apt_cfg_dir('Dir::Etc::sourceparts'), filename))
 204 
 205     def _suggest_filename(self, line):
 206         def _cleanup_filename(s):
 207             filename = self.module.params['filename']
 208             if filename is not None:
 209                 return filename
 210             return '_'.join(re.sub('[^a-zA-Z0-9]', ' ', s).split())
 211 
 212         def _strip_username_password(s):
 213             if '@' in s:
 214                 s = s.split('@', 1)
 215                 s = s[-1]
 216             return s
 217 
 218         # Drop options and protocols.
 219         line = re.sub(r'\[[^\]]+\]', '', line)
 220         line = re.sub(r'\w+://', '', line)
 221 
 222         # split line into valid keywords
 223         parts = [part for part in line.split() if part not in VALID_SOURCE_TYPES]
 224 
 225         # Drop usernames and passwords
 226         parts[0] = _strip_username_password(parts[0])
 227 
 228         return '%s.list' % _cleanup_filename(' '.join(parts[:1]))
 229 
 230     def _parse(self, line, raise_if_invalid_or_disabled=False):
 231         valid = False
 232         enabled = True
 233         source = ''
 234         comment = ''
 235 
 236         line = line.strip()
 237         if line.startswith('#'):
 238             enabled = False
 239             line = line[1:]
 240 
 241         # Check for another "#" in the line and treat a part after it as a comment.
 242         i = line.find('#')
 243         if i > 0:
 244             comment = line[i + 1:].strip()
 245             line = line[:i]
 246 
 247         # Split a source into substring to make sure that it is source spec.
 248         # Duplicated whitespaces in a valid source spec will be removed.
 249         source = line.strip()
 250         if source:
 251             chunks = source.split()
 252             if chunks[0] in VALID_SOURCE_TYPES:
 253                 valid = True
 254                 source = ' '.join(chunks)
 255 
 256         if raise_if_invalid_or_disabled and (not valid or not enabled):
 257             raise InvalidSource(line)
 258 
 259         return valid, enabled, source, comment
 260 
 261     @staticmethod
 262     def _apt_cfg_file(filespec):
 263         '''
 264         Wrapper for `apt_pkg` module for running with Python 2.5
 265         '''
 266         try:
 267             result = apt_pkg.config.find_file(filespec)
 268         except AttributeError:
 269             result = apt_pkg.Config.FindFile(filespec)
 270         return result
 271 
 272     @staticmethod
 273     def _apt_cfg_dir(dirspec):
 274         '''
 275         Wrapper for `apt_pkg` module for running with Python 2.5
 276         '''
 277         try:
 278             result = apt_pkg.config.find_dir(dirspec)
 279         except AttributeError:
 280             result = apt_pkg.Config.FindDir(dirspec)
 281         return result

 283     def load(self, file):
 284         group = []
 285         f = open(file, 'r')
 286         for n, line in enumerate(f):
 287             valid, enabled, source, comment = self._parse(line)
 288             group.append((n, valid, enabled, source, comment))
 289         self.files[file] = group
 290 
 291     def save(self):
 292         for filename, sources in list(self.files.items()):
 293             if sources:
 294                 d, fn = os.path.split(filename)
 295                 try:
 296                     os.makedirs(d)
 297                 except OSError as err:
 298                     if not os.path.isdir(d):
 299                         self.module.fail_json("Failed to create directory %s: %s" % (d, to_native(err)))
 300                 fd, tmp_path = tempfile.mkstemp(prefix=".%s-" % fn, dir=d)
 301 
 302                 f = os.fdopen(fd, 'w')
 303                 for n, valid, enabled, source, comment in sources:
 304                     chunks = []
 305                     if not enabled:
 306                         chunks.append('# ')
 307                     chunks.append(source)
 308                     if comment:
 309                         chunks.append(' # ')
 310                         chunks.append(comment)
 311                     chunks.append('\n')
 312                     line = ''.join(chunks)
 313 
 314                     try:
 315                         f.write(line)
 316                     except IOError as err:
 317                         self.module.fail_json(msg="Failed to write to file %s: %s" % (tmp_path, to_native(err)))
 318                 self.module.atomic_move(tmp_path, filename)
 319 
 320                 # allow the user to override the default mode
 321                 if filename in self.new_repos:
 322                     this_mode = self.module.params.get('mode', DEFAULT_SOURCES_PERM)
 323                     self.module.set_mode_if_different(filename, this_mode, False)
 324             else:
 325                 del self.files[filename]
 326                 if os.path.exists(filename):
 327                     os.remove(filename)
 328 
 329     def dump(self):
 330         dumpstruct = {}
 331         for filename, sources in self.files.items():
 332             if sources:
 333                 lines = []
 334                 for n, valid, enabled, source, comment in sources:
 335                     chunks = []
 336                     if not enabled:
 337                         chunks.append('# ')
 338                     chunks.append(source)
 339                     if comment:
 340                         chunks.append(' # ')
 341                         chunks.append(comment)
 342                     chunks.append('\n')
 343                     lines.append(''.join(chunks))
 344                 dumpstruct[filename] = ''.join(lines)
 345         return dumpstruct
 346 
 347     def _choice(self, new, old):
 348         if new is None:
 349             return old
 350         return new
 351 
 352     def modify(self, file, n, enabled=None, source=None, comment=None):
 353         '''
 354         This function to be used with iterator, so we don't care of invalid sources.
 355         If source, enabled, or comment is None, original value from line ``n`` will be preserved.
 356         '''
 357         valid, enabled_old, source_old, comment_old = self.files[file][n][1:]
 358         self.files[file][n] = (n, valid, self._choice(enabled, enabled_old), self._choice(source, source_old), self._choice(comment, comment_old))
 359 
 360     def _add_valid_source(self, source_new, comment_new, file):
 361         # We'll try to reuse disabled source if we have it.
 362         # If we have more than one entry, we will enable them all - no advanced logic, remember.
 363         found = False
 364         for filename, n, enabled, source, comment in self:
 365             if source == source_new:
 366                 self.modify(filename, n, enabled=True)
 367                 found = True
 368 
 369         if not found:
 370             if file is None:
 371                 file = self.default_file
 372             else:
 373                 file = self._expand_path(file)
 374 
 375             if file not in self.files:
 376                 self.files[file] = []
 377 
 378             files = self.files[file]
 379             files.append((len(files), True, True, source_new, comment_new))
 380             self.new_repos.add(file)
 381 
 382     def add_source(self, line, comment='', file=None):
 383         source = self._parse(line, raise_if_invalid_or_disabled=True)[2]
 384 
 385         # Prefer separate files for new sources.
 386         self._add_valid_source(source, comment, file=file or self._suggest_filename(source))
 387 
 388     def _remove_valid_source(self, source):
 389         # If we have more than one entry, we will remove them all (not comment, remove!)
 390         for filename, n, enabled, src, comment in self:
 391             if source == src and enabled:
 392                 self.files[filename].pop(n)
 393 
 394     def remove_source(self, line):
 395         source = self._parse(line, raise_if_invalid_or_disabled=True)[2]
 396         self._remove_valid_source(source)

 399 class UbuntuSourcesList(SourcesList):
 400 
 401     LP_API = 'https://launchpad.net/api/1.0/~%s/+archive/%s'
 402 
 403     def __init__(self, module, add_ppa_signing_keys_callback=None):
 404         self.module = module
 405         self.add_ppa_signing_keys_callback = add_ppa_signing_keys_callback
 406         self.codename = module.params['codename'] or distro.codename
 407         super(UbuntuSourcesList, self).__init__(module)
 408 
 409     def __deepcopy__(self, memo=None):
 410         return UbuntuSourcesList(
 411             self.module,
 412             add_ppa_signing_keys_callback=self.add_ppa_signing_keys_callback
 413         )
 414 
 415     def _get_ppa_info(self, owner_name, ppa_name):
 416         lp_api = self.LP_API % (owner_name, ppa_name)
 417 
 418         headers = dict(Accept='application/json')
 419         response, info = fetch_url(self.module, lp_api, headers=headers)
 420         if info['status'] != 200:
 421             self.module.fail_json(msg="failed to fetch PPA information, error was: %s" % info['msg'])
 422         return json.loads(to_native(response.read()))
 423 
 424     def _expand_ppa(self, path):
 425         ppa = path.split(':')[1]
 426         ppa_owner = ppa.split('/')[0]
 427         try:
 428             ppa_name = ppa.split('/')[1]
 429         except IndexError:
 430             ppa_name = 'ppa'
 431 
 432         line = 'deb http://ppa.launchpad.net/%s/%s/ubuntu %s main' % (ppa_owner, ppa_name, self.codename)
 433         return line, ppa_owner, ppa_name
 434 
 435     def _key_already_exists(self, key_fingerprint):
 436         rc, out, err = self.module.run_command('apt-key export %s' % key_fingerprint, check_rc=True)
 437         return len(err) == 0
 438 
 439     def add_source(self, line, comment='', file=None):
 440         if line.startswith('ppa:'):
 441             source, ppa_owner, ppa_name = self._expand_ppa(line)
 442 
 443             if source in self.repos_urls:
 444                 # repository already exists
 445                 return
 446 
 447             if self.add_ppa_signing_keys_callback is not None:
 448                 info = self._get_ppa_info(ppa_owner, ppa_name)
 449                 if not self._key_already_exists(info['signing_key_fingerprint']):
 450                     command = ['apt-key', 'adv', '--recv-keys', '--no-tty', '--keyserver', 'hkp://keyserver.ubuntu.com:80', info['signing_key_fingerprint']]
 451                     self.add_ppa_signing_keys_callback(command)
 452 
 453             file = file or self._suggest_filename('%s_%s' % (line, self.codename))
 454         else:
 455             source = self._parse(line, raise_if_invalid_or_disabled=True)[2]
 456             file = file or self._suggest_filename(source)
 457         self._add_valid_source(source, comment, file)
 458 
 459     def remove_source(self, line):
 460         if line.startswith('ppa:'):
 461             source = self._expand_ppa(line)[0]
 462         else:
 463             source = self._parse(line, raise_if_invalid_or_disabled=True)[2]
 464         self._remove_valid_source(source)
 465 
 466     @property
 467     def repos_urls(self):
 468         _repositories = []
 469         for parsed_repos in self.files.values():
 470             for parsed_repo in parsed_repos:
 471                 valid = parsed_repo[1]
 472                 enabled = parsed_repo[2]
 473                 source_line = parsed_repo[3]
 474 
 475                 if not valid or not enabled:
 476                     continue
 477 
 478                 if source_line.startswith('ppa:'):
 479                     source, ppa_owner, ppa_name = self._expand_ppa(source_line)
 480                     _repositories.append(source)
 481                 else:
 482                     _repositories.append(source_line)
 483 
 484         return _repositories

 489         module.run_command(command, check_rc=True)

 492         return None

 494         return _run_command

 497 def revert_sources_list(sources_before, sources_after, sourceslist_before):
 498     '''Revert the sourcelist files to their previous state.'''
 499 
 500     # First remove any new files that were created:
 501     for filename in set(sources_after.keys()).difference(sources_before.keys()):
 502         if os.path.exists(filename):
 503             os.remove(filename)
 504     # Now revert the existing files to their former state:
 505     sourceslist_before.save()

 508 def main():
 509     module = AnsibleModule(
 510         argument_spec=dict(
 511             repo=dict(type='str', required=True),
 512             state=dict(type='str', default='present', choices=['absent', 'present']),
 513             mode=dict(type='raw'),
 514             update_cache=dict(type='bool', default=True, aliases=['update-cache']),
 515             update_cache_retries=dict(type='int', default=5),
 516             update_cache_retry_max_delay=dict(type='int', default=12),
 517             filename=dict(type='str'),
 518             # This should not be needed, but exists as a failsafe
 519             install_python_apt=dict(type='bool', default=True),
 520             validate_certs=dict(type='bool', default=True),
 521             codename=dict(type='str'),
 522         ),
 523         supports_check_mode=True,
 524     )
 525 
 526     params = module.params
 527     repo = module.params['repo']
 528     state = module.params['state']
 529     update_cache = module.params['update_cache']
 530     # Note: mode is referenced in SourcesList class via the passed in module (self here)
 531 
 532     sourceslist = None
 533 
 534     if not HAVE_PYTHON_APT:
 535         if params['install_python_apt']:
 536             install_python_apt(module)
 537         else:
 538             module.fail_json(msg='%s is not installed, and install_python_apt is False' % PYTHON_APT)
 539 
 540     if not repo:
 541         module.fail_json(msg='Please set argument \'repo\' to a non-empty value')
 542 
 543     if isinstance(distro, aptsources_distro.Distribution):
 544         sourceslist = UbuntuSourcesList(module, add_ppa_signing_keys_callback=get_add_ppa_signing_key_callback(module))
 545     else:
 546         module.fail_json(msg='Module apt_repository is not supported on target.')
 547 
 548     sourceslist_before = copy.deepcopy(sourceslist)
 549     sources_before = sourceslist.dump()
 550 
 551     try:
 552         if state == 'present':
 553             sourceslist.add_source(repo)
 554         elif state == 'absent':
 555             sourceslist.remove_source(repo)
 556     except InvalidSource as err:
 557         module.fail_json(msg='Invalid repository string: %s' % to_native(err))
 558 
 559     sources_after = sourceslist.dump()
 560     changed = sources_before != sources_after
 561 
 562     if changed and module._diff:
 563         diff = []
 564         for filename in set(sources_before.keys()).union(sources_after.keys()):
 565             diff.append({'before': sources_before.get(filename, ''),
 566                          'after': sources_after.get(filename, ''),
 567                          'before_header': (filename, '/dev/null')[filename not in sources_before],
 568                          'after_header': (filename, '/dev/null')[filename not in sources_after]})
 569     else:
 570         diff = {}
 571 
 572     if changed and not module.check_mode:
 573         try:
 574             sourceslist.save()
 575             if update_cache:
 576                 err = ''
 577                 update_cache_retries = module.params.get('update_cache_retries')
 578                 update_cache_retry_max_delay = module.params.get('update_cache_retry_max_delay')
 579                 randomize = random.randint(0, 1000) / 1000.0
 580 
 581                 for retry in range(update_cache_retries):
 582                     try:
 583                         cache = apt.Cache()
 584                         cache.update()
 585                         break
 586                     except apt.cache.FetchFailedException as e:
 587                         err = to_native(e)
 588 
 589                     # Use exponential backoff with a max fail count, plus a little bit of randomness
 590                     delay = 2 ** retry + randomize
 591                     if delay > update_cache_retry_max_delay:
 592                         delay = update_cache_retry_max_delay + randomize
 593                     time.sleep(delay)
 594                 else:
 595                     revert_sources_list(sources_before, sources_after, sourceslist_before)
 596                     module.fail_json(msg='Failed to update apt cache: %s' % (err if err else 'unknown reason'))
 597 
 598         except (OSError, IOError) as err:
 599             revert_sources_list(sources_before, sources_after, sourceslist_before)
 600             module.fail_json(msg=to_native(err))
 601 
 602     module.exit_json(changed=changed, repo=repo, state=state, diff=diff)
 603 
 604 
 605 if __name__ == '__main__':
 606     main()
