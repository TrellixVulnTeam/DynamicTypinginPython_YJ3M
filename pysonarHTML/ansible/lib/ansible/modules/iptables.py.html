 484 def append_param(rule, param, flag, is_list):
 485     if is_list:
 486         for item in param:
 487             append_param(rule, item, flag, False)
 488     else:
 489         if param is not None:
 490             if param[0] == '!':
 491                 rule.extend(['!', flag, param[1:]])
 492             else:
 493                 rule.extend([flag, param])

 496 def append_tcp_flags(rule, param, flag):
 497     if param:
 498         if 'flags' in param and 'flags_set' in param:
 499             rule.extend([flag, ','.join(param['flags']), ','.join(param['flags_set'])])

 502 def append_match_flag(rule, param, flag, negatable):
 503     if param == 'match':
 504         rule.extend([flag])
 505     elif negatable and param == 'negate':
 506         rule.extend(['!', flag])

 529 def construct_rule(params):
 530     rule = []
 531     append_wait(rule, params['wait'], '-w')
 532     append_param(rule, params['protocol'], '-p', False)
 533     append_param(rule, params['source'], '-s', False)
 534     append_param(rule, params['destination'], '-d', False)
 535     append_param(rule, params['match'], '-m', True)
 536     append_tcp_flags(rule, params['tcp_flags'], '--tcp-flags')
 537     append_param(rule, params['jump'], '-j', False)
 538     if params.get('jump') and params['jump'].lower() == 'tee':
 539         append_param(rule, params['gateway'], '--gateway', False)
 540     append_param(rule, params['log_prefix'], '--log-prefix', False)
 541     append_param(rule, params['log_level'], '--log-level', False)
 542     append_param(rule, params['to_destination'], '--to-destination', False)
 543     append_param(rule, params['to_source'], '--to-source', False)
 544     append_param(rule, params['goto'], '-g', False)
 545     append_param(rule, params['in_interface'], '-i', False)
 546     append_param(rule, params['out_interface'], '-o', False)
 547     append_param(rule, params['fragment'], '-f', False)
 548     append_param(rule, params['set_counters'], '-c', False)
 549     append_param(rule, params['source_port'], '--source-port', False)
 550     append_param(rule, params['destination_port'], '--destination-port', False)
 551     append_param(rule, params['to_ports'], '--to-ports', False)
 552     append_param(rule, params['set_dscp_mark'], '--set-dscp', False)
 553     append_param(
 554         rule,
 555         params['set_dscp_mark_class'],
 556         '--set-dscp-class',
 557         False)
 558     append_match_flag(rule, params['syn'], '--syn', True)
 559     append_match(rule, params['comment'], 'comment')
 560     append_param(rule, params['comment'], '--comment', False)
 561     if 'conntrack' in params['match']:
 562         append_csv(rule, params['ctstate'], '--ctstate')
 563     elif 'state' in params['match']:
 564         append_csv(rule, params['ctstate'], '--state')
 565     elif params['ctstate']:
 566         append_match(rule, params['ctstate'], 'conntrack')
 567         append_csv(rule, params['ctstate'], '--ctstate')
 568     if 'iprange' in params['match']:
 569         append_param(rule, params['src_range'], '--src-range', False)
 570         append_param(rule, params['dst_range'], '--dst-range', False)
 571     elif params['src_range'] or params['dst_range']:
 572         append_match(rule, params['src_range'] or params['dst_range'], 'iprange')
 573         append_param(rule, params['src_range'], '--src-range', False)
 574         append_param(rule, params['dst_range'], '--dst-range', False)
 575     append_match(rule, params['limit'] or params['limit_burst'], 'limit')
 576     append_param(rule, params['limit'], '--limit', False)
 577     append_param(rule, params['limit_burst'], '--limit-burst', False)
 578     append_match(rule, params['uid_owner'], 'owner')
 579     append_match_flag(rule, params['uid_owner'], '--uid-owner', True)
 580     append_param(rule, params['uid_owner'], '--uid-owner', False)
 581     append_match(rule, params['gid_owner'], 'owner')
 582     append_match_flag(rule, params['gid_owner'], '--gid-owner', True)
 583     append_param(rule, params['gid_owner'], '--gid-owner', False)
 584     if params['jump'] is None:
 585         append_jump(rule, params['reject_with'], 'REJECT')
 586     append_param(rule, params['reject_with'], '--reject-with', False)
 587     append_param(
 588         rule,
 589         params['icmp_type'],
 590         ICMP_TYPE_OPTIONS[params['ip_version']],
 591         False)
 592     return rule

 595 def push_arguments(iptables_path, action, params, make_rule=True):
 596     cmd = [iptables_path]
 597     cmd.extend(['-t', params['table']])
 598     cmd.extend([action, params['chain']])
 599     if action == '-I' and params['rule_num']:
 600         cmd.extend([params['rule_num']])
 601     if make_rule:
 602         cmd.extend(construct_rule(params))
 603     return cmd

 606 def check_present(iptables_path, module, params):
 607     cmd = push_arguments(iptables_path, '-C', params)
 608     rc, _, __ = module.run_command(cmd, check_rc=False)
 609     return (rc == 0)

 638 def get_chain_policy(iptables_path, module, params):
 639     cmd = push_arguments(iptables_path, '-L', params)
 640     rc, out, _ = module.run_command(cmd, check_rc=True)
 641     chain_header = out.split("\n")[0]
 642     result = re.search(r'\(policy ([A-Z]+)\)', chain_header)
 643     if result:
 644         return result.group(1)
 645     return None
