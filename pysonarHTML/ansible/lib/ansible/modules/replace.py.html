 176 def write_changes(module, contents, path):
 177 
 178     tmpfd, tmpfile = tempfile.mkstemp(dir=module.tmpdir)
 179     f = os.fdopen(tmpfd, 'wb')
 180     f.write(contents)
 181     f.close()
 182 
 183     validate = module.params.get('validate', None)
 184     valid = not validate
 185     if validate:
 186         if "%s" not in validate:
 187             module.fail_json(msg="validate must contain %%s: %s" % (validate))
 188         (rc, out, err) = module.run_command(validate % tmpfile)
 189         valid = rc == 0
 190         if rc != 0:
 191             module.fail_json(msg='failed to validate: '
 192                                  'rc:%s error:%s' % (rc, err))
 193     if valid:
 194         module.atomic_move(tmpfile, path, unsafe_writes=module.params['unsafe_writes'])

 197 def check_file_attrs(module, changed, message):
 198 
 199     file_args = module.load_file_common_arguments(module.params)
 200     if module.set_file_attributes_if_different(file_args, False):
 201 
 202         if changed:
 203             message += " and "
 204         changed = True
 205         message += "ownership, perms or SE linux context changed"
 206 
 207     return message, changed

 210 def main():
 211     module = AnsibleModule(
 212         argument_spec=dict(
 213             path=dict(type='path', required=True, aliases=['dest', 'destfile', 'name']),
 214             regexp=dict(type='str', required=True),
 215             replace=dict(type='str', default=''),
 216             after=dict(type='str'),
 217             before=dict(type='str'),
 218             backup=dict(type='bool', default=False),
 219             validate=dict(type='str'),
 220             encoding=dict(type='str', default='utf-8'),
 221         ),
 222         add_file_common_args=True,
 223         supports_check_mode=True,
 224     )
 225 
 226     params = module.params
 227     path = params['path']
 228     encoding = params['encoding']
 229     res_args = dict()
 230 
 231     params['after'] = to_text(params['after'], errors='surrogate_or_strict', nonstring='passthru')
 232     params['before'] = to_text(params['before'], errors='surrogate_or_strict', nonstring='passthru')
 233     params['regexp'] = to_text(params['regexp'], errors='surrogate_or_strict', nonstring='passthru')
 234     params['replace'] = to_text(params['replace'], errors='surrogate_or_strict', nonstring='passthru')
 235 
 236     if os.path.isdir(path):
 237         module.fail_json(rc=256, msg='Path %s is a directory !' % path)
 238 
 239     if not os.path.exists(path):
 240         module.fail_json(rc=257, msg='Path %s does not exist !' % path)
 241     else:
 242         f = open(path, 'rb')
 243         contents = to_text(f.read(), errors='surrogate_or_strict', encoding=encoding)
 244         f.close()
 245 
 246     pattern = u''
 247     if params['after'] and params['before']:
 248         pattern = u'%s(?P<subsection>.*?)%s' % (params['after'], params['before'])
 249     elif params['after']:
 250         pattern = u'%s(?P<subsection>.*)' % params['after']
 251     elif params['before']:
 252         pattern = u'(?P<subsection>.*)%s' % params['before']
 253 
 254     if pattern:
 255         section_re = re.compile(pattern, re.DOTALL)
 256         match = re.search(section_re, contents)
 257         if match:
 258             section = match.group('subsection')
 259             indices = [match.start('subsection'), match.end('subsection')]
 260         else:
 261             res_args['msg'] = 'Pattern for before/after params did not match the given file: %s' % pattern
 262             res_args['changed'] = False
 263             module.exit_json(**res_args)
 264     else:
 265         section = contents
 266 
 267     mre = re.compile(params['regexp'], re.MULTILINE)
 268     result = re.subn(mre, params['replace'], section, 0)
 269 
 270     if result[1] > 0 and section != result[0]:
 271         if pattern:
 272             result = (contents[:indices[0]] + result[0] + contents[indices[1]:], result[1])
 273         msg = '%s replacements made' % result[1]
 274         changed = True
 275         if module._diff:
 276             res_args['diff'] = {
 277                 'before_header': path,
 278                 'before': contents,
 279                 'after_header': path,
 280                 'after': result[0],
 281             }
 282     else:
 283         msg = ''
 284         changed = False
 285 
 286     if changed and not module.check_mode:
 287         if params['backup'] and os.path.exists(path):
 288             res_args['backup_file'] = module.backup_local(path)
 289         # We should always follow symlinks so that we change the real file
 290         path = os.path.realpath(path)
 291         write_changes(module, to_bytes(result[0], encoding=encoding), path)
 292 
 293     res_args['msg'], res_args['changed'] = check_file_attrs(module, changed, msg)
 294     module.exit_json(**res_args)
 295 
 296 
 297 if __name__ == '__main__':
 298     main()
