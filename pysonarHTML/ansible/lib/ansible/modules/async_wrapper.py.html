  38 def daemonize_self():
  39     # daemonizing code: http://aspn.activestate.com/ASPN/Cookbook/Python/Recipe/66012
  40     try:
  41         pid = os.fork()
  42         if pid > 0:
  43             # exit first parent
  44             sys.exit(0)
  45     except OSError:
  46         e = sys.exc_info()[1]
  47         sys.exit("fork #1 failed: %d (%s)\n" % (e.errno, e.strerror))
  48 
  49     # decouple from parent environment (does not chdir / to keep the directory context the same as for non async tasks)
  50     os.setsid()
  51     os.umask(int('022', 8))
  52 
  53     # do second fork
  54     try:
  55         pid = os.fork()
  56         if pid > 0:
  57             # print "Daemon PID %d" % pid
  58             sys.exit(0)
  59     except OSError:
  60         e = sys.exc_info()[1]
  61         sys.exit("fork #2 failed: %d (%s)\n" % (e.errno, e.strerror))
  62 
  63     dev_null = open('/dev/null', 'w')
  64     os.dup2(dev_null.fileno(), sys.stdin.fileno())
  65     os.dup2(dev_null.fileno(), sys.stdout.fileno())
  66     os.dup2(dev_null.fileno(), sys.stderr.fileno())

  69 # NB: this function copied from module_utils/json_utils.py. Ensure any changes are propagated there.
  70 # FUTURE: AnsibleModule-ify this module so it's Ansiballz-compatible and can use the module_utils copy of this function.
  71 def _filter_non_json_lines(data):
  72     '''
  73     Used to filter unrelated output around module JSON output, like messages from
  74     tcagetattr, or where dropbear spews MOTD on every single command (which is nuts).
  75 
  76     Filters leading lines before first line-starting occurrence of '{' or '[', and filter all
  77     trailing lines after matching close character (working from the bottom of output).
  78     '''
  79     warnings = []
  80 
  81     # Filter initial junk
  82     lines = data.splitlines()
  83 
  84     for start, line in enumerate(lines):
  85         line = line.strip()
  86         if line.startswith(u'{'):
  87             endchar = u'}'
  88             break
  89         elif line.startswith(u'['):
  90             endchar = u']'
  91             break
  92     else:
  93         raise ValueError('No start of json char found')
  94 
  95     # Filter trailing junk
  96     lines = lines[start:]
  97 
  98     for reverse_end_offset, line in enumerate(reversed(lines)):
  99         if line.strip().endswith(endchar):
 100             break
 101     else:
 102         raise ValueError('No end of json char found')
 103 
 104     if reverse_end_offset > 0:
 105         # Trailing junk is uncommon and can point to things the user might
 106         # want to change.  So print a warning if we find any
 107         trailing_junk = lines[len(lines) - reverse_end_offset:]
 108         warnings.append('Module invocation had junk after the JSON data: %s' % '\n'.join(trailing_junk))
 109 
 110     lines = lines[:(len(lines) - reverse_end_offset)]
 111 
 112     return ('\n'.join(lines), warnings)

 115 def _get_interpreter(module_path):
 116     module_fd = open(module_path, 'rb')
 117     try:
 118         head = module_fd.read(1024)
 119         if head[0:2] != '#!':
 120             return None
 121         return head[2:head.index('\n')].strip().split(' ')
 122     finally:
 123         module_fd.close()

 126 def _run_module(wrapped_cmd, jid, job_path):
 127 
 128     tmp_job_path = job_path + ".tmp"
 129     jobfile = open(tmp_job_path, "w")
 130     jobfile.write(json.dumps({"started": 1, "finished": 0, "ansible_job_id": jid}))
 131     jobfile.close()
 132     os.rename(tmp_job_path, job_path)
 133     jobfile = open(tmp_job_path, "w")
 134     result = {}
 135 
 136     # signal grandchild process started and isolated from being terminated
 137     # by the connection being closed sending a signal to the job group
 138     ipc_notifier.send(True)
 139     ipc_notifier.close()
 140 
 141     outdata = ''
 142     filtered_outdata = ''
 143     stderr = ''
 144     try:
 145         cmd = shlex.split(wrapped_cmd)
 146         # call the module interpreter directly (for non-binary modules)
 147         # this permits use of a script for an interpreter on non-Linux platforms
 148         interpreter = _get_interpreter(cmd[0])
 149         if interpreter:
 150             cmd = interpreter + cmd
 151         script = subprocess.Popen(cmd, shell=False, stdin=subprocess.PIPE, stdout=subprocess.PIPE,
 152                                   stderr=subprocess.PIPE)
 153 
 154         (outdata, stderr) = script.communicate()
 155         if PY3:
 156             outdata = outdata.decode('utf-8', 'surrogateescape')
 157             stderr = stderr.decode('utf-8', 'surrogateescape')
 158 
 159         (filtered_outdata, json_warnings) = _filter_non_json_lines(outdata)
 160 
 161         result = json.loads(filtered_outdata)
 162 
 163         if json_warnings:
 164             # merge JSON junk warnings with any existing module warnings
 165             module_warnings = result.get('warnings', [])
 166             if not isinstance(module_warnings, list):
 167                 module_warnings = [module_warnings]
 168             module_warnings.extend(json_warnings)
 169             result['warnings'] = module_warnings
 170 
 171         if stderr:
 172             result['stderr'] = stderr
 173         jobfile.write(json.dumps(result))
 174 
 175     except (OSError, IOError):
 176         e = sys.exc_info()[1]
 177         result = {
 178             "failed": 1,
 179             "cmd": wrapped_cmd,
 180             "msg": to_text(e),
 181             "outdata": outdata,  # temporary notice only
 182             "stderr": stderr
 183         }
 184         result['ansible_job_id'] = jid
 185         jobfile.write(json.dumps(result))
 186 
 187     except (ValueError, Exception):
 188         result = {
 189             "failed": 1,
 190             "cmd": wrapped_cmd,
 191             "data": outdata,  # temporary notice only
 192             "stderr": stderr,
 193             "msg": traceback.format_exc()
 194         }
 195         result['ansible_job_id'] = jid
 196         jobfile.write(json.dumps(result))
 197 
 198     jobfile.close()
 199     os.rename(tmp_job_path, job_path)

 202 def main():
 203     if len(sys.argv) < 5:
 204         print(json.dumps({
 205             "failed": True,
 206             "msg": "usage: async_wrapper <jid> <time_limit> <modulescript> <argsfile> [-preserve_tmp]  "
 207                    "Humans, do not call directly!"
 208         }))
 209         sys.exit(1)
 210 
 211     jid = "%s.%d" % (sys.argv[1], os.getpid())
 212     time_limit = sys.argv[2]
 213     wrapped_module = sys.argv[3]
 214     argsfile = sys.argv[4]
 215     if '-tmp-' not in os.path.dirname(wrapped_module):
 216         preserve_tmp = True
 217     elif len(sys.argv) > 5:
 218         preserve_tmp = sys.argv[5] == '-preserve_tmp'
 219     else:
 220         preserve_tmp = False
 221     # consider underscore as no argsfile so we can support passing of additional positional parameters
 222     if argsfile != '_':
 223         cmd = "%s %s" % (wrapped_module, argsfile)
 224     else:
 225         cmd = wrapped_module
 226     step = 5
 227 
 228     async_dir = os.environ.get('ANSIBLE_ASYNC_DIR', '~/.ansible_async')
 229 
 230     # setup job output directory
 231     jobdir = os.path.expanduser(async_dir)
 232     job_path = os.path.join(jobdir, jid)
 233 
 234     if not os.path.exists(jobdir):
 235         try:
 236             os.makedirs(jobdir)
 237         except Exception:
 238             print(json.dumps({
 239                 "failed": 1,
 240                 "msg": "could not create: %s" % jobdir
 241             }))
 242     # immediately exit this process, leaving an orphaned process
 243     # running which immediately forks a supervisory timing process
 244 
 245     try:
 246         pid = os.fork()
 247         if pid:
 248             # Notify the overlord that the async process started
 249 
 250             # we need to not return immediately such that the launched command has an attempt
 251             # to initialize PRIOR to ansible trying to clean up the launch directory (and argsfile)
 252             # this probably could be done with some IPC later.  Modules should always read
 253             # the argsfile at the very first start of their execution anyway
 254 
 255             # close off notifier handle in grandparent, probably unnecessary as
 256             # this process doesn't hang around long enough
 257             ipc_notifier.close()
 258 
 259             # allow waiting up to 2.5 seconds in total should be long enough for worst
 260             # loaded environment in practice.
 261             retries = 25
 262             while retries > 0:
 263                 if ipc_watcher.poll(0.1):
 264                     break
 265                 else:
 266                     retries = retries - 1
 267                     continue
 268 
 269             notice("Return async_wrapper task started.")
 270             print(json.dumps({"started": 1, "finished": 0, "ansible_job_id": jid, "results_file": job_path,
 271                               "_ansible_suppress_tmpdir_delete": not preserve_tmp}))
 272             sys.stdout.flush()
 273             sys.exit(0)
 274         else:
 275             # The actual wrapper process
 276 
 277             # close off the receiving end of the pipe from child process
 278             ipc_watcher.close()
 279 
 280             # Daemonize, so we keep on running
 281             daemonize_self()
 282 
 283             # we are now daemonized, create a supervisory process
 284             notice("Starting module and watcher")
 285 
 286             sub_pid = os.fork()
 287             if sub_pid:
 288                 # close off inherited pipe handles
 289                 ipc_watcher.close()
 290                 ipc_notifier.close()
 291 
 292                 # the parent stops the process after the time limit
 293                 remaining = int(time_limit)
 294 
 295                 # set the child process group id to kill all children
 296                 os.setpgid(sub_pid, sub_pid)
 297 
 298                 notice("Start watching %s (%s)" % (sub_pid, remaining))
 299                 time.sleep(step)
 300                 while os.waitpid(sub_pid, os.WNOHANG) == (0, 0):
 301                     notice("%s still running (%s)" % (sub_pid, remaining))
 302                     time.sleep(step)
 303                     remaining = remaining - step
 304                     if remaining <= 0:
 305                         notice("Now killing %s" % (sub_pid))
 306                         os.killpg(sub_pid, signal.SIGKILL)
 307                         notice("Sent kill to group %s " % sub_pid)
 308                         time.sleep(1)
 309                         if not preserve_tmp:
 310                             shutil.rmtree(os.path.dirname(wrapped_module), True)
 311                         sys.exit(0)
 312                 notice("Done in kid B.")
 313                 if not preserve_tmp:
 314                     shutil.rmtree(os.path.dirname(wrapped_module), True)
 315                 sys.exit(0)
 316             else:
 317                 # the child process runs the actual module
 318                 notice("Start module (%s)" % os.getpid())
 319                 _run_module(cmd, jid, job_path)
 320                 notice("Module complete (%s)" % os.getpid())
 321                 sys.exit(0)
 322 
 323     except SystemExit:
 324         # On python2.4, SystemExit is a subclass of Exception.
 325         # This block makes python2.4 behave the same as python2.5+
 326         raise
 327 
 328     except Exception:
 329         e = sys.exc_info()[1]
 330         notice("error: %s" % e)
 331         print(json.dumps({
 332             "failed": True,
 333             "msg": "FATAL ERROR: %s" % e
 334         }))
 335         sys.exit(1)
 336 
 337 
 338 if __name__ == '__main__':
 339     main()
