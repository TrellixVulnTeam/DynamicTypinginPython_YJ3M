  15 module: assemble
  16 short_description: Assemble configuration files from fragments
  17 description:
  18 - Assembles a configuration file from fragments.
  19 - Often a particular program will take a single configuration file and does not support a
  20   C(conf.d) style structure where it is easy to build up the configuration
  21   from multiple sources. C(assemble) will take a directory of files that can be

  36     type: path
  37     required: true

  39     description:
  40     - Create a backup file (if C(yes)), including the timestamp information so
  41       you can get the original file back if you somehow clobbered it
  42       incorrectly.
  43     type: bool

  47     - A delimiter to separate the file contents.
  48     type: str
  49     version_added: '1.4'
  50   remote_src:
  51     description:

 117 def assemble_from_fragments(src_path, delimiter=None, compiled_regexp=None, ignore_hidden=False, tmpdir=None):
 118     ''' assemble a file from a directory of fragments '''
 119     tmpfd, temp_path = tempfile.mkstemp(dir=tmpdir)
 120     tmp = os.fdopen(tmpfd, 'wb')
 121     delimit_me = False
 122     add_newline = False
 123 
 124     for f in sorted(os.listdir(src_path)):
 125         if compiled_regexp and not compiled_regexp.search(f):
 126             continue
 127         fragment = os.path.join(src_path, f)
 128         if not os.path.isfile(fragment) or (ignore_hidden and os.path.basename(fragment).startswith('.')):
 129             continue
 130         with open(fragment, 'rb') as fragment_fh:
 131             fragment_content = fragment_fh.read()
 132 
 133         # always put a newline between fragments if the previous fragment didn't end with a newline.
 134         if add_newline:
 135             tmp.write(b('\n'))
 136 
 137         # delimiters should only appear between fragments
 138         if delimit_me:
 139             if delimiter:
 140                 # un-escape anything like newlines
 141                 delimiter = codecs.escape_decode(delimiter)[0]
 142                 tmp.write(delimiter)
 143                 # always make sure there's a newline after the
 144                 # delimiter, so lines don't run together
 145 
 146                 # byte indexing differs on Python 2 and 3,
 147                 # use indexbytes for compat
 148                 # chr(10) == '\n'
 149                 if indexbytes(delimiter, -1) != 10:
 150                     tmp.write(b('\n'))
 151 
 152         tmp.write(fragment_content)
 153         delimit_me = True
 154         if fragment_content.endswith(b('\n')):
 155             add_newline = False
 156         else:
 157             add_newline = True
 158 
 159     tmp.close()
 160     return temp_path

 163 def cleanup(path, result=None):
 164     # cleanup just in case
 165     if os.path.exists(path):
 166         try:
 167             os.remove(path)
 168         except (IOError, OSError) as e:
 169             # don't error on possible race conditions, but keep warning
 170             if result is not None:
 171                 result['warnings'] = ['Unable to remove temp file (%s): %s' % (path, to_native(e))]

 174 def main():
 175 
 176     module = AnsibleModule(
 177         # not checking because of daisy chain to file module
 178         argument_spec=dict(
 179             src=dict(type='path', required=True),
 180             delimiter=dict(type='str'),
 181             dest=dict(type='path', required=True),
 182             backup=dict(type='bool', default=False),
 183             remote_src=dict(type='bool', default=True),
 184             regexp=dict(type='str'),
 185             ignore_hidden=dict(type='bool', default=False),
 186             validate=dict(type='str'),
 187         ),
 188         add_file_common_args=True,
 189     )
 190 
 191     changed = False
 192     path_hash = None
 193     dest_hash = None
 194     src = module.params['src']
 195     dest = module.params['dest']
 196     backup = module.params['backup']
 197     delimiter = module.params['delimiter']
 198     regexp = module.params['regexp']
 199     compiled_regexp = None
 200     ignore_hidden = module.params['ignore_hidden']
 201     validate = module.params.get('validate', None)
 202 
 203     result = dict(src=src, dest=dest)
 204     if not os.path.exists(src):
 205         module.fail_json(msg="Source (%s) does not exist" % src)
 206 
 207     if not os.path.isdir(src):
 208         module.fail_json(msg="Source (%s) is not a directory" % src)
 209 
 210     if regexp is not None:
 211         try:
 212             compiled_regexp = re.compile(regexp)
 213         except re.error as e:
 214             module.fail_json(msg="Invalid Regexp (%s) in \"%s\"" % (to_native(e), regexp))
 215 
 216     if validate and "%s" not in validate:
 217         module.fail_json(msg="validate must contain %%s: %s" % validate)
 218 
 219     path = assemble_from_fragments(src, delimiter, compiled_regexp, ignore_hidden, module.tmpdir)
 220     path_hash = module.sha1(path)
 221     result['checksum'] = path_hash
 222 
 223     # Backwards compat.  This won't return data if FIPS mode is active
 224     try:
 225         pathmd5 = module.md5(path)
 226     except ValueError:
 227         pathmd5 = None
 228     result['md5sum'] = pathmd5
 229 
 230     if os.path.exists(dest):
 231         dest_hash = module.sha1(dest)
 232 
 233     if path_hash != dest_hash:
 234         if validate:
 235             (rc, out, err) = module.run_command(validate % path)
 236             result['validation'] = dict(rc=rc, stdout=out, stderr=err)
 237             if rc != 0:
 238                 cleanup(path)
 239                 module.fail_json(msg="failed to validate: rc:%s error:%s" % (rc, err))
 240         if backup and dest_hash is not None:
 241             result['backup_file'] = module.backup_local(dest)
 242 
 243         module.atomic_move(path, dest, unsafe_writes=module.params['unsafe_writes'])
 244         changed = True
 245 
 246     cleanup(path, result)
 247 
 248     # handle file permissions
 249     file_args = module.load_file_common_arguments(module.params)
 250     result['changed'] = module.set_fs_attributes_if_different(file_args, changed)
 251 
 252     # Mission complete
 253     result['msg'] = "OK"
 254     module.exit_json(**result)
 255 
 256 
 257 if __name__ == '__main__':
 258     main()
