 218 def write_changes(module, b_lines, dest):
 219 
 220     tmpfd, tmpfile = tempfile.mkstemp(dir=module.tmpdir)
 221     with os.fdopen(tmpfd, 'wb') as f:
 222         f.writelines(b_lines)
 223 
 224     validate = module.params.get('validate', None)
 225     valid = not validate
 226     if validate:
 227         if "%s" not in validate:
 228             module.fail_json(msg="validate must contain %%s: %s" % (validate))
 229         (rc, out, err) = module.run_command(to_bytes(validate % tmpfile, errors='surrogate_or_strict'))
 230         valid = rc == 0
 231         if rc != 0:
 232             module.fail_json(msg='failed to validate: '
 233                                  'rc:%s error:%s' % (rc, err))
 234     if valid:
 235         module.atomic_move(tmpfile,
 236                            to_native(os.path.realpath(to_bytes(dest, errors='surrogate_or_strict')), errors='surrogate_or_strict'),
 237                            unsafe_writes=module.params['unsafe_writes'])

 253 def present(module, dest, regexp, line, insertafter, insertbefore, create,
 254             backup, backrefs, firstmatch):
 255 
 256     diff = {'before': '',
 257             'after': '',
 258             'before_header': '%s (content)' % dest,
 259             'after_header': '%s (content)' % dest}
 260 
 261     b_dest = to_bytes(dest, errors='surrogate_or_strict')
 262     if not os.path.exists(b_dest):
 263         if not create:
 264             module.fail_json(rc=257, msg='Destination %s does not exist !' % dest)
 265         b_destpath = os.path.dirname(b_dest)
 266         if b_destpath and not os.path.exists(b_destpath) and not module.check_mode:
 267             try:
 268                 os.makedirs(b_destpath)
 269             except Exception as e:
 270                 module.fail_json(msg='Error creating %s Error code: %s Error description: %s' % (b_destpath, e[0], e[1]))
 271 
 272         b_lines = []
 273     else:
 274         with open(b_dest, 'rb') as f:
 275             b_lines = f.readlines()
 276 
 277     if module._diff:
 278         diff['before'] = to_native(b''.join(b_lines))
 279 
 280     if regexp is not None:
 281         bre_m = re.compile(to_bytes(regexp, errors='surrogate_or_strict'))
 282 
 283     if insertafter not in (None, 'BOF', 'EOF'):
 284         bre_ins = re.compile(to_bytes(insertafter, errors='surrogate_or_strict'))
 285     elif insertbefore not in (None, 'BOF'):
 286         bre_ins = re.compile(to_bytes(insertbefore, errors='surrogate_or_strict'))
 287     else:
 288         bre_ins = None
 289 
 290     # index[0] is the line num where regexp has been found
 291     # index[1] is the line num where insertafter/insertbefore has been found
 292     index = [-1, -1]
 293     match = None
 294     exact_line_match = False
 295     b_line = to_bytes(line, errors='surrogate_or_strict')
 296 
 297     # The module's doc says
 298     # "If regular expressions are passed to both regexp and
 299     # insertafter, insertafter is only honored if no match for regexp is found."
 300     # Therefore:
 301     # 1. regexp was found -> ignore insertafter, replace the founded line
 302     # 2. regexp was not found -> insert the line after 'insertafter' or 'insertbefore' line
 303 
 304     # Given the above:
 305     # 1. First check that there is no match for regexp:
 306     if regexp is not None:
 307         for lineno, b_cur_line in enumerate(b_lines):
 308             match_found = bre_m.search(b_cur_line)
 309             if match_found:
 310                 index[0] = lineno
 311                 match = match_found
 312                 if firstmatch:
 313                     break
 314 
 315     # 2. When no match found on the previous step,
 316     # parse for searching insertafter/insertbefore:
 317     if not match:
 318         for lineno, b_cur_line in enumerate(b_lines):
 319             if b_line == b_cur_line.rstrip(b'\r\n'):
 320                 index[0] = lineno
 321                 exact_line_match = True
 322 
 323             elif bre_ins is not None and bre_ins.search(b_cur_line):
 324                 if insertafter:
 325                     # + 1 for the next line
 326                     index[1] = lineno + 1
 327                     if firstmatch:
 328                         break
 329 
 330                 if insertbefore:
 331                     # index[1] for the previous line
 332                     index[1] = lineno
 333                     if firstmatch:
 334                         break
 335 
 336     msg = ''
 337     changed = False
 338     b_linesep = to_bytes(os.linesep, errors='surrogate_or_strict')
 339     # Exact line or Regexp matched a line in the file
 340     if index[0] != -1:
 341         if backrefs and match:
 342             b_new_line = match.expand(b_line)
 343         else:
 344             # Don't do backref expansion if not asked.
 345             b_new_line = b_line
 346 
 347         if not b_new_line.endswith(b_linesep):
 348             b_new_line += b_linesep
 349 
 350         # If no regexp was given and no line match is found anywhere in the file,
 351         # insert the line appropriately if using insertbefore or insertafter
 352         if regexp is None and match is None and not exact_line_match:
 353 
 354             # Insert lines
 355             if insertafter and insertafter != 'EOF':
 356                 # Ensure there is a line separator after the found string
 357                 # at the end of the file.
 358                 if b_lines and not b_lines[-1][-1:] in (b'\n', b'\r'):
 359                     b_lines[-1] = b_lines[-1] + b_linesep
 360 
 361                 # If the line to insert after is at the end of the file
 362                 # use the appropriate index value.
 363                 if len(b_lines) == index[1]:
 364                     if b_lines[index[1] - 1].rstrip(b'\r\n') != b_line:
 365                         b_lines.append(b_line + b_linesep)
 366                         msg = 'line added'
 367                         changed = True
 368                 elif b_lines[index[1]].rstrip(b'\r\n') != b_line:
 369                     b_lines.insert(index[1], b_line + b_linesep)
 370                     msg = 'line added'
 371                     changed = True
 372 
 373             elif insertbefore and insertbefore != 'BOF':
 374                 # If the line to insert before is at the beginning of the file
 375                 # use the appropriate index value.
 376                 if index[1] <= 0:
 377                     if b_lines[index[1]].rstrip(b'\r\n') != b_line:
 378                         b_lines.insert(index[1], b_line + b_linesep)
 379                         msg = 'line added'
 380                         changed = True
 381 
 382                 elif b_lines[index[1] - 1].rstrip(b'\r\n') != b_line:
 383                     b_lines.insert(index[1], b_line + b_linesep)
 384                     msg = 'line added'
 385                     changed = True
 386 
 387         elif b_lines[index[0]] != b_new_line:
 388             b_lines[index[0]] = b_new_line
 389             msg = 'line replaced'
 390             changed = True
 391 
 392     elif backrefs:
 393         # Do absolutely nothing, since it's not safe generating the line
 394         # without the regexp matching to populate the backrefs.
 395         pass
 396     # Add it to the beginning of the file
 397     elif insertbefore == 'BOF' or insertafter == 'BOF':
 398         b_lines.insert(0, b_line + b_linesep)
 399         msg = 'line added'
 400         changed = True
 401     # Add it to the end of the file if requested or
 402     # if insertafter/insertbefore didn't match anything
 403     # (so default behaviour is to add at the end)
 404     elif insertafter == 'EOF' or index[1] == -1:
 405 
 406         # If the file is not empty then ensure there's a newline before the added line
 407         if b_lines and not b_lines[-1][-1:] in (b'\n', b'\r'):
 408             b_lines.append(b_linesep)
 409 
 410         b_lines.append(b_line + b_linesep)
 411         msg = 'line added'
 412         changed = True
 413 
 414     elif insertafter and index[1] != -1:
 415 
 416         # Don't insert the line if it already matches at the index.
 417         # If the line to insert after is at the end of the file use the appropriate index value.
 418         if len(b_lines) == index[1]:
 419             if b_lines[index[1] - 1].rstrip(b'\r\n') != b_line:
 420                 b_lines.append(b_line + b_linesep)
 421                 msg = 'line added'
 422                 changed = True
 423         elif b_line != b_lines[index[1]].rstrip(b'\n\r'):
 424             b_lines.insert(index[1], b_line + b_linesep)
 425             msg = 'line added'
 426             changed = True
 427 
 428     # insert matched, but not the regexp
 429     else:
 430         b_lines.insert(index[1], b_line + b_linesep)
 431         msg = 'line added'
 432         changed = True
 433 
 434     if module._diff:
 435         diff['after'] = to_native(b''.join(b_lines))
 436 
 437     backupdest = ""
 438     if changed and not module.check_mode:
 439         if backup and os.path.exists(b_dest):
 440             backupdest = module.backup_local(dest)
 441         write_changes(module, b_lines, dest)
 442 
 443     if module.check_mode and not os.path.exists(b_dest):
 444         module.exit_json(changed=changed, msg=msg, backup=backupdest, diff=diff)
 445 
 446     attr_diff = {}
 447     msg, changed = check_file_attrs(module, changed, msg, attr_diff)
 448 
 449     attr_diff['before_header'] = '%s (file attributes)' % dest
 450     attr_diff['after_header'] = '%s (file attributes)' % dest
 451 
 452     difflist = [diff, attr_diff]
 453     module.exit_json(changed=changed, msg=msg, backup=backupdest, diff=difflist)

 456 def absent(module, dest, regexp, line, backup):
 457 
 458     b_dest = to_bytes(dest, errors='surrogate_or_strict')
 459     if not os.path.exists(b_dest):
 460         module.exit_json(changed=False, msg="file not present")
 461 
 462     msg = ''
 463     diff = {'before': '',
 464             'after': '',
 465             'before_header': '%s (content)' % dest,
 466             'after_header': '%s (content)' % dest}
 467 
 468     with open(b_dest, 'rb') as f:
 469         b_lines = f.readlines()
 470 
 471     if module._diff:
 472         diff['before'] = to_native(b''.join(b_lines))
 473 
 474     if regexp is not None:
 475         bre_c = re.compile(to_bytes(regexp, errors='surrogate_or_strict'))
 476     found = []
 477 
 478     b_line = to_bytes(line, errors='surrogate_or_strict')
 479 
 480     def matcher(b_cur_line):
 481         if regexp is not None:
 482             match_found = bre_c.search(b_cur_line)
 483         else:
 484             match_found = b_line == b_cur_line.rstrip(b'\r\n')
 485         if match_found:
 486             found.append(b_cur_line)
 487         return not match_found
 488 
 489     b_lines = [l for l in b_lines if matcher(l)]
 490     changed = len(found) > 0
 491 
 492     if module._diff:
 493         diff['after'] = to_native(b''.join(b_lines))
 494 
 495     backupdest = ""
 496     if changed and not module.check_mode:
 497         if backup:
 498             backupdest = module.backup_local(dest)
 499         write_changes(module, b_lines, dest)
 500 
 501     if changed:
 502         msg = "%s line(s) removed" % len(found)
 503 
 504     attr_diff = {}
 505     msg, changed = check_file_attrs(module, changed, msg, attr_diff)
 506 
 507     attr_diff['before_header'] = '%s (file attributes)' % dest
 508     attr_diff['after_header'] = '%s (file attributes)' % dest
 509 
 510     difflist = [diff, attr_diff]
 511 
 512     module.exit_json(changed=changed, found=len(found), msg=msg, backup=backupdest, diff=difflist)

 515 def main():
 516     module = AnsibleModule(
 517         argument_spec=dict(
 518             path=dict(type='path', required=True, aliases=['dest', 'destfile', 'name']),
 519             state=dict(type='str', default='present', choices=['absent', 'present']),
 520             regexp=dict(type='str', aliases=['regex']),
 521             line=dict(type='str', aliases=['value']),
 522             insertafter=dict(type='str'),
 523             insertbefore=dict(type='str'),
 524             backrefs=dict(type='bool', default=False),
 525             create=dict(type='bool', default=False),
 526             backup=dict(type='bool', default=False),
 527             firstmatch=dict(type='bool', default=False),
 528             validate=dict(type='str'),
 529         ),
 530         mutually_exclusive=[['insertbefore', 'insertafter']],
 531         add_file_common_args=True,
 532         supports_check_mode=True,
 533     )
 534 
 535     params = module.params
 536     create = params['create']
 537     backup = params['backup']
 538     backrefs = params['backrefs']
 539     path = params['path']
 540     firstmatch = params['firstmatch']
 541     regexp = params['regexp']
 542     line = params['line']
 543 
 544     if regexp == '':
 545         module.warn(
 546             "The regular expression is an empty string, which will match every line in the file. "
 547             "This may have unintended consequences, such as replacing the last line in the file rather than appending. "
 548             "If this is desired, use '^' to match every line in the file and avoid this warning.")
 549 
 550     b_path = to_bytes(path, errors='surrogate_or_strict')
 551     if os.path.isdir(b_path):
 552         module.fail_json(rc=256, msg='Path %s is a directory !' % path)
 553 
 554     if params['state'] == 'present':
 555         if backrefs and regexp is None:
 556             module.fail_json(msg='regexp is required with backrefs=true')
 557 
 558         if line is None:
 559             module.fail_json(msg='line is required with state=present')
 560 
 561         # Deal with the insertafter default value manually, to avoid errors
 562         # because of the mutually_exclusive mechanism.
 563         ins_bef, ins_aft = params['insertbefore'], params['insertafter']
 564         if ins_bef is None and ins_aft is None:
 565             ins_aft = 'EOF'
 566 
 567         present(module, path, regexp, line,
 568                 ins_aft, ins_bef, create, backup, backrefs, firstmatch)
 569     else:
 570         if regexp is None and line is None:
 571             module.fail_json(msg='one of line or regexp is required with state=absent')
 572 
 573         absent(module, path, regexp, line, backup)
 574 
 575 
 576 if __name__ == '__main__':
 577     main()
