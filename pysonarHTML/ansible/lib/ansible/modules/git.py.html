 297 def relocate_repo(module, result, repo_dir, old_repo_dir, worktree_dir):
 298     if os.path.exists(repo_dir):
 299         module.fail_json(msg='Separate-git-dir path %s already exists.' % repo_dir)
 300     if worktree_dir:
 301         dot_git_file_path = os.path.join(worktree_dir, '.git')
 302         try:
 303             shutil.move(old_repo_dir, repo_dir)
 304             with open(dot_git_file_path, 'w') as dot_git_file:
 305                 dot_git_file.write('gitdir: %s' % repo_dir)
 306             result['git_dir_before'] = old_repo_dir
 307             result['git_dir_now'] = repo_dir
 308         except (IOError, OSError) as err:
 309             # if we already moved the .git dir, roll it back
 310             if os.path.exists(repo_dir):
 311                 shutil.move(repo_dir, old_repo_dir)
 312             module.fail_json(msg=u'Unable to move git dir. %s' % to_text(err))

 315 def head_splitter(headfile, remote, module=None, fail_on_error=False):
 316     '''Extract the head reference'''
 317     # https://github.com/ansible/ansible-modules-core/pull/907
 318 
 319     res = None
 320     if os.path.exists(headfile):
 321         rawdata = None
 322         try:
 323             f = open(headfile, 'r')
 324             rawdata = f.readline()
 325             f.close()
 326         except Exception:
 327             if fail_on_error and module:
 328                 module.fail_json(msg="Unable to read %s" % headfile)
 329         if rawdata:
 330             try:
 331                 rawdata = rawdata.replace('refs/remotes/%s' % remote, '', 1)
 332                 refparts = rawdata.split(' ')
 333                 newref = refparts[-1]
 334                 nrefparts = newref.split('/', 2)
 335                 res = nrefparts[-1].rstrip('\n')
 336             except Exception:
 337                 if fail_on_error and module:
 338                     module.fail_json(msg="Unable to split head from '%s'" % rawdata)
 339     return res

 350 def get_submodule_update_params(module, git_path, cwd):
 351     # or: git submodule [--quiet] update [--init] [-N|--no-fetch]
 352     # [-f|--force] [--rebase] [--reference <repository>] [--merge]
 353     # [--recursive] [--] [<path>...]
 354 
 355     params = []
 356 
 357     # run a bad submodule command to get valid params
 358     cmd = "%s submodule update --help" % (git_path)
 359     rc, stdout, stderr = module.run_command(cmd, cwd=cwd)
 360     lines = stderr.split('\n')
 361     update_line = None
 362     for line in lines:
 363         if 'git submodule [--quiet] update ' in line:
 364             update_line = line
 365     if update_line:
 366         update_line = update_line.replace('[', '')
 367         update_line = update_line.replace(']', '')
 368         update_line = update_line.replace('|', ' ')
 369         parts = shlex.split(update_line)
 370         for part in parts:
 371             if part.startswith('--'):
 372                 part = part.replace('--', '')
 373                 params.append(part)
 374 
 375     return params

 378 def write_ssh_wrapper(module_tmpdir):
 379     try:
 380         # make sure we have full permission to the module_dir, which
 381         # may not be the case if we're sudo'ing to a non-root user
 382         if os.access(module_tmpdir, os.W_OK | os.R_OK | os.X_OK):
 383             fd, wrapper_path = tempfile.mkstemp(prefix=module_tmpdir + '/')
 384         else:
 385             raise OSError
 386     except (IOError, OSError):
 387         fd, wrapper_path = tempfile.mkstemp()
 388     fh = os.fdopen(fd, 'w+b')
 389     template = b("""#!/bin/sh
 390 if [ -z "$GIT_SSH_OPTS" ]; then
 391     BASEOPTS=""
 392 else
 393     BASEOPTS=$GIT_SSH_OPTS
 394 fi
 395 
 396 # Let ssh fail rather than prompt
 397 BASEOPTS="$BASEOPTS -o BatchMode=yes"
 398 
 399 if [ -z "$GIT_KEY" ]; then
 400     ssh $BASEOPTS "$@"
 401 else
 402     ssh -i "$GIT_KEY" -o IdentitiesOnly=yes $BASEOPTS "$@"
 403 fi
 404 """)
 405     fh.write(template)
 406     fh.close()
 407     st = os.stat(wrapper_path)
 408     os.chmod(wrapper_path, st.st_mode | stat.S_IEXEC)
 409     return wrapper_path

 431 def get_version(module, git_path, dest, ref="HEAD"):
 432     ''' samples the version of the git repo '''
 433 
 434     cmd = "%s rev-parse %s" % (git_path, ref)
 435     rc, stdout, stderr = module.run_command(cmd, cwd=dest)
 436     sha = to_native(stdout).rstrip('\n')
 437     return sha

 440 def get_submodule_versions(git_path, module, dest, version='HEAD'):
 441     cmd = [git_path, 'submodule', 'foreach', git_path, 'rev-parse', version]
 442     (rc, out, err) = module.run_command(cmd, cwd=dest)
 443     if rc != 0:
 444         module.fail_json(
 445             msg='Unable to determine hashes of submodules',
 446             stdout=out,
 447             stderr=err,
 448             rc=rc)
 449     submodules = {}
 450     subm_name = None
 451     for line in out.splitlines():
 452         if line.startswith("Entering '"):
 453             subm_name = line[10:-1]
 454         elif len(line.strip()) == 40:
 455             if subm_name is None:
 456                 module.fail_json()
 457             submodules[subm_name] = line.strip()
 458             subm_name = None
 459         else:
 460             module.fail_json(msg='Unable to parse submodule hash line: %s' % line.strip())
 461     if subm_name is not None:
 462         module.fail_json(msg='Unable to find hash for submodule: %s' % subm_name)
 463 
 464     return submodules

 467 def clone(git_path, module, repo, dest, remote, depth, version, bare,
 468           reference, refspec, git_version_used, verify_commit, separate_git_dir, result, gpg_whitelist, single_branch):
 469     ''' makes a new git repo if it does not already exist '''
 470     dest_dirname = os.path.dirname(dest)
 471     try:
 472         os.makedirs(dest_dirname)
 473     except Exception:
 474         pass
 475     cmd = [git_path, 'clone']
 476 
 477     if bare:
 478         cmd.append('--bare')
 479     else:
 480         cmd.extend(['--origin', remote])
 481 
 482     is_branch_or_tag = is_remote_branch(git_path, module, dest, repo, version) or is_remote_tag(git_path, module, dest, repo, version)
 483     if depth:
 484         if version == 'HEAD' or refspec:
 485             cmd.extend(['--depth', str(depth)])
 486         elif is_branch_or_tag:
 487             cmd.extend(['--depth', str(depth)])
 488             cmd.extend(['--branch', version])
 489         else:
 490             # only use depth if the remote object is branch or tag (i.e. fetchable)
 491             module.warn("Ignoring depth argument. "
 492                         "Shallow clones are only available for "
 493                         "HEAD, branches, tags or in combination with refspec.")
 494     if reference:
 495         cmd.extend(['--reference', str(reference)])
 496 
 497     if single_branch:
 498         if git_version_used is None:
 499             module.fail_json(msg='Cannot find git executable at %s' % git_path)
 500 
 501         if git_version_used < LooseVersion('1.7.10'):
 502             module.warn("git version '%s' is too old to use 'single-branch'. Ignoring." % git_version_used)
 503         else:
 504             cmd.append("--single-branch")
 505 
 506             if is_branch_or_tag:
 507                 cmd.extend(['--branch', version])
 508 
 509     needs_separate_git_dir_fallback = False
 510     if separate_git_dir:
 511         if git_version_used is None:
 512             module.fail_json(msg='Cannot find git executable at %s' % git_path)
 513         if git_version_used < LooseVersion('1.7.5'):
 514             # git before 1.7.5 doesn't have separate-git-dir argument, do fallback
 515             needs_separate_git_dir_fallback = True
 516         else:
 517             cmd.append('--separate-git-dir=%s' % separate_git_dir)
 518 
 519     cmd.extend([repo, dest])
 520     module.run_command(cmd, check_rc=True, cwd=dest_dirname)
 521     if needs_separate_git_dir_fallback:
 522         relocate_repo(module, result, separate_git_dir, os.path.join(dest, ".git"), dest)
 523 
 524     if bare and remote != 'origin':
 525         module.run_command([git_path, 'remote', 'add', remote, repo], check_rc=True, cwd=dest)
 526 
 527     if refspec:
 528         cmd = [git_path, 'fetch']
 529         if depth:
 530             cmd.extend(['--depth', str(depth)])
 531         cmd.extend([remote, refspec])
 532         module.run_command(cmd, check_rc=True, cwd=dest)
 533 
 534     if verify_commit:
 535         verify_commit_sign(git_path, module, dest, version, gpg_whitelist)

 538 def has_local_mods(module, git_path, dest, bare):
 539     if bare:
 540         return False
 541 
 542     cmd = "%s status --porcelain" % (git_path)
 543     rc, stdout, stderr = module.run_command(cmd, cwd=dest)
 544     lines = stdout.splitlines()
 545     lines = list(filter(lambda c: not re.search('^\\?\\?.*$', c), lines))
 546 
 547     return len(lines) > 0

 552     Resets the index and working tree to HEAD.

 560 def get_diff(module, git_path, dest, repo, remote, depth, bare, before, after):
 561     ''' Return the difference between 2 versions '''
 562     if before is None:
 563         return {'prepared': '>> Newly checked out %s' % after}
 564     elif before != after:
 565         # Ensure we have the object we are referring to during git diff !
 566         git_version_used = git_version(git_path, module)
 567         fetch(git_path, module, repo, dest, after, remote, depth, bare, '', git_version_used)
 568         cmd = '%s diff %s %s' % (git_path, before, after)
 569         (rc, out, err) = module.run_command(cmd, cwd=dest)
 570         if rc == 0 and out:
 571             return {'prepared': out}
 572         elif rc == 0:
 573             return {'prepared': '>> No visual differences between %s and %s' % (before, after)}
 574         elif err:
 575             return {'prepared': '>> Failed to get proper diff between %s and %s:\n>> %s' % (before, after, err)}
 576         else:
 577             return {'prepared': '>> Failed to get proper diff between %s and %s' % (before, after)}
 578     return {}

 581 def get_remote_head(git_path, module, dest, version, remote, bare):
 582     cloning = False
 583     cwd = None
 584     tag = False
 585     if remote == module.params['repo']:
 586         cloning = True
 587     elif remote == 'file://' + os.path.expanduser(module.params['repo']):
 588         cloning = True
 589     else:
 590         cwd = dest
 591     if version == 'HEAD':
 592         if cloning:
 593             # cloning the repo, just get the remote's HEAD version
 594             cmd = '%s ls-remote %s -h HEAD' % (git_path, remote)
 595         else:
 596             head_branch = get_head_branch(git_path, module, dest, remote, bare)
 597             cmd = '%s ls-remote %s -h refs/heads/%s' % (git_path, remote, head_branch)
 598     elif is_remote_branch(git_path, module, dest, remote, version):
 599         cmd = '%s ls-remote %s -h refs/heads/%s' % (git_path, remote, version)
 600     elif is_remote_tag(git_path, module, dest, remote, version):
 601         tag = True
 602         cmd = '%s ls-remote %s -t refs/tags/%s*' % (git_path, remote, version)
 603     else:
 604         # appears to be a sha1.  return as-is since it appears
 605         # cannot check for a specific sha1 on remote
 606         return version
 607     (rc, out, err) = module.run_command(cmd, check_rc=True, cwd=cwd)
 608     if len(out) < 1:
 609         module.fail_json(msg="Could not determine remote revision for %s" % version, stdout=out, stderr=err, rc=rc)
 610 
 611     out = to_native(out)
 612 
 613     if tag:
 614         # Find the dereferenced tag if this is an annotated tag.
 615         for tag in out.split('\n'):
 616             if tag.endswith(version + '^{}'):
 617                 out = tag
 618                 break
 619             elif tag.endswith(version):
 620                 out = tag
 621 
 622     rev = out.split()[0]

 689 def get_repo_path(dest, bare):
 690     if bare:
 691         repo_path = dest
 692     else:
 693         repo_path = os.path.join(dest, '.git')
 694     # Check if the .git is a file. If it is a file, it means that the repository is in external directory respective to the working copy (e.g. we are in a
 695     # submodule structure).
 696     if os.path.isfile(repo_path):
 697         with open(repo_path, 'r') as gitfile:
 698             data = gitfile.read()
 699         ref_prefix, gitdir = data.rstrip().split('gitdir: ', 1)
 700         if ref_prefix:
 701             raise ValueError('.git file has invalid git dir reference format')
 702 
 703         # There is a possibility the .git file to have an absolute path.
 704         if os.path.isabs(gitdir):
 705             repo_path = gitdir
 706         else:
 707             repo_path = os.path.join(repo_path.split('.git')[0], gitdir)
 708         if not os.path.isdir(repo_path):
 709             raise ValueError('%s is not a directory' % repo_path)
 710     return repo_path

 713 def get_head_branch(git_path, module, dest, remote, bare=False):
 714     '''
 715     Determine what branch HEAD is associated with.  This is partly
 716     taken from lib/ansible/utils/__init__.py.  It finds the correct
 717     path to .git/HEAD and reads from that file the branch that HEAD is
 718     associated with.  In the case of a detached HEAD, this will look
 719     up the branch in .git/refs/remotes/<remote>/HEAD.
 720     '''
 721     try:
 722         repo_path = get_repo_path(dest, bare)
 723     except (IOError, ValueError) as err:
 724         # No repo path found
 725         """``.git`` file does not have a valid format for detached Git dir."""
 726         module.fail_json(
 727             msg='Current repo does not have a valid reference to a '
 728             'separate Git dir or it refers to the invalid path',
 729             details=to_text(err),
 730         )
 731     # Read .git/HEAD for the name of the branch.
 732     # If we're in a detached HEAD state, look up the branch associated with
 733     # the remote HEAD in .git/refs/remotes/<remote>/HEAD
 734     headfile = os.path.join(repo_path, "HEAD")
 735     if is_not_a_branch(git_path, module, dest):
 736         headfile = os.path.join(repo_path, 'refs', 'remotes', remote, 'HEAD')
 737     branch = head_splitter(headfile, remote, module=module, fail_on_error=True)
 738     return branch

 770 def fetch(git_path, module, repo, dest, version, remote, depth, bare, refspec, git_version_used, force=False):
 771     ''' updates repo from remote sources '''
 772     set_remote_url(git_path, module, repo, dest, remote)
 773     commands = []
 774 
 775     fetch_str = 'download remote objects and refs'
 776     fetch_cmd = [git_path, 'fetch']
 777 
 778     refspecs = []
 779     if depth:
 780         # try to find the minimal set of refs we need to fetch to get a
 781         # successful checkout
 782         currenthead = get_head_branch(git_path, module, dest, remote)
 783         if refspec:
 784             refspecs.append(refspec)
 785         elif version == 'HEAD':
 786             refspecs.append(currenthead)
 787         elif is_remote_branch(git_path, module, dest, repo, version):
 788             if currenthead != version:
 789                 # this workaround is only needed for older git versions
 790                 # 1.8.3 is broken, 1.9.x works
 791                 # ensure that remote branch is available as both local and remote ref
 792                 refspecs.append('+refs/heads/%s:refs/heads/%s' % (version, version))
 793             refspecs.append('+refs/heads/%s:refs/remotes/%s/%s' % (version, remote, version))
 794         elif is_remote_tag(git_path, module, dest, repo, version):
 795             refspecs.append('+refs/tags/' + version + ':refs/tags/' + version)
 796         if refspecs:
 797             # if refspecs is empty, i.e. version is neither heads nor tags
 798             # assume it is a version hash
 799             # fall back to a full clone, otherwise we might not be able to checkout
 800             # version
 801             fetch_cmd.extend(['--depth', str(depth)])
 802 
 803     if not depth or not refspecs:
 804         # don't try to be minimalistic but do a full clone
 805         # also do this if depth is given, but version is something that can't be fetched directly
 806         if bare:
 807             refspecs = ['+refs/heads/*:refs/heads/*', '+refs/tags/*:refs/tags/*']
 808         else:
 809             # ensure all tags are fetched
 810             if git_version_used >= LooseVersion('1.9'):
 811                 fetch_cmd.append('--tags')
 812             else:
 813                 # old git versions have a bug in --tags that prevents updating existing tags
 814                 commands.append((fetch_str, fetch_cmd + [remote]))
 815                 refspecs = ['+refs/tags/*:refs/tags/*']
 816         if refspec:
 817             refspecs.append(refspec)
 818 
 819     if force:
 820         fetch_cmd.append('--force')
 821 
 822     fetch_cmd.extend([remote])
 823 
 824     commands.append((fetch_str, fetch_cmd + refspecs))
 825 
 826     for (label, command) in commands:
 827         (rc, out, err) = module.run_command(command, cwd=dest)
 828         if rc != 0:
 829             module.fail_json(msg="Failed to %s: %s %s" % (label, out, err), cmd=command)

 832 def submodules_fetch(git_path, module, remote, track_submodules, dest):
 833     changed = False
 834 
 835     if not os.path.exists(os.path.join(dest, '.gitmodules')):
 836         # no submodules
 837         return changed
 838 
 839     gitmodules_file = open(os.path.join(dest, '.gitmodules'), 'r')
 840     for line in gitmodules_file:
 841         # Check for new submodules
 842         if not changed and line.strip().startswith('path'):
 843             path = line.split('=', 1)[1].strip()
 844             # Check that dest/path/.git exists
 845             if not os.path.exists(os.path.join(dest, path, '.git')):
 846                 changed = True
 847 
 848     # Check for updates to existing modules
 849     if not changed:
 850         # Fetch updates
 851         begin = get_submodule_versions(git_path, module, dest)
 852         cmd = [git_path, 'submodule', 'foreach', git_path, 'fetch']
 853         (rc, out, err) = module.run_command(cmd, check_rc=True, cwd=dest)
 854         if rc != 0:
 855             module.fail_json(msg="Failed to fetch submodules: %s" % out + err)
 856 
 857         if track_submodules:
 858             # Compare against submodule HEAD
 859             # FIXME: determine this from .gitmodules
 860             version = 'master'
 861             after = get_submodule_versions(git_path, module, dest, '%s/%s' % (remote, version))
 862             if begin != after:
 863                 changed = True
 864         else:
 865             # Compare against the superproject's expectation
 866             cmd = [git_path, 'submodule', 'status']
 867             (rc, out, err) = module.run_command(cmd, check_rc=True, cwd=dest)
 868             if rc != 0:
 869                 module.fail_json(msg='Failed to retrieve submodule status: %s' % out + err)
 870             for line in out.splitlines():
 871                 if line[0] != ' ':
 872                     changed = True
 873                     break
 874     return changed

 915 def switch_version(git_path, module, dest, remote, version, verify_commit, depth, gpg_whitelist):
 916     cmd = ''
 917     if version == 'HEAD':
 918         branch = get_head_branch(git_path, module, dest, remote)
 919         (rc, out, err) = module.run_command("%s checkout --force %s" % (git_path, branch), cwd=dest)
 920         if rc != 0:
 921             module.fail_json(msg="Failed to checkout branch %s" % branch,
 922                              stdout=out, stderr=err, rc=rc)
 923         cmd = "%s reset --hard %s/%s --" % (git_path, remote, branch)
 924     else:
 925         # FIXME check for local_branch first, should have been fetched already
 926         if is_remote_branch(git_path, module, dest, remote, version):
 927             if depth and not is_local_branch(git_path, module, dest, version):
 928                 # git clone --depth implies --single-branch, which makes
 929                 # the checkout fail if the version changes
 930                 # fetch the remote branch, to be able to check it out next
 931                 set_remote_branch(git_path, module, dest, remote, version, depth)
 932             if not is_local_branch(git_path, module, dest, version):
 933                 cmd = "%s checkout --track -b %s %s/%s" % (git_path, version, remote, version)
 934             else:
 935                 (rc, out, err) = module.run_command("%s checkout --force %s" % (git_path, version), cwd=dest)
 936                 if rc != 0:
 937                     module.fail_json(msg="Failed to checkout branch %s" % version, stdout=out, stderr=err, rc=rc)
 938                 cmd = "%s reset --hard %s/%s" % (git_path, remote, version)
 939         else:
 940             cmd = "%s checkout --force %s" % (git_path, version)
 941     (rc, out1, err1) = module.run_command(cmd, cwd=dest)
 942     if rc != 0:
 943         if version != 'HEAD':
 944             module.fail_json(msg="Failed to checkout %s" % (version),
 945                              stdout=out1, stderr=err1, rc=rc, cmd=cmd)
 946         else:
 947             module.fail_json(msg="Failed to checkout branch %s" % (branch),
 948                              stdout=out1, stderr=err1, rc=rc, cmd=cmd)
 949 
 950     if verify_commit:
 951         verify_commit_sign(git_path, module, dest, version, gpg_whitelist)
 952 
 953     return (rc, out1, err1)

 956 def verify_commit_sign(git_path, module, dest, version, gpg_whitelist):
 957     if version in get_annotated_tags(git_path, module, dest):
 958         git_sub = "verify-tag"
 959     else:
 960         git_sub = "verify-commit"
 961     cmd = "%s %s %s --raw" % (git_path, git_sub, version)
 962     (rc, out, err) = module.run_command(cmd, cwd=dest)
 963     if rc != 0:
 964         module.fail_json(msg='Failed to verify GPG signature of commit/tag "%s"' % version, stdout=out, stderr=err, rc=rc)
 965     if gpg_whitelist:
 966         fingerprint = get_gpg_fingerprint(err)
 967         if fingerprint not in gpg_whitelist:
 968             module.fail_json(msg='The gpg_whitelist does not include the public key "%s" for this commit' % fingerprint, stdout=out, stderr=err, rc=rc)
 969     return (rc, out, err)

 972 def get_gpg_fingerprint(output):
 973     """Return a fingerprint of the primary key.
 974 
 975     Ref:
 976     https://git.gnupg.org/cgi-bin/gitweb.cgi?p=gnupg.git;a=blob;f=doc/DETAILS;hb=HEAD#l482
 977     """
 978     for line in output.splitlines():
 979         data = line.split()
 980         if data[1] != 'VALIDSIG':
 981             continue
 982 
 983         # if signed with a subkey, this contains the primary key fingerprint
 984         data_id = 11 if len(data) == 11 else 2
 985         return data[data_id]

 989     """return the installed version of git"""

1017 def create_archive(git_path, module, dest, archive, archive_prefix, version, repo, result):
1018     """ Helper function for creating archive using git_archive """
1019     all_archive_fmt = {'.zip': 'zip', '.gz': 'tar.gz', '.tar': 'tar',
1020                        '.tgz': 'tgz'}
1021     _, archive_ext = os.path.splitext(archive)
1022     archive_fmt = all_archive_fmt.get(archive_ext, None)
1023     if archive_fmt is None:
1024         module.fail_json(msg="Unable to get file extension from "
1025                              "archive file name : %s" % archive,
1026                          details="Please specify archive as filename with "
1027                                  "extension. File extension can be one "
1028                                  "of ['tar', 'tar.gz', 'zip', 'tgz']")
1029 
1030     repo_name = repo.split("/")[-1].replace(".git", "")
1031 
1032     if os.path.exists(archive):
1033         # If git archive file exists, then compare it with new git archive file.
1034         # if match, do nothing
1035         # if does not match, then replace existing with temp archive file.
1036         tempdir = tempfile.mkdtemp()
1037         new_archive_dest = os.path.join(tempdir, repo_name)
1038         new_archive = new_archive_dest + '.' + archive_fmt
1039         git_archive(git_path, module, dest, new_archive, archive_fmt, archive_prefix, version)
1040 
1041         # filecmp is supposed to be efficient than md5sum checksum
1042         if filecmp.cmp(new_archive, archive):
1043             result.update(changed=False)
1044             # Cleanup before exiting
1045             try:
1046                 shutil.rmtree(tempdir)
1047             except OSError:
1048                 pass
1049         else:
1050             try:
1051                 shutil.move(new_archive, archive)
1052                 shutil.rmtree(tempdir)
1053                 result.update(changed=True)
1054             except OSError as e:
1055                 module.fail_json(msg="Failed to move %s to %s" %
1056                                      (new_archive, archive),
1057                                  details=u"Error occurred while moving : %s"
1058                                          % to_text(e))
1059     else:
1060         # Perform archive from local directory
1061         git_archive(git_path, module, dest, archive, archive_fmt, archive_prefix, version)
1062         result.update(changed=True)
