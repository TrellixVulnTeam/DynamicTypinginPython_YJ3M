  11 DOCUMENTATION = r'''
  12 ---
  13 module: add_host
  14 short_description: Add a host (and alternatively a group) to the ansible-playbook in-memory inventory
  15 description:
  16 - Use variables to create new hosts and groups in inventory for use in later plays of the same playbook.
  17 - Takes variables so you can define the new hosts more fully.
  18 - This module is also supported for Windows targets.
  19 version_added: "0.9"
  20 options:
  21   name:
  22     description:
  23     - The hostname/ip of the host to add to the inventory, can include a colon and a port number.
  24     type: str
  25     required: true
  26     aliases: [ host, hostname ]
  27   groups:
  28     description:
  29     - The groups to add the hostname to.
  30     type: list
  31     aliases: [ group, groupname ]
  32 notes:
  33 - This module bypasses the play host loop and only runs once for all the hosts in the play, if you need it
  34   to iterate use a C(loop) construct. If you need to dynamically add all hosts targeted by a playbook for
  35   later use, the C(group_by) module is potentially a better choice.
  36 - The alias C(host) of the parameter C(name) is only available on Ansible 2.4 and newer.
  37 - Since Ansible 2.4, the C(inventory_dir) variable is now set to C(None) instead of the 'global inventory source',
  38   because you can now have multiple sources.  An example was added that shows how to partially restore the previous behaviour.
  39 - Windows targets are supported by this module.
  40 seealso:
  41 - module: group_by
  42 author:
  43 - Ansible Core Team
  44 - Seth Vidal (@skvidal)
  45 '''
  46 
  47 EXAMPLES = r'''
  48 - name: Add host to group 'just_created' with variable foo=42
  49   add_host:
  50     name: '{{ ip_from_ec2 }}'
  51     groups: just_created
  52     foo: 42
  53 
  54 - name: Add host to multiple groups
  55   add_host:
  56     hostname: '{{ new_ip }}'
  57     groups:
  58     - group1
  59     - group2
  60 
  61 - name: Add a host with a non-standard port local to your machines
  62   add_host:
  63     name: '{{ new_ip }}:{{ new_port }}'
  64 
  65 - name: Add a host alias that we reach through a tunnel (Ansible 1.9 and older)
  66   add_host:
  67     hostname: '{{ new_ip }}'
  68     ansible_ssh_host: '{{ inventory_hostname }}'
  69     ansible_ssh_port: '{{ new_port }}'
  70 
  71 - name: Add a host alias that we reach through a tunnel (Ansible 2.0 and newer)
  72   add_host:
  73     hostname: '{{ new_ip }}'
  74     ansible_host: '{{ inventory_hostname }}'
  75     ansible_port: '{{ new_port }}'
  76 
  77 - name: Ensure inventory vars are set to the same value as the inventory_hostname has (close to pre Ansible 2.4 behaviour)
  78   add_host:
  79     hostname: charlie
  80     inventory_dir: '{{ inventory_dir }}'
  81 
  82 - name: Add all hosts running this playbook to the done group
  83   add_host:
  84     name: '{{ item }}'
  85     groups: done
  86   loop: "{{ ansible_play_hosts }}"
  87 '''
