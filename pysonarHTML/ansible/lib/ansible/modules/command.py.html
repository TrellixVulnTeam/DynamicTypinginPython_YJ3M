 214 def check_command(module, commandline):
 215     arguments = {'chown': 'owner', 'chmod': 'mode', 'chgrp': 'group',
 216                  'ln': 'state=link', 'mkdir': 'state=directory',
 217                  'rmdir': 'state=absent', 'rm': 'state=absent', 'touch': 'state=touch'}
 218     commands = {'curl': 'get_url or uri', 'wget': 'get_url or uri',
 219                 'svn': 'subversion', 'service': 'service',
 220                 'mount': 'mount', 'rpm': 'yum, dnf or zypper', 'yum': 'yum', 'apt-get': 'apt',
 221                 'tar': 'unarchive', 'unzip': 'unarchive', 'sed': 'replace, lineinfile or template',
 222                 'dnf': 'dnf', 'zypper': 'zypper'}
 223     become = ['sudo', 'su', 'pbrun', 'pfexec', 'runas', 'pmrun', 'machinectl']
 224     if isinstance(commandline, list):
 225         command = commandline[0]
 226     else:
 227         command = commandline.split()[0]
 228     command = os.path.basename(command)
 229 
 230     disable_suffix = "If you need to use command because {mod} is insufficient you can add" \
 231                      " 'warn: false' to this command task or set 'command_warnings=False' in" \
 232                      " ansible.cfg to get rid of this message."
 233     substitutions = {'mod': None, 'cmd': command}
 234 
 235     if command in arguments:
 236         msg = "Consider using the {mod} module with {subcmd} rather than running '{cmd}'.  " + disable_suffix
 237         substitutions['mod'] = 'file'
 238         substitutions['subcmd'] = arguments[command]
 239         module.warn(msg.format(**substitutions))
 240 
 241     if command in commands:
 242         msg = "Consider using the {mod} module rather than running '{cmd}'.  " + disable_suffix
 243         substitutions['mod'] = commands[command]
 244         module.warn(msg.format(**substitutions))
 245 
 246     if command in become:
 247         module.warn("Consider using 'become', 'become_method', and 'become_user' rather than running %s" % (command,))

 250 def main():
 251 
 252     # the command module is the one ansible module that does not take key=value args
 253     # hence don't copy this one if you are looking to build others!
 254     module = AnsibleModule(
 255         argument_spec=dict(
 256             _raw_params=dict(),
 257             _uses_shell=dict(type='bool', default=False),
 258             argv=dict(type='list'),
 259             chdir=dict(type='path'),
 260             executable=dict(),
 261             creates=dict(type='path'),
 262             removes=dict(type='path'),
 263             # The default for this really comes from the action plugin
 264             warn=dict(type='bool', default=False, removed_in_version='2.14', removed_from_collection='ansible.builtin'),
 265             stdin=dict(required=False),
 266             stdin_add_newline=dict(type='bool', default=True),
 267             strip_empty_ends=dict(type='bool', default=True),
 268         ),
 269         supports_check_mode=True,
 270     )
 271     shell = module.params['_uses_shell']
 272     chdir = module.params['chdir']
 273     executable = module.params['executable']
 274     args = module.params['_raw_params']
 275     argv = module.params['argv']
 276     creates = module.params['creates']
 277     removes = module.params['removes']
 278     warn = module.params['warn']
 279     stdin = module.params['stdin']
 280     stdin_add_newline = module.params['stdin_add_newline']
 281     strip = module.params['strip_empty_ends']
 282 
 283     if not shell and executable:
 284         module.warn("As of Ansible 2.4, the parameter 'executable' is no longer supported with the 'command' module. Not using '%s'." % executable)
 285         executable = None
 286 
 287     if (not args or args.strip() == '') and not argv:
 288         module.fail_json(rc=256, msg="no command given")
 289 
 290     if args and argv:
 291         module.fail_json(rc=256, msg="only command or argv can be given, not both")
 292 
 293     if not shell and args:
 294         args = shlex.split(args)
 295 
 296     args = args or argv
 297 
 298     # All args must be strings
 299     if is_iterable(args, include_strings=False):
 300         args = [to_native(arg, errors='surrogate_or_strict', nonstring='simplerepr') for arg in args]
 301 
 302     if chdir:
 303         try:
 304             chdir = to_bytes(os.path.abspath(chdir), errors='surrogate_or_strict')
 305         except ValueError as e:
 306             module.fail_json(msg='Unable to use supplied chdir: %s' % to_text(e))
 307 
 308         try:
 309             os.chdir(chdir)
 310         except (IOError, OSError) as e:
 311             module.fail_json(msg='Unable to change directory before execution: %s' % to_text(e))
 312 
 313     if creates:
 314         # do not run the command if the line contains creates=filename
 315         # and the filename already exists.  This allows idempotence
 316         # of command executions.
 317         if glob.glob(creates):
 318             module.exit_json(
 319                 cmd=args,
 320                 stdout="skipped, since %s exists" % creates,
 321                 changed=False,
 322                 rc=0
 323             )
 324 
 325     if removes:
 326         # do not run the command if the line contains removes=filename
 327         # and the filename does not exist.  This allows idempotence
 328         # of command executions.
 329         if not glob.glob(removes):
 330             module.exit_json(
 331                 cmd=args,
 332                 stdout="skipped, since %s does not exist" % removes,
 333                 changed=False,
 334                 rc=0
 335             )
 336 
 337     if warn:
 338         check_command(module, args)
 339 
 340     startd = datetime.datetime.now()
 341 
 342     if not module.check_mode:
 343         rc, out, err = module.run_command(args, executable=executable, use_unsafe_shell=shell, encoding=None, data=stdin, binary_data=(not stdin_add_newline))
 344     elif creates or removes:
 345         rc = 0
 346         out = err = b'Command would have run if not in check mode'
 347     else:
 348         module.exit_json(msg="skipped, running in check mode", skipped=True)
 349 
 350     endd = datetime.datetime.now()
 351     delta = endd - startd
 352 
 353     if strip:
 354         out = out.rstrip(b"\r\n")
 355         err = err.rstrip(b"\r\n")
 356 
 357     result = dict(
 358         cmd=args,
 359         stdout=out,
 360         stderr=err,
 361         rc=rc,
 362         start=str(startd),
 363         end=str(endd),
 364         delta=str(delta),
 365         changed=True,
 366     )
 367 
 368     if rc != 0:
 369         module.fail_json(msg='non-zero return code', **result)
 370 
 371     module.exit_json(**result)
 372 
 373 
 374 if __name__ == '__main__':
 375     main()
