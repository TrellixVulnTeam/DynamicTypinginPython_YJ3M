  74   stat:

  81 # Determine if a path exists and is a symlink. Note that if the path does
  82 # not exist, and we test sym.stat.islnk, it will fail with an error. So
  83 # therefore, we must test whether it is defined.
  84 # Run this to understand the structure, the skipped ones do not pass the
  85 # check performed by 'when'
  86 - name: Get stats of the FS object
  87   stat:
  88     path: /path/to/something
  89   register: sym
  90 
  91 - debug:
  92     msg: "islnk isn't defined (path doesn't exist)"
  93   when: sym.stat.islnk is not defined
  94 
  95 - debug:
  96     msg: "islnk is defined (path must exist)"
  97   when: sym.stat.islnk is defined
  98 
  99 - debug:
 100     msg: "Path exists and is a symlink"
 101   when: sym.stat.islnk is defined and sym.stat.islnk

 103 - debug:
 104     msg: "Path exists and isn't a symlink"

 108 # Determine if a path exists and is a directory.  Note that we need to test

 115     msg: "Path exists and is a directory"

 370 def format_output(module, path, st):
 371     mode = st.st_mode
 372 
 373     # back to ansible
 374     output = dict(
 375         exists=True,
 376         path=path,
 377         mode="%04o" % stat.S_IMODE(mode),
 378         isdir=stat.S_ISDIR(mode),
 379         ischr=stat.S_ISCHR(mode),
 380         isblk=stat.S_ISBLK(mode),
 381         isreg=stat.S_ISREG(mode),
 382         isfifo=stat.S_ISFIFO(mode),
 383         islnk=stat.S_ISLNK(mode),
 384         issock=stat.S_ISSOCK(mode),
 385         uid=st.st_uid,
 386         gid=st.st_gid,
 387         size=st.st_size,
 388         inode=st.st_ino,
 389         dev=st.st_dev,
 390         nlink=st.st_nlink,
 391         atime=st.st_atime,
 392         mtime=st.st_mtime,
 393         ctime=st.st_ctime,
 394         wusr=bool(mode & stat.S_IWUSR),
 395         rusr=bool(mode & stat.S_IRUSR),
 396         xusr=bool(mode & stat.S_IXUSR),
 397         wgrp=bool(mode & stat.S_IWGRP),
 398         rgrp=bool(mode & stat.S_IRGRP),
 399         xgrp=bool(mode & stat.S_IXGRP),
 400         woth=bool(mode & stat.S_IWOTH),
 401         roth=bool(mode & stat.S_IROTH),
 402         xoth=bool(mode & stat.S_IXOTH),
 403         isuid=bool(mode & stat.S_ISUID),
 404         isgid=bool(mode & stat.S_ISGID),
 405     )
 406 
 407     # Platform dependent flags:
 408     for other in [
 409             # Some Linux
 410             ('st_blocks', 'blocks'),
 411             ('st_blksize', 'block_size'),
 412             ('st_rdev', 'device_type'),
 413             ('st_flags', 'flags'),
 414             # Some Berkley based
 415             ('st_gen', 'generation'),
 416             ('st_birthtime', 'birthtime'),
 417             # RISCOS
 418             ('st_ftype', 'file_type'),
 419             ('st_attrs', 'attrs'),
 420             ('st_obtype', 'object_type'),
 421             # macOS
 422             ('st_rsize', 'real_size'),
 423             ('st_creator', 'creator'),
 424             ('st_type', 'file_type'),
 425     ]:
 426         if hasattr(st, other[0]):
 427             output[other[1]] = getattr(st, other[0])
 428 
 429     return output

 432 def main():
 433     module = AnsibleModule(
 434         argument_spec=dict(
 435             path=dict(type='path', required=True, aliases=['dest', 'name']),
 436             follow=dict(type='bool', default=False),
 437             get_md5=dict(type='bool', default=False),
 438             get_checksum=dict(type='bool', default=True),
 439             get_mime=dict(type='bool', default=True, aliases=['mime', 'mime_type', 'mime-type']),
 440             get_attributes=dict(type='bool', default=True, aliases=['attr', 'attributes']),
 441             checksum_algorithm=dict(type='str', default='sha1',
 442                                     choices=['md5', 'sha1', 'sha224', 'sha256', 'sha384', 'sha512'],
 443                                     aliases=['checksum', 'checksum_algo']),
 444         ),
 445         supports_check_mode=True,
 446     )
 447 
 448     path = module.params.get('path')
 449     b_path = to_bytes(path, errors='surrogate_or_strict')
 450     follow = module.params.get('follow')
 451     get_mime = module.params.get('get_mime')
 452     get_attr = module.params.get('get_attributes')
 453     get_checksum = module.params.get('get_checksum')
 454     checksum_algorithm = module.params.get('checksum_algorithm')
 455 
 456     # NOTE: undocumented option since 2.9 to be removed at a later date if possible (3.0+)
 457     # no real reason for keeping other than fear we may break older content.
 458     get_md5 = module.params.get('get_md5')
 459 
 460     # main stat data
 461     try:
 462         if follow:
 463             st = os.stat(b_path)
 464         else:
 465             st = os.lstat(b_path)
 466     except OSError as e:
 467         if e.errno == errno.ENOENT:
 468             output = {'exists': False}
 469             module.exit_json(changed=False, stat=output)
 470 
 471         module.fail_json(msg=e.strerror)
 472 
 473     # process base results
 474     output = format_output(module, path, st)
 475 
 476     # resolved permissions
 477     for perm in [('readable', os.R_OK), ('writeable', os.W_OK), ('executable', os.X_OK)]:
 478         output[perm[0]] = os.access(b_path, perm[1])
 479 
 480     # symlink info
 481     if output.get('islnk'):
 482         output['lnk_source'] = os.path.realpath(b_path)
 483         output['lnk_target'] = os.readlink(b_path)
 484 
 485     try:  # user data
 486         pw = pwd.getpwuid(st.st_uid)
 487         output['pw_name'] = pw.pw_name
 488     except (TypeError, KeyError):
 489         pass
 490 
 491     try:  # group data
 492         grp_info = grp.getgrgid(st.st_gid)
 493         output['gr_name'] = grp_info.gr_name
 494     except (KeyError, ValueError, OverflowError):
 495         pass
 496 
 497     # checksums
 498     if output.get('isreg') and output.get('readable'):
 499 
 500         # NOTE: see above about get_md5
 501         if get_md5:
 502             # Will fail on FIPS-140 compliant systems
 503             try:
 504                 output['md5'] = module.md5(b_path)
 505             except ValueError:
 506                 output['md5'] = None
 507 
 508         if get_checksum:
 509             output['checksum'] = module.digest_from_file(b_path, checksum_algorithm)
 510 
 511     # try to get mime data if requested
 512     if get_mime:
 513         output['mimetype'] = output['charset'] = 'unknown'
 514         mimecmd = module.get_bin_path('file')
 515         if mimecmd:
 516             mimecmd = [mimecmd, '--mime-type', '--mime-encoding', b_path]
 517             try:
 518                 rc, out, err = module.run_command(mimecmd)
 519                 if rc == 0:
 520                     mimetype, charset = out.rsplit(':', 1)[1].split(';')
 521                     output['mimetype'] = mimetype.strip()
 522                     output['charset'] = charset.split('=')[1].strip()
 523             except Exception:
 524                 pass
 525 
 526     # try to get attr data
 527     if get_attr:
 528         output['version'] = None
 529         output['attributes'] = []
 530         output['attr_flags'] = ''
 531         out = module.get_file_attributes(b_path)
 532         for x in ('version', 'attributes', 'attr_flags'):
 533             if x in out:
 534                 output[x] = out[x]
 535 
 536     module.exit_json(changed=False, stat=output)
 537 
 538 
 539 if __name__ == '__main__':
 540     main()
