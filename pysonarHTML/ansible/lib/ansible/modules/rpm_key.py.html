  78 def is_pubkey(string):
  79     """Verifies if string is a pubkey"""
  80     pgp_regex = ".*?(-----BEGIN PGP PUBLIC KEY BLOCK-----.*?-----END PGP PUBLIC KEY BLOCK-----).*"
  81     return bool(re.match(pgp_regex, to_native(string, errors='surrogate_or_strict'), re.DOTALL))

  84 class RpmKey(object):
  85 
  86     def __init__(self, module):
  87         # If the key is a url, we need to check if it's present to be idempotent,
  88         # to do that, we need to check the keyid, which we can get from the armor.
  89         keyfile = None
  90         should_cleanup_keyfile = False
  91         self.module = module
  92         self.rpm = self.module.get_bin_path('rpm', True)
  93         state = module.params['state']
  94         key = module.params['key']
  95         fingerprint = module.params['fingerprint']
  96         if fingerprint:
  97             fingerprint = fingerprint.replace(' ', '').upper()
  98 
  99         self.gpg = self.module.get_bin_path('gpg')
 100         if not self.gpg:
 101             self.gpg = self.module.get_bin_path('gpg2', required=True)
 102 
 103         if '://' in key:
 104             keyfile = self.fetch_key(key)
 105             keyid = self.getkeyid(keyfile)
 106             should_cleanup_keyfile = True
 107         elif self.is_keyid(key):
 108             keyid = key
 109         elif os.path.isfile(key):
 110             keyfile = key
 111             keyid = self.getkeyid(keyfile)
 112         else:
 113             self.module.fail_json(msg="Not a valid key %s" % key)
 114         keyid = self.normalize_keyid(keyid)
 115 
 116         if state == 'present':
 117             if self.is_key_imported(keyid):
 118                 module.exit_json(changed=False)
 119             else:
 120                 if not keyfile:
 121                     self.module.fail_json(msg="When importing a key, a valid file must be given")
 122                 if fingerprint:
 123                     has_fingerprint = self.getfingerprint(keyfile)
 124                     if fingerprint != has_fingerprint:
 125                         self.module.fail_json(
 126                             msg="The specified fingerprint, '%s', does not match the key fingerprint '%s'" % (fingerprint, has_fingerprint)
 127                         )
 128                 self.import_key(keyfile)
 129                 if should_cleanup_keyfile:
 130                     self.module.cleanup(keyfile)
 131                 module.exit_json(changed=True)
 132         else:
 133             if self.is_key_imported(keyid):
 134                 self.drop_key(keyid)
 135                 module.exit_json(changed=True)
 136             else:
 137                 module.exit_json(changed=False)

 139     def fetch_key(self, url):
 140         """Downloads a key from url, returns a valid path to a gpg key"""
 141         rsp, info = fetch_url(self.module, url)
 142         if info['status'] != 200:
 143             self.module.fail_json(msg="failed to fetch key at %s , error was: %s" % (url, info['msg']))
 144 
 145         key = rsp.read()
 146         if not is_pubkey(key):
 147             self.module.fail_json(msg="Not a public key: %s" % url)
 148         tmpfd, tmpname = tempfile.mkstemp()
 149         self.module.add_cleanup_file(tmpname)
 150         tmpfile = os.fdopen(tmpfd, "w+b")
 151         tmpfile.write(key)
 152         tmpfile.close()
 153         return tmpname
 154 
 155     def normalize_keyid(self, keyid):
 156         """Ensure a keyid doesn't have a leading 0x, has leading or trailing whitespace, and make sure is uppercase"""
 157         ret = keyid.strip().upper()
 158         if ret.startswith('0x'):
 159             return ret[2:]
 160         elif ret.startswith('0X'):
 161             return ret[2:]
 162         else:
 163             return ret
 164 
 165     def getkeyid(self, keyfile):
 166         stdout, stderr = self.execute_command([self.gpg, '--no-tty', '--batch', '--with-colons', '--fixed-list-mode', keyfile])
 167         for line in stdout.splitlines():
 168             line = line.strip()
 169             if line.startswith('pub:'):
 170                 return line.split(':')[4]
 171 
 172         self.module.fail_json(msg="Unexpected gpg output")
 173 
 174     def getfingerprint(self, keyfile):
 175         stdout, stderr = self.execute_command([
 176             self.gpg, '--no-tty', '--batch', '--with-colons',
 177             '--fixed-list-mode', '--with-fingerprint', keyfile
 178         ])
 179         for line in stdout.splitlines():
 180             line = line.strip()
 181             if line.startswith('fpr:'):
 182                 # As mentioned here,
 183                 #
 184                 # https://git.gnupg.org/cgi-bin/gitweb.cgi?p=gnupg.git;a=blob_plain;f=doc/DETAILS
 185                 #
 186                 # The description of the `fpr` field says
 187                 #
 188                 # "fpr :: Fingerprint (fingerprint is in field 10)"
 189                 #
 190                 return line.split(':')[9]
 191 
 192         self.module.fail_json(msg="Unexpected gpg output")
 193 
 194     def is_keyid(self, keystr):
 195         """Verifies if a key, as provided by the user is a keyid"""
 196         return re.match('(0x)?[0-9a-f]{8}', keystr, flags=re.IGNORECASE)
 197 
 198     def execute_command(self, cmd):
 199         rc, stdout, stderr = self.module.run_command(cmd, use_unsafe_shell=True)
 200         if rc != 0:
 201             self.module.fail_json(msg=stderr)
 202         return stdout, stderr
 203 
 204     def is_key_imported(self, keyid):
 205         cmd = self.rpm + ' -q  gpg-pubkey'
 206         rc, stdout, stderr = self.module.run_command(cmd)
 207         if rc != 0:  # No key is installed on system
 208             return False
 209         cmd += ' --qf "%{description}" | ' + self.gpg + ' --no-tty --batch --with-colons --fixed-list-mode -'
 210         stdout, stderr = self.execute_command(cmd)
 211         for line in stdout.splitlines():
 212             if keyid in line.split(':')[4]:
 213                 return True
 214         return False
 215 
 216     def import_key(self, keyfile):
 217         if not self.module.check_mode:
 218             self.execute_command([self.rpm, '--import', keyfile])
 219 
 220     def drop_key(self, keyid):
 221         if not self.module.check_mode:
 222             self.execute_command([self.rpm, '--erase', '--allmatches', "gpg-pubkey-%s" % keyid[-8:].lower()])
