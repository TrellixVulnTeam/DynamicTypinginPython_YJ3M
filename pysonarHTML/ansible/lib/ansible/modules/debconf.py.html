  91 def get_selections(module, pkg):
  92     cmd = [module.get_bin_path('debconf-show', True), pkg]
  93     rc, out, err = module.run_command(' '.join(cmd))
  94 
  95     if rc != 0:
  96         module.fail_json(msg=err)
  97 
  98     selections = {}
  99 
 100     for line in out.splitlines():
 101         (key, value) = line.split(':', 1)
 102         selections[key.strip('*').strip()] = value.strip()
 103 
 104     return selections
 105 
 106 
 107 def set_selection(module, pkg, question, vtype, value, unseen):
 108     setsel = module.get_bin_path('debconf-set-selections', True)
 109     cmd = [setsel]
 110     if unseen:
 111         cmd.append('-u')
 112 
 113     if vtype == 'boolean':
 114         if value == 'True':
 115             value = 'true'
 116         elif value == 'False':
 117             value = 'false'
 118     data = ' '.join([pkg, question, vtype, value])
 119 
 120     return module.run_command(cmd, data=data)

 123 def main():
 124     module = AnsibleModule(
 125         argument_spec=dict(
 126             name=dict(type='str', required=True, aliases=['pkg']),
 127             question=dict(type='str', aliases=['selection', 'setting']),
 128             vtype=dict(type='str', choices=['boolean', 'error', 'multiselect', 'note', 'password', 'seen', 'select', 'string', 'text', 'title']),
 129             value=dict(type='str', aliases=['answer']),
 130             unseen=dict(type='bool'),
 131         ),
 132         required_together=(['question', 'vtype', 'value'],),
 133         supports_check_mode=True,
 134     )
 135 
 136     # TODO: enable passing array of options and/or debconf file from get-selections dump
 137     pkg = module.params["name"]
 138     question = module.params["question"]
 139     vtype = module.params["vtype"]
 140     value = module.params["value"]
 141     unseen = module.params["unseen"]
 142 
 143     prev = get_selections(module, pkg)
 144 
 145     changed = False
 146     msg = ""
 147 
 148     if question is not None:
 149         if vtype is None or value is None:
 150             module.fail_json(msg="when supplying a question you must supply a valid vtype and value")
 151 
 152         # if question doesn't exist, value cannot match
 153         if question not in prev:
 154             changed = True
 155         else:
 156 
 157             existing = prev[question]
 158 
 159             # ensure we compare booleans supplied to the way debconf sees them (true/false strings)
 160             if vtype == 'boolean':
 161                 value = to_text(value).lower()
 162                 existing = to_text(prev[question]).lower()
 163 
 164             if value != existing:
 165                 changed = True
 166 
 167     if changed:
 168         if not module.check_mode:
 169             rc, msg, e = set_selection(module, pkg, question, vtype, value, unseen)
 170             if rc:
 171                 module.fail_json(msg=e)
 172 
 173         curr = {question: value}
 174         if question in prev:
 175             prev = {question: prev[question]}
 176         else:
 177             prev[question] = ''
 178         if module._diff:
 179             after = prev.copy()
 180             after.update(curr)
 181             diff_dict = {'before': prev, 'after': after}
 182         else:
 183             diff_dict = {}
 184 
 185         module.exit_json(changed=changed, msg=msg, current=curr, previous=prev, diff=diff_dict)
 186 
 187     module.exit_json(changed=changed, msg=msg, current=prev)
 188 
 189 
 190 if __name__ == '__main__':
 191     main()
