  97 class Hostname(object):
  98     """
  99     This is a generic Hostname manipulation class that is subclassed
 100     based on platform.
 101 
 102     A subclass may wish to set different strategy instance to self.strategy.
 103 
 104     All subclasses MUST define platform and distribution (which may be None).
 105     """
 106 
 107     platform = 'Generic'
 108     distribution = None
 109     strategy_class = UnimplementedStrategy
 110 
 111     def __new__(cls, *args, **kwargs):
 112         new_cls = get_platform_subclass(Hostname)
 113         return super(cls, new_cls).__new__(new_cls)
 114 
 115     def __init__(self, module):
 116         self.module = module
 117         self.name = module.params['name']
 118         self.use = module.params['use']
 119 
 120         if self.use is not None:
 121             strat = globals()['%sStrategy' % STRATS[self.use]]
 122             self.strategy = strat(module)
 123         elif self.platform == 'Linux' and ServiceMgrFactCollector.is_systemd_managed(module):
 124             self.strategy = SystemdStrategy(module)
 125         else:
 126             self.strategy = self.strategy_class(module)
 127 
 128     def update_current_and_permanent_hostname(self):
 129         return self.strategy.update_current_and_permanent_hostname()
 130 
 131     def get_current_hostname(self):
 132         return self.strategy.get_current_hostname()
 133 
 134     def set_current_hostname(self, name):
 135         self.strategy.set_current_hostname(name)
 136 
 137     def get_permanent_hostname(self):
 138         return self.strategy.get_permanent_hostname()
 139 
 140     def set_permanent_hostname(self, name):
 141         self.strategy.set_permanent_hostname(name)

 144 class GenericStrategy(object):
 145     """
 146     This is a generic Hostname manipulation strategy class.
 147 
 148     A subclass may wish to override some or all of these methods.
 149       - get_current_hostname()
 150       - get_permanent_hostname()
 151       - set_current_hostname(name)
 152       - set_permanent_hostname(name)
 153     """
 154 
 155     def __init__(self, module):
 156         self.module = module
 157         self.changed = False
 158         self.hostname_cmd = self.module.get_bin_path('hostname', True)
 159 
 160     def update_current_and_permanent_hostname(self):
 161         self.update_current_hostname()
 162         self.update_permanent_hostname()
 163         return self.changed
 164 
 165     def update_current_hostname(self):
 166         name = self.module.params['name']
 167         current_name = self.get_current_hostname()
 168         if current_name != name:
 169             if not self.module.check_mode:
 170                 self.set_current_hostname(name)
 171             self.changed = True
 172 
 173     def update_permanent_hostname(self):
 174         name = self.module.params['name']
 175         permanent_name = self.get_permanent_hostname()
 176         if permanent_name != name:
 177             if not self.module.check_mode:
 178                 self.set_permanent_hostname(name)
 179             self.changed = True
 180 
 181     def get_current_hostname(self):
 182         cmd = [self.hostname_cmd]
 183         rc, out, err = self.module.run_command(cmd)
 184         if rc != 0:
 185             self.module.fail_json(msg="Command failed rc=%d, out=%s, err=%s" % (rc, out, err))
 186         return to_native(out).strip()
 187 
 188     def set_current_hostname(self, name):
 189         cmd = [self.hostname_cmd, name]
 190         rc, out, err = self.module.run_command(cmd)
 191         if rc != 0:
 192             self.module.fail_json(msg="Command failed rc=%d, out=%s, err=%s" % (rc, out, err))
 193 
 194     def get_permanent_hostname(self):
 195         return 'UNKNOWN'
 196 
 197     def set_permanent_hostname(self, name):
 198         pass

 201 class DebianStrategy(GenericStrategy):
 202     """
 203     This is a Debian family Hostname manipulation strategy class - it edits
 204     the /etc/hostname file.
 205     """
 206 
 207     HOSTNAME_FILE = '/etc/hostname'
 208 
 209     def get_permanent_hostname(self):
 210         if not os.path.isfile(self.HOSTNAME_FILE):
 211             try:
 212                 open(self.HOSTNAME_FILE, "a").write("")
 213             except IOError as e:
 214                 self.module.fail_json(msg="failed to write file: %s" %
 215                                           to_native(e), exception=traceback.format_exc())
 216         try:
 217             f = open(self.HOSTNAME_FILE)
 218             try:
 219                 return f.read().strip()
 220             finally:
 221                 f.close()
 222         except Exception as e:
 223             self.module.fail_json(msg="failed to read hostname: %s" %
 224                                       to_native(e), exception=traceback.format_exc())
 225 
 226     def set_permanent_hostname(self, name):
 227         try:
 228             f = open(self.HOSTNAME_FILE, 'w+')
 229             try:

 231             finally:
 232                 f.close()
 233         except Exception as e:
 234             self.module.fail_json(msg="failed to update hostname: %s" %
 235                                       to_native(e), exception=traceback.format_exc())

 274 class RedHatStrategy(GenericStrategy):
 275     """
 276     This is a Redhat Hostname strategy class - it edits the
 277     /etc/sysconfig/network file.
 278     """
 279     NETWORK_FILE = '/etc/sysconfig/network'
 280 
 281     def get_permanent_hostname(self):
 282         try:
 283             f = open(self.NETWORK_FILE, 'rb')
 284             try:
 285                 for line in f.readlines():
 286                     if line.startswith('HOSTNAME'):
 287                         k, v = line.split('=')
 288                         return v.strip()
 289             finally:
 290                 f.close()
 291         except Exception as e:
 292             self.module.fail_json(msg="failed to read hostname: %s" %
 293                                       to_native(e), exception=traceback.format_exc())
 294 
 295     def set_permanent_hostname(self, name):
 296         try:
 297             lines = []
 298             found = False
 299             f = open(self.NETWORK_FILE, 'rb')
 300             try:
 301                 for line in f.readlines():
 302                     if line.startswith('HOSTNAME'):
 303                         lines.append("HOSTNAME=%s\n" % name)
 304                         found = True
 305                     else:
 306                         lines.append(line)
 307             finally:
 308                 f.close()
 309             if not found:
 310                 lines.append("HOSTNAME=%s\n" % name)
 311             f = open(self.NETWORK_FILE, 'w+')
 312             try:
 313                 f.writelines(lines)
 314             finally:
 315                 f.close()
 316         except Exception as e:
 317             self.module.fail_json(msg="failed to update hostname: %s" %
 318                                       to_native(e), exception=traceback.format_exc())

 414 class OpenRCStrategy(GenericStrategy):
 415     """
 416     This is a Gentoo (OpenRC) Hostname manipulation strategy class - it edits
 417     the /etc/conf.d/hostname file.
 418     """
 419 
 420     HOSTNAME_FILE = '/etc/conf.d/hostname'
 421 
 422     def get_permanent_hostname(self):
 423         name = 'UNKNOWN'
 424         try:
 425             try:
 426                 f = open(self.HOSTNAME_FILE, 'r')
 427                 for line in f:
 428                     line = line.strip()
 429                     if line.startswith('hostname='):
 430                         name = line[10:].strip('"')
 431                         break
 432             except Exception as e:
 433                 self.module.fail_json(msg="failed to read hostname: %s" %
 434                                           to_native(e), exception=traceback.format_exc())
 435         finally:
 436             f.close()
 437 
 438         return name
 439 
 440     def set_permanent_hostname(self, name):
 441         try:
 442             try:
 443                 f = open(self.HOSTNAME_FILE, 'r')
 444                 lines = [x.strip() for x in f]
 445 
 446                 for i, line in enumerate(lines):
 447                     if line.startswith('hostname='):
 448                         lines[i] = 'hostname="%s"' % name
 449                         break
 450                 f.close()
 451 
 452                 f = open(self.HOSTNAME_FILE, 'w')
 453                 f.write('\n'.join(lines) + '\n')
 454             except Exception as e:
 455                 self.module.fail_json(msg="failed to update hostname: %s" %
 456                                           to_native(e), exception=traceback.format_exc())
 457         finally:
 458             f.close()

 512         fmri = 'svc:/system/identity:node'

 527 class FreeBSDStrategy(GenericStrategy):
 528     """
 529     This is a FreeBSD hostname manipulation strategy class - it edits
 530     the /etc/rc.conf.d/hostname file.
 531     """
 532 
 533     HOSTNAME_FILE = '/etc/rc.conf.d/hostname'
 534 
 535     def get_permanent_hostname(self):
 536 
 537         name = 'UNKNOWN'
 538         if not os.path.isfile(self.HOSTNAME_FILE):
 539             try:
 540                 open(self.HOSTNAME_FILE, "a").write("hostname=temporarystub\n")
 541             except IOError as e:
 542                 self.module.fail_json(msg="failed to write file: %s" %
 543                                           to_native(e), exception=traceback.format_exc())
 544         try:
 545             try:
 546                 f = open(self.HOSTNAME_FILE, 'r')
 547                 for line in f:
 548                     line = line.strip()
 549                     if line.startswith('hostname='):
 550                         name = line[10:].strip('"')
 551                         break
 552             except Exception as e:
 553                 self.module.fail_json(msg="failed to read hostname: %s" %
 554                                           to_native(e), exception=traceback.format_exc())
 555         finally:
 556             f.close()
 557 
 558         return name
 559 
 560     def set_permanent_hostname(self, name):
 561         try:
 562             try:
 563                 f = open(self.HOSTNAME_FILE, 'r')
 564                 lines = [x.strip() for x in f]
 565 
 566                 for i, line in enumerate(lines):
 567                     if line.startswith('hostname='):
 568                         lines[i] = 'hostname="%s"' % name
 569                         break
 570                 f.close()
 571 
 572                 f = open(self.HOSTNAME_FILE, 'w')
 573                 f.write('\n'.join(lines) + '\n')
 574             except Exception as e:
 575                 self.module.fail_json(msg="failed to update hostname: %s" %
 576                                           to_native(e), exception=traceback.format_exc())
 577         finally:
 578             f.close()

 851

 853         name_before = current_hostname
 854     elif name != permanent_hostname:

 859                                  ansible_nodename=name,

 861                                  ansible_domain='.'.join(socket.getfqdn().split('.')[1:])))

 864         kw['diff'] = {'after': 'hostname = ' + name + '\n',
 865                       'before': 'hostname = ' + name_before + '\n'}

 867     module.exit_json(**kw)
