 260 def additional_parameter_handling(params):
 261     """Additional parameter validation and reformatting"""
 262     # When path is a directory, rewrite the pathname to be the file inside of the directory
 263     # TODO: Why do we exclude link?  Why don't we exclude directory?  Should we exclude touch?
 264     # I think this is where we want to be in the future:
 265     # when isdir(path):
 266     # if state == absent:  Remove the directory
 267     # if state == touch:   Touch the directory
 268     # if state == directory: Assert the directory is the same as the one specified
 269     # if state == file:    place inside of the directory (use _original_basename)
 270     # if state == link:    place inside of the directory (use _original_basename.  Fallback to src?)
 271     # if state == hard:    place inside of the directory (use _original_basename.  Fallback to src?)
 272     if (params['state'] not in ("link", "absent") and os.path.isdir(to_bytes(params['path'], errors='surrogate_or_strict'))):
 273         basename = None
 274 
 275         if params['_original_basename']:
 276             basename = params['_original_basename']
 277         elif params['src']:
 278             basename = os.path.basename(params['src'])
 279 
 280         if basename:
 281             params['path'] = os.path.join(params['path'], basename)
 282 
 283     # state should default to file, but since that creates many conflicts,
 284     # default state to 'current' when it exists.
 285     prev_state = get_state(to_bytes(params['path'], errors='surrogate_or_strict'))
 286 
 287     if params['state'] is None:
 288         if prev_state != 'absent':
 289             params['state'] = prev_state
 290         elif params['recurse']:
 291             params['state'] = 'directory'
 292         else:
 293             params['state'] = 'file'
 294 
 295     # make sure the target path is a directory when we're doing a recursive operation
 296     if params['recurse'] and params['state'] != 'directory':
 297         raise ParameterError(results={"msg": "recurse option requires state to be 'directory'",
 298                                       "path": params["path"]})
 299 
 300     # Fail if 'src' but no 'state' is specified
 301     if params['src'] and params['state'] not in ('link', 'hard'):
 302         raise ParameterError(results={'msg': "src option requires state to be 'link' or 'hard'",
 303                                       'path': params['path']})

 306 def get_state(path):
 307     ''' Find out current state '''
 308 
 309     b_path = to_bytes(path, errors='surrogate_or_strict')
 310     try:
 311         if os.path.lexists(b_path):
 312             if os.path.islink(b_path):
 313                 return 'link'
 314             elif os.path.isdir(b_path):
 315                 return 'directory'
 316             elif os.stat(b_path).st_nlink > 1:
 317                 return 'hard'
 318 
 319             # could be many other things, but defaulting to file
 320             return 'file'
 321 
 322         return 'absent'
 323     except OSError as e:
 324         if e.errno == errno.ENOENT:  # It may already have been removed
 325             return 'absent'
 326         else:
 327             raise

 331 def recursive_set_attributes(b_path, follow, file_args, mtime, atime):
 332     changed = False
 333 
 334     try:
 335         for b_root, b_dirs, b_files in os.walk(b_path):
 336             for b_fsobj in b_dirs + b_files:
 337                 b_fsname = os.path.join(b_root, b_fsobj)
 338                 if not os.path.islink(b_fsname):
 339                     tmp_file_args = file_args.copy()
 340                     tmp_file_args['path'] = to_native(b_fsname, errors='surrogate_or_strict')
 341                     changed |= module.set_fs_attributes_if_different(tmp_file_args, changed, expand=False)
 342                     changed |= update_timestamp_for_file(tmp_file_args['path'], mtime, atime)
 343 
 344                 else:
 345                     # Change perms on the link
 346                     tmp_file_args = file_args.copy()
 347                     tmp_file_args['path'] = to_native(b_fsname, errors='surrogate_or_strict')
 348                     changed |= module.set_fs_attributes_if_different(tmp_file_args, changed, expand=False)
 349                     changed |= update_timestamp_for_file(tmp_file_args['path'], mtime, atime)
 350 
 351                     if follow:
 352                         b_fsname = os.path.join(b_root, os.readlink(b_fsname))
 353                         # The link target could be nonexistent
 354                         if os.path.exists(b_fsname):
 355                             if os.path.isdir(b_fsname):
 356                                 # Link is a directory so change perms on the directory's contents
 357                                 changed |= recursive_set_attributes(b_fsname, follow, file_args, mtime, atime)
 358 
 359                             # Change perms on the file pointed to by the link
 360                             tmp_file_args = file_args.copy()
 361                             tmp_file_args['path'] = to_native(b_fsname, errors='surrogate_or_strict')
 362                             changed |= module.set_fs_attributes_if_different(tmp_file_args, changed, expand=False)
 363                             changed |= update_timestamp_for_file(tmp_file_args['path'], mtime, atime)
 364     except RuntimeError as e:
 365         # on Python3 "RecursionError" is raised which is derived from "RuntimeError"
 366         # TODO once this function is moved into the common file utilities, this should probably raise more general exception
 367         raise AnsibleModuleError(
 368             results={'msg': "Could not recursively set attributes on %s. Original error was: '%s'" % (to_native(b_path), to_native(e))}
 369         )
 370 
 371     return changed

 374 def initial_diff(path, state, prev_state):
 375     diff = {'before': {'path': path},
 376             'after': {'path': path},
 377             }
 378 
 379     if prev_state != state:
 380         diff['before']['state'] = prev_state
 381         diff['after']['state'] = state
 382         if state == 'absent' and prev_state == 'directory':
 383             walklist = {
 384                 'directories': [],
 385                 'files': [],
 386             }
 387             b_path = to_bytes(path, errors='surrogate_or_strict')
 388             for base_path, sub_folders, files in os.walk(b_path):
 389                 for folder in sub_folders:
 390                     folderpath = os.path.join(base_path, folder)
 391                     walklist['directories'].append(folderpath)
 392 
 393                 for filename in files:
 394                     filepath = os.path.join(base_path, filename)
 395                     walklist['files'].append(filepath)
 396 
 397             diff['before']['path_content'] = walklist
 398 
 399     return diff

 406 def get_timestamp_for_time(formatted_time, time_format):
 407     if formatted_time == 'preserve':
 408         return None
 409     elif formatted_time == 'now':
 410         return Sentinel
 411     else:
 412         try:
 413             struct = time.strptime(formatted_time, time_format)
 414             struct_time = time.mktime(struct)
 415         except (ValueError, OverflowError) as e:
 416             raise AnsibleModuleError(results={'msg': 'Error while obtaining timestamp for time %s using format %s: %s'
 417                                               % (formatted_time, time_format, to_native(e, nonstring='simplerepr'))})
 418 
 419         return struct_time

 422 def update_timestamp_for_file(path, mtime, atime, diff=None):
 423     b_path = to_bytes(path, errors='surrogate_or_strict')
 424 
 425     try:
 426         # When mtime and atime are set to 'now', rely on utime(path, None) which does not require ownership of the file
 427         # https://github.com/ansible/ansible/issues/50943
 428         if mtime is Sentinel and atime is Sentinel:
 429             # It's not exact but we can't rely on os.stat(path).st_mtime after setting os.utime(path, None) as it may
 430             # not be updated. Just use the current time for the diff values
 431             mtime = atime = time.time()
 432 
 433             previous_mtime = os.stat(b_path).st_mtime
 434             previous_atime = os.stat(b_path).st_atime
 435 
 436             set_time = None
 437         else:
 438             # If both parameters are None 'preserve', nothing to do
 439             if mtime is None and atime is None:
 440                 return False
 441 
 442             previous_mtime = os.stat(b_path).st_mtime
 443             previous_atime = os.stat(b_path).st_atime
 444 
 445             if mtime is None:
 446                 mtime = previous_mtime
 447             elif mtime is Sentinel:
 448                 mtime = time.time()
 449 
 450             if atime is None:
 451                 atime = previous_atime
 452             elif atime is Sentinel:
 453                 atime = time.time()
 454 
 455             # If both timestamps are already ok, nothing to do
 456             if mtime == previous_mtime and atime == previous_atime:
 457                 return False
 458 
 459             set_time = (atime, mtime)
 460 
 461         os.utime(b_path, set_time)
 462 
 463         if diff is not None:
 464             if 'before' not in diff:
 465                 diff['before'] = {}
 466             if 'after' not in diff:
 467                 diff['after'] = {}
 468             if mtime != previous_mtime:
 469                 diff['before']['mtime'] = previous_mtime
 470                 diff['after']['mtime'] = mtime
 471             if atime != previous_atime:
 472                 diff['before']['atime'] = previous_atime
 473                 diff['after']['atime'] = atime
 474     except OSError as e:
 475         raise AnsibleModuleError(results={'msg': 'Error while updating modification or access time: %s'
 476                                           % to_native(e, nonstring='simplerepr'), 'path': path})
 477     return True

 480 def keep_backward_compatibility_on_timestamps(parameter, state):
 481     if state in ['file', 'hard', 'directory', 'link'] and parameter is None:
 482         return 'preserve'
 483     elif state == 'touch' and parameter is None:
 484         return 'now'
 485     else:
 486         return parameter

 489 def execute_diff_peek(path):
 490     """Take a guess as to whether a file is a binary file"""
 491     b_path = to_bytes(path, errors='surrogate_or_strict')
 492     appears_binary = False
 493     try:
 494         with open(b_path, 'rb') as f:
 495             head = f.read(8192)
 496     except Exception:
 497         # If we can't read the file, we're okay assuming it's text
 498         pass
 499     else:
 500         if b"\x00" in head:
 501             appears_binary = True
 502 
 503     return appears_binary

 506 def ensure_absent(path):
 507     b_path = to_bytes(path, errors='surrogate_or_strict')
 508     prev_state = get_state(b_path)
 509     result = {}
 510 
 511     if prev_state != 'absent':
 512         diff = initial_diff(path, 'absent', prev_state)
 513 
 514         if not module.check_mode:
 515             if prev_state == 'directory':
 516                 try:
 517                     shutil.rmtree(b_path, ignore_errors=False)
 518                 except Exception as e:
 519                     raise AnsibleModuleError(results={'msg': "rmtree failed: %s" % to_native(e)})
 520             else:
 521                 try:
 522                     os.unlink(b_path)
 523                 except OSError as e:
 524                     if e.errno != errno.ENOENT:  # It may already have been removed
 525                         raise AnsibleModuleError(results={'msg': "unlinking failed: %s " % to_native(e),
 526                                                           'path': path})
 527 
 528         result.update({'path': path, 'changed': True, 'diff': diff, 'state': 'absent'})
 529     else:
 530         result.update({'path': path, 'changed': False, 'state': 'absent'})
 531 
 532     return result

 535 def execute_touch(path, follow, timestamps):
 536     b_path = to_bytes(path, errors='surrogate_or_strict')
 537     prev_state = get_state(b_path)
 538     changed = False
 539     result = {'dest': path}
 540     mtime = get_timestamp_for_time(timestamps['modification_time'], timestamps['modification_time_format'])
 541     atime = get_timestamp_for_time(timestamps['access_time'], timestamps['access_time_format'])
 542 
 543     if not module.check_mode:
 544         if prev_state == 'absent':
 545             # Create an empty file if the filename did not already exist
 546             try:
 547                 open(b_path, 'wb').close()
 548                 changed = True
 549             except (OSError, IOError) as e:
 550                 raise AnsibleModuleError(results={'msg': 'Error, could not touch target: %s'
 551                                                          % to_native(e, nonstring='simplerepr'),
 552                                                   'path': path})
 553 
 554         # Update the attributes on the file
 555         diff = initial_diff(path, 'touch', prev_state)
 556         file_args = module.load_file_common_arguments(module.params)
 557         try:
 558             changed = module.set_fs_attributes_if_different(file_args, changed, diff, expand=False)
 559             changed |= update_timestamp_for_file(file_args['path'], mtime, atime, diff)
 560         except SystemExit as e:
 561             if e.code:  # this is the exit code passed to sys.exit, not a constant -- pylint: disable=using-constant-test
 562                 # We take this to mean that fail_json() was called from
 563                 # somewhere in basic.py
 564                 if prev_state == 'absent':
 565                     # If we just created the file we can safely remove it
 566                     os.remove(b_path)
 567             raise
 568 
 569         result['changed'] = changed
 570         result['diff'] = diff
 571     return result

 574 def ensure_file_attributes(path, follow, timestamps):
 575     b_path = to_bytes(path, errors='surrogate_or_strict')
 576     prev_state = get_state(b_path)
 577     file_args = module.load_file_common_arguments(module.params)
 578     mtime = get_timestamp_for_time(timestamps['modification_time'], timestamps['modification_time_format'])
 579     atime = get_timestamp_for_time(timestamps['access_time'], timestamps['access_time_format'])
 580 
 581     if prev_state != 'file':
 582         if follow and prev_state == 'link':
 583             # follow symlink and operate on original
 584             b_path = os.path.realpath(b_path)
 585             path = to_native(b_path, errors='strict')
 586             prev_state = get_state(b_path)
 587             file_args['path'] = path
 588 
 589     if prev_state not in ('file', 'hard'):
 590         # file is not absent and any other state is a conflict
 591         raise AnsibleModuleError(results={'msg': 'file (%s) is %s, cannot continue' % (path, prev_state),
 592                                           'path': path, 'state': prev_state})
 593 
 594     diff = initial_diff(path, 'file', prev_state)
 595     changed = module.set_fs_attributes_if_different(file_args, False, diff, expand=False)
 596     changed |= update_timestamp_for_file(file_args['path'], mtime, atime, diff)
 597     return {'path': path, 'changed': changed, 'diff': diff}

 600 def ensure_directory(path, follow, recurse, timestamps):
 601     b_path = to_bytes(path, errors='surrogate_or_strict')
 602     prev_state = get_state(b_path)
 603     file_args = module.load_file_common_arguments(module.params)
 604     mtime = get_timestamp_for_time(timestamps['modification_time'], timestamps['modification_time_format'])
 605     atime = get_timestamp_for_time(timestamps['access_time'], timestamps['access_time_format'])
 606 
 607     # For followed symlinks, we need to operate on the target of the link
 608     if follow and prev_state == 'link':
 609         b_path = os.path.realpath(b_path)
 610         path = to_native(b_path, errors='strict')
 611         file_args['path'] = path
 612         prev_state = get_state(b_path)
 613 
 614     changed = False
 615     diff = initial_diff(path, 'directory', prev_state)
 616 
 617     if prev_state == 'absent':
 618         # Create directory and assign permissions to it
 619         if module.check_mode:
 620             return {'path': path, 'changed': True, 'diff': diff}
 621         curpath = ''
 622 
 623         try:
 624             # Split the path so we can apply filesystem attributes recursively
 625             # from the root (/) directory for absolute paths or the base path
 626             # of a relative path.  We can then walk the appropriate directory
 627             # path to apply attributes.
 628             # Something like mkdir -p with mode applied to all of the newly created directories
 629             for dirname in path.strip('/').split('/'):
 630                 curpath = '/'.join([curpath, dirname])
 631                 # Remove leading slash if we're creating a relative path
 632                 if not os.path.isabs(path):
 633                     curpath = curpath.lstrip('/')
 634                 b_curpath = to_bytes(curpath, errors='surrogate_or_strict')
 635                 if not os.path.exists(b_curpath):
 636                     try:
 637                         os.mkdir(b_curpath)
 638                         changed = True
 639                     except OSError as ex:
 640                         # Possibly something else created the dir since the os.path.exists
 641                         # check above. As long as it's a dir, we don't need to error out.
 642                         if not (ex.errno == errno.EEXIST and os.path.isdir(b_curpath)):
 643                             raise
 644                     tmp_file_args = file_args.copy()
 645                     tmp_file_args['path'] = curpath
 646                     changed = module.set_fs_attributes_if_different(tmp_file_args, changed, diff, expand=False)
 647                     changed |= update_timestamp_for_file(file_args['path'], mtime, atime, diff)
 648         except Exception as e:
 649             raise AnsibleModuleError(results={'msg': 'There was an issue creating %s as requested:'
 650                                                      ' %s' % (curpath, to_native(e)),
 651                                               'path': path})
 652         return {'path': path, 'changed': changed, 'diff': diff}
 653 
 654     elif prev_state != 'directory':
 655         # We already know prev_state is not 'absent', therefore it exists in some form.
 656         raise AnsibleModuleError(results={'msg': '%s already exists as a %s' % (path, prev_state),
 657                                           'path': path})
 658 
 659     #
 660     # previous state == directory
 661     #
 662 
 663     changed = module.set_fs_attributes_if_different(file_args, changed, diff, expand=False)
 664     changed |= update_timestamp_for_file(file_args['path'], mtime, atime, diff)
 665     if recurse:
 666         changed |= recursive_set_attributes(b_path, follow, file_args, mtime, atime)
 667 
 668     return {'path': path, 'changed': changed, 'diff': diff}

 671 def ensure_symlink(path, src, follow, force, timestamps):
 672     b_path = to_bytes(path, errors='surrogate_or_strict')
 673     b_src = to_bytes(src, errors='surrogate_or_strict')
 674     prev_state = get_state(b_path)
 675     mtime = get_timestamp_for_time(timestamps['modification_time'], timestamps['modification_time_format'])
 676     atime = get_timestamp_for_time(timestamps['access_time'], timestamps['access_time_format'])
 677     # source is both the source of a symlink or an informational passing of the src for a template module
 678     # or copy module, even if this module never uses it, it is needed to key off some things
 679     if src is None:
 680         if follow:
 681             # use the current target of the link as the source
 682             src = to_native(os.path.realpath(b_path), errors='strict')
 683             b_src = to_bytes(src, errors='surrogate_or_strict')
 684 
 685     if not os.path.islink(b_path) and os.path.isdir(b_path):
 686         relpath = path
 687     else:
 688         b_relpath = os.path.dirname(b_path)
 689         relpath = to_native(b_relpath, errors='strict')
 690 
 691     absrc = os.path.join(relpath, src)
 692     b_absrc = to_bytes(absrc, errors='surrogate_or_strict')
 693     if not force and not os.path.exists(b_absrc):
 694         raise AnsibleModuleError(results={'msg': 'src file does not exist, use "force=yes" if you'
 695                                                  ' really want to create the link: %s' % absrc,
 696                                           'path': path, 'src': src})
 697 
 698     if prev_state == 'directory':
 699         if not force:
 700             raise AnsibleModuleError(results={'msg': 'refusing to convert from %s to symlink for %s'
 701                                                      % (prev_state, path),
 702                                               'path': path})
 703         elif os.listdir(b_path):
 704             # refuse to replace a directory that has files in it
 705             raise AnsibleModuleError(results={'msg': 'the directory %s is not empty, refusing to'
 706                                                      ' convert it' % path,
 707                                               'path': path})
 708     elif prev_state in ('file', 'hard') and not force:
 709         raise AnsibleModuleError(results={'msg': 'refusing to convert from %s to symlink for %s'
 710                                                  % (prev_state, path),
 711                                           'path': path})
 712 
 713     diff = initial_diff(path, 'link', prev_state)
 714     changed = False
 715 
 716     if prev_state in ('hard', 'file', 'directory', 'absent'):
 717         changed = True
 718     elif prev_state == 'link':
 719         b_old_src = os.readlink(b_path)
 720         if b_old_src != b_src:
 721             diff['before']['src'] = to_native(b_old_src, errors='strict')
 722             diff['after']['src'] = src
 723             changed = True
 724     else:
 725         raise AnsibleModuleError(results={'msg': 'unexpected position reached', 'dest': path, 'src': src})
 726 
 727     if changed and not module.check_mode:
 728         if prev_state != 'absent':
 729             # try to replace atomically
 730             b_tmppath = to_bytes(os.path.sep).join(
 731                 [os.path.dirname(b_path), to_bytes(".%s.%s.tmp" % (os.getpid(), time.time()))]
 732             )
 733             try:
 734                 if prev_state == 'directory':
 735                     os.rmdir(b_path)
 736                 os.symlink(b_src, b_tmppath)
 737                 os.rename(b_tmppath, b_path)
 738             except OSError as e:
 739                 if os.path.exists(b_tmppath):
 740                     os.unlink(b_tmppath)
 741                 raise AnsibleModuleError(results={'msg': 'Error while replacing: %s'
 742                                                          % to_native(e, nonstring='simplerepr'),
 743                                                   'path': path})
 744         else:
 745             try:
 746                 os.symlink(b_src, b_path)
 747             except OSError as e:
 748                 raise AnsibleModuleError(results={'msg': 'Error while linking: %s'
 749                                                          % to_native(e, nonstring='simplerepr'),
 750                                                   'path': path})
 751 
 752     if module.check_mode and not os.path.exists(b_path):
 753         return {'dest': path, 'src': src, 'changed': changed, 'diff': diff}
 754 
 755     # Now that we might have created the symlink, get the arguments.
 756     # We need to do it now so we can properly follow the symlink if needed
 757     # because load_file_common_arguments sets 'path' according
 758     # the value of follow and the symlink existence.
 759     file_args = module.load_file_common_arguments(module.params)
 760 
 761     # Whenever we create a link to a nonexistent target we know that the nonexistent target
 762     # cannot have any permissions set on it.  Skip setting those and emit a warning (the user
 763     # can set follow=False to remove the warning)
 764     if follow and os.path.islink(b_path) and not os.path.exists(file_args['path']):
 765         module.warn('Cannot set fs attributes on a non-existent symlink target. follow should be'
 766                     ' set to False to avoid this.')
 767     else:
 768         changed = module.set_fs_attributes_if_different(file_args, changed, diff, expand=False)
 769         changed |= update_timestamp_for_file(file_args['path'], mtime, atime, diff)
 770 
 771     return {'dest': path, 'src': src, 'changed': changed, 'diff': diff}

 774 def ensure_hardlink(path, src, follow, force, timestamps):
 775     b_path = to_bytes(path, errors='surrogate_or_strict')
 776     b_src = to_bytes(src, errors='surrogate_or_strict')
 777     prev_state = get_state(b_path)
 778     file_args = module.load_file_common_arguments(module.params)
 779     mtime = get_timestamp_for_time(timestamps['modification_time'], timestamps['modification_time_format'])
 780     atime = get_timestamp_for_time(timestamps['access_time'], timestamps['access_time_format'])
 781 
 782     # src is the source of a hardlink.  We require it if we are creating a new hardlink.
 783     # We require path in the argument_spec so we know it is present at this point.
 784     if src is None:
 785         raise AnsibleModuleError(results={'msg': 'src is required for creating new hardlinks'})
 786 
 787     if not os.path.exists(b_src):
 788         raise AnsibleModuleError(results={'msg': 'src does not exist', 'dest': path, 'src': src})
 789 
 790     diff = initial_diff(path, 'hard', prev_state)
 791     changed = False
 792 
 793     if prev_state == 'absent':
 794         changed = True
 795     elif prev_state == 'link':
 796         b_old_src = os.readlink(b_path)
 797         if b_old_src != b_src:
 798             diff['before']['src'] = to_native(b_old_src, errors='strict')
 799             diff['after']['src'] = src
 800             changed = True
 801     elif prev_state == 'hard':
 802         if not os.stat(b_path).st_ino == os.stat(b_src).st_ino:
 803             changed = True
 804             if not force:
 805                 raise AnsibleModuleError(results={'msg': 'Cannot link, different hard link exists at destination',
 806                                                   'dest': path, 'src': src})
 807     elif prev_state == 'file':
 808         changed = True
 809         if not force:
 810             raise AnsibleModuleError(results={'msg': 'Cannot link, %s exists at destination' % prev_state,
 811                                               'dest': path, 'src': src})
 812     elif prev_state == 'directory':
 813         changed = True
 814         if os.path.exists(b_path):
 815             if os.stat(b_path).st_ino == os.stat(b_src).st_ino:
 816                 return {'path': path, 'changed': False}
 817             elif not force:
 818                 raise AnsibleModuleError(results={'msg': 'Cannot link: different hard link exists at destination',
 819                                                   'dest': path, 'src': src})
 820     else:
 821         raise AnsibleModuleError(results={'msg': 'unexpected position reached', 'dest': path, 'src': src})
 822 
 823     if changed and not module.check_mode:
 824         if prev_state != 'absent':
 825             # try to replace atomically
 826             b_tmppath = to_bytes(os.path.sep).join(
 827                 [os.path.dirname(b_path), to_bytes(".%s.%s.tmp" % (os.getpid(), time.time()))]
 828             )
 829             try:
 830                 if prev_state == 'directory':
 831                     if os.path.exists(b_path):
 832                         try:
 833                             os.unlink(b_path)
 834                         except OSError as e:
 835                             if e.errno != errno.ENOENT:  # It may already have been removed
 836                                 raise
 837                 os.link(b_src, b_tmppath)
 838                 os.rename(b_tmppath, b_path)
 839             except OSError as e:
 840                 if os.path.exists(b_tmppath):
 841                     os.unlink(b_tmppath)
 842                 raise AnsibleModuleError(results={'msg': 'Error while replacing: %s'
 843                                                          % to_native(e, nonstring='simplerepr'),
 844                                                   'path': path})
 845         else:
 846             try:
 847                 os.link(b_src, b_path)
 848             except OSError as e:
 849                 raise AnsibleModuleError(results={'msg': 'Error while linking: %s'
 850                                                          % to_native(e, nonstring='simplerepr'),
 851                                                   'path': path})
 852 
 853     if module.check_mode and not os.path.exists(b_path):
 854         return {'dest': path, 'src': src, 'changed': changed, 'diff': diff}
 855 
 856     changed = module.set_fs_attributes_if_different(file_args, changed, diff, expand=False)
 857     changed |= update_timestamp_for_file(file_args['path'], mtime, atime, diff)
 858 
 859     return {'dest': path, 'src': src, 'changed': changed, 'diff': diff}
