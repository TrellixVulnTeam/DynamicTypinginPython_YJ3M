 221     """
 222         CronTab object to write time based crontab file

 228     def __init__(self, module, user=None, cron_file=None):
 229         self.module = module
 230         self.user = user
 231         self.root = (os.getuid() == 0)
 232         self.lines = None
 233         self.ansible = "#Ansible: "
 234         self.existing = ''
 235         self.cron_cmd = self.module.get_bin_path('crontab', required=True)
 236 
 237         if cron_file:
 238             if os.path.isabs(cron_file):
 239                 self.cron_file = cron_file
 240             else:
 241                 self.cron_file = os.path.join('/etc/cron.d', cron_file)
 242         else:
 243             self.cron_file = None
 244 
 245         self.read()

 247     def read(self):
 248         # Read in the crontab from the system
 249         self.lines = []
 250         if self.cron_file:
 251             # read the cronfile
 252             try:
 253                 f = open(self.cron_file, 'r')
 254                 self.existing = f.read()
 255                 self.lines = self.existing.splitlines()
 256                 f.close()
 257             except IOError:
 258                 # cron file does not exist
 259                 return
 260             except Exception:
 261                 raise CronTabError("Unexpected error:", sys.exc_info()[0])
 262         else:
 263             # using safely quoted shell for now, but this really should be two non-shell calls instead.  FIXME
 264             (rc, out, err) = self.module.run_command(self._read_user_execute(), use_unsafe_shell=True)
 265 
 266             if rc != 0 and rc != 1:  # 1 can mean that there are no jobs.
 267                 raise CronTabError("Unable to read crontab")
 268 
 269             self.existing = out
 270 
 271             lines = out.splitlines()
 272             count = 0
 273             for l in lines:
 274                 if count > 2 or (not re.match(r'# DO NOT EDIT THIS FILE - edit the master and reinstall.', l) and
 275                                  not re.match(r'# \(/tmp/.*installed on.*\)', l) and
 276                                  not re.match(r'# \(.*version.*\)', l)):
 277                     self.lines.append(l)
 278                 else:
 279                     pattern = re.escape(l) + '[\r\n]?'
 280                     self.existing = re.sub(pattern, '', self.existing, 1)
 281                 count += 1

 283     def is_empty(self):
 284         if len(self.lines) == 0:
 285             return True
 286         else:
 287             return False

 289     def write(self, backup_file=None):
 290         """
 291         Write the crontab to the system. Saves all information.
 292         """
 293         if backup_file:
 294             fileh = open(backup_file, 'w')
 295         elif self.cron_file:
 296             fileh = open(self.cron_file, 'w')
 297         else:
 298             filed, path = tempfile.mkstemp(prefix='crontab')
 299             os.chmod(path, int('0644', 8))
 300             fileh = os.fdopen(filed, 'w')
 301 
 302         fileh.write(self.render())
 303         fileh.close()
 304 
 305         # return if making a backup
 306         if backup_file:
 307             return
 308 
 309         # Add the entire crontab back to the user crontab
 310         if not self.cron_file:
 311             # quoting shell args for now but really this should be two non-shell calls.  FIXME
 312             (rc, out, err) = self.module.run_command(self._write_execute(path), use_unsafe_shell=True)
 313             os.unlink(path)
 314 
 315             if rc != 0:
 316                 self.module.fail_json(msg=err)
 317 
 318         # set SELinux permissions
 319         if self.module.selinux_enabled() and self.cron_file:
 320             self.module.set_default_selinux_context(self.cron_file, False)

 346     def add_env(self, decl, insertafter=None, insertbefore=None):
 347         if not (insertafter or insertbefore):
 348             self.lines.insert(0, decl)
 349             return
 350 
 351         if insertafter:
 352             other_name = insertafter
 353         elif insertbefore:
 354             other_name = insertbefore
 355         other_decl = self.find_env(other_name)
 356         if len(other_decl) > 0:
 357             if insertafter:
 358                 index = other_decl[0] + 1
 359             elif insertbefore:
 360                 index = other_decl[0]
 361             self.lines.insert(index, decl)
 362             return
 363 
 364         self.module.fail_json(msg="Variable named '%s' not found." % other_name)

 388     def find_job(self, name, job=None):
 389         # attempt to find job by 'Ansible:' header comment
 390         comment = None
 391         for l in self.lines:
 392             if comment is not None:
 393                 if comment == name:
 394                     return [comment, l]
 395                 else:
 396                     comment = None
 397             elif re.match(r'%s' % self.ansible, l):
 398                 comment = re.sub(r'%s' % self.ansible, '', l)
 399 
 400         # failing that, attempt to find job by exact match
 401         if job:
 402             for i, l in enumerate(self.lines):
 403                 if l == job:
 404                     # if no leading ansible header, insert one
 405                     if not re.match(r'%s' % self.ansible, self.lines[i - 1]):
 406                         self.lines.insert(i, self.do_comment(name))
 407                         return [self.lines[i], l, True]
 408                     # if a leading blank ansible header AND job has a name, update header
 409                     elif name and self.lines[i - 1] == self.do_comment(None):
 410                         self.lines[i - 1] = self.do_comment(name)
 411                         return [self.lines[i - 1], l, True]
 412 
 413         return []

 415     def find_env(self, name):
 416         for index, l in enumerate(self.lines):
 417             if re.match(r'^%s=' % name, l):
 418                 return [index, l]
 419 
 420         return []

 422     def get_cron_job(self, minute, hour, day, month, weekday, job, special, disabled):
 423         # normalize any leading/trailing newlines (ansible/ansible-modules-core#3791)
 424         job = job.strip('\r\n')
 425 
 426         if disabled:
 427             disable_prefix = '#'
 428         else:
 429             disable_prefix = ''
 430 
 431         if special:
 432             if self.cron_file:
 433                 return "%s@%s %s %s" % (disable_prefix, special, self.user, job)
 434             else:
 435                 return "%s@%s %s" % (disable_prefix, special, job)
 436         else:
 437             if self.cron_file:
 438                 return "%s%s %s %s %s %s %s %s" % (disable_prefix, minute, hour, day, month, weekday, self.user, job)
 439             else:
 440                 return "%s%s %s %s %s %s %s" % (disable_prefix, minute, hour, day, month, weekday, job)

 442     def get_jobnames(self):
 443         jobnames = []
 444 
 445         for l in self.lines:
 446             if re.match(r'%s' % self.ansible, l):
 447                 jobnames.append(re.sub(r'%s' % self.ansible, '', l))
 448 
 449         return jobnames

 451     def get_envnames(self):
 452         envnames = []
 453 
 454         for l in self.lines:
 455             if re.match(r'^\S+=', l):
 456                 envnames.append(l.split('=')[0])
 457 
 458         return envnames

 460     def _update_job(self, name, job, addlinesfunction):
 461         ansiblename = self.do_comment(name)
 462         newlines = []
 463         comment = None
 464 
 465         for l in self.lines:
 466             if comment is not None:
 467                 addlinesfunction(newlines, comment, job)
 468                 comment = None
 469             elif l == ansiblename:
 470                 comment = l
 471             else:
 472                 newlines.append(l)
 473 
 474         self.lines = newlines
 475 
 476         if len(newlines) == 0:
 477             return True
 478         else:
 479             return False  # TODO add some more error testing

 481     def _update_env(self, name, decl, addenvfunction):
 482         newlines = []
 483 
 484         for l in self.lines:
 485             if re.match(r'^%s=' % name, l):
 486                 addenvfunction(newlines, decl)
 487             else:
 488                 newlines.append(l)
 489 
 490         self.lines = newlines

 492     def render(self):
 493         """
 494         Render this crontab as it would be in the crontab.
 495         """
 496         crons = []
 497         for cron in self.lines:
 498             crons.append(cron)
 499 
 500         result = '\n'.join(crons)
 501         if result:
 502             result = result.rstrip('\r\n') + '\n'
 503         return result

 505     def _read_user_execute(self):
 506         """
 507         Returns the command line for reading a crontab
 508         """
 509         user = ''
 510         if self.user:
 511             if platform.system() == 'SunOS':
 512                 return "su %s -c '%s -l'" % (shlex_quote(self.user), shlex_quote(self.cron_cmd))
 513             elif platform.system() == 'AIX':
 514                 return "%s -l %s" % (shlex_quote(self.cron_cmd), shlex_quote(self.user))
 515             elif platform.system() == 'HP-UX':
 516                 return "%s %s %s" % (self.cron_cmd, '-l', shlex_quote(self.user))
 517             elif pwd.getpwuid(os.getuid())[0] != self.user:
 518                 user = '-u %s' % shlex_quote(self.user)
 519         return "%s %s %s" % (self.cron_cmd, user, '-l')
 520 
 521     def _write_execute(self, path):
 522         """
 523         Return the command line for writing a crontab
 524         """
 525         user = ''
 526         if self.user:
 527             if platform.system() in ['SunOS', 'HP-UX', 'AIX']:
 528                 return "chown %s %s ; su '%s' -c '%s %s'" % (
 529                     shlex_quote(self.user), shlex_quote(path), shlex_quote(self.user), self.cron_cmd, shlex_quote(path))
 530             elif pwd.getpwuid(os.getuid())[0] != self.user:
 531                 user = '-u %s' % shlex_quote(self.user)
 532         return "%s %s %s" % (self.cron_cmd, user, shlex_quote(path))

 593     weekday = module.params['weekday']

 597     env = module.params['env']

 599     insertbefore = module.params['insertbefore']

 602     changed = False
 603     res_args = dict()
 604     warnings = list()

 606     if cron_file:
 607         cron_file_basename = os.path.basename(cron_file)
 608         if not re.search(r'^[A-Z0-9_-]+$', cron_file_basename, re.I):
 609             warnings.append('Filename portion of cron_file ("%s") should consist' % cron_file_basename +
 610                             ' solely of upper- and lower-case letters, digits, underscores, and hyphens')

 613     os.umask(int('022', 8))

 617 
 618     if not name:
 619         module.deprecate(
 620             msg="The 'name' parameter will be required in future releases.",
 621             version='2.12', collection_name='ansible.builtin'

 626             version='2.12', collection_name='ansible.builtin'

 632         if crontab.cron_file:

 641 
 642     if env and not name:
 643         module.fail_json(msg="You must specify 'name' while working with environment variables (env=yes)")
 644 
 645     if (special_time or reboot) and \
 646        (True in [(x != '*') for x in [minute, hour, day, month, weekday]]):
 647         module.fail_json(msg="You must specify time and date fields or special time.")
 648 
 649     # cannot support special_time on solaris
 650     if (special_time or reboot) and platform.system() == 'SunOS':
 651         module.fail_json(msg="Solaris does not support special_time=... or @reboot")
 652 
 653     if cron_file and do_install:
 654         if not user:
 655             module.fail_json(msg="To use cron_file=... parameter you must specify user=... as well")
 656 
 657     if job is None and do_install:
 658         module.fail_json(msg="You must specify 'job' to install a new cron job or variable")
 659 
 660     if (insertafter or insertbefore) and not env and do_install:
 661         module.fail_json(msg="Insertafter and insertbefore parameters are valid only with env=yes")
 662 
 663     if reboot:
 664         special_time = "reboot"
 665 
 666     # if requested make a backup before making a change
 667     if backup and not module.check_mode:
 668         (backuph, backup_file) = tempfile.mkstemp(prefix='crontab')
 669         crontab.write(backup_file)
 670 
 671     if crontab.cron_file and not do_install:
 672         if module._diff:
 673             diff['after'] = ''
 674             diff['after_header'] = '/dev/null'
 675         else:
 676             diff = dict()
 677         if module.check_mode:
 678             changed = os.path.isfile(crontab.cron_file)
 679         else:
 680             changed = crontab.remove_job_file()
 681         module.exit_json(changed=changed, cron_file=cron_file, state=state, diff=diff)
 682 
 683     if env:
 684         if ' ' in name:
 685             module.fail_json(msg="Invalid name for environment variable")
 686         decl = '%s="%s"' % (name, job)
 687         old_decl = crontab.find_env(name)
 688 
 689         if do_install:
 690             if len(old_decl) == 0:
 691                 crontab.add_env(decl, insertafter, insertbefore)
 692                 changed = True
 693             if len(old_decl) > 0 and old_decl[1] != decl:
 694                 crontab.update_env(name, decl)
 695                 changed = True
 696         else:
 697             if len(old_decl) > 0:
 698                 crontab.remove_env(name)
 699                 changed = True
 700     else:
 701         if do_install:
 702             for char in ['\r', '\n']:
 703                 if char in job.strip('\r\n'):
 704                     warnings.append('Job should not contain line breaks')
 705                     break
 706 
 707             job = crontab.get_cron_job(minute, hour, day, month, weekday, job, special_time, disabled)
 708             old_job = crontab.find_job(name, job)
 709 
 710             if len(old_job) == 0:
 711                 crontab.add_job(name, job)
 712                 changed = True
 713             if len(old_job) > 0 and old_job[1] != job:
 714                 crontab.update_job(name, job)
 715                 changed = True
 716             if len(old_job) > 2:
 717                 crontab.update_job(name, job)
 718                 changed = True
 719         else:
 720             old_job = crontab.find_job(name)
 721 
 722             if len(old_job) > 0:
 723                 crontab.remove_job(name)
 724                 changed = True
 725 
 726     # no changes to env/job, but existing crontab needs a terminating newline
 727     if not changed and crontab.existing != '':
 728         if not (crontab.existing.endswith('\r') or crontab.existing.endswith('\n')):
 729             changed = True
 730 
 731     res_args = dict(
 732         jobs=crontab.get_jobnames(),
 733         envs=crontab.get_envnames(),
 734         warnings=warnings,
 735         changed=changed
 736     )
 737 
 738     if changed:
 739         if not module.check_mode:
 740             crontab.write()
 741         if module._diff:
 742             diff['after'] = crontab.render()
 743             if crontab.cron_file:
 744                 diff['after_header'] = crontab.cron_file
 745             else:
 746                 if crontab.user:
 747                     diff['after_header'] = 'crontab for user "%s"' % crontab.user
 748                 else:
 749                     diff['after_header'] = 'crontab'
 750 
 751             res_args['diff'] = diff
 752 
 753     # retain the backup only if crontab or cron file have changed
 754     if backup and not module.check_mode:
 755         if changed:
 756             res_args['backup_file'] = backup_file
 757         else:
 758             os.unlink(backup_file)
 759 
 760     if cron_file:
 761         res_args['cron_file'] = cron_file
 762 
 763     module.exit_json(**res_args)
 764 
 765     # --- should never get here
 766     module.exit_json(msg="Unable to execute cron task.")
 767

 769 if __name__ == '__main__':
 770     main()
