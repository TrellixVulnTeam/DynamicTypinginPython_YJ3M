 212     def execute_command(self, cmd, daemonize=False):
 213 
 214         # Most things don't need to be daemonized
 215         if not daemonize:
 216             # chkconfig localizes messages and we're screen scraping so make
 217             # sure we use the C locale
 218             lang_env = dict(LANG='C', LC_ALL='C', LC_MESSAGES='C')
 219             return self.module.run_command(cmd, environ_update=lang_env)
 220 
 221         # This is complex because daemonization is hard for people.
 222         # What we do is daemonize a part of this module, the daemon runs the
 223         # command, picks up the return code and output, and returns it to the
 224         # main process.
 225         pipe = os.pipe()
 226         pid = os.fork()
 227         if pid == 0:
 228             os.close(pipe[0])
 229             # Set stdin/stdout/stderr to /dev/null
 230             fd = os.open(os.devnull, os.O_RDWR)
 231             if fd != 0:
 232                 os.dup2(fd, 0)
 233             if fd != 1:
 234                 os.dup2(fd, 1)
 235             if fd != 2:
 236                 os.dup2(fd, 2)
 237             if fd not in (0, 1, 2):
 238                 os.close(fd)
 239 
 240             # Make us a daemon. Yes, that's all it takes.
 241             pid = os.fork()
 242             if pid > 0:
 243                 os._exit(0)
 244             os.setsid()
 245             os.chdir("/")
 246             pid = os.fork()
 247             if pid > 0:
 248                 os._exit(0)
 249 
 250             # Start the command
 251             if PY2:
 252                 # Python 2.6's shlex.split can't handle text strings correctly
 253                 cmd = to_bytes(cmd, errors='surrogate_or_strict')
 254                 cmd = shlex.split(cmd)
 255             else:
 256                 # Python3.x shex.split text strings.
 257                 cmd = to_text(cmd, errors='surrogate_or_strict')
 258                 cmd = [to_bytes(c, errors='surrogate_or_strict') for c in shlex.split(cmd)]
 259             # In either of the above cases, pass a list of byte strings to Popen
 260 
 261             # chkconfig localizes messages and we're screen scraping so make
 262             # sure we use the C locale
 263             lang_env = dict(LANG='C', LC_ALL='C', LC_MESSAGES='C')
 264             p = subprocess.Popen(cmd, shell=False, stdout=subprocess.PIPE, stderr=subprocess.PIPE, env=lang_env, preexec_fn=lambda: os.close(pipe[1]))
 265             stdout = b("")
 266             stderr = b("")
 267             fds = [p.stdout, p.stderr]
 268             # Wait for all output, or until the main process is dead and its output is done.
 269             while fds:
 270                 rfd, wfd, efd = select.select(fds, [], fds, 1)
 271                 if not (rfd + wfd + efd) and p.poll() is not None:
 272                     break
 273                 if p.stdout in rfd:
 274                     dat = os.read(p.stdout.fileno(), 4096)
 275                     if not dat:
 276                         fds.remove(p.stdout)
 277                     stdout += dat
 278                 if p.stderr in rfd:
 279                     dat = os.read(p.stderr.fileno(), 4096)
 280                     if not dat:
 281                         fds.remove(p.stderr)
 282                     stderr += dat
 283             p.wait()
 284             # Return a JSON blob to parent
 285             blob = json.dumps([p.returncode, to_text(stdout), to_text(stderr)])
 286             os.write(pipe[1], to_bytes(blob, errors='surrogate_or_strict'))
 287             os.close(pipe[1])
 288             os._exit(0)
 289         elif pid == -1:
 290             self.module.fail_json(msg="unable to fork")
 291         else:
 292             os.close(pipe[1])
 293             os.waitpid(pid, 0)
 294             # Wait for data from daemon process and process it.
 295             data = b("")
 296             while True:
 297                 rfd, wfd, efd = select.select([pipe[0]], [], [pipe[0]])
 298                 if pipe[0] in rfd:
 299                     dat = os.read(pipe[0], 4096)
 300                     if not dat:
 301                         break
 302                     data += dat
 303             return json.loads(to_text(data, errors='surrogate_or_strict'))

 305     def check_ps(self):
 306         # Set ps flags
 307         if platform.system() == 'SunOS':
 308             psflags = '-ef'
 309         else:
 310             psflags = 'auxww'
 311 
 312         # Find ps binary
 313         psbin = self.module.get_bin_path('ps', True)
 314 
 315         (rc, psout, pserr) = self.execute_command('%s %s' % (psbin, psflags))
 316         # If rc is 0, set running as appropriate
 317         if rc == 0:
 318             self.running = False
 319             lines = psout.split("\n")
 320             for line in lines:
 321                 if self.pattern in line and "pattern=" not in line:
 322                     # so as to not confuse ./hacking/test-module.py
 323                     self.running = True
 324                     break

 326     def check_service_changed(self):
 327         if self.state and self.running is None:
 328             self.module.fail_json(msg="failed determining service state, possible typo of service name?")
 329         # Find out if state has changed
 330         if not self.running and self.state in ["reloaded", "started"]:
 331             self.svc_change = True
 332         elif self.running and self.state in ["reloaded", "stopped"]:
 333             self.svc_change = True
 334         elif self.state == "restarted":
 335             self.svc_change = True
 336         if self.module.check_mode and self.svc_change:
 337             self.module.exit_json(changed=True, msg='service state changed')

 339     def modify_service_state(self):
 340 
 341         # Only do something if state will change
 342         if self.svc_change:
 343             # Control service
 344             if self.state in ['started']:
 345                 self.action = "start"
 346             elif not self.running and self.state == 'reloaded':
 347                 self.action = "start"
 348             elif self.state == 'stopped':
 349                 self.action = "stop"
 350             elif self.state == 'reloaded':
 351                 self.action = "reload"
 352             elif self.state == 'restarted':
 353                 self.action = "restart"
 354 
 355             if self.module.check_mode:
 356                 self.module.exit_json(changed=True, msg='changing service state')
 357 
 358             return self.service_control()
 359 
 360         else:
 361             # If nothing needs to change just say all is well
 362             rc = 0
 363             err = ''
 364             out = ''
 365             return rc, out, err

 367     def service_enable_rcconf(self):
 368         if self.rcconf_file is None or self.rcconf_key is None or self.rcconf_value is None:
 369             self.module.fail_json(msg="service_enable_rcconf() requires rcconf_file, rcconf_key and rcconf_value")
 370 
 371         self.changed = None
 372         entry = '%s="%s"\n' % (self.rcconf_key, self.rcconf_value)
 373         RCFILE = open(self.rcconf_file, "r")
 374         new_rc_conf = []
 375 
 376         # Build a list containing the possibly modified file.
 377         for rcline in RCFILE:
 378             # Parse line removing whitespaces, quotes, etc.
 379             rcarray = shlex.split(rcline, comments=True)
 380             if len(rcarray) >= 1 and '=' in rcarray[0]:
 381                 (key, value) = rcarray[0].split("=", 1)
 382                 if key == self.rcconf_key:
 383                     if value.upper() == self.rcconf_value:
 384                         # Since the proper entry already exists we can stop iterating.
 385                         self.changed = False
 386                         break
 387                     else:
 388                         # We found the key but the value is wrong, replace with new entry.
 389                         rcline = entry
 390                         self.changed = True
 391 
 392             # Add line to the list.
 393             new_rc_conf.append(rcline.strip() + '\n')
 394 
 395         # We are done with reading the current rc.conf, close it.
 396         RCFILE.close()
 397 
 398         # If we did not see any trace of our entry we need to add it.
 399         if self.changed is None:
 400             new_rc_conf.append(entry)
 401             self.changed = True
 402 
 403         if self.changed is True:
 404 
 405             if self.module.check_mode:
 406                 self.module.exit_json(changed=True, msg="changing service enablement")
 407 
 408             # Create a temporary file next to the current rc.conf (so we stay on the same filesystem).
 409             # This way the replacement operation is atomic.
 410             rcconf_dir = os.path.dirname(self.rcconf_file)
 411             rcconf_base = os.path.basename(self.rcconf_file)
 412             (TMP_RCCONF, tmp_rcconf_file) = tempfile.mkstemp(dir=rcconf_dir, prefix="%s-" % rcconf_base)
 413 
 414             # Write out the contents of the list into our temporary file.
 415             for rcline in new_rc_conf:
 416                 os.write(TMP_RCCONF, rcline)
 417 
 418             # Close temporary file.
 419             os.close(TMP_RCCONF)
 420 
 421             # Replace previous rc.conf.
 422             self.module.atomic_move(tmp_rcconf_file, self.rcconf_file)

 425 class LinuxService(Service):
 426     """
 427     This is the Linux Service manipulation class - it is currently supporting
 428     a mixture of binaries and init scripts for controlling services started at
 429     boot, as well as for controlling the current state.
 430     """

 435     def get_service_tools(self):
 436 
 437         paths = ['/sbin', '/usr/sbin', '/bin', '/usr/bin']
 438         binaries = ['service', 'chkconfig', 'update-rc.d', 'rc-service', 'rc-update', 'initctl', 'systemctl', 'start', 'stop', 'restart', 'insserv']
 439         initpaths = ['/etc/init.d']
 440         location = dict()
 441 
 442         for binary in binaries:
 443             location[binary] = self.module.get_bin_path(binary, opt_dirs=paths)
 444 
 445         for initdir in initpaths:
 446             initscript = "%s/%s" % (initdir, self.name)
 447             if os.path.isfile(initscript):
 448                 self.svc_initscript = initscript
 449 
 450         def check_systemd():
 451 
 452             # tools must be installed
 453             if location.get('systemctl', False):
 454 
 455                 # this should show if systemd is the boot init system
 456                 # these mirror systemd's own sd_boot test http://www.freedesktop.org/software/systemd/man/sd_booted.html
 457                 for canary in ["/run/systemd/system/", "/dev/.run/systemd/", "/dev/.systemd/"]:
 458                     if os.path.exists(canary):
 459                         return True
 460 
 461                 # If all else fails, check if init is the systemd command, using comm as cmdline could be symlink
 462                 try:
 463                     f = open('/proc/1/comm', 'r')
 464                 except IOError:
 465                     # If comm doesn't exist, old kernel, no systemd
 466                     return False
 467 
 468                 for line in f:
 469                     if 'systemd' in line:
 470                         return True
 471 
 472             return False
 473 
 474         # Locate a tool to enable/disable a service
 475         if check_systemd():
 476             # service is managed by systemd
 477             self.__systemd_unit = self.name
 478             self.svc_cmd = location['systemctl']
 479             self.enable_cmd = location['systemctl']
 480 
 481         elif location.get('initctl', False) and os.path.exists("/etc/init/%s.conf" % self.name):
 482             # service is managed by upstart
 483             self.enable_cmd = location['initctl']
 484             # set the upstart version based on the output of 'initctl version'
 485             self.upstart_version = LooseVersion('0.0.0')
 486             try:
 487                 version_re = re.compile(r'\(upstart (.*)\)')
 488                 rc, stdout, stderr = self.module.run_command('%s version' % location['initctl'])
 489                 if rc == 0:
 490                     res = version_re.search(stdout)
 491                     if res:
 492                         self.upstart_version = LooseVersion(res.groups()[0])
 493             except Exception:
 494                 pass  # we'll use the default of 0.0.0
 495 
 496             self.svc_cmd = location['initctl']
 497 
 498         elif location.get('rc-service', False):
 499             # service is managed by OpenRC
 500             self.svc_cmd = location['rc-service']
 501             self.enable_cmd = location['rc-update']
 502             return  # already have service start/stop tool too!
 503 
 504         elif self.svc_initscript:
 505             # service is managed by with SysV init scripts
 506             if location.get('update-rc.d', False):
 507                 # and uses update-rc.d
 508                 self.enable_cmd = location['update-rc.d']
 509             elif location.get('insserv', None):
 510                 # and uses insserv
 511                 self.enable_cmd = location['insserv']
 512             elif location.get('chkconfig', False):
 513                 # and uses chkconfig
 514                 self.enable_cmd = location['chkconfig']
 515 
 516         if self.enable_cmd is None:
 517             fail_if_missing(self.module, False, self.name, msg='host')
 518 
 519         # If no service control tool selected yet, try to see if 'service' is available
 520         if self.svc_cmd is None and location.get('service', False):
 521             self.svc_cmd = location['service']
 522 
 523         # couldn't find anything yet
 524         if self.svc_cmd is None and not self.svc_initscript:
 525             self.module.fail_json(msg='cannot find \'service\' binary or init script for service,  possible typo in service name?, aborting')
 526 
 527         if location.get('initctl', False):
 528             self.svc_initctl = location['initctl']

 530     def get_systemd_service_enabled(self):
 531         def sysv_exists(name):
 532             script = '/etc/init.d/' + name
 533             return os.access(script, os.X_OK)
 534 
 535         def sysv_is_enabled(name):
 536             return bool(glob.glob('/etc/rc?.d/S??' + name))
 537 
 538         service_name = self.__systemd_unit
 539         (rc, out, err) = self.execute_command("%s is-enabled %s" % (self.enable_cmd, service_name,))
 540         if rc == 0:
 541             return True
 542         elif out.startswith('disabled'):
 543             return False
 544         elif sysv_exists(service_name):
 545             return sysv_is_enabled(service_name)
 546         else:
 547             return False

 549     def get_systemd_status_dict(self):
 550 
 551         # Check status first as show will not fail if service does not exist
 552         (rc, out, err) = self.execute_command("%s show '%s'" % (self.enable_cmd, self.__systemd_unit,))
 553         if rc != 0:
 554             self.module.fail_json(msg='failure %d running systemctl show for %r: %s' % (rc, self.__systemd_unit, err))
 555         elif 'LoadState=not-found' in out:
 556             self.module.fail_json(msg='systemd could not find the requested service "%r": %s' % (self.__systemd_unit, err))
 557 
 558         key = None
 559         value_buffer = []
 560         status_dict = {}
 561         for line in out.splitlines():
 562             if '=' in line:
 563                 if not key:
 564                     key, value = line.split('=', 1)
 565                     # systemd fields that are shell commands can be multi-line
 566                     # We take a value that begins with a "{" as the start of
 567                     # a shell command and a line that ends with "}" as the end of
 568                     # the command
 569                     if value.lstrip().startswith('{'):
 570                         if value.rstrip().endswith('}'):
 571                             status_dict[key] = value
 572                             key = None
 573                         else:
 574                             value_buffer.append(value)
 575                     else:
 576                         status_dict[key] = value
 577                         key = None
 578                 else:
 579                     if line.rstrip().endswith('}'):
 580                         status_dict[key] = '\n'.join(value_buffer)
 581                         key = None
 582                     else:
 583                         value_buffer.append(value)
 584             else:
 585                 value_buffer.append(value)
 586 
 587         return status_dict

 589     def get_systemd_service_status(self):
 590         d = self.get_systemd_status_dict()
 591         if d.get('ActiveState') == 'active':
 592             # run-once services (for which a single successful exit indicates
 593             # that they are running as designed) should not be restarted here.
 594             # Thus, we are not checking d['SubState'].
 595             self.running = True
 596             self.crashed = False
 597         elif d.get('ActiveState') == 'failed':
 598             self.running = False
 599             self.crashed = True
 600         elif d.get('ActiveState') is None:
 601             self.module.fail_json(msg='No ActiveState value in systemctl show output for %r' % (self.__systemd_unit,))
 602         else:
 603             self.running = False
 604             self.crashed = False
 605         return self.running

 608         if self.svc_cmd and self.svc_cmd.endswith('systemctl'):
 609             return self.get_systemd_service_status()

 611         self.action = "status"
 612         rc, status_stdout, status_stderr = self.service_control()
 613 
 614         # if we have decided the service is managed by upstart, we check for some additional output...
 615         if self.svc_initctl and self.running is None:
 616             # check the job status by upstart response
 617             initctl_rc, initctl_status_stdout, initctl_status_stderr = self.execute_command("%s status %s %s" % (self.svc_initctl, self.name, self.arguments))
 618             if "stop/waiting" in initctl_status_stdout:
 619                 self.running = False
 620             elif "start/running" in initctl_status_stdout:
 621                 self.running = True
 622 
 623         if self.svc_cmd and self.svc_cmd.endswith("rc-service") and self.running is None:
 624             openrc_rc, openrc_status_stdout, openrc_status_stderr = self.execute_command("%s %s status" % (self.svc_cmd, self.name))
 625             self.running = "started" in openrc_status_stdout
 626             self.crashed = "crashed" in openrc_status_stderr
 627 
 628         # Prefer a non-zero return code. For reference, see:
 629         # http://refspecs.linuxbase.org/LSB_4.1.0/LSB-Core-generic/LSB-Core-generic/iniscrptact.html
 630         if self.running is None and rc in [1, 2, 3, 4, 69]:
 631             self.running = False
 632 
 633         # if the job status is still not known check it by status output keywords
 634         # Only check keywords if there's only one line of output (some init
 635         # scripts will output verbosely in case of error and those can emit
 636         # keywords that are picked up as false positives
 637         if self.running is None and status_stdout.count('\n') <= 1:
 638             # first transform the status output that could irritate keyword matching
 639             cleanout = status_stdout.lower().replace(self.name.lower(), '')
 640             if "stop" in cleanout:
 641                 self.running = False
 642             elif "run" in cleanout:
 643                 self.running = not ("not " in cleanout)
 644             elif "start" in cleanout and "not " not in cleanout:
 645                 self.running = True
 646             elif 'could not access pid file' in cleanout:
 647                 self.running = False
 648             elif 'is dead and pid file exists' in cleanout:
 649                 self.running = False
 650             elif 'dead but subsys locked' in cleanout:
 651                 self.running = False
 652             elif 'dead but pid file exists' in cleanout:
 653                 self.running = False
 654 
 655         # if the job status is still not known and we got a zero for the
 656         # return code, assume here that the service is running
 657         if self.running is None and rc == 0:
 658             self.running = True
 659 
 660         # if the job status is still not known check it by special conditions
 661         if self.running is None:
 662             if self.name == 'iptables' and "ACCEPT" in status_stdout:
 663                 # iptables status command output is lame
 664                 # TODO: lookup if we can use a return code for this instead?
 665                 self.running = True
 666 
 667         return self.running

 669     def service_enable(self):

 672             self.module.fail_json(msg='cannot detect command to enable service %s, typo or init system potentially unknown' % self.name)
 673

 678         # Upstart's initctl

 700                 self.module.fail_json(msg="manual stanza not supported in a .conf file")
 701 
 702             self.changed = False
 703             if os.path.exists(override_file_name):
 704                 with open(override_file_name) as override_fh:
 705                     override_file_contents = override_fh.read()

 707                 if self.enable and manreg.search(override_file_contents):
 708                     self.changed = True
 709                     override_state = manreg.sub('', override_file_contents)
 710                 # Add manual stanza if not present and service disabled
 711                 elif not (self.enable) and not (manreg.search(override_file_contents)):
 712                     self.changed = True
 713                     override_state = '\n'.join((override_file_contents, config_line))
 714                 # service already in desired state
 715                 else:
 716                     pass
 717             # Add file with manual stanza if service disabled
 718             elif not (self.enable):
 719                 self.changed = True
 720                 override_state = config_line
 721             else:
 722                 # service already in desired state
 723                 pass
 724 
 725             if self.module.check_mode:
 726                 self.module.exit_json(changed=self.changed)
 727 
 728             # The initctl method of enabling and disabling services is much
 729             # different than for the other service methods.  So actually
 730             # committing the change is done in this conditional and then we
 731             # skip the boilerplate at the bottom of the method
 732             if self.changed:
 733                 try:
 734                     write_to_override_file(override_file_name, override_state)
 735                 except Exception:
 736                     self.module.fail_json(msg='Could not modify override file')
 737 
 738             return
 739

 741         # SysV's chkconfig
 742         #
 743         if self.enable_cmd.endswith("chkconfig"):
 744             if self.enable:
 745                 action = 'on'
 746             else:
 747                 action = 'off'
 748 
 749             (rc, out, err) = self.execute_command("%s --list %s" % (self.enable_cmd, self.name))
 750             if 'chkconfig --add %s' % self.name in err:
 751                 self.execute_command("%s --add %s" % (self.enable_cmd, self.name))
 752                 (rc, out, err) = self.execute_command("%s --list %s" % (self.enable_cmd, self.name))

 755             # TODO: look back on why this is here
 756             # state = out.split()[-1]
 757 
 758             # Check if we're already in the correct state
 759             if "3:%s" % action in out and "5:%s" % action in out:

 761                 return
 762

 764         # Systemd's systemctl
 765         #
 766         if self.enable_cmd.endswith("systemctl"):
 767             if self.enable:
 768                 action = 'enable'
 769             else:
 770                 action = 'disable'
 771 
 772             # Check if we're already in the correct state
 773             service_enabled = self.get_systemd_service_enabled()
 774 
 775             # self.changed should already be true
 776             if self.enable == service_enabled:

 778                 return

 781         # OpenRC's rc-update
 782         #

 788

 791                 service_name, runlevels = line.split('|')
 792                 service_name = service_name.strip()
 793                 if service_name != self.name:
 794                     continue

 796                 # service already enabled for the runlevel
 797                 if self.enable and self.runlevel in runlevels:
 798                     self.changed = False
 799                 # service already disabled for the runlevel
 800                 elif not self.enable and self.runlevel not in runlevels:
 801                     self.changed = False
 802                 break
 803             else:
 804                 # service already disabled altogether
 805                 if not self.enable:
 806                     self.changed = False
 807 
 808             if not self.changed:
 809                 return
 810 
 811         #
 812         # update-rc.d style
 813         #
 814         if self.enable_cmd.endswith("update-rc.d"):
 815 
 816             enabled = False
 817             slinks = glob.glob('/etc/rc?.d/S??' + self.name)
 818             if slinks:
 819                 enabled = True
 820 
 821             if self.enable != enabled:
 822                 self.changed = True
 823 
 824                 if self.enable:
 825                     action = 'enable'
 826                     klinks = glob.glob('/etc/rc?.d/K??' + self.name)
 827                     if not klinks:
 828                         if not self.module.check_mode:
 829                             (rc, out, err) = self.execute_command("%s %s defaults" % (self.enable_cmd, self.name))
 830                             if rc != 0:
 831                                 if err:
 832                                     self.module.fail_json(msg=err)
 833                                 else:
 834                                     self.module.fail_json(msg=out) % (self.enable_cmd, self.name, action)
 835                 else:
 836                     action = 'disable'
 837 
 838                 if not self.module.check_mode:
 839                     (rc, out, err) = self.execute_command("%s %s %s" % (self.enable_cmd, self.name, action))
 840                     if rc != 0:
 841                         if err:
 842                             self.module.fail_json(msg=err)
 843                         else:
 844                             self.module.fail_json(msg=out) % (self.enable_cmd, self.name, action)
 845             else:
 846                 self.changed = False
 847 
 848             return
 849 
 850         #
 851         # insserv (Debian <=7, SLES, others)
 852         #
 853         if self.enable_cmd.endswith("insserv"):
 854             if self.enable:
 855                 (rc, out, err) = self.execute_command("%s -n -v %s" % (self.enable_cmd, self.name))
 856             else:
 857                 (rc, out, err) = self.execute_command("%s -n -r -v %s" % (self.enable_cmd, self.name))
 858 
 859             self.changed = False
 860             for line in err.splitlines():
 861                 if self.enable and line.find('enable service') != -1:
 862                     self.changed = True
 863                     break
 864                 if not self.enable and line.find('remove service') != -1:
 865                     self.changed = True
 866                     break
 867 
 868             if self.module.check_mode:
 869                 self.module.exit_json(changed=self.changed)
 870 
 871             if not self.changed:
 872                 return
 873 
 874             if self.enable:
 875                 (rc, out, err) = self.execute_command("%s %s" % (self.enable_cmd, self.name))
 876                 if (rc != 0) or (err != ''):
 877                     self.module.fail_json(msg=("Failed to install service. rc: %s, out: %s, err: %s" % (rc, out, err)))
 878                 return (rc, out, err)
 879             else:
 880                 (rc, out, err) = self.execute_command("%s -r %s" % (self.enable_cmd, self.name))
 881                 if (rc != 0) or (err != ''):
 882                     self.module.fail_json(msg=("Failed to remove service. rc: %s, out: %s, err: %s" % (rc, out, err)))
 883                 return (rc, out, err)
 884 
 885         #
 886         # If we've gotten to the end, the service needs to be updated
 887         #
 888         self.changed = True
 889 
 890         # we change argument order depending on real binary used:
 891         # rc-update and systemctl need the argument order reversed
 892 
 893         if self.enable_cmd.endswith("rc-update"):
 894             args = (self.enable_cmd, action, self.name + " " + self.runlevel)
 895         elif self.enable_cmd.endswith("systemctl"):
 896             args = (self.enable_cmd, action, self.__systemd_unit)
 897         else:
 898             args = (self.enable_cmd, self.name, action)
 899

 901             self.module.exit_json(changed=self.changed)
 902 
 903         (rc, out, err) = self.execute_command("%s %s %s" % args)
 904         if rc != 0:
 905             if err:
 906                 self.module.fail_json(msg="Error when trying to %s %s: rc=%s %s" % (action, self.name, rc, err))
 907             else:
 908                 self.module.fail_json(msg="Failure for %s %s: rc=%s %s" % (action, self.name, rc, out))
 909 
 910         return (rc, out, err)

 912     def service_control(self):
 913 
 914         # Decide what command to run
 915         svc_cmd = ''
 916         arguments = self.arguments
 917         if self.svc_cmd:
 918             if not self.svc_cmd.endswith("systemctl"):
 919                 if self.svc_cmd.endswith("initctl"):
 920                     # initctl commands take the form <cmd> <action> <name>
 921                     svc_cmd = self.svc_cmd
 922                     arguments = "%s %s" % (self.name, arguments)
 923                 else:
 924                     # SysV and OpenRC take the form <cmd> <name> <action>
 925                     svc_cmd = "%s %s" % (self.svc_cmd, self.name)
 926             else:
 927                 # systemd commands take the form <cmd> <action> <name>
 928                 svc_cmd = self.svc_cmd
 929                 arguments = "%s %s" % (self.__systemd_unit, arguments)
 930         elif self.svc_cmd is None and self.svc_initscript:
 931             # upstart
 932             svc_cmd = "%s" % self.svc_initscript
 933 
 934         # In OpenRC, if a service crashed, we need to reset its status to
 935         # stopped with the zap command, before we can start it back.
 936         if self.svc_cmd and self.svc_cmd.endswith('rc-service') and self.action == 'start' and self.crashed:
 937             self.execute_command("%s zap" % svc_cmd, daemonize=True)
 938 
 939         if self.action != "restart":
 940             if svc_cmd != '':
 941                 # upstart or systemd or OpenRC
 942                 rc_state, stdout, stderr = self.execute_command("%s %s %s" % (svc_cmd, self.action, arguments), daemonize=True)
 943             else:
 944                 # SysV
 945                 rc_state, stdout, stderr = self.execute_command("%s %s %s" % (self.action, self.name, arguments), daemonize=True)
 946         elif self.svc_cmd and self.svc_cmd.endswith('rc-service'):
 947             # All services in OpenRC support restart.
 948             rc_state, stdout, stderr = self.execute_command("%s %s %s" % (svc_cmd, self.action, arguments), daemonize=True)
 949         else:
 950             # In other systems, not all services support restart. Do it the hard way.
 951             if svc_cmd != '':
 952                 # upstart or systemd
 953                 rc1, stdout1, stderr1 = self.execute_command("%s %s %s" % (svc_cmd, 'stop', arguments), daemonize=True)
 954             else:
 955                 # SysV
 956                 rc1, stdout1, stderr1 = self.execute_command("%s %s %s" % ('stop', self.name, arguments), daemonize=True)
 957 
 958             if self.sleep:
 959                 time.sleep(self.sleep)
 960 
 961             if svc_cmd != '':
 962                 # upstart or systemd
 963                 rc2, stdout2, stderr2 = self.execute_command("%s %s %s" % (svc_cmd, 'start', arguments), daemonize=True)
 964             else:
 965                 # SysV
 966                 rc2, stdout2, stderr2 = self.execute_command("%s %s %s" % ('start', self.name, arguments), daemonize=True)
 967 
 968             # merge return information
 969             if rc1 != 0 and rc2 == 0:
 970                 rc_state = rc2
 971                 stdout = stdout2
 972                 stderr = stderr2
 973             else:
 974                 rc_state = rc1 + rc2
 975                 stdout = stdout1 + stdout2
 976                 stderr = stderr1 + stderr2
 977 
 978         return (rc_state, stdout, stderr)

1068

1182

1192         else:

1199         # If the user has not supplied any arguments and the current flags
1200         # differ from the default we reset them.
1201         elif not self.arguments and current_flags != default_flags:
1202             changed_flags = ' '
1203         # Otherwise there is no need to modify flags.
1204         else:
1205             changed_flags = ''

1211                 return
1212

1214                 status_action = "set %s status on" % (self.name)
1215             else:
1216                 status_action = ''
1217             if changed_flags:
1218                 flags_action = "set %s flags %s" % (self.name, changed_flags)
1219             else:
1220                 flags_action = ''
1221         else:
1222             if rc == 1:
1223                 return
1224 
1225             status_action = "set %s status off" % self.name
1226             flags_action = ''
1227 
1228         # Verify state assumption
1229         if not status_action and not flags_action:
1230             self.module.fail_json(msg="neither status_action or status_flags is set, this should never happen")
1231 
1232         if self.module.check_mode:
1233             self.module.exit_json(changed=True, msg="changing service enablement")
1234 
1235         status_modified = 0
1236         if status_action:
1237             rc, stdout, stderr = self.execute_command("%s %s" % (self.enable_cmd, status_action))
1238 
1239             if rc != 0:
1240                 if stderr:
1241                     self.module.fail_json(msg=stderr)
1242                 else:
1243                     self.module.fail_json(msg="rcctl failed to modify service status")
1244 
1245             status_modified = 1
1246 
1247         if flags_action:
1248             rc, stdout, stderr = self.execute_command("%s %s" % (self.enable_cmd, flags_action))
1249 
1250             if rc != 0:
1251                 if stderr:
1252                     if status_modified:
1253                         error_message = "rcctl modified service status but failed to set flags: " + stderr
1254                     else:
1255                         error_message = stderr
1256                 else:
1257                     if status_modified:
1258                         error_message = "rcctl modified service status but failed to set flags"
1259                     else:
1260                         error_message = "rcctl failed to modify service flags"
1261

1349     def svcadm_supports_sync(self):
1350         # Support for synchronous restart/refresh is only supported on
1351         # Oracle Solaris >= 11.2
1352         for line in open('/etc/release', 'r').readlines():
1353             m = re.match(r'\s+Oracle Solaris (\d+\.\d+).*', line.rstrip())
1354             if m and m.groups()[0] >= 11.2:

1358         status = self.get_sunos_svcs_status()

1366     def get_sunos_svcs_status(self):
1367         rc, stdout, stderr = self.execute_command("%s %s" % (self.svcs_cmd, self.name))
1368         if rc == 1:
1369             if stderr:
1370                 self.module.fail_json(msg=stderr)
1371             else:
1372                 self.module.fail_json(msg=stdout)
1373 
1374         lines = stdout.rstrip("\n").split("\n")
1375         status = lines[-1].split(" ")[0]
1376         # status is one of: online, offline, degraded, disabled, maintenance, uninitialized
1377         # see man svcs(1)
1378         return status

1380     def service_enable(self):
1381         # Get current service enablement status
1382         rc, stdout, stderr = self.execute_command("%s -l %s" % (self.svcs_cmd, self.name))
1383 
1384         if rc != 0:
1385             if stderr:
1386                 self.module.fail_json(msg=stderr)
1387             else:
1388                 self.module.fail_json(msg=stdout)
1389 
1390         enabled = False
1391         temporary = False
1392 
1393         # look for enabled line, which could be one of:
1394         #    enabled   true (temporary)
1395         #    enabled   false (temporary)
1396         #    enabled   true
1397         #    enabled   false
1398         for line in stdout.split("\n"):
1399             if line.startswith("enabled"):
1400                 if "true" in line:
1401                     enabled = True
1402                 if "temporary" in line:
1403                     temporary = True
1404 
1405         startup_enabled = (enabled and not temporary) or (not enabled and temporary)
1406 
1407         if self.enable and startup_enabled:
1408             return
1409         elif (not self.enable) and (not startup_enabled):
1410             return
1411 
1412         if not self.module.check_mode:
1413             # Mark service as started or stopped (this will have the side effect of
1414             # actually stopping or starting the service)
1415             if self.enable:
1416                 subcmd = "enable -rs"
1417             else:
1418                 subcmd = "disable -s"
1419 
1420             rc, stdout, stderr = self.execute_command("%s %s %s" % (self.svcadm_cmd, subcmd, self.name))
1421 
1422             if rc != 0:
1423                 if stderr:
1424                     self.module.fail_json(msg=stderr)
1425                 else:
1426                     self.module.fail_json(msg=stdout)
1427 
1428         self.changed = True

1430     def service_control(self):
1431         status = self.get_sunos_svcs_status()
1432 
1433         # if starting or reloading, clear maintenance states
1434         if self.action in ['start', 'reload', 'restart'] and status in ['maintenance', 'degraded']:
1435             rc, stdout, stderr = self.execute_command("%s clear %s" % (self.svcadm_cmd, self.name))
1436             if rc != 0:
1437                 return rc, stdout, stderr
1438             status = self.get_sunos_svcs_status()
1439 
1440         if status in ['maintenance', 'degraded']:
1441             self.module.fail_json(msg="Failed to bring service out of %s status." % status)
1442 
1443         if self.action == 'start':
1444             subcmd = "enable -rst"
1445         elif self.action == 'stop':
1446             subcmd = "disable -st"
1447         elif self.action == 'reload':
1448             subcmd = "refresh %s" % (self.svcadm_sync)
1449         elif self.action == 'restart' and status == 'online':
1450             subcmd = "restart %s" % (self.svcadm_sync)
1451         elif self.action == 'restart' and status != 'online':
1452             subcmd = "enable -rst"
1453 
1454         return self.execute_command("%s %s %s" % (self.svcadm_cmd, subcmd, self.name))
