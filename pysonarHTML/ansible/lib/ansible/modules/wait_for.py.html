 243 class TCPConnectionInfo(object):
 244     """
 245     This is a generic TCP Connection Info strategy class that relies
 246     on the psutil module, which is not ideal for targets, but necessary
 247     for cross platform support.
 248 
 249     A subclass may wish to override some or all of these methods.
 250       - _get_exclude_ips()
 251       - get_active_connections()
 252 
 253     All subclasses MUST define platform and distribution (which may be None).
 254     """
 255     platform = 'Generic'
 256     distribution = None
 257 
 258     match_all_ips = {
 259         socket.AF_INET: '0.0.0.0',
 260         socket.AF_INET6: '::',
 261     }
 262     ipv4_mapped_ipv6_address = {
 263         'prefix': '::ffff',
 264         'match_all': '::ffff:0.0.0.0'
 265     }
 266 
 267     def __new__(cls, *args, **kwargs):
 268         new_cls = get_platform_subclass(TCPConnectionInfo)
 269         return super(cls, new_cls).__new__(new_cls)
 270 
 271     def __init__(self, module):
 272         self.module = module
 273         self.ips = _convert_host_to_ip(module.params['host'])
 274         self.port = int(self.module.params['port'])
 275         self.exclude_ips = self._get_exclude_ips()
 276         if not HAS_PSUTIL:
 277             module.fail_json(msg=missing_required_lib('psutil'), exception=PSUTIL_IMP_ERR)
 278 
 279     def _get_exclude_ips(self):
 280         exclude_hosts = self.module.params['exclude_hosts']
 281         exclude_ips = []
 282         if exclude_hosts is not None:
 283             for host in exclude_hosts:
 284                 exclude_ips.extend(_convert_host_to_ip(host))
 285         return exclude_ips
 286 
 287     def get_active_connections_count(self):
 288         active_connections = 0
 289         for p in psutil.process_iter():
 290             if hasattr(p, 'get_connections'):
 291                 connections = p.get_connections(kind='inet')
 292             else:
 293                 connections = p.connections(kind='inet')
 294             for conn in connections:
 295                 if conn.status not in self.module.params['active_connection_states']:
 296                     continue
 297                 if hasattr(conn, 'local_address'):
 298                     (local_ip, local_port) = conn.local_address
 299                 else:
 300                     (local_ip, local_port) = conn.laddr
 301                 if self.port != local_port:
 302                     continue
 303                 if hasattr(conn, 'remote_address'):
 304                     (remote_ip, remote_port) = conn.remote_address
 305                 else:
 306                     (remote_ip, remote_port) = conn.raddr
 307                 if (conn.family, remote_ip) in self.exclude_ips:
 308                     continue
 309                 if any((
 310                     (conn.family, local_ip) in self.ips,
 311                     (conn.family, self.match_all_ips[conn.family]) in self.ips,
 312                     local_ip.startswith(self.ipv4_mapped_ipv6_address['prefix']) and
 313                         (conn.family, self.ipv4_mapped_ipv6_address['match_all']) in self.ips,
 314                 )):
 315                     active_connections += 1
 316         return active_connections

 322 class LinuxTCPConnectionInfo(TCPConnectionInfo):
 323     """
 324     This is a TCP Connection Info evaluation strategy class
 325     that utilizes information from Linux's procfs. While less universal,
 326     does allow Linux targets to not require an additional library.
 327     """
 328     platform = 'Linux'
 329     distribution = None
 330 
 331     source_file = {
 332         socket.AF_INET: '/proc/net/tcp',
 333         socket.AF_INET6: '/proc/net/tcp6'
 334     }
 335     match_all_ips = {
 336         socket.AF_INET: '00000000',
 337         socket.AF_INET6: '00000000000000000000000000000000',
 338     }
 339     ipv4_mapped_ipv6_address = {
 340         'prefix': '0000000000000000FFFF0000',
 341         'match_all': '0000000000000000FFFF000000000000'
 342     }
 343     local_address_field = 1
 344     remote_address_field = 2
 345     connection_state_field = 3
 346 
 347     def __init__(self, module):
 348         self.module = module
 349         self.ips = _convert_host_to_hex(module.params['host'])
 350         self.port = "%0.4X" % int(module.params['port'])
 351         self.exclude_ips = self._get_exclude_ips()
 352 
 353     def _get_exclude_ips(self):
 354         exclude_hosts = self.module.params['exclude_hosts']
 355         exclude_ips = []
 356         if exclude_hosts is not None:
 357             for host in exclude_hosts:
 358                 exclude_ips.extend(_convert_host_to_hex(host))
 359         return exclude_ips
 360 
 361     def get_active_connections_count(self):
 362         active_connections = 0
 363         for family in self.source_file.keys():
 364             if not os.path.isfile(self.source_file[family]):
 365                 continue
 366             f = open(self.source_file[family])
 367             for tcp_connection in f.readlines():
 368                 tcp_connection = tcp_connection.strip().split()
 369                 if tcp_connection[self.local_address_field] == 'local_address':
 370                     continue
 371                 if (tcp_connection[self.connection_state_field] not in
 372                         [get_connection_state_id(_connection_state) for _connection_state in self.module.params['active_connection_states']]):
 373                     continue
 374                 (local_ip, local_port) = tcp_connection[self.local_address_field].split(':')
 375                 if self.port != local_port:
 376                     continue
 377                 (remote_ip, remote_port) = tcp_connection[self.remote_address_field].split(':')
 378                 if (family, remote_ip) in self.exclude_ips:
 379                     continue
 380                 if any((
 381                     (family, local_ip) in self.ips,
 382                     (family, self.match_all_ips[family]) in self.ips,
 383                     local_ip.startswith(self.ipv4_mapped_ipv6_address['prefix']) and
 384                         (family, self.ipv4_mapped_ipv6_address['match_all']) in self.ips,
 385                 )):
 386                     active_connections += 1
 387             f.close()
 388         return active_connections

 391 def _convert_host_to_ip(host):
 392     """
 393     Perform forward DNS resolution on host, IP will give the same IP
 394 
 395     Args:
 396         host: String with either hostname, IPv4, or IPv6 address
 397 
 398     Returns:
 399         List of tuples containing address family and IP
 400     """
 401     addrinfo = socket.getaddrinfo(host, 80, 0, 0, socket.SOL_TCP)
 402     ips = []
 403     for family, socktype, proto, canonname, sockaddr in addrinfo:
 404         ip = sockaddr[0]
 405         ips.append((family, ip))
 406         if family == socket.AF_INET:
 407             ips.append((socket.AF_INET6, "::ffff:" + ip))
 408     return ips

 411 def _convert_host_to_hex(host):
 412     """
 413     Convert the provided host to the format in /proc/net/tcp*
 414 
 415     /proc/net/tcp uses little-endian four byte hex for ipv4
 416     /proc/net/tcp6 uses little-endian per 4B word for ipv6
 417 
 418     Args:
 419         host: String with either hostname, IPv4, or IPv6 address
 420 
 421     Returns:
 422         List of tuples containing address family and the
 423         little-endian converted host
 424     """
 425     ips = []
 426     if host is not None:
 427         for family, ip in _convert_host_to_ip(host):
 428             hexip_nf = binascii.b2a_hex(socket.inet_pton(family, ip))
 429             hexip_hf = ""
 430             for i in range(0, len(hexip_nf), 8):
 431                 ipgroup_nf = hexip_nf[i:i + 8]
 432                 ipgroup_hf = socket.ntohl(int(ipgroup_nf, base=16))
 433                 hexip_hf = "%s%08X" % (hexip_hf, ipgroup_hf)
 434             ips.append((family, hexip_hf))
 435     return ips

 438 def _timedelta_total_seconds(timedelta):
 439     return (
 440         timedelta.microseconds + 0.0 +
 441         (timedelta.seconds + timedelta.days * 24 * 3600) * 10 ** 6) / 10 ** 6

 444 def get_connection_state_id(state):
 445     connection_state_id = {
 446         'ESTABLISHED': '01',
 447         'SYN_SENT': '02',
 448         'SYN_RECV': '03',
 449         'FIN_WAIT1': '04',
 450         'FIN_WAIT2': '05',
 451         'TIME_WAIT': '06',
 452     }
 453     return connection_state_id[state]

 456 def main():
 457 
 458     module = AnsibleModule(
 459         argument_spec=dict(
 460             host=dict(type='str', default='127.0.0.1'),
 461             timeout=dict(type='int', default=300),
 462             connect_timeout=dict(type='int', default=5),
 463             delay=dict(type='int', default=0),
 464             port=dict(type='int'),
 465             active_connection_states=dict(type='list', default=['ESTABLISHED', 'FIN_WAIT1', 'FIN_WAIT2', 'SYN_RECV', 'SYN_SENT', 'TIME_WAIT']),
 466             path=dict(type='path'),
 467             search_regex=dict(type='str'),
 468             state=dict(type='str', default='started', choices=['absent', 'drained', 'present', 'started', 'stopped']),
 469             exclude_hosts=dict(type='list'),
 470             sleep=dict(type='int', default=1),
 471             msg=dict(type='str'),
 472         ),
 473     )
 474 
 475     host = module.params['host']
 476     timeout = module.params['timeout']
 477     connect_timeout = module.params['connect_timeout']
 478     delay = module.params['delay']
 479     port = module.params['port']
 480     state = module.params['state']
 481     path = module.params['path']
 482     search_regex = module.params['search_regex']
 483     msg = module.params['msg']
 484 
 485     if search_regex is not None:
 486         compiled_search_re = re.compile(search_regex, re.MULTILINE)
 487     else:
 488         compiled_search_re = None
 489 
 490     match_groupdict = {}
 491     match_groups = ()
 492 
 493     if port and path:
 494         module.fail_json(msg="port and path parameter can not both be passed to wait_for", elapsed=0)
 495     if path and state == 'stopped':
 496         module.fail_json(msg="state=stopped should only be used for checking a port in the wait_for module", elapsed=0)
 497     if path and state == 'drained':
 498         module.fail_json(msg="state=drained should only be used for checking a port in the wait_for module", elapsed=0)
 499     if module.params['exclude_hosts'] is not None and state != 'drained':
 500         module.fail_json(msg="exclude_hosts should only be with state=drained", elapsed=0)
 501     for _connection_state in module.params['active_connection_states']:
 502         try:
 503             get_connection_state_id(_connection_state)
 504         except Exception:
 505             module.fail_json(msg="unknown active_connection_state (%s) defined" % _connection_state, elapsed=0)
 506 
 507     start = datetime.datetime.utcnow()
 508 
 509     if delay:
 510         time.sleep(delay)
 511 
 512     if not port and not path and state != 'drained':
 513         time.sleep(timeout)
 514     elif state in ['absent', 'stopped']:
 515         # first wait for the stop condition
 516         end = start + datetime.timedelta(seconds=timeout)
 517 
 518         while datetime.datetime.utcnow() < end:
 519             if path:
 520                 try:
 521                     if not os.access(path, os.F_OK):
 522                         break
 523                 except IOError:
 524                     break
 525             elif port:
 526                 try:
 527                     s = socket.create_connection((host, port), connect_timeout)
 528                     s.shutdown(socket.SHUT_RDWR)
 529                     s.close()
 530                 except Exception:
 531                     break
 532             # Conditions not yet met, wait and try again
 533             time.sleep(module.params['sleep'])
 534         else:
 535             elapsed = datetime.datetime.utcnow() - start
 536             if port:
 537                 module.fail_json(msg=msg or "Timeout when waiting for %s:%s to stop." % (host, port), elapsed=elapsed.seconds)
 538             elif path:
 539                 module.fail_json(msg=msg or "Timeout when waiting for %s to be absent." % (path), elapsed=elapsed.seconds)
 540 
 541     elif state in ['started', 'present']:
 542         # wait for start condition
 543         end = start + datetime.timedelta(seconds=timeout)
 544         while datetime.datetime.utcnow() < end:
 545             if path:
 546                 try:
 547                     os.stat(path)
 548                 except OSError as e:
 549                     # If anything except file not present, throw an error
 550                     if e.errno != 2:
 551                         elapsed = datetime.datetime.utcnow() - start
 552                         module.fail_json(msg=msg or "Failed to stat %s, %s" % (path, e.strerror), elapsed=elapsed.seconds)
 553                     # file doesn't exist yet, so continue
 554                 else:
 555                     # File exists.  Are there additional things to check?
 556                     if not compiled_search_re:
 557                         # nope, succeed!
 558                         break
 559                     try:
 560                         f = open(path)
 561                         try:
 562                             search = re.search(compiled_search_re, f.read())
 563                             if search:
 564                                 if search.groupdict():
 565                                     match_groupdict = search.groupdict()
 566                                 if search.groups():
 567                                     match_groups = search.groups()
 568 
 569                                 break
 570                         finally:
 571                             f.close()
 572                     except IOError:
 573                         pass
 574             elif port:
 575                 alt_connect_timeout = math.ceil(_timedelta_total_seconds(end - datetime.datetime.utcnow()))
 576                 try:
 577                     s = socket.create_connection((host, port), min(connect_timeout, alt_connect_timeout))
 578                 except Exception:
 579                     # Failed to connect by connect_timeout. wait and try again
 580                     pass
 581                 else:
 582                     # Connected -- are there additional conditions?
 583                     if compiled_search_re:
 584                         data = ''
 585                         matched = False
 586                         while datetime.datetime.utcnow() < end:
 587                             max_timeout = math.ceil(_timedelta_total_seconds(end - datetime.datetime.utcnow()))
 588                             (readable, w, e) = select.select([s], [], [], max_timeout)
 589                             if not readable:
 590                                 # No new data.  Probably means our timeout
 591                                 # expired
 592                                 continue
 593                             response = s.recv(1024)
 594                             if not response:
 595                                 # Server shutdown
 596                                 break
 597                             data += to_native(response, errors='surrogate_or_strict')
 598                             if re.search(compiled_search_re, data):
 599                                 matched = True
 600                                 break
 601 
 602                         # Shutdown the client socket
 603                         try:
 604                             s.shutdown(socket.SHUT_RDWR)
 605                         except socket.error as e:
 606                             if e.errno != errno.ENOTCONN:
 607                                 raise
 608                         # else, the server broke the connection on its end, assume it's not ready
 609                         else:
 610                             s.close()
 611                         if matched:
 612                             # Found our string, success!
 613                             break
 614                     else:
 615                         # Connection established, success!
 616                         try:
 617                             s.shutdown(socket.SHUT_RDWR)
 618                         except socket.error as e:
 619                             if e.errno != errno.ENOTCONN:
 620                                 raise
 621                         # else, the server broke the connection on its end, assume it's not ready
 622                         else:
 623                             s.close()
 624                         break
 625 
 626             # Conditions not yet met, wait and try again
 627             time.sleep(module.params['sleep'])
 628 
 629         else:   # while-else
 630             # Timeout expired
 631             elapsed = datetime.datetime.utcnow() - start
 632             if port:
 633                 if search_regex:
 634                     module.fail_json(msg=msg or "Timeout when waiting for search string %s in %s:%s" % (search_regex, host, port), elapsed=elapsed.seconds)
 635                 else:
 636                     module.fail_json(msg=msg or "Timeout when waiting for %s:%s" % (host, port), elapsed=elapsed.seconds)
 637             elif path:
 638                 if search_regex:
 639                     module.fail_json(msg=msg or "Timeout when waiting for search string %s in %s" % (search_regex, path), elapsed=elapsed.seconds)
 640                 else:
 641                     module.fail_json(msg=msg or "Timeout when waiting for file %s" % (path), elapsed=elapsed.seconds)
 642 
 643     elif state == 'drained':
 644         # wait until all active connections are gone
 645         end = start + datetime.timedelta(seconds=timeout)
 646         tcpconns = TCPConnectionInfo(module)
 647         while datetime.datetime.utcnow() < end:
 648             try:
 649                 if tcpconns.get_active_connections_count() == 0:
 650                     break
 651             except IOError:
 652                 pass
 653             # Conditions not yet met, wait and try again
 654             time.sleep(module.params['sleep'])
 655         else:
 656             elapsed = datetime.datetime.utcnow() - start
 657             module.fail_json(msg=msg or "Timeout when waiting for %s:%s to drain" % (host, port), elapsed=elapsed.seconds)
 658 
 659     elapsed = datetime.datetime.utcnow() - start
 660     module.exit_json(state=state, port=port, search_regex=search_regex, match_groups=match_groups, match_groupdict=match_groupdict, path=path,
 661                      elapsed=elapsed.seconds)
