 109 def enforce_state(module, params):
 110     """
 111     Add or remove key.
 112     """
 113 
 114     host = params["name"].lower()
 115     key = params.get("key", None)
 116     path = params.get("path")
 117     hash_host = params.get("hash_host")
 118     state = params.get("state")
 119     # Find the ssh-keygen binary
 120     sshkeygen = module.get_bin_path("ssh-keygen", True)
 121 
 122     if not key and state != "absent":
 123         module.fail_json(msg="No key specified when adding a host")
 124 
 125     if key and hash_host:
 126         key = hash_host_key(host, key)
 127 
 128     # Trailing newline in files gets lost, so re-add if necessary
 129     if key and not key.endswith('\n'):
 130         key += '\n'
 131 
 132     sanity_check(module, host, key, sshkeygen)
 133 
 134     found, replace_or_add, found_line = search_for_host_key(module, host, key, path, sshkeygen)
 135 
 136     params['diff'] = compute_diff(path, found_line, replace_or_add, state, key)
 137 
 138     # We will change state if found==True & state!="present"
 139     # or found==False & state=="present"
 140     # i.e found XOR (state=="present")
 141     # Alternatively, if replace is true (i.e. key present, and we must change
 142     # it)
 143     if module.check_mode:
 144         module.exit_json(changed=replace_or_add or (state == "present") != found,
 145                          diff=params['diff'])
 146 
 147     # Now do the work.
 148 
 149     # Only remove whole host if found and no key provided
 150     if found and not key and state == "absent":
 151         module.run_command([sshkeygen, '-R', host, '-f', path], check_rc=True)
 152         params['changed'] = True
 153 
 154     # Next, add a new (or replacing) entry
 155     if replace_or_add or found != (state == "present"):
 156         try:
 157             inf = open(path, "r")
 158         except IOError as e:
 159             if e.errno == errno.ENOENT:
 160                 inf = None
 161             else:
 162                 module.fail_json(msg="Failed to read %s: %s" % (path, str(e)))
 163         try:
 164             with tempfile.NamedTemporaryFile(mode='w+', dir=os.path.dirname(path), delete=False) as outf:
 165                 if inf is not None:
 166                     for line_number, line in enumerate(inf):
 167                         if found_line == (line_number + 1) and (replace_or_add or state == 'absent'):
 168                             continue  # skip this line to replace its key
 169                         outf.write(line)
 170                     inf.close()
 171                 if state == 'present':
 172                     outf.write(key)
 173         except (IOError, OSError) as e:
 174             module.fail_json(msg="Failed to write to file %s: %s" % (path, to_native(e)))
 175         else:
 176             module.atomic_move(outf.name, path)
 177 
 178         params['changed'] = True
 179 
 180     return params

 183 def sanity_check(module, host, key, sshkeygen):
 184     '''Check supplied key is sensible
 185 
 186     host and key are parameters provided by the user; If the host
 187     provided is inconsistent with the key supplied, then this function
 188     quits, providing an error to the user.
 189     sshkeygen is the path to ssh-keygen, found earlier with get_bin_path
 190     '''
 191     # If no key supplied, we're doing a removal, and have nothing to check here.
 192     if not key:
 193         return
 194     # Rather than parsing the key ourselves, get ssh-keygen to do it
 195     # (this is essential for hashed keys, but otherwise useful, as the
 196     # key question is whether ssh-keygen thinks the key matches the host).
 197 
 198     # The approach is to write the key to a temporary file,
 199     # and then attempt to look up the specified host in that file.
 200 
 201     if re.search(r'\S+(\s+)?,(\s+)?', host):
 202         module.fail_json(msg="Comma separated list of names is not supported. "
 203                              "Please pass a single name to lookup in the known_hosts file.")
 204 
 205     with tempfile.NamedTemporaryFile(mode='w+') as outf:
 206         try:
 207             outf.write(key)
 208             outf.flush()
 209         except IOError as e:
 210             module.fail_json(msg="Failed to write to temporary file %s: %s" %
 211                              (outf.name, to_native(e)))
 212 
 213         sshkeygen_command = [sshkeygen, '-F', host, '-f', outf.name]
 214         rc, stdout, stderr = module.run_command(sshkeygen_command)
 215 
 216     if stdout == '':  # host not found
 217         module.fail_json(msg="Host parameter does not match hashed host field in supplied key")

 220 def search_for_host_key(module, host, key, path, sshkeygen):
 221     '''search_for_host_key(module,host,key,path,sshkeygen) -> (found,replace_or_add,found_line)
 222 
 223     Looks up host and keytype in the known_hosts file path; if it's there, looks to see
 224     if one of those entries matches key. Returns:
 225     found (Boolean): is host found in path?
 226     replace_or_add (Boolean): is the key in path different to that supplied by user?
 227     found_line (int or None): the line where a key of the same type was found
 228     if found=False, then replace is always False.
 229     sshkeygen is the path to ssh-keygen, found earlier with get_bin_path
 230     '''
 231     if os.path.exists(path) is False:
 232         return False, False, None
 233 
 234     sshkeygen_command = [sshkeygen, '-F', host, '-f', path]
 235 
 236     # openssh >=6.4 has changed ssh-keygen behaviour such that it returns
 237     # 1 if no host is found, whereas previously it returned 0
 238     rc, stdout, stderr = module.run_command(sshkeygen_command, check_rc=False)
 239     if stdout == '' and stderr == '' and (rc == 0 or rc == 1):
 240         return False, False, None  # host not found, no other errors
 241     if rc != 0:  # something went wrong
 242         module.fail_json(msg="ssh-keygen failed (rc=%d, stdout='%s',stderr='%s')" % (rc, stdout, stderr))
 243 
 244     # If user supplied no key, we don't want to try and replace anything with it
 245     if not key:
 246         return True, False, None
 247 
 248     lines = stdout.split('\n')
 249     new_key = normalize_known_hosts_key(key)
 250 
 251     for lnum, l in enumerate(lines):
 252         if l == '':
 253             continue
 254         elif l[0] == '#':  # info output from ssh-keygen; contains the line number where key was found
 255             try:
 256                 # This output format has been hardcoded in ssh-keygen since at least OpenSSH 4.0
 257                 # It always outputs the non-localized comment before the found key
 258                 found_line = int(re.search(r'found: line (\d+)', l).group(1))
 259             except IndexError:
 260                 module.fail_json(msg="failed to parse output of ssh-keygen for line number: '%s'" % l)
 261         else:
 262             found_key = normalize_known_hosts_key(l)
 263             if new_key['host'][:3] == '|1|' and found_key['host'][:3] == '|1|':  # do not change host hash if already hashed
 264                 new_key['host'] = found_key['host']
 265             if new_key == found_key:  # found a match
 266                 return True, False, found_line  # found exactly the same key, don't replace
 267             elif new_key['type'] == found_key['type']:  # found a different key for the same key type
 268                 return True, True, found_line
 269 
 270     # No match found, return found and replace, but no line
 271     return True, True, None

 274 def hash_host_key(host, key):
 275     hmac_key = os.urandom(20)
 276     hashed_host = hmac.new(hmac_key, to_bytes(host), hashlib.sha1).digest()
 277     parts = key.strip().split()
 278     # @ indicates the optional marker field used for @cert-authority or @revoked
 279     i = 1 if parts[0][0] == '@' else 0
 280     parts[i] = '|1|%s|%s' % (to_native(base64.b64encode(hmac_key)), to_native(base64.b64encode(hashed_host)))
 281     return ' '.join(parts)

 284 def normalize_known_hosts_key(key):
 285     '''
 286     Transform a key, either taken from a known_host file or provided by the
 287     user, into a normalized form.
 288     The host part (which might include multiple hostnames or be hashed) gets
 289     replaced by the provided host. Also, any spurious information gets removed
 290     from the end (like the username@host tag usually present in hostkeys, but
 291     absent in known_hosts files)
 292     '''
 293     key = key.strip()  # trim trailing newline
 294     k = key.split()
 295     d = dict()
 296     # The optional "marker" field, used for @cert-authority or @revoked
 297     if k[0][0] == '@':
 298         d['options'] = k[0]
 299         d['host'] = k[1]
 300         d['type'] = k[2]
 301         d['key'] = k[3]
 302     else:
 303         d['host'] = k[0]
 304         d['type'] = k[1]
 305         d['key'] = k[2]
 306     return d

 309 def compute_diff(path, found_line, replace_or_add, state, key):
 310     diff = {
 311         'before_header': path,
 312         'after_header': path,
 313         'before': '',
 314         'after': '',
 315     }
 316     try:
 317         inf = open(path, "r")
 318     except IOError as e:
 319         if e.errno == errno.ENOENT:
 320             diff['before_header'] = '/dev/null'
 321     else:
 322         diff['before'] = inf.read()
 323         inf.close()
 324     lines = diff['before'].splitlines(1)
 325     if (replace_or_add or state == 'absent') and found_line is not None and 1 <= found_line <= len(lines):
 326         del lines[found_line - 1]
 327     if state == 'present' and (replace_or_add or found_line is None):
 328         lines.append(key)
 329     diff['after'] = ''.join(lines)
 330     return diff
