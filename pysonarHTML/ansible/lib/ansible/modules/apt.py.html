 347 class PolicyRcD(object):
 348     """
 349     This class is a context manager for the /usr/sbin/policy-rc.d file.
 350     It allow the user to prevent dpkg to start the corresponding service when installing
 351     a package.
 352     https://people.debian.org/~hmh/invokerc.d-policyrc.d-specification.txt
 353     """
 354 
 355     def __init__(self, module):
 356         # we need the module for later use (eg. fail_json)
 357         self.m = module
 358 
 359         # if policy_rc_d is null then we don't need to modify policy-rc.d
 360         if self.m.params['policy_rc_d'] is None:
 361             return
 362 
 363         # if the /usr/sbin/policy-rc.d already exist
 364         # we will back it up during package installation
 365         # then restore it
 366         if os.path.exists('/usr/sbin/policy-rc.d'):
 367             self.backup_dir = tempfile.mkdtemp(prefix="ansible")
 368         else:
 369             self.backup_dir = None
 370 
 371     def __enter__(self):
 372         """
 373         This method will be call when we enter the context, before we call `apt-get …`
 374         """
 375 
 376         # if policy_rc_d is null then we don't need to modify policy-rc.d
 377         if self.m.params['policy_rc_d'] is None:
 378             return
 379 
 380         # if the /usr/sbin/policy-rc.d already exist we back it up
 381         if self.backup_dir:
 382             try:
 383                 shutil.move('/usr/sbin/policy-rc.d', self.backup_dir)
 384             except Exception:
 385                 self.m.fail_json(msg="Fail to move /usr/sbin/policy-rc.d to %s" % self.backup_dir)
 386 
 387         # we write /usr/sbin/policy-rc.d so it always exit with code policy_rc_d
 388         try:
 389             with open('/usr/sbin/policy-rc.d', 'w') as policy_rc_d:
 390                 policy_rc_d.write('#!/bin/sh\nexit %d\n' % self.m.params['policy_rc_d'])
 391 
 392             os.chmod('/usr/sbin/policy-rc.d', 0o0755)
 393         except Exception:
 394             self.m.fail_json(msg="Failed to create or chmod /usr/sbin/policy-rc.d")
 395 
 396     def __exit__(self, type, value, traceback):
 397         """
 398         This method will be call when we enter the context, before we call `apt-get …`
 399         """
 400 
 401         # if policy_rc_d is null then we don't need to modify policy-rc.d
 402         if self.m.params['policy_rc_d'] is None:
 403             return
 404 
 405         if self.backup_dir:
 406             # if /usr/sbin/policy-rc.d already exists before the call to __enter__
 407             # we restore it (from the backup done in __enter__)
 408             try:
 409                 shutil.move(os.path.join(self.backup_dir, 'policy-rc.d'),
 410                             '/usr/sbin/policy-rc.d')
 411                 os.rmdir(self.tmpdir_name)
 412             except Exception:
 413                 self.m.fail_json(msg="Fail to move back %s to /usr/sbin/policy-rc.d"
 414                                      % os.path.join(self.backup_dir, 'policy-rc.d'))
 415         else:
 416             # if they wheren't any /usr/sbin/policy-rc.d file before the call to __enter__
 417             # we just remove the file
 418             try:
 419                 os.remove('/usr/sbin/policy-rc.d')
 420             except Exception:
 421                 self.m.fail_json(msg="Fail to remove /usr/sbin/policy-rc.d (after package manipulation)")

 424 def package_split(pkgspec):
 425     parts = pkgspec.split('=', 1)
 426     version = None
 427     if len(parts) > 1:
 428         version = parts[1]
 429     return parts[0], version

 432 def package_versions(pkgname, pkg, pkg_cache):
 433     try:
 434         versions = set(p.version for p in pkg.versions)
 435     except AttributeError:
 436         # assume older version of python-apt is installed
 437         # apt.package.Package#versions require python-apt >= 0.7.9.
 438         pkg_cache_list = (p for p in pkg_cache.Packages if p.Name == pkgname)
 439         pkg_versions = (p.VersionList for p in pkg_cache_list)
 440         versions = set(p.VerStr for p in itertools.chain(*pkg_versions))
 441 
 442     return versions

 452 def package_status(m, pkgname, version, cache, state):
 453     try:
 454         # get the package from the cache, as well as the
 455         # low-level apt_pkg.Package object which contains
 456         # state fields not directly accessible from the
 457         # higher-level apt.package.Package object.
 458         pkg = cache[pkgname]
 459         ll_pkg = cache._cache[pkgname]  # the low-level package object
 460     except KeyError:
 461         if state == 'install':
 462             try:
 463                 provided_packages = cache.get_providing_packages(pkgname)
 464                 if provided_packages:
 465                     is_installed = False
 466                     upgradable = False
 467                     version_ok = False
 468                     # when virtual package providing only one package, look up status of target package
 469                     if cache.is_virtual_package(pkgname) and len(provided_packages) == 1:
 470                         package = provided_packages[0]
 471                         installed, version_ok, upgradable, has_files = package_status(m, package.name, version, cache, state='install')
 472                         if installed:
 473                             is_installed = True
 474                     return is_installed, version_ok, upgradable, False
 475                 m.fail_json(msg="No package matching '%s' is available" % pkgname)
 476             except AttributeError:
 477                 # python-apt version too old to detect virtual packages
 478                 # mark as upgradable and let apt-get install deal with it
 479                 return False, False, True, False
 480         else:
 481             return False, False, False, False
 482     try:
 483         has_files = len(pkg.installed_files) > 0
 484     except UnicodeDecodeError:
 485         has_files = True
 486     except AttributeError:
 487         has_files = False  # older python-apt cannot be used to determine non-purged
 488 
 489     try:
 490         package_is_installed = ll_pkg.current_state == apt_pkg.CURSTATE_INSTALLED
 491     except AttributeError:  # python-apt 0.7.X has very weak low-level object
 492         try:
 493             # might not be necessary as python-apt post-0.7.X should have current_state property
 494             package_is_installed = pkg.is_installed
 495         except AttributeError:
 496             # assume older version of python-apt is installed
 497             package_is_installed = pkg.isInstalled
 498 
 499     version_is_installed = package_is_installed
 500     if version:
 501         versions = package_versions(pkgname, pkg, cache._cache)
 502         avail_upgrades = fnmatch.filter(versions, version)
 503 
 504         if package_is_installed:
 505             try:
 506                 installed_version = pkg.installed.version
 507             except AttributeError:
 508                 installed_version = pkg.installedVersion
 509 
 510             # check if the version is matched as well
 511             version_is_installed = fnmatch.fnmatch(installed_version, version)
 512 
 513             # Only claim the package is upgradable if a candidate matches the version
 514             package_is_upgradable = False
 515             for candidate in avail_upgrades:
 516                 if package_version_compare(candidate, installed_version) > 0:
 517                     package_is_upgradable = True
 518                     break
 519         else:
 520             package_is_upgradable = bool(avail_upgrades)
 521     else:
 522         try:
 523             package_is_upgradable = pkg.is_upgradable
 524         except AttributeError:
 525             # assume older version of python-apt is installed
 526             package_is_upgradable = pkg.isUpgradable
 527 
 528     return package_is_installed, version_is_installed, package_is_upgradable, has_files

 531 def expand_dpkg_options(dpkg_options_compressed):
 532     options_list = dpkg_options_compressed.split(',')
 533     dpkg_options = ""
 534     for dpkg_option in options_list:
 535         dpkg_options = '%s -o "Dpkg::Options::=--%s"' \
 536                        % (dpkg_options, dpkg_option)
 537     return dpkg_options.strip()

 540 def expand_pkgspec_from_fnmatches(m, pkgspec, cache):
 541     # Note: apt-get does implicit regex matching when an exact package name
 542     # match is not found.  Something like this:
 543     # matches = [pkg.name for pkg in cache if re.match(pkgspec, pkg.name)]
 544     # (Should also deal with the ':' for multiarch like the fnmatch code below)
 545     #
 546     # We have decided not to do similar implicit regex matching but might take
 547     # a PR to add some sort of explicit regex matching:
 548     # https://github.com/ansible/ansible-modules-core/issues/1258
 549     new_pkgspec = []
 550     if pkgspec:
 551         for pkgspec_pattern in pkgspec:
 552             pkgname_pattern, version = package_split(pkgspec_pattern)
 553 
 554             # note that none of these chars is allowed in a (debian) pkgname
 555             if frozenset('*?[]!').intersection(pkgname_pattern):
 556                 # handle multiarch pkgnames, the idea is that "apt*" should
 557                 # only select native packages. But "apt*:i386" should still work
 558                 if ":" not in pkgname_pattern:
 559                     # Filter the multiarch packages from the cache only once
 560                     try:
 561                         pkg_name_cache = _non_multiarch
 562                     except NameError:
 563                         pkg_name_cache = _non_multiarch = [pkg.name for pkg in cache if ':' not in pkg.name]  # noqa: F841
 564                 else:
 565                     # Create a cache of pkg_names including multiarch only once
 566                     try:
 567                         pkg_name_cache = _all_pkg_names
 568                     except NameError:
 569                         pkg_name_cache = _all_pkg_names = [pkg.name for pkg in cache]  # noqa: F841
 570 
 571                 matches = fnmatch.filter(pkg_name_cache, pkgname_pattern)
 572 
 573                 if not matches:
 574                     m.fail_json(msg="No package(s) matching '%s' available" % str(pkgname_pattern))
 575                 else:
 576                     new_pkgspec.extend(matches)
 577             else:
 578                 # No wildcards in name
 579                 new_pkgspec.append(pkgspec_pattern)
 580     return new_pkgspec

 583 def parse_diff(output):
 584     diff = to_native(output).splitlines()
 585     try:
 586         # check for start marker from aptitude
 587         diff_start = diff.index('Resolving dependencies...')
 588     except ValueError:
 589         try:
 590             # check for start marker from apt-get
 591             diff_start = diff.index('Reading state information...')
 592         except ValueError:
 593             # show everything
 594             diff_start = -1
 595     try:
 596         # check for end marker line from both apt-get and aptitude
 597         diff_end = next(i for i, item in enumerate(diff) if re.match('[0-9]+ (packages )?upgraded', item))
 598     except StopIteration:
 599         diff_end = len(diff)
 600     diff_start += 1
 601     diff_end += 1
 602     return {'prepared': '\n'.join(diff[diff_start:diff_end])}

 627 def install(m, pkgspec, cache, upgrade=False, default_release=None,
 628             install_recommends=None, force=False,
 629             dpkg_options=expand_dpkg_options(DPKG_OPTIONS),
 630             build_dep=False, fixed=False, autoremove=False, fail_on_autoremove=False, only_upgrade=False,
 631             allow_unauthenticated=False):
 632     pkg_list = []
 633     packages = ""
 634     pkgspec = expand_pkgspec_from_fnmatches(m, pkgspec, cache)
 635     package_names = []
 636     for package in pkgspec:
 637         if build_dep:
 638             # Let apt decide what to install
 639             pkg_list.append("'%s'" % package)
 640             continue
 641 
 642         name, version = package_split(package)
 643         package_names.append(name)
 644         installed, installed_version, upgradable, has_files = package_status(m, name, version, cache, state='install')
 645         if (not installed and not only_upgrade) or (installed and not installed_version) or (upgrade and upgradable):
 646             pkg_list.append("'%s'" % package)
 647         if installed_version and upgradable and version:
 648             # This happens when the package is installed, a newer version is
 649             # available, and the version is a wildcard that matches both
 650             #
 651             # We do not apply the upgrade flag because we cannot specify both
 652             # a version and state=latest.  (This behaviour mirrors how apt
 653             # treats a version with wildcard in the package)
 654             pkg_list.append("'%s'" % package)
 655     packages = ' '.join(pkg_list)
 656 
 657     if packages:
 658         if force:
 659             force_yes = '--force-yes'
 660         else:
 661             force_yes = ''
 662 
 663         if m.check_mode:
 664             check_arg = '--simulate'
 665         else:
 666             check_arg = ''
 667 
 668         if autoremove:
 669             autoremove = '--auto-remove'
 670         else:
 671             autoremove = ''
 672 
 673         if fail_on_autoremove:
 674             fail_on_autoremove = '--no-remove'
 675         else:
 676             fail_on_autoremove = ''
 677 
 678         if only_upgrade:
 679             only_upgrade = '--only-upgrade'
 680         else:
 681             only_upgrade = ''
 682 
 683         if fixed:
 684             fixed = '--fix-broken'
 685         else:
 686             fixed = ''
 687 
 688         if build_dep:
 689             cmd = "%s -y %s %s %s %s %s %s build-dep %s" % (APT_GET_CMD, dpkg_options, only_upgrade, fixed, force_yes, fail_on_autoremove, check_arg, packages)
 690         else:
 691             cmd = "%s -y %s %s %s %s %s %s %s install %s" % \
 692                   (APT_GET_CMD, dpkg_options, only_upgrade, fixed, force_yes, autoremove, fail_on_autoremove, check_arg, packages)
 693 
 694         if default_release:
 695             cmd += " -t '%s'" % (default_release,)
 696 
 697         if install_recommends is False:
 698             cmd += " -o APT::Install-Recommends=no"
 699         elif install_recommends is True:
 700             cmd += " -o APT::Install-Recommends=yes"
 701         # install_recommends is None uses the OS default
 702 
 703         if allow_unauthenticated:
 704             cmd += " --allow-unauthenticated"
 705 
 706         with PolicyRcD(m):
 707             rc, out, err = m.run_command(cmd)
 708 
 709         if m._diff:
 710             diff = parse_diff(out)
 711         else:
 712             diff = {}
 713         status = True
 714 
 715         changed = True
 716         if build_dep:
 717             changed = APT_GET_ZERO not in out
 718 
 719         data = dict(changed=changed, stdout=out, stderr=err, diff=diff)
 720         if rc:
 721             status = False
 722             data = dict(msg="'%s' failed: %s" % (cmd, err), stdout=out, stderr=err, rc=rc)
 723     else:
 724         status = True
 725         data = dict(changed=False)
 726 
 727     if not build_dep:
 728         mark_installed_manually(m, package_names)
 729 
 730     return (status, data)

 742 def install_deb(m, debs, cache, force, fail_on_autoremove, install_recommends, allow_unauthenticated, dpkg_options):
 743     changed = False
 744     deps_to_install = []
 745     pkgs_to_install = []
 746     for deb_file in debs.split(','):
 747         try:
 748             pkg = apt.debfile.DebPackage(deb_file)
 749             pkg_name = get_field_of_deb(m, deb_file, "Package")
 750             pkg_version = get_field_of_deb(m, deb_file, "Version")
 751             if len(apt_pkg.get_architectures()) > 1:
 752                 pkg_arch = get_field_of_deb(m, deb_file, "Architecture")
 753                 pkg_key = "%s:%s" % (pkg_name, pkg_arch)
 754             else:
 755                 pkg_key = pkg_name
 756             try:
 757                 installed_pkg = apt.Cache()[pkg_key]
 758                 installed_version = installed_pkg.installed.version
 759                 if package_version_compare(pkg_version, installed_version) == 0:
 760                     # Does not need to down-/upgrade, move on to next package
 761                     continue
 762             except Exception:
 763                 # Must not be installed, continue with installation
 764                 pass
 765             # Check if package is installable
 766             if not pkg.check() and not force:
 767                 m.fail_json(msg=pkg._failure_string)
 768 
 769             # add any missing deps to the list of deps we need
 770             # to install so they're all done in one shot
 771             deps_to_install.extend(pkg.missing_deps)
 772 
 773         except Exception as e:
 774             m.fail_json(msg="Unable to install package: %s" % to_native(e))
 775 
 776         # and add this deb to the list of packages to install
 777         pkgs_to_install.append(deb_file)
 778 
 779     # install the deps through apt
 780     retvals = {}
 781     if deps_to_install:
 782         (success, retvals) = install(m=m, pkgspec=deps_to_install, cache=cache,
 783                                      install_recommends=install_recommends,
 784                                      fail_on_autoremove=fail_on_autoremove,
 785                                      allow_unauthenticated=allow_unauthenticated,
 786                                      dpkg_options=expand_dpkg_options(dpkg_options))
 787         if not success:
 788             m.fail_json(**retvals)
 789         changed = retvals.get('changed', False)
 790 
 791     if pkgs_to_install:
 792         options = ' '.join(["--%s" % x for x in dpkg_options.split(",")])
 793         if m.check_mode:
 794             options += " --simulate"
 795         if force:
 796             options += " --force-all"
 797 
 798         cmd = "dpkg %s -i %s" % (options, " ".join(pkgs_to_install))
 799 
 800         with PolicyRcD(m):
 801             rc, out, err = m.run_command(cmd)
 802 
 803         if "stdout" in retvals:
 804             stdout = retvals["stdout"] + out
 805         else:
 806             stdout = out
 807         if "diff" in retvals:
 808             diff = retvals["diff"]
 809             if 'prepared' in diff:
 810                 diff['prepared'] += '\n\n' + out
 811         else:
 812             diff = parse_diff(out)
 813         if "stderr" in retvals:
 814             stderr = retvals["stderr"] + err
 815         else:
 816             stderr = err
 817 
 818         if rc == 0:
 819             m.exit_json(changed=True, stdout=stdout, stderr=stderr, diff=diff)
 820         else:
 821             m.fail_json(msg="%s failed" % cmd, stdout=stdout, stderr=stderr)
 822     else:
 823         m.exit_json(changed=changed, stdout=retvals.get('stdout', ''), stderr=retvals.get('stderr', ''), diff=retvals.get('diff', ''))

 912 def upgrade(m, mode="yes", force=False, default_release=None,
 913             use_apt_get=False,
 914             dpkg_options=expand_dpkg_options(DPKG_OPTIONS), autoremove=False, fail_on_autoremove=False,
 915             allow_unauthenticated=False,
 916             ):
 917 
 918     if autoremove:
 919         autoremove = '--auto-remove'
 920     else:
 921         autoremove = ''
 922 
 923     if m.check_mode:
 924         check_arg = '--simulate'
 925     else:
 926         check_arg = ''
 927 
 928     apt_cmd = None
 929     prompt_regex = None
 930     if mode == "dist" or (mode == "full" and use_apt_get):
 931         # apt-get dist-upgrade
 932         apt_cmd = APT_GET_CMD
 933         upgrade_command = "dist-upgrade %s" % (autoremove)
 934     elif mode == "full" and not use_apt_get:
 935         # aptitude full-upgrade
 936         apt_cmd = APTITUDE_CMD
 937         upgrade_command = "full-upgrade"
 938     else:
 939         if use_apt_get:
 940             apt_cmd = APT_GET_CMD
 941             upgrade_command = "upgrade --with-new-pkgs %s" % (autoremove)
 942         else:
 943             # aptitude safe-upgrade # mode=yes # default
 944             apt_cmd = APTITUDE_CMD
 945             upgrade_command = "safe-upgrade"
 946             prompt_regex = r"(^Do you want to ignore this warning and proceed anyway\?|^\*\*\*.*\[default=.*\])"
 947 
 948     if force:
 949         if apt_cmd == APT_GET_CMD:
 950             force_yes = '--force-yes'
 951         else:
 952             force_yes = '--assume-yes --allow-untrusted'
 953     else:
 954         force_yes = ''
 955 
 956     if fail_on_autoremove:
 957         fail_on_autoremove = '--no-remove'
 958     else:
 959         fail_on_autoremove = ''
 960 
 961     allow_unauthenticated = '--allow-unauthenticated' if allow_unauthenticated else ''
 962 
 963     if apt_cmd is None:
 964         if use_apt_get:
 965             apt_cmd = APT_GET_CMD
 966         else:
 967             m.fail_json(msg="Unable to find APTITUDE in path. Please make sure "
 968                             "to have APTITUDE in path or use 'force_apt_get=True'")
 969     apt_cmd_path = m.get_bin_path(apt_cmd, required=True)
 970 
 971     cmd = '%s -y %s %s %s %s %s %s' % (apt_cmd_path, dpkg_options, force_yes, fail_on_autoremove, allow_unauthenticated, check_arg, upgrade_command)
 972 
 973     if default_release:
 974         cmd += " -t '%s'" % (default_release,)
 975 
 976     with PolicyRcD(m):
 977         rc, out, err = m.run_command(cmd, prompt_regex=prompt_regex)
 978 
 979     if m._diff:
 980         diff = parse_diff(out)
 981     else:
 982         diff = {}
 983     if rc:
 984         m.fail_json(msg="'%s %s' failed: %s" % (apt_cmd, upgrade_command, err), stdout=out, rc=rc)
 985     if (apt_cmd == APT_GET_CMD and APT_GET_ZERO in out) or (apt_cmd == APTITUDE_CMD and APTITUDE_ZERO in out):
 986         m.exit_json(changed=False, msg=out, stdout=out, stderr=err)
 987     m.exit_json(changed=True, msg=out, stdout=out, stderr=err, diff=diff)

 993     :returns: ``int``

1003 def get_updated_cache_time():
1004     """Return the mtime time stamp and the updated cache time.
1005     Always retrieve the mtime of the apt cache or set the `cache_mtime`
1006     variable to 0
1007     :returns: ``tuple``
1008     """
1009     cache_mtime = get_cache_mtime()

1011     updated_cache_time = int(time.mktime(mtimestamp.timetuple()))

1015 # https://github.com/ansible/ansible-modules-core/issues/2951
1016 def get_cache(module):
1017     '''Attempt to get the cache object and update till it works'''
1018     cache = None
1019     try:
1020         cache = apt.Cache()
1021     except SystemError as e:
1022         if '/var/lib/apt/lists/' in to_native(e).lower():
1023             # update cache until files are fixed or retries exceeded
1024             retries = 0
1025             while retries < 2:
1026                 (rc, so, se) = module.run_command(['apt-get', 'update', '-q'])
1027                 retries += 1
1028                 if rc == 0:
1029                     break
1030             if rc != 0:
1031                 module.fail_json(msg='Updating the cache to correct corrupt package lists failed:\n%s\n%s' % (to_native(e), so + se), rc=rc)
1032             # try again
1033             cache = apt.Cache()
1034         else:
1035             module.fail_json(msg=to_native(e))
1036     return cache

1039 def main():

1067 
1068     module.run_command_environ_update = APT_ENV_VARS
1069 
1070     if not HAS_PYTHON_APT:
1071         if module.check_mode:
1072             module.fail_json(msg="%s must be installed to use check mode. "
1073                                  "If run normally this module can auto-install it." % PYTHON_APT)
1074         try:
1075             # We skip cache update in auto install the dependency if the
1076             # user explicitly declared it with update_cache=no.
1077             if module.params.get('update_cache') is False:
1078                 module.warn("Auto-installing missing dependency without updating cache: %s" % PYTHON_APT)
1079             else:
1080                 module.warn("Updating cache and auto-installing missing dependency: %s" % PYTHON_APT)
1081                 module.run_command(['apt-get', 'update'], check_rc=True)
1082 
1083             module.run_command(['apt-get', 'install', '--no-install-recommends', PYTHON_APT, '-y', '-q'], check_rc=True)
1084             global apt, apt_pkg
1085             import apt
1086             import apt.debfile
1087             import apt_pkg
1088         except ImportError:
1089             module.fail_json(msg="Could not import python modules: apt, apt_pkg. "
1090                                  "Please install %s package." % PYTHON_APT)
1091 
1092     global APTITUDE_CMD
1093     APTITUDE_CMD = module.get_bin_path("aptitude", False)
1094     global APT_GET_CMD
1095     APT_GET_CMD = module.get_bin_path("apt-get")
1096 
1097     p = module.params
1098 
1099     if p['upgrade'] == 'no':
1100         p['upgrade'] = None
1101 
1102     use_apt_get = p['force_apt_get']
1103 
1104     if not use_apt_get and not APTITUDE_CMD:
1105         use_apt_get = True
1106 
1107     updated_cache = False
1108     updated_cache_time = 0
1109     install_recommends = p['install_recommends']
1110     allow_unauthenticated = p['allow_unauthenticated']
1111     dpkg_options = expand_dpkg_options(p['dpkg_options'])
1112     autoremove = p['autoremove']
1113     fail_on_autoremove = p['fail_on_autoremove']
1114     autoclean = p['autoclean']
1115 
1116     # Get the cache object
1117     cache = get_cache(module)
1118 
1119     try:
1120         if p['default_release']:
1121             try:
1122                 apt_pkg.config['APT::Default-Release'] = p['default_release']
1123             except AttributeError:
1124                 apt_pkg.Config['APT::Default-Release'] = p['default_release']
1125             # reopen cache w/ modified config
1126             cache.open(progress=None)
1127 
1128         mtimestamp, updated_cache_time = get_updated_cache_time()
1129         # Cache valid time is default 0, which will update the cache if
1130         #  needed and `update_cache` was set to true
1131         updated_cache = False
1132         if p['update_cache'] or p['cache_valid_time']:
1133             now = datetime.datetime.now()
1134             tdelta = datetime.timedelta(seconds=p['cache_valid_time'])
1135             if not mtimestamp + tdelta >= now:
1136                 # Retry to update the cache with exponential backoff
1137                 err = ''
1138                 update_cache_retries = module.params.get('update_cache_retries')
1139                 update_cache_retry_max_delay = module.params.get('update_cache_retry_max_delay')
1140                 randomize = random.randint(0, 1000) / 1000.0
1141 
1142                 for retry in range(update_cache_retries):
1143                     try:
1144                         cache.update()
1145                         break
1146                     except apt.cache.FetchFailedException as e:
1147                         err = to_native(e)
1148 
1149                     # Use exponential backoff plus a little bit of randomness
1150                     delay = 2 ** retry + randomize
1151                     if delay > update_cache_retry_max_delay:
1152                         delay = update_cache_retry_max_delay + randomize
1153                     time.sleep(delay)
1154                 else:
1155                     module.fail_json(msg='Failed to update apt cache: %s' % (err if err else 'unknown reason'))
1156 
1157                 cache.open(progress=None)
1158                 mtimestamp, post_cache_update_time = get_updated_cache_time()
1159                 if updated_cache_time != post_cache_update_time:
1160                     updated_cache = True
1161                 updated_cache_time = post_cache_update_time
1162 
1163             # If there is nothing else to do exit. This will set state as
1164             #  changed based on if the cache was updated.
1165             if not p['package'] and not p['upgrade'] and not p['deb']:
1166                 module.exit_json(
1167                     changed=updated_cache,
1168                     cache_updated=updated_cache,
1169                     cache_update_time=updated_cache_time
1170                 )
1171
