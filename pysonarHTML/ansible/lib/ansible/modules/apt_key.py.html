 133 def parse_key_id(key_id):
 134     """validate the key_id and break it into segments
 135 
 136     :arg key_id: The key_id as supplied by the user.  A valid key_id will be
 137         8, 16, or more hexadecimal chars with an optional leading ``0x``.
 138     :returns: The portion of key_id suitable for apt-key del, the portion
 139         suitable for comparisons with --list-public-keys, and the portion that
 140         can be used with --recv-key.  If key_id is long enough, these will be
 141         the last 8 characters of key_id, the last 16 characters, and all of
 142         key_id.  If key_id is not long enough, some of the values will be the
 143         same.
 144 
 145     * apt-key del <= 1.10 has a bug with key_id != 8 chars
 146     * apt-key adv --list-public-keys prints 16 chars
 147     * apt-key adv --recv-key can take more chars
 148 
 149     """
 150     # Make sure the key_id is valid hexadecimal
 151     int(key_id, 16)
 152 
 153     key_id = key_id.upper()
 154     if key_id.startswith('0X'):
 155         key_id = key_id[2:]
 156 
 157     key_id_len = len(key_id)
 158     if (key_id_len != 8 and key_id_len != 16) and key_id_len <= 16:
 159         raise ValueError('key_id must be 8, 16, or 16+ hexadecimal characters in length')
 160 
 161     short_key_id = key_id[-8:]
 162 
 163     fingerprint = key_id
 164     if key_id_len > 16:
 165         fingerprint = key_id[-16:]
 166 
 167     return short_key_id, fingerprint, key_id

 170 def all_keys(module, keyring, short_format):
 171     if keyring:
 172         cmd = "%s --keyring %s adv --list-public-keys --keyid-format=long" % (apt_key_bin, keyring)
 173     else:
 174         cmd = "%s adv --list-public-keys --keyid-format=long" % apt_key_bin
 175     (rc, out, err) = module.run_command(cmd)
 176     results = []
 177     lines = to_native(out).split('\n')
 178     for line in lines:
 179         if (line.startswith("pub") or line.startswith("sub")) and "expired" not in line:
 180             tokens = line.split()
 181             code = tokens[1]
 182             (len_type, real_code) = code.split("/")
 183             results.append(real_code)
 184     if short_format:
 185         results = shorten_key_ids(results)
 186     return results

 189 def shorten_key_ids(key_id_list):
 190     """
 191     Takes a list of key ids, and converts them to the 'short' format,
 192     by reducing them to their last 8 characters.
 193     """
 194     short = []
 195     for key in key_id_list:
 196         short.append(key[-8:])
 197     return short

 200 def download_key(module, url):
 201     # FIXME: move get_url code to common, allow for in-memory D/L, support proxies
 202     # and reuse here
 203     if url is None:
 204         module.fail_json(msg="needed a URL but was not specified")
 205 
 206     try:
 207         rsp, info = fetch_url(module, url)
 208         if info['status'] != 200:
 209             module.fail_json(msg="Failed to download key at %s: %s" % (url, info['msg']))
 210 
 211         return rsp.read()
 212     except Exception:
 213         module.fail_json(msg="error getting key id from url: %s" % url, traceback=format_exc())

 216 def import_key(module, keyring, keyserver, key_id):
 217     if keyring:
 218         cmd = "%s --keyring %s adv --no-tty --keyserver %s --recv %s" % (apt_key_bin, keyring, keyserver, key_id)
 219     else:
 220         cmd = "%s adv --no-tty --keyserver %s --recv %s" % (apt_key_bin, keyserver, key_id)
 221     for retry in range(5):
 222         lang_env = dict(LANG='C', LC_ALL='C', LC_MESSAGES='C')
 223         (rc, out, err) = module.run_command(cmd, environ_update=lang_env)
 224         if rc == 0:
 225             break
 226     else:
 227         # Out of retries
 228         if rc == 2 and 'not found on keyserver' in out:
 229             msg = 'Key %s not found on keyserver %s' % (key_id, keyserver)
 230             module.fail_json(cmd=cmd, msg=msg)
 231         else:
 232             msg = "Error fetching key %s from keyserver: %s" % (key_id, keyserver)
 233             module.fail_json(cmd=cmd, msg=msg, rc=rc, stdout=out, stderr=err)
 234     return True

 237 def add_key(module, keyfile, keyring, data=None):
 238     if data is not None:
 239         if keyring:
 240             cmd = "%s --keyring %s add -" % (apt_key_bin, keyring)
 241         else:
 242             cmd = "%s add -" % apt_key_bin
 243         (rc, out, err) = module.run_command(cmd, data=data, check_rc=True, binary_data=True)
 244     else:
 245         if keyring:
 246             cmd = "%s --keyring %s add %s" % (apt_key_bin, keyring, keyfile)
 247         else:
 248             cmd = "%s add %s" % (apt_key_bin, keyfile)
 249         (rc, out, err) = module.run_command(cmd, check_rc=True)
 250     return True

 253 def remove_key(module, key_id, keyring):
 254     # FIXME: use module.run_command, fail at point of error and don't discard useful stdin/stdout
 255     if keyring:
 256         cmd = '%s --keyring %s del %s' % (apt_key_bin, keyring, key_id)
 257     else:
 258         cmd = '%s del %s' % (apt_key_bin, key_id)
 259     (rc, out, err) = module.run_command(cmd, check_rc=True)
 260     return True

 263 def main():
 264     module = AnsibleModule(
 265         argument_spec=dict(
 266             id=dict(type='str'),
 267             url=dict(type='str'),
 268             data=dict(type='str'),
 269             file=dict(type='path'),
 270             key=dict(type='str'),
 271             keyring=dict(type='path'),
 272             validate_certs=dict(type='bool', default=True),
 273             keyserver=dict(type='str'),
 274             state=dict(type='str', default='present', choices=['absent', 'present']),
 275         ),
 276         supports_check_mode=True,
 277         mutually_exclusive=(('data', 'file', 'keyserver', 'url'),),
 278     )
 279 
 280     key_id = module.params['id']
 281     url = module.params['url']
 282     data = module.params['data']
 283     filename = module.params['file']
 284     keyring = module.params['keyring']
 285     state = module.params['state']
 286     keyserver = module.params['keyserver']
 287     changed = False
 288 
 289     fingerprint = short_key_id = key_id
 290     short_format = False
 291     if key_id:
 292         try:
 293             short_key_id, fingerprint, key_id = parse_key_id(key_id)
 294         except ValueError:
 295             module.fail_json(msg='Invalid key_id', id=key_id)
 296 
 297         if len(fingerprint) == 8:
 298             short_format = True
 299 
 300     find_needed_binaries(module)
 301 
 302     keys = all_keys(module, keyring, short_format)
 303     return_values = {}
 304 
 305     if state == 'present':
 306         if fingerprint and fingerprint in keys:
 307             module.exit_json(changed=False)
 308         elif fingerprint and fingerprint not in keys and module.check_mode:
 309             # TODO: Someday we could go further -- write keys out to
 310             # a temporary file and then extract the key id from there via gpg
 311             # to decide if the key is installed or not.
 312             module.exit_json(changed=True)
 313         else:
 314             if not filename and not data and not keyserver:
 315                 data = download_key(module, url)
 316 
 317             if filename:
 318                 add_key(module, filename, keyring)
 319             elif keyserver:
 320                 import_key(module, keyring, keyserver, key_id)
 321             else:
 322                 add_key(module, "-", keyring, data)
 323 
 324             changed = False
 325             keys2 = all_keys(module, keyring, short_format)
 326             if len(keys) != len(keys2):
 327                 changed = True
 328 
 329             if fingerprint and fingerprint not in keys2:
 330                 module.fail_json(msg="key does not seem to have been added", id=key_id)
 331             module.exit_json(changed=changed)
 332 
 333     elif state == 'absent':
 334         if not key_id:
 335             module.fail_json(msg="key is required")
 336         if fingerprint in keys:
 337             if module.check_mode:
 338                 module.exit_json(changed=True)
 339 
 340             # we use the "short" id: key_id[-8:], short_format=True
 341             # it's a workaround for https://bugs.launchpad.net/ubuntu/+source/apt/+bug/1481871
 342             if remove_key(module, short_key_id, keyring):
 343                 keys = all_keys(module, keyring, short_format)
 344                 if fingerprint in keys:
 345                     module.fail_json(msg="apt-key del did not return an error but the key was not removed (check that the id is correct and *not* a subkey)",
 346                                      id=key_id)
 347                 changed = True
 348             else:
 349                 # FIXME: module.fail_json or exit-json immediately at point of failure
 350                 module.fail_json(msg="error removing key_id", **return_values)
 351 
 352     module.exit_json(changed=changed, **return_values)
 353 
 354 
 355 if __name__ == '__main__':
 356     main()
