 454     def __init__(self, module):
 455         # To be able to use fail_json
 456         self.module = module
 457         # Shortcut for the params
 458         self.params = self.module.params
 459         # Section is always the repoid
 460         self.section = self.params['repoid']
 461 
 462         # Check if repo directory exists
 463         repos_dir = self.params['reposdir']
 464         if not os.path.isdir(repos_dir):
 465             self.module.fail_json(
 466                 msg="Repo directory '%s' does not exist." % repos_dir)
 467 
 468         # Set dest; also used to set dest parameter for the FS attributes
 469         self.params['dest'] = os.path.join(
 470             repos_dir, "%s.repo" % self.params['file'])
 471 
 472         # Read the repo file if it exists
 473         if os.path.isfile(self.params['dest']):
 474             self.repofile.read(self.params['dest'])
 475 
 476     def add(self):
 477         # Remove already existing repo and create a new one
 478         if self.repofile.has_section(self.section):
 479             self.repofile.remove_section(self.section)
 480 
 481         # Add section
 482         self.repofile.add_section(self.section)
 483 
 484         # Baseurl/mirrorlist is not required because for removal we need only
 485         # the repo name. This is why we check if the baseurl/mirrorlist is
 486         # defined.
 487         req_params = (self.params['baseurl'], self.params['metalink'], self.params['mirrorlist'])
 488         if req_params == (None, None, None):
 489             self.module.fail_json(
 490                 msg="Parameter 'baseurl', 'metalink' or 'mirrorlist' is required for "
 491                     "adding a new repo.")
 492 
 493         # Set options
 494         for key, value in sorted(self.params.items()):
 495             if key in self.list_params and isinstance(value, list):
 496                 # Join items into one string for specific parameters
 497                 value = ' '.join(value)
 498             elif isinstance(value, bool):
 499                 # Convert boolean value to integer
 500                 value = int(value)
 501 
 502             # Set the value only if it was defined (default is None)
 503             if value is not None and key in self.allowed_params:
 504                 self.repofile.set(self.section, key, value)
 505 
 506     def save(self):
 507         if len(self.repofile.sections()):
 508             # Write data into the file
 509             try:
 510                 with open(self.params['dest'], 'w') as fd:
 511                     self.repofile.write(fd)
 512             except IOError as e:
 513                 self.module.fail_json(
 514                     msg="Problems handling file %s." % self.params['dest'],
 515                     details=to_native(e))
 516         else:
 517             # Remove the file if there are not repos
 518             try:
 519                 os.remove(self.params['dest'])
 520             except OSError as e:
 521                 self.module.fail_json(
 522                     msg=(
 523                         "Cannot remove empty repo file %s." %
 524                         self.params['dest']),
 525                     details=to_native(e))
 526 
 527     def remove(self):
 528         # Remove section if exists
 529         if self.repofile.has_section(self.section):
 530             self.repofile.remove_section(self.section)
 531 
 532     def dump(self):
 533         repo_string = ""
 534 
 535         # Compose the repo file
 536         for section in sorted(self.repofile.sections()):
 537             repo_string += "[%s]\n" % section
 538 
 539             for key, value in sorted(self.repofile.items(section)):
 540                 repo_string += "%s = %s\n" % (key, value)
 541 
 542             repo_string += "\n"
 543 
 544         return repo_string
