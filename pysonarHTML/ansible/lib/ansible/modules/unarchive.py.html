 179 def crc32(path):
 180     ''' Return a CRC32 checksum of a file '''
 181     with open(path, 'rb') as f:
 182         file_content = f.read()
 183     return binascii.crc32(file_content) & 0xffffffff

 186 def shell_escape(string):
 187     ''' Quote meta-characters in the args for the unix shell '''
 188     return re.sub(r'([^A-Za-z0-9_])', r'\\\1', string)

 210     def _permstr_to_octal(self, modestr, umask):
 211         ''' Convert a Unix permission string (rw-r--r--) into a mode (0644) '''
 212         revstr = modestr[::-1]
 213         mode = 0
 214         for j in range(0, 3):
 215             for i in range(0, 3):
 216                 if revstr[i + 3 * j] in ['r', 'w', 'x', 's', 't']:
 217                     mode += 2 ** (i + 3 * j)
 218         # The unzip utility does not support setting the stST bits
 219 #                if revstr[i + 3 * j] in ['s', 't', 'S', 'T' ]:
 220 #                    mode += 2 ** (9 + j)
 221         return (mode & ~umask)

 223     def _legacy_file_list(self):
 224         unzip_bin = self.module.get_bin_path('unzip')
 225         if not unzip_bin:
 226             raise UnarchiveError('Python Zipfile cannot read %s and unzip not found' % self.src)
 227 
 228         rc, out, err = self.module.run_command([unzip_bin, '-v', self.src])
 229         if rc:
 230             raise UnarchiveError('Neither python zipfile nor unzip can read %s' % self.src)
 231 
 232         for line in out.splitlines()[3:-2]:
 233             fields = line.split(None, 7)
 234             self._files_in_archive.append(fields[7])
 235             self._infodict[fields[7]] = int(fields[6])

 237     def _crc32(self, path):
 238         if self._infodict:
 239             return self._infodict[path]
 240 
 241         try:
 242             archive = ZipFile(self.src)
 243         except BadZipfile as e:
 244             if e.args[0].lower().startswith('bad magic number'):
 245                 # Python2.4 can't handle zipfiles with > 64K files.  Try using
 246                 # /usr/bin/unzip instead
 247                 self._legacy_file_list()
 248             else:
 249                 raise
 250         else:
 251             try:
 252                 for item in archive.infolist():
 253                     self._infodict[item.filename] = int(item.CRC)
 254             except Exception:
 255                 archive.close()
 256                 raise UnarchiveError('Unable to list files in the archive')
 257 
 258         return self._infodict[path]

 260     @property
 261     def files_in_archive(self):
 262         if self._files_in_archive:
 263             return self._files_in_archive
 264 
 265         self._files_in_archive = []
 266         try:
 267             archive = ZipFile(self.src)
 268         except BadZipfile as e:
 269             if e.args[0].lower().startswith('bad magic number'):
 270                 # Python2.4 can't handle zipfiles with > 64K files.  Try using
 271                 # /usr/bin/unzip instead
 272                 self._legacy_file_list()
 273             else:
 274                 raise
 275         else:
 276             try:
 277                 for member in archive.namelist():
 278                     exclude_flag = False
 279                     if self.excludes:
 280                         for exclude in self.excludes:
 281                             if fnmatch.fnmatch(member, exclude):
 282                                 exclude_flag = True
 283                                 break
 284                     if not exclude_flag:
 285                         self._files_in_archive.append(to_native(member))
 286             except Exception:
 287                 archive.close()
 288                 raise UnarchiveError('Unable to list files in the archive')
 289 
 290             archive.close()
 291         return self._files_in_archive

 293     def is_unarchived(self):
 294         # BSD unzip doesn't support zipinfo listings with timestamp.
 295         cmd = [self.zipinfocmd_path, '-T', '-s', self.src]
 296         if self.excludes:
 297             cmd.extend(['-x', ] + self.excludes)
 298         rc, out, err = self.module.run_command(cmd)
 299 
 300         old_out = out
 301         diff = ''
 302         out = ''
 303         if rc == 0:
 304             unarchived = True
 305         else:
 306             unarchived = False
 307 
 308         # Get some information related to user/group ownership
 309         umask = os.umask(0)
 310         os.umask(umask)
 311         systemtype = platform.system()
 312 
 313         # Get current user and group information
 314         groups = os.getgroups()
 315         run_uid = os.getuid()
 316         run_gid = os.getgid()
 317         try:
 318             run_owner = pwd.getpwuid(run_uid).pw_name
 319         except (TypeError, KeyError):
 320             run_owner = run_uid
 321         try:
 322             run_group = grp.getgrgid(run_gid).gr_name
 323         except (KeyError, ValueError, OverflowError):
 324             run_group = run_gid
 325 
 326         # Get future user ownership
 327         fut_owner = fut_uid = None
 328         if self.file_args['owner']:
 329             try:
 330                 tpw = pwd.getpwnam(self.file_args['owner'])
 331             except KeyError:
 332                 try:
 333                     tpw = pwd.getpwuid(self.file_args['owner'])
 334                 except (TypeError, KeyError):
 335                     tpw = pwd.getpwuid(run_uid)
 336             fut_owner = tpw.pw_name
 337             fut_uid = tpw.pw_uid
 338         else:
 339             try:
 340                 fut_owner = run_owner
 341             except Exception:
 342                 pass
 343             fut_uid = run_uid
 344 
 345         # Get future group ownership
 346         fut_group = fut_gid = None
 347         if self.file_args['group']:
 348             try:
 349                 tgr = grp.getgrnam(self.file_args['group'])
 350             except (ValueError, KeyError):
 351                 try:
 352                     tgr = grp.getgrgid(self.file_args['group'])
 353                 except (KeyError, ValueError, OverflowError):
 354                     tgr = grp.getgrgid(run_gid)
 355             fut_group = tgr.gr_name
 356             fut_gid = tgr.gr_gid
 357         else:
 358             try:
 359                 fut_group = run_group
 360             except Exception:
 361                 pass
 362             fut_gid = run_gid
 363 
 364         for line in old_out.splitlines():
 365             change = False
 366 
 367             pcs = line.split(None, 7)
 368             if len(pcs) != 8:
 369                 # Too few fields... probably a piece of the header or footer
 370                 continue
 371 
 372             # Check first and seventh field in order to skip header/footer
 373             if len(pcs[0]) != 7 and len(pcs[0]) != 10:
 374                 continue
 375             if len(pcs[6]) != 15:
 376                 continue
 377 
 378             # Possible entries:
 379             #   -rw-rws---  1.9 unx    2802 t- defX 11-Aug-91 13:48 perms.2660
 380             #   -rw-a--     1.0 hpf    5358 Tl i4:3  4-Dec-91 11:33 longfilename.hpfs
 381             #   -r--ahs     1.1 fat    4096 b- i4:2 14-Jul-91 12:58 EA DATA. SF
 382             #   --w-------  1.0 mac   17357 bx i8:2  4-May-92 04:02 unzip.macr
 383             if pcs[0][0] not in 'dl-?' or not frozenset(pcs[0][1:]).issubset('rwxstah-'):
 384                 continue
 385 
 386             ztype = pcs[0][0]
 387             permstr = pcs[0][1:]
 388             version = pcs[1]
 389             ostype = pcs[2]
 390             size = int(pcs[3])
 391             path = to_text(pcs[7], errors='surrogate_or_strict')
 392 
 393             # Skip excluded files
 394             if path in self.excludes:
 395                 out += 'Path %s is excluded on request\n' % path
 396                 continue
 397 
 398             # Itemized change requires L for symlink
 399             if path[-1] == '/':
 400                 if ztype != 'd':
 401                     err += 'Path %s incorrectly tagged as "%s", but is a directory.\n' % (path, ztype)
 402                 ftype = 'd'
 403             elif ztype == 'l':
 404                 ftype = 'L'
 405             elif ztype == '-':
 406                 ftype = 'f'
 407             elif ztype == '?':
 408                 ftype = 'f'
 409 
 410             # Some files may be storing FAT permissions, not Unix permissions
 411             # For FAT permissions, we will use a base permissions set of 777 if the item is a directory or has the execute bit set.  Otherwise, 666.
 412             #     This permission will then be modified by the system UMask.
 413             # BSD always applies the Umask, even to Unix permissions.
 414             # For Unix style permissions on Linux or Mac, we want to use them directly.
 415             #     So we set the UMask for this file to zero.  That permission set will then be unchanged when calling _permstr_to_octal
 416 
 417             if len(permstr) == 6:
 418                 if path[-1] == '/':
 419                     permstr = 'rwxrwxrwx'
 420                 elif permstr == 'rwx---':
 421                     permstr = 'rwxrwxrwx'
 422                 else:
 423                     permstr = 'rw-rw-rw-'
 424                 file_umask = umask
 425             elif 'bsd' in systemtype.lower():
 426                 file_umask = umask
 427             else:
 428                 file_umask = 0
 429 
 430             # Test string conformity
 431             if len(permstr) != 9 or not ZIP_FILE_MODE_RE.match(permstr):
 432                 raise UnarchiveError('ZIP info perm format incorrect, %s' % permstr)
 433 
 434             # DEBUG
 435 #            err += "%s%s %10d %s\n" % (ztype, permstr, size, path)
 436 
 437             b_dest = os.path.join(self.b_dest, to_bytes(path, errors='surrogate_or_strict'))
 438             try:
 439                 st = os.lstat(b_dest)
 440             except Exception:
 441                 change = True
 442                 self.includes.append(path)
 443                 err += 'Path %s is missing\n' % path
 444                 diff += '>%s++++++.?? %s\n' % (ftype, path)
 445                 continue
 446 
 447             # Compare file types
 448             if ftype == 'd' and not stat.S_ISDIR(st.st_mode):
 449                 change = True
 450                 self.includes.append(path)
 451                 err += 'File %s already exists, but not as a directory\n' % path
 452                 diff += 'c%s++++++.?? %s\n' % (ftype, path)
 453                 continue
 454 
 455             if ftype == 'f' and not stat.S_ISREG(st.st_mode):
 456                 change = True
 457                 unarchived = False
 458                 self.includes.append(path)
 459                 err += 'Directory %s already exists, but not as a regular file\n' % path
 460                 diff += 'c%s++++++.?? %s\n' % (ftype, path)
 461                 continue
 462 
 463             if ftype == 'L' and not stat.S_ISLNK(st.st_mode):
 464                 change = True
 465                 self.includes.append(path)
 466                 err += 'Directory %s already exists, but not as a symlink\n' % path
 467                 diff += 'c%s++++++.?? %s\n' % (ftype, path)
 468                 continue
 469 
 470             itemized = list('.%s.......??' % ftype)
 471 
 472             # Note: this timestamp calculation has a rounding error
 473             # somewhere... unzip and this timestamp can be one second off
 474             # When that happens, we report a change and re-unzip the file
 475             dt_object = datetime.datetime(*(time.strptime(pcs[6], '%Y%m%d.%H%M%S')[0:6]))
 476             timestamp = time.mktime(dt_object.timetuple())
 477 
 478             # Compare file timestamps
 479             if stat.S_ISREG(st.st_mode):
 480                 if self.module.params['keep_newer']:
 481                     if timestamp > st.st_mtime:
 482                         change = True
 483                         self.includes.append(path)
 484                         err += 'File %s is older, replacing file\n' % path
 485                         itemized[4] = 't'
 486                     elif stat.S_ISREG(st.st_mode) and timestamp < st.st_mtime:
 487                         # Add to excluded files, ignore other changes
 488                         out += 'File %s is newer, excluding file\n' % path
 489                         self.excludes.append(path)
 490                         continue
 491                 else:
 492                     if timestamp != st.st_mtime:
 493                         change = True
 494                         self.includes.append(path)
 495                         err += 'File %s differs in mtime (%f vs %f)\n' % (path, timestamp, st.st_mtime)
 496                         itemized[4] = 't'
 497 
 498             # Compare file sizes
 499             if stat.S_ISREG(st.st_mode) and size != st.st_size:
 500                 change = True
 501                 err += 'File %s differs in size (%d vs %d)\n' % (path, size, st.st_size)
 502                 itemized[3] = 's'
 503 
 504             # Compare file checksums
 505             if stat.S_ISREG(st.st_mode):
 506                 crc = crc32(b_dest)
 507                 if crc != self._crc32(path):
 508                     change = True
 509                     err += 'File %s differs in CRC32 checksum (0x%08x vs 0x%08x)\n' % (path, self._crc32(path), crc)
 510                     itemized[2] = 'c'
 511 
 512             # Compare file permissions
 513 
 514             # Do not handle permissions of symlinks
 515             if ftype != 'L':
 516 
 517                 # Use the new mode provided with the action, if there is one
 518                 if self.file_args['mode']:
 519                     if isinstance(self.file_args['mode'], int):
 520                         mode = self.file_args['mode']
 521                     else:
 522                         try:
 523                             mode = int(self.file_args['mode'], 8)
 524                         except Exception as e:
 525                             try:
 526                                 mode = AnsibleModule._symbolic_mode_to_octal(st, self.file_args['mode'])
 527                             except ValueError as e:
 528                                 self.module.fail_json(path=path, msg="%s" % to_native(e), exception=traceback.format_exc())
 529                 # Only special files require no umask-handling
 530                 elif ztype == '?':
 531                     mode = self._permstr_to_octal(permstr, 0)
 532                 else:
 533                     mode = self._permstr_to_octal(permstr, file_umask)
 534 
 535                 if mode != stat.S_IMODE(st.st_mode):
 536                     change = True
 537                     itemized[5] = 'p'
 538                     err += 'Path %s differs in permissions (%o vs %o)\n' % (path, mode, stat.S_IMODE(st.st_mode))
 539 
 540             # Compare file user ownership
 541             owner = uid = None
 542             try:
 543                 owner = pwd.getpwuid(st.st_uid).pw_name
 544             except (TypeError, KeyError):
 545                 uid = st.st_uid
 546 
 547             # If we are not root and requested owner is not our user, fail
 548             if run_uid != 0 and (fut_owner != run_owner or fut_uid != run_uid):
 549                 raise UnarchiveError('Cannot change ownership of %s to %s, as user %s' % (path, fut_owner, run_owner))
 550 
 551             if owner and owner != fut_owner:
 552                 change = True
 553                 err += 'Path %s is owned by user %s, not by user %s as expected\n' % (path, owner, fut_owner)
 554                 itemized[6] = 'o'
 555             elif uid and uid != fut_uid:
 556                 change = True
 557                 err += 'Path %s is owned by uid %s, not by uid %s as expected\n' % (path, uid, fut_uid)
 558                 itemized[6] = 'o'
 559 
 560             # Compare file group ownership
 561             group = gid = None
 562             try:
 563                 group = grp.getgrgid(st.st_gid).gr_name
 564             except (KeyError, ValueError, OverflowError):
 565                 gid = st.st_gid
 566 
 567             if run_uid != 0 and fut_gid not in groups:
 568                 raise UnarchiveError('Cannot change group ownership of %s to %s, as user %s' % (path, fut_group, run_owner))
 569 
 570             if group and group != fut_group:
 571                 change = True
 572                 err += 'Path %s is owned by group %s, not by group %s as expected\n' % (path, group, fut_group)
 573                 itemized[6] = 'g'
 574             elif gid and gid != fut_gid:
 575                 change = True
 576                 err += 'Path %s is owned by gid %s, not by gid %s as expected\n' % (path, gid, fut_gid)
 577                 itemized[6] = 'g'
 578 
 579             # Register changed files and finalize diff output
 580             if change:
 581                 if path not in self.includes:
 582                     self.includes.append(path)
 583                 diff += '%s %s\n' % (''.join(itemized), path)
 584 
 585         if self.includes:
 586             unarchived = False
 587 
 588         # DEBUG
 589 #        out = old_out + out
 590

 642     def _get_tar_type(self):
 643         cmd = [self.cmd_path, '--version']
 644         (rc, out, err) = self.module.run_command(cmd)
 645         tar_type = None
 646         if out.startswith('bsdtar'):
 647             tar_type = 'bsd'
 648         elif out.startswith('tar') and 'GNU' in out:
 649             tar_type = 'gnu'
 650         return tar_type

 652     @property
 653     def files_in_archive(self):
 654         if self._files_in_archive:
 655             return self._files_in_archive
 656 
 657         cmd = [self.cmd_path, '--list', '-C', self.b_dest]
 658         if self.zipflag:
 659             cmd.append(self.zipflag)
 660         if self.opts:
 661             cmd.extend(['--show-transformed-names'] + self.opts)
 662         if self.excludes:
 663             cmd.extend(['--exclude=' + f for f in self.excludes])
 664         cmd.extend(['-f', self.src])
 665         rc, out, err = self.module.run_command(cmd, cwd=self.b_dest, environ_update=dict(LANG='C', LC_ALL='C', LC_MESSAGES='C'))
 666 
 667         if rc != 0:
 668             raise UnarchiveError('Unable to list files in the archive')
 669 
 670         for filename in out.splitlines():
 671             # Compensate for locale-related problems in gtar output (octal unicode representation) #11348
 672             # filename = filename.decode('string_escape')
 673             filename = to_native(codecs.escape_decode(filename)[0])
 674 
 675             # We don't allow absolute filenames.  If the user wants to unarchive rooted in "/"
 676             # they need to use "dest: '/'".  This follows the defaults for gtar, pax, etc.
 677             # Allowing absolute filenames here also causes bugs: https://github.com/ansible/ansible/issues/21397
 678             if filename.startswith('/'):
 679                 filename = filename[1:]
 680 
 681             exclude_flag = False
 682             if self.excludes:
 683                 for exclude in self.excludes:
 684                     if fnmatch.fnmatch(filename, exclude):
 685                         exclude_flag = True
 686                         break
 687 
 688             if not exclude_flag:
 689                 self._files_in_archive.append(to_native(filename))
 690 
 691         return self._files_in_archive

 693     def is_unarchived(self):
 694         cmd = [self.cmd_path, '--diff', '-C', self.b_dest]
 695         if self.zipflag:
 696             cmd.append(self.zipflag)
 697         if self.opts:
 698             cmd.extend(['--show-transformed-names'] + self.opts)
 699         if self.file_args['owner']:
 700             cmd.append('--owner=' + quote(self.file_args['owner']))
 701         if self.file_args['group']:
 702             cmd.append('--group=' + quote(self.file_args['group']))
 703         if self.module.params['keep_newer']:
 704             cmd.append('--keep-newer-files')
 705         if self.excludes:
 706             cmd.extend(['--exclude=' + f for f in self.excludes])
 707         cmd.extend(['-f', self.src])
 708         rc, out, err = self.module.run_command(cmd, cwd=self.b_dest, environ_update=dict(LANG='C', LC_ALL='C', LC_MESSAGES='C'))
 709 
 710         # Check whether the differences are in something that we're
 711         # setting anyway
 712 
 713         # What is different
 714         unarchived = True
 715         old_out = out
 716         out = ''
 717         run_uid = os.getuid()
 718         # When unarchiving as a user, or when owner/group/mode is supplied --diff is insufficient
 719         # Only way to be sure is to check request with what is on disk (as we do for zip)
 720         # Leave this up to set_fs_attributes_if_different() instead of inducing a (false) change
 721         for line in old_out.splitlines() + err.splitlines():
 722             # FIXME: Remove the bogus lines from error-output as well !
 723             # Ignore bogus errors on empty filenames (when using --split-component)
 724             if EMPTY_FILE_RE.search(line):
 725                 continue
 726             if run_uid == 0 and not self.file_args['owner'] and OWNER_DIFF_RE.search(line):
 727                 out += line + '\n'
 728             if run_uid == 0 and not self.file_args['group'] and GROUP_DIFF_RE.search(line):
 729                 out += line + '\n'
 730             if not self.file_args['mode'] and MODE_DIFF_RE.search(line):
 731                 out += line + '\n'
 732             if MOD_TIME_DIFF_RE.search(line):
 733                 out += line + '\n'
 734             if MISSING_FILE_RE.search(line):
 735                 out += line + '\n'
 736             if INVALID_OWNER_RE.search(line):
 737                 out += line + '\n'
 738             if INVALID_GROUP_RE.search(line):
 739                 out += line + '\n'
 740         if out:
 741             unarchived = False
 742         return dict(unarchived=unarchived, rc=rc, out=out, err=err, cmd=cmd)

 763         if not self.cmd_path:
 764             return False, 'Commands "gtar" and "tar" not found.'
 765 
 766         if self.tar_type != 'gnu':
 767             return False, 'Command "%s" detected as tar type %s. GNU tar required.' % (self.cmd_path, self.tar_type)

 769         try:
 770             if self.files_in_archive:
 771                 return True, None
 772         except UnarchiveError:

 801 # try handlers in order and return the one that works or bail if none work
 802 def pick_handler(src, dest, file_args, module):
 803     handlers = [ZipArchive, TgzArchive, TarArchive, TarBzipArchive, TarXzArchive]
 804     reasons = set()
 805     for handler in handlers:
 806         obj = handler(src, dest, file_args, module)
 807         (can_handle, reason) = obj.can_handle_archive()
 808         if can_handle:
 809             return obj
 810         reasons.add(reason)
 811     reason_msg = ' '.join(reasons)
 812     module.fail_json(msg='Failed to find handler for "%s". Make sure the required command to extract the file is installed. %s' % (src, reason_msg))

 815 def main():
 816     module = AnsibleModule(
 817         # not checking because of daisy chain to file module
 818         argument_spec=dict(
 819             src=dict(type='path', required=True),
 820             dest=dict(type='path', required=True),
 821             remote_src=dict(type='bool', default=False),
 822             creates=dict(type='path'),
 823             list_files=dict(type='bool', default=False),
 824             keep_newer=dict(type='bool', default=False),
 825             exclude=dict(type='list', default=[]),
 826             extra_opts=dict(type='list', default=[]),
 827             validate_certs=dict(type='bool', default=True),
 828         ),
 829         add_file_common_args=True,
 830         # check-mode only works for zip files, we cover that later
 831         supports_check_mode=True,
 832     )
 833 
 834     src = module.params['src']
 835     dest = module.params['dest']
 836     b_dest = to_bytes(dest, errors='surrogate_or_strict')
 837     remote_src = module.params['remote_src']
 838     file_args = module.load_file_common_arguments(module.params)
 839 
 840     # did tar file arrive?
 841     if not os.path.exists(src):
 842         if not remote_src:
 843             module.fail_json(msg="Source '%s' failed to transfer" % src)
 844         # If remote_src=true, and src= contains ://, try and download the file to a temp directory.
 845         elif '://' in src:
 846             src = fetch_file(module, src)
 847         else:
 848             module.fail_json(msg="Source '%s' does not exist" % src)
 849     if not os.access(src, os.R_OK):
 850         module.fail_json(msg="Source '%s' not readable" % src)
 851 
 852     # skip working with 0 size archives
 853     try:
 854         if os.path.getsize(src) == 0:
 855             module.fail_json(msg="Invalid archive '%s', the file is 0 bytes" % src)
 856     except Exception as e:
 857         module.fail_json(msg="Source '%s' not readable, %s" % (src, to_native(e)))
 858 
 859     # is dest OK to receive tar file?
 860     if not os.path.isdir(b_dest):
 861         module.fail_json(msg="Destination '%s' is not a directory" % dest)
 862 
 863     handler = pick_handler(src, b_dest, file_args, module)
 864 
 865     res_args = dict(handler=handler.__class__.__name__, dest=dest, src=src)
 866 
 867     # do we need to do unpack?
 868     check_results = handler.is_unarchived()
 869 
 870     # DEBUG
 871     # res_args['check_results'] = check_results
 872 
 873     if module.check_mode:
 874         res_args['changed'] = not check_results['unarchived']
 875     elif check_results['unarchived']:
 876         res_args['changed'] = False
 877     else:
 878         # do the unpack
 879         try:
 880             res_args['extract_results'] = handler.unarchive()
 881             if res_args['extract_results']['rc'] != 0:
 882                 module.fail_json(msg="failed to unpack %s to %s" % (src, dest), **res_args)
 883         except IOError:
 884             module.fail_json(msg="failed to unpack %s to %s" % (src, dest), **res_args)
 885         else:
 886             res_args['changed'] = True
 887 
 888     # Get diff if required
 889     if check_results.get('diff', False):
 890         res_args['diff'] = {'prepared': check_results['diff']}
 891 
 892     # Run only if we found differences (idempotence) or diff was missing
 893     if res_args.get('diff', True) and not module.check_mode:
 894         # do we need to change perms?
 895         for filename in handler.files_in_archive:
 896             file_args['path'] = os.path.join(b_dest, to_bytes(filename, errors='surrogate_or_strict'))
 897 
 898             try:
 899                 res_args['changed'] = module.set_fs_attributes_if_different(file_args, res_args['changed'], expand=False)
 900             except (IOError, OSError) as e:
 901                 module.fail_json(msg="Unexpected error when accessing exploded file: %s" % to_native(e), **res_args)
 902 
 903     if module.params['list_files']:
 904         res_args['files'] = handler.files_in_archive
 905 
 906     module.exit_json(**res_args)
 907

 909 if __name__ == '__main__':
 910     main()
