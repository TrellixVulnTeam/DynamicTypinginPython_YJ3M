  11 DOCUMENTATION = r'''
  12 ---
  13 module: group_by
  14 short_description: Create Ansible groups based on facts
  15 description:
  16 - Use facts to create ad-hoc groups that can be used later in a playbook.
  17 - This module is also supported for Windows targets.
  18 version_added: "0.9"
  19 options:
  20   key:
  21     description:
  22     - The variables whose values will be used as groups.
  23     type: str
  24     required: true
  25   parents:
  26     description:
  27     - The list of the parent groups.
  28     type: list
  29     default: all
  30     version_added: "2.4"
  31 notes:
  32 - Spaces in group names are converted to dashes '-'.
  33 - This module is also supported for Windows targets.

  35 - module: add_host
  36 author:
  37 - Jeroen Hoekx (@jhoekx)
  38 '''

  40 EXAMPLES = r'''
  41 - name: Create groups based on the machine architecture
  42   group_by:
  43     key: machine_{{ ansible_machine }}
  44 
  45 - name: Create groups like 'virt_kvm_host'
  46   group_by:
  47     key: virt_{{ ansible_virtualization_type }}_{{ ansible_virtualization_role }}
  48 
  49 - name: Create nested groups
  50   group_by:
  51     key: el{{ ansible_distribution_major_version }}-{{ ansible_architecture }}
  52     parents:
  53       - el{{ ansible_distribution_major_version }}
  54 
  55 # Add all active hosts to a static group
  56 - group_by:
  57     key: done
  58 '''
