 276 def parse_systemctl_show(lines):
 277     # The output of 'systemctl show' can contain values that span multiple lines. At first glance it
 278     # appears that such values are always surrounded by {}, so the previous version of this code
 279     # assumed that any value starting with { was a multi-line value; it would then consume lines
 280     # until it saw a line that ended with }. However, it is possible to have a single-line value
 281     # that starts with { but does not end with } (this could happen in the value for Description=,
 282     # for example), and the previous version of this code would then consume all remaining lines as
 283     # part of that value. Cryptically, this would lead to Ansible reporting that the service file
 284     # couldn't be found.
 285     #
 286     # To avoid this issue, the following code only accepts multi-line values for keys whose names
 287     # start with Exec (e.g., ExecStart=), since these are the only keys whose values are known to
 288     # span multiple lines.
 289     parsed = {}
 290     multival = []
 291     k = None
 292     for line in lines:
 293         if k is None:
 294             if '=' in line:
 295                 k, v = line.split('=', 1)
 296                 if k.startswith('Exec') and v.lstrip().startswith('{'):
 297                     if not v.rstrip().endswith('}'):
 298                         multival.append(v)
 299                         continue
 300                 parsed[k] = v.strip()
 301                 k = None
 302         else:
 303             multival.append(line)
 304             if line.rstrip().endswith('}'):
 305                 parsed[k] = '\n'.join(multival).strip()
 306                 multival = []
 307                 k = None
 308     return parsed

 314 def main():
 315     # initialize
 316     module = AnsibleModule(
 317         argument_spec=dict(
 318             name=dict(type='str', aliases=['service', 'unit']),
 319             state=dict(type='str', choices=['reloaded', 'restarted', 'started', 'stopped']),
 320             enabled=dict(type='bool'),
 321             force=dict(type='bool'),
 322             masked=dict(type='bool'),
 323             daemon_reload=dict(type='bool', default=False, aliases=['daemon-reload']),
 324             daemon_reexec=dict(type='bool', default=False, aliases=['daemon-reexec']),
 325             scope=dict(type='str', default='system', choices=['system', 'user', 'global']),
 326             no_block=dict(type='bool', default=False),
 327         ),
 328         supports_check_mode=True,
 329         required_one_of=[['state', 'enabled', 'masked', 'daemon_reload', 'daemon_reexec']],
 330         required_by=dict(
 331             state=('name', ),
 332             enabled=('name', ),
 333             masked=('name', ),
 334         ),
 335     )
 336 
 337     unit = module.params['name']
 338     if unit is not None:
 339         for globpattern in (r"*", r"?", r"["):
 340             if globpattern in unit:
 341                 module.fail_json(msg="This module does not currently support using glob patterns, found '%s' in service name: %s" % (globpattern, unit))
 342 
 343     systemctl = module.get_bin_path('systemctl', True)
 344 
 345     if os.getenv('XDG_RUNTIME_DIR') is None:
 346         os.environ['XDG_RUNTIME_DIR'] = '/run/user/%s' % os.geteuid()
 347 
 348     ''' Set CLI options depending on params '''
 349     # if scope is 'system' or None, we can ignore as there is no extra switch.
 350     # The other choices match the corresponding switch
 351     if module.params['scope'] != 'system':
 352         systemctl += " --%s" % module.params['scope']
 353 
 354     if module.params['no_block']:
 355         systemctl += " --no-block"
 356 
 357     if module.params['force']:
 358         systemctl += " --force"
 359 
 360     rc = 0
 361     out = err = ''
 362     result = dict(
 363         name=unit,
 364         changed=False,
 365         status=dict(),
 366     )
 367 
 368     # Run daemon-reload first, if requested
 369     if module.params['daemon_reload'] and not module.check_mode:
 370         (rc, out, err) = module.run_command("%s daemon-reload" % (systemctl))
 371         if rc != 0:
 372             module.fail_json(msg='failure %d during daemon-reload: %s' % (rc, err))
 373 
 374     # Run daemon-reexec
 375     if module.params['daemon_reexec'] and not module.check_mode:
 376         (rc, out, err) = module.run_command("%s daemon-reexec" % (systemctl))
 377         if rc != 0:
 378             module.fail_json(msg='failure %d during daemon-reexec: %s' % (rc, err))
 379 
 380     if unit:
 381         found = False
 382         is_initd = sysv_exists(unit)
 383         is_systemd = False
 384 
 385         # check service data, cannot error out on rc as it changes across versions, assume not found
 386         (rc, out, err) = module.run_command("%s show '%s'" % (systemctl, unit))
 387 
 388         if rc == 0 and not (request_was_ignored(out) or request_was_ignored(err)):
 389             # load return of systemctl show into dictionary for easy access and return
 390             if out:
 391                 result['status'] = parse_systemctl_show(to_native(out).split('\n'))
 392 
 393                 is_systemd = 'LoadState' in result['status'] and result['status']['LoadState'] != 'not-found'
 394 
 395                 is_masked = 'LoadState' in result['status'] and result['status']['LoadState'] == 'masked'
 396 
 397                 # Check for loading error
 398                 if is_systemd and not is_masked and 'LoadError' in result['status']:
 399                     module.fail_json(msg="Error loading unit file '%s': %s" % (unit, result['status']['LoadError']))
 400         else:
 401             # list taken from man systemctl(1) for systemd 244
 402             valid_enabled_states = [
 403                 "enabled",
 404                 "enabled-runtime",
 405                 "linked",
 406                 "linked-runtime",
 407                 "masked",
 408                 "masked-runtime",
 409                 "static",
 410                 "indirect",
 411                 "disabled",
 412                 "generated",
 413                 "transient"]
 414 
 415             (rc, out, err) = module.run_command("%s is-enabled '%s'" % (systemctl, unit))
 416             if out.strip() in valid_enabled_states:
 417                 is_systemd = True
 418             else:
 419                 # fallback list-unit-files as show does not work on some systems (chroot)
 420                 # not used as primary as it skips some services (like those using init.d) and requires .service/etc notation
 421                 (rc, out, err) = module.run_command("%s list-unit-files '%s'" % (systemctl, unit))
 422                 if rc == 0:
 423                     is_systemd = True
 424                 else:
 425                     # Check for systemctl command
 426                     module.run_command(systemctl, check_rc=True)
 427 
 428         # Does service exist?
 429         found = is_systemd or is_initd
 430         if is_initd and not is_systemd:
 431             module.warn('The service (%s) is actually an init script but the system is managed by systemd' % unit)
 432 
 433         # mask/unmask the service, if requested, can operate on services before they are installed
 434         if module.params['masked'] is not None:
 435             # state is not masked unless systemd affirms otherwise
 436             (rc, out, err) = module.run_command("%s is-enabled '%s'" % (systemctl, unit))
 437             masked = out.strip() == "masked"
 438 
 439             if masked != module.params['masked']:
 440                 result['changed'] = True
 441                 if module.params['masked']:
 442                     action = 'mask'
 443                 else:
 444                     action = 'unmask'
 445 
 446                 if not module.check_mode:
 447                     (rc, out, err) = module.run_command("%s %s '%s'" % (systemctl, action, unit))
 448                     if rc != 0:
 449                         # some versions of system CAN mask/unmask non existing services, we only fail on missing if they don't
 450                         fail_if_missing(module, found, unit, msg='host')
 451 
 452         # Enable/disable service startup at boot if requested
 453         if module.params['enabled'] is not None:
 454 
 455             if module.params['enabled']:
 456                 action = 'enable'
 457             else:
 458                 action = 'disable'
 459 
 460             fail_if_missing(module, found, unit, msg='host')
 461 
 462             # do we need to enable the service?
 463             enabled = False
 464             (rc, out, err) = module.run_command("%s is-enabled '%s'" % (systemctl, unit))
 465 
 466             # check systemctl result or if it is a init script
 467             if rc == 0:
 468                 enabled = True
 469             elif rc == 1:
 470                 # if not a user or global user service and both init script and unit file exist stdout should have enabled/disabled, otherwise use rc entries
 471                 if module.params['scope'] == 'system' and \
 472                         is_initd and \
 473                         not out.strip().endswith('disabled') and \
 474                         sysv_is_enabled(unit):
 475                     enabled = True
 476 
 477             # default to current state
 478             result['enabled'] = enabled
 479 
 480             # Change enable/disable if needed
 481             if enabled != module.params['enabled']:
 482                 result['changed'] = True
 483                 if not module.check_mode:
 484                     (rc, out, err) = module.run_command("%s %s '%s'" % (systemctl, action, unit))
 485                     if rc != 0:
 486                         module.fail_json(msg="Unable to %s service %s: %s" % (action, unit, out + err))
 487 
 488                 result['enabled'] = not enabled
 489 
 490         # set service state if requested
 491         if module.params['state'] is not None:
 492             fail_if_missing(module, found, unit, msg="host")
 493 
 494             # default to desired state
 495             result['state'] = module.params['state']
 496 
 497             # What is current service state?
 498             if 'ActiveState' in result['status']:
 499                 action = None
 500                 if module.params['state'] == 'started':
 501                     if not is_running_service(result['status']):
 502                         action = 'start'
 503                 elif module.params['state'] == 'stopped':
 504                     if is_running_service(result['status']) or is_deactivating_service(result['status']):
 505                         action = 'stop'
 506                 else:
 507                     if not is_running_service(result['status']):
 508                         action = 'start'
 509                     else:
 510                         action = module.params['state'][:-2]  # remove 'ed' from restarted/reloaded
 511                     result['state'] = 'started'
 512 
 513                 if action:
 514                     result['changed'] = True
 515                     if not module.check_mode:
 516                         (rc, out, err) = module.run_command("%s %s '%s'" % (systemctl, action, unit))
 517                         if rc != 0:
 518                             module.fail_json(msg="Unable to %s service %s: %s" % (action, unit, err))
 519             # check for chroot
 520             elif is_chroot(module):
 521                 module.warn("Target is a chroot. This can lead to false positives or prevent the init system tools from working.")
 522             else:
 523                 # this should not happen?
 524                 module.fail_json(msg="Service is in unknown state", status=result['status'])
 525 
 526     module.exit_json(**result)
 527 
 528 
 529 if __name__ == '__main__':
 530     main()
