  11 DOCUMENTATION = r'''
  12 ---
  13 module: debug
  14 short_description: Print statements during execution
  15 description:
  16 - This module prints statements during execution and can be useful
  17   for debugging variables or expressions without necessarily halting
  18   the playbook.
  19 - Useful for debugging together with the 'when:' directive.
  20 - This module is also supported for Windows targets.
  21 version_added: '0.8'
  22 options:
  23   msg:
  24     description:
  25     - The customized message that is printed. If omitted, prints a generic message.
  26     type: str
  27     default: 'Hello world!'
  28   var:
  29     description:
  30     - A variable name to debug.
  31     - Mutually exclusive with the C(msg) option.
  32     - Be aware that this option already runs in Jinja2 context and has an implicit C({{ }}) wrapping,
  33       so you should not be using Jinja2 delimiters unless you are looking for double interpolation.
  34     type: str
  35   verbosity:
  36     description:
  37     - A number that controls when the debug is run, if you set to 3 it will only run debug when -vvv or above
  38     type: int
  39     default: 0
  40     version_added: '2.1'
  41 notes:
  42     - This module is also supported for Windows targets.
  43 seealso:
  44 - module: assert
  45 - module: fail
  46 author:
  47 - Dag Wieers (@dagwieers)
  48 - Michael DeHaan
  49 '''
