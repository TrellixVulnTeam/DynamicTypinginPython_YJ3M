  14 version_added: '2.0'
  15 short_description: Executes a command and responds to prompts.
  16 description:
  17      - The C(expect) module executes a command and responds to prompts.
  18      - The given command will be executed on all selected nodes. It will not be

  20        like C("<"), C(">"), C("|"), and C("&") will not work.

  22   command:

  26   creates:
  27     type: path

  30   removes:
  31     type: path
  32     description:
  33       - A filename, when it does not exist, this step will B(not) be run.
  34   chdir:
  35     type: path
  36     description:
  37       - Change into this directory before running the command.
  38   responses:
  39     type: dict
  40     description:
  41       - Mapping of expected string/regex and string to respond with. If the
  42         response is a list, successive matches return successive
  43         responses. List functionality is new in 2.1.
  44     required: true
  45   timeout:
  46     type: int
  47     description:
  48       - Amount of time in seconds to wait for the expected strings. Use
  49         C(null) to disable timeout.
  50     default: 30
  51   echo:
  52     description:
  53       - Whether or not to echo out your response strings.
  54     default: false
  55     type: bool
  56 requirements:
  57   - python >= 2.6
  58   - pexpect >= 3.3
  59 notes:
  60   - If you want to run a command through the shell (say you are using C(<),

  97 import datetime
  98 import os
  99 import traceback
 100 
 101 PEXPECT_IMP_ERR = None
 102 try:
 103     import pexpect
 104     HAS_PEXPECT = True
 105 except ImportError:
 106     PEXPECT_IMP_ERR = traceback.format_exc()
 107     HAS_PEXPECT = False
 108 
 109 from ansible.module_utils.basic import AnsibleModule, missing_required_lib
 110 from ansible.module_utils._text import to_native, to_text
 111 
 112 
 113 def response_closure(module, question, responses):
 114     resp_gen = (u'%s\n' % to_text(r).rstrip(u'\n') for r in responses)
 115 
 116     def wrapped(info):
 117         try:
 118             return next(resp_gen)
 119         except StopIteration:
 120             module.fail_json(msg="No remaining responses for '%s', "
 121                                  "output was '%s'" %
 122                                  (question,
 123                                   info['child_result_list'][-1]))
 124 
 125     return wrapped
 126 
 127 
 128 def main():
 129     module = AnsibleModule(
 130         argument_spec=dict(
 131             command=dict(required=True),
 132             chdir=dict(type='path'),
 133             creates=dict(type='path'),
 134             removes=dict(type='path'),
 135             responses=dict(type='dict', required=True),
 136             timeout=dict(type='int', default=30),
 137             echo=dict(type='bool', default=False),
 138         )
 139     )
 140 
 141     if not HAS_PEXPECT:
 142         module.fail_json(msg=missing_required_lib("pexpect"),
 143                          exception=PEXPECT_IMP_ERR)
 144 
 145     chdir = module.params['chdir']
 146     args = module.params['command']
 147     creates = module.params['creates']
 148     removes = module.params['removes']
 149     responses = module.params['responses']
 150     timeout = module.params['timeout']
 151     echo = module.params['echo']
 152 
 153     events = dict()
 154     for key, value in responses.items():
 155         if isinstance(value, list):
 156             response = response_closure(module, key, value)
 157         else:
 158             response = u'%s\n' % to_text(value).rstrip(u'\n')
 159 
 160         events[to_text(key)] = response
 161 
 162     if args.strip() == '':
 163         module.fail_json(rc=256, msg="no command given")
 164 
 165     if chdir:
 166         chdir = os.path.abspath(chdir)
 167         os.chdir(chdir)
 168 
 169     if creates:
 170         # do not run the command if the line contains creates=filename
 171         # and the filename already exists.  This allows idempotence
 172         # of command executions.
 173         if os.path.exists(creates):
 174             module.exit_json(
 175                 cmd=args,
 176                 stdout="skipped, since %s exists" % creates,
 177                 changed=False,
 178                 rc=0
 179             )
 180 
 181     if removes:
 182         # do not run the command if the line contains removes=filename
 183         # and the filename does not exist.  This allows idempotence
 184         # of command executions.
 185         if not os.path.exists(removes):
 186             module.exit_json(
 187                 cmd=args,
 188                 stdout="skipped, since %s does not exist" % removes,
 189                 changed=False,
 190                 rc=0
 191             )
 192 
 193     startd = datetime.datetime.now()
 194 
 195     try:
 196         try:
 197             # Prefer pexpect.run from pexpect>=4
 198             out, rc = pexpect.run(args, timeout=timeout, withexitstatus=True,
 199                                   events=events, cwd=chdir, echo=echo,
 200                                   encoding='utf-8')
 201         except TypeError:
 202             # Use pexpect.runu in pexpect>=3.3,<4
 203             out, rc = pexpect.runu(args, timeout=timeout, withexitstatus=True,
 204                                    events=events, cwd=chdir, echo=echo)
 205     except (TypeError, AttributeError) as e:
 206         # This should catch all insufficient versions of pexpect
 207         # We deem them insufficient for their lack of ability to specify
 208         # to not echo responses via the run/runu functions, which would
 209         # potentially leak sensitive information
 210         module.fail_json(msg='Insufficient version of pexpect installed '
 211                              '(%s), this module requires pexpect>=3.3. '
 212                              'Error was %s' % (pexpect.__version__, to_native(e)))
 213     except pexpect.ExceptionPexpect as e:
 214         module.fail_json(msg='%s' % to_native(e), exception=traceback.format_exc())
 215 
 216     endd = datetime.datetime.now()
 217     delta = endd - startd
 218 
 219     if out is None:
 220         out = ''
 221 
 222     result = dict(
 223         cmd=args,
 224         stdout=out.rstrip('\r\n'),
 225         rc=rc,
 226         start=str(startd),
 227         end=str(endd),
 228         delta=str(delta),
 229         changed=True,
 230     )
 231 
 232     if rc is None:
 233         module.fail_json(msg='command exceeded timeout', **result)
 234     elif rc != 0:
 235         module.fail_json(msg='non-zero return code', **result)
 236 
 237     module.exit_json(**result)
 238 
 239 
 240 if __name__ == '__main__':
 241     main()
