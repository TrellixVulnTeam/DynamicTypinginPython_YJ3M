  93 class ServiceScanService(BaseService):
  94 
  95     def gather_services(self):
  96         services = {}
  97         service_path = self.module.get_bin_path("service")
  98         if service_path is None:
  99             return None
 100         initctl_path = self.module.get_bin_path("initctl")
 101         chkconfig_path = self.module.get_bin_path("chkconfig")
 102 
 103         # sysvinit
 104         if service_path is not None and chkconfig_path is None:
 105             rc, stdout, stderr = self.module.run_command("%s --status-all 2>&1 | grep -E \"\\[ (\\+|\\-) \\]\"" % service_path, use_unsafe_shell=True)
 106             for line in stdout.split("\n"):
 107                 line_data = line.split()
 108                 if len(line_data) < 4:
 109                     continue  # Skipping because we expected more data
 110                 service_name = " ".join(line_data[3:])
 111                 if line_data[1] == "+":
 112                     service_state = "running"
 113                 else:
 114                     service_state = "stopped"
 115                 services[service_name] = {"name": service_name, "state": service_state, "source": "sysv"}
 116 
 117         # Upstart
 118         if initctl_path is not None and chkconfig_path is None:
 119             p = re.compile(r'^\s?(?P<name>.*)\s(?P<goal>\w+)\/(?P<state>\w+)(\,\sprocess\s(?P<pid>[0-9]+))?\s*$')
 120             rc, stdout, stderr = self.module.run_command("%s list" % initctl_path)
 121             real_stdout = stdout.replace("\r", "")
 122             for line in real_stdout.split("\n"):
 123                 m = p.match(line)
 124                 if not m:
 125                     continue
 126                 service_name = m.group('name')
 127                 service_goal = m.group('goal')
 128                 service_state = m.group('state')
 129                 if m.group('pid'):
 130                     pid = m.group('pid')
 131                 else:
 132                     pid = None  # NOQA
 133                 payload = {"name": service_name, "state": service_state, "goal": service_goal, "source": "upstart"}
 134                 services[service_name] = payload
 135 
 136         # RH sysvinit
 137         elif chkconfig_path is not None:
 138             # print '%s --status-all | grep -E "is (running|stopped)"' % service_path
 139             p = re.compile(
 140                 r'(?P<service>.*?)\s+[0-9]:(?P<rl0>on|off)\s+[0-9]:(?P<rl1>on|off)\s+[0-9]:(?P<rl2>on|off)\s+'
 141                 r'[0-9]:(?P<rl3>on|off)\s+[0-9]:(?P<rl4>on|off)\s+[0-9]:(?P<rl5>on|off)\s+[0-9]:(?P<rl6>on|off)')
 142             rc, stdout, stderr = self.module.run_command('%s' % chkconfig_path, use_unsafe_shell=True)
 143             # Check for special cases where stdout does not fit pattern
 144             match_any = False
 145             for line in stdout.split('\n'):
 146                 if p.match(line):
 147                     match_any = True
 148             if not match_any:
 149                 p_simple = re.compile(r'(?P<service>.*?)\s+(?P<rl0>on|off)')
 150                 match_any = False
 151                 for line in stdout.split('\n'):
 152                     if p_simple.match(line):
 153                         match_any = True
 154                 if match_any:
 155                     # Try extra flags " -l --allservices" needed for SLES11
 156                     rc, stdout, stderr = self.module.run_command('%s -l --allservices' % chkconfig_path, use_unsafe_shell=True)
 157                 elif '--list' in stderr:
 158                     # Extra flag needed for RHEL5
 159                     rc, stdout, stderr = self.module.run_command('%s --list' % chkconfig_path, use_unsafe_shell=True)
 160             for line in stdout.split('\n'):
 161                 m = p.match(line)
 162                 if m:
 163                     service_name = m.group('service')
 164                     service_state = 'stopped'
 165                     service_status = "disabled"
 166                     if m.group('rl3') == 'on':
 167                         service_status = "enabled"
 168                     rc, stdout, stderr = self.module.run_command('%s %s status' % (service_path, service_name), use_unsafe_shell=True)
 169                     service_state = rc
 170                     if rc in (0,):
 171                         service_state = 'running'
 172                     # elif rc in (1,3):
 173                     else:
 174                         if 'root' in stderr or 'permission' in stderr.lower() or 'not in sudoers' in stderr.lower():
 175                             self.incomplete_warning = True
 176                             continue
 177                         else:
 178                             service_state = 'stopped'
 179                     service_data = {"name": service_name, "state": service_state, "status": service_status, "source": "sysv"}
 180                     services[service_name] = service_data
 181         return services

 184 class SystemctlScanService(BaseService):
 185 
 186     def systemd_enabled(self):
 187         # Check if init is the systemd command, using comm as cmdline could be symlink
 188         try:
 189             f = open('/proc/1/comm', 'r')
 190         except IOError:
 191             # If comm doesn't exist, old kernel, no systemd
 192             return False
 193         for line in f:
 194             if 'systemd' in line:
 195                 return True
 196         return False
 197 
 198     def gather_services(self):
 199         services = {}
 200         if not self.systemd_enabled():
 201             return None
 202         systemctl_path = self.module.get_bin_path("systemctl", opt_dirs=["/usr/bin", "/usr/local/bin"])
 203         if systemctl_path is None:
 204             return None
 205         rc, stdout, stderr = self.module.run_command("%s list-units --no-pager --type service --all" % systemctl_path, use_unsafe_shell=True)
 206         for line in [svc_line for svc_line in stdout.split('\n') if '.service' in svc_line and 'not-found' not in svc_line]:
 207             service_name = line.split()[0]
 208             if "running" in line:
 209                 state_val = "running"
 210             else:
 211                 if 'failed' in line:
 212                     service_name = line.split()[1]
 213                 state_val = "stopped"
 214             services[service_name] = {"name": service_name, "state": state_val, "status": "unknown", "source": "systemd"}
 215         rc, stdout, stderr = self.module.run_command("%s list-unit-files --no-pager --type service --all" % systemctl_path, use_unsafe_shell=True)
 216         for line in [svc_line for svc_line in stdout.split('\n') if '.service' in svc_line and 'not-found' not in svc_line]:
 217             # there is one more column (VENDOR PRESET) from `systemctl list-unit-files` for systemd >= 245
 218             try:
 219                 service_name, status_val = line.split()[:2]
 220             except IndexError:
 221                 self.module.fail_json(msg="Malformed output discovered from systemd list-unit-files: {0}".format(line))
 222             if service_name not in services:
 223                 rc, stdout, stderr = self.module.run_command("%s show %s --property=ActiveState" % (systemctl_path, service_name), use_unsafe_shell=True)
 224                 state = 'unknown'
 225                 if not rc and stdout != '':
 226                     state = stdout.replace('ActiveState=', '').rstrip()
 227                 services[service_name] = {"name": service_name, "state": state, "status": status_val, "source": "systemd"}
 228             else:
 229                 services[service_name]["status"] = status_val
 230         return services

 233 def main():
 234     module = AnsibleModule(argument_spec=dict(), supports_check_mode=True)
 235     module.run_command_environ_update = dict(LANG="C", LC_ALL="C")
 236     service_modules = (ServiceScanService, SystemctlScanService)
 237     all_services = {}
 238     incomplete_warning = False
 239     for svc_module in service_modules:
 240         svcmod = svc_module(module)
 241         svc = svcmod.gather_services()
 242         if svc is not None:
 243             all_services.update(svc)
 244             if svcmod.incomplete_warning:
 245                 incomplete_warning = True
 246     if len(all_services) == 0:
 247         results = dict(skipped=True, msg="Failed to find any services. Sometimes this is due to insufficient privileges.")
 248     else:
 249         results = dict(ansible_facts=dict(services=all_services))
 250         if incomplete_warning:
 251             results['msg'] = "WARNING: Could not find status for all services. Sometimes this is due to insufficient privileges."
 252     module.exit_json(**results)
