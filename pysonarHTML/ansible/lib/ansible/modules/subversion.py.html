 121 class Subversion(object):
 122     def __init__(self, module, dest, repo, revision, username, password, svn_path):
 123         self.module = module
 124         self.dest = dest
 125         self.repo = repo
 126         self.revision = revision
 127         self.username = username
 128         self.password = password
 129         self.svn_path = svn_path
 130 
 131     def has_option_password_from_stdin(self):
 132         rc, version, err = self.module.run_command([self.svn_path, '--version', '--quiet'], check_rc=True)
 133         return LooseVersion(version) >= LooseVersion('1.10.0')
 134 
 135     def _exec(self, args, check_rc=True):
 136         '''Execute a subversion command, and return output. If check_rc is False, returns the return code instead of the output.'''
 137         bits = [
 138             self.svn_path,
 139             '--non-interactive',
 140             '--trust-server-cert',
 141             '--no-auth-cache',
 142         ]
 143         stdin_data = None
 144         if self.username:
 145             bits.extend(["--username", self.username])
 146         if self.password:
 147             if self.has_option_password_from_stdin():
 148                 bits.append("--password-from-stdin")
 149                 stdin_data = self.password
 150             else:
 151                 self.module.warn("The authentication provided will be used on the svn command line and is not secure. "
 152                                  "To securely pass credentials, upgrade svn to version 1.10.0 or greater.")
 153                 bits.extend(["--password", self.password])
 154         bits.extend(args)
 155         rc, out, err = self.module.run_command(bits, check_rc, data=stdin_data)
 156 
 157         if check_rc:
 158             return out.splitlines()
 159         else:
 160             return rc
 161 
 162     def is_svn_repo(self):
 163         '''Checks if path is a SVN Repo.'''
 164         rc = self._exec(["info", self.dest], check_rc=False)
 165         return rc == 0
 166 
 167     def checkout(self, force=False):
 168         '''Creates new svn working directory if it does not already exist.'''
 169         cmd = ["checkout"]
 170         if force:
 171             cmd.append("--force")
 172         cmd.extend(["-r", self.revision, self.repo, self.dest])
 173         self._exec(cmd)
 174 
 175     def export(self, force=False):
 176         '''Export svn repo to directory'''
 177         cmd = ["export"]
 178         if force:
 179             cmd.append("--force")
 180         cmd.extend(["-r", self.revision, self.repo, self.dest])
 181 
 182         self._exec(cmd)
 183 
 184     def switch(self):
 185         '''Change working directory's repo.'''
 186         # switch to ensure we are pointing at correct repo.
 187         # it also updates!
 188         output = self._exec(["switch", "--revision", self.revision, self.repo, self.dest])
 189         for line in output:
 190             if re.search(r'^[ABDUCGE]\s', line):
 191                 return True
 192         return False
 193 
 194     def update(self):
 195         '''Update existing svn working directory.'''
 196         output = self._exec(["update", "-r", self.revision, self.dest])
 197 
 198         for line in output:
 199             if re.search(r'^[ABDUCGE]\s', line):
 200                 return True
 201         return False
 202 
 203     def revert(self):
 204         '''Revert svn working directory.'''
 205         output = self._exec(["revert", "-R", self.dest])
 206         for line in output:
 207             if re.search(r'^Reverted ', line) is None:
 208                 return True
 209         return False
 210 
 211     def get_revision(self):
 212         '''Revision and URL of subversion working directory.'''
 213         text = '\n'.join(self._exec(["info", self.dest]))
 214         rev = re.search(r'^Revision:.*$', text, re.MULTILINE).group(0)
 215         url = re.search(r'^URL:.*$', text, re.MULTILINE).group(0)
 216         return rev, url
 217 
 218     def get_remote_revision(self):
 219         '''Revision and URL of subversion working directory.'''
 220         text = '\n'.join(self._exec(["info", self.repo]))
 221         rev = re.search(r'^Revision:.*$', text, re.MULTILINE).group(0)
 222         return rev
 223 
 224     def has_local_mods(self):
 225         '''True if revisioned files have been added or modified. Unrevisioned files are ignored.'''
 226         lines = self._exec(["status", "--quiet", "--ignore-externals", self.dest])
 227         # The --quiet option will return only modified files.
 228         # Match only revisioned files, i.e. ignore status '?'.
 229         regex = re.compile(r'^[^?X]')
 230         # Has local mods if more than 0 modified revisioned files.
 231         return len(list(filter(regex.match, lines))) > 0
 232 
 233     def needs_update(self):
 234         curr, url = self.get_revision()
 235         out2 = '\n'.join(self._exec(["info", "-r", self.revision, self.dest]))
 236         head = re.search(r'^Revision:.*$', out2, re.MULTILINE).group(0)
 237         rev1 = int(curr.split(':')[1].strip())
 238         rev2 = int(head.split(':')[1].strip())
 239         change = False
 240         if rev1 < rev2:
 241             change = True
 242         return change, curr, head

 245 def main():
 246     module = AnsibleModule(
 247         argument_spec=dict(
 248             dest=dict(type='path'),
 249             repo=dict(type='str', required=True, aliases=['name', 'repository']),
 250             revision=dict(type='str', default='HEAD', aliases=['rev', 'version']),
 251             force=dict(type='bool', default=False),
 252             username=dict(type='str'),
 253             password=dict(type='str', no_log=True),
 254             executable=dict(type='path'),
 255             export=dict(type='bool', default=False),
 256             checkout=dict(type='bool', default=True),
 257             update=dict(type='bool', default=True),
 258             switch=dict(type='bool', default=True),
 259             in_place=dict(type='bool', default=False),
 260         ),
 261         supports_check_mode=True,
 262     )
 263 
 264     dest = module.params['dest']
 265     repo = module.params['repo']
 266     revision = module.params['revision']
 267     force = module.params['force']
 268     username = module.params['username']
 269     password = module.params['password']
 270     svn_path = module.params['executable'] or module.get_bin_path('svn', True)
 271     export = module.params['export']
 272     switch = module.params['switch']
 273     checkout = module.params['checkout']
 274     update = module.params['update']
 275     in_place = module.params['in_place']
 276 
 277     # We screenscrape a huge amount of svn commands so use C locale anytime we
 278     # call run_command()
 279     module.run_command_environ_update = dict(LANG='C', LC_MESSAGES='C')
 280 
 281     if not dest and (checkout or update or export):
 282         module.fail_json(msg="the destination directory must be specified unless checkout=no, update=no, and export=no")
 283 
 284     svn = Subversion(module, dest, repo, revision, username, password, svn_path)
 285 
 286     if not export and not update and not checkout:
 287         module.exit_json(changed=False, after=svn.get_remote_revision())
 288     if export or not os.path.exists(dest):
 289         before = None
 290         local_mods = False
 291         if module.check_mode:
 292             module.exit_json(changed=True)
 293         elif not export and not checkout:
 294             module.exit_json(changed=False)
 295         if not export and checkout:
 296             svn.checkout()
 297             files_changed = True
 298         else:
 299             svn.export(force=force)
 300             files_changed = True
 301     elif svn.is_svn_repo():
 302         # Order matters. Need to get local mods before switch to avoid false
 303         # positives. Need to switch before revert to ensure we are reverting to
 304         # correct repo.
 305         if not update:
 306             module.exit_json(changed=False)
 307         if module.check_mode:
 308             if svn.has_local_mods() and not force:
 309                 module.fail_json(msg="ERROR: modified files exist in the repository.")
 310             check, before, after = svn.needs_update()
 311             module.exit_json(changed=check, before=before, after=after)
 312         files_changed = False
 313         before = svn.get_revision()
 314         local_mods = svn.has_local_mods()
 315         if switch:
 316             files_changed = svn.switch() or files_changed
 317         if local_mods:
 318             if force:
 319                 files_changed = svn.revert() or files_changed
 320             else:
 321                 module.fail_json(msg="ERROR: modified files exist in the repository.")
 322         files_changed = svn.update() or files_changed
 323     elif in_place:
 324         before = None
 325         svn.checkout(force=True)
 326         files_changed = True
 327         local_mods = svn.has_local_mods()
 328         if local_mods and force:
 329             svn.revert()
 330     else:
 331         module.fail_json(msg="ERROR: %s folder already exists, but its not a subversion repository." % (dest,))
 332 
 333     if export:
 334         module.exit_json(changed=True)
 335     else:
 336         after = svn.get_revision()
 337         changed = files_changed or local_mods
 338         module.exit_json(changed=changed, before=before, after=after)
 339 
 340 
 341 if __name__ == '__main__':
 342     main()
