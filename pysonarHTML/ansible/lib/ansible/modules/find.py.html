 225 def pfilter(f, patterns=None, excludes=None, use_regex=False):
 226     '''filter using glob patterns'''
 227     if patterns is None and excludes is None:
 228         return True
 229 
 230     if use_regex:
 231         if patterns and excludes is None:
 232             for p in patterns:
 233                 r = re.compile(p)
 234                 if r.match(f):
 235                     return True
 236 
 237         elif patterns and excludes:
 238             for p in patterns:
 239                 r = re.compile(p)
 240                 if r.match(f):
 241                     for e in excludes:
 242                         r = re.compile(e)
 243                         if r.match(f):
 244                             return False
 245                     return True
 246 
 247     else:
 248         if patterns and excludes is None:
 249             for p in patterns:
 250                 if fnmatch.fnmatch(f, p):
 251                     return True
 252 
 253         elif patterns and excludes:
 254             for p in patterns:
 255                 if fnmatch.fnmatch(f, p):
 256                     for e in excludes:
 257                         if fnmatch.fnmatch(f, e):
 258                             return False
 259                     return True
 260 
 261     return False

 264 def agefilter(st, now, age, timestamp):
 265     '''filter files older than age'''
 266     if age is None:
 267         return True
 268     elif age >= 0 and now - st.__getattribute__("st_%s" % timestamp) >= abs(age):
 269         return True
 270     elif age < 0 and now - st.__getattribute__("st_%s" % timestamp) <= abs(age):
 271         return True
 272     return False

 275 def sizefilter(st, size):
 276     '''filter files greater than size'''
 277     if size is None:
 278         return True
 279     elif size >= 0 and st.st_size >= abs(size):
 280         return True
 281     elif size < 0 and st.st_size <= abs(size):
 282         return True
 283     return False

 286 def contentfilter(fsname, pattern):
 287     """
 288     Filter files which contain the given expression
 289     :arg fsname: Filename to scan for lines matching a pattern
 290     :arg pattern: Pattern to look for inside of line
 291     :rtype: bool
 292     :returns: True if one of the lines in fsname matches the pattern. Otherwise False
 293     """
 294     if pattern is None:
 295         return True
 296 
 297     prog = re.compile(pattern)
 298 
 299     try:
 300         with open(fsname) as f:
 301             for line in f:
 302                 if prog.match(line):
 303                     return True
 304 
 305     except Exception:
 306         pass
 307 
 308     return False

 311 def statinfo(st):
 312     pw_name = ""
 313     gr_name = ""
 314 
 315     try:  # user data
 316         pw_name = pwd.getpwuid(st.st_uid).pw_name
 317     except Exception:
 318         pass
 319 
 320     try:  # group data
 321         gr_name = grp.getgrgid(st.st_gid).gr_name
 322     except Exception:
 323         pass
 324 
 325     return {
 326         'mode': "%04o" % stat.S_IMODE(st.st_mode),
 327         'isdir': stat.S_ISDIR(st.st_mode),
 328         'ischr': stat.S_ISCHR(st.st_mode),
 329         'isblk': stat.S_ISBLK(st.st_mode),
 330         'isreg': stat.S_ISREG(st.st_mode),
 331         'isfifo': stat.S_ISFIFO(st.st_mode),
 332         'islnk': stat.S_ISLNK(st.st_mode),
 333         'issock': stat.S_ISSOCK(st.st_mode),
 334         'uid': st.st_uid,
 335         'gid': st.st_gid,
 336         'size': st.st_size,
 337         'inode': st.st_ino,
 338         'dev': st.st_dev,
 339         'nlink': st.st_nlink,
 340         'atime': st.st_atime,
 341         'mtime': st.st_mtime,
 342         'ctime': st.st_ctime,
 343         'gr_name': gr_name,
 344         'pw_name': pw_name,
 345         'wusr': bool(st.st_mode & stat.S_IWUSR),
 346         'rusr': bool(st.st_mode & stat.S_IRUSR),
 347         'xusr': bool(st.st_mode & stat.S_IXUSR),
 348         'wgrp': bool(st.st_mode & stat.S_IWGRP),
 349         'rgrp': bool(st.st_mode & stat.S_IRGRP),
 350         'xgrp': bool(st.st_mode & stat.S_IXGRP),
 351         'woth': bool(st.st_mode & stat.S_IWOTH),
 352         'roth': bool(st.st_mode & stat.S_IROTH),
 353         'xoth': bool(st.st_mode & stat.S_IXOTH),
 354         'isuid': bool(st.st_mode & stat.S_ISUID),
 355         'isgid': bool(st.st_mode & stat.S_ISGID),
 356     }

 359 def main():
 360     module = AnsibleModule(
 361         argument_spec=dict(
 362             paths=dict(type='list', required=True, aliases=['name', 'path'], elements='str'),
 363             patterns=dict(type='list', default=['*'], aliases=['pattern'], elements='str'),
 364             excludes=dict(type='list', aliases=['exclude'], elements='str'),
 365             contains=dict(type='str'),
 366             file_type=dict(type='str', default="file", choices=['any', 'directory', 'file', 'link']),
 367             age=dict(type='str'),
 368             age_stamp=dict(type='str', default="mtime", choices=['atime', 'ctime', 'mtime']),
 369             size=dict(type='str'),
 370             recurse=dict(type='bool', default=False),
 371             hidden=dict(type='bool', default=False),
 372             follow=dict(type='bool', default=False),
 373             get_checksum=dict(type='bool', default=False),
 374             use_regex=dict(type='bool', default=False),
 375             depth=dict(type='int'),
 376         ),
 377         supports_check_mode=True,
 378     )
 379 
 380     params = module.params
 381 
 382     filelist = []
 383 
 384     if params['age'] is None:
 385         age = None
 386     else:
 387         # convert age to seconds:
 388         m = re.match(r"^(-?\d+)(s|m|h|d|w)?$", params['age'].lower())
 389         seconds_per_unit = {"s": 1, "m": 60, "h": 3600, "d": 86400, "w": 604800}
 390         if m:
 391             age = int(m.group(1)) * seconds_per_unit.get(m.group(2), 1)
 392         else:
 393             module.fail_json(age=params['age'], msg="failed to process age")
 394 
 395     if params['size'] is None:
 396         size = None
 397     else:
 398         # convert size to bytes:
 399         m = re.match(r"^(-?\d+)(b|k|m|g|t)?$", params['size'].lower())
 400         bytes_per_unit = {"b": 1, "k": 1024, "m": 1024**2, "g": 1024**3, "t": 1024**4}
 401         if m:
 402             size = int(m.group(1)) * bytes_per_unit.get(m.group(2), 1)
 403         else:
 404             module.fail_json(size=params['size'], msg="failed to process size")
 405 
 406     now = time.time()
 407     msg = ''
 408     looked = 0
 409     for npath in params['paths']:
 410         npath = os.path.expanduser(os.path.expandvars(npath))
 411         if os.path.isdir(npath):
 412             for root, dirs, files in os.walk(npath, followlinks=params['follow']):
 413                 looked = looked + len(files) + len(dirs)
 414                 for fsobj in (files + dirs):
 415                     fsname = os.path.normpath(os.path.join(root, fsobj))
 416                     if params['depth']:
 417                         wpath = npath.rstrip(os.path.sep) + os.path.sep
 418                         depth = int(fsname.count(os.path.sep)) - int(wpath.count(os.path.sep)) + 1
 419                         if depth > params['depth']:
 420                             continue
 421                     if os.path.basename(fsname).startswith('.') and not params['hidden']:
 422                         continue
 423 
 424                     try:
 425                         st = os.lstat(fsname)
 426                     except Exception:
 427                         msg += "%s was skipped as it does not seem to be a valid file or it cannot be accessed\n" % fsname
 428                         continue
 429 
 430                     r = {'path': fsname}
 431                     if params['file_type'] == 'any':
 432                         if pfilter(fsobj, params['patterns'], params['excludes'], params['use_regex']) and agefilter(st, now, age, params['age_stamp']):
 433 
 434                             r.update(statinfo(st))
 435                             if stat.S_ISREG(st.st_mode) and params['get_checksum']:
 436                                 r['checksum'] = module.sha1(fsname)
 437                             filelist.append(r)
 438 
 439                     elif stat.S_ISDIR(st.st_mode) and params['file_type'] == 'directory':
 440                         if pfilter(fsobj, params['patterns'], params['excludes'], params['use_regex']) and agefilter(st, now, age, params['age_stamp']):
 441 
 442                             r.update(statinfo(st))
 443                             filelist.append(r)
 444 
 445                     elif stat.S_ISREG(st.st_mode) and params['file_type'] == 'file':
 446                         if pfilter(fsobj, params['patterns'], params['excludes'], params['use_regex']) and \
 447                            agefilter(st, now, age, params['age_stamp']) and \
 448                            sizefilter(st, size) and contentfilter(fsname, params['contains']):
 449 
 450                             r.update(statinfo(st))
 451                             if params['get_checksum']:
 452                                 r['checksum'] = module.sha1(fsname)
 453                             filelist.append(r)
 454 
 455                     elif stat.S_ISLNK(st.st_mode) and params['file_type'] == 'link':
 456                         if pfilter(fsobj, params['patterns'], params['excludes'], params['use_regex']) and agefilter(st, now, age, params['age_stamp']):
 457 
 458                             r.update(statinfo(st))
 459                             filelist.append(r)
 460 
 461                 if not params['recurse']:
 462                     break
 463         else:
 464             msg += "%s was skipped as it does not seem to be a valid directory or it cannot be accessed\n" % npath
 465 
 466     matched = len(filelist)
 467     module.exit_json(files=filelist, changed=False, msg=msg, matched=matched, examined=looked)
