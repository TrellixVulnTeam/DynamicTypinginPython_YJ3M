 401 def write_file(module, url, dest, content, resp):
 402     # create a tempfile with some test content
 403     fd, tmpsrc = tempfile.mkstemp(dir=module.tmpdir)
 404     f = open(tmpsrc, 'wb')
 405     try:
 406         f.write(content)
 407     except Exception as e:
 408         os.remove(tmpsrc)
 409         msg = format_message("Failed to create temporary content file: %s" % to_native(e), resp)
 410         module.fail_json(msg=msg, **resp)
 411     f.close()
 412 
 413     checksum_src = None
 414     checksum_dest = None
 415 
 416     # raise an error if there is no tmpsrc file
 417     if not os.path.exists(tmpsrc):
 418         os.remove(tmpsrc)
 419         msg = format_message("Source '%s' does not exist" % tmpsrc, resp)
 420         module.fail_json(msg=msg, **resp)
 421     if not os.access(tmpsrc, os.R_OK):
 422         os.remove(tmpsrc)
 423         msg = format_message("Source '%s' not readable" % tmpsrc, resp)
 424         module.fail_json(msg=msg, **resp)
 425     checksum_src = module.sha1(tmpsrc)
 426 
 427     # check if there is no dest file
 428     if os.path.exists(dest):
 429         # raise an error if copy has no permission on dest
 430         if not os.access(dest, os.W_OK):
 431             os.remove(tmpsrc)
 432             msg = format_message("Destination '%s' not writable" % dest, resp)
 433             module.fail_json(msg=msg, **resp)
 434         if not os.access(dest, os.R_OK):
 435             os.remove(tmpsrc)
 436             msg = format_message("Destination '%s' not readable" % dest, resp)
 437             module.fail_json(msg=msg, **resp)
 438         checksum_dest = module.sha1(dest)
 439     else:
 440         if not os.access(os.path.dirname(dest), os.W_OK):
 441             os.remove(tmpsrc)
 442             msg = format_message("Destination dir '%s' not writable" % os.path.dirname(dest), resp)
 443             module.fail_json(msg=msg, **resp)
 444 
 445     if checksum_src != checksum_dest:
 446         try:
 447             shutil.copyfile(tmpsrc, dest)
 448         except Exception as e:
 449             os.remove(tmpsrc)
 450             msg = format_message("failed to copy %s to %s: %s" % (tmpsrc, dest, to_native(e)), resp)
 451             module.fail_json(msg=msg, **resp)
 452 
 453     os.remove(tmpsrc)

 456 def url_filename(url):

 463 def absolute_location(url, location):
 464     """Attempts to create an absolute URL based on initial URL, and
 465     next URL, specifically in the case of a ``Location`` header.
 466     """
 467 
 468     if '://' in location:
 469         return location
 470 
 471     elif location.startswith('/'):
 472         parts = urlsplit(url)
 473         base = url.replace(parts[2], '')
 474         return '%s%s' % (base, location)
 475 
 476     elif not location.startswith('/'):
 477         base = os.path.dirname(url)
 478         return '%s/%s' % (base, location)
 479 
 480     else:
 481         return location

 484 def kv_list(data):
 485     ''' Convert data into a list of key-value tuples '''
 486     if data is None:
 487         return None
 488 
 489     if isinstance(data, Sequence):
 490         return list(data)
 491 
 492     if isinstance(data, Mapping):
 493         return list(data.items())
 494 
 495     raise TypeError('cannot form-urlencode body, expect list or dict')
 496

 498 def form_urlencoded(body):
 499     ''' Convert data into a form-urlencoded string '''
 500     if isinstance(body, string_types):
 501         return body
 502 
 503     if isinstance(body, (Mapping, Sequence)):
 504         result = []
 505         # Turn a list of lists into a list of tuples that urlencode accepts
 506         for key, values in kv_list(body):
 507             if isinstance(values, string_types) or not isinstance(values, (Mapping, Sequence)):
 508                 values = [values]
 509             for value in values:
 510                 if value is not None:
 511                     result.append((to_text(key), to_text(value)))
 512         return urlencode(result, doseq=True)
 513 
 514     return body

 517 def uri(module, url, dest, body, body_format, method, headers, socket_timeout):
 518     # is dest is set and is a directory, let's check if we get redirected and
 519     # set the filename from that url
 520     redirected = False
 521     redir_info = {}
 522     r = {}
 523 
 524     src = module.params['src']
 525     if src:
 526         try:
 527             headers.update({
 528                 'Content-Length': os.stat(src).st_size
 529             })
 530             data = open(src, 'rb')
 531         except OSError:
 532             module.fail_json(msg='Unable to open source file %s' % src, elapsed=0)
 533     else:
 534         data = body
 535 
 536     kwargs = {}
 537     if dest is not None:
 538         # Stash follow_redirects, in this block we don't want to follow
 539         # we'll reset back to the supplied value soon
 540         follow_redirects = module.params['follow_redirects']
 541         module.params['follow_redirects'] = False
 542         if os.path.isdir(dest):
 543             # first check if we are redirected to a file download
 544             _, redir_info = fetch_url(module, url, data=body,
 545                                       headers=headers,
 546                                       method=method,
 547                                       timeout=socket_timeout, unix_socket=module.params['unix_socket'])
 548             # if we are redirected, update the url with the location header,
 549             # and update dest with the new url filename
 550             if redir_info['status'] in (301, 302, 303, 307):
 551                 url = redir_info['location']
 552                 redirected = True
 553             dest = os.path.join(dest, url_filename(url))
 554         # if destination file already exist, only download if file newer
 555         if os.path.exists(dest):
 556             kwargs['last_mod_time'] = datetime.datetime.utcfromtimestamp(os.path.getmtime(dest))
 557 
 558         # Reset follow_redirects back to the stashed value
 559         module.params['follow_redirects'] = follow_redirects
 560 
 561     resp, info = fetch_url(module, url, data=data, headers=headers,
 562                            method=method, timeout=socket_timeout, unix_socket=module.params['unix_socket'],
 563                            **kwargs)
 564 
 565     try:
 566         content = resp.read()
 567     except AttributeError:
 568         # there was no content, but the error read()
 569         # may have been stored in the info as 'body'
 570         content = info.pop('body', '')
 571 
 572     if src:
 573         # Try to close the open file handle
 574         try:
 575             data.close()
 576         except Exception:
 577             pass
 578 
 579     r['redirected'] = redirected or info['url'] != url
 580     r.update(redir_info)
 581     r.update(info)
 582 
 583     return r, content, dest

 586 def main():
 587     argument_spec = url_argument_spec()
 588     argument_spec.update(
 589         dest=dict(type='path'),
 590         url_username=dict(type='str', aliases=['user']),
 591         url_password=dict(type='str', aliases=['password'], no_log=True),
 592         body=dict(type='raw'),
 593         body_format=dict(type='str', default='raw', choices=['form-urlencoded', 'json', 'raw', 'form-multipart']),
 594         src=dict(type='path'),
 595         method=dict(type='str', default='GET'),
 596         return_content=dict(type='bool', default=False),
 597         follow_redirects=dict(type='str', default='safe', choices=['all', 'no', 'none', 'safe', 'urllib2', 'yes']),
 598         creates=dict(type='path'),
 599         removes=dict(type='path'),
 600         status_code=dict(type='list', default=[200]),
 601         timeout=dict(type='int', default=30),
 602         headers=dict(type='dict', default={}),
 603         unix_socket=dict(type='path'),
 604         remote_src=dict(type='bool', default=False),
 605     )
 606 
 607     module = AnsibleModule(
 608         argument_spec=argument_spec,
 609         add_file_common_args=True,
 610         mutually_exclusive=[['body', 'src']],
 611     )
 612 
 613     if module.params.get('thirsty'):
 614         module.deprecate('The alias "thirsty" has been deprecated and will be removed, use "force" instead',
 615                          version='2.13', collection_name='ansible.builtin')
 616 
 617     url = module.params['url']
 618     body = module.params['body']
 619     body_format = module.params['body_format'].lower()
 620     method = module.params['method'].upper()
 621     dest = module.params['dest']
 622     return_content = module.params['return_content']
 623     creates = module.params['creates']
 624     removes = module.params['removes']
 625     status_code = [int(x) for x in list(module.params['status_code'])]
 626     socket_timeout = module.params['timeout']
 627 
 628     dict_headers = module.params['headers']
 629 
 630     if not re.match('^[A-Z]+$', method):
 631         module.fail_json(msg="Parameter 'method' needs to be a single word in uppercase, like GET or POST.")
 632 
 633     if body_format == 'json':
 634         # Encode the body unless its a string, then assume it is pre-formatted JSON
 635         if not isinstance(body, string_types):
 636             body = json.dumps(body)
 637         if 'content-type' not in [header.lower() for header in dict_headers]:
 638             dict_headers['Content-Type'] = 'application/json'
 639     elif body_format == 'form-urlencoded':
 640         if not isinstance(body, string_types):
 641             try:
 642                 body = form_urlencoded(body)
 643             except ValueError as e:
 644                 module.fail_json(msg='failed to parse body as form_urlencoded: %s' % to_native(e), elapsed=0)
 645         if 'content-type' not in [header.lower() for header in dict_headers]:
 646             dict_headers['Content-Type'] = 'application/x-www-form-urlencoded'
 647     elif body_format == 'form-multipart':
 648         try:
 649             content_type, body = prepare_multipart(body)
 650         except (TypeError, ValueError) as e:
 651             module.fail_json(msg='failed to parse body as form-multipart: %s' % to_native(e))
 652         dict_headers['Content-Type'] = content_type
 653 
 654     if creates is not None:
 655         # do not run the command if the line contains creates=filename
 656         # and the filename already exists.  This allows idempotence
 657         # of uri executions.
 658         if os.path.exists(creates):
 659             module.exit_json(stdout="skipped, since '%s' exists" % creates, changed=False)
 660 
 661     if removes is not None:
 662         # do not run the command if the line contains removes=filename
 663         # and the filename does not exist.  This allows idempotence
 664         # of uri executions.
 665         if not os.path.exists(removes):
 666             module.exit_json(stdout="skipped, since '%s' does not exist" % removes, changed=False)
 667 
 668     # Make the request
 669     start = datetime.datetime.utcnow()
 670     resp, content, dest = uri(module, url, dest, body, body_format, method,
 671                               dict_headers, socket_timeout)
 672     resp['elapsed'] = (datetime.datetime.utcnow() - start).seconds
 673     resp['status'] = int(resp['status'])
 674     resp['changed'] = False
 675 
 676     # Write the file out if requested
 677     if dest is not None:
 678         if resp['status'] in status_code and resp['status'] != 304:
 679             write_file(module, url, dest, content, resp)
 680             # allow file attribute changes
 681             resp['changed'] = True
 682             module.params['path'] = dest
 683             file_args = module.load_file_common_arguments(module.params, path=dest)
 684             resp['changed'] = module.set_fs_attributes_if_different(file_args, resp['changed'])
 685         resp['path'] = dest
 686 
 687     # Transmogrify the headers, replacing '-' with '_', since variables don't
 688     # work with dashes.
 689     # In python3, the headers are title cased.  Lowercase them to be
 690     # compatible with the python2 behaviour.
 691     uresp = {}
 692     for key, value in iteritems(resp):
 693         ukey = key.replace("-", "_").lower()
 694         uresp[ukey] = value
 695 
 696     if 'location' in uresp:
 697         uresp['location'] = absolute_location(url, uresp['location'])
 698 
 699     # Default content_encoding to try
 700     content_encoding = 'utf-8'
 701     if 'content_type' in uresp:
 702         # Handle multiple Content-Type headers
 703         charsets = []
 704         content_types = []
 705         for value in uresp['content_type'].split(','):
 706             ct, params = cgi.parse_header(value)
 707             if ct not in content_types:
 708                 content_types.append(ct)
 709             if 'charset' in params:
 710                 if params['charset'] not in charsets:
 711                     charsets.append(params['charset'])
 712 
 713         if content_types:
 714             content_type = content_types[0]
 715             if len(content_types) > 1:
 716                 module.warn(
 717                     'Received multiple conflicting Content-Type values (%s), using %s' % (', '.join(content_types), content_type)
 718                 )
 719         if charsets:
 720             content_encoding = charsets[0]
 721             if len(charsets) > 1:
 722                 module.warn(
 723                     'Received multiple conflicting charset values (%s), using %s' % (', '.join(charsets), content_encoding)
 724                 )
 725 
 726         u_content = to_text(content, encoding=content_encoding)
 727         if any(candidate in content_type for candidate in JSON_CANDIDATES):
 728             try:
 729                 js = json.loads(u_content)
 730                 uresp['json'] = js
 731             except Exception:
 732                 if PY2:
 733                     sys.exc_clear()  # Avoid false positive traceback in fail_json() on Python 2
 734     else:
 735         u_content = to_text(content, encoding=content_encoding)
 736 
 737     if resp['status'] not in status_code:
 738         uresp['msg'] = 'Status code was %s and not %s: %s' % (resp['status'], status_code, uresp.get('msg', ''))
 739         if return_content:
 740             module.fail_json(content=u_content, **uresp)
 741         else:
 742             module.fail_json(**uresp)
 743     elif return_content:
 744         module.exit_json(content=u_content, **uresp)
 745     else:
 746         module.exit_json(**uresp)
 747 
 748 
 749 if __name__ == '__main__':
 750     main()
