 429     def is_lockfile_pid_valid(self):
 430         try:
 431             try:
 432                 with open(self.lockfile, 'r') as f:
 433                     oldpid = int(f.readline())
 434             except ValueError:
 435                 # invalid data
 436                 os.unlink(self.lockfile)
 437                 return False
 438 
 439             if oldpid == os.getpid():
 440                 # that's us?
 441                 os.unlink(self.lockfile)
 442                 return False
 443 
 444             try:
 445                 with open("/proc/%d/stat" % oldpid, 'r') as f:
 446                     stat = f.readline()
 447 
 448                 if stat.split()[2] == 'Z':
 449                     # Zombie
 450                     os.unlink(self.lockfile)
 451                     return False
 452             except IOError:
 453                 # either /proc is not mounted or the process is already dead
 454                 try:
 455                     # check the state of the process
 456                     os.kill(oldpid, 0)
 457                 except OSError as e:
 458                     if e.errno == errno.ESRCH:
 459                         # No such process
 460                         os.unlink(self.lockfile)
 461                         return False
 462 
 463                     self.module.fail_json(msg="Unable to check PID %s in  %s: %s" % (oldpid, self.lockfile, to_native(e)))
 464         except (IOError, OSError) as e:
 465             # lockfile disappeared?
 466             return False
 467 
 468         # another copy seems to be running
 469         return True

 524     def is_group_env_installed(self, name):
 525         name_lower = name.lower()
 526 
 527         if yum.__version_info__ >= (3, 4):
 528             groups_list = self.yum_base.doGroupLists(return_evgrps=True)
 529         else:
 530             groups_list = self.yum_base.doGroupLists()
 531 
 532         # list of the installed groups on the first index
 533         groups = groups_list[0]
 534         for group in groups:
 535             if name_lower.endswith(group.name.lower()) or name_lower.endswith(group.groupid.lower()):
 536                 return True
 537 
 538         if yum.__version_info__ >= (3, 4):
 539             # list of the installed env_groups on the third index
 540             envs = groups_list[2]
 541             for env in envs:
 542                 if name_lower.endswith(env.name.lower()) or name_lower.endswith(env.environmentid.lower()):
 543                     return True
 544 
 545         return False

 547     def is_installed(self, repoq, pkgspec, qf=None, is_pkg=False):
 548         if qf is None:
 549             qf = "%{epoch}:%{name}-%{version}-%{release}.%{arch}\n"
 550 
 551         if not repoq:
 552             pkgs = []
 553             try:
 554                 e, m, _ = self.yum_base.rpmdb.matchPackageNames([pkgspec])
 555                 pkgs = e + m
 556                 if not pkgs and not is_pkg:
 557                     pkgs.extend(self.yum_base.returnInstalledPackagesByDep(pkgspec))
 558             except Exception as e:
 559                 self.module.fail_json(msg="Failure talking to yum: %s" % to_native(e))
 560 
 561             return [self.po_to_envra(p) for p in pkgs]
 562 
 563         else:
 564             global rpmbin
 565             if not rpmbin:
 566                 rpmbin = self.module.get_bin_path('rpm', required=True)
 567 
 568             cmd = [rpmbin, '-q', '--qf', qf, pkgspec]
 569             if self.installroot != '/':
 570                 cmd.extend(['--root', self.installroot])
 571             # rpm localizes messages and we're screen scraping so make sure we use
 572             # the C locale
 573             lang_env = dict(LANG='C', LC_ALL='C', LC_MESSAGES='C')
 574             rc, out, err = self.module.run_command(cmd, environ_update=lang_env)
 575             if rc != 0 and 'is not installed' not in out:
 576                 self.module.fail_json(msg='Error from rpm: %s: %s' % (cmd, err))
 577             if 'is not installed' in out:
 578                 out = ''
 579 
 580             pkgs = [p for p in out.replace('(none)', '0').split('\n') if p.strip()]
 581             if not pkgs and not is_pkg:
 582                 cmd = [rpmbin, '-q', '--qf', qf, '--whatprovides', pkgspec]
 583                 if self.installroot != '/':
 584                     cmd.extend(['--root', self.installroot])
 585                 rc2, out2, err2 = self.module.run_command(cmd, environ_update=lang_env)
 586             else:
 587                 rc2, out2, err2 = (0, '', '')
 588 
 589             if rc2 != 0 and 'no package provides' not in out2:
 590                 self.module.fail_json(msg='Error from rpm: %s: %s' % (cmd, err + err2))
 591             if 'no package provides' in out2:
 592                 out2 = ''
 593             pkgs += [p for p in out2.replace('(none)', '0').split('\n') if p.strip()]
 594             return pkgs
 595 
 596         return []

 735     def transaction_exists(self, pkglist):
 736         """
 737         checks the package list to see if any packages are
 738         involved in an incomplete transaction
 739         """
 740 
 741         conflicts = []
 742         if not transaction_helpers:
 743             return conflicts
 744 
 745         # first, we create a list of the package 'nvreas'
 746         # so we can compare the pieces later more easily
 747         pkglist_nvreas = (splitFilename(pkg) for pkg in pkglist)

 749         # next, we build the list of packages that are
 750         # contained within an unfinished transaction
 751         unfinished_transactions = find_unfinished_transactions()
 752         for trans in unfinished_transactions:
 753             steps = find_ts_remaining(trans)
 754             for step in steps:
 755                 # the action is install/erase/etc., but we only
 756                 # care about the package spec contained in the step
 757                 (action, step_spec) = step

 760                 # requested for installation/updating
 761                 for pkg in pkglist_nvreas:
 762                     # if the name and arch match, we're going to assume

 765                     label = "%s-%s" % (n, a)
 766                     if n == pkg[0] and a == pkg[4]:
 767                         if label not in conflicts:
 768                             conflicts.append("%s-%s" % (n, a))
 769                         break

 948     def install(self, items, repoq):
 949 
 950         pkgs = []
 951         downgrade_pkgs = []
 952         res = {}
 953         res['results'] = []
 954         res['msg'] = ''
 955         res['rc'] = 0
 956         res['changed'] = False
 957 
 958         for spec in items:
 959             pkg = None
 960             downgrade_candidate = False
 961 
 962             # check if pkgspec is installed (if possible for idempotence)
 963             if spec.endswith('.rpm') or '://' in spec:
 964                 if '://' not in spec and not os.path.exists(spec):
 965                     res['msg'] += "No RPM file matching '%s' found on system" % spec
 966                     res['results'].append("No RPM file matching '%s' found on system" % spec)
 967                     res['rc'] = 127  # Ensure the task fails in with-loop
 968                     self.module.fail_json(**res)
 969 
 970                 if '://' in spec:
 971                     with self.set_env_proxy():
 972                         package = fetch_file(self.module, spec)
 973                         if not package.endswith('.rpm'):
 974                             # yum requires a local file to have the extension of .rpm and we
 975                             # can not guarantee that from an URL (redirects, proxies, etc)
 976                             new_package_path = '%s.rpm' % package
 977                             os.rename(package, new_package_path)
 978                             package = new_package_path
 979                 else:
 980                     package = spec
 981 
 982                 # most common case is the pkg is already installed
 983                 envra = self.local_envra(package)
 984                 if envra is None:
 985                     self.module.fail_json(msg="Failed to get nevra information from RPM package: %s" % spec)
 986                 installed_pkgs = self.is_installed(repoq, envra)

 988                     res['results'].append('%s providing %s is already installed' % (installed_pkgs[0], package))
 989                     continue
 990 
 991                 (name, ver, rel, epoch, arch) = splitFilename(envra)
 992                 installed_pkgs = self.is_installed(repoq, name)
 993

 995                 if len(installed_pkgs) == 2:
 996                     (cur_name0, cur_ver0, cur_rel0, cur_epoch0, cur_arch0) = splitFilename(installed_pkgs[0])
 997                     (cur_name1, cur_ver1, cur_rel1, cur_epoch1, cur_arch1) = splitFilename(installed_pkgs[1])
 998                     cur_epoch0 = cur_epoch0 or '0'
 999                     cur_epoch1 = cur_epoch1 or '0'
1000                     compare = compareEVR((cur_epoch0, cur_ver0, cur_rel0), (cur_epoch1, cur_ver1, cur_rel1))
1001                     if compare == 0 and cur_arch0 != cur_arch1:
1002                         for installed_pkg in installed_pkgs:
1003                             if installed_pkg.endswith(arch):
1004                                 installed_pkgs = [installed_pkg]
1005 
1006                 if len(installed_pkgs) == 1:
1007                     installed_pkg = installed_pkgs[0]
1008                     (cur_name, cur_ver, cur_rel, cur_epoch, cur_arch) = splitFilename(installed_pkg)
1009                     cur_epoch = cur_epoch or '0'
1010                     compare = compareEVR((cur_epoch, cur_ver, cur_rel), (epoch, ver, rel))
1011 
1012                     # compare > 0 -> higher version is installed
1013                     # compare == 0 -> exact version is installed
1014                     # compare < 0 -> lower version is installed
1015                     if compare > 0 and self.allow_downgrade:
1016                         downgrade_candidate = True
1017                     elif compare >= 0:
1018                         continue
1019 
1020                 # else: if there are more installed packages with the same name, that would mean
1021                 # kernel, gpg-pubkey or like, so just let yum deal with it and try to install it
1022 
1023                 pkg = package
1024 
1025             # groups
1026             elif spec.startswith('@'):
1027                 if self.is_group_env_installed(spec):
1028                     continue
1029 
1030                 pkg = spec
1031 
1032             # range requires or file-requires or pkgname :(
1033             else:
1034                 # most common case is the pkg is already installed and done
1035                 # short circuit all the bs - and search for it as a pkg in is_installed
1036                 # if you find it then we're done
1037                 if not set(['*', '?']).intersection(set(spec)):
1038                     installed_pkgs = self.is_installed(repoq, spec, is_pkg=True)
1039                     if installed_pkgs:
1040                         res['results'].append('%s providing %s is already installed' % (installed_pkgs[0], spec))
1041                         continue
1042 
1043                 # look up what pkgs provide this
1044                 pkglist = self.what_provides(repoq, spec)
1045                 if not pkglist:
1046                     res['msg'] += "No package matching '%s' found available, installed or updated" % spec
1047                     res['results'].append("No package matching '%s' found available, installed or updated" % spec)
1048                     res['rc'] = 126  # Ensure the task fails in with-loop
1049                     self.module.fail_json(**res)

1052                 # so that we don't hang on the yum operation later

1060                 # then nothing to do
1061

1064                     if self.is_installed(repoq, this, is_pkg=True):

1066                         res['results'].append('%s providing %s is already installed' % (this, spec))

1069                 # if the version of the pkg you have installed is not in ANY repo, but there are
1070                 # other versions in the repos (both higher and lower) then the previous checks won't work.
1071                 # so we check one more time. This really only works for pkgname - not for file provides or virt provides
1072                 # but virt provides should be all caught in what_provides on its own.
1073                 # highly irritating
1074                 if not found:
1075                     if self.is_installed(repoq, spec):
1076                         found = True
1077                         res['results'].append('package providing %s is already installed' % (spec))
1078 
1079                 if found:
1080                     continue
1081 
1082                 # Downgrade - The yum install command will only install or upgrade to a spec version, it will
1083                 # not install an older version of an RPM even if specified by the install spec. So we need to
1084                 # determine if this is a downgrade, and then use the yum downgrade command to install the RPM.
1085                 if self.allow_downgrade:
1086                     for package in pkglist:
1087                         # Get the NEVRA of the requested package using pkglist instead of spec because pkglist
1088                         #  contains consistently-formatted package names returned by yum, rather than user input
1089                         #  that is often not parsed correctly by splitFilename().
1090                         (name, ver, rel, epoch, arch) = splitFilename(package)
1091 
1092                         # Check if any version of the requested package is installed
1093                         inst_pkgs = self.is_installed(repoq, name, is_pkg=True)
1094                         if inst_pkgs:
1095                             (cur_name, cur_ver, cur_rel, cur_epoch, cur_arch) = splitFilename(inst_pkgs[0])
1096                             compare = compareEVR((cur_epoch, cur_ver, cur_rel), (epoch, ver, rel))
1097                             if compare > 0:
1098                                 downgrade_candidate = True
1099                             else:
1100                                 downgrade_candidate = False
1101                                 break
1102 
1103                 # If package needs to be installed/upgraded/downgraded, then pass in the spec
1104                 # we could get here if nothing provides it but that's not
1105                 # the error we're catching here
1106                 pkg = spec
1107 
1108             if downgrade_candidate and self.allow_downgrade:
1109                 downgrade_pkgs.append(pkg)
1110             else:
1111                 pkgs.append(pkg)
1112 
1113         if downgrade_pkgs:
1114             res = self.exec_install(items, 'downgrade', downgrade_pkgs, res)
1115 
1116         if pkgs:
1117             res = self.exec_install(items, 'install', pkgs, res)
1118 
1119         return res

1197     def parse_check_update(check_update_output):
1198         updates = {}
1199         obsoletes = {}
1200 
1201         # remove incorrect new lines in longer columns in output from yum check-update
1202         # yum line wrapping can move the repo to the next line
1203         #
1204         # Meant to filter out sets of lines like:
1205         #  some_looooooooooooooooooooooooooooooooooooong_package_name   1:1.2.3-1.el7
1206         #                                                                    some-repo-label
1207         #
1208         # But it also needs to avoid catching lines like:
1209         # Loading mirror speeds from cached hostfile
1210         #
1211         # ceph.x86_64                               1:11.2.0-0.el7                    ceph
1212 
1213         # preprocess string and filter out empty lines so the regex below works
1214         out = re.sub(r'\n[^\w]\W+(.*)', r' \1', check_update_output)
1215 
1216         available_updates = out.split('\n')
1217 
1218         # build update dictionary
1219         for line in available_updates:
1220             line = line.split()
1221             # ignore irrelevant lines
1222             # '*' in line matches lines like mirror lists:
1223             #      * base: mirror.corbina.net
1224             # len(line) != 3 or 6 could be junk or a continuation
1225             # len(line) = 6 is package obsoletes
1226             #
1227             # FIXME: what is  the '.' not in line  conditional for?
1228 
1229             if '*' in line or len(line) not in [3, 6] or '.' not in line[0]:
1230                 continue
1231 
1232             pkg, version, repo = line[0], line[1], line[2]
1233             name, dist = pkg.rsplit('.', 1)
1234             updates.update({name: {'version': version, 'dist': dist, 'repo': repo}})
1235 
1236             if len(line) == 6:
1237                 obsolete_pkg, obsolete_version, obsolete_repo = line[3], line[4], line[5]
1238                 obsolete_name, obsolete_dist = obsolete_pkg.rsplit('.', 1)
1239                 obsoletes.update({obsolete_name: {'version': obsolete_version, 'dist': obsolete_dist, 'repo': obsolete_repo}})
1240 
1241         return updates, obsoletes
