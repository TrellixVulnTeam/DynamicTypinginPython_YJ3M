  11 DOCUMENTATION = r'''
  12 ---
  13 author: Ansible Core Team (@ansible)
  14 module: include_role
  15 short_description: Load and execute a role
  16 description:
  17   - Dynamically loads and executes a specified role as a task.
  18   - May be used only where Ansible tasks are allowed - inside C(pre_tasks), C(tasks), or C(post_tasks) playbook objects, or as a task inside a role.
  19   - Task-level keywords, loops, and conditionals apply only to the C(include_role) statement itself.
  20   - To apply keywords to the tasks within the role, pass them using the C(apply) option or use M(import_role) instead.
  21   - Ignores some keywords, like C(until) and C(retries).
  22   - This module is also supported for Windows targets.
  23 version_added: "2.2"
  24 options:
  25   apply:
  26     description:
  27       - Accepts a hash of task keywords (e.g. C(tags), C(become)) that will be applied to all tasks within the included role.
  28     version_added: '2.7'
  29   name:
  30     description:
  31       - The name of the role to be executed.
  32     type: str
  33     required: True
  34   tasks_from:
  35     description:
  36       - File to load from a role's C(tasks/) directory.
  37     type: str
  38     default: main
  39   vars_from:
  40     description:
  41       - File to load from a role's C(vars/) directory.
  42     type: str
  43     default: main
  44   defaults_from:
  45     description:
  46       - File to load from a role's C(defaults/) directory.
  47     type: str
  48     default: main
  49   allow_duplicates:

  51       - Overrides the role's metadata setting to allow using a role more than once with the same parameters.
  52     type: bool

  58         standard variable exposure for roles listed under the C(roles) header or C(import_role) as they are exposed at
  59         playbook parsing time, and available to earlier roles and tasks as well.
  60     type: bool

  62     version_added: '2.7'
  63   handlers_from:
  64     description:

  73     autodetect what is needed, but you can set C(static) to C(yes) or C(no) at task level to control this.
