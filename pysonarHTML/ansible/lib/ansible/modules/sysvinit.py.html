  44             - "This option is mainly for use with init scripts that don't support the 'status' option."
  45     runlevels:

 111 import re
 112 from time import sleep
 113 from ansible.module_utils.basic import AnsibleModule
 114 from ansible.module_utils.service import sysv_is_enabled, get_sysv_script, sysv_exists, fail_if_missing, get_ps, daemonize
 115 
 116 
 117 def main():
 118 
 119     module = AnsibleModule(
 120         argument_spec=dict(
 121             name=dict(required=True, type='str', aliases=['service']),
 122             state=dict(choices=['started', 'stopped', 'restarted', 'reloaded'], type='str'),
 123             enabled=dict(type='bool'),
 124             sleep=dict(type='int', default=1),
 125             pattern=dict(type='str'),
 126             arguments=dict(type='str', aliases=['args']),
 127             runlevels=dict(type='list'),
 128             daemonize=dict(type='bool', default=False),
 129         ),
 130         supports_check_mode=True,
 131         required_one_of=[['state', 'enabled']],
 132     )
 133 
 134     name = module.params['name']
 135     action = module.params['state']
 136     enabled = module.params['enabled']
 137     runlevels = module.params['runlevels']
 138     pattern = module.params['pattern']
 139     sleep_for = module.params['sleep']
 140     rc = 0
 141     out = err = ''
 142     result = {
 143         'name': name,
 144         'changed': False,
 145         'status': {}
 146     }
 147 
 148     # ensure service exists, get script name
 149     fail_if_missing(module, sysv_exists(name), name)
 150     script = get_sysv_script(name)
 151 
 152     # locate binaries for service management
 153     paths = ['/sbin', '/usr/sbin', '/bin', '/usr/bin']
 154     binaries = ['chkconfig', 'update-rc.d', 'insserv', 'service']
 155 
 156     # Keeps track of the service status for various runlevels because we can
 157     # operate on multiple runlevels at once
 158     runlevel_status = {}
 159 
 160     location = {}
 161     for binary in binaries:
 162         location[binary] = module.get_bin_path(binary, opt_dirs=paths)
 163 
 164     # figure out enable status
 165     if runlevels:
 166         for rl in runlevels:
 167             runlevel_status.setdefault(rl, {})
 168             runlevel_status[rl]["enabled"] = sysv_is_enabled(name, runlevel=rl)
 169     else:
 170         runlevel_status["enabled"] = sysv_is_enabled(name)
 171 
 172     # figure out started status, everyone does it different!
 173     is_started = False
 174     worked = False
 175 
 176     # user knows other methods fail and supplied pattern
 177     if pattern:
 178         worked = is_started = get_ps(module, pattern)
 179     else:
 180         if location.get('service'):
 181             # standard tool that has been 'destandarized' by reimplementation in other OS/distros
 182             cmd = '%s %s status' % (location['service'], name)
 183         elif script:
 184             # maybe script implements status (not LSB)
 185             cmd = '%s status' % script
 186         else:
 187             module.fail_json(msg="Unable to determine service status")
 188 
 189         (rc, out, err) = module.run_command(cmd)
 190         if not rc == -1:
 191             # special case
 192             if name == 'iptables' and "ACCEPT" in out:
 193                 worked = True
 194                 is_started = True
 195 
 196             # check output messages, messy but sadly more reliable than rc
 197             if not worked and out.count('\n') <= 1:
 198 
 199                 cleanout = out.lower().replace(name.lower(), '')
 200 
 201                 for stopped in ['stop', 'is dead ', 'dead but ', 'could not access pid file', 'inactive']:
 202                     if stopped in cleanout:
 203                         worked = True
 204                         break
 205 
 206                 if not worked:
 207                     for started_status in ['run', 'start', 'active']:
 208                         if started_status in cleanout and "not " not in cleanout:
 209                             is_started = True
 210                             worked = True
 211                             break
 212 
 213             # hope rc is not lying to us, use often used 'bad' returns
 214             if not worked and rc in [1, 2, 3, 4, 69]:
 215                 worked = True
 216 
 217         if not worked:
 218             # hail mary
 219             if rc == 0:
 220                 is_started = True
 221                 worked = True
 222             # ps for luck, can only assure positive match
 223             elif get_ps(module, name):
 224                 is_started = True
 225                 worked = True
 226                 module.warn("Used ps output to match service name and determine it is up, this is very unreliable")
 227 
 228     if not worked:
 229         module.warn("Unable to determine if service is up, assuming it is down")
 230 
 231     ###########################################################################
 232     # BEGIN: Enable/Disable
 233     result['status'].setdefault('enabled', {})
 234     result['status']['enabled']['changed'] = False
 235     result['status']['enabled']['rc'] = None
 236     result['status']['enabled']['stdout'] = None
 237     result['status']['enabled']['stderr'] = None
 238     if runlevels:
 239         result['status']['enabled']['runlevels'] = runlevels
 240         for rl in runlevels:
 241             if enabled != runlevel_status[rl]["enabled"]:
 242                 result['changed'] = True
 243                 result['status']['enabled']['changed'] = True
 244 
 245         if not module.check_mode and result['changed']:
 246             # Perform enable/disable here
 247             if enabled:
 248                 if location.get('update-rc.d'):
 249                     (rc, out, err) = module.run_command("%s %s enable %s" % (location['update-rc.d'], name, ' '.join(runlevels)))
 250                 elif location.get('chkconfig'):
 251                     (rc, out, err) = module.run_command("%s --level %s %s on" % (location['chkconfig'], ''.join(runlevels), name))
 252             else:
 253                 if location.get('update-rc.d'):
 254                     (rc, out, err) = module.run_command("%s %s disable %s" % (location['update-rc.d'], name, ' '.join(runlevels)))
 255                 elif location.get('chkconfig'):
 256                     (rc, out, err) = module.run_command("%s --level %s %s off" % (location['chkconfig'], ''.join(runlevels), name))
 257     else:
 258         if enabled is not None and enabled != runlevel_status["enabled"]:
 259             result['changed'] = True
 260             result['status']['enabled']['changed'] = True
 261 
 262         if not module.check_mode and result['changed']:
 263             # Perform enable/disable here
 264             if enabled:
 265                 if location.get('update-rc.d'):
 266                     (rc, out, err) = module.run_command("%s %s defaults" % (location['update-rc.d'], name))
 267                 elif location.get('chkconfig'):
 268                     (rc, out, err) = module.run_command("%s %s on" % (location['chkconfig'], name))
 269             else:
 270                 if location.get('update-rc.d'):
 271                     (rc, out, err) = module.run_command("%s %s disable" % (location['update-rc.d'], name))
 272                 elif location.get('chkconfig'):
 273                     (rc, out, err) = module.run_command("%s %s off" % (location['chkconfig'], name))
 274 
 275     # Assigned above, might be useful is something goes sideways
 276     if not module.check_mode and result['status']['enabled']['changed']:
 277         result['status']['enabled']['rc'] = rc
 278         result['status']['enabled']['stdout'] = out
 279         result['status']['enabled']['stderr'] = err
 280         rc, out, err = None, None, None
 281 
 282         if "illegal runlevel specified" in result['status']['enabled']['stderr']:
 283             module.fail_json(msg="Illegal runlevel specified for enable operation on service %s" % name, **result)
 284     # END: Enable/Disable
 285     ###########################################################################
 286 
 287     ###########################################################################
 288     # BEGIN: state
 289     result['status'].setdefault(module.params['state'], {})
 290     result['status'][module.params['state']]['changed'] = False
 291     result['status'][module.params['state']]['rc'] = None
 292     result['status'][module.params['state']]['stdout'] = None
 293     result['status'][module.params['state']]['stderr'] = None
 294     if action:
 295         action = re.sub(r'p?ed$', '', action.lower())
 296 
 297         def runme(doit):
 298 
 299             args = module.params['arguments']
 300             cmd = "%s %s %s" % (script, doit, "" if args is None else args)
 301 
 302             # how to run
 303             if module.params['daemonize']:
 304                 (rc, out, err) = daemonize(module, cmd)
 305             else:
 306                 (rc, out, err) = module.run_command(cmd)
 307             # FIXME: ERRORS
 308 
 309             if rc != 0:
 310                 module.fail_json(msg="Failed to %s service: %s" % (action, name), rc=rc, stdout=out, stderr=err)
 311 
 312             return (rc, out, err)
 313 
 314         if action == 'restart':
 315             result['changed'] = True
 316             result['status'][module.params['state']]['changed'] = True
 317             if not module.check_mode:
 318 
 319                 # cannot rely on existing 'restart' in init script
 320                 for dothis in ['stop', 'start']:
 321                     (rc, out, err) = runme(dothis)
 322                     if sleep_for:
 323                         sleep(sleep_for)
 324 
 325         elif is_started != (action == 'start'):
 326             result['changed'] = True
 327             result['status'][module.params['state']]['changed'] = True
 328             if not module.check_mode:
 329                 rc, out, err = runme(action)
 330 
 331         elif is_started == (action == 'stop'):
 332             result['changed'] = True
 333             result['status'][module.params['state']]['changed'] = True
 334             if not module.check_mode:
 335                 rc, out, err = runme(action)
 336 
 337         if not module.check_mode and result['status'][module.params['state']]['changed']:
 338             result['status'][module.params['state']]['rc'] = rc
 339             result['status'][module.params['state']]['stdout'] = out
 340             result['status'][module.params['state']]['stderr'] = err
 341             rc, out, err = None, None, None
 342     # END: state
 343     ###########################################################################
 344 
 345     module.exit_json(**result)
 346 
 347 
 348 if __name__ == '__main__':
 349     main()
