 304 def _recover_package_name(names):
 305     """Recover package names as list from user's raw input.
 306 
 307     :input: a mixed and invalid list of names or version specifiers
 308     :return: a list of valid package name
 309 
 310     eg.
 311     input: ['django>1.11.1', '<1.11.3', 'ipaddress', 'simpleproject>1.1.0', '<2.0.0']
 312     return: ['django>1.11.1,<1.11.3', 'ipaddress', 'simpleproject>1.1.0,<2.0.0']
 313 
 314     input: ['django>1.11.1,<1.11.3,ipaddress', 'simpleproject>1.1.0,<2.0.0']
 315     return: ['django>1.11.1,<1.11.3', 'ipaddress', 'simpleproject>1.1.0,<2.0.0']
 316     """
 317     # rebuild input name to a flat list so we can tolerate any combination of input
 318     tmp = []
 319     for one_line in names:
 320         tmp.extend(one_line.split(","))
 321     names = tmp
 322 
 323     # reconstruct the names
 324     name_parts = []
 325     package_names = []
 326     in_brackets = False
 327     for name in names:
 328         if _is_package_name(name) and not in_brackets:
 329             if name_parts:
 330                 package_names.append(",".join(name_parts))
 331             name_parts = []
 332         if "[" in name:
 333             in_brackets = True
 334         if in_brackets and "]" in name:
 335             in_brackets = False
 336         name_parts.append(name)
 337     package_names.append(",".join(name_parts))
 338     return package_names

 341 def _get_cmd_options(module, cmd):
 342     thiscmd = cmd + " --help"
 343     rc, stdout, stderr = module.run_command(thiscmd)
 344     if rc != 0:
 345         module.fail_json(msg="Could not get output from %s: %s" % (thiscmd, stdout + stderr))
 346 
 347     words = stdout.strip().split()
 348     cmd_options = [x for x in words if x.startswith('--')]
 349     return cmd_options

 352 def _get_packages(module, pip, chdir):
 353     '''Return results of pip command to get packages.'''
 354     # Try 'pip list' command first.
 355     command = '%s list --format=freeze' % pip
 356     lang_env = {'LANG': 'C', 'LC_ALL': 'C', 'LC_MESSAGES': 'C'}
 357     rc, out, err = module.run_command(command, cwd=chdir, environ_update=lang_env)
 358 
 359     # If there was an error (pip version too old) then use 'pip freeze'.
 360     if rc != 0:
 361         command = '%s freeze' % pip
 362         rc, out, err = module.run_command(command, cwd=chdir)
 363         if rc != 0:
 364             _fail(module, command, out, err)
 365 
 366     return command, out, err

 369 def _is_present(module, req, installed_pkgs, pkg_command):
 370     '''Return whether or not package is installed.'''
 371     for pkg in installed_pkgs:
 372         if '==' in pkg:
 373             pkg_name, pkg_version = pkg.split('==')
 374             pkg_name = Package.canonicalize_name(pkg_name)
 375         else:
 376             continue
 377 
 378         if pkg_name == req.package_name and req.is_satisfied_by(pkg_version):
 379             return True
 380 
 381     return False

 384 def _get_pip(module, env=None, executable=None):
 385     # Older pip only installed under the "/usr/bin/pip" name.  Many Linux
 386     # distros install it there.
 387     # By default, we try to use pip required for the current python
 388     # interpreter, so people can use pip to install modules dependencies
 389     candidate_pip_basenames = ('pip2', 'pip')
 390     if PY3:
 391         # pip under python3 installs the "/usr/bin/pip3" name
 392         candidate_pip_basenames = ('pip3',)
 393 
 394     pip = None
 395     if executable is not None:
 396         if os.path.isabs(executable):
 397             pip = executable
 398         else:
 399             # If you define your own executable that executable should be the only candidate.
 400             # As noted in the docs, executable doesn't work with virtualenvs.
 401             candidate_pip_basenames = (executable,)
 402 
 403     if pip is None:
 404         if env is None:
 405             opt_dirs = []
 406             for basename in candidate_pip_basenames:
 407                 pip = module.get_bin_path(basename, False, opt_dirs)
 408                 if pip is not None:
 409                     break
 410             else:
 411                 # For-else: Means that we did not break out of the loop
 412                 # (therefore, that pip was not found)
 413                 module.fail_json(msg='Unable to find any of %s to use.  pip'
 414                                      ' needs to be installed.' % ', '.join(candidate_pip_basenames))
 415         else:
 416             # If we're using a virtualenv we must use the pip from the
 417             # virtualenv
 418             venv_dir = os.path.join(env, 'bin')
 419             candidate_pip_basenames = (candidate_pip_basenames[0], 'pip')
 420             for basename in candidate_pip_basenames:
 421                 candidate = os.path.join(venv_dir, basename)
 422                 if os.path.exists(candidate) and is_executable(candidate):
 423                     pip = candidate
 424                     break
 425             else:
 426                 # For-else: Means that we did not break out of the loop
 427                 # (therefore, that pip was not found)
 428                 module.fail_json(msg='Unable to find pip in the virtualenv, %s, ' % env +
 429                                      'under any of these names: %s. ' % (', '.join(candidate_pip_basenames)) +
 430                                      'Make sure pip is present in the virtualenv.')
 431 
 432     return pip

 435 def _fail(module, cmd, out, err):
 436     msg = ''
 437     if out:
 438         msg += "stdout: %s" % (out, )
 439     if err:
 440         msg += "\n:stderr: %s" % (err, )
 441     module.fail_json(cmd=cmd, msg=msg)

 444 def _get_package_info(module, package, env=None):
 445     """This is only needed for special packages which do not show up in pip freeze
 446 
 447     pip and setuptools fall into this category.
 448 
 449     :returns: a string containing the version number if the package is
 450         installed.  None if the package is not installed.
 451     """
 452     if env:
 453         opt_dirs = ['%s/bin' % env]
 454     else:
 455         opt_dirs = []
 456     python_bin = module.get_bin_path('python', False, opt_dirs)
 457 
 458     if python_bin is None:
 459         formatted_dep = None
 460     else:
 461         rc, out, err = module.run_command([python_bin, '-c', _SPECIAL_PACKAGE_CHECKERS[package]])
 462         if rc:
 463             formatted_dep = None
 464         else:
 465             formatted_dep = '%s==%s' % (package, out.strip())
 466     return formatted_dep

 469 def setup_virtualenv(module, env, chdir, out, err):
 470     if module.check_mode:
 471         module.exit_json(changed=True)
 472 
 473     cmd = shlex.split(module.params['virtualenv_command'])
 474 
 475     # Find the binary for the command in the PATH
 476     # and switch the command for the explicit path.
 477     if os.path.basename(cmd[0]) == cmd[0]:
 478         cmd[0] = module.get_bin_path(cmd[0], True)
 479 
 480     # Add the system-site-packages option if that
 481     # is enabled, otherwise explicitly set the option
 482     # to not use system-site-packages if that is an
 483     # option provided by the command's help function.
 484     if module.params['virtualenv_site_packages']:
 485         cmd.append('--system-site-packages')
 486     else:
 487         cmd_opts = _get_cmd_options(module, cmd[0])
 488         if '--no-site-packages' in cmd_opts:
 489             cmd.append('--no-site-packages')
 490 
 491     virtualenv_python = module.params['virtualenv_python']
 492     # -p is a virtualenv option, not compatible with pyenv or venv
 493     # this conditional validates if the command being used is not any of them
 494     if not any(ex in module.params['virtualenv_command'] for ex in ('pyvenv', '-m venv')):
 495         if virtualenv_python:
 496             cmd.append('-p%s' % virtualenv_python)
 497         elif PY3:
 498             # Ubuntu currently has a patch making virtualenv always
 499             # try to use python2.  Since Ubuntu16 works without
 500             # python2 installed, this is a problem.  This code mimics
 501             # the upstream behaviour of using the python which invoked
 502             # virtualenv to determine which python is used inside of
 503             # the virtualenv (when none are specified).
 504             cmd.append('-p%s' % sys.executable)
 505 
 506     # if venv or pyvenv are used and virtualenv_python is defined, then
 507     # virtualenv_python is ignored, this has to be acknowledged
 508     elif module.params['virtualenv_python']:
 509         module.fail_json(
 510             msg='virtualenv_python should not be used when'
 511                 ' using the venv module or pyvenv as virtualenv_command'
 512         )
 513 
 514     cmd.append(env)
 515     rc, out_venv, err_venv = module.run_command(cmd, cwd=chdir)
 516     out += out_venv
 517     err += err_venv
 518     if rc != 0:
 519         _fail(module, cmd, out, err)
 520     return out, err

 523 class Package:
 524     """Python distribution package metadata wrapper.
 525 
 526     A wrapper class for Requirement, which provides
 527     API to parse package name, version specifier,
 528     test whether a package is already satisfied.
 529     """
 530 
 531     _CANONICALIZE_RE = re.compile(r'[-_.]+')
 532 
 533     def __init__(self, name_string, version_string=None):
 534         self._plain_package = False
 535         self.package_name = name_string
 536         self._requirement = None
 537 
 538         if version_string:
 539             version_string = version_string.lstrip()
 540             separator = '==' if version_string[0].isdigit() else ' '
 541             name_string = separator.join((name_string, version_string))
 542         try:
 543             self._requirement = Requirement.parse(name_string)
 544             # old pkg_resource will replace 'setuptools' with 'distribute' when it's already installed
 545             if self._requirement.project_name == "distribute" and "setuptools" in name_string:
 546                 self.package_name = "setuptools"
 547                 self._requirement.project_name = "setuptools"
 548             else:
 549                 self.package_name = Package.canonicalize_name(self._requirement.project_name)
 550             self._plain_package = True
 551         except ValueError as e:
 552             pass
 553 
 554     @property
 555     def has_version_specifier(self):
 556         if self._plain_package:
 557             return bool(self._requirement.specs)
 558         return False
 559 
 560     def is_satisfied_by(self, version_to_test):
 561         if not self._plain_package:
 562             return False
 563         try:
 564             return self._requirement.specifier.contains(version_to_test, prereleases=True)
 565         except AttributeError:
 566             # old setuptools has no specifier, do fallback
 567             version_to_test = LooseVersion(version_to_test)
 568             return all(
 569                 op_dict[op](version_to_test, LooseVersion(ver))
 570                 for op, ver in self._requirement.specs
 571             )
 572 
 573     @staticmethod
 574     def canonicalize_name(name):
 575         # This is taken from PEP 503.
 576         return Package._CANONICALIZE_RE.sub("-", name).lower()
 577 
 578     def __str__(self):
 579         if self._plain_package:
 580             return to_native(self._requirement)
 581         return self.package_name

 636 
 637     old_umask = None
 638     if umask is not None:
 639         old_umask = os.umask(umask)
 640     try:
 641         if state == 'latest' and version is not None:
 642             module.fail_json(msg='version is incompatible with state=latest')
 643 
 644         if chdir is None:
 645             # this is done to avoid permissions issues with privilege escalation and virtualenvs
 646             chdir = tempfile.gettempdir()
 647 
 648         err = ''
 649         out = ''
 650 
 651         if env:
 652             if not os.path.exists(os.path.join(env, 'bin', 'activate')):
 653                 venv_created = True
 654                 out, err = setup_virtualenv(module, env, chdir, out, err)
 655 
 656         pip = _get_pip(module, env, module.params['executable'])
 657 
 658         cmd = [pip] + state_map[state]
 659 
 660         # If there's a virtualenv we want things we install to be able to use other
 661         # installations that exist as binaries within this virtualenv. Example: we
 662         # install cython and then gevent -- gevent needs to use the cython binary,
 663         # not just a python package that will be found by calling the right python.
 664         # So if there's a virtualenv, we add that bin/ to the beginning of the PATH
 665         # in run_command by setting path_prefix here.
 666         path_prefix = None
 667         if env:
 668             path_prefix = "/".join(pip.split('/')[:-1])
 669 
 670         # Automatically apply -e option to extra_args when source is a VCS url. VCS
 671         # includes those beginning with svn+, git+, hg+ or bzr+
 672         has_vcs = False
 673         if name:
 674             for pkg in name:
 675                 if pkg and _is_vcs_url(pkg):
 676                     has_vcs = True
 677                     break
 678 
 679             # convert raw input package names to Package instances
 680             packages = [Package(pkg) for pkg in _recover_package_name(name)]
 681             # check invalid combination of arguments
 682             if version is not None:
 683                 if len(packages) > 1:
 684                     module.fail_json(
 685                         msg="'version' argument is ambiguous when installing multiple package distributions. "
 686                             "Please specify version restrictions next to each package in 'name' argument."
 687                     )
 688                 if packages[0].has_version_specifier:
 689                     module.fail_json(
 690                         msg="The 'version' argument conflicts with any version specifier provided along with a package name. "
 691                             "Please keep the version specifier, but remove the 'version' argument."
 692                     )
 693                 # if the version specifier is provided by version, append that into the package
 694                 packages[0] = Package(to_native(packages[0]), version)
 695 
 696         if module.params['editable']:
 697             args_list = []  # used if extra_args is not used at all
 698             if extra_args:
 699                 args_list = extra_args.split(' ')
 700             if '-e' not in args_list:
 701                 args_list.append('-e')
 702                 # Ok, we will reconstruct the option string
 703                 extra_args = ' '.join(args_list)
 704 
 705         if extra_args:
 706             cmd.extend(shlex.split(extra_args))
 707 
 708         if name:
 709             cmd.extend(to_native(p) for p in packages)
 710         elif requirements:
 711             cmd.extend(['-r', requirements])
 712         else:
 713             module.exit_json(
 714                 changed=False,
 715                 warnings=["No valid name or requirements file found."],
 716             )
 717 
 718         if module.check_mode:
 719             if extra_args or requirements or state == 'latest' or not name:
 720                 module.exit_json(changed=True)
 721 
 722             pkg_cmd, out_pip, err_pip = _get_packages(module, pip, chdir)
 723 
 724             out += out_pip
 725             err += err_pip
 726 
 727             changed = False
 728             if name:
 729                 pkg_list = [p for p in out.split('\n') if not p.startswith('You are using') and not p.startswith('You should consider') and p]
 730 
 731                 if pkg_cmd.endswith(' freeze') and ('pip' in name or 'setuptools' in name):
 732                     # Older versions of pip (pre-1.3) do not have pip list.
 733                     # pip freeze does not list setuptools or pip in its output
 734                     # So we need to get those via a specialcase
 735                     for pkg in ('setuptools', 'pip'):
 736                         if pkg in name:
 737                             formatted_dep = _get_package_info(module, pkg, env)
 738                             if formatted_dep is not None:
 739                                 pkg_list.append(formatted_dep)
 740                                 out += '%s\n' % formatted_dep
 741 
 742                 for package in packages:
 743                     is_present = _is_present(module, package, pkg_list, pkg_cmd)
 744                     if (state == 'present' and not is_present) or (state == 'absent' and is_present):
 745                         changed = True
 746                         break
 747             module.exit_json(changed=changed, cmd=pkg_cmd, stdout=out, stderr=err)
 748 
 749         out_freeze_before = None
 750         if requirements or has_vcs:
 751             _, out_freeze_before, _ = _get_packages(module, pip, chdir)
 752 
 753         rc, out_pip, err_pip = module.run_command(cmd, path_prefix=path_prefix, cwd=chdir)
 754         out += out_pip
 755         err += err_pip
 756         if rc == 1 and state == 'absent' and \
 757            ('not installed' in out_pip or 'not installed' in err_pip):
 758             pass  # rc is 1 when attempting to uninstall non-installed package
 759         elif rc != 0:
 760             _fail(module, cmd, out, err)
 761 
 762         if state == 'absent':
 763             changed = 'Successfully uninstalled' in out_pip
 764         else:
 765             if out_freeze_before is None:
 766                 changed = 'Successfully installed' in out_pip
 767             else:
 768                 _, out_freeze_after, _ = _get_packages(module, pip, chdir)
 769                 changed = out_freeze_before != out_freeze_after
 770 
 771         changed = changed or venv_created
 772 
 773         module.exit_json(changed=changed, cmd=cmd, name=name, version=version,
 774                          state=state, requirements=requirements, virtualenv=env,
 775                          stdout=out, stderr=err)
 776     finally:
 777         if old_umask is not None:
 778             os.umask(old_umask)
 779 
 780 
 781 if __name__ == '__main__':
 782     main()
