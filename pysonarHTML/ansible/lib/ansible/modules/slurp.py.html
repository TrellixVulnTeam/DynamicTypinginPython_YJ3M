  59 import base64
  60 import os
  61 
  62 from ansible.module_utils.basic import AnsibleModule
  63 
  64 
  65 def main():
  66     module = AnsibleModule(
  67         argument_spec=dict(
  68             src=dict(type='path', required=True, aliases=['path']),
  69         ),
  70         supports_check_mode=True,
  71     )
  72     source = module.params['src']
  73 
  74     if not os.path.exists(source):
  75         module.fail_json(msg="file not found: %s" % source)
  76     if not os.access(source, os.R_OK):
  77         module.fail_json(msg="file is not readable: %s" % source)
  78 
  79     with open(source, 'rb') as source_fh:
  80         source_content = source_fh.read()
  81     data = base64.b64encode(source_content)
  82 
  83     module.exit_json(content=data, source=source, encoding='base64')
  84 
  85 
  86 if __name__ == '__main__':
  87     main()
