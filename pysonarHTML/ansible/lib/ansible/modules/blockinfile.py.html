  13 ---
  14 module: blockinfile
  15 short_description: Insert/update/remove a text block surrounded by marker lines
  16 version_added: '2.0'
  17 description:
  18 - This module will insert/update/remove a block of multi-line text surrounded by customizable marker lines.
  19 author:
  20 - Yaegashi Takeshi (@yaegashi)
  21 options:
  22   path:
  23     description:
  24     - The file to modify.
  25     - Before Ansible 2.3 this option was only usable as I(dest), I(destfile) and I(name).
  26     type: path
  27     required: yes
  28     aliases: [ dest, destfile, name ]
  29   state:
  30     description:
  31     - Whether the block should be there or not.
  32     type: str
  33     choices: [ absent, present ]
  34     default: present
  35   marker:
  36     description:
  37     - The marker line template.
  38     - C({mark}) will be replaced with the values in C(marker_begin) (default="BEGIN") and C(marker_end) (default="END").
  39     - Using a custom marker without the C({mark}) variable may result in the block being repeatedly inserted on subsequent playbook runs.
  40     type: str
  41     default: '# {mark} ANSIBLE MANAGED BLOCK'
  42   block:
  43     description:
  44     - The text to insert inside the marker lines.
  45     - If it is missing or an empty string, the block will be removed as if C(state) were specified to C(absent).
  46     type: str
  47     default: ''
  48     aliases: [ content ]
  49   insertafter:
  50     description:
  51     - If specified and no begin/ending C(marker) lines are found, the block will be inserted after the last match of specified regular expression.
  52     - A special value is available; C(EOF) for inserting the block at the end of the file.
  53     - If specified regular expression has no matches, C(EOF) will be used instead.
  54     type: str
  55     choices: [ EOF, '*regex*' ]
  56     default: EOF
  57   insertbefore:
  58     description:
  59     - If specified and no begin/ending C(marker) lines are found, the block will be inserted before the last match of specified regular expression.
  60     - A special value is available; C(BOF) for inserting the block at the beginning of the file.
  61     - If specified regular expression has no matches, the block will be inserted at the end of the file.
  62     type: str
  63     choices: [ BOF, '*regex*' ]
  64   create:
  65     description:
  66     - Create a new file if it does not exist.
  67     type: bool
  68     default: no
  69   backup:
  70     description:
  71     - Create a backup file including the timestamp information so you can
  72       get the original file back if you somehow clobbered it incorrectly.
  73     type: bool
  74     default: no
  75   marker_begin:
  76     description:
  77     - This will be inserted at C({mark}) in the opening ansible block marker.
  78     type: str
  79     default: BEGIN
  80     version_added: '2.5'
  81   marker_end:
  82     required: false
  83     description:
  84     - This will be inserted at C({mark}) in the closing ansible block marker.
  85     type: str
  86     default: END
  87     version_added: '2.5'
  88 notes:
  89   - This module supports check mode.
  90   - When using 'with_*' loops be aware that if you do not set a unique mark the block will be overwritten on each iteration.
  91   - As of Ansible 2.3, the I(dest) option has been changed to I(path) as default, but I(dest) still works as well.
  92   - Option I(follow) has been removed in Ansible 2.5, because this module modifies the contents of the file so I(follow=no) doesn't make sense.
  93   - When more then one block should be handled in one file you must change the I(marker) per task.

 155 from ansible.module_utils.basic import AnsibleModule
 156 from ansible.module_utils._text import to_bytes

 159 def write_changes(module, contents, path):
 160 
 161     tmpfd, tmpfile = tempfile.mkstemp(dir=module.tmpdir)
 162     f = os.fdopen(tmpfd, 'wb')
 163     f.write(contents)
 164     f.close()
 165 
 166     validate = module.params.get('validate', None)
 167     valid = not validate
 168     if validate:
 169         if "%s" not in validate:
 170             module.fail_json(msg="validate must contain %%s: %s" % (validate))
 171         (rc, out, err) = module.run_command(validate % tmpfile)
 172         valid = rc == 0
 173         if rc != 0:
 174             module.fail_json(msg='failed to validate: '
 175                                  'rc:%s error:%s' % (rc, err))
 176     if valid:
 177         module.atomic_move(tmpfile, path, unsafe_writes=module.params['unsafe_writes'])

 180 def check_file_attrs(module, changed, message, diff):
 181 
 182     file_args = module.load_file_common_arguments(module.params)
 183     if module.set_file_attributes_if_different(file_args, False, diff=diff):
 184 
 185         if changed:
 186             message += " and "
 187         changed = True
 188         message += "ownership, perms or SE linux context changed"
 189 
 190     return message, changed

 193 def main():
 194     module = AnsibleModule(
 195         argument_spec=dict(
 196             path=dict(type='path', required=True, aliases=['dest', 'destfile', 'name']),
 197             state=dict(type='str', default='present', choices=['absent', 'present']),
 198             marker=dict(type='str', default='# {mark} ANSIBLE MANAGED BLOCK'),
 199             block=dict(type='str', default='', aliases=['content']),
 200             insertafter=dict(type='str'),
 201             insertbefore=dict(type='str'),
 202             create=dict(type='bool', default=False),
 203             backup=dict(type='bool', default=False),
 204             validate=dict(type='str'),
 205             marker_begin=dict(type='str', default='BEGIN'),
 206             marker_end=dict(type='str', default='END'),
 207         ),
 208         mutually_exclusive=[['insertbefore', 'insertafter']],
 209         add_file_common_args=True,
 210         supports_check_mode=True
 211     )
 212 
 213     params = module.params
 214     path = params['path']
 215 
 216     if os.path.isdir(path):
 217         module.fail_json(rc=256,
 218                          msg='Path %s is a directory !' % path)
 219 
 220     path_exists = os.path.exists(path)
 221     if not path_exists:
 222         if not module.boolean(params['create']):
 223             module.fail_json(rc=257,
 224                              msg='Path %s does not exist !' % path)
 225         destpath = os.path.dirname(path)
 226         if not os.path.exists(destpath) and not module.check_mode:
 227             try:
 228                 os.makedirs(destpath)
 229             except Exception as e:
 230                 module.fail_json(msg='Error creating %s Error code: %s Error description: %s' % (destpath, e[0], e[1]))
 231         original = None
 232         lines = []
 233     else:
 234         f = open(path, 'rb')
 235         original = f.read()
 236         f.close()
 237         lines = original.splitlines(True)
 238 
 239     diff = {'before': '',
 240             'after': '',
 241             'before_header': '%s (content)' % path,
 242             'after_header': '%s (content)' % path}
 243 
 244     if module._diff and original:
 245         diff['before'] = original
 246 
 247     insertbefore = params['insertbefore']
 248     insertafter = params['insertafter']
 249     block = to_bytes(params['block'])
 250     marker = to_bytes(params['marker'])
 251     present = params['state'] == 'present'
 252 
 253     if not present and not path_exists:
 254         module.exit_json(changed=False, msg="File %s not present" % path)
 255 
 256     if insertbefore is None and insertafter is None:
 257         insertafter = 'EOF'
 258 
 259     if insertafter not in (None, 'EOF'):
 260         insertre = re.compile(to_bytes(insertafter, errors='surrogate_or_strict'))
 261     elif insertbefore not in (None, 'BOF'):
 262         insertre = re.compile(to_bytes(insertbefore, errors='surrogate_or_strict'))
 263     else:
 264         insertre = None
 265 
 266     marker0 = re.sub(b(r'{mark}'), b(params['marker_begin']), marker) + b(os.linesep)
 267     marker1 = re.sub(b(r'{mark}'), b(params['marker_end']), marker) + b(os.linesep)
 268     if present and block:
 269         # Escape sequences like '\n' need to be handled in Ansible 1.x
 270         if module.ansible_version.startswith('1.'):
 271             block = re.sub('', block, '')
 272         if not block.endswith(b(os.linesep)):
 273             block += b(os.linesep)
 274         blocklines = [marker0] + block.splitlines(True) + [marker1]
 275     else:
 276         blocklines = []
 277 
 278     n0 = n1 = None
 279     for i, line in enumerate(lines):
 280         if line == marker0:
 281             n0 = i
 282         if line == marker1:
 283             n1 = i
 284 
 285     if None in (n0, n1):
 286         n0 = None
 287         if insertre is not None:
 288             for i, line in enumerate(lines):
 289                 if insertre.search(line):
 290                     n0 = i
 291             if n0 is None:
 292                 n0 = len(lines)
 293             elif insertafter is not None:
 294                 n0 += 1
 295         elif insertbefore is not None:
 296             n0 = 0  # insertbefore=BOF
 297         else:
 298             n0 = len(lines)  # insertafter=EOF
 299     elif n0 < n1:
 300         lines[n0:n1 + 1] = []
 301     else:
 302         lines[n1:n0 + 1] = []
 303         n0 = n1
 304 
 305     lines[n0:n0] = blocklines
 306 
 307     if lines:
 308         result = b''.join(lines)
 309     else:
 310         result = b''
 311 
 312     if module._diff:
 313         diff['after'] = result
 314 
 315     if original == result:
 316         msg = ''
 317         changed = False
 318     elif original is None:
 319         msg = 'File created'
 320         changed = True
 321     elif not blocklines:
 322         msg = 'Block removed'
 323         changed = True
 324     else:
 325         msg = 'Block inserted'
 326         changed = True
 327 
 328     if changed and not module.check_mode:
 329         if module.boolean(params['backup']) and path_exists:
 330             module.backup_local(path)
 331         # We should always follow symlinks so that we change the real file
 332         real_path = os.path.realpath(params['path'])
 333         write_changes(module, result, real_path)
 334 
 335     if module.check_mode and not path_exists:
 336         module.exit_json(changed=changed, msg=msg, diff=diff)
 337 
 338     attr_diff = {}
 339     msg, changed = check_file_attrs(module, changed, msg, attr_diff)
 340 
 341     attr_diff['before_header'] = '%s (file attributes)' % path
 342     attr_diff['after_header'] = '%s (file attributes)' % path
 343 
 344     difflist = [diff, attr_diff]
 345     module.exit_json(changed=changed, msg=msg, diff=difflist)
 346 
 347 
 348 if __name__ == '__main__':
 349     main()
