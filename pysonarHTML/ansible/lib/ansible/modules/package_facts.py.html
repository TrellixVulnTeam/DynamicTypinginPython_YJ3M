 215 class RPM(LibMgr):
 216 
 217     LIB = 'rpm'
 218 
 219     def list_installed(self):
 220         return self._lib.TransactionSet().dbMatch()
 221 
 222     def get_package_details(self, package):
 223         return dict(name=package[self._lib.RPMTAG_NAME],
 224                     version=package[self._lib.RPMTAG_VERSION],
 225                     release=package[self._lib.RPMTAG_RELEASE],
 226                     epoch=package[self._lib.RPMTAG_EPOCH],
 227                     arch=package[self._lib.RPMTAG_ARCH],)
 228 
 229     def is_available(self):
 230         ''' we expect the python bindings installed, but this gives warning if they are missing and we have rpm cli'''
 231         we_have_lib = super(RPM, self).is_available()
 232 
 233         try:
 234             get_bin_path('rpm')
 235             if not we_have_lib:
 236                 module.warn('Found "rpm" but %s' % (missing_required_lib('rpm')))
 237         except ValueError:
 238             pass
 239 
 240         return we_have_lib

 243 class APT(LibMgr):
 244 
 245     LIB = 'apt'
 246 
 247     def __init__(self):
 248         self._cache = None
 249         super(APT, self).__init__()
 250 
 251     @property
 252     def pkg_cache(self):
 253         if self._cache is not None:
 254             return self._cache
 255 
 256         self._cache = self._lib.Cache()
 257         return self._cache
 258 
 259     def is_available(self):
 260         ''' we expect the python bindings installed, but if there is apt/apt-get give warning about missing bindings'''
 261         we_have_lib = super(APT, self).is_available()
 262         if not we_have_lib:
 263             for exe in ('apt', 'apt-get', 'aptitude'):
 264                 try:
 265                     get_bin_path(exe)
 266                 except ValueError:
 267                     continue
 268                 else:
 269                     module.warn('Found "%s" but %s' % (exe, missing_required_lib('apt')))
 270                     break
 271         return we_have_lib
 272 
 273     def list_installed(self):
 274         # Store the cache to avoid running pkg_cache() for each item in the comprehension, which is very slow
 275         cache = self.pkg_cache
 276         return [pk for pk in cache.keys() if cache[pk].is_installed]
 277 
 278     def get_package_details(self, package):
 279         ac_pkg = self.pkg_cache[package].installed
 280         return dict(name=package, version=ac_pkg.version, arch=ac_pkg.architecture, category=ac_pkg.section, origin=ac_pkg.origins[0].origin)

 283 class PACMAN(CLIMgr):
 284 
 285     CLI = 'pacman'
 286 
 287     def list_installed(self):
 288         rc, out, err = module.run_command([self._cli, '-Qi'], environ_update=dict(LC_ALL='C'))
 289         if rc != 0 or err:
 290             raise Exception("Unable to list packages rc=%s : %s" % (rc, err))
 291         return out.split("\n\n")[:-1]
 292 
 293     def get_package_details(self, package):
 294         # parse values of details that might extend over several lines
 295         raw_pkg_details = {}
 296         last_detail = None
 297         for line in package.splitlines():
 298             m = re.match(r"([\w ]*[\w]) +: (.*)", line)
 299             if m:
 300                 last_detail = m.group(1)
 301                 raw_pkg_details[last_detail] = m.group(2)
 302             else:
 303                 # append value to previous detail
 304                 raw_pkg_details[last_detail] = raw_pkg_details[last_detail] + "  " + line.lstrip()
 305 
 306         provides = None
 307         if raw_pkg_details['Provides'] != 'None':
 308             provides = [
 309                 p.split('=')[0]
 310                 for p in raw_pkg_details['Provides'].split('  ')
 311             ]
 312 
 313         return {
 314             'name': raw_pkg_details['Name'],
 315             'version': raw_pkg_details['Version'],
 316             'arch': raw_pkg_details['Architecture'],
 317             'provides': provides,
 318         }

 321 class PKG(CLIMgr):
 322 
 323     CLI = 'pkg'
 324     atoms = ['name', 'version', 'origin', 'installed', 'automatic', 'arch', 'category', 'prefix', 'vital']
 325 
 326     def list_installed(self):
 327         rc, out, err = module.run_command([self._cli, 'query', "%%%s" % '\t%'.join(['n', 'v', 'R', 't', 'a', 'q', 'o', 'p', 'V'])])
 328         if rc != 0 or err:
 329             raise Exception("Unable to list packages rc=%s : %s" % (rc, err))
 330         return out.splitlines()
 331 
 332     def get_package_details(self, package):
 333 
 334         pkg = dict(zip(self.atoms, package.split('\t')))
 335 
 336         if 'arch' in pkg:
 337             try:
 338                 pkg['arch'] = pkg['arch'].split(':')[2]
 339             except IndexError:
 340                 pass
 341 
 342         if 'automatic' in pkg:
 343             pkg['automatic'] = bool(int(pkg['automatic']))
 344 
 345         if 'category' in pkg:
 346             pkg['category'] = pkg['category'].split('/', 1)[0]
 347 
 348         if 'version' in pkg:
 349             if ',' in pkg['version']:
 350                 pkg['version'], pkg['port_epoch'] = pkg['version'].split(',', 1)
 351             else:
 352                 pkg['port_epoch'] = 0
 353 
 354             if '_' in pkg['version']:
 355                 pkg['version'], pkg['revision'] = pkg['version'].split('_', 1)
 356             else:
 357                 pkg['revision'] = '0'
 358 
 359         if 'vital' in pkg:
 360             pkg['vital'] = bool(int(pkg['vital']))
 361 
 362         return pkg

 365 class PORTAGE(CLIMgr):
 366 
 367     CLI = 'qlist'
 368     atoms = ['category', 'name', 'version', 'ebuild_revision', 'slots', 'prefixes', 'sufixes']
 369 
 370     def list_installed(self):
 371         rc, out, err = module.run_command(' '.join([self._cli, '-Iv', '|', 'xargs', '-n', '1024', 'qatom']), use_unsafe_shell=True)
 372         if rc != 0:
 373             raise RuntimeError("Unable to list packages rc=%s : %s" % (rc, to_native(err)))
 374         return out.splitlines()
 375 
 376     def get_package_details(self, package):
 377         return dict(zip(self.atoms, package.split()))

 380 class APK(CLIMgr):
 381 
 382     CLI = 'apk'
 383     atoms = ['name', 'version']
 384 
 385     def list_installed(self):
 386         rc, out, err = module.run_command([self._cli, 'info', '-v'])
 387         if rc != 0 or err:
 388             raise Exception("Unable to list packages rc=%s : %s" % (rc, err))
 389         return out.splitlines()
 390 
 391     def get_package_details(self, package):
 392         raw_pkg_details = {}
 393         for line in package.splitlines():
 394             m = re.match(r"([\w ].*?)-([0-9-\.]+[0-9a-z-\.]*-r[0-9]+)", to_native(line))
 395             if m:
 396                 raw_pkg_details['name'] = m.group(1)
 397                 raw_pkg_details['version'] = m.group(2)
 398 
 399         return raw_pkg_details

 402 def main():
 403 
 404     # get supported pkg managers
 405     PKG_MANAGERS = get_all_pkg_managers()
 406     PKG_MANAGER_NAMES = [x.lower() for x in PKG_MANAGERS.keys()]
 407 
 408     # start work
 409     global module
 410     module = AnsibleModule(argument_spec=dict(manager={'type': 'list', 'default': ['auto']},
 411                                               strategy={'choices': ['first', 'all'], 'default': 'first'}),
 412                            supports_check_mode=True)
 413     packages = {}
 414     results = {'ansible_facts': {}}
 415     managers = [x.lower() for x in module.params['manager']]
 416     strategy = module.params['strategy']
 417 
 418     if 'auto' in managers:
 419         # keep order from user, we do dedupe below
 420         managers.extend(PKG_MANAGER_NAMES)
 421         managers.remove('auto')
 422 
 423     unsupported = set(managers).difference(PKG_MANAGER_NAMES)
 424     if unsupported:
 425         if 'auto' in module.params['manager']:
 426             msg = 'Could not auto detect a usable package manager, check warnings for details.'
 427         else:
 428             msg = 'Unsupported package managers requested: %s' % (', '.join(unsupported))
 429         module.fail_json(msg=msg)
 430 
 431     found = 0
 432     seen = set()
 433     for pkgmgr in managers:
 434 
 435         if found and strategy == 'first':
 436             break
 437 
 438         # dedupe as per above
 439         if pkgmgr in seen:
 440             continue
 441         seen.add(pkgmgr)
 442         try:
 443             try:
 444                 # manager throws exception on init (calls self.test) if not usable.
 445                 manager = PKG_MANAGERS[pkgmgr]()
 446                 if manager.is_available():
 447                     found += 1
 448                     packages.update(manager.get_packages())
 449 
 450             except Exception as e:
 451                 if pkgmgr in module.params['manager']:
 452                     module.warn('Requested package manager %s was not usable by this module: %s' % (pkgmgr, to_text(e)))
 453                 continue
 454 
 455         except Exception as e:
 456             if pkgmgr in module.params['manager']:
 457                 module.warn('Failed to retrieve packages with %s: %s' % (pkgmgr, to_text(e)))
 458 
 459     if found == 0:
 460         msg = ('Could not detect a supported package manager from the following list: %s, '
 461                'or the required Python library is not installed. Check warnings for details.' % managers)
 462         module.fail_json(msg=msg)
 463 
 464     # Set the facts, this will override the facts in ansible_facts that might exist from previous runs
 465     # when using operating system level or distribution package managers
 466     results['ansible_facts']['packages'] = packages
 467 
 468     module.exit_json(**results)
 469 
 470 
 471 if __name__ == '__main__':
 472     main()
