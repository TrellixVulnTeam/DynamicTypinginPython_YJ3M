  40 from ansible.module_utils.basic import AnsibleModule
  41 
  42 
  43 def main():
  44     module = AnsibleModule(
  45         argument_spec=dict(
  46             name=dict(required=True),
  47             selection=dict(choices=['install', 'hold', 'deinstall', 'purge'])
  48         ),
  49         supports_check_mode=True,
  50     )
  51 
  52     dpkg = module.get_bin_path('dpkg', True)
  53 
  54     name = module.params['name']
  55     selection = module.params['selection']
  56 
  57     # Get current settings.
  58     rc, out, err = module.run_command([dpkg, '--get-selections', name], check_rc=True)
  59     if not out:
  60         current = 'not present'
  61     else:
  62         current = out.split()[1]
  63 
  64     changed = current != selection
  65 
  66     if module.check_mode or not changed:
  67         module.exit_json(changed=changed, before=current, after=selection)
  68 
  69     module.run_command([dpkg, '--set-selections'], data="%s %s" % (name, selection), check_rc=True)
  70     module.exit_json(changed=changed, before=current, after=selection)
  71 
  72 
  73 if __name__ == '__main__':
  74     main()
