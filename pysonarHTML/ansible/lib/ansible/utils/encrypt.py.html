  40 def random_password(length=DEFAULT_PASSWORD_LENGTH, chars=C.DEFAULT_PASSWORD_CHARS):
  41     '''Return a random password string of length containing only chars
  42 
  43     :kwarg length: The number of characters in the new password.  Defaults to 20.
  44     :kwarg chars: The characters to choose from.  The default is all ascii
  45         letters, ascii digits, and these symbols ``.,:-_``
  46     '''
  47     if not isinstance(chars, text_type):
  48         raise AnsibleAssertionError('%s (%s) is not a text_type' % (chars, type(chars)))
  49 
  50     random_generator = random.SystemRandom()
  51     return u''.join(random_generator.choice(chars) for dummy in range(length))

  54 def random_salt(length=8):
  55     """Return a text string suitable for use as a salt for the hash functions we use to encrypt passwords.
  56     """
  57     # Note passlib salt values must be pure ascii so we can't let the user
  58     # configure this
  59     salt_chars = string.ascii_letters + string.digits + u'./'
  60     return random_password(length=length, chars=salt_chars)

  63 class BaseHash(object):
  64     algo = namedtuple('algo', ['crypt_id', 'salt_size', 'implicit_rounds'])
  65     algorithms = {
  66         'md5_crypt': algo(crypt_id='1', salt_size=8, implicit_rounds=None),
  67         'bcrypt': algo(crypt_id='2a', salt_size=22, implicit_rounds=None),
  68         'sha256_crypt': algo(crypt_id='5', salt_size=16, implicit_rounds=5000),
  69         'sha512_crypt': algo(crypt_id='6', salt_size=16, implicit_rounds=5000),
  70     }
  71 
  72     def __init__(self, algorithm):
  73         self.algorithm = algorithm
  74 
  75 
  76 class CryptHash(BaseHash):
  77     def __init__(self, algorithm):
  78         super(CryptHash, self).__init__(algorithm)
  79 
  80         if sys.platform.startswith('darwin'):
  81             raise AnsibleError("crypt.crypt not supported on Mac OS X/Darwin, install passlib python module")
  82 
  83         if algorithm not in self.algorithms:
  84             raise AnsibleError("crypt.crypt does not support '%s' algorithm" % self.algorithm)
  85         self.algo_data = self.algorithms[algorithm]
  86 
  87     def hash(self, secret, salt=None, salt_size=None, rounds=None):
  88         salt = self._salt(salt, salt_size)
  89         rounds = self._rounds(rounds)
  90         return self._hash(secret, salt, rounds)
  91 
  92     def _salt(self, salt, salt_size):
  93         salt_size = salt_size or self.algo_data.salt_size
  94         return salt or random_salt(salt_size)
  95 
  96     def _rounds(self, rounds):
  97         if rounds == self.algo_data.implicit_rounds:
  98             # Passlib does not include the rounds if it is the same as implicit_rounds.
  99             # Make crypt lib behave the same, by not explicitly specifying the rounds in that case.
 100             return None
 101         else:
 102             return rounds
 103 
 104     def _hash(self, secret, salt, rounds):
 105         if rounds is None:
 106             saltstring = "$%s$%s" % (self.algo_data.crypt_id, salt)
 107         else:
 108             saltstring = "$%s$rounds=%d$%s" % (self.algo_data.crypt_id, rounds, salt)
 109 
 110         # crypt.crypt on Python < 3.9 returns None if it cannot parse saltstring
 111         # On Python >= 3.9, it throws OSError.
 112         try:
 113             result = crypt.crypt(secret, saltstring)
 114             orig_exc = None
 115         except OSError as e:
 116             result = None
 117             orig_exc = e
 118 
 119         # None as result would be interpreted by the some modules (user module)
 120         # as no password at all.
 121         if not result:
 122             raise AnsibleError(
 123                 "crypt.crypt does not support '%s' algorithm" % self.algorithm,
 124                 orig_exc=orig_exc,
 125             )
 126 
 127         return result

 130 class PasslibHash(BaseHash):
 131     def __init__(self, algorithm):
 132         super(PasslibHash, self).__init__(algorithm)
 133 
 134         if not PASSLIB_AVAILABLE:
 135             raise AnsibleError("passlib must be installed to hash with '%s'" % algorithm)
 136 
 137         try:
 138             self.crypt_algo = getattr(passlib.hash, algorithm)
 139         except Exception:
 140             raise AnsibleError("passlib does not support '%s' algorithm" % algorithm)
 141 
 142     def hash(self, secret, salt=None, salt_size=None, rounds=None):
 143         salt = self._clean_salt(salt)
 144         rounds = self._clean_rounds(rounds)
 145         return self._hash(secret, salt=salt, salt_size=salt_size, rounds=rounds)
 146 
 147     def _clean_salt(self, salt):
 148         if not salt:
 149             return None
 150         elif issubclass(self.crypt_algo, HasRawSalt):
 151             return to_bytes(salt, encoding='ascii', errors='strict')
 152         else:
 153             return to_text(salt, encoding='ascii', errors='strict')
 154 
 155     def _clean_rounds(self, rounds):
 156         algo_data = self.algorithms.get(self.algorithm)
 157         if rounds:
 158             return rounds
 159         elif algo_data and algo_data.implicit_rounds:
 160             # The default rounds used by passlib depend on the passlib version.
 161             # For consistency ensure that passlib behaves the same as crypt in case no rounds were specified.
 162             # Thus use the crypt defaults.
 163             return algo_data.implicit_rounds
 164         else:
 165             return None
 166 
 167     def _hash(self, secret, salt, salt_size, rounds):
 168         # Not every hash algorithm supports every parameter.
 169         # Thus create the settings dict only with set parameters.
 170         settings = {}
 171         if salt:
 172             settings['salt'] = salt
 173         if salt_size:
 174             settings['salt_size'] = salt_size
 175         if rounds:
 176             settings['rounds'] = rounds
 177 
 178         # starting with passlib 1.7 'using' and 'hash' should be used instead of 'encrypt'
 179         if hasattr(self.crypt_algo, 'hash'):
 180             result = self.crypt_algo.using(**settings).hash(secret)
 181         elif hasattr(self.crypt_algo, 'encrypt'):
 182             result = self.crypt_algo.encrypt(secret, **settings)
 183         else:
 184             raise AnsibleError("installed passlib version %s not supported" % passlib.__version__)
 185 
 186         # passlib.hash should always return something or raise an exception.
 187         # Still ensure that there is always a result.
 188         # Otherwise an empty password might be assumed by some modules, like the user module.
 189         if not result:
 190             raise AnsibleError("failed to hash with algorithm '%s'" % self.algorithm)
 191 
 192         # Hashes from passlib.hash should be represented as ascii strings of hex
 193         # digits so this should not traceback.  If it's not representable as such
 194         # we need to traceback and then blacklist such algorithms because it may
 195         # impact calling code.
 196         return to_text(result, errors='strict')
 197

 199 def passlib_or_crypt(secret, algorithm, salt=None, salt_size=None, rounds=None):
 200     if PASSLIB_AVAILABLE:
 201         return PasslibHash(algorithm).hash(secret, salt=salt, salt_size=salt_size, rounds=rounds)
 202     else:
 203         return CryptHash(algorithm).hash(secret, salt=salt, salt_size=salt_size, rounds=rounds)

 206 def do_encrypt(result, encrypt, salt_size=None, salt=None):
 207     return passlib_or_crypt(result, encrypt, salt_size=salt_size, salt=salt)
