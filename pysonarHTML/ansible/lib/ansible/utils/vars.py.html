  58 def _validate_mutable_mappings(a, b):
  59     """
  60     Internal convenience function to ensure arguments are MutableMappings
  61 
  62     This checks that all arguments are MutableMappings or raises an error
  63 
  64     :raises AnsibleError: if one of the arguments is not a MutableMapping
  65     """
  66 
  67     # If this becomes generally needed, change the signature to operate on
  68     # a variable number of arguments instead.
  69 
  70     if not (isinstance(a, MutableMapping) and isinstance(b, MutableMapping)):
  71         myvars = []
  72         for x in [a, b]:
  73             try:
  74                 myvars.append(dumps(x))
  75             except Exception:
  76                 myvars.append(to_native(x))
  77         raise AnsibleError("failed to combine variables, expected dicts but got a '{0}' and a '{1}': \n{2}\n{3}".format(
  78             a.__class__.__name__, b.__class__.__name__, myvars[0], myvars[1])
  79         )

  82 def combine_vars(a, b):
  83     """
  84     Return a copy of dictionaries of variables based on configured hash behavior
  85     """
  86 
  87     if C.DEFAULT_HASH_BEHAVIOUR == "merge":
  88         return merge_hash(a, b)
  89     else:
  90         # HASH_BEHAVIOUR == 'replace'
  91         _validate_mutable_mappings(a, b)
  92         result = a.copy()
  93         result.update(b)
  94         return result

  97 def merge_hash(x, y, recursive=True, list_merge='replace'):
  98     """
  99     Return a new dictionary result of the merges of y into x,
 100     so that keys from y take precedence over keys from x.
 101     (x and y aren't modified)
 102     """
 103     if list_merge not in ('replace', 'keep', 'append', 'prepend', 'append_rp', 'prepend_rp'):
 104         raise AnsibleError("merge_hash: 'list_merge' argument can only be equal to 'replace', 'keep', 'append', 'prepend', 'append_rp' or 'prepend_rp'")
 105 
 106     # verify x & y are dicts
 107     _validate_mutable_mappings(x, y)
 108 
 109     # to speed things up: if x is empty or equal to y, return y
 110     # (this `if` can be remove without impact on the function
 111     #  except performance)
 112     if x == {} or x == y:
 113         return y.copy()
 114 
 115     # in the following we will copy elements from y to x, but
 116     # we don't want to modify x, so we create a copy of it
 117     x = x.copy()
 118 
 119     # to speed things up: use dict.update if possible
 120     # (this `if` can be remove without impact on the function
 121     #  except performance)
 122     if not recursive and list_merge == 'replace':
 123         x.update(y)
 124         return x
 125 
 126     # insert each element of y in x, overriding the one in x
 127     # (as y has higher priority)
 128     # we copy elements from y to x instead of x to y because
 129     # there is a high probability x will be the "default" dict the user
 130     # want to "patch" with y
 131     # therefore x will have much more elements than y
 132     for key, y_value in iteritems(y):
 133         # if `key` isn't in x
 134         # update x and move on to the next element of y
 135         if key not in x:
 136             x[key] = y_value
 137             continue
 138         # from this point we know `key` is in x
 139 
 140         x_value = x[key]
 141 
 142         # if both x's element and y's element are dicts
 143         # recursively "combine" them or override x's with y's element
 144         # depending on the `recursive` argument
 145         # and move on to the next element of y
 146         if isinstance(x_value, MutableMapping) and isinstance(y_value, MutableMapping):
 147             if recursive:
 148                 x[key] = merge_hash(x_value, y_value, recursive, list_merge)
 149             else:
 150                 x[key] = y_value
 151             continue
 152 
 153         # if both x's element and y's element are lists
 154         # "merge" them depending on the `list_merge` argument
 155         # and move on to the next element of y
 156         if isinstance(x_value, MutableSequence) and isinstance(y_value, MutableSequence):
 157             if list_merge == 'replace':
 158                 # replace x value by y's one as it has higher priority
 159                 x[key] = y_value
 160             elif list_merge == 'append':
 161                 x[key] = x_value + y_value
 162             elif list_merge == 'prepend':
 163                 x[key] = y_value + x_value
 164             elif list_merge == 'append_rp':
 165                 # append all elements from y_value (high prio) to x_value (low prio)
 166                 # and remove x_value elements that are also in y_value
 167                 # we don't remove elements from x_value nor y_value that were already in double
 168                 # (we assume that there is a reason if there where such double elements)
 169                 # _rp stands for "remove present"
 170                 x[key] = [z for z in x_value if z not in y_value] + y_value
 171             elif list_merge == 'prepend_rp':
 172                 # same as 'append_rp' but y_value elements are prepend
 173                 x[key] = y_value + [z for z in x_value if z not in y_value]
 174             # else 'keep'
 175             #   keep x value even if y it's of higher priority
 176             #   it's done by not changing x[key]
 177             continue
 178 
 179         # else just override x's element with y's one
 180         x[key] = y_value
 181 
 182     return x

 185 def load_extra_vars(loader):
 186     extra_vars = {}
 187     for extra_vars_opt in context.CLIARGS.get('extra_vars', tuple()):
 188         data = None
 189         extra_vars_opt = to_text(extra_vars_opt, errors='surrogate_or_strict')
 190         if extra_vars_opt is None or not extra_vars_opt:
 191             continue
 192 
 193         if extra_vars_opt.startswith(u"@"):
 194             # Argument is a YAML file (JSON is a subset of YAML)
 195             data = loader.load_from_file(extra_vars_opt[1:])
 196         elif extra_vars_opt[0] in [u'/', u'.']:
 197             raise AnsibleOptionsError("Please prepend extra_vars filename '%s' with '@'" % extra_vars_opt)
 198         elif extra_vars_opt[0] in [u'[', u'{']:
 199             # Arguments as YAML
 200             data = loader.load(extra_vars_opt)
 201         else:
 202             # Arguments as Key-value
 203             data = parse_kv(extra_vars_opt)
 204 
 205         if isinstance(data, MutableMapping):
 206             extra_vars = combine_vars(extra_vars, data)
 207         else:
 208             raise AnsibleOptionsError("Invalid extra vars data supplied. '%s' could not be made into a dictionary" % extra_vars_opt)
 209 
 210     return extra_vars

 213 def load_options_vars(version):
 214 
 215     if version is None:
 216         version = 'Unknown'
 217     options_vars = {'ansible_version': version}
 218     attrs = {'check': 'check_mode',
 219              'diff': 'diff_mode',
 220              'forks': 'forks',
 221              'inventory': 'inventory_sources',
 222              'skip_tags': 'skip_tags',
 223              'subset': 'limit',
 224              'tags': 'run_tags',
 225              'verbosity': 'verbosity'}
 226 
 227     for attr, alias in attrs.items():
 228         opt = context.CLIARGS.get(attr)
 229         if opt is not None:
 230             options_vars['ansible_%s' % alias] = opt
 231 
 232     return options_vars

 235 def _isidentifier_PY3(ident):
 236     if not isinstance(ident, string_types):
 237         return False
 238 
 239     # NOTE Python 3.7 offers str.isascii() so switch over to using it once
 240     # we stop supporting 3.5 and 3.6 on the controller
 241     try:
 242         # Python 2 does not allow non-ascii characters in identifiers so unify
 243         # the behavior for Python 3
 244         ident.encode('ascii')
 245     except UnicodeEncodeError:
 246         return False
 247 
 248     if not ident.isidentifier():
 249         return False
 250 
 251     if keyword.iskeyword(ident):
 252         return False
 253 
 254     return True
 255 
 256 
 257 def _isidentifier_PY2(ident):
 258     if not isinstance(ident, string_types):
 259         return False
 260 
 261     if not ident:
 262         return False
 263 
 264     if C.INVALID_VARIABLE_NAMES.search(ident):
 265         return False
 266 
 267     if keyword.iskeyword(ident) or ident in ADDITIONAL_PY2_KEYWORDS:
 268         return False
 269 
 270     return True
 271

 273 if PY3:
 274     isidentifier = _isidentifier_PY3
 275 else:
 276     isidentifier = _isidentifier_PY2

 279 isidentifier.__doc__ = """Determine if string is valid identifier.
 280 
 281 The purpose of this function is to be used to validate any variables created in
 282 a play to be valid Python identifiers and to not conflict with Python keywords
 283 to prevent unexpected behavior. Since Python 2 and Python 3 differ in what
 284 a valid identifier is, this function unifies the validation so playbooks are
 285 portable between the two. The following changes were made:
 286 
 287     * disallow non-ascii characters (Python 3 allows for them as opposed to Python 2)
 288     * True, False and None are reserved keywords (these are reserved keywords
 289       on Python 3 as opposed to Python 2)
 290 
 291 :arg ident: A text string of indentifier to check. Note: It is callers
 292     responsibility to convert ident to text if it is not already.
 293 
 294 Originally posted at http://stackoverflow.com/a/29586366
 295 """
