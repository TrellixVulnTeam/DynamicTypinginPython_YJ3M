  25 ANSIBLE_COLOR = True
  26 if C.ANSIBLE_NOCOLOR:
  27     ANSIBLE_COLOR = False
  28 elif not hasattr(sys.stdout, 'isatty') or not sys.stdout.isatty():
  29     ANSIBLE_COLOR = False
  30 else:
  31     try:
  32         import curses
  33         curses.setupterm()
  34         if curses.tigetnum('colors') < 0:
  35             ANSIBLE_COLOR = False
  36     except ImportError:
  37         # curses library was not found
  38         pass
  39     except curses.error:
  40         # curses returns an error (e.g. could not find terminal)
  41         ANSIBLE_COLOR = False

  57 codeCodes = {
  58     'black': u'0;30', 'bright gray': u'0;37',
  59     'blue': u'0;34', 'white': u'1;37',
  60     'green': u'0;32', 'bright blue': u'1;34',
  61     'cyan': u'0;36', 'bright green': u'1;32',
  62     'red': u'0;31', 'bright cyan': u'1;36',
  63     'purple': u'0;35', 'bright red': u'1;31',
  64     'yellow': u'0;33', 'bright purple': u'1;35',
  65     'dark gray': u'1;30', 'bright yellow': u'1;33',
  66     'magenta': u'0;35', 'bright magenta': u'1;35',
  67     'normal': u'0',
  68 }

  71 def parsecolor(color):
  72     """SGR parameter string for the specified color name."""
  73     matches = re.match(r"color(?P<color>[0-9]+)"
  74                        r"|(?P<rgb>rgb(?P<red>[0-5])(?P<green>[0-5])(?P<blue>[0-5]))"
  75                        r"|gray(?P<gray>[0-9]+)", color)
  76     if not matches:
  77         return codeCodes[color]
  78     if matches.group('color'):
  79         return u'38;5;%d' % int(matches.group('color'))
  80     if matches.group('rgb'):
  81         return u'38;5;%d' % (16 + 36 * int(matches.group('red')) +
  82                              6 * int(matches.group('green')) +
  83                              int(matches.group('blue')))
  84     if matches.group('gray'):
  85         return u'38;5;%d' % (232 + int(matches.group('gray')))

  88 def stringc(text, color, wrap_nonvisible_chars=False):
  89     """String in color."""
  90 
  91     if ANSIBLE_COLOR:
  92         color_code = parsecolor(color)
  93         fmt = u"\033[%sm%s\033[0m"
  94         if wrap_nonvisible_chars:
  95             # This option is provided for use in cases when the
  96             # formatting of a command line prompt is needed, such as
  97             # `ansible-console`. As said in `readline` sources:
  98             # readline/display.c:321
  99             # /* Current implementation:
 100             #         \001 (^A) start non-visible characters
 101             #         \002 (^B) end non-visible characters
 102             #    all characters except \001 and \002 (following a \001) are copied to
 103             #    the returned string; all characters except those between \001 and
 104             #    \002 are assumed to be `visible'. */
 105             fmt = u"\001\033[%sm\002%s\001\033[0m\002"
 106         return u"\n".join([fmt % (color_code, t) for t in text.split(u'\n')])
 107     else:
 108         return text

 111 def colorize(lead, num, color):
 112     """ Print 'lead' = 'num' in 'color' """
 113     s = u"%s=%-4s" % (lead, str(num))
 114     if num != 0 and ANSIBLE_COLOR and color is not None:
 115         s = stringc(s, color)
 116     return s

 119 def hostcolor(host, stats, color=True):
 120     if ANSIBLE_COLOR and color:
 121         if stats['failures'] != 0 or stats['unreachable'] != 0:
 122             return u"%-37s" % stringc(host, C.COLOR_ERROR)
 123         elif stats['changed'] != 0:
 124             return u"%-37s" % stringc(host, C.COLOR_CHANGED)
 125         else:
 126             return u"%-37s" % stringc(host, C.COLOR_OK)
 127     return u"%-26s" % host
