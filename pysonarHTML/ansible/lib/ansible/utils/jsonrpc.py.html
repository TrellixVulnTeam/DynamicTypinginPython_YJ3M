   7 import json
   8 import traceback

  10 from ansible.module_utils._text import to_text
  11 from ansible.module_utils.connection import ConnectionError
  12 from ansible.module_utils.six import binary_type, text_type
  13 from ansible.module_utils.six.moves import cPickle
  14 from ansible.utils.display import Display
  15 
  16 display = Display()
  17 
  18 
  19 class JsonRpcServer(object):
  20 
  21     _objects = set()
  22 
  23     def handle_request(self, request):
  24         request = json.loads(to_text(request, errors='surrogate_then_replace'))
  25 
  26         method = request.get('method')
  27 
  28         if method.startswith('rpc.') or method.startswith('_'):
  29             error = self.invalid_request()
  30             return json.dumps(error)
  31 
  32         args, kwargs = request.get('params')
  33         setattr(self, '_identifier', request.get('id'))
  34 
  35         rpc_method = None
  36         for obj in self._objects:
  37             rpc_method = getattr(obj, method, None)
  38             if rpc_method:
  39                 break
  40 
  41         if not rpc_method:
  42             error = self.method_not_found()
  43             response = json.dumps(error)
  44         else:
  45             try:
  46                 result = rpc_method(*args, **kwargs)
  47             except ConnectionError as exc:
  48                 display.vvv(traceback.format_exc())
  49                 try:
  50                     error = self.error(code=exc.code, message=to_text(exc))
  51                 except AttributeError:
  52                     error = self.internal_error(data=to_text(exc))
  53                 response = json.dumps(error)
  54             except Exception as exc:
  55                 display.vvv(traceback.format_exc())
  56                 error = self.internal_error(data=to_text(exc, errors='surrogate_then_replace'))
  57                 response = json.dumps(error)
  58             else:
  59                 if isinstance(result, dict) and 'jsonrpc' in result:
  60                     response = result
  61                 else:
  62                     response = self.response(result)
  63 
  64                 try:
  65                     response = json.dumps(response)
  66                 except Exception as exc:
  67                     display.vvv(traceback.format_exc())
  68                     error = self.internal_error(data=to_text(exc, errors='surrogate_then_replace'))
  69                     response = json.dumps(error)
  70 
  71         delattr(self, '_identifier')
  72 
  73         return response
  74 
  75     def register(self, obj):
  76         self._objects.add(obj)
  77 
  78     def header(self):
  79         return {'jsonrpc': '2.0', 'id': self._identifier}
  80 
  81     def response(self, result=None):
  82         response = self.header()
  83         if isinstance(result, binary_type):
  84             result = to_text(result)
  85         if not isinstance(result, text_type):
  86             response["result_type"] = "pickle"
  87             result = to_text(cPickle.dumps(result, protocol=0))
  88         response['result'] = result
  89         return response
  90 
  91     def error(self, code, message, data=None):
  92         response = self.header()
  93         error = {'code': code, 'message': message}
  94         if data:
  95             error['data'] = data
  96         response['error'] = error
  97         return response
  98 
  99     # json-rpc standard errors (-32768 .. -32000)
 100     def parse_error(self, data=None):
 101         return self.error(-32700, 'Parse error', data)
 102 
 103     def method_not_found(self, data=None):
 104         return self.error(-32601, 'Method not found', data)
 105 
 106     def invalid_request(self, data=None):
 107         return self.error(-32600, 'Invalid request', data)
 108 
 109     def invalid_params(self, data=None):
 110         return self.error(-32602, 'Invalid params', data)
 111 
 112     def internal_error(self, data=None):
 113         return self.error(-32603, 'Internal error', data)
