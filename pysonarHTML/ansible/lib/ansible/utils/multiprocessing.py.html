  10 # Explicit multiprocessing context using the fork start method
  11 # This exists as a compat layer now that Python3.8 has changed the default
  12 # start method for macOS to ``spawn`` which is incompatible with our
  13 # code base currently
  14 #
  15 # This exists in utils to allow it to be easily imported into various places
  16 # without causing circular import or dependency problems
  17 try:
  18     context = multiprocessing.get_context('fork')
  19 except AttributeError:
  20     # Py2 has no context functionality, and only supports fork
  21     context = multiprocessing
