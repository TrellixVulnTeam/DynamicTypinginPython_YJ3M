  65 class AnsibleUnsafe(object):
  66     __UNSAFE__ = True
  67 
  68 
  69 class AnsibleUnsafeBytes(binary_type, AnsibleUnsafe):
  70     def decode(self, *args, **kwargs):
  71         """Wrapper method to ensure type conversions maintain unsafe context"""
  72         return AnsibleUnsafeText(super(AnsibleUnsafeBytes, self).decode(*args, **kwargs))
  73 
  74 
  75 class AnsibleUnsafeText(text_type, AnsibleUnsafe):
  76     def encode(self, *args, **kwargs):
  77         """Wrapper method to ensure type conversions maintain unsafe context"""
  78         return AnsibleUnsafeBytes(super(AnsibleUnsafeText, self).encode(*args, **kwargs))

  81 class UnsafeProxy(object):
  82     def __new__(cls, obj, *args, **kwargs):
  83         from ansible.utils.display import Display
  84         Display().deprecated(
  85             'UnsafeProxy is being deprecated. Use wrap_var or AnsibleUnsafeBytes/AnsibleUnsafeText directly instead',
  86             version='2.13', collection_name='ansible.builtin'
  87         )
  88         # In our usage we should only receive unicode strings.
  89         # This conditional and conversion exists to sanity check the values
  90         # we're given but we may want to take it out for testing and sanitize
  91         # our input instead.
  92         if isinstance(obj, AnsibleUnsafe):
  93             return obj
  94 
  95         if isinstance(obj, string_types):
  96             obj = AnsibleUnsafeText(to_text(obj, errors='surrogate_or_strict'))
  97         return obj

 100 def _wrap_dict(v):
 101     return dict((wrap_var(k), wrap_var(item)) for k, item in v.items())
 102 
 103 
 104 def _wrap_sequence(v):
 105     """Wraps a sequence with unsafe, not meant for strings, primarily
 106     ``tuple`` and ``list``
 107     """
 108     v_type = type(v)
 109     return v_type(wrap_var(item) for item in v)
 110 
 111 
 112 def _wrap_set(v):
 113     return set(wrap_var(item) for item in v)
 114 
 115 
 116 def wrap_var(v):
 117     if v is None or isinstance(v, AnsibleUnsafe):
 118         return v
 119 
 120     if isinstance(v, Mapping):
 121         v = _wrap_dict(v)
 122     elif isinstance(v, Set):
 123         v = _wrap_set(v)
 124     elif is_sequence(v):
 125         v = _wrap_sequence(v)
 126     elif isinstance(v, binary_type):
 127         v = AnsibleUnsafeBytes(v)
 128     elif isinstance(v, text_type):
 129         v = AnsibleUnsafeText(v)
 130 
 131     return v
 132 
 133 
 134 def to_unsafe_bytes(*args, **kwargs):
 135     return wrap_var(to_bytes(*args, **kwargs))
 136 
 137 
 138 def to_unsafe_text(*args, **kwargs):
 139     return wrap_var(to_text(*args, **kwargs))
