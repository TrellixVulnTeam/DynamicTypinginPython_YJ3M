  27 def merge_fragment(target, source):
  28 
  29     for key, value in source.items():
  30         if key in target:
  31             # assumes both structures have same type
  32             if isinstance(target[key], MutableMapping):
  33                 value.update(target[key])
  34             elif isinstance(target[key], MutableSet):
  35                 value.add(target[key])
  36             elif isinstance(target[key], MutableSequence):
  37                 value = sorted(frozenset(value + target[key]))
  38             else:
  39                 raise Exception("Attempt to extend a documentation fragement, invalid type for %s" % key)
  40         target[key] = value

  43 def _process_versions_and_dates(fragment, is_module, return_docs, callback):
  44     def process_deprecation(deprecation, top_level=False):
  45         if not isinstance(deprecation, MutableMapping):
  46             return
  47         if (is_module or top_level) and 'removed_in' in deprecation:  # used in module deprecations
  48             callback(deprecation, 'removed_in', 'removed_from_collection')
  49         if 'removed_at_date' in deprecation:
  50             callback(deprecation, 'removed_at_date', 'removed_from_collection')
  51         if not (is_module or top_level) and 'version' in deprecation:  # used in plugin option deprecations
  52             callback(deprecation, 'version', 'removed_from_collection')
  53 
  54     def process_option_specifiers(specifiers):
  55         for specifier in specifiers:
  56             if not isinstance(specifier, MutableMapping):
  57                 continue
  58             if 'version_added' in specifier:
  59                 callback(specifier, 'version_added', 'version_added_collection')
  60             if isinstance(specifier.get('deprecated'), MutableMapping):
  61                 process_deprecation(specifier['deprecated'])
  62 
  63     def process_options(options):
  64         for option in options.values():
  65             if not isinstance(option, MutableMapping):
  66                 continue
  67             if 'version_added' in option:
  68                 callback(option, 'version_added', 'version_added_collection')
  69             if not is_module:
  70                 if isinstance(option.get('env'), list):
  71                     process_option_specifiers(option['env'])
  72                 if isinstance(option.get('ini'), list):
  73                     process_option_specifiers(option['ini'])
  74                 if isinstance(option.get('vars'), list):
  75                     process_option_specifiers(option['vars'])
  76             if isinstance(option.get('suboptions'), MutableMapping):
  77                 process_options(option['suboptions'])
  78 
  79     def process_return_values(return_values):
  80         for return_value in return_values.values():
  81             if not isinstance(return_value, MutableMapping):
  82                 continue
  83             if 'version_added' in return_value:
  84                 callback(return_value, 'version_added', 'version_added_collection')
  85             if isinstance(return_value.get('contains'), MutableMapping):
  86                 process_return_values(return_value['contains'])
  87 
  88     if not fragment:
  89         return
  90 
  91     if return_docs:
  92         process_return_values(fragment)
  93         return
  94 
  95     if 'version_added' in fragment:
  96         callback(fragment, 'version_added', 'version_added_collection')
  97     if isinstance(fragment.get('deprecated'), MutableMapping):
  98         process_deprecation(fragment['deprecated'], top_level=True)
  99     if isinstance(fragment.get('options'), MutableMapping):
 100         process_options(fragment['options'])

 103 def add_collection_to_versions_and_dates(fragment, collection_name, is_module, return_docs=False):
 104     def add(options, option, collection_name_field):
 105         if collection_name_field not in options:
 106             options[collection_name_field] = collection_name
 107 
 108     _process_versions_and_dates(fragment, is_module, return_docs, add)

 111 def remove_current_collection_from_versions_and_dates(fragment, collection_name, is_module, return_docs=False):
 112     def remove(options, option, collection_name_field):
 113         if options.get(collection_name_field) == collection_name:
 114             del options[collection_name_field]
 115 
 116     _process_versions_and_dates(fragment, is_module, return_docs, remove)

 119 def add_fragments(doc, filename, fragment_loader, is_module=False):
 120 
 121     fragments = doc.pop('extends_documentation_fragment', [])
 122 
 123     if isinstance(fragments, string_types):
 124         fragments = [fragments]
 125 
 126     unknown_fragments = []
 127 
 128     # doc_fragments are allowed to specify a fragment var other than DOCUMENTATION
 129     # with a . separator; this is complicated by collections-hosted doc_fragments that
 130     # use the same separator. Assume it's collection-hosted normally first, try to load
 131     # as-specified. If failure, assume the right-most component is a var, split it off,
 132     # and retry the load.
 133     for fragment_slug in fragments:
 134         fragment_name = fragment_slug
 135         fragment_var = 'DOCUMENTATION'
 136 
 137         fragment_class = fragment_loader.get(fragment_name)
 138         if fragment_class is None and '.' in fragment_slug:
 139             splitname = fragment_slug.rsplit('.', 1)
 140             fragment_name = splitname[0]
 141             fragment_var = splitname[1].upper()
 142             fragment_class = fragment_loader.get(fragment_name)
 143 
 144         if fragment_class is None:
 145             unknown_fragments.append(fragment_slug)
 146             continue
 147 
 148         fragment_yaml = getattr(fragment_class, fragment_var, None)
 149         if fragment_yaml is None:
 150             if fragment_var != 'DOCUMENTATION':
 151                 # if it's asking for something specific that's missing, that's an error
 152                 unknown_fragments.append(fragment_slug)
 153                 continue
 154             else:
 155                 fragment_yaml = '{}'  # TODO: this is still an error later since we require 'options' below...
 156 
 157         fragment = AnsibleLoader(fragment_yaml, file_name=filename).get_single_data()
 158 
 159         real_collection_name = 'ansible.builtin'
 160         real_fragment_name = getattr(fragment_class, '_load_name')
 161         if real_fragment_name.startswith('ansible_collections.'):
 162             real_collection_name = '.'.join(real_fragment_name.split('.')[1:3])
 163         add_collection_to_versions_and_dates(fragment, real_collection_name, is_module=is_module)
 164 
 165         if 'notes' in fragment:
 166             notes = fragment.pop('notes')
 167             if notes:
 168                 if 'notes' not in doc:
 169                     doc['notes'] = []
 170                 doc['notes'].extend(notes)
 171 
 172         if 'seealso' in fragment:
 173             seealso = fragment.pop('seealso')
 174             if seealso:
 175                 if 'seealso' not in doc:
 176                     doc['seealso'] = []
 177                 doc['seealso'].extend(seealso)
 178 
 179         if 'options' not in fragment:
 180             raise Exception("missing options in fragment (%s), possibly misformatted?: %s" % (fragment_name, filename))
 181 
 182         # ensure options themselves are directly merged
 183         if 'options' in doc:
 184             try:
 185                 merge_fragment(doc['options'], fragment.pop('options'))
 186             except Exception as e:
 187                 raise AnsibleError("%s options (%s) of unknown type: %s" % (to_native(e), fragment_name, filename))
 188         else:
 189             doc['options'] = fragment.pop('options')
 190 
 191         # merge rest of the sections
 192         try:
 193             merge_fragment(doc, fragment)
 194         except Exception as e:
 195             raise AnsibleError("%s (%s) of unknown type: %s" % (to_native(e), fragment_name, filename))
 196 
 197     if unknown_fragments:
 198         raise AnsibleError('unknown doc_fragment(s) in file {0}: {1}'.format(filename, to_native(', '.join(unknown_fragments))))

 201 def get_docstring(filename, fragment_loader, verbose=False, ignore_errors=False, collection_name=None, is_module=False):
 202     """
 203     DOCUMENTATION can be extended using documentation fragments loaded by the PluginLoader from the doc_fragments plugins.
 204     """
 205 
 206     data = read_docstring(filename, verbose=verbose, ignore_errors=ignore_errors)
 207 
 208     if data.get('doc', False):
 209         # add collection name to versions and dates
 210         if collection_name is not None:
 211             add_collection_to_versions_and_dates(data['doc'], collection_name, is_module=is_module)
 212 
 213         # add fragments to documentation
 214         add_fragments(data['doc'], filename, fragment_loader=fragment_loader, is_module=is_module)
 215 
 216     if data.get('returndocs', False):
 217         # add collection name to versions and dates
 218         if collection_name is not None:
 219             add_collection_to_versions_and_dates(data['returndocs'], collection_name, is_module=is_module, return_docs=True)
 220 
 221     return data['doc'], data['plainexamples'], data['returndocs'], data['metadata']

 224 def get_versioned_doclink(path):
 225     """
 226     returns a versioned documentation link for the current Ansible major.minor version; used to generate
 227     in-product warning/error links to the configured DOCSITE_ROOT_URL
 228     (eg, https://docs.ansible.com/ansible/2.8/somepath/doc.html)
 229 
 230     :param path: relative path to a document under docs/docsite/rst;
 231     :return: absolute URL to the specified doc for the current version of Ansible
 232     """
 233     path = to_native(path)
 234     try:
 235         base_url = C.config.get_config_value('DOCSITE_ROOT_URL')
 236         if not base_url.endswith('/'):
 237             base_url += '/'
 238         if path.startswith('/'):
 239             path = path[1:]
 240         split_ver = ansible_version.split('.')
 241         if len(split_ver) < 3:
 242             raise RuntimeError('invalid version ({0})'.format(ansible_version))
 243 
 244         doc_version = '{0}.{1}'.format(split_ver[0], split_ver[1])
 245 
 246         # check to see if it's a X.Y.0 non-rc prerelease or dev release, if so, assume devel (since the X.Y doctree
 247         # isn't published until beta-ish)
 248         if split_ver[2].startswith('0'):
 249             # exclude rc; we should have the X.Y doctree live by rc1
 250             if any((pre in split_ver[2]) for pre in ['a', 'b']) or len(split_ver) > 3 and 'dev' in split_ver[3]:
 251                 doc_version = 'devel'
 252 
 253         return '{0}{1}/{2}'.format(base_url, doc_version, path)
 254     except Exception as ex:
 255         return '(unable to create versioned doc link for path {0}: {1})'.format(path, to_native(ex))
