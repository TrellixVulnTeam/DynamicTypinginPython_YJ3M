  28 def unicode_wrap(func, *args, **kwargs):
  29     """If a function returns a string, force it to be a text string.
  30 
  31     Use with partial to ensure that filter plugins will return text values.
  32     """
  33     return to_text(func(*args, **kwargs), nonstring='passthru')
