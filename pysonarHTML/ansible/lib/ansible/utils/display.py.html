  64 def initialize_locale():
  65     """Set the locale to the users default setting
  66     and set ``_LOCALE_INITIALIZED`` to indicate whether
  67     ``get_text_width`` may run into trouble
  68     """
  69     global _LOCALE_INITIALIZED, _LOCALE_INITIALIZATION_ERR
  70     if _LOCALE_INITIALIZED is False:
  71         try:
  72             locale.setlocale(locale.LC_ALL, '')
  73         except locale.Error as e:
  74             _LOCALE_INITIALIZATION_ERR = e
  75         else:
  76             _LOCALE_INITIALIZED = True

  79 def get_text_width(text):
  80     """Function that utilizes ``wcswidth`` or ``wcwidth`` to determine the
  81     number of columns used to display a text string.
  82 
  83     We try first with ``wcswidth``, and fallback to iterating each
  84     character and using wcwidth individually, falling back to a value of 0
  85     for non-printable wide characters
  86 
  87     On Py2, this depends on ``locale.setlocale(locale.LC_ALL, '')``,
  88     that in the case of Ansible is done in ``bin/ansible``
  89     """
  90     if not isinstance(text, text_type):
  91         raise TypeError('get_text_width requires text, not %s' % type(text))
  92 
  93     if _LOCALE_INITIALIZATION_ERR:
  94         Display().warning(
  95             'An error occurred while calling ansible.utils.display.initialize_locale '
  96             '(%s). This may result in incorrectly calculated text widths that can '
  97             'cause Display to print incorrect line lengths' % _LOCALE_INITIALIZATION_ERR
  98         )
  99     elif not _LOCALE_INITIALIZED:
 100         Display().warning(
 101             'ansible.utils.display.initialize_locale has not been called, '
 102             'this may result in incorrectly calculated text widths that can '
 103             'cause Display to print incorrect line lengths'
 104         )
 105 
 106     try:
 107         width = _LIBC.wcswidth(text, _MAX_INT)
 108     except ctypes.ArgumentError:
 109         width = -1
 110     if width != -1:
 111         return width
 112 
 113     width = 0
 114     counter = 0
 115     for c in text:
 116         counter += 1
 117         if c in (u'\x08', u'\x7f', u'\x94', u'\x1b'):
 118             # A few characters result in a subtraction of length:
 119             # BS, DEL, CCH, ESC
 120             # ESC is slightly different in that it's part of an escape sequence, and
 121             # while ESC is non printable, it's part of an escape sequence, which results
 122             # in a single non printable length
 123             width -= 1
 124             counter -= 1
 125             continue
 126 
 127         try:
 128             w = _LIBC.wcwidth(c)
 129         except ctypes.ArgumentError:
 130             w = -1
 131         if w == -1:
 132             # -1 signifies a non-printable character
 133             # use 0 here as a best effort
 134             w = 0
 135         width += w
 136 
 137     if width == 0 and counter and not _LOCALE_INITIALIZED:
 138         raise EnvironmentError(
 139             'ansible.utils.display.initialize_locale has not been called, '
 140             'and get_text_width could not calculate text width of %r' % text
 141         )
 142 
 143     # It doesn't make sense to have a negative printable width
 144     return width if width >= 0 else 0

 147 class FilterBlackList(logging.Filter):
 148     def __init__(self, blacklist):
 149         self.blacklist = [logging.Filter(name) for name in blacklist]

 151     def filter(self, record):
 152         return not any(f.filter(record) for f in self.blacklist)

 155 class FilterUserInjector(logging.Filter):
 156     """
 157     This is a filter which injects the current user as the 'user' attribute on each record. We need to add this filter
 158     to all logger handlers so that 3rd party libraries won't print an exception due to user not being defined.
 159     """
 160 
 161     try:
 162         username = getpass.getuser()
 163     except KeyError:
 164         # people like to make containers w/o actual valid passwd/shadow and use host uids
 165         username = 'uid=%s' % os.getuid()
 166 
 167     def filter(self, record):
 168         record.user = FilterUserInjector.username
 169         return True

 208     def __init__(self, verbosity=0):

 231                 # could not execute cowsay for some reason
 232                 self.b_cowsay = False

 247     def display(self, msg, color=None, stderr=False, screen_only=False, log_only=False, newline=True):
 248         """ Display a message to the user
 249 
 250         Note: msg *must* be a unicode string to prevent UnicodeError tracebacks.
 251         """
 252 
 253         nocolor = msg
 254 
 255         if not log_only:
 256 
 257             has_newline = msg.endswith(u'\n')
 258             if has_newline:
 259                 msg2 = msg[:-1]
 260             else:
 261                 msg2 = msg
 262 
 263             if color:
 264                 msg2 = stringc(msg2, color)
 265 
 266             if has_newline or newline:
 267                 msg2 = msg2 + u'\n'
 268 
 269             msg2 = to_bytes(msg2, encoding=self._output_encoding(stderr=stderr))
 270             if sys.version_info >= (3,):
 271                 # Convert back to text string on python3
 272                 # We first convert to a byte string so that we get rid of
 273                 # characters that are invalid in the user's locale
 274                 msg2 = to_text(msg2, self._output_encoding(stderr=stderr), errors='replace')
 275 
 276             # Note: After Display() class is refactored need to update the log capture
 277             # code in 'bin/ansible-connection' (and other relevant places).
 278             if not stderr:
 279                 fileobj = sys.stdout
 280             else:
 281                 fileobj = sys.stderr
 282 
 283             fileobj.write(msg2)
 284 
 285             try:
 286                 fileobj.flush()
 287             except IOError as e:
 288                 # Ignore EPIPE in case fileobj has been prematurely closed, eg.
 289                 # when piping to "head -n1"
 290                 if e.errno != errno.EPIPE:
 291                     raise
 292 
 293         if logger and not screen_only:
 294             # We first convert to a byte string so that we get rid of
 295             # color and characters that are invalid in the user's locale
 296             msg2 = to_bytes(nocolor.lstrip(u'\n'))
 297 
 298             if sys.version_info >= (3,):
 299                 # Convert back to text string on python3
 300                 msg2 = to_text(msg2, self._output_encoding(stderr=stderr))
 301 
 302             lvl = logging.INFO
 303             if color:
 304                 # set logger level based on color (not great)
 305                 try:
 306                     lvl = color_to_log_level[color]
 307                 except KeyError:
 308                     # this should not happen, but JIC
 309                     raise AnsibleAssertionError('Invalid color supplied to display: %s' % color)
 310             # actually log
 311             logger.log(lvl, msg2)

 347     def get_deprecation_message(self, msg, version=None, removed=False, date=None, collection_name=None):
 348         ''' used to print out a deprecation message.'''
 349         msg = msg.strip()
 350         if msg and msg[-1] not in ['!', '?', '.']:
 351             msg += '.'
 352 
 353         if collection_name == 'ansible.builtin':
 354             collection_name = 'ansible-base'
 355 
 356         if removed:
 357             header = '[DEPRECATED]: {0}'.format(msg)
 358             removal_fragment = 'This feature was removed'
 359             help_text = 'Please update your playbooks.'
 360         else:
 361             header = '[DEPRECATION WARNING]: {0}'.format(msg)
 362             removal_fragment = 'This feature will be removed'
 363             # FUTURE: make this a standalone warning so it only shows up once?
 364             help_text = 'Deprecation warnings can be disabled by setting deprecation_warnings=False in ansible.cfg.'
 365 
 366         if collection_name:
 367             from_fragment = 'from {0}'.format(collection_name)
 368         else:
 369             from_fragment = ''
 370 
 371         if date:
 372             when = 'in a release after {0}.'.format(date)
 373         elif version:
 374             when = 'in version {0}.'.format(version)
 375         else:
 376             when = 'in a future release.'
 377 
 378         message_text = ' '.join(f for f in [header, removal_fragment, from_fragment, when, help_text] if f)
 379 
 380         return message_text

 382     def deprecated(self, msg, version=None, removed=False, date=None, collection_name=None):
 383         if not removed and not C.DEPRECATION_WARNINGS:
 384             return
 385 
 386         message_text = self.get_deprecation_message(msg, version=version, removed=removed, date=date, collection_name=collection_name)
 387 
 388         if removed:
 389             raise AnsibleError(message_text)
 390 
 391         wrapped = textwrap.wrap(message_text, self.columns, drop_whitespace=False)
 392         message_text = "\n".join(wrapped) + "\n"
 393 
 394         if message_text not in self._deprecations:
 395             self.display(message_text.strip(), color=C.COLOR_DEPRECATE, stderr=True)
 396             self._deprecations[message_text] = 1
 397 
 398     def warning(self, msg, formatted=False):
 399 
 400         if not formatted:
 401             new_msg = "[WARNING]: %s" % msg
 402             wrapped = textwrap.wrap(new_msg, self.columns)
 403             new_msg = "\n".join(wrapped) + "\n"
 404         else:
 405             new_msg = "\n[WARNING]: \n%s" % msg
 406 
 407         if new_msg not in self._warns:
 408             self.display(new_msg, color=C.COLOR_WARN, stderr=True)
 409             self._warns[new_msg] = 1

 415     def banner(self, msg, color=None, cows=True):
 416         '''
 417         Prints a header-looking line with cowsay or stars with length depending on terminal width (3 minimum)
 418         '''
 419         msg = to_text(msg)
 420 
 421         if self.b_cowsay and cows:
 422             try:
 423                 self.banner_cowsay(msg)
 424                 return
 425             except OSError:
 426                 self.warning("somebody cleverly deleted cowsay or something during the PB run.  heh.")
 427 
 428         msg = msg.strip()
 429         try:
 430             star_len = self.columns - get_text_width(msg)
 431         except EnvironmentError:
 432             star_len = self.columns - len(msg)
 433         if star_len <= 3:
 434             star_len = 3
 435         stars = u"*" * star_len
 436         self.display(u"\n%s %s" % (msg, stars), color=color)
 437 
 438     def banner_cowsay(self, msg, color=None):
 439         if u": [" in msg:
 440             msg = msg.replace(u"[", u"")
 441             if msg.endswith(u"]"):
 442                 msg = msg[:-1]
 443         runcmd = [self.b_cowsay, b"-W", b"60"]
 444         if self.noncow:
 445             thecow = self.noncow
 446             if thecow == 'random':
 447                 thecow = random.choice(list(self.cows_available))
 448             runcmd.append(b'-f')
 449             runcmd.append(to_bytes(thecow))
 450         runcmd.append(to_bytes(msg))
 451         cmd = subprocess.Popen(runcmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
 452         (out, err) = cmd.communicate()
 453         self.display(u"%s\n" % to_text(out), color=color)

 455     def error(self, msg, wrap_text=True):
 456         if wrap_text:
 457             new_msg = u"\n[ERROR]: %s" % msg
 458             wrapped = textwrap.wrap(new_msg, self.columns)
 459             new_msg = u"\n".join(wrapped) + u"\n"
 460         else:
 461             new_msg = u"ERROR! %s" % msg
 462         if new_msg not in self._errors:
 463             self.display(new_msg, color=C.COLOR_ERROR, stderr=True)
 464             self._errors[new_msg] = 1

 466     @staticmethod
 467     def prompt(msg, private=False):
 468         prompt_string = to_bytes(msg, encoding=Display._output_encoding())
 469         if sys.version_info >= (3,):
 470             # Convert back into text on python3.  We do this double conversion
 471             # to get rid of characters that are illegal in the user's locale
 472             prompt_string = to_text(prompt_string)
 473 
 474         if private:
 475             return getpass.getpass(prompt_string)
 476         else:
 477             return input(prompt_string)

 479     def do_var_prompt(self, varname, private=True, prompt=None, encrypt=None, confirm=False, salt_size=None, salt=None, default=None, unsafe=None):
 480 
 481         result = None
 482         if sys.__stdin__.isatty():
 483 
 484             do_prompt = self.prompt
 485 
 486             if prompt and default is not None:
 487                 msg = "%s [%s]: " % (prompt, default)
 488             elif prompt:
 489                 msg = "%s: " % prompt
 490             else:
 491                 msg = 'input for %s: ' % varname
 492 
 493             if confirm:
 494                 while True:
 495                     result = do_prompt(msg, private)
 496                     second = do_prompt("confirm " + msg, private)
 497                     if result == second:
 498                         break
 499                     self.display("***** VALUES ENTERED DO NOT MATCH ****")
 500             else:
 501                 result = do_prompt(msg, private)
 502         else:
 503             result = None
 504             self.warning("Not prompting as we are not in interactive mode")
 505 
 506         # if result is false and default is not None
 507         if not result and default is not None:
 508             result = default
 509 
 510         if encrypt:
 511             # Circular import because encrypt needs a display class
 512             from ansible.utils.encrypt import do_encrypt
 513             result = do_encrypt(result, encrypt, salt_size, salt)
 514 
 515         # handle utf-8 chars
 516         result = to_text(result, errors='surrogate_or_strict')
 517 
 518         if unsafe:
 519             result = wrap_var(result)
 520         return result

 522     @staticmethod
 523     def _output_encoding(stderr=False):
 524         encoding = locale.getpreferredencoding()
 525         # https://bugs.python.org/issue6202
 526         # Python2 hardcodes an obsolete value on Mac.  Use MacOSX defaults
 527         # instead.
 528         if encoding in ('mac-roman',):
 529             encoding = 'utf-8'
 530         return encoding

 532     def _set_column_width(self):
 533         if os.isatty(1):
 534             tty_size = unpack('HHHH', fcntl.ioctl(1, TIOCGWINSZ, pack('HHHH', 0, 0, 0, 0)))[1]
 535         else:
 536             tty_size = 0
 537         self.columns = max(79, tty_size - 1)
