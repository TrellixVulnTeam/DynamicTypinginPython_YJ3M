  20 def _make_immutable(obj):
  21     """Recursively convert a container and objects inside of it into immutable data types"""
  22     if isinstance(obj, (text_type, binary_type)):
  23         # Strings first because they are also sequences
  24         return obj
  25     elif isinstance(obj, Mapping):
  26         temp_dict = {}
  27         for key, value in obj.items():
  28             if isinstance(value, Container):
  29                 temp_dict[key] = _make_immutable(value)
  30             else:
  31                 temp_dict[key] = value
  32         return ImmutableDict(temp_dict)
  33     elif isinstance(obj, Set):
  34         temp_set = set()
  35         for value in obj:
  36             if isinstance(value, Container):
  37                 temp_set.add(_make_immutable(value))
  38             else:
  39                 temp_set.add(value)
  40         return frozenset(temp_set)
  41     elif isinstance(obj, Sequence):
  42         temp_sequence = []
  43         for value in obj:
  44             if isinstance(value, Container):
  45                 temp_sequence.append(_make_immutable(value))
  46             else:
  47                 temp_sequence.append(value)
  48         return tuple(temp_sequence)
  49 
  50     return obj

  63 class CLIArgs(ImmutableDict):
  64     """
  65     Hold a parsed copy of cli arguments
  66 
  67     We have both this non-Singleton version and the Singleton, GlobalCLIArgs, version to leave us
  68     room to implement a Context object in the future.  Whereas there should only be one set of args
  69     in a global context, individual Context objects might want to pretend that they have different
  70     command line switches to trigger different behaviour when they run.  So if we support Contexts
  71     in the future, they would use CLIArgs instead of GlobalCLIArgs to store their version of command
  72     line flags.
  73     """
  74     def __init__(self, mapping):
  75         toplevel = {}
  76         for key, value in mapping.items():
  77             toplevel[key] = _make_immutable(value)
  78         super(CLIArgs, self).__init__(toplevel)
  79 
  80     @classmethod
  81     def from_options(cls, options):
  82         return cls(vars(options))
  83
