  11 class _EventSource:
  12     def __init__(self):
  13         self._handlers = set()
  14 
  15     def __iadd__(self, handler):
  16         if not callable(handler):
  17             raise ValueError('handler must be callable')
  18         self._handlers.add(handler)
  19         return self
  20 
  21     def __isub__(self, handler):
  22         try:
  23             self._handlers.remove(handler)
  24         except KeyError:
  25             pass
  26 
  27         return self
  28 
  29     def _on_exception(self, handler, exc, *args, **kwargs):
  30         # if we return True, we want the caller to re-raise
  31         return True
  32 
  33     def fire(self, *args, **kwargs):
  34         for h in self._handlers:
  35             try:
  36                 h(*args, **kwargs)
  37             except Exception as ex:
  38                 if self._on_exception(h, ex, *args, **kwargs):
  39                     raise
