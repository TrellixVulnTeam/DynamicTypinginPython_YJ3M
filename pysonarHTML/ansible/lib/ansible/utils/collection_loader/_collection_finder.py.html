 102     def _ansible_collection_path_hook(self, path):
 103         path = to_native(path)
 104         interesting_paths = self._n_cached_collection_qualified_paths
 105         if not interesting_paths:
 106             interesting_paths = [os.path.join(p, 'ansible_collections') for p in

 108             interesting_paths.insert(0, self._ansible_pkg_path)

 145     def find_module(self, fullname, path=None):
 146         # Figure out what's being asked for, and delegate to a special-purpose loader
 147 
 148         split_name = fullname.split('.')
 149         toplevel_pkg = split_name[0]
 150         module_to_find = split_name[-1]
 151         part_count = len(split_name)
 152 
 153         if toplevel_pkg not in ['ansible', 'ansible_collections']:
 154             # not interested in anything other than ansible_collections (and limited cases under ansible)
 155             return None
 156 
 157         # sanity check what we're getting from import, canonicalize path values
 158         if part_count == 1:
 159             if path:
 160                 raise ValueError('path should not be specified for top-level packages (trying to find {0})'.format(fullname))
 161             else:
 162                 # seed the path to the configured collection roots
 163                 path = self._n_collection_paths
 164 
 165         if part_count > 1 and path is None:
 166             raise ValueError('path must be specified for subpackages (trying to find {0})'.format(fullname))
 167 
 168         # NB: actual "find"ing is delegated to the constructors on the various loaders; they'll ImportError if not found
 169         try:
 170             if toplevel_pkg == 'ansible':
 171                 # something under the ansible package, delegate to our internal loader in case of redirections
 172                 return _AnsibleInternalRedirectLoader(fullname=fullname, path_list=path)
 173             if part_count == 1:
 174                 return _AnsibleCollectionRootPkgLoader(fullname=fullname, path_list=path)
 175             if part_count == 2:  # ns pkg eg, ansible_collections, ansible_collections.somens
 176                 return _AnsibleCollectionNSPkgLoader(fullname=fullname, path_list=path)
 177             elif part_count == 3:  # collection pkg eg, ansible_collections.somens.somecoll
 178                 return _AnsibleCollectionPkgLoader(fullname=fullname, path_list=path)
 179             # anything below the collection
 180             return _AnsibleCollectionLoader(fullname=fullname, path_list=path)
 181         except ImportError:
 182             # TODO: log attempt to load context
 183             return None

 186 # Implements a path_hook finder for iter_modules (since it's only path based). This finder does not need to actually
 187 # function as a finder in most cases, since our meta_path finder is consulted first for *almost* everything, except
 188 # pkgutil.iter_modules, and under py2, pkgutil.get_data if the parent package passed has not been loaded yet.
 189 class _AnsiblePathHookFinder:
 190     def __init__(self, collection_finder, pathctx):
 191         # when called from a path_hook, find_module doesn't usually get the path arg, so this provides our context
 192         self._pathctx = to_native(pathctx)
 193         self._collection_finder = collection_finder
 194         if PY3:
 195             # cache the native FileFinder (take advantage of its filesystem cache for future find/load requests)
 196             self._file_finder = None
 197 
 198     # class init is fun- this method has a self arg that won't get used
 199     def _get_filefinder_path_hook(self=None):
 200         _file_finder_hook = None
 201         if PY3:
 202             # try to find the FileFinder hook to call for fallback path-based imports in Py3
 203             _file_finder_hook = [ph for ph in sys.path_hooks if 'FileFinder' in repr(ph)]
 204             if len(_file_finder_hook) != 1:
 205                 raise Exception('need exactly one FileFinder import hook (found {0})'.format(len(_file_finder_hook)))
 206             _file_finder_hook = _file_finder_hook[0]
 207 
 208         return _file_finder_hook
 209 
 210     _filefinder_path_hook = _get_filefinder_path_hook()

 212     def find_module(self, fullname, path=None):
 213         # we ignore the passed in path here- use what we got from the path hook init
 214         split_name = fullname.split('.')
 215         toplevel_pkg = split_name[0]
 216 
 217         if toplevel_pkg == 'ansible_collections':
 218             # collections content? delegate to the collection finder
 219             return self._collection_finder.find_module(fullname, path=[self._pathctx])
 220         else:
 221             # Something else; we'd normally restrict this to `ansible` descendent modules so that any weird loader
 222             # behavior that arbitrary Python modules have can be serviced by those loaders. In some dev/test
 223             # scenarios (eg a venv under a collection) our path_hook signs us up to load non-Ansible things, and
 224             # it's too late by the time we've reached this point, but also too expensive for the path_hook to figure
 225             # out what we *shouldn't* be loading with the limited info it has. So we'll just delegate to the
 226             # normal path-based loader as best we can to service it. This also allows us to take advantage of Python's
 227             # built-in FS caching and byte-compilation for most things.
 228             if PY3:
 229                 # create or consult our cached file finder for this path
 230                 if not self._file_finder:
 231                     try:
 232                         self._file_finder = _AnsiblePathHookFinder._filefinder_path_hook(self._pathctx)
 233                     except ImportError:
 234                         # FUTURE: log at a high logging level? This is normal for things like python36.zip on the path, but
 235                         # might not be in some other situation...
 236                         return None
 237 
 238                 spec = self._file_finder.find_spec(fullname)
 239                 if not spec:
 240                     return None
 241                 return spec.loader
 242             else:
 243                 # call py2's internal loader
 244                 return pkgutil.ImpImporter(self._pathctx).find_module(fullname)
 245 
 246     def iter_modules(self, prefix):
 247         # NB: this currently represents only what's on disk, and does not handle package redirection
 248         return _iter_modules_impl([self._pathctx], prefix)

 250     def __repr__(self):
 251         return "{0}(path='{1}')".format(self.__class__.__name__, self._pathctx)

 294     @staticmethod
 295     @contextmanager
 296     def _new_or_existing_module(name, **kwargs):
 297         # handle all-or-nothing sys.modules creation/use-existing/delete-on-exception-if-created behavior
 298         created_module = False
 299         module = sys.modules.get(name)
 300         try:
 301             if not module:
 302                 module = ModuleType(name)
 303                 created_module = True
 304                 sys.modules[name] = module
 305             # always override the values passed, except name (allow reference aliasing)
 306             for attr, value in kwargs.items():
 307                 setattr(module, attr, value)
 308             yield module
 309         except Exception:
 310             if created_module:
 311                 if sys.modules.get(name):
 312                     sys.modules.pop(name)
 313             raise

 315     # basic module/package location support
 316     # NB: this does not support distributed packages!
 317     @staticmethod
 318     def _module_file_from_path(leaf_name, path):
 319         has_code = True
 320         package_path = os.path.join(to_native(path), to_native(leaf_name))
 321         module_path = None
 322 
 323         # if the submodule is a package, assemble valid submodule paths, but stop looking for a module
 324         if os.path.isdir(to_bytes(package_path)):
 325             # is there a package init?
 326             module_path = os.path.join(package_path, '__init__.py')
 327             if not os.path.isfile(to_bytes(module_path)):
 328                 module_path = os.path.join(package_path, '__synthetic__')
 329                 has_code = False
 330         else:
 331             module_path = package_path + '.py'
 332             package_path = None
 333             if not os.path.isfile(to_bytes(module_path)):
 334                 raise ImportError('{0} not found at {1}'.format(leaf_name, path))
 335 
 336         return module_path, has_code, package_path

 338     def load_module(self, fullname):
 339         # short-circuit redirect; we've already imported the redirected module, so just alias it and return it
 340         if self._redirect_module:
 341             sys.modules[self._fullname] = self._redirect_module
 342             return self._redirect_module
 343 
 344         # we're actually loading a module/package
 345         module_attrs = dict(
 346             __loader__=self,
 347             __file__=self.get_filename(fullname),
 348             __package__=self._parent_package_name  # sane default for non-packages
 349         )
 350 
 351         # eg, I am a package
 352         if self._subpackage_search_paths is not None:  # empty is legal
 353             module_attrs['__path__'] = self._subpackage_search_paths
 354             module_attrs['__package__'] = fullname  # per PEP366
 355 
 356         with self._new_or_existing_module(fullname, **module_attrs) as module:
 357             # execute the module's code in its namespace
 358             exec(self.get_code(fullname), module.__dict__)
 359 
 360             return module

 367     def get_source(self, fullname):
 368         if self._decoded_source:
 369             return self._decoded_source
 370         if fullname != self._fullname:
 371             raise ValueError('this loader cannot load source for {0}, only {1}'.format(fullname, self._fullname))
 372         if not self._source_code_path:
 373             return None
 374         # FIXME: what do we want encoding/newline requirements to be?
 375         self._decoded_source = self.get_data(self._source_code_path)
 376         return self._decoded_source

 378     def get_data(self, path):
 379         if not path:
 380             raise ValueError('a path must be specified')
 381 
 382         # TODO: ensure we're being asked for a path below something we own
 383         # TODO: try to handle redirects internally?
 384 
 385         if not path[0] == '/':
 386             # relative to current package, search package paths if possible (this may not be necessary)
 387             # candidate_paths = [os.path.join(ssp, path) for ssp in self._subpackage_search_paths]
 388             raise ValueError('relative resource paths not supported')
 389         else:
 390             candidate_paths = [path]
 391 
 392         for p in candidate_paths:
 393             b_path = to_bytes(p)
 394             if os.path.isfile(b_path):
 395                 with open(b_path, 'rb') as fd:
 396                     return fd.read()
 397 
 398         return None

 403     def get_filename(self, fullname):
 404         if fullname != self._fullname:
 405             raise ValueError('this loader cannot find files for {0}, only {1}'.format(fullname, self._fullname))
 406 
 407         filename = self._source_code_path
 408 
 409         if not filename and self.is_package(fullname):
 410             if len(self._subpackage_search_paths) == 1:
 411                 filename = os.path.join(self._subpackage_search_paths[0], '__synthetic__')
 412             else:
 413                 filename = self._synthetic_filename(fullname)
 414 
 415         return filename

 417     def get_code(self, fullname):
 418         if self._compiled_code:
 419             return self._compiled_code
 420 
 421         # this may or may not be an actual filename, but it's the value we'll use for __file__
 422         filename = self.get_filename(fullname)
 423         if not filename:
 424             filename = '<string>'
 425 
 426         source_code = self.get_source(fullname)
 427         if not source_code:
 428             source_code = ''
 429 
 430         self._compiled_code = compile(source=source_code, filename=filename, mode='exec', flags=0, dont_inherit=True)
 431 
 432         return self._compiled_code

 483     def load_module(self, fullname):
 484         if not _meta_yml_to_dict:
 485             raise ValueError('ansible.utils.collection_loader._meta_yml_to_dict is not set')
 486 
 487         module = super(_AnsibleCollectionPkgLoader, self).load_module(fullname)
 488 
 489         module._collection_meta = {}
 490         # TODO: load collection metadata, cache in __loader__ state
 491 
 492         collection_name = '.'.join(self._split_name[1:3])
 493 
 494         if collection_name == 'ansible.builtin':
 495             # ansible.builtin is a synthetic collection, get its routing config from the Ansible distro
 496             ansible_pkg_path = os.path.dirname(import_module('ansible').__file__)
 497             metadata_path = os.path.join(ansible_pkg_path, 'config/ansible_builtin_runtime.yml')
 498             with open(to_bytes(metadata_path), 'rb') as fd:
 499                 raw_routing = fd.read()
 500         else:
 501             b_routing_meta_path = to_bytes(os.path.join(module.__path__[0], 'meta/runtime.yml'))
 502             if os.path.isfile(b_routing_meta_path):
 503                 with open(b_routing_meta_path, 'rb') as fd:
 504                     raw_routing = fd.read()
 505             else:
 506                 raw_routing = ''
 507         try:
 508             if raw_routing:
 509                 routing_dict = _meta_yml_to_dict(raw_routing, (collection_name, 'runtime.yml'))
 510                 module._collection_meta = self._canonicalize_meta(routing_dict)
 511         except Exception as ex:
 512             raise ValueError('error parsing collection metadata: {0}'.format(to_native(ex)))
 513 
 514         AnsibleCollectionConfig.on_collection_load.fire(collection_name=collection_name, collection_path=os.path.dirname(module.__file__))
 515 
 516         return module

 546 # loads everything under a collection, including handling redirections defined by the collection
 547 class _AnsibleCollectionLoader(_AnsibleCollectionPkgLoaderBase):
 548     # HACK: stash this in a better place
 549     _redirected_package_map = {}
 550     _allows_package_code = True

 552     def _validate_args(self):
 553         super(_AnsibleCollectionLoader, self)._validate_args()
 554         if len(self._split_name) < 4:
 555             raise ValueError('this loader is only for sub-collection modules/packages, not {0}'.format(self._fullname))

 557     def _get_candidate_paths(self, path_list):
 558         if len(path_list) != 1 and self._split_name[1:3] != ['ansible', 'builtin']:
 559             raise ValueError('this loader requires exactly one path to search')
 560 
 561         return path_list

 563     def _get_subpackage_search_paths(self, candidate_paths):
 564         collection_name = '.'.join(self._split_name[1:3])
 565         collection_meta = _get_collection_metadata(collection_name)
 566 
 567         # check for explicit redirection, as well as ancestor package-level redirection (only load the actual code once!)
 568         redirect = None
 569         explicit_redirect = False
 570 
 571         routing_entry = _nested_dict_get(collection_meta, ['import_redirection', self._fullname])
 572         if routing_entry:
 573             redirect = routing_entry.get('redirect')
 574 
 575         if redirect:
 576             explicit_redirect = True
 577         else:
 578             redirect = _get_ancestor_redirect(self._redirected_package_map, self._fullname)
 579 
 580         # NB: package level redirection requires hooking all future imports beneath the redirected source package
 581         # in order to ensure sanity on future relative imports. We always import everything under its "real" name,
 582         # then add a sys.modules entry with the redirected name using the same module instance. If we naively imported
 583         # the source for each redirection, most submodules would import OK, but we'd have N runtime copies of the module
 584         # (one for each name), and relative imports that ascend above the redirected package would break (since they'd
 585         # see the redirected ancestor package contents instead of the package where they actually live).
 586         if redirect:
 587             # FIXME: wrap this so we can be explicit about a failed redirection
 588             self._redirect_module = import_module(redirect)
 589             if explicit_redirect and hasattr(self._redirect_module, '__path__') and self._redirect_module.__path__:
 590                 # if the import target looks like a package, store its name so we can rewrite future descendent loads
 591                 self._redirected_package_map[self._fullname] = redirect
 592 
 593             # if we redirected, don't do any further custom package logic
 594             return None
 595 
 596         # we're not doing a redirect- try to find what we need to actually load a module/package
 597 
 598         # this will raise ImportError if we can't find the requested module/package at all
 599         if not candidate_paths:
 600             # noplace to look, just ImportError
 601             raise ImportError('package has no paths')
 602 
 603         found_path, has_code, package_path = self._module_file_from_path(self._package_to_load, candidate_paths[0])
 604 
 605         # still here? we found something to load...
 606         if has_code:
 607             self._source_code_path = found_path
 608 
 609         if package_path:
 610             return [package_path]  # always needs to be a list
 611 
 612         return None

 652 class AnsibleCollectionRef:
 653     # FUTURE: introspect plugin loaders to get these dynamically?
 654     VALID_REF_TYPES = frozenset(to_text(r) for r in ['action', 'become', 'cache', 'callback', 'cliconf', 'connection',
 655                                                      'doc_fragments', 'filter', 'httpapi', 'inventory', 'lookup',
 656                                                      'module_utils', 'modules', 'netconf', 'role', 'shell', 'strategy',
 657                                                      'terminal', 'test', 'vars'])
 658 
 659     # FIXME: tighten this up to match Python identifier reqs, etc
 660     VALID_COLLECTION_NAME_RE = re.compile(to_text(r'^(\w+)\.(\w+)$'))
 661     VALID_SUBDIRS_RE = re.compile(to_text(r'^\w+(\.\w+)*$'))
 662     VALID_FQCR_RE = re.compile(to_text(r'^\w+\.\w+\.\w+(\.\w+)*$'))  # can have 0-N included subdirs as well
 663 
 664     def __init__(self, collection_name, subdirs, resource, ref_type):
 665         """
 666         Create an AnsibleCollectionRef from components
 667         :param collection_name: a collection name of the form 'namespace.collectionname'
 668         :param subdirs: optional subdir segments to be appended below the plugin type (eg, 'subdir1.subdir2')
 669         :param resource: the name of the resource being references (eg, 'mymodule', 'someaction', 'a_role')
 670         :param ref_type: the type of the reference, eg 'module', 'role', 'doc_fragment'
 671         """
 672         collection_name = to_text(collection_name, errors='strict')
 673         if subdirs is not None:
 674             subdirs = to_text(subdirs, errors='strict')
 675         resource = to_text(resource, errors='strict')
 676         ref_type = to_text(ref_type, errors='strict')
 677 
 678         if not self.is_valid_collection_name(collection_name):
 679             raise ValueError('invalid collection name (must be of the form namespace.collection): {0}'.format(to_native(collection_name)))
 680 
 681         if ref_type not in self.VALID_REF_TYPES:
 682             raise ValueError('invalid collection ref_type: {0}'.format(ref_type))
 683 
 684         self.collection = collection_name
 685         if subdirs:
 686             if not re.match(self.VALID_SUBDIRS_RE, subdirs):
 687                 raise ValueError('invalid subdirs entry: {0} (must be empty/None or of the form subdir1.subdir2)'.format(to_native(subdirs)))
 688             self.subdirs = subdirs
 689         else:
 690             self.subdirs = u''
 691 
 692         self.resource = resource
 693         self.ref_type = ref_type
 694 
 695         package_components = [u'ansible_collections', self.collection]
 696         fqcr_components = [self.collection]
 697 
 698         self.n_python_collection_package_name = to_native('.'.join(package_components))
 699 
 700         if self.ref_type == u'role':
 701             package_components.append(u'roles')
 702         else:
 703             # we assume it's a plugin
 704             package_components += [u'plugins', self.ref_type]
 705 
 706         if self.subdirs:
 707             package_components.append(self.subdirs)
 708             fqcr_components.append(self.subdirs)
 709 
 710         if self.ref_type == u'role':
 711             # roles are their own resource
 712             package_components.append(self.resource)
 713 
 714         fqcr_components.append(self.resource)
 715 
 716         self.n_python_package_name = to_native('.'.join(package_components))
 717         self._fqcr = u'.'.join(fqcr_components)
 718 
 719     def __repr__(self):
 720         return 'AnsibleCollectionRef(collection={0!r}, subdirs={1!r}, resource={2!r})'.format(self.collection, self.subdirs, self.resource)

 726     @staticmethod
 727     def from_fqcr(ref, ref_type):
 728         """
 729         Parse a string as a fully-qualified collection reference, raises ValueError if invalid
 730         :param ref: collection reference to parse (a valid ref is of the form 'ns.coll.resource' or 'ns.coll.subdir1.subdir2.resource')
 731         :param ref_type: the type of the reference, eg 'module', 'role', 'doc_fragment'
 732         :return: a populated AnsibleCollectionRef object
 733         """
 734         # assuming the fq_name is of the form (ns).(coll).(optional_subdir_N).(resource_name),
 735         # we split the resource name off the right, split ns and coll off the left, and we're left with any optional
 736         # subdirs that need to be added back below the plugin-specific subdir we'll add. So:
 737         # ns.coll.resource -> ansible_collections.ns.coll.plugins.(plugintype).resource
 738         # ns.coll.subdir1.resource -> ansible_collections.ns.coll.plugins.subdir1.(plugintype).resource
 739         # ns.coll.rolename -> ansible_collections.ns.coll.roles.rolename
 740         if not AnsibleCollectionRef.is_valid_fqcr(ref):
 741             raise ValueError('{0} is not a valid collection reference'.format(to_native(ref)))
 742 
 743         ref = to_text(ref, errors='strict')
 744         ref_type = to_text(ref_type, errors='strict')
 745 
 746         resource_splitname = ref.rsplit(u'.', 1)
 747         package_remnant = resource_splitname[0]
 748         resource = resource_splitname[1]
 749 
 750         # split the left two components of the collection package name off, anything remaining is plugin-type
 751         # specific subdirs to be added back on below the plugin type
 752         package_splitname = package_remnant.split(u'.', 2)
 753         if len(package_splitname) == 3:
 754             subdirs = package_splitname[2]
 755         else:
 756             subdirs = u''
 757 
 758         collection_name = u'.'.join(package_splitname[0:2])
 759 
 760         return AnsibleCollectionRef(collection_name, subdirs, resource, ref_type)
 761 
 762     @staticmethod
 763     def try_parse_fqcr(ref, ref_type):
 764         """
 765         Attempt to parse a string as a fully-qualified collection reference, returning None on failure (instead of raising an error)
 766         :param ref: collection reference to parse (a valid ref is of the form 'ns.coll.resource' or 'ns.coll.subdir1.subdir2.resource')
 767         :param ref_type: the type of the reference, eg 'module', 'role', 'doc_fragment'
 768         :return: a populated AnsibleCollectionRef object on successful parsing, else None
 769         """
 770         try:
 771             return AnsibleCollectionRef.from_fqcr(ref, ref_type)
 772         except ValueError:
 773             pass

 796         """
 797         Validates if is string is a well-formed fully-qualified collection reference (does not look up the collection itself)
 798         :param ref: candidate collection reference to validate (a valid ref is of the form 'ns.coll.resource' or 'ns.coll.subdir1.subdir2.resource')

 857 def _get_collection_name_from_path(path):
 858     """
 859     Return the containing collection name for a given path, or None if the path is not below a configured collection, or
 860     the collection cannot be loaded (eg, the collection is masked by another of the same name higher in the configured
 861     collection roots).
 862     :param path: path to evaluate for collection containment
 863     :return: collection name or None
 864     """
 865 
 866     # FIXME: mess with realpath canonicalization or not?
 867     path = to_native(path)
 868 
 869     path_parts = path.split('/')
 870     if path_parts.count('ansible_collections') != 1:
 871         return None
 872 
 873     ac_pos = path_parts.index('ansible_collections')
 874 
 875     # make sure it's followed by at least a namespace and collection name
 876     if len(path_parts) < ac_pos + 3:
 877         return None
 878 
 879     candidate_collection_name = '.'.join(path_parts[ac_pos + 1:ac_pos + 3])
 880 
 881     try:
 882         # we've got a name for it, now see if the path prefix matches what the loader sees
 883         imported_pkg_path = to_native(os.path.dirname(to_bytes(import_module('ansible_collections.' + candidate_collection_name).__file__)))
 884     except ImportError:
 885         return None
 886 
 887     # reassemble the original path prefix up the collection name, and it should match what we just imported. If not
 888     # this is probably a collection root that's not configured.
 889 
 890     original_path_prefix = os.path.join('/', *path_parts[0:ac_pos + 3])
 891 
 892     if original_path_prefix != imported_pkg_path:
 893         return None
 894 
 895     return candidate_collection_name

 905 def _get_ancestor_redirect(redirected_package_map, fullname):
 906     # walk the requested module's ancestor packages to see if any have been previously redirected
 907     cur_pkg = fullname
 908     while cur_pkg:
 909         cur_pkg = cur_pkg.rpartition('.')[0]
 910         ancestor_redirect = redirected_package_map.get(cur_pkg)
 911         if ancestor_redirect:
 912             # rewrite the prefix on fullname so we import the target first, then alias it
 913             redirect = ancestor_redirect + fullname[len(cur_pkg):]
 914             return redirect
 915     return None
