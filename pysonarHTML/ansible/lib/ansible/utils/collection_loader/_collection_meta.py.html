  19 def _meta_yml_to_dict(yaml_string_data, content_id):
  20     """
  21     Converts string YAML dictionary to a Python dictionary. This function may be monkeypatched to another implementation
  22     by some tools (eg the import sanity test).
  23     :param yaml_string_data: a bytes-ish YAML dictionary
  24     :param content_id: a unique ID representing the content to allow other implementations to cache the output
  25     :return: a Python dictionary representing the YAML dictionary content
  26     """
  27     # NB: content_id is passed in, but not used by this implementation
  28     routing_dict = load(yaml_string_data, Loader=SafeLoader)
  29     if not routing_dict:
  30         routing_dict = {}
  31     if not isinstance(routing_dict, Mapping):
  32         raise ValueError('collection metadata must be an instance of Python Mapping')
  33     return routing_dict
