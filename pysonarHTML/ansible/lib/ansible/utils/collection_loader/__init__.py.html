  13 def resource_from_fqcr(ref):
  14     """
  15     Return resource from a fully-qualified collection reference,
  16     or from a simple resource name.
  17     For fully-qualified collection references, this is equivalent to
  18     ``AnsibleCollectionRef.from_fqcr(ref).resource``.
  19     :param ref: collection reference to parse
  20     :return: the resource as a unicode string
  21     """
  22     ref = to_text(ref, errors='strict')
  23     return ref.split(u'.')[-1]
