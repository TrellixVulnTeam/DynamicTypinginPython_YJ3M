  33 class AnsibleJ2Vars(Mapping):
  34     '''
  35     Helper class to template all variable content before jinja2 sees it. This is
  36     done by hijacking the variable storage that jinja2 uses, and overriding __contains__
  37     and __getitem__ to look like a dict. Added bonus is avoiding duplicating the large
  38     hashes that inject tends to be.
  39 
  40     To facilitate using builtin jinja2 things like range, globals are also handled here.
  41     '''
  42 
  43     def __init__(self, templar, globals, locals=None, *extras):
  44         '''
  45         Initializes this object with a valid Templar() object, as
  46         well as several dictionaries of variables representing
  47         different scopes (in jinja2 terminology).
  48         '''
  49 
  50         self._templar = templar
  51         self._globals = globals
  52         self._extras = extras
  53         self._locals = dict()
  54         if isinstance(locals, dict):
  55             for key, val in iteritems(locals):
  56                 if val is not missing:
  57                     if key[:2] == 'l_':
  58                         self._locals[key[2:]] = val
  59                     elif key not in ('context', 'environment', 'template'):
  60                         self._locals[key] = val
  61 
  62     def __contains__(self, k):
  63         if k in self._templar.available_variables:
  64             return True
  65         if k in self._locals:
  66             return True
  67         for i in self._extras:
  68             if k in i:
  69                 return True
  70         if k in self._globals:
  71             return True
  72         return False
  73 
  74     def __iter__(self):
  75         keys = set()
  76         keys.update(self._templar.available_variables, self._locals, self._globals, *self._extras)
  77         return iter(keys)
  78 
  79     def __len__(self):
  80         keys = set()
  81         keys.update(self._templar.available_variables, self._locals, self._globals, *self._extras)
  82         return len(keys)
  83 
  84     def __getitem__(self, varname):
  85         if varname not in self._templar.available_variables:
  86             if varname in self._locals:
  87                 return self._locals[varname]
  88             for i in self._extras:
  89                 if varname in i:
  90                     return i[varname]
  91             if varname in self._globals:
  92                 return self._globals[varname]
  93             else:
  94                 raise KeyError("undefined variable: %s" % varname)
  95 
  96         variable = self._templar.available_variables[varname]
  97 
  98         # HostVars is special, return it as-is, as is the special variable
  99         # 'vars', which contains the vars structure
 100         from ansible.vars.hostvars import HostVars
 101         if isinstance(variable, dict) and varname == "vars" or isinstance(variable, HostVars) or hasattr(variable, '__UNSAFE__'):
 102             return variable
 103         else:
 104             value = None
 105             try:
 106                 value = self._templar.template(variable)
 107             except AnsibleUndefinedVariable:
 108                 raise
 109             except Exception as e:
 110                 msg = getattr(e, 'message', None) or to_native(e)
 111                 raise AnsibleError("An unhandled exception occurred while templating '%s'. "
 112                                    "Error was a %s, original message: %s" % (to_native(variable), type(e), msg))
 113 
 114             return value
 115 
 116     def add_locals(self, locals):
 117         '''
 118         If locals are provided, create a copy of self containing those
 119         locals in addition to what is already in this variable proxy.
 120         '''
 121         if locals is None:
 122             return self
 123 
 124         # FIXME run this only on jinja2>=2.9?
 125         # prior to version 2.9, locals contained all of the vars and not just the current
 126         # local vars so this was not necessary for locals to propagate down to nested includes
 127         new_locals = self._locals.copy()
 128         new_locals.update(locals)
 129 
 130         return AnsibleJ2Vars(self._templar, self._globals, locals=new_locals, *self._extras)
