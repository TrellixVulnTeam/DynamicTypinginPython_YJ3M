 101 def generate_ansible_template_vars(path, dest_path=None):
 102     b_path = to_bytes(path)
 103     try:
 104         template_uid = pwd.getpwuid(os.stat(b_path).st_uid).pw_name
 105     except (KeyError, TypeError):
 106         template_uid = os.stat(b_path).st_uid
 107 
 108     temp_vars = {
 109         'template_host': to_text(os.uname()[1]),
 110         'template_path': path,
 111         'template_mtime': datetime.datetime.fromtimestamp(os.path.getmtime(b_path)),
 112         'template_uid': to_text(template_uid),
 113         'template_fullpath': os.path.abspath(path),
 114         'template_run_date': datetime.datetime.now(),
 115         'template_destpath': to_native(dest_path) if dest_path else None,
 116     }
 117 
 118     managed_default = C.DEFAULT_MANAGED_STR
 119     managed_str = managed_default.format(
 120         host=temp_vars['template_host'],
 121         uid=temp_vars['template_uid'],
 122         file=temp_vars['template_path'],
 123     )
 124     temp_vars['ansible_managed'] = to_text(time.strftime(to_native(managed_str), time.localtime(os.path.getmtime(b_path))))
 125 
 126     return temp_vars

 129 def _escape_backslashes(data, jinja_env):
 130     """Double backslashes within jinja2 expressions
 131 
 132     A user may enter something like this in a playbook::
 133 
 134       debug:
 135         msg: "Test Case 1\\3; {{ test1_name | regex_replace('^(.*)_name$', '\\1')}}"
 136 
 137     The string inside of the {{ gets interpreted multiple times First by yaml.
 138     Then by python.  And finally by jinja2 as part of it's variable.  Because
 139     it is processed by both python and jinja2, the backslash escaped
 140     characters get unescaped twice.  This means that we'd normally have to use
 141     four backslashes to escape that.  This is painful for playbook authors as
 142     they have to remember different rules for inside vs outside of a jinja2
 143     expression (The backslashes outside of the "{{ }}" only get processed by
 144     yaml and python.  So they only need to be escaped once).  The following
 145     code fixes this by automatically performing the extra quoting of
 146     backslashes inside of a jinja2 expression.
 147 
 148     """
 149     if '\\' in data and '{{' in data:
 150         new_data = []
 151         d2 = jinja_env.preprocess(data)
 152         in_var = False
 153 
 154         for token in jinja_env.lex(d2):
 155             if token[1] == 'variable_begin':
 156                 in_var = True
 157                 new_data.append(token[2])
 158             elif token[1] == 'variable_end':
 159                 in_var = False
 160                 new_data.append(token[2])
 161             elif in_var and token[1] == 'string':
 162                 # Double backslashes only if we're inside of a jinja2 variable
 163                 new_data.append(token[2].replace('\\', '\\\\'))
 164             else:
 165                 new_data.append(token[2])
 166 
 167         data = ''.join(new_data)
 168 
 169     return data

 172 def is_template(data, jinja_env):
 173     """This function attempts to quickly detect whether a value is a jinja2
 174     template. To do so, we look for the first 2 matching jinja2 tokens for
 175     start and end delimiters.
 176     """
 177     found = None
 178     start = True
 179     comment = False
 180     d2 = jinja_env.preprocess(data)
 181 
 182     # This wraps a lot of code, but this is due to lex returing a generator
 183     # so we may get an exception at any part of the loop
 184     try:
 185         for token in jinja_env.lex(d2):
 186             if token[1] in JINJA2_BEGIN_TOKENS:
 187                 if start and token[1] == 'comment_begin':
 188                     # Comments can wrap other token types
 189                     comment = True
 190                 start = False
 191                 # Example: variable_end -> variable
 192                 found = token[1].split('_')[0]
 193             elif token[1] in JINJA2_END_TOKENS:
 194                 if token[1].split('_')[0] == found:
 195                     return True
 196                 elif comment:
 197                     continue
 198                 return False
 199     except TemplateSyntaxError:
 200         return False
 201 
 202     return False

 205 def _count_newlines_from_end(in_str):
 206     '''
 207     Counts the number of newlines at the end of a string. This is used during
 208     the jinja2 templating to ensure the count matches the input, since some newlines
 209     may be thrown away during the templating.
 210     '''
 211 
 212     try:
 213         i = len(in_str)
 214         j = i - 1
 215         while in_str[j] == '\n':
 216             j -= 1
 217         return i - 1 - j
 218     except IndexError:
 219         # Uncommon cases: zero length string and string containing only newlines
 220         return i

 223 def recursive_check_defined(item):
 224     from jinja2.runtime import Undefined
 225 
 226     if isinstance(item, MutableMapping):
 227         for key in item:
 228             recursive_check_defined(item[key])
 229     elif isinstance(item, list):
 230         for i in item:
 231             recursive_check_defined(i)
 232     else:
 233         if isinstance(item, Undefined):
 234             raise AnsibleFilterError("{0} is undefined".format(item))

 238     """Helper method to determine if something is an unrolled generator,
 239     iterator, or similar object
 240     """

 243         isinstance(value, MappingView) or
 244         isinstance(value, RANGE_TYPE)

 248 def _unroll_iterator(func):
 249     """Wrapper function, that intercepts the result of a filter
 250     and auto unrolls a generator, so that users are not required to
 251     explicitly use ``|list`` to unroll.
 252     """
 253     def wrapper(*args, **kwargs):
 254         ret = func(*args, **kwargs)
 255         if _is_rolled(ret):
 256             return list(ret)
 257         return ret
 258 
 259     # This code is duplicated from ``functools.update_wrapper`` from Py3.7.
 260     # ``functools.update_wrapper`` was failing when the func was ``functools.partial``
 261     for attr in ('__module__', '__name__', '__qualname__', '__doc__', '__annotations__'):
 262         try:
 263             value = getattr(func, attr)
 264         except AttributeError:
 265             pass
 266         else:
 267             setattr(wrapper, attr, value)
 268     for attr in ('__dict__',):
 269         getattr(wrapper, attr).update(getattr(func, attr, {}))
 270     wrapper.__wrapped__ = func
 271     return wrapper

 274 class AnsibleUndefined(StrictUndefined):
 275     '''
 276     A custom Undefined class, which returns further Undefined objects on access,
 277     rather than throwing an exception.
 278     '''
 279     def __getattr__(self, name):
 280         if name == '__UNSAFE__':
 281             # AnsibleUndefined should never be assumed to be unsafe
 282             # This prevents ``hasattr(val, '__UNSAFE__')`` from evaluating to ``True``
 283             raise AttributeError(name)
 284         # Return original Undefined object to preserve the first failure context
 285         return self
 286 
 287     def __getitem__(self, key):
 288         # Return original Undefined object to preserve the first failure context
 289         return self
 290 
 291     def __repr__(self):
 292         return 'AnsibleUndefined'

 295 class AnsibleContext(Context):
 296     '''
 297     A custom context, which intercepts resolve() calls and sets a flag
 298     internally if any variable lookup returns an AnsibleUnsafe value. This
 299     flag is checked post-templating, and (when set) will result in the
 300     final templated result being wrapped in AnsibleUnsafe.
 301     '''
 302     def __init__(self, *args, **kwargs):
 303         super(AnsibleContext, self).__init__(*args, **kwargs)
 304         self.unsafe = False
 305 
 306     def _is_unsafe(self, val):
 307         '''
 308         Our helper function, which will also recursively check dict and
 309         list entries due to the fact that they may be repr'd and contain
 310         a key or value which contains jinja2 syntax and would otherwise
 311         lose the AnsibleUnsafe value.
 312         '''
 313         if isinstance(val, dict):
 314             for key in val.keys():
 315                 if self._is_unsafe(val[key]):
 316                     return True
 317         elif isinstance(val, list):
 318             for item in val:
 319                 if self._is_unsafe(item):
 320                     return True
 321         elif getattr(val, '__UNSAFE__', False) is True:
 322             return True
 323         return False
 324 
 325     def _update_unsafe(self, val):
 326         if val is not None and not self.unsafe and self._is_unsafe(val):
 327             self.unsafe = True
 328 
 329     def resolve(self, key):
 330         '''
 331         The intercepted resolve(), which uses the helper above to set the
 332         internal flag whenever an unsafe variable value is returned.
 333         '''
 334         val = super(AnsibleContext, self).resolve(key)
 335         self._update_unsafe(val)
 336         return val
 337 
 338     def resolve_or_missing(self, key):
 339         val = super(AnsibleContext, self).resolve_or_missing(key)
 340         self._update_unsafe(val)
 341         return val

 343     def get_all(self):
 344         """Return the complete context as a dict including the exported
 345         variables. For optimizations reasons this might not return an
 346         actual copy so be careful with using it.
 347 
 348         This is to prevent from running ``AnsibleJ2Vars`` through dict():
 349 
 350             ``dict(self.parent, **self.vars)``
 351 
 352         In Ansible this means that ALL variables would be templated in the
 353         process of re-creating the parent because ``AnsibleJ2Vars`` templates
 354         each variable in its ``__getitem__`` method. Instead we re-create the
 355         parent via ``AnsibleJ2Vars.add_locals`` that creates a new
 356         ``AnsibleJ2Vars`` copy without templating each variable.
 357 
 358         This will prevent unnecessarily templating unused variables in cases
 359         like setting a local variable and passing it to {% include %}
 360         in a template.
 361 
 362         Also see ``AnsibleJ2Template``and
 363         https://github.com/pallets/jinja/commit/d67f0fd4cc2a4af08f51f4466150d49da7798729
 364         """
 365         if LooseVersion(j2_version) >= LooseVersion('2.9'):
 366             if not self.vars:
 367                 return self.parent
 368             if not self.parent:
 369                 return self.vars
 370 
 371         if isinstance(self.parent, AnsibleJ2Vars):
 372             return self.parent.add_locals(self.vars)
 373         else:
 374             # can this happen in Ansible?
 375             return dict(self.parent, **self.vars)

 378 class JinjaPluginIntercept(MutableMapping):
 379     def __init__(self, delegatee, pluginloader, *args, **kwargs):
 380         super(JinjaPluginIntercept, self).__init__(*args, **kwargs)
 381         self._delegatee = delegatee
 382         self._pluginloader = pluginloader
 383 
 384         if self._pluginloader.class_name == 'FilterModule':
 385             self._method_map_name = 'filters'
 386             self._dirname = 'filter'
 387         elif self._pluginloader.class_name == 'TestModule':
 388             self._method_map_name = 'tests'
 389             self._dirname = 'test'
 390 
 391         self._collection_jinja_func_cache = {}
 392 
 393     # FUTURE: we can cache FQ filter/test calls for the entire duration of a run, since a given collection's impl's
 394     # aren't supposed to change during a run
 395     def __getitem__(self, key):
 396         try:
 397             if not isinstance(key, string_types):
 398                 raise ValueError('key must be a string')
 399 
 400             key = to_native(key)
 401 
 402             if '.' not in key:  # might be a built-in or legacy, check the delegatee dict first, then try for a last-chance base redirect
 403                 func = self._delegatee.get(key)
 404 
 405                 if func:
 406                     return func
 407 
 408                 # didn't find it in the pre-built Jinja env, assume it's a former builtin and follow the normal routing path
 409                 leaf_key = key
 410                 key = 'ansible.builtin.' + key
 411             else:
 412                 leaf_key = key.split('.')[-1]
 413 
 414             acr = AnsibleCollectionRef.try_parse_fqcr(key, self._dirname)
 415 
 416             if not acr:
 417                 raise KeyError('invalid plugin name: {0}'.format(key))
 418 
 419             ts = _get_collection_metadata(acr.collection)
 420 
 421             # TODO: implement support for collection-backed redirect (currently only builtin)
 422             # TODO: implement cycle detection (unified across collection redir as well)
 423 
 424             routing_entry = ts.get('plugin_routing', {}).get(self._dirname, {}).get(leaf_key, {})
 425 
 426             deprecation_entry = routing_entry.get('deprecation')
 427             if deprecation_entry:
 428                 warning_text = deprecation_entry.get('warning_text')
 429                 removal_date = deprecation_entry.get('removal_date')
 430                 removal_version = deprecation_entry.get('removal_version')
 431 
 432                 if not warning_text:
 433                     warning_text = '{0} "{1}" is deprecated'.format(self._dirname, key)
 434 
 435                 display.deprecated(warning_text, version=removal_version, date=removal_date, collection_name=acr.collection)
 436 
 437             tombstone_entry = routing_entry.get('tombstone')
 438 
 439             if tombstone_entry:
 440                 warning_text = tombstone_entry.get('warning_text')
 441                 removal_date = tombstone_entry.get('removal_date')
 442                 removal_version = tombstone_entry.get('removal_version')
 443 
 444                 if not warning_text:
 445                     warning_text = '{0} "{1}" has been removed'.format(self._dirname, key)
 446 
 447                 exc_msg = display.get_deprecation_message(warning_text, version=removal_version, date=removal_date,
 448                                                           collection_name=acr.collection, removed=True)
 449 
 450                 raise AnsiblePluginRemovedError(exc_msg)
 451 
 452             redirect_fqcr = routing_entry.get('redirect', None)
 453             if redirect_fqcr:
 454                 acr = AnsibleCollectionRef.from_fqcr(ref=redirect_fqcr, ref_type=self._dirname)
 455                 display.vvv('redirecting {0} {1} to {2}.{3}'.format(self._dirname, key, acr.collection, acr.resource))
 456                 key = redirect_fqcr
 457             # TODO: handle recursive forwarding (not necessary for builtin, but definitely for further collection redirs)
 458 
 459             func = self._collection_jinja_func_cache.get(key)
 460 
 461             if func:
 462                 return func
 463 
 464             try:
 465                 pkg = import_module(acr.n_python_package_name)
 466             except ImportError:
 467                 raise KeyError()
 468 
 469             parent_prefix = acr.collection
 470 
 471             if acr.subdirs:
 472                 parent_prefix = '{0}.{1}'.format(parent_prefix, acr.subdirs)
 473 
 474             # TODO: implement collection-level redirect
 475 
 476             for dummy, module_name, ispkg in pkgutil.iter_modules(pkg.__path__, prefix=parent_prefix + '.'):
 477                 if ispkg:
 478                     continue
 479 
 480                 try:
 481                     plugin_impl = self._pluginloader.get(module_name)
 482                 except Exception as e:
 483                     raise TemplateSyntaxError(to_native(e), 0)
 484 
 485                 method_map = getattr(plugin_impl, self._method_map_name)
 486 
 487                 for f in iteritems(method_map()):
 488                     fq_name = '.'.join((parent_prefix, f[0]))
 489                     # FIXME: detect/warn on intra-collection function name collisions
 490                     self._collection_jinja_func_cache[fq_name] = _unroll_iterator(f[1])
 491 
 492             function_impl = self._collection_jinja_func_cache[key]
 493             return function_impl
 494         except AnsiblePluginRemovedError as apre:
 495             raise TemplateSyntaxError(to_native(apre), 0)
 496         except KeyError:
 497             raise
 498         except Exception as ex:
 499             display.warning('an unexpected error occurred during Jinja2 environment setup: {0}'.format(to_native(ex)))
 500             display.vvv('exception during Jinja2 environment setup: {0}'.format(format_exc()))
 501             raise TemplateSyntaxError(to_native(ex), 0)
 502 
 503     def __setitem__(self, key, value):
 504         return self._delegatee.__setitem__(key, value)
 505

 507         raise NotImplementedError()

 510         # not strictly accurate since we're not counting dynamically-loaded values
 511         return iter(self._delegatee)
 512

 514         # not strictly accurate since we're not counting dynamically-loaded values

 668     @contextmanager
 669     def set_temporary_context(self, **kwargs):
 670         """Context manager used to set temporary templating context, without having to worry about resetting
 671         original values afterward
 672 
 673         Use a keyword that maps to the attr you are setting. Applies to ``self.environment`` by default, to
 674         set context on another object, it must be in ``mapping``.
 675         """
 676         mapping = {
 677             'available_variables': self,
 678             'searchpath': self.environment.loader,
 679         }
 680         original = {}
 681 
 682         for key, value in kwargs.items():
 683             obj = mapping.get(key, self.environment)
 684             try:
 685                 original[key] = getattr(obj, key)
 686                 if value is not None:
 687                     setattr(obj, key, value)
 688             except AttributeError:
 689                 # Ignore invalid attrs, lstrip_blocks was added in jinja2==2.7
 690                 pass
 691 
 692         yield
 693 
 694         for key in original:
 695             obj = mapping.get(key, self.environment)
 696             setattr(obj, key, original[key])

 698     def template(self, variable, convert_bare=False, preserve_trailing_newlines=True, escape_backslashes=True, fail_on_undefined=None, overrides=None,
 699                  convert_data=True, static_vars=None, cache=True, disable_lookups=False):
 700         '''
 701         Templates (possibly recursively) any given data as input. If convert_bare is
 702         set to True, the given data will be wrapped as a jinja2 variable ('{{foo}}')
 703         before being sent through the template engine.
 704         '''
 705         static_vars = [''] if static_vars is None else static_vars
 706 
 707         # Don't template unsafe variables, just return them.
 708         if hasattr(variable, '__UNSAFE__'):
 709             return variable
 710 
 711         if fail_on_undefined is None:
 712             fail_on_undefined = self._fail_on_undefined_errors
 713 
 714         try:
 715             if convert_bare:
 716                 variable = self._convert_bare_variable(variable)
 717 
 718             if isinstance(variable, string_types):
 719                 result = variable
 720 
 721                 if self.is_possibly_template(variable):
 722                     # Check to see if the string we are trying to render is just referencing a single
 723                     # var.  In this case we don't want to accidentally change the type of the variable
 724                     # to a string by using the jinja template renderer. We just want to pass it.
 725                     only_one = self.SINGLE_VAR.match(variable)
 726                     if only_one:
 727                         var_name = only_one.group(1)
 728                         if var_name in self._available_variables:
 729                             resolved_val = self._available_variables[var_name]
 730                             if isinstance(resolved_val, NON_TEMPLATED_TYPES):
 731                                 return resolved_val
 732                             elif resolved_val is None:
 733                                 return C.DEFAULT_NULL_REPRESENTATION
 734 
 735                     # Using a cache in order to prevent template calls with already templated variables
 736                     sha1_hash = None
 737                     if cache:
 738                         variable_hash = sha1(text_type(variable).encode('utf-8'))
 739                         options_hash = sha1(
 740                             (
 741                                 text_type(preserve_trailing_newlines) +
 742                                 text_type(escape_backslashes) +
 743                                 text_type(fail_on_undefined) +
 744                                 text_type(overrides)
 745                             ).encode('utf-8')
 746                         )
 747                         sha1_hash = variable_hash.hexdigest() + options_hash.hexdigest()
 748                     if cache and sha1_hash in self._cached_result:
 749                         result = self._cached_result[sha1_hash]
 750                     else:
 751                         result = self.do_template(
 752                             variable,
 753                             preserve_trailing_newlines=preserve_trailing_newlines,
 754                             escape_backslashes=escape_backslashes,
 755                             fail_on_undefined=fail_on_undefined,
 756                             overrides=overrides,
 757                             disable_lookups=disable_lookups,
 758                         )
 759 
 760                         if not USE_JINJA2_NATIVE:
 761                             unsafe = hasattr(result, '__UNSAFE__')
 762                             if convert_data and not self._no_type_regex.match(variable):
 763                                 # if this looks like a dictionary or list, convert it to such using the safe_eval method
 764                                 if (result.startswith("{") and not result.startswith(self.environment.variable_start_string)) or \
 765                                         result.startswith("[") or result in ("True", "False"):
 766                                     eval_results = safe_eval(result, include_exceptions=True)
 767                                     if eval_results[1] is None:
 768                                         result = eval_results[0]
 769                                         if unsafe:
 770                                             result = wrap_var(result)
 771                                     else:
 772                                         # FIXME: if the safe_eval raised an error, should we do something with it?
 773                                         pass
 774 
 775                         # we only cache in the case where we have a single variable
 776                         # name, to make sure we're not putting things which may otherwise
 777                         # be dynamic in the cache (filters, lookups, etc.)
 778                         if cache and only_one:
 779                             self._cached_result[sha1_hash] = result
 780 
 781                 return result
 782 
 783             elif is_sequence(variable):
 784                 return [self.template(
 785                     v,
 786                     preserve_trailing_newlines=preserve_trailing_newlines,
 787                     fail_on_undefined=fail_on_undefined,
 788                     overrides=overrides,
 789                     disable_lookups=disable_lookups,
 790                 ) for v in variable]
 791             elif isinstance(variable, Mapping):
 792                 d = {}
 793                 # we don't use iteritems() here to avoid problems if the underlying dict
 794                 # changes sizes due to the templating, which can happen with hostvars
 795                 for k in variable.keys():
 796                     if k not in static_vars:
 797                         d[k] = self.template(
 798                             variable[k],
 799                             preserve_trailing_newlines=preserve_trailing_newlines,
 800                             fail_on_undefined=fail_on_undefined,
 801                             overrides=overrides,
 802                             disable_lookups=disable_lookups,
 803                         )
 804                     else:
 805                         d[k] = variable[k]
 806                 return d
 807             else:
 808                 return variable
 809 
 810         except AnsibleFilterError:
 811             if self._fail_on_filter_errors:
 812                 raise
 813             else:
 814                 return variable

 816     def is_template(self, data):
 817         '''lets us know if data has a template'''
 818         if isinstance(data, string_types):
 819             return is_template(data, self.environment)
 820         elif isinstance(data, (list, tuple)):
 821             for v in data:
 822                 if self.is_template(v):
 823                     return True
 824         elif isinstance(data, dict):
 825             for k in data:
 826                 if self.is_template(k) or self.is_template(data[k]):
 827                     return True
 828         return False

 832     def is_possibly_template(self, data):
 833         '''Determines if a string looks like a template, by seeing if it
 834         contains a jinja2 start delimiter. Does not guarantee that the string
 835         is actually a template.
 836 
 837         This is different than ``is_template`` which is more strict.
 838         This method may return ``True`` on a string that is not templatable.
 839 
 840         Useful when guarding passing a string for templating, but when
 841         you want to allow the templating engine to make the final
 842         assessment which may result in ``TemplateSyntaxError``.
 843         '''
 844         env = self.environment
 845         if isinstance(data, string_types):
 846             for marker in (env.block_start_string, env.variable_start_string, env.comment_start_string):
 847                 if marker in data:
 848                     return True
 849         return False

 851     def _convert_bare_variable(self, variable):
 852         '''
 853         Wraps a bare string, which may have an attribute portion (ie. foo.bar)
 854         in jinja2 variable braces so that it is evaluated properly.
 855         '''
 856 
 857         if isinstance(variable, string_types):
 858             contains_filters = "|" in variable
 859             first_part = variable.split("|")[0].split(".")[0].split("[")[0]
 860             if (contains_filters or first_part in self._available_variables) and self.environment.variable_start_string not in variable:
 861                 return "%s%s%s" % (self.environment.variable_start_string, variable, self.environment.variable_end_string)
 862 
 863         # the variable didn't meet the conditions to be converted,
 864         # so just return it as-is
 865         return variable

 867     def _finalize(self, thing):
 868         '''
 869         A custom finalize method for jinja2, which prevents None from being returned. This
 870         avoids a string of ``"None"`` as ``None`` has no importance in YAML.
 871 
 872         If using ANSIBLE_JINJA2_NATIVE we bypass this and return the actual value always
 873         '''
 874         if _is_rolled(thing):
 875             # Auto unroll a generator, so that users are not required to
 876             # explicitly use ``|list`` to unroll
 877             # This only affects the scenario where the final result of templating
 878             # is a generator, and not where a filter creates a generator in the middle
 879             # of a template. See ``_unroll_iterator`` for the other case. This is probably
 880             # unncessary
 881             return list(thing)
 882 
 883         if USE_JINJA2_NATIVE:
 884             return thing
 885 
 886         return thing if thing is not None else ''

 908     def _lookup(self, name, *args, **kwargs):
 909         instance = self._lookup_loader.get(name, loader=self._loader, templar=self)
 910 
 911         if instance is not None:
 912             wantlist = kwargs.pop('wantlist', False)
 913             allow_unsafe = kwargs.pop('allow_unsafe', C.DEFAULT_ALLOW_UNSAFE_LOOKUPS)
 914             errors = kwargs.pop('errors', 'strict')
 915 
 916             from ansible.utils.listify import listify_lookup_plugin_terms
 917             loop_terms = listify_lookup_plugin_terms(terms=args, templar=self, loader=self._loader, fail_on_undefined=True, convert_bare=False)
 918             # safely catch run failures per #5059
 919             try:
 920                 ran = instance.run(loop_terms, variables=self._available_variables, **kwargs)
 921             except (AnsibleUndefinedVariable, UndefinedError) as e:
 922                 raise AnsibleUndefinedVariable(e)
 923             except Exception as e:
 924                 if self._fail_on_lookup_errors:
 925                     msg = u"An unhandled exception occurred while running the lookup plugin '%s'. Error was a %s, original message: %s" % \
 926                           (name, type(e), to_text(e))
 927                     if errors == 'warn':
 928                         display.warning(msg)
 929                     elif errors == 'ignore':
 930                         display.display(msg, log_only=True)
 931                     else:
 932                         raise AnsibleError(to_native(msg))
 933                 ran = [] if wantlist else None
 934 
 935             if ran and not allow_unsafe:
 936                 if wantlist:
 937                     ran = wrap_var(ran)
 938                 else:
 939                     try:
 940                         ran = wrap_var(",".join(ran))
 941                     except TypeError:
 942                         # Lookup Plugins should always return lists.  Throw an error if that's not
 943                         # the case:
 944                         if not isinstance(ran, Sequence):
 945                             raise AnsibleError("The lookup plugin '%s' did not return a list."
 946                                                % name)
 947 
 948                         # The TypeError we can recover from is when the value *inside* of the list
 949                         # is not a string
 950                         if len(ran) == 1:
 951                             ran = wrap_var(ran[0])
 952                         else:
 953                             ran = wrap_var(ran)
 954 
 955                 if self.cur_context:
 956                     self.cur_context.unsafe = True
 957             return ran
 958         else:
 959             raise AnsibleError("lookup plugin (%s) not found" % name)

 961     def do_template(self, data, preserve_trailing_newlines=True, escape_backslashes=True, fail_on_undefined=None, overrides=None, disable_lookups=False):
 962         if USE_JINJA2_NATIVE and not isinstance(data, string_types):
 963             return data
 964 
 965         # For preserving the number of input newlines in the output (used
 966         # later in this method)
 967         data_newlines = _count_newlines_from_end(data)
 968 
 969         if fail_on_undefined is None:
 970             fail_on_undefined = self._fail_on_undefined_errors
 971 
 972         try:
 973             # allows template header overrides to change jinja2 options.
 974             if overrides is None:
 975                 myenv = self.environment.overlay()
 976             else:
 977                 myenv = self.environment.overlay(overrides)
 978 
 979             # Get jinja env overrides from template
 980             if hasattr(data, 'startswith') and data.startswith(JINJA2_OVERRIDE):
 981                 eol = data.find('\n')
 982                 line = data[len(JINJA2_OVERRIDE):eol]
 983                 data = data[eol + 1:]
 984                 for pair in line.split(','):
 985                     (key, val) = pair.split(':')
 986                     key = key.strip()
 987                     setattr(myenv, key, ast.literal_eval(val.strip()))
 988 
 989             # Adds Ansible custom filters and tests
 990             myenv.filters.update(self._get_filters())
 991             for k in myenv.filters:
 992                 myenv.filters[k] = _unroll_iterator(myenv.filters[k])
 993             myenv.tests.update(self._get_tests())
 994 
 995             if escape_backslashes:
 996                 # Allow users to specify backslashes in playbooks as "\\" instead of as "\\\\".
 997                 data = _escape_backslashes(data, myenv)
 998 
 999             try:
1000                 t = myenv.from_string(data)
1001             except TemplateSyntaxError as e:
1002                 raise AnsibleError("template error while templating string: %s. String: %s" % (to_native(e), to_native(data)))
1003             except Exception as e:
1004                 if 'recursion' in to_native(e):
1005                     raise AnsibleError("recursive loop detected in template string: %s" % to_native(data))
1006                 else:
1007                     return data
1008 
1009             if disable_lookups:
1010                 t.globals['query'] = t.globals['q'] = t.globals['lookup'] = self._fail_lookup
1011 
1012             jvars = AnsibleJ2Vars(self, t.globals)
1013 
1014             self.cur_context = new_context = t.new_context(jvars, shared=True)
1015             rf = t.root_render_func(new_context)
1016 
1017             try:
1018                 res = j2_concat(rf)
1019                 if getattr(new_context, 'unsafe', False):
1020                     res = wrap_var(res)
1021             except TypeError as te:
1022                 if 'AnsibleUndefined' in to_native(te):
1023                     errmsg = "Unable to look up a name or access an attribute in template string (%s).\n" % to_native(data)
1024                     errmsg += "Make sure your variable name does not contain invalid characters like '-': %s" % to_native(te)
1025                     raise AnsibleUndefinedVariable(errmsg)
1026                 else:
1027                     display.debug("failing because of a type error, template data is: %s" % to_text(data))
1028                     raise AnsibleError("Unexpected templating type error occurred on (%s): %s" % (to_native(data), to_native(te)))
1029 
1030             if USE_JINJA2_NATIVE and not isinstance(res, string_types):
1031                 return res
1032 
1033             if preserve_trailing_newlines:
1034                 # The low level calls above do not preserve the newline
1035                 # characters at the end of the input data, so we use the
1036                 # calculate the difference in newlines and append them
1037                 # to the resulting output for parity
1038                 #
1039                 # jinja2 added a keep_trailing_newline option in 2.7 when
1040                 # creating an Environment.  That would let us make this code
1041                 # better (remove a single newline if
1042                 # preserve_trailing_newlines is False).  Once we can depend on
1043                 # that version being present, modify our code to set that when
1044                 # initializing self.environment and remove a single trailing
1045                 # newline here if preserve_newlines is False.
1046                 res_newlines = _count_newlines_from_end(res)
1047                 if data_newlines > res_newlines:
1048                     res += self.environment.newline_sequence * (data_newlines - res_newlines)
1049             return res
1050         except (UndefinedError, AnsibleUndefinedVariable) as e:
1051             if fail_on_undefined:
1052                 raise AnsibleUndefinedVariable(e)
1053             else:
1054                 display.debug("Ignoring undefined failure: %s" % to_text(e))
1055                 return data
