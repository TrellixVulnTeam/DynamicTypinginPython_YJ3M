  21 def ansible_native_concat(nodes):
  22     """Return a native Python type from the list of compiled nodes. If the
  23     result is a single node, its value is returned. Otherwise, the nodes are
  24     concatenated as strings. If the result can be parsed with
  25     :func:`ast.literal_eval`, the parsed value is returned. Otherwise, the
  26     string is returned.
  27     """
  28 
  29     # https://github.com/pallets/jinja/blob/master/jinja2/nativetypes.py
  30 
  31     head = list(islice(nodes, 2))
  32 
  33     if not head:
  34         return None
  35 
  36     if len(head) == 1:
  37         out = head[0]
  38 
  39         # TODO send unvaulted data to literal_eval?
  40         if isinstance(out, AnsibleVaultEncryptedUnicode):
  41             return out.data
  42 
  43         if isinstance(out, StrictUndefined):
  44             # A hack to raise proper UndefinedError/AnsibleUndefinedVariable exception.
  45             # We need to access the AnsibleUndefined(StrictUndefined) object by either of the following:
  46             # __iter__, __str__, __len__, __nonzero__, __eq__, __ne__, __bool__, __hash__
  47             # to actually raise the exception.
  48             # (see Jinja2 source of StrictUndefined to get up to date info)
  49             # Otherwise the undefined error would be raised on the next access which might not be properly handled.
  50             # See https://github.com/ansible/ansible/issues/52158
  51             # We do that only here because it is taken care of by to_text() in the else block below already.
  52             str(out)
  53     else:
  54         if isinstance(nodes, types.GeneratorType):
  55             nodes = chain(head, nodes)
  56         # Stringifying the nodes is important as it takes care of
  57         # StrictUndefined by side-effect - by raising an exception.
  58         out = u''.join([to_text(v) for v in nodes])
  59 
  60     try:
  61         out = literal_eval(out)
  62         if PY2:
  63             # ensure bytes are not returned back into Ansible from templating
  64             out = container_to_text(out)
  65         return out
  66     except (ValueError, SyntaxError, MemoryError):
  67         return out
