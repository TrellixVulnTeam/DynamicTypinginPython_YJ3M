  27 class AnsibleJ2Template(jinja2.environment.Template):
  28     '''
  29     A helper class, which prevents Jinja2 from running AnsibleJ2Vars through dict().
  30     Without this, {% include %} and similar will create new contexts unlike the special
  31     one created in Templar.template. This ensures they are all alike, except for
  32     potential locals.
  33     '''
  34 
  35     def new_context(self, vars=None, shared=False, locals=None):
  36         if vars is not None:
  37             if isinstance(vars, dict):
  38                 vars = vars.copy()
  39                 if locals is not None:
  40                     vars.update(locals)
  41             else:
  42                 vars = vars.add_locals(locals)
  43         return self.environment.context_class(self.environment, vars, self.name, self.blocks)
