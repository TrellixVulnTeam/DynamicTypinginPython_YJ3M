  46 class AnsibleOutputPrimaryLexer(RegexLexer):
  47     name = 'Ansible-output-primary'
  48 
  49     # The following definitions are borrowed from Pygment's JSON lexer.
  50     # It has been originally authored by Norman Richards.
  51 
  52     # integer part of a number
  53     int_part = r'-?(0|[1-9]\d*)'
  54 
  55     # fractional part of a number
  56     frac_part = r'\.\d+'
  57 
  58     # exponential part of a number
  59     exp_part = r'[eE](\+|-)?\d+'
  60 
  61     tokens = {
  62         # #########################################
  63         # # BEGIN: states from JSON lexer #########
  64         # #########################################
  65         'whitespace': [
  66             (r'\s+', token.Text),
  67         ],
  68 
  69         # represents a simple terminal value
  70         'simplevalue': [
  71             (r'(true|false|null)\b', token.Keyword.Constant),
  72             (('%(int_part)s(%(frac_part)s%(exp_part)s|'
  73               '%(exp_part)s|%(frac_part)s)') % vars(),
  74              token.Number.Float),
  75             (int_part, token.Number.Integer),
  76             (r'"(\\\\|\\"|[^"])*"', token.String),
  77         ],
  78 
  79 
  80         # the right hand side of an object, after the attribute name
  81         'objectattribute': [
  82             include('value'),
  83             (r':', token.Punctuation),
  84             # comma terminates the attribute but expects more
  85             (r',', token.Punctuation, '#pop'),
  86             # a closing bracket terminates the entire object, so pop twice
  87             (r'\}', token.Punctuation, '#pop:2'),
  88         ],
  89 
  90         # a json object - { attr, attr, ... }
  91         'objectvalue': [
  92             include('whitespace'),
  93             (r'"(\\\\|\\"|[^"])*"', token.Name.Tag, 'objectattribute'),
  94             (r'\}', token.Punctuation, '#pop'),
  95         ],
  96 
  97         # json array - [ value, value, ... }
  98         'arrayvalue': [
  99             include('whitespace'),
 100             include('value'),
 101             (r',', token.Punctuation),
 102             (r'\]', token.Punctuation, '#pop'),
 103         ],
 104 
 105         # a json value - either a simple value or a complex value (object or array)
 106         'value': [
 107             include('whitespace'),
 108             include('simplevalue'),
 109             (r'\{', token.Punctuation, 'objectvalue'),
 110             (r'\[', token.Punctuation, 'arrayvalue'),
 111         ],
 112         # #########################################
 113         # # END: states from JSON lexer ###########
 114         # #########################################
 115 
 116         'host-postfix': [
 117             (r'\n', token.Text, '#pop:3'),
 118             (r'( )(=>)( )(\{)',
 119                 bygroups(token.Text, token.Punctuation, token.Text, token.Punctuation),
 120                 'objectvalue'),
 121         ],
 122 
 123         'host-error': [
 124             (r'(?:(:)( )(UNREACHABLE|FAILED)(!))?',
 125                 bygroups(token.Punctuation, token.Text, token.Keyword, token.Punctuation),
 126                 'host-postfix'),
 127             (r'', token.Text, 'host-postfix'),
 128         ],
 129 
 130         'host-name': [
 131             (r'(\[)([^ \]]+)(?:( )(=>)( )([^\]]+))?(\])',
 132                 bygroups(token.Punctuation, token.Name.Variable, token.Text, token.Punctuation, token.Text, token.Name.Variable, token.Punctuation),
 133                 'host-error')
 134         ],
 135 
 136         'host-result': [
 137             (r'\n', token.Text, '#pop'),
 138             (r'( +)(ok|changed|failed|skipped|unreachable)(=)([0-9]+)',
 139                 bygroups(token.Text, token.Keyword, token.Punctuation, token.Number.Integer)),
 140         ],
 141 
 142         'root': [
 143             (r'(PLAY|TASK|PLAY RECAP)(?:( )(\[)([^\]]+)(\]))?( )(\*+)(\n)',
 144                 bygroups(token.Keyword, token.Text, token.Punctuation, token.Literal, token.Punctuation, token.Text, token.Name.Variable, token.Text)),
 145             (r'(fatal|ok|changed|skipping)(:)( )',
 146                 bygroups(token.Keyword, token.Punctuation, token.Text),
 147                 'host-name'),
 148             (r'(\[)(WARNING)(\]:)([^\n]+)',
 149                 bygroups(token.Punctuation, token.Keyword, token.Punctuation, token.Text)),
 150             (r'([^ ]+)( +)(:)',
 151                 bygroups(token.Name, token.Text, token.Punctuation),
 152                 'host-result'),
 153             (r'(\tto retry, use: )(.*)(\n)', bygroups(token.Text, token.Literal.String, token.Text)),
 154             (r'.*\n', token.Other),
 155         ],
 156     }
