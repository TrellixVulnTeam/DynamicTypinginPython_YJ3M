  20 # Purpose:
  21 # The purpose of this script is to convert uses of tests as filters to proper jinja test syntax
  22 # as part of https://github.com/ansible/proposals/issues/83
  23 
  24 # Notes:
  25 # This script is imperfect, but was close enough to "fix" all integration tests
  26 # with the exception of:
  27 #
  28 # 1. One file needed manual remediation, where \\\\ was ultimately replace with \\ in 8 locations.
  29 # 2. Multiple filter pipeline is unsupported. Example:
  30 #        var|string|search('foo')
  31 #    Which should be converted to:
  32 #        var|string is search('foo')
  33 
  34 from __future__ import (absolute_import, division, print_function)
  35 __metaclass__ = type
  36 
  37 import argparse
  38 import os
  39 import re
  40 
  41 from ansible.plugins.test import core, files, mathstuff
  42 
  43 
  44 TESTS = list(core.TestModule().tests().keys()) + list(files.TestModule().tests().keys()) + list(mathstuff.TestModule().tests().keys())
  45 
  46 
  47 TEST_MAP = {
  48     'version_compare': 'version',
  49     'is_dir': 'directory',
  50     'is_file': 'file',
  51     'is_link': 'link',
  52     'is_abs': 'abs',
  53     'is_same_file': 'same_file',
  54     'is_mount': 'mount',
  55     'issubset': 'subset',
  56     'issuperset': 'superset',
  57     'isnan': 'nan',
  58     'succeeded': 'successful',
  59     'success': 'successful',
  60     'change': 'changed',
  61     'skip': 'skipped',
  62 }
  63 
  64 
  65 FILTER_RE = re.compile(r'((.+?)\s*([\w \.\'"]+)(\s*)\|(\s*)(\w+))')
  66 NOT_RE = re.compile(r'( ?)not ')
  67 ASSERT_SPACE_RE = re.compile(r'- ([\'"])\s+')
  68 
  69 parser = argparse.ArgumentParser()
  70 parser.add_argument(
  71     'path',
  72     help='Path to a directory that will be recursively walked. All .yml and .yaml files will be evaluated '
  73          'and uses of tests as filters will be conveted to proper jinja test syntax files to have test syntax '
  74          'fixed'
  75 )
  76 args = parser.parse_args()
  77 
  78 for root, dirs, filenames in os.walk(args.path):
  79     for name in filenames:
  80         if os.path.splitext(name)[1] not in ('.yml', '.yaml'):
  81             continue
  82         path = os.path.join(root, name)
  83 
  84         print(path)
  85         with open(path) as f:
  86             text = f.read()
  87 
  88         for match in FILTER_RE.findall(text):
  89             filter_name = match[5]
  90 
  91             is_not = match[2].strip(' "\'').startswith('not ')
  92 
  93             try:
  94                 test_name = TEST_MAP[filter_name]
  95             except KeyError:
  96                 test_name = filter_name
  97 
  98             if test_name not in TESTS:
  99                 continue
 100 
 101             if is_not:
 102                 before = NOT_RE.sub(r'\1', match[2]).rstrip()
 103                 text = re.sub(
 104                     re.escape(match[0]),
 105                     '%s %s is not %s' % (match[1], before, test_name,),
 106                     text
 107                 )
 108             else:
 109                 text = re.sub(
 110                     re.escape(match[0]),
 111                     '%s %s is %s' % (match[1], match[2].rstrip(), test_name,),
 112                     text
 113                 )
 114 
 115         with open(path, 'w+') as f:
 116             f.write(text)
