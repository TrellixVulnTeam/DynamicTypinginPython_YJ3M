  80 def populate_modules():
  81     module_dir = os.path.join(BASE_PATH, 'lib/ansible/modules/')
  82 
  83     modules_rows = []
  84 
  85     for root, dir_names, file_names in os.walk(module_dir):
  86         for file_name in file_names:
  87             module, extension = os.path.splitext(file_name)
  88 
  89             if module == '__init__' or extension != '.py':
  90                 continue
  91 
  92             if module.startswith('_'):
  93                 module = module[1:]
  94 
  95             namespace = os.path.join(root.replace(module_dir, '')).replace('/', '.')
  96 
  97             path = os.path.join(root, file_name)
  98 
  99             result = read_docstring(path)
 100 
 101             doc = result['doc']
 102 
 103             modules_rows.append(dict(
 104                 module=module,
 105                 namespace=namespace,
 106                 path=path.replace(BASE_PATH, ''),
 107                 version_added=str(doc.get('version_added', '')) if doc else '',
 108             ))
 109 
 110     populate_data(dict(
 111         modules=dict(
 112             rows=modules_rows,
 113             schema=(
 114                 ('module', 'TEXT'),
 115                 ('namespace', 'TEXT'),
 116                 ('path', 'TEXT'),
 117                 ('version_added', 'TEXT'),
 118             )),
 119     ))

 122 def populate_coverage():
 123     response = open_url('https://codecov.io/api/gh/ansible/ansible/tree/devel/?src=extension')
 124     data = json.load(response)
 125     files = data['commit']['report']['files']
 126     coverage_rows = []
 127 
 128     for path, data in files.items():
 129         report = data['t']
 130         coverage_rows.append(dict(
 131             path=path,
 132             coverage=float(report['c']),
 133             lines=report['n'],
 134             hit=report['h'],
 135             partial=report['p'],
 136             missed=report['m'],
 137         ))
 138 
 139     populate_data(dict(
 140         coverage=dict(
 141             rows=coverage_rows,
 142             schema=(
 143                 ('path', 'TEXT'),
 144                 ('coverage', 'REAL'),
 145                 ('lines', 'INTEGER'),
 146                 ('hit', 'INTEGER'),
 147                 ('partial', 'INTEGER'),
 148                 ('missed', 'INTEGER'),
 149             )),
 150     ))

 153 def populate_integration_targets():
 154     targets = list(walk_integration_targets())
 155 
 156     integration_targets_rows = [dict(
 157         target=target.name,
 158         type=target.type,
 159         path=target.path,
 160         script_path=target.script_path,
 161     ) for target in targets]
 162 
 163     integration_target_aliases_rows = [dict(
 164         target=target.name,
 165         alias=alias,
 166     ) for target in targets for alias in target.aliases]
 167 
 168     integration_target_modules_rows = [dict(
 169         target=target.name,
 170         module=module,
 171     ) for target in targets for module in target.modules]
 172 
 173     populate_data(dict(
 174         integration_targets=dict(
 175             rows=integration_targets_rows,
 176             schema=(
 177                 ('target', 'TEXT'),
 178                 ('type', 'TEXT'),
 179                 ('path', 'TEXT'),
 180                 ('script_path', 'TEXT'),
 181             )),
 182         integration_target_aliases=dict(
 183             rows=integration_target_aliases_rows,
 184             schema=(
 185                 ('target', 'TEXT'),
 186                 ('alias', 'TEXT'),
 187             )),
 188         integration_target_modules=dict(
 189             rows=integration_target_modules_rows,
 190             schema=(
 191                 ('target', 'TEXT'),
 192                 ('module', 'TEXT'),
 193             )),
 194     ))

 197 def create_table(cursor, name, columns):
 198     schema = ', '.join('%s %s' % column for column in columns)
 199 
 200     cursor.execute('DROP TABLE IF EXISTS %s' % name)
 201     cursor.execute('CREATE TABLE %s (%s)' % (name, schema))
 202 
 203 
 204 def populate_table(cursor, rows, name, columns):
 205     create_table(cursor, name, columns)
 206 
 207     values = ', '.join([':%s' % column[0] for column in columns])
 208 
 209     for row in rows:
 210         cursor.execute('INSERT INTO %s VALUES (%s)' % (name, values), row)

 213 def populate_data(data):
 214     connection = sqlite3.connect(DATABASE_PATH)
 215     cursor = connection.cursor()
 216 
 217     for table in data:
 218         populate_table(cursor, data[table]['rows'], table, data[table]['schema'])
 219 
 220     connection.commit()
 221     connection.close()
