   1 #!/usr/bin/env python
   2 
   3 """
   4 This script generated test_cases for test_distribution_version.py.
   5 
   6 To do so it outputs the relevant files from /etc/*release, the output of distro.linux_distribution()
   7 and the current ansible_facts regarding the distribution version.
   8 
   9 This assumes a working ansible version in the path.
  10 """
  11 
  12 from __future__ import (absolute_import, division, print_function)
  13 __metaclass__ = type
  14 
  15 import os.path
  16 import subprocess
  17 import json
  18 import sys
  19 
  20 from ansible.module_utils import distro
  21 from ansible.module_utils._text import to_text
  22 
  23 
  24 filelist = [
  25     '/etc/oracle-release',
  26     '/etc/slackware-version',
  27     '/etc/redhat-release',
  28     '/etc/vmware-release',
  29     '/etc/openwrt_release',
  30     '/etc/system-release',
  31     '/etc/alpine-release',
  32     '/etc/release',
  33     '/etc/arch-release',
  34     '/etc/os-release',
  35     '/etc/SuSE-release',
  36     '/etc/gentoo-release',
  37     '/etc/os-release',
  38     '/etc/lsb-release',
  39     '/etc/altlinux-release',
  40     '/etc/os-release',
  41     '/etc/coreos/update.conf',
  42     '/etc/flatcar/update.conf',
  43     '/usr/lib/os-release',
  44 ]
  45 
  46 fcont = {}
  47 
  48 for f in filelist:
  49     if os.path.exists(f):
  50         s = os.path.getsize(f)
  51         if s > 0 and s < 10000:
  52             with open(f) as fh:
  53                 fcont[f] = fh.read()
  54 
  55 dist = distro.linux_distribution(full_distribution_name=False)
  56 
  57 facts = ['distribution', 'distribution_version', 'distribution_release', 'distribution_major_version', 'os_family']
  58 
  59 try:
  60     b_ansible_out = subprocess.check_output(
  61         ['ansible', 'localhost', '-m', 'setup'])
  62 except subprocess.CalledProcessError as e:
  63     print("ERROR: ansible run failed, output was: \n")
  64     print(e.output)
  65     sys.exit(e.returncode)
  66 
  67 ansible_out = to_text(b_ansible_out)
  68 parsed = json.loads(ansible_out[ansible_out.index('{'):])
  69 ansible_facts = {}
  70 for fact in facts:
  71     try:
  72         ansible_facts[fact] = parsed['ansible_facts']['ansible_' + fact]
  73     except Exception:
  74         ansible_facts[fact] = "N/A"
  75 
  76 nicename = ansible_facts['distribution'] + ' ' + ansible_facts['distribution_version']
  77 
  78 output = {
  79     'name': nicename,
  80     'distro': {
  81         'codename': distro.codename(),
  82         'id': distro.id(),
  83         'name': distro.name(),
  84         'version': distro.version(),
  85         'version_best': distro.version(best=True),
  86         'lsb_release_info': distro.lsb_release_info(),
  87         'os_release_info': distro.os_release_info(),
  88     },
  89     'input': fcont,
  90     'platform.dist': dist,
  91     'result': ansible_facts,
  92 }
  93 
  94 print(json.dumps(output, indent=4))
