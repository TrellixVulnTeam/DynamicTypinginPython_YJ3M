 150 async def calculate_hash_from_tarball(session, version):
 151     tar_url = f'https://pypi.python.org/packages/source/a/ansible-base/ansible-base-{version}.tar.gz'
 152     tar_task = asyncio.create_task(session.get(tar_url))
 153     tar_response = await tar_task
 154 
 155     tar_hash = hashlib.sha256()
 156     while True:
 157         chunk = await tar_response.content.read(1024)
 158         if not chunk:
 159             break
 160         tar_hash.update(chunk)
 161 
 162     return tar_hash.hexdigest()

 165 async def parse_hash_from_file(session, version):
 166     filename = f'ansible-base-{version}.tar.gz'
 167     hash_url = f'https://releases.ansible.com/ansible-base/{filename}.sha'
 168     hash_task = asyncio.create_task(session.get(hash_url))
 169     hash_response = await hash_task
 170 
 171     hash_content = await hash_response.read()
 172     precreated_hash, precreated_filename = hash_content.split(None, 1)
 173     if filename != precreated_filename.strip().decode('utf-8'):
 174         raise ValueError(f'Hash file contains hash for a different file: {precreated_filename}')
 175 
 176     return precreated_hash.decode('utf-8')

 179 async def get_hash(session, version):
 180     calculated_hash = await calculate_hash_from_tarball(session, version)
 181     precreated_hash = await parse_hash_from_file(session, version)
 182 
 183     if calculated_hash != precreated_hash:
 184         raise ValueError(f'Hash in file ansible-base-{version}.tar.gz.sha {precreated_hash} does not'
 185                          f' match hash of tarball from pypi {calculated_hash}')
 186 
 187     return calculated_hash

 190 async def get_hashes(versions):
 191     hashes = {}
 192     requestors = {}
 193     async with aiohttp.ClientSession() as aio_session:
 194         for version in versions:
 195             requestors[version] = asyncio.create_task(get_hash(aio_session, version))
 196 
 197         for version, request in requestors.items():
 198             await request
 199             hashes[version] = request.result()
 200 
 201     return hashes

 204 def next_release_date(weeks=3):
 205     days_in_the_future = weeks * 7
 206     today = datetime.datetime.now()
 207     numeric_today = today.weekday()
 208 
 209     # We release on Thursdays
 210     if numeric_today == 3:
 211         # 3 is Thursday
 212         pass
 213     elif numeric_today == 4:
 214         # If this is Friday, we can adjust back to Thursday for the next release
 215         today -= datetime.timedelta(days=1)
 216     elif numeric_today < 3:
 217         # Otherwise, slide forward to Thursday
 218         today += datetime.timedelta(days=(3 - numeric_today))
 219     else:
 220         # slightly different formula if it's past Thursday this week.  We need to go forward to
 221         # Thursday of next week
 222         today += datetime.timedelta(days=(10 - numeric_today))
 223 
 224     next_release = today + datetime.timedelta(days=days_in_the_future)
 225     return next_release

 228 def is_ansible_base(version):
 229     '''
 230     Determines if a version is an ansible-base version or not, by checking
 231     if it is >= 2.10.0. Stops comparing when it gets to the first non-numeric
 232     component to allow for .dev and .beta suffixes.
 233     '''
 234     # Ignore .beta/.dev suffixes
 235     ver_split = []
 236     for component in version.split('.'):
 237         if not component.isdigit():
 238             # Take everything up until the first non-numeric component
 239             break
 240         ver_split.append(int(component))
 241     return tuple(ver_split) >= (2, 10, 0)

 244 # Currently only use with a single element list, but left general for later
 245 # in case we need to refer to the releases collectively.
 246 def release_variants(versions):
 247     if all(is_ansible_base(v) for v in versions):
 248         return 'ansible-base'
 249 
 250     if all(not is_ansible_base(v) for v in versions):
 251         return 'Ansible'
 252 
 253     return 'Ansible and ansible-base'

 256 def pretty_version(version):
 257     return '{0} {1}'.format(
 258         release_variants([version]),
 259         version,
 260     )

 263 def create_long_message(versions, name):
 264     hashes = asyncio.run(get_hashes(versions))
 265 
 266     version_template = JINJA_ENV.get_template('version_string')
 267     version_str = version_template.render(versions=versions,
 268                                           pretty_version=pretty_version).strip()
 269 
 270     next_release = next_release_date()
 271 
 272     template = JINJA_ENV.get_template('long')
 273     message = template.render(versions=versions, version_str=version_str,
 274                               name=name, hashes=hashes, next_release=next_release,
 275                               is_ansible_base=is_ansible_base,
 276                               pretty_version=pretty_version)
 277     return message

 280 def create_short_message(versions):
 281     version_template = JINJA_ENV.get_template('version_string')
 282     version_str = version_template.render(versions=versions,
 283                                           pretty_version=pretty_version).strip()
 284 
 285     template = JINJA_ENV.get_template('short')
 286     message = template.render(versions=versions, version_str=version_str,
 287                               is_ansible_base=is_ansible_base,
 288                               pretty_version=pretty_version)
 289     message = ' '.join(message.split()) + '\n'
 290     return message
