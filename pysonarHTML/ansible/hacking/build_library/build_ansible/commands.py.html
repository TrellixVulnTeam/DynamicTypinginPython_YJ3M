  10 from abc import ABCMeta, abstractmethod, abstractproperty
  11 
  12 
  13 class Command(metaclass=ABCMeta):
  14     """
  15     Subcommands of :program:`build-ansible.py`.
  16 
  17     This defines an interface that all subcommands must conform to.  :program:`build-ansible.py`
  18     will require that these things are present in order to proceed.
  19     """
  20     @staticmethod
  21     @abstractproperty
  22     def name():
  23         """Name of the subcommand.  It's the string to invoked it via on the command line"""
  24 
  25     @staticmethod
  26     @abstractmethod
  27     def init_parser(add_parser):
  28         """
  29         Initialize and register an argparse ArgumentParser
  30 
  31         :arg add_parser: function which creates an ArgumentParser for the main program.
  32 
  33         Implementations should first create an ArgumentParser using `add_parser` and then populate
  34         it with the command line arguments that are needed.
  35 
  36         .. seealso:
  37             `add_parser` information in the :py:meth:`ArgumentParser.add_subparsers` documentation.
  38         """
  39 
  40     @staticmethod
  41     @abstractmethod
  42     def main(arguments):
  43         """
  44         Run the command
  45 
  46         :arg arguments: The **parsed** command line args
  47 
  48         This is the Command's entrypoint.  The command line args are already parsed but from here
  49         on, the command can do its work.
  50         """
