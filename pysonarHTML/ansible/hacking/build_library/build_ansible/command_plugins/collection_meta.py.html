  25 DEFAULT_TEMPLATE_DIR = pathlib.Path(__file__).parents[4] / 'docs/templates'

  28 def normalize_options(options):
  29     """Normalize the options to make for easy templating"""
  30     for opt in options:
  31         if isinstance(opt['description'], string_types):
  32             opt['description'] = [opt['description']]

  35 class DocumentCollectionMeta(Command):
  36     name = 'collection-meta'
  37 
  38     @classmethod
  39     def init_parser(cls, add_parser):
  40         parser = add_parser(cls.name, description='Generate collection galaxy.yml documentation from shared metadata')
  41         parser.add_argument("-t", "--template-file", action="store", dest="template_file",
  42                             default=DEFAULT_TEMPLATE_FILE,
  43                             help="Jinja2 template to use for the config")
  44         parser.add_argument("-T", "--template-dir", action="store", dest="template_dir",
  45                             default=str(DEFAULT_TEMPLATE_DIR),
  46                             help="directory containing Jinja2 templates")
  47         parser.add_argument("-o", "--output-dir", action="store", dest="output_dir", default='/tmp/',
  48                             help="Output directory for rst files")
  49         parser.add_argument("collection_defs", metavar="COLLECTION-OPTION-DEFINITIONS.yml", type=str,
  50                             help="Source for collection metadata option docs")
  51 
  52     @staticmethod
  53     def main(args):
  54         output_dir = os.path.abspath(args.output_dir)
  55         template_file_full_path = os.path.abspath(os.path.join(args.template_dir, args.template_file))
  56         template_file = os.path.basename(template_file_full_path)
  57         template_dir = os.path.dirname(template_file_full_path)
  58 
  59         with open(args.collection_defs) as f:
  60             options = yaml.safe_load(f)
  61 
  62         normalize_options(options)
  63 
  64         env = Environment(loader=FileSystemLoader(template_dir),
  65                           variable_start_string="@{",
  66                           variable_end_string="}@",
  67                           trim_blocks=True)
  68         env.filters['documented_type'] = documented_type
  69         env.filters['rst_ify'] = rst_ify
  70 
  71         template = env.get_template(template_file)
  72         output_name = os.path.join(output_dir, template_file.replace('.j2', ''))
  73         temp_vars = {'options': options}
  74 
  75         data = to_bytes(template.render(temp_vars))
  76         update_file_if_different(output_name, data)
  77 
  78         return 0
