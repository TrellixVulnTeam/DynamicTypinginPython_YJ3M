  27 # from https://www.python.org/dev/peps/pep-0257/
  28 def trim_docstring(docstring):
  29     if not docstring:
  30         return ''
  31     # Convert tabs to spaces (following the normal Python rules)
  32     # and split into a list of lines:
  33     lines = docstring.expandtabs().splitlines()
  34     # Determine minimum indentation (first line doesn't count):
  35     indent = sys.maxsize
  36     for line in lines[1:]:
  37         stripped = line.lstrip()
  38         if stripped:
  39             indent = min(indent, len(line) - len(stripped))
  40     # Remove indentation (first line is special):
  41     trimmed = [lines[0].strip()]
  42     if indent < sys.maxsize:
  43         for line in lines[1:]:
  44             trimmed.append(line[indent:].rstrip())
  45     # Strip off trailing and leading blank lines:
  46     while trimmed and not trimmed[-1]:
  47         trimmed.pop()
  48     while trimmed and not trimmed[0]:
  49         trimmed.pop(0)
  50     # Return a single string:
  51     return '\n'.join(trimmed)

  54 def get_options(optlist):
  55     ''' get actual options '''
  56 
  57     opts = []
  58     for opt in optlist:
  59         res = {
  60             'desc': opt.help,
  61             'options': opt.option_strings
  62         }
  63         if isinstance(opt, argparse._StoreAction):
  64             res['arg'] = opt.dest.upper()
  65         elif not res['options']:
  66             continue
  67         opts.append(res)
  68 
  69     return opts

  72 def dedupe_groups(parser):
  73     action_groups = []
  74     for action_group in parser._action_groups:
  75         found = False
  76         for a in action_groups:
  77             if a._actions == action_group._actions:
  78                 found = True
  79                 break
  80         if not found:
  81             action_groups.append(action_group)
  82     return action_groups
  83

  85 def get_option_groups(option_parser):
  86     groups = []
  87     for action_group in dedupe_groups(option_parser)[1:]:
  88         group_info = {}
  89         group_info['desc'] = action_group.description
  90         group_info['options'] = action_group._actions
  91         group_info['group_obj'] = action_group
  92         groups.append(group_info)
  93     return groups

  96 def opt_doc_list(parser):
  97     ''' iterate over options lists '''
  98 
  99     results = []
 100     for option_group in dedupe_groups(parser)[1:]:
 101         results.extend(get_options(option_group._actions))
 102 
 103     results.extend(get_options(parser._actions))
 104 
 105     return results

 109 def opts_docs(cli_class_name, cli_module_name):
 110     ''' generate doc structure from options '''
 111 
 112     cli_name = 'ansible-%s' % cli_module_name
 113     if cli_module_name == 'adhoc':
 114         cli_name = 'ansible'
 115 
 116     # WIth no action/subcommand
 117     # shared opts set
 118     # instantiate each cli and ask its options
 119     cli_klass = getattr(__import__("ansible.cli.%s" % cli_module_name,
 120                                    fromlist=[cli_class_name]), cli_class_name)
 121     cli = cli_klass([cli_name])
 122 
 123     # parse the common options
 124     try:
 125         cli.init_parser()
 126     except Exception:
 127         pass
 128 
 129     # base/common cli info
 130     docs = {
 131         'cli': cli_module_name,
 132         'cli_name': cli_name,
 133         'usage': cli.parser.format_usage(),
 134         'short_desc': cli.parser.description,
 135         'long_desc': trim_docstring(cli.__doc__),
 136         'actions': {},
 137         'content_depth': 2,
 138     }
 139     option_info = {'option_names': [],
 140                    'options': [],
 141                    'groups': []}
 142 
 143     for extras in ('ARGUMENTS'):
 144         if hasattr(cli, extras):
 145             docs[extras.lower()] = getattr(cli, extras)
 146 
 147     common_opts = opt_doc_list(cli.parser)
 148     groups_info = get_option_groups(cli.parser)
 149     shared_opt_names = []
 150     for opt in common_opts:
 151         shared_opt_names.extend(opt.get('options', []))
 152 
 153     option_info['options'] = common_opts
 154     option_info['option_names'] = shared_opt_names
 155 
 156     option_info['groups'].extend(groups_info)
 157 
 158     docs.update(option_info)
 159 
 160     # now for each action/subcommand
 161     # force populate parser with per action options
 162 
 163     def get_actions(parser, docs):
 164         # use class attrs not the attrs on a instance (not that it matters here...)
 165         try:
 166             subparser = parser._subparsers._group_actions[0].choices
 167         except AttributeError:
 168             subparser = {}
 169 
 170         depth = 0
 171 
 172         for action, parser in subparser.items():
 173             action_info = {'option_names': [],
 174                            'options': [],
 175                            'actions': {}}
 176             # docs['actions'][action] = {}
 177             # docs['actions'][action]['name'] = action
 178             action_info['name'] = action
 179             action_info['desc'] = trim_docstring(getattr(cli, 'execute_%s' % action).__doc__)
 180 
 181             # docs['actions'][action]['desc'] = getattr(cli, 'execute_%s' % action).__doc__.strip()
 182             action_doc_list = opt_doc_list(parser)
 183 
 184             uncommon_options = []
 185             for action_doc in action_doc_list:
 186                 # uncommon_options = []
 187 
 188                 option_aliases = action_doc.get('options', [])
 189                 for option_alias in option_aliases:
 190 
 191                     if option_alias in shared_opt_names:
 192                         continue
 193 
 194                     # TODO: use set
 195                     if option_alias not in action_info['option_names']:
 196                         action_info['option_names'].append(option_alias)
 197 
 198                     if action_doc in action_info['options']:
 199                         continue
 200 
 201                     uncommon_options.append(action_doc)
 202 
 203                 action_info['options'] = uncommon_options
 204 
 205             depth = 1 + get_actions(parser, action_info)
 206 
 207             docs['actions'][action] = action_info
 208 
 209         return depth
 210 
 211     action_depth = get_actions(cli.parser, docs)
 212     docs['content_depth'] = action_depth + 1
 213 
 214     docs['options'] = opt_doc_list(cli.parser)
 215     return docs

 218 class GenerateMan(Command):
 219     name = 'generate-man'
 220 
 221     @classmethod
 222     def init_parser(cls, add_parser):
 223         parser = add_parser(name=cls.name,
 224                             description='Generate cli documentation from cli docstrings')
 225 
 226         parser.add_argument("-t", "--template-file", action="store", dest="template_file",
 227                             default=DEFAULT_TEMPLATE_FILE, help="path to jinja2 template")
 228         parser.add_argument("-o", "--output-dir", action="store", dest="output_dir",
 229                             default='/tmp/', help="Output directory for rst files")
 230         parser.add_argument("-f", "--output-format", action="store", dest="output_format",
 231                             default='man',
 232                             help="Output format for docs (the default 'man' or 'rst')")
 233         parser.add_argument('cli_modules', help='CLI module name(s)', metavar='MODULE_NAME', nargs='*')
 234 
 235     @staticmethod
 236     def main(args):
 237         template_file = args.template_file
 238         template_path = os.path.expanduser(template_file)
 239         template_dir = os.path.abspath(os.path.dirname(template_path))
 240         template_basename = os.path.basename(template_file)
 241 
 242         output_dir = os.path.abspath(args.output_dir)
 243         output_format = args.output_format
 244 
 245         cli_modules = args.cli_modules
 246 
 247         # various cli parsing things checks sys.argv if the 'args' that are passed in are []
 248         # so just remove any args so the cli modules dont try to parse them resulting in warnings
 249         sys.argv = [sys.argv[0]]
 250 
 251         allvars = {}
 252         output = {}
 253         cli_list = []
 254         cli_bin_name_list = []
 255 
 256         # for binary in os.listdir('../../lib/ansible/cli'):
 257         for cli_module_name in cli_modules:
 258             binary = os.path.basename(os.path.expanduser(cli_module_name))
 259 
 260             if not binary.endswith('.py'):
 261                 continue
 262             elif binary == '__init__.py':
 263                 continue
 264 
 265             cli_name = os.path.splitext(binary)[0]
 266 
 267             if cli_name == 'adhoc':
 268                 cli_class_name = 'AdHocCLI'
 269                 # myclass = 'AdHocCLI'
 270                 output[cli_name] = 'ansible.1.rst.in'
 271                 cli_bin_name = 'ansible'
 272             else:
 273                 # myclass = "%sCLI" % libname.capitalize()
 274                 cli_class_name = "%sCLI" % cli_name.capitalize()
 275                 output[cli_name] = 'ansible-%s.1.rst.in' % cli_name
 276                 cli_bin_name = 'ansible-%s' % cli_name
 277 
 278             # FIXME:
 279             allvars[cli_name] = opts_docs(cli_class_name, cli_name)
 280             cli_bin_name_list.append(cli_bin_name)
 281 
 282         cli_list = allvars.keys()
 283 
 284         doc_name_formats = {'man': '%s.1.rst.in',
 285                             'rst': '%s.rst'}
 286 
 287         for cli_name in cli_list:
 288 
 289             # template it!
 290             env = Environment(loader=FileSystemLoader(template_dir))
 291             template = env.get_template(template_basename)
 292 
 293             # add rest to vars
 294             tvars = allvars[cli_name]
 295             tvars['cli_list'] = cli_list
 296             tvars['cli_bin_name_list'] = cli_bin_name_list
 297             tvars['cli'] = cli_name
 298             if '-i' in tvars['options']:
 299                 print('uses inventory')
 300 
 301             manpage = template.render(tvars)
 302             filename = os.path.join(output_dir, doc_name_formats[output_format] % tvars['cli_name'])
 303             update_file_if_different(filename, to_bytes(manpage))
