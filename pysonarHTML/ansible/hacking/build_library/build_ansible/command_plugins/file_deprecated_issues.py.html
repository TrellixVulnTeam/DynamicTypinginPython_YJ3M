  26 def get_token(token_file):
  27     if token_file:
  28         return token_file.read().strip()
  29 
  30     token = os.getenv('GITHUB_TOKEN').strip()
  31     if not token:
  32         raise errors.MissingUserInput(
  33             'Please provide a file containing a github oauth token with public_repo scope'
  34             ' via the --github-token argument or set the GITHUB_TOKEN env var with your'
  35             ' github oauth token'
  36         )
  37     return token

  40 def parse_deprecations(problems_file_handle):
  41     deprecated = defaultdict(list)
  42     deprecation_errors = problems_file_handle.read()
  43     for line in deprecation_errors.splitlines():
  44         path = line.split(':')[0]
  45         if path.endswith('__init__.py'):
  46             component = os.path.basename(os.path.dirname(path))
  47         else:
  48             component, dummy = os.path.splitext(os.path.basename(path).lstrip('_'))
  49 
  50         title = (
  51             '%s contains deprecated call to be removed in %s' %
  52             (component, ANSIBLE_MAJOR_VERSION)
  53         )
  54         deprecated[component].append(
  55             dict(title=title, path=path, line=line)
  56         )
  57     return deprecated

  60 def find_project_todo_column(repo, project_name):
  61     project = None
  62     for project in repo.projects():
  63         if project.name.lower() == project_name:
  64             break
  65     else:
  66         raise errors.InvalidUserInput('%s was an invalid project name' % project_name)
  67 
  68     for project_column in project.columns():
  69         column_name = project_column.name.lower()
  70         if 'todo' in column_name or 'backlog' in column_name or 'to do' in column_name:
  71             return project_column
  72 
  73     raise Exception('Unable to determine the todo column in'
  74                     ' project %s' % project_name)

  77 def create_issues(deprecated, body_tmpl, repo):
  78     issues = []
  79 
  80     for component, items in deprecated.items():
  81         title = items[0]['title']
  82         path = '\n'.join(set((i['path']) for i in items))
  83         line = '\n'.join(i['line'] for i in items)
  84         body = body_tmpl % dict(component=component, path=path,
  85                                 line=line,
  86                                 version=ANSIBLE_MAJOR_VERSION)
  87 
  88         issue = repo.create_issue(title, body=body, labels=['deprecated'])
  89         print(issue)
  90         issues.append(issue)
  91 
  92         # Sleep a little, so that the API doesn't block us
  93         time.sleep(0.5)
  94 
  95     return issues

  98 class FileDeprecationTickets(Command):
  99     name = 'file-deprecation-tickets'
 100 
 101     @classmethod
 102     def init_parser(cls, add_parser):
 103         parser = add_parser(cls.name, description='File tickets to cleanup deprecated features for'
 104                             ' the next release')
 105         parser.add_argument('--template', default='deprecated_issue_template.md',
 106                             type=argparse.FileType('r'),
 107                             help='Path to markdown file template to be used for issue '
 108                                  'body. Default: %(default)s')
 109         parser.add_argument('--project-name', default='', type=str,
 110                             help='Name of a github project to assign all issues to')
 111         parser.add_argument('--github-token', type=argparse.FileType('r'),
 112                             help='Path to file containing a github token with public_repo scope.'
 113                                  ' This token in this file will be used to open the deprcation'
 114                                  ' tickets and add them to the github project.  If not given,'
 115                                  ' the GITHUB_TOKEN environment variable will be tried')
 116         parser.add_argument('problems', type=argparse.FileType('r'),
 117                             help='Path to file containing pylint output for the '
 118                                  'ansible-deprecated-version check')
 119 
 120     @staticmethod
 121     def main(args):
 122         try:
 123             from github3 import GitHub
 124         except ImportError:
 125             raise errors.DependencyError(
 126                 'This command needs the github3.py library installed to work'
 127             )
 128 
 129         token = get_token(args.github_token)
 130         args.github_token.close()
 131 
 132         deprecated = parse_deprecations(args.problems)
 133         args.problems.close()
 134 
 135         body_tmpl = args.template.read()
 136         args.template.close()
 137 
 138         project_name = args.project_name.strip().lower()
 139 
 140         gh_conn = GitHub(token=token)
 141         repo = gh_conn.repository('abadger', 'ansible')
 142 
 143         if project_name:
 144             project_column = find_project_todo_column(repo, project_name)
 145 
 146         issues = create_issues(deprecated, body_tmpl, repo)
 147 
 148         if project_column:
 149             for issue in issues:
 150                 project_column.create_card_with_issue(issue)
 151                 time.sleep(0.5)
 152 
 153         return 0
