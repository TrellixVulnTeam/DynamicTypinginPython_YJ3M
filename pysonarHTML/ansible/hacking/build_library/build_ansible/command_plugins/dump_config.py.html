  26 def fix_description(config_options):
  27     '''some descriptions are strings, some are lists. workaround it...'''
  28 
  29     for config_key in config_options:
  30         description = config_options[config_key].get('description', [])
  31         if isinstance(description, list):
  32             desc_list = description
  33         else:
  34             desc_list = [description]
  35         config_options[config_key]['description'] = desc_list
  36     return config_options

  39 class DocumentConfig(Command):
  40     name = 'document-config'
  41 
  42     @classmethod
  43     def init_parser(cls, add_parser):
  44         parser = add_parser(cls.name, description='Generate module documentation from metadata')
  45         parser.add_argument("-t", "--template-file", action="store", dest="template_file",
  46                             default=DEFAULT_TEMPLATE_FILE,
  47                             help="Jinja2 template to use for the config")
  48         parser.add_argument("-T", "--template-dir", action="store", dest="template_dir",
  49                             default=str(DEFAULT_TEMPLATE_DIR),
  50                             help="directory containing Jinja2 templates")
  51         parser.add_argument("-o", "--output-dir", action="store", dest="output_dir", default='/tmp/',
  52                             help="Output directory for rst files")
  53         parser.add_argument("config_defs", metavar="CONFIG-OPTION-DEFINITIONS.yml", type=str,
  54                             help="Source for config option docs")
  55 
  56     @staticmethod
  57     def main(args):
  58         output_dir = os.path.abspath(args.output_dir)
  59         template_file_full_path = os.path.abspath(os.path.join(args.template_dir, args.template_file))
  60         template_file = os.path.basename(template_file_full_path)
  61         template_dir = os.path.dirname(template_file_full_path)
  62 
  63         with open(args.config_defs) as f:
  64             config_options = yaml.safe_load(f)
  65 
  66         config_options = fix_description(config_options)
  67 
  68         env = Environment(loader=FileSystemLoader(template_dir), trim_blocks=True,)
  69         template = env.get_template(template_file)
  70         output_name = os.path.join(output_dir, template_file.replace('.j2', ''))
  71         temp_vars = {'config_options': config_options}
  72 
  73         data = to_bytes(template.render(temp_vars))
  74         update_file_if_different(output_name, data)
  75 
  76         return 0
