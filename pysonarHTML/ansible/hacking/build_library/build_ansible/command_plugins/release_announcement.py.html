  19 class VersionStr(UserString):
  20     def __init__(self, string):
  21         super().__init__(string.strip())
  22         self.ver_obj = LooseVersion(string)

  25 def transform_args(args):
  26     # Make it possible to sort versions in the jinja2 templates
  27     new_versions = []
  28     for version in args.versions:
  29         new_versions.append(VersionStr(version))
  30     args.versions = new_versions
  31 
  32     return args

  35 def write_message(filename, message):
  36     if filename != '-':
  37         with open(filename, 'w') as out_file:
  38             out_file.write(message)
  39     else:
  40         sys.stdout.write('\n\n')
  41         sys.stdout.write(message)

  44 class ReleaseAnnouncementCommand(Command):
  45     name = 'release-announcement'
  46 
  47     @classmethod
  48     def init_parser(cls, add_parser):
  49         parser = add_parser(cls.name,
  50                             description="Generate email and twitter announcements from template")
  51 
  52         parser.add_argument("--version", dest="versions", type=str, required=True, action='append',
  53                             help="Versions of Ansible to announce")
  54         parser.add_argument("--name", type=str, required=True, help="Real name to use on emails")
  55         parser.add_argument("--email-out", type=str, default="-",
  56                             help="Filename to place the email announcement into")
  57         parser.add_argument("--twitter-out", type=str, default="-",
  58                             help="Filename to place the twitter announcement into")
  59 
  60     @classmethod
  61     def main(cls, args):
  62         if sys.version_info < (3, 6):
  63             raise errors.DependencyError('The {0} subcommand needs Python-3.6+'
  64                                          ' to run'.format(cls.name))
  65 
  66         # Import here because these functions are invalid on Python-3.5 and the command plugins and
  67         # init_parser() method need to be compatible with Python-3.4+ for now.
  68         # Pylint doesn't understand Python3 namespace modules.
  69         from .. announce import create_short_message, create_long_message  # pylint: disable=relative-beyond-top-level
  70 
  71         args = transform_args(args)
  72 
  73         twitter_message = create_short_message(args.versions)
  74         email_message = create_long_message(args.versions, args.name)
  75 
  76         write_message(args.twitter_out, twitter_message)
  77         write_message(args.email_out, email_message)
  78         return 0
