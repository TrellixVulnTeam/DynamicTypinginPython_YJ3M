  26 DEFAULT_TEMPLATE_DIR = str(pathlib.Path(__file__).resolve().parents[4] / 'docs/templates')
  27 TEMPLATE_FILE = 'playbooks_keywords.rst.j2'
  28 PLAYBOOK_CLASS_NAMES = ['Play', 'Role', 'Block', 'Task']

  31 def load_definitions(keyword_definitions_file):
  32     docs = {}
  33     with open(keyword_definitions_file) as f:
  34         docs = yaml.safe_load(f)
  35 
  36     return docs

  39 def extract_keywords(keyword_definitions):
  40     pb_keywords = {}
  41     for pb_class_name in PLAYBOOK_CLASS_NAMES:
  42         if pb_class_name == 'Play':
  43             module_name = 'ansible.playbook'
  44         else:
  45             module_name = 'ansible.playbook.{0}'.format(pb_class_name.lower())
  46         module = importlib.import_module(module_name)
  47         playbook_class = getattr(module, pb_class_name, None)
  48         if playbook_class is None:
  49             raise ImportError("We weren't able to import the module {0}".format(module_name))
  50 
  51         # Maintain order of the actual class names for our output
  52         # Build up a mapping of playbook classes to the attributes that they hold
  53         pb_keywords[pb_class_name] = {k: v for (k, v) in playbook_class._valid_attrs.items()
  54                                       # Filter private attributes as they're not usable in playbooks
  55                                       if not v.private}
  56 
  57         # pick up definitions if they exist
  58         for keyword in tuple(pb_keywords[pb_class_name]):
  59             if keyword in keyword_definitions:
  60                 pb_keywords[pb_class_name][keyword] = keyword_definitions[keyword]
  61             else:
  62                 # check if there is an alias, otherwise undocumented
  63                 alias = getattr(getattr(playbook_class, '_%s' % keyword), 'alias', None)
  64                 if alias and alias in keyword_definitions:
  65                     pb_keywords[pb_class_name][alias] = keyword_definitions[alias]
  66                     del pb_keywords[pb_class_name][keyword]
  67                 else:
  68                     pb_keywords[pb_class_name][keyword] = ' UNDOCUMENTED!! '
  69 
  70         # loop is really with_ for users
  71         if pb_class_name == 'Task':
  72             pb_keywords[pb_class_name]['with_<lookup_plugin>'] = (
  73                 'The same as ``loop`` but magically adds the output of any lookup plugin to'
  74                 ' generate the item list.')
  75 
  76         # local_action is implicit with action
  77         if 'action' in pb_keywords[pb_class_name]:
  78             pb_keywords[pb_class_name]['local_action'] = ('Same as action but also implies'
  79                                                           ' ``delegate_to: localhost``')
  80 
  81     return pb_keywords

  84 def generate_page(pb_keywords, template_dir):
  85     env = Environment(loader=FileSystemLoader(template_dir), trim_blocks=True,)
  86     template = env.get_template(TEMPLATE_FILE)
  87     tempvars = {'pb_keywords': pb_keywords, 'playbook_class_names': PLAYBOOK_CLASS_NAMES}
  88 
  89     keyword_page = template.render(tempvars)
  90     if LooseVersion(jinja2.__version__) < LooseVersion('2.10'):
  91         # jinja2 < 2.10's indent filter indents blank lines.  Cleanup
  92         keyword_page = re.sub(' +\n', '\n', keyword_page)
  93 
  94     return keyword_page

  97 class DocumentKeywords(Command):
  98     name = 'document-keywords'
  99 
 100     @classmethod
 101     def init_parser(cls, add_parser):
 102         parser = add_parser(cls.name, description='Generate playbook keyword documentation from'
 103                             ' code and descriptions')
 104         parser.add_argument("-T", "--template-dir", action="store", dest="template_dir",
 105                             default=DEFAULT_TEMPLATE_DIR,
 106                             help="directory containing Jinja2 templates")
 107         parser.add_argument("-o", "--output-dir", action="store", dest="output_dir",
 108                             default='/tmp/', help="Output directory for rst files")
 109         parser.add_argument("keyword_defs", metavar="KEYWORD-DEFINITIONS.yml", type=str,
 110                             help="Source for playbook keyword docs")
 111 
 112     @staticmethod
 113     def main(args):
 114         keyword_definitions = load_definitions(args.keyword_defs)
 115         pb_keywords = extract_keywords(keyword_definitions)
 116 
 117         keyword_page = generate_page(pb_keywords, args.template_dir)
 118         outputname = os.path.join(args.output_dir, TEMPLATE_FILE.replace('.j2', ''))
 119         update_file_if_different(outputname, to_bytes(keyword_page))
 120 
 121         return 0
