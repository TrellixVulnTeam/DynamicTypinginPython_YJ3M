   8 def update_file_if_different(filename, b_data):
   9     """
  10     Replace file content only if content is different.
  11 
  12     This preserves timestamps in case the file content has not changed.  It performs multiple
  13     operations on the file so it is not atomic and may be slower than simply writing to the file.
  14 
  15     :arg filename: The filename to write to
  16     :b_data: Byte string containing the data to write to the file
  17     """
  18     try:
  19         with open(filename, 'rb') as f:
  20             b_data_old = f.read()
  21     except IOError as e:
  22         if e.errno != 2:
  23             raise
  24         # File did not exist, set b_data_old to a sentinel value so that
  25         # b_data gets written to the filename
  26         b_data_old = None
  27 
  28     if b_data_old != b_data:
  29         with open(filename, 'wb') as f:
  30             f.write(b_data)
  31         return True
  32 
  33     return False
