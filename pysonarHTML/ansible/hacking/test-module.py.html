  55 def parse():
  56     """parse command line
  57 
  58     :return : (options, args)"""
  59     parser = optparse.OptionParser()
  60 
  61     parser.usage = "%prog -[options] (-h for help)"
  62 
  63     parser.add_option('-m', '--module-path', dest='module_path',
  64                       help="REQUIRED: full path of module source to execute")
  65     parser.add_option('-a', '--args', dest='module_args', default="",
  66                       help="module argument string")
  67     parser.add_option('-D', '--debugger', dest='debugger',
  68                       help="path to python debugger (e.g. /usr/bin/pdb)")
  69     parser.add_option('-I', '--interpreter', dest='interpreter',
  70                       help="path to interpreter to use for this module"
  71                       " (e.g. ansible_python_interpreter=/usr/bin/python)",
  72                       metavar='INTERPRETER_TYPE=INTERPRETER_PATH',
  73                       default="ansible_python_interpreter=%s" %
  74                       (sys.executable if sys.executable else '/usr/bin/python'))
  75     parser.add_option('-c', '--check', dest='check', action='store_true',
  76                       help="run the module in check mode")
  77     parser.add_option('-n', '--noexecute', dest='execute', action='store_false',
  78                       default=True, help="do not run the resulting module")
  79     parser.add_option('-o', '--output', dest='filename',
  80                       help="Filename for resulting module",
  81                       default="~/.ansible_module_generated")
  82     options, args = parser.parse_args()
  83     if not options.module_path:
  84         parser.print_help()
  85         sys.exit(1)
  86     else:
  87         return options, args

  90 def write_argsfile(argstring, json=False):
  91     """ Write args to a file for old-style module's use. """
  92     argspath = os.path.expanduser("~/.ansible_test_module_arguments")
  93     argsfile = open(argspath, 'w')
  94     if json:
  95         args = parse_kv(argstring)
  96         argstring = jsonify(args)
  97     argsfile.write(argstring)
  98     argsfile.close()
  99     return argspath

 102 def get_interpreters(interpreter):
 103     result = dict()
 104     if interpreter:
 105         if '=' not in interpreter:
 106             print("interpreter must by in the form of ansible_python_interpreter=/usr/bin/python")
 107             sys.exit(1)
 108         interpreter_type, interpreter_path = interpreter.split('=')
 109         if not interpreter_type.startswith('ansible_'):
 110             interpreter_type = 'ansible_%s' % interpreter_type
 111         if not interpreter_type.endswith('_interpreter'):
 112             interpreter_type = '%s_interpreter' % interpreter_type
 113         result[interpreter_type] = interpreter_path
 114     return result

 117 def boilerplate_module(modfile, args, interpreters, check, destfile):
 118     """ simulate what ansible does with new style modules """
 119 
 120     # module_fh = open(modfile)
 121     # module_data = module_fh.read()
 122     # module_fh.close()
 123 
 124     # replacer = module_common.ModuleReplacer()
 125     loader = DataLoader()
 126 
 127     # included_boilerplate = module_data.find(module_common.REPLACER) != -1 or module_data.find("import ansible.module_utils") != -1
 128 
 129     complex_args = {}
 130 
 131     # default selinux fs list is pass in as _ansible_selinux_special_fs arg
 132     complex_args['_ansible_selinux_special_fs'] = C.DEFAULT_SELINUX_SPECIAL_FS
 133     complex_args['_ansible_tmpdir'] = C.DEFAULT_LOCAL_TMP
 134     complex_args['_ansible_keep_remote_files'] = C.DEFAULT_KEEP_REMOTE_FILES
 135     complex_args['_ansible_version'] = __version__
 136 
 137     if args.startswith("@"):
 138         # Argument is a YAML file (JSON is a subset of YAML)
 139         complex_args = utils_vars.combine_vars(complex_args, loader.load_from_file(args[1:]))
 140         args = ''
 141     elif args.startswith("{"):
 142         # Argument is a YAML document (not a file)
 143         complex_args = utils_vars.combine_vars(complex_args, loader.load(args))
 144         args = ''
 145 
 146     if args:
 147         parsed_args = parse_kv(args)
 148         complex_args = utils_vars.combine_vars(complex_args, parsed_args)
 149 
 150     task_vars = interpreters
 151 
 152     if check:
 153         complex_args['_ansible_check_mode'] = True
 154 
 155     modname = os.path.basename(modfile)
 156     modname = os.path.splitext(modname)[0]
 157     (module_data, module_style, shebang) = module_common.modify_module(
 158         modname,
 159         modfile,
 160         complex_args,
 161         Templar(loader=loader),
 162         task_vars=task_vars
 163     )
 164 
 165     if module_style == 'new' and '_ANSIBALLZ_WRAPPER = True' in to_native(module_data):
 166         module_style = 'ansiballz'
 167 
 168     modfile2_path = os.path.expanduser(destfile)
 169     print("* including generated source, if any, saving to: %s" % modfile2_path)
 170     if module_style not in ('ansiballz', 'old'):
 171         print("* this may offset any line numbers in tracebacks/debuggers!")
 172     modfile2 = open(modfile2_path, 'wb')
 173     modfile2.write(module_data)
 174     modfile2.close()
 175     modfile = modfile2_path
 176 
 177     return (modfile2_path, modname, module_style)

 180 def ansiballz_setup(modfile, modname, interpreters):
 181     os.system("chmod +x %s" % modfile)
 182 
 183     if 'ansible_python_interpreter' in interpreters:
 184         command = [interpreters['ansible_python_interpreter']]
 185     else:
 186         command = []
 187     command.extend([modfile, 'explode'])
 188 
 189     cmd = subprocess.Popen(command, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
 190     out, err = cmd.communicate()
 191     out, err = to_text(out, errors='surrogate_or_strict'), to_text(err)
 192     lines = out.splitlines()
 193     if len(lines) != 2 or 'Module expanded into' not in lines[0]:
 194         print("*" * 35)
 195         print("INVALID OUTPUT FROM ANSIBALLZ MODULE WRAPPER")
 196         print(out)
 197         sys.exit(err)
 198     debug_dir = lines[1].strip()
 199 
 200     # All the directories in an AnsiBallZ that modules can live
 201     core_dirs = glob.glob(os.path.join(debug_dir, 'ansible/modules'))
 202     collection_dirs = glob.glob(os.path.join(debug_dir, 'ansible_collections/*/*/plugins/modules'))
 203 
 204     # There's only one module in an AnsiBallZ payload so look for the first module and then exit
 205     for module_dir in core_dirs + collection_dirs:
 206         for dirname, directories, filenames in os.walk(module_dir):
 207             for filename in filenames:
 208                 if filename == modname + '.py':
 209                     modfile = os.path.join(dirname, filename)
 210                     break
 211 
 212     argsfile = os.path.join(debug_dir, 'args')
 213 
 214     print("* ansiballz module detected; extracted module source to: %s" % debug_dir)
 215     return modfile, argsfile

 218 def runtest(modfile, argspath, modname, module_style, interpreters):
 219     """Test run a module, piping it's output for reporting."""
 220     invoke = ""
 221     if module_style == 'ansiballz':
 222         modfile, argspath = ansiballz_setup(modfile, modname, interpreters)
 223         if 'ansible_python_interpreter' in interpreters:
 224             invoke = "%s " % interpreters['ansible_python_interpreter']
 225 
 226     os.system("chmod +x %s" % modfile)
 227 
 228     invoke = "%s%s" % (invoke, modfile)
 229     if argspath is not None:
 230         invoke = "%s %s" % (invoke, argspath)
 231 
 232     cmd = subprocess.Popen(invoke, shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
 233     (out, err) = cmd.communicate()
 234     out, err = to_text(out), to_text(err)
 235 
 236     try:
 237         print("*" * 35)
 238         print("RAW OUTPUT")
 239         print(out)
 240         print(err)
 241         results = json.loads(out)
 242     except Exception:
 243         print("*" * 35)
 244         print("INVALID OUTPUT FORMAT")
 245         print(out)
 246         traceback.print_exc()
 247         sys.exit(1)
 248 
 249     print("*" * 35)
 250     print("PARSED OUTPUT")
 251     print(jsonify(results, format=True))

 254 def rundebug(debugger, modfile, argspath, modname, module_style, interpreters):
 255     """Run interactively with console debugger."""
 256 
 257     if module_style == 'ansiballz':
 258         modfile, argspath = ansiballz_setup(modfile, modname, interpreters)
 259 
 260     if argspath is not None:
 261         subprocess.call("%s %s %s" % (debugger, modfile, argspath), shell=True)
 262     else:
 263         subprocess.call("%s %s" % (debugger, modfile), shell=True)

 266 def main():
 267 
 268     options, args = parse()
 269     interpreters = get_interpreters(options.interpreter)
 270     (modfile, modname, module_style) = boilerplate_module(options.module_path, options.module_args, interpreters, options.check, options.filename)
 271 
 272     argspath = None
 273     if module_style not in ('new', 'ansiballz'):
 274         if module_style in ('non_native_want_json', 'binary'):
 275             argspath = write_argsfile(options.module_args, json=True)
 276         elif module_style == 'old':
 277             argspath = write_argsfile(options.module_args, json=False)
 278         else:
 279             raise Exception("internal error, unexpected module style: %s" % module_style)
 280 
 281     if options.execute:
 282         if options.debugger:
 283             rundebug(options.debugger, modfile, argspath, modname, module_style, interpreters)
 284         else:
 285             runtest(modfile, argspath, modname, module_style, interpreters)
 286 
 287 
 288 if __name__ == "__main__":
 289     try:
 290         main()
 291     finally:
 292         shutil.rmtree(C.DEFAULT_LOCAL_TMP, True)
