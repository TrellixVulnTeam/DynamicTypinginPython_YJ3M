  24 def build_lib_path(this_script=__file__):
  25     """Return path to the common build library directory"""
  26     hacking_dir = os.path.dirname(this_script)
  27     libdir = os.path.abspath(os.path.join(hacking_dir, 'build_library'))
  28 
  29     return libdir
  30

  32 sys.path.insert(0, build_lib_path())

  34 from build_ansible import commands, errors

  36 
  37 def create_arg_parser(program_name):
  38     """
  39     Creates a command line argument parser
  40 
  41     :arg program_name: The name of the script.  Used in help texts
  42     """
  43     parser = argparse.ArgumentParser(prog=program_name,
  44                                      description="Implements utilities to build Ansible")
  45     return parser
  46 
  47 
  48 def main():
  49     """
  50     Main entrypoint of the script
  51 
  52     "It all starts here"
  53     """
  54     subcommands = load('build_ansible.command_plugins', subclasses=commands.Command)
  55 
  56     arg_parser = create_arg_parser(os.path.basename(sys.argv[0]))
  57     arg_parser.add_argument('--debug', dest='debug', required=False, default=False, action='store_true',
  58                             help='Show tracebacks and other debugging information')
  59     subparsers = arg_parser.add_subparsers(title='Subcommands', dest='command',
  60                                            help='for help use build-ansible.py SUBCOMMANDS -h')
  61     subcommands.pipe('init_parser', subparsers.add_parser)
  62 
  63     if argcomplete:
  64         argcomplete.autocomplete(arg_parser)
  65 
  66     args = arg_parser.parse_args(sys.argv[1:])
  67     if args.command is None:
  68         print('Please specify a subcommand to run')
  69         sys.exit(1)
  70 
  71     for subcommand in subcommands:
  72         if subcommand.name == args.command:
  73             command = subcommand
  74             break
  75     else:
  76         # Note: We should never trigger this because argparse should shield us from it
  77         print('Error: {0} was not a recognized subcommand'.format(args.command))
  78         sys.exit(1)
  79 
  80     try:
  81         retval = command.main(args)
  82     except (errors.DependencyError, errors.MissingUserInput, errors.InvalidUserInput) as e:
  83         print(e)
  84         if args.debug:
  85             raise
  86         sys.exit(2)
  87 
  88     sys.exit(retval)
  89 
  90 
  91 if __name__ == '__main__':
  92     main()
