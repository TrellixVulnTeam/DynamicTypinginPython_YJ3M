  29 PULL_URL_RE = re.compile(r'(?P<user>\S+)/(?P<repo>\S+)#(?P<ticket>\d+)')
  30 PULL_HTTP_URL_RE = re.compile(r'https?://(?:www\.|)github.com/(?P<user>\S+)/(?P<repo>\S+)/pull/(?P<ticket>\d+)')
  31 PULL_BACKPORT_IN_TITLE = re.compile(r'.*\(#?(?P<ticket1>\d+)\)|\(backport of #?(?P<ticket2>\d+)\).*', re.I)
  32 PULL_CHERRY_PICKED_FROM = re.compile(r'\(?cherry(?:\-| )picked from(?: ?commit|) (?P<hash>\w+)(?:\)|\.|$)')
  33 TICKET_NUMBER = re.compile(r'(?:^|\s)#(\d+)')

  36 def normalize_pr_url(pr, allow_non_ansible_ansible=False, only_number=False):
  37     '''
  38     Given a PullRequest, or a string containing a PR number, PR URL,
  39     or internal PR URL (e.g. ansible-collections/community.general#1234),
  40     return either a full github URL to the PR (if only_number is False),
  41     or an int containing the PR number (if only_number is True).
  42 
  43     Throws if it can't parse the input.
  44     '''
  45     if isinstance(pr, PullRequest):
  46         return pr.html_url
  47 
  48     if pr.isnumeric():
  49         if only_number:
  50             return int(pr)
  51         return 'https://github.com/ansible/ansible/pull/{0}'.format(pr)
  52 
  53     # Allow for forcing ansible/ansible
  54     if not allow_non_ansible_ansible and 'ansible/ansible' not in pr:
  55         raise Exception('Non ansible/ansible repo given where not expected')
  56 
  57     re_match = PULL_HTTP_URL_RE.match(pr)
  58     if re_match:
  59         if only_number:
  60             return int(re_match.group('ticket'))
  61         return pr
  62 
  63     re_match = PULL_URL_RE.match(pr)
  64     if re_match:
  65         if only_number:
  66             return int(re_match.group('ticket'))
  67         return 'https://github.com/{0}/{1}/pull/{2}'.format(
  68             re_match.group('user'),
  69             re_match.group('repo'),
  70             re_match.group('ticket'))
  71 
  72     raise Exception('Did not understand given PR')

  75 def url_to_org_repo(url):
  76     '''
  77     Given a full Github PR URL, extract the user/org and repo name.
  78     Return them in the form: "user/repo"
  79     '''
  80     match = PULL_HTTP_URL_RE.match(url)
  81     if not match:
  82         return ''
  83     return '{0}/{1}'.format(match.group('user'), match.group('repo'))

  86 def generate_new_body(pr, source_pr):
  87     '''
  88     Given the new PR (the backport) and the originating (source) PR,
  89     construct the new body for the backport PR.
  90 
  91     If the backport follows the usual ansible/ansible template, we look for the
  92     '##### SUMMARY'-type line and add our "Backport of" line right below that.
  93 
  94     If we can't find the SUMMARY line, we add our line at the very bottom.
  95 
  96     This function does not side-effect, it simply returns the new body as a
  97     string.
  98     '''
  99     backport_text = '\nBackport of {0}\n'.format(source_pr)
 100     body_lines = pr.body.split('\n')
 101     new_body_lines = []
 102 
 103     added = False
 104     for line in body_lines:
 105         if 'Backport of http' in line:
 106             raise Exception('Already has a backport line, aborting.')
 107         new_body_lines.append(line)
 108         if line.startswith('#') and line.strip().endswith('SUMMARY'):
 109             # This would be a fine place to add it
 110             new_body_lines.append(backport_text)
 111             added = True
 112     if not added:
 113         # Otherwise, no '#### SUMMARY' line, so just add it at the bottom
 114         new_body_lines.append(backport_text)
 115 
 116     return '\n'.join(new_body_lines)

 119 def get_prs_for_commit(g, commit):
 120     '''
 121     Given a commit hash, attempt to find the hash in any repo in the
 122     ansible orgs, and then use it to determine what, if any, PR it appeared in.
 123     '''
 124 
 125     commits = g.search_commits(
 126         'hash:{0} org:ansible org:ansible-collections is:public'.format(commit)
 127     ).get_page(0)
 128     if not commits or len(commits) == 0:
 129         return []
 130     pulls = commits[0].get_pulls().get_page(0)
 131     if not pulls or len(pulls) == 0:
 132         return []
 133     return pulls

 136 def search_backport(pr, g, ansible_ansible):
 137     '''
 138     Do magic. This is basically the "brain" of 'auto'.
 139     It will search the PR (the newest PR - the backport) and try to find where
 140     it originated.
 141 
 142     First it will search in the title. Some titles include things like
 143     "foo bar change (#12345)" or "foo bar change (backport of #54321)"
 144     so we search for those and pull them out.
 145 
 146     Next it will scan the body of the PR and look for:
 147       - cherry-pick reference lines (e.g. "cherry-picked from commit XXXXX")
 148       - other PRs (#nnnnnn) and (foo/bar#nnnnnnn)
 149       - full URLs to other PRs
 150 
 151     It will take all of the above, and return a list of "possibilities",
 152     which is a list of PullRequest objects.
 153     '''
 154 
 155     possibilities = []
 156 
 157     # 1. Try searching for it in the title.
 158     title_search = PULL_BACKPORT_IN_TITLE.match(pr.title)
 159     if title_search:
 160         ticket = title_search.group('ticket1')
 161         if not ticket:
 162             ticket = title_search.group('ticket2')
 163         try:
 164             possibilities.append(ansible_ansible.get_pull(int(ticket)))
 165         except Exception:
 166             pass
 167 
 168     # 2. Search for clues in the body of the PR
 169     body_lines = pr.body.split('\n')
 170     for line in body_lines:
 171         # a. Try searching for a `git cherry-pick` line
 172         cherrypick = PULL_CHERRY_PICKED_FROM.match(line)
 173         if cherrypick:
 174             prs = get_prs_for_commit(g, cherrypick.group('hash'))
 175             possibilities.extend(prs)
 176             continue
 177 
 178         # b. Try searching for other referenced PRs (by #nnnnn or full URL)
 179         tickets = [('ansible', 'ansible', ticket) for ticket in TICKET_NUMBER.findall(line)]
 180         tickets.extend(PULL_HTTP_URL_RE.findall(line))
 181         tickets.extend(PULL_URL_RE.findall(line))
 182         if tickets:
 183             for ticket in tickets:
 184                 # Is it a PR (even if not in ansible/ansible)?
 185                 # TODO: As a small optimization/to avoid extra calls to GitHub,
 186                 # we could limit this check to non-URL matches. If it's a URL,
 187                 # we know it's definitely a pull request.
 188                 try:
 189                     repo_path = '{0}/{1}'.format(ticket[0], ticket[1])
 190                     repo = ansible_ansible
 191                     if repo_path != 'ansible/ansible':
 192                         repo = g.get_repo(repo_path)
 193                     ticket_pr = repo.get_pull(int(ticket))
 194                     possibilities.append(ticket_pr)
 195                 except Exception:
 196                     pass
 197             continue  # Future-proofing
 198 
 199     return possibilities

 202 def prompt_add():
 203     '''
 204     Prompt the user and return whether or not they agree.
 205     '''
 206     res = input('Shall I add the reference? [Y/n]: ')
 207     return res.lower() in ('', 'y', 'yes')

 210 def commit_edit(new_pr, pr):
 211     '''
 212     Given the new PR (the backport), and the "possibility" that we have decided
 213     on, prompt the user and then add the reference to the body of the new PR.
 214 
 215     This method does the actual "destructive" work of editing the PR body.
 216     '''
 217     print('I think this PR might have come from:')
 218     print(pr.title)
 219     print('-' * 50)
 220     print(pr.html_url)
 221     if prompt_add():
 222         new_body = generate_new_body(new_pr, pr.html_url)
 223         new_pr.edit(body=new_body)
 224         print('I probably added the reference successfully.')

 227 if __name__ == '__main__':
 228     if (
 229         len(sys.argv) != 3 or
 230         not sys.argv[1].isnumeric()
 231     ):
 232         print('Usage: <new backport PR> <already merged PR, or "auto">')
 233         sys.exit(1)
 234 
 235     token = os.environ.get('GITHUB_TOKEN')
 236     if not token:
 237         print('Go to https://github.com/settings/tokens/new and generate a '
 238               'token with "repo" access, then set GITHUB_TOKEN to that token.')
 239         sys.exit(1)
 240 
 241     # https://github.com/settings/tokens/new
 242     g = Github(token)
 243     ansible_ansible = g.get_repo('ansible/ansible')
 244 
 245     try:
 246         pr_num = normalize_pr_url(sys.argv[1], only_number=True)
 247         new_pr = ansible_ansible.get_pull(pr_num)
 248     except Exception:
 249         print('Could not load PR {0}'.format(sys.argv[1]))
 250         sys.exit(1)
 251 
 252     if sys.argv[2] == 'auto':
 253         print('Trying to find originating PR...')
 254         possibilities = search_backport(new_pr, g, ansible_ansible)
 255         if not possibilities:
 256             print('No match found, manual review required.')
 257             sys.exit(1)
 258         # TODO: Logic above can return multiple possibilities/guesses, but we
 259         # only handle one here. We can cycle/prompt through them or something.
 260         # For now, use the first match, which is also the most likely
 261         # candidate.
 262         pr = possibilities[0]
 263         commit_edit(new_pr, pr)
 264     else:
 265         try:
 266             # TODO: Fix having to call this twice to save some regex evals
 267             pr_num = normalize_pr_url(sys.argv[2], only_number=True, allow_non_ansible_ansible=True)
 268             pr_url = normalize_pr_url(sys.argv[2], allow_non_ansible_ansible=True)
 269             pr_repo = g.get_repo(url_to_org_repo(pr_url))
 270             pr = pr_repo.get_pull(pr_num)
 271         except Exception as e:
 272             print(e)
 273             print('Could not load PR {0}'.format(sys.argv[2]))
 274             sys.exit(1)
 275         commit_edit(new_pr, pr)
