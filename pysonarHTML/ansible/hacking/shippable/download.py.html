 132 def download_run(args):
 133     """Download a Shippable run."""
 134     headers = dict(
 135         Authorization='apiToken %s' % args.api_key,
 136     )
 137 
 138     match = re.search(
 139         r'^https://app.shippable.com/github/(?P<account>[^/]+)/(?P<project>[^/]+)/runs/(?P<run_number>[0-9]+)(?:/summary|(/(?P<job_number>[0-9]+)))?$',
 140         args.run_id)
 141 
 142     if not match:
 143         match = re.search(r'^(?P<account>[^/]+)/(?P<project>[^/]+)/(?P<run_number>[0-9]+)$', args.run_id)
 144 
 145     if match:
 146         account = match.group('account')
 147         project = match.group('project')
 148         run_number = int(match.group('run_number'))
 149         job_number = int(match.group('job_number')) if match.group('job_number') else None
 150 
 151         if job_number:
 152             if args.job_number:
 153                 sys.exit('ERROR: job number found in url and specified with --job-number')
 154 
 155             args.job_number = [job_number]
 156 
 157         url = 'https://api.shippable.com/projects'
 158         response = requests.get(url, dict(projectFullNames='%s/%s' % (account, project)), headers=headers)
 159 
 160         if response.status_code != 200:
 161             raise Exception(response.content)
 162 
 163         project_id = response.json()[0]['id']
 164 
 165         url = 'https://api.shippable.com/runs?projectIds=%s&runNumbers=%s' % (project_id, run_number)
 166 
 167         response = requests.get(url, headers=headers)
 168 
 169         if response.status_code != 200:
 170             raise Exception(response.content)
 171 
 172         run = [run for run in response.json() if run['runNumber'] == run_number][0]
 173 
 174         args.run_id = run['id']
 175     elif re.search('^[a-f0-9]+$', args.run_id):
 176         url = 'https://api.shippable.com/runs/%s' % args.run_id
 177 
 178         response = requests.get(url, headers=headers)
 179 
 180         if response.status_code != 200:
 181             raise Exception(response.content)
 182 
 183         run = response.json()
 184 
 185         account = run['subscriptionOrgName']
 186         project = run['projectName']
 187         run_number = run['runNumber']
 188     else:
 189         sys.exit('ERROR: invalid run: %s' % args.run_id)
 190 
 191     output_dir = '%s/%s/%s' % (account, project, run_number)
 192 
 193     if not args.test:
 194         if not os.path.exists(output_dir):
 195             os.makedirs(output_dir)
 196 
 197     if args.run_metadata:
 198         path = os.path.join(output_dir, 'run.json')
 199         contents = json.dumps(run, sort_keys=True, indent=4)
 200 
 201         if args.verbose or args.test:
 202             print(path)
 203 
 204         if not args.test:
 205             with open(path, 'w') as metadata_fd:
 206                 metadata_fd.write(contents)
 207 
 208     download_run_recursive(args, headers, output_dir, run, True)
 209

 211 def download_run_recursive(args, headers, output_dir, run, is_given=False):
 212     # Notes:
 213     # - The /runs response tells us if we need to eventually go up another layer
 214     #   or not (if we are a re-run attempt or not).
 215     # - Given a run id, /jobs will tell us all the jobs in that run, and whether
 216     #   or not we can pull results from them.
 217     #
 218     # When we initially run (i.e., in download_run), we'll have a /runs output
 219     # which we can use to get a /jobs output. Using the /jobs output, we filter
 220     # on the jobs we need to fetch (usually only successful ones unless we are
 221     # processing the initial/given run and not one of its parent runs) and
 222     # download them accordingly.
 223     #
 224     # Lastly, we check if the run we are currently processing has another
 225     # parent (reRunBatchId). If it does, we pull that /runs result and
 226     # recurse using it to start the process over again.
 227     response = requests.get('https://api.shippable.com/jobs?runIds=%s' % run['id'], headers=headers)
 228 
 229     if response.status_code != 200:
 230         raise Exception(response.content)
 231 
 232     jobs = sorted(response.json(), key=lambda job: int(job['jobNumber']))
 233 
 234     if is_given:
 235         needed_jobs = [j for j in jobs if j['isConsoleArchived']]
 236     else:
 237         needed_jobs = [j for j in jobs if j['isConsoleArchived'] and j['statusCode'] == 30]
 238 
 239     if not args.test:
 240         if not os.path.exists(output_dir):
 241             os.makedirs(output_dir)
 242 
 243     download_jobs(args, needed_jobs, headers, output_dir)
 244 
 245     rerun_batch_id = run.get('reRunBatchId')
 246     if rerun_batch_id:
 247         print('Downloading previous run: %s' % rerun_batch_id)
 248         response = requests.get('https://api.shippable.com/runs/%s' % rerun_batch_id, headers=headers)
 249 
 250         if response.status_code != 200:
 251             raise Exception(response.content)
 252 
 253         run = response.json()
 254         download_run_recursive(args, headers, output_dir, run)

 257 def download_jobs(args, jobs, headers, output_dir):
 258     """Download Shippable jobs."""
 259     for j in jobs:
 260         job_id = j['id']
 261         job_number = j['jobNumber']
 262 
 263         if args.job_number and job_number not in args.job_number:
 264             continue
 265 
 266         if args.job_metadata:
 267             path = os.path.join(output_dir, '%s/job.json' % job_number)
 268             contents = json.dumps(j, sort_keys=True, indent=4).encode('utf-8')
 269 
 270             if args.verbose or args.test:
 271                 print(path)
 272 
 273             if not args.test:
 274                 directory = os.path.dirname(path)
 275 
 276                 if not os.path.exists(directory):
 277                     os.makedirs(directory)
 278 
 279                 with open(path, 'wb') as metadata_fd:
 280                     metadata_fd.write(contents)
 281 
 282         if args.console_logs:
 283             path = os.path.join(output_dir, '%s/console.log' % job_number)
 284             url = 'https://api.shippable.com/jobs/%s/consoles?download=true' % job_id
 285             download(args, headers, path, url, is_json=False)
 286 
 287         if args.test_results:
 288             path = os.path.join(output_dir, '%s/test.json' % job_number)
 289             url = 'https://api.shippable.com/jobs/%s/jobTestReports' % job_id
 290             download(args, headers, path, url)
 291             extract_contents(args, path, os.path.join(output_dir, '%s/test' % job_number))
 292 
 293         if args.coverage_results:
 294             path = os.path.join(output_dir, '%s/coverage.json' % job_number)
 295             url = 'https://api.shippable.com/jobs/%s/jobCoverageReports' % job_id
 296             download(args, headers, path, url)
 297             extract_contents(args, path, os.path.join(output_dir, '%s/coverage' % job_number))

 300 def extract_contents(args, path, output_dir):
 301     """
 302     :type args: any
 303     :type path: str
 304     :type output_dir: str
 305     """
 306     if not args.test:
 307         if not os.path.exists(path):
 308             return
 309 
 310         with open(path, 'r') as json_fd:
 311             items = json.load(json_fd)
 312 
 313             for item in items:
 314                 contents = item['contents'].encode('utf-8')
 315                 path = output_dir + '/' + re.sub('^/*', '', item['path'])
 316 
 317                 directory = os.path.dirname(path)
 318 
 319                 if not os.path.exists(directory):
 320                     os.makedirs(directory)
 321 
 322                 if args.verbose:
 323                     print(path)
 324 
 325                 if path.endswith('.json'):
 326                     contents = json.dumps(json.loads(contents), sort_keys=True, indent=4).encode('utf-8')
 327 
 328                 if not os.path.exists(path):
 329                     with open(path, 'wb') as output_fd:
 330                         output_fd.write(contents)

 333 def download(args, headers, path, url, is_json=True):
 334     """
 335     :type args: any
 336     :type headers: dict[str, str]
 337     :type path: str
 338     :type url: str
 339     :type is_json: bool
 340     """
 341     if args.verbose or args.test:
 342         print(path)
 343 
 344     if os.path.exists(path):
 345         return
 346 
 347     if not args.test:
 348         response = requests.get(url, headers=headers)
 349 
 350         if response.status_code != 200:
 351             path += '.error'
 352 
 353         if is_json:
 354             content = json.dumps(response.json(), sort_keys=True, indent=4).encode(response.encoding)
 355         else:
 356             content = response.content
 357 
 358         directory = os.path.dirname(path)
 359 
 360         if not os.path.exists(directory):
 361             os.makedirs(directory)
 362 
 363         with open(path, 'wb') as content_fd:
 364             content_fd.write(content)

 367 def get_api_key():
 368     """
 369     rtype: str
 370     """
 371     key = os.environ.get('SHIPPABLE_KEY', None)
 372 
 373     if key:
 374         return key
 375 
 376     path = os.path.join(os.environ['HOME'], '.shippable.key')
 377 
 378     try:
 379         with open(path, 'r') as key_fd:
 380             return key_fd.read().strip()
 381     except IOError:
 382         return None
