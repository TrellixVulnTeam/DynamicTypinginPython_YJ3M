  51 def parse_args():
  52     """Parse and return args."""
  53     source = os.path.dirname(os.path.dirname(os.path.dirname(os.path.abspath(__file__))))
  54 
  55     parser = argparse.ArgumentParser(description='Report on incidental test coverage downloaded from Shippable.')
  56 
  57     parser.add_argument('result',
  58                         type=directory,
  59                         help='path to directory containing test results downloaded from Shippable')
  60 
  61     parser.add_argument('--output',
  62                         type=optional_directory,
  63                         default=os.path.join(source, 'test', 'results', '.tmp', 'incidental'),
  64                         help='path to directory where reports should be written')
  65 
  66     parser.add_argument('--source',
  67                         type=optional_directory,
  68                         default=source,
  69                         help='path to git repository containing Ansible source')
  70 
  71     parser.add_argument('--skip-checks',
  72                         action='store_true',
  73                         help='skip integrity checks, use only for debugging')
  74 
  75     parser.add_argument('--ignore-cache',
  76                         dest='use_cache',
  77                         action='store_false',
  78                         help='ignore cached files')
  79 
  80     parser.add_argument('-v', '--verbose',
  81                         action='store_true',
  82                         help='increase verbosity')
  83 
  84     targets = parser.add_mutually_exclusive_group()
  85 
  86     targets.add_argument('--targets',
  87                          type=regex,
  88                          default='^incidental_',
  89                          help='regex for targets to analyze, default: %(default)s')
  90 
  91     targets.add_argument('--plugin-path',
  92                          help='path to plugin to report incidental coverage on')
  93 
  94     if argcomplete:
  95         argcomplete.autocomplete(parser)
  96 
  97     args = parser.parse_args()
  98 
  99     return args

 102 def optional_directory(value):
 103     if not os.path.exists(value):
 104         return value
 105 
 106     return directory(value)

 109 def directory(value):
 110     if not os.path.isdir(value):
 111         raise argparse.ArgumentTypeError('"%s" is not a directory' % value)
 112 
 113     return value

 120         raise argparse.ArgumentTypeError('"%s" is not a valid regex: %s' % (value, ex))

 123 def incidental_report(args):
 124     """Generate incidental coverage report."""
 125     ct = CoverageTool()
 126     git = Git(os.path.abspath(args.source))
 127     coverage_data = CoverageData(os.path.abspath(args.result))
 128 
 129     try:
 130         git.show([coverage_data.result_sha, '--'])
 131     except subprocess.CalledProcessError:
 132         raise ApplicationError('%s: commit not found: %s\n'
 133                                'make sure your source repository is up-to-date' % (git.path, coverage_data.result_sha))
 134 
 135     if coverage_data.status_code != 30:
 136         check_failed(args, 'results from Shippable indicate tests did not pass (status code: %d)\n'
 137                            're-run until passing, then download the latest results and re-run the report using those results' % coverage_data.status_code)
 138 
 139     if coverage_data.missing_jobs or coverage_data.extra_jobs:
 140         check_failed(args, 'unexpected results from Shippable -- missing jobs: %s, extra jobs: %s\n'
 141                            'make sure the tests were successfull and the all results were downloaded\n' % (
 142                                sorted(coverage_data.missing_jobs), sorted(coverage_data.extra_jobs)))
 143 
 144     if not coverage_data.paths:
 145         raise ApplicationError('no coverage data found\n'
 146                                'make sure the downloaded results are from a code coverage run on Shippable')
 147 
 148     # generate a unique subdirectory in the output directory based on the input files being used
 149     path_hash = hashlib.sha256(b'\n'.join(p.encode() for p in coverage_data.paths)).hexdigest()
 150     output_path = os.path.abspath(os.path.join(args.output, path_hash))
 151 
 152     data_path = os.path.join(output_path, 'data')
 153     reports_path = os.path.join(output_path, 'reports')
 154 
 155     for path in [data_path, reports_path]:
 156         if not os.path.exists(path):
 157             os.makedirs(path)
 158 
 159     # combine coverage results into a single file
 160     combined_path = os.path.join(output_path, 'combined.json')
 161     cached(combined_path, args.use_cache, args.verbose,
 162            lambda: ct.combine(coverage_data.paths, combined_path))
 163 
 164     with open(combined_path) as combined_file:
 165         combined = json.load(combined_file)
 166 
 167     if args.plugin_path:
 168         # reporting on coverage missing from the test target for the specified plugin
 169         # the report will be on a single target
 170         cache_path_format = '%s' + '-for-%s' % os.path.splitext(os.path.basename(args.plugin_path))[0]
 171         target_pattern = '^%s$' % get_target_name_from_plugin_path(args.plugin_path)
 172         include_path = args.plugin_path
 173         missing = True
 174         target_name = get_target_name_from_plugin_path(args.plugin_path)
 175     else:
 176         # reporting on coverage exclusive to the matched targets
 177         # the report can contain multiple targets
 178         cache_path_format = '%s'
 179         target_pattern = args.targets
 180         include_path = None
 181         missing = False
 182         target_name = None
 183 
 184     # identify integration test targets to analyze
 185     target_names = sorted(combined['targets'])
 186     incidental_target_names = [target for target in target_names if re.search(target_pattern, target)]
 187 
 188     if not incidental_target_names:
 189         if target_name:
 190             # if the plugin has no tests we still want to know what coverage is missing
 191             incidental_target_names = [target_name]
 192         else:
 193             raise ApplicationError('no targets to analyze')
 194 
 195     # exclude test support plugins from analysis
 196     # also exclude six, which for an unknown reason reports bogus coverage lines (indicating coverage of comments)
 197     exclude_path = '^(test/support/|lib/ansible/module_utils/six/)'
 198 
 199     # process coverage for each target and then generate a report
 200     # save sources for generating a summary report at the end
 201     summary = {}
 202     report_paths = {}
 203 
 204     for target_name in incidental_target_names:
 205         cache_name = cache_path_format % target_name
 206 
 207         only_target_path = os.path.join(data_path, 'only-%s.json' % cache_name)
 208         cached(only_target_path, args.use_cache, args.verbose,
 209                lambda: ct.filter(combined_path, only_target_path, include_targets=[target_name], include_path=include_path, exclude_path=exclude_path))
 210 
 211         without_target_path = os.path.join(data_path, 'without-%s.json' % cache_name)
 212         cached(without_target_path, args.use_cache, args.verbose,
 213                lambda: ct.filter(combined_path, without_target_path, exclude_targets=[target_name], include_path=include_path, exclude_path=exclude_path))
 214 
 215         if missing:
 216             source_target_path = missing_target_path = os.path.join(data_path, 'missing-%s.json' % cache_name)
 217             cached(missing_target_path, args.use_cache, args.verbose,
 218                    lambda: ct.missing(without_target_path, only_target_path, missing_target_path, only_gaps=True))
 219         else:
 220             source_target_path = exclusive_target_path = os.path.join(data_path, 'exclusive-%s.json' % cache_name)
 221             cached(exclusive_target_path, args.use_cache, args.verbose,
 222                    lambda: ct.missing(only_target_path, without_target_path, exclusive_target_path, only_gaps=True))
 223 
 224         source_expanded_target_path = os.path.join(os.path.dirname(source_target_path), 'expanded-%s' % os.path.basename(source_target_path))
 225         cached(source_expanded_target_path, args.use_cache, args.verbose,
 226                lambda: ct.expand(source_target_path, source_expanded_target_path))
 227 
 228         summary[target_name] = sources = collect_sources(source_expanded_target_path, git, coverage_data)
 229 
 230         txt_report_path = os.path.join(reports_path, '%s.txt' % cache_name)
 231         cached(txt_report_path, args.use_cache, args.verbose,
 232                lambda: generate_report(sources, txt_report_path, coverage_data, target_name, missing=missing))
 233 
 234         report_paths[target_name] = txt_report_path
 235 
 236     # provide a summary report of results
 237     for target_name in incidental_target_names:
 238         sources = summary[target_name]
 239         report_path = os.path.relpath(report_paths[target_name])
 240 
 241         print('%s: %d arcs, %d lines, %d files - %s' % (
 242             target_name,
 243             sum(len(s.covered_arcs) for s in sources),
 244             sum(len(s.covered_lines) for s in sources),
 245             len(sources),
 246             report_path,
 247         ))
 248 
 249     if not missing:
 250         sys.stderr.write('NOTE: This report shows only coverage exclusive to the reported targets. '
 251                          'As targets are removed, exclusive coverage on the remaining targets will increase.\n')
 252

 254 def get_target_name_from_plugin_path(path):  # type: (str) -> str
 255     """Return the integration test target name for the given plugin path."""
 256     parts = os.path.splitext(path)[0].split(os.path.sep)
 257     plugin_name = parts[-1]
 258 
 259     if path.startswith('lib/ansible/modules/'):
 260         plugin_type = None
 261     elif path.startswith('lib/ansible/plugins/'):
 262         plugin_type = parts[3]
 263     elif path.startswith('lib/ansible/module_utils/'):
 264         plugin_type = parts[2]
 265     elif path.startswith('plugins/'):
 266         plugin_type = parts[1]
 267     else:
 268         raise ApplicationError('Cannot determine plugin type from plugin path: %s' % path)
 269 
 270     if plugin_type is None:
 271         target_name = plugin_name
 272     else:
 273         target_name = '%s_%s' % (plugin_type, plugin_name)
 274 
 275     return target_name

 278 class CoverageData:
 279     def __init__(self, result_path):
 280         with open(os.path.join(result_path, 'run.json')) as run_file:
 281             run = json.load(run_file)
 282 
 283         self.org_name = run['subscriptionOrgName']
 284         self.project_name = run['projectName']
 285         self.result_sha = run['commitSha']
 286         self.status_code = run['statusCode']
 287 
 288         self.github_base_url = 'https://github.com/%s/%s/blob/%s/' % (self.org_name, self.project_name, self.result_sha)
 289 
 290         # locate available results
 291         self.paths = sorted(glob.glob(os.path.join(result_path, '*', 'test', 'testresults', 'coverage-analyze-targets.json')))
 292 
 293         # make sure the test matrix is complete
 294         matrix_include = run['cleanRunYml']['matrix']['include']
 295         matrix_jobs = list((idx, dict(tuple(item.split('=', 1)) for item in value['env'])) for idx, value in enumerate(matrix_include, start=1))
 296         sanity_job_numbers = set(idx for idx, env in matrix_jobs if env['T'].startswith('sanity/'))
 297         units_job_numbers = set(idx for idx, env in matrix_jobs if env['T'].startswith('units/'))
 298         expected_job_numbers = set(idx for idx, env in matrix_jobs)
 299         actual_job_numbers = set(int(os.path.relpath(path, result_path).split(os.path.sep)[0]) for path in self.paths)
 300 
 301         self.missing_jobs = expected_job_numbers - actual_job_numbers - sanity_job_numbers - units_job_numbers
 302         self.extra_jobs = actual_job_numbers - expected_job_numbers - sanity_job_numbers - units_job_numbers

 305 class Git:
 306     def __init__(self, path):
 307         self.git = 'git'
 308         self.path = path
 309 
 310         try:
 311             self.show()
 312         except subprocess.CalledProcessError:
 313             raise ApplicationError('%s: not a git repository' % path)
 314 
 315     def show(self, args=None):
 316         return self.run(['show'] + (args or []))
 317 
 318     def run(self, command):
 319         return subprocess.check_output([self.git] + command, cwd=self.path)

 322 class CoverageTool:
 323     def __init__(self):
 324         self.analyze_cmd = ['ansible-test', 'coverage', 'analyze', 'targets']
 325 
 326     def combine(self, input_paths, output_path):
 327         subprocess.check_call(self.analyze_cmd + ['combine'] + input_paths + [output_path])
 328 
 329     def filter(self, input_path, output_path, include_targets=None, exclude_targets=None, include_path=None, exclude_path=None):
 330         args = []
 331 
 332         if include_targets:
 333             for target in include_targets:
 334                 args.extend(['--include-target', target])
 335 
 336         if exclude_targets:
 337             for target in exclude_targets:
 338                 args.extend(['--exclude-target', target])
 339 
 340         if include_path:
 341             args.extend(['--include-path', include_path])
 342 
 343         if exclude_path:
 344             args.extend(['--exclude-path', exclude_path])
 345 
 346         subprocess.check_call(self.analyze_cmd + ['filter', input_path, output_path] + args)
 347 
 348     def missing(self, from_path, to_path, output_path, only_gaps=False):
 349         args = []
 350 
 351         if only_gaps:
 352             args.append('--only-gaps')
 353 
 354         subprocess.check_call(self.analyze_cmd + ['missing', from_path, to_path, output_path] + args)
 355 
 356     def expand(self, input_path, output_path):
 357         subprocess.check_call(self.analyze_cmd + ['expand', input_path, output_path])

 360 class SourceFile:
 361     def __init__(self, path, source, coverage_data, coverage_points):
 362         self.path = path
 363         self.lines = source.decode().splitlines()
 364         self.coverage_data = coverage_data
 365         self.coverage_points = coverage_points
 366         self.github_url = coverage_data.github_base_url + path
 367 
 368         is_arcs = ':' in dict(coverage_points).popitem()[0]
 369 
 370         if is_arcs:
 371             parse = parse_arc
 372         else:
 373             parse = int
 374 
 375         self.covered_points = set(parse(v) for v in coverage_points)
 376         self.covered_arcs = self.covered_points if is_arcs else None
 377         self.covered_lines = set(abs(p[0]) for p in self.covered_points) | set(abs(p[1]) for p in self.covered_points)

 380 def collect_sources(data_path, git, coverage_data):
 381     with open(data_path) as data_file:
 382         data = json.load(data_file)
 383 
 384     sources = []
 385 
 386     for path_coverage in data.values():
 387         for path, path_data in path_coverage.items():
 388             sources.append(SourceFile(path, git.show(['%s:%s' % (coverage_data.result_sha, path)]), coverage_data, path_data))
 389 
 390     return sources

 393 def generate_report(sources, report_path, coverage_data, target_name, missing):
 394     output = [
 395         'Target: %s (%s coverage)' % (target_name, 'missing' if missing else 'exclusive'),
 396         'GitHub: %stest/integration/targets/%s' % (coverage_data.github_base_url, target_name),
 397     ]
 398 
 399     for source in sources:
 400         if source.covered_arcs:
 401             output.extend([
 402                 '',
 403                 'Source: %s (%d arcs, %d/%d lines):' % (source.path, len(source.covered_arcs), len(source.covered_lines), len(source.lines)),
 404                 'GitHub: %s' % source.github_url,
 405                 '',
 406             ])
 407         else:
 408             output.extend([
 409                 '',
 410                 'Source: %s (%d/%d lines):' % (source.path, len(source.covered_lines), len(source.lines)),
 411                 'GitHub: %s' % source.github_url,
 412                 '',
 413             ])
 414 
 415         last_line_no = 0
 416 
 417         for line_no, line in enumerate(source.lines, start=1):
 418             if line_no not in source.covered_lines:
 419                 continue
 420 
 421             if last_line_no and last_line_no != line_no - 1:
 422                 output.append('')
 423 
 424             notes = ''
 425 
 426             if source.covered_arcs:
 427                 from_lines = sorted(p[0] for p in source.covered_points if abs(p[1]) == line_no)
 428                 to_lines = sorted(p[1] for p in source.covered_points if abs(p[0]) == line_no)
 429 
 430                 if from_lines:
 431                     notes += '  ### %s -> (here)' % ', '.join(str(from_line) for from_line in from_lines)
 432 
 433                 if to_lines:
 434                     notes += '  ### (here) -> %s' % ', '.join(str(to_line) for to_line in to_lines)
 435 
 436             output.append('%4d  %s%s' % (line_no, line, notes))
 437             last_line_no = line_no
 438 
 439     with open(report_path, 'w') as report_file:
 440         report_file.write('\n'.join(output) + '\n')

 444     return tuple(int(v) for v in value.split(':'))

 447 def cached(path, use_cache, show_messages, func):
 448     if os.path.exists(path) and use_cache:
 449         if show_messages:
 450             sys.stderr.write('%s: cached\n' % path)
 451             sys.stderr.flush()
 452         return
 453 
 454     if show_messages:
 455         sys.stderr.write('%s: generating ... ' % path)
 456         sys.stderr.flush()
 457 
 458     func()
 459 
 460     if show_messages:
 461         sys.stderr.write('done\n')
 462         sys.stderr.flush()

 465 def check_failed(args, message):
 466     if args.skip_checks:
 467         sys.stderr.write('WARNING: %s\n' % message)
 468         return
 469 
 470     raise ApplicationError(message)
