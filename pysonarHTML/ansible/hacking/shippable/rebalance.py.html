   7 """
   8 CLI tool that analyses a Shippable run's test result and re-balances the test targets into new groups.
   9 
  10 Before running this script you must run download.py like:
  11 
  12     ./download.py https://app.shippable.com/github/<team>/<repo>/runs/<run_num> --test-results --job-number x --job-number y
  13 
  14 Or to get all job results from a run:
  15 
  16     ./download.py https://app.shippable.com/github/<team>/<repo>/runs/<run_num> --test-results --all
  17 
  18 
  19 Set the dir <team>/<repo>/<run_num> as the value of '-p/--test-path' for this script.
  20 """

  77 def get_raw_test_targets(args, test_path):
  78     """Scans the test directory for all the test targets that was run and get's the max runtime for each target."""
  79     target_times = {}
  80 
  81     for job_id in os.listdir(test_path):
  82         json_path = os.path.join(test_path, job_id, 'test', 'testresults', 'data')
  83 
  84         # Some tests to do not have a data directory
  85         if not os.path.exists(json_path):
  86             continue
  87 
  88         json_file = glob(os.path.join(json_path, '*integration-*.json'))[0]
  89         if not os.path.isfile(json_file):
  90             if args.verbose:
  91                 print("The test json file '%s' does not exist or is not a file, skipping test job run" % json_file)
  92             continue
  93 
  94         with open(json_file, mode='rb') as fd:
  95             test_info = json.loads(fd.read().decode('utf-8'))
  96 
  97         targets = test_info.get('targets', {})
  98 
  99         for target_name, target_info in targets.items():
 100             target_runtime = int(target_info.get('run_time_seconds', 0))
 101 
 102             # If that target already is found and has a higher runtime than the current one, ignore this entry.
 103             if target_times.get(target_name, 0) > target_runtime:
 104                 continue
 105 
 106             target_times[target_name] = target_runtime
 107 
 108     return dict(sorted(target_times.items(), key=lambda i: i[1], reverse=True))

 111 def print_test_runtime(target_times):
 112     """Prints a nice summary of a dict containing test target names and their runtime."""
 113     target_name_max_len = 0
 114     for target_name in target_times.keys():
 115         target_name_max_len = max(target_name_max_len, len(target_name))
 116 
 117     print("%s | Seconds |" % ("Target Name".ljust(target_name_max_len),))
 118     print("%s | ------- |" % ("-" * target_name_max_len,))
 119     for target_name, target_time in target_times.items():
 120         print("%s | %s |" % (target_name.ljust(target_name_max_len), str(target_time).ljust(7)))

 123 def rebalance(args):
 124     """Prints a nice summary of a proposed rebalanced configuration based on the downloaded Shippable result."""
 125     test_path = os.path.expanduser(os.path.expandvars(args.test_results_path))
 126     target_times = get_raw_test_targets(args, test_path)
 127 
 128     group_info = dict([(i, {'targets': [], 'total_time': 0}) for i in range(1, int(args.group_count) + 1)])
 129 
 130     # Now add each test to the group with the lowest running time.
 131     for target_name, target_time in target_times.items():
 132         index, total_time = min(enumerate([g['total_time'] for g in group_info.values()]), key=operator.itemgetter(1))
 133         group_info[index + 1]['targets'].append(target_name)
 134         group_info[index + 1]['total_time'] = total_time + target_time
 135 
 136     # Print a summary of the proposed test split.
 137     for group_number, test_info in group_info.items():
 138         print("Group %d - Total Runtime (s): %d" % (group_number, test_info['total_time']))
 139         print_test_runtime(dict([(n, target_times[n]) for n in test_info['targets']]))
 140         print()
 141 
 142     if args.target_path:
 143         target_path = os.path.expanduser(os.path.expandvars(args.target_path))
 144 
 145         for test_root in ['test', 'tests']:  # ansible/ansible uses 'test' but collections use 'tests'.
 146             integration_root = os.path.join(target_path, test_root, 'integration', 'targets')
 147             if os.path.isdir(integration_root):
 148                 if args.verbose:
 149                     print("Found test integration target dir at '%s'" % integration_root)
 150                 break
 151 
 152         else:
 153             # Failed to find test integration target folder
 154             raise ValueError("Failed to find the test target folder on test/integration/targets or "
 155                              "tests/integration/targets under '%s'." % target_path)
 156 
 157         for group_number, test_info in group_info.items():
 158             for test_target in test_info['targets']:
 159                 test_target_aliases = os.path.join(integration_root, test_target, 'aliases')
 160                 if not os.path.isfile(test_target_aliases):
 161                     if args.verbose:
 162                         print("Cannot find test target alias file at '%s', skipping." % test_target_aliases)
 163                     continue
 164 
 165                 with open(test_target_aliases, mode='r') as fd:
 166                     test_aliases = fd.readlines()
 167 
 168                 changed = False
 169                 for idx, line in enumerate(test_aliases):
 170                     group_match = re.match(r'shippable/(.*)/group(\d+)', line)
 171                     if group_match:
 172                         if int(group_match.group(2)) != group_number:
 173                             new_group = 'shippable/%s/group%d\n' % (group_match.group(1), group_number)
 174                             if args.verbose:
 175                                 print("Changing %s group from '%s' to '%s'" % (test_target, group_match.group(0),
 176                                                                                new_group.rstrip()))
 177                             test_aliases[idx] = new_group
 178                             changed = True
 179                             break
 180                 else:
 181                     if args.verbose:
 182                         print("Test target %s matches proposed group number, no changed required" % test_target)
 183 
 184                 if changed:
 185                     with open(test_target_aliases, mode='w') as fd:
 186                         fd.writelines(test_aliases)
 187 
 188 
 189 if __name__ == '__main__':
 190     main()
