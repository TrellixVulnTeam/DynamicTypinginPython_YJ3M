  40 def _find_symlinks(topdir, extension=''):
  41     """Find symlinks that should be maintained
  42 
  43     Maintained symlinks exist in the bin dir or are modules which have
  44     aliases.  Our heuristic is that they are a link in a certain path which
  45     point to a file in the same directory.
  46 
  47     .. warn::
  48 
  49         We want the symlinks in :file:`bin/` that link into :file:`lib/ansible/*` (currently,
  50         :command:`ansible`, :command:`ansible-test`, and :command:`ansible-connection`) to become
  51         real files on install.  Updates to the heuristic here *must not* add them to the symlink
  52         cache.
  53     """
  54     symlinks = defaultdict(list)
  55     for base_path, dirs, files in os.walk(topdir):
  56         for filename in files:
  57             filepath = os.path.join(base_path, filename)
  58             if os.path.islink(filepath) and filename.endswith(extension):
  59                 target = os.readlink(filepath)
  60                 if target.startswith('/'):
  61                     # We do not support absolute symlinks at all
  62                     continue
  63 
  64                 if os.path.dirname(target) == '':
  65                     link = filepath[len(topdir):]
  66                     if link.startswith('/'):
  67                         link = link[1:]
  68                     symlinks[os.path.basename(target)].append(link)
  69                 else:
  70                     # Count how many directory levels from the topdir we are
  71                     levels_deep = os.path.dirname(filepath).count('/')
  72 
  73                     # Count the number of directory levels higher we walk up the tree in target
  74                     target_depth = 0
  75                     for path_component in target.split('/'):
  76                         if path_component == '..':
  77                             target_depth += 1
  78                             # If we walk past the topdir, then don't store
  79                             if target_depth >= levels_deep:
  80                                 break
  81                         else:
  82                             target_depth -= 1
  83                     else:
  84                         # If we managed to stay within the tree, store the symlink
  85                         link = filepath[len(topdir):]
  86                         if link.startswith('/'):
  87                             link = link[1:]
  88                         symlinks[target].append(link)
  89 
  90     return symlinks

  98 def _maintain_symlinks(symlink_type, base_path):
  99     """Switch a real file into a symlink"""
 100     try:
 101         # Try the cache first because going from git checkout to sdist is the
 102         # only time we know that we're going to cache correctly
 103         with open(SYMLINK_CACHE, 'r') as f:
 104             symlink_data = json.load(f)
 105     except (IOError, OSError) as e:
 106         # IOError on py2, OSError on py3.  Both have errno
 107         if e.errno == 2:
 108             # SYMLINKS_CACHE doesn't exist.  Fallback to trying to create the
 109             # cache now.  Will work if we're running directly from a git
 110             # checkout or from an sdist created earlier.
 111             library_symlinks = _find_symlinks('lib', '.py')
 112             library_symlinks.update(_find_symlinks('test/lib'))
 113 
 114             symlink_data = {'script': _find_symlinks('bin'),
 115                             'library': library_symlinks,
 116                             }
 117 
 118             # Sanity check that something we know should be a symlink was
 119             # found.  We'll take that to mean that the current directory
 120             # structure properly reflects symlinks in the git repo
 121             if 'ansible-playbook' in symlink_data['script']['ansible']:
 122                 _cache_symlinks(symlink_data)
 123             else:
 124                 raise RuntimeError(
 125                     "Pregenerated symlink list was not present and expected "
 126                     "symlinks in ./bin were missing or broken. "
 127                     "Perhaps this isn't a git checkout?"
 128                 )
 129         else:
 130             raise
 131     symlinks = symlink_data[symlink_type]
 132 
 133     for source in symlinks:
 134         for dest in symlinks[source]:
 135             dest_path = os.path.join(base_path, dest)
 136             if not os.path.islink(dest_path):
 137                 try:
 138                     os.unlink(dest_path)
 139                 except OSError as e:
 140                     if e.errno == 2:
 141                         # File does not exist which is all we wanted
 142                         pass
 143                 os.symlink(source, dest_path)

 170 class SDistCommand(SDist):
 171     def run(self):
 172         # have to generate the cache of symlinks for release as sdist is the
 173         # only command that has access to symlinks from the git repo
 174         library_symlinks = _find_symlinks('lib', '.py')
 175         library_symlinks.update(_find_symlinks('test/lib'))
 176 
 177         symlinks = {'script': _find_symlinks('bin'),
 178                     'library': library_symlinks,
 179                     }
 180         _cache_symlinks(symlinks)
 181 
 182         SDist.run(self)
 183 
 184         # Print warnings at the end because no one will see warnings before all the normal status
 185         # output
 186         if os.environ.get('_ANSIBLE_SDIST_FROM_MAKEFILE', False) != '1':
 187             warnings.warn('When setup.py sdist is run from outside of the Makefile,'
 188                           ' the generated tarball may be incomplete.  Use `make snapshot`'
 189                           ' to create a tarball from an arbitrary checkout or use'
 190                           ' `cd packaging/release && make release version=[..]` for official builds.',
 191                           RuntimeWarning)

 200 def read_requirements(file_name):
 201     """Read requirements file as a list."""
 202     reqs = read_file(file_name).splitlines()
 203     if not reqs:
 204         raise RuntimeError(
 205             "Unable to read requirements from the %s file"
 206             "That indicates this copy of the source code is incomplete."
 207             % file_name
 208         )
 209     return reqs

 215 def get_crypto_req():
 216     """Detect custom crypto from ANSIBLE_CRYPTO_BACKEND env var.
 217 
 218     pycrypto or cryptography. We choose a default but allow the user to
 219     override it. This translates into pip install of the sdist deciding what
 220     package to install and also the runtime dependencies that pkg_resources
 221     knows about.
 222     """
 223     crypto_backend = os.environ.get('ANSIBLE_CRYPTO_BACKEND', '').strip()
 224 
 225     if crypto_backend == PYCRYPTO_DIST:
 226         # Attempt to set version requirements
 227         return '%s >= 2.6' % PYCRYPTO_DIST
 228 
 229     return crypto_backend or None

 232 def substitute_crypto_to_req(req):
 233     """Replace crypto requirements if customized."""
 234     crypto_backend = get_crypto_req()
 235 
 236     if crypto_backend is None:
 237         return req
 238 
 239     def is_not_crypto(r):
 240         CRYPTO_LIBS = PYCRYPTO_DIST, 'cryptography'
 241         return not any(r.lower().startswith(c) for c in CRYPTO_LIBS)
 242 
 243     return [r for r in req if is_not_crypto(r)] + [crypto_backend]

 328 def main():
 329     """Invoke installation process using setuptools."""
 330     setup_params = dict(static_setup_params, **get_dynamic_setup_params())
 331     ignore_warning_regex = (
 332         r"Unknown distribution option: '(project_urls|python_requires)'"
 333     )
 334     warnings.filterwarnings(
 335         'ignore',
 336         message=ignore_warning_regex,
 337         category=UserWarning,
 338         module='distutils.dist',
 339     )
 340     setup(**setup_params)
