  60 def regex_match(s, pattern):
  61     '''Custom filter for regex matching'''
  62     reg = re.compile(pattern)
  63     if reg.match(s):
  64         return True
  65     else:
  66         return False

  69 def select_chain_match(inlist, key, pattern):
  70     '''Get a key from a list of dicts, squash values to a single list, then filter'''
  71     outlist = [x[key] for x in inlist]
  72     outlist = list(itertools.chain(*outlist))
  73     outlist = [x for x in outlist if regex_match(x, pattern)]
  74     return outlist

 172     def is_cache_valid(self):
 173         ''' Determines if the cache files have expired, or if it is still valid '''
 174 
 175         valid = False
 176 
 177         if os.path.isfile(self.cache_path_cache):
 178             mod_time = os.path.getmtime(self.cache_path_cache)
 179             current_time = time()
 180             if (mod_time + self.cache_max_age) > current_time:
 181                 valid = True
 182 
 183         return valid

 342     def get_instances(self):
 343         ''' Get a list of vm instances with pyvmomi '''
 344         kwargs = {'host': self.server,
 345                   'user': self.username,
 346                   'pwd': self.password,
 347                   'port': int(self.port)}
 348 
 349         if self.validate_certs and hasattr(ssl, 'SSLContext'):
 350             context = ssl.SSLContext(ssl.PROTOCOL_SSLv23)
 351             context.verify_mode = ssl.CERT_REQUIRED
 352             context.check_hostname = True
 353             kwargs['sslContext'] = context
 354         elif self.validate_certs and not hasattr(ssl, 'SSLContext'):
 355             sys.exit('pyVim does not support changing verification mode with python < 2.7.9. Either update '
 356                      'python or use validate_certs=false.')
 357         elif not self.validate_certs and hasattr(ssl, 'SSLContext'):
 358             context = ssl.SSLContext(ssl.PROTOCOL_SSLv23)
 359             context.verify_mode = ssl.CERT_NONE
 360             context.check_hostname = False
 361             kwargs['sslContext'] = context
 362         elif not self.validate_certs and not hasattr(ssl, 'SSLContext'):
 363             # Python 2.7.9 < or RHEL/CentOS 7.4 <
 364             pass
 365 
 366         return self._get_instances(kwargs)

 368     def _get_instances(self, inkwargs):
 369         ''' Make API calls '''
 370         instances = []
 371         si = None
 372         try:
 373             si = SmartConnect(**inkwargs)
 374         except ssl.SSLError as connection_error:
 375             if '[SSL: CERTIFICATE_VERIFY_FAILED]' in str(connection_error) and self.validate_certs:
 376                 sys.exit("Unable to connect to ESXi server due to %s, "
 377                          "please specify validate_certs=False and try again" % connection_error)
 378 
 379         except Exception as exc:
 380             self.debugl("Unable to connect to ESXi server due to %s" % exc)
 381             sys.exit("Unable to connect to ESXi server due to %s" % exc)
 382 
 383         self.debugl('retrieving all instances')
 384         if not si:
 385             sys.exit("Could not connect to the specified host using specified "
 386                      "username and password")
 387         atexit.register(Disconnect, si)
 388         content = si.RetrieveContent()
 389 
 390         # Create a search container for virtualmachines
 391         self.debugl('creating containerview for virtualmachines')
 392         container = content.rootFolder
 393         viewType = [vim.VirtualMachine]
 394         recursive = True
 395         containerView = content.viewManager.CreateContainerView(container, viewType, recursive)
 396         children = containerView.view
 397         for child in children:
 398             # If requested, limit the total number of instances
 399             if self.args.max_instances:
 400                 if len(instances) >= self.args.max_instances:
 401                     break
 402             instances.append(child)
 403         self.debugl("%s total instances in container view" % len(instances))
 404 
 405         if self.args.host:
 406             instances = [x for x in instances if x.name == self.args.host]
 407 
 408         instance_tuples = []
 409         for instance in instances:
 410             if self.guest_props:
 411                 ifacts = self.facts_from_proplist(instance)
 412             else:
 413                 ifacts = self.facts_from_vobj(instance)
 414             instance_tuples.append((instance, ifacts))
 415         self.debugl('facts collected for all instances')
 416 
 417         try:
 418             cfm = content.customFieldsManager
 419             if cfm is not None and cfm.field:
 420                 for f in cfm.field:
 421                     if not f.managedObjectType or f.managedObjectType == vim.VirtualMachine:
 422                         self.custom_fields[f.key] = f.name
 423                 self.debugl('%d custom fields collected' % len(self.custom_fields))
 424         except vmodl.RuntimeFault as exc:
 425             self.debugl("Unable to gather custom fields due to %s" % exc.msg)
 426         except IndexError as exc:
 427             self.debugl("Unable to gather custom fields due to %s" % exc)
 428 
 429         return instance_tuples

 431     def instances_to_inventory(self, instances):
 432         ''' Convert a list of vm objects into a json compliant inventory '''
 433         self.debugl('re-indexing instances based on ini settings')
 434         inventory = VMWareInventory._empty_inventory()
 435         inventory['all'] = {}
 436         inventory['all']['hosts'] = []
 437         for idx, instance in enumerate(instances):
 438             # make a unique id for this object to avoid vmware's
 439             # numerous uuid's which aren't all unique.
 440             thisid = str(uuid.uuid4())
 441             idata = instance[1]
 442 
 443             # Put it in the inventory
 444             inventory['all']['hosts'].append(thisid)
 445             inventory['_meta']['hostvars'][thisid] = idata.copy()
 446             inventory['_meta']['hostvars'][thisid]['ansible_uuid'] = thisid
 447 
 448         # Make a map of the uuid to the alias the user wants
 449         name_mapping = self.create_template_mapping(
 450             inventory,
 451             self.config.get('vmware', 'alias_pattern')
 452         )
 453 
 454         # Make a map of the uuid to the ssh hostname the user wants
 455         host_mapping = self.create_template_mapping(
 456             inventory,
 457             self.config.get('vmware', 'host_pattern')
 458         )
 459 
 460         # Reset the inventory keys
 461         for k, v in name_mapping.items():
 462 
 463             if not host_mapping or k not in host_mapping:
 464                 continue
 465 
 466             # set ansible_host (2.x)
 467             try:
 468                 inventory['_meta']['hostvars'][k]['ansible_host'] = host_mapping[k]
 469                 # 1.9.x backwards compliance
 470                 inventory['_meta']['hostvars'][k]['ansible_ssh_host'] = host_mapping[k]
 471             except Exception:
 472                 continue
 473 
 474             if k == v:
 475                 continue
 476 
 477             # add new key
 478             inventory['all']['hosts'].append(v)
 479             inventory['_meta']['hostvars'][v] = inventory['_meta']['hostvars'][k]
 480 
 481             # cleanup old key
 482             inventory['all']['hosts'].remove(k)
 483             inventory['_meta']['hostvars'].pop(k, None)
 484 
 485         self.debugl('pre-filtered hosts:')
 486         for i in inventory['all']['hosts']:
 487             self.debugl('  * %s' % i)
 488         # Apply host filters
 489         for hf in self.host_filters:
 490             if not hf:
 491                 continue
 492             self.debugl('filter: %s' % hf)
 493             filter_map = self.create_template_mapping(inventory, hf, dtype='boolean')
 494             for k, v in filter_map.items():
 495                 if not v:
 496                     # delete this host
 497                     inventory['all']['hosts'].remove(k)
 498                     inventory['_meta']['hostvars'].pop(k, None)
 499 
 500         self.debugl('post-filter hosts:')
 501         for i in inventory['all']['hosts']:
 502             self.debugl('  * %s' % i)
 503 
 504         # Create groups
 505         for gbp in self.groupby_patterns:
 506             groupby_map = self.create_template_mapping(inventory, gbp)
 507             for k, v in groupby_map.items():
 508                 if v not in inventory:
 509                     inventory[v] = {}
 510                     inventory[v]['hosts'] = []
 511                 if k not in inventory[v]['hosts']:
 512                     inventory[v]['hosts'].append(k)
 513 
 514         if self.config.get('vmware', 'groupby_custom_field'):
 515             for k, v in inventory['_meta']['hostvars'].items():
 516                 if 'customvalue' in v:
 517                     for tv in v['customvalue']:
 518                         newkey = None
 519                         field_name = self.custom_fields[tv['key']] if tv['key'] in self.custom_fields else tv['key']
 520                         if field_name in self.groupby_custom_field_excludes:
 521                             continue
 522                         values = []
 523                         keylist = map(lambda x: x.strip(), tv['value'].split(','))
 524                         for kl in keylist:
 525                             try:
 526                                 newkey = "%s%s_%s" % (self.config.get('vmware', 'custom_field_group_prefix'), str(field_name), kl)
 527                                 newkey = newkey.strip()
 528                             except Exception as e:
 529                                 self.debugl(e)
 530                             values.append(newkey)
 531                         for tag in values:
 532                             if not tag:
 533                                 continue
 534                             if tag not in inventory:
 535                                 inventory[tag] = {}
 536                                 inventory[tag]['hosts'] = []
 537                             if k not in inventory[tag]['hosts']:
 538                                 inventory[tag]['hosts'].append(k)
 539 
 540         return inventory

 542     def create_template_mapping(self, inventory, pattern, dtype='string'):
 543         ''' Return a hash of uuid to templated string from pattern '''
 544         mapping = {}
 545         for k, v in inventory['_meta']['hostvars'].items():
 546             t = self.env.from_string(pattern)
 547             newkey = None
 548             try:
 549                 newkey = t.render(v)
 550                 newkey = newkey.strip()
 551             except Exception as e:
 552                 self.debugl(e)
 553             if not newkey:
 554                 continue
 555             elif dtype == 'integer':
 556                 newkey = int(newkey)
 557             elif dtype == 'boolean':
 558                 if newkey.lower() == 'false':
 559                     newkey = False
 560                 elif newkey.lower() == 'true':
 561                     newkey = True
 562             elif dtype == 'string':
 563                 pass
 564             mapping[k] = newkey
 565         return mapping

 567     def facts_from_proplist(self, vm):
 568         '''Get specific properties instead of serializing everything'''
 569 
 570         rdata = {}
 571         for prop in self.guest_props:
 572             self.debugl('getting %s property for %s' % (prop, vm.name))
 573             key = prop
 574             if self.lowerkeys:
 575                 key = key.lower()
 576 
 577             if '.' not in prop:
 578                 # props without periods are direct attributes of the parent
 579                 vm_property = getattr(vm, prop)
 580                 if isinstance(vm_property, vim.CustomFieldsManager.Value.Array):
 581                     temp_vm_property = []
 582                     for vm_prop in vm_property:
 583                         temp_vm_property.append({'key': vm_prop.key,
 584                                                  'value': vm_prop.value})
 585                     rdata[key] = temp_vm_property
 586                 else:
 587                     rdata[key] = vm_property
 588             else:
 589                 # props with periods are subkeys of parent attributes
 590                 parts = prop.split('.')
 591                 total = len(parts) - 1
 592 
 593                 # pointer to the current object
 594                 val = None
 595                 # pointer to the current result key
 596                 lastref = rdata
 597 
 598                 for idx, x in enumerate(parts):
 599 
 600                     if isinstance(val, dict):
 601                         if x in val:
 602                             val = val.get(x)
 603                         elif x.lower() in val:
 604                             val = val.get(x.lower())
 605                     else:
 606                         # if the val wasn't set yet, get it from the parent
 607                         if not val:
 608                             try:
 609                                 val = getattr(vm, x)
 610                             except AttributeError as e:
 611                                 self.debugl(e)
 612                         else:
 613                             # in a subkey, get the subprop from the previous attrib
 614                             try:
 615                                 val = getattr(val, x)
 616                             except AttributeError as e:
 617                                 self.debugl(e)
 618 
 619                         # make sure it serializes
 620                         val = self._process_object_types(val)
 621 
 622                     # lowercase keys if requested
 623                     if self.lowerkeys:
 624                         x = x.lower()
 625 
 626                     # change the pointer or set the final value
 627                     if idx != total:
 628                         if x not in lastref:
 629                             lastref[x] = {}
 630                         lastref = lastref[x]
 631                     else:
 632                         lastref[x] = val
 633         if self.args.debug:
 634             self.debugl("For %s" % vm.name)
 635             for key in list(rdata.keys()):
 636                 if isinstance(rdata[key], dict):
 637                     for ikey in list(rdata[key].keys()):
 638                         self.debugl("Property '%s.%s' has value '%s'" % (key, ikey, rdata[key][ikey]))
 639                 else:
 640                     self.debugl("Property '%s' has value '%s'" % (key, rdata[key]))
 641         return rdata

 643     def facts_from_vobj(self, vobj, level=0):
 644         ''' Traverse a VM object and return a json compliant data structure '''
 645 
 646         # pyvmomi objects are not yet serializable, but may be one day ...
 647         # https://github.com/vmware/pyvmomi/issues/21
 648 
 649         # WARNING:
 650         # Accessing an object attribute will trigger a SOAP call to the remote.
 651         # Increasing the attributes collected or the depth of recursion greatly
 652         # increases runtime duration and potentially memory+network utilization.
 653 
 654         if level == 0:
 655             try:
 656                 self.debugl("get facts for %s" % vobj.name)
 657             except Exception as e:
 658                 self.debugl(e)
 659 
 660         rdata = {}
 661 
 662         methods = dir(vobj)
 663         methods = [str(x) for x in methods if not x.startswith('_')]
 664         methods = [x for x in methods if x not in self.bad_types]
 665         methods = [x for x in methods if not x.lower() in self.skip_keys]
 666         methods = sorted(methods)
 667 
 668         for method in methods:
 669             # Attempt to get the method, skip on fail
 670             try:
 671                 methodToCall = getattr(vobj, method)
 672             except Exception as e:
 673                 continue
 674 
 675             # Skip callable methods
 676             if callable(methodToCall):
 677                 continue
 678 
 679             if self.lowerkeys:
 680                 method = method.lower()
 681 
 682             rdata[method] = self._process_object_types(
 683                 methodToCall,
 684                 thisvm=vobj,
 685                 inkey=method,
 686             )
 687 
 688         return rdata

 690     def _process_object_types(self, vobj, thisvm=None, inkey='', level=0):
 691         ''' Serialize an object '''
 692         rdata = {}
 693 
 694         if type(vobj).__name__ in self.vimTableMaxDepth and level >= self.vimTableMaxDepth[type(vobj).__name__]:
 695             return rdata
 696 
 697         if vobj is None:
 698             rdata = None
 699         elif type(vobj) in self.vimTable:
 700             rdata = {}
 701             for key in self.vimTable[type(vobj)]:
 702                 try:
 703                     rdata[key] = getattr(vobj, key)
 704                 except Exception as e:
 705                     self.debugl(e)
 706 
 707         elif issubclass(type(vobj), str) or isinstance(vobj, str):
 708             if vobj.isalnum():
 709                 rdata = vobj
 710             else:
 711                 rdata = vobj.encode('utf-8').decode('utf-8')
 712         elif issubclass(type(vobj), bool) or isinstance(vobj, bool):
 713             rdata = vobj
 714         elif issubclass(type(vobj), integer_types) or isinstance(vobj, integer_types):
 715             rdata = vobj
 716         elif issubclass(type(vobj), float) or isinstance(vobj, float):
 717             rdata = vobj
 718         elif issubclass(type(vobj), list) or issubclass(type(vobj), tuple):
 719             rdata = []
 720             try:
 721                 vobj = sorted(vobj)
 722             except Exception:
 723                 pass
 724 
 725             for idv, vii in enumerate(vobj):
 726                 if level + 1 <= self.maxlevel:
 727                     vid = self._process_object_types(
 728                         vii,
 729                         thisvm=thisvm,
 730                         inkey=inkey + '[' + str(idv) + ']',
 731                         level=(level + 1)
 732                     )
 733 
 734                     if vid:
 735                         rdata.append(vid)
 736 
 737         elif issubclass(type(vobj), dict):
 738             pass
 739 
 740         elif issubclass(type(vobj), object):
 741             methods = dir(vobj)
 742             methods = [str(x) for x in methods if not x.startswith('_')]
 743             methods = [x for x in methods if x not in self.bad_types]
 744             methods = [x for x in methods if not inkey + '.' + x.lower() in self.skip_keys]
 745             methods = sorted(methods)
 746 
 747             for method in methods:
 748                 # Attempt to get the method, skip on fail
 749                 try:
 750                     methodToCall = getattr(vobj, method)
 751                 except Exception as e:
 752                     continue
 753 
 754                 if callable(methodToCall):
 755                     continue
 756 
 757                 if self.lowerkeys:
 758                     method = method.lower()
 759                 if level + 1 <= self.maxlevel:
 760                     try:
 761                         rdata[method] = self._process_object_types(
 762                             methodToCall,
 763                             thisvm=thisvm,
 764                             inkey=inkey + '.' + method,
 765                             level=(level + 1)
 766                         )
 767                     except vim.fault.NoPermission:
 768                         self.debugl("Skipping method %s (NoPermission)" % method)
 769         else:
 770             pass
 771 
 772         return rdata

 774     def get_host_info(self, host):
 775         ''' Return hostvars for a single host '''
 776 
 777         if host in self.inventory['_meta']['hostvars']:
 778             return self.inventory['_meta']['hostvars'][host]
 779         elif self.args.host and self.inventory['_meta']['hostvars']:
 780             match = None
 781             for k, v in self.inventory['_meta']['hostvars'].items():
 782                 if self.inventory['_meta']['hostvars'][k]['name'] == self.args.host:
 783                     match = k
 784                     break
 785             if match:
 786                 return self.inventory['_meta']['hostvars'][match]
 787             else:
 788                 raise VMwareMissingHostException('%s not found' % host)
 789         else:
 790             raise VMwareMissingHostException('%s not found' % host)
