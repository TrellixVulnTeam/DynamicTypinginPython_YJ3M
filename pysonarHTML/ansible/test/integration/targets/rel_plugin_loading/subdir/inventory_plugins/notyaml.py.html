  72 class InventoryModule(BaseFileInventoryPlugin):
  73 
  74     NAME = 'yaml'
  75 
  76     def __init__(self):
  77 
  78         super(InventoryModule, self).__init__()
  79 
  80     def verify_file(self, path):
  81 
  82         valid = False
  83         if super(InventoryModule, self).verify_file(path):
  84             file_name, ext = os.path.splitext(path)
  85             if not ext or ext in self.get_option('yaml_extensions'):
  86                 valid = True
  87         return valid
  88 
  89     def parse(self, inventory, loader, path, cache=True):
  90         ''' parses the inventory file '''
  91 
  92         super(InventoryModule, self).parse(inventory, loader, path)
  93         self.set_options()
  94 
  95         try:
  96             data = self.loader.load_from_file(path, cache=False)
  97         except Exception as e:
  98             raise AnsibleParserError(e)
  99 
 100         if not data:
 101             raise AnsibleParserError('Parsed empty YAML file')
 102         elif not isinstance(data, MutableMapping):
 103             raise AnsibleParserError('YAML inventory has invalid structure, it should be a dictionary, got: %s' % type(data))
 104         elif data.get('plugin'):
 105             raise AnsibleParserError('Plugin configuration YAML file, not YAML inventory')
 106 
 107         # We expect top level keys to correspond to groups, iterate over them
 108         # to get host, vars and subgroups (which we iterate over recursivelly)
 109         if isinstance(data, MutableMapping):
 110             for group_name in data:
 111                 self._parse_group(group_name, data[group_name])
 112         else:
 113             raise AnsibleParserError("Invalid data from file, expected dictionary and got:\n\n%s" % to_native(data))
 114 
 115     def _parse_group(self, group, group_data):
 116 
 117         if isinstance(group_data, (MutableMapping, NoneType)):
 118 
 119             try:
 120                 self.inventory.add_group(group)
 121             except AnsibleError as e:
 122                 raise AnsibleParserError("Unable to add group %s: %s" % (group, to_text(e)))
 123 
 124             if group_data is not None:
 125                 # make sure they are dicts
 126                 for section in ['vars', 'children', 'hosts']:
 127                     if section in group_data:
 128                         # convert strings to dicts as these are allowed
 129                         if isinstance(group_data[section], string_types):
 130                             group_data[section] = {group_data[section]: None}
 131 
 132                         if not isinstance(group_data[section], (MutableMapping, NoneType)):
 133                             raise AnsibleParserError('Invalid "%s" entry for "%s" group, requires a dictionary, found "%s" instead.' %
 134                                                      (section, group, type(group_data[section])))
 135 
 136                 for key in group_data:
 137 
 138                     if not isinstance(group_data[key], (MutableMapping, NoneType)):
 139                         self.display.warning('Skipping key (%s) in group (%s) as it is not a mapping, it is a %s' % (key, group, type(group_data[key])))
 140                         continue
 141 
 142                     if isinstance(group_data[key], NoneType):
 143                         self.display.vvv('Skipping empty key (%s) in group (%s)' % (key, group))
 144                     elif key == 'vars':
 145                         for var in group_data[key]:
 146                             self.inventory.set_variable(group, var, group_data[key][var])
 147                     elif key == 'children':
 148                         for subgroup in group_data[key]:
 149                             self._parse_group(subgroup, group_data[key][subgroup])
 150                             self.inventory.add_child(group, subgroup)
 151 
 152                     elif key == 'hosts':
 153                         for host_pattern in group_data[key]:
 154                             hosts, port = self._parse_host(host_pattern)
 155                             self._populate_host_vars(hosts, group_data[key][host_pattern] or {}, group, port)
 156                     else:
 157                         self.display.warning('Skipping unexpected key (%s) in group (%s), only "vars", "children" and "hosts" are valid' % (key, group))
 158 
 159         else:
 160             self.display.warning("Skipping '%s' as this is not a valid group definition" % group)
 161 
 162     def _parse_host(self, host_pattern):
 163         '''
 164         Each host key can be a pattern, try to process it and add variables as needed
 165         '''
 166         (hostnames, port) = self._expand_hostpattern(host_pattern)
 167 
 168         return hostnames, port
