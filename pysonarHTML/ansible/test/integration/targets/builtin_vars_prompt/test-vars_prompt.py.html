  24 def run_test(playbook, test_spec, args=None, timeout=10, env=None):
  25 
  26     if not env:
  27         env = os.environ.copy()
  28     env.update(env_vars)
  29 
  30     if not args:
  31         args = sys.argv[1:]

  33     vars_prompt_test = pexpect.spawn(
  34         'ansible-playbook',
  35         args=[playbook] + args,
  36         timeout=timeout,
  37         env=env,
  38     )
  39 
  40     vars_prompt_test.logfile = log_buffer
  41     for item in test_spec[0]:

  46     vars_prompt_test.expect(pexpect.EOF)
  47     vars_prompt_test.close()
