 147     def write_script(self):
 148         fdir = os.path.join(TESTDIR, 'inventory')

 179     def write_playbook(self):
 180         fname = os.path.join(TESTDIR, 'site.yml')
 181         pb_copy = self.playbook.copy()
 182 
 183         if self.playvars:
 184             pb_copy['vars'] = self.playvars
 185         if self.varsfiles:
 186             pb_copy['vars_files'] = self.varsfiles
 187         if self.roles:
 188             pb_copy['roles'] = []
 189             for role in self.roles:
 190                 role.write_role()
 191                 role_def = dict(role=role.name)
 192                 role_def.update(role.params)
 193                 pb_copy['roles'].append(role_def)
 194         if self.tasks:
 195             pb_copy['tasks'] = self.tasks
 196 
 197         with open(fname, 'w') as f:
 198             pb_yaml = yaml.dump([pb_copy], f, default_flow_style=False, indent=2)

 377     def run(self):
 378         '''
 379         if self.dynamic_inventory:
 380             cmd = 'ansible-playbook -c local -i inventory/hosts site.yml'
 381         else:
 382             cmd = 'ansible-playbook -c local -i inventory site.yml'
 383         '''
 384         cmd = 'ansible-playbook -c local -i inventory site.yml'
 385         if 'extra_vars' in self.features:
 386             cmd += ' --extra-vars="findme=extra_vars"'
 387         cmd = cmd + ' -vvvvv'
 388         self.ansible_command = cmd
 389         (rc, so, se) = run_command(cmd, cwd=TESTDIR)
 390         self.stdout = so
 391 
 392         if rc != 0:
 393             raise Exception("playbook failed (rc=%s), stdout: '%s' stderr: '%s'" % (rc, so, se))

 435         # 'ini_group_vars_file_child', #FIXME: this contradicts documented precedence pb group vars files should override inventory ones

 446 
 447     parser = OptionParser()
 448     parser.add_option('-f', '--feature', action='append')
 449     parser.add_option('--use_dynamic_inventory', action='store_true')
 450     parser.add_option('--show_tree', action='store_true')
 451     parser.add_option('--show_content', action='store_true')
 452     parser.add_option('--show_stdout', action='store_true')
 453     parser.add_option('--copy_testcases_to_local_dir', action='store_true')
 454     (options, args) = parser.parse_args()
 455 
 456     if options.feature:
 457         for f in options.feature:
 458             if f not in features:
 459                 print('%s is not a valid feature' % f)
 460                 sys.exit(1)
 461         features = list(options.feature)
 462 
 463     fdesc = {
 464         'ini_host': 'host var inside the ini',
 465         'script_host': 'host var inside the script _meta',
 466         'ini_child': 'child group var inside the ini',
 467         'script_child': 'child group var inside the script',
 468         'ini_parent': 'parent group var inside the ini',
 469         'script_parent': 'parent group var inside the script',
 470         'ini_all': 'all group var inside the ini',
 471         'script_all': 'all group var inside the script',
 472         'ini_host_vars_file': 'var in inventory/host_vars/host',
 473         'ini_group_vars_file_parent': 'var in inventory/group_vars/parent',
 474         'ini_group_vars_file_child': 'var in inventory/group_vars/child',
 475         'ini_group_vars_file_all': 'var in inventory/group_vars/all',
 476         'pb_group_vars_file_parent': 'var in playbook/group_vars/parent',
 477         'pb_group_vars_file_child': 'var in playbook/group_vars/child',
 478         'pb_group_vars_file_all': 'var in playbook/group_vars/all',
 479         'pb_host_vars_file': 'var in playbook/host_vars/host',
 480         'play_var': 'var set in playbook header',
 481         'role_parent_default': 'var in roles/role_parent/defaults/main.yml',
 482         'role_default': 'var in roles/role/defaults/main.yml',
 483         'role_var': 'var in ???',
 484         'include_vars': 'var in included file',
 485         'set_fact': 'var made by set_fact',
 486         'vars_file': 'var in file added by vars_file',
 487         'block_vars': 'vars defined on the block',
 488         'task_vars': 'vars defined on the task',
 489         'extra_vars': 'var passed via the cli'
 490     }
 491 
 492     dinv = options.use_dynamic_inventory
 493     if dinv:
 494         # some features are specific to ini, so swap those
 495         for (idx, x) in enumerate(features):
 496             if x.startswith('ini_') and 'vars_file' not in x:
 497                 features[idx] = x.replace('ini_', 'script_')
 498 
 499     dinv = options.use_dynamic_inventory
 500 
 501     index = 1
 502     while features:
 503         VTM = VarTestMaker(features, dynamic_inventory=dinv)
 504         VTM.build()
 505 
 506         if options.show_tree or options.show_content or options.show_stdout:
 507             print('')
 508         if options.show_tree:
 509             VTM.show_tree()
 510         if options.show_content:
 511             VTM.show_content()
 512 
 513         try:
 514             print("CHECKING: %s (%s)" % (features[0], fdesc.get(features[0], '')))
 515             res = VTM.run()
 516             if options.show_stdout:
 517                 VTM.show_stdout()
 518 
 519             features.pop(0)
 520 
 521             if options.copy_testcases_to_local_dir:
 522                 topdir = 'testcases'
 523                 if index == 1 and os.path.isdir(topdir):
 524                     shutil.rmtree(topdir)
 525                 if not os.path.isdir(topdir):
 526                     os.makedirs(topdir)
 527                 thisindex = str(index)
 528                 if len(thisindex) == 1:
 529                     thisindex = '0' + thisindex
 530                 thisdir = os.path.join(topdir, '%s.%s' % (thisindex, res))
 531                 shutil.copytree(TESTDIR, thisdir)
 532 
 533         except Exception as e:
 534             print("ERROR !!!")
 535             print(e)
 536             print('feature: %s failed' % features[0])
 537             sys.exit(1)
 538         finally:
 539             shutil.rmtree(TESTDIR)
 540             index += 1
