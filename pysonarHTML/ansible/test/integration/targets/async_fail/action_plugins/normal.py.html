  25 class ActionModule(ActionBase):
  26 
  27     def run(self, tmp=None, task_vars=None):
  28 
  29         # individual modules might disagree but as the generic the action plugin, pass at this point.
  30         self._supports_check_mode = True
  31         self._supports_async = True
  32 
  33         result = super(ActionModule, self).run(tmp, task_vars)
  34         del tmp  # tmp no longer has any effect
  35 
  36         if not result.get('skipped'):
  37 
  38             if result.get('invocation', {}).get('module_args'):
  39                 # avoid passing to modules in case of no_log
  40                 # should not be set anymore but here for backwards compatibility
  41                 del result['invocation']['module_args']
  42 
  43             # FUTURE: better to let _execute_module calculate this internally?
  44             wrap_async = self._task.async_val and not self._connection.has_native_async
  45 
  46             # do work!
  47             result = merge_hash(result, self._execute_module(task_vars=task_vars, wrap_async=wrap_async))
  48 
  49             # hack to keep --verbose from showing all the setup module result
  50             # moved from setup module as now we filter out all _ansible_ from result
  51             if self._task.action == 'setup':
  52                 result['_ansible_verbose_override'] = True
  53 
  54         # Simulate a transient network failure
  55         if self._task.action == 'async_status' and 'finished' in result and result['finished'] != 1:
  56             raise AnsibleError('Pretend to fail somewher ein executing async_status')
  57 
  58         if not wrap_async:
  59             # remove a temporary path we created
  60             self._remove_tmp_path(self._connection._shell.tmpdir)
  61 
  62         return result
