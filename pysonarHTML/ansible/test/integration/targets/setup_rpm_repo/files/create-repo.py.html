  15     RPM('dinginessentail', '1.0', '1', None, None),

  27 def main():
  28     try:
  29         arch = sys.argv[1]
  30     except IndexError:
  31         arch = 'x86_64'
  32 
  33     pkgs = []
  34     for spec in SPECS:

  36         pkg.epoch = spec.epoch
  37 
  38         if spec.recommends:
  39             # Skip packages that require weak deps but an older version of RPM is being used
  40             if not hasattr(rpmfluff, "can_use_rpm_weak_deps") or not rpmfluff.can_use_rpm_weak_deps():
  41                 continue
  42 
  43             for recommend in spec.recommends:
  44                 pkg.add_recommends(recommend)
  45 
  46         pkgs.append(pkg)

  48     repo = rpmfluff.YumRepoBuild(pkgs)
  49     repo.make(arch)
  50 
  51     for pkg in pkgs:
  52         pkg.clean()
  53 
  54     print(repo.repoDir)

  56 
  57 if __name__ == "__main__":
