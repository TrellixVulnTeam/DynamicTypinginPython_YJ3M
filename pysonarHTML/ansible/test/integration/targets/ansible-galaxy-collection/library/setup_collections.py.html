  80 import os
  81 import tempfile
  82 import yaml

  84 from ansible.module_utils.basic import AnsibleModule
  85 from ansible.module_utils._text import to_bytes

  88 def run_module():
  89     module_args = dict(
  90         server=dict(type='str', required=True),
  91         token=dict(type='str'),
  92         collections=dict(
  93             type='list',
  94             elements='dict',
  95             required=True,
  96             options=dict(
  97                 namespace=dict(type='str', required=True),
  98                 name=dict(type='str', required=True),
  99                 version=dict(type='str', default='1.0.0'),
 100                 dependencies=dict(type='dict', default={}),
 101                 use_symlink=dict(type='bool', default=False),
 102             ),
 103         ),
 104     )
 105 
 106     module = AnsibleModule(
 107         argument_spec=module_args,
 108         supports_check_mode=False
 109     )
 110 
 111     result = dict(changed=True, results=[])
 112 
 113     for idx, collection in enumerate(module.params['collections']):
 114         collection_dir = os.path.join(module.tmpdir, "%s-%s-%s" % (collection['namespace'], collection['name'],
 115                                                                    collection['version']))
 116         b_collection_dir = to_bytes(collection_dir, errors='surrogate_or_strict')
 117         os.mkdir(b_collection_dir)
 118 
 119         with open(os.path.join(b_collection_dir, b'README.md'), mode='wb') as fd:
 120             fd.write(b"Collection readme")
 121 
 122         galaxy_meta = {
 123             'namespace': collection['namespace'],
 124             'name': collection['name'],
 125             'version': collection['version'],
 126             'readme': 'README.md',
 127             'authors': ['Collection author <name@email.com'],
 128             'dependencies': collection['dependencies'],
 129             'license': ['GPL-3.0-or-later'],
 130         }
 131         with open(os.path.join(b_collection_dir, b'galaxy.yml'), mode='wb') as fd:
 132             fd.write(to_bytes(yaml.safe_dump(galaxy_meta), errors='surrogate_or_strict'))
 133 
 134         with tempfile.NamedTemporaryFile(mode='wb') as temp_fd:
 135             temp_fd.write(b"data")
 136 
 137             if collection['use_symlink']:
 138                 os.mkdir(os.path.join(b_collection_dir, b'docs'))
 139                 os.mkdir(os.path.join(b_collection_dir, b'plugins'))
 140                 b_target_file = b'RE\xc3\x85DM\xc3\x88.md'
 141                 with open(os.path.join(b_collection_dir, b_target_file), mode='wb') as fd:
 142                     fd.write(b'data')
 143 
 144                 os.symlink(b_target_file, os.path.join(b_collection_dir, b_target_file + b'-link'))
 145                 os.symlink(temp_fd.name, os.path.join(b_collection_dir, b_target_file + b'-outside-link'))
 146                 os.symlink(os.path.join(b'..', b_target_file), os.path.join(b_collection_dir, b'docs', b_target_file))
 147                 os.symlink(os.path.join(b_collection_dir, b_target_file),
 148                            os.path.join(b_collection_dir, b'plugins', b_target_file))
 149                 os.symlink(b'docs', os.path.join(b_collection_dir, b'docs-link'))
 150 
 151             release_filename = '%s-%s-%s.tar.gz' % (collection['namespace'], collection['name'], collection['version'])
 152             collection_path = os.path.join(collection_dir, release_filename)
 153             rc, stdout, stderr = module.run_command(['ansible-galaxy', 'collection', 'build'], cwd=collection_dir)
 154             result['results'].append({
 155                 'build': {
 156                     'rc': rc,
 157                     'stdout': stdout,
 158                     'stderr': stderr,
 159                 }
 160             })
 161 
 162         # To save on time, skip the import wait until the last collection is being uploaded.
 163         publish_args = ['ansible-galaxy', 'collection', 'publish', collection_path, '--server',
 164                         module.params['server']]
 165         if module.params['token']:
 166             publish_args.extend(['--token', module.params['token']])
 167         if idx != (len(module.params['collections']) - 1):
 168             publish_args.append('--no-wait')
 169         rc, stdout, stderr = module.run_command(publish_args)
 170         result['results'][-1]['publish'] = {
 171             'rc': rc,
 172             'stdout': stdout,
 173             'stderr': stderr,
 174         }
 175 
 176     failed = bool(sum(
 177         r['build']['rc'] + r['publish']['rc'] for r in result['results']
 178     ))
 179 
 180     module.exit_json(failed=failed, **result)
 181
