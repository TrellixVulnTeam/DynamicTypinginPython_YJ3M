  98 def assemble_files_to_install(complete_file_list):
  99     """
 100     This looks for all of the files which should show up in an installation of ansible
 101     """
 102     ignore_patterns = tuple()
 103 
 104     pkg_data_files = []
 105     for path in complete_file_list:
 106 
 107         if path.startswith("lib/ansible"):
 108             prefix = 'lib'
 109         elif path.startswith("test/lib/ansible_test"):
 110             prefix = 'test/lib'
 111         else:
 112             continue
 113 
 114         for ignore in ignore_patterns:
 115             if fnmatch.fnmatch(path, ignore):
 116                 break
 117         else:
 118             pkg_data_files.append(os.path.relpath(path, prefix))
 119 
 120     return pkg_data_files

 124 def clean_repository(file_list):
 125     """Copy the repository to clean it of artifacts"""
 126     # Create a tempdir that will be the clean repo
 127     with tempfile.TemporaryDirectory() as repo_root:
 128         directories = set((repo_root + os.path.sep,))
 129 
 130         for filename in file_list:
 131             # Determine if we need to create the directory
 132             directory = os.path.dirname(filename)
 133             dest_dir = os.path.join(repo_root, directory)
 134             if dest_dir not in directories:
 135                 os.makedirs(dest_dir)
 136 
 137                 # Keep track of all the directories that now exist
 138                 path_components = directory.split(os.path.sep)
 139                 path = repo_root
 140                 for component in path_components:
 141                     path = os.path.join(path, component)
 142                     if path not in directories:
 143                         directories.add(path)
 144 
 145             # Copy the file
 146             shutil.copy2(filename, dest_dir, follow_symlinks=False)
 147 
 148         yield repo_root

 151 def create_sdist(tmp_dir):
 152     """Create an sdist in the repository"""
 153     create = subprocess.Popen(
 154         ['make', 'snapshot', 'SDIST_DIR=%s' % tmp_dir],
 155         stdout=subprocess.PIPE,
 156         stderr=subprocess.PIPE,
 157         universal_newlines=True,
 158     )
 159 
 160     stderr = create.communicate()[1]
 161 
 162     if create.returncode != 0:
 163         raise Exception('make snapshot failed:\n%s' % stderr)
 164 
 165     # Determine path to sdist
 166     tmp_dir_files = os.listdir(tmp_dir)
 167 
 168     if not tmp_dir_files:
 169         raise Exception('sdist was not created in the temp dir')
 170     elif len(tmp_dir_files) > 1:
 171         raise Exception('Unexpected extra files in the temp dir')
 172 
 173     return os.path.join(tmp_dir, tmp_dir_files[0])

 176 def extract_sdist(sdist_path, tmp_dir):
 177     """Untar the sdist"""
 178     # Untar the sdist from the tmp_dir
 179     with tarfile.open(os.path.join(tmp_dir, sdist_path), 'r|*') as sdist:
 180         sdist.extractall(path=tmp_dir)
 181 
 182     # Determine the sdist directory name
 183     sdist_filename = os.path.basename(sdist_path)
 184     tmp_dir_files = os.listdir(tmp_dir)
 185     try:
 186         tmp_dir_files.remove(sdist_filename)
 187     except ValueError:
 188         # Unexpected could not find original sdist in temp dir
 189         raise
 190 
 191     if len(tmp_dir_files) > 1:
 192         raise Exception('Unexpected extra files in the temp dir')
 193     elif len(tmp_dir_files) < 1:
 194         raise Exception('sdist extraction did not occur i nthe temp dir')
 195 
 196     return os.path.join(tmp_dir, tmp_dir_files[0])

 199 def install_sdist(tmp_dir, sdist_dir):
 200     """Install the extracted sdist into the temporary directory"""
 201     install = subprocess.Popen(
 202         ['python', 'setup.py', 'install', '--root=%s' % tmp_dir],
 203         stdout=subprocess.PIPE,
 204         stderr=subprocess.PIPE,
 205         universal_newlines=True,
 206         cwd=os.path.join(tmp_dir, sdist_dir),
 207     )
 208 
 209     stdout, stderr = install.communicate()
 210 
 211     if install.returncode != 0:
 212         raise Exception('sdist install failed:\n%s' % stderr)
 213 
 214     # Determine the prefix for the installed files
 215     match = re.search('^creating (%s/.*?/(?:site|dist)-packages)/ansible$' %
 216                       tmp_dir, stdout, flags=re.M)
 217     return match.group(1)

 220 def check_sdist_contains_expected(sdist_dir, to_ship_files):
 221     """Check that the files we expect to ship are present in the sdist"""
 222     results = []
 223     for filename in to_ship_files:
 224         path = os.path.join(sdist_dir, filename)
 225         if not os.path.exists(path):
 226             results.append('%s: File was not added to sdist' % filename)
 227 
 228     # Also changelog
 229     changelog_files = glob.glob(os.path.join(sdist_dir, 'changelogs/CHANGELOG-v2.[0-9]*.rst'))
 230     if not changelog_files:
 231         results.append('changelogs/CHANGELOG-v2.*.rst: Changelog file was not added to the sdist')
 232     elif len(changelog_files) > 1:
 233         results.append('changelogs/CHANGELOG-v2.*.rst: Too many changelog files: %s'
 234                        % changelog_files)
 235 
 236     return results

 239 def check_sdist_files_are_wanted(sdist_dir, to_ship_files):
 240     """Check that all files in the sdist are desired"""
 241     results = []
 242     for dirname, dummy, files in os.walk(sdist_dir):
 243         dirname = os.path.relpath(dirname, start=sdist_dir)
 244         if dirname == '.':
 245             dirname = ''
 246 
 247         for filename in files:
 248             path = os.path.join(dirname, filename)
 249             if path not in to_ship_files:
 250                 if fnmatch.fnmatch(path, 'changelogs/CHANGELOG-v2.[0-9]*.rst'):
 251                     # changelog files are expected
 252                     continue
 253 
 254                 # FIXME: ansible-test doesn't pass the paths of symlinks to us so we aren't
 255                 # checking those
 256                 if not os.path.islink(os.path.join(sdist_dir, path)):
 257                     results.append('%s: File in sdist was not in the repository' % path)
 258 
 259     return results

 273 EGG_RE = re.compile('ansible[^/]+\\.egg-info/(PKG-INFO|SOURCES.txt|'
 274                     'dependency_links.txt|not-zip-safe|requires.txt|top_level.txt)$')

 277 def check_installed_files_are_wanted(install_dir, to_install_files):
 278     """Check that all installed files were desired"""
 279     results = []
 280 
 281     for dirname, dummy, files in os.walk(install_dir):
 282         dirname = os.path.relpath(dirname, start=install_dir)
 283         if dirname == '.':
 284             dirname = ''
 285 
 286         for filename in files:
 287             # If this is a byte code cache, look for the python file's name
 288             directory = dirname
 289             if filename.endswith('.pyc') or filename.endswith('.pyo'):
 290                 # Remove the trailing "o" or c"
 291                 filename = filename[:-1]
 292 
 293                 if directory.endswith('%s__pycache__' % os.path.sep):
 294                     # Python3 byte code cache, look for the basename of
 295                     # __pycache__/__init__.cpython-36.py
 296                     segments = filename.rsplit('.', 2)
 297                     if len(segments) >= 3:
 298                         filename = '.'.join((segments[0], segments[2]))
 299                         directory = os.path.dirname(directory)
 300 
 301             path = os.path.join(directory, filename)
 302 
 303             # Test that the file was listed for installation
 304             if path not in to_install_files:
 305                 # FIXME: ansible-test doesn't pass the paths of symlinks to us so we
 306                 # aren't checking those
 307                 if not os.path.islink(os.path.join(install_dir, path)):
 308                     if not EGG_RE.match(path):
 309                         results.append('%s: File was installed but was not supposed to be' % path)
 310 
 311     return results

 314 def _find_symlinks():
 315     symlink_list = []
 316     for dirname, directories, filenames in os.walk('.'):
 317         for filename in filenames:
 318             path = os.path.join(dirname, filename)
 319             # Strip off "./" from the front
 320             path = path[2:]
 321             if os.path.islink(path):
 322                 symlink_list.append(path)
 323 
 324     return symlink_list

 327 def main():
 328     """All of the files in the repository"""
 329     complete_file_list = []
 330     for path in sys.argv[1:] or sys.stdin.read().splitlines():
 331         complete_file_list.append(path)
 332 
 333     # ansible-test isn't currently passing symlinks to us so construct those ourselves for now
 334     for filename in _find_symlinks():
 335         if filename not in complete_file_list:
 336             # For some reason ansible-test is passing us lib/ansible/module_utils/ansible_release.py
 337             # which is a symlink even though it doesn't pass any others
 338             complete_file_list.append(filename)
 339 
 340     # We may run this after docs sanity tests so get a clean repository to run in
 341     with clean_repository(complete_file_list) as clean_repo_dir:
 342         os.chdir(clean_repo_dir)
 343 
 344         to_ship_files = assemble_files_to_ship(complete_file_list)
 345         to_install_files = assemble_files_to_install(complete_file_list)
 346 
 347         results = []
 348         with tempfile.TemporaryDirectory() as tmp_dir:
 349             sdist_path = create_sdist(tmp_dir)
 350             sdist_dir = extract_sdist(sdist_path, tmp_dir)
 351 
 352             # Check that the files that are supposed to be in the sdist are there
 353             results.extend(check_sdist_contains_expected(sdist_dir, to_ship_files))
 354 
 355             # Check that the files that are in the sdist are in the repository
 356             results.extend(check_sdist_files_are_wanted(sdist_dir, to_ship_files))
 357 
 358             # install the sdist
 359             install_dir = install_sdist(tmp_dir, sdist_dir)
 360 
 361             # Check that the files that are supposed to be installed are there
 362             results.extend(check_installed_contains_expected(install_dir, to_install_files))
 363 
 364             # Check that the files that are installed are supposed to be installed
 365             results.extend(check_installed_files_are_wanted(install_dir, to_install_files))
 366 
 367         for message in results:
 368             print(message)
