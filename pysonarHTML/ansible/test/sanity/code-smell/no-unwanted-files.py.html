   1 #!/usr/bin/env python
   2 """Prevent unwanted files from being added to the source tree."""
   3 from __future__ import (absolute_import, division, print_function)

  10 def main():
  11     """Main entry point."""
  12     paths = sys.argv[1:] or sys.stdin.read().splitlines()
  13 
  14     allowed_extensions = (
  15         '.cs',
  16         '.ps1',
  17         '.psm1',
  18         '.py',
  19     )
  20 
  21     skip_paths = set([
  22         'lib/ansible/config/ansible_builtin_runtime.yml',  # not included in the sanity ignore file since it won't exist until after migration
  23     ])
  24 
  25     skip_directories = (
  26         'lib/ansible/galaxy/data/',
  27     )
  28 
  29     for path in paths:
  30         if path in skip_paths:
  31             continue
  32 
  33         if any(path.startswith(skip_directory) for skip_directory in skip_directories):
  34             continue
  35 
  36         if path.startswith('lib/') and not path.startswith('lib/ansible/'):
  37             print('%s: all "lib" content must reside in the "lib/ansible" directory' % path)
  38             continue
  39 
  40         ext = os.path.splitext(path)[1]
  41 
  42         if ext not in allowed_extensions:
  43             print('%s: extension must be one of: %s' % (path, ', '.join(allowed_extensions)))
  44 
  45 
  46 if __name__ == '__main__':
  47     main()
