  18 def main():
  19     """Validate BOTMETA"""
  20     path = '.github/BOTMETA.yml'
  21 
  22     try:
  23         with open(path, 'r') as f_path:
  24             botmeta = yaml.safe_load(f_path)
  25     except yaml.error.MarkedYAMLError as ex:
  26         print('%s:%d:%d: YAML load failed: %s' % (path, ex.context_mark.line + 1, ex.context_mark.column + 1, re.sub(r'\s+', ' ', str(ex))))
  27         sys.exit()
  28     except Exception as ex:  # pylint: disable=broad-except
  29         print('%s:%d:%d: YAML load failed: %s' % (path, 0, 0, re.sub(r'\s+', ' ', str(ex))))
  30         sys.exit()
  31 
  32     list_string_types = list(string_types)
  33 
  34     files_schema = Any(
  35         Schema(*string_types),
  36         Schema({
  37             'ignored': Any(list_string_types, *string_types),
  38             'keywords': Any(list_string_types, *string_types),
  39             'labels': Any(list_string_types, *string_types),
  40             'maintainers': Any(list_string_types, *string_types),
  41             'migrated_to': All(
  42                 Any(*string_types),
  43                 Match(r'^\w+\.\w+$'),
  44             ),
  45             'notified': Any(list_string_types, *string_types),
  46             'supershipit': Any(list_string_types, *string_types),
  47             'support': Any("core", "network", "community"),
  48         })
  49     )
  50 
  51     list_dict_file_schema = [{str_type: files_schema}
  52                              for str_type in string_types]
  53 
  54     schema = Schema({
  55         Required('automerge'): bool,
  56         Required('collection_redirect'): bool,
  57         Required('notifications'): bool,
  58         Required('files'): Any(None, *list_dict_file_schema),
  59         Required('macros'): dict,  # Any(*list_macros_schema),
  60     })
  61 
  62     # Ensure schema is valid
  63 
  64     try:
  65         schema(botmeta)
  66     except MultipleInvalid as ex:
  67         for error in ex.errors:
  68             # No way to get line numbers
  69             print('%s:%d:%d: %s' % (path, 0, 0, humanize_error(botmeta, error)))
  70 
  71     # Ensure botmeta is always support:core
  72     botmeta_support = botmeta.get('files', {}).get('.github/BOTMETA.yml', {}).get('support', '')
  73     if botmeta_support != 'core':
  74         print('%s:%d:%d: .github/BOTMETA.yml MUST be support: core' % (path, 0, 0))
  75 
  76     # Find all path (none-team) macros so we can substitute them
  77     macros = botmeta.get('macros', {})
  78     path_macros = []
  79     for macro in macros:
  80         if macro.startswith('team_'):
  81             continue
  82         path_macros.append(macro)
  83 
  84 
  85 if __name__ == '__main__':
  86     main()
