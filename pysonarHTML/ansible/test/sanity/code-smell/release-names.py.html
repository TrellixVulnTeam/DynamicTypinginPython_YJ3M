  19 """
  20 Test that the release name is present in the list of used up release names
  21 """
  22 
  23 
  24 from __future__ import (absolute_import, division, print_function)
  25 __metaclass__ = type
  26 
  27 from yaml import safe_load
  28 
  29 from ansible.release import __codename__
  30 
  31 
  32 def main():
  33     """Entrypoint to the script"""
  34 
  35     with open('.github/RELEASE_NAMES.yml') as f:
  36         releases = safe_load(f.read())
  37 
  38     # Why this format?  The file's sole purpose is to be read by a human when they need to know
  39     # which release names have already been used.  So:
  40     # 1) It's easier for a human to find the release names when there's one on each line
  41     # 2) It helps keep other people from using the file and then asking for new features in it
  42     for name in (r.split(maxsplit=1)[1] for r in releases):
  43         if __codename__ == name:
  44             break
  45     else:
  46         print('.github/RELEASE_NAMES.yml: Current codename was not present in the file')
  47 
  48 
  49 if __name__ == '__main__':
  50     main()
