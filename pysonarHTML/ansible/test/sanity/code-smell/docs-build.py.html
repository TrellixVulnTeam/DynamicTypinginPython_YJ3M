  98 def simplify_stdout(value):
  99     """Simplify output by omitting earlier 'rendering: ...' messages."""
 100     lines = value.strip().splitlines()
 101 
 102     rendering = []
 103     keep = []
 104 
 105     def truncate_rendering():
 106         """Keep last rendering line (if any) with a message about omitted lines as needed."""
 107         if not rendering:
 108             return
 109 
 110         notice = rendering[-1]
 111 
 112         if len(rendering) > 1:
 113             notice += ' (%d previous rendering line(s) omitted)' % (len(rendering) - 1)
 114 
 115         keep.append(notice)
 116         # Could change to rendering.clear() if we do not support python2
 117         rendering[:] = []
 118 
 119     for line in lines:
 120         if line.startswith('rendering: '):
 121             rendering.append(line)
 122             continue
 123 
 124         truncate_rendering()
 125         keep.append(line)
 126 
 127     truncate_rendering()
 128 
 129     result = '\n'.join(keep)
 130 
 131     return result
