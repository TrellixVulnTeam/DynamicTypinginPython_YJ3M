  91 def show_dump_env(args):
  92     """
  93     :type args: EnvConfig
  94     """
  95     if not args.show and not args.dump:
  96         return
  97 
  98     data = dict(
  99         ansible=dict(
 100             version=get_ansible_version(),
 101         ),
 102         docker=get_docker_details(args),
 103         environ=os.environ.copy(),
 104         location=dict(
 105             pwd=os.environ.get('PWD', None),
 106             cwd=os.getcwd(),
 107         ),
 108         git=get_ci_provider().get_git_details(args),
 109         platform=dict(
 110             datetime=datetime.datetime.utcnow().strftime('%Y-%m-%dT%H:%M:%SZ'),
 111             platform=platform.platform(),
 112             uname=platform.uname(),
 113         ),
 114         python=dict(
 115             executable=sys.executable,
 116             version=platform.python_version(),
 117         ),
 118         interpreters=get_available_python_versions(SUPPORTED_PYTHON_VERSIONS),
 119     )
 120 
 121     if args.show:
 122         verbose = {
 123             'docker': 3,
 124             'docker.executable': 0,
 125             'environ': 2,
 126             'platform.uname': 1,
 127         }
 128 
 129         show_dict(data, verbose)
 130 
 131     if args.dump and not args.explain:
 132         write_json_test_results(ResultType.BOT, 'data-environment.json', data)

 144 def set_timeout(args):
 145     """
 146     :type args: EnvConfig
 147     """
 148     if args.timeout is None:
 149         return
 150 
 151     if args.timeout:
 152         deadline = (datetime.datetime.utcnow() + datetime.timedelta(minutes=args.timeout)).strftime('%Y-%m-%dT%H:%M:%SZ')
 153 
 154         display.info('Setting a %d minute test timeout which will end at: %s' % (args.timeout, deadline), verbosity=1)
 155     else:
 156         deadline = None
 157 
 158         display.info('Clearing existing test timeout.', verbosity=1)
 159 
 160     if args.explain:
 161         return
 162 
 163     if deadline:
 164         data = dict(
 165             duration=args.timeout,
 166             deadline=deadline,
 167         )
 168 
 169         write_json_file(TIMEOUT_PATH, data)
 170     elif os.path.exists(TIMEOUT_PATH):
 171         os.remove(TIMEOUT_PATH)
 172

 174 def get_timeout():
 175     """
 176     :rtype: dict[str, any] | None
 177     """
 178     if not os.path.exists(TIMEOUT_PATH):
 179         return None
 180 
 181     data = read_json_file(TIMEOUT_PATH)
 182     data['deadline'] = datetime.datetime.strptime(data['deadline'], '%Y-%m-%dT%H:%M:%SZ')
 183 
 184     return data

 195 def configure_test_timeout(args):
 196     """
 197     :type args: TestConfig
 198     """
 199     timeout = get_timeout()
 200 
 201     if not timeout:
 202         return
 203 
 204     timeout_start = datetime.datetime.utcnow()
 205     timeout_duration = timeout['duration']
 206     timeout_deadline = timeout['deadline']
 207     timeout_remaining = timeout_deadline - timeout_start
 208 
 209     test_timeout = TestTimeout(timeout_duration)
 210 
 211     if timeout_remaining <= datetime.timedelta():
 212         test_timeout.write(args)
 213 
 214         raise ApplicationError('The %d minute test timeout expired %s ago at %s.' % (
 215             timeout_duration, timeout_remaining * -1, timeout_deadline))
 216 
 217     display.info('The %d minute test timeout expires in %s at %s.' % (
 218         timeout_duration, timeout_remaining, timeout_deadline), verbosity=1)
 219 
 220     def timeout_handler(_dummy1, _dummy2):
 221         """Runs when SIGUSR1 is received."""
 222         test_timeout.write(args)
 223 
 224         raise ApplicationError('Tests aborted after exceeding the %d minute time limit.' % timeout_duration)
 225 
 226     def timeout_waiter(timeout_seconds):
 227         """
 228         :type timeout_seconds: int
 229         """
 230         time.sleep(timeout_seconds)
 231         os.kill(os.getpid(), signal.SIGUSR1)
 232 
 233     signal.signal(signal.SIGUSR1, timeout_handler)
 234 
 235     instance = WrappedThread(functools.partial(timeout_waiter, timeout_remaining.seconds))
 236     instance.daemon = True
 237     instance.start()

 240 def show_dict(data, verbose, root_verbosity=0, path=None):
 241     """
 242     :type data: dict[str, any]
 243     :type verbose: dict[str, int]
 244     :type root_verbosity: int
 245     :type path: list[str] | None
 246     """
 247     path = path if path else []
 248 
 249     for key, value in sorted(data.items()):
 250         indent = '  ' * len(path)
 251         key_path = path + [key]
 252         key_name = '.'.join(key_path)
 253         verbosity = verbose.get(key_name, root_verbosity)
 254 
 255         if isinstance(value, (tuple, list)):
 256             display.info(indent + '%s:' % key, verbosity=verbosity)
 257             for item in value:
 258                 display.info(indent + '  - %s' % item, verbosity=verbosity)
 259         elif isinstance(value, dict):
 260             min_verbosity = min([verbosity] + [v for k, v in verbose.items() if k.startswith('%s.' % key)])
 261             display.info(indent + '%s:' % key, verbosity=min_verbosity)
 262             show_dict(value, verbose, verbosity, key_path)
 263         else:
 264             display.info(indent + '%s: %s' % (key, value), verbosity=verbosity)

 267 def get_docker_details(args):
 268     """
 269     :type args: CommonConfig
 270     :rtype: dict[str, any]
 271     """
 272     docker = find_executable('docker', required=False)
 273     info = None
 274     version = None
 275 
 276     if docker:
 277         try:
 278             info = docker_info(args)
 279         except SubprocessError as ex:
 280             display.warning('Failed to collect docker info:\n%s' % ex)
 281 
 282         try:
 283             version = docker_version(args)
 284         except SubprocessError as ex:
 285             display.warning('Failed to collect docker version:\n%s' % ex)
 286 
 287     docker_details = dict(
 288         executable=docker,
 289         info=info,
 290         version=version,
 291     )
 292 
 293     return docker_details
