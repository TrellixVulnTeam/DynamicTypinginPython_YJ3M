 130 def read_lines_without_comments(path, remove_blank_lines=False, optional=False):  # type: (str, bool, bool) -> t.List[str]
 131     """
 132     Returns lines from the specified text file with comments removed.
 133     Comments are any content from a hash symbol to the end of a line.
 134     Any spaces immediately before a comment are also removed.
 135     """
 136     if optional and not os.path.exists(path):
 137         return []
 138 
 139     lines = read_text_file(path).splitlines()
 140 
 141     lines = [re.sub(r' *#.*$', '', line) for line in lines]
 142 
 143     if remove_blank_lines:
 144         lines = [line for line in lines if line]
 145 
 146     return lines

 149 def find_executable(executable, cwd=None, path=None, required=True):
 150     """
 151     :type executable: str
 152     :type cwd: str
 153     :type path: str
 154     :type required: bool | str
 155     :rtype: str | None
 156     """
 157     match = None
 158     real_cwd = os.getcwd()
 159 
 160     if not cwd:
 161         cwd = real_cwd
 162 
 163     if os.path.dirname(executable):
 164         target = os.path.join(cwd, executable)
 165         if os.path.exists(target) and os.access(target, os.F_OK | os.X_OK):
 166             match = executable
 167     else:
 168         if path is None:
 169             path = os.environ.get('PATH', os.path.defpath)
 170 
 171         if path:
 172             path_dirs = path.split(os.path.pathsep)
 173             seen_dirs = set()
 174 
 175             for path_dir in path_dirs:
 176                 if path_dir in seen_dirs:
 177                     continue
 178 
 179                 seen_dirs.add(path_dir)
 180 
 181                 if os.path.abspath(path_dir) == real_cwd:
 182                     path_dir = cwd
 183 
 184                 candidate = os.path.join(path_dir, executable)
 185 
 186                 if os.path.exists(candidate) and os.access(candidate, os.F_OK | os.X_OK):
 187                     match = candidate
 188                     break
 189 
 190     if not match and required:
 191         message = 'Required program "%s" not found.' % executable
 192 
 193         if required != 'warning':
 194             raise ApplicationError(message)
 195 
 196         display.warning(message)
 197 
 198     return match

 258 def raw_command(cmd, capture=False, env=None, data=None, cwd=None, explain=False, stdin=None, stdout=None,
 259                 cmd_verbosity=1, str_errors='strict'):
 260     """
 261     :type cmd: collections.Iterable[str]
 262     :type capture: bool
 263     :type env: dict[str, str] | None
 264     :type data: str | None
 265     :type cwd: str | None
 266     :type explain: bool
 267     :type stdin: file | None
 268     :type stdout: file | None
 269     :type cmd_verbosity: int
 270     :type str_errors: str
 271     :rtype: str | None, str | None
 272     """
 273     if not cwd:
 274         cwd = os.getcwd()
 275 
 276     if not env:
 277         env = common_environment()
 278 
 279     cmd = list(cmd)
 280 
 281     escaped_cmd = ' '.join(cmd_quote(c) for c in cmd)
 282 
 283     display.info('Run command: %s' % escaped_cmd, verbosity=cmd_verbosity, truncate=True)
 284     display.info('Working directory: %s' % cwd, verbosity=2)
 285 
 286     program = find_executable(cmd[0], cwd=cwd, path=env['PATH'], required='warning')
 287 
 288     if program:
 289         display.info('Program found: %s' % program, verbosity=2)
 290 
 291     for key in sorted(env.keys()):
 292         display.info('%s=%s' % (key, env[key]), verbosity=2)
 293 
 294     if explain:
 295         return None, None
 296 
 297     communicate = False
 298 
 299     if stdin is not None:
 300         data = None
 301         communicate = True
 302     elif data is not None:
 303         stdin = subprocess.PIPE
 304         communicate = True
 305 
 306     if stdout:
 307         communicate = True
 308 
 309     if capture:
 310         stdout = stdout or subprocess.PIPE
 311         stderr = subprocess.PIPE
 312         communicate = True
 313     else:
 314         stderr = None
 315 
 316     start = time.time()
 317     process = None
 318 
 319     try:
 320         try:
 321             cmd_bytes = [to_bytes(c) for c in cmd]
 322             env_bytes = dict((to_bytes(k), to_bytes(v)) for k, v in env.items())
 323             process = subprocess.Popen(cmd_bytes, env=env_bytes, stdin=stdin, stdout=stdout, stderr=stderr, cwd=cwd)
 324         except OSError as ex:
 325             if ex.errno == errno.ENOENT:
 326                 raise ApplicationError('Required program "%s" not found.' % cmd[0])
 327             raise
 328 
 329         if communicate:
 330             data_bytes = to_optional_bytes(data)
 331             stdout_bytes, stderr_bytes = process.communicate(data_bytes)
 332             stdout_text = to_optional_text(stdout_bytes, str_errors) or u''
 333             stderr_text = to_optional_text(stderr_bytes, str_errors) or u''
 334         else:
 335             process.wait()
 336             stdout_text, stderr_text = None, None
 337     finally:
 338         if process and process.returncode is None:
 339             process.kill()
 340             display.info('')  # the process we're interrupting may have completed a partial line of output
 341             display.notice('Killed command to avoid an orphaned child process during handling of an unexpected exception.')
 342 
 343     status = process.returncode
 344     runtime = time.time() - start
 345 
 346     display.info('Command exited with status %s after %s seconds.' % (status, runtime), verbosity=4)
 347 
 348     if status == 0:
 349         return stdout_text, stderr_text
 350 
 351     raise SubprocessError(cmd, status, stdout_text, stderr_text, runtime)

 354 def common_environment():
 355     """Common environment used for executing all programs."""
 356     env = dict(
 357         LC_ALL='en_US.UTF-8',
 358         PATH=os.environ.get('PATH', os.path.defpath),
 359     )
 360 
 361     required = (
 362         'HOME',
 363     )
 364 
 365     optional = (
 366         'HTTPTESTER',
 367         'LD_LIBRARY_PATH',
 368         'SSH_AUTH_SOCK',
 369         # MacOS High Sierra Compatibility
 370         # http://sealiesoftware.com/blog/archive/2017/6/5/Objective-C_and_fork_in_macOS_1013.html
 371         # Example configuration for macOS:
 372         # export OBJC_DISABLE_INITIALIZE_FORK_SAFETY=YES
 373         'OBJC_DISABLE_INITIALIZE_FORK_SAFETY',
 374         'ANSIBLE_KEEP_REMOTE_FILES',
 375         # MacOS Homebrew Compatibility
 376         # https://cryptography.io/en/latest/installation/#building-cryptography-on-macos
 377         # This may also be required to install pyyaml with libyaml support when installed in non-standard locations.
 378         # Example configuration for brew on macOS:
 379         # export LDFLAGS="-L$(brew --prefix openssl)/lib/     -L$(brew --prefix libyaml)/lib/"
 380         # export  CFLAGS="-I$(brew --prefix openssl)/include/ -I$(brew --prefix libyaml)/include/"
 381         # However, this is not adequate for PyYAML 3.13, which is the latest version supported on Python 2.6.
 382         # For that version the standard location must be used, or `pip install` must be invoked with additional options:
 383         # --global-option=build_ext --global-option=-L{path_to_lib_dir}
 384         'LDFLAGS',
 385         'CFLAGS',
 386     )
 387 
 388     env.update(pass_vars(required=required, optional=optional))
 389 
 390     return env

 393 def pass_vars(required, optional):
 394     """
 395     :type required: collections.Iterable[str]
 396     :type optional: collections.Iterable[str]
 397     :rtype: dict[str, str]
 398     """
 399     env = {}
 400 
 401     for name in required:
 402         if name not in os.environ:
 403             raise MissingEnvironmentVariable(name)
 404         env[name] = os.environ[name]
 405 
 406     for name in optional:
 407         if name not in os.environ:
 408             continue
 409         env[name] = os.environ[name]
 410 
 411     return env

 414 def deepest_path(path_a, path_b):
 415     """Return the deepest of two paths, or None if the paths are unrelated.
 416     :type path_a: str
 417     :type path_b: str
 418     :rtype: str | None
 419     """
 420     if path_a == '.':
 421         path_a = ''
 422 
 423     if path_b == '.':
 424         path_b = ''
 425 
 426     if path_a.startswith(path_b):
 427         return path_a or '.'
 428 
 429     if path_b.startswith(path_a):
 430         return path_b or '.'
 431 
 432     return None

 452         '.cfg',

 454         '.crt',
 455         '.cs',

 458         '.ini',
 459         '.j2',
 460         '.js',

 466         '.py',
 467         '.rst',
 468         '.sh',
 469         '.txt',
 470         '.xml',
 471         '.yaml',
 472         '.yml',

 476         '.bin',
 477         '.eot',
 478         '.gz',
 479         '.ico',

 484         '.png',

 525 class Display:
 526     """Manages color console output."""
 527     clear = '\033[0m'
 528     red = '\033[31m'
 529     green = '\033[32m'
 530     yellow = '\033[33m'
 531     blue = '\033[34m'
 532     purple = '\033[35m'
 533     cyan = '\033[36m'
 534 
 535     verbosity_colors = {
 536         0: None,
 537         1: green,
 538         2: blue,
 539         3: cyan,
 540     }
 541 
 542     def __init__(self):
 543         self.verbosity = 0
 544         self.color = sys.stdout.isatty()
 545         self.warnings = []
 546         self.warnings_unique = set()
 547         self.info_stderr = False
 548         self.rows = 0
 549         self.columns = 0
 550         self.truncate = 0
 551         self.redact = True
 552         self.sensitive = set()
 553 
 554         if os.isatty(0):
 555             self.rows, self.columns = unpack('HHHH', fcntl.ioctl(0, TIOCGWINSZ, pack('HHHH', 0, 0, 0, 0)))[:2]
 556 
 557     def __warning(self, message):
 558         """
 559         :type message: str
 560         """
 561         self.print_message('WARNING: %s' % message, color=self.purple, fd=sys.stderr)
 562 
 563     def review_warnings(self):
 564         """Review all warnings which previously occurred."""
 565         if not self.warnings:
 566             return
 567 
 568         self.__warning('Reviewing previous %d warning(s):' % len(self.warnings))
 569 
 570         for warning in self.warnings:
 571             self.__warning(warning)
 572 
 573     def warning(self, message, unique=False, verbosity=0):
 574         """
 575         :type message: str
 576         :type unique: bool
 577         :type verbosity: int
 578         """
 579         if verbosity > self.verbosity:
 580             return
 581 
 582         if unique:
 583             if message in self.warnings_unique:
 584                 return
 585 
 586             self.warnings_unique.add(message)
 587 
 588         self.__warning(message)
 589         self.warnings.append(message)
 590 
 591     def notice(self, message):
 592         """
 593         :type message: str
 594         """
 595         self.print_message('NOTICE: %s' % message, color=self.purple, fd=sys.stderr)
 596 
 597     def error(self, message):
 598         """
 599         :type message: str
 600         """
 601         self.print_message('ERROR: %s' % message, color=self.red, fd=sys.stderr)
 602 
 603     def info(self, message, verbosity=0, truncate=False):
 604         """
 605         :type message: str
 606         :type verbosity: int
 607         :type truncate: bool
 608         """
 609         if self.verbosity >= verbosity:
 610             color = self.verbosity_colors.get(verbosity, self.yellow)
 611             self.print_message(message, color=color, fd=sys.stderr if self.info_stderr else sys.stdout, truncate=truncate)
 612 
 613     def print_message(self, message, color=None, fd=sys.stdout, truncate=False):  # pylint: disable=locally-disabled, invalid-name
 614         """
 615         :type message: str
 616         :type color: str | None
 617         :type fd: file
 618         :type truncate: bool
 619         """
 620         if self.redact and self.sensitive:
 621             for item in self.sensitive:
 622                 if not item:
 623                     continue
 624 
 625                 message = message.replace(item, '*' * len(item))
 626 
 627         if truncate:
 628             if len(message) > self.truncate > 5:
 629                 message = message[:self.truncate - 5] + ' ...'
 630 
 631         if color and self.color:
 632             # convert color resets in message to desired color
 633             message = message.replace(self.clear, color)
 634             message = '%s%s%s' % (color, message, self.clear)
 635 
 636         if sys.version_info[0] == 2:
 637             message = to_bytes(message)
 638 
 639         print(message, file=fd)
 640         fd.flush()

 651 class SubprocessError(ApplicationError):
 652     """Error resulting from failed subprocess execution."""
 653     def __init__(self, cmd, status=0, stdout=None, stderr=None, runtime=None):
 654         """
 655         :type cmd: list[str]
 656         :type status: int
 657         :type stdout: str | None
 658         :type stderr: str | None
 659         :type runtime: float | None
 660         """
 661         message = 'Command "%s" returned exit status %s.\n' % (' '.join(cmd_quote(c) for c in cmd), status)
 662 
 663         if stderr:
 664             message += '>>> Standard Error\n'
 665             message += '%s%s\n' % (stderr.strip(), Display.clear)
 666 
 667         if stdout:
 668             message += '>>> Standard Output\n'
 669             message += '%s%s\n' % (stdout.strip(), Display.clear)
 670 
 671         message = message.strip()
 672 
 673         super(SubprocessError, self).__init__(message)
 674 
 675         self.cmd = cmd
 676         self.message = message
 677         self.status = status
 678         self.stdout = stdout
 679         self.stderr = stderr
 680         self.runtime = runtime

 694 def parse_to_list_of_dict(pattern, value):
 695     """
 696     :type pattern: str
 697     :type value: str
 698     :return: list[dict[str, str]]
 699     """
 700     matched = []
 701     unmatched = []
 702 
 703     for line in value.splitlines():
 704         match = re.search(pattern, line)
 705 
 706         if match:
 707             matched.append(match.groupdict())
 708         else:
 709             unmatched.append(line)
 710 
 711     if unmatched:

 729 def get_subclasses(class_type):  # type: (t.Type[C]) -> t.Set[t.Type[C]]
 730     """Returns the set of types that are concrete subclasses of the given type."""
 731     subclasses = set()  # type: t.Set[t.Type[C]]
 732     queue = [class_type]  # type: t.List[t.Type[C]]
 733 
 734     while queue:
 735         parent = queue.pop()
 736 
 737         for child in parent.__subclasses__():
 738             if child not in subclasses:
 739                 if not inspect.isabstract(child):
 740                     subclasses.add(child)
 741                 queue.append(child)
 742 
 743     return subclasses

 757 def paths_to_dirs(paths):  # type: (t.List[str]) -> t.List[str]
 758     """Returns a list of directories extracted from the given list of paths."""

 762         while True:

 766                 break

 783 def import_plugins(directory, root=None):  # type: (str, t.Optional[str]) -> None
 784     """
 785     Import plugins from the given directory relative to the given root.
 786     If the root is not provided, the 'lib' directory for the test runner will be used.
 787     """
 788     if root is None:
 789         root = os.path.dirname(__file__)
 790 
 791     path = os.path.join(root, directory)
 792     package = __name__.rsplit('.', 1)[0]
 793     prefix = '%s.%s.' % (package, directory.replace(os.path.sep, '.'))
 794 
 795     for (_module_loader, name, _ispkg) in pkgutil.iter_modules([path], prefix=prefix):
 796         module_path = os.path.join(root, name[len(package) + 1:].replace('.', os.path.sep) + '.py')
 797         load_module(module_path, name)

 800 def load_plugins(base_type, database):  # type: (t.Type[C], t.Dict[str, t.Type[C]]) -> None
 801     """

 803     Only plugins which have already been imported will be loaded.

 811 def load_module(path, name):  # type: (str, str) -> None
 812     """Load a Python module using the given name and path."""
 813     if name in sys.modules:
 814         return
 815 
 816     if sys.version_info >= (3, 4):
 817         # noinspection PyUnresolvedReferences
 818         import importlib.util
 819 
 820         # noinspection PyUnresolvedReferences
 821         spec = importlib.util.spec_from_file_location(name, path)
 822         # noinspection PyUnresolvedReferences
 823         module = importlib.util.module_from_spec(spec)
 824         spec.loader.exec_module(module)
 825 
 826         sys.modules[name] = module
 827     else:
 828         # noinspection PyDeprecation
 829         import imp
 830 
 831         # load_source (and thus load_module) require a file opened with `open` in text mode
 832         with open(to_bytes(path)) as module_file:
 833             # noinspection PyDeprecation
 834             imp.load_module(name, module_file, path, ('.py', 'r', imp.PY_SOURCE))
