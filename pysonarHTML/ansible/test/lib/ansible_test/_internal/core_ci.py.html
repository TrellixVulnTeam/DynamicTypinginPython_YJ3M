 230     def _get_parallels_endpoints(self):
 231         """
 232         :rtype: tuple[str]
 233         """
 234         client = HttpClient(self.args, always=True)
 235         display.info('Getting available endpoints...', verbosity=1)
 236         sleep = 3
 237 
 238         for _iteration in range(1, 10):
 239             response = client.get('https://s3.amazonaws.com/ansible-ci-files/ansible-test/parallels-endpoints.txt')
 240 
 241             if response.status_code == 200:
 242                 endpoints = tuple(response.response.splitlines())
 243                 display.info('Available endpoints (%d):\n%s' % (len(endpoints), '\n'.join(' - %s' % endpoint for endpoint in endpoints)), verbosity=1)
 244                 return endpoints
 245 
 246             display.warning('HTTP %d error getting endpoints, trying again in %d seconds.' % (response.status_code, sleep))
 247             time.sleep(sleep)

 249         raise ApplicationError('Unable to get available endpoints.')

 257         """Start instance."""
 258         if self.started:
 259             display.info('Skipping started %s/%s instance %s.' % (self.platform, self.version, self.instance_id),
 260                          verbosity=1)
 261             return None
 262 
 263         return self._start(self.ci_provider.prepare_core_ci_auth(self.auth_context))

 265     def stop(self):
 266         """Stop instance."""
 267         if not self.started:
 268             display.info('Skipping invalid %s/%s instance %s.' % (self.platform, self.version, self.instance_id),
 269                          verbosity=1)
 270             return
 271 
 272         response = self.client.delete(self._uri)
 273 
 274         if response.status_code == 404:
 275             self._clear()
 276             display.info('Cleared invalid %s/%s instance %s.' % (self.platform, self.version, self.instance_id),
 277                          verbosity=1)
 278             return
 279 
 280         if response.status_code == 200:
 281             self._clear()
 282             display.info('Stopped running %s/%s instance %s.' % (self.platform, self.version, self.instance_id),
 283                          verbosity=1)
 284             return
 285 
 286         raise self._create_http_error(response)

 288     def get(self, tries=3, sleep=15, always_raise_on=None):
 289         """
 290         Get instance connection information.
 291         :type tries: int
 292         :type sleep: int
 293         :type always_raise_on: list[int] | None
 294         :rtype: InstanceConnection
 295         """
 296         if not self.started:
 297             display.info('Skipping invalid %s/%s instance %s.' % (self.platform, self.version, self.instance_id),
 298                          verbosity=1)
 299             return None
 300 
 301         if not always_raise_on:
 302             always_raise_on = []
 303 
 304         if self.connection and self.connection.running:
 305             return self.connection
 306 
 307         while True:
 308             tries -= 1
 309             response = self.client.get(self._uri)
 310 
 311             if response.status_code == 200:
 312                 break
 313 
 314             error = self._create_http_error(response)
 315 
 316             if not tries or response.status_code in always_raise_on:
 317                 raise error
 318 
 319             display.warning('%s. Trying again after %d seconds.' % (error, sleep))
 320             time.sleep(sleep)
 321 
 322         if self.args.explain:
 323             self.connection = InstanceConnection(
 324                 running=True,
 325                 hostname='cloud.example.com',
 326                 port=self.port or 12345,
 327                 username='username',
 328                 password='password' if self.platform == 'windows' else None,
 329             )
 330         else:
 331             response_json = response.json()
 332             status = response_json['status']
 333             con = response_json.get('connection')
 334 
 335             if con:
 336                 self.connection = InstanceConnection(
 337                     running=status == 'running',
 338                     hostname=con['hostname'],
 339                     port=int(con.get('port', self.port)),
 340                     username=con['username'],
 341                     password=con.get('password'),
 342                     response_json=response_json,
 343                 )
 344             else:
 345                 self.connection = InstanceConnection(
 346                     running=status == 'running',
 347                     response_json=response_json,
 348                 )
 349 
 350         if self.connection.password:
 351             display.sensitive.add(str(self.connection.password))
 352 
 353         status = 'running' if self.connection.running else 'starting'
 354 
 355         display.info('Status update: %s/%s on instance %s is %s.' %
 356                      (self.platform, self.version, self.instance_id, status),
 357                      verbosity=1)
 358 
 359         return self.connection

 361     def wait(self, iterations=90):  # type: (t.Optional[int]) -> None
 362         """Wait for the instance to become ready."""
 363         for _iteration in range(1, iterations):
 364             if self.get().running:
 365                 return
 366             time.sleep(10)
 367 
 368         raise ApplicationError('Timeout waiting for %s/%s instance %s.' %
 369                                (self.platform, self.version, self.instance_id))

 375     def _start(self, auth):
 376         """Start instance."""
 377         display.info('Initializing new %s/%s instance %s.' % (self.platform, self.version, self.instance_id), verbosity=1)
 378 
 379         if self.platform == 'windows':
 380             winrm_config = read_text_file(os.path.join(ANSIBLE_TEST_DATA_ROOT, 'setup', 'ConfigureRemotingForAnsible.ps1'))
 381         else:
 382             winrm_config = None
 383 
 384         data = dict(
 385             config=dict(
 386                 platform=self.platform,
 387                 version=self.version,
 388                 public_key=self.ssh_key.pub_contents if self.ssh_key else None,
 389                 query=False,
 390                 winrm_config=winrm_config,
 391             )
 392         )
 393 
 394         data.update(dict(auth=auth))
 395 
 396         headers = {
 397             'Content-Type': 'application/json',
 398         }
 399 
 400         response = self._start_try_endpoints(data, headers)
 401 
 402         self.started = True
 403         self._save()
 404 
 405         display.info('Started %s/%s from: %s' % (self.platform, self.version, self._uri), verbosity=1)
 406 
 407         if self.args.explain:
 408             return {}
 409 
 410         return response.json()

 412     def _start_try_endpoints(self, data, headers):
 413         """
 414         :type data: dict[str, any]
 415         :type headers: dict[str, str]
 416         :rtype: HttpResponse
 417         """
 418         threshold = 1
 419 
 420         while threshold <= self.max_threshold:
 421             for self.endpoint in self.endpoints:
 422                 try:
 423                     return self._start_at_threshold(data, headers, threshold)
 424                 except CoreHttpError as ex:
 425                     if ex.status == 503:
 426                         display.info('Service Unavailable: %s' % ex.remote_message, verbosity=1)
 427                         continue
 428                     display.error(ex.remote_message)
 429                 except HttpError as ex:
 430                     display.error(u'%s' % ex)
 431 
 432                 time.sleep(3)
 433 
 434             threshold += 1
 435 
 436         raise ApplicationError('Maximum threshold reached and all endpoints exhausted.')
 437 
 438     def _start_at_threshold(self, data, headers, threshold):
 439         """
 440         :type data: dict[str, any]
 441         :type headers: dict[str, str]
 442         :type threshold: int
 443         :rtype: HttpResponse | None
 444         """
 445         tries = self.retries
 446         sleep = 15
 447 
 448         data['threshold'] = threshold
 449 
 450         display.info('Trying endpoint: %s (threshold %d)' % (self.endpoint, threshold), verbosity=1)
 451 
 452         while True:
 453             tries -= 1
 454             response = self.client.put(self._uri, data=json.dumps(data), headers=headers)
 455 
 456             if response.status_code == 200:
 457                 return response
 458 
 459             error = self._create_http_error(response)
 460 
 461             if response.status_code == 503:
 462                 raise error
 463 
 464             if not tries:
 465                 raise error
 466 
 467             display.warning('%s. Trying again after %d seconds.' % (error, sleep))
 468             time.sleep(sleep)

 470     def _clear(self):
 471         """Clear instance information."""
 472         try:
 473             self.connection = None
 474             os.remove(self.path)
 475         except OSError as ex:
 476             if ex.errno != errno.ENOENT:
 477                 raise

 479     def _load(self):
 480         """Load instance information."""
 481         try:
 482             data = read_text_file(self.path)
 483         except IOError as ex:
 484             if ex.errno != errno.ENOENT:
 485                 raise
 486 
 487             return False
 488 
 489         if not data.startswith('{'):
 490             return False  # legacy format
 491 
 492         config = json.loads(data)
 493 
 494         return self.load(config)

 528     @staticmethod
 529     def _create_http_error(response):
 530         """
 531         :type response: HttpResponse
 532         :rtype: ApplicationError
 533         """
 534         response_json = response.json()
 535         stack_trace = ''
 536 
 537         if 'message' in response_json:
 538             message = response_json['message']
 539         elif 'errorMessage' in response_json:
 540             message = response_json['errorMessage'].strip()
 541             if 'stackTrace' in response_json:
 542                 traceback_lines = response_json['stackTrace']
 543 
 544                 # AWS Lambda on Python 2.7 returns a list of tuples
 545                 # AWS Lambda on Python 3.7 returns a list of strings
 546                 if traceback_lines and isinstance(traceback_lines[0], list):
 547                     traceback_lines = traceback.format_list(traceback_lines)
 548 
 549                 trace = '\n'.join([x.rstrip() for x in traceback_lines])
 550                 stack_trace = ('\nTraceback (from remote server):\n%s' % trace)
 551         else:
 552             message = str(response_json)
 553 
 554         return CoreHttpError(response.status_code, message, stack_trace)

 571 class SshKey:
 572     """Container for SSH key used to connect to remote instances."""
 573     KEY_NAME = 'id_rsa'
 574     PUB_NAME = 'id_rsa.pub'
 575 
 576     def __init__(self, args):
 577         """
 578         :type args: EnvironmentConfig
 579         """
 580         key_pair = self.get_key_pair()
 581 
 582         if not key_pair:
 583             key_pair = self.generate_key_pair(args)
 584 
 585         key, pub = key_pair
 586         key_dst, pub_dst = self.get_in_tree_key_pair_paths()
 587 
 588         def ssh_key_callback(files):  # type: (t.List[t.Tuple[str, str]]) -> None
 589             """
 590             Add the SSH keys to the payload file list.
 591             They are either outside the source tree or in the cache dir which is ignored by default.
 592             """
 593             files.append((key, os.path.relpath(key_dst, data_context().content.root)))
 594             files.append((pub, os.path.relpath(pub_dst, data_context().content.root)))
 595 
 596         data_context().register_payload_callback(ssh_key_callback)
 597 
 598         self.key, self.pub = key, pub
 599 
 600         if args.explain:
 601             self.pub_contents = None
 602         else:
 603             self.pub_contents = read_text_file(self.pub).strip()
 604 
 605     def get_in_tree_key_pair_paths(self):  # type: () -> t.Optional[t.Tuple[str, str]]
 606         """Return the ansible-test SSH key pair paths from the content tree."""
 607         temp_dir = ResultType.TMP.path
 608 
 609         key = os.path.join(temp_dir, self.KEY_NAME)
 610         pub = os.path.join(temp_dir, self.PUB_NAME)
 611 
 612         return key, pub
 613 
 614     def get_source_key_pair_paths(self):  # type: () -> t.Optional[t.Tuple[str, str]]
 615         """Return the ansible-test SSH key pair paths for the current user."""
 616         base_dir = os.path.expanduser('~/.ansible/test/')
 617 
 618         key = os.path.join(base_dir, self.KEY_NAME)
 619         pub = os.path.join(base_dir, self.PUB_NAME)
 620 
 621         return key, pub
 622 
 623     def get_key_pair(self):  # type: () -> t.Optional[t.Tuple[str, str]]
 624         """Return the ansible-test SSH key pair paths if present, otherwise return None."""
 625         key, pub = self.get_in_tree_key_pair_paths()
 626 
 627         if os.path.isfile(key) and os.path.isfile(pub):
 628             return key, pub
 629 
 630         key, pub = self.get_source_key_pair_paths()
 631 
 632         if os.path.isfile(key) and os.path.isfile(pub):
 633             return key, pub
 634 
 635         return None
 636 
 637     def generate_key_pair(self, args):  # type: (EnvironmentConfig) -> t.Tuple[str, str]
 638         """Generate an SSH key pair for use by all ansible-test invocations for the current user."""
 639         key, pub = self.get_source_key_pair_paths()
 640 
 641         if not args.explain:
 642             make_dirs(os.path.dirname(key))
 643 
 644         if not os.path.isfile(key) or not os.path.isfile(pub):
 645             run_command(args, ['ssh-keygen', '-m', 'PEM', '-q', '-t', 'rsa', '-N', '', '-f', key])
 646 
 647             # newer ssh-keygen PEM output (such as on RHEL 8.1) is not recognized by paramiko
 648             key_contents = read_text_file(key)
 649             key_contents = re.sub(r'(BEGIN|END) PRIVATE KEY', r'\1 RSA PRIVATE KEY', key_contents)
 650 
 651             write_text_file(key, key_contents)
 652 
 653         return key, pub

 656 class InstanceConnection:
 657     """Container for remote instance status and connection details."""
 658     def __init__(self,
 659                  running,  # type: bool
 660                  hostname=None,  # type: t.Optional[str]
 661                  port=None,  # type: t.Optional[int]
 662                  username=None,  # type: t.Optional[str]
 663                  password=None,  # type: t.Optional[str]
 664                  response_json=None,  # type: t.Optional[t.Dict[str, t.Any]]
 665                  ):  # type: (...) -> None
 666         self.running = running
 667         self.hostname = hostname
 668         self.port = port
 669         self.username = username
 670         self.password = password
 671         self.response_json = response_json or {}
 672 
 673     def __str__(self):
 674         if self.password:
 675             return '%s:%s [%s:%s]' % (self.hostname, self.port, self.username, self.password)
 676 
 677         return '%s:%s [%s]' % (self.hostname, self.port, self.username)
