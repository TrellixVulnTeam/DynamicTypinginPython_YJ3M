  22 class UnversionedSource(SourceProvider):
  23     """Fallback source provider when no other provider matches the content root."""
  24     sequence = 0  # disable automatic detection
  25 
  26     @staticmethod
  27     def is_content_root(path):  # type: (str) -> bool
  28         """Return True if the given path is a content root for this provider."""
  29         return False
  30 
  31     def get_paths(self, path):  # type: (str) -> t.List[str]
  32         """Return the list of available content paths under the given path."""
  33         paths = []
  34 
  35         kill_any_dir = (
  36             '.idea',
  37             '.pytest_cache',
  38             '__pycache__',
  39             'ansible.egg-info',
  40             'ansible_base.egg-info',
  41         )
  42 
  43         kill_sub_dir = {
  44             'test': (
  45                 'results',
  46                 'cache',
  47                 'output',
  48             ),
  49             'tests': (
  50                 'output',
  51             ),
  52             'docs/docsite': (
  53                 '_build',
  54             ),
  55         }
  56 
  57         kill_sub_file = {
  58             '': (
  59                 TIMEOUT_PATH,
  60             ),
  61         }
  62 
  63         kill_extensions = (
  64             '.pyc',
  65             '.pyo',
  66             '.retry',
  67         )
  68 
  69         for root, dir_names, file_names in os.walk(path):
  70             rel_root = os.path.relpath(root, path)
  71 
  72             if rel_root == '.':
  73                 rel_root = ''
  74 
  75             for kill in kill_any_dir + kill_sub_dir.get(rel_root, ()):
  76                 if kill in dir_names:
  77                     dir_names.remove(kill)
  78 
  79             kill_files = kill_sub_file.get(rel_root, ())
  80 
  81             paths.extend([os.path.join(rel_root, file_name) for file_name in file_names
  82                           if not os.path.splitext(file_name)[1] in kill_extensions and file_name not in kill_files])
  83 
  84             # include directory symlinks since they will not be traversed and would otherwise go undetected
  85             paths.extend([os.path.join(rel_root, dir_name) + os.path.sep for dir_name in dir_names if os.path.islink(to_bytes(dir_name))])
  86 
  87         return paths
