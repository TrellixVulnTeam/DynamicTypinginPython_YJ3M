  24         """Return True if the given path is a content root for this provider."""
  25         if os.path.basename(os.path.dirname(os.path.dirname(path))) == 'ansible_collections':
  26             return True

  28         return False

  30     def create(self, root, paths):  # type: (str, t.List[str]) -> ContentLayout
  31         """Create a Layout using the given root and paths."""
  32         plugin_paths = dict((p, os.path.join('plugins', p)) for p in self.PLUGIN_TYPES)
  33 
  34         collection_root = os.path.dirname(os.path.dirname(root))
  35         collection_dir = os.path.relpath(root, collection_root)
  36         collection_namespace, collection_name = collection_dir.split(os.path.sep)
  37 
  38         collection_root = os.path.dirname(collection_root)
  39 
  40         sanity_messages = LayoutMessages()
  41         integration_messages = LayoutMessages()
  42         unit_messages = LayoutMessages()
  43 
  44         # these apply to all test commands
  45         self.__check_test_path(paths, sanity_messages)
  46         self.__check_test_path(paths, integration_messages)
  47         self.__check_test_path(paths, unit_messages)
  48 
  49         # these apply to specific test commands
  50         integration_targets_path = self.__check_integration_path(paths, integration_messages)
  51         self.__check_unit_path(paths, unit_messages)
  52 
  53         return ContentLayout(root,
  54                              paths,
  55                              plugin_paths=plugin_paths,
  56                              collection=CollectionDetail(
  57                                  name=collection_name,
  58                                  namespace=collection_namespace,
  59                                  root=collection_root,
  60                              ),
  61                              test_path='tests',
  62                              results_path='tests/output',
  63                              sanity_path='tests/sanity',
  64                              sanity_messages=sanity_messages,
  65                              integration_path='tests/integration',
  66                              integration_targets_path=integration_targets_path.rstrip(os.path.sep),
  67                              integration_vars_path='tests/integration/integration_config.yml',
  68                              integration_messages=integration_messages,
  69                              unit_path='tests/unit',
  70                              unit_module_path='tests/unit/plugins/modules',
  71                              unit_module_utils_path='tests/unit/plugins/module_utils',
  72                              unit_messages=unit_messages,
  73                              )

  75     @staticmethod
  76     def __check_test_path(paths, messages):  # type: (t.List[str], LayoutMessages) -> None
  77         modern_test_path = 'tests/'
  78         modern_test_path_found = any(path.startswith(modern_test_path) for path in paths)
  79         legacy_test_path = 'test/'
  80         legacy_test_path_found = any(path.startswith(legacy_test_path) for path in paths)
  81 
  82         if modern_test_path_found and legacy_test_path_found:
  83             messages.warning.append('Ignoring tests in "%s" in favor of "%s".' % (legacy_test_path, modern_test_path))
  84         elif legacy_test_path_found:
  85             messages.warning.append('Ignoring tests in "%s" that should be in "%s".' % (legacy_test_path, modern_test_path))
  86 
  87     @staticmethod
  88     def __check_integration_path(paths, messages):  # type: (t.List[str], LayoutMessages) -> str
  89         modern_integration_path = 'roles/test/'
  90         modern_integration_path_found = any(path.startswith(modern_integration_path) for path in paths)
  91         legacy_integration_path = 'tests/integration/targets/'
  92         legacy_integration_path_found = any(path.startswith(legacy_integration_path) for path in paths)
  93 
  94         if modern_integration_path_found and legacy_integration_path_found:
  95             messages.warning.append('Ignoring tests in "%s" in favor of "%s".' % (legacy_integration_path, modern_integration_path))
  96             integration_targets_path = modern_integration_path
  97         elif legacy_integration_path_found:
  98             messages.info.append('Falling back to tests in "%s" because "%s" was not found.' % (legacy_integration_path, modern_integration_path))
  99             integration_targets_path = legacy_integration_path
 100         elif modern_integration_path_found:
 101             messages.info.append('Loading tests from "%s".' % modern_integration_path)
 102             integration_targets_path = modern_integration_path
 103         else:
 104             messages.error.append('Cannot run integration tests without "%s" or "%s".' % (modern_integration_path, legacy_integration_path))
 105             integration_targets_path = modern_integration_path
 106 
 107         return integration_targets_path
 108 
 109     @staticmethod
 110     def __check_unit_path(paths, messages):  # type: (t.List[str], LayoutMessages) -> None
 111         modern_unit_path = 'tests/unit/'
 112         modern_unit_path_found = any(path.startswith(modern_unit_path) for path in paths)
 113         legacy_unit_path = 'tests/units/'  # test/units/ will be covered by the warnings for test/ vs tests/
 114         legacy_unit_path_found = any(path.startswith(legacy_unit_path) for path in paths)
 115 
 116         if modern_unit_path_found and legacy_unit_path_found:
 117             messages.warning.append('Ignoring tests in "%s" in favor of "%s".' % (legacy_unit_path, modern_unit_path))
 118         elif legacy_unit_path_found:
 119             messages.warning.append('Rename "%s" to "%s" to run unit tests.' % (legacy_unit_path, modern_unit_path))
 120         elif modern_unit_path_found:
 121             pass  # unit tests only run from one directory so no message is needed
 122         else:
 123             messages.error.append('Cannot run unit tests without "%s".' % modern_unit_path)
