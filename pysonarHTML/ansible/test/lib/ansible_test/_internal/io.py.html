  70 def open_text_file(path, mode='r'):  # type: (str, str) -> t.TextIO
  71     """Open the given path for text access."""
  72     if 'b' in mode:
  73         raise Exception('mode cannot include "b" for text files: %s' % mode)
  74 
  75     # noinspection PyTypeChecker

  79 def open_binary_file(path, mode='rb'):  # type: (str, str) -> t.BinaryIO
  80     """Open the given path for binary access."""

  82         raise Exception('mode must include "b" for binary files: %s' % mode)
  83 
  84     # noinspection PyTypeChecker
