  29 def get_python_module_utils_imports(compile_targets):
  30     """Return a dictionary of module_utils names mapped to sets of python file paths.
  31     :type compile_targets: list[TestTarget]
  32     :rtype: dict[str, set[str]]
  33     """
  34 
  35     module_utils = enumerate_module_utils()
  36 
  37     virtual_utils = set(m for m in module_utils if any(m.startswith('%s.' % v) for v in VIRTUAL_PACKAGES))
  38     module_utils -= virtual_utils
  39 
  40     imports_by_target_path = {}
  41 
  42     for target in compile_targets:
  43         imports_by_target_path[target.path] = extract_python_module_utils_imports(target.path, module_utils)
  44 
  45     def recurse_import(import_name, depth=0, seen=None):  # type: (str, int, t.Optional[t.Set[str]]) -> t.Set[str]
  46         """Recursively expand module_utils imports from module_utils files."""
  47         display.info('module_utils import: %s%s' % ('  ' * depth, import_name), verbosity=4)
  48 
  49         if seen is None:
  50             seen = set([import_name])
  51 
  52         results = set([import_name])
  53 
  54         # virtual packages depend on the modules they contain instead of the reverse
  55         if import_name in VIRTUAL_PACKAGES:
  56             for sub_import in sorted(virtual_utils):
  57                 if sub_import.startswith('%s.' % import_name):
  58                     if sub_import in seen:
  59                         continue
  60 
  61                     seen.add(sub_import)
  62 
  63                     matches = sorted(recurse_import(sub_import, depth + 1, seen))
  64 
  65                     for result in matches:
  66                         results.add(result)
  67 
  68         import_path = get_import_path(import_name)
  69 
  70         if import_path not in imports_by_target_path:
  71             import_path = get_import_path(import_name, package=True)
  72 
  73             if import_path not in imports_by_target_path:
  74                 raise ApplicationError('Cannot determine path for module_utils import: %s' % import_name)
  75 
  76         # process imports in reverse so the deepest imports come first
  77         for name in sorted(imports_by_target_path[import_path], reverse=True):
  78             if name in virtual_utils:
  79                 continue
  80 
  81             if name in seen:
  82                 continue
  83 
  84             seen.add(name)
  85 
  86             matches = sorted(recurse_import(name, depth + 1, seen))
  87 
  88             for result in matches:
  89                 results.add(result)
  90 
  91         return results
  92 
  93     for module_util in module_utils:
  94         # recurse over module_utils imports while excluding self
  95         module_util_imports = recurse_import(module_util)
  96         module_util_imports.remove(module_util)
  97 
  98         # add recursive imports to all path entries which import this module_util
  99         for target_path in imports_by_target_path:
 100             if module_util in imports_by_target_path[target_path]:
 101                 for module_util_import in sorted(module_util_imports):
 102                     if module_util_import not in imports_by_target_path[target_path]:
 103                         display.info('%s inherits import %s via %s' % (target_path, module_util_import, module_util), verbosity=6)
 104                         imports_by_target_path[target_path].add(module_util_import)
 105 
 106     imports = dict([(module_util, set()) for module_util in module_utils | virtual_utils])
 107 
 108     for target_path in imports_by_target_path:
 109         for module_util in imports_by_target_path[target_path]:
 110             imports[module_util].add(target_path)
 111 
 112     # for purposes of mapping module_utils to paths, treat imports of virtual utils the same as the parent package
 113     for virtual_util in virtual_utils:
 114         parent_package = '.'.join(virtual_util.split('.')[:-1])
 115         imports[virtual_util] = imports[parent_package]
 116         display.info('%s reports imports from parent package %s' % (virtual_util, parent_package), verbosity=6)
 117 
 118     for module_util in sorted(imports):
 119         if not imports[module_util]:
 120             package_path = get_import_path(module_util, package=True)
 121 
 122             if os.path.exists(package_path) and not os.path.getsize(package_path):
 123                 continue  # ignore empty __init__.py files
 124 
 125             display.warning('No imports found which use the "%s" module_util.' % module_util)
 126 
 127     return imports

 130 def get_python_module_utils_name(path):  # type: (str) -> str
 131     """Return a namespace and name from the given module_utils path."""
 132     base_path = data_context().content.module_utils_path
 133 
 134     if data_context().content.collection:
 135         prefix = 'ansible_collections.' + data_context().content.collection.prefix + 'plugins.module_utils'
 136     else:
 137         prefix = 'ansible.module_utils'
 138 
 139     if path.endswith('/__init__.py'):
 140         path = os.path.dirname(path)
 141 
 142     if path == base_path:
 143         name = prefix
 144     else:
 145         name = prefix + '.' + os.path.splitext(os.path.relpath(path, base_path))[0].replace(os.path.sep, '.')
 146 
 147     return name

 150 def enumerate_module_utils():
 151     """Return a list of available module_utils imports.
 152     :rtype: set[str]
 153     """
 154     module_utils = []
 155 
 156     for path in data_context().content.walk_files(data_context().content.module_utils_path):
 157         ext = os.path.splitext(path)[1]
 158 
 159         if ext != '.py':
 160             continue
 161 
 162         module_utils.append(get_python_module_utils_name(path))
 163 
 164     return set(module_utils)

 167 def extract_python_module_utils_imports(path, module_utils):
 168     """Return a list of module_utils imports found in the specified source file.
 169     :type path: str
 170     :type module_utils: set[str]
 171     :rtype: set[str]
 172     """
 173     # Python code must be read as bytes to avoid a SyntaxError when the source uses comments to declare the file encoding.
 174     # See: https://www.python.org/dev/peps/pep-0263
 175     # Specifically: If a Unicode string with a coding declaration is passed to compile(), a SyntaxError will be raised.
 176     code = read_binary_file(path)
 177 
 178     try:
 179         tree = ast.parse(code)
 180     except SyntaxError as ex:
 181         # Treat this error as a warning so tests can be executed as best as possible.
 182         # The compile test will detect and report this syntax error.
 183         display.warning('%s:%s Syntax error extracting module_utils imports: %s' % (path, ex.lineno, ex.msg))
 184         return set()
 185 
 186     finder = ModuleUtilFinder(path, module_utils)
 187     finder.visit(tree)
 188     return finder.imports

 191 def get_import_path(name, package=False):  # type: (str, bool) -> str
 192     """Return a path from an import name."""
 193     if package:
 194         filename = os.path.join(name.replace('.', '/'), '__init__.py')
 195     else:
 196         filename = '%s.py' % name.replace('.', '/')
 197 
 198     if name.startswith('ansible.module_utils.') or name == 'ansible.module_utils':
 199         path = os.path.join('lib', filename)
 200     elif data_context().content.collection and (
 201             name.startswith('ansible_collections.%s.plugins.module_utils.' % data_context().content.collection.full_name) or
 202             name == 'ansible_collections.%s.plugins.module_utils' % data_context().content.collection.full_name):
 203         path = '/'.join(filename.split('/')[3:])
 204     else:
 205         raise Exception('Unexpected import name: %s' % name)
 206 
 207     return path

 210 class ModuleUtilFinder(ast.NodeVisitor):
 211     """AST visitor to find valid module_utils imports."""
 212     def __init__(self, path, module_utils):
 213         """Return a list of module_utils imports found in the specified source file.
 214         :type path: str
 215         :type module_utils: set[str]
 216         """
 217         self.path = path
 218         self.module_utils = module_utils
 219         self.imports = set()
 220 
 221         # implicitly import parent package
 222 
 223         if path.endswith('/__init__.py'):
 224             path = os.path.split(path)[0]
 225 
 226         if path.startswith('lib/ansible/module_utils/'):
 227             package = os.path.split(path)[0].replace('/', '.')[4:]
 228 
 229             if package != 'ansible.module_utils' and package not in VIRTUAL_PACKAGES:
 230                 self.add_import(package, 0)
 231 
 232     # noinspection PyPep8Naming
 233     # pylint: disable=locally-disabled, invalid-name
 234     def visit_Import(self, node):
 235         """
 236         :type node: ast.Import
 237         """
 238         self.generic_visit(node)
 239 
 240         # import ansible.module_utils.MODULE[.MODULE]
 241         # import ansible_collections.{ns}.{col}.plugins.module_utils.module_utils.MODULE[.MODULE]
 242         self.add_imports([alias.name for alias in node.names], node.lineno)
 243 
 244     # noinspection PyPep8Naming
 245     # pylint: disable=locally-disabled, invalid-name
 246     def visit_ImportFrom(self, node):
 247         """
 248         :type node: ast.ImportFrom
 249         """
 250         self.generic_visit(node)
 251 
 252         if not node.module:
 253             return
 254 
 255         if not node.module.startswith('ansible'):
 256             return
 257 
 258         # from ansible.module_utils import MODULE[, MODULE]
 259         # from ansible.module_utils.MODULE[.MODULE] import MODULE[, MODULE]
 260         # from ansible_collections.{ns}.{col}.plugins.module_utils import MODULE[, MODULE]
 261         # from ansible_collections.{ns}.{col}.plugins.module_utils.MODULE[.MODULE] import MODULE[, MODULE]
 262         self.add_imports(['%s.%s' % (node.module, alias.name) for alias in node.names], node.lineno)
 263 
 264     def add_import(self, name, line_number):
 265         """
 266         :type name: str
 267         :type line_number: int
 268         """
 269         import_name = name
 270 
 271         while self.is_module_util_name(name):
 272             if name in self.module_utils:
 273                 if name not in self.imports:
 274                     display.info('%s:%d imports module_utils: %s' % (self.path, line_number, name), verbosity=5)
 275                     self.imports.add(name)
 276 
 277                 return  # duplicate imports are ignored
 278 
 279             name = '.'.join(name.split('.')[:-1])
 280 
 281         if is_subdir(self.path, data_context().content.test_path):
 282             return  # invalid imports in tests are ignored
 283 
 284         # Treat this error as a warning so tests can be executed as best as possible.
 285         # This error should be detected by unit or integration tests.
 286         display.warning('%s:%d Invalid module_utils import: %s' % (self.path, line_number, import_name))
 287 
 288     def add_imports(self, names, line_no):  # type: (t.List[str], int) -> None
 289         """Add the given import names if they are module_utils imports."""
 290         for name in names:
 291             if self.is_module_util_name(name):
 292                 self.add_import(name, line_no)
 293 
 294     @staticmethod
 295     def is_module_util_name(name):  # type: (str) -> bool
 296         """Return True if the given name is a module_util name for the content under test. External module_utils are ignored."""
 297         if data_context().content.is_ansible and name.startswith('ansible.module_utils.'):
 298             return True
 299 
 300         if data_context().content.collection and name.startswith('ansible_collections.%s.plugins.module_utils.' % data_context().content.collection.full_name):
 301             return True
 302 
 303         return False
