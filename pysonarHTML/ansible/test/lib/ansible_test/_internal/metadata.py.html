  22 class Metadata:
  23     """Metadata object for passing data to delegated tests."""
  24     def __init__(self):
  25         """Initialize metadata."""
  26         self.changes = {}  # type: t.Dict[str, t.Tuple[t.Tuple[int, int]]]
  27         self.cloud_config = None  # type: t.Optional[t.Dict[str, str]]
  28         self.instance_config = None  # type: t.Optional[t.List[t.Dict[str, str]]]
  29         self.change_description = None  # type: t.Optional[ChangeDescription]
  30         self.ci_provider = None  # type: t.Optional[str]
  31 
  32     def populate_changes(self, diff):
  33         """
  34         :type diff: list[str] | None
  35         """
  36         patches = parse_diff(diff)
  37         patches = sorted(patches, key=lambda k: k.new.path)  # type: t.List[FileDiff]
  38 
  39         self.changes = dict((patch.new.path, tuple(patch.new.ranges)) for patch in patches)
  40 
  41         renames = [patch.old.path for patch in patches if patch.old.path != patch.new.path and patch.old.exists and patch.new.exists]
  42         deletes = [patch.old.path for patch in patches if not patch.new.exists]
  43 
  44         # make sure old paths which were renamed or deleted are registered in changes
  45         for path in renames + deletes:
  46             if path in self.changes:
  47                 # old path was replaced with another file
  48                 continue
  49 
  50             # failed tests involving deleted files should be using line 0 since there is no content remaining
  51             self.changes[path] = ((0, 0),)
  52 
  53     def to_dict(self):
  54         """
  55         :rtype: dict[str, any]
  56         """
  57         return dict(
  58             changes=self.changes,
  59             cloud_config=self.cloud_config,
  60             instance_config=self.instance_config,

  63         )
