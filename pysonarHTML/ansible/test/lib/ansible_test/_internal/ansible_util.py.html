  45 def ansible_environment(args, color=True, ansible_config=None):
  46     """
  47     :type args: CommonConfig
  48     :type color: bool
  49     :type ansible_config: str | None
  50     :rtype: dict[str, str]
  51     """
  52     env = common_environment()
  53     path = env['PATH']
  54 
  55     if not path.startswith(ANSIBLE_BIN_PATH + os.path.pathsep):
  56         path = ANSIBLE_BIN_PATH + os.path.pathsep + path
  57 
  58     if not ansible_config:
  59         # use the default empty configuration unless one has been provided
  60         ansible_config = args.get_ansible_config()
  61 
  62     if not args.explain and not os.path.exists(ansible_config):
  63         raise ApplicationError('Configuration not found: %s' % ansible_config)
  64 
  65     ansible = dict(
  66         ANSIBLE_PYTHON_MODULE_RLIMIT_NOFILE=str(SOFT_RLIMIT_NOFILE),
  67         ANSIBLE_FORCE_COLOR='%s' % 'true' if args.color and color else 'false',
  68         ANSIBLE_FORCE_HANDLERS='true',  # allow cleanup handlers to run when tests fail
  69         ANSIBLE_HOST_PATTERN_MISMATCH='error',  # prevent tests from unintentionally passing when hosts are not found
  70         ANSIBLE_INVENTORY='/dev/null',  # force tests to provide inventory
  71         ANSIBLE_DEPRECATION_WARNINGS='false',
  72         ANSIBLE_HOST_KEY_CHECKING='false',
  73         ANSIBLE_RETRY_FILES_ENABLED='false',
  74         ANSIBLE_CONFIG=ansible_config,
  75         ANSIBLE_LIBRARY='/dev/null',
  76         ANSIBLE_DEVEL_WARNING='false',  # Don't show warnings that CI is running devel
  77         PYTHONPATH=get_ansible_python_path(),
  78         PAGER='/bin/cat',
  79         PATH=path,
  80         # give TQM worker processes time to report code coverage results
  81         # without this the last task in a play may write no coverage file, an empty file, or an incomplete file
  82         # enabled even when not using code coverage to surface warnings when worker processes do not exit cleanly
  83         ANSIBLE_WORKER_SHUTDOWN_POLL_COUNT='100',
  84         ANSIBLE_WORKER_SHUTDOWN_POLL_DELAY='0.1',
  85     )
  86 
  87     if isinstance(args, IntegrationConfig) and args.coverage:
  88         # standard path injection is not effective for ansible-connection, instead the location must be configured
  89         # ansible-connection only requires the injector for code coverage
  90         # the correct python interpreter is already selected using the sys.executable used to invoke ansible
  91         ansible.update(dict(
  92             ANSIBLE_CONNECTION_PATH=os.path.join(ANSIBLE_TEST_DATA_ROOT, 'injector', 'ansible-connection'),
  93         ))
  94 
  95     if isinstance(args, PosixIntegrationConfig):
  96         ansible.update(dict(
  97             ANSIBLE_PYTHON_INTERPRETER='/set/ansible_python_interpreter/in/inventory',  # force tests to set ansible_python_interpreter in inventory
  98         ))
  99 
 100     env.update(ansible)
 101 
 102     if args.debug:
 103         env.update(dict(
 104             ANSIBLE_DEBUG='true',
 105             ANSIBLE_LOG_PATH=os.path.join(ResultType.LOGS.name, 'debug.log'),
 106         ))
 107 
 108     if data_context().content.collection:
 109         env.update(dict(
 110             ANSIBLE_COLLECTIONS_PATH=data_context().content.collection.root,
 111         ))
 112 
 113     if data_context().content.is_ansible:
 114         env.update(configure_plugin_paths(args))
 115 
 116     return env

 119 def configure_plugin_paths(args):  # type: (CommonConfig) -> t.Dict[str, str]
 120     """Return environment variables with paths to plugins relevant for the current command."""
 121     if not isinstance(args, IntegrationConfig):
 122         return {}
 123 
 124     support_path = os.path.join(ANSIBLE_SOURCE_ROOT, 'test', 'support', args.command)
 125 
 126     # provide private copies of collections for integration tests
 127     collection_root = os.path.join(support_path, 'collections')
 128 
 129     env = dict(
 130         ANSIBLE_COLLECTIONS_PATH=collection_root,
 131     )
 132 
 133     # provide private copies of plugins for integration tests
 134     plugin_root = os.path.join(support_path, 'plugins')
 135 
 136     plugin_list = [
 137         'action',
 138         'become',
 139         'cache',
 140         'callback',
 141         'cliconf',
 142         'connection',
 143         'filter',
 144         'httpapi',
 145         'inventory',
 146         'lookup',
 147         'netconf',
 148         # 'shell' is not configurable
 149         'strategy',
 150         'terminal',
 151         'test',
 152         'vars',
 153     ]
 154 
 155     # most plugins follow a standard naming convention
 156     plugin_map = dict(('%s_plugins' % name, name) for name in plugin_list)
 157 
 158     # these plugins do not follow the standard naming convention
 159     plugin_map.update(
 160         doc_fragment='doc_fragments',
 161         library='modules',
 162         module_utils='module_utils',
 163     )
 164 
 165     env.update(dict(('ANSIBLE_%s' % key.upper(), os.path.join(plugin_root, value)) for key, value in plugin_map.items()))
 166 
 167     # only configure directories which exist
 168     env = dict((key, value) for key, value in env.items() if os.path.isdir(value))
 169 
 170     return env

 173 def get_ansible_python_path():  # type: () -> str
 174     """
 175     Return a directory usable for PYTHONPATH, containing only the ansible package.
 176     If a temporary directory is required, it will be cached for the lifetime of the process and cleaned up at exit.
 177     """
 178     if ANSIBLE_SOURCE_ROOT:
 179         # when running from source there is no need for a temporary directory to isolate the ansible package
 180         return os.path.dirname(ANSIBLE_LIB_ROOT)
 181 
 182     try:
 183         return get_ansible_python_path.python_path
 184     except AttributeError:
 185         pass
 186 
 187     python_path = create_temp_dir(prefix='ansible-test-')
 188     get_ansible_python_path.python_path = python_path
 189 
 190     os.symlink(ANSIBLE_LIB_ROOT, os.path.join(python_path, 'ansible'))
 191 
 192     return python_path

 195 def check_pyyaml(args, version, required=True, quiet=False):
 196     """
 197     :type args: EnvironmentConfig
 198     :type version: str
 199     :type required: bool
 200     :type quiet: bool
 201     """
 202     try:
 203         return CHECK_YAML_VERSIONS[version]
 204     except KeyError:
 205         pass
 206 
 207     python = find_python(version)
 208     stdout, _dummy = run_command(args, [python, os.path.join(ANSIBLE_TEST_DATA_ROOT, 'yamlcheck.py')],
 209                                  capture=True, always=True)
 210 
 211     result = json.loads(stdout)
 212 
 213     yaml = result['yaml']
 214     cloader = result['cloader']
 215 
 216     if yaml or required:
 217         # results are cached only if pyyaml is required or present
 218         # it is assumed that tests will not uninstall/re-install pyyaml -- if they do, those changes will go undetected
 219         CHECK_YAML_VERSIONS[version] = result
 220 
 221     if not quiet:
 222         if not yaml and required:
 223             display.warning('PyYAML is not installed for interpreter: %s' % python)
 224         elif not cloader:
 225             display.warning('PyYAML will be slow due to installation without libyaml support for interpreter: %s' % python)
 226 
 227     return result

 243 def get_collection_detail(args, python):  # type: (EnvironmentConfig, str) -> CollectionDetail
 244     """Return collection detail."""
 245     collection = data_context().content.collection
 246     directory = os.path.join(collection.root, collection.directory)
 247 
 248     stdout = run_command(args, [python, os.path.join(ANSIBLE_TEST_DATA_ROOT, 'collection_detail.py'), directory], capture=True, always=True)[0]
 249     result = json.loads(stdout)
 250     error = result.get('error')
 251 
 252     if error:
 253         raise CollectionDetailError(error)
 254 
 255     version = result.get('version')
 256 
 257     detail = CollectionDetail()
 258     detail.version = str(version) if version is not None else None
 259 
 260     return detail
