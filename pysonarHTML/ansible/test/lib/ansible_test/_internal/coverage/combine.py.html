  40 def command_coverage_combine(args):
  41     """Patch paths in coverage files and merge into a single file.
  42     :type args: CoverageConfig
  43     :rtype: list[str]
  44     """
  45     paths = _command_coverage_combine_powershell(args) + _command_coverage_combine_python(args)
  46 
  47     for path in paths:
  48         display.info('Generated combined output: %s' % path, verbosity=1)
  49 
  50     return paths

  53 def _command_coverage_combine_python(args):
  54     """
  55     :type args: CoverageConfig
  56     :rtype: list[str]
  57     """
  58     coverage = initialize_coverage(args)
  59 
  60     modules = get_python_modules()
  61 
  62     coverage_files = get_python_coverage_files()
  63 
  64     counter = 0
  65     sources = _get_coverage_targets(args, walk_compile_targets)
  66     groups = _build_stub_groups(args, sources, lambda line_count: set())
  67 
  68     collection_search_re, collection_sub_re = get_collection_path_regexes()
  69 
  70     for coverage_file in coverage_files:
  71         counter += 1
  72         display.info('[%4d/%4d] %s' % (counter, len(coverage_files), coverage_file), verbosity=2)
  73 
  74         group = get_coverage_group(args, coverage_file)
  75 
  76         if group is None:
  77             display.warning('Unexpected name for coverage file: %s' % coverage_file)
  78             continue
  79 
  80         for filename, arcs in enumerate_python_arcs(coverage_file, coverage, modules, collection_search_re, collection_sub_re):
  81             if group not in groups:
  82                 groups[group] = {}
  83 
  84             arc_data = groups[group]
  85 
  86             if filename not in arc_data:
  87                 arc_data[filename] = set()
  88 
  89             arc_data[filename].update(arcs)
  90 
  91     output_files = []
  92 
  93     coverage_file = os.path.join(ResultType.COVERAGE.path, COVERAGE_OUTPUT_FILE_NAME)
  94 
  95     path_checker = PathChecker(args, collection_search_re)
  96 
  97     for group in sorted(groups):
  98         arc_data = groups[group]
  99 
 100         updated = coverage.CoverageData()
 101 
 102         for filename in arc_data:
 103             if not path_checker.check_path(filename):
 104                 continue
 105 
 106             updated.add_arcs({filename: list(arc_data[filename])})
 107 
 108         if args.all:
 109             updated.add_arcs(dict((source[0], []) for source in sources))
 110 
 111         if not args.explain:
 112             output_file = coverage_file + group
 113             updated.write_file(output_file)  # always write files to make sure stale files do not exist
 114 
 115             if updated:
 116                 # only report files which are non-empty to prevent coverage from reporting errors
 117                 output_files.append(output_file)
 118 
 119     path_checker.report()
 120 
 121     return sorted(output_files)

 124 def _command_coverage_combine_powershell(args):
 125     """
 126     :type args: CoverageConfig
 127     :rtype: list[str]
 128     """
 129     coverage_files = get_powershell_coverage_files()
 130 
 131     def _default_stub_value(lines):
 132         val = {}
 133         for line in range(lines):
 134             val[line] = 0
 135         return val
 136 
 137     counter = 0
 138     sources = _get_coverage_targets(args, walk_powershell_targets)
 139     groups = _build_stub_groups(args, sources, _default_stub_value)
 140 
 141     collection_search_re, collection_sub_re = get_collection_path_regexes()
 142 
 143     for coverage_file in coverage_files:
 144         counter += 1
 145         display.info('[%4d/%4d] %s' % (counter, len(coverage_files), coverage_file), verbosity=2)
 146 
 147         group = get_coverage_group(args, coverage_file)
 148 
 149         if group is None:
 150             display.warning('Unexpected name for coverage file: %s' % coverage_file)
 151             continue
 152 
 153         for filename, hits in enumerate_powershell_lines(coverage_file, collection_search_re, collection_sub_re):
 154             if group not in groups:
 155                 groups[group] = {}
 156 
 157             coverage_data = groups[group]
 158 
 159             if filename not in coverage_data:
 160                 coverage_data[filename] = {}
 161 
 162             file_coverage = coverage_data[filename]
 163 
 164             for line_no, hit_count in hits.items():
 165                 file_coverage[line_no] = file_coverage.get(line_no, 0) + hit_count
 166 
 167     output_files = []
 168 
 169     path_checker = PathChecker(args)
 170 
 171     for group in sorted(groups):
 172         coverage_data = dict((filename, data) for filename, data in groups[group].items() if path_checker.check_path(filename))
 173 
 174         if args.all:
 175             # Add 0 line entries for files not in coverage_data
 176             for source, source_line_count in sources:
 177                 if source in coverage_data:
 178                     continue
 179 
 180                 coverage_data[source] = _default_stub_value(source_line_count)
 181 
 182         if not args.explain:
 183             output_file = COVERAGE_OUTPUT_FILE_NAME + group + '-powershell'
 184 
 185             write_json_test_results(ResultType.COVERAGE, output_file, coverage_data)
 186 
 187             output_files.append(os.path.join(ResultType.COVERAGE.path, output_file))
 188 
 189     path_checker.report()
 190 
 191     return sorted(output_files)

 194 def _get_coverage_targets(args, walk_func):
 195     """
 196     :type args: CoverageConfig
 197     :type walk_func: Func
 198     :rtype: list[tuple[str, int]]
 199     """
 200     sources = []
 201 
 202     if args.all or args.stub:
 203         # excludes symlinks of regular files to avoid reporting on the same file multiple times
 204         # in the future it would be nice to merge any coverage for symlinks into the real files
 205         for target in walk_func(include_symlinks=False):
 206             target_path = os.path.abspath(target.path)
 207 
 208             target_lines = len(read_text_file(target_path).splitlines())
 209 
 210             sources.append((target_path, target_lines))
 211 
 212         sources.sort()
 213 
 214     return sources

 217 def _build_stub_groups(args, sources, default_stub_value):
 218     """
 219     :type args: CoverageConfig
 220     :type sources: List[tuple[str, int]]
 221     :type default_stub_value: Func[int]
 222     :rtype: dict
 223     """
 224     groups = {}
 225 
 226     if args.stub:
 227         stub_group = []
 228         stub_groups = [stub_group]
 229         stub_line_limit = 500000
 230         stub_line_count = 0
 231 
 232         for source, source_line_count in sources:
 233             stub_group.append((source, source_line_count))
 234             stub_line_count += source_line_count
 235 
 236             if stub_line_count > stub_line_limit:
 237                 stub_line_count = 0
 238                 stub_group = []
 239                 stub_groups.append(stub_group)
 240 
 241         for stub_index, stub_group in enumerate(stub_groups):
 242             if not stub_group:
 243                 continue
 244 
 245             groups['=stub-%02d' % (stub_index + 1)] = dict((source, default_stub_value(line_count))
 246                                                            for source, line_count in stub_group)
 247 
 248     return groups

 251 def get_coverage_group(args, coverage_file):
 252     """
 253     :type args: CoverageConfig
 254     :type coverage_file: str
 255     :rtype: str
 256     """
 257     parts = os.path.basename(coverage_file).split('=', 4)
 258 
 259     # noinspection PyTypeChecker
 260     if len(parts) != 5 or not parts[4].startswith('coverage.'):
 261         return None
 262 
 263     names = dict(
 264         command=parts[0],
 265         target=parts[1],
 266         environment=parts[2],
 267         version=parts[3],
 268     )
 269 
 270     group = ''
 271 
 272     for part in COVERAGE_GROUPS:
 273         if part in args.group_by:
 274             group += '=%s' % names[part]
 275 
 276     return group
