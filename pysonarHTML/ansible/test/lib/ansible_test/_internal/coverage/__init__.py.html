  56 class CoverageConfig(EnvironmentConfig):
  57     """Configuration for the coverage command."""
  58     def __init__(self, args):  # type: (t.Any) -> None
  59         super(CoverageConfig, self).__init__(args, 'coverage')
  60 
  61         self.group_by = frozenset(args.group_by) if 'group_by' in args and args.group_by else set()  # type: t.FrozenSet[str]
  62         self.all = args.all if 'all' in args else False  # type: bool
  63         self.stub = args.stub if 'stub' in args else False  # type: bool
  64         self.coverage = False  # temporary work-around to support intercept_command in cover.py

  67 def initialize_coverage(args):  # type: (CoverageConfig) -> coverage_module
  68     """Delegate execution if requested, install requirements, then import and return the coverage module. Raises an exception if coverage is not available."""
  69     if args.delegate:
  70         raise Delegate()
  71 
  72     if args.requirements:
  73         install_command_requirements(args)
  74 
  75     try:
  76         import coverage
  77     except ImportError:
  78         coverage = None
  79 
  80     if not coverage:
  81         raise ApplicationError('You must install the "coverage" python module to use this command.')
  82 
  83     coverage_version_string = coverage.__version__
  84     coverage_version = tuple(int(v) for v in coverage_version_string.split('.'))
  85 
  86     min_version = (4, 2)
  87     max_version = (5, 0)
  88 
  89     supported_version = True
  90     recommended_version = '4.5.4'
  91 
  92     if coverage_version < min_version or coverage_version >= max_version:
  93         supported_version = False
  94 
  95     if not supported_version:
  96         raise ApplicationError('Version %s of "coverage" is not supported. Version %s is known to work and is recommended.' % (
  97             coverage_version_string, recommended_version))
  98 
  99     return coverage

 122 def get_coverage_files(language, path=None):  # type: (str, t.Optional[str]) -> t.List[str]
 123     """Return the list of coverage file paths for the given language."""
 124     coverage_dir = path or ResultType.COVERAGE.path
 125     coverage_files = [os.path.join(coverage_dir, f) for f in os.listdir(coverage_dir)
 126                       if '=coverage.' in f and '=%s' % language in f]
 127 
 128     return coverage_files

 148 def enumerate_python_arcs(
 149         path,  # type: str
 150         coverage,  # type: coverage_module
 151         modules,  # type: t.Dict[str, str]
 152         collection_search_re,  # type: t.Optional[t.Pattern]
 153         collection_sub_re,  # type: t.Optional[t.Pattern]
 154 ):  # type: (...) -> t.Generator[t.Tuple[str, t.Set[t.Tuple[int, int]]]]
 155     """Enumerate Python code coverage arcs in the given file."""
 156     if os.path.getsize(path) == 0:
 157         display.warning('Empty coverage file: %s' % path, verbosity=2)
 158         return
 159 
 160     original = coverage.CoverageData()
 161 
 162     try:
 163         original.read_file(path)
 164     except Exception as ex:  # pylint: disable=locally-disabled, broad-except
 165         with open_binary_file(path) as file:
 166             header = file.read(6)
 167 
 168         if header == b'SQLite':
 169             display.error('File created by "coverage" 5.0+: %s' % os.path.relpath(path))
 170         else:
 171             display.error(u'%s' % ex)
 172 
 173         return
 174 
 175     for filename in original.measured_files():
 176         arcs = original.arcs(filename)
 177 
 178         if not arcs:
 179             # This is most likely due to using an unsupported version of coverage.
 180             display.warning('No arcs found for "%s" in coverage file: %s' % (filename, path))
 181             continue
 182 
 183         filename = sanitize_filename(filename, modules=modules, collection_search_re=collection_search_re, collection_sub_re=collection_sub_re)
 184 
 185         if not filename:
 186             continue
 187 
 188         yield filename, set(arcs)

 191 def enumerate_powershell_lines(
 192         path,  # type: str
 193         collection_search_re,  # type: t.Optional[t.Pattern]
 194         collection_sub_re,  # type: t.Optional[t.Pattern]
 195 ):  # type: (...) -> t.Generator[t.Tuple[str, t.Dict[int, int]]]
 196     """Enumerate PowerShell code coverage lines in the given file."""
 197     if os.path.getsize(path) == 0:
 198         display.warning('Empty coverage file: %s' % path, verbosity=2)
 199         return
 200 
 201     try:
 202         coverage_run = read_json_file(path)
 203     except Exception as ex:  # pylint: disable=locally-disabled, broad-except
 204         display.error(u'%s' % ex)
 205         return
 206 
 207     for filename, hits in coverage_run.items():
 208         filename = sanitize_filename(filename, collection_search_re=collection_search_re, collection_sub_re=collection_sub_re)
 209 
 210         if not filename:
 211             continue
 212 
 213         # PowerShell unpacks arrays if there's only a single entry so this is a defensive check on that
 214         if not isinstance(hits, list):
 215             hits = [hits]
 216 
 217         hits = dict((hit['Line'], hit['HitCount']) for hit in hits if hit)
 218 
 219         yield filename, hits

 222 def sanitize_filename(
 223         filename,  # type: str
 224         modules=None,  # type: t.Optional[t.Dict[str, str]]
 225         collection_search_re=None,  # type: t.Optional[t.Pattern]
 226         collection_sub_re=None,  # type: t.Optional[t.Pattern]
 227 ):  # type: (...) -> t.Optional[str]
 228     """Convert the given code coverage path to a local absolute path and return its, or None if the path is not valid."""
 229     ansible_path = os.path.abspath('lib/ansible/') + '/'
 230     root_path = data_context().content.root + '/'
 231     integration_temp_path = os.path.sep + os.path.join(ResultType.TMP.relative_path, 'integration') + os.path.sep
 232 
 233     if modules is None:
 234         modules = {}
 235 
 236     if '/ansible_modlib.zip/ansible/' in filename:
 237         # Rewrite the module_utils path from the remote host to match the controller. Ansible 2.6 and earlier.
 238         new_name = re.sub('^.*/ansible_modlib.zip/ansible/', ansible_path, filename)
 239         display.info('%s -> %s' % (filename, new_name), verbosity=3)
 240         filename = new_name
 241     elif collection_search_re and collection_search_re.search(filename):
 242         new_name = os.path.abspath(collection_sub_re.sub('', filename))
 243         display.info('%s -> %s' % (filename, new_name), verbosity=3)
 244         filename = new_name
 245     elif re.search(r'/ansible_[^/]+_payload\.zip/ansible/', filename):
 246         # Rewrite the module_utils path from the remote host to match the controller. Ansible 2.7 and later.
 247         new_name = re.sub(r'^.*/ansible_[^/]+_payload\.zip/ansible/', ansible_path, filename)
 248         display.info('%s -> %s' % (filename, new_name), verbosity=3)
 249         filename = new_name
 250     elif '/ansible_module_' in filename:
 251         # Rewrite the module path from the remote host to match the controller. Ansible 2.6 and earlier.
 252         module_name = re.sub('^.*/ansible_module_(?P<module>.*).py$', '\\g<module>', filename)
 253         if module_name not in modules:
 254             display.warning('Skipping coverage of unknown module: %s' % module_name)
 255             return None
 256         new_name = os.path.abspath(modules[module_name])
 257         display.info('%s -> %s' % (filename, new_name), verbosity=3)
 258         filename = new_name
 259     elif re.search(r'/ansible_[^/]+_payload(_[^/]+|\.zip)/__main__\.py$', filename):
 260         # Rewrite the module path from the remote host to match the controller. Ansible 2.7 and later.
 261         # AnsiballZ versions using zipimporter will match the `.zip` portion of the regex.
 262         # AnsiballZ versions not using zipimporter will match the `_[^/]+` portion of the regex.
 263         module_name = re.sub(r'^.*/ansible_(?P<module>[^/]+)_payload(_[^/]+|\.zip)/__main__\.py$',
 264                              '\\g<module>', filename).rstrip('_')
 265         if module_name not in modules:
 266             display.warning('Skipping coverage of unknown module: %s' % module_name)
 267             return None
 268         new_name = os.path.abspath(modules[module_name])
 269         display.info('%s -> %s' % (filename, new_name), verbosity=3)
 270         filename = new_name
 271     elif re.search('^(/.*?)?/root/ansible/', filename):
 272         # Rewrite the path of code running on a remote host or in a docker container as root.
 273         new_name = re.sub('^(/.*?)?/root/ansible/', root_path, filename)
 274         display.info('%s -> %s' % (filename, new_name), verbosity=3)
 275         filename = new_name
 276     elif integration_temp_path in filename:
 277         # Rewrite the path of code running from an integration test temporary directory.
 278         new_name = re.sub(r'^.*' + re.escape(integration_temp_path) + '[^/]+/', root_path, filename)
 279         display.info('%s -> %s' % (filename, new_name), verbosity=3)
 280         filename = new_name
 281 
 282     return filename

 285 class PathChecker:
 286     """Checks code coverage paths to verify they are valid and reports on the findings."""
 287     def __init__(self, args, collection_search_re=None):  # type: (CoverageConfig, t.Optional[t.Pattern]) -> None
 288         self.args = args
 289         self.collection_search_re = collection_search_re
 290         self.invalid_paths = []
 291         self.invalid_path_chars = 0
 292 
 293     def check_path(self, path):  # type: (str) -> bool
 294         """Return True if the given coverage path is valid, otherwise display a warning and return False."""
 295         if os.path.isfile(to_bytes(path)):
 296             return True
 297 
 298         if self.collection_search_re and self.collection_search_re.search(path) and os.path.basename(path) == '__init__.py':
 299             # the collection loader uses implicit namespace packages, so __init__.py does not need to exist on disk
 300             # coverage is still reported for these non-existent files, but warnings are not needed
 301             return False
 302 
 303         self.invalid_paths.append(path)
 304         self.invalid_path_chars += len(path)
 305 
 306         if self.args.verbosity > 1:
 307             display.warning('Invalid coverage path: %s' % path)
 308 
 309         return False
 310 
 311     def report(self):  # type: () -> None
 312         """Display a warning regarding invalid paths if any were found."""
 313         if self.invalid_paths:
 314             display.warning('Ignored %d characters from %d invalid coverage path(s).' % (self.invalid_path_chars, len(self.invalid_paths)))
