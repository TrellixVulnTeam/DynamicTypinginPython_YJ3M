  47 def command_coverage_xml(args):
  48     """
  49     :type args: CoverageConfig
  50     """
  51     output_files = command_coverage_combine(args)
  52 
  53     for output_file in output_files:
  54         xml_name = '%s.xml' % os.path.basename(output_file)
  55         if output_file.endswith('-powershell'):
  56             report = _generate_powershell_xml(output_file)
  57 
  58             rough_string = tostring(report, 'utf-8')
  59             reparsed = minidom.parseString(rough_string)
  60             pretty = reparsed.toprettyxml(indent='    ')
  61 
  62             write_text_test_results(ResultType.REPORTS, xml_name, pretty)
  63         else:
  64             xml_path = os.path.join(ResultType.REPORTS.path, xml_name)
  65             make_dirs(ResultType.REPORTS.path)
  66             run_coverage(args, output_file, 'xml', ['-i', '-o', xml_path])

  69 def _generate_powershell_xml(coverage_file):
  70     """
  71     :type coverage_file: str
  72     :rtype: Element
  73     """
  74     coverage_info = read_json_file(coverage_file)
  75 
  76     content_root = data_context().content.root
  77     is_ansible = data_context().content.is_ansible
  78 
  79     packages = {}
  80     for path, results in coverage_info.items():
  81         filename = os.path.splitext(os.path.basename(path))[0]
  82 
  83         if filename.startswith('Ansible.ModuleUtils'):
  84             package = 'ansible.module_utils'
  85         elif is_ansible:
  86             package = 'ansible.modules'
  87         else:
  88             rel_path = path[len(content_root) + 1:]
  89             plugin_type = "modules" if rel_path.startswith("plugins/modules") else "module_utils"
  90             package = 'ansible_collections.%splugins.%s' % (data_context().content.collection.prefix, plugin_type)
  91 
  92         if package not in packages:
  93             packages[package] = {}
  94 
  95         packages[package][path] = results
  96 
  97     elem_coverage = Element('coverage')
  98     elem_coverage.append(
  99         Comment(' Generated by ansible-test from the Ansible project: https://www.ansible.com/ '))
 100     elem_coverage.append(
 101         Comment(' Based on https://raw.githubusercontent.com/cobertura/web/master/htdocs/xml/coverage-04.dtd '))
 102 
 103     elem_sources = SubElement(elem_coverage, 'sources')
 104 
 105     elem_source = SubElement(elem_sources, 'source')
 106     elem_source.text = data_context().content.root
 107 
 108     elem_packages = SubElement(elem_coverage, 'packages')
 109 
 110     total_lines_hit = 0
 111     total_line_count = 0
 112 
 113     for package_name, package_data in packages.items():
 114         lines_hit, line_count = _add_cobertura_package(elem_packages, package_name, package_data)
 115 
 116         total_lines_hit += lines_hit
 117         total_line_count += line_count
 118 
 119     elem_coverage.attrib.update({
 120         'branch-rate': '0',
 121         'branches-covered': '0',
 122         'branches-valid': '0',
 123         'complexity': '0',
 124         'line-rate': str(round(total_lines_hit / total_line_count, 4)) if total_line_count else "0",
 125         'lines-covered': str(total_line_count),
 126         'lines-valid': str(total_lines_hit),
 127         'timestamp': str(int(time.time())),
 128         'version': get_ansible_version(),
 129     })
 130 
 131     return elem_coverage

 134 def _add_cobertura_package(packages, package_name, package_data):
 135     """
 136     :type packages: SubElement
 137     :type package_name: str
 138     :type package_data: Dict[str, Dict[str, int]]
 139     :rtype: Tuple[int, int]
 140     """
 141     elem_package = SubElement(packages, 'package')
 142     elem_classes = SubElement(elem_package, 'classes')
 143 
 144     total_lines_hit = 0
 145     total_line_count = 0
 146 
 147     for path, results in package_data.items():
 148         lines_hit = len([True for hits in results.values() if hits])
 149         line_count = len(results)
 150 
 151         total_lines_hit += lines_hit
 152         total_line_count += line_count
 153 
 154         elem_class = SubElement(elem_classes, 'class')
 155 
 156         class_name = os.path.splitext(os.path.basename(path))[0]
 157         if class_name.startswith("Ansible.ModuleUtils"):
 158             class_name = class_name[20:]
 159 
 160         content_root = data_context().content.root
 161         filename = path
 162         if filename.startswith(content_root):
 163             filename = filename[len(content_root) + 1:]
 164 
 165         elem_class.attrib.update({
 166             'branch-rate': '0',
 167             'complexity': '0',
 168             'filename': filename,
 169             'line-rate': str(round(lines_hit / line_count, 4)) if line_count else "0",
 170             'name': class_name,
 171         })
 172 
 173         SubElement(elem_class, 'methods')
 174 
 175         elem_lines = SubElement(elem_class, 'lines')
 176 
 177         for number, hits in results.items():
 178             elem_line = SubElement(elem_lines, 'line')
 179             elem_line.attrib.update(
 180                 hits=str(hits),
 181                 number=str(number),
 182             )
 183 
 184     elem_package.attrib.update({
 185         'branch-rate': '0',
 186         'complexity': '0',
 187         'line-rate': str(round(total_lines_hit / total_line_count, 4)) if total_line_count else "0",
 188         'name': package_name,
 189     })
 190 
 191     return total_lines_hit, total_line_count
