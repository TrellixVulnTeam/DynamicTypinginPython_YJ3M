  29 def command_coverage_report(args):
  30     """
  31     :type args: CoverageReportConfig
  32     """
  33     output_files = command_coverage_combine(args)
  34 
  35     for output_file in output_files:
  36         if args.group_by or args.stub:
  37             display.info('>>> Coverage Group: %s' % ' '.join(os.path.basename(output_file).split('=')[1:]))
  38 
  39         if output_file.endswith('-powershell'):
  40             display.info(_generate_powershell_output_report(args, output_file))
  41         else:
  42             options = []
  43 
  44             if args.show_missing:
  45                 options.append('--show-missing')
  46 
  47             if args.include:
  48                 options.extend(['--include', args.include])
  49 
  50             if args.omit:
  51                 options.extend(['--omit', args.omit])
  52 
  53             run_coverage(args, output_file, 'report', options)

  56 def _generate_powershell_output_report(args, coverage_file):
  57     """
  58     :type args: CoverageReportConfig
  59     :type coverage_file: str
  60     :rtype: str
  61     """
  62     coverage_info = read_json_file(coverage_file)
  63 
  64     root_path = data_context().content.root + '/'
  65 
  66     name_padding = 7
  67     cover_padding = 8
  68 
  69     file_report = []
  70     total_stmts = 0
  71     total_miss = 0
  72 
  73     for filename in sorted(coverage_info.keys()):
  74         hit_info = coverage_info[filename]
  75 
  76         if filename.startswith(root_path):
  77             filename = filename[len(root_path):]
  78 
  79         if args.omit and filename in args.omit:
  80             continue
  81         if args.include and filename not in args.include:
  82             continue
  83 
  84         stmts = len(hit_info)
  85         miss = len([c for c in hit_info.values() if c == 0])
  86 
  87         name_padding = max(name_padding, len(filename) + 3)
  88 
  89         total_stmts += stmts
  90         total_miss += miss
  91 
  92         cover = "{0}%".format(int((stmts - miss) / stmts * 100))
  93 
  94         missing = []
  95         current_missing = None
  96         sorted_lines = sorted([int(x) for x in hit_info.keys()])
  97         for idx, line in enumerate(sorted_lines):
  98             hit = hit_info[str(line)]
  99             if hit == 0 and current_missing is None:
 100                 current_missing = line
 101             elif hit != 0 and current_missing is not None:
 102                 end_line = sorted_lines[idx - 1]
 103                 if current_missing == end_line:
 104                     missing.append(str(current_missing))
 105                 else:
 106                     missing.append('%s-%s' % (current_missing, end_line))
 107                 current_missing = None
 108 
 109         if current_missing is not None:
 110             end_line = sorted_lines[-1]
 111             if current_missing == end_line:
 112                 missing.append(str(current_missing))
 113             else:
 114                 missing.append('%s-%s' % (current_missing, end_line))
 115 
 116         file_report.append({'name': filename, 'stmts': stmts, 'miss': miss, 'cover': cover, 'missing': missing})
 117 
 118     if total_stmts == 0:
 119         return ''
 120 
 121     total_percent = '{0}%'.format(int((total_stmts - total_miss) / total_stmts * 100))
 122     stmts_padding = max(8, len(str(total_stmts)))
 123     miss_padding = max(7, len(str(total_miss)))
 124 
 125     line_length = name_padding + stmts_padding + miss_padding + cover_padding
 126 
 127     header = 'Name'.ljust(name_padding) + 'Stmts'.rjust(stmts_padding) + 'Miss'.rjust(miss_padding) + \
 128              'Cover'.rjust(cover_padding)
 129 
 130     if args.show_missing:
 131         header += 'Lines Missing'.rjust(16)
 132         line_length += 16
 133 
 134     line_break = '-' * line_length
 135     lines = ['%s%s%s%s%s' % (f['name'].ljust(name_padding), str(f['stmts']).rjust(stmts_padding),
 136                              str(f['miss']).rjust(miss_padding), f['cover'].rjust(cover_padding),
 137                              '   ' + ', '.join(f['missing']) if args.show_missing else '')
 138              for f in file_report]
 139     totals = 'TOTAL'.ljust(name_padding) + str(total_stmts).rjust(stmts_padding) + \
 140              str(total_miss).rjust(miss_padding) + total_percent.rjust(cover_padding)
 141 
 142     report = '{0}\n{1}\n{2}\n{1}\n{3}'.format(header, line_break, "\n".join(lines), totals)
 143     return report
