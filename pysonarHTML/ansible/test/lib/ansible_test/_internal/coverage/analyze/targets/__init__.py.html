  41 def make_report(target_indexes, arcs, lines):  # type: (TargetIndexes, Arcs, Lines) -> t.Dict[str, t.Any]
  42     """Condense target indexes, arcs and lines into a compact report."""
  43     set_indexes = {}
  44     arc_refs = dict((path, dict((format_arc(arc), get_target_set_index(indexes, set_indexes)) for arc, indexes in data.items())) for path, data in arcs.items())
  45     line_refs = dict((path, dict((line, get_target_set_index(indexes, set_indexes)) for line, indexes in data.items())) for path, data in lines.items())
  46 
  47     report = dict(
  48         targets=[name for name, index in sorted(target_indexes.items(), key=lambda kvp: kvp[1])],
  49         target_sets=[sorted(data) for data, index in sorted(set_indexes.items(), key=lambda kvp: kvp[1])],
  50         arcs=arc_refs,
  51         lines=line_refs,
  52     )
  53 
  54     return report

  57 def load_report(report):  # type: (t.Dict[str, t.Any]) -> t.Tuple[t.List[str], Arcs, Lines]
  58     """Extract target indexes, arcs and lines from an existing report."""
  59     try:
  60         target_indexes = report['targets']  # type: t.List[str]
  61         target_sets = report['target_sets']  # type: t.List[t.List[int]]
  62         arc_data = report['arcs']  # type: t.Dict[str, t.Dict[str, int]]
  63         line_data = report['lines']  # type: t.Dict[str, t.Dict[int, int]]
  64     except KeyError as ex:
  65         raise ApplicationError('Document is missing key "%s".' % ex.args)
  66     except TypeError:
  67         raise ApplicationError('Document is type "%s" instead of "dict".' % type(report).__name__)
  68 
  69     arcs = dict((path, dict((parse_arc(arc), set(target_sets[index])) for arc, index in data.items())) for path, data in arc_data.items())
  70     lines = dict((path, dict((int(line), set(target_sets[index])) for line, index in data.items())) for path, data in line_data.items())
  71 
  72     return target_indexes, arcs, lines

 121 def expand_indexes(
 122         source_data,  # type: IndexedPoints
 123         source_index,  # type: t.List[str]
 124         format_func,  # type: t.Callable[t.Tuple[t.Any], str]
 125 ):  # type: (...) -> NamedPoints
 126     """Expand indexes from the source into target names for easier processing of the data (arcs or lines)."""
 127     combined_data = {}  # type: t.Dict[str, t.Dict[t.Any, t.Set[str]]]
 128 
 129     for covered_path, covered_points in source_data.items():
 130         combined_points = combined_data.setdefault(covered_path, {})
 131 
 132         for covered_point, covered_target_indexes in covered_points.items():
 133             combined_point = combined_points.setdefault(format_func(covered_point), set())
 134 
 135             for covered_target_index in covered_target_indexes:
 136                 combined_point.add(source_index[covered_target_index])
 137 
 138     return combined_data

 141 def generate_indexes(target_indexes, data):  # type: (TargetIndexes, NamedPoints) -> IndexedPoints
 142     """Return an indexed version of the given data (arcs or points)."""
 143     results = {}  # type: IndexedPoints
 144 
 145     for path, points in data.items():
 146         result_points = results[path] = {}
 147 
 148         for point, target_names in points.items():
 149             result_point = result_points[point] = set()
 150 
 151             for target_name in target_names:
 152                 result_point.add(get_target_index(target_name, target_indexes))
 153 
 154     return results
