  56 def command_coverage_analyze_targets_generate(args):  # type: (CoverageAnalyzeTargetsGenerateConfig) -> None
  57     """Analyze code coverage data to determine which integration test targets provide coverage for each arc or line."""
  58     root = data_context().content.root
  59     target_indexes = {}
  60     arcs = dict((os.path.relpath(path, root), data) for path, data in analyze_python_coverage(args, args.input_dir, target_indexes).items())
  61     lines = dict((os.path.relpath(path, root), data) for path, data in analyze_powershell_coverage(args, args.input_dir, target_indexes).items())
  62     report = make_report(target_indexes, arcs, lines)
  63     write_report(args, report, args.output_file)

  66 def analyze_python_coverage(
  67         args,  # type: CoverageAnalyzeTargetsGenerateConfig
  68         path,  # type: str
  69         target_indexes,  # type: TargetIndexes
  70 ):  # type: (...) -> Arcs
  71     """Analyze Python code coverage."""
  72     results = {}  # type: Arcs
  73     collection_search_re, collection_sub_re = get_collection_path_regexes()
  74     modules = get_python_modules()
  75     python_files = get_python_coverage_files(path)
  76     coverage = initialize_coverage(args)
  77 
  78     for python_file in python_files:
  79         if not is_integration_coverage_file(python_file):
  80             continue
  81 
  82         target_name = get_target_name(python_file)
  83         target_index = get_target_index(target_name, target_indexes)
  84 
  85         for filename, covered_arcs in enumerate_python_arcs(python_file, coverage, modules, collection_search_re, collection_sub_re):
  86             arcs = results.setdefault(filename, {})
  87 
  88             for covered_arc in covered_arcs:
  89                 arc = arcs.setdefault(covered_arc, set())
  90                 arc.add(target_index)
  91 
  92     prune_invalid_filenames(args, results, collection_search_re=collection_search_re)
  93 
  94     return results

  97 def analyze_powershell_coverage(
  98         args,  # type: CoverageAnalyzeTargetsGenerateConfig
  99         path,  # type: str
 100         target_indexes,  # type: TargetIndexes
 101 ):  # type: (...) -> Lines
 102     """Analyze PowerShell code coverage"""

 105     powershell_files = get_powershell_coverage_files(path)

 107     for powershell_file in powershell_files:

 113 
 114         for filename, hits in enumerate_powershell_lines(powershell_file, collection_search_re, collection_sub_re):
 115             lines = results.setdefault(filename, {})

 117             for covered_line in hits:

 119                 line.add(target_index)
 120 
 121     prune_invalid_filenames(args, results)
 122 
 123     return results

 126 def prune_invalid_filenames(
 127         args,  # type: CoverageAnalyzeTargetsGenerateConfig
 128         results,  # type: t.Dict[str, t.Any]
 129         collection_search_re=None,  # type: t.Optional[str]
 130 ):  # type: (...) -> None
 131     """Remove invalid filenames from the given result set."""
 132     path_checker = PathChecker(args, collection_search_re)
 133 
 134     for path in list(results.keys()):
 135         if not path_checker.check_path(path):
 136             del results[path]

 139 def get_target_name(path):  # type: (str) -> str
 140     """Extract the test target name from the given coverage path."""
 141     return to_text(os.path.basename(path).split('=')[1])

 144 def is_integration_coverage_file(path):  # type: (str) -> bool
 145     """Returns True if the coverage file came from integration tests, otherwise False."""
 146     return os.path.basename(path).split('=')[0] in ('integration', 'windows-integration', 'network-integration')
