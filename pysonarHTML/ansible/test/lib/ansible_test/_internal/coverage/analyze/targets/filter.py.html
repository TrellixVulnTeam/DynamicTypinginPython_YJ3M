  39     """Filter target names in an aggregated coverage file."""

  42     filtered_path_arcs = expand_indexes(covered_path_arcs, covered_targets, lambda v: v)

  44 
  45     include_targets = set(args.include_targets) if args.include_targets else None
  46     exclude_targets = set(args.exclude_targets) if args.exclude_targets else None

  49     exclude_path = re.compile(args.exclude_path) if args.exclude_path else None
  50 
  51     def path_filter_func(path):
  52         if include_path and not re.search(include_path, path):
  53             return False
  54 
  55         if exclude_path and re.search(exclude_path, path):
  56             return False
  57 
  58         return True
  59 
  60     def target_filter_func(targets):
  61         if include_targets:
  62             targets &= include_targets
  63 
  64         if exclude_targets:
  65             targets -= exclude_targets
  66 
  67         return targets
  68 
  69     filtered_path_arcs = filter_data(filtered_path_arcs, path_filter_func, target_filter_func)
  70     filtered_path_lines = filter_data(filtered_path_lines, path_filter_func, target_filter_func)
