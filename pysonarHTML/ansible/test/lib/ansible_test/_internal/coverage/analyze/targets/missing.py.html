  41 def command_coverage_analyze_targets_missing(args):  # type: (CoverageAnalyzeTargetsMissingConfig) -> None
  42     """Identify aggregated coverage in one file missing from another."""
  43     from_targets, from_path_arcs, from_path_lines = read_report(args.from_file)
  44     to_targets, to_path_arcs, to_path_lines = read_report(args.to_file)
  45     target_indexes = {}
  46 
  47     if args.only_gaps:

  50     else:

  53

  58 def find_gaps(
  59         from_data,  # type: IndexedPoints
  60         from_index,  # type: t.List[str]
  61         to_data,  # type: IndexedPoints
  62         target_indexes,  # type: TargetIndexes
  63         only_exists,  # type: bool
  64 ):  # type: (...) -> IndexedPoints
  65     """Find gaps in coverage between the from and to data sets."""
  66     target_data = {}
  67 
  68     for from_path, from_points in from_data.items():
  69         if only_exists and not os.path.isfile(to_bytes(from_path)):
  70             continue
  71 
  72         to_points = to_data.get(from_path, {})
  73 
  74         gaps = set(from_points.keys()) - set(to_points.keys())
  75 
  76         if gaps:
  77             gap_points = dict((key, value) for key, value in from_points.items() if key in gaps)
  78             target_data[from_path] = dict((gap, set(get_target_index(from_index[i], target_indexes) for i in indexes)) for gap, indexes in gap_points.items())
  79 
  80     return target_data

  83 def find_missing(
  84         from_data,  # type: IndexedPoints
  85         from_index,  # type: t.List[str]
  86         to_data,  # type: IndexedPoints
  87         to_index,  # type: t.List[str]
  88         target_indexes,  # type: TargetIndexes
  89         only_exists,  # type: bool
  90 ):  # type: (...) -> IndexedPoints
  91     """Find coverage in from_data not present in to_data (arcs or lines)."""
  92     target_data = {}
  93 
  94     for from_path, from_points in from_data.items():
  95         if only_exists and not os.path.isfile(to_bytes(from_path)):
  96             continue
  97 
  98         to_points = to_data.get(from_path, {})
  99 
 100         for from_point, from_target_indexes in from_points.items():
 101             to_target_indexes = to_points.get(from_point, set())
 102 
 103             remaining_targets = set(from_index[i] for i in from_target_indexes) - set(to_index[i] for i in to_target_indexes)
 104 
 105             if remaining_targets:
 106                 target_index = target_data.setdefault(from_path, {}).setdefault(from_point, set())
 107                 target_index.update(get_target_index(name, target_indexes) for name in remaining_targets)
 108 
 109     return target_data
