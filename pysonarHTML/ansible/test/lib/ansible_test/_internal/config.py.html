  37 class ParsedRemote:
  38     """A parsed version of a "remote" string."""
  39     def __init__(self, arch, platform, version):  # type: (t.Optional[str], str, str) -> None
  40         self.arch = arch
  41         self.platform = platform
  42         self.version = version
  43 
  44     @staticmethod
  45     def parse(value):  # type: (str) -> t.Optional['ParsedRemote']
  46         """Return a ParsedRemote from the given value or None if the syntax is invalid."""
  47         parts = value.split('/')
  48 
  49         if len(parts) == 2:
  50             arch = None
  51             platform, version = parts
  52         elif len(parts) == 3:
  53             arch, platform, version = parts
  54         else:
  55             return None
  56 
  57         return ParsedRemote(arch, platform, version)

  60 class EnvironmentConfig(CommonConfig):
  61     """Configuration common to all commands which execute in an environment."""
  62     def __init__(self, args, command):
  63         """
  64         :type args: any
  65         :type command: str
  66         """
  67         super(EnvironmentConfig, self).__init__(args, command)
  68 
  69         self.local = args.local is True
  70         self.venv = args.venv
  71         self.venv_system_site_packages = args.venv_system_site_packages
  72 
  73         self.python = args.python if 'python' in args else None  # type: str
  74 
  75         self.docker = docker_qualify_image(args.docker)  # type: str
  76         self.docker_raw = args.docker  # type: str
  77         self.remote = args.remote  # type: str
  78 
  79         if self.remote:
  80             self.parsed_remote = ParsedRemote.parse(self.remote)
  81 
  82             if not self.parsed_remote or not self.parsed_remote.platform or not self.parsed_remote.version:
  83                 raise ApplicationError('Unrecognized remote "%s" syntax. Use "platform/version" or "arch/platform/version".' % self.remote)
  84         else:
  85             self.parsed_remote = None
  86 
  87         self.docker_privileged = args.docker_privileged if 'docker_privileged' in args else False  # type: bool
  88         self.docker_pull = args.docker_pull if 'docker_pull' in args else False  # type: bool
  89         self.docker_keep_git = args.docker_keep_git if 'docker_keep_git' in args else False  # type: bool
  90         self.docker_seccomp = args.docker_seccomp if 'docker_seccomp' in args else None  # type: str
  91         self.docker_memory = args.docker_memory if 'docker_memory' in args else None
  92         self.docker_terminate = args.docker_terminate if 'docker_terminate' in args else None  # type: str
  93 
  94         if self.docker_seccomp is None:
  95             self.docker_seccomp = get_docker_completion().get(self.docker_raw, {}).get('seccomp', 'default')
  96 
  97         self.remote_stage = args.remote_stage  # type: str
  98         self.remote_provider = args.remote_provider  # type: str
  99         self.remote_aws_region = args.remote_aws_region  # type: str
 100         self.remote_terminate = args.remote_terminate  # type: str
 101 
 102         if self.remote_provider == 'default':
 103             self.remote_provider = None
 104 
 105         self.requirements = args.requirements  # type: bool
 106 
 107         if self.python == 'default':
 108             self.python = None
 109 
 110         actual_major_minor = '.'.join(str(i) for i in sys.version_info[:2])
 111 
 112         self.python_version = self.python or actual_major_minor
 113         self.python_interpreter = args.python_interpreter
 114 
 115         self.pip_check = args.pip_check
 116 
 117         self.delegate = self.docker or self.remote or self.venv
 118         self.delegate_args = []  # type: t.List[str]
 119 
 120         if self.delegate:
 121             self.requirements = True
 122 
 123         self.inject_httptester = args.inject_httptester if 'inject_httptester' in args else False  # type: bool
 124         self.httptester = docker_qualify_image(args.httptester if 'httptester' in args else '')  # type: str
 125 
 126         if self.get_delegated_completion().get('httptester', 'enabled') == 'disabled':
 127             self.httptester = False
 128 
 129         if self.get_delegated_completion().get('pip-check', 'enabled') == 'disabled':
 130             self.pip_check = False
 131 
 132         if args.check_python and args.check_python != actual_major_minor:
 133             raise ApplicationError('Running under Python %s instead of Python %s as expected.' % (actual_major_minor, args.check_python))
 134 
 135         if self.docker_keep_git:
 136             def git_callback(files):  # type: (t.List[t.Tuple[str, str]]) -> None
 137                 """Add files from the content root .git directory to the payload file list."""
 138                 for dirpath, _dirnames, filenames in os.walk(os.path.join(data_context().content.root, '.git')):
 139                     paths = [os.path.join(dirpath, filename) for filename in filenames]
 140                     files.extend((path, os.path.relpath(path, data_context().content.root)) for path in paths)
 141 
 142             data_context().register_payload_callback(git_callback)
 143 
 144     @property
 145     def python_executable(self):
 146         """
 147         :rtype: str
 148         """
 149         return find_python(self.python_version)
 150 
 151     @property
 152     def pip_command(self):
 153         """
 154         :rtype: list[str]
 155         """
 156         return generate_pip_command(self.python_executable)
 157 
 158     def get_delegated_completion(self):
 159         """Returns a dictionary of settings specific to the selected delegation system, if any. Otherwise returns an empty dictionary.
 160         :rtype: dict[str, str]
 161         """
 162         if self.docker:
 163             return get_docker_completion().get(self.docker_raw, {})
 164 
 165         if self.remote:
 166             return get_remote_completion().get(self.remote, {})
 167 
 168         return {}

 284     def get_ansible_config(self):  # type: () -> str
 285         """Return the path to the Ansible config for the given config."""
 286         ansible_config_relative_path = os.path.join(data_context().content.integration_path, '%s.cfg' % self.command)
 287         ansible_config_path = os.path.join(data_context().content.root, ansible_config_relative_path)
