  63 class ImportTest(SanityMultipleVersion):
  64     """Sanity test for proper import exception handling."""
  65     def filter_targets(self, targets):  # type: (t.List[TestTarget]) -> t.List[TestTarget]
  66         """Return the given list of test targets, filtered to include only those relevant for the test."""
  67         return [target for target in targets if os.path.splitext(target.path)[1] == '.py' and
  68                 (is_subdir(target.path, data_context().content.module_path) or is_subdir(target.path, data_context().content.module_utils_path))]
  69 
  70     def test(self, args, targets, python_version):
  71         """
  72         :type args: SanityConfig
  73         :type targets: SanityTargets
  74         :type python_version: str
  75         :rtype: TestResult
  76         """
  77         capture_pip = args.verbosity < 2
  78 
  79         python = find_python(python_version)
  80 
  81         if python_version.startswith('2.') and args.requirements:
  82             # hack to make sure that virtualenv is available under Python 2.x
  83             # on Python 3.x we can use the built-in venv
  84             pip = generate_pip_command(python)
  85             run_command(args, generate_pip_install(pip, '', packages=['virtualenv']), capture=capture_pip)
  86 
  87         settings = self.load_processor(args, python_version)
  88 
  89         paths = [target.path for target in targets.include]
  90 
  91         env = ansible_environment(args, color=False)
  92 
  93         temp_root = os.path.join(ResultType.TMP.path, 'sanity', 'import')
  94 
  95         # create a clean virtual environment to minimize the available imports beyond the python standard library
  96         virtual_environment_path = os.path.join(temp_root, 'minimal-py%s' % python_version.replace('.', ''))
  97         virtual_environment_bin = os.path.join(virtual_environment_path, 'bin')
  98 
  99         remove_tree(virtual_environment_path)
 100 
 101         if not create_virtual_environment(args, python_version, virtual_environment_path):
 102             display.warning("Skipping sanity test '%s' on Python %s due to missing virtual environment support." % (self.name, python_version))
 103             return SanitySkipped(self.name, python_version)
 104 
 105         # add the importer to our virtual environment so it can be accessed through the coverage injector
 106         importer_path = os.path.join(virtual_environment_bin, 'importer.py')
 107         yaml_to_json_path = os.path.join(virtual_environment_bin, 'yaml_to_json.py')
 108         if not args.explain:
 109             os.symlink(os.path.abspath(os.path.join(SANITY_ROOT, 'import', 'importer.py')), importer_path)
 110             os.symlink(os.path.abspath(os.path.join(SANITY_ROOT, 'import', 'yaml_to_json.py')), yaml_to_json_path)
 111 
 112         # activate the virtual environment
 113         env['PATH'] = '%s:%s' % (virtual_environment_bin, env['PATH'])
 114 
 115         env.update(
 116             SANITY_TEMP_PATH=ResultType.TMP.path,
 117         )
 118 
 119         if data_context().content.collection:
 120             env.update(
 121                 SANITY_COLLECTION_FULL_NAME=data_context().content.collection.full_name,
 122                 SANITY_EXTERNAL_PYTHON=python,
 123             )
 124 
 125         virtualenv_python = os.path.join(virtual_environment_bin, 'python')
 126         virtualenv_pip = generate_pip_command(virtualenv_python)
 127 
 128         # make sure coverage is available in the virtual environment if needed
 129         if args.coverage:
 130             run_command(args, generate_pip_install(virtualenv_pip, '', packages=['setuptools']), env=env, capture=capture_pip)
 131             run_command(args, generate_pip_install(virtualenv_pip, '', packages=['coverage']), env=env, capture=capture_pip)
 132 
 133         try:
 134             # In some environments pkg_resources is installed as a separate pip package which needs to be removed.
 135             # For example, using Python 3.8 on Ubuntu 18.04 a virtualenv is created with only pip and setuptools.
 136             # However, a venv is created with an additional pkg-resources package which is independent of setuptools.
 137             # Making sure pkg-resources is removed preserves the import test consistency between venv and virtualenv.
 138             # Additionally, in the above example, the pyparsing package vendored with pkg-resources is out-of-date and generates deprecation warnings.
 139             # Thus it is important to remove pkg-resources to prevent system installed packages from generating deprecation warnings.
 140             run_command(args, virtualenv_pip + ['uninstall', '--disable-pip-version-check', '-y', 'pkg-resources'], env=env, capture=capture_pip)
 141         except SubprocessError:
 142             pass
 143 
 144         run_command(args, virtualenv_pip + ['uninstall', '--disable-pip-version-check', '-y', 'setuptools'], env=env, capture=capture_pip)
 145         run_command(args, virtualenv_pip + ['uninstall', '--disable-pip-version-check', '-y', 'pip'], env=env, capture=capture_pip)
 146 
 147         cmd = ['importer.py']
 148 
 149         data = '\n'.join(paths)
 150 
 151         display.info(data, verbosity=4)
 152 
 153         results = []
 154 
 155         try:
 156             with coverage_context(args):
 157                 stdout, stderr = intercept_command(args, cmd, self.name, env, capture=True, data=data, python_version=python_version,
 158                                                    virtualenv=virtualenv_python)
 159 
 160             if stdout or stderr:
 161                 raise SubprocessError(cmd, stdout=stdout, stderr=stderr)
 162         except SubprocessError as ex:
 163             if ex.status != 10 or ex.stderr or not ex.stdout:
 164                 raise
 165 
 166             pattern = r'^(?P<path>[^:]*):(?P<line>[0-9]+):(?P<column>[0-9]+): (?P<message>.*)$'
 167 
 168             results = parse_to_list_of_dict(pattern, ex.stdout)
 169 
 170             relative_temp_root = os.path.relpath(temp_root, data_context().content.root) + os.path.sep
 171 
 172             results = [SanityMessage(
 173                 message=r['message'],
 174                 path=os.path.relpath(r['path'], relative_temp_root) if r['path'].startswith(relative_temp_root) else r['path'],
 175                 line=int(r['line']),
 176                 column=int(r['column']),
 177             ) for r in results]
 178 
 179         results = settings.process_errors(results, paths)
 180 
 181         if results:
 182             return SanityFailure(self.name, messages=results, python_version=python_version)
 183 
 184         return SanitySuccess(self.name, python_version=python_version)
