  48 class AnsibleDocTest(SanitySingleVersion):
  49     """Sanity test for ansible-doc."""
  50     def filter_targets(self, targets):  # type: (t.List[TestTarget]) -> t.List[TestTarget]
  51         """Return the given list of test targets, filtered to include only those relevant for the test."""
  52         # This should use documentable plugins from constants instead
  53         unsupported_plugin_types = set([
  54             # not supported by ansible-doc
  55             'action',
  56             'doc_fragments',
  57             'filter',
  58             'module_utils',
  59             'terminal',
  60             'test',
  61         ])
  62 
  63         plugin_paths = [plugin_path for plugin_type, plugin_path in data_context().content.plugin_paths.items() if plugin_type not in unsupported_plugin_types]
  64 
  65         return [target for target in targets
  66                 if os.path.splitext(target.path)[1] == '.py'
  67                 and os.path.basename(target.path) != '__init__.py'
  68                 and any(is_subdir(target.path, path) for path in plugin_paths)
  69                 ]
  70 
  71     def test(self, args, targets, python_version):
  72         """
  73         :type args: SanityConfig
  74         :type targets: SanityTargets
  75         :type python_version: str
  76         :rtype: TestResult
  77         """
  78         settings = self.load_processor(args)
  79 
  80         paths = [target.path for target in targets.include]
  81 
  82         doc_targets = collections.defaultdict(list)
  83         target_paths = collections.defaultdict(dict)
  84 
  85         remap_types = dict(
  86             modules='module',
  87         )
  88 
  89         for plugin_type, plugin_path in data_context().content.plugin_paths.items():
  90             plugin_type = remap_types.get(plugin_type, plugin_type)
  91 
  92             for plugin_file_path in [target.name for target in targets.include if is_subdir(target.path, plugin_path)]:
  93                 plugin_name = os.path.splitext(os.path.basename(plugin_file_path))[0]
  94 
  95                 if plugin_name.startswith('_'):
  96                     plugin_name = plugin_name[1:]
  97 
  98                 doc_targets[plugin_type].append(data_context().content.prefix + plugin_name)
  99                 target_paths[plugin_type][data_context().content.prefix + plugin_name] = plugin_file_path
 100 
 101         env = ansible_environment(args, color=False)
 102         error_messages = []
 103 
 104         for doc_type in sorted(doc_targets):
 105             for format_option in [None, '--json']:
 106                 cmd = ['ansible-doc', '-t', doc_type]
 107                 if format_option is not None:
 108                     cmd.append(format_option)
 109                 cmd.extend(sorted(doc_targets[doc_type]))
 110 
 111                 try:
 112                     with coverage_context(args):
 113                         stdout, stderr = intercept_command(args, cmd, target_name='ansible-doc', env=env, capture=True, python_version=python_version)
 114 
 115                     status = 0
 116                 except SubprocessError as ex:
 117                     stdout = ex.stdout
 118                     stderr = ex.stderr
 119                     status = ex.status
 120 
 121                 if status:
 122                     summary = u'%s' % SubprocessError(cmd=cmd, status=status, stderr=stderr)
 123                     return SanityFailure(self.name, summary=summary)
 124 
 125                 if stdout:
 126                     display.info(stdout.strip(), verbosity=3)
 127 
 128                 if stderr:
 129                     # ignore removed module/plugin warnings
 130                     stderr = re.sub(r'\[WARNING\]: [^ ]+ [^ ]+ has been removed\n', '', stderr).strip()
 131 
 132                 if stderr:
 133                     summary = u'Output on stderr from ansible-doc is considered an error.\n\n%s' % SubprocessError(cmd, stderr=stderr)
 134                     return SanityFailure(self.name, summary=summary)
 135 
 136         if args.explain:
 137             return SanitySuccess(self.name)
 138 
 139         error_messages = settings.process_errors(error_messages, paths)
 140 
 141         if error_messages:
 142             return SanityFailure(self.name, messages=error_messages)
 143 
 144         return SanitySuccess(self.name)
