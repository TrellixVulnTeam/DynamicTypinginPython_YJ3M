  71     def test(self, args, targets, python_version):
  72         """
  73         :type args: SanityConfig
  74         :type targets: SanityTargets
  75         :type python_version: str
  76         :rtype: TestResult
  77         """
  78         plugin_dir = os.path.join(SANITY_ROOT, 'pylint', 'plugins')
  79         plugin_names = sorted(p[0] for p in [
  80             os.path.splitext(p) for p in os.listdir(plugin_dir)] if p[1] == '.py' and p[0] != '__init__')
  81 
  82         settings = self.load_processor(args)
  83 
  84         paths = [target.path for target in targets.include]
  85 
  86         module_paths = [os.path.relpath(p, data_context().content.module_path).split(os.path.sep) for p in
  87                         paths if is_subdir(p, data_context().content.module_path)]
  88         module_dirs = sorted(set([p[0] for p in module_paths if len(p) > 1]))
  89 
  90         large_module_group_threshold = 500
  91         large_module_groups = [key for key, value in
  92                                itertools.groupby(module_paths, lambda p: p[0] if len(p) > 1 else '') if len(list(value)) > large_module_group_threshold]
  93 
  94         large_module_group_paths = [os.path.relpath(p, data_context().content.module_path).split(os.path.sep) for p in paths
  95                                     if any(is_subdir(p, os.path.join(data_context().content.module_path, g)) for g in large_module_groups)]
  96         large_module_group_dirs = sorted(set([os.path.sep.join(p[:2]) for p in large_module_group_paths if len(p) > 2]))
  97 
  98         contexts = []
  99         remaining_paths = set(paths)
 100 
 101         def add_context(available_paths, context_name, context_filter):
 102             """
 103             :type available_paths: set[str]
 104             :type context_name: str
 105             :type context_filter: (str) -> bool
 106             """
 107             filtered_paths = set(p for p in available_paths if context_filter(p))
 108             contexts.append((context_name, sorted(filtered_paths)))
 109             available_paths -= filtered_paths
 110 
 111         def filter_path(path_filter=None):
 112             """
 113             :type path_filter: str
 114             :rtype: (str) -> bool
 115             """
 116             def context_filter(path_to_filter):
 117                 """
 118                 :type path_to_filter: str
 119                 :rtype: bool
 120                 """
 121                 return is_subdir(path_to_filter, path_filter)
 122 
 123             return context_filter
 124 
 125         for large_module_dir in large_module_group_dirs:
 126             add_context(remaining_paths, 'modules/%s' % large_module_dir, filter_path(os.path.join(data_context().content.module_path, large_module_dir)))
 127 
 128         for module_dir in module_dirs:
 129             add_context(remaining_paths, 'modules/%s' % module_dir, filter_path(os.path.join(data_context().content.module_path, module_dir)))
 130 
 131         add_context(remaining_paths, 'modules', filter_path(data_context().content.module_path))
 132         add_context(remaining_paths, 'module_utils', filter_path(data_context().content.module_utils_path))
 133 
 134         add_context(remaining_paths, 'units', filter_path(data_context().content.unit_path))
 135 
 136         if data_context().content.collection:
 137             add_context(remaining_paths, 'collection', lambda p: True)
 138         else:
 139             add_context(remaining_paths, 'validate-modules', filter_path('test/lib/ansible_test/_data/sanity/validate-modules/'))
 140             add_context(remaining_paths, 'validate-modules-unit', filter_path('test/lib/ansible_test/tests/validate-modules-unit/'))
 141             add_context(remaining_paths, 'sanity', filter_path('test/lib/ansible_test/_data/sanity/'))
 142             add_context(remaining_paths, 'ansible-test', filter_path('test/lib/'))
 143             add_context(remaining_paths, 'test', filter_path('test/'))
 144             add_context(remaining_paths, 'hacking', filter_path('hacking/'))
 145             add_context(remaining_paths, 'ansible', lambda p: True)
 146 
 147         messages = []
 148         context_times = []
 149 
 150         python = find_python(python_version)
 151 
 152         collection_detail = None
 153 
 154         if data_context().content.collection:
 155             try:
 156                 collection_detail = get_collection_detail(args, python)
 157 
 158                 if not collection_detail.version:
 159                     display.warning('Skipping pylint collection version checks since no collection version was found.')
 160             except CollectionDetailError as ex:
 161                 display.warning('Skipping pylint collection version checks since collection detail loading failed: %s' % ex.reason)
 162 
 163         test_start = datetime.datetime.utcnow()
 164 
 165         for context, context_paths in sorted(contexts):
 166             if not context_paths:
 167                 continue
 168 
 169             context_start = datetime.datetime.utcnow()
 170             messages += self.pylint(args, context, context_paths, plugin_dir, plugin_names, python, collection_detail)
 171             context_end = datetime.datetime.utcnow()
 172 
 173             context_times.append('%s: %d (%s)' % (context, len(context_paths), context_end - context_start))
 174 
 175         test_end = datetime.datetime.utcnow()
 176 
 177         for context_time in context_times:
 178             display.info(context_time, verbosity=4)
 179 
 180         display.info('total: %d (%s)' % (len(paths), test_end - test_start), verbosity=4)
 181 
 182         errors = [SanityMessage(
 183             message=m['message'].replace('\n', ' '),
 184             path=m['path'],
 185             line=int(m['line']),
 186             column=int(m['column']),
 187             level=m['type'],
 188             code=m['symbol'],
 189         ) for m in messages]
 190 
 191         if args.explain:
 192             return SanitySuccess(self.name)
 193 
 194         errors = settings.process_errors(errors, paths)
 195 
 196         if errors:
 197             return SanityFailure(self.name, messages=errors)
 198 
 199         return SanitySuccess(self.name)

 201     @staticmethod
 202     def pylint(
 203             args,  # type: SanityConfig
 204             context,  # type: str
 205             paths,  # type: t.List[str]
 206             plugin_dir,  # type: str
 207             plugin_names,  # type: t.List[str]
 208             python,  # type: str
 209             collection_detail,  # type: CollectionDetail
 210     ):  # type: (...) -> t.List[t.Dict[str, str]]
 211         """Run pylint using the config specified by the context on the specified paths."""
 212         rcfile = os.path.join(SANITY_ROOT, 'pylint', 'config', context.split('/')[0] + '.cfg')
 213 
 214         if not os.path.exists(rcfile):
 215             if data_context().content.collection:
 216                 rcfile = os.path.join(SANITY_ROOT, 'pylint', 'config', 'collection.cfg')
 217             else:
 218                 rcfile = os.path.join(SANITY_ROOT, 'pylint', 'config', 'default.cfg')
 219 
 220         parser = ConfigParser()
 221         parser.read(rcfile)
 222 
 223         if parser.has_section('ansible-test'):
 224             config = dict(parser.items('ansible-test'))
 225         else:
 226             config = dict()
 227 
 228         disable_plugins = set(i.strip() for i in config.get('disable-plugins', '').split(',') if i)
 229         load_plugins = set(plugin_names + ['pylint.extensions.mccabe']) - disable_plugins
 230 
 231         cmd = [
 232             python,
 233             '-m', 'pylint',
 234             '--jobs', '0',
 235             '--reports', 'n',
 236             '--max-line-length', '160',
 237             '--max-complexity', '20',
 238             '--rcfile', rcfile,
 239             '--output-format', 'json',
 240             '--load-plugins', ','.join(load_plugins),
 241         ] + paths
 242 
 243         if data_context().content.collection:
 244             cmd.extend(['--collection-name', data_context().content.collection.full_name])
 245 
 246             if collection_detail and collection_detail.version:
 247                 cmd.extend(['--collection-version', collection_detail.version])
 248 
 249         append_python_path = [plugin_dir]
 250 
 251         if data_context().content.collection:
 252             append_python_path.append(data_context().content.collection.root)
 253 
 254         env = ansible_environment(args)
 255         env['PYTHONPATH'] += os.path.pathsep + os.path.pathsep.join(append_python_path)
 256 
 257         # expose plugin paths for use in custom plugins
 258         env.update(dict(('ANSIBLE_TEST_%s_PATH' % k.upper(), os.path.abspath(v) + os.path.sep) for k, v in data_context().content.plugin_paths.items()))
 259 
 260         if paths:
 261             display.info('Checking %d file(s) in context "%s" with config: %s' % (len(paths), context, rcfile), verbosity=1)
 262 
 263             try:
 264                 stdout, stderr = run_command(args, cmd, env=env, capture=True)
 265                 status = 0
 266             except SubprocessError as ex:
 267                 stdout = ex.stdout
 268                 stderr = ex.stderr
 269                 status = ex.status
 270 
 271             if stderr or status >= 32:
 272                 raise SubprocessError(cmd=cmd, status=status, stderr=stderr, stdout=stdout)
 273         else:
 274             stdout = None
 275 
 276         if not args.explain and stdout:
 277             messages = json.loads(stdout)
 278         else:
 279             messages = []
 280 
 281         return messages
