 109     @property
 110     def shippable_yml_lines(self):
 111         """
 112         :rtype: list[str]
 113         """
 114         if not self._shippable_yml_lines:
 115             self._shippable_yml_lines = read_text_file(self.SHIPPABLE_YML).splitlines()
 116 
 117         return self._shippable_yml_lines

 119     @property
 120     def shippable_test_groups(self):
 121         """
 122         :rtype: dict[str, set[int]]
 123         """
 124         if not self._shippable_test_groups:
 125             matches = [re.search(r'^[ #]+- env: T=(?P<group>[^/]+)/(?P<params>.+)/(?P<number>[1-9][0-9]?)$', line) for line in self.shippable_yml_lines]
 126             entries = [(match.group('group'), int(match.group('number'))) for match in matches if match]
 127 
 128             for key, value in entries:
 129                 if key not in self._shippable_test_groups:
 130                     self._shippable_test_groups[key] = set()
 131 
 132                 self._shippable_test_groups[key].add(value)
 133 
 134         return self._shippable_test_groups

 136     def format_shippable_group_alias(self, name, fallback=''):
 137         """
 138         :type name: str
 139         :type fallback: str
 140         :rtype: str
 141         """
 142         group_numbers = self.shippable_test_groups.get(name, None)
 143 
 144         if group_numbers:
 145             if min(group_numbers) != 1:
 146                 display.warning('Min test group "%s" in shippable.yml is %d instead of 1.' % (name, min(group_numbers)), unique=True)
 147 
 148             if max(group_numbers) != len(group_numbers):
 149                 display.warning('Max test group "%s" in shippable.yml is %d instead of %d.' % (name, max(group_numbers), len(group_numbers)), unique=True)
 150 
 151             if max(group_numbers) > 9:
 152                 alias = 'shippable/%s/group(%s)/' % (name, '|'.join(str(i) for i in range(min(group_numbers), max(group_numbers) + 1)))
 153             elif len(group_numbers) > 1:
 154                 alias = 'shippable/%s/group[%d-%d]/' % (name, min(group_numbers), max(group_numbers))
 155             else:
 156                 alias = 'shippable/%s/group%d/' % (name, min(group_numbers))
 157         elif fallback:
 158             alias = 'shippable/%s/group%d/' % (fallback, 1)
 159         else:
 160             raise Exception('cannot find test group "%s" in shippable.yml' % name)
 161 
 162         return alias

 164     def test(self, args, targets):
 165         """
 166         :type args: SanityConfig
 167         :type targets: SanityTargets
 168         :rtype: TestResult
 169         """
 170         if args.explain:
 171             return SanitySuccess(self.name)
 172 
 173         if not os.path.isfile(self.SHIPPABLE_YML):
 174             return SanityFailure(self.name, messages=[SanityMessage(
 175                 message='file missing',
 176                 path=self.SHIPPABLE_YML,
 177             )])
 178 
 179         results = dict(
 180             comments=[],
 181             labels={},
 182         )
 183 
 184         self.check_changes(args, results)
 185 
 186         write_json_test_results(ResultType.BOT, 'data-sanity-ci.json', results)
 187 
 188         messages = []
 189 
 190         messages += self.check_posix_targets(args)
 191         messages += self.check_windows_targets()
 192 
 193         if messages:
 194             return SanityFailure(self.name, messages=messages)
 195 
 196         return SanitySuccess(self.name)

 198     def check_posix_targets(self, args):
 199         """
 200         :type args: SanityConfig
 201         :rtype: list[SanityMessage]
 202         """
 203         posix_targets = tuple(walk_posix_integration_targets())
 204 
 205         clouds = get_cloud_platforms(args, posix_targets)
 206         cloud_targets = ['cloud/%s/' % cloud for cloud in clouds]
 207 
 208         all_cloud_targets = tuple(filter_targets(posix_targets, ['cloud/'], include=True, directories=False, errors=False))
 209         invalid_cloud_targets = tuple(filter_targets(all_cloud_targets, cloud_targets, include=False, directories=False, errors=False))
 210 
 211         messages = []
 212 
 213         for target in invalid_cloud_targets:
 214             for alias in target.aliases:
 215                 if alias.startswith('cloud/') and alias != 'cloud/':
 216                     if any(alias.startswith(cloud_target) for cloud_target in cloud_targets):
 217                         continue
 218 
 219                     messages.append(SanityMessage('invalid alias `%s`' % alias, '%s/aliases' % target.path))
 220 
 221         messages += self.check_ci_group(
 222             targets=tuple(filter_targets(posix_targets, ['cloud/', 'shippable/generic/'], include=False,
 223                                          directories=False, errors=False)),
 224             find=self.format_shippable_group_alias('linux').replace('linux', 'posix'),
 225             find_incidental=['shippable/posix/incidental/'],
 226         )
 227 
 228         messages += self.check_ci_group(
 229             targets=tuple(filter_targets(posix_targets, ['shippable/generic/'], include=True, directories=False,
 230                                          errors=False)),
 231             find=self.format_shippable_group_alias('generic'),
 232         )
 233 
 234         for cloud in clouds:
 235             messages += self.check_ci_group(
 236                 targets=tuple(filter_targets(posix_targets, ['cloud/%s/' % cloud], include=True, directories=False, errors=False)),
 237                 find=self.format_shippable_group_alias(cloud, 'cloud'),
 238                 find_incidental=['shippable/%s/incidental/' % cloud, 'shippable/cloud/incidental/'],
 239             )
 240 
 241         return messages

 259     def check_ci_group(self, targets, find, find_incidental=None):
 260         """
 261         :type targets: tuple[CompletionTarget]
 262         :type find: str
 263         :type find_incidental: list[str] | None
 264         :rtype: list[SanityMessage]
 265         """
 266         all_paths = set(target.path for target in targets)
 267         supported_paths = set(target.path for target in filter_targets(targets, [find], include=True, directories=False, errors=False))
 268         unsupported_paths = set(target.path for target in filter_targets(targets, [self.UNSUPPORTED], include=True, directories=False, errors=False))
 269 
 270         if find_incidental:
 271             incidental_paths = set(target.path for target in filter_targets(targets, find_incidental, include=True, directories=False, errors=False))
 272         else:
 273             incidental_paths = set()
 274 
 275         unassigned_paths = all_paths - supported_paths - unsupported_paths - incidental_paths
 276         conflicting_paths = supported_paths & unsupported_paths
 277 
 278         unassigned_message = 'missing alias `%s` or `%s`' % (find.strip('/'), self.UNSUPPORTED.strip('/'))
 279         conflicting_message = 'conflicting alias `%s` and `%s`' % (find.strip('/'), self.UNSUPPORTED.strip('/'))
 280 
 281         messages = []
 282 
 283         for path in unassigned_paths:
 284             messages.append(SanityMessage(unassigned_message, '%s/aliases' % path))
 285 
 286         for path in conflicting_paths:
 287             messages.append(SanityMessage(conflicting_message, '%s/aliases' % path))
 288 
 289         return messages

 291     def check_changes(self, args, results):
 292         """
 293         :type args: SanityConfig
 294         :type results: dict[str, any]
 295         """
 296         integration_targets = list(walk_integration_targets())
 297         module_targets = list(walk_module_targets())
 298 
 299         integration_targets_by_name = dict((target.name, target) for target in integration_targets)
 300         module_names_by_path = dict((target.path, target.module) for target in module_targets)
 301 
 302         disabled_targets = []
 303         unstable_targets = []
 304         unsupported_targets = []
 305 
 306         for command in [command for command in args.metadata.change_description.focused_command_targets if 'integration' in command]:
 307             for target in args.metadata.change_description.focused_command_targets[command]:
 308                 if self.DISABLED in integration_targets_by_name[target].aliases:
 309                     disabled_targets.append(target)
 310                 elif self.UNSTABLE in integration_targets_by_name[target].aliases:
 311                     unstable_targets.append(target)
 312                 elif self.UNSUPPORTED in integration_targets_by_name[target].aliases:
 313                     unsupported_targets.append(target)
 314 
 315         untested_modules = []
 316 
 317         for path in args.metadata.change_description.no_integration_paths:
 318             module = module_names_by_path.get(path)
 319 
 320             if module:
 321                 untested_modules.append(module)
 322 
 323         comments = [
 324             self.format_comment(self.TEMPLATE_DISABLED, disabled_targets),
 325             self.format_comment(self.TEMPLATE_UNSTABLE, unstable_targets),
 326             self.format_comment(self.TEMPLATE_UNSUPPORTED, unsupported_targets),
 327             self.format_comment(self.TEMPLATE_UNTESTED, untested_modules),
 328         ]
 329 
 330         comments = [comment for comment in comments if comment]
 331 
 332         labels = dict(
 333             needs_tests=bool(untested_modules),
 334             disabled_tests=bool(disabled_targets),
 335             unstable_tests=bool(unstable_targets),
 336             unsupported_tests=bool(unsupported_targets),
 337         )
 338 
 339         results['comments'] += comments
 340         results['labels'].update(labels)

 342     def format_comment(self, template, targets):
 343         """
 344         :type template: str
 345         :type targets: list[str]
 346         :rtype: str | None
 347         """
 348         if not targets:
 349             return None
 350 
 351         tests = '\n'.join('- %s' % target for target in targets)
 352 
 353         data = dict(
 354             explain_url=self.EXPLAIN_URL,
 355             tests=tests,
 356         )
 357 
 358         message = textwrap.dedent(template).strip().format(**data)
 359 
 360         return message
