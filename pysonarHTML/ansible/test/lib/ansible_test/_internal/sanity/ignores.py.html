  25 class IgnoresTest(SanityVersionNeutral):
  26     """Sanity test for sanity test ignore entries."""
  27     @property

  36 
  37     # noinspection PyUnusedLocal
  38     def test(self, args, targets):  # pylint: disable=locally-disabled, unused-argument
  39         """
  40         :type args: SanityConfig
  41         :type targets: SanityTargets
  42         :rtype: TestResult
  43         """
  44         sanity_ignore = SanityIgnoreParser.load(args)
  45 
  46         messages = []
  47 
  48         # parse errors
  49 
  50         messages.extend(SanityMessage(
  51             message=message,
  52             path=sanity_ignore.relative_path,
  53             line=line,
  54             column=column,
  55             confidence=calculate_confidence(sanity_ignore.path, line, args.metadata) if args.metadata.changes else None,
  56         ) for line, column, message in sanity_ignore.parse_errors)
  57 
  58         # file not found errors
  59 
  60         messages.extend(SanityMessage(
  61             message="%s '%s' does not exist" % ("Directory" if path.endswith(os.path.sep) else "File", path),
  62             path=sanity_ignore.relative_path,
  63             line=line,
  64             column=1,
  65             confidence=calculate_best_confidence(((sanity_ignore.path, line), (path, 0)), args.metadata) if args.metadata.changes else None,
  66         ) for line, path in sanity_ignore.file_not_found_errors)
  67 
  68         # conflicting ignores and skips
  69 
  70         for test_name, ignores in sanity_ignore.ignores.items():
  71             for ignore_path, ignore_entry in ignores.items():
  72                 skip_line_no = sanity_ignore.skips.get(test_name, {}).get(ignore_path)
  73 
  74                 if not skip_line_no:
  75                     continue
  76 
  77                 for ignore_line_no in ignore_entry.values():
  78                     messages.append(SanityMessage(
  79                         message="Ignoring '%s' is unnecessary due to skip entry on line %d" % (ignore_path, skip_line_no),
  80                         path=sanity_ignore.relative_path,
  81                         line=ignore_line_no,
  82                         column=1,
  83                         confidence=calculate_confidence(sanity_ignore.path, ignore_line_no, args.metadata) if args.metadata.changes else None,
  84                     ))
  85 
  86         if messages:
  87             return SanityFailure(self.name, messages=messages)
  88 
  89         return SanitySuccess(self.name)
