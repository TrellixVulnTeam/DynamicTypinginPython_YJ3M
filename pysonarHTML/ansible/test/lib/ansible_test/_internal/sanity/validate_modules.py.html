  51 class ValidateModulesTest(SanitySingleVersion):
  52     """Sanity test using validate-modules."""
  53 
  54     def __init__(self):
  55         super(ValidateModulesTest, self).__init__()
  56         self.optional_error_codes.update([
  57             'deprecated-date',
  58         ])
  59 
  60     @property
  61     def error_code(self):  # type: () -> t.Optional[str]
  62         """Error code for ansible-test matching the format used by the underlying test program, or None if the program does not use error codes."""
  63         return 'A100'
  64 
  65     def filter_targets(self, targets):  # type: (t.List[TestTarget]) -> t.List[TestTarget]
  66         """Return the given list of test targets, filtered to include only those relevant for the test."""
  67         return [target for target in targets if target.module]
  68 
  69     def test(self, args, targets, python_version):
  70         """
  71         :type args: SanityConfig
  72         :type targets: SanityTargets
  73         :type python_version: str
  74         :rtype: TestResult
  75         """
  76         env = ansible_environment(args, color=False)
  77 
  78         settings = self.load_processor(args)
  79 
  80         paths = [target.path for target in targets.include]
  81 
  82         python = find_python(python_version)
  83 
  84         cmd = [
  85             python,
  86             os.path.join(SANITY_ROOT, 'validate-modules', 'validate-modules'),
  87             '--format', 'json',
  88             '--arg-spec',
  89         ] + paths
  90 
  91         if data_context().content.collection:
  92             cmd.extend(['--collection', data_context().content.collection.directory])
  93 
  94             try:
  95                 collection_detail = get_collection_detail(args, python)
  96 
  97                 if collection_detail.version:
  98                     cmd.extend(['--collection-version', collection_detail.version])
  99                 else:
 100                     display.warning('Skipping validate-modules collection version checks since no collection version was found.')
 101             except CollectionDetailError as ex:
 102                 display.warning('Skipping validate-modules collection version checks since collection detail loading failed: %s' % ex.reason)
 103         else:
 104             base_branch = args.base_branch or get_ci_provider().get_base_branch()
 105 
 106             if base_branch:
 107                 cmd.extend([
 108                     '--base-branch', base_branch,
 109                 ])
 110             else:
 111                 display.warning('Cannot perform module comparison against the base branch because the base branch was not detected.')
 112 
 113         try:
 114             stdout, stderr = run_command(args, cmd, env=env, capture=True)
 115             status = 0
 116         except SubprocessError as ex:
 117             stdout = ex.stdout
 118             stderr = ex.stderr
 119             status = ex.status
 120 
 121         if stderr or status not in (0, 3):
 122             raise SubprocessError(cmd=cmd, status=status, stderr=stderr, stdout=stdout)
 123 
 124         if args.explain:
 125             return SanitySuccess(self.name)
 126 
 127         messages = json.loads(stdout)
 128 
 129         errors = []
 130 
 131         for filename in messages:
 132             output = messages[filename]
 133 
 134             for item in output['errors']:
 135                 errors.append(SanityMessage(
 136                     path=filename,
 137                     line=int(item['line']) if 'line' in item else 0,
 138                     column=int(item['column']) if 'column' in item else 0,
 139                     level='error',
 140                     code='%s' % item['code'],
 141                     message=item['msg'],
 142                 ))
 143 
 144         errors = settings.process_errors(errors, paths)
 145 
 146         if errors:
 147             return SanityFailure(self.name, messages=errors)
 148 
 149         return SanitySuccess(self.name)
