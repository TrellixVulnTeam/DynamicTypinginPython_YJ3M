  30 def create_virtual_environment(args,  # type: EnvironmentConfig
  31                                version,  # type: str
  32                                path,  # type: str
  33                                system_site_packages=False,  # type: bool
  34                                pip=True,  # type: bool
  35                                ):  # type: (...) -> bool
  36     """Create a virtual environment using venv or virtualenv for the requested Python version."""
  37     if os.path.isdir(path):
  38         display.info('Using existing Python %s virtual environment: %s' % (version, path), verbosity=1)
  39         return True
  40 
  41     python = find_python(version, required=False)
  42     python_version = tuple(int(v) for v in version.split('.'))
  43 
  44     if not python:
  45         # the requested python version could not be found
  46         return False
  47 
  48     if python_version >= (3, 0):
  49         # use the built-in 'venv' module on Python 3.x
  50         # creating a virtual environment using 'venv' when running in a virtual environment created by 'virtualenv' results
  51         # in a copy of the original virtual environment instead of creation of a new one
  52         # avoid this issue by only using "real" python interpreters to invoke 'venv'
  53         for real_python in iterate_real_pythons(args, version):
  54             if run_venv(args, real_python, system_site_packages, pip, path):
  55                 display.info('Created Python %s virtual environment using "venv": %s' % (version, path), verbosity=1)
  56                 return True
  57 
  58         # something went wrong, most likely the package maintainer for the Python installation removed ensurepip
  59         # which will prevent creation of a virtual environment without installation of other OS packages
  60 
  61     # use the installed 'virtualenv' module on the Python requested version
  62     if run_virtualenv(args, python, python, system_site_packages, pip, path):
  63         display.info('Created Python %s virtual environment using "virtualenv": %s' % (version, path), verbosity=1)
  64         return True
  65 
  66     available_pythons = get_available_python_versions(SUPPORTED_PYTHON_VERSIONS)
  67 
  68     for available_python_version, available_python_interpreter in sorted(available_pythons.items()):
  69         virtualenv_version = get_virtualenv_version(args, available_python_interpreter)
  70 
  71         if not virtualenv_version:
  72             # virtualenv not available for this Python or we were unable to detect the version
  73             continue
  74 
  75         if python_version == (2, 6) and virtualenv_version >= (16, 0, 0):
  76             # virtualenv 16.0.0 dropped python 2.6 support: https://virtualenv.pypa.io/en/latest/changes/#v16-0-0-2018-05-16
  77             continue
  78 
  79         # try using 'virtualenv' from another Python to setup the desired version
  80         if run_virtualenv(args, available_python_interpreter, python, system_site_packages, pip, path):
  81             display.info('Created Python %s virtual environment using "virtualenv" on Python %s: %s' % (version, available_python_version, path), verbosity=1)
  82             return True
  83 
  84     # no suitable 'virtualenv' available
  85     return False

  88 def iterate_real_pythons(args, version):  # type: (EnvironmentConfig, str) -> t.Iterable[str]
  89     """
  90     Iterate through available real python interpreters of the requested version.
  91     The current interpreter will be checked and then the path will be searched.
  92     """
  93     version_info = tuple(int(n) for n in version.split('.'))
  94     current_python = None
  95 
  96     if version_info == sys.version_info[:len(version_info)]:
  97         current_python = sys.executable
  98         real_prefix = get_python_real_prefix(args, current_python)
  99 
 100         if real_prefix:
 101             current_python = find_python(version, os.path.join(real_prefix, 'bin'))
 102 
 103         if current_python:
 104             yield current_python
 105 
 106     path = os.environ.get('PATH', os.path.defpath)
 107 
 108     if not path:
 109         return
 110 
 111     found_python = find_python(version, path)
 112 
 113     if not found_python:
 114         return
 115 
 116     if found_python == current_python:
 117         return
 118 
 119     real_prefix = get_python_real_prefix(args, found_python)
 120 
 121     if real_prefix:
 122         found_python = find_python(version, os.path.join(real_prefix, 'bin'))
 123 
 124     if found_python:
 125         yield found_python

 128 def get_python_real_prefix(args, path):  # type: (EnvironmentConfig, str) -> t.Optional[str]
 129     """
 130     Return the real prefix of the specified interpreter or None if the interpreter is not a virtual environment created by 'virtualenv'.
 131     """
 132     cmd = [path, os.path.join(os.path.join(ANSIBLE_TEST_DATA_ROOT, 'virtualenvcheck.py'))]
 133     check_result = json.loads(run_command(args, cmd, capture=True, always=True)[0])
 134     real_prefix = check_result['real_prefix']
 135     return real_prefix

 138 def run_venv(args,  # type: EnvironmentConfig
 139              run_python,  # type: str
 140              system_site_packages,  # type: bool
 141              pip,  # type: bool
 142              path,  # type: str
 143              ):  # type: (...) -> bool
 144     """Create a virtual environment using the 'venv' module. Not available on Python 2.x."""
 145     cmd = [run_python, '-m', 'venv']
 146 
 147     if system_site_packages:
 148         cmd.append('--system-site-packages')
 149 
 150     if not pip:
 151         cmd.append('--without-pip')
 152 
 153     cmd.append(path)
 154 
 155     try:
 156         run_command(args, cmd, capture=True)
 157     except SubprocessError as ex:
 158         remove_tree(path)
 159 
 160         if args.verbosity > 1:
 161             display.error(ex)
 162 
 163         return False
 164 
 165     return True

 168 def run_virtualenv(args,  # type: EnvironmentConfig
 169                    run_python,  # type: str
 170                    env_python,  # type: str
 171                    system_site_packages,  # type: bool
 172                    pip,  # type: bool
 173                    path,  # type: str
 174                    ):  # type: (...) -> bool
 175     """Create a virtual environment using the 'virtualenv' module."""
 176     # always specify --python to guarantee the desired interpreter is provided
 177     # otherwise virtualenv may select a different interpreter than the one running virtualenv
 178     cmd = [run_python, '-m', 'virtualenv', '--python', env_python]
 179 
 180     if system_site_packages:
 181         cmd.append('--system-site-packages')
 182 
 183     if not pip:
 184         cmd.append('--no-pip')
 185 
 186     cmd.append(path)
 187 
 188     try:
 189         run_command(args, cmd, capture=True)
 190     except SubprocessError as ex:
 191         remove_tree(path)
 192 
 193         if args.verbosity > 1:
 194             display.error(ex)
 195 
 196         return False
 197 
 198     return True

 201 def get_virtualenv_version(args, python):  # type: (EnvironmentConfig, str) -> t.Optional[t.Tuple[int, ...]]
 202     """Get the virtualenv version for the given python intepreter, if available."""
 203     try:
 204         return get_virtualenv_version.result
 205     except AttributeError:
 206         pass
 207 
 208     get_virtualenv_version.result = None
 209 
 210     cmd = [python, '-m', 'virtualenv', '--version']
 211 
 212     try:
 213         stdout = run_command(args, cmd, capture=True)[0]
 214     except SubprocessError as ex:
 215         if args.verbosity > 1:
 216             display.error(ex)
 217 
 218         stdout = ''
 219 
 220     if stdout:
 221         # noinspection PyBroadException
 222         try:
 223             get_virtualenv_version.result = tuple(int(v) for v in stdout.strip().split('.'))
 224         except Exception:  # pylint: disable=broad-except
 225             pass
 226 
 227     return get_virtualenv_version.result
