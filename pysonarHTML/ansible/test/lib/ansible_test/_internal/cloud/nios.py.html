  27 class NiosProvider(CloudProvider):
  28     """Nios plugin.
  29 
  30     Sets up NIOS mock server for tests.
  31     """
  32 
  33     DOCKER_SIMULATOR_NAME = 'nios-simulator'
  34 
  35     DOCKER_IMAGE = 'quay.io/ansible/nios-test-container:1.3.0'
  36     """Default image to run the nios simulator.
  37 
  38     The simulator must be pinned to a specific version
  39     to guarantee CI passes with the version used.
  40 
  41     It's source source itself resides at:
  42     https://github.com/ansible/nios-test-container
  43     """
  44 
  45     def __init__(self, args):
  46         """Set up container references for provider.
  47 
  48         :type args: TestConfig
  49         """
  50         super(NiosProvider, self).__init__(args)
  51 
  52         self.__container_from_env = os.environ.get('ANSIBLE_NIOSSIM_CONTAINER')
  53         """Overrides target container, might be used for development.
  54 
  55         Use ANSIBLE_NIOSSIM_CONTAINER=whatever_you_want if you want
  56         to use other image. Omit/empty otherwise.
  57         """
  58 
  59         self.image = self.__container_from_env or self.DOCKER_IMAGE
  60         self.container_name = ''
  61 
  62     def filter(self, targets, exclude):
  63         """Filter out the tests with the necessary config and res unavailable.
  64 
  65         :type targets: tuple[TestTarget]
  66         :type exclude: list[str]
  67         """
  68         docker_cmd = 'docker'
  69         docker = find_executable(docker_cmd, required=False)
  70 
  71         if docker:
  72             return
  73 
  74         skip = 'cloud/%s/' % self.platform
  75         skipped = [target.name for target in targets if skip in target.aliases]
  76 
  77         if skipped:
  78             exclude.append(skip)
  79             display.warning(
  80                 'Excluding tests marked "%s" '
  81                 'which require the "%s" command: %s'
  82                 % (skip.rstrip('/'), docker_cmd, ', '.join(skipped))
  83             )
  84 
  85     def setup(self):
  86         """Setup cloud resource before delegation and reg cleanup callback."""
  87         super(NiosProvider, self).setup()
  88 
  89         if self._use_static_config():
  90             self._setup_static()
  91         else:
  92             self._setup_dynamic()
  93 
  94     def get_docker_run_options(self):
  95         """Get additional options needed when delegating tests to a container.
  96 
  97         :rtype: list[str]
  98         """
  99         return ['--link', self.DOCKER_SIMULATOR_NAME] if self.managed else []
 100 
 101     def cleanup(self):
 102         """Clean up the resource and temporary configs files after tests."""
 103         if self.container_name:
 104             docker_rm(self.args, self.container_name)
 105 
 106         super(NiosProvider, self).cleanup()
 107 
 108     def _setup_dynamic(self):
 109         """Spawn a NIOS simulator within docker container."""
 110         nios_port = 443
 111         container_id = get_docker_container_id()
 112 
 113         if container_id:
 114             display.info(
 115                 'Running in docker container: %s' % container_id,
 116                 verbosity=1,
 117             )
 118 
 119         self.container_name = self.DOCKER_SIMULATOR_NAME
 120 
 121         results = docker_inspect(self.args, self.container_name)
 122 
 123         if results and not results[0].get('State', {}).get('Running'):
 124             docker_rm(self.args, self.container_name)
 125             results = []
 126 
 127         display.info(
 128             '%s NIOS simulator docker container.'
 129             % ('Using the existing' if results else 'Starting a new'),
 130             verbosity=1,
 131         )
 132 
 133         if not results:
 134             if self.args.docker or container_id:
 135                 publish_ports = []
 136             else:
 137                 # publish the simulator ports when not running inside docker
 138                 publish_ports = [
 139                     '-p', ':'.join((str(nios_port), ) * 2),
 140                 ]
 141 
 142             if not self.__container_from_env:
 143                 docker_pull(self.args, self.image)
 144 
 145             docker_run(
 146                 self.args,
 147                 self.image,
 148                 ['-d', '--name', self.container_name] + publish_ports,
 149             )
 150 
 151         if self.args.docker:
 152             nios_host = self.DOCKER_SIMULATOR_NAME
 153         elif container_id:
 154             nios_host = self._get_simulator_address()
 155             display.info(
 156                 'Found NIOS simulator container address: %s'
 157                 % nios_host, verbosity=1
 158             )
 159         else:
 160             nios_host = 'localhost'
 161 
 162         self._set_cloud_config('NIOS_HOST', nios_host)
 163 
 164     def _get_simulator_address(self):
 165         results = docker_inspect(self.args, self.container_name)
 166         ip_address = results[0]['NetworkSettings']['IPAddress']
 167         return ip_address
 168 
 169     def _setup_static(self):
 170         raise NotImplementedError

 173 class NiosEnvironment(CloudEnvironment):
 174     """NIOS environment plugin.
 175 
 176     Updates integration test environment after delegation.
 177     """
 178     def get_environment_config(self):
 179         """
 180         :rtype: CloudEnvironmentConfig
 181         """
 182         ansible_vars = dict(
 183             nios_provider=dict(
 184                 host=self._get_cloud_config('NIOS_HOST'),
 185                 username='admin',
 186                 password='infoblox',
 187             ),
 188         )
 189 
 190         return CloudEnvironmentConfig(
 191             ansible_vars=ansible_vars,
 192         )
