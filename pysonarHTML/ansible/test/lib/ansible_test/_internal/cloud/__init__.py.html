  65 def get_cloud_platforms(args, targets=None):
  66     """
  67     :type args: TestConfig
  68     :type targets: tuple[IntegrationTarget] | None
  69     :rtype: list[str]
  70     """
  71     if isinstance(args, IntegrationConfig):
  72         if args.list_targets:
  73             return []
  74 
  75     if targets is None:
  76         cloud_platforms = set(args.metadata.cloud_config or [])
  77     else:
  78         cloud_platforms = set(get_cloud_platform(target) for target in targets)
  79 
  80     cloud_platforms.discard(None)
  81 
  82     return sorted(cloud_platforms)
  83

  85 def get_cloud_platform(target):
  86     """
  87     :type target: IntegrationTarget
  88     :rtype: str | None
  89     """
  90     cloud_platforms = set(a.split('/')[1] for a in target.aliases if a.startswith('cloud/') and a.endswith('/') and a != 'cloud/')
  91 
  92     if not cloud_platforms:
  93         return None
  94 
  95     if len(cloud_platforms) == 1:
  96         cloud_platform = cloud_platforms.pop()
  97 
  98         if cloud_platform not in PROVIDERS:
  99             raise ApplicationError('Target %s aliases contains unknown cloud platform: %s' % (target.name, cloud_platform))
 100 
 101         return cloud_platform
 102 
 103     raise ApplicationError('Target %s aliases contains multiple cloud platforms: %s' % (target.name, ', '.join(sorted(cloud_platforms))))

 129 def cloud_filter(args, targets):
 130     """
 131     :type args: IntegrationConfig
 132     :type targets: tuple[IntegrationTarget]
 133     :return: list[str]
 134     """
 135     if args.metadata.cloud_config is not None:
 136         return []  # cloud filter already performed prior to delegation
 137 
 138     exclude = []
 139 
 140     for provider in get_cloud_providers(args, targets):
 141         provider.filter(targets, exclude)
 142 
 143     return exclude

 146 def cloud_init(args, targets):
 147     """
 148     :type args: IntegrationConfig
 149     :type targets: tuple[IntegrationTarget]
 150     """
 151     if args.metadata.cloud_config is not None:
 152         return  # cloud configuration already established prior to delegation
 153 
 154     args.metadata.cloud_config = {}
 155 
 156     results = {}
 157 
 158     for provider in get_cloud_providers(args, targets):
 159         args.metadata.cloud_config[provider.platform] = {}
 160 
 161         start_time = time.time()
 162         provider.setup()
 163         end_time = time.time()
 164 
 165         results[provider.platform] = dict(
 166             platform=provider.platform,
 167             setup_seconds=int(end_time - start_time),
 168             targets=[target.name for target in targets],
 169         )
 170 
 171     if not args.explain and results:
 172         result_name = '%s-%s.json' % (
 173             args.command, re.sub(r'[^0-9]', '-', str(datetime.datetime.utcnow().replace(microsecond=0))))
 174 
 175         data = dict(
 176             clouds=results,
 177         )
 178 
 179         write_json_test_results(ResultType.DATA, result_name, data)

 182 class CloudBase(ABC):
 183     """Base class for cloud plugins."""
 184     __metaclass__ = abc.ABCMeta
 185 
 186     _CONFIG_PATH = 'config_path'
 187     _RESOURCE_PREFIX = 'resource_prefix'
 188     _MANAGED = 'managed'
 189     _SETUP_EXECUTED = 'setup_executed'
 190 
 191     def __init__(self, args):
 192         """
 193         :type args: IntegrationConfig
 194         """
 195         self.args = args
 196         self.platform = self.__module__.split('.')[-1]
 197 
 198         def config_callback(files):  # type: (t.List[t.Tuple[str, str]]) -> None
 199             """Add the config file to the payload file list."""
 200             if self._get_cloud_config(self._CONFIG_PATH, ''):
 201                 pair = (self.config_path, os.path.relpath(self.config_path, data_context().content.root))
 202 
 203                 if pair not in files:
 204                     display.info('Including %s config: %s -> %s' % (self.platform, pair[0], pair[1]), verbosity=3)
 205                     files.append(pair)
 206 
 207         data_context().register_payload_callback(config_callback)
 208 
 209     @property
 210     def setup_executed(self):
 211         """
 212         :rtype: bool
 213         """
 214         return self._get_cloud_config(self._SETUP_EXECUTED, False)
 215 
 216     @setup_executed.setter
 217     def setup_executed(self, value):
 218         """
 219         :type value: bool
 220         """
 221         self._set_cloud_config(self._SETUP_EXECUTED, value)
 222 
 223     @property
 224     def config_path(self):
 225         """
 226         :rtype: str
 227         """
 228         return os.path.join(data_context().content.root, self._get_cloud_config(self._CONFIG_PATH))
 229 
 230     @config_path.setter
 231     def config_path(self, value):
 232         """
 233         :type value: str
 234         """
 235         self._set_cloud_config(self._CONFIG_PATH, value)
 236 
 237     @property
 238     def resource_prefix(self):
 239         """
 240         :rtype: str
 241         """
 242         return self._get_cloud_config(self._RESOURCE_PREFIX)
 243 
 244     @resource_prefix.setter
 245     def resource_prefix(self, value):
 246         """
 247         :type value: str
 248         """
 249         self._set_cloud_config(self._RESOURCE_PREFIX, value)
 250 
 251     @property
 252     def managed(self):
 253         """
 254         :rtype: bool
 255         """
 256         return self._get_cloud_config(self._MANAGED)
 257 
 258     @managed.setter
 259     def managed(self, value):
 260         """
 261         :type value: bool
 262         """
 263         self._set_cloud_config(self._MANAGED, value)
 264 
 265     def _get_cloud_config(self, key, default=None):
 266         """
 267         :type key: str
 268         :type default: str | int | bool | None
 269         :rtype: str | int | bool
 270         """
 271         if default is not None:
 272             return self.args.metadata.cloud_config[self.platform].get(key, default)
 273 
 274         return self.args.metadata.cloud_config[self.platform][key]
 275 
 276     def _set_cloud_config(self, key, value):
 277         """
 278         :type key: str
 279         :type value: str | int | bool
 280         """
 281         self.args.metadata.cloud_config[self.platform][key] = value

 284 class CloudProvider(CloudBase):
 285     """Base class for cloud provider plugins. Sets up cloud resources before delegation."""
 286     def __init__(self, args, config_extension='.ini'):
 287         """
 288         :type args: IntegrationConfig
 289         :type config_extension: str
 290         """
 291         super(CloudProvider, self).__init__(args)
 292 
 293         self.ci_provider = get_ci_provider()
 294         self.remove_config = False
 295         self.config_static_name = 'cloud-config-%s%s' % (self.platform, config_extension)
 296         self.config_static_path = os.path.join(data_context().content.integration_path, self.config_static_name)
 297         self.config_template_path = os.path.join(ANSIBLE_TEST_CONFIG_ROOT, '%s.template' % self.config_static_name)
 298         self.config_extension = config_extension
 299 
 300     def filter(self, targets, exclude):
 301         """Filter out the cloud tests when the necessary config and resources are not available.
 302         :type targets: tuple[TestTarget]
 303         :type exclude: list[str]
 304         """
 305         skip = 'cloud/%s/' % self.platform
 306         skipped = [target.name for target in targets if skip in target.aliases]
 307 
 308         if skipped:
 309             exclude.append(skip)
 310             display.warning('Excluding tests marked "%s" which require config (see "%s"): %s'
 311                             % (skip.rstrip('/'), self.config_template_path, ', '.join(skipped)))
 312 
 313     def setup(self):
 314         """Setup the cloud resource before delegation and register a cleanup callback."""
 315         self.resource_prefix = self.ci_provider.generate_resource_prefix()
 316 
 317         atexit.register(self.cleanup)
 318 
 319     def get_remote_ssh_options(self):
 320         """Get any additional options needed when delegating tests to a remote instance via SSH.
 321         :rtype: list[str]
 322         """
 323         return []
 324 
 325     def get_docker_run_options(self):
 326         """Get any additional options needed when delegating tests to a docker container.
 327         :rtype: list[str]
 328         """
 329         return []
 330 
 331     def cleanup(self):
 332         """Clean up the cloud resource and any temporary configuration files after tests complete."""
 333         if self.remove_config:
 334             os.remove(self.config_path)
 335 
 336     def _use_static_config(self):
 337         """
 338         :rtype: bool
 339         """
 340         if os.path.isfile(self.config_static_path):
 341             display.info('Using existing %s cloud config: %s' % (self.platform, self.config_static_path), verbosity=1)
 342             self.config_path = self.config_static_path
 343             static = True
 344         else:
 345             static = False
 346 
 347         self.managed = not static
 348 
 349         return static
 350 
 351     def _write_config(self, content):
 352         """
 353         :type content: str
 354         """
 355         prefix = '%s-' % os.path.splitext(os.path.basename(self.config_static_path))[0]
 356 
 357         with tempfile.NamedTemporaryFile(dir=data_context().content.integration_path, prefix=prefix, suffix=self.config_extension, delete=False) as config_fd:
 358             filename = os.path.join(data_context().content.integration_path, os.path.basename(config_fd.name))
 359 
 360             self.config_path = filename
 361             self.remove_config = True
 362 
 363             display.info('>>> Config: %s\n%s' % (filename, content.strip()), verbosity=3)
 364 
 365             config_fd.write(to_bytes(content))
 366             config_fd.flush()
 367 
 368     def _read_config_template(self):
 369         """
 370         :rtype: str
 371         """
 372         lines = read_text_file(self.config_template_path).splitlines()
 373         lines = [line for line in lines if not line.startswith('#')]
 374         config = '\n'.join(lines).strip() + '\n'
 375         return config
 376 
 377     @staticmethod
 378     def _populate_config_template(template, values):
 379         """
 380         :type template: str
 381         :type values: dict[str, str]
 382         :rtype: str
 383         """
 384         for key in sorted(values):
 385             value = values[key]
 386             template = template.replace('@%s' % key, value)
 387 
 388         return template
