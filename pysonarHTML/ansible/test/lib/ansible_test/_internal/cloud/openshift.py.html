  42 class OpenShiftCloudProvider(CloudProvider):
  43     """OpenShift cloud provider plugin. Sets up cloud resources before delegation."""
  44     DOCKER_CONTAINER_NAME = 'openshift-origin'
  45 
  46     def __init__(self, args):
  47         """
  48         :type args: TestConfig
  49         """
  50         super(OpenShiftCloudProvider, self).__init__(args, config_extension='.kubeconfig')
  51 
  52         # The image must be pinned to a specific version to guarantee CI passes with the version used.
  53         self.image = 'openshift/origin:v3.9.0'
  54         self.container_name = ''
  55 
  56     def filter(self, targets, exclude):
  57         """Filter out the cloud tests when the necessary config and resources are not available.
  58         :type targets: tuple[TestTarget]
  59         :type exclude: list[str]
  60         """
  61         if os.path.isfile(self.config_static_path):
  62             return
  63 
  64         docker = find_executable('docker', required=False)
  65 
  66         if docker:
  67             return
  68 
  69         skip = 'cloud/%s/' % self.platform
  70         skipped = [target.name for target in targets if skip in target.aliases]
  71 
  72         if skipped:
  73             exclude.append(skip)
  74             display.warning('Excluding tests marked "%s" which require the "docker" command or config (see "%s"): %s'
  75                             % (skip.rstrip('/'), self.config_template_path, ', '.join(skipped)))
  76 
  77     def setup(self):
  78         """Setup the cloud resource before delegation and register a cleanup callback."""
  79         super(OpenShiftCloudProvider, self).setup()
  80 
  81         if self._use_static_config():
  82             self._setup_static()
  83         else:
  84             self._setup_dynamic()
  85 
  86     def get_remote_ssh_options(self):
  87         """Get any additional options needed when delegating tests to a remote instance via SSH.
  88         :rtype: list[str]
  89         """
  90         if self.managed:
  91             return ['-R', '8443:localhost:8443']
  92 
  93         return []
  94 
  95     def get_docker_run_options(self):
  96         """Get any additional options needed when delegating tests to a docker container.
  97         :rtype: list[str]
  98         """
  99         if self.managed:
 100             return ['--link', self.DOCKER_CONTAINER_NAME]
 101 
 102         return []

 104     def cleanup(self):
 105         """Clean up the cloud resource and any temporary configuration files after tests complete."""
 106         if self.container_name:
 107             docker_rm(self.args, self.container_name)
 108 
 109         super(OpenShiftCloudProvider, self).cleanup()
 110 
 111     def _setup_static(self):
 112         """Configure OpenShift tests for use with static configuration."""
 113         config = read_text_file(self.config_static_path)
 114 
 115         match = re.search(r'^ *server: (?P<server>.*)$', config, flags=re.MULTILINE)
 116 
 117         if match:
 118             endpoint = match.group('server')
 119             self._wait_for_service(endpoint)
 120         else:
 121             display.warning('Could not find OpenShift endpoint in kubeconfig. Skipping check for OpenShift service availability.')
 122 
 123     def _setup_dynamic(self):
 124         """Create a OpenShift container using docker."""
 125         self.container_name = self.DOCKER_CONTAINER_NAME
 126 
 127         results = docker_inspect(self.args, self.container_name)
 128 
 129         if results and not results[0]['State']['Running']:
 130             docker_rm(self.args, self.container_name)
 131             results = []
 132 
 133         if results:
 134             display.info('Using the existing OpenShift docker container.', verbosity=1)
 135         else:
 136             display.info('Starting a new OpenShift docker container.', verbosity=1)
 137             docker_pull(self.args, self.image)
 138             cmd = ['start', 'master', '--listen', 'https://0.0.0.0:8443']
 139             docker_run(self.args, self.image, ['-d', '-p', '8443:8443', '--name', self.container_name], cmd)
 140 
 141         container_id = get_docker_container_id()
 142 
 143         if container_id:
 144             display.info('Running in docker container: %s' % container_id, verbosity=1)
 145             host = self._get_container_address()
 146             display.info('Found OpenShift container address: %s' % host, verbosity=1)
 147         else:
 148             host = 'localhost'
 149 
 150         port = 8443
 151         endpoint = 'https://%s:%s/' % (host, port)
 152 
 153         self._wait_for_service(endpoint)
 154 
 155         if self.args.explain:
 156             config = '# Unknown'
 157         else:
 158             if self.args.docker:
 159                 host = self.DOCKER_CONTAINER_NAME
 160 
 161             server = 'https://%s:%s' % (host, port)
 162             config = self._get_config(server)
 163 
 164         self._write_config(config)
 165 
 166     def _get_container_address(self):
 167         networks = docker_network_inspect(self.args, 'bridge')
 168 
 169         try:
 170             bridge = [network for network in networks if network['Name'] == 'bridge'][0]
 171             containers = bridge['Containers']
 172             container = [containers[container] for container in containers if containers[container]['Name'] == self.DOCKER_CONTAINER_NAME][0]
 173             return re.sub(r'/[0-9]+$', '', container['IPv4Address'])
 174         except Exception:
 175             display.error('Failed to process the following docker network inspect output:\n%s' %
 176                           json.dumps(networks, indent=4, sort_keys=True))
 177             raise
 178 
 179     def _wait_for_service(self, endpoint):
 180         """Wait for the OpenShift service endpoint to accept connections.
 181         :type endpoint: str
 182         """
 183         if self.args.explain:
 184             return
 185 
 186         client = HttpClient(self.args, always=True, insecure=True)
 187 
 188         for dummy in range(1, 30):
 189             display.info('Waiting for OpenShift service: %s' % endpoint, verbosity=1)
 190 
 191             try:
 192                 client.get(endpoint)
 193                 return
 194             except SubprocessError:
 195                 pass
 196 
 197             time.sleep(10)
 198 
 199         raise ApplicationError('Timeout waiting for OpenShift service.')
 200 
 201     def _get_config(self, server):
 202         """Get OpenShift config from container.
 203         :type server: str
 204         :rtype: dict[str, str]
 205         """
 206         cmd = ['cat', '/var/lib/origin/openshift.local.config/master/admin.kubeconfig']
 207 
 208         stdout, dummy = docker_exec(self.args, self.container_name, cmd, capture=True)
 209 
 210         config = stdout
 211         config = re.sub(r'^( *)certificate-authority-data: .*$', r'\1insecure-skip-tls-verify: true', config, flags=re.MULTILINE)
 212         config = re.sub(r'^( *)server: .*$', r'\1server: %s' % server, config, flags=re.MULTILINE)
 213 
 214         return config
