  41 class CsCloudProvider(CloudProvider):
  42     """CloudStack cloud provider plugin. Sets up cloud resources before delegation."""
  43     DOCKER_SIMULATOR_NAME = 'cloudstack-sim'
  44 
  45     def __init__(self, args):
  46         """
  47         :type args: TestConfig
  48         """
  49         super(CsCloudProvider, self).__init__(args)
  50 
  51         # The simulator must be pinned to a specific version to guarantee CI passes with the version used.
  52         self.image = 'quay.io/ansible/cloudstack-test-container:1.2.0'
  53         self.container_name = ''
  54         self.endpoint = ''
  55         self.host = ''
  56         self.port = 0
  57 
  58     def filter(self, targets, exclude):
  59         """Filter out the cloud tests when the necessary config and resources are not available.
  60         :type targets: tuple[TestTarget]
  61         :type exclude: list[str]
  62         """
  63         if os.path.isfile(self.config_static_path):
  64             return
  65 
  66         docker = find_executable('docker', required=False)
  67 
  68         if docker:
  69             return
  70 
  71         skip = 'cloud/%s/' % self.platform
  72         skipped = [target.name for target in targets if skip in target.aliases]
  73 
  74         if skipped:
  75             exclude.append(skip)
  76             display.warning('Excluding tests marked "%s" which require the "docker" command or config (see "%s"): %s'
  77                             % (skip.rstrip('/'), self.config_template_path, ', '.join(skipped)))
  78 
  79     def setup(self):
  80         """Setup the cloud resource before delegation and register a cleanup callback."""
  81         super(CsCloudProvider, self).setup()
  82 
  83         if self._use_static_config():
  84             self._setup_static()
  85         else:
  86             self._setup_dynamic()
  87 
  88     def get_remote_ssh_options(self):
  89         """Get any additional options needed when delegating tests to a remote instance via SSH.
  90         :rtype: list[str]
  91         """
  92         if self.managed:
  93             return ['-R', '8888:localhost:8888']
  94 
  95         return []
  96 
  97     def get_docker_run_options(self):
  98         """Get any additional options needed when delegating tests to a docker container.
  99         :rtype: list[str]
 100         """
 101         if self.managed:
 102             return ['--link', self.DOCKER_SIMULATOR_NAME]
 103 
 104         return []
 105 
 106     def cleanup(self):
 107         """Clean up the cloud resource and any temporary configuration files after tests complete."""
 108         if self.container_name:
 109             if self.ci_provider.code:
 110                 docker_rm(self.args, self.container_name)
 111             elif not self.args.explain:
 112                 display.notice('Remember to run `docker rm -f %s` when finished testing.' % self.container_name)
 113 
 114         super(CsCloudProvider, self).cleanup()
 115 
 116     def _setup_static(self):
 117         """Configure CloudStack tests for use with static configuration."""
 118         parser = ConfigParser()
 119         parser.read(self.config_static_path)
 120 
 121         self.endpoint = parser.get('cloudstack', 'endpoint')
 122 
 123         parts = urlparse(self.endpoint)
 124 
 125         self.host = parts.hostname
 126 
 127         if not self.host:
 128             raise ApplicationError('Could not determine host from endpoint: %s' % self.endpoint)
 129 
 130         if parts.port:
 131             self.port = parts.port
 132         elif parts.scheme == 'http':
 133             self.port = 80
 134         elif parts.scheme == 'https':
 135             self.port = 443
 136         else:
 137             raise ApplicationError('Could not determine port from endpoint: %s' % self.endpoint)
 138 
 139         display.info('Read cs host "%s" and port %d from config: %s' % (self.host, self.port, self.config_static_path), verbosity=1)
 140 
 141         self._wait_for_service()
 142 
 143     def _setup_dynamic(self):
 144         """Create a CloudStack simulator using docker."""
 145         config = self._read_config_template()
 146 
 147         self.container_name = self.DOCKER_SIMULATOR_NAME
 148 
 149         results = docker_inspect(self.args, self.container_name)
 150 
 151         if results and not results[0]['State']['Running']:
 152             docker_rm(self.args, self.container_name)
 153             results = []
 154 
 155         if results:
 156             display.info('Using the existing CloudStack simulator docker container.', verbosity=1)
 157         else:
 158             display.info('Starting a new CloudStack simulator docker container.', verbosity=1)
 159             docker_pull(self.args, self.image)
 160             docker_run(self.args, self.image, ['-d', '-p', '8888:8888', '--name', self.container_name])
 161 
 162             # apply work-around for OverlayFS issue
 163             # https://github.com/docker/for-linux/issues/72#issuecomment-319904698
 164             docker_exec(self.args, self.container_name, ['find', '/var/lib/mysql', '-type', 'f', '-exec', 'touch', '{}', ';'])
 165 
 166             if not self.args.explain:
 167                 display.notice('The CloudStack simulator will probably be ready in 2 - 4 minutes.')
 168 
 169         container_id = get_docker_container_id()
 170 
 171         if container_id:
 172             display.info('Running in docker container: %s' % container_id, verbosity=1)
 173             self.host = self._get_simulator_address()
 174             display.info('Found CloudStack simulator container address: %s' % self.host, verbosity=1)
 175         else:
 176             self.host = 'localhost'
 177 
 178         self.port = 8888
 179         self.endpoint = 'http://%s:%d' % (self.host, self.port)
 180 
 181         self._wait_for_service()
 182 
 183         if self.args.explain:
 184             values = dict(
 185                 HOST=self.host,
 186                 PORT=str(self.port),
 187             )
 188         else:
 189             credentials = self._get_credentials()
 190 
 191             if self.args.docker:
 192                 host = self.DOCKER_SIMULATOR_NAME
 193             else:
 194                 host = self.host
 195 
 196             values = dict(
 197                 HOST=host,
 198                 PORT=str(self.port),
 199                 KEY=credentials['apikey'],
 200                 SECRET=credentials['secretkey'],
 201             )
 202 
 203             display.sensitive.add(values['SECRET'])
 204 
 205         config = self._populate_config_template(config, values)
 206 
 207         self._write_config(config)
 208 
 209     def _get_simulator_address(self):
 210         networks = docker_network_inspect(self.args, 'bridge')
 211 
 212         try:
 213             bridge = [network for network in networks if network['Name'] == 'bridge'][0]
 214             containers = bridge['Containers']
 215             container = [containers[container] for container in containers if containers[container]['Name'] == self.DOCKER_SIMULATOR_NAME][0]
 216             return re.sub(r'/[0-9]+$', '', container['IPv4Address'])
 217         except Exception:
 218             display.error('Failed to process the following docker network inspect output:\n%s' %
 219                           json.dumps(networks, indent=4, sort_keys=True))
 220             raise
 221 
 222     def _wait_for_service(self):
 223         """Wait for the CloudStack service endpoint to accept connections."""
 224         if self.args.explain:
 225             return
 226 
 227         client = HttpClient(self.args, always=True)
 228         endpoint = self.endpoint
 229 
 230         for _iteration in range(1, 30):
 231             display.info('Waiting for CloudStack service: %s' % endpoint, verbosity=1)
 232 
 233             try:
 234                 client.get(endpoint)
 235                 return
 236             except SubprocessError:
 237                 pass
 238 
 239             time.sleep(10)
 240 
 241         raise ApplicationError('Timeout waiting for CloudStack service.')
 242 
 243     def _get_credentials(self):
 244         """Wait for the CloudStack simulator to return credentials.
 245         :rtype: dict[str, str]
 246         """
 247         client = HttpClient(self.args, always=True)
 248         endpoint = '%s/admin.json' % self.endpoint
 249 
 250         for _iteration in range(1, 30):
 251             display.info('Waiting for CloudStack credentials: %s' % endpoint, verbosity=1)
 252 
 253             response = client.get(endpoint)
 254 
 255             if response.status_code == 200:
 256                 try:
 257                     return response.json()
 258                 except HttpError as ex:
 259                     display.error(ex)
 260 
 261             time.sleep(10)
 262 
 263         raise ApplicationError('Timeout waiting for CloudStack credentials.')

 266 class CsCloudEnvironment(CloudEnvironment):
 267     """CloudStack cloud environment plugin. Updates integration test environment after delegation."""
 268     def get_environment_config(self):
 269         """
 270         :rtype: CloudEnvironmentConfig
 271         """
 272         parser = ConfigParser()
 273         parser.read(self.config_path)
 274 
 275         config = dict(parser.items('default'))
 276 
 277         env_vars = dict(
 278             CLOUDSTACK_ENDPOINT=config['endpoint'],
 279             CLOUDSTACK_KEY=config['key'],
 280             CLOUDSTACK_SECRET=config['secret'],
 281             CLOUDSTACK_TIMEOUT=config['timeout'],
 282         )
 283 
 284         display.sensitive.add(env_vars['CLOUDSTACK_SECRET'])
 285 
 286         ansible_vars = dict(
 287             cs_resource_prefix=self.resource_prefix,
 288         )
 289 
 290         return CloudEnvironmentConfig(
 291             env_vars=env_vars,
 292             ansible_vars=ansible_vars,
 293         )
