  30 class TowerCloudProvider(CloudProvider):
  31     """Tower cloud provider plugin. Sets up cloud resources before delegation."""
  32     def __init__(self, args):
  33         """
  34         :type args: TestConfig
  35         """
  36         super(TowerCloudProvider, self).__init__(args)
  37 
  38         self.aci = None
  39         self.version = ''
  40 
  41     def filter(self, targets, exclude):
  42         """Filter out the cloud tests when the necessary config and resources are not available.
  43         :type targets: tuple[TestTarget]
  44         :type exclude: list[str]
  45         """
  46         if os.path.isfile(self.config_static_path):
  47             return
  48 
  49         aci = get_tower_aci(self.args)
  50 
  51         if aci.available:
  52             return
  53 
  54         super(TowerCloudProvider, self).filter(targets, exclude)
  55 
  56     def setup(self):
  57         """Setup the cloud resource before delegation and register a cleanup callback."""
  58         super(TowerCloudProvider, self).setup()
  59 
  60         if self._use_static_config():
  61             self._setup_static()
  62         else:
  63             self._setup_dynamic()
  64 
  65     def check_tower_version(self, fallback=None):
  66         """Check the Tower version being tested and determine the correct CLI version to use.
  67         :type fallback: str | None
  68         """
  69         tower_cli_version_map = {
  70             '3.1.5': '3.1.8',
  71             '3.2.3': '3.3.0',
  72             '3.3.5': '3.3.3',
  73             '3.4.3': '3.3.3',
  74             '3.6.3': '3.3.8',
  75         }
  76 
  77         cli_version = tower_cli_version_map.get(self.version, fallback)
  78 
  79         if not cli_version:
  80             raise ApplicationError('Mapping to ansible-tower-cli version required for Tower version: %s' % self.version)
  81 
  82         self._set_cloud_config('tower_cli_version', cli_version)
  83 
  84     def cleanup(self):
  85         """Clean up the cloud resource and any temporary configuration files after tests complete."""
  86         # cleanup on success or failure is not yet supported due to how cleanup is called
  87         if self.aci and self.args.remote_terminate == 'always':
  88             self.aci.stop()
  89 
  90         super(TowerCloudProvider, self).cleanup()
  91 
  92     def _setup_static(self):
  93         config = TowerConfig.parse(self.config_static_path)
  94 
  95         self.version = config.version
  96         self.check_tower_version()
  97 
  98     def _setup_dynamic(self):
  99         """Request Tower credentials through the Ansible Core CI service."""
 100         display.info('Provisioning %s cloud environment.' % self.platform, verbosity=1)
 101 
 102         # temporary solution to allow version selection
 103         self.version = os.environ.get('TOWER_VERSION', '3.6.3')
 104         self.check_tower_version(os.environ.get('TOWER_CLI_VERSION'))
 105 
 106         aci = get_tower_aci(self.args, self.version)
 107         aci.start()
 108         aci.wait()
 109 
 110         connection = aci.get()
 111 
 112         config = self._read_config_template()
 113 
 114         if not self.args.explain:
 115             self.aci = aci
 116 
 117             values = dict(
 118                 VERSION=self.version,
 119                 HOST=connection.hostname,
 120                 USERNAME=connection.username,
 121                 PASSWORD=connection.password,
 122             )
 123 
 124             display.sensitive.add(values['PASSWORD'])
 125 
 126             config = self._populate_config_template(config, values)
 127 
 128         self._write_config(config)

 131 class TowerCloudEnvironment(CloudEnvironment):
 132     """Tower cloud environment plugin. Updates integration test environment after delegation."""
 133     def setup(self):
 134         """Setup which should be done once per environment instead of once per test target."""
 135         self.setup_cli()
 136         self.disable_pendo()
 137 
 138     def setup_cli(self):
 139         """Install the correct Tower CLI for the version of Tower being tested."""
 140         tower_cli_version = self._get_cloud_config('tower_cli_version')
 141 
 142         display.info('Installing Tower CLI version: %s' % tower_cli_version)
 143 
 144         cmd = self.args.pip_command + ['install', '--disable-pip-version-check', 'ansible-tower-cli==%s' % tower_cli_version]
 145 
 146         run_command(self.args, cmd)
 147 
 148         cmd = ['tower-cli', 'config', 'verify_ssl', 'false']
 149         run_command(self.args, cmd, capture=True)
 150 
 151     def disable_pendo(self):
 152         """Disable Pendo tracking."""
 153         display.info('Disable Pendo tracking')
 154 
 155         config = TowerConfig.parse(self.config_path)
 156 
 157         # tower-cli does not recognize TOWER_ environment variables
 158         cmd = ['tower-cli', 'setting', 'modify', 'PENDO_TRACKING_STATE', 'off',
 159                '-h', config.host, '-u', config.username, '-p', config.password]
 160 
 161         attempts = 60
 162 
 163         while True:
 164             attempts -= 1
 165 
 166             try:
 167                 run_command(self.args, cmd, capture=True)
 168                 return
 169             except SubprocessError as ex:
 170                 if not attempts:
 171                     raise ApplicationError('Timed out trying to disable Pendo tracking:\n%s' % ex)
 172 
 173             time.sleep(5)
 174 
 175     def get_environment_config(self):
 176         """
 177         :rtype: CloudEnvironmentConfig
 178         """
 179         config = TowerConfig.parse(self.config_path)
 180 
 181         env_vars = config.environment
 182 
 183         ansible_vars = dict((key.lower(), value) for key, value in env_vars.items())
 184 
 185         return CloudEnvironmentConfig(
 186             env_vars=env_vars,
 187             ansible_vars=ansible_vars,
 188         )

 191 class TowerConfig:
 192     """Tower settings."""
 193     def __init__(self, values):
 194         self.version = values.get('version')
 195         self.host = values.get('host')
 196         self.username = values.get('username')
 197         self.password = values.get('password')
 198 
 199         if self.password:
 200             display.sensitive.add(self.password)
 201 
 202     @property
 203     def environment(self):
 204         """Tower settings as environment variables.
 205         :rtype: dict[str, str]
 206         """
 207         env = dict(
 208             TOWER_VERSION=self.version,
 209             TOWER_HOST=self.host,
 210             TOWER_USERNAME=self.username,
 211             TOWER_PASSWORD=self.password,
 212         )
 213 
 214         return env
 215 
 216     @staticmethod
 217     def parse(path):
 218         """
 219         :type path: str
 220         :rtype: TowerConfig
 221         """
 222         parser = ConfigParser()
 223         parser.read(path)
 224 
 225         keys = (
 226             'version',
 227             'host',
 228             'username',
 229             'password',
 230         )
 231 
 232         values = dict((k, parser.get('default', k)) for k in keys)
 233         config = TowerConfig(values)
 234 
 235         missing = [k for k in keys if not values.get(k)]
 236 
 237         if missing:
 238             raise ApplicationError('Missing or empty Tower configuration value(s): %s' % ', '.join(missing))
 239 
 240         return config

 243 def get_tower_aci(args, version=None):
 244     """
 245     :type args: EnvironmentConfig
 246     :type version: str | None
 247     :rtype: AnsibleCoreCI
 248     """
 249     if version:
 250         persist = True
 251     else:
 252         version = ''
 253         persist = False
 254 
 255     return AnsibleCoreCI(args, 'tower', version, persist=persist, stage=args.remote_stage, provider=args.remote_provider)
