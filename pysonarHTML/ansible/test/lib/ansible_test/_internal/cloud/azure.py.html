  35 class AzureCloudProvider(CloudProvider):
  36     """Azure cloud provider plugin. Sets up cloud resources before delegation."""
  37     SHERLOCK_CONFIG_PATH = os.path.expanduser('~/.ansible-sherlock-ci.cfg')
  38 
  39     def __init__(self, args):
  40         """
  41         :type args: TestConfig
  42         """
  43         super(AzureCloudProvider, self).__init__(args)
  44 
  45         self.aci = None
  46 
  47     def filter(self, targets, exclude):
  48         """Filter out the cloud tests when the necessary config and resources are not available.
  49         :type targets: tuple[TestTarget]
  50         :type exclude: list[str]
  51         """
  52         if os.path.isfile(self.config_static_path):
  53             return
  54 
  55         aci = self._create_ansible_core_ci()
  56 
  57         if aci.available:
  58             return
  59 
  60         if os.path.isfile(self.SHERLOCK_CONFIG_PATH):
  61             return
  62 
  63         super(AzureCloudProvider, self).filter(targets, exclude)
  64 
  65     def setup(self):
  66         """Setup the cloud resource before delegation and register a cleanup callback."""
  67         super(AzureCloudProvider, self).setup()
  68 
  69         if not self._use_static_config():
  70             self._setup_dynamic()
  71 
  72         get_config(self.config_path)  # check required variables
  73 
  74     def cleanup(self):
  75         """Clean up the cloud resource and any temporary configuration files after tests complete."""
  76         if self.aci:
  77             self.aci.stop()
  78 
  79         super(AzureCloudProvider, self).cleanup()
  80 
  81     def _setup_dynamic(self):
  82         """Request Azure credentials through Sherlock."""
  83         display.info('Provisioning %s cloud environment.' % self.platform, verbosity=1)
  84 
  85         config = self._read_config_template()
  86         response = {}
  87 
  88         if os.path.isfile(self.SHERLOCK_CONFIG_PATH):
  89             sherlock_uri = read_text_file(self.SHERLOCK_CONFIG_PATH).splitlines()[0].strip() + '&rgcount=2'
  90 
  91             parts = urlparse(sherlock_uri)
  92             query_string = parse_qs(parts.query)
  93             base_uri = urlunparse(parts[:4] + ('', ''))
  94 
  95             if 'code' not in query_string:
  96                 example_uri = 'https://example.azurewebsites.net/api/sandbox-provisioning'
  97                 raise ApplicationError('The Sherlock URI must include the API key in the query string. Example: %s?code=xxx' % example_uri)
  98 
  99             display.info('Initializing azure/sherlock from: %s' % base_uri, verbosity=1)
 100 
 101             http = HttpClient(self.args)
 102             result = http.get(sherlock_uri)
 103 
 104             display.info('Started azure/sherlock from: %s' % base_uri, verbosity=1)
 105 
 106             if not self.args.explain:
 107                 response = result.json()
 108         else:
 109             aci = self._create_ansible_core_ci()
 110 
 111             aci_result = aci.start()
 112 
 113             if not self.args.explain:
 114                 response = aci_result['azure']
 115                 self.aci = aci
 116 
 117         if not self.args.explain:
 118             values = dict(
 119                 AZURE_CLIENT_ID=response['clientId'],
 120                 AZURE_SECRET=response['clientSecret'],
 121                 AZURE_SUBSCRIPTION_ID=response['subscriptionId'],
 122                 AZURE_TENANT=response['tenantId'],
 123                 RESOURCE_GROUP=response['resourceGroupNames'][0],
 124                 RESOURCE_GROUP_SECONDARY=response['resourceGroupNames'][1],
 125             )
 126 
 127             display.sensitive.add(values['AZURE_SECRET'])
 128 
 129             config = '\n'.join('%s: %s' % (key, values[key]) for key in sorted(values))
 130 
 131             config = '[default]\n' + config
 132 
 133         self._write_config(config)
 134 
 135     def _create_ansible_core_ci(self):
 136         """
 137         :rtype: AnsibleCoreCI
 138         """
 139         return AnsibleCoreCI(self.args, 'azure', 'azure', persist=False, stage=self.args.remote_stage, provider=self.args.remote_provider)

 142 class AzureCloudEnvironment(CloudEnvironment):
 143     """Azure cloud environment plugin. Updates integration test environment after delegation."""
 144     def get_environment_config(self):
 145         """
 146         :rtype: CloudEnvironmentConfig
 147         """
 148         env_vars = get_config(self.config_path)
 149 
 150         display.sensitive.add(env_vars.get('AZURE_SECRET'))
 151         display.sensitive.add(env_vars.get('AZURE_PASSWORD'))
 152 
 153         ansible_vars = dict(
 154             resource_prefix=self.resource_prefix,
 155         )
 156 
 157         ansible_vars.update(dict((key.lower(), value) for key, value in env_vars.items()))
 158 
 159         return CloudEnvironmentConfig(
 160             env_vars=env_vars,
 161             ansible_vars=ansible_vars,
 162         )

 164     def on_failure(self, target, tries):
 165         """
 166         :type target: TestTarget
 167         :type tries: int
 168         """
 169         if not tries and self.managed:
 170             display.notice('If %s failed due to permissions, the test policy may need to be updated. '
 171                            'For help, consult @mattclay or @gundalow on GitHub or #ansible-devel on IRC.' % target.name)

 174 def get_config(config_path):
 175     """
 176     :type config_path: str
 177     :rtype: dict[str, str]
 178     """
 179     parser = ConfigParser()
 180     parser.read(config_path)
 181 
 182     config = dict((key.upper(), value) for key, value in parser.items('default'))
 183 
 184     rg_vars = (
 185         'RESOURCE_GROUP',
 186         'RESOURCE_GROUP_SECONDARY',
 187     )
 188 
 189     sp_vars = (
 190         'AZURE_CLIENT_ID',
 191         'AZURE_SECRET',
 192         'AZURE_SUBSCRIPTION_ID',
 193         'AZURE_TENANT',
 194     )
 195 
 196     ad_vars = (
 197         'AZURE_AD_USER',
 198         'AZURE_PASSWORD',
 199         'AZURE_SUBSCRIPTION_ID',
 200     )
 201 
 202     rg_ok = all(var in config for var in rg_vars)
 203     sp_ok = all(var in config for var in sp_vars)
 204     ad_ok = all(var in config for var in ad_vars)
 205 
 206     if not rg_ok:
 207         raise ApplicationError('Resource groups must be defined with: %s' % ', '.join(sorted(rg_vars)))
 208 
 209     if not sp_ok and not ad_ok:
 210         raise ApplicationError('Credentials must be defined using either:\nService Principal: %s\nActive Directory: %s' % (
 211             ', '.join(sorted(sp_vars)), ', '.join(sorted(ad_vars))))
 212 
 213     return config
