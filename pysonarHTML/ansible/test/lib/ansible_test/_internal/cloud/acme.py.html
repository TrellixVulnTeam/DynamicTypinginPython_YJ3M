  34 class ACMEProvider(CloudProvider):
  35     """ACME plugin. Sets up cloud resources for tests."""
  36     DOCKER_SIMULATOR_NAME = 'acme-simulator'
  37 
  38     def __init__(self, args):
  39         """
  40         :type args: TestConfig
  41         """
  42         super(ACMEProvider, self).__init__(args)
  43 
  44         # The simulator must be pinned to a specific version to guarantee CI passes with the version used.
  45         if os.environ.get('ANSIBLE_ACME_CONTAINER'):
  46             self.image = os.environ.get('ANSIBLE_ACME_CONTAINER')
  47         else:
  48             self.image = 'quay.io/ansible/acme-test-container:1.9.0'
  49         self.container_name = ''
  50 
  51     def _wait_for_service(self, protocol, acme_host, port, local_part, name):
  52         """Wait for an endpoint to accept connections."""
  53         if self.args.explain:
  54             return
  55 
  56         client = HttpClient(self.args, always=True, insecure=True)
  57         endpoint = '%s://%s:%d/%s' % (protocol, acme_host, port, local_part)
  58 
  59         for dummy in range(1, 30):
  60             display.info('Waiting for %s: %s' % (name, endpoint), verbosity=1)
  61 
  62             try:
  63                 client.get(endpoint)
  64                 return
  65             except SubprocessError:
  66                 pass
  67 
  68             time.sleep(1)
  69 
  70         raise ApplicationError('Timeout waiting for %s.' % name)
  71 
  72     def filter(self, targets, exclude):
  73         """Filter out the cloud tests when the necessary config and resources are not available.
  74         :type targets: tuple[TestTarget]
  75         :type exclude: list[str]
  76         """
  77         docker = find_executable('docker', required=False)
  78 
  79         if docker:
  80             return
  81 
  82         skip = 'cloud/%s/' % self.platform
  83         skipped = [target.name for target in targets if skip in target.aliases]
  84 
  85         if skipped:
  86             exclude.append(skip)
  87             display.warning('Excluding tests marked "%s" which require the "docker" command: %s'
  88                             % (skip.rstrip('/'), ', '.join(skipped)))
  89 
  90     def setup(self):
  91         """Setup the cloud resource before delegation and register a cleanup callback."""
  92         super(ACMEProvider, self).setup()
  93 
  94         if self._use_static_config():
  95             self._setup_static()
  96         else:
  97             self._setup_dynamic()
  98 
  99     def get_docker_run_options(self):
 100         """Get any additional options needed when delegating tests to a docker container.
 101         :rtype: list[str]
 102         """
 103         if self.managed:
 104             return ['--link', self.DOCKER_SIMULATOR_NAME]
 105 
 106         return []
 107 
 108     def cleanup(self):
 109         """Clean up the cloud resource and any temporary configuration files after tests complete."""
 110         if self.container_name:
 111             docker_rm(self.args, self.container_name)
 112 
 113         super(ACMEProvider, self).cleanup()
 114 
 115     def _setup_dynamic(self):
 116         """Create a ACME test container using docker."""
 117         container_id = get_docker_container_id()
 118 
 119         if container_id:
 120             display.info('Running in docker container: %s' % container_id, verbosity=1)
 121 
 122         self.container_name = self.DOCKER_SIMULATOR_NAME
 123 
 124         results = docker_inspect(self.args, self.container_name)
 125 
 126         if results and not results[0].get('State', {}).get('Running'):
 127             docker_rm(self.args, self.container_name)
 128             results = []
 129 
 130         if results:
 131             display.info('Using the existing ACME docker test container.', verbosity=1)
 132         else:
 133             display.info('Starting a new ACME docker test container.', verbosity=1)
 134 
 135             if not self.args.docker and not container_id:
 136                 # publish the simulator ports when not running inside docker
 137                 publish_ports = [
 138                     '-p', '5000:5000',  # control port for flask app in container
 139                     '-p', '14000:14000',  # Pebble ACME CA
 140                 ]
 141             else:
 142                 publish_ports = []
 143 
 144             if not os.environ.get('ANSIBLE_ACME_CONTAINER'):
 145                 docker_pull(self.args, self.image)
 146 
 147             docker_run(
 148                 self.args,
 149                 self.image,
 150                 ['-d', '--name', self.container_name] + publish_ports,
 151             )
 152 
 153         if self.args.docker:
 154             acme_host = self.DOCKER_SIMULATOR_NAME
 155             acme_host_ip = self._get_simulator_address()
 156         elif container_id:
 157             acme_host = self._get_simulator_address()
 158             acme_host_ip = acme_host
 159             display.info('Found ACME test container address: %s' % acme_host, verbosity=1)
 160         else:
 161             acme_host = 'localhost'
 162             acme_host_ip = acme_host
 163 
 164         self._set_cloud_config('acme_host', acme_host)
 165 
 166         self._wait_for_service('http', acme_host_ip, 5000, '', 'ACME controller')
 167         self._wait_for_service('https', acme_host_ip, 14000, 'dir', 'ACME CA endpoint')
 168 
 169     def _get_simulator_address(self):
 170         results = docker_inspect(self.args, self.container_name)
 171         ipaddress = results[0]['NetworkSettings']['IPAddress']
 172         return ipaddress
 173 
 174     def _setup_static(self):
 175         raise NotImplementedError()
