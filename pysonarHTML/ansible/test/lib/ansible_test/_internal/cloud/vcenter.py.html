  29 class VcenterProvider(CloudProvider):
  30     """VMware vcenter/esx plugin. Sets up cloud resources for tests."""
  31     DOCKER_SIMULATOR_NAME = 'vcenter-simulator'
  32 
  33     def __init__(self, args):
  34         """
  35         :type args: TestConfig
  36         """
  37         super(VcenterProvider, self).__init__(args)
  38 
  39         # The simulator must be pinned to a specific version to guarantee CI passes with the version used.
  40         if os.environ.get('ANSIBLE_VCSIM_CONTAINER'):
  41             self.image = os.environ.get('ANSIBLE_VCSIM_CONTAINER')
  42         else:
  43             self.image = 'quay.io/ansible/vcenter-test-container:1.7.0'
  44         self.container_name = ''
  45 
  46         # VMware tests can be run on govcsim or BYO with a static config file.
  47         # The simulator is the default if no config is provided.
  48         self.vmware_test_platform = os.environ.get('VMWARE_TEST_PLATFORM', 'govcsim')
  49         self.insecure = False
  50         self.proxy = None
  51         self.platform = 'vcenter'
  52 
  53     def filter(self, targets, exclude):
  54         """Filter out the cloud tests when the necessary config and resources are not available.
  55         :type targets: tuple[TestTarget]
  56         :type exclude: list[str]
  57         """
  58         if self.vmware_test_platform == 'govcsim' or (self.vmware_test_platform == '' and not os.path.isfile(self.config_static_path)):
  59             docker = find_executable('docker', required=False)
  60 
  61             if docker:
  62                 return
  63 
  64             skip = 'cloud/%s/' % self.platform
  65             skipped = [target.name for target in targets if skip in target.aliases]
  66 
  67             if skipped:
  68                 exclude.append(skip)
  69                 display.warning('Excluding tests marked "%s" which require the "docker" command or config (see "%s"): %s'
  70                                 % (skip.rstrip('/'), self.config_template_path, ', '.join(skipped)))
  71         elif self.vmware_test_platform == 'static':
  72             if os.path.isfile(self.config_static_path):
  73                 return
  74 
  75             super(VcenterProvider, self).filter(targets, exclude)
  76 
  77     def setup(self):
  78         """Setup the cloud resource before delegation and register a cleanup callback."""
  79         super(VcenterProvider, self).setup()
  80 
  81         self._set_cloud_config('vmware_test_platform', self.vmware_test_platform)
  82         if self.vmware_test_platform == 'govcsim':
  83             self._setup_dynamic_simulator()
  84             self.managed = True
  85         elif self.vmware_test_platform == 'static':
  86             self._use_static_config()
  87             self._setup_static()
  88         else:
  89             raise ApplicationError('Unknown vmware_test_platform: %s' % self.vmware_test_platform)
  90 
  91     def get_docker_run_options(self):
  92         """Get any additional options needed when delegating tests to a docker container.
  93         :rtype: list[str]
  94         """
  95         if self.managed:
  96             return ['--link', self.DOCKER_SIMULATOR_NAME]
  97 
  98         return []
  99 
 100     def cleanup(self):
 101         """Clean up the cloud resource and any temporary configuration files after tests complete."""
 102         if self.container_name:
 103             docker_rm(self.args, self.container_name)
 104 
 105         super(VcenterProvider, self).cleanup()

 107     def _setup_dynamic_simulator(self):
 108         """Create a vcenter simulator using docker."""
 109         container_id = get_docker_container_id()
 110 
 111         if container_id:
 112             display.info('Running in docker container: %s' % container_id, verbosity=1)
 113 
 114         self.container_name = self.DOCKER_SIMULATOR_NAME
 115 
 116         results = docker_inspect(self.args, self.container_name)
 117 
 118         if results and not results[0].get('State', {}).get('Running'):
 119             docker_rm(self.args, self.container_name)
 120             results = []
 121 
 122         if results:
 123             display.info('Using the existing vCenter simulator docker container.', verbosity=1)
 124         else:
 125             display.info('Starting a new vCenter simulator docker container.', verbosity=1)
 126 
 127             if not self.args.docker and not container_id:
 128                 # publish the simulator ports when not running inside docker
 129                 publish_ports = [
 130                     '-p', '1443:443',
 131                     '-p', '8080:8080',
 132                     '-p', '8989:8989',
 133                     '-p', '5000:5000',  # control port for flask app in simulator
 134                 ]
 135             else:
 136                 publish_ports = []
 137 
 138             if not os.environ.get('ANSIBLE_VCSIM_CONTAINER'):
 139                 docker_pull(self.args, self.image)
 140 
 141             docker_run(
 142                 self.args,
 143                 self.image,
 144                 ['-d', '--name', self.container_name] + publish_ports,
 145             )
 146 
 147         if self.args.docker:
 148             vcenter_hostname = self.DOCKER_SIMULATOR_NAME
 149         elif container_id:
 150             vcenter_hostname = self._get_simulator_address()
 151             display.info('Found vCenter simulator container address: %s' % vcenter_hostname, verbosity=1)
 152         else:
 153             vcenter_hostname = 'localhost'
 154 
 155         self._set_cloud_config('vcenter_hostname', vcenter_hostname)
 156 
 157     def _get_simulator_address(self):
 158         results = docker_inspect(self.args, self.container_name)
 159         ipaddress = results[0]['NetworkSettings']['IPAddress']
 160         return ipaddress
 161 
 162     def _setup_static(self):
 163         if not os.path.exists(self.config_static_path):
 164             raise ApplicationError('Configuration file does not exist: %s' % self.config_static_path)
 165 
 166         parser = ConfigParser({
 167             'vcenter_port': '443',
 168             'vmware_proxy_host': '',
 169             'vmware_proxy_port': '8080'})
 170         parser.read(self.config_static_path)
 171 
 172         if parser.get('DEFAULT', 'vmware_validate_certs').lower() in ('no', 'false'):
 173             self.insecure = True
 174         proxy_host = parser.get('DEFAULT', 'vmware_proxy_host')
 175         proxy_port = int(parser.get('DEFAULT', 'vmware_proxy_port'))
 176         if proxy_host and proxy_port:
 177             self.proxy = 'http://%s:%d' % (proxy_host, proxy_port)

 180 class VcenterEnvironment(CloudEnvironment):
 181     """VMware vcenter/esx environment plugin. Updates integration test environment after delegation."""
 182     def get_environment_config(self):
 183         """
 184         :rtype: CloudEnvironmentConfig
 185         """
 186         try:
 187             # We may be in a container, so we cannot just reach VMWARE_TEST_PLATFORM,
 188             # We do a try/except instead
 189             parser = ConfigParser()
 190             parser.read(self.config_path)  # static
 191 
 192             env_vars = dict()
 193             ansible_vars = dict(
 194                 resource_prefix=self.resource_prefix,
 195             )
 196             ansible_vars.update(dict(parser.items('DEFAULT', raw=True)))
 197         except KeyError:  # govcsim
 198             env_vars = dict(
 199                 VCENTER_HOSTNAME=self._get_cloud_config('vcenter_hostname'),
 200                 VCENTER_USERNAME='user',
 201                 VCENTER_PASSWORD='pass',
 202             )
 203 
 204             ansible_vars = dict(
 205                 vcsim=self._get_cloud_config('vcenter_hostname'),
 206                 vcenter_hostname=self._get_cloud_config('vcenter_hostname'),
 207                 vcenter_username='user',
 208                 vcenter_password='pass',
 209             )
 210             # Shippable starts ansible-test from withing an existing container,
 211             # and in this case, we don't have to change the vcenter port.
 212             if not self.args.docker and not get_docker_container_id():
 213                 ansible_vars['vcenter_port'] = '1443'
 214 
 215         for key, value in ansible_vars.items():
 216             if key.endswith('_password'):
 217                 display.sensitive.add(value)
 218 
 219         return CloudEnvironmentConfig(
 220             env_vars=env_vars,
 221             ansible_vars=ansible_vars,
 222             module_defaults={
 223                 'group/vmware': {
 224                     'hostname': ansible_vars['vcenter_hostname'],
 225                     'username': ansible_vars['vcenter_username'],
 226                     'password': ansible_vars['vcenter_password'],
 227                     'port': ansible_vars.get('vcenter_port', '443'),
 228                     'validate_certs': ansible_vars.get('vmware_validate_certs', 'no'),
 229                 },
 230             },
 231         )
