  59 class GalaxyProvider(CloudProvider):
  60     """Galaxy plugin.
  61 
  62     Sets up pulp (ansible-galaxy) servers for tests.
  63 
  64     The pulp source itself resides at: https://github.com/pulp/pulp-oci-images
  65     """
  66 
  67     def __init__(self, args):
  68         """
  69         :type args: TestConfig
  70         """
  71         super(GalaxyProvider, self).__init__(args)
  72 
  73         self.pulp = os.environ.get(
  74             'ANSIBLE_PULP_CONTAINER',
  75             'docker.io/pulp/pulp-galaxy-ng@sha256:69b4c4cba4908539b56c5592f40d282f938dd1bdf4de5a81e0a8d04ac3e6e326'
  76         )
  77 
  78         self.containers = []
  79 
  80     def filter(self, targets, exclude):
  81         """Filter out the tests with the necessary config and res unavailable.
  82 
  83         :type targets: tuple[TestTarget]
  84         :type exclude: list[str]
  85         """
  86         docker_cmd = 'docker'
  87         docker = find_executable(docker_cmd, required=False)
  88 
  89         if docker:
  90             return
  91 
  92         skip = 'cloud/%s/' % self.platform
  93         skipped = [target.name for target in targets if skip in target.aliases]
  94 
  95         if skipped:
  96             exclude.append(skip)
  97             display.warning('Excluding tests marked "%s" which require the "%s" command: %s'
  98                             % (skip.rstrip('/'), docker_cmd, ', '.join(skipped)))
  99 
 100     def setup(self):
 101         """Setup cloud resource before delegation and reg cleanup callback."""
 102         super(GalaxyProvider, self).setup()
 103 
 104         container_id = get_docker_container_id()
 105 
 106         if container_id:
 107             display.info('Running in docker container: %s' % container_id, verbosity=1)
 108 
 109         p_results = docker_inspect(self.args, 'ansible-ci-pulp')
 110 
 111         if p_results and not p_results[0].get('State', {}).get('Running'):
 112             docker_rm(self.args, 'ansible-ci-pulp')
 113             p_results = []
 114 
 115         display.info('%s ansible-ci-pulp docker container.'
 116                      % ('Using the existing' if p_results else 'Starting a new'),
 117                      verbosity=1)
 118 
 119         pulp_port = 80
 120 
 121         if not p_results:
 122             if self.args.docker or container_id:
 123                 publish_ports = []
 124             else:
 125                 # publish the simulator ports when not running inside docker
 126                 publish_ports = [
 127                     '-p', ':'.join((str(pulp_port),) * 2),
 128                 ]
 129 
 130             docker_pull(self.args, self.pulp)
 131 
 132             # Create the container, don't run it, we need to inject configs before it starts
 133             stdout, _dummy = docker_run(
 134                 self.args,
 135                 self.pulp,
 136                 ['--name', 'ansible-ci-pulp'] + publish_ports,
 137                 create_only=True
 138             )
 139 
 140             pulp_id = stdout.strip()
 141 
 142             try:
 143                 # Inject our settings.py file
 144                 with tempfile.NamedTemporaryFile(delete=False) as settings:
 145                     settings.write(SETTINGS)
 146                 docker_command(self.args, ['cp', settings.name, '%s:/etc/pulp/settings.py' % pulp_id])
 147             finally:
 148                 os.unlink(settings.name)
 149 
 150             try:
 151                 # Inject our settings.py file
 152                 with tempfile.NamedTemporaryFile(delete=False) as admin_pass:
 153                     admin_pass.write(SET_ADMIN_PASSWORD)
 154                 docker_command(self.args, ['cp', admin_pass.name, '%s:/etc/cont-init.d/111-postgres' % pulp_id])
 155             finally:
 156                 os.unlink(admin_pass.name)
 157 
 158             # Start the container
 159             docker_start(self.args, 'ansible-ci-pulp', [])
 160 
 161             self.containers.append('ansible-ci-pulp')
 162 
 163         if self.args.docker:
 164             pulp_host = 'ansible-ci-pulp'
 165         elif container_id:
 166             pulp_host = self._get_simulator_address('ansible-ci-pulp')
 167             display.info('Found Galaxy simulator container address: %s' % pulp_host, verbosity=1)
 168         else:
 169             pulp_host = 'localhost'
 170 
 171         self._set_cloud_config('PULP_HOST', pulp_host)
 172         self._set_cloud_config('PULP_PORT', str(pulp_port))
 173         self._set_cloud_config('PULP_USER', 'admin')
 174         self._set_cloud_config('PULP_PASSWORD', 'password')
 175 
 176     def get_docker_run_options(self):
 177         """Get additional options needed when delegating tests to a container.
 178 
 179         :rtype: list[str]
 180         """
 181         return ['--link', 'ansible-ci-pulp']  # if self.managed else []
 182 
 183     def cleanup(self):
 184         """Clean up the resource and temporary configs files after tests."""
 185         for container_name in self.containers:
 186             docker_rm(self.args, container_name)
 187 
 188         super(GalaxyProvider, self).cleanup()
 189 
 190     def _get_simulator_address(self, container_name):
 191         results = docker_inspect(self.args, container_name)
 192         ipaddress = results[0]['NetworkSettings']['IPAddress']
 193         return ipaddress

 196 class GalaxyEnvironment(CloudEnvironment):
 197     """Galaxy environment plugin.
 198 
 199     Updates integration test environment after delegation.
 200     """
 201     def get_environment_config(self):
 202         """
 203         :rtype: CloudEnvironmentConfig
 204         """
 205         pulp_user = self._get_cloud_config('PULP_USER')
 206         pulp_password = self._get_cloud_config('PULP_PASSWORD')
 207         pulp_host = self._get_cloud_config('PULP_HOST')
 208         pulp_port = self._get_cloud_config('PULP_PORT')
 209 
 210         return CloudEnvironmentConfig(
 211             ansible_vars=dict(
 212                 pulp_user=pulp_user,
 213                 pulp_password=pulp_password,
 214                 pulp_v2_server='http://%s:%s/pulp_ansible/galaxy/automation-hub/api/' % (pulp_host, pulp_port),
 215                 pulp_v3_server='http://%s:%s/pulp_ansible/galaxy/automation-hub/api/' % (pulp_host, pulp_port),
 216                 pulp_api='http://%s:%s' % (pulp_host, pulp_port),
 217                 galaxy_ng_server='http://%s:%s/api/galaxy/' % (pulp_host, pulp_port),
 218             ),
 219             env_vars=dict(
 220                 PULP_USER=pulp_user,
 221                 PULP_PASSWORD=pulp_password,
 222                 PULP_V2_SERVER='http://%s:%s/pulp_ansible/galaxy/automation-hub/api/' % (pulp_host, pulp_port),
 223                 PULP_V3_SERVER='http://%s:%s/pulp_ansible/galaxy/automation-hub/api/' % (pulp_host, pulp_port),
 224                 GALAXY_NG_SERVER='http://%s:%s/api/galaxy/' % (pulp_host, pulp_port),
 225             ),
 226         )
