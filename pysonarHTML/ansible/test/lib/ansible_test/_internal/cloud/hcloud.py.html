  23 class HcloudCloudProvider(CloudProvider):
  24     """Hetzner Cloud provider plugin. Sets up cloud resources before
  25        delegation.
  26     """
  27 
  28     def __init__(self, args):
  29         """
  30         :type args: TestConfig
  31         """
  32         super(HcloudCloudProvider, self).__init__(args)
  33 
  34     def filter(self, targets, exclude):
  35         """Filter out the cloud tests when the necessary config and resources are not available.
  36         :type targets: tuple[TestTarget]
  37         :type exclude: list[str]
  38         """
  39         if os.path.isfile(self.config_static_path):
  40             return
  41 
  42         aci = self._create_ansible_core_ci()
  43 
  44         if aci.available:
  45             return
  46 
  47         super(HcloudCloudProvider, self).filter(targets, exclude)
  48 
  49     def setup(self):
  50         """Setup the cloud resource before delegation and register a cleanup callback."""
  51         super(HcloudCloudProvider, self).setup()
  52 
  53         if not self._use_static_config():
  54             self._setup_dynamic()
  55 
  56     def _setup_dynamic(self):
  57         """Request Hetzner credentials through the Ansible Core CI service."""
  58         display.info('Provisioning %s cloud environment.' % self.platform, verbosity=1)
  59 
  60         config = self._read_config_template()
  61 
  62         aci = self._create_ansible_core_ci()
  63 
  64         response = aci.start()
  65 
  66         if not self.args.explain:
  67             token = response['hetzner']['token']
  68 
  69             display.sensitive.add(token)
  70             display.info('Hetzner Cloud Token: %s' % token, verbosity=1)
  71 
  72             values = dict(
  73                 TOKEN=token,
  74             )
  75 
  76             display.sensitive.add(values['TOKEN'])
  77 
  78             config = self._populate_config_template(config, values)
  79 
  80         self._write_config(config)
  81 
  82     def _create_ansible_core_ci(self):
  83         """
  84         :rtype: AnsibleCoreCI
  85         """
  86         return AnsibleCoreCI(self.args, 'hetzner', 'hetzner', persist=False, stage=self.args.remote_stage, provider=self.args.remote_provider)

  89 class HcloudCloudEnvironment(CloudEnvironment):
  90     """Hetzner Cloud cloud environment plugin. Updates integration test environment
  91        after delegation.
  92     """
  93 
  94     def get_environment_config(self):
  95         """
  96         :rtype: CloudEnvironmentConfig
  97         """
  98         parser = ConfigParser()
  99         parser.read(self.config_path)
 100 
 101         env_vars = dict(
 102             HCLOUD_TOKEN=parser.get('default', 'hcloud_api_token'),
 103         )
 104 
 105         display.sensitive.add(env_vars['HCLOUD_TOKEN'])
 106 
 107         ansible_vars = dict(
 108             hcloud_prefix=self.resource_prefix,
 109         )
 110 
 111         ansible_vars.update(dict((key.lower(), value) for key, value in env_vars.items()))
 112 
 113         return CloudEnvironmentConfig(
 114             env_vars=env_vars,
 115             ansible_vars=ansible_vars,
 116         )
