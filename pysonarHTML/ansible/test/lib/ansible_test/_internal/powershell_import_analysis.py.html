  25 def get_powershell_module_utils_imports(powershell_targets):
  26     """Return a dictionary of module_utils names mapped to sets of powershell file paths.
  27     :type powershell_targets: list[TestTarget]
  28     :rtype: dict[str, set[str]]
  29     """
  30 
  31     module_utils = enumerate_module_utils()
  32 
  33     imports_by_target_path = {}
  34 
  35     for target in powershell_targets:
  36         imports_by_target_path[target.path] = extract_powershell_module_utils_imports(target.path, module_utils)
  37 
  38     imports = dict([(module_util, set()) for module_util in module_utils])
  39 
  40     for target_path in imports_by_target_path:
  41         for module_util in imports_by_target_path[target_path]:
  42             imports[module_util].add(target_path)
  43 
  44     for module_util in sorted(imports):
  45         if not imports[module_util]:
  46             display.warning('No imports found which use the "%s" module_util.' % module_util)
  47 
  48     return imports

  74 def extract_powershell_module_utils_imports(path, module_utils):
  75     """Return a list of module_utils imports found in the specified source file.
  76     :type path: str
  77     :type module_utils: set[str]
  78     :rtype: set[str]
  79     """
  80     imports = set()
  81 
  82     code = read_text_file(path)
  83 
  84     if data_context().content.is_ansible and '# POWERSHELL_COMMON' in code:
  85         imports.add('Ansible.ModuleUtils.Legacy')
  86 
  87     lines = code.splitlines()
  88     line_number = 0
  89 
  90     for line in lines:
  91         line_number += 1
  92         match = re.search(r'(?i)^#\s*(?:requires\s+-module(?:s?)|ansiblerequires\s+-powershell)\s*((?:Ansible|ansible_collections|\.)\..+)', line)
  93 
  94         if not match:
  95             continue
  96 
  97         import_name = resolve_csharp_ps_util(match.group(1), path)
  98 
  99         if import_name in module_utils:
 100             imports.add(import_name)
 101         elif data_context().content.is_ansible or \
 102                 import_name.startswith('ansible_collections.%s' % data_context().content.prefix):
 103             display.warning('%s:%d Invalid module_utils import: %s' % (path, line_number, import_name))
 104 
 105     return imports
