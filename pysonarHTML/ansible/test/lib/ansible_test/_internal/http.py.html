  36 class HttpClient:
  37     """Make HTTP requests via curl."""
  38     def __init__(self, args, always=False, insecure=False, proxy=None):
  39         """
  40         :type args: CommonConfig
  41         :type always: bool
  42         :type insecure: bool
  43         """
  44         self.args = args
  45         self.always = always
  46         self.insecure = insecure
  47         self.proxy = proxy
  48 
  49         self.username = None
  50         self.password = None
  51 
  52     def get(self, url):
  53         """
  54         :type url: str
  55         :rtype: HttpResponse
  56         """
  57         return self.request('GET', url)
  58 
  59     def delete(self, url):
  60         """
  61         :type url: str
  62         :rtype: HttpResponse
  63         """
  64         return self.request('DELETE', url)
  65 
  66     def put(self, url, data=None, headers=None):
  67         """
  68         :type url: str
  69         :type data: str | None
  70         :type headers: dict[str, str] | None
  71         :rtype: HttpResponse
  72         """

  75     def request(self, method, url, data=None, headers=None):
  76         """
  77         :type method: str
  78         :type url: str
  79         :type data: str | None
  80         :type headers: dict[str, str] | None
  81         :rtype: HttpResponse
  82         """
  83         cmd = ['curl', '-s', '-S', '-i', '-X', method]
  84 
  85         if self.insecure:
  86             cmd += ['--insecure']
  87 
  88         if headers is None:
  89             headers = {}
  90 
  91         headers['Expect'] = ''  # don't send expect continue header
  92 
  93         if self.username:
  94             if self.password:
  95                 display.sensitive.add(self.password)
  96                 cmd += ['-u', '%s:%s' % (self.username, self.password)]
  97             else:
  98                 cmd += ['-u', self.username]
  99 
 100         for header in headers.keys():
 101             cmd += ['-H', '%s: %s' % (header, headers[header])]
 102 
 103         if data is not None:
 104             cmd += ['-d', data]
 105 
 106         if self.proxy:
 107             cmd += ['-x', self.proxy]
 108 
 109         cmd += [url]
 110 
 111         attempts = 0
 112         max_attempts = 3
 113         sleep_seconds = 3
 114 
 115         # curl error codes which are safe to retry (request never sent to server)
 116         retry_on_status = (
 117             6,  # CURLE_COULDNT_RESOLVE_HOST
 118         )
 119 
 120         stdout = ''
 121 
 122         while True:
 123             attempts += 1
 124 
 125             try:
 126                 stdout = run_command(self.args, cmd, capture=True, always=self.always, cmd_verbosity=2)[0]
 127                 break
 128             except SubprocessError as ex:
 129                 if ex.status in retry_on_status and attempts < max_attempts:
 130                     display.warning(u'%s' % ex)
 131                     time.sleep(sleep_seconds)
 132                     continue
 133 
 134                 raise
 135 
 136         if self.args.explain and not self.always:
 137             return HttpResponse(method, url, 200, '')
 138 
 139         header, body = stdout.split('\r\n\r\n', 1)
 140 
 141         response_headers = header.split('\r\n')
 142         first_line = response_headers[0]
 143         http_response = first_line.split(' ')
 144         status_code = int(http_response[1])
 145 
 146         return HttpResponse(method, url, status_code, body)

 149 class HttpResponse:
 150     """HTTP response from curl."""
 151     def __init__(self, method, url, status_code, response):
 152         """
 153         :type method: str
 154         :type url: str
 155         :type status_code: int
 156         :type response: str
 157         """
 158         self.method = method
 159         self.url = url
 160         self.status_code = status_code
 161         self.response = response
 162 
 163     def json(self):
 164         """
 165         :rtype: any
 166         """
 167         try:
 168             return json.loads(self.response)
 169         except ValueError:
 170             raise HttpError(self.status_code, 'Cannot parse response to %s %s as JSON:\n%s' % (self.method, self.url, self.response))
