  65     def wait(self):
  66         """Wait for instance to respond to ansible ping."""
  67         extra_vars = [
  68             'ansible_connection=winrm',
  69             'ansible_host=%s' % self.core_ci.connection.hostname,
  70             'ansible_user=%s' % self.core_ci.connection.username,
  71             'ansible_password=%s' % self.core_ci.connection.password,
  72             'ansible_port=%s' % self.core_ci.connection.port,
  73             'ansible_winrm_server_cert_validation=ignore',
  74         ]
  75 
  76         name = 'windows_%s' % self.core_ci.version
  77 
  78         env = ansible_environment(self.core_ci.args)
  79         cmd = ['ansible', '-m', 'ansible.windows.win_ping', '-i', '%s,' % name, name, '-e', ' '.join(extra_vars)]
  80 
  81         for dummy in range(1, 120):
  82             try:
  83                 intercept_command(self.core_ci.args, cmd, 'ping', env=env, disable_coverage=True)
  84                 return
  85             except SubprocessError:
  86                 time.sleep(10)
  87 
  88         raise ApplicationError('Timeout waiting for %s/%s instance %s.' %
  89                                (self.core_ci.platform, self.core_ci.version, self.core_ci.instance_id))

 105     def ssh(self, command, options=None, force_pty=True):
 106         """
 107         :type command: str | list[str]
 108         :type options: list[str] | None
 109         :type force_pty: bool
 110         """
 111         if not options:
 112             options = []
 113         if force_pty:
 114             options.append('-tt')
 115 
 116         if isinstance(command, list):
 117             command = ' '.join(cmd_quote(c) for c in command)
 118 
 119         run_command(self.core_ci.args,
 120                     ['ssh', '-q'] + self.ssh_args +
 121                     options +
 122                     ['-p', '22',
 123                      '%s@%s' % (self.core_ci.connection.username, self.core_ci.connection.hostname)] +
 124                     [command])
 125 
 126     def scp(self, src, dst):
 127         """
 128         :type src: str
 129         :type dst: str
 130         """
 131         for dummy in range(1, 10):
 132             try:
 133                 run_command(self.core_ci.args,
 134                             ['scp'] + self.ssh_args +
 135                             ['-P', '22', '-q', '-r', src, dst])
 136                 return
 137             except SubprocessError:
 138                 time.sleep(10)
 139 
 140         raise ApplicationError('Failed transfer: %s -> %s' % (src, dst))

 151     def wait(self):
 152         """Wait for instance to respond to ansible ping."""
 153         settings = get_network_settings(self.core_ci.args, self.core_ci.platform, self.core_ci.version)
 154 
 155         extra_vars = [
 156             'ansible_host=%s' % self.core_ci.connection.hostname,
 157             'ansible_port=%s' % self.core_ci.connection.port,
 158             'ansible_ssh_private_key_file=%s' % self.core_ci.ssh_key.key,
 159         ] + [
 160             '%s=%s' % (key, value) for key, value in settings.inventory_vars.items()
 161         ]
 162 
 163         name = '%s-%s' % (self.core_ci.platform, self.core_ci.version.replace('.', '-'))
 164 
 165         env = ansible_environment(self.core_ci.args)
 166         cmd = [
 167             'ansible',
 168             '-m', '%s%s_command' % (settings.collection + '.' if settings.collection else '', self.core_ci.platform),
 169             '-a', 'commands=?',
 170             '-u', self.core_ci.connection.username,
 171             '-i', '%s,' % name,
 172             '-e', ' '.join(extra_vars),
 173             name,
 174         ]
 175 
 176         for dummy in range(1, 90):
 177             try:
 178                 intercept_command(self.core_ci.args, cmd, 'ping', env=env, disable_coverage=True)
 179                 return
 180             except SubprocessError:
 181                 time.sleep(10)
 182 
 183         raise ApplicationError('Timeout waiting for %s/%s instance %s.' %
 184                                (self.core_ci.platform, self.core_ci.version, self.core_ci.instance_id))

 187 class ManagePosixCI:
 188     """Manage access to a POSIX instance provided by Ansible Core CI."""
 189     def __init__(self, core_ci):
 190         """
 191         :type core_ci: AnsibleCoreCI
 192         """
 193         self.core_ci = core_ci
 194         self.ssh_args = ['-i', self.core_ci.ssh_key.key]
 195 
 196         ssh_options = dict(
 197             BatchMode='yes',
 198             StrictHostKeyChecking='no',
 199             UserKnownHostsFile='/dev/null',
 200             ServerAliveInterval=15,
 201             ServerAliveCountMax=4,
 202         )
 203 
 204         for ssh_option in sorted(ssh_options):
 205             self.ssh_args += ['-o', '%s=%s' % (ssh_option, ssh_options[ssh_option])]
 206 
 207         if self.core_ci.platform == 'freebsd':
 208             if self.core_ci.provider == 'aws':
 209                 self.become = ['su', '-l', 'root', '-c']
 210             elif self.core_ci.provider == 'azure':
 211                 self.become = ['sudo', '-in', 'sh', '-c']
 212             else:
 213                 raise NotImplementedError('provider %s has not been implemented' % self.core_ci.provider)
 214         elif self.core_ci.platform == 'osx':
 215             self.become = ['sudo', '-in', 'PATH=/usr/local/bin:$PATH']
 216         elif self.core_ci.platform == 'rhel' or self.core_ci.platform == 'centos':
 217             self.become = ['sudo', '-in', 'bash', '-c']
 218         elif self.core_ci.platform in ['aix', 'ibmi']:
 219             self.become = []
 220 
 221     def setup(self, python_version):
 222         """Start instance and wait for it to become ready and respond to an ansible ping.
 223         :type python_version: str
 224         :rtype: str
 225         """
 226         pwd = self.wait()
 227 
 228         display.info('Remote working directory: %s' % pwd, verbosity=1)
 229 
 230         if isinstance(self.core_ci.args, ShellConfig):
 231             if self.core_ci.args.raw:
 232                 return pwd
 233 
 234         self.configure(python_version)
 235         self.upload_source()
 236 
 237         return pwd
 238 
 239     def wait(self):  # type: () -> str
 240         """Wait for instance to respond to SSH."""
 241         for dummy in range(1, 90):
 242             try:
 243                 stdout = self.ssh('pwd', capture=True)[0]
 244 
 245                 if self.core_ci.args.explain:
 246                     return '/pwd'
 247 
 248                 pwd = stdout.strip().splitlines()[-1]
 249 
 250                 if not pwd.startswith('/'):
 251                     raise Exception('Unexpected current working directory "%s" from "pwd" command output:\n%s' % (pwd, stdout))
 252 
 253                 return pwd
 254             except SubprocessError:
 255                 time.sleep(10)
 256 
 257         raise ApplicationError('Timeout waiting for %s/%s instance %s.' %
 258                                (self.core_ci.platform, self.core_ci.version, self.core_ci.instance_id))
 259 
 260     def configure(self, python_version):
 261         """Configure remote host for testing.
 262         :type python_version: str
 263         """
 264         self.upload(os.path.join(ANSIBLE_TEST_DATA_ROOT, 'setup', 'remote.sh'), '/tmp')
 265         self.ssh('chmod +x /tmp/remote.sh && /tmp/remote.sh %s %s' % (self.core_ci.platform, python_version))
 266 
 267     def upload_source(self):
 268         """Upload and extract source."""
 269         with tempfile.NamedTemporaryFile(prefix='ansible-source-', suffix='.tgz') as local_source_fd:
 270             remote_source_dir = '/tmp'
 271             remote_source_path = os.path.join(remote_source_dir, os.path.basename(local_source_fd.name))
 272 
 273             create_payload(self.core_ci.args, local_source_fd.name)
 274 
 275             self.upload(local_source_fd.name, remote_source_dir)
 276             # AIX does not provide the GNU tar version, leading to parameters
 277             # being different and -z not being recognized. This pattern works
 278             # with both versions of tar.
 279             self.ssh(
 280                 'rm -rf ~/ansible ~/ansible_collections && cd ~/ && gunzip --stdout %s | tar oxf - && rm %s' %
 281                 (remote_source_path, remote_source_path)
 282             )

 298     def ssh(self, command, options=None, capture=False):
 299         """
 300         :type command: str | list[str]
 301         :type options: list[str] | None
 302         :type capture: bool
 303         :rtype: str | None, str | None
 304         """
 305         if not options:
 306             options = []
 307 
 308         if isinstance(command, list):
 309             command = ' '.join(cmd_quote(c) for c in command)
 310 
 311         command = cmd_quote(command) if self.become else command
 312         return run_command(self.core_ci.args,
 313                            ['ssh', '-tt', '-q'] + self.ssh_args +
 314                            options +
 315                            ['-p', str(self.core_ci.connection.port),
 316                             '%s@%s' % (self.core_ci.connection.username, self.core_ci.connection.hostname)] +
 317                            self.become + [command], capture=capture)
 318
