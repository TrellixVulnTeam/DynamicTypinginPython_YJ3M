  66 def setup_common_temp_dir(args, path):
  67     """
  68     :type args: IntegrationConfig
  69     :type path: str
  70     """
  71     if args.explain:
  72         return
  73 
  74     os.mkdir(path)
  75     os.chmod(path, MODE_DIRECTORY)
  76 
  77     if args.coverage:
  78         coverage_config_path = os.path.join(path, COVERAGE_CONFIG_NAME)
  79 
  80         coverage_config = generate_coverage_config(args)
  81 
  82         write_text_file(coverage_config_path, coverage_config)
  83 
  84         os.chmod(coverage_config_path, MODE_FILE)
  85 
  86         coverage_output_path = os.path.join(path, ResultType.COVERAGE.name)
  87 
  88         os.mkdir(coverage_output_path)
  89         os.chmod(coverage_output_path, MODE_DIRECTORY_WRITE)

  92 def generate_dependency_map(integration_targets):
  93     """
  94     :type integration_targets: list[IntegrationTarget]
  95     :rtype: dict[str, set[IntegrationTarget]]
  96     """
  97     targets_dict = dict((target.name, target) for target in integration_targets)
  98     target_dependencies = analyze_integration_target_dependencies(integration_targets)
  99     dependency_map = {}
 100 
 101     invalid_targets = set()
 102 
 103     for dependency, dependents in target_dependencies.items():
 104         dependency_target = targets_dict.get(dependency)
 105 
 106         if not dependency_target:
 107             invalid_targets.add(dependency)
 108             continue
 109 
 110         for dependent in dependents:
 111             if dependent not in dependency_map:
 112                 dependency_map[dependent] = set()
 113 
 114             dependency_map[dependent].add(dependency_target)
 115 
 116     if invalid_targets:
 117         raise ApplicationError('Non-existent target dependencies: %s' % ', '.join(sorted(invalid_targets)))
 118 
 119     return dependency_map

 122 def get_files_needed(target_dependencies):
 123     """
 124     :type target_dependencies: list[IntegrationTarget]
 125     :rtype: list[str]
 126     """
 127     files_needed = []
 128 
 129     for target_dependency in target_dependencies:
 130         files_needed += target_dependency.needs_file
 131 
 132     files_needed = sorted(set(files_needed))
 133 
 134     invalid_paths = [path for path in files_needed if not os.path.isfile(path)]
 135 
 136     if invalid_paths:
 137         raise ApplicationError('Invalid "needs/file/*" aliases:\n%s' % '\n'.join(invalid_paths))
 138 
 139     return files_needed

 152 def get_inventory_relative_path(args):  # type: (IntegrationConfig) -> str
 153     """Return the inventory path used for the given integration configuration relative to the content root."""
 154     inventory_names = {
 155         PosixIntegrationConfig: 'inventory',
 156         WindowsIntegrationConfig: 'inventory.winrm',
 157         NetworkIntegrationConfig: 'inventory.networking',
 158     }  # type: t.Dict[t.Type[IntegrationConfig], str]
 159 
 160     return os.path.join(data_context().content.integration_path, inventory_names[type(args)])

 163 def delegate_inventory(args, inventory_path_src):  # type: (IntegrationConfig, str) -> None
 164     """Make the given inventory available during delegation."""
 165     if isinstance(args, PosixIntegrationConfig):
 166         return
 167 
 168     def inventory_callback(files):  # type: (t.List[t.Tuple[str, str]]) -> None
 169         """
 170         Add the inventory file to the payload file list.
 171         This will preserve the file during delegation even if it is ignored or is outside the content and install roots.
 172         """
 173         inventory_path = get_inventory_relative_path(args)
 174         inventory_tuple = inventory_path_src, inventory_path
 175 
 176         if os.path.isfile(inventory_path_src) and inventory_tuple not in files:
 177             originals = [item for item in files if item[1] == inventory_path]
 178 
 179             if originals:
 180                 for original in originals:
 181                     files.remove(original)
 182 
 183                 display.warning('Overriding inventory file "%s" with "%s".' % (inventory_path, inventory_path_src))
 184             else:
 185                 display.notice('Sourcing inventory file "%s" from "%s".' % (inventory_path, inventory_path_src))
 186 
 187             files.append(inventory_tuple)
 188 
 189     data_context().register_payload_callback(inventory_callback)

 192 @contextlib.contextmanager
 193 def integration_test_environment(args, target, inventory_path_src):
 194     """
 195     :type args: IntegrationConfig
 196     :type target: IntegrationTarget
 197     :type inventory_path_src: str
 198     """
 199     ansible_config_src = args.get_ansible_config()
 200     ansible_config_relative = os.path.join(data_context().content.integration_path, '%s.cfg' % args.command)
 201 
 202     if args.no_temp_workdir or 'no/temp_workdir/' in target.aliases:
 203         display.warning('Disabling the temp work dir is a temporary debugging feature that may be removed in the future without notice.')
 204 
 205         integration_dir = os.path.join(data_context().content.root, data_context().content.integration_path)
 206         targets_dir = os.path.join(data_context().content.root, data_context().content.integration_targets_path)
 207         inventory_path = inventory_path_src
 208         ansible_config = ansible_config_src
 209         vars_file = os.path.join(data_context().content.root, data_context().content.integration_vars_path)
 210 
 211         yield IntegrationEnvironment(integration_dir, targets_dir, inventory_path, ansible_config, vars_file)
 212         return
 213 
 214     # When testing a collection, the temporary directory must reside within the collection.
 215     # This is necessary to enable support for the default collection for non-collection content (playbooks and roles).
 216     root_temp_dir = os.path.join(ResultType.TMP.path, 'integration')
 217 
 218     prefix = '%s-' % target.name
 219     suffix = u'-\u00c5\u00d1\u015a\u00cc\u03b2\u0141\u00c8'
 220 
 221     if args.no_temp_unicode or 'no/temp_unicode/' in target.aliases:
 222         display.warning('Disabling unicode in the temp work dir is a temporary debugging feature that may be removed in the future without notice.')
 223         suffix = '-ansible'
 224 
 225     if args.explain:
 226         temp_dir = os.path.join(root_temp_dir, '%stemp%s' % (prefix, suffix))
 227     else:
 228         make_dirs(root_temp_dir)
 229         temp_dir = tempfile.mkdtemp(prefix=prefix, suffix=suffix, dir=root_temp_dir)
 230 
 231     try:
 232         display.info('Preparing temporary directory: %s' % temp_dir, verbosity=2)
 233 
 234         inventory_relative_path = get_inventory_relative_path(args)
 235         inventory_path = os.path.join(temp_dir, inventory_relative_path)
 236 
 237         cache = IntegrationCache(args)
 238 
 239         target_dependencies = sorted([target] + list(cache.dependency_map.get(target.name, set())))
 240 
 241         files_needed = get_files_needed(target_dependencies)
 242 
 243         integration_dir = os.path.join(temp_dir, data_context().content.integration_path)
 244         targets_dir = os.path.join(temp_dir, data_context().content.integration_targets_path)
 245         ansible_config = os.path.join(temp_dir, ansible_config_relative)
 246 
 247         vars_file_src = os.path.join(data_context().content.root, data_context().content.integration_vars_path)
 248         vars_file = os.path.join(temp_dir, data_context().content.integration_vars_path)
 249 
 250         file_copies = [
 251             (ansible_config_src, ansible_config),
 252             (inventory_path_src, inventory_path),
 253         ]
 254 
 255         if os.path.exists(vars_file_src):
 256             file_copies.append((vars_file_src, vars_file))
 257 
 258         file_copies += [(path, os.path.join(temp_dir, path)) for path in files_needed]
 259 
 260         integration_targets_relative_path = data_context().content.integration_targets_path
 261 
 262         directory_copies = [
 263             (
 264                 os.path.join(integration_targets_relative_path, target.relative_path),
 265                 os.path.join(temp_dir, integration_targets_relative_path, target.relative_path)
 266             )
 267             for target in target_dependencies
 268         ]
 269 
 270         directory_copies = sorted(set(directory_copies))
 271         file_copies = sorted(set(file_copies))
 272 
 273         if not args.explain:
 274             make_dirs(integration_dir)
 275 
 276         for dir_src, dir_dst in directory_copies:
 277             display.info('Copying %s/ to %s/' % (dir_src, dir_dst), verbosity=2)
 278 
 279             if not args.explain:
 280                 shutil.copytree(to_bytes(dir_src), to_bytes(dir_dst), symlinks=True)
 281 
 282         for file_src, file_dst in file_copies:
 283             display.info('Copying %s to %s' % (file_src, file_dst), verbosity=2)
 284 
 285             if not args.explain:
 286                 make_dirs(os.path.dirname(file_dst))
 287                 shutil.copy2(file_src, file_dst)
 288 
 289         yield IntegrationEnvironment(integration_dir, targets_dir, inventory_path, ansible_config, vars_file)
 290     finally:
 291         if not args.explain:
 292             shutil.rmtree(temp_dir)

 295 @contextlib.contextmanager
 296 def integration_test_config_file(args, env_config, integration_dir):
 297     """
 298     :type args: IntegrationConfig
 299     :type env_config: CloudEnvironmentConfig
 300     :type integration_dir: str
 301     """
 302     if not env_config:
 303         yield None
 304         return
 305 
 306     config_vars = (env_config.ansible_vars or {}).copy()
 307 
 308     config_vars.update(dict(
 309         ansible_test=dict(
 310             environment=env_config.env_vars,
 311             module_defaults=env_config.module_defaults,
 312         )
 313     ))
 314 
 315     config_file = json.dumps(config_vars, indent=4, sort_keys=True)
 316 
 317     with named_temporary_file(args, 'config-file-', '.json', integration_dir, config_file) as path:
 318         filename = os.path.relpath(path, integration_dir)
 319 
 320         display.info('>>> Config File: %s\n%s' % (filename, config_file), verbosity=3)
 321 
 322         yield path

 335 class IntegrationCache(CommonCache):
 336     """Integration cache."""
 337     @property
 338     def integration_targets(self):
 339         """
 340         :rtype: list[IntegrationTarget]
 341         """
 342         return self.get('integration_targets', lambda: list(walk_integration_targets()))
 343 
 344     @property
 345     def dependency_map(self):
 346         """
 347         :rtype: dict[str, set[IntegrationTarget]]
