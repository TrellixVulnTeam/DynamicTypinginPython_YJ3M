  89     def detect_changes(self, args):  # type: (TestConfig) -> t.Optional[t.List[str]]
  90         """Initialize change detection."""
  91         result = AzurePipelinesChanges(args)
  92 
  93         if result.is_pr:
  94             job_type = 'pull request'
  95         else:
  96             job_type = 'merge commit'
  97 
  98         display.info('Processing %s for branch %s commit %s' % (job_type, result.branch, result.commit))
  99 
 100         if not args.metadata.changes:
 101             args.metadata.populate_changes(result.diff)
 102 
 103         if result.paths is None:
 104             # There are several likely causes of this:
 105             # - First run on a new branch.
 106             # - Too many pull requests passed since the last merge run passed.
 107             display.warning('No successful commit found. All tests will be executed.')
 108 
 109         return result.paths

 147 class AzurePipelinesAuthHelper(CryptographyAuthHelper):
 148     """
 149     Authentication helper for Azure Pipelines.
 150     Based on cryptography since it is provided by the default Azure Pipelines environment.
 151     """
 152     def publish_public_key(self, public_key_pem):  # type: (str) -> None
 153         """Publish the given public key."""
 154         # the temporary file cannot be deleted because we do not know when the agent has processed it
 155         with tempfile.NamedTemporaryFile(prefix='public-key-', suffix='.pem', delete=False) as public_key_file:
 156             public_key_file.write(to_bytes(public_key_pem))
 157             public_key_file.flush()
 158 
 159         # make the agent aware of the public key by declaring it as an attachment
 160         vso_add_attachment('ansible-core-ci', 'public-key.pem', public_key_file.name)

 163 class AzurePipelinesChanges:
 164     """Change information for an Azure Pipelines build."""
 165     def __init__(self, args):  # type: (CommonConfig) -> None
 166         self.args = args
 167         self.git = Git()
 168 
 169         try:
 170             self.org_uri = os.environ['SYSTEM_COLLECTIONURI']  # ex: https://dev.azure.com/{org}/
 171             self.project = os.environ['SYSTEM_TEAMPROJECT']
 172             self.repo_type = os.environ['BUILD_REPOSITORY_PROVIDER']  # ex: GitHub
 173             self.source_branch = os.environ['BUILD_SOURCEBRANCH']
 174             self.source_branch_name = os.environ['BUILD_SOURCEBRANCHNAME']
 175             self.pr_branch_name = os.environ.get('SYSTEM_PULLREQUEST_TARGETBRANCH')
 176         except KeyError as ex:
 177             raise MissingEnvironmentVariable(name=ex.args[0])
 178 
 179         if self.source_branch.startswith('refs/tags/'):
 180             raise ChangeDetectionNotSupported('Change detection is not supported for tags.')
 181 
 182         self.org = self.org_uri.strip('/').split('/')[-1]
 183         self.is_pr = self.pr_branch_name is not None
 184 
 185         if self.is_pr:
 186             # HEAD is a merge commit of the PR branch into the target branch
 187             # HEAD^1 is HEAD of the target branch (first parent of merge commit)
 188             # HEAD^2 is HEAD of the PR branch (second parent of merge commit)
 189             # see: https://git-scm.com/docs/gitrevisions
 190             self.branch = self.pr_branch_name
 191             self.base_commit = 'HEAD^1'
 192             self.commit = 'HEAD^2'
 193         else:
 194             commits = self.get_successful_merge_run_commits()
 195 
 196             self.branch = self.source_branch_name
 197             self.base_commit = self.get_last_successful_commit(commits)
 198             self.commit = 'HEAD'
 199 
 200         self.commit = self.git.run_git(['rev-parse', self.commit]).strip()
 201 
 202         if self.base_commit:
 203             self.base_commit = self.git.run_git(['rev-parse', self.base_commit]).strip()
 204 
 205             # <rev1>...<rev2>
 206             # Include commits that are reachable from <rev2> but exclude those that are reachable from <rev1>.
 207             # see: https://git-scm.com/docs/gitrevisions
 208             dot_range = '%s..%s' % (self.base_commit, self.commit)
 209 
 210             self.paths = sorted(self.git.get_diff_names([dot_range]))
 211             self.diff = self.git.get_diff([dot_range])
 212         else:
 213             self.paths = None  # act as though change detection not enabled, do not filter targets
 214             self.diff = []
 215 
 216     def get_successful_merge_run_commits(self):  # type: () -> t.Set[str]
 217         """Return a set of recent successsful merge commits from Azure Pipelines."""
 218         parameters = dict(
 219             maxBuildsPerDefinition=100,  # max 5000
 220             queryOrder='queueTimeDescending',  # assumes under normal circumstances that later queued jobs are for later commits
 221             resultFilter='succeeded',
 222             reasonFilter='batchedCI',  # may miss some non-PR reasons, the alternative is to filter the list after receiving it
 223             repositoryType=self.repo_type,
 224             repositoryId='%s/%s' % (self.org, self.project),
 225         )
 226 
 227         url = '%s%s/build/builds?%s' % (self.org_uri, self.project, urlencode(parameters))
 228 
 229         http = HttpClient(self.args)
 230         response = http.get(url)
 231 
 232         # noinspection PyBroadException
 233         try:
 234             result = response.json()
 235         except Exception:  # pylint: disable=broad-except
 236             # most likely due to a private project, which returns an HTTP 203 response with HTML
 237             display.warning('Unable to find project. Cannot determine changes. All tests will be executed.')
 238             return set()
 239 
 240         commits = set(build['sourceVersion'] for build in result['value'])
 241 
 242         return commits
 243 
 244     def get_last_successful_commit(self, commits):  # type: (t.Set[str]) -> t.Optional[str]
 245         """Return the last successful commit from git history that is found in the given commit list, or None."""
 246         commit_history = self.git.get_rev_list(max_count=100)
 247         ordered_successful_commits = [commit for commit in commit_history if commit in commits]
 248         last_successful_commit = ordered_successful_commits[0] if ordered_successful_commits else None
 249         return last_successful_commit
