  71     def detect_changes(self, args):  # type: (TestConfig) -> t.Optional[t.List[str]]
  72         """Initialize change detection."""
  73         result = LocalChanges(args)
  74 
  75         display.info('Detected branch %s forked from %s at commit %s' % (
  76             result.current_branch, result.fork_branch, result.fork_point))
  77 
  78         if result.untracked and not args.untracked:
  79             display.warning('Ignored %s untracked file(s). Use --untracked to include them.' %
  80                             len(result.untracked))
  81 
  82         if result.committed and not args.committed:
  83             display.warning('Ignored %s committed change(s). Omit --ignore-committed to include them.' %
  84                             len(result.committed))
  85 
  86         if result.staged and not args.staged:
  87             display.warning('Ignored %s staged change(s). Omit --ignore-staged to include them.' %
  88                             len(result.staged))
  89 
  90         if result.unstaged and not args.unstaged:
  91             display.warning('Ignored %s unstaged change(s). Omit --ignore-unstaged to include them.' %
  92                             len(result.unstaged))
  93 
  94         names = set()
  95 
  96         if args.tracked:
  97             names |= set(result.tracked)
  98         if args.untracked:
  99             names |= set(result.untracked)
 100         if args.committed:
 101             names |= set(result.committed)
 102         if args.staged:
 103             names |= set(result.staged)
 104         if args.unstaged:
 105             names |= set(result.unstaged)
 106 
 107         if not args.metadata.changes:
 108             args.metadata.populate_changes(result.diff)
 109 
 110             for path in result.untracked:
 111                 if is_binary_file(path):
 112                     args.metadata.changes[path] = ((0, 0),)
 113                     continue
 114 
 115                 line_count = len(read_text_file(path).splitlines())
 116 
 117                 args.metadata.changes[path] = ((1, line_count),)
 118 
 119         return sorted(names)

 121     def supports_core_ci_auth(self, context):  # type: (AuthContext) -> bool
 122         """Return True if Ansible Core CI is supported."""
 123         path = self._get_aci_key_path(context)

 126     def prepare_core_ci_auth(self, context):  # type: (AuthContext) -> t.Dict[str, t.Any]
 127         """Return authentication details for Ansible Core CI."""
 128         path = self._get_aci_key_path(context)
 129         auth_key = read_text_file(path).strip()
 130 
 131         request = dict(
 132             key=auth_key,
 133             nonce=None,
 134         )
 135 
 136         auth = dict(
 137             remote=request,
 138         )
 139 
 140         return auth

 165 class LocalChanges:
 166     """Change information for local work."""
 167     def __init__(self, args):  # type: (TestConfig) -> None
 168         self.args = args
 169         self.git = Git()
 170 
 171         self.current_branch = self.git.get_branch()
 172 
 173         if self.is_official_branch(self.current_branch):
 174             raise InvalidBranch(branch=self.current_branch,
 175                                 reason='Current branch is not a feature branch.')
 176 
 177         self.fork_branch = None
 178         self.fork_point = None
 179 
 180         self.local_branches = sorted(self.git.get_branches())
 181         self.official_branches = sorted([b for b in self.local_branches if self.is_official_branch(b)])
 182 
 183         for self.fork_branch in self.official_branches:
 184             try:
 185                 self.fork_point = self.git.get_branch_fork_point(self.fork_branch)
 186                 break
 187             except SubprocessError:
 188                 pass
 189 
 190         if self.fork_point is None:
 191             raise ApplicationError('Unable to auto-detect fork branch and fork point.')
 192 
 193         # tracked files (including unchanged)
 194         self.tracked = sorted(self.git.get_file_names(['--cached']))
 195         # untracked files (except ignored)
 196         self.untracked = sorted(self.git.get_file_names(['--others', '--exclude-standard']))
 197         # tracked changes (including deletions) committed since the branch was forked
 198         self.committed = sorted(self.git.get_diff_names([self.fork_point, 'HEAD']))
 199         # tracked changes (including deletions) which are staged
 200         self.staged = sorted(self.git.get_diff_names(['--cached']))
 201         # tracked changes (including deletions) which are not staged
 202         self.unstaged = sorted(self.git.get_diff_names([]))
 203         # diff of all tracked files from fork point to working copy
 204         self.diff = self.git.get_diff([self.fork_point])
 205 
 206     def is_official_branch(self, name):  # type: (str) -> bool
 207         """Return True if the given branch name an official branch for development or releases."""
 208         if self.args.base_branch:
 209             return name == self.args.base_branch
 210 
 211         if name == 'devel':
 212             return True
 213 
 214         if re.match(r'^stable-[0-9]+\.[0-9]+$', name):
 215             return True
 216 
 217         return False
