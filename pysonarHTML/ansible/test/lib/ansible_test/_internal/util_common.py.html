 146 def get_parameterized_completion(cache, name):
 147     """
 148     :type cache: dict[str, dict[str, str]]
 149     :type name: str
 150     :rtype: dict[str, dict[str, str]]
 151     """
 152     if not cache:
 153         if data_context().content.collection:
 154             context = 'collection'
 155         else:
 156             context = 'ansible-base'
 157 
 158         images = read_lines_without_comments(os.path.join(ANSIBLE_TEST_DATA_ROOT, 'completion', '%s.txt' % name), remove_blank_lines=True)
 159 
 160         cache.update(dict(kvp for kvp in [parse_parameterized_completion(i) for i in images] if kvp and kvp[1].get('context', context) == context))
 161 
 162     return cache

 165 def parse_parameterized_completion(value):  # type: (str) -> t.Optional[t.Tuple[str, t.Dict[str, str]]]
 166     """Parse the given completion entry, returning the entry name and a dictionary of key/value settings."""
 167     values = value.split()
 168 
 169     if not values:
 170         return None
 171 
 172     name = values[0]
 173     data = dict((kvp[0], kvp[1] if len(kvp) > 1 else '') for kvp in [item.split('=', 1) for item in values[1:]])
 174 
 175     return name, data

 205 def handle_layout_messages(messages):  # type: (t.Optional[LayoutMessages]) -> None
 206     """Display the given layout messages."""
 207     if not messages:
 208         return
 209 
 210     for message in messages.info:
 211         display.info(message, verbosity=1)
 212 
 213     for message in messages.warning:
 214         display.warning(message)
 215

 217         raise ApplicationError('\n'.join(messages.error))

 220 @contextlib.contextmanager
 221 def named_temporary_file(args, prefix, suffix, directory, content):
 222     """
 223     :param args: CommonConfig
 224     :param prefix: str
 225     :param suffix: str
 226     :param directory: str
 227     :param content: str | bytes | unicode
 228     :rtype: str
 229     """
 230     if args.explain:
 231         yield os.path.join(directory, '%stemp%s' % (prefix, suffix))
 232     else:
 233         with tempfile.NamedTemporaryFile(prefix=prefix, suffix=suffix, dir=directory) as tempfile_fd:
 234             tempfile_fd.write(to_bytes(content))
 235             tempfile_fd.flush()
 236 
 237             yield tempfile_fd.name

 257 def get_python_path(args, interpreter):
 258     """
 259     :type args: TestConfig
 260     :type interpreter: str
 261     :rtype: str
 262     """
 263     python_path = PYTHON_PATHS.get(interpreter)
 264 
 265     if python_path:
 266         return python_path
 267 
 268     prefix = 'python-'
 269     suffix = '-ansible'
 270 
 271     root_temp_dir = '/tmp'
 272 
 273     if args.explain:
 274         return os.path.join(root_temp_dir, ''.join((prefix, 'temp', suffix)))
 275 
 276     python_path = tempfile.mkdtemp(prefix=prefix, suffix=suffix, dir=root_temp_dir)
 277     injected_interpreter = os.path.join(python_path, 'python')
 278 
 279     # A symlink is faster than the execv wrapper, but isn't compatible with virtual environments.
 280     # Attempt to detect when it is safe to use a symlink by checking the real path of the interpreter.
 281     use_symlink = os.path.dirname(os.path.realpath(interpreter)) == os.path.dirname(interpreter)
 282 
 283     if use_symlink:
 284         display.info('Injecting "%s" as a symlink to the "%s" interpreter.' % (injected_interpreter, interpreter), verbosity=1)
 285 
 286         os.symlink(interpreter, injected_interpreter)
 287     else:
 288         display.info('Injecting "%s" as a execv wrapper for the "%s" interpreter.' % (injected_interpreter, interpreter), verbosity=1)
 289 
 290         create_interpreter_wrapper(interpreter, injected_interpreter)
 291 
 292     os.chmod(python_path, MODE_DIRECTORY)
 293 
 294     if not PYTHON_PATHS:
 295         atexit.register(cleanup_python_paths)
 296 
 297     PYTHON_PATHS[interpreter] = python_path
 298 
 299     return python_path

 302 def create_temp_dir(prefix=None, suffix=None, base_dir=None):  # type: (t.Optional[str], t.Optional[str], t.Optional[str]) -> str
 303     """Create a temporary directory that persists until the current process exits."""
 304     temp_path = tempfile.mkdtemp(prefix=prefix or 'tmp', suffix=suffix or '', dir=base_dir)
 305     atexit.register(remove_tree, temp_path)

 309 def create_interpreter_wrapper(interpreter, injected_interpreter):  # type: (str, str) -> None
 310     """Create a wrapper for the given Python interpreter at the specified path."""
 311     # sys.executable is used for the shebang to guarantee it is a binary instead of a script
 312     # injected_interpreter could be a script from the system or our own wrapper created for the --venv option
 313     shebang_interpreter = sys.executable
 314 
 315     code = textwrap.dedent('''
 316     #!%s
 317 
 318     from __future__ import absolute_import
 319 
 320     from os import execv
 321     from sys import argv
 322 
 323     python = '%s'
 324 
 325     execv(python, [python] + argv[1:])
 326     ''' % (shebang_interpreter, interpreter)).lstrip()
 327 
 328     write_text_file(injected_interpreter, code)
 329 
 330     os.chmod(injected_interpreter, MODE_FILE_EXECUTE)

 340 def get_coverage_environment(args, target_name, version, temp_path, module_coverage, remote_temp_path=None):
 341     """
 342     :type args: TestConfig
 343     :type target_name: str
 344     :type version: str
 345     :type temp_path: str
 346     :type module_coverage: bool
 347     :type remote_temp_path: str | None
 348     :rtype: dict[str, str]
 349     """
 350     if temp_path:
 351         # integration tests (both localhost and the optional testhost)
 352         # config and results are in a temporary directory
 353         coverage_config_base_path = temp_path
 354         coverage_output_base_path = temp_path
 355     elif args.coverage_config_base_path:
 356         # unit tests, sanity tests and other special cases (localhost only)
 357         # config is in a temporary directory
 358         # results are in the source tree
 359         coverage_config_base_path = args.coverage_config_base_path
 360         coverage_output_base_path = os.path.join(data_context().content.root, data_context().content.results_path)
 361     else:
 362         raise Exception('No temp path and no coverage config base path. Check for missing coverage_context usage.')
 363 
 364     config_file = os.path.join(coverage_config_base_path, COVERAGE_CONFIG_NAME)
 365     coverage_file = os.path.join(coverage_output_base_path, ResultType.COVERAGE.name, '%s=%s=%s=%s=coverage' % (
 366         args.command, target_name, args.coverage_label or 'local-%s' % version, 'python-%s' % version))
 367 
 368     if not args.explain and not os.path.exists(config_file):
 369         raise Exception('Missing coverage config file: %s' % config_file)
 370 
 371     if args.coverage_check:
 372         # cause the 'coverage' module to be found, but not imported or enabled
 373         coverage_file = ''
 374 
 375     # Enable code coverage collection on local Python programs (this does not include Ansible modules).
 376     # Used by the injectors to support code coverage.
 377     # Used by the pytest unit test plugin to support code coverage.
 378     # The COVERAGE_FILE variable is also used directly by the 'coverage' module.
 379     env = dict(
 380         COVERAGE_CONF=config_file,
 381         COVERAGE_FILE=coverage_file,
 382     )
 383 
 384     if module_coverage:
 385         # Enable code coverage collection on Ansible modules (both local and remote).
 386         # Used by the AnsiballZ wrapper generator in lib/ansible/executor/module_common.py to support code coverage.
 387         env.update(dict(
 388             _ANSIBLE_COVERAGE_CONFIG=config_file,
 389             _ANSIBLE_COVERAGE_OUTPUT=coverage_file,
 390         ))
 391 
 392         if remote_temp_path:
 393             # Include the command, target and label so the remote host can create a filename with that info. The remote
 394             # is responsible for adding '={language version}=coverage.{hostname}.{pid}.{id}'
 395             env['_ANSIBLE_COVERAGE_REMOTE_OUTPUT'] = os.path.join(remote_temp_path, '%s=%s=%s' % (
 396                 args.command, target_name, args.coverage_label or 'remote'))
 397             env['_ANSIBLE_COVERAGE_REMOTE_PATH_FILTER'] = os.path.join(data_context().content.root, '*')
 398 
 399     return env

 402 def intercept_command(args, cmd, target_name, env, capture=False, data=None, cwd=None, python_version=None, temp_path=None, module_coverage=True,
 403                       virtualenv=None, disable_coverage=False, remote_temp_path=None):
 404     """
 405     :type args: TestConfig
 406     :type cmd: collections.Iterable[str]
 407     :type target_name: str
 408     :type env: dict[str, str]
 409     :type capture: bool
 410     :type data: str | None
 411     :type cwd: str | None
 412     :type python_version: str | None
 413     :type temp_path: str | None
 414     :type module_coverage: bool
 415     :type virtualenv: str | None
 416     :type disable_coverage: bool
 417     :type remote_temp_path: str | None
 418     :rtype: str | None, str | None
 419     """
 420     if not env:
 421         env = common_environment()
 422     else:
 423         env = env.copy()
 424 
 425     cmd = list(cmd)
 426     version = python_version or args.python_version
 427     interpreter = virtualenv or find_python(version)
 428     inject_path = os.path.join(ANSIBLE_TEST_DATA_ROOT, 'injector')
 429 
 430     if not virtualenv:
 431         # injection of python into the path is required when not activating a virtualenv
 432         # otherwise scripts may find the wrong interpreter or possibly no interpreter
 433         python_path = get_python_path(args, interpreter)
 434         inject_path = python_path + os.path.pathsep + inject_path
 435 
 436     env['PATH'] = inject_path + os.path.pathsep + env['PATH']
 437     env['ANSIBLE_TEST_PYTHON_VERSION'] = version
 438     env['ANSIBLE_TEST_PYTHON_INTERPRETER'] = interpreter
 439 
 440     if args.coverage and not disable_coverage:
 441         # add the necessary environment variables to enable code coverage collection
 442         env.update(get_coverage_environment(args, target_name, version, temp_path, module_coverage,
 443                                             remote_temp_path=remote_temp_path))
 444 
 445     return run_command(args, cmd, capture=capture, env=env, data=data, cwd=cwd)

 448 def resolve_csharp_ps_util(import_name, path):
 449     """
 450     :type import_name: str
 451     :type path: str
 452     """
 453     if data_context().content.is_ansible or not import_name.startswith('.'):
 454         # We don't support relative paths for builtin utils, there's no point.
 455         return import_name
 456 
 457     packages = import_name.split('.')
 458     module_packages = path.split(os.path.sep)
 459 
 460     for package in packages:
 461         if not module_packages or package:
 462             break
 463         del module_packages[-1]
 464 
 465     return 'ansible_collections.%s%s' % (data_context().content.prefix,
 466                                          '.'.join(module_packages + [p for p in packages if p]))

 469 def run_command(args, cmd, capture=False, env=None, data=None, cwd=None, always=False, stdin=None, stdout=None,
 470                 cmd_verbosity=1, str_errors='strict'):
 471     """
 472     :type args: CommonConfig
 473     :type cmd: collections.Iterable[str]
 474     :type capture: bool
 475     :type env: dict[str, str] | None
 476     :type data: str | None
 477     :type cwd: str | None
 478     :type always: bool
 479     :type stdin: file | None
 480     :type stdout: file | None
 481     :type cmd_verbosity: int
 482     :type str_errors: str
 483     :rtype: str | None, str | None
 484     """
 485     explain = args.explain and not always
 486     return raw_command(cmd, capture=capture, env=env, data=data, cwd=cwd, explain=explain, stdin=stdin, stdout=stdout,
 487                        cmd_verbosity=cmd_verbosity, str_errors=str_errors)
