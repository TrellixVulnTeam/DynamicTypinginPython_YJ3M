 157 def check_legacy_modules():
 158     """Detect conflicts with legacy core/extras module directories to avoid problems later."""
 159     for directory in 'core', 'extras':
 160         path = 'lib/ansible/modules/%s' % directory
 161 
 162         for root, _dir_names, file_names in os.walk(path):
 163             if file_names:
 164                 # the directory shouldn't exist, but if it does, it must contain no files
 165                 raise ApplicationError('Files prohibited in "%s". '
 166                                        'These are most likely legacy modules from version 2.2 or earlier.' % root)

 196 def get_cryptography_requirement(args, python_version):  # type: (EnvironmentConfig, str) -> str
 197     """
 198     Return the correct cryptography requirement for the given python version.
 199     The version of cryptograpy installed depends on the python version and setuptools version.
 200     """
 201     python = find_python(python_version)
 202     setuptools_version = get_setuptools_version(args, python)
 203 
 204     if setuptools_version >= (18, 5):
 205         if python_version == '2.6':
 206             # cryptography 2.2+ requires python 2.7+
 207             # see https://github.com/pyca/cryptography/blob/master/CHANGELOG.rst#22---2018-03-19
 208             cryptography = 'cryptography < 2.2'
 209         else:
 210             cryptography = 'cryptography'
 211     else:
 212         # cryptography 2.1+ requires setuptools 18.5+
 213         # see https://github.com/pyca/cryptography/blob/62287ae18383447585606b9d0765c0f1b8a9777c/setup.py#L26
 214         cryptography = 'cryptography < 2.1'
 215 
 216     return cryptography

 219 def install_command_requirements(args, python_version=None, context=None, enable_pyyaml_check=False):
 220     """
 221     :type args: EnvironmentConfig
 222     :type python_version: str | None
 223     :type context: str | None
 224     :type enable_pyyaml_check: bool
 225     """
 226     if not args.explain:
 227         make_dirs(ResultType.COVERAGE.path)
 228         make_dirs(ResultType.DATA.path)
 229 
 230     if isinstance(args, ShellConfig):
 231         if args.raw:
 232             return
 233 
 234     generate_egg_info(args)
 235 
 236     if not args.requirements:
 237         return
 238 
 239     if isinstance(args, ShellConfig):
 240         return
 241 
 242     packages = []
 243 
 244     if isinstance(args, TestConfig):
 245         if args.coverage:
 246             packages.append('coverage')
 247         if args.junit:
 248             packages.append('junit-xml')
 249 
 250     if not python_version:
 251         python_version = args.python_version
 252 
 253     pip = generate_pip_command(find_python(python_version))
 254 
 255     # skip packages which have aleady been installed for python_version
 256 
 257     try:
 258         package_cache = install_command_requirements.package_cache
 259     except AttributeError:
 260         package_cache = install_command_requirements.package_cache = {}
 261 
 262     installed_packages = package_cache.setdefault(python_version, set())
 263     skip_packages = [package for package in packages if package in installed_packages]
 264 
 265     for package in skip_packages:
 266         packages.remove(package)
 267 
 268     installed_packages.update(packages)
 269 
 270     if args.command != 'sanity':
 271         install_ansible_test_requirements(args, pip)
 272 
 273         # make sure setuptools is available before trying to install cryptography
 274         # the installed version of setuptools affects the version of cryptography to install
 275         run_command(args, generate_pip_install(pip, '', packages=['setuptools']))
 276 
 277         # install the latest cryptography version that the current requirements can support
 278         # use a custom constraints file to avoid the normal constraints file overriding the chosen version of cryptography
 279         # if not installed here later install commands may try to install an unsupported version due to the presence of older setuptools
 280         # this is done instead of upgrading setuptools to allow tests to function with older distribution provided versions of setuptools
 281         run_command(args, generate_pip_install(pip, '',
 282                                                packages=[get_cryptography_requirement(args, python_version)],
 283                                                constraints=os.path.join(ANSIBLE_TEST_DATA_ROOT, 'cryptography-constraints.txt')))
 284 
 285     commands = [generate_pip_install(pip, args.command, packages=packages, context=context)]
 286 
 287     if isinstance(args, IntegrationConfig):
 288         for cloud_platform in get_cloud_platforms(args):
 289             commands.append(generate_pip_install(pip, '%s.cloud.%s' % (args.command, cloud_platform)))
 290 
 291     commands = [cmd for cmd in commands if cmd]
 292 
 293     if not commands:
 294         return  # no need to detect changes or run pip check since we are not making any changes
 295 
 296     # only look for changes when more than one requirements file is needed
 297     detect_pip_changes = len(commands) > 1
 298 
 299     # first pass to install requirements, changes expected unless environment is already set up
 300     install_ansible_test_requirements(args, pip)
 301     changes = run_pip_commands(args, pip, commands, detect_pip_changes)
 302 
 303     if changes:
 304         # second pass to check for conflicts in requirements, changes are not expected here
 305         changes = run_pip_commands(args, pip, commands, detect_pip_changes)
 306 
 307         if changes:
 308             raise ApplicationError('Conflicts detected in requirements. The following commands reported changes during verification:\n%s' %
 309                                    '\n'.join((' '.join(cmd_quote(c) for c in cmd) for cmd in changes)))
 310 
 311     if args.pip_check:
 312         # ask pip to check for conflicts between installed packages
 313         try:
 314             run_command(args, pip + ['check', '--disable-pip-version-check'], capture=True)
 315         except SubprocessError as ex:
 316             if ex.stderr.strip() == 'ERROR: unknown command "check"':
 317                 display.warning('Cannot check pip requirements for conflicts because "pip check" is not supported.')
 318             else:
 319                 raise
 320 
 321     if enable_pyyaml_check:
 322         # pyyaml may have been one of the requirements that was installed, so perform an optional check for it
 323         check_pyyaml(args, python_version, required=False)

 326 def install_ansible_test_requirements(args, pip):  # type: (EnvironmentConfig, t.List[str]) -> None
 327     """Install requirements for ansible-test for the given pip if not already installed."""
 328     try:
 329         installed = install_command_requirements.installed
 330     except AttributeError:
 331         installed = install_command_requirements.installed = set()
 332 
 333     if tuple(pip) in installed:
 334         return
 335 
 336     # make sure basic ansible-test requirements are met, including making sure that pip is recent enough to support constraints
 337     # virtualenvs created by older distributions may include very old pip versions, such as those created in the centos6 test container (pip 6.0.8)
 338     run_command(args, generate_pip_install(pip, 'ansible-test', use_constraints=False))
 339 
 340     installed.add(tuple(pip))

 343 def run_pip_commands(args, pip, commands, detect_pip_changes=False):
 344     """
 345     :type args: EnvironmentConfig
 346     :type pip: list[str]
 347     :type commands: list[list[str]]
 348     :type detect_pip_changes: bool
 349     :rtype: list[list[str]]
 350     """
 351     changes = []
 352 
 353     after_list = pip_list(args, pip) if detect_pip_changes else None
 354 
 355     for cmd in commands:
 356         if not cmd:
 357             continue
 358 
 359         before_list = after_list
 360 
 361         run_command(args, cmd)
 362 
 363         after_list = pip_list(args, pip) if detect_pip_changes else None
 364 
 365         if before_list != after_list:
 366             changes.append(cmd)
 367 
 368     return changes

 381 def generate_egg_info(args):
 382     """
 383     :type args: EnvironmentConfig
 384     """
 385     if args.explain:
 386         return
 387 
 388     ansible_version = get_ansible_version()
 389 
 390     # inclusion of the version number in the path is optional
 391     # see: https://setuptools.readthedocs.io/en/latest/formats.html#filename-embedded-metadata
 392     egg_info_path = ANSIBLE_LIB_ROOT + '_base-%s.egg-info' % ansible_version
 393 
 394     if os.path.exists(egg_info_path):
 395         return
 396 
 397     egg_info_path = ANSIBLE_LIB_ROOT + '_base.egg-info'
 398 
 399     if os.path.exists(egg_info_path):
 400         return
 401 
 402     # minimal PKG-INFO stub following the format defined in PEP 241
 403     # required for older setuptools versions to avoid a traceback when importing pkg_resources from packages like cryptography
 404     # newer setuptools versions are happy with an empty directory
 405     # including a stub here means we don't need to locate the existing file or have setup.py generate it when running from source
 406     pkg_info = '''
 407 Metadata-Version: 1.0
 408 Name: ansible
 409 Version: %s
 410 Platform: UNKNOWN
 411 Summary: Radically simple IT automation
 412 Author-email: info@ansible.com
 413 License: GPLv3+
 414 ''' % get_ansible_version()
 415 
 416     pkg_info_path = os.path.join(egg_info_path, 'PKG-INFO')
 417 
 418     write_text_file(pkg_info_path, pkg_info.lstrip(), create_directories=True)

 421 def generate_pip_install(pip, command, packages=None, constraints=None, use_constraints=True, context=None):
 422     """
 423     :type pip: list[str]
 424     :type command: str
 425     :type packages: list[str] | None
 426     :type constraints: str | None
 427     :type use_constraints: bool
 428     :type context: str | None
 429     :rtype: list[str] | None
 430     """
 431     constraints = constraints or os.path.join(ANSIBLE_TEST_DATA_ROOT, 'requirements', 'constraints.txt')
 432     requirements = os.path.join(ANSIBLE_TEST_DATA_ROOT, 'requirements', '%s.txt' % ('%s.%s' % (command, context) if context else command))
 433 
 434     options = []
 435 
 436     if os.path.exists(requirements) and os.path.getsize(requirements):
 437         options += ['-r', requirements]
 438 
 439     if command == 'sanity' and data_context().content.is_ansible:
 440         requirements = os.path.join(data_context().content.sanity_path, 'code-smell', '%s.requirements.txt' % context)
 441 
 442         if os.path.exists(requirements) and os.path.getsize(requirements):
 443             options += ['-r', requirements]
 444 
 445     if command == 'units':
 446         requirements = os.path.join(data_context().content.unit_path, 'requirements.txt')
 447 
 448         if os.path.exists(requirements) and os.path.getsize(requirements):
 449             options += ['-r', requirements]
 450 
 451     if command in ('integration', 'windows-integration', 'network-integration'):
 452         requirements = os.path.join(data_context().content.integration_path, 'requirements.txt')
 453 
 454         if os.path.exists(requirements) and os.path.getsize(requirements):
 455             options += ['-r', requirements]
 456 
 457         requirements = os.path.join(data_context().content.integration_path, '%s.requirements.txt' % command)
 458 
 459         if os.path.exists(requirements) and os.path.getsize(requirements):
 460             options += ['-r', requirements]
 461 
 462     if packages:
 463         options += packages
 464 
 465     if not options:
 466         return None
 467 
 468     if use_constraints:
 469         options.extend(['-c', constraints])
 470 
 471     return pip + ['install', '--disable-pip-version-check'] + options

 504 def command_network_integration(args):
 505     """
 506     :type args: NetworkIntegrationConfig
 507     """
 508     handle_layout_messages(data_context().content.integration_messages)
 509 
 510     inventory_relative_path = get_inventory_relative_path(args)
 511     template_path = os.path.join(ANSIBLE_TEST_CONFIG_ROOT, os.path.basename(inventory_relative_path)) + '.template'
 512 
 513     if args.inventory:
 514         inventory_path = os.path.join(data_context().content.root, data_context().content.integration_path, args.inventory)
 515     else:
 516         inventory_path = os.path.join(data_context().content.root, inventory_relative_path)
 517 
 518     if args.no_temp_workdir:
 519         # temporary solution to keep DCI tests working
 520         inventory_exists = os.path.exists(inventory_path)
 521     else:
 522         inventory_exists = os.path.isfile(inventory_path)
 523 
 524     if not args.explain and not args.platform and not inventory_exists:
 525         raise ApplicationError(
 526             'Inventory not found: %s\n'
 527             'Use --inventory to specify the inventory path.\n'
 528             'Use --platform to provision resources and generate an inventory file.\n'
 529             'See also inventory template: %s' % (inventory_path, template_path)
 530         )
 531 
 532     check_inventory(args, inventory_path)
 533     delegate_inventory(args, inventory_path)
 534 
 535     all_targets = tuple(walk_network_integration_targets(include_hidden=True))
 536     internal_targets = command_integration_filter(args, all_targets, init_callback=network_init)
 537     instances = []  # type: t.List[WrappedThread]
 538 
 539     if args.platform:
 540         get_python_path(args, args.python_executable)  # initialize before starting threads
 541 
 542         configs = dict((config['platform_version'], config) for config in args.metadata.instance_config)
 543 
 544         for platform_version in args.platform:
 545             platform, version = platform_version.split('/', 1)
 546             config = configs.get(platform_version)
 547 
 548             if not config:
 549                 continue
 550 
 551             instance = WrappedThread(functools.partial(network_run, args, platform, version, config))
 552             instance.daemon = True
 553             instance.start()
 554             instances.append(instance)
 555 
 556         while any(instance.is_alive() for instance in instances):
 557             time.sleep(1)
 558 
 559         remotes = [instance.wait_for_result() for instance in instances]
 560         inventory = network_inventory(remotes)
 561 
 562         display.info('>>> Inventory: %s\n%s' % (inventory_path, inventory.strip()), verbosity=3)
 563 
 564         if not args.explain:
 565             write_text_file(inventory_path, inventory)
 566 
 567     success = False
 568 
 569     try:
 570         command_integration_filtered(args, internal_targets, all_targets, inventory_path)
 571         success = True
 572     finally:
 573         if args.remote_terminate == 'always' or (args.remote_terminate == 'success' and success):
 574             for instance in instances:
 575                 instance.result.stop()

 578 def network_init(args, internal_targets):  # type: (NetworkIntegrationConfig, t.Tuple[IntegrationTarget, ...]) -> None
 579     """Initialize platforms for network integration tests."""
 580     if not args.platform:
 581         return
 582 
 583     if args.metadata.instance_config is not None:
 584         return
 585 
 586     platform_targets = set(a for target in internal_targets for a in target.aliases if a.startswith('network/'))
 587 
 588     instances = []  # type: t.List[WrappedThread]
 589 
 590     # generate an ssh key (if needed) up front once, instead of for each instance
 591     SshKey(args)
 592 
 593     for platform_version in args.platform:
 594         platform, version = platform_version.split('/', 1)
 595         platform_target = 'network/%s/' % platform
 596 
 597         if platform_target not in platform_targets:
 598             display.warning('Skipping "%s" because selected tests do not target the "%s" platform.' % (
 599                 platform_version, platform))
 600             continue
 601 
 602         instance = WrappedThread(functools.partial(network_start, args, platform, version))
 603         instance.daemon = True
 604         instance.start()
 605         instances.append(instance)
 606 
 607     while any(instance.is_alive() for instance in instances):
 608         time.sleep(1)
 609 
 610     args.metadata.instance_config = [instance.wait_for_result() for instance in instances]

 644 def network_inventory(remotes):
 645     """
 646     :type remotes: list[AnsibleCoreCI]
 647     :rtype: str
 648     """
 649     groups = dict([(remote.platform, []) for remote in remotes])
 650     net = []
 651 
 652     for remote in remotes:
 653         options = dict(
 654             ansible_host=remote.connection.hostname,
 655             ansible_user=remote.connection.username,
 656             ansible_ssh_private_key_file=os.path.abspath(remote.ssh_key.key),
 657         )
 658 
 659         settings = get_network_settings(remote.args, remote.platform, remote.version)
 660 
 661         options.update(settings.inventory_vars)
 662 
 663         groups[remote.platform].append(
 664             '%s %s' % (
 665                 remote.name.replace('.', '-'),
 666                 ' '.join('%s="%s"' % (k, options[k]) for k in sorted(options)),
 667             )
 668         )
 669 
 670         net.append(remote.platform)
 671 
 672     groups['net:children'] = net
 673 
 674     template = ''
 675 
 676     for group in groups:
 677         hosts = '\n'.join(groups[group])
 678 
 679         template += textwrap.dedent("""
 680         [%s]
 681         %s

1294 def inject_httptester(args):
1295     """
1296     :type args: CommonConfig
1297     """
1298     comment = ' # ansible-test httptester\n'
1299     append_lines = ['127.0.0.1 %s%s' % (host, comment) for host in HTTPTESTER_HOSTS]
1300     hosts_path = '/etc/hosts'
1301 
1302     original_lines = read_text_file(hosts_path).splitlines(True)
1303 
1304     if not any(line.endswith(comment) for line in original_lines):
1305         write_text_file(hosts_path, ''.join(original_lines + append_lines))
1306 
1307     # determine which forwarding mechanism to use
1308     pfctl = find_executable('pfctl', required=False)
1309     iptables = find_executable('iptables', required=False)
1310 
1311     if pfctl:
1312         kldload = find_executable('kldload', required=False)
1313 
1314         if kldload:
1315             try:
1316                 run_command(args, ['kldload', 'pf'], capture=True)
1317             except SubprocessError:
1318                 pass  # already loaded
1319 
1320         rules = '''
1321 rdr pass inet proto tcp from any to any port 80 -> 127.0.0.1 port 8080
1322 rdr pass inet proto tcp from any to any port 443 -> 127.0.0.1 port 8443
1323 '''
1324         cmd = ['pfctl', '-ef', '-']
1325 
1326         try:
1327             run_command(args, cmd, capture=True, data=rules)
1328         except SubprocessError:
1329             pass  # non-zero exit status on success
1330 
1331     elif iptables:
1332         ports = [
1333             (80, 8080),
1334             (443, 8443),
1335         ]
1336 
1337         for src, dst in ports:
1338             rule = ['-o', 'lo', '-p', 'tcp', '--dport', str(src), '-j', 'REDIRECT', '--to-port', str(dst)]
1339 
1340             try:
1341                 # check for existing rule
1342                 cmd = ['iptables', '-t', 'nat', '-C', 'OUTPUT'] + rule
1343                 run_command(args, cmd, capture=True)
1344             except SubprocessError:

1947     @staticmethod
1948     def check_python_pip_association(version, python_paths, pip_paths, pip_interpreters, warnings):
1949         """
1950         :type version: str
1951         :param python_paths: dict[str, str]
1952         :param pip_paths:  dict[str, str]
1953         :param pip_interpreters:  dict[str, str]
1954         :param warnings: list[str]
1955         """
1956         python_label = 'Python%s' % (' %s' % version if version else '')
1957 
1958         pip_path = pip_paths.get(version)
1959         python_path = python_paths.get(version)
1960 
1961         if not python_path and not pip_path:
1962             # neither python or pip is present for this version
1963             return
1964 
1965         if not python_path:
1966             warnings.append('A %s interpreter was not found, yet a matching pip was found at "%s".' % (python_label, pip_path))
1967             return
1968 
1969         if not pip_path:
1970             warnings.append('A %s interpreter was found at "%s", yet a matching pip was not found.' % (python_label, python_path))
1971             return
1972 
1973         pip_shebang = pip_interpreters.get(version)
1974 
1975         match = re.search(r'#!\s*(?P<command>[^\s]+)', pip_shebang)
1976 
1977         if not match:
1978             warnings.append('A %s pip was found at "%s", but it does not have a valid shebang: %s' % (python_label, pip_path, pip_shebang))
1979             return
1980 
1981         pip_interpreter = os.path.realpath(match.group('command'))
1982         python_interpreter = os.path.realpath(python_path)
1983 
1984         if pip_interpreter == python_interpreter:
1985             return
1986 
1987         try:
1988             identical = filecmp.cmp(pip_interpreter, python_interpreter)
1989         except OSError:
1990             identical = False
1991 
1992         if identical:
1993             return
1994 
1995         warnings.append('A %s pip was found at "%s", but it uses interpreter "%s" instead of "%s".' % (
1996             python_label, pip_path, pip_interpreter, python_interpreter))
