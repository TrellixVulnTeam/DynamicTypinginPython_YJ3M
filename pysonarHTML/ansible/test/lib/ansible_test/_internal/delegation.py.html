 102 def check_delegation_args(args):
 103     """
 104     :type args: CommonConfig
 105     """
 106     if not isinstance(args, EnvironmentConfig):
 107         return
 108 
 109     if args.docker:
 110         get_python_version(args, get_docker_completion(), args.docker_raw)
 111     elif args.remote:
 112         get_python_version(args, get_remote_completion(), args.remote)

 115 def delegate(args, exclude, require, integration_targets):
 116     """
 117     :type args: EnvironmentConfig
 118     :type exclude: list[str]
 119     :type require: list[str]
 120     :type integration_targets: tuple[IntegrationTarget]
 121     :rtype: bool
 122     """
 123     if isinstance(args, TestConfig):
 124         args.metadata.ci_provider = get_ci_provider().code
 125 
 126         make_dirs(ResultType.TMP.path)
 127 
 128         with tempfile.NamedTemporaryFile(prefix='metadata-', suffix='.json', dir=ResultType.TMP.path) as metadata_fd:
 129             args.metadata_path = os.path.join(ResultType.TMP.relative_path, os.path.basename(metadata_fd.name))
 130             args.metadata.to_file(args.metadata_path)
 131 
 132             try:
 133                 return delegate_command(args, exclude, require, integration_targets)
 134             finally:
 135                 args.metadata_path = None
 136     else:
 137         return delegate_command(args, exclude, require, integration_targets)

 140 def delegate_command(args, exclude, require, integration_targets):
 141     """
 142     :type args: EnvironmentConfig
 143     :type exclude: list[str]
 144     :type require: list[str]
 145     :type integration_targets: tuple[IntegrationTarget]
 146     :rtype: bool
 147     """
 148     if args.venv:
 149         delegate_venv(args, exclude, require, integration_targets)
 150         return True
 151 
 152     if args.docker:
 153         delegate_docker(args, exclude, require, integration_targets)
 154         return True
 155 
 156     if args.remote:
 157         delegate_remote(args, exclude, require, integration_targets)
 158         return True
 159 
 160     return False

 163 def delegate_venv(args,  # type: EnvironmentConfig
 164                   exclude,  # type: t.List[str]
 165                   require,  # type: t.List[str]
 166                   integration_targets,  # type: t.Tuple[IntegrationTarget, ...]
 167                   ):  # type: (...) -> None
 168     """Delegate ansible-test execution to a virtual environment using venv or virtualenv."""
 169     if args.python:
 170         versions = (args.python_version,)
 171     else:
 172         versions = SUPPORTED_PYTHON_VERSIONS
 173 
 174     if args.httptester:
 175         needs_httptester = sorted(target.name for target in integration_targets if 'needs/httptester/' in target.aliases)
 176 
 177         if needs_httptester:
 178             display.warning('Use --docker or --remote to enable httptester for tests marked "needs/httptester": %s' % ', '.join(needs_httptester))
 179 
 180     if args.venv_system_site_packages:
 181         suffix = '-ssp'
 182     else:
 183         suffix = ''
 184 
 185     venvs = dict((version, os.path.join(ResultType.TMP.path, 'delegation', 'python%s%s' % (version, suffix))) for version in versions)
 186     venvs = dict((version, path) for version, path in venvs.items() if create_virtual_environment(args, version, path, args.venv_system_site_packages))
 187 
 188     if not venvs:
 189         raise ApplicationError('No usable virtual environment support found.')
 190 
 191     options = {
 192         '--venv': 0,
 193         '--venv-system-site-packages': 0,
 194     }
 195 
 196     with tempdir() as inject_path:
 197         for version, path in venvs.items():
 198             create_interpreter_wrapper(os.path.join(path, 'bin', 'python'), os.path.join(inject_path, 'python%s' % version))
 199 
 200         python_interpreter = os.path.join(inject_path, 'python%s' % args.python_version)
 201 
 202         cmd = generate_command(args, python_interpreter, ANSIBLE_BIN_PATH, data_context().content.root, options, exclude, require)
 203 
 204         if isinstance(args, TestConfig):
 205             if args.coverage and not args.coverage_label:
 206                 cmd += ['--coverage-label', 'venv']
 207 
 208         env = common_environment()
 209 
 210         with tempdir() as library_path:
 211             # expose ansible and ansible_test to the virtual environment (only required when running from an install)
 212             os.symlink(ANSIBLE_LIB_ROOT, os.path.join(library_path, 'ansible'))
 213             os.symlink(ANSIBLE_TEST_ROOT, os.path.join(library_path, 'ansible_test'))
 214 
 215             env.update(
 216                 PATH=inject_path + os.path.pathsep + env['PATH'],
 217                 PYTHONPATH=library_path,
 218             )
 219 
 220             run_command(args, cmd, env=env)

 223 def delegate_docker(args, exclude, require, integration_targets):
 224     """
 225     :type args: EnvironmentConfig
 226     :type exclude: list[str]
 227     :type require: list[str]
 228     :type integration_targets: tuple[IntegrationTarget]
 229     """
 230     test_image = args.docker
 231     privileged = args.docker_privileged
 232 
 233     if isinstance(args, ShellConfig):
 234         use_httptester = args.httptester
 235     else:
 236         use_httptester = args.httptester and any('needs/httptester/' in target.aliases for target in integration_targets)
 237 
 238     if use_httptester:
 239         docker_pull(args, args.httptester)
 240 
 241     docker_pull(args, test_image)
 242 
 243     httptester_id = None
 244     test_id = None
 245     success = False
 246 
 247     options = {
 248         '--docker': 1,
 249         '--docker-privileged': 0,
 250         '--docker-util': 1,
 251     }
 252 
 253     python_interpreter = get_python_interpreter(args, get_docker_completion(), args.docker_raw)
 254 
 255     pwd = '/root'
 256     ansible_root = os.path.join(pwd, 'ansible')
 257 
 258     if data_context().content.collection:
 259         content_root = os.path.join(pwd, data_context().content.collection.directory)
 260     else:
 261         content_root = ansible_root
 262 
 263     remote_results_root = os.path.join(content_root, data_context().content.results_path)
 264 
 265     cmd = generate_command(args, python_interpreter, os.path.join(ansible_root, 'bin'), content_root, options, exclude, require)
 266 
 267     if isinstance(args, TestConfig):
 268         if args.coverage and not args.coverage_label:
 269             image_label = args.docker_raw
 270             image_label = re.sub('[^a-zA-Z0-9]+', '-', image_label)
 271             cmd += ['--coverage-label', 'docker-%s' % image_label]
 272 
 273     if isinstance(args, IntegrationConfig):
 274         if not args.allow_destructive:
 275             cmd.append('--allow-destructive')
 276 
 277     cmd_options = []
 278 
 279     if isinstance(args, ShellConfig) or (isinstance(args, IntegrationConfig) and args.debug_strategy):
 280         cmd_options.append('-it')
 281 
 282     with tempfile.NamedTemporaryFile(prefix='ansible-source-', suffix='.tgz') as local_source_fd:
 283         try:
 284             create_payload(args, local_source_fd.name)
 285 
 286             if use_httptester:
 287                 httptester_id = run_httptester(args)
 288             else:
 289                 httptester_id = None
 290 
 291             test_options = [
 292                 '--detach',
 293                 '--volume', '/sys/fs/cgroup:/sys/fs/cgroup:ro',
 294                 '--privileged=%s' % str(privileged).lower(),
 295             ]
 296 
 297             if args.docker_memory:
 298                 test_options.extend([
 299                     '--memory=%d' % args.docker_memory,
 300                     '--memory-swap=%d' % args.docker_memory,
 301                 ])
 302 
 303             docker_socket = '/var/run/docker.sock'
 304 
 305             if args.docker_seccomp != 'default':
 306                 test_options += ['--security-opt', 'seccomp=%s' % args.docker_seccomp]
 307 
 308             if os.path.exists(docker_socket):
 309                 test_options += ['--volume', '%s:%s' % (docker_socket, docker_socket)]
 310 
 311             if httptester_id:
 312                 test_options += ['--env', 'HTTPTESTER=1']
 313 
 314                 for host in HTTPTESTER_HOSTS:
 315                     test_options += ['--link', '%s:%s' % (httptester_id, host)]
 316 
 317             if isinstance(args, IntegrationConfig):
 318                 cloud_platforms = get_cloud_providers(args)
 319 
 320                 for cloud_platform in cloud_platforms:
 321                     test_options += cloud_platform.get_docker_run_options()
 322 
 323             test_id = docker_run(args, test_image, options=test_options)[0]
 324 
 325             if args.explain:
 326                 test_id = 'test_id'
 327             else:
 328                 test_id = test_id.strip()
 329 
 330             # write temporary files to /root since /tmp isn't ready immediately on container start
 331             docker_put(args, test_id, os.path.join(ANSIBLE_TEST_DATA_ROOT, 'setup', 'docker.sh'), '/root/docker.sh')
 332             docker_exec(args, test_id, ['/bin/bash', '/root/docker.sh'])
 333             docker_put(args, test_id, local_source_fd.name, '/root/test.tgz')
 334             docker_exec(args, test_id, ['tar', 'oxzf', '/root/test.tgz', '-C', '/root'])
 335 
 336             # docker images are only expected to have a single python version available
 337             if isinstance(args, UnitsConfig) and not args.python:
 338                 cmd += ['--python', 'default']
 339 
 340             # run unit tests unprivileged to prevent stray writes to the source tree
 341             # also disconnect from the network once requirements have been installed
 342             if isinstance(args, UnitsConfig):
 343                 writable_dirs = [
 344                     os.path.join(content_root, ResultType.JUNIT.relative_path),
 345                     os.path.join(content_root, ResultType.COVERAGE.relative_path),
 346                 ]
 347 
 348                 docker_exec(args, test_id, ['mkdir', '-p'] + writable_dirs)
 349                 docker_exec(args, test_id, ['chmod', '777'] + writable_dirs)
 350                 docker_exec(args, test_id, ['chmod', '755', '/root'])
 351                 docker_exec(args, test_id, ['chmod', '644', os.path.join(content_root, args.metadata_path)])
 352 
 353                 docker_exec(args, test_id, ['useradd', 'pytest', '--create-home'])
 354 
 355                 docker_exec(args, test_id, cmd + ['--requirements-mode', 'only'], options=cmd_options)
 356 
 357                 networks = get_docker_networks(args, test_id)
 358 
 359                 if networks is not None:
 360                     for network in networks:
 361                         docker_network_disconnect(args, test_id, network)
 362                 else:
 363                     display.warning('Network disconnection is not supported (this is normal under podman). '
 364                                     'Tests will not be isolated from the network. Network-related tests may misbehave.')
 365 
 366                 cmd += ['--requirements-mode', 'skip']
 367 
 368                 cmd_options += ['--user', 'pytest']
 369 
 370             try:
 371                 docker_exec(args, test_id, cmd, options=cmd_options)
 372                 # docker_exec will throw SubprocessError if not successful
 373                 # If we make it here, all the prep work earlier and the docker_exec line above were all successful.
 374                 success = True
 375             finally:
 376                 local_test_root = os.path.dirname(os.path.join(data_context().content.root, data_context().content.results_path))
 377 
 378                 remote_test_root = os.path.dirname(remote_results_root)
 379                 remote_results_name = os.path.basename(remote_results_root)
 380                 remote_temp_file = os.path.join('/root', remote_results_name + '.tgz')
 381 
 382                 make_dirs(local_test_root)  # make sure directory exists for collections which have no tests
 383 
 384                 with tempfile.NamedTemporaryFile(prefix='ansible-result-', suffix='.tgz') as local_result_fd:
 385                     docker_exec(args, test_id, ['tar', 'czf', remote_temp_file, '--exclude', ResultType.TMP.name, '-C', remote_test_root, remote_results_name])
 386                     docker_get(args, test_id, remote_temp_file, local_result_fd.name)
 387                     run_command(args, ['tar', 'oxzf', local_result_fd.name, '-C', local_test_root])
 388         finally:
 389             if httptester_id:
 390                 docker_rm(args, httptester_id)
 391 
 392             if test_id:
 393                 if args.docker_terminate == 'always' or (args.docker_terminate == 'success' and success):
 394                     docker_rm(args, test_id)

 397 def delegate_remote(args, exclude, require, integration_targets):
 398     """
 399     :type args: EnvironmentConfig
 400     :type exclude: list[str]
 401     :type require: list[str]
 402     :type integration_targets: tuple[IntegrationTarget]
 403     """
 404     remote = args.parsed_remote
 405 
 406     core_ci = AnsibleCoreCI(args, remote.platform, remote.version, stage=args.remote_stage, provider=args.remote_provider, arch=remote.arch)
 407     success = False
 408     raw = False
 409 
 410     if isinstance(args, ShellConfig):
 411         use_httptester = args.httptester
 412         raw = args.raw
 413     else:
 414         use_httptester = args.httptester and any('needs/httptester/' in target.aliases for target in integration_targets)
 415 
 416     if use_httptester and not docker_available():
 417         display.warning('Assuming --disable-httptester since `docker` is not available.')
 418         use_httptester = False
 419 
 420     httptester_id = None
 421     ssh_options = []
 422     content_root = None
 423 
 424     try:
 425         core_ci.start()
 426 
 427         if use_httptester:
 428             httptester_id, ssh_options = start_httptester(args)
 429 
 430         core_ci.wait()
 431 
 432         python_version = get_python_version(args, get_remote_completion(), args.remote)
 433 
 434         if remote.platform == 'windows':
 435             # Windows doesn't need the ansible-test fluff, just run the SSH command
 436             manage = ManageWindowsCI(core_ci)
 437             manage.setup(python_version)
 438 
 439             cmd = ['powershell.exe']
 440         elif raw:
 441             manage = ManagePosixCI(core_ci)
 442             manage.setup(python_version)
 443 
 444             cmd = create_shell_command(['bash'])
 445         else:
 446             manage = ManagePosixCI(core_ci)
 447             pwd = manage.setup(python_version)
 448 
 449             options = {
 450                 '--remote': 1,
 451             }
 452 
 453             python_interpreter = get_python_interpreter(args, get_remote_completion(), args.remote)
 454 
 455             ansible_root = os.path.join(pwd, 'ansible')
 456 
 457             if data_context().content.collection:
 458                 content_root = os.path.join(pwd, data_context().content.collection.directory)
 459             else:
 460                 content_root = ansible_root
 461 
 462             cmd = generate_command(args, python_interpreter, os.path.join(ansible_root, 'bin'), content_root, options, exclude, require)
 463 
 464             if httptester_id:
 465                 cmd += ['--inject-httptester']
 466 
 467             if isinstance(args, TestConfig):
 468                 if args.coverage and not args.coverage_label:
 469                     cmd += ['--coverage-label', 'remote-%s-%s' % (remote.platform, remote.version)]
 470 
 471             if isinstance(args, IntegrationConfig):
 472                 if not args.allow_destructive:
 473                     cmd.append('--allow-destructive')
 474 
 475             # remote instances are only expected to have a single python version available
 476             if isinstance(args, UnitsConfig) and not args.python:
 477                 cmd += ['--python', 'default']
 478 
 479         if isinstance(args, IntegrationConfig):
 480             cloud_platforms = get_cloud_providers(args)
 481 
 482             for cloud_platform in cloud_platforms:
 483                 ssh_options += cloud_platform.get_remote_ssh_options()
 484 
 485         try:
 486             manage.ssh(cmd, ssh_options)
 487             success = True
 488         finally:
 489             download = False
 490 
 491             if remote.platform != 'windows':
 492                 download = True
 493 
 494             if isinstance(args, ShellConfig):
 495                 if args.raw:
 496                     download = False
 497 
 498             if download and content_root:
 499                 local_test_root = os.path.dirname(os.path.join(data_context().content.root, data_context().content.results_path))
 500 
 501                 remote_results_root = os.path.join(content_root, data_context().content.results_path)
 502                 remote_results_name = os.path.basename(remote_results_root)
 503                 remote_temp_path = os.path.join('/tmp', remote_results_name)
 504 
 505                 # AIX cp and GNU cp provide different options, no way could be found to have a common
 506                 # pattern and achieve the same goal
 507                 cp_opts = '-hr' if remote.platform in ['aix', 'ibmi'] else '-a'
 508 
 509                 manage.ssh('rm -rf {0} && mkdir {0} && cp {1} {2}/* {0}/ && chmod -R a+r {0}'.format(remote_temp_path, cp_opts, remote_results_root))
 510                 manage.download(remote_temp_path, local_test_root)
 511     finally:
 512         if args.remote_terminate == 'always' or (args.remote_terminate == 'success' and success):
 513             core_ci.stop()
 514 
 515         if httptester_id:
 516             docker_rm(args, httptester_id)
 517

 519 def generate_command(args, python_interpreter, ansible_bin_path, content_root, options, exclude, require):
 520     """
 521     :type args: EnvironmentConfig
 522     :type python_interpreter: str | None
 523     :type ansible_bin_path: str
 524     :type content_root: str
 525     :type options: dict[str, int]
 526     :type exclude: list[str]
 527     :type require: list[str]
 528     :rtype: list[str]
 529     """
 530     options['--color'] = 1
 531 
 532     cmd = [os.path.join(ansible_bin_path, 'ansible-test')]
 533 
 534     if python_interpreter:
 535         cmd = [python_interpreter] + cmd
 536 
 537     # Force the encoding used during delegation.
 538     # This is only needed because ansible-test relies on Python's file system encoding.
 539     # Environments that do not have the locale configured are thus unable to work with unicode file paths.
 540     # Examples include FreeBSD and some Linux containers.
 541     env_vars = dict(
 542         LC_ALL='en_US.UTF-8',
 543         ANSIBLE_TEST_CONTENT_ROOT=content_root,
 544     )
 545 
 546     env_args = ['%s=%s' % (key, env_vars[key]) for key in sorted(env_vars)]
 547 
 548     cmd = ['/usr/bin/env'] + env_args + cmd
 549 
 550     cmd += list(filter_options(args, sys.argv[1:], options, exclude, require))
 551     cmd += ['--color', 'yes' if args.color else 'no']
 552 
 553     if args.requirements:
 554         cmd += ['--requirements']
 555 
 556     if isinstance(args, ShellConfig):
 557         cmd = create_shell_command(cmd)
 558     elif isinstance(args, SanityConfig):
 559         base_branch = args.base_branch or get_ci_provider().get_base_branch()
 560 
 561         if base_branch:
 562             cmd += ['--base-branch', base_branch]
 563 
 564     return cmd

 567 def filter_options(args, argv, options, exclude, require):
 568     """
 569     :type args: EnvironmentConfig
 570     :type argv: list[str]
 571     :type options: dict[str, int]
 572     :type exclude: list[str]
 573     :type require: list[str]
 574     :rtype: collections.Iterable[str]
 575     """
 576     options = options.copy()
 577 
 578     options['--requirements'] = 0
 579     options['--truncate'] = 1
 580     options['--redact'] = 0
 581     options['--no-redact'] = 0
 582 
 583     if isinstance(args, TestConfig):
 584         options.update({
 585             '--changed': 0,
 586             '--tracked': 0,
 587             '--untracked': 0,
 588             '--ignore-committed': 0,
 589             '--ignore-staged': 0,
 590             '--ignore-unstaged': 0,
 591             '--changed-from': 1,
 592             '--changed-path': 1,
 593             '--metadata': 1,
 594             '--exclude': 1,
 595             '--require': 1,
 596         })
 597     elif isinstance(args, SanityConfig):
 598         options.update({
 599             '--base-branch': 1,
 600         })
 601 
 602     if isinstance(args, IntegrationConfig):
 603         options.update({
 604             '--no-temp-unicode': 0,
 605             '--no-pip-check': 0,
 606         })
 607 
 608     if isinstance(args, (NetworkIntegrationConfig, WindowsIntegrationConfig)):
 609         options.update({
 610             '--inventory': 1,
 611         })
 612 
 613     remaining = 0
 614 
 615     for arg in argv:
 616         if not arg.startswith('-') and remaining:
 617             remaining -= 1
 618             continue
 619 
 620         remaining = 0
 621 
 622         parts = arg.split('=', 1)
 623         key = parts[0]
 624 
 625         if key in options:
 626             remaining = options[key] - len(parts) + 1
 627             continue
 628 
 629         yield arg
 630 
 631     for arg in args.delegate_args:
 632         yield arg
 633 
 634     for target in exclude:
 635         yield '--exclude'
 636         yield target
 637 
 638     for target in require:
 639         yield '--require'
 640         yield target
 641 
 642     if isinstance(args, TestConfig):
 643         if args.metadata_path:
 644             yield '--metadata'
 645             yield args.metadata_path
 646 
 647     yield '--truncate'
 648     yield '%d' % args.truncate
 649 
 650     if args.redact:
 651         yield '--redact'
 652     else:
 653         yield '--no-redact'
 654 
 655     if isinstance(args, IntegrationConfig):
 656         if args.no_temp_unicode:
 657             yield '--no-temp-unicode'
 658 
 659         if not args.pip_check:
 660             yield '--no-pip-check'
