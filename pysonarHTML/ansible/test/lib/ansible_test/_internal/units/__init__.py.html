  54 def command_units(args):
  55     """
  56     :type args: UnitsConfig
  57     """
  58     handle_layout_messages(data_context().content.unit_messages)
  59 
  60     changes = get_changes_filter(args)
  61     require = args.require + changes
  62     include = walk_internal_targets(walk_units_targets(), args.include, args.exclude, require)
  63 
  64     paths = [target.path for target in include]
  65     remote_paths = [path for path in paths
  66                     if is_subdir(path, data_context().content.unit_module_path)
  67                     or is_subdir(path, data_context().content.unit_module_utils_path)]
  68 
  69     if not paths:
  70         raise AllTargetsSkipped()
  71 
  72     if args.python and args.python in REMOTE_ONLY_PYTHON_VERSIONS and not remote_paths:
  73         raise AllTargetsSkipped()
  74 
  75     if args.delegate:
  76         raise Delegate(require=changes, exclude=args.exclude)
  77 
  78     version_commands = []
  79 
  80     available_versions = sorted(get_available_python_versions(list(SUPPORTED_PYTHON_VERSIONS)).keys())
  81 
  82     for version in SUPPORTED_PYTHON_VERSIONS:
  83         # run all versions unless version given, in which case run only that version
  84         if args.python and version != args.python_version:
  85             continue
  86 
  87         if not args.python and version not in available_versions:
  88             display.warning("Skipping unit tests on Python %s due to missing interpreter." % version)
  89             continue
  90 
  91         if args.requirements_mode != 'skip':
  92             install_command_requirements(args, version)
  93 
  94         env = ansible_environment(args)
  95 
  96         cmd = [
  97             'pytest',
  98             '--boxed',
  99             '-r', 'a',
 100             '-n', str(args.num_workers) if args.num_workers else 'auto',
 101             '--color',
 102             'yes' if args.color else 'no',
 103             '-p', 'no:cacheprovider',
 104             '-c', os.path.join(ANSIBLE_TEST_DATA_ROOT, 'pytest.ini'),
 105             '--junit-xml', os.path.join(ResultType.JUNIT.path, 'python%s-units.xml' % version),
 106         ]
 107 
 108         if not data_context().content.collection:
 109             cmd.append('--durations=25')
 110 
 111         if version != '2.6':
 112             # added in pytest 4.5.0, which requires python 2.7+
 113             cmd.append('--strict-markers')
 114 
 115         plugins = []
 116 
 117         if args.coverage:
 118             plugins.append('ansible_pytest_coverage')
 119 
 120         if data_context().content.collection:
 121             plugins.append('ansible_pytest_collections')
 122 
 123         if plugins:
 124             env['PYTHONPATH'] += ':%s' % os.path.join(ANSIBLE_TEST_DATA_ROOT, 'pytest/plugins')
 125             env['PYTEST_PLUGINS'] = ','.join(plugins)
 126 
 127         if args.collect_only:
 128             cmd.append('--collect-only')
 129 
 130         if args.verbosity:
 131             cmd.append('-' + ('v' * args.verbosity))
 132 
 133         if version in REMOTE_ONLY_PYTHON_VERSIONS:
 134             test_paths = remote_paths
 135         else:
 136             test_paths = paths
 137 
 138         if not test_paths:
 139             continue
 140 
 141         cmd.extend(test_paths)
 142 
 143         version_commands.append((version, cmd, env))
 144 
 145     if args.requirements_mode == 'only':
 146         sys.exit()
 147 
 148     for version, command, env in version_commands:
 149         check_pyyaml(args, version)
 150 
 151         display.info('Unit test with Python %s' % version)
 152 
 153         try:
 154             with coverage_context(args):
 155                 intercept_command(args, command, target_name='units', env=env, python_version=version)
 156         except SubprocessError as ex:
 157             # pytest exits with status code 5 when all tests are skipped, which isn't an error for our use case
 158             if ex.status != 5:
 159                 raise
