 152 def main():
 153     """Main program function."""
 154     try:
 155         os.chdir(data_context().content.root)
 156         initialize_cloud_plugins()
 157         sanity_init()
 158         args = parse_args()
 159         config = args.config(args)  # type: CommonConfig
 160         display.verbosity = config.verbosity
 161         display.truncate = config.truncate
 162         display.redact = config.redact
 163         display.color = config.color
 164         display.info_stderr = config.info_stderr
 165         check_startup()
 166         check_delegation_args(config)
 167         configure_timeout(config)
 168 
 169         display.info('RLIMIT_NOFILE: %s' % (CURRENT_RLIMIT_NOFILE,), verbosity=2)
 170         display.info('MAXFD: %d' % MAXFD, verbosity=2)
 171 
 172         try:
 173             args.func(config)
 174             delegate_args = None
 175         except Delegate as ex:
 176             # save delegation args for use once we exit the exception handler
 177             delegate_args = (ex.exclude, ex.require, ex.integration_targets)
 178 
 179         if delegate_args:
 180             # noinspection PyTypeChecker
 181             delegate(config, *delegate_args)
 182 
 183         display.review_warnings()
 184     except ApplicationWarning as ex:
 185         display.warning(u'%s' % ex)
 186         sys.exit(0)
 187     except ApplicationError as ex:
 188         display.error(u'%s' % ex)
 189         sys.exit(1)
 190     except KeyboardInterrupt:
 191         sys.exit(2)
 192     except IOError as ex:
 193         if ex.errno == errno.EPIPE:
 194             sys.exit(3)
 195         raise

 671 def key_value(argparse, value):  # type: (argparse_module, str) -> t.Tuple[str, str]
 672     """Type parsing and validation for argparse key/value pairs separated by an '=' character."""
 673     parts = value.split('=')
 674 
 675     if len(parts) != 2:
 676         raise argparse.ArgumentTypeError('"%s" must be in the format "key=value"' % value)
 677 
 678     return parts[0], parts[1]

1161 def complete_network_testcase(prefix, parsed_args, **_):
1162     """
1163     :type prefix: unicode
1164     :type parsed_args: any
1165     :rtype: list[str]
1166     """
1167     testcases = []
1168 
1169     # since testcases are module specific, don't autocomplete if more than one
1170     # module is specidied
1171     if len(parsed_args.include) != 1:
1172         return []
1173 
1174     test_dir = os.path.join(data_context().content.integration_targets_path, parsed_args.include[0], 'tests')
1175     connection_dirs = data_context().content.get_dirs(test_dir)
1176 
1177     for connection_dir in connection_dirs:
1178         for testcase in [os.path.basename(path) for path in data_context().content.get_files(connection_dir)]:
1179             if testcase.startswith(prefix):
1180                 testcases.append(testcase.split('.')[0])
1181 
1182     return testcases
