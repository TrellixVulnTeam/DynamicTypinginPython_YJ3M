  16 class WrappedThread(threading.Thread):
  17     """Wrapper around Thread which captures results and exceptions."""
  18     def __init__(self, action):
  19         """
  20         :type action: () -> any
  21         """
  22         # noinspection PyOldStyleClasses
  23         super(WrappedThread, self).__init__()
  24         self._result = queue.Queue()
  25         self.action = action
  26         self.result = None
  27 
  28     def run(self):
  29         """
  30         Run action and capture results or exception.
  31         Do not override. Do not call directly. Executed by the start() method.
  32         """
  33         # We truly want to catch anything that the worker thread might do including call sys.exit.
  34         # Therefore we catch *everything* (including old-style class exceptions)
  35         # noinspection PyBroadException, PyPep8
  36         try:
  37             self._result.put((self.action(), None))
  38         # pylint: disable=locally-disabled, bare-except
  39         except:  # noqa
  40             self._result.put((None, sys.exc_info()))
  41 
  42     def wait_for_result(self):
  43         """
  44         Wait for thread to exit and return the result or raise an exception.
  45         :rtype: any
  46         """
  47         result, exception = self._result.get()
  48 
  49         if exception:
  50             if sys.version_info[0] > 2:
  51                 raise exception[1].with_traceback(exception[2])
  52             # noinspection PyRedundantParentheses
  53             exec('raise exception[0], exception[1], exception[2]')  # pylint: disable=locally-disabled, exec-used
  54 
  55         self.result = result
  56 
  57         return result
