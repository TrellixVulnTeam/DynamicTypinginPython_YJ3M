  45 def find_target_completion(target_func, prefix):
  46     """
  47     :type target_func: () -> collections.Iterable[CompletionTarget]
  48     :type prefix: unicode
  49     :rtype: list[str]
  50     """
  51     try:
  52         targets = target_func()
  53         short = os.environ.get('COMP_TYPE') == '63'  # double tab completion from bash
  54         matches = walk_completion_targets(targets, prefix, short)
  55         return matches
  56     except Exception as ex:  # pylint: disable=locally-disabled, broad-except
  57         return [u'%s' % ex]

  60 def walk_completion_targets(targets, prefix, short=False):
  61     """
  62     :type targets: collections.Iterable[CompletionTarget]
  63     :type prefix: str
  64     :type short: bool
  65     :rtype: tuple[str]
  66     """
  67     aliases = set(alias for target in targets for alias in target.aliases)
  68 
  69     if prefix.endswith('/') and prefix in aliases:
  70         aliases.remove(prefix)
  71 
  72     matches = [alias for alias in aliases if alias.startswith(prefix) and '/' not in alias[len(prefix):-1]]
  73 
  74     if short:
  75         offset = len(os.path.dirname(prefix))
  76         if offset:
  77             offset += 1
  78             relative_matches = [match[offset:] for match in matches if len(match) > offset]
  79             if len(relative_matches) > 1:
  80                 matches = relative_matches
  81 
  82     return tuple(sorted(matches))

  85 def walk_internal_targets(targets, includes=None, excludes=None, requires=None):
  86     """
  87     :type targets: collections.Iterable[T <= CompletionTarget]
  88     :type includes: list[str]
  89     :type excludes: list[str]
  90     :type requires: list[str]
  91     :rtype: tuple[T <= CompletionTarget]
  92     """
  93     targets = tuple(targets)
  94 
  95     include_targets = sorted(filter_targets(targets, includes, errors=True, directories=False), key=lambda include_target: include_target.name)
  96 
  97     if requires:
  98         require_targets = set(filter_targets(targets, requires, errors=True, directories=False))
  99         include_targets = [require_target for require_target in include_targets if require_target in require_targets]
 100 
 101     if excludes:
 102         list(filter_targets(targets, excludes, errors=True, include=False, directories=False))
 103 
 104     internal_targets = set(filter_targets(include_targets, excludes, errors=False, include=False, directories=False))
 105     return tuple(sorted(internal_targets, key=lambda sort_target: sort_target.name))

 108 def filter_targets(targets,  # type: t.Iterable[TCompletionTarget]
 109                    patterns,  # type: t.List[str]
 110                    include=True,  # type: bool
 111                    directories=True,  # type: bool
 112                    errors=True,  # type: bool
 113                    ):  # type: (...) -> t.Iterable[TCompletionTarget]
 114     """Iterate over the given targets and filter them based on the supplied arguments."""
 115     unmatched = set(patterns or ())
 116     compiled_patterns = dict((p, re.compile('^%s$' % p)) for p in patterns) if patterns else None
 117 
 118     for target in targets:
 119         matched_directories = set()
 120         match = False
 121 
 122         if patterns:
 123             for alias in target.aliases:
 124                 for pattern in patterns:
 125                     if compiled_patterns[pattern].match(alias):
 126                         match = True
 127 
 128                         try:
 129                             unmatched.remove(pattern)
 130                         except KeyError:
 131                             pass
 132 
 133                         if alias.endswith('/'):
 134                             if target.base_path and len(target.base_path) > len(alias):
 135                                 matched_directories.add(target.base_path)
 136                             else:
 137                                 matched_directories.add(alias)
 138         elif include:
 139             match = True
 140             if not target.base_path:
 141                 matched_directories.add('.')
 142             for alias in target.aliases:
 143                 if alias.endswith('/'):
 144                     if target.base_path and len(target.base_path) > len(alias):
 145                         matched_directories.add(target.base_path)
 146                     else:
 147                         matched_directories.add(alias)
 148 
 149         if match != include:
 150             continue
 151 
 152         if directories and matched_directories:
 153             yield DirectoryTarget(sorted(matched_directories, key=len)[0], target.modules)
 154         else:
 155             yield target
 156 
 157     if errors:
 158         if unmatched:
 159             raise TargetPatternsNotMatched(unmatched)

 232 def walk_integration_targets():
 233     """
 234     :rtype: collections.Iterable[IntegrationTarget]
 235     """
 236     path = data_context().content.integration_targets_path
 237     modules = frozenset(target.module for target in walk_module_targets())
 238     paths = data_context().content.walk_files(path)
 239     prefixes = load_integration_prefixes()
 240     targets_path_tuple = tuple(path.split(os.path.sep))
 241 
 242     entry_dirs = (
 243         'defaults',
 244         'files',
 245         'handlers',
 246         'meta',
 247         'tasks',
 248         'templates',
 249         'vars',
 250     )
 251 
 252     entry_files = (
 253         'main.yml',
 254         'main.yaml',
 255     )
 256 
 257     entry_points = []
 258 
 259     for entry_dir in entry_dirs:
 260         for entry_file in entry_files:
 261             entry_points.append(os.path.join(os.path.sep, entry_dir, entry_file))
 262 
 263     # any directory with at least one file is a target
 264     path_tuples = set(tuple(os.path.dirname(p).split(os.path.sep))
 265                       for p in paths)
 266 
 267     # also detect targets which are ansible roles, looking for standard entry points
 268     path_tuples.update(tuple(os.path.dirname(os.path.dirname(p)).split(os.path.sep))
 269                        for p in paths if any(p.endswith(entry_point) for entry_point in entry_points))
 270 
 271     # remove the top-level directory if it was included
 272     if targets_path_tuple in path_tuples:
 273         path_tuples.remove(targets_path_tuple)
 274 
 275     previous_path_tuple = None
 276     paths = []
 277 
 278     for path_tuple in sorted(path_tuples):
 279         if previous_path_tuple and previous_path_tuple == path_tuple[:len(previous_path_tuple)]:
 280             # ignore nested directories
 281             continue
 282 
 283         previous_path_tuple = path_tuple
 284         paths.append(os.path.sep.join(path_tuple))
 285 
 286     for path in paths:
 287         yield IntegrationTarget(path, modules, prefixes)

 290 def load_integration_prefixes():
 291     """
 292     :rtype: dict[str, str]
 293     """
 294     path = data_context().content.integration_path
 295     file_paths = sorted(f for f in data_context().content.get_files(path) if os.path.splitext(os.path.basename(f))[0] == 'target-prefixes')
 296     prefixes = {}
 297 
 298     for file_path in file_paths:
 299         prefix = os.path.splitext(file_path)[1][1:]
 300         prefixes.update(dict((k, prefix) for k in read_text_file(file_path).splitlines()))
 301 
 302     return prefixes

 305 def walk_test_targets(path=None, module_path=None, extensions=None, prefix=None, extra_dirs=None, include_symlinks=False, include_symlinked_directories=False):
 306     """
 307     :type path: str | None
 308     :type module_path: str | None
 309     :type extensions: tuple[str] | None
 310     :type prefix: str | None
 311     :type extra_dirs: tuple[str] | None
 312     :type include_symlinks: bool
 313     :type include_symlinked_directories: bool
 314     :rtype: collections.Iterable[TestTarget]
 315     """
 316     if path:
 317         file_paths = data_context().content.walk_files(path, include_symlinked_directories=include_symlinked_directories)
 318     else:
 319         file_paths = data_context().content.all_files(include_symlinked_directories=include_symlinked_directories)
 320 
 321     for file_path in file_paths:
 322         name, ext = os.path.splitext(os.path.basename(file_path))
 323 
 324         if extensions and ext not in extensions:
 325             continue
 326 
 327         if prefix and not name.startswith(prefix):
 328             continue
 329 
 330         symlink = os.path.islink(to_bytes(file_path.rstrip(os.path.sep)))
 331 
 332         if symlink and not include_symlinks:
 333             continue
 334 
 335         yield TestTarget(file_path, module_path, prefix, path, symlink)
 336 
 337     file_paths = []
 338 
 339     if extra_dirs:
 340         for extra_dir in extra_dirs:
 341             for file_path in data_context().content.get_files(extra_dir):
 342                 file_paths.append(file_path)
 343 
 344     for file_path in file_paths:
 345         symlink = os.path.islink(to_bytes(file_path.rstrip(os.path.sep)))
 346 
 347         if symlink and not include_symlinks:
 348             continue
 349 
 350         yield TestTarget(file_path, module_path, prefix, path, symlink)

 353 def analyze_integration_target_dependencies(integration_targets):
 354     """
 355     :type integration_targets: list[IntegrationTarget]
 356     :rtype: dict[str,set[str]]
 357     """
 358     real_target_root = os.path.realpath(data_context().content.integration_targets_path) + '/'
 359 
 360     role_targets = [target for target in integration_targets if target.type == 'role']
 361     hidden_role_target_names = set(target.name for target in role_targets if 'hidden/' in target.aliases)
 362 
 363     dependencies = collections.defaultdict(set)
 364 
 365     # handle setup dependencies
 366     for target in integration_targets:
 367         for setup_target_name in target.setup_always + target.setup_once:
 368             dependencies[setup_target_name].add(target.name)
 369 
 370     # handle target dependencies
 371     for target in integration_targets:
 372         for need_target in target.needs_target:
 373             dependencies[need_target].add(target.name)
 374 
 375     # handle symlink dependencies between targets
 376     # this use case is supported, but discouraged
 377     for target in integration_targets:
 378         for path in data_context().content.walk_files(target.path):
 379             if not os.path.islink(to_bytes(path.rstrip(os.path.sep))):
 380                 continue
 381 
 382             real_link_path = os.path.realpath(path)
 383 
 384             if not real_link_path.startswith(real_target_root):
 385                 continue
 386 
 387             link_target = real_link_path[len(real_target_root):].split('/')[0]
 388 
 389             if link_target == target.name:
 390                 continue
 391 
 392             dependencies[link_target].add(target.name)
 393 
 394     # intentionally primitive analysis of role meta to avoid a dependency on pyyaml
 395     # script based targets are scanned as they may execute a playbook with role dependencies
 396     for target in integration_targets:
 397         meta_dir = os.path.join(target.path, 'meta')
 398 
 399         if not os.path.isdir(meta_dir):
 400             continue
 401 
 402         meta_paths = data_context().content.get_files(meta_dir)
 403 
 404         for meta_path in meta_paths:
 405             if os.path.exists(meta_path):
 406                 # try and decode the file as a utf-8 string, skip if it contains invalid chars (binary file)
 407                 try:
 408                     meta_lines = read_text_file(meta_path).splitlines()
 409                 except UnicodeDecodeError:
 410                     continue
 411 
 412                 for meta_line in meta_lines:
 413                     if re.search(r'^ *#.*$', meta_line):
 414                         continue
 415 
 416                     if not meta_line.strip():
 417                         continue
 418 
 419                     for hidden_target_name in hidden_role_target_names:
 420                         if hidden_target_name in meta_line:
 421                             dependencies[hidden_target_name].add(target.name)
 422 
 423     while True:
 424         changes = 0
 425 
 426         for dummy, dependent_target_names in dependencies.items():
 427             for dependent_target_name in list(dependent_target_names):
 428                 new_target_names = dependencies.get(dependent_target_name)
 429 
 430                 if new_target_names:
 431                     for new_target_name in new_target_names:
 432                         if new_target_name not in dependent_target_names:
 433                             dependent_target_names.add(new_target_name)
 434                             changes += 1
 435 
 436         if not changes:
 437             break
 438 
 439     for target_name in sorted(dependencies):
 440         consumers = dependencies[target_name]
 441 
 442         if not consumers:
 443             continue
 444 
 445         display.info('%s:' % target_name, verbosity=4)
 446 
 447         for consumer in sorted(consumers):
 448             display.info('  %s' % consumer, verbosity=4)
 449 
 450     return dependencies

 453 class CompletionTarget:
 454     """Command-line argument completion target base class."""
 455     __metaclass__ = abc.ABCMeta
 456 
 457     def __init__(self):
 458         self.name = None
 459         self.path = None
 460         self.base_path = None
 461         self.modules = tuple()
 462         self.aliases = tuple()
 463 
 464     def __eq__(self, other):
 465         if isinstance(other, CompletionTarget):
 466             return self.__repr__() == other.__repr__()
 467 
 468         return False
 469 
 470     def __ne__(self, other):
 471         return not self.__eq__(other)
 472 
 473     def __lt__(self, other):
 474         return self.name.__lt__(other.name)
 475 
 476     def __gt__(self, other):
 477         return self.name.__gt__(other.name)
 478 
 479     def __hash__(self):
 480         return hash(self.__repr__())
 481 
 482     def __repr__(self):
 483         if self.modules:
 484             return '%s (%s)' % (self.name, ', '.join(self.modules))
 485 
 486         return self.name

 503 class TestTarget(CompletionTarget):
 504     """Generic test target."""
 505     def __init__(self, path, module_path, module_prefix, base_path, symlink=None):
 506         """
 507         :type path: str
 508         :type module_path: str | None
 509         :type module_prefix: str | None
 510         :type base_path: str
 511         :type symlink: bool | None
 512         """
 513         super(TestTarget, self).__init__()
 514 
 515         if symlink is None:
 516             symlink = os.path.islink(to_bytes(path.rstrip(os.path.sep)))
 517 
 518         self.name = path
 519         self.path = path
 520         self.base_path = base_path + '/' if base_path else None
 521         self.symlink = symlink
 522 
 523         name, ext = os.path.splitext(os.path.basename(self.path))
 524 
 525         if module_path and is_subdir(path, module_path) and name != '__init__' and ext in MODULE_EXTENSIONS:
 526             self.module = name[len(module_prefix or ''):].lstrip('_')
 527             self.modules = (self.module,)
 528         else:
 529             self.module = None
 530             self.modules = tuple()
 531 
 532         aliases = [self.path, self.module]
 533         parts = self.path.split('/')
 534 
 535         for i in range(1, len(parts)):
 536             alias = '%s/' % '/'.join(parts[:i])
 537             aliases.append(alias)
 538 
 539         aliases = [a for a in aliases if a]
 540 
 541         self.aliases = tuple(sorted(aliases))

 544 class IntegrationTarget(CompletionTarget):
 545     """Integration test target."""
 546     non_posix = frozenset((
 547         'network',
 548         'windows',
 549     ))
 550 
 551     categories = frozenset(non_posix | frozenset((
 552         'posix',
 553         'module',
 554         'needs',
 555         'skip',
 556     )))
 557 
 558     def __init__(self, path, modules, prefixes):
 559         """
 560         :type path: str
 561         :type modules: frozenset[str]
 562         :type prefixes: dict[str, str]
 563         """
 564         super(IntegrationTarget, self).__init__()
 565 
 566         self.relative_path = os.path.relpath(path, data_context().content.integration_targets_path)
 567         self.name = self.relative_path.replace(os.path.sep, '.')
 568         self.path = path
 569 
 570         # script_path and type
 571 
 572         file_paths = data_context().content.get_files(path)
 573         runme_path = os.path.join(path, 'runme.sh')
 574 
 575         if runme_path in file_paths:
 576             self.type = 'script'
 577             self.script_path = runme_path
 578         else:
 579             self.type = 'role'  # ansible will consider these empty roles, so ansible-test should as well
 580             self.script_path = None
 581 
 582         # static_aliases
 583 
 584         aliases_path = os.path.join(path, 'aliases')
 585 
 586         if aliases_path in file_paths:
 587             static_aliases = tuple(read_lines_without_comments(aliases_path, remove_blank_lines=True))
 588         else:
 589             static_aliases = tuple()
 590 
 591         # modules
 592 
 593         if self.name in modules:
 594             module_name = self.name
 595         elif self.name.startswith('win_') and self.name[4:] in modules:
 596             module_name = self.name[4:]
 597         else:
 598             module_name = None
 599 
 600         self.modules = tuple(sorted(a for a in static_aliases + tuple([module_name]) if a in modules))
 601 
 602         # groups
 603 
 604         groups = [self.type]
 605         groups += [a for a in static_aliases if a not in modules]
 606         groups += ['module/%s' % m for m in self.modules]
 607 
 608         if not self.modules:
 609             groups.append('non_module')
 610 
 611         if 'destructive' not in groups:
 612             groups.append('non_destructive')
 613 
 614         if '_' in self.name:
 615             prefix = self.name[:self.name.find('_')]
 616         else:
 617             prefix = None
 618 
 619         if prefix in prefixes:
 620             group = prefixes[prefix]
 621 
 622             if group != prefix:
 623                 group = '%s/%s' % (group, prefix)
 624 
 625             groups.append(group)
 626 
 627         if self.name.startswith('win_'):
 628             groups.append('windows')
 629 
 630         if self.name.startswith('connection_'):
 631             groups.append('connection')
 632 
 633         if self.name.startswith('setup_') or self.name.startswith('prepare_'):
 634             groups.append('hidden')
 635 
 636         if self.type not in ('script', 'role'):
 637             groups.append('hidden')
 638 
 639         targets_relative_path = data_context().content.integration_targets_path
 640 
 641         # Collect skip entries before group expansion to avoid registering more specific skip entries as less specific versions.
 642         self.skips = tuple(g for g in groups if g.startswith('skip/'))
 643 
 644         # Collect file paths before group expansion to avoid including the directories.
 645         # Ignore references to test targets, as those must be defined using `needs/target/*` or other target references.
 646         self.needs_file = tuple(sorted(set('/'.join(g.split('/')[2:]) for g in groups if
 647                                            g.startswith('needs/file/') and not g.startswith('needs/file/%s/' % targets_relative_path))))
 648 
 649         # network platform
 650         networks = [g.split('/')[1] for g in groups if g.startswith('network/')]
 651         self.network_platform = networks[0] if networks else None
 652 
 653         for group in itertools.islice(groups, 0, len(groups)):
 654             if '/' in group:
 655                 parts = group.split('/')
 656                 for i in range(1, len(parts)):
 657                     groups.append('/'.join(parts[:i]))
 658 
 659         if not any(g in self.non_posix for g in groups):
 660             groups.append('posix')
 661 
 662         # aliases
 663 
 664         aliases = [self.name] + \
 665                   ['%s/' % g for g in groups] + \
 666                   ['%s/%s' % (g, self.name) for g in groups if g not in self.categories]
 667 
 668         if 'hidden/' in aliases:
 669             aliases = ['hidden/'] + ['hidden/%s' % a for a in aliases if not a.startswith('hidden/')]
 670 
 671         self.aliases = tuple(sorted(set(aliases)))
 672 
 673         # configuration
 674 
 675         self.setup_once = tuple(sorted(set(g.split('/')[2] for g in groups if g.startswith('setup/once/'))))
 676         self.setup_always = tuple(sorted(set(g.split('/')[2] for g in groups if g.startswith('setup/always/'))))
