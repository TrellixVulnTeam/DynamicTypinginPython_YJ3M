  55 def create_payload(args, dst_path):  # type: (CommonConfig, str) -> None
  56     """Create a payload for delegation."""
  57     if args.explain:
  58         return
  59 
  60     files = list(data_context().ansible_source)
  61     filters = {}
  62 
  63     def make_executable(tar_info):  # type: (tarfile.TarInfo) -> t.Optional[tarfile.TarInfo]
  64         """Make the given file executable."""
  65         tar_info.mode |= stat.S_IXUSR | stat.S_IXOTH | stat.S_IXGRP
  66         return tar_info
  67 
  68     if not ANSIBLE_SOURCE_ROOT:
  69         # reconstruct the bin directory which is not available when running from an ansible install
  70         files.extend(create_temporary_bin_files(args))
  71         filters.update(dict((os.path.join('ansible', path[3:]), make_executable) for path in ANSIBLE_BIN_SYMLINK_MAP.values() if path.startswith('../')))
  72 
  73     if not data_context().content.is_ansible:
  74         # exclude unnecessary files when not testing ansible itself
  75         files = [f for f in files if
  76                  is_subdir(f[1], 'bin/') or
  77                  is_subdir(f[1], 'lib/ansible/') or
  78                  is_subdir(f[1], 'test/lib/ansible_test/')]
  79 
  80         if not isinstance(args, (ShellConfig, IntegrationConfig)):
  81             # exclude built-in ansible modules when they are not needed
  82             files = [f for f in files if not is_subdir(f[1], 'lib/ansible/modules/') or f[1] == 'lib/ansible/modules/__init__.py']
  83 
  84         collection_layouts = data_context().create_collection_layouts()
  85 
  86         content_files = []
  87         extra_files = []
  88 
  89         for layout in collection_layouts:
  90             if layout == data_context().content:
  91                 # include files from the current collection (layout.collection.directory will be added later)
  92                 content_files.extend((os.path.join(layout.root, path), path) for path in data_context().content.all_files())
  93             else:
  94                 # include files from each collection in the same collection root as the content being tested
  95                 extra_files.extend((os.path.join(layout.root, path), os.path.join(layout.collection.directory, path)) for path in layout.all_files())
  96     else:
  97         # when testing ansible itself the ansible source is the content
  98         content_files = files
  99         # there are no extra files when testing ansible itself
 100         extra_files = []
 101 
 102     for callback in data_context().payload_callbacks:
 103         # execute callbacks only on the content paths
 104         # this is done before placing them in the appropriate subdirectory (see below)
 105         callback(content_files)
 106 
 107     # place ansible source files under the 'ansible' directory on the delegated host
 108     files = [(src, os.path.join('ansible', dst)) for src, dst in files]
 109 
 110     if data_context().content.collection:
 111         # place collection files under the 'ansible_collections/{namespace}/{collection}' directory on the delegated host
 112         files.extend((src, os.path.join(data_context().content.collection.directory, dst)) for src, dst in content_files)
 113         # extra files already have the correct destination path
 114         files.extend(extra_files)
 115 
 116     # maintain predictable file order
 117     files = sorted(set(files))
 118 
 119     display.info('Creating a payload archive containing %d files...' % len(files), verbosity=1)
 120 
 121     start = time.time()
 122 
 123     with tarfile.TarFile.open(dst_path, mode='w:gz', compresslevel=4, format=tarfile.GNU_FORMAT) as tar:
 124         for src, dst in files:
 125             display.info('%s -> %s' % (src, dst), verbosity=4)
 126             tar.add(src, dst, filter=filters.get(dst))
 127 
 128     duration = time.time() - start
 129     payload_size_bytes = os.path.getsize(dst_path)
 130 
 131     display.info('Created a %d byte payload archive containing %d files in %d seconds.' % (payload_size_bytes, len(files), duration), verbosity=1)

 134 def create_temporary_bin_files(args):  # type: (CommonConfig) -> t.Tuple[t.Tuple[str, str], ...]
 135     """Create a temporary ansible bin directory populated using the symlink map."""
 136     if args.explain:
 137         temp_path = '/tmp/ansible-tmp-bin'
 138     else:
 139         temp_path = tempfile.mkdtemp(prefix='ansible', suffix='bin')
 140         atexit.register(remove_tree, temp_path)
 141 
 142         for name, dest in ANSIBLE_BIN_SYMLINK_MAP.items():
 143             path = os.path.join(temp_path, name)
 144             os.symlink(dest, path)
 145 
 146     return tuple((os.path.join(temp_path, name), os.path.join('bin', name)) for name in sorted(ANSIBLE_BIN_SYMLINK_MAP))
