  58 def categorize_changes(args, paths, verbose_command=None):
  59     """
  60     :type args: TestConfig
  61     :type paths: list[str]
  62     :type verbose_command: str
  63     :rtype: ChangeDescription
  64     """
  65     mapper = PathMapper(args)
  66 
  67     commands = {
  68         'sanity': set(),
  69         'units': set(),
  70         'integration': set(),
  71         'windows-integration': set(),
  72         'network-integration': set(),
  73     }
  74 
  75     focused_commands = collections.defaultdict(set)
  76 
  77     deleted_paths = set()
  78     original_paths = set()
  79     additional_paths = set()
  80     no_integration_paths = set()
  81 
  82     for path in paths:
  83         if not os.path.exists(path):
  84             deleted_paths.add(path)
  85             continue
  86 
  87         original_paths.add(path)
  88 
  89         dependent_paths = mapper.get_dependent_paths(path)
  90 
  91         if not dependent_paths:
  92             continue
  93 
  94         display.info('Expanded "%s" to %d dependent file(s):' % (path, len(dependent_paths)), verbosity=2)
  95 
  96         for dependent_path in dependent_paths:
  97             display.info(dependent_path, verbosity=2)
  98             additional_paths.add(dependent_path)
  99 
 100     additional_paths -= set(paths)  # don't count changed paths as additional paths
 101 
 102     if additional_paths:
 103         display.info('Expanded %d changed file(s) into %d additional dependent file(s).' % (len(paths), len(additional_paths)))
 104         paths = sorted(set(paths) | additional_paths)
 105 
 106     display.info('Mapping %d changed file(s) to tests.' % len(paths))
 107 
 108     none_count = 0
 109 
 110     for path in paths:
 111         tests = mapper.classify(path)
 112 
 113         if tests is None:
 114             focused_target = False
 115 
 116             display.info('%s -> all' % path, verbosity=1)
 117             tests = all_tests(args)  # not categorized, run all tests
 118             display.warning('Path not categorized: %s' % path)
 119         else:
 120             focused_target = tests.pop(FOCUSED_TARGET, False) and path in original_paths
 121 
 122             tests = dict((key, value) for key, value in tests.items() if value)
 123 
 124             if focused_target and not any('integration' in command for command in tests):
 125                 no_integration_paths.add(path)  # path triggers no integration tests
 126 
 127             if verbose_command:
 128                 result = '%s: %s' % (verbose_command, tests.get(verbose_command) or 'none')
 129 
 130                 # identify targeted integration tests (those which only target a single integration command)
 131                 if 'integration' in verbose_command and tests.get(verbose_command):
 132                     if not any('integration' in command for command in tests if command != verbose_command):
 133                         if focused_target:
 134                             result += ' (focused)'
 135 
 136                         result += ' (targeted)'
 137             else:
 138                 result = '%s' % tests
 139 
 140             if not tests.get(verbose_command):
 141                 # minimize excessive output from potentially thousands of files which do not trigger tests
 142                 none_count += 1
 143                 verbosity = 2
 144             else:
 145                 verbosity = 1
 146 
 147             if args.verbosity >= verbosity:
 148                 display.info('%s -> %s' % (path, result), verbosity=1)
 149 
 150         for command, target in tests.items():
 151             commands[command].add(target)
 152 
 153             if focused_target:
 154                 focused_commands[command].add(target)
 155 
 156     if none_count > 0 and args.verbosity < 2:
 157         display.notice('Omitted %d file(s) that triggered no tests.' % none_count)
 158 
 159     for command in commands:
 160         commands[command].discard('none')
 161 
 162         if any(target == 'all' for target in commands[command]):
 163             commands[command] = set(['all'])
 164 
 165     commands = dict((c, sorted(commands[c])) for c in commands if commands[c])
 166     focused_commands = dict((c, sorted(focused_commands[c])) for c in focused_commands)
 167 
 168     for command in commands:
 169         if commands[command] == ['all']:
 170             commands[command] = []  # changes require testing all targets, do not filter targets
 171 
 172     changes = ChangeDescription()
 173     changes.command = verbose_command
 174     changes.changed_paths = sorted(original_paths)
 175     changes.deleted_paths = sorted(deleted_paths)
 176     changes.regular_command_targets = commands
 177     changes.focused_command_targets = focused_commands
 178     changes.no_integration_paths = sorted(no_integration_paths)
 179 
 180     return changes

 183 class PathMapper:
 184     """Map file paths to test commands and targets."""
 185     def __init__(self, args):
 186         """
 187         :type args: TestConfig
 188         """
 189         self.args = args
 190         self.integration_all_target = get_integration_all_target(self.args)
 191 
 192         self.integration_targets = list(walk_integration_targets())
 193         self.module_targets = list(walk_module_targets())
 194         self.compile_targets = list(walk_compile_targets())
 195         self.units_targets = list(walk_units_targets())
 196         self.sanity_targets = list(walk_sanity_targets())
 197         self.powershell_targets = [target for target in self.sanity_targets if os.path.splitext(target.path)[1] in ('.ps1', '.psm1')]
 198         self.csharp_targets = [target for target in self.sanity_targets if os.path.splitext(target.path)[1] == '.cs']
 199 
 200         self.units_modules = set(target.module for target in self.units_targets if target.module)
 201         self.units_paths = set(a for target in self.units_targets for a in target.aliases)
 202         self.sanity_paths = set(target.path for target in self.sanity_targets)
 203 
 204         self.module_names_by_path = dict((target.path, target.module) for target in self.module_targets)
 205         self.integration_targets_by_name = dict((target.name, target) for target in self.integration_targets)
 206         self.integration_targets_by_alias = dict((a, target) for target in self.integration_targets for a in target.aliases)
 207 
 208         self.posix_integration_by_module = dict((m, target.name) for target in self.integration_targets
 209                                                 if 'posix/' in target.aliases for m in target.modules)
 210         self.windows_integration_by_module = dict((m, target.name) for target in self.integration_targets
 211                                                   if 'windows/' in target.aliases for m in target.modules)
 212         self.network_integration_by_module = dict((m, target.name) for target in self.integration_targets
 213                                                   if 'network/' in target.aliases for m in target.modules)
 214 
 215         self.prefixes = load_integration_prefixes()
 216         self.integration_dependencies = analyze_integration_target_dependencies(self.integration_targets)
 217 
 218         self.python_module_utils_imports = {}  # populated on first use to reduce overhead when not needed
 219         self.powershell_module_utils_imports = {}  # populated on first use to reduce overhead when not needed
 220         self.csharp_module_utils_imports = {}  # populated on first use to reduce overhead when not needed
 221 
 222         self.paths_to_dependent_targets = {}
 223 
 224         for target in self.integration_targets:
 225             for path in target.needs_file:
 226                 if path not in self.paths_to_dependent_targets:
 227                     self.paths_to_dependent_targets[path] = set()
 228 
 229                 self.paths_to_dependent_targets[path].add(target)

 231     def get_dependent_paths(self, path):
 232         """
 233         :type path: str
 234         :rtype: list[str]
 235         """
 236         unprocessed_paths = set(self.get_dependent_paths_non_recursive(path))
 237         paths = set()
 238 
 239         while unprocessed_paths:
 240             queued_paths = list(unprocessed_paths)
 241             paths |= unprocessed_paths
 242             unprocessed_paths = set()
 243 
 244             for queued_path in queued_paths:
 245                 new_paths = self.get_dependent_paths_non_recursive(queued_path)
 246 
 247                 for new_path in new_paths:
 248                     if new_path not in paths:
 249                         unprocessed_paths.add(new_path)
 250 
 251         return sorted(paths)
 252 
 253     def get_dependent_paths_non_recursive(self, path):
 254         """
 255         :type path: str
 256         :rtype: list[str]
 257         """
 258         paths = self.get_dependent_paths_internal(path)
 259         paths += [target.path + '/' for target in self.paths_to_dependent_targets.get(path, set())]
 260         paths = sorted(set(paths))
 261 
 262         return paths

 264     def get_dependent_paths_internal(self, path):
 265         """
 266         :type path: str
 267         :rtype: list[str]
 268         """
 269         ext = os.path.splitext(os.path.split(path)[1])[1]
 270 
 271         if is_subdir(path, data_context().content.module_utils_path):
 272             if ext == '.py':
 273                 return self.get_python_module_utils_usage(path)
 274 
 275             if ext == '.psm1':
 276                 return self.get_powershell_module_utils_usage(path)
 277 
 278             if ext == '.cs':
 279                 return self.get_csharp_module_utils_usage(path)
 280 
 281         if is_subdir(path, data_context().content.integration_targets_path):
 282             return self.get_integration_target_usage(path)
 283 
 284         return []

 334     def get_integration_target_usage(self, path):
 335         """
 336         :type path: str

 338         """
 339         target_name = path.split('/')[3]
 340         dependents = [os.path.join(data_context().content.integration_targets_path, target) + os.path.sep
 341                       for target in sorted(self.integration_dependencies.get(target_name, set()))]
 342

 345     def classify(self, path):
 346         """
 347         :type path: str
 348         :rtype: dict[str, str] | None
 349         """
 350         result = self._classify(path)
 351 
 352         # run all tests when no result given
 353         if result is None:
 354             return None
 355 
 356         # run sanity on path unless result specified otherwise
 357         if path in self.sanity_paths and 'sanity' not in result:
 358             result['sanity'] = path
 359 
 360         return result

 362     def _classify(self, path):  # type: (str) -> t.Optional[t.Dict[str, str]]
 363         """Return the classification for the given path."""
 364         if data_context().content.is_ansible:
 365             return self._classify_ansible(path)
 366 
 367         if data_context().content.collection:
 368             return self._classify_collection(path)
 369 
 370         return None

 372     def _classify_common(self, path):  # type: (str) -> t.Optional[t.Dict[str, str]]
 373         """Return the classification for the given path using rules common to all layouts."""
 374         dirname = os.path.dirname(path)
 375         filename = os.path.basename(path)
 376         name, ext = os.path.splitext(filename)
 377 
 378         minimal = {}
 379 
 380         if is_subdir(path, '.github'):
 381             return minimal
 382 
 383         if is_subdir(path, data_context().content.integration_targets_path):
 384             if not os.path.exists(path):
 385                 return minimal
 386 
 387             target = self.integration_targets_by_name.get(path.split('/')[3])
 388 
 389             if not target:
 390                 display.warning('Unexpected non-target found: %s' % path)
 391                 return minimal
 392 
 393             if 'hidden/' in target.aliases:
 394                 return minimal  # already expanded using get_dependent_paths
 395 
 396             return {
 397                 'integration': target.name if 'posix/' in target.aliases else None,
 398                 'windows-integration': target.name if 'windows/' in target.aliases else None,
 399                 'network-integration': target.name if 'network/' in target.aliases else None,
 400                 FOCUSED_TARGET: True,
 401             }
 402 
 403         if is_subdir(path, data_context().content.integration_path):
 404             if dirname == data_context().content.integration_path:
 405                 for command in (
 406                         'integration',
 407                         'windows-integration',
 408                         'network-integration',
 409                 ):
 410                     if name == command and ext == '.cfg':
 411                         return {
 412                             command: self.integration_all_target,
 413                         }
 414 
 415                     if name == command + '.requirements' and ext == '.txt':
 416                         return {
 417                             command: self.integration_all_target,
 418                         }
 419 
 420             return {
 421                 'integration': self.integration_all_target,
 422                 'windows-integration': self.integration_all_target,
 423                 'network-integration': self.integration_all_target,
 424             }
 425 
 426         if is_subdir(path, data_context().content.sanity_path):
 427             return {
 428                 'sanity': 'all',  # test infrastructure, run all sanity checks
 429             }
 430 
 431         if is_subdir(path, data_context().content.unit_path):
 432             if path in self.units_paths:
 433                 return {
 434                     'units': path,
 435                 }
 436 
 437             # changes to files which are not unit tests should trigger tests from the nearest parent directory
 438 
 439             test_path = os.path.dirname(path)
 440 
 441             while test_path:
 442                 if test_path + '/' in self.units_paths:
 443                     return {
 444                         'units': test_path + '/',
 445                     }
 446 
 447                 test_path = os.path.dirname(test_path)
 448 
 449         if is_subdir(path, data_context().content.module_path):
 450             module_name = self.module_names_by_path.get(path)
 451 
 452             if module_name:
 453                 return {
 454                     'units': module_name if module_name in self.units_modules else None,
 455                     'integration': self.posix_integration_by_module.get(module_name) if ext == '.py' else None,
 456                     'windows-integration': self.windows_integration_by_module.get(module_name) if ext in ['.cs', '.ps1'] else None,
 457                     'network-integration': self.network_integration_by_module.get(module_name),
 458                     FOCUSED_TARGET: True,
 459                 }
 460 
 461             return minimal
 462 
 463         if is_subdir(path, data_context().content.module_utils_path):
 464             if ext == '.cs':
 465                 return minimal  # already expanded using get_dependent_paths
 466 
 467             if ext == '.psm1':
 468                 return minimal  # already expanded using get_dependent_paths
 469 
 470             if ext == '.py':
 471                 return minimal  # already expanded using get_dependent_paths
 472 
 473         if is_subdir(path, data_context().content.plugin_paths['action']):
 474             if ext == '.py':
 475                 if name.startswith('net_'):
 476                     network_target = 'network/.*_%s' % name[4:]
 477 
 478                     if any(re.search(r'^%s$' % network_target, alias) for alias in self.integration_targets_by_alias):
 479                         return {
 480                             'network-integration': network_target,
 481                             'units': 'all',
 482                         }
 483 
 484                     return {
 485                         'network-integration': self.integration_all_target,
 486                         'units': 'all',
 487                     }
 488 
 489                 if self.prefixes.get(name) == 'network':
 490                     network_platform = name
 491                 elif name.endswith('_config') and self.prefixes.get(name[:-7]) == 'network':
 492                     network_platform = name[:-7]
 493                 elif name.endswith('_template') and self.prefixes.get(name[:-9]) == 'network':
 494                     network_platform = name[:-9]
 495                 else:
 496                     network_platform = None
 497 
 498                 if network_platform:
 499                     network_target = 'network/%s/' % network_platform
 500 
 501                     if network_target in self.integration_targets_by_alias:
 502                         return {
 503                             'network-integration': network_target,
 504                             'units': 'all',
 505                         }
 506 
 507                     display.warning('Integration tests for "%s" not found.' % network_target, unique=True)
 508 
 509                     return {
 510                         'units': 'all',
 511                     }
 512 
 513         if is_subdir(path, data_context().content.plugin_paths['connection']):
 514             if name == '__init__':
 515                 return {
 516                     'integration': self.integration_all_target,
 517                     'windows-integration': self.integration_all_target,
 518                     'network-integration': self.integration_all_target,
 519                     'units': 'test/units/plugins/connection/',
 520                 }
 521 
 522             units_path = 'test/units/plugins/connection/test_%s.py' % name
 523 
 524             if units_path not in self.units_paths:
 525                 units_path = None
 526 
 527             integration_name = 'connection_%s' % name
 528 
 529             if integration_name not in self.integration_targets_by_name:
 530                 integration_name = None
 531 
 532             windows_integration_name = 'connection_windows_%s' % name
 533 
 534             if windows_integration_name not in self.integration_targets_by_name:
 535                 windows_integration_name = None
 536 
 537             # entire integration test commands depend on these connection plugins
 538 
 539             if name in ['winrm', 'psrp']:
 540                 return {
 541                     'windows-integration': self.integration_all_target,
 542                     'units': units_path,
 543                 }
 544 
 545             if name == 'local':
 546                 return {
 547                     'integration': self.integration_all_target,
 548                     'network-integration': self.integration_all_target,
 549                     'units': units_path,
 550                 }
 551 
 552             if name == 'network_cli':
 553                 return {
 554                     'network-integration': self.integration_all_target,
 555                     'units': units_path,
 556                 }
 557 
 558             if name == 'paramiko_ssh':
 559                 return {
 560                     'integration': integration_name,
 561                     'network-integration': self.integration_all_target,
 562                     'units': units_path,
 563                 }
 564 
 565             # other connection plugins have isolated integration and unit tests
 566 
 567             return {
 568                 'integration': integration_name,
 569                 'windows-integration': windows_integration_name,
 570                 'units': units_path,
 571             }
 572 
 573         if is_subdir(path, data_context().content.plugin_paths['doc_fragments']):
 574             return {
 575                 'sanity': 'all',
 576             }
 577 
 578         if is_subdir(path, data_context().content.plugin_paths['inventory']):
 579             if name == '__init__':
 580                 return all_tests(self.args)  # broad impact, run all tests
 581 
 582             # These inventory plugins are enabled by default (see INVENTORY_ENABLED).
 583             # Without dedicated integration tests for these we must rely on the incidental coverage from other tests.
 584             test_all = [
 585                 'host_list',
 586                 'script',
 587                 'yaml',
 588                 'ini',
 589                 'auto',
 590             ]
 591 
 592             if name in test_all:
 593                 posix_integration_fallback = get_integration_all_target(self.args)
 594             else:
 595                 posix_integration_fallback = None
 596 
 597             target = self.integration_targets_by_name.get('inventory_%s' % name)
 598             units_path = 'test/units/plugins/inventory/test_%s.py' % name
 599 
 600             if units_path not in self.units_paths:
 601                 units_path = None
 602 
 603             return {
 604                 'integration': target.name if target and 'posix/' in target.aliases else posix_integration_fallback,
 605                 'windows-integration': target.name if target and 'windows/' in target.aliases else None,
 606                 'network-integration': target.name if target and 'network/' in target.aliases else None,
 607                 'units': units_path,
 608                 FOCUSED_TARGET: target is not None,
 609             }
 610 
 611         if is_subdir(path, data_context().content.plugin_paths['filter']):
 612             return self._simple_plugin_tests('filter', name)
 613 
 614         if is_subdir(path, data_context().content.plugin_paths['lookup']):
 615             return self._simple_plugin_tests('lookup', name)
 616 
 617         if (is_subdir(path, data_context().content.plugin_paths['terminal']) or
 618                 is_subdir(path, data_context().content.plugin_paths['cliconf']) or
 619                 is_subdir(path, data_context().content.plugin_paths['netconf'])):
 620             if ext == '.py':
 621                 if name in self.prefixes and self.prefixes[name] == 'network':
 622                     network_target = 'network/%s/' % name
 623 
 624                     if network_target in self.integration_targets_by_alias:
 625                         return {
 626                             'network-integration': network_target,
 627                             'units': 'all',
 628                         }
 629 
 630                     display.warning('Integration tests for "%s" not found.' % network_target, unique=True)
 631 
 632                     return {
 633                         'units': 'all',
 634                     }
 635 
 636                 return {
 637                     'network-integration': self.integration_all_target,
 638                     'units': 'all',
 639                 }
 640 
 641         if is_subdir(path, data_context().content.plugin_paths['test']):
 642             return self._simple_plugin_tests('test', name)
 643 
 644         return None

 646     def _classify_collection(self, path):  # type: (str) -> t.Optional[t.Dict[str, str]]
 647         """Return the classification for the given path using rules specific to collections."""
 648         result = self._classify_common(path)
 649 
 650         if result is not None:
 651             return result
 652 
 653         minimal = {}
 654 
 655         if path.startswith('changelogs/'):
 656             return minimal
 657 
 658         if path.startswith('docs/'):
 659             return minimal
 660 
 661         return None

 663     def _classify_ansible(self, path):  # type: (str) -> t.Optional[t.Dict[str, str]]
 664         """Return the classification for the given path using rules specific to Ansible."""
 665         if path.startswith('test/units/compat/'):
 666             return {
 667                 'units': 'test/units/',
 668             }
 669 
 670         result = self._classify_common(path)
 671
