  40 def get_docker_container_id():
  41     """
  42     :rtype: str | None
  43     """
  44     path = '/proc/self/cgroup'
  45 
  46     if not os.path.exists(path):
  47         return None
  48 
  49     contents = read_text_file(path)
  50 
  51     paths = [line.split(':')[2] for line in contents.splitlines()]
  52     container_ids = set(path.split('/')[2] for path in paths if path.startswith('/docker/'))
  53 
  54     if not container_ids:
  55         return None
  56 
  57     if len(container_ids) == 1:
  58         return container_ids.pop()
  59 
  60     raise ApplicationError('Found multiple container_id candidates: %s\n%s' % (sorted(container_ids), contents))

  75     """

  88 def docker_pull(args, image):
  89     """
  90     :type args: EnvironmentConfig
  91     :type image: str
  92     """
  93     if ('@' in image or ':' in image) and docker_images(args, image):
  94         display.info('Skipping docker pull of existing image with tag or digest: %s' % image, verbosity=2)
  95         return
  96 
  97     if not args.docker_pull:
  98         display.warning('Skipping docker pull for "%s". Image may be out-of-date.' % image)
  99         return
 100 
 101     for _iteration in range(1, 10):
 102         try:
 103             docker_command(args, ['pull', image])
 104             return
 105         except SubprocessError:
 106             display.warning('Failed to pull docker image "%s". Waiting a few seconds before trying again.' % image)
 107             time.sleep(3)
 108 
 109     raise ApplicationError('Failed to pull docker image "%s".' % image)

 112 def docker_put(args, container_id, src, dst):
 113     """
 114     :type args: EnvironmentConfig
 115     :type container_id: str
 116     :type src: str
 117     :type dst: str
 118     """
 119     # avoid 'docker cp' due to a bug which causes 'docker rm' to fail
 120     with open_binary_file(src) as src_fd:
 121         docker_exec(args, container_id, ['dd', 'of=%s' % dst, 'bs=%s' % BUFFER_SIZE],
 122                     options=['-i'], stdin=src_fd, capture=True)

 125 def docker_get(args, container_id, src, dst):
 126     """
 127     :type args: EnvironmentConfig
 128     :type container_id: str
 129     :type src: str
 130     :type dst: str
 131     """
 132     # avoid 'docker cp' due to a bug which causes 'docker rm' to fail
 133     with open_binary_file(dst, 'wb') as dst_fd:
 134         docker_exec(args, container_id, ['dd', 'if=%s' % src, 'bs=%s' % BUFFER_SIZE],
 135                     options=['-i'], stdout=dst_fd, capture=True)

 138 def docker_run(args, image, options, cmd=None, create_only=False):
 139     """
 140     :type args: EnvironmentConfig
 141     :type image: str
 142     :type options: list[str] | None
 143     :type cmd: list[str] | None
 144     :type create_only[bool] | False
 145     :rtype: str | None, str | None
 146     """
 147     if not options:
 148         options = []
 149 
 150     if not cmd:
 151         cmd = []
 152 
 153     if create_only:
 154         command = 'create'
 155     else:
 156         command = 'run'
 157 
 158     for _iteration in range(1, 3):
 159         try:
 160             return docker_command(args, [command] + options + [image] + cmd, capture=True)
 161         except SubprocessError as ex:
 162             display.error(ex)
 163             display.warning('Failed to run docker image "%s". Waiting a few seconds before trying again.' % image)
 164             time.sleep(3)
 165 
 166     raise ApplicationError('Failed to run docker image "%s".' % image)

 187 def docker_images(args, image):
 188     """
 189     :param args: CommonConfig
 190     :param image: str
 191     :rtype: list[dict[str, any]]
 192     """
 193     try:
 194         stdout, _dummy = docker_command(args, ['images', image, '--format', '{{json .}}'], capture=True, always=True)
 195     except SubprocessError as ex:
 196         if 'no such image' in ex.stderr:
 197             stdout = ''  # podman does not handle this gracefully, exits 125
 198         else:
 199             raise ex
 200     results = [json.loads(line) for line in stdout.splitlines()]
 201     return results

 218 def docker_inspect(args, container_id):
 219     """
 220     :type args: EnvironmentConfig
 221     :type container_id: str
 222     :rtype: list[dict]
 223     """
 224     if args.explain:
 225         return []
 226 
 227     try:
 228         stdout = docker_command(args, ['inspect', container_id], capture=True)[0]
 229         return json.loads(stdout)
 230     except SubprocessError as ex:
 231         if 'no such image' in ex.stderr:
 232             return []  # podman does not handle this gracefully, exits 125
 233         try:
 234             return json.loads(ex.stdout)
 235         except Exception:
 236             raise ex

 302 def docker_command(args, cmd, capture=False, stdin=None, stdout=None, always=False, data=None):
 303     """
 304     :type args: CommonConfig
 305     :type cmd: list[str]
 306     :type capture: bool
 307     :type stdin: file | None
 308     :type stdout: file | None
 309     :type always: bool
 310     :type data: str | None
 311     :rtype: str | None, str | None
 312     """
 313     env = docker_environment()
 314     return run_command(args, ['docker'] + cmd, env=env, capture=capture, stdin=stdin, stdout=stdout, always=always, data=data)

 317 def docker_environment():
 318     """
 319     :rtype: dict[str, str]
 320     """
 321     env = common_environment()
 322     env.update(dict((key, os.environ[key]) for key in os.environ if key.startswith('DOCKER_')))
 323     return env
