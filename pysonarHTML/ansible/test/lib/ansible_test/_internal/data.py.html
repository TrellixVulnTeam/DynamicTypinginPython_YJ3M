  67     def create_collection_layouts(self):  # type: () -> t.List[ContentLayout]
  68         """
  69         Return a list of collection layouts, one for each collection in the same collection root as the current collection layout.
  70         An empty list is returned if the current content layout is not a collection layout.
  71         """
  72         layout = self.content
  73         collection = layout.collection
  74 
  75         if not collection:
  76             return []
  77 
  78         root_path = os.path.join(collection.root, 'ansible_collections')
  79         display.info('Scanning collection root: %s' % root_path, verbosity=1)
  80         namespace_names = sorted(name for name in os.listdir(root_path) if os.path.isdir(os.path.join(root_path, name)))
  81         collections = []
  82 
  83         for namespace_name in namespace_names:
  84             namespace_path = os.path.join(root_path, namespace_name)
  85             collection_names = sorted(name for name in os.listdir(namespace_path) if os.path.isdir(os.path.join(namespace_path, name)))
  86 
  87             for collection_name in collection_names:
  88                 collection_path = os.path.join(namespace_path, collection_name)
  89 
  90                 if collection_path == os.path.join(collection.root, collection.directory):
  91                     collection_layout = layout
  92                 else:
  93                     collection_layout = self.__create_content_layout(self.__layout_providers, self.__source_providers, collection_path, False)
  94 
  95                 file_count = len(collection_layout.all_files())
  96 
  97                 if not file_count:
  98                     continue
  99 
 100                 display.info('Including collection: %s (%d files)' % (collection_layout.collection.full_name, file_count), verbosity=1)
 101                 collections.append(collection_layout)
 102 
 103         return collections

 105     @staticmethod
 106     def __create_content_layout(layout_providers,  # type: t.List[t.Type[LayoutProvider]]
 107                                 source_providers,  # type: t.List[t.Type[SourceProvider]]
 108                                 root,  # type: str
 109                                 walk,  # type: bool
 110                                 ):  # type: (...) -> ContentLayout
 111         """Create a content layout using the given providers and root path."""
 112         layout_provider = find_path_provider(LayoutProvider, layout_providers, root, walk)
 113 
 114         try:
 115             # Begin the search for the source provider at the layout provider root.
 116             # This intentionally ignores version control within subdirectories of the layout root, a condition which was previously an error.
 117             # Doing so allows support for older git versions for which it is difficult to distinguish between a super project and a sub project.
 118             # It also provides a better user experience, since the solution for the user would effectively be the same -- to remove the nested version control.
 119             source_provider = find_path_provider(SourceProvider, source_providers, layout_provider.root, walk)
 120         except ProviderNotFoundForPath:
 121             source_provider = UnversionedSource(layout_provider.root)
 122 
 123         layout = layout_provider.create(layout_provider.root, source_provider.get_paths(layout_provider.root))
 124 
 125         return layout

 127     def __create_ansible_source(self):
 128         """Return a tuple of Ansible source files with both absolute and relative paths."""
 129         if not ANSIBLE_SOURCE_ROOT:
 130             sources = []
 131 
 132             source_provider = InstalledSource(ANSIBLE_LIB_ROOT)
 133             sources.extend((os.path.join(source_provider.root, path), os.path.join('lib', 'ansible', path))
 134                            for path in source_provider.get_paths(source_provider.root))
 135 
 136             source_provider = InstalledSource(ANSIBLE_TEST_ROOT)
 137             sources.extend((os.path.join(source_provider.root, path), os.path.join('test', 'lib', 'ansible_test', path))
 138                            for path in source_provider.get_paths(source_provider.root))
 139 
 140             return tuple(sources)
 141 
 142         if self.content.is_ansible:
 143             return tuple((os.path.join(self.content.root, path), path) for path in self.content.all_files())
 144 
 145         try:
 146             source_provider = find_path_provider(SourceProvider, self.__source_providers, ANSIBLE_SOURCE_ROOT, False)
 147         except ProviderNotFoundForPath:
 148             source_provider = UnversionedSource(ANSIBLE_SOURCE_ROOT)
 149 
 150         return tuple((os.path.join(source_provider.root, path), path) for path in source_provider.get_paths(source_provider.root))
 151 
 152     @property
 153     def ansible_source(self):  # type: () -> t.Tuple[t.Tuple[str, str], ...]
 154         """Return a tuple of Ansible source files with both absolute and relative paths."""
 155         if not self.__ansible_source:
 156             self.__ansible_source = self.__create_ansible_source()
 157 
 158         return self.__ansible_source

 165 def data_init():  # type: () -> DataContext
 166     """Initialize provider plugins."""
 167     provider_types = (
 168         'layout',
 169         'source',
 170     )
 171 
 172     for provider_type in provider_types:
 173         import_plugins('provider/%s' % provider_type)
 174 
 175     try:
 176         context = DataContext()
 177     except ProviderNotFoundForPath:
 178         options = [
 179             ' - an Ansible collection: {...}/ansible_collections/{namespace}/{collection}/',
 180         ]
 181 
 182         if ANSIBLE_SOURCE_ROOT:
 183             options.insert(0, ' - the Ansible source: %s/' % ANSIBLE_SOURCE_ROOT)
 184 
 185         raise ApplicationError('''The current working directory must be at or below:
 186 
 187 %s
 188 
 189 Current working directory: %s''' % ('\n'.join(options), os.getcwd()))
 190 
 191     return context
