  26 def calculate_best_confidence(choices, metadata):
  27     """
  28     :type choices: tuple[tuple[str, int]]
  29     :type metadata: Metadata
  30     :rtype: int
  31     """
  32     best_confidence = 0
  33 
  34     for path, line in choices:
  35         confidence = calculate_confidence(path, line, metadata)
  36         best_confidence = max(confidence, best_confidence)
  37 
  38     return best_confidence

  41 def calculate_confidence(path, line, metadata):
  42     """
  43     :type path: str
  44     :type line: int
  45     :type metadata: Metadata
  46     :rtype: int
  47     """
  48     ranges = metadata.changes.get(path)
  49 
  50     # no changes were made to the file
  51     if not ranges:
  52         return 0
  53 
  54     # changes were made to the same file and line
  55     if any(r[0] <= line <= r[1] in r for r in ranges):
  56         return 100
  57 
  58     # changes were made to the same file and the line number is unknown
  59     if line == 0:
  60         return 75
  61 
  62     # changes were made to the same file and the line number is different
  63     return 50

  66 class TestResult:

  68     def __init__(self, command, test, python_version=None):
  69         """
  70         :type command: str

 121     def create_result_name(self, extension):

 138     def save_junit(self, args, test_case, properties=None):
 139         """
 140         :type args: TestConfig
 141         :type test_case: junit_xml.TestCase
 142         :type properties: dict[str, str] | None
 143         :rtype: str | None
 144         """
 145         test_suites = [
 146             self.junit.TestSuite(
 147                 name='ansible-test',
 148                 test_cases=[test_case],
 149                 timestamp=datetime.datetime.utcnow().replace(microsecond=0).isoformat(),
 150                 properties=properties,
 151             ),
 152         ]
 153 
 154         # the junit_xml API is changing in version 2.0.0
 155         # TestSuite.to_xml_string is being replaced with to_xml_report_string
 156         # see: https://github.com/kyrus/python-junit-xml/blob/63db26da353790500642fd02cae1543eb41aab8b/junit_xml/__init__.py#L249-L261
 157         try:
 158             to_xml_string = self.junit.to_xml_report_string
 159         except AttributeError:
 160             to_xml_string = self.junit.TestSuite.to_xml_string
 161 
 162         report = to_xml_string(test_suites=test_suites, prettyprint=True, encoding='utf-8')
 163 
 164         if args.explain:
 165             return
 166 
 167         write_text_test_results(ResultType.JUNIT, self.create_result_name('.xml'), report)

 170 class TestTimeout(TestResult):
 171     """Test timeout."""
 172     def __init__(self, timeout_duration):
 173         """
 174         :type timeout_duration: int
 175         """
 176         super(TestTimeout, self).__init__(command='timeout', test='')
 177 
 178         self.timeout_duration = timeout_duration
 179 
 180     def write(self, args):
 181         """
 182         :type args: TestConfig
 183         """
 184         message = 'Tests were aborted after exceeding the %d minute time limit.' % self.timeout_duration
 185 
 186         # Include a leading newline to improve readability on Shippable "Tests" tab.
 187         # Without this, the first line becomes indented.
 188         output = '''
 189 One or more of the following situations may be responsible:
 190 
 191 - Code changes have resulted in tests that hang or run for an excessive amount of time.
 192 - Tests have been added which exceed the time limit when combined with existing tests.
 193 - Test infrastructure and/or external dependencies are operating slower than normal.'''
 194 
 195         if args.coverage:
 196             output += '\n- Additional overhead from collecting code coverage has resulted in tests exceeding the time limit.'
 197 
 198         output += '\n\nConsult the console log for additional details on where the timeout occurred.'
 199 
 200         timestamp = datetime.datetime.utcnow().replace(microsecond=0).isoformat()
 201 
 202         # hack to avoid requiring junit-xml, which may not be pre-installed outside our test containers
 203         xml = '''
 204 <?xml version="1.0" encoding="utf-8"?>
 205 <testsuites disabled="0" errors="1" failures="0" tests="1" time="0.0">
 206 \t<testsuite disabled="0" errors="1" failures="0" file="None" log="None" name="ansible-test" skipped="0" tests="1" time="0" timestamp="%s" url="None">
 207 \t\t<testcase classname="timeout" name="timeout">
 208 \t\t\t<error message="%s" type="error">%s</error>
 209 \t\t</testcase>
 210 \t</testsuite>
 211 </testsuites>
 212 ''' % (timestamp, message, output)
 213 
 214         write_text_test_results(ResultType.JUNIT, self.create_result_name('.xml'), xml.lstrip())

 244 class TestFailure(TestResult):
 245     """Test failure."""

 249         :type test: str

 273     def write_console(self):
 274         """Write results to console."""
 275         if self.summary:
 276             display.error(self.summary)
 277         else:
 278             if self.python_version:
 279                 specifier = ' on python %s' % self.python_version
 280             else:
 281                 specifier = ''
 282 
 283             display.error('Found %d %s issue(s)%s which need to be resolved:' % (len(self.messages), self.test or self.command, specifier))
 284 
 285             for message in self.messages:
 286                 display.error(message.format(show_confidence=True))
 287 
 288             doc_url = self.find_docs()
 289             if doc_url:
 290                 display.info('See documentation for help: %s' % doc_url)
 291 
 292     def write_lint(self):
 293         """Write lint results to stdout."""
 294         if self.summary:
 295             command = self.format_command()
 296             message = 'The test `%s` failed. See stderr output for details.' % command
 297             path = ''
 298             message = TestMessage(message, path)
 299             print(message)
 300         else:
 301             for message in self.messages:
 302                 print(message)
 303 
 304     def write_junit(self, args):
 305         """
 306         :type args: TestConfig
 307         """
 308         title = self.format_title()
 309         output = self.format_block()
 310 
 311         test_case = self.junit.TestCase(classname=self.command, name=self.name)
 312 
 313         # Include a leading newline to improve readability on Shippable "Tests" tab.
 314         # Without this, the first line becomes indented.
 315         test_case.add_failure_info(message=title, output='\n%s' % output)
 316 
 317         self.save_junit(args, test_case)
 318 
 319     def write_bot(self, args):
 320         """
 321         :type args: TestConfig
 322         """
 323         docs = self.find_docs()
 324         message = self.format_title(help_link=docs)
 325         output = self.format_block()
 326 
 327         if self.messages:
 328             verified = all((m.confidence or 0) >= 50 for m in self.messages)
 329         else:
 330             verified = False
 331 
 332         bot_data = dict(
 333             verified=verified,
 334             docs=docs,
 335             results=[
 336                 dict(
 337                     message=message,
 338                     output=output,
 339                 ),
 340             ],
 341         )
 342 
 343         if args.explain:
 344             return
 345 
 346         write_json_test_results(ResultType.BOT, self.create_result_name('.json'), bot_data)

 348     def populate_confidence(self, metadata):
 349         """
 350         :type metadata: Metadata
 351         """
 352         for message in self.messages:
 353             if message.confidence is None:
 354                 message.confidence = calculate_confidence(message.path, message.line, metadata)

 356     def format_command(self):
 357         """
 358         :rtype: str
 359         """
 360         command = 'ansible-test %s' % self.command
 361 
 362         if self.test:
 363             command += ' --test %s' % self.test
 364 
 365         if self.python_version:
 366             command += ' --python %s' % self.python_version
 367 
 368         return command
 369 
 370     def find_docs(self):
 371         """
 372         :rtype: str
 373         """
 374         if self.command != 'sanity':
 375             return None  # only sanity tests have docs links
 376 
 377         # Use the major.minor version for the URL only if this a release that
 378         # matches the pattern 2.4.0, otherwise, use 'devel'
 379         ansible_version = get_ansible_version()
 380         url_version = 'devel'
 381         if re.search(r'^[0-9.]+$', ansible_version):
 382             url_version = '.'.join(ansible_version.split('.')[:2])
 383 
 384         testing_docs_url = 'https://docs.ansible.com/ansible/%s/dev_guide/testing' % url_version
 385 
 386         url = '%s/%s/' % (testing_docs_url, self.command)
 387 
 388         if self.test:
 389             url += '%s.html' % self.test
 390 
 391         return url
 392 
 393     def format_title(self, help_link=None):
 394         """
 395         :type help_link: str | None
 396         :rtype: str
 397         """
 398         command = self.format_command()
 399 
 400         if self.summary:
 401             reason = 'the error'
 402         else:
 403             reason = '1 error' if len(self.messages) == 1 else '%d errors' % len(self.messages)
 404 
 405         if help_link:
 406             help_link_markup = ' [[explain](%s)]' % help_link
 407         else:
 408             help_link_markup = ''
 409 
 410         title = 'The test `%s`%s failed with %s:' % (command, help_link_markup, reason)
 411 
 412         return title
 413 
 414     def format_block(self):
 415         """
 416         :rtype: str
 417         """
 418         if self.summary:
 419             block = self.summary
 420         else:
 421             block = '\n'.join(m.format() for m in self.messages)
 422 
 423         message = block.strip()
 424 
 425         # Hack to remove ANSI color reset code from SubprocessError messages.
 426         message = message.replace(display.clear, '')
 427 
 428         return message

 431 class TestMessage:
 432     """Single test message for one file."""
 433     def __init__(self, message, path, line=0, column=0, level='error', code=None, confidence=None):
 434         """
 435         :type message: str
 436         :type path: str
 437         :type line: int
 438         :type column: int
 439         :type level: str
 440         :type code: str | None
 441         :type confidence: int | None
 442         """
 443         self.__path = path
 444         self.__line = line
 445         self.__column = column
 446         self.__level = level
 447         self.__code = code
 448         self.__message = message
 449 
 450         self.confidence = confidence
 451 
 452     @property
 453     def path(self):  # type: () -> str
 454         """Return the path."""
 455         return self.__path
 456 
 457     @property
 458     def line(self):  # type: () -> int
 459         """Return the line number, or 0 if none is available."""
 460         return self.__line
 461 
 462     @property
 463     def column(self):  # type: () -> int
 464         """Return the column number, or 0 if none is available."""
 465         return self.__column
 466 
 467     @property
 468     def level(self):  # type: () -> str
 469         """Return the level."""
 470         return self.__level
 471 
 472     @property
 473     def code(self):  # type: () -> t.Optional[str]
 474         """Return the code, if any."""
 475         return self.__code
 476 
 477     @property
 478     def message(self):  # type: () -> str
 479         """Return the message."""
 480         return self.__message
 481 
 482     @property
 483     def tuple(self):  # type: () -> t.Tuple[str, int, int, str, t.Optional[str], str]
 484         """Return a tuple with all the immutable values of this test message."""
 485         return self.__path, self.__line, self.__column, self.__level, self.__code, self.__message

 508     def __str__(self):

 511     def format(self, show_confidence=False):
 512         """
 513         :type show_confidence: bool
 514         :rtype: str
 515         """
 516         if self.__code:
 517             msg = '%s: %s' % (self.__code, self.__message)
 518         else:
 519             msg = self.__message
 520 
 521         if show_confidence and self.confidence is not None:
 522             msg += ' (%d%%)' % self.confidence
 523 
 524         return '%s:%s:%s: %s' % (self.__path, self.__line, self.__column, msg)
