  10 def main():
  11     """Main entry point."""
  12     name = os.path.basename(__file__)
  13     args = [sys.executable]
  14 
  15     coverage_config = os.environ.get('COVERAGE_CONF')
  16     coverage_output = os.environ.get('COVERAGE_FILE')
  17 
  18     if coverage_config:
  19         if coverage_output:
  20             args += ['-m', 'coverage.__main__', 'run', '--rcfile', coverage_config]
  21         else:
  22             if sys.version_info >= (3, 4):
  23                 # noinspection PyUnresolvedReferences
  24                 import importlib.util
  25 
  26                 # noinspection PyUnresolvedReferences
  27                 found = bool(importlib.util.find_spec('coverage'))
  28             else:
  29                 # noinspection PyDeprecation
  30                 import imp
  31 
  32                 try:
  33                     # noinspection PyDeprecation
  34                     imp.find_module('coverage')
  35                     found = True
  36                 except ImportError:
  37                     found = False
  38 
  39             if not found:
  40                 sys.exit('ERROR: Could not find `coverage` module. '
  41                          'Did you use a virtualenv created without --system-site-packages or with the wrong interpreter?')
  42 
  43     if name == 'python.py':
  44         if sys.argv[1] == '-c':
  45             # prevent simple misuse of python.py with -c which does not work with coverage
  46             sys.exit('ERROR: Use `python -c` instead of `python.py -c` to avoid errors when code coverage is collected.')
  47     elif name == 'pytest':
  48         args += ['-m', 'pytest']
  49     else:
  50         args += [find_executable(name)]
  51 
  52     args += sys.argv[1:]
  53 
  54     os.execv(args[0], args)

  57 def find_executable(name):
  58     """
  59     :type name: str
  60     :rtype: str
  61     """
  62     path = os.environ.get('PATH', os.path.defpath)
  63     seen = set([os.path.abspath(__file__)])
  64 
  65     for base in path.split(os.path.pathsep):
  66         candidate = os.path.abspath(os.path.join(base, name))
  67 
  68         if candidate in seen:
  69             continue
  70 
  71         seen.add(candidate)
  72 
  73         if os.path.exists(candidate) and os.access(candidate, os.F_OK | os.X_OK):
  74             return candidate
  75 
  76     raise Exception('Executable "%s" not found in path: %s' % (name, path))
  77

  79 if __name__ == '__main__':
  80     main()
