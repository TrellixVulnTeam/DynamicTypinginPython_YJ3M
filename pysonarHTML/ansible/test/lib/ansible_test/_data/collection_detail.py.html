  13 # See semantic versioning specification (https://semver.org/)
  14 NUMERIC_IDENTIFIER = r'(?:0|[1-9][0-9]*)'
  15 ALPHANUMERIC_IDENTIFIER = r'(?:[0-9]*[a-zA-Z-][a-zA-Z0-9-]*)'
  16 
  17 PRE_RELEASE_IDENTIFIER = r'(?:' + NUMERIC_IDENTIFIER + r'|' + ALPHANUMERIC_IDENTIFIER + r')'
  18 BUILD_IDENTIFIER = r'[a-zA-Z0-9-]+'  # equivalent to r'(?:[0-9]+|' + ALPHANUMERIC_IDENTIFIER + r')'
  19 
  20 VERSION_CORE = NUMERIC_IDENTIFIER + r'\.' + NUMERIC_IDENTIFIER + r'\.' + NUMERIC_IDENTIFIER
  21 PRE_RELEASE = r'(?:-' + PRE_RELEASE_IDENTIFIER + r'(?:\.' + PRE_RELEASE_IDENTIFIER + r')*)?'
  22 BUILD = r'(?:\+' + BUILD_IDENTIFIER + r'(?:\.' + BUILD_IDENTIFIER + r')*)?'
  23 
  24 SEMVER_REGULAR_EXPRESSION = r'^' + VERSION_CORE + PRE_RELEASE + BUILD + r'$'
  25

  27 def validate_version(version):
  28     """Raise exception if the provided version is not None or a valid semantic version."""
  29     if version is None:
  30         return
  31     if not re.match(SEMVER_REGULAR_EXPRESSION, version):
  32         raise Exception('Invalid version number "{0}". Collection version numbers must '
  33                         'follow semantic versioning (https://semver.org/).'.format(version))

  36 def read_manifest_json(collection_path):
  37     """Return collection information from the MANIFEST.json file."""
  38     manifest_path = os.path.join(collection_path, 'MANIFEST.json')
  39 
  40     if not os.path.exists(manifest_path):
  41         return None
  42 
  43     try:
  44         with open(manifest_path) as manifest_file:
  45             manifest = json.load(manifest_file)
  46 
  47         collection_info = manifest.get('collection_info') or dict()
  48 
  49         result = dict(
  50             version=collection_info.get('version'),
  51         )
  52         validate_version(result['version'])
  53     except Exception as ex:  # pylint: disable=broad-except
  54         raise Exception('{0}: {1}'.format(os.path.basename(manifest_path), ex))
  55 
  56     return result
  57

  59 def read_galaxy_yml(collection_path):
  60     """Return collection information from the galaxy.yml file."""
  61     galaxy_path = os.path.join(collection_path, 'galaxy.yml')
  62 
  63     if not os.path.exists(galaxy_path):
  64         return None
  65 
  66     try:
  67         with open(galaxy_path) as galaxy_file:
  68             galaxy = yaml.safe_load(galaxy_file)
  69 
  70         result = dict(
  71             version=galaxy.get('version'),
  72         )
  73         validate_version(result['version'])
  74     except Exception as ex:  # pylint: disable=broad-except
  75         raise Exception('{0}: {1}'.format(os.path.basename(galaxy_path), ex))
  76 
  77     return result

  80 def main():
  81     """Retrieve collection detail."""

  84     try:
