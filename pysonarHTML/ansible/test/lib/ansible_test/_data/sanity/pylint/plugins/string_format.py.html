   7 import sys
   8 
   9 import six
  10 
  11 import astroid
  12 from pylint.interfaces import IAstroidChecker
  13 from pylint.checkers import BaseChecker
  14 from pylint.checkers import utils
  15 from pylint.checkers.utils import check_messages
  16 try:
  17     from pylint.checkers.utils import parse_format_method_string
  18 except ImportError:
  19     # noinspection PyUnresolvedReferences
  20     from pylint.checkers.strings import parse_format_method_string
  21 
  22 _PY3K = sys.version_info[:2] >= (3, 0)
  23 
  24 MSGS = {
  25     'E9305': ("Format string contains automatic field numbering "
  26               "specification",
  27               "ansible-format-automatic-specification",
  28               "Used when a PEP 3101 format string contains automatic "
  29               "field numbering (e.g. '{}').",
  30               {'minversion': (2, 6)}),
  31     'E9390': ("bytes object has no .format attribute",
  32               "ansible-no-format-on-bytestring",
  33               "Used when a bytestring was used as a PEP 3101 format string "
  34               "as Python3 bytestrings do not have a .format attribute",
  35               {'minversion': (3, 0)}),
  36 }
  37 
  38 
  39 class AnsibleStringFormatChecker(BaseChecker):
  40     """Checks string formatting operations to ensure that the format string
  41     is valid and the arguments match the format string.
  42     """
  43 
  44     __implements__ = (IAstroidChecker,)
  45     name = 'string'
  46     msgs = MSGS
  47 
  48     @check_messages(*(MSGS.keys()))
  49     def visit_call(self, node):
  50         func = utils.safe_infer(node.func)
  51         if (isinstance(func, astroid.BoundMethod)
  52                 and isinstance(func.bound, astroid.Instance)
  53                 and func.bound.name in ('str', 'unicode', 'bytes')):
  54             if func.name == 'format':
  55                 self._check_new_format(node, func)
  56 
  57     def _check_new_format(self, node, func):
  58         """ Check the new string formatting """
  59         if (isinstance(node.func, astroid.Attribute)
  60                 and not isinstance(node.func.expr, astroid.Const)):
  61             return
  62         try:
  63             strnode = next(func.bound.infer())
  64         except astroid.InferenceError:
  65             return
  66         if not isinstance(strnode, astroid.Const):
  67             return
  68 
  69         if _PY3K and isinstance(strnode.value, six.binary_type):
  70             self.add_message('ansible-no-format-on-bytestring', node=node)
  71             return
  72         if not isinstance(strnode.value, six.string_types):
  73             return
  74 
  75         if node.starargs or node.kwargs:
  76             return
  77         try:
  78             num_args = parse_format_method_string(strnode.value)[1]
  79         except utils.IncompleteFormatString:
  80             return
  81 
  82         if num_args:
  83             self.add_message('ansible-format-automatic-specification',
  84                              node=node)
  85             return
  86 
  87 
  88 def register(linter):
  89     """required method to auto register this checker """
  90     linter.register_checker(AnsibleStringFormatChecker(linter))
