  17 class UnwantedEntry:
  18     """Defines an unwanted import."""
  19     def __init__(self, alternative, modules_only=False, names=None, ignore_paths=None):
  20         """
  21         :type alternative: str
  22         :type modules_only: bool
  23         :type names: tuple[str] | None
  24         :type ignore_paths: tuple[str] | None
  25         """
  26         self.alternative = alternative
  27         self.modules_only = modules_only
  28         self.names = set(names) if names else set()
  29         self.ignore_paths = ignore_paths
  30 
  31     def applies_to(self, path, name=None):
  32         """
  33         :type path: str
  34         :type name: str | None
  35         :rtype: bool
  36         """
  37         if self.names:
  38             if not name:
  39                 return False
  40 
  41             if name not in self.names:
  42                 return False
  43 
  44         if self.ignore_paths and any(path.endswith(ignore_path) for ignore_path in self.ignore_paths):
  45             return False
  46 
  47         if self.modules_only:
  48             return is_module_path(path)
  49 
  50         return True
  51 
  52 
  53 def is_module_path(path):
  54     """
  55     :type path: str
  56     :rtype: bool
  57     """
  58     return path.startswith(ANSIBLE_TEST_MODULES_PATH) or path.startswith(ANSIBLE_TEST_MODULE_UTILS_PATH)

 141     def visit_import(self, node):
 142         """
 143         :type node: astroid.node_classes.Import
 144         """
 145         for name in node.names:
 146             self._check_import(node, name[0])
 147 
 148     def visit_importfrom(self, node):
 149         """
 150         :type node: astroid.node_classes.ImportFrom
 151         """
 152         self._check_importfrom(node, node.modname, node.names)
 153 
 154     def visit_attribute(self, node):
 155         """
 156         :type node: astroid.node_classes.Attribute
 157         """
 158         last_child = node.last_child()
 159 
 160         # this is faster than using type inference and will catch the most common cases
 161         if not isinstance(last_child, astroid.node_classes.Name):
 162             return
 163 
 164         module = last_child.name
 165 
 166         entry = self.unwanted_imports.get(module)
 167 
 168         if entry and entry.names:
 169             if entry.applies_to(self.linter.current_file, node.attrname):
 170                 self.add_message(self.BAD_IMPORT_FROM, args=(node.attrname, entry.alternative, module), node=node)
 171 
 172     def visit_call(self, node):
 173         """
 174         :type node: astroid.node_classes.Call
 175         """
 176         try:
 177             for i in node.func.inferred():
 178                 func = None
 179 
 180                 if isinstance(i, astroid.scoped_nodes.FunctionDef) and isinstance(i.parent, astroid.scoped_nodes.Module):
 181                     func = '%s.%s' % (i.parent.name, i.name)
 182 
 183                 if not func:
 184                     continue
 185 
 186                 entry = self.unwanted_functions.get(func)
 187 
 188                 if entry and entry.applies_to(self.linter.current_file):
 189                     self.add_message(self.BAD_FUNCTION, args=(entry.alternative, func), node=node)
 190         except astroid.exceptions.InferenceError:
 191             pass
 192 
 193     def _check_import(self, node, modname):
 194         """
 195         :type node: astroid.node_classes.Import
 196         :type modname: str
 197         """
 198         self._check_module_import(node, modname)
 199 
 200         entry = self.unwanted_imports.get(modname)
 201 
 202         if not entry:
 203             return
 204 
 205         if entry.applies_to(self.linter.current_file):
 206             self.add_message(self.BAD_IMPORT, args=(entry.alternative, modname), node=node)
 207 
 208     def _check_importfrom(self, node, modname, names):
 209         """
 210         :type node: astroid.node_classes.ImportFrom
 211         :type modname: str
 212         :type names:  list[str[
 213         """
 214         self._check_module_import(node, modname)
 215 
 216         entry = self.unwanted_imports.get(modname)
 217 
 218         if not entry:
 219             return
 220 
 221         for name in names:
 222             if entry.applies_to(self.linter.current_file, name[0]):
 223                 self.add_message(self.BAD_IMPORT_FROM, args=(name[0], entry.alternative, modname), node=node)
 224 
 225     def _check_module_import(self, node, modname):
 226         """
 227         :type node: astroid.node_classes.Import | astroid.node_classes.ImportFrom
 228         :type modname: str
 229         """
 230         if not is_module_path(self.linter.current_file):
 231             return
 232 
 233         if modname == 'ansible.module_utils' or modname.startswith('ansible.module_utils.'):
 234             return
 235 
 236         if modname == 'ansible' or modname.startswith('ansible.'):
 237             self.add_message(self.BAD_MODULE_IMPORT, args=(modname,), node=node)
