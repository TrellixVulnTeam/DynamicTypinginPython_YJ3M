  90 def _get_expr_name(node):
  91     """Funciton to get either ``attrname`` or ``name`` from ``node.func.expr``
  92 
  93     Created specifically for the case of ``display.deprecated`` or ``self._display.deprecated``
  94     """
  95     try:
  96         return node.func.expr.attrname
  97     except AttributeError:
  98         # If this fails too, we'll let it raise, the caller should catch it
  99         return node.func.expr.name

 102 def parse_isodate(value):
 103     msg = 'Expected ISO 8601 date string (YYYY-MM-DD)'
 104     if not isinstance(value, string_types):
 105         raise ValueError(msg)
 106     # From Python 3.7 in, there is datetime.date.fromisoformat(). For older versions,
 107     # we have to do things manually.
 108     if not re.match('^[0-9]{4}-[0-9]{2}-[0-9]{2}$', value):
 109         raise ValueError(msg)
 110     try:
 111         return datetime.datetime.strptime(value, '%Y-%m-%d').date()
 112     except ValueError:
 113         raise ValueError(msg)

 116 class AnsibleDeprecatedChecker(BaseChecker):
 117     """Checks for Display.deprecated calls to ensure that the ``version``
 118     has not passed or met the time for removal
 119     """
 120 
 121     __implements__ = (IAstroidChecker,)
 122     name = 'deprecated'
 123     msgs = MSGS
 124 
 125     options = (
 126         ('collection-name', {
 127             'default': None,
 128             'type': 'string',
 129             'metavar': '<name>',
 130             'help': 'The collection\'s name used to check collection names in deprecations.',
 131         }),
 132         ('collection-version', {
 133             'default': None,
 134             'type': 'string',
 135             'metavar': '<version>',
 136             'help': 'The collection\'s version number used to check deprecations.',
 137         }),
 138     )
 139 
 140     def __init__(self, *args, **kwargs):
 141         self.collection_version = None
 142         self.collection_name = None
 143         super(AnsibleDeprecatedChecker, self).__init__(*args, **kwargs)
 144 
 145     def set_option(self, optname, value, action=None, optdict=None):
 146         super(AnsibleDeprecatedChecker, self).set_option(optname, value, action, optdict)
 147         if optname == 'collection-version' and value is not None:
 148             self.collection_version = SemanticVersion(self.config.collection_version)
 149         if optname == 'collection-name' and value is not None:
 150             self.collection_name = self.config.collection_name
 151 
 152     def _check_date(self, node, date):
 153         if not isinstance(date, str):
 154             self.add_message('invalid-date', node=node, args=(date,))
 155             return
 156 
 157         try:
 158             date_parsed = parse_isodate(date)
 159         except ValueError:
 160             self.add_message('ansible-invalid-deprecated-date', node=node, args=(date,))
 161             return
 162 
 163         if date_parsed < datetime.date.today():
 164             self.add_message('ansible-deprecated-date', node=node, args=(date,))
 165 
 166     def _check_version(self, node, version, collection_name):
 167         if not isinstance(version, (str, float)):
 168             self.add_message('invalid-version', node=node, args=(version,))
 169             return
 170 
 171         version_no = str(version)
 172 
 173         if collection_name == 'ansible.builtin':
 174             # Ansible-base
 175             try:
 176                 if not version_no:
 177                     raise ValueError('Version string should not be empty')
 178                 loose_version = LooseVersion(str(version_no))
 179                 if ANSIBLE_VERSION >= loose_version:
 180                     self.add_message('ansible-deprecated-version', node=node, args=(version,))
 181             except ValueError:
 182                 self.add_message('ansible-invalid-deprecated-version', node=node, args=(version,))
 183         elif collection_name:
 184             # Collections
 185             try:
 186                 if not version_no:
 187                     raise ValueError('Version string should not be empty')
 188                 semantic_version = SemanticVersion(version_no)
 189                 if collection_name == self.collection_name and self.collection_version is not None:
 190                     if self.collection_version >= semantic_version:
 191                         self.add_message('collection-deprecated-version', node=node, args=(version,))
 192             except ValueError:
 193                 self.add_message('collection-invalid-deprecated-version', node=node, args=(version,))
 194 
 195     @check_messages(*(MSGS.keys()))
 196     def visit_call(self, node):
 197         version = None
 198         date = None
 199         collection_name = None
 200         try:
 201             if (node.func.attrname == 'deprecated' and 'display' in _get_expr_name(node) or
 202                     node.func.attrname == 'deprecate' and _get_expr_name(node)):
 203                 if node.keywords:
 204                     for keyword in node.keywords:
 205                         if len(node.keywords) == 1 and keyword.arg is None:
 206                             # This is likely a **kwargs splat
 207                             return
 208                         if keyword.arg == 'version':
 209                             if isinstance(keyword.value.value, astroid.Name):
 210                                 # This is likely a variable
 211                                 return
 212                             version = keyword.value.value
 213                         if keyword.arg == 'date':
 214                             if isinstance(keyword.value.value, astroid.Name):
 215                                 # This is likely a variable
 216                                 return
 217                             date = keyword.value.value
 218                         if keyword.arg == 'collection_name':
 219                             if isinstance(keyword.value.value, astroid.Name):
 220                                 # This is likely a variable
 221                                 return
 222                             collection_name = keyword.value.value
 223                 if not version and not date:
 224                     try:
 225                         version = node.args[1].value
 226                     except IndexError:
 227                         self.add_message('ansible-deprecated-no-version', node=node)
 228                         return
 229                 if version and date:
 230                     self.add_message('ansible-deprecated-both-version-and-date', node=node)
 231 
 232                 if collection_name:
 233                     this_collection = collection_name == (self.collection_name or 'ansible.builtin')
 234                     if not this_collection:
 235                         self.add_message('wrong-collection-deprecated', node=node, args=(collection_name,))
 236                 else:
 237                     self.add_message('ansible-deprecated-no-collection-name', node=node)
 238 
 239                 if date:
 240                     self._check_date(node, date)
 241                 elif version:
 242                     self._check_version(node, version, collection_name)
 243         except AttributeError:
 244             # Not the type of node we are interested in
 245             pass

 248 def register(linter):
 249     """required method to auto register this checker """
 250     linter.register_checker(AnsibleDeprecatedChecker(linter))
