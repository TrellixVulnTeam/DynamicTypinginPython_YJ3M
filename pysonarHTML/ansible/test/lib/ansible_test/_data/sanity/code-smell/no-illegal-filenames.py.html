   1 #!/usr/bin/env python

   3 # a script to check for illegal filenames on various Operating Systems. The
   4 # main rules are derived from restrictions on Windows
   5 # https://msdn.microsoft.com/en-us/library/aa365247#naming_conventions
   6 from __future__ import (absolute_import, division, print_function)
   7 __metaclass__ = type
   8 
   9 import os
  10 import struct
  11 import sys
  12 
  13 from ansible.module_utils.basic import to_bytes
  14 
  15 ILLEGAL_CHARS = [
  16     b'<',
  17     b'>',
  18     b':',
  19     b'"',
  20     b'/',
  21     b'\\',
  22     b'|',
  23     b'?',
  24     b'*'
  25 ] + [struct.pack("b", i) for i in range(32)]
  26 
  27 ILLEGAL_NAMES = [
  28     "CON",
  29     "PRN",
  30     "AUX",
  31     "NUL",
  32     "COM1",
  33     "COM2",
  34     "COM3",
  35     "COM4",
  36     "COM5",
  37     "COM6",
  38     "COM7",
  39     "COM8",
  40     "COM9",
  41     "LPT1",
  42     "LPT2",
  43     "LPT3",
  44     "LPT4",
  45     "LPT5",
  46     "LPT6",
  47     "LPT7",
  48     "LPT8",
  49     "LPT9",
  50 ]
  51 
  52 ILLEGAL_END_CHARS = [
  53     '.',
  54     ' ',
  55 ]
  56 
  57 
  58 def check_path(path, is_dir=False):
  59     type_name = 'directory' if is_dir else 'file'
  60     file_name = os.path.basename(path.rstrip(os.path.sep))
  61     name = os.path.splitext(file_name)[0]
  62 
  63     if name.upper() in ILLEGAL_NAMES:
  64         print("%s: illegal %s name %s" % (path, type_name, name.upper()))
  65 
  66     if file_name[-1] in ILLEGAL_END_CHARS:
  67         print("%s: illegal %s name end-char '%s'" % (path, type_name, file_name[-1]))
  68 
  69     bfile = to_bytes(file_name, encoding='utf-8')
  70     for char in ILLEGAL_CHARS:
  71         if char in bfile:
  72             bpath = to_bytes(path, encoding='utf-8')
  73             print("%s: illegal char '%s' in %s name" % (bpath, char, type_name))
  74 
  75 
  76 def main():
  77     for path in sys.argv[1:] or sys.stdin.read().splitlines():
  78         check_path(path, is_dir=path.endswith(os.path.sep))
  79 
  80 
  81 if __name__ == '__main__':
  82     main()
