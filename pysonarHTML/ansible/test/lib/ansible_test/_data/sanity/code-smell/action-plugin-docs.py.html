  40     """Return the full name of the plugin at the given path by matching against the given path prefixes, or None if no match is found."""
  41     for prefix, flat in prefixes.items():

  44 
  45             if flat:
  46                 full_name = os.path.basename(relative_path)

  48                 full_name = relative_path

  56 
  57             if name.startswith('_'):

  59
