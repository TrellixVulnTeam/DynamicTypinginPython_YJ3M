  19 def isodate(value):
  20     """Validate a datetime.date or ISO 8601 date string."""
  21     # datetime.date objects come from YAML dates, these are ok
  22     if isinstance(value, datetime.date):
  23         return value
  24     # make sure we have a string
  25     msg = 'Expected ISO 8601 date string (YYYY-MM-DD), or YAML date'
  26     if not isinstance(value, string_types):
  27         raise Invalid(msg)
  28     try:
  29         datetime.datetime.strptime(value, '%Y-%m-%d').date()
  30     except ValueError:
  31         raise Invalid(msg)
  32     return value

  35 def validate_metadata_file(path):
  36     """Validate explicit runtime metadata file"""
  37     try:
  38         with open(path, 'r') as f_path:
  39             routing = yaml.safe_load(f_path)
  40     except yaml.error.MarkedYAMLError as ex:
  41         print('%s:%d:%d: YAML load failed: %s' % (path, ex.context_mark.line +
  42                                                   1, ex.context_mark.column + 1, re.sub(r'\s+', ' ', str(ex))))
  43         return
  44     except Exception as ex:  # pylint: disable=broad-except
  45         print('%s:%d:%d: YAML load failed: %s' %
  46               (path, 0, 0, re.sub(r'\s+', ' ', str(ex))))
  47         return

 134 def main():
 135     """Validate runtime metadata"""
 136     paths = sys.argv[1:] or sys.stdin.read().splitlines()
 137 
 138     collection_legacy_file = 'meta/routing.yml'
 139     collection_runtime_file = 'meta/runtime.yml'
 140 
 141     for path in paths:
 142         if path == collection_legacy_file:
 143             print('%s:%d:%d: %s' % (path, 0, 0, ("Should be called '%s'" % collection_runtime_file)))
