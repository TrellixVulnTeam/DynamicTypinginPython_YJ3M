 124     class RestrictedModuleLoader:
 125         """Python module loader that restricts inappropriate imports."""
 126         def __init__(self, path, name):
 127             self.path = path
 128             self.name = name
 129             self.loaded_modules = set()
 130 
 131         def find_module(self, fullname, path=None):
 132             """Return self if the given fullname is restricted, otherwise return None.
 133             :param fullname: str
 134             :param path: str
 135             :return: RestrictedModuleLoader | None
 136             """
 137             if fullname in self.loaded_modules:
 138                 return None  # ignore modules that are already being loaded
 139 
 140             if is_name_in_namepace(fullname, ['ansible']):
 141                 if fullname in ('ansible.module_utils.basic', 'ansible.module_utils.common.removed'):
 142                     return self  # intercept loading so we can modify the result
 143 
 144                 if is_name_in_namepace(fullname, ['ansible.module_utils', self.name]):
 145                     return None  # module_utils and module under test are always allowed
 146 
 147                 if any(os.path.exists(candidate_path) for candidate_path in convert_ansible_name_to_absolute_paths(fullname)):
 148                     return self  # restrict access to ansible files that exist
 149 
 150                 return None  # ansible file does not exist, do not restrict access
 151 
 152             if is_name_in_namepace(fullname, ['ansible_collections']):
 153                 if not collection_loader:
 154                     return self  # restrict access to collections when we are not testing a collection
 155 
 156                 if is_name_in_namepace(fullname, ['ansible_collections...plugins.module_utils', self.name]):
 157                     return None  # module_utils and module under test are always allowed
 158 
 159                 if collection_loader.find_module(fullname, path):
 160                     return self  # restrict access to collection files that exist
 161 
 162                 return None  # collection file does not exist, do not restrict access
 163 
 164             # not a namespace we care about
 165             return None
 166 
 167         def load_module(self, fullname):
 168             """Raise an ImportError.
 169             :type fullname: str
 170             """
 171             if fullname == 'ansible.module_utils.basic':
 172                 module = self.__load_module(fullname)
 173 
 174                 # stop Ansible module execution during AnsibleModule instantiation
 175                 module.AnsibleModule = ImporterAnsibleModule
 176                 # no-op for _load_params since it may be called before instantiating AnsibleModule
 177                 module._load_params = lambda *args, **kwargs: {}  # pylint: disable=protected-access
 178 
 179                 return module
 180 
 181             if fullname == 'ansible.module_utils.common.removed':
 182                 module = self.__load_module(fullname)
 183 
 184                 # no-op for removed_module since it is called in place of AnsibleModule instantiation
 185                 module.removed_module = lambda *args, **kwargs: None
 186 
 187                 return module
 188 
 189             raise ImportError('import of "%s" is not allowed in this context' % fullname)
 190 
 191         def __load_module(self, fullname):
 192             """Load the requested module while avoiding infinite recursion.
 193             :type fullname: str
 194             :rtype: module
 195             """
 196             self.loaded_modules.add(fullname)
 197             return import_module(fullname)

 211     def test_python_module(path, name, base_dir, messages):
 212         """Test the given python module by importing it.
 213         :type path: str
 214         :type name: str
 215         :type base_dir: str
 216         :type messages: set[str]
 217         """
 218         if name in sys.modules:
 219             return  # cannot be tested because it has already been loaded
 220 
 221         is_ansible_module = (path.startswith('lib/ansible/modules/') or path.startswith('plugins/modules/')) and os.path.basename(path) != '__init__.py'
 222         run_main = is_ansible_module
 223 
 224         if path == 'lib/ansible/modules/async_wrapper.py':
 225             # async_wrapper is a non-standard Ansible module (does not use AnsibleModule) so we cannot test the main function
 226             run_main = False
 227 
 228         capture_normal = Capture()
 229         capture_main = Capture()
 230 
 231         try:
 232             with monitor_sys_modules(path, messages):
 233                 with restrict_imports(path, name, messages):
 234                     with capture_output(capture_normal):
 235                         import_module(name)
 236 
 237             if run_main:
 238                 with monitor_sys_modules(path, messages):
 239                     with restrict_imports(path, name, messages):
 240                         with capture_output(capture_main):
 241                             runpy.run_module(name, run_name='__main__', alter_sys=True)
 242         except ImporterAnsibleModuleException:
 243             # module instantiated AnsibleModule without raising an exception
 244             pass
 245         except BaseException as ex:  # pylint: disable=locally-disabled, broad-except
 246             # intentionally catch all exceptions, including calls to sys.exit
 247             exc_type, _exc, exc_tb = sys.exc_info()
 248             message = str(ex)
 249             results = list(reversed(traceback.extract_tb(exc_tb)))
 250             line = 0
 251             offset = 0
 252             full_path = os.path.join(base_dir, path)
 253             base_path = base_dir + os.path.sep
 254             source = None
 255 
 256             # avoid line wraps in messages
 257             message = re.sub(r'\n *', ': ', message)
 258 
 259             for result in results:
 260                 if result[0] == full_path:
 261                     # save the line number for the file under test
 262                     line = result[1] or 0
 263 
 264                 if not source and result[0].startswith(base_path) and not result[0].startswith(temp_path):
 265                     # save the first path and line number in the traceback which is in our source tree
 266                     source = (os.path.relpath(result[0], base_path), result[1] or 0, 0)
 267 
 268             if isinstance(ex, SyntaxError):
 269                 # SyntaxError has better information than the traceback
 270                 if ex.filename == full_path:  # pylint: disable=locally-disabled, no-member
 271                     # syntax error was reported in the file under test
 272                     line = ex.lineno or 0  # pylint: disable=locally-disabled, no-member
 273                     offset = ex.offset or 0  # pylint: disable=locally-disabled, no-member
 274                 elif ex.filename.startswith(base_path) and not ex.filename.startswith(temp_path):  # pylint: disable=locally-disabled, no-member
 275                     # syntax error was reported in our source tree
 276                     source = (os.path.relpath(ex.filename, base_path), ex.lineno or 0, ex.offset or 0)  # pylint: disable=locally-disabled, no-member
 277 
 278                 # remove the filename and line number from the message
 279                 # either it was extracted above, or it's not really useful information
 280                 message = re.sub(r' \(.*?, line [0-9]+\)$', '', message)
 281 
 282             if source and source[0] != path:
 283                 message += ' (at %s:%d:%d)' % (source[0], source[1], source[2])
 284 
 285             report_message(path, line, offset, 'traceback', '%s: %s' % (exc_type.__name__, message), messages)
 286         finally:
 287             capture_report(path, capture_normal, messages)
 288             capture_report(path, capture_main, messages)

 290     def is_name_in_namepace(name, namespaces):
 291         """Returns True if the given name is one of the given namespaces, otherwise returns False."""
 292         name_parts = name.split('.')
 293 
 294         for namespace in namespaces:
 295             namespace_parts = namespace.split('.')
 296             length = min(len(name_parts), len(namespace_parts))
 297 
 298             truncated_name = name_parts[0:length]
 299             truncated_namespace = namespace_parts[0:length]
 300 
 301             # empty parts in the namespace are treated as wildcards
 302             # to simplify the comparison, use those empty parts to indicate the positions in the name to be empty as well
 303             for idx, part in enumerate(truncated_namespace):
 304                 if not part:
 305                     truncated_name[idx] = part
 306 
 307             # example: name=ansible, allowed_name=ansible.module_utils
 308             # example: name=ansible.module_utils.system.ping, allowed_name=ansible.module_utils
 309             if truncated_name == truncated_namespace:
 310                 return True
 311 
 312         return False

 314     def check_sys_modules(path, before, messages):
 315         """Check for unwanted changes to sys.modules.
 316         :type path: str
 317         :type before: dict[str, module]
 318         :type messages: set[str]
 319         """
 320         after = sys.modules
 321         removed = set(before.keys()) - set(after.keys())
 322         changed = set(key for key, value in before.items() if key in after and value != after[key])
 323 
 324         # additions are checked by our custom PEP 302 loader, so we don't need to check them again here
 325 
 326         for module in sorted(removed):
 327             report_message(path, 0, 0, 'unload', 'unloading of "%s" in sys.modules is not supported' % module, messages)
 328 
 329         for module in sorted(changed):
 330             report_message(path, 0, 0, 'reload', 'reloading of "%s" in sys.modules is not supported' % module, messages)

 332     def convert_ansible_name_to_absolute_paths(name):

 337         return [

 339             os.path.join(ansible_path, name.replace('.', os.path.sep)) + '.py',

 342     def convert_relative_path_to_name(path):
 343         """Calculate the module name from the given path.
 344         :type path: str
 345         :rtype: str
 346         """
 347         if path.endswith('/__init__.py'):
 348             clean_path = os.path.dirname(path)
 349         else:
 350             clean_path = path
 351 
 352         clean_path = os.path.splitext(clean_path)[0]
 353 
 354         name = clean_path.replace(os.path.sep, '.')
 355 
 356         if collection_loader:
 357             # when testing collections the relative paths (and names) being tested are within the collection under test
 358             name = 'ansible_collections.%s.%s' % (collection_full_name, name)
 359         else:
 360             # when testing ansible all files being imported reside under the lib directory
 361             name = name[len('lib/'):]
 362 
 363         return name

 365     class Capture:
 366         """Captured output and/or exception."""
 367         def __init__(self):
 368             self.stdout = StringIO()
 369             self.stderr = StringIO()

 371     def capture_report(path, capture, messages):
 372         """Report on captured output.
 373         :type path: str
 374         :type capture: Capture
 375         :type messages: set[str]
 376         """
 377         if capture.stdout.getvalue():
 378             first = capture.stdout.getvalue().strip().splitlines()[0].strip()
 379             report_message(path, 0, 0, 'stdout', first, messages)
 380 
 381         if capture.stderr.getvalue():
 382             first = capture.stderr.getvalue().strip().splitlines()[0].strip()
 383             report_message(path, 0, 0, 'stderr', first, messages)

 385     def report_message(path, line, column, code, message, messages):
 386         """Report message if not already reported.
 387         :type path: str
 388         :type line: int
 389         :type column: int
 390         :type code: str
 391         :type message: str
 392         :type messages: set[str]
 393         """
 394         message = '%s:%d:%d: %s: %s' % (path, line, column, code, message)
 395 
 396         if message not in messages:
 397             messages.add(message)
 398             print(message)

 400     @contextlib.contextmanager
 401     def restrict_imports(path, name, messages):
 402         """Restrict available imports.
 403         :type path: str
 404         :type name: str
 405         :type messages: set[str]
 406         """
 407         restricted_loader = RestrictedModuleLoader(path, name)
 408 
 409         # noinspection PyTypeChecker
 410         sys.meta_path.insert(0, restricted_loader)
 411         sys.path_importer_cache.clear()
 412 
 413         try:
 414             yield
 415         finally:
 416             if sys.meta_path[0] != restricted_loader:
 417                 report_message(path, 0, 0, 'metapath', 'changes to sys.meta_path[0] are not permitted', messages)
 418 
 419             while restricted_loader in sys.meta_path:
 420                 # noinspection PyTypeChecker
 421                 sys.meta_path.remove(restricted_loader)
 422 
 423             sys.path_importer_cache.clear()
 424 
 425     @contextlib.contextmanager
 426     def monitor_sys_modules(path, messages):
 427         """Monitor sys.modules for unwanted changes, reverting any additions made to our own namespaces."""
 428         snapshot = sys.modules.copy()
 429 
 430         try:
 431             yield
 432         finally:
 433             check_sys_modules(path, snapshot, messages)
 434 
 435             for key in set(sys.modules.keys()) - set(snapshot.keys()):
 436                 if is_name_in_namepace(key, ('ansible', 'ansible_collections')):
 437                     del sys.modules[key]  # only unload our own code since we know it's native Python
 438 
 439     @contextlib.contextmanager
 440     def capture_output(capture):
 441         """Capture sys.stdout and sys.stderr.
 442         :type capture: Capture
 443         """
 444         old_stdout = sys.stdout
 445         old_stderr = sys.stderr
 446 
 447         sys.stdout = capture.stdout
 448         sys.stderr = capture.stderr
 449 
 450         # clear all warnings registries to make all warnings available
 451         for module in sys.modules.values():
 452             try:
 453                 # noinspection PyUnresolvedReferences
 454                 module.__warningregistry__.clear()
 455             except AttributeError:
 456                 pass
 457 
 458         with warnings.catch_warnings():
 459             warnings.simplefilter('error')
 460 
 461             try:
 462                 yield
 463             finally:
 464                 sys.stdout = old_stdout
 465                 sys.stderr = old_stderr
 466
