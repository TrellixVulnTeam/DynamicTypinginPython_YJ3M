  56 class YamlChecker:
  57     """Wrapper around yamllint that supports YAML embedded in Ansible modules."""
  58     def __init__(self):
  59         self.messages = []
  60 
  61     def report(self):
  62         """Print yamllint report to stdout."""
  63         report = dict(
  64             messages=self.messages,
  65         )
  66 
  67         print(json.dumps(report, indent=4, sort_keys=True))

  69     def check(self, paths):
  70         """
  71         :type paths: str
  72         """
  73         config_path = os.path.join(os.path.dirname(os.path.abspath(__file__)), 'config')
  74 
  75         yaml_conf = YamlLintConfig(file=os.path.join(config_path, 'default.yml'))
  76         module_conf = YamlLintConfig(file=os.path.join(config_path, 'modules.yml'))
  77         plugin_conf = YamlLintConfig(file=os.path.join(config_path, 'plugins.yml'))
  78 
  79         for path in paths:
  80             extension = os.path.splitext(path)[1]
  81 
  82             with open(path) as f:
  83                 contents = f.read()
  84 
  85             if extension in ('.yml', '.yaml'):
  86                 self.check_yaml(yaml_conf, path, contents)
  87             elif extension == '.py':
  88                 if path.startswith('lib/ansible/modules/') or path.startswith('plugins/modules/'):
  89                     conf = module_conf
  90                 else:
  91                     conf = plugin_conf
  92 
  93                 self.check_module(conf, path, contents)
  94             else:
  95                 raise Exception('unsupported extension: %s' % extension)
  96 
  97     def check_yaml(self, conf, path, contents):
  98         """
  99         :type conf: YamlLintConfig
 100         :type path: str
 101         :type contents: str
 102         """
 103         self.check_parsable(path, contents)
 104         self.messages += [self.result_to_message(r, path) for r in linter.run(contents, conf, path)]
 105 
 106     def check_module(self, conf, path, contents):
 107         """
 108         :type conf: YamlLintConfig
 109         :type path: str
 110         :type contents: str
 111         """
 112         docs = self.get_module_docs(path, contents)
 113 
 114         for key, value in docs.items():
 115             yaml_data = value['yaml']
 116             lineno = value['lineno']
 117 
 118             if yaml_data.startswith('\n'):
 119                 yaml_data = yaml_data[1:]
 120                 lineno += 1
 121 
 122             self.check_parsable(path, yaml_data, lineno)
 123 
 124             messages = list(linter.run(yaml_data, conf, path))
 125 
 126             self.messages += [self.result_to_message(r, path, lineno - 1, key) for r in messages]
 127 
 128     def check_parsable(self, path, contents, lineno=1):
 129         """
 130         :type path: str
 131         :type contents: str
 132         :type lineno: int
 133         """
 134         try:
 135             yaml.load(contents, Loader=TestLoader)
 136         except MarkedYAMLError as e:
 137             self.messages += [{'code': 'unparsable-with-libyaml',
 138                                'message': '%s - %s' % (e.args[0], e.args[2]),
 139                                'path': path,
 140                                'line': e.problem_mark.line + lineno,
 141                                'column': e.problem_mark.column + 1,
 142                                'level': 'error',
 143                                }]
 144 
 145     @staticmethod
 146     def result_to_message(result, path, line_offset=0, prefix=''):
 147         """
 148         :type result: any
 149         :type path: str
 150         :type line_offset: int
 151         :type prefix: str
 152         :rtype: dict[str, any]
 153         """
 154         if prefix:
 155             prefix = '%s: ' % prefix
 156 
 157         return dict(
 158             code=result.rule or result.level,
 159             message=prefix + result.desc,
 160             path=path,
 161             line=result.line + line_offset,
 162             column=result.column,
 163             level=result.level,
 164         )
 165 
 166     def get_module_docs(self, path, contents):
 167         """
 168         :type path: str
 169         :type contents: str
 170         :rtype: dict[str, any]
 171         """
 172         module_doc_types = [
 173             'DOCUMENTATION',
 174             'EXAMPLES',
 175             'RETURN',
 176         ]
 177 
 178         docs = {}
 179 
 180         def check_assignment(statement, doc_types=None):
 181             """Check the given statement for a documentation assignment."""
 182             for target in statement.targets:
 183                 if isinstance(target, ast.Tuple):
 184                     continue
 185 
 186                 if doc_types and target.id not in doc_types:
 187                     continue
 188 
 189                 docs[target.id] = dict(
 190                     yaml=statement.value.s,
 191                     lineno=statement.lineno,
 192                     end_lineno=statement.lineno + len(statement.value.s.splitlines())
 193                 )
 194 
 195         module_ast = self.parse_module(path, contents)
 196 
 197         if not module_ast:
 198             return {}
 199 
 200         is_plugin = path.startswith('lib/ansible/modules/') or path.startswith('lib/ansible/plugins/') or path.startswith('plugins/')
 201         is_doc_fragment = path.startswith('lib/ansible/plugins/doc_fragments/') or path.startswith('plugins/doc_fragments/')
 202 
 203         if is_plugin and not is_doc_fragment:
 204             for body_statement in module_ast.body:
 205                 if isinstance(body_statement, ast.Assign):
 206                     check_assignment(body_statement, module_doc_types)
 207         elif is_doc_fragment:
 208             for body_statement in module_ast.body:
 209                 if isinstance(body_statement, ast.ClassDef):
 210                     for class_statement in body_statement.body:
 211                         if isinstance(class_statement, ast.Assign):
 212                             check_assignment(class_statement)
 213         else:
 214             raise Exception('unsupported path: %s' % path)
 215 
 216         return docs
 217 
 218     def parse_module(self, path, contents):
 219         """
 220         :type path: str
 221         :type contents: str
 222         :rtype: ast.Module | None
 223         """
 224         try:
 225             return ast.parse(contents)
 226         except SyntaxError as ex:
 227             self.messages.append(dict(
 228                 code='python-syntax-error',
 229                 message=str(ex),
 230                 path=path,
 231                 line=ex.lineno,
 232                 column=ex.offset,
 233                 level='error',
 234             ))
 235         except Exception as ex:  # pylint: disable=broad-except
 236             self.messages.append(dict(
 237                 code='python-parse-error',
 238                 message=str(ex),
 239                 path=path,
 240                 line=0,
 241                 column=0,
 242                 level='error',
 243             ))
 244 
 245         return None
