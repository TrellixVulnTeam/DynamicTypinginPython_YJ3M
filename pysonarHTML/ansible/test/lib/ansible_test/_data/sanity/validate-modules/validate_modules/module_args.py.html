  49 class _FakeAnsibleModuleInit:
  50     def __init__(self):
  51         self.args = tuple()
  52         self.kwargs = {}
  53         self.called = False
  54 
  55     def __call__(self, *args, **kwargs):
  56         self.args = args
  57         self.kwargs = kwargs
  58         self.called = True
  59         raise AnsibleModuleCallError('AnsibleModuleCallError')

  66 @contextmanager
  67 def setup_env(filename):
  68     # Used to clean up imports later
  69     pre_sys_modules = list(sys.modules.keys())
  70 
  71     fake = _FakeAnsibleModuleInit()
  72     module = __import__('ansible.module_utils.basic').module_utils.basic
  73     _original_init = module.AnsibleModule.__init__
  74     _original_load_params = module._load_params
  75     setattr(module.AnsibleModule, '__init__', fake)
  76     setattr(module, '_load_params', _fake_load_params)
  77 
  78     try:
  79         yield fake
  80     finally:
  81         setattr(module.AnsibleModule, '__init__', _original_init)
  82         setattr(module, '_load_params', _original_load_params)
  83 
  84         # Clean up imports to prevent issues with mutable data being used in modules
  85         for k in list(sys.modules.keys()):
  86             # It's faster if we limit to items in ansible.module_utils
  87             # But if this causes problems later, we should remove it
  88             if k not in pre_sys_modules and k.startswith('ansible.module_utils.'):
  89                 del sys.modules[k]

  92 def get_ps_argument_spec(filename, collection):
  93     fqc_name = get_module_name_from_filename(filename, collection)
  94 
  95     pwsh = find_executable('pwsh')
  96     if not pwsh:
  97         raise FileNotFoundError('Required program for PowerShell arg spec inspection "pwsh" not found.')
  98 
  99     module_path = os.path.join(os.getcwd(), filename)
 100     b_module_path = to_bytes(module_path, errors='surrogate_or_strict')
 101     with open(b_module_path, mode='rb') as module_fd:
 102         b_module_data = module_fd.read()
 103 
 104     ps_dep_finder = PSModuleDepFinder()
 105     ps_dep_finder.scan_module(b_module_data, fqn=fqc_name)
 106 
 107     # For ps_argspec.ps1 to compile Ansible.Basic it also needs the AddType module_util.
 108     ps_dep_finder._add_module((b"Ansible.ModuleUtils.AddType", ".psm1", None), wrapper=False)
 109 
 110     util_manifest = json.dumps({
 111         'module_path': to_text(module_path, errors='surrogiate_or_strict'),
 112         'ansible_basic': ps_dep_finder.cs_utils_module["Ansible.Basic"]['path'],
 113         'ps_utils': dict([(name, info['path']) for name, info in ps_dep_finder.ps_modules.items()]),
 114     })
 115 
 116     script_path = os.path.join(os.path.dirname(os.path.realpath(__file__)), 'ps_argspec.ps1')
 117     proc = subprocess.Popen([script_path, util_manifest], stdout=subprocess.PIPE, stderr=subprocess.PIPE,
 118                             shell=False)
 119     stdout, stderr = proc.communicate()
 120 
 121     if proc.returncode != 0:
 122         raise AnsibleModuleImportError("STDOUT:\n%s\nSTDERR:\n%s" % (stdout.decode('utf-8'), stderr.decode('utf-8')))
 123 
 124     kwargs = json.loads(stdout)
 125 
 126     # the validate-modules code expects the options spec to be under the argument_spec key not options as set in PS
 127     kwargs['argument_spec'] = kwargs.pop('options', {})
 128 
 129     return kwargs['argument_spec'], (), kwargs

 132 def get_py_argument_spec(filename, collection):
 133     name = get_module_name_from_filename(filename, collection)
 134 
 135     with setup_env(filename) as fake:
 136         try:
 137             with CaptureStd():
 138                 runpy.run_module(name, run_name='__main__', alter_sys=True)
 139         except AnsibleModuleCallError:
 140             pass
 141         except BaseException as e:
 142             # we want to catch all exceptions here, including sys.exit
 143             reraise(AnsibleModuleImportError, AnsibleModuleImportError('%s' % e), sys.exc_info()[2])
 144 
 145         if not fake.called:
 146             raise AnsibleModuleNotInitialized()
 147 
 148     try:
 149         try:
 150             # for ping kwargs == {'argument_spec':{'data':{'type':'str','default':'pong'}}, 'supports_check_mode':True}
 151             argument_spec = fake.kwargs['argument_spec']
 152             # If add_file_common_args is truish, add options from FILE_COMMON_ARGUMENTS when not present.
 153             # This is the only modification to argument_spec done by AnsibleModule itself, and which is
 154             # not caught by setup_env's AnsibleModule replacement
 155             if fake.kwargs.get('add_file_common_args'):
 156                 for k, v in FILE_COMMON_ARGUMENTS.items():
 157                     if k not in argument_spec:
 158                         argument_spec[k] = v
 159             return argument_spec, fake.args, fake.kwargs
 160         except KeyError:
 161             return fake.args[0], fake.args, fake.kwargs
 162     except (TypeError, IndexError):
 163         return {}, (), {}

 166 def get_argument_spec(filename, collection):
 167     if filename.endswith('.py'):
 168         return get_py_argument_spec(filename, collection)
 169     else:
 170         return get_ps_argument_spec(filename, collection)
