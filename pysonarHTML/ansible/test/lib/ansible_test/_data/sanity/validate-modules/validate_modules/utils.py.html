  42 def find_executable(executable, cwd=None, path=None):
  43     """Finds the full path to the executable specified"""
  44     match = None
  45     real_cwd = os.getcwd()
  46 
  47     if not cwd:
  48         cwd = real_cwd
  49 
  50     if os.path.dirname(executable):
  51         target = os.path.join(cwd, executable)
  52         if os.path.exists(target) and os.access(target, os.F_OK | os.X_OK):
  53             match = executable
  54     else:
  55         path = os.environ.get('PATH', os.path.defpath)
  56 
  57         path_dirs = path.split(os.path.pathsep)
  58         seen_dirs = set()
  59 
  60         for path_dir in path_dirs:
  61             if path_dir in seen_dirs:
  62                 continue
  63 
  64             seen_dirs.add(path_dir)
  65 
  66             if os.path.abspath(path_dir) == real_cwd:
  67                 path_dir = cwd
  68 
  69             candidate = os.path.join(path_dir, executable)
  70 
  71             if os.path.exists(candidate) and os.access(candidate, os.F_OK | os.X_OK):
  72                 match = candidate
  73                 break
  74 
  75     return match

  78 def find_globals(g, tree):
  79     """Uses AST to find globals in an ast tree"""
  80     for child in tree:
  81         if hasattr(child, 'body') and isinstance(child.body, list):
  82             find_globals(g, child.body)
  83         elif isinstance(child, (ast.FunctionDef, ast.ClassDef)):
  84             g.add(child.name)
  85             continue
  86         elif isinstance(child, ast.Assign):
  87             try:
  88                 g.add(child.targets[0].id)
  89             except (IndexError, AttributeError):
  90                 pass
  91         elif isinstance(child, ast.Import):
  92             g.add(child.names[0].name)
  93         elif isinstance(child, ast.ImportFrom):
  94             for name in child.names:
  95                 g_name = name.asname or name.name
  96                 if g_name == '*':
  97                     continue
  98                 g.add(g_name)

 101 class CaptureStd():
 102     """Context manager to handle capturing stderr and stdout"""
 103 
 104     def __enter__(self):
 105         self.sys_stdout = sys.stdout
 106         self.sys_stderr = sys.stderr
 107         sys.stdout = self.stdout = AnsibleTextIOWrapper(BytesIO(), encoding=self.sys_stdout.encoding)
 108         sys.stderr = self.stderr = AnsibleTextIOWrapper(BytesIO(), encoding=self.sys_stderr.encoding)
 109         return self
 110 
 111     def __exit__(self, exc_type, exc_value, traceback):
 112         sys.stdout = self.sys_stdout
 113         sys.stderr = self.sys_stderr
 114 
 115     def get(self):
 116         """Return ``(stdout, stderr)``"""
 117 
 118         return self.stdout.buffer.getvalue(), self.stderr.buffer.getvalue()

 121 def get_module_name_from_filename(filename, collection):
 122     # Calculate the module's name so that relative imports work correctly
 123     if collection:
 124         # collection is a relative path, example: ansible_collections/my_namespace/my_collection
 125         # filename is a relative path, example: plugins/modules/my_module.py
 126         path = os.path.join(collection, filename)
 127     else:
 128         # filename is a relative path, example: lib/ansible/modules/system/ping.py
 129         path = os.path.relpath(filename, 'lib')
 130 
 131     name = os.path.splitext(path)[0].replace(os.path.sep, '.')
 132 
 133     return name

 136 def parse_yaml(value, lineno, module, name, load_all=False):
 137     traces = []
 138     errors = []
 139     data = None
 140 
 141     if load_all:
 142         loader = yaml.safe_load_all
 143     else:
 144         loader = yaml.safe_load
 145 
 146     try:
 147         data = loader(value)
 148         if load_all:
 149             data = list(data)
 150     except yaml.MarkedYAMLError as e:
 151         e.problem_mark.line += lineno - 1
 152         e.problem_mark.name = '%s.%s' % (module, name)
 153         errors.append({
 154             'msg': '%s is not valid YAML' % name,
 155             'line': e.problem_mark.line + 1,
 156             'column': e.problem_mark.column + 1
 157         })
 158         traces.append(e)
 159     except yaml.reader.ReaderError as e:
 160         traces.append(e)
 161         # TODO: Better line/column detection
 162         errors.append({
 163             'msg': ('%s is not valid YAML. Character '
 164                     '0x%x at position %d.' % (name, e.character, e.position)),
 165             'line': lineno
 166         })
 167     except yaml.YAMLError as e:
 168         traces.append(e)
 169         errors.append({
 170             'msg': '%s is not valid YAML: %s: %s' % (name, type(e), e),
 171             'line': lineno
 172         })
 173 
 174     return data, errors, traces

 177 def is_empty(value):
 178     """Evaluate null like values excluding False"""
 179     if value is False:
 180         return False
 181     return not bool(value)

 184 def compare_unordered_lists(a, b):
 185     """Safe list comparisons
 186 
 187     Supports:
 188       - unordered lists
 189       - unhashable elements
 190     """
 191     return len(a) == len(b) and all(x in b for x in a)

 202 def parse_isodate(v, allow_date):
 203     if allow_date:
 204         if isinstance(v, datetime.date):
 205             return v
 206         msg = 'Expected ISO 8601 date string (YYYY-MM-DD) or YAML date'
 207     else:
 208         msg = 'Expected ISO 8601 date string (YYYY-MM-DD)'
 209     if not isinstance(v, string_types):
 210         raise ValueError(msg)
 211     # From Python 3.7 in, there is datetime.date.fromisoformat(). For older versions,
 212     # we have to do things manually.
 213     if not re.match('^[0-9]{4}-[0-9]{2}-[0-9]{2}$', v):
 214         raise ValueError(msg)
 215     try:
 216         return datetime.datetime.strptime(v, '%Y-%m-%d').date()
 217     except ValueError:
 218         raise ValueError(msg)
