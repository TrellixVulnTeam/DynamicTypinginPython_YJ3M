  50 def collection_name(v, error_code=None):
  51     if not isinstance(v, string_types):
  52         raise _add_ansible_error_code(
  53             Invalid('Collection name must be a string'), error_code or 'collection-invalid-name')
  54     m = COLLECTION_NAME_RE.match(v)
  55     if not m:
  56         raise _add_ansible_error_code(
  57             Invalid('Collection name must be of format `<namespace>.<name>`'), error_code or 'collection-invalid-name')
  58     return v

 127 def no_required_with_default(v):

 129         raise Invalid('required=True cannot be supplied with a default')

 133 def elements_with_list(v):

 135         raise Invalid('type must be list to use elements')

 145 def option_deprecation(v):
 146     if v.get('removed_in_version') or v.get('removed_at_date'):
 147         if v.get('removed_in_version') and v.get('removed_at_date'):
 148             raise _add_ansible_error_code(
 149                 Invalid('Only one of removed_in_version and removed_at_date must be specified'),
 150                 error_code='deprecation-either-date-or-version')
 151         if not v.get('removed_from_collection'):
 152             raise _add_ansible_error_code(
 153                 Invalid('If removed_in_version or removed_at_date is specified, '
 154                         'removed_from_collection must be specified as well'),
 155                 error_code='deprecation-collection-missing')
 156         return
 157     if v.get('removed_from_collection'):
 158         raise Invalid('removed_from_collection cannot be specified without either '
 159                       'removed_in_version or removed_at_date')

 162 def argument_spec_schema(for_collection):
 163     any_string_types = Any(*string_types)
 164     schema = {
 165         any_string_types: {
 166             'type': Any(is_callable, *argument_spec_types),
 167             'elements': Any(*argument_spec_types),
 168             'default': object,
 169             'fallback': Any(
 170                 (is_callable, list_string_types),
 171                 [is_callable, list_string_types],
 172             ),
 173             'choices': Any([object], (object,)),
 174             'required': bool,
 175             'no_log': bool,
 176             'aliases': Any(list_string_types, tuple(list_string_types)),
 177             'apply_defaults': bool,
 178             'removed_in_version': version(for_collection),
 179             'removed_at_date': date(),
 180             'removed_from_collection': collection_name,
 181             'options': Self,
 182             'deprecated_aliases': Any([Any(
 183                 {
 184                     Required('name'): Any(*string_types),
 185                     Required('date'): date(),
 186                     Required('collection_name'): collection_name,
 187                 },
 188                 {
 189                     Required('name'): Any(*string_types),
 190                     Required('version'): version(for_collection),
 191                     Required('collection_name'): collection_name,
 192                 },
 193             )]),
 194         }
 195     }
 196     schema[any_string_types].update(argument_spec_modifiers)
 197     schemas = All(
 198         schema,
 199         Schema({any_string_types: no_required_with_default}),
 200         Schema({any_string_types: elements_with_list}),
 201         Schema({any_string_types: options_with_apply_defaults}),
 202         Schema({any_string_types: option_deprecation}),
 203     )
 204     return Schema(schemas)

 229 def version_added(v):
 230     if 'version_added' in v:
 231         version_added = v.get('version_added')
 232         if isinstance(version_added, string_types):
 233             # If it is not a string, schema validation will have already complained
 234             # - or we have a float and we are in ansible/ansible, in which case we're
 235             # also happy.
 236             if v.get('version_added_collection') == 'ansible.builtin':
 237                 try:
 238                     version = StrictVersion()
 239                     version.parse(version_added)
 240                 except ValueError as exc:
 241                     raise _add_ansible_error_code(
 242                         Invalid('version_added (%r) is not a valid ansible-base version: '
 243                                 '%s' % (version_added, exc)),
 244                         error_code='deprecation-either-date-or-version')
 245             else:
 246                 try:
 247                     version = SemanticVersion()
 248                     version.parse(version_added)
 249                 except ValueError as exc:
 250                     raise _add_ansible_error_code(
 251                         Invalid('version_added (%r) is not a valid collection version '
 252                                 '(see specification at https://semver.org/): '
 253                                 '%s' % (version_added, exc)),
 254                         error_code='deprecation-either-date-or-version')
 255     elif 'version_added_collection' in v:
 256         # Must have been manual intervention, since version_added_collection is only
 257         # added automatically when version_added is present
 258         raise Invalid('version_added_collection cannot be specified without version_added')
 259     return v

 262 def list_dict_option_schema(for_collection):
 263     suboption_schema = Schema(
 264         {
 265             Required('description'): Any(list_string_types, *string_types),
 266             'required': bool,
 267             'choices': list,
 268             'aliases': Any(list_string_types),
 269             'version_added': version(for_collection),
 270             'version_added_collection': collection_name,
 271             'default': json_value,
 272             # Note: Types are strings, not literal bools, such as True or False
 273             'type': Any(None, 'bits', 'bool', 'bytes', 'dict', 'float', 'int', 'json', 'jsonarg', 'list', 'path', 'raw', 'sid', 'str'),
 274             # in case of type='list' elements define type of individual item in list
 275             'elements': Any(None, 'bits', 'bool', 'bytes', 'dict', 'float', 'int', 'json', 'jsonarg', 'list', 'path', 'raw', 'sid', 'str'),
 276             # Recursive suboptions
 277             'suboptions': Any(None, *list({str_type: Self} for str_type in string_types)),
 278         },
 279         extra=PREVENT_EXTRA
 280     )
 281 
 282     # This generates list of dicts with keys from string_types and suboption_schema value
 283     # for example in Python 3: {str: suboption_schema}
 284     list_dict_suboption_schema = [{str_type: suboption_schema} for str_type in string_types]
 285 
 286     option_schema = Schema(
 287         {
 288             Required('description'): Any(list_string_types, *string_types),
 289             'required': bool,
 290             'choices': list,
 291             'aliases': Any(list_string_types),
 292             'version_added': version(for_collection),
 293             'version_added_collection': collection_name,
 294             'default': json_value,
 295             'suboptions': Any(None, *list_dict_suboption_schema),
 296             # Note: Types are strings, not literal bools, such as True or False
 297             'type': Any(None, 'bits', 'bool', 'bytes', 'dict', 'float', 'int', 'json', 'jsonarg', 'list', 'path', 'raw', 'sid', 'str'),
 298             # in case of type='list' elements define type of individual item in list
 299             'elements': Any(None, 'bits', 'bool', 'bytes', 'dict', 'float', 'int', 'json', 'jsonarg', 'list', 'path', 'raw', 'sid', 'str'),
 300         },
 301         extra=PREVENT_EXTRA
 302     )
 303 
 304     option_version_added = Schema(
 305         All({
 306             'suboptions': Any(None, *[{str_type: Self} for str_type in string_types]),
 307         }, version_added),
 308         extra=ALLOW_EXTRA
 309     )
 310 
 311     # This generates list of dicts with keys from string_types and option_schema value
 312     # for example in Python 3: {str: option_schema}
 313     return [{str_type: All(option_schema, option_version_added)} for str_type in string_types]

 380 def deprecation_schema(for_collection):
 381     main_fields = {
 382         Required('why'): Any(*string_types),
 383         Required('alternative'): Any(*string_types),
 384         Required('removed_from_collection'): collection_name,
 385         'removed': Any(True),
 386     }
 387 
 388     date_schema = {
 389         Required('removed_at_date'): date(),
 390     }
 391     date_schema.update(main_fields)
 392 
 393     if for_collection:
 394         version_schema = {
 395             Required('removed_in'): version(for_collection),
 396         }
 397     else:
 398         version_schema = {
 399             # Only list branches that are deprecated or may have docs stubs in
 400             # Deprecation cycle changed at 2.4 (though not retroactively)
 401             # 2.3 -> removed_in: "2.5" + n for docs stub
 402             # 2.4 -> removed_in: "2.8" + n for docs stub
 403             Required('removed_in'): Any(
 404                 "2.2", "2.3", "2.4", "2.5", "2.6", "2.8", "2.9", "2.10", "2.11", "2.12", "2.13", "2.14"),
 405         }
 406     version_schema.update(main_fields)
 407 
 408     return Any(
 409         Schema(version_schema, extra=PREVENT_EXTRA),
 410         Schema(date_schema, extra=PREVENT_EXTRA),
 411     )

 414 def author(value):
 415     if value is None:
 416         return value  # let schema checks handle
 417 
 418     if not is_iterable(value):
 419         value = [value]
 420 
 421     for line in value:
 422         if not isinstance(line, string_types):
 423             continue  # let schema checks handle
 424         m = author_line.search(line)
 425         if not m:
 426             raise Invalid("Invalid author")
 427 
 428     return value

 431 def doc_schema(module_name, for_collection=False, deprecated_module=False):
 432 
 433     if module_name.startswith('_'):
 434         module_name = module_name[1:]
 435         deprecated_module = True
 436     doc_schema_dict = {
 437         Required('module'): module_name,
 438         Required('short_description'): Any(*string_types),
 439         Required('description'): Any(list_string_types, *string_types),
 440         Required('author'): All(Any(None, list_string_types, *string_types), author),
 441         'notes': Any(None, list_string_types),
 442         'seealso': Any(None, seealso_schema),
 443         'requirements': list_string_types,
 444         'todo': Any(None, list_string_types, *string_types),
 445         'options': Any(None, *list_dict_option_schema(for_collection)),
 446         'extends_documentation_fragment': Any(list_string_types, *string_types),
 447         'version_added_collection': collection_name,
 448     }
 449 
 450     if for_collection:
 451         # Optional
 452         doc_schema_dict['version_added'] = version(for_collection=True)
 453     else:
 454         doc_schema_dict[Required('version_added')] = version(for_collection=False)
 455 
 456     if deprecated_module:
 457         deprecation_required_scheme = {
 458             Required('deprecated'): Any(deprecation_schema(for_collection=for_collection)),
 459         }
 460 
 461         doc_schema_dict.update(deprecation_required_scheme)
 462     return Schema(
 463         doc_schema_dict,
 464         extra=PREVENT_EXTRA
 465     )
