  16 module: win_find
  17 version_added: "2.3"
  18 short_description: Return a list of files based on specific criteria
  19 description:
  20     - Return a list of files based on specified criteria.
  21     - Multiple criteria are AND'd together.
  22     - For non-Windows targets, use the M(find) module instead.
  23 options:
  24     age:
  25         description:
  26             - Select files or folders whose age is equal to or greater than
  27               the specified time.
  28             - Use a negative age to find files equal to or less than
  29               the specified time.
  30             - You can choose seconds, minutes, hours, days or weeks
  31               by specifying the first letter of an of
  32               those words (e.g., "2s", "10d", 1w").
  33         type: str
  34     age_stamp:
  35         description:
  36             - Choose the file property against which we compare C(age).
  37             - The default attribute we compare with is the last modification time.
  38         type: str
  39         choices: [ atime, ctime, mtime ]
  40         default: mtime
  41     checksum_algorithm:
  42         description:
  43             - Algorithm to determine the checksum of a file.
  44             - Will throw an error if the host is unable to use specified algorithm.
  45         type: str
  46         choices: [ md5, sha1, sha256, sha384, sha512 ]
  47         default: sha1
  48     file_type:
  49         description: Type of file to search for.
  50         type: str
  51         choices: [ directory, file ]
  52         default: file
  53     follow:
  54         description:
  55             - Set this to C(yes) to follow symlinks in the path.
  56             - This needs to be used in conjunction with C(recurse).
  57         type: bool
  58         default: no
  59     get_checksum:
  60         description:
  61             - Whether to return a checksum of the file in the return info (default sha1),
  62               use C(checksum_algorithm) to change from the default.
  63         type: bool
  64         default: yes
  65     hidden:
  66         description: Set this to include hidden files or folders.
  67         type: bool
  68         default: no
  69     paths:
  70         description:
  71             - List of paths of directories to search for files or folders in.
  72             - This can be supplied as a single path or a list of paths.
  73         type: list
  74         required: yes
  75     patterns:
  76         description:
  77             - One or more (powershell or regex) patterns to compare filenames with.
  78             - The type of pattern matching is controlled by C(use_regex) option.
  79             - The patterns restrict the list of files or folders to be returned based on the filenames.
  80             - For a file to be matched it only has to match with one pattern in a list provided.
  81         type: list
  82         aliases: [ "regex", "regexp" ]
  83     recurse:
  84         description:
  85             - Will recursively descend into the directory looking for files or folders.
  86         type: bool
  87         default: no
  88     size:
  89         description:
  90             - Select files or folders whose size is equal to or greater than the specified size.
  91             - Use a negative value to find files equal to or less than the specified size.
  92             - You can specify the size with a suffix of the byte type i.e. kilo = k, mega = m...
  93             - Size is not evaluated for symbolic links.
  94         type: str
  95     use_regex:
  96         description:
  97             - Will set patterns to run as a regex check if set to C(yes).
  98         type: bool
  99         default: no
