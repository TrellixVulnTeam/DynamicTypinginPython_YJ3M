  27 def _walk_dirs(topdir, loader, decrypt=True, base_path=None, local_follow=False, trailing_slash_detector=None, checksum_check=False):
  28     """
  29     Walk a filesystem tree returning enough information to copy the files.
  30     This is similar to the _walk_dirs function in ``copy.py`` but returns
  31     a dict instead of a tuple for each entry and includes the checksum of
  32     a local file if wanted.
  33 
  34     :arg topdir: The directory that the filesystem tree is rooted at
  35     :arg loader: The self._loader object from ActionBase
  36     :kwarg decrypt: Whether to decrypt a file encrypted with ansible-vault
  37     :kwarg base_path: The initial directory structure to strip off of the
  38         files for the destination directory.  If this is None (the default),
  39         the base_path is set to ``top_dir``.
  40     :kwarg local_follow: Whether to follow symlinks on the source.  When set
  41         to False, no symlinks are dereferenced.  When set to True (the
  42         default), the code will dereference most symlinks.  However, symlinks
  43         can still be present if needed to break a circular link.
  44     :kwarg trailing_slash_detector: Function to determine if a path has
  45         a trailing directory separator. Only needed when dealing with paths on
  46         a remote machine (in which case, pass in a function that is aware of the
  47         directory separator conventions on the remote machine).
  48     :kawrg whether to get the checksum of the local file and add to the dict
  49     :returns: dictionary of dictionaries. All of the path elements in the structure are text string.
  50             This separates all the files, directories, and symlinks along with
  51             import information about each::
  52 
  53                 {
  54                     'files'; [{
  55                         src: '/absolute/path/to/copy/from',
  56                         dest: 'relative/path/to/copy/to',
  57                         checksum: 'b54ba7f5621240d403f06815f7246006ef8c7d43'
  58                     }, ...],
  59                     'directories'; [{
  60                         src: '/absolute/path/to/copy/from',
  61                         dest: 'relative/path/to/copy/to'
  62                     }, ...],
  63                     'symlinks'; [{
  64                         src: '/symlink/target/path',
  65                         dest: 'relative/path/to/copy/to'
  66                     }, ...],
  67 
  68                 }
  69 
  70         The ``symlinks`` field is only populated if ``local_follow`` is set to False
  71         *or* a circular symlink cannot be dereferenced. The ``checksum`` entry is set
  72         to None if checksum_check=False.
  73 
  74     """
  75     # Convert the path segments into byte strings
  76 
  77     r_files = {'files': [], 'directories': [], 'symlinks': []}
  78 
  79     def _recurse(topdir, rel_offset, parent_dirs, rel_base=u'', checksum_check=False):
  80         """
  81         This is a closure (function utilizing variables from it's parent
  82         function's scope) so that we only need one copy of all the containers.
  83         Note that this function uses side effects (See the Variables used from
  84         outer scope).
  85 
  86         :arg topdir: The directory we are walking for files
  87         :arg rel_offset: Integer defining how many characters to strip off of
  88             the beginning of a path
  89         :arg parent_dirs: Directories that we're copying that this directory is in.
  90         :kwarg rel_base: String to prepend to the path after ``rel_offset`` is
  91             applied to form the relative path.
  92 
  93         Variables used from the outer scope
  94         ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
  95 
  96         :r_files: Dictionary of files in the hierarchy.  See the return value
  97             for :func:`walk` for the structure of this dictionary.
  98         :local_follow: Read-only inside of :func:`_recurse`. Whether to follow symlinks
  99         """
 100         for base_path, sub_folders, files in os.walk(topdir):
 101             for filename in files:
 102                 filepath = os.path.join(base_path, filename)
 103                 dest_filepath = os.path.join(rel_base, filepath[rel_offset:])
 104 
 105                 if os.path.islink(filepath):
 106                     # Dereference the symlnk
 107                     real_file = loader.get_real_file(os.path.realpath(filepath), decrypt=decrypt)
 108                     if local_follow and os.path.isfile(real_file):
 109                         # Add the file pointed to by the symlink
 110                         r_files['files'].append(
 111                             {
 112                                 "src": real_file,
 113                                 "dest": dest_filepath,
 114                                 "checksum": _get_local_checksum(checksum_check, real_file)
 115                             }
 116                         )
 117                     else:
 118                         # Mark this file as a symlink to copy
 119                         r_files['symlinks'].append({"src": os.readlink(filepath), "dest": dest_filepath})
 120                 else:
 121                     # Just a normal file
 122                     real_file = loader.get_real_file(filepath, decrypt=decrypt)
 123                     r_files['files'].append(
 124                         {
 125                             "src": real_file,
 126                             "dest": dest_filepath,
 127                             "checksum": _get_local_checksum(checksum_check, real_file)
 128                         }
 129                     )
 130 
 131             for dirname in sub_folders:
 132                 dirpath = os.path.join(base_path, dirname)
 133                 dest_dirpath = os.path.join(rel_base, dirpath[rel_offset:])
 134                 real_dir = os.path.realpath(dirpath)
 135                 dir_stats = os.stat(real_dir)
 136 
 137                 if os.path.islink(dirpath):
 138                     if local_follow:
 139                         if (dir_stats.st_dev, dir_stats.st_ino) in parent_dirs:
 140                             # Just insert the symlink if the target directory
 141                             # exists inside of the copy already
 142                             r_files['symlinks'].append({"src": os.readlink(dirpath), "dest": dest_dirpath})
 143                         else:
 144                             # Walk the dirpath to find all parent directories.
 145                             new_parents = set()
 146                             parent_dir_list = os.path.dirname(dirpath).split(os.path.sep)
 147                             for parent in range(len(parent_dir_list), 0, -1):
 148                                 parent_stat = os.stat(u'/'.join(parent_dir_list[:parent]))
 149                                 if (parent_stat.st_dev, parent_stat.st_ino) in parent_dirs:
 150                                     # Reached the point at which the directory
 151                                     # tree is already known.  Don't add any
 152                                     # more or we might go to an ancestor that
 153                                     # isn't being copied.
 154                                     break
 155                                 new_parents.add((parent_stat.st_dev, parent_stat.st_ino))
 156 
 157                             if (dir_stats.st_dev, dir_stats.st_ino) in new_parents:
 158                                 # This was a a circular symlink.  So add it as
 159                                 # a symlink
 160                                 r_files['symlinks'].append({"src": os.readlink(dirpath), "dest": dest_dirpath})
 161                             else:
 162                                 # Walk the directory pointed to by the symlink
 163                                 r_files['directories'].append({"src": real_dir, "dest": dest_dirpath})
 164                                 offset = len(real_dir) + 1
 165                                 _recurse(real_dir, offset, parent_dirs.union(new_parents),
 166                                          rel_base=dest_dirpath,
 167                                          checksum_check=checksum_check)
 168                     else:
 169                         # Add the symlink to the destination
 170                         r_files['symlinks'].append({"src": os.readlink(dirpath), "dest": dest_dirpath})
 171                 else:
 172                     # Just a normal directory
 173                     r_files['directories'].append({"src": dirpath, "dest": dest_dirpath})
 174 
 175     # Check if the source ends with a "/" so that we know which directory
 176     # level to work at (similar to rsync)
 177     source_trailing_slash = False
 178     if trailing_slash_detector:
 179         source_trailing_slash = trailing_slash_detector(topdir)
 180     else:
 181         source_trailing_slash = topdir.endswith(os.path.sep)
 182 
 183     # Calculate the offset needed to strip the base_path to make relative
 184     # paths
 185     if base_path is None:
 186         base_path = topdir
 187     if not source_trailing_slash:
 188         base_path = os.path.dirname(base_path)
 189     if topdir.startswith(base_path):
 190         offset = len(base_path)
 191 
 192     # Make sure we're making the new paths relative
 193     if trailing_slash_detector and not trailing_slash_detector(base_path):
 194         offset += 1
 195     elif not base_path.endswith(os.path.sep):
 196         offset += 1
 197 
 198     if os.path.islink(topdir) and not local_follow:
 199         r_files['symlinks'] = {"src": os.readlink(topdir), "dest": os.path.basename(topdir)}
 200         return r_files
 201 
 202     dir_stats = os.stat(topdir)
 203     parents = frozenset(((dir_stats.st_dev, dir_stats.st_ino),))
 204     # Actually walk the directory hierarchy
 205     _recurse(topdir, offset, parents, checksum_check=checksum_check)
 206 
 207     return r_files
 208

 210 def _get_local_checksum(get_checksum, local_path):
 211     if get_checksum:
 212         return checksum(local_path)
 213     else:
 214         return None

 221     def _create_content_tempfile(self, content):
 222         ''' Create a tempfile containing defined content '''
 223         fd, content_tempfile = tempfile.mkstemp(dir=C.DEFAULT_LOCAL_TMP)
 224         f = os.fdopen(fd, 'wb')
 225         content = to_bytes(content)
 226         try:
 227             f.write(content)
 228         except Exception as err:
 229             os.remove(content_tempfile)
 230             raise Exception(err)
 231         finally:
 232             f.close()
 233         return content_tempfile
 234 
 235     def _create_zip_tempfile(self, files, directories):
 236         tmpdir = tempfile.mkdtemp(dir=C.DEFAULT_LOCAL_TMP)
 237         zip_file_path = os.path.join(tmpdir, "win_copy.zip")
 238         zip_file = zipfile.ZipFile(zip_file_path, "w", zipfile.ZIP_STORED, True)
 239 
 240         # encoding the file/dir name with base64 so Windows can unzip a unicode
 241         # filename and get the right name, Windows doesn't handle unicode names
 242         # very well
 243         for directory in directories:
 244             directory_path = to_bytes(directory['src'], errors='surrogate_or_strict')
 245             archive_path = to_bytes(directory['dest'], errors='surrogate_or_strict')
 246 
 247             encoded_path = to_text(base64.b64encode(archive_path), errors='surrogate_or_strict')
 248             zip_file.write(directory_path, encoded_path, zipfile.ZIP_DEFLATED)
 249 
 250         for file in files:
 251             file_path = to_bytes(file['src'], errors='surrogate_or_strict')
 252             archive_path = to_bytes(file['dest'], errors='surrogate_or_strict')
 253 
 254             encoded_path = to_text(base64.b64encode(archive_path), errors='surrogate_or_strict')
 255             zip_file.write(file_path, encoded_path, zipfile.ZIP_DEFLATED)
 256 
 257         return zip_file_path
 258 
 259     def _remove_tempfile_if_content_defined(self, content, content_tempfile):
 260         if content is not None:
 261             os.remove(content_tempfile)

 263     def _copy_single_file(self, local_file, dest, source_rel, task_vars, tmp, backup):
 264         if self._play_context.check_mode:
 265             module_return = dict(changed=True)
 266             return module_return
 267 
 268         # copy the file across to the server
 269         tmp_src = self._connection._shell.join_path(tmp, 'source')
 270         self._transfer_file(local_file, tmp_src)
 271 
 272         copy_args = self._task.args.copy()
 273         copy_args.update(
 274             dict(
 275                 dest=dest,
 276                 src=tmp_src,
 277                 _original_basename=source_rel,
 278                 _copy_mode="single",
 279                 backup=backup,
 280             )
 281         )
 282         copy_args.pop('content', None)
 283 
 284         copy_result = self._execute_module(module_name="copy",
 285                                            module_args=copy_args,
 286                                            task_vars=task_vars)
 287 
 288         return copy_result

 290     def _copy_zip_file(self, dest, files, directories, task_vars, tmp, backup):
 291         # create local zip file containing all the files and directories that
 292         # need to be copied to the server
 293         if self._play_context.check_mode:
 294             module_return = dict(changed=True)
 295             return module_return
 296 
 297         try:
 298             zip_file = self._create_zip_tempfile(files, directories)
 299         except Exception as e:
 300             module_return = dict(
 301                 changed=False,
 302                 failed=True,
 303                 msg="failed to create tmp zip file: %s" % to_text(e),
 304                 exception=traceback.format_exc()
 305             )
 306             return module_return
 307 
 308         zip_path = self._loader.get_real_file(zip_file)
 309 
 310         # send zip file to remote, file must end in .zip so
 311         # Com Shell.Application works
 312         tmp_src = self._connection._shell.join_path(tmp, 'source.zip')
 313         self._transfer_file(zip_path, tmp_src)
 314 
 315         # run the explode operation of win_copy on remote
 316         copy_args = self._task.args.copy()
 317         copy_args.update(
 318             dict(
 319                 src=tmp_src,
 320                 dest=dest,
 321                 _copy_mode="explode",
 322                 backup=backup,
 323             )
 324         )
 325         copy_args.pop('content', None)
 326         module_return = self._execute_module(module_name='copy',
 327                                              module_args=copy_args,
 328                                              task_vars=task_vars)
 329         shutil.rmtree(os.path.dirname(zip_path))
 330         return module_return

 332     def run(self, tmp=None, task_vars=None):
 333         ''' handler for file transfer operations '''
 334         if task_vars is None:
 335             task_vars = dict()
 336 
 337         result = super(ActionModule, self).run(tmp, task_vars)
 338         del tmp  # tmp no longer has any effect
 339 
 340         source = self._task.args.get('src', None)
 341         content = self._task.args.get('content', None)
 342         dest = self._task.args.get('dest', None)
 343         remote_src = boolean(self._task.args.get('remote_src', False), strict=False)
 344         local_follow = boolean(self._task.args.get('local_follow', False), strict=False)
 345         force = boolean(self._task.args.get('force', True), strict=False)
 346         decrypt = boolean(self._task.args.get('decrypt', True), strict=False)
 347         backup = boolean(self._task.args.get('backup', False), strict=False)
 348 
 349         result['src'] = source
 350         result['dest'] = dest
 351 
 352         result['failed'] = True
 353         if (source is None and content is None) or dest is None:
 354             result['msg'] = "src (or content) and dest are required"
 355         elif source is not None and content is not None:
 356             result['msg'] = "src and content are mutually exclusive"
 357         elif content is not None and dest is not None and (
 358                 dest.endswith(os.path.sep) or dest.endswith(self.WIN_PATH_SEPARATOR)):
 359             result['msg'] = "dest must be a file if content is defined"
 360         else:
 361             del result['failed']
 362 
 363         if result.get('failed'):
 364             return result
 365 
 366         # If content is defined make a temp file and write the content into it
 367         content_tempfile = None
 368         if content is not None:
 369             try:
 370                 # if content comes to us as a dict it should be decoded json.
 371                 # We need to encode it back into a string and write it out
 372                 if isinstance(content, dict) or isinstance(content, list):
 373                     content_tempfile = self._create_content_tempfile(json.dumps(content))
 374                 else:
 375                     content_tempfile = self._create_content_tempfile(content)
 376                 source = content_tempfile
 377             except Exception as err:
 378                 result['failed'] = True
 379                 result['msg'] = "could not write content tmp file: %s" % to_native(err)
 380                 return result
 381         # all actions should occur on the remote server, run win_copy module
 382         elif remote_src:
 383             new_module_args = self._task.args.copy()
 384             new_module_args.update(
 385                 dict(
 386                     _copy_mode="remote",
 387                     dest=dest,
 388                     src=source,
 389                     force=force,
 390                     backup=backup,
 391                 )
 392             )
 393             new_module_args.pop('content', None)
 394             result.update(self._execute_module(module_args=new_module_args, task_vars=task_vars))
 395             return result
 396         # find_needle returns a path that may not have a trailing slash on a
 397         # directory so we need to find that out first and append at the end
 398         else:
 399             trailing_slash = source.endswith(os.path.sep)
 400             try:
 401                 # find in expected paths
 402                 source = self._find_needle('files', source)
 403             except AnsibleError as e:
 404                 result['failed'] = True
 405                 result['msg'] = to_text(e)
 406                 result['exception'] = traceback.format_exc()
 407                 return result
 408 
 409             if trailing_slash != source.endswith(os.path.sep):
 410                 if source[-1] == os.path.sep:
 411                     source = source[:-1]
 412                 else:
 413                     source = source + os.path.sep
 414 
 415         # A list of source file tuples (full_path, relative_path) which will try to copy to the destination
 416         source_files = {'files': [], 'directories': [], 'symlinks': []}
 417 
 418         # If source is a directory populate our list else source is a file and translate it to a tuple.
 419         if os.path.isdir(to_bytes(source, errors='surrogate_or_strict')):
 420             result['operation'] = 'folder_copy'
 421 
 422             # Get a list of the files we want to replicate on the remote side
 423             source_files = _walk_dirs(source, self._loader, decrypt=decrypt, local_follow=local_follow,
 424                                       trailing_slash_detector=self._connection._shell.path_has_trailing_slash,
 425                                       checksum_check=force)
 426 
 427             # If it's recursive copy, destination is always a dir,
 428             # explicitly mark it so (note - win_copy module relies on this).
 429             if not self._connection._shell.path_has_trailing_slash(dest):
 430                 dest = "%s%s" % (dest, self.WIN_PATH_SEPARATOR)
 431 
 432             check_dest = dest
 433         # Source is a file, add details to source_files dict
 434         else:
 435             result['operation'] = 'file_copy'
 436 
 437             # If the local file does not exist, get_real_file() raises AnsibleFileNotFound
 438             try:
 439                 source_full = self._loader.get_real_file(source, decrypt=decrypt)
 440             except AnsibleFileNotFound as e:
 441                 result['failed'] = True
 442                 result['msg'] = "could not find src=%s, %s" % (source_full, to_text(e))
 443                 return result
 444 
 445             original_basename = os.path.basename(source)
 446             result['original_basename'] = original_basename
 447 
 448             # check if dest ends with / or \ and append source filename to dest
 449             if self._connection._shell.path_has_trailing_slash(dest):
 450                 check_dest = dest
 451                 filename = original_basename
 452                 result['dest'] = self._connection._shell.join_path(dest, filename)
 453             else:
 454                 # replace \\ with / so we can use os.path to get the filename or dirname
 455                 unix_path = dest.replace(self.WIN_PATH_SEPARATOR, os.path.sep)
 456                 filename = os.path.basename(unix_path)
 457                 check_dest = os.path.dirname(unix_path)
 458 
 459             file_checksum = _get_local_checksum(force, source_full)
 460             source_files['files'].append(
 461                 dict(
 462                     src=source_full,
 463                     dest=filename,
 464                     checksum=file_checksum
 465                 )
 466             )
 467             result['checksum'] = file_checksum
 468             result['size'] = os.path.getsize(to_bytes(source_full, errors='surrogate_or_strict'))
 469 
 470         # find out the files/directories/symlinks that we need to copy to the server
 471         query_args = self._task.args.copy()
 472         query_args.update(
 473             dict(
 474                 _copy_mode="query",
 475                 dest=check_dest,
 476                 force=force,
 477                 files=source_files['files'],
 478                 directories=source_files['directories'],
 479                 symlinks=source_files['symlinks'],
 480             )
 481         )
 482         # src is not required for query, will fail path validation is src has unix allowed chars
 483         query_args.pop('src', None)
 484 
 485         query_args.pop('content', None)
 486         query_return = self._execute_module(module_args=query_args,
 487                                             task_vars=task_vars)
 488 
 489         if query_return.get('failed') is True:
 490             result.update(query_return)
 491             return result
 492 
 493         if len(query_return['files']) > 0 or len(query_return['directories']) > 0 and self._connection._shell.tmpdir is None:
 494             self._connection._shell.tmpdir = self._make_tmp_path()
 495 
 496         if len(query_return['files']) == 1 and len(query_return['directories']) == 0:
 497             # we only need to copy 1 file, don't mess around with zips
 498             file_src = query_return['files'][0]['src']
 499             file_dest = query_return['files'][0]['dest']
 500             result.update(self._copy_single_file(file_src, dest, file_dest,
 501                                                  task_vars, self._connection._shell.tmpdir, backup))
 502             if result.get('failed') is True:
 503                 result['msg'] = "failed to copy file %s: %s" % (file_src, result['msg'])
 504             result['changed'] = True
 505 
 506         elif len(query_return['files']) > 0 or len(query_return['directories']) > 0:
 507             # either multiple files or directories need to be copied, compress
 508             # to a zip and 'explode' the zip on the server
 509             # TODO: handle symlinks
 510             result.update(self._copy_zip_file(dest, source_files['files'],
 511                                               source_files['directories'],
 512                                               task_vars, self._connection._shell.tmpdir, backup))
 513             result['changed'] = True
 514         else:
 515             # no operations need to occur
 516             result['failed'] = False
 517             result['changed'] = False
 518 
 519         # remove the content tmp file and remote tmp file if it was created
 520         self._remove_tempfile_if_content_defined(content, content_tempfile)
 521         self._remove_tmp_path(self._connection._shell.tmpdir)
 522         return result
