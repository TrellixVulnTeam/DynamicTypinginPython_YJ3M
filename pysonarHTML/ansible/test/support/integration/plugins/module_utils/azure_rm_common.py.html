 277 def azure_id_to_dict(id):
 278     pieces = re.sub(r'^\/', '', id).split('/')
 279     result = {}
 280     index = 0
 281     while index < len(pieces) - 1:
 282         result[pieces[index]] = pieces[index + 1]
 283         index += 1
 284     return result

 461     def validate_tags(self, tags):
 462         '''
 463         Check if tags dictionary contains string:string pairs.
 464

 466         :return: None
 467         '''
 468         if not self.facts_module:

 475     def update_tags(self, tags):
 476         '''
 477         Call from the module to update metadata tags. Returns tuple
 478         with bool indicating if there was a change and dict of new
 479         tags to assign to the object.
 480 
 481         :param tags: metadata tags from the object
 482         :return: bool, dict
 483         '''
 484         tags = tags or dict()
 485         new_tags = copy.copy(tags) if isinstance(tags, dict) else dict()
 486         param_tags = self.module.params.get('tags') if isinstance(self.module.params.get('tags'), dict) else dict()
 487         append_tags = self.module.params.get('append_tags') if self.module.params.get('append_tags') is not None else True
 488         changed = False
 489         # check add or update
 490         for key, value in param_tags.items():
 491             if not new_tags.get(key) or new_tags[key] != value:
 492                 changed = True
 493                 new_tags[key] = value
 494         # check remove
 495         if not append_tags:
 496             for key, value in tags.items():
 497                 if not param_tags.get(key):
 498                     new_tags.pop(key)
 499                     changed = True
 500         return changed, new_tags

 502     def has_tags(self, obj_tags, tag_list):
 503         '''
 504         Used in fact modules to compare object tags to list of parameter tags. Return true if list of parameter tags
 505         exists in object tags.
 506 
 507         :param obj_tags: dictionary of tags from an Azure object.
 508         :param tag_list: list of tag keys or tag key:value pairs
 509         :return: bool
 510         '''
 511 
 512         if not obj_tags and tag_list:
 513             return False
 514 
 515         if not tag_list:
 516             return True
 517 
 518         matches = 0
 519         result = False
 520         for tag in tag_list:
 521             tag_key = tag
 522             tag_value = None
 523             if ':' in tag:
 524                 tag_key, tag_value = tag.split(':')
 525             if tag_value and obj_tags.get(tag_key) == tag_value:
 526                 matches += 1

 559         '''
 560         Return a JSON representation of an Azure object.

 570         if enum_modules:
 571             for module_name in enum_modules:

 576             self.log(str(dependencies))

 619                     azure_object.name, azure_object.properties.provisioning_state, AZURE_SUCCESS_STATE))

 819     def get_mgmt_svc_client(self, client_type, base_url=None, api_version=None):
 820         self.log('Getting management service client {0}'.format(client_type.__name__))
 821         self.check_client_version(client_type)
 822 
 823         client_argspec = inspect.getargspec(client_type.__init__)
 824 
 825         if not base_url:
 826             # most things are resource_manager, don't make everyone specify
 827             base_url = self.azure_auth._cloud_environment.endpoints.resource_manager
 828 
 829         client_kwargs = dict(credentials=self.azure_auth.azure_credentials, subscription_id=self.azure_auth.subscription_id, base_url=base_url)
 830 
 831         api_profile_dict = {}
 832 
 833         if self.api_profile:
 834             api_profile_dict = self.get_api_profile(client_type.__name__, self.api_profile)
 835 
 836         # unversioned clients won't accept profile; only send it if necessary
 837         # clients without a version specified in the profile will use the default
 838         if api_profile_dict and 'profile' in client_argspec.args:
 839             client_kwargs['profile'] = api_profile_dict
 840 
 841         # If the client doesn't accept api_version, it's unversioned.
 842         # If it does, favor explicitly-specified api_version, fall back to api_profile
 843         if 'api_version' in client_argspec.args:
 844             profile_default_version = api_profile_dict.get('default_api_version', None)
 845             if api_version or profile_default_version:
 846                 client_kwargs['api_version'] = api_version or profile_default_version
 847                 if 'profile' in client_kwargs:
 848                     # remove profile; only pass API version if specified
 849                     client_kwargs.pop('profile')
 850 
 851         client = client_type(**client_kwargs)
 852 
 853         # FUTURE: remove this once everything exposes models directly (eg, containerinstance)
 854         try:
 855             getattr(client, "models")
 856         except AttributeError:
 857             def _ansible_get_models(self, *arg, **kwarg):
 858                 return self._ansible_models
 859 
 860             setattr(client, '_ansible_models', importlib.import_module(client_type.__module__).models)
 861             client.models = types.MethodType(_ansible_get_models, client)
 862 
 863         client.config = self.add_user_agent(client.config)
 864 
 865         if self.azure_auth._cert_validation_mode == 'ignore':
 866             client.config.session_configuration_callback = self._validation_ignore_callback
 867 
 868         return client

 881     def generate_sas_token(self, **kwags):
 882         base_url = kwags.get('base_url', None)
 883         expiry = kwags.get('expiry', time() + 3600)
 884         key = kwags.get('key', None)
 885         policy = kwags.get('policy', None)
 886         url = quote_plus(base_url)
 887         ttl = int(expiry)
 888         sign_key = '{0}\n{1}'.format(url, ttl)
 889         signature = b64encode(HMAC(b64decode(key), sign_key.encode('utf-8'), sha256).digest())
 890         result = {
 891             'sr': url,
 892             'sig': signature,
 893             'se': str(ttl),
 894         }
 895         if policy:
 896             result['skn'] = policy
 897         return 'SharedAccessSignature ' + urlencode(result)

1426             return credentials
