  81 def get_connector(*args, **kwargs):
  82     ''' Returns an instance of infoblox_client.connector.Connector
  83     :params args: positional arguments are silently ignored
  84     :params kwargs: dict that is passed to Connector init
  85     :returns: Connector
  86     '''
  87     if not HAS_INFOBLOX_CLIENT:
  88         raise Exception('infoblox-client is required but does not appear '
  89                         'to be installed.  It can be installed using the '
  90                         'command `pip install infoblox-client`')
  91 
  92     if not set(kwargs.keys()).issubset(list(NIOS_PROVIDER_SPEC.keys()) + ['ssl_verify']):
  93         raise Exception('invalid or unsupported keyword argument for connector')
  94     for key, value in iteritems(NIOS_PROVIDER_SPEC):
  95         if key not in kwargs:
  96             # apply default values from NIOS_PROVIDER_SPEC since we cannot just
  97             # assume the provider values are coming from AnsibleModule
  98             if 'default' in value:
  99                 kwargs[key] = value['default']
 100 
 101             # override any values with env variables unless they were
 102             # explicitly set
 103             env = ('INFOBLOX_%s' % key).upper()
 104             if env in os.environ:
 105                 kwargs[key] = os.environ.get(env)
 106 
 107     if 'validate_certs' in kwargs.keys():
 108         kwargs['ssl_verify'] = kwargs['validate_certs']
 109         kwargs.pop('validate_certs', None)
 110 
 111     return Connector(kwargs)

 114 def normalize_extattrs(value):
 115     ''' Normalize extattrs field to expected format
 116     The module accepts extattrs as key/value pairs.  This method will
 117     transform the key/value pairs into a structure suitable for
 118     sending across WAPI in the format of:
 119         extattrs: {
 120             key: {
 121                 value: <value>
 122             }
 123         }
 124     '''
 125     return dict([(k, {'value': v}) for k, v in iteritems(value)])
 126 
 127 
 128 def flatten_extattrs(value):
 129     ''' Flatten the key/value struct for extattrs
 130     WAPI returns extattrs field as a dict in form of:
 131         extattrs: {
 132             key: {
 133                 value: <value>
 134             }
 135         }
 136     This method will flatten the structure to:
 137         extattrs: {
 138             key: value
 139         }
 140     '''
 141     return dict([(k, v['value']) for k, v in iteritems(value)])

 144 def member_normalize(member_spec):
 145     ''' Transforms the member module arguments into a valid WAPI struct
 146     This function will transform the arguments into a structure that
 147     is a valid WAPI structure in the format of:
 148         {
 149             key: <value>,
 150         }
 151     It will remove any arguments that are set to None since WAPI will error on
 152     that condition.
 153     The remainder of the value validation is performed by WAPI
 154     Some parameters in ib_spec are passed as a list in order to pass the validation for elements.
 155     In this function, they are converted to dictionary.
 156     '''
 157     member_elements = ['vip_setting', 'ipv6_setting', 'lan2_port_setting', 'mgmt_port_setting',
 158                        'pre_provisioning', 'network_setting', 'v6_network_setting',
 159                        'ha_port_setting', 'lan_port_setting', 'lan2_physical_setting',
 160                        'lan_ha_port_setting', 'mgmt_network_setting', 'v6_mgmt_network_setting']
 161     for key in member_spec.keys():
 162         if key in member_elements and member_spec[key] is not None:
 163             member_spec[key] = member_spec[key][0]
 164         if isinstance(member_spec[key], dict):
 165             member_spec[key] = member_normalize(member_spec[key])
 166         elif isinstance(member_spec[key], list):
 167             for x in member_spec[key]:
 168                 if isinstance(x, dict):
 169                     x = member_normalize(x)
 170         elif member_spec[key] is None:
 171             del member_spec[key]
 172     return member_spec

 175 class WapiBase(object):
 176     ''' Base class for implementing Infoblox WAPI API '''
 177     provider_spec = {'provider': dict(type='dict', options=NIOS_PROVIDER_SPEC)}
 178 
 179     def __init__(self, provider):
 180         self.connector = get_connector(**provider)
 181 
 182     def __getattr__(self, name):
 183         try:
 184             return self.__dict__[name]
 185         except KeyError:
 186             if name.startswith('_'):
 187                 raise AttributeError("'%s' object has no attribute '%s'" % (self.__class__.__name__, name))
 188             return partial(self._invoke_method, name)
 189 
 190     def _invoke_method(self, name, *args, **kwargs):
 191         try:
 192             method = getattr(self.connector, name)
 193             return method(*args, **kwargs)
 194         except InfobloxException as exc:
 195             if hasattr(self, 'handle_exception'):
 196                 self.handle_exception(name, exc)
 197             else:
 198                 raise

 215 class WapiModule(WapiBase):
 216     ''' Implements WapiBase for executing a NIOS module '''
 217     def __init__(self, module):
 218         self.module = module
 219         provider = module.params['provider']
 220         try:
 221             super(WapiModule, self).__init__(provider)
 222         except Exception as exc:
 223             self.module.fail_json(msg=to_text(exc))

 225     def handle_exception(self, method_name, exc):
 226         ''' Handles any exceptions raised
 227         This method will be called if an InfobloxException is raised for
 228         any call to the instance of Connector and also, in case of generic
 229         exception. This method will then gracefully fail the module.
 230         :args exc: instance of InfobloxException
 231         '''
 232         if ('text' in exc.response):
 233             self.module.fail_json(
 234                 msg=exc.response['text'],
 235                 type=exc.response['Error'].split(':')[0],
 236                 code=exc.response.get('code'),
 237                 operation=method_name
 238             )
 239         else:
 240             self.module.fail_json(msg=to_native(exc))

 242     def run(self, ib_obj_type, ib_spec):
 243         ''' Runs the module and performans configuration tasks
 244         :args ib_obj_type: the WAPI object type to operate against
 245         :args ib_spec: the specification for the WAPI object as a dict
 246         :returns: a results dict
 247         '''
 248 
 249         update = new_name = None
 250         state = self.module.params['state']
 251         if state not in ('present', 'absent'):
 252             self.module.fail_json(msg='state must be one of `present`, `absent`, got `%s`' % state)
 253 
 254         result = {'changed': False}
 255 
 256         obj_filter = dict([(k, self.module.params[k]) for k, v in iteritems(ib_spec) if v.get('ib_req')])
 257 
 258         # get object reference
 259         ib_obj_ref, update, new_name = self.get_object_ref(self.module, ib_obj_type, obj_filter, ib_spec)
 260         proposed_object = {}
 261         for key, value in iteritems(ib_spec):
 262             if self.module.params[key] is not None:
 263                 if 'transform' in value:
 264                     proposed_object[key] = value['transform'](self.module)
 265                 else:
 266                     proposed_object[key] = self.module.params[key]
 267 
 268         # If configure_by_dns is set to False, then delete the default dns set in the param else throw exception
 269         if not proposed_object.get('configure_for_dns') and proposed_object.get('view') == 'default'\
 270                 and ib_obj_type == NIOS_HOST_RECORD:
 271             del proposed_object['view']
 272         elif not proposed_object.get('configure_for_dns') and proposed_object.get('view') != 'default'\
 273                 and ib_obj_type == NIOS_HOST_RECORD:
 274             self.module.fail_json(msg='DNS Bypass is not allowed if DNS view is set other than \'default\'')
 275 
 276         if ib_obj_ref:
 277             if len(ib_obj_ref) > 1:
 278                 for each in ib_obj_ref:
 279                     # To check for existing A_record with same name with input A_record by IP
 280                     if each.get('ipv4addr') and each.get('ipv4addr') == proposed_object.get('ipv4addr'):
 281                         current_object = each
 282                     # To check for existing Host_record with same name with input Host_record by IP
 283                     elif each.get('ipv4addrs')[0].get('ipv4addr') and each.get('ipv4addrs')[0].get('ipv4addr')\
 284                             == proposed_object.get('ipv4addrs')[0].get('ipv4addr'):
 285                         current_object = each
 286                     # Else set the current_object with input value
 287                     else:
 288                         current_object = obj_filter
 289                         ref = None
 290             else:
 291                 current_object = ib_obj_ref[0]
 292             if 'extattrs' in current_object:
 293                 current_object['extattrs'] = flatten_extattrs(current_object['extattrs'])
 294             if current_object.get('_ref'):
 295                 ref = current_object.pop('_ref')
 296         else:
 297             current_object = obj_filter
 298             ref = None
 299         # checks if the object type is member to normalize the attributes being passed
 300         if (ib_obj_type == NIOS_MEMBER):
 301             proposed_object = member_normalize(proposed_object)
 302 
 303         # checks if the name's field has been updated
 304         if update and new_name:
 305             proposed_object['name'] = new_name
 306 
 307         check_remove = []
 308         if (ib_obj_type == NIOS_HOST_RECORD):
 309             # this check is for idempotency, as if the same ip address shall be passed
 310             # add param will be removed, and same exists true for remove case as well.
 311             if 'ipv4addrs' in [current_object and proposed_object]:
 312                 for each in current_object['ipv4addrs']:
 313                     if each['ipv4addr'] == proposed_object['ipv4addrs'][0]['ipv4addr']:
 314                         if 'add' in proposed_object['ipv4addrs'][0]:
 315                             del proposed_object['ipv4addrs'][0]['add']
 316                             break
 317                     check_remove += each.values()
 318                 if proposed_object['ipv4addrs'][0]['ipv4addr'] not in check_remove:
 319                     if 'remove' in proposed_object['ipv4addrs'][0]:
 320                         del proposed_object['ipv4addrs'][0]['remove']
 321 
 322         res = None
 323         modified = not self.compare_objects(current_object, proposed_object)
 324         if 'extattrs' in proposed_object:
 325             proposed_object['extattrs'] = normalize_extattrs(proposed_object['extattrs'])
 326 
 327         # Checks if nios_next_ip param is passed in ipv4addrs/ipv4addr args
 328         proposed_object = self.check_if_nios_next_ip_exists(proposed_object)
 329 
 330         if state == 'present':
 331             if ref is None:
 332                 if not self.module.check_mode:
 333                     self.create_object(ib_obj_type, proposed_object)
 334                 result['changed'] = True
 335             # Check if NIOS_MEMBER and the flag to call function create_token is set
 336             elif (ib_obj_type == NIOS_MEMBER) and (proposed_object['create_token']):
 337                 proposed_object = None
 338                 # the function creates a token that can be used by a pre-provisioned member to join the grid
 339                 result['api_results'] = self.call_func('create_token', ref, proposed_object)
 340                 result['changed'] = True
 341             elif modified:
 342                 if 'ipv4addrs' in proposed_object:
 343                     if ('add' not in proposed_object['ipv4addrs'][0]) and ('remove' not in proposed_object['ipv4addrs'][0]):
 344                         self.check_if_recordname_exists(obj_filter, ib_obj_ref, ib_obj_type, current_object, proposed_object)
 345 
 346                 if (ib_obj_type in (NIOS_HOST_RECORD, NIOS_NETWORK_VIEW, NIOS_DNS_VIEW)):
 347                     run_update = True
 348                     proposed_object = self.on_update(proposed_object, ib_spec)
 349                     if 'ipv4addrs' in proposed_object:
 350                         if ('add' or 'remove') in proposed_object['ipv4addrs'][0]:
 351                             run_update, proposed_object = self.check_if_add_remove_ip_arg_exists(proposed_object)
 352                             if run_update:
 353                                 res = self.update_object(ref, proposed_object)
 354                                 result['changed'] = True
 355                             else:
 356                                 res = ref
 357                 if (ib_obj_type in (NIOS_A_RECORD, NIOS_AAAA_RECORD, NIOS_PTR_RECORD, NIOS_SRV_RECORD)):
 358                     # popping 'view' key as update of 'view' is not supported with respect to a:record/aaaa:record/srv:record/ptr:record
 359                     proposed_object = self.on_update(proposed_object, ib_spec)
 360                     del proposed_object['view']
 361                     if not self.module.check_mode:
 362                         res = self.update_object(ref, proposed_object)
 363                     result['changed'] = True
 364                 elif 'network_view' in proposed_object:
 365                     proposed_object.pop('network_view')
 366                     result['changed'] = True
 367                 if not self.module.check_mode and res is None:
 368                     proposed_object = self.on_update(proposed_object, ib_spec)
 369                     self.update_object(ref, proposed_object)
 370                     result['changed'] = True
 371 
 372         elif state == 'absent':
 373             if ref is not None:
 374                 if 'ipv4addrs' in proposed_object:
 375                     if 'remove' in proposed_object['ipv4addrs'][0]:
 376                         self.check_if_add_remove_ip_arg_exists(proposed_object)
 377                         self.update_object(ref, proposed_object)
 378                         result['changed'] = True
 379                 elif not self.module.check_mode:
 380                     self.delete_object(ref)
 381                     result['changed'] = True
 382 
 383         return result

 385     def check_if_recordname_exists(self, obj_filter, ib_obj_ref, ib_obj_type, current_object, proposed_object):
 386         ''' Send POST request if host record input name and retrieved ref name is same,
 387             but input IP and retrieved IP is different'''
 388 
 389         if 'name' in (obj_filter and ib_obj_ref[0]) and ib_obj_type == NIOS_HOST_RECORD:
 390             obj_host_name = obj_filter['name']
 391             ref_host_name = ib_obj_ref[0]['name']
 392             if 'ipv4addrs' in (current_object and proposed_object):
 393                 current_ip_addr = current_object['ipv4addrs'][0]['ipv4addr']
 394                 proposed_ip_addr = proposed_object['ipv4addrs'][0]['ipv4addr']
 395             elif 'ipv6addrs' in (current_object and proposed_object):
 396                 current_ip_addr = current_object['ipv6addrs'][0]['ipv6addr']
 397                 proposed_ip_addr = proposed_object['ipv6addrs'][0]['ipv6addr']
 398 
 399             if obj_host_name == ref_host_name and current_ip_addr != proposed_ip_addr:
 400                 self.create_object(ib_obj_type, proposed_object)

 402     def check_if_nios_next_ip_exists(self, proposed_object):
 403         ''' Check if nios_next_ip argument is passed in ipaddr while creating
 404             host record, if yes then format proposed object ipv4addrs and pass
 405             func:nextavailableip and ipaddr range to create hostrecord with next
 406              available ip in one call to avoid any race condition '''
 407 
 408         if 'ipv4addrs' in proposed_object:
 409             if 'nios_next_ip' in proposed_object['ipv4addrs'][0]['ipv4addr']:
 410                 ip_range = self.module._check_type_dict(proposed_object['ipv4addrs'][0]['ipv4addr'])['nios_next_ip']
 411                 proposed_object['ipv4addrs'][0]['ipv4addr'] = NIOS_NEXT_AVAILABLE_IP + ':' + ip_range
 412         elif 'ipv4addr' in proposed_object:
 413             if 'nios_next_ip' in proposed_object['ipv4addr']:
 414                 ip_range = self.module._check_type_dict(proposed_object['ipv4addr'])['nios_next_ip']
 415                 proposed_object['ipv4addr'] = NIOS_NEXT_AVAILABLE_IP + ':' + ip_range
 416 
 417         return proposed_object

 419     def check_if_add_remove_ip_arg_exists(self, proposed_object):
 420         '''
 421             This function shall check if add/remove param is set to true and
 422             is passed in the args, then we will update the proposed dictionary
 423             to add/remove IP to existing host_record, if the user passes false
 424             param with the argument nothing shall be done.
 425             :returns: True if param is changed based on add/remove, and also the
 426             changed proposed_object.
 427         '''
 428         update = False
 429         if 'add' in proposed_object['ipv4addrs'][0]:
 430             if proposed_object['ipv4addrs'][0]['add']:
 431                 proposed_object['ipv4addrs+'] = proposed_object['ipv4addrs']
 432                 del proposed_object['ipv4addrs']
 433                 del proposed_object['ipv4addrs+'][0]['add']
 434                 update = True
 435             else:
 436                 del proposed_object['ipv4addrs'][0]['add']
 437         elif 'remove' in proposed_object['ipv4addrs'][0]:
 438             if proposed_object['ipv4addrs'][0]['remove']:
 439                 proposed_object['ipv4addrs-'] = proposed_object['ipv4addrs']
 440                 del proposed_object['ipv4addrs']
 441                 del proposed_object['ipv4addrs-'][0]['remove']
 442                 update = True
 443             else:
 444                 del proposed_object['ipv4addrs'][0]['remove']
 445         return update, proposed_object

 447     def issubset(self, item, objects):
 448         ''' Checks if item is a subset of objects
 449         :args item: the subset item to validate
 450         :args objects: superset list of objects to validate against
 451         :returns: True if item is a subset of one entry in objects otherwise
 452             this method will return None
 453         '''
 454         for obj in objects:
 455             if isinstance(item, dict):
 456                 if all(entry in obj.items() for entry in item.items()):
 457                     return True
 458             else:
 459                 if item in obj:
 460                     return True
 461 
 462     def compare_objects(self, current_object, proposed_object):
 463         for key, proposed_item in iteritems(proposed_object):
 464             current_item = current_object.get(key)
 465 
 466             # if proposed has a key that current doesn't then the objects are
 467             # not equal and False will be immediately returned
 468             if current_item is None:
 469                 return False
 470 
 471             elif isinstance(proposed_item, list):
 472                 for subitem in proposed_item:
 473                     if not self.issubset(subitem, current_item):
 474                         return False
 475 
 476             elif isinstance(proposed_item, dict):
 477                 return self.compare_objects(current_item, proposed_item)
 478 
 479             else:
 480                 if current_item != proposed_item:
 481                     return False
 482 
 483         return True

 485     def get_object_ref(self, module, ib_obj_type, obj_filter, ib_spec):
 486         ''' this function gets the reference object of pre-existing nios objects '''
 487 
 488         update = False
 489         old_name = new_name = None
 490         if ('name' in obj_filter):
 491             # gets and returns the current object based on name/old_name passed
 492             try:
 493                 name_obj = self.module._check_type_dict(obj_filter['name'])
 494                 old_name = name_obj['old_name']
 495                 new_name = name_obj['new_name']
 496             except TypeError:
 497                 name = obj_filter['name']
 498 
 499             if old_name and new_name:
 500                 if (ib_obj_type == NIOS_HOST_RECORD):
 501                     test_obj_filter = dict([('name', old_name), ('view', obj_filter['view'])])
 502                 elif (ib_obj_type in (NIOS_AAAA_RECORD, NIOS_A_RECORD)):
 503                     test_obj_filter = obj_filter
 504                 else:
 505                     test_obj_filter = dict([('name', old_name)])
 506                 # get the object reference
 507                 ib_obj = self.get_object(ib_obj_type, test_obj_filter, return_fields=ib_spec.keys())
 508                 if ib_obj:
 509                     obj_filter['name'] = new_name
 510                 else:
 511                     test_obj_filter['name'] = new_name

 587     def on_update(self, proposed_object, ib_spec):
 588         ''' Event called before the update is sent to the API endpoing
 589         This method will allow the final proposed object to be changed
 590         and/or keys filtered before it is sent to the API endpoint to
 591         be processed.
 592         :args proposed_object: A dict item that will be encoded and sent
 593             the API endpoint with the updated data structure
 594         :returns: updated object to be sent to API endpoint
 595         '''
 596         keys = set()
 597         for key, value in iteritems(proposed_object):
 598             update = ib_spec[key].get('update', True)
 599             if not update:
 600                 keys.add(key)
 601         return dict([(k, v) for k, v in iteritems(proposed_object) if k not in keys])
