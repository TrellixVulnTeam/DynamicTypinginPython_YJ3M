  10 from ansible.module_utils.ansible_release import __version__
  11 from ansible.module_utils.basic import env_fallback, missing_required_lib
  12 
  13 try:
  14     import hcloud
  15 
  16     HAS_HCLOUD = True
  17 except ImportError:
  18     HAS_HCLOUD = False
  19 
  20 
  21 class Hcloud(object):
  22     def __init__(self, module, represent):
  23         self.module = module
  24         self.represent = represent
  25         self.result = {"changed": False, self.represent: None}
  26         if not HAS_HCLOUD:
  27             module.fail_json(msg=missing_required_lib("hcloud-python"))
  28         self._build_client()
  29 
  30     def _build_client(self):
  31         self.client = hcloud.Client(
  32             token=self.module.params["api_token"],
  33             api_endpoint=self.module.params["endpoint"],
  34             application_name="ansible-module",
  35             application_version=__version__,
  36         )
  37 
  38     def _mark_as_changed(self):
  39         self.result["changed"] = True
  40 
  41     @staticmethod
  42     def base_module_arguments():
  43         return {
  44             "api_token": {
  45                 "type": "str",
  46                 "required": True,
  47                 "fallback": (env_fallback, ["HCLOUD_TOKEN"]),
  48                 "no_log": True,
  49             },
  50             "endpoint": {"type": "str", "default": "https://api.hetzner.cloud/v1"},
  51         }
  52 
  53     def _prepare_result(self):
  54         """Prepare the result for every module
  55 
  56         :return: dict
  57         """
  58         return {}
  59 
  60     def get_result(self):
  61         if getattr(self, self.represent) is not None:
  62             self.result[self.represent] = self._prepare_result()
  63         return self.result
