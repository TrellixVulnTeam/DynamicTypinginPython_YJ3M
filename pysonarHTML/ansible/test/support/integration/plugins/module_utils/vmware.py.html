  51 def wait_for_task(task, max_backoff=64, timeout=3600):
  52     """Wait for given task using exponential back-off algorithm.
  53 
  54     Args:
  55         task: VMware task object
  56         max_backoff: Maximum amount of sleep time in seconds
  57         timeout: Timeout for the given task in seconds
  58 
  59     Returns: Tuple with True and result for successful task
  60     Raises: TaskError on failure
  61     """
  62     failure_counter = 0
  63     start_time = time.time()
  64 
  65     while True:
  66         if time.time() - start_time >= timeout:
  67             raise TaskError("Timeout")
  68         if task.info.state == vim.TaskInfo.State.success:
  69             return True, task.info.result
  70         if task.info.state == vim.TaskInfo.State.error:
  71             error_msg = task.info.error
  72             host_thumbprint = None
  73             try:
  74                 error_msg = error_msg.msg
  75                 if hasattr(task.info.error, 'thumbprint'):
  76                     host_thumbprint = task.info.error.thumbprint
  77             except AttributeError:
  78                 pass
  79             finally:
  80                 raise_from(TaskError(error_msg, host_thumbprint), task.info.error)
  81         if task.info.state in [vim.TaskInfo.State.running, vim.TaskInfo.State.queued]:
  82             sleep_time = min(2 ** failure_counter + randint(1, 1000) / 1000, max_backoff)
  83             time.sleep(sleep_time)
  84             failure_counter += 1

  87 def wait_for_vm_ip(content, vm, timeout=300):
  88     facts = dict()
  89     interval = 15
  90     while timeout > 0:
  91         _facts = gather_vm_facts(content, vm)
  92         if _facts['ipv4'] or _facts['ipv6']:

  96         timeout -= interval
  97 
  98     return facts

 197 def find_vm_by_id(content, vm_id, vm_id_type="vm_name", datacenter=None,
 198                   cluster=None, folder=None, match_first=False):
 199     """ UUID is unique to a VM, every other id returns the first match. """
 200     si = content.searchIndex
 201     vm = None
 202 
 203     if vm_id_type == 'dns_name':
 204         vm = si.FindByDnsName(datacenter=datacenter, dnsName=vm_id, vmSearch=True)
 205     elif vm_id_type == 'uuid':
 206         # Search By BIOS UUID rather than instance UUID
 207         vm = si.FindByUuid(datacenter=datacenter, instanceUuid=False, uuid=vm_id, vmSearch=True)
 208     elif vm_id_type == 'instance_uuid':
 209         vm = si.FindByUuid(datacenter=datacenter, instanceUuid=True, uuid=vm_id, vmSearch=True)
 210     elif vm_id_type == 'ip':
 211         vm = si.FindByIp(datacenter=datacenter, ip=vm_id, vmSearch=True)
 212     elif vm_id_type == 'vm_name':
 213         folder = None
 214         if cluster:
 215             folder = cluster
 216         elif datacenter:
 217             folder = datacenter.hostFolder
 218         vm = find_vm_by_name(content, vm_id, folder)
 219     elif vm_id_type == 'inventory_path':
 220         searchpath = folder
 221         # get all objects for this path
 222         f_obj = si.FindByInventoryPath(searchpath)
 223         if f_obj:
 224             if isinstance(f_obj, vim.Datacenter):
 225                 f_obj = f_obj.vmFolder
 226             for c_obj in f_obj.childEntity:
 227                 if not isinstance(c_obj, vim.VirtualMachine):
 228                     continue
 229                 if c_obj.name == vm_id:
 230                     vm = c_obj
 231                     if match_first:
 232                         break
 233     return vm

 248 def compile_folder_path_for_object(vobj):
 249     """ make a /vm/foo/bar/baz like folder path for an object """
 250 
 251     paths = []
 252     if isinstance(vobj, vim.Folder):
 253         paths.append(vobj.name)
 254 
 255     thisobj = vobj
 256     while hasattr(thisobj, 'parent'):
 257         thisobj = thisobj.parent
 258         try:
 259             moid = thisobj._moId
 260         except AttributeError:
 261             moid = None
 262         if moid in ['group-d1', 'ha-folder-root']:
 263             break
 264         if isinstance(thisobj, vim.Folder):
 265             paths.append(thisobj.name)
 266     paths.reverse()
 267     return '/' + '/'.join(paths)

 281 def gather_vm_facts(content, vm):
 282     """ Gather facts from vim.VirtualMachine object. """
 283     facts = {
 284         'module_hw': True,
 285         'hw_name': vm.config.name,
 286         'hw_power_status': vm.summary.runtime.powerState,
 287         'hw_guest_full_name': vm.summary.guest.guestFullName,
 288         'hw_guest_id': vm.summary.guest.guestId,
 289         'hw_product_uuid': vm.config.uuid,
 290         'hw_processor_count': vm.config.hardware.numCPU,
 291         'hw_cores_per_socket': vm.config.hardware.numCoresPerSocket,
 292         'hw_memtotal_mb': vm.config.hardware.memoryMB,
 293         'hw_interfaces': [],
 294         'hw_datastores': [],
 295         'hw_files': [],
 296         'hw_esxi_host': None,
 297         'hw_guest_ha_state': None,
 298         'hw_is_template': vm.config.template,
 299         'hw_folder': None,
 300         'hw_version': vm.config.version,
 301         'instance_uuid': vm.config.instanceUuid,
 302         'guest_tools_status': _get_vm_prop(vm, ('guest', 'toolsRunningStatus')),
 303         'guest_tools_version': _get_vm_prop(vm, ('guest', 'toolsVersion')),
 304         'guest_question': vm.summary.runtime.question,
 305         'guest_consolidation_needed': vm.summary.runtime.consolidationNeeded,
 306         'ipv4': None,
 307         'ipv6': None,
 308         'annotation': vm.config.annotation,
 309         'customvalues': {},
 310         'snapshots': [],
 311         'current_snapshot': None,
 312         'vnc': {},
 313         'moid': vm._moId,
 314         'vimref': "vim.VirtualMachine:%s" % vm._moId,
 315     }
 316 
 317     # facts that may or may not exist
 318     if vm.summary.runtime.host:
 319         try:
 320             host = vm.summary.runtime.host
 321             facts['hw_esxi_host'] = host.summary.config.name
 322             facts['hw_cluster'] = host.parent.name if host.parent and isinstance(host.parent, vim.ClusterComputeResource) else None
 323 
 324         except vim.fault.NoPermission:
 325             # User does not have read permission for the host system,
 326             # proceed without this value. This value does not contribute or hamper
 327             # provisioning or power management operations.
 328             pass
 329     if vm.summary.runtime.dasVmProtection:
 330         facts['hw_guest_ha_state'] = vm.summary.runtime.dasVmProtection.dasProtected
 331 
 332     datastores = vm.datastore
 333     for ds in datastores:
 334         facts['hw_datastores'].append(ds.info.name)
 335 
 336     try:
 337         files = vm.config.files
 338         layout = vm.layout
 339         if files:
 340             facts['hw_files'] = [files.vmPathName]
 341             for item in layout.snapshot:
 342                 for snap in item.snapshotFile:
 343                     if 'vmsn' in snap:
 344                         facts['hw_files'].append(snap)
 345             for item in layout.configFile:
 346                 facts['hw_files'].append(os.path.join(os.path.dirname(files.vmPathName), item))
 347             for item in vm.layout.logFile:
 348                 facts['hw_files'].append(os.path.join(files.logDirectory, item))
 349             for item in vm.layout.disk:
 350                 for disk in item.diskFile:
 351                     facts['hw_files'].append(disk)
 352     except Exception:
 353         pass
 354 
 355     facts['hw_folder'] = PyVmomi.get_vm_path(content, vm)
 356 
 357     cfm = content.customFieldsManager
 358     # Resolve custom values
 359     for value_obj in vm.summary.customValue:
 360         kn = value_obj.key
 361         if cfm is not None and cfm.field:
 362             for f in cfm.field:
 363                 if f.key == value_obj.key:
 364                     kn = f.name
 365                     # Exit the loop immediately, we found it
 366                     break
 367 
 368         facts['customvalues'][kn] = value_obj.value
 369 
 370     net_dict = {}
 371     vmnet = _get_vm_prop(vm, ('guest', 'net'))
 372     if vmnet:
 373         for device in vmnet:
 374             if device.deviceConfigId > 0:
 375                 net_dict[device.macAddress] = list(device.ipAddress)
 376 
 377     if vm.guest.ipAddress:
 378         if ':' in vm.guest.ipAddress:
 379             facts['ipv6'] = vm.guest.ipAddress
 380         else:
 381             facts['ipv4'] = vm.guest.ipAddress
 382 
 383     ethernet_idx = 0
 384     for entry in vm.config.hardware.device:
 385         if not hasattr(entry, 'macAddress'):
 386             continue
 387 
 388         if entry.macAddress:
 389             mac_addr = entry.macAddress
 390             mac_addr_dash = mac_addr.replace(':', '-')
 391         else:
 392             mac_addr = mac_addr_dash = None
 393 
 394         if (hasattr(entry, 'backing') and hasattr(entry.backing, 'port') and
 395                 hasattr(entry.backing.port, 'portKey') and hasattr(entry.backing.port, 'portgroupKey')):
 396             port_group_key = entry.backing.port.portgroupKey
 397             port_key = entry.backing.port.portKey
 398         else:
 399             port_group_key = None
 400             port_key = None
 401 
 402         factname = 'hw_eth' + str(ethernet_idx)
 403         facts[factname] = {
 404             'addresstype': entry.addressType,
 405             'label': entry.deviceInfo.label,
 406             'macaddress': mac_addr,
 407             'ipaddresses': net_dict.get(entry.macAddress, None),
 408             'macaddress_dash': mac_addr_dash,
 409             'summary': entry.deviceInfo.summary,
 410             'portgroup_portkey': port_key,
 411             'portgroup_key': port_group_key,
 412         }
 413         facts['hw_interfaces'].append('eth' + str(ethernet_idx))
 414         ethernet_idx += 1
 415 
 416     snapshot_facts = list_snapshots(vm)
 417     if 'snapshots' in snapshot_facts:
 418         facts['snapshots'] = snapshot_facts['snapshots']
 419         facts['current_snapshot'] = snapshot_facts['current_snapshot']
 420 
 421     facts['vnc'] = get_vnc_extraconfig(vm)
 422     return facts

 450 def list_snapshots(vm):
 451     result = {}
 452     snapshot = _get_vm_prop(vm, ('snapshot',))
 453     if not snapshot:
 454         return result
 455     if vm.snapshot is None:
 456         return result
 457 
 458     result['snapshots'] = list_snapshots_recursively(vm.snapshot.rootSnapshotList)
 459     current_snapref = vm.snapshot.currentSnapshot
 460     current_snap_obj = get_current_snap_obj(vm.snapshot.rootSnapshotList, current_snapref)
 461     if current_snap_obj:
 462         result['current_snapshot'] = deserialize_snapshot_obj(current_snap_obj[0])
 463     else:
 464         result['current_snapshot'] = dict()
 465     return result

 511 def connect_to_api(module, disconnect_atexit=True, return_si=False, hostname=None, username=None, password=None, port=None, validate_certs=None):
 512     hostname = hostname if hostname else module.params['hostname']
 513     username = username if username else module.params['username']
 514     password = password if password else module.params['password']
 515     port = port if port else module.params.get('port', 443)
 516     validate_certs = validate_certs if validate_certs else module.params['validate_certs']
 517 
 518     if not hostname:
 519         module.fail_json(msg="Hostname parameter is missing."
 520                              " Please specify this parameter in task or"
 521                              " export environment variable like 'export VMWARE_HOST=ESXI_HOSTNAME'")
 522 
 523     if not username:
 524         module.fail_json(msg="Username parameter is missing."
 525                              " Please specify this parameter in task or"
 526                              " export environment variable like 'export VMWARE_USER=ESXI_USERNAME'")
 527 
 528     if not password:
 529         module.fail_json(msg="Password parameter is missing."
 530                              " Please specify this parameter in task or"
 531                              " export environment variable like 'export VMWARE_PASSWORD=ESXI_PASSWORD'")
 532 
 533     if validate_certs and not hasattr(ssl, 'SSLContext'):
 534         module.fail_json(msg='pyVim does not support changing verification mode with python < 2.7.9. Either update '
 535                              'python or use validate_certs=false.')
 536     elif validate_certs:
 537         ssl_context = ssl.SSLContext(ssl.PROTOCOL_SSLv23)
 538         ssl_context.verify_mode = ssl.CERT_REQUIRED
 539         ssl_context.check_hostname = True
 540         ssl_context.load_default_certs()
 541     elif hasattr(ssl, 'SSLContext'):
 542         ssl_context = ssl.SSLContext(ssl.PROTOCOL_SSLv23)
 543         ssl_context.verify_mode = ssl.CERT_NONE
 544         ssl_context.check_hostname = False
 545     else:  # Python < 2.7.9 or RHEL/Centos < 7.4
 546         ssl_context = None
 547 
 548     service_instance = None
 549     proxy_host = module.params.get('proxy_host')
 550     proxy_port = module.params.get('proxy_port')
 551 
 552     connect_args = dict(
 553         host=hostname,
 554         port=port,
 555     )
 556     if ssl_context:
 557         connect_args.update(sslContext=ssl_context)
 558 
 559     msg_suffix = ''
 560     try:
 561         if proxy_host:
 562             msg_suffix = " [proxy: %s:%d]" % (proxy_host, proxy_port)
 563             connect_args.update(httpProxyHost=proxy_host, httpProxyPort=proxy_port)
 564             smart_stub = connect.SmartStubAdapter(**connect_args)
 565             session_stub = connect.VimSessionOrientedStub(smart_stub, connect.VimSessionOrientedStub.makeUserLoginMethod(username, password))
 566             service_instance = vim.ServiceInstance('ServiceInstance', session_stub)
 567         else:
 568             connect_args.update(user=username, pwd=password)
 569             service_instance = connect.SmartConnect(**connect_args)
 570     except vim.fault.InvalidLogin as invalid_login:

 665 def serialize_spec(clonespec):
 666     """Serialize a clonespec or a relocation spec"""
 667     data = {}
 668     attrs = dir(clonespec)
 669     attrs = [x for x in attrs if not x.startswith('_')]
 670     for x in attrs:
 671         xo = getattr(clonespec, x)
 672         if callable(xo):
 673             continue
 674         xt = type(xo)
 675         if xo is None:
 676             data[x] = None
 677         elif isinstance(xo, vim.vm.ConfigSpec):
 678             data[x] = serialize_spec(xo)
 679         elif isinstance(xo, vim.vm.RelocateSpec):
 680             data[x] = serialize_spec(xo)
 681         elif isinstance(xo, vim.vm.device.VirtualDisk):
 682             data[x] = serialize_spec(xo)
 683         elif isinstance(xo, vim.vm.device.VirtualDeviceSpec.FileOperation):
 684             data[x] = to_text(xo)
 685         elif isinstance(xo, vim.Description):
 686             data[x] = {
 687                 'dynamicProperty': serialize_spec(xo.dynamicProperty),
 688                 'dynamicType': serialize_spec(xo.dynamicType),
 689                 'label': serialize_spec(xo.label),
 690                 'summary': serialize_spec(xo.summary),
 691             }
 692         elif hasattr(xo, 'name'):
 693             data[x] = to_text(xo) + ':' + to_text(xo.name)
 694         elif isinstance(xo, vim.vm.ProfileSpec):
 695             pass
 696         elif issubclass(xt, list):
 697             data[x] = []
 698             for xe in xo:
 699                 data[x].append(serialize_spec(xe))
 700         elif issubclass(xt, string_types + integer_types + (float, bool)):
 701             if issubclass(xt, integer_types):
 702                 data[x] = int(xo)
 703             else:
 704                 data[x] = to_text(xo)
 705         elif issubclass(xt, bool):
 706             data[x] = xo
 707         elif issubclass(xt, dict):
 708             data[to_text(x)] = {}
 709             for k, v in xo.items():
 710                 k = to_text(k)
 711                 data[x][k] = serialize_spec(v)
 712         else:
 713             data[x] = str(xt)
 714 
 715     return data

 733 def set_vm_power_state(content, vm, state, force, timeout=0):
 734     """
 735     Set the power status for a VM determined by the current and
 736     requested states. force is forceful
 737     """
 738     facts = gather_vm_facts(content, vm)
 739     expected_state = state.replace('_', '').replace('-', '').lower()
 740     current_state = facts['hw_power_status'].lower()
 741     result = dict(
 742         changed=False,
 743         failed=False,
 744     )
 745 
 746     # Need Force
 747     if not force and current_state not in ['poweredon', 'poweredoff']:
 748         result['failed'] = True
 749         result['msg'] = "Virtual Machine is in %s power state. Force is required!" % current_state
 750         result['instance'] = gather_vm_facts(content, vm)
 751         return result
 752 
 753     # State is not already true
 754     if current_state != expected_state:
 755         task = None
 756         try:
 757             if expected_state == 'poweredoff':
 758                 task = vm.PowerOff()
 759 
 760             elif expected_state == 'poweredon':
 761                 task = vm.PowerOn()
 762 
 763             elif expected_state == 'restarted':
 764                 if current_state in ('poweredon', 'poweringon', 'resetting', 'poweredoff'):
 765                     task = vm.Reset()
 766                 else:
 767                     result['failed'] = True
 768                     result['msg'] = "Cannot restart virtual machine in the current state %s" % current_state
 769 
 770             elif expected_state == 'suspended':
 771                 if current_state in ('poweredon', 'poweringon'):
 772                     task = vm.Suspend()
 773                 else:
 774                     result['failed'] = True
 775                     result['msg'] = 'Cannot suspend virtual machine in the current state %s' % current_state
 776 
 777             elif expected_state in ['shutdownguest', 'rebootguest']:
 778                 if current_state == 'poweredon':
 779                     if vm.guest.toolsRunningStatus == 'guestToolsRunning':
 780                         if expected_state == 'shutdownguest':
 781                             task = vm.ShutdownGuest()
 782                             if timeout > 0:
 783                                 result.update(wait_for_poweroff(vm, timeout))
 784                         else:
 785                             task = vm.RebootGuest()
 786                         # Set result['changed'] immediately because
 787                         # shutdown and reboot return None.
 788                         result['changed'] = True
 789                     else:
 790                         result['failed'] = True
 791                         result['msg'] = "VMware tools should be installed for guest shutdown/reboot"
 792                 else:
 793                     result['failed'] = True
 794                     result['msg'] = "Virtual machine %s must be in poweredon state for guest shutdown/reboot" % vm.name
 795 
 796             else:
 797                 result['failed'] = True
 798                 result['msg'] = "Unsupported expected state provided: %s" % expected_state
 799 
 800         except Exception as e:
 801             result['failed'] = True
 802             result['msg'] = to_text(e)
 803 
 804         if task:
 805             wait_for_task(task)
 806             if task.info.state == 'error':
 807                 result['failed'] = True
 808                 result['msg'] = task.info.error.msg
 809             else:
 810                 result['changed'] = True
 811 
 812     # need to get new metadata if changed
 813     result['instance'] = gather_vm_facts(content, vm)
 814 
 815     return result

1033                 if not dcpath.endswith('/'):
1034                     dcpath += '/'

1040                                           "name [%s], please specify folder path other than blank "

1045                 else:

1072         if vm_obj:
1073             self.current_vm_obj = vm_obj
1074

1576     def to_json(self, obj, properties=None):
1577         """
1578         Convert a vSphere (pyVmomi) Object into JSON.  This is a deep
1579         transformation.  The list of properties is optional - if not
1580         provided then all properties are deeply converted.  The resulting
1581         JSON is sorted to improve human readability.
1582 
1583         Requires upstream support from pyVmomi > 6.7.1
1584         (https://github.com/vmware/pyvmomi/pull/732)
1585 
1586         Args:
1587           - obj (object): vim object
1588           - properties (list, optional): list of properties following
1589                 the property collector specification, for example:
1590                 ["config.hardware.memoryMB", "name", "overallStatus"]
1591                 default is a complete object dump, which can be large
1592 
1593         Return:
1594           dict
1595         """
1596         if not HAS_PYVMOMIJSON:
1597             self.module.fail_json(msg='The installed version of pyvmomi lacks JSON output support; need pyvmomi>6.7.1')
1598 
1599         result = dict()
1600         if properties:
1601             for prop in properties:
1602                 try:
1603                     if '.' in prop:
1604                         key, remainder = prop.split('.', 1)
1605                         tmp = dict()
1606                         tmp[key] = self._extract(self._jsonify(getattr(obj, key)), remainder)
1607                         self._deepmerge(result, tmp)
1608                     else:
1609                         result[prop] = self._jsonify(getattr(obj, prop))
1610                         # To match gather_vm_facts output
1611                         prop_name = prop
1612                         if prop.lower() == '_moid':
1613                             prop_name = 'moid'
1614                         elif prop.lower() == '_vimref':
1615                             prop_name = 'vimref'
1616                         result[prop_name] = result[prop]
1617                 except (AttributeError, KeyError):
1618                     self.module.fail_json(msg="Property '{0}' not found.".format(prop))
1619         else:
1620             result = self._jsonify(obj)
1621         return result
