  46 def tower_auth_config(module):
  47     '''tower_auth_config attempts to load the tower-cli.cfg file
  48     specified from the `tower_config_file` parameter. If found,
  49     if returns the contents of the file as a dictionary, else
  50     it will attempt to fetch values from the module params and
  51     only pass those values that have been set.
  52     '''
  53     config_file = module.params.pop('tower_config_file', None)
  54     if config_file:
  55         if not os.path.exists(config_file):
  56             module.fail_json(msg='file not found: %s' % config_file)
  57         if os.path.isdir(config_file):
  58             module.fail_json(msg='directory can not be used as config file: %s' % config_file)
  59 
  60         with open(config_file, 'rb') as f:
  61             return parser.string_to_dict(f.read())
  62     else:
  63         auth_config = {}
  64         host = module.params.pop('tower_host', None)
  65         if host:
  66             auth_config['host'] = host
  67         username = module.params.pop('tower_username', None)
  68         if username:
  69             auth_config['username'] = username
  70         password = module.params.pop('tower_password', None)
  71         if password:
  72             auth_config['password'] = password
  73         module.params.pop('tower_verify_ssl', None)  # pop alias if used
  74         verify_ssl = module.params.pop('validate_certs', None)
  75         if verify_ssl is not None:
  76             auth_config['verify_ssl'] = verify_ssl
  77         return auth_config

  80 def tower_check_mode(module):
  81     '''Execute check mode logic for Ansible Tower modules'''
  82     if module.check_mode:
  83         try:
  84             result = client.get('/ping').json()
  85             module.exit_json(changed=True, tower_version='{0}'.format(result['version']))
  86         except (exc.ServerError, exc.ConnectionError, exc.BadRequest) as excinfo:
  87             module.fail_json(changed=False, msg='Failed check mode: {0}'.format(excinfo))

  90 class TowerModule(AnsibleModule):
  91     def __init__(self, argument_spec, **kwargs):
  92         args = dict(
  93             tower_host=dict(),
  94             tower_username=dict(),
  95             tower_password=dict(no_log=True),
  96             validate_certs=dict(type='bool', aliases=['tower_verify_ssl']),
  97             tower_config_file=dict(type='path'),
  98         )
  99         args.update(argument_spec)
 100 
 101         mutually_exclusive = kwargs.get('mutually_exclusive', [])
 102         kwargs['mutually_exclusive'] = mutually_exclusive.extend((
 103             ('tower_config_file', 'tower_host'),
 104             ('tower_config_file', 'tower_username'),
 105             ('tower_config_file', 'tower_password'),
 106             ('tower_config_file', 'validate_certs'),
 107         ))
 108 
 109         super(TowerModule, self).__init__(argument_spec=args, **kwargs)
 110 
 111         if not HAS_TOWER_CLI:
 112             self.fail_json(msg=missing_required_lib('ansible-tower-cli'),
 113                            exception=TOWER_CLI_IMP_ERR)
