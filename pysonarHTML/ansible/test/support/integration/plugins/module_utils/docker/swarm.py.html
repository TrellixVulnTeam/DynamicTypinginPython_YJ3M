  30     def get_swarm_node_id(self):
  31         """
  32         Get the 'NodeID' of the Swarm node or 'None' if host is not in Swarm. It returns the NodeID
  33         of Docker host the module is executed on
  34         :return:
  35             NodeID of host or 'None' if not part of Swarm
  36         """
  37 
  38         try:
  39             info = self.info()
  40         except APIError as exc:
  41             self.fail("Failed to get node information for %s" % to_native(exc))
  42 
  43         if info:
  44             json_str = json.dumps(info, ensure_ascii=False)
  45             swarm_info = json.loads(json_str)
  46             if swarm_info['Swarm']['NodeID']:
  47                 return swarm_info['Swarm']['NodeID']
  48         return None

  50     def check_if_swarm_node(self, node_id=None):
  51         """
  52         Checking if host is part of Docker Swarm. If 'node_id' is not provided it reads the Docker host
  53         system information looking if specific key in output exists. If 'node_id' is provided then it tries to
  54         read node information assuming it is run on Swarm manager. The get_node_inspect() method handles exception if
  55         it is not executed on Swarm manager
  56 
  57         :param node_id: Node identifier
  58         :return:
  59             bool: True if node is part of Swarm, False otherwise
  60         """
  61 
  62         if node_id is None:
  63             try:
  64                 info = self.info()
  65             except APIError:
  66                 self.fail("Failed to get host information.")
  67 
  68             if info:
  69                 json_str = json.dumps(info, ensure_ascii=False)
  70                 swarm_info = json.loads(json_str)
  71                 if swarm_info['Swarm']['NodeID']:
  72                     return True
  73                 if swarm_info['Swarm']['LocalNodeState'] in ('active', 'pending', 'locked'):
  74                     return True
  75             return False
  76         else:
  77             try:
  78                 node_info = self.get_node_inspect(node_id=node_id)
  79             except APIError:
  80                 return
  81 
  82             if node_info['ID'] is not None:
  83                 return True
  84             return False

  86     def check_if_swarm_manager(self):
  87         """
  88         Checks if node role is set as Manager in Swarm. The node is the docker host on which module action
  89         is performed. The inspect_swarm() will fail if node is not a manager
  90 
  91         :return: True if node is Swarm Manager, False otherwise
  92         """
  93 
  94         try:
  95             self.inspect_swarm()
  96             return True
  97         except APIError:
  98             return False

 100     def fail_task_if_not_swarm_manager(self):
 101         """
 102         If host is not a swarm manager then Ansible task on this host should end with 'failed' state
 103         """
 104         if not self.check_if_swarm_manager():
 105             self.fail("Error running docker swarm module: must run on swarm manager node")
 106 
 107     def check_if_swarm_worker(self):
 108         """
 109         Checks if node role is set as Worker in Swarm. The node is the docker host on which module action
 110         is performed. Will fail if run on host that is not part of Swarm via check_if_swarm_node()
 111 
 112         :return: True if node is Swarm Worker, False otherwise
 113         """
 114 
 115         if self.check_if_swarm_node() and not self.check_if_swarm_manager():
 116             return True
 117         return False

 119     def check_if_swarm_node_is_down(self, node_id=None, repeat_check=1):
 120         """
 121         Checks if node status on Swarm manager is 'down'. If node_id is provided it query manager about
 122         node specified in parameter, otherwise it query manager itself. If run on Swarm Worker node or
 123         host that is not part of Swarm it will fail the playbook
 124 
 125         :param repeat_check: number of check attempts with 5 seconds delay between them, by default check only once
 126         :param node_id: node ID or name, if None then method will try to get node_id of host module run on
 127         :return:
 128             True if node is part of swarm but its state is down, False otherwise
 129         """
 130 
 131         if repeat_check < 1:
 132             repeat_check = 1
 133 
 134         if node_id is None:
 135             node_id = self.get_swarm_node_id()
 136 
 137         for retry in range(0, repeat_check):
 138             if retry > 0:
 139                 sleep(5)
 140             node_info = self.get_node_inspect(node_id=node_id)
 141             if node_info['Status']['State'] == 'down':
 142                 return True
 143         return False

 145     def get_node_inspect(self, node_id=None, skip_missing=False):
 146         """
 147         Returns Swarm node info as in 'docker node inspect' command about single node
 148 
 149         :param skip_missing: if True then function will return None instead of failing the task
 150         :param node_id: node ID or name, if None then method will try to get node_id of host module run on
 151         :return:
 152             Single node information structure
 153         """
 154 
 155         if node_id is None:
 156             node_id = self.get_swarm_node_id()
 157 
 158         if node_id is None:
 159             self.fail("Failed to get node information.")
 160 
 161         try:
 162             node_info = self.inspect_node(node_id=node_id)
 163         except APIError as exc:
 164             if exc.status_code == 503:
 165                 self.fail("Cannot inspect node: To inspect node execute module on Swarm Manager")
 166             if exc.status_code == 404:
 167                 if skip_missing:
 168                     return None
 169             self.fail("Error while reading from Swarm manager: %s" % to_native(exc))
 170         except Exception as exc:
 171             self.fail("Error inspecting swarm node: %s" % exc)
 172 
 173         json_str = json.dumps(node_info, ensure_ascii=False)
 174         node_info = json.loads(json_str)
 175 
 176         if 'ManagerStatus' in node_info:
 177             if node_info['ManagerStatus'].get('Leader'):
 178                 # This is workaround of bug in Docker when in some cases the Leader IP is 0.0.0.0
 179                 # Check moby/moby#35437 for details
 180                 count_colons = node_info['ManagerStatus']['Addr'].count(":")
 181                 if count_colons == 1:
 182                     swarm_leader_ip = node_info['ManagerStatus']['Addr'].split(":", 1)[0] or node_info['Status']['Addr']
 183                 else:
 184                     swarm_leader_ip = node_info['Status']['Addr']
 185                 node_info['Status']['Addr'] = swarm_leader_ip
 186         return node_info

 188     def get_all_nodes_inspect(self):
 189         """
 190         Returns Swarm node info as in 'docker node inspect' command about all registered nodes
 191 
 192         :return:
 193             Structure with information about all nodes
 194         """
 195         try:
 196             node_info = self.nodes()
 197         except APIError as exc:
 198             if exc.status_code == 503:
 199                 self.fail("Cannot inspect node: To inspect node execute module on Swarm Manager")
 200             self.fail("Error while reading from Swarm manager: %s" % to_native(exc))
 201         except Exception as exc:
 202             self.fail("Error inspecting swarm node: %s" % exc)
 203 
 204         json_str = json.dumps(node_info, ensure_ascii=False)
 205         node_info = json.loads(json_str)
 206         return node_info

 208     def get_all_nodes_list(self, output='short'):
 209         """
 210         Returns list of nodes registered in Swarm
 211 
 212         :param output: Defines format of returned data
 213         :return:
 214             If 'output' is 'short' then return data is list of nodes hostnames registered in Swarm,
 215             if 'output' is 'long' then returns data is list of dict containing the attributes as in
 216             output of command 'docker node ls'
 217         """
 218         nodes_list = []
 219 
 220         nodes_inspect = self.get_all_nodes_inspect()
 221         if nodes_inspect is None:
 222             return None
 223 
 224         if output == 'short':
 225             for node in nodes_inspect:
 226                 nodes_list.append(node['Description']['Hostname'])
 227         elif output == 'long':
 228             for node in nodes_inspect:
 229                 node_property = {}
 230 
 231                 node_property.update({'ID': node['ID']})
 232                 node_property.update({'Hostname': node['Description']['Hostname']})
 233                 node_property.update({'Status': node['Status']['State']})
 234                 node_property.update({'Availability': node['Spec']['Availability']})
 235                 if 'ManagerStatus' in node:
 236                     if node['ManagerStatus']['Leader'] is True:
 237                         node_property.update({'Leader': True})
 238                     node_property.update({'ManagerStatus': node['ManagerStatus']['Reachability']})
 239                 node_property.update({'EngineVersion': node['Description']['Engine']['EngineVersion']})
 240 
 241                 nodes_list.append(node_property)
 242         else:
 243             return None
 244 
 245         return nodes_list

 255     def get_service_inspect(self, service_id, skip_missing=False):
 256         """
 257         Returns Swarm service info as in 'docker service inspect' command about single service
 258 
 259         :param service_id: service ID or name
 260         :param skip_missing: if True then function will return None instead of failing the task
 261         :return:
 262             Single service information structure
 263         """
 264         try:
 265             service_info = self.inspect_service(service_id)
 266         except NotFound as exc:
 267             if skip_missing is False:
 268                 self.fail("Error while reading from Swarm manager: %s" % to_native(exc))
 269             else:
 270                 return None
 271         except APIError as exc:
 272             if exc.status_code == 503:
 273                 self.fail("Cannot inspect service: To inspect service execute module on Swarm Manager")
 274             self.fail("Error inspecting swarm service: %s" % exc)
 275         except Exception as exc:
 276             self.fail("Error inspecting swarm service: %s" % exc)
 277 
 278         json_str = json.dumps(service_info, ensure_ascii=False)
 279         service_info = json.loads(json_str)
 280         return service_info
