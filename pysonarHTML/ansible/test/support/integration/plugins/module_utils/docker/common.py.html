 139 def is_image_name_id(name):
 140     """Check whether the given image name is in fact an image ID (hash)."""
 141     if re.match('^sha256:[0-9a-fA-F]{64}$', name):
 142         return True
 143     return False

 146 def is_valid_tag(tag, allow_empty=False):
 147     """Check whether the given string is a valid docker tag name."""
 148     if not tag:
 149         return allow_empty
 150     # See here ("Extended description") for a definition what tags can be:
 151     # https://docs.docker.com/engine/reference/commandline/tag/
 152     return bool(re.match('^[a-zA-Z0-9_][a-zA-Z0-9_.-]{0,127}$', tag))

 155 def sanitize_result(data):
 156     """Sanitize data object for return to Ansible.
 157 
 158     When the data object contains types such as docker.types.containers.HostConfig,
 159     Ansible will fail when these are returned via exit_json or fail_json.
 160     HostConfig is derived from dict, but its constructor requires additional
 161     arguments. This function sanitizes data structures by recursively converting
 162     everything derived from dict to dict and everything derived from list (and tuple)
 163     to a list.
 164     """
 165     if isinstance(data, dict):
 166         return dict((k, sanitize_result(v)) for k, v in data.items())
 167     elif isinstance(data, (list, tuple)):
 168         return [sanitize_result(v) for v in data]
 169     else:
 170         return data

 189 def update_tls_hostname(result):
 190     if result['tls_hostname'] is None:
 191         # get default machine name from the url
 192         parsed_url = urlparse(result['docker_host'])
 193         if ':' in parsed_url.netloc:
 194             result['tls_hostname'] = parsed_url.netloc[:parsed_url.netloc.rindex(':')]
 195         else:
 196             result['tls_hostname'] = parsed_url

 207 def get_connect_params(auth, fail_function):
 208     if auth['tls'] or auth['tls_verify']:
 209         auth['docker_host'] = auth['docker_host'].replace('tcp://', 'https://')
 210 
 211     if auth['tls_verify'] and auth['cert_path'] and auth['key_path']:
 212         # TLS with certs and host verification
 213         if auth['cacert_path']:
 214             tls_config = _get_tls_config(client_cert=(auth['cert_path'], auth['key_path']),
 215                                          ca_cert=auth['cacert_path'],
 216                                          verify=True,
 217                                          assert_hostname=auth['tls_hostname'],
 218                                          ssl_version=auth['ssl_version'],
 219                                          fail_function=fail_function)
 220         else:
 221             tls_config = _get_tls_config(client_cert=(auth['cert_path'], auth['key_path']),
 222                                          verify=True,
 223                                          assert_hostname=auth['tls_hostname'],
 224                                          ssl_version=auth['ssl_version'],
 225                                          fail_function=fail_function)
 226 
 227         return dict(base_url=auth['docker_host'],
 228                     tls=tls_config,
 229                     version=auth['api_version'],
 230                     timeout=auth['timeout'])
 231 
 232     if auth['tls_verify'] and auth['cacert_path']:
 233         # TLS with cacert only
 234         tls_config = _get_tls_config(ca_cert=auth['cacert_path'],
 235                                      assert_hostname=auth['tls_hostname'],
 236                                      verify=True,
 237                                      ssl_version=auth['ssl_version'],
 238                                      fail_function=fail_function)
 239         return dict(base_url=auth['docker_host'],
 240                     tls=tls_config,
 241                     version=auth['api_version'],
 242                     timeout=auth['timeout'])
 243 
 244     if auth['tls_verify']:
 245         # TLS with verify and no certs
 246         tls_config = _get_tls_config(verify=True,
 247                                      assert_hostname=auth['tls_hostname'],
 248                                      ssl_version=auth['ssl_version'],
 249                                      fail_function=fail_function)
 250         return dict(base_url=auth['docker_host'],
 251                     tls=tls_config,
 252                     version=auth['api_version'],
 253                     timeout=auth['timeout'])
 254 
 255     if auth['tls'] and auth['cert_path'] and auth['key_path']:
 256         # TLS with certs and no host verification
 257         tls_config = _get_tls_config(client_cert=(auth['cert_path'], auth['key_path']),
 258                                      verify=False,
 259                                      ssl_version=auth['ssl_version'],
 260                                      fail_function=fail_function)
 261         return dict(base_url=auth['docker_host'],
 262                     tls=tls_config,
 263                     version=auth['api_version'],
 264                     timeout=auth['timeout'])
 265 
 266     if auth['tls']:
 267         # TLS with no certs and not host verification
 268         tls_config = _get_tls_config(verify=False,
 269                                      ssl_version=auth['ssl_version'],
 270                                      fail_function=fail_function)
 271         return dict(base_url=auth['docker_host'],
 272                     tls=tls_config,
 273                     version=auth['api_version'],
 274                     timeout=auth['timeout'])
 275 
 276     # No TLS
 277     return dict(base_url=auth['docker_host'],
 278                 version=auth['api_version'],
 279                 timeout=auth['timeout'])

 292                  required_together=None, required_if=None, min_docker_version=MIN_DOCKER_VERSION,
 293                  min_docker_api_version=None, option_minimal_versions=None,

 296         # Modules can put information in here which will always be returned

 298         self.fail_results = fail_results or {}
 299 
 300         merged_arg_spec = dict()
 301         merged_arg_spec.update(DOCKER_COMMON_ARGS)

 304             self.arg_spec = merged_arg_spec

 307         mutually_exclusive_params += DOCKER_MUTUALLY_EXCLUSIVE
 308         if mutually_exclusive:
 309             mutually_exclusive_params += mutually_exclusive
 310 
 311         required_together_params = []
 312         required_together_params += DOCKER_REQUIRED_TOGETHER
 313         if required_together:
 314             required_together_params += required_together
 315 
 316         self.module = AnsibleModule(
 317             argument_spec=merged_arg_spec,
 318             supports_check_mode=supports_check_mode,
 319             mutually_exclusive=mutually_exclusive_params,
 320             required_together=required_together_params,
 321             required_if=required_if)
 322 
 323         NEEDS_DOCKER_PY2 = (LooseVersion(min_docker_version) >= LooseVersion('2.0.0'))
 324 
 325         self.docker_py_version = LooseVersion(docker_version)
 326 
 327         if HAS_DOCKER_MODELS and HAS_DOCKER_SSLADAPTER:
 328             self.fail("Cannot have both the docker-py and docker python modules (old and new version of Docker "

 330                       "installation. Please uninstall both packages, and re-install only the docker-py or docker "

 336             if NEEDS_DOCKER_PY2:
 337                 msg = missing_required_lib("Docker SDK for Python: docker")
 338                 msg = msg + ", for example via `pip install docker`. The error was: %s"

 340                 msg = missing_required_lib("Docker SDK for Python: docker (Python >= 2.7) or docker-py (Python 2.6)")
 341                 msg = msg + ", for example via `pip install docker` or `pip install docker-py` (Python 2.6). The error was: %s"
 342             self.fail(msg % HAS_DOCKER_ERROR)
 343 
 344         if self.docker_py_version < LooseVersion(min_docker_version):
 345             msg = "Error: Docker SDK for Python version is %s (%s's Python %s). Minimum version required is %s."
 346             if not NEEDS_DOCKER_PY2:
 347                 # The minimal required version is < 2.0 (and the current version as well).
 348                 # Advertise docker (instead of docker-py) for non-Python-2.6 users.

 357         self.check_mode = self.module.check_mode

 390     @staticmethod
 391     def _get_value(param_name, param_value, env_variable, default_value):
 392         if param_value is not None:
 393             # take module parameter value
 394             if param_value in BOOLEANS_TRUE:
 395                 return True
 396             if param_value in BOOLEANS_FALSE:
 397                 return False
 398             return param_value
 399 
 400         if env_variable is not None:
 401             env_value = os.environ.get(env_variable)
 402             if env_value is not None:
 403                 # take the env variable value
 404                 if param_name == 'cert_path':
 405                     return os.path.join(env_value, 'cert.pem')
 406                 if param_name == 'cacert_path':
 407                     return os.path.join(env_value, 'ca.pem')
 408                 if param_name == 'key_path':
 409                     return os.path.join(env_value, 'key.pem')
 410                 if env_value in BOOLEANS_TRUE:
 411                     return True
 412                 if env_value in BOOLEANS_FALSE:
 413                     return False
 414                 return env_value
 415 
 416         # take the default
 417         return default_value

 419     @property
 420     def auth_params(self):
 421         # Get authentication credentials.
 422         # Precedence: module parameters-> environment variables-> defaults.
 423 
 424         self.log('Getting credentials')
 425 
 426         params = dict()
 427         for key in DOCKER_COMMON_ARGS:
 428             params[key] = self.module.params.get(key)
 429 
 430         if self.module.params.get('use_tls'):
 431             # support use_tls option in docker_image.py. This will be deprecated.
 432             use_tls = self.module.params.get('use_tls')
 433             if use_tls == 'encrypt':
 434                 params['tls'] = True
 435             if use_tls == 'verify':
 436                 params['validate_certs'] = True
 437 
 438         result = dict(
 439             docker_host=self._get_value('docker_host', params['docker_host'], 'DOCKER_HOST',
 440                                         DEFAULT_DOCKER_HOST),
 441             tls_hostname=self._get_value('tls_hostname', params['tls_hostname'],
 442                                          'DOCKER_TLS_HOSTNAME', DEFAULT_TLS_HOSTNAME),
 443             api_version=self._get_value('api_version', params['api_version'], 'DOCKER_API_VERSION',
 444                                         'auto'),
 445             cacert_path=self._get_value('cacert_path', params['ca_cert'], 'DOCKER_CERT_PATH', None),
 446             cert_path=self._get_value('cert_path', params['client_cert'], 'DOCKER_CERT_PATH', None),
 447             key_path=self._get_value('key_path', params['client_key'], 'DOCKER_CERT_PATH', None),
 448             ssl_version=self._get_value('ssl_version', params['ssl_version'], 'DOCKER_SSL_VERSION', None),
 449             tls=self._get_value('tls', params['tls'], 'DOCKER_TLS', DEFAULT_TLS),
 450             tls_verify=self._get_value('tls_verfy', params['validate_certs'], 'DOCKER_TLS_VERIFY',
 451                                        DEFAULT_TLS_VERIFY),
 452             timeout=self._get_value('timeout', params['timeout'], 'DOCKER_TIMEOUT',
 453                                     DEFAULT_TIMEOUT_SECONDS),
 454         )
 455 
 456         update_tls_hostname(result)
 457 
 458         return result

 460     def _handle_ssl_error(self, error):
 461         match = re.match(r"hostname.*doesn\'t match (\'.*\')", str(error))
 462         if match:
 463             self.fail("You asked for verification that Docker daemons certificate's hostname matches %s. "
 464                       "The actual certificate's hostname is %s. Most likely you need to set DOCKER_TLS_HOSTNAME "
 465                       "or pass `tls_hostname` with a value of %s. You may also use TLS without verification by "
 466                       "setting the `tls` parameter to true."
 467                       % (self.auth_params['tls_hostname'], match.group(1), match.group(1)))
 468         self.fail("SSL Exception: %s" % (error))

 470     def _get_minimal_versions(self, option_minimal_versions, ignore_params=None):
 471         self.option_minimal_versions = dict()
 472         for option in self.module.argument_spec:
 473             if ignore_params is not None:
 474                 if option in ignore_params:
 475                     continue
 476             self.option_minimal_versions[option] = dict()
 477         self.option_minimal_versions.update(option_minimal_versions)
 478 
 479         for option, data in self.option_minimal_versions.items():
 480             # Test whether option is supported, and store result
 481             support_docker_py = True
 482             support_docker_api = True
 483             if 'docker_py_version' in data:
 484                 support_docker_py = self.docker_py_version >= LooseVersion(data['docker_py_version'])
 485             if 'docker_api_version' in data:
 486                 support_docker_api = self.docker_api_version >= LooseVersion(data['docker_api_version'])
 487             data['supported'] = support_docker_py and support_docker_api
 488             # Fail if option is not supported but used
 489             if not data['supported']:
 490                 # Test whether option is specified
 491                 if 'detect_usage' in data:
 492                     used = data['detect_usage'](self)
 493                 else:
 494                     used = self.module.params.get(option) is not None
 495                     if used and 'default' in self.module.argument_spec[option]:
 496                         used = self.module.params[option] != self.module.argument_spec[option]['default']
 497                 if used:
 498                     # If the option is used, compose error message.
 499                     if 'usage_msg' in data:
 500                         usg = data['usage_msg']
 501                     else:
 502                         usg = 'set %s option' % (option, )
 503                     if not support_docker_api:
 504                         msg = 'Docker API version is %s. Minimum version required is %s to %s.'
 505                         msg = msg % (self.docker_api_version_str, data['docker_api_version'], usg)
 506                     elif not support_docker_py:
 507                         msg = "Docker SDK for Python version is %s (%s's Python %s). Minimum version required is %s to %s. "
 508                         if LooseVersion(data['docker_py_version']) < LooseVersion('2.0.0'):
 509                             msg += DOCKERPYUPGRADE_RECOMMEND_DOCKER
 510                         elif self.docker_py_version < LooseVersion('2.0.0'):

 604     def find_image(self, name, tag):
 605         '''
 606         Lookup an image (by name and tag) and return the inspection results.
 607         '''
 608         if not name:
 609             return None
 610 
 611         self.log("Find image %s:%s" % (name, tag))
 612         images = self._image_lookup(name, tag)
 613         if not images:
 614             # In API <= 1.20 seeing 'docker.io/<name>' as the name of images pulled from docker hub
 615             registry, repo_name = auth.resolve_repository_name(name)
 616             if registry == 'docker.io':
 617                 # If docker.io is explicitly there in name, the image
 618                 # isn't found in some cases (#41509)
 619                 self.log("Check for docker.io image: %s" % repo_name)
 620                 images = self._image_lookup(repo_name, tag)
 621                 if not images and repo_name.startswith('library/'):
 622                     # Sometimes library/xxx images are not found
 623                     lookup = repo_name[len('library/'):]
 624                     self.log("Check for docker.io image: %s" % lookup)
 625                     images = self._image_lookup(lookup, tag)
 626                 if not images:
 627                     # Last case: if docker.io wasn't there, it can be that
 628                     # the image wasn't found either (#15586)
 629                     lookup = "%s/%s" % (registry, repo_name)
 630                     self.log("Check for docker.io image: %s" % lookup)
 631                     images = self._image_lookup(lookup, tag)
 632 
 633         if len(images) > 1:
 634             self.fail("Registry returned more than one result for %s:%s" % (name, tag))
 635 
 636         if len(images) == 1:
 637             try:
 638                 inspection = self.inspect_image(images[0]['Id'])
 639             except Exception as exc:
 640                 self.fail("Error inspecting image %s:%s - %s" % (name, tag, str(exc)))
 641             return inspection
 642 
 643         self.log("Image %s:%s not found." % (name, tag))
 644         return None

 660     def _image_lookup(self, name, tag):
 661         '''
 662         Including a tag in the name parameter sent to the Docker SDK for Python images method
 663         does not work consistently. Instead, get the result set for name and manually check
 664         if the tag exists.
 665         '''
 666         try:
 667             response = self.images(name=name)
 668         except Exception as exc:
 669             self.fail("Error searching for image %s - %s" % (name, str(exc)))
 670         images = response
 671         if tag:
 672             lookup = "%s:%s" % (name, tag)
 673             lookup_digest = "%s@%s" % (name, tag)
 674             images = []
 675             for image in response:
 676                 tags = image.get('RepoTags')
 677                 digests = image.get('RepoDigests')
 678                 if (tags and lookup in tags) or (digests and lookup_digest in digests):
 679                     images = [image]
 680                     break
 681         return images

 683     def pull_image(self, name, tag="latest"):
 684         '''
 685         Pull an image
 686         '''
 687         self.log("Pulling image %s:%s" % (name, tag))
 688         old_tag = self.find_image(name, tag)
 689         try:
 690             for line in self.pull(name, tag=tag, stream=True, decode=True):
 691                 self.log(line, pretty_print=True)
 692                 if line.get('error'):
 693                     if line.get('errorDetail'):
 694                         error_detail = line.get('errorDetail')
 695                         self.fail("Error pulling %s - code: %s message: %s" % (name,
 696                                                                                error_detail.get('code'),
 697                                                                                error_detail.get('message')))
 698                     else:
 699                         self.fail("Error pulling %s - %s" % (name, line.get('error')))
 700         except Exception as exc:
 701             self.fail("Error pulling image %s:%s - %s" % (name, tag, str(exc)))
 702 
 703         new_tag = self.find_image(name, tag)
 704 
 705         return new_tag, old_tag == new_tag

 707     def report_warnings(self, result, warnings_key=None):
 708         '''
 709         Checks result of client operation for warnings, and if present, outputs them.
 710 
 711         warnings_key should be a list of keys used to crawl the result dictionary.
 712         For example, if warnings_key == ['a', 'b'], the function will consider
 713         result['a']['b'] if these keys exist. If the result is a non-empty string, it
 714         will be reported as a warning. If the result is a list, every entry will be
 715         reported as a warning.
 716 
 717         In most cases (if warnings are returned at all), warnings_key should be
 718         ['Warnings'] or ['Warning']. The default value (if not specified) is ['Warnings'].
 719         '''
 720         if warnings_key is None:
 721             warnings_key = ['Warnings']
 722         for key in warnings_key:
 723             if not isinstance(result, Mapping):
 724                 return
 725             result = result.get(key)
 726         if isinstance(result, Sequence):
 727             for warning in result:
 728                 self.module.warn('Docker warning: {0}'.format(warning))
 729         elif isinstance(result, string_types) and result:
 730             self.module.warn('Docker warning: {0}'.format(result))

 748 def compare_dict_allow_more_present(av, bv):
 749     '''
 750     Compare two dictionaries for whether every entry of the first is in the second.
 751     '''
 752     for key, value in av.items():
 753         if key not in bv:
 754             return False
 755         if bv[key] != value:
 756             return False
 757     return True

 760 def compare_generic(a, b, method, datatype):
 761     '''
 762     Compare values a and b as described by method and datatype.
 763 
 764     Returns ``True`` if the values compare equal, and ``False`` if not.
 765 
 766     ``a`` is usually the module's parameter, while ``b`` is a property
 767     of the current object. ``a`` must not be ``None`` (except for
 768     ``datatype == 'value'``).
 769 
 770     Valid values for ``method`` are:
 771     - ``ignore`` (always compare as equal);
 772     - ``strict`` (only compare if really equal)
 773     - ``allow_more_present`` (allow b to have elements which a does not have).
 774 
 775     Valid values for ``datatype`` are:
 776     - ``value``: for simple values (strings, numbers, ...);
 777     - ``list``: for ``list``s or ``tuple``s where order matters;
 778     - ``set``: for ``list``s, ``tuple``s or ``set``s where order does not
 779       matter;
 780     - ``set(dict)``: for ``list``s, ``tuple``s or ``sets`` where order does
 781       not matter and which contain ``dict``s; ``allow_more_present`` is used
 782       for the ``dict``s, and these are assumed to be dictionaries of values;
 783     - ``dict``: for dictionaries of values.
 784     '''
 785     if method == 'ignore':
 786         return True
 787     # If a or b is None:
 788     if a is None or b is None:
 789         # If both are None: equality
 790         if a == b:
 791             return True
 792         # Otherwise, not equal for values, and equal
 793         # if the other is empty for set/list/dict
 794         if datatype == 'value':
 795             return False
 796         # For allow_more_present, allow a to be None
 797         if method == 'allow_more_present' and a is None:
 798             return True
 799         # Otherwise, the iterable object which is not None must have length 0
 800         return len(b if a is None else a) == 0
 801     # Do proper comparison (both objects not None)
 802     if datatype == 'value':
 803         return a == b
 804     elif datatype == 'list':
 805         if method == 'strict':
 806             return a == b
 807         else:
 808             i = 0
 809             for v in a:
 810                 while i < len(b) and b[i] != v:
 811                     i += 1
 812                 if i == len(b):
 813                     return False
 814                 i += 1
 815             return True
 816     elif datatype == 'dict':
 817         if method == 'strict':
 818             return a == b
 819         else:
 820             return compare_dict_allow_more_present(a, b)
 821     elif datatype == 'set':
 822         set_a = set(a)
 823         set_b = set(b)
 824         if method == 'strict':
 825             return set_a == set_b
 826         else:
 827             return set_b >= set_a
 828     elif datatype == 'set(dict)':
 829         for av in a:
 830             found = False
 831             for bv in b:
 832                 if compare_dict_allow_more_present(av, bv):
 833                     found = True
 834                     break
 835             if not found:
 836                 return False
 837         if method == 'strict':
 838             # If we would know that both a and b do not contain duplicates,
 839             # we could simply compare len(a) to len(b) to finish this test.
 840             # We can assume that b has no duplicates (as it is returned by
 841             # docker), but we don't know for a.
 842             for bv in b:
 843                 found = False
 844                 for av in a:
 845                     if compare_dict_allow_more_present(av, bv):
 846                         found = True
 847                         break
 848                 if not found:
 849                     return False
 850         return True

 910 def clean_dict_booleans_for_docker_api(data):
 911     '''
 912     Go doesn't like Python booleans 'True' or 'False', while Ansible is just
 913     fine with them in YAML. As such, they need to be converted in cases where
 914     we pass dictionaries to the Docker API (e.g. docker_network's
 915     driver_options and docker_prune's filters).
 916     '''
 917     result = dict()
 918     if data is not None:
 919         for k, v in data.items():
 920             if v is True:
 921                 v = 'true'
 922             elif v is False:
 923                 v = 'false'
 924             else:
 925                 v = str(v)
 926             result[str(k)] = v
 927     return result

 930 def convert_duration_to_nanosecond(time_str):
 931     """
 932     Return time duration in nanosecond.
 933     """
 934     if not isinstance(time_str, str):
 935         raise ValueError('Missing unit in duration - %s' % time_str)
 936 
 937     regex = re.compile(
 938         r'^(((?P<hours>\d+)h)?'
 939         r'((?P<minutes>\d+)m(?!s))?'
 940         r'((?P<seconds>\d+)s)?'
 941         r'((?P<milliseconds>\d+)ms)?'
 942         r'((?P<microseconds>\d+)us)?)$'
 943     )
 944     parts = regex.match(time_str)
 945 
 946     if not parts:
 947         raise ValueError('Invalid time duration - %s' % time_str)
 948 
 949     parts = parts.groupdict()
 950     time_params = {}
 951     for (name, value) in parts.items():
 952         if value:
 953             time_params[name] = int(value)
 954 
 955     delta = timedelta(**time_params)
 956     time_in_nanoseconds = (
 957         delta.microseconds + (delta.seconds + delta.days * 24 * 3600) * 10 ** 6
 958     ) * 10 ** 3
 959 
 960     return time_in_nanoseconds

 963 def parse_healthcheck(healthcheck):
 964     """
 965     Return dictionary of healthcheck parameters and boolean if
 966     healthcheck defined in image was requested to be disabled.
 967     """
 968     if (not healthcheck) or (not healthcheck.get('test')):
 969         return None, None
 970 
 971     result = dict()
 972 
 973     # All supported healthcheck parameters
 974     options = dict(
 975         test='test',
 976         interval='interval',
 977         timeout='timeout',
 978         start_period='start_period',
 979         retries='retries'
 980     )
 981 
 982     duration_options = ['interval', 'timeout', 'start_period']
 983 
 984     for (key, value) in options.items():
 985         if value in healthcheck:
 986             if healthcheck.get(value) is None:
 987                 # due to recursive argument_spec, all keys are always present
 988                 # (but have default value None if not specified)
 989                 continue
 990             if value in duration_options:
 991                 time = convert_duration_to_nanosecond(healthcheck.get(value))
 992                 if time:
 993                     result[key] = time
 994             elif healthcheck.get(value):
 995                 result[key] = healthcheck.get(value)
 996                 if key == 'test':
 997                     if isinstance(result[key], (tuple, list)):
 998                         result[key] = [str(e) for e in result[key]]
 999                     else:
1000                         result[key] = ['CMD-SHELL', str(result[key])]
1001                 elif key == 'retries':
1002                     try:
1003                         result[key] = int(result[key])
1004                     except ValueError:
1005                         raise ValueError(
1006                             'Cannot parse number of retries for healthcheck. '
1007                             'Expected an integer, got "{0}".'.format(result[key])
1008                         )
1009 
1010     if result['test'] == ['NONE']:
1011         # If the user explicitly disables the healthcheck, return None
1012         # as the healthcheck object, and set disable_healthcheck to True
1013         return None, True
1014 
1015     return result, False
