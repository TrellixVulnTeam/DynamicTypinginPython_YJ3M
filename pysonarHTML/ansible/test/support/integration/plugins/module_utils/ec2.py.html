  91     @staticmethod

  95     @staticmethod
  96     def found(response_code, catch_extra_error_codes=None):
  97         # This list of failures is based on this API Reference
  98         # http://docs.aws.amazon.com/AWSEC2/latest/APIReference/errors-overview.html
  99         #
 100         # TooManyRequestsException comes from inside botocore when it
 101         # does retrys, unfortunately however it does not try long
 102         # enough to allow some services such as API Gateway to
 103         # complete configuration.  At the moment of writing there is a
 104         # botocore/boto3 bug open to fix this.
 105         #
 106         # https://github.com/boto/boto3/issues/876 (and linked PRs etc)
 107         retry_on = [
 108             'RequestLimitExceeded', 'Unavailable', 'ServiceUnavailable',
 109             'InternalFailure', 'InternalError', 'TooManyRequestsException',
 110             'Throttling'
 111         ]
 112         if catch_extra_error_codes:
 113             retry_on.extend(catch_extra_error_codes)
 114 
 115         return response_code in retry_on

 118 def boto3_conn(module, conn_type=None, resource=None, region=None, endpoint=None, **params):
 119     try:
 120         return _boto3_conn(conn_type=conn_type, resource=resource, region=region, endpoint=endpoint, **params)
 121     except ValueError as e:
 122         module.fail_json(msg="Couldn't connect to AWS: %s" % to_native(e))
 123     except (botocore.exceptions.ProfileNotFound, botocore.exceptions.PartialCredentialsError,
 124             botocore.exceptions.NoCredentialsError, botocore.exceptions.ConfigParseError) as e:
 125         module.fail_json(msg=to_native(e))
 126     except botocore.exceptions.NoRegionError as e:
 127         module.fail_json(msg="The %s module requires a region and none was found in configuration, "
 128                          "environment variables or module parameters" % module._name)

 131 def _boto3_conn(conn_type=None, resource=None, region=None, endpoint=None, **params):
 132     profile = params.pop('profile_name', None)
 133 
 134     if conn_type not in ['both', 'resource', 'client']:
 135         raise ValueError('There is an issue in the calling code. You '
 136                          'must specify either both, resource, or client to '
 137                          'the conn_type parameter in the boto3_conn function '
 138                          'call')
 139 
 140     config = botocore.config.Config(
 141         user_agent_extra='Ansible/{0}'.format(__version__),
 142     )
 143 
 144     if params.get('config') is not None:
 145         config = config.merge(params.pop('config'))
 146     if params.get('aws_config') is not None:
 147         config = config.merge(params.pop('aws_config'))
 148 
 149     session = boto3.session.Session(
 150         profile_name=profile,
 151     )
 152 
 153     if conn_type == 'resource':
 154         return session.resource(resource, config=config, region_name=region, endpoint_url=endpoint, **params)
 155     elif conn_type == 'client':
 156         return session.client(resource, config=config, region_name=region, endpoint_url=endpoint, **params)
 157     else:
 158         client = session.client(resource, region_name=region, endpoint_url=endpoint, **params)
 159         resource = session.resource(resource, region_name=region, endpoint_url=endpoint, **params)
 160         return client, resource

 166 def boto_exception(err):
 167     """
 168     Extracts the error message from a boto exception.
 169 
 170     :param err: Exception from boto
 171     :return: Error message
 172     """
 173     if hasattr(err, 'error_message'):
 174         error = err.error_message
 175     elif hasattr(err, 'message'):
 176         error = str(err.message) + ' ' + str(err) + ' - ' + str(type(err))
 177     else:
 178         error = '%s: %s' % (Exception, err)
 179 
 180     return error

 206 def get_aws_region(module, boto3=False):
 207     region = module.params.get('region')
 208 
 209     if region:
 210         return region
 211 
 212     if 'AWS_REGION' in os.environ:
 213         return os.environ['AWS_REGION']
 214     if 'AWS_DEFAULT_REGION' in os.environ:
 215         return os.environ['AWS_DEFAULT_REGION']
 216     if 'EC2_REGION' in os.environ:
 217         return os.environ['EC2_REGION']
 218 
 219     if not boto3:
 220         if not HAS_BOTO:
 221             module.fail_json(msg=missing_required_lib('boto'), exception=BOTO_IMP_ERR)
 222         # boto.config.get returns None if config not found
 223         region = boto.config.get('Boto', 'aws_region')
 224         if region:
 225             return region
 226         return boto.config.get('Boto', 'ec2_region')
 227 
 228     if not HAS_BOTO3:
 229         module.fail_json(msg=missing_required_lib('boto3'), exception=BOTO3_IMP_ERR)
 230 
 231     # here we don't need to make an additional call, will default to 'us-east-1' if the below evaluates to None.
 232     try:
 233         profile_name = module.params.get('profile')
 234         return botocore.session.Session(profile=profile_name).get_config_variable('region')
 235     except botocore.exceptions.ProfileNotFound as e:
 236         return None

 239 def get_aws_connection_info(module, boto3=False):
 240 
 241     # Check module args for credentials, then check environment vars
 242     # access_key
 243 
 244     ec2_url = module.params.get('ec2_url')
 245     access_key = module.params.get('aws_access_key')
 246     secret_key = module.params.get('aws_secret_key')
 247     security_token = module.params.get('security_token')
 248     region = get_aws_region(module, boto3)
 249     profile_name = module.params.get('profile')
 250     validate_certs = module.params.get('validate_certs')
 251     config = module.params.get('aws_config')
 252 
 253     if not ec2_url:
 254         if 'AWS_URL' in os.environ:
 255             ec2_url = os.environ['AWS_URL']
 256         elif 'EC2_URL' in os.environ:
 257             ec2_url = os.environ['EC2_URL']
 258 
 259     if not access_key:
 260         if os.environ.get('AWS_ACCESS_KEY_ID'):
 261             access_key = os.environ['AWS_ACCESS_KEY_ID']
 262         elif os.environ.get('AWS_ACCESS_KEY'):
 263             access_key = os.environ['AWS_ACCESS_KEY']
 264         elif os.environ.get('EC2_ACCESS_KEY'):
 265             access_key = os.environ['EC2_ACCESS_KEY']
 266         elif HAS_BOTO and boto.config.get('Credentials', 'aws_access_key_id'):
 267             access_key = boto.config.get('Credentials', 'aws_access_key_id')
 268         elif HAS_BOTO and boto.config.get('default', 'aws_access_key_id'):
 269             access_key = boto.config.get('default', 'aws_access_key_id')
 270         else:
 271             # in case access_key came in as empty string
 272             access_key = None
 273 
 274     if not secret_key:
 275         if os.environ.get('AWS_SECRET_ACCESS_KEY'):
 276             secret_key = os.environ['AWS_SECRET_ACCESS_KEY']
 277         elif os.environ.get('AWS_SECRET_KEY'):
 278             secret_key = os.environ['AWS_SECRET_KEY']
 279         elif os.environ.get('EC2_SECRET_KEY'):
 280             secret_key = os.environ['EC2_SECRET_KEY']
 281         elif HAS_BOTO and boto.config.get('Credentials', 'aws_secret_access_key'):
 282             secret_key = boto.config.get('Credentials', 'aws_secret_access_key')
 283         elif HAS_BOTO and boto.config.get('default', 'aws_secret_access_key'):
 284             secret_key = boto.config.get('default', 'aws_secret_access_key')
 285         else:
 286             # in case secret_key came in as empty string
 287             secret_key = None
 288 
 289     if not security_token:
 290         if os.environ.get('AWS_SECURITY_TOKEN'):
 291             security_token = os.environ['AWS_SECURITY_TOKEN']
 292         elif os.environ.get('AWS_SESSION_TOKEN'):
 293             security_token = os.environ['AWS_SESSION_TOKEN']
 294         elif os.environ.get('EC2_SECURITY_TOKEN'):
 295             security_token = os.environ['EC2_SECURITY_TOKEN']
 296         elif HAS_BOTO and boto.config.get('Credentials', 'aws_security_token'):
 297             security_token = boto.config.get('Credentials', 'aws_security_token')
 298         elif HAS_BOTO and boto.config.get('default', 'aws_security_token'):
 299             security_token = boto.config.get('default', 'aws_security_token')
 300         else:
 301             # in case secret_token came in as empty string
 302             security_token = None
 303 
 304     if HAS_BOTO3 and boto3:
 305         boto_params = dict(aws_access_key_id=access_key,
 306                            aws_secret_access_key=secret_key,
 307                            aws_session_token=security_token)
 308         boto_params['verify'] = validate_certs
 309 
 310         if profile_name:
 311             boto_params = dict(aws_access_key_id=None, aws_secret_access_key=None, aws_session_token=None)
 312             boto_params['profile_name'] = profile_name
 313 
 314     else:
 315         boto_params = dict(aws_access_key_id=access_key,
 316                            aws_secret_access_key=secret_key,
 317                            security_token=security_token)
 318 
 319         # only set profile_name if passed as an argument
 320         if profile_name:
 321             boto_params['profile_name'] = profile_name
 322 
 323         boto_params['validate_certs'] = validate_certs
 324 
 325     if config is not None:
 326         if HAS_BOTO3 and boto3:
 327             boto_params['aws_config'] = botocore.config.Config(**config)
 328         elif HAS_BOTO and not boto3:
 329             if 'user_agent' in config:
 330                 sys.modules["boto.connection"].UserAgent = config['user_agent']
 331 
 332     for param, value in boto_params.items():
 333         if isinstance(value, binary_type):
 334             boto_params[param] = text_type(value, 'utf-8', 'strict')
 335 
 336     return region, ec2_url, boto_params

 354 def connect_to_aws(aws_module, region, **params):
 355     try:
 356         conn = aws_module.connect_to_region(region, **params)
 357     except(boto.provider.ProfileNotFoundError):
 358         raise AnsibleAWSError("Profile given for AWS was not found.  Please fix and retry.")
 359     if not conn:
 360         if region not in [aws_module_region.name for aws_module_region in aws_module.regions()]:
 361             raise AnsibleAWSError("Region %s does not seem to be available for aws module %s. If the region definitely exists, you may need to upgrade "
 362                                   "boto or extend with endpoints_path" % (region, aws_module.__name__))
 363         else:
 364             raise AnsibleAWSError("Unknown problem connecting to region %s for aws module %s." % (region, aws_module.__name__))
 365     if params.get('profile_name'):
 366         conn = boto_fix_security_token_in_profile(conn, params['profile_name'])
 367     return conn

 370 def ec2_connect(module):
 371 
 372     """ Return an ec2 connection"""
 373 
 374     region, ec2_url, boto_params = get_aws_connection_info(module)
 375 
 376     # If we have a region specified, connect to its endpoint.
 377     if region:
 378         try:
 379             ec2 = connect_to_aws(boto.ec2, region, **boto_params)
 380         except (boto.exception.NoAuthHandlerFound, AnsibleAWSError, boto.provider.ProfileNotFoundError) as e:
 381             module.fail_json(msg=str(e))
 382     # Otherwise, no region so we fallback to the old connection method
 383     elif ec2_url:
 384         try:
 385             ec2 = boto.connect_ec2_endpoint(ec2_url, **boto_params)
 386         except (boto.exception.NoAuthHandlerFound, AnsibleAWSError, boto.provider.ProfileNotFoundError) as e:
 387             module.fail_json(msg=str(e))
 388     else:
 389         module.fail_json(msg="Either region or ec2_url must be specified")
 390 
 391     return ec2

 394 def ansible_dict_to_boto3_filter_list(filters_dict):
 395 
 396     """ Convert an Ansible dict of filters to list of dicts that boto3 can use
 397     Args:
 398         filters_dict (dict): Dict of AWS filters.
 399     Basic Usage:
 400         >>> filters = {'some-aws-id': 'i-01234567'}
 401         >>> ansible_dict_to_boto3_filter_list(filters)
 402         {
 403             'some-aws-id': 'i-01234567'
 404         }
 405     Returns:
 406         List: List of AWS filters and their values
 407         [
 408             {
 409                 'Name': 'some-aws-id',
 410                 'Values': [
 411                     'i-01234567',
 412                 ]
 413             }
 414         ]
 415     """
 416 
 417     filters_list = []
 418     for k, v in filters_dict.items():
 419         filter_dict = {'Name': k}
 420         if isinstance(v, string_types):
 421             filter_dict['Values'] = [v]
 422         else:
 423             filter_dict['Values'] = v
 424 
 425         filters_list.append(filter_dict)
 426 
 427     return filters_list

 430 def boto3_tag_list_to_ansible_dict(tags_list, tag_name_key_name=None, tag_value_key_name=None):
 431 
 432     """ Convert a boto3 list of resource tags to a flat dict of key:value pairs
 433     Args:
 434         tags_list (list): List of dicts representing AWS tags.
 435         tag_name_key_name (str): Value to use as the key for all tag keys (useful because boto3 doesn't always use "Key")
 436         tag_value_key_name (str): Value to use as the key for all tag values (useful because boto3 doesn't always use "Value")
 437     Basic Usage:
 438         >>> tags_list = [{'Key': 'MyTagKey', 'Value': 'MyTagValue'}]
 439         >>> boto3_tag_list_to_ansible_dict(tags_list)
 440         [
 441             {
 442                 'Key': 'MyTagKey',
 443                 'Value': 'MyTagValue'
 444             }
 445         ]
 446     Returns:
 447         Dict: Dict of key:value pairs representing AWS tags
 448          {
 449             'MyTagKey': 'MyTagValue',
 450         }
 451     """
 452 
 453     if tag_name_key_name and tag_value_key_name:
 454         tag_candidates = {tag_name_key_name: tag_value_key_name}
 455     else:
 456         tag_candidates = {'key': 'value', 'Key': 'Value'}
 457 
 458     if not tags_list:
 459         return {}
 460     for k, v in tag_candidates.items():
 461         if k in tags_list[0] and v in tags_list[0]:
 462             return dict((tag[k], tag[v]) for tag in tags_list)
 463     raise ValueError("Couldn't find tag key (candidates %s) in tag list %s" % (str(tag_candidates), str(tags_list)))

 466 def ansible_dict_to_boto3_tag_list(tags_dict, tag_name_key_name='Key', tag_value_key_name='Value'):
 467 
 468     """ Convert a flat dict of key:value pairs representing AWS resource tags to a boto3 list of dicts
 469     Args:
 470         tags_dict (dict): Dict representing AWS resource tags.
 471         tag_name_key_name (str): Value to use as the key for all tag keys (useful because boto3 doesn't always use "Key")
 472         tag_value_key_name (str): Value to use as the key for all tag values (useful because boto3 doesn't always use "Value")

 496 def get_ec2_security_group_ids_from_names(sec_group_list, ec2_connection, vpc_id=None, boto3=True):
 497 
 498     """ Return list of security group IDs from security group names. Note that security group names are not unique
 499      across VPCs.  If a name exists across multiple VPCs and no VPC ID is supplied, all matching IDs will be returned. This
 500      will probably lead to a boto exception if you attempt to assign both IDs to a resource so ensure you wrap the call in
 501      a try block
 502      """
 503 
 504     def get_sg_name(sg, boto3):
 505 
 506         if boto3:
 507             return sg['GroupName']
 508         else:
 509             return sg.name
 510 
 511     def get_sg_id(sg, boto3):
 512 
 513         if boto3:
 514             return sg['GroupId']
 515         else:
 516             return sg.id
 517 
 518     sec_group_id_list = []
 519 
 520     if isinstance(sec_group_list, string_types):
 521         sec_group_list = [sec_group_list]
 522 
 523     # Get all security groups
 524     if boto3:
 525         if vpc_id:
 526             filters = [
 527                 {
 528                     'Name': 'vpc-id',
 529                     'Values': [
 530                         vpc_id,
 531                     ]
 532                 }
 533             ]
 534             all_sec_groups = ec2_connection.describe_security_groups(Filters=filters)['SecurityGroups']
 535         else:
 536             all_sec_groups = ec2_connection.describe_security_groups()['SecurityGroups']
 537     else:
 538         if vpc_id:
 539             filters = {'vpc-id': vpc_id}
 540             all_sec_groups = ec2_connection.get_all_security_groups(filters=filters)
 541         else:
 542             all_sec_groups = ec2_connection.get_all_security_groups()
 543 
 544     unmatched = set(sec_group_list).difference(str(get_sg_name(all_sg, boto3)) for all_sg in all_sec_groups)
 545     sec_group_name_list = list(set(sec_group_list) - set(unmatched))
 546 
 547     if len(unmatched) > 0:
 548         # If we have unmatched names that look like an ID, assume they are
 549         import re
 550         sec_group_id_list = [sg for sg in unmatched if re.match('sg-[a-fA-F0-9]+$', sg)]
 551         still_unmatched = [sg for sg in unmatched if not re.match('sg-[a-fA-F0-9]+$', sg)]
 552         if len(still_unmatched) > 0:
 553             raise ValueError("The following group names are not valid: %s" % ', '.join(still_unmatched))
 554 
 555     sec_group_id_list += [str(get_sg_id(all_sg, boto3)) for all_sg in all_sec_groups if str(get_sg_name(all_sg, boto3)) in sec_group_name_list]
 556 
 557     return sec_group_id_list

 560 def _hashable_policy(policy, policy_list):
 561     """
 562         Takes a policy and returns a list, the contents of which are all hashable and sorted.
 563         Example input policy:
 564         {'Version': '2012-10-17',
 565          'Statement': [{'Action': 's3:PutObjectAcl',
 566                         'Sid': 'AddCannedAcl2',
 567                         'Resource': 'arn:aws:s3:::test_policy/*',
 568                         'Effect': 'Allow',
 569                         'Principal': {'AWS': ['arn:aws:iam::XXXXXXXXXXXX:user/username1', 'arn:aws:iam::XXXXXXXXXXXX:user/username2']}
 570                        }]}
 571         Returned value:
 572         [('Statement',  ((('Action', (u's3:PutObjectAcl',)),
 573                           ('Effect', (u'Allow',)),
 574                           ('Principal', ('AWS', ((u'arn:aws:iam::XXXXXXXXXXXX:user/username1',), (u'arn:aws:iam::XXXXXXXXXXXX:user/username2',)))),
 575                           ('Resource', (u'arn:aws:s3:::test_policy/*',)), ('Sid', (u'AddCannedAcl2',)))),
 576          ('Version', (u'2012-10-17',)))]
 577 
 578     """
 579     # Amazon will automatically convert bool and int to strings for us
 580     if isinstance(policy, bool):
 581         return tuple([str(policy).lower()])
 582     elif isinstance(policy, int):
 583         return tuple([str(policy)])
 584 
 585     if isinstance(policy, list):
 586         for each in policy:
 587             tupleified = _hashable_policy(each, [])
 588             if isinstance(tupleified, list):
 589                 tupleified = tuple(tupleified)
 590             policy_list.append(tupleified)
 591     elif isinstance(policy, string_types) or isinstance(policy, binary_type):
 592         policy = to_text(policy)
 593         # convert root account ARNs to just account IDs
 594         if policy.startswith('arn:aws:iam::') and policy.endswith(':root'):
 595             policy = policy.split(':')[4]
 596         return [policy]
 597     elif isinstance(policy, dict):
 598         sorted_keys = list(policy.keys())
 599         sorted_keys.sort()
 600         for key in sorted_keys:
 601             tupleified = _hashable_policy(policy[key], [])
 602             if isinstance(tupleified, list):
 603                 tupleified = tuple(tupleified)
 604             policy_list.append((key, tupleified))
 605 
 606     # ensure we aren't returning deeply nested structures of length 1
 607     if len(policy_list) == 1 and isinstance(policy_list[0], tuple):
 608         policy_list = policy_list[0]
 609     if isinstance(policy_list, list):
 610         if PY3_COMPARISON:
 611             policy_list.sort(key=cmp_to_key(py3cmp))
 612         else:
 613             policy_list.sort()
 614     return policy_list

 617 def py3cmp(a, b):
 618     """ Python 2 can sort lists of mixed types. Strings < tuples. Without this function this fails on Python 3."""
 619     try:
 620         if a > b:
 621             return 1
 622         elif a < b:
 623             return -1
 624         else:
 625             return 0
 626     except TypeError as e:
 627         # check to see if they're tuple-string
 628         # always say strings are less than tuples (to maintain compatibility with python2)
 629         str_ind = to_text(e).find('str')
 630         tup_ind = to_text(e).find('tuple')
 631         if -1 not in (str_ind, tup_ind):
 632             if str_ind < tup_ind:
 633                 return -1
 634             elif tup_ind < str_ind:
 635                 return 1
 636         raise

 646 def sort_json_policy_dict(policy_dict):
 647 
 648     """ Sort any lists in an IAM JSON policy so that comparison of two policies with identical values but
 649     different orders will return true
 650     Args:
 651         policy_dict (dict): Dict representing IAM JSON policy.
 652     Basic Usage:
 653         >>> my_iam_policy = {'Principle': {'AWS':["31","7","14","101"]}
 654         >>> sort_json_policy_dict(my_iam_policy)
 655     Returns:
 656         Dict: Will return a copy of the policy as a Dict but any List will be sorted
 657         {
 658             'Principle': {
 659                 'AWS': [ '7', '14', '31', '101' ]
 660             }
 661         }
 662     """
 663 
 664     def value_is_list(my_list):
 665 
 666         checked_list = []
 667         for item in my_list:
 668             if isinstance(item, dict):
 669                 checked_list.append(sort_json_policy_dict(item))
 670             elif isinstance(item, list):
 671                 checked_list.append(value_is_list(item))
 672             else:
 673                 checked_list.append(item)
 674 
 675         # Sort list. If it's a list of dictionaries, sort by tuple of key-value
 676         # pairs, since Python 3 doesn't allow comparisons such as `<` between dictionaries.
 677         checked_list.sort(key=lambda x: sorted(x.items()) if isinstance(x, dict) else x)
 678         return checked_list
 679 
 680     ordered_policy_dict = {}
 681     for key, value in policy_dict.items():
 682         if isinstance(value, dict):
 683             ordered_policy_dict[key] = sort_json_policy_dict(value)
 684         elif isinstance(value, list):
 685             ordered_policy_dict[key] = value_is_list(value)
 686         else:
 687             ordered_policy_dict[key] = value
 688 
 689     return ordered_policy_dict

 692 def map_complex_type(complex_type, type_map):
 693     """
 694         Allows to cast elements within a dictionary to a specific type
 695         Example of usage:
 696 
 697         DEPLOYMENT_CONFIGURATION_TYPE_MAP = {
 698             'maximum_percent': 'int',
 699             'minimum_healthy_percent': 'int'
 700         }
 701 
 702         deployment_configuration = map_complex_type(module.params['deployment_configuration'],
 703                                                     DEPLOYMENT_CONFIGURATION_TYPE_MAP)
 704 
 705         This ensures all keys within the root element are casted and valid integers
 706     """
 707 
 708     if complex_type is None:
 709         return
 710     new_type = type(complex_type)()
 711     if isinstance(complex_type, dict):
 712         for key in complex_type:
 713             if key in type_map:
 714                 if isinstance(type_map[key], list):
 715                     new_type[key] = map_complex_type(
 716                         complex_type[key],
 717                         type_map[key][0])
 718                 else:
 719                     new_type[key] = map_complex_type(
 720                         complex_type[key],
 721                         type_map[key])
 722             else:
 723                 return complex_type
 724     elif isinstance(complex_type, list):
 725         for i in range(len(complex_type)):
 726             new_type.append(map_complex_type(
 727                 complex_type[i],
 728                 type_map))
 729     elif type_map:
 730         return globals()['__builtins__'][type_map](complex_type)
 731     return new_type

 734 def compare_aws_tags(current_tags_dict, new_tags_dict, purge_tags=True):
 735     """
 736     Compare two dicts of AWS tags. Dicts are expected to of been created using 'boto3_tag_list_to_ansible_dict' helper function.
 737     Two dicts are returned - the first is tags to be set, the second is any tags to remove. Since the AWS APIs differ
 738     these may not be able to be used out of the box.
 739 
 740     :param current_tags_dict:
 741     :param new_tags_dict:
 742     :param purge_tags:
 743     :return: tag_key_value_pairs_to_set: a dict of key value pairs that need to be set in AWS. If all tags are identical this dict will be empty
 744     :return: tag_keys_to_unset: a list of key names (type str) that need to be unset in AWS. If no tags need to be unset this list will be empty
 745     """
 746 
 747     tag_key_value_pairs_to_set = {}
 748     tag_keys_to_unset = []
 749 
 750     for key in current_tags_dict.keys():
 751         if key not in new_tags_dict and purge_tags:
 752             tag_keys_to_unset.append(key)
 753 
 754     for key in set(new_tags_dict.keys()) - set(tag_keys_to_unset):
 755         if to_text(new_tags_dict[key]) != current_tags_dict.get(key):
 756             tag_key_value_pairs_to_set[key] = new_tags_dict[key]
 757 
 758     return tag_key_value_pairs_to_set, tag_keys_to_unset
