 400 def get_waiter(client, waiter_name):
 401     try:
 402         return waiters_by_name[(client.__class__.__name__, waiter_name)](client)
 403     except KeyError:
 404         raise NotImplementedError("Waiter {0} could not be found for client {1}. Available waiters: {2}".format(
 405             waiter_name, type(client), ', '.join(repr(k) for k in waiters_by_name.keys())))
