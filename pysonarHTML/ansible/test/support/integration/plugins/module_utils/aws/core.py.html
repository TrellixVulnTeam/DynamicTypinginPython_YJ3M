  19 """This module adds shared support for generic Amazon AWS modules
  20 
  21 **This code is not yet ready for use in user modules.  As of 2017**
  22 **and through to 2018, the interface is likely to change**
  23 **aggressively as the exact correct interface for ansible AWS modules**
  24 **is identified.  In particular, until this notice goes away or is**
  25 **changed, methods may disappear from the interface.  Please don't**
  26 **publish modules using this except directly to the main Ansible**
  27 **development repository.**
  28 
  29 In order to use this module, include it as part of a custom
  30 module as shown below.
  31 
  32   from ansible.module_utils.aws import AnsibleAWSModule
  33   module = AnsibleAWSModule(argument_spec=dictionary, supports_check_mode=boolean
  34                             mutually_exclusive=list1, required_together=list2)
  35 
  36 The 'AnsibleAWSModule' module provides similar, but more restricted,
  37 interfaces to the normal Ansible module.  It also includes the
  38 additional methods for connecting to AWS using the standard module arguments
  39 
  40     m.resource('lambda') # - get an AWS connection as a boto3 resource.
  41 
  42 or
  43 
  44     m.client('sts') # - get an AWS connection as a boto3 client.
  45 
  46 To make use of AWSRetry easier, it can now be wrapped around any call from a
  47 module-created client. To add retries to a client, create a client:
  48 
  49     m.client('ec2', retry_decorator=AWSRetry.jittered_backoff(retries=10))
  50 
  51 Any calls from that client can be made to use the decorator passed at call-time
  52 using the `aws_retry` argument. By default, no retries are used.
  53 
  54     ec2 = m.client('ec2', retry_decorator=AWSRetry.jittered_backoff(retries=10))
  55     ec2.describe_instances(InstanceIds=['i-123456789'], aws_retry=True)
  56 
  57 The call will be retried the specified number of times, so the calling functions
  58 don't need to be wrapped in the backoff decorator.
  59 """

  85 class AnsibleAWSModule(object):
  86     """An ansible module class for AWS modules
  87 
  88     AnsibleAWSModule provides an a class for building modules which
  89     connect to Amazon Web Services.  The interface is currently more
  90     restricted than the basic module class with the aim that later the
  91     basic module class can be reduced.  If you find that any key
  92     feature is missing please contact the author/Ansible AWS team
  93     (available on #ansible-aws on IRC) to request the additional
  94     features needed.
  95     """
  96     default_settings = {
  97         "default_args": True,
  98         "check_boto3": True,
  99         "auto_retry": True,
 100         "module_class": AnsibleModule
 101     }
 102 
 103     def __init__(self, **kwargs):
 104         local_settings = {}
 105         for key in AnsibleAWSModule.default_settings:
 106             try:
 107                 local_settings[key] = kwargs.pop(key)
 108             except KeyError:
 109                 local_settings[key] = AnsibleAWSModule.default_settings[key]
 110         self.settings = local_settings
 111 
 112         if local_settings["default_args"]:
 113             # ec2_argument_spec contains the region so we use that; there's a patch coming which
 114             # will add it to aws_argument_spec so if that's accepted then later we should change
 115             # over
 116             argument_spec_full = ec2_argument_spec()
 117             try:
 118                 argument_spec_full.update(kwargs["argument_spec"])
 119             except (TypeError, NameError):
 120                 pass
 121             kwargs["argument_spec"] = argument_spec_full
 122 
 123         self._module = AnsibleAWSModule.default_settings["module_class"](**kwargs)
 124 
 125         if local_settings["check_boto3"] and not HAS_BOTO3:
 126             self._module.fail_json(
 127                 msg=missing_required_lib('botocore or boto3'))
 128 
 129         self.check_mode = self._module.check_mode
 130         self._diff = self._module._diff
 131         self._name = self._module._name
 132 
 133         self._botocore_endpoint_log_stream = StringIO()
 134         self.logger = None
 135         if self.params.get('debug_botocore_endpoint_logs'):
 136             self.logger = logging.getLogger('botocore.endpoint')
 137             self.logger.setLevel(logging.DEBUG)
 138             self.logger.addHandler(logging.StreamHandler(self._botocore_endpoint_log_stream))
 139 
 140     @property
 141     def params(self):
 142         return self._module.params
 143 
 144     def _get_resource_action_list(self):
 145         actions = []
 146         for ln in self._botocore_endpoint_log_stream.getvalue().split('\n'):
 147             ln = ln.strip()
 148             if not ln:
 149                 continue
 150             found_operational_request = re.search(r"OperationModel\(name=.*?\)", ln)
 151             if found_operational_request:
 152                 operation_request = found_operational_request.group(0)[20:-1]
 153                 resource = re.search(r"https://.*?\.", ln).group(0)[8:-1]
 154                 actions.append("{0}:{1}".format(resource, operation_request))
 155         return list(set(actions))
 156 
 157     def exit_json(self, *args, **kwargs):
 158         if self.params.get('debug_botocore_endpoint_logs'):
 159             kwargs['resource_actions'] = self._get_resource_action_list()
 160         return self._module.exit_json(*args, **kwargs)
 161 
 162     def fail_json(self, *args, **kwargs):
 163         if self.params.get('debug_botocore_endpoint_logs'):
 164             kwargs['resource_actions'] = self._get_resource_action_list()
 165         return self._module.fail_json(*args, **kwargs)
 166 
 167     def debug(self, *args, **kwargs):

 170     def warn(self, *args, **kwargs):

 173     def deprecate(self, *args, **kwargs):

 176     def boolean(self, *args, **kwargs):

 178 
 179     def md5(self, *args, **kwargs):

 182     def client(self, service, retry_decorator=None):
 183         region, ec2_url, aws_connect_kwargs = get_aws_connection_info(self, boto3=True)
 184         conn = boto3_conn(self, conn_type='client', resource=service,
 185                           region=region, endpoint=ec2_url, **aws_connect_kwargs)
 186         return conn if retry_decorator is None else _RetryingBotoClientWrapper(conn, retry_decorator)
 187 
 188     def resource(self, service):
 189         region, ec2_url, aws_connect_kwargs = get_aws_connection_info(self, boto3=True)
 190         return boto3_conn(self, conn_type='resource', resource=service,
 191                           region=region, endpoint=ec2_url, **aws_connect_kwargs)

 194     def region(self, boto3=True):
 195         return get_aws_region(self, boto3)

 197     def fail_json_aws(self, exception, msg=None):
 198         """call fail_json with processed exception
 199 
 200         function for converting exceptions thrown by AWS SDK modules,
 201         botocore, boto3 and boto, into nice error messages.
 202         """
 203         last_traceback = traceback.format_exc()
 204 
 205         # to_native is trusted to handle exceptions that str() could
 206         # convert to text.
 207         try:
 208             except_msg = to_native(exception.message)
 209         except AttributeError:
 210             except_msg = to_native(exception)
 211 
 212         if msg is not None:
 213             message = '{0}: {1}'.format(msg, except_msg)
 214         else:
 215             message = except_msg
 216 
 217         try:
 218             response = exception.response
 219         except AttributeError:
 220             response = None
 221 
 222         failure = dict(
 223             msg=message,
 224             exception=last_traceback,
 225             **self._gather_versions()
 226         )
 227 
 228         if response is not None:
 229             failure.update(**camel_dict_to_snake_dict(response))
 230 
 231         self.fail_json(**failure)
 232 
 233     def _gather_versions(self):
 234         """Gather AWS SDK (boto3 and botocore) dependency versions
 235 
 236         Returns {'boto3_version': str, 'botocore_version': str}
 237         Returns {} if neither are installed
 238         """
 239         if not HAS_BOTO3:
 240             return {}
 241         import boto3
 242         import botocore
 243         return dict(boto3_version=boto3.__version__,
 244                     botocore_version=botocore.__version__)
 245 
 246     def boto3_at_least(self, desired):
 247         """Check if the available boto3 version is greater than or equal to a desired version.
 248 
 249         Usage:
 250             if module.params.get('assign_ipv6_address') and not module.boto3_at_least('1.4.4'):
 251                 # conditionally fail on old boto3 versions if a specific feature is not supported
 252                 module.fail_json(msg="Boto3 can't deal with EC2 IPv6 addresses before version 1.4.4.")
 253         """
 254         existing = self._gather_versions()
 255         return LooseVersion(existing['boto3_version']) >= LooseVersion(desired)
 256 
 257     def botocore_at_least(self, desired):
 258         """Check if the available botocore version is greater than or equal to a desired version.
 259 
 260         Usage:
 261             if not module.botocore_at_least('1.2.3'):
 262                 module.fail_json(msg='The Serverless Elastic Load Compute Service is not in botocore before v1.2.3')
 263             if not module.botocore_at_least('1.5.3'):
 264                 module.warn('Botocore did not include waiters for Service X before 1.5.3. '
 265                             'To wait until Service X resources are fully available, update botocore.')
 266         """
 267         existing = self._gather_versions()
 268         return LooseVersion(existing['botocore_version']) >= LooseVersion(desired)

 271 class _RetryingBotoClientWrapper(object):
 272     __never_wait = (
 273         'get_paginator', 'can_paginate',
 274         'get_waiter', 'generate_presigned_url',
 275     )
 276 
 277     def __init__(self, client, retry):
 278         self.client = client
 279         self.retry = retry
 280 
 281     def _create_optional_retry_wrapper_function(self, unwrapped):
 282         retrying_wrapper = self.retry(unwrapped)
 283 
 284         @wraps(unwrapped)
 285         def deciding_wrapper(aws_retry=False, *args, **kwargs):
 286             if aws_retry:
 287                 return retrying_wrapper(*args, **kwargs)
 288             else:
 289                 return unwrapped(*args, **kwargs)
 290         return deciding_wrapper
 291 
 292     def __getattr__(self, name):
 293         unwrapped = getattr(self.client, name)
 294         if name in self.__never_wait:
 295             return unwrapped
 296         elif callable(unwrapped):
 297             wrapped = self._create_optional_retry_wrapper_function(unwrapped)
 298             setattr(self, name, wrapped)
 299             return wrapped
 300         else:
 301             return unwrapped

 304 def is_boto3_error_code(code, e=None):
 305     """Check if the botocore exception is raised by a specific error code.
 306 
 307     Returns ClientError if the error code matches, a dummy exception if it does not have an error code or does not match
 308 
 309     Example:
 310     try:
 311         ec2.describe_instances(InstanceIds=['potato'])
 312     except is_boto3_error_code('InvalidInstanceID.Malformed'):
 313         # handle the error for that code case
 314     except botocore.exceptions.ClientError as e:
 315         # handle the generic error case for all other codes
 316     """
 317     from botocore.exceptions import ClientError
 318     if e is None:
 319         import sys
 320         dummy, e, dummy = sys.exc_info()
 321     if isinstance(e, ClientError) and e.response['Error']['Code'] == code:
 322         return ClientError
 323     return type('NeverEverRaisedException', (Exception,), {})

 326 def get_boto3_client_method_parameters(client, method_name, required=False):
 327     op = client.meta.method_to_api_mapping.get(method_name)
 328     input_shape = client._service_model.operation_model(op).input_shape
 329     if not input_shape:
 330         parameters = []
 331     elif required:
 332         parameters = list(input_shape.required_members)
 333     else:
 334         parameters = list(input_shape.members.keys())
 335     return parameters
