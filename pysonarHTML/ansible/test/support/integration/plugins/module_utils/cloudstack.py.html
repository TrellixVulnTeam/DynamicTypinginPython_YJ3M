 115     def get_api_config(self):
 116         api_region = self.module.params.get('api_region') or os.environ.get('CLOUDSTACK_REGION')
 117         try:
 118             config = read_config(api_region)
 119         except KeyError:
 120             config = {}
 121 
 122         api_config = {
 123             'endpoint': self.module.params.get('api_url') or config.get('endpoint'),
 124             'key': self.module.params.get('api_key') or config.get('key'),
 125             'secret': self.module.params.get('api_secret') or config.get('secret'),
 126             'timeout': self.module.params.get('api_timeout') or config.get('timeout') or 10,
 127             'method': self.module.params.get('api_http_method') or config.get('method') or 'get',
 128         }
 129         self.result.update({
 130             'api_region': api_region,
 131             'api_url': api_config['endpoint'],
 132             'api_key': api_config['key'],
 133             'api_timeout': int(api_config['timeout']),
 134             'api_http_method': api_config['method'],
 135         })
 136         if not all([api_config['endpoint'], api_config['key'], api_config['secret']]):
 137             self.fail_json(msg="Missing api credentials: can not authenticate")
 138         return api_config

 150     def has_changed(self, want_dict, current_dict, only_keys=None, skip_diff_for_keys=None):
 151         result = False
 152         for key, value in want_dict.items():
 153 
 154             # Optionally limit by a list of keys
 155             if only_keys and key not in only_keys:
 156                 continue
 157 
 158             # Skip None values
 159             if value is None:
 160                 continue
 161 
 162             if key in current_dict:
 163                 if isinstance(value, (int, float, long, complex)):
 164 
 165                     # ensure we compare the same type
 166                     if isinstance(value, int):
 167                         current_dict[key] = int(current_dict[key])
 168                     elif isinstance(value, float):
 169                         current_dict[key] = float(current_dict[key])
 170                     elif isinstance(value, long):
 171                         current_dict[key] = long(current_dict[key])
 172                     elif isinstance(value, complex):
 173                         current_dict[key] = complex(current_dict[key])
 174 
 175                     if value != current_dict[key]:
 176                         if skip_diff_for_keys and key not in skip_diff_for_keys:
 177                             self.result['diff']['before'][key] = current_dict[key]
 178                             self.result['diff']['after'][key] = value
 179                         result = True
 180                 else:
 181                     before_value = to_text(current_dict[key])
 182                     after_value = to_text(value)
 183 
 184                     if self.case_sensitive_keys and key in self.case_sensitive_keys:
 185                         if before_value != after_value:
 186                             if skip_diff_for_keys and key not in skip_diff_for_keys:
 187                                 self.result['diff']['before'][key] = before_value
 188                                 self.result['diff']['after'][key] = after_value
 189                             result = True
 190 
 191                     # Test for diff in case insensitive way
 192                     elif before_value.lower() != after_value.lower():
 193                         if skip_diff_for_keys and key not in skip_diff_for_keys:
 194                             self.result['diff']['before'][key] = before_value
 195                             self.result['diff']['after'][key] = after_value
 196                         result = True
 197             else:
 198                 if skip_diff_for_keys and key not in skip_diff_for_keys:
 199                     self.result['diff']['before'][key] = None
 200                     self.result['diff']['after'][key] = to_text(value)
 201                 result = True
 202         return result

 204     def _get_by_key(self, key=None, my_dict=None):
 205         if my_dict is None:
 206             my_dict = {}
 207         if key:
 208             if key in my_dict:
 209                 return my_dict[key]
 210             self.fail_json(msg="Something went wrong: %s not found" % key)
 211         return my_dict

 213     def query_api(self, command, **args):
 214         try:
 215             res = getattr(self.cs, command)(**args)
 216 
 217             if 'errortext' in res:
 218                 self.fail_json(msg="Failed: '%s'" % res['errortext'])
 219 
 220         except CloudStackException as e:
 221             self.fail_json(msg='CloudStackException: %s' % to_native(e))
 222 
 223         except Exception as e:
 224             self.fail_json(msg=to_native(e))
 225 
 226         return res

 243     def get_vpc(self, key=None):
 244         """Return a VPC dictionary or the value of given key of."""
 245         if self.vpc:
 246             return self._get_by_key(key, self.vpc)
 247 
 248         vpc = self.module.params.get('vpc')
 249         if not vpc:
 250             vpc = os.environ.get('CLOUDSTACK_VPC')
 251         if not vpc:
 252             return None
 253 
 254         args = {
 255             'account': self.get_account(key='name'),
 256             'domainid': self.get_domain(key='id'),
 257             'projectid': self.get_project(key='id'),
 258             'zoneid': self.get_zone(key='id'),
 259         }
 260         vpcs = self.query_api('listVPCs', **args)
 261         if not vpcs:
 262             self.fail_json(msg="No VPCs available.")
 263 
 264         for v in vpcs['vpc']:
 265             if vpc in [v['name'], v['displaytext'], v['id']]:
 266                 # Fail if the identifyer matches more than one VPC
 267                 if self.vpc:
 268                     self.fail_json(msg="More than one VPC found with the provided identifyer '%s'" % vpc)
 269                 else:
 270                     self.vpc = v
 271                     self.result['vpc'] = v['name']
 272         if self.vpc:
 273             return self._get_by_key(key, self.vpc)
 274         self.fail_json(msg="VPC '%s' not found" % vpc)

 276     def is_vpc_network(self, network_id):
 277         """Returns True if network is in VPC."""
 278         # This is an efficient way to query a lot of networks at a time
 279         if self._vpc_networks_ids is None:
 280             args = {
 281                 'account': self.get_account(key='name'),
 282                 'domainid': self.get_domain(key='id'),
 283                 'projectid': self.get_project(key='id'),
 284                 'zoneid': self.get_zone(key='id'),
 285             }
 286             vpcs = self.query_api('listVPCs', **args)
 287             self._vpc_networks_ids = []
 288             if vpcs:
 289                 for vpc in vpcs['vpc']:
 290                     for n in vpc.get('network', []):
 291                         self._vpc_networks_ids.append(n['id'])
 292         return network_id in self._vpc_networks_ids

 312     def get_network(self, key=None):
 313         """Return a network dictionary or the value of given key of."""
 314         if self.network:
 315             return self._get_by_key(key, self.network)
 316 
 317         network = self.module.params.get('network')
 318         if not network:
 319             vpc_name = self.get_vpc(key='name')
 320             if vpc_name:
 321                 self.fail_json(msg="Could not find network for VPC '%s' due missing argument: network" % vpc_name)
 322             return None
 323 
 324         args = {
 325             'account': self.get_account(key='name'),
 326             'domainid': self.get_domain(key='id'),
 327             'projectid': self.get_project(key='id'),
 328             'zoneid': self.get_zone(key='id'),
 329             'vpcid': self.get_vpc(key='id')
 330         }
 331         networks = self.query_api('listNetworks', **args)
 332         if not networks:
 333             self.fail_json(msg="No networks available.")
 334 
 335         for n in networks['network']:
 336             # ignore any VPC network if vpc param is not given
 337             if 'vpcid' in n and not self.get_vpc(key='id'):
 338                 continue
 339             if network in [n['displaytext'], n['name'], n['id']]:
 340                 self.result['network'] = n['name']
 341                 self.network = n
 342                 return self._get_by_key(key, self.network)
 343         self.fail_json(msg="Network '%s' not found" % network)

 391     def get_vm_guest_ip(self):
 392         vm_guest_ip = self.module.params.get('vm_guest_ip')
 393         default_nic = self.get_vm_default_nic()
 394 
 395         if not vm_guest_ip:
 396             return default_nic['ipaddress']
 397 
 398         for secondary_ip in default_nic['secondaryip']:
 399             if vm_guest_ip == secondary_ip['ipaddress']:
 400                 return vm_guest_ip
 401         self.fail_json(msg="Secondary IP '%s' not assigned to VM" % vm_guest_ip)

 567     def get_tags(self, resource=None, key='tags'):
 568         existing_tags = []
 569         for tag in resource.get(key) or []:
 570             existing_tags.append({'key': tag['key'], 'value': tag['value']})
 571         return existing_tags

 573     def _process_tags(self, resource, resource_type, tags, operation="create"):
 574         if tags:
 575             self.result['changed'] = True
 576             if not self.module.check_mode:
 577                 args = {
 578                     'resourceids': resource['id'],
 579                     'resourcetype': resource_type,
 580                     'tags': tags,
 581                 }
 582                 if operation == "create":
 583                     response = self.query_api('createTags', **args)
 584                 else:
 585                     response = self.query_api('deleteTags', **args)
 586                 self.poll_job(response)

 588     def _tags_that_should_exist_or_be_updated(self, resource, tags):
 589         existing_tags = self.get_tags(resource)
 590         return [tag for tag in tags if tag not in existing_tags]
 591 
 592     def _tags_that_should_not_exist(self, resource, tags):
 593         existing_tags = self.get_tags(resource)
 594         return [tag for tag in existing_tags if tag not in tags]

 597         if not resource_type or not resource:
 598             self.fail_json(msg="Error: Missing resource or resource_type for tags.")
 599 
 600         if 'tags' in resource:
 601             tags = self.module.params.get('tags')
 602             if tags is not None:
 603                 self._process_tags(resource, resource_type, self._tags_that_should_not_exist(resource, tags), operation="delete")
 604                 self._process_tags(resource, resource_type, self._tags_that_should_exist_or_be_updated(resource, tags))
 605                 resource['tags'] = self.query_tags(resource=resource, resource_type=resource_type)
 606         return resource

 615     def poll_job(self, job=None, key=None):
 616         if 'jobid' in job:
 617             while True:
 618                 res = self.query_api('queryAsyncJobResult', jobid=job['jobid'])
 619                 if res['jobstatus'] != 0 and 'jobresult' in res:
 620 
 621                     if 'errortext' in res['jobresult']:
 622                         self.fail_json(msg="Failed: '%s'" % res['jobresult']['errortext'])
 623 
 624                     if key and key in res['jobresult']:
 625                         job = res['jobresult'][key]
 626 
 627                     break
 628                 time.sleep(2)
 629         return job

 631     def update_result(self, resource, result=None):
 632         if result is None:
 633             result = dict()
 634         if resource:
 635             returns = self.common_returns.copy()
 636             returns.update(self.returns)
 637             for search_key, return_key in returns.items():
 638                 if search_key in resource:
 639                     result[return_key] = resource[search_key]
 640 
 641             # Bad bad API does not always return int when it should.
 642             for search_key, return_key in self.returns_to_int.items():
 643                 if search_key in resource:
 644                     result[return_key] = int(resource[search_key])
 645 
 646             if 'tags' in resource:
 647                 result['tags'] = resource['tags']
 648         return result

 650     def get_result(self, resource):

 653     def get_result_and_facts(self, facts_name, resource):
