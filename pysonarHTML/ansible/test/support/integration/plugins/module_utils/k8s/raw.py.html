  97     def __init__(self, k8s_kind=None, *args, **kwargs):
  98         self.client = None
  99         self.warnings = []
 100 
 101         mutually_exclusive = [
 102             ('resource_definition', 'src'),
 103             ('merge_type', 'apply'),
 104         ]
 105 
 106         KubernetesAnsibleModule.__init__(self, *args,
 107                                          mutually_exclusive=mutually_exclusive,
 108                                          supports_check_mode=True,
 109                                          **kwargs)
 110         self.kind = k8s_kind or self.params.get('kind')
 111         self.api_version = self.params.get('api_version')
 112         self.name = self.params.get('name')
 113         self.namespace = self.params.get('namespace')
 114         resource_definition = self.params.get('resource_definition')
 115         validate = self.params.get('validate')
 116         if validate:
 117             if LooseVersion(self.openshift_version) < LooseVersion("0.8.0"):
 118                 self.fail_json(msg="openshift >= 0.8.0 is required for validate")
 119         self.append_hash = self.params.get('append_hash')
 120         if self.append_hash:
 121             if not HAS_K8S_CONFIG_HASH:
 122                 self.fail_json(msg=missing_required_lib("openshift >= 0.7.2", reason="for append_hash"),
 123                                exception=K8S_CONFIG_HASH_IMP_ERR)
 124         if self.params['merge_type']:
 125             if LooseVersion(self.openshift_version) < LooseVersion("0.6.2"):
 126                 self.fail_json(msg=missing_required_lib("openshift >= 0.6.2", reason="for merge_type"))
 127         self.apply = self.params.get('apply', False)
 128         if self.apply:
 129             if not HAS_K8S_APPLY:
 130                 self.fail_json(msg=missing_required_lib("openshift >= 0.9.2", reason="for apply"))
 131 
 132         if resource_definition:
 133             if isinstance(resource_definition, string_types):
 134                 try:
 135                     self.resource_definitions = yaml.safe_load_all(resource_definition)
 136                 except (IOError, yaml.YAMLError) as exc:
 137                     self.fail(msg="Error loading resource_definition: {0}".format(exc))
 138             elif isinstance(resource_definition, list):
 139                 self.resource_definitions = resource_definition
 140             else:
 141                 self.resource_definitions = [resource_definition]
 142         src = self.params.get('src')
 143         if src:
 144             self.resource_definitions = self.load_resource_definitions(src)
 145         try:
 146             self.resource_definitions = [item for item in self.resource_definitions if item]
 147         except AttributeError:
 148             pass
 149 
 150         if not resource_definition and not src:
 151             implicit_definition = dict(
 152                 kind=self.kind,
 153                 apiVersion=self.api_version,
 154                 metadata=dict(name=self.name)
 155             )
 156             if self.namespace:
 157                 implicit_definition['metadata']['namespace'] = self.namespace
 158             self.resource_definitions = [implicit_definition]

 160     def flatten_list_kind(self, list_resource, definitions):
 161         flattened = []
 162         parent_api_version = list_resource.group_version if list_resource else None
 163         parent_kind = list_resource.kind[:-4] if list_resource else None
 164         for definition in definitions.get('items', []):
 165             resource = self.find_resource(definition.get('kind', parent_kind), definition.get('apiVersion', parent_api_version), fail=True)
 166             flattened.append((resource, self.set_defaults(resource, definition)))
 167         return flattened

 169     def execute_module(self):
 170         changed = False
 171         results = []
 172         try:
 173             self.client = self.get_api_client()
 174         # Hopefully the kubernetes client will provide its own exception class one day
 175         except (urllib3.exceptions.RequestError) as e:
 176             self.fail_json(msg="Couldn't connect to Kubernetes: %s" % str(e))
 177 
 178         flattened_definitions = []
 179         for definition in self.resource_definitions:
 180             kind = definition.get('kind', self.kind)
 181             api_version = definition.get('apiVersion', self.api_version)
 182             if kind.endswith('List'):
 183                 resource = self.find_resource(kind, api_version, fail=False)
 184                 flattened_definitions.extend(self.flatten_list_kind(resource, definition))
 185             else:
 186                 resource = self.find_resource(kind, api_version, fail=True)
 187                 flattened_definitions.append((resource, definition))
 188 
 189         for (resource, definition) in flattened_definitions:
 190             kind = definition.get('kind', self.kind)
 191             api_version = definition.get('apiVersion', self.api_version)
 192             definition = self.set_defaults(resource, definition)
 193             self.warnings = []
 194             if self.params['validate'] is not None:
 195                 self.warnings = self.validate(definition)
 196             result = self.perform_action(resource, definition)
 197             result['warnings'] = self.warnings
 198             changed = changed or result['changed']
 199             results.append(result)
 200 
 201         if len(results) == 1:
 202             self.exit_json(**results[0])
 203 
 204         self.exit_json(**{
 205             'changed': changed,
 206             'result': {
 207                 'results': results
 208             }
 209         })

 211     def validate(self, resource):
 212         def _prepend_resource_info(resource, msg):
 213             return "%s %s: %s" % (resource['kind'], resource['metadata']['name'], msg)
 214 
 215         try:
 216             warnings, errors = self.client.validate(resource, self.params['validate'].get('version'), self.params['validate'].get('strict'))
 217         except KubernetesValidateMissing:
 218             self.fail_json(msg="kubernetes-validate python library is required to validate resources")
 219 
 220         if errors and self.params['validate']['fail_on_error']:
 221             self.fail_json(msg="\n".join([_prepend_resource_info(resource, error) for error in errors]))
 222         else:
 223             return [_prepend_resource_info(resource, msg) for msg in warnings + errors]
 224 
 225     def set_defaults(self, resource, definition):
 226         definition['kind'] = resource.kind
 227         definition['apiVersion'] = resource.group_version
 228         metadata = definition.get('metadata', {})
 229         if self.name and not metadata.get('name'):
 230             metadata['name'] = self.name
 231         if resource.namespaced and self.namespace and not metadata.get('namespace'):
 232             metadata['namespace'] = self.namespace
 233         definition['metadata'] = metadata
 234         return definition

 236     def perform_action(self, resource, definition):
 237         result = {'changed': False, 'result': {}}
 238         state = self.params.get('state', None)
 239         force = self.params.get('force', False)
 240         name = definition['metadata'].get('name')
 241         namespace = definition['metadata'].get('namespace')
 242         existing = None
 243         wait = self.params.get('wait')
 244         wait_sleep = self.params.get('wait_sleep')
 245         wait_timeout = self.params.get('wait_timeout')
 246         wait_condition = None
 247         if self.params.get('wait_condition') and self.params['wait_condition'].get('type'):
 248             wait_condition = self.params['wait_condition']
 249 
 250         self.remove_aliases()
 251 
 252         try:
 253             # ignore append_hash for resources other than ConfigMap and Secret
 254             if self.append_hash and definition['kind'] in ['ConfigMap', 'Secret']:
 255                 name = '%s-%s' % (name, generate_hash(definition))
 256                 definition['metadata']['name'] = name
 257             params = dict(name=name)
 258             if namespace:
 259                 params['namespace'] = namespace
 260             existing = resource.get(**params)
 261         except NotFoundError:
 262             # Remove traceback so that it doesn't show up in later failures
 263             try:
 264                 sys.exc_clear()
 265             except AttributeError:
 266                 # no sys.exc_clear on python3
 267                 pass
 268         except ForbiddenError as exc:
 269             if definition['kind'] in ['Project', 'ProjectRequest'] and state != 'absent':
 270                 return self.create_project_request(definition)
 271             self.fail_json(msg='Failed to retrieve requested object: {0}'.format(exc.body),
 272                            error=exc.status, status=exc.status, reason=exc.reason)
 273         except DynamicApiError as exc:
 274             self.fail_json(msg='Failed to retrieve requested object: {0}'.format(exc.body),
 275                            error=exc.status, status=exc.status, reason=exc.reason)
 276 
 277         if state == 'absent':
 278             result['method'] = "delete"
 279             if not existing:
 280                 # The object already does not exist
 281                 return result
 282             else:
 283                 # Delete the object
 284                 result['changed'] = True
 285                 if not self.check_mode:
 286                     try:
 287                         k8s_obj = resource.delete(**params)
 288                         result['result'] = k8s_obj.to_dict()
 289                     except DynamicApiError as exc:
 290                         self.fail_json(msg="Failed to delete object: {0}".format(exc.body),
 291                                        error=exc.status, status=exc.status, reason=exc.reason)
 292                     if wait:
 293                         success, resource, duration = self.wait(resource, definition, wait_sleep, wait_timeout, 'absent')
 294                         result['duration'] = duration
 295                         if not success:
 296                             self.fail_json(msg="Resource deletion timed out", **result)
 297                 return result
 298         else:
 299             if self.apply:
 300                 if self.check_mode:
 301                     ignored, k8s_obj = apply_object(resource, definition)
 302                 else:
 303                     try:
 304                         k8s_obj = resource.apply(definition, namespace=namespace).to_dict()
 305                     except DynamicApiError as exc:
 306                         msg = "Failed to apply object: {0}".format(exc.body)
 307                         if self.warnings:
 308                             msg += "\n" + "\n    ".join(self.warnings)
 309                         self.fail_json(msg=msg, error=exc.status, status=exc.status, reason=exc.reason)
 310                 success = True
 311                 result['result'] = k8s_obj
 312                 if wait:
 313                     success, result['result'], result['duration'] = self.wait(resource, definition, wait_sleep, wait_timeout, condition=wait_condition)
 314                 if existing:
 315                     existing = existing.to_dict()
 316                 else:
 317                     existing = {}
 318                 match, diffs = self.diff_objects(existing, result['result'])
 319                 result['changed'] = not match
 320                 result['diff'] = diffs
 321                 result['method'] = 'apply'
 322                 if not success:
 323                     self.fail_json(msg="Resource apply timed out", **result)
 324                 return result
 325 
 326             if not existing:
 327                 if self.check_mode:
 328                     k8s_obj = definition
 329                 else:
 330                     try:
 331                         k8s_obj = resource.create(definition, namespace=namespace).to_dict()
 332                     except ConflictError:
 333                         # Some resources, like ProjectRequests, can't be created multiple times,
 334                         # because the resources that they create don't match their kind
 335                         # In this case we'll mark it as unchanged and warn the user
 336                         self.warn("{0} was not found, but creating it returned a 409 Conflict error. This can happen \
 337                                   if the resource you are creating does not directly create a resource of the same kind.".format(name))
 338                         return result
 339                     except DynamicApiError as exc:
 340                         msg = "Failed to create object: {0}".format(exc.body)
 341                         if self.warnings:
 342                             msg += "\n" + "\n    ".join(self.warnings)
 343                         self.fail_json(msg=msg, error=exc.status, status=exc.status, reason=exc.reason)
 344                 success = True
 345                 result['result'] = k8s_obj
 346                 if wait and not self.check_mode:
 347                     success, result['result'], result['duration'] = self.wait(resource, definition, wait_sleep, wait_timeout, condition=wait_condition)
 348                 result['changed'] = True
 349                 result['method'] = 'create'
 350                 if not success:
 351                     self.fail_json(msg="Resource creation timed out", **result)
 352                 return result
 353 
 354             match = False
 355             diffs = []
 356 
 357             if existing and force:
 358                 if self.check_mode:
 359                     k8s_obj = definition
 360                 else:
 361                     try:
 362                         k8s_obj = resource.replace(definition, name=name, namespace=namespace, append_hash=self.append_hash).to_dict()
 363                     except DynamicApiError as exc:
 364                         msg = "Failed to replace object: {0}".format(exc.body)
 365                         if self.warnings:
 366                             msg += "\n" + "\n    ".join(self.warnings)
 367                         self.fail_json(msg=msg, error=exc.status, status=exc.status, reason=exc.reason)
 368                 match, diffs = self.diff_objects(existing.to_dict(), k8s_obj)
 369                 success = True
 370                 result['result'] = k8s_obj
 371                 if wait:
 372                     success, result['result'], result['duration'] = self.wait(resource, definition, wait_sleep, wait_timeout, condition=wait_condition)
 373                 match, diffs = self.diff_objects(existing.to_dict(), result['result'])
 374                 result['changed'] = not match
 375                 result['method'] = 'replace'
 376                 result['diff'] = diffs
 377                 if not success:
 378                     self.fail_json(msg="Resource replacement timed out", **result)
 379                 return result
 380 
 381             # Differences exist between the existing obj and requested params
 382             if self.check_mode:
 383                 k8s_obj = dict_merge(existing.to_dict(), definition)
 384             else:
 385                 if LooseVersion(self.openshift_version) < LooseVersion("0.6.2"):
 386                     k8s_obj, error = self.patch_resource(resource, definition, existing, name,
 387                                                          namespace)
 388                 else:
 389                     for merge_type in self.params['merge_type'] or ['strategic-merge', 'merge']:
 390                         k8s_obj, error = self.patch_resource(resource, definition, existing, name,
 391                                                              namespace, merge_type=merge_type)
 392                         if not error:
 393                             break
 394                 if error:
 395                     self.fail_json(**error)
 396 
 397             success = True
 398             result['result'] = k8s_obj
 399             if wait:
 400                 success, result['result'], result['duration'] = self.wait(resource, definition, wait_sleep, wait_timeout, condition=wait_condition)
 401             match, diffs = self.diff_objects(existing.to_dict(), result['result'])
 402             result['changed'] = not match
 403             result['method'] = 'patch'
 404             result['diff'] = diffs
 405 
 406             if not success:
 407                 self.fail_json(msg="Resource update timed out", **result)
 408             return result

 410     def patch_resource(self, resource, definition, existing, name, namespace, merge_type=None):
 411         try:
 412             params = dict(name=name, namespace=namespace)
 413             if merge_type:
 414                 params['content_type'] = 'application/{0}-patch+json'.format(merge_type)
 415             k8s_obj = resource.patch(definition, **params).to_dict()
 416             match, diffs = self.diff_objects(existing.to_dict(), k8s_obj)
 417             error = {}
 418             return k8s_obj, {}
 419         except DynamicApiError as exc:
 420             msg = "Failed to patch object: {0}".format(exc.body)
 421             if self.warnings:
 422                 msg += "\n" + "\n    ".join(self.warnings)
 423             error = dict(msg=msg, error=exc.status, status=exc.status, reason=exc.reason, warnings=self.warnings)
 424             return None, error

 426     def create_project_request(self, definition):
 427         definition['kind'] = 'ProjectRequest'
 428         result = {'changed': False, 'result': {}}
 429         resource = self.find_resource('ProjectRequest', definition['apiVersion'], fail=True)
 430         if not self.check_mode:
 431             try:
 432                 k8s_obj = resource.create(definition)
 433                 result['result'] = k8s_obj.to_dict()
 434             except DynamicApiError as exc:
 435                 self.fail_json(msg="Failed to create object: {0}".format(exc.body),
 436                                error=exc.status, status=exc.status, reason=exc.reason)
 437         result['changed'] = True
 438         result['method'] = 'create'
 439         return result

 441     def _wait_for(self, resource, name, namespace, predicate, sleep, timeout, state):
 442         start = datetime.now()
 443 
 444         def _wait_for_elapsed():
 445             return (datetime.now() - start).seconds
 446 
 447         response = None
 448         while _wait_for_elapsed() < timeout:
 449             try:
 450                 response = resource.get(name=name, namespace=namespace)
 451                 if predicate(response):
 452                     if response:
 453                         return True, response.to_dict(), _wait_for_elapsed()
 454                     else:
 455                         return True, {}, _wait_for_elapsed()
 456                 time.sleep(sleep)
 457             except NotFoundError:
 458                 if state == 'absent':
 459                     return True, {}, _wait_for_elapsed()
 460         if response:
 461             response = response.to_dict()
 462         return False, response, _wait_for_elapsed()

 464     def wait(self, resource, definition, sleep, timeout, state='present', condition=None):
 465 
 466         def _deployment_ready(deployment):
 467             # FIXME: frustratingly bool(deployment.status) is True even if status is empty
 468             # Furthermore deployment.status.availableReplicas == deployment.status.replicas == None if status is empty
 469             return (deployment.status and deployment.status.replicas is not None and
 470                     deployment.status.availableReplicas == deployment.status.replicas and
 471                     deployment.status.observedGeneration == deployment.metadata.generation)
 472 
 473         def _pod_ready(pod):
 474             return (pod.status and pod.status.containerStatuses is not None and
 475                     all([container.ready for container in pod.status.containerStatuses]))
 476 
 477         def _daemonset_ready(daemonset):
 478             return (daemonset.status and daemonset.status.desiredNumberScheduled is not None and
 479                     daemonset.status.numberReady == daemonset.status.desiredNumberScheduled and
 480                     daemonset.status.observedGeneration == daemonset.metadata.generation)
 481 
 482         def _custom_condition(resource):
 483             if not resource.status or not resource.status.conditions:
 484                 return False
 485             match = [x for x in resource.status.conditions if x.type == condition['type']]
 486             if not match:
 487                 return False
 488             # There should never be more than one condition of a specific type
 489             match = match[0]
 490             if match.status == 'Unknown':
 491                 if match.status == condition['status']:
 492                     if 'reason' not in condition:
 493                         return True
 494                     if condition['reason']:
 495                         return match.reason == condition['reason']
 496                 return False
 497             status = True if match.status == 'True' else False
 498             if status == condition['status']:
 499                 if condition.get('reason'):
 500                     return match.reason == condition['reason']
 501                 return True
 502             return False
 503 
 504         def _resource_absent(resource):
 505             return not resource
 506 
 507         waiter = dict(
 508             Deployment=_deployment_ready,
 509             DaemonSet=_daemonset_ready,
 510             Pod=_pod_ready
 511         )
 512         kind = definition['kind']
 513         if state == 'present' and not condition:
 514             predicate = waiter.get(kind, lambda x: x)
 515         elif state == 'present' and condition:
 516             predicate = _custom_condition
 517         else:
 518             predicate = _resource_absent
 519         return self._wait_for(resource, definition['metadata']['name'], definition['metadata'].get('namespace'), predicate, sleep, timeout, state)
