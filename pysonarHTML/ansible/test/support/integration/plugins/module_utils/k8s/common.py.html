  59 def list_dict_str(value):
  60     if isinstance(value, list):
  61         return value
  62     elif isinstance(value, dict):
  63         return value
  64     elif isinstance(value, string_types):
  65         return value
  66     raise TypeError

 148 class K8sAnsibleMixin(object):
 149     _argspec_cache = None
 150 
 151     @property
 152     def argspec(self):
 153         """
 154         Introspect the model properties, and return an Ansible module arg_spec dict.
 155         :return: dict
 156         """
 157         if self._argspec_cache:
 158             return self._argspec_cache
 159         argument_spec = copy.deepcopy(COMMON_ARG_SPEC)
 160         argument_spec.update(copy.deepcopy(AUTH_ARG_SPEC))
 161         self._argspec_cache = argument_spec
 162         return self._argspec_cache
 163 
 164     def get_api_client(self, **auth_params):
 165         auth_params = auth_params or getattr(self, 'params', {})
 166         auth = {}
 167 
 168         # If authorization variables aren't defined, look for them in environment variables
 169         for true_name, arg_name in AUTH_ARG_MAP.items():
 170             if auth_params.get(arg_name) is None:
 171                 env_value = os.getenv('K8S_AUTH_{0}'.format(arg_name.upper()), None) or os.getenv('K8S_AUTH_{0}'.format(true_name.upper()), None)
 172                 if env_value is not None:
 173                     if AUTH_ARG_SPEC[arg_name].get('type') == 'bool':
 174                         env_value = env_value.lower() not in ['0', 'false', 'no']
 175                     auth[true_name] = env_value
 176             else:
 177                 auth[true_name] = auth_params[arg_name]
 178 
 179         def auth_set(*names):
 180             return all([auth.get(name) for name in names])
 181 
 182         if auth_set('username', 'password', 'host') or auth_set('api_key', 'host'):
 183             # We have enough in the parameters to authenticate, no need to load incluster or kubeconfig
 184             pass
 185         elif auth_set('kubeconfig') or auth_set('context'):
 186             kubernetes.config.load_kube_config(auth.get('kubeconfig'), auth.get('context'), persist_config=auth.get('persist_config'))
 187         else:
 188             # First try to do incluster config, then kubeconfig
 189             try:
 190                 kubernetes.config.load_incluster_config()
 191             except kubernetes.config.ConfigException:
 192                 kubernetes.config.load_kube_config(auth.get('kubeconfig'), auth.get('context'), persist_config=auth.get('persist_config'))
 193 
 194         # Override any values in the default configuration with Ansible parameters
 195         configuration = kubernetes.client.Configuration()
 196         for key, value in iteritems(auth):
 197             if key in AUTH_ARG_MAP.keys() and value is not None:
 198                 if key == 'api_key':
 199                     setattr(configuration, key, {'authorization': "Bearer {0}".format(value)})
 200                 else:
 201                     setattr(configuration, key, value)
 202 
 203         kubernetes.client.Configuration.set_default(configuration)
 204         return DynamicClient(kubernetes.client.ApiClient(configuration))
 205 
 206     def find_resource(self, kind, api_version, fail=False):
 207         for attribute in ['kind', 'name', 'singular_name']:
 208             try:
 209                 return self.client.resources.get(**{'api_version': api_version, attribute: kind})
 210             except (ResourceNotFoundError, ResourceNotUniqueError):
 211                 pass
 212         try:
 213             return self.client.resources.get(api_version=api_version, short_names=[kind])
 214         except (ResourceNotFoundError, ResourceNotUniqueError):
 215             if fail:
 216                 self.fail(msg='Failed to find exact match for {0}.{1} by [kind, name, singularName, shortNames]'.format(api_version, kind))
 217 
 218     def kubernetes_facts(self, kind, api_version, name=None, namespace=None, label_selectors=None, field_selectors=None):
 219         resource = self.find_resource(kind, api_version)
 220         if not resource:
 221             return dict(resources=[])
 222         try:
 223             result = resource.get(name=name,
 224                                   namespace=namespace,
 225                                   label_selector=','.join(label_selectors),
 226                                   field_selector=','.join(field_selectors)).to_dict()
 227         except openshift.dynamic.exceptions.NotFoundError:
 228             return dict(resources=[])
 229 
 230         if 'items' in result:
 231             return dict(resources=result['items'])
 232         else:
 233             return dict(resources=[result])
 234 
 235     def remove_aliases(self):
 236         """
 237         The helper doesn't know what to do with aliased keys
 238         """
 239         for k, v in iteritems(self.argspec):
 240             if 'aliases' in v:
 241                 for alias in v['aliases']:
 242                     if alias in self.params:
 243                         self.params.pop(alias)
 244 
 245     def load_resource_definitions(self, src):
 246         """ Load the requested src path """
 247         result = None
 248         path = os.path.normpath(src)
 249         if not os.path.exists(path):
 250             self.fail(msg="Error accessing {0}. Does the file exist?".format(path))
 251         try:
 252             with open(path, 'r') as f:
 253                 result = list(yaml.safe_load_all(f))
 254         except (IOError, yaml.YAMLError) as exc:
 255             self.fail(msg="Error loading resource_definition: {0}".format(exc))
 256         return result
 257 
 258     @staticmethod
 259     def diff_objects(existing, new):
 260         result = dict()
 261         diff = recursive_diff(existing, new)
 262         if diff:
 263             result['before'] = diff[0]
 264             result['after'] = diff[1]
 265         return not diff, result

 268 class KubernetesAnsibleModule(AnsibleModule, K8sAnsibleMixin):
 269     resource_definition = None
 270     api_version = None
 271     kind = None
 272 
 273     def __init__(self, *args, **kwargs):
 274 
 275         kwargs['argument_spec'] = self.argspec
 276         AnsibleModule.__init__(self, *args, **kwargs)
 277 
 278         if not HAS_K8S_MODULE_HELPER:
 279             self.fail_json(msg=missing_required_lib('openshift'), exception=K8S_IMP_ERR,
 280                            error=to_native(k8s_import_exception))
 281         self.openshift_version = openshift.__version__
 282 
 283         if not HAS_YAML:
 284             self.fail_json(msg=missing_required_lib("PyYAML"), exception=YAML_IMP_ERR)
 285 
 286     def execute_module(self):
 287         raise NotImplementedError()
 288 
 289     def fail(self, msg=None):
 290         self.fail_json(msg=msg)
