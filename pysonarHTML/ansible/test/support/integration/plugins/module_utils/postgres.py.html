  60 def ensure_required_libs(module):
  61     """Check required libraries."""
  62     if not HAS_PSYCOPG2:
  63         module.fail_json(msg=missing_required_lib('psycopg2'))
  64 
  65     if module.params.get('ca_cert') and LooseVersion(psycopg2.__version__) < LooseVersion('2.4.3'):
  66         module.fail_json(msg='psycopg2 must be at least 2.4.3 in order to use the ca_cert parameter')

  69 def connect_to_db(module, conn_params, autocommit=False, fail_on_conn=True):
  70     """Connect to a PostgreSQL database.
  71 
  72     Return psycopg2 connection object.
  73 
  74     Args:
  75         module (AnsibleModule) -- object of ansible.module_utils.basic.AnsibleModule class
  76         conn_params (dict) -- dictionary with connection parameters
  77 
  78     Kwargs:
  79         autocommit (bool) -- commit automatically (default False)
  80         fail_on_conn (bool) -- fail if connection failed or just warn and return None (default True)
  81     """
  82     ensure_required_libs(module)
  83 
  84     db_connection = None
  85     try:
  86         db_connection = psycopg2.connect(**conn_params)
  87         if autocommit:
  88             if LooseVersion(psycopg2.__version__) >= LooseVersion('2.4.2'):
  89                 db_connection.set_session(autocommit=True)
  90             else:
  91                 db_connection.set_isolation_level(psycopg2.extensions.ISOLATION_LEVEL_AUTOCOMMIT)
  92 
  93         # Switch role, if specified:
  94         if module.params.get('session_role'):
  95             cursor = db_connection.cursor(cursor_factory=psycopg2.extras.DictCursor)
  96 
  97             try:
  98                 cursor.execute('SET ROLE "%s"' % module.params['session_role'])
  99             except Exception as e:
 100                 module.fail_json(msg="Could not switch role: %s" % to_native(e))
 101             finally:
 102                 cursor.close()
 103 
 104     except TypeError as e:
 105         if 'sslrootcert' in e.args[0]:
 106             module.fail_json(msg='Postgresql server must be at least '
 107                                  'version 8.4 to support sslrootcert')
 108 
 109         if fail_on_conn:
 110             module.fail_json(msg="unable to connect to database: %s" % to_native(e))
 111         else:
 112             module.warn("PostgreSQL server is unavailable: %s" % to_native(e))
 113             db_connection = None
 114 
 115     except Exception as e:
 116         if fail_on_conn:
 117             module.fail_json(msg="unable to connect to database: %s" % to_native(e))
 118         else:
 119             module.warn("PostgreSQL server is unavailable: %s" % to_native(e))
 120             db_connection = None
 121 
 122     return db_connection

 125 def exec_sql(obj, query, query_params=None, ddl=False, add_to_executed=True, dont_exec=False):
 126     """Execute SQL.
 127 
 128     Auxiliary function for PostgreSQL user classes.
 129 
 130     Returns a query result if possible or True/False if ddl=True arg was passed.
 131     It necessary for statements that don't return any result (like DDL queries).
 132 
 133     Args:
 134         obj (obj) -- must be an object of a user class.
 135             The object must have module (AnsibleModule class object) and
 136             cursor (psycopg cursor object) attributes
 137         query (str) -- SQL query to execute
 138 
 139     Kwargs:
 140         query_params (dict or tuple) -- Query parameters to prevent SQL injections,
 141             could be a dict or tuple
 142         ddl (bool) -- must return True or False instead of rows (typical for DDL queries)
 143             (default False)
 144         add_to_executed (bool) -- append the query to obj.executed_queries attribute
 145         dont_exec (bool) -- used with add_to_executed=True to generate a query, add it
 146             to obj.executed_queries list and return True (default False)
 147     """
 148 
 149     if dont_exec:
 150         # This is usually needed to return queries in check_mode
 151         # without execution
 152         query = obj.cursor.mogrify(query, query_params)
 153         if add_to_executed:
 154             obj.executed_queries.append(query)
 155 
 156         return True
 157 
 158     try:
 159         if query_params is not None:
 160             obj.cursor.execute(query, query_params)
 161         else:
 162             obj.cursor.execute(query)
 163 
 164         if add_to_executed:
 165             if query_params is not None:
 166                 obj.executed_queries.append(obj.cursor.mogrify(query, query_params))
 167             else:
 168                 obj.executed_queries.append(query)
 169 
 170         if not ddl:
 171             res = obj.cursor.fetchall()
 172             return res
 173         return True
 174     except Exception as e:
 175         obj.module.fail_json(msg="Cannot execute SQL '%s': %s" % (query, to_native(e)))
 176     return False

 179 def get_conn_params(module, params_dict, warn_db_default=True):
 180     """Get connection parameters from the passed dictionary.
 181 
 182     Return a dictionary with parameters to connect to PostgreSQL server.
 183 
 184     Args:
 185         module (AnsibleModule) -- object of ansible.module_utils.basic.AnsibleModule class
 186         params_dict (dict) -- dictionary with variables
 187 
 188     Kwargs:
 189         warn_db_default (bool) -- warn that the default DB is used (default True)
 190     """
 191     # To use defaults values, keyword arguments must be absent, so
 192     # check which values are empty and don't include in the return dictionary
 193     params_map = {
 194         "login_host": "host",
 195         "login_user": "user",
 196         "login_password": "password",
 197         "port": "port",
 198         "ssl_mode": "sslmode",
 199         "ca_cert": "sslrootcert"
 200     }
 201 
 202     # Might be different in the modules:
 203     if params_dict.get('db'):
 204         params_map['db'] = 'database'
 205     elif params_dict.get('database'):
 206         params_map['database'] = 'database'
 207     elif params_dict.get('login_db'):
 208         params_map['login_db'] = 'database'
 209     else:
 210         if warn_db_default:
 211             module.warn('Database name has not been passed, '
 212                         'used default database to connect to.')
 213 
 214     kw = dict((params_map[k], v) for (k, v) in iteritems(params_dict)
 215               if k in params_map and v != '' and v is not None)
 216 
 217     # If a login_unix_socket is specified, incorporate it here.
 218     is_localhost = "host" not in kw or kw["host"] is None or kw["host"] == "localhost"
 219     if is_localhost and params_dict["login_unix_socket"] != "":
 220         kw["host"] = params_dict["login_unix_socket"]
 221 
 222     return kw

 225 class PgMembership(object):
 226     def __init__(self, module, cursor, groups, target_roles, fail_on_role=True):
 227         self.module = module
 228         self.cursor = cursor
 229         self.target_roles = [r.strip() for r in target_roles]
 230         self.groups = [r.strip() for r in groups]
 231         self.executed_queries = []
 232         self.granted = {}
 233         self.revoked = {}
 234         self.fail_on_role = fail_on_role
 235         self.non_existent_roles = []
 236         self.changed = False
 237         self.__check_roles_exist()
 238 
 239     def grant(self):
 240         for group in self.groups:
 241             self.granted[group] = []
 242 
 243             for role in self.target_roles:
 244                 # If role is in a group now, pass:
 245                 if self.__check_membership(group, role):
 246                     continue
 247 
 248                 query = 'GRANT "%s" TO "%s"' % (group, role)
 249                 self.changed = exec_sql(self, query, ddl=True)
 250 
 251                 if self.changed:
 252                     self.granted[group].append(role)
 253 
 254         return self.changed
 255 
 256     def revoke(self):
 257         for group in self.groups:
 258             self.revoked[group] = []
 259 
 260             for role in self.target_roles:
 261                 # If role is not in a group now, pass:
 262                 if not self.__check_membership(group, role):
 263                     continue
 264 
 265                 query = 'REVOKE "%s" FROM "%s"' % (group, role)
 266                 self.changed = exec_sql(self, query, ddl=True)
 267 
 268                 if self.changed:
 269                     self.revoked[group].append(role)
 270 
 271         return self.changed
 272 
 273     def __check_membership(self, src_role, dst_role):
 274         query = ("SELECT ARRAY(SELECT b.rolname FROM "
 275                  "pg_catalog.pg_auth_members m "
 276                  "JOIN pg_catalog.pg_roles b ON (m.roleid = b.oid) "
 277                  "WHERE m.member = r.oid) "
 278                  "FROM pg_catalog.pg_roles r "
 279                  "WHERE r.rolname = %(dst_role)s")
 280 
 281         res = exec_sql(self, query, query_params={'dst_role': dst_role}, add_to_executed=False)
 282         membership = []
 283         if res:
 284             membership = res[0][0]
 285 
 286         if not membership:
 287             return False
 288 
 289         if src_role in membership:
 290             return True
 291 
 292         return False
 293 
 294     def __check_roles_exist(self):
 295         existent_groups = self.__roles_exist(self.groups)
 296         existent_roles = self.__roles_exist(self.target_roles)
 297 
 298         for group in self.groups:
 299             if group not in existent_groups:
 300                 if self.fail_on_role:
 301                     self.module.fail_json(msg="Role %s does not exist" % group)
 302                 else:
 303                     self.module.warn("Role %s does not exist, pass" % group)
 304                     self.non_existent_roles.append(group)
 305 
 306         for role in self.target_roles:
 307             if role not in existent_roles:
 308                 if self.fail_on_role:
 309                     self.module.fail_json(msg="Role %s does not exist" % role)
 310                 else:
 311                     self.module.warn("Role %s does not exist, pass" % role)
 312 
 313                 if role not in self.groups:
 314                     self.non_existent_roles.append(role)
 315 
 316                 else:
 317                     if self.fail_on_role:
 318                         self.module.exit_json(msg="Role role '%s' is a member of role '%s'" % (role, role))
 319                     else:
 320                         self.module.warn("Role role '%s' is a member of role '%s', pass" % (role, role))
 321 
 322         # Update role lists, excluding non existent roles:
 323         self.groups = [g for g in self.groups if g not in self.non_existent_roles]
 324 
 325         self.target_roles = [r for r in self.target_roles if r not in self.non_existent_roles]
 326 
 327     def __roles_exist(self, roles):
 328         tmp = ["'" + x + "'" for x in roles]
 329         query = "SELECT rolname FROM pg_roles WHERE rolname IN (%s)" % ','.join(tmp)
 330         return [x[0] for x in exec_sql(self, query, add_to_executed=False)]
