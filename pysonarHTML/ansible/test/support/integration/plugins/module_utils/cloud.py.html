  48 def _exponential_backoff(retries=10, delay=2, backoff=2, max_delay=60):
  49     """ Customizable exponential backoff strategy.
  50     Args:
  51         retries (int): Maximum number of times to retry a request.
  52         delay (float): Initial (base) delay.
  53         backoff (float): base of the exponent to use for exponential
  54             backoff.
  55         max_delay (int): Optional. If provided each delay generated is capped
  56             at this amount. Defaults to 60 seconds.
  57     Returns:
  58         Callable that returns a generator. This generator yields durations in
  59         seconds to be used as delays for an exponential backoff strategy.
  60     Usage:
  61         >>> backoff = _exponential_backoff()
  62         >>> backoff
  63         <function backoff_backoff at 0x7f0d939facf8>
  64         >>> list(backoff())
  65         [2, 4, 8, 16, 32, 60, 60, 60, 60, 60]
  66     """
  67     def backoff_gen():
  68         for retry in range(0, retries):
  69             sleep = delay * backoff ** retry
  70             yield sleep if max_delay is None else min(sleep, max_delay)
  71     return backoff_gen

  74 def _full_jitter_backoff(retries=10, delay=3, max_delay=60, _random=random):
  75     """ Implements the "Full Jitter" backoff strategy described here
  76     https://www.awsarchitectureblog.com/2015/03/backoff.html
  77     Args:
  78         retries (int): Maximum number of times to retry a request.
  79         delay (float): Approximate number of seconds to sleep for the first
  80             retry.
  81         max_delay (int): The maximum number of seconds to sleep for any retry.
  82             _random (random.Random or None): Makes this generator testable by
  83             allowing developers to explicitly pass in the a seeded Random.
  84     Returns:
  85         Callable that returns a generator. This generator yields durations in
  86         seconds to be used as delays for a full jitter backoff strategy.
  87     Usage:
  88         >>> backoff = _full_jitter_backoff(retries=5)
  89         >>> backoff
  90         <function backoff_backoff at 0x7f0d939facf8>
  91         >>> list(backoff())
  92         [3, 6, 5, 23, 38]
  93         >>> list(backoff())
  94         [2, 1, 6, 6, 31]
  95     """
  96     def backoff_gen():
  97         for retry in range(0, retries):
  98             yield _random.randint(0, min(max_delay, delay * 2 ** retry))
  99     return backoff_gen

 102 class CloudRetry(object):
 103     """ CloudRetry can be used by any cloud provider, in order to implement a
 104         backoff algorithm/retry effect based on Status Code from Exceptions.
 105     """
 106     # This is the base class of the exception.
 107     # AWS Example botocore.exceptions.ClientError
 108     base_class = None
 109 
 110     @staticmethod
 111     def status_code_from_exception(error):
 112         """ Return the status code from the exception object
 113         Args:
 114             error (object): The exception itself.
 115         """
 116         pass
 117 
 118     @staticmethod
 119     def found(response_code, catch_extra_error_codes=None):
 120         """ Return True if the Response Code to retry on was found.
 121         Args:
 122             response_code (str): This is the Response Code that is being matched against.
 123         """
 124         pass
 125 
 126     @classmethod
 127     def _backoff(cls, backoff_strategy, catch_extra_error_codes=None):
 128         """ Retry calling the Cloud decorated function using the provided
 129         backoff strategy.
 130         Args:
 131             backoff_strategy (callable): Callable that returns a generator. The
 132             generator should yield sleep times for each retry of the decorated
 133             function.
 134         """
 135         def deco(f):
 136             @wraps(f)
 137             def retry_func(*args, **kwargs):
 138                 for delay in backoff_strategy():
 139                     try:
 140                         return f(*args, **kwargs)
 141                     except Exception as e:
 142                         if isinstance(e, cls.base_class):
 143                             response_code = cls.status_code_from_exception(e)
 144                             if cls.found(response_code, catch_extra_error_codes):
 145                                 msg = "{0}: Retrying in {1} seconds...".format(str(e), delay)
 146                                 syslog.syslog(syslog.LOG_INFO, msg)
 147                                 time.sleep(delay)
 148                             else:
 149                                 # Return original exception if exception is not a ClientError
 150                                 raise e
 151                         else:
 152                             # Return original exception if exception is not a ClientError
 153                             raise e
 154                 return f(*args, **kwargs)
 155 
 156             return retry_func  # true decorator
 157 
 158         return deco
 159 
 160     @classmethod
 161     def exponential_backoff(cls, retries=10, delay=3, backoff=2, max_delay=60, catch_extra_error_codes=None):
 162         """
 163         Retry calling the Cloud decorated function using an exponential backoff.
 164 
 165         Kwargs:
 166             retries (int): Number of times to retry a failed request before giving up
 167                 default=10
 168             delay (int or float): Initial delay between retries in seconds
 169                 default=3
 170             backoff (int or float): backoff multiplier e.g. value of 2 will
 171                 double the delay each retry
 172                 default=1.1
 173             max_delay (int or None): maximum amount of time to wait between retries.
 174                 default=60
 175         """
 176         return cls._backoff(_exponential_backoff(
 177             retries=retries, delay=delay, backoff=backoff, max_delay=max_delay), catch_extra_error_codes)
 178 
 179     @classmethod
 180     def jittered_backoff(cls, retries=10, delay=3, max_delay=60, catch_extra_error_codes=None):
 181         """
 182         Retry calling the Cloud decorated function using a jittered backoff
 183         strategy. More on this strategy here:
 184 
 185         https://www.awsarchitectureblog.com/2015/03/backoff.html
 186 
 187         Kwargs:
 188             retries (int): Number of times to retry a failed request before giving up
 189                 default=10
 190             delay (int): Initial delay between retries in seconds
 191                 default=3
 192             max_delay (int): maximum amount of time to wait between retries.
 193                 default=60
 194         """
 195         return cls._backoff(_full_jitter_backoff(
 196             retries=retries, delay=delay, max_delay=max_delay), catch_extra_error_codes)
 197 
 198     @classmethod
 199     def backoff(cls, tries=10, delay=3, backoff=1.1, catch_extra_error_codes=None):
 200         """
 201         Retry calling the Cloud decorated function using an exponential backoff.
 202 
 203         Compatibility for the original implementation of CloudRetry.backoff that
 204         did not provide configurable backoff strategies. Developers should use
 205         CloudRetry.exponential_backoff instead.
 206 
 207         Kwargs:
 208             tries (int): Number of times to try (not retry) before giving up
 209                 default=10
 210             delay (int or float): Initial delay between retries in seconds
 211                 default=3
 212             backoff (int or float): backoff multiplier e.g. value of 2 will
 213                 double the delay each retry
 214                 default=1.1
 215         """
 216         return cls.exponential_backoff(
 217             retries=tries - 1, delay=delay, backoff=backoff, max_delay=None, catch_extra_error_codes=catch_extra_error_codes)
