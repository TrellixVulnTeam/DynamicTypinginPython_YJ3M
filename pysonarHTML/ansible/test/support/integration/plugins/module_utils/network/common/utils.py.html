 100 class Entity(object):
 101     """Transforms a dict to with an argument spec
 102 
 103     This class will take a dict and apply an Ansible argument spec to the
 104     values.  The resulting dict will contain all of the keys in the param
 105     with appropriate values set.
 106 
 107     Example::
 108 
 109         argument_spec = dict(
 110             command=dict(key=True),
 111             display=dict(default='text', choices=['text', 'json']),
 112             validate=dict(type='bool')
 113         )
 114         transform = Entity(module, argument_spec)
 115         value = dict(command='foo')
 116         result = transform(value)
 117         print result
 118         {'command': 'foo', 'display': 'text', 'validate': None}
 119 
 120     Supported argument spec:
 121         * key - specifies how to map a single value to a dict
 122         * read_from - read and apply the argument_spec from the module
 123         * required - a value is required
 124         * type - type of value (uses AnsibleModule type checker)
 125         * fallback - implements fallback function
 126         * choices - set of valid options
 127         * default - default value
 128     """
 129 
 130     def __init__(self, module, attrs=None, args=None, keys=None, from_argspec=False):
 131         args = [] if args is None else args
 132 
 133         self._attributes = attrs or {}
 134         self._module = module
 135 
 136         for arg in args:
 137             self._attributes[arg] = dict()
 138             if from_argspec:
 139                 self._attributes[arg]['read_from'] = arg
 140             if keys and arg in keys:
 141                 self._attributes[arg]['key'] = True
 142 
 143         self.attr_names = frozenset(self._attributes.keys())
 144 
 145         _has_key = False
 146 
 147         for name, attr in iteritems(self._attributes):
 148             if attr.get('read_from'):
 149                 if attr['read_from'] not in self._module.argument_spec:
 150                     module.fail_json(msg='argument %s does not exist' % attr['read_from'])
 151                 spec = self._module.argument_spec.get(attr['read_from'])
 152                 for key, value in iteritems(spec):
 153                     if key not in attr:
 154                         attr[key] = value
 155 
 156             if attr.get('key'):
 157                 if _has_key:
 158                     module.fail_json(msg='only one key value can be specified')
 159                 _has_key = True
 160                 attr['required'] = True
 161 
 162     def serialize(self):
 163         return self._attributes
 164 
 165     def to_dict(self, value):
 166         obj = {}
 167         for name, attr in iteritems(self._attributes):
 168             if attr.get('key'):
 169                 obj[name] = value
 170             else:
 171                 obj[name] = attr.get('default')
 172         return obj
 173 
 174     def __call__(self, value, strict=True):
 175         if not isinstance(value, dict):
 176             value = self.to_dict(value)
 177 
 178         if strict:
 179             unknown = set(value).difference(self.attr_names)
 180             if unknown:
 181                 self._module.fail_json(msg='invalid keys: %s' % ','.join(unknown))
 182 
 183         for name, attr in iteritems(self._attributes):
 184             if value.get(name) is None:
 185                 value[name] = attr.get('default')
 186 
 187             if attr.get('fallback') and not value.get(name):
 188                 fallback = attr.get('fallback', (None,))
 189                 fallback_strategy = fallback[0]
 190                 fallback_args = []
 191                 fallback_kwargs = {}
 192                 if fallback_strategy is not None:
 193                     for item in fallback[1:]:
 194                         if isinstance(item, dict):
 195                             fallback_kwargs = item
 196                         else:
 197                             fallback_args = item
 198                     try:
 199                         value[name] = fallback_strategy(*fallback_args, **fallback_kwargs)
 200                     except basic.AnsibleFallbackNotFound:
 201                         continue
 202 
 203             if attr.get('required') and value.get(name) is None:
 204                 self._module.fail_json(msg='missing required attribute %s' % name)
 205 
 206             if 'choices' in attr:
 207                 if value[name] not in attr['choices']:
 208                     self._module.fail_json(msg='%s must be one of %s, got %s' % (name, ', '.join(attr['choices']), value[name]))
 209 
 210             if value[name] is not None:
 211                 value_type = attr.get('type', 'str')
 212                 type_checker = self._module._CHECK_ARGUMENT_TYPES_DISPATCHER[value_type]
 213                 type_checker(value[name])
 214             elif value.get(name):
 215                 value[name] = self._module.params[name]
 216 
 217         return value

 220 class EntityCollection(Entity):
 221     """Extends ```Entity``` to handle a list of dicts """
 222 
 223     def __call__(self, iterable, strict=True):
 224         if iterable is None:
 225             iterable = [super(EntityCollection, self).__call__(self._module.params, strict)]
 226 
 227         if not isinstance(iterable, (list, tuple)):
 228             self._module.fail_json(msg='value must be an iterable')
 229 
 230         return [(super(EntityCollection, self).__call__(i, strict)) for i in iterable]

 245 def dict_diff(base, comparable):
 246     """ Generate a dict object of differences
 247 
 248     This function will compare two dict objects and return the difference
 249     between them as a dict object.  For scalar values, the key will reflect
 250     the updated value.  If the key does not exist in `comparable`, then then no
 251     key will be returned.  For lists, the value in comparable will wholly replace
 252     the value in base for the key.  For dicts, the returned value will only
 253     return keys that are different.
 254 
 255     :param base: dict object to base the diff on
 256     :param comparable: dict object to compare against base
 257 
 258     :returns: new dict object with differences
 259     """
 260     if not isinstance(base, dict):
 261         raise AssertionError("`base` must be of type <dict>")
 262     if not isinstance(comparable, dict):
 263         if comparable is None:
 264             comparable = dict()
 265         else:
 266             raise AssertionError("`comparable` must be of type <dict>")
 267 
 268     updates = dict()
 269 
 270     for key, value in iteritems(base):
 271         if isinstance(value, dict):
 272             item = comparable.get(key)
 273             if item is not None:
 274                 sub_diff = dict_diff(value, comparable[key])
 275                 if sub_diff:
 276                     updates[key] = sub_diff
 277         else:
 278             comparable_value = comparable.get(key)
 279             if comparable_value is not None:
 280                 if sort_list(base[key]) != sort_list(comparable_value):
 281                     updates[key] = comparable_value
 282 
 283     for key in set(comparable.keys()).difference(base.keys()):
 284         updates[key] = comparable.get(key)
 285 
 286     return updates

 289 def dict_merge(base, other):
 290     """ Return a new dict object that combines base and other
 291 
 292     This will create a new dict object that is a combination of the key/value
 293     pairs from base and other.  When both keys exist, the value will be
 294     selected from other.  If the value is a list object, the two lists will
 295     be combined and duplicate entries removed.
 296 
 297     :param base: dict object to serve as base
 298     :param other: dict object to combine with base
 299 
 300     :returns: new combined dict object
 301     """
 302     if not isinstance(base, dict):
 303         raise AssertionError("`base` must be of type <dict>")
 304     if not isinstance(other, dict):
 305         raise AssertionError("`other` must be of type <dict>")
 306 
 307     combined = dict()
 308 
 309     for key, value in iteritems(base):
 310         if isinstance(value, dict):
 311             if key in other:
 312                 item = other.get(key)
 313                 if item is not None:
 314                     if isinstance(other[key], Mapping):
 315                         combined[key] = dict_merge(value, other[key])
 316                     else:
 317                         combined[key] = other[key]
 318                 else:
 319                     combined[key] = item
 320             else:
 321                 combined[key] = value
 322         elif isinstance(value, list):
 323             if key in other:
 324                 item = other.get(key)
 325                 if item is not None:
 326                     try:
 327                         combined[key] = list(set(chain(value, item)))
 328                     except TypeError:
 329                         value.extend([i for i in item if i not in value])
 330                         combined[key] = value
 331                 else:
 332                     combined[key] = item
 333             else:
 334                 combined[key] = value
 335         else:
 336             if key in other:
 337                 other_value = other.get(key)
 338                 if other_value is not None:
 339                     if sort_list(base[key]) != sort_list(other_value):
 340                         combined[key] = other_value
 341                     else:
 342                         combined[key] = value
 343                 else:
 344                     combined[key] = other_value
 345             else:
 346                 combined[key] = value
 347 
 348     for key in set(other.keys()).difference(base.keys()):
 349         combined[key] = other.get(key)
 350 
 351     return combined

 354 def param_list_to_dict(param_list, unique_key="name", remove_key=True):
 355     """Rotates a list of dictionaries to be a dictionary of dictionaries.
 356 
 357     :param param_list: The aforementioned list of dictionaries
 358     :param unique_key: The name of a key which is present and unique in all of param_list's dictionaries. The value
 359     behind this key will be the key each dictionary can be found at in the new root dictionary
 360     :param remove_key: If True, remove unique_key from the individual dictionaries before returning.
 361     """
 362     param_dict = {}
 363     for params in param_list:
 364         params = params.copy()
 365         if remove_key:
 366             name = params.pop(unique_key)
 367         else:
 368             name = params.get(unique_key)
 369         param_dict[name] = params
 370 
 371     return param_dict

 374 def conditional(expr, val, cast=None):
 375     match = re.match(r'^(.+)\((.+)\)$', str(expr), re.I)
 376     if match:
 377         op, arg = match.groups()
 378     else:
 379         op = 'eq'
 380         if ' ' in str(expr):
 381             raise AssertionError('invalid expression: cannot contain spaces')
 382         arg = expr
 383 
 384     if cast is None and val is not None:
 385         arg = type(val)(arg)
 386     elif callable(cast):
 387         arg = cast(arg)
 388         val = cast(val)
 389 
 390     op = next((oper for alias, oper in ALIASES if op == alias), op)
 391 
 392     if not hasattr(operator, op) and op not in OPERATORS:
 393         raise ValueError('unknown operator: %s' % op)
 394 
 395     func = getattr(operator, op)
 396     return func(val, arg)

 435 def load_provider(spec, args):
 436     provider = args.get('provider') or {}
 437     for key, value in iteritems(spec):
 438         if key not in provider:
 439             if 'fallback' in value:
 440                 provider[key] = _fallback(value['fallback'])
 441             elif 'default' in value:
 442                 provider[key] = value['default']
 443             else:
 444                 provider[key] = None
 445     if 'authorize' in provider:
 446         # Coerce authorize to provider if a string has somehow snuck in.
 447         provider['authorize'] = boolean(provider['authorize'] or False)
 448     args['provider'] = provider
 449     return provider

 452 def _fallback(fallback):
 453     strategy = fallback[0]
 454     args = []
 455     kwargs = {}
 456 
 457     for item in fallback[1:]:
 458         if isinstance(item, dict):
 459             kwargs = item
 460         else:
 461             args = item
 462     try:
 463         return strategy(*args, **kwargs)
 464     except basic.AnsibleFallbackNotFound:
 465         pass

 468 def generate_dict(spec):
 469     """
 470     Generate dictionary which is in sync with argspec
 471 
 472     :param spec: A dictionary that is the argspec of the module
 473     :rtype: A dictionary
 474     :returns: A dictionary in sync with argspec with default value
 475     """
 476     obj = {}
 477     if not spec:
 478         return obj
 479 
 480     for key, val in iteritems(spec):
 481         if 'default' in val:
 482             dct = {key: val['default']}
 483         elif 'type' in val and val['type'] == 'dict':
 484             dct = {key: generate_dict(val['options'])}
 485         else:
 486             dct = {key: None}
 487         obj.update(dct)
 488     return obj

 491 def parse_conf_arg(cfg, arg):
 492     """
 493     Parse config based on argument
 494 
 495     :param cfg: A text string which is a line of configuration.
 496     :param arg: A text string which is to be matched.
 497     :rtype: A text string
 498     :returns: A text string if match is found
 499     """
 500     match = re.search(r'%s (.+)(\n|$)' % arg, cfg, re.M)
 501     if match:
 502         result = match.group(1).strip()
 503     else:
 504         result = None
 505     return result

 508 def parse_conf_cmd_arg(cfg, cmd, res1, res2=None, delete_str='no'):
 509     """
 510     Parse config based on command
 511 
 512     :param cfg: A text string which is a line of configuration.
 513     :param cmd: A text string which is the command to be matched
 514     :param res1: A text string to be returned if the command is present
 515     :param res2: A text string to be returned if the negate command
 516                  is present
 517     :param delete_str: A text string to identify the start of the
 518                  negate command
 519     :rtype: A text string
 520     :returns: A text string if match is found
 521     """
 522     match = re.search(r'\n\s+%s(\n|$)' % cmd, cfg)
 523     if match:
 524         return res1
 525     if res2 is not None:
 526         match = re.search(r'\n\s+%s %s(\n|$)' % (delete_str, cmd), cfg)
 527         if match:
 528             return res2
 529     return None

 532 def get_xml_conf_arg(cfg, path, data='text'):
 533     """
 534     :param cfg: The top level configuration lxml Element tree object
 535     :param path: The relative xpath w.r.t to top level element (cfg)
 536            to be searched in the xml hierarchy
 537     :param data: The type of data to be returned for the matched xml node.
 538         Valid values are text, tag, attrib, with default as text.
 539     :return: Returns the required type for the matched xml node or else None
 540     """
 541     match = cfg.xpath(path)
 542     if len(match):
 543         if data == 'tag':
 544             result = getattr(match[0], 'tag')
 545         elif data == 'attrib':
 546             result = getattr(match[0], 'attrib')
 547         else:
 548             result = getattr(match[0], 'text')
 549     else:
 550         result = None
 551     return result

 554 def remove_empties(cfg_dict):
 555     """
 556     Generate final config dictionary
 557 
 558     :param cfg_dict: A dictionary parsed in the facts system
 559     :rtype: A dictionary
 560     :returns: A dictionary by eliminating keys that have null values
 561     """
 562     final_cfg = {}
 563     if not cfg_dict:
 564         return final_cfg
 565 
 566     for key, val in iteritems(cfg_dict):
 567         dct = None
 568         if isinstance(val, dict):
 569             child_val = remove_empties(val)
 570             if child_val:
 571                 dct = {key: child_val}
 572         elif (isinstance(val, list) and val
 573               and all([isinstance(x, dict) for x in val])):
 574             child_val = [remove_empties(x) for x in val]
 575             if child_val:
 576                 dct = {key: child_val}
 577         elif val not in [None, [], {}, (), '']:
 578             dct = {key: val}
 579         if dct:
 580             final_cfg.update(dct)
 581     return final_cfg

 607 class Template:
 608 
 609     def __init__(self):
 610         if not HAS_JINJA2:
 611             raise ImportError("jinja2 is required but does not appear to be installed.  "
 612                               "It can be installed using `pip install jinja2`")
 613 
 614         self.env = Environment(undefined=StrictUndefined)
 615         self.env.filters.update({'ternary': ternary})
 616 
 617     def __call__(self, value, variables=None, fail_on_undefined=True):
 618         variables = variables or {}
 619 
 620         if not self.contains_vars(value):
 621             return value
 622 
 623         try:
 624             value = self.env.from_string(value).render(variables)
 625         except UndefinedError:
 626             if not fail_on_undefined:
 627                 return None
 628             raise
 629 
 630         if value:
 631             try:
 632                 return ast.literal_eval(value)
 633             except Exception:
 634                 return str(value)
 635         else:
 636             return None
 637 
 638     def contains_vars(self, data):
 639         if isinstance(data, string_types):
 640             for marker in (self.env.block_start_string, self.env.variable_start_string, self.env.comment_start_string):
 641                 if marker in data:
 642                     return True
 643         return False
