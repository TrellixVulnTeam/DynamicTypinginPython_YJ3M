  46 class RabbitClient():
  47     def __init__(self, module):
  48         self.module = module
  49         self.params = module.params
  50         self.check_required_library()
  51         self.check_host_params()
  52         self.url = self.params['url']
  53         self.proto = self.params['proto']
  54         self.username = self.params['username']
  55         self.password = self.params['password']
  56         self.host = self.params['host']
  57         self.port = self.params['port']
  58         self.vhost = self.params['vhost']
  59         self.queue = self.params['queue']
  60         self.headers = self.params['headers']
  61         self.cafile = self.params['cafile']
  62         self.certfile = self.params['certfile']
  63         self.keyfile = self.params['keyfile']
  64 
  65         if self.host is not None:
  66             self.build_url()
  67 
  68         if self.cafile is not None:
  69             self.append_ssl_certs()
  70 
  71         self.connect_to_rabbitmq()

  73     def check_required_library(self):
  74         if not HAS_PIKA:
  75             self.module.fail_json(msg=missing_required_lib("pika"), exception=PIKA_IMP_ERR)
  76 
  77     def check_host_params(self):
  78         # Fail if url is specified and other conflicting parameters have been specified
  79         if self.params['url'] is not None and any(self.params[k] is not None for k in ['proto', 'host', 'port', 'password', 'username', 'vhost']):
  80             self.module.fail_json(msg="url and proto, host, port, vhost, username or password cannot be specified at the same time.")
  81 
  82         # Fail if url not specified and there is a missing parameter to build the url
  83         if self.params['url'] is None and any(self.params[k] is None for k in ['proto', 'host', 'port', 'password', 'username', 'vhost']):
  84             self.module.fail_json(msg="Connection parameters must be passed via url, or,  proto, host, port, vhost, username or password.")
  85 
  86     def append_ssl_certs(self):
  87         ssl_options = {}
  88         if self.cafile:
  89             ssl_options['cafile'] = self.cafile
  90         if self.certfile:
  91             ssl_options['certfile'] = self.certfile
  92         if self.keyfile:
  93             ssl_options['keyfile'] = self.keyfile
  94 
  95         self.url = self.url + '?ssl_options=' + urllib_parse.quote(json.dumps(ssl_options))

 110     ''' Consider some file size limits here '''
 111     def _read_file(self, path):
 112         try:
 113             with open(path, "rb") as file_handle:
 114                 return file_handle.read()
 115         except IOError as e:
 116             self.module.fail_json(msg="Unable to open file %s: %s" % (path, to_native(e)))

 118     @staticmethod
 119     def _check_file_mime_type(path):
 120         mime = MimeTypes()
 121         return mime.guess_type(path)

 123     def build_url(self):
 124         self.url = '{0}://{1}:{2}@{3}:{4}/{5}'.format(self.proto,
 125                                                       self.username,
 126                                                       self.password,
 127                                                       self.host,
 128                                                       self.port,
 129                                                       self.vhost)

 131     def connect_to_rabbitmq(self):
 132         """
 133         Function to connect to rabbitmq using username and password
 134         """
 135         try:
 136             parameters = pika.URLParameters(self.url)
 137         except Exception as e:
 138             self.module.fail_json(msg="URL malformed: %s" % to_native(e))
 139 
 140         try:
 141             self.connection = pika.BlockingConnection(parameters)
 142         except Exception as e:
 143             self.module.fail_json(msg="Connection issue: %s" % to_native(e))
 144 
 145         try:
 146             self.conn_channel = self.connection.channel()
 147         except pika.exceptions.AMQPChannelError as e:
 148             self.close_connection()
 149             self.module.fail_json(msg="Channel issue: %s" % to_native(e))
 150 
 151     def close_connection(self):
 152         try:
 153             self.connection.close()
 154         except pika.exceptions.AMQPConnectionError:
 155             pass

 157     def basic_publish(self):
 158         self.content_type = self.params.get("content_type")
 159 
 160         if self.params.get("body") is not None:
 161             args = dict(
 162                 body=self.params.get("body"),
 163                 exchange=self.params.get("exchange"),
 164                 routing_key=self.params.get("routing_key"),
 165                 properties=pika.BasicProperties(content_type=self.content_type, delivery_mode=1, headers=self.headers))
 166 
 167         # If src (file) is defined and content_type is left as default, do a mime lookup on the file
 168         if self.params.get("src") is not None and self.content_type == 'text/plain':
 169             self.content_type = RabbitClient._check_file_mime_type(self.params.get("src"))[0]
 170             self.headers.update(
 171                 filename=os.path.basename(self.params.get("src"))
 172             )
 173 
 174             args = dict(
 175                 body=self._read_file(self.params.get("src")),
 176                 exchange=self.params.get("exchange"),
 177                 routing_key=self.params.get("routing_key"),
 178                 properties=pika.BasicProperties(content_type=self.content_type,
 179                                                 delivery_mode=1,
 180                                                 headers=self.headers
 181                                                 ))
 182         elif self.params.get("src") is not None:
 183             args = dict(
 184                 body=self._read_file(self.params.get("src")),
 185                 exchange=self.params.get("exchange"),
 186                 routing_key=self.params.get("routing_key"),
 187                 properties=pika.BasicProperties(content_type=self.content_type,
 188                                                 delivery_mode=1,
 189                                                 headers=self.headers
 190                                                 ))
 191 
 192         try:
 193             # If queue is not defined, RabbitMQ will return the queue name of the automatically generated queue.
 194             if self.queue is None:
 195                 result = self.conn_channel.queue_declare(durable=self.params.get("durable"),
 196                                                          exclusive=self.params.get("exclusive"),
 197                                                          auto_delete=self.params.get("auto_delete"))
 198                 self.conn_channel.confirm_delivery()
 199                 self.queue = result.method.queue
 200             else:
 201                 self.conn_channel.queue_declare(queue=self.queue,
 202                                                 durable=self.params.get("durable"),
 203                                                 exclusive=self.params.get("exclusive"),
 204                                                 auto_delete=self.params.get("auto_delete"))
 205                 self.conn_channel.confirm_delivery()
 206         except Exception as e:
 207             self.module.fail_json(msg="Queue declare issue: %s" % to_native(e))
 208 
 209         # https://github.com/ansible/ansible/blob/devel/lib/ansible/module_utils/cloudstack.py#L150
 210         if args['routing_key'] is None:
 211             args['routing_key'] = self.queue
 212 
 213         if args['exchange'] is None:
 214             args['exchange'] = ''
 215 
 216         try:
 217             self.conn_channel.basic_publish(**args)
 218             return True
 219         except pika.exceptions.UnroutableError:
 220             return False
