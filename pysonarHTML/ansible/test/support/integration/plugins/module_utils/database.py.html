  38 # maps a type of identifier to the maximum number of dot levels that are
  39 # allowed to specify that identifier.  For example, a database column can be
  40 # specified by up to 4 levels: database.schema.table.column
  41 _PG_IDENTIFIER_TO_DOT_LEVEL = dict(
  42     database=1,
  43     schema=2,
  44     table=3,
  45     column=4,
  46     role=1,
  47     tablespace=1,
  48     sequence=3,
  49     publication=1,
  50 )
  51 _MYSQL_IDENTIFIER_TO_DOT_LEVEL = dict(database=1, table=2, column=3, role=1, vars=1)

  54 def _find_end_quote(identifier, quote_char):
  55     accumulate = 0
  56     while True:
  57         try:
  58             quote = identifier.index(quote_char)
  59         except ValueError:
  60             raise UnclosedQuoteError
  61         accumulate = accumulate + quote
  62         try:
  63             next_char = identifier[quote + 1]
  64         except IndexError:
  65             return accumulate
  66         if next_char == quote_char:
  67             try:
  68                 identifier = identifier[quote + 2:]
  69                 accumulate = accumulate + 2
  70             except IndexError:
  71                 raise UnclosedQuoteError
  72         else:
  73             return accumulate

  76 def _identifier_parse(identifier, quote_char):
  77     if not identifier:
  78         raise SQLParseError('Identifier name unspecified or unquoted trailing dot')
  79 
  80     already_quoted = False
  81     if identifier.startswith(quote_char):
  82         already_quoted = True
  83         try:
  84             end_quote = _find_end_quote(identifier[1:], quote_char=quote_char) + 1
  85         except UnclosedQuoteError:
  86             already_quoted = False
  87         else:
  88             if end_quote < len(identifier) - 1:
  89                 if identifier[end_quote + 1] == '.':
  90                     dot = end_quote + 1
  91                     first_identifier = identifier[:dot]
  92                     next_identifier = identifier[dot + 1:]
  93                     further_identifiers = _identifier_parse(next_identifier, quote_char)
  94                     further_identifiers.insert(0, first_identifier)
  95                 else:
  96                     raise SQLParseError('User escaped identifiers must escape extra quotes')
  97             else:
  98                 further_identifiers = [identifier]
  99 
 100     if not already_quoted:
 101         try:
 102             dot = identifier.index('.')
 103         except ValueError:
 104             identifier = identifier.replace(quote_char, quote_char * 2)
 105             identifier = ''.join((quote_char, identifier, quote_char))
 106             further_identifiers = [identifier]
 107         else:
 108             if dot == 0 or dot >= len(identifier) - 1:
 109                 identifier = identifier.replace(quote_char, quote_char * 2)
 110                 identifier = ''.join((quote_char, identifier, quote_char))
 111                 further_identifiers = [identifier]
 112             else:
 113                 first_identifier = identifier[:dot]
 114                 next_identifier = identifier[dot + 1:]
 115                 further_identifiers = _identifier_parse(next_identifier, quote_char)
 116                 first_identifier = first_identifier.replace(quote_char, quote_char * 2)
 117                 first_identifier = ''.join((quote_char, first_identifier, quote_char))
 118                 further_identifiers.insert(0, first_identifier)
 119 
 120     return further_identifiers

 123 def pg_quote_identifier(identifier, id_type):
 124     identifier_fragments = _identifier_parse(identifier, quote_char='"')
 125     if len(identifier_fragments) > _PG_IDENTIFIER_TO_DOT_LEVEL[id_type]:
 126         raise SQLParseError('PostgreSQL does not support %s with more than %i dots' % (id_type, _PG_IDENTIFIER_TO_DOT_LEVEL[id_type]))
 127     return '.'.join(identifier_fragments)

 130 def mysql_quote_identifier(identifier, id_type):
 131     identifier_fragments = _identifier_parse(identifier, quote_char='`')
 132     if (len(identifier_fragments) - 1) > _MYSQL_IDENTIFIER_TO_DOT_LEVEL[id_type]:
 133         raise SQLParseError('MySQL does not support %s with more than %i dots' % (id_type, _MYSQL_IDENTIFIER_TO_DOT_LEVEL[id_type]))
 134 
 135     special_cased_fragments = []
 136     for fragment in identifier_fragments:
 137         if fragment == '`*`':
 138             special_cased_fragments.append('*')
 139         else:
 140             special_cased_fragments.append(fragment)
 141 
 142     return '.'.join(special_cased_fragments)
