  66 class Connection(ConnectionBase):
  67     ''' Local chroot based connections '''
  68 
  69     transport = 'chroot'
  70     has_pipelining = True
  71     # su currently has an undiagnosed issue with calculating the file
  72     # checksums (so copy, for instance, doesn't work right)
  73     # Have to look into that before re-enabling this
  74     has_tty = False
  75 
  76     default_user = 'root'
  77 
  78     def __init__(self, play_context, new_stdin, *args, **kwargs):
  79         super(Connection, self).__init__(play_context, new_stdin, *args, **kwargs)
  80 
  81         self.chroot = self._play_context.remote_addr
  82 
  83         if os.geteuid() != 0:
  84             raise AnsibleError("chroot connection requires running as root")
  85 
  86         # we're running as root on the local system so do some
  87         # trivial checks for ensuring 'host' is actually a chroot'able dir
  88         if not os.path.isdir(self.chroot):
  89             raise AnsibleError("%s is not a directory" % self.chroot)
  90 
  91         chrootsh = os.path.join(self.chroot, 'bin/sh')
  92         # Want to check for a usable bourne shell inside the chroot.
  93         # is_executable() == True is sufficient.  For symlinks it
  94         # gets really complicated really fast.  So we punt on finding that
  95         # out.  As long as it's a symlink we assume that it will work
  96         if not (is_executable(chrootsh) or (os.path.lexists(chrootsh) and os.path.islink(chrootsh))):
  97             raise AnsibleError("%s does not look like a chrootable dir (/bin/sh missing)" % self.chroot)
  98 
  99     def _connect(self):
 100         ''' connect to the chroot '''
 101         if os.path.isabs(self.get_option('chroot_exe')):
 102             self.chroot_cmd = self.get_option('chroot_exe')
 103         else:
 104             try:
 105                 self.chroot_cmd = get_bin_path(self.get_option('chroot_exe'))
 106             except ValueError as e:
 107                 raise AnsibleError(to_native(e))
 108 
 109         super(Connection, self)._connect()
 110         if not self._connected:
 111             display.vvv("THIS IS A LOCAL CHROOT DIR", host=self.chroot)
 112             self._connected = True
 113 
 114     def _buffered_exec_command(self, cmd, stdin=subprocess.PIPE):
 115         ''' run a command on the chroot.  This is only needed for implementing
 116         put_file() get_file() so that we don't have to read the whole file
 117         into memory.
 118 
 119         compared to exec_command() it looses some niceties like being able to
 120         return the process's exit code immediately.
 121         '''
 122         executable = self.get_option('executable')
 123         local_cmd = [self.chroot_cmd, self.chroot, executable, '-c', cmd]
 124 
 125         display.vvv("EXEC %s" % (local_cmd), host=self.chroot)
 126         local_cmd = [to_bytes(i, errors='surrogate_or_strict') for i in local_cmd]
 127         p = subprocess.Popen(local_cmd, shell=False, stdin=stdin,
 128                              stdout=subprocess.PIPE, stderr=subprocess.PIPE)
 129 
 130         return p
 131 
 132     def exec_command(self, cmd, in_data=None, sudoable=False):
 133         ''' run a command on the chroot '''
 134         super(Connection, self).exec_command(cmd, in_data=in_data, sudoable=sudoable)
 135 
 136         p = self._buffered_exec_command(cmd)
 137 
 138         stdout, stderr = p.communicate(in_data)
 139         return (p.returncode, stdout, stderr)
 140 
 141     def _prefix_login_path(self, remote_path):
 142         ''' Make sure that we put files into a standard path
 143 
 144             If a path is relative, then we need to choose where to put it.
 145             ssh chooses $HOME but we aren't guaranteed that a home dir will
 146             exist in any given chroot.  So for now we're choosing "/" instead.
 147             This also happens to be the former default.
 148 
 149             Can revisit using $HOME instead if it's a problem
 150         '''
 151         if not remote_path.startswith(os.path.sep):
 152             remote_path = os.path.join(os.path.sep, remote_path)
 153         return os.path.normpath(remote_path)
 154 
 155     def put_file(self, in_path, out_path):
 156         ''' transfer a file from local to chroot '''
 157         super(Connection, self).put_file(in_path, out_path)
 158         display.vvv("PUT %s TO %s" % (in_path, out_path), host=self.chroot)
 159 
 160         out_path = shlex_quote(self._prefix_login_path(out_path))
 161         try:
 162             with open(to_bytes(in_path, errors='surrogate_or_strict'), 'rb') as in_file:
 163                 if not os.fstat(in_file.fileno()).st_size:
 164                     count = ' count=0'
 165                 else:
 166                     count = ''
 167                 try:
 168                     p = self._buffered_exec_command('dd of=%s bs=%s%s' % (out_path, BUFSIZE, count), stdin=in_file)
 169                 except OSError:
 170                     raise AnsibleError("chroot connection requires dd command in the chroot")
 171                 try:
 172                     stdout, stderr = p.communicate()
 173                 except Exception:
 174                     traceback.print_exc()
 175                     raise AnsibleError("failed to transfer file %s to %s" % (in_path, out_path))
 176                 if p.returncode != 0:
 177                     raise AnsibleError("failed to transfer file %s to %s:\n%s\n%s" % (in_path, out_path, stdout, stderr))
 178         except IOError:
 179             raise AnsibleError("file or module does not exist at: %s" % in_path)
 180 
 181     def fetch_file(self, in_path, out_path):
 182         ''' fetch a file from chroot to local '''
 183         super(Connection, self).fetch_file(in_path, out_path)
 184         display.vvv("FETCH %s TO %s" % (in_path, out_path), host=self.chroot)
 185 
 186         in_path = shlex_quote(self._prefix_login_path(in_path))
 187         try:
 188             p = self._buffered_exec_command('dd if=%s bs=%s' % (in_path, BUFSIZE))
 189         except OSError:
 190             raise AnsibleError("chroot connection requires dd command in the chroot")
 191 
 192         with open(to_bytes(out_path, errors='surrogate_or_strict'), 'wb+') as out_file:
 193             try:
 194                 chunk = p.stdout.read(BUFSIZE)
 195                 while chunk:
 196                     out_file.write(chunk)
 197                     chunk = p.stdout.read(BUFSIZE)
 198             except Exception:
 199                 traceback.print_exc()
 200                 raise AnsibleError("failed to transfer file %s to %s" % (in_path, out_path))
 201             stdout, stderr = p.communicate()
 202             if p.returncode != 0:
 203                 raise AnsibleError("failed to transfer file %s to %s:\n%s\n%s" % (in_path, out_path, stdout, stderr))
 204 
 205     def close(self):
 206         ''' terminate the connection; nothing to do here '''
 207         super(Connection, self).close()
 208         self._connected = False
