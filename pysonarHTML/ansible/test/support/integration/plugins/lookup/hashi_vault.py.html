 131 ANSIBLE_HASHI_VAULT_ADDR = 'http://127.0.0.1:8200'

 137 class HashiVault:
 138     def __init__(self, **kwargs):
 139 
 140         self.url = kwargs.get('url', ANSIBLE_HASHI_VAULT_ADDR)
 141         self.namespace = kwargs.get('namespace', None)
 142         self.avail_auth_method = ['approle', 'userpass', 'ldap']
 143 
 144         # split secret arg, which has format 'secret/hello:value' into secret='secret/hello' and secret_field='value'
 145         s = kwargs.get('secret')
 146         if s is None:
 147             raise AnsibleError("No secret specified for hashi_vault lookup")
 148 
 149         s_f = s.rsplit(':', 1)
 150         self.secret = s_f[0]
 151         if len(s_f) >= 2:
 152             self.secret_field = s_f[1]
 153         else:
 154             self.secret_field = ''
 155 
 156         self.verify = self.boolean_or_cacert(kwargs.get('validate_certs', True), kwargs.get('cacert', ''))
 157 
 158         # If a particular backend is asked for (and its method exists) we call it, otherwise drop through to using
 159         # token auth. This means if a particular auth backend is requested and a token is also given, then we
 160         # ignore the token and attempt authentication against the specified backend.
 161         #
 162         # to enable a new auth backend, simply add a new 'def auth_<type>' method below.
 163         #
 164         self.auth_method = kwargs.get('auth_method', os.environ.get('VAULT_AUTH_METHOD'))
 165         self.verify = self.boolean_or_cacert(kwargs.get('validate_certs', True), kwargs.get('cacert', ''))
 166         if self.auth_method and self.auth_method != 'token':
 167             try:
 168                 if self.namespace is not None:
 169                     self.client = hvac.Client(url=self.url, verify=self.verify, namespace=self.namespace)
 170                 else:
 171                     self.client = hvac.Client(url=self.url, verify=self.verify)
 172                 # prefixing with auth_ to limit which methods can be accessed
 173                 getattr(self, 'auth_' + self.auth_method)(**kwargs)
 174             except AttributeError:
 175                 raise AnsibleError("Authentication method '%s' not supported."
 176                                    " Available options are %r" % (self.auth_method, self.avail_auth_method))
 177         else:
 178             self.token = kwargs.get('token', os.environ.get('VAULT_TOKEN', None))
 179             if self.token is None and os.environ.get('HOME'):
 180                 token_filename = os.path.join(
 181                     os.environ.get('HOME'),
 182                     '.vault-token'
 183                 )
 184                 if os.path.exists(token_filename):
 185                     with open(token_filename) as token_file:
 186                         self.token = token_file.read().strip()
 187 
 188             if self.token is None:
 189                 raise AnsibleError("No Vault Token specified")
 190 
 191             if self.namespace is not None:
 192                 self.client = hvac.Client(url=self.url, token=self.token, verify=self.verify, namespace=self.namespace)
 193             else:
 194                 self.client = hvac.Client(url=self.url, token=self.token, verify=self.verify)
 195 
 196         if not self.client.is_authenticated():
 197             raise AnsibleError("Invalid Hashicorp Vault Token Specified for hashi_vault lookup")
 198 
 199     def get(self):
 200         data = self.client.read(self.secret)
 201 
 202         # Check response for KV v2 fields and flatten nested secret data.
 203         #
 204         # https://vaultproject.io/api/secret/kv/kv-v2.html#sample-response-1
 205         try:
 206             # sentinel field checks
 207             check_dd = data['data']['data']
 208             check_md = data['data']['metadata']
 209             # unwrap nested data
 210             data = data['data']
 211         except KeyError:
 212             pass
 213 
 214         if data is None:
 215             raise AnsibleError("The secret %s doesn't seem to exist for hashi_vault lookup" % self.secret)
 216 
 217         if self.secret_field == '':
 218             return data['data']
 219 
 220         if self.secret_field not in data['data']:
 221             raise AnsibleError("The secret %s does not contain the field '%s'. for hashi_vault lookup" % (self.secret, self.secret_field))
 222 
 223         return data['data'][self.secret_field]
 224 
 225     def check_params(self, **kwargs):
 226         username = kwargs.get('username')
 227         if username is None:
 228             raise AnsibleError("Authentication method %s requires a username" % self.auth_method)
 229 
 230         password = kwargs.get('password')
 231         if password is None:
 232             raise AnsibleError("Authentication method %s requires a password" % self.auth_method)
 233 
 234         mount_point = kwargs.get('mount_point')
 235 
 236         return username, password, mount_point
 237 
 238     def auth_userpass(self, **kwargs):
 239         username, password, mount_point = self.check_params(**kwargs)
 240         if mount_point is None:
 241             mount_point = 'userpass'
 242 
 243         self.client.auth_userpass(username, password, mount_point=mount_point)
 244 
 245     def auth_ldap(self, **kwargs):
 246         username, password, mount_point = self.check_params(**kwargs)
 247         if mount_point is None:
 248             mount_point = 'ldap'
 249 
 250         self.client.auth.ldap.login(username, password, mount_point=mount_point)
 251 
 252     def boolean_or_cacert(self, validate_certs, cacert):
 253         validate_certs = boolean(validate_certs, strict=False)
 254         '''' return a bool or cacert '''
 255         if validate_certs is True:
 256             if cacert != '':
 257                 return cacert
 258             else:
 259                 return True
 260         else:
 261             return False
 262 
 263     def auth_approle(self, **kwargs):
 264         role_id = kwargs.get('role_id', os.environ.get('VAULT_ROLE_ID', None))
 265         if role_id is None:
 266             raise AnsibleError("Authentication method app role requires a role_id")
 267 
 268         secret_id = kwargs.get('secret_id', os.environ.get('VAULT_SECRET_ID', None))
 269         if secret_id is None:
 270             raise AnsibleError("Authentication method app role requires a secret_id")
 271 
 272         self.client.auth_approle(role_id, secret_id)

 275 class LookupModule(LookupBase):
 276     def run(self, terms, variables=None, **kwargs):
 277         if not HAS_HVAC:
 278             raise AnsibleError("Please pip install hvac to use the hashi_vault lookup module.")
 279 
 280         vault_args = terms[0].split()
 281         vault_dict = {}
 282         ret = []
 283 
 284         for param in vault_args:
 285             try:
 286                 key, value = param.split('=')
 287             except ValueError:
 288                 raise AnsibleError("hashi_vault lookup plugin needs key=value pairs, but received %s" % terms)
 289             vault_dict[key] = value
 290 
 291         if 'ca_cert' in vault_dict.keys():
 292             vault_dict['cacert'] = vault_dict['ca_cert']
 293             vault_dict.pop('ca_cert', None)
 294 
 295         vault_conn = HashiVault(**vault_dict)
 296 
 297         for term in terms:
 298             key = term.split()[0]
 299             value = vault_conn.get()
 300             ret.append(value)
 301 
 302         return ret
