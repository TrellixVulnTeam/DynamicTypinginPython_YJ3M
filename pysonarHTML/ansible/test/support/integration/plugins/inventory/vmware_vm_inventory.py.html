 134 class BaseVMwareInventory:
 135     def __init__(self, hostname, username, password, port, validate_certs, with_tags):
 136         self.hostname = hostname
 137         self.username = username
 138         self.password = password
 139         self.port = port
 140         self.with_tags = with_tags
 141         self.validate_certs = validate_certs
 142         self.content = None
 143         self.rest_content = None
 144 
 145     def do_login(self):
 146         """
 147         Check requirements and do login
 148         """
 149         self.check_requirements()
 150         self.content = self._login()
 151         if self.with_tags:
 152             self.rest_content = self._login_vapi()
 153 
 154     def _login_vapi(self):
 155         """
 156         Login to vCenter API using REST call
 157         Returns: connection object
 158 
 159         """
 160         session = requests.Session()
 161         session.verify = self.validate_certs
 162         if not self.validate_certs:
 163             # Disable warning shown at stdout
 164             requests.packages.urllib3.disable_warnings()
 165 
 166         server = self.hostname
 167         if self.port:
 168             server += ":" + str(self.port)
 169         client = create_vsphere_client(server=server,
 170                                        username=self.username,
 171                                        password=self.password,
 172                                        session=session)
 173         if client is None:
 174             raise AnsibleError("Failed to login to %s using %s" % (server, self.username))
 175         return client
 176 
 177     def _login(self):
 178         """
 179         Login to vCenter or ESXi server
 180         Returns: connection object
 181 
 182         """
 183         if self.validate_certs and not hasattr(ssl, 'SSLContext'):
 184             raise AnsibleError('pyVim does not support changing verification mode with python < 2.7.9. Either update '
 185                                'python or set validate_certs to false in configuration YAML file.')
 186 
 187         ssl_context = None
 188         if not self.validate_certs and hasattr(ssl, 'SSLContext'):
 189             ssl_context = ssl.SSLContext(ssl.PROTOCOL_SSLv23)
 190             ssl_context.verify_mode = ssl.CERT_NONE
 191 
 192         service_instance = None
 193         try:
 194             service_instance = connect.SmartConnect(host=self.hostname, user=self.username,
 195                                                     pwd=self.password, sslContext=ssl_context,
 196                                                     port=self.port)
 197         except vim.fault.InvalidLogin as e:
 198             raise AnsibleParserError("Unable to log on to vCenter or ESXi API at %s:%s as %s: %s" % (self.hostname, self.port, self.username, e.msg))
 199         except vim.fault.NoPermission as e:
 200             raise AnsibleParserError("User %s does not have required permission"
 201                                      " to log on to vCenter or ESXi API at %s:%s : %s" % (self.username, self.hostname, self.port, e.msg))
 202         except (requests.ConnectionError, ssl.SSLError) as e:
 203             raise AnsibleParserError("Unable to connect to vCenter or ESXi API at %s on TCP/%s: %s" % (self.hostname, self.port, e))
 204         except vmodl.fault.InvalidRequest as e:
 205             # Request is malformed
 206             raise AnsibleParserError("Failed to get a response from server %s:%s as "
 207                                      "request is malformed: %s" % (self.hostname, self.port, e.msg))
 208         except Exception as e:
 209             raise AnsibleParserError("Unknown error while connecting to vCenter or ESXi API at %s:%s : %s" % (self.hostname, self.port, e))
 210 
 211         if service_instance is None:
 212             raise AnsibleParserError("Unknown error while connecting to vCenter or ESXi API at %s:%s" % (self.hostname, self.port))
 213 
 214         atexit.register(connect.Disconnect, service_instance)
 215         return service_instance.RetrieveContent()

 217     def check_requirements(self):
 218         """ Check all requirements for this inventory are satisified"""
 219         if not HAS_REQUESTS:
 220             raise AnsibleParserError('Please install "requests" Python module as this is required'
 221                                      ' for VMware Guest dynamic inventory plugin.')
 222         elif not HAS_PYVMOMI:
 223             raise AnsibleParserError('Please install "PyVmomi" Python module as this is required'
 224                                      ' for VMware Guest dynamic inventory plugin.')
 225         if HAS_REQUESTS:
 226             # Pyvmomi 5.5 and onwards requires requests 2.3
 227             # https://github.com/vmware/pyvmomi/blob/master/requirements.txt
 228             required_version = (2, 3)
 229             requests_version = requests.__version__.split(".")[:2]
 230             try:
 231                 requests_major_minor = tuple(map(int, requests_version))
 232             except ValueError:
 233                 raise AnsibleParserError("Failed to parse 'requests' library version.")
 234 
 235             if requests_major_minor < required_version:
 236                 raise AnsibleParserError("'requests' library version should"
 237                                          " be >= %s, found: %s." % (".".join([str(w) for w in required_version]),
 238                                                                     requests.__version__))
 239 
 240         if not HAS_VSPHERE and self.with_tags:
 241             raise AnsibleError("Unable to find 'vSphere Automation SDK' Python library which is required."
 242                                " Please refer this URL for installation steps"
 243                                " - https://code.vmware.com/web/sdk/65/vsphere-automation-python")
 244 
 245         if not all([self.hostname, self.username, self.password]):
 246             raise AnsibleError("Missing one of the following : hostname, username, password. Please read "
 247                                "the documentation for more information.")

 249     def _get_managed_objects_properties(self, vim_type, properties=None):
 250         """
 251         Look up a Managed Object Reference in vCenter / ESXi Environment
 252         :param vim_type: Type of vim object e.g, for datacenter - vim.Datacenter
 253         :param properties: List of properties related to vim object e.g. Name
 254         :return: local content object
 255         """
 256         # Get Root Folder
 257         root_folder = self.content.rootFolder
 258 
 259         if properties is None:
 260             properties = ['name']
 261 
 262         # Create Container View with default root folder
 263         mor = self.content.viewManager.CreateContainerView(root_folder, [vim_type], True)
 264 
 265         # Create Traversal spec
 266         traversal_spec = vmodl.query.PropertyCollector.TraversalSpec(
 267             name="traversal_spec",
 268             path='view',
 269             skip=False,
 270             type=vim.view.ContainerView
 271         )
 272 
 273         # Create Property Spec
 274         property_spec = vmodl.query.PropertyCollector.PropertySpec(
 275             type=vim_type,  # Type of object to retrieved
 276             all=False,
 277             pathSet=properties
 278         )
 279 
 280         # Create Object Spec
 281         object_spec = vmodl.query.PropertyCollector.ObjectSpec(
 282             obj=mor,
 283             skip=True,
 284             selectSet=[traversal_spec]
 285         )
 286 
 287         # Create Filter Spec
 288         filter_spec = vmodl.query.PropertyCollector.FilterSpec(
 289             objectSet=[object_spec],
 290             propSet=[property_spec],
 291             reportMissingObjectsInResults=False
 292         )
 293 
 294         return self.content.propertyCollector.RetrieveContents([filter_spec])

 296     @staticmethod
 297     def _get_object_prop(vm, attributes):
 298         """Safely get a property or return None"""
 299         result = vm
 300         for attribute in attributes:
 301             try:
 302                 result = getattr(result, attribute)
 303             except (AttributeError, IndexError):
 304                 return None
 305         return result

 312     def verify_file(self, path):
 313         """
 314         Verify plugin configuration file and mark this plugin active
 315         Args:
 316             path: Path of configuration YAML file
 317         Returns: True if everything is correct, else False
 318         """
 319         valid = False
 320         if super(InventoryModule, self).verify_file(path):
 321             if path.endswith(('vmware.yaml', 'vmware.yml', 'vmware_vm_inventory.yaml', 'vmware_vm_inventory.yml')):
 322                 valid = True
 323 
 324         return valid

 326     def parse(self, inventory, loader, path, cache=True):
 327         """
 328         Parses the inventory file
 329         """
 330         super(InventoryModule, self).parse(inventory, loader, path, cache=cache)
 331 
 332         cache_key = self.get_cache_key(path)
 333 
 334         config_data = self._read_config_data(path)
 335 
 336         # set _options from config data
 337         self._consume_options(config_data)
 338 
 339         self.pyv = BaseVMwareInventory(
 340             hostname=self.get_option('hostname'),
 341             username=self.get_option('username'),
 342             password=self.get_option('password'),
 343             port=self.get_option('port'),
 344             with_tags=self.get_option('with_tags'),
 345             validate_certs=self.get_option('validate_certs')
 346         )
 347 
 348         self.pyv.do_login()
 349 
 350         self.pyv.check_requirements()
 351 
 352         source_data = None
 353         if cache:
 354             cache = self.get_option('cache')
 355 
 356         update_cache = False
 357         if cache:
 358             try:
 359                 source_data = self._cache[cache_key]
 360             except KeyError:
 361                 update_cache = True
 362 
 363         using_current_cache = cache and not update_cache
 364         cacheable_results = self._populate_from_source(source_data, using_current_cache)
 365 
 366         if update_cache:
 367             self._cache[cache_key] = cacheable_results
 368 
 369     def _populate_from_cache(self, source_data):
 370         """ Populate cache using source data """
 371         hostvars = source_data.pop('_meta', {}).get('hostvars', {})
 372         for group in source_data:
 373             if group == 'all':
 374                 continue
 375             else:
 376                 self.inventory.add_group(group)
 377                 hosts = source_data[group].get('hosts', [])
 378                 for host in hosts:
 379                     self._populate_host_vars([host], hostvars.get(host, {}), group)
 380                 self.inventory.add_child('all', group)
 381 
 382     def _populate_from_source(self, source_data, using_current_cache):
 383         """
 384         Populate inventory data from direct source
 385 
 386         """
 387         if using_current_cache:
 388             self._populate_from_cache(source_data)
 389             return source_data
 390 
 391         cacheable_results = {'_meta': {'hostvars': {}}}
 392         hostvars = {}
 393         objects = self.pyv._get_managed_objects_properties(vim_type=vim.VirtualMachine,
 394                                                            properties=['name'])
 395 
 396         if self.pyv.with_tags:
 397             tag_svc = self.pyv.rest_content.tagging.Tag
 398             tag_association = self.pyv.rest_content.tagging.TagAssociation
 399 
 400             tags_info = dict()
 401             tags = tag_svc.list()
 402             for tag in tags:
 403                 tag_obj = tag_svc.get(tag)
 404                 tags_info[tag_obj.id] = tag_obj.name
 405                 if tag_obj.name not in cacheable_results:
 406                     cacheable_results[tag_obj.name] = {'hosts': []}
 407                     self.inventory.add_group(tag_obj.name)
 408 
 409         for vm_obj in objects:
 410             for vm_obj_property in vm_obj.propSet:
 411                 # VMware does not provide a way to uniquely identify VM by its name
 412                 # i.e. there can be two virtual machines with same name
 413                 # Appending "_" and VMware UUID to make it unique
 414                 if not vm_obj.obj.config:
 415                     # Sometime orphaned VMs return no configurations
 416                     continue
 417 
 418                 current_host = vm_obj_property.val + "_" + vm_obj.obj.config.uuid
 419 
 420                 if current_host not in hostvars:
 421                     hostvars[current_host] = {}
 422                     self.inventory.add_host(current_host)
 423 
 424                     host_ip = vm_obj.obj.guest.ipAddress
 425                     if host_ip:
 426                         self.inventory.set_variable(current_host, 'ansible_host', host_ip)
 427 
 428                     self._populate_host_properties(vm_obj, current_host)
 429 
 430                     # Only gather facts related to tag if vCloud and vSphere is installed.
 431                     if HAS_VSPHERE and self.pyv.with_tags:
 432                         # Add virtual machine to appropriate tag group
 433                         vm_mo_id = vm_obj.obj._GetMoId()
 434                         vm_dynamic_id = DynamicID(type='VirtualMachine', id=vm_mo_id)
 435                         attached_tags = tag_association.list_attached_tags(vm_dynamic_id)
 436 
 437                         for tag_id in attached_tags:
 438                             self.inventory.add_child(tags_info[tag_id], current_host)
 439                             cacheable_results[tags_info[tag_id]]['hosts'].append(current_host)
 440 
 441                     # Based on power state of virtual machine
 442                     vm_power = str(vm_obj.obj.summary.runtime.powerState)
 443                     if vm_power not in cacheable_results:
 444                         cacheable_results[vm_power] = {'hosts': []}
 445                         self.inventory.add_group(vm_power)
 446                     cacheable_results[vm_power]['hosts'].append(current_host)
 447                     self.inventory.add_child(vm_power, current_host)
 448 
 449                     # Based on guest id
 450                     vm_guest_id = vm_obj.obj.config.guestId
 451                     if vm_guest_id and vm_guest_id not in cacheable_results:
 452                         cacheable_results[vm_guest_id] = {'hosts': []}
 453                         self.inventory.add_group(vm_guest_id)
 454                     cacheable_results[vm_guest_id]['hosts'].append(current_host)
 455                     self.inventory.add_child(vm_guest_id, current_host)
 456 
 457         for host in hostvars:
 458             h = self.inventory.get_host(host)
 459             cacheable_results['_meta']['hostvars'][h.name] = h.vars
 460 
 461         return cacheable_results
 462 
 463     def _populate_host_properties(self, vm_obj, current_host):
 464         # Load VM properties in host_vars
 465         vm_properties = self.get_option('properties') or []
 466 
 467         field_mgr = self.pyv.content.customFieldsManager.field
 468 
 469         for vm_prop in vm_properties:
 470             if vm_prop == 'customValue':
 471                 for cust_value in vm_obj.obj.customValue:
 472                     self.inventory.set_variable(current_host,
 473                                                 [y.name for y in field_mgr if y.key == cust_value.key][0],
 474                                                 cust_value.value)
 475             else:
 476                 vm_value = self.pyv._get_object_prop(vm_obj.obj, vm_prop.split("."))
 477                 self.inventory.set_variable(current_host, vm_prop, vm_value)
