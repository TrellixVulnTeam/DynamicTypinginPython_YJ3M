 102 class InventoryModule(BaseInventoryPlugin, Cacheable, Constructable):
 103     ''' Host inventory parser for ansible using foreman as source. '''
 104 
 105     NAME = 'foreman'
 106 
 107     def __init__(self):
 108 
 109         super(InventoryModule, self).__init__()
 110 
 111         # from config
 112         self.foreman_url = None
 113 
 114         self.session = None
 115         self.cache_key = None
 116         self.use_cache = None
 117 
 118     def verify_file(self, path):
 119 
 120         valid = False
 121         if super(InventoryModule, self).verify_file(path):
 122             if path.endswith(('foreman.yaml', 'foreman.yml')):
 123                 valid = True
 124             else:
 125                 self.display.vvv('Skipping due to inventory source not ending in "foreman.yaml" nor "foreman.yml"')
 126         return valid
 127 
 128     def _get_session(self):
 129         if not self.session:
 130             self.session = requests.session()
 131             self.session.auth = HTTPBasicAuth(self.get_option('user'), to_bytes(self.get_option('password')))
 132             self.session.verify = self.get_option('validate_certs')
 133         return self.session
 134 
 135     def _get_json(self, url, ignore_errors=None):
 136 
 137         if not self.use_cache or url not in self._cache.get(self.cache_key, {}):
 138 
 139             if self.cache_key not in self._cache:
 140                 self._cache[self.cache_key] = {url: ''}
 141 
 142             results = []
 143             s = self._get_session()
 144             params = {'page': 1, 'per_page': 250}
 145             while True:
 146                 ret = s.get(url, params=params)
 147                 if ignore_errors and ret.status_code in ignore_errors:
 148                     break
 149                 ret.raise_for_status()
 150                 json = ret.json()
 151 
 152                 # process results
 153                 # FIXME: This assumes 'return type' matches a specific query,
 154                 #        it will break if we expand the queries and they dont have different types
 155                 if 'results' not in json:
 156                     # /hosts/:id dos not have a 'results' key
 157                     results = json
 158                     break
 159                 elif isinstance(json['results'], MutableMapping):
 160                     # /facts are returned as dict in 'results'
 161                     results = json['results']
 162                     break
 163                 else:
 164                     # /hosts 's 'results' is a list of all hosts, returned is paginated
 165                     results = results + json['results']
 166 
 167                     # check for end of paging
 168                     if len(results) >= json['subtotal']:
 169                         break
 170                     if len(json['results']) == 0:
 171                         self.display.warning("Did not make any progress during loop. expected %d got %d" % (json['subtotal'], len(results)))
 172                         break
 173 
 174                     # get next page
 175                     params['page'] += 1
 176 
 177             self._cache[self.cache_key][url] = results
 178 
 179         return self._cache[self.cache_key][url]
 180 
 181     def _get_hosts(self):
 182         return self._get_json("%s/api/v2/hosts" % self.foreman_url)
 183 
 184     def _get_all_params_by_id(self, hid):
 185         url = "%s/api/v2/hosts/%s" % (self.foreman_url, hid)
 186         ret = self._get_json(url, [404])
 187         if not ret or not isinstance(ret, MutableMapping) or not ret.get('all_parameters', False):
 188             return {}
 189         return ret.get('all_parameters')
 190 
 191     def _get_facts_by_id(self, hid):
 192         url = "%s/api/v2/hosts/%s/facts" % (self.foreman_url, hid)
 193         return self._get_json(url)
 194 
 195     def _get_host_data_by_id(self, hid):
 196         url = "%s/api/v2/hosts/%s" % (self.foreman_url, hid)
 197         return self._get_json(url)
 198 
 199     def _get_facts(self, host):
 200         """Fetch all host facts of the host"""
 201 
 202         ret = self._get_facts_by_id(host['id'])
 203         if len(ret.values()) == 0:
 204             facts = {}
 205         elif len(ret.values()) == 1:
 206             facts = list(ret.values())[0]
 207         else:
 208             raise ValueError("More than one set of facts returned for '%s'" % host)
 209         return facts
 210 
 211     def _populate(self):
 212 
 213         for host in self._get_hosts():
 214 
 215             if host.get('name'):
 216                 host_name = self.inventory.add_host(host['name'])
 217 
 218                 # create directly mapped groups
 219                 group_name = host.get('hostgroup_title', host.get('hostgroup_name'))
 220                 if group_name:
 221                     group_name = to_safe_group_name('%s%s' % (self.get_option('group_prefix'), group_name.lower().replace(" ", "")))
 222                     group_name = self.inventory.add_group(group_name)
 223                     self.inventory.add_child(group_name, host_name)
 224 
 225                 # set host vars from host info
 226                 try:
 227                     for k, v in host.items():
 228                         if k not in ('name', 'hostgroup_title', 'hostgroup_name'):
 229                             try:
 230                                 self.inventory.set_variable(host_name, self.get_option('vars_prefix') + k, v)
 231                             except ValueError as e:
 232                                 self.display.warning("Could not set host info hostvar for %s, skipping %s: %s" % (host, k, to_text(e)))
 233                 except ValueError as e:
 234                     self.display.warning("Could not get host info for %s, skipping: %s" % (host_name, to_text(e)))
 235 
 236                 # set host vars from params
 237                 if self.get_option('want_params'):
 238                     for p in self._get_all_params_by_id(host['id']):
 239                         try:
 240                             self.inventory.set_variable(host_name, p['name'], p['value'])
 241                         except ValueError as e:
 242                             self.display.warning("Could not set hostvar %s to '%s' for the '%s' host, skipping:  %s" %
 243                                                  (p['name'], to_native(p['value']), host, to_native(e)))
 244 
 245                 # set host vars from facts
 246                 if self.get_option('want_facts'):
 247                     self.inventory.set_variable(host_name, 'foreman_facts', self._get_facts(host))
 248 
 249                 # create group for host collections
 250                 if self.get_option('want_hostcollections'):
 251                     host_data = self._get_host_data_by_id(host['id'])
 252                     hostcollections = host_data.get('host_collections')
 253                     if hostcollections:
 254                         # Create Ansible groups for host collections
 255                         for hostcollection in hostcollections:
 256                             try:
 257                                 hostcollection_group = to_safe_group_name('%shostcollection_%s' % (self.get_option('group_prefix'),
 258                                                                           hostcollection['name'].lower().replace(" ", "")))
 259                                 hostcollection_group = self.inventory.add_group(hostcollection_group)
 260                                 self.inventory.add_child(hostcollection_group, host_name)
 261                             except ValueError as e:
 262                                 self.display.warning("Could not create groups for host collections for %s, skipping: %s" % (host_name, to_text(e)))
 263 
 264                 # put ansible_ssh_host as hostvar
 265                 if self.get_option('want_ansible_ssh_host'):
 266                     for key in ('ip', 'ipv4', 'ipv6'):
 267                         if host.get(key):
 268                             try:
 269                                 self.inventory.set_variable(host_name, 'ansible_ssh_host', host[key])
 270                                 break
 271                             except ValueError as e:
 272                                 self.display.warning("Could not set hostvar ansible_ssh_host to '%s' for the '%s' host, skipping: %s" %
 273                                                      (host[key], host_name, to_text(e)))
 274 
 275                 strict = self.get_option('strict')
 276 
 277                 hostvars = self.inventory.get_host(host_name).get_vars()
 278                 self._set_composite_vars(self.get_option('compose'), hostvars, host_name, strict)
 279                 self._add_host_to_composed_groups(self.get_option('groups'), hostvars, host_name, strict)
 280                 self._add_host_to_keyed_groups(self.get_option('keyed_groups'), hostvars, host_name, strict)
 281 
 282     def parse(self, inventory, loader, path, cache=True):
 283 
 284         super(InventoryModule, self).parse(inventory, loader, path)
 285 
 286         # read config from file, this sets 'options'
 287         self._read_config_data(path)
 288 
 289         # get connection host
 290         self.foreman_url = self.get_option('url')
 291         self.cache_key = self.get_cache_key(path)
 292         self.use_cache = cache and self.get_option('cache')
 293 
 294         # actually populate inventory
 295         self._populate()
