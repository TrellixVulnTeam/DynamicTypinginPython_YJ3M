 252 def get_encoding_id(cursor, encoding):
 253     query = "SELECT pg_char_to_encoding(%(encoding)s) AS encoding_id;"
 254     cursor.execute(query, {'encoding': encoding})
 255     return cursor.fetchone()['encoding_id']

 258 def get_db_info(cursor, db):
 259     query = """
 260     SELECT rolname AS owner,
 261     pg_encoding_to_char(encoding) AS encoding, encoding AS encoding_id,
 262     datcollate AS lc_collate, datctype AS lc_ctype, pg_database.datconnlimit AS conn_limit,
 263     spcname AS tablespace
 264     FROM pg_database
 265     JOIN pg_roles ON pg_roles.oid = pg_database.datdba
 266     JOIN pg_tablespace ON pg_tablespace.oid = pg_database.dattablespace
 267     WHERE datname = %(db)s
 268     """
 269     cursor.execute(query, {'db': db})
 270     return cursor.fetchone()

 289 def db_create(cursor, db, owner, template, encoding, lc_collate, lc_ctype, conn_limit, tablespace):
 290     params = dict(enc=encoding, collate=lc_collate, ctype=lc_ctype, conn_limit=conn_limit, tablespace=tablespace)
 291     if not db_exists(cursor, db):
 292         query_fragments = ['CREATE DATABASE %s' % pg_quote_identifier(db, 'database')]
 293         if owner:
 294             query_fragments.append('OWNER "%s"' % owner)
 295         if template:
 296             query_fragments.append('TEMPLATE %s' % pg_quote_identifier(template, 'database'))
 297         if encoding:
 298             query_fragments.append('ENCODING %(enc)s')
 299         if lc_collate:
 300             query_fragments.append('LC_COLLATE %(collate)s')
 301         if lc_ctype:
 302             query_fragments.append('LC_CTYPE %(ctype)s')
 303         if tablespace:
 304             query_fragments.append('TABLESPACE %s' % pg_quote_identifier(tablespace, 'tablespace'))
 305         if conn_limit:
 306             query_fragments.append("CONNECTION LIMIT %(conn_limit)s" % {"conn_limit": conn_limit})
 307         query = ' '.join(query_fragments)
 308         executed_commands.append(cursor.mogrify(query, params))
 309         cursor.execute(query, params)
 310         return True
 311     else:
 312         db_info = get_db_info(cursor, db)
 313         if (encoding and get_encoding_id(cursor, encoding) != db_info['encoding_id']):
 314             raise NotSupportedError(
 315                 'Changing database encoding is not supported. '
 316                 'Current encoding: %s' % db_info['encoding']
 317             )
 318         elif lc_collate and lc_collate != db_info['lc_collate']:
 319             raise NotSupportedError(
 320                 'Changing LC_COLLATE is not supported. '
 321                 'Current LC_COLLATE: %s' % db_info['lc_collate']
 322             )
 323         elif lc_ctype and lc_ctype != db_info['lc_ctype']:
 324             raise NotSupportedError(
 325                 'Changing LC_CTYPE is not supported.'
 326                 'Current LC_CTYPE: %s' % db_info['lc_ctype']
 327             )
 328         else:
 329             changed = False
 330 
 331             if owner and owner != db_info['owner']:
 332                 changed = set_owner(cursor, db, owner)
 333 
 334             if conn_limit and conn_limit != str(db_info['conn_limit']):
 335                 changed = set_conn_limit(cursor, db, conn_limit)
 336 
 337             if tablespace and tablespace != db_info['tablespace']:
 338                 changed = set_tablespace(cursor, db, tablespace)
 339 
 340             return changed

 343 def db_matches(cursor, db, owner, template, encoding, lc_collate, lc_ctype, conn_limit, tablespace):
 344     if not db_exists(cursor, db):
 345         return False
 346     else:
 347         db_info = get_db_info(cursor, db)
 348         if (encoding and get_encoding_id(cursor, encoding) != db_info['encoding_id']):
 349             return False
 350         elif lc_collate and lc_collate != db_info['lc_collate']:
 351             return False
 352         elif lc_ctype and lc_ctype != db_info['lc_ctype']:
 353             return False
 354         elif owner and owner != db_info['owner']:
 355             return False
 356         elif conn_limit and conn_limit != str(db_info['conn_limit']):
 357             return False
 358         elif tablespace and tablespace != db_info['tablespace']:
 359             return False
 360         else:
 361             return True

 364 def db_dump(module, target, target_opts="",
 365             db=None,
 366             dump_extra_args=None,
 367             user=None,
 368             password=None,
 369             host=None,
 370             port=None,
 371             **kw):
 372 
 373     flags = login_flags(db, host, port, user, db_prefix=False)
 374     cmd = module.get_bin_path('pg_dump', True)
 375     comp_prog_path = None
 376 
 377     if os.path.splitext(target)[-1] == '.tar':
 378         flags.append(' --format=t')
 379     elif os.path.splitext(target)[-1] == '.pgc':
 380         flags.append(' --format=c')
 381     if os.path.splitext(target)[-1] == '.gz':
 382         if module.get_bin_path('pigz'):
 383             comp_prog_path = module.get_bin_path('pigz', True)
 384         else:
 385             comp_prog_path = module.get_bin_path('gzip', True)
 386     elif os.path.splitext(target)[-1] == '.bz2':
 387         comp_prog_path = module.get_bin_path('bzip2', True)
 388     elif os.path.splitext(target)[-1] == '.xz':
 389         comp_prog_path = module.get_bin_path('xz', True)
 390 
 391     cmd += "".join(flags)
 392 
 393     if dump_extra_args:
 394         cmd += " {0} ".format(dump_extra_args)
 395 
 396     if target_opts:
 397         cmd += " {0} ".format(target_opts)
 398 
 399     if comp_prog_path:
 400         # Use a fifo to be notified of an error in pg_dump
 401         # Using shell pipe has no way to return the code of the first command
 402         # in a portable way.
 403         fifo = os.path.join(module.tmpdir, 'pg_fifo')
 404         os.mkfifo(fifo)
 405         cmd = '{1} <{3} > {2} & {0} >{3}'.format(cmd, comp_prog_path, shlex_quote(target), fifo)
 406     else:
 407         cmd = '{0} > {1}'.format(cmd, shlex_quote(target))
 408 
 409     return do_with_password(module, cmd, password)

 412 def db_restore(module, target, target_opts="",
 413                db=None,
 414                user=None,
 415                password=None,
 416                host=None,
 417                port=None,
 418                **kw):
 419 
 420     flags = login_flags(db, host, port, user)
 421     comp_prog_path = None
 422     cmd = module.get_bin_path('psql', True)
 423 
 424     if os.path.splitext(target)[-1] == '.sql':
 425         flags.append(' --file={0}'.format(target))
 426 
 427     elif os.path.splitext(target)[-1] == '.tar':
 428         flags.append(' --format=Tar')
 429         cmd = module.get_bin_path('pg_restore', True)
 430 
 431     elif os.path.splitext(target)[-1] == '.pgc':
 432         flags.append(' --format=Custom')
 433         cmd = module.get_bin_path('pg_restore', True)
 434 
 435     elif os.path.splitext(target)[-1] == '.gz':
 436         comp_prog_path = module.get_bin_path('zcat', True)
 437 
 438     elif os.path.splitext(target)[-1] == '.bz2':
 439         comp_prog_path = module.get_bin_path('bzcat', True)
 440 
 441     elif os.path.splitext(target)[-1] == '.xz':
 442         comp_prog_path = module.get_bin_path('xzcat', True)
 443 
 444     cmd += "".join(flags)
 445     if target_opts:
 446         cmd += " {0} ".format(target_opts)
 447 
 448     if comp_prog_path:
 449         env = os.environ.copy()
 450         if password:
 451             env = {"PGPASSWORD": password}
 452         p1 = subprocess.Popen([comp_prog_path, target], stdout=subprocess.PIPE, stderr=subprocess.PIPE)
 453         p2 = subprocess.Popen(cmd, stdin=p1.stdout, stdout=subprocess.PIPE, stderr=subprocess.PIPE, shell=True, env=env)
 454         (stdout2, stderr2) = p2.communicate()
 455         p1.stdout.close()
 456         p1.wait()
 457         if p1.returncode != 0:
 458             stderr1 = p1.stderr.read()
 459             return p1.returncode, '', stderr1, 'cmd: ****'
 460         else:
 461             return p2.returncode, '', stderr2, 'cmd: ****'
 462     else:
 463         cmd = '{0} < {1}'.format(cmd, shlex_quote(target))
 464 
 465     return do_with_password(module, cmd, password)

 468 def login_flags(db, host, port, user, db_prefix=True):
 469     """
 470     returns a list of connection argument strings each prefixed
 471     with a space and quoted where necessary to later be combined
 472     in a single shell string with `"".join(rv)`
 473 
 474     db_prefix determines if "--dbname" is prefixed to the db argument,
 475     since the argument was introduced in 9.3.
 476     """
 477     flags = []
 478     if db:
 479         if db_prefix:
 480             flags.append(' --dbname={0}'.format(shlex_quote(db)))
 481         else:
 482             flags.append(' {0}'.format(shlex_quote(db)))
 483     if host:
 484         flags.append(' --host={0}'.format(host))
 485     if port:
 486         flags.append(' --port={0}'.format(port))
 487     if user:
 488         flags.append(' --username={0}'.format(user))
 489     return flags

 492 def do_with_password(module, cmd, password):
 493     env = {}
 494     if password:
 495         env = {"PGPASSWORD": password}
 496     executed_commands.append(cmd)
 497     rc, stderr, stdout = module.run_command(cmd, use_unsafe_shell=True, environ_update=env)
 498     return rc, stderr, stdout, cmd

 514 def main():
 515     argument_spec = pgutils.postgres_common_argument_spec()
 516     argument_spec.update(
 517         db=dict(type='str', required=True, aliases=['name']),
 518         owner=dict(type='str', default=''),
 519         template=dict(type='str', default=''),
 520         encoding=dict(type='str', default=''),
 521         lc_collate=dict(type='str', default=''),
 522         lc_ctype=dict(type='str', default=''),
 523         state=dict(type='str', default='present', choices=['absent', 'dump', 'present', 'restore']),
 524         target=dict(type='path', default=''),
 525         target_opts=dict(type='str', default=''),
 526         maintenance_db=dict(type='str', default="postgres"),
 527         session_role=dict(type='str'),
 528         conn_limit=dict(type='str', default=''),
 529         tablespace=dict(type='path', default=''),
 530         dump_extra_args=dict(type='str', default=None),
 531     )
 532 
 533     module = AnsibleModule(
 534         argument_spec=argument_spec,
 535         supports_check_mode=True
 536     )
 537 
 538     db = module.params["db"]
 539     owner = module.params["owner"]
 540     template = module.params["template"]
 541     encoding = module.params["encoding"]
 542     lc_collate = module.params["lc_collate"]
 543     lc_ctype = module.params["lc_ctype"]
 544     target = module.params["target"]
 545     target_opts = module.params["target_opts"]
 546     state = module.params["state"]
 547     changed = False
 548     maintenance_db = module.params['maintenance_db']
 549     session_role = module.params["session_role"]
 550     conn_limit = module.params['conn_limit']
 551     tablespace = module.params['tablespace']
 552     dump_extra_args = module.params['dump_extra_args']
 553 
 554     raw_connection = state in ("dump", "restore")
 555 
 556     if not raw_connection:
 557         pgutils.ensure_required_libs(module)
 558 
 559     # To use defaults values, keyword arguments must be absent, so
 560     # check which values are empty and don't include in the **kw
 561     # dictionary
 562     params_map = {
 563         "login_host": "host",
 564         "login_user": "user",
 565         "login_password": "password",
 566         "port": "port",
 567         "ssl_mode": "sslmode",
 568         "ca_cert": "sslrootcert"
 569     }
 570     kw = dict((params_map[k], v) for (k, v) in iteritems(module.params)
 571               if k in params_map and v != '' and v is not None)
 572 
 573     # If a login_unix_socket is specified, incorporate it here.
 574     is_localhost = "host" not in kw or kw["host"] == "" or kw["host"] == "localhost"
 575 
 576     if is_localhost and module.params["login_unix_socket"] != "":
 577         kw["host"] = module.params["login_unix_socket"]
 578 
 579     if target == "":
 580         target = "{0}/{1}.sql".format(os.getcwd(), db)
 581         target = os.path.expanduser(target)
 582 
 583     if not raw_connection:
 584         try:
 585             db_connection = psycopg2.connect(database=maintenance_db, **kw)
 586 
 587             # Enable autocommit so we can create databases
 588             if psycopg2.__version__ >= '2.4.2':
 589                 db_connection.autocommit = True
 590             else:
 591                 db_connection.set_isolation_level(psycopg2.extensions.ISOLATION_LEVEL_AUTOCOMMIT)
 592             cursor = db_connection.cursor(cursor_factory=psycopg2.extras.DictCursor)
 593 
 594         except TypeError as e:
 595             if 'sslrootcert' in e.args[0]:
 596                 module.fail_json(msg='Postgresql server must be at least version 8.4 to support sslrootcert. Exception: {0}'.format(to_native(e)),
 597                                  exception=traceback.format_exc())
 598             module.fail_json(msg="unable to connect to database: %s" % to_native(e), exception=traceback.format_exc())
 599 
 600         except Exception as e:
 601             module.fail_json(msg="unable to connect to database: %s" % to_native(e), exception=traceback.format_exc())
 602 
 603         if session_role:
 604             try:
 605                 cursor.execute('SET ROLE "%s"' % session_role)
 606             except Exception as e:
 607                 module.fail_json(msg="Could not switch role: %s" % to_native(e), exception=traceback.format_exc())
 608 
 609     try:
 610         if module.check_mode:
 611             if state == "absent":
 612                 changed = db_exists(cursor, db)
 613             elif state == "present":
 614                 changed = not db_matches(cursor, db, owner, template, encoding, lc_collate, lc_ctype, conn_limit, tablespace)
 615             module.exit_json(changed=changed, db=db, executed_commands=executed_commands)
 616 
 617         if state == "absent":
 618             try:
 619                 changed = db_delete(cursor, db)
 620             except SQLParseError as e:
 621                 module.fail_json(msg=to_native(e), exception=traceback.format_exc())
 622 
 623         elif state == "present":
 624             try:
 625                 changed = db_create(cursor, db, owner, template, encoding, lc_collate, lc_ctype, conn_limit, tablespace)
 626             except SQLParseError as e:
 627                 module.fail_json(msg=to_native(e), exception=traceback.format_exc())
 628 
 629         elif state in ("dump", "restore"):
 630             method = state == "dump" and db_dump or db_restore
 631             try:
 632                 if state == 'dump':
 633                     rc, stdout, stderr, cmd = method(module, target, target_opts, db, dump_extra_args, **kw)
 634                 else:
 635                     rc, stdout, stderr, cmd = method(module, target, target_opts, db, **kw)
 636 
 637                 if rc != 0:
 638                     module.fail_json(msg=stderr, stdout=stdout, rc=rc, cmd=cmd)
 639                 else:
 640                     module.exit_json(changed=True, msg=stdout, stderr=stderr, rc=rc, cmd=cmd,
 641                                      executed_commands=executed_commands)
 642             except SQLParseError as e:
 643                 module.fail_json(msg=to_native(e), exception=traceback.format_exc())
 644 
 645     except NotSupportedError as e:
 646         module.fail_json(msg=to_native(e), exception=traceback.format_exc())
 647     except SystemExit:
 648         # Avoid catching this on Python 2.4
 649         raise
 650     except Exception as e:
 651         module.fail_json(msg="Database query failed: %s" % to_native(e), exception=traceback.format_exc())
 652 
 653     module.exit_json(changed=changed, db=db, executed_commands=executed_commands)
