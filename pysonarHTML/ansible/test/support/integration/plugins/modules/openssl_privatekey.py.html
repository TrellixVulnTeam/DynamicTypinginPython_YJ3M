 339 class PrivateKeyBase(crypto_utils.OpenSSLObject):
 340 
 341     def __init__(self, module):
 342         super(PrivateKeyBase, self).__init__(
 343             module.params['path'],
 344             module.params['state'],
 345             module.params['force'],
 346             module.check_mode
 347         )
 348         self.size = module.params['size']
 349         self.passphrase = module.params['passphrase']
 350         self.cipher = module.params['cipher']
 351         self.privatekey = None
 352         self.fingerprint = {}
 353         self.format = module.params['format']
 354         self.format_mismatch = module.params['format_mismatch']
 355         self.privatekey_bytes = None
 356         self.return_content = module.params['return_content']
 357         self.regenerate = module.params['regenerate']
 358         if self.regenerate == 'always':
 359             self.force = True

 362         self.backup_file = None

 386     def generate(self, module):
 387         """Generate a keypair."""
 388 
 389         if not self.check(module, perms_required=False, ignore_conversion=True) or self.force:
 390             # Regenerate
 391             if self.backup:
 392                 self.backup_file = module.backup_local(self.path)
 393             self._generate_private_key()
 394             privatekey_data = self._get_private_key_data()
 395             if self.return_content:
 396                 self.privatekey_bytes = privatekey_data
 397             crypto_utils.write_file(module, privatekey_data, 0o600)
 398             self.changed = True
 399         elif not self.check(module, perms_required=False, ignore_conversion=False):
 400             # Convert
 401             if self.backup:
 402                 self.backup_file = module.backup_local(self.path)
 403             self._ensure_private_key_loaded()
 404             privatekey_data = self._get_private_key_data()
 405             if self.return_content:
 406                 self.privatekey_bytes = privatekey_data
 407             crypto_utils.write_file(module, privatekey_data, 0o600)
 408             self.changed = True
 409 
 410         self.fingerprint = self._get_fingerprint()
 411         file_args = module.load_file_common_arguments(module.params)
 412         if module.set_fs_attributes_if_different(file_args, False):
 413             self.changed = True
 414 
 415     def remove(self, module):
 416         if self.backup:
 417             self.backup_file = module.backup_local(self.path)
 418         super(PrivateKeyBase, self).remove(module)

 432     def check(self, module, perms_required=True, ignore_conversion=True):
 433         """Ensure the resource is in its desired state."""
 434 
 435         state_and_perms = super(PrivateKeyBase, self).check(module, perms_required=False)
 436 
 437         if not state_and_perms:
 438             # key does not exist
 439             return False
 440 
 441         if not self._check_passphrase():
 442             if self.regenerate in ('full_idempotence', 'always'):
 443                 return False
 444             module.fail_json(msg='Unable to read the key. The key is protected with a another passphrase / no passphrase or broken.'
 445                                  ' Will not proceed. To force regeneration, call the module with `generate`'
 446                                  ' set to `full_idempotence` or `always`, or with `force=yes`.')
 447 
 448         if self.regenerate != 'never':
 449             if not self._check_size_and_type():
 450                 if self.regenerate in ('partial_idempotence', 'full_idempotence', 'always'):
 451                     return False
 452                 module.fail_json(msg='Key has wrong type and/or size.'
 453                                      ' Will not proceed. To force regeneration, call the module with `generate`'
 454                                      ' set to `partial_idempotence`, `full_idempotence` or `always`, or with `force=yes`.')
 455 
 456         if not self._check_format():
 457             # During conversion step, convert if format does not match and format_mismatch == 'convert'
 458             if not ignore_conversion and self.format_mismatch == 'convert':
 459                 return False
 460             # During generation step, regenerate if format does not match and format_mismatch == 'regenerate'
 461             if ignore_conversion and self.format_mismatch == 'regenerate' and self.regenerate != 'never':
 462                 if not ignore_conversion or self.regenerate in ('partial_idempotence', 'full_idempotence', 'always'):
 463                     return False
 464                 module.fail_json(msg='Key has wrong format.'
 465                                      ' Will not proceed. To force regeneration, call the module with `generate`'
 466                                      ' set to `partial_idempotence`, `full_idempotence` or `always`, or with `force=yes`.'
 467                                      ' To convert the key, set `format_mismatch` to `convert`.')
 468 
 469         # check whether permissions are correct (in case that needs to be checked)
 470         return not perms_required or super(PrivateKeyBase, self).check(module, perms_required=perms_required)
 471 
 472     def dump(self):
 473         """Serialize the object into a dictionary."""
 474 
 475         result = {
 476             'size': self.size,
 477             'filename': self.path,
 478             'changed': self.changed,
 479             'fingerprint': self.fingerprint,
 480         }
 481         if self.backup_file:
 482             result['backup_file'] = self.backup_file
 483         if self.return_content:
 484             if self.privatekey_bytes is None:
 485                 self.privatekey_bytes = crypto_utils.load_file_if_exists(self.path, ignore_errors=True)
 486             if self.privatekey_bytes:
 487                 if crypto_utils.identify_private_key_format(self.privatekey_bytes) == 'raw':
 488                     result['privatekey'] = base64.b64encode(self.privatekey_bytes)
 489                 else:
 490                     result['privatekey'] = self.privatekey_bytes.decode('utf-8')
 491             else:
 492                 result['privatekey'] = None
 493 
 494         return result

 497 # Implementation with using pyOpenSSL
 498 class PrivateKeyPyOpenSSL(PrivateKeyBase):
 499 
 500     def __init__(self, module):
 501         super(PrivateKeyPyOpenSSL, self).__init__(module)
 502 
 503         if module.params['type'] == 'RSA':
 504             self.type = crypto.TYPE_RSA
 505         elif module.params['type'] == 'DSA':
 506             self.type = crypto.TYPE_DSA
 507         else:
 508             module.fail_json(msg="PyOpenSSL backend only supports RSA and DSA keys.")
 509 
 510         if self.format != 'auto_ignore':
 511             module.fail_json(msg="PyOpenSSL backend only supports auto_ignore format.")
 512 
 513     def _generate_private_key(self):
 514         """(Re-)Generate private key."""
 515         self.privatekey = crypto.PKey()
 516         try:
 517             self.privatekey.generate_key(self.type, self.size)
 518         except (TypeError, ValueError) as exc:
 519             raise PrivateKeyError(exc)
 520 
 521     def _ensure_private_key_loaded(self):
 522         """Make sure that the private key has been loaded."""
 523         if self.privatekey is None:
 524             try:
 525                 self.privatekey = privatekey = crypto_utils.load_privatekey(self.path, self.passphrase)
 526             except crypto_utils.OpenSSLBadPassphraseError as exc:
 527                 raise PrivateKeyError(exc)
 528 
 529     def _get_private_key_data(self):
 530         """Return bytes for self.privatekey"""
 531         if self.cipher and self.passphrase:
 532             return crypto.dump_privatekey(crypto.FILETYPE_PEM, self.privatekey,
 533                                           self.cipher, to_bytes(self.passphrase))
 534         else:
 535             return crypto.dump_privatekey(crypto.FILETYPE_PEM, self.privatekey)
 536 
 537     def _get_fingerprint(self):
 538         return crypto_utils.get_fingerprint(self.path, self.passphrase)
 539 
 540     def _check_passphrase(self):
 541         try:

 577     def _get_ec_class(self, ectype):
 578         ecclass = cryptography.hazmat.primitives.asymmetric.ec.__dict__.get(ectype)
 579         if ecclass is None:
 580             self.module.fail_json(msg='Your cryptography version does not support {0}'.format(ectype))
 581         return ecclass

 583     def _add_curve(self, name, ectype, deprecated=False):
 584         def create(size):
 585             ecclass = self._get_ec_class(ectype)
 586             return ecclass()
 587 
 588         def verify(privatekey):
 589             ecclass = self._get_ec_class(ectype)
 590             return isinstance(privatekey.private_numbers().public_numbers.curve, ecclass)
 591 
 592         self.curves[name] = {
 593             'create': create,
 594             'verify': verify,
 595             'deprecated': deprecated,
 596         }

 638     def _get_wanted_format(self):
 639         if self.format not in ('auto', 'auto_ignore'):
 640             return self.format
 641         if self.type in ('X25519', 'X448', 'Ed25519', 'Ed448'):
 642             return 'pkcs8'
 643         else:
 644             return 'pkcs1'

 646     def _generate_private_key(self):
 647         """(Re-)Generate private key."""
 648         try:
 649             if self.type == 'RSA':
 650                 self.privatekey = cryptography.hazmat.primitives.asymmetric.rsa.generate_private_key(
 651                     public_exponent=65537,  # OpenSSL always uses this
 652                     key_size=self.size,
 653                     backend=self.cryptography_backend
 654                 )
 655             if self.type == 'DSA':
 656                 self.privatekey = cryptography.hazmat.primitives.asymmetric.dsa.generate_private_key(
 657                     key_size=self.size,
 658                     backend=self.cryptography_backend
 659                 )
 660             if CRYPTOGRAPHY_HAS_X25519_FULL and self.type == 'X25519':
 661                 self.privatekey = cryptography.hazmat.primitives.asymmetric.x25519.X25519PrivateKey.generate()
 662             if CRYPTOGRAPHY_HAS_X448 and self.type == 'X448':
 663                 self.privatekey = cryptography.hazmat.primitives.asymmetric.x448.X448PrivateKey.generate()
 664             if CRYPTOGRAPHY_HAS_ED25519 and self.type == 'Ed25519':
 665                 self.privatekey = cryptography.hazmat.primitives.asymmetric.ed25519.Ed25519PrivateKey.generate()
 666             if CRYPTOGRAPHY_HAS_ED448 and self.type == 'Ed448':
 667                 self.privatekey = cryptography.hazmat.primitives.asymmetric.ed448.Ed448PrivateKey.generate()
 668             if self.type == 'ECC' and self.curve in self.curves:
 669                 if self.curves[self.curve]['deprecated']:
 670                     self.module.warn('Elliptic curves of type {0} should not be used for new keys!'.format(self.curve))
 671                 self.privatekey = cryptography.hazmat.primitives.asymmetric.ec.generate_private_key(
 672                     curve=self.curves[self.curve]['create'](self.size),
 673                     backend=self.cryptography_backend
 674                 )
 675         except cryptography.exceptions.UnsupportedAlgorithm as dummy:
 676             self.module.fail_json(msg='Cryptography backend does not support the algorithm required for {0}'.format(self.type))
 677 
 678     def _ensure_private_key_loaded(self):
 679         """Make sure that the private key has been loaded."""
 680         if self.privatekey is None:
 681             self.privatekey = self._load_privatekey()

 683     def _get_private_key_data(self):
 684         """Return bytes for self.privatekey"""
 685         # Select export format and encoding
 686         try:
 687             export_format = self._get_wanted_format()
 688             export_encoding = cryptography.hazmat.primitives.serialization.Encoding.PEM
 689             if export_format == 'pkcs1':
 690                 # "TraditionalOpenSSL" format is PKCS1
 691                 export_format = cryptography.hazmat.primitives.serialization.PrivateFormat.TraditionalOpenSSL
 692             elif export_format == 'pkcs8':
 693                 export_format = cryptography.hazmat.primitives.serialization.PrivateFormat.PKCS8
 694             elif export_format == 'raw':
 695                 export_format = cryptography.hazmat.primitives.serialization.PrivateFormat.Raw
 696                 export_encoding = cryptography.hazmat.primitives.serialization.Encoding.Raw
 697         except AttributeError:
 698             self.module.fail_json(msg='Cryptography backend does not support the selected output format "{0}"'.format(self.format))
 699 
 700         # Select key encryption
 701         encryption_algorithm = cryptography.hazmat.primitives.serialization.NoEncryption()
 702         if self.cipher and self.passphrase:
 703             if self.cipher == 'auto':
 704                 encryption_algorithm = cryptography.hazmat.primitives.serialization.BestAvailableEncryption(to_bytes(self.passphrase))
 705             else:
 706                 self.module.fail_json(msg='Cryptography backend can only use "auto" for cipher option.')
 707 
 708         # Serialize key
 709         try:
 710             return self.privatekey.private_bytes(
 711                 encoding=export_encoding,
 712                 format=export_format,
 713                 encryption_algorithm=encryption_algorithm
 714             )
 715         except ValueError as dummy:
 716             self.module.fail_json(
 717                 msg='Cryptography backend cannot serialize the private key in the required format "{0}"'.format(self.format)
 718             )
 719         except Exception as dummy:
 720             self.module.fail_json(
 721                 msg='Error while serializing the private key in the required format "{0}"'.format(self.format),
 722                 exception=traceback.format_exc()
 723             )

 725     def _load_privatekey(self):
 726         try:
 727             # Read bytes
 728             with open(self.path, 'rb') as f:
 729                 data = f.read()
 730             # Interpret bytes depending on format.
 731             format = crypto_utils.identify_private_key_format(data)
 732             if format == 'raw':
 733                 if len(data) == 56 and CRYPTOGRAPHY_HAS_X448:
 734                     return cryptography.hazmat.primitives.asymmetric.x448.X448PrivateKey.from_private_bytes(data)
 735                 if len(data) == 57 and CRYPTOGRAPHY_HAS_ED448:
 736                     return cryptography.hazmat.primitives.asymmetric.ed448.Ed448PrivateKey.from_private_bytes(data)
 737                 if len(data) == 32:
 738                     if CRYPTOGRAPHY_HAS_X25519 and (self.type == 'X25519' or not CRYPTOGRAPHY_HAS_ED25519):
 739                         return cryptography.hazmat.primitives.asymmetric.x25519.X25519PrivateKey.from_private_bytes(data)
 740                     if CRYPTOGRAPHY_HAS_ED25519 and (self.type == 'Ed25519' or not CRYPTOGRAPHY_HAS_X25519):
 741                         return cryptography.hazmat.primitives.asymmetric.ed25519.Ed25519PrivateKey.from_private_bytes(data)
 742                     if CRYPTOGRAPHY_HAS_X25519 and CRYPTOGRAPHY_HAS_ED25519:
 743                         try:
 744                             return cryptography.hazmat.primitives.asymmetric.x25519.X25519PrivateKey.from_private_bytes(data)
 745                         except Exception:
 746                             return cryptography.hazmat.primitives.asymmetric.ed25519.Ed25519PrivateKey.from_private_bytes(data)
 747                 raise PrivateKeyError('Cannot load raw key')
 748             else:
 749                 return cryptography.hazmat.primitives.serialization.load_pem_private_key(
 750                     data,
 751                     None if self.passphrase is None else to_bytes(self.passphrase),
 752                     backend=self.cryptography_backend
 753                 )
 754         except Exception as e:
 755             raise PrivateKeyError(e)

 757     def _get_fingerprint(self):
 758         # Get bytes of public key
 759         private_key = self._load_privatekey()
 760         public_key = private_key.public_key()
 761         public_key_bytes = public_key.public_bytes(
 762             cryptography.hazmat.primitives.serialization.Encoding.DER,
 763             cryptography.hazmat.primitives.serialization.PublicFormat.SubjectPublicKeyInfo
 764         )
 765         # Get fingerprints of public_key_bytes
 766         return crypto_utils.get_fingerprint_of_bytes(public_key_bytes)

 768     def _check_passphrase(self):
 769         try:
 770             with open(self.path, 'rb') as f:
 771                 data = f.read()
 772             format = crypto_utils.identify_private_key_format(data)
 773             if format == 'raw':
 774                 # Raw keys cannot be encrypted. To avoid incompatibilities, we try to
 775                 # actually load the key (and return False when this fails).
 776                 self._load_privatekey()
 777                 # Loading the key succeeded. Only return True when no passphrase was
 778                 # provided.
 779                 return self.passphrase is None
 780             else:
 781                 return cryptography.hazmat.primitives.serialization.load_pem_private_key(
 782                     data,
 783                     None if self.passphrase is None else to_bytes(self.passphrase),
 784                     backend=self.cryptography_backend
 785                 )
 786         except Exception as dummy:
 787             return False

 789     def _check_size_and_type(self):
 790         self._ensure_private_key_loaded()
 791 
 792         if isinstance(self.privatekey, cryptography.hazmat.primitives.asymmetric.rsa.RSAPrivateKey):
 793             return self.type == 'RSA' and self.size == self.privatekey.key_size
 794         if isinstance(self.privatekey, cryptography.hazmat.primitives.asymmetric.dsa.DSAPrivateKey):
 795             return self.type == 'DSA' and self.size == self.privatekey.key_size
 796         if CRYPTOGRAPHY_HAS_X25519 and isinstance(self.privatekey, cryptography.hazmat.primitives.asymmetric.x25519.X25519PrivateKey):
 797             return self.type == 'X25519'
 798         if CRYPTOGRAPHY_HAS_X448 and isinstance(self.privatekey, cryptography.hazmat.primitives.asymmetric.x448.X448PrivateKey):
 799             return self.type == 'X448'
 800         if CRYPTOGRAPHY_HAS_ED25519 and isinstance(self.privatekey, cryptography.hazmat.primitives.asymmetric.ed25519.Ed25519PrivateKey):
 801             return self.type == 'Ed25519'
 802         if CRYPTOGRAPHY_HAS_ED448 and isinstance(self.privatekey, cryptography.hazmat.primitives.asymmetric.ed448.Ed448PrivateKey):
 803             return self.type == 'Ed448'
 804         if isinstance(self.privatekey, cryptography.hazmat.primitives.asymmetric.ec.EllipticCurvePrivateKey):
 805             if self.type != 'ECC':
 806                 return False
 807             if self.curve not in self.curves:
 808                 return False
 809             return self.curves[self.curve]['verify'](self.privatekey)
 810 
 811         return False

 813     def _check_format(self):
 814         if self.format == 'auto_ignore':
 815             return True
 816         try:
 817             with open(self.path, 'rb') as f:
 818                 content = f.read()
 819             format = crypto_utils.identify_private_key_format(content)
 820             return format == self._get_wanted_format()
 821         except Exception as dummy:
 822             return False
