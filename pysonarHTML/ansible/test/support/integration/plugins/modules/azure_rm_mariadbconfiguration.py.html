 132     def exec_module(self, **kwargs):
 133 
 134         for key in list(self.module_arg_spec.keys()):
 135             if hasattr(self, key):
 136                 setattr(self, key, kwargs[key])
 137 
 138         old_response = None
 139         response = None
 140 
 141         old_response = self.get_configuration()
 142 
 143         if not old_response:
 144             self.log("Configuration instance doesn't exist")
 145             if self.state == 'absent':
 146                 self.log("Old instance didn't exist")
 147             else:
 148                 self.to_do = Actions.Create
 149         else:
 150             self.log("Configuration instance already exists")
 151             if self.state == 'absent' and old_response['source'] == 'user-override':
 152                 self.to_do = Actions.Delete
 153             elif self.state == 'present':
 154                 self.log("Need to check if Configuration instance has to be deleted or may be updated")
 155                 if self.value != old_response.get('value'):
 156                     self.to_do = Actions.Update
 157 
 158         if (self.to_do == Actions.Create) or (self.to_do == Actions.Update):
 159             self.log("Need to Create / Update the Configuration instance")
 160 
 161             if self.check_mode:
 162                 self.results['changed'] = True
 163                 return self.results
 164 
 165             response = self.create_update_configuration()
 166 
 167             self.results['changed'] = True
 168             self.log("Creation / Update done")
 169         elif self.to_do == Actions.Delete:
 170             self.log("Configuration instance deleted")
 171             self.results['changed'] = True
 172 
 173             if self.check_mode:
 174                 return self.results
 175 
 176             self.delete_configuration()
 177         else:
 178             self.log("Configuration instance unchanged")
 179             self.results['changed'] = False
 180             response = old_response
 181 
 182         if response:
 183             self.results["id"] = response["id"]
 184 
 185         return self.results
 186 
 187     def create_update_configuration(self):
 188         self.log("Creating / Updating the Configuration instance {0}".format(self.name))
 189 
 190         try:
 191             response = self.mariadb_client.configurations.create_or_update(resource_group_name=self.resource_group,
 192                                                                            server_name=self.server_name,
 193                                                                            configuration_name=self.name,
 194                                                                            value=self.value,
 195                                                                            source='user-override')
 196             if isinstance(response, LROPoller):
 197                 response = self.get_poller_result(response)
 198 
 199         except CloudError as exc:
 200             self.log('Error attempting to create the Configuration instance.')
 201             self.fail("Error creating the Configuration instance: {0}".format(str(exc)))
 202         return response.as_dict()
 203 
 204     def delete_configuration(self):
 205         self.log("Deleting the Configuration instance {0}".format(self.name))
 206         try:
 207             response = self.mariadb_client.configurations.create_or_update(resource_group_name=self.resource_group,
 208                                                                            server_name=self.server_name,
 209                                                                            configuration_name=self.name,
 210                                                                            source='system-default')
 211         except CloudError as e:
 212             self.log('Error attempting to delete the Configuration instance.')
 213             self.fail("Error deleting the Configuration instance: {0}".format(str(e)))
 214 
 215         return True
 216 
 217     def get_configuration(self):
 218         self.log("Checking if the Configuration instance {0} is present".format(self.name))
 219         found = False
 220         try:
 221             response = self.mariadb_client.configurations.get(resource_group_name=self.resource_group,
 222                                                               server_name=self.server_name,
 223                                                               configuration_name=self.name)
 224             found = True
 225             self.log("Response : {0}".format(response))
 226             self.log("Configuration instance : {0} found".format(response.name))
 227         except CloudError as e:
 228             self.log('Did not find the Configuration instance.')
 229         if found is True:
 230             return response.as_dict()
 231 
 232         return False
