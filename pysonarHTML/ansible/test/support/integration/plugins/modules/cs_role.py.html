  98 from ansible.module_utils.basic import AnsibleModule
  99 from ansible.module_utils.cloudstack import (
 100     AnsibleCloudStack,
 101     cs_argument_spec,
 102     cs_required_together,
 103 )
 104 
 105 
 106 class AnsibleCloudStackRole(AnsibleCloudStack):
 107 
 108     def __init__(self, module):
 109         super(AnsibleCloudStackRole, self).__init__(module)
 110         self.returns = {
 111             'type': 'role_type',
 112         }
 113 
 114     def get_role(self):
 115         uuid = self.module.params.get('uuid')
 116         if uuid:
 117             args = {
 118                 'id': uuid,
 119             }
 120             roles = self.query_api('listRoles', **args)
 121             if roles:
 122                 return roles['role'][0]
 123         else:
 124             args = {
 125                 'name': self.module.params.get('name'),
 126             }
 127             roles = self.query_api('listRoles', **args)
 128             if roles:
 129                 return roles['role'][0]
 130         return None
 131 
 132     def present_role(self):
 133         role = self.get_role()
 134         if role:
 135             role = self._update_role(role)
 136         else:
 137             role = self._create_role(role)
 138         return role
 139 
 140     def _create_role(self, role):
 141         self.result['changed'] = True
 142         args = {
 143             'name': self.module.params.get('name'),
 144             'type': self.module.params.get('role_type'),
 145             'description': self.module.params.get('description'),
 146         }
 147         if not self.module.check_mode:
 148             res = self.query_api('createRole', **args)
 149             role = res['role']
 150         return role
 151 
 152     def _update_role(self, role):
 153         args = {
 154             'id': role['id'],
 155             'name': self.module.params.get('name'),
 156             'description': self.module.params.get('description'),
 157         }
 158         if self.has_changed(args, role):
 159             self.result['changed'] = True
 160             if not self.module.check_mode:
 161                 res = self.query_api('updateRole', **args)
 162 
 163                 # The API as in 4.9 does not return an updated role yet
 164                 if 'role' not in res:
 165                     role = self.get_role()
 166                 else:
 167                     role = res['role']
 168         return role
 169 
 170     def absent_role(self):
 171         role = self.get_role()
 172         if role:
 173             self.result['changed'] = True
 174             args = {
 175                 'id': role['id'],
 176             }
 177             if not self.module.check_mode:
 178                 self.query_api('deleteRole', **args)
 179         return role
 180 
 181 
 182 def main():
 183     argument_spec = cs_argument_spec()
 184     argument_spec.update(dict(
 185         uuid=dict(aliases=['id']),
 186         name=dict(required=True),
 187         description=dict(),
 188         role_type=dict(choices=['User', 'DomainAdmin', 'ResourceAdmin', 'Admin'], default='User'),
 189         state=dict(choices=['present', 'absent'], default='present'),
 190     ))
 191 
 192     module = AnsibleModule(
 193         argument_spec=argument_spec,
 194         required_together=cs_required_together(),
 195         supports_check_mode=True
 196     )
 197 
 198     acs_role = AnsibleCloudStackRole(module)
 199     state = module.params.get('state')
 200     if state == 'absent':
 201         role = acs_role.absent_role()
 202     else:
 203         role = acs_role.present_role()
 204 
 205     result = acs_role.get_result(role)
 206 
 207     module.exit_json(**result)
