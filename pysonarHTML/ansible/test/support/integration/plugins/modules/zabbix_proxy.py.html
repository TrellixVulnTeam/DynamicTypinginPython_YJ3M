 223 class Proxy(object):
 224     def __init__(self, module, zbx):
 225         self._module = module
 226         self._zapi = zbx
 227         self.existing_data = None
 228 
 229     def proxy_exists(self, proxy_name):
 230         result = self._zapi.proxy.get({
 231             'output': 'extend', 'selectInterface': 'extend',
 232             'filter': {'host': proxy_name}})
 233 
 234         if len(result) > 0 and 'proxyid' in result[0]:
 235             self.existing_data = result[0]
 236             return result[0]['proxyid']
 237         else:
 238             return result
 239 
 240     def add_proxy(self, data):
 241         try:
 242             if self._module.check_mode:
 243                 self._module.exit_json(changed=True)
 244 
 245             parameters = {}
 246             for item in data:
 247                 if data[item]:
 248                     parameters[item] = data[item]
 249 
 250             if 'proxy_address' in data and data['status'] != '5':
 251                 parameters.pop('proxy_address', False)
 252 
 253             if 'interface' in data and data['status'] != '6':
 254                 parameters.pop('interface', False)
 255 
 256             proxy_ids_list = self._zapi.proxy.create(parameters)
 257             self._module.exit_json(changed=True,
 258                                    result="Successfully added proxy %s (%s)" %
 259                                           (data['host'], data['status']))
 260             if len(proxy_ids_list) >= 1:
 261                 return proxy_ids_list['proxyids'][0]
 262         except Exception as e:
 263             self._module.fail_json(msg="Failed to create proxy %s: %s" %
 264                                    (data['host'], e))
 265 
 266     def delete_proxy(self, proxy_id, proxy_name):
 267         try:
 268             if self._module.check_mode:
 269                 self._module.exit_json(changed=True)
 270             self._zapi.proxy.delete([proxy_id])
 271             self._module.exit_json(changed=True,
 272                                    result="Successfully deleted"
 273                                           + " proxy %s" % proxy_name)
 274         except Exception as e:
 275             self._module.fail_json(msg="Failed to delete proxy %s: %s" %
 276                                        (proxy_name, str(e)))
 277 
 278     def compile_interface_params(self, new_interface):
 279         old_interface = {}
 280         if 'interface' in self.existing_data and \
 281            len(self.existing_data['interface']) > 0:
 282             old_interface = self.existing_data['interface']
 283 
 284         for item in ['type', 'main']:
 285             new_interface.pop(item, False)
 286 
 287         final_interface = old_interface.copy()
 288         final_interface.update(new_interface)
 289         final_interface = dict((k, str(v)) for k, v in final_interface.items())
 290 
 291         if final_interface != old_interface:
 292             return final_interface
 293         else:
 294             return {}
 295 
 296     def update_proxy(self, proxy_id, data):
 297         try:
 298             if self._module.check_mode:
 299                 self._module.exit_json(changed=True)
 300             parameters = {'proxyid': proxy_id}
 301 
 302             for item in data:
 303                 if data[item] and item in self.existing_data and \
 304                    self.existing_data[item] != data[item]:
 305                     parameters[item] = data[item]
 306 
 307             if 'interface' in parameters:
 308                 parameters.pop('interface')
 309 
 310             if 'proxy_address' in data and data['status'] != '5':
 311                 parameters.pop('proxy_address', False)
 312 
 313             if 'interface' in data and data['status'] != '6':
 314                 parameters.pop('interface', False)
 315 
 316             if 'interface' in data and data['status'] == '6':
 317                 new_interface = self.compile_interface_params(data['interface'])
 318                 if len(new_interface) > 0:
 319                     parameters['interface'] = new_interface
 320 
 321             if len(parameters) > 1:
 322                 self._zapi.proxy.update(parameters)
 323                 self._module.exit_json(
 324                     changed=True,
 325                     result="Successfully updated proxy %s (%s)" %
 326                            (data['host'], proxy_id)
 327                 )
 328             else:
 329                 self._module.exit_json(changed=False)
 330         except Exception as e:
 331             self._module.fail_json(msg="Failed to update proxy %s: %s" %
 332                                        (data['host'], e))

 380     login_user = module.params['login_user']
 381     login_password = module.params['login_password']
 382     http_login_user = module.params['http_login_user']
 383     http_login_password = module.params['http_login_password']
 384     validate_certs = module.params['validate_certs']
 385     proxy_name = module.params['proxy_name']
 386     proxy_address = module.params['proxy_address']

 395     state = module.params['state']
 396     timeout = module.params['timeout']
 397     interface = module.params['interface']
 398 
 399     # convert enabled to 0; disabled to 1
 400     status = 6 if status == "passive" else 5
 401 
 402     if tls_connect == 'certificate':
 403         tls_connect = 4
 404     elif tls_connect == 'PSK':
 405         tls_connect = 2
 406     else:
 407         tls_connect = 1
 408 
 409     if tls_accept == 'certificate':
 410         tls_accept = 4
 411     elif tls_accept == 'PSK':
 412         tls_accept = 2
 413     else:
 414         tls_accept = 1
 415 
 416     zbx = None
 417     # login to zabbix
 418     try:
 419         zbx = ZabbixAPI(server_url, timeout=timeout,
 420                         user=http_login_user,
 421                         passwd=http_login_password,
 422                         validate_certs=validate_certs)
 423         zbx.login(login_user, login_password)
 424         atexit.register(zbx.logout)
 425     except Exception as e:
 426         module.fail_json(msg="Failed to connect to Zabbix server: %s" % e)
 427 
 428     proxy = Proxy(module, zbx)
 429 
 430     # check if proxy already exists
 431     proxy_id = proxy.proxy_exists(proxy_name)
 432 
 433     if proxy_id:
 434         if state == "absent":
 435             # remove proxy
 436             proxy.delete_proxy(proxy_id, proxy_name)
 437         else:
 438             proxy.update_proxy(proxy_id, {
 439                 'host': proxy_name,
 440                 'description': description,
 441                 'status': str(status),
 442                 'tls_connect': str(tls_connect),
 443                 'tls_accept': str(tls_accept),
 444                 'tls_issuer': tls_issuer,
 445                 'tls_subject': tls_subject,
 446                 'tls_psk_identity': tls_psk_identity,
 447                 'tls_psk': tls_psk,
 448                 'interface': interface,
 449                 'proxy_address': proxy_address
 450             })
 451     else:
 452         if state == "absent":
 453             # the proxy is already deleted.
 454             module.exit_json(changed=False)
 455 
 456         proxy_id = proxy.add_proxy(data={
 457             'host': proxy_name,
 458             'description': description,
 459             'status': str(status),
 460             'tls_connect': str(tls_connect),
 461             'tls_accept': str(tls_accept),
 462             'tls_issuer': tls_issuer,
 463             'tls_subject': tls_subject,
 464             'tls_psk_identity': tls_psk_identity,
 465             'tls_psk': tls_psk,
 466             'interface': interface,
 467             'proxy_address': proxy_address
 468         })
