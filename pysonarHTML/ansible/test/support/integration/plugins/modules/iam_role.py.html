 230 def convert_friendly_names_to_arns(connection, module, policy_names):
 231     if not any([not policy.startswith('arn:') for policy in policy_names]):
 232         return policy_names
 233     allpolicies = {}
 234     policies = _list_policies(connection)
 235 
 236     for policy in policies:
 237         allpolicies[policy['PolicyName']] = policy['Arn']
 238         allpolicies[policy['Arn']] = policy['Arn']
 239     try:
 240         return [allpolicies[policy] for policy in policy_names]
 241     except KeyError as e:
 242         module.fail_json_aws(e, msg="Couldn't find policy")

 245 def attach_policies(connection, module, policies_to_attach, params):
 246     changed = False
 247     for policy_arn in policies_to_attach:
 248         try:
 249             if not module.check_mode:
 250                 connection.attach_role_policy(RoleName=params['RoleName'], PolicyArn=policy_arn, aws_retry=True)
 251         except (BotoCoreError, ClientError) as e:
 252             module.fail_json_aws(e, msg="Unable to attach policy {0} to role {1}".format(policy_arn, params['RoleName']))
 253         changed = True
 254     return changed

 286 def create_basic_role(connection, module, params):
 287     """
 288     Perform the Role creation.
 289     Assumes tests for the role existing have already been performed.
 290     """
 291 
 292     try:
 293         if not module.check_mode:
 294             role = connection.create_role(aws_retry=True, **params)
 295             # 'Description' is documented as key of the role returned by create_role
 296             # but appears to be an AWS bug (the value is not returned using the AWS CLI either).
 297             # Get the role after creating it.
 298             role = get_role_with_backoff(connection, module, params['RoleName'])
 299         else:
 300             role = {'MadeInCheckMode': True}
 301             role['AssumeRolePolicyDocument'] = json.loads(params['AssumeRolePolicyDocument'])
 302     except (BotoCoreError, ClientError) as e:
 303         module.fail_json_aws(e, msg="Unable to create role")
 304 
 305     return role

 360 def update_role_permissions_boundary(connection, module, params, role):
 361     # Check PermissionsBoundary
 362     if params.get('PermissionsBoundary') is None:
 363         return False
 364     if params.get('PermissionsBoundary') == role.get('PermissionsBoundary', {}).get('PermissionsBoundaryArn', ''):
 365         return False
 366 
 367     if module.check_mode:
 368         return True
 369 
 370     if params.get('PermissionsBoundary') == '':
 371         try:
 372             connection.delete_role_permissions_boundary(RoleName=params['RoleName'], aws_retry=True)
 373         except (BotoCoreError, ClientError) as e:
 374             module.fail_json_aws(e, msg="Unable to remove permission boundary for role {0}".format(params['RoleName']))
 375     else:
 376         try:
 377             connection.put_role_permissions_boundary(RoleName=params['RoleName'], PermissionsBoundary=params['PermissionsBoundary'], aws_retry=True)
 378         except (BotoCoreError, ClientError) as e:
 379             module.fail_json_aws(e, msg="Unable to update permission boundary for role {0}".format(params['RoleName']))
 380     return True

 383 def update_managed_policies(connection, module, params, role, managed_policies, purge_policies):
 384     # Check Managed Policies
 385     if managed_policies is None:
 386         return False
 387 
 388     # If we're manipulating a fake role
 389     if role.get('MadeInCheckMode', False):
 390         role['AttachedPolicies'] = list(map(lambda x: {'PolicyArn': x, 'PolicyName': x.split(':')[5]}, managed_policies))
 391         return True
 392 
 393     # Get list of current attached managed policies
 394     current_attached_policies = get_attached_policy_list(connection, module, params['RoleName'])
 395     current_attached_policies_arn_list = [policy['PolicyArn'] for policy in current_attached_policies]
 396 
 397     if len(managed_policies) == 1 and managed_policies[0] is None:
 398         managed_policies = []
 399 
 400     policies_to_remove = set(current_attached_policies_arn_list) - set(managed_policies)
 401     policies_to_attach = set(managed_policies) - set(current_attached_policies_arn_list)
 402 
 403     changed = False
 404 
 405     if purge_policies:
 406         changed |= remove_policies(connection, module, policies_to_remove, params)
 407 
 408     changed |= attach_policies(connection, module, policies_to_attach, params)
 409 
 410     return changed

 413 def create_or_update_role(connection, module):
 414 
 415     params = generate_create_params(module)
 416     role_name = params['RoleName']
 417     create_instance_profile = module.params.get('create_instance_profile')
 418     purge_policies = module.params.get('purge_policies')
 419     if purge_policies is None:
 420         purge_policies = True
 421     managed_policies = module.params.get('managed_policies')
 422     if managed_policies:
 423         # Attempt to list the policies early so we don't leave things behind if we can't find them.
 424         managed_policies = convert_friendly_names_to_arns(connection, module, managed_policies)
 425 
 426     changed = False
 427 
 428     # Get role
 429     role = get_role(connection, module, role_name)
 430 
 431     # If role is None, create it
 432     if role is None:
 433         role = create_basic_role(connection, module, params)
 434         changed = True
 435     else:
 436         changed |= update_role_tags(connection, module, params, role)
 437         changed |= update_role_assumed_policy(connection, module, params, role)
 438         changed |= update_role_description(connection, module, params, role)
 439         changed |= update_role_max_session_duration(connection, module, params, role)
 440         changed |= update_role_permissions_boundary(connection, module, params, role)
 441 
 442     if create_instance_profile:
 443         changed |= create_instance_profiles(connection, module, params, role)
 444 
 445     changed |= update_managed_policies(connection, module, params, role, managed_policies, purge_policies)
 446 
 447     # Get the role again
 448     if not role.get('MadeInCheckMode', False):
 449         role = get_role(connection, module, params['RoleName'])
 450         role['AttachedPolicies'] = get_attached_policy_list(connection, module, params['RoleName'])
 451         role['tags'] = get_role_tags(connection, module)
 452 
 453     module.exit_json(
 454         changed=changed, iam_role=camel_dict_to_snake_dict(role, ignore_list=['tags']),
 455         **camel_dict_to_snake_dict(role, ignore_list=['tags']))

 458 def create_instance_profiles(connection, module, params, role):
 459 
 460     if role.get('MadeInCheckMode', False):
 461         return False
 462 
 463     # Fetch existing Profiles
 464     try:
 465         instance_profiles = connection.list_instance_profiles_for_role(RoleName=params['RoleName'], aws_retry=True)['InstanceProfiles']
 466     except (BotoCoreError, ClientError) as e:
 467         module.fail_json_aws(e, msg="Unable to list instance profiles for role {0}".format(params['RoleName']))
 468 
 469     # Profile already exists
 470     if any(p['InstanceProfileName'] == params['RoleName'] for p in instance_profiles):
 471         return False
 472 
 473     if module.check_mode:
 474         return True
 475 
 476     # Make sure an instance profile is created
 477     try:
 478         connection.create_instance_profile(InstanceProfileName=params['RoleName'], Path=params['Path'], aws_retry=True)
 479     except ClientError as e:
 480         # If the profile already exists, no problem, move on.
 481         # Implies someone's changing things at the same time...
 482         if e.response['Error']['Code'] == 'EntityAlreadyExists':
 483             return False
 484         else:
 485             module.fail_json_aws(e, msg="Unable to create instance profile for role {0}".format(params['RoleName']))
 486     except BotoCoreError as e:
 487         module.fail_json_aws(e, msg="Unable to create instance profile for role {0}".format(params['RoleName']))
 488 
 489     # And attach the role to the profile
 490     try:
 491         connection.add_role_to_instance_profile(InstanceProfileName=params['RoleName'], RoleName=params['RoleName'], aws_retry=True)
 492     except (BotoCoreError, ClientError) as e:
 493         module.fail_json_aws(e, msg="Unable to attach role {0} to instance profile {0}".format(params['RoleName']))
 494 
 495     return True

 498 def remove_instance_profiles(connection, module, role_params, role):
 499     role_name = module.params.get('name')
 500     delete_profiles = module.params.get("delete_instance_profile")
 501 
 502     try:
 503         instance_profiles = connection.list_instance_profiles_for_role(aws_retry=True, **role_params)['InstanceProfiles']
 504     except (BotoCoreError, ClientError) as e:
 505         module.fail_json_aws(e, msg="Unable to list instance profiles for role {0}".format(role_name))
 506 
 507     # Remove the role from the instance profile(s)
 508     for profile in instance_profiles:
 509         profile_name = profile['InstanceProfileName']
 510         try:
 511             if not module.check_mode:
 512                 connection.remove_role_from_instance_profile(aws_retry=True, InstanceProfileName=profile_name, **role_params)
 513                 if profile_name == role_name:
 514                     if delete_profiles:
 515                         try:
 516                             connection.delete_instance_profile(InstanceProfileName=profile_name, aws_retry=True)
 517                         except (BotoCoreError, ClientError) as e:
 518                             module.fail_json_aws(e, msg="Unable to remove instance profile {0}".format(profile_name))
 519         except (BotoCoreError, ClientError) as e:
 520             module.fail_json_aws(e, msg="Unable to remove role {0} from instance profile {1}".format(role_name, profile_name))

 523 def destroy_role(connection, module):
 524 
 525     role_name = module.params.get('name')
 526     role = get_role(connection, module, role_name)
 527     role_params = dict()
 528     role_params['RoleName'] = role_name
 529     boundary_params = dict(role_params)
 530     boundary_params['PermissionsBoundary'] = ''
 531 
 532     if role is None:
 533         module.exit_json(changed=False)
 534 
 535     # Before we try to delete the role we need to remove any
 536     # - attached instance profiles
 537     # - attached managed policies
 538     # - permissions boundary
 539     remove_instance_profiles(connection, module, role_params, role)
 540     update_managed_policies(connection, module, role_params, role, [], True)
 541     update_role_permissions_boundary(connection, module, boundary_params, role)
 542 
 543     try:
 544         if not module.check_mode:
 545             connection.delete_role(aws_retry=True, **role_params)
 546     except (BotoCoreError, ClientError) as e:
 547         module.fail_json_aws(e, msg="Unable to delete role")
 548 
 549     module.exit_json(changed=True)

 595     purge_tags = module.params.get('purge_tags')
 596 
 597     try:
 598         existing_tags = boto3_tag_list_to_ansible_dict(connection.list_role_tags(RoleName=role_name, aws_retry=True)['Tags'])
 599     except (ClientError, KeyError):
 600         existing_tags = {}
 601 
 602     tags_to_add, tags_to_remove = compare_aws_tags(existing_tags, new_tags, purge_tags=purge_tags)
 603 
 604     if not module.check_mode:
 605         try:
 606             if tags_to_remove:
 607                 connection.untag_role(RoleName=role_name, TagKeys=tags_to_remove, aws_retry=True)
 608             if tags_to_add:
 609                 connection.tag_role(RoleName=role_name, Tags=ansible_dict_to_boto3_tag_list(tags_to_add), aws_retry=True)
 610         except (ClientError, BotoCoreError) as e:
 611             module.fail_json_aws(e, msg='Unable to set tags for role %s' % role_name)
 612 
 613     changed = bool(tags_to_add) or bool(tags_to_remove)
 614     return changed
