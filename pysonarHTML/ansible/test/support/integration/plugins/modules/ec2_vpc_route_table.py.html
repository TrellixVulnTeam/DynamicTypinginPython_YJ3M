 259 def find_subnets(connection, module, vpc_id, identified_subnets):
 260     """
 261     Finds a list of subnets, each identified either by a raw ID, a unique
 262     'Name' tag, or a CIDR such as 10.0.0.0/8.
 263 
 264     Note that this function is duplicated in other ec2 modules, and should
 265     potentially be moved into a shared module_utils
 266     """
 267     subnet_ids = []
 268     subnet_names = []
 269     subnet_cidrs = []
 270     for subnet in (identified_subnets or []):
 271         if re.match(SUBNET_RE, subnet):
 272             subnet_ids.append(subnet)
 273         elif re.match(CIDR_RE, subnet):
 274             subnet_cidrs.append(subnet)
 275         else:
 276             subnet_names.append(subnet)
 277 
 278     subnets_by_id = []
 279     if subnet_ids:
 280         filters = ansible_dict_to_boto3_filter_list({'vpc-id': vpc_id})
 281         try:
 282             subnets_by_id = describe_subnets_with_backoff(connection, SubnetIds=subnet_ids, Filters=filters)
 283         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 284             module.fail_json_aws(e, msg="Couldn't find subnet with id %s" % subnet_ids)
 285 
 286     subnets_by_cidr = []
 287     if subnet_cidrs:
 288         filters = ansible_dict_to_boto3_filter_list({'vpc-id': vpc_id, 'cidr': subnet_cidrs})
 289         try:
 290             subnets_by_cidr = describe_subnets_with_backoff(connection, Filters=filters)
 291         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 292             module.fail_json_aws(e, msg="Couldn't find subnet with cidr %s" % subnet_cidrs)
 293 
 294     subnets_by_name = []
 295     if subnet_names:
 296         filters = ansible_dict_to_boto3_filter_list({'vpc-id': vpc_id, 'tag:Name': subnet_names})
 297         try:
 298             subnets_by_name = describe_subnets_with_backoff(connection, Filters=filters)
 299         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 300             module.fail_json_aws(e, msg="Couldn't find subnet with names %s" % subnet_names)
 301 
 302         for name in subnet_names:
 303             matching_count = len([1 for s in subnets_by_name for t in s.get('Tags', []) if t['Key'] == 'Name' and t['Value'] == name])
 304             if matching_count == 0:
 305                 module.fail_json(msg='Subnet named "{0}" does not exist'.format(name))
 306             elif matching_count > 1:
 307                 module.fail_json(msg='Multiple subnets named "{0}"'.format(name))
 308 
 309     return subnets_by_id + subnets_by_cidr + subnets_by_name

 312 def find_igw(connection, module, vpc_id):
 313     """
 314     Finds the Internet gateway for the given VPC ID.
 315     """
 316     filters = ansible_dict_to_boto3_filter_list({'attachment.vpc-id': vpc_id})
 317     try:
 318         igw = connection.describe_internet_gateways(Filters=filters)['InternetGateways']
 319     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 320         module.fail_json_aws(e, msg='No IGW found for VPC {0}'.format(vpc_id))
 321     if len(igw) == 1:
 322         return igw[0]['InternetGatewayId']
 323     elif len(igw) == 0:
 324         module.fail_json(msg='No IGWs found for VPC {0}'.format(vpc_id))
 325     else:
 326         module.fail_json(msg='Multiple IGWs found for VPC {0}'.format(vpc_id))

 337 def tags_match(match_tags, candidate_tags):
 338     return all((k in candidate_tags and candidate_tags[k] == v
 339                 for k, v in match_tags.items()))

 342 def ensure_tags(connection=None, module=None, resource_id=None, tags=None, purge_tags=None, check_mode=None):
 343     try:
 344         cur_tags = describe_tags_with_backoff(connection, resource_id)
 345     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 346         module.fail_json_aws(e, msg='Unable to list tags for VPC')
 347 
 348     to_add, to_delete = compare_aws_tags(cur_tags, tags, purge_tags)
 349 
 350     if not to_add and not to_delete:
 351         return {'changed': False, 'tags': cur_tags}
 352     if check_mode:
 353         if not purge_tags:
 354             tags = cur_tags.update(tags)
 355         return {'changed': True, 'tags': tags}
 356 
 357     if to_delete:
 358         try:
 359             connection.delete_tags(Resources=[resource_id], Tags=[{'Key': k} for k in to_delete])
 360         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 361             module.fail_json_aws(e, msg="Couldn't delete tags")
 362     if to_add:
 363         try:
 364             connection.create_tags(Resources=[resource_id], Tags=ansible_dict_to_boto3_tag_list(to_add))
 365         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 366             module.fail_json_aws(e, msg="Couldn't create tags")
 367 
 368     try:
 369         latest_tags = describe_tags_with_backoff(connection, resource_id)
 370     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 371         module.fail_json_aws(e, msg='Unable to list tags for VPC')
 372     return {'changed': True, 'tags': latest_tags}

 399 def get_route_table_by_tags(connection, module, vpc_id, tags):
 400     count = 0
 401     route_table = None
 402     filters = ansible_dict_to_boto3_filter_list({'vpc-id': vpc_id})
 403     try:
 404         route_tables = describe_route_tables_with_backoff(connection, Filters=filters)
 405     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 406         module.fail_json_aws(e, msg="Couldn't get route table")
 407     for table in route_tables:
 408         this_tags = describe_tags_with_backoff(connection, table['RouteTableId'])
 409         if tags_match(tags, this_tags):
 410             route_table = table
 411             count += 1
 412 
 413     if count > 1:
 414         module.fail_json(msg="Tags provided do not identify a unique route table")
 415     else:
 416         return route_table

 419 def route_spec_matches_route(route_spec, route):
 420     if route_spec.get('GatewayId') and 'nat-' in route_spec['GatewayId']:
 421         route_spec['NatGatewayId'] = route_spec.pop('GatewayId')
 422     if route_spec.get('GatewayId') and 'vpce-' in route_spec['GatewayId']:
 423         if route_spec.get('DestinationCidrBlock', '').startswith('pl-'):
 424             route_spec['DestinationPrefixListId'] = route_spec.pop('DestinationCidrBlock')
 425 
 426     return set(route_spec.items()).issubset(route.items())

 437 def index_of_matching_route(route_spec, routes_to_match):
 438     for i, route in enumerate(routes_to_match):
 439         if route_spec_matches_route(route_spec, route):
 440             return "exact", i
 441         elif 'Origin' in route_spec and route_spec['Origin'] != 'EnableVgwRoutePropagation':
 442             if route_spec_matches_route_cidr(route_spec, route):
 443                 return "replace", i

 446 def ensure_routes(connection=None, module=None, route_table=None, route_specs=None,
 447                   propagating_vgw_ids=None, check_mode=None, purge_routes=None):
 448     routes_to_match = [route for route in route_table['Routes']]
 449     route_specs_to_create = []
 450     route_specs_to_recreate = []
 451     for route_spec in route_specs:
 452         match = index_of_matching_route(route_spec, routes_to_match)
 453         if match is None:
 454             if route_spec.get('DestinationCidrBlock'):
 455                 route_specs_to_create.append(route_spec)
 456             else:
 457                 module.warn("Skipping creating {0} because it has no destination cidr block. "
 458                             "To add VPC endpoints to route tables use the ec2_vpc_endpoint module.".format(route_spec))
 459         else:
 460             if match[0] == "replace":
 461                 if route_spec.get('DestinationCidrBlock'):
 462                     route_specs_to_recreate.append(route_spec)
 463                 else:
 464                     module.warn("Skipping recreating route {0} because it has no destination cidr block.".format(route_spec))
 465             del routes_to_match[match[1]]
 466 
 467     routes_to_delete = []
 468     if purge_routes:
 469         for r in routes_to_match:
 470             if not r.get('DestinationCidrBlock'):
 471                 module.warn("Skipping purging route {0} because it has no destination cidr block. "
 472                             "To remove VPC endpoints from route tables use the ec2_vpc_endpoint module.".format(r))
 473                 continue
 474             if r['Origin'] == 'CreateRoute':
 475                 routes_to_delete.append(r)
 476 
 477     changed = bool(routes_to_delete or route_specs_to_create or route_specs_to_recreate)
 478     if changed and not check_mode:
 479         for route in routes_to_delete:
 480             try:
 481                 connection.delete_route(RouteTableId=route_table['RouteTableId'], DestinationCidrBlock=route['DestinationCidrBlock'])
 482             except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 483                 module.fail_json_aws(e, msg="Couldn't delete route")
 484 
 485         for route_spec in route_specs_to_recreate:
 486             try:
 487                 connection.replace_route(RouteTableId=route_table['RouteTableId'],
 488                                          **route_spec)
 489             except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 490                 module.fail_json_aws(e, msg="Couldn't recreate route")
 491 
 492         for route_spec in route_specs_to_create:
 493             try:
 494                 connection.create_route(RouteTableId=route_table['RouteTableId'],
 495                                         **route_spec)
 496             except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 497                 module.fail_json_aws(e, msg="Couldn't create route")
 498 
 499     return {'changed': bool(changed)}

 502 def ensure_subnet_association(connection=None, module=None, vpc_id=None, route_table_id=None, subnet_id=None,
 503                               check_mode=None):
 504     filters = ansible_dict_to_boto3_filter_list({'association.subnet-id': subnet_id, 'vpc-id': vpc_id})
 505     try:
 506         route_tables = describe_route_tables_with_backoff(connection, Filters=filters)
 507     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 508         module.fail_json_aws(e, msg="Couldn't get route tables")
 509     for route_table in route_tables:
 510         if route_table['RouteTableId'] is None:
 511             continue
 512         for a in route_table['Associations']:
 513             if a['Main']:
 514                 continue
 515             if a['SubnetId'] == subnet_id:
 516                 if route_table['RouteTableId'] == route_table_id:
 517                     return {'changed': False, 'association_id': a['RouteTableAssociationId']}
 518                 else:
 519                     if check_mode:
 520                         return {'changed': True}
 521                     try:
 522                         connection.disassociate_route_table(AssociationId=a['RouteTableAssociationId'])
 523                     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 524                         module.fail_json_aws(e, msg="Couldn't disassociate subnet from route table")
 525 
 526     try:
 527         association_id = connection.associate_route_table(RouteTableId=route_table_id, SubnetId=subnet_id)
 528     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 529         module.fail_json_aws(e, msg="Couldn't associate subnet with route table")
 530     return {'changed': True, 'association_id': association_id}

 533 def ensure_subnet_associations(connection=None, module=None, route_table=None, subnets=None,
 534                                check_mode=None, purge_subnets=None):
 535     current_association_ids = [a['RouteTableAssociationId'] for a in route_table['Associations'] if not a['Main']]
 536     new_association_ids = []
 537     changed = False
 538     for subnet in subnets:
 539         result = ensure_subnet_association(connection=connection, module=module, vpc_id=route_table['VpcId'],
 540                                            route_table_id=route_table['RouteTableId'], subnet_id=subnet['SubnetId'], check_mode=check_mode)
 541         changed = changed or result['changed']
 542         if changed and check_mode:
 543             return {'changed': True}
 544         new_association_ids.append(result['association_id'])
 545 
 546     if purge_subnets:
 547         to_delete = [a_id for a_id in current_association_ids
 548                      if a_id not in new_association_ids]
 549 
 550         for a_id in to_delete:
 551             changed = True
 552             if not check_mode:
 553                 try:
 554                     connection.disassociate_route_table(AssociationId=a_id)
 555                 except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 556                     module.fail_json_aws(e, msg="Couldn't disassociate subnet from route table")
 557 
 558     return {'changed': changed}

 561 def ensure_propagation(connection=None, module=None, route_table=None, propagating_vgw_ids=None,
 562                        check_mode=None):
 563     changed = False
 564     gateways = [gateway['GatewayId'] for gateway in route_table['PropagatingVgws']]
 565     to_add = set(propagating_vgw_ids) - set(gateways)
 566     if to_add:
 567         changed = True
 568         if not check_mode:
 569             for vgw_id in to_add:
 570                 try:
 571                     connection.enable_vgw_route_propagation(RouteTableId=route_table['RouteTableId'],
 572                                                             GatewayId=vgw_id)
 573                 except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 574                     module.fail_json_aws(e, msg="Couldn't enable route propagation")
 575 
 576     return {'changed': changed}

 579 def ensure_route_table_absent(connection, module):
 580 
 581     lookup = module.params.get('lookup')
 582     route_table_id = module.params.get('route_table_id')
 583     tags = module.params.get('tags')
 584     vpc_id = module.params.get('vpc_id')
 585     purge_subnets = module.params.get('purge_subnets')
 586 
 587     if lookup == 'tag':
 588         if tags is not None:
 589             route_table = get_route_table_by_tags(connection, module, vpc_id, tags)
 590         else:
 591             route_table = None
 592     elif lookup == 'id':
 593         route_table = get_route_table_by_id(connection, module, route_table_id)
 594 
 595     if route_table is None:
 596         return {'changed': False}
 597 
 598     # disassociate subnets before deleting route table
 599     if not module.check_mode:
 600         ensure_subnet_associations(connection=connection, module=module, route_table=route_table,
 601                                    subnets=[], check_mode=False, purge_subnets=purge_subnets)
 602         try:
 603             connection.delete_route_table(RouteTableId=route_table['RouteTableId'])
 604         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 605             module.fail_json_aws(e, msg="Error deleting route table")
 606 
 607     return {'changed': True}

 610 def get_route_table_info(connection, module, route_table):
 611     result = get_route_table_by_id(connection, module, route_table['RouteTableId'])
 612     try:
 613         result['Tags'] = describe_tags_with_backoff(connection, route_table['RouteTableId'])
 614     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 615         module.fail_json_aws(e, msg="Couldn't get tags for route table")
 616     result = camel_dict_to_snake_dict(result, ignore_list=['Tags'])
 617     # backwards compatibility
 618     result['id'] = result['route_table_id']
 619     return result

 622 def create_route_spec(connection, module, vpc_id):
 623     routes = module.params.get('routes')
 624 
 625     for route_spec in routes:
 626         rename_key(route_spec, 'dest', 'destination_cidr_block')
 627 
 628         if route_spec.get('gateway_id') and route_spec['gateway_id'].lower() == 'igw':
 629             igw = find_igw(connection, module, vpc_id)
 630             route_spec['gateway_id'] = igw
 631         if route_spec.get('gateway_id') and route_spec['gateway_id'].startswith('nat-'):
 632             rename_key(route_spec, 'gateway_id', 'nat_gateway_id')
 633 
 634     return snake_dict_to_camel_dict(routes, capitalize_first=True)

 637 def ensure_route_table_present(connection, module):
 638 
 639     lookup = module.params.get('lookup')
 640     propagating_vgw_ids = module.params.get('propagating_vgw_ids')
 641     purge_routes = module.params.get('purge_routes')
 642     purge_subnets = module.params.get('purge_subnets')
 643     purge_tags = module.params.get('purge_tags')
 644     route_table_id = module.params.get('route_table_id')
 645     subnets = module.params.get('subnets')
 646     tags = module.params.get('tags')
 647     vpc_id = module.params.get('vpc_id')
 648     routes = create_route_spec(connection, module, vpc_id)
 649 
 650     changed = False
 651     tags_valid = False
 652 
 653     if lookup == 'tag':
 654         if tags is not None:
 655             try:
 656                 route_table = get_route_table_by_tags(connection, module, vpc_id, tags)
 657             except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 658                 module.fail_json_aws(e, msg="Error finding route table with lookup 'tag'")
 659         else:
 660             route_table = None
 661     elif lookup == 'id':
 662         try:
 663             route_table = get_route_table_by_id(connection, module, route_table_id)
 664         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 665             module.fail_json_aws(e, msg="Error finding route table with lookup 'id'")
 666 
 667     # If no route table returned then create new route table
 668     if route_table is None:
 669         changed = True
 670         if not module.check_mode:
 671             try:
 672                 route_table = connection.create_route_table(VpcId=vpc_id)['RouteTable']
 673                 # try to wait for route table to be present before moving on
 674                 get_waiter(
 675                     connection, 'route_table_exists'
 676                 ).wait(
 677                     RouteTableIds=[route_table['RouteTableId']],
 678                 )
 679             except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 680                 module.fail_json_aws(e, msg="Error creating route table")
 681         else:
 682             route_table = {"id": "rtb-xxxxxxxx", "route_table_id": "rtb-xxxxxxxx", "vpc_id": vpc_id}
 683             module.exit_json(changed=changed, route_table=route_table)
 684 
 685     if routes is not None:
 686         result = ensure_routes(connection=connection, module=module, route_table=route_table,
 687                                route_specs=routes, propagating_vgw_ids=propagating_vgw_ids,
 688                                check_mode=module.check_mode, purge_routes=purge_routes)
 689         changed = changed or result['changed']
 690 
 691     if propagating_vgw_ids is not None:
 692         result = ensure_propagation(connection=connection, module=module, route_table=route_table,
 693                                     propagating_vgw_ids=propagating_vgw_ids, check_mode=module.check_mode)
 694         changed = changed or result['changed']
 695 
 696     if not tags_valid and tags is not None:
 697         result = ensure_tags(connection=connection, module=module, resource_id=route_table['RouteTableId'], tags=tags,
 698                              purge_tags=purge_tags, check_mode=module.check_mode)
 699         route_table['Tags'] = result['tags']
 700         changed = changed or result['changed']
 701 
 702     if subnets is not None:
 703         associated_subnets = find_subnets(connection, module, vpc_id, subnets)
 704 
 705         result = ensure_subnet_associations(connection=connection, module=module, route_table=route_table,
 706                                             subnets=associated_subnets, check_mode=module.check_mode,
 707                                             purge_subnets=purge_subnets)
 708         changed = changed or result['changed']
 709 
 710     if changed:
 711         # pause to allow route table routes/subnets/associations to be updated before exiting with final state
 712         sleep(5)
 713     module.exit_json(changed=changed, route_table=get_route_table_info(connection, module, route_table))
