 277 def get_eni_info(interface):
 278 
 279     # Private addresses
 280     private_addresses = []
 281     for ip in interface.private_ip_addresses:
 282         private_addresses.append({'private_ip_address': ip.private_ip_address, 'primary_address': ip.primary})
 283 
 284     interface_info = {'id': interface.id,
 285                       'subnet_id': interface.subnet_id,
 286                       'vpc_id': interface.vpc_id,
 287                       'description': interface.description,
 288                       'owner_id': interface.owner_id,
 289                       'status': interface.status,
 290                       'mac_address': interface.mac_address,
 291                       'private_ip_address': interface.private_ip_address,
 292                       'source_dest_check': interface.source_dest_check,
 293                       'groups': dict((group.id, group.name) for group in interface.groups),
 294                       'private_ip_addresses': private_addresses
 295                       }
 296 
 297     if interface.attachment is not None:
 298         interface_info['attachment'] = {'attachment_id': interface.attachment.id,
 299                                         'instance_id': interface.attachment.instance_id,
 300                                         'device_index': interface.attachment.device_index,
 301                                         'status': interface.attachment.status,
 302                                         'attach_time': interface.attachment.attach_time,
 303                                         'delete_on_termination': interface.attachment.delete_on_termination,
 304                                         }
 305 
 306     return interface_info

 309 def wait_for_eni(eni, status):
 310 
 311     while True:
 312         time.sleep(3)
 313         eni.update()
 314         # If the status is detached we just need attachment to disappear
 315         if eni.attachment is None:
 316             if status == "detached":
 317                 break
 318         else:
 319             if status == "attached" and eni.attachment.status == "attached":
 320                 break

 323 def create_eni(connection, vpc_id, module):
 324 
 325     instance_id = module.params.get("instance_id")
 326     attached = module.params.get("attached")
 327     if instance_id == 'None':
 328         instance_id = None
 329     device_index = module.params.get("device_index")
 330     subnet_id = module.params.get('subnet_id')
 331     private_ip_address = module.params.get('private_ip_address')
 332     description = module.params.get('description')
 333     security_groups = get_ec2_security_group_ids_from_names(module.params.get('security_groups'), connection, vpc_id=vpc_id, boto3=False)
 334     secondary_private_ip_addresses = module.params.get("secondary_private_ip_addresses")
 335     secondary_private_ip_address_count = module.params.get("secondary_private_ip_address_count")
 336     changed = False
 337 
 338     try:
 339         eni = connection.create_network_interface(subnet_id, private_ip_address, description, security_groups)
 340         if attached and instance_id is not None:
 341             try:
 342                 eni.attach(instance_id, device_index)
 343             except BotoServerError:
 344                 eni.delete()
 345                 raise
 346             # Wait to allow creation / attachment to finish
 347             wait_for_eni(eni, "attached")
 348             eni.update()
 349 
 350         if secondary_private_ip_address_count is not None:
 351             try:
 352                 connection.assign_private_ip_addresses(network_interface_id=eni.id, secondary_private_ip_address_count=secondary_private_ip_address_count)
 353             except BotoServerError:
 354                 eni.delete()
 355                 raise
 356 
 357         if secondary_private_ip_addresses is not None:
 358             try:
 359                 connection.assign_private_ip_addresses(network_interface_id=eni.id, private_ip_addresses=secondary_private_ip_addresses)
 360             except BotoServerError:
 361                 eni.delete()
 362                 raise
 363 
 364         changed = True
 365 
 366     except BotoServerError as e:
 367         module.fail_json(msg=e.message)
 368 
 369     module.exit_json(changed=changed, interface=get_eni_info(eni))

 372 def modify_eni(connection, vpc_id, module, eni):
 373 
 374     instance_id = module.params.get("instance_id")
 375     attached = module.params.get("attached")
 376     do_detach = module.params.get('state') == 'detached'
 377     device_index = module.params.get("device_index")
 378     description = module.params.get('description')
 379     security_groups = module.params.get('security_groups')
 380     force_detach = module.params.get("force_detach")
 381     source_dest_check = module.params.get("source_dest_check")
 382     delete_on_termination = module.params.get("delete_on_termination")
 383     secondary_private_ip_addresses = module.params.get("secondary_private_ip_addresses")
 384     purge_secondary_private_ip_addresses = module.params.get("purge_secondary_private_ip_addresses")
 385     secondary_private_ip_address_count = module.params.get("secondary_private_ip_address_count")
 386     allow_reassignment = module.params.get("allow_reassignment")
 387     changed = False
 388 
 389     try:
 390         if description is not None:
 391             if eni.description != description:
 392                 connection.modify_network_interface_attribute(eni.id, "description", description)
 393                 changed = True
 394         if len(security_groups) > 0:
 395             groups = get_ec2_security_group_ids_from_names(security_groups, connection, vpc_id=vpc_id, boto3=False)
 396             if sorted(get_sec_group_list(eni.groups)) != sorted(groups):
 397                 connection.modify_network_interface_attribute(eni.id, "groupSet", groups)
 398                 changed = True
 399         if source_dest_check is not None:
 400             if eni.source_dest_check != source_dest_check:
 401                 connection.modify_network_interface_attribute(eni.id, "sourceDestCheck", source_dest_check)
 402                 changed = True
 403         if delete_on_termination is not None and eni.attachment is not None:
 404             if eni.attachment.delete_on_termination is not delete_on_termination:
 405                 connection.modify_network_interface_attribute(eni.id, "deleteOnTermination", delete_on_termination, eni.attachment.id)
 406                 changed = True
 407 
 408         current_secondary_addresses = [i.private_ip_address for i in eni.private_ip_addresses if not i.primary]
 409         if secondary_private_ip_addresses is not None:
 410             secondary_addresses_to_remove = list(set(current_secondary_addresses) - set(secondary_private_ip_addresses))
 411             if secondary_addresses_to_remove and purge_secondary_private_ip_addresses:
 412                 connection.unassign_private_ip_addresses(network_interface_id=eni.id,
 413                                                          private_ip_addresses=list(set(current_secondary_addresses) -
 414                                                                                    set(secondary_private_ip_addresses)),
 415                                                          dry_run=False)
 416                 changed = True
 417 
 418             secondary_addresses_to_add = list(set(secondary_private_ip_addresses) - set(current_secondary_addresses))
 419             if secondary_addresses_to_add:
 420                 connection.assign_private_ip_addresses(network_interface_id=eni.id,
 421                                                        private_ip_addresses=secondary_addresses_to_add,
 422                                                        secondary_private_ip_address_count=None,
 423                                                        allow_reassignment=allow_reassignment, dry_run=False)
 424                 changed = True
 425         if secondary_private_ip_address_count is not None:
 426             current_secondary_address_count = len(current_secondary_addresses)
 427 
 428             if secondary_private_ip_address_count > current_secondary_address_count:
 429                 connection.assign_private_ip_addresses(network_interface_id=eni.id,
 430                                                        private_ip_addresses=None,
 431                                                        secondary_private_ip_address_count=(secondary_private_ip_address_count -
 432                                                                                            current_secondary_address_count),
 433                                                        allow_reassignment=allow_reassignment, dry_run=False)
 434                 changed = True
 435             elif secondary_private_ip_address_count < current_secondary_address_count:
 436                 # How many of these addresses do we want to remove
 437                 secondary_addresses_to_remove_count = current_secondary_address_count - secondary_private_ip_address_count
 438                 connection.unassign_private_ip_addresses(network_interface_id=eni.id,
 439                                                          private_ip_addresses=current_secondary_addresses[:secondary_addresses_to_remove_count],
 440                                                          dry_run=False)
 441 
 442         if attached is True:
 443             if eni.attachment and eni.attachment.instance_id != instance_id:
 444                 detach_eni(eni, module)
 445                 eni.attach(instance_id, device_index)
 446                 wait_for_eni(eni, "attached")
 447                 changed = True
 448             if eni.attachment is None:
 449                 eni.attach(instance_id, device_index)
 450                 wait_for_eni(eni, "attached")
 451                 changed = True
 452         elif attached is False:
 453             detach_eni(eni, module)
 454 
 455     except BotoServerError as e:
 456         module.fail_json(msg=e.message)
 457 
 458     eni.update()
 459     module.exit_json(changed=changed, interface=get_eni_info(eni))

 462 def delete_eni(connection, module):
 463 
 464     eni_id = module.params.get("eni_id")
 465     force_detach = module.params.get("force_detach")
 466 
 467     try:
 468         eni_result_set = connection.get_all_network_interfaces(eni_id)
 469         eni = eni_result_set[0]
 470 
 471         if force_detach is True:
 472             if eni.attachment is not None:
 473                 eni.detach(force_detach)
 474                 # Wait to allow detachment to finish
 475                 wait_for_eni(eni, "detached")
 476                 eni.update()
 477             eni.delete()
 478             changed = True
 479         else:
 480             eni.delete()
 481             changed = True
 482 
 483         module.exit_json(changed=changed)
 484     except BotoServerError as e:
 485         regex = re.compile('The networkInterface ID \'.*\' does not exist')
 486         if regex.search(e.message) is not None:
 487             module.exit_json(changed=False)
 488         else:
 489             module.fail_json(msg=e.message)

 492 def detach_eni(eni, module):
 493 
 494     attached = module.params.get("attached")
 495 
 496     force_detach = module.params.get("force_detach")
 497     if eni.attachment is not None:
 498         eni.detach(force_detach)
 499         wait_for_eni(eni, "detached")
 500         if attached:
 501             return
 502         eni.update()
 503         module.exit_json(changed=True, interface=get_eni_info(eni))
 504     else:
 505         module.exit_json(changed=False, interface=get_eni_info(eni))

 508 def uniquely_find_eni(connection, module):
 509 
 510     eni_id = module.params.get("eni_id")
 511     private_ip_address = module.params.get('private_ip_address')
 512     subnet_id = module.params.get('subnet_id')
 513     instance_id = module.params.get('instance_id')
 514     device_index = module.params.get('device_index')
 515     attached = module.params.get('attached')
 516 
 517     try:
 518         filters = {}
 519 
 520         # proceed only if we're univocally specifying an ENI
 521         if eni_id is None and private_ip_address is None and (instance_id is None and device_index is None):
 522             return None
 523 
 524         if private_ip_address and subnet_id:
 525             filters['private-ip-address'] = private_ip_address
 526             filters['subnet-id'] = subnet_id
 527 
 528         if not attached and instance_id and device_index:
 529             filters['attachment.instance-id'] = instance_id
 530             filters['attachment.device-index'] = device_index
 531 
 532         if eni_id is None and len(filters) == 0:
 533             return None
 534 
 535         eni_result = connection.get_all_network_interfaces(eni_id, filters=filters)
 536         if len(eni_result) == 1:
 537             return eni_result[0]
 538         else:
 539             return None
 540 
 541     except BotoServerError as e:
 542         module.fail_json(msg=e.message)
 543 
 544     return None

 547 def get_sec_group_list(groups):
 548 
 549     # Build list of remote security groups
 550     remote_security_groups = []
 551     for group in groups:
 552         remote_security_groups.append(group.id.encode())
 553 
 554     return remote_security_groups

 565 def main():
 566     argument_spec = ec2_argument_spec()
 567     argument_spec.update(
 568         dict(
 569             eni_id=dict(default=None, type='str'),
 570             instance_id=dict(default=None, type='str'),
 571             private_ip_address=dict(type='str'),
 572             subnet_id=dict(type='str'),
 573             description=dict(type='str'),
 574             security_groups=dict(default=[], type='list'),
 575             device_index=dict(default=0, type='int'),
 576             state=dict(default='present', choices=['present', 'absent']),
 577             force_detach=dict(default='no', type='bool'),
 578             source_dest_check=dict(default=None, type='bool'),
 579             delete_on_termination=dict(default=None, type='bool'),
 580             secondary_private_ip_addresses=dict(default=None, type='list'),
 581             purge_secondary_private_ip_addresses=dict(default=False, type='bool'),
 582             secondary_private_ip_address_count=dict(default=None, type='int'),
 583             allow_reassignment=dict(default=False, type='bool'),
 584             attached=dict(default=None, type='bool')
 585         )
 586     )
 587 
 588     module = AnsibleModule(argument_spec=argument_spec,
 589                            mutually_exclusive=[
 590                                ['secondary_private_ip_addresses', 'secondary_private_ip_address_count']
 591                            ],
 592                            required_if=([
 593                                ('state', 'absent', ['eni_id']),
 594                                ('attached', True, ['instance_id']),
 595                                ('purge_secondary_private_ip_addresses', True, ['secondary_private_ip_addresses'])
 596                            ])
 597                            )
 598 
 599     if not HAS_BOTO:
 600         module.fail_json(msg='boto required for this module')
 601 
 602     region, ec2_url, aws_connect_params = get_aws_connection_info(module)
 603 
 604     if region:
 605         try:
 606             connection = connect_to_aws(boto.ec2, region, **aws_connect_params)
 607             vpc_connection = connect_to_aws(boto.vpc, region, **aws_connect_params)
 608         except (boto.exception.NoAuthHandlerFound, AnsibleAWSError) as e:
 609             module.fail_json(msg=str(e))
 610     else:
 611         module.fail_json(msg="region must be specified")
 612 
 613     state = module.params.get("state")
 614 
 615     if state == 'present':
 616         eni = uniquely_find_eni(connection, module)
 617         if eni is None:
 618             subnet_id = module.params.get("subnet_id")
 619             if subnet_id is None:
 620                 module.fail_json(msg="subnet_id is required when creating a new ENI")
 621 
 622             vpc_id = _get_vpc_id(vpc_connection, module, subnet_id)
 623             create_eni(connection, vpc_id, module)
 624         else:
 625             vpc_id = eni.vpc_id
 626             modify_eni(connection, vpc_id, module, eni)
 627 
 628     elif state == 'absent':
 629         delete_eni(connection, module)
 630 
 631 
 632 if __name__ == '__main__':
 633     main()
