 398             return template_ids

 481             if tags is not None:
 482                 parameters['tags'] = tags

 484             self._zapi.host.update(parameters)
 485             interface_list_copy = exist_interface_list
 486             if interfaces:
 487                 for interface in interfaces:
 488                     flag = False
 489                     interface_str = interface
 490                     for exist_interface in exist_interface_list:
 491                         interface_type = int(interface['type'])
 492                         exist_interface_type = int(exist_interface['type'])
 493                         if interface_type == exist_interface_type:
 494                             # update
 495                             interface_str['interfaceid'] = exist_interface['interfaceid']
 496                             self._zapi.hostinterface.update(interface_str)
 497                             flag = True
 498                             interface_list_copy.remove(exist_interface)
 499                             break
 500                     if not flag:
 501                         # add
 502                         interface_str['hostid'] = host_id
 503                         self._zapi.hostinterface.create(interface_str)
 504                         # remove
 505                 remove_interface_ids = []
 506                 for remove_interface in interface_list_copy:
 507                     interface_id = remove_interface['interfaceid']

 567     # check the exist_interfaces whether it equals the interfaces or not
 568     def check_interface_properties(self, exist_interface_list, interfaces):
 569         interfaces_port_list = []
 570 
 571         if interfaces is not None:
 572             if len(interfaces) >= 1:
 573                 for interface in interfaces:
 574                     interfaces_port_list.append(str(interface['port']))
 575 
 576         exist_interface_ports = []
 577         if len(exist_interface_list) >= 1:
 578             for exist_interface in exist_interface_list:
 579                 exist_interface_ports.append(str(exist_interface['port']))
 580 
 581         if set(interfaces_port_list) != set(exist_interface_ports):
 582             return True
 583 
 584         for exist_interface in exist_interface_list:
 585             exit_interface_port = str(exist_interface['port'])
 586             for interface in interfaces:
 587                 interface_port = str(interface['port'])
 588                 if interface_port == exit_interface_port:
 589                     for key in interface.keys():
 590                         if str(exist_interface[key]) != str(interface[key]):
 591                             return True
 592 
 593         return False

 599     # check all the properties before link or clear template
 600     def check_all_properties(self, host_id, group_ids, status, interfaces, template_ids,
 601                              exist_interfaces, host, proxy_id, visible_name, description, host_name,
 602                              inventory_mode, inventory_zabbix, tls_accept, tls_psk_identity, tls_psk,
 603                              tls_issuer, tls_subject, tls_connect, ipmi_authtype, ipmi_privilege,
 604                              ipmi_username, ipmi_password, macros, tags):
 605         # get the existing host's groups
 606         exist_host_groups = sorted(self.get_group_ids_by_host_id(host_id), key=lambda k: k['groupid'])
 607         if sorted(group_ids, key=lambda k: k['groupid']) != exist_host_groups:
 608             return True
 609 
 610         # get the existing status
 611         exist_status = self.get_host_status_by_host(host)
 612         if int(status) != int(exist_status):
 613             return True
 614 
 615         # check the exist_interfaces whether it equals the interfaces or not
 616         if self.check_interface_properties(exist_interfaces, interfaces):
 617             return True
 618 
 619         # get the existing templates
 620         exist_template_ids = self.get_host_templates_by_host_id(host_id)
 621         if set(list(template_ids)) != set(exist_template_ids):
 622             return True
 623 
 624         if int(host['proxy_hostid']) != int(proxy_id):
 625             return True
 626 
 627         # Check whether the visible_name has changed; Zabbix defaults to the technical hostname if not set.
 628         if visible_name:
 629             if host['name'] != visible_name:
 630                 return True
 631 
 632         # Only compare description if it is given as a module parameter
 633         if description:
 634             if host['description'] != description:

 640                     if int(host['inventory']['inventory_mode']) != self.inventory_mode_numeric(inventory_mode):
 641                         return True
 642                 elif inventory_mode != 'disabled':
 643                     return True
 644             else:
 645                 if int(host['inventory_mode']) != self.inventory_mode_numeric(inventory_mode):
 646                     return True
 647 
 648         if inventory_zabbix:
 649             proposed_inventory = copy.deepcopy(host['inventory'])
 650             proposed_inventory.update(inventory_zabbix)
 651             if proposed_inventory != host['inventory']:
 652                 return True
 653 
 654         if tls_accept is not None and 'tls_accept' in host:
 655             if int(host['tls_accept']) != tls_accept:
 656                 return True
 657 
 658         if tls_psk_identity is not None and 'tls_psk_identity' in host:
 659             if host['tls_psk_identity'] != tls_psk_identity:
 660                 return True
 661 
 662         if tls_psk is not None and 'tls_psk' in host:
 663             if host['tls_psk'] != tls_psk:
 664                 return True
 665 
 666         if tls_issuer is not None and 'tls_issuer' in host:
 667             if host['tls_issuer'] != tls_issuer:
 668                 return True
 669 
 670         if tls_subject is not None and 'tls_subject' in host:
 671             if host['tls_subject'] != tls_subject:
 672                 return True
 673 
 674         if tls_connect is not None and 'tls_connect' in host:
 675             if int(host['tls_connect']) != tls_connect:
 676                 return True
 677         if ipmi_authtype is not None:
 678             if int(host['ipmi_authtype']) != ipmi_authtype:
 679                 return True
 680         if ipmi_privilege is not None:
 681             if int(host['ipmi_privilege']) != ipmi_privilege:
 682                 return True
 683         if ipmi_username is not None:
 684             if host['ipmi_username'] != ipmi_username:
 685                 return True
 686         if ipmi_password is not None:
 687             if host['ipmi_password'] != ipmi_password:
 688                 return True
 689 
 690         # hostmacroid and hostid are present in every item of host['macros'] and need to be removed
 691         if macros is not None and 'macros' in host:
 692             existing_macros = sorted(host['macros'], key=lambda k: k['macro'])
 693             for macro in existing_macros:
 694                 macro.pop('hostid', False)
 695                 macro.pop('hostmacroid', False)
 696 
 697             if sorted(macros, key=lambda k: k['macro']) != existing_macros:
 698                 return True
 699 
 700         if tags is not None and 'tags' in host:
 701             if sorted(tags, key=lambda k: k['tag']) != sorted(host['tags'], key=lambda k: k['tag']):
 702                 return True
 703 
 704         return False

 707     def link_or_clear_template(self, host_id, template_id_list, tls_connect, tls_accept, tls_psk_identity, tls_psk,
 708                                tls_issuer, tls_subject, ipmi_authtype, ipmi_privilege, ipmi_username, ipmi_password):
 709         # get host's exist template ids

 714         template_id_list = list(template_ids)

 716         # get unlink and clear templates
 717         templates_clear = exist_template_ids.difference(template_ids)
 718         templates_clear_list = list(templates_clear)
 719         request_str = {'hostid': host_id, 'templates': template_id_list, 'templates_clear': templates_clear_list,
 720                        'tls_connect': tls_connect, 'tls_accept': tls_accept, 'ipmi_authtype': ipmi_authtype,
 721                        'ipmi_privilege': ipmi_privilege, 'ipmi_username': ipmi_username, 'ipmi_password': ipmi_password}
 722         if tls_psk_identity is not None:
 723             request_str['tls_psk_identity'] = tls_psk_identity
 724         if tls_psk is not None:
 725             request_str['tls_psk'] = tls_psk
 726         if tls_issuer is not None:
 727             request_str['tls_issuer'] = tls_issuer
 728         if tls_subject is not None:
 729             request_str['tls_subject'] = tls_subject
 730         try:
 731             if self._module.check_mode:
 732                 self._module.exit_json(changed=True)
 733             self._zapi.host.update(request_str)
 734         except Exception as e:
 735             self._module.fail_json(msg="Failed to link template to host: %s" % e)
