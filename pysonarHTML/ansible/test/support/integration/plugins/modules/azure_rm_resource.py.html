 315     def exec_module(self, **kwargs):
 316         for key in self.module_arg_spec:
 317             setattr(self, key, kwargs[key])
 318         self.mgmt_client = self.get_mgmt_svc_client(GenericRestClient,
 319                                                     base_url=self._cloud_environment.endpoints.resource_manager)
 320 
 321         if self.state == 'absent':
 322             self.method = 'DELETE'
 323             self.status_code.append(204)
 324 
 325         if self.url is None:
 326             orphan = None
 327             rargs = dict()
 328             rargs['subscription'] = self.subscription_id
 329             rargs['resource_group'] = self.resource_group
 330             if not (self.provider is None or self.provider.lower().startswith('.microsoft')):
 331                 rargs['namespace'] = "Microsoft." + self.provider
 332             else:
 333                 rargs['namespace'] = self.provider
 334 
 335             if self.resource_type is not None and self.resource_name is not None:
 336                 rargs['type'] = self.resource_type
 337                 rargs['name'] = self.resource_name
 338                 for i in range(len(self.subresource)):
 339                     resource_ns = self.subresource[i].get('namespace', None)
 340                     resource_type = self.subresource[i].get('type', None)
 341                     resource_name = self.subresource[i].get('name', None)
 342                     if resource_type is not None and resource_name is not None:
 343                         rargs['child_namespace_' + str(i + 1)] = resource_ns
 344                         rargs['child_type_' + str(i + 1)] = resource_type
 345                         rargs['child_name_' + str(i + 1)] = resource_name
 346                     else:
 347                         orphan = resource_type
 348             else:
 349                 orphan = self.resource_type
 350 
 351             self.url = resource_id(**rargs)
 352 
 353             if orphan is not None:
 354                 self.url += '/' + orphan
 355 
 356         # if api_version was not specified, get latest one
 357         if not self.api_version:
 358             try:
 359                 # extract provider and resource type
 360                 if "/providers/" in self.url:
 361                     provider = self.url.split("/providers/")[1].split("/")[0]
 362                     resourceType = self.url.split(provider + "/")[1].split("/")[0]
 363                     url = "/subscriptions/" + self.subscription_id + "/providers/" + provider
 364                     api_versions = json.loads(self.mgmt_client.query(url, "GET", {'api-version': '2015-01-01'}, None, None, [200], 0, 0).text)
 365                     for rt in api_versions['resourceTypes']:
 366                         if rt['resourceType'].lower() == resourceType.lower():
 367                             self.api_version = rt['apiVersions'][0]
 368                             break
 369                 else:
 370                     # if there's no provider in API version, assume Microsoft.Resources
 371                     self.api_version = '2018-05-01'
 372                 if not self.api_version:
 373                     self.fail("Couldn't find api version for {0}/{1}".format(provider, resourceType))
 374             except Exception as exc:
 375                 self.fail("Failed to obtain API version: {0}".format(str(exc)))
 376 
 377         query_parameters = {}
 378         query_parameters['api-version'] = self.api_version
 379 
 380         header_parameters = {}
 381         header_parameters['Content-Type'] = 'application/json; charset=utf-8'
 382 
 383         needs_update = True
 384         response = None
 385 
 386         if self.idempotency:
 387             original = self.mgmt_client.query(self.url, "GET", query_parameters, None, None, [200, 404], 0, 0)
 388 
 389             if original.status_code == 404:
 390                 if self.state == 'absent':
 391                     needs_update = False
 392             else:
 393                 try:
 394                     response = json.loads(original.text)
 395                     needs_update = (dict_merge(response, self.body) != response)
 396                 except Exception:
 397                     pass
 398 
 399         if needs_update:
 400             response = self.mgmt_client.query(self.url,
 401                                               self.method,
 402                                               query_parameters,
 403                                               header_parameters,
 404                                               self.body,
 405                                               self.status_code,
 406                                               self.polling_timeout,
 407                                               self.polling_interval)
 408             if self.state == 'present':
 409                 try:
 410                     response = json.loads(response.text)
 411                 except Exception:
 412                     response = response.text
 413             else:
 414                 response = None
 415 
 416         self.results['response'] = response
 417         self.results['changed'] = needs_update
 418 
 419         return self.results
