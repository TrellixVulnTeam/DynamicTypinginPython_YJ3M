 145 def extract_key_data(key):
 146 
 147     data = {
 148         'name': key['KeyName'],
 149         'fingerprint': key['KeyFingerprint']
 150     }
 151     if 'KeyMaterial' in key:
 152         data['private_key'] = key['KeyMaterial']
 153     return data
 154

 156 def get_key_fingerprint(module, ec2_client, key_material):
 157     '''
 158     EC2's fingerprints are non-trivial to generate, so push this key
 159     to a temporary name and make ec2 calculate the fingerprint for us.
 160     http://blog.jbrowne.com/?p=23
 161     https://forums.aws.amazon.com/thread.jspa?messageID=352828
 162     '''
 163 
 164     # find an unused name
 165     name_in_use = True
 166     while name_in_use:
 167         random_name = "ansible-" + str(uuid.uuid4())
 168         name_in_use = find_key_pair(module, ec2_client, random_name)
 169 
 170     temp_key = import_key_pair(module, ec2_client, random_name, key_material)
 171     delete_key_pair(module, ec2_client, random_name, finish_task=False)
 172     return temp_key['KeyFingerprint']

 175 def find_key_pair(module, ec2_client, name):
 176 
 177     try:
 178         key = ec2_client.describe_key_pairs(KeyNames=[name])['KeyPairs'][0]
 179     except ClientError as err:
 180         if err.response['Error']['Code'] == "InvalidKeyPair.NotFound":
 181             return None
 182         module.fail_json_aws(err, msg="error finding keypair")
 183     except IndexError:
 184         key = None
 185     return key
 186

 188 def create_key_pair(module, ec2_client, name, key_material, force):
 189 
 190     key = find_key_pair(module, ec2_client, name)
 191     if key:
 192         if key_material and force:
 193             if not module.check_mode:
 194                 new_fingerprint = get_key_fingerprint(module, ec2_client, key_material)
 195                 if key['KeyFingerprint'] != new_fingerprint:
 196                     delete_key_pair(module, ec2_client, name, finish_task=False)
 197                     key = import_key_pair(module, ec2_client, name, key_material)
 198                     key_data = extract_key_data(key)
 199                     module.exit_json(changed=True, key=key_data, msg="key pair updated")
 200             else:
 201                 # Assume a change will be made in check mode since a comparison can't be done
 202                 module.exit_json(changed=True, key=extract_key_data(key), msg="key pair updated")
 203         key_data = extract_key_data(key)
 204         module.exit_json(changed=False, key=key_data, msg="key pair already exists")
 205     else:
 206         # key doesn't exist, create it now
 207         key_data = None
 208         if not module.check_mode:
 209             if key_material:
 210                 key = import_key_pair(module, ec2_client, name, key_material)
 211             else:
 212                 try:
 213                     key = ec2_client.create_key_pair(KeyName=name)
 214                 except ClientError as err:
 215                     module.fail_json_aws(err, msg="error creating key")
 216             key_data = extract_key_data(key)
 217         module.exit_json(changed=True, key=key_data, msg="key pair created")

 220 def import_key_pair(module, ec2_client, name, key_material):
 221 
 222     try:
 223         key = ec2_client.import_key_pair(KeyName=name, PublicKeyMaterial=to_bytes(key_material))
 224     except ClientError as err:
 225         module.fail_json_aws(err, msg="error importing key")
 226     return key
 227 
 228 
 229 def delete_key_pair(module, ec2_client, name, finish_task=True):
 230 
 231     key = find_key_pair(module, ec2_client, name)
 232     if key:
 233         if not module.check_mode:
 234             try:
 235                 ec2_client.delete_key_pair(KeyName=name)
 236             except ClientError as err:
 237                 module.fail_json_aws(err, msg="error deleting key")
 238         if not finish_task:
 239             return
 240         module.exit_json(changed=True, key=None, msg="key deleted")
 241     module.exit_json(key=None, msg="key did not exist")

 244 def main():
 245 
 246     argument_spec = dict(
 247         name=dict(required=True),
 248         key_material=dict(),
 249         force=dict(type='bool', default=True),
 250         state=dict(default='present', choices=['present', 'absent']),
 251         wait=dict(type='bool', removed_in_version='2.14'),
 252         wait_timeout=dict(type='int', removed_in_version='2.14')
 253     )
 254 
 255     module = AnsibleAWSModule(argument_spec=argument_spec, supports_check_mode=True)
 256 
 257     ec2_client = module.client('ec2')
 258 
 259     name = module.params['name']
 260     state = module.params.get('state')
 261     key_material = module.params.get('key_material')
 262     force = module.params.get('force')
 263 
 264     if state == 'absent':
 265         delete_key_pair(module, ec2_client, name)
 266     elif state == 'present':
 267         create_key_pair(module, ec2_client, name, key_material, force)
