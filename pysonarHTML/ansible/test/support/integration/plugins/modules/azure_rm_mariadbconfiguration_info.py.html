 139     def exec_module(self, **kwargs):
 140         is_old_facts = self.module._name == 'azure_rm_mariadbconfiguration_facts'
 141         if is_old_facts:
 142             self.module.deprecate("The 'azure_rm_mariadbconfiguration_facts' module has been renamed to 'azure_rm_mariadbconfiguration_info'",
 143                                   version='2.13', collection_name='ansible.builtin')
 144 
 145         for key in self.module_arg_spec:
 146             setattr(self, key, kwargs[key])
 147         self.mgmt_client = self.get_mgmt_svc_client(MariaDBManagementClient,
 148                                                     base_url=self._cloud_environment.endpoints.resource_manager)
 149 
 150         if self.name is not None:
 151             self.results['settings'] = self.get()
 152         else:
 153             self.results['settings'] = self.list_by_server()
 154         return self.results
 155 
 156     def get(self):
 157         '''
 158         Gets facts of the specified MariaDB Configuration.
 159 
 160         :return: deserialized MariaDB Configurationinstance state dictionary
 161         '''
 162         response = None
 163         results = []
 164         try:
 165             response = self.mgmt_client.configurations.get(resource_group_name=self.resource_group,
 166                                                            server_name=self.server_name,
 167                                                            configuration_name=self.name)
 168             self.log("Response : {0}".format(response))
 169         except CloudError as e:
 170             self.log('Could not get facts for Configurations.')
 171 
 172         if response is not None:
 173             results.append(self.format_item(response))
 174 
 175         return results
 176 
 177     def list_by_server(self):
 178         '''
 179         Gets facts of the specified MariaDB Configuration.
 180 
 181         :return: deserialized MariaDB Configurationinstance state dictionary
 182         '''
 183         response = None
 184         results = []
 185         try:
 186             response = self.mgmt_client.configurations.list_by_server(resource_group_name=self.resource_group,
 187                                                                       server_name=self.server_name)
 188             self.log("Response : {0}".format(response))
 189         except CloudError as e:
 190             self.log('Could not get facts for Configurations.')
 191 
 192         if response is not None:
 193             for item in response:
 194                 results.append(self.format_item(item))
 195 
 196         return results
 197 
 198     def format_item(self, item):
 199         d = item.as_dict()
 200         d = {
 201             'resource_group': self.resource_group,
 202             'server_name': self.server_name,
 203             'id': d['id'],
 204             'name': d['name'],
 205             'value': d['value'],
 206             'description': d['description'],
 207             'source': d['source']
 208         }
 209         return d
