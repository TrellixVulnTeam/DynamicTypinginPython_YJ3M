 290 # User Authentication Management changed in MySQL 5.7 and MariaDB 10.2.0
 291 def use_old_user_mgmt(cursor):
 292     cursor.execute("SELECT VERSION()")
 293     result = cursor.fetchone()
 294     version_str = result[0]
 295     version = version_str.split('.')
 296 
 297     if 'mariadb' in version_str.lower():
 298         # Prior to MariaDB 10.2
 299         if int(version[0]) * 1000 + int(version[1]) < 10002:
 300             return True
 301         else:
 302             return False
 303     else:
 304         # Prior to MySQL 5.7
 305         if int(version[0]) * 1000 + int(version[1]) < 5007:
 306             return True
 307         else:
 308             return False

 311 def get_mode(cursor):
 312     cursor.execute('SELECT @@GLOBAL.sql_mode')
 313     result = cursor.fetchone()
 314     mode_str = result[0]
 315     if 'ANSI' in mode_str:
 316         mode = 'ANSI'
 317     else:
 318         mode = 'NOTANSI'
 319     return mode

 332 def user_add(cursor, user, host, host_all, password, encrypted,
 333              plugin, plugin_hash_string, plugin_auth_string, new_priv, check_mode):
 334     # we cannot create users without a proper hostname
 335     if host_all:
 336         return False
 337 
 338     if check_mode:
 339         return True
 340 
 341     if password and encrypted:
 342         cursor.execute("CREATE USER %s@%s IDENTIFIED BY PASSWORD %s", (user, host, password))
 343     elif password and not encrypted:
 344         cursor.execute("CREATE USER %s@%s IDENTIFIED BY %s", (user, host, password))
 345     elif plugin and plugin_hash_string:
 346         cursor.execute("CREATE USER %s@%s IDENTIFIED WITH %s AS %s", (user, host, plugin, plugin_hash_string))
 347     elif plugin and plugin_auth_string:
 348         cursor.execute("CREATE USER %s@%s IDENTIFIED WITH %s BY %s", (user, host, plugin, plugin_auth_string))
 349     elif plugin:
 350         cursor.execute("CREATE USER %s@%s IDENTIFIED WITH %s", (user, host, plugin))
 351     else:
 352         cursor.execute("CREATE USER %s@%s", (user, host))
 353     if new_priv is not None:
 354         for db_table, priv in iteritems(new_priv):
 355             privileges_grant(cursor, user, host, db_table, priv)
 356     return True

 359 def is_hash(password):
 360     ishash = False
 361     if len(password) == 41 and password[0] == '*':
 362         if frozenset(password[1:]).issubset(string.hexdigits):
 363             ishash = True
 364     return ishash

 367 def user_mod(cursor, user, host, host_all, password, encrypted,
 368              plugin, plugin_hash_string, plugin_auth_string, new_priv, append_privs, module):
 369     changed = False
 370     msg = "User unchanged"
 371     grant_option = False
 372 
 373     if host_all:
 374         hostnames = user_get_hostnames(cursor, [user])
 375     else:
 376         hostnames = [host]
 377 
 378     for host in hostnames:
 379         # Handle clear text and hashed passwords.
 380         if bool(password):
 381             # Determine what user management method server uses
 382             old_user_mgmt = use_old_user_mgmt(cursor)
 383 
 384             # Get a list of valid columns in mysql.user table to check if Password and/or authentication_string exist
 385             cursor.execute("""
 386                 SELECT COLUMN_NAME FROM information_schema.COLUMNS
 387                 WHERE TABLE_SCHEMA = 'mysql' AND TABLE_NAME = 'user' AND COLUMN_NAME IN ('Password', 'authentication_string')
 388                 ORDER BY COLUMN_NAME DESC LIMIT 1
 389             """)
 390             colA = cursor.fetchone()
 391 
 392             cursor.execute("""
 393                 SELECT COLUMN_NAME FROM information_schema.COLUMNS
 394                 WHERE TABLE_SCHEMA = 'mysql' AND TABLE_NAME = 'user' AND COLUMN_NAME IN ('Password', 'authentication_string')
 395                 ORDER BY COLUMN_NAME ASC  LIMIT 1
 396             """)
 397             colB = cursor.fetchone()
 398 
 399             # Select hash from either Password or authentication_string, depending which one exists and/or is filled
 400             cursor.execute("""
 401                 SELECT COALESCE(
 402                         CASE WHEN %s = '' THEN NULL ELSE %s END,
 403                         CASE WHEN %s = '' THEN NULL ELSE %s END
 404                     )
 405                 FROM mysql.user WHERE user = %%s AND host = %%s
 406                 """ % (colA[0], colA[0], colB[0], colB[0]), (user, host))
 407             current_pass_hash = cursor.fetchone()[0]
 408             if isinstance(current_pass_hash, bytes):
 409                 current_pass_hash = current_pass_hash.decode('ascii')
 410 
 411             if encrypted:
 412                 encrypted_password = password
 413                 if not is_hash(encrypted_password):
 414                     module.fail_json(msg="encrypted was specified however it does not appear to be a valid hash expecting: *SHA1(SHA1(your_password))")
 415             else:
 416                 if old_user_mgmt:
 417                     cursor.execute("SELECT PASSWORD(%s)", (password,))
 418                 else:
 419                     cursor.execute("SELECT CONCAT('*', UCASE(SHA1(UNHEX(SHA1(%s)))))", (password,))
 420                 encrypted_password = cursor.fetchone()[0]
 421 
 422             if current_pass_hash != encrypted_password:
 423                 msg = "Password updated"
 424                 if module.check_mode:
 425                     return (True, msg)
 426                 if old_user_mgmt:
 427                     cursor.execute("SET PASSWORD FOR %s@%s = %s", (user, host, encrypted_password))
 428                     msg = "Password updated (old style)"
 429                 else:
 430                     try:
 431                         cursor.execute("ALTER USER %s@%s IDENTIFIED WITH mysql_native_password AS %s", (user, host, encrypted_password))
 432                         msg = "Password updated (new style)"
 433                     except (mysql_driver.Error) as e:
 434                         # https://stackoverflow.com/questions/51600000/authentication-string-of-root-user-on-mysql
 435                         # Replacing empty root password with new authentication mechanisms fails with error 1396
 436                         if e.args[0] == 1396:
 437                             cursor.execute(
 438                                 "UPDATE user SET plugin = %s, authentication_string = %s, Password = '' WHERE User = %s AND Host = %s",
 439                                 ('mysql_native_password', encrypted_password, user, host)
 440                             )
 441                             cursor.execute("FLUSH PRIVILEGES")
 442                             msg = "Password forced update"
 443                         else:
 444                             raise e
 445                 changed = True
 446 
 447         # Handle plugin authentication
 448         if plugin:
 449             cursor.execute("SELECT plugin, authentication_string FROM mysql.user "
 450                            "WHERE user = %s AND host = %s", (user, host))
 451             current_plugin = cursor.fetchone()
 452 
 453             update = False
 454 
 455             if current_plugin[0] != plugin:
 456                 update = True
 457 
 458             if plugin_hash_string and current_plugin[1] != plugin_hash_string:
 459                 update = True
 460 
 461             if plugin_auth_string and current_plugin[1] != plugin_auth_string:
 462                 # this case can cause more updates than expected,
 463                 # as plugin can hash auth_string in any way it wants
 464                 # and there's no way to figure it out for
 465                 # a check, so I prefer to update more often than never
 466                 update = True
 467 
 468             if update:
 469                 if plugin_hash_string:
 470                     cursor.execute("ALTER USER %s@%s IDENTIFIED WITH %s AS %s", (user, host, plugin, plugin_hash_string))
 471                 elif plugin_auth_string:
 472                     cursor.execute("ALTER USER %s@%s IDENTIFIED WITH %s BY %s", (user, host, plugin, plugin_auth_string))
 473                 else:
 474                     cursor.execute("ALTER USER %s@%s IDENTIFIED WITH %s", (user, host, plugin))
 475                 changed = True
 476 
 477         # Handle privileges
 478         if new_priv is not None:
 479             curr_priv = privileges_get(cursor, user, host)
 480 
 481             # If the user has privileges on a db.table that doesn't appear at all in
 482             # the new specification, then revoke all privileges on it.
 483             for db_table, priv in iteritems(curr_priv):
 484                 # If the user has the GRANT OPTION on a db.table, revoke it first.
 485                 if "GRANT" in priv:
 486                     grant_option = True
 487                 if db_table not in new_priv:
 488                     if user != "root" and "PROXY" not in priv and not append_privs:
 489                         msg = "Privileges updated"
 490                         if module.check_mode:
 491                             return (True, msg)
 492                         privileges_revoke(cursor, user, host, db_table, priv, grant_option)
 493                         changed = True
 494 
 495             # If the user doesn't currently have any privileges on a db.table, then
 496             # we can perform a straight grant operation.
 497             for db_table, priv in iteritems(new_priv):
 498                 if db_table not in curr_priv:
 499                     msg = "New privileges granted"
 500                     if module.check_mode:
 501                         return (True, msg)
 502                     privileges_grant(cursor, user, host, db_table, priv)
 503                     changed = True
 504 
 505             # If the db.table specification exists in both the user's current privileges
 506             # and in the new privileges, then we need to see if there's a difference.
 507             db_table_intersect = set(new_priv.keys()) & set(curr_priv.keys())
 508             for db_table in db_table_intersect:
 509                 priv_diff = set(new_priv[db_table]) ^ set(curr_priv[db_table])
 510                 if len(priv_diff) > 0:
 511                     msg = "Privileges updated"
 512                     if module.check_mode:
 513                         return (True, msg)
 514                     if not append_privs:
 515                         privileges_revoke(cursor, user, host, db_table, curr_priv[db_table], grant_option)
 516                     privileges_grant(cursor, user, host, db_table, new_priv[db_table])
 517                     changed = True
 518 
 519     return (changed, msg)

 548 def privileges_get(cursor, user, host):
 549     """ MySQL doesn't have a better method of getting privileges aside from the
 550     SHOW GRANTS query syntax, which requires us to then parse the returned string.
 551     Here's an example of the string that is returned from MySQL:
 552 
 553      GRANT USAGE ON *.* TO 'user'@'localhost' IDENTIFIED BY 'pass';
 554 
 555     This function makes the query and returns a dictionary containing the results.
 556     The dictionary format is the same as that returned by privileges_unpack() below.
 557     """
 558     output = {}
 559     cursor.execute("SHOW GRANTS FOR %s@%s", (user, host))
 560     grants = cursor.fetchall()
 561 
 562     def pick(x):
 563         if x == 'ALL PRIVILEGES':
 564             return 'ALL'
 565         else:
 566             return x
 567 
 568     for grant in grants:
 569         res = re.match("""GRANT (.+) ON (.+) TO (['`"]).*\\3@(['`"]).*\\4( IDENTIFIED BY PASSWORD (['`"]).+\\6)? ?(.*)""", grant[0])
 570         if res is None:
 571             raise InvalidPrivsError('unable to parse the MySQL grant string: %s' % grant[0])
 572         privileges = res.group(1).split(", ")
 573         privileges = [pick(x) for x in privileges]
 574         if "WITH GRANT OPTION" in res.group(7):
 575             privileges.append('GRANT')
 576         if "REQUIRE SSL" in res.group(7):
 577             privileges.append('REQUIRESSL')
 578         db = res.group(2)
 579         output[db] = privileges
 580     return output

 583 def privileges_unpack(priv, mode):
 584     """ Take a privileges string, typically passed as a parameter, and unserialize
 585     it into a dictionary, the same format as privileges_get() above. We have this
 586     custom format to avoid using YAML/JSON strings inside YAML playbooks. Example
 587     of a privileges string:
 588 
 589      mydb.*:INSERT,UPDATE/anotherdb.*:SELECT/yetanother.*:ALL
 590 
 591     The privilege USAGE stands for no privileges, so we add that in on *.* if it's
 592     not specified in the string, as MySQL will always provide this by default.
 593     """
 594     if mode == 'ANSI':
 595         quote = '"'
 596     else:
 597         quote = '`'
 598     output = {}
 599     privs = []
 600     for item in priv.strip().split('/'):
 601         pieces = item.strip().rsplit(':', 1)
 602         dbpriv = pieces[0].rsplit(".", 1)
 603 
 604         # Check for FUNCTION or PROCEDURE object types
 605         parts = dbpriv[0].split(" ", 1)
 606         object_type = ''
 607         if len(parts) > 1 and (parts[0] == 'FUNCTION' or parts[0] == 'PROCEDURE'):
 608             object_type = parts[0] + ' '
 609             dbpriv[0] = parts[1]
 610 
 611         # Do not escape if privilege is for database or table, i.e.
 612         # neither quote *. nor .*
 613         for i, side in enumerate(dbpriv):
 614             if side.strip('`') != '*':
 615                 dbpriv[i] = '%s%s%s' % (quote, side.strip('`'), quote)
 616         pieces[0] = object_type + '.'.join(dbpriv)
 617 
 618         if '(' in pieces[1]:
 619             output[pieces[0]] = re.split(r',\s*(?=[^)]*(?:\(|$))', pieces[1].upper())
 620             for i in output[pieces[0]]:
 621                 privs.append(re.sub(r'\s*\(.*\)', '', i))
 622         else:
 623             output[pieces[0]] = pieces[1].upper().split(',')
 624             privs = output[pieces[0]]
 625         new_privs = frozenset(privs)
 626         if not new_privs.issubset(VALID_PRIVS):
 627             raise InvalidPrivsError('Invalid privileges specified: %s' % new_privs.difference(VALID_PRIVS))
 628 
 629     if '*.*' not in output:
 630         output['*.*'] = ['USAGE']
 631 
 632     # if we are only specifying something like REQUIRESSL and/or GRANT (=WITH GRANT OPTION) in *.*
 633     # we still need to add USAGE as a privilege to avoid syntax errors
 634     if 'REQUIRESSL' in priv and not set(output['*.*']).difference(set(['GRANT', 'REQUIRESSL'])):
 635         output['*.*'].append('USAGE')
 636 
 637     return output

 640 def privileges_revoke(cursor, user, host, db_table, priv, grant_option):
 641     # Escape '%' since mysql db.execute() uses a format string
 642     db_table = db_table.replace('%', '%%')
 643     if grant_option:
 644         query = ["REVOKE GRANT OPTION ON %s" % db_table]
 645         query.append("FROM %s@%s")
 646         query = ' '.join(query)
 647         cursor.execute(query, (user, host))
 648     priv_string = ",".join([p for p in priv if p not in ('GRANT', 'REQUIRESSL')])
 649     query = ["REVOKE %s ON %s" % (priv_string, db_table)]
 650     query.append("FROM %s@%s")
 651     query = ' '.join(query)
 652     cursor.execute(query, (user, host))

 655 def privileges_grant(cursor, user, host, db_table, priv):
 656     # Escape '%' since mysql db.execute uses a format string and the
 657     # specification of db and table often use a % (SQL wildcard)
 658     db_table = db_table.replace('%', '%%')
 659     priv_string = ",".join([p for p in priv if p not in ('GRANT', 'REQUIRESSL')])
 660     query = ["GRANT %s ON %s" % (priv_string, db_table)]
 661     query.append("TO %s@%s")
 662     if 'REQUIRESSL' in priv:
 663         query.append("REQUIRE SSL")
 664     if 'GRANT' in priv:
 665         query.append("WITH GRANT OPTION")
 666     query = ' '.join(query)
 667     cursor.execute(query, (user, host))

 670 def convert_priv_dict_to_str(priv):
 671     """Converts privs dictionary to string of certain format.
 672 
 673     Args:
 674         priv (dict): Dict of privileges that needs to be converted to string.
 675 
 676     Returns:
 677         priv (str): String representation of input argument.
 678     """
 679     priv_list = ['%s:%s' % (key, val) for key, val in iteritems(priv)]
 680 
 681     return '/'.join(priv_list)
