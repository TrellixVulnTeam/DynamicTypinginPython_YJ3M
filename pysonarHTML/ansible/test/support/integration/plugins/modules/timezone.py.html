  90 class Timezone(object):
  91     """This is a generic Timezone manipulation class that is subclassed based on platform.
  92 
  93     A subclass may wish to override the following action methods:
  94         - get(key, phase)   ... get the value from the system at `phase`
  95         - set(key, value)   ... set the value to the current system
  96     """
  97 
  98     def __new__(cls, module):
  99         """Return the platform-specific subclass.
 100 
 101         It does not use load_platform_subclass() because it needs to judge based
 102         on whether the `timedatectl` command exists and is available.
 103 
 104         Args:
 105             module: The AnsibleModule.
 106         """
 107         if platform.system() == 'Linux':
 108             timedatectl = module.get_bin_path('timedatectl')
 109             if timedatectl is not None:
 110                 rc, stdout, stderr = module.run_command(timedatectl)
 111                 if rc == 0:
 112                     return super(Timezone, SystemdTimezone).__new__(SystemdTimezone)
 113                 else:
 114                     module.warn('timedatectl command was found but not usable: %s. using other method.' % stderr)
 115                     return super(Timezone, NosystemdTimezone).__new__(NosystemdTimezone)
 116             else:
 117                 return super(Timezone, NosystemdTimezone).__new__(NosystemdTimezone)
 118         elif re.match('^joyent_.*Z', platform.version()):
 119             # platform.system() returns SunOS, which is too broad. So look at the
 120             # platform version instead. However we have to ensure that we're not
 121             # running in the global zone where changing the timezone has no effect.
 122             zonename_cmd = module.get_bin_path('zonename')
 123             if zonename_cmd is not None:
 124                 (rc, stdout, _) = module.run_command(zonename_cmd)
 125                 if rc == 0 and stdout.strip() == 'global':
 126                     module.fail_json(msg='Adjusting timezone is not supported in Global Zone')
 127 
 128             return super(Timezone, SmartOSTimezone).__new__(SmartOSTimezone)
 129         elif re.match('^Darwin', platform.platform()):
 130             return super(Timezone, DarwinTimezone).__new__(DarwinTimezone)
 131         elif re.match('^(Free|Net|Open)BSD', platform.platform()):
 132             return super(Timezone, BSDTimezone).__new__(BSDTimezone)
 133         elif platform.system() == 'AIX':
 134             AIXoslevel = int(platform.version() + platform.release())
 135             if AIXoslevel >= 61:
 136                 return super(Timezone, AIXTimezone).__new__(AIXTimezone)
 137             else:
 138                 module.fail_json(msg='AIX os level must be >= 61 for timezone module (Target: %s).' % AIXoslevel)
 139         else:
 140             # Not supported yet
 141             return super(Timezone, Timezone).__new__(Timezone)

 143     def __init__(self, module):
 144         """Initialize of the class.
 145 
 146         Args:
 147             module: The AnsibleModule.
 148         """
 149         super(Timezone, self).__init__()
 150         self.msg = []
 151         # `self.value` holds the values for each params on each phases.
 152         # Initially there's only info of "planned" phase, but the
 153         # `self.check()` function will fill out it.
 154         self.value = dict()
 155         for key in module.argument_spec:
 156             value = module.params[key]
 157             if value is not None:
 158                 self.value[key] = dict(planned=value)
 159         self.module = module
 160 
 161     def abort(self, msg):
 162         """Abort the process with error message.
 163 
 164         This is just the wrapper of module.fail_json().
 165 
 166         Args:
 167             msg: The error message.
 168         """
 169         error_msg = ['Error message:', msg]
 170         if len(self.msg) > 0:
 171             error_msg.append('Other message(s):')
 172             error_msg.extend(self.msg)
 173         self.module.fail_json(msg='\n'.join(error_msg))
 174 
 175     def execute(self, *commands, **kwargs):
 176         """Execute the shell command.
 177 
 178         This is just the wrapper of module.run_command().
 179 
 180         Args:
 181             *commands: The command to execute.
 182                 It will be concatenated with single space.
 183             **kwargs:  Only 'log' key is checked.
 184                 If kwargs['log'] is true, record the command to self.msg.
 185 
 186         Returns:
 187             stdout: Standard output of the command.
 188         """
 189         command = ' '.join(commands)
 190         (rc, stdout, stderr) = self.module.run_command(command, check_rc=True)
 191         if kwargs.get('log', False):
 192             self.msg.append('executed `%s`' % command)
 193         return stdout
 194 
 195     def diff(self, phase1='before', phase2='after'):
 196         """Calculate the difference between given 2 phases.
 197 
 198         Args:
 199             phase1, phase2: The names of phase to compare.
 200 
 201         Returns:
 202             diff: The difference of value between phase1 and phase2.
 203                 This is in the format which can be used with the
 204                 `--diff` option of ansible-playbook.
 205         """
 206         diff = {phase1: {}, phase2: {}}
 207         for key, value in iteritems(self.value):
 208             diff[phase1][key] = value[phase1]
 209             diff[phase2][key] = value[phase2]
 210         return diff
 211 
 212     def check(self, phase):
 213         """Check the state in given phase and set it to `self.value`.
 214 
 215         Args:
 216             phase: The name of the phase to check.
 217 
 218         Returns:
 219             NO RETURN VALUE
 220         """
 221         if phase == 'planned':
 222             return
 223         for key, value in iteritems(self.value):
 224             value[phase] = self.get(key, phase)
 225 
 226     def change(self):
 227         """Make the changes effect based on `self.value`."""
 228         for key, value in iteritems(self.value):
 229             if value['before'] != value['planned']:
 230                 self.set(key, value['planned'])
 231

 233     # Platform specific methods (must be replaced by subclass).

 235     def get(self, key, phase):
 236         """Get the value for the key at the given phase.
 237 
 238         Called from self.check().
 239 
 240         Args:
 241             key:   The key to get the value
 242             phase: The phase to get the value
 243 
 244         Return:
 245             value: The value for the key at the given phase.
 246         """
 247         self.abort('get(key, phase) is not implemented on target platform')

 249     def set(self, key, value):
 250         """Set the value for the key (of course, for the phase 'after').
 251 
 252         Called from self.change().
 253 
 254         Args:
 255             key: Key to set the value
 256             value: Value to set
 257         """
 258         self.abort('set(key, value) is not implemented on target platform')

 260     def _verify_timezone(self):
 261         tz = self.value['name']['planned']
 262         tzfile = '/usr/share/zoneinfo/%s' % tz
 263         if not os.path.isfile(tzfile):
 264             self.abort('given timezone "%s" is not available' % tz)
 265         return tzfile

 268 class SystemdTimezone(Timezone):
 269     """This is a Timezone manipulation class for systemd-powered Linux.
 270 
 271     It uses the `timedatectl` command to check/set all arguments.
 272     """
 273 
 274     regexps = dict(
 275         hwclock=re.compile(r'^\s*RTC in local TZ\s*:\s*([^\s]+)', re.MULTILINE),
 276         name=re.compile(r'^\s*Time ?zone\s*:\s*([^\s]+)', re.MULTILINE)
 277     )
 278 
 279     subcmds = dict(
 280         hwclock='set-local-rtc',
 281         name='set-timezone'
 282     )
 283 
 284     def __init__(self, module):
 285         super(SystemdTimezone, self).__init__(module)
 286         self.timedatectl = module.get_bin_path('timedatectl', required=True)
 287         self.status = dict()
 288         # Validate given timezone
 289         if 'name' in self.value:
 290             self._verify_timezone()
 291 
 292     def _get_status(self, phase):
 293         if phase not in self.status:
 294             self.status[phase] = self.execute(self.timedatectl, 'status')
 295         return self.status[phase]
 296 
 297     def get(self, key, phase):
 298         status = self._get_status(phase)
 299         value = self.regexps[key].search(status).group(1)
 300         if key == 'hwclock':
 301             # For key='hwclock'; convert yes/no -> local/UTC
 302             if self.module.boolean(value):
 303                 value = 'local'
 304             else:
 305                 value = 'UTC'
 306         return value
 307 
 308     def set(self, key, value):
 309         # For key='hwclock'; convert UTC/local -> yes/no
 310         if key == 'hwclock':
 311             if value == 'local':
 312                 value = 'yes'
 313             else:
 314                 value = 'no'
 315         self.execute(self.timedatectl, self.subcmds[key], value, log=True)

 318 class NosystemdTimezone(Timezone):
 319     """This is a Timezone manipulation class for non systemd-powered Linux.
 320 
 321     For timezone setting, it edits the following file and reflect changes:
 322         - /etc/sysconfig/clock  ... RHEL/CentOS
 323         - /etc/timezone         ... Debian/Ubuntu
 324     For hwclock setting, it executes `hwclock --systohc` command with the
 325     '--utc' or '--localtime' option.
 326     """
 327 
 328     conf_files = dict(
 329         name=None,  # To be set in __init__
 330         hwclock=None,  # To be set in __init__
 331         adjtime='/etc/adjtime'
 332     )
 333 
 334     # It's fine if all tree config files don't exist
 335     allow_no_file = dict(
 336         name=True,
 337         hwclock=True,
 338         adjtime=True
 339     )
 340 
 341     regexps = dict(
 342         name=None,  # To be set in __init__
 343         hwclock=re.compile(r'^UTC\s*=\s*([^\s]+)', re.MULTILINE),
 344         adjtime=re.compile(r'^(UTC|LOCAL)$', re.MULTILINE)
 345     )
 346 
 347     dist_regexps = dict(
 348         SuSE=re.compile(r'^TIMEZONE\s*=\s*"?([^"\s]+)"?', re.MULTILINE),
 349         redhat=re.compile(r'^ZONE\s*=\s*"?([^"\s]+)"?', re.MULTILINE)
 350     )
 351 
 352     dist_tzline_format = dict(
 353         SuSE='TIMEZONE="%s"\n',
 354         redhat='ZONE="%s"\n'
 355     )
 356 
 357     def __init__(self, module):
 358         super(NosystemdTimezone, self).__init__(module)
 359         # Validate given timezone
 360         if 'name' in self.value:
 361             tzfile = self._verify_timezone()
 362             # `--remove-destination` is needed if /etc/localtime is a symlink so
 363             # that it overwrites it instead of following it.
 364             self.update_timezone = ['%s --remove-destination %s /etc/localtime' % (self.module.get_bin_path('cp', required=True), tzfile)]
 365         self.update_hwclock = self.module.get_bin_path('hwclock', required=True)
 366         # Distribution-specific configurations
 367         if self.module.get_bin_path('dpkg-reconfigure') is not None:
 368             # Debian/Ubuntu
 369             if 'name' in self.value:
 370                 self.update_timezone = ['%s -sf %s /etc/localtime' % (self.module.get_bin_path('ln', required=True), tzfile),
 371                                         '%s --frontend noninteractive tzdata' % self.module.get_bin_path('dpkg-reconfigure', required=True)]
 372             self.conf_files['name'] = '/etc/timezone'
 373             self.conf_files['hwclock'] = '/etc/default/rcS'
 374             self.regexps['name'] = re.compile(r'^([^\s]+)', re.MULTILINE)
 375             self.tzline_format = '%s\n'
 376         else:
 377             # RHEL/CentOS/SUSE
 378             if self.module.get_bin_path('tzdata-update') is not None:
 379                 # tzdata-update cannot update the timezone if /etc/localtime is
 380                 # a symlink so we have to use cp to update the time zone which
 381                 # was set above.
 382                 if not os.path.islink('/etc/localtime'):
 383                     self.update_timezone = [self.module.get_bin_path('tzdata-update', required=True)]
 384                 # else:
 385                 #   self.update_timezone       = 'cp --remove-destination ...' <- configured above
 386             self.conf_files['name'] = '/etc/sysconfig/clock'
 387             self.conf_files['hwclock'] = '/etc/sysconfig/clock'
 388             try:
 389                 f = open(self.conf_files['name'], 'r')
 390             except IOError as err:
 391                 if self._allow_ioerror(err, 'name'):
 392                     # If the config file doesn't exist detect the distribution and set regexps.
 393                     distribution = get_distribution()
 394                     if distribution == 'SuSE':
 395                         # For SUSE
 396                         self.regexps['name'] = self.dist_regexps['SuSE']
 397                         self.tzline_format = self.dist_tzline_format['SuSE']
 398                     else:
 399                         # For RHEL/CentOS
 400                         self.regexps['name'] = self.dist_regexps['redhat']
 401                         self.tzline_format = self.dist_tzline_format['redhat']
 402                 else:
 403                     self.abort('could not read configuration file "%s"' % self.conf_files['name'])
 404             else:
 405                 # The key for timezone might be `ZONE` or `TIMEZONE`
 406                 # (the former is used in RHEL/CentOS and the latter is used in SUSE linux).
 407                 # So check the content of /etc/sysconfig/clock and decide which key to use.
 408                 sysconfig_clock = f.read()
 409                 f.close()
 410                 if re.search(r'^TIMEZONE\s*=', sysconfig_clock, re.MULTILINE):
 411                     # For SUSE
 412                     self.regexps['name'] = self.dist_regexps['SuSE']
 413                     self.tzline_format = self.dist_tzline_format['SuSE']
 414                 else:
 415                     # For RHEL/CentOS
 416                     self.regexps['name'] = self.dist_regexps['redhat']
 417                     self.tzline_format = self.dist_tzline_format['redhat']
 418 
 419     def _allow_ioerror(self, err, key):
 420         # In some cases, even if the target file does not exist,
 421         # simply creating it may solve the problem.
 422         # In such cases, we should continue the configuration rather than aborting.
 423         if err.errno != errno.ENOENT:
 424             # If the error is not ENOENT ("No such file or directory"),
 425             # (e.g., permission error, etc), we should abort.
 426             return False
 427         return self.allow_no_file.get(key, False)

 429     def _edit_file(self, filename, regexp, value, key):
 430         """Replace the first matched line with given `value`.
 431 
 432         If `regexp` matched more than once, other than the first line will be deleted.
 433 
 434         Args:
 435             filename: The name of the file to edit.
 436             regexp:   The regular expression to search with.
 437             value:    The line which will be inserted.
 438             key:      For what key the file is being editted.
 439         """
 440         # Read the file
 441         try:
 442             file = open(filename, 'r')
 443         except IOError as err:
 444             if self._allow_ioerror(err, key):
 445                 lines = []
 446             else:
 447                 self.abort('tried to configure %s using a file "%s", but could not read it' % (key, filename))
 448         else:
 449             lines = file.readlines()
 450             file.close()
 451         # Find the all matched lines
 452         matched_indices = []
 453         for i, line in enumerate(lines):
 454             if regexp.search(line):
 455                 matched_indices.append(i)
 456         if len(matched_indices) > 0:
 457             insert_line = matched_indices[0]
 458         else:
 459             insert_line = 0
 460         # Remove all matched lines
 461         for i in matched_indices[::-1]:
 462             del lines[i]
 463         # ...and insert the value
 464         lines.insert(insert_line, value)
 465         # Write the changes
 466         try:
 467             file = open(filename, 'w')
 468         except IOError:
 469             self.abort('tried to configure %s using a file "%s", but could not write to it' % (key, filename))
 470         else:
 471             file.writelines(lines)
 472             file.close()
 473         self.msg.append('Added 1 line and deleted %s line(s) on %s' % (len(matched_indices), filename))
 474 
 475     def _get_value_from_config(self, key, phase):
 476         filename = self.conf_files[key]
 477         try:
 478             file = open(filename, mode='r')
 479         except IOError as err:
 480             if self._allow_ioerror(err, key):
 481                 if key == 'hwclock':
 482                     return 'n/a'
 483                 elif key == 'adjtime':
 484                     return 'UTC'
 485                 elif key == 'name':
 486                     return 'n/a'
 487             else:
 488                 self.abort('tried to configure %s using a file "%s", but could not read it' % (key, filename))
 489         else:
 490             status = file.read()
 491             file.close()
 492             try:
 493                 value = self.regexps[key].search(status).group(1)
 494             except AttributeError:
 495                 if key == 'hwclock':
 496                     # If we cannot find UTC in the config that's fine.
 497                     return 'n/a'
 498                 elif key == 'adjtime':
 499                     # If we cannot find UTC/LOCAL in /etc/cannot that means UTC
 500                     # will be used by default.
 501                     return 'UTC'
 502                 elif key == 'name':
 503                     if phase == 'before':
 504                         # In 'before' phase UTC/LOCAL doesn't need to be set in
 505                         # the timezone config file, so we ignore this error.
 506                         return 'n/a'
 507                     else:
 508                         self.abort('tried to configure %s using a file "%s", but could not find a valid value in it' % (key, filename))
 509             else:
 510                 if key == 'hwclock':
 511                     # convert yes/no -> UTC/local
 512                     if self.module.boolean(value):
 513                         value = 'UTC'
 514                     else:
 515                         value = 'local'
 516                 elif key == 'adjtime':
 517                     # convert LOCAL -> local
 518                     if value != 'UTC':
 519                         value = value.lower()
 520         return value
 521 
 522     def get(self, key, phase):
 523         planned = self.value[key]['planned']
 524         if key == 'hwclock':
 525             value = self._get_value_from_config(key, phase)
 526             if value == planned:
 527                 # If the value in the config file is the same as the 'planned'
 528                 # value, we need to check /etc/adjtime.
 529                 value = self._get_value_from_config('adjtime', phase)
 530         elif key == 'name':
 531             value = self._get_value_from_config(key, phase)
 532             if value == planned:
 533                 # If the planned values is the same as the one in the config file
 534                 # we need to check if /etc/localtime is also set to the 'planned' zone.
 535                 if os.path.islink('/etc/localtime'):
 536                     # If /etc/localtime is a symlink and is not set to the TZ we 'planned'
 537                     # to set, we need to return the TZ which the symlink points to.
 538                     if os.path.exists('/etc/localtime'):
 539                         # We use readlink() because on some distros zone files are symlinks
 540                         # to other zone files, so it's hard to get which TZ is actually set
 541                         # if we follow the symlink.
 542                         path = os.readlink('/etc/localtime')
 543                         linktz = re.search(r'/usr/share/zoneinfo/(.*)', path, re.MULTILINE)
 544                         if linktz:
 545                             valuelink = linktz.group(1)
 546                             if valuelink != planned:
 547                                 value = valuelink
 548                         else:
 549                             # Set current TZ to 'n/a' if the symlink points to a path
 550                             # which isn't a zone file.
 551                             value = 'n/a'
 552                     else:
 553                         # Set current TZ to 'n/a' if the symlink to the zone file is broken.
 554                         value = 'n/a'
 555                 else:
 556                     # If /etc/localtime is not a symlink best we can do is compare it with
 557                     # the 'planned' zone info file and return 'n/a' if they are different.
 558                     try:
 559                         if not filecmp.cmp('/etc/localtime', '/usr/share/zoneinfo/' + planned):
 560                             return 'n/a'
 561                     except Exception:
 562                         return 'n/a'
 563         else:
 564             self.abort('unknown parameter "%s"' % key)
 565         return value

 567     def set_timezone(self, value):
 568         self._edit_file(filename=self.conf_files['name'],
 569                         regexp=self.regexps['name'],
 570                         value=self.tzline_format % value,
 571                         key='name')
 572         for cmd in self.update_timezone:
 573             self.execute(cmd)
 574 
 575     def set_hwclock(self, value):
 576         if value == 'local':
 577             option = '--localtime'
 578             utc = 'no'
 579         else:
 580             option = '--utc'
 581             utc = 'yes'
 582         if self.conf_files['hwclock'] is not None:
 583             self._edit_file(filename=self.conf_files['hwclock'],
 584                             regexp=self.regexps['hwclock'],
 585                             value='UTC=%s\n' % utc,
 586                             key='hwclock')

 592         elif key == 'hwclock':

 624                         return m.groups()[0]

 651 class DarwinTimezone(Timezone):
 652     """This is the timezone implementation for Darwin which, unlike other *BSD
 653     implementations, uses the `systemsetup` command on Darwin to check/set
 654     the timezone.
 655     """
 656 
 657     regexps = dict(
 658         name=re.compile(r'^\s*Time ?Zone\s*:\s*([^\s]+)', re.MULTILINE)
 659     )
 660 
 661     def __init__(self, module):
 662         super(DarwinTimezone, self).__init__(module)
 663         self.systemsetup = module.get_bin_path('systemsetup', required=True)
 664         self.status = dict()
 665         # Validate given timezone
 666         if 'name' in self.value:
 667             self._verify_timezone()
 668 
 669     def _get_current_timezone(self, phase):
 670         """Lookup the current timezone via `systemsetup -gettimezone`."""
 671         if phase not in self.status:
 672             self.status[phase] = self.execute(self.systemsetup, '-gettimezone')
 673         return self.status[phase]
 674 
 675     def _verify_timezone(self):
 676         tz = self.value['name']['planned']
 677         # Lookup the list of supported timezones via `systemsetup -listtimezones`.
 678         # Note: Skip the first line that contains the label 'Time Zones:'
 679         out = self.execute(self.systemsetup, '-listtimezones').splitlines()[1:]
 680         tz_list = list(map(lambda x: x.strip(), out))
 681         if tz not in tz_list:
 682             self.abort('given timezone "%s" is not available' % tz)
 683         return tz
 684

 686         if key == 'name':
 687             status = self._get_current_timezone(phase)
 688             value = self.regexps[key].search(status).group(1)
 689             return value
 690         else:
 691             self.module.fail_json(msg='%s is not a supported option on target platform' % key)
 692 
 693     def set(self, key, value):
 694         if key == 'name':
 695             self.execute(self.systemsetup, '-settimezone', value, log=True)
 696         else:
 697             self.module.fail_json(msg='%s is not a supported option on target platform' % key)

 700 class BSDTimezone(Timezone):
 701     """This is the timezone implementation for *BSD which works simply through
 702     updating the `/etc/localtime` symlink to point to a valid timezone name under
 703     `/usr/share/zoneinfo`.
 704     """
 705 
 706     def __init__(self, module):
 707         super(BSDTimezone, self).__init__(module)
 708 
 709     def __get_timezone(self):
 710         zoneinfo_dir = '/usr/share/zoneinfo/'
 711         localtime_file = '/etc/localtime'
 712 
 713         # Strategy 1:
 714         #   If /etc/localtime does not exist, assum the timezone is UTC.
 715         if not os.path.exists(localtime_file):
 716             self.module.warn('Could not read /etc/localtime. Assuming UTC.')
 717             return 'UTC'
 718 
 719         # Strategy 2:
 720         #   Follow symlink of /etc/localtime
 721         zoneinfo_file = localtime_file
 722         while not zoneinfo_file.startswith(zoneinfo_dir):
 723             try:
 724                 zoneinfo_file = os.readlink(localtime_file)
 725             except OSError:
 726                 # OSError means "end of symlink chain" or broken link.
 727                 break
 728         else:
 729             return zoneinfo_file.replace(zoneinfo_dir, '')
 730 
 731         # Strategy 3:
 732         #   (If /etc/localtime is not symlinked)
 733         #   Check all files in /usr/share/zoneinfo and return first non-link match.
 734         for dname, _, fnames in sorted(os.walk(zoneinfo_dir)):
 735             for fname in sorted(fnames):
 736                 zoneinfo_file = os.path.join(dname, fname)
 737                 if not os.path.islink(zoneinfo_file) and filecmp.cmp(zoneinfo_file, localtime_file):
 738                     return zoneinfo_file.replace(zoneinfo_dir, '')
 739 
 740         # Strategy 4:
 741         #   As a fall-back, return 'UTC' as default assumption.
 742         self.module.warn('Could not identify timezone name from /etc/localtime. Assuming UTC.')
 743         return 'UTC'
 744 
 745     def get(self, key, phase):
 746         """Lookup the current timezone by resolving `/etc/localtime`."""
 747         if key == 'name':
 748             return self.__get_timezone()
 749         else:
 750             self.module.fail_json(msg='%s is not a supported option on target platform' % key)
 751 
 752     def set(self, key, value):
 753         if key == 'name':
 754             # First determine if the requested timezone is valid by looking in
 755             # the zoneinfo directory.
 756             zonefile = '/usr/share/zoneinfo/' + value
 757             try:
 758                 if not os.path.isfile(zonefile):
 759                     self.module.fail_json(msg='%s is not a recognized timezone' % value)
 760             except Exception:
 761                 self.module.fail_json(msg='Failed to stat %s' % zonefile)
 762 
 763             # Now (somewhat) atomically update the symlink by creating a new
 764             # symlink and move it into place. Otherwise we have to remove the
 765             # original symlink and create the new symlink, however that would
 766             # create a race condition in case another process tries to read
 767             # /etc/localtime between removal and creation.
 768             suffix = "".join([random.choice(string.ascii_letters + string.digits) for x in range(0, 10)])
 769             new_localtime = '/etc/localtime.' + suffix
 770 
 771             try:
 772                 os.symlink(zonefile, new_localtime)
 773                 os.rename(new_localtime, '/etc/localtime')
 774             except Exception:
 775                 os.remove(new_localtime)
 776                 self.module.fail_json(msg='Could not update /etc/localtime')
 777         else:
 778             self.module.fail_json(msg='%s is not a supported option on target platform' % key)

 781 class AIXTimezone(Timezone):
 782     """This is a Timezone manipulation class for AIX instances.
 783 
 784     It uses the C(chtz) utility to set the timezone, and
 785     inspects C(/etc/environment) to determine the current timezone.
 786 
 787     While AIX time zones can be set using two formats (POSIX and
 788     Olson) the prefered method is Olson.
 789     See the following article for more information:
 790     https://developer.ibm.com/articles/au-aix-posix/
 791 
 792     NB: AIX needs to be rebooted in order for the change to be
 793     activated.
 794     """
 795 
 796     def __init__(self, module):
 797         super(AIXTimezone, self).__init__(module)
 798         self.settimezone = self.module.get_bin_path('chtz', required=True)
 799 
 800     def __get_timezone(self):
 801         """ Return the current value of TZ= in /etc/environment """
 802         try:
 803             f = open('/etc/environment', 'r')
 804             etcenvironment = f.read()
 805             f.close()
 806         except Exception:
 807             self.module.fail_json(msg='Issue reading contents of /etc/environment')
 808 
 809         match = re.search(r'^TZ=(.*)$', etcenvironment, re.MULTILINE)
 810         if match:
 811             return match.group(1)
 812         else:
 813             return None
 814 
 815     def get(self, key, phase):
 816         """Lookup the current timezone name in `/etc/environment`. If anything else
 817         is requested, or if the TZ field is not set we fail.
 818         """
 819         if key == 'name':
 820             return self.__get_timezone()
 821         else:
 822             self.module.fail_json(msg='%s is not a supported option on target platform' % key)
 823 
 824     def set(self, key, value):
 825         """Set the requested timezone through chtz, an invalid timezone name
 826         will be rejected and we have no further input validation to perform.
 827         """
 828         if key == 'name':
 829             # chtz seems to always return 0 on AIX 7.2, even for invalid timezone values.
 830             # It will only return non-zero if the chtz command itself fails, it does not check for
 831             #  valid timezones. We need to perform a basic check to confirm that the timezone
 832             #  definition exists in /usr/share/lib/zoneinfo
 833             # This does mean that we can only support Olson for now. The below commented out regex
 834             #  detects Olson date formats, so in the future we could detect Posix or Olson and
 835             #  act accordingly.
 836 
 837             # regex_olson = re.compile('^([a-z0-9_\-\+]+\/?)+$', re.IGNORECASE)
 838             # if not regex_olson.match(value):
 839             #     msg = 'Supplied timezone (%s) does not appear to a be valid Olson string' % value
 840             #     self.module.fail_json(msg=msg)
 841 
 842             # First determine if the requested timezone is valid by looking in the zoneinfo
 843             #  directory.
 844             zonefile = '/usr/share/lib/zoneinfo/' + value
 845             try:
 846                 if not os.path.isfile(zonefile):
 847                     self.module.fail_json(msg='%s is not a recognized timezone.' % value)
 848             except Exception:
 849                 self.module.fail_json(msg='Failed to check %s.' % zonefile)
 850 
 851             # Now set the TZ using chtz
 852             cmd = 'chtz %s' % value
 853             (rc, stdout, stderr) = self.module.run_command(cmd)
 854 
 855             if rc != 0:
 856                 self.module.fail_json(msg=stderr)
 857 
 858             # The best condition check we can do is to check the value of TZ after making the
 859             #  change.
 860             TZ = self.__get_timezone()
 861             if TZ != value:
 862                 msg = 'TZ value does not match post-change (Actual: %s, Expected: %s).' % (TZ, value)
 863                 self.module.fail_json(msg=msg)
 864 
 865         else:
 866             self.module.fail_json(msg='%s is not a supported option on target platform' % key)

 869 def main():
 870     # Construct 'module' and 'tz'
 871     module = AnsibleModule(
 872         argument_spec=dict(
 873             hwclock=dict(type='str', choices=['local', 'UTC'], aliases=['rtc']),
 874             name=dict(type='str'),
 875         ),
 876         required_one_of=[
 877             ['hwclock', 'name']
 878         ],
 879         supports_check_mode=True,
 880     )
 881     tz = Timezone(module)
 882 
 883     # Check the current state
 884     tz.check(phase='before')
 885     if module.check_mode:
 886         diff = tz.diff('before', 'planned')
 887         # In check mode, 'planned' state is treated as 'after' state
 888         diff['after'] = diff.pop('planned')
 889     else:
 890         # Make change
 891         tz.change()
 892         # Check the current state
 893         tz.check(phase='after')
 894         # Examine if the current state matches planned state
 895         (after, planned) = tz.diff('after', 'planned').values()
 896         if after != planned:
 897             tz.abort('still not desired state, though changes have made - '
 898                      'planned: %s, after: %s' % (str(planned), str(after)))
 899         diff = tz.diff('before', 'after')
 900 
 901     changed = (diff['before'] != diff['after'])
 902     if len(tz.msg) > 0:
 903         module.exit_json(changed=changed, diff=diff, msg='\n'.join(tz.msg))
 904     else:
 905         module.exit_json(changed=changed, diff=diff)
 906 
 907 
 908 if __name__ == '__main__':
 909     main()
