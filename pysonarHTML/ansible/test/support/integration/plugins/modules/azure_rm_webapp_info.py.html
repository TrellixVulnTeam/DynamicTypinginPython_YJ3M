 265     def exec_module(self, **kwargs):
 266         is_old_facts = self.module._name == 'azure_rm_webapp_facts'
 267         if is_old_facts:
 268             self.module.deprecate("The 'azure_rm_webapp_facts' module has been renamed to 'azure_rm_webapp_info'",
 269                                   version='2.13', collection_name='ansible.builtin')
 270 
 271         for key in self.module_arg_spec:
 272             setattr(self, key, kwargs[key])
 273 
 274         if self.name:
 275             self.results['webapps'] = self.list_by_name()
 276         elif self.resource_group:
 277             self.results['webapps'] = self.list_by_resource_group()
 278         else:
 279             self.results['webapps'] = self.list_all()
 280 
 281         return self.results

 283     def list_by_name(self):
 284         self.log('Get web app {0}'.format(self.name))
 285         item = None
 286         result = []
 287 
 288         try:
 289             item = self.web_client.web_apps.get(self.resource_group, self.name)
 290         except CloudError:
 291             pass
 292 
 293         if item and self.has_tags(item.tags, self.tags):
 294             curated_result = self.get_curated_webapp(self.resource_group, self.name, item)
 295             result = [curated_result]
 296 
 297         return result
 298 
 299     def list_by_resource_group(self):
 300         self.log('List web apps in resource groups {0}'.format(self.resource_group))
 301         try:
 302             response = list(self.web_client.web_apps.list_by_resource_group(self.resource_group))
 303         except CloudError as exc:
 304             request_id = exc.request_id if exc.request_id else ''
 305             self.fail("Error listing web apps in resource groups {0}, request id: {1} - {2}".format(self.resource_group, request_id, str(exc)))
 306 
 307         results = []
 308         for item in response:
 309             if self.has_tags(item.tags, self.tags):
 310                 curated_output = self.get_curated_webapp(self.resource_group, item.name, item)
 311                 results.append(curated_output)
 312         return results

 314     def list_all(self):

 316         try:
 317             response = list(self.web_client.web_apps.list())

 319             request_id = exc.request_id if exc.request_id else ''
 320             self.fail("Error listing web apps, request id {0} - {1}".format(request_id, str(exc)))
 321 
 322         results = []
 323         for item in response:
 324             if self.has_tags(item.tags, self.tags):
 325                 curated_output = self.get_curated_webapp(item.resource_group, item.name, item)
 326                 results.append(curated_output)
 327         return results

 355     def get_publish_credentials(self, resource_group, name):
 356         self.log('Get web app {0} publish credentials'.format(name))
 357         try:
 358             poller = self.web_client.web_apps.list_publishing_credentials(resource_group, name)
 359             if isinstance(poller, LROPoller):
 360                 response = self.get_poller_result(poller)
 361         except CloudError as ex:
 362             request_id = ex.request_id if ex.request_id else ''
 363             self.fail('Error getting web app {0} publishing credentials - {1}'.format(request_id, str(ex)))
 364         return response

 366     def get_webapp_ftp_publish_url(self, resource_group, name):
 367         import xmltodict
 368 
 369         self.log('Get web app {0} app publish profile'.format(name))
 370 
 371         url = None
 372         try:
 373             content = self.web_client.web_apps.list_publishing_profile_xml_with_secrets(resource_group_name=resource_group, name=name)
 374             if not content:
 375                 return url
 376 
 377             full_xml = ''
 378             for f in content:
 379                 full_xml += f.decode()
 380             profiles = xmltodict.parse(full_xml, xml_attribs=True)['publishData']['publishProfile']
 381 
 382             if not profiles:
 383                 return url
 384 
 385             for profile in profiles:
 386                 if profile['@publishMethod'] == 'FTP':
 387                     url = profile['@publishUrl']
 388 
 389         except CloudError as ex:
 390             self.fail('Error getting web app {0} app settings'.format(name))
 391 
 392         return url

 394     def get_curated_webapp(self, resource_group, name, webapp):
 395         pip = self.serialize_obj(webapp, AZURE_OBJECT_CLASS)
 396 
 397         try:
 398             site_config = self.list_webapp_configuration(resource_group, name)
 399             app_settings = self.list_webapp_appsettings(resource_group, name)
 400             publish_cred = self.get_publish_credentials(resource_group, name)
 401             ftp_publish_url = self.get_webapp_ftp_publish_url(resource_group, name)
 402         except CloudError as ex:
 403             pass
 404         return self.construct_curated_webapp(webapp=pip,
 405                                              configuration=site_config,
 406                                              app_settings=app_settings,
 407                                              deployment_slot=None,
 408                                              ftp_publish_url=ftp_publish_url,
 409                                              publish_credentials=publish_cred)

 411     def construct_curated_webapp(self,
 412                                  webapp,
 413                                  configuration=None,
 414                                  app_settings=None,
 415                                  deployment_slot=None,
 416                                  ftp_publish_url=None,
 417                                  publish_credentials=None):
 418         curated_output = dict()
 419         curated_output['id'] = webapp['id']
 420         curated_output['name'] = webapp['name']
 421         curated_output['resource_group'] = webapp['properties']['resourceGroup']
 422         curated_output['location'] = webapp['location']
 423         curated_output['plan'] = webapp['properties']['serverFarmId']
 424         curated_output['tags'] = webapp.get('tags', None)
 425 
 426         # important properties from output. not match input arguments.
 427         curated_output['app_state'] = webapp['properties']['state']
 428         curated_output['availability_state'] = webapp['properties']['availabilityState']
 429         curated_output['default_host_name'] = webapp['properties']['defaultHostName']
 430         curated_output['host_names'] = webapp['properties']['hostNames']
 431         curated_output['enabled'] = webapp['properties']['enabled']
 432         curated_output['enabled_host_names'] = webapp['properties']['enabledHostNames']
 433         curated_output['host_name_ssl_states'] = webapp['properties']['hostNameSslStates']
 434         curated_output['outbound_ip_addresses'] = webapp['properties']['outboundIpAddresses']
 435 
 436         # curated site_config
 437         if configuration:
 438             curated_output['frameworks'] = []
 439             for fx_name in self.framework_names:
 440                 fx_version = configuration.get(fx_name + '_version', None)
 441                 if fx_version:
 442                     fx = {
 443                         'name': fx_name,
 444                         'version': fx_version
 445                     }
 446                     # java container setting
 447                     if fx_name == 'java':
 448                         if configuration['java_container'] and configuration['java_container_version']:
 449                             settings = {
 450                                 'java_container': configuration['java_container'].lower(),
 451                                 'java_container_version': configuration['java_container_version']
 452                             }
 453                             fx['settings'] = settings
 454 
 455                     curated_output['frameworks'].append(fx)
 456 
 457             # linux_fx_version
 458             if configuration.get('linux_fx_version', None):
 459                 tmp = configuration.get('linux_fx_version').split("|")
 460                 if len(tmp) == 2:
 461                     curated_output['frameworks'].append({'name': tmp[0].lower(), 'version': tmp[1]})
 462 
 463         # curated app_settings
 464         if app_settings and app_settings.get('properties', None):
 465             curated_output['app_settings'] = dict()
 466             for item in app_settings['properties']:
 467                 curated_output['app_settings'][item] = app_settings['properties'][item]
 468 
 469         # curated deploymenet_slot
 470         if deployment_slot:
 471             curated_output['deployment_slot'] = deployment_slot
 472 
 473         # ftp_publish_url
 474         if ftp_publish_url:
 475             curated_output['ftp_publish_url'] = ftp_publish_url
 476 
 477         # curated publish credentials
 478         if publish_credentials and self.return_publish_profile:
 479             curated_output['publishing_username'] = publish_credentials.publishing_user_name
 480             curated_output['publishing_password'] = publish_credentials.publishing_password
 481         return curated_output
