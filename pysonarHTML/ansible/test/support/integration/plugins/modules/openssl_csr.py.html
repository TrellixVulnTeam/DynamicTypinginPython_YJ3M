 493     def __init__(self, module):
 494         super(CertificateSigningRequestBase, self).__init__(
 495             module.params['path'],
 496             module.params['state'],
 497             module.params['force'],
 498             module.check_mode
 499         )
 500         self.digest = module.params['digest']
 501         self.privatekey_path = module.params['privatekey_path']
 502         self.privatekey_content = module.params['privatekey_content']
 503         if self.privatekey_content is not None:
 504             self.privatekey_content = self.privatekey_content.encode('utf-8')
 505         self.privatekey_passphrase = module.params['privatekey_passphrase']
 506         self.version = module.params['version']
 507         self.subjectAltName = module.params['subject_alt_name']
 508         self.subjectAltName_critical = module.params['subject_alt_name_critical']
 509         self.keyUsage = module.params['key_usage']
 510         self.keyUsage_critical = module.params['key_usage_critical']
 511         self.extendedKeyUsage = module.params['extended_key_usage']
 512         self.extendedKeyUsage_critical = module.params['extended_key_usage_critical']
 513         self.basicConstraints = module.params['basic_constraints']
 514         self.basicConstraints_critical = module.params['basic_constraints_critical']
 515         self.ocspMustStaple = module.params['ocsp_must_staple']
 516         self.ocspMustStaple_critical = module.params['ocsp_must_staple_critical']
 517         self.create_subject_key_identifier = module.params['create_subject_key_identifier']
 518         self.subject_key_identifier = module.params['subject_key_identifier']
 519         self.authority_key_identifier = module.params['authority_key_identifier']
 520         self.authority_cert_issuer = module.params['authority_cert_issuer']
 521         self.authority_cert_serial_number = module.params['authority_cert_serial_number']
 522         self.request = None
 523         self.privatekey = None
 524         self.csr_bytes = None
 525         self.return_content = module.params['return_content']
 526 
 527         if self.create_subject_key_identifier and self.subject_key_identifier is not None:
 528             module.fail_json(msg='subject_key_identifier cannot be specified if create_subject_key_identifier is true')
 529 
 530         self.backup = module.params['backup']
 531         self.backup_file = None
 532 
 533         self.subject = [
 534             ('C', module.params['country_name']),
 535             ('ST', module.params['state_or_province_name']),
 536             ('L', module.params['locality_name']),
 537             ('O', module.params['organization_name']),
 538             ('OU', module.params['organizational_unit_name']),
 539             ('CN', module.params['common_name']),
 540             ('emailAddress', module.params['email_address']),
 541         ]
 542 
 543         if module.params['subject']:
 544             self.subject = self.subject + crypto_utils.parse_name_field(module.params['subject'])
 545         self.subject = [(entry[0], entry[1]) for entry in self.subject if entry[1]]
 546 
 547         if not self.subjectAltName and module.params['use_common_name_for_san']:
 548             for sub in self.subject:
 549                 if sub[0] in ('commonName', 'CN'):
 550                     self.subjectAltName = ['DNS:%s' % sub[1]]
 551                     break
 552 
 553         if self.subject_key_identifier is not None:
 554             try:
 555                 self.subject_key_identifier = binascii.unhexlify(self.subject_key_identifier.replace(':', ''))
 556             except Exception as e:
 557                 raise CertificateSigningRequestError('Cannot parse subject_key_identifier: {0}'.format(e))
 558 
 559         if self.authority_key_identifier is not None:
 560             try:
 561                 self.authority_key_identifier = binascii.unhexlify(self.authority_key_identifier.replace(':', ''))
 562             except Exception as e:
 563                 raise CertificateSigningRequestError('Cannot parse authority_key_identifier: {0}'.format(e))
 564 
 565     @abc.abstractmethod
 566     def _generate_csr(self):
 567         pass
 568 
 569     def generate(self, module):
 570         '''Generate the certificate signing request.'''
 571         if not self.check(module, perms_required=False) or self.force:
 572             result = self._generate_csr()
 573             if self.backup:
 574                 self.backup_file = module.backup_local(self.path)
 575             if self.return_content:
 576                 self.csr_bytes = result
 577             crypto_utils.write_file(module, result)
 578             self.changed = True
 579 
 580         file_args = module.load_file_common_arguments(module.params)
 581         if module.set_fs_attributes_if_different(file_args, False):
 582             self.changed = True
 583 
 584     @abc.abstractmethod
 585     def _load_private_key(self):
 586         pass
 587 
 588     @abc.abstractmethod
 589     def _check_csr(self):
 590         pass
 591 
 592     def check(self, module, perms_required=True):
 593         """Ensure the resource is in its desired state."""
 594         state_and_perms = super(CertificateSigningRequestBase, self).check(module, perms_required)
 595 
 596         self._load_private_key()
 597 
 598         if not state_and_perms:
 599             return False
 600 
 601         return self._check_csr()
 602 
 603     def remove(self, module):
 604         if self.backup:
 605             self.backup_file = module.backup_local(self.path)
 606         super(CertificateSigningRequestBase, self).remove(module)
 607 
 608     def dump(self):
 609         '''Serialize the object into a dictionary.'''
 610 
 611         result = {
 612             'privatekey': self.privatekey_path,
 613             'filename': self.path,
 614             'subject': self.subject,
 615             'subjectAltName': self.subjectAltName,
 616             'keyUsage': self.keyUsage,
 617             'extendedKeyUsage': self.extendedKeyUsage,
 618             'basicConstraints': self.basicConstraints,
 619             'ocspMustStaple': self.ocspMustStaple,
 620             'changed': self.changed
 621         }
 622         if self.backup_file:
 623             result['backup_file'] = self.backup_file
 624         if self.return_content:
 625             if self.csr_bytes is None:
 626                 self.csr_bytes = crypto_utils.load_file_if_exists(self.path, ignore_errors=True)
 627             result['csr'] = self.csr_bytes.decode('utf-8') if self.csr_bytes else None
 628 
 629         return result

 634     def __init__(self, module):

 638             if module.params[o] is not None:

 642     def _generate_csr(self):
 643         req = crypto.X509Req()
 644         req.set_version(self.version - 1)
 645         subject = req.get_subject()
 646         for entry in self.subject:
 647             if entry[1] is not None:
 648                 # Workaround for https://github.com/pyca/pyopenssl/issues/165
 649                 nid = OpenSSL._util.lib.OBJ_txt2nid(to_bytes(entry[0]))
 650                 if nid == 0:
 651                     raise CertificateSigningRequestError('Unknown subject field identifier "{0}"'.format(entry[0]))
 652                 res = OpenSSL._util.lib.X509_NAME_add_entry_by_NID(subject._name, nid, OpenSSL._util.lib.MBSTRING_UTF8, to_bytes(entry[1]), -1, -1, 0)
 653                 if res == 0:
 654                     raise CertificateSigningRequestError('Invalid value for subject field identifier "{0}": {1}'.format(entry[0], entry[1]))
 655 
 656         extensions = []
 657         if self.subjectAltName:
 658             altnames = ', '.join(self.subjectAltName)
 659             try:
 660                 extensions.append(crypto.X509Extension(b"subjectAltName", self.subjectAltName_critical, altnames.encode('ascii')))
 661             except OpenSSL.crypto.Error as e:
 662                 raise CertificateSigningRequestError(
 663                     'Error while parsing Subject Alternative Names {0} (check for missing type prefix, such as "DNS:"!): {1}'.format(
 664                         ', '.join(["{0}".format(san) for san in self.subjectAltName]), str(e)
 665                     )
 666                 )
 667 
 668         if self.keyUsage:
 669             usages = ', '.join(self.keyUsage)
 670             extensions.append(crypto.X509Extension(b"keyUsage", self.keyUsage_critical, usages.encode('ascii')))
 671 
 672         if self.extendedKeyUsage:
 673             usages = ', '.join(self.extendedKeyUsage)
 674             extensions.append(crypto.X509Extension(b"extendedKeyUsage", self.extendedKeyUsage_critical, usages.encode('ascii')))
 675 
 676         if self.basicConstraints:
 677             usages = ', '.join(self.basicConstraints)
 678             extensions.append(crypto.X509Extension(b"basicConstraints", self.basicConstraints_critical, usages.encode('ascii')))
 679 
 680         if self.ocspMustStaple:
 681             extensions.append(crypto.X509Extension(OPENSSL_MUST_STAPLE_NAME, self.ocspMustStaple_critical, OPENSSL_MUST_STAPLE_VALUE))
 682 
 683         if extensions:
 684             req.add_extensions(extensions)
 685 
 686         req.set_pubkey(self.privatekey)
 687         req.sign(self.privatekey, self.digest)
 688         self.request = req
 689 
 690         return crypto.dump_certificate_request(crypto.FILETYPE_PEM, self.request)

 702     def _normalize_san(self, san):
 703         # Apparently OpenSSL returns 'IP address' not 'IP' as specifier when converting the subjectAltName to string
 704         # although it won't accept this specifier when generating the CSR. (https://github.com/openssl/openssl/issues/4004)
 705         if san.startswith('IP Address:'):
 706             san = 'IP:' + san[len('IP Address:'):]
 707         if san.startswith('IP:'):
 708             ip = compat_ipaddress.ip_address(san[3:])
 709             san = 'IP:{0}'.format(ip.compressed)
 710         return san

 712     def _check_csr(self):
 713         def _check_subject(csr):
 714             subject = [(OpenSSL._util.lib.OBJ_txt2nid(to_bytes(sub[0])), to_bytes(sub[1])) for sub in self.subject]
 715             current_subject = [(OpenSSL._util.lib.OBJ_txt2nid(to_bytes(sub[0])), to_bytes(sub[1])) for sub in csr.get_subject().get_components()]
 716             if not set(subject) == set(current_subject):
 717                 return False
 718 
 719             return True
 720 
 721         def _check_subjectAltName(extensions):
 722             altnames_ext = next((ext for ext in extensions if ext.get_short_name() == b'subjectAltName'), '')
 723             altnames = [self._normalize_san(altname.strip()) for altname in
 724                         to_text(altnames_ext, errors='surrogate_or_strict').split(',') if altname.strip()]
 725             if self.subjectAltName:
 726                 if (set(altnames) != set([self._normalize_san(to_text(name)) for name in self.subjectAltName]) or
 727                         altnames_ext.get_critical() != self.subjectAltName_critical):
 728                     return False
 729             else:
 730                 if altnames:
 731                     return False
 732 
 733             return True
 734 
 735         def _check_keyUsage_(extensions, extName, expected, critical):
 736             usages_ext = [ext for ext in extensions if ext.get_short_name() == extName]
 737             if (not usages_ext and expected) or (usages_ext and not expected):
 738                 return False
 739             elif not usages_ext and not expected:
 740                 return True
 741             else:
 742                 current = [OpenSSL._util.lib.OBJ_txt2nid(to_bytes(usage.strip())) for usage in str(usages_ext[0]).split(',')]
 743                 expected = [OpenSSL._util.lib.OBJ_txt2nid(to_bytes(usage)) for usage in expected]
 744                 return set(current) == set(expected) and usages_ext[0].get_critical() == critical
 745 
 746         def _check_keyUsage(extensions):
 747             usages_ext = [ext for ext in extensions if ext.get_short_name() == b'keyUsage']
 748             if (not usages_ext and self.keyUsage) or (usages_ext and not self.keyUsage):
 749                 return False
 750             elif not usages_ext and not self.keyUsage:
 751                 return True
 752             else:
 753                 # OpenSSL._util.lib.OBJ_txt2nid() always returns 0 for all keyUsage values
 754                 # (since keyUsage has a fixed bitfield for these values and is not extensible).
 755                 # Therefore, we create an extension for the wanted values, and compare the
 756                 # data of the extensions (which is the serialized bitfield).
 757                 expected_ext = crypto.X509Extension(b"keyUsage", False, ', '.join(self.keyUsage).encode('ascii'))
 758                 return usages_ext[0].get_data() == expected_ext.get_data() and usages_ext[0].get_critical() == self.keyUsage_critical
 759 
 760         def _check_extenededKeyUsage(extensions):
 761             return _check_keyUsage_(extensions, b'extendedKeyUsage', self.extendedKeyUsage, self.extendedKeyUsage_critical)
 762 
 763         def _check_basicConstraints(extensions):
 764             return _check_keyUsage_(extensions, b'basicConstraints', self.basicConstraints, self.basicConstraints_critical)
 765 
 766         def _check_ocspMustStaple(extensions):
 767             oms_ext = [ext for ext in extensions if to_bytes(ext.get_short_name()) == OPENSSL_MUST_STAPLE_NAME and to_bytes(ext) == OPENSSL_MUST_STAPLE_VALUE]
 768             if OpenSSL.SSL.OPENSSL_VERSION_NUMBER < 0x10100000:
 769                 # Older versions of libssl don't know about OCSP Must Staple
 770                 oms_ext.extend([ext for ext in extensions if ext.get_short_name() == b'UNDEF' and ext.get_data() == b'\x30\x03\x02\x01\x05'])
 771             if self.ocspMustStaple:
 772                 return len(oms_ext) > 0 and oms_ext[0].get_critical() == self.ocspMustStaple_critical
 773             else:
 774                 return len(oms_ext) == 0
 775 
 776         def _check_extensions(csr):
 777             extensions = csr.get_extensions()
 778             return (_check_subjectAltName(extensions) and _check_keyUsage(extensions) and
 779                     _check_extenededKeyUsage(extensions) and _check_basicConstraints(extensions) and
 780                     _check_ocspMustStaple(extensions))
 781 
 782         def _check_signature(csr):
 783             try:
 784                 return csr.verify(self.privatekey)
 785             except crypto.Error:
 786                 return False
 787 
 788         try:
 789             csr = crypto_utils.load_certificate_request(self.path, backend='pyopenssl')
 790         except Exception as dummy:
 791             return False
 792 
 793         return _check_subject(csr) and _check_extensions(csr) and _check_signature(csr)

 796 class CertificateSigningRequestCryptography(CertificateSigningRequestBase):
 797 
 798     def __init__(self, module):
 799         super(CertificateSigningRequestCryptography, self).__init__(module)
 800         self.cryptography_backend = cryptography.hazmat.backends.default_backend()
 801         self.module = module
 802         if self.version != 1:
 803             module.warn('The cryptography backend only supports version 1. (The only valid value according to RFC 2986.)')
 804 
 805     def _generate_csr(self):
 806         csr = cryptography.x509.CertificateSigningRequestBuilder()
 807         try:
 808             csr = csr.subject_name(cryptography.x509.Name([
 809                 cryptography.x509.NameAttribute(crypto_utils.cryptography_name_to_oid(entry[0]), to_text(entry[1])) for entry in self.subject
 810             ]))
 811         except ValueError as e:
 812             raise CertificateSigningRequestError(e)
 813 
 814         if self.subjectAltName:
 815             csr = csr.add_extension(cryptography.x509.SubjectAlternativeName([
 816                 crypto_utils.cryptography_get_name(name) for name in self.subjectAltName
 817             ]), critical=self.subjectAltName_critical)
 818 
 819         if self.keyUsage:
 820             params = crypto_utils.cryptography_parse_key_usage_params(self.keyUsage)
 821             csr = csr.add_extension(cryptography.x509.KeyUsage(**params), critical=self.keyUsage_critical)
 822 
 823         if self.extendedKeyUsage:
 824             usages = [crypto_utils.cryptography_name_to_oid(usage) for usage in self.extendedKeyUsage]
 825             csr = csr.add_extension(cryptography.x509.ExtendedKeyUsage(usages), critical=self.extendedKeyUsage_critical)
 826 
 827         if self.basicConstraints:
 828             params = {}
 829             ca, path_length = crypto_utils.cryptography_get_basic_constraints(self.basicConstraints)
 830             csr = csr.add_extension(cryptography.x509.BasicConstraints(ca, path_length), critical=self.basicConstraints_critical)
 831 
 832         if self.ocspMustStaple:
 833             try:
 834                 # This only works with cryptography >= 2.1
 835                 csr = csr.add_extension(cryptography.x509.TLSFeature([cryptography.x509.TLSFeatureType.status_request]), critical=self.ocspMustStaple_critical)
 836             except AttributeError as dummy:
 837                 csr = csr.add_extension(
 838                     cryptography.x509.UnrecognizedExtension(CRYPTOGRAPHY_MUST_STAPLE_NAME, CRYPTOGRAPHY_MUST_STAPLE_VALUE),
 839                     critical=self.ocspMustStaple_critical
 840                 )
 841 
 842         if self.create_subject_key_identifier:
 843             csr = csr.add_extension(
 844                 cryptography.x509.SubjectKeyIdentifier.from_public_key(self.privatekey.public_key()),
 845                 critical=False
 846             )
 847         elif self.subject_key_identifier is not None:
 848             csr = csr.add_extension(cryptography.x509.SubjectKeyIdentifier(self.subject_key_identifier), critical=False)
 849 
 850         if self.authority_key_identifier is not None or self.authority_cert_issuer is not None or self.authority_cert_serial_number is not None:
 851             issuers = None
 852             if self.authority_cert_issuer is not None:
 853                 issuers = [crypto_utils.cryptography_get_name(n) for n in self.authority_cert_issuer]
 854             csr = csr.add_extension(
 855                 cryptography.x509.AuthorityKeyIdentifier(self.authority_key_identifier, issuers, self.authority_cert_serial_number),
 856                 critical=False
 857             )
 858 
 859         digest = None
 860         if crypto_utils.cryptography_key_needs_digest_for_signing(self.privatekey):
 861             if self.digest == 'sha256':
 862                 digest = cryptography.hazmat.primitives.hashes.SHA256()
 863             elif self.digest == 'sha384':
 864                 digest = cryptography.hazmat.primitives.hashes.SHA384()
 865             elif self.digest == 'sha512':
 866                 digest = cryptography.hazmat.primitives.hashes.SHA512()
 867             elif self.digest == 'sha1':
 868                 digest = cryptography.hazmat.primitives.hashes.SHA1()
 869             elif self.digest == 'md5':
 870                 digest = cryptography.hazmat.primitives.hashes.MD5()
 871             # FIXME
 872             else:
 873                 raise CertificateSigningRequestError('Unsupported digest "{0}"'.format(self.digest))
 874         try:
 875             self.request = csr.sign(self.privatekey, digest, self.cryptography_backend)
 876         except TypeError as e:
 877             if str(e) == 'Algorithm must be a registered hash algorithm.' and digest is None:
 878                 self.module.fail_json(msg='Signing with Ed25519 and Ed448 keys requires cryptography 2.8 or newer.')
 879             raise
 880 
 881         return self.request.public_bytes(cryptography.hazmat.primitives.serialization.Encoding.PEM)

 883     def _load_private_key(self):
 884         try:
 885             if self.privatekey_content is not None:
 886                 content = self.privatekey_content
 887             else:
 888                 with open(self.privatekey_path, 'rb') as f:
 889                     content = f.read()
 890             self.privatekey = cryptography.hazmat.primitives.serialization.load_pem_private_key(
 891                 content,
 892                 None if self.privatekey_passphrase is None else to_bytes(self.privatekey_passphrase),
 893                 backend=self.cryptography_backend
 894             )
 895         except Exception as e:
 896             raise CertificateSigningRequestError(e)

 898     def _check_csr(self):
 899         def _check_subject(csr):
 900             subject = [(crypto_utils.cryptography_name_to_oid(entry[0]), entry[1]) for entry in self.subject]
 901             current_subject = [(sub.oid, sub.value) for sub in csr.subject]
 902             return set(subject) == set(current_subject)
 903 
 904         def _find_extension(extensions, exttype):
 905             return next(
 906                 (ext for ext in extensions if isinstance(ext.value, exttype)),
 907                 None
 908             )
 909 
 910         def _check_subjectAltName(extensions):
 911             current_altnames_ext = _find_extension(extensions, cryptography.x509.SubjectAlternativeName)
 912             current_altnames = [str(altname) for altname in current_altnames_ext.value] if current_altnames_ext else []
 913             altnames = [str(crypto_utils.cryptography_get_name(altname)) for altname in self.subjectAltName] if self.subjectAltName else []
 914             if set(altnames) != set(current_altnames):
 915                 return False
 916             if altnames:
 917                 if current_altnames_ext.critical != self.subjectAltName_critical:
 918                     return False
 919             return True
 920 
 921         def _check_keyUsage(extensions):
 922             current_keyusage_ext = _find_extension(extensions, cryptography.x509.KeyUsage)
 923             if not self.keyUsage:
 924                 return current_keyusage_ext is None
 925             elif current_keyusage_ext is None:
 926                 return False
 927             params = crypto_utils.cryptography_parse_key_usage_params(self.keyUsage)
 928             for param in params:
 929                 if getattr(current_keyusage_ext.value, '_' + param) != params[param]:
 930                     return False
 931             if current_keyusage_ext.critical != self.keyUsage_critical:
 932                 return False
 933             return True
 934 
 935         def _check_extenededKeyUsage(extensions):
 936             current_usages_ext = _find_extension(extensions, cryptography.x509.ExtendedKeyUsage)
 937             current_usages = [str(usage) for usage in current_usages_ext.value] if current_usages_ext else []
 938             usages = [str(crypto_utils.cryptography_name_to_oid(usage)) for usage in self.extendedKeyUsage] if self.extendedKeyUsage else []
 939             if set(current_usages) != set(usages):
 940                 return False
 941             if usages:
 942                 if current_usages_ext.critical != self.extendedKeyUsage_critical:
 943                     return False
 944             return True
 945 
 946         def _check_basicConstraints(extensions):
 947             bc_ext = _find_extension(extensions, cryptography.x509.BasicConstraints)
 948             current_ca = bc_ext.value.ca if bc_ext else False
 949             current_path_length = bc_ext.value.path_length if bc_ext else None
 950             ca, path_length = crypto_utils.cryptography_get_basic_constraints(self.basicConstraints)
 951             # Check CA flag
 952             if ca != current_ca:
 953                 return False
 954             # Check path length
 955             if path_length != current_path_length:
 956                 return False
 957             # Check criticality
 958             if self.basicConstraints:
 959                 if bc_ext.critical != self.basicConstraints_critical:
 960                     return False
 961             return True
 962 
 963         def _check_ocspMustStaple(extensions):
 964             try:
 965                 # This only works with cryptography >= 2.1
 966                 tlsfeature_ext = _find_extension(extensions, cryptography.x509.TLSFeature)
 967                 has_tlsfeature = True
 968             except AttributeError as dummy:
 969                 tlsfeature_ext = next(
 970                     (ext for ext in extensions if ext.value.oid == CRYPTOGRAPHY_MUST_STAPLE_NAME),
 971                     None
 972                 )
 973                 has_tlsfeature = False
 974             if self.ocspMustStaple:
 975                 if not tlsfeature_ext or tlsfeature_ext.critical != self.ocspMustStaple_critical:
 976                     return False
 977                 if has_tlsfeature:
 978                     return cryptography.x509.TLSFeatureType.status_request in tlsfeature_ext.value
 979                 else:
 980                     return tlsfeature_ext.value.value == CRYPTOGRAPHY_MUST_STAPLE_VALUE
 981             else:
 982                 return tlsfeature_ext is None
 983 
 984         def _check_subject_key_identifier(extensions):
 985             ext = _find_extension(extensions, cryptography.x509.SubjectKeyIdentifier)
 986             if self.create_subject_key_identifier or self.subject_key_identifier is not None:
 987                 if not ext or ext.critical:
 988                     return False
 989                 if self.create_subject_key_identifier:
 990                     digest = cryptography.x509.SubjectKeyIdentifier.from_public_key(self.privatekey.public_key()).digest
 991                     return ext.value.digest == digest
 992                 else:
 993                     return ext.value.digest == self.subject_key_identifier
 994             else:
 995                 return ext is None
 996 
 997         def _check_authority_key_identifier(extensions):
 998             ext = _find_extension(extensions, cryptography.x509.AuthorityKeyIdentifier)
 999             if self.authority_key_identifier is not None or self.authority_cert_issuer is not None or self.authority_cert_serial_number is not None:
1000                 if not ext or ext.critical:
1001                     return False
1002                 aci = None
1003                 csr_aci = None
1004                 if self.authority_cert_issuer is not None:
1005                     aci = [str(crypto_utils.cryptography_get_name(n)) for n in self.authority_cert_issuer]
1006                 if ext.value.authority_cert_issuer is not None:
1007                     csr_aci = [str(n) for n in ext.value.authority_cert_issuer]
1008                 return (ext.value.key_identifier == self.authority_key_identifier
1009                         and csr_aci == aci
1010                         and ext.value.authority_cert_serial_number == self.authority_cert_serial_number)
1011             else:
1012                 return ext is None
1013 
1014         def _check_extensions(csr):
1015             extensions = csr.extensions
1016             return (_check_subjectAltName(extensions) and _check_keyUsage(extensions) and
1017                     _check_extenededKeyUsage(extensions) and _check_basicConstraints(extensions) and
1018                     _check_ocspMustStaple(extensions) and _check_subject_key_identifier(extensions) and
1019                     _check_authority_key_identifier(extensions))
1020 
1021         def _check_signature(csr):
1022             if not csr.is_signature_valid:
1023                 return False
1024             # To check whether public key of CSR belongs to private key,
1025             # encode both public keys and compare PEMs.
1026             key_a = csr.public_key().public_bytes(
1027                 cryptography.hazmat.primitives.serialization.Encoding.PEM,
1028                 cryptography.hazmat.primitives.serialization.PublicFormat.SubjectPublicKeyInfo
1029             )
1030             key_b = self.privatekey.public_key().public_bytes(
1031                 cryptography.hazmat.primitives.serialization.Encoding.PEM,
1032                 cryptography.hazmat.primitives.serialization.PublicFormat.SubjectPublicKeyInfo
1033             )
1034             return key_a == key_b
1035 
1036         try:
1037             csr = crypto_utils.load_certificate_request(self.path, backend='cryptography')
1038         except Exception as dummy:
1039             return False
1040 
1041         return _check_subject(csr) and _check_extensions(csr) and _check_signature(csr)
