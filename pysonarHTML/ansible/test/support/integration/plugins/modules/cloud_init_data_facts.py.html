  90 import os
  91 
  92 from ansible.module_utils.basic import AnsibleModule
  93 from ansible.module_utils._text import to_text
  94 
  95 
  96 CLOUD_INIT_PATH = "/var/lib/cloud/data/"
  97 
  98 
  99 def gather_cloud_init_data_facts(module):
 100     res = {
 101         'cloud_init_data_facts': dict()
 102     }
 103 
 104     for i in ['result', 'status']:
 105         filter = module.params.get('filter')
 106         if filter is None or filter == i:
 107             res['cloud_init_data_facts'][i] = dict()
 108             json_file = CLOUD_INIT_PATH + i + '.json'
 109 
 110             if os.path.exists(json_file):
 111                 f = open(json_file, 'rb')
 112                 contents = to_text(f.read(), errors='surrogate_or_strict')
 113                 f.close()
 114 
 115                 if contents:
 116                     res['cloud_init_data_facts'][i] = module.from_json(contents)
 117     return res
 118 
 119 
 120 def main():
 121     module = AnsibleModule(
 122         argument_spec=dict(
 123             filter=dict(choices=['result', 'status']),
 124         ),
 125         supports_check_mode=True,
 126     )
 127 
 128     facts = gather_cloud_init_data_facts(module)
 129     result = dict(changed=False, ansible_facts=facts, **facts)
 130     module.exit_json(**result)
