 162 class VmAttributeManager(PyVmomi):
 163     def __init__(self, module):
 164         super(VmAttributeManager, self).__init__(module)
 165 
 166     def set_custom_field(self, vm, user_fields):
 167         result_fields = dict()
 168         change_list = list()
 169         changed = False
 170 
 171         for field in user_fields:
 172             field_key = self.check_exists(field['name'])
 173             found = False
 174             field_value = field.get('value', '')
 175 
 176             for k, v in [(x.name, v.value) for x in self.custom_field_mgr for v in vm.customValue if x.key == v.key]:
 177                 if k == field['name']:
 178                     found = True
 179                     if v != field_value:
 180                         if not self.module.check_mode:
 181                             self.content.customFieldsManager.SetField(entity=vm, key=field_key.key, value=field_value)
 182                             result_fields[k] = field_value
 183                         change_list.append(True)
 184             if not found and field_value != "":
 185                 if not field_key and not self.module.check_mode:
 186                     field_key = self.content.customFieldsManager.AddFieldDefinition(name=field['name'], moType=vim.VirtualMachine)
 187                 change_list.append(True)
 188                 if not self.module.check_mode:
 189                     self.content.customFieldsManager.SetField(entity=vm, key=field_key.key, value=field_value)
 190                 result_fields[field['name']] = field_value
 191 
 192         if any(change_list):
 193             changed = True
 194 
 195         return {'changed': changed, 'failed': False, 'custom_attributes': result_fields}
 196 
 197     def check_exists(self, field):
 198         for x in self.custom_field_mgr:
 199             if x.name == field:
 200                 return x
 201         return False

 204 def main():
 205     argument_spec = vmware_argument_spec()
 206     argument_spec.update(
 207         datacenter=dict(type='str'),
 208         name=dict(type='str'),
 209         folder=dict(type='str'),
 210         uuid=dict(type='str'),
 211         moid=dict(type='str'),
 212         use_instance_uuid=dict(type='bool', default=False),
 213         state=dict(type='str', default='present',
 214                    choices=['absent', 'present']),
 215         attributes=dict(
 216             type='list',
 217             default=[],
 218             options=dict(
 219                 name=dict(type='str', required=True),
 220                 value=dict(type='str'),
 221             )
 222         ),
 223     )
 224 
 225     module = AnsibleModule(
 226         argument_spec=argument_spec,
 227         supports_check_mode=True,
 228         required_one_of=[
 229             ['name', 'uuid', 'moid']
 230         ],
 231     )
 232 
 233     if module.params.get('folder'):
 234         # FindByInventoryPath() does not require an absolute path
 235         # so we should leave the input folder path unmodified
 236         module.params['folder'] = module.params['folder'].rstrip('/')
 237 
 238     pyv = VmAttributeManager(module)
 239     results = {'changed': False, 'failed': False, 'instance': dict()}
 240 
 241     # Check if the virtual machine exists before continuing
 242     vm = pyv.get_vm()
 243 
 244     if vm:
 245         # virtual machine already exists
 246         if module.params['state'] == "present":
 247             results = pyv.set_custom_field(vm, module.params['attributes'])
 248         elif module.params['state'] == "absent":
 249             results = pyv.set_custom_field(vm, module.params['attributes'])
 250         module.exit_json(**results)
 251     else:
 252         # virtual machine does not exists
 253         vm_id = (module.params.get('name') or module.params.get('uuid') or module.params.get('moid'))
 254         module.fail_json(msg="Unable to manage custom attributes for non-existing"
 255                              " virtual machine %s" % vm_id)
 256 
 257 
 258 if __name__ == '__main__':
 259     main()
