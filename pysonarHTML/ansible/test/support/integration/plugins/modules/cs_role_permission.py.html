 117 from distutils.version import LooseVersion
 118 
 119 from ansible.module_utils.basic import AnsibleModule
 120 from ansible.module_utils.cloudstack import (
 121     AnsibleCloudStack,
 122     cs_argument_spec,
 123     cs_required_together,
 124 )

 127 class AnsibleCloudStackRolePermission(AnsibleCloudStack):
 128 
 129     def __init__(self, module):
 130         super(AnsibleCloudStackRolePermission, self).__init__(module)
 131         cloudstack_min_version = LooseVersion('4.9.2')
 132 
 133         self.returns = {
 134             'id': 'id',
 135             'roleid': 'role_id',
 136             'rule': 'name',
 137             'permission': 'permission',
 138             'description': 'description',
 139         }
 140         self.role_permission = None
 141 
 142         self.cloudstack_version = self._cloudstack_ver()
 143 
 144         if self.cloudstack_version < cloudstack_min_version:
 145             self.fail_json(msg="This module requires CloudStack >= %s." % cloudstack_min_version)
 146 
 147     def _cloudstack_ver(self):
 148         capabilities = self.get_capabilities()
 149         return LooseVersion(capabilities['cloudstackversion'])
 150 
 151     def _get_role_id(self):
 152         role = self.module.params.get('role')
 153         if not role:
 154             return None
 155 
 156         res = self.query_api('listRoles')
 157         roles = res['role']
 158         if roles:
 159             for r in roles:
 160                 if role in [r['name'], r['id']]:
 161                     return r['id']
 162         self.fail_json(msg="Role '%s' not found" % role)
 163 
 164     def _get_role_perm(self):
 165         role_permission = self.role_permission
 166 
 167         args = {
 168             'roleid': self._get_role_id(),
 169         }
 170 
 171         rp = self.query_api('listRolePermissions', **args)
 172 
 173         if rp:
 174             role_permission = rp['rolepermission']
 175 
 176         return role_permission
 177 
 178     def _get_rule(self, rule=None):
 179         if not rule:
 180             rule = self.module.params.get('name')
 181 
 182         if self._get_role_perm():
 183             for _rule in self._get_role_perm():
 184                 if rule == _rule['rule'] or rule == _rule['id']:
 185                     return _rule
 186 
 187         return None
 188 
 189     def _get_rule_order(self):
 190         perms = self._get_role_perm()
 191         rules = []
 192 
 193         if perms:
 194             for i, rule in enumerate(perms):
 195                 rules.append(rule['id'])
 196 
 197         return rules
 198 
 199     def replace_rule(self):
 200         old_rule = self._get_rule()
 201 
 202         if old_rule:
 203             rules_order = self._get_rule_order()
 204             old_pos = rules_order.index(old_rule['id'])
 205 
 206             self.remove_role_perm()
 207 
 208             new_rule = self.create_role_perm()
 209 
 210             if new_rule:
 211                 perm_order = self.order_permissions(int(old_pos - 1), new_rule['id'])
 212 
 213                 return perm_order
 214 
 215         return None
 216 
 217     def order_permissions(self, parent, rule_id):
 218         rules = self._get_rule_order()
 219 
 220         if isinstance(parent, int):
 221             parent_pos = parent
 222         elif parent == '0':
 223             parent_pos = -1
 224         else:
 225             parent_rule = self._get_rule(parent)
 226             if not parent_rule:
 227                 self.fail_json(msg="Parent rule '%s' not found" % parent)
 228 
 229             parent_pos = rules.index(parent_rule['id'])
 230 
 231         r_id = rules.pop(rules.index(rule_id))
 232 
 233         rules.insert((parent_pos + 1), r_id)
 234         rules = ','.join(map(str, rules))
 235 
 236         return rules
 237 
 238     def create_or_update_role_perm(self):
 239         role_permission = self._get_rule()
 240 
 241         if not role_permission:
 242             role_permission = self.create_role_perm()
 243         else:
 244             role_permission = self.update_role_perm(role_permission)
 245 
 246         return role_permission
 247 
 248     def create_role_perm(self):
 249         role_permission = None
 250 
 251         self.result['changed'] = True
 252 
 253         args = {
 254             'rule': self.module.params.get('name'),
 255             'description': self.module.params.get('description'),
 256             'roleid': self._get_role_id(),
 257             'permission': self.module.params.get('permission'),
 258         }
 259 
 260         if not self.module.check_mode:
 261             res = self.query_api('createRolePermission', **args)
 262             role_permission = res['rolepermission']
 263 
 264         return role_permission
 265 
 266     def update_role_perm(self, role_perm):
 267         perm_order = None
 268 
 269         if not self.module.params.get('parent'):
 270             args = {
 271                 'ruleid': role_perm['id'],
 272                 'roleid': role_perm['roleid'],
 273                 'permission': self.module.params.get('permission'),
 274             }
 275 
 276             if self.has_changed(args, role_perm, only_keys=['permission']):
 277                 self.result['changed'] = True
 278 
 279                 if not self.module.check_mode:
 280                     if self.cloudstack_version >= LooseVersion('4.11.0'):
 281                         self.query_api('updateRolePermission', **args)
 282                         role_perm = self._get_rule()
 283                     else:
 284                         perm_order = self.replace_rule()
 285         else:
 286             perm_order = self.order_permissions(self.module.params.get('parent'), role_perm['id'])
 287 
 288         if perm_order:
 289             args = {
 290                 'roleid': role_perm['roleid'],
 291                 'ruleorder': perm_order,
 292             }
 293 
 294             self.result['changed'] = True
 295 
 296             if not self.module.check_mode:
 297                 self.query_api('updateRolePermission', **args)
 298                 role_perm = self._get_rule()
 299 
 300         return role_perm
 301 
 302     def remove_role_perm(self):
 303         role_permission = self._get_rule()
 304 
 305         if role_permission:
 306             self.result['changed'] = True
 307 
 308             args = {
 309                 'id': role_permission['id'],
 310             }
 311 
 312             if not self.module.check_mode:
 313                 self.query_api('deleteRolePermission', **args)
 314 
 315         return role_permission

 318 def main():
 319     argument_spec = cs_argument_spec()
 320     argument_spec.update(dict(
 321         role=dict(required=True),
 322         name=dict(required=True),
 323         permission=dict(choices=['allow', 'deny'], default='deny'),
 324         description=dict(),
 325         state=dict(choices=['present', 'absent'], default='present'),
 326         parent=dict(),
 327     ))
 328 
 329     module = AnsibleModule(
 330         argument_spec=argument_spec,
 331         required_together=cs_required_together(),
 332         mutually_exclusive=(
 333             ['permission', 'parent'],
 334         ),
 335         supports_check_mode=True
 336     )
 337 
 338     acs_role_perm = AnsibleCloudStackRolePermission(module)
 339 
 340     state = module.params.get('state')
 341     if state in ['absent']:
 342         role_permission = acs_role_perm.remove_role_perm()
 343     else:
 344         role_permission = acs_role_perm.create_or_update_role_perm()
 345 
 346     result = acs_role_perm.get_result(role_permission)
 347     module.exit_json(**result)
