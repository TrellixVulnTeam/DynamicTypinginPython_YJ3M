 429 def rule_cmp(a, b):
 430     """Compare rules without descriptions"""
 431     for prop in ['port_range', 'protocol', 'target', 'target_type']:
 432         if prop == 'port_range' and to_text(a.protocol) == to_text(b.protocol):
 433             # equal protocols can interchange `(-1, -1)` and `(None, None)`
 434             if a.port_range in ((None, None), (-1, -1)) and b.port_range in ((None, None), (-1, -1)):
 435                 continue
 436             elif getattr(a, prop) != getattr(b, prop):
 437                 return False
 438         elif getattr(a, prop) != getattr(b, prop):
 439             return False
 440     return True

 447 def to_permission(rule):
 448     # take a Rule, output the serialized grant
 449     perm = {
 450         'IpProtocol': rule.protocol,
 451     }
 452     perm['FromPort'], perm['ToPort'] = rule.port_range
 453     if rule.target_type == 'ipv4':
 454         perm['IpRanges'] = [{
 455             'CidrIp': rule.target,
 456         }]
 457         if rule.description:
 458             perm['IpRanges'][0]['Description'] = rule.description
 459     elif rule.target_type == 'ipv6':
 460         perm['Ipv6Ranges'] = [{
 461             'CidrIpv6': rule.target,
 462         }]
 463         if rule.description:
 464             perm['Ipv6Ranges'][0]['Description'] = rule.description
 465     elif rule.target_type == 'group':
 466         if isinstance(rule.target, tuple):
 467             pair = {}
 468             if rule.target[0]:
 469                 pair['UserId'] = rule.target[0]
 470             # group_id/group_name are mutually exclusive - give group_id more precedence as it is more specific
 471             if rule.target[1]:
 472                 pair['GroupId'] = rule.target[1]
 473             elif rule.target[2]:
 474                 pair['GroupName'] = rule.target[2]
 475             perm['UserIdGroupPairs'] = [pair]
 476         else:
 477             perm['UserIdGroupPairs'] = [{
 478                 'GroupId': rule.target
 479             }]
 480         if rule.description:
 481             perm['UserIdGroupPairs'][0]['Description'] = rule.description
 482     elif rule.target_type == 'ip_prefix':
 483         perm['PrefixListIds'] = [{
 484             'PrefixListId': rule.target,
 485         }]
 486         if rule.description:
 487             perm['PrefixListIds'][0]['Description'] = rule.description
 488     elif rule.target_type not in valid_targets:
 489         raise ValueError('Invalid target type for rule {0}'.format(rule))
 490     return fix_port_and_protocol(perm)

 493 def rule_from_group_permission(perm):
 494     def ports_from_permission(p):
 495         if 'FromPort' not in p and 'ToPort' not in p:
 496             return (None, None)
 497         return (int(perm['FromPort']), int(perm['ToPort']))
 498 
 499     # outputs a rule tuple
 500     for target_key, target_subkey, target_type in [
 501         ('IpRanges', 'CidrIp', 'ipv4'),
 502         ('Ipv6Ranges', 'CidrIpv6', 'ipv6'),
 503         ('PrefixListIds', 'PrefixListId', 'ip_prefix'),
 504     ]:
 505         if target_key not in perm:
 506             continue
 507         for r in perm[target_key]:
 508             # there may be several IP ranges here, which is ok
 509             yield Rule(
 510                 ports_from_permission(perm),
 511                 to_text(perm['IpProtocol']),
 512                 r[target_subkey],
 513                 target_type,
 514                 r.get('Description')
 515             )
 516     if 'UserIdGroupPairs' in perm and perm['UserIdGroupPairs']:
 517         for pair in perm['UserIdGroupPairs']:
 518             target = (
 519                 pair.get('UserId', None),
 520                 pair.get('GroupId', None),
 521                 pair.get('GroupName', None),
 522             )
 523             if pair.get('UserId', '').startswith('amazon-'):
 524                 # amazon-elb and amazon-prefix rules don't need
 525                 # group-id specified, so remove it when querying
 526                 # from permission
 527                 target = (
 528                     target[0],
 529                     None,
 530                     target[2],
 531                 )
 532             elif 'VpcPeeringConnectionId' in pair or pair['UserId'] != current_account_id:
 533                 target = (
 534                     pair.get('UserId', None),
 535                     pair.get('GroupId', None),
 536                     pair.get('GroupName', None),
 537                 )
 538 
 539             yield Rule(
 540                 ports_from_permission(perm),
 541                 to_text(perm['IpProtocol']),
 542                 target,
 543                 'group',
 544                 pair.get('Description')
 545             )

 568 def validate_rule(module, rule):
 569     VALID_PARAMS = ('cidr_ip', 'cidr_ipv6', 'ip_prefix',
 570                     'group_id', 'group_name', 'group_desc',
 571                     'proto', 'from_port', 'to_port', 'rule_desc')
 572     if not isinstance(rule, dict):
 573         module.fail_json(msg='Invalid rule parameter type [%s].' % type(rule))
 574     for k in rule:
 575         if k not in VALID_PARAMS:
 576             module.fail_json(msg='Invalid rule parameter \'{0}\' for rule: {1}'.format(k, rule))
 577 
 578     if 'group_id' in rule and 'cidr_ip' in rule:
 579         module.fail_json(msg='Specify group_id OR cidr_ip, not both')
 580     elif 'group_name' in rule and 'cidr_ip' in rule:
 581         module.fail_json(msg='Specify group_name OR cidr_ip, not both')
 582     elif 'group_id' in rule and 'cidr_ipv6' in rule:
 583         module.fail_json(msg="Specify group_id OR cidr_ipv6, not both")
 584     elif 'group_name' in rule and 'cidr_ipv6' in rule:
 585         module.fail_json(msg="Specify group_name OR cidr_ipv6, not both")
 586     elif 'cidr_ip' in rule and 'cidr_ipv6' in rule:
 587         module.fail_json(msg="Specify cidr_ip OR cidr_ipv6, not both")
 588     elif 'group_id' in rule and 'group_name' in rule:
 589         module.fail_json(msg='Specify group_id OR group_name, not both')

 606     group_name = None

 616         # group_id/group_name are mutually exclusive - give group_id more precedence as it is more specific
 617         if group_id and group_name:
 618             group_name = None
 619         return 'group', (owner_id, group_id, group_name), False
 620     elif 'group_id' in rule:
 621         return 'group', rule['group_id'], False
 622     elif 'group_name' in rule:
 623         group_name = rule['group_name']
 624         if group_name == name:

 626             groups[group_id] = group
 627             groups[group_name] = group
 628         elif group_name in groups and group.get('VpcId') and groups[group_name].get('VpcId'):
 629             # both are VPC groups, this is ok

 631         elif group_name in groups and not (group.get('VpcId') or groups[group_name].get('VpcId')):
 632             # both are EC2 classic, this is ok
 633             group_id = groups[group_name]['GroupId']
 634         else:
 635             auto_group = None
 636             filters = {'group-name': group_name}
 637             if vpc_id:
 638                 filters['vpc-id'] = vpc_id
 639             # if we got here, either the target group does not exist, or there
 640             # is a mix of EC2 classic + VPC groups. Mixing of EC2 classic + VPC

 642             # exists
 643             if not rule.get('group_desc', '').strip():

 645                 try:
 646                     auto_group = get_security_groups_with_backoff(client, Filters=ansible_dict_to_boto3_filter_list(filters)).get('SecurityGroups', [])[0]

 649                                          "no description was provided" % (group_name, rule))

 651                     module.fail_json_aws(e)
 652             elif not module.check_mode:
 653                 params = dict(GroupName=group_name, Description=rule['group_desc'])
 654                 if vpc_id:
 655                     params['VpcId'] = vpc_id
 656                 try:
 657                     auto_group = client.create_security_group(**params)
 658                     get_waiter(
 659                         client, 'security_group_exists',
 660                     ).wait(
 661                         GroupIds=[auto_group['GroupId']],
 662                     )
 663                 except is_boto3_error_code('InvalidGroup.Duplicate'):
 664                     # The group exists, but didn't show up in any of our describe-security-groups calls
 665                     # Try searching on a filter for the name, and allow a retry window for AWS to update
 666                     # the model on their end.
 667                     try:
 668                         auto_group = get_security_groups_with_backoff(client, Filters=ansible_dict_to_boto3_filter_list(filters)).get('SecurityGroups', [])[0]
 669                     except IndexError as e:

 671                     except ClientError as e:
 672                         module.fail_json_aws(
 673                             e,
 674                             msg="Could not create or use existing group '{0}' in rule. Make sure the group exists".format(group_name))

 677                 groups[group_id] = auto_group
 678                 groups[group_name] = auto_group

 833 def validate_ip(module, cidr_ip):
 834     split_addr = cidr_ip.split('/')
 835     if len(split_addr) == 2:
 836         # this_ip is a IPv4 or IPv6 CIDR that may or may not have host bits set
 837         # Get the network bits if IPv4, and validate if IPv6.
 838         try:
 839             ip = to_subnet(split_addr[0], split_addr[1])
 840             if ip != cidr_ip:
 841                 module.warn("One of your CIDR addresses ({0}) has host bits set. To get rid of this warning, "
 842                             "check the network mask and make sure that only network bits are set: {1}.".format(
 843                                 cidr_ip, ip))
 844         except ValueError:
 845             # to_subnet throws a ValueError on IPv6 networks, so we should be working with v6 if we get here
 846             try:
 847                 isinstance(ip_network(to_text(cidr_ip)), IPv6Network)
 848                 ip = cidr_ip
 849             except ValueError:
 850                 # If a host bit is set on something other than a /128, IPv6Network will throw a ValueError
 851                 # The ipv6_cidr in this case probably looks like "2001:DB8:A0B:12F0::1/64" and we just want the network bits
 852                 ip6 = to_ipv6_subnet(split_addr[0]) + "/" + split_addr[1]
 853                 if ip6 != cidr_ip:
 854                     module.warn("One of your IPv6 CIDR addresses ({0}) has host bits set. To get rid of this warning, "
 855                                 "check the network mask and make sure that only network bits are set: {1}.".format(cidr_ip, ip6))
 856                 return ip6
 857         return ip
 858     return cidr_ip

 932 def wait_for_rule_propagation(module, group, desired_ingress, desired_egress, purge_ingress, purge_egress):
 933     group_id = group['GroupId']
 934     tries = 6
 935 
 936     def await_rules(group, desired_rules, purge, rule_key):
 937         for i in range(tries):
 938             current_rules = set(sum([list(rule_from_group_permission(p)) for p in group[rule_key]], []))
 939             if purge and len(current_rules ^ set(desired_rules)) == 0:
 940                 return group
 941             elif purge:
 942                 conflicts = current_rules ^ set(desired_rules)
 943                 # For cases where set comparison is equivalent, but invalid port/proto exist
 944                 for a, b in itertools.combinations(conflicts, 2):
 945                     if rule_cmp(a, b):
 946                         conflicts.discard(a)
 947                         conflicts.discard(b)
 948                 if not len(conflicts):
 949                     return group
 950             elif current_rules.issuperset(desired_rules) and not purge:
 951                 return group
 952             sleep(10)
 953             group = get_security_groups_with_backoff(module.client('ec2'), GroupIds=[group_id])['SecurityGroups'][0]
 954         module.warn("Ran out of time waiting for {0} {1}. Current: {2}, Desired: {3}".format(group_id, rule_key, current_rules, desired_rules))
 955         return group
 956 
 957     group = get_security_groups_with_backoff(module.client('ec2'), GroupIds=[group_id])['SecurityGroups'][0]
 958     if 'VpcId' in group and module.params.get('rules_egress') is not None:
 959         group = await_rules(group, desired_egress, purge_egress, 'IpPermissionsEgress')
 960     return await_rules(group, desired_ingress, purge_ingress, 'IpPermissions')
