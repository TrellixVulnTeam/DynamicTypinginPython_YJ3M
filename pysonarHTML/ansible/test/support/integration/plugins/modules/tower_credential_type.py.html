 104 KIND_CHOICES = {
 105     'ssh': 'Machine',
 106     'vault': 'Ansible Vault',
 107     'net': 'Network',
 108     'scm': 'Source Control',
 109     'cloud': 'Lots of others',
 110     'insights': 'Insights'

 114 def main():
 115     argument_spec = dict(
 116         name=dict(required=True),
 117         description=dict(required=False),
 118         kind=dict(required=False, choices=KIND_CHOICES.keys()),
 119         inputs=dict(type='dict', required=False),
 120         injectors=dict(type='dict', required=False),
 121         state=dict(choices=['present', 'absent'], default='present'),
 122     )
 123 
 124     module = TowerModule(
 125         argument_spec=argument_spec,
 126         supports_check_mode=False
 127     )
 128 
 129     name = module.params.get('name')
 130     kind = module.params.get('kind')
 131     state = module.params.get('state')
 132 
 133     json_output = {'credential_type': name, 'state': state}
 134 
 135     tower_auth = tower_auth_config(module)
 136     with settings.runtime_values(**tower_auth):
 137         tower_check_mode(module)
 138         credential_type_res = tower_cli.get_resource('credential_type')
 139 
 140         params = {}
 141         params['name'] = name
 142         params['kind'] = kind
 143         params['managed_by_tower'] = False
 144 
 145         if module.params.get('description'):
 146             params['description'] = module.params.get('description')
 147 
 148         if module.params.get('inputs'):
 149             params['inputs'] = module.params.get('inputs')
 150 
 151         if module.params.get('injectors'):
 152             params['injectors'] = module.params.get('injectors')
 153 
 154         try:
 155             if state == 'present':
 156                 params['create_on_missing'] = True
 157                 result = credential_type_res.modify(**params)
 158                 json_output['id'] = result['id']
 159             elif state == 'absent':
 160                 params['fail_on_missing'] = False
 161                 result = credential_type_res.delete(**params)
 162 
 163         except (exc.ConnectionError, exc.BadRequest, exc.AuthError) as excinfo:
 164             module.fail_json(
 165                 msg='Failed to update credential type: {0}'.format(excinfo),
 166                 changed=False
 167             )
 168 
 169     json_output['changed'] = result['changed']
 170     module.exit_json(**json_output)
 171 
 172 
 173 if __name__ == '__main__':
 174     main()
