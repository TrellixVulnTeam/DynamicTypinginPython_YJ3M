 389     def __init__(self, module):
 390         super(CRL, self).__init__(
 391             module.params['path'],
 392             module.params['state'],
 393             module.params['force'],
 394             module.check_mode
 395         )
 396 
 397         self.update = module.params['mode'] == 'update'
 398         self.ignore_timestamps = module.params['ignore_timestamps']
 399         self.return_content = module.params['return_content']
 400         self.crl_content = None
 401 
 402         self.privatekey_path = module.params['privatekey_path']
 403         self.privatekey_content = module.params['privatekey_content']
 404         if self.privatekey_content is not None:
 405             self.privatekey_content = self.privatekey_content.encode('utf-8')
 406         self.privatekey_passphrase = module.params['privatekey_passphrase']
 407 
 408         self.issuer = crypto_utils.parse_name_field(module.params['issuer'])
 409         self.issuer = [(entry[0], entry[1]) for entry in self.issuer if entry[1]]
 410 
 411         self.last_update = crypto_utils.get_relative_time_option(module.params['last_update'], 'last_update')
 412         self.next_update = crypto_utils.get_relative_time_option(module.params['next_update'], 'next_update')
 413 
 414         self.digest = crypto_utils.select_message_digest(module.params['digest'])
 415         if self.digest is None:
 416             raise CRLError('The digest "{0}" is not supported'.format(module.params['digest']))
 417 
 418         self.revoked_certificates = []
 419         for i, rc in enumerate(module.params['revoked_certificates']):
 420             result = {
 421                 'serial_number': None,
 422                 'revocation_date': None,
 423                 'issuer': None,
 424                 'issuer_critical': False,
 425                 'reason': None,
 426                 'reason_critical': False,
 427                 'invalidity_date': None,
 428                 'invalidity_date_critical': False,
 429             }
 430             path_prefix = 'revoked_certificates[{0}].'.format(i)
 431             if rc['path'] is not None or rc['content'] is not None:
 432                 # Load certificate from file or content
 433                 try:
 434                     if rc['content'] is not None:
 435                         rc['content'] = rc['content'].encode('utf-8')
 436                     cert = crypto_utils.load_certificate(rc['path'], content=rc['content'], backend='cryptography')
 437                     try:
 438                         result['serial_number'] = cert.serial_number
 439                     except AttributeError:
 440                         # The property was called "serial" before cryptography 1.4
 441                         result['serial_number'] = cert.serial
 442                 except crypto_utils.OpenSSLObjectError as e:
 443                     if rc['content'] is not None:
 444                         module.fail_json(
 445                             msg='Cannot parse certificate from {0}content: {1}'.format(path_prefix, to_native(e))
 446                         )
 447                     else:
 448                         module.fail_json(
 449                             msg='Cannot read certificate "{1}" from {0}path: {2}'.format(path_prefix, rc['path'], to_native(e))
 450                         )
 451             else:
 452                 # Specify serial_number (and potentially issuer) directly
 453                 result['serial_number'] = rc['serial_number']
 454             # All other options
 455             if rc['issuer']:
 456                 result['issuer'] = [crypto_utils.cryptography_get_name(issuer) for issuer in rc['issuer']]
 457                 result['issuer_critical'] = rc['issuer_critical']
 458             result['revocation_date'] = crypto_utils.get_relative_time_option(
 459                 rc['revocation_date'],
 460                 path_prefix + 'revocation_date'
 461             )
 462             if rc['reason']:
 463                 result['reason'] = crypto_utils.REVOCATION_REASON_MAP[rc['reason']]
 464                 result['reason_critical'] = rc['reason_critical']
 465             if rc['invalidity_date']:
 466                 result['invalidity_date'] = crypto_utils.get_relative_time_option(
 467                     rc['invalidity_date'],
 468                     path_prefix + 'invalidity_date'
 469                 )
 470                 result['invalidity_date_critical'] = rc['invalidity_date_critical']
 471             self.revoked_certificates.append(result)
 472 
 473         self.module = module
 474 
 475         self.backup = module.params['backup']
 476         self.backup_file = None
 477 
 478         try:
 479             self.privatekey = crypto_utils.load_privatekey(
 480                 path=self.privatekey_path,
 481                 content=self.privatekey_content,
 482                 passphrase=self.privatekey_passphrase,
 483                 backend='cryptography'
 484             )
 485         except crypto_utils.OpenSSLBadPassphraseError as exc:
 486             raise CRLError(exc)
 487 
 488         self.crl = None
 489         try:
 490             with open(self.path, 'rb') as f:
 491                 data = f.read()
 492             self.crl = x509.load_pem_x509_crl(data, default_backend())
 493             if self.return_content:
 494                 self.crl_content = data
 495         except Exception as dummy:
 496             self.crl_content = None

 503     def _compress_entry(self, entry):
 504         if self.ignore_timestamps:
 505             # Throw out revocation_date
 506             return (
 507                 entry['serial_number'],
 508                 tuple(entry['issuer']) if entry['issuer'] is not None else None,
 509                 entry['issuer_critical'],
 510                 entry['reason'],
 511                 entry['reason_critical'],
 512                 entry['invalidity_date'],
 513                 entry['invalidity_date_critical'],
 514             )
 515         else:
 516             return (
 517                 entry['serial_number'],
 518                 entry['revocation_date'],
 519                 tuple(entry['issuer']) if entry['issuer'] is not None else None,
 520                 entry['issuer_critical'],
 521                 entry['reason'],
 522                 entry['reason_critical'],
 523                 entry['invalidity_date'],
 524                 entry['invalidity_date_critical'],
 525             )

 527     def check(self, perms_required=True):
 528         """Ensure the resource is in its desired state."""
 529 
 530         state_and_perms = super(CRL, self).check(self.module, perms_required)
 531 
 532         if not state_and_perms:
 533             return False
 534 
 535         if self.crl is None:
 536             return False
 537 
 538         if self.last_update != self.crl.last_update and not self.ignore_timestamps:
 539             return False
 540         if self.next_update != self.crl.next_update and not self.ignore_timestamps:
 541             return False
 542         if self.digest.name != self.crl.signature_hash_algorithm.name:
 543             return False
 544 
 545         want_issuer = [(crypto_utils.cryptography_name_to_oid(entry[0]), entry[1]) for entry in self.issuer]
 546         if want_issuer != [(sub.oid, sub.value) for sub in self.crl.issuer]:
 547             return False
 548 
 549         old_entries = [self._compress_entry(crypto_utils.cryptography_decode_revoked_certificate(cert)) for cert in self.crl]
 550         new_entries = [self._compress_entry(cert) for cert in self.revoked_certificates]
 551         if self.update:
 552             # We don't simply use a set so that duplicate entries are treated correctly
 553             for entry in new_entries:
 554                 try:
 555                     old_entries.remove(entry)
 556                 except ValueError:
 557                     return False
 558         else:
 559             if old_entries != new_entries:
 560                 return False
 561 
 562         return True

 564     def _generate_crl(self):
 565         backend = default_backend()
 566         crl = CertificateRevocationListBuilder()
 567 
 568         try:
 569             crl = crl.issuer_name(Name([
 570                 NameAttribute(crypto_utils.cryptography_name_to_oid(entry[0]), to_text(entry[1]))
 571                 for entry in self.issuer
 572             ]))
 573         except ValueError as e:
 574             raise CRLError(e)
 575 
 576         crl = crl.last_update(self.last_update)
 577         crl = crl.next_update(self.next_update)
 578 
 579         if self.update and self.crl:
 580             new_entries = set([self._compress_entry(entry) for entry in self.revoked_certificates])
 581             for entry in self.crl:
 582                 decoded_entry = self._compress_entry(crypto_utils.cryptography_decode_revoked_certificate(entry))
 583                 if decoded_entry not in new_entries:
 584                     crl = crl.add_revoked_certificate(entry)
 585         for entry in self.revoked_certificates:
 586             revoked_cert = RevokedCertificateBuilder()
 587             revoked_cert = revoked_cert.serial_number(entry['serial_number'])
 588             revoked_cert = revoked_cert.revocation_date(entry['revocation_date'])
 589             if entry['issuer'] is not None:
 590                 revoked_cert = revoked_cert.add_extension(
 591                     x509.CertificateIssuer([
 592                         crypto_utils.cryptography_get_name(name) for name in self.entry['issuer']
 593                     ]),
 594                     entry['issuer_critical']
 595                 )
 596             if entry['reason'] is not None:
 597                 revoked_cert = revoked_cert.add_extension(
 598                     x509.CRLReason(entry['reason']),
 599                     entry['reason_critical']
 600                 )
 601             if entry['invalidity_date'] is not None:
 602                 revoked_cert = revoked_cert.add_extension(
 603                     x509.InvalidityDate(entry['invalidity_date']),
 604                     entry['invalidity_date_critical']
 605                 )
 606             crl = crl.add_revoked_certificate(revoked_cert.build(backend))
 607 
 608         self.crl = crl.sign(self.privatekey, self.digest, backend=backend)
 609         return self.crl.public_bytes(Encoding.PEM)
 610 
 611     def generate(self):
 612         if not self.check(perms_required=False) or self.force:
 613             result = self._generate_crl()
 614             if self.return_content:
 615                 self.crl_content = result
 616             if self.backup:
 617                 self.backup_file = self.module.backup_local(self.path)
 618             crypto_utils.write_file(self.module, result)
 619             self.changed = True
 620 
 621         file_args = self.module.load_file_common_arguments(self.module.params)
 622         if self.module.set_fs_attributes_if_different(file_args, False):
 623             self.changed = True

 634             'reason_critical': entry['reason_critical'],
 635             'invalidity_date':
 636                 entry['invalidity_date'].strftime(TIMESTAMP_FORMAT)
 637                 if entry['invalidity_date'] is not None else None,
 638             'invalidity_date_critical': entry['invalidity_date_critical'],
 639         }
 640 
 641     def dump(self, check_mode=False):
 642         result = {
 643             'changed': self.changed,
 644             'filename': self.path,
 645             'privatekey': self.privatekey_path,
 646             'last_update': None,
 647             'next_update': None,
 648             'digest': None,
 649             'issuer_ordered': None,
 650             'issuer': None,
 651             'revoked_certificates': [],
 652         }
 653         if self.backup_file:
 654             result['backup_file'] = self.backup_file
 655 
 656         if check_mode:
 657             result['last_update'] = self.last_update.strftime(TIMESTAMP_FORMAT)
 658             result['next_update'] = self.next_update.strftime(TIMESTAMP_FORMAT)
 659             # result['digest'] = crypto_utils.cryptography_oid_to_name(self.crl.signature_algorithm_oid)
 660             result['digest'] = self.module.params['digest']
 661             result['issuer_ordered'] = self.issuer
 662             result['issuer'] = {}
 663             for k, v in self.issuer:
 664                 result['issuer'][k] = v
 665             result['revoked_certificates'] = []
 666             for entry in self.revoked_certificates:
 667                 result['revoked_certificates'].append(self._dump_revoked(entry))
 668         elif self.crl:
 669             result['last_update'] = self.crl.last_update.strftime(TIMESTAMP_FORMAT)
 670             result['next_update'] = self.crl.next_update.strftime(TIMESTAMP_FORMAT)
 671             try:
 672                 result['digest'] = crypto_utils.cryptography_oid_to_name(self.crl.signature_algorithm_oid)
 673             except AttributeError:
 674                 # Older cryptography versions don't have signature_algorithm_oid yet
 675                 dotted = crypto_utils._obj2txt(
 676                     self.crl._backend._lib,
 677                     self.crl._backend._ffi,
 678                     self.crl._x509_crl.sig_alg.algorithm
 679                 )
 680                 oid = x509.oid.ObjectIdentifier(dotted)
 681                 result['digest'] = crypto_utils.cryptography_oid_to_name(oid)
 682             issuer = []
 683             for attribute in self.crl.issuer:
 684                 issuer.append([crypto_utils.cryptography_oid_to_name(attribute.oid), attribute.value])
 685             result['issuer_ordered'] = issuer
 686             result['issuer'] = {}
 687             for k, v in issuer:
 688                 result['issuer'][k] = v
 689             result['revoked_certificates'] = []
 690             for cert in self.crl:
 691                 entry = crypto_utils.cryptography_decode_revoked_certificate(cert)
 692                 result['revoked_certificates'].append(self._dump_revoked(entry))
 693 
 694         if self.return_content:
 695             result['crl'] = self.crl_content
 696 
 697         return result
