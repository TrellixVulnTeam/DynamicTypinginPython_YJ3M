 120 class AzureRMMariaDbDatabaseInfo(AzureRMModuleBase):
 121     def __init__(self):
 122         # define user inputs into argument

 126                 required=True

 128             server_name=dict(
 129                 type='str',

 133                 type='str'
 134             )
 135         )
 136         # store the results of the module operation
 137         self.results = dict(
 138             changed=False
 139         )
 140         self.resource_group = None
 141         self.server_name = None
 142         self.name = None
 143         super(AzureRMMariaDbDatabaseInfo, self).__init__(self.module_arg_spec, supports_tags=False)

 145     def exec_module(self, **kwargs):
 146         is_old_facts = self.module._name == 'azure_rm_mariadbdatabase_facts'
 147         if is_old_facts:
 148             self.module.deprecate("The 'azure_rm_mariadbdatabase_facts' module has been renamed to 'azure_rm_mariadbdatabase_info'",
 149                                   version='2.13', collection_name='ansible.builtin')
 150 
 151         for key in self.module_arg_spec:
 152             setattr(self, key, kwargs[key])
 153 
 154         if (self.resource_group is not None and
 155                 self.server_name is not None and
 156                 self.name is not None):
 157             self.results['databases'] = self.get()
 158         elif (self.resource_group is not None and
 159               self.server_name is not None):
 160             self.results['databases'] = self.list_by_server()
 161         return self.results
 162 
 163     def get(self):
 164         response = None
 165         results = []
 166         try:
 167             response = self.mariadb_client.databases.get(resource_group_name=self.resource_group,
 168                                                          server_name=self.server_name,
 169                                                          database_name=self.name)
 170             self.log("Response : {0}".format(response))
 171         except CloudError as e:
 172             self.log('Could not get facts for Databases.')
 173 
 174         if response is not None:
 175             results.append(self.format_item(response))
 176 
 177         return results
 178 
 179     def list_by_server(self):
 180         response = None
 181         results = []
 182         try:
 183             response = self.mariadb_client.databases.list_by_server(resource_group_name=self.resource_group,
 184                                                                     server_name=self.server_name)
 185             self.log("Response : {0}".format(response))
 186         except CloudError as e:
 187             self.fail("Error listing for server {0} - {1}".format(self.server_name, str(e)))
 188 
 189         if response is not None:
 190             for item in response:
 191                 results.append(self.format_item(item))
 192 
 193         return results
 194 
 195     def format_item(self, item):
 196         d = item.as_dict()
 197         d = {
 198             'resource_group': self.resource_group,
 199             'server_name': self.server_name,
 200             'name': d['name'],
 201             'charset': d['charset'],
 202             'collation': d['collation']
 203         }
 204         return d
