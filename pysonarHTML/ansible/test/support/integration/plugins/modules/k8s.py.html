 155 EXAMPLES = '''
 156 - name: Create a k8s namespace
 157   k8s:
 158     name: testing
 159     api_version: v1
 160     kind: Namespace
 161     state: present
 162 
 163 - name: Create a Service object from an inline definition
 164   k8s:
 165     state: present
 166     definition:
 167       apiVersion: v1
 168       kind: Service
 169       metadata:
 170         name: web
 171         namespace: testing
 172         labels:
 173           app: galaxy
 174           service: web
 175       spec:
 176         selector:
 177           app: galaxy
 178           service: web
 179         ports:
 180         - protocol: TCP
 181           targetPort: 8000
 182           name: port-8000-tcp
 183           port: 8000
 184 
 185 - name: Remove an existing Service object
 186   k8s:
 187     state: absent
 188     api_version: v1
 189     kind: Service
 190     namespace: testing
 191     name: web
 192 
 193 # Passing the object definition from a file
 194 
 195 - name: Create a Deployment by reading the definition from a local file
 196   k8s:
 197     state: present
 198     src: /testing/deployment.yml
 199 
 200 - name: >-
 201     Read definition file from the Ansible controller file system.
 202     If the definition file has been encrypted with Ansible Vault it will automatically be decrypted.
 203   k8s:
 204     state: present
 205     definition: "{{ lookup('file', '/testing/deployment.yml') | from_yaml }}"
 206 
 207 - name: Read definition file from the Ansible controller file system after Jinja templating
 208   k8s:
 209     state: present
 210     definition: "{{ lookup('template', '/testing/deployment.yml') | from_yaml }}"
 211 
 212 - name: fail on validation errors
 213   k8s:
 214     state: present
 215     definition: "{{ lookup('template', '/testing/deployment.yml')  | from_yaml }}"
 216     validate:
 217       fail_on_error: yes
 218 
 219 - name: warn on validation errors, check for unexpected properties
 220   k8s:
 221     state: present
 222     definition: "{{ lookup('template', '/testing/deployment.yml') | from_yaml }}"
 223     validate:
 224       fail_on_error: no
 225       strict: yes
