 107 def parse_vgs(data):
 108     vgs = []
 109     for line in data.splitlines():
 110         parts = line.strip().split(';')
 111         vgs.append({
 112             'name': parts[0],
 113             'pv_count': int(parts[1]),
 114             'lv_count': int(parts[2]),
 115         })
 116     return vgs

 119 def find_mapper_device_name(module, dm_device):
 120     dmsetup_cmd = module.get_bin_path('dmsetup', True)
 121     mapper_prefix = '/dev/mapper/'
 122     rc, dm_name, err = module.run_command("%s info -C --noheadings -o name %s" % (dmsetup_cmd, dm_device))
 123     if rc != 0:
 124         module.fail_json(msg="Failed executing dmsetup command.", rc=rc, err=err)
 125     mapper_device = mapper_prefix + dm_name.rstrip()
 126     return mapper_device
 127 
 128 
 129 def parse_pvs(module, data):
 130     pvs = []
 131     dm_prefix = '/dev/dm-'
 132     for line in data.splitlines():
 133         parts = line.strip().split(';')
 134         if parts[0].startswith(dm_prefix):
 135             parts[0] = find_mapper_device_name(module, parts[0])
 136         pvs.append({
 137             'name': parts[0],
 138             'vg_name': parts[1],
 139         })
 140     return pvs

 143 def main():
 144     module = AnsibleModule(
 145         argument_spec=dict(
 146             vg=dict(type='str', required=True),
 147             pvs=dict(type='list'),
 148             pesize=dict(type='str', default='4'),
 149             pv_options=dict(type='str', default=''),
 150             vg_options=dict(type='str', default=''),
 151             state=dict(type='str', default='present', choices=['absent', 'present']),
 152             force=dict(type='bool', default=False),
 153         ),
 154         supports_check_mode=True,
 155     )
 156 
 157     vg = module.params['vg']
 158     state = module.params['state']
 159     force = module.boolean(module.params['force'])
 160     pesize = module.params['pesize']
 161     pvoptions = module.params['pv_options'].split()
 162     vgoptions = module.params['vg_options'].split()
 163 
 164     dev_list = []
 165     if module.params['pvs']:
 166         dev_list = list(module.params['pvs'])
 167     elif state == 'present':
 168         module.fail_json(msg="No physical volumes given.")
 169 
 170     # LVM always uses real paths not symlinks so replace symlinks with actual path
 171     for idx, dev in enumerate(dev_list):
 172         dev_list[idx] = os.path.realpath(dev)
 173 
 174     if state == 'present':
 175         # check given devices
 176         for test_dev in dev_list:
 177             if not os.path.exists(test_dev):
 178                 module.fail_json(msg="Device %s not found." % test_dev)
 179 
 180         # get pv list
 181         pvs_cmd = module.get_bin_path('pvs', True)
 182         if dev_list:
 183             pvs_filter_pv_name = ' || '.join(
 184                 'pv_name = {0}'.format(x)
 185                 for x in itertools.chain(dev_list, module.params['pvs'])
 186             )
 187             pvs_filter_vg_name = 'vg_name = {0}'.format(vg)
 188             pvs_filter = "--select '{0} || {1}' ".format(pvs_filter_pv_name, pvs_filter_vg_name)
 189         else:
 190             pvs_filter = ''
 191         rc, current_pvs, err = module.run_command("%s --noheadings -o pv_name,vg_name --separator ';' %s" % (pvs_cmd, pvs_filter))
 192         if rc != 0:
 193             module.fail_json(msg="Failed executing pvs command.", rc=rc, err=err)
 194 
 195         # check pv for devices
 196         pvs = parse_pvs(module, current_pvs)
 197         used_pvs = [pv for pv in pvs if pv['name'] in dev_list and pv['vg_name'] and pv['vg_name'] != vg]
 198         if used_pvs:
 199             module.fail_json(msg="Device %s is already in %s volume group." % (used_pvs[0]['name'], used_pvs[0]['vg_name']))
 200 
 201     vgs_cmd = module.get_bin_path('vgs', True)
 202     rc, current_vgs, err = module.run_command("%s --noheadings -o vg_name,pv_count,lv_count --separator ';'" % vgs_cmd)
 203 
 204     if rc != 0:
 205         module.fail_json(msg="Failed executing vgs command.", rc=rc, err=err)
 206 
 207     changed = False
 208 
 209     vgs = parse_vgs(current_vgs)
 210 
 211     for test_vg in vgs:
 212         if test_vg['name'] == vg:
 213             this_vg = test_vg
 214             break
 215     else:
 216         this_vg = None
 217 
 218     if this_vg is None:
 219         if state == 'present':
 220             # create VG
 221             if module.check_mode:
 222                 changed = True
 223             else:
 224                 # create PV
 225                 pvcreate_cmd = module.get_bin_path('pvcreate', True)
 226                 for current_dev in dev_list:
 227                     rc, _, err = module.run_command([pvcreate_cmd] + pvoptions + ['-f', str(current_dev)])
 228                     if rc == 0:
 229                         changed = True
 230                     else:
 231                         module.fail_json(msg="Creating physical volume '%s' failed" % current_dev, rc=rc, err=err)
 232                 vgcreate_cmd = module.get_bin_path('vgcreate')
 233                 rc, _, err = module.run_command([vgcreate_cmd] + vgoptions + ['-s', pesize, vg] + dev_list)
 234                 if rc == 0:
 235                     changed = True
 236                 else:
 237                     module.fail_json(msg="Creating volume group '%s' failed" % vg, rc=rc, err=err)
 238     else:
 239         if state == 'absent':
 240             if module.check_mode:
 241                 module.exit_json(changed=True)
 242             else:
 243                 if this_vg['lv_count'] == 0 or force:
 244                     # remove VG
 245                     vgremove_cmd = module.get_bin_path('vgremove', True)
 246                     rc, _, err = module.run_command("%s --force %s" % (vgremove_cmd, vg))
 247                     if rc == 0:
 248                         module.exit_json(changed=True)
 249                     else:
 250                         module.fail_json(msg="Failed to remove volume group %s" % (vg), rc=rc, err=err)
 251                 else:
 252                     module.fail_json(msg="Refuse to remove non-empty volume group %s without force=yes" % (vg))
 253 
 254         # resize VG
 255         current_devs = [os.path.realpath(pv['name']) for pv in pvs if pv['vg_name'] == vg]
 256         devs_to_remove = list(set(current_devs) - set(dev_list))
 257         devs_to_add = list(set(dev_list) - set(current_devs))
 258 
 259         if devs_to_add or devs_to_remove:
 260             if module.check_mode:
 261                 changed = True
 262             else:
 263                 if devs_to_add:
 264                     devs_to_add_string = ' '.join(devs_to_add)
 265                     # create PV
 266                     pvcreate_cmd = module.get_bin_path('pvcreate', True)
 267                     for current_dev in devs_to_add:
 268                         rc, _, err = module.run_command([pvcreate_cmd] + pvoptions + ['-f', str(current_dev)])
 269                         if rc == 0:
 270                             changed = True
 271                         else:
 272                             module.fail_json(msg="Creating physical volume '%s' failed" % current_dev, rc=rc, err=err)
 273                     # add PV to our VG
 274                     vgextend_cmd = module.get_bin_path('vgextend', True)
 275                     rc, _, err = module.run_command("%s %s %s" % (vgextend_cmd, vg, devs_to_add_string))
 276                     if rc == 0:
 277                         changed = True
 278                     else:
 279                         module.fail_json(msg="Unable to extend %s by %s." % (vg, devs_to_add_string), rc=rc, err=err)
 280 
 281                 # remove some PV from our VG
 282                 if devs_to_remove:
 283                     devs_to_remove_string = ' '.join(devs_to_remove)
 284                     vgreduce_cmd = module.get_bin_path('vgreduce', True)
 285                     rc, _, err = module.run_command("%s --force %s %s" % (vgreduce_cmd, vg, devs_to_remove_string))
 286                     if rc == 0:
 287                         changed = True
 288                     else:
 289                         module.fail_json(msg="Unable to reduce %s by %s." % (vg, devs_to_remove_string), rc=rc, err=err)
 290 
 291     module.exit_json(changed=changed)
 292 
 293 
 294 if __name__ == '__main__':
 295     main()
