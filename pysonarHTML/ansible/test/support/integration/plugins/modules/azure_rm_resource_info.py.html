 336     def exec_module(self, **kwargs):
 337         is_old_facts = self.module._name == 'azure_rm_resource_facts'
 338         if is_old_facts:
 339             self.module.deprecate("The 'azure_rm_resource_facts' module has been renamed to 'azure_rm_resource_info'",
 340                                   version='2.13', collection_name='ansible.builtin')
 341 
 342         for key in self.module_arg_spec:
 343             setattr(self, key, kwargs[key])
 344         self.mgmt_client = self.get_mgmt_svc_client(GenericRestClient,
 345                                                     base_url=self._cloud_environment.endpoints.resource_manager)
 346 
 347         if self.url is None:
 348             orphan = None
 349             rargs = dict()
 350             rargs['subscription'] = self.subscription_id
 351             rargs['resource_group'] = self.resource_group
 352             if not (self.provider is None or self.provider.lower().startswith('.microsoft')):
 353                 rargs['namespace'] = "Microsoft." + self.provider
 354             else:
 355                 rargs['namespace'] = self.provider
 356 
 357             if self.resource_type is not None and self.resource_name is not None:
 358                 rargs['type'] = self.resource_type
 359                 rargs['name'] = self.resource_name
 360                 for i in range(len(self.subresource)):
 361                     resource_ns = self.subresource[i].get('namespace', None)
 362                     resource_type = self.subresource[i].get('type', None)
 363                     resource_name = self.subresource[i].get('name', None)
 364                     if resource_type is not None and resource_name is not None:
 365                         rargs['child_namespace_' + str(i + 1)] = resource_ns
 366                         rargs['child_type_' + str(i + 1)] = resource_type
 367                         rargs['child_name_' + str(i + 1)] = resource_name
 368                     else:
 369                         orphan = resource_type
 370             else:
 371                 orphan = self.resource_type
 372 
 373             self.url = resource_id(**rargs)
 374 
 375             if orphan is not None:
 376                 self.url += '/' + orphan
 377 
 378         # if api_version was not specified, get latest one
 379         if not self.api_version:
 380             try:
 381                 # extract provider and resource type
 382                 if "/providers/" in self.url:
 383                     provider = self.url.split("/providers/")[1].split("/")[0]
 384                     resourceType = self.url.split(provider + "/")[1].split("/")[0]
 385                     url = "/subscriptions/" + self.subscription_id + "/providers/" + provider
 386                     api_versions = json.loads(self.mgmt_client.query(url, "GET", {'api-version': '2015-01-01'}, None, None, [200], 0, 0).text)
 387                     for rt in api_versions['resourceTypes']:
 388                         if rt['resourceType'].lower() == resourceType.lower():
 389                             self.api_version = rt['apiVersions'][0]
 390                             break
 391                 else:
 392                     # if there's no provider in API version, assume Microsoft.Resources
 393                     self.api_version = '2018-05-01'
 394                 if not self.api_version:
 395                     self.fail("Couldn't find api version for {0}/{1}".format(provider, resourceType))
 396             except Exception as exc:
 397                 self.fail("Failed to obtain API version: {0}".format(str(exc)))
 398 
 399         self.results['url'] = self.url
 400 
 401         query_parameters = {}
 402         query_parameters['api-version'] = self.api_version
 403 
 404         header_parameters = {}
 405         header_parameters['Content-Type'] = 'application/json; charset=utf-8'
 406         skiptoken = None
 407 
 408         while True:
 409             if skiptoken:
 410                 query_parameters['skiptoken'] = skiptoken
 411             response = self.mgmt_client.query(self.url, "GET", query_parameters, header_parameters, None, [200, 404], 0, 0)
 412             try:
 413                 response = json.loads(response.text)
 414                 if isinstance(response, dict):
 415                     if response.get('value'):
 416                         self.results['response'] = self.results['response'] + response['value']
 417                         skiptoken = response.get('nextLink')
 418                     else:
 419                         self.results['response'] = self.results['response'] + [response]
 420             except Exception as e:
 421                 self.fail('Failed to parse response: ' + str(e))
 422             if not skiptoken:
 423                 break
 424         return self.results
