 231 def check_compatibility(module, client):
 232     """Check the compatibility between the driver and the database.
 233 
 234        See: https://docs.mongodb.com/ecosystem/drivers/driver-compatibility-reference/#python-driver-compatibility
 235 
 236     Args:
 237         module: Ansible module.
 238         client (cursor): Mongodb cursor on admin database.
 239     """
 240     loose_srv_version = LooseVersion(client.server_info()['version'])
 241     loose_driver_version = LooseVersion(PyMongoVersion)
 242 
 243     if loose_srv_version >= LooseVersion('3.2') and loose_driver_version < LooseVersion('3.2'):
 244         module.fail_json(msg=' (Note: you must use pymongo 3.2+ with MongoDB >= 3.2)')
 245 
 246     elif loose_srv_version >= LooseVersion('3.0') and loose_driver_version <= LooseVersion('2.8'):
 247         module.fail_json(msg=' (Note: you must use pymongo 2.8+ with MongoDB 3.0)')
 248 
 249     elif loose_srv_version >= LooseVersion('2.6') and loose_driver_version <= LooseVersion('2.7'):
 250         module.fail_json(msg=' (Note: you must use pymongo 2.7+ with MongoDB 2.6)')
 251 
 252     elif LooseVersion(PyMongoVersion) <= LooseVersion('2.5'):
 253         module.fail_json(msg=' (Note: you must be on mongodb 2.4+ and pymongo 2.5+ to use the roles param)')

 256 def user_find(client, user, db_name):
 257     """Check if the user exists.
 258 
 259     Args:
 260         client (cursor): Mongodb cursor on admin database.
 261         user (str): User to check.
 262         db_name (str): User's database.
 263 
 264     Returns:
 265         dict: when user exists, False otherwise.
 266     """
 267     for mongo_user in client["admin"].system.users.find():
 268         if mongo_user['user'] == user:
 269             # NOTE: there is no 'db' field in mongo 2.4.
 270             if 'db' not in mongo_user:
 271                 return mongo_user
 272 
 273             if mongo_user["db"] == db_name:
 274                 return mongo_user
 275     return False

 278 def user_add(module, client, db_name, user, password, roles):
 279     # pymongo's user_add is a _create_or_update_user so we won't know if it was changed or updated
 280     # without reproducing a lot of the logic in database.py of pymongo
 281     db = client[db_name]
 282 
 283     if roles is None:
 284         db.add_user(user, password, False)
 285     else:
 286         db.add_user(user, password, None, roles=roles)
 287 
 288 
 289 def user_remove(module, client, db_name, user):
 290     exists = user_find(client, user, db_name)
 291     if exists:
 292         if module.check_mode:
 293             module.exit_json(changed=True, user=user)
 294         db = client[db_name]
 295         db.remove_user(user)
 296     else:
 297         module.exit_json(changed=False, user=user)

 300 def load_mongocnf():
 301     config = configparser.RawConfigParser()
 302     mongocnf = os.path.expanduser('~/.mongodb.cnf')
 303 
 304     try:
 305         config.readfp(open(mongocnf))
 306         creds = dict(
 307             user=config.get('client', 'user'),
 308             password=config.get('client', 'pass')
 309         )
 310     except (configparser.NoOptionError, IOError):
 311         return False
 312 
 313     return creds

 316 def check_if_roles_changed(uinfo, roles, db_name):
 317     # We must be aware of users which can read the oplog on a replicaset
 318     # Such users must have access to the local DB, but since this DB does not store users credentials
 319     # and is not synchronized among replica sets, the user must be stored on the admin db
 320     # Therefore their structure is the following :
 321     # {
 322     #     "_id" : "admin.oplog_reader",
 323     #     "user" : "oplog_reader",
 324     #     "db" : "admin",                    # <-- admin DB
 325     #     "roles" : [
 326     #         {
 327     #             "role" : "read",
 328     #             "db" : "local"             # <-- local DB
 329     #         }
 330     #     ]
 331     # }
 332 
 333     def make_sure_roles_are_a_list_of_dict(roles, db_name):
 334         output = list()
 335         for role in roles:
 336             if isinstance(role, (binary_type, text_type)):
 337                 new_role = {"role": role, "db": db_name}
 338                 output.append(new_role)
 339             else:
 340                 output.append(role)
 341         return output
 342 
 343     roles_as_list_of_dict = make_sure_roles_are_a_list_of_dict(roles, db_name)
 344     uinfo_roles = uinfo.get('roles', [])
 345 
 346     if sorted(roles_as_list_of_dict, key=itemgetter('db')) == sorted(uinfo_roles, key=itemgetter('db')):
 347         return False
 348     return True

 355 def main():
 356     module = AnsibleModule(
 357         argument_spec=dict(
 358             login_user=dict(default=None),
 359             login_password=dict(default=None, no_log=True),
 360             login_host=dict(default='localhost'),
 361             login_port=dict(default='27017'),
 362             login_database=dict(default=None),
 363             replica_set=dict(default=None),
 364             database=dict(required=True, aliases=['db']),
 365             name=dict(required=True, aliases=['user']),
 366             password=dict(aliases=['pass'], no_log=True),
 367             ssl=dict(default=False, type='bool'),
 368             roles=dict(default=None, type='list', elements='raw'),
 369             state=dict(default='present', choices=['absent', 'present']),
 370             update_password=dict(default="always", choices=["always", "on_create"]),
 371             ssl_cert_reqs=dict(default='CERT_REQUIRED', choices=['CERT_NONE', 'CERT_OPTIONAL', 'CERT_REQUIRED']),
 372         ),
 373         supports_check_mode=True
 374     )
 375 
 376     if not pymongo_found:
 377         module.fail_json(msg=missing_required_lib('pymongo'))
 378 
 379     login_user = module.params['login_user']
 380     login_password = module.params['login_password']
 381     login_host = module.params['login_host']
 382     login_port = module.params['login_port']
 383     login_database = module.params['login_database']
 384 
 385     replica_set = module.params['replica_set']
 386     db_name = module.params['database']
 387     user = module.params['name']
 388     password = module.params['password']
 389     ssl = module.params['ssl']
 390     roles = module.params['roles'] or []
 391     state = module.params['state']
 392     update_password = module.params['update_password']
 393 
 394     try:
 395         connection_params = {
 396             "host": login_host,
 397             "port": int(login_port),
 398         }
 399 
 400         if replica_set:
 401             connection_params["replicaset"] = replica_set
 402 
 403         if ssl:
 404             connection_params["ssl"] = ssl
 405             connection_params["ssl_cert_reqs"] = getattr(ssl_lib, module.params['ssl_cert_reqs'])
 406 
 407         client = MongoClient(**connection_params)
 408 
 409         # NOTE: this check must be done ASAP.
 410         # We doesn't need to be authenticated (this ability has lost in PyMongo 3.6)
 411         if LooseVersion(PyMongoVersion) <= LooseVersion('3.5'):
 412             check_compatibility(module, client)
 413 
 414         if login_user is None and login_password is None:
 415             mongocnf_creds = load_mongocnf()
 416             if mongocnf_creds is not False:
 417                 login_user = mongocnf_creds['user']
 418                 login_password = mongocnf_creds['password']
 419         elif login_password is None or login_user is None:
 420             module.fail_json(msg='when supplying login arguments, both login_user and login_password must be provided')
 421 
 422         if login_user is not None and login_password is not None:
 423             client.admin.authenticate(login_user, login_password, source=login_database)
 424         elif LooseVersion(PyMongoVersion) >= LooseVersion('3.0'):
 425             if db_name != "admin":
 426                 module.fail_json(msg='The localhost login exception only allows the first admin account to be created')
 427             # else: this has to be the first admin user added
 428 
 429     except Exception as e:
 430         module.fail_json(msg='unable to connect to database: %s' % to_native(e), exception=traceback.format_exc())
 431 
 432     if state == 'present':
 433         if password is None and update_password == 'always':
 434             module.fail_json(msg='password parameter required when adding a user unless update_password is set to on_create')
 435 
 436         try:
 437             if update_password != 'always':
 438                 uinfo = user_find(client, user, db_name)
 439                 if uinfo:
 440                     password = None
 441                     if not check_if_roles_changed(uinfo, roles, db_name):
 442                         module.exit_json(changed=False, user=user)
 443 
 444             if module.check_mode:
 445                 module.exit_json(changed=True, user=user)
 446 
 447             user_add(module, client, db_name, user, password, roles)
 448         except Exception as e:
 449             module.fail_json(msg='Unable to add or update user: %s' % to_native(e), exception=traceback.format_exc())
 450         finally:
 451             try:
 452                 client.close()
 453             except Exception:
 454                 pass
 455             # Here we can  check password change if mongo provide a query for that : https://jira.mongodb.org/browse/SERVER-22848
 456             # newuinfo = user_find(client, user, db_name)
 457             # if uinfo['role'] == newuinfo['role'] and CheckPasswordHere:
 458             #    module.exit_json(changed=False, user=user)
 459 
 460     elif state == 'absent':
 461         try:
 462             user_remove(module, client, db_name, user)
 463         except Exception as e:
 464             module.fail_json(msg='Unable to remove user: %s' % to_native(e), exception=traceback.format_exc())
 465         finally:
 466             try:
 467                 client.close()
 468             except Exception:
 469                 pass
 470     module.exit_json(changed=True, user=user)
 471

 473 if __name__ == '__main__':
 474     main()
