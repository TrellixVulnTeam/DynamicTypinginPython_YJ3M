 150 
 151         self.exit_json(changed=False,
 152                        **self.kubernetes_facts(self.params['kind'],
 153                                                self.params['api_version'],
 154                                                self.params['name'],
 155                                                self.params['namespace'],
 156                                                self.params['label_selectors'],
 157                                                self.params['field_selectors']))

 166                 name=dict(),
