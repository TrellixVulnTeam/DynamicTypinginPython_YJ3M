 350 _IDENT = r"[a-zA-Z-][a-zA-Z0-9_\-\.]*"
 351 _NSIDENT = _IDENT + "|" + _IDENT + ":" + _IDENT
 352 # Note: we can't reasonably support the 'if you need to put both ' and " in a string, concatenate
 353 # strings wrapped by the other delimiter' XPath trick, especially as simple XPath.
 354 _XPSTR = "('(?:.*)'|\"(?:.*)\")"
 355 
 356 _RE_SPLITSIMPLELAST = re.compile("^(.*)/(" + _NSIDENT + ")$")
 357 _RE_SPLITSIMPLELASTEQVALUE = re.compile("^(.*)/(" + _NSIDENT + ")/text\\(\\)=" + _XPSTR + "$")
 358 _RE_SPLITSIMPLEATTRLAST = re.compile("^(.*)/(@(?:" + _NSIDENT + "))$")
 359 _RE_SPLITSIMPLEATTRLASTEQVALUE = re.compile("^(.*)/(@(?:" + _NSIDENT + "))=" + _XPSTR + "$")
 360 _RE_SPLITSUBLAST = re.compile("^(.*)/(" + _NSIDENT + ")\\[(.*)\\]$")
 361 _RE_SPLITONLYEQVALUE = re.compile("^(.*)/text\\(\\)=" + _XPSTR + "$")

 387 def is_node(tree, xpath, namespaces):
 388     """ Test if a given xpath matches anything and if that match is a node.
 389 
 390     For now we just assume you're only searching for one specific thing."""
 391     if xpath_matches(tree, xpath, namespaces):
 392         # OK, it found something
 393         match = tree.xpath(xpath, namespaces=namespaces)
 394         if isinstance(match[0], etree._Element):
 395             return True
 396 
 397     return False

 400 def is_attribute(tree, xpath, namespaces):
 401     """ Test if a given xpath matches and that match is an attribute
 402 
 403     An xpath attribute search will only match one item"""
 404     if xpath_matches(tree, xpath, namespaces):
 405         match = tree.xpath(xpath, namespaces=namespaces)
 406         if isinstance(match[0], etree._ElementStringResult):
 407             return True
 408         elif isinstance(match[0], etree._ElementUnicodeResult):
 409             return True
 410     return False

 420 def delete_xpath_target(module, tree, xpath, namespaces):
 421     """ Delete an attribute or element from a tree """
 422     try:
 423         for result in tree.xpath(xpath, namespaces=namespaces):
 424             # Get the xpath for this result
 425             if is_attribute(tree, xpath, namespaces):
 426                 # Delete an attribute
 427                 parent = result.getparent()
 428                 # Pop this attribute match out of the parent
 429                 # node's 'attrib' dict by using this match's
 430                 # 'attrname' attribute for the key
 431                 parent.attrib.pop(result.attrname)
 432             elif is_node(tree, xpath, namespaces):
 433                 # Delete an element
 434                 result.getparent().remove(result)
 435             else:
 436                 raise Exception("Impossible error")
 437     except Exception as e:
 438         module.fail_json(msg="Couldn't delete xpath target: %s (%s)" % (xpath, e))
 439     else:
 440         finish(module, tree, xpath, namespaces, changed=True)

 449 def set_target_children_inner(module, tree, xpath, namespaces, children, in_type):
 450     matches = tree.xpath(xpath, namespaces=namespaces)
 451 
 452     # Create a list of our new children
 453     children = children_to_nodes(module, children, in_type)
 454     children_as_string = [etree.tostring(c) for c in children]
 455 
 456     changed = False
 457 
 458     # xpaths always return matches as a list, so....
 459     for match in matches:
 460         # Check if elements differ
 461         if len(list(match)) == len(children):
 462             for idx, element in enumerate(list(match)):
 463                 if etree.tostring(element) != children_as_string[idx]:
 464                     replace_children_of(children, match)
 465                     changed = True
 466                     break
 467         else:
 468             replace_children_of(children, match)
 469             changed = True
 470 
 471     return changed

 480 def add_target_children(module, tree, xpath, namespaces, children, in_type, insertbefore, insertafter):
 481     if is_node(tree, xpath, namespaces):
 482         new_kids = children_to_nodes(module, children, in_type)
 483         if insertbefore or insertafter:
 484             insert_target_children(tree, xpath, namespaces, new_kids, insertbefore, insertafter)
 485         else:
 486             for node in tree.xpath(xpath, namespaces=namespaces):
 487                 node.extend(new_kids)
 488         finish(module, tree, xpath, namespaces, changed=True)
 489     else:
 490         finish(module, tree, xpath, namespaces)

 493 def insert_target_children(tree, xpath, namespaces, children, insertbefore, insertafter):
 494     """
 495     Insert the given children before or after the given xpath. If insertbefore is True, it is inserted before the
 496     first xpath hit, with insertafter, it is inserted after the last xpath hit.
 497     """
 498     insert_target = tree.xpath(xpath, namespaces=namespaces)
 499     loc_index = 0 if insertbefore else -1
 500     index_in_parent = insert_target[loc_index].getparent().index(insert_target[loc_index])
 501     parent = insert_target[0].getparent()
 502     if insertafter:
 503         index_in_parent += 1
 504     for child in children:
 505         parent.insert(index_in_parent, child)
 506         index_in_parent += 1

 513 def split_xpath_last(xpath):
 514     """split an XPath of the form /foo/bar/baz into /foo/bar and baz"""
 515     xpath = xpath.strip()
 516     m = _RE_SPLITSIMPLELAST.match(xpath)
 517     if m:
 518         # requesting an element to exist
 519         return (m.group(1), [(m.group(2), None)])
 520     m = _RE_SPLITSIMPLELASTEQVALUE.match(xpath)
 521     if m:
 522         # requesting an element to exist with an inner text
 523         return (m.group(1), [(m.group(2), _extract_xpstr(m.group(3)))])
 524 
 525     m = _RE_SPLITSIMPLEATTRLAST.match(xpath)
 526     if m:
 527         # requesting an attribute to exist
 528         return (m.group(1), [(m.group(2), None)])
 529     m = _RE_SPLITSIMPLEATTRLASTEQVALUE.match(xpath)
 530     if m:
 531         # requesting an attribute to exist with a value
 532         return (m.group(1), [(m.group(2), _extract_xpstr(m.group(3)))])
 533 
 534     m = _RE_SPLITSUBLAST.match(xpath)
 535     if m:
 536         content = [x.strip() for x in m.group(3).split(" and ")]
 537         return (m.group(1), [('/' + m.group(2), content)])
 538 
 539     m = _RE_SPLITONLYEQVALUE.match(xpath)
 540     if m:
 541         # requesting a change of inner text
 542         return (m.group(1), [("", _extract_xpstr(m.group(2)))])
 543     return (xpath, [])

 546 def nsnameToClark(name, namespaces):
 547     if ":" in name:
 548         (nsname, rawname) = name.split(":")
 549         # return "{{%s}}%s" % (namespaces[nsname], rawname)
 550         return "{{{0}}}{1}".format(namespaces[nsname], rawname)
 551 
 552     # no namespace name here
 553     return name

 556 def check_or_make_target(module, tree, xpath, namespaces):
 557     (inner_xpath, changes) = split_xpath_last(xpath)
 558     if (inner_xpath == xpath) or (changes is None):
 559         module.fail_json(msg="Can't process Xpath %s in order to spawn nodes! tree is %s" %
 560                              (xpath, etree.tostring(tree, pretty_print=True)))
 561         return False
 562 
 563     changed = False
 564 
 565     if not is_node(tree, inner_xpath, namespaces):
 566         changed = check_or_make_target(module, tree, inner_xpath, namespaces)
 567 
 568     # we test again after calling check_or_make_target
 569     if is_node(tree, inner_xpath, namespaces) and changes:
 570         for (eoa, eoa_value) in changes:
 571             if eoa and eoa[0] != '@' and eoa[0] != '/':
 572                 # implicitly creating an element
 573                 new_kids = children_to_nodes(module, [nsnameToClark(eoa, namespaces)], "yaml")
 574                 if eoa_value:
 575                     for nk in new_kids:
 576                         nk.text = eoa_value
 577 
 578                 for node in tree.xpath(inner_xpath, namespaces=namespaces):
 579                     node.extend(new_kids)
 580                     changed = True
 581                 # module.fail_json(msg="now tree=%s" % etree.tostring(tree, pretty_print=True))
 582             elif eoa and eoa[0] == '/':
 583                 element = eoa[1:]
 584                 new_kids = children_to_nodes(module, [nsnameToClark(element, namespaces)], "yaml")
 585                 for node in tree.xpath(inner_xpath, namespaces=namespaces):
 586                     node.extend(new_kids)
 587                     for nk in new_kids:
 588                         for subexpr in eoa_value:
 589                             # module.fail_json(msg="element=%s subexpr=%s node=%s now tree=%s" %
 590                             #                      (element, subexpr, etree.tostring(node, pretty_print=True), etree.tostring(tree, pretty_print=True))
 591                             check_or_make_target(module, nk, "./" + subexpr, namespaces)
 592                     changed = True
 593 
 594                 # module.fail_json(msg="now tree=%s" % etree.tostring(tree, pretty_print=True))
 595             elif eoa == "":
 596                 for node in tree.xpath(inner_xpath, namespaces=namespaces):
 597                     if (node.text != eoa_value):
 598                         node.text = eoa_value
 599                         changed = True
 600 
 601             elif eoa and eoa[0] == '@':
 602                 attribute = nsnameToClark(eoa[1:], namespaces)
 603 
 604                 for element in tree.xpath(inner_xpath, namespaces=namespaces):
 605                     changing = (attribute not in element.attrib or element.attrib[attribute] != eoa_value)
 606 
 607                     if changing:
 608                         changed = changed or changing
 609                         if eoa_value is None:
 610                             value = ""
 611                         else:
 612                             value = eoa_value
 613                         element.attrib[attribute] = value
 614 
 615                     # module.fail_json(msg="arf %s changing=%s as curval=%s changed tree=%s" %
 616                     #       (xpath, changing, etree.tostring(tree, changing, element[attribute], pretty_print=True)))
 617 
 618             else:
 619                 module.fail_json(msg="unknown tree transformation=%s" % etree.tostring(tree, pretty_print=True))
 620 
 621     return changed

 624 def ensure_xpath_exists(module, tree, xpath, namespaces):
 625     changed = False
 626 
 627     if not is_node(tree, xpath, namespaces):
 628         changed = check_or_make_target(module, tree, xpath, namespaces)
 629 
 630     finish(module, tree, xpath, namespaces, changed)

 633 def set_target_inner(module, tree, xpath, namespaces, attribute, value):
 634     changed = False
 635 
 636     try:
 637         if not is_node(tree, xpath, namespaces):
 638             changed = check_or_make_target(module, tree, xpath, namespaces)
 639     except Exception as e:
 640         missing_namespace = ""
 641         # NOTE: This checks only the namespaces defined in root element!
 642         # TODO: Implement a more robust check to check for child namespaces' existence
 643         if tree.getroot().nsmap and ":" not in xpath:
 644             missing_namespace = "XML document has namespace(s) defined, but no namespace prefix(es) used in xpath!\n"
 645         module.fail_json(msg="%sXpath %s causes a failure: %s\n  -- tree is %s" %
 646                              (missing_namespace, xpath, e, etree.tostring(tree, pretty_print=True)), exception=traceback.format_exc())
 647 
 648     if not is_node(tree, xpath, namespaces):
 649         module.fail_json(msg="Xpath %s does not reference a node! tree is %s" %
 650                              (xpath, etree.tostring(tree, pretty_print=True)))
 651 
 652     for element in tree.xpath(xpath, namespaces=namespaces):
 653         if not attribute:
 654             changed = changed or (element.text != value)
 655             if element.text != value:
 656                 element.text = value
 657         else:
 658             changed = changed or (element.get(attribute) != value)
 659             if ":" in attribute:
 660                 attr_ns, attr_name = attribute.split(":")
 661                 # attribute = "{{%s}}%s" % (namespaces[attr_ns], attr_name)
 662                 attribute = "{{{0}}}{1}".format(namespaces[attr_ns], attr_name)
 663             if element.get(attribute) != value:
 664                 element.set(attribute, value)
 665 
 666     return changed

 670     changed = set_target_inner(module, tree, xpath, namespaces, attribute, value)
 671     finish(module, tree, xpath, namespaces, changed)

 674 def get_element_text(module, tree, xpath, namespaces):
 675     if not is_node(tree, xpath, namespaces):
 676         module.fail_json(msg="Xpath %s does not reference a node!" % xpath)
 677 
 678     elements = []
 679     for element in tree.xpath(xpath, namespaces=namespaces):
 680         elements.append({element.tag: element.text})
 681 
 682     finish(module, tree, xpath, namespaces, changed=False, msg=len(elements), hitcount=len(elements), matches=elements)

 685 def get_element_attr(module, tree, xpath, namespaces):
 686     if not is_node(tree, xpath, namespaces):
 687         module.fail_json(msg="Xpath %s does not reference a node!" % xpath)
 688 
 689     elements = []
 690     for element in tree.xpath(xpath, namespaces=namespaces):
 691         child = {}
 692         for key in element.keys():
 693             value = element.get(key)
 694             child.update({key: value})
 695         elements.append({element.tag: child})
 696 
 697     finish(module, tree, xpath, namespaces, changed=False, msg=len(elements), hitcount=len(elements), matches=elements)

 700 def child_to_element(module, child, in_type):
 701     if in_type == 'xml':
 702         infile = BytesIO(to_bytes(child, errors='surrogate_or_strict'))
 703 
 704         try:
 705             parser = etree.XMLParser()
 706             node = etree.parse(infile, parser)
 707             return node.getroot()
 708         except etree.XMLSyntaxError as e:
 709             module.fail_json(msg="Error while parsing child element: %s" % e)
 710     elif in_type == 'yaml':
 711         if isinstance(child, string_types):
 712             return etree.Element(child)
 713         elif isinstance(child, MutableMapping):
 714             if len(child) > 1:
 715                 module.fail_json(msg="Can only create children from hashes with one key")
 716 
 717             (key, value) = next(iteritems(child))
 718             if isinstance(value, MutableMapping):
 719                 children = value.pop('_', None)
 720 
 721                 node = etree.Element(key, value)
 722 
 723                 if children is not None:
 724                     if not isinstance(children, list):
 725                         module.fail_json(msg="Invalid children type: %s, must be list." % type(children))
 726 
 727                     subnodes = children_to_nodes(module, children)
 728                     node.extend(subnodes)
 729             else:
 730                 node = etree.Element(key)
 731                 node.text = value
 732             return node
 733         else:
 734             module.fail_json(msg="Invalid child type: %s. Children must be either strings or hashes." % type(child))
 735     else:
 736         module.fail_json(msg="Invalid child input type: %s. Type must be either xml or yaml." % in_type)

 739 def children_to_nodes(module=None, children=None, type='yaml'):
 740     """turn a str/hash/list of str&hash into a list of elements"""
 741     children = [] if children is None else children
 742 
 743     return [child_to_element(module, child, type) for child in children]

 746 def make_pretty(module, tree):
 747     xml_string = etree.tostring(tree, xml_declaration=True, encoding='UTF-8', pretty_print=module.params['pretty_print'])
 748 
 749     result = dict(
 750         changed=False,
 751     )
 752 
 753     if module.params['path']:
 754         xml_file = module.params['path']
 755         with open(xml_file, 'rb') as xml_content:
 756             if xml_string != xml_content.read():
 757                 result['changed'] = True
 758                 if not module.check_mode:
 759                     if module.params['backup']:
 760                         result['backup_file'] = module.backup_local(module.params['path'])
 761                     tree.write(xml_file, xml_declaration=True, encoding='UTF-8', pretty_print=module.params['pretty_print'])
 762 
 763     elif module.params['xmlstring']:
 764         result['xmlstring'] = xml_string
 765         # NOTE: Modifying a string is not considered a change !
 766         if xml_string != module.params['xmlstring']:
 767             result['changed'] = True
 768 
 769     module.exit_json(**result)

 772 def finish(module, tree, xpath, namespaces, changed=False, msg='', hitcount=0, matches=tuple()):
 773 
 774     result = dict(
 775         actions=dict(
 776             xpath=xpath,
 777             namespaces=namespaces,
 778             state=module.params['state']
 779         ),
 780         changed=has_changed(tree),
 781     )
 782 
 783     if module.params['count'] or hitcount:
 784         result['count'] = hitcount
 785 
 786     if module.params['print_match'] or matches:
 787         result['matches'] = matches
 788 
 789     if msg:
 790         result['msg'] = msg
 791 
 792     if result['changed']:
 793         if module._diff:
 794             result['diff'] = dict(
 795                 before=etree.tostring(orig_doc, xml_declaration=True, encoding='UTF-8', pretty_print=True),
 796                 after=etree.tostring(tree, xml_declaration=True, encoding='UTF-8', pretty_print=True),
 797             )
 798 
 799         if module.params['path'] and not module.check_mode:
 800             if module.params['backup']:
 801                 result['backup_file'] = module.backup_local(module.params['path'])
 802             tree.write(module.params['path'], xml_declaration=True, encoding='UTF-8', pretty_print=module.params['pretty_print'])
 803 
 804     if module.params['xmlstring']:
 805         result['xmlstring'] = etree.tostring(tree, xml_declaration=True, encoding='UTF-8', pretty_print=module.params['pretty_print'])
 806 
 807     module.exit_json(**result)

 857 
 858     xml_file = module.params['path']
 859     xml_string = module.params['xmlstring']
 860     xpath = module.params['xpath']
 861     namespaces = module.params['namespaces']
 862     state = module.params['state']
 863     value = json_dict_bytes_to_unicode(module.params['value'])
 864     attribute = module.params['attribute']
 865     set_children = json_dict_bytes_to_unicode(module.params['set_children'])
 866     add_children = json_dict_bytes_to_unicode(module.params['add_children'])
 867     pretty_print = module.params['pretty_print']
 868     content = module.params['content']
 869     input_type = module.params['input_type']
 870     print_match = module.params['print_match']
 871     count = module.params['count']
 872     backup = module.params['backup']
 873     strip_cdata_tags = module.params['strip_cdata_tags']
 874     insertbefore = module.params['insertbefore']
 875     insertafter = module.params['insertafter']
 876 
 877     # Check if we have lxml 2.3.0 or newer installed
 878     if not HAS_LXML:
 879         module.fail_json(msg=missing_required_lib("lxml"), exception=LXML_IMP_ERR)
 880     elif LooseVersion('.'.join(to_native(f) for f in etree.LXML_VERSION)) < LooseVersion('2.3.0'):
 881         module.fail_json(msg='The xml ansible module requires lxml 2.3.0 or newer installed on the managed machine')
 882     elif LooseVersion('.'.join(to_native(f) for f in etree.LXML_VERSION)) < LooseVersion('3.0.0'):
 883         module.warn('Using lxml version lower than 3.0.0 does not guarantee predictable element attribute order.')
 884 
 885     # Report wrongly used attribute parameter when using content=attribute
 886     # TODO: Remove this in Ansible v2.12 (and reinstate strict parameter test above) and remove the integration test example
 887     if content == 'attribute' and attribute is not None:
 888         module.deprecate("Parameter 'attribute=%s' is ignored when using 'content=attribute' only 'xpath' is used. Please remove entry." % attribute,
 889                          '2.12', collection_name='ansible.builtin')
 890 
 891     # Check if the file exists
 892     if xml_string:
 893         infile = BytesIO(to_bytes(xml_string, errors='surrogate_or_strict'))
 894     elif os.path.isfile(xml_file):
 895         infile = open(xml_file, 'rb')
 896     else:
 897         module.fail_json(msg="The target XML source '%s' does not exist." % xml_file)
 898

 900     if xpath is not None:
 901         try:
 902             etree.XPath(xpath)
 903         except etree.XPathSyntaxError as e:

 905         except etree.XPathEvalError as e:
 906             module.fail_json(msg="Evaluation error in xpath expression: %s (%s)" % (xpath, e))
 907 
 908     # Try to parse in the target XML file
 909     try:
 910         parser = etree.XMLParser(remove_blank_text=pretty_print, strip_cdata=strip_cdata_tags)
 911         doc = etree.parse(infile, parser)
 912     except etree.XMLSyntaxError as e:
 913         module.fail_json(msg="Error while parsing document: %s (%s)" % (xml_file or 'xml_string', e))
 914 
 915     # Ensure we have the original copy to compare
 916     global orig_doc
 917     orig_doc = copy.deepcopy(doc)
 918 
 919     if print_match:
 920         do_print_match(module, doc, xpath, namespaces)
 921 
 922     if count:
 923         count_nodes(module, doc, xpath, namespaces)
 924 
 925     if content == 'attribute':
 926         get_element_attr(module, doc, xpath, namespaces)
 927     elif content == 'text':
 928         get_element_text(module, doc, xpath, namespaces)
 929 
 930     # File exists:
 931     if state == 'absent':
 932         # - absent: delete xpath target
 933         delete_xpath_target(module, doc, xpath, namespaces)
 934 
 935     # - present: carry on
 936 
 937     # children && value both set?: should have already aborted by now
 938     # add_children && set_children both set?: should have already aborted by now
 939 
 940     # set_children set?
 941     if set_children:
 942         set_target_children(module, doc, xpath, namespaces, set_children, input_type)
 943 
 944     # add_children set?
 945     if add_children:
 946         add_target_children(module, doc, xpath, namespaces, add_children, input_type, insertbefore, insertafter)
 947 
 948     # No?: Carry on
 949 
 950     # Is the xpath target an attribute selector?
 951     if value is not None:
 952         set_target(module, doc, xpath, namespaces, attribute, value)
 953 
 954     # If an xpath was provided, we need to do something with the data
 955     if xpath is not None:
 956         ensure_xpath_exists(module, doc, xpath, namespaces)
 957 
 958     # Otherwise only reformat the xml data?
 959     if pretty_print:
 960         make_pretty(module, doc)
 961 
 962     module.fail_json(msg="Don't know what to do")
 963 
 964 
 965 if __name__ == '__main__':
 966     main()
