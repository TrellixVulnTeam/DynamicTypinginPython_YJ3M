 134 from ansible.module_utils import crypto as crypto_utils
 135 from ansible.module_utils._text import to_native

 138 MINIMAL_CRYPTOGRAPHY_VERSION = '1.2'
 139 
 140 CRYPTOGRAPHY_IMP_ERR = None
 141 try:
 142     import cryptography
 143     from cryptography import x509
 144     from cryptography.hazmat.backends import default_backend
 145     CRYPTOGRAPHY_VERSION = LooseVersion(cryptography.__version__)
 146 except ImportError:
 147     CRYPTOGRAPHY_IMP_ERR = traceback.format_exc()
 148     CRYPTOGRAPHY_FOUND = False
 149 else:
 150     CRYPTOGRAPHY_FOUND = True
 151 
 152 
 153 TIMESTAMP_FORMAT = "%Y%m%d%H%M%SZ"
 154 
 155 
 156 class CRLError(crypto_utils.OpenSSLObjectError):
 157     pass
 158 
 159 
 160 class CRLInfo(crypto_utils.OpenSSLObject):
 161     """The main module implementation."""
 162 
 163     def __init__(self, module):
 164         super(CRLInfo, self).__init__(
 165             module.params['path'] or '',
 166             'present',
 167             False,
 168             module.check_mode
 169         )
 170 
 171         self.content = module.params['content']
 172 
 173         self.module = module
 174 
 175         self.crl = None
 176         if self.content is None:
 177             try:
 178                 with open(self.path, 'rb') as f:
 179                     data = f.read()
 180             except Exception as e:
 181                 self.module.fail_json(msg='Error while reading CRL file from disk: {0}'.format(e))
 182         else:
 183             data = self.content.encode('utf-8')
 184 
 185         try:
 186             self.crl = x509.load_pem_x509_crl(data, default_backend())
 187         except Exception as e:
 188             self.module.fail_json(msg='Error while decoding CRL: {0}'.format(e))
 189 
 190     def _dump_revoked(self, entry):
 191         return {
 192             'serial_number': entry['serial_number'],
 193             'revocation_date': entry['revocation_date'].strftime(TIMESTAMP_FORMAT),
 194             'issuer':
 195                 [crypto_utils.cryptography_decode_name(issuer) for issuer in entry['issuer']]
 196                 if entry['issuer'] is not None else None,
 197             'issuer_critical': entry['issuer_critical'],
 198             'reason': crypto_utils.REVOCATION_REASON_MAP_INVERSE.get(entry['reason']) if entry['reason'] is not None else None,
 199             'reason_critical': entry['reason_critical'],
 200             'invalidity_date':
 201                 entry['invalidity_date'].strftime(TIMESTAMP_FORMAT)
 202                 if entry['invalidity_date'] is not None else None,
 203             'invalidity_date_critical': entry['invalidity_date_critical'],
 204         }
 205 
 206     def get_info(self):
 207         result = {
 208             'changed': False,
 209             'last_update': None,
 210             'next_update': None,
 211             'digest': None,
 212             'issuer_ordered': None,
 213             'issuer': None,
 214             'revoked_certificates': [],
 215         }
 216 
 217         result['last_update'] = self.crl.last_update.strftime(TIMESTAMP_FORMAT)
 218         result['next_update'] = self.crl.next_update.strftime(TIMESTAMP_FORMAT)
 219         try:
 220             result['digest'] = crypto_utils.cryptography_oid_to_name(self.crl.signature_algorithm_oid)
 221         except AttributeError:
 222             # Older cryptography versions don't have signature_algorithm_oid yet
 223             dotted = crypto_utils._obj2txt(
 224                 self.crl._backend._lib,
 225                 self.crl._backend._ffi,
 226                 self.crl._x509_crl.sig_alg.algorithm
 227             )
 228             oid = x509.oid.ObjectIdentifier(dotted)
 229             result['digest'] = crypto_utils.cryptography_oid_to_name(oid)
 230         issuer = []
 231         for attribute in self.crl.issuer:
 232             issuer.append([crypto_utils.cryptography_oid_to_name(attribute.oid), attribute.value])
 233         result['issuer_ordered'] = issuer
 234         result['issuer'] = {}
 235         for k, v in issuer:
 236             result['issuer'][k] = v
 237         result['revoked_certificates'] = []
 238         for cert in self.crl:
 239             entry = crypto_utils.cryptography_decode_revoked_certificate(cert)
 240             result['revoked_certificates'].append(self._dump_revoked(entry))
 241 
 242         return result

 253 def main():
 254     module = AnsibleModule(
 255         argument_spec=dict(
 256             path=dict(type='path'),
 257             content=dict(type='str'),
 258         ),
 259         required_one_of=(
 260             ['path', 'content'],
 261         ),
 262         mutually_exclusive=(
 263             ['path', 'content'],
 264         ),
 265         supports_check_mode=True,
 266     )
 267 
 268     if not CRYPTOGRAPHY_FOUND:
 269         module.fail_json(msg=missing_required_lib('cryptography >= {0}'.format(MINIMAL_CRYPTOGRAPHY_VERSION)),
 270                          exception=CRYPTOGRAPHY_IMP_ERR)
 271 
 272     try:
 273         crl = CRLInfo(module)
 274         result = crl.get_info()
 275         module.exit_json(**result)
 276     except crypto_utils.OpenSSLObjectError as e:
 277         module.fail_json(msg=to_native(e))

 280 if __name__ == "__main__":
 281     main()
