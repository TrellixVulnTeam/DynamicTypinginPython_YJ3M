 143 def execute(module):
 144 
 145     state = module.params.get('state')
 146 
 147     if state in ['info', 'list', 'node']:
 148         lookup_sessions(module)
 149     elif state == 'present':
 150         update_session(module)
 151     else:
 152         remove_session(module)
 153 
 154 
 155 def lookup_sessions(module):
 156 
 157     datacenter = module.params.get('datacenter')
 158 
 159     state = module.params.get('state')
 160     consul_client = get_consul_api(module)
 161     try:
 162         if state == 'list':
 163             sessions_list = consul_client.session.list(dc=datacenter)
 164             # Ditch the index, this can be grabbed from the results
 165             if sessions_list and len(sessions_list) >= 2:
 166                 sessions_list = sessions_list[1]
 167             module.exit_json(changed=True,
 168                              sessions=sessions_list)
 169         elif state == 'node':
 170             node = module.params.get('node')
 171             sessions = consul_client.session.node(node, dc=datacenter)
 172             module.exit_json(changed=True,
 173                              node=node,
 174                              sessions=sessions)
 175         elif state == 'info':
 176             session_id = module.params.get('id')
 177 
 178             session_by_id = consul_client.session.info(session_id, dc=datacenter)
 179             module.exit_json(changed=True,
 180                              session_id=session_id,
 181                              sessions=session_by_id)
 182 
 183     except Exception as e:
 184         module.fail_json(msg="Could not retrieve session info %s" % e)

 187 def update_session(module):
 188 
 189     name = module.params.get('name')
 190     delay = module.params.get('delay')
 191     checks = module.params.get('checks')
 192     datacenter = module.params.get('datacenter')
 193     node = module.params.get('node')
 194     behavior = module.params.get('behavior')
 195 
 196     consul_client = get_consul_api(module)
 197 
 198     try:
 199         session = consul_client.session.create(
 200             name=name,
 201             behavior=behavior,
 202             node=node,
 203             lock_delay=delay,
 204             dc=datacenter,
 205             checks=checks
 206         )
 207         module.exit_json(changed=True,
 208                          session_id=session,
 209                          name=name,
 210                          behavior=behavior,
 211                          delay=delay,
 212                          checks=checks,
 213                          node=node)
 214     except Exception as e:
 215         module.fail_json(msg="Could not create/update session %s" % e)
 216 
 217 
 218 def remove_session(module):
 219     session_id = module.params.get('id')
 220 
 221     consul_client = get_consul_api(module)
 222 
 223     try:
 224         consul_client.session.destroy(session_id)
 225 
 226         module.exit_json(changed=True,
 227                          session_id=session_id)
 228     except Exception as e:
 229         module.fail_json(msg="Could not remove session with id '%s' %s" % (
 230                          session_id, e))

 233 def get_consul_api(module):
 234     return consul.Consul(host=module.params.get('host'),
 235                          port=module.params.get('port'),
 236                          scheme=module.params.get('scheme'),
 237                          verify=module.params.get('validate_certs'))
