 269 class AnsibleHcloudServer(Hcloud):
 270     def __init__(self, module):
 271         Hcloud.__init__(self, module, "hcloud_server")
 272         self.hcloud_server = None
 273 
 274     def _prepare_result(self):
 275         image = None if self.hcloud_server.image is None else to_native(self.hcloud_server.image.name)
 276         return {
 277             "id": to_native(self.hcloud_server.id),
 278             "name": to_native(self.hcloud_server.name),
 279             "ipv4_address": to_native(self.hcloud_server.public_net.ipv4.ip),
 280             "ipv6": to_native(self.hcloud_server.public_net.ipv6.ip),
 281             "image": image,
 282             "server_type": to_native(self.hcloud_server.server_type.name),
 283             "datacenter": to_native(self.hcloud_server.datacenter.name),
 284             "location": to_native(self.hcloud_server.datacenter.location.name),
 285             "rescue_enabled": self.hcloud_server.rescue_enabled,
 286             "backup_window": to_native(self.hcloud_server.backup_window),
 287             "labels": self.hcloud_server.labels,
 288             "delete_protection": self.hcloud_server.protection["delete"],
 289             "rebuild_protection": self.hcloud_server.protection["rebuild"],
 290             "status": to_native(self.hcloud_server.status),
 291         }
 292 
 293     def _get_server(self):
 294         try:
 295             if self.module.params.get("id") is not None:
 296                 self.hcloud_server = self.client.servers.get_by_id(
 297                     self.module.params.get("id")
 298                 )
 299             else:
 300                 self.hcloud_server = self.client.servers.get_by_name(
 301                     self.module.params.get("name")
 302                 )
 303         except APIException as e:
 304             self.module.fail_json(msg=e.message)
 305 
 306     def _create_server(self):
 307 
 308         self.module.fail_on_missing_params(
 309             required_params=["name", "server_type", "image"]
 310         )
 311         params = {
 312             "name": self.module.params.get("name"),
 313             "server_type": self.client.server_types.get_by_name(
 314                 self.module.params.get("server_type")
 315             ),
 316             "user_data": self.module.params.get("user_data"),
 317             "labels": self.module.params.get("labels"),
 318         }
 319         if self.client.images.get_by_name(self.module.params.get("image")) is not None:
 320             # When image name is not available look for id instead
 321             params["image"] = self.client.images.get_by_name(self.module.params.get("image"))
 322         else:
 323             params["image"] = self.client.images.get_by_id(self.module.params.get("image"))
 324 
 325         if self.module.params.get("ssh_keys") is not None:
 326             params["ssh_keys"] = [
 327                 SSHKey(name=ssh_key_name)
 328                 for ssh_key_name in self.module.params.get("ssh_keys")
 329             ]
 330 
 331         if self.module.params.get("volumes") is not None:
 332             params["volumes"] = [
 333                 Volume(id=volume_id) for volume_id in self.module.params.get("volumes")
 334             ]
 335 
 336         if self.module.params.get("location") is None and self.module.params.get("datacenter") is None:
 337             # When not given, the API will choose the location.
 338             params["location"] = None
 339             params["datacenter"] = None
 340         elif self.module.params.get("location") is not None and self.module.params.get("datacenter") is None:
 341             params["location"] = self.client.locations.get_by_name(
 342                 self.module.params.get("location")
 343             )
 344         elif self.module.params.get("location") is None and self.module.params.get("datacenter") is not None:
 345             params["datacenter"] = self.client.datacenters.get_by_name(
 346                 self.module.params.get("datacenter")
 347             )
 348 
 349         if not self.module.check_mode:
 350             resp = self.client.servers.create(**params)
 351             self.result["root_password"] = resp.root_password
 352             resp.action.wait_until_finished(max_retries=1000)
 353             [action.wait_until_finished() for action in resp.next_actions]
 354 
 355             rescue_mode = self.module.params.get("rescue_mode")
 356             if rescue_mode:
 357                 self._get_server()
 358                 self._set_rescue_mode(rescue_mode)
 359 
 360         self._mark_as_changed()
 361         self._get_server()
 362 
 363     def _update_server(self):
 364         try:
 365             rescue_mode = self.module.params.get("rescue_mode")
 366             if rescue_mode and self.hcloud_server.rescue_enabled is False:
 367                 if not self.module.check_mode:
 368                     self._set_rescue_mode(rescue_mode)
 369                 self._mark_as_changed()
 370             elif not rescue_mode and self.hcloud_server.rescue_enabled is True:
 371                 if not self.module.check_mode:
 372                     self.hcloud_server.disable_rescue().wait_until_finished()
 373                 self._mark_as_changed()
 374 
 375             if self.module.params.get("backups") and self.hcloud_server.backup_window is None:
 376                 if not self.module.check_mode:
 377                     self.hcloud_server.enable_backup().wait_until_finished()
 378                 self._mark_as_changed()
 379             elif not self.module.params.get("backups") and self.hcloud_server.backup_window is not None:
 380                 if not self.module.check_mode:
 381                     self.hcloud_server.disable_backup().wait_until_finished()
 382                 self._mark_as_changed()
 383 
 384             labels = self.module.params.get("labels")
 385             if labels is not None and labels != self.hcloud_server.labels:
 386                 if not self.module.check_mode:
 387                     self.hcloud_server.update(labels=labels)
 388                 self._mark_as_changed()
 389 
 390             server_type = self.module.params.get("server_type")
 391             if server_type is not None and self.hcloud_server.server_type.name != server_type:
 392                 previous_server_status = self.hcloud_server.status
 393                 state = self.module.params.get("state")
 394                 if previous_server_status == Server.STATUS_RUNNING:
 395                     if not self.module.check_mode:
 396                         if self.module.params.get("force_upgrade") or state == "stopped":
 397                             self.stop_server()  # Only stopped server can be upgraded
 398                         else:
 399                             self.module.warn(
 400                                 "You can not upgrade a running instance %s. You need to stop the instance or use force_upgrade=yes."
 401                                 % self.hcloud_server.name
 402                             )
 403                 timeout = 100
 404                 if self.module.params.get("upgrade_disk"):
 405                     timeout = (
 406                         1000
 407                     )  # When we upgrade the disk too the resize progress takes some more time.
 408                 if not self.module.check_mode:
 409                     self.hcloud_server.change_type(
 410                         server_type=self.client.server_types.get_by_name(server_type),
 411                         upgrade_disk=self.module.params.get("upgrade_disk"),
 412                     ).wait_until_finished(timeout)
 413                     if state == "present" and previous_server_status == Server.STATUS_RUNNING or state == "started":
 414                         self.start_server()
 415 
 416                 self._mark_as_changed()
 417 
 418             delete_protection = self.module.params.get("delete_protection")
 419             rebuild_protection = self.module.params.get("rebuild_protection")
 420             if (delete_protection is not None and rebuild_protection is not None) and (
 421                     delete_protection != self.hcloud_server.protection["delete"] or rebuild_protection !=
 422                     self.hcloud_server.protection["rebuild"]):
 423                 if not self.module.check_mode:
 424                     self.hcloud_server.change_protection(delete=delete_protection,
 425                                                          rebuild=rebuild_protection).wait_until_finished()
 426                 self._mark_as_changed()
 427             self._get_server()
 428         except APIException as e:
 429             self.module.fail_json(msg=e.message)
 430 
 431     def _set_rescue_mode(self, rescue_mode):
 432         if self.module.params.get("ssh_keys"):
 433             resp = self.hcloud_server.enable_rescue(type=rescue_mode,
 434                                                     ssh_keys=[self.client.ssh_keys.get_by_name(ssh_key_name).id
 435                                                               for
 436                                                               ssh_key_name in
 437                                                               self.module.params.get("ssh_keys")])
 438         else:
 439             resp = self.hcloud_server.enable_rescue(type=rescue_mode)
 440         resp.action.wait_until_finished()
 441         self.result["root_password"] = resp.root_password
 442 
 443     def start_server(self):
 444         try:
 445             if self.hcloud_server.status != Server.STATUS_RUNNING:
 446                 if not self.module.check_mode:
 447                     self.client.servers.power_on(self.hcloud_server).wait_until_finished()
 448                 self._mark_as_changed()
 449             self._get_server()
 450         except APIException as e:
 451             self.module.fail_json(msg=e.message)
 452 
 453     def stop_server(self):
 454         try:
 455             if self.hcloud_server.status != Server.STATUS_OFF:
 456                 if not self.module.check_mode:
 457                     self.client.servers.power_off(self.hcloud_server).wait_until_finished()
 458                 self._mark_as_changed()
 459             self._get_server()
 460         except APIException as e:
 461             self.module.fail_json(msg=e.message)
 462 
 463     def rebuild_server(self):
 464         self.module.fail_on_missing_params(
 465             required_params=["image"]
 466         )
 467         try:
 468             if not self.module.check_mode:
 469                 self.client.servers.rebuild(self.hcloud_server, self.client.images.get_by_name(
 470                     self.module.params.get("image"))).wait_until_finished()
 471             self._mark_as_changed()
 472 
 473             self._get_server()
 474         except APIException as e:
 475             self.module.fail_json(msg=e.message)
 476 
 477     def present_server(self):
 478         self._get_server()
 479         if self.hcloud_server is None:
 480             self._create_server()
 481         else:
 482             self._update_server()
 483 
 484     def delete_server(self):
 485         try:
 486             self._get_server()
 487             if self.hcloud_server is not None:
 488                 if not self.module.check_mode:
 489                     self.client.servers.delete(self.hcloud_server).wait_until_finished()
 490                 self._mark_as_changed()
 491             self.hcloud_server = None
 492         except APIException as e:
 493             self.module.fail_json(msg=e.message)
 494 
 495     @staticmethod
 496     def define_module():
 497         return AnsibleModule(
 498             argument_spec=dict(
 499                 id={"type": "int"},
 500                 name={"type": "str"},
 501                 image={"type": "str"},
 502                 server_type={"type": "str"},
 503                 location={"type": "str"},
 504                 datacenter={"type": "str"},
 505                 user_data={"type": "str"},
 506                 ssh_keys={"type": "list"},
 507                 volumes={"type": "list"},
 508                 labels={"type": "dict"},
 509                 backups={"type": "bool", "default": False},
 510                 upgrade_disk={"type": "bool", "default": False},
 511                 force_upgrade={"type": "bool", "default": False},
 512                 rescue_mode={"type": "str"},
 513                 delete_protection={"type": "bool"},
 514                 rebuild_protection={"type": "bool"},
 515                 state={
 516                     "choices": ["absent", "present", "restarted", "started", "stopped", "rebuild"],
 517                     "default": "present",
 518                 },
 519                 **Hcloud.base_module_arguments()
 520             ),
 521             required_one_of=[['id', 'name']],
 522             mutually_exclusive=[["location", "datacenter"]],
 523             required_together=[["delete_protection", "rebuild_protection"]],
 524             supports_check_mode=True,
 525         )
 526 
 527 
 528 def main():
 529     module = AnsibleHcloudServer.define_module()
 530 
 531     hcloud = AnsibleHcloudServer(module)
 532     state = module.params.get("state")
 533     if state == "absent":
 534         hcloud.delete_server()
 535     elif state == "present":
 536         hcloud.present_server()
 537     elif state == "started":
 538         hcloud.present_server()
 539         hcloud.start_server()
 540     elif state == "stopped":
 541         hcloud.present_server()
 542         hcloud.stop_server()
 543     elif state == "restarted":
 544         hcloud.present_server()
 545         hcloud.stop_server()
 546         hcloud.start_server()
 547     elif state == "rebuild":
 548         hcloud.present_server()
 549         hcloud.rebuild_server()
 550 
 551     module.exit_json(**hcloud.get_result())
 552

 554 if __name__ == "__main__":
 555     main()
