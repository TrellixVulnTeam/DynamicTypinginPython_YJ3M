 458     def exec_module(self, **kwargs):
 459         """Main module execution method"""
 460 
 461         for key in list(self.module_arg_spec.keys()) + ['tags']:
 462             if hasattr(self, key):
 463                 setattr(self, key, kwargs[key])
 464             elif kwargs[key] is not None:
 465                 if key == "scm_type":
 466                     self.site_config[key] = kwargs[key]
 467 
 468         old_response = None
 469         response = None
 470         to_be_updated = False
 471 
 472         # set location
 473         resource_group = self.get_resource_group(self.resource_group)
 474         if not self.location:
 475             self.location = resource_group.location
 476 
 477         # get web app
 478         webapp_response = self.get_webapp()
 479 
 480         if not webapp_response:
 481             self.fail("Web app {0} does not exist in resource group {1}.".format(self.webapp_name, self.resource_group))
 482 
 483         # get slot
 484         old_response = self.get_slot()
 485 
 486         # set is_linux
 487         is_linux = True if webapp_response['reserved'] else False
 488 
 489         if self.state == 'present':
 490             if self.frameworks:
 491                 # java is mutually exclusive with other frameworks
 492                 if len(self.frameworks) > 1 and any(f['name'] == 'java' for f in self.frameworks):
 493                     self.fail('Java is mutually exclusive with other frameworks.')
 494 
 495                 if is_linux:
 496                     if len(self.frameworks) != 1:
 497                         self.fail('Can specify one framework only for Linux web app.')
 498 
 499                     if self.frameworks[0]['name'] not in self.supported_linux_frameworks:
 500                         self.fail('Unsupported framework {0} for Linux web app.'.format(self.frameworks[0]['name']))
 501 
 502                     self.site_config['linux_fx_version'] = (self.frameworks[0]['name'] + '|' + self.frameworks[0]['version']).upper()
 503 
 504                     if self.frameworks[0]['name'] == 'java':
 505                         if self.frameworks[0]['version'] != '8':
 506                             self.fail("Linux web app only supports java 8.")
 507 
 508                         if self.frameworks[0].get('settings', {}) and self.frameworks[0]['settings'].get('java_container', None) and \
 509                            self.frameworks[0]['settings']['java_container'].lower() != 'tomcat':
 510                             self.fail("Linux web app only supports tomcat container.")
 511 
 512                         if self.frameworks[0].get('settings', {}) and self.frameworks[0]['settings'].get('java_container', None) and \
 513                            self.frameworks[0]['settings']['java_container'].lower() == 'tomcat':
 514                             self.site_config['linux_fx_version'] = 'TOMCAT|' + self.frameworks[0]['settings']['java_container_version'] + '-jre8'
 515                         else:
 516                             self.site_config['linux_fx_version'] = 'JAVA|8-jre8'
 517                 else:
 518                     for fx in self.frameworks:
 519                         if fx.get('name') not in self.supported_windows_frameworks:
 520                             self.fail('Unsupported framework {0} for Windows web app.'.format(fx.get('name')))
 521                         else:
 522                             self.site_config[fx.get('name') + '_version'] = fx.get('version')
 523 
 524                         if 'settings' in fx and fx['settings'] is not None:
 525                             for key, value in fx['settings'].items():
 526                                 self.site_config[key] = value
 527 
 528             if not self.app_settings:
 529                 self.app_settings = dict()
 530 
 531             if self.container_settings:
 532                 linux_fx_version = 'DOCKER|'
 533 
 534                 if self.container_settings.get('registry_server_url'):
 535                     self.app_settings['DOCKER_REGISTRY_SERVER_URL'] = 'https://' + self.container_settings['registry_server_url']
 536 
 537                     linux_fx_version += self.container_settings['registry_server_url'] + '/'
 538 
 539                 linux_fx_version += self.container_settings['name']
 540 
 541                 self.site_config['linux_fx_version'] = linux_fx_version
 542 
 543                 if self.container_settings.get('registry_server_user'):
 544                     self.app_settings['DOCKER_REGISTRY_SERVER_USERNAME'] = self.container_settings['registry_server_user']
 545 
 546                 if self.container_settings.get('registry_server_password'):
 547                     self.app_settings['DOCKER_REGISTRY_SERVER_PASSWORD'] = self.container_settings['registry_server_password']
 548 
 549             # set auto_swap_slot_name
 550             if self.auto_swap_slot_name and isinstance(self.auto_swap_slot_name, str):
 551                 self.site_config['auto_swap_slot_name'] = self.auto_swap_slot_name
 552             if self.auto_swap_slot_name is False:
 553                 self.site_config['auto_swap_slot_name'] = None
 554 
 555             # init site
 556             self.site = Site(location=self.location, site_config=self.site_config)
 557 
 558             # check if the slot already present in the webapp
 559             if not old_response:
 560                 self.log("Web App slot doesn't exist")
 561 
 562                 to_be_updated = True
 563                 self.to_do = Actions.CreateOrUpdate
 564                 self.site.tags = self.tags
 565 
 566                 # if linux, setup startup_file
 567                 if self.startup_file:
 568                     self.site_config['app_command_line'] = self.startup_file
 569 
 570                 # set app setting
 571                 if self.app_settings:
 572                     app_settings = []
 573                     for key in self.app_settings.keys():
 574                         app_settings.append(NameValuePair(name=key, value=self.app_settings[key]))
 575 
 576                     self.site_config['app_settings'] = app_settings
 577 
 578                 # clone slot
 579                 if self.configuration_source:
 580                     self.clone = True
 581 
 582             else:
 583                 # existing slot, do update
 584                 self.log("Web App slot already exists")
 585 
 586                 self.log('Result: {0}'.format(old_response))
 587 
 588                 update_tags, self.site.tags = self.update_tags(old_response.get('tags', None))
 589 
 590                 if update_tags:
 591                     to_be_updated = True
 592 
 593                 # check if site_config changed
 594                 old_config = self.get_configuration_slot(self.name)
 595 
 596                 if self.is_site_config_changed(old_config):
 597                     to_be_updated = True
 598                     self.to_do = Actions.CreateOrUpdate
 599 
 600                 self.app_settings_strDic = self.list_app_settings_slot(self.name)
 601 
 602                 # purge existing app_settings:
 603                 if self.purge_app_settings:
 604                     to_be_updated = True
 605                     self.to_do = Actions.UpdateAppSettings
 606                     self.app_settings_strDic = dict()
 607 
 608                 # check if app settings changed
 609                 if self.purge_app_settings or self.is_app_settings_changed():
 610                     to_be_updated = True
 611                     self.to_do = Actions.UpdateAppSettings
 612 
 613                     if self.app_settings:
 614                         for key in self.app_settings.keys():
 615                             self.app_settings_strDic[key] = self.app_settings[key]
 616 
 617         elif self.state == 'absent':
 618             if old_response:
 619                 self.log("Delete Web App slot")
 620                 self.results['changed'] = True
 621 
 622                 if self.check_mode:
 623                     return self.results
 624 
 625                 self.delete_slot()
 626 
 627                 self.log('Web App slot deleted')
 628 
 629             else:
 630                 self.log("Web app slot {0} not exists.".format(self.name))
 631 
 632         if to_be_updated:
 633             self.log('Need to Create/Update web app')
 634             self.results['changed'] = True
 635 
 636             if self.check_mode:
 637                 return self.results
 638 
 639             if self.to_do == Actions.CreateOrUpdate:
 640                 response = self.create_update_slot()
 641 
 642                 self.results['id'] = response['id']
 643 
 644                 if self.clone:
 645                     self.clone_slot()
 646 
 647             if self.to_do == Actions.UpdateAppSettings:
 648                 self.update_app_settings_slot()
 649 
 650         slot = None
 651         if response:
 652             slot = response
 653         if old_response:
 654             slot = old_response
 655 
 656         if slot:
 657             if (slot['state'] != 'Stopped' and self.app_state == 'stopped') or \
 658                (slot['state'] != 'Running' and self.app_state == 'started') or \
 659                self.app_state == 'restarted':
 660 
 661                 self.results['changed'] = True
 662                 if self.check_mode:
 663                     return self.results
 664 
 665                 self.set_state_slot(self.app_state)
 666 
 667             if self.swap:
 668                 self.results['changed'] = True
 669                 if self.check_mode:
 670                     return self.results
 671 
 672                 self.swap_slot()
 673 
 674         return self.results

 678         for fx_version in self.site_config_updatable_frameworks:
 679             if self.site_config.get(fx_version):
 680                 if not getattr(existing_config, fx_version) or \
 681                         getattr(existing_config, fx_version).upper() != self.site_config.get(fx_version).upper():
