 302 def user_exists(cursor, user):
 303     # The PUBLIC user is a special case that is always there
 304     if user == 'PUBLIC':
 305         return True
 306     query = "SELECT rolname FROM pg_roles WHERE rolname=%(user)s"
 307     cursor.execute(query, {'user': user})
 308     return cursor.rowcount > 0

 311 def user_add(cursor, user, password, role_attr_flags, encrypted, expires, conn_limit):
 312     """Create a new database user (role)."""
 313     # Note: role_attr_flags escaped by parse_role_attrs and encrypted is a
 314     # literal
 315     query_password_data = dict(password=password, expires=expires)
 316     query = ['CREATE USER "%(user)s"' %
 317              {"user": user}]
 318     if password is not None and password != '':
 319         query.append("WITH %(crypt)s" % {"crypt": encrypted})
 320         query.append("PASSWORD %(password)s")
 321     if expires is not None:
 322         query.append("VALID UNTIL %(expires)s")
 323     if conn_limit is not None:
 324         query.append("CONNECTION LIMIT %(conn_limit)s" % {"conn_limit": conn_limit})
 325     query.append(role_attr_flags)
 326     query = ' '.join(query)
 327     executed_queries.append(query)
 328     cursor.execute(query, query_password_data)
 329     return True

 332 def user_should_we_change_password(current_role_attrs, user, password, encrypted):
 333     """Check if we should change the user's password.
 334 
 335     Compare the proposed password with the existing one, comparing
 336     hashes if encrypted. If we can't access it assume yes.
 337     """
 338 
 339     if current_role_attrs is None:
 340         # on some databases, E.g. AWS RDS instances, there is no access to
 341         # the pg_authid relation to check the pre-existing password, so we
 342         # just assume password is different
 343         return True
 344 
 345     # Do we actually need to do anything?
 346     pwchanging = False
 347     if password is not None:
 348         # Empty password means that the role shouldn't have a password, which
 349         # means we need to check if the current password is None.
 350         if password == '':
 351             if current_role_attrs['rolpassword'] is not None:
 352                 pwchanging = True
 353         # 32: MD5 hashes are represented as a sequence of 32 hexadecimal digits
 354         #  3: The size of the 'md5' prefix
 355         # When the provided password looks like a MD5-hash, value of
 356         # 'encrypted' is ignored.
 357         elif (password.startswith('md5') and len(password) == 32 + 3) or encrypted == 'UNENCRYPTED':
 358             if password != current_role_attrs['rolpassword']:
 359                 pwchanging = True
 360         elif encrypted == 'ENCRYPTED':
 361             hashed_password = 'md5{0}'.format(md5(to_bytes(password) + to_bytes(user)).hexdigest())
 362             if hashed_password != current_role_attrs['rolpassword']:
 363                 pwchanging = True
 364 
 365     return pwchanging

 368 def user_alter(db_connection, module, user, password, role_attr_flags, encrypted, expires, no_password_changes, conn_limit):
 369     """Change user password and/or attributes. Return True if changed, False otherwise."""
 370     changed = False
 371 
 372     cursor = db_connection.cursor(cursor_factory=DictCursor)
 373     # Note: role_attr_flags escaped by parse_role_attrs and encrypted is a
 374     # literal
 375     if user == 'PUBLIC':
 376         if password is not None:
 377             module.fail_json(msg="cannot change the password for PUBLIC user")
 378         elif role_attr_flags != '':
 379             module.fail_json(msg="cannot change the role_attr_flags for PUBLIC user")
 380         else:
 381             return False
 382 
 383     # Handle passwords.
 384     if not no_password_changes and (password is not None or role_attr_flags != '' or expires is not None or conn_limit is not None):
 385         # Select password and all flag-like columns in order to verify changes.
 386         try:
 387             select = "SELECT * FROM pg_authid where rolname=%(user)s"
 388             cursor.execute(select, {"user": user})
 389             # Grab current role attributes.
 390             current_role_attrs = cursor.fetchone()
 391         except psycopg2.ProgrammingError:
 392             current_role_attrs = None
 393             db_connection.rollback()
 394 
 395         pwchanging = user_should_we_change_password(current_role_attrs, user, password, encrypted)
 396 
 397         if current_role_attrs is None:
 398             try:
 399                 # AWS RDS instances does not allow user to access pg_authid
 400                 # so try to get current_role_attrs from pg_roles tables
 401                 select = "SELECT * FROM pg_roles where rolname=%(user)s"
 402                 cursor.execute(select, {"user": user})
 403                 # Grab current role attributes from pg_roles
 404                 current_role_attrs = cursor.fetchone()
 405             except psycopg2.ProgrammingError as e:
 406                 db_connection.rollback()
 407                 module.fail_json(msg="Failed to get role details for current user %s: %s" % (user, e))
 408 
 409         role_attr_flags_changing = False
 410         if role_attr_flags:
 411             role_attr_flags_dict = {}
 412             for r in role_attr_flags.split(' '):
 413                 if r.startswith('NO'):
 414                     role_attr_flags_dict[r.replace('NO', '', 1)] = False
 415                 else:
 416                     role_attr_flags_dict[r] = True
 417 
 418             for role_attr_name, role_attr_value in role_attr_flags_dict.items():
 419                 if current_role_attrs[PRIV_TO_AUTHID_COLUMN[role_attr_name]] != role_attr_value:
 420                     role_attr_flags_changing = True
 421 
 422         if expires is not None:
 423             cursor.execute("SELECT %s::timestamptz;", (expires,))
 424             expires_with_tz = cursor.fetchone()[0]
 425             expires_changing = expires_with_tz != current_role_attrs.get('rolvaliduntil')
 426         else:
 427             expires_changing = False
 428 
 429         conn_limit_changing = (conn_limit is not None and conn_limit != current_role_attrs['rolconnlimit'])
 430 
 431         if not pwchanging and not role_attr_flags_changing and not expires_changing and not conn_limit_changing:
 432             return False
 433 
 434         alter = ['ALTER USER "%(user)s"' % {"user": user}]
 435         if pwchanging:
 436             if password != '':
 437                 alter.append("WITH %(crypt)s" % {"crypt": encrypted})
 438                 alter.append("PASSWORD %(password)s")
 439             else:
 440                 alter.append("WITH PASSWORD NULL")
 441             alter.append(role_attr_flags)
 442         elif role_attr_flags:
 443             alter.append('WITH %s' % role_attr_flags)
 444         if expires is not None:
 445             alter.append("VALID UNTIL %(expires)s")
 446         if conn_limit is not None:
 447             alter.append("CONNECTION LIMIT %(conn_limit)s" % {"conn_limit": conn_limit})
 448 
 449         query_password_data = dict(password=password, expires=expires)
 450         try:
 451             cursor.execute(' '.join(alter), query_password_data)
 452             changed = True
 453         except psycopg2.InternalError as e:
 454             if e.pgcode == '25006':
 455                 # Handle errors due to read-only transactions indicated by pgcode 25006
 456                 # ERROR:  cannot execute ALTER ROLE in a read-only transaction
 457                 changed = False
 458                 module.fail_json(msg=e.pgerror, exception=traceback.format_exc())
 459                 return changed
 460             else:
 461                 raise psycopg2.InternalError(e)
 462         except psycopg2.NotSupportedError as e:
 463             module.fail_json(msg=e.pgerror, exception=traceback.format_exc())
 464 
 465     elif no_password_changes and role_attr_flags != '':
 466         # Grab role information from pg_roles instead of pg_authid
 467         select = "SELECT * FROM pg_roles where rolname=%(user)s"
 468         cursor.execute(select, {"user": user})
 469         # Grab current role attributes.
 470         current_role_attrs = cursor.fetchone()
 471 
 472         role_attr_flags_changing = False
 473 
 474         if role_attr_flags:
 475             role_attr_flags_dict = {}
 476             for r in role_attr_flags.split(' '):
 477                 if r.startswith('NO'):
 478                     role_attr_flags_dict[r.replace('NO', '', 1)] = False
 479                 else:
 480                     role_attr_flags_dict[r] = True
 481 
 482             for role_attr_name, role_attr_value in role_attr_flags_dict.items():
 483                 if current_role_attrs[PRIV_TO_AUTHID_COLUMN[role_attr_name]] != role_attr_value:
 484                     role_attr_flags_changing = True
 485 
 486         if not role_attr_flags_changing:
 487             return False
 488 
 489         alter = ['ALTER USER "%(user)s"' %
 490                  {"user": user}]
 491         if role_attr_flags:
 492             alter.append('WITH %s' % role_attr_flags)
 493 
 494         try:
 495             cursor.execute(' '.join(alter))
 496         except psycopg2.InternalError as e:
 497             if e.pgcode == '25006':
 498                 # Handle errors due to read-only transactions indicated by pgcode 25006
 499                 # ERROR:  cannot execute ALTER ROLE in a read-only transaction
 500                 changed = False
 501                 module.fail_json(msg=e.pgerror, exception=traceback.format_exc())
 502                 return changed
 503             else:
 504                 raise psycopg2.InternalError(e)
 505 
 506         # Grab new role attributes.
 507         cursor.execute(select, {"user": user})
 508         new_role_attrs = cursor.fetchone()
 509 
 510         # Detect any differences between current_ and new_role_attrs.
 511         changed = current_role_attrs != new_role_attrs
 512 
 513     return changed

 516 def user_delete(cursor, user):
 517     """Try to remove a user. Returns True if successful otherwise False"""
 518     cursor.execute("SAVEPOINT ansible_pgsql_user_delete")
 519     try:
 520         query = 'DROP USER "%s"' % user
 521         executed_queries.append(query)
 522         cursor.execute(query)
 523     except Exception:
 524         cursor.execute("ROLLBACK TO SAVEPOINT ansible_pgsql_user_delete")
 525         cursor.execute("RELEASE SAVEPOINT ansible_pgsql_user_delete")
 526         return False
 527 
 528     cursor.execute("RELEASE SAVEPOINT ansible_pgsql_user_delete")
 529     return True

 532 def has_table_privileges(cursor, user, table, privs):
 533     """
 534     Return the difference between the privileges that a user already has and
 535     the privileges that they desire to have.
 536 
 537     :returns: tuple of:
 538         * privileges that they have and were requested
 539         * privileges they currently hold but were not requested
 540         * privileges requested that they do not hold
 541     """
 542     cur_privs = get_table_privileges(cursor, user, table)
 543     have_currently = cur_privs.intersection(privs)
 544     other_current = cur_privs.difference(privs)
 545     desired = privs.difference(cur_privs)
 546     return (have_currently, other_current, desired)

 549 def get_table_privileges(cursor, user, table):
 550     if '.' in table:
 551         schema, table = table.split('.', 1)
 552     else:
 553         schema = 'public'
 554     query = ("SELECT privilege_type FROM information_schema.role_table_grants "
 555              "WHERE grantee=%(user)s AND table_name=%(table)s AND table_schema=%(schema)s")
 556     cursor.execute(query, {'user': user, 'table': table, 'schema': schema})
 557     return frozenset([x[0] for x in cursor.fetchall()])

 578 def get_database_privileges(cursor, user, db):
 579     priv_map = {
 580         'C': 'CREATE',
 581         'T': 'TEMPORARY',
 582         'c': 'CONNECT',
 583     }
 584     query = 'SELECT datacl FROM pg_database WHERE datname = %s'
 585     cursor.execute(query, (db,))
 586     datacl = cursor.fetchone()[0]
 587     if datacl is None:
 588         return set()
 589     r = re.search(r'%s\\?"?=(C?T?c?)/[^,]+,?' % user, datacl)
 590     if r is None:
 591         return set()
 592     o = set()
 593     for v in r.group(1):
 594         o.add(priv_map[v])
 595     return normalize_privileges(o, 'database')

 606         * privileges requested that they do not hold
 607     """
 608     cur_privs = get_database_privileges(cursor, user, db)
 609     have_currently = cur_privs.intersection(privs)
 610     other_current = cur_privs.difference(privs)
 611     desired = privs.difference(cur_privs)
 612     return (have_currently, other_current, desired)

 615 def grant_database_privileges(cursor, user, db, privs):
 616     # Note: priv escaped by parse_privs
 617     privs = ', '.join(privs)
 618     if user == "PUBLIC":
 619         query = 'GRANT %s ON DATABASE %s TO PUBLIC' % (
 620                 privs, pg_quote_identifier(db, 'database'))
 621     else:
 622         query = 'GRANT %s ON DATABASE %s TO "%s"' % (
 623                 privs, pg_quote_identifier(db, 'database'), user)
 624 
 625     executed_queries.append(query)
 626     cursor.execute(query)

 629 def revoke_database_privileges(cursor, user, db, privs):
 630     # Note: priv escaped by parse_privs
 631     privs = ', '.join(privs)
 632     if user == "PUBLIC":
 633         query = 'REVOKE %s ON DATABASE %s FROM PUBLIC' % (
 634                 privs, pg_quote_identifier(db, 'database'))
 635     else:
 636         query = 'REVOKE %s ON DATABASE %s FROM "%s"' % (
 637                 privs, pg_quote_identifier(db, 'database'), user)
 638 
 639     executed_queries.append(query)
 640     cursor.execute(query)

 643 def revoke_privileges(cursor, user, privs):
 644     if privs is None:
 645         return False
 646 
 647     revoke_funcs = dict(table=revoke_table_privileges,
 648                         database=revoke_database_privileges)
 649     check_funcs = dict(table=has_table_privileges,
 650                        database=has_database_privileges)
 651 
 652     changed = False
 653     for type_ in privs:
 654         for name, privileges in iteritems(privs[type_]):
 655             # Check that any of the privileges requested to be removed are
 656             # currently granted to the user
 657             differences = check_funcs[type_](cursor, user, name, privileges)
 658             if differences[0]:
 659                 revoke_funcs[type_](cursor, user, name, privileges)
 660                 changed = True
 661     return changed
 662 
 663 
 664 def grant_privileges(cursor, user, privs):
 665     if privs is None:
 666         return False
 667 
 668     grant_funcs = dict(table=grant_table_privileges,
 669                        database=grant_database_privileges)
 670     check_funcs = dict(table=has_table_privileges,
 671                        database=has_database_privileges)
 672 
 673     changed = False
 674     for type_ in privs:
 675         for name, privileges in iteritems(privs[type_]):
 676             # Check that any of the privileges requested for the user are
 677             # currently missing
 678             differences = check_funcs[type_](cursor, user, name, privileges)
 679             if differences[2]:
 680                 grant_funcs[type_](cursor, user, name, privileges)
 681                 changed = True
 682     return changed

 685 def parse_role_attrs(cursor, role_attr_flags):
 686     """
 687     Parse role attributes string for user creation.
 688     Format:
 689 
 690         attributes[,attributes,...]
 691 
 692     Where:
 693 
 694         attributes := CREATEDB,CREATEROLE,NOSUPERUSER,...
 695         [ "[NO]SUPERUSER","[NO]CREATEROLE", "[NO]CREATEDB",
 696                             "[NO]INHERIT", "[NO]LOGIN", "[NO]REPLICATION",
 697                             "[NO]BYPASSRLS" ]
 698 
 699     Note: "[NO]BYPASSRLS" role attribute introduced in 9.5
 700     Note: "[NO]CREATEUSER" role attribute is deprecated.
 701 
 702     """
 703     flags = frozenset(role.upper() for role in role_attr_flags.split(',') if role)
 704 
 705     valid_flags = frozenset(itertools.chain(FLAGS, get_valid_flags_by_version(cursor)))
 706     valid_flags = frozenset(itertools.chain(valid_flags, ('NO%s' % flag for flag in valid_flags)))
 707 
 708     if not flags.issubset(valid_flags):
 709         raise InvalidFlagsError('Invalid role_attr_flags specified: %s' %
 710                                 ' '.join(flags.difference(valid_flags)))
 711 
 712     return ' '.join(flags)
 713

 715 def normalize_privileges(privs, type_):
 716     new_privs = set(privs)
 717     if 'ALL' in new_privs:
 718         new_privs.update(VALID_PRIVS[type_])
 719         new_privs.remove('ALL')
 720     if 'TEMP' in new_privs:
 721         new_privs.add('TEMPORARY')
 722         new_privs.remove('TEMP')
 723 
 724     return new_privs

 727 def parse_privs(privs, db):
 728     """
 729     Parse privilege string to determine permissions for database db.
 730     Format:
 731 
 732         privileges[/privileges/...]
 733 
 734     Where:
 735 
 736         privileges := DATABASE_PRIVILEGES[,DATABASE_PRIVILEGES,...] |
 737             TABLE_NAME:TABLE_PRIVILEGES[,TABLE_PRIVILEGES,...]
 738     """
 739     if privs is None:
 740         return privs
 741 
 742     o_privs = {
 743         'database': {},
 744         'table': {}
 745     }
 746     for token in privs.split('/'):
 747         if ':' not in token:
 748             type_ = 'database'
 749             name = db
 750             priv_set = frozenset(x.strip().upper()
 751                                  for x in token.split(',') if x.strip())
 752         else:
 753             type_ = 'table'
 754             name, privileges = token.split(':', 1)
 755             priv_set = frozenset(x.strip().upper()
 756                                  for x in privileges.split(',') if x.strip())
 757 
 758         if not priv_set.issubset(VALID_PRIVS[type_]):
 759             raise InvalidPrivsError('Invalid privs specified for %s: %s' %
 760                                     (type_, ' '.join(priv_set.difference(VALID_PRIVS[type_]))))
 761 
 762         priv_set = normalize_privileges(priv_set, type_)
 763         o_privs[type_][name] = priv_set
 764 
 765     return o_privs

 768 def get_valid_flags_by_version(cursor):
 769     """
 770     Some role attributes were introduced after certain versions. We want to
 771     compile a list of valid flags against the current Postgres version.
 772     """
 773     current_version = cursor.connection.server_version
 774 
 775     return [
 776         flag
 777         for flag, version_introduced in FLAGS_BY_VERSION.items()
 778         if current_version >= version_introduced
 779     ]

 782 def get_comment(cursor, user):
 783     """Get user's comment."""
 784     query = ("SELECT pg_catalog.shobj_description(r.oid, 'pg_authid') "
 785              "FROM pg_catalog.pg_roles r "
 786              "WHERE r.rolname = %(user)s")
 787     cursor.execute(query, {'user': user})
 788     return cursor.fetchone()[0]

 791 def add_comment(cursor, user, comment):
 792     """Add comment on user."""
 793     if comment != get_comment(cursor, user):
 794         query = 'COMMENT ON ROLE "%s" IS ' % user
 795         cursor.execute(query + '%(comment)s', {'comment': comment})
 796         executed_queries.append(cursor.mogrify(query + '%(comment)s', {'comment': comment}))
 797         return True
 798     else:
 799         return False

 806 def main():
 807     argument_spec = postgres_common_argument_spec()
 808     argument_spec.update(
 809         user=dict(type='str', required=True, aliases=['name']),
 810         password=dict(type='str', default=None, no_log=True),
 811         state=dict(type='str', default='present', choices=['absent', 'present']),
 812         priv=dict(type='str', default=None),
 813         db=dict(type='str', default='', aliases=['login_db']),
 814         fail_on_user=dict(type='bool', default='yes', aliases=['fail_on_role']),
 815         role_attr_flags=dict(type='str', default=''),
 816         encrypted=dict(type='bool', default='yes'),
 817         no_password_changes=dict(type='bool', default='no'),
 818         expires=dict(type='str', default=None),
 819         conn_limit=dict(type='int', default=None),
 820         session_role=dict(type='str'),
 821         groups=dict(type='list', elements='str'),
 822         comment=dict(type='str', default=None),
 823     )
 824     module = AnsibleModule(
 825         argument_spec=argument_spec,
 826         supports_check_mode=True
 827     )
 828 
 829     user = module.params["user"]
 830     password = module.params["password"]
 831     state = module.params["state"]
 832     fail_on_user = module.params["fail_on_user"]
 833     if module.params['db'] == '' and module.params["priv"] is not None:
 834         module.fail_json(msg="privileges require a database to be specified")
 835     privs = parse_privs(module.params["priv"], module.params["db"])
 836     no_password_changes = module.params["no_password_changes"]
 837     if module.params["encrypted"]:
 838         encrypted = "ENCRYPTED"
 839     else:
 840         encrypted = "UNENCRYPTED"
 841     expires = module.params["expires"]
 842     conn_limit = module.params["conn_limit"]
 843     role_attr_flags = module.params["role_attr_flags"]
 844     groups = module.params["groups"]
 845     if groups:
 846         groups = [e.strip() for e in groups]
 847     comment = module.params["comment"]
 848 
 849     conn_params = get_conn_params(module, module.params, warn_db_default=False)
 850     db_connection = connect_to_db(module, conn_params)
 851     cursor = db_connection.cursor(cursor_factory=DictCursor)
 852 
 853     try:
 854         role_attr_flags = parse_role_attrs(cursor, role_attr_flags)
 855     except InvalidFlagsError as e:
 856         module.fail_json(msg=to_native(e), exception=traceback.format_exc())
 857 
 858     kw = dict(user=user)
 859     changed = False
 860     user_removed = False
 861 
 862     if state == "present":
 863         if user_exists(cursor, user):
 864             try:
 865                 changed = user_alter(db_connection, module, user, password,
 866                                      role_attr_flags, encrypted, expires, no_password_changes, conn_limit)
 867             except SQLParseError as e:
 868                 module.fail_json(msg=to_native(e), exception=traceback.format_exc())
 869         else:
 870             try:
 871                 changed = user_add(cursor, user, password,
 872                                    role_attr_flags, encrypted, expires, conn_limit)
 873             except psycopg2.ProgrammingError as e:
 874                 module.fail_json(msg="Unable to add user with given requirement "
 875                                      "due to : %s" % to_native(e),
 876                                  exception=traceback.format_exc())
 877             except SQLParseError as e:
 878                 module.fail_json(msg=to_native(e), exception=traceback.format_exc())
 879         try:
 880             changed = grant_privileges(cursor, user, privs) or changed
 881         except SQLParseError as e:
 882             module.fail_json(msg=to_native(e), exception=traceback.format_exc())
 883 
 884         if groups:
 885             target_roles = []
 886             target_roles.append(user)
 887             pg_membership = PgMembership(module, cursor, groups, target_roles)
 888             changed = pg_membership.grant() or changed
 889             executed_queries.extend(pg_membership.executed_queries)
 890 
 891         if comment is not None:
 892             try:
 893                 changed = add_comment(cursor, user, comment) or changed
 894             except Exception as e:
 895                 module.fail_json(msg='Unable to add comment on role: %s' % to_native(e),
 896                                  exception=traceback.format_exc())
 897 
 898     else:
 899         if user_exists(cursor, user):
 900             if module.check_mode:
 901                 changed = True
 902                 kw['user_removed'] = True
 903             else:
 904                 try:
 905                     changed = revoke_privileges(cursor, user, privs)
 906                     user_removed = user_delete(cursor, user)
 907                 except SQLParseError as e:
 908                     module.fail_json(msg=to_native(e), exception=traceback.format_exc())
 909                 changed = changed or user_removed
 910                 if fail_on_user and not user_removed:
 911                     msg = "Unable to remove user"
 912                     module.fail_json(msg=msg)
 913                 kw['user_removed'] = user_removed
 914 
 915     if changed:
 916         if module.check_mode:
 917             db_connection.rollback()
 918         else:
 919             db_connection.commit()
 920 
 921     kw['changed'] = changed
 922     kw['queries'] = executed_queries
 923     module.exit_json(**kw)
