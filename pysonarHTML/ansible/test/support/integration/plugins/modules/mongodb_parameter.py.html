 125 # =========================================
 126 # MongoDB module specific support methods.
 127 #
 128 
 129 def load_mongocnf():
 130     config = configparser.RawConfigParser()
 131     mongocnf = os.path.expanduser('~/.mongodb.cnf')
 132 
 133     try:
 134         config.readfp(open(mongocnf))
 135         creds = dict(
 136             user=config.get('client', 'user'),
 137             password=config.get('client', 'pass')
 138         )
 139     except (configparser.NoOptionError, IOError):
 140         return False
 141 
 142     return creds
 143

 149 def main():
 150     module = AnsibleModule(
 151         argument_spec=dict(
 152             login_user=dict(default=None),
 153             login_password=dict(default=None, no_log=True),
 154             login_host=dict(default='localhost'),
 155             login_port=dict(default=27017, type='int'),
 156             login_database=dict(default=None),
 157             replica_set=dict(default=None),
 158             param=dict(required=True),
 159             value=dict(required=True),
 160             param_type=dict(default="str", choices=['str', 'int']),
 161             ssl=dict(default=False, type='bool'),
 162         )
 163     )
 164 
 165     if not pymongo_found:
 166         module.fail_json(msg=missing_required_lib('pymongo'))
 167 
 168     login_user = module.params['login_user']
 169     login_password = module.params['login_password']
 170     login_host = module.params['login_host']
 171     login_port = module.params['login_port']
 172     login_database = module.params['login_database']
 173 
 174     replica_set = module.params['replica_set']
 175     ssl = module.params['ssl']
 176 
 177     param = module.params['param']
 178     param_type = module.params['param_type']
 179     value = module.params['value']
 180 
 181     # Verify parameter is coherent with specified type
 182     try:
 183         if param_type == 'int':
 184             value = int(value)
 185     except ValueError:
 186         module.fail_json(msg="value '%s' is not %s" % (value, param_type))
 187 
 188     try:
 189         if replica_set:
 190             client = MongoClient(login_host, int(login_port), replicaset=replica_set, ssl=ssl)
 191         else:
 192             client = MongoClient(login_host, int(login_port), ssl=ssl)
 193 
 194         if login_user is None and login_password is None:
 195             mongocnf_creds = load_mongocnf()
 196             if mongocnf_creds is not False:
 197                 login_user = mongocnf_creds['user']
 198                 login_password = mongocnf_creds['password']
 199         elif login_password is None or login_user is None:
 200             module.fail_json(msg='when supplying login arguments, both login_user and login_password must be provided')
 201 
 202         if login_user is not None and login_password is not None:
 203             client.admin.authenticate(login_user, login_password, source=login_database)
 204 
 205     except ConnectionFailure as e:
 206         module.fail_json(msg='unable to connect to database: %s' % to_native(e), exception=traceback.format_exc())
 207 
 208     db = client.admin
 209 
 210     try:
 211         after_value = db.command("setParameter", **{param: value})
 212     except OperationFailure as e:
 213         module.fail_json(msg="unable to change parameter: %s" % to_native(e), exception=traceback.format_exc())
 214 
 215     if "was" not in after_value:
 216         module.exit_json(changed=True, msg="Unable to determine old value, assume it changed.")
 217     else:
 218         module.exit_json(changed=(value != after_value["was"]), before=after_value["was"],
 219                          after=value)
 220 
 221 
 222 if __name__ == '__main__':
 223     main()
