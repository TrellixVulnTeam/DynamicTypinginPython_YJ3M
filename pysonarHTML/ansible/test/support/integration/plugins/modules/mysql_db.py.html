 287 def db_exists(cursor, db):
 288     res = 0
 289     for each_db in db:
 290         res += cursor.execute("SHOW DATABASES LIKE %s", (each_db.replace("_", r"\_"),))
 291     return res == len(db)

 294 def db_delete(cursor, db):
 295     if not db:
 296         return False
 297     for each_db in db:
 298         query = "DROP DATABASE %s" % mysql_quote_identifier(each_db, 'database')
 299         executed_commands.append(query)
 300         cursor.execute(query)
 301     return True

 304 def db_dump(module, host, user, password, db_name, target, all_databases, port,
 305             config_file, socket=None, ssl_cert=None, ssl_key=None, ssl_ca=None,
 306             single_transaction=None, quick=None, ignore_tables=None, hex_blob=None,
 307             encoding=None, force=False, master_data=0, skip_lock_tables=False, dump_extra_args=None):
 308     cmd = module.get_bin_path('mysqldump', True)
 309     # If defined, mysqldump demands --defaults-extra-file be the first option
 310     if config_file:
 311         cmd += " --defaults-extra-file=%s" % shlex_quote(config_file)
 312     if user is not None:
 313         cmd += " --user=%s" % shlex_quote(user)
 314     if password is not None:
 315         cmd += " --password=%s" % shlex_quote(password)
 316     if ssl_cert is not None:
 317         cmd += " --ssl-cert=%s" % shlex_quote(ssl_cert)
 318     if ssl_key is not None:
 319         cmd += " --ssl-key=%s" % shlex_quote(ssl_key)
 320     if ssl_ca is not None:
 321         cmd += " --ssl-ca=%s" % shlex_quote(ssl_ca)
 322     if force:
 323         cmd += " --force"
 324     if socket is not None:
 325         cmd += " --socket=%s" % shlex_quote(socket)
 326     else:
 327         cmd += " --host=%s --port=%i" % (shlex_quote(host), port)
 328 
 329     if all_databases:
 330         cmd += " --all-databases"
 331     elif len(db_name) > 1:
 332         cmd += " --databases {0}".format(' '.join(db_name))
 333     else:
 334         cmd += " %s" % shlex_quote(' '.join(db_name))
 335 
 336     if skip_lock_tables:
 337         cmd += " --skip-lock-tables"
 338     if (encoding is not None) and (encoding != ""):
 339         cmd += " --default-character-set=%s" % shlex_quote(encoding)
 340     if single_transaction:
 341         cmd += " --single-transaction=true"
 342     if quick:
 343         cmd += " --quick"
 344     if ignore_tables:
 345         for an_ignored_table in ignore_tables:
 346             cmd += " --ignore-table={0}".format(an_ignored_table)
 347     if hex_blob:
 348         cmd += " --hex-blob"
 349     if master_data:
 350         cmd += " --master-data=%s" % master_data
 351     if dump_extra_args is not None:
 352         cmd += " " + dump_extra_args
 353 
 354     path = None
 355     if os.path.splitext(target)[-1] == '.gz':
 356         path = module.get_bin_path('gzip', True)
 357     elif os.path.splitext(target)[-1] == '.bz2':
 358         path = module.get_bin_path('bzip2', True)
 359     elif os.path.splitext(target)[-1] == '.xz':
 360         path = module.get_bin_path('xz', True)
 361 
 362     if path:
 363         cmd = '%s | %s > %s' % (cmd, path, shlex_quote(target))
 364     else:
 365         cmd += " > %s" % shlex_quote(target)
 366 
 367     executed_commands.append(cmd)
 368     rc, stdout, stderr = module.run_command(cmd, use_unsafe_shell=True)
 369     return rc, stdout, stderr

 372 def db_import(module, host, user, password, db_name, target, all_databases, port, config_file,
 373               socket=None, ssl_cert=None, ssl_key=None, ssl_ca=None, encoding=None, force=False):
 374     if not os.path.exists(target):
 375         return module.fail_json(msg="target %s does not exist on the host" % target)
 376 
 377     cmd = [module.get_bin_path('mysql', True)]
 378     # --defaults-file must go first, or errors out
 379     if config_file:
 380         cmd.append("--defaults-extra-file=%s" % shlex_quote(config_file))
 381     if user:
 382         cmd.append("--user=%s" % shlex_quote(user))
 383     if password:
 384         cmd.append("--password=%s" % shlex_quote(password))
 385     if ssl_cert is not None:
 386         cmd.append("--ssl-cert=%s" % shlex_quote(ssl_cert))
 387     if ssl_key is not None:
 388         cmd.append("--ssl-key=%s" % shlex_quote(ssl_key))
 389     if ssl_ca is not None:
 390         cmd.append("--ssl-ca=%s" % shlex_quote(ssl_ca))
 391     if force:
 392         cmd.append("-f")
 393     if socket is not None:
 394         cmd.append("--socket=%s" % shlex_quote(socket))
 395     else:
 396         cmd.append("--host=%s" % shlex_quote(host))
 397         cmd.append("--port=%i" % port)
 398     if (encoding is not None) and (encoding != ""):
 399         cmd.append("--default-character-set=%s" % shlex_quote(encoding))
 400     if not all_databases:
 401         cmd.append("--one-database")
 402         cmd.append(shlex_quote(''.join(db_name)))
 403 
 404     comp_prog_path = None
 405     if os.path.splitext(target)[-1] == '.gz':
 406         comp_prog_path = module.get_bin_path('gzip', required=True)
 407     elif os.path.splitext(target)[-1] == '.bz2':
 408         comp_prog_path = module.get_bin_path('bzip2', required=True)
 409     elif os.path.splitext(target)[-1] == '.xz':
 410         comp_prog_path = module.get_bin_path('xz', required=True)
 411     if comp_prog_path:
 412         # The line above is for returned data only:
 413         executed_commands.append('%s -dc %s | %s' % (comp_prog_path, target, ' '.join(cmd)))
 414         p1 = subprocess.Popen([comp_prog_path, '-dc', target], stdout=subprocess.PIPE, stderr=subprocess.PIPE)
 415         p2 = subprocess.Popen(cmd, stdin=p1.stdout, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
 416         (stdout2, stderr2) = p2.communicate()
 417         p1.stdout.close()
 418         p1.wait()
 419         if p1.returncode != 0:
 420             stderr1 = p1.stderr.read()
 421             return p1.returncode, '', stderr1
 422         else:
 423             return p2.returncode, stdout2, stderr2
 424     else:
 425         cmd = ' '.join(cmd)
 426         cmd += " < %s" % shlex_quote(target)
 427         executed_commands.append(cmd)
 428         rc, stdout, stderr = module.run_command(cmd, use_unsafe_shell=True)
 429         return rc, stdout, stderr

 432 def db_create(cursor, db, encoding, collation):
 433     if not db:
 434         return False
 435     query_params = dict(enc=encoding, collate=collation)
 436     res = 0
 437     for each_db in db:
 438         query = ['CREATE DATABASE %s' % mysql_quote_identifier(each_db, 'database')]
 439         if encoding:
 440             query.append("CHARACTER SET %(enc)s")
 441         if collation:
 442             query.append("COLLATE %(collate)s")
 443         query = ' '.join(query)
 444         res += cursor.execute(query, query_params)
 445         try:
 446             executed_commands.append(cursor.mogrify(query, query_params))
 447         except AttributeError:
 448             executed_commands.append(cursor._executed)
 449         except Exception:
 450             executed_commands.append(query)
 451     return res > 0

 459 def main():
 460     module = AnsibleModule(
 461         argument_spec=dict(
 462             login_user=dict(type='str'),
 463             login_password=dict(type='str', no_log=True),
 464             login_host=dict(type='str', default='localhost'),
 465             login_port=dict(type='int', default=3306),
 466             login_unix_socket=dict(type='str'),
 467             name=dict(type='list', required=True, aliases=['db']),
 468             encoding=dict(type='str', default=''),
 469             collation=dict(type='str', default=''),
 470             target=dict(type='path'),
 471             state=dict(type='str', default='present', choices=['absent', 'dump', 'import', 'present']),
 472             client_cert=dict(type='path', aliases=['ssl_cert']),
 473             client_key=dict(type='path', aliases=['ssl_key']),
 474             ca_cert=dict(type='path', aliases=['ssl_ca']),
 475             connect_timeout=dict(type='int', default=30),
 476             config_file=dict(type='path', default='~/.my.cnf'),
 477             single_transaction=dict(type='bool', default=False),
 478             quick=dict(type='bool', default=True),
 479             ignore_tables=dict(type='list', default=[]),
 480             hex_blob=dict(default=False, type='bool'),
 481             force=dict(type='bool', default=False),
 482             master_data=dict(type='int', default=0, choices=[0, 1, 2]),
 483             skip_lock_tables=dict(type='bool', default=False),
 484             dump_extra_args=dict(type='str'),
 485         ),
 486         supports_check_mode=True,
 487     )
 488 
 489     if mysql_driver is None:
 490         module.fail_json(msg=mysql_driver_fail_msg)
 491 
 492     db = module.params["name"]
 493     if not db:
 494         module.exit_json(changed=False, db=db, db_list=[])
 495     db = [each_db.strip() for each_db in db]
 496 
 497     encoding = module.params["encoding"]
 498     collation = module.params["collation"]
 499     state = module.params["state"]
 500     target = module.params["target"]
 501     socket = module.params["login_unix_socket"]
 502     login_port = module.params["login_port"]
 503     if login_port < 0 or login_port > 65535:
 504         module.fail_json(msg="login_port must be a valid unix port number (0-65535)")
 505     ssl_cert = module.params["client_cert"]
 506     ssl_key = module.params["client_key"]
 507     ssl_ca = module.params["ca_cert"]
 508     connect_timeout = module.params['connect_timeout']
 509     config_file = module.params['config_file']
 510     login_password = module.params["login_password"]
 511     login_user = module.params["login_user"]
 512     login_host = module.params["login_host"]
 513     ignore_tables = module.params["ignore_tables"]
 514     for a_table in ignore_tables:
 515         if a_table == "":
 516             module.fail_json(msg="Name of ignored table cannot be empty")
 517     single_transaction = module.params["single_transaction"]
 518     quick = module.params["quick"]
 519     hex_blob = module.params["hex_blob"]
 520     force = module.params["force"]
 521     master_data = module.params["master_data"]
 522     skip_lock_tables = module.params["skip_lock_tables"]
 523     dump_extra_args = module.params["dump_extra_args"]
 524 
 525     if len(db) > 1 and state == 'import':
 526         module.fail_json(msg="Multiple databases are not supported with state=import")
 527     db_name = ' '.join(db)
 528 
 529     all_databases = False
 530     if state in ['dump', 'import']:
 531         if target is None:
 532             module.fail_json(msg="with state=%s target is required" % state)
 533         if db == ['all']:
 534             all_databases = True
 535     else:
 536         if db == ['all']:
 537             module.fail_json(msg="name is not allowed to equal 'all' unless state equals import, or dump.")
 538     try:
 539         cursor, db_conn = mysql_connect(module, login_user, login_password, config_file, ssl_cert, ssl_key, ssl_ca,
 540                                         connect_timeout=connect_timeout)
 541     except Exception as e:
 542         if os.path.exists(config_file):
 543             module.fail_json(msg="unable to connect to database, check login_user and login_password are correct or %s has the credentials. "
 544                                  "Exception message: %s" % (config_file, to_native(e)))
 545         else:
 546             module.fail_json(msg="unable to find %s. Exception message: %s" % (config_file, to_native(e)))
 547 
 548     changed = False
 549     if not os.path.exists(config_file):
 550         config_file = None
 551 
 552     existence_list = []
 553     non_existence_list = []
 554 
 555     if not all_databases:
 556         for each_database in db:
 557             if db_exists(cursor, [each_database]):
 558                 existence_list.append(each_database)
 559             else:
 560                 non_existence_list.append(each_database)
 561 
 562     if state == "absent":
 563         if module.check_mode:
 564             module.exit_json(changed=bool(existence_list), db=db_name, db_list=db)
 565         try:
 566             changed = db_delete(cursor, existence_list)
 567         except Exception as e:
 568             module.fail_json(msg="error deleting database: %s" % to_native(e))
 569         module.exit_json(changed=changed, db=db_name, db_list=db, executed_commands=executed_commands)
 570     elif state == "present":
 571         if module.check_mode:
 572             module.exit_json(changed=bool(non_existence_list), db=db_name, db_list=db)
 573         changed = False
 574         if non_existence_list:
 575             try:
 576                 changed = db_create(cursor, non_existence_list, encoding, collation)
 577             except Exception as e:
 578                 module.fail_json(msg="error creating database: %s" % to_native(e),
 579                                  exception=traceback.format_exc())
 580         module.exit_json(changed=changed, db=db_name, db_list=db, executed_commands=executed_commands)
 581     elif state == "dump":
 582         if non_existence_list and not all_databases:
 583             module.fail_json(msg="Cannot dump database(s) %r - not found" % (', '.join(non_existence_list)))
 584         if module.check_mode:
 585             module.exit_json(changed=True, db=db_name, db_list=db)
 586         rc, stdout, stderr = db_dump(module, login_host, login_user,
 587                                      login_password, db, target, all_databases,
 588                                      login_port, config_file, socket, ssl_cert, ssl_key,
 589                                      ssl_ca, single_transaction, quick, ignore_tables,
 590                                      hex_blob, encoding, force, master_data, skip_lock_tables,
 591                                      dump_extra_args)
 592         if rc != 0:
 593             module.fail_json(msg="%s" % stderr)
 594         module.exit_json(changed=True, db=db_name, db_list=db, msg=stdout,
 595                          executed_commands=executed_commands)
 596     elif state == "import":
 597         if module.check_mode:
 598             module.exit_json(changed=True, db=db_name, db_list=db)
 599         if non_existence_list and not all_databases:
 600             try:
 601                 db_create(cursor, non_existence_list, encoding, collation)
 602             except Exception as e:
 603                 module.fail_json(msg="error creating database: %s" % to_native(e),
 604                                  exception=traceback.format_exc())
 605         rc, stdout, stderr = db_import(module, login_host, login_user,
 606                                        login_password, db, target,
 607                                        all_databases,
 608                                        login_port, config_file,
 609                                        socket, ssl_cert, ssl_key, ssl_ca, encoding, force)
 610         if rc != 0:
