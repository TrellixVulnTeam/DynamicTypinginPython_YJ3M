 400     def compare_to_active(self, other, client, differences):
 401         for k in self.__dict__:
 402             if k in ('advertise_addr', 'listen_addr', 'remote_addrs', 'join_token',
 403                      'rotate_worker_token', 'rotate_manager_token', 'spec',
 404                      'default_addr_pool', 'subnet_size'):
 405                 continue
 406             if not client.option_minimal_versions[k]['supported']:
 407                 continue
 408             value = getattr(self, k)
 409             if value is None:
 410                 continue
 411             other_value = getattr(other, k)
 412             if value != other_value:
 413                 differences.add(k, parameter=value, active=other_value)
 414         if self.rotate_worker_token:
 415             differences.add('rotate_worker_token', parameter=True, active=False)
 416         if self.rotate_manager_token:
 417             differences.add('rotate_manager_token', parameter=True, active=False)
 418         return differences

 421 class SwarmManager(DockerBaseClass):
 422 
 423     def __init__(self, client, results):
 424 
 425         super(SwarmManager, self).__init__()
 426 
 427         self.client = client
 428         self.results = results
 429         self.check_mode = self.client.check_mode
 430         self.swarm_info = {}
 431 
 432         self.state = client.module.params['state']
 433         self.force = client.module.params['force']
 434         self.node_id = client.module.params['node_id']
 435 
 436         self.differences = DifferenceTracker()
 437         self.parameters = TaskParameters.from_ansible_params(client)
 438 
 439         self.created = False
 440 
 441     def __call__(self):
 442         choice_map = {
 443             "present": self.init_swarm,
 444             "join": self.join,
 445             "absent": self.leave,
 446             "remove": self.remove,
 447             "inspect": self.inspect_swarm
 448         }
 449 
 450         if self.state == 'inspect':
 451             self.client.module.deprecate(
 452                 "The 'inspect' state is deprecated, please use 'docker_swarm_info' to inspect swarm cluster",
 453                 version='2.12', collection_name='ansible.builtin')
 454 
 455         choice_map.get(self.state)()
 456 
 457         if self.client.module._diff or self.parameters.debug:
 458             diff = dict()
 459             diff['before'], diff['after'] = self.differences.get_before_after()
 460             self.results['diff'] = diff
 461 
 462     def inspect_swarm(self):
 463         try:
 464             data = self.client.inspect_swarm()
 465             json_str = json.dumps(data, ensure_ascii=False)
 466             self.swarm_info = json.loads(json_str)
 467 
 468             self.results['changed'] = False
 469             self.results['swarm_facts'] = self.swarm_info
 470 
 471             unlock_key = self.get_unlock_key()
 472             self.swarm_info.update(unlock_key)
 473         except APIError:
 474             return
 475 
 476     def get_unlock_key(self):
 477         default = {'UnlockKey': None}
 478         if not self.has_swarm_lock_changed():
 479             return default
 480         try:
 481             return self.client.get_unlock_key() or default
 482         except APIError:
 483             return default
 484 
 485     def has_swarm_lock_changed(self):
 486         return self.parameters.autolock_managers and (
 487             self.created or self.differences.has_difference_for('autolock_managers')
 488         )
 489 
 490     def init_swarm(self):
 491         if not self.force and self.client.check_if_swarm_manager():
 492             self.__update_swarm()
 493             return
 494 
 495         if not self.check_mode:
 496             init_arguments = {
 497                 'advertise_addr': self.parameters.advertise_addr,
 498                 'listen_addr': self.parameters.listen_addr,
 499                 'force_new_cluster': self.force,
 500                 'swarm_spec': self.parameters.spec,
 501             }
 502             if self.parameters.default_addr_pool is not None:
 503                 init_arguments['default_addr_pool'] = self.parameters.default_addr_pool
 504             if self.parameters.subnet_size is not None:
 505                 init_arguments['subnet_size'] = self.parameters.subnet_size
 506             try:
 507                 self.client.init_swarm(**init_arguments)
 508             except APIError as exc:
 509                 self.client.fail("Can not create a new Swarm Cluster: %s" % to_native(exc))
 510 
 511         if not self.client.check_if_swarm_manager():
 512             if not self.check_mode:
 513                 self.client.fail("Swarm not created or other error!")
 514 
 515         self.created = True
 516         self.inspect_swarm()
 517         self.results['actions'].append("New Swarm cluster created: %s" % (self.swarm_info.get('ID')))
 518         self.differences.add('state', parameter='present', active='absent')
 519         self.results['changed'] = True
 520         self.results['swarm_facts'] = {
 521             'JoinTokens': self.swarm_info.get('JoinTokens'),
 522             'UnlockKey': self.swarm_info.get('UnlockKey')
 523         }
 524 
 525     def __update_swarm(self):
 526         try:
 527             self.inspect_swarm()
 528             version = self.swarm_info['Version']['Index']
 529             self.parameters.update_from_swarm_info(self.swarm_info)
 530             old_parameters = TaskParameters()
 531             old_parameters.update_from_swarm_info(self.swarm_info)
 532             self.parameters.compare_to_active(old_parameters, self.client, self.differences)
 533             if self.differences.empty:
 534                 self.results['actions'].append("No modification")
 535                 self.results['changed'] = False
 536                 return
 537             update_parameters = TaskParameters.from_ansible_params(self.client)
 538             update_parameters.update_parameters(self.client)
 539             if not self.check_mode:
 540                 self.client.update_swarm(
 541                     version=version, swarm_spec=update_parameters.spec,
 542                     rotate_worker_token=self.parameters.rotate_worker_token,
 543                     rotate_manager_token=self.parameters.rotate_manager_token)
 544         except APIError as exc:
 545             self.client.fail("Can not update a Swarm Cluster: %s" % to_native(exc))
 546             return
 547 
 548         self.inspect_swarm()
 549         self.results['actions'].append("Swarm cluster updated")
 550         self.results['changed'] = True
 551 
 552     def join(self):
 553         if self.client.check_if_swarm_node():
 554             self.results['actions'].append("This node is already part of a swarm.")
 555             return
 556         if not self.check_mode:
 557             try:
 558                 self.client.join_swarm(
 559                     remote_addrs=self.parameters.remote_addrs, join_token=self.parameters.join_token,
 560                     listen_addr=self.parameters.listen_addr, advertise_addr=self.parameters.advertise_addr)
 561             except APIError as exc:
 562                 self.client.fail("Can not join the Swarm Cluster: %s" % to_native(exc))
 563         self.results['actions'].append("New node is added to swarm cluster")
 564         self.differences.add('joined', parameter=True, active=False)
 565         self.results['changed'] = True
 566 
 567     def leave(self):
 568         if not self.client.check_if_swarm_node():
 569             self.results['actions'].append("This node is not part of a swarm.")
 570             return
 571         if not self.check_mode:
 572             try:
 573                 self.client.leave_swarm(force=self.force)
 574             except APIError as exc:
 575                 self.client.fail("This node can not leave the Swarm Cluster: %s" % to_native(exc))
 576         self.results['actions'].append("Node has left the swarm cluster")
 577         self.differences.add('joined', parameter='absent', active='present')
 578         self.results['changed'] = True
 579 
 580     def remove(self):
 581         if not self.client.check_if_swarm_manager():
 582             self.client.fail("This node is not a manager.")
 583 
 584         try:
 585             status_down = self.client.check_if_swarm_node_is_down(node_id=self.node_id, repeat_check=5)
 586         except APIError:
 587             return
 588 
 589         if not status_down:
 590             self.client.fail("Can not remove the node. The status node is ready and not down.")
 591 
 592         if not self.check_mode:
 593             try:
 594                 self.client.remove_node(node_id=self.node_id, force=self.force)
 595             except APIError as exc:
 596                 self.client.fail("Can not remove the node from the Swarm Cluster: %s" % to_native(exc))
 597         self.results['actions'].append("Node is removed from swarm cluster.")
 598         self.differences.add('joined', parameter=False, active=True)
 599         self.results['changed'] = True
