 345 def _normalize_sku(sku):
 346     if sku is None:
 347         return sku
 348 
 349     sku = sku.upper()
 350     if sku == 'FREE':
 351         return 'F1'
 352     elif sku == 'SHARED':
 353         return 'D1'
 354     return sku
 355 
 356 
 357 def get_sku_name(tier):
 358     tier = tier.upper()
 359     if tier == 'F1' or tier == "FREE":
 360         return 'FREE'
 361     elif tier == 'D1' or tier == "SHARED":
 362         return 'SHARED'
 363     elif tier in ['B1', 'B2', 'B3', 'BASIC']:
 364         return 'BASIC'
 365     elif tier in ['S1', 'S2', 'S3']:
 366         return 'STANDARD'
 367     elif tier in ['P1', 'P2', 'P3']:
 368         return 'PREMIUM'
 369     elif tier in ['P1V2', 'P2V2', 'P3V2']:
 370         return 'PREMIUMV2'
 371     else:
 372         return None

 550     def exec_module(self, **kwargs):
 551         """Main module execution method"""
 552 
 553         for key in list(self.module_arg_spec.keys()) + ['tags']:
 554             if hasattr(self, key):
 555                 setattr(self, key, kwargs[key])
 556             elif kwargs[key] is not None:
 557                 if key == "scm_type":
 558                     self.site_config[key] = kwargs[key]
 559 
 560         old_response = None
 561         response = None
 562         to_be_updated = False
 563 
 564         # set location
 565         resource_group = self.get_resource_group(self.resource_group)
 566         if not self.location:
 567             self.location = resource_group.location
 568 
 569         # get existing web app
 570         old_response = self.get_webapp()
 571 
 572         if old_response:
 573             self.results['id'] = old_response['id']
 574 
 575         if self.state == 'present':
 576             if not self.plan and not old_response:
 577                 self.fail("Please specify plan for newly created web app.")
 578 
 579             if not self.plan:
 580                 self.plan = old_response['server_farm_id']
 581 
 582             self.plan = self.parse_resource_to_dict(self.plan)
 583 
 584             # get app service plan
 585             is_linux = False
 586             old_plan = self.get_app_service_plan()
 587             if old_plan:
 588                 is_linux = old_plan['reserved']
 589             else:
 590                 is_linux = self.plan['is_linux'] if 'is_linux' in self.plan else False
 591 
 592             if self.frameworks:
 593                 # java is mutually exclusive with other frameworks
 594                 if len(self.frameworks) > 1 and any(f['name'] == 'java' for f in self.frameworks):
 595                     self.fail('Java is mutually exclusive with other frameworks.')
 596 
 597                 if is_linux:
 598                     if len(self.frameworks) != 1:
 599                         self.fail('Can specify one framework only for Linux web app.')
 600 
 601                     if self.frameworks[0]['name'] not in self.supported_linux_frameworks:
 602                         self.fail('Unsupported framework {0} for Linux web app.'.format(self.frameworks[0]['name']))
 603 
 604                     self.site_config['linux_fx_version'] = (self.frameworks[0]['name'] + '|' + self.frameworks[0]['version']).upper()
 605 
 606                     if self.frameworks[0]['name'] == 'java':
 607                         if self.frameworks[0]['version'] != '8':
 608                             self.fail("Linux web app only supports java 8.")
 609                         if self.frameworks[0]['settings'] and self.frameworks[0]['settings']['java_container'].lower() != 'tomcat':
 610                             self.fail("Linux web app only supports tomcat container.")
 611 
 612                         if self.frameworks[0]['settings'] and self.frameworks[0]['settings']['java_container'].lower() == 'tomcat':
 613                             self.site_config['linux_fx_version'] = 'TOMCAT|' + self.frameworks[0]['settings']['java_container_version'] + '-jre8'
 614                         else:
 615                             self.site_config['linux_fx_version'] = 'JAVA|8-jre8'
 616                 else:
 617                     for fx in self.frameworks:
 618                         if fx.get('name') not in self.supported_windows_frameworks:
 619                             self.fail('Unsupported framework {0} for Windows web app.'.format(fx.get('name')))
 620                         else:
 621                             self.site_config[fx.get('name') + '_version'] = fx.get('version')
 622 
 623                         if 'settings' in fx and fx['settings'] is not None:
 624                             for key, value in fx['settings'].items():
 625                                 self.site_config[key] = value
 626 
 627             if not self.app_settings:
 628                 self.app_settings = dict()
 629 
 630             if self.container_settings:
 631                 linux_fx_version = 'DOCKER|'
 632 
 633                 if self.container_settings.get('registry_server_url'):
 634                     self.app_settings['DOCKER_REGISTRY_SERVER_URL'] = 'https://' + self.container_settings['registry_server_url']
 635 
 636                     linux_fx_version += self.container_settings['registry_server_url'] + '/'
 637 
 638                 linux_fx_version += self.container_settings['name']
 639 
 640                 self.site_config['linux_fx_version'] = linux_fx_version
 641 
 642                 if self.container_settings.get('registry_server_user'):
 643                     self.app_settings['DOCKER_REGISTRY_SERVER_USERNAME'] = self.container_settings['registry_server_user']
 644 
 645                 if self.container_settings.get('registry_server_password'):
 646                     self.app_settings['DOCKER_REGISTRY_SERVER_PASSWORD'] = self.container_settings['registry_server_password']
 647 
 648             # init site
 649             self.site = Site(location=self.location, site_config=self.site_config)
 650 
 651             if self.https_only is not None:
 652                 self.site.https_only = self.https_only
 653 
 654             if self.client_affinity_enabled:
 655                 self.site.client_affinity_enabled = self.client_affinity_enabled
 656 
 657             # check if the web app already present in the resource group
 658             if not old_response:
 659                 self.log("Web App instance doesn't exist")
 660 
 661                 to_be_updated = True
 662                 self.to_do.append(Actions.CreateOrUpdate)
 663                 self.site.tags = self.tags
 664 
 665                 # service plan is required for creation
 666                 if not self.plan:
 667                     self.fail("Please specify app service plan in plan parameter.")
 668 
 669                 if not old_plan:
 670                     # no existing service plan, create one
 671                     if (not self.plan.get('name') or not self.plan.get('sku')):
 672                         self.fail('Please specify name, is_linux, sku in plan')
 673 
 674                     if 'location' not in self.plan:
 675                         plan_resource_group = self.get_resource_group(self.plan['resource_group'])
 676                         self.plan['location'] = plan_resource_group.location
 677 
 678                     old_plan = self.create_app_service_plan()
 679 
 680                 self.site.server_farm_id = old_plan['id']
 681 
 682                 # if linux, setup startup_file
 683                 if old_plan['is_linux']:
 684                     if hasattr(self, 'startup_file'):
 685                         self.site_config['app_command_line'] = self.startup_file
 686 
 687                 # set app setting
 688                 if self.app_settings:
 689                     app_settings = []
 690                     for key in self.app_settings.keys():
 691                         app_settings.append(NameValuePair(name=key, value=self.app_settings[key]))
 692 
 693                     self.site_config['app_settings'] = app_settings
 694             else:
 695                 # existing web app, do update
 696                 self.log("Web App instance already exists")
 697 
 698                 self.log('Result: {0}'.format(old_response))
 699 
 700                 update_tags, self.site.tags = self.update_tags(old_response.get('tags', None))
 701 
 702                 if update_tags:
 703                     to_be_updated = True
 704 
 705                 # check if root level property changed
 706                 if self.is_updatable_property_changed(old_response):
 707                     to_be_updated = True
 708                     self.to_do.append(Actions.CreateOrUpdate)
 709 
 710                 # check if site_config changed
 711                 old_config = self.get_webapp_configuration()
 712 
 713                 if self.is_site_config_changed(old_config):
 714                     to_be_updated = True
 715                     self.to_do.append(Actions.CreateOrUpdate)
 716 
 717                 # check if linux_fx_version changed
 718                 if old_config.linux_fx_version != self.site_config.get('linux_fx_version', ''):
 719                     to_be_updated = True
 720                     self.to_do.append(Actions.CreateOrUpdate)
 721 
 722                 self.app_settings_strDic = self.list_app_settings()
 723 
 724                 # purge existing app_settings:
 725                 if self.purge_app_settings:
 726                     to_be_updated = True
 727                     self.app_settings_strDic = dict()
 728                     self.to_do.append(Actions.UpdateAppSettings)
 729 
 730                 # check if app settings changed
 731                 if self.purge_app_settings or self.is_app_settings_changed():
 732                     to_be_updated = True
 733                     self.to_do.append(Actions.UpdateAppSettings)
 734 
 735                     if self.app_settings:
 736                         for key in self.app_settings.keys():
 737                             self.app_settings_strDic[key] = self.app_settings[key]
 738 
 739         elif self.state == 'absent':
 740             if old_response:
 741                 self.log("Delete Web App instance")
 742                 self.results['changed'] = True
 743 
 744                 if self.check_mode:
 745                     return self.results
 746 
 747                 self.delete_webapp()
 748 
 749                 self.log('Web App instance deleted')
 750 
 751             else:
 752                 self.fail("Web app {0} not exists.".format(self.name))
 753 
 754         if to_be_updated:
 755             self.log('Need to Create/Update web app')
 756             self.results['changed'] = True
 757 
 758             if self.check_mode:
 759                 return self.results
 760 
 761             if Actions.CreateOrUpdate in self.to_do:
 762                 response = self.create_update_webapp()
 763 
 764                 self.results['id'] = response['id']
 765 
 766             if Actions.UpdateAppSettings in self.to_do:
 767                 update_response = self.update_app_settings()
 768                 self.results['id'] = update_response.id
 769 
 770         webapp = None
 771         if old_response:
 772             webapp = old_response
 773         if response:
 774             webapp = response
 775 
 776         if webapp:
 777             if (webapp['state'] != 'Stopped' and self.app_state == 'stopped') or \
 778                (webapp['state'] != 'Running' and self.app_state == 'started') or \
 779                self.app_state == 'restarted':
 780 
 781                 self.results['changed'] = True
 782                 if self.check_mode:
 783                     return self.results
 784 
 785                 self.set_webapp_state(self.app_state)
 786 
 787         return self.results

 789     # compare existing web app with input, determine weather it's update operation
 790     def is_updatable_property_changed(self, existing_webapp):
 791         for property_name in self.updatable_properties:
 792             if hasattr(self, property_name) and getattr(self, property_name) is not None and \
 793                     getattr(self, property_name) != existing_webapp.get(property_name, None):
 794                 return True
 795 
 796         return False

 798     # compare xxx_version
 799     def is_site_config_changed(self, existing_config):
 800         for fx_version in self.site_config_updatable_properties:
 801             if self.site_config.get(fx_version):
 802                 if not getattr(existing_config, fx_version) or \
 803                         getattr(existing_config, fx_version).upper() != self.site_config.get(fx_version).upper():
 804                     return True
 805 
 806         return False

 808     # comparing existing app setting with input, determine whether it's changed
 809     def is_app_settings_changed(self):
 810         if self.app_settings:
 811             if self.app_settings_strDic:
 812                 for key in self.app_settings.keys():
 813                     if self.app_settings[key] != self.app_settings_strDic.get(key, None):
 814                         return True
 815             else:
 816                 return True
 817         return False

 823                     and self.deployment_source['url'] != existing_webapp.get('site_source_control')['url']:

 827                     and self.deployment_source['branch'] != existing_webapp.get('site_source_control')['branch']:
 828                 return True
 829
