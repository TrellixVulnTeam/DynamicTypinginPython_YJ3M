 319 def key_check(module, s3, bucket, obj, version=None, validate=True):
 320     exists = True
 321     try:
 322         if version:
 323             s3.head_object(Bucket=bucket, Key=obj, VersionId=version)
 324         else:
 325             s3.head_object(Bucket=bucket, Key=obj)
 326     except botocore.exceptions.ClientError as e:
 327         # if a client error is thrown, check if it's a 404 error
 328         # if it's a 404 error, then the object does not exist
 329         error_code = int(e.response['Error']['Code'])
 330         if error_code == 404:
 331             exists = False
 332         elif error_code == 403 and validate is False:
 333             pass
 334         else:
 335             module.fail_json_aws(e, msg="Failed while looking up object (during key check) %s." % obj)
 336     except botocore.exceptions.BotoCoreError as e:
 337         module.fail_json_aws(e, msg="Failed while looking up object (during key check) %s." % obj)
 338     return exists

 341 def etag_compare(module, local_file, s3, bucket, obj, version=None):
 342     s3_etag = get_etag(s3, bucket, obj, version=version)
 343     local_etag = calculate_etag(module, local_file, s3_etag, s3, bucket, obj, version)
 344 
 345     return s3_etag == local_etag

 348 def get_etag(s3, bucket, obj, version=None):
 349     if version:
 350         key_check = s3.head_object(Bucket=bucket, Key=obj, VersionId=version)
 351     else:
 352         key_check = s3.head_object(Bucket=bucket, Key=obj)
 353     if not key_check:
 354         return None
 355     return key_check['ETag']

 358 def bucket_check(module, s3, bucket, validate=True):
 359     exists = True
 360     try:
 361         s3.head_bucket(Bucket=bucket)
 362     except botocore.exceptions.ClientError as e:
 363         # If a client error is thrown, then check that it was a 404 error.
 364         # If it was a 404 error, then the bucket does not exist.
 365         error_code = int(e.response['Error']['Code'])
 366         if error_code == 404:
 367             exists = False
 368         elif error_code == 403 and validate is False:
 369             pass
 370         else:
 371             module.fail_json_aws(e, msg="Failed while looking up bucket (during bucket_check) %s." % bucket)
 372     except botocore.exceptions.EndpointConnectionError as e:
 373         module.fail_json_aws(e, msg="Invalid endpoint provided")
 374     except botocore.exceptions.BotoCoreError as e:
 375         module.fail_json_aws(e, msg="Failed while looking up bucket (during bucket_check) %s." % bucket)
 376     return exists

 379 def create_bucket(module, s3, bucket, location=None):
 380     if module.check_mode:
 381         module.exit_json(msg="CREATE operation skipped - running in check mode", changed=True)
 382     configuration = {}
 383     if location not in ('us-east-1', None):
 384         configuration['LocationConstraint'] = location
 385     try:
 386         if len(configuration) > 0:
 387             s3.create_bucket(Bucket=bucket, CreateBucketConfiguration=configuration)
 388         else:
 389             s3.create_bucket(Bucket=bucket)
 390         if module.params.get('permission'):
 391             # Wait for the bucket to exist before setting ACLs
 392             s3.get_waiter('bucket_exists').wait(Bucket=bucket)
 393         for acl in module.params.get('permission'):
 394             s3.put_bucket_acl(ACL=acl, Bucket=bucket)
 395     except botocore.exceptions.ClientError as e:
 396         if e.response['Error']['Code'] in IGNORE_S3_DROP_IN_EXCEPTIONS:
 397             module.warn("PutBucketAcl is not implemented by your storage provider. Set the permission parameters to the empty list to avoid this warning")
 398         else:
 399             module.fail_json_aws(e, msg="Failed while creating bucket or setting acl (check that you have CreateBucket and PutBucketAcl permission).")
 400     except botocore.exceptions.BotoCoreError as e:
 401         module.fail_json_aws(e, msg="Failed while creating bucket or setting acl (check that you have CreateBucket and PutBucketAcl permission).")
 402 
 403     if bucket:
 404         return True

 407 def paginated_list(s3, **pagination_params):
 408     pg = s3.get_paginator('list_objects_v2')
 409     for page in pg.paginate(**pagination_params):
 410         yield [data['Key'] for data in page.get('Contents', [])]

 413 def paginated_versioned_list_with_fallback(s3, **pagination_params):
 414     try:
 415         versioned_pg = s3.get_paginator('list_object_versions')
 416         for page in versioned_pg.paginate(**pagination_params):
 417             delete_markers = [{'Key': data['Key'], 'VersionId': data['VersionId']} for data in page.get('DeleteMarkers', [])]
 418             current_objects = [{'Key': data['Key'], 'VersionId': data['VersionId']} for data in page.get('Versions', [])]
 419             yield delete_markers + current_objects
 420     except botocore.exceptions.ClientError as e:
 421         if to_text(e.response['Error']['Code']) in IGNORE_S3_DROP_IN_EXCEPTIONS + ['AccessDenied']:
 422             for page in paginated_list(s3, **pagination_params):
 423                 yield [{'Key': data['Key']} for data in page]

 426 def list_keys(module, s3, bucket, prefix, marker, max_keys):
 427     pagination_params = {'Bucket': bucket}
 428     for param_name, param_value in (('Prefix', prefix), ('StartAfter', marker), ('MaxKeys', max_keys)):
 429         pagination_params[param_name] = param_value
 430     try:
 431         keys = sum(paginated_list(s3, **pagination_params), [])
 432         module.exit_json(msg="LIST operation complete", s3_keys=keys)
 433     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 434         module.fail_json_aws(e, msg="Failed while listing the keys in the bucket {0}".format(bucket))

 437 def delete_bucket(module, s3, bucket):
 438     if module.check_mode:
 439         module.exit_json(msg="DELETE operation skipped - running in check mode", changed=True)
 440     try:
 441         exists = bucket_check(module, s3, bucket)
 442         if exists is False:
 443             return False
 444         # if there are contents then we need to delete them before we can delete the bucket
 445         for keys in paginated_versioned_list_with_fallback(s3, Bucket=bucket):
 446             if keys:
 447                 s3.delete_objects(Bucket=bucket, Delete={'Objects': keys})
 448         s3.delete_bucket(Bucket=bucket)
 449         return True
 450     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 451         module.fail_json_aws(e, msg="Failed while deleting bucket %s." % bucket)

 464 def create_dirkey(module, s3, bucket, obj, encrypt):
 465     if module.check_mode:
 466         module.exit_json(msg="PUT operation skipped - running in check mode", changed=True)
 467     try:
 468         params = {'Bucket': bucket, 'Key': obj, 'Body': b''}
 469         if encrypt:
 470             params['ServerSideEncryption'] = module.params['encryption_mode']
 471         if module.params['encryption_kms_key_id'] and module.params['encryption_mode'] == 'aws:kms':
 472             params['SSEKMSKeyId'] = module.params['encryption_kms_key_id']
 473 
 474         s3.put_object(**params)
 475         for acl in module.params.get('permission'):
 476             s3.put_object_acl(ACL=acl, Bucket=bucket, Key=obj)
 477     except botocore.exceptions.ClientError as e:
 478         if e.response['Error']['Code'] in IGNORE_S3_DROP_IN_EXCEPTIONS:
 479             module.warn("PutObjectAcl is not implemented by your storage provider. Set the permissions parameters to the empty list to avoid this warning")
 480         else:
 481             module.fail_json_aws(e, msg="Failed while creating object %s." % obj)
 482     except botocore.exceptions.BotoCoreError as e:
 483         module.fail_json_aws(e, msg="Failed while creating object %s." % obj)
 484     module.exit_json(msg="Virtual directory %s created in bucket %s" % (obj, bucket), changed=True)

 509 def upload_s3file(module, s3, bucket, obj, src, expiry, metadata, encrypt, headers):
 510     if module.check_mode:
 511         module.exit_json(msg="PUT operation skipped - running in check mode", changed=True)
 512     try:
 513         extra = {}
 514         if encrypt:
 515             extra['ServerSideEncryption'] = module.params['encryption_mode']
 516         if module.params['encryption_kms_key_id'] and module.params['encryption_mode'] == 'aws:kms':
 517             extra['SSEKMSKeyId'] = module.params['encryption_kms_key_id']
 518         if metadata:
 519             extra['Metadata'] = {}
 520 
 521             # determine object metadata and extra arguments
 522             for option in metadata:
 523                 extra_args_option = option_in_extra_args(option)
 524                 if extra_args_option is not None:
 525                     extra[extra_args_option] = metadata[option]
 526                 else:
 527                     extra['Metadata'][option] = metadata[option]
 528 
 529         if 'ContentType' not in extra:
 530             content_type = mimetypes.guess_type(src)[0]
 531             if content_type is None:
 532                 # s3 default content type
 533                 content_type = 'binary/octet-stream'
 534             extra['ContentType'] = content_type
 535 
 536         s3.upload_file(Filename=src, Bucket=bucket, Key=obj, ExtraArgs=extra)
 537     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 538         module.fail_json_aws(e, msg="Unable to complete PUT operation.")
 539     try:
 540         for acl in module.params.get('permission'):
 541             s3.put_object_acl(ACL=acl, Bucket=bucket, Key=obj)
 542     except botocore.exceptions.ClientError as e:
 543         if e.response['Error']['Code'] in IGNORE_S3_DROP_IN_EXCEPTIONS:
 544             module.warn("PutObjectAcl is not implemented by your storage provider. Set the permission parameters to the empty list to avoid this warning")
 545         else:
 546             module.fail_json_aws(e, msg="Unable to set object ACL")
 547     except botocore.exceptions.BotoCoreError as e:
 548         module.fail_json_aws(e, msg="Unable to set object ACL")
 549     try:
 550         url = s3.generate_presigned_url(ClientMethod='put_object',
 551                                         Params={'Bucket': bucket, 'Key': obj},
 552                                         ExpiresIn=expiry)
 553     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 554         module.fail_json_aws(e, msg="Unable to generate presigned URL")
 555     module.exit_json(msg="PUT operation complete", url=url, changed=True)

 558 def download_s3file(module, s3, bucket, obj, dest, retries, version=None):
 559     if module.check_mode:
 560         module.exit_json(msg="GET operation skipped - running in check mode", changed=True)
 561     # retries is the number of loops; range/xrange needs to be one
 562     # more to get that count of loops.
 563     try:
 564         if version:
 565             key = s3.get_object(Bucket=bucket, Key=obj, VersionId=version)
 566         else:
 567             key = s3.get_object(Bucket=bucket, Key=obj)
 568     except botocore.exceptions.ClientError as e:
 569         if e.response['Error']['Code'] == 'InvalidArgument' and 'require AWS Signature Version 4' in to_text(e):
 570             raise Sigv4Required()
 571         elif e.response['Error']['Code'] not in ("403", "404"):
 572             # AccessDenied errors may be triggered if 1) file does not exist or 2) file exists but
 573             # user does not have the s3:GetObject permission. 404 errors are handled by download_file().
 574             module.fail_json_aws(e, msg="Could not find the key %s." % obj)
 575     except botocore.exceptions.BotoCoreError as e:
 576         module.fail_json_aws(e, msg="Could not find the key %s." % obj)
 577 
 578     optional_kwargs = {'ExtraArgs': {'VersionId': version}} if version else {}
 579     for x in range(0, retries + 1):
 580         try:
 581             s3.download_file(bucket, obj, dest, **optional_kwargs)
 582             module.exit_json(msg="GET operation complete", changed=True)
 583         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 584             # actually fail on last pass through the loop.
 585             if x >= retries:
 586                 module.fail_json_aws(e, msg="Failed while downloading %s." % obj)
 587             # otherwise, try again, this may be a transient timeout.
 588         except SSLError as e:  # will ClientError catch SSLError?
 589             # actually fail on last pass through the loop.
 590             if x >= retries:
 591                 module.fail_json_aws(e, msg="s3 download failed")
 592             # otherwise, try again, this may be a transient timeout.

 595 def download_s3str(module, s3, bucket, obj, version=None, validate=True):
 596     if module.check_mode:
 597         module.exit_json(msg="GET operation skipped - running in check mode", changed=True)
 598     try:
 599         if version:
 600             contents = to_native(s3.get_object(Bucket=bucket, Key=obj, VersionId=version)["Body"].read())
 601         else:
 602             contents = to_native(s3.get_object(Bucket=bucket, Key=obj)["Body"].read())
 603         module.exit_json(msg="GET operation complete", contents=contents, changed=True)
 604     except botocore.exceptions.ClientError as e:
 605         if e.response['Error']['Code'] == 'InvalidArgument' and 'require AWS Signature Version 4' in to_text(e):
 606             raise Sigv4Required()
 607         else:
 608             module.fail_json_aws(e, msg="Failed while getting contents of object %s as a string." % obj)
 609     except botocore.exceptions.BotoCoreError as e:
 610         module.fail_json_aws(e, msg="Failed while getting contents of object %s as a string." % obj)

 613 def get_download_url(module, s3, bucket, obj, expiry, changed=True):
 614     try:
 615         url = s3.generate_presigned_url(ClientMethod='get_object',
 616                                         Params={'Bucket': bucket, 'Key': obj},
 617                                         ExpiresIn=expiry)
 618         module.exit_json(msg="Download url:", url=url, expiry=expiry, changed=changed)
 619     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 620         module.fail_json_aws(e, msg="Failed while getting download url.")

 623 def is_fakes3(s3_url):
 624     """ Return True if s3_url has scheme fakes3:// """
 625     if s3_url is not None:
 626         return urlparse(s3_url).scheme in ('fakes3', 'fakes3s')
 627     else:
 628         return False

 631 def get_s3_connection(module, aws_connect_kwargs, location, rgw, s3_url, sig_4=False):
 632     if s3_url and rgw:  # TODO - test this
 633         rgw = urlparse(s3_url)
 634         params = dict(module=module, conn_type='client', resource='s3', use_ssl=rgw.scheme == 'https', region=location, endpoint=s3_url, **aws_connect_kwargs)
 635     elif is_fakes3(s3_url):
 636         fakes3 = urlparse(s3_url)
 637         port = fakes3.port
 638         if fakes3.scheme == 'fakes3s':
 639             protocol = "https"
 640             if port is None:
 641                 port = 443
 642         else:
 643             protocol = "http"
 644             if port is None:
 645                 port = 80
 646         params = dict(module=module, conn_type='client', resource='s3', region=location,
 647                       endpoint="%s://%s:%s" % (protocol, fakes3.hostname, to_text(port)),
 648                       use_ssl=fakes3.scheme == 'fakes3s', **aws_connect_kwargs)
 649     else:
 650         params = dict(module=module, conn_type='client', resource='s3', region=location, endpoint=s3_url, **aws_connect_kwargs)
 651         if module.params['mode'] == 'put' and module.params['encryption_mode'] == 'aws:kms':
 652             params['config'] = botocore.client.Config(signature_version='s3v4')
 653         elif module.params['mode'] in ('get', 'getstr') and sig_4:
 654             params['config'] = botocore.client.Config(signature_version='s3v4')
 655         if module.params['dualstack']:
 656             dualconf = botocore.client.Config(s3={'use_dualstack_endpoint': True})
 657             if 'config' in params:
 658                 params['config'] = params['config'].merge(dualconf)
 659             else:
 660                 params['config'] = dualconf
 661     return boto3_conn(**params)

 719     bucket_canned_acl = ["private", "public-read", "public-read-write", "authenticated-read"]
