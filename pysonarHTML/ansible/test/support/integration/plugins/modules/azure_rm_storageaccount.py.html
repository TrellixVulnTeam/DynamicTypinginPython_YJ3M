 317 def compare_cors(cors1, cors2):
 318     if len(cors1) != len(cors2):
 319         return False
 320     copy2 = copy.copy(cors2)
 321     for rule1 in cors1:
 322         matched = False
 323         for rule2 in copy2:
 324             if (rule1['max_age_in_seconds'] == rule2['max_age_in_seconds']
 325                     and set(rule1['allowed_methods']) == set(rule2['allowed_methods'])
 326                     and set(rule1['allowed_origins']) == set(rule2['allowed_origins'])
 327                     and set(rule1['allowed_headers']) == set(rule2['allowed_headers'])
 328                     and set(rule1['exposed_headers']) == set(rule2['exposed_headers'])):
 329                 matched = True
 330                 copy2.remove(rule2)
 331         if not matched:
 332             return False
 333     return True

 379     def exec_module(self, **kwargs):
 380 
 381         for key in list(self.module_arg_spec.keys()) + ['tags']:
 382             setattr(self, key, kwargs[key])
 383 
 384         resource_group = self.get_resource_group(self.resource_group)
 385         if not self.location:
 386             # Set default location
 387             self.location = resource_group.location
 388 
 389         if len(self.name) < 3 or len(self.name) > 24:
 390             self.fail("Parameter error: name length must be between 3 and 24 characters.")
 391 
 392         if self.custom_domain:
 393             if self.custom_domain.get('name', None) is None:
 394                 self.fail("Parameter error: expecting custom_domain to have a name attribute of type string.")
 395             if self.custom_domain.get('use_sub_domain', None) is None:
 396                 self.fail("Parameter error: expecting custom_domain to have a use_sub_domain "
 397                           "attribute of type boolean.")

 399         self.account_dict = self.get_account()
 400 
 401         if self.state == 'present' and self.account_dict and \
 402            self.account_dict['provisioning_state'] != AZURE_SUCCESS_STATE:
 403             self.fail("Error: storage account {0} has not completed provisioning. State is {1}. Expecting state "
 404                       "to be {2}.".format(self.name, self.account_dict['provisioning_state'], AZURE_SUCCESS_STATE))
 405 
 406         if self.account_dict is not None:
 407             self.results['state'] = self.account_dict
 408         else:
 409             self.results['state'] = dict()
 410 
 411         if self.state == 'present':
 412             if not self.account_dict:
 413                 self.results['state'] = self.create_account()
 414             else:
 415                 self.update_account()
 416         elif self.state == 'absent' and self.account_dict:
 417             self.delete_account()
 418             self.results['state'] = dict(Status='Deleted')
 419

 504     def update_account(self):
 505         self.log('Update storage account {0}'.format(self.name))
 506         if bool(self.https_only) != bool(self.account_dict.get('https_only')):
 507             self.results['changed'] = True
 508             self.account_dict['https_only'] = self.https_only
 509             if not self.check_mode:
 510                 try:
 511                     parameters = self.storage_models.StorageAccountUpdateParameters(enable_https_traffic_only=self.https_only)
 512                     self.storage_client.storage_accounts.update(self.resource_group,
 513                                                                 self.name,
 514                                                                 parameters)
 515                 except Exception as exc:
 516                     self.fail("Failed to update account type: {0}".format(str(exc)))
 517 
 518         if self.account_type:
 519             if self.account_type != self.account_dict['sku_name']:
 520                 # change the account type
 521                 SkuName = self.storage_models.SkuName
 522                 if self.account_dict['sku_name'] in [SkuName.premium_lrs, SkuName.standard_zrs]:
 523                     self.fail("Storage accounts of type {0} and {1} cannot be changed.".format(
 524                         SkuName.premium_lrs, SkuName.standard_zrs))
 525                 if self.account_type in [SkuName.premium_lrs, SkuName.standard_zrs]:
 526                     self.fail("Storage account of type {0} cannot be changed to a type of {1} or {2}.".format(
 527                         self.account_dict['sku_name'], SkuName.premium_lrs, SkuName.standard_zrs))
 528 
 529                 self.results['changed'] = True
 530                 self.account_dict['sku_name'] = self.account_type
 531 
 532                 if self.results['changed'] and not self.check_mode:
 533                     # Perform the update. The API only allows changing one attribute per call.
 534                     try:
 535                         self.log("sku_name: %s" % self.account_dict['sku_name'])
 536                         self.log("sku_tier: %s" % self.account_dict['sku_tier'])
 537                         sku = self.storage_models.Sku(name=SkuName(self.account_dict['sku_name']))
 538                         sku.tier = self.storage_models.SkuTier(self.account_dict['sku_tier'])
 539                         parameters = self.storage_models.StorageAccountUpdateParameters(sku=sku)
 540                         self.storage_client.storage_accounts.update(self.resource_group,
 541                                                                     self.name,
 542                                                                     parameters)
 543                     except Exception as exc:
 544                         self.fail("Failed to update account type: {0}".format(str(exc)))
 545 
 546         if self.custom_domain:
 547             if not self.account_dict['custom_domain'] or self.account_dict['custom_domain'] != self.custom_domain:
 548                 self.results['changed'] = True
 549                 self.account_dict['custom_domain'] = self.custom_domain
 550 
 551             if self.results['changed'] and not self.check_mode:
 552                 new_domain = self.storage_models.CustomDomain(name=self.custom_domain['name'],
 553                                                               use_sub_domain=self.custom_domain['use_sub_domain'])
 554                 parameters = self.storage_models.StorageAccountUpdateParameters(custom_domain=new_domain)
 555                 try:
 556                     self.storage_client.storage_accounts.update(self.resource_group, self.name, parameters)
 557                 except Exception as exc:
 558                     self.fail("Failed to update custom domain: {0}".format(str(exc)))
 559 
 560         if self.access_tier:
 561             if not self.account_dict['access_tier'] or self.account_dict['access_tier'] != self.access_tier:
 562                 self.results['changed'] = True
 563                 self.account_dict['access_tier'] = self.access_tier
 564 
 565             if self.results['changed'] and not self.check_mode:
 566                 parameters = self.storage_models.StorageAccountUpdateParameters(access_tier=self.access_tier)
 567                 try:
 568                     self.storage_client.storage_accounts.update(self.resource_group, self.name, parameters)
 569                 except Exception as exc:
 570                     self.fail("Failed to update access tier: {0}".format(str(exc)))
 571 
 572         update_tags, self.account_dict['tags'] = self.update_tags(self.account_dict['tags'])
 573         if update_tags:
 574             self.results['changed'] = True
 575             if not self.check_mode:
 576                 parameters = self.storage_models.StorageAccountUpdateParameters(tags=self.account_dict['tags'])
 577                 try:
 578                     self.storage_client.storage_accounts.update(self.resource_group, self.name, parameters)
 579                 except Exception as exc:
 580                     self.fail("Failed to update tags: {0}".format(str(exc)))
 581 
 582         if self.blob_cors and not compare_cors(self.account_dict.get('blob_cors', []), self.blob_cors):
 583             self.results['changed'] = True
 584             if not self.check_mode:
 585                 self.set_blob_cors()

 587     def create_account(self):
 588         self.log("Creating account {0}".format(self.name))
 589 
 590         if not self.location:
 591             self.fail('Parameter error: location required when creating a storage account.')
 592 
 593         if not self.account_type:
 594             self.fail('Parameter error: account_type required when creating a storage account.')
 595 
 596         if not self.access_tier and self.kind == 'BlobStorage':
 597             self.fail('Parameter error: access_tier required when creating a storage account of type BlobStorage.')
 598 
 599         self.check_name_availability()
 600         self.results['changed'] = True
 601 
 602         if self.check_mode:
 603             account_dict = dict(
 604                 location=self.location,
 605                 account_type=self.account_type,
 606                 name=self.name,
 607                 resource_group=self.resource_group,
 608                 enable_https_traffic_only=self.https_only,
 609                 tags=dict()
 610             )
 611             if self.tags:
 612                 account_dict['tags'] = self.tags
 613             if self.blob_cors:
 614                 account_dict['blob_cors'] = self.blob_cors
 615             return account_dict
 616         sku = self.storage_models.Sku(name=self.storage_models.SkuName(self.account_type))
 617         sku.tier = self.storage_models.SkuTier.standard if 'Standard' in self.account_type else \
 618             self.storage_models.SkuTier.premium
 619         parameters = self.storage_models.StorageAccountCreateParameters(sku=sku,
 620                                                                         kind=self.kind,
 621                                                                         location=self.location,
 622                                                                         tags=self.tags,
 623                                                                         access_tier=self.access_tier)
 624         self.log(str(parameters))
 625         try:
 626             poller = self.storage_client.storage_accounts.create(self.resource_group, self.name, parameters)
 627             self.get_poller_result(poller)
 628         except CloudError as e:
 629             self.log('Error creating storage account.')
 630             self.fail("Failed to create account: {0}".format(str(e)))
 631         if self.blob_cors:
 632             self.set_blob_cors()
 633         # the poller doesn't actually return anything
 634         return self.get_account()
 635 
 636     def delete_account(self):
 637         if self.account_dict['provisioning_state'] == self.storage_models.ProvisioningState.succeeded.value and \
 638            not self.force_delete_nonempty and self.account_has_blob_containers():
 639             self.fail("Account contains blob containers. Is it in use? Use the force_delete_nonempty option to attempt deletion.")
 640 
 641         self.log('Delete storage account {0}'.format(self.name))
 642         self.results['changed'] = True
 643         if not self.check_mode:
 644             try:
 645                 status = self.storage_client.storage_accounts.delete(self.resource_group, self.name)
 646                 self.log("delete status: ")
 647                 self.log(str(status))
 648             except CloudError as e:
 649                 self.fail("Failed to delete the account: {0}".format(str(e)))
 650         return True
 651 
 652     def account_has_blob_containers(self):
 653         '''
 654         If there are blob containers, then there are likely VMs depending on this account and it should
 655         not be deleted.
 656         '''
 657         self.log('Checking for existing blob containers')
 658         blob_service = self.get_blob_client(self.resource_group, self.name)
 659         try:
 660             response = blob_service.list_containers()
 661         except AzureMissingResourceHttpError:
 662             # No blob storage available?
 663             return False
 664 
 665         if len(response.items) > 0:
 666             return True
 667         return False

 669     def set_blob_cors(self):
 670         try:
 671             cors_rules = self.storage_models.CorsRules(cors_rules=[self.storage_models.CorsRule(**x) for x in self.blob_cors])
 672             self.storage_client.blob_services.set_service_properties(self.resource_group,
 673                                                                      self.name,
 674                                                                      self.storage_models.BlobServiceProperties(cors=cors_rules))
 675         except Exception as exc:
 676             self.fail("Failed to set CORS rules: {0}".format(str(exc)))
