 180 from ansible.module_utils.basic import AnsibleModule
 181 from ansible.module_utils.net_tools.nios.api import WapiModule
 182 from ansible.module_utils.net_tools.nios.api import NIOS_ZONE
 183 
 184 
 185 def main():
 186     ''' Main entry point for module execution
 187     '''
 188     grid_spec = dict(
 189         name=dict(required=True),
 190     )
 191 
 192     ib_spec = dict(
 193         fqdn=dict(required=True, aliases=['name'], ib_req=True, update=False),
 194         zone_format=dict(default='FORWARD', aliases=['zone_format'], ib_req=False),
 195         view=dict(default='default', aliases=['dns_view'], ib_req=True),
 196 
 197         grid_primary=dict(type='list', elements='dict', options=grid_spec),
 198         grid_secondaries=dict(type='list', elements='dict', options=grid_spec),
 199         ns_group=dict(),
 200         restart_if_needed=dict(type='bool'),
 201 
 202         extattrs=dict(type='dict'),
 203         comment=dict()
 204     )
 205 
 206     argument_spec = dict(
 207         provider=dict(required=True),
 208         state=dict(default='present', choices=['present', 'absent'])
 209     )
 210 
 211     argument_spec.update(ib_spec)
 212     argument_spec.update(WapiModule.provider_spec)
 213 
 214     module = AnsibleModule(argument_spec=argument_spec,
 215                            supports_check_mode=True,
 216                            mutually_exclusive=[
 217                                ['ns_group', 'grid_primary'],
 218                                ['ns_group', 'grid_secondaries']
 219                            ])
 220 
 221     wapi = WapiModule(module)
 222     result = wapi.run(NIOS_ZONE, ib_spec)
 223 
 224     module.exit_json(**result)
 225 
 226 
 227 if __name__ == '__main__':
 228     main()
