 123 def _normalize_sku(sku):
 124     if sku is None:
 125         return sku
 126 
 127     sku = sku.upper()
 128     if sku == 'FREE':
 129         return 'F1'
 130     elif sku == 'SHARED':
 131         return 'D1'
 132     return sku
 133 
 134 
 135 def get_sku_name(tier):
 136     tier = tier.upper()
 137     if tier == 'F1' or tier == "FREE":
 138         return 'FREE'
 139     elif tier == 'D1' or tier == "SHARED":
 140         return 'SHARED'
 141     elif tier in ['B1', 'B2', 'B3', 'BASIC']:
 142         return 'BASIC'
 143     elif tier in ['S1', 'S2', 'S3']:
 144         return 'STANDARD'
 145     elif tier in ['P1', 'P2', 'P3']:
 146         return 'PREMIUM'
 147     elif tier in ['P1V2', 'P2V2', 'P3V2']:
 148         return 'PREMIUMV2'
 149     else:
 150         return None

 153 def appserviceplan_to_dict(plan):

 155         id=plan.id,
 156         name=plan.name,
 157         kind=plan.kind,
 158         location=plan.location,

 161         provisioning_state=plan.provisioning_state,
 162         status=plan.status,
 163         target_worker_count=plan.target_worker_count,
 164         sku=dict(
 165             name=plan.sku.name,
 166             size=plan.sku.size,
 167             tier=plan.sku.tier,
 168             family=plan.sku.family,
 169             capacity=plan.sku.capacity
 170         ),
 171         resource_group=plan.resource_group,
 172         number_of_sites=plan.number_of_sites,
 173         tags=plan.tags if plan.tags else None
 174     )

 200             number_of_workers=dict(
 201                 type='str'
 202             ),
 203             state=dict(
 204                 type='str',
 205                 default='present',

 207             )

 210         self.resource_group = None
 211         self.name = None
 212         self.location = None

 214         self.sku = None

 222             ansible_facts=dict(azure_appserviceplan=None)

 227                                                      supports_check_mode=True,

 230     def exec_module(self, **kwargs):
 231         """Main module execution method"""
 232 
 233         for key in list(self.module_arg_spec.keys()) + ['tags']:
 234             if kwargs[key]:
 235                 setattr(self, key, kwargs[key])
 236 
 237         old_response = None
 238         response = None
 239         to_be_updated = False
 240 
 241         # set location
 242         resource_group = self.get_resource_group(self.resource_group)
 243         if not self.location:
 244             self.location = resource_group.location
 245 
 246         # get app service plan
 247         old_response = self.get_plan()
 248 
 249         # if not existing
 250         if not old_response:
 251             self.log("App Service plan doesn't exist")
 252 
 253             if self.state == "present":
 254                 to_be_updated = True
 255 
 256                 if not self.sku:
 257                     self.fail('Please specify sku in plan when creation')
 258 
 259         else:
 260             # existing app service plan, do update
 261             self.log("App Service Plan already exists")
 262 
 263             if self.state == 'present':
 264                 self.log('Result: {0}'.format(old_response))
 265 
 266                 update_tags, newtags = self.update_tags(old_response.get('tags', dict()))
 267 
 268                 if update_tags:
 269                     to_be_updated = True
 270                     self.tags = newtags
 271 
 272                 # check if sku changed
 273                 if self.sku and _normalize_sku(self.sku) != old_response['sku']['size']:
 274                     to_be_updated = True
 275 
 276                 # check if number_of_workers changed
 277                 if self.number_of_workers and int(self.number_of_workers) != old_response['sku']['capacity']:
 278                     to_be_updated = True
 279 
 280                 if self.is_linux and self.is_linux != old_response['reserved']:
 281                     self.fail("Operation not allowed: cannot update reserved of app service plan.")
 282 
 283         if old_response:
 284             self.results['id'] = old_response['id']
 285 
 286         if to_be_updated:
 287             self.log('Need to Create/Update app service plan')
 288             self.results['changed'] = True
 289 
 290             if self.check_mode:
 291                 return self.results
 292 
 293             response = self.create_or_update_plan()
 294             self.results['id'] = response['id']
 295 
 296         if self.state == 'absent' and old_response:
 297             self.log("Delete app service plan")
 298             self.results['changed'] = True
 299 
 300             if self.check_mode:
 301                 return self.results
 302 
 303             self.delete_plan()
 304 
 305             self.log('App service plan instance deleted')
 306 
 307         return self.results

 309     def get_plan(self):
 310         '''
 311         Gets app service plan
 312         :return: deserialized app service plan dictionary
 313         '''
 314         self.log("Get App Service Plan {0}".format(self.name))
 315 
 316         try:
 317             response = self.web_client.app_service_plans.get(self.resource_group, self.name)
 318             if response:
 319                 self.log("Response : {0}".format(response))
 320                 self.log("App Service Plan : {0} found".format(response.name))
 321 
 322                 return appserviceplan_to_dict(response)
 323         except CloudError as ex:
 324             self.log("Didn't find app service plan {0} in resource group {1}".format(self.name, self.resource_group))
 325 
 326         return False
 327 
 328     def create_or_update_plan(self):
 329         '''
 330         Creates app service plan
 331         :return: deserialized app service plan dictionary
 332         '''
 333         self.log("Create App Service Plan {0}".format(self.name))
 334 
 335         try:
 336             # normalize sku
 337             sku = _normalize_sku(self.sku)
 338 
 339             sku_def = SkuDescription(tier=get_sku_name(
 340                 sku), name=sku, capacity=self.number_of_workers)
 341             plan_def = AppServicePlan(
 342                 location=self.location, app_service_plan_name=self.name, sku=sku_def, reserved=self.is_linux, tags=self.tags if self.tags else None)
 343 
 344             response = self.web_client.app_service_plans.create_or_update(self.resource_group, self.name, plan_def)
 345 
 346             if isinstance(response, LROPoller) or isinstance(response, AzureOperationPoller):
 347                 response = self.get_poller_result(response)
 348 
 349             self.log("Response : {0}".format(response))
 350 
 351             return appserviceplan_to_dict(response)
 352         except CloudError as ex:
 353             self.fail("Failed to create app service plan {0} in resource group {1}: {2}".format(self.name, self.resource_group, str(ex)))
 354 
 355     def delete_plan(self):
 356         '''
 357         Deletes specified App service plan in the specified subscription and resource group.
 358 
 359         :return: True
 360         '''
 361         self.log("Deleting the App service plan {0}".format(self.name))
 362         try:
 363             response = self.web_client.app_service_plans.delete(resource_group_name=self.resource_group,
 364                                                                 name=self.name)
 365         except CloudError as e:
 366             self.log('Error attempting to delete App service plan.')
 367             self.fail(
 368                 "Error deleting the App service plan : {0}".format(str(e)))
 369 
 370         return True
