 445 def role_exists(module, cursor, rolname):
 446     """Check user exists or not"""
 447     query = "SELECT 1 FROM pg_roles WHERE rolname = '%s'" % rolname
 448     try:
 449         cursor.execute(query)
 450         return cursor.rowcount > 0
 451 
 452     except Exception as e:
 453         module.fail_json(msg="Cannot execute SQL '%s': %s" % (query, to_native(e)))
 454 
 455     return False

 458 # We don't have functools.partial in Python < 2.5
 459 def partial(f, *args, **kwargs):
 460     """Partial function application"""
 461 
 462     def g(*g_args, **g_kwargs):
 463         new_kwargs = kwargs.copy()
 464         new_kwargs.update(g_kwargs)
 465         return f(*(args + g_args), **g_kwargs)
 466 
 467     g.f = f
 468     g.args = args
 469     g.kwargs = kwargs
 470     return g

 473 class Connection(object):
 474     """Wrapper around a psycopg2 connection with some convenience methods"""
 475 
 476     def __init__(self, params, module):
 477         self.database = params.database
 478         self.module = module
 479         # To use defaults values, keyword arguments must be absent, so
 480         # check which values are empty and don't include in the **kw
 481         # dictionary
 482         params_map = {
 483             "host": "host",
 484             "login": "user",
 485             "password": "password",
 486             "port": "port",
 487             "database": "database",
 488             "ssl_mode": "sslmode",
 489             "ca_cert": "sslrootcert"
 490         }
 491 
 492         kw = dict((params_map[k], getattr(params, k)) for k in params_map
 493                   if getattr(params, k) != '' and getattr(params, k) is not None)
 494 
 495         # If a unix_socket is specified, incorporate it here.
 496         is_localhost = "host" not in kw or kw["host"] == "" or kw["host"] == "localhost"
 497         if is_localhost and params.unix_socket != "":
 498             kw["host"] = params.unix_socket
 499 
 500         sslrootcert = params.ca_cert
 501         if psycopg2.__version__ < '2.4.3' and sslrootcert is not None:
 502             raise ValueError('psycopg2 must be at least 2.4.3 in order to user the ca_cert parameter')
 503 
 504         self.connection = psycopg2.connect(**kw)
 505         self.cursor = self.connection.cursor()
 506 
 507     def commit(self):
 508         self.connection.commit()
 509 
 510     def rollback(self):
 511         self.connection.rollback()

 518     # Methods for querying database objects
 519 
 520     # PostgreSQL < 9.0 doesn't support "ALL TABLES IN SCHEMA schema"-like
 521     # phrases in GRANT or REVOKE statements, therefore alternative methods are
 522     # provided here.
 523 
 524     def schema_exists(self, schema):
 525         query = """SELECT count(*)
 526                    FROM pg_catalog.pg_namespace WHERE nspname = %s"""
 527         self.cursor.execute(query, (schema,))
 528         return self.cursor.fetchone()[0] > 0
 529 
 530     def get_all_tables_in_schema(self, schema):
 531         if not self.schema_exists(schema):
 532             raise Error('Schema "%s" does not exist.' % schema)
 533         query = """SELECT relname
 534                    FROM pg_catalog.pg_class c
 535                    JOIN pg_catalog.pg_namespace n ON n.oid = c.relnamespace
 536                    WHERE nspname = %s AND relkind in ('r', 'v', 'm', 'p')"""
 537         self.cursor.execute(query, (schema,))
 538         return [t[0] for t in self.cursor.fetchall()]
 539 
 540     def get_all_sequences_in_schema(self, schema):
 541         if not self.schema_exists(schema):
 542             raise Error('Schema "%s" does not exist.' % schema)
 543         query = """SELECT relname
 544                    FROM pg_catalog.pg_class c
 545                    JOIN pg_catalog.pg_namespace n ON n.oid = c.relnamespace
 546                    WHERE nspname = %s AND relkind = 'S'"""
 547         self.cursor.execute(query, (schema,))
 548         return [t[0] for t in self.cursor.fetchall()]
 549 
 550     def get_all_functions_in_schema(self, schema):
 551         if not self.schema_exists(schema):
 552             raise Error('Schema "%s" does not exist.' % schema)
 553         query = """SELECT p.proname, oidvectortypes(p.proargtypes)
 554                     FROM pg_catalog.pg_proc p
 555                     JOIN pg_namespace n ON n.oid = p.pronamespace
 556                     WHERE nspname = %s"""
 557         self.cursor.execute(query, (schema,))
 558         return ["%s(%s)" % (t[0], t[1]) for t in self.cursor.fetchall()]

 560     # Methods for getting access control lists and group membership info
 561 
 562     # To determine whether anything has changed after granting/revoking
 563     # privileges, we compare the access control lists of the specified database
 564     # objects before and afterwards. Python's list/string comparison should
 565     # suffice for change detection, we should not actually have to parse ACLs.
 566     # The same should apply to group membership information.
 567 
 568     def get_table_acls(self, schema, tables):
 569         query = """SELECT relacl
 570                    FROM pg_catalog.pg_class c
 571                    JOIN pg_catalog.pg_namespace n ON n.oid = c.relnamespace
 572                    WHERE nspname = %s AND relkind in ('r','p','v','m') AND relname = ANY (%s)
 573                    ORDER BY relname"""
 574         self.cursor.execute(query, (schema, tables))
 575         return [t[0] for t in self.cursor.fetchall()]
 576 
 577     def get_sequence_acls(self, schema, sequences):
 578         query = """SELECT relacl
 579                    FROM pg_catalog.pg_class c
 580                    JOIN pg_catalog.pg_namespace n ON n.oid = c.relnamespace
 581                    WHERE nspname = %s AND relkind = 'S' AND relname = ANY (%s)
 582                    ORDER BY relname"""
 583         self.cursor.execute(query, (schema, sequences))
 584         return [t[0] for t in self.cursor.fetchall()]
 585 
 586     def get_function_acls(self, schema, function_signatures):
 587         funcnames = [f.split('(', 1)[0] for f in function_signatures]
 588         query = """SELECT proacl
 589                    FROM pg_catalog.pg_proc p
 590                    JOIN pg_catalog.pg_namespace n ON n.oid = p.pronamespace
 591                    WHERE nspname = %s AND proname = ANY (%s)
 592                    ORDER BY proname, proargtypes"""
 593         self.cursor.execute(query, (schema, funcnames))
 594         return [t[0] for t in self.cursor.fetchall()]
 595 
 596     def get_schema_acls(self, schemas):
 597         query = """SELECT nspacl FROM pg_catalog.pg_namespace
 598                    WHERE nspname = ANY (%s) ORDER BY nspname"""
 599         self.cursor.execute(query, (schemas,))
 600         return [t[0] for t in self.cursor.fetchall()]
 601 
 602     def get_language_acls(self, languages):
 603         query = """SELECT lanacl FROM pg_catalog.pg_language
 604                    WHERE lanname = ANY (%s) ORDER BY lanname"""
 605         self.cursor.execute(query, (languages,))
 606         return [t[0] for t in self.cursor.fetchall()]
 607 
 608     def get_tablespace_acls(self, tablespaces):
 609         query = """SELECT spcacl FROM pg_catalog.pg_tablespace
 610                    WHERE spcname = ANY (%s) ORDER BY spcname"""
 611         self.cursor.execute(query, (tablespaces,))
 612         return [t[0] for t in self.cursor.fetchall()]
 613 
 614     def get_database_acls(self, databases):
 615         query = """SELECT datacl FROM pg_catalog.pg_database
 616                    WHERE datname = ANY (%s) ORDER BY datname"""
 617         self.cursor.execute(query, (databases,))
 618         return [t[0] for t in self.cursor.fetchall()]

 621         query = """SELECT roleid, grantor, member, admin_option
 622                    FROM pg_catalog.pg_auth_members am

 625                    ORDER BY roleid, grantor, member"""
 626         self.cursor.execute(query, (groups,))

 658     def manipulate_privs(self, obj_type, privs, objs, roles, target_roles,
 659                          state, grant_option, schema_qualifier=None, fail_on_role=True):
 660         """Manipulate database object privileges.
 661 
 662         :param obj_type: Type of database object to grant/revoke
 663                          privileges for.
 664         :param privs: Either a list of privileges to grant/revoke
 665                       or None if type is "group".
 666         :param objs: List of database objects to grant/revoke
 667                      privileges for.
 668         :param roles: Either a list of role names or "PUBLIC"
 669                       for the implicitly defined "PUBLIC" group
 670         :param target_roles: List of role names to grant/revoke
 671                              default privileges as.
 672         :param state: "present" to grant privileges, "absent" to revoke.
 673         :param grant_option: Only for state "present": If True, set
 674                              grant/admin option. If False, revoke it.
 675                              If None, don't change grant option.
 676         :param schema_qualifier: Some object types ("TABLE", "SEQUENCE",
 677                                  "FUNCTION") must be qualified by schema.
 678                                  Ignored for other Types.
 679         """
 680         # get_status: function to get current status
 681         if obj_type == 'table':
 682             get_status = partial(self.get_table_acls, schema_qualifier)
 683         elif obj_type == 'sequence':
 684             get_status = partial(self.get_sequence_acls, schema_qualifier)
 685         elif obj_type == 'function':
 686             get_status = partial(self.get_function_acls, schema_qualifier)
 687         elif obj_type == 'schema':
 688             get_status = self.get_schema_acls
 689         elif obj_type == 'language':
 690             get_status = self.get_language_acls
 691         elif obj_type == 'tablespace':
 692             get_status = self.get_tablespace_acls
 693         elif obj_type == 'database':
 694             get_status = self.get_database_acls
 695         elif obj_type == 'group':
 696             get_status = self.get_group_memberships
 697         elif obj_type == 'default_privs':
 698             get_status = partial(self.get_default_privs, schema_qualifier)
 699         elif obj_type == 'foreign_data_wrapper':
 700             get_status = self.get_foreign_data_wrapper_acls
 701         elif obj_type == 'foreign_server':
 702             get_status = self.get_foreign_server_acls
 703         elif obj_type == 'type':
 704             get_status = partial(self.get_type_acls, schema_qualifier)
 705         else:
 706             raise Error('Unsupported database object type "%s".' % obj_type)
 707 
 708         # Return False (nothing has changed) if there are no objs to work on.
 709         if not objs:
 710             return False
 711 
 712         # obj_ids: quoted db object identifiers (sometimes schema-qualified)
 713         if obj_type == 'function':
 714             obj_ids = []
 715             for obj in objs:
 716                 try:
 717                     f, args = obj.split('(', 1)
 718                 except Exception:
 719                     raise Error('Illegal function signature: "%s".' % obj)
 720                 obj_ids.append('"%s"."%s"(%s' % (schema_qualifier, f, args))
 721         elif obj_type in ['table', 'sequence', 'type']:
 722             obj_ids = ['"%s"."%s"' % (schema_qualifier, o) for o in objs]
 723         else:
 724             obj_ids = ['"%s"' % o for o in objs]
 725 
 726         # set_what: SQL-fragment specifying what to set for the target roles:
 727         # Either group membership or privileges on objects of a certain type
 728         if obj_type == 'group':
 729             set_what = ','.join('"%s"' % i for i in obj_ids)
 730         elif obj_type == 'default_privs':
 731             # We don't want privs to be quoted here
 732             set_what = ','.join(privs)
 733         else:
 734             # function types are already quoted above
 735             if obj_type != 'function':
 736                 obj_ids = [pg_quote_identifier(i, 'table') for i in obj_ids]
 737             # Note: obj_type has been checked against a set of string literals
 738             # and privs was escaped when it was parsed
 739             # Note: Underscores are replaced with spaces to support multi-word obj_type
 740             set_what = '%s ON %s %s' % (','.join(privs), obj_type.replace('_', ' '),
 741                                         ','.join(obj_ids))
 742 
 743         # for_whom: SQL-fragment specifying for whom to set the above
 744         if roles == 'PUBLIC':
 745             for_whom = 'PUBLIC'
 746         else:
 747             for_whom = []
 748             for r in roles:
 749                 if not role_exists(self.module, self.cursor, r):
 750                     if fail_on_role:
 751                         self.module.fail_json(msg="Role '%s' does not exist" % r.strip())
 752 
 753                     else:
 754                         self.module.warn("Role '%s' does not exist, pass it" % r.strip())
 755                 else:
 756                     for_whom.append('"%s"' % r)
 757 
 758             if not for_whom:
 759                 return False
 760 
 761             for_whom = ','.join(for_whom)
 762 
 763         # as_who:
 764         as_who = None
 765         if target_roles:
 766             as_who = ','.join('"%s"' % r for r in target_roles)
 767 
 768         status_before = get_status(objs)
 769 
 770         query = QueryBuilder(state) \
 771             .for_objtype(obj_type) \
 772             .with_grant_option(grant_option) \
 773             .for_whom(for_whom) \
 774             .as_who(as_who) \
 775             .for_schema(schema_qualifier) \
 776             .set_what(set_what) \
 777             .for_objs(objs) \
 778             .build()
 779 
 780         executed_queries.append(query)
 781         self.cursor.execute(query)
 782         status_after = get_status(objs)
 783 
 784         def nonesorted(e):
 785             # For python 3+ that can fail trying
 786             # to compare NoneType elements by sort method.
 787             if e is None:
 788                 return ''
 789             return e
 790 
 791         status_before.sort(key=nonesorted)
 792         status_after.sort(key=nonesorted)
 793         return status_before != status_after

 796 class QueryBuilder(object):
 797     def __init__(self, state):
 798         self._grant_option = None
 799         self._for_whom = None
 800         self._as_who = None
 801         self._set_what = None
 802         self._obj_type = None
 803         self._state = state
 804         self._schema = None
 805         self._objs = None
 806         self.query = []
 807 
 808     def for_objs(self, objs):
 809         self._objs = objs
 810         return self
 811 
 812     def for_schema(self, schema):
 813         self._schema = schema
 814         return self
 815 
 816     def with_grant_option(self, option):
 817         self._grant_option = option
 818         return self
 819 
 820     def for_whom(self, who):
 821         self._for_whom = who
 822         return self
 823 
 824     def as_who(self, target_roles):
 825         self._as_who = target_roles
 826         return self
 827 
 828     def set_what(self, what):
 829         self._set_what = what
 830         return self
 831 
 832     def for_objtype(self, objtype):
 833         self._obj_type = objtype
 834         return self
 835 
 836     def build(self):
 837         if self._state == 'present':
 838             self.build_present()
 839         elif self._state == 'absent':
 840             self.build_absent()
 841         else:
 842             self.build_absent()
 843         return '\n'.join(self.query)
 844 
 845     def add_default_revoke(self):
 846         for obj in self._objs:
 847             if self._as_who:
 848                 self.query.append(
 849                     'ALTER DEFAULT PRIVILEGES FOR ROLE {0} IN SCHEMA {1} REVOKE ALL ON {2} FROM {3};'.format(self._as_who,
 850                                                                                                              self._schema, obj,
 851                                                                                                              self._for_whom))
 852             else:
 853                 self.query.append(
 854                     'ALTER DEFAULT PRIVILEGES IN SCHEMA {0} REVOKE ALL ON {1} FROM {2};'.format(self._schema, obj,
 855                                                                                                 self._for_whom))
 856 
 857     def add_grant_option(self):
 858         if self._grant_option:
 859             if self._obj_type == 'group':
 860                 self.query[-1] += ' WITH ADMIN OPTION;'
 861             else:
 862                 self.query[-1] += ' WITH GRANT OPTION;'
 863         else:
 864             self.query[-1] += ';'
 865             if self._obj_type == 'group':
 866                 self.query.append('REVOKE ADMIN OPTION FOR {0} FROM {1};'.format(self._set_what, self._for_whom))
 867             elif not self._obj_type == 'default_privs':
 868                 self.query.append('REVOKE GRANT OPTION FOR {0} FROM {1};'.format(self._set_what, self._for_whom))
 869 
 870     def add_default_priv(self):
 871         for obj in self._objs:
 872             if self._as_who:
 873                 self.query.append(
 874                     'ALTER DEFAULT PRIVILEGES FOR ROLE {0} IN SCHEMA {1} GRANT {2} ON {3} TO {4}'.format(self._as_who,
 875                                                                                                          self._schema,
 876                                                                                                          self._set_what,
 877                                                                                                          obj,
 878                                                                                                          self._for_whom))
 879             else:
 880                 self.query.append(
 881                     'ALTER DEFAULT PRIVILEGES IN SCHEMA {0} GRANT {1} ON {2} TO {3}'.format(self._schema,
 882                                                                                             self._set_what,
 883                                                                                             obj,
 884                                                                                             self._for_whom))
 885             self.add_grant_option()
 886         if self._as_who:
 887             self.query.append(
 888                 'ALTER DEFAULT PRIVILEGES FOR ROLE {0} IN SCHEMA {1} GRANT USAGE ON TYPES TO {2}'.format(self._as_who,
 889                                                                                                          self._schema,
 890                                                                                                          self._for_whom))
 891         else:
 892             self.query.append(
 893                 'ALTER DEFAULT PRIVILEGES IN SCHEMA {0} GRANT USAGE ON TYPES TO {1}'.format(self._schema, self._for_whom))
 894         self.add_grant_option()
 895 
 896     def build_present(self):
 897         if self._obj_type == 'default_privs':
 898             self.add_default_revoke()
 899             self.add_default_priv()
 900         else:
 901             self.query.append('GRANT {0} TO {1}'.format(self._set_what, self._for_whom))
 902             self.add_grant_option()
 903 
 904     def build_absent(self):
 905         if self._obj_type == 'default_privs':
 906             self.query = []
 907             for obj in ['TABLES', 'SEQUENCES', 'TYPES']:
 908                 if self._as_who:
 909                     self.query.append(
 910                         'ALTER DEFAULT PRIVILEGES FOR ROLE {0} IN SCHEMA {1} REVOKE ALL ON {2} FROM {3};'.format(self._as_who,
 911                                                                                                                  self._schema, obj,
 912                                                                                                                  self._for_whom))
 913                 else:
 914                     self.query.append(
 915                         'ALTER DEFAULT PRIVILEGES IN SCHEMA {0} REVOKE ALL ON {1} FROM {2};'.format(self._schema, obj,
 916                                                                                                     self._for_whom))
 917         else:
 918             self.query.append('REVOKE {0} FROM {1};'.format(self._set_what, self._for_whom))

 921 def main():
 922     argument_spec = postgres_common_argument_spec()
 923     argument_spec.update(
 924         database=dict(required=True, aliases=['db', 'login_db']),
 925         state=dict(default='present', choices=['present', 'absent']),
 926         privs=dict(required=False, aliases=['priv']),
 927         type=dict(default='table',
 928                   choices=['table',
 929                            'sequence',
 930                            'function',
 931                            'database',
 932                            'schema',
 933                            'language',
 934                            'tablespace',
 935                            'group',
 936                            'default_privs',
 937                            'foreign_data_wrapper',
 938                            'foreign_server',
 939                            'type', ]),
 940         objs=dict(required=False, aliases=['obj']),
 941         schema=dict(required=False),
 942         roles=dict(required=True, aliases=['role']),
 943         session_role=dict(required=False),
 944         target_roles=dict(required=False),
 945         grant_option=dict(required=False, type='bool',
 946                           aliases=['admin_option']),
 947         host=dict(default='', aliases=['login_host']),
 948         unix_socket=dict(default='', aliases=['login_unix_socket']),
 949         login=dict(default='postgres', aliases=['login_user']),
 950         password=dict(default='', aliases=['login_password'], no_log=True),
 951         fail_on_role=dict(type='bool', default=True),
 952     )
 953 
 954     module = AnsibleModule(
 955         argument_spec=argument_spec,
 956         supports_check_mode=True,
 957     )
 958 
 959     fail_on_role = module.params['fail_on_role']
 960 
 961     # Create type object as namespace for module params
 962     p = type('Params', (), module.params)
 963     # param "schema": default, allowed depends on param "type"
 964     if p.type in ['table', 'sequence', 'function', 'type', 'default_privs']:
 965         p.schema = p.schema or 'public'
 966     elif p.schema:
 967         module.fail_json(msg='Argument "schema" is not allowed '
 968                              'for type "%s".' % p.type)
 969 
 970     # param "objs": default, required depends on param "type"
 971     if p.type == 'database':
 972         p.objs = p.objs or p.database
 973     elif not p.objs:
 974         module.fail_json(msg='Argument "objs" is required '
 975                              'for type "%s".' % p.type)
 976 
 977     # param "privs": allowed, required depends on param "type"
 978     if p.type == 'group':
 979         if p.privs:
 980             module.fail_json(msg='Argument "privs" is not allowed '
 981                                  'for type "group".')
 982     elif not p.privs:
 983         module.fail_json(msg='Argument "privs" is required '
 984                              'for type "%s".' % p.type)
 985 
 986     # Connect to Database
 987     if not psycopg2:
 988         module.fail_json(msg=missing_required_lib('psycopg2'), exception=PSYCOPG2_IMP_ERR)
 989     try:
 990         conn = Connection(p, module)
 991     except psycopg2.Error as e:
 992         module.fail_json(msg='Could not connect to database: %s' % to_native(e), exception=traceback.format_exc())
 993     except TypeError as e:
 994         if 'sslrootcert' in e.args[0]:
 995             module.fail_json(msg='Postgresql server must be at least version 8.4 to support sslrootcert')
 996         module.fail_json(msg="unable to connect to database: %s" % to_native(e), exception=traceback.format_exc())
 997     except ValueError as e:
 998         # We raise this when the psycopg library is too old
 999         module.fail_json(msg=to_native(e))
1000 
1001     if p.session_role:
1002         try:
1003             conn.cursor.execute('SET ROLE "%s"' % p.session_role)
1004         except Exception as e:
1005             module.fail_json(msg="Could not switch to role %s: %s" % (p.session_role, to_native(e)), exception=traceback.format_exc())
1006 
1007     try:
1008         # privs
1009         if p.privs:
1010             privs = frozenset(pr.upper() for pr in p.privs.split(','))
1011             if not privs.issubset(VALID_PRIVS):
1012                 module.fail_json(msg='Invalid privileges specified: %s' % privs.difference(VALID_PRIVS))
1013         else:
1014             privs = None
1015         # objs:
1016         if p.type == 'table' and p.objs == 'ALL_IN_SCHEMA':
1017             objs = conn.get_all_tables_in_schema(p.schema)
1018         elif p.type == 'sequence' and p.objs == 'ALL_IN_SCHEMA':
1019             objs = conn.get_all_sequences_in_schema(p.schema)
1020         elif p.type == 'function' and p.objs == 'ALL_IN_SCHEMA':
1021             objs = conn.get_all_functions_in_schema(p.schema)
1022         elif p.type == 'default_privs':
1023             if p.objs == 'ALL_DEFAULT':
1024                 objs = frozenset(VALID_DEFAULT_OBJS.keys())
1025             else:
1026                 objs = frozenset(obj.upper() for obj in p.objs.split(','))
1027                 if not objs.issubset(VALID_DEFAULT_OBJS):
1028                     module.fail_json(
1029                         msg='Invalid Object set specified: %s' % objs.difference(VALID_DEFAULT_OBJS.keys()))
1030             # Again, do we have valid privs specified for object type:
1031             valid_objects_for_priv = frozenset(obj for obj in objs if privs.issubset(VALID_DEFAULT_OBJS[obj]))
1032             if not valid_objects_for_priv == objs:
1033                 module.fail_json(
1034                     msg='Invalid priv specified. Valid object for priv: {0}. Objects: {1}'.format(
1035                         valid_objects_for_priv, objs))
1036         else:
1037             objs = p.objs.split(',')
1038 
1039             # function signatures are encoded using ':' to separate args
1040             if p.type == 'function':
1041                 objs = [obj.replace(':', ',') for obj in objs]
1042 
1043         # roles
1044         if p.roles == 'PUBLIC':
1045             roles = 'PUBLIC'
1046         else:
1047             roles = p.roles.split(',')
1048 
1049             if len(roles) == 1 and not role_exists(module, conn.cursor, roles[0]):
1050                 module.exit_json(changed=False)
1051 
1052                 if fail_on_role:
1053                     module.fail_json(msg="Role '%s' does not exist" % roles[0].strip())
1054 
1055                 else:
1056                     module.warn("Role '%s' does not exist, nothing to do" % roles[0].strip())
1057 
1058         # check if target_roles is set with type: default_privs
1059         if p.target_roles and not p.type == 'default_privs':
1060             module.warn('"target_roles" will be ignored '
1061                         'Argument "type: default_privs" is required for usage of "target_roles".')
1062 
1063         # target roles
1064         if p.target_roles:
1065             target_roles = p.target_roles.split(',')
1066         else:
1067             target_roles = None
1068 
1069         changed = conn.manipulate_privs(
1070             obj_type=p.type,
1071             privs=privs,
1072             objs=objs,
1073             roles=roles,
1074             target_roles=target_roles,
1075             state=p.state,
1076             grant_option=p.grant_option,
1077             schema_qualifier=p.schema,
1078             fail_on_role=fail_on_role,
1079         )
1080 
1081     except Error as e:
1082         conn.rollback()
1083         module.fail_json(msg=e.message, exception=traceback.format_exc())
1084 
1085     except psycopg2.Error as e:
1086         conn.rollback()
1087         module.fail_json(msg=to_native(e.message))
1088 
1089     if module.check_mode:
1090         conn.rollback()
1091     else:
1092         conn.commit()
1093     module.exit_json(changed=changed, queries=executed_queries)
