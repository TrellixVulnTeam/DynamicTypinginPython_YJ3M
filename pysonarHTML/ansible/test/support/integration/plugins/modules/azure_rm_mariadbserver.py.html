 220     def exec_module(self, **kwargs):
 221         """Main module execution method"""
 222 
 223         for key in list(self.module_arg_spec.keys()) + ['tags']:
 224             if hasattr(self, key):
 225                 setattr(self, key, kwargs[key])
 226             elif kwargs[key] is not None:
 227                 if key == "sku":
 228                     ev = kwargs[key]
 229                     if 'tier' in ev:
 230                         if ev['tier'] == 'basic':
 231                             ev['tier'] = 'Basic'
 232                         elif ev['tier'] == 'standard':
 233                             ev['tier'] = 'Standard'
 234                     self.parameters["sku"] = ev
 235                 elif key == "location":
 236                     self.parameters["location"] = kwargs[key]
 237                 elif key == "storage_mb":
 238                     self.parameters.setdefault("properties", {}).setdefault("storage_profile", {})["storage_mb"] = kwargs[key]
 239                 elif key == "version":
 240                     self.parameters.setdefault("properties", {})["version"] = kwargs[key]
 241                 elif key == "enforce_ssl":
 242                     self.parameters.setdefault("properties", {})["ssl_enforcement"] = 'Enabled' if kwargs[key] else 'Disabled'
 243                 elif key == "create_mode":
 244                     self.parameters.setdefault("properties", {})["create_mode"] = kwargs[key]
 245                 elif key == "admin_username":
 246                     self.parameters.setdefault("properties", {})["administrator_login"] = kwargs[key]
 247                 elif key == "admin_password":
 248                     self.parameters.setdefault("properties", {})["administrator_login_password"] = kwargs[key]
 249 
 250         old_response = None
 251         response = None
 252 
 253         resource_group = self.get_resource_group(self.resource_group)
 254 
 255         if "location" not in self.parameters:
 256             self.parameters["location"] = resource_group.location
 257 
 258         old_response = self.get_mariadbserver()
 259 
 260         if not old_response:
 261             self.log("MariaDB Server instance doesn't exist")
 262             if self.state == 'absent':
 263                 self.log("Old instance didn't exist")
 264             else:
 265                 self.to_do = Actions.Create
 266         else:
 267             self.log("MariaDB Server instance already exists")
 268             if self.state == 'absent':
 269                 self.to_do = Actions.Delete
 270             elif self.state == 'present':
 271                 self.log("Need to check if MariaDB Server instance has to be deleted or may be updated")
 272                 update_tags, newtags = self.update_tags(old_response.get('tags', {}))
 273                 if update_tags:
 274                     self.tags = newtags
 275                 self.to_do = Actions.Update
 276 
 277         if (self.to_do == Actions.Create) or (self.to_do == Actions.Update):
 278             self.log("Need to Create / Update the MariaDB Server instance")
 279 
 280             if self.check_mode:
 281                 self.results['changed'] = True
 282                 return self.results
 283 
 284             response = self.create_update_mariadbserver()
 285 
 286             if not old_response:
 287                 self.results['changed'] = True
 288             else:
 289                 self.results['changed'] = old_response.__ne__(response)
 290             self.log("Creation / Update done")
 291         elif self.to_do == Actions.Delete:
 292             self.log("MariaDB Server instance deleted")
 293             self.results['changed'] = True
 294 
 295             if self.check_mode:
 296                 return self.results
 297 
 298             self.delete_mariadbserver()
 299             # make sure instance is actually deleted, for some Azure resources, instance is hanging around
 300             # for some time after deletion -- this should be really fixed in Azure
 301             while self.get_mariadbserver():
 302                 time.sleep(20)
 303         else:
 304             self.log("MariaDB Server instance unchanged")
 305             self.results['changed'] = False
 306             response = old_response
 307 
 308         if response:
 309             self.results["id"] = response["id"]
 310             self.results["version"] = response["version"]
 311             self.results["state"] = response["user_visible_state"]
 312             self.results["fully_qualified_domain_name"] = response["fully_qualified_domain_name"]
 313 
 314         return self.results

 316     def create_update_mariadbserver(self):
 317         '''
 318         Creates or updates MariaDB Server with the specified configuration.
 319 
 320         :return: deserialized MariaDB Server instance state dictionary
 321         '''
 322         self.log("Creating / Updating the MariaDB Server instance {0}".format(self.name))
 323 
 324         try:
 325             self.parameters['tags'] = self.tags
 326             if self.to_do == Actions.Create:
 327                 response = self.mariadb_client.servers.create(resource_group_name=self.resource_group,
 328                                                               server_name=self.name,
 329                                                               parameters=self.parameters)
 330             else:
 331                 # structure of parameters for update must be changed
 332                 self.parameters.update(self.parameters.pop("properties", {}))
 333                 response = self.mariadb_client.servers.update(resource_group_name=self.resource_group,
 334                                                               server_name=self.name,
 335                                                               parameters=self.parameters)
 336             if isinstance(response, LROPoller):
 337                 response = self.get_poller_result(response)
 338 
 339         except CloudError as exc:
 340             self.log('Error attempting to create the MariaDB Server instance.')
 341             self.fail("Error creating the MariaDB Server instance: {0}".format(str(exc)))
 342         return response.as_dict()
 343 
 344     def delete_mariadbserver(self):
 345         '''
 346         Deletes specified MariaDB Server instance in the specified subscription and resource group.
 347 
 348         :return: True
 349         '''
 350         self.log("Deleting the MariaDB Server instance {0}".format(self.name))
 351         try:
 352             response = self.mariadb_client.servers.delete(resource_group_name=self.resource_group,
 353                                                           server_name=self.name)
 354         except CloudError as e:
 355             self.log('Error attempting to delete the MariaDB Server instance.')
 356             self.fail("Error deleting the MariaDB Server instance: {0}".format(str(e)))
 357 
 358         return True
 359 
 360     def get_mariadbserver(self):
 361         '''
 362         Gets the properties of the specified MariaDB Server.
 363 
 364         :return: deserialized MariaDB Server instance state dictionary
 365         '''
 366         self.log("Checking if the MariaDB Server instance {0} is present".format(self.name))
 367         found = False
 368         try:
 369             response = self.mariadb_client.servers.get(resource_group_name=self.resource_group,
 370                                                        server_name=self.name)
 371             found = True
 372             self.log("Response : {0}".format(response))
 373             self.log("MariaDB Server instance : {0} found".format(response.name))
 374         except CloudError as e:
 375             self.log('Did not find the MariaDB Server instance.')
 376         if found is True:
 377             return response.as_dict()
 378 
 379         return False
