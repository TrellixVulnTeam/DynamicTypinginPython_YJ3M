 203      encounters an error. Those synchronizing large numbers of files that are willing to trade safety for performance should call rsync directly.

 352 def substitute_controller(path):
 353     global client_addr
 354     if not client_addr:
 355         ssh_env_string = os.environ.get('SSH_CLIENT', None)
 356         try:
 357             client_addr, _ = ssh_env_string.split(None, 1)
 358         except AttributeError:
 359             ssh_env_string = os.environ.get('SSH_CONNECTION', None)
 360             try:
 361                 client_addr, _ = ssh_env_string.split(None, 1)
 362             except AttributeError:
 363                 pass
 364         if not client_addr:
 365             raise ValueError
 366 
 367     if path.startswith('localhost:'):
 368         path = path.replace('localhost', client_addr, 1)
 369     return path
 370

 372 def is_rsh_needed(source, dest):
 373     if source.startswith('rsync://') or dest.startswith('rsync://'):
 374         return False
 375     if ':' in source or ':' in dest:
 376         return True
 377     return False

 380 def main():
 381     module = AnsibleModule(
 382         argument_spec=dict(
 383             src=dict(type='str', required=True),
 384             dest=dict(type='str', required=True),
 385             dest_port=dict(type='int'),
 386             delete=dict(type='bool', default=False),
 387             private_key=dict(type='path'),
 388             rsync_path=dict(type='str'),
 389             _local_rsync_path=dict(type='path', default='rsync'),
 390             _local_rsync_password=dict(type='str', no_log=True),
 391             _substitute_controller=dict(type='bool', default=False),
 392             archive=dict(type='bool', default=True),
 393             checksum=dict(type='bool', default=False),
 394             compress=dict(type='bool', default=True),
 395             existing_only=dict(type='bool', default=False),
 396             dirs=dict(type='bool', default=False),
 397             recursive=dict(type='bool'),
 398             links=dict(type='bool'),
 399             copy_links=dict(type='bool', default=False),
 400             perms=dict(type='bool'),
 401             times=dict(type='bool'),
 402             owner=dict(type='bool'),
 403             group=dict(type='bool'),
 404             set_remote_user=dict(type='bool', default=True),
 405             rsync_timeout=dict(type='int', default=0),
 406             rsync_opts=dict(type='list', default=[]),
 407             ssh_args=dict(type='str'),
 408             partial=dict(type='bool', default=False),
 409             verify_host=dict(type='bool', default=False),
 410             mode=dict(type='str', default='push', choices=['pull', 'push']),
 411             link_dest=dict(type='list')
 412         ),
 413         supports_check_mode=True,
 414     )
 415 
 416     if module.params['_substitute_controller']:
 417         try:
 418             source = substitute_controller(module.params['src'])
 419             dest = substitute_controller(module.params['dest'])
 420         except ValueError:
 421             module.fail_json(msg='Could not determine controller hostname for rsync to send to')
 422     else:
 423         source = module.params['src']
 424         dest = module.params['dest']
 425     dest_port = module.params['dest_port']
 426     delete = module.params['delete']
 427     private_key = module.params['private_key']
 428     rsync_path = module.params['rsync_path']
 429     rsync = module.params.get('_local_rsync_path', 'rsync')
 430     rsync_password = module.params.get('_local_rsync_password')
 431     rsync_timeout = module.params.get('rsync_timeout', 'rsync_timeout')
 432     archive = module.params['archive']
 433     checksum = module.params['checksum']
 434     compress = module.params['compress']
 435     existing_only = module.params['existing_only']
 436     dirs = module.params['dirs']
 437     partial = module.params['partial']
 438     # the default of these params depends on the value of archive
 439     recursive = module.params['recursive']
 440     links = module.params['links']
 441     copy_links = module.params['copy_links']
 442     perms = module.params['perms']
 443     times = module.params['times']
 444     owner = module.params['owner']
 445     group = module.params['group']
 446     rsync_opts = module.params['rsync_opts']
 447     ssh_args = module.params['ssh_args']
 448     verify_host = module.params['verify_host']
 449     link_dest = module.params['link_dest']
 450 
 451     if '/' not in rsync:
 452         rsync = module.get_bin_path(rsync, required=True)
 453 
 454     cmd = [rsync, '--delay-updates', '-F']
 455     _sshpass_pipe = None
 456     if rsync_password:
 457         try:
 458             module.run_command(["sshpass"])
 459         except OSError:
 460             module.fail_json(
 461                 msg="to use rsync connection with passwords, you must install the sshpass program"
 462             )
 463         _sshpass_pipe = os.pipe()
 464         cmd = ['sshpass', '-d' + to_native(_sshpass_pipe[0], errors='surrogate_or_strict')] + cmd
 465     if compress:
 466         cmd.append('--compress')
 467     if rsync_timeout:
 468         cmd.append('--timeout=%s' % rsync_timeout)
 469     if module.check_mode:
 470         cmd.append('--dry-run')
 471     if delete:
 472         cmd.append('--delete-after')
 473     if existing_only:
 474         cmd.append('--existing')
 475     if checksum:
 476         cmd.append('--checksum')
 477     if copy_links:
 478         cmd.append('--copy-links')
 479     if archive:
 480         cmd.append('--archive')
 481         if recursive is False:
 482             cmd.append('--no-recursive')
 483         if links is False:
 484             cmd.append('--no-links')
 485         if perms is False:
 486             cmd.append('--no-perms')
 487         if times is False:
 488             cmd.append('--no-times')
 489         if owner is False:
 490             cmd.append('--no-owner')
 491         if group is False:
 492             cmd.append('--no-group')
 493     else:
 494         if recursive is True:
 495             cmd.append('--recursive')
 496         if links is True:
 497             cmd.append('--links')
 498         if perms is True:
 499             cmd.append('--perms')
 500         if times is True:
 501             cmd.append('--times')
 502         if owner is True:
 503             cmd.append('--owner')
 504         if group is True:
 505             cmd.append('--group')
 506     if dirs:
 507         cmd.append('--dirs')
 508 
 509     if source.startswith('rsync://') and dest.startswith('rsync://'):
 510         module.fail_json(msg='either src or dest must be a localhost', rc=1)
 511 
 512     if is_rsh_needed(source, dest):
 513 
 514         # https://github.com/ansible/ansible/issues/15907
 515         has_rsh = False
 516         for rsync_opt in rsync_opts:
 517             if '--rsh' in rsync_opt:
 518                 has_rsh = True
 519                 break
 520 
 521         # if the user has not supplied an --rsh option go ahead and add ours
 522         if not has_rsh:
 523             ssh_cmd = [module.get_bin_path('ssh', required=True), '-S', 'none']
 524             if private_key is not None:
 525                 ssh_cmd.extend(['-i', private_key])
 526             # If the user specified a port value
 527             # Note:  The action plugin takes care of setting this to a port from
 528             # inventory if the user didn't specify an explicit dest_port
 529             if dest_port is not None:
 530                 ssh_cmd.extend(['-o', 'Port=%s' % dest_port])
 531             if not verify_host:
 532                 ssh_cmd.extend(['-o', 'StrictHostKeyChecking=no', '-o', 'UserKnownHostsFile=/dev/null'])
 533             ssh_cmd_str = ' '.join(shlex_quote(arg) for arg in ssh_cmd)
 534             if ssh_args:
 535                 ssh_cmd_str += ' %s' % ssh_args
 536             cmd.append('--rsh=%s' % ssh_cmd_str)
 537 
 538     if rsync_path:
 539         cmd.append('--rsync-path=%s' % rsync_path)
 540 
 541     if rsync_opts:
 542         if '' in rsync_opts:
 543             module.warn('The empty string is present in rsync_opts which will cause rsync to'
 544                         ' transfer the current working directory. If this is intended, use "."'
 545                         ' instead to get rid of this warning. If this is unintended, check for'
 546                         ' problems in your playbook leading to empty string in rsync_opts.')
 547         cmd.extend(rsync_opts)
 548 
 549     if partial:
 550         cmd.append('--partial')
 551 
 552     if link_dest:
 553         cmd.append('-H')
 554         # verbose required because rsync does not believe that adding a
 555         # hardlink is actually a change
 556         cmd.append('-vv')
 557         for x in link_dest:
 558             link_path = os.path.abspath(os.path.expanduser(x))
 559             destination_path = os.path.abspath(os.path.dirname(dest))
 560             if destination_path.find(link_path) == 0:
 561                 module.fail_json(msg='Hardlinking into a subdirectory of the source would cause recursion. %s and %s' % (destination_path, dest))
 562             cmd.append('--link-dest=%s' % link_path)
 563 
 564     changed_marker = '<<CHANGED>>'
 565     cmd.append('--out-format=' + changed_marker + '%i %n%L')
 566 
 567     # expand the paths
 568     if '@' not in source:
 569         source = os.path.expanduser(source)
 570     if '@' not in dest:
 571         dest = os.path.expanduser(dest)
 572 
 573     cmd.append(source)
 574     cmd.append(dest)
 575     cmdstr = ' '.join(cmd)
 576 
 577     # If we are using password authentication, write the password into the pipe
 578     if rsync_password:
 579         def _write_password_to_pipe(proc):
 580             os.close(_sshpass_pipe[0])
 581             try:
 582                 os.write(_sshpass_pipe[1], to_bytes(rsync_password) + b'\n')
 583             except OSError as exc:
 584                 # Ignore broken pipe errors if the sshpass process has exited.
 585                 if exc.errno != errno.EPIPE or proc.poll() is None:
 586                     raise
 587 
 588         (rc, out, err) = module.run_command(
 589             cmd, pass_fds=_sshpass_pipe,
 590             before_communicate_callback=_write_password_to_pipe)
 591     else:
 592         (rc, out, err) = module.run_command(cmd)
 593 
 594     if rc:
 595         return module.fail_json(msg=err, rc=rc, cmd=cmdstr)
 596 
 597     if link_dest:
 598         # a leading period indicates no change
 599         changed = (changed_marker + '.') not in out
 600     else:
 601         changed = changed_marker in out
 602 
 603     out_clean = out.replace(changed_marker, '')
 604     out_lines = out_clean.split('\n')
 605     while '' in out_lines:
 606         out_lines.remove('')
 607     if module._diff:
 608         diff = {'prepared': out_clean}
 609         return module.exit_json(changed=changed, msg=out_clean,
 610                                 rc=rc, cmd=cmdstr, stdout_lines=out_lines,
 611                                 diff=diff)
 612 
 613     return module.exit_json(changed=changed, msg=out_clean,
 614                             rc=rc, cmd=cmdstr, stdout_lines=out_lines)
 615

 618     main()
