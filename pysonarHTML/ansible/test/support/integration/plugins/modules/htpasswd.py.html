 105     from passlib.apache import HtpasswdFile, htpasswd_context

 114 apache_hashes = ["apr_md5_crypt", "des_crypt", "ldap_sha1", "plaintext"]

 117 def create_missing_directories(dest):
 118     destpath = os.path.dirname(dest)
 119     if not os.path.exists(destpath):
 120         os.makedirs(destpath)

 123 def present(dest, username, password, crypt_scheme, create, check_mode):
 124     """ Ensures user is present
 125 
 126     Returns (msg, changed) """
 127     if crypt_scheme in apache_hashes:
 128         context = htpasswd_context
 129     else:
 130         context = CryptContext(schemes=[crypt_scheme] + apache_hashes)
 131     if not os.path.exists(dest):
 132         if not create:
 133             raise ValueError('Destination %s does not exist' % dest)
 134         if check_mode:
 135             return ("Create %s" % dest, True)
 136         create_missing_directories(dest)
 137         if LooseVersion(passlib.__version__) >= LooseVersion('1.6'):
 138             ht = HtpasswdFile(dest, new=True, default_scheme=crypt_scheme, context=context)
 139         else:
 140             ht = HtpasswdFile(dest, autoload=False, default=crypt_scheme, context=context)
 141         if getattr(ht, 'set_password', None):
 142             ht.set_password(username, password)
 143         else:
 144             ht.update(username, password)
 145         ht.save()
 146         return ("Created %s and added %s" % (dest, username), True)
 147     else:
 148         if LooseVersion(passlib.__version__) >= LooseVersion('1.6'):
 149             ht = HtpasswdFile(dest, new=False, default_scheme=crypt_scheme, context=context)
 150         else:
 151             ht = HtpasswdFile(dest, default=crypt_scheme, context=context)
 152 
 153         found = None
 154         if getattr(ht, 'check_password', None):
 155             found = ht.check_password(username, password)
 156         else:
 157             found = ht.verify(username, password)
 158 
 159         if found:
 160             return ("%s already present" % username, False)
 161         else:
 162             if not check_mode:
 163                 if getattr(ht, 'set_password', None):
 164                     ht.set_password(username, password)
 165                 else:
 166                     ht.update(username, password)
 167                 ht.save()
 168             return ("Add/update %s" % username, True)
 169 
 170 
 171 def absent(dest, username, check_mode):
 172     """ Ensures user is absent
 173 
 174     Returns (msg, changed) """
 175     if LooseVersion(passlib.__version__) >= LooseVersion('1.6'):
 176         ht = HtpasswdFile(dest, new=False)
 177     else:
 178         ht = HtpasswdFile(dest)
 179 
 180     if username not in ht.users():
 181         return ("%s not present" % username, False)
 182     else:
 183         if not check_mode:
 184             ht.delete(username)
 185             ht.save()
 186         return ("Remove %s" % username, True)

 199     return message, changed

 202 def main():
 203     arg_spec = dict(
 204         path=dict(required=True, aliases=["dest", "destfile"]),
 205         name=dict(required=True, aliases=["username"]),
 206         password=dict(required=False, default=None, no_log=True),
 207         crypt_scheme=dict(required=False, default="apr_md5_crypt"),
 208         state=dict(required=False, default="present"),
 209         create=dict(type='bool', default='yes'),
 210 
 211     )
 212     module = AnsibleModule(argument_spec=arg_spec,
 213                            add_file_common_args=True,
 214                            supports_check_mode=True)
 215 
 216     path = module.params['path']
 217     username = module.params['name']
 218     password = module.params['password']
 219     crypt_scheme = module.params['crypt_scheme']
 220     state = module.params['state']
 221     create = module.params['create']
 222     check_mode = module.check_mode
 223 
 224     if not passlib_installed:
 225         module.fail_json(msg=missing_required_lib("passlib"), exception=PASSLIB_IMP_ERR)
 226 
 227     # Check file for blank lines in effort to avoid "need more than 1 value to unpack" error.
 228     try:
 229         f = open(path, "r")
 230     except IOError:
 231         # No preexisting file to remove blank lines from
 232         f = None
 233     else:
 234         try:
 235             lines = f.readlines()
 236         finally:
 237             f.close()
 238 
 239         # If the file gets edited, it returns true, so only edit the file if it has blank lines
 240         strip = False
 241         for line in lines:
 242             if not line.strip():
 243                 strip = True
 244                 break
 245 
 246         if strip:
 247             # If check mode, create a temporary file
 248             if check_mode:
 249                 temp = tempfile.NamedTemporaryFile()
 250                 path = temp.name
 251             f = open(path, "w")
 252             try:
 253                 [f.write(line) for line in lines if line.strip()]
 254             finally:
 255                 f.close()
 256 
 257     try:
 258         if state == 'present':
 259             (msg, changed) = present(path, username, password, crypt_scheme, create, check_mode)
 260         elif state == 'absent':
 261             if not os.path.exists(path):
 262                 module.exit_json(msg="%s not present" % username,
 263                                  warnings="%s does not exist" % path, changed=False)
 264             (msg, changed) = absent(path, username, check_mode)
 265         else:
 266             module.fail_json(msg="Invalid state: %s" % state)
 267 
 268         check_file_attrs(module, changed, msg)
 269         module.exit_json(msg=msg, changed=changed)
 270     except Exception as e:
 271         module.fail_json(msg=to_native(e))
 272 
 273 
 274 if __name__ == '__main__':
 275     main()
