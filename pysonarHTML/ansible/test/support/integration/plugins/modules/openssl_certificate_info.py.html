 450     def get_info(self):
 451         result = dict()
 452         self.cert = crypto_utils.load_certificate(self.path, content=self.content, backend=self.backend)
 453 
 454         result['signature_algorithm'] = self._get_signature_algorithm()
 455         subject = self._get_subject_ordered()
 456         issuer = self._get_issuer_ordered()
 457         result['subject'] = dict()
 458         for k, v in subject:
 459             result['subject'][k] = v
 460         result['subject_ordered'] = subject
 461         result['issuer'] = dict()
 462         for k, v in issuer:
 463             result['issuer'][k] = v
 464         result['issuer_ordered'] = issuer
 465         result['version'] = self._get_version()
 466         result['key_usage'], result['key_usage_critical'] = self._get_key_usage()
 467         result['extended_key_usage'], result['extended_key_usage_critical'] = self._get_extended_key_usage()
 468         result['basic_constraints'], result['basic_constraints_critical'] = self._get_basic_constraints()
 469         result['ocsp_must_staple'], result['ocsp_must_staple_critical'] = self._get_ocsp_must_staple()
 470         result['subject_alt_name'], result['subject_alt_name_critical'] = self._get_subject_alt_name()
 471 
 472         not_before = self._get_not_before()
 473         not_after = self._get_not_after()
 474         result['not_before'] = not_before.strftime(TIMESTAMP_FORMAT)
 475         result['not_after'] = not_after.strftime(TIMESTAMP_FORMAT)
 476         result['expired'] = not_after < datetime.datetime.utcnow()
 477 
 478         result['valid_at'] = dict()
 479         if self.valid_at:
 480             for k, v in self.valid_at.items():
 481                 result['valid_at'][k] = not_before <= v <= not_after
 482 
 483         result['public_key'] = self._get_public_key(binary=False)
 484         pk = self._get_public_key(binary=True)
 485         result['public_key_fingerprints'] = crypto_utils.get_fingerprint_of_bytes(pk) if pk is not None else dict()
 486 
 487         if self.backend != 'pyopenssl':
 488             ski = self._get_subject_key_identifier()
 489             if ski is not None:
 490                 ski = to_native(binascii.hexlify(ski))
 491                 ski = ':'.join([ski[i:i + 2] for i in range(0, len(ski), 2)])
 492             result['subject_key_identifier'] = ski
 493 
 494             aki, aci, acsn = self._get_authority_key_identifier()
 495             if aki is not None:
 496                 aki = to_native(binascii.hexlify(aki))
 497                 aki = ':'.join([aki[i:i + 2] for i in range(0, len(aki), 2)])
 498             result['authority_key_identifier'] = aki
 499             result['authority_cert_issuer'] = aci
 500             result['authority_cert_serial_number'] = acsn
 501 
 502         result['serial_number'] = self._get_serial_number()
 503         result['extensions_by_oid'] = self._get_all_extensions()
 504         result['ocsp_uri'] = self._get_ocsp_uri()
 505 
 506         return result

 509 class CertificateInfoCryptography(CertificateInfo):
 510     """Validate the supplied cert, using the cryptography backend"""
 511     def __init__(self, module):
 512         super(CertificateInfoCryptography, self).__init__(module, 'cryptography')
 513 
 514     def _get_signature_algorithm(self):
 515         return crypto_utils.cryptography_oid_to_name(self.cert.signature_algorithm_oid)
 516 
 517     def _get_subject_ordered(self):
 518         result = []
 519         for attribute in self.cert.subject:
 520             result.append([crypto_utils.cryptography_oid_to_name(attribute.oid), attribute.value])
 521         return result
 522 
 523     def _get_issuer_ordered(self):
 524         result = []
 525         for attribute in self.cert.issuer:
 526             result.append([crypto_utils.cryptography_oid_to_name(attribute.oid), attribute.value])
 527         return result
 528 
 529     def _get_version(self):
 530         if self.cert.version == x509.Version.v1:
 531             return 1
 532         if self.cert.version == x509.Version.v3:
 533             return 3
 534         return "unknown"
 535 
 536     def _get_key_usage(self):
 537         try:
 538             current_key_ext = self.cert.extensions.get_extension_for_class(x509.KeyUsage)
 539             current_key_usage = current_key_ext.value
 540             key_usage = dict(
 541                 digital_signature=current_key_usage.digital_signature,
 542                 content_commitment=current_key_usage.content_commitment,
 543                 key_encipherment=current_key_usage.key_encipherment,
 544                 data_encipherment=current_key_usage.data_encipherment,
 545                 key_agreement=current_key_usage.key_agreement,
 546                 key_cert_sign=current_key_usage.key_cert_sign,
 547                 crl_sign=current_key_usage.crl_sign,
 548                 encipher_only=False,
 549                 decipher_only=False,
 550             )
 551             if key_usage['key_agreement']:
 552                 key_usage.update(dict(
 553                     encipher_only=current_key_usage.encipher_only,
 554                     decipher_only=current_key_usage.decipher_only
 555                 ))
 556 
 557             key_usage_names = dict(
 558                 digital_signature='Digital Signature',
 559                 content_commitment='Non Repudiation',
 560                 key_encipherment='Key Encipherment',
 561                 data_encipherment='Data Encipherment',
 562                 key_agreement='Key Agreement',
 563                 key_cert_sign='Certificate Sign',
 564                 crl_sign='CRL Sign',

 696                 ]

 709     def _get_ocsp_must_staple(self):
 710         extensions = [self.cert.get_extension(i) for i in range(0, self.cert.get_extension_count())]
 711         oms_ext = [
 712             ext for ext in extensions
 713             if to_bytes(ext.get_short_name()) == OPENSSL_MUST_STAPLE_NAME and to_bytes(ext) == OPENSSL_MUST_STAPLE_VALUE
 714         ]
 715         if OpenSSL.SSL.OPENSSL_VERSION_NUMBER < 0x10100000:
 716             # Older versions of libssl don't know about OCSP Must Staple
 717             oms_ext.extend([ext for ext in extensions if ext.get_short_name() == b'UNDEF' and ext.get_data() == b'\x30\x03\x02\x01\x05'])
 718         if oms_ext:
 719             return True, bool(oms_ext[0].get_critical())
 720         else:
 721             return None, False

 723     def _normalize_san(self, san):
 724         if san.startswith('IP Address:'):

 726         if san.startswith('IP:'):
 727             ip = compat_ipaddress.ip_address(san[3:])
 728             san = 'IP:{0}'.format(ip.compressed)

 748     def _get_public_key(self, binary):
 749         try:
 750             return crypto.dump_publickey(
 751                 crypto.FILETYPE_ASN1 if binary else crypto.FILETYPE_PEM,
 752                 self.cert.get_pubkey()
 753             )
 754         except AttributeError:
 755             try:
 756                 # pyOpenSSL < 16.0:
 757                 bio = crypto._new_mem_buf()
 758                 if binary:
 759                     rc = crypto._lib.i2d_PUBKEY_bio(bio, self.cert.get_pubkey()._pkey)
 760                 else:
 761                     rc = crypto._lib.PEM_write_bio_PUBKEY(bio, self.cert.get_pubkey()._pkey)
 762                 if rc != 1:
 763                     crypto._raise_current_error()
 764                 return crypto._bio_to_string(bio)
 765             except AttributeError:
 766                 self.module.warn('Your pyOpenSSL version does not support dumping public keys. '
 767                                  'Please upgrade to version 16.0 or newer, or use the cryptography backend.')

 794 def main():
 795     module = AnsibleModule(
 796         argument_spec=dict(
 797             path=dict(type='path'),
 798             content=dict(type='str'),
 799             valid_at=dict(type='dict'),
 800             select_crypto_backend=dict(type='str', default='auto', choices=['auto', 'cryptography', 'pyopenssl']),
 801         ),
 802         required_one_of=(
 803             ['path', 'content'],
 804         ),
 805         mutually_exclusive=(
 806             ['path', 'content'],
 807         ),
 808         supports_check_mode=True,
 809     )
 810 
 811     try:
 812         if module.params['path'] is not None:
 813             base_dir = os.path.dirname(module.params['path']) or '.'
 814             if not os.path.isdir(base_dir):
 815                 module.fail_json(
 816                     name=base_dir,
 817                     msg='The directory %s does not exist or the file is not a directory' % base_dir
 818                 )
 819 
 820         backend = module.params['select_crypto_backend']
 821         if backend == 'auto':
 822             # Detect what backend we can use
 823             can_use_cryptography = CRYPTOGRAPHY_FOUND and CRYPTOGRAPHY_VERSION >= LooseVersion(MINIMAL_CRYPTOGRAPHY_VERSION)
 824             can_use_pyopenssl = PYOPENSSL_FOUND and PYOPENSSL_VERSION >= LooseVersion(MINIMAL_PYOPENSSL_VERSION)
 825 
 826             # If cryptography is available we'll use it
 827             if can_use_cryptography:
 828                 backend = 'cryptography'
 829             elif can_use_pyopenssl:
 830                 backend = 'pyopenssl'
 831 
 832             # Fail if no backend has been found
 833             if backend == 'auto':
 834                 module.fail_json(msg=("Can't detect any of the required Python libraries "
 835                                       "cryptography (>= {0}) or PyOpenSSL (>= {1})").format(
 836                                           MINIMAL_CRYPTOGRAPHY_VERSION,
 837                                           MINIMAL_PYOPENSSL_VERSION))
 838 
 839         if backend == 'pyopenssl':
 840             if not PYOPENSSL_FOUND:
 841                 module.fail_json(msg=missing_required_lib('pyOpenSSL >= {0}'.format(MINIMAL_PYOPENSSL_VERSION)),
 842                                  exception=PYOPENSSL_IMP_ERR)
 843             try:
 844                 getattr(crypto.X509Req, 'get_extensions')
 845             except AttributeError:
 846                 module.fail_json(msg='You need to have PyOpenSSL>=0.15')
 847 
 848             module.deprecate('The module is using the PyOpenSSL backend. This backend has been deprecated',
 849                              version='2.13', collection_name='ansible.builtin')
 850             certificate = CertificateInfoPyOpenSSL(module)
 851         elif backend == 'cryptography':
 852             if not CRYPTOGRAPHY_FOUND:
 853                 module.fail_json(msg=missing_required_lib('cryptography >= {0}'.format(MINIMAL_CRYPTOGRAPHY_VERSION)),
 854                                  exception=CRYPTOGRAPHY_IMP_ERR)
 855             certificate = CertificateInfoCryptography(module)
 856 
 857         result = certificate.get_info()
 858         module.exit_json(**result)
 859     except crypto_utils.OpenSSLObjectError as exc:
 860         module.fail_json(msg=to_native(exc))
