 187 class CloudFormationServiceManager:
 188     """Handles CloudFormation Services"""
 189 
 190     def __init__(self, module):
 191         self.module = module
 192         self.client = module.client('cloudformation')
 193 
 194     @AWSRetry.exponential_backoff(retries=5, delay=5)
 195     def describe_stacks_with_backoff(self, **kwargs):
 196         paginator = self.client.get_paginator('describe_stacks')
 197         return paginator.paginate(**kwargs).build_full_result()['Stacks']
 198 
 199     def describe_stacks(self, stack_name=None):
 200         try:
 201             kwargs = {'StackName': stack_name} if stack_name else {}
 202             response = self.describe_stacks_with_backoff(**kwargs)
 203             if response is not None:
 204                 return response
 205             self.module.fail_json(msg="Error describing stack(s) - an empty response was returned")
 206         except (botocore.exceptions.BotoCoreError, botocore.exceptions.ClientError) as e:
 207             if 'does not exist' in e.response['Error']['Message']:
 208                 # missing stack, don't bail.
 209                 return {}
 210             self.module.fail_json_aws(e, msg="Error describing stack " + stack_name)
 211 
 212     @AWSRetry.exponential_backoff(retries=5, delay=5)
 213     def list_stack_resources_with_backoff(self, stack_name):
 214         paginator = self.client.get_paginator('list_stack_resources')
 215         return paginator.paginate(StackName=stack_name).build_full_result()['StackResourceSummaries']
 216 
 217     def list_stack_resources(self, stack_name):
 218         try:
 219             return self.list_stack_resources_with_backoff(stack_name)
 220         except (botocore.exceptions.BotoCoreError, botocore.exceptions.ClientError) as e:
 221             self.module.fail_json_aws(e, msg="Error listing stack resources for stack " + stack_name)
 222 
 223     @AWSRetry.exponential_backoff(retries=5, delay=5)
 224     def describe_stack_events_with_backoff(self, stack_name):
 225         paginator = self.client.get_paginator('describe_stack_events')
 226         return paginator.paginate(StackName=stack_name).build_full_result()['StackEvents']
 227 
 228     def describe_stack_events(self, stack_name):
 229         try:
 230             return self.describe_stack_events_with_backoff(stack_name)
 231         except (botocore.exceptions.BotoCoreError, botocore.exceptions.ClientError) as e:
 232             self.module.fail_json_aws(e, msg="Error listing stack events for stack " + stack_name)
 233 
 234     @AWSRetry.exponential_backoff(retries=5, delay=5)
 235     def list_stack_change_sets_with_backoff(self, stack_name):
 236         paginator = self.client.get_paginator('list_change_sets')
 237         return paginator.paginate(StackName=stack_name).build_full_result()['Summaries']
 238 
 239     @AWSRetry.exponential_backoff(retries=5, delay=5)
 240     def describe_stack_change_set_with_backoff(self, **kwargs):
 241         paginator = self.client.get_paginator('describe_change_set')
 242         return paginator.paginate(**kwargs).build_full_result()
 243 
 244     def describe_stack_change_sets(self, stack_name):
 245         changes = []
 246         try:
 247             change_sets = self.list_stack_change_sets_with_backoff(stack_name)
 248             for item in change_sets:
 249                 changes.append(self.describe_stack_change_set_with_backoff(
 250                                StackName=stack_name,
 251                                ChangeSetName=item['ChangeSetName']))
 252             return changes
 253         except (botocore.exceptions.BotoCoreError, botocore.exceptions.ClientError) as e:
 254             self.module.fail_json_aws(e, msg="Error describing stack change sets for stack " + stack_name)
 255 
 256     @AWSRetry.exponential_backoff(retries=5, delay=5)
 257     def get_stack_policy_with_backoff(self, stack_name):
 258         return self.client.get_stack_policy(StackName=stack_name)
 259 
 260     def get_stack_policy(self, stack_name):
 261         try:
 262             response = self.get_stack_policy_with_backoff(stack_name)
 263             stack_policy = response.get('StackPolicyBody')
 264             if stack_policy:
 265                 return json.loads(stack_policy)
 266             return dict()
 267         except (botocore.exceptions.BotoCoreError, botocore.exceptions.ClientError) as e:
 268             self.module.fail_json_aws(e, msg="Error getting stack policy for stack " + stack_name)
 269 
 270     @AWSRetry.exponential_backoff(retries=5, delay=5)
 271     def get_template_with_backoff(self, stack_name):
 272         return self.client.get_template(StackName=stack_name)
 273 
 274     def get_template(self, stack_name):
 275         try:
 276             response = self.get_template_with_backoff(stack_name)
 277             return response.get('TemplateBody')
 278         except (botocore.exceptions.BotoCoreError, botocore.exceptions.ClientError) as e:
 279             self.module.fail_json_aws(e, msg="Error getting stack template for stack " + stack_name)

 282 def to_dict(items, key, value):
 283     ''' Transforms a list of items to a Key/Value dictionary '''
 284     if items:
 285         return dict(zip([i.get(key) for i in items], [i.get(value) for i in items]))
 286     else:
 287         return dict()

 290 def main():
 291     argument_spec = dict(
 292         stack_name=dict(),
 293         all_facts=dict(required=False, default=False, type='bool'),
 294         stack_policy=dict(required=False, default=False, type='bool'),
 295         stack_events=dict(required=False, default=False, type='bool'),
 296         stack_resources=dict(required=False, default=False, type='bool'),
 297         stack_template=dict(required=False, default=False, type='bool'),
 298         stack_change_sets=dict(required=False, default=False, type='bool'),
 299     )
 300     module = AnsibleAWSModule(argument_spec=argument_spec, supports_check_mode=True)
 301 
 302     is_old_facts = module._name == 'cloudformation_facts'
 303     if is_old_facts:
 304         module.deprecate("The 'cloudformation_facts' module has been renamed to 'cloudformation_info', "
 305                          "and the renamed one no longer returns ansible_facts",
 306                          version='2.13', collection_name='ansible.builtin')
 307 
 308     service_mgr = CloudFormationServiceManager(module)
 309 
 310     if is_old_facts:
 311         result = {'ansible_facts': {'cloudformation': {}}}
 312     else:
 313         result = {'cloudformation': {}}
 314 
 315     for stack_description in service_mgr.describe_stacks(module.params.get('stack_name')):
 316         facts = {'stack_description': stack_description}
 317         stack_name = stack_description.get('StackName')
 318 
 319         # Create stack output and stack parameter dictionaries
 320         if facts['stack_description']:
 321             facts['stack_outputs'] = to_dict(facts['stack_description'].get('Outputs'), 'OutputKey', 'OutputValue')
 322             facts['stack_parameters'] = to_dict(facts['stack_description'].get('Parameters'),
 323                                                 'ParameterKey', 'ParameterValue')
 324             facts['stack_tags'] = boto3_tag_list_to_ansible_dict(facts['stack_description'].get('Tags'))
 325 
 326         # Create optional stack outputs
 327         all_facts = module.params.get('all_facts')
 328         if all_facts or module.params.get('stack_resources'):
 329             facts['stack_resource_list'] = service_mgr.list_stack_resources(stack_name)
 330             facts['stack_resources'] = to_dict(facts.get('stack_resource_list'),
 331                                                'LogicalResourceId', 'PhysicalResourceId')
 332         if all_facts or module.params.get('stack_template'):
 333             facts['stack_template'] = service_mgr.get_template(stack_name)
 334         if all_facts or module.params.get('stack_policy'):
 335             facts['stack_policy'] = service_mgr.get_stack_policy(stack_name)
 336         if all_facts or module.params.get('stack_events'):
 337             facts['stack_events'] = service_mgr.describe_stack_events(stack_name)
 338         if all_facts or module.params.get('stack_change_sets'):
 339             facts['stack_change_sets'] = service_mgr.describe_stack_change_sets(stack_name)
 340 
 341         if is_old_facts:
 342             result['ansible_facts']['cloudformation'][stack_name] = facts
 343         else:
 344             result['cloudformation'][stack_name] = camel_dict_to_snake_dict(facts, ignore_list=('stack_outputs',
 345                                                                                                 'stack_parameters',
 346                                                                                                 'stack_policy',
 347                                                                                                 'stack_resources',
 348                                                                                                 'stack_tags',
 349                                                                                                 'stack_template'))
 350 
 351     module.exit_json(changed=False, **result)
 352 
 353 
 354 if __name__ == '__main__':
 355     main()
