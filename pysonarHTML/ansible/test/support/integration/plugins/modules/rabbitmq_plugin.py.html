  86 import os
  87 from ansible.module_utils.basic import AnsibleModule
  88 
  89 
  90 class RabbitMqPlugins(object):
  91 
  92     def __init__(self, module):
  93         self.module = module
  94         bin_path = ''
  95         if module.params['prefix']:
  96             if os.path.isdir(os.path.join(module.params['prefix'], 'bin')):
  97                 bin_path = os.path.join(module.params['prefix'], 'bin')
  98             elif os.path.isdir(os.path.join(module.params['prefix'], 'sbin')):
  99                 bin_path = os.path.join(module.params['prefix'], 'sbin')
 100             else:
 101                 # No such path exists.
 102                 module.fail_json(msg="No binary folder in prefix %s" % module.params['prefix'])
 103 
 104             self._rabbitmq_plugins = os.path.join(bin_path, "rabbitmq-plugins")
 105         else:
 106             self._rabbitmq_plugins = module.get_bin_path('rabbitmq-plugins', True)
 107 
 108     def _exec(self, args, run_in_check_mode=False):
 109         if not self.module.check_mode or (self.module.check_mode and run_in_check_mode):
 110             cmd = [self._rabbitmq_plugins]
 111             rc, out, err = self.module.run_command(cmd + args, check_rc=True)
 112             return out.splitlines()
 113         return list()
 114 
 115     def get_all(self):
 116         list_output = self._exec(['list', '-E', '-m'], True)
 117         plugins = []
 118         for plugin in list_output:
 119             if not plugin:
 120                 break
 121             plugins.append(plugin)
 122 
 123         return plugins
 124 
 125     def enable(self, name):
 126         self._exec(['enable', name])
 127 
 128     def disable(self, name):
 129         self._exec(['disable', name])
 130 
 131 
 132 def main():
 133     arg_spec = dict(
 134         names=dict(required=True, aliases=['name']),
 135         new_only=dict(default='no', type='bool'),
 136         state=dict(default='enabled', choices=['enabled', 'disabled']),
 137         prefix=dict(required=False, default=None)
 138     )
 139     module = AnsibleModule(
 140         argument_spec=arg_spec,
 141         supports_check_mode=True
 142     )
 143 
 144     result = dict()
 145     names = module.params['names'].split(',')
 146     new_only = module.params['new_only']
 147     state = module.params['state']
 148 
 149     rabbitmq_plugins = RabbitMqPlugins(module)
 150     enabled_plugins = rabbitmq_plugins.get_all()
 151 
 152     enabled = []
 153     disabled = []
 154     if state == 'enabled':
 155         if not new_only:
 156             for plugin in enabled_plugins:
 157                 if " " in plugin:
 158                     continue
 159                 if plugin not in names:
 160                     rabbitmq_plugins.disable(plugin)
 161                     disabled.append(plugin)
 162 
 163         for name in names:
 164             if name not in enabled_plugins:
 165                 rabbitmq_plugins.enable(name)
 166                 enabled.append(name)
 167     else:
 168         for plugin in enabled_plugins:
 169             if plugin in names:
 170                 rabbitmq_plugins.disable(plugin)
 171                 disabled.append(plugin)
 172 
 173     result['changed'] = len(enabled) > 0 or len(disabled) > 0
 174     result['enabled'] = enabled
 175     result['disabled'] = disabled
 176     module.exit_json(**result)
 177 
 178 
 179 if __name__ == '__main__':
 180     main()
