 109 class AzureRMMariaDbDatabase(AzureRMModuleBase):
 110     """Configuration class for an Azure RM MariaDB Database resource"""
 111 
 112     def __init__(self):
 113         self.module_arg_spec = dict(
 114             resource_group=dict(
 115                 type='str',
 116                 required=True
 117             ),
 118             server_name=dict(
 119                 type='str',
 120                 required=True
 121             ),
 122             name=dict(
 123                 type='str',
 124                 required=True
 125             ),
 126             charset=dict(
 127                 type='str'
 128             ),
 129             collation=dict(
 130                 type='str'
 131             ),
 132             force_update=dict(
 133                 type='bool',
 134                 default=False
 135             ),
 136             state=dict(
 137                 type='str',
 138                 default='present',
 139                 choices=['present', 'absent']
 140             )
 141         )
 142 
 143         self.resource_group = None
 144         self.server_name = None
 145         self.name = None
 146         self.force_update = None
 147         self.parameters = dict()
 148 
 149         self.results = dict(changed=False)
 150         self.mgmt_client = None
 151         self.state = None
 152         self.to_do = Actions.NoAction
 153 
 154         super(AzureRMMariaDbDatabase, self).__init__(derived_arg_spec=self.module_arg_spec,
 155                                                      supports_check_mode=True,
 156                                                      supports_tags=False)
 157 
 158     def exec_module(self, **kwargs):
 159         """Main module execution method"""
 160 
 161         for key in list(self.module_arg_spec.keys()):
 162             if hasattr(self, key):
 163                 setattr(self, key, kwargs[key])
 164             elif kwargs[key] is not None:
 165                 if key == "charset":
 166                     self.parameters["charset"] = kwargs[key]
 167                 elif key == "collation":
 168                     self.parameters["collation"] = kwargs[key]
 169 
 170         old_response = None
 171         response = None
 172 
 173         self.mgmt_client = self.get_mgmt_svc_client(MariaDBManagementClient,
 174                                                     base_url=self._cloud_environment.endpoints.resource_manager)
 175 
 176         resource_group = self.get_resource_group(self.resource_group)
 177 
 178         old_response = self.get_mariadbdatabase()
 179 
 180         if not old_response:
 181             self.log("MariaDB Database instance doesn't exist")
 182             if self.state == 'absent':
 183                 self.log("Old instance didn't exist")
 184             else:
 185                 self.to_do = Actions.Create
 186         else:
 187             self.log("MariaDB Database instance already exists")
 188             if self.state == 'absent':
 189                 self.to_do = Actions.Delete
 190             elif self.state == 'present':
 191                 self.log("Need to check if MariaDB Database instance has to be deleted or may be updated")
 192                 if ('collation' in self.parameters) and (self.parameters['collation'] != old_response['collation']):
 193                     self.to_do = Actions.Update
 194                 if ('charset' in self.parameters) and (self.parameters['charset'] != old_response['charset']):
 195                     self.to_do = Actions.Update
 196         if self.to_do == Actions.Update:
 197             if self.force_update:
 198                 if not self.check_mode:
 199                     self.delete_mariadbdatabase()
 200             else:
 201                 self.fail("Database properties cannot be updated without setting 'force_update' option")
 202                 self.to_do = Actions.NoAction
 203 
 204         if (self.to_do == Actions.Create) or (self.to_do == Actions.Update):
 205             self.log("Need to Create / Update the MariaDB Database instance")
 206 
 207             if self.check_mode:
 208                 self.results['changed'] = True
 209                 return self.results
 210 
 211             response = self.create_update_mariadbdatabase()
 212             self.results['changed'] = True
 213             self.log("Creation / Update done")
 214         elif self.to_do == Actions.Delete:
 215             self.log("MariaDB Database instance deleted")
 216             self.results['changed'] = True
 217 
 218             if self.check_mode:
 219                 return self.results
 220 
 221             self.delete_mariadbdatabase()
 222             # make sure instance is actually deleted, for some Azure resources, instance is hanging around
 223             # for some time after deletion -- this should be really fixed in Azure
 224             while self.get_mariadbdatabase():
 225                 time.sleep(20)
 226         else:
 227             self.log("MariaDB Database instance unchanged")
 228             self.results['changed'] = False
 229             response = old_response
 230 
 231         if response:
 232             self.results["id"] = response["id"]
 233             self.results["name"] = response["name"]
 234 
 235         return self.results
 236 
 237     def create_update_mariadbdatabase(self):
 238         '''
 239         Creates or updates MariaDB Database with the specified configuration.
 240 
 241         :return: deserialized MariaDB Database instance state dictionary
 242         '''
 243         self.log("Creating / Updating the MariaDB Database instance {0}".format(self.name))
 244 
 245         try:
 246             response = self.mgmt_client.databases.create_or_update(resource_group_name=self.resource_group,
 247                                                                    server_name=self.server_name,
 248                                                                    database_name=self.name,
 249                                                                    parameters=self.parameters)
 250             if isinstance(response, LROPoller):
 251                 response = self.get_poller_result(response)
 252 
 253         except CloudError as exc:
 254             self.log('Error attempting to create the MariaDB Database instance.')
 255             self.fail("Error creating the MariaDB Database instance: {0}".format(str(exc)))
 256         return response.as_dict()
 257 
 258     def delete_mariadbdatabase(self):
 259         '''
 260         Deletes specified MariaDB Database instance in the specified subscription and resource group.
 261 
 262         :return: True
 263         '''
 264         self.log("Deleting the MariaDB Database instance {0}".format(self.name))
 265         try:
 266             response = self.mgmt_client.databases.delete(resource_group_name=self.resource_group,
 267                                                          server_name=self.server_name,
 268                                                          database_name=self.name)
 269         except CloudError as e:
 270             self.log('Error attempting to delete the MariaDB Database instance.')
 271             self.fail("Error deleting the MariaDB Database instance: {0}".format(str(e)))
 272 
 273         return True
 274 
 275     def get_mariadbdatabase(self):
 276         '''
 277         Gets the properties of the specified MariaDB Database.
 278 
 279         :return: deserialized MariaDB Database instance state dictionary
 280         '''
 281         self.log("Checking if the MariaDB Database instance {0} is present".format(self.name))
 282         found = False
 283         try:
 284             response = self.mgmt_client.databases.get(resource_group_name=self.resource_group,
 285                                                       server_name=self.server_name,
 286                                                       database_name=self.name)
 287             found = True
 288             self.log("Response : {0}".format(response))
 289             self.log("MariaDB Database instance : {0} found".format(response.name))
 290         except CloudError as e:
 291             self.log('Did not find the MariaDB Database instance.')
 292         if found is True:
 293             return response.as_dict()
 294 
 295         return False
