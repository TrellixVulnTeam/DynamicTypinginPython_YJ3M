 105 def start_execution(module, sfn_client):
 106     '''
 107     start_execution uses execution name to determine if a previous execution already exists.
 108     If an execution by the provided name exists, call client.start_execution will not be called.
 109     '''
 110 
 111     state_machine_arn = module.params.get('state_machine_arn')
 112     name = module.params.get('name')
 113     execution_input = module.params.get('execution_input')
 114 
 115     try:
 116         # list_executions is eventually consistent
 117         page_iterators = sfn_client.get_paginator('list_executions').paginate(stateMachineArn=state_machine_arn)
 118 
 119         for execution in page_iterators.build_full_result()['executions']:
 120             if name == execution['name']:
 121                 check_mode(module, msg='State machine execution already exists.', changed=False)
 122                 module.exit_json(changed=False)
 123 
 124         check_mode(module, msg='State machine execution would be started.', changed=True)
 125         res_execution = sfn_client.start_execution(
 126             stateMachineArn=state_machine_arn,
 127             name=name,
 128             input=execution_input
 129         )
 130     except (ClientError, BotoCoreError) as e:
 131         if e.response['Error']['Code'] == 'ExecutionAlreadyExists':
 132             # this will never be executed anymore
 133             module.exit_json(changed=False)
 134         module.fail_json_aws(e, msg="Failed to start execution.")
 135 
 136     module.exit_json(changed=True, **camel_dict_to_snake_dict(res_execution))
 137

 139 def stop_execution(module, sfn_client):
 140 
 141     cause = module.params.get('cause')
 142     error = module.params.get('error')
 143     execution_arn = module.params.get('execution_arn')
 144 
 145     try:
 146         # describe_execution is eventually consistent
 147         execution_status = sfn_client.describe_execution(executionArn=execution_arn)['status']
 148         if execution_status != 'RUNNING':
 149             check_mode(module, msg='State machine execution is not running.', changed=False)
 150             module.exit_json(changed=False)
 151 
 152         check_mode(module, msg='State machine execution would be stopped.', changed=True)
 153         res = sfn_client.stop_execution(
 154             executionArn=execution_arn,
 155             cause=cause,
 156             error=error
 157         )
 158     except (ClientError, BotoCoreError) as e:
 159         module.fail_json_aws(e, msg="Failed to stop execution.")
 160 
 161     module.exit_json(changed=True, **camel_dict_to_snake_dict(res))
 162

 164 def check_mode(module, msg='', changed=False):
 165     if module.check_mode:
 166         module.exit_json(changed=changed, output=msg)

 169 def main():
 170     module_args = dict(
 171         action=dict(choices=['start', 'stop'], default='start'),
 172         name=dict(type='str'),
 173         execution_input=dict(type='json', default={}),
 174         state_machine_arn=dict(type='str'),
 175         cause=dict(type='str', default=''),
 176         error=dict(type='str', default=''),
 177         execution_arn=dict(type='str')
 178     )
 179     module = AnsibleAWSModule(
 180         argument_spec=module_args,
 181         required_if=[('action', 'start', ['name', 'state_machine_arn']),
 182                      ('action', 'stop', ['execution_arn']),
 183                      ],
 184         supports_check_mode=True
 185     )
 186 
 187     sfn_client = module.client('stepfunctions')
 188 
 189     action = module.params.get('action')
 190     if action == "start":
 191         start_execution(module, sfn_client)
 192     else:
 193         stop_execution(module, sfn_client)
