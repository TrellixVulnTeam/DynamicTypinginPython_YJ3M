 497 import traceback

 499 try:
 500     import boto3
 501     from botocore.exceptions import ClientError
 502     HAS_BOTO3 = True
 503 except ImportError:
 504     HAS_BOTO3 = False

 506 from ansible.module_utils.basic import AnsibleModule
 507 from ansible.module_utils.ec2 import (ansible_dict_to_boto3_filter_list,
 508                                       boto3_conn, boto3_tag_list_to_ansible_dict, camel_dict_to_snake_dict,
 509                                       ec2_argument_spec, get_aws_connection_info)
 510 
 511 
 512 def list_ec2_instances(connection, module):
 513 
 514     instance_ids = module.params.get("instance_ids")
 515     filters = ansible_dict_to_boto3_filter_list(module.params.get("filters"))
 516 
 517     try:
 518         reservations_paginator = connection.get_paginator('describe_instances')
 519         reservations = reservations_paginator.paginate(InstanceIds=instance_ids, Filters=filters).build_full_result()
 520     except ClientError as e:
 521         module.fail_json(msg=e.message, exception=traceback.format_exc(), **camel_dict_to_snake_dict(e.response))
 522 
 523     # Get instances from reservations
 524     instances = []
 525     for reservation in reservations['Reservations']:
 526         instances = instances + reservation['Instances']
 527 
 528     # Turn the boto3 result in to ansible_friendly_snaked_names
 529     snaked_instances = [camel_dict_to_snake_dict(instance) for instance in instances]
 530 
 531     # Turn the boto3 result in to ansible friendly tag dictionary
 532     for instance in snaked_instances:
 533         instance['tags'] = boto3_tag_list_to_ansible_dict(instance.get('tags', []), 'key', 'value')
 534 
 535     module.exit_json(instances=snaked_instances)
