 193     def exec_module(self, **kwargs):
 194         is_old_facts = self.module._name == 'azure_rm_mariadbserver_facts'
 195         if is_old_facts:
 196             self.module.deprecate("The 'azure_rm_mariadbserver_facts' module has been renamed to 'azure_rm_mariadbserver_info'",
 197                                   version='2.13', collection_name='ansible.builtin')
 198 
 199         for key in self.module_arg_spec:
 200             setattr(self, key, kwargs[key])
 201 
 202         if (self.resource_group is not None and
 203                 self.name is not None):
 204             self.results['servers'] = self.get()
 205         elif (self.resource_group is not None):
 206             self.results['servers'] = self.list_by_resource_group()
 207         return self.results
 208 
 209     def get(self):
 210         response = None
 211         results = []
 212         try:
 213             response = self.mariadb_client.servers.get(resource_group_name=self.resource_group,
 214                                                        server_name=self.name)
 215             self.log("Response : {0}".format(response))
 216         except CloudError as e:
 217             self.log('Could not get facts for MariaDB Server.')
 218 
 219         if response and self.has_tags(response.tags, self.tags):
 220             results.append(self.format_item(response))
 221 
 222         return results
 223 
 224     def list_by_resource_group(self):
 225         response = None
 226         results = []
 227         try:
 228             response = self.mariadb_client.servers.list_by_resource_group(resource_group_name=self.resource_group)
 229             self.log("Response : {0}".format(response))
 230         except CloudError as e:
 231             self.log('Could not get facts for MariaDB Servers.')
 232 
 233         if response is not None:
 234             for item in response:
 235                 if self.has_tags(item.tags, self.tags):
 236                     results.append(self.format_item(item))
 237 
 238         return results
 239 
 240     def format_item(self, item):
 241         d = item.as_dict()
 242         d = {
 243             'id': d['id'],
 244             'resource_group': self.resource_group,
 245             'name': d['name'],
 246             'sku': d['sku'],
 247             'location': d['location'],
 248             'storage_mb': d['storage_profile']['storage_mb'],
 249             'version': d['version'],
 250             'enforce_ssl': (d['ssl_enforcement'] == 'Enabled'),
 251             'admin_username': d['administrator_login'],
 252             'user_visible_state': d['user_visible_state'],
 253             'fully_qualified_domain_name': d['fully_qualified_domain_name'],
 254             'tags': d.get('tags')
 255         }
 256 
 257         return d
