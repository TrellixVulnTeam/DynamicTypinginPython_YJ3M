 106 class AnsibleEc2Igw(object):
 107 
 108     def __init__(self, module, results):
 109         self._module = module
 110         self._results = results
 111         self._connection = self._module.client('ec2')
 112         self._check_mode = self._module.check_mode
 113 
 114     def process(self):
 115         vpc_id = self._module.params.get('vpc_id')
 116         state = self._module.params.get('state', 'present')
 117         tags = self._module.params.get('tags')
 118 
 119         if state == 'present':
 120             self.ensure_igw_present(vpc_id, tags)
 121         elif state == 'absent':
 122             self.ensure_igw_absent(vpc_id)
 123 
 124     def get_matching_igw(self, vpc_id):
 125         filters = ansible_dict_to_boto3_filter_list({'attachment.vpc-id': vpc_id})
 126         igws = []
 127         try:
 128             response = self._connection.describe_internet_gateways(Filters=filters)
 129             igws = response.get('InternetGateways', [])
 130         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 131             self._module.fail_json_aws(e)
 132 
 133         igw = None
 134         if len(igws) > 1:
 135             self._module.fail_json(
 136                 msg='EC2 returned more than one Internet Gateway for VPC {0}, aborting'.format(vpc_id))
 137         elif igws:
 138             igw = camel_dict_to_snake_dict(igws[0])
 139 
 140         return igw
 141 
 142     def check_input_tags(self, tags):
 143         nonstring_tags = [k for k, v in tags.items() if not isinstance(v, string_types)]
 144         if nonstring_tags:
 145             self._module.fail_json(msg='One or more tags contain non-string values: {0}'.format(nonstring_tags))
 146 
 147     def ensure_tags(self, igw_id, tags, add_only):
 148         final_tags = []
 149 
 150         filters = ansible_dict_to_boto3_filter_list({'resource-id': igw_id, 'resource-type': 'internet-gateway'})
 151         cur_tags = None
 152         try:
 153             cur_tags = self._connection.describe_tags(Filters=filters)
 154         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 155             self._module.fail_json_aws(e, msg="Couldn't describe tags")
 156 
 157         purge_tags = bool(not add_only)
 158         to_update, to_delete = compare_aws_tags(boto3_tag_list_to_ansible_dict(cur_tags.get('Tags')), tags, purge_tags)
 159         final_tags = boto3_tag_list_to_ansible_dict(cur_tags.get('Tags'))
 160 
 161         if to_update:
 162             try:
 163                 if self._check_mode:
 164                     # update tags
 165                     final_tags.update(to_update)
 166                 else:
 167                     AWSRetry.exponential_backoff()(self._connection.create_tags)(
 168                         Resources=[igw_id],
 169                         Tags=ansible_dict_to_boto3_tag_list(to_update)
 170                     )
 171 
 172                 self._results['changed'] = True
 173             except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 174                 self._module.fail_json_aws(e, msg="Couldn't create tags")
 175 
 176         if to_delete:
 177             try:
 178                 if self._check_mode:
 179                     # update tags
 180                     for key in to_delete:
 181                         del final_tags[key]
 182                 else:
 183                     tags_list = []
 184                     for key in to_delete:
 185                         tags_list.append({'Key': key})
 186 
 187                     AWSRetry.exponential_backoff()(self._connection.delete_tags)(Resources=[igw_id], Tags=tags_list)
 188 
 189                 self._results['changed'] = True
 190             except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 191                 self._module.fail_json_aws(e, msg="Couldn't delete tags")
 192 
 193         if not self._check_mode and (to_update or to_delete):
 194             try:
 195                 response = self._connection.describe_tags(Filters=filters)
 196                 final_tags = boto3_tag_list_to_ansible_dict(response.get('Tags'))
 197             except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 198                 self._module.fail_json_aws(e, msg="Couldn't describe tags")
 199 
 200         return final_tags
 201 
 202     @staticmethod
 203     def get_igw_info(igw):
 204         return {
 205             'gateway_id': igw['internet_gateway_id'],
 206             'tags': igw['tags'],
 207             'vpc_id': igw['vpc_id']
 208         }
 209 
 210     def ensure_igw_absent(self, vpc_id):
 211         igw = self.get_matching_igw(vpc_id)
 212         if igw is None:
 213             return self._results
 214 
 215         if self._check_mode:
 216             self._results['changed'] = True
 217             return self._results
 218 
 219         try:
 220             self._results['changed'] = True
 221             self._connection.detach_internet_gateway(InternetGatewayId=igw['internet_gateway_id'], VpcId=vpc_id)
 222             self._connection.delete_internet_gateway(InternetGatewayId=igw['internet_gateway_id'])
 223         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 224             self._module.fail_json_aws(e, msg="Unable to delete Internet Gateway")
 225 
 226         return self._results
 227 
 228     def ensure_igw_present(self, vpc_id, tags):
 229         self.check_input_tags(tags)
 230 
 231         igw = self.get_matching_igw(vpc_id)
 232 
 233         if igw is None:
 234             if self._check_mode:
 235                 self._results['changed'] = True
 236                 self._results['gateway_id'] = None
 237                 return self._results
 238 
 239             try:
 240                 response = self._connection.create_internet_gateway()
 241 
 242                 # Ensure the gateway exists before trying to attach it or add tags
 243                 waiter = get_waiter(self._connection, 'internet_gateway_exists')
 244                 waiter.wait(InternetGatewayIds=[response['InternetGateway']['InternetGatewayId']])
 245 
 246                 igw = camel_dict_to_snake_dict(response['InternetGateway'])
 247                 self._connection.attach_internet_gateway(InternetGatewayId=igw['internet_gateway_id'], VpcId=vpc_id)
 248                 self._results['changed'] = True
 249             except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 250                 self._module.fail_json_aws(e, msg='Unable to create Internet Gateway')
 251 
 252         igw['vpc_id'] = vpc_id
 253 
 254         igw['tags'] = self.ensure_tags(igw_id=igw['internet_gateway_id'], tags=tags, add_only=False)
 255 
 256         igw_info = self.get_igw_info(igw)
 257         self._results.update(igw_info)
 258 
 259         return self._results
