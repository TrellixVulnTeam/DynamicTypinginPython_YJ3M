 359 def get_stack_events(cfn, stack_name, events_limit, token_filter=None):
 360     '''This event data was never correct, it worked as a side effect. So the v2.3 format is different.'''
 361     ret = {'events': [], 'log': []}
 362 
 363     try:
 364         pg = cfn.get_paginator(
 365             'describe_stack_events'
 366         ).paginate(
 367             StackName=stack_name,
 368             PaginationConfig={'MaxItems': events_limit}
 369         )
 370         if token_filter is not None:
 371             events = list(pg.search(
 372                 "StackEvents[?ClientRequestToken == '{0}']".format(token_filter)
 373             ))
 374         else:
 375             events = list(pg.search("StackEvents[*]"))
 376     except (botocore.exceptions.ValidationError, botocore.exceptions.ClientError) as err:
 377         error_msg = boto_exception(err)
 378         if 'does not exist' in error_msg:
 379             # missing stack, don't bail.
 380             ret['log'].append('Stack does not exist.')
 381             return ret
 382         ret['log'].append('Unknown error: ' + str(error_msg))
 383         return ret
 384 
 385     for e in events:
 386         eventline = 'StackEvent {ResourceType} {LogicalResourceId} {ResourceStatus}'.format(**e)
 387         ret['events'].append(eventline)
 388 
 389         if e['ResourceStatus'].endswith('FAILED'):
 390             failline = '{ResourceType} {LogicalResourceId} {ResourceStatus}: {ResourceStatusReason}'.format(**e)
 391             ret['log'].append(failline)
 392 
 393     return ret

 396 def create_stack(module, stack_params, cfn, events_limit):
 397     if 'TemplateBody' not in stack_params and 'TemplateURL' not in stack_params:
 398         module.fail_json(msg="Either 'template', 'template_body' or 'template_url' is required when the stack does not exist.")
 399 
 400     # 'DisableRollback', 'TimeoutInMinutes', 'EnableTerminationProtection' and
 401     # 'OnFailure' only apply on creation, not update.
 402     if module.params.get('on_create_failure') is not None:
 403         stack_params['OnFailure'] = module.params['on_create_failure']
 404     else:
 405         stack_params['DisableRollback'] = module.params['disable_rollback']
 406 
 407     if module.params.get('create_timeout') is not None:
 408         stack_params['TimeoutInMinutes'] = module.params['create_timeout']
 409     if module.params.get('termination_protection') is not None:
 410         if boto_supports_termination_protection(cfn):
 411             stack_params['EnableTerminationProtection'] = bool(module.params.get('termination_protection'))
 412         else:
 413             module.fail_json(msg="termination_protection parameter requires botocore >= 1.7.18")
 414 
 415     try:
 416         response = cfn.create_stack(**stack_params)
 417         # Use stack ID to follow stack state in case of on_create_failure = DELETE
 418         result = stack_operation(cfn, response['StackId'], 'CREATE', events_limit, stack_params.get('ClientRequestToken', None))
 419     except Exception as err:
 420         error_msg = boto_exception(err)
 421         module.fail_json(msg="Failed to create stack {0}: {1}.".format(stack_params.get('StackName'), error_msg), exception=traceback.format_exc())
 422     if not result:
 423         module.fail_json(msg="empty result")
 424     return result

 432 def create_changeset(module, stack_params, cfn, events_limit):
 433     if 'TemplateBody' not in stack_params and 'TemplateURL' not in stack_params:
 434         module.fail_json(msg="Either 'template' or 'template_url' is required.")
 435     if module.params['changeset_name'] is not None:
 436         stack_params['ChangeSetName'] = module.params['changeset_name']
 437 
 438     # changesets don't accept ClientRequestToken parameters
 439     stack_params.pop('ClientRequestToken', None)
 440 
 441     try:
 442         changeset_name = build_changeset_name(stack_params)
 443         stack_params['ChangeSetName'] = changeset_name
 444 
 445         # Determine if this changeset already exists
 446         pending_changesets = list_changesets(cfn, stack_params['StackName'])
 447         if changeset_name in pending_changesets:
 448             warning = 'WARNING: %d pending changeset(s) exist(s) for this stack!' % len(pending_changesets)
 449             result = dict(changed=False, output='ChangeSet %s already exists.' % changeset_name, warnings=[warning])
 450         else:
 451             cs = cfn.create_change_set(**stack_params)
 452             # Make sure we don't enter an infinite loop
 453             time_end = time.time() + 600
 454             while time.time() < time_end:
 455                 try:
 456                     newcs = cfn.describe_change_set(ChangeSetName=cs['Id'])
 457                 except botocore.exceptions.BotoCoreError as err:
 458                     error_msg = boto_exception(err)
 459                     module.fail_json(msg=error_msg)
 460                 if newcs['Status'] == 'CREATE_PENDING' or newcs['Status'] == 'CREATE_IN_PROGRESS':
 461                     time.sleep(1)
 462                 elif newcs['Status'] == 'FAILED' and "The submitted information didn't contain changes" in newcs['StatusReason']:
 463                     cfn.delete_change_set(ChangeSetName=cs['Id'])
 464                     result = dict(changed=False,
 465                                   output='The created Change Set did not contain any changes to this stack and was deleted.')
 466                     # a failed change set does not trigger any stack events so we just want to
 467                     # skip any further processing of result and just return it directly
 468                     return result
 469                 else:
 470                     break
 471                 # Lets not hog the cpu/spam the AWS API
 472                 time.sleep(1)
 473             result = stack_operation(cfn, stack_params['StackName'], 'CREATE_CHANGESET', events_limit)
 474             result['change_set_id'] = cs['Id']
 475             result['warnings'] = ['Created changeset named %s for stack %s' % (changeset_name, stack_params['StackName']),
 476                                   'You can execute it using: aws cloudformation execute-change-set --change-set-name %s' % cs['Id'],
 477                                   'NOTE that dependencies on this stack might fail due to pending changes!']
 478     except Exception as err:
 479         error_msg = boto_exception(err)
 480         if 'No updates are to be performed.' in error_msg:
 481             result = dict(changed=False, output='Stack is already up-to-date.')
 482         else:
 483             module.fail_json(msg="Failed to create change set: {0}".format(error_msg), exception=traceback.format_exc())
 484 
 485     if not result:
 486         module.fail_json(msg="empty result")
 487     return result

 490 def update_stack(module, stack_params, cfn, events_limit):
 491     if 'TemplateBody' not in stack_params and 'TemplateURL' not in stack_params:
 492         stack_params['UsePreviousTemplate'] = True
 493 
 494     # if the state is present and the stack already exists, we try to update it.
 495     # AWS will tell us if the stack template and parameters are the same and
 496     # don't need to be updated.
 497     try:
 498         cfn.update_stack(**stack_params)
 499         result = stack_operation(cfn, stack_params['StackName'], 'UPDATE', events_limit, stack_params.get('ClientRequestToken', None))
 500     except Exception as err:
 501         error_msg = boto_exception(err)
 502         if 'No updates are to be performed.' in error_msg:
 503             result = dict(changed=False, output='Stack is already up-to-date.')
 504         else:
 505             module.fail_json(msg="Failed to update stack {0}: {1}".format(stack_params.get('StackName'), error_msg), exception=traceback.format_exc())
 506     if not result:
 507         module.fail_json(msg="empty result")
 508     return result

 511 def update_termination_protection(module, cfn, stack_name, desired_termination_protection_state):
 512     '''updates termination protection of a stack'''
 513     if not boto_supports_termination_protection(cfn):
 514         module.fail_json(msg="termination_protection parameter requires botocore >= 1.7.18")
 515     stack = get_stack_facts(cfn, stack_name)
 516     if stack:
 517         if stack['EnableTerminationProtection'] is not desired_termination_protection_state:
 518             try:
 519                 cfn.update_termination_protection(
 520                     EnableTerminationProtection=desired_termination_protection_state,
 521                     StackName=stack_name)
 522             except botocore.exceptions.ClientError as e:
 523                 module.fail_json(msg=boto_exception(e), exception=traceback.format_exc())

 526 def boto_supports_termination_protection(cfn):
 527     '''termination protection was added in botocore 1.7.18'''
 528     return hasattr(cfn, "update_termination_protection")

 531 def stack_operation(cfn, stack_name, operation, events_limit, op_token=None):
 532     '''gets the status of a stack while it is created/updated/deleted'''
 533     existed = []
 534     while True:
 535         try:
 536             stack = get_stack_facts(cfn, stack_name)
 537             existed.append('yes')
 538         except Exception:
 539             # If the stack previously existed, and now can't be found then it's
 540             # been deleted successfully.
 541             if 'yes' in existed or operation == 'DELETE':  # stacks may delete fast, look in a few ways.
 542                 ret = get_stack_events(cfn, stack_name, events_limit, op_token)
 543                 ret.update({'changed': True, 'output': 'Stack Deleted'})
 544                 return ret
 545             else:
 546                 return {'changed': True, 'failed': True, 'output': 'Stack Not Found', 'exception': traceback.format_exc()}
 547         ret = get_stack_events(cfn, stack_name, events_limit, op_token)
 548         if not stack:
 549             if 'yes' in existed or operation == 'DELETE':  # stacks may delete fast, look in a few ways.
 550                 ret = get_stack_events(cfn, stack_name, events_limit, op_token)
 551                 ret.update({'changed': True, 'output': 'Stack Deleted'})
 552                 return ret
 553             else:
 554                 ret.update({'changed': False, 'failed': True, 'output': 'Stack not found.'})
 555                 return ret
 556         # it covers ROLLBACK_COMPLETE and UPDATE_ROLLBACK_COMPLETE
 557         # Possible states: https://docs.aws.amazon.com/AWSCloudFormation/latest/UserGuide/using-cfn-describing-stacks.html#w1ab2c15c17c21c13
 558         elif stack['StackStatus'].endswith('ROLLBACK_COMPLETE') and operation != 'CREATE_CHANGESET':
 559             ret.update({'changed': True, 'failed': True, 'output': 'Problem with %s. Rollback complete' % operation})
 560             return ret
 561         elif stack['StackStatus'] == 'DELETE_COMPLETE' and operation == 'CREATE':
 562             ret.update({'changed': True, 'failed': True, 'output': 'Stack create failed. Delete complete.'})
 563             return ret
 564         # note the ordering of ROLLBACK_COMPLETE, DELETE_COMPLETE, and COMPLETE, because otherwise COMPLETE will match all cases.
 565         elif stack['StackStatus'].endswith('_COMPLETE'):
 566             ret.update({'changed': True, 'output': 'Stack %s complete' % operation})
 567             return ret
 568         elif stack['StackStatus'].endswith('_ROLLBACK_FAILED'):
 569             ret.update({'changed': True, 'failed': True, 'output': 'Stack %s rollback failed' % operation})
 570             return ret
 571         # note the ordering of ROLLBACK_FAILED and FAILED, because otherwise FAILED will match both cases.
 572         elif stack['StackStatus'].endswith('_FAILED'):
 573             ret.update({'changed': True, 'failed': True, 'output': 'Stack %s failed' % operation})
 574             return ret
 575         else:
 576             # this can loop forever :/
 577             time.sleep(5)
 578     return {'failed': True, 'output': 'Failed for unknown reasons.'}

 581 def build_changeset_name(stack_params):
 582     if 'ChangeSetName' in stack_params:
 583         return stack_params['ChangeSetName']
 584 
 585     json_params = json.dumps(stack_params, sort_keys=True)
 586 
 587     return 'Ansible-{0}-{1}'.format(
 588         stack_params['StackName'],
 589         sha1(to_bytes(json_params, errors='surrogate_or_strict')).hexdigest()
 590     )

 593 def check_mode_changeset(module, stack_params, cfn):
 594     """Create a change set, describe it and delete it before returning check mode outputs."""
 595     stack_params['ChangeSetName'] = build_changeset_name(stack_params)
 596     # changesets don't accept ClientRequestToken parameters
 597     stack_params.pop('ClientRequestToken', None)
 598 
 599     try:
 600         change_set = cfn.create_change_set(**stack_params)
 601         for i in range(60):  # total time 5 min
 602             description = cfn.describe_change_set(ChangeSetName=change_set['Id'])
 603             if description['Status'] in ('CREATE_COMPLETE', 'FAILED'):
 604                 break
 605             time.sleep(5)
 606         else:
 607             # if the changeset doesn't finish in 5 mins, this `else` will trigger and fail
 608             module.fail_json(msg="Failed to create change set %s" % stack_params['ChangeSetName'])
 609 
 610         cfn.delete_change_set(ChangeSetName=change_set['Id'])
 611 
 612         reason = description.get('StatusReason')
 613 
 614         if description['Status'] == 'FAILED' and "didn't contain changes" in description['StatusReason']:
 615             return {'changed': False, 'msg': reason, 'meta': description['StatusReason']}
 616         return {'changed': True, 'msg': reason, 'meta': description['Changes']}
 617 
 618     except (botocore.exceptions.ValidationError, botocore.exceptions.ClientError) as err:
 619         error_msg = boto_exception(err)
 620         module.fail_json(msg=error_msg, exception=traceback.format_exc())

 623 def get_stack_facts(cfn, stack_name):
 624     try:
 625         stack_response = cfn.describe_stacks(StackName=stack_name)
 626         stack_info = stack_response['Stacks'][0]
 627     except (botocore.exceptions.ValidationError, botocore.exceptions.ClientError) as err:
 628         error_msg = boto_exception(err)
 629         if 'does not exist' in error_msg:
 630             # missing stack, don't bail.
 631             return None
 632 
 633         # other error, bail.
 634         raise err
 635 
 636     if stack_response and stack_response.get('Stacks', None):
 637         stacks = stack_response['Stacks']
 638         if len(stacks):
 639             stack_info = stacks[0]
 640 
 641     return stack_info

 644 def main():
 645     argument_spec = ec2_argument_spec()
 646     argument_spec.update(dict(
 647         stack_name=dict(required=True),
 648         template_parameters=dict(required=False, type='dict', default={}),
 649         state=dict(default='present', choices=['present', 'absent']),
 650         template=dict(default=None, required=False, type='path'),
 651         notification_arns=dict(default=None, required=False),
 652         stack_policy=dict(default=None, required=False),
 653         disable_rollback=dict(default=False, type='bool'),
 654         on_create_failure=dict(default=None, required=False, choices=['DO_NOTHING', 'ROLLBACK', 'DELETE']),
 655         create_timeout=dict(default=None, type='int'),
 656         template_url=dict(default=None, required=False),
 657         template_body=dict(default=None, required=False),
 658         template_format=dict(removed_in_version='2.14'),
 659         create_changeset=dict(default=False, type='bool'),
 660         changeset_name=dict(default=None, required=False),
 661         role_arn=dict(default=None, required=False),
 662         tags=dict(default=None, type='dict'),
 663         termination_protection=dict(default=None, type='bool'),
 664         events_limit=dict(default=200, type='int'),
 665         backoff_retries=dict(type='int', default=10, required=False),
 666         backoff_delay=dict(type='int', default=3, required=False),
 667         backoff_max_delay=dict(type='int', default=30, required=False),
 668         capabilities=dict(type='list', default=['CAPABILITY_IAM', 'CAPABILITY_NAMED_IAM'])
 669     )
 670     )
 671 
 672     module = AnsibleModule(
 673         argument_spec=argument_spec,
 674         mutually_exclusive=[['template_url', 'template', 'template_body'],
 675                             ['disable_rollback', 'on_create_failure']],
 676         supports_check_mode=True
 677     )
 678     if not HAS_BOTO3:
 679         module.fail_json(msg='boto3 and botocore are required for this module')
 680 
 681     invalid_capabilities = []
 682     user_capabilities = module.params.get('capabilities')
 683     for user_cap in user_capabilities:
 684         if user_cap not in ['CAPABILITY_IAM', 'CAPABILITY_NAMED_IAM', 'CAPABILITY_AUTO_EXPAND']:
 685             invalid_capabilities.append(user_cap)
 686 
 687     if invalid_capabilities:
 688         module.fail_json(msg="Specified capabilities are invalid : %r,"
 689                              " please check documentation for valid capabilities" % invalid_capabilities)
 690 
 691     # collect the parameters that are passed to boto3. Keeps us from having so many scalars floating around.
 692     stack_params = {
 693         'Capabilities': user_capabilities,
 694         'ClientRequestToken': to_native(uuid.uuid4()),
 695     }
 696     state = module.params['state']
 697     stack_params['StackName'] = module.params['stack_name']
 698 
 699     if module.params['template'] is not None:
 700         with open(module.params['template'], 'r') as template_fh:
 701             stack_params['TemplateBody'] = template_fh.read()
 702     elif module.params['template_body'] is not None:
 703         stack_params['TemplateBody'] = module.params['template_body']
 704     elif module.params['template_url'] is not None:
 705         stack_params['TemplateURL'] = module.params['template_url']
 706 
 707     if module.params.get('notification_arns'):
 708         stack_params['NotificationARNs'] = module.params['notification_arns'].split(',')
 709     else:
 710         stack_params['NotificationARNs'] = []
 711 
 712     # can't check the policy when verifying.
 713     if module.params['stack_policy'] is not None and not module.check_mode and not module.params['create_changeset']:
 714         with open(module.params['stack_policy'], 'r') as stack_policy_fh:
 715             stack_params['StackPolicyBody'] = stack_policy_fh.read()
 716 
 717     template_parameters = module.params['template_parameters']
 718 
 719     stack_params['Parameters'] = []
 720     for k, v in template_parameters.items():
 721         if isinstance(v, dict):
 722             # set parameter based on a dict to allow additional CFN Parameter Attributes
 723             param = dict(ParameterKey=k)
 724 
 725             if 'value' in v:
 726                 param['ParameterValue'] = str(v['value'])
 727 
 728             if 'use_previous_value' in v and bool(v['use_previous_value']):
 729                 param['UsePreviousValue'] = True
 730                 param.pop('ParameterValue', None)
 731 
 732             stack_params['Parameters'].append(param)
 733         else:
 734             # allow default k/v configuration to set a template parameter
 735             stack_params['Parameters'].append({'ParameterKey': k, 'ParameterValue': str(v)})
 736 
 737     if isinstance(module.params.get('tags'), dict):
 738         stack_params['Tags'] = ansible_dict_to_boto3_tag_list(module.params['tags'])
 739 
 740     if module.params.get('role_arn'):
 741         stack_params['RoleARN'] = module.params['role_arn']
 742 
 743     result = {}
 744 
 745     try:
 746         region, ec2_url, aws_connect_kwargs = get_aws_connection_info(module, boto3=True)
 747         cfn = boto3_conn(module, conn_type='client', resource='cloudformation', region=region, endpoint=ec2_url, **aws_connect_kwargs)
 748     except botocore.exceptions.NoCredentialsError as e:
 749         module.fail_json(msg=boto_exception(e))
 750 
 751     # Wrap the cloudformation client methods that this module uses with
 752     # automatic backoff / retry for throttling error codes
 753     backoff_wrapper = AWSRetry.jittered_backoff(
 754         retries=module.params.get('backoff_retries'),
 755         delay=module.params.get('backoff_delay'),
 756         max_delay=module.params.get('backoff_max_delay')
 757     )
 758     cfn.describe_stack_events = backoff_wrapper(cfn.describe_stack_events)
 759     cfn.create_stack = backoff_wrapper(cfn.create_stack)
 760     cfn.list_change_sets = backoff_wrapper(cfn.list_change_sets)
 761     cfn.create_change_set = backoff_wrapper(cfn.create_change_set)
 762     cfn.update_stack = backoff_wrapper(cfn.update_stack)
 763     cfn.describe_stacks = backoff_wrapper(cfn.describe_stacks)
 764     cfn.list_stack_resources = backoff_wrapper(cfn.list_stack_resources)
 765     cfn.delete_stack = backoff_wrapper(cfn.delete_stack)
 766     if boto_supports_termination_protection(cfn):
 767         cfn.update_termination_protection = backoff_wrapper(cfn.update_termination_protection)
 768 
 769     stack_info = get_stack_facts(cfn, stack_params['StackName'])
 770 
 771     if module.check_mode:
 772         if state == 'absent' and stack_info:
 773             module.exit_json(changed=True, msg='Stack would be deleted', meta=[])
 774         elif state == 'absent' and not stack_info:
 775             module.exit_json(changed=False, msg='Stack doesn\'t exist', meta=[])
 776         elif state == 'present' and not stack_info:
 777             module.exit_json(changed=True, msg='New stack would be created', meta=[])
 778         else:
 779             module.exit_json(**check_mode_changeset(module, stack_params, cfn))
 780 
 781     if state == 'present':
 782         if not stack_info:
 783             result = create_stack(module, stack_params, cfn, module.params.get('events_limit'))
 784         elif module.params.get('create_changeset'):
 785             result = create_changeset(module, stack_params, cfn, module.params.get('events_limit'))
 786         else:
 787             if module.params.get('termination_protection') is not None:
 788                 update_termination_protection(module, cfn, stack_params['StackName'],
 789                                               bool(module.params.get('termination_protection')))
 790             result = update_stack(module, stack_params, cfn, module.params.get('events_limit'))
 791 
 792         # format the stack output
 793 
 794         stack = get_stack_facts(cfn, stack_params['StackName'])
 795         if stack is not None:
 796             if result.get('stack_outputs') is None:
 797                 # always define stack_outputs, but it may be empty
 798                 result['stack_outputs'] = {}
 799             for output in stack.get('Outputs', []):
 800                 result['stack_outputs'][output['OutputKey']] = output['OutputValue']
 801             stack_resources = []
 802             reslist = cfn.list_stack_resources(StackName=stack_params['StackName'])
 803             for res in reslist.get('StackResourceSummaries', []):
 804                 stack_resources.append({
 805                     "logical_resource_id": res['LogicalResourceId'],
 806                     "physical_resource_id": res.get('PhysicalResourceId', ''),
 807                     "resource_type": res['ResourceType'],
 808                     "last_updated_time": res['LastUpdatedTimestamp'],
 809                     "status": res['ResourceStatus'],
 810                     "status_reason": res.get('ResourceStatusReason')  # can be blank, apparently
 811                 })
 812             result['stack_resources'] = stack_resources
 813 
 814     elif state == 'absent':
 815         # absent state is different because of the way delete_stack works.
 816         # problem is it it doesn't give an error if stack isn't found
 817         # so must describe the stack first
 818 
 819         try:
 820             stack = get_stack_facts(cfn, stack_params['StackName'])
 821             if not stack:
 822                 result = {'changed': False, 'output': 'Stack not found.'}
 823             else:
 824                 if stack_params.get('RoleARN') is None:
 825                     cfn.delete_stack(StackName=stack_params['StackName'])
 826                 else:
 827                     cfn.delete_stack(StackName=stack_params['StackName'], RoleARN=stack_params['RoleARN'])
 828                 result = stack_operation(cfn, stack_params['StackName'], 'DELETE', module.params.get('events_limit'),
 829                                          stack_params.get('ClientRequestToken', None))
 830         except Exception as err:
 831             module.fail_json(msg=boto_exception(err), exception=traceback.format_exc())
 832 
 833     module.exit_json(**result)
