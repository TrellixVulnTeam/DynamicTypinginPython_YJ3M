 183 def create_or_update_bucket(s3_client, module, location):
 184 
 185     policy = module.params.get("policy")
 186     name = module.params.get("name")
 187     requester_pays = module.params.get("requester_pays")
 188     tags = module.params.get("tags")
 189     purge_tags = module.params.get("purge_tags")
 190     versioning = module.params.get("versioning")
 191     encryption = module.params.get("encryption")
 192     encryption_key_id = module.params.get("encryption_key_id")
 193     changed = False
 194     result = {}
 195 
 196     try:
 197         bucket_is_present = bucket_exists(s3_client, name)
 198     except EndpointConnectionError as e:
 199         module.fail_json_aws(e, msg="Invalid endpoint provided: %s" % to_text(e))
 200     except (BotoCoreError, ClientError) as e:
 201         module.fail_json_aws(e, msg="Failed to check bucket presence")
 202 
 203     if not bucket_is_present:
 204         try:
 205             bucket_changed = create_bucket(s3_client, name, location)
 206             s3_client.get_waiter('bucket_exists').wait(Bucket=name)
 207             changed = changed or bucket_changed
 208         except WaiterError as e:
 209             module.fail_json_aws(e, msg='An error occurred waiting for the bucket to become available')
 210         except (BotoCoreError, ClientError) as e:
 211             module.fail_json_aws(e, msg="Failed while creating bucket")
 212 
 213     # Versioning
 214     try:
 215         versioning_status = get_bucket_versioning(s3_client, name)
 216     except BotoCoreError as exp:
 217         module.fail_json_aws(exp, msg="Failed to get bucket versioning")
 218     except ClientError as exp:
 219         if exp.response['Error']['Code'] != 'NotImplemented' or versioning is not None:
 220             module.fail_json_aws(exp, msg="Failed to get bucket versioning")
 221     else:
 222         if versioning is not None:
 223             required_versioning = None
 224             if versioning and versioning_status.get('Status') != "Enabled":
 225                 required_versioning = 'Enabled'
 226             elif not versioning and versioning_status.get('Status') == "Enabled":
 227                 required_versioning = 'Suspended'
 228 
 229             if required_versioning:
 230                 try:
 231                     put_bucket_versioning(s3_client, name, required_versioning)
 232                     changed = True
 233                 except (BotoCoreError, ClientError) as e:
 234                     module.fail_json_aws(e, msg="Failed to update bucket versioning")
 235 
 236                 versioning_status = wait_versioning_is_applied(module, s3_client, name, required_versioning)
 237 
 238         # This output format is there to ensure compatibility with previous versions of the module
 239         result['versioning'] = {
 240             'Versioning': versioning_status.get('Status', 'Disabled'),
 241             'MfaDelete': versioning_status.get('MFADelete', 'Disabled'),
 242         }
 243 
 244     # Requester pays
 245     try:
 246         requester_pays_status = get_bucket_request_payment(s3_client, name)
 247     except BotoCoreError as exp:
 248         module.fail_json_aws(exp, msg="Failed to get bucket request payment")
 249     except ClientError as exp:
 250         if exp.response['Error']['Code'] not in ('NotImplemented', 'XNotImplemented') or requester_pays:
 251             module.fail_json_aws(exp, msg="Failed to get bucket request payment")
 252     else:
 253         if requester_pays:
 254             payer = 'Requester' if requester_pays else 'BucketOwner'
 255             if requester_pays_status != payer:
 256                 put_bucket_request_payment(s3_client, name, payer)
 257                 requester_pays_status = wait_payer_is_applied(module, s3_client, name, payer, should_fail=False)
 258                 if requester_pays_status is None:
 259                     # We have seen that it happens quite a lot of times that the put request was not taken into
 260                     # account, so we retry one more time
 261                     put_bucket_request_payment(s3_client, name, payer)
 262                     requester_pays_status = wait_payer_is_applied(module, s3_client, name, payer, should_fail=True)
 263                 changed = True
 264 
 265         result['requester_pays'] = requester_pays
 266 
 267     # Policy
 268     try:
 269         current_policy = get_bucket_policy(s3_client, name)
 270     except BotoCoreError as exp:
 271         module.fail_json_aws(exp, msg="Failed to get bucket policy")
 272     except ClientError as exp:
 273         if exp.response['Error']['Code'] != 'NotImplemented' or policy is not None:
 274             module.fail_json_aws(exp, msg="Failed to get bucket policy")
 275     else:
 276         if policy is not None:
 277             if isinstance(policy, string_types):
 278                 policy = json.loads(policy)
 279 
 280             if not policy and current_policy:
 281                 try:
 282                     delete_bucket_policy(s3_client, name)
 283                 except (BotoCoreError, ClientError) as e:
 284                     module.fail_json_aws(e, msg="Failed to delete bucket policy")
 285                 current_policy = wait_policy_is_applied(module, s3_client, name, policy)
 286                 changed = True
 287             elif compare_policies(current_policy, policy):
 288                 try:
 289                     put_bucket_policy(s3_client, name, policy)
 290                 except (BotoCoreError, ClientError) as e:
 291                     module.fail_json_aws(e, msg="Failed to update bucket policy")
 292                 current_policy = wait_policy_is_applied(module, s3_client, name, policy, should_fail=False)
 293                 if current_policy is None:
 294                     # As for request payement, it happens quite a lot of times that the put request was not taken into
 295                     # account, so we retry one more time
 296                     put_bucket_policy(s3_client, name, policy)
 297                     current_policy = wait_policy_is_applied(module, s3_client, name, policy, should_fail=True)
 298                 changed = True
 299 
 300         result['policy'] = current_policy
 301 
 302     # Tags
 303     try:
 304         current_tags_dict = get_current_bucket_tags_dict(s3_client, name)
 305     except BotoCoreError as exp:
 306         module.fail_json_aws(exp, msg="Failed to get bucket tags")
 307     except ClientError as exp:
 308         if exp.response['Error']['Code'] not in ('NotImplemented', 'XNotImplemented') or tags is not None:
 309             module.fail_json_aws(exp, msg="Failed to get bucket tags")
 310     else:
 311         if tags is not None:
 312             # Tags are always returned as text
 313             tags = dict((to_text(k), to_text(v)) for k, v in tags.items())
 314             if not purge_tags:
 315                 # Ensure existing tags that aren't updated by desired tags remain
 316                 current_copy = current_tags_dict.copy()
 317                 current_copy.update(tags)
 318                 tags = current_copy
 319             if current_tags_dict != tags:
 320                 if tags:
 321                     try:
 322                         put_bucket_tagging(s3_client, name, tags)
 323                     except (BotoCoreError, ClientError) as e:
 324                         module.fail_json_aws(e, msg="Failed to update bucket tags")
 325                 else:
 326                     if purge_tags:
 327                         try:
 328                             delete_bucket_tagging(s3_client, name)
 329                         except (BotoCoreError, ClientError) as e:
 330                             module.fail_json_aws(e, msg="Failed to delete bucket tags")
 331                 current_tags_dict = wait_tags_are_applied(module, s3_client, name, tags)
 332                 changed = True
 333 
 334         result['tags'] = current_tags_dict
 335 
 336     # Encryption
 337     if hasattr(s3_client, "get_bucket_encryption"):
 338         try:
 339             current_encryption = get_bucket_encryption(s3_client, name)
 340         except (ClientError, BotoCoreError) as e:
 341             module.fail_json_aws(e, msg="Failed to get bucket encryption")
 342     elif encryption is not None:
 343         module.fail_json(msg="Using bucket encryption requires botocore version >= 1.7.41")
 344 
 345     if encryption is not None:
 346         current_encryption_algorithm = current_encryption.get('SSEAlgorithm') if current_encryption else None
 347         current_encryption_key = current_encryption.get('KMSMasterKeyID') if current_encryption else None
 348         if encryption == 'none' and current_encryption_algorithm is not None:
 349             try:
 350                 delete_bucket_encryption(s3_client, name)
 351             except (BotoCoreError, ClientError) as e:
 352                 module.fail_json_aws(e, msg="Failed to delete bucket encryption")
 353             current_encryption = wait_encryption_is_applied(module, s3_client, name, None)
 354             changed = True
 355         elif encryption != 'none' and (encryption != current_encryption_algorithm) or (encryption == 'aws:kms' and current_encryption_key != encryption_key_id):
 356             expected_encryption = {'SSEAlgorithm': encryption}
 357             if encryption == 'aws:kms' and encryption_key_id is not None:
 358                 expected_encryption.update({'KMSMasterKeyID': encryption_key_id})
 359             try:
 360                 put_bucket_encryption(s3_client, name, expected_encryption)
 361             except (BotoCoreError, ClientError) as e:
 362                 module.fail_json_aws(e, msg="Failed to set bucket encryption")
 363             current_encryption = wait_encryption_is_applied(module, s3_client, name, expected_encryption)
 364             changed = True
 365 
 366         result['encryption'] = current_encryption
 367 
 368     module.exit_json(changed=changed, name=name, **result)

 371 def bucket_exists(s3_client, bucket_name):
 372     # head_bucket appeared to be really inconsistent, so we use list_buckets instead,
 373     # and loop over all the buckets, even if we know it's less performant :(
 374     all_buckets = s3_client.list_buckets(Bucket=bucket_name)['Buckets']
 375     return any(bucket['Name'] == bucket_name for bucket in all_buckets)

 378 @AWSRetry.exponential_backoff(max_delay=120)
 379 def create_bucket(s3_client, bucket_name, location):
 380     try:
 381         configuration = {}
 382         if location not in ('us-east-1', None):
 383             configuration['LocationConstraint'] = location
 384         if len(configuration) > 0:
 385             s3_client.create_bucket(Bucket=bucket_name, CreateBucketConfiguration=configuration)
 386         else:
 387             s3_client.create_bucket(Bucket=bucket_name)
 388         return True
 389     except ClientError as e:
 390         error_code = e.response['Error']['Code']
 391         if error_code == 'BucketAlreadyOwnedByYou':
 392             # We should never get there since we check the bucket presence before calling the create_or_update_bucket
 393             # method. However, the AWS Api sometimes fails to report bucket presence, so we catch this exception
 394             return False
 395         else:
 396             raise e

 489 def wait_policy_is_applied(module, s3_client, bucket_name, expected_policy, should_fail=True):
 490     for dummy in range(0, 12):
 491         try:
 492             current_policy = get_bucket_policy(s3_client, bucket_name)
 493         except (ClientError, BotoCoreError) as e:
 494             module.fail_json_aws(e, msg="Failed to get bucket policy")
 495 
 496         if compare_policies(current_policy, expected_policy):
 497             time.sleep(5)
 498         else:
 499             return current_policy
 500     if should_fail:
 501         module.fail_json(msg="Bucket policy failed to apply in the expected time")
 502     else:
 503         return None
 504 
 505 
 506 def wait_payer_is_applied(module, s3_client, bucket_name, expected_payer, should_fail=True):
 507     for dummy in range(0, 12):
 508         try:
 509             requester_pays_status = get_bucket_request_payment(s3_client, bucket_name)
 510         except (BotoCoreError, ClientError) as e:
 511             module.fail_json_aws(e, msg="Failed to get bucket request payment")
 512         if requester_pays_status != expected_payer:
 513             time.sleep(5)
 514         else:
 515             return requester_pays_status
 516     if should_fail:
 517         module.fail_json(msg="Bucket request payment failed to apply in the expected time")
 518     else:
 519         return None

 522 def wait_encryption_is_applied(module, s3_client, bucket_name, expected_encryption):
 523     for dummy in range(0, 12):
 524         try:
 525             encryption = get_bucket_encryption(s3_client, bucket_name)
 526         except (BotoCoreError, ClientError) as e:
 527             module.fail_json_aws(e, msg="Failed to get updated encryption for bucket")
 528         if encryption != expected_encryption:
 529             time.sleep(5)
 530         else:
 531             return encryption
 532     module.fail_json(msg="Bucket encryption failed to apply in the expected time")
 533 
 534 
 535 def wait_versioning_is_applied(module, s3_client, bucket_name, required_versioning):
 536     for dummy in range(0, 24):
 537         try:
 538             versioning_status = get_bucket_versioning(s3_client, bucket_name)
 539         except (BotoCoreError, ClientError) as e:
 540             module.fail_json_aws(e, msg="Failed to get updated versioning for bucket")
 541         if versioning_status.get('Status') != required_versioning:
 542             time.sleep(8)
 543         else:
 544             return versioning_status
 545     module.fail_json(msg="Bucket versioning failed to apply in the expected time")

 555             time.sleep(5)
 556         else:
 557             return current_tags_dict
 558     module.fail_json(msg="Bucket tags failed to apply in the expected time")

 572 def paginated_list(s3_client, **pagination_params):
 573     pg = s3_client.get_paginator('list_objects_v2')
 574     for page in pg.paginate(**pagination_params):
 575         yield [data['Key'] for data in page.get('Contents', [])]

 578 def paginated_versions_list(s3_client, **pagination_params):
 579     try:
 580         pg = s3_client.get_paginator('list_object_versions')
 581         for page in pg.paginate(**pagination_params):
 582             # We have to merge the Versions and DeleteMarker lists here, as DeleteMarkers can still prevent a bucket deletion
 583             yield [(data['Key'], data['VersionId']) for data in (page.get('Versions', []) + page.get('DeleteMarkers', []))]
 584     except is_boto3_error_code('NoSuchBucket'):
 585         yield []

 588 def destroy_bucket(s3_client, module):
 589 
 590     force = module.params.get("force")
 591     name = module.params.get("name")
 592     try:
 593         bucket_is_present = bucket_exists(s3_client, name)
 594     except EndpointConnectionError as e:
 595         module.fail_json_aws(e, msg="Invalid endpoint provided: %s" % to_text(e))
 596     except (BotoCoreError, ClientError) as e:
 597         module.fail_json_aws(e, msg="Failed to check bucket presence")
 598 
 599     if not bucket_is_present:
 600         module.exit_json(changed=False)
 601 
 602     if force:
 603         # if there are contents then we need to delete them (including versions) before we can delete the bucket
 604         try:
 605             for key_version_pairs in paginated_versions_list(s3_client, Bucket=name):
 606                 formatted_keys = [{'Key': key, 'VersionId': version} for key, version in key_version_pairs]
 607                 for fk in formatted_keys:
 608                     # remove VersionId from cases where they are `None` so that
 609                     # unversioned objects are deleted using `DeleteObject`
 610                     # rather than `DeleteObjectVersion`, improving backwards
 611                     # compatibility with older IAM policies.
 612                     if not fk.get('VersionId'):
 613                         fk.pop('VersionId')
 614 
 615                 if formatted_keys:
 616                     resp = s3_client.delete_objects(Bucket=name, Delete={'Objects': formatted_keys})
 617                     if resp.get('Errors'):
 618                         module.fail_json(
 619                             msg='Could not empty bucket before deleting. Could not delete objects: {0}'.format(
 620                                 ', '.join([k['Key'] for k in resp['Errors']])
 621                             ),
 622                             errors=resp['Errors'], response=resp
 623                         )
 624         except (BotoCoreError, ClientError) as e:
 625             module.fail_json_aws(e, msg="Failed while deleting bucket")
 626 
 627     try:
 628         delete_bucket(s3_client, name)
 629         s3_client.get_waiter('bucket_not_exists').wait(Bucket=name, WaiterConfig=dict(Delay=5, MaxAttempts=60))
 630     except WaiterError as e:
 631         module.fail_json_aws(e, msg='An error occurred waiting for the bucket to be deleted.')
 632     except (BotoCoreError, ClientError) as e:
 633         module.fail_json_aws(e, msg="Failed to delete bucket")
 634 
 635     module.exit_json(changed=True)

 638 def is_fakes3(s3_url):
 639     """ Return True if s3_url has scheme fakes3:// """
 640     if s3_url is not None:
 641         return urlparse(s3_url).scheme in ('fakes3', 'fakes3s')
 642     else:
 643         return False

 646 def get_s3_client(module, aws_connect_kwargs, location, ceph, s3_url):
 647     if s3_url and ceph:  # TODO - test this
 648         ceph = urlparse(s3_url)
 649         params = dict(module=module, conn_type='client', resource='s3', use_ssl=ceph.scheme == 'https', region=location, endpoint=s3_url, **aws_connect_kwargs)
 650     elif is_fakes3(s3_url):
 651         fakes3 = urlparse(s3_url)
 652         port = fakes3.port
 653         if fakes3.scheme == 'fakes3s':
 654             protocol = "https"
 655             if port is None:
 656                 port = 443
 657         else:
 658             protocol = "http"
 659             if port is None:
 660                 port = 80
 661         params = dict(module=module, conn_type='client', resource='s3', region=location,
 662                       endpoint="%s://%s:%s" % (protocol, fakes3.hostname, to_text(port)),
 663                       use_ssl=fakes3.scheme == 'fakes3s', **aws_connect_kwargs)
 664     else:
 665         params = dict(module=module, conn_type='client', resource='s3', region=location, endpoint=s3_url, **aws_connect_kwargs)
 666     return boto3_conn(**params)

 669 def main():
 670 
 671     argument_spec = ec2_argument_spec()
 672     argument_spec.update(
 673         dict(
 674             force=dict(default=False, type='bool'),
 675             policy=dict(type='json'),
 676             name=dict(required=True),
 677             requester_pays=dict(default=False, type='bool'),
 678             s3_url=dict(aliases=['S3_URL']),
 679             state=dict(default='present', choices=['present', 'absent']),
 680             tags=dict(type='dict'),
 681             purge_tags=dict(type='bool', default=True),
 682             versioning=dict(type='bool'),
 683             ceph=dict(default=False, type='bool'),
 684             encryption=dict(choices=['none', 'AES256', 'aws:kms']),
 685             encryption_key_id=dict()
 686         )
 687     )
 688 
 689     module = AnsibleAWSModule(
 690         argument_spec=argument_spec,
 691     )
 692 
 693     region, ec2_url, aws_connect_kwargs = get_aws_connection_info(module, boto3=True)
 694 
 695     if region in ('us-east-1', '', None):
 696         # default to US Standard region
 697         location = 'us-east-1'
 698     else:
 699         # Boto uses symbolic names for locations but region strings will
 700         # actually work fine for everything except us-east-1 (US Standard)
 701         location = region
 702 
 703     s3_url = module.params.get('s3_url')
 704     ceph = module.params.get('ceph')
 705 
 706     # allow eucarc environment variables to be used if ansible vars aren't set
 707     if not s3_url and 'S3_URL' in os.environ:
 708         s3_url = os.environ['S3_URL']
 709 
 710     if ceph and not s3_url:
 711         module.fail_json(msg='ceph flavour requires s3_url')
 712 
 713     # Look at s3_url and tweak connection settings
 714     # if connecting to Ceph RGW, Walrus or fakes3
 715     if s3_url:
 716         for key in ['validate_certs', 'security_token', 'profile_name']:
 717             aws_connect_kwargs.pop(key, None)
 718     s3_client = get_s3_client(module, aws_connect_kwargs, location, ceph, s3_url)
 719 
 720     if s3_client is None:  # this should never happen
 721         module.fail_json(msg='Unknown error, failed to create s3 connection, no information from boto.')
 722 
 723     state = module.params.get("state")
 724     encryption = module.params.get("encryption")
 725     encryption_key_id = module.params.get("encryption_key_id")
 726 
 727     # Parameter validation
 728     if encryption_key_id is not None and encryption is None:
 729         module.fail_json(msg="You must specify encryption parameter along with encryption_key_id.")
 730     elif encryption_key_id is not None and encryption != 'aws:kms':
 731         module.fail_json(msg="Only 'aws:kms' is a valid option for encryption parameter when you specify encryption_key_id.")
 732 
 733     if state == 'present':
 734         create_or_update_bucket(s3_client, module, location)
 735     elif state == 'absent':
 736         destroy_bucket(s3_client, module)
 737 
 738 
 739 if __name__ == '__main__':
 740     main()
