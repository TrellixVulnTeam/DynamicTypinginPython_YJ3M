 299 def compile_ipv4_regexp():
 300     r = r"((25[0-5]|(2[0-4]|1{0,1}[0-9]){0,1}[0-9])\.){3,3}"
 301     r += r"(25[0-5]|(2[0-4]|1{0,1}[0-9]){0,1}[0-9])"
 302     return re.compile(r)

 305 def compile_ipv6_regexp():
 306     """
 307     validation pattern provided by :
 308     https://stackoverflow.com/questions/53497/regular-expression-that-matches-
 309     valid-ipv6-addresses#answer-17871737
 310     """
 311     r = r"(([0-9a-fA-F]{1,4}:){7,7}[0-9a-fA-F]{1,4}|([0-9a-fA-F]{1,4}:){1,7}:"
 312     r += r"|([0-9a-fA-F]{1,4}:){1,6}:[0-9a-fA-F]{1,4}|([0-9a-fA-F]{1,4}:){1,5}"
 313     r += r"(:[0-9a-fA-F]{1,4}){1,2}|([0-9a-fA-F]{1,4}:){1,4}(:[0-9a-fA-F]{1,4})"
 314     r += r"{1,3}|([0-9a-fA-F]{1,4}:){1,3}(:[0-9a-fA-F]{1,4}){1,4}|([0-9a-fA-F]"
 315     r += r"{1,4}:){1,2}(:[0-9a-fA-F]{1,4}){1,5}|[0-9a-fA-F]{1,4}:((:[0-9a-fA-F]"
 316     r += r"{1,4}){1,6})|:((:[0-9a-fA-F]{1,4}){1,7}|:)|fe80:(:[0-9a-fA-F]{0,4})"
 317     r += r"{0,4}%[0-9a-zA-Z]{1,}|::(ffff(:0{1,4}){0,1}:){0,1}((25[0-5]|(2[0-4]"
 318     r += r"|1{0,1}[0-9]){0,1}[0-9])\.){3,3}(25[0-5]|(2[0-4]|1{0,1}[0-9]){0,1}"
 319     r += r"[0-9])|([0-9a-fA-F]{1,4}:){1,4}:((25[0-5]|(2[0-4]|1{0,1}[0-9]){0,1}"
 320     r += r"[0-9])\.){3,3}(25[0-5]|(2[0-4]|1{0,1}[0-9]){0,1}[0-9]))"
 321     return re.compile(r)

 324 def main():
 325     command_keys = ['state', 'default', 'rule', 'logging']
 326 
 327     module = AnsibleModule(
 328         argument_spec=dict(
 329             state=dict(type='str', choices=['enabled', 'disabled', 'reloaded', 'reset']),
 330             default=dict(type='str', aliases=['policy'], choices=['allow', 'deny', 'reject']),
 331             logging=dict(type='str', choices=['full', 'high', 'low', 'medium', 'off', 'on']),
 332             direction=dict(type='str', choices=['in', 'incoming', 'out', 'outgoing', 'routed']),
 333             delete=dict(type='bool', default=False),
 334             route=dict(type='bool', default=False),
 335             insert=dict(type='int'),
 336             insert_relative_to=dict(choices=['zero', 'first-ipv4', 'last-ipv4', 'first-ipv6', 'last-ipv6'], default='zero'),
 337             rule=dict(type='str', choices=['allow', 'deny', 'limit', 'reject']),
 338             interface=dict(type='str', aliases=['if']),
 339             interface_in=dict(type='str', aliases=['if_in']),
 340             interface_out=dict(type='str', aliases=['if_out']),
 341             log=dict(type='bool', default=False),
 342             from_ip=dict(type='str', default='any', aliases=['from', 'src']),
 343             from_port=dict(type='str'),
 344             to_ip=dict(type='str', default='any', aliases=['dest', 'to']),
 345             to_port=dict(type='str', aliases=['port']),
 346             proto=dict(type='str', aliases=['protocol'], choices=['ah', 'any', 'esp', 'ipv6', 'tcp', 'udp', 'gre', 'igmp']),
 347             name=dict(type='str', aliases=['app']),
 348             comment=dict(type='str'),
 349         ),
 350         supports_check_mode=True,
 351         mutually_exclusive=[
 352             ['name', 'proto', 'logging'],
 353             # Mutual exclusivity with `interface` implied by `required_by`.
 354             ['direction', 'interface_in'],
 355             ['direction', 'interface_out'],
 356         ],
 357         required_one_of=([command_keys]),
 358         required_by=dict(
 359             interface=('direction', ),
 360         ),
 361     )
 362 
 363     cmds = []
 364 
 365     ipv4_regexp = compile_ipv4_regexp()
 366     ipv6_regexp = compile_ipv6_regexp()
 367 
 368     def filter_line_that_not_start_with(pattern, content):
 369         return ''.join([line for line in content.splitlines(True) if line.startswith(pattern)])
 370 
 371     def filter_line_that_contains(pattern, content):
 372         return [line for line in content.splitlines(True) if pattern in line]
 373 
 374     def filter_line_that_not_contains(pattern, content):
 375         return ''.join([line for line in content.splitlines(True) if not line.contains(pattern)])
 376 
 377     def filter_line_that_match_func(match_func, content):
 378         return ''.join([line for line in content.splitlines(True) if match_func(line) is not None])
 379 
 380     def filter_line_that_contains_ipv4(content):
 381         return filter_line_that_match_func(ipv4_regexp.search, content)
 382 
 383     def filter_line_that_contains_ipv6(content):
 384         return filter_line_that_match_func(ipv6_regexp.search, content)
 385 
 386     def is_starting_by_ipv4(ip):
 387         return ipv4_regexp.match(ip) is not None
 388 
 389     def is_starting_by_ipv6(ip):
 390         return ipv6_regexp.match(ip) is not None

 392     def execute(cmd, ignore_error=False):
 393         cmd = ' '.join(map(itemgetter(-1), filter(itemgetter(0), cmd)))
 394 
 395         cmds.append(cmd)
 396         (rc, out, err) = module.run_command(cmd, environ_update={"LANG": "C"})
 397 
 398         if rc != 0 and not ignore_error:
 399             module.fail_json(msg=err or out, commands=cmds)
 400 
 401         return out
 402 
 403     def get_current_rules():
 404         user_rules_files = ["/lib/ufw/user.rules",
 405                             "/lib/ufw/user6.rules",
 406                             "/etc/ufw/user.rules",
 407                             "/etc/ufw/user6.rules",
 408                             "/var/lib/ufw/user.rules",
 409                             "/var/lib/ufw/user6.rules"]
 410 
 411         cmd = [[grep_bin], ["-h"], ["'^### tuple'"]]
 412 
 413         cmd.extend([[f] for f in user_rules_files])
 414         return execute(cmd, ignore_error=True)
 415 
 416     def ufw_version():
 417         """
 418         Returns the major and minor version of ufw installed on the system.
 419         """
 420         out = execute([[ufw_bin], ["--version"]])
 421 
 422         lines = [x for x in out.split('\n') if x.strip() != '']
 423         if len(lines) == 0:
 424             module.fail_json(msg="Failed to get ufw version.", rc=0, out=out)
 425 
 426         matches = re.search(r'^ufw.+(\d+)\.(\d+)(?:\.(\d+))?.*$', lines[0])
 427         if matches is None:
 428             module.fail_json(msg="Failed to get ufw version.", rc=0, out=out)
 429 
 430         # Convert version to numbers
 431         major = int(matches.group(1))
 432         minor = int(matches.group(2))
 433         rev = 0
 434         if matches.group(3) is not None:
 435             rev = int(matches.group(3))
 436 
 437         return major, minor, rev
 438 
 439     params = module.params
 440 
 441     commands = dict((key, params[key]) for key in command_keys if params[key])
 442 
 443     # Ensure ufw is available
 444     ufw_bin = module.get_bin_path('ufw', True)
 445     grep_bin = module.get_bin_path('grep', True)
 446 
 447     # Save the pre state and rules in order to recognize changes
 448     pre_state = execute([[ufw_bin], ['status verbose']])
 449     pre_rules = get_current_rules()
 450 
 451     changed = False
 452 
 453     # Execute filter
 454     for (command, value) in commands.items():
 455 
 456         cmd = [[ufw_bin], [module.check_mode, '--dry-run']]
 457 
 458         if command == 'state':
 459             states = {'enabled': 'enable', 'disabled': 'disable',
 460                       'reloaded': 'reload', 'reset': 'reset'}
 461 
 462             if value in ['reloaded', 'reset']:
 463                 changed = True
 464 
 465             if module.check_mode:
 466                 # "active" would also match "inactive", hence the space
 467                 ufw_enabled = pre_state.find(" active") != -1
 468                 if (value == 'disabled' and ufw_enabled) or (value == 'enabled' and not ufw_enabled):
 469                     changed = True
 470             else:
 471                 execute(cmd + [['-f'], [states[value]]])
 472 
 473         elif command == 'logging':
 474             extract = re.search(r'Logging: (on|off)(?: \(([a-z]+)\))?', pre_state)
 475             if extract:
 476                 current_level = extract.group(2)
 477                 current_on_off_value = extract.group(1)
 478                 if value != "off":
 479                     if current_on_off_value == "off":
 480                         changed = True
 481                     elif value != "on" and value != current_level:
 482                         changed = True
 483                 elif current_on_off_value != "off":
 484                     changed = True
 485             else:
 486                 changed = True
 487 
 488             if not module.check_mode:
 489                 execute(cmd + [[command], [value]])
 490 
 491         elif command == 'default':
 492             if params['direction'] not in ['outgoing', 'incoming', 'routed', None]:
 493                 module.fail_json(msg='For default, direction must be one of "outgoing", "incoming" and "routed", or direction must not be specified.')
 494             if module.check_mode:
 495                 regexp = r'Default: (deny|allow|reject) \(incoming\), (deny|allow|reject) \(outgoing\), (deny|allow|reject|disabled) \(routed\)'
 496                 extract = re.search(regexp, pre_state)
 497                 if extract is not None:
 498                     current_default_values = {}
 499                     current_default_values["incoming"] = extract.group(1)
 500                     current_default_values["outgoing"] = extract.group(2)
 501                     current_default_values["routed"] = extract.group(3)
 502                     v = current_default_values[params['direction'] or 'incoming']
 503                     if v not in (value, 'disabled'):
 504                         changed = True
 505                 else:
 506                     changed = True
 507             else:
 508                 execute(cmd + [[command], [value], [params['direction']]])
 509 
 510         elif command == 'rule':
 511             if params['direction'] not in ['in', 'out', None]:
 512                 module.fail_json(msg='For rules, direction must be one of "in" and "out", or direction must not be specified.')
 513             if not params['route'] and params['interface_in'] and params['interface_out']:
 514                 module.fail_json(msg='Only route rules can combine '
 515                                  'interface_in and interface_out')
 516             # Rules are constructed according to the long format
 517             #
 518             # ufw [--dry-run] [route] [delete] [insert NUM] allow|deny|reject|limit [in|out on INTERFACE] [log|log-all] \
 519             #     [from ADDRESS [port PORT]] [to ADDRESS [port PORT]] \
 520             #     [proto protocol] [app application] [comment COMMENT]
 521             cmd.append([module.boolean(params['route']), 'route'])
 522             cmd.append([module.boolean(params['delete']), 'delete'])
 523             if params['insert'] is not None:
 524                 relative_to_cmd = params['insert_relative_to']
 525                 if relative_to_cmd == 'zero':
 526                     insert_to = params['insert']
 527                 else:
 528                     (dummy, numbered_state, dummy) = module.run_command([ufw_bin, 'status', 'numbered'])
 529                     numbered_line_re = re.compile(R'^\[ *([0-9]+)\] ')
 530                     lines = [(numbered_line_re.match(line), '(v6)' in line) for line in numbered_state.splitlines()]
 531                     lines = [(int(matcher.group(1)), ipv6) for (matcher, ipv6) in lines if matcher]
 532                     last_number = max([no for (no, ipv6) in lines]) if lines else 0
 533                     has_ipv4 = any([not ipv6 for (no, ipv6) in lines])
 534                     has_ipv6 = any([ipv6 for (no, ipv6) in lines])
 535                     if relative_to_cmd == 'first-ipv4':
 536                         relative_to = 1
 537                     elif relative_to_cmd == 'last-ipv4':
 538                         relative_to = max([no for (no, ipv6) in lines if not ipv6]) if has_ipv4 else 1
 539                     elif relative_to_cmd == 'first-ipv6':
 540                         relative_to = max([no for (no, ipv6) in lines if not ipv6]) + 1 if has_ipv4 else 1
 541                     elif relative_to_cmd == 'last-ipv6':
 542                         relative_to = last_number if has_ipv6 else last_number + 1
 543                     insert_to = params['insert'] + relative_to
 544                     if insert_to > last_number:
 545                         # ufw does not like it when the insert number is larger than the
 546                         # maximal rule number for IPv4/IPv6.
 547                         insert_to = None
 548                 cmd.append([insert_to is not None, "insert %s" % insert_to])
 549             cmd.append([value])
 550             cmd.append([params['direction'], "%s" % params['direction']])
 551             cmd.append([params['interface'], "on %s" % params['interface']])
 552             cmd.append([params['interface_in'], "in on %s" % params['interface_in']])
 553             cmd.append([params['interface_out'], "out on %s" % params['interface_out']])
 554             cmd.append([module.boolean(params['log']), 'log'])
 555 
 556             for (key, template) in [('from_ip', "from %s"), ('from_port', "port %s"),
 557                                     ('to_ip', "to %s"), ('to_port', "port %s"),
 558                                     ('proto', "proto %s"), ('name', "app '%s'")]:
 559                 value = params[key]
 560                 cmd.append([value, template % (value)])
 561 
 562             ufw_major, ufw_minor, dummy = ufw_version()
 563             # comment is supported only in ufw version after 0.35
 564             if (ufw_major == 0 and ufw_minor >= 35) or ufw_major > 0:
 565                 cmd.append([params['comment'], "comment '%s'" % params['comment']])
 566 
 567             rules_dry = execute(cmd)
 568 
 569             if module.check_mode:
 570 
 571                 nb_skipping_line = len(filter_line_that_contains("Skipping", rules_dry))
 572 
 573                 if not (nb_skipping_line > 0 and nb_skipping_line == len(rules_dry.splitlines(True))):
 574 
 575                     rules_dry = filter_line_that_not_start_with("### tuple", rules_dry)
 576                     # ufw dry-run doesn't send all rules so have to compare ipv4 or ipv6 rules
 577                     if is_starting_by_ipv4(params['from_ip']) or is_starting_by_ipv4(params['to_ip']):
 578                         if filter_line_that_contains_ipv4(pre_rules) != filter_line_that_contains_ipv4(rules_dry):
 579                             changed = True
 580                     elif is_starting_by_ipv6(params['from_ip']) or is_starting_by_ipv6(params['to_ip']):
 581                         if filter_line_that_contains_ipv6(pre_rules) != filter_line_that_contains_ipv6(rules_dry):
 582                             changed = True
 583                     elif pre_rules != rules_dry:
 584                         changed = True
 585 
 586     # Get the new state
 587     if module.check_mode:
 588         return module.exit_json(changed=changed, commands=cmds)
 589     else:
 590         post_state = execute([[ufw_bin], ['status'], ['verbose']])
 591         if not changed:
 592             post_rules = get_current_rules()
 593             changed = (pre_state != post_state) or (pre_rules != post_rules)
 594         return module.exit_json(changed=changed, commands=cmds, msg=post_state.rstrip())
