  93 import json
  94 import traceback
  95 
  96 REQUESTS_IMP_ERR = None
  97 try:
  98     import requests
  99     HAS_REQUESTS = True
 100 except ImportError:
 101     REQUESTS_IMP_ERR = traceback.format_exc()
 102     HAS_REQUESTS = False
 103 
 104 from ansible.module_utils.basic import AnsibleModule, missing_required_lib
 105 from ansible.module_utils.six.moves.urllib import parse as urllib_parse
 106 from ansible.module_utils.rabbitmq import rabbitmq_argument_spec
 107 
 108 
 109 def main():
 110 
 111     argument_spec = rabbitmq_argument_spec()
 112     argument_spec.update(
 113         dict(
 114             state=dict(default='present', choices=['present', 'absent'], type='str'),
 115             name=dict(required=True, type='str'),
 116             durable=dict(default=True, type='bool'),
 117             auto_delete=dict(default=False, type='bool'),
 118             message_ttl=dict(default=None, type='int'),
 119             auto_expires=dict(default=None, type='int'),
 120             max_length=dict(default=None, type='int'),
 121             dead_letter_exchange=dict(default=None, type='str'),
 122             dead_letter_routing_key=dict(default=None, type='str'),
 123             arguments=dict(default=dict(), type='dict'),
 124             max_priority=dict(default=None, type='int')
 125         )
 126     )
 127     module = AnsibleModule(argument_spec=argument_spec, supports_check_mode=True)
 128 
 129     url = "%s://%s:%s/api/queues/%s/%s" % (
 130         module.params['login_protocol'],
 131         module.params['login_host'],
 132         module.params['login_port'],
 133         urllib_parse.quote(module.params['vhost'], ''),
 134         module.params['name']
 135     )
 136 
 137     if not HAS_REQUESTS:
 138         module.fail_json(msg=missing_required_lib("requests"), exception=REQUESTS_IMP_ERR)
 139 
 140     result = dict(changed=False, name=module.params['name'])
 141 
 142     # Check if queue already exists
 143     r = requests.get(url, auth=(module.params['login_user'], module.params['login_password']),
 144                      verify=module.params['ca_cert'], cert=(module.params['client_cert'], module.params['client_key']))
 145 
 146     if r.status_code == 200:
 147         queue_exists = True
 148         response = r.json()
 149     elif r.status_code == 404:
 150         queue_exists = False
 151         response = r.text
 152     else:
 153         module.fail_json(
 154             msg="Invalid response from RESTAPI when trying to check if queue exists",
 155             details=r.text
 156         )
 157 
 158     if module.params['state'] == 'present':
 159         change_required = not queue_exists
 160     else:
 161         change_required = queue_exists
 162 
 163     # Check if attributes change on existing queue
 164     if not change_required and r.status_code == 200 and module.params['state'] == 'present':
 165         if not (
 166             response['durable'] == module.params['durable'] and
 167             response['auto_delete'] == module.params['auto_delete'] and
 168             (
 169                 ('x-message-ttl' in response['arguments'] and response['arguments']['x-message-ttl'] == module.params['message_ttl']) or
 170                 ('x-message-ttl' not in response['arguments'] and module.params['message_ttl'] is None)
 171             ) and
 172             (
 173                 ('x-expires' in response['arguments'] and response['arguments']['x-expires'] == module.params['auto_expires']) or
 174                 ('x-expires' not in response['arguments'] and module.params['auto_expires'] is None)
 175             ) and
 176             (
 177                 ('x-max-length' in response['arguments'] and response['arguments']['x-max-length'] == module.params['max_length']) or
 178                 ('x-max-length' not in response['arguments'] and module.params['max_length'] is None)
 179             ) and
 180             (
 181                 ('x-dead-letter-exchange' in response['arguments'] and
 182                  response['arguments']['x-dead-letter-exchange'] == module.params['dead_letter_exchange']) or
 183                 ('x-dead-letter-exchange' not in response['arguments'] and module.params['dead_letter_exchange'] is None)
 184             ) and
 185             (
 186                 ('x-dead-letter-routing-key' in response['arguments'] and
 187                  response['arguments']['x-dead-letter-routing-key'] == module.params['dead_letter_routing_key']) or
 188                 ('x-dead-letter-routing-key' not in response['arguments'] and module.params['dead_letter_routing_key'] is None)
 189             ) and
 190             (
 191                 ('x-max-priority' in response['arguments'] and
 192                  response['arguments']['x-max-priority'] == module.params['max_priority']) or
 193                 ('x-max-priority' not in response['arguments'] and module.params['max_priority'] is None)
 194             )
 195         ):
 196             module.fail_json(
 197                 msg="RabbitMQ RESTAPI doesn't support attribute changes for existing queues",
 198             )
 199 
 200     # Copy parameters to arguments as used by RabbitMQ
 201     for k, v in {
 202         'message_ttl': 'x-message-ttl',
 203         'auto_expires': 'x-expires',
 204         'max_length': 'x-max-length',
 205         'dead_letter_exchange': 'x-dead-letter-exchange',
 206         'dead_letter_routing_key': 'x-dead-letter-routing-key',
 207         'max_priority': 'x-max-priority'
 208     }.items():
 209         if module.params[k] is not None:
 210             module.params['arguments'][v] = module.params[k]
 211 
 212     # Exit if check_mode
 213     if module.check_mode:
 214         result['changed'] = change_required
 215         result['details'] = response
 216         result['arguments'] = module.params['arguments']
 217         module.exit_json(**result)
 218 
 219     # Do changes
 220     if change_required:
 221         if module.params['state'] == 'present':
 222             r = requests.put(
 223                 url,
 224                 auth=(module.params['login_user'], module.params['login_password']),
 225                 headers={"content-type": "application/json"},
 226                 data=json.dumps({
 227                     "durable": module.params['durable'],
 228                     "auto_delete": module.params['auto_delete'],
 229                     "arguments": module.params['arguments']
 230                 }),
 231                 verify=module.params['ca_cert'],
 232                 cert=(module.params['client_cert'], module.params['client_key'])
 233             )
 234         elif module.params['state'] == 'absent':
 235             r = requests.delete(url, auth=(module.params['login_user'], module.params['login_password']),
 236                                 verify=module.params['ca_cert'], cert=(module.params['client_cert'], module.params['client_key']))
 237 
 238         # RabbitMQ 3.6.7 changed this response code from 204 to 201
 239         if r.status_code == 204 or r.status_code == 201:
 240             result['changed'] = True
 241             module.exit_json(**result)
 242         else:
 243             module.fail_json(
 244                 msg="Error creating queue",
 245                 status=r.status_code,
 246                 details=r.text
 247             )
 248 
 249     else:
 250         module.exit_json(
 251             changed=False,
 252             name=module.params['name']
 253         )
 254 
 255 
 256 if __name__ == '__main__':
 257     main()
