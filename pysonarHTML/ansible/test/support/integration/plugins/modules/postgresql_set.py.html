 185 def param_get(cursor, module, name):
 186     query = ("SELECT name, setting, unit, context, boot_val "
 187              "FROM pg_settings WHERE name = %(name)s")
 188     try:
 189         cursor.execute(query, {'name': name})
 190         info = cursor.fetchall()
 191         cursor.execute("SHOW %s" % name)
 192         val = cursor.fetchone()
 193 
 194     except Exception as e:
 195         module.fail_json(msg="Unable to get %s value due to : %s" % (name, to_native(e)))
 196 
 197     raw_val = info[0][1]
 198     unit = info[0][2]
 199     context = info[0][3]
 200     boot_val = info[0][4]
 201 
 202     if val[0] == 'True':
 203         val[0] = 'on'
 204     elif val[0] == 'False':
 205         val[0] = 'off'
 206 
 207     if unit == 'kB':
 208         if int(raw_val) > 0:
 209             raw_val = int(raw_val) * 1024
 210         if int(boot_val) > 0:
 211             boot_val = int(boot_val) * 1024
 212 
 213         unit = 'b'
 214 
 215     elif unit == 'MB':
 216         if int(raw_val) > 0:
 217             raw_val = int(raw_val) * 1024 * 1024
 218         if int(boot_val) > 0:
 219             boot_val = int(boot_val) * 1024 * 1024
 220 
 221         unit = 'b'
 222 
 223     return (val[0], raw_val, unit, boot_val, context)

 226 def pretty_to_bytes(pretty_val):
 227     # The function returns a value in bytes
 228     # if the value contains 'B', 'kB', 'MB', 'GB', 'TB'.
 229     # Otherwise it returns the passed argument.
 230 
 231     val_in_bytes = None
 232 
 233     if 'kB' in pretty_val:
 234         num_part = int(''.join(d for d in pretty_val if d.isdigit()))
 235         val_in_bytes = num_part * 1024
 236 
 237     elif 'MB' in pretty_val.upper():
 238         num_part = int(''.join(d for d in pretty_val if d.isdigit()))
 239         val_in_bytes = num_part * 1024 * 1024
 240 
 241     elif 'GB' in pretty_val.upper():
 242         num_part = int(''.join(d for d in pretty_val if d.isdigit()))
 243         val_in_bytes = num_part * 1024 * 1024 * 1024
 244 
 245     elif 'TB' in pretty_val.upper():
 246         num_part = int(''.join(d for d in pretty_val if d.isdigit()))
 247         val_in_bytes = num_part * 1024 * 1024 * 1024 * 1024
 248 
 249     elif 'B' in pretty_val.upper():
 250         num_part = int(''.join(d for d in pretty_val if d.isdigit()))
 251         val_in_bytes = num_part
 252 
 253     else:
 254         return pretty_val
 255 
 256     return val_in_bytes

 259 def param_set(cursor, module, name, value, context):
 260     try:
 261         if str(value).lower() == 'default':
 262             query = "ALTER SYSTEM SET %s = DEFAULT" % name
 263         else:
 264             query = "ALTER SYSTEM SET %s = '%s'" % (name, value)
 265         cursor.execute(query)
 266 
 267         if context != 'postmaster':
 268             cursor.execute("SELECT pg_reload_conf()")
 269 
 270     except Exception as e:
 271         module.fail_json(msg="Unable to get %s value due to : %s" % (name, to_native(e)))
 272 
 273     return True

 281 def main():
 282     argument_spec = postgres_common_argument_spec()
 283     argument_spec.update(
 284         name=dict(type='str', required=True),
 285         db=dict(type='str', aliases=['login_db']),
 286         value=dict(type='str'),
 287         reset=dict(type='bool'),
 288         session_role=dict(type='str'),
 289     )
 290     module = AnsibleModule(
 291         argument_spec=argument_spec,
 292         supports_check_mode=True,
 293     )
 294 
 295     name = module.params["name"]
 296     value = module.params["value"]
 297     reset = module.params["reset"]
 298 
 299     # Allow to pass values like 1mb instead of 1MB, etc:
 300     if value:
 301         for unit in POSSIBLE_SIZE_UNITS:
 302             if value[:-2].isdigit() and unit in value[-2:]:
 303                 value = value.upper()
 304 
 305     if value and reset:
 306         module.fail_json(msg="%s: value and reset params are mutually exclusive" % name)
 307 
 308     if not value and not reset:
 309         module.fail_json(msg="%s: at least one of value or reset param must be specified" % name)
 310 
 311     conn_params = get_conn_params(module, module.params, warn_db_default=False)
 312     db_connection = connect_to_db(module, conn_params, autocommit=True)
 313     cursor = db_connection.cursor(cursor_factory=DictCursor)
 314 
 315     kw = {}
 316     # Check server version (needs 9.4 or later):
 317     ver = db_connection.server_version
 318     if ver < PG_REQ_VER:
 319         module.warn("PostgreSQL is %s version but %s or later is required" % (ver, PG_REQ_VER))
 320         kw = dict(
 321             changed=False,
 322             restart_required=False,
 323             value_pretty="",
 324             prev_val_pretty="",
 325             value={"value": "", "unit": ""},
 326         )
 327         kw['name'] = name
 328         db_connection.close()
 329         module.exit_json(**kw)
 330 
 331     # Set default returned values:
 332     restart_required = False
 333     changed = False
 334     kw['name'] = name
 335     kw['restart_required'] = False
 336 
 337     # Get info about param state:
 338     res = param_get(cursor, module, name)
 339     current_value = res[0]
 340     raw_val = res[1]
 341     unit = res[2]
 342     boot_val = res[3]
 343     context = res[4]
 344 
 345     if value == 'True':
 346         value = 'on'
 347     elif value == 'False':
 348         value = 'off'
 349 
 350     kw['prev_val_pretty'] = current_value
 351     kw['value_pretty'] = deepcopy(kw['prev_val_pretty'])
 352     kw['context'] = context
 353 
 354     # Do job
 355     if context == "internal":
 356         module.fail_json(msg="%s: cannot be changed (internal context). See "
 357                              "https://www.postgresql.org/docs/current/runtime-config-preset.html" % name)
 358 
 359     if context == "postmaster":
 360         restart_required = True
 361 
 362     # If check_mode, just compare and exit:
 363     if module.check_mode:
 364         if pretty_to_bytes(value) == pretty_to_bytes(current_value):
 365             kw['changed'] = False
 366 
 367         else:
 368             kw['value_pretty'] = value
 369             kw['changed'] = True
 370 
 371         # Anyway returns current raw value in the check_mode:
 372         kw['value'] = dict(
 373             value=raw_val,
 374             unit=unit,
 375         )
 376         kw['restart_required'] = restart_required
 377         module.exit_json(**kw)
 378 
 379     # Set param:
 380     if value and value != current_value:
 381         changed = param_set(cursor, module, name, value, context)
 382 
 383         kw['value_pretty'] = value
 384 
 385     # Reset param:
 386     elif reset:
 387         if raw_val == boot_val:
 388             # nothing to change, exit:
 389             kw['value'] = dict(
 390                 value=raw_val,
 391                 unit=unit,
 392             )
 393             module.exit_json(**kw)
 394 
 395         changed = param_set(cursor, module, name, boot_val, context)
 396 
 397     if restart_required:
 398         module.warn("Restart of PostgreSQL is required for setting %s" % name)
 399 
 400     cursor.close()
 401     db_connection.close()
 402 
 403     # Reconnect and recheck current value:
 404     if context in ('sighup', 'superuser-backend', 'backend', 'superuser', 'user'):
 405         db_connection = connect_to_db(module, conn_params, autocommit=True)
 406         cursor = db_connection.cursor(cursor_factory=DictCursor)
 407 
 408         res = param_get(cursor, module, name)
 409         # f_ means 'final'
 410         f_value = res[0]
 411         f_raw_val = res[1]
 412 
 413         if raw_val == f_raw_val:
 414             changed = False
 415 
 416         else:
 417             changed = True
 418 
 419         kw['value_pretty'] = f_value
 420         kw['value'] = dict(
 421             value=f_raw_val,
 422             unit=unit,
 423         )
 424 
 425         cursor.close()
 426         db_connection.close()
 427 
 428     kw['changed'] = changed
 429     kw['restart_required'] = restart_required
 430     module.exit_json(**kw)
 431 
 432 
 433 if __name__ == '__main__':
 434     main()
