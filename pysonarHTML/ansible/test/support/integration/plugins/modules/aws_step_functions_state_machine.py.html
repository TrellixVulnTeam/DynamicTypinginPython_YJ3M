 104 from ansible.module_utils.aws.core import AnsibleAWSModule
 105 from ansible.module_utils.ec2 import ansible_dict_to_boto3_tag_list, AWSRetry, compare_aws_tags, boto3_tag_list_to_ansible_dict
 106 
 107 try:
 108     from botocore.exceptions import ClientError, BotoCoreError
 109 except ImportError:
 110     pass  # caught by AnsibleAWSModule
 111 
 112 
 113 def manage_state_machine(state, sfn_client, module):
 114     state_machine_arn = get_state_machine_arn(sfn_client, module)
 115 
 116     if state == 'present':
 117         if state_machine_arn is None:
 118             create(sfn_client, module)
 119         else:
 120             update(state_machine_arn, sfn_client, module)
 121     elif state == 'absent':
 122         if state_machine_arn is not None:
 123             remove(state_machine_arn, sfn_client, module)
 124 
 125     check_mode(module, msg='State is up-to-date.')
 126     module.exit_json(changed=False)
 127 
 128 
 129 def create(sfn_client, module):
 130     check_mode(module, msg='State machine would be created.', changed=True)
 131 
 132     tags = module.params.get('tags')
 133     sfn_tags = ansible_dict_to_boto3_tag_list(tags, tag_name_key_name='key', tag_value_key_name='value') if tags else []
 134 
 135     state_machine = sfn_client.create_state_machine(
 136         name=module.params.get('name'),
 137         definition=module.params.get('definition'),
 138         roleArn=module.params.get('role_arn'),
 139         tags=sfn_tags
 140     )
 141     module.exit_json(changed=True, state_machine_arn=state_machine.get('stateMachineArn'))
 142 
 143 
 144 def remove(state_machine_arn, sfn_client, module):
 145     check_mode(module, msg='State machine would be deleted: {0}'.format(state_machine_arn), changed=True)
 146 
 147     sfn_client.delete_state_machine(stateMachineArn=state_machine_arn)
 148     module.exit_json(changed=True, state_machine_arn=state_machine_arn)
 149 
 150 
 151 def update(state_machine_arn, sfn_client, module):
 152     tags_to_add, tags_to_remove = compare_tags(state_machine_arn, sfn_client, module)
 153 
 154     if params_changed(state_machine_arn, sfn_client, module) or tags_to_add or tags_to_remove:
 155         check_mode(module, msg='State machine would be updated: {0}'.format(state_machine_arn), changed=True)
 156 
 157         sfn_client.update_state_machine(
 158             stateMachineArn=state_machine_arn,
 159             definition=module.params.get('definition'),
 160             roleArn=module.params.get('role_arn')
 161         )
 162         sfn_client.untag_resource(
 163             resourceArn=state_machine_arn,
 164             tagKeys=tags_to_remove
 165         )
 166         sfn_client.tag_resource(
 167             resourceArn=state_machine_arn,
 168             tags=ansible_dict_to_boto3_tag_list(tags_to_add, tag_name_key_name='key', tag_value_key_name='value')
 169         )
 170 
 171         module.exit_json(changed=True, state_machine_arn=state_machine_arn)
 172 
 173 
 174 def compare_tags(state_machine_arn, sfn_client, module):
 175     new_tags = module.params.get('tags')
 176     current_tags = sfn_client.list_tags_for_resource(resourceArn=state_machine_arn).get('tags')
 177     return compare_aws_tags(boto3_tag_list_to_ansible_dict(current_tags), new_tags if new_tags else {}, module.params.get('purge_tags'))
 178 
 179 
 180 def params_changed(state_machine_arn, sfn_client, module):
 181     """
 182     Check whether the state machine definition or IAM Role ARN is different
 183     from the existing state machine parameters.
 184     """
 185     current = sfn_client.describe_state_machine(stateMachineArn=state_machine_arn)
 186     return current.get('definition') != module.params.get('definition') or current.get('roleArn') != module.params.get('role_arn')
 187 
 188 
 189 def get_state_machine_arn(sfn_client, module):
 190     """
 191     Finds the state machine ARN based on the name parameter. Returns None if
 192     there is no state machine with this name.
 193     """
 194     target_name = module.params.get('name')
 195     all_state_machines = sfn_client.list_state_machines(aws_retry=True).get('stateMachines')
 196 
 197     for state_machine in all_state_machines:
 198         if state_machine.get('name') == target_name:
 199             return state_machine.get('stateMachineArn')
 200 
 201 
 202 def check_mode(module, msg='', changed=False):
 203     if module.check_mode:
 204         module.exit_json(changed=changed, output=msg)
 205

 207 def main():
 208     module_args = dict(
 209         name=dict(type='str', required=True),
 210         definition=dict(type='json'),
 211         role_arn=dict(type='str'),
 212         state=dict(choices=['present', 'absent'], default='present'),
 213         tags=dict(default=None, type='dict'),
 214         purge_tags=dict(default=True, type='bool'),
 215     )
 216     module = AnsibleAWSModule(
 217         argument_spec=module_args,
 218         required_if=[('state', 'present', ['role_arn']), ('state', 'present', ['definition'])],
 219         supports_check_mode=True
 220     )
 221 
 222     sfn_client = module.client('stepfunctions', retry_decorator=AWSRetry.jittered_backoff(retries=5))
 223     state = module.params.get('state')
 224 
 225     try:
 226         manage_state_machine(state, sfn_client, module)
 227     except (BotoCoreError, ClientError) as e:
 228         module.fail_json_aws(e, msg='Failed to manage state machine')
