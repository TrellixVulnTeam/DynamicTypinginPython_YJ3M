 112 from ansible.module_utils.ansible_tower import TowerModule, tower_auth_config, HAS_TOWER_CLI
 113 
 114 try:
 115     from tower_cli.cli.transfer.receive import Receiver
 116     from tower_cli.cli.transfer.common import SEND_ORDER
 117     from tower_cli.utils.exceptions import TowerCLIError
 118 
 119     from tower_cli.conf import settings
 120     TOWER_CLI_HAS_EXPORT = True
 121 except ImportError:
 122     TOWER_CLI_HAS_EXPORT = False
 123 
 124 
 125 def main():
 126     argument_spec = dict(
 127         all=dict(type='bool', default=False),
 128         credential=dict(type='list', default=[]),
 129         credential_type=dict(type='list', default=[]),
 130         inventory=dict(type='list', default=[]),
 131         inventory_script=dict(type='list', default=[]),
 132         job_template=dict(type='list', default=[]),
 133         notification_template=dict(type='list', default=[]),
 134         organization=dict(type='list', default=[]),
 135         project=dict(type='list', default=[]),
 136         team=dict(type='list', default=[]),
 137         user=dict(type='list', default=[]),
 138         workflow=dict(type='list', default=[]),
 139     )
 140 
 141     module = TowerModule(argument_spec=argument_spec, supports_check_mode=False)
 142 
 143     if not HAS_TOWER_CLI:
 144         module.fail_json(msg='ansible-tower-cli required for this module')
 145 
 146     if not TOWER_CLI_HAS_EXPORT:
 147         module.fail_json(msg='ansible-tower-cli version does not support export')
 148 
 149     export_all = module.params.get('all')
 150     assets_to_export = {}
 151     for asset_type in SEND_ORDER:
 152         assets_to_export[asset_type] = module.params.get(asset_type)
 153 
 154     result = dict(
 155         assets=None,
 156         changed=False,
 157         message='',
 158     )
 159 
 160     tower_auth = tower_auth_config(module)
 161     with settings.runtime_values(**tower_auth):
 162         try:
 163             receiver = Receiver()
 164             result['assets'] = receiver.export_assets(all=export_all, asset_input=assets_to_export)
 165             module.exit_json(**result)
 166         except TowerCLIError as e:
 167             result['message'] = e.message
 168             module.fail_json(msg='Receive Failed', **result)
 169 
 170 
 171 if __name__ == '__main__':
 172     main()
