  73 from ansible.module_utils.aws.core import AnsibleAWSModule
  74 from ansible.module_utils.ec2 import AWSRetry, ansible_dict_to_boto3_filter_list, camel_dict_to_snake_dict
  75 
  76 try:
  77     from botocore.exceptions import ClientError, BotoCoreError
  78 except ImportError:
  79     pass  # Handled by AnsibleAWSModule
  80 
  81 
  82 def main():
  83     argument_spec = dict(
  84         filters=dict(default={}, type='dict')
  85     )
  86 
  87     module = AnsibleAWSModule(argument_spec=argument_spec)
  88     if module._name == 'aws_az_facts':
  89         module.deprecate("The 'aws_az_facts' module has been renamed to 'aws_az_info'",
  90                          version='2.14', collection_name='ansible.builtin')
  91 
  92     connection = module.client('ec2', retry_decorator=AWSRetry.jittered_backoff())
  93 
  94     # Replace filter key underscores with dashes, for compatibility
  95     sanitized_filters = dict((k.replace('_', '-'), v) for k, v in module.params.get('filters').items())
  96 
  97     try:
  98         availability_zones = connection.describe_availability_zones(
  99             Filters=ansible_dict_to_boto3_filter_list(sanitized_filters)
 100         )
 101     except (BotoCoreError, ClientError) as e:
 102         module.fail_json_aws(e, msg="Unable to describe availability zones.")
 103 
 104     # Turn the boto3 result into ansible_friendly_snaked_names
 105     snaked_availability_zones = [camel_dict_to_snake_dict(az) for az in availability_zones['AvailabilityZones']]
 106 
 107     module.exit_json(availability_zones=snaked_availability_zones)
