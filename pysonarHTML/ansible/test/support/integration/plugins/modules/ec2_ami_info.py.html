 214 def list_ec2_images(ec2_client, module):
 215 
 216     image_ids = module.params.get("image_ids")
 217     owners = module.params.get("owners")
 218     executable_users = module.params.get("executable_users")
 219     filters = module.params.get("filters")
 220     owner_param = []
 221 
 222     # describe_images is *very* slow if you pass the `Owners`
 223     # param (unless it's self), for some reason.
 224     # Converting the owners to filters and removing from the
 225     # owners param greatly speeds things up.
 226     # Implementation based on aioue's suggestion in #24886
 227     for owner in owners:
 228         if owner.isdigit():
 229             if 'owner-id' not in filters:
 230                 filters['owner-id'] = list()
 231             filters['owner-id'].append(owner)
 232         elif owner == 'self':
 233             # self not a valid owner-alias filter (https://docs.aws.amazon.com/AWSEC2/latest/APIReference/API_DescribeImages.html)
 234             owner_param.append(owner)
 235         else:
 236             if 'owner-alias' not in filters:
 237                 filters['owner-alias'] = list()
 238             filters['owner-alias'].append(owner)
 239 
 240     filters = ansible_dict_to_boto3_filter_list(filters)
 241 
 242     try:
 243         images = ec2_client.describe_images(ImageIds=image_ids, Filters=filters, Owners=owner_param, ExecutableUsers=executable_users)
 244         images = [camel_dict_to_snake_dict(image) for image in images["Images"]]
 245     except (ClientError, BotoCoreError) as err:
 246         module.fail_json_aws(err, msg="error describing images")
 247     for image in images:
 248         try:
 249             image['tags'] = boto3_tag_list_to_ansible_dict(image.get('tags', []))
 250             if module.params.get("describe_image_attributes"):
 251                 launch_permissions = ec2_client.describe_image_attribute(Attribute='launchPermission', ImageId=image['image_id'])['LaunchPermissions']
 252                 image['launch_permissions'] = [camel_dict_to_snake_dict(perm) for perm in launch_permissions]
 253         except (ClientError, BotoCoreError) as err:
 254             # describing launch permissions of images owned by others is not permitted, but shouldn't cause failures
 255             pass
 256 
 257     images.sort(key=lambda e: e.get('creation_date', ''))  # it may be possible that creation_date does not always exist
 258     module.exit_json(images=images)
