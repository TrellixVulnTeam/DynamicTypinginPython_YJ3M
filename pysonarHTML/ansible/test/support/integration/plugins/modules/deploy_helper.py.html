  23 short_description: Manages some of the steps common in deploying projects.
  24 description:
  25   - The Deploy Helper manages some of the steps common in deploying software.
  26     It creates a folder structure, manages a symlink for the current release
  27     and cleans up old releases.
  28   - "Running it with the C(state=query) or C(state=present) will return the C(deploy_helper) fact.
  29     C(project_path), whatever you set in the path parameter,
  30     C(current_path), the path to the symlink that points to the active release,
  31     C(releases_path), the path to the folder to keep releases in,
  32     C(shared_path), the path to the folder to keep shared resources in,
  33     C(unfinished_filename), the file to check for to recognize unfinished builds,
  34     C(previous_release), the release the 'current' symlink is pointing to,
  35     C(previous_release_path), the full path to the 'current' symlink target,
  36     C(new_release), either the 'release' parameter or a generated timestamp,
  37     C(new_release_path), the path to the new release folder (not created by the module)."
  38 
  39 options:
  40   path:
  41     required: True
  42     aliases: ['dest']
  43     description:
  44       - the root path of the project. Alias I(dest).
  45         Returned in the C(deploy_helper.project_path) fact.
  46 
  47   state:
  48     description:
  49       - the state of the project.
  50         C(query) will only gather facts,
  51         C(present) will create the project I(root) folder, and in it the I(releases) and I(shared) folders,
  52         C(finalize) will remove the unfinished_filename file, create a symlink to the newly
  53           deployed release and optionally clean old releases,
  54         C(clean) will remove failed & old releases,
  55         C(absent) will remove the project folder (synonymous to the M(file) module with C(state=absent))
  56     choices: [ present, finalize, absent, clean, query ]
  57     default: present
  58 
  59   release:
  60     description:
  61       - the release version that is being deployed. Defaults to a timestamp format %Y%m%d%H%M%S (i.e. '20141119223359').
  62         This parameter is optional during C(state=present), but needs to be set explicitly for C(state=finalize).
  63         You can use the generated fact C(release={{ deploy_helper.new_release }}).
  64 
  65   releases_path:
  66     description:
  67       - the name of the folder that will hold the releases. This can be relative to C(path) or absolute.
  68         Returned in the C(deploy_helper.releases_path) fact.
  69     default: releases
  70

  76     default: shared
  77 
  78   current_path:
  79     description:
  80       - the name of the symlink that is created when the deploy is finalized. Used in C(finalize) and C(clean).
  81         Returned in the C(deploy_helper.current_path) fact.

  83

  85     description:

  88         automatically deleted from the I(new_release_path) during C(state=finalize).
  89     default: DEPLOY_UNFINISHED

  92     description:

  99       - the number of old releases to keep when cleaning. Used in C(finalize) and C(clean). Any unfinished builds

 114 
 115 # General explanation, starting with an example folder structure for a project:
 116 
 117 # root:
 118 #     releases:
 119 #         - 20140415234508
 120 #         - 20140415235146
 121 #         - 20140416082818
 122 #
 123 #     shared:
 124 #         - sessions
 125 #         - uploads
 126 #
 127 #     current: releases/20140416082818
 128 
 129 
 130 # The 'releases' folder holds all the available releases. A release is a complete build of the application being
 131 # deployed. This can be a clone of a repository for example, or a sync of a local folder on your filesystem.
 132 # Having timestamped folders is one way of having distinct releases, but you could choose your own strategy like
 133 # git tags or commit hashes.
 134 #
 135 # During a deploy, a new folder should be created in the releases folder and any build steps required should be
 136 # performed. Once the new build is ready, the deploy procedure is 'finalized' by replacing the 'current' symlink
 137 # with a link to this build.
 138 #
 139 # The 'shared' folder holds any resource that is shared between releases. Examples of this are web-server
 140 # session files, or files uploaded by users of your application. It's quite common to have symlinks from a release
 141 # folder pointing to a shared/subfolder, and creating these links would be automated as part of the build steps.
 142 #
 143 # The 'current' symlink points to one of the releases. Probably the latest one, unless a deploy is in progress.
 144 # The web-server's root for the project will go through this symlink, so the 'downtime' when switching to a new
 145 # release is reduced to the time it takes to switch the link.
 146 #
 147 # To distinguish between successful builds and unfinished ones, a file can be placed in the folder of the release
 148 # that is currently in progress. The existence of this file will mark it as unfinished, and allow an automated
 149 # procedure to remove it during cleanup.
 150 
 151 
 152 # Typical usage
 153 - name: Initialize the deploy root and gather facts
 154   deploy_helper:
 155     path: /path/to/root
 156 - name: Clone the project to the new release folder
 157   git:
 158     repo: git://foosball.example.org/path/to/repo.git
 159     dest: '{{ deploy_helper.new_release_path }}'
 160     version: v1.1.1
 161 - name: Add an unfinished file, to allow cleanup on successful finalize
 162   file:
 163     path: '{{ deploy_helper.new_release_path }}/{{ deploy_helper.unfinished_filename }}'
 164     state: touch
 165 - name: Perform some build steps, like running your dependency manager for example
 166   composer:
 167     command: install
 168     working_dir: '{{ deploy_helper.new_release_path }}'
 169 - name: Create some folders in the shared folder
 170   file:
 171     path: '{{ deploy_helper.shared_path }}/{{ item }}'
 172     state: directory
 173   with_items:
 174     - sessions
 175     - uploads
 176 - name: Add symlinks from the new release to the shared folder
 177   file:
 178     path: '{{ deploy_helper.new_release_path }}/{{ item.path }}'
 179     src: '{{ deploy_helper.shared_path }}/{{ item.src }}'
 180     state: link
 181   with_items:
 182       - path: app/sessions
 183         src: sessions
 184       - path: web/uploads
 185         src: uploads
 186 - name: Finalize the deploy, removing the unfinished file and switching the symlink
 187   deploy_helper:
 188     path: /path/to/root
 189     release: '{{ deploy_helper.new_release }}'
 190     state: finalize
 191 
 192 # Retrieving facts before running a deploy
 193 - name: Run 'state=query' to gather facts without changing anything
 194   deploy_helper:
 195     path: /path/to/root
 196     state: query
 197 # Remember to set the 'release' parameter when you actually call 'state=present' later
 198 - name: Initialize the deploy root
 199   deploy_helper:
 200     path: /path/to/root
 201     release: '{{ deploy_helper.new_release }}'
 202     state: present
 203 
 204 # all paths can be absolute or relative (to the 'path' parameter)
 205 - deploy_helper:
 206     path: /path/to/root
 207     releases_path: /var/www/project/releases
 208     shared_path: /var/www/shared
 209     current_path: /var/www/active
 210 
 211 # Using your own naming strategy for releases (a version tag in this case):
 212 - deploy_helper:
 213     path: /path/to/root
 214     release: v1.1.1
 215     state: present
 216 - deploy_helper:
 217     path: /path/to/root
 218     release: '{{ deploy_helper.new_release }}'
 219     state: finalize
 220 
 221 # Using a different unfinished_filename:
 222 - deploy_helper:
 223     path: /path/to/root
 224     unfinished_filename: README.md
 225     release: '{{ deploy_helper.new_release }}'
 226     state: finalize
 227 
 228 # Postponing the cleanup of older builds:
 229 - deploy_helper:
 230     path: /path/to/root
 231     release: '{{ deploy_helper.new_release }}'
 232     state: finalize
 233     clean: False
 234 - deploy_helper:
 235     path: /path/to/root
 236     state: clean
 237 # Or running the cleanup ahead of the new deploy
 238 - deploy_helper:
 239     path: /path/to/root
 240     state: clean
 241 - deploy_helper:
 242     path: /path/to/root
 243     state: present
 244 
 245 # Keeping more old releases:
 246 - deploy_helper:
 247     path: /path/to/root
 248     release: '{{ deploy_helper.new_release }}'
 249     state: finalize
 250     keep_releases: 10

 268 import os
 269 import shutil
 270 import time
 271 import traceback
 272 
 273 from ansible.module_utils.basic import AnsibleModule
 274 from ansible.module_utils._text import to_native
 275 
 276 
 277 class DeployHelper(object):
 278 
 279     def __init__(self, module):
 280         self.module = module
 281         self.file_args = module.load_file_common_arguments(module.params)
 282 
 283         self.clean = module.params['clean']
 284         self.current_path = module.params['current_path']
 285         self.keep_releases = module.params['keep_releases']
 286         self.path = module.params['path']
 287         self.release = module.params['release']
 288         self.releases_path = module.params['releases_path']
 289         self.shared_path = module.params['shared_path']
 290         self.state = module.params['state']
 291         self.unfinished_filename = module.params['unfinished_filename']
 292 
 293     def gather_facts(self):
 294         current_path = os.path.join(self.path, self.current_path)
 295         releases_path = os.path.join(self.path, self.releases_path)
 296         if self.shared_path:
 297             shared_path = os.path.join(self.path, self.shared_path)
 298         else:
 299             shared_path = None
 300 
 301         previous_release, previous_release_path = self._get_last_release(current_path)
 302 
 303         if not self.release and (self.state == 'query' or self.state == 'present'):
 304             self.release = time.strftime("%Y%m%d%H%M%S")
 305 
 306         if self.release:
 307             new_release_path = os.path.join(releases_path, self.release)
 308         else:
 309             new_release_path = None
 310 
 311         return {
 312             'project_path': self.path,
 313             'current_path': current_path,
 314             'releases_path': releases_path,
 315             'shared_path': shared_path,
 316             'previous_release': previous_release,
 317             'previous_release_path': previous_release_path,
 318             'new_release': self.release,
 319             'new_release_path': new_release_path,
 320             'unfinished_filename': self.unfinished_filename
 321         }
 322 
 323     def delete_path(self, path):
 324         if not os.path.lexists(path):
 325             return False
 326 
 327         if not os.path.isdir(path):
 328             self.module.fail_json(msg="%s exists but is not a directory" % path)
 329 
 330         if not self.module.check_mode:
 331             try:
 332                 shutil.rmtree(path, ignore_errors=False)
 333             except Exception as e:
 334                 self.module.fail_json(msg="rmtree failed: %s" % to_native(e), exception=traceback.format_exc())
 335 
 336         return True
 337 
 338     def create_path(self, path):
 339         changed = False
 340 
 341         if not os.path.lexists(path):
 342             changed = True
 343             if not self.module.check_mode:
 344                 os.makedirs(path)
 345 
 346         elif not os.path.isdir(path):
 347             self.module.fail_json(msg="%s exists but is not a directory" % path)
 348 
 349         changed += self.module.set_directory_attributes_if_different(self._get_file_args(path), changed)
 350 
 351         return changed
 352 
 353     def check_link(self, path):
 354         if os.path.lexists(path):
 355             if not os.path.islink(path):
 356                 self.module.fail_json(msg="%s exists but is not a symbolic link" % path)
 357 
 358     def create_link(self, source, link_name):
 359         changed = False
 360 
 361         if os.path.islink(link_name):
 362             norm_link = os.path.normpath(os.path.realpath(link_name))
 363             norm_source = os.path.normpath(os.path.realpath(source))
 364             if norm_link == norm_source:
 365                 changed = False
 366             else:
 367                 changed = True
 368                 if not self.module.check_mode:
 369                     if not os.path.lexists(source):
 370                         self.module.fail_json(msg="the symlink target %s doesn't exists" % source)
 371                     tmp_link_name = link_name + '.' + self.unfinished_filename
 372                     if os.path.islink(tmp_link_name):
 373                         os.unlink(tmp_link_name)
 374                     os.symlink(source, tmp_link_name)
 375                     os.rename(tmp_link_name, link_name)
 376         else:
 377             changed = True
 378             if not self.module.check_mode:
 379                 os.symlink(source, link_name)
 380 
 381         return changed
 382 
 383     def remove_unfinished_file(self, new_release_path):
 384         changed = False
 385         unfinished_file_path = os.path.join(new_release_path, self.unfinished_filename)
 386         if os.path.lexists(unfinished_file_path):
 387             changed = True
 388             if not self.module.check_mode:
 389                 os.remove(unfinished_file_path)
 390 
 391         return changed
 392 
 393     def remove_unfinished_builds(self, releases_path):
 394         changes = 0
 395 
 396         for release in os.listdir(releases_path):
 397             if os.path.isfile(os.path.join(releases_path, release, self.unfinished_filename)):
 398                 if self.module.check_mode:
 399                     changes += 1
 400                 else:
 401                     changes += self.delete_path(os.path.join(releases_path, release))
 402 
 403         return changes
 404 
 405     def remove_unfinished_link(self, path):
 406         changed = False
 407 
 408         tmp_link_name = os.path.join(path, self.release + '.' + self.unfinished_filename)
 409         if not self.module.check_mode and os.path.exists(tmp_link_name):
 410             changed = True
 411             os.remove(tmp_link_name)
 412 
 413         return changed
 414 
 415     def cleanup(self, releases_path, reserve_version):
 416         changes = 0
 417 
 418         if os.path.lexists(releases_path):
 419             releases = [f for f in os.listdir(releases_path) if os.path.isdir(os.path.join(releases_path, f))]
 420             try:
 421                 releases.remove(reserve_version)
 422             except ValueError:
 423                 pass
 424 
 425             if not self.module.check_mode:
 426                 releases.sort(key=lambda x: os.path.getctime(os.path.join(releases_path, x)), reverse=True)
 427                 for release in releases[self.keep_releases:]:
 428                     changes += self.delete_path(os.path.join(releases_path, release))
 429             elif len(releases) > self.keep_releases:
 430                 changes += (len(releases) - self.keep_releases)
 431 
 432         return changes
 433 
 434     def _get_file_args(self, path):
 435         file_args = self.file_args.copy()
 436         file_args['path'] = path
 437         return file_args
 438 
 439     def _get_last_release(self, current_path):
 440         previous_release = None
 441         previous_release_path = None
 442 
 443         if os.path.lexists(current_path):
 444             previous_release_path = os.path.realpath(current_path)
 445             previous_release = os.path.basename(previous_release_path)
 446 
 447         return previous_release, previous_release_path

 450 def main():
 451 
 452     module = AnsibleModule(
 453         argument_spec=dict(
 454             path=dict(aliases=['dest'], required=True, type='path'),
 455             release=dict(required=False, type='str', default=None),
 456             releases_path=dict(required=False, type='str', default='releases'),
 457             shared_path=dict(required=False, type='path', default='shared'),
 458             current_path=dict(required=False, type='path', default='current'),
 459             keep_releases=dict(required=False, type='int', default=5),
 460             clean=dict(required=False, type='bool', default=True),
 461             unfinished_filename=dict(required=False, type='str', default='DEPLOY_UNFINISHED'),
 462             state=dict(required=False, choices=['present', 'absent', 'clean', 'finalize', 'query'], default='present')
 463         ),
 464         add_file_common_args=True,
 465         supports_check_mode=True
 466     )
 467 
 468     deploy_helper = DeployHelper(module)
 469     facts = deploy_helper.gather_facts()
 470 
 471     result = {
 472         'state': deploy_helper.state
 473     }
 474 
 475     changes = 0
 476 
 477     if deploy_helper.state == 'query':
 478         result['ansible_facts'] = {'deploy_helper': facts}
 479 
 480     elif deploy_helper.state == 'present':
 481         deploy_helper.check_link(facts['current_path'])
 482         changes += deploy_helper.create_path(facts['project_path'])
 483         changes += deploy_helper.create_path(facts['releases_path'])
 484         if deploy_helper.shared_path:
 485             changes += deploy_helper.create_path(facts['shared_path'])
 486 
 487         result['ansible_facts'] = {'deploy_helper': facts}
 488 
 489     elif deploy_helper.state == 'finalize':
 490         if not deploy_helper.release:
 491             module.fail_json(msg="'release' is a required parameter for state=finalize (try the 'deploy_helper.new_release' fact)")
 492         if deploy_helper.keep_releases <= 0:
 493             module.fail_json(msg="'keep_releases' should be at least 1")
 494 
 495         changes += deploy_helper.remove_unfinished_file(facts['new_release_path'])
 496         changes += deploy_helper.create_link(facts['new_release_path'], facts['current_path'])
 497         if deploy_helper.clean:
 498             changes += deploy_helper.remove_unfinished_link(facts['project_path'])
 499             changes += deploy_helper.remove_unfinished_builds(facts['releases_path'])
 500             changes += deploy_helper.cleanup(facts['releases_path'], facts['new_release'])
 501 
 502     elif deploy_helper.state == 'clean':
 503         changes += deploy_helper.remove_unfinished_link(facts['project_path'])
 504         changes += deploy_helper.remove_unfinished_builds(facts['releases_path'])
 505         changes += deploy_helper.cleanup(facts['releases_path'], facts['new_release'])
 506 
 507     elif deploy_helper.state == 'absent':
 508         # destroy the facts
 509         result['ansible_facts'] = {'deploy_helper': []}
 510         changes += deploy_helper.delete_path(facts['project_path'])
 511 
 512     if changes > 0:
 513         result['changed'] = True
 514     else:
 515         result['changed'] = False
 516 
 517     module.exit_json(**result)
 518 
 519 
 520 if __name__ == '__main__':
 521     main()
