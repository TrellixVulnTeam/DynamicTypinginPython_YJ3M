 189 class AzureRMFunctionApp(AzureRMModuleBase):
 190 
 191     def __init__(self):
 192 
 193         self.module_arg_spec = dict(
 194             resource_group=dict(type='str', required=True, aliases=['resource_group_name']),
 195             name=dict(type='str', required=True),
 196             state=dict(type='str', default='present', choices=['present', 'absent']),
 197             location=dict(type='str'),
 198             storage_account=dict(
 199                 type='str',
 200                 aliases=['storage', 'storage_account_name']
 201             ),
 202             app_settings=dict(type='dict'),
 203             plan=dict(
 204                 type='raw'
 205             ),
 206             container_settings=dict(
 207                 type='dict',
 208                 options=container_settings_spec
 209             )
 210         )
 211 
 212         self.results = dict(
 213             changed=False,
 214             state=dict()
 215         )
 216 
 217         self.resource_group = None
 218         self.name = None
 219         self.state = None
 220         self.location = None
 221         self.storage_account = None
 222         self.app_settings = None
 223         self.plan = None
 224         self.container_settings = None
 225 
 226         required_if = [('state', 'present', ['storage_account'])]
 227 
 228         super(AzureRMFunctionApp, self).__init__(
 229             self.module_arg_spec,
 230             supports_check_mode=True,
 231             required_if=required_if
 232         )
 233 
 234     def exec_module(self, **kwargs):
 235 
 236         for key in self.module_arg_spec:
 237             setattr(self, key, kwargs[key])
 238         if self.app_settings is None:
 239             self.app_settings = dict()
 240 
 241         try:
 242             resource_group = self.rm_client.resource_groups.get(self.resource_group)
 243         except CloudError:
 244             self.fail('Unable to retrieve resource group')
 245 
 246         self.location = self.location or resource_group.location
 247 
 248         try:
 249             function_app = self.web_client.web_apps.get(
 250                 resource_group_name=self.resource_group,
 251                 name=self.name
 252             )
 253             # Newer SDK versions (0.40.0+) seem to return None if it doesn't exist instead of raising CloudError
 254             exists = function_app is not None
 255         except CloudError as exc:
 256             exists = False
 257 
 258         if self.state == 'absent':
 259             if exists:
 260                 if self.check_mode:
 261                     self.results['changed'] = True
 262                     return self.results
 263                 try:
 264                     self.web_client.web_apps.delete(
 265                         resource_group_name=self.resource_group,
 266                         name=self.name
 267                     )
 268                     self.results['changed'] = True
 269                 except CloudError as exc:
 270                     self.fail('Failure while deleting web app: {0}'.format(exc))
 271             else:
 272                 self.results['changed'] = False
 273         else:
 274             kind = 'functionapp'
 275             linux_fx_version = None
 276             if self.container_settings and self.container_settings.get('name'):
 277                 kind = 'functionapp,linux,container'
 278                 linux_fx_version = 'DOCKER|'
 279                 if self.container_settings.get('registry_server_url'):
 280                     self.app_settings['DOCKER_REGISTRY_SERVER_URL'] = 'https://' + self.container_settings['registry_server_url']
 281                     linux_fx_version += self.container_settings['registry_server_url'] + '/'
 282                 linux_fx_version += self.container_settings['name']
 283                 if self.container_settings.get('registry_server_user'):
 284                     self.app_settings['DOCKER_REGISTRY_SERVER_USERNAME'] = self.container_settings.get('registry_server_user')
 285 
 286                 if self.container_settings.get('registry_server_password'):
 287                     self.app_settings['DOCKER_REGISTRY_SERVER_PASSWORD'] = self.container_settings.get('registry_server_password')
 288 
 289             if not self.plan and function_app:
 290                 self.plan = function_app.server_farm_id
 291 
 292             if not exists:
 293                 function_app = Site(
 294                     location=self.location,
 295                     kind=kind,
 296                     site_config=SiteConfig(
 297                         app_settings=self.aggregated_app_settings(),
 298                         scm_type='LocalGit'
 299                     )
 300                 )
 301                 self.results['changed'] = True
 302             else:
 303                 self.results['changed'], function_app = self.update(function_app)
 304 
 305             # get app service plan
 306             if self.plan:
 307                 if isinstance(self.plan, dict):
 308                     self.plan = "/subscriptions/{0}/resourceGroups/{1}/providers/Microsoft.Web/serverfarms/{2}".format(
 309                         self.subscription_id,
 310                         self.plan.get('resource_group', self.resource_group),
 311                         self.plan.get('name')
 312                     )
 313                 function_app.server_farm_id = self.plan
 314 
 315             # set linux fx version
 316             if linux_fx_version:
 317                 function_app.site_config.linux_fx_version = linux_fx_version
 318 
 319             if self.check_mode:
 320                 self.results['state'] = function_app.as_dict()
 321             elif self.results['changed']:
 322                 try:
 323                     new_function_app = self.web_client.web_apps.create_or_update(
 324                         resource_group_name=self.resource_group,
 325                         name=self.name,
 326                         site_envelope=function_app
 327                     ).result()
 328                     self.results['state'] = new_function_app.as_dict()
 329                 except CloudError as exc:
 330                     self.fail('Error creating or updating web app: {0}'.format(exc))
 331 
 332         return self.results
 333 
 334     def update(self, source_function_app):
 335         """Update the Site object if there are any changes"""
 336 
 337         source_app_settings = self.web_client.web_apps.list_application_settings(
 338             resource_group_name=self.resource_group,
 339             name=self.name
 340         )
 341 
 342         changed, target_app_settings = self.update_app_settings(source_app_settings.properties)
 343 
 344         source_function_app.site_config = SiteConfig(
 345             app_settings=target_app_settings,
 346             scm_type='LocalGit'
 347         )
 348 
 349         return changed, source_function_app
 350 
 351     def update_app_settings(self, source_app_settings):
 352         """Update app settings"""
 353 
 354         target_app_settings = self.aggregated_app_settings()
 355         target_app_settings_dict = dict([(i.name, i.value) for i in target_app_settings])
 356         return target_app_settings_dict != source_app_settings, target_app_settings
 357 
 358     def necessary_functionapp_settings(self):
 359         """Construct the necessary app settings required for an Azure Function App"""
 360 
 361         function_app_settings = []
 362 
 363         if self.container_settings is None:
 364             for key in ['AzureWebJobsStorage', 'WEBSITE_CONTENTAZUREFILECONNECTIONSTRING', 'AzureWebJobsDashboard']:
 365                 function_app_settings.append(NameValuePair(name=key, value=self.storage_connection_string))
 366             function_app_settings.append(NameValuePair(name='FUNCTIONS_EXTENSION_VERSION', value='~1'))
 367             function_app_settings.append(NameValuePair(name='WEBSITE_NODE_DEFAULT_VERSION', value='6.5.0'))
 368             function_app_settings.append(NameValuePair(name='WEBSITE_CONTENTSHARE', value=self.name))
 369         else:
 370             function_app_settings.append(NameValuePair(name='FUNCTIONS_EXTENSION_VERSION', value='~2'))
 371             function_app_settings.append(NameValuePair(name='WEBSITES_ENABLE_APP_SERVICE_STORAGE', value=False))
 372             function_app_settings.append(NameValuePair(name='AzureWebJobsStorage', value=self.storage_connection_string))
 373 
 374         return function_app_settings
 375 
 376     def aggregated_app_settings(self):
 377         """Combine both system and user app settings"""
 378 
 379         function_app_settings = self.necessary_functionapp_settings()
 380         for app_setting_key in self.app_settings:
 381             found_setting = None
 382             for s in function_app_settings:
 383                 if s.name == app_setting_key:
 384                     found_setting = s
 385                     break
 386             if found_setting:
 387                 found_setting.value = self.app_settings[app_setting_key]
 388             else:
 389                 function_app_settings.append(NameValuePair(
 390                     name=app_setting_key,
 391                     value=self.app_settings[app_setting_key]
 392                 ))
 393         return function_app_settings
 394 
 395     @property
 396     def storage_connection_string(self):
 397         """Construct the storage account connection string"""
 398 
 399         return 'DefaultEndpointsProtocol=https;AccountName={0};AccountKey={1}'.format(
 400             self.storage_account,
 401             self.storage_key
 402         )
 403 
 404     @property
 405     def storage_key(self):
 406         """Retrieve the storage account key"""
 407 
 408         return self.storage_client.storage_accounts.list_keys(
 409             resource_group_name=self.resource_group,
 410             account_name=self.storage_account
 411         ).keys[0].value
