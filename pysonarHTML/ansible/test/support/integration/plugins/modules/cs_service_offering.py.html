 396 class AnsibleCloudStackServiceOffering(AnsibleCloudStack):
 397 
 398     def __init__(self, module):
 399         super(AnsibleCloudStackServiceOffering, self).__init__(module)
 400         self.returns = {
 401             'cpunumber': 'cpu_number',
 402             'cpuspeed': 'cpu_speed',
 403             'deploymentplanner': 'deployment_planner',
 404             'diskBytesReadRate': 'disk_bytes_read_rate',
 405             'diskBytesWriteRate': 'disk_bytes_write_rate',
 406             'diskIopsReadRate': 'disk_iops_read_rate',
 407             'diskIopsWriteRate': 'disk_iops_write_rate',
 408             'maxiops': 'disk_iops_max',
 409             'miniops': 'disk_iops_min',
 410             'hypervisorsnapshotreserve': 'hypervisor_snapshot_reserve',
 411             'iscustomized': 'is_customized',
 412             'iscustomizediops': 'is_iops_customized',
 413             'issystem': 'is_system',
 414             'isvolatile': 'is_volatile',
 415             'limitcpuuse': 'limit_cpu_usage',
 416             'memory': 'memory',
 417             'networkrate': 'network_rate',
 418             'offerha': 'offer_ha',
 419             'provisioningtype': 'provisioning_type',
 420             'serviceofferingdetails': 'service_offering_details',
 421             'storagetype': 'storage_type',
 422             'systemvmtype': 'system_vm_type',
 423             'tags': 'storage_tags',
 424         }
 425 
 426     def get_service_offering(self):
 427         args = {
 428             'name': self.module.params.get('name'),
 429             'domainid': self.get_domain(key='id'),
 430             'issystem': self.module.params.get('is_system'),
 431             'systemvmtype': self.module.params.get('system_vm_type'),
 432         }
 433         service_offerings = self.query_api('listServiceOfferings', **args)
 434         if service_offerings:
 435             return service_offerings['serviceoffering'][0]
 436 
 437     def present_service_offering(self):
 438         service_offering = self.get_service_offering()
 439         if not service_offering:
 440             service_offering = self._create_offering(service_offering)
 441         else:
 442             service_offering = self._update_offering(service_offering)
 443 
 444         return service_offering
 445 
 446     def absent_service_offering(self):
 447         service_offering = self.get_service_offering()
 448         if service_offering:
 449             self.result['changed'] = True
 450             if not self.module.check_mode:
 451                 args = {
 452                     'id': service_offering['id'],
 453                 }
 454                 self.query_api('deleteServiceOffering', **args)
 455         return service_offering
 456 
 457     def _create_offering(self, service_offering):
 458         self.result['changed'] = True
 459 
 460         system_vm_type = self.module.params.get('system_vm_type')
 461         is_system = self.module.params.get('is_system')
 462 
 463         required_params = []
 464         if is_system and not system_vm_type:
 465             required_params.append('system_vm_type')
 466         self.module.fail_on_missing_params(required_params=required_params)
 467 
 468         args = {
 469             'name': self.module.params.get('name'),
 470             'displaytext': self.get_or_fallback('display_text', 'name'),
 471             'bytesreadrate': self.module.params.get('disk_bytes_read_rate'),
 472             'byteswriterate': self.module.params.get('disk_bytes_write_rate'),
 473             'cpunumber': self.module.params.get('cpu_number'),
 474             'cpuspeed': self.module.params.get('cpu_speed'),
 475             'customizediops': self.module.params.get('is_iops_customized'),
 476             'deploymentplanner': self.module.params.get('deployment_planner'),
 477             'domainid': self.get_domain(key='id'),
 478             'hosttags': self.module.params.get('host_tags'),
 479             'hypervisorsnapshotreserve': self.module.params.get('hypervisor_snapshot_reserve'),
 480             'iopsreadrate': self.module.params.get('disk_iops_read_rate'),
 481             'iopswriterate': self.module.params.get('disk_iops_write_rate'),
 482             'maxiops': self.module.params.get('disk_iops_max'),
 483             'miniops': self.module.params.get('disk_iops_min'),
 484             'issystem': is_system,
 485             'isvolatile': self.module.params.get('is_volatile'),
 486             'memory': self.module.params.get('memory'),
 487             'networkrate': self.module.params.get('network_rate'),
 488             'offerha': self.module.params.get('offer_ha'),
 489             'provisioningtype': self.module.params.get('provisioning_type'),
 490             'serviceofferingdetails': self.module.params.get('service_offering_details'),
 491             'storagetype': self.module.params.get('storage_type'),
 492             'systemvmtype': system_vm_type,
 493             'tags': self.module.params.get('storage_tags'),
 494             'limitcpuuse': self.module.params.get('limit_cpu_usage'),
 495             'customized': self.module.params.get('is_customized')
 496         }
 497         if not self.module.check_mode:
 498             res = self.query_api('createServiceOffering', **args)
 499             service_offering = res['serviceoffering']
 500         return service_offering
 501 
 502     def _update_offering(self, service_offering):
 503         args = {
 504             'id': service_offering['id'],
 505             'name': self.module.params.get('name'),
 506             'displaytext': self.get_or_fallback('display_text', 'name'),
 507         }
 508         if self.has_changed(args, service_offering):
 509             self.result['changed'] = True
 510 
 511             if not self.module.check_mode:
 512                 res = self.query_api('updateServiceOffering', **args)
 513                 service_offering = res['serviceoffering']
 514         return service_offering
 515 
 516     def get_result(self, service_offering):
 517         super(AnsibleCloudStackServiceOffering, self).get_result(service_offering)
 518         if service_offering:
 519             if 'hosttags' in service_offering:
 520                 self.result['host_tags'] = service_offering['hosttags'].split(',') or [service_offering['hosttags']]
 521 
 522             # Prevent confusion, the api returns a tags key for storage tags.
 523             if 'tags' in service_offering:
 524                 self.result['storage_tags'] = service_offering['tags'].split(',') or [service_offering['tags']]
 525             if 'tags' in self.result:
 526                 del self.result['tags']
 527 
 528         return self.result

 531 def main():
 532     argument_spec = cs_argument_spec()
 533     argument_spec.update(dict(
 534         name=dict(required=True),
 535         display_text=dict(),
 536         cpu_number=dict(type='int'),
 537         cpu_speed=dict(type='int'),
 538         limit_cpu_usage=dict(type='bool'),
 539         deployment_planner=dict(),
 540         domain=dict(),
 541         host_tags=dict(type='list', aliases=['host_tag']),
 542         hypervisor_snapshot_reserve=dict(type='int'),
 543         disk_bytes_read_rate=dict(type='int', aliases=['bytes_read_rate']),
 544         disk_bytes_write_rate=dict(type='int', aliases=['bytes_write_rate']),
 545         disk_iops_read_rate=dict(type='int'),
 546         disk_iops_write_rate=dict(type='int'),
 547         disk_iops_max=dict(type='int'),
 548         disk_iops_min=dict(type='int'),
 549         is_system=dict(type='bool', default=False),
 550         is_volatile=dict(type='bool'),
 551         is_iops_customized=dict(type='bool', aliases=['disk_iops_customized']),
 552         memory=dict(type='int'),
 553         network_rate=dict(type='int'),
 554         offer_ha=dict(type='bool'),
 555         provisioning_type=dict(choices=['thin', 'sparse', 'fat']),
 556         service_offering_details=dict(type='list'),
 557         storage_type=dict(choices=['local', 'shared']),
 558         system_vm_type=dict(choices=['domainrouter', 'consoleproxy', 'secondarystoragevm']),
 559         storage_tags=dict(type='list', aliases=['storage_tag']),
 560         state=dict(choices=['present', 'absent'], default='present'),
 561         is_customized=dict(type='bool'),
 562     ))
 563 
 564     module = AnsibleModule(
 565         argument_spec=argument_spec,
 566         required_together=cs_required_together(),
 567         supports_check_mode=True
 568     )
 569 
 570     acs_so = AnsibleCloudStackServiceOffering(module)
 571 
 572     state = module.params.get('state')
 573     if state == "absent":
 574         service_offering = acs_so.absent_service_offering()
 575     else:
 576         service_offering = acs_so.present_service_offering()
 577 
 578     result = acs_so.get_result(service_offering)
 579     module.exit_json(**result)
