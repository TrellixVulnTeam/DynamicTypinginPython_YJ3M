 302 def create_or_update_project(client, params, module):
 303     resp = {}
 304     name = params['name']
 305     # clean up params
 306     formatted_params = snake_dict_to_camel_dict(dict((k, v) for k, v in params.items() if v is not None))
 307     permitted_create_params = get_boto3_client_method_parameters(client, 'create_project')
 308     permitted_update_params = get_boto3_client_method_parameters(client, 'update_project')
 309 
 310     formatted_create_params = dict((k, v) for k, v in formatted_params.items() if k in permitted_create_params)
 311     formatted_update_params = dict((k, v) for k, v in formatted_params.items() if k in permitted_update_params)
 312 
 313     # Check if project with that name already exists and if so update existing:
 314     found = describe_project(client=client, name=name, module=module)
 315     changed = False
 316 
 317     if 'name' in found:
 318         found_project = found
 319         resp = update_project(client=client, params=formatted_update_params, module=module)
 320         updated_project = resp['project']
 321 
 322         # Prep both dicts for sensible change comparison:
 323         found_project.pop('lastModified')
 324         updated_project.pop('lastModified')
 325         if 'tags' not in updated_project:
 326             updated_project['tags'] = []
 327 
 328         if updated_project != found_project:
 329             changed = True
 330         return resp, changed
 331     # Or create new project:
 332     try:
 333         resp = client.create_project(**formatted_create_params)
 334         changed = True
 335         return resp, changed
 336     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 337         module.fail_json_aws(e, msg="Unable to create CodeBuild project")

 350 def delete_project(client, name, module):
 351     found = describe_project(client=client, name=name, module=module)

 363 def describe_project(client, name, module):
 364     project = {}
 365     try:
 366         projects = client.batch_get_projects(names=[name])['projects']
 367         if len(projects) > 0:
 368             project = projects[0]
 369         return project
 370     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 371         module.fail_json_aws(e, msg="Unable to describe CodeBuild projects")

 374 def main():
 375     argument_spec = dict(
 376         name=dict(required=True),
 377         description=dict(),
 378         source=dict(required=True, type='dict'),
 379         artifacts=dict(required=True, type='dict'),
 380         cache=dict(type='dict'),
 381         environment=dict(type='dict'),
 382         service_role=dict(),
 383         timeout_in_minutes=dict(type='int', default=60),
 384         encryption_key=dict(),
 385         tags=dict(type='list'),
 386         vpc_config=dict(type='dict'),
 387         state=dict(choices=['present', 'absent'], default='present')
 388     )
 389 
 390     module = AnsibleAWSModule(argument_spec=argument_spec)
 391     client_conn = module.client('codebuild')
 392 
 393     state = module.params.get('state')
 394     changed = False
 395 
 396     if state == 'present':
 397         project_result, changed = create_or_update_project(
 398             client=client_conn,
 399             params=module.params,
 400             module=module)
 401     elif state == 'absent':
 402         project_result, changed = delete_project(client=client_conn, name=module.params['name'], module=module)
 403 
 404     module.exit_json(changed=changed, **camel_dict_to_snake_dict(project_result))
