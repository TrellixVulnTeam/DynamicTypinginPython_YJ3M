 121 def semanage_login_add(module, login, seuser, do_reload, serange='s0', sestore=''):
 122     """ Add linux user to SELinux user mapping
 123 
 124     :type module: AnsibleModule
 125     :param module: Ansible module
 126 
 127     :type login: str
 128     :param login: a Linux User or a Linux group if it begins with %
 129 
 130     :type seuser: str
 131     :param proto: An SELinux user ('__default__', 'unconfined_u', 'staff_u', ...), see 'semanage login -l'
 132 
 133     :type serange: str
 134     :param serange: SELinux MLS/MCS range (defaults to 's0')
 135 
 136     :type do_reload: bool
 137     :param do_reload: Whether to reload SELinux policy after commit
 138 
 139     :type sestore: str
 140     :param sestore: SELinux store
 141 
 142     :rtype: bool
 143     :return: True if the policy was changed, otherwise False
 144     """
 145     try:
 146         selogin = seobject.loginRecords(sestore)
 147         selogin.set_reload(do_reload)
 148         change = False
 149         all_logins = selogin.get_all()
 150         # module.fail_json(msg="%s: %s %s" % (all_logins, login, sestore))
 151         # for local_login in all_logins:
 152         if login not in all_logins.keys():
 153             change = True
 154             if not module.check_mode:
 155                 selogin.add(login, seuser, serange)
 156         else:
 157             if all_logins[login][0] != seuser or all_logins[login][1] != serange:
 158                 change = True
 159                 if not module.check_mode:
 160                     selogin.modify(login, seuser, serange)
 161 
 162     except (ValueError, KeyError, OSError, RuntimeError) as e:
 163         module.fail_json(msg="%s: %s\n" % (e.__class__.__name__, to_native(e)), exception=traceback.format_exc())
 164 
 165     return change

 168 def semanage_login_del(module, login, seuser, do_reload, sestore=''):
 169     """ Delete linux user to SELinux user mapping
 170 
 171     :type module: AnsibleModule
 172     :param module: Ansible module
 173 
 174     :type login: str
 175     :param login: a Linux User or a Linux group if it begins with %
 176 
 177     :type seuser: str
 178     :param proto: An SELinux user ('__default__', 'unconfined_u', 'staff_u', ...), see 'semanage login -l'
 179 
 180     :type do_reload: bool
 181     :param do_reload: Whether to reload SELinux policy after commit
 182 
 183     :type sestore: str
 184     :param sestore: SELinux store
 185 
 186     :rtype: bool
 187     :return: True if the policy was changed, otherwise False
 188     """
 189     try:
 190         selogin = seobject.loginRecords(sestore)
 191         selogin.set_reload(do_reload)
 192         change = False
 193         all_logins = selogin.get_all()
 194         # module.fail_json(msg="%s: %s %s" % (all_logins, login, sestore))
 195         if login in all_logins.keys():
 196             change = True
 197             if not module.check_mode:
 198                 selogin.delete(login)
 199 
 200     except (ValueError, KeyError, OSError, RuntimeError) as e:
 201         module.fail_json(msg="%s: %s\n" % (e.__class__.__name__, to_native(e)), exception=traceback.format_exc())
 202 
 203     return change

 206 def get_runtime_status(ignore_selinux_state=False):
 207     return True if ignore_selinux_state is True else selinux.is_selinux_enabled()
 208

 210 def main():
 211     module = AnsibleModule(
 212         argument_spec=dict(
 213             ignore_selinux_state=dict(type='bool', default=False),
 214             login=dict(type='str', required=True),
 215             seuser=dict(type='str'),
 216             selevel=dict(type='str', aliases=['serange'], default='s0'),
 217             state=dict(type='str', default='present', choices=['absent', 'present']),
 218             reload=dict(type='bool', default=True),
 219         ),
 220         required_if=[
 221             ["state", "present", ["seuser"]]
 222         ],
 223         supports_check_mode=True
 224     )
 225     if not HAVE_SELINUX:
 226         module.fail_json(msg=missing_required_lib("libselinux"), exception=SELINUX_IMP_ERR)
 227 
 228     if not HAVE_SEOBJECT:
 229         module.fail_json(msg=missing_required_lib("seobject from policycoreutils"), exception=SEOBJECT_IMP_ERR)
 230 
 231     ignore_selinux_state = module.params['ignore_selinux_state']
 232 
 233     if not get_runtime_status(ignore_selinux_state):
 234         module.fail_json(msg="SELinux is disabled on this host.")
 235 
 236     login = module.params['login']
 237     seuser = module.params['seuser']
 238     serange = module.params['selevel']
 239     state = module.params['state']
 240     do_reload = module.params['reload']
 241 
 242     result = {
 243         'login': login,
 244         'seuser': seuser,
 245         'serange': serange,
 246         'state': state,
 247     }
 248 
 249     if state == 'present':
 250         result['changed'] = semanage_login_add(module, login, seuser, do_reload, serange)
 251     elif state == 'absent':
 252         result['changed'] = semanage_login_del(module, login, seuser, do_reload)
 253     else:
 254         module.fail_json(msg='Invalid value of argument "state": {0}'.format(state))
 255 
 256     module.exit_json(**result)
 257

 259 if __name__ == '__main__':
 260     main()
