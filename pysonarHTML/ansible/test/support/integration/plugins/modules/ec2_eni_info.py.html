 191 def list_eni(connection, module):
 192 
 193     if module.params.get("filters") is None:
 194         filters = []
 195     else:
 196         filters = ansible_dict_to_boto3_filter_list(module.params.get("filters"))
 197 
 198     try:
 199         network_interfaces_result = connection.describe_network_interfaces(Filters=filters)['NetworkInterfaces']
 200     except (ClientError, NoCredentialsError) as e:
 201         module.fail_json(msg=e.message)
 202 
 203     # Modify boto3 tags list to be ansible friendly dict and then camel_case
 204     camel_network_interfaces = []
 205     for network_interface in network_interfaces_result:
 206         network_interface['TagSet'] = boto3_tag_list_to_ansible_dict(network_interface['TagSet'])
 207         # Added id to interface info to be compatible with return values of ec2_eni module:
 208         network_interface['Id'] = network_interface['NetworkInterfaceId']
 209         camel_network_interfaces.append(camel_dict_to_snake_dict(network_interface))
 210 
 211     module.exit_json(network_interfaces=camel_network_interfaces)

 214 def get_eni_info(interface):
 215 
 216     # Private addresses
 217     private_addresses = []
 218     for ip in interface.private_ip_addresses:
 219         private_addresses.append({'private_ip_address': ip.private_ip_address, 'primary_address': ip.primary})
 220 
 221     interface_info = {'id': interface.id,
 222                       'subnet_id': interface.subnet_id,
 223                       'vpc_id': interface.vpc_id,
 224                       'description': interface.description,
 225                       'owner_id': interface.owner_id,
 226                       'status': interface.status,
 227                       'mac_address': interface.mac_address,
 228                       'private_ip_address': interface.private_ip_address,
 229                       'source_dest_check': interface.source_dest_check,
 230                       'groups': dict((group.id, group.name) for group in interface.groups),
 231                       'private_ip_addresses': private_addresses
 232                       }
 233 
 234     if hasattr(interface, 'publicDnsName'):
 235         interface_info['association'] = {'public_ip_address': interface.publicIp,
 236                                          'public_dns_name': interface.publicDnsName,
 237                                          'ip_owner_id': interface.ipOwnerId
 238                                          }
 239 
 240     if interface.attachment is not None:
 241         interface_info['attachment'] = {'attachment_id': interface.attachment.id,
 242                                         'instance_id': interface.attachment.instance_id,
 243                                         'device_index': interface.attachment.device_index,
 244                                         'status': interface.attachment.status,
 245                                         'attach_time': interface.attachment.attach_time,
 246                                         'delete_on_termination': interface.attachment.delete_on_termination,
 247                                         }
 248 
 249     return interface_info
