 105 # getter subroutines
 106 def get_config_state(configfile):
 107     lines = get_file_lines(configfile, strip=False)
 108 
 109     for line in lines:
 110         stateline = re.match(r'^SELINUX=.*$', line)
 111         if stateline:
 112             return line.split('=')[1].strip()
 113 
 114 
 115 def get_config_policy(configfile):
 116     lines = get_file_lines(configfile, strip=False)
 117 
 118     for line in lines:
 119         stateline = re.match(r'^SELINUXTYPE=.*$', line)
 120         if stateline:
 121             return line.split('=')[1].strip()

 124 # setter subroutines
 125 def set_config_state(module, state, configfile):
 126     # SELINUX=permissive
 127     # edit config file with state value
 128     stateline = 'SELINUX=%s' % state
 129     lines = get_file_lines(configfile, strip=False)
 130 
 131     tmpfd, tmpfile = tempfile.mkstemp()
 132 
 133     with open(tmpfile, "w") as write_file:
 134         for line in lines:
 135             write_file.write(re.sub(r'^SELINUX=.*', stateline, line) + '\n')
 136 
 137     module.atomic_move(tmpfile, configfile)
 138 
 139 
 140 def set_state(module, state):
 141     if state == 'enforcing':
 142         selinux.security_setenforce(1)
 143     elif state == 'permissive':
 144         selinux.security_setenforce(0)
 145     elif state == 'disabled':
 146         pass
 147     else:
 148         msg = 'trying to set invalid runtime state %s' % state
 149         module.fail_json(msg=msg)
 150 
 151 
 152 def set_config_policy(module, policy, configfile):
 153     if not os.path.exists('/etc/selinux/%s/policy' % policy):
 154         module.fail_json(msg='Policy %s does not exist in /etc/selinux/' % policy)
 155 
 156     # edit config file with state value
 157     # SELINUXTYPE=targeted
 158     policyline = 'SELINUXTYPE=%s' % policy
 159     lines = get_file_lines(configfile, strip=False)
 160 
 161     tmpfd, tmpfile = tempfile.mkstemp()
 162 
 163     with open(tmpfile, "w") as write_file:
 164         for line in lines:
 165             write_file.write(re.sub(r'^SELINUXTYPE=.*', policyline, line) + '\n')
 166 
 167     module.atomic_move(tmpfile, configfile)

 170 def main():
 171     module = AnsibleModule(
 172         argument_spec=dict(
 173             policy=dict(type='str'),
 174             state=dict(type='str', required='True', choices=['enforcing', 'permissive', 'disabled']),
 175             configfile=dict(type='str', default='/etc/selinux/config', aliases=['conf', 'file']),
 176         ),
 177         supports_check_mode=True,
 178     )
 179 
 180     if not HAS_SELINUX:
 181         module.fail_json(msg=missing_required_lib('libselinux-python'), exception=SELINUX_IMP_ERR)
 182 
 183     # global vars
 184     changed = False
 185     msgs = []
 186     configfile = module.params['configfile']
 187     policy = module.params['policy']
 188     state = module.params['state']
 189     runtime_enabled = selinux.is_selinux_enabled()
 190     runtime_policy = selinux.selinux_getpolicytype()[1]
 191     runtime_state = 'disabled'
 192     reboot_required = False
 193 
 194     if runtime_enabled:
 195         # enabled means 'enforcing' or 'permissive'
 196         if selinux.security_getenforce():
 197             runtime_state = 'enforcing'
 198         else:
 199             runtime_state = 'permissive'
 200 
 201     if not os.path.isfile(configfile):
 202         module.fail_json(msg="Unable to find file {0}".format(configfile),
 203                          details="Please install SELinux-policy package, "
 204                                  "if this package is not installed previously.")
 205 
 206     config_policy = get_config_policy(configfile)
 207     config_state = get_config_state(configfile)
 208 
 209     # check to see if policy is set if state is not 'disabled'
 210     if state != 'disabled':
 211         if not policy:
 212             module.fail_json(msg="Policy is required if state is not 'disabled'")
 213     else:
 214         if not policy:
 215             policy = config_policy
 216 
 217     # check changed values and run changes
 218     if policy != runtime_policy:
 219         if module.check_mode:
 220             module.exit_json(changed=True)
 221         # cannot change runtime policy
 222         msgs.append("Running SELinux policy changed from '%s' to '%s'" % (runtime_policy, policy))
 223         changed = True
 224 
 225     if policy != config_policy:
 226         if module.check_mode:
 227             module.exit_json(changed=True)
 228         set_config_policy(module, policy, configfile)
 229         msgs.append("SELinux policy configuration in '%s' changed from '%s' to '%s'" % (configfile, config_policy, policy))
 230         changed = True
 231 
 232     if state != runtime_state:
 233         if runtime_enabled:
 234             if state == 'disabled':
 235                 if runtime_state != 'permissive':
 236                     # Temporarily set state to permissive
 237                     if not module.check_mode:
 238                         set_state(module, 'permissive')
 239                     module.warn("SELinux state temporarily changed from '%s' to 'permissive'. State change will take effect next reboot." % (runtime_state))
 240                     changed = True
 241                 else:
 242                     module.warn('SELinux state change will take effect next reboot')
 243                 reboot_required = True
 244             else:
 245                 if not module.check_mode:
 246                     set_state(module, state)
 247                 msgs.append("SELinux state changed from '%s' to '%s'" % (runtime_state, state))
 248 
 249                 # Only report changes if the file is changed.
 250                 # This prevents the task from reporting changes every time the task is run.
 251                 changed = True
 252         else:
 253             module.warn("Reboot is required to set SELinux state to '%s'" % state)
 254             reboot_required = True
 255 
 256     if state != config_state:
 257         if not module.check_mode:
 258             set_config_state(module, state, configfile)
 259         msgs.append("Config SELinux state changed from '%s' to '%s'" % (config_state, state))
 260         changed = True
 261 
 262     module.exit_json(changed=changed, msg=', '.join(msgs), configfile=configfile, policy=policy, state=state, reboot_required=reboot_required)
 263 
 264 
 265 if __name__ == '__main__':
 266     main()
