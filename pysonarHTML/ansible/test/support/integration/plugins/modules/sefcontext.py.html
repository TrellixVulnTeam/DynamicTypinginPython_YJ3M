 155 def get_runtime_status(ignore_selinux_state=False):
 156     return True if ignore_selinux_state is True else selinux.is_selinux_enabled()

 159 def semanage_fcontext_exists(sefcontext, target, ftype):
 160     ''' Get the SELinux file context mapping definition from policy. Return None if it does not exist. '''
 161 
 162     # Beware that records comprise of a string representation of the file_type
 163     record = (target, option_to_file_type_str[ftype])
 164     records = sefcontext.get_all()
 165     try:
 166         return records[record]
 167     except KeyError:
 168         return None

 171 def semanage_fcontext_modify(module, result, target, ftype, setype, do_reload, serange, seuser, sestore=''):
 172     ''' Add or modify SELinux file context mapping definition to the policy. '''
 173 
 174     changed = False
 175     prepared_diff = ''
 176 
 177     try:
 178         sefcontext = seobject.fcontextRecords(sestore)
 179         sefcontext.set_reload(do_reload)
 180         exists = semanage_fcontext_exists(sefcontext, target, ftype)
 181         if exists:
 182             # Modify existing entry
 183             orig_seuser, orig_serole, orig_setype, orig_serange = exists
 184 
 185             if seuser is None:
 186                 seuser = orig_seuser
 187             if serange is None:
 188                 serange = orig_serange
 189 
 190             if setype != orig_setype or seuser != orig_seuser or serange != orig_serange:
 191                 if not module.check_mode:
 192                     sefcontext.modify(target, setype, ftype, serange, seuser)
 193                 changed = True
 194 
 195                 if module._diff:
 196                     prepared_diff += '# Change to semanage file context mappings\n'
 197                     prepared_diff += '-%s      %s      %s:%s:%s:%s\n' % (target, ftype, orig_seuser, orig_serole, orig_setype, orig_serange)
 198                     prepared_diff += '+%s      %s      %s:%s:%s:%s\n' % (target, ftype, seuser, orig_serole, setype, serange)
 199         else:
 200             # Add missing entry
 201             if seuser is None:
 202                 seuser = 'system_u'
 203             if serange is None:
 204                 serange = 's0'
 205 
 206             if not module.check_mode:
 207                 sefcontext.add(target, setype, ftype, serange, seuser)
 208             changed = True
 209 
 210             if module._diff:
 211                 prepared_diff += '# Addition to semanage file context mappings\n'
 212                 prepared_diff += '+%s      %s      %s:%s:%s:%s\n' % (target, ftype, seuser, 'object_r', setype, serange)
 213 
 214     except Exception as e:
 215         module.fail_json(msg="%s: %s\n" % (e.__class__.__name__, to_native(e)))
 216 
 217     if module._diff and prepared_diff:
 218         result['diff'] = dict(prepared=prepared_diff)
 219 
 220     module.exit_json(changed=changed, seuser=seuser, serange=serange, **result)

 223 def semanage_fcontext_delete(module, result, target, ftype, do_reload, sestore=''):
 224     ''' Delete SELinux file context mapping definition from the policy. '''
 225 
 226     changed = False
 227     prepared_diff = ''
 228 
 229     try:
 230         sefcontext = seobject.fcontextRecords(sestore)
 231         sefcontext.set_reload(do_reload)
 232         exists = semanage_fcontext_exists(sefcontext, target, ftype)
 233         if exists:
 234             # Remove existing entry
 235             orig_seuser, orig_serole, orig_setype, orig_serange = exists
 236 
 237             if not module.check_mode:
 238                 sefcontext.delete(target, ftype)
 239             changed = True
 240 
 241             if module._diff:
 242                 prepared_diff += '# Deletion to semanage file context mappings\n'
 243                 prepared_diff += '-%s      %s      %s:%s:%s:%s\n' % (target, ftype, exists[0], exists[1], exists[2], exists[3])
 244 
 245     except Exception as e:
 246         module.fail_json(msg="%s: %s\n" % (e.__class__.__name__, to_native(e)))
 247 
 248     if module._diff and prepared_diff:
 249         result['diff'] = dict(prepared=prepared_diff)
 250 
 251     module.exit_json(changed=changed, **result)

 254 def main():
 255     module = AnsibleModule(
 256         argument_spec=dict(
 257             ignore_selinux_state=dict(type='bool', default=False),
 258             target=dict(type='str', required=True, aliases=['path']),
 259             ftype=dict(type='str', default='a', choices=option_to_file_type_str.keys()),
 260             setype=dict(type='str', required=True),
 261             seuser=dict(type='str'),
 262             selevel=dict(type='str', aliases=['serange']),
 263             state=dict(type='str', default='present', choices=['absent', 'present']),
 264             reload=dict(type='bool', default=True),
 265         ),
 266         supports_check_mode=True,
 267     )
 268     if not HAVE_SELINUX:
 269         module.fail_json(msg=missing_required_lib("libselinux-python"), exception=SELINUX_IMP_ERR)
 270 
 271     if not HAVE_SEOBJECT:
 272         module.fail_json(msg=missing_required_lib("policycoreutils-python"), exception=SEOBJECT_IMP_ERR)
 273 
 274     ignore_selinux_state = module.params['ignore_selinux_state']
 275 
 276     if not get_runtime_status(ignore_selinux_state):
 277         module.fail_json(msg="SELinux is disabled on this host.")
 278 
 279     target = module.params['target']
 280     ftype = module.params['ftype']
 281     setype = module.params['setype']
 282     seuser = module.params['seuser']
 283     serange = module.params['selevel']
 284     state = module.params['state']
 285     do_reload = module.params['reload']
 286 
 287     result = dict(target=target, ftype=ftype, setype=setype, state=state)
 288 
 289     if state == 'present':
 290         semanage_fcontext_modify(module, result, target, ftype, setype, do_reload, serange, seuser)
 291     elif state == 'absent':
 292         semanage_fcontext_delete(module, result, target, ftype, do_reload)
 293     else:
 294         module.fail_json(msg='Invalid value of argument "state": {0}'.format(state))
 295 
 296 
 297 if __name__ == '__main__':
 298     main()
