  90 import re
  91 import sys
  92 import operator
  93 
  94 HAS_DISTUTILS = False
  95 try:
  96     import pkg_resources
  97     from distutils.version import LooseVersion
  98     HAS_DISTUTILS = True
  99 except ImportError:
 100     pass
 101 
 102 from ansible.module_utils.basic import AnsibleModule
 103 
 104 operations = {
 105     '<=': operator.le,
 106     '>=': operator.ge,
 107     '<': operator.lt,
 108     '>': operator.gt,
 109     '==': operator.eq,
 110 }
 111 
 112 
 113 def main():
 114     module = AnsibleModule(
 115         argument_spec=dict(
 116             dependencies=dict(type='list')
 117         ),
 118         supports_check_mode=True,
 119     )
 120     if module._name == 'python_requirements_facts':
 121         module.deprecate("The 'python_requirements_facts' module has been renamed to 'python_requirements_info'",
 122                          version='2.13', collection_name='ansible.builtin')
 123     if not HAS_DISTUTILS:
 124         module.fail_json(
 125             msg='Could not import "distutils" and "pkg_resources" libraries to introspect python environment.',
 126             python=sys.executable,
 127             python_version=sys.version,
 128             python_system_path=sys.path,
 129         )
 130     pkg_dep_re = re.compile(r'(^[a-zA-Z][a-zA-Z0-9_-]+)(==|[><]=?)?([0-9.]+)?$')
 131 
 132     results = dict(
 133         not_found=[],
 134         mismatched={},
 135         valid={},
 136     )
 137 
 138     for dep in (module.params.get('dependencies') or []):
 139         match = pkg_dep_re.match(dep)
 140         if match is None:
 141             module.fail_json(msg="Failed to parse version requirement '{0}'. Must be formatted like 'ansible>2.6'".format(dep))
 142         pkg, op, version = match.groups()
 143         if op is not None and op not in operations:
 144             module.fail_json(msg="Failed to parse version requirement '{0}'. Operator must be one of >, <, <=, >=, or ==".format(dep))
 145         try:
 146             existing = pkg_resources.get_distribution(pkg).version
 147         except pkg_resources.DistributionNotFound:
 148             # not there
 149             results['not_found'].append(pkg)
 150             continue
 151         if op is None and version is None:
 152             results['valid'][pkg] = {
 153                 'installed': existing,
 154                 'desired': None,
 155             }
 156         elif operations[op](LooseVersion(existing), LooseVersion(version)):
 157             results['valid'][pkg] = {
 158                 'installed': existing,
 159                 'desired': dep,
 160             }
 161         else:
 162             results['mismatched'] = {
 163                 'installed': existing,
 164                 'desired': dep,
 165             }
 166 
 167     module.exit_json(
 168         python=sys.executable,
 169         python_version=sys.version,
 170         python_system_path=sys.path,
 171         **results
 172     )
 173 
 174 
 175 if __name__ == '__main__':
 176     main()
