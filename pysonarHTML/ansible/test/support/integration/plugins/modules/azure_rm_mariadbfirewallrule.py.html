 142     def exec_module(self, **kwargs):
 143         """Main module execution method"""
 144 
 145         for key in list(self.module_arg_spec.keys()):
 146             if hasattr(self, key):
 147                 setattr(self, key, kwargs[key])
 148 
 149         old_response = None
 150         response = None
 151 
 152         resource_group = self.get_resource_group(self.resource_group)
 153 
 154         old_response = self.get_firewallrule()
 155 
 156         if not old_response:
 157             self.log("MariaDB firewall rule instance doesn't exist")
 158             if self.state == 'absent':
 159                 self.log("Old instance didn't exist")
 160             else:
 161                 self.to_do = Actions.Create
 162         else:
 163             self.log("MariaDB firewall rule instance already exists")
 164             if self.state == 'absent':
 165                 self.to_do = Actions.Delete
 166             elif self.state == 'present':
 167                 self.log("Need to check if MariaDB firewall rule instance has to be deleted or may be updated")
 168                 if (self.start_ip_address is not None) and (self.start_ip_address != old_response['start_ip_address']):
 169                     self.to_do = Actions.Update
 170                 if (self.end_ip_address is not None) and (self.end_ip_address != old_response['end_ip_address']):
 171                     self.to_do = Actions.Update
 172 
 173         if (self.to_do == Actions.Create) or (self.to_do == Actions.Update):
 174             self.log("Need to Create / Update the MariaDB firewall rule instance")
 175 
 176             if self.check_mode:
 177                 self.results['changed'] = True
 178                 return self.results
 179 
 180             response = self.create_update_firewallrule()
 181 
 182             if not old_response:
 183                 self.results['changed'] = True
 184             else:
 185                 self.results['changed'] = old_response.__ne__(response)
 186             self.log("Creation / Update done")
 187         elif self.to_do == Actions.Delete:
 188             self.log("MariaDB firewall rule instance deleted")
 189             self.results['changed'] = True
 190 
 191             if self.check_mode:
 192                 return self.results
 193 
 194             self.delete_firewallrule()
 195             # make sure instance is actually deleted, for some Azure resources, instance is hanging around
 196             # for some time after deletion -- this should be really fixed in Azure
 197             while self.get_firewallrule():
 198                 time.sleep(20)
 199         else:
 200             self.log("MariaDB firewall rule instance unchanged")
 201             self.results['changed'] = False
 202             response = old_response
 203 
 204         if response:
 205             self.results["id"] = response["id"]
 206 
 207         return self.results

 209     def create_update_firewallrule(self):
 210         '''
 211         Creates or updates MariaDB firewall rule with the specified configuration.
 212 
 213         :return: deserialized MariaDB firewall rule instance state dictionary
 214         '''
 215         self.log("Creating / Updating the MariaDB firewall rule instance {0}".format(self.name))
 216 
 217         try:
 218             response = self.mariadb_client.firewall_rules.create_or_update(resource_group_name=self.resource_group,
 219                                                                            server_name=self.server_name,
 220                                                                            firewall_rule_name=self.name,
 221                                                                            start_ip_address=self.start_ip_address,
 222                                                                            end_ip_address=self.end_ip_address)
 223             if isinstance(response, LROPoller):
 224                 response = self.get_poller_result(response)
 225 
 226         except CloudError as exc:
 227             self.log('Error attempting to create the MariaDB firewall rule instance.')
 228             self.fail("Error creating the MariaDB firewall rule instance: {0}".format(str(exc)))
 229         return response.as_dict()
 230 
 231     def delete_firewallrule(self):
 232         '''
 233         Deletes specified MariaDB firewall rule instance in the specified subscription and resource group.
 234 
 235         :return: True
 236         '''
 237         self.log("Deleting the MariaDB firewall rule instance {0}".format(self.name))
 238         try:
 239             response = self.mariadb_client.firewall_rules.delete(resource_group_name=self.resource_group,
 240                                                                  server_name=self.server_name,
 241                                                                  firewall_rule_name=self.name)
 242         except CloudError as e:
 243             self.log('Error attempting to delete the MariaDB firewall rule instance.')
 244             self.fail("Error deleting the MariaDB firewall rule instance: {0}".format(str(e)))
 245 
 246         return True
 247 
 248     def get_firewallrule(self):
 249         '''
 250         Gets the properties of the specified MariaDB firewall rule.
 251 
 252         :return: deserialized MariaDB firewall rule instance state dictionary
 253         '''
 254         self.log("Checking if the MariaDB firewall rule instance {0} is present".format(self.name))
 255         found = False
 256         try:
 257             response = self.mariadb_client.firewall_rules.get(resource_group_name=self.resource_group,
 258                                                               server_name=self.server_name,
 259                                                               firewall_rule_name=self.name)
 260             found = True
 261             self.log("Response : {0}".format(response))
 262             self.log("MariaDB firewall rule instance : {0} found".format(response.name))
 263         except CloudError as e:
 264             self.log('Did not find the MariaDB firewall rule instance.')
 265         if found is True:
 266             return response.as_dict()
 267 
 268         return False
