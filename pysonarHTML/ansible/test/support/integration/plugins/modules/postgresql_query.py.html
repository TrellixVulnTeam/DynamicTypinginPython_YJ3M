 203 def list_to_pg_array(elem):
 204     """Convert the passed list to PostgreSQL array
 205     represented as a string.
 206 
 207     Args:
 208         elem (list): List that needs to be converted.
 209 
 210     Returns:
 211         elem (str): String representation of PostgreSQL array.
 212     """
 213     elem = str(elem).strip('[]')
 214     elem = '{' + elem + '}'
 215     return elem
 216 
 217 
 218 def convert_elements_to_pg_arrays(obj):
 219     """Convert list elements of the passed object
 220     to PostgreSQL arrays represented as strings.
 221 
 222     Args:
 223         obj (dict or list): Object whose elements need to be converted.
 224 
 225     Returns:
 226         obj (dict or list): Object with converted elements.
 227     """
 228     if isinstance(obj, dict):
 229         for (key, elem) in iteritems(obj):
 230             if isinstance(elem, list):
 231                 obj[key] = list_to_pg_array(elem)
 232 
 233     elif isinstance(obj, list):
 234         for i, elem in enumerate(obj):
 235             if isinstance(elem, list):
 236                 obj[i] = list_to_pg_array(elem)
 237 
 238     return obj

 241 def main():
 242     argument_spec = postgres_common_argument_spec()
 243     argument_spec.update(
 244         query=dict(type='str'),
 245         db=dict(type='str', aliases=['login_db']),
 246         positional_args=dict(type='list', elements='raw'),
 247         named_args=dict(type='dict'),
 248         session_role=dict(type='str'),
 249         path_to_script=dict(type='path'),
 250         autocommit=dict(type='bool', default=False),
 251         encoding=dict(type='str'),
 252     )
 253 
 254     module = AnsibleModule(
 255         argument_spec=argument_spec,
 256         mutually_exclusive=(('positional_args', 'named_args'),),
 257         supports_check_mode=True,
 258     )
 259 
 260     query = module.params["query"]
 261     positional_args = module.params["positional_args"]
 262     named_args = module.params["named_args"]
 263     path_to_script = module.params["path_to_script"]
 264     autocommit = module.params["autocommit"]
 265     encoding = module.params["encoding"]
 266 
 267     if autocommit and module.check_mode:
 268         module.fail_json(msg="Using autocommit is mutually exclusive with check_mode")
 269 
 270     if path_to_script and query:
 271         module.fail_json(msg="path_to_script is mutually exclusive with query")
 272 
 273     if positional_args:
 274         positional_args = convert_elements_to_pg_arrays(positional_args)
 275 
 276     elif named_args:
 277         named_args = convert_elements_to_pg_arrays(named_args)
 278 
 279     if path_to_script:
 280         try:
 281             with open(path_to_script, 'rb') as f:
 282                 query = to_native(f.read())
 283         except Exception as e:
 284             module.fail_json(msg="Cannot read file '%s' : %s" % (path_to_script, to_native(e)))
 285 
 286     conn_params = get_conn_params(module, module.params)
 287     db_connection = connect_to_db(module, conn_params, autocommit=autocommit)
 288     if encoding is not None:
 289         db_connection.set_client_encoding(encoding)
 290     cursor = db_connection.cursor(cursor_factory=DictCursor)
 291 
 292     # Prepare args:
 293     if module.params.get("positional_args"):
 294         arguments = module.params["positional_args"]
 295     elif module.params.get("named_args"):
 296         arguments = module.params["named_args"]
 297     else:
 298         arguments = None
 299 
 300     # Set defaults:
 301     changed = False
 302 
 303     # Execute query:
 304     try:
 305         cursor.execute(query, arguments)
 306     except Exception as e:
 307         if not autocommit:
 308             db_connection.rollback()
 309 
 310         cursor.close()
 311         db_connection.close()
 312         module.fail_json(msg="Cannot execute SQL '%s' %s: %s" % (query, arguments, to_native(e)))
 313 
 314     statusmessage = cursor.statusmessage
 315     rowcount = cursor.rowcount
 316 
 317     try:
 318         query_result = [dict(row) for row in cursor.fetchall()]
 319     except Psycopg2ProgrammingError as e:
 320         if to_native(e) == 'no results to fetch':
 321             query_result = {}
 322 
 323     except Exception as e:
 324         module.fail_json(msg="Cannot fetch rows from cursor: %s" % to_native(e))
 325 
 326     if 'SELECT' not in statusmessage:
 327         if 'UPDATE' in statusmessage or 'INSERT' in statusmessage or 'DELETE' in statusmessage:
 328             s = statusmessage.split()
 329             if len(s) == 3:
 330                 if statusmessage.split()[2] != '0':
 331                     changed = True
 332 
 333             elif len(s) == 2:
 334                 if statusmessage.split()[1] != '0':
 335                     changed = True
 336 
 337             else:
 338                 changed = True
 339 
 340         else:
 341             changed = True
 342 
 343     if module.check_mode:
 344         db_connection.rollback()
 345     else:
 346         if not autocommit:
 347             db_connection.commit()
 348 
 349     kw = dict(
 350         changed=changed,
 351         query=cursor.query,
 352         statusmessage=statusmessage,
 353         query_result=query_result,
 354         rowcount=rowcount if rowcount >= 0 else 0,
 355     )
 356 
 357     cursor.close()
 358     db_connection.close()
 359 
 360     module.exit_json(**kw)
 361 
 362 
 363 if __name__ == '__main__':
 364     main()
