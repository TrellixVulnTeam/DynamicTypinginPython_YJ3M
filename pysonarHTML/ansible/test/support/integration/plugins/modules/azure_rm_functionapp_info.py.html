 111 class AzureRMFunctionAppInfo(AzureRMModuleBase):
 112     def __init__(self):
 113 
 114         self.module_arg_spec = dict(
 115             name=dict(type='str'),
 116             resource_group=dict(type='str', aliases=['resource_group_name']),
 117             tags=dict(type='list'),
 118         )
 119 
 120         self.results = dict(
 121             changed=False,
 122             ansible_info=dict(azure_functionapps=[])
 123         )
 124 
 125         self.name = None
 126         self.resource_group = None
 127         self.tags = None
 128 
 129         super(AzureRMFunctionAppInfo, self).__init__(
 130             self.module_arg_spec,
 131             supports_tags=False,
 132             facts_module=True
 133         )
 134 
 135     def exec_module(self, **kwargs):
 136 
 137         is_old_facts = self.module._name == 'azure_rm_functionapp_facts'
 138         if is_old_facts:
 139             self.module.deprecate("The 'azure_rm_functionapp_facts' module has been renamed to 'azure_rm_functionapp_info'",
 140                                   version='2.13', collection_name='ansible.builtin')
 141 
 142         for key in self.module_arg_spec:
 143             setattr(self, key, kwargs[key])
 144 
 145         if self.name and not self.resource_group:
 146             self.fail("Parameter error: resource group required when filtering by name.")
 147 
 148         if self.name:
 149             self.results['ansible_info']['azure_functionapps'] = self.get_functionapp()
 150         elif self.resource_group:
 151             self.results['ansible_info']['azure_functionapps'] = self.list_resource_group()
 152         else:
 153             self.results['ansible_info']['azure_functionapps'] = self.list_all()
 154 
 155         return self.results
 156 
 157     def get_functionapp(self):
 158         self.log('Get properties for Function App {0}'.format(self.name))
 159         function_app = None
 160         result = []
 161 
 162         try:
 163             function_app = self.web_client.web_apps.get(
 164                 self.resource_group,
 165                 self.name
 166             )
 167         except CloudError:
 168             pass
 169 
 170         if function_app and self.has_tags(function_app.tags, self.tags):
 171             result = function_app.as_dict()
 172 
 173         return [result]
 174 
 175     def list_resource_group(self):
 176         self.log('List items')
 177         try:
 178             response = self.web_client.web_apps.list_by_resource_group(self.resource_group)
 179         except Exception as exc:
 180             self.fail("Error listing for resource group {0} - {1}".format(self.resource_group, str(exc)))
 181 
 182         results = []
 183         for item in response:
 184             if self.has_tags(item.tags, self.tags):
 185                 results.append(item.as_dict())
 186         return results
 187 
 188     def list_all(self):
 189         self.log('List all items')
 190         try:
 191             response = self.web_client.web_apps.list_by_resource_group(self.resource_group)
 192         except Exception as exc:
 193             self.fail("Error listing all items - {0}".format(str(exc)))
 194 
 195         results = []
 196         for item in response:
 197             if self.has_tags(item.tags, self.tags):
 198                 results.append(item.as_dict())
 199         return results
