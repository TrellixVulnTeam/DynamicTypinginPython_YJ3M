 230 def get_subnet_info(subnet):
 231     if 'Subnets' in subnet:
 232         return [get_subnet_info(s) for s in subnet['Subnets']]
 233     elif 'Subnet' in subnet:
 234         subnet = camel_dict_to_snake_dict(subnet['Subnet'])
 235     else:
 236         subnet = camel_dict_to_snake_dict(subnet)
 237 
 238     if 'tags' in subnet:
 239         subnet['tags'] = boto3_tag_list_to_ansible_dict(subnet['tags'])
 240     else:
 241         subnet['tags'] = dict()
 242 
 243     if 'subnet_id' in subnet:
 244         subnet['id'] = subnet['subnet_id']
 245         del subnet['subnet_id']
 246 
 247     subnet['ipv6_cidr_block'] = ''
 248     subnet['ipv6_association_id'] = ''
 249     ipv6set = subnet.get('ipv6_cidr_block_association_set')
 250     if ipv6set:
 251         for item in ipv6set:
 252             if item.get('ipv6_cidr_block_state', {}).get('state') in ('associated', 'associating'):
 253                 subnet['ipv6_cidr_block'] = item['ipv6_cidr_block']
 254                 subnet['ipv6_association_id'] = item['association_id']
 255 
 256     return subnet

 264 def waiter_params(module, params, start_time):

 282 def create_subnet(conn, module, vpc_id, cidr, ipv6_cidr=None, az=None, start_time=None):
 283     wait = module.params['wait']
 284     wait_timeout = module.params['wait_timeout']
 285 
 286     params = dict(VpcId=vpc_id,
 287                   CidrBlock=cidr)
 288 
 289     if ipv6_cidr:
 290         params['Ipv6CidrBlock'] = ipv6_cidr
 291 
 292     if az:
 293         params['AvailabilityZone'] = az
 294 
 295     try:
 296         subnet = get_subnet_info(conn.create_subnet(**params))
 297     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 298         module.fail_json_aws(e, msg="Couldn't create subnet")
 299 
 300     # Sometimes AWS takes its time to create a subnet and so using
 301     # new subnets's id to do things like create tags results in
 302     # exception.
 303     if wait and subnet.get('state') != 'available':
 304         handle_waiter(conn, module, 'subnet_exists', {'SubnetIds': [subnet['id']]}, start_time)
 305         try:
 306             conn.get_waiter('subnet_available').wait(
 307                 **waiter_params(module, {'SubnetIds': [subnet['id']]}, start_time)
 308             )
 309             subnet['state'] = 'available'
 310         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 311             module.fail_json_aws(e, "Create subnet action timed out waiting for subnet to become available")
 312 
 313     return subnet

 316 def ensure_tags(conn, module, subnet, tags, purge_tags, start_time):
 317     changed = False
 318 
 319     filters = ansible_dict_to_boto3_filter_list({'resource-id': subnet['id'], 'resource-type': 'subnet'})
 320     try:
 321         cur_tags = conn.describe_tags(Filters=filters)
 322     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 323         module.fail_json_aws(e, msg="Couldn't describe tags")
 324 
 325     to_update, to_delete = compare_aws_tags(boto3_tag_list_to_ansible_dict(cur_tags.get('Tags')), tags, purge_tags)
 326 
 327     if to_update:
 328         try:
 329             if not module.check_mode:
 330                 AWSRetry.exponential_backoff(
 331                     catch_extra_error_codes=['InvalidSubnetID.NotFound']
 332                 )(conn.create_tags)(
 333                     Resources=[subnet['id']],
 334                     Tags=ansible_dict_to_boto3_tag_list(to_update)
 335                 )
 336 
 337             changed = True
 338         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 339             module.fail_json_aws(e, msg="Couldn't create tags")
 340 
 341     if to_delete:
 342         try:
 343             if not module.check_mode:
 344                 tags_list = []
 345                 for key in to_delete:
 346                     tags_list.append({'Key': key})
 347 
 348                 AWSRetry.exponential_backoff(
 349                     catch_extra_error_codes=['InvalidSubnetID.NotFound']
 350                 )(conn.delete_tags)(Resources=[subnet['id']], Tags=tags_list)
 351 
 352             changed = True
 353         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 354             module.fail_json_aws(e, msg="Couldn't delete tags")
 355 
 356     if module.params['wait'] and not module.check_mode:
 357         # Wait for tags to be updated
 358         filters = [{'Name': 'tag:{0}'.format(k), 'Values': [v]} for k, v in tags.items()]
 359         handle_waiter(conn, module, 'subnet_exists',
 360                       {'SubnetIds': [subnet['id']], 'Filters': filters}, start_time)
 361 
 362     return changed

 383 def disassociate_ipv6_cidr(conn, module, subnet, start_time):
 384     if subnet.get('assign_ipv6_address_on_creation'):
 385         ensure_assign_ipv6_on_create(conn, module, subnet, False, False, start_time)
 386 
 387     try:
 388         conn.disassociate_subnet_cidr_block(AssociationId=subnet['ipv6_association_id'])
 389     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 390         module.fail_json_aws(e, msg="Couldn't disassociate ipv6 cidr block id {0} from subnet {1}"
 391                              .format(subnet['ipv6_association_id'], subnet['id']))
 392 
 393     # Wait for cidr block to be disassociated
 394     if module.params['wait']:
 395         filters = ansible_dict_to_boto3_filter_list(
 396             {'ipv6-cidr-block-association.state': ['disassociated'],
 397              'vpc-id': subnet['vpc_id']}
 398         )
 399         handle_waiter(conn, module, 'subnet_exists',
 400                       {'SubnetIds': [subnet['id']], 'Filters': filters}, start_time)

 403 def ensure_ipv6_cidr_block(conn, module, subnet, ipv6_cidr, check_mode, start_time):
 404     wait = module.params['wait']
 405     changed = False
 406 
 407     if subnet['ipv6_association_id'] and not ipv6_cidr:
 408         if not check_mode:
 409             disassociate_ipv6_cidr(conn, module, subnet, start_time)
 410         changed = True
 411 
 412     if ipv6_cidr:
 413         filters = ansible_dict_to_boto3_filter_list({'ipv6-cidr-block-association.ipv6-cidr-block': ipv6_cidr,
 414                                                      'vpc-id': subnet['vpc_id']})
 415 
 416         try:
 417             check_subnets = get_subnet_info(describe_subnets_with_backoff(conn, Filters=filters))
 418         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 419             module.fail_json_aws(e, msg="Couldn't get subnet info")
 420 
 421         if check_subnets and check_subnets[0]['ipv6_cidr_block']:
 422             module.fail_json(msg="The IPv6 CIDR '{0}' conflicts with another subnet".format(ipv6_cidr))
 423 
 424         if subnet['ipv6_association_id']:
 425             if not check_mode:
 426                 disassociate_ipv6_cidr(conn, module, subnet, start_time)
 427             changed = True
 428 
 429         try:
 430             if not check_mode:
 431                 associate_resp = conn.associate_subnet_cidr_block(SubnetId=subnet['id'], Ipv6CidrBlock=ipv6_cidr)
 432             changed = True
 433         except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 434             module.fail_json_aws(e, msg="Couldn't associate ipv6 cidr {0} to {1}".format(ipv6_cidr, subnet['id']))
 435         else:
 436             if not check_mode and wait:
 437                 filters = ansible_dict_to_boto3_filter_list(
 438                     {'ipv6-cidr-block-association.state': ['associated'],
 439                      'vpc-id': subnet['vpc_id']}
 440                 )
 441                 handle_waiter(conn, module, 'subnet_exists',
 442                               {'SubnetIds': [subnet['id']], 'Filters': filters}, start_time)
 443 
 444         if associate_resp.get('Ipv6CidrBlockAssociation', {}).get('AssociationId'):
 445             subnet['ipv6_association_id'] = associate_resp['Ipv6CidrBlockAssociation']['AssociationId']
 446             subnet['ipv6_cidr_block'] = associate_resp['Ipv6CidrBlockAssociation']['Ipv6CidrBlock']
 447             if subnet['ipv6_cidr_block_association_set']:
 448                 subnet['ipv6_cidr_block_association_set'][0] = camel_dict_to_snake_dict(associate_resp['Ipv6CidrBlockAssociation'])
 449             else:
 450                 subnet['ipv6_cidr_block_association_set'].append(camel_dict_to_snake_dict(associate_resp['Ipv6CidrBlockAssociation']))
 451 
 452     return changed

 455 def get_matching_subnet(conn, module, vpc_id, cidr):
 456     filters = ansible_dict_to_boto3_filter_list({'vpc-id': vpc_id, 'cidr-block': cidr})
 457     try:
 458         subnets = get_subnet_info(describe_subnets_with_backoff(conn, Filters=filters))
 459     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
 460         module.fail_json_aws(e, msg="Couldn't get matching subnet")
 461 
 462     if subnets:
 463         return subnets[0]
 464 
 465     return None

 468 def ensure_subnet_present(conn, module):
 469     subnet = get_matching_subnet(conn, module, module.params['vpc_id'], module.params['cidr'])
 470     changed = False
 471 
 472     # Initialize start so max time does not exceed the specified wait_timeout for multiple operations
 473     start_time = time.time()
 474 
 475     if subnet is None:
 476         if not module.check_mode:
 477             subnet = create_subnet(conn, module, module.params['vpc_id'], module.params['cidr'],
 478                                    ipv6_cidr=module.params['ipv6_cidr'], az=module.params['az'], start_time=start_time)
 479         changed = True
 480         # Subnet will be None when check_mode is true
 481         if subnet is None:
 482             return {
 483                 'changed': changed,
 484                 'subnet': {}
 485             }
 486     if module.params['wait']:
 487         handle_waiter(conn, module, 'subnet_exists', {'SubnetIds': [subnet['id']]}, start_time)
 488 
 489     if module.params['ipv6_cidr'] != subnet.get('ipv6_cidr_block'):
 490         if ensure_ipv6_cidr_block(conn, module, subnet, module.params['ipv6_cidr'], module.check_mode, start_time):
 491             changed = True
 492 
 493     if module.params['map_public'] != subnet['map_public_ip_on_launch']:
 494         ensure_map_public(conn, module, subnet, module.params['map_public'], module.check_mode, start_time)
 495         changed = True
 496 
 497     if module.params['assign_instances_ipv6'] != subnet.get('assign_ipv6_address_on_creation'):
 498         ensure_assign_ipv6_on_create(conn, module, subnet, module.params['assign_instances_ipv6'], module.check_mode, start_time)
 499         changed = True
 500 
 501     if module.params['tags'] != subnet['tags']:
 502         stringified_tags_dict = dict((to_text(k), to_text(v)) for k, v in module.params['tags'].items())
 503         if ensure_tags(conn, module, subnet, stringified_tags_dict, module.params['purge_tags'], start_time):
 504             changed = True
 505 
 506     subnet = get_matching_subnet(conn, module, module.params['vpc_id'], module.params['cidr'])
 507     if not module.check_mode and module.params['wait']:
 508         # GET calls are not monotonic for map_public_ip_on_launch and assign_ipv6_address_on_creation
 509         # so we only wait for those if necessary just before returning the subnet
 510         subnet = ensure_final_subnet(conn, module, subnet, start_time)
 511 
 512     return {
 513         'changed': changed,
 514         'subnet': subnet
 515     }

 518 def ensure_final_subnet(conn, module, subnet, start_time):
 519     for rewait in range(0, 30):
 520         map_public_correct = False
 521         assign_ipv6_correct = False
 522 
 523         if module.params['map_public'] == subnet['map_public_ip_on_launch']:
 524             map_public_correct = True
 525         else:
 526             if module.params['map_public']:
 527                 handle_waiter(conn, module, 'subnet_has_map_public', {'SubnetIds': [subnet['id']]}, start_time)
 528             else:
 529                 handle_waiter(conn, module, 'subnet_no_map_public', {'SubnetIds': [subnet['id']]}, start_time)
 530 
 531         if module.params['assign_instances_ipv6'] == subnet.get('assign_ipv6_address_on_creation'):
 532             assign_ipv6_correct = True
 533         else:
 534             if module.params['assign_instances_ipv6']:
 535                 handle_waiter(conn, module, 'subnet_has_assign_ipv6', {'SubnetIds': [subnet['id']]}, start_time)
 536             else:
 537                 handle_waiter(conn, module, 'subnet_no_assign_ipv6', {'SubnetIds': [subnet['id']]}, start_time)
 538 
 539         if map_public_correct and assign_ipv6_correct:
 540             break
 541 
 542         time.sleep(5)
 543         subnet = get_matching_subnet(conn, module, module.params['vpc_id'], module.params['cidr'])
 544 
 545     return subnet

 548 def ensure_subnet_absent(conn, module):
 549     subnet = get_matching_subnet(conn, module, module.params['vpc_id'], module.params['cidr'])
 550     if subnet is None:
 551         return {'changed': False}
 552 
 553     try:
 554         if not module.check_mode:
 555             conn.delete_subnet(SubnetId=subnet['id'])
 556             if module.params['wait']:
 557                 handle_waiter(conn, module, 'subnet_deleted', {'SubnetIds': [subnet['id']]}, time.time())
 558         return {'changed': True}
 559     except (botocore.exceptions.ClientError, botocore.exceptions.BotoCoreError) as e:
