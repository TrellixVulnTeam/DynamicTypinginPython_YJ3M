 349 def check_args(module, warnings):
 350     if module.params["multiline_delimiter"]:
 351         if len(module.params["multiline_delimiter"]) != 1:
 352             module.fail_json(
 353                 msg="multiline_delimiter value can only be a "
 354                 "single character"
 355             )

 358 def edit_config_or_macro(connection, commands):
 359     # only catch the macro configuration command,
 360     # not negated 'no' variation.
 361     if commands[0].startswith("macro name"):
 362         connection.edit_macro(candidate=commands)
 363     else:
 364         connection.edit_config(candidate=commands)
 365 
 366 
 367 def get_candidate_config(module):
 368     candidate = ""
 369     if module.params["src"]:
 370         candidate = module.params["src"]
 371 
 372     elif module.params["lines"]:
 373         candidate_obj = NetworkConfig(indent=1)
 374         parents = module.params["parents"] or list()
 375         candidate_obj.add(module.params["lines"], parents=parents)
 376         candidate = dumps(candidate_obj, "raw")
 377 
 378     return candidate
 379 
 380 
 381 def get_running_config(module, current_config=None, flags=None):
 382     running = module.params["running_config"]
 383     if not running:
 384         if not module.params["defaults"] and current_config:
 385             running = current_config
 386         else:
 387             running = get_config(module, flags=flags)
 388 
 389     return running
 390

 392 def save_config(module, result):
 393     result["changed"] = True
 394     if not module.check_mode:
 395         run_commands(module, "copy running-config startup-config\r")
 396     else:
 397         module.warn(
 398             "Skipping command `copy running-config startup-config` "
 399             "due to check_mode.  Configuration not copied to "
 400             "non-volatile storage"
 401         )

 404 def main():
 405     """ main entry point for module execution
 406     """
 407     backup_spec = dict(filename=dict(), dir_path=dict(type="path"))
 408     argument_spec = dict(
 409         src=dict(type="path"),
 410         lines=dict(aliases=["commands"], type="list"),
 411         parents=dict(type="list"),
 412         before=dict(type="list"),
 413         after=dict(type="list"),
 414         match=dict(
 415             default="line", choices=["line", "strict", "exact", "none"]
 416         ),
 417         replace=dict(default="line", choices=["line", "block"]),
 418         multiline_delimiter=dict(default="@"),
 419         running_config=dict(aliases=["config"]),
 420         intended_config=dict(),
 421         defaults=dict(type="bool", default=False),
 422         backup=dict(type="bool", default=False),
 423         backup_options=dict(type="dict", options=backup_spec),
 424         save_when=dict(
 425             choices=["always", "never", "modified", "changed"], default="never"
 426         ),
 427         diff_against=dict(choices=["startup", "intended", "running"]),
 428         diff_ignore_lines=dict(type="list"),
 429     )
 430 
 431     argument_spec.update(ios_argument_spec)
 432 
 433     mutually_exclusive = [("lines", "src"), ("parents", "src")]
 434 
 435     required_if = [
 436         ("match", "strict", ["lines"]),
 437         ("match", "exact", ["lines"]),
 438         ("replace", "block", ["lines"]),
 439         ("diff_against", "intended", ["intended_config"]),
 440     ]
 441 
 442     module = AnsibleModule(
 443         argument_spec=argument_spec,
 444         mutually_exclusive=mutually_exclusive,
 445         required_if=required_if,
 446         supports_check_mode=True,
 447     )
 448 
 449     result = {"changed": False}
 450 
 451     warnings = list()
 452     check_args(module, warnings)
 453     result["warnings"] = warnings
 454 
 455     diff_ignore_lines = module.params["diff_ignore_lines"]
 456     config = None
 457     contents = None
 458     flags = get_defaults_flag(module) if module.params["defaults"] else []
 459     connection = get_connection(module)
 460 
 461     if module.params["backup"] or (
 462         module._diff and module.params["diff_against"] == "running"
 463     ):
 464         contents = get_config(module, flags=flags)
 465         config = NetworkConfig(indent=1, contents=contents)
 466         if module.params["backup"]:
 467             result["__backup__"] = contents
 468 
 469     if any((module.params["lines"], module.params["src"])):
 470         match = module.params["match"]
 471         replace = module.params["replace"]
 472         path = module.params["parents"]
 473 
 474         candidate = get_candidate_config(module)
 475         running = get_running_config(module, contents, flags=flags)
 476         try:
 477             response = connection.get_diff(
 478                 candidate=candidate,
 479                 running=running,
 480                 diff_match=match,
 481                 diff_ignore_lines=diff_ignore_lines,
 482                 path=path,
 483                 diff_replace=replace,
 484             )
 485         except ConnectionError as exc:
 486             module.fail_json(msg=to_text(exc, errors="surrogate_then_replace"))
 487 
 488         config_diff = response["config_diff"]
 489         banner_diff = response["banner_diff"]
 490 
 491         if config_diff or banner_diff:
 492             commands = config_diff.split("\n")
 493 
 494             if module.params["before"]:
 495                 commands[:0] = module.params["before"]
 496 
 497             if module.params["after"]:
 498                 commands.extend(module.params["after"])
 499 
 500             result["commands"] = commands
 501             result["updates"] = commands
 502             result["banners"] = banner_diff
 503 
 504             # send the configuration commands to the device and merge
 505             # them with the current running config
 506             if not module.check_mode:
 507                 if commands:
 508                     edit_config_or_macro(connection, commands)
 509                 if banner_diff:
 510                     connection.edit_banner(
 511                         candidate=json.dumps(banner_diff),
 512                         multiline_delimiter=module.params[
 513                             "multiline_delimiter"
 514                         ],
 515                     )
 516 
 517             result["changed"] = True
 518 
 519     running_config = module.params["running_config"]
 520     startup_config = None
 521 
 522     if module.params["save_when"] == "always":
 523         save_config(module, result)
 524     elif module.params["save_when"] == "modified":
 525         output = run_commands(
 526             module, ["show running-config", "show startup-config"]
 527         )
 528 
 529         running_config = NetworkConfig(
 530             indent=1, contents=output[0], ignore_lines=diff_ignore_lines
 531         )
 532         startup_config = NetworkConfig(
 533             indent=1, contents=output[1], ignore_lines=diff_ignore_lines
 534         )
 535 
 536         if running_config.sha1 != startup_config.sha1:
 537             save_config(module, result)
 538     elif module.params["save_when"] == "changed" and result["changed"]:
 539         save_config(module, result)
 540 
 541     if module._diff:
 542         if not running_config:
 543             output = run_commands(module, "show running-config")
 544             contents = output[0]
 545         else:
 546             contents = running_config
 547 
 548         # recreate the object in order to process diff_ignore_lines
 549         running_config = NetworkConfig(
 550             indent=1, contents=contents, ignore_lines=diff_ignore_lines
 551         )
 552 
 553         if module.params["diff_against"] == "running":
 554             if module.check_mode:
 555                 module.warn(
 556                     "unable to perform diff against running-config due to check mode"
 557                 )
 558                 contents = None
 559             else:
 560                 contents = config.config_text
 561 
 562         elif module.params["diff_against"] == "startup":
 563             if not startup_config:
 564                 output = run_commands(module, "show startup-config")
 565                 contents = output[0]
 566             else:
 567                 contents = startup_config.config_text
 568 
 569         elif module.params["diff_against"] == "intended":
 570             contents = module.params["intended_config"]
 571 
 572         if contents is not None:
 573             base_config = NetworkConfig(
 574                 indent=1, contents=contents, ignore_lines=diff_ignore_lines
 575             )
 576 
 577             if running_config.sha1 != base_config.sha1:
 578                 if module.params["diff_against"] == "intended":
 579                     before = running_config
 580                     after = base_config
 581                 elif module.params["diff_against"] in ("startup", "running"):
 582                     before = base_config
 583                     after = running_config
 584 
 585                 result.update(
 586                     {
 587                         "changed": True,
 588                         "diff": {"before": str(before), "after": str(after)},
 589                     }
 590                 )
 591 
 592     module.exit_json(**result)
 593 
 594 
 595 if __name__ == "__main__":
 596     main()
