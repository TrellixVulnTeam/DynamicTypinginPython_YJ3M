  34 class TerminalModule(TerminalBase):
  35 
  36     terminal_stdout_re = [
  37         re.compile(br"[\r\n]?[\w\+\-\.:\/\[\]]+(?:\([^\)]+\)){0,3}(?:[>#]) ?$")
  38     ]
  39 
  40     terminal_stderr_re = [
  41         re.compile(br"% ?Error"),
  42         # re.compile(br"^% \w+", re.M),
  43         re.compile(br"% ?Bad secret"),
  44         re.compile(br"[\r\n%] Bad passwords"),
  45         re.compile(br"invalid input", re.I),
  46         re.compile(br"(?:incomplete|ambiguous) command", re.I),
  47         re.compile(br"connection timed out", re.I),
  48         re.compile(br"[^\r\n]+ not found"),
  49         re.compile(br"'[^']' +returned error code: ?\d+"),
  50         re.compile(br"Bad mask", re.I),
  51         re.compile(br"% ?(\S+) ?overlaps with ?(\S+)", re.I),
  52         re.compile(br"[%\S] ?Error: ?[\s]+", re.I),
  53         re.compile(br"[%\S] ?Informational: ?[\s]+", re.I),
  54         re.compile(br"Command authorization failed"),
  55     ]

  57     def on_open_shell(self):
  58         try:
  59             self._exec_cli_command(b"terminal length 0")
  60         except AnsibleConnectionFailure:
  61             raise AnsibleConnectionFailure("unable to set terminal parameters")
  62 
  63         try:
  64             self._exec_cli_command(b"terminal width 512")
  65             try:
  66                 self._exec_cli_command(b"terminal width 0")
  67             except AnsibleConnectionFailure:
  68                 pass
  69         except AnsibleConnectionFailure:
  70             display.display(
  71                 "WARNING: Unable to set terminal width, command responses may be truncated"
  72             )
  73 
  74     def on_become(self, passwd=None):
  75         if self._get_prompt().endswith(b"#"):
  76             return
  77 
  78         cmd = {u"command": u"enable"}
  79         if passwd:
  80             # Note: python-3.5 cannot combine u"" and r"" together.  Thus make
  81             # an r string and use to_text to ensure it's text on both py2 and py3.
  82             cmd[u"prompt"] = to_text(
  83                 r"[\r\n]?(?:.*)?[Pp]assword: ?$", errors="surrogate_or_strict"
  84             )
  85             cmd[u"answer"] = passwd
  86             cmd[u"prompt_retry_check"] = True
  87         try:
  88             self._exec_cli_command(
  89                 to_bytes(json.dumps(cmd), errors="surrogate_or_strict")
  90             )
  91             prompt = self._get_prompt()
  92             if prompt is None or not prompt.endswith(b"#"):
  93                 raise AnsibleConnectionFailure(
  94                     "failed to elevate privilege to enable mode still at prompt [%s]"
  95                     % prompt
  96                 )
  97         except AnsibleConnectionFailure as e:
  98             prompt = self._get_prompt()
  99             raise AnsibleConnectionFailure(
 100                 "unable to elevate privilege to enable mode, at prompt [%s] with error: %s"
 101                 % (prompt, e.message)
 102             )
 103 
 104     def on_unbecome(self):
 105         prompt = self._get_prompt()
 106         if prompt is None:
 107             # if prompt is None most likely the terminal is hung up at a prompt
 108             return
 109 
 110         if b"(config" in prompt:
 111             self._exec_cli_command(b"end")
 112             self._exec_cli_command(b"disable")
 113 
 114         elif prompt.endswith(b"#"):
 115             self._exec_cli_command(b"disable")
