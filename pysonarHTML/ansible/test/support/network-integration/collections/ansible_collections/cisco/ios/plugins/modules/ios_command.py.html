 154 def parse_commands(module, warnings):
 155     commands = transform_commands(module)
 156 
 157     if module.check_mode:
 158         for item in list(commands):
 159             if not item["command"].startswith("show"):
 160                 warnings.append(
 161                     "Only show commands are supported when using check mode, not "
 162                     "executing %s" % item["command"]
 163                 )
 164                 commands.remove(item)
 165 
 166     return commands
 167

 169 def main():
 170     """main entry point for module execution
 171     """
 172     argument_spec = dict(
 173         commands=dict(type="list", required=True),
 174         wait_for=dict(type="list", aliases=["waitfor"]),
 175         match=dict(default="all", choices=["all", "any"]),
 176         retries=dict(default=10, type="int"),
 177         interval=dict(default=1, type="int"),
 178     )
 179 
 180     argument_spec.update(ios_argument_spec)
 181 
 182     module = AnsibleModule(
 183         argument_spec=argument_spec, supports_check_mode=True
 184     )
 185 
 186     warnings = list()
 187     result = {"changed": False, "warnings": warnings}
 188     commands = parse_commands(module, warnings)
 189     wait_for = module.params["wait_for"] or list()
 190 
 191     try:
 192         conditionals = [Conditional(c) for c in wait_for]
 193     except AttributeError as exc:
 194         module.fail_json(msg=to_text(exc))
 195 
 196     retries = module.params["retries"]
 197     interval = module.params["interval"]
 198     match = module.params["match"]
 199 
 200     while retries > 0:
 201         responses = run_commands(module, commands)
 202 
 203         for item in list(conditionals):
 204             if item(responses):
 205                 if match == "any":
 206                     conditionals = list()
 207                     break
 208                 conditionals.remove(item)
 209 
 210         if not conditionals:
 211             break
 212 
 213         time.sleep(interval)
 214         retries -= 1
 215 
 216     if conditionals:
 217         failed_conditions = [item.raw for item in conditionals]
 218         msg = "One or more conditional statements have not been satisfied"
 219         module.fail_json(msg=msg, failed_conditions=failed_conditions)
 220 
 221     result.update(
 222         {"stdout": responses, "stdout_lines": list(to_lines(responses))}
 223     )
 224 
 225     module.exit_json(**result)
