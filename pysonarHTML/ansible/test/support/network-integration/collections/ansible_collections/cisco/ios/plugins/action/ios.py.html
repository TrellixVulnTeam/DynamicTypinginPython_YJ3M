  40 class ActionModule(ActionNetworkModule):
  41     def run(self, tmp=None, task_vars=None):
  42         del tmp  # tmp no longer has any effect
  43 
  44         module_name = self._task.action.split(".")[-1]
  45         self._config_module = True if module_name == "ios_config" else False
  46         persistent_connection = self._play_context.connection.split(".")[-1]
  47         warnings = []
  48 
  49         if persistent_connection == "network_cli":
  50             provider = self._task.args.get("provider", {})
  51             if any(provider.values()):
  52                 display.warning(
  53                     "provider is unnecessary when using network_cli and will be ignored"
  54                 )
  55                 del self._task.args["provider"]
  56         elif self._play_context.connection == "local":
  57             provider = load_provider(ios_provider_spec, self._task.args)
  58             pc = copy.deepcopy(self._play_context)
  59             pc.connection = "ansible.netcommon.network_cli"
  60             pc.network_os = "cisco.ios.ios"
  61             pc.remote_addr = provider["host"] or self._play_context.remote_addr
  62             pc.port = int(provider["port"] or self._play_context.port or 22)
  63             pc.remote_user = (
  64                 provider["username"] or self._play_context.connection_user
  65             )
  66             pc.password = provider["password"] or self._play_context.password
  67             pc.private_key_file = (
  68                 provider["ssh_keyfile"] or self._play_context.private_key_file
  69             )
  70             pc.become = provider["authorize"] or False
  71             if pc.become:
  72                 pc.become_method = "enable"
  73             pc.become_pass = provider["auth_pass"]
  74 
  75             connection = self._shared_loader_obj.connection_loader.get(
  76                 "ansible.netcommon.persistent",
  77                 pc,
  78                 sys.stdin,
  79                 task_uuid=self._task._uuid,
  80             )
  81 
  82             # TODO: Remove below code after ansible minimal is cut out
  83             if connection is None:
  84                 pc.connection = "network_cli"
  85                 pc.network_os = "ios"
  86                 connection = self._shared_loader_obj.connection_loader.get(
  87                     "persistent", pc, sys.stdin, task_uuid=self._task._uuid
  88                 )
  89 
  90             display.vvv(
  91                 "using connection plugin %s (was local)" % pc.connection,
  92                 pc.remote_addr,
  93             )
  94 
  95             command_timeout = (
  96                 int(provider["timeout"])
  97                 if provider["timeout"]
  98                 else connection.get_option("persistent_command_timeout")
  99             )
 100             connection.set_options(
 101                 direct={"persistent_command_timeout": command_timeout}
 102             )
 103 
 104             socket_path = connection.run()
 105             display.vvvv("socket_path: %s" % socket_path, pc.remote_addr)
 106             if not socket_path:
 107                 return {
 108                     "failed": True,
 109                     "msg": "unable to open shell. Please see: "
 110                     + "https://docs.ansible.com/ansible/network_debug_troubleshooting.html#unable-to-open-shell",
 111                 }
 112 
 113             task_vars["ansible_socket"] = socket_path
 114             warnings.append(
 115                 [
 116                     "connection local support for this module is deprecated and will be removed in version 2.14, use connection %s"
 117                     % pc.connection
 118                 ]
 119             )
 120         else:
 121             return {
 122                 "failed": True,
 123                 "msg": "Connection type %s is not valid for this module"
 124                 % self._play_context.connection,
 125             }
 126 
 127         result = super(ActionModule, self).run(task_vars=task_vars)
 128         if warnings:
 129             if "warnings" in result:
 130                 result["warnings"].extend(warnings)
 131             else:
 132                 result["warnings"] = warnings
 133         return result
