 153 def parse_commands(module, warnings):
 154     commands = transform_commands(module)
 155 
 156     if module.check_mode:
 157         for item in list(commands):
 158             if not item["command"].startswith("show"):
 159                 warnings.append(
 160                     "Only show commands are supported when using check mode, not "
 161                     "executing %s" % item["command"]
 162                 )
 163                 commands.remove(item)
 164 
 165     return commands
 166 
 167 
 168 def main():
 169     spec = dict(
 170         commands=dict(type="list", required=True),
 171         wait_for=dict(type="list", aliases=["waitfor"]),
 172         match=dict(default="all", choices=["all", "any"]),
 173         retries=dict(default=10, type="int"),
 174         interval=dict(default=1, type="int"),
 175     )
 176 
 177     spec.update(vyos_argument_spec)
 178 
 179     module = AnsibleModule(argument_spec=spec, supports_check_mode=True)
 180 
 181     warnings = list()
 182     result = {"changed": False, "warnings": warnings}
 183     commands = parse_commands(module, warnings)
 184     wait_for = module.params["wait_for"] or list()
 185 
 186     try:
 187         conditionals = [Conditional(c) for c in wait_for]
 188     except AttributeError as exc:
 189         module.fail_json(msg=to_text(exc))
 190 
 191     retries = module.params["retries"]
 192     interval = module.params["interval"]
 193     match = module.params["match"]
 194 
 195     for _ in range(retries):
 196         responses = run_commands(module, commands)
 197 
 198         for item in list(conditionals):
 199             if item(responses):
 200                 if match == "any":
 201                     conditionals = list()
 202                     break
 203                 conditionals.remove(item)
 204 
 205         if not conditionals:
 206             break
 207 
 208         time.sleep(interval)
 209 
 210     if conditionals:
 211         failed_conditions = [item.raw for item in conditionals]
 212         msg = "One or more conditional statements have not been satisfied"
 213         module.fail_json(msg=msg, failed_conditions=failed_conditions)
 214 
 215     result.update(
 216         {"stdout": responses, "stdout_lines": list(to_lines(responses)),}
 217     )
 218 
 219     module.exit_json(**result)
 220 
 221 
 222 if __name__ == "__main__":
 223     main()
