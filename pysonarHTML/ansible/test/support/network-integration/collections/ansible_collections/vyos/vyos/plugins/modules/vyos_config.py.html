 197 CONFIG_FILTERS = [
 198     re.compile(r"set system login user \S+ authentication encrypted-password")
 199 ]

 202 def get_candidate(module):
 203     contents = module.params["src"] or module.params["lines"]
 204 
 205     if module.params["src"]:
 206         contents = format_commands(contents.splitlines())
 207 
 208     contents = "\n".join(contents)
 209     return contents
 210 
 211 
 212 def format_commands(commands):
 213     """
 214     This function format the input commands and removes the prepend white spaces
 215     for command lines having 'set' or 'delete' and it skips empty lines.
 216     :param commands:
 217     :return: list of commands
 218     """
 219     return [
 220         line.strip() if line.split()[0] in ("set", "delete") else line
 221         for line in commands
 222         if len(line.strip()) > 0
 223     ]

 226 def diff_config(commands, config):
 227     config = [str(c).replace("'", "") for c in config.splitlines()]
 228 
 229     updates = list()
 230     visited = set()
 231 
 232     for line in commands:
 233         item = str(line).replace("'", "")
 234 
 235         if not item.startswith("set") and not item.startswith("delete"):
 236             raise ValueError("line must start with either `set` or `delete`")
 237 
 238         elif item.startswith("set") and item not in config:
 239             updates.append(line)
 240 
 241         elif item.startswith("delete"):
 242             if not config:
 243                 updates.append(line)
 244             else:
 245                 item = re.sub(r"delete", "set", item)
 246                 for entry in config:
 247                     if entry.startswith(item) and line not in visited:
 248                         updates.append(line)
 249                         visited.add(line)
 250 
 251     return list(updates)

 254 def sanitize_config(config, result):
 255     result["filtered"] = list()
 256     index_to_filter = list()
 257     for regex in CONFIG_FILTERS:
 258         for index, line in enumerate(list(config)):
 259             if regex.search(line):
 260                 result["filtered"].append(line)
 261                 index_to_filter.append(index)
 262     # Delete all filtered configs
 263     for filter_index in sorted(index_to_filter, reverse=True):
 264         del config[filter_index]

 267 def run(module, result):
 268     # get the current active config from the node or passed in via
 269     # the config param
 270     config = module.params["config"] or get_config(module)
 271 
 272     # create the candidate config object from the arguments
 273     candidate = get_candidate(module)
 274 
 275     # create loadable config that includes only the configuration updates
 276     connection = get_connection(module)
 277     try:
 278         response = connection.get_diff(
 279             candidate=candidate,
 280             running=config,
 281             diff_match=module.params["match"],
 282         )
 283     except ConnectionError as exc:
 284         module.fail_json(msg=to_text(exc, errors="surrogate_then_replace"))
 285 
 286     commands = response.get("config_diff")
 287     sanitize_config(commands, result)
 288 
 289     result["commands"] = commands
 290 
 291     commit = not module.check_mode
 292     comment = module.params["comment"]
 293 
 294     diff = None
 295     if commands:
 296         diff = load_config(module, commands, commit=commit, comment=comment)
 297 
 298         if result.get("filtered"):
 299             result["warnings"].append(
 300                 "Some configuration commands were "
 301                 "removed, please see the filtered key"
 302             )
 303 
 304         result["changed"] = True
 305 
 306     if module._diff:
 307         result["diff"] = {"prepared": diff}

 310 def main():
 311     backup_spec = dict(filename=dict(), dir_path=dict(type="path"))
 312     argument_spec = dict(
 313         src=dict(type="path"),
 314         lines=dict(type="list"),
 315         match=dict(default="line", choices=["line", "none"]),
 316         comment=dict(default=DEFAULT_COMMENT),
 317         config=dict(),
 318         backup=dict(type="bool", default=False),
 319         backup_options=dict(type="dict", options=backup_spec),
 320         save=dict(type="bool", default=False),
 321     )
 322 
 323     argument_spec.update(vyos_argument_spec)
 324 
 325     mutually_exclusive = [("lines", "src")]
 326 
 327     module = AnsibleModule(
 328         argument_spec=argument_spec,
 329         mutually_exclusive=mutually_exclusive,
 330         supports_check_mode=True,
 331     )
 332 
 333     warnings = list()
 334 
 335     result = dict(changed=False, warnings=warnings)
 336 
 337     if module.params["backup"]:
 338         result["__backup__"] = get_config(module=module)
 339 
 340     if any((module.params["src"], module.params["lines"])):
 341         run(module, result)
 342 
 343     if module.params["save"]:
 344         diff = run_commands(module, commands=["configure", "compare saved"])[1]
 345         if diff != "[edit]":
 346             run_commands(module, commands=["save"])
 347             result["changed"] = True
 348         run_commands(module, commands=["exit"])
 349 
 350     module.exit_json(**result)
