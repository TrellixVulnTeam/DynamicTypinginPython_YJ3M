  59 def get_connection(module):
  60     if hasattr(module, "_vyos_connection"):
  61         return module._vyos_connection
  62 
  63     capabilities = get_capabilities(module)
  64     network_api = capabilities.get("network_api")
  65     if network_api == "cliconf":
  66         module._vyos_connection = Connection(module._socket_path)
  67     else:
  68         module.fail_json(msg="Invalid connection type %s" % network_api)
  69 
  70     return module._vyos_connection
  71 
  72 
  73 def get_capabilities(module):
  74     if hasattr(module, "_vyos_capabilities"):
  75         return module._vyos_capabilities
  76 
  77     try:
  78         capabilities = Connection(module._socket_path).get_capabilities()
  79     except ConnectionError as exc:
  80         module.fail_json(msg=to_text(exc, errors="surrogate_then_replace"))
  81 
  82     module._vyos_capabilities = json.loads(capabilities)
  83     return module._vyos_capabilities

  86 def get_config(module, flags=None, format=None):
  87     flags = [] if flags is None else flags
  88     global _DEVICE_CONFIGS
  89 
  90     if _DEVICE_CONFIGS != {}:
  91         return _DEVICE_CONFIGS
  92     else:
  93         connection = get_connection(module)
  94         try:
  95             out = connection.get_config(flags=flags, format=format)
  96         except ConnectionError as exc:
  97             module.fail_json(msg=to_text(exc, errors="surrogate_then_replace"))
  98         cfg = to_text(out, errors="surrogate_then_replace").strip()
  99         _DEVICE_CONFIGS = cfg
 100         return cfg

 103 def run_commands(module, commands, check_rc=True):
 104     connection = get_connection(module)
 105     try:
 106         response = connection.run_commands(
 107             commands=commands, check_rc=check_rc
 108         )
 109     except ConnectionError as exc:
 110         module.fail_json(msg=to_text(exc, errors="surrogate_then_replace"))
 111     return response

 114 def load_config(module, commands, commit=False, comment=None):
 115     connection = get_connection(module)
 116 
 117     try:
 118         response = connection.edit_config(
 119             candidate=commands, commit=commit, comment=comment
 120         )
 121     except ConnectionError as exc:
 122         module.fail_json(msg=to_text(exc, errors="surrogate_then_replace"))
 123 
 124     return response.get("diff")
