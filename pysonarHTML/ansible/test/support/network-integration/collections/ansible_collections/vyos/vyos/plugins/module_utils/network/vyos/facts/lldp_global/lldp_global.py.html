  31     def __init__(self, module, subspec="config", options="options"):
  32         self._module = module
  33         self.argument_spec = Lldp_globalArgs.argument_spec
  34         spec = deepcopy(self.argument_spec)
  35         if subspec:
  36             if options:
  37                 facts_argument_spec = spec[subspec][options]
  38             else:
  39                 facts_argument_spec = spec[subspec]
  40         else:
  41             facts_argument_spec = spec
  42 
  43         self.generated_spec = utils.generate_dict(facts_argument_spec)
  44 
  45     def populate_facts(self, connection, ansible_facts, data=None):
  46         """ Populate the facts for lldp_global
  47         :param connection: the device connection
  48         :param ansible_facts: Facts dictionary
  49         :param data: previously collected conf
  50         :rtype: dictionary
  51         :returns: facts
  52         """
  53         if not data:
  54             data = connection.get_config()
  55 
  56         objs = {}
  57         lldp_output = findall(r"^set service lldp (\S+)", data, M)
  58         if lldp_output:
  59             for item in set(lldp_output):
  60                 lldp_regex = r" %s .+$" % item
  61                 cfg = findall(lldp_regex, data, M)
  62                 obj = self.render_config(cfg)
  63                 if obj:
  64                     objs.update(obj)
  65         lldp_service = findall(r"^set service (lldp)?('lldp')", data, M)
  66         if lldp_service or lldp_output:
  67             lldp_obj = {}
  68             lldp_obj["enable"] = True
  69             objs.update(lldp_obj)
  70 
  71         facts = {}
  72         params = utils.validate_config(self.argument_spec, {"config": objs})
  73         facts["lldp_global"] = utils.remove_empties(params["config"])
  74 
  75         ansible_facts["ansible_network_resources"].update(facts)
  76 
  77         return ansible_facts
  78 
  79     def render_config(self, conf):
  80         """
  81          Render config as dictionary structure and delete keys
  82            from spec for null values
  83          :param spec: The facts tree, generated from the argspec
  84          :param conf: The configuration
  85          :rtype: dictionary
  86          :returns: The generated config
  87          """
  88         protocol_conf = "\n".join(
  89             filter(lambda x: ("legacy-protocols" in x), conf)
  90         )
  91         att_conf = "\n".join(
  92             filter(lambda x: ("legacy-protocols" not in x), conf)
  93         )
  94         config = self.parse_attribs(["snmp", "address"], att_conf)
  95         config["legacy_protocols"] = self.parse_protocols(protocol_conf)
  96         return utils.remove_empties(config)
  97 
  98     def parse_protocols(self, conf):
  99         protocol_support = None
 100         if conf:
 101             protocols = findall(r"^.*legacy-protocols (.+)", conf, M)
 102             if protocols:
 103                 protocol_support = []
 104                 for protocol in protocols:
 105                     protocol_support.append(protocol.strip("'"))
 106         return protocol_support
 107 
 108     def parse_attribs(self, attribs, conf):
 109         config = {}
 110         for item in attribs:
 111             value = utils.parse_conf_arg(conf, item)
 112             if value:
 113                 config[item] = value.strip("'")
 114             else:
 115                 config[item] = None
 116         return utils.remove_empties(config)
