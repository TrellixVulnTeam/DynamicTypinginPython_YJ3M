  47     def populate_facts(self, connection, ansible_facts, data=None):
  48         """ Populate the facts for firewall_rules
  49         :param connection: the device connection
  50         :param ansible_facts: Facts dictionary
  51         :param data: previously collected conf
  52         :rtype: dictionary
  53         :returns: facts
  54         """
  55         if not data:
  56             # typically data is populated from the current device configuration
  57             # data = connection.get('show running-config | section ^interface')
  58             # using mock data instead
  59             data = self.get_device_data(connection)
  60         # split the config into instances of the resource
  61         objs = []
  62         v6_rules = findall(
  63             r"^set firewall ipv6-name (?:\'*)(\S+)(?:\'*)", data, M
  64         )
  65         v4_rules = findall(r"^set firewall name (?:\'*)(\S+)(?:\'*)", data, M)
  66         if v6_rules:
  67             config = self.get_rules(data, v6_rules, type="ipv6")
  68             if config:
  69                 config = utils.remove_empties(config)
  70                 objs.append(config)
  71         if v4_rules:
  72             config = self.get_rules(data, v4_rules, type="ipv4")
  73             if config:
  74                 config = utils.remove_empties(config)
  75                 objs.append(config)
  76 
  77         ansible_facts["ansible_network_resources"].pop("firewall_rules", None)
  78         facts = {}
  79         if objs:
  80             facts["firewall_rules"] = []
  81             params = utils.validate_config(
  82                 self.argument_spec, {"config": objs}
  83             )
  84             for cfg in params["config"]:
  85                 facts["firewall_rules"].append(utils.remove_empties(cfg))
  86 
  87         ansible_facts["ansible_network_resources"].update(facts)
  88         return ansible_facts

  90     def get_rules(self, data, rules, type):
  91         """
  92         This function performs following:
  93         - Form regex to fetch 'rule-sets' specific config from data.
  94         - Form the rule-set list based on ip address.
  95         :param data: configuration.
  96         :param rules: list of rule-sets.
  97         :param type: ip address type.
  98         :return: generated rule-sets configuration.
  99         """
 100         r_v4 = []
 101         r_v6 = []
 102         for r in set(rules):
 103             rule_regex = r" %s .+$" % r.strip("'")
 104             cfg = findall(rule_regex, data, M)
 105             fr = self.render_config(cfg, r.strip("'"))
 106             fr["name"] = r.strip("'")
 107             if type == "ipv6":
 108                 r_v6.append(fr)
 109             else:
 110                 r_v4.append(fr)
 111         if r_v4:
 112             config = {"afi": "ipv4", "rule_sets": r_v4}
 113         if r_v6:
 114             config = {"afi": "ipv6", "rule_sets": r_v6}
 115         return config

 117     def render_config(self, conf, match):
 118         """
 119         Render config as dictionary structure and delete keys
 120           from spec for null values
 121 
 122         :param spec: The facts tree, generated from the argspec
 123         :param conf: The configuration
 124         :rtype: dictionary
 125         :returns: The generated config
 126         """
 127         conf = "\n".join(filter(lambda x: x, conf))
 128         a_lst = ["description", "default_action", "enable_default_log"]
 129         config = self.parse_attr(conf, a_lst, match)
 130         if not config:
 131             config = {}
 132         config["rules"] = self.parse_rules_lst(conf)
 133         return config

 135     def parse_rules_lst(self, conf):
 136         """
 137         This function forms the regex to fetch the 'rules' with in
 138         'rule-sets'
 139         :param conf: configuration data.
 140         :return: generated rule list configuration.
 141         """
 142         r_lst = []
 143         rules = findall(r"rule (?:\'*)(\d+)(?:\'*)", conf, M)
 144         if rules:
 145             rules_lst = []
 146             for r in set(rules):
 147                 r_regex = r" %s .+$" % r
 148                 cfg = "\n".join(findall(r_regex, conf, M))
 149                 obj = self.parse_rules(cfg)
 150                 obj["number"] = int(r)
 151                 if obj:
 152                     rules_lst.append(obj)
 153             r_lst = sorted(rules_lst, key=lambda i: i["number"])
 154         return r_lst

 156     def parse_rules(self, conf):
 157         """
 158         This function triggers the parsing of 'rule' attributes.
 159         a_lst is a list having rule attributes which doesn't
 160         have further sub attributes.
 161         :param conf: configuration
 162         :return: generated rule configuration dictionary.
 163         """
 164         a_lst = [
 165             "ipsec",
 166             "action",
 167             "protocol",
 168             "fragment",
 169             "disabled",
 170             "description",
 171         ]
 172         rule = self.parse_attr(conf, a_lst)
 173         r_sub = {
 174             "p2p": self.parse_p2p(conf),
 175             "tcp": self.parse_tcp(conf, "tcp"),
 176             "icmp": self.parse_icmp(conf, "icmp"),
 177             "time": self.parse_time(conf, "time"),
 178             "limit": self.parse_limit(conf, "limit"),
 179             "state": self.parse_state(conf, "state"),
 180             "recent": self.parse_recent(conf, "recent"),
 181             "source": self.parse_src_or_dest(conf, "source"),
 182             "destination": self.parse_src_or_dest(conf, "destination"),
 183         }
 184         rule.update(r_sub)
 185         return rule

 187     def parse_p2p(self, conf):

 190         'rules'

 193         """
 194         a_lst = []
 195         applications = findall(r"p2p (?:\'*)(\d+)(?:\'*)", conf, M)

 197             app_lst = []

 202         return a_lst

 311     def parse_attr(self, conf, attr_list, match=None):
 312         """
 313         This function peforms the following:
 314         - Form the regex to fetch the required attribute config.
 315         - Type cast the output in desired format.
 316         :param conf: configuration.
 317         :param attr_list: list of attributes.
 318         :param match: parent node/attribute name.
 319         :return: generated config dictionary.
 320         """
 321         config = {}
 322         for attrib in attr_list:
 323             regex = self.map_regex(attrib)
 324             if match:
 325                 regex = match + " " + regex
 326             if conf:
 327                 if self.is_bool(attrib):
 328                     out = conf.find(attrib.replace("_", "-"))
 329 
 330                     dis = conf.find(attrib.replace("_", "-") + " 'disable'")
 331                     if out >= 1:
 332                         if dis >= 1:
 333                             config[attrib] = False
 334                         else:
 335                             config[attrib] = True
 336                 else:
 337                     out = search(r"^.*" + regex + " (.+)", conf, M)
 338                     if out:
 339                         val = out.group(1).strip("'")
 340                         if self.is_num(attrib):
 341                             val = int(val)
 342                         config[attrib] = val
 343         return config

 345     def map_regex(self, attrib):
 346         """
 347         - This function construct the regex string.
 348         - replace the underscore with hyphen.
 349         :param attrib: attribute
 350         :return: regex string
 351         """
 352         regex = attrib.replace("_", "-")
 353         if attrib == "disabled":
 354             regex = "disable"
 355         return regex
