 130 def spec_to_commands(updates, module):
 131     commands = list()
 132     want, have = updates
 133 
 134     for w in want:
 135         dest = w["dest"]
 136         name = w["name"]
 137         facility = w["facility"]
 138         level = w["level"]
 139         state = w["state"]
 140         del w["state"]
 141 
 142         if state == "absent" and w in have:
 143             if w["name"]:
 144                 commands.append(
 145                     "delete system syslog {0} {1} facility {2} level {3}".format(
 146                         dest, name, facility, level
 147                     )
 148                 )
 149             else:
 150                 commands.append(
 151                     "delete system syslog {0} facility {1} level {2}".format(
 152                         dest, facility, level
 153                     )
 154                 )
 155         elif state == "present" and w not in have:
 156             if w["name"]:
 157                 commands.append(
 158                     "set system syslog {0} {1} facility {2} level {3}".format(
 159                         dest, name, facility, level
 160                     )
 161                 )
 162             else:
 163                 commands.append(
 164                     "set system syslog {0} facility {1} level {2}".format(
 165                         dest, facility, level
 166                     )
 167                 )
 168 
 169     return commands

 172 def config_to_dict(module):
 173     data = get_config(module)
 174     obj = []
 175 
 176     for line in data.split("\n"):
 177         if line.startswith("set system syslog"):
 178             match = re.search(r"set system syslog (\S+)", line, re.M)
 179             dest = match.group(1)
 180             if dest == "host":
 181                 match = re.search(r"host (\S+)", line, re.M)
 182                 name = match.group(1)
 183             elif dest == "file":
 184                 match = re.search(r"file (\S+)", line, re.M)
 185                 name = match.group(1)
 186             elif dest == "user":
 187                 match = re.search(r"user (\S+)", line, re.M)
 188                 name = match.group(1)
 189             else:
 190                 name = None
 191 
 192             if "facility" in line:
 193                 match = re.search(r"facility (\S+)", line, re.M)
 194                 facility = match.group(1)
 195             if "level" in line:
 196                 match = re.search(r"level (\S+)", line, re.M)
 197                 level = match.group(1).strip("'")
 198 
 199                 obj.append(
 200                     {
 201                         "dest": dest,
 202                         "name": name,
 203                         "facility": facility,
 204                         "level": level,
 205                     }
 206                 )
 207 
 208     return obj

 211 def map_params_to_obj(module, required_if=None):
 212     obj = []
 213 
 214     aggregate = module.params.get("aggregate")
 215     if aggregate:
 216         for item in aggregate:
 217             for key in item:
 218                 if item.get(key) is None:
 219                     item[key] = module.params[key]
 220 
 221             module._check_required_if(required_if, item)
 222             obj.append(item.copy())
 223 
 224     else:
 225         if module.params["dest"] not in ("host", "file", "user"):
 226             module.params["name"] = None
 227 
 228         obj.append(
 229             {
 230                 "dest": module.params["dest"],
 231                 "name": module.params["name"],
 232                 "facility": module.params["facility"],
 233                 "level": module.params["level"],
 234                 "state": module.params["state"],
 235             }
 236         )
 237 
 238     return obj

 241 def main():
 242     """ main entry point for module execution
 243     """
 244     element_spec = dict(
 245         dest=dict(
 246             type="str", choices=["console", "file", "global", "host", "user"]
 247         ),
 248         name=dict(type="str"),
 249         facility=dict(type="str"),
 250         level=dict(type="str"),
 251         state=dict(default="present", choices=["present", "absent"]),
 252     )
 253 
 254     aggregate_spec = deepcopy(element_spec)
 255 
 256     # remove default in aggregate spec, to handle common arguments
 257     remove_default_spec(aggregate_spec)
 258 
 259     argument_spec = dict(
 260         aggregate=dict(type="list", elements="dict", options=aggregate_spec),
 261     )
 262 
 263     argument_spec.update(element_spec)
 264 
 265     argument_spec.update(vyos_argument_spec)
 266     required_if = [
 267         ("dest", "host", ["name", "facility", "level"]),
 268         ("dest", "file", ["name", "facility", "level"]),
 269         ("dest", "user", ["name", "facility", "level"]),
 270         ("dest", "console", ["facility", "level"]),
 271         ("dest", "global", ["facility", "level"]),
 272     ]
 273 
 274     module = AnsibleModule(
 275         argument_spec=argument_spec,
 276         required_if=required_if,
 277         supports_check_mode=True,
 278     )
 279 
 280     warnings = list()
 281 
 282     result = {"changed": False}
 283     if warnings:
 284         result["warnings"] = warnings
 285     want = map_params_to_obj(module, required_if=required_if)
 286     have = config_to_dict(module)
 287 
 288     commands = spec_to_commands((want, have), module)
 289     result["commands"] = commands
 290 
 291     if commands:
 292         commit = not module.check_mode
 293         load_config(module, commands, commit=commit)
 294         result["changed"] = True
 295 
 296     module.exit_json(**result)
