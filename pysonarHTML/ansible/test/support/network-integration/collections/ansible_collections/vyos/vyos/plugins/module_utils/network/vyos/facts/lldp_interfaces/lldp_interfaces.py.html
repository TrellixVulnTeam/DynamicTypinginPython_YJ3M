  33     def __init__(self, module, subspec="config", options="options"):
  34         self._module = module
  35         self.argument_spec = Lldp_interfacesArgs.argument_spec
  36         spec = deepcopy(self.argument_spec)
  37         if subspec:
  38             if options:
  39                 facts_argument_spec = spec[subspec][options]
  40             else:
  41                 facts_argument_spec = spec[subspec]
  42         else:
  43             facts_argument_spec = spec
  44 
  45         self.generated_spec = utils.generate_dict(facts_argument_spec)

  47     def populate_facts(self, connection, ansible_facts, data=None):
  48         """ Populate the facts for lldp_interfaces
  49         :param connection: the device connection
  50         :param ansible_facts: Facts dictionary
  51         :param data: previously collected conf
  52         :rtype: dictionary
  53         :returns: facts
  54         """
  55         if not data:
  56             data = connection.get_config()
  57 
  58         objs = []
  59         lldp_names = findall(r"^set service lldp interface (\S+)", data, M)
  60         if lldp_names:
  61             for lldp in set(lldp_names):
  62                 lldp_regex = r" %s .+$" % lldp
  63                 cfg = findall(lldp_regex, data, M)
  64                 obj = self.render_config(cfg)
  65                 obj["name"] = lldp.strip("'")
  66                 if obj:
  67                     objs.append(obj)
  68         facts = {}
  69         if objs:
  70             facts["lldp_interfaces"] = objs
  71             ansible_facts["ansible_network_resources"].update(facts)
  72 
  73         ansible_facts["ansible_network_resources"].update(facts)
  74         return ansible_facts

  76     def render_config(self, conf):
  77         """
  78         Render config as dictionary structure and delete keys
  79           from spec for null values
  80 
  81         :param spec: The facts tree, generated from the argspec
  82         :param conf: The configuration
  83         :rtype: dictionary
  84         :returns: The generated config
  85         """
  86         config = {}
  87         location = {}
  88 
  89         civic_conf = "\n".join(filter(lambda x: ("civic-based" in x), conf))
  90         elin_conf = "\n".join(filter(lambda x: ("elin" in x), conf))
  91         coordinate_conf = "\n".join(
  92             filter(lambda x: ("coordinate-based" in x), conf)
  93         )
  94         disable = "\n".join(filter(lambda x: ("disable" in x), conf))
  95 
  96         coordinate_based_conf = self.parse_attribs(
  97             ["altitude", "datum", "longitude", "latitude"], coordinate_conf
  98         )
  99         elin_based_conf = self.parse_lldp_elin_based(elin_conf)
 100         civic_based_conf = self.parse_lldp_civic_based(civic_conf)
 101         if disable:
 102             config["enable"] = False
 103         if coordinate_conf:
 104             location["coordinate_based"] = coordinate_based_conf
 105             config["location"] = location
 106         elif civic_based_conf:
 107             location["civic_based"] = civic_based_conf
 108             config["location"] = location
 109         elif elin_conf:
 110             location["elin"] = elin_based_conf
 111             config["location"] = location
 112 
 113         return utils.remove_empties(config)
 114 
 115     def parse_attribs(self, attribs, conf):
 116         config = {}
 117         for item in attribs:
 118             value = utils.parse_conf_arg(conf, item)
 119             if value:
 120                 value = value.strip("'")
 121                 if item == "altitude":
 122                     value = int(value)
 123                 config[item] = value
 124             else:
 125                 config[item] = None
 126         return utils.remove_empties(config)
 127 
 128     def parse_lldp_civic_based(self, conf):
 129         civic_based = None
 130         if conf:
 131             civic_info_list = []
 132             civic_add_list = findall(r"^.*civic-based ca-type (.+)", conf, M)
 133             if civic_add_list:
 134                 for civic_add in civic_add_list:
 135                     ca = civic_add.split(" ")
 136                     c_add = {}
 137                     c_add["ca_type"] = int(ca[0].strip("'"))
 138                     c_add["ca_value"] = ca[2].strip("'")
 139                     civic_info_list.append(c_add)
 140 
 141                 country_code = search(
 142                     r"^.*civic-based country-code (.+)", conf, M
 143                 )
 144                 civic_based = {}
 145                 civic_based["ca_info"] = civic_info_list
 146                 civic_based["country_code"] = country_code.group(1).strip("'")
 147         return civic_based
 148 
 149     def parse_lldp_elin_based(self, conf):
 150         elin_based = None
 151         if conf:
 152             e_num = search(r"^.* elin (.+)", conf, M)
 153             elin_based = e_num.group(1).strip("'")
 154 
 155         return elin_based
