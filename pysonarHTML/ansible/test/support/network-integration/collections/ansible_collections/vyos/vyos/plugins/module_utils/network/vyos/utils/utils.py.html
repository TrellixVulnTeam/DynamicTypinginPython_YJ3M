  23 def get_interface_type(interface):
  24     """Gets the type of interface
  25     """
  26     if interface.startswith("eth"):
  27         return "ethernet"
  28     elif interface.startswith("bond"):
  29         return "bonding"
  30     elif interface.startswith("vti"):
  31         return "vti"
  32     elif interface.startswith("lo"):
  33         return "loopback"

  36 def dict_delete(base, comparable):
  37     """
  38     This function generates a dict containing key, value pairs for keys
  39     that are present in the `base` dict but not present in the `comparable`
  40     dict.
  41 
  42     :param base: dict object to base the diff on
  43     :param comparable: dict object to compare against base
  44     :returns: new dict object with key, value pairs that needs to be deleted.
  45 
  46     """
  47     to_delete = dict()
  48 
  49     for key in base:
  50         if isinstance(base[key], dict):
  51             sub_diff = dict_delete(base[key], comparable.get(key, {}))
  52             if sub_diff:
  53                 to_delete[key] = sub_diff
  54         else:
  55             if key not in comparable:
  56                 to_delete[key] = base[key]
  57 
  58     return to_delete

  61 def diff_list_of_dicts(want, have):
  62     diff = []
  63 
  64     set_w = set(tuple(d.items()) for d in want)
  65     set_h = set(tuple(d.items()) for d in have)
  66     difference = set_w.difference(set_h)
  67 
  68     for element in difference:
  69         diff.append(dict((x, y) for x, y in element))
  70 
  71     return diff

  93 def get_lst_same_for_dicts(want, have, lst):
  94     """
  95     This function generates a list containing values
  96     that are common for list in want and list in have dict
  97     :param want: dict object to want
  98     :param have: dict object to have
  99     :param lst: list the comparison on
 100     :return: new list object with values which are common in want and have.
 101     """
 102     diff = None
 103     if want and have:
 104         want_list = want.get(lst) or {}
 105         have_list = have.get(lst) or {}
 106         diff = [
 107             i
 108             for i in want_list and have_list
 109             if i in have_list and i in want_list
 110         ]
 111     return diff

 114 def list_diff_have_only(want_list, have_list):
 115     """
 116     This function generated the list containing values
 117     that are only in have list.
 118     :param want_list:
 119     :param have_list:
 120     :return: new list with values which are only in have list
 121     """
 122     if have_list and not want_list:
 123         diff = have_list
 124     elif not have_list:
 125         diff = None
 126     else:
 127         diff = [
 128             i
 129             for i in have_list + want_list
 130             if i in have_list and i not in want_list
 131         ]
 132     return diff

 156 def search_dict_tv_in_list(d_val1, d_val2, lst, key1, key2):
 157     """
 158     This function return the dict object if it exist in list.
 159     :param d_val1:
 160     :param d_val2:
 161     :param lst:
 162     :param key1:
 163     :param key2:
 164     :return:
 165     """
 166     obj = next(
 167         (
 168             item
 169             for item in lst
 170             if item[key1] == d_val1 and item[key2] == d_val2
 171         ),
 172         None,
 173     )
 174     if obj:
 175         return obj
 176     else:
 177         return None

 180 def key_value_in_dict(have_key, have_value, want_dict):
 181     """
 182     This function checks whether the key and values exist in dict
 183     :param have_key:
 184     :param have_value:
 185     :param want_dict:
 186     :return:
 187     """
 188     for key, value in iteritems(want_dict):
 189         if key == have_key and value == have_value:
 190             return True
 191     return False

 207 def get_ip_address_version(address):
 208     """
 209     This function returns the version of IP address
 210     :param address: IP address
 211     :return:
 212     """
 213     try:
 214         address = unicode(address)
 215     except NameError:
 216         address = str(address)
 217     version = ipaddress.ip_address(address.split("/")[0]).version
 218     return version

 221 def get_route_type(address):
 222     """
 223     This function returns the route type based on IP address
 224     :param address:
 225     :return:
 226     """
 227     version = get_ip_address_version(address)
 228     if version == 6:
 229         return "route6"
 230     elif version == 4:
 231         return "route"
