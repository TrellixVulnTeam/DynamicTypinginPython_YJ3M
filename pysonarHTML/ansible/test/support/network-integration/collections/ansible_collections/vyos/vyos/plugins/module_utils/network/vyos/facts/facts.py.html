  55 class Facts(FactsBase):
  56     """ The fact class for vyos
  57     """
  58 
  59     VALID_LEGACY_GATHER_SUBSETS = frozenset(FACT_LEGACY_SUBSETS.keys())
  60     VALID_RESOURCE_SUBSETS = frozenset(FACT_RESOURCE_SUBSETS.keys())
  61 
  62     def __init__(self, module):
  63         super(Facts, self).__init__(module)
  64 
  65     def get_facts(
  66         self, legacy_facts_type=None, resource_facts_type=None, data=None
  67     ):
  68         """ Collect the facts for vyos
  69         :param legacy_facts_type: List of legacy facts types
  70         :param resource_facts_type: List of resource fact types
  71         :param data: previously collected conf
  72         :rtype: dict
  73         :return: the facts gathered
  74         """
  75         if self.VALID_RESOURCE_SUBSETS:
  76             self.get_network_resources_facts(
  77                 FACT_RESOURCE_SUBSETS, resource_facts_type, data
  78             )
  79         if self.VALID_LEGACY_GATHER_SUBSETS:
  80             self.get_network_legacy_facts(
  81                 FACT_LEGACY_SUBSETS, legacy_facts_type
  82             )
  83         return self.ansible_facts, self._warnings
