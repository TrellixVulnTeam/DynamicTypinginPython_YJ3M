  57     def get_lldp_interfaces_facts(self):
  58         """ Get the 'facts' (the current configuration)
  59 
  60         :rtype: A dictionary
  61         :returns: The current configuration as a dictionary
  62         """
  63         facts, _warnings = Facts(self._module).get_facts(
  64             self.gather_subset, self.gather_network_resources
  65         )
  66         lldp_interfaces_facts = facts["ansible_network_resources"].get(
  67             "lldp_interfaces"
  68         )
  69         if not lldp_interfaces_facts:
  70             return []
  71         return lldp_interfaces_facts

  73     def execute_module(self):
  74         """ Execute the module
  75 
  76         :rtype: A dictionary
  77         :returns: The result from module execution
  78         """
  79         result = {"changed": False}
  80         commands = list()
  81         warnings = list()
  82         existing_lldp_interfaces_facts = self.get_lldp_interfaces_facts()
  83         commands.extend(self.set_config(existing_lldp_interfaces_facts))
  84         if commands:
  85             if self._module.check_mode:
  86                 resp = self._connection.edit_config(commands, commit=False)
  87             else:
  88                 resp = self._connection.edit_config(commands)
  89             result["changed"] = True
  90 
  91         result["commands"] = commands
  92 
  93         if self._module._diff:
  94             result["diff"] = resp["diff"] if result["changed"] else None
  95 
  96         changed_lldp_interfaces_facts = self.get_lldp_interfaces_facts()
  97         result["before"] = existing_lldp_interfaces_facts
  98         if result["changed"]:
  99             result["after"] = changed_lldp_interfaces_facts
 100 
 101         result["warnings"] = warnings
 102         return result

 104     def set_config(self, existing_lldp_interfaces_facts):
 105         """ Collect the configuration from the args passed to the module,
 106             collect the current configuration (as a dict from facts)
 107 
 108         :rtype: A list
 109         :returns: the commands necessary to migrate the current configuration
 110                   to the desired configuration
 111         """
 112         want = self._module.params["config"]
 113         have = existing_lldp_interfaces_facts
 114         resp = self.set_state(want, have)
 115         return to_list(resp)

 117     def set_state(self, want, have):
 118         """ Select the appropriate function based on the state provided
 119 
 120         :param want: the desired configuration as a dictionary
 121         :param have: the current configuration as a dictionary
 122         :rtype: A list
 123         :returns: the commands necessary to migrate the current configuration
 124                   to the desired configuration
 125         """
 126         commands = []
 127         state = self._module.params["state"]
 128         if state in ("merged", "replaced", "overridden") and not want:
 129             self._module.fail_json(
 130                 msg="value of config parameter must not be empty for state {0}".format(
 131                     state
 132                 )
 133             )
 134         if state == "overridden":
 135             commands.extend(self._state_overridden(want=want, have=have))
 136         elif state == "deleted":
 137             if want:
 138                 for item in want:
 139                     name = item["name"]
 140                     have_item = search_obj_in_list(name, have)
 141                     commands.extend(
 142                         self._state_deleted(want=None, have=have_item)
 143                     )
 144             else:
 145                 for have_item in have:
 146                     commands.extend(
 147                         self._state_deleted(want=None, have=have_item)
 148                     )
 149         else:
 150             for want_item in want:
 151                 name = want_item["name"]
 152                 have_item = search_obj_in_list(name, have)
 153                 if state == "merged":
 154                     commands.extend(
 155                         self._state_merged(want=want_item, have=have_item)
 156                     )
 157                 else:
 158                     commands.extend(
 159                         self._state_replaced(want=want_item, have=have_item)
 160                     )
 161         return commands
 162 
 163     def _state_replaced(self, want, have):
 164         """ The command generator when state is replaced
 165 
 166         :rtype: A list
 167         :returns: the commands necessary to migrate the current configuration
 168                   to the desired configuration
 169         """
 170         commands = []
 171         if have:
 172             commands.extend(self._state_deleted(want, have))
 173         commands.extend(self._state_merged(want, have))
 174         return commands
 175 
 176     def _state_overridden(self, want, have):
 177         """ The command generator when state is overridden
 178 
 179         :rtype: A list
 180         :returns: the commands necessary to migrate the current configuration
 181                   to the desired configuration
 182         """
 183         commands = []
 184         for have_item in have:
 185             lldp_name = have_item["name"]
 186             lldp_in_want = search_obj_in_list(lldp_name, want)
 187             if not lldp_in_want:
 188                 commands.append(
 189                     self._compute_command(have_item["name"], remove=True)
 190                 )
 191 
 192         for want_item in want:
 193             name = want_item["name"]
 194             lldp_in_have = search_obj_in_list(name, have)
 195             commands.extend(self._state_replaced(want_item, lldp_in_have))
 196         return commands
 197 
 198     def _state_merged(self, want, have):
 199         """ The command generator when state is merged
 200 
 201         :rtype: A list
 202         :returns: the commands necessary to merge the provided into
 203                   the current configuration
 204         """
 205         commands = []
 206         if have:
 207             commands.extend(self._render_updates(want, have))
 208         else:
 209             commands.extend(self._render_set_commands(want))
 210         return commands
 211 
 212     def _state_deleted(self, want, have):
 213         """ The command generator when state is deleted
 214 
 215         :rtype: A list
 216         :returns: the commands necessary to remove the current configuration
 217                   of the provided objects
 218         """
 219         commands = []
 220         if want:
 221             params = Lldp_interfaces.params
 222             for attrib in params:
 223                 if attrib == "location":
 224                     commands.extend(
 225                         self._update_location(have["name"], want, have)
 226                     )
 227 
 228         elif have:
 229             commands.append(self._compute_command(have["name"], remove=True))
 230         return commands

 232     def _render_updates(self, want, have):
 233         commands = []
 234         lldp_name = have["name"]
 235         commands.extend(self._configure_status(lldp_name, want, have))
 236         commands.extend(self._add_location(lldp_name, want, have))
 237 
 238         return commands

 240     def _render_set_commands(self, want):
 241         commands = []
 242         have = {}
 243         lldp_name = want["name"]
 244         params = Lldp_interfaces.params
 245 
 246         commands.extend(self._add_location(lldp_name, want, have))
 247         for attrib in params:
 248             value = want[attrib]
 249             if value:
 250                 if attrib == "location":
 251                     commands.extend(self._add_location(lldp_name, want, have))
 252                 elif attrib == "enable":
 253                     if not value:
 254                         commands.append(
 255                             self._compute_command(lldp_name, value="disable")
 256                         )
 257                 else:
 258                     commands.append(self._compute_command(lldp_name))
 259 
 260         return commands

 262     def _configure_status(self, name, want_item, have_item):
 263         commands = []
 264         if is_dict_element_present(have_item, "enable"):
 265             temp_have_item = False
 266         else:
 267             temp_have_item = True
 268         if want_item["enable"] != temp_have_item:
 269             if want_item["enable"]:
 270                 commands.append(
 271                     self._compute_command(name, value="disable", remove=True)
 272                 )
 273             else:
 274                 commands.append(self._compute_command(name, value="disable"))
 275         return commands

 277     def _add_location(self, name, want_item, have_item):
 278         commands = []
 279         have_dict = {}
 280         have_ca = {}
 281         set_cmd = name + " location "
 282         want_location_type = want_item.get("location") or {}
 283         have_location_type = have_item.get("location") or {}
 284 
 285         if want_location_type["coordinate_based"]:
 286             want_dict = want_location_type.get("coordinate_based") or {}
 287             if is_dict_element_present(have_location_type, "coordinate_based"):
 288                 have_dict = have_location_type.get("coordinate_based") or {}
 289             location_type = "coordinate-based"
 290             updates = dict_diff(have_dict, want_dict)
 291             for key, value in iteritems(updates):
 292                 if value:
 293                     commands.append(
 294                         self._compute_command(
 295                             set_cmd + location_type, key, str(value)
 296                         )
 297                     )
 298 
 299         elif want_location_type["civic_based"]:
 300             location_type = "civic-based"
 301             want_dict = want_location_type.get("civic_based") or {}
 302             want_ca = want_dict.get("ca_info") or []
 303             if is_dict_element_present(have_location_type, "civic_based"):
 304                 have_dict = have_location_type.get("civic_based") or {}
 305                 have_ca = have_dict.get("ca_info") or []
 306                 if want_dict["country_code"] != have_dict["country_code"]:
 307                     commands.append(
 308                         self._compute_command(
 309                             set_cmd + location_type,
 310                             "country-code",
 311                             str(want_dict["country_code"]),
 312                         )
 313                     )
 314             else:
 315                 commands.append(
 316                     self._compute_command(
 317                         set_cmd + location_type,
 318                         "country-code",
 319                         str(want_dict["country_code"]),
 320                     )
 321                 )
 322             commands.extend(self._add_civic_address(name, want_ca, have_ca))
 323 
 324         elif want_location_type["elin"]:
 325             location_type = "elin"
 326             if is_dict_element_present(have_location_type, "elin"):
 327                 if want_location_type.get("elin") != have_location_type.get(
 328                     "elin"
 329                 ):
 330                     commands.append(
 331                         self._compute_command(
 332                             set_cmd + location_type,
 333                             value=str(want_location_type["elin"]),
 334                         )
 335                     )
 336             else:
 337                 commands.append(
 338                     self._compute_command(
 339                         set_cmd + location_type,
 340                         value=str(want_location_type["elin"]),
 341                     )
 342                 )
 343         return commands

 345     def _update_location(self, name, want_item, have_item):
 346         commands = []
 347         del_cmd = name + " location"
 348         want_location_type = want_item.get("location") or {}
 349         have_location_type = have_item.get("location") or {}
 350 
 351         if want_location_type["coordinate_based"]:
 352             want_dict = want_location_type.get("coordinate_based") or {}
 353             if is_dict_element_present(have_location_type, "coordinate_based"):
 354                 have_dict = have_location_type.get("coordinate_based") or {}
 355                 location_type = "coordinate-based"
 356                 for key, value in iteritems(have_dict):
 357                     only_in_have = key_value_in_dict(key, value, want_dict)
 358                     if not only_in_have:
 359                         commands.append(
 360                             self._compute_command(
 361                                 del_cmd + location_type, key, str(value), True
 362                             )
 363                         )
 364             else:
 365                 commands.append(self._compute_command(del_cmd, remove=True))
 366 
 367         elif want_location_type["civic_based"]:
 368             want_dict = want_location_type.get("civic_based") or {}
 369             want_ca = want_dict.get("ca_info") or []
 370             if is_dict_element_present(have_location_type, "civic_based"):
 371                 have_dict = have_location_type.get("civic_based") or {}
 372                 have_ca = have_dict.get("ca_info")
 373                 commands.extend(
 374                     self._update_civic_address(name, want_ca, have_ca)
 375                 )
 376             else:
 377                 commands.append(self._compute_command(del_cmd, remove=True))
 378 
 379         else:
 380             if is_dict_element_present(have_location_type, "elin"):
 381                 if want_location_type.get("elin") != have_location_type.get(
 382                     "elin"
 383                 ):
 384                     commands.append(
 385                         self._compute_command(del_cmd, remove=True)
 386                     )
 387             else:
 388                 commands.append(self._compute_command(del_cmd, remove=True))
 389         return commands

 391     def _add_civic_address(self, name, want, have):
 392         commands = []
 393         for item in want:
 394             ca_type = item["ca_type"]
 395             ca_value = item["ca_value"]
 396             obj_in_have = search_dict_tv_in_list(
 397                 ca_type, ca_value, have, "ca_type", "ca_value"
 398             )
 399             if not obj_in_have:
 400                 commands.append(
 401                     self._compute_command(
 402                         key=name + " location civic-based ca-type",
 403                         attrib=str(ca_type) + " ca-value",
 404                         value=ca_value,
 405                     )
 406                 )
 407         return commands

 409     def _update_civic_address(self, name, want, have):
 410         commands = []
 411         for item in have:
 412             ca_type = item["ca_type"]
 413             ca_value = item["ca_value"]
 414             in_want = search_dict_tv_in_list(
 415                 ca_type, ca_value, want, "ca_type", "ca_value"
 416             )
 417             if not in_want:
 418                 commands.append(
 419                     self._compute_command(
 420                         name,
 421                         "location civic-based ca-type",
 422                         str(ca_type),
 423                         remove=True,
 424                     )
 425                 )
 426         return commands

 428     def _compute_command(self, key, attrib=None, value=None, remove=False):
 429         if remove:
 430             cmd = "delete service lldp interface "
 431         else:
 432             cmd = "set service lldp interface "
 433         cmd += key
 434         if attrib:
 435             cmd += " " + attrib
 436         if value:
 437             cmd += " '" + value + "'"
 438         return cmd
