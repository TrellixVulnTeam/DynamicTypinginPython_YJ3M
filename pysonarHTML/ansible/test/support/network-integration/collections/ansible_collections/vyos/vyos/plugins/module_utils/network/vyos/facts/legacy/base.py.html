  23 class LegacyFactsBase(object):
  24 
  25     COMMANDS = frozenset()
  26 
  27     def __init__(self, module):
  28         self.module = module
  29         self.facts = dict()
  30         self.warnings = list()
  31         self.responses = None
  32 
  33     def populate(self):
  34         self.responses = run_commands(self.module, list(self.COMMANDS))

  37 class Default(LegacyFactsBase):
  38 
  39     COMMANDS = [
  40         "show version",
  41     ]
  42 
  43     def populate(self):
  44         super(Default, self).populate()
  45         data = self.responses[0]
  46         self.facts["serialnum"] = self.parse_serialnum(data)
  47         self.facts.update(self.platform_facts())
  48 
  49     def parse_serialnum(self, data):
  50         match = re.search(r"HW S/N:\s+(\S+)", data)
  51         if match:
  52             return match.group(1)
  53 
  54     def platform_facts(self):
  55         platform_facts = {}
  56 
  57         resp = get_capabilities(self.module)
  58         device_info = resp["device_info"]
  59 
  60         platform_facts["system"] = device_info["network_os"]
  61 
  62         for item in ("model", "image", "version", "platform", "hostname"):
  63             val = device_info.get("network_os_%s" % item)
  64             if val:
  65                 platform_facts[item] = val
  66 
  67         platform_facts["api"] = resp["network_api"]
  68         platform_facts["python_version"] = platform.python_version()
  69 
  70         return platform_facts

  73 class Config(LegacyFactsBase):
  74 
  75     COMMANDS = [
  76         "show configuration commands",
  77         "show system commit",
  78     ]
  79 
  80     def populate(self):
  81         super(Config, self).populate()
  82 
  83         self.facts["config"] = self.responses
  84 
  85         commits = self.responses[1]
  86         entries = list()
  87         entry = None
  88 
  89         for line in commits.split("\n"):
  90             match = re.match(r"(\d+)\s+(.+)by(.+)via(.+)", line)
  91             if match:
  92                 if entry:
  93                     entries.append(entry)
  94 
  95                 entry = dict(
  96                     revision=match.group(1),
  97                     datetime=match.group(2),
  98                     by=str(match.group(3)).strip(),
  99                     via=str(match.group(4)).strip(),
 100                     comment=None,
 101                 )
 102             else:
 103                 entry["comment"] = line.strip()
 104 
 105         self.facts["commits"] = entries

 108 class Neighbors(LegacyFactsBase):
 109 
 110     COMMANDS = [
 111         "show lldp neighbors",
 112         "show lldp neighbors detail",
 113     ]
 114 
 115     def populate(self):
 116         super(Neighbors, self).populate()
 117 
 118         all_neighbors = self.responses[0]
 119         if "LLDP not configured" not in all_neighbors:
 120             neighbors = self.parse(self.responses[1])
 121             self.facts["neighbors"] = self.parse_neighbors(neighbors)
 122 
 123     def parse(self, data):
 124         parsed = list()
 125         values = None
 126         for line in data.split("\n"):
 127             if not line:
 128                 continue
 129             elif line[0] == " ":
 130                 values += "\n%s" % line
 131             elif line.startswith("Interface"):
 132                 if values:
 133                     parsed.append(values)
 134                 values = line
 135         if values:
 136             parsed.append(values)
 137         return parsed
 138 
 139     def parse_neighbors(self, data):
 140         facts = dict()
 141         for item in data:
 142             interface = self.parse_interface(item)
 143             host = self.parse_host(item)
 144             port = self.parse_port(item)
 145             if interface not in facts:
 146                 facts[interface] = list()
 147             facts[interface].append(dict(host=host, port=port))
 148         return facts
 149 
 150     def parse_interface(self, data):
 151         match = re.search(r"^Interface:\s+(\S+),", data)
 152         return match.group(1)
 153 
 154     def parse_host(self, data):
 155         match = re.search(r"SysName:\s+(.+)$", data, re.M)
 156         if match:
 157             return match.group(1)
 158 
 159     def parse_port(self, data):
 160         match = re.search(r"PortDescr:\s+(.+)$", data, re.M)
 161         if match:
 162             return match.group(1)
