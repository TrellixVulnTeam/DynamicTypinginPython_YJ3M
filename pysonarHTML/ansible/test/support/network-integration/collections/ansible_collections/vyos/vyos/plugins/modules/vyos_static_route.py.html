 134 def spec_to_commands(updates, module):
 135     commands = list()
 136     want, have = updates
 137     for w in want:
 138         prefix = w["prefix"]
 139         mask = w["mask"]
 140         next_hop = w["next_hop"]
 141         admin_distance = w["admin_distance"]
 142         state = w["state"]
 143         del w["state"]
 144 
 145         if state == "absent" and w in have:
 146             commands.append(
 147                 "delete protocols static route %s/%s" % (prefix, mask)
 148             )
 149         elif state == "present" and w not in have:
 150             cmd = "set protocols static route %s/%s next-hop %s" % (
 151                 prefix,
 152                 mask,
 153                 next_hop,
 154             )
 155             if admin_distance != "None":
 156                 cmd += " distance %s" % (admin_distance)
 157             commands.append(cmd)
 158 
 159     return commands

 162 def config_to_dict(module):
 163     data = get_config(module)
 164     obj = []
 165 
 166     for line in data.split("\n"):
 167         if line.startswith("set protocols static route"):
 168             match = re.search(r"static route (\S+)", line, re.M)
 169             prefix = match.group(1).split("/")[0]
 170             mask = match.group(1).split("/")[1]
 171             if "next-hop" in line:
 172                 match_hop = re.search(r"next-hop (\S+)", line, re.M)
 173                 next_hop = match_hop.group(1).strip("'")
 174 
 175                 match_distance = re.search(r"distance (\S+)", line, re.M)
 176                 if match_distance is not None:
 177                     admin_distance = match_distance.group(1)[1:-1]
 178                 else:
 179                     admin_distance = None
 180 
 181                 if admin_distance is not None:
 182                     obj.append(
 183                         {
 184                             "prefix": prefix,
 185                             "mask": mask,
 186                             "next_hop": next_hop,
 187                             "admin_distance": admin_distance,
 188                         }
 189                     )
 190                 else:
 191                     obj.append(
 192                         {
 193                             "prefix": prefix,
 194                             "mask": mask,
 195                             "next_hop": next_hop,
 196                             "admin_distance": "None",
 197                         }
 198                     )
 199 
 200     return obj

 203 def map_params_to_obj(module, required_together=None):
 204     obj = []
 205     aggregate = module.params.get("aggregate")
 206     if aggregate:
 207         for item in aggregate:
 208             for key in item:
 209                 if item.get(key) is None:
 210                     item[key] = module.params[key]
 211 
 212             module._check_required_together(required_together, item)
 213             d = item.copy()
 214             if "/" in d["prefix"]:
 215                 d["mask"] = d["prefix"].split("/")[1]
 216                 d["prefix"] = d["prefix"].split("/")[0]
 217 
 218             if "admin_distance" in d:
 219                 d["admin_distance"] = str(d["admin_distance"])
 220 
 221             obj.append(d)
 222     else:
 223         prefix = module.params["prefix"].strip()
 224         if "/" in prefix:
 225             mask = prefix.split("/")[1]
 226             prefix = prefix.split("/")[0]
 227         else:
 228             mask = module.params["mask"].strip()
 229         next_hop = module.params["next_hop"].strip()
 230         admin_distance = str(module.params["admin_distance"])
 231         state = module.params["state"]
 232 
 233         obj.append(
 234             {
 235                 "prefix": prefix,
 236                 "mask": mask,
 237                 "next_hop": next_hop,
 238                 "admin_distance": admin_distance,
 239                 "state": state,
 240             }
 241         )
 242 
 243     return obj

 246 def main():
 247     """ main entry point for module execution
 248     """
 249     element_spec = dict(
 250         prefix=dict(type="str"),
 251         mask=dict(type="str"),
 252         next_hop=dict(type="str"),
 253         admin_distance=dict(type="int"),
 254         state=dict(default="present", choices=["present", "absent"]),
 255     )
 256 
 257     aggregate_spec = deepcopy(element_spec)
 258     aggregate_spec["prefix"] = dict(required=True)
 259 
 260     # remove default in aggregate spec, to handle common arguments
 261     remove_default_spec(aggregate_spec)
 262 
 263     argument_spec = dict(
 264         aggregate=dict(type="list", elements="dict", options=aggregate_spec),
 265     )
 266 
 267     argument_spec.update(element_spec)
 268     argument_spec.update(vyos_argument_spec)
 269 
 270     required_one_of = [["aggregate", "prefix"]]
 271     required_together = [["prefix", "next_hop"]]
 272     mutually_exclusive = [["aggregate", "prefix"]]
 273 
 274     module = AnsibleModule(
 275         argument_spec=argument_spec,
 276         required_one_of=required_one_of,
 277         required_together=required_together,
 278         mutually_exclusive=mutually_exclusive,
 279         supports_check_mode=True,
 280     )
 281 
 282     warnings = list()
 283 
 284     result = {"changed": False}
 285     if warnings:
 286         result["warnings"] = warnings
 287     want = map_params_to_obj(module, required_together=required_together)
 288     have = config_to_dict(module)
 289 
 290     commands = spec_to_commands((want, have), module)
 291     result["commands"] = commands
 292 
 293     if commands:
 294         commit = not module.check_mode
 295         load_config(module, commands, commit=commit)
 296         result["changed"] = True
 297 
 298     module.exit_json(**result)
