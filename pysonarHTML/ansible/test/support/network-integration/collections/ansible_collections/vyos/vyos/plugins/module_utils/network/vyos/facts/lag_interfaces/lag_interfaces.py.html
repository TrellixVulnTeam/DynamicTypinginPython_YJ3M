  44     def populate_facts(self, connection, ansible_facts, data=None):
  45         """ Populate the facts for lag_interfaces
  46         :param module: the module instance
  47         :param connection: the device connection
  48         :param data: previously collected conf
  49         :rtype: dictionary
  50         :returns: facts
  51         """
  52         if not data:
  53             data = connection.get_config()
  54 
  55         objs = []
  56         lag_names = findall(r"^set interfaces bonding (\S+)", data, M)
  57         if lag_names:
  58             for lag in set(lag_names):
  59                 lag_regex = r" %s .+$" % lag
  60                 cfg = findall(lag_regex, data, M)
  61                 obj = self.render_config(cfg)
  62 
  63                 output = connection.run_commands(
  64                     ["show interfaces bonding " + lag + " slaves"]
  65                 )
  66                 lines = output[0].splitlines()
  67                 members = []
  68                 member = {}
  69                 if len(lines) > 1:
  70                     for line in lines[2:]:
  71                         splitted_line = line.split()
  72 
  73                         if len(splitted_line) > 1:
  74                             member["member"] = splitted_line[0]
  75                             members.append(member)
  76                         else:
  77                             members = []
  78                         member = {}
  79                 obj["name"] = lag.strip("'")
  80                 if members:
  81                     obj["members"] = members
  82 
  83                 if obj:
  84                     objs.append(obj)
  85 
  86         facts = {}
  87         if objs:
  88             facts["lag_interfaces"] = []
  89             params = utils.validate_config(
  90                 self.argument_spec, {"config": objs}
  91             )
  92             for cfg in params["config"]:
  93                 facts["lag_interfaces"].append(utils.remove_empties(cfg))
  94 
  95         ansible_facts["ansible_network_resources"].update(facts)
  96         return ansible_facts

  98     def render_config(self, conf):
  99         """
 100         Render config as dictionary structure and delete keys
 101           from spec for null values
 102 
 103         :param spec: The facts tree, generated from the argspec
 104         :param conf: The configuration
 105         :rtype: dictionary
 106         :returns: The generated config
 107         """
 108         arp_monitor_conf = "\n".join(
 109             filter(lambda x: ("arp-monitor" in x), conf)
 110         )
 111         hash_policy_conf = "\n".join(
 112             filter(lambda x: ("hash-policy" in x), conf)
 113         )
 114         lag_conf = "\n".join(filter(lambda x: ("bond" in x), conf))
 115         config = self.parse_attribs(["mode", "primary"], lag_conf)
 116         config["arp_monitor"] = self.parse_arp_monitor(arp_monitor_conf)
 117         config["hash_policy"] = self.parse_hash_policy(hash_policy_conf)
 118 
 119         return utils.remove_empties(config)
 120 
 121     def parse_attribs(self, attribs, conf):
 122         config = {}
 123         for item in attribs:
 124             value = utils.parse_conf_arg(conf, item)
 125             if value:
 126                 config[item] = value.strip("'")
 127             else:
 128                 config[item] = None
 129         return utils.remove_empties(config)
 130 
 131     def parse_arp_monitor(self, conf):
 132         arp_monitor = None
 133         if conf:
 134             arp_monitor = {}
 135             target_list = []
 136             interval = search(r"^.*arp-monitor interval (.+)", conf, M)
 137             targets = findall(r"^.*arp-monitor target '(.+)'", conf, M)
 138             if targets:
 139                 for target in targets:
 140                     target_list.append(target)
 141                 arp_monitor["target"] = target_list
 142             if interval:
 143                 value = interval.group(1).strip("'")
 144                 arp_monitor["interval"] = int(value)
 145         return arp_monitor
 146 
 147     def parse_hash_policy(self, conf):
 148         hash_policy = None
 149         if conf:
 150             hash_policy = search(r"^.*hash-policy (.+)", conf, M)
 151             hash_policy = hash_policy.group(1).strip("'")
 152         return hash_policy
