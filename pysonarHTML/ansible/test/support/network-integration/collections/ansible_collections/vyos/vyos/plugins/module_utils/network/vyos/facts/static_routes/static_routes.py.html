  33     def __init__(self, module, subspec="config", options="options"):

  50     def populate_facts(self, connection, ansible_facts, data=None):
  51         """ Populate the facts for static_routes
  52         :param connection: the device connection
  53         :param ansible_facts: Facts dictionary
  54         :param data: previously collected conf
  55         :rtype: dictionary
  56         :returns: facts
  57         """
  58         if not data:
  59             data = self.get_device_data(connection)
  60             # typically data is populated from the current device configuration
  61             # data = connection.get('show running-config | section ^interface')
  62             # using mock data instead
  63         objs = []
  64         r_v4 = []
  65         r_v6 = []
  66         af = []
  67         static_routes = findall(
  68             r"set protocols static route(6)? (\S+)", data, M
  69         )
  70         if static_routes:
  71             for route in set(static_routes):
  72                 route_regex = r" %s .+$" % route[1]
  73                 cfg = findall(route_regex, data, M)
  74                 sr = self.render_config(cfg)
  75                 sr["dest"] = route[1].strip("'")
  76                 afi = self.get_afi(sr["dest"])
  77                 if afi == "ipv4":
  78                     r_v4.append(sr)
  79                 else:
  80                     r_v6.append(sr)
  81             if r_v4:
  82                 afi_v4 = {"afi": "ipv4", "routes": r_v4}
  83                 af.append(afi_v4)
  84             if r_v6:
  85                 afi_v6 = {"afi": "ipv6", "routes": r_v6}
  86                 af.append(afi_v6)
  87             config = {"address_families": af}
  88             if config:
  89                 objs.append(config)
  90 
  91         ansible_facts["ansible_network_resources"].pop("static_routes", None)
  92         facts = {}
  93         if objs:
  94             facts["static_routes"] = []
  95             params = utils.validate_config(
  96                 self.argument_spec, {"config": objs}
  97             )
  98             for cfg in params["config"]:
  99                 facts["static_routes"].append(utils.remove_empties(cfg))
 100 
 101         ansible_facts["ansible_network_resources"].update(facts)
 102         return ansible_facts

 104     def render_config(self, conf):
 105         """
 106         Render config as dictionary structure and delete keys
 107           from spec for null values
 108 
 109         :param spec: The facts tree, generated from the argspec
 110         :param conf: The configuration
 111         :rtype: dictionary
 112         :returns: The generated config
 113         """
 114         next_hops_conf = "\n".join(filter(lambda x: ("next-hop" in x), conf))
 115         blackhole_conf = "\n".join(filter(lambda x: ("blackhole" in x), conf))
 116         routes_dict = {
 117             "blackhole_config": self.parse_blackhole(blackhole_conf),
 118             "next_hops": self.parse_next_hop(next_hops_conf),
 119         }
 120         return routes_dict
 121 
 122     def parse_blackhole(self, conf):
 123         blackhole = None
 124         if conf:
 125             distance = search(r"^.*blackhole distance (.\S+)", conf, M)
 126             bh = conf.find("blackhole")
 127             if distance is not None:
 128                 blackhole = {}
 129                 value = distance.group(1).strip("'")
 130                 blackhole["distance"] = int(value)
 131             elif bh:
 132                 blackhole = {}
 133                 blackhole["type"] = "blackhole"
 134         return blackhole
 135 
 136     def get_afi(self, address):
 137         route_type = get_route_type(address)
 138         if route_type == "route":
 139             return "ipv4"
 140         elif route_type == "route6":
 141             return "ipv6"
 142 
 143     def parse_next_hop(self, conf):
 144         nh_list = None
 145         if conf:
 146             nh_list = []
 147             hop_list = findall(r"^.*next-hop (.+)", conf, M)
 148             if hop_list:
 149                 for hop in hop_list:
 150                     distance = search(r"^.*distance (.\S+)", hop, M)
 151                     interface = search(r"^.*interface (.\S+)", hop, M)
 152 
 153                     dis = hop.find("disable")
 154                     hop_info = hop.split(" ")
 155                     nh_info = {
 156                         "forward_router_address": hop_info[0].strip("'")
 157                     }
 158                     if interface:
 159                         nh_info["interface"] = interface.group(1).strip("'")
 160                     if distance:
 161                         value = distance.group(1).strip("'")
 162                         nh_info["admin_distance"] = int(value)
 163                     elif dis >= 1:
 164                         nh_info["enabled"] = False
 165                     for element in nh_list:
 166                         if (
 167                             element["forward_router_address"]
 168                             == nh_info["forward_router_address"]
 169                         ):
 170                             if "interface" in nh_info.keys():
 171                                 element["interface"] = nh_info["interface"]
 172                             if "admin_distance" in nh_info.keys():
 173                                 element["admin_distance"] = nh_info[
 174                                     "admin_distance"
 175                                 ]
 176                             if "enabled" in nh_info.keys():
 177                                 element["enabled"] = nh_info["enabled"]
 178                             nh_info = None
 179                     if nh_info is not None:
 180                         nh_list.append(nh_info)
 181         return nh_list
