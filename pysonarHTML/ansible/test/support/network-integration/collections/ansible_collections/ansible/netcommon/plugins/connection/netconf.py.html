 204 class Connection(NetworkConnectionBase):
 205     """NetConf connections"""
 206 
 207     transport = "ansible.netcommon.netconf"
 208     has_pipelining = False
 209 
 210     def __init__(self, play_context, new_stdin, *args, **kwargs):
 211         super(Connection, self).__init__(
 212             play_context, new_stdin, *args, **kwargs
 213         )
 214 
 215         # If network_os is not specified then set the network os to auto
 216         # This will be used to trigger the use of guess_network_os when connecting.
 217         self._network_os = self._network_os or "auto"
 218 
 219         self.netconf = netconf_loader.get(self._network_os, self)
 220         if self.netconf:
 221             self._sub_plugin = {
 222                 "type": "netconf",
 223                 "name": self.netconf._load_name,
 224                 "obj": self.netconf,
 225             }
 226             self.queue_message(
 227                 "vvvv",
 228                 "loaded netconf plugin %s from path %s for network_os %s"
 229                 % (
 230                     self.netconf._load_name,
 231                     self.netconf._original_path,
 232                     self._network_os,
 233                 ),
 234             )
 235         else:
 236             self.netconf = netconf_loader.get("default", self)
 237             self._sub_plugin = {
 238                 "type": "netconf",
 239                 "name": "default",
 240                 "obj": self.netconf,
 241             }
 242             self.queue_message(
 243                 "display",
 244                 "unable to load netconf plugin for network_os %s, falling back to default plugin"
 245                 % self._network_os,
 246             )
 247 
 248         self.queue_message("log", "network_os is set to %s" % self._network_os)
 249         self._manager = None
 250         self.key_filename = None
 251         self._ssh_config = None
 252 
 253     def exec_command(self, cmd, in_data=None, sudoable=True):
 254         """Sends the request to the node and returns the reply
 255         The method accepts two forms of request.  The first form is as a byte
 256         string that represents xml string be send over netconf session.
 257         The second form is a json-rpc (2.0) byte string.
 258         """
 259         if self._manager:
 260             # to_ele operates on native strings
 261             request = to_ele(to_native(cmd, errors="surrogate_or_strict"))
 262 
 263             if request is None:
 264                 return "unable to parse request"
 265 
 266             try:
 267                 reply = self._manager.rpc(request)
 268             except RPCError as exc:
 269                 error = self.internal_error(
 270                     data=to_text(to_xml(exc.xml), errors="surrogate_or_strict")
 271                 )
 272                 return json.dumps(error)
 273 
 274             return reply.data_xml
 275         else:
 276             return super(Connection, self).exec_command(cmd, in_data, sudoable)
 277 
 278     @property
 279     @ensure_connect
 280     def manager(self):
 281         return self._manager
 282 
 283     def _connect(self):
 284         if not HAS_NCCLIENT:
 285             raise AnsibleError(
 286                 "%s: %s"
 287                 % (
 288                     missing_required_lib("ncclient"),
 289                     to_native(NCCLIENT_IMP_ERR),
 290                 )
 291             )
 292 
 293         self.queue_message("log", "ssh connection done, starting ncclient")
 294 
 295         allow_agent = True
 296         if self._play_context.password is not None:
 297             allow_agent = False
 298         setattr(self._play_context, "allow_agent", allow_agent)
 299 
 300         self.key_filename = (
 301             self._play_context.private_key_file
 302             or self.get_option("private_key_file")
 303         )
 304         if self.key_filename:
 305             self.key_filename = str(os.path.expanduser(self.key_filename))
 306 
 307         self._ssh_config = self.get_option("netconf_ssh_config")
 308         if self._ssh_config in BOOLEANS_TRUE:
 309             self._ssh_config = True
 310         elif self._ssh_config in BOOLEANS_FALSE:
 311             self._ssh_config = None
 312 
 313         # Try to guess the network_os if the network_os is set to auto
 314         if self._network_os == "auto":
 315             for cls in netconf_loader.all(class_only=True):
 316                 network_os = cls.guess_network_os(self)
 317                 if network_os:
 318                     self.queue_message(
 319                         "vvv", "discovered network_os %s" % network_os
 320                     )
 321                     self._network_os = network_os
 322 
 323         # If we have tried to detect the network_os but were unable to i.e. network_os is still 'auto'
 324         # then use default as the network_os
 325 
 326         if self._network_os == "auto":
 327             # Network os not discovered. Set it to default
 328             self.queue_message(
 329                 "vvv",
 330                 "Unable to discover network_os. Falling back to default.",
 331             )
 332             self._network_os = "default"
 333         try:
 334             ncclient_device_handler = self.netconf.get_option(
 335                 "ncclient_device_handler"
 336             )
 337         except KeyError:
 338             ncclient_device_handler = "default"
 339         self.queue_message(
 340             "vvv",
 341             "identified ncclient device handler: %s."
 342             % ncclient_device_handler,
 343         )
 344         device_params = {"name": ncclient_device_handler}
 345 
 346         try:
 347             port = self._play_context.port or 830
 348             self.queue_message(
 349                 "vvv",
 350                 "ESTABLISH NETCONF SSH CONNECTION FOR USER: %s on PORT %s TO %s WITH SSH_CONFIG = %s"
 351                 % (
 352                     self._play_context.remote_user,
 353                     port,
 354                     self._play_context.remote_addr,
 355                     self._ssh_config,
 356                 ),
 357             )
 358             self._manager = manager.connect(
 359                 host=self._play_context.remote_addr,
 360                 port=port,
 361                 username=self._play_context.remote_user,
 362                 password=self._play_context.password,
 363                 key_filename=self.key_filename,
 364                 hostkey_verify=self.get_option("host_key_checking"),
 365                 look_for_keys=self.get_option("look_for_keys"),
 366                 device_params=device_params,
 367                 allow_agent=self._play_context.allow_agent,
 368                 timeout=self.get_option("persistent_connect_timeout"),
 369                 ssh_config=self._ssh_config,
 370             )
 371 
 372             self._manager._timeout = self.get_option(
 373                 "persistent_command_timeout"
 374             )
 375         except SSHUnknownHostError as exc:
 376             raise AnsibleConnectionFailure(to_native(exc))
 377         except ImportError:
 378             raise AnsibleError(
 379                 "connection=netconf is not supported on {0}".format(
 380                     self._network_os
 381                 )
 382             )
 383 
 384         if not self._manager.connected:
 385             return 1, b"", b"not connected"
 386 
 387         self.queue_message(
 388             "log", "ncclient manager object created successfully"
 389         )
 390 
 391         self._connected = True
 392 
 393         super(Connection, self)._connect()
 394 
 395         return (
 396             0,
 397             to_bytes(self._manager.session_id, errors="surrogate_or_strict"),
 398             b"",
 399         )
 400 
 401     def close(self):
 402         if self._manager:
 403             self._manager.close_session()
 404         super(Connection, self).close()
