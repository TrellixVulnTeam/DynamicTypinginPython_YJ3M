  51 class ConfigLine(object):
  52     def __init__(self, raw):
  53         self.text = str(raw).strip()
  54         self.raw = raw
  55         self._children = list()
  56         self._parents = list()
  57 
  58     def __str__(self):
  59         return self.raw
  60 
  61     def __eq__(self, other):
  62         return self.line == other.line
  63 
  64     def __ne__(self, other):
  65         return not self.__eq__(other)
  66 
  67     def __getitem__(self, key):
  68         for item in self._children:
  69             if item.text == key:
  70                 return item
  71         raise KeyError(key)
  72 
  73     @property
  74     def line(self):
  75         line = self.parents
  76         line.append(self.text)
  77         return " ".join(line)
  78 
  79     @property
  80     def children(self):
  81         return _obj_to_text(self._children)
  82 
  83     @property
  84     def child_objs(self):
  85         return self._children
  86 
  87     @property
  88     def parents(self):
  89         return _obj_to_text(self._parents)
  90 
  91     @property
  92     def path(self):
  93         config = _obj_to_raw(self._parents)
  94         config.append(self.raw)
  95         return "\n".join(config)
  96 
  97     @property
  98     def has_children(self):
  99         return len(self._children) > 0

 102     def has_parents(self):
 103         return len(self._parents) > 0

 105     def add_child(self, obj):
 106         if not isinstance(obj, ConfigLine):
 107             raise AssertionError("child must be of type `ConfigLine`")
 108         self._children.append(obj)

 111 def ignore_line(text, tokens=None):
 112     for item in tokens or DEFAULT_COMMENT_TOKENS:
 113         if text.startswith(item):
 114             return True
 115     for regex in DEFAULT_IGNORE_LINES_RE:
 116         if regex.match(text):
 117             return True

 128 def _obj_to_block(objects, visited=None):
 129     items = list()
 130     for o in objects:
 131         if o not in items:
 132             items.append(o)
 133             for child in o._children:
 134                 if child not in items:
 135                     items.append(child)
 136     return _obj_to_raw(items)

 139 def dumps(objects, output="block", comments=False):
 140     if output == "block":
 141         items = _obj_to_block(objects)
 142     elif output == "commands":
 143         items = _obj_to_text(objects)
 144     elif output == "raw":
 145         items = _obj_to_raw(objects)
 146     else:
 147         raise TypeError("unknown value supplied for keyword output")
 148 
 149     if output == "block":
 150         if comments:
 151             for index, item in enumerate(items):
 152                 nextitem = index + 1
 153                 if (
 154                     nextitem < len(items)
 155                     and not item.startswith(" ")
 156                     and items[nextitem].startswith(" ")
 157                 ):
 158                     item = "!\n%s" % item
 159                 items[index] = item
 160             items.append("!")
 161         items.append("end")
 162 
 163     return "\n".join(items)

 166 class NetworkConfig(object):
 167     def __init__(self, indent=1, contents=None, ignore_lines=None):
 168         self._indent = indent
 169         self._items = list()
 170         self._config_text = None
 171 
 172         if ignore_lines:
 173             for item in ignore_lines:
 174                 if not isinstance(item, Pattern):
 175                     item = re.compile(item)
 176                 DEFAULT_IGNORE_LINES_RE.add(item)
 177 
 178         if contents:
 179             self.load(contents)
 180 
 181     @property
 182     def items(self):
 183         return self._items
 184 
 185     @property
 186     def config_text(self):
 187         return self._config_text

 189     @property
 190     def sha1(self):
 191         sha1 = hashlib.sha1()
 192         sha1.update(to_bytes(str(self), errors="surrogate_or_strict"))
 193         return sha1.digest()
 194 
 195     def __getitem__(self, key):
 196         for line in self:
 197             if line.text == key:
 198                 return line
 199         raise KeyError(key)
 200 
 201     def __iter__(self):
 202         return iter(self._items)
 203 
 204     def __str__(self):
 205         return "\n".join([c.raw for c in self.items])

 207     def __len__(self):
 208         return len(self._items)

 210     def load(self, s):
 211         self._config_text = s
 212         self._items = self.parse(s)

 214     def loadfp(self, fp):
 215         with open(fp) as f:
 216             return self.load(f.read())

 218     def parse(self, lines, comment_tokens=None):
 219         toplevel = re.compile(r"\S")
 220         childline = re.compile(r"^\s*(.+)$")
 221         entry_reg = re.compile(r"([{};])")
 222 
 223         ancestors = list()
 224         config = list()
 225 
 226         indents = [0]
 227 
 228         for linenum, line in enumerate(
 229             to_native(lines, errors="surrogate_or_strict").split("\n")
 230         ):
 231             text = entry_reg.sub("", line).strip()
 232 
 233             cfg = ConfigLine(line)
 234 
 235             if not text or ignore_line(text, comment_tokens):
 236                 continue
 237 
 238             # handle top level commands
 239             if toplevel.match(line):
 240                 ancestors = [cfg]
 241                 indents = [0]
 242 
 243             # handle sub level commands
 244             else:
 245                 match = childline.match(line)
 246                 line_indent = match.start(1)
 247 
 248                 if line_indent < indents[-1]:
 249                     while indents[-1] > line_indent:
 250                         indents.pop()
 251 
 252                 if line_indent > indents[-1]:
 253                     indents.append(line_indent)
 254 
 255                 curlevel = len(indents) - 1
 256                 parent_level = curlevel - 1
 257 
 258                 cfg._parents = ancestors[:curlevel]
 259 
 260                 if curlevel > len(ancestors):
 261                     config.append(cfg)
 262                     continue
 263 
 264                 for i in range(curlevel, len(ancestors)):
 265                     ancestors.pop()
 266 
 267                 ancestors.append(cfg)
 268                 ancestors[parent_level].add_child(cfg)
 269 
 270             config.append(cfg)
 271 
 272         return config

 274     def get_object(self, path):
 275         for item in self.items:
 276             if item.text == path[-1]:
 277                 if item.parents == path[:-1]:
 278                     return item

 280     def get_block(self, path):
 281         if not isinstance(path, list):
 282             raise AssertionError("path argument must be a list object")
 283         obj = self.get_object(path)
 284         if not obj:
 285             raise ValueError("path does not exist in config")
 286         return self._expand_block(obj)

 288     def get_block_config(self, path):
 289         block = self.get_block(path)
 290         return dumps(block, "block")

 292     def _expand_block(self, configobj, S=None):
 293         if S is None:
 294             S = list()
 295         S.append(configobj)
 296         for child in configobj._children:
 297             if child in S:
 298                 continue
 299             self._expand_block(child, S)
 300         return S

 302     def _diff_line(self, other):
 303         updates = list()
 304         for item in self.items:
 305             if item not in other:
 306                 updates.append(item)
 307         return updates
 308 
 309     def _diff_strict(self, other):
 310         updates = list()
 311         # block extracted from other does not have all parents
 312         # but the last one. In case of multiple parents we need
 313         # to add additional parents.
 314         if other and isinstance(other, list) and len(other) > 0:
 315             start_other = other[0]
 316             if start_other.parents:
 317                 for parent in start_other.parents:
 318                     other.insert(0, ConfigLine(parent))
 319         for index, line in enumerate(self.items):
 320             try:
 321                 if str(line).strip() != str(other[index]).strip():
 322                     updates.append(line)
 323             except (AttributeError, IndexError):
 324                 updates.append(line)
 325         return updates
 326 
 327     def _diff_exact(self, other):
 328         updates = list()
 329         if len(other) != len(self.items):
 330             updates.extend(self.items)
 331         else:
 332             for ours, theirs in zip(self.items, other):
 333                 if ours != theirs:
 334                     updates.extend(self.items)
 335                     break
 336         return updates

 338     def difference(self, other, match="line", path=None, replace=None):
 339         """Perform a config diff against the another network config
 340 
 341         :param other: instance of NetworkConfig to diff against
 342         :param match: type of diff to perform.  valid values are 'line',
 343             'strict', 'exact'
 344         :param path: context in the network config to filter the diff
 345         :param replace: the method used to generate the replacement lines.
 346             valid values are 'block', 'line'
 347 
 348         :returns: a string of lines that are different
 349         """
 350         if path and match != "line":
 351             try:
 352                 other = other.get_block(path)
 353             except ValueError:
 354                 other = list()
 355         else:
 356             other = other.items
 357 
 358         # generate a list of ConfigLines that aren't in other
 359         meth = getattr(self, "_diff_%s" % match)
 360         updates = meth(other)
 361 
 362         if replace == "block":
 363             parents = list()
 364             for item in updates:
 365                 if not item.has_parents:
 366                     parents.append(item)
 367                 else:
 368                     for p in item._parents:
 369                         if p not in parents:
 370                             parents.append(p)
 371 
 372             updates = list()
 373             for item in parents:
 374                 updates.extend(self._expand_block(item))
 375 
 376         visited = set()
 377         expanded = list()
 378 
 379         for item in updates:
 380             for p in item._parents:
 381                 if p.line not in visited:
 382                     visited.add(p.line)
 383                     expanded.append(p)
 384             expanded.append(item)
 385             visited.add(item.line)
 386 
 387         return expanded

 389     def add(self, lines, parents=None):
 390         ancestors = list()
 391         offset = 0
 392         obj = None
 393 
 394         # global config command
 395         if not parents:
 396             for line in lines:
 397                 # handle ignore lines
 398                 if ignore_line(line):
 399                     continue
 400 
 401                 item = ConfigLine(line)
 402                 item.raw = line
 403                 if item not in self.items:
 404                     self.items.append(item)
 405 
 406         else:
 407             for index, p in enumerate(parents):
 408                 try:
 409                     i = index + 1
 410                     obj = self.get_block(parents[:i])[0]
 411                     ancestors.append(obj)
 412 
 413                 except ValueError:
 414                     # add parent to config
 415                     offset = index * self._indent
 416                     obj = ConfigLine(p)
 417                     obj.raw = p.rjust(len(p) + offset)
 418                     if ancestors:
 419                         obj._parents = list(ancestors)
 420                         ancestors[-1]._children.append(obj)
 421                     self.items.append(obj)
 422                     ancestors.append(obj)
 423 
 424             # add child objects
 425             for line in lines:
 426                 # handle ignore lines
 427                 if ignore_line(line):
 428                     continue
 429 
 430                 # check if child already exists
 431                 for child in ancestors[-1]._children:
 432                     if child.text == line:
 433                         break
 434                 else:
 435                     offset = len(parents) * self._indent
 436                     item = ConfigLine(line)
 437                     item.raw = line.rjust(len(line) + offset)
 438                     item._parents = ancestors
 439                     ancestors[-1]._children.append(item)
 440                     self.items.append(item)

 443 class CustomNetworkConfig(NetworkConfig):
 444     def items_text(self):
 445         return [item.text for item in self.items]
 446 
 447     def expand_section(self, configobj, S=None):
 448         if S is None:
 449             S = list()
 450         S.append(configobj)
 451         for child in configobj.child_objs:
 452             if child in S:
 453                 continue
 454             self.expand_section(child, S)
 455         return S

 457     def to_block(self, section):
 458         return "\n".join([item.raw for item in section])
 459 
 460     def get_section(self, path):
 461         try:
 462             section = self.get_section_objects(path)
 463             return self.to_block(section)
 464         except ValueError:
 465             return list()
 466 
 467     def get_section_objects(self, path):
 468         if not isinstance(path, list):
 469             path = [path]
 470         obj = self.get_object(path)
 471         if not obj:
 472             raise ValueError("path does not exist in config")
 473         return self.expand_section(obj)
