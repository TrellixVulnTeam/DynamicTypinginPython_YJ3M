  36 class ActionModule(ActionBase):
  37     def run(self, tmp=None, task_vars=None):
  38         socket_path = None
  39         self._get_network_os(task_vars)
  40         persistent_connection = self._play_context.connection.split(".")[-1]
  41 
  42         result = super(ActionModule, self).run(task_vars=task_vars)
  43 
  44         if persistent_connection != "network_cli":
  45             # It is supported only with network_cli
  46             result["failed"] = True
  47             result["msg"] = (
  48                 "connection type %s is not valid for net_get module,"
  49                 " please use fully qualified name of network_cli connection type"
  50                 % self._play_context.connection
  51             )
  52             return result
  53 
  54         try:
  55             src = self._task.args["src"]
  56         except KeyError as exc:
  57             return {
  58                 "failed": True,
  59                 "msg": "missing required argument: %s" % exc,
  60             }
  61 
  62         # Get destination file if specified
  63         dest = self._task.args.get("dest")
  64 
  65         if dest is None:
  66             dest = self._get_default_dest(src)
  67         else:
  68             dest = self._handle_dest_path(dest)
  69 
  70         # Get proto
  71         proto = self._task.args.get("protocol")
  72         if proto is None:
  73             proto = "scp"
  74 
  75         if socket_path is None:
  76             socket_path = self._connection.socket_path
  77 
  78         conn = Connection(socket_path)
  79         sock_timeout = conn.get_option("persistent_command_timeout")
  80 
  81         try:
  82             changed = self._handle_existing_file(
  83                 conn, src, dest, proto, sock_timeout
  84             )
  85             if changed is False:
  86                 result["changed"] = changed
  87                 result["destination"] = dest
  88                 return result
  89         except Exception as exc:
  90             result["msg"] = (
  91                 "Warning: %s idempotency check failed. Check dest" % exc
  92             )
  93 
  94         try:
  95             conn.get_file(
  96                 source=src, destination=dest, proto=proto, timeout=sock_timeout
  97             )
  98         except Exception as exc:
  99             result["failed"] = True
 100             result["msg"] = "Exception received: %s" % exc
 101 
 102         result["changed"] = changed
 103         result["destination"] = dest
 104         return result

 106     def _handle_dest_path(self, dest):
 107         working_path = self._get_working_path()
 108 
 109         if os.path.isabs(dest) or urlsplit("dest").scheme:
 110             dst = dest
 111         else:
 112             dst = self._loader.path_dwim_relative(working_path, "", dest)
 113 
 114         return dst
 115 
 116     def _get_src_filename_from_path(self, src_path):
 117         filename_list = re.split("/|:", src_path)
 118         return filename_list[-1]
 119 
 120     def _get_default_dest(self, src_path):
 121         dest_path = self._get_working_path()
 122         src_fname = self._get_src_filename_from_path(src_path)
 123         filename = "%s/%s" % (dest_path, src_fname)
 124         return filename

 126     def _handle_existing_file(self, conn, source, dest, proto, timeout):
 127         """
 128         Determines whether the source and destination file match.
 129 
 130         :return: False if source and dest both exist and have matching sha1 sums, True otherwise.
 131         """
 132         if not os.path.exists(dest):
 133             return True
 134 
 135         cwd = self._loader.get_basedir()
 136         filename = str(uuid.uuid4())
 137         tmp_dest_file = os.path.join(cwd, filename)
 138         try:
 139             conn.get_file(
 140                 source=source,
 141                 destination=tmp_dest_file,
 142                 proto=proto,
 143                 timeout=timeout,
 144             )
 145         except ConnectionError as exc:
 146             error = to_text(exc)
 147             if error.endswith("No such file or directory"):
 148                 if os.path.exists(tmp_dest_file):
 149                     os.remove(tmp_dest_file)
 150                 return True
 151 
 152         try:
 153             with open(tmp_dest_file, "r") as f:
 154                 new_content = f.read()
 155             with open(dest, "r") as f:
 156                 old_content = f.read()
 157         except (IOError, OSError):
 158             os.remove(tmp_dest_file)
 159             raise
 160 
 161         sha1 = hashlib.sha1()
 162         old_content_b = to_bytes(old_content, errors="surrogate_or_strict")
 163         sha1.update(old_content_b)
 164         checksum_old = sha1.digest()
 165 
 166         sha1 = hashlib.sha1()
 167         new_content_b = to_bytes(new_content, errors="surrogate_or_strict")
 168         sha1.update(new_content_b)
 169         checksum_new = sha1.digest()
 170         os.remove(tmp_dest_file)
 171         if checksum_old == checksum_new:
 172             return False
 173         return True

 175     def _get_working_path(self):
 176         cwd = self._loader.get_basedir()
 177         if self._task._role is not None:
 178             cwd = self._task._role._role_path
 179         return cwd

 181     def _get_network_os(self, task_vars):
 182         if "network_os" in self._task.args and self._task.args["network_os"]:
 183             display.vvvv("Getting network OS from task argument")
 184             network_os = self._task.args["network_os"]
 185         elif self._play_context.network_os:
 186             display.vvvv("Getting network OS from inventory")
 187             network_os = self._play_context.network_os
 188         elif (
 189             "network_os" in task_vars.get("ansible_facts", {})
 190             and task_vars["ansible_facts"]["network_os"]
 191         ):
 192             display.vvvv("Getting network OS from fact")
 193             network_os = task_vars["ansible_facts"]["network_os"]
 194         else:
 195             raise AnsibleError(
 196                 "ansible_network_os must be specified on this host"
 197             )
 198 
 199         return network_os
