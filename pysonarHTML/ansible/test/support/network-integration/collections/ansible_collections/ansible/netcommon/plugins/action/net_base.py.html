  10 from ansible.errors import AnsibleError
  11 from ansible.plugins.action import ActionBase
  12 from ansible.utils.display import Display

  14 display = Display()

  17 class ActionModule(ActionBase):
  18     def run(self, tmp=None, task_vars=None):
  19         del tmp  # tmp no longer has any effect
  20 
  21         result = {}
  22         play_context = copy.deepcopy(self._play_context)
  23         play_context.network_os = self._get_network_os(task_vars)
  24         new_task = self._task.copy()
  25 
  26         module = self._get_implementation_module(
  27             play_context.network_os, self._task.action
  28         )
  29         if not module:
  30             if self._task.args["fail_on_missing_module"]:
  31                 result["failed"] = True
  32             else:
  33                 result["failed"] = False
  34 
  35             result["msg"] = (
  36                 "Could not find implementation module %s for %s"
  37                 % (self._task.action, play_context.network_os)
  38             )
  39             return result
  40 
  41         new_task.action = module
  42 
  43         action = self._shared_loader_obj.action_loader.get(
  44             play_context.network_os,
  45             task=new_task,
  46             connection=self._connection,
  47             play_context=play_context,
  48             loader=self._loader,
  49             templar=self._templar,
  50             shared_loader_obj=self._shared_loader_obj,
  51         )
  52         display.vvvv("Running implementation module %s" % module)
  53         return action.run(task_vars=task_vars)
  54 
  55     def _get_network_os(self, task_vars):
  56         if "network_os" in self._task.args and self._task.args["network_os"]:
  57             display.vvvv("Getting network OS from task argument")
  58             network_os = self._task.args["network_os"]
  59         elif self._play_context.network_os:
  60             display.vvvv("Getting network OS from inventory")
  61             network_os = self._play_context.network_os
  62         elif (
  63             "network_os" in task_vars.get("ansible_facts", {})
  64             and task_vars["ansible_facts"]["network_os"]
  65         ):
  66             display.vvvv("Getting network OS from fact")
  67             network_os = task_vars["ansible_facts"]["network_os"]
  68         else:
  69             raise AnsibleError(
  70                 "ansible_network_os must be specified on this host to use platform agnostic modules"
  71             )
  72 
  73         return network_os
  74 
  75     def _get_implementation_module(self, network_os, platform_agnostic_module):
  76         module_name = (
  77             network_os.split(".")[-1]
  78             + "_"
  79             + platform_agnostic_module.partition("_")[2]
  80         )
  81         if "." in network_os:
  82             fqcn_module = ".".join(network_os.split(".")[0:-1])
  83             implementation_module = fqcn_module + "." + module_name
  84         else:
  85             implementation_module = module_name
  86 
  87         if implementation_module not in self._shared_loader_obj.module_loader:
  88             implementation_module = None
  89 
  90         return implementation_module
