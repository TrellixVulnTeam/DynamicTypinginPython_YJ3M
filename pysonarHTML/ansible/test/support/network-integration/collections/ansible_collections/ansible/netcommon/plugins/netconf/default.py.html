  39 import json
  40 
  41 from ansible.module_utils._text import to_text
  42 from ansible.plugins.netconf import NetconfBase
  43 
  44 
  45 class Netconf(NetconfBase):
  46     def get_text(self, ele, tag):
  47         try:
  48             return to_text(
  49                 ele.find(tag).text, errors="surrogate_then_replace"
  50             ).strip()
  51         except AttributeError:
  52             pass
  53 
  54     def get_device_info(self):
  55         device_info = dict()
  56         device_info["network_os"] = "default"
  57         return device_info
  58 
  59     def get_capabilities(self):
  60         result = dict()
  61         result["rpc"] = self.get_base_rpc()
  62         result["network_api"] = "netconf"
  63         result["device_info"] = self.get_device_info()
  64         result["server_capabilities"] = [c for c in self.m.server_capabilities]
  65         result["client_capabilities"] = [c for c in self.m.client_capabilities]
  66         result["session_id"] = self.m.session_id
  67         result["device_operations"] = self.get_device_operations(
  68             result["server_capabilities"]
  69         )
  70         return json.dumps(result)
