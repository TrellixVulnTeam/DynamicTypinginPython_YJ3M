  21     def __init__(self, module):
  22         self._module = module
  23         self._warnings = []
  24         self._gather_subset = module.params.get("gather_subset")
  25         self._gather_network_resources = module.params.get(
  26             "gather_network_resources"
  27         )
  28         self._connection = None
  29         if module.params.get("state") not in ["rendered", "parsed"]:
  30             self._connection = get_resource_connection(module)
  31 
  32         self.ansible_facts = {"ansible_network_resources": {}}
  33         self.ansible_facts["ansible_net_gather_network_resources"] = list()
  34         self.ansible_facts["ansible_net_gather_subset"] = list()
  35 
  36         if not self._gather_subset:
  37             self._gather_subset = ["!config"]
  38         if not self._gather_network_resources:
  39             self._gather_network_resources = ["!all"]

  41     def gen_runable(self, subsets, valid_subsets, resource_facts=False):
  42         """ Generate the runable subset
  43 
  44         :param module: The module instance
  45         :param subsets: The provided subsets
  46         :param valid_subsets: The valid subsets
  47         :param resource_facts: A boolean flag
  48         :rtype: list
  49         :returns: The runable subsets
  50         """
  51         runable_subsets = set()
  52         exclude_subsets = set()
  53         minimal_gather_subset = set()
  54         if not resource_facts:
  55             minimal_gather_subset = frozenset(["default"])
  56 
  57         for subset in subsets:
  58             if subset == "all":
  59                 runable_subsets.update(valid_subsets)
  60                 continue
  61             if subset == "min" and minimal_gather_subset:
  62                 runable_subsets.update(minimal_gather_subset)
  63                 continue
  64             if subset.startswith("!"):
  65                 subset = subset[1:]
  66                 if subset == "min":
  67                     exclude_subsets.update(minimal_gather_subset)
  68                     continue
  69                 if subset == "all":
  70                     exclude_subsets.update(
  71                         valid_subsets - minimal_gather_subset
  72                     )
  73                     continue
  74                 exclude = True
  75             else:
  76                 exclude = False
  77 
  78             if subset not in valid_subsets:
  79                 self._module.fail_json(
  80                     msg="Subset must be one of [%s], got %s"
  81                     % (
  82                         ", ".join(sorted([item for item in valid_subsets])),
  83                         subset,
  84                     )
  85                 )
  86 
  87             if exclude:
  88                 exclude_subsets.add(subset)
  89             else:
  90                 runable_subsets.add(subset)
  91 
  92         if not runable_subsets:
  93             runable_subsets.update(valid_subsets)
  94         runable_subsets.difference_update(exclude_subsets)
  95         return runable_subsets

  97     def get_network_resources_facts(
  98         self, facts_resource_obj_map, resource_facts_type=None, data=None
  99     ):
 100         """
 101         :param fact_resource_subsets:
 102         :param data: previously collected configuration
 103         :return:
 104         """
 105         if not resource_facts_type:
 106             resource_facts_type = self._gather_network_resources
 107 
 108         restorun_subsets = self.gen_runable(
 109             resource_facts_type,
 110             frozenset(facts_resource_obj_map.keys()),
 111             resource_facts=True,
 112         )
 113         if restorun_subsets:
 114             self.ansible_facts["ansible_net_gather_network_resources"] = list(
 115                 restorun_subsets
 116             )
 117             instances = list()
 118             for key in restorun_subsets:
 119                 fact_cls_obj = facts_resource_obj_map.get(key)
 120                 if fact_cls_obj:
 121                     instances.append(fact_cls_obj(self._module))
 122                 else:
 123                     self._warnings.extend(
 124                         [
 125                             "network resource fact gathering for '%s' is not supported"
 126                             % key
 127                         ]
 128                     )
 129 
 130             for inst in instances:
 131                 inst.populate_facts(self._connection, self.ansible_facts, data)
 132 
 133     def get_network_legacy_facts(
 134         self, fact_legacy_obj_map, legacy_facts_type=None
 135     ):
 136         if not legacy_facts_type:
 137             legacy_facts_type = self._gather_subset
 138 
 139         runable_subsets = self.gen_runable(
 140             legacy_facts_type, frozenset(fact_legacy_obj_map.keys())
 141         )
 142         if runable_subsets:
 143             facts = dict()
 144             # default subset should always returned be with legacy facts subsets
 145             if "default" not in runable_subsets:
 146                 runable_subsets.add("default")
 147             self.ansible_facts["ansible_net_gather_subset"] = list(
 148                 runable_subsets
 149             )
 150 
 151             instances = list()
 152             for key in runable_subsets:
 153                 instances.append(fact_legacy_obj_map[key](self._module))
 154 
 155             for inst in instances:
 156                 inst.populate()
 157                 facts.update(inst.facts)
 158                 self._warnings.extend(inst.warnings)
 159 
 160             for key, value in iteritems(facts):
 161                 key = "ansible_net_%s" % key
 162                 self.ansible_facts[key] = value
