  58 class NetconfConnection(Connection):
  59     def __init__(self, socket_path):
  60         super(NetconfConnection, self).__init__(socket_path)
  61 
  62     def __rpc__(self, name, *args, **kwargs):
  63         """Executes the json-rpc and returns the output received
  64            from remote device.
  65            :name: rpc method to be executed over connection plugin that implements jsonrpc 2.0
  66            :args: Ordered list of params passed as arguments to rpc method
  67            :kwargs: Dict of valid key, value pairs passed as arguments to rpc method
  68 
  69            For usage refer the respective connection plugin docs.
  70         """
  71         self.check_rc = kwargs.pop("check_rc", True)
  72         self.ignore_warning = kwargs.pop("ignore_warning", True)
  73 
  74         response = self._exec_jsonrpc(name, *args, **kwargs)
  75         if "error" in response:
  76             rpc_error = response["error"].get("data")
  77             return self.parse_rpc_error(
  78                 to_bytes(rpc_error, errors="surrogate_then_replace")
  79             )
  80 
  81         return fromstring(
  82             to_bytes(response["result"], errors="surrogate_then_replace")
  83         )
  84 
  85     def parse_rpc_error(self, rpc_error):
  86         if self.check_rc:
  87             try:
  88                 error_root = fromstring(rpc_error)
  89                 root = Element("root")
  90                 root.append(error_root)
  91 
  92                 error_list = root.findall(".//nc:rpc-error", NS_MAP)
  93                 if not error_list:
  94                     raise ConnectionError(
  95                         to_text(rpc_error, errors="surrogate_then_replace")
  96                     )
  97 
  98                 warnings = []
  99                 for error in error_list:
 100                     message_ele = error.find("./nc:error-message", NS_MAP)
 101 
 102                     if message_ele is None:
 103                         message_ele = error.find("./nc:error-info", NS_MAP)
 104 
 105                     message = (
 106                         message_ele.text if message_ele is not None else None
 107                     )
 108 
 109                     severity = error.find("./nc:error-severity", NS_MAP).text
 110 
 111                     if (
 112                         severity == "warning"
 113                         and self.ignore_warning
 114                         and message is not None
 115                     ):
 116                         warnings.append(message)
 117                     else:
 118                         raise ConnectionError(
 119                             to_text(rpc_error, errors="surrogate_then_replace")
 120                         )
 121                 return warnings
 122             except XMLSyntaxError:
 123                 raise ConnectionError(rpc_error)

 126 def transform_reply():

 154         raise ImportError(
 155             "ncclient is required but does not appear to be installed.  "

 165 def build_child_xml_node(parent, tag, text=None, attrib=None):
 166     element = sub_ele(parent, tag)
 167     if text:
 168         element.text = to_text(text)
 169     if attrib:
 170         element.attrib.update(attrib)
 171     return element

 174 def build_subtree(parent, path):
 175     element = parent
 176     for field in path.split("/"):
 177         sub_element = build_child_xml_node(element, field)
 178         element = sub_element
 179     return element
