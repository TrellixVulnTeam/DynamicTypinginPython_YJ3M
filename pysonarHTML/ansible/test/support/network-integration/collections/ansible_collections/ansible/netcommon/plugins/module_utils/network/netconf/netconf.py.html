  39 def get_connection(module):
  40     if hasattr(module, "_netconf_connection"):
  41         return module._netconf_connection
  42 
  43     capabilities = get_capabilities(module)
  44     network_api = capabilities.get("network_api")
  45     if network_api == "netconf":
  46         module._netconf_connection = NetconfConnection(module._socket_path)
  47     else:
  48         module.fail_json(msg="Invalid connection type %s" % network_api)
  49 
  50     return module._netconf_connection
  51

  53 def get_capabilities(module):
  54     if hasattr(module, "_netconf_capabilities"):
  55         return module._netconf_capabilities
  56 
  57     capabilities = Connection(module._socket_path).get_capabilities()
  58     module._netconf_capabilities = json.loads(capabilities)
  59     return module._netconf_capabilities

  62 def lock_configuration(module, target=None):
  63     conn = get_connection(module)
  64     return conn.lock(target=target)
  65 
  66 
  67 def unlock_configuration(module, target=None):
  68     conn = get_connection(module)
  69     return conn.unlock(target=target)

  72 @contextmanager
  73 def locked_config(module, target=None):
  74     try:
  75         lock_configuration(module, target=target)
  76         yield
  77     finally:
  78         unlock_configuration(module, target=target)

  81 def get_config(module, source, filter=None, lock=False):
  82     conn = get_connection(module)
  83     try:
  84         locked = False
  85         if lock:
  86             conn.lock(target=source)
  87             locked = True
  88         response = conn.get_config(source=source, filter=filter)
  89 
  90     except ConnectionError as e:
  91         module.fail_json(
  92             msg=to_text(e, errors="surrogate_then_replace").strip()
  93         )
  94 
  95     finally:
  96         if locked:
  97             conn.unlock(target=source)
  98 
  99     return response

 102 def get(module, filter, lock=False):
 103     conn = get_connection(module)
 104     try:
 105         locked = False
 106         if lock:
 107             conn.lock(target="running")
 108             locked = True
 109 
 110         response = conn.get(filter=filter)
 111 
 112     except ConnectionError as e:
 113         module.fail_json(
 114             msg=to_text(e, errors="surrogate_then_replace").strip()
 115         )
 116 
 117     finally:
 118         if locked:
 119             conn.unlock(target="running")
 120 
 121     return response

 124 def dispatch(module, request):
 125     conn = get_connection(module)
 126     try:
 127         response = conn.dispatch(request)
 128     except ConnectionError as e:
 129         module.fail_json(
 130             msg=to_text(e, errors="surrogate_then_replace").strip()
 131         )
 132 
 133     return response

 136 def sanitize_xml(data):
 137     tree = fromstring(
 138         to_bytes(deepcopy(data), errors="surrogate_then_replace")
 139     )
 140     for element in tree.getiterator():
 141         # remove attributes
 142         attribute = element.attrib
 143         if attribute:
 144             for key in list(attribute):
 145                 if key not in IGNORE_XML_ATTRIBUTE:
 146                     attribute.pop(key)
 147     return to_text(tostring(tree), errors="surrogate_then_replace").strip()
