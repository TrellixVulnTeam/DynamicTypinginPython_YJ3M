 122 class NetworkModule(AnsibleModule):
 123     def __init__(self, *args, **kwargs):
 124         connect_on_load = kwargs.pop("connect_on_load", True)
 125 
 126         argument_spec = NET_TRANSPORT_ARGS.copy()
 127         argument_spec["transport"]["choices"] = NET_CONNECTIONS.keys()
 128         argument_spec.update(NET_CONNECTION_ARGS.copy())
 129 
 130         if kwargs.get("argument_spec"):
 131             argument_spec.update(kwargs["argument_spec"])
 132         kwargs["argument_spec"] = argument_spec
 133 
 134         super(NetworkModule, self).__init__(*args, **kwargs)
 135 
 136         self.connection = None
 137         self._cli = None
 138         self._config = None
 139 
 140         try:
 141             transport = self.params["transport"] or "__default__"
 142             cls = NET_CONNECTIONS[transport]
 143             self.connection = cls()
 144         except KeyError:
 145             self.fail_json(
 146                 msg="Unknown transport or no default transport specified"
 147             )
 148         except (TypeError, NetworkError) as exc:
 149             self.fail_json(
 150                 msg=to_native(exc), exception=traceback.format_exc()
 151             )
 152 
 153         if connect_on_load:
 154             self.connect()
 155 
 156     @property
 157     def cli(self):
 158         if not self.connected:
 159             self.connect()
 160         if self._cli:
 161             return self._cli
 162         self._cli = Cli(self.connection)
 163         return self._cli
 164 
 165     @property
 166     def config(self):
 167         if not self.connected:
 168             self.connect()
 169         if self._config:
 170             return self._config
 171         self._config = Config(self.connection)
 172         return self._config
 173 
 174     @property
 175     def connected(self):
 176         return self.connection._connected
 177 
 178     def _load_params(self):
 179         super(NetworkModule, self)._load_params()
 180         provider = self.params.get("provider") or dict()
 181         for key, value in provider.items():
 182             for args in [NET_TRANSPORT_ARGS, NET_CONNECTION_ARGS]:
 183                 if key in args:
 184                     if self.params.get(key) is None and value is not None:
 185                         self.params[key] = value
 186 
 187     def connect(self):
 188         try:
 189             if not self.connected:
 190                 self.connection.connect(self.params)
 191                 if self.params["authorize"]:
 192                     self.connection.authorize(self.params)
 193                 self.log(
 194                     "connected to %s:%s using %s"
 195                     % (
 196                         self.params["host"],
 197                         self.params["port"],
 198                         self.params["transport"],
 199                     )
 200                 )
 201         except NetworkError as exc:
 202             self.fail_json(
 203                 msg=to_native(exc), exception=traceback.format_exc()
 204             )
 205 
 206     def disconnect(self):
 207         try:
 208             if self.connected:
 209                 self.connection.disconnect()
 210             self.log("disconnected from %s" % self.params["host"])
 211         except NetworkError as exc:
 212             self.fail_json(
 213                 msg=to_native(exc), exception=traceback.format_exc()
 214             )

 222         return cls

 231 def get_resource_connection(module):
 232     if hasattr(module, "_connection"):
 233         return module._connection
 234 
 235     capabilities = get_capabilities(module)
 236     network_api = capabilities.get("network_api")
 237     if network_api in ("cliconf", "nxapi", "eapi", "exosapi"):
 238         module._connection = Connection(module._socket_path)
 239     elif network_api == "netconf":
 240         module._connection = NetconfConnection(module._socket_path)
 241     elif network_api == "local":
 242         # This isn't supported, but we shouldn't fail here.
 243         # Set the connection to a fake connection so it fails sensibly.
 244         module._connection = LocalResourceConnection(module)
 245     else:
 246         module.fail_json(
 247             msg="Invalid connection type {0!s}".format(network_api)
 248         )
 249 
 250     return module._connection

 253 def get_capabilities(module):
 254     if hasattr(module, "capabilities"):
 255         return module._capabilities
 256     try:
 257         capabilities = Connection(module._socket_path).get_capabilities()
 258     except ConnectionError as exc:
 259         module.fail_json(msg=to_text(exc, errors="surrogate_then_replace"))
 260     except AssertionError:
 261         # No socket_path, connection most likely local.
 262         return dict(network_api="local")
 263     module._capabilities = json.loads(capabilities)
 264 
 265     return module._capabilities
