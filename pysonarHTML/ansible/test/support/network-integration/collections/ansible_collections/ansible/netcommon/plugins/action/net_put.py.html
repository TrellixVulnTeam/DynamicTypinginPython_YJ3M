  36     def run(self, tmp=None, task_vars=None):
  37         socket_path = None
  38         network_os = self._get_network_os(task_vars).split(".")[-1]
  39         persistent_connection = self._play_context.connection.split(".")[-1]
  40 
  41         result = super(ActionModule, self).run(task_vars=task_vars)
  42 
  43         if persistent_connection != "network_cli":
  44             # It is supported only with network_cli
  45             result["failed"] = True
  46             result["msg"] = (
  47                 "connection type %s is not valid for net_put module,"
  48                 " please use fully qualified name of network_cli connection type"
  49                 % self._play_context.connection
  50             )
  51             return result
  52 
  53         try:
  54             src = self._task.args["src"]
  55         except KeyError as exc:
  56             return {
  57                 "failed": True,
  58                 "msg": "missing required argument: %s" % exc,
  59             }
  60 
  61         src_file_path_name = src
  62 
  63         # Get destination file if specified
  64         dest = self._task.args.get("dest")
  65 
  66         # Get proto
  67         proto = self._task.args.get("protocol")
  68         if proto is None:
  69             proto = "scp"
  70 
  71         # Get mode if set
  72         mode = self._task.args.get("mode")
  73         if mode is None:
  74             mode = "binary"
  75 
  76         if mode == "text":
  77             try:
  78                 self._handle_template(convert_data=False)
  79             except ValueError as exc:
  80                 return dict(failed=True, msg=to_text(exc))
  81 
  82             # Now src has resolved file write to disk in current diectory for scp
  83             src = self._task.args.get("src")
  84             filename = str(uuid.uuid4())
  85             cwd = self._loader.get_basedir()
  86             output_file = os.path.join(cwd, filename)
  87             try:
  88                 with open(output_file, "wb") as f:
  89                     f.write(to_bytes(src, encoding="utf-8"))
  90             except Exception:
  91                 os.remove(output_file)
  92                 raise
  93         else:
  94             try:
  95                 output_file = self._get_binary_src_file(src)
  96             except ValueError as exc:
  97                 return dict(failed=True, msg=to_text(exc))
  98 
  99         if socket_path is None:
 100             socket_path = self._connection.socket_path
 101 
 102         conn = Connection(socket_path)
 103         sock_timeout = conn.get_option("persistent_command_timeout")
 104 
 105         if dest is None:
 106             dest = src_file_path_name
 107 
 108         try:
 109             changed = self._handle_existing_file(
 110                 conn, output_file, dest, proto, sock_timeout
 111             )
 112             if changed is False:
 113                 result["changed"] = changed
 114                 result["destination"] = dest
 115                 return result
 116         except Exception as exc:
 117             result["msg"] = (
 118                 "Warning: %s idempotency check failed. Check dest" % exc
 119             )
 120 
 121         try:
 122             conn.copy_file(
 123                 source=output_file,
 124                 destination=dest,
 125                 proto=proto,
 126                 timeout=sock_timeout,
 127             )
 128         except Exception as exc:
 129             if to_text(exc) == "No response from server":
 130                 if network_os == "iosxr":
 131                     # IOSXR sometimes closes socket prematurely after completion
 132                     # of file transfer
 133                     result[
 134                         "msg"
 135                     ] = "Warning: iosxr scp server pre close issue. Please check dest"
 136             else:
 137                 result["failed"] = True
 138                 result["msg"] = "Exception received: %s" % exc
 139 
 140         if mode == "text":
 141             # Cleanup tmp file expanded wih ansible vars
 142             os.remove(output_file)
 143 
 144         result["changed"] = changed
 145         result["destination"] = dest
 146         return result

 148     def _handle_existing_file(self, conn, source, dest, proto, timeout):
 149         """
 150         Determines whether the source and destination file match.
 151 
 152         :return: False if source and dest both exist and have matching sha1 sums, True otherwise.
 153         """
 154         cwd = self._loader.get_basedir()
 155         filename = str(uuid.uuid4())
 156         tmp_source_file = os.path.join(cwd, filename)
 157         try:
 158             conn.get_file(
 159                 source=dest,
 160                 destination=tmp_source_file,
 161                 proto=proto,
 162                 timeout=timeout,
 163             )
 164         except ConnectionError as exc:
 165             error = to_text(exc)
 166             if error.endswith("No such file or directory"):
 167                 if os.path.exists(tmp_source_file):
 168                     os.remove(tmp_source_file)
 169                 return True
 170 
 171         try:
 172             with open(source, "r") as f:
 173                 new_content = f.read()
 174             with open(tmp_source_file, "r") as f:
 175                 old_content = f.read()
 176         except (IOError, OSError):
 177             os.remove(tmp_source_file)
 178             raise
 179 
 180         sha1 = hashlib.sha1()
 181         old_content_b = to_bytes(old_content, errors="surrogate_or_strict")
 182         sha1.update(old_content_b)
 183         checksum_old = sha1.digest()
 184 
 185         sha1 = hashlib.sha1()
 186         new_content_b = to_bytes(new_content, errors="surrogate_or_strict")
 187         sha1.update(new_content_b)
 188         checksum_new = sha1.digest()
 189         os.remove(tmp_source_file)
 190         if checksum_old == checksum_new:
 191             return False
 192         return True

 194     def _get_binary_src_file(self, src):
 195         working_path = self._get_working_path()
 196 
 197         if os.path.isabs(src) or urlsplit("src").scheme:
 198             source = src
 199         else:
 200             source = self._loader.path_dwim_relative(
 201                 working_path, "templates", src
 202             )
 203             if not source:
 204                 source = self._loader.path_dwim_relative(working_path, src)
 205 
 206         if not os.path.exists(source):
 207             raise ValueError("path specified in src not found")
 208 
 209         return source
 210 
 211     def _get_working_path(self):
 212         cwd = self._loader.get_basedir()
 213         if self._task._role is not None:
 214             cwd = self._task._role._role_path
 215         return cwd

 217     def _get_network_os(self, task_vars):
 218         if "network_os" in self._task.args and self._task.args["network_os"]:
 219             display.vvvv("Getting network OS from task argument")
 220             network_os = self._task.args["network_os"]
 221         elif self._play_context.network_os:
 222             display.vvvv("Getting network OS from inventory")
 223             network_os = self._play_context.network_os
 224         elif (
 225             "network_os" in task_vars.get("ansible_facts", {})
 226             and task_vars["ansible_facts"]["network_os"]
 227         ):
 228             display.vvvv("Getting network OS from fact")
 229             network_os = task_vars["ansible_facts"]["network_os"]
 230         else:
 231             raise AnsibleError(
 232                 "ansible_network_os must be specified on this host"
 233             )
 234 
 235         return network_os
