  38 import json

  40 from ansible.module_utils._text import to_text
  41 from ansible.module_utils.connection import ConnectionError
  42 from ansible.module_utils.six.moves.urllib.error import HTTPError
  43 from ansible.plugins.httpapi import HttpApiBase
  44 
  45 
  46 CONTENT_TYPE = "application/yang-data+json"
  47 
  48 
  49 class HttpApi(HttpApiBase):
  50     def send_request(self, data, **message_kwargs):
  51         if data:
  52             data = json.dumps(data)
  53 
  54         path = "/".join(
  55             [
  56                 self.get_option("root_path").rstrip("/"),
  57                 message_kwargs.get("path", "").lstrip("/"),
  58             ]
  59         )
  60 
  61         headers = {
  62             "Content-Type": message_kwargs.get("content_type") or CONTENT_TYPE,
  63             "Accept": message_kwargs.get("accept") or CONTENT_TYPE,
  64         }
  65         response, response_data = self.connection.send(
  66             path, data, headers=headers, method=message_kwargs.get("method")
  67         )
  68 
  69         return handle_response(response, response_data)
  70 
  71 
  72 def handle_response(response, response_data):
  73     try:
  74         response_data = json.loads(response_data.read())
  75     except ValueError:
  76         response_data = response_data.read()
  77 
  78     if isinstance(response, HTTPError):
  79         if response_data:
  80             if "errors" in response_data:
  81                 errors = response_data["errors"]["error"]
  82                 error_text = "\n".join(
  83                     (error["error-message"] for error in errors)
  84                 )
  85             else:
  86                 error_text = response_data
  87 
  88             raise ConnectionError(error_text, code=response.code)
  89         raise ConnectionError(to_text(response), code=response.code)
  90 
  91     return response_data
