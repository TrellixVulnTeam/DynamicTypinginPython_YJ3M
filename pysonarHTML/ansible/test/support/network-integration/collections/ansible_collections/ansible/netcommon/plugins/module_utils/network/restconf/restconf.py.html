  32 def get(module, path=None, content=None, fields=None, output="json"):
  33     if path is None:
  34         raise ValueError("path value must be provided")
  35     if content:
  36         path += "?" + "content=%s" % content
  37     if fields:
  38         path += "?" + "field=%s" % fields
  39 
  40     accept = None
  41     if output == "xml":
  42         accept = "application/yang-data+xml"
  43 
  44     connection = Connection(module._socket_path)
  45     return connection.send_request(
  46         None, path=path, method="GET", accept=accept
  47     )

  50 def edit_config(module, path=None, content=None, method="GET", format="json"):
  51     if path is None:
  52         raise ValueError("path value must be provided")

  54     content_type = None
  55     if format == "xml":
  56         content_type = "application/yang-data+xml"
  57 
  58     connection = Connection(module._socket_path)
  59     return connection.send_request(
  60         content, path=path, method=method, content_type=content_type
