  58 def re_matchall(regex, value):
  59     objects = list()
  60     for match in re.findall(regex.pattern, value, re.M):
  61         obj = {}
  62         if regex.groupindex:
  63             for name, index in iteritems(regex.groupindex):
  64                 if len(regex.groupindex) == 1:
  65                     obj[name] = match
  66                 else:
  67                     obj[name] = match[index - 1]
  68             objects.append(obj)
  69     return objects

  72 def re_search(regex, value):
  73     obj = {}
  74     match = regex.search(value, re.M)
  75     if match:
  76         items = list(match.groups())
  77         if regex.groupindex:
  78             for name, index in iteritems(regex.groupindex):
  79                 obj[name] = items[index - 1]
  80     return obj

  83 def parse_cli(output, tmpl):
  84     if not isinstance(output, string_types):
  85         raise AnsibleError(
  86             "parse_cli input should be a string, but was given a input of %s"
  87             % (type(output))
  88         )
  89 
  90     if not os.path.exists(tmpl):
  91         raise AnsibleError("unable to locate parse_cli template: %s" % tmpl)
  92 
  93     try:
  94         template = Template()
  95     except ImportError as exc:
  96         raise AnsibleError(to_native(exc))
  97 
  98     with open(tmpl) as tmpl_fh:
  99         tmpl_content = tmpl_fh.read()
 100 
 101     spec = yaml.safe_load(tmpl_content)
 102     obj = {}
 103 
 104     for name, attrs in iteritems(spec["keys"]):
 105         value = attrs["value"]
 106 
 107         try:
 108             variables = spec.get("vars", {})
 109             value = template(value, variables)
 110         except Exception:
 111             pass
 112 
 113         if "start_block" in attrs and "end_block" in attrs:
 114             start_block = re.compile(attrs["start_block"])
 115             end_block = re.compile(attrs["end_block"])
 116 
 117             blocks = list()
 118             lines = None
 119             block_started = False
 120 
 121             for line in output.split("\n"):
 122                 match_start = start_block.match(line)
 123                 match_end = end_block.match(line)
 124 
 125                 if match_start:
 126                     lines = list()
 127                     lines.append(line)
 128                     block_started = True
 129 
 130                 elif match_end:
 131                     if lines:
 132                         lines.append(line)
 133                         blocks.append("\n".join(lines))
 134                     block_started = False
 135 
 136                 elif block_started:
 137                     if lines:
 138                         lines.append(line)
 139 
 140             regex_items = [re.compile(r) for r in attrs["items"]]
 141             objects = list()
 142 
 143             for block in blocks:
 144                 if isinstance(value, Mapping) and "key" not in value:
 145                     items = list()
 146                     for regex in regex_items:
 147                         match = regex.search(block)
 148                         if match:
 149                             item_values = match.groupdict()
 150                             item_values["match"] = list(match.groups())
 151                             items.append(item_values)
 152                         else:
 153                             items.append(None)
 154 
 155                     obj = {}
 156                     for k, v in iteritems(value):
 157                         try:
 158                             obj[k] = template(
 159                                 v, {"item": items}, fail_on_undefined=False
 160                             )
 161                         except Exception:
 162                             obj[k] = None
 163                     objects.append(obj)
 164 
 165                 elif isinstance(value, Mapping):
 166                     items = list()
 167                     for regex in regex_items:
 168                         match = regex.search(block)
 169                         if match:
 170                             item_values = match.groupdict()
 171                             item_values["match"] = list(match.groups())
 172                             items.append(item_values)
 173                         else:
 174                             items.append(None)
 175 
 176                     key = template(value["key"], {"item": items})
 177                     values = dict(
 178                         [
 179                             (k, template(v, {"item": items}))
 180                             for k, v in iteritems(value["values"])
 181                         ]
 182                     )
 183                     objects.append({key: values})
 184 
 185             return objects
 186 
 187         elif "items" in attrs:
 188             regexp = re.compile(attrs["items"])
 189             when = attrs.get("when")
 190             conditional = (
 191                 "{%% if %s %%}True{%% else %%}False{%% endif %%}" % when
 192             )
 193 
 194             if isinstance(value, Mapping) and "key" not in value:
 195                 values = list()
 196 
 197                 for item in re_matchall(regexp, output):
 198                     entry = {}
 199 
 200                     for item_key, item_value in iteritems(value):
 201                         entry[item_key] = template(item_value, {"item": item})
 202 
 203                     if when:
 204                         if template(conditional, {"item": entry}):
 205                             values.append(entry)
 206                     else:
 207                         values.append(entry)
 208 
 209                 obj[name] = values
 210 
 211             elif isinstance(value, Mapping):
 212                 values = dict()
 213 
 214                 for item in re_matchall(regexp, output):
 215                     entry = {}
 216 
 217                     for item_key, item_value in iteritems(value["values"]):
 218                         entry[item_key] = template(item_value, {"item": item})
 219 
 220                     key = template(value["key"], {"item": item})
 221 
 222                     if when:
 223                         if template(
 224                             conditional, {"item": {"key": key, "value": entry}}
 225                         ):
 226                             values[key] = entry
 227                     else:
 228                         values[key] = entry
 229 
 230                 obj[name] = values
 231 
 232             else:
 233                 item = re_search(regexp, output)
 234                 obj[name] = template(value, {"item": item})
 235 
 236         else:
 237             obj[name] = value
 238 
 239     return obj

 242 def parse_cli_textfsm(value, template):
 243     if not HAS_TEXTFSM:
 244         raise AnsibleError(
 245             "parse_cli_textfsm filter requires TextFSM library to be installed"
 246         )
 247 
 248     if not isinstance(value, string_types):
 249         raise AnsibleError(
 250             "parse_cli_textfsm input should be a string, but was given a input of %s"
 251             % (type(value))
 252         )
 253 
 254     if not os.path.exists(template):
 255         raise AnsibleError(
 256             "unable to locate parse_cli_textfsm template: %s" % template
 257         )
 258 
 259     try:
 260         template = open(template)
 261     except IOError as exc:
 262         raise AnsibleError(to_native(exc))
 263 
 264     re_table = textfsm.TextFSM(template)
 265     fsm_results = re_table.ParseText(value)
 266 
 267     results = list()
 268     for item in fsm_results:
 269         results.append(dict(zip(re_table.header, item)))
 270 
 271     return results

 274 def _extract_param(template, root, attrs, value):
 275 
 276     key = None
 277     when = attrs.get("when")
 278     conditional = "{%% if %s %%}True{%% else %%}False{%% endif %%}" % when
 279     param_to_xpath_map = attrs["items"]
 280 
 281     if isinstance(value, Mapping):
 282         key = value.get("key", None)
 283         if key:
 284             value = value["values"]
 285 
 286     entries = dict() if key else list()
 287 
 288     for element in root.findall(attrs["top"]):
 289         entry = dict()
 290         item_dict = dict()
 291         for param, param_xpath in iteritems(param_to_xpath_map):
 292             fields = None
 293             try:
 294                 fields = element.findall(param_xpath)
 295             except Exception:
 296                 display.warning(
 297                     "Failed to evaluate value of '%s' with XPath '%s'.\nUnexpected error: %s."
 298                     % (param, param_xpath, traceback.format_exc())
 299                 )
 300 
 301             tags = param_xpath.split("/")
 302 
 303             # check if xpath ends with attribute.
 304             # If yes set attribute key/value dict to param value in case attribute matches
 305             # else if it is a normal xpath assign matched element text value.
 306             if len(tags) and tags[-1].endswith("]"):
 307                 if fields:
 308                     if len(fields) > 1:
 309                         item_dict[param] = [field.attrib for field in fields]
 310                     else:
 311                         item_dict[param] = fields[0].attrib
 312                 else:
 313                     item_dict[param] = {}
 314             else:
 315                 if fields:
 316                     if len(fields) > 1:
 317                         item_dict[param] = [field.text for field in fields]
 318                     else:
 319                         item_dict[param] = fields[0].text
 320                 else:
 321                     item_dict[param] = None
 322 
 323         if isinstance(value, Mapping):
 324             for item_key, item_value in iteritems(value):
 325                 entry[item_key] = template(item_value, {"item": item_dict})
 326         else:
 327             entry = template(value, {"item": item_dict})
 328 
 329         if key:
 330             expanded_key = template(key, {"item": item_dict})
 331             if when:
 332                 if template(
 333                     conditional,
 334                     {"item": {"key": expanded_key, "value": entry}},
 335                 ):
 336                     entries[expanded_key] = entry
 337             else:
 338                 entries[expanded_key] = entry
 339         else:
 340             if when:
 341                 if template(conditional, {"item": entry}):
 342                     entries.append(entry)
 343             else:
 344                 entries.append(entry)
 345 
 346     return entries

 349 def parse_xml(output, tmpl):
 350     if not os.path.exists(tmpl):
 351         raise AnsibleError("unable to locate parse_xml template: %s" % tmpl)
 352 
 353     if not isinstance(output, string_types):
 354         raise AnsibleError(
 355             "parse_xml works on string input, but given input of : %s"
 356             % type(output)
 357         )
 358 
 359     root = fromstring(output)
 360     try:
 361         template = Template()
 362     except ImportError as exc:
 363         raise AnsibleError(to_native(exc))
 364 
 365     with open(tmpl) as tmpl_fh:
 366         tmpl_content = tmpl_fh.read()
 367 
 368     spec = yaml.safe_load(tmpl_content)
 369     obj = {}
 370 
 371     for name, attrs in iteritems(spec["keys"]):
 372         value = attrs["value"]
 373 
 374         try:
 375             variables = spec.get("vars", {})
 376             value = template(value, variables)
 377         except Exception:
 378             pass
 379 
 380         if "items" in attrs:
 381             obj[name] = _extract_param(template, root, attrs, value)
 382         else:
 383             obj[name] = value
 384 
 385     return obj

 388 def type5_pw(password, salt=None):
 389     if not isinstance(password, string_types):
 390         raise AnsibleFilterError(
 391             "type5_pw password input should be a string, but was given a input of %s"
 392             % (type(password).__name__)
 393         )
 394 
 395     salt_chars = u"".join(
 396         (to_text(string.ascii_letters), to_text(string.digits), u"./")
 397     )
 398     if salt is not None and not isinstance(salt, string_types):
 399         raise AnsibleFilterError(
 400             "type5_pw salt input should be a string, but was given a input of %s"
 401             % (type(salt).__name__)
 402         )
 403     elif not salt:
 404         salt = random_password(length=4, chars=salt_chars)
 405     elif not set(salt) <= set(salt_chars):
 406         raise AnsibleFilterError(
 407             "type5_pw salt used inproper characters, must be one of %s"
 408             % (salt_chars)
 409         )
 410 
 411     encrypted_password = passlib_or_crypt(password, "md5_crypt", salt=salt)
 412 
 413     return encrypted_password

 416 def hash_salt(password):
 417 
 418     split_password = password.split("$")
 419     if len(split_password) != 4:
 420         raise AnsibleFilterError(
 421             "Could not parse salt out password correctly from {0}".format(
 422                 password
 423             )
 424         )
 425     else:
 426         return split_password[2]

 429 def comp_type5(
 430     unencrypted_password, encrypted_password, return_original=False
 431 ):
 432 
 433     salt = hash_salt(encrypted_password)
 434     if type5_pw(unencrypted_password, salt) == encrypted_password:
 435         if return_original is True:
 436             return encrypted_password
 437         else:
 438             return True
 439     return False

 442 def vlan_parser(vlan_list, first_line_len=48, other_line_len=44):
 443 
 444     """
 445         Input: Unsorted list of vlan integers
 446         Output: Sorted string list of integers according to IOS-like vlan list rules
 447 
 448         1. Vlans are listed in ascending order
 449         2. Runs of 3 or more consecutive vlans are listed with a dash
 450         3. The first line of the list can be first_line_len characters long
 451         4. Subsequent list lines can be other_line_len characters
 452     """
 453 
 454     # Sort and remove duplicates
 455     sorted_list = sorted(set(vlan_list))
 456 
 457     if sorted_list[0] < 1 or sorted_list[-1] > 4094:
 458         raise AnsibleFilterError("Valid VLAN range is 1-4094")
 459 
 460     parse_list = []
 461     idx = 0
 462     while idx < len(sorted_list):
 463         start = idx
 464         end = start
 465         while end < len(sorted_list) - 1:
 466             if sorted_list[end + 1] - sorted_list[end] == 1:
 467                 end += 1
 468             else:
 469                 break
 470 
 471         if start == end:
 472             # Single VLAN
 473             parse_list.append(str(sorted_list[idx]))
 474         elif start + 1 == end:
 475             # Run of 2 VLANs
 476             parse_list.append(str(sorted_list[start]))
 477             parse_list.append(str(sorted_list[end]))
 478         else:
 479             # Run of 3 or more VLANs
 480             parse_list.append(
 481                 str(sorted_list[start]) + "-" + str(sorted_list[end])
 482             )
 483         idx = end + 1
 484 
 485     line_count = 0
 486     result = [""]
 487     for vlans in parse_list:
 488         # First line (" switchport trunk allowed vlan ")
 489         if line_count == 0:
 490             if len(result[line_count] + vlans) > first_line_len:
 491                 result.append("")
 492                 line_count += 1
 493                 result[line_count] += vlans + ","
 494             else:
 495                 result[line_count] += vlans + ","
 496 
 497         # Subsequent lines (" switchport trunk allowed vlan add ")
 498         else:
 499             if len(result[line_count] + vlans) > other_line_len:
 500                 result.append("")
 501                 line_count += 1
 502                 result[line_count] += vlans + ","
 503             else:
 504                 result[line_count] += vlans + ","
 505 
 506     # Remove trailing orphan commas
 507     for idx in range(0, len(result)):
 508         result[idx] = result[idx].rstrip(",")
 509 
 510     # Sometimes text wraps to next line, but there are no remaining VLANs
 511     if "" in result:
 512         result.remove("")
 513 
 514     return result
