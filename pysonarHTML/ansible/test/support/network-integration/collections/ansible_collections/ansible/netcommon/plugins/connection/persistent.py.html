  30 from ansible.executor.task_executor import start_connection
  31 from ansible.plugins.connection import ConnectionBase
  32 from ansible.module_utils._text import to_text
  33 from ansible.module_utils.connection import Connection as SocketConnection
  34 from ansible.utils.display import Display
  35 
  36 display = Display()
  37 
  38 
  39 class Connection(ConnectionBase):
  40     """ Local based connections """
  41 
  42     transport = "ansible.netcommon.persistent"
  43     has_pipelining = False
  44 
  45     def __init__(self, play_context, new_stdin, *args, **kwargs):
  46         super(Connection, self).__init__(
  47             play_context, new_stdin, *args, **kwargs
  48         )
  49         self._task_uuid = to_text(kwargs.get("task_uuid", ""))
  50 
  51     def _connect(self):
  52         self._connected = True
  53         return self
  54 
  55     def exec_command(self, cmd, in_data=None, sudoable=True):
  56         display.vvvv(
  57             "exec_command(), socket_path=%s" % self.socket_path,
  58             host=self._play_context.remote_addr,
  59         )
  60         connection = SocketConnection(self.socket_path)
  61         out = connection.exec_command(cmd, in_data=in_data, sudoable=sudoable)
  62         return 0, out, ""
  63 
  64     def put_file(self, in_path, out_path):
  65         pass
  66 
  67     def fetch_file(self, in_path, out_path):
  68         pass
  69 
  70     def close(self):
  71         self._connected = False
  72 
  73     def run(self):
  74         """Returns the path of the persistent connection socket.
  75 
  76         Attempts to ensure (within playcontext.timeout seconds) that the
  77         socket path exists. If the path exists (or the timeout has expired),
  78         returns the socket path.
  79         """
  80         display.vvvv(
  81             "starting connection from persistent connection plugin",
  82             host=self._play_context.remote_addr,
  83         )
  84         variables = {
  85             "ansible_command_timeout": self.get_option(
  86                 "persistent_command_timeout"
  87             )
  88         }
  89         socket_path = start_connection(
  90             self._play_context, variables, self._task_uuid
  91         )
  92         display.vvvv(
  93             "local domain socket path is %s" % socket_path,
  94             host=self._play_context.remote_addr,
  95         )
  96         setattr(self, "_socket_path", socket_path)
  97         return socket_path
