  38 class ActionModule(_ActionModule):
  39     def run(self, task_vars=None):
  40         config_module = hasattr(self, "_config_module") and self._config_module
  41         if config_module and self._task.args.get("src"):
  42             try:
  43                 self._handle_src_option()
  44             except AnsibleError as e:
  45                 return {"failed": True, "msg": e.message, "changed": False}
  46 
  47         result = super(ActionModule, self).run(task_vars=task_vars)
  48 
  49         if (
  50             config_module
  51             and self._task.args.get("backup")
  52             and not result.get("failed")
  53         ):
  54             self._handle_backup_option(result, task_vars)
  55 
  56         return result
  57 
  58     def _handle_backup_option(self, result, task_vars):
  59 
  60         filename = None
  61         backup_path = None
  62         try:
  63             content = result["__backup__"]
  64         except KeyError:
  65             raise AnsibleError("Failed while reading configuration backup")
  66 
  67         backup_options = self._task.args.get("backup_options")
  68         if backup_options:
  69             filename = backup_options.get("filename")
  70             backup_path = backup_options.get("dir_path")
  71 
  72         if not backup_path:
  73             cwd = self._get_working_path()
  74             backup_path = os.path.join(cwd, "backup")
  75         if not filename:
  76             tstamp = time.strftime(
  77                 "%Y-%m-%d@%H:%M:%S", time.localtime(time.time())
  78             )
  79             filename = "%s_config.%s" % (
  80                 task_vars["inventory_hostname"],
  81                 tstamp,
  82             )
  83 
  84         dest = os.path.join(backup_path, filename)
  85         backup_path = os.path.expanduser(
  86             os.path.expandvars(
  87                 to_bytes(backup_path, errors="surrogate_or_strict")
  88             )
  89         )
  90 
  91         if not os.path.exists(backup_path):
  92             os.makedirs(backup_path)
  93 
  94         new_task = self._task.copy()
  95         for item in self._task.args:
  96             if not item.startswith("_"):
  97                 new_task.args.pop(item, None)
  98 
  99         new_task.args.update(dict(content=content, dest=dest))
 100         copy_action = self._shared_loader_obj.action_loader.get(
 101             "copy",
 102             task=new_task,
 103             connection=self._connection,
 104             play_context=self._play_context,
 105             loader=self._loader,
 106             templar=self._templar,
 107             shared_loader_obj=self._shared_loader_obj,
 108         )
 109         copy_result = copy_action.run(task_vars=task_vars)
 110         if copy_result.get("failed"):
 111             result["failed"] = copy_result["failed"]
 112             result["msg"] = copy_result.get("msg")
 113             return
 114 
 115         result["backup_path"] = dest
 116         if copy_result.get("changed", False):
 117             result["changed"] = copy_result["changed"]
 118 
 119         if backup_options and backup_options.get("filename"):
 120             result["date"] = time.strftime(
 121                 "%Y-%m-%d",
 122                 time.gmtime(os.stat(result["backup_path"]).st_ctime),
 123             )
 124             result["time"] = time.strftime(
 125                 "%H:%M:%S",
 126                 time.gmtime(os.stat(result["backup_path"]).st_ctime),
 127             )
 128 
 129         else:
 130             result["date"] = tstamp.split("@")[0]
 131             result["time"] = tstamp.split("@")[1]
 132             result["shortname"] = result["backup_path"][::-1].split(".", 1)[1][
 133                 ::-1
 134             ]
 135             result["filename"] = result["backup_path"].split("/")[-1]
 136 
 137         # strip out any keys that have two leading and two trailing
 138         # underscore characters
 139         for key in list(result.keys()):
 140             if PRIVATE_KEYS_RE.match(key):
 141                 del result[key]

 143     def _get_working_path(self):
 144         cwd = self._loader.get_basedir()
 145         if self._task._role is not None:
 146             cwd = self._task._role._role_path
 147         return cwd
 148 
 149     def _handle_src_option(self, convert_data=True):
 150         src = self._task.args.get("src")
 151         working_path = self._get_working_path()
 152 
 153         if os.path.isabs(src) or urlsplit("src").scheme:
 154             source = src
 155         else:
 156             source = self._loader.path_dwim_relative(
 157                 working_path, "templates", src
 158             )
 159             if not source:
 160                 source = self._loader.path_dwim_relative(working_path, src)
 161 
 162         if not os.path.exists(source):
 163             raise AnsibleError("path specified in src not found")
 164 
 165         try:
 166             with open(source, "r") as f:
 167                 template_data = to_text(f.read())
 168         except IOError as e:
 169             raise AnsibleError(
 170                 "unable to load src file {0}, I/O error({1}): {2}".format(
 171                     source, e.errno, e.strerror
 172                 )
 173             )
 174 
 175         # Create a template search path in the following order:
 176         # [working_path, self_role_path, dependent_role_paths, dirname(source)]
 177         searchpath = [working_path]
 178         if self._task._role is not None:
 179             searchpath.append(self._task._role._role_path)
 180             if hasattr(self._task, "_block:"):
 181                 dep_chain = self._task._block.get_dep_chain()
 182                 if dep_chain is not None:
 183                     for role in dep_chain:
 184                         searchpath.append(role._role_path)
 185         searchpath.append(os.path.dirname(source))
 186         with self._templar.set_temporary_context(searchpath=searchpath):
 187             self._task.args["src"] = self._templar.template(
 188                 template_data, convert_data=convert_data
 189             )

 191     def _get_network_os(self, task_vars):
 192         if "network_os" in self._task.args and self._task.args["network_os"]:
 193             display.vvvv("Getting network OS from task argument")
 194             network_os = self._task.args["network_os"]
 195         elif self._play_context.network_os:
 196             display.vvvv("Getting network OS from inventory")
 197             network_os = self._play_context.network_os
 198         elif (
 199             "network_os" in task_vars.get("ansible_facts", {})
 200             and task_vars["ansible_facts"]["network_os"]
 201         ):
 202             display.vvvv("Getting network OS from fact")
 203             network_os = task_vars["ansible_facts"]["network_os"]
 204         else:
 205             raise AnsibleError(
 206                 "ansible_network_os must be specified on this host"
 207             )
 208 
 209         return network_os
