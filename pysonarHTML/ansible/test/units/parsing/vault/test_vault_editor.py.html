 188     def _assert_file_is_encrypted(self, vault_editor, src_file_path, src_contents):
 189         new_src_file = open(src_file_path, 'rb')
 190         new_src_file_contents = new_src_file.read()
 191 
 192         # TODO: assert that it is encrypted
 193         self.assertTrue(vault.is_encrypted(new_src_file_contents))
 194 
 195         src_file_plaintext = vault_editor.vault.decrypt(new_src_file_contents)
 196 
 197         # the plaintext should not be encrypted
 198         self.assertFalse(vault.is_encrypted(src_file_plaintext))
 199 
 200         # and the new plaintext should match the original
 201         self.assertEqual(src_file_plaintext, src_contents)
