  22     INVALID_MULTIPLE_ACTIONS = (
  23         ({'action': 'shell echo hi', 'local_action': 'shell echo hi'}, "action and local_action are mutually exclusive"),
  24         ({'action': 'shell echo hi', 'shell': 'echo hi'}, "conflicting action statements: shell, shell"),
  25         ({'local_action': 'shell echo hi', 'shell': 'echo hi'}, "conflicting action statements: shell, shell"),

  28     def _debug(self, mod, args, to):
  29         print("RETURNED module = {0}".format(mod))
  30         print("           args = {0}".format(args))
  31         print("             to = {0}".format(to))

  33     def test_basic_shell(self):
  34         m = ModuleArgsParser(dict(shell='echo hi'))
  35         mod, args, to = m.parse()
  36         self._debug(mod, args, to)
  37 
  38         assert mod == 'shell'
  39         assert args == dict(
  40             _raw_params='echo hi',
  41         )
  42         assert to is Sentinel

 113     @pytest.mark.parametrize("args_dict, msg", INVALID_MULTIPLE_ACTIONS)
 114     def test_multiple_actions(self, args_dict, msg):
 115         m = ModuleArgsParser(args_dict)
 116         with pytest.raises(AnsibleParserError) as err:
 117             m.parse()
 118 
 119         assert err.value.args[0] == msg
 120 
 121     def test_multiple_actions(self):
 122         args_dict = {'ping': 'data=hi', 'shell': 'echo hi'}
 123         m = ModuleArgsParser(args_dict)
 124         with pytest.raises(AnsibleParserError) as err:
 125             m.parse()
 126 
 127         assert err.value.args[0].startswith("conflicting action statements: ")
 128         actions = set(re.search(r'(\w+), (\w+)', err.value.args[0]).groups())
 129         assert actions == set(['ping', 'shell'])
 130 
 131     def test_bogus_action(self):
