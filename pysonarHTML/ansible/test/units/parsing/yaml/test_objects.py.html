  74 class TestAnsibleVaultEncryptedUnicode(unittest.TestCase, YamlTestUtils):
  75     def setUp(self):
  76         self.good_vault_password = "hunter42"
  77         good_vault_secret = TextVaultSecret(self.good_vault_password)
  78         self.good_vault_secrets = [('good_vault_password', good_vault_secret)]
  79         self.good_vault = vault.VaultLib(self.good_vault_secrets)
  80 
  81         # TODO: make this use two vault secret identities instead of two vaultSecrets
  82         self.wrong_vault_password = 'not-hunter42'
  83         wrong_vault_secret = TextVaultSecret(self.wrong_vault_password)
  84         self.wrong_vault_secrets = [('wrong_vault_password', wrong_vault_secret)]
  85         self.wrong_vault = vault.VaultLib(self.wrong_vault_secrets)
  86 
  87         self.vault = self.good_vault
  88         self.vault_secrets = self.good_vault_secrets
  89 
  90     def _loader(self, stream):
  91         return AnsibleLoader(stream, vault_secrets=self.vault_secrets)
  92 
  93     def test_dump_load_cycle(self):
  94         aveu = self._from_plaintext('the test string for TestAnsibleVaultEncryptedUnicode.test_dump_load_cycle')
  95         self._dump_load_cycle(aveu)
  96 
  97     def assert_values(self, avu, seq):
  98         self.assertIsInstance(avu, objects.AnsibleVaultEncryptedUnicode)
  99 
 100         self.assertEqual(avu, seq)
 101         self.assertTrue(avu.vault is self.vault)
 102         self.assertIsInstance(avu.vault, vault.VaultLib)
 103 
 104     def _from_plaintext(self, seq):
 105         id_secret = vault.match_encrypt_secret(self.good_vault_secrets)
 106         return objects.AnsibleVaultEncryptedUnicode.from_plaintext(seq, vault=self.vault, secret=id_secret[1])
 107 
 108     def _from_ciphertext(self, ciphertext):
 109         avu = objects.AnsibleVaultEncryptedUnicode(ciphertext)
 110         avu.vault = self.vault
 111         return avu
 112 
 113     def test_empty_init(self):
 114         self.assertRaises(TypeError, objects.AnsibleVaultEncryptedUnicode)
 115 
 116     def test_empty_string_init_from_plaintext(self):
 117         seq = ''
 118         avu = self._from_plaintext(seq)
 119         self.assert_values(avu, seq)
 120 
 121     def test_empty_unicode_init_from_plaintext(self):
 122         seq = u''
 123         avu = self._from_plaintext(seq)
 124         self.assert_values(avu, seq)
 125 
 126     def test_string_from_plaintext(self):
 127         seq = 'some letters'
 128         avu = self._from_plaintext(seq)
 129         self.assert_values(avu, seq)
 130 
 131     def test_unicode_from_plaintext(self):
 132         seq = u'some letters'
 133         avu = self._from_plaintext(seq)
 134         self.assert_values(avu, seq)
 135 
 136     def test_unicode_from_plaintext_encode(self):
 137         seq = u'some text here'
 138         avu = self._from_plaintext(seq)
 139         b_avu = avu.encode('utf-8', 'strict')
 140         self.assertIsInstance(avu, objects.AnsibleVaultEncryptedUnicode)
 141         self.assertEqual(b_avu, seq.encode('utf-8', 'strict'))
 142         self.assertTrue(avu.vault is self.vault)
 143         self.assertIsInstance(avu.vault, vault.VaultLib)
 144 
 145     # TODO/FIXME: make sure bad password fails differently than 'thats not encrypted'
 146     def test_empty_string_wrong_password(self):
 147         seq = ''
 148         self.vault = self.wrong_vault
 149         avu = self._from_plaintext(seq)
 150 
 151         def compare(avu, seq):
 152             return avu == seq
 153 
 154         self.assertRaises(AnsibleError, compare, avu, seq)
 155 
 156     def test_vaulted_utf8_value_37258(self):

 158         avu = self._from_plaintext(seq)
 159         self.assert_values(avu, seq)
 160 
 161     def test_str_vaulted_utf8_value_37258(self):
 162         seq = u"aöffü"
 163         avu = self._from_plaintext(seq)
 164         assert str(avu) == to_native(seq)
