  34 class TestAnsibleDumper(unittest.TestCase, YamlTestUtils):
  35     def setUp(self):
  36         self.vault_password = "hunter42"
  37         vault_secret = TextVaultSecret(self.vault_password)
  38         self.vault_secrets = [('vault_secret', vault_secret)]
  39         self.good_vault = vault.VaultLib(self.vault_secrets)
  40         self.vault = self.good_vault
  41         self.stream = self._build_stream()
  42         self.dumper = dumper.AnsibleDumper
  43 
  44     def _build_stream(self, yaml_text=None):
  45         text = yaml_text or u''
  46         stream = io.StringIO(text)
  47         return stream
  48 
  49     def _loader(self, stream):
  50         return AnsibleLoader(stream, vault_secrets=self.vault.secrets)
  51 
  52     def test_ansible_vault_encrypted_unicode(self):
  53         plaintext = 'This is a string we are going to encrypt.'
  54         avu = objects.AnsibleVaultEncryptedUnicode.from_plaintext(plaintext, vault=self.vault,
  55                                                                   secret=vault.match_secrets(self.vault_secrets, ['vault_secret'])[0][1])
  56 
  57         yaml_out = self._dump_string(avu, dumper=self.dumper)
  58         stream = self._build_stream(yaml_out)
  59         loader = self._loader(stream)
  60 
  61         data_from_yaml = loader.get_single_data()
  62 
  63         self.assertEqual(plaintext, data_from_yaml.data)
  64 
  65     def test_bytes(self):
  66         b_text = u'tréma'.encode('utf-8')
  67         unsafe_object = AnsibleUnsafeBytes(b_text)
  68         yaml_out = self._dump_string(unsafe_object, dumper=self.dumper)
  69 
  70         stream = self._build_stream(yaml_out)
  71         loader = self._loader(stream)
  72 
  73         data_from_yaml = loader.get_single_data()
  74 
  75         result = b_text
  76         if PY2:
  77             # https://pyyaml.org/wiki/PyYAMLDocumentation#string-conversion-python-2-only
  78             # pyyaml on Python 2 can return either unicode or bytes when given byte strings.
  79             # We normalize that to always return unicode on Python2 as that's right most of the
  80             # time.  However, this means byte strings can round trip through yaml on Python3 but
  81             # not on Python2.  To make this code work the same on Python2 and Python3 (we want
  82             # the Python3 behaviour) we need to change the methods in Ansible to:
  83             # (1) Let byte strings pass through yaml without being converted on Python2
  84             # (2) Convert byte strings to text strings before being given to pyyaml  (Without this,
  85             #       strings would end up as byte strings most of the time which would mostly be wrong)
  86             # In practice, we mostly read bytes in from files and then pass that to pyyaml, for which
  87             # the present behavior is correct.
  88             # This is a workaround for the current behavior.
  89             result = u'tr\xe9ma'
  90 
  91         self.assertEqual(result, data_from_yaml)
  92 
  93     def test_unicode(self):
