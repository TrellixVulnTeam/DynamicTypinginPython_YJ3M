 182     def test_wrong_password(self):

 194             self.vault.decrypt(ciphertext)

 199     def _encrypt_plaintext(self, plaintext):
 200         # Construct a yaml repr of a vault by hand
 201         vaulted_var_bytes = self.vault.encrypt(plaintext, self.vault_secret)
 202 
 203         # add yaml tag
 204         vaulted_var = vaulted_var_bytes.decode()
 205         lines = vaulted_var.splitlines()
 206         lines2 = []
 207         for line in lines:
 208             lines2.append('        %s' % line)
 209 
 210         vaulted_var = '\n'.join(lines2)
 211         tagged_vaulted_var = u"""!vault |\n%s""" % vaulted_var
 212         return tagged_vaulted_var
 213 
 214     def _build_stream(self, yaml_text):
 215         stream = NameStringIO(yaml_text)
 216         stream.name = 'my.yml'
 217         return stream

 219     def _loader(self, stream):
 220         return AnsibleLoader(stream, vault_secrets=self.vault.secrets)

 222     def _load_yaml(self, yaml_text, password):
 223         stream = self._build_stream(yaml_text)
 224         loader = self._loader(stream)
 225 
 226         data_from_yaml = loader.get_single_data()
 227 
 228         return data_from_yaml

 360     def walk(self, data):
 361         # Make sure there's no str in the data
 362         self.assertNotIsInstance(data, binary_type)
 363 
 364         # Descend into various container types
 365         if isinstance(data, text_type):
 366             # strings are a sequence so we have to be explicit here
 367             return
 368         elif isinstance(data, (Sequence, Set)):
 369             for element in data:
 370                 self.walk(element)
 371         elif isinstance(data, Mapping):
 372             for k, v in data.items():
 373                 self.walk(k)
 374                 self.walk(v)
 375 
 376         # Scalars were all checked so we're good to go
 377         return
