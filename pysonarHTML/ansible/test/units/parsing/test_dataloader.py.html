  83     @patch('ansible.parsing.dataloader.unfrackpath', mock_unfrackpath_noop)
  84     @patch.object(DataLoader, '_is_role')
  85     def test_path_dwim_relative(self, mock_is_role):
  86         """
  87         simulate a nested dynamic include:
  88 
  89         playbook.yml:
  90         - hosts: localhost
  91           roles:
  92             - { role: 'testrole' }
  93 
  94         testrole/tasks/main.yml:
  95         - include: "include1.yml"
  96           static: no
  97 
  98         testrole/tasks/include1.yml:
  99         - include: include2.yml
 100           static: no
 101 
 102         testrole/tasks/include2.yml:
 103         - debug: msg="blah"
 104         """
 105         mock_is_role.return_value = False
 106         with patch('os.path.exists') as mock_os_path_exists:
 107             mock_os_path_exists.return_value = False
 108             self._loader.path_dwim_relative('/tmp/roles/testrole/tasks', 'tasks', 'included2.yml')
 109 
 110             # Fetch first args for every call
 111             # mock_os_path_exists.assert_any_call isn't used because os.path.normpath must be used in order to compare paths
 112             called_args = [os.path.normpath(to_text(call[0][0])) for call in mock_os_path_exists.call_args_list]
 113 
 114             # 'path_dwim_relative' docstrings say 'with or without explicitly named dirname subdirs':
 115             self.assertIn('/tmp/roles/testrole/tasks/included2.yml', called_args)
 116             self.assertIn('/tmp/roles/testrole/tasks/tasks/included2.yml', called_args)
 117 
 118             # relative directories below are taken in account too:
 119             self.assertIn('tasks/included2.yml', called_args)
 120             self.assertIn('included2.yml', called_args)
 121

 171     def test_none(self):
 172         self.assertRaisesRegexp(AnsibleFileNotFound, 'on the Ansible Controller', self._loader.path_dwim_relative_stack, None, None, None)
