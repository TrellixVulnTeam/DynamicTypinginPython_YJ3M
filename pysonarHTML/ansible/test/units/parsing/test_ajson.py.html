  31 def test_encode_decode_unsafe():
  32     data = {
  33         'key_value': AnsibleUnsafeText(u'{#NOTACOMMENT#}'),
  34         'list': [AnsibleUnsafeText(u'{#NOTACOMMENT#}')],
  35         'list_dict': [{'key_value': AnsibleUnsafeText(u'{#NOTACOMMENT#}')}]}
  36     json_expected = (
  37         '{"key_value": {"__ansible_unsafe": "{#NOTACOMMENT#}"}, '
  38         '"list": [{"__ansible_unsafe": "{#NOTACOMMENT#}"}], '
  39         '"list_dict": [{"key_value": {"__ansible_unsafe": "{#NOTACOMMENT#}"}}]}'
  40     )
  41     assert json.dumps(data, cls=AnsibleJSONEncoder, preprocess_unsafe=True, sort_keys=True) == json_expected
  42     assert json.loads(json_expected, cls=AnsibleJSONDecoder) == data

  87     def mapping(self, request):
  88         """
  89         Returns object of Mapping mock class.
  90 
  91         The object is used for testing handling of Mapping objects
  92         in AnsibleJSONEncoder.default().
  93         Using a plain dictionary instead is not suitable because
  94         it is handled by default encoder of the superclass (json.JSONEncoder).
  95         """
  96 
  97         class M(Mapping):
  98 
  99             """Mock mapping class."""
 100 
 101             def __init__(self, *args, **kwargs):
 102                 self.__dict__.update(*args, **kwargs)
 103 
 104             def __getitem__(self, key):
 105                 return self.__dict__[key]
 106 
 107             def __iter__(self):
 108                 return iter(self.__dict__)
 109 
 110             def __len__(self):
 111                 return len(self.__dict__)
 112 
 113         return M(request.param)
