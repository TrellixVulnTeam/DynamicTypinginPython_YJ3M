  28 class TestGroup(unittest.TestCase):

  30     def test_depth_update(self):
  31         A = Group('A')
  32         B = Group('B')
  33         Z = Group('Z')
  34         A.add_child_group(B)
  35         A.add_child_group(Z)
  36         self.assertEqual(A.depth, 0)
  37         self.assertEqual(Z.depth, 1)
  38         self.assertEqual(B.depth, 1)
  39 
  40     def test_depth_update_dual_branches(self):
  41         alpha = Group('alpha')
  42         A = Group('A')
  43         alpha.add_child_group(A)
  44         B = Group('B')
  45         A.add_child_group(B)
  46         Z = Group('Z')
  47         alpha.add_child_group(Z)
  48         beta = Group('beta')
  49         B.add_child_group(beta)
  50         Z.add_child_group(beta)
  51 
  52         self.assertEqual(alpha.depth, 0)  # apex
  53         self.assertEqual(beta.depth, 3)  # alpha -> A -> B -> beta
  54 
  55         omega = Group('omega')
  56         omega.add_child_group(alpha)
  57 
  58         # verify that both paths are traversed to get the max depth value
  59         self.assertEqual(B.depth, 3)  # omega -> alpha -> A -> B
  60         self.assertEqual(beta.depth, 4)  # B -> beta
  61 
  62     def test_depth_recursion(self):
  63         A = Group('A')
  64         B = Group('B')
  65         A.add_child_group(B)
  66         # hypothetical of adding B as child group to A
  67         A.parent_groups.append(B)
  68         B.child_groups.append(A)
  69         # can't update depths of groups, because of loop
  70         with self.assertRaises(AnsibleError):
  71             B._check_children_depth()
  72 
  73     def test_loop_detection(self):
  74         A = Group('A')
  75         B = Group('B')
  76         C = Group('C')
  77         A.add_child_group(B)
  78         B.add_child_group(C)
  79         with self.assertRaises(AnsibleError):
  80             C.add_child_group(A)

  82     def test_direct_host_ordering(self):
  83         """Hosts are returned in order they are added
  84         """
  85         group = Group('A')
  86         # host names not added in alphabetical order
  87         host_name_list = ['z', 'b', 'c', 'a', 'p', 'q']
  88         expected_hosts = []
  89         for host_name in host_name_list:
  90             h = Host(host_name)
  91             group.add_host(h)
  92             expected_hosts.append(h)
  93         assert group.get_hosts() == expected_hosts
  94 
  95     def test_sub_group_host_ordering(self):
  96         """With multiple nested groups, asserts that hosts are returned
  97         in deterministic order
  98         """
  99         top_group = Group('A')
 100         expected_hosts = []
 101         for name in ['z', 'b', 'c', 'a', 'p', 'q']:
 102             child = Group('group_{0}'.format(name))
 103             top_group.add_child_group(child)
 104             host = Host('host_{0}'.format(name))
 105             child.add_host(host)
 106             expected_hosts.append(host)
 107         assert top_group.get_hosts() == expected_hosts

 109     def test_populates_descendant_hosts(self):
 110         A = Group('A')
 111         B = Group('B')
 112         C = Group('C')
 113         h = Host('h')
 114         C.add_host(h)
 115         A.add_child_group(B)  # B is child of A
 116         B.add_child_group(C)  # C is descendant of A
 117         A.add_child_group(B)
 118         self.assertEqual(set(h.groups), set([C, B, A]))
 119         h2 = Host('h2')
 120         C.add_host(h2)
 121         self.assertEqual(set(h2.groups), set([C, B, A]))

 123     def test_ancestor_example(self):
 124         # see docstring for Group._walk_relationship
 125         groups = {}
 126         for name in ['A', 'B', 'C', 'D', 'E', 'F']:
 127             groups[name] = Group(name)
 128         # first row
 129         groups['A'].add_child_group(groups['D'])
 130         groups['B'].add_child_group(groups['D'])
 131         groups['B'].add_child_group(groups['E'])
 132         groups['C'].add_child_group(groups['D'])
 133         # second row
 134         groups['D'].add_child_group(groups['E'])
 135         groups['D'].add_child_group(groups['F'])
 136         groups['E'].add_child_group(groups['F'])
 137 
 138         self.assertEqual(
 139             set(groups['F'].get_ancestors()),
 140             set([
 141                 groups['A'], groups['B'], groups['C'], groups['D'], groups['E']
 142             ])
 143         )
 144 
 145     def test_ancestors_recursive_loop_safe(self):
 146         '''
 147         The get_ancestors method may be referenced before circular parenting
 148         checks, so the method is expected to be stable even with loops
 149         '''
 150         A = Group('A')
 151         B = Group('B')
 152         A.parent_groups.append(B)
 153         B.parent_groups.append(A)
 154         # finishes in finite time
 155         self.assertEqual(A.get_ancestors(), set([A, B]))
