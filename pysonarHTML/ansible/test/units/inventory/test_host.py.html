  32 class TestHost(unittest.TestCase):
  33     ansible_port = 22
  34 
  35     def setUp(self):
  36         self.hostA = Host('a')
  37         self.hostB = Host('b')
  38 
  39     def test_equality(self):
  40         self.assertEqual(self.hostA, self.hostA)
  41         self.assertNotEqual(self.hostA, self.hostB)
  42         self.assertNotEqual(self.hostA, Host('a'))
  43 
  44     def test_hashability(self):
  45         # equality implies the hash values are the same
  46         self.assertEqual(hash(self.hostA), hash(Host('a')))
  47 
  48     def test_get_vars(self):
  49         host_vars = self.hostA.get_vars()
  50         self.assertIsInstance(host_vars, dict)
  51 
  52     def test_repr(self):
  53         host_repr = repr(self.hostA)
  54         self.assertIsInstance(host_repr, string_types)
  55 
  56     def test_add_group(self):
  57         group = Group('some_group')
  58         group_len = len(self.hostA.groups)
  59         self.hostA.add_group(group)
  60         self.assertEqual(len(self.hostA.groups), group_len + 1)
  61 
  62     def test_get_groups(self):
  63         group = Group('some_group')
  64         self.hostA.add_group(group)
  65         groups = self.hostA.get_groups()
  66         self.assertEqual(len(groups), 1)
  67         for _group in groups:
  68             self.assertIsInstance(_group, Group)
  69 
  70     def test_equals_none(self):
  71         other = None
  72         self.hostA == other
  73         other == self.hostA
  74         self.hostA != other
  75         other != self.hostA
  76         self.assertNotEqual(self.hostA, other)
  77 
  78     def test_serialize(self):
  79         group = Group('some_group')
  80         self.hostA.add_group(group)
  81         data = self.hostA.serialize()
  82         self.assertIsInstance(data, dict)
  83 
  84     def test_serialize_then_deserialize(self):
  85         group = Group('some_group')
  86         self.hostA.add_group(group)
  87         hostA_data = self.hostA.serialize()
  88 
  89         hostA_clone = Host()
  90         hostA_clone.deserialize(hostA_data)
  91         self.assertEqual(self.hostA, hostA_clone)
  92 
  93     def test_set_state(self):
  94         group = Group('some_group')
  95         self.hostA.add_group(group)
  96 
  97         pickled_hostA = pickle.dumps(self.hostA)
  98 
  99         hostA_clone = pickle.loads(pickled_hostA)
 100         self.assertEqual(self.hostA, hostA_clone)

 103 class TestHostWithPort(TestHost):
 104     ansible_port = 8822
 105 
 106     def setUp(self):
 107         self.hostA = Host(name='a', port=self.ansible_port)
 108         self.hostB = Host(name='b', port=self.ansible_port)
 109
