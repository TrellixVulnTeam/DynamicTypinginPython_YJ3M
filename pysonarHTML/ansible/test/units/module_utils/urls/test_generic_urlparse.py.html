  12 def test_generic_urlparse():
  13     url = 'https://ansible.com/blog'
  14     parts = urlparse(url)
  15     generic_parts = generic_urlparse(parts)
  16     assert generic_parts.as_list() == list(parts)
  17 
  18     assert urlunparse(generic_parts.as_list()) == url

  21 def test_generic_urlparse_netloc():
  22     url = 'https://ansible.com:443/blog'
  23     parts = urlparse(url)
  24     generic_parts = generic_urlparse(parts)
  25     assert generic_parts.hostname == parts.hostname
  26     assert generic_parts.hostname == 'ansible.com'
  27     assert generic_parts.port == 443
  28     assert urlunparse(generic_parts.as_list()) == url

  31 def test_generic_urlparse_no_netloc():
  32     url = 'https://user:passwd@ansible.com:443/blog'
  33     parts = list(urlparse(url))
  34     generic_parts = generic_urlparse(parts)
  35     assert generic_parts.hostname == 'ansible.com'
  36     assert generic_parts.port == 443
  37     assert generic_parts.username == 'user'
  38     assert generic_parts.password == 'passwd'
  39     assert urlunparse(generic_parts.as_list()) == url
