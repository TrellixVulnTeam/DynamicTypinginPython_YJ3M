  95 def test_fetch_url_cookies(mocker, fake_ansible_module):
  96     def make_cookies(*args, **kwargs):
  97         cookies = kwargs['cookies']
  98         r = MagicMock()
  99         try:
 100             r.headers = HTTPMessage()
 101             add_header = r.headers.add_header
 102         except TypeError:
 103             # PY2
 104             r.headers = HTTPMessage(StringIO())
 105             add_header = r.headers.addheader
 106         r.info.return_value = r.headers
 107         for name, value in (('Foo', 'bar'), ('Baz', 'qux')):
 108             cookie = Cookie(
 109                 version=0,
 110                 name=name,
 111                 value=value,
 112                 port=None,
 113                 port_specified=False,
 114                 domain="ansible.com",
 115                 domain_specified=True,
 116                 domain_initial_dot=False,
 117                 path="/",
 118                 path_specified=True,
 119                 secure=False,
 120                 expires=None,
 121                 discard=False,
 122                 comment=None,
 123                 comment_url=None,
 124                 rest=None
 125             )
 126             cookies.set_cookie(cookie)
 127             add_header('Set-Cookie', '%s=%s' % (name, value))
 128 
 129         return r
 130 
 131     mocker = mocker.patch('ansible.module_utils.urls.open_url', new=make_cookies)
 132 
 133     r, info = fetch_url(fake_ansible_module, 'http://ansible.com/')
 134 
 135     assert info['cookies'] == {'Baz': 'qux', 'Foo': 'bar'}
 136     # Python sorts cookies in order of most specific (ie. longest) path first
 137     # items with the same path are reversed from response order
 138     assert info['cookies_string'] == 'Baz=qux; Foo=bar'
 139     # The key here has a `-` as opposed to what we see in the `uri` module that converts to `_`
 140     # Note: this is response order, which differs from cookies_string
 141     assert info['set-cookie'] == 'Foo=bar, Baz=qux'

 144 def test_fetch_url_nossl(open_url_mock, fake_ansible_module, mocker):
 145     mocker.patch('ansible.module_utils.urls.get_distribution', return_value='notredhat')
 146 
 147     open_url_mock.side_effect = NoSSLError
 148     with pytest.raises(FailJson) as excinfo:
 149         fetch_url(fake_ansible_module, 'http://ansible.com/')
 150 
 151     assert 'python-ssl' not in excinfo.value.kwargs['msg']
 152 
 153     mocker.patch('ansible.module_utils.urls.get_distribution', return_value='redhat')
 154 
 155     open_url_mock.side_effect = NoSSLError
 156     with pytest.raises(FailJson) as excinfo:
 157         fetch_url(fake_ansible_module, 'http://ansible.com/')
 158 
 159     assert 'python-ssl' in excinfo.value.kwargs['msg']
 160     assert 'http://ansible.com/' == excinfo.value.kwargs['url']
 161     assert excinfo.value.kwargs['status'] == -1
