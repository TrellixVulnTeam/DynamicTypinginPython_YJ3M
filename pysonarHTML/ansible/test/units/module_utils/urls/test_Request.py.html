  33 def test_Request_fallback(urlopen_mock, install_opener_mock, mocker):
  34     cookies = cookiejar.CookieJar()
  35     request = Request(
  36         headers={'foo': 'bar'},
  37         use_proxy=False,
  38         force=True,
  39         timeout=100,
  40         validate_certs=False,
  41         url_username='user',
  42         url_password='passwd',
  43         http_agent='ansible-tests',
  44         force_basic_auth=True,
  45         follow_redirects='all',
  46         client_cert='/tmp/client.pem',
  47         client_key='/tmp/client.key',
  48         cookies=cookies,
  49         unix_socket='/foo/bar/baz.sock',
  50         ca_path='/foo/bar/baz.pem',
  51     )
  52     fallback_mock = mocker.spy(request, '_fallback')
  53 
  54     r = request.open('GET', 'https://ansible.com')
  55 
  56     calls = [
  57         call(None, False),  # use_proxy
  58         call(None, True),  # force
  59         call(None, 100),  # timeout
  60         call(None, False),  # validate_certs
  61         call(None, 'user'),  # url_username
  62         call(None, 'passwd'),  # url_password
  63         call(None, 'ansible-tests'),  # http_agent
  64         call(None, True),  # force_basic_auth
  65         call(None, 'all'),  # follow_redirects
  66         call(None, '/tmp/client.pem'),  # client_cert
  67         call(None, '/tmp/client.key'),  # client_key
  68         call(None, cookies),  # cookies
  69         call(None, '/foo/bar/baz.sock'),  # unix_socket
  70         call(None, '/foo/bar/baz.pem'),  # ca_path
  71     ]
  72     fallback_mock.assert_has_calls(calls)
  73 
  74     assert fallback_mock.call_count == 14  # All but headers use fallback
  75 
  76     args = urlopen_mock.call_args[0]
  77     assert args[1] is None  # data, this is handled in the Request not urlopen
  78     assert args[2] == 100  # timeout
  79 
  80     req = args[0]
  81     assert req.headers == {
  82         'Authorization': b'Basic dXNlcjpwYXNzd2Q=',
  83         'Cache-control': 'no-cache',
  84         'Foo': 'bar',
  85         'User-agent': 'ansible-tests'
  86     }
  87     assert req.data is None
  88     assert req.get_method() == 'GET'

  91 def test_Request_open(urlopen_mock, install_opener_mock):
  92     r = Request().open('GET', 'https://ansible.com/')
  93     args = urlopen_mock.call_args[0]
  94     assert args[1] is None  # data, this is handled in the Request not urlopen
  95     assert args[2] == 10  # timeout

 101 
 102     opener = install_opener_mock.call_args[0][0]
 103     handlers = opener.handlers
 104 
 105     if not HAS_SSLCONTEXT:
 106         expected_handlers = (
 107             SSLValidationHandler,
 108             RedirectHandlerFactory(),  # factory, get handler
 109         )
 110     else:
 111         expected_handlers = (
 112             RedirectHandlerFactory(),  # factory, get handler
 113         )
 114 
 115     found_handlers = []
 116     for handler in handlers:
 117         if isinstance(handler, SSLValidationHandler) or handler.__class__.__name__ == 'RedirectHandler':
 118             found_handlers.append(handler)
 119 
 120     assert len(found_handlers) == len(expected_handlers)

 302 @pytest.mark.skipif(not HAS_SSLCONTEXT, reason="requires SSLContext")
 303 def test_Request_open_no_validate_certs(urlopen_mock, install_opener_mock):
 304     r = Request().open('GET', 'https://ansible.com/', validate_certs=False)
 305 
 306     opener = install_opener_mock.call_args[0][0]
 307     handlers = opener.handlers
 308 
 309     ssl_handler = None
 310     for handler in handlers:
 311         if isinstance(handler, HTTPSClientAuthHandler):
 312             ssl_handler = handler
 313             break
 314 
 315     assert ssl_handler is not None
 316 
 317     inst = ssl_handler._build_https_connection('foo')
 318     assert isinstance(inst, httplib.HTTPSConnection)
 319 
 320     context = ssl_handler._context
 321     assert context.protocol == ssl.PROTOCOL_SSLv23
 322     if ssl.OP_NO_SSLv2:
 323         assert context.options & ssl.OP_NO_SSLv2
 324     assert context.options & ssl.OP_NO_SSLv3
 325     assert context.verify_mode == ssl.CERT_NONE
 326     assert context.check_hostname is False
