  28 def test_no_redirs(urllib_req, request_body):
  29     handler = RedirectHandlerFactory('none', False)
  30     inst = handler()
  31     with pytest.raises(urllib_error.HTTPError):
  32         inst.redirect_request(urllib_req, request_body, 301, '301 Moved Permanently', {}, 'https://docs.ansible.com/')
