  19 def test_prepare_multipart():
  20     fixture_boundary = b'===============3996062709511591449=='
  21 
  22     here = os.path.dirname(__file__)
  23     multipart = os.path.join(here, 'fixtures/multipart.txt')
  24 
  25     client_cert = os.path.join(here, 'fixtures/client.pem')
  26     client_key = os.path.join(here, 'fixtures/client.key')
  27     client_txt = os.path.join(here, 'fixtures/client.txt')
  28     fields = {
  29         'form_field_1': 'form_value_1',
  30         'form_field_2': {
  31             'content': 'form_value_2',
  32         },
  33         'form_field_3': {
  34             'content': '<html></html>',
  35             'mime_type': 'text/html',
  36         },
  37         'form_field_4': {
  38             'content': '{"foo": "bar"}',
  39             'mime_type': 'application/json',
  40         },
  41         'file1': {
  42             'content': 'file_content_1',
  43             'filename': 'fake_file1.txt',
  44         },
  45         'file2': {
  46             'content': '<html></html>',
  47             'mime_type': 'text/html',
  48             'filename': 'fake_file2.html',
  49         },
  50         'file3': {
  51             'content': '{"foo": "bar"}',
  52             'mime_type': 'application/json',
  53             'filename': 'fake_file3.json',
  54         },
  55         'file4': {
  56             'filename': client_cert,
  57             'mime_type': 'text/plain',
  58         },
  59         'file5': {
  60             'filename': client_key,
  61         },
  62         'file6': {
  63             'filename': client_txt,
  64         },
  65     }
  66 
  67     content_type, b_data = prepare_multipart(fields)
  68 
  69     headers = Message()
  70     headers['Content-Type'] = content_type
  71     assert headers.get_content_type() == 'multipart/form-data'
  72     boundary = headers.get_boundary()
  73     assert boundary is not None
  74 
  75     with open(multipart, 'rb') as f:
  76         b_expected = f.read().replace(fixture_boundary, boundary.encode())
  77 
  78     # Depending on Python version, there may or may not be a trailing newline
  79     assert b_data.rstrip(b'\r\n') == b_expected.rstrip(b'\r\n')

  91 def test_unknown_mime(mocker):
  92     fields = {'foo': {'filename': 'foo.boom', 'content': 'foo'}}
  93     mocker.patch('mimetypes.guess_type', return_value=(None, None))
  94     content_type, b_data = prepare_multipart(fields)
  95     assert b'Content-Type: application/octet-stream' in b_data

  98 def test_bad_mime(mocker):
  99     fields = {'foo': {'filename': 'foo.boom', 'content': 'foo'}}
 100     mocker.patch('mimetypes.guess_type', side_effect=TypeError)
 101     content_type, b_data = prepare_multipart(fields)
 102     assert b'Content-Type: application/octet-stream' in b_data
