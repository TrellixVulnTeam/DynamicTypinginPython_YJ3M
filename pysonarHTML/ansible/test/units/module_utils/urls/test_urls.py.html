  14 def test_build_ssl_validation_error(mocker):
  15     mocker.patch.object(urls, 'HAS_SSLCONTEXT', new=False)
  16     mocker.patch.object(urls, 'HAS_URLLIB3_PYOPENSSLCONTEXT', new=False)
  17     mocker.patch.object(urls, 'HAS_URLLIB3_SSL_WRAP_SOCKET', new=False)
  18     with pytest.raises(urls.SSLValidationError) as excinfo:
  19         urls.build_ssl_validation_error('hostname', 'port', 'paths', exc=None)
  20 
  21     assert 'python >= 2.7.9' in to_native(excinfo.value)
  22     assert 'the python executable used' in to_native(excinfo.value)
  23     assert 'urllib3' in to_native(excinfo.value)
  24     assert 'python >= 2.6' in to_native(excinfo.value)
  25     assert 'validate_certs=False' in to_native(excinfo.value)
  26 
  27     mocker.patch.object(urls, 'HAS_SSLCONTEXT', new=True)
  28     with pytest.raises(urls.SSLValidationError) as excinfo:
  29         urls.build_ssl_validation_error('hostname', 'port', 'paths', exc=None)
  30 
  31     assert 'validate_certs=False' in to_native(excinfo.value)
  32 
  33     mocker.patch.object(urls, 'HAS_SSLCONTEXT', new=False)
  34     mocker.patch.object(urls, 'HAS_URLLIB3_PYOPENSSLCONTEXT', new=True)
  35     mocker.patch.object(urls, 'HAS_URLLIB3_SSL_WRAP_SOCKET', new=True)
  36 
  37     mocker.patch.object(urls, 'HAS_SSLCONTEXT', new=True)
  38     with pytest.raises(urls.SSLValidationError) as excinfo:
  39         urls.build_ssl_validation_error('hostname', 'port', 'paths', exc=None)
  40 
  41     assert 'urllib3' not in to_native(excinfo.value)
  42 
  43     with pytest.raises(urls.SSLValidationError) as excinfo:
  44         urls.build_ssl_validation_error('hostname', 'port', 'paths', exc='BOOM')
  45 
  46     assert 'BOOM' in to_native(excinfo.value)

  49 def test_maybe_add_ssl_handler(mocker):
  50     mocker.patch.object(urls, 'HAS_SSL', new=False)
  51     with pytest.raises(urls.NoSSLError):
  52         urls.maybe_add_ssl_handler('https://ansible.com/', True)
  53 
  54     mocker.patch.object(urls, 'HAS_SSL', new=True)
  55     url = 'https://user:passwd@ansible.com/'
  56     handler = urls.maybe_add_ssl_handler(url, True)
  57     assert handler.hostname == 'ansible.com'
  58     assert handler.port == 443
  59 
  60     url = 'https://ansible.com:4433/'
  61     handler = urls.maybe_add_ssl_handler(url, True)
  62     assert handler.hostname == 'ansible.com'
  63     assert handler.port == 4433
  64 
  65     url = 'https://user:passwd@ansible.com:4433/'
  66     handler = urls.maybe_add_ssl_handler(url, True)
  67     assert handler.hostname == 'ansible.com'
  68     assert handler.port == 4433
  69 
  70     url = 'https://ansible.com/'
  71     handler = urls.maybe_add_ssl_handler(url, True)
  72     assert handler.hostname == 'ansible.com'
  73     assert handler.port == 443
  74 
  75     url = 'http://ansible.com/'
  76     handler = urls.maybe_add_ssl_handler(url, True)
  77     assert handler is None
  78 
  79     url = 'https://[2a00:16d8:0:7::205]:4443/'
  80     handler = urls.maybe_add_ssl_handler(url, True)
  81     assert handler.hostname == '2a00:16d8:0:7::205'
  82     assert handler.port == 4443
  83 
  84     url = 'https://[2a00:16d8:0:7::205]/'
  85     handler = urls.maybe_add_ssl_handler(url, True)
  86     assert handler.hostname == '2a00:16d8:0:7::205'
  87     assert handler.port == 443

  95 def test_ParseResultDottedDict():
  96     url = 'https://ansible.com/blog'
  97     parts = urls.urlparse(url)
  98     dotted_parts = urls.ParseResultDottedDict(parts._asdict())
  99     assert parts[0] == dotted_parts.scheme
 100 
 101     assert dotted_parts.as_list() == list(parts)

 104 def test_unix_socket_patch_httpconnection_connect(mocker):
