   8 from ansible.module_utils.urls import RequestWithMethod

  10 
  11 def test_RequestWithMethod():
  12     get = RequestWithMethod('https://ansible.com/', 'GET')
  13     assert get.get_method() == 'GET'
  14 
  15     post = RequestWithMethod('https://ansible.com/', 'POST', data='foo', headers={'Bar': 'baz'})
  16     assert post.get_method() == 'POST'
  17     assert post.get_full_url() == 'https://ansible.com/'
  18     assert post.data == 'foo'
  19     assert post.headers == {'Bar': 'baz'}
  20 
  21     none = RequestWithMethod('https://ansible.com/', '')
  22     assert none.get_method() == 'GET'
