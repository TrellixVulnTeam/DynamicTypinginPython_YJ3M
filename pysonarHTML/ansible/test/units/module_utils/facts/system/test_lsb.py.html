  50 class TestLSBFacts(BaseFactsTest):
  51     __test__ = True
  52     gather_subset = ['!all', 'lsb']
  53     valid_subsets = ['lsb']
  54     fact_namespace = 'ansible_lsb'
  55     collector_class = LSBFactCollector
  56 
  57     def _mock_module(self):
  58         mock_module = Mock()
  59         mock_module.params = {'gather_subset': self.gather_subset,
  60                               'gather_timeout': 10,
  61                               'filter': '*'}
  62         mock_module.get_bin_path = Mock(return_value='/usr/bin/lsb_release')
  63         mock_module.run_command = Mock(return_value=(0, lsb_release_a_fedora_output, ''))
  64         return mock_module
  65 
  66     def test_lsb_release_bin(self):
  67         module = self._mock_module()
  68         fact_collector = self.collector_class()
  69         facts_dict = fact_collector.collect(module=module)
  70 
  71         self.assertIsInstance(facts_dict, dict)
  72         self.assertEqual(facts_dict['lsb']['release'], '25')
  73         self.assertEqual(facts_dict['lsb']['id'], 'Fedora')
  74         self.assertEqual(facts_dict['lsb']['description'], 'Fedora release 25 (Twenty Five)')
  75         self.assertEqual(facts_dict['lsb']['codename'], 'TwentyFive')
  76         self.assertEqual(facts_dict['lsb']['major_release'], '25')
  77 
  78     def test_etc_lsb_release(self):
  79         module = self._mock_module()
  80         module.get_bin_path = Mock(return_value=None)
  81         with patch('ansible.module_utils.facts.system.lsb.os.path.exists',
  82                    return_value=True):
  83             with patch('ansible.module_utils.facts.system.lsb.get_file_lines',
  84                        return_value=etc_lsb_release_ubuntu14.splitlines()):
  85                 fact_collector = self.collector_class()
  86                 facts_dict = fact_collector.collect(module=module)
  87 
  88         self.assertIsInstance(facts_dict, dict)
  89         self.assertEqual(facts_dict['lsb']['release'], '14.04')
  90         self.assertEqual(facts_dict['lsb']['id'], 'Ubuntu')
  91         self.assertEqual(facts_dict['lsb']['description'], 'Ubuntu 14.04.3 LTS')
  92         self.assertEqual(facts_dict['lsb']['codename'], 'trusty')
  93
