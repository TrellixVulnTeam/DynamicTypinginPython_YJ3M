  27 @pytest.mark.parametrize("stdin, testcase", product([{}], TESTSETS), ids=lambda x: x.get('name'), indirect=['stdin'])
  28 def test_distribution_version(am, mocker, testcase):
  29     """tests the distribution parsing code of the Facts class
  30 
  31     testsets have
  32     * a name (for output/debugging only)
  33     * input files that are faked
  34       * those should be complete and also include "irrelevant" files that might be mistaken as coming from other distributions
  35       * all files that are not listed here are assumed to not exist at all
  36     * the output of ansible.module_utils.distro.linux_distribution() [called platform.dist() for historical reasons]
  37     * results for the ansible variables distribution* and os_family
  38 
  39     """
  40 
  41     # prepare some mock functions to get the testdata in
  42     def mock_get_file_content(fname, default=None, strip=True):
  43         """give fake content if it exists, otherwise pretend the file is empty"""
  44         data = default
  45         if fname in testcase['input']:
  46             # for debugging
  47             print('faked %s for %s' % (fname, testcase['name']))
  48             data = testcase['input'][fname].strip()
  49         if strip and data is not None:
  50             data = data.strip()
  51         return data
  52 
  53     def mock_get_uname(am, flags):
  54         if '-v' in flags:
  55             return testcase.get('uname_v', None)
  56         elif '-r' in flags:
  57             return testcase.get('uname_r', None)
  58         else:
  59             return None
  60 
  61     def mock_file_exists(fname, allow_empty=False):
  62         if fname not in testcase['input']:
  63             return False
  64 
  65         if allow_empty:
  66             return True
  67         return bool(len(testcase['input'][fname]))
  68 
  69     def mock_platform_system():
  70         return testcase.get('platform.system', 'Linux')
  71 
  72     def mock_platform_release():
  73         return testcase.get('platform.release', '')
  74 
  75     def mock_platform_version():
  76         return testcase.get('platform.version', '')
  77 
  78     def mock_distro_name():
  79         return testcase['distro']['name']
  80 
  81     def mock_distro_id():
  82         return testcase['distro']['id']
  83 
  84     def mock_distro_version(best=False):
  85         if best:
  86             return testcase['distro']['version_best']
  87         return testcase['distro']['version']
  88 
  89     def mock_distro_codename():
  90         return testcase['distro']['codename']
  91 
  92     def mock_distro_os_release_info():
  93         return testcase['distro']['os_release_info']
  94 
  95     def mock_distro_lsb_release_info():
  96         return testcase['distro']['lsb_release_info']
  97 
  98     def mock_open(filename, mode='r'):
  99         if filename in testcase['input']:
 100             file_object = mocker.mock_open(read_data=testcase['input'][filename]).return_value
 101             file_object.__iter__.return_value = testcase['input'][filename].splitlines(True)
 102         else:
 103             file_object = real_open(filename, mode)
 104         return file_object
 105 
 106     def mock_os_path_is_file(filename):
 107         if filename in testcase['input']:
 108             return True
 109         return False
 110 
 111     mocker.patch('ansible.module_utils.facts.system.distribution.get_file_content', mock_get_file_content)
 112     mocker.patch('ansible.module_utils.facts.system.distribution.get_uname', mock_get_uname)
 113     mocker.patch('ansible.module_utils.facts.system.distribution._file_exists', mock_file_exists)
 114     mocker.patch('ansible.module_utils.distro.name', mock_distro_name)
 115     mocker.patch('ansible.module_utils.distro.id', mock_distro_id)
 116     mocker.patch('ansible.module_utils.distro.version', mock_distro_version)
 117     mocker.patch('ansible.module_utils.distro.codename', mock_distro_codename)
 118     mocker.patch(
 119         'ansible.module_utils.common.sys_info.distro.os_release_info',
 120         mock_distro_os_release_info)
 121     mocker.patch(
 122         'ansible.module_utils.common.sys_info.distro.lsb_release_info',
 123         mock_distro_lsb_release_info)
 124     mocker.patch('os.path.isfile', mock_os_path_is_file)
 125     mocker.patch('platform.system', mock_platform_system)
 126     mocker.patch('platform.release', mock_platform_release)
 127     mocker.patch('platform.version', mock_platform_version)
 128 
 129     real_open = builtins.open
 130     mocker.patch.object(builtins, 'open', new=mock_open)
 131 
 132     # run Facts()
 133     distro_collector = DistributionFactCollector()
 134     generated_facts = distro_collector.collect(am)
 135 
 136     # compare with the expected output
 137 
 138     # testcase['result'] has a list of variables and values it expects Facts() to set
 139     for key, val in testcase['result'].items():
 140         assert key in generated_facts
 141         msg = 'Comparing value of %s on %s, should: %s, is: %s' %\
 142             (key, testcase['name'], val, generated_facts[key])
 143         assert generated_facts[key] == val, msg
