  78 def mock_module(gather_subset=None,
  79                 filter=None):
  80     if gather_subset is None:
  81         gather_subset = ['all', '!facter', '!ohai']
  82     if filter is None:
  83         filter = '*'
  84     mock_module = Mock()
  85     mock_module.params = {'gather_subset': gather_subset,
  86                           'gather_timeout': 5,
  87                           'filter': filter}
  88     mock_module.get_bin_path = Mock(return_value=None)
  89     return mock_module

  92 def _collectors(module,
  93                 all_collector_classes=None,
  94                 minimal_gather_subset=None):
  95     gather_subset = module.params.get('gather_subset')
  96     if all_collector_classes is None:
  97         all_collector_classes = ALL_COLLECTOR_CLASSES
  98     if minimal_gather_subset is None:
  99         minimal_gather_subset = frozenset([])
 100 
 101     collector_classes = \
 102         collector.collector_classes_from_gather_subset(all_collector_classes=all_collector_classes,
 103                                                        minimal_gather_subset=minimal_gather_subset,
 104                                                        gather_subset=gather_subset)
 105 
 106     collectors = []
 107     for collector_class in collector_classes:
 108         collector_obj = collector_class()
 109         collectors.append(collector_obj)
 110 
 111     # Add a collector that knows what gather_subset we used so it it can provide a fact
 112     collector_meta_data_collector = \
 113         ansible_collector.CollectorMetaDataCollector(gather_subset=gather_subset,
 114                                                      module_setup=True)
 115     collectors.append(collector_meta_data_collector)
 116 
 117     return collectors

 238     def _assert_basics(self, facts):
 239         self.assertIsInstance(facts, dict)
 240         # just assert it's not almost empty

 242         # and that is not huge number of keys
 243         self.assertLess(len(facts), self.max_fact_count)

 245     # everything starts with ansible_ namespace
 246     def _assert_ansible_namespace(self, facts):
 247 
 248         # FIXME: kluge for non-namespace fact
 249         facts.pop('module_setup', None)
 250         facts.pop('gather_subset', None)
 251 
 252         for fact_key in facts:
 253             self.assertTrue(fact_key.startswith('ansible_'),
 254                             'The fact name "%s" does not startwith "ansible_"' % fact_key)

 277 class RequiresOtherFactCollector(collector.BaseFactCollector):
 278     name = 'requires_something'
 279 
 280     def collect(self, module=None, collected_facts=None):
 281         collected_facts = collected_facts or {}
 282         fact_dict = {}
 283         fact_dict['needed_fact'] = collected_facts['needed_fact']

 288 class ConCatFactCollector(collector.BaseFactCollector):
 289     name = 'concat_collected'
 290 
 291     def collect(self, module=None, collected_facts=None):
 292         collected_facts = collected_facts or {}
 293         fact_dict = {}
 294         con_cat_list = []
 295         for key, value in collected_facts.items():
 296             con_cat_list.append(value)
 297 
 298         fact_dict['concat_fact'] = '-'.join(con_cat_list)
 299         return fact_dict
