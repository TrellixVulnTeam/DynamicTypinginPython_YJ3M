  52 class ExceptionThrowingCollector(collector.BaseFactCollector):
  53     name = 'exc_throwing'
  54 
  55     def __init__(self, collectors=None, namespace=None, exception=None):
  56         super(ExceptionThrowingCollector, self).__init__(collectors, namespace)
  57         self._exception = exception or CollectorException('collection failed')
  58 
  59     def collect(self, module=None, collected_facts=None):
  60         raise self._exception

 142                     'ro': True,
