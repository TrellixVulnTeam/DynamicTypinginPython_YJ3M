 185 class TestFacterCollector(BaseFactsTest):
 186     __test__ = True
 187     gather_subset = ['!all', 'facter']
 188     valid_subsets = ['facter']
 189     fact_namespace = 'ansible_facter'
 190     collector_class = FacterFactCollector
 191 
 192     def _mock_module(self):
 193         mock_module = Mock()
 194         mock_module.params = {'gather_subset': self.gather_subset,
 195                               'gather_timeout': 10,
 196                               'filter': '*'}
 197         mock_module.get_bin_path = Mock(return_value='/not/actually/facter')
 198         mock_module.run_command = Mock(return_value=(0, facter_json_output, ''))
 199         return mock_module
 200 
 201     @patch('ansible.module_utils.facts.other.facter.FacterFactCollector.get_facter_output')
 202     def test_bogus_json(self, mock_get_facter_output):
 203         module = self._mock_module()
 204 
 205         # bogus json
 206         mock_get_facter_output.return_value = '{'
 207 
 208         fact_collector = self.collector_class()
 209         facts_dict = fact_collector.collect(module=module)
 210 
 211         self.assertIsInstance(facts_dict, dict)
 212         self.assertEqual(facts_dict, {})
 213 
 214     @patch('ansible.module_utils.facts.other.facter.FacterFactCollector.run_facter')
 215     def test_facter_non_zero_return_code(self, mock_run_facter):
 216         module = self._mock_module()
 217 
 218         # bogus json
 219         mock_run_facter.return_value = (1, '{}', '')
 220 
 221         fact_collector = self.collector_class()
 222         facts_dict = fact_collector.collect(module=module)
 223 
 224         self.assertIsInstance(facts_dict, dict)
 225 
 226         # This assumes no 'facter' entry at all is correct
 227         self.assertNotIn('facter', facts_dict)
 228         self.assertEqual(facts_dict, {})
