  26 class BaseFactsTest(unittest.TestCase):
  27     # just a base class, not an actual test
  28     __test__ = False
  29 
  30     gather_subset = ['all']
  31     valid_subsets = None
  32     fact_namespace = None
  33     collector_class = None
  34 
  35     # a dict ansible_facts. Some fact collectors depend on facts gathered by
  36     # other collectors (like 'ansible_architecture' or 'ansible_system') which
  37     # can be passed via the collected_facts arg to collect()
  38     collected_facts = None
  39 
  40     def _mock_module(self):
  41         mock_module = Mock()
  42         mock_module.params = {'gather_subset': self.gather_subset,
  43                               'gather_timeout': 5,
  44                               'filter': '*'}
  45         mock_module.get_bin_path = Mock(return_value=None)
  46         return mock_module
  47 
  48     @patch('platform.system', return_value='Linux')
  49     @patch('ansible.module_utils.facts.system.service_mgr.get_file_content', return_value='systemd')
  50     def test_collect(self, mock_gfc, mock_ps):
  51         module = self._mock_module()
  52         fact_collector = self.collector_class()
  53         facts_dict = fact_collector.collect(module=module, collected_facts=self.collected_facts)
  54         self.assertIsInstance(facts_dict, dict)
  55         return facts_dict
  56 
  57     @patch('platform.system', return_value='Linux')
  58     @patch('ansible.module_utils.facts.system.service_mgr.get_file_content', return_value='systemd')
  59     def test_collect_with_namespace(self, mock_gfc, mock_ps):
  60         module = self._mock_module()
  61         fact_collector = self.collector_class()
  62         facts_dict = fact_collector.collect_with_namespace(module=module,
  63                                                            collected_facts=self.collected_facts)
  64         self.assertIsInstance(facts_dict, dict)
  65         return facts_dict
