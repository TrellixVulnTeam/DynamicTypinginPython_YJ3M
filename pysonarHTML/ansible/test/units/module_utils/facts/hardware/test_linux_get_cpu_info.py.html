  13 def test_get_cpu_info(mocker):
  14     module = mocker.Mock()
  15     inst = linux.LinuxHardware(module)
  16 
  17     mocker.patch('os.path.exists', return_value=False)
  18     mocker.patch('os.access', return_value=True)
  19     for test in CPU_INFO_TEST_SCENARIOS:
  20         mocker.patch('ansible.module_utils.facts.hardware.linux.get_file_lines', side_effect=[[], test['cpuinfo']])
  21         mocker.patch('os.sched_getaffinity', create=True, return_value=test['sched_getaffinity'])
  22         module.run_command.return_value = (0, test['nproc_out'], '')
  23         collected_facts = {'ansible_architecture': test['architecture']}
  24 
  25         assert test['expected_result'] == inst.get_cpu_facts(collected_facts=collected_facts)

  44 def test_get_cpu_info_missing_arch(mocker):
  45     module = mocker.Mock()
  46     inst = linux.LinuxHardware(module)
  47 
  48     # ARM and Power will report incorrect processor count if architecture is not available
  49     mocker.patch('os.path.exists', return_value=False)
  50     mocker.patch('os.access', return_value=True)
  51     for test in CPU_INFO_TEST_SCENARIOS:
  52         mocker.patch('ansible.module_utils.facts.hardware.linux.get_file_lines', side_effect=[[], test['cpuinfo']])
  53         mocker.patch('os.sched_getaffinity', create=True, return_value=test['sched_getaffinity'])
  54 
  55         module.run_command.return_value = (0, test['nproc_out'], '')
  56 
  57         test_result = inst.get_cpu_facts()
  58 
  59         if test['architecture'].startswith(('armv', 'aarch', 'ppc')):
  60             assert test['expected_result'] != test_result
  61         else:
  62             assert test['expected_result'] == test_result
