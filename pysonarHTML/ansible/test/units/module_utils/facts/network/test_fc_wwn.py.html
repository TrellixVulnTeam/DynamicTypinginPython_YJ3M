  58 def mock_get_bin_path(cmd, required=False):
  59     result = None
  60     if cmd == 'lsdev':
  61         result = '/usr/sbin/lsdev'
  62     elif cmd == 'lscfg':
  63         result = '/usr/sbin/lscfg'
  64     elif cmd == 'fcinfo':
  65         result = '/usr/sbin/fcinfo'
  66     return result
  67 
  68 
  69 def mock_run_command(cmd):
  70     rc = 0
  71     if 'lsdev' in cmd:
  72         result = LSDEV_OUTPUT
  73     elif 'lscfg' in cmd:
  74         result = LSCFG_OUTPUT
  75     elif 'fcinfo' in cmd:
  76         result = FCINFO_OUTPUT
  77     else:
  78         rc = 1
  79         result = 'Error'
  80     return (rc, result, '')

  83 def test_get_fc_wwn_info(mocker):
  84     module = Mock()
  85     inst = fc_wwn.FcWwnInitiatorFactCollector()
  86 
  87     mocker.patch.object(module, 'get_bin_path', side_effect=mock_get_bin_path)
  88     mocker.patch.object(module, 'run_command', side_effect=mock_run_command)
  89 
  90     d = {'aix6': ['10000090FA551508'], 'sunos5': ['10000090fa1658de']}
  91     for key, value in d.items():
  92         mocker.patch('sys.platform', key)
  93         wwn_expected = {"fibre_channel_wwn": value}
  94         assert wwn_expected == inst.collect(module=module)
