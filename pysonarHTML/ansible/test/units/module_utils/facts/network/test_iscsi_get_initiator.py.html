  40 def test_get_iscsi_info(mocker):
  41     module = Mock()
  42     inst = iscsi.IscsiInitiatorNetworkCollector()
  43 
  44     mocker.patch('sys.platform', 'aix6')
  45     mocker.patch('ansible.module_utils.facts.network.iscsi.get_bin_path', return_value='/usr/sbin/lsattr')
  46     mocker.patch.object(module, 'run_command', return_value=(0, LSATTR_OUTPUT, ''))
  47     aix_iscsi_expected = {"iscsi_iqn": "iqn.localhost.hostid.7f000002"}
  48     assert aix_iscsi_expected == inst.collect(module=module)
  49 
  50     mocker.patch('sys.platform', 'hp-ux')
  51     mocker.patch('ansible.module_utils.facts.network.iscsi.get_bin_path', return_value='/opt/iscsi/bin/iscsiutil')
  52     mocker.patch.object(module, 'run_command', return_value=(0, ISCSIUTIL_OUTPUT, ''))
  53     hpux_iscsi_expected = {"iscsi_iqn": " iqn.2001-04.com.hp.stor:svcio"}
  54     assert hpux_iscsi_expected == inst.collect(module=module)
