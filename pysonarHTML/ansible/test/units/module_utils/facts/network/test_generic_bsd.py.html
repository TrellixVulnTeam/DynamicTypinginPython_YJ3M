 122 class TestGenericBsdNetworkNetBSD(unittest.TestCase):
 123     gather_subset = ['all']
 124 
 125     def setUp(self):
 126         self.maxDiff = None
 127         self.longMessage = True
 128 
 129     # TODO: extract module run_command/get_bin_path usage to methods I can mock without mocking all of run_command
 130     def test(self):
 131         module = self._mock_module()
 132         module.get_bin_path.side_effect = get_bin_path
 133         module.run_command.side_effect = run_command_old_ifconfig
 134 
 135         bsd_net = generic_bsd.GenericBsdIfconfigNetwork(module)
 136 
 137         res = bsd_net.populate()
 138         self.assertDictEqual(res, NETBSD_EXPECTED)
 139 
 140     def test_ifconfig_post_7_1(self):
 141         module = self._mock_module()
 142         module.get_bin_path.side_effect = get_bin_path
 143         module.run_command.side_effect = run_command_post_7_1_ifconfig
 144 
 145         bsd_net = generic_bsd.GenericBsdIfconfigNetwork(module)
 146 
 147         res = bsd_net.populate()
 148         self.assertDictEqual(res, NETBSD_EXPECTED)
 149 
 150     def test_netbsd_ifconfig_old_and_new(self):
 151         module_new = self._mock_module()
 152         module_new.get_bin_path.side_effect = get_bin_path
 153         module_new.run_command.side_effect = run_command_post_7_1_ifconfig
 154 
 155         bsd_net_new = generic_bsd.GenericBsdIfconfigNetwork(module_new)
 156         res_new = bsd_net_new.populate()
 157 
 158         module_old = self._mock_module()
 159         module_old.get_bin_path.side_effect = get_bin_path
 160         module_old.run_command.side_effect = run_command_old_ifconfig
 161 
 162         bsd_net_old = generic_bsd.GenericBsdIfconfigNetwork(module_old)
 163         res_old = bsd_net_old.populate()
 164 
 165         self.assertDictEqual(res_old, res_new)
 166         self.assertDictEqual(res_old, NETBSD_EXPECTED)
 167         self.assertDictEqual(res_new, NETBSD_EXPECTED)
 168 
 169     def _mock_module(self):
 170         mock_module = Mock()
 171         mock_module.params = {'gather_subset': self.gather_subset,
 172                               'gather_timeout': 5,
 173                               'filter': '*'}
 174         mock_module.get_bin_path = Mock(return_value=None)
 175         return mock_module
