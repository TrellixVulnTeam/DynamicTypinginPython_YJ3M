  23     def clear_modules(self, mods):
  24         for mod in mods:
  25             if mod in sys.modules:
  26                 del sys.modules[mod]

  28     @patch.object(builtins, '__import__')
  29     def test_module_utils_basic_import_syslog(self, mock_import):
  30         def _mock_import(name, *args, **kwargs):
  31             if name == 'syslog':
  32                 raise ImportError
  33             return realimport(name, *args, **kwargs)
  34 
  35         self.clear_modules(['syslog', 'ansible.module_utils.basic'])
  36         mod = builtins.__import__('ansible.module_utils.basic')
  37         self.assertTrue(mod.module_utils.basic.HAS_SYSLOG)
  38 
  39         self.clear_modules(['syslog', 'ansible.module_utils.basic'])
  40         mock_import.side_effect = _mock_import
  41         mod = builtins.__import__('ansible.module_utils.basic')
  42         self.assertFalse(mod.module_utils.basic.HAS_SYSLOG)

  84     @patch.object(builtins, '__import__')
  85     @unittest.skipIf(sys.version_info[0] >= 3, "literal_eval is available in every version of Python3")
  86     def test_module_utils_basic_import_literal_eval(self, mock_import):
  87         def _mock_import(name, *args, **kwargs):
  88             try:
  89                 fromlist = kwargs.get('fromlist', args[2])
  90             except IndexError:
  91                 fromlist = []
  92             if name == 'ast' and 'literal_eval' in fromlist:
  93                 raise ImportError
  94             return realimport(name, *args, **kwargs)
  95 
  96         mock_import.side_effect = _mock_import
  97         self.clear_modules(['ast', 'ansible.module_utils.basic'])
  98         mod = builtins.__import__('ansible.module_utils.basic')
  99         self.assertEqual(mod.module_utils.basic.literal_eval("'1'"), "1")
 100         self.assertEqual(mod.module_utils.basic.literal_eval("1"), 1)
 101         self.assertEqual(mod.module_utils.basic.literal_eval("-1"), -1)
 102         self.assertEqual(mod.module_utils.basic.literal_eval("(1,2,3)"), (1, 2, 3))
 103         self.assertEqual(mod.module_utils.basic.literal_eval("[1]"), [1])
 104         self.assertEqual(mod.module_utils.basic.literal_eval("True"), True)
 105         self.assertEqual(mod.module_utils.basic.literal_eval("False"), False)
 106         self.assertEqual(mod.module_utils.basic.literal_eval("None"), None)
 107         # self.assertEqual(mod.module_utils.basic.literal_eval('{"a": 1}'), dict(a=1))
 108         self.assertRaises(ValueError, mod.module_utils.basic.literal_eval, "asdfasdfasdf")
