 101 @pytest.mark.skipif(not ansible.module_utils.basic.has_journal, reason='python systemd bindings not installed')
 102 class TestAnsibleModuleLogJournal:
 103     """Test the AnsibleModule Log Method"""
 104 
 105     OUTPUT_DATA = [
 106         (u'Text string', u'Text string'),
 107         (u'Toshio くらとみ non-ascii test', u'Toshio くらとみ non-ascii test'),
 108         (b'Byte string', u'Byte string'),
 109         (u'Toshio くらとみ non-ascii test'.encode('utf-8'), u'Toshio くらとみ non-ascii test'),
 110         (b'non-utf8 :\xff: test', b'non-utf8 :\xff: test'.decode('utf-8', 'replace')),
 111     ]
 112 
 113     @pytest.mark.parametrize('no_log, stdin', (product((True, False), [{}])), indirect=['stdin'])
 114     def test_no_log(self, am, mocker, no_log):
 115         journal_send = mocker.patch('systemd.journal.send')
 116         am.no_log = no_log
 117         am.log('unittest no_log')
 118         if no_log:
 119             assert not journal_send.called
 120         else:
 121             assert journal_send.called == 1
 122             # Message
 123             # call_args is a 2-tuple of (arg_list, kwarg_dict)
 124             assert journal_send.call_args[1]['MESSAGE'].endswith('unittest no_log'), 'Message was not sent to log'
 125             # log adds this journal field
 126             assert 'MODULE' in journal_send.call_args[1]
 127             assert 'basic.py' in journal_send.call_args[1]['MODULE']
 128 
 129     # pylint bug: https://github.com/PyCQA/pylint/issues/511
 130     @pytest.mark.parametrize('msg, param, stdin',
 131                              ((m, p, {}) for m, p in OUTPUT_DATA),  # pylint: disable=undefined-variable
 132                              indirect=['stdin'])
 133     def test_output_matches(self, am, mocker, msg, param):
 134         journal_send = mocker.patch('systemd.journal.send')
 135         am.log(msg)
 136         assert journal_send.call_count == 1, 'journal.send not called exactly once'
 137         assert journal_send.call_args[1]['MESSAGE'].endswith(param)
 138 
 139     @pytest.mark.parametrize('stdin', ({},), indirect=['stdin'])
 140     def test_log_args(self, am, mocker):
 141         journal_send = mocker.patch('systemd.journal.send')
 142         am.log('unittest log_args', log_args=dict(TEST='log unittest'))
 143         assert journal_send.called == 1
 144         assert journal_send.call_args[1]['MESSAGE'].endswith('unittest log_args'), 'Message was not sent to log'
 145 
 146         # log adds this journal field
 147         assert 'MODULE' in journal_send.call_args[1]
 148         assert 'basic.py' in journal_send.call_args[1]['MODULE']
 149 
 150         # We added this journal field
 151         assert 'TEST' in journal_send.call_args[1]
 152         assert 'log unittest' in journal_send.call_args[1]['TEST']
