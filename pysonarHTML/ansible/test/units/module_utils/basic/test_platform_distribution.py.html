 133 class TestLoadPlatformSubclass:
 134     class LinuxTest:
 135         pass
 136 
 137     class Foo(LinuxTest):
 138         platform = "Linux"
 139         distribution = None
 140 
 141     class Bar(LinuxTest):
 142         platform = "Linux"
 143         distribution = "Bar"
 144 
 145     def test_not_linux(self):
 146         # if neither match, the fallback should be the top-level class
 147         with patch('platform.system', return_value="Foo"):
 148             with patch('ansible.module_utils.common.sys_info.get_distribution', return_value=None):
 149                 assert isinstance(load_platform_subclass(self.LinuxTest), self.LinuxTest)
 150 
 151     @pytest.mark.usefixtures("platform_linux")
 152     def test_get_distribution_none(self):
 153         # match just the platform class, not a specific distribution
 154         with patch('ansible.module_utils.common.sys_info.get_distribution', return_value=None):
 155             assert isinstance(load_platform_subclass(self.LinuxTest), self.Foo)
 156 
 157     @pytest.mark.usefixtures("platform_linux")
 158     def test_get_distribution_found(self):
 159         # match both the distribution and platform class
 160         with patch('ansible.module_utils.common.sys_info.get_distribution', return_value="Bar"):
 161             assert isinstance(load_platform_subclass(self.LinuxTest), self.Bar)
