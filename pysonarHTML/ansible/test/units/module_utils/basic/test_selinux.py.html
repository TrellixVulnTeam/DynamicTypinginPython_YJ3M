  56     def test_module_utils_basic_ansible_module_selinux_enabled(self):
  57         from ansible.module_utils import basic
  58         basic._ANSIBLE_ARGS = None
  59 
  60         am = basic.AnsibleModule(
  61             argument_spec=dict(),
  62         )
  63 
  64         # we first test the cases where the python selinux lib is
  65         # not installed, which has two paths: one in which the system
  66         # does have selinux installed (and the selinuxenabled command
  67         # is present and returns 0 when run), or selinux is not installed
  68         basic.HAVE_SELINUX = False
  69         am.get_bin_path = MagicMock()
  70         am.get_bin_path.return_value = '/path/to/selinuxenabled'
  71         am.run_command = MagicMock()
  72         am.run_command.return_value = (0, '', '')
  73         self.assertRaises(SystemExit, am.selinux_enabled)
  74         am.get_bin_path.return_value = None
  75         self.assertEqual(am.selinux_enabled(), False)
  76 
  77         # finally we test the case where the python selinux lib is installed,
  78         # and both possibilities there (enabled vs. disabled)
  79         basic.HAVE_SELINUX = True
  80         basic.selinux = Mock()
  81         with patch.dict('sys.modules', {'selinux': basic.selinux}):
  82             with patch('selinux.is_selinux_enabled', return_value=0):
  83                 self.assertEqual(am.selinux_enabled(), False)
  84             with patch('selinux.is_selinux_enabled', return_value=1):
  85                 self.assertEqual(am.selinux_enabled(), True)
  86         delattr(basic, 'selinux')

  88     def test_module_utils_basic_ansible_module_selinux_default_context(self):
  89         from ansible.module_utils import basic
  90         basic._ANSIBLE_ARGS = None
  91 
  92         am = basic.AnsibleModule(
  93             argument_spec=dict(),
  94         )
  95 
  96         am.selinux_initial_context = MagicMock(return_value=[None, None, None, None])
  97         am.selinux_enabled = MagicMock(return_value=True)
  98 
  99         # we first test the cases where the python selinux lib is not installed
 100         basic.HAVE_SELINUX = False
 101         self.assertEqual(am.selinux_default_context(path='/foo/bar'), [None, None, None, None])
 102 
 103         # all following tests assume the python selinux bindings are installed
 104         basic.HAVE_SELINUX = True
 105 
 106         basic.selinux = Mock()
 107 
 108         with patch.dict('sys.modules', {'selinux': basic.selinux}):
 109             # next, we test with a mocked implementation of selinux.matchpathcon to simulate
 110             # an actual context being found
 111             with patch('selinux.matchpathcon', return_value=[0, 'unconfined_u:object_r:default_t:s0']):
 112                 self.assertEqual(am.selinux_default_context(path='/foo/bar'), ['unconfined_u', 'object_r', 'default_t', 's0'])
 113 
 114             # we also test the case where matchpathcon returned a failure
 115             with patch('selinux.matchpathcon', return_value=[-1, '']):
 116                 self.assertEqual(am.selinux_default_context(path='/foo/bar'), [None, None, None, None])
 117 
 118             # finally, we test where an OSError occurred during matchpathcon's call
 119             with patch('selinux.matchpathcon', side_effect=OSError):
 120                 self.assertEqual(am.selinux_default_context(path='/foo/bar'), [None, None, None, None])
 121 
 122         delattr(basic, 'selinux')

 124     def test_module_utils_basic_ansible_module_selinux_context(self):
 125         from ansible.module_utils import basic
 126         basic._ANSIBLE_ARGS = None
 127 
 128         am = basic.AnsibleModule(
 129             argument_spec=dict(),
 130         )
 131 
 132         am.selinux_initial_context = MagicMock(return_value=[None, None, None, None])
 133         am.selinux_enabled = MagicMock(return_value=True)
 134 
 135         # we first test the cases where the python selinux lib is not installed
 136         basic.HAVE_SELINUX = False
 137         self.assertEqual(am.selinux_context(path='/foo/bar'), [None, None, None, None])
 138 
 139         # all following tests assume the python selinux bindings are installed
 140         basic.HAVE_SELINUX = True
 141 
 142         basic.selinux = Mock()
 143 
 144         with patch.dict('sys.modules', {'selinux': basic.selinux}):
 145             # next, we test with a mocked implementation of selinux.lgetfilecon_raw to simulate
 146             # an actual context being found
 147             with patch('selinux.lgetfilecon_raw', return_value=[0, 'unconfined_u:object_r:default_t:s0']):
 148                 self.assertEqual(am.selinux_context(path='/foo/bar'), ['unconfined_u', 'object_r', 'default_t', 's0'])
 149 
 150             # we also test the case where matchpathcon returned a failure
 151             with patch('selinux.lgetfilecon_raw', return_value=[-1, '']):
 152                 self.assertEqual(am.selinux_context(path='/foo/bar'), [None, None, None, None])
 153 
 154             # finally, we test where an OSError occurred during matchpathcon's call
 155             e = OSError()
 156             e.errno = errno.ENOENT
 157             with patch('selinux.lgetfilecon_raw', side_effect=e):
 158                 self.assertRaises(SystemExit, am.selinux_context, path='/foo/bar')
 159 
 160             e = OSError()
 161             with patch('selinux.lgetfilecon_raw', side_effect=e):
 162                 self.assertRaises(SystemExit, am.selinux_context, path='/foo/bar')
 163 
 164         delattr(basic, 'selinux')

 166     def test_module_utils_basic_ansible_module_is_special_selinux_path(self):
 167         from ansible.module_utils import basic
 168 
 169         args = json.dumps(dict(ANSIBLE_MODULE_ARGS={'_ansible_selinux_special_fs': "nfs,nfsd,foos",
 170                                                     '_ansible_remote_tmp': "/tmp",
 171                                                     '_ansible_keep_remote_files': False}))
 172 
 173         with swap_stdin_and_argv(stdin_data=args):
 174             basic._ANSIBLE_ARGS = None
 175             am = basic.AnsibleModule(
 176                 argument_spec=dict(),
 177             )
 178 
 179             def _mock_find_mount_point(path):
 180                 if path.startswith('/some/path'):
 181                     return '/some/path'
 182                 elif path.startswith('/weird/random/fstype'):
 183                     return '/weird/random/fstype'
 184                 return '/'
 185 
 186             am.find_mount_point = MagicMock(side_effect=_mock_find_mount_point)
 187             am.selinux_context = MagicMock(return_value=['foo_u', 'foo_r', 'foo_t', 's0'])
 188 
 189             m = mock_open()
 190             m.side_effect = OSError
 191 
 192             with patch.object(builtins, 'open', m, create=True):
 193                 self.assertEqual(am.is_special_selinux_path('/some/path/that/should/be/nfs'), (False, None))
 194 
 195             mount_data = [
 196                 '/dev/disk1 / ext4 rw,seclabel,relatime,data=ordered 0 0\n',
 197                 '10.1.1.1:/path/to/nfs /some/path nfs ro 0 0\n',
 198                 'whatever /weird/random/fstype foos rw 0 0\n',
 199             ]
 200 
 201             # mock_open has a broken readlines() implementation apparently...
 202             # this should work by default but doesn't, so we fix it
 203             m = mock_open(read_data=''.join(mount_data))
 204             m.return_value.readlines.return_value = mount_data
 205 
 206             with patch.object(builtins, 'open', m, create=True):
 207                 self.assertEqual(am.is_special_selinux_path('/some/random/path'), (False, None))
 208                 self.assertEqual(am.is_special_selinux_path('/some/path/that/should/be/nfs'), (True, ['foo_u', 'foo_r', 'foo_t', 's0']))
 209                 self.assertEqual(am.is_special_selinux_path('/weird/random/fstype/path'), (True, ['foo_u', 'foo_r', 'foo_t', 's0']))

 211     def test_module_utils_basic_ansible_module_set_context_if_different(self):
 212         from ansible.module_utils import basic
 213         basic._ANSIBLE_ARGS = None
 214 
 215         am = basic.AnsibleModule(
 216             argument_spec=dict(),
 217         )
 218 
 219         basic.HAVE_SELINUX = False
 220 
 221         am.selinux_enabled = MagicMock(return_value=False)
 222         self.assertEqual(am.set_context_if_different('/path/to/file', ['foo_u', 'foo_r', 'foo_t', 's0'], True), True)
 223         self.assertEqual(am.set_context_if_different('/path/to/file', ['foo_u', 'foo_r', 'foo_t', 's0'], False), False)
 224 
 225         basic.HAVE_SELINUX = True
 226 
 227         am.selinux_enabled = MagicMock(return_value=True)
 228         am.selinux_context = MagicMock(return_value=['bar_u', 'bar_r', None, None])
 229         am.is_special_selinux_path = MagicMock(return_value=(False, None))
 230 
 231         basic.selinux = Mock()
 232         with patch.dict('sys.modules', {'selinux': basic.selinux}):
 233             with patch('selinux.lsetfilecon', return_value=0) as m:
 234                 self.assertEqual(am.set_context_if_different('/path/to/file', ['foo_u', 'foo_r', 'foo_t', 's0'], False), True)
 235                 m.assert_called_with('/path/to/file', 'foo_u:foo_r:foo_t:s0')
 236                 m.reset_mock()
 237                 am.check_mode = True
 238                 self.assertEqual(am.set_context_if_different('/path/to/file', ['foo_u', 'foo_r', 'foo_t', 's0'], False), True)
 239                 self.assertEqual(m.called, False)
 240                 am.check_mode = False
 241 
 242             with patch('selinux.lsetfilecon', return_value=1) as m:
 243                 self.assertRaises(SystemExit, am.set_context_if_different, '/path/to/file', ['foo_u', 'foo_r', 'foo_t', 's0'], True)
 244 
 245             with patch('selinux.lsetfilecon', side_effect=OSError) as m:
 246                 self.assertRaises(SystemExit, am.set_context_if_different, '/path/to/file', ['foo_u', 'foo_r', 'foo_t', 's0'], True)
 247 
 248             am.is_special_selinux_path = MagicMock(return_value=(True, ['sp_u', 'sp_r', 'sp_t', 's0']))
 249 
 250             with patch('selinux.lsetfilecon', return_value=0) as m:
 251                 self.assertEqual(am.set_context_if_different('/path/to/file', ['foo_u', 'foo_r', 'foo_t', 's0'], False), True)
 252                 m.assert_called_with('/path/to/file', 'sp_u:sp_r:sp_t:s0')
 253 
 254         delattr(basic, 'selinux')
