  60     def test_module_utils_basic_ansible_module_set_owner_if_different(self):
  61         from ansible.module_utils import basic
  62         basic._ANSIBLE_ARGS = None
  63 
  64         am = basic.AnsibleModule(
  65             argument_spec=dict(),
  66         )
  67 
  68         self.assertEqual(am.set_owner_if_different('/path/to/file', None, True), True)
  69         self.assertEqual(am.set_owner_if_different('/path/to/file', None, False), False)
  70 
  71         am.user_and_group = MagicMock(return_value=(500, 500))
  72 
  73         with patch('os.lchown', return_value=None) as m:
  74             self.assertEqual(am.set_owner_if_different('/path/to/file', 0, False), True)
  75             m.assert_called_with(b'/path/to/file', 0, -1)
  76 
  77             def _mock_getpwnam(*args, **kwargs):
  78                 mock_pw = MagicMock()
  79                 mock_pw.pw_uid = 0
  80                 return mock_pw
  81 
  82             m.reset_mock()
  83             with patch('pwd.getpwnam', side_effect=_mock_getpwnam):
  84                 self.assertEqual(am.set_owner_if_different('/path/to/file', 'root', False), True)
  85                 m.assert_called_with(b'/path/to/file', 0, -1)
  86 
  87             with patch('pwd.getpwnam', side_effect=KeyError):
  88                 self.assertRaises(SystemExit, am.set_owner_if_different, '/path/to/file', 'root', False)
  89 
  90             m.reset_mock()
  91             am.check_mode = True
  92             self.assertEqual(am.set_owner_if_different('/path/to/file', 0, False), True)
  93             self.assertEqual(m.called, False)
  94             am.check_mode = False
  95 
  96         with patch('os.lchown', side_effect=OSError) as m:
  97             self.assertRaises(SystemExit, am.set_owner_if_different, '/path/to/file', 'root', False)
