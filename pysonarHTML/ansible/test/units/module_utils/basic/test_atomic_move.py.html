 162 @pytest.mark.parametrize('stdin', [{}], indirect=['stdin'])
 163 def test_rename_failure(atomic_am, atomic_mocks, mocker, capfd):

 220     else:
 221         assert not atomic_am.selinux_default_context.called
 222         assert not atomic_am.set_context_if_different.called
