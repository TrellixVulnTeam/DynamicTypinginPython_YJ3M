  14 @pytest.mark.parametrize('stdin', [{}], indirect=['stdin'])
  15 def test_warn(am, capfd):
  16 
  17     am.warn('warning1')
  18

  25 @pytest.mark.parametrize('stdin', [{}], indirect=['stdin'])
  26 def test_deprecate(am, capfd):
  27     am.deprecate('deprecation1')
  28     am.deprecate('deprecation2', '2.3')  # pylint: disable=ansible-deprecated-no-collection-name
  29     am.deprecate('deprecation3', version='2.4')  # pylint: disable=ansible-deprecated-no-collection-name
  30     am.deprecate('deprecation4', date='2020-03-10')  # pylint: disable=ansible-deprecated-no-collection-name
  31     am.deprecate('deprecation5', collection_name='ansible.builtin')
  32     am.deprecate('deprecation6', '2.3', collection_name='ansible.builtin')
  33     am.deprecate('deprecation7', version='2.4', collection_name='ansible.builtin')
  34     am.deprecate('deprecation8', date='2020-03-10', collection_name='ansible.builtin')
  35 
  36     with pytest.raises(SystemExit):
  37         am.exit_json(deprecations=['deprecation9', ('deprecation10', '2.4')])
  38 
  39     out, err = capfd.readouterr()
  40     output = json.loads(out)
  41     assert ('warnings' not in output or output['warnings'] == [])
  42     assert output['deprecations'] == [
  43         {u'msg': u'deprecation1', u'version': None, u'collection_name': None},
  44         {u'msg': u'deprecation2', u'version': '2.3', u'collection_name': None},
  45         {u'msg': u'deprecation3', u'version': '2.4', u'collection_name': None},
  46         {u'msg': u'deprecation4', u'date': '2020-03-10', u'collection_name': None},
  47         {u'msg': u'deprecation5', u'version': None, u'collection_name': 'ansible.builtin'},
  48         {u'msg': u'deprecation6', u'version': '2.3', u'collection_name': 'ansible.builtin'},
  49         {u'msg': u'deprecation7', u'version': '2.4', u'collection_name': 'ansible.builtin'},
  50         {u'msg': u'deprecation8', u'date': '2020-03-10', u'collection_name': 'ansible.builtin'},
  51         {u'msg': u'deprecation9', u'version': None, u'collection_name': None},
  52         {u'msg': u'deprecation10', u'version': '2.4', u'collection_name': None},
  53     ]
