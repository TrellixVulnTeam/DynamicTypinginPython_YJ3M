 187 @pytest.mark.parametrize('argspec, expected, stdin', [(s[0], s[2], s[1]) for s in VALID_SPECS],
 188                          indirect=['stdin'])
 189 def test_validator_basic_types(argspec, expected, stdin):
 190 
 191     am = basic.AnsibleModule(argspec)
 192 
 193     if 'type' in argspec['arg']:
 194         if argspec['arg']['type'] == 'int':
 195             type_ = integer_types
 196         else:
 197             type_ = getattr(builtins, argspec['arg']['type'])
 198     else:
 199         type_ = str
 200 
 201     assert isinstance(am.params['arg'], type_)
 202     assert am.params['arg'] == expected
 203
