  47 @pytest.mark.parametrize('previous_changes, check_mode, exists, stdin',
  48                          product((True, False), (True, False), (True, False), ({},)),
  49                          indirect=['stdin'])
  50 def test_no_mode_given_returns_previous_changes(am, mock_stats, mock_lchmod, mocker, previous_changes, check_mode, exists):
  51     am.check_mode = check_mode
  52     mocker.patch('os.lstat', side_effect=[mock_stats['before']])
  53     m_lchmod = mocker.patch('os.lchmod', return_value=None, create=True)
  54     m_path_exists = mocker.patch('os.path.exists', return_value=exists)
  55 
  56     assert am.set_mode_if_different('/path/to/file', None, previous_changes) == previous_changes
  57     assert not m_lchmod.called
  58     assert not m_path_exists.called

  61 @pytest.mark.parametrize('mode, check_mode, stdin',
  62                          product(SYNONYMS_0660, (True, False), ({},)),
  63                          indirect=['stdin'])
  64 def test_mode_changed_to_0660(am, mock_stats, mocker, mode, check_mode):
  65     # Note: This is for checking that all the different ways of specifying
  66     # 0660 mode work.  It cannot be used to check that setting a mode that is
  67     # not equivalent to 0660 works.
  68     am.check_mode = check_mode
  69     mocker.patch('os.lstat', side_effect=[mock_stats['before'], mock_stats['after'], mock_stats['after']])
  70     m_lchmod = mocker.patch('os.lchmod', return_value=None, create=True)
  71     mocker.patch('os.path.exists', return_value=True)
  72 
  73     assert am.set_mode_if_different('/path/to/file', mode, False)
  74     if check_mode:
  75         assert not m_lchmod.called
  76     else:
  77         m_lchmod.assert_called_with(b'/path/to/file', 0o660)

  99 def test_missing_lchmod_is_not_link(am, mock_stats, mocker, monkeypatch, check_mode):

 101
