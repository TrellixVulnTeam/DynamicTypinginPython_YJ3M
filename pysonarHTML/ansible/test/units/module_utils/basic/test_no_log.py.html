 140     def test_hit_recursion_limit(self):
 141         """ Check that we do not hit a recursion limit"""
 142         data_list = []
 143         inner_list = data_list
 144         for i in range(0, 10000):
 145             new_list = []
 146             inner_list.append(new_list)
 147             inner_list = new_list
 148         inner_list.append('secret')
 149 
 150         # Check that this does not hit a recursion limit
 151         actual_data_list = remove_values(data_list, frozenset(('secret',)))
 152 
 153         levels = 0
 154         inner_list = actual_data_list
 155         while inner_list:
 156             if isinstance(inner_list, list):
 157                 self.assertEqual(len(inner_list), 1)
 158             else:
 159                 levels -= 1
 160                 break
 161             inner_list = inner_list[0]
 162             levels += 1
 163 
 164         self.assertEqual(inner_list, self.OMIT)
 165         self.assertEqual(levels, 10000)
