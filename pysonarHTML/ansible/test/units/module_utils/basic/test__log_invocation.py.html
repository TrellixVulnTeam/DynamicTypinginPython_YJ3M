   9 import pytest

  12 ARGS = dict(foo=False, bar=[1, 2, 3], bam="bam", baz=u'baz')
  13 ARGUMENT_SPEC = dict(
  14     foo=dict(default=True, type='bool'),
  15     bar=dict(default=[], type='list'),
  16     bam=dict(default="bam"),
  17     baz=dict(default=u"baz"),
  18     password=dict(default=True),
  19     no_log=dict(default="you shouldn't see me", no_log=True),
  20 )
  21 
  22 
  23 @pytest.mark.parametrize('am, stdin', [(ARGUMENT_SPEC, ARGS)], indirect=['am', 'stdin'])
  24 def test_module_utils_basic__log_invocation(am, mocker):
  25 
  26     am.log = mocker.MagicMock()
  27     am._log_invocation()
  28 
  29     # Message is generated from a dict so it will be in an unknown order.
  30     # have to check this manually rather than with assert_called_with()
  31     args = am.log.call_args[0]
  32     assert len(args) == 1
  33     message = args[0]
  34 
  35     assert len(message) == \
  36         len('Invoked with bam=bam bar=[1, 2, 3] foo=False baz=baz no_log=NOT_LOGGING_PARAMETER password=NOT_LOGGING_PASSWORD')
  37 
  38     assert message.startswith('Invoked with ')
  39     assert ' bam=bam' in message
  40     assert ' bar=[1, 2, 3]' in message
  41     assert ' foo=False' in message
  42     assert ' baz=baz' in message
  43     assert ' no_log=NOT_LOGGING_PARAMETER' in message
  44     assert ' password=NOT_LOGGING_PASSWORD' in message
  45 
  46     kwargs = am.log.call_args[1]
  47     assert kwargs == \
  48         dict(log_args={
  49             'foo': 'False',
  50             'bar': '[1, 2, 3]',
  51             'bam': 'bam',
  52             'baz': 'baz',
  53             'password': 'NOT_LOGGING_PASSWORD',
  54             'no_log': 'NOT_LOGGING_PARAMETER',
  55         })
