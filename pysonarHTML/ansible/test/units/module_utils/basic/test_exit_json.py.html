  20 class TestAnsibleModuleExitJson:
  21     """
  22     Test that various means of calling exitJson and FailJson return the messages they've been given
  23     """
  24     DATA = (
  25         ({}, {'invocation': EMPTY_INVOCATION}),
  26         ({'msg': 'message'}, {'msg': 'message', 'invocation': EMPTY_INVOCATION}),
  27         ({'msg': 'success', 'changed': True},
  28          {'msg': 'success', 'changed': True, 'invocation': EMPTY_INVOCATION}),
  29         ({'msg': 'nochange', 'changed': False},
  30          {'msg': 'nochange', 'changed': False, 'invocation': EMPTY_INVOCATION}),
  31         ({'msg': 'message', 'date': DATETIME.date()},
  32          {'msg': 'message', 'date': DATETIME.date().isoformat(), 'invocation': EMPTY_INVOCATION}),
  33         ({'msg': 'message', 'datetime': DATETIME},
  34          {'msg': 'message', 'datetime': DATETIME.isoformat(), 'invocation': EMPTY_INVOCATION}),
  35     )
  36 
  37     # pylint bug: https://github.com/PyCQA/pylint/issues/511
  38     # pylint: disable=undefined-variable
  39     @pytest.mark.parametrize('args, expected, stdin', ((a, e, {}) for a, e in DATA), indirect=['stdin'])
  40     def test_exit_json_exits(self, am, capfd, args, expected):
  41         with pytest.raises(SystemExit) as ctx:
  42             am.exit_json(**args)
  43         assert ctx.value.code == 0
  44 
  45         out, err = capfd.readouterr()
  46         return_val = json.loads(out)
  47         assert return_val == expected
  48 
  49     # Fail_json is only legal if it's called with a message
  50     # pylint bug: https://github.com/PyCQA/pylint/issues/511
  51     @pytest.mark.parametrize('args, expected, stdin',
  52                              ((a, e, {}) for a, e in DATA if 'msg' in a),  # pylint: disable=undefined-variable
  53                              indirect=['stdin'])
  54     def test_fail_json_exits(self, am, capfd, args, expected):
  55         with pytest.raises(SystemExit) as ctx:
  56             am.fail_json(**args)
  57         assert ctx.value.code == 1
  58 
  59         out, err = capfd.readouterr()
  60         return_val = json.loads(out)
  61         # Fail_json should add failed=True
  62         expected['failed'] = True
  63         assert return_val == expected
  64 
  65     @pytest.mark.parametrize('stdin', [{}], indirect=['stdin'])
  66     def test_fail_json_msg_positional(self, am, capfd):
  67         with pytest.raises(SystemExit) as ctx:
  68             am.fail_json('This is the msg')
  69         assert ctx.value.code == 1
  70 
  71         out, err = capfd.readouterr()
  72         return_val = json.loads(out)
  73         # Fail_json should add failed=True
  74         assert return_val == {'msg': 'This is the msg', 'failed': True,
  75                               'invocation': EMPTY_INVOCATION}
  76 
  77     @pytest.mark.parametrize('stdin', [{}], indirect=['stdin'])
  78     def test_fail_json_msg_as_kwarg_after(self, am, capfd):
  79         """Test that msg as a kwarg after other kwargs works"""
  80         with pytest.raises(SystemExit) as ctx:
  81             am.fail_json(arbitrary=42, msg='This is the msg')
  82         assert ctx.value.code == 1
  83 
  84         out, err = capfd.readouterr()
  85         return_val = json.loads(out)
  86         # Fail_json should add failed=True
  87         assert return_val == {'msg': 'This is the msg', 'failed': True,
  88                               'arbitrary': 42,
  89                               'invocation': EMPTY_INVOCATION}
  90 
  91     @pytest.mark.parametrize('stdin', [{}], indirect=['stdin'])
  92     def test_fail_json_no_msg(self, am):
  93         with pytest.raises(TypeError) as ctx:
  94             am.fail_json()
  95 
  96         if sys.version_info < (3,):
  97             error_msg = "fail_json() takes exactly 2 arguments (1 given)"
  98         else:
  99             error_msg = "fail_json() missing 1 required positional argument: 'msg'"
 100 
 101         assert ctx.value.args[0] == error_msg

 104 class TestAnsibleModuleExitValuesRemoved:
 105     """
 106     Test that ExitJson and FailJson remove password-like values
 107     """
 108     OMIT = 'VALUE_SPECIFIED_IN_NO_LOG_PARAMETER'
 109 
 110     DATA = (
 111         (
 112             dict(username='person', password='$ecret k3y'),
 113             dict(one=1, pwd='$ecret k3y', url='https://username:password12345@foo.com/login/',
 114                  not_secret='following the leader', msg='here'),
 115             dict(one=1, pwd=OMIT, url='https://username:password12345@foo.com/login/',
 116                  not_secret='following the leader', msg='here',
 117                  invocation=dict(module_args=dict(password=OMIT, token=None, username='person'))),
 118         ),
 119         (
 120             dict(username='person', password='password12345'),
 121             dict(one=1, pwd='$ecret k3y', url='https://username:password12345@foo.com/login/',
 122                  not_secret='following the leader', msg='here'),
 123             dict(one=1, pwd='$ecret k3y', url='https://username:********@foo.com/login/',
 124                  not_secret='following the leader', msg='here',
 125                  invocation=dict(module_args=dict(password=OMIT, token=None, username='person'))),
 126         ),
 127         (
 128             dict(username='person', password='$ecret k3y'),
 129             dict(one=1, pwd='$ecret k3y', url='https://username:$ecret k3y@foo.com/login/',
 130                  not_secret='following the leader', msg='here'),
 131             dict(one=1, pwd=OMIT, url='https://username:********@foo.com/login/',
 132                  not_secret='following the leader', msg='here',
 133                  invocation=dict(module_args=dict(password=OMIT, token=None, username='person'))),
 134         ),
 135     )
 136 
 137     # pylint bug: https://github.com/PyCQA/pylint/issues/511
 138     @pytest.mark.parametrize('am, stdin, return_val, expected',
 139                              (({'username': {}, 'password': {'no_log': True}, 'token': {'no_log': True}}, s, r, e)
 140                               for s, r, e in DATA),  # pylint: disable=undefined-variable
 141                              indirect=['am', 'stdin'])
 142     def test_exit_json_removes_values(self, am, capfd, return_val, expected):
 143         with pytest.raises(SystemExit):
 144             am.exit_json(**return_val)
 145         out, err = capfd.readouterr()
 146 
 147         assert json.loads(out) == expected
 148 
 149     # pylint bug: https://github.com/PyCQA/pylint/issues/511
 150     @pytest.mark.parametrize('am, stdin, return_val, expected',
 151                              (({'username': {}, 'password': {'no_log': True}, 'token': {'no_log': True}}, s, r, e)
 152                               for s, r, e in DATA),  # pylint: disable=undefined-variable
 153                              indirect=['am', 'stdin'])
 154     def test_fail_json_removes_values(self, am, capfd, return_val, expected):
 155         expected['failed'] = True
 156         with pytest.raises(SystemExit):
 157             am.fail_json(**return_val) == expected
 158         out, err = capfd.readouterr()
