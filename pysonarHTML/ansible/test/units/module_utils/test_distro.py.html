  22 # Generic test case with minimal assertions about specific returned values.
  23 class TestDistro():
  24     # should run on any platform without errors, even if non-linux without any
  25     # useful info to return
  26     def test_info(self):
  27         info = distro.info()
  28         assert isinstance(info, dict), \
  29             'distro.info() returned %s (%s) which is not a dist' % (info, type(info))
  30 
  31     def test_linux_distribution(self):
  32         linux_dist = distro.linux_distribution()
  33         assert isinstance(linux_dist, tuple), \
  34             'linux_distrution() returned %s (%s) which is not a tuple' % (linux_dist, type(linux_dist))
  35 
  36     def test_id(self):
  37         id = distro.id()
  38         assert isinstance(id, string_types), 'distro.id() returned %s (%s) which is not a string' % (id, type(id))
