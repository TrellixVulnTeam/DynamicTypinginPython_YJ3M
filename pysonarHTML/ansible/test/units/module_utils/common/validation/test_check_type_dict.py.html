  13 def test_check_type_dict():
  14     test_cases = (
  15         ({'k1': 'v1'}, {'k1': 'v1'}),
  16         ('k1=v1,k2=v2', {'k1': 'v1', 'k2': 'v2'}),
  17         ('k1=v1, k2=v2', {'k1': 'v1', 'k2': 'v2'}),
  18         ('k1=v1,     k2=v2,  k3=v3', {'k1': 'v1', 'k2': 'v2', 'k3': 'v3'}),
  19         ('{"key": "value", "list": ["one", "two"]}', {'key': 'value', 'list': ['one', 'two']})
  20     )
  21     for case in test_cases:
  22         assert case[1] == check_type_dict(case[0])
