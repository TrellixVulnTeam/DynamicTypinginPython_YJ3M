  14 def test_check_type_bool():
  15     test_cases = (
  16         (True, True),
  17         (False, False),
  18         ('1', True),
  19         ('on', True),
  20         (1, True),
  21         ('0', False),
  22         (0, False),
  23         ('n', False),
  24         ('f', False),
  25         ('false', False),
  26         ('true', True),
  27         ('y', True),
  28         ('t', True),
  29         ('yes', True),
  30         ('no', False),
  31         ('off', False),
  32     )
  33     for case in test_cases:
  34         assert case[1] == check_type_bool(case[0])

  37 def test_check_type_bool_fail():
  38     default_test_msg = 'cannot be converted to a bool'
  39     test_cases = (
  40         ({'k1': 'v1'}, 'is not a valid bool'),
  41         (3.14159, default_test_msg),
  42         (-1, default_test_msg),
  43         (-90810398401982340981023948192349081, default_test_msg),
  44         (90810398401982340981023948192349081, default_test_msg),
  45     )
  46     for case in test_cases:
  47         with pytest.raises(TypeError) as e:
  48             check_type_bool(case)
  49         assert 'cannot be converted to a bool' in to_native(e.value)
