  14 def test_check_type_float():
  15     test_cases = (
  16         ('1.5', 1.5),
  17         ('''1.5''', 1.5),
  18         (u'1.5', 1.5),
  19         (1002, 1002.0),
  20         (1.0, 1.0),
  21         (3.141592653589793, 3.141592653589793),
  22         ('3.141592653589793', 3.141592653589793),
  23         (b'3.141592653589793', 3.141592653589793),
  24     )
  25     for case in test_cases:
  26         assert case[1] == check_type_float(case[0])

  29 def test_check_type_float_fail():
  30     test_cases = (
  31         {'k1': 'v1'},
  32         ['a', 'b'],
  33         'b',
  34     )
  35     for case in test_cases:
  36         with pytest.raises(TypeError) as e:
  37             check_type_float(case)
  38         assert 'cannot be converted to a float' in to_native(e.value)
