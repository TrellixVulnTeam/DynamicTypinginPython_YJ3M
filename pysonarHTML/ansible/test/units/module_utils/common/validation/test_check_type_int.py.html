  14 def test_check_type_int():
  15     test_cases = (
  16         ('1', 1),
  17         (u'1', 1),
  18         (1002, 1002),
  19     )
  20     for case in test_cases:
  21         assert case[1] == check_type_int(case[0])

  24 def test_check_type_int_fail():

  28         (3.14159, 3),
  29         'b',
  30     )
