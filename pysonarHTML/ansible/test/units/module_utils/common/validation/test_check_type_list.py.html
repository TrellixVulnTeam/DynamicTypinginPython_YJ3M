  13 def test_check_type_list():
  14     test_cases = (
  15         ([1, 2], [1, 2]),
  16         (1, ['1']),
  17         (['a', 'b'], ['a', 'b']),
  18         ('a', ['a']),
  19         (3.14159, ['3.14159']),
  20         ('a,b,1,2', ['a', 'b', '1', '2'])

  22     for case in test_cases:
  23         assert case[1] == check_type_list(case[0])
