  14 def test_check_type_jsonarg():
  15     test_cases = (
  16         ('a', 'a'),
  17         ('a  ', 'a'),
  18         (b'99', b'99'),
  19         (b'99  ', b'99'),
  20         ({'k1': 'v1'}, '{"k1": "v1"}'),
  21         ([1, 'a'], '[1, "a"]'),
  22         ((1, 2, 'three'), '[1, 2, "three"]'),
  23     )
  24     for case in test_cases:
  25         assert case[1] == check_type_jsonarg(case[0])

  28 def test_check_type_jsonarg_fail():

  30         1.5,
  31         910313498012384012341982374109384098,
  32     )

  34         with pytest.raises(TypeError) as e:
  35             check_type_jsonarg(case)
  36         assert 'cannot be converted to a json string' in to_native(e.value)
