  14 def test_check_type_bits():
  15     test_cases = (
  16         ('1', 1),
  17         (99, 99),
  18         (1.5, 2),
  19         ('1.5', 2),
  20         ('2b', 2),
  21         ('2k', 2048),
  22         ('2K', 2048),
  23         ('1m', 1048576),
  24         ('1M', 1048576),
  25         ('1g', 1073741824),
  26         ('1G', 1073741824),
  27         (1073741824, 1073741824),
  28     )
  29     for case in test_cases:
  30         assert case[1] == check_type_bits(case[0])

  33 def test_check_type_bits_fail():

  35         'foo',
  36         '2KB',
  37         '1MB',
  38         '1GB',
  39     )
