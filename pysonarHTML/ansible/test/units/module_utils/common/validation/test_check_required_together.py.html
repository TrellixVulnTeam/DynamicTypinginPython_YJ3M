  10 from ansible.module_utils._text import to_native

  14 @pytest.fixture
  15 def together_terms():
  16     return [
  17         ['bananas', 'potatoes'],
  18         ['cats', 'wolves']

  22 def test_check_required_together(together_terms):
  23     params = {
  24         'bananas': 'hello',
  25         'potatoes': 'this is here too',
  26         'dogs': 'haha',
  27     }
  28     assert check_required_together(together_terms, params) == []
  29 
  30 
  31 def test_check_required_together_missing(together_terms):
  32     params = {
  33         'bananas': 'woohoo',
  34         'wolves': 'uh oh',
  35     }
  36     expected = "parameters are required together: bananas, potatoes"
  37 
  38     with pytest.raises(TypeError) as e:
  39         check_required_together(together_terms, params)
  40 
  41     assert to_native(e.value) == expected
  42 
  43 
  44 def test_check_required_together_missing_none():
  45     terms = None
  46     params = {
  47         'foo': 'bar',
  48         'baz': 'buzz',
  49     }
  50     assert check_required_together(terms, params) == []
  51 
  52 
  53 def test_check_required_together_no_params(together_terms):
  54     with pytest.raises(TypeError) as te:
  55         check_required_together(together_terms, None)
  56 
  57     assert "'NoneType' object is not iterable" in to_native(te.value)
