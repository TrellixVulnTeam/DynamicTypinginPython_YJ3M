   8 import pytest

  10 from ansible.module_utils._text import to_native
  11 from ansible.module_utils.common.validation import check_mutually_exclusive
  12 
  13 
  14 @pytest.fixture
  15 def mutually_exclusive_terms():
  16     return [
  17         ('string1', 'string2',),
  18         ('box', 'fox', 'socks'),
  19     ]
  20 
  21 
  22 def test_check_mutually_exclusive(mutually_exclusive_terms):
  23     params = {
  24         'string1': 'cat',
  25         'fox': 'hat',
  26     }
  27     assert check_mutually_exclusive(mutually_exclusive_terms, params) == []
  28 
  29 
  30 def test_check_mutually_exclusive_found(mutually_exclusive_terms):
  31     params = {
  32         'string1': 'cat',
  33         'string2': 'hat',
  34         'fox': 'red',
  35         'socks': 'blue',
  36     }
  37     expected = "parameters are mutually exclusive: string1|string2, box|fox|socks"
  38 
  39     with pytest.raises(TypeError) as e:
  40         check_mutually_exclusive(mutually_exclusive_terms, params)
  41 
  42     assert to_native(e.value) == expected
  43 
  44 
  45 def test_check_mutually_exclusive_none():
  46     terms = None
  47     params = {
  48         'string1': 'cat',
  49         'fox': 'hat',
  50     }
  51     assert check_mutually_exclusive(terms, params) == []
  52 
  53 
  54 def test_check_mutually_exclusive_no_params(mutually_exclusive_terms):
  55     with pytest.raises(TypeError) as te:
  56         check_mutually_exclusive(mutually_exclusive_terms, None)
  57     assert "'NoneType' object is not iterable" in to_native(te.value)
