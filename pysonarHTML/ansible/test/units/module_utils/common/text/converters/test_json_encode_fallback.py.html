  17 @pytest.mark.parametrize(
  18     'test_input,expected',
  19     [
  20         (set([1]), [1]),
  21         (datetime(2019, 5, 14, 13, 39, 38, 569047), '2019-05-14T13:39:38.569047'),
  22         (datetime(2019, 5, 14, 13, 47, 16, 923866), '2019-05-14T13:47:16.923866'),
  23         (datetime(2019, 6, 15, 14, 45, tzinfo=tz('UTC')), '2019-06-15T14:45:00+00:00'),
  24         (datetime(2019, 6, 15, 14, 45, tzinfo=tz('Europe/Helsinki')), '2019-06-15T14:45:00+01:40'),
  25     ]
  26 )
  27 def test_json_encode_fallback(test_input, expected):
  28     """
  29     Test for passing expected objects to _json_encode_fallback().
