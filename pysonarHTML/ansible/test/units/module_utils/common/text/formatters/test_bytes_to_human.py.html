 109 @pytest.mark.parametrize('input_data', [0j, u'1B', [1], {1: 1}, None, b'1B'])
 110 def test_bytes_to_human_illegal_size(input_data):
 111     """Test of bytes_to_human function, illegal objects are passed as a size."""
 112     e_regexp = (r'(no ordering relation is defined for complex numbers)|'
 113                 r'(unsupported operand type\(s\) for /)|(unorderable types)|'
 114                 r'(not supported between instances of)')
 115     with pytest.raises(TypeError, match=e_regexp):
 116         bytes_to_human(input_data)
