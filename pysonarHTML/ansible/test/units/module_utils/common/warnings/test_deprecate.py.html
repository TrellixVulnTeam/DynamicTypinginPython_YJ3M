  29 def test_deprecate_message_only():

  65 def test_multiple_deprecations(deprecation_messages):
  66     for d in deprecation_messages:
  67         deprecate(**d)
  68 
  69     assert deprecation_messages == warnings._global_deprecations
  70 
  71 
  72 def test_get_deprecation_messages(deprecation_messages):
  73     for d in deprecation_messages:
  74         deprecate(**d)
  75 
  76     accessor_deprecations = get_deprecation_messages()
  77     assert isinstance(accessor_deprecations, tuple)
  78     assert len(accessor_deprecations) == 7
