  13 def test_get_bin_path(mocker):
  14     path = '/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin'
  15     mocker.patch.dict('os.environ', {'PATH': path})
  16     mocker.patch('os.pathsep', ':')
  17 
  18     mocker.patch('os.path.isdir', return_value=False)
  19     mocker.patch('ansible.module_utils.common.process.is_executable', return_value=True)
  20 
  21     # pytest-mock 2.0.0 will throw when os.path.exists is messed with
  22     # and then another method is patched afterwards. Likely
  23     # something in the pytest-mock chain uses os.path.exists internally, and
  24     # since pytest-mock prohibits context-specific patching, there's not a
  25     # good solution. For now, just patch os.path.exists last.
  26     mocker.patch('os.path.exists', side_effect=[False, True])
  27 
  28     assert '/usr/local/bin/notacommand' == get_bin_path('notacommand')

  31 def test_get_path_path_raise_valueerror(mocker):
  32     mocker.patch.dict('os.environ', {'PATH': ''})
  33 
  34     mocker.patch('os.path.exists', return_value=False)
  35     mocker.patch('os.path.isdir', return_value=False)
  36     mocker.patch('ansible.module_utils.common.process.is_executable', return_value=True)
  37 
  38     with pytest.raises(ValueError, match='Failed to find required executable notacommand'):
  39         get_bin_path('notacommand')
