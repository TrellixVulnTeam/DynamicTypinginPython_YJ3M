  33 def test_handle_aliases_no_aliases():
  34     argument_spec = {
  35         'name': {'type': 'str'},
  36     }
  37 
  38     params = {
  39         'name': 'foo',
  40         'path': 'bar'
  41     }
  42 
  43     expected = (
  44         {},
  45         DEFAULT_LEGAL_INPUTS + ['name'],
  46     )
  47     expected[1].sort()
  48 
  49     result = handle_aliases(argument_spec, params)
  50     result[1].sort()
  51     assert expected == result

  54 def test_handle_aliases_basic():
  55     argument_spec = {
  56         'name': {'type': 'str', 'aliases': ['surname', 'nick']},
  57     }
  58 
  59     params = {
  60         'name': 'foo',
  61         'path': 'bar',
  62         'surname': 'foo',
  63         'nick': 'foo',
  64     }
  65 
  66     expected = (
  67         {'surname': 'name', 'nick': 'name'},
  68         DEFAULT_LEGAL_INPUTS + ['name', 'surname', 'nick'],
  69     )
  70     expected[1].sort()
  71 
  72     result = handle_aliases(argument_spec, params)
  73     result[1].sort()
  74     assert expected == result

  77 def test_handle_aliases_value_error():
  78     argument_spec = {
  79         'name': {'type': 'str', 'aliases': ['surname', 'nick'], 'default': 'bob', 'required': True},
  80     }
  81 
  82     params = {
  83         'name': 'foo',
  84     }
  85 
  86     with pytest.raises(ValueError) as ve:
  87         handle_aliases(argument_spec, params)
  88         assert 'internal error: aliases must be a list or tuple' == to_native(ve.error)
  89
