  23 def test_list_deprecations():
  24     argument_spec = {
  25         'old': {'type': 'str', 'removed_in_version': '2.5'},
  26         'foo': {'type': 'dict', 'options': {'old': {'type': 'str', 'removed_in_version': 1.0}}},
  27         'bar': {'type': 'list', 'elements': 'dict', 'options': {'old': {'type': 'str', 'removed_in_version': '2.10'}}},
  28     }
  29 
  30     params = {
  31         'name': 'rod',
  32         'old': 'option',
  33         'foo': {'old': 'value'},
  34         'bar': [{'old': 'value'}, {}],
  35     }
  36     result = list_deprecations(argument_spec, params)
  37     assert len(result) == 3
  38     result.sort(key=lambda entry: entry['msg'])
  39     assert result[0]['msg'] == """Param 'bar["old"]' is deprecated. See the module docs for more information"""
  40     assert result[0]['version'] == '2.10'
  41     assert result[1]['msg'] == """Param 'foo["old"]' is deprecated. See the module docs for more information"""
  42     assert result[1]['version'] == 1.0
  43     assert result[2]['msg'] == "Param 'old' is deprecated. See the module docs for more information"
  44     assert result[2]['version'] == '2.5'
