  69 class CamelDictToSnakeDictTestCase(unittest.TestCase):
  70     def test_ignore_list(self):
  71         camel_dict = dict(Hello=dict(One='one', Two='two'), World=dict(Three='three', Four='four'))
  72         snake_dict = camel_dict_to_snake_dict(camel_dict, ignore_list='World')

  74         self.assertEqual(snake_dict['world'], dict(Three='three', Four='four'))

  77 class DictMergeTestCase(unittest.TestCase):
  78     def test_dict_merge(self):
  79         base = dict(obj2=dict(), b1=True, b2=False, b3=False,
  80                     one=1, two=2, three=3, obj1=dict(key1=1, key2=2),
  81                     l1=[1, 3], l2=[1, 2, 3], l4=[4],
  82                     nested=dict(n1=dict(n2=2)))
  83 
  84         other = dict(b1=True, b2=False, b3=True, b4=True,
  85                      one=1, three=4, four=4, obj1=dict(key1=2),
  86                      l1=[2, 1], l2=[3, 2, 1], l3=[1],
  87                      nested=dict(n1=dict(n2=2, n3=3)))
  88 
  89         result = dict_merge(base, other)
  90 
  91         # string assertions
  92         self.assertTrue('one' in result)
  93         self.assertTrue('two' in result)
  94         self.assertEqual(result['three'], 4)
  95         self.assertEqual(result['four'], 4)
  96 
  97         # dict assertions
  98         self.assertTrue('obj1' in result)
  99         self.assertTrue('key1' in result['obj1'])
 100         self.assertTrue('key2' in result['obj1'])
 101 
 102         # list assertions
 103         # this line differs from the network_utils/common test of the function of the
 104         # same name as this method does not merge lists
 105         self.assertEqual(result['l1'], [2, 1])
 106         self.assertTrue('l2' in result)
 107         self.assertEqual(result['l3'], [1])
 108         self.assertTrue('l4' in result)
 109 
 110         # nested assertions
 111         self.assertTrue('obj1' in result)
 112         self.assertEqual(result['obj1']['key1'], 2)
 113         self.assertTrue('key2' in result['obj1'])
 114 
 115         # bool assertions
 116         self.assertTrue('b1' in result)
 117         self.assertTrue('b2' in result)
 118         self.assertTrue(result['b3'])
 119         self.assertTrue(result['b4'])

 122 class AzureIncidentalTestCase(unittest.TestCase):
 123 
 124     def test_dict_merge_invalid_dict(self):
 125         ''' if b is not a dict, return b '''
 126         res = dict_merge({}, None)
 127         self.assertEqual(res, None)

 129     def test_merge_sub_dicts(self):
 130         '''merge sub dicts '''
 131         a = {'a': {'a1': 1}}
 132         b = {'a': {'b1': 2}}
 133         c = {'a': {'a1': 1, 'b1': 2}}
 134         res = dict_merge(a, b)
 135         self.assertEqual(res, c)
