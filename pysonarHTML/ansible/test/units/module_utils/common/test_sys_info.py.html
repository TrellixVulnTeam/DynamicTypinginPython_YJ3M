 122 class TestGetPlatformSubclass:
 123     class LinuxTest:
 124         pass
 125 
 126     class Foo(LinuxTest):
 127         platform = "Linux"
 128         distribution = None
 129 
 130     class Bar(LinuxTest):
 131         platform = "Linux"
 132         distribution = "Bar"
 133 
 134     def test_not_linux(self):
 135         # if neither match, the fallback should be the top-level class
 136         with patch('platform.system', return_value="Foo"):
 137             with patch('ansible.module_utils.common.sys_info.get_distribution', return_value=None):
 138                 assert get_platform_subclass(self.LinuxTest) is self.LinuxTest
 139 
 140     @pytest.mark.usefixtures("platform_linux")
 141     def test_get_distribution_none(self):
 142         # match just the platform class, not a specific distribution
 143         with patch('ansible.module_utils.common.sys_info.get_distribution', return_value=None):
 144             assert get_platform_subclass(self.LinuxTest) is self.Foo
 145 
 146     @pytest.mark.usefixtures("platform_linux")
 147     def test_get_distribution_found(self):
 148         # match both the distribution and platform class
 149         with patch('ansible.module_utils.common.sys_info.get_distribution', return_value="Bar"):
 150             assert get_platform_subclass(self.LinuxTest) is self.Bar
