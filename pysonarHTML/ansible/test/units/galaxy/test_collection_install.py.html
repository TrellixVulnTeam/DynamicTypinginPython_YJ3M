  42 def artifact_json(namespace, name, version, dependencies, server):
  43     json_str = json.dumps({
  44         'artifact': {
  45             'filename': '%s-%s-%s.tar.gz' % (namespace, name, version),
  46             'sha256': '2d76f3b8c4bab1072848107fb3914c345f71a12a1722f25c08f5d3f51f4ab5fd',
  47             'size': 1234,
  48         },
  49         'download_url': '%s/download/%s-%s-%s.tar.gz' % (server, namespace, name, version),
  50         'metadata': {
  51             'namespace': namespace,
  52             'name': name,
  53             'dependencies': dependencies,
  54         },
  55         'version': version
  56     })
  57     return to_text(json_str)

  60 def artifact_versions_json(namespace, name, versions, galaxy_api, available_api_versions=None):
  61     results = []
  62     available_api_versions = available_api_versions or {}
  63     api_version = 'v2'
  64     if 'v3' in available_api_versions:
  65         api_version = 'v3'
  66     for version in versions:
  67         results.append({
  68             'href': '%s/api/%s/%s/%s/versions/%s/' % (galaxy_api.api_server, api_version, namespace, name, version),
  69             'version': version,
  70         })
  71 
  72     if api_version == 'v2':
  73         json_str = json.dumps({
  74             'count': len(versions),
  75             'next': None,
  76             'previous': None,
  77             'results': results
  78         })
  79 
  80     if api_version == 'v3':
  81         response = {'meta': {'count': len(versions)},
  82                     'data': results,
  83                     'links': {'first': None,
  84                               'last': None,
  85                               'next': None,
  86                               'previous': None},
  87                     }
  88         json_str = json.dumps(response)
  89     return to_text(json_str)

  92 def error_json(galaxy_api, errors_to_return=None, available_api_versions=None):
  93     errors_to_return = errors_to_return or []
  94     available_api_versions = available_api_versions or {}
  95 
  96     response = {}
  97 
  98     api_version = 'v2'
  99     if 'v3' in available_api_versions:
 100         api_version = 'v3'
 101 
 102     if api_version == 'v2':
 103         assert len(errors_to_return) <= 1
 104         if errors_to_return:
 105             response = errors_to_return[0]
 106 
 107     if api_version == 'v3':
 108         response['errors'] = errors_to_return
 109 
 110     json_str = json.dumps(response)
 111     return to_text(json_str)
