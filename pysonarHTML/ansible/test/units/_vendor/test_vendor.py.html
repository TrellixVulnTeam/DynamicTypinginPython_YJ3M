  15 def reset_internal_vendor_package():
  16     import ansible
  17     ansible_vendor_path = os.path.join(os.path.dirname(ansible.__file__), '_vendor')
  18 
  19     if ansible_vendor_path in sys.path:
  20         sys.path.remove(ansible_vendor_path)
  21 
  22     for pkg in ['ansible._vendor', 'ansible']:
  23         if pkg in sys.modules:
  24             del sys.modules[pkg]

  44 def test_vendored(vendored_pkg_names=None):
  45     if not vendored_pkg_names:
  46         vendored_pkg_names = ['boguspkg']
  47     reset_internal_vendor_package()
  48     with patch.object(pkgutil, 'iter_modules', return_value=list((None, p, None) for p in vendored_pkg_names)):
  49         previous_path = list(sys.path)
  50         import ansible
  51         ansible_vendor_path = os.path.join(os.path.dirname(ansible.__file__), '_vendor')
  52         assert sys.path[0] == ansible_vendor_path
  53 
  54         if ansible_vendor_path in previous_path:
  55             previous_path.remove(ansible_vendor_path)
  56 
  57         assert sys.path[1:] == previous_path
