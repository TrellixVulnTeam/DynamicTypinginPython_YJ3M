  89     def _assert_is_task_list(self, results):
  90         for result in results:
  91             self.assertIsInstance(result, Task)

  93     def _assert_is_task_list_or_blocks(self, results):
  94         self.assertIsInstance(results, list)
  95         for result in results:
  96             self.assertIsInstance(result, (Task, Block))
