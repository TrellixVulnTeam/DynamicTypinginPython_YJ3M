  22 from units.compat import unittest
  23 from ansible.errors import AnsibleParserError
  24 from ansible.playbook import Playbook
  25 from ansible.vars.manager import VariableManager
  26 
  27 from units.mock.loader import DictDataLoader
  28 
  29 
  30 class TestPlaybook(unittest.TestCase):
  31 
  32     def test_empty_playbook(self):
  33         fake_loader = DictDataLoader({})
  34         p = Playbook(loader=fake_loader)
  35 
  36     def test_basic_playbook(self):
  37         fake_loader = DictDataLoader({
  38             "test_file.yml": """
  39             - hosts: all
  40             """,
  41         })
  42         p = Playbook.load("test_file.yml", loader=fake_loader)
  43         plays = p.get_plays()
  44 
  45     def test_bad_playbook_files(self):
  46         fake_loader = DictDataLoader({
  47             # represents a playbook which is not a list of plays
  48             "bad_list.yml": """
  49             foo: bar
  50 
  51             """,
  52             # represents a playbook where a play entry is mis-formatted
  53             "bad_entry.yml": """
  54             -
  55               - "This should be a mapping..."
  56 
  57             """,
  58         })
  59         vm = VariableManager()
  60         self.assertRaises(AnsibleParserError, Playbook.load, "bad_list.yml", vm, fake_loader)
  61         self.assertRaises(AnsibleParserError, Playbook.load, "bad_entry.yml", vm, fake_loader)
