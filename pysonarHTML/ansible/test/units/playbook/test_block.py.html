  27 class TestBlock(unittest.TestCase):
  28 
  29     def test_construct_empty_block(self):

  32     def test_construct_block_with_role(self):
  33         pass

  35     def test_load_block_simple(self):
  36         ds = dict(
  37             block=[],
  38             rescue=[],
  39             always=[],
  40             # otherwise=[],
  41         )
  42         b = Block.load(ds)
  43         self.assertEqual(b.block, [])
  44         self.assertEqual(b.rescue, [])
  45         self.assertEqual(b.always, [])
  46         # not currently used
  47         # self.assertEqual(b.otherwise, [])
  48 
  49     def test_load_block_with_tasks(self):
  50         ds = dict(
  51             block=[dict(action='block')],
  52             rescue=[dict(action='rescue')],
  53             always=[dict(action='always')],
  54             # otherwise=[dict(action='otherwise')],
  55         )
  56         b = Block.load(ds)
  57         self.assertEqual(len(b.block), 1)
  58         self.assertIsInstance(b.block[0], Task)
  59         self.assertEqual(len(b.rescue), 1)
  60         self.assertIsInstance(b.rescue[0], Task)
  61         self.assertEqual(len(b.always), 1)
  62         self.assertIsInstance(b.always[0], Task)
  63         # not currently used
  64         # self.assertEqual(len(b.otherwise), 1)
  65         # self.assertIsInstance(b.otherwise[0], Task)
  66 
  67     def test_load_implicit_block(self):
  68         ds = [dict(action='foo')]
  69         b = Block.load(ds)
  70         self.assertEqual(len(b.block), 1)
  71         self.assertIsInstance(b.block[0], Task)
  72 
  73     def test_deserialize(self):
  74         ds = dict(
  75             block=[dict(action='block')],
  76             rescue=[dict(action='rescue')],
  77             always=[dict(action='always')],
  78         )
  79         b = Block.load(ds)
  80         data = dict(parent=ds, parent_type='Block')
  81         b.deserialize(data)
  82         self.assertIsInstance(b._parent, Block)
