  30 def test_collection_static_warning(capsys):
  31     """Test that collection name is not templated.
  32 
  33     Also, make sure that users see the warning message for the referenced name.
  34     """
  35     collection_name = "foo.{{bar}}"
  36     p = Play.load(dict(
  37         name="test play",
  38         hosts=['foo'],
  39         gather_facts=False,
  40         connection='local',
  41         collections=collection_name,
  42     ))
  43     assert collection_name in p.collections
  44     std_out, std_err = capsys.readouterr()
  45     assert '[WARNING]: "collections" is not templatable, but we found: %s' % collection_name in std_err
  46     assert '' == std_out

  49 def test_collection_invalid_data_play():
  50     """Test that collection as a dict at the play level fails with parser error"""
  51     collection_name = {'name': 'foo'}
  52     with pytest.raises(AnsibleParserError):
  53         Play.load(dict(
  54             name="test play",
  55             hosts=['foo'],
  56             gather_facts=False,
  57             connection='local',
  58             collections=collection_name,
