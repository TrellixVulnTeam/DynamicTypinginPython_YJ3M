  63     def test_load_task_simple(self):
  64         t = Task.load(basic_command_task)
  65         assert t is not None
  66         self.assertEqual(t.name, basic_command_task['name'])

  70     def test_load_task_kv_form(self):

  75     @patch.object(errors.AnsibleError, '_get_error_lines_from_file')
  76     def test_load_task_kv_form_error_36848(self, mock_get_err_lines):
  77         ds = objects.AnsibleMapping(kv_bad_args_ds)
  78         ds.ansible_pos = ('test_task_faux_playbook.yml', 1, 1)
  79         mock_get_err_lines.return_value = (kv_bad_args_str, '')
  80 
  81         with self.assertRaises(errors.AnsibleParserError) as cm:
  82             Task.load(ds)
  83 
  84         self.assertIsInstance(cm.exception, errors.AnsibleParserError)
  85         self.assertEqual(cm.exception._obj, ds)
  86         self.assertEqual(cm.exception._obj, kv_bad_args_ds)
  87         self.assertIn("The error appears to be in 'test_task_faux_playbook.yml", cm.exception.message)
  88         self.assertIn(kv_bad_args_str, cm.exception.message)
  89         self.assertIn('apk', cm.exception.message)
  90         self.assertEqual(cm.exception.message.count('The offending line'), 1)
  91         self.assertEqual(cm.exception.message.count('The error appears to be in'), 1)
