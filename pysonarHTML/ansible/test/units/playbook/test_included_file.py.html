 108 def test_process_include_results(mock_iterator, mock_variable_manager):
 109     hostname = "testhost1"
 110     hostname2 = "testhost2"
 111 
 112     parent_task_ds = {'debug': 'msg=foo'}
 113     parent_task = Task.load(parent_task_ds)
 114     parent_task._play = None
 115 
 116     task_ds = {'include': 'include_test.yml'}
 117     loaded_task = TaskInclude.load(task_ds, task_include=parent_task)
 118 
 119     return_data = {'include': 'include_test.yml'}
 120     # The task in the TaskResult has to be a TaskInclude so it has a .static attr
 121     result1 = task_result.TaskResult(host=hostname, task=loaded_task, return_data=return_data)
 122     result2 = task_result.TaskResult(host=hostname2, task=loaded_task, return_data=return_data)
 123     results = [result1, result2]
 124 
 125     fake_loader = DictDataLoader({'include_test.yml': ""})
 126 
 127     res = IncludedFile.process_include_results(results, mock_iterator, fake_loader, mock_variable_manager)
 128     assert isinstance(res, list)
 129     assert len(res) == 1
 130     assert res[0]._filename == os.path.join(os.getcwd(), 'include_test.yml')
 131     assert res[0]._hosts == ['testhost1', 'testhost2']
 132     assert res[0]._args == {}
 133     assert res[0]._vars == {}

 217 def test_process_include_simulate_free_block_role_tasks(mock_iterator,
 218                                                         mock_variable_manager):
 219     """Test loading the same role returns different included files
 220 
 221     In the case of free, we may end up with included files from roles that
 222     have the same parent but are different tasks. Previously the comparison
 223     for equality did not check if the tasks were the same and only checked
 224     that the parents were the same. This lead to some tasks being run
 225     incorrectly and some tasks being silient dropped."""
 226 
 227     fake_loader = DictDataLoader({
 228         'include_test.yml': "",
 229         '/etc/ansible/roles/foo_role/tasks/task1.yml': """
 230             - debug: msg=task1
 231         """,
 232         '/etc/ansible/roles/foo_role/tasks/task2.yml': """
 233             - debug: msg=task2
 234         """,
 235     })
 236 
 237     hostname = "testhost1"
 238     hostname2 = "testhost2"
 239 
 240     role1_ds = {
 241         'name': 'task1 include',
 242         'include_role': {
 243             'name': 'foo_role',
 244             'tasks_from': 'task1.yml'
 245         }
 246     }
 247     role2_ds = {
 248         'name': 'task2 include',
 249         'include_role': {
 250             'name': 'foo_role',
 251             'tasks_from': 'task2.yml'
 252         }
 253     }
 254     parent_task_ds = {
 255         'block': [
 256             role1_ds,
 257             role2_ds
 258         ]
 259     }
 260     parent_block = Block.load(parent_task_ds, loader=fake_loader)
 261 
 262     parent_block._play = None
 263 
 264     include_role1_ds = {
 265         'include_args': {
 266             'name': 'foo_role',
 267             'tasks_from': 'task1.yml'
 268         }
 269     }
 270     include_role2_ds = {
 271         'include_args': {
 272             'name': 'foo_role',
 273             'tasks_from': 'task2.yml'
 274         }
 275     }
 276 
 277     include_role1 = IncludeRole.load(role1_ds,
 278                                      block=parent_block,
 279                                      loader=fake_loader)
 280     include_role2 = IncludeRole.load(role2_ds,
 281                                      block=parent_block,
 282                                      loader=fake_loader)
 283 
 284     result1 = task_result.TaskResult(host=hostname,
 285                                      task=include_role1,
 286                                      return_data=include_role1_ds)
 287     result2 = task_result.TaskResult(host=hostname2,
 288                                      task=include_role2,
 289                                      return_data=include_role2_ds)
 290     results = [result1, result2]
 291 
 292     res = IncludedFile.process_include_results(results,
 293                                                mock_iterator,
 294                                                fake_loader,
 295                                                mock_variable_manager)
 296     assert isinstance(res, list)
 297     # we should get two different includes
 298     assert len(res) == 2
 299     assert res[0]._filename == 'foo_role'
 300     assert res[1]._filename == 'foo_role'
 301     # with different tasks
 302     assert res[0]._task != res[1]._task
 303 
 304     assert res[0]._hosts == ['testhost1']
 305     assert res[1]._hosts == ['testhost2']
 306 
 307     assert res[0]._args == {}
 308     assert res[1]._args == {}
 309 
 310     assert res[0]._vars == {}
 311     assert res[1]._vars == {}

 314 def test_empty_raw_params():
 315     parent_task_ds = {'debug': 'msg=foo'}
 316     parent_task = Task.load(parent_task_ds)
 317     parent_task._play = None
 318 
 319     task_ds_list = [
 320         {
 321             'include': ''
 322         },
 323         {
 324             'include_tasks': ''
 325         },
 326         {
 327             'import_tasks': ''
 328         }
 329     ]
 330     for task_ds in task_ds_list:
 331         with pytest.raises(AnsibleParserError):
 332             TaskInclude.load(task_ds, task_include=parent_task)
