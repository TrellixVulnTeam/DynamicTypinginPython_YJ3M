 383     def _get_parent_attribute(self, attr, extend=False, prepend=False):
 384         value = None
 385         try:
 386             value = self._attributes[attr]
 387             if self._parent and (value is None or extend):
 388                 parent_value = getattr(self._parent, attr, None)
 389                 if extend:
 390                     value = self._extend_value(value, parent_value, prepend)
 391                 else:
 392                     value = parent_value
 393         except KeyError:
 394             pass
 395 
 396         return value
