  18 from __future__ import (absolute_import, division, print_function)
  19 __metaclass__ = type
  20 
  21 from units.compat import unittest
  22 from ansible.playbook.attribute import Attribute
  23 
  24 
  25 class TestAttribute(unittest.TestCase):
  26 
  27     def setUp(self):
  28         self.one = Attribute(priority=100)
  29         self.two = Attribute(priority=0)
  30 
  31     def test_eq(self):
  32         self.assertTrue(self.one == self.one)
  33         self.assertFalse(self.one == self.two)
  34 
  35     def test_ne(self):
  36         self.assertFalse(self.one != self.one)
  37         self.assertTrue(self.one != self.two)
  38 
  39     def test_lt(self):
  40         self.assertFalse(self.one < self.one)
  41         self.assertTrue(self.one < self.two)
  42         self.assertFalse(self.two < self.one)
  43 
  44     def test_gt(self):
  45         self.assertFalse(self.one > self.one)
  46         self.assertFalse(self.one > self.two)
  47         self.assertTrue(self.two > self.one)
  48 
  49     def test_le(self):
  50         self.assertTrue(self.one <= self.one)
  51         self.assertTrue(self.one <= self.two)
  52         self.assertFalse(self.two <= self.one)
  53 
  54     def test_ge(self):
  55         self.assertTrue(self.one >= self.one)
  56         self.assertFalse(self.one >= self.two)
  57         self.assertTrue(self.two >= self.one)
