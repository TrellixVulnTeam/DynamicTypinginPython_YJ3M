  15     def setUp(self):
  16         self.loader = DictDataLoader({})

  20     def _eval_con(self, when=None, variables=None):
  21         when = when or []
  22         variables = variables or {}
  23         self.cond.when = when
  24         ret = self.cond.evaluate_conditional(self.templar, variables)
  25         return ret
