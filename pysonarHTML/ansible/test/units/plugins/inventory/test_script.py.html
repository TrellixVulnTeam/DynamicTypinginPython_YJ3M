  34 class TestInventoryModule(unittest.TestCase):
  35 
  36     def setUp(self):
  37 
  38         class Inventory():
  39             cache = dict()
  40 
  41         class PopenResult():
  42             returncode = 0
  43             stdout = b""
  44             stderr = b""
  45 
  46             def communicate(self):
  47                 return (self.stdout, self.stderr)
  48 
  49         self.popen_result = PopenResult()
  50         self.inventory = Inventory()
  51         self.loader = mock.MagicMock()
  52         self.loader.load = mock.MagicMock()
  53 
  54         inv_loader = PluginLoader('InventoryModule', 'ansible.plugins.inventory', C.DEFAULT_INVENTORY_PLUGIN_PATH, 'inventory_plugins')
  55         self.inventory_module = inv_loader.get('script')
  56         self.inventory_module.set_options()
  57 
  58         def register_patch(name):
  59             patcher = mock.patch(name)
  60             self.addCleanup(patcher.stop)
  61             return patcher.start()
  62 
  63         self.popen = register_patch('subprocess.Popen')
  64         self.popen.return_value = self.popen_result
  65 
  66         self.BaseInventoryPlugin = register_patch('ansible.plugins.inventory.BaseInventoryPlugin')
  67         self.BaseInventoryPlugin.get_cache_prefix.return_value = 'abc123'
  68 
  69     def test_parse_subprocess_path_not_found_fail(self):
  70         self.popen.side_effect = OSError("dummy text")
  71 
  72         with pytest.raises(AnsibleError) as e:
  73             self.inventory_module.parse(self.inventory, self.loader, '/foo/bar/foobar.py')
  74         assert e.value.message == "problem running /foo/bar/foobar.py --list (dummy text)"
  75 
  76     def test_parse_subprocess_err_code_fail(self):
  77         self.popen_result.stdout = to_bytes(u"fooébar", errors='surrogate_escape')
  78         self.popen_result.stderr = to_bytes(u"dummyédata")
  79 
  80         self.popen_result.returncode = 1
  81 
  82         with pytest.raises(AnsibleError) as e:
  83             self.inventory_module.parse(self.inventory, self.loader, '/foo/bar/foobar.py')
  84         assert e.value.message == to_native("Inventory script (/foo/bar/foobar.py) had an execution error: "
  85                                             "dummyédata\n ")
  86 
  87     def test_parse_utf8_fail(self):
  88         self.popen_result.returncode = 0
  89         self.popen_result.stderr = to_bytes("dummyédata")
  90         self.loader.load.side_effect = TypeError('obj must be string')
  91 
  92         with pytest.raises(AnsibleError) as e:
  93             self.inventory_module.parse(self.inventory, self.loader, '/foo/bar/foobar.py')
  94         assert e.value.message == to_native("failed to parse executable inventory script results from "
  95                                             "/foo/bar/foobar.py: obj must be string\ndummyédata\n")
  96 
  97     def test_parse_dict_fail(self):
  98         self.popen_result.returncode = 0
  99         self.popen_result.stderr = to_bytes("dummyédata")
 100         self.loader.load.return_value = 'i am not a dict'
 101 
 102         with pytest.raises(AnsibleError) as e:
 103             self.inventory_module.parse(self.inventory, self.loader, '/foo/bar/foobar.py')
 104         assert e.value.message == to_native("failed to parse executable inventory script results from "
 105                                             "/foo/bar/foobar.py: needs to be a json dict\ndummyédata\n")
