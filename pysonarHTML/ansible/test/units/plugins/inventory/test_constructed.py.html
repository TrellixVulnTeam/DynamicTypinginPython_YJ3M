  39 def test_group_by_value_only(inventory_module):
  40     inventory_module.inventory.add_host('foohost')
  41     inventory_module.inventory.set_variable('foohost', 'bar', 'my_group_name')
  42     host = inventory_module.inventory.get_host('foohost')
  43     keyed_groups = [
  44         {
  45             'prefix': '',
  46             'separator': '',
  47             'key': 'bar'
  48         }
  49     ]
  50     inventory_module._add_host_to_keyed_groups(
  51         keyed_groups, host.vars, host.name, strict=False
  52     )
  53     assert 'my_group_name' in inventory_module.inventory.groups
  54     group = inventory_module.inventory.groups['my_group_name']
  55     assert group.hosts == [host]

  58 def test_keyed_group_separator(inventory_module):
  59     inventory_module.inventory.add_host('farm')
  60     inventory_module.inventory.set_variable('farm', 'farmer', 'mcdonald')
  61     inventory_module.inventory.set_variable('farm', 'barn', {'cow': 'betsy'})
  62     host = inventory_module.inventory.get_host('farm')
  63     keyed_groups = [
  64         {
  65             'prefix': 'farmer',
  66             'separator': '_old_',
  67             'key': 'farmer'
  68         },
  69         {
  70             'separator': 'mmmmmmmmmm',
  71             'key': 'barn'
  72         }
  73     ]
  74     inventory_module._add_host_to_keyed_groups(
  75         keyed_groups, host.vars, host.name, strict=False
  76     )
  77     for group_name in ('farmer_old_mcdonald', 'mmmmmmmmmmcowmmmmmmmmmmbetsy'):
  78         assert group_name in inventory_module.inventory.groups
  79         group = inventory_module.inventory.groups[group_name]
  80         assert group.hosts == [host]

 119 def test_keyed_parent_groups(inventory_module):
 120     inventory_module.inventory.add_host('web1')
 121     inventory_module.inventory.add_host('web2')
 122     inventory_module.inventory.set_variable('web1', 'region', 'japan')
 123     inventory_module.inventory.set_variable('web2', 'region', 'japan')
 124     host1 = inventory_module.inventory.get_host('web1')
 125     host2 = inventory_module.inventory.get_host('web2')
 126     keyed_groups = [
 127         {
 128             'prefix': 'region',
 129             'key': 'region',
 130             'parent_group': 'region_list'
 131         }
 132     ]
 133     for host in [host1, host2]:
 134         inventory_module._add_host_to_keyed_groups(
 135             keyed_groups, host.vars, host.name, strict=False
 136         )
 137     assert 'region_japan' in inventory_module.inventory.groups
 138     assert 'region_list' in inventory_module.inventory.groups
 139     region_group = inventory_module.inventory.groups['region_japan']
 140     all_regions = inventory_module.inventory.groups['region_list']
 141     assert all_regions.child_groups == [region_group]
 142     assert region_group.hosts == [host1, host2]

 145 def test_parent_group_templating(inventory_module):
 146     inventory_module.inventory.add_host('cow')
 147     inventory_module.inventory.set_variable('cow', 'sound', 'mmmmmmmmmm')
 148     inventory_module.inventory.set_variable('cow', 'nickname', 'betsy')
 149     host = inventory_module.inventory.get_host('cow')
 150     keyed_groups = [
 151         {
 152             'key': 'sound',
 153             'prefix': 'sound',
 154             'parent_group': '{{ nickname }}'
 155         },
 156         {
 157             'key': 'nickname',
 158             'prefix': '',
 159             'separator': '',
 160             'parent_group': 'nickname'  # statically-named parent group, conflicting with hostvar
 161         },
 162         {
 163             'key': 'nickname',
 164             'separator': '',
 165             'parent_group': '{{ location | default("field") }}'
 166         }
 167     ]
 168     inventory_module._add_host_to_keyed_groups(
 169         keyed_groups, host.vars, host.name, strict=True
 170     )
 171     # first keyed group, "betsy" is a parent group name dynamically generated
 172     betsys_group = inventory_module.inventory.groups['betsy']
 173     assert [child.name for child in betsys_group.child_groups] == ['sound_mmmmmmmmmm']
 174     # second keyed group, "nickname" is a statically-named root group
 175     nicknames_group = inventory_module.inventory.groups['nickname']
 176     assert [child.name for child in nicknames_group.child_groups] == ['betsy']
 177     # second keyed group actually generated the parent group of the first keyed group
 178     # assert that these are, in fact, the same object
 179     assert nicknames_group.child_groups[0] == betsys_group
 180     # second keyed group has two parents
 181     locations_group = inventory_module.inventory.groups['field']
 182     assert [child.name for child in locations_group.child_groups] == ['betsy']

 185 def test_parent_group_templating_error(inventory_module):
 186     inventory_module.inventory.add_host('cow')
 187     inventory_module.inventory.set_variable('cow', 'nickname', 'betsy')
 188     host = inventory_module.inventory.get_host('cow')
 189     keyed_groups = [
 190         {
 191             'key': 'nickname',
 192             'separator': '',
 193             'parent_group': '{{ location.barn-yard }}'
 194         }
 195     ]
 196     with pytest.raises(AnsibleParserError) as err_message:
 197         inventory_module._add_host_to_keyed_groups(
 198             keyed_groups, host.vars, host.name, strict=True
 199         )
 200         assert 'Could not generate parent group' in err_message
 201     # invalid parent group did not raise an exception with strict=False
 202     inventory_module._add_host_to_keyed_groups(
 203         keyed_groups, host.vars, host.name, strict=False
 204     )
 205     # assert group was never added with invalid parent
 206     assert 'betsy' not in inventory_module.inventory.groups
