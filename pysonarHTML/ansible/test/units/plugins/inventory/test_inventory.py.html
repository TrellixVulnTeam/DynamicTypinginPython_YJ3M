  37 class TestInventory(unittest.TestCase):
  38 
  39     patterns = {
  40         'a': ['a'],
  41         'a, b': ['a', 'b'],
  42         'a , b': ['a', 'b'],
  43         ' a,b ,c[1:2] ': ['a', 'b', 'c[1:2]'],
  44         '9a01:7f8:191:7701::9': ['9a01:7f8:191:7701::9'],
  45         '9a01:7f8:191:7701::9,9a01:7f8:191:7701::9': ['9a01:7f8:191:7701::9', '9a01:7f8:191:7701::9'],
  46         '9a01:7f8:191:7701::9,9a01:7f8:191:7701::9,foo': ['9a01:7f8:191:7701::9', '9a01:7f8:191:7701::9', 'foo'],
  47         'foo[1:2]': ['foo[1:2]'],
  48         'a::b': ['a::b'],
  49         'a:b': ['a', 'b'],
  50         ' a : b ': ['a', 'b'],
  51         'foo:bar:baz[1:2]': ['foo', 'bar', 'baz[1:2]'],
  52         'a,,b': ['a', 'b'],
  53         'a,  ,b,,c, ,': ['a', 'b', 'c'],
  54         ',': [],
  55         '': [],
  56     }
  57 
  58     pattern_lists = [
  59         [['a'], ['a']],
  60         [['a', 'b'], ['a', 'b']],
  61         [['a, b'], ['a', 'b']],
  62         [['9a01:7f8:191:7701::9', '9a01:7f8:191:7701::9,foo'],
  63          ['9a01:7f8:191:7701::9', '9a01:7f8:191:7701::9', 'foo']]
  64     ]
  65 
  66     # pattern_string: [ ('base_pattern', (a,b)), ['x','y','z'] ]
  67     # a,b are the bounds of the subscript; x..z are the results of the subscript
  68     # when applied to string.ascii_letters.
  69 
  70     subscripts = {
  71         'a': [('a', None), list(string.ascii_letters)],
  72         'a[0]': [('a', (0, None)), ['a']],
  73         'a[1]': [('a', (1, None)), ['b']],
  74         'a[2:3]': [('a', (2, 3)), ['c', 'd']],
  75         'a[-1]': [('a', (-1, None)), ['Z']],
  76         'a[-2]': [('a', (-2, None)), ['Y']],
  77         'a[48:]': [('a', (48, -1)), ['W', 'X', 'Y', 'Z']],
  78         'a[49:]': [('a', (49, -1)), ['X', 'Y', 'Z']],
  79         'a[1:]': [('a', (1, -1)), list(string.ascii_letters[1:])],
  80     }
  81 
  82     ranges_to_expand = {
  83         'a[1:2]': ['a1', 'a2'],
  84         'a[1:10:2]': ['a1', 'a3', 'a5', 'a7', 'a9'],
  85         'a[a:b]': ['aa', 'ab'],
  86         'a[a:i:3]': ['aa', 'ad', 'ag'],
  87         'a[a:b][c:d]': ['aac', 'aad', 'abc', 'abd'],
  88         'a[0:1][2:3]': ['a02', 'a03', 'a12', 'a13'],
  89         'a[a:b][2:3]': ['aa2', 'aa3', 'ab2', 'ab3'],
  90     }
  91 
  92     def setUp(self):
  93         fake_loader = DictDataLoader({})
  94 
  95         self.i = InventoryManager(loader=fake_loader, sources=[None])

  97     def test_split_patterns(self):
  98 
  99         for p in self.patterns:
 100             r = self.patterns[p]
 101             self.assertEqual(r, split_host_pattern(p))
 102 
 103         for p, r in self.pattern_lists:
 104             self.assertEqual(r, split_host_pattern(p))
 105 
 106     def test_ranges(self):
 107 
 108         for s in self.subscripts:
 109             r = self.subscripts[s]
 110             self.assertEqual(r[0], self.i._split_subscript(s))
 111             self.assertEqual(
 112                 r[1],
 113                 self.i._apply_subscript(
 114                     list(string.ascii_letters),
 115                     r[0][1]
 116                 )
 117             )

 153     def test_ini_variables_stringify(self):
 154         values = ['string', 'no', 'No', 'false', 'FALSE', [], False, 0]
 155 
 156         inventory_content = "host1 "
 157         inventory_content += ' '.join(['var%s=%s' % (i, to_text(x)) for i, x in enumerate(values)])
 158         inventory = self._get_inventory(inventory_content)
 159 
 160         variables = inventory.get_host('host1').vars
 161         for i in range(len(values)):
 162             if isinstance(values[i], string_types):
 163                 self.assertIsInstance(variables['var%s' % i], string_types)
 164             else:
 165                 self.assertIsInstance(variables['var%s' % i], type(values[i]))

 167     @mock.patch('ansible.inventory.manager.unfrackpath', mock_unfrackpath_noop)
 168     @mock.patch('os.path.exists', lambda x: True)
 169     @mock.patch('os.access', lambda x, y: True)
 170     def test_yaml_inventory(self, filename="test.yaml"):
 171         inventory_content = {filename: textwrap.dedent("""\
 172         ---
 173         all:
 174             hosts:
 175                 test1:
 176                 test2:
 177         """)}
 178         C.INVENTORY_ENABLED = ['yaml']
 179         fake_loader = DictDataLoader(inventory_content)
 180         im = InventoryManager(loader=fake_loader, sources=filename)
 181         self.assertTrue(im._inventory.hosts)
 182         self.assertIn('test1', im._inventory.hosts)
 183         self.assertIn('test2', im._inventory.hosts)
 184         self.assertIn(im._inventory.get_host('test1'), im._inventory.groups['all'].hosts)
 185         self.assertIn(im._inventory.get_host('test2'), im._inventory.groups['all'].hosts)
 186         self.assertEqual(len(im._inventory.groups['all'].hosts), 2)
 187         self.assertIn(im._inventory.get_host('test1'), im._inventory.groups['ungrouped'].hosts)
 188         self.assertIn(im._inventory.get_host('test2'), im._inventory.groups['ungrouped'].hosts)
 189         self.assertEqual(len(im._inventory.groups['ungrouped'].hosts), 2)

 191     def _get_inventory(self, inventory_content):
 192 
 193         fake_loader = DictDataLoader({__file__: inventory_content})
 194 
 195         return InventoryManager(loader=fake_loader, sources=[__file__])

 197     def _test_default_groups(self, inventory_content):
 198         inventory = self._get_inventory(inventory_content)
 199 
 200         self.assertIn('all', inventory.groups)
 201         self.assertIn('ungrouped', inventory.groups)
 202         all_hosts = set(host.name for host in inventory.groups['all'].get_hosts())
 203         self.assertEqual(set(['host1', 'host2', 'host3', 'host4', 'host5']), all_hosts)
 204         ungrouped_hosts = set(host.name for host in inventory.groups['ungrouped'].get_hosts())
 205         self.assertEqual(set(['host1', 'host2']), ungrouped_hosts)
 206         servers_hosts = set(host.name for host in inventory.groups['servers'].get_hosts())
 207         self.assertEqual(set(['host3', 'host4', 'host5']), servers_hosts)
