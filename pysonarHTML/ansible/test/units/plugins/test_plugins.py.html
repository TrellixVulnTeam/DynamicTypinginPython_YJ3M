  45     def test_plugins__get_package_paths_with_package(self):
  46         # the _get_package_paths() call uses __import__ to load a
  47         # python library, and then uses the __file__ attribute of
  48         # the result for that to get the library path, so we mock
  49         # that here and patch the builtin to use our mocked result
  50         foo = MagicMock()
  51         bar = MagicMock()
  52         bam = MagicMock()
  53         bam.__file__ = '/path/to/my/foo/bar/bam/__init__.py'
  54         bar.bam = bam
  55         foo.return_value.bar = bar
  56         pl = PluginLoader('test', 'foo.bar.bam', 'test', 'test_plugin')
  57         with patch('{0}.__import__'.format(BUILTINS), foo):
  58             self.assertEqual(pl._get_package_paths(), ['/path/to/my/foo/bar/bam'])

  91     def test__load_module_source_no_duplicate_names(self):
  92         '''
  93         This test simulates importing 2 plugins with the same name,
  94         and validating that the import is short circuited if a file with the same name
  95         has already been imported
  96         '''
  97 
  98         fixture_path = os.path.join(os.path.dirname(__file__), 'loader_fixtures')
  99 
 100         pl = PluginLoader('test', '', 'test', 'test_plugin')
 101         one = pl._load_module_source('import_fixture', os.path.join(fixture_path, 'import_fixture.py'))
 102         # This line wouldn't even succeed if we didn't short circuit on finding a duplicate name
 103         two = pl._load_module_source('import_fixture', '/path/to/import_fixture.py')
 104 
 105         self.assertEqual(one, two)

 107     @patch('ansible.plugins.loader.glob')
 108     @patch.object(PluginLoader, '_get_paths')
 109     def test_all_no_duplicate_names(self, gp_mock, glob_mock):
 110         '''
 111         This test goes along with ``test__load_module_source_no_duplicate_names``
 112         and ensures that we ignore duplicate imports on multiple paths
 113         '''
 114 
 115         fixture_path = os.path.join(os.path.dirname(__file__), 'loader_fixtures')
 116 
 117         gp_mock.return_value = [
 118             fixture_path,
 119             '/path/to'
 120         ]
 121 
 122         glob_mock.glob.side_effect = [
 123             [os.path.join(fixture_path, 'import_fixture.py')],
 124             ['/path/to/import_fixture.py']
 125         ]
 126 
 127         pl = PluginLoader('test', '', 'test', 'test_plugin')
 128         # Aside from needing ``list()`` so we can do a len, ``PluginLoader.all`` returns a generator
 129         # so ``list()`` actually causes ``PluginLoader.all`` to run.
 130         plugins = list(pl.all())
 131         self.assertEqual(len(plugins), 1)
 132 
 133         self.assertIn(os.path.join(fixture_path, 'import_fixture.py'), pl._module_cache)
 134         self.assertNotIn('/path/to/import_fixture.py', pl._module_cache)
