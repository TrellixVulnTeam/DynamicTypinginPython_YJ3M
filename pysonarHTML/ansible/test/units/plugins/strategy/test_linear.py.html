  22 class TestStrategyLinear(unittest.TestCase):
  23 
  24     @patch('ansible.playbook.role.definition.unfrackpath', mock_unfrackpath_noop)
  25     def test_noop(self):
  26         fake_loader = DictDataLoader({
  27             "test_play.yml": """
  28             - hosts: all
  29               gather_facts: no
  30               tasks:
  31                 - block:
  32                    - block:
  33                      - name: task1
  34                        debug: msg='task1'
  35                        failed_when: inventory_hostname == 'host01'
  36 
  37                      - name: task2
  38                        debug: msg='task2'
  39 
  40                      rescue:
  41                        - name: rescue1
  42                          debug: msg='rescue1'
  43 
  44                        - name: rescue2
  45                          debug: msg='rescue2'
  46             """,
  47         })
  48 
  49         mock_var_manager = MagicMock()
  50         mock_var_manager._fact_cache = dict()
  51         mock_var_manager.get_vars.return_value = dict()
  52 
  53         p = Playbook.load('test_play.yml', loader=fake_loader, variable_manager=mock_var_manager)
  54 
  55         inventory = MagicMock()
  56         inventory.hosts = {}
  57         hosts = []
  58         for i in range(0, 2):
  59             host = MagicMock()
  60             host.name = host.get_name.return_value = 'host%02d' % i
  61             hosts.append(host)
  62             inventory.hosts[host.name] = host
  63         inventory.get_hosts.return_value = hosts
  64         inventory.filter_hosts.return_value = hosts
  65 
  66         mock_var_manager._fact_cache['host00'] = dict()
  67 
  68         play_context = PlayContext(play=p._entries[0])
  69 
  70         itr = PlayIterator(
  71             inventory=inventory,
  72             play=p._entries[0],
  73             play_context=play_context,
  74             variable_manager=mock_var_manager,
  75             all_vars=dict(),
  76         )
  77 
  78         tqm = TaskQueueManager(
  79             inventory=inventory,
  80             variable_manager=mock_var_manager,
  81             loader=fake_loader,
  82             passwords=None,
  83             forks=5,
  84         )
  85         tqm._initialize_processes(3)
  86         strategy = StrategyModule(tqm)
  87         strategy._hosts_cache = [h.name for h in hosts]
  88         strategy._hosts_cache_all = [h.name for h in hosts]
  89 
  90         # implicit meta: flush_handlers
  91         hosts_left = strategy.get_hosts_left(itr)
  92         hosts_tasks = strategy._get_next_task_lockstep(hosts_left, itr)
  93         host1_task = hosts_tasks[0][1]
  94         host2_task = hosts_tasks[1][1]
  95         self.assertIsNotNone(host1_task)
  96         self.assertIsNotNone(host2_task)
  97         self.assertEqual(host1_task.action, 'meta')
  98         self.assertEqual(host2_task.action, 'meta')
  99

 112         # mark the second host failed

 114

 116         hosts_left = strategy.get_hosts_left(itr)
