  23 from units.compat import unittest
  24 from ansible.plugins.lookup.ini import _parse_params
  25 
  26 
  27 class TestINILookup(unittest.TestCase):
  28 
  29     # Currently there isn't a new-style
  30     old_style_params_data = (
  31         # Simple case
  32         dict(
  33             term=u'keyA section=sectionA file=/path/to/file',
  34             expected=[u'file=/path/to/file', u'keyA', u'section=sectionA'],
  35         ),
  36         dict(
  37             term=u'keyB section=sectionB with space file=/path/with/embedded spaces and/file',
  38             expected=[u'file=/path/with/embedded spaces and/file', u'keyB', u'section=sectionB with space'],
  39         ),
  40         dict(
  41             term=u'keyC section=sectionC file=/path/with/equals/cn=com.ansible',
  42             expected=[u'file=/path/with/equals/cn=com.ansible', u'keyC', u'section=sectionC'],
  43         ),
  44         dict(
  45             term=u'keyD section=sectionD file=/path/with space and/equals/cn=com.ansible',
  46             expected=[u'file=/path/with space and/equals/cn=com.ansible', u'keyD', u'section=sectionD'],
  47         ),
  48         dict(
  49             term=u'keyE section=sectionE file=/path/with/unicode/くらとみ/file',
  50             expected=[u'file=/path/with/unicode/くらとみ/file', u'keyE', u'section=sectionE'],
  51         ),
  52         dict(
  53             term=u'keyF section=sectionF file=/path/with/utf 8 and spaces/くらとみ/file',
  54             expected=[u'file=/path/with/utf 8 and spaces/くらとみ/file', u'keyF', u'section=sectionF'],
  55         ),
  56     )
  57 
  58     def test_parse_parameters(self):
  59         for testcase in self.old_style_params_data:
  60             # print(testcase)
  61             params = _parse_params(testcase['term'])
  62             params.sort()
  63             self.assertEqual(params, testcase['expected'])
