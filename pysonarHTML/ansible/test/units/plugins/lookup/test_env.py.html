  14 @pytest.mark.parametrize('env_var,exp_value', [
  15     ('foo', 'bar'),
  16     ('equation', 'a=b*100')
  17 ])
  18 def test_env_var_value(monkeypatch, env_var, exp_value):
  19     monkeypatch.setattr('ansible.utils.py3compat.environ.get', lambda x, y: exp_value)
  20 
  21     env_lookup = lookup_loader.get('env')
  22     retval = env_lookup.run([env_var], None)
  23     assert retval == [exp_value]
