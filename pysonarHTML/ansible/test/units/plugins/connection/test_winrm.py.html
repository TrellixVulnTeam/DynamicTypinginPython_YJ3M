 204     @pytest.mark.parametrize('options, direct, expected, kerb',
 205                              ((o, d, e, k) for o, d, e, k in OPTIONS_DATA))
 206     def test_set_options(self, options, direct, expected, kerb):
 207         winrm.HAVE_KERBEROS = kerb
 208 
 209         pc = PlayContext()
 210         new_stdin = StringIO()
 211 
 212         conn = connection_loader.get('winrm', pc, new_stdin)
 213         conn.set_options(var_options=options, direct=direct)
 214         conn._build_winrm_kwargs()
 215 
 216         for attr, expected in expected.items():
 217             actual = getattr(conn, attr)
 218             assert actual == expected, \
 219                 "winrm attr '%s', actual '%s' != expected '%s'"\
 220                 % (attr, actual, expected)

 225     @pytest.mark.parametrize('options, expected', [
 226         [{"_extras": {}},
 227          (["kinit", "user@domain"],)],
 228         [{"_extras": {}, 'ansible_winrm_kinit_cmd': 'kinit2'},
 229          (["kinit2", "user@domain"],)],
 230         [{"_extras": {'ansible_winrm_kerberos_delegation': True}},
 231          (["kinit", "-f", "user@domain"],)],
 232         [{"_extras": {}, 'ansible_winrm_kinit_args': '-f -p'},
 233          (["kinit", "-f", "-p", "user@domain"],)],
 234         [{"_extras": {}, 'ansible_winrm_kerberos_delegation': True, 'ansible_winrm_kinit_args': '-p'},
 235          (["kinit", "-p", "user@domain"],)]

 237     def test_kinit_success_subprocess(self, monkeypatch, options, expected):
 238         def mock_communicate(input=None, timeout=None):
 239             return b"", b""
 240 
 241         mock_popen = MagicMock()
 242         mock_popen.return_value.communicate = mock_communicate
 243         mock_popen.return_value.returncode = 0
 244         monkeypatch.setattr("subprocess.Popen", mock_popen)
 245 
 246         winrm.HAS_PEXPECT = False
 247         pc = PlayContext()
 248         new_stdin = StringIO()
 249         conn = connection_loader.get('winrm', pc, new_stdin)
 250         conn.set_options(var_options=options)
 251         conn._build_winrm_kwargs()
 252 
 253         conn._kerb_auth("user@domain", "pass")
 254         mock_calls = mock_popen.mock_calls
 255         assert len(mock_calls) == 1
 256         assert mock_calls[0][1] == expected
 257         actual_env = mock_calls[0][2]['env']
 258         assert list(actual_env.keys()) == ['KRB5CCNAME']
 259         assert actual_env['KRB5CCNAME'].startswith("FILE:/")

 273     def test_kinit_success_pexpect(self, monkeypatch, options, expected):
 274         pytest.importorskip("pexpect")
 275         mock_pexpect = MagicMock()
 276         mock_pexpect.return_value.exitstatus = 0
 277         monkeypatch.setattr("pexpect.spawn", mock_pexpect)
 278 
 279         winrm.HAS_PEXPECT = True
 280         pc = PlayContext()
 281         new_stdin = StringIO()
 282         conn = connection_loader.get('winrm', pc, new_stdin)
 283         conn.set_options(var_options=options)
 284         conn._build_winrm_kwargs()
 285 
 286         conn._kerb_auth("user@domain", "pass")
 287         mock_calls = mock_pexpect.mock_calls
 288         assert mock_calls[0][1] == expected
 289         actual_env = mock_calls[0][2]['env']
 290         assert list(actual_env.keys()) == ['KRB5CCNAME']
 291         assert actual_env['KRB5CCNAME'].startswith("FILE:/")
 292         assert mock_calls[0][2]['echo'] is False
 293         assert mock_calls[1][0] == "().expect"
 294         assert mock_calls[1][1] == (".*:",)
 295         assert mock_calls[2][0] == "().sendline"
 296         assert mock_calls[2][1] == ("pass",)
 297         assert mock_calls[3][0] == "().read"
 298         assert mock_calls[4][0] == "().wait"

 300     def test_kinit_with_missing_executable_subprocess(self, monkeypatch):
 301         expected_err = "[Errno 2] No such file or directory: " \
 302                        "'/fake/kinit': '/fake/kinit'"
 303         mock_popen = MagicMock(side_effect=OSError(expected_err))
 304 
 305         monkeypatch.setattr("subprocess.Popen", mock_popen)
 306 
 307         winrm.HAS_PEXPECT = False
 308         pc = PlayContext()
 309         new_stdin = StringIO()
 310         conn = connection_loader.get('winrm', pc, new_stdin)
 311         options = {"_extras": {}, "ansible_winrm_kinit_cmd": "/fake/kinit"}
 312         conn.set_options(var_options=options)
 313         conn._build_winrm_kwargs()
 314 
 315         with pytest.raises(AnsibleConnectionFailure) as err:
 316             conn._kerb_auth("user@domain", "pass")
 317         assert str(err.value) == "Kerberos auth failure when calling " \
 318                                  "kinit cmd '/fake/kinit': %s" % expected_err
