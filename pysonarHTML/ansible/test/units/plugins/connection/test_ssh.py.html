 222     def test_plugins_connection_ssh_put_file(self, mock_ospe, mock_sleep):
 223         pc = PlayContext()
 224         new_stdin = StringIO()
 225         conn = connection_loader.get('ssh', pc, new_stdin)
 226         conn._build_command = MagicMock()
 227         conn._bare_run = MagicMock()
 228 
 229         mock_ospe.return_value = True
 230         conn._build_command.return_value = 'some command to run'
 231         conn._bare_run.return_value = (0, '', '')
 232         conn.host = "some_host"
 233 
 234         C.ANSIBLE_SSH_RETRIES = 9
 235 
 236         # Test with C.DEFAULT_SCP_IF_SSH set to smart
 237         # Test when SFTP works
 238         C.DEFAULT_SCP_IF_SSH = 'smart'
 239         expected_in_data = b' '.join((b'put', to_bytes(shlex_quote('/path/to/in/file')), to_bytes(shlex_quote('/path/to/dest/file')))) + b'\n'
 240         conn.put_file('/path/to/in/file', '/path/to/dest/file')
 241         conn._bare_run.assert_called_with('some command to run', expected_in_data, checkrc=False)
 242 
 243         # Test when SFTP doesn't work but SCP does
 244         conn._bare_run.side_effect = [(1, 'stdout', 'some errors'), (0, '', '')]
 245         conn.put_file('/path/to/in/file', '/path/to/dest/file')
 246         conn._bare_run.assert_called_with('some command to run', None, checkrc=False)
 247         conn._bare_run.side_effect = None
 248 
 249         # test with C.DEFAULT_SCP_IF_SSH enabled
 250         C.DEFAULT_SCP_IF_SSH = True
 251         conn.put_file('/path/to/in/file', '/path/to/dest/file')
 252         conn._bare_run.assert_called_with('some command to run', None, checkrc=False)
 253 
 254         conn.put_file(u'/path/to/in/file/with/unicode-fö〩', u'/path/to/dest/file/with/unicode-fö〩')
 255         conn._bare_run.assert_called_with('some command to run', None, checkrc=False)
 256 
 257         # test with C.DEFAULT_SCP_IF_SSH disabled
 258         C.DEFAULT_SCP_IF_SSH = False
 259         expected_in_data = b' '.join((b'put', to_bytes(shlex_quote('/path/to/in/file')), to_bytes(shlex_quote('/path/to/dest/file')))) + b'\n'
 260         conn.put_file('/path/to/in/file', '/path/to/dest/file')
 261         conn._bare_run.assert_called_with('some command to run', expected_in_data, checkrc=False)
 262 
 263         expected_in_data = b' '.join((b'put',
 264                                       to_bytes(shlex_quote('/path/to/in/file/with/unicode-fö〩')),
 265                                       to_bytes(shlex_quote('/path/to/dest/file/with/unicode-fö〩')))) + b'\n'
 266         conn.put_file(u'/path/to/in/file/with/unicode-fö〩', u'/path/to/dest/file/with/unicode-fö〩')
 267         conn._bare_run.assert_called_with('some command to run', expected_in_data, checkrc=False)
 268 
 269         # test that a non-zero rc raises an error
 270         conn._bare_run.return_value = (1, 'stdout', 'some errors')
 271         self.assertRaises(AnsibleError, conn.put_file, '/path/to/bad/file', '/remote/path/to/file')
 272 
 273         # test that a not-found path raises an error
 274         mock_ospe.return_value = False
 275         conn._bare_run.return_value = (0, 'stdout', '')
 276         self.assertRaises(AnsibleFileNotFound, conn.put_file, '/path/to/bad/file', '/remote/path/to/file')
