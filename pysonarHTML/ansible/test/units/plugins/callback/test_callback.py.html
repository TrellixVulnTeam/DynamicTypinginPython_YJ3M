 120 class TestCallbackDumpResults(object):
 121     def test_internal_keys(self):
 122         cb = CallbackBase()
 123         result = {'item': 'some_item',
 124                   '_ansible_some_var': 'SENTINEL',
 125                   'testing_ansible_out': 'should_be_left_in LEFTIN',
 126                   'invocation': 'foo --bar whatever [some_json]',
 127                   'some_dict_key': {'a_sub_dict_for_key': 'baz'},
 128                   'bad_dict_key': {'_ansible_internal_blah': 'SENTINEL'},
 129                   'changed': True}
 130         json_out = cb._dump_results(result)
 131         assert '"_ansible_' not in json_out
 132         assert 'SENTINEL' not in json_out
 133         assert 'LEFTIN' in json_out
 134 
 135     def test_exception(self):
 136         cb = CallbackBase()
 137         result = {'item': 'some_item LEFTIN',
 138                   'exception': ['frame1', 'SENTINEL']}
 139         json_out = cb._dump_results(result)
 140         assert 'SENTINEL' not in json_out
 141         assert 'exception' not in json_out
 142         assert 'LEFTIN' in json_out
 143 
 144     def test_verbose(self):
 145         cb = CallbackBase()
 146         result = {'item': 'some_item LEFTIN',
 147                   '_ansible_verbose_always': 'chicane'}
 148         json_out = cb._dump_results(result)
 149         assert 'SENTINEL' not in json_out
 150         assert 'LEFTIN' in json_out
 151 
 152     def test_diff(self):
 153         cb = CallbackBase()
 154         result = {'item': 'some_item LEFTIN',
 155                   'diff': ['remove stuff', 'added LEFTIN'],
 156                   '_ansible_verbose_always': 'chicane'}
 157         json_out = cb._dump_results(result)
 158         assert 'SENTINEL' not in json_out
 159         assert 'LEFTIN' in json_out
 160 
 161     def test_mixed_keys(self):
 162         cb = CallbackBase()
 163         result = {3: 'pi',
 164                   'tau': 6}
 165         json_out = cb._dump_results(result)
 166         round_trip_result = json.loads(json_out)
 167         assert len(round_trip_result) == 2
 168         assert '3' in round_trip_result
 169         assert 'tau' in round_trip_result
 170         assert round_trip_result['3'] == 'pi'
 171         assert round_trip_result['tau'] == 6

 179     def _strip_color(self, s):
 180         return re.sub('\033\\[[^m]*m', '', s)

 182     def test_difflist(self):
 183         # TODO: split into smaller tests?
 184         difflist = [{'before': u'preface\nThe Before String\npostscript',
 185                      'after': u'preface\nThe After String\npostscript',
 186                      'before_header': u'just before',
 187                      'after_header': u'just after'
 188                      },
 189                     {'before': u'preface\nThe Before String\npostscript',
 190                      'after': u'preface\nThe After String\npostscript',
 191                      },
 192                     {'src_binary': 'chicane'},
 193                     {'dst_binary': 'chicanery'},
 194                     {'dst_larger': 1},
 195                     {'src_larger': 2},
 196                     {'prepared': u'what does prepared do?'},
 197                     {'before_header': u'just before'},
 198                     {'after_header': u'just after'}]
 199 
 200         res = self.cb._get_diff(difflist)
 201 
 202         self.assertIn(u'Before String', res)
 203         self.assertIn(u'After String', res)
 204         self.assertIn(u'just before', res)
 205         self.assertIn(u'just after', res)

 381 class TestCallbackOnMethods(unittest.TestCase):
 382     def _find_on_methods(self, callback):
 383         cb_dir = dir(callback)
 384         method_names = [x for x in cb_dir if '_on_' in x]
 385         methods = [getattr(callback, mn) for mn in method_names]
 386         return methods
 387 
 388     def test_are_methods(self):
 389         cb = CallbackBase()
 390         for method in self._find_on_methods(cb):
 391             self.assertIsInstance(method, types.MethodType)
 392 
 393     def test_on_any(self):
 394         cb = CallbackBase()
 395         cb.v2_on_any('whatever', some_keyword='blippy')
 396         cb.on_any('whatever', some_keyword='blippy')
