  17 def test_sudo(mocker, parser, reset_cli_args):
  18     options = parser.parse_args([])
  19     context._init_global_context(options)
  20     play_context = PlayContext()
  21 
  22     default_cmd = "/bin/foo"
  23     default_exe = "/bin/bash"
  24     sudo_exe = 'sudo'
  25     sudo_flags = '-H -s -n'
  26 
  27     cmd = play_context.make_become_cmd(cmd=default_cmd, executable=default_exe)
  28     assert cmd == default_cmd
  29 
  30     success = 'BECOME-SUCCESS-.+?'
  31 
  32     play_context.become = True
  33     play_context.become_user = 'foo'
  34     play_context.set_become_plugin(become_loader.get('sudo'))
  35     play_context.become_flags = sudo_flags
  36     cmd = play_context.make_become_cmd(cmd=default_cmd, executable=default_exe)
  37 
  38     assert (re.match("""%s %s  -u %s %s -c 'echo %s; %s'""" % (sudo_exe, sudo_flags, play_context.become_user,
  39                                                                default_exe, success, default_cmd), cmd) is not None)
  40 
  41     play_context.become_pass = 'testpass'
  42     cmd = play_context.make_become_cmd(cmd=default_cmd, executable=default_exe)
  43     assert (re.match("""%s %s -p "%s" -u %s %s -c 'echo %s; %s'""" % (sudo_exe, sudo_flags.replace('-n', ''),
  44                                                                       r"\[sudo via ansible, key=.+?\] password:", play_context.become_user,
  45                                                                       default_exe, success, default_cmd), cmd) is not None)
