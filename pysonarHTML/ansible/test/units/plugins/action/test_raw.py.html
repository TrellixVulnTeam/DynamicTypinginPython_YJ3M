  47     def test_raw_executable_is_not_empty_string(self):
  48 
  49         task = MagicMock(Task)
  50         task.async_val = False
  51 
  52         task.args = {'_raw_params': 'Args1'}
  53         self.play_context.check_mode = False
  54 
  55         self.mock_am = ActionModule(task, self.connection, self.play_context, loader=None, templar=None, shared_loader_obj=None)
  56         self.mock_am._low_level_execute_command = Mock(return_value={})
  57         self.mock_am.display = Mock()
  58         self.mock_am._admin_users = ['root', 'toor']
  59 
  60         self.mock_am.run()
