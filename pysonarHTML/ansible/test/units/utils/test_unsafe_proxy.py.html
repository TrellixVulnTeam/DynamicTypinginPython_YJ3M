  86 def test_wrap_var_no_ref():
  87     thing = {
  88         'foo': {
  89             'bar': 'baz'
  90         },
  91         'bar': ['baz', 'qux'],
  92         'baz': ('qux',),
  93         'none': None,
  94         'text': 'text',
  95     }
  96     wrapped_thing = wrap_var(thing)
  97     thing is not wrapped_thing
  98     thing['foo'] is not wrapped_thing['foo']
  99     thing['bar'][0] is not wrapped_thing['bar'][0]
 100     thing['baz'][0] is not wrapped_thing['baz'][0]
 101     thing['none'] is not wrapped_thing['none']
 102     thing['text'] is not wrapped_thing['text']
