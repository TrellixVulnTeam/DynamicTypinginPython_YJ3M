  59 def test_cliargs_argparse():
  60     parser = argparse.ArgumentParser(description='Process some integers.')
  61     parser.add_argument('integers', metavar='N', type=int, nargs='+',
  62                         help='an integer for the accumulator')
  63     parser.add_argument('--sum', dest='accumulate', action='store_const',
  64                         const=sum, default=max,
  65                         help='sum the integers (default: find the max)')
  66     args = parser.parse_args([u'--sum', u'1', u'2'])
  67 
  68     expected = frozenset((('accumulate', sum), ('integers', (1, 2))))
  69 
  70     assert frozenset(co.CLIArgs.from_options(args).items()) == expected
