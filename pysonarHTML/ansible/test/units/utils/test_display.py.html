  16 def test_get_text_width():
  17     initialize_locale()
  18     assert get_text_width(u'„Ç≥„É≥„Éã„ÉÅ„Éè') == 10
  19     assert get_text_width(u'ab„Ç≥cd') == 6
  20     assert get_text_width(u'caf√©') == 4
  21     assert get_text_width(u'four') == 4
  22     assert get_text_width(u'\u001B') == 0
  23     assert get_text_width(u'ab\u0000') == 2
  24     assert get_text_width(u'ab„Ç≥\u0000') == 4
  25     assert get_text_width(u'üöÄüêÆ') == 4
  26     assert get_text_width(u'\x08') == 0
  27     assert get_text_width(u'\x08\x08') == 0
  28     assert get_text_width(u'ab\x08cd') == 3
  29     assert get_text_width(u'ab\x1bcd') == 3
  30     assert get_text_width(u'ab\x7fcd') == 3
  31     assert get_text_width(u'ab\x94cd') == 3
  32 
  33     pytest.raises(TypeError, get_text_width, 1)
  34     pytest.raises(TypeError, get_text_width, b'four')
