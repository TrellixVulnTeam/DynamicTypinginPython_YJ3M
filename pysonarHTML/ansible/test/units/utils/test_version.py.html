 223 def test_example_precedence():
 224     # https://semver.org/#spec-item-11
 225     sv = SemanticVersion
 226     assert sv('1.0.0') < sv('2.0.0') < sv('2.1.0') < sv('2.1.1')
 227     assert sv('1.0.0-alpha') < sv('1.0.0')
 228     assert sv('1.0.0-alpha') < sv('1.0.0-alpha.1') < sv('1.0.0-alpha.beta')
 229     assert sv('1.0.0-beta') < sv('1.0.0-beta.2') < sv('1.0.0-beta.11') < sv('1.0.0-rc.1') < sv('1.0.0')

 256 def test_alpha():
 257     assert _Alpha('a') == _Alpha('a')
 258     assert _Alpha('a') == 'a'
 259     assert _Alpha('a') != _Alpha('b')
 260     assert _Alpha('a') != 1
 261     assert _Alpha('a') < _Alpha('b')
 262     assert _Alpha('a') < 'c'
 263     assert _Alpha('a') > _Numeric(1)
 264     with pytest.raises(ValueError):
 265         _Alpha('a') < None
 266     assert _Alpha('a') <= _Alpha('a')
 267     assert _Alpha('a') <= _Alpha('b')
 268     assert _Alpha('b') >= _Alpha('a')
 269     assert _Alpha('b') >= _Alpha('b')
