  72     def test_combine_vars_improper_args(self):
  73         with mock.patch('ansible.constants.DEFAULT_HASH_BEHAVIOUR', 'replace'):
  74             with self.assertRaises(AnsibleError):
  75                 combine_vars([1, 2, 3], dict(a=1))
  76             with self.assertRaises(AnsibleError):
  77                 combine_vars(dict(a=1), [1, 2, 3])
  78 
  79         with mock.patch('ansible.constants.DEFAULT_HASH_BEHAVIOUR', 'merge'):
  80             with self.assertRaises(AnsibleError):
  81                 combine_vars([1, 2, 3], dict(a=1))
  82             with self.assertRaises(AnsibleError):
  83                 combine_vars(dict(a=1), [1, 2, 3])

  85     def test_combine_vars_replace(self):
  86         with mock.patch('ansible.constants.DEFAULT_HASH_BEHAVIOUR', 'replace'):
  87             for test in self.combine_vars_replace_data:
  88                 self.assertEqual(combine_vars(test['a'], test['b']), test['result'])

 118     def test_merge_hash_simple(self):
