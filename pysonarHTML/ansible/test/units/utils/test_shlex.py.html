  18 from __future__ import (absolute_import, division, print_function)
  19 __metaclass__ = type
  20 
  21 import unittest
  22 
  23 from ansible.utils.shlex import shlex_split
  24 
  25 
  26 class TestSplit(unittest.TestCase):
  27 
  28     def test_trivial(self):
  29         self.assertEqual(shlex_split("a b c"), ["a", "b", "c"])
  30 
  31     def test_unicode(self):
  32         self.assertEqual(shlex_split(u"a b \u010D"), [u"a", u"b", u"\u010D"])
  33 
  34     def test_quoted(self):
  35         self.assertEqual(shlex_split('"a b" c'), ["a b", "c"])
  36 
  37     def test_comments(self):
  38         self.assertEqual(shlex_split('"a b" c # d', comments=True), ["a b", "c"])
  39 
  40     def test_error(self):
  41         self.assertRaises(ValueError, shlex_split, 'a "b')
