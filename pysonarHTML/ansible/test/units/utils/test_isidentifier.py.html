  17 @pytest.mark.parametrize(
  18     "identifier", [
  19         "foo", "foo1_23",
  20     ]
  21 )
  22 def test_valid_identifier(identifier):
  23     assert isidentifier(identifier)
  24 
  25 
  26 @pytest.mark.parametrize(
  27     "identifier", [
  28         "pass", "foo ", " foo", "1234", "1234abc", "", "   ", "foo bar", "no-dashed-names-for-you",
  29     ]
  30 )
  31 def test_invalid_identifier(identifier):
  32     assert not isidentifier(identifier)

  35 def test_keywords_not_in_PY2():
  36     """In Python 2 ("True", "False", "None") are not keywords. The isidentifier
  37     method ensures that those are treated as keywords on both Python 2 and 3.
  38     """
  39     assert not isidentifier("True")
  40     assert not isidentifier("False")
  41     assert not isidentifier("None")
  42 
  43 
  44 def test_non_ascii():
  45     """In Python 3 non-ascii characters are allowed as opposed to Python 2. The
  46     isidentifier method ensures that those are treated as keywords on both
  47     Python 2 and 3.
  48     """
  49     assert not isidentifier("křížek")
