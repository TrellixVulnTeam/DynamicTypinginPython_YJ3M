 138     @patch('ansible.playbook.role.definition.unfrackpath', mock_unfrackpath_noop)
 139     def test_variable_manager_precedence(self):
 140         # FIXME: this needs to be redone as dataloader is not the automatic source of data anymore
 141         return
 142 
 143         # pylint: disable=unreachable
 144         '''
 145         Tests complex variations and combinations of get_vars() with different
 146         objects to modify the context under which variables are merged.
 147         '''
 148         # FIXME: BCS makethiswork
 149         # return True
 150 
 151         mock_inventory = MagicMock()
 152 
 153         inventory1_filedata = """
 154             [group2:children]
 155             group1
 156 
 157             [group1]
 158             host1 host_var=host_var_from_inventory_host1
 159 
 160             [group1:vars]
 161             group_var = group_var_from_inventory_group1
 162 
 163             [group2:vars]
 164             group_var = group_var_from_inventory_group2
 165             """
 166 
 167         fake_loader = DictDataLoader({
 168             # inventory1
 169             '/etc/ansible/inventory1': inventory1_filedata,
 170             # role defaults_only1
 171             '/etc/ansible/roles/defaults_only1/defaults/main.yml': """
 172             default_var: "default_var_from_defaults_only1"
 173             host_var: "host_var_from_defaults_only1"
 174             group_var: "group_var_from_defaults_only1"
 175             group_var_all: "group_var_all_from_defaults_only1"
 176             extra_var: "extra_var_from_defaults_only1"
 177             """,
 178             '/etc/ansible/roles/defaults_only1/tasks/main.yml': """
 179             - debug: msg="here i am"
 180             """,
 181 
 182             # role defaults_only2
 183             '/etc/ansible/roles/defaults_only2/defaults/main.yml': """
 184             default_var: "default_var_from_defaults_only2"
 185             host_var: "host_var_from_defaults_only2"
 186             group_var: "group_var_from_defaults_only2"
 187             group_var_all: "group_var_all_from_defaults_only2"
 188             extra_var: "extra_var_from_defaults_only2"
 189             """,
 190         })
 191 
 192         inv1 = InventoryManager(loader=fake_loader, sources=['/etc/ansible/inventory1'])
 193         v = VariableManager(inventory=mock_inventory, loader=fake_loader)
 194 
 195         play1 = Play.load(dict(
 196             hosts=['all'],
 197             roles=['defaults_only1', 'defaults_only2'],
 198         ), loader=fake_loader, variable_manager=v)
 199 
 200         # first we assert that the defaults as viewed as a whole are the merged results
 201         # of the defaults from each role, with the last role defined "winning" when
 202         # there is a variable naming conflict
 203         res = v.get_vars(play=play1)
 204         self.assertEqual(res['default_var'], 'default_var_from_defaults_only2')
 205 
 206         # next, we assert that when vars are viewed from the context of a task within a
 207         # role, that task will see its own role defaults before any other role's
 208         blocks = play1.compile()
 209         task = blocks[1].block[0]
 210         res = v.get_vars(play=play1, task=task)
 211         self.assertEqual(res['default_var'], 'default_var_from_defaults_only1')
 212 
 213         # next we assert the precedence of inventory variables
 214         v.set_inventory(inv1)
 215         h1 = inv1.get_host('host1')
 216 
 217         res = v.get_vars(play=play1, host=h1)
 218         self.assertEqual(res['group_var'], 'group_var_from_inventory_group1')
 219         self.assertEqual(res['host_var'], 'host_var_from_inventory_host1')
 220 
 221         # next we test with group_vars/ files loaded
 222         fake_loader.push("/etc/ansible/group_vars/all", """
 223         group_var_all: group_var_all_from_group_vars_all
 224         """)
 225         fake_loader.push("/etc/ansible/group_vars/group1", """
 226         group_var: group_var_from_group_vars_group1
 227         """)
 228         fake_loader.push("/etc/ansible/group_vars/group3", """
 229         # this is a dummy, which should not be used anywhere
 230         group_var: group_var_from_group_vars_group3
 231         """)
 232         fake_loader.push("/etc/ansible/host_vars/host1", """
 233         host_var: host_var_from_host_vars_host1
 234         """)
 235         fake_loader.push("group_vars/group1", """
 236         playbook_group_var: playbook_group_var
 237         """)
 238         fake_loader.push("host_vars/host1", """
 239         playbook_host_var: playbook_host_var
 240         """)
 241 
 242         res = v.get_vars(play=play1, host=h1)
 243         # self.assertEqual(res['group_var'], 'group_var_from_group_vars_group1')
 244         # self.assertEqual(res['group_var_all'], 'group_var_all_from_group_vars_all')
 245         # self.assertEqual(res['playbook_group_var'], 'playbook_group_var')
 246         # self.assertEqual(res['host_var'], 'host_var_from_host_vars_host1')
 247         # self.assertEqual(res['playbook_host_var'], 'playbook_host_var')
 248 
 249         # add in the fact cache
 250         v._fact_cache['host1'] = dict(fact_cache_var="fact_cache_var_from_fact_cache")
 251 
 252         res = v.get_vars(play=play1, host=h1)
 253         self.assertEqual(res['fact_cache_var'], 'fact_cache_var_from_fact_cache')

 255     @patch('ansible.playbook.role.definition.unfrackpath', mock_unfrackpath_noop)
 256     def test_variable_manager_role_vars_dependencies(self):
 257         '''
 258         Tests vars from role dependencies with duplicate dependencies.
 259         '''
 260         mock_inventory = MagicMock()
 261 
 262         fake_loader = DictDataLoader({
 263             # role common-role
 264             '/etc/ansible/roles/common-role/tasks/main.yml': """
 265             - debug: msg="{{role_var}}"
 266             """,
 267             # We do not need allow_duplicates: yes for this role
 268             # because eliminating duplicates is done by the execution
 269             # strategy, which we do not test here.
 270 
 271             # role role1
 272             '/etc/ansible/roles/role1/vars/main.yml': """
 273             role_var: "role_var_from_role1"
 274             """,
 275             '/etc/ansible/roles/role1/meta/main.yml': """
 276             dependencies:
 277               - { role: common-role }
 278             """,
 279 
 280             # role role2
 281             '/etc/ansible/roles/role2/vars/main.yml': """
 282             role_var: "role_var_from_role2"
 283             """,
 284             '/etc/ansible/roles/role2/meta/main.yml': """
 285             dependencies:
 286               - { role: common-role }
 287             """,
 288         })
 289 
 290         v = VariableManager(loader=fake_loader, inventory=mock_inventory)
 291 
 292         play1 = Play.load(dict(
 293             hosts=['all'],
 294             roles=['role1', 'role2'],
 295         ), loader=fake_loader, variable_manager=v)
 296 
 297         # The task defined by common-role exists twice because role1
 298         # and role2 depend on common-role.  Check that the tasks see
 299         # different values of role_var.
 300         blocks = play1.compile()
 301         task = blocks[1].block[0]
 302         res = v.get_vars(play=play1, task=task)
 303         self.assertEqual(res['role_var'], 'role_var_from_role1')
 304 
 305         task = blocks[2].block[0]
 306         res = v.get_vars(play=play1, task=task)
 307         self.assertEqual(res['role_var'], 'role_var_from_role2')
