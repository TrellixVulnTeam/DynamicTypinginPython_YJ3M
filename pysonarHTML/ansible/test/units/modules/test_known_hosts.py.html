  17     def _create_file(self, content):
  18         tmp_file = tempfile.NamedTemporaryFile(prefix='ansible-test-', suffix='-known_hosts', delete=False)
  19         tmp_file.write(to_bytes(content))
  20         tmp_file.close()
  21         self.addCleanup(os.unlink, tmp_file.name)
  22         return tmp_file.name

  24     def test_no_existing_file(self):
  25         path = "/tmp/this_file_does_not_exists_known_hosts"
  26         key = 'example.com ssh-rsa AAAAetc\n'
  27         diff = compute_diff(path, found_line=None, replace_or_add=False, state='present', key=key)
  28         self.assertEqual(diff, {
  29             'before_header': '/dev/null',
  30             'after_header': path,
  31             'before': '',
  32             'after': 'example.com ssh-rsa AAAAetc\n',
  33         })
