  25     def test_multiline_exec(self):
  26         # This was taken from a real service that specified "ExecStart=/bin/echo foo\nbar"
  27         lines = [
  28             'Type=simple',
  29             'ExecStart={ path=/bin/echo ; argv[]=/bin/echo foo',
  30             'bar ; ignore_errors=no ; start_time=[n/a] ; stop_time=[n/a] ; pid=0 ; code=(null) ; status=0/0 }',
  31             'Description=blah',
  32         ]
  33         parsed = parse_systemctl_show(lines)
  34         self.assertEqual(parsed, {
  35             'Type': 'simple',
  36             'ExecStart': '{ path=/bin/echo ; argv[]=/bin/echo foo\n'
  37                          'bar ; ignore_errors=no ; start_time=[n/a] ; stop_time=[n/a] ; pid=0 ; code=(null) ; status=0/0 }',
  38             'Description': 'blah',
  39         })
