 147 class TestYumUpdateCheckParse(unittest.TestCase):
 148     def _assert_expected(self, expected_pkgs, result):
 149 
 150         for expected_pkg in expected_pkgs:
 151             self.assertIn(expected_pkg, result)
 152         self.assertEqual(len(result), len(expected_pkgs))
 153         self.assertIsInstance(result, dict)
 154 
 155     def test_empty_output(self):
 156         res, obs = YumModule.parse_check_update("")
 157         expected_pkgs = []
 158         self._assert_expected(expected_pkgs, res)
 159 
 160     def test_longname(self):
 161         res, obs = YumModule.parse_check_update(longname)
 162         expected_pkgs = ['xxxxxxxxxxxxxxxxxxxxxxxxxx', 'glibc']
 163         self._assert_expected(expected_pkgs, res)
