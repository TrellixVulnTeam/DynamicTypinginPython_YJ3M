  31 class TestBackslashEscape(unittest.TestCase):
  32 
  33     test_data = (
  34         # Test backslashes in a filter arg are double escaped
  35         dict(
  36             template=u"{{ 'test2 %s' | format('\\1') }}",
  37             intermediate=u"{{ 'test2 %s' | format('\\\\1') }}",
  38             expectation=u"test2 \\1",
  39             args=dict()
  40         ),
  41         # Test backslashes inside the jinja2 var itself are double
  42         # escaped
  43         dict(
  44             template=u"Test 2\\3: {{ '\\1 %s' | format('\\2') }}",
  45             intermediate=u"Test 2\\3: {{ '\\\\1 %s' | format('\\\\2') }}",
  46             expectation=u"Test 2\\3: \\1 \\2",
  47             args=dict()
  48         ),
  49         # Test backslashes outside of the jinja2 var are not double
  50         # escaped
  51         dict(
  52             template=u"Test 2\\3: {{ 'test2 %s' | format('\\1') }}; \\done",
  53             intermediate=u"Test 2\\3: {{ 'test2 %s' | format('\\\\1') }}; \\done",
  54             expectation=u"Test 2\\3: test2 \\1; \\done",
  55             args=dict()
  56         ),
  57         # Test backslashes in a variable sent to a filter are handled
  58         dict(
  59             template=u"{{ 'test2 %s' | format(var1) }}",
  60             intermediate=u"{{ 'test2 %s' | format(var1) }}",
  61             expectation=u"test2 \\1",
  62             args=dict(var1=u'\\1')
  63         ),
  64         # Test backslashes in a variable expanded by jinja2 are double
  65         # escaped
  66         dict(
  67             template=u"Test 2\\3: {{ var1 | format('\\2') }}",
  68             intermediate=u"Test 2\\3: {{ var1 | format('\\\\2') }}",
  69             expectation=u"Test 2\\3: \\1 \\2",
  70             args=dict(var1=u'\\1 %s')
  71         ),
  72     )
  73 
  74     def setUp(self):
  75         self.env = jinja2.Environment()
  76 
  77     def test_backslash_escaping(self):
  78 
  79         for test in self.test_data:
  80             intermediate = _escape_backslashes(test['template'], self.env)
  81             self.assertEqual(intermediate, test['intermediate'])
  82             template = jinja2.Template(intermediate)
  83             args = test['args']
  84             self.assertEqual(template.render(**args), test['expectation'])

  87 class TestCountNewlines(unittest.TestCase):
  88 
  89     def test_zero_length_string(self):
  90         self.assertEqual(_count_newlines_from_end(u''), 0)
  91 
  92     def test_short_string(self):
  93         self.assertEqual(_count_newlines_from_end(u'The quick\n'), 1)
  94 
  95     def test_one_newline(self):
  96         self.assertEqual(_count_newlines_from_end(u'The quick brown fox jumped over the lazy dog' * 1000 + u'\n'), 1)
  97 
  98     def test_multiple_newlines(self):
  99         self.assertEqual(_count_newlines_from_end(u'The quick brown fox jumped over the lazy dog' * 1000 + u'\n\n\n'), 3)
 100 
 101     def test_zero_newlines(self):
 102         self.assertEqual(_count_newlines_from_end(u'The quick brown fox jumped over the lazy dog' * 1000), 0)
 103 
 104     def test_all_newlines(self):
 105         self.assertEqual(_count_newlines_from_end(u'\n' * 10), 10)
 106 
 107     def test_mostly_newlines(self):
 108         self.assertEqual(_count_newlines_from_end(u'The quick brown fox jumped over the lazy dog' + u'\n' * 1000), 1000)
