 233 class TestTemplarMisc(BaseTemplar, unittest.TestCase):
 234     def test_templar_simple(self):
 235 
 236         templar = self.templar
 237         # test some basic templating
 238         self.assertEqual(templar.template("{{foo}}"), "bar")
 239         self.assertEqual(templar.template("{{foo}}\n"), "bar\n")
 240         self.assertEqual(templar.template("{{foo}}\n", preserve_trailing_newlines=True), "bar\n")
 241         self.assertEqual(templar.template("{{foo}}\n", preserve_trailing_newlines=False), "bar")
 242         self.assertEqual(templar.template("{{bam}}"), "bar")
 243         self.assertEqual(templar.template("{{num}}"), 1)
 244         self.assertEqual(templar.template("{{var_true}}"), True)
 245         self.assertEqual(templar.template("{{var_false}}"), False)
 246         self.assertEqual(templar.template("{{var_dict}}"), dict(a="b"))
 247         self.assertEqual(templar.template("{{bad_dict}}"), "{a='b'")
 248         self.assertEqual(templar.template("{{var_list}}"), [1])
 249         self.assertEqual(templar.template(1, convert_bare=True), 1)
 250 
 251         # force errors
 252         self.assertRaises(AnsibleUndefinedVariable, templar.template, "{{bad_var}}")
 253         self.assertRaises(AnsibleUndefinedVariable, templar.template, "{{lookup('file', bad_var)}}")
 254         self.assertRaises(AnsibleError, templar.template, "{{lookup('bad_lookup')}}")
 255         self.assertRaises(AnsibleError, templar.template, "{{recursive}}")
 256         self.assertRaises(AnsibleUndefinedVariable, templar.template, "{{foo-bar}}")
 257 
 258         # test with fail_on_undefined=False
 259         self.assertEqual(templar.template("{{bad_var}}", fail_on_undefined=False), "{{bad_var}}")
 260 
 261         # test setting available_variables
 262         templar.available_variables = dict(foo="bam")
 263         self.assertEqual(templar.template("{{foo}}"), "bam")
 264         # variables must be a dict() for available_variables setter
 265         # FIXME Use assertRaises() as a context manager (added in 2.7) once we do not run tests on Python 2.6 anymore.
 266         try:
 267             templar.available_variables = "foo=bam"
 268         except AssertionError:
 269             pass
 270         except Exception as e:
 271             self.fail(e)

 273     def test_templar_escape_backslashes(self):
 274         # Rule of thumb: If escape backslashes is True you should end up with
 275         # the same number of backslashes as when you started.
 276         self.assertEqual(self.templar.template("\t{{foo}}", escape_backslashes=True), "\tbar")
 277         self.assertEqual(self.templar.template("\t{{foo}}", escape_backslashes=False), "\tbar")
 278         self.assertEqual(self.templar.template("\\{{foo}}", escape_backslashes=True), "\\bar")
 279         self.assertEqual(self.templar.template("\\{{foo}}", escape_backslashes=False), "\\bar")
 280         self.assertEqual(self.templar.template("\\{{foo + '\t' }}", escape_backslashes=True), "\\bar\t")
 281         self.assertEqual(self.templar.template("\\{{foo + '\t' }}", escape_backslashes=False), "\\bar\t")
 282         self.assertEqual(self.templar.template("\\{{foo + '\\t' }}", escape_backslashes=True), "\\bar\\t")
 283         self.assertEqual(self.templar.template("\\{{foo + '\\t' }}", escape_backslashes=False), "\\bar\t")
 284         self.assertEqual(self.templar.template("\\{{foo + '\\\\t' }}", escape_backslashes=True), "\\bar\\\\t")
 285         self.assertEqual(self.templar.template("\\{{foo + '\\\\t' }}", escape_backslashes=False), "\\bar\\t")

 388 class TestAnsibleContext(BaseTemplar, unittest.TestCase):
 389     def _context(self, variables=None):
 390         variables = variables or {}
 391 
 392         env = AnsibleEnvironment()
 393         context = AnsibleContext(env, parent={}, name='some_context',
 394                                  blocks={})
 395 
 396         for key, value in variables.items():
 397             context.vars[key] = value
 398 
 399         return context
 400 
 401     def test(self):
 402         context = self._context()
 403         self.assertIsInstance(context, AnsibleContext)
 404         self.assertIsInstance(context, Context)
 405 
 406     def test_resolve_unsafe(self):
 407         context = self._context(variables={'some_unsafe_key': wrap_var('some_unsafe_string')})
 408         res = context.resolve('some_unsafe_key')
 409         # self.assertIsInstance(res, AnsibleUnsafe)
 410         self.assertTrue(self.is_unsafe(res),
 411                         'return of AnsibleContext.resolve (%s) was expected to be marked unsafe but was not' % res)
 412 
 413     def test_resolve_unsafe_list(self):
 414         context = self._context(variables={'some_unsafe_key': [wrap_var('some unsafe string 1')]})
 415         res = context.resolve('some_unsafe_key')
 416         # self.assertIsInstance(res[0], AnsibleUnsafe)
 417         self.assertTrue(self.is_unsafe(res),
 418                         'return of AnsibleContext.resolve (%s) was expected to be marked unsafe but was not' % res)
 419 
 420     def test_resolve_unsafe_dict(self):
 421         context = self._context(variables={'some_unsafe_key':

 428     def test_resolve(self):
 429         context = self._context(variables={'some_key': 'some_string'})
 430         res = context.resolve('some_key')

 432         # self.assertNotIsInstance(res, AnsibleUnsafe)
