  87         ''' used for making a tarfile from a role directory '''
  88         # adding directory into a tar file
  89         try:
  90             tar = tarfile.open(output_file, "w:gz")
  91             tar.add(source_dir, arcname=os.path.basename(source_dir))
  92         except AttributeError:  # tarfile obj. has no attribute __exit__ prior to python 2.    7

  94         finally:  # ensuring closure of tarfile obj
