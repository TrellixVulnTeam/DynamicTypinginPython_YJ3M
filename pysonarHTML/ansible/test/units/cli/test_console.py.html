  22 from units.compat import unittest
  23 from units.compat.mock import patch
  24 
  25 from ansible.cli.console import ConsoleCLI
  26 
  27 
  28 class TestConsoleCLI(unittest.TestCase):
  29     def test_parse(self):
  30         cli = ConsoleCLI(['ansible test'])
  31         cli.parse()
  32         self.assertTrue(cli.parser is not None)
  33 
  34     def test_module_args(self):
  35         cli = ConsoleCLI(['ansible test'])
  36         cli.parse()
  37         res = cli.module_args('copy')
  38         self.assertTrue(cli.parser is not None)
  39         self.assertIn('src', res)
  40         self.assertIn('backup', res)
  41         self.assertIsInstance(res, list)
  42 
  43     @patch('ansible.utils.display.Display.display')
  44     def test_helpdefault(self, mock_display):
  45         cli = ConsoleCLI(['ansible test'])
  46         cli.parse()
  47         cli.modules = set(['copy'])
  48         cli.helpdefault('copy')
  49         self.assertTrue(cli.parser is not None)
  50         self.assertTrue(len(mock_display.call_args_list) > 0,
  51                         "display.display should have been called but was not")
