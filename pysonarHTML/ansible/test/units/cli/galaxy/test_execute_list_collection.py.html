  16 def path_exists(path):
  17     if to_native(path) == '/root/.ansible/collections/ansible_collections/sandwiches/ham':
  18         return False
  19     elif to_native(path) == '/usr/share/ansible/collections/ansible_collections/sandwiches/reuben':
  20         return False

  22         return False
  23     else:
  24         return True

 138 def test_execute_list_collection_all(mocker, capsys, mock_collection_objects):
 139     """Test listing all collections from multiple paths"""
 140 
 141     cliargs()
 142 
 143     mocker.patch('os.path.exists', return_value=True)
 144     mocker.patch('os.path.isdir', return_value=True)
 145     gc = GalaxyCLI(['ansible-galaxy', 'collection', 'list'])
 146     gc.execute_list_collection()
 147 
 148     out, err = capsys.readouterr()
 149     out_lines = out.splitlines()
 150 
 151     assert len(out_lines) == 12
 152     assert out_lines[0] == ''
 153     assert out_lines[1] == '# /root/.ansible/collections/ansible_collections'
 154     assert out_lines[2] == 'Collection        Version'
 155     assert out_lines[3] == '----------------- -------'
 156     assert out_lines[4] == 'sandwiches.pbj    1.5.0  '
 157     assert out_lines[5] == 'sandwiches.reuben 2.5.0  '
 158     assert out_lines[6] == ''
 159     assert out_lines[7] == '# /usr/share/ansible/collections/ansible_collections'
 160     assert out_lines[8] == 'Collection     Version'
 161     assert out_lines[9] == '-------------- -------'
 162     assert out_lines[10] == 'sandwiches.ham 1.0.0  '
 163     assert out_lines[11] == 'sandwiches.pbj 1.0.0  '
