  43 def test_extract_tar_dir_exists(mocker, fake_tar_obj):
  44     mocker.patch('os.makedirs', return_value=None)
  45     m_makedir = mocker.patch('os.mkdir', return_value=None)
  46     mocker.patch('os.path.isdir', return_value=True)

  49 
  50     assert not m_makedir.called

  53 def test_extract_tar_dir_does_not_exist(mocker, fake_tar_obj):
