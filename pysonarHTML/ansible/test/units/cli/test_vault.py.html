 192 @pytest.mark.parametrize('cli_args, expected', [
 193     (['ansible-vault', 'view', 'vault.txt'], 0),
 194     (['ansible-vault', 'view', 'vault.txt', '-vvv'], 3),
 195     (['ansible-vault', '-vv', 'view', 'vault.txt'], 2),
 196     # Due to our manual parsing we want to verify that -v set in the sub parser takes precedence. This behaviour is
 197     # deprecated and tests should be removed when the code that handles it is removed
 198     (['ansible-vault', '-vv', 'view', 'vault.txt', '-v'], 1),
 199     (['ansible-vault', '-vv', 'view', 'vault.txt', '-vvvv'], 4),
 200 ])
 201 def test_verbosity_arguments(cli_args, expected, tmp_path_factory, monkeypatch):
 202     # Add a password file so we don't get a prompt in the test
 203     test_dir = to_text(tmp_path_factory.mktemp('test-ansible-vault'))
 204     pass_file = os.path.join(test_dir, 'pass.txt')
 205     with open(pass_file, 'w') as pass_fd:
 206         pass_fd.write('password')
 207 
 208     cli_args.extend(['--vault-id', pass_file])
 209 
 210     # Mock out the functions so we don't actually execute anything
 211     for func_name in [f for f in dir(VaultCLI) if f.startswith("execute_")]:
 212         monkeypatch.setattr(VaultCLI, func_name, MagicMock())
 213 
 214     cli = VaultCLI(args=cli_args)
 215     cli.run()
 216 
 217     assert context.CLIARGS['verbosity'] == expected
