 183     @patch('ansible.cli.get_file_vault_secret')

 185     def test_prompt_no_tty_and_password_file(self, mock_prompt_secret, mock_file_secret):
 186         self.mock_isatty.return_value = False
 187         mock_prompt_secret.return_value = MagicMock(bytes=b'prompt1_password',
 188                                                     vault_id='prompt1')

 205     def _assert_ids(self, vault_id_names, res, password=b'prompt1_password'):
 206         self.assertIsInstance(res, list)
 207         len_ids = len(vault_id_names)
 208         matches = vault.match_secrets(res, vault_id_names)
 209         self.assertEqual(len(res), len_ids, 'len(res):%s does not match len_ids:%s' % (len(res), len_ids))
 210         self.assertEqual(len(matches), len_ids)
 211         for index, prompt in enumerate(vault_id_names):
 212             self.assertIn(prompt, [x[0] for x in matches])
 213             # simple mock, same password/prompt for each mock_prompt_secret
 214             self.assertEqual(matches[index][1].bytes, password)

 246     @patch('ansible.cli.C')
 247     @patch('ansible.cli.get_file_vault_secret')
 248     @patch('ansible.cli.PromptVaultSecret')
 249     def test_default_file_vault(self, mock_prompt_secret,
 250                                 mock_file_secret,
 251                                 mock_config):
 252         mock_prompt_secret.return_value = MagicMock(bytes=b'prompt1_password',
 253                                                     vault_id='default')
 254         mock_file_secret.return_value = MagicMock(bytes=b'file1_password',
 255                                                   vault_id='default')
 256         mock_config.DEFAULT_VAULT_PASSWORD_FILE = '/dev/null/faux/vault_password_file'
 257         mock_config.DEFAULT_VAULT_IDENTITY = 'default'
 258 
 259         res = cli.CLI.setup_vault_secrets(loader=self.fake_loader,
 260                                           vault_ids=[],
 261                                           create_new_password=False,
 262                                           ask_vault_pass=False)
 263 
 264         self.assertIsInstance(res, list)
 265         matches = vault.match_secrets(res, ['default'])
 266         # --vault-password-file/DEFAULT_VAULT_PASSWORD_FILE is higher precendce than prompts
 267         # if the same vault-id ('default') regardless of cli order since it didn't matter in 2.3
 268 
 269         self.assertEqual(matches[0][1].bytes, b'file1_password')
 270         self.assertEqual(len(matches), 1)
 271 
 272         res = cli.CLI.setup_vault_secrets(loader=self.fake_loader,
 273                                           vault_ids=[],
 274                                           create_new_password=False,
 275                                           ask_vault_pass=True,
 276                                           auto_prompt=True)
 277 
 278         self.assertIsInstance(res, list)
 279         matches = vault.match_secrets(res, ['default'])
 280         self.assertEqual(matches[0][1].bytes, b'file1_password')
 281         self.assertEqual(matches[1][1].bytes, b'prompt1_password')
 282         self.assertEqual(len(matches), 2)
