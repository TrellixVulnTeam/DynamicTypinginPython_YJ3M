  25 def test_with_command():
  26     """ Test simple adhoc command"""
  27     module_name = 'command'
  28     adhoc_cli = AdHocCLI(args=['ansible', '-m', module_name, '-vv', 'localhost'])
  29     adhoc_cli.parse()
  30     assert context.CLIARGS['module_name'] == module_name
  31     assert display.verbosity == 2

  60 def test_play_ds_positive():
  61     """ Test _play_ds"""
  62     adhoc_cli = AdHocCLI(args=['/bin/ansible', 'localhost', '-m', 'command'])
  63     adhoc_cli.parse()
  64     ret = adhoc_cli._play_ds('command', 10, 2)
  65     assert ret['name'] == 'Ansible Ad-Hoc'
  66     assert ret['tasks'] == [{'action': {'module': 'command', 'args': {}}, 'async_val': 10, 'poll': 2}]

  78 def test_run_import_playbook():
  79     """ Test import_playbook which is not allowed with ad-hoc command"""
  80     import_playbook = 'import_playbook'
  81     adhoc_cli = AdHocCLI(args=['/bin/ansible', '-m', import_playbook, 'localhost'])
  82     adhoc_cli.parse()
  83     with pytest.raises(AnsibleOptionsError) as exec_info:
  84         adhoc_cli.run()
  85     assert context.CLIARGS['module_name'] == import_playbook
  86     assert "'%s' is not a valid action for ad-hoc commands" % import_playbook == str(exec_info.value)

  96 def test_ansible_version(capsys, mocker):
  97     adhoc_cli = AdHocCLI(args=['/bin/ansible', '--version'])
  98     with pytest.raises(SystemExit):
  99         adhoc_cli.run()
 100     version = capsys.readouterr()
 101     try:
 102         version_lines = version.out.splitlines()
 103     except AttributeError:
 104         # Python 2.6 does return a named tuple, so get the first item
 105         version_lines = version[0].splitlines()
 106 
 107     assert len(version_lines) == 7, 'Incorrect number of lines in "ansible --version" output'
 108     assert re.match('ansible [0-9.a-z]+$', version_lines[0]), 'Incorrect ansible version line in "ansible --version" output'
 109     assert re.match('  config file = .*$', version_lines[1]), 'Incorrect config file line in "ansible --version" output'
 110     assert re.match('  configured module search path = .*$', version_lines[2]), 'Incorrect module search path in "ansible --version" output'
 111     assert re.match('  ansible python module location = .*$', version_lines[3]), 'Incorrect python module location in "ansible --version" output'
 112     assert re.match('  ansible collection location = .*$', version_lines[4]), 'Incorrect collection location in "ansible --version" output'
 113     assert re.match('  executable location = .*$', version_lines[5]), 'Incorrect executable locaction in "ansible --version" output'
 114     assert re.match('  python version = .*$', version_lines[6]), 'Incorrect python version in "ansible --version" output'
