  22 from units.compat import unittest
  23 from units.mock.loader import DictDataLoader

  25 from ansible import context
  26 from ansible.inventory.manager import InventoryManager
  27 from ansible.vars.manager import VariableManager
  28 
  29 from ansible.cli.playbook import PlaybookCLI
  30 
  31 
  32 class TestPlaybookCLI(unittest.TestCase):
  33     def test_flush_cache(self):
  34         cli = PlaybookCLI(args=["ansible-playbook", "--flush-cache", "foobar.yml"])
  35         cli.parse()
  36         self.assertTrue(context.CLIARGS['flush_cache'])
  37 
  38         variable_manager = VariableManager()
  39         fake_loader = DictDataLoader({'foobar.yml': ""})
  40         inventory = InventoryManager(loader=fake_loader, sources='testhost,')
  41 
  42         variable_manager.set_host_facts('testhost', {'canary': True})
  43         self.assertTrue('testhost' in variable_manager._fact_cache)
  44 
  45         cli._flush_cache(inventory, variable_manager)
  46         self.assertFalse('testhost' in variable_manager._fact_cache)
