 100     def test_task_executor_run_clean_res(self):
 101         te = TaskExecutor(None, MagicMock(), None, None, None, None, None, None)
 102         te._get_loop_items = MagicMock(return_value=[1])
 103         te._run_loop = MagicMock(
 104             return_value=[
 105                 {
 106                     'unsafe_bytes': AnsibleUnsafeBytes(b'{{ $bar }}'),
 107                     'unsafe_text': AnsibleUnsafeText(u'{{ $bar }}'),

 109                     'text': u'text',
 110                     'int': 1,
 111                 }

 368     def test_task_executor_poll_async_result(self):
 369         fake_loader = DictDataLoader({})
 370 
 371         mock_host = MagicMock()
 372 
 373         mock_task = MagicMock()
 374         mock_task.async_val = 0.1
 375         mock_task.poll = 0.05
 376 
 377         mock_play_context = MagicMock()
 378 
 379         mock_connection = MagicMock()
 380 
 381         mock_action = MagicMock()
 382         mock_queue = MagicMock()
 383 
 384         shared_loader = MagicMock()
 385         shared_loader.action_loader = action_loader
 386 
 387         new_stdin = None
 388         job_vars = dict(omit="XXXXXXXXXXXXXXXXXXX")
 389 
 390         te = TaskExecutor(
 391             host=mock_host,
 392             task=mock_task,
 393             job_vars=job_vars,
 394             play_context=mock_play_context,
 395             new_stdin=new_stdin,
 396             loader=fake_loader,
 397             shared_loader_obj=shared_loader,
 398             final_q=mock_queue,
 399         )
 400 
 401         te._connection = MagicMock()
 402 
 403         def _get(*args, **kwargs):
 404             mock_action = MagicMock()
 405             mock_action.run.return_value = dict(stdout='')
 406             return mock_action
 407 
 408         # testing with some bad values in the result passed to poll async,
 409         # and with a bad value returned from the mock action
 410         with patch.object(action_loader, 'get', _get):
 411             mock_templar = MagicMock()
 412             res = te._poll_async_result(result=dict(), templar=mock_templar)
 413             self.assertIn('failed', res)
 414             res = te._poll_async_result(result=dict(ansible_job_id=1), templar=mock_templar)
 415             self.assertIn('failed', res)
 416 
 417         def _get(*args, **kwargs):
 418             mock_action = MagicMock()
 419             mock_action.run.return_value = dict(finished=1)
 420             return mock_action
 421 
 422         # now testing with good values
 423         with patch.object(action_loader, 'get', _get):
 424             mock_templar = MagicMock()
 425             res = te._poll_async_result(result=dict(ansible_job_id=1), templar=mock_templar)
 426             self.assertEqual(res, dict(finished=1))

 428     def test_recursive_remove_omit(self):
 429         omit_token = 'POPCORN'
 430 
 431         data = {
 432             'foo': 'bar',
 433             'baz': 1,
 434             'qux': ['one', 'two', 'three'],
 435             'subdict': {
 436                 'remove': 'POPCORN',
 437                 'keep': 'not_popcorn',
 438                 'subsubdict': {
 439                     'remove': 'POPCORN',
 440                     'keep': 'not_popcorn',
 441                 },
 442                 'a_list': ['POPCORN'],
 443             },
 444             'a_list': ['POPCORN'],
 445             'list_of_lists': [
 446                 ['some', 'thing'],
 447             ],
 448             'list_of_dicts': [
 449                 {
 450                     'remove': 'POPCORN',
 451                 }
 452             ],
 453         }
 454 
 455         expected = {
 456             'foo': 'bar',
 457             'baz': 1,
 458             'qux': ['one', 'two', 'three'],
 459             'subdict': {
 460                 'keep': 'not_popcorn',
 461                 'subsubdict': {
 462                     'keep': 'not_popcorn',
 463                 },
 464                 'a_list': ['POPCORN'],
 465             },
 466             'a_list': ['POPCORN'],
 467             'list_of_lists': [
 468                 ['some', 'thing'],
 469             ],
 470             'list_of_dicts': [{}],
 471         }
 472 
 473         self.assertEqual(remove_omit(data, omit_token), expected)
