  40     def test_task_result_is_changed(self):
  41         mock_host = MagicMock()
  42         mock_task = MagicMock()
  43 
  44         # test with no changed in result
  45         tr = TaskResult(mock_host, mock_task, dict())
  46         self.assertFalse(tr.is_changed())
  47 
  48         # test with changed in the result
  49         tr = TaskResult(mock_host, mock_task, dict(changed=True))
  50         self.assertTrue(tr.is_changed())
  51 
  52         # test with multiple results but none changed
  53         mock_task.loop = 'foo'
  54         tr = TaskResult(mock_host, mock_task, dict(results=[dict(foo='bar'), dict(bam='baz'), True]))
  55         self.assertFalse(tr.is_changed())
  56 
  57         # test with multiple results and one changed
  58         mock_task.loop = 'foo'
  59         tr = TaskResult(mock_host, mock_task, dict(results=[dict(changed=False), dict(changed=True), dict(some_key=False)]))
  60         self.assertTrue(tr.is_changed())

  71         tr = TaskResult(mock_host, mock_task, dict(skipped=True))
  72         self.assertTrue(tr.is_skipped())
  73 
  74         # test with multiple results but none skipped
  75         mock_task.loop = 'foo'
  76         tr = TaskResult(mock_host, mock_task, dict(results=[dict(foo='bar'), dict(bam='baz'), True]))
  77         self.assertFalse(tr.is_skipped())

  84         # test with multiple results and all skipped
  85         mock_task.loop = 'foo'
  86         tr = TaskResult(mock_host, mock_task, dict(results=[dict(skipped=True), dict(skipped=True), dict(skipped=True)]))
  87         self.assertTrue(tr.is_skipped())
  88 
  89         # test with multiple squashed results (list of strings)
  90         # first with the main result having skipped=False

  93         self.assertFalse(tr.is_skipped())

  95         tr = TaskResult(mock_host, mock_task, dict(results=["a", "b", "c"], skipped=True))

 120     def test_task_result_is_failed(self):
 121         mock_host = MagicMock()
 122         mock_task = MagicMock()
 123 
 124         # test with no failed in result
 125         tr = TaskResult(mock_host, mock_task, dict())
 126         self.assertFalse(tr.is_failed())
 127 
 128         # test failed result with rc values (should not matter)
 129         tr = TaskResult(mock_host, mock_task, dict(rc=0))
 130         self.assertFalse(tr.is_failed())
 131         tr = TaskResult(mock_host, mock_task, dict(rc=1))
 132         self.assertFalse(tr.is_failed())
 133 
 134         # test with failed in result
 135         tr = TaskResult(mock_host, mock_task, dict(failed=True))
 136         self.assertTrue(tr.is_failed())
 137 
 138         # test with failed_when in result
 139         tr = TaskResult(mock_host, mock_task, dict(failed_when_result=True))
 140         self.assertTrue(tr.is_failed())

 142     def test_task_result_no_log(self):
 143         mock_host = MagicMock()
 144         mock_task = MagicMock()
 145 
 146         # no_log should remove secrets
 147         tr = TaskResult(mock_host, mock_task, dict(_ansible_no_log=True, secret='DONTSHOWME'))
 148         clean = tr.clean_copy()
 149         self.assertTrue('secret' not in clean._result)
 150 
 151     def test_task_result_no_log_preserve(self):
 152         mock_host = MagicMock()
 153         mock_task = MagicMock()
 154 
 155         # no_log should not remove presrved keys
 156         tr = TaskResult(
 157             mock_host,
 158             mock_task,
 159             dict(
 160                 _ansible_no_log=True,
 161                 retries=5,
 162                 attempts=5,
 163                 changed=False,
 164                 foo='bar',
 165             )
 166         )
 167         clean = tr.clean_copy()
 168         self.assertTrue('retries' in clean._result)
 169         self.assertTrue('attempts' in clean._result)
 170         self.assertTrue('changed' in clean._result)
 171         self.assertTrue('foo' not in clean._result)
