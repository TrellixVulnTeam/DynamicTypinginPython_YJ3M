  22 def test_discovery_interpreter_linux_auto_legacy():
  23     res1 = u'PLATFORM\nLinux\nFOUND\n/usr/bin/python\n/usr/bin/python3.5\n/usr/bin/python3\nENDFOUND'
  24 
  25     mock_action = MagicMock()
  26     mock_action._low_level_execute_command.side_effect = [{'stdout': res1}, {'stdout': mock_ubuntu_platform_res}]
  27 
  28     actual = discover_interpreter(mock_action, 'python', 'auto_legacy', {'inventory_hostname': u'host-fóöbär'})
  29 
  30     assert actual == u'/usr/bin/python'
  31     assert len(mock_action.method_calls) == 3
  32     assert mock_action.method_calls[2][0] == '_discovery_deprecation_warnings.append'
  33     assert u'Distribution Ubuntu 16.04 on host host-fóöbär should use /usr/bin/python3, but is using /usr/bin/python' \
  34            u' for backward compatibility' in mock_action.method_calls[2][1][0]['msg']
  35     assert mock_action.method_calls[2][1][0]['version'] == '2.12'

  62 def test_discovery_interpreter_non_linux():
  63     mock_action = MagicMock()
  64     mock_action._low_level_execute_command.return_value = \
  65         {'stdout': u'PLATFORM\nDarwin\nFOUND\n/usr/bin/python\nENDFOUND'}
  66 
  67     actual = discover_interpreter(mock_action, 'python', 'auto_legacy', {'inventory_hostname': u'host-fóöbär'})
  68 
  69     assert actual == u'/usr/bin/python'
  70     assert len(mock_action.method_calls) == 2
  71     assert mock_action.method_calls[1][0] == '_discovery_warnings.append'
  72     assert u'Platform darwin on host host-fóöbär is using the discovered Python interpreter at /usr/bin/python, ' \
  73            u'but future installation of another Python interpreter could change the meaning of that path' \
  74            in mock_action.method_calls[1][1][0]
