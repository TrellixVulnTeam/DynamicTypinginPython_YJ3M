  33 class TestStripComments:
  34     def test_no_changes(self):
  35         no_comments = u"""def some_code():
  36     return False"""
  37         assert amc._strip_comments(no_comments) == no_comments
  38 
  39     def test_all_comments(self):
  40         all_comments = u"""# This is a test
  41             # Being as it is
  42             # To be
  43             """
  44         assert amc._strip_comments(all_comments) == u""
  45 
  46     def test_all_whitespace(self):
  47         # Note: Do not remove the spaces on the blank lines below.  They're
  48         # test data to show that the lines get removed despite having spaces
  49         # on them
  50         all_whitespace = u"""
  51               
  52 
  53                 
  54 \t\t\r\n
  55             """  # nopep8
  56         assert amc._strip_comments(all_whitespace) == u""
  57 
  58     def test_somewhat_normal(self):
  59         mixed = u"""#!/usr/bin/python
  60 
  61 # here we go
  62 def test(arg):
  63     # this is a thing
  64     thing = '# test'
  65     return thing
  66 # End
  67 """
  68         mixed_results = u"""def test(arg):
  69     thing = '# test'
  70     return thing"""
  71         assert amc._strip_comments(mixed) == mixed_results

  74 class TestSlurp:
  75     def test_slurp_nonexistent(self, mocker):
  76         mocker.patch('os.path.exists', side_effect=lambda x: False)
  77         with pytest.raises(ansible.errors.AnsibleError):
  78             amc._slurp('no_file')
  79 
  80     def test_slurp_file(self, mocker):
  81         mocker.patch('os.path.exists', side_effect=lambda x: True)
  82         m = mocker.mock_open(read_data='This is a test')
  83         if PY2:
  84             mocker.patch('__builtin__.open', m)
  85         else:
  86             mocker.patch('builtins.open', m)
  87         assert amc._slurp('some_file') == 'This is a test'
  88 
  89     def test_slurp_file_with_newlines(self, mocker):
  90         mocker.patch('os.path.exists', side_effect=lambda x: True)
  91         m = mocker.mock_open(read_data='#!/usr/bin/python\ndef test(args):\nprint("hi")\n')
  92         if PY2:
  93             mocker.patch('__builtin__.open', m)
  94         else:
  95             mocker.patch('builtins.open', m)
  96         assert amc._slurp('some_file') == '#!/usr/bin/python\ndef test(args):\nprint("hi")\n'

 108 class TestGetShebang:
 109     """Note: We may want to change the API of this function in the future.  It isn't a great API"""
 110     def test_no_interpreter_set(self, templar):
 111         # normally this would return /usr/bin/python, but so long as we're defaulting to auto python discovery, we'll get
 112         # an InterpreterDiscoveryRequiredError here instead
 113         with pytest.raises(InterpreterDiscoveryRequiredError):
 114             amc._get_shebang(u'/usr/bin/python', {}, templar)
 115 
 116     def test_non_python_interpreter(self, templar):
 117         assert amc._get_shebang(u'/usr/bin/ruby', {}, templar) == (None, u'/usr/bin/ruby')
 118 
 119     def test_interpreter_set_in_task_vars(self, templar):
 120         assert amc._get_shebang(u'/usr/bin/python', {u'ansible_python_interpreter': u'/usr/bin/pypy'}, templar) == \
 121             (u'#!/usr/bin/pypy', u'/usr/bin/pypy')
 122 
 123     def test_non_python_interpreter_in_task_vars(self, templar):
 124         assert amc._get_shebang(u'/usr/bin/ruby', {u'ansible_ruby_interpreter': u'/usr/local/bin/ruby'}, templar) == \
 125             (u'#!/usr/local/bin/ruby', u'/usr/local/bin/ruby')
 126 
 127     def test_with_args(self, templar):
 128         assert amc._get_shebang(u'/usr/bin/python', {u'ansible_python_interpreter': u'/usr/bin/python3'}, templar, args=('-tt', '-OO')) == \
 129             (u'#!/usr/bin/python3 -tt -OO', u'/usr/bin/python3')
 130 
 131     def test_python_via_env(self, templar):
 132         assert amc._get_shebang(u'/usr/bin/python', {u'ansible_python_interpreter': u'/usr/bin/env python'}, templar) == \
 133             (u'#!/usr/bin/env python', u'/usr/bin/env python')

 136 class TestDetectionRegexes:
 137     ANSIBLE_MODULE_UTIL_STRINGS = (
 138         # Absolute collection imports
 139         b'import ansible_collections.my_ns.my_col.plugins.module_utils.my_util',
 140         b'from ansible_collections.my_ns.my_col.plugins.module_utils import my_util',
 141         b'from ansible_collections.my_ns.my_col.plugins.module_utils.my_util import my_func',
 142         # Absolute core imports
 143         b'import ansible.module_utils.basic',
 144         b'from ansible.module_utils import basic',
 145         b'from ansible.module_utils.basic import AnsibleModule',
 146         # Relative imports
 147         b'from ..module_utils import basic',
 148         b'from .. module_utils import basic',
 149         b'from ....module_utils import basic',
 150         b'from ..module_utils.basic import AnsibleModule',
 151     )
 152     NOT_ANSIBLE_MODULE_UTIL_STRINGS = (
 153         b'from ansible import release',
 154         b'from ..release import __version__',
 155         b'from .. import release',
 156         b'from ansible.modules.system import ping',
 157         b'from ansible_collecitons.my_ns.my_col.plugins.modules import function',
 158     )
 159 
 160     OFFSET = os.path.dirname(os.path.dirname(amc.__file__))
 161     CORE_PATHS = (
 162         ('%s/modules/from_role.py' % OFFSET, 'ansible/modules/from_role'),
 163         ('%s/modules/system/ping.py' % OFFSET, 'ansible/modules/system/ping'),
 164         ('%s/modules/cloud/amazon/s3.py' % OFFSET, 'ansible/modules/cloud/amazon/s3'),
 165     )
 166 
 167     COLLECTION_PATHS = (
 168         ('/root/ansible_collections/ns/col/plugins/modules/ping.py',
 169          'ansible_collections/ns/col/plugins/modules/ping'),
 170         ('/root/ansible_collections/ns/col/plugins/modules/subdir/ping.py',
 171          'ansible_collections/ns/col/plugins/modules/subdir/ping'),
 172     )
 173 
 174     @pytest.mark.parametrize('testcase', ANSIBLE_MODULE_UTIL_STRINGS)
 175     def test_detect_new_style_python_module_re(self, testcase):
 176         assert amc.NEW_STYLE_PYTHON_MODULE_RE.search(testcase)
 177 
 178     @pytest.mark.parametrize('testcase', NOT_ANSIBLE_MODULE_UTIL_STRINGS)
 179     def test_no_detect_new_style_python_module_re(self, testcase):
 180         assert not amc.NEW_STYLE_PYTHON_MODULE_RE.search(testcase)
 181 
 182     # pylint bug: https://github.com/PyCQA/pylint/issues/511
 183     @pytest.mark.parametrize('testcase, result', CORE_PATHS)  # pylint: disable=undefined-variable
 184     def test_detect_core_library_path_re(self, testcase, result):
 185         assert amc.CORE_LIBRARY_PATH_RE.search(testcase).group('path') == result
 186 
 187     @pytest.mark.parametrize('testcase', (p[0] for p in COLLECTION_PATHS))  # pylint: disable=undefined-variable
 188     def test_no_detect_core_library_path_re(self, testcase):
 189         assert not amc.CORE_LIBRARY_PATH_RE.search(testcase)
 190 
 191     @pytest.mark.parametrize('testcase, result', COLLECTION_PATHS)  # pylint: disable=undefined-variable
 192     def test_detect_collection_path_re(self, testcase, result):
 193         assert amc.COLLECTION_PATH_RE.search(testcase).group('path') == result
 194 
 195     @pytest.mark.parametrize('testcase', (p[0] for p in CORE_PATHS))  # pylint: disable=undefined-variable
 196     def test_no_detect_collection_path_re(self, testcase):
 197         assert not amc.COLLECTION_PATH_RE.search(testcase)
