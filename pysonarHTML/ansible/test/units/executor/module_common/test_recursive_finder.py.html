 121 class TestRecursiveFinder(object):
 122     def test_no_module_utils(self, finder_containers):
 123         name = 'ping'
 124         data = b'#!/usr/bin/python\nreturn \'{\"changed\": false}\''
 125         recursive_finder(name, os.path.join(ANSIBLE_LIB, 'modules', 'system', 'ping.py'), data, *finder_containers)
 126         assert finder_containers.py_module_names == set(()).union(MODULE_UTILS_BASIC_IMPORTS)
 127         assert finder_containers.py_module_cache == {}
 128         assert frozenset(finder_containers.zf.namelist()) == MODULE_UTILS_BASIC_FILES
 129 
 130     def test_module_utils_with_syntax_error(self, finder_containers):
 131         name = 'fake_module'
 132         data = b'#!/usr/bin/python\ndef something(:\n   pass\n'
 133         with pytest.raises(ansible.errors.AnsibleError) as exec_info:
 134             recursive_finder(name, os.path.join(ANSIBLE_LIB, 'modules', 'system', 'fake_module.py'), data, *finder_containers)
 135         assert 'Unable to import fake_module due to invalid syntax' in str(exec_info.value)
 136 
 137     def test_module_utils_with_identation_error(self, finder_containers):
 138         name = 'fake_module'
 139         data = b'#!/usr/bin/python\n    def something():\n    pass\n'
 140         with pytest.raises(ansible.errors.AnsibleError) as exec_info:
 141             recursive_finder(name, os.path.join(ANSIBLE_LIB, 'modules', 'system', 'fake_module.py'), data, *finder_containers)
 142         assert 'Unable to import fake_module due to unexpected indent' in str(exec_info.value)

 144     def test_from_import_toplevel_package(self, finder_containers, mocker):
 145         if PY2:
 146             module_utils_data = b'# License\ndef do_something():\n    pass\n'
 147         else:
 148             module_utils_data = u'# License\ndef do_something():\n    pass\n'
 149         mi_mock = mocker.patch('ansible.executor.module_common.ModuleInfo')
 150         mi_inst = mi_mock()
 151         mi_inst.pkg_dir = True
 152         mi_inst.py_src = False
 153         mi_inst.path = '/path/to/ansible/module_utils/foo/__init__.py'
 154         mi_inst.get_source.return_value = module_utils_data
 155 
 156         name = 'ping'
 157         data = b'#!/usr/bin/python\nfrom ansible.module_utils import foo'
 158         recursive_finder(name, os.path.join(ANSIBLE_LIB, 'modules', 'system', 'ping.py'), data, *finder_containers)
 159         mocker.stopall()
 160 
 161         assert finder_containers.py_module_names == set((('ansible', 'module_utils', 'foo', '__init__'),)).union(ONLY_BASIC_IMPORT)
 162         assert finder_containers.py_module_cache == {}
 163         assert frozenset(finder_containers.zf.namelist()) == frozenset(('ansible/module_utils/foo/__init__.py',)).union(ONLY_BASIC_FILE)
