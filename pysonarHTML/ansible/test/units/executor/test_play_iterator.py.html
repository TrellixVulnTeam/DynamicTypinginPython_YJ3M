 374     def test_play_iterator_add_tasks(self):
 375         fake_loader = DictDataLoader({
 376             'test_play.yml': """
 377             - hosts: all
 378               gather_facts: no
 379               tasks:
 380               - debug: msg="dummy task"
 381             """,
 382         })
 383 
 384         mock_var_manager = MagicMock()
 385         mock_var_manager._fact_cache = dict()
 386         mock_var_manager.get_vars.return_value = dict()
 387 
 388         p = Playbook.load('test_play.yml', loader=fake_loader, variable_manager=mock_var_manager)
 389 
 390         hosts = []
 391         for i in range(0, 10):
 392             host = MagicMock()
 393             host.name = host.get_name.return_value = 'host%02d' % i
 394             hosts.append(host)
 395 
 396         inventory = MagicMock()
 397         inventory.get_hosts.return_value = hosts
 398         inventory.filter_hosts.return_value = hosts
 399 
 400         play_context = PlayContext(play=p._entries[0])
 401 
 402         itr = PlayIterator(
 403             inventory=inventory,
 404             play=p._entries[0],
 405             play_context=play_context,
 406             variable_manager=mock_var_manager,
 407             all_vars=dict(),
 408         )
 409 
 410         # test the high-level add_tasks() method
 411         s = HostState(blocks=[0, 1, 2])
 412         itr._insert_tasks_into_state = MagicMock(return_value=s)
 413         itr.add_tasks(hosts[0], [MagicMock(), MagicMock(), MagicMock()])
 414         self.assertEqual(itr._host_states[hosts[0].name], s)
 415 
 416         # now actually test the lower-level method that does the work
 417         itr = PlayIterator(
 418             inventory=inventory,
 419             play=p._entries[0],
 420             play_context=play_context,
 421             variable_manager=mock_var_manager,
 422             all_vars=dict(),
 423         )
 424 
 425         # iterate past first task
 426         _, task = itr.get_next_task_for_host(hosts[0])
 427         while(task and task.action != 'debug'):
 428             _, task = itr.get_next_task_for_host(hosts[0])
 429 
 430         if task is None:
 431             raise Exception("iterated past end of play while looking for place to insert tasks")
 432 
 433         # get the current host state and copy it so we can mutate it
 434         s = itr.get_host_state(hosts[0])
 435         s_copy = s.copy()
 436 
 437         # assert with an empty task list, or if we're in a failed state, we simply return the state as-is
 438         res_state = itr._insert_tasks_into_state(s_copy, task_list=[])
 439         self.assertEqual(res_state, s_copy)
 440 
 441         s_copy.fail_state = itr.FAILED_TASKS
 442         res_state = itr._insert_tasks_into_state(s_copy, task_list=[MagicMock()])
 443         self.assertEqual(res_state, s_copy)
 444 
 445         # but if we've failed with a rescue/always block
 446         mock_task = MagicMock()
 447         s_copy.run_state = itr.ITERATING_RESCUE
 448         res_state = itr._insert_tasks_into_state(s_copy, task_list=[mock_task])
 449         self.assertEqual(res_state, s_copy)
 450         self.assertIn(mock_task, res_state._blocks[res_state.cur_block].rescue)
 451         itr._host_states[hosts[0].name] = res_state
 452         (next_state, next_task) = itr.get_next_task_for_host(hosts[0], peek=True)
 453         self.assertEqual(next_task, mock_task)
 454         itr._host_states[hosts[0].name] = s
 455 
 456         # test a regular insertion
 457         s_copy = s.copy()
 458         res_state = itr._insert_tasks_into_state(s_copy, task_list=[MagicMock()])
