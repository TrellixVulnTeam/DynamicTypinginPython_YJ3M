 158                              [[('/etc/ansible/ansible.cfg', cfg_in_homedir, cfg_file, alt_cfg_file)]],
 159                              indirect=['setup_existing_files'])
 160     def test_no_cwd_cfg_no_warning_on_writable(self, setup_env, setup_existing_files, monkeypatch):
 161         """If the cwd is writable but there is no config file there, move on with no warning"""
 162         real_stat = os.stat
 163 
 164         def _os_stat(path):
 165             if path == working_dir:
 166                 from posix import stat_result
 167                 stat_info = list(real_stat(path))
 168                 stat_info[stat.ST_MODE] |= stat.S_IWOTH
 169                 return stat_result(stat_info)
 170             else:
 171                 return real_stat(path)
 172 
 173         monkeypatch.setattr('os.stat', _os_stat)
 174 
 175         warnings = set()
 176         assert find_ini_config_file(warnings) == cfg_in_homedir
 177         assert len(warnings) == 0
