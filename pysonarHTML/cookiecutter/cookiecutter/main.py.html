   1 """
   2 Main entry point for the `cookiecutter` command.
   3 
   4 The code in this module is also a good example of how to use Cookiecutter as a
   5 library rather than a script.
   6 """
   7 import logging
   8 import os
   9 
  10 from cookiecutter.config import get_user_config
  11 from cookiecutter.exceptions import InvalidModeException
  12 from cookiecutter.generate import generate_context, generate_files
  13 from cookiecutter.prompt import prompt_for_config
  14 from cookiecutter.replay import dump, load
  15 from cookiecutter.repository import determine_repo_dir
  16 from cookiecutter.utils import rmtree
  17 
  18 logger = logging.getLogger(__name__)
  19 
  20 
  21 def cookiecutter(
  22     template,
  23     checkout=None,
  24     no_input=False,
  25     extra_context=None,
  26     replay=None,
  27     overwrite_if_exists=False,
  28     output_dir='.',
  29     config_file=None,
  30     default_config=False,
  31     password=None,
  32     directory=None,
  33     skip_if_file_exists=False,
  34     accept_hooks=True,
  35 ):
  36     """
  37     Run Cookiecutter just as if using it from the command line.
  38 
  39     :param template: A directory containing a project template directory,
  40         or a URL to a git repository.
  41     :param checkout: The branch, tag or commit ID to checkout after clone.
  42     :param no_input: Prompt the user at command line for manual configuration?
  43     :param extra_context: A dictionary of context that overrides default
  44         and user configuration.
  45     :param replay: Do not prompt for input, instead read from saved json. If
  46         ``True`` read from the ``replay_dir``.
  47         if it exists
  48     :param output_dir: Where to output the generated project dir into.
  49     :param config_file: User configuration file path.
  50     :param default_config: Use default values rather than a config file.
  51     :param password: The password to use when extracting the repository.
  52     :param directory: Relative path to a cookiecutter template in a repository.
  53     :param accept_hooks: Accept pre and post hooks if set to `True`.
  54     """
  55     if replay and ((no_input is not False) or (extra_context is not None)):
  56         err_msg = (
  57             "You can not use both replay and no_input or extra_context "
  58             "at the same time."
  59         )
  60         raise InvalidModeException(err_msg)
  61 
  62     config_dict = get_user_config(
  63         config_file=config_file, default_config=default_config,
  64     )
  65 
  66     repo_dir, cleanup = determine_repo_dir(
  67         template=template,
  68         abbreviations=config_dict['abbreviations'],
  69         clone_to_dir=config_dict['cookiecutters_dir'],
  70         checkout=checkout,
  71         no_input=no_input,
  72         password=password,
  73         directory=directory,
  74     )
  75 
  76     template_name = os.path.basename(os.path.abspath(repo_dir))
  77 
  78     if replay:
  79         if isinstance(replay, bool):
  80             context = load(config_dict['replay_dir'], template_name)
  81         else:
  82             path, template_name = os.path.split(os.path.splitext(replay)[0])
  83             context = load(path, template_name)
  84     else:
  85         context_file = os.path.join(repo_dir, 'cookiecutter.json')
  86         logger.debug('context_file is %s', context_file)
  87 
  88         context = generate_context(
  89             context_file=context_file,
  90             default_context=config_dict['default_context'],
  91             extra_context=extra_context,
  92         )
  93 
  94         # prompt the user to manually configure at the command line.
  95         # except when 'no-input' flag is set
  96         context['cookiecutter'] = prompt_for_config(context, no_input)
  97 
  98         # include template dir or url in the context dict
  99         context['cookiecutter']['_template'] = template
 100 
 101         # include output+dir in the context dict
 102         context['cookiecutter']['_output_dir'] = os.path.abspath(output_dir)
 103 
 104         dump(config_dict['replay_dir'], template_name, context)
 105 
 106     # Create project from local context and project template.
 107     result = generate_files(
 108         repo_dir=repo_dir,
 109         context=context,
 110         overwrite_if_exists=overwrite_if_exists,
 111         skip_if_file_exists=skip_if_file_exists,
 112         output_dir=output_dir,
 113         accept_hooks=accept_hooks,
 114     )
 115 
 116     # Cleanup (if required)
 117     if cleanup:
 118         rmtree(repo_dir)
 119 
 120     return result
