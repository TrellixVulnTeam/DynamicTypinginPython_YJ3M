   4 class CookiecutterException(Exception):
   5     """
   6     Base exception class.
   7 
   8     All Cookiecutter-specific exceptions should subclass this class.
   9     """

  12 class NonTemplatedInputDirException(CookiecutterException):
  13     """
  14     Exception for when a project's input dir is not templated.
  15 
  16     The name of the input directory should always contain a string that is
  17     rendered to something else, so that input_dir != output_dir.
  18     """
  19 
  20 
  21 class UnknownTemplateDirException(CookiecutterException):
  22     """
  23     Exception for ambiguous project template directory.
  24 
  25     Raised when Cookiecutter cannot determine which directory is the project
  26     template, e.g. more than one dir appears to be a template dir.
  27     """

 110 class UndefinedVariableInTemplate(CookiecutterException):
 111     """
 112     Exception for out-of-scope variables.
 113 
 114     Raised when a template uses a variable which is not defined in the
 115     context.
 116     """
 117 
 118     def __init__(self, message, error, context):
 119         """Exception for out-of-scope variables."""
 120         self.message = message
 121         self.error = error
 122         self.context = context
 123 
 124     def __str__(self):
 125         """Text representation of UndefinedVariableInTemplate."""
 126         return (
 127             "{self.message}. "
 128             "Error message: {self.error.message}. "
 129             "Context: {self.context}"
 130         ).format(**locals())
