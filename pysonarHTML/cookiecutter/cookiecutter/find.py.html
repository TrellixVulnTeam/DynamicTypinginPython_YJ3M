   1 """Functions for finding Cookiecutter templates and other components."""
   2 import logging
   3 import os
   4 
   5 from cookiecutter.exceptions import NonTemplatedInputDirException
   6 
   7 logger = logging.getLogger(__name__)
   8 
   9 
  10 def find_template(repo_dir):
  11     """Determine which child directory of `repo_dir` is the project template.
  12 
  13     :param repo_dir: Local directory of newly cloned repo.
  14     :returns project_template: Relative path to project template.
  15     """
  16     logger.debug('Searching %s for the project template.', repo_dir)
  17 
  18     repo_dir_contents = os.listdir(repo_dir)
  19 
  20     project_template = None
  21     for item in repo_dir_contents:
  22         if 'cookiecutter' in item and '{{' in item and '}}' in item:
  23             project_template = item
  24             break
  25 
  26     if project_template:
  27         project_template = os.path.join(repo_dir, project_template)
  28         logger.debug('The project template appears to be %s', project_template)
  29         return project_template
  30     else:
  31         raise NonTemplatedInputDirException
