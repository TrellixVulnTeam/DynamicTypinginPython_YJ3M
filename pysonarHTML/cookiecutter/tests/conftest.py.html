  17 def backup_dir(original_dir, backup_dir):
  18     """Generate backup directory based on original directory."""
  19     # If the default original_dir is pre-existing, move it to a temp location
  20     if not os.path.isdir(original_dir):
  21         return False
  22 
  23     # Remove existing backups before backing up. If they exist, they're stale.
  24     if os.path.isdir(backup_dir):
  25         utils.rmtree(backup_dir)
  26 
  27     shutil.copytree(original_dir, backup_dir)
  28     return True
  29

  31 def restore_backup_dir(original_dir, backup_dir, original_dir_found):
  32     """Restore default contents."""
  33     # Carefully delete the created original_dir only in certain
  34     # conditions.
  35     original_dir_is_dir = os.path.isdir(original_dir)
  36     if original_dir_found:
  37         # Delete the created original_dir as long as a backup
  38         # exists
  39         if original_dir_is_dir and os.path.isdir(backup_dir):
  40             utils.rmtree(original_dir)
  41     else:
  42         # Delete the created original_dir.
  43         # There's no backup because it never existed
  44         if original_dir_is_dir:
  45             utils.rmtree(original_dir)
  46 
  47     # Restore the user's default original_dir contents
  48     if os.path.isdir(backup_dir):
  49         shutil.copytree(backup_dir, original_dir)
  50     if os.path.isdir(original_dir):
  51         utils.rmtree(backup_dir)
