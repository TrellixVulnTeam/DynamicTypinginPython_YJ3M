  20     @pytest.mark.parametrize(
  21         'raw_var, rendered_var',
  22         [
  23             (1, '1'),

  30     def test_convert_to_str(self, mocker, raw_var, rendered_var):
  31         """Verify simple items correctly rendered to strings."""
  32         env = environment.StrictEnvironment()
  33         from_string = mocker.patch(
  34             'cookiecutter.prompt.StrictEnvironment.from_string', wraps=env.from_string
  35         )
  36         context = {'project': 'foobar'}
  37 
  38         result = prompt.render_variable(env, raw_var, context)
  39         assert result == rendered_var
  40 
  41         # Make sure that non None non str variables are converted beforehand
  42         if raw_var is not None:
  43             if not isinstance(raw_var, str):
  44                 raw_var = str(raw_var)
  45             from_string.assert_called_once_with(raw_var)
  46         else:
  47             assert not from_string.called

 198     def test_should_render_private_variables_with_two_underscores(self):
 199         """Test rendering of private variables with two underscores.
 200 
 201         There are three cases:
 202         1. Variables beginning with a single underscore are private and not rendered.
 203         2. Variables beginning with a double underscore are private and are rendered.
 204         3. Variables beginning with anything other than underscores are not private and
 205            are rendered.
 206         """
 207         context = {
 208             'cookiecutter': OrderedDict(
 209                 [
 210                     ('foo', 'Hello world'),
 211                     ('bar', 123),
 212                     ('rendered_foo', u'{{ cookiecutter.foo|lower }}'),
 213                     ('rendered_bar', 123),
 214                     ('_hidden_foo', u'{{ cookiecutter.foo|lower }}'),
 215                     ('_hidden_bar', 123),
 216                     ('__rendered_hidden_foo', u'{{ cookiecutter.foo|lower }}'),
 217                     ('__rendered_hidden_bar', 123),
 218                 ]
 219             )
 220         }
 221         cookiecutter_dict = prompt.prompt_for_config(context, no_input=True)
 222         assert cookiecutter_dict == OrderedDict(
 223             [
 224                 ('foo', 'Hello world'),
 225                 ('bar', '123'),
 226                 ('rendered_foo', 'hello world'),
 227                 ('rendered_bar', '123'),
 228                 ('_hidden_foo', u'{{ cookiecutter.foo|lower }}'),
 229                 ('_hidden_bar', 123),
 230                 ('__rendered_hidden_foo', 'hello world'),
 231                 ('__rendered_hidden_bar', '123'),
 232             ]
 233         )
