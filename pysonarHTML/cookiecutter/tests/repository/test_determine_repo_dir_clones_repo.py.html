   9 @pytest.mark.parametrize(
  10     'template, is_url',
  11     [
  12         ('/path/to/zipfile.zip', False),

  17 def test_zipfile_unzip(mocker, template, is_url, user_config_data):
  18     """Verify zip files correctly handled for different source locations.
  19 
  20     `unzip()` should be called with correct args when `determine_repo_dir()`
  21     is passed a zipfile, or a URL to a zipfile.
  22     """
  23     mock_clone = mocker.patch(
  24         'cookiecutter.repository.unzip',
  25         return_value='tests/fake-repo-tmpl',
  26         autospec=True,
  27     )
  28 
  29     project_dir, cleanup = repository.determine_repo_dir(
  30         template,
  31         abbreviations={},
  32         clone_to_dir=user_config_data['cookiecutters_dir'],
  33         checkout=None,
  34         no_input=True,
  35         password=None,
  36     )
  37 
  38     mock_clone.assert_called_once_with(
  39         zip_uri=template,
  40         is_url=is_url,
  41         clone_to_dir=user_config_data['cookiecutters_dir'],
  42         no_input=True,
  43         password=None,
  44     )
  45 
  46     assert os.path.isdir(project_dir)
  47     assert cleanup
  48     assert 'tests/fake-repo-tmpl' == project_dir
  49

  92 def test_repository_url_with_no_context_file(mocker, template_url, user_config_data):
  93     """Verify cloned repository without `cookiecutter.json` file raises error."""
  94     mocker.patch(
  95         'cookiecutter.repository.clone',
  96         return_value='tests/fake-repo-bad',
  97         autospec=True,
  98     )
  99 
 100     with pytest.raises(exceptions.RepositoryNotFound) as err:
 101         repository.determine_repo_dir(
 102             template_url,
 103             abbreviations={},
 104             clone_to_dir=None,
 105             checkout=None,
 106             no_input=True,
 107         )
 108 
 109     assert str(err.value) == (
 110         'A valid repository for "{}" could not be found in the following '
 111         'locations:\n{}'.format(template_url, 'tests/fake-repo-bad')
