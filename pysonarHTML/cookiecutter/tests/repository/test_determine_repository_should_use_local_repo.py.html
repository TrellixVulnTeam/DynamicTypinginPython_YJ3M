   1 """Tests around using locally cached cookiecutter template repositories."""
   2 
   3 import os
   4 
   5 import pytest
   6 
   7 from cookiecutter import repository, exceptions
   8 
   9 
  10 def test_finds_local_repo(tmpdir):
  11     """A valid local repository should be returned."""
  12     project_dir, cleanup = repository.determine_repo_dir(
  13         'tests/fake-repo',
  14         abbreviations={},
  15         clone_to_dir=str(tmpdir),
  16         checkout=None,
  17         no_input=True,
  18     )
  19 
  20     assert 'tests/fake-repo' == project_dir
  21     assert not cleanup
  22 
  23 
  24 def test_local_repo_with_no_context_raises(tmpdir):
  25     """A local repository without a cookiecutter.json should raise a \
  26     `RepositoryNotFound` exception."""
  27     template_path = os.path.join('tests', 'fake-repo-bad')
  28     with pytest.raises(exceptions.RepositoryNotFound) as err:
  29         repository.determine_repo_dir(
  30             template_path,
  31             abbreviations={},
  32             clone_to_dir=str(tmpdir),
  33             checkout=None,
  34             no_input=True,
  35         )
  36 
  37     assert str(err.value) == (
  38         'A valid repository for "{}" could not be found in the following '
  39         'locations:\n{}'.format(
  40             template_path,
  41             '\n'.join([template_path, str(tmpdir / 'tests/fake-repo-bad')]),
  42         )
  43     )
  44

  46 def test_local_repo_typo(tmpdir):

  48     exception."""
