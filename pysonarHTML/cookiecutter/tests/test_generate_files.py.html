 239 def test_generate_files_with_overwrite_if_exists_with_skip_if_file_exists(tmp_path):

 242     simple_with_new_line_file = Path(tmp_path, 'inputpizzä/simple-with-newline.txt')

 246         f.write('temp')

 248     generate.generate_files(

 252         skip_if_file_exists=True,
 253         output_dir=tmp_path,
 254     )
 255 
 256     assert Path(simple_file).is_file()
 257     assert Path(simple_file).exists()
 258     assert Path(simple_with_new_line_file).is_file()
 259     assert Path(simple_with_new_line_file).exists()
 260 
 261     simple_text = open(simple_file, 'rt', encoding='utf-8').read()
 262     assert simple_text == 'temp'
