  42 def test_run_json_load(
  43     mocker, mock_user_config, template_name, context, replay_test_dir, replay_file
  44 ):
  45     """Test that replay.load runs json.load under the hood and that the context \
  46     is correctly loaded from the file in replay_dir."""
  47     spy_get_replay_file = mocker.spy(replay, 'get_file_name')
  48 
  49     mock_json_load = mocker.patch('json.load', side_effect=json.load)
  50 
  51     loaded_context = replay.load(replay_test_dir, template_name)
  52 
  53     assert not mock_user_config.called
  54     spy_get_replay_file.assert_called_once_with(replay_test_dir, template_name)
  55 
  56     assert mock_json_load.call_count == 1
  57     (infile_handler,), kwargs = mock_json_load.call_args
  58     assert infile_handler.name == replay_file
  59     assert loaded_context == context
