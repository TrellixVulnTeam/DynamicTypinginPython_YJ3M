   1 """Collection of tests around loading extensions."""
   2 import pytest
   3 
   4 from cookiecutter.environment import StrictEnvironment

   7 
   8 def test_env_should_raise_for_unknown_extension():
   9     """Test should raise if extension not installed in system."""
  10     context = {'cookiecutter': {'_extensions': ['foobar']}}
  11 
  12     with pytest.raises(UnknownExtension) as err:
  13         StrictEnvironment(context=context, keep_trailing_newline=True)
  14 
  15     assert 'Unable to load extension: ' in str(err.value)
  16 
  17

  24     assert 'cookiecutter.extensions.SlugifyExtension' in env.extensions
