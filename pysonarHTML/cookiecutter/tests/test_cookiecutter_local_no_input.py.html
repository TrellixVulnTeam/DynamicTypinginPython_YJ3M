  33 @pytest.mark.parametrize('path', ['tests/fake-repo-pre/', 'tests/fake-repo-pre'])
  34 @pytest.mark.usefixtures('clean_system', 'remove_additional_dirs')
  35 def test_cookiecutter_no_input_return_project_dir(path):
  36     """Verify `cookiecutter` create project dir on input with or without slash."""
  37     project_dir = main.cookiecutter(path, no_input=True)
  38     assert os.path.isdir('tests/fake-repo-pre/{{cookiecutter.repo_name}}')
  39     assert not os.path.isdir('tests/fake-repo-pre/fake-project')
  40     assert os.path.isdir(project_dir)
  41     assert os.path.isfile('fake-project/README.rst')
  42     assert not os.path.exists('fake-project/json/')

 125 @pytest.mark.usefixtures('clean_system', 'remove_additional_dirs')
 126 def test_cookiecutter_template_cleanup(mocker):
 127     """Verify temporary folder for zip unpacking dropped."""
 128     mocker.patch('tempfile.mkdtemp', return_value='fake-tmp', autospec=True)
 129 
 130     mocker.patch(
 131         'cookiecutter.utils.prompt_and_delete', return_value=True, autospec=True
 132     )
 133 
 134     main.cookiecutter('tests/files/fake-repo-tmpl.zip', no_input=True)
 135     assert os.path.isdir('fake-project-templated')
 136 
 137     # The tmp directory will still exist, but the
 138     # extracted template directory *in* the temp directory will not.
 139     assert os.path.exists('fake-tmp')
 140     assert not os.path.exists('fake-tmp/fake-repo-tmpl')
