  15 def nested_tree():
  16     """This tree has a few unique properties:
  17 
  18     - children is mixed strings and components (as in c2)
  19     - children is just components (as in c)
  20     - children is just strings (as in c1)
  21     - children is just a single component (as in c3, c4)
  22     - children contains numbers (as in c2)
  23     - children contains "None" items (as in c2)
  24     """
  25     c1 = Component(id="0.1.x.x.0", children="string")
  26     c2 = Component(
  27         id="0.1.x.x", children=[10, None, "wrap string", c1, "another string", 4.51]
  28     )
  29     c3 = Component(
  30         id="0.1.x",
  31         # children is just a component
  32         children=c2,
  33     )
  34     c4 = Component(id="0.1", children=c3)
  35     c5 = Component(id="0.0")
  36     c = Component(id="0", children=[c5, c4])
  37     return c, c1, c2, c3, c4, c5

 372 def test_debc025_iter():
 373     # The mixin methods from MutableMapping were cute but probably never
 374     # used - at least not by us. Test that they're gone
 375 
 376     # keys, __contains__, items, values, and more are all mixin methods
 377     # that we get for free by inheriting from the MutableMapping
 378     # and behave as according to our implementation of __iter__
 379 
 380     c = Component(
 381         id="1",
 382         children=[
 383             Component(id="2", children=[Component(id="3", children=Component(id="4"))]),
 384             Component(id="5", children=[Component(id="6", children="Hello World")]),
 385             Component(),
 386             Component(children="Hello World"),
 387             Component(children=Component(id="7")),
 388             Component(children=[Component(id="8")]),
 389         ],
 390     )
 391 
 392     mixins = [
 393         "clear",
 394         "get",
 395         "items",
 396         "keys",
 397         "pop",
 398         "popitem",
 399         "setdefault",
 400         "update",
 401         "values",
 402     ]
 403 
 404     for m in mixins:
 405         assert not hasattr(c, m), "should not have method " + m
 406 
 407     keys = ["2", "3", "4", "5", "6", "7", "8"]
 408 
 409     for k in keys:
 410         # test __contains__()
 411         assert k in c, "should find key " + k
 412         # test __getitem__()
 413         assert c[k].id == k, "key {} points to the right item".format(k)
 414 
 415     # test __iter__()
 416     keys2 = []
 417     for k in c:
 418         keys2.append(k)
 419         assert k in keys, "iteration produces key " + k
 420 
 421     assert len(keys) == len(keys2), "iteration produces no extra keys"

 424 def test_debc026_component_not_children():
 425     children = [Component(id="a"), html.Div(id="b"), "c", 1]
 426     for i in range(len(children)):
 427         # cycle through each component in each position
 428         children = children[1:] + [children[0]]
 429 
 430         # use html.Div because only real components accept positional args
 431         html.Div(children)
 432         # the first arg is children, and a single component works there
 433         html.Div(children[0], id="x")
 434 
 435         with pytest.raises(TypeError):
 436             # If you forget the `[]` around children you get this:
 437             html.Div(children[0], children[1], children[2], children[3])

 440 def test_debc027_component_error_message():
