 125 def test_attrs_match_forbidden_props(component_class):
 126     assert "_.*" in reserved_words, "props cannot have leading underscores"
 127 
 128     # props are not added as attrs unless explicitly provided
 129     # except for children, which is always set if it's a prop at all.
 130     expected_attrs = set(reserved_words + ["children"]) - {"_.*"}
 131     c = component_class()
 132     base_attrs = set(dir(c))
 133     extra_attrs = set(a for a in base_attrs if a[0] != "_")
 134 
 135     assert (
 136         extra_attrs == expected_attrs
 137     ), "component has only underscored and reserved word attrs"
 138 
 139     # setting props causes them to show up as attrs
 140     c2 = component_class("children", id="c2", optionalArray=[1])
 141     prop_attrs = set(dir(c2))
 142 
 143     assert base_attrs - prop_attrs == set([]), "no attrs were removed"
 144     assert prop_attrs - base_attrs == {
 145         "id",
 146         "optionalArray",
 147     }, "explicit props were added as attrs"
