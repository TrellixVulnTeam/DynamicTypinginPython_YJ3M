   8 def wait_for_queue(dash_duo):
   9     # mostly for cases where no callbacks should fire:
  10     # just wait until we have the button and the queue is empty
  11     dash_duo.wait_for_text_to_equal("#btn", "click")
  12     wait.until(lambda: not dash_duo.redux_state_is_loading, 3)

  15 def test_cbmi001_all_missing_inputs(dash_duo):
  16     app = dash.Dash(__name__, suppress_callback_exceptions=True)
  17     app.layout = html.Div(
  18         [
  19             html.Div("Title", id="title"),
  20             html.Button("click", id="btn"),
  21             html.Div(id="content"),
  22             html.Div("output1 init", id="out1"),
  23             html.Div("output2 init", id="out2"),
  24             html.Div("output3 init", id="out3"),
  25         ]
  26     )
  27 
  28     @app.callback(Output("content", "children"), [Input("btn", "n_clicks")])
  29     def content(n):
  30         return (
  31             [html.Div("A", id="a"), html.Div("B", id="b"), html.Div("C", id="c")]
  32             if n
  33             else "content init"
  34         )
  35 
  36     @app.callback(
  37         Output("out1", "children"),
  38         [Input("a", "children"), Input("b", "children")],
  39         [State("c", "children"), State("title", "children")],
  40     )
  41     def out1(a, b, c, title):
  42         # title exists at the start but State isn't enough to fire the callback
  43         assert c == "C"
  44         assert title == "Title"
  45         return a + b
  46 
  47     @app.callback(
  48         Output("out2", "children"),
  49         [Input("out1", "children")],
  50         [State("title", "children")],
  51     )
  52     def out2(out1, title):
  53         return out1 + " - 2 - " + title
  54 
  55     @app.callback(
  56         Output("out3", "children"),
  57         [Input("out1", "children"), Input("title", "children")],
  58     )
  59     def out3(out1, title):
  60         return out1 + " - 3 - " + title
  61 
  62     dash_duo.start_server(app)
  63     wait_for_queue(dash_duo)
  64 
  65     # out3 fires because it has another Input besides out1
  66     dash_duo.wait_for_text_to_equal("#out3", "output1 init - 3 - Title")
  67 
  68     assert dash_duo.find_element("#out1").text == "output1 init"
  69 
  70     # out2 doesn't fire because its only input (out1) is "prevented"
  71     # State items don't matter for this.
  72     assert dash_duo.find_element("#out2").text == "output2 init"
  73 
  74     dash_duo.find_element("#btn").click()
  75 
  76     # after a button click all inputs are present so all callbacks fire.
  77     dash_duo.wait_for_text_to_equal("#out1", "AB")
  78     dash_duo.wait_for_text_to_equal("#out2", "AB - 2 - Title")
  79     dash_duo.wait_for_text_to_equal("#out3", "AB - 3 - Title")

 399 def test_cbmi009_match_wildcards_all_missing(dash_duo):
 400     # Kind of contrived - MATCH will always be 0. Just want to make sure
 401     # that this behaves the same as cbmi001
 402     app = dash.Dash(__name__, suppress_callback_exceptions=True)
 403     app.layout = html.Div(
 404         [
 405             html.Div("Title", id={"i": 0, "id": "title"}),
 406             html.Button("click", id="btn"),
 407             html.Div(id="content"),
 408             html.Div("output1 init", id={"i": 0, "id": "out1"}),
 409             html.Div("output2 init", id={"i": 0, "id": "out2"}),
 410             html.Div("output3 init", id={"i": 0, "id": "out3"}),
 411         ]
 412     )
 413 
 414     @app.callback(Output("content", "children"), [Input("btn", "n_clicks")])
 415     def content(n):
 416         return (
 417             [
 418                 html.Div("A", id={"i": 0, "id": "a"}),
 419                 html.Div("B", id={"i": 0, "id": "b"}),
 420                 html.Div("C", id={"i": 0, "id": "c"}),
 421             ]
 422             if n
 423             else "content init"
 424         )
 425 
 426     @app.callback(
 427         Output({"i": MATCH, "id": "out1"}, "children"),
 428         [
 429             Input({"i": MATCH, "id": "a"}, "children"),
 430             Input({"i": MATCH, "id": "b"}, "children"),
 431         ],
 432         [
 433             State({"i": MATCH, "id": "c"}, "children"),
 434             State({"i": MATCH, "id": "title"}, "children"),
 435         ],
 436     )
 437     def out1(a, b, c, title):
 438         # title exists at the start but State isn't enough to fire the callback
 439         assert c == "C"
 440         assert title == "Title"
 441         return a + b
 442 
 443     @app.callback(
 444         Output({"i": MATCH, "id": "out2"}, "children"),
 445         [Input({"i": MATCH, "id": "out1"}, "children")],
 446         [State({"i": MATCH, "id": "title"}, "children")],
 447     )
 448     def out2(out1, title):
 449         return out1 + " - 2 - " + title
 450 
 451     @app.callback(
 452         Output({"i": MATCH, "id": "out3"}, "children"),
 453         [
 454             Input({"i": MATCH, "id": "out1"}, "children"),
 455             Input({"i": MATCH, "id": "title"}, "children"),
 456         ],
 457     )
 458     def out3(out1, title):
 459         return out1 + " - 3 - " + title

 464     def cssid(v):
 465         # escaped CSS for object IDs
 466         return r"#\{\"i\"\:0\,\"id\"\:\"" + v + r"\"\}"
 467
