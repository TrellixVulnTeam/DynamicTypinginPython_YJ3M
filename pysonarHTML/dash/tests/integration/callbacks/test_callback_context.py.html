  14 def test_cbcx001_modified_response(dash_duo):
  15     app = Dash(__name__)
  16     app.layout = html.Div([dcc.Input(id="input", value="ab"), html.Div(id="output")])
  17 
  18     @app.callback(Output("output", "children"), [Input("input", "value")])

  20         callback_context.response.set_cookie("dash_cookie", value + " - cookie")

  37 def test_cbcx002_triggered(dash_duo):
  38     app = Dash(__name__)
  39 
  40     btns = ["btn-{}".format(x) for x in range(1, 6)]
  41 
  42     app.layout = html.Div(
  43         [html.Div([html.Button(btn, id=btn) for btn in btns]), html.Div(id="output")]
  44     )
  45 
  46     @app.callback(Output("output", "children"), [Input(x, "n_clicks") for x in btns])
  47     def on_click(*args):
  48         if not callback_context.triggered:
  49             raise PreventUpdate
  50         trigger = callback_context.triggered[0]
  51         return "Just clicked {} for the {} time!".format(
  52             trigger["prop_id"].split(".")[0], trigger["value"]
  53         )
  54 
  55     dash_duo.start_server(app)
  56 
  57     for i in range(1, 5):
  58         for btn in btns:
  59             dash_duo.find_element("#" + btn).click()
  60             dash_duo.wait_for_text_to_equal(
  61                 "#output", "Just clicked {} for the {} time!".format(btn, i)
  62             )

  71 def test_cbcx004_triggered_backward_compat(dash_duo):
  72     app = Dash(__name__)
  73     app.layout = html.Div([html.Button("click!", id="btn"), html.Div(id="out")])
  74 
  75     @app.callback(Output("out", "children"), [Input("btn", "n_clicks")])
  76     def report_triggered(n):
  77         triggered = callback_context.triggered
  78         bool_val = "truthy" if triggered else "falsy"
  79         split_propid = json.dumps(triggered[0]["prop_id"].split("."))
  80         full_val = json.dumps(triggered)
  81         return "triggered is {}, has prop/id {}, and full value {}".format(
  82             bool_val, split_propid, full_val
  83         )
  84 
  85     dash_duo.start_server(app)
  86 
  87     dash_duo.wait_for_text_to_equal(
  88         "#out",
  89         'triggered is falsy, has prop/id ["", ""], and full value '
  90         '[{"prop_id": ".", "value": null}]',
  91     )
  92 
  93     dash_duo.find_element("#btn").click()
  94     dash_duo.wait_for_text_to_equal(
  95         "#out",
  96         'triggered is truthy, has prop/id ["btn", "n_clicks"], and full value '
  97         '[{"prop_id": "btn.n_clicks", "value": 1}]',
  98     )
