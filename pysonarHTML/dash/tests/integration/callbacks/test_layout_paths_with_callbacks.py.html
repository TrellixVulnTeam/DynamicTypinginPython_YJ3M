 100     @app.callback(Output("body", "children"), [Input("toc", "value")])
 101     def display_chapter(toc_value):
 102         call_counts["body"].value += 1
 103         return chapters[toc_value]

 105     app.config.suppress_callback_exceptions = True
 106 
 107     def generate_graph_callback(counterId):
 108         def callback(value):
 109             call_counts[counterId].value += 1
 110             return {
 111                 "data": [
 112                     {
 113                         "x": ["Call Counter for: {}".format(counterId)],
 114                         "y": [call_counts[counterId].value],
 115                         "type": "bar",
 116                     }
 117                 ],
 118                 "layout": {
 119                     "title": value,
 120                     "width": 500,
 121                     "height": 400,
 122                     "margin": {"autoexpand": False},
 123                 },
 124             }
 125 
 126         return callback
 127 
 128     def generate_label_callback(id_):
 129         def update_label(value):
 130             call_counts[id_].value += 1
 131             return value
 132 
 133         return update_label
 134 
 135     for chapter in ["chapter1", "chapter2", "chapter3"]:

 141         app.callback(
