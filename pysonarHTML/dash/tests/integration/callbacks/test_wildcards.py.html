 237 def fibonacci_app(clientside):
 238     global fibonacci_count
 239     global fibonacci_sum_count
 240 
 241     fibonacci_count = 0
 242     fibonacci_sum_count = 0

 244     # This app tests 2 things in particular:
 245     # - clientside callbacks work the same as server-side
 246     # - callbacks using ALLSMALLER as an input to MATCH of the exact same id/prop
 247     app = dash.Dash(__name__)
 248     app.layout = html.Div(
 249         [
 250             dcc.Input(id="n", type="number", min=0, max=10, value=4),
 251             html.Div(id="series"),
 252             html.Div(id="sum"),
 253         ]
 254     )
 255 
 256     @app.callback(Output("series", "children"), [Input("n", "value")])
 257     def items(n):
 258         return [html.Div(id={"i": i}) for i in range(n)]
 259 
 260     if clientside:
 261         app.clientside_callback(
 262             """
 263             function(vals) {
 264                 var len = vals.length;
 265                 return len < 2 ? len : +(vals[len - 1] || 0) + +(vals[len - 2] || 0);
 266             }
 267             """,
 268             Output({"i": MATCH}, "children"),
 269             [Input({"i": ALLSMALLER}, "children")],
 270         )
 271 
 272         app.clientside_callback(
 273             """
 274             function(vals) {
 275                 var sum = vals.reduce(function(a, b) { return +a + +b; }, 0);
 276                 return vals.length + ' elements, sum: ' + sum;
 277             }
 278             """,
 279             Output("sum", "children"),
 280             [Input({"i": ALL}, "children")],
 281         )
 282 
 283     else:
 284 
 285         @app.callback(
 286             Output({"i": MATCH}, "children"), [Input({"i": ALLSMALLER}, "children")]
 287         )
 288         def sequence(prev):
 289             global fibonacci_count
 290             fibonacci_count = fibonacci_count + 1
 291             print(fibonacci_count)
 292 
 293             if len(prev) < 2:
 294                 return len(prev)
 295             return int(prev[-1] or 0) + int(prev[-2] or 0)
 296 
 297         @app.callback(Output("sum", "children"), [Input({"i": ALL}, "children")])
 298         def show_sum(seq):
 299             global fibonacci_sum_count
 300             fibonacci_sum_count = fibonacci_sum_count + 1
 301             print("fibonacci_sum_count: ", fibonacci_sum_count)
 302 
 303             return "{} elements, sum: {}".format(
 304                 len(seq), sum(int(v or 0) for v in seq)
 305             )
 306 
 307     return app
