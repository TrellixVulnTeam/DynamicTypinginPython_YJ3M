  15     app = dash.Dash(__name__)
  16     app.layout = html.Div([html.Button(id="input", n_clicks=0), html.Div(id="output")])
  17

  19

  42 def test_cbmt002_canceled_intermediate_callback(dash_duo):
  43     # see https://github.com/plotly/dash/issues/1053
  44     app = dash.Dash(__name__)
  45     app.layout = html.Div(
  46         [
  47             dcc.Input(id="a", value="x"),
  48             html.Div("b", id="b"),
  49             html.Div("c", id="c"),
  50             html.Div(id="out"),
  51         ]
  52     )
  53 
  54     @app.callback(
  55         Output("out", "children"),
  56         [Input("a", "value"), Input("b", "children"), Input("c", "children")],
  57     )
  58     def set_out(a, b, c):
  59         return "{}/{}/{}".format(a, b, c)
  60 
  61     @app.callback(Output("b", "children"), [Input("a", "value")])
  62     def set_b(a):
  63         raise PreventUpdate
  64 
  65     @app.callback(Output("c", "children"), [Input("a", "value")])
  66     def set_c(a):
  67         return a
  68 
  69     dash_duo.start_server(app)
  70     dash_duo.wait_for_text_to_equal("#out", "x/b/x")
  71     chars = "x"
  72     for i in list(range(10)) * 2:
  73         dash_duo.find_element("#a").send_keys(str(i))
  74         chars += str(i)
  75         dash_duo.wait_for_text_to_equal("#out", "{0}/b/{0}".format(chars))

  78 def test_cbmt003_chain_with_table(dash_duo):
  79     # see https://github.com/plotly/dash/issues/1071
  80     app = dash.Dash(__name__)
  81     app.layout = html.Div(
  82         [
  83             html.Div(id="a1"),

  88             dash_table.DataTable(id="table"),
  89         ]
  90     )
  91 
  92     @app.callback(
  93         # Changing the order of outputs here fixes the issue
  94         [Output("a2", "children"), Output("a1", "children")],
  95         [Input("button", "n_clicks")],
  96     )
  97     def a12(n):
  98         return "a2: {!s}".format(n), "a1: {!s}".format(n)
  99 
 100     @app.callback(Output("b1", "children"), [Input("a1", "children")])
 101     def b1(a1):
 102         return "b1: '{!s}'".format(a1)
 103 
 104     @app.callback(
 105         Output("b2", "children"),
 106         [Input("a2", "children"), Input("table", "selected_cells")],
 107     )
 108     def b2(a2, selected_cells):
 109         return "b2: '{!s}', {!s}".format(a2, selected_cells)

 121     dash_duo.wait_for_text_to_equal("#b1", "b1: 'a1: 1'")

 199 def test_cbmt005_multi_converging_chain(dash_duo):
 200     app = dash.Dash(__name__)
 201     app.layout = html.Div(
 202         [
 203             html.Button("Button 1", id="b1"),
 204             html.Button("Button 2", id="b2"),
 205             dcc.Slider(id="slider1", min=-5, max=5),
 206             dcc.Slider(id="slider2", min=-5, max=5),
 207             html.Div(id="out"),
 208         ]
 209     )
 210 
 211     @app.callback(
 212         [Output("slider1", "value"), Output("slider2", "value")],
 213         [Input("b1", "n_clicks"), Input("b2", "n_clicks")],
 214     )
 215     def update_sliders(button1, button2):
 216         if not dash.callback_context.triggered:
 217             raise PreventUpdate
 218 
 219         if dash.callback_context.triggered[0]["prop_id"] == "b1.n_clicks":
 220             return -1, -1
 221         else:
 222             return 1, 1
 223 
 224     @app.callback(
 225         Output("out", "children"),
 226         [Input("slider1", "value"), Input("slider2", "value")],
 227     )
 228     def update_graph(s1, s2):
 229         return "x={}, y={}".format(s1, s2)
 230 
 231     dash_duo.start_server(app)
 232     dash_duo.wait_for_text_to_equal("#out", "")
 233 
 234     dash_duo.find_element("#b1").click()
 235     dash_duo.wait_for_text_to_equal("#out", "x=-1, y=-1")
 236 
 237     dash_duo.find_element("#b2").click()
 238     dash_duo.wait_for_text_to_equal("#out", "x=1, y=1")

 267 def test_cbmt007_early_preventupdate_inputs_above_below(dash_duo):
 268     app = dash.Dash(__name__, suppress_callback_exceptions=True)
 269     app.layout = html.Div(id="content")
 270 
 271     @app.callback(Output("content", "children"), [Input("content", "style")])
 272     def content(_):
 273         return html.Div(
 274             [
 275                 html.Div(42, id="above-in"),
 276                 html.Div(id="above-dummy"),
 277                 html.Hr(),
 278                 html.Div(0, id="above-out"),
 279                 html.Div(0, id="below-out"),
 280                 html.Hr(),
 281                 html.Div(id="below-dummy"),
 282                 html.Div(44, id="below-in"),
 283             ]
 284         )
 285 
 286     # Create 4 callbacks - 2 above, 2 below.
 287     for pos in ("above", "below"):
 288 
 289         @app.callback(
 290             Output("{}-dummy".format(pos), "children"),
 291             [Input("{}-dummy".format(pos), "style")],
 292         )
 293         def dummy(_):
 294             raise PreventUpdate
 295 
 296         @app.callback(
 297             Output("{}-out".format(pos), "children"),
 298             [Input("{}-in".format(pos), "children")],
 299         )
 300         def out(v):
 301             return v
 302 
 303     dash_duo.start_server(app)
 304 
 305     # as of https://github.com/plotly/dash/issues/1223, above-out would be 0
 306     dash_duo.wait_for_text_to_equal("#above-out", "42")
 307     dash_duo.wait_for_text_to_equal("#below-out", "44")
