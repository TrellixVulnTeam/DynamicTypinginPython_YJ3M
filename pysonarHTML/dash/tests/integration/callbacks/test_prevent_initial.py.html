  19 def make_app(flavor):
  20     kwargs = {}
  21     if flavor["global"]:
  22         kwargs["prevent_initial_callbacks"] = True
  23     return dash.Dash(__name__, **kwargs)

  26 def content_callback(app, flavor, layout):
  27     kwargs = {}
  28     if flavor["global"]:
  29         kwargs["prevent_initial_call"] = False
  30 
  31     if flavor["content"]:
  32         app.layout = html.Div(id="content")
  33 
  34         @app.callback(
  35             Output("content", "children"), [Input("content", "style")], **kwargs
  36         )
  37         def set_content(_):
  38             return layout
  39 
  40     else:
  41         app.layout = layout

  44 def const_callback(app, flavor, val, outputs, inputs, prevent_initial_call=False):
  45     kwargs = {}
  46     if prevent_initial_call != flavor["global"]:
  47         kwargs["prevent_initial_call"] = prevent_initial_call
  48 
  49     if flavor["clientside"]:
  50         vstr = json.dumps(val)
  51         app.clientside_callback(
  52             "function() { return " + vstr + "; }", outputs, inputs, **kwargs
  53         )
  54     else:
  55 
  56         @app.callback(outputs, inputs, **kwargs)
  57         def f(*args):
  58             return val

  61 def concat_callback(app, flavor, outputs, inputs, prevent_initial_call=False):
  62     kwargs = {}
  63     if prevent_initial_call != flavor["global"]:
  64         kwargs["prevent_initial_call"] = prevent_initial_call
  65 
  66     multi_out = isinstance(outputs, (list, tuple))
  67 
  68     if flavor["clientside"]:
  69         app.clientside_callback(
  70             """
  71             function() {
  72                 var out = '';
  73                 for(var i = 0; i < arguments.length; i++) {
  74                     out += String(arguments[i]);
  75                 }
  76                 return X;
  77             }
  78             """.replace(
  79                 "X",
  80                 ("[" + ",".join(["out"] * len(outputs)) + "]") if multi_out else "out",
  81             ),
  82             outputs,
  83             inputs,
  84             **kwargs
  85         )
  86     else:
  87 
  88         @app.callback(outputs, inputs, **kwargs)
  89         def f(*args):
  90             out = "".join(str(arg) for arg in args)
  91             return [out] * len(outputs) if multi_out else out

  94 @pytest.mark.parametrize("flavor", flavors)
  95 def test_cbpi001_prevent_initial_call(flavor, dash_duo):
  96     app = make_app(flavor)
  97 
  98     layout = html.Div(
  99         [
 100             html.Button("click", id="btn"),
 101             html.Div("A", id="a"),
 102             html.Div("B", id="b"),
 103             html.Div("C", id="c"),
 104             html.Div("D", id="d"),
 105             html.Div("E", id="e"),
 106             html.Div("F", id="f"),
 107         ]
 108     )
 109     content_callback(app, flavor, layout)
 110 
 111     # Prevented, so A will only change after the button is clicked
 112     const_callback(
 113         app,
 114         flavor,
 115         "Click",
 116         Output("a", "children"),
 117         [Input("btn", "n_clicks")],
 118         prevent_initial_call=True,
 119     )
 120 
 121     # B depends on A - this *will* run, because prevent_initial_call is
 122     # not equivalent to PreventUpdate within the callback, it's treated as if
 123     # that callback was never in the initialization chain.
 124     concat_callback(app, flavor, Output("b", "children"), [Input("a", "children")])
 125 
 126     # C matches B except that it also has prevent_initial_call itself, not just
 127     # its input A - so it will not run initially
 128     concat_callback(
 129         app,
 130         flavor,
 131         Output("c", "children"),
 132         [Input("a", "children")],
 133         prevent_initial_call=True,
 134     )
 135 
 136     @app.callback(Output("d", "children"), [Input("d", "style")])
 137     def d(_):
 138         raise PreventUpdate
 139 
 140     # E depends on A and D - one prevent_initial_call and one PreventUpdate
 141     # the prevent_initial_call means it *will* run
 142     concat_callback(
 143         app,
 144         flavor,
 145         Output("e", "children"),
 146         [Input("a", "children"), Input("d", "children")],
 147     )
 148 
 149     # F has prevent_initial_call but DOES fire during init, because one of its
 150     # inputs (B) was changed by a callback
 151     concat_callback(
 152         app,
 153         flavor,
 154         Output("f", "children"),
 155         [Input("a", "children"), Input("b", "children"), Input("d", "children")],
 156         prevent_initial_call=True,
 157     )
 158 
 159     dash_duo.start_server(app)
 160 
 161     # check from the end, to ensure the callbacks are all done
 162     dash_duo.wait_for_text_to_equal("#f", "AAD")
 163     dash_duo.wait_for_text_to_equal("#e", "AD"),
 164     dash_duo.wait_for_text_to_equal("#d", "D")
 165     dash_duo.wait_for_text_to_equal("#c", "C")
 166     dash_duo.wait_for_text_to_equal("#b", "A")
 167     dash_duo.wait_for_text_to_equal("#a", "A")
 168 
 169     dash_duo.find_element("#btn").click()
 170 
 171     dash_duo.wait_for_text_to_equal("#f", "ClickClickD")
 172     dash_duo.wait_for_text_to_equal("#e", "ClickD"),
 173     dash_duo.wait_for_text_to_equal("#d", "D")
 174     dash_duo.wait_for_text_to_equal("#c", "Click")
 175     dash_duo.wait_for_text_to_equal("#b", "Click")
 176     dash_duo.wait_for_text_to_equal("#a", "Click")
