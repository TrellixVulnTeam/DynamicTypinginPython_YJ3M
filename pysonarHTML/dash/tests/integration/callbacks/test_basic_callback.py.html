  14 def test_cbsc001_simple_callback(dash_duo):
  15     app = dash.Dash(__name__)
  16     app.layout = html.Div(
  17         [
  18             dcc.Input(id="input", value="initial value"),
  19             html.Div(html.Div([1.5, None, "string", html.Div(id="output-1")])),
  20         ]
  21     )
  22     call_count = Value("i", 0)
  23 
  24     @app.callback(Output("output-1", "children"), [Input("input", "value")])
  25     def update_output(value):
  26         call_count.value = call_count.value + 1
  27         return value
  28 
  29     dash_duo.start_server(app)
  30 
  31     assert dash_duo.find_element("#output-1").text == "initial value"
  32     dash_duo.percy_snapshot(name="simple-callback-initial")
  33 
  34     input_ = dash_duo.find_element("#input")
  35     dash_duo.clear_input(input_)
  36 
  37     input_.send_keys("hello world")
  38 
  39     assert dash_duo.find_element("#output-1").text == "hello world"
  40     dash_duo.percy_snapshot(name="simple-callback-hello-world")
  41 
  42     assert call_count.value == 2 + len("hello world"), "initial count + each key stroke"
  43 
  44     assert not dash_duo.redux_state_is_loading
  45 
  46     assert dash_duo.get_logs() == []

  49 def test_cbsc002_callbacks_generating_children(dash_duo):
  50     """Modify the DOM tree by adding new components in the callbacks."""
  51 
  52     # some components don't exist in the initial render
  53     app = dash.Dash(__name__, suppress_callback_exceptions=True)
  54     app.layout = html.Div(
  55         [dcc.Input(id="input", value="initial value"), html.Div(id="output")]
  56     )
  57 
  58     @app.callback(Output("output", "children"), [Input("input", "value")])
  59     def pad_output(input):
  60         return html.Div(
  61             [
  62                 dcc.Input(id="sub-input-1", value="sub input initial value"),
  63                 html.Div(id="sub-output-1"),
  64             ]
  65         )
  66 
  67     call_count = Value("i", 0)
  68 
  69     @app.callback(Output("sub-output-1", "children"), [Input("sub-input-1", "value")])
  70     def update_input(value):
  71         call_count.value = call_count.value + 1
  72         return value
  73 
  74     dash_duo.start_server(app)
  75 
  76     dash_duo.wait_for_text_to_equal("#sub-output-1", "sub input initial value")
  77 
  78     assert call_count.value == 1, "called once at initial stage"
  79 
  80     pad_input, pad_div = dash_duo.dash_innerhtml_dom.select_one(
  81         "#output > div"
  82     ).contents
  83 
  84     assert (
  85         pad_input.attrs["value"] == "sub input initial value"
  86         and pad_input.attrs["id"] == "sub-input-1"
  87     )
  88     assert pad_input.name == "input"
  89 
  90     assert (
  91         pad_div.text == pad_input.attrs["value"] and pad_div.get("id") == "sub-output-1"
  92     ), "the sub-output-1 content reflects to sub-input-1 value"
  93 
  94     dash_duo.percy_snapshot(name="callback-generating-function-1")
  95 
  96     paths = dash_duo.redux_state_paths
  97     assert paths["objs"] == {}
  98     assert paths["strs"] == {
  99         "input": ["props", "children", 0],
 100         "output": ["props", "children", 1],
 101         "sub-input-1": [
 102             "props",
 103             "children",
 104             1,
 105             "props",
 106             "children",
 107             "props",
 108             "children",
 109             0,
 110         ],
 111         "sub-output-1": [
 112             "props",
 113             "children",
 114             1,
 115             "props",
 116             "children",
 117             "props",
 118             "children",
 119             1,
 120         ],
 121     }, "the paths should include these new output IDs"
 122 
 123     # editing the input should modify the sub output
 124     dash_duo.find_element("#sub-input-1").send_keys("deadbeef")
 125 
 126     assert (
 127         dash_duo.find_element("#sub-output-1").text
 128         == pad_input.attrs["value"] + "deadbeef"
 129     ), "deadbeef is added"
 130 
 131     # the total updates is initial one + the text input changes
 132     dash_duo.wait_for_text_to_equal(
 133         "#sub-output-1", pad_input.attrs["value"] + "deadbeef"
 134     )
 135 
 136     assert not dash_duo.redux_state_is_loading, "loadingMap is empty"
 137 
 138     dash_duo.percy_snapshot(name="callback-generating-function-2")
 139     assert dash_duo.get_logs() == [], "console is clean"

 142 def test_cbsc003_callback_with_unloaded_async_component(dash_duo):
 143     app = dash.Dash()
 144     app.layout = html.Div(
 145         children=[
 146             dcc.Tabs(
 147                 children=[
 148                     dcc.Tab(
 149                         children=[
 150                             html.Button(id="btn", children="Update Input"),
 151                             html.Div(id="output", children=["Hello"]),
 152                         ]
 153                     ),
 154                     dcc.Tab(children=dash_table.DataTable(id="other-table")),
 155                 ]
 156             )
 157         ]
 158     )
 159 
 160     @app.callback(Output("output", "children"), [Input("btn", "n_clicks")])
 161     def update_out(n_clicks):
 162         if n_clicks is None:
 163             raise PreventUpdate
 164 
 165         return "Bye"
 166 
 167     dash_duo.start_server(app)
 168 
 169     dash_duo.wait_for_text_to_equal("#output", "Hello")
 170     dash_duo.find_element("#btn").click()
 171     dash_duo.wait_for_text_to_equal("#output", "Bye")
 172     assert dash_duo.get_logs() == []

 175 def test_cbsc004_callback_using_unloaded_async_component(dash_duo):
 176     app = dash.Dash()
 177     app.layout = html.Div(
 178         [
 179             dcc.Tabs(
 180                 [
 181                     dcc.Tab("boo!"),
 182                     dcc.Tab(
 183                         dash_table.DataTable(
 184                             id="table",
 185                             columns=[{"id": "a", "name": "A"}],
 186                             data=[{"a": "b"}],
 187                         )
 188                     ),
 189                 ]
 190             ),
 191             html.Button("Update Input", id="btn"),
 192             html.Div("Hello", id="output"),
 193             html.Div(id="output2"),
 194         ]
 195     )
 196 
 197     @app.callback(
 198         Output("output", "children"),
 199         [Input("btn", "n_clicks")],
 200         [State("table", "data")],
 201     )
 202     def update_out(n_clicks, data):
 203         return json.dumps(data) + " - " + str(n_clicks)
 204 
 205     @app.callback(
 206         Output("output2", "children"),
 207         [Input("btn", "n_clicks")],
 208         [State("table", "derived_viewport_data")],
 209     )
 210     def update_out2(n_clicks, data):
 211         return json.dumps(data) + " - " + str(n_clicks)
 212 
 213     dash_duo.start_server(app)
 214 
 215     dash_duo.wait_for_text_to_equal("#output", '[{"a": "b"}] - None')
 216     dash_duo.wait_for_text_to_equal("#output2", "null - None")
 217 
 218     dash_duo.find_element("#btn").click()
 219     dash_duo.wait_for_text_to_equal("#output", '[{"a": "b"}] - 1')
 220     dash_duo.wait_for_text_to_equal("#output2", "null - 1")
 221 
 222     dash_duo.find_element(".tab:not(.tab--selected)").click()
 223     dash_duo.wait_for_text_to_equal("#table th", "A")
 224     # table props are in state so no change yet
 225     dash_duo.wait_for_text_to_equal("#output2", "null - 1")
 226 
 227     # repeat a few times, since one of the failure modes I saw during dev was
 228     # intermittent - but predictably so?
 229     for i in range(2, 10):
 230         expected = '[{"a": "b"}] - ' + str(i)
 231         dash_duo.find_element("#btn").click()
 232         dash_duo.wait_for_text_to_equal("#output", expected)
 233         # now derived props are available
 234         dash_duo.wait_for_text_to_equal("#output2", expected)
 235 
 236     assert dash_duo.get_logs() == []

 239 def test_cbsc005_children_types(dash_duo):
 240     app = dash.Dash()
 241     app.layout = html.Div([html.Button(id="btn"), html.Div("init", id="out")])
 242 
 243     outputs = [
 244         [None, ""],
 245         ["a string", "a string"],
 246         [123, "123"],
 247         [123.45, "123.45"],
 248         [[6, 7, 8], "678"],
 249         [["a", "list", "of", "strings"], "alistofstrings"],
 250         [["strings", 2, "numbers"], "strings2numbers"],
 251         [["a string", html.Div("and a div")], "a string\nand a div"],
 252     ]
 253 
 254     @app.callback(Output("out", "children"), [Input("btn", "n_clicks")])
 255     def set_children(n):
 256         if n is None or n > len(outputs):
 257             return dash.no_update
 258         return outputs[n - 1][0]
 259 
 260     dash_duo.start_server(app)
 261     dash_duo.wait_for_text_to_equal("#out", "init")
 262 
 263     for children, text in outputs:
 264         dash_duo.find_element("#btn").click()
 265         dash_duo.wait_for_text_to_equal("#out", text)

 268 def test_cbsc006_array_of_objects(dash_duo):
 269     app = dash.Dash()
 270     app.layout = html.Div(
 271         [html.Button(id="btn"), dcc.Dropdown(id="dd"), html.Div(id="out")]
 272     )
 273 
 274     @app.callback(Output("dd", "options"), [Input("btn", "n_clicks")])
 275     def set_options(n):
 276         return [{"label": "opt{}".format(i), "value": i} for i in range(n or 0)]
 277 
 278     @app.callback(Output("out", "children"), [Input("dd", "options")])
 279     def set_out(opts):
 280         print(repr(opts))
 281         return len(opts)
 282 
 283     dash_duo.start_server(app)
 284 
 285     dash_duo.wait_for_text_to_equal("#out", "0")
 286     for i in range(5):
 287         dash_duo.find_element("#btn").click()
 288         dash_duo.wait_for_text_to_equal("#out", str(i + 1))
 289         dash_duo.select_dcc_dropdown("#dd", "opt{}".format(i))

 292 @pytest.mark.parametrize("refresh", [False, True])
 293 def test_cbsc007_parallel_updates(refresh, dash_duo):
 294     # This is a funny case, that seems to mostly happen with dcc.Location
 295     # but in principle could happen in other cases too:
 296     # A callback chain (in this case the initial hydration) is set to update a
 297     # value, but after that callback is queued and before it returns, that value
 298     # is also set explicitly from the front end (in this case Location.pathname,
 299     # which gets set in its componentDidMount during the render process, and
 300     # callbacks are delayed until after rendering is finished because of the
 301     # async table)
 302     # At one point in the wildcard PR #1103, changing from requestQueue to
 303     # pendingCallbacks, calling PreventUpdate in the callback would also skip
 304     # any callbacks that depend on pathname, despite the new front-end-provided
 305     # value.
 306 
 307     app = dash.Dash()
 308 
 309     app.layout = html.Div(
 310         [
 311             dcc.Location(id="loc", refresh=refresh),
 312             html.Button("Update path", id="btn"),
 313             dash_table.DataTable(id="t", columns=[{"name": "a", "id": "a"}]),
 314             html.Div(id="out"),
 315         ]
 316     )
 317 
 318     @app.callback(Output("t", "data"), [Input("loc", "pathname")])
 319     def set_data(path):
 320         return [{"a": (path or repr(path)) + ":a"}]
 321 
 322     @app.callback(
 323         Output("out", "children"), [Input("loc", "pathname"), Input("t", "data")]
 324     )
 325     def set_out(path, data):
 326         return json.dumps(data) + " - " + (path or repr(path))
 327 
 328     @app.callback(Output("loc", "pathname"), [Input("btn", "n_clicks")])
 329     def set_path(n):
 330         if not n:
 331             raise PreventUpdate
 332 
 333         return "/{0}".format(n)
 334 
 335     dash_duo.start_server(app)
 336 
 337     dash_duo.wait_for_text_to_equal("#out", '[{"a": "/:a"}] - /')
 338     dash_duo.find_element("#btn").click()
 339     # the refresh=True case here is testing that we really do get the right
 340     # pathname, not the prevented default value from the layout.
 341     dash_duo.wait_for_text_to_equal("#out", '[{"a": "/1:a"}] - /1')
 342     if not refresh:
 343         dash_duo.find_element("#btn").click()
 344         dash_duo.wait_for_text_to_equal("#out", '[{"a": "/2:a"}] - /2')
