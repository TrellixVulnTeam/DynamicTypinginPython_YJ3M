 187 def test_cbmo003_multi_all(dash_duo):
 188     app = dash.Dash(__name__)
 189 
 190     app.layout = html.Div(
 191         children=[
 192             html.Button("items", id="items"),
 193             html.Button("values", id="values"),
 194             html.Div(id="content1"),
 195             html.Hr(),
 196             html.Div(id="content2"),
 197             html.Hr(),
 198             html.Div("Output init", id="output"),
 199         ]
 200     )
 201 
 202     @app.callback(
 203         [Output("content1", "children"), Output("content2", "children")],
 204         [Input("items", "n_clicks")],
 205     )
 206     def content(n1):
 207         c1 = [html.Div(id={"i": i}) for i in range(((n1 or 0) + 2) % 4)]
 208         c2 = [html.Div(id={"j": j}) for j in range((n1 or 0) % 3)]
 209         return c1, c2
 210 
 211     @app.callback(
 212         [Output({"i": ALL}, "children"), Output({"j": ALL}, "children")],
 213         [Input("values", "n_clicks")],
 214     )
 215     def content_inner(n2):
 216         # this variant does NOT get called with empty ALL
 217         # the second callback handles outputting 0 in that case.
 218         # if it were to be called throw an error so we'll see it in get_logs
 219         n1i = len(dash.callback_context.outputs_list[0])
 220         n1j = len(dash.callback_context.outputs_list[1])
 221         if not n1i + n1j:
 222             raise ValueError("should not be called with no outputs!")
 223         return [n2 or 0] * n1i, [(n2 or 0) + 2] * n1j
 224 
 225     @app.callback(
 226         Output("output", "children"),
 227         [Input({"i": ALL}, "children"), Input({"j": ALL}, "children")],
 228     )
 229     def out2(ci, cj):
 230         return sum(ci) + sum(cj)
 231 
 232     dash_duo.start_server(app)
 233 
 234     dash_duo.wait_for_text_to_equal("#content1", "0\n0")
 235     dash_duo.wait_for_text_to_equal("#content2", "")
 236     dash_duo.wait_for_text_to_equal("#output", "0")
 237 
 238     actions = [
 239         ["#values", "1\n1", "", "2"],
 240         ["#items", "1\n1\n1", "3", "6"],
 241         ["#values", "2\n2\n2", "4", "10"],
 242         ["#items", "", "4\n4", "8"],
 243         ["#values", "", "5\n5", "10"],
 244         ["#items", "3", "", "3"],
 245         ["#values", "4", "", "4"],
 246         ["#items", "4\n4", "6", "14"],
 247         ["#values", "5\n5", "7", "17"],
 248         ["#items", "5\n5\n5", "7\n7", "29"],
 249         ["#values", "6\n6\n6", "8\n8", "34"],
 250         # all empty! we'll see an error logged if the callback was fired
 251         ["#items", "", "", "0"],
 252         ["#values", "", "", "0"],
 253         ["#items", "7", "9", "16"],
 254     ]
 255     for selector, content1, content2, output in actions:
 256         dash_duo.find_element(selector).click()
 257         dash_duo.wait_for_text_to_equal("#content1", content1)
 258         dash_duo.wait_for_text_to_equal("#content2", content2)
 259         dash_duo.wait_for_text_to_equal("#output", output)
 260 
 261     assert not dash_duo.get_logs()
