  28     def wait_for_element_by_css_selector(self, selector, timeout=TIMEOUT):
  29         return WebDriverWait(self.driver, timeout).until(
  30             EC.presence_of_element_located((By.CSS_SELECTOR, selector)),
  31             'Could not find element with selector "{}"'.format(selector),
  32         )

  34     def wait_for_text_to_equal(self, selector, assertion_text, timeout=TIMEOUT):
  35         self.wait_for_element_by_css_selector(selector)
  36         WebDriverWait(self.driver, timeout).until(
  37             lambda *args: (
  38                 (
  39                     str(self.wait_for_element_by_css_selector(selector).text)
  40                     == assertion_text
  41                 )
  42                 or (
  43                     str(
  44                         self.wait_for_element_by_css_selector(selector).get_attribute(
  45                             "value"
  46                         )
  47                     )
  48                     == assertion_text
  49                 )
  50             ),
  51             "Element '{}' text expects to equal '{}' but it didn't".format(
  52                 selector, assertion_text
  53             ),
  54         )

 293     def test_removing_component_while_its_getting_updated(self):
 294         app = Dash(__name__)
 295         app.layout = html.Div(
 296             [
 297                 dcc.RadioItems(
 298                     id="toc",
 299                     options=[{"label": i, "value": i} for i in ["1", "2"]],
 300                     value="1",
 301                 ),
 302                 html.Div(id="body"),
 303             ]
 304         )
 305         app.config.suppress_callback_exceptions = True
 306 
 307         call_counts = {"body": Value("i", 0), "button-output": Value("i", 0)}
 308 
 309         @app.callback(Output("body", "children"), [Input("toc", "value")])
 310         def update_body(chapter):
 311             call_counts["body"].value += 1
 312             if chapter == "1":
 313                 return [
 314                     html.Div("Chapter 1"),
 315                     html.Button("clicking this button takes forever", id="button"),
 316                     html.Div(id="button-output"),
 317                 ]
 318             elif chapter == "2":
 319                 return "Chapter 2"
 320             else:
 321                 raise Exception("chapter is {}".format(chapter))
 322 
 323         @app.callback(
 324             Output("button-output", "children"), [Input("button", "n_clicks")]
 325         )
 326         def this_callback_takes_forever(n_clicks):
 327             if not n_clicks:
 328                 # initial value is quick, only new value is slow
 329                 # also don't let the initial value increment call_counts
 330                 return "Initial Value"
 331             time.sleep(5)
 332             call_counts["button-output"].value += 1
 333             return "New value!"
 334 
 335         body = lambda: self.driver.find_element_by_id("body")
 336         self.startServer(app)
 337 
 338         wait_for(lambda: call_counts["body"].value == 1)
 339         time.sleep(0.5)
 340         self.driver.find_element_by_id("button").click()
 341 
 342         # while that callback is resolving, switch the chapter,
 343         # hiding the `button-output` tag
 344         def chapter2_assertions():
 345             wait_for(lambda: body().text == "Chapter 2")
 346 
 347             layout = self.driver.execute_script(
 348                 "return JSON.parse(JSON.stringify("
 349                 "window.store.getState().layout"
 350                 "))"
 351             )
 352 
 353             dcc_radio = layout["props"]["children"][0]
 354             html_body = layout["props"]["children"][1]
 355 
 356             self.assertEqual(dcc_radio["props"]["id"], "toc")
 357             self.assertEqual(dcc_radio["props"]["value"], "2")
 358 
 359             self.assertEqual(html_body["props"]["id"], "body")
 360             self.assertEqual(html_body["props"]["children"], "Chapter 2")
 361 
 362         (self.driver.find_elements_by_css_selector('input[type="radio"]')[1]).click()
 363         chapter2_assertions()
 364         self.assertEqual(call_counts["button-output"].value, 0)
 365         time.sleep(5)
 366         wait_for(lambda: call_counts["button-output"].value == 1)
 367         time.sleep(2)  # liberally wait for the front-end to process request
 368         chapter2_assertions()
 369         self.assertTrue(self.is_console_clean())
