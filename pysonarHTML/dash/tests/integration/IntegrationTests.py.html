  97     def wait_for_element_by_css_selector(self, selector, timeout=TIMEOUT):
  98         return WebDriverWait(self.driver, timeout).until(
  99             EC.presence_of_element_located((By.CSS_SELECTOR, selector)),
 100             'Could not find element with selector "{}"'.format(selector),
 101         )

 103     def wait_for_text_to_equal(self, selector, assertion_text, timeout=TIMEOUT):
 104         el = self.wait_for_element_by_css_selector(selector)
 105         WebDriverWait(self.driver, timeout).until(
 106             lambda *args: (
 107                 (str(el.text) == assertion_text)
 108                 or (str(el.get_attribute("value")) == assertion_text)
 109             ),
 110             "Element '{}' text was supposed to equal '{}' but it didn't".format(
 111                 selector, assertion_text
 112             ),
 113         )

 115     def clear_log(self):
 116         entries = self.driver.get_log("browser")
 117         if entries:

 120     def get_log(self):
 121         entries = self.driver.get_log("browser")
 122         return [entry for entry in entries if entry["timestamp"] > self.last_timestamp]

 124     def wait_until_get_log(self, timeout=10):
 125 
 126         logs = None
 127         cnt, poll = 0, 0.1
 128         while not logs:
 129             logs = self.get_log()
 130             time.sleep(poll)
 131             cnt += 1
 132             if cnt * poll >= timeout * 1000:
 133                 raise SeleniumDriverTimeout("cannot get log in {}".format(timeout))
 134 
 135         return logs
 136 
 137     def is_console_clean(self):
