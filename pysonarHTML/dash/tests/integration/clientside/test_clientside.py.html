  69 
  70     app.clientside_callback(

  75 
  76     call_counts = {"divide": Value("i", 0), "display": Value("i", 0)}
  77

  89             Input("x-plus-y-div-2", "value"),
  90         ],
  91     )
  92     def display_all(*args):
  93         call_counts["display"].value += 1
  94         return "\n".join([str(a) for a in args])
  95 
  96     app.clientside_callback(
  97         ClientsideFunction("clientside", "mean"),
  98         Output("mean-of-all-values", "value"),

 100             Input("x", "value"),
