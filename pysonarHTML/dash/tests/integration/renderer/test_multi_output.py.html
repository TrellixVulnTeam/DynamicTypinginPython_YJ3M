  11 def test_rdmo001_single_input_multi_outputs_on_multiple_components(dash_duo):
  12     call_count = Value("i")
  13 
  14     app = dash.Dash(__name__)
  15 
  16     N_OUTPUTS = 50
  17 
  18     app.layout = html.Div(
  19         [html.Button("click me", id="btn")]
  20         + [html.Div(id="output-{}".format(i)) for i in range(N_OUTPUTS)]
  21     )
  22 
  23     @app.callback(
  24         [Output("output-{}".format(i), "children") for i in range(N_OUTPUTS)],
  25         [Input("btn", "n_clicks")],
  26     )
  27     def update_output(n_clicks):
  28         if n_clicks is None:
  29             raise PreventUpdate
  30 
  31         call_count.value += 1
  32         return ["{}={}".format(i, i + n_clicks) for i in range(N_OUTPUTS)]
  33 
  34     dash_duo.start_server(app)
  35 
  36     btn = dash_duo.wait_for_element("#btn")
  37 
  38     for click in range(1, 20):
  39         btn.click()
  40 
  41         for i in range(N_OUTPUTS):
  42             dash_duo.wait_for_text_to_equal(
  43                 "#output-{}".format(i), "{}={}".format(i, i + click)
  44             )
  45 
  46         assert call_count.value == click
  47

  49 def test_rdmo002_multi_outputs_on_single_component(dash_duo):
  50     call_count = Value("i")
  51     app = dash.Dash(__name__)
  52 
  53     app.layout = html.Div(
  54         [
  55             dcc.Input(id="input", value="dash"),
  56             html.Div(html.Div(id="output"), id="output-container"),
  57         ]
  58     )
  59 
  60     @app.callback(
  61         [
  62             Output("output", "children"),
  63             Output("output", "style"),
  64             Output("output", "className"),
  65         ],
  66         [Input("input", "value")],
  67     )
  68     def update_output(value):
  69         call_count.value += 1
  70         return [value, {"fontFamily": value}, value]
  71 
  72     dash_duo.start_server(app)
  73 
  74     dash_duo.wait_for_text_to_equal("#output-container", "dash")
  75     _html = dash_duo.find_element("#output-container").get_property("innerHTML")
  76     assert _html == (
  77         '<div id="output" class="dash" style="font-family: dash;">dash</div>'
  78     )
  79 
  80     assert call_count.value == 1
  81 
  82     dash_duo.find_element("#input").send_keys(" hello")
  83 
  84     dash_duo.wait_for_text_to_equal("#output-container", "dash hello")
  85     _html = dash_duo.find_element("#output-container").get_property("innerHTML")
  86     assert _html == (
  87         '<div id="output" class="dash hello" '
  88         'style="font-family: &quot;dash hello&quot;;">dash hello</div>'
  89     )
  90 
  91     assert call_count.value == 7

 114 def test_rdmo004_multi_output_circular_dependencies(dash_duo):
 115     app = dash.Dash(__name__)
 116     app.layout = html.Div([dcc.Input(id="a"), dcc.Input(id="b"), html.P(id="c")])
 117 
 118     @app.callback(Output("a", "value"), [Input("b", "value")])
 119     def set_a(b):
 120         return ((b or "") + "X")[:100]
 121 
 122     @app.callback(
 123         [Output("b", "value"), Output("c", "children")], [Input("a", "value")]
 124     )
 125     def set_bc(a):
 126         return [a, a]
 127 
 128     dash_duo.start_server(
 129         app,
 130         debug=True,
 131         use_debugger=True,
 132         use_reloader=False,
 133         dev_tools_hot_reload=False,
 134     )
 135 
 136     # the UI still renders the output triggered by callback.
 137     # The new system does NOT loop infinitely like it used to, each callback
 138     # is invoked no more than once.
 139     dash_duo.wait_for_text_to_equal("#c", "X")
 140 
 141     err_text = dash_duo.find_element("span.dash-fe-error__title").text
 142     assert err_text == "Circular Dependencies"
