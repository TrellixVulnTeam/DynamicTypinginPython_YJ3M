  10 def test_rdsi001_state_and_inputs(dash_duo):
  11     app = dash.Dash(__name__)
  12     app.layout = html.Div(
  13         [
  14             dcc.Input(value="Initial Input", id="input"),
  15             dcc.Input(value="Initial State", id="state"),
  16             html.Div(id="output"),
  17         ]
  18     )
  19 
  20     call_count = Value("i", 0)
  21 
  22     @app.callback(
  23         Output("output", "children"),
  24         [Input("input", "value")],
  25         [State("state", "value")],
  26     )
  27     def update_output(input, state):
  28         call_count.value += 1
  29         return 'input="{}", state="{}"'.format(input, state)
  30 
  31     dash_duo.start_server(app)
  32 
  33     def input_():
  34         return dash_duo.find_element("#input")
  35 
  36     def output_():
  37         return dash_duo.find_element("#output")
  38 
  39     assert (
  40         output_().text == 'input="Initial Input", state="Initial State"'
  41     ), "callback gets called with initial input"
  42 
  43     input_().send_keys("x")
  44     wait.until(lambda: call_count.value == 2, timeout=1)
  45     assert (
  46         output_().text == 'input="Initial Inputx", state="Initial State"'
  47     ), "output get updated with key `x`"
  48 
  49     dash_duo.find_element("#state").send_keys("z")
  50     time.sleep(0.5)
  51     assert call_count.value == 2, "state not trigger callback with 0.5 wait"
  52     assert (
  53         output_().text == 'input="Initial Inputx", state="Initial State"'

  55

  58     assert (
  59         output_().text == 'input="Initial Inputxy", state="Initial Statez"'
  60     ), "both input and state value get updated by input callback"
