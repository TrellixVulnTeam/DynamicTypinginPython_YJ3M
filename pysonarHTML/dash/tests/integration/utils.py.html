  21 def wait_for(condition_function, get_message=lambda: "", *args, **kwargs):
  22     """Waits for condition_function to return True or raises WaitForTimeout.
  23 
  24     :param (function) condition_function: Should return True on success.
  25     :param (function) get_message:
  26     :param args: Optional args to pass to condition_function.
  27     :param kwargs: Optional kwargs to pass to condition_function.
  28         if `timeout` is in kwargs, it will be used to override TIMEOUT
  29     :raises: WaitForTimeout If condition_function doesn't return True in time.
  30     Usage:
  31         def get_element(selector):
  32             # some code to get some element or return a `False`-y value.
  33         selector = '.js-plotly-plot'
  34         try:
  35             wait_for(get_element, selector)
  36         except WaitForTimeout:
  37             self.fail('element never appeared...')
  38         plot = get_element(selector)  # we know it exists.
  39     """

  41     def wrapped_condition_function():
  42         """We wrap this to alter the call base on the closure."""
  43         if args and kwargs:
  44             return condition_function(*args, **kwargs)
  45         if args:
  46             return condition_function(*args)
  47         if kwargs:
  48             return condition_function(**kwargs)
  49         return condition_function()

  51     if "timeout" in kwargs:
  52         timeout = kwargs["timeout"]
  53         del kwargs["timeout"]
  54     else:
  55         timeout = TIMEOUT
  56

  59         if wrapped_condition_function():
  60             return True
