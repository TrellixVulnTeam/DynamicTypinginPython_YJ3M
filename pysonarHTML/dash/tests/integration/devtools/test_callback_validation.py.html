  14 def check_errors(dash_duo, specs):
  15     # Order-agnostic check of all the errors shown.
  16     # This is not fully general - despite the selectors below, it only applies
  17     # to front-end errors with no back-end errors in the list.
  18     cnt = len(specs)
  19     dash_duo.wait_for_text_to_equal(dash_duo.devtools_error_count_locator, str(cnt))
  20 
  21     found = []
  22     for i in range(cnt):
  23         msg = dash_duo.find_elements(".dash-fe-error__title")[i].text
  24         dash_duo.find_elements(".test-devtools-error-toggle")[i].click()
  25         dash_duo.wait_for_element(".dash-backend-error,.dash-fe-error__info")
  26         has_BE = dash_duo.driver.execute_script(
  27             "return document.querySelectorAll('.dash-backend-error').length"
  28         )
  29         txt_selector = ".dash-backend-error" if has_BE else ".dash-fe-error__info"
  30         txt = dash_duo.wait_for_element(txt_selector).text
  31         dash_duo.find_elements(".test-devtools-error-toggle")[i].click()
  32         dash_duo.wait_for_no_elements(".dash-backend-error")
  33         found.append((msg, txt))
  34 
  35     orig_found = found[:]
  36 
  37     for i, (message, snippets) in enumerate(specs):
  38         for j, (msg, txt) in enumerate(found):
  39             if msg == message and all(snip in txt for snip in snippets):
  40                 print(j)
  41                 found.pop(j)
  42                 break
  43         else:
  44             raise AssertionError(
  45                 (
  46                     "error {} ({}) not found with text:\n"
  47                     "  {}\nThe found messages were:\n---\n{}"
  48                 ).format(
  49                     i,
  50                     message,
  51                     "\n  ".join(snippets),
  52                     "\n---\n".join(
  53                         "{}\n{}".format(msg, txt) for msg, txt in orig_found
  54                     ),
  55                 )
  56             )
  57 
  58     # ensure the errors didn't leave items in the pendingCallbacks queue
  59     assert dash_duo.driver.execute_script("return document.title") == "Dash"

 698 def multipage_app(validation=False):
 699     app = Dash(__name__, suppress_callback_exceptions=(validation == "suppress"))
 700 
 701     skeleton = html.Div(
 702         [dcc.Location(id="url", refresh=False), html.Div(id="page-content")]
 703     )
 704 
 705     layout_index = html.Div(
 706         [
 707             dcc.Link('Navigate to "/page-1"', id="index_p1", href="/page-1"),
 708             dcc.Link('Navigate to "/page-2"', id="index_p2", href="/page-2"),
 709         ]
 710     )
 711 
 712     layout_page_1 = html.Div(
 713         [
 714             html.H2("Page 1"),
 715             dcc.Input(id="input-1-state", type="text", value="Montreal"),
 716             dcc.Input(id="input-2-state", type="text", value="Canada"),
 717             html.Button(id="submit-button", n_clicks=0, children="Submit"),
 718             html.Div(id="output-state"),
 719             html.Br(),
 720             dcc.Link('Navigate to "/"', id="p1_index", href="/"),
 721             dcc.Link('Navigate to "/page-2"', id="p1_p2", href="/page-2"),
 722         ]
 723     )
 724 
 725     layout_page_2 = html.Div(
 726         [
 727             html.H2("Page 2"),
 728             dcc.Input(id="page-2-input", value="LA"),
 729             html.Div(id="page-2-display-value"),
 730             html.Br(),
 731             dcc.Link('Navigate to "/"', id="p2_index", href="/"),
 732             dcc.Link('Navigate to "/page-1"', id="p2_p1", href="/page-1"),
 733         ]
 734     )
 735 
 736     validation_layout = html.Div([skeleton, layout_index, layout_page_1, layout_page_2])
 737 
 738     def validation_function():
 739         return skeleton if flask.has_request_context() else validation_layout
 740 
 741     app.layout = validation_function if validation == "function" else skeleton
 742     if validation == "attribute":
 743         app.validation_layout = validation_layout
 744 
 745     # Index callbacks
 746     @app.callback(Output("page-content", "children"), [Input("url", "pathname")])
 747     def display_page(pathname):
 748         if pathname == "/page-1":
 749             return layout_page_1
 750         elif pathname == "/page-2":
 751             return layout_page_2
 752         else:
 753             return layout_index
 754 
 755     # Page 1 callbacks
 756     @app.callback(
 757         Output("output-state", "children"),
 758         [Input("submit-button", "n_clicks")],
 759         [State("input-1-state", "value"), State("input-2-state", "value")],
 760     )
 761     def update_output(n_clicks, input1, input2):
 762         return (
 763             "The Button has been pressed {} times,"
 764             'Input 1 is "{}",'
 765             'and Input 2 is "{}"'
 766         ).format(n_clicks, input1, input2)
 767 
 768     # Page 2 callbacks
 769     @app.callback(
 770         Output("page-2-display-value", "children"), [Input("page-2-input", "value")]
 771     )
 772     def display_value(value):
 773         print("display_value")
 774         return 'You have selected "{}"'.format(value)
 775 
 776     return app
