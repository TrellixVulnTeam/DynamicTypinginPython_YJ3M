 178 def test_dvpc001_prop_check_errors_with_path(dash_duo):
 179     app = dash.Dash(__name__, eager_loading=True)
 180 
 181     app.layout = html.Div([html.Div(id="content"), dcc.Location(id="location")])
 182 
 183     @app.callback(Output("content", "children"), [Input("location", "pathname")])
 184     def display_content(pathname):
 185         if pathname is None or pathname == "/":
 186             return "Initial state"
 187         test_case = test_cases[pathname.strip("/")]
 188         return html.Div(
 189             id="new-component", children=test_case["component"](**test_case["props"])
 190         )
 191 
 192     dash_duo.start_server(
 193         app,
 194         debug=True,
 195         use_reloader=False,
 196         use_debugger=True,
 197         dev_tools_hot_reload=False,
 198     )
 199 
 200     for tc in test_cases:
 201         route_url = "{}/{}".format(dash_duo.server_url, tc)
 202         dash_duo.wait_for_page(url=route_url)
 203 
 204         if test_cases[tc]["fail"]:
 205             dash_duo.wait_for_element(".test-devtools-error-toggle").click()
 206             dash_duo.percy_snapshot(
 207                 "devtools validation exception: {}".format(test_cases[tc]["name"])
 208             )
 209         else:
 210             dash_duo.wait_for_element("#new-component")
 211             dash_duo.percy_snapshot(
 212                 "devtools validation no exception: {}".format(test_cases[tc]["name"])
 213             )
