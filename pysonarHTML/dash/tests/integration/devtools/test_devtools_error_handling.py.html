   9 def app_with_errors():
  10     app = dash.Dash(__name__)
  11 
  12     app.layout = html.Div(
  13         [
  14             html.Button(id="python", children="Python exception", n_clicks=0),
  15             html.Div(id="output"),
  16         ]
  17     )
  18 
  19     @app.callback(Output("output", "children"), [Input("python", "n_clicks")])
  20     def update_output(n_clicks):
  21         if n_clicks == 1:
  22             return bad_sub()

  24             raise Exception("Special 2 clicks exception")

 122 def test_dveh002_prevent_update_not_in_error_msg(dash_duo):
 123     # raising PreventUpdate shouldn't display the error message
 124     app = dash.Dash(__name__)
 125 
 126     app.layout = html.Div(
 127         [
 128             html.Button(id="python", children="Prevent update", n_clicks=0),
 129             html.Div(id="output"),
 130         ]
 131     )
 132 
 133     @app.callback(Output("output", "children"), [Input("python", "n_clicks")])
 134     def update_output(n_clicks):
 135         if n_clicks == 1:
 136             raise PreventUpdate
 137         if n_clicks == 2:
 138             raise Exception("An actual Python exception")
 139 
 140         return "button clicks: {}".format(n_clicks)
 141 
 142     dash_duo.start_server(
 143         app,
 144         debug=True,
 145         use_reloader=False,
 146         use_debugger=True,
 147         dev_tools_hot_reload=False,
 148     )
 149 
 150     for _ in range(3):
 151         dash_duo.find_element("#python").click()
 152 
 153     assert (
 154         dash_duo.find_element("#output").text == "button clicks: 3"
 155     ), "the click counts correctly in output"
 156 
 157     # two exceptions fired, but only a single exception appeared in the UI:
 158     # the prevent default was not displayed
 159     dash_duo.wait_for_text_to_equal(dash_duo.devtools_error_count_locator, "1")
 160     dash_duo.percy_snapshot("devtools - prevent update - only a single exception")
