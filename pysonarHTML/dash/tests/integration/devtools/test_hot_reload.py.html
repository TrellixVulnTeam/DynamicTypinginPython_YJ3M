  22 def replace_file(filename, new_content):
  23     path = os.path.join(os.path.dirname(__file__), "hr_assets", filename)
  24     with open(path, "r+") as fp:
  25         sleep(1)  # ensure a new mod time
  26         old_content = fp.read()
  27         fp.truncate(0)
  28         fp.seek(0)
  29         fp.write(new_content)
  30 
  31     return path, old_content

  34 def test_dvhr001_hot_reload(dash_duo):
  35     app = dash.Dash(__name__, assets_folder="hr_assets")
  36     app.layout = html.Div(
  37         [html.H3("Hot reload", id="text"), html.Button("Click", id="btn")],
  38         id="hot-reload-content",
  39     )
  40 
  41     @app.callback(Output("text", "children"), [Input("btn", "n_clicks")])
  42     def new_text(n):
  43         if not n:
  44             raise PreventUpdate
  45         return n
  46 
  47     hot_reload_settings = dict(
  48         dev_tools_hot_reload=True,
  49         dev_tools_ui=True,
  50         dev_tools_serve_dev_bundles=True,
  51         dev_tools_hot_reload_interval=0.1,
  52         dev_tools_hot_reload_max_retry=100,
  53     )
  54 
  55     dash_duo.start_server(app, **hot_reload_settings)
  56 
  57     # default overload color is blue
  58     dash_duo.wait_for_style_to_equal(
  59         "#hot-reload-content", "background-color", "rgba(0, 0, 255, 1)"
  60     )
  61 
  62     # set a global var - if we soft reload it should still be there,
  63     # hard reload will delete it
  64     dash_duo.driver.execute_script("window.someVar = 42;")
  65     assert dash_duo.driver.execute_script("return window.someVar") == 42
  66 
  67     soft_reload_file, old_soft = replace_file("hot_reload.css", RED_BG)
  68 
  69     try:
  70         # red is live changed during the test execution
  71         dash_duo.wait_for_style_to_equal(
  72             "#hot-reload-content", "background-color", "rgba(255, 0, 0, 1)"
  73         )
  74     finally:
  75         sleep(1)  # ensure a new mod time
  76         with open(soft_reload_file, "w") as f:
  77             f.write(old_soft)
  78 
  79     dash_duo.wait_for_style_to_equal(
  80         "#hot-reload-content", "background-color", "rgba(0, 0, 255, 1)"
  81     )
  82 
  83     # only soft reload, someVar is still there
  84     assert dash_duo.driver.execute_script("return window.someVar") == 42
  85 
  86     assert dash_duo.driver.execute_script("return window.cheese") == "roquefort"
  87 
  88     hard_reload_file, old_hard = replace_file("hot_reload.js", GOUDA)
  89 
  90     try:
  91         until(
  92             lambda: dash_duo.driver.execute_script("return window.cheese") == "gouda",
  93             timeout=3,
  94         )
  95     finally:
  96         sleep(1)  # ensure a new mod time
  97         with open(hard_reload_file, "w") as f:
  98             f.write(old_hard)
  99 
 100     until(
 101         lambda: dash_duo.driver.execute_script("return window.cheese") == "roquefort",
 102         timeout=3,
 103     )
 104 
 105     # we've done a hard reload so someVar is gone
 106     assert dash_duo.driver.execute_script("return window.someVar") is None
