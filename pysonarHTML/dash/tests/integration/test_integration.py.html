 114 def test_inin003_aborted_callback(dash_duo):
 115     """Raising PreventUpdate OR returning no_update prevents update and
 116     triggering dependencies."""
 117 
 118     initial_input = "initial input"
 119     initial_output = "initial output"
 120 
 121     app = Dash(__name__)
 122     app.layout = html.Div(
 123         [
 124             dcc.Input(id="input", value=initial_input),
 125             html.Div(initial_output, id="output1"),
 126             html.Div(initial_output, id="output2"),
 127         ]
 128     )
 129 
 130     callback1_count = Value("i", 0)
 131     callback2_count = Value("i", 0)
 132 
 133     @app.callback(Output("output1", "children"), [Input("input", "value")])
 134     def callback1(value):
 135         callback1_count.value += 1
 136         if callback1_count.value > 2:
 137             return no_update
 138         raise PreventUpdate("testing callback does not update")
 139         return value
 140 
 141     @app.callback(Output("output2", "children"), [Input("output1", "children")])
 142     def callback2(value):
 143         callback2_count.value += 1
 144         return value
 145 
 146     dash_duo.start_server(app)
 147 
 148     input_ = dash_duo.find_element("#input")
 149     input_.send_keys("xyz")
 150     dash_duo.wait_for_text_to_equal("#input", "initial inputxyz")
 151 
 152     until(
 153         lambda: callback1_count.value == 4,
 154         timeout=3,
 155         msg="callback1 runs 4x (initial page load and 3x through send_keys)",
 156     )
 157 
 158     assert (
 159         callback2_count.value == 0
 160     ), "callback2 is never triggered, even on initial load"
 161 
 162     # double check that output1 and output2 children were not updated
 163     assert dash_duo.find_element("#output1").text == initial_output
 164     assert dash_duo.find_element("#output2").text == initial_output
 165 
 166     assert not dash_duo.get_logs()
 167 
 168     dash_duo.percy_snapshot(name="aborted")

 409 def test_inin013_no_update_chains(dash_duo):
 410     app = Dash(__name__)
 411 
 412     app.layout = html.Div(
 413         [
 414             dcc.Input(id="a_in", value="a"),
 415             dcc.Input(id="b_in", value="b"),
 416             html.P("", id="a_out"),
 417             html.P("", id="a_out_short"),
 418             html.P("", id="b_out"),
 419             html.P("", id="ab_out"),
 420         ]
 421     )
 422 
 423     @app.callback(
 424         [Output("a_out", "children"), Output("a_out_short", "children")],
 425         [Input("a_in", "value")],
 426     )
 427     def a_out(a):
 428         return a, a if len(a) < 3 else no_update
 429 
 430     @app.callback(Output("b_out", "children"), [Input("b_in", "value")])
 431     def b_out(b):
 432         return b
 433 
 434     @app.callback(
 435         Output("ab_out", "children"),
 436         [Input("a_out_short", "children")],
 437         [State("b_out", "children")],
 438     )
 439     def ab_out(a, b):
 440         return a + " " + b
 441 
 442     dash_duo.start_server(app)
 443 
 444     a_in = dash_duo.find_element("#a_in")
 445     b_in = dash_duo.find_element("#b_in")
 446 
 447     b_in.send_keys("b")
 448     a_in.send_keys("a")
 449     dash_duo.wait_for_text_to_equal("#a_out", "aa")
 450     dash_duo.wait_for_text_to_equal("#b_out", "bb")
 451     dash_duo.wait_for_text_to_equal("#a_out_short", "aa")
 452     dash_duo.wait_for_text_to_equal("#ab_out", "aa bb")
 453 
 454     b_in.send_keys("b")
 455     a_in.send_keys("a")
 456     dash_duo.wait_for_text_to_equal("#a_out", "aaa")
 457     dash_duo.wait_for_text_to_equal("#b_out", "bbb")
 458     dash_duo.wait_for_text_to_equal("#a_out_short", "aa")
 459     # ab_out has not been triggered because a_out_short received no_update
 460     dash_duo.wait_for_text_to_equal("#ab_out", "aa bb")
 461 
 462     b_in.send_keys("b")
 463     a_in.send_keys(Keys.END)
 464     a_in.send_keys(Keys.BACKSPACE)
 465     dash_duo.wait_for_text_to_equal("#a_out", "aa")
 466     dash_duo.wait_for_text_to_equal("#b_out", "bbbb")
 467     dash_duo.wait_for_text_to_equal("#a_out_short", "aa")
 468     # now ab_out *is* triggered - a_out_short got a new value
 469     # even though that value is the same as the last value it got
 470     dash_duo.wait_for_text_to_equal("#ab_out", "aa bbbb")
