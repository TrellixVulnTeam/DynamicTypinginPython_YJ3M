 447         self._layout = value

 452             self._layout_is_function

 456 
 457             def simple_clone(c, children=None):
 458                 cls = type(c)
 459                 # in Py3 we can use the __init__ signature to reduce to just
 460                 # required args and id; in Py2 this doesn't work so we just
 461                 # empty out children.
 462                 sig = getattr(cls.__init__, "__signature__", None)
 463                 props = {
 464                     p: getattr(c, p)
 465                     for p in c._prop_names  # pylint: disable=protected-access
 466                     if hasattr(c, p)
 467                     and (
 468                         p == "id" or not sig or sig.parameters[p].default == c.REQUIRED
 469                     )
 470                 }
 471                 if props.get("children", children):
 472                     props["children"] = children or []
 473                 return cls(**props)
 474 
 475             layout_value = self._layout_value()
 476             _validate.validate_layout(value, layout_value)
 477             self.validation_layout = simple_clone(
 478                 # pylint: disable=protected-access
 479                 layout_value,
 480                 [simple_clone(c) for c in layout_value._traverse_ids()],

 542     def _collect_and_register_resources(self, resources):
 543         # now needs the app context.
 544         # template in the necessary component suite JS bundles
 545         # add the version number of the package as a query parameter
 546         # for cache busting
 547         def _relative_url_path(relative_package_path="", namespace=""):
 548 
 549             module_path = os.path.join(
 550                 os.path.dirname(sys.modules[namespace].__file__), relative_package_path
 551             )
 552 
 553             modified = int(os.stat(module_path).st_mtime)
 554 
 555             return "{}_dash-component-suites/{}/{}".format(
 556                 self.config.requests_pathname_prefix,
 557                 namespace,
 558                 build_fingerprint(
 559                     relative_package_path,
 560                     importlib.import_module(namespace).__version__,
 561                     modified,
 562                 ),
 563             )
 564 
 565         srcs = []
 566         for resource in resources:
 567             is_dynamic_resource = resource.get("dynamic", False)
 568 
 569             if "relative_package_path" in resource:
 570                 paths = resource["relative_package_path"]
 571                 paths = [paths] if isinstance(paths, str) else paths
 572 
 573                 for rel_path in paths:
 574                     self.registered_paths[resource["namespace"]].add(rel_path)
 575 
 576                     if not is_dynamic_resource:
 577                         srcs.append(
 578                             _relative_url_path(
 579                                 relative_package_path=rel_path,
 580                                 namespace=resource["namespace"],
 581                             )
 582                         )
 583             elif "external_url" in resource:
 584                 if not is_dynamic_resource:
 585                     if isinstance(resource["external_url"], str):
 586                         srcs.append(resource["external_url"])
 587                     else:
 588                         srcs += resource["external_url"]
 589             elif "absolute_path" in resource:
 590                 raise Exception("Serving files from absolute_path isn't supported yet")
 591             elif "asset_path" in resource:
 592                 static_url = self.get_asset_url(resource["asset_path"])
 593                 # Add a cache-busting query param
 594                 static_url += "?m={}".format(resource["ts"])
 595                 srcs.append(static_url)
 596         return srcs

 611     def _generate_scripts_html(self):
 612         # Dash renderer has dependencies like React which need to be rendered
 613         # before every other script. However, the dash renderer bundle
 614         # itself needs to be rendered after all of the component's
 615         # scripts have rendered.
 616         # The rest of the scripts can just be loaded after React but before
 617         # dash renderer.
 618         # pylint: disable=protected-access
 619 
 620         mode = "dev" if self._dev_tools["props_check"] is True else "prod"
 621 
 622         deps = []
 623         for js_dist_dependency in dash_renderer._js_dist_dependencies:
 624             dep = {}
 625             for key, value in js_dist_dependency.items():
 626                 dep[key] = value[mode] if isinstance(value, dict) else value
 627 
 628             deps.append(dep)
 629 
 630         dev = self._dev_tools.serve_dev_bundles

 850     def clientside_callback(
 851         self, clientside_function, output, inputs, state=(), prevent_initial_call=None
 852     ):
 853         """Create a callback that updates the output by calling a clientside
 854         (JavaScript) function instead of a Python function.
 855 
 856         Unlike `@app.callback`, `clientside_callback` is not a decorator:
 857         it takes either a
 858         `dash.dependencies.ClientsideFunction(namespace, function_name)`
 859         argument that describes which JavaScript function to call
 860         (Dash will look for the JavaScript function at
 861         `window.dash_clientside[namespace][function_name]`), or it may take
 862         a string argument that contains the clientside function source.
 863 
 864         For example, when using a `dash.dependencies.ClientsideFunction`:
 865         ```
 866         app.clientside_callback(
 867             ClientsideFunction('my_clientside_library', 'my_function'),
 868             Output('my-div' 'children'),
 869             [Input('my-input', 'value'),
 870              Input('another-input', 'value')]
 871         )
 872         ```
 873 
 874         With this signature, Dash's front-end will call
 875         `window.dash_clientside.my_clientside_library.my_function` with the
 876         current values of the `value` properties of the components `my-input`
 877         and `another-input` whenever those values change.
 878 
 879         Include a JavaScript file by including it your `assets/` folder. The
 880         file can be named anything but you'll need to assign the function's
 881         namespace to the `window.dash_clientside` namespace. For example,
 882         this file might look:
 883         ```
 884         window.dash_clientside = window.dash_clientside || {};
 885         window.dash_clientside.my_clientside_library = {
 886             my_function: function(input_value_1, input_value_2) {
 887                 return (
 888                     parseFloat(input_value_1, 10) +
 889                     parseFloat(input_value_2, 10)
 890                 );
 891             }
 892         }
 893         ```
 894 
 895         Alternatively, you can pass the JavaScript source directly to
 896         `clientside_callback`. In this case, the same example would look like:
 897         ```
 898         app.clientside_callback(
 899             '''
 900             function(input_value_1, input_value_2) {
 901                 return (
 902                     parseFloat(input_value_1, 10) +
 903                     parseFloat(input_value_2, 10)
 904                 );
 905             }
 906             ''',
 907             Output('my-div' 'children'),
 908             [Input('my-input', 'value'),
 909              Input('another-input', 'value')]
 910         )
 911         ```
 912 
 913         The last, optional argument `prevent_initial_call` causes the callback
 914         not to fire when its outputs are first added to the page. Defaults to
 915         `False` unless `prevent_initial_callbacks=True` at the app level.
 916         """
 917         self._insert_callback(output, inputs, state, prevent_initial_call)
 918 
 919         # If JS source is explicitly given, create a namespace and function
 920         # name, then inject the code.
 921         if isinstance(clientside_function, str):
 922 
 923             out0 = output
 924             if isinstance(output, (list, tuple)):
 925                 out0 = output[0]
 926 
 927             namespace = "_dashprivate_{}".format(out0.component_id)
 928             function_name = "{}".format(out0.component_property)
 929 
 930             self._inline_scripts.append(
 931                 _inline_clientside_template.format(
 932                     namespace=namespace.replace('"', '\\"'),
 933                     function_name=function_name.replace('"', '\\"'),
 934                     clientside_function=clientside_function,
 935                 )
 936             )
 937 
 938         # Callback is stored in an external asset.
 939         else:
 940             namespace = clientside_function.namespace
 941             function_name = clientside_function.function_name
 942 
 943         self._callback_list[-1]["clientside_function"] = {
 944             "namespace": namespace,
 945             "function_name": function_name,
 946         }

 948     def callback(self, output, inputs, state=(), prevent_initial_call=None):
 949         """
 950         Normally used as a decorator, `@app.callback` provides a server-side
 951         callback relating the values of one or more `output` items to one or
 952         more `input` items which will trigger the callback when they change,
 953         and optionally `state` items which provide additional information but
 954         do not trigger the callback directly.
 955 
 956         The last, optional argument `prevent_initial_call` causes the callback
 957         not to fire when its outputs are first added to the page. Defaults to
 958         `False` unless `prevent_initial_callbacks=True` at the app level.
 959         """
 960         callback_id = self._insert_callback(output, inputs, state, prevent_initial_call)
 961         multi = isinstance(output, (list, tuple))
 962 
 963         def wrap_func(func):
 964             @wraps(func)
 965             def add_context(*args, **kwargs):
 966                 output_spec = kwargs.pop("outputs_list")
 967 
 968                 # don't touch the comment on the next line - used by debugger
 969                 output_value = func(*args, **kwargs)  # %% callback invoked %%
 970 
 971                 if isinstance(output_value, _NoUpdate):
 972                     raise PreventUpdate
 973 
 974                 # wrap single outputs so we can treat them all the same
 975                 # for validation and response creation
 976                 if not multi:
 977                     output_value, output_spec = [output_value], [output_spec]
 978 
 979                 _validate.validate_multi_return(output_spec, output_value, callback_id)
 980 
 981                 component_ids = collections.defaultdict(dict)
 982                 has_update = False
 983                 for val, spec in zip(output_value, output_spec):
 984                     if isinstance(val, _NoUpdate):
 985                         continue
 986                     for vali, speci in (
 987                         zip(val, spec) if isinstance(spec, list) else [[val, spec]]
 988                     ):
 989                         if not isinstance(vali, _NoUpdate):
 990                             has_update = True
 991                             id_str = stringify_id(speci["id"])
 992                             component_ids[id_str][speci["property"]] = vali
 993 
 994                 if not has_update:
 995                     raise PreventUpdate
 996 
 997                 response = {"response": component_ids, "multi": True}
 998 
 999                 try:
1000                     jsonResponse = json.dumps(
1001                         response, cls=plotly.utils.PlotlyJSONEncoder
1002                     )
1003                 except TypeError:
1004                     _validate.fail_callback_output(output_value, output)
1005 
1006                 return jsonResponse
1007 
1008             self.callback_map[callback_id]["callback"] = add_context
1009 
1010             return add_context
1011 
1012         return wrap_func

1401         _reload = self._hot_reload
1402         with _reload.lock:
1403             _reload.hard = True
1404             _reload.hash = generate_hash()

1406             if self.config.assets_folder in filename:

1408                     os.path.relpath(
1409                         filename,
1410                         os.path.commonprefix([self.config.assets_folder, filename]),
1411                     )
1412                     .replace("\\", "/")
1413                     .lstrip("/")
1414                 )
1415 
1416                 _reload.changed_assets.append(
1417                     {
1418                         "url": self.get_asset_url(asset_path),
1419                         "modified": int(modified),
1420                         "is_css": filename.endswith("css"),
1421                     }
1422                 )
1423 
1424                 if filename not in self._assets_files and not deleted:
1425                     res = self._add_assets_resource(asset_path, filename)
1426                     if filename.endswith("js"):
1427                         self.scripts.append_script(res)
1428                     elif filename.endswith("css"):
1429                         self.css.append_css(res)
1430 
1431                 if deleted:
1432                     if filename in self._assets_files:
1433                         self._assets_files.remove(filename)

1436                         to_delete = None

1438                             if r.get("asset_path") == asset_path:
1439                                 to_delete = r
1440                                 break
1441                         if to_delete:
1442                             resources.remove(to_delete)
1443 
1444                     if filename.endswith("js"):
1445                         # pylint: disable=protected-access
