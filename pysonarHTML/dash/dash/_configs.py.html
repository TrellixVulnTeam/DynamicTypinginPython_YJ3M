  43 def get_combined_config(name, val, default=None):
  44     """Consolidate the config with priority from high to low provided init
  45     value > OS environ > default."""
  46 
  47     if val is not None:
  48         return val
  49 
  50     env = load_dash_env_vars().get("DASH_{}".format(name.upper()))
  51     if env is None:
  52         return default
  53 
  54     return env.lower() == "true" if env.lower() in {"true", "false"} else env

  57 def pathname_configs(
  58     url_base_pathname=None, routes_pathname_prefix=None, requests_pathname_prefix=None
  59 ):
  60     _pathname_config_error_message = """
  61     {} This is ambiguous.
  62     To fix this, set `routes_pathname_prefix` instead of `url_base_pathname`.
  63 
  64     Note that `requests_pathname_prefix` is the prefix for the AJAX calls that
  65     originate from the client (the web browser) and `routes_pathname_prefix` is
  66     the prefix for the API routes on the backend (this flask server).
  67     `url_base_pathname` will set `requests_pathname_prefix` and
  68     `routes_pathname_prefix` to the same value.
  69     If you need these to be different values then you should set
  70     `requests_pathname_prefix` and `routes_pathname_prefix`,
  71     not `url_base_pathname`.
  72     """
  73     url_base_pathname = get_combined_config("url_base_pathname", url_base_pathname)
  74 
  75     routes_pathname_prefix = get_combined_config(
  76         "routes_pathname_prefix", routes_pathname_prefix
  77     )
  78 
  79     requests_pathname_prefix = get_combined_config(
  80         "requests_pathname_prefix", requests_pathname_prefix
  81     )
  82 
  83     if url_base_pathname is not None and requests_pathname_prefix is not None:
  84         raise exceptions.InvalidConfig(
  85             _pathname_config_error_message.format(
  86                 "You supplied `url_base_pathname` and `requests_pathname_prefix`."
  87             )
  88         )
  89 
  90     if url_base_pathname is not None and routes_pathname_prefix is not None:
  91         raise exceptions.InvalidConfig(
  92             _pathname_config_error_message.format(
  93                 "You supplied `url_base_pathname` and `routes_pathname_prefix`."
  94             )
  95         )
  96 
  97     if url_base_pathname is not None and routes_pathname_prefix is None:
  98         routes_pathname_prefix = url_base_pathname
  99     elif routes_pathname_prefix is None:
 100         routes_pathname_prefix = "/"
 101 
 102     if not routes_pathname_prefix.startswith("/"):
 103         raise exceptions.InvalidConfig(
 104             "`routes_pathname_prefix` needs to start with `/`"
 105         )
 106     if not routes_pathname_prefix.endswith("/"):
 107         raise exceptions.InvalidConfig("`routes_pathname_prefix` needs to end with `/`")
 108 
 109     app_name = load_dash_env_vars().DASH_APP_NAME
 110 
 111     if not requests_pathname_prefix and app_name:
 112         requests_pathname_prefix = "/" + app_name + routes_pathname_prefix
 113     elif requests_pathname_prefix is None:
 114         requests_pathname_prefix = routes_pathname_prefix
 115 
 116     if not requests_pathname_prefix.startswith("/"):
 117         raise exceptions.InvalidConfig(
 118             "`requests_pathname_prefix` needs to start with `/`"
 119         )
 120     if not requests_pathname_prefix.endswith(routes_pathname_prefix):
 121         raise exceptions.InvalidConfig(
 122             "`requests_pathname_prefix` needs to ends with `routes_pathname_prefix`."
 123         )
 124 
 125     return url_base_pathname, routes_pathname_prefix, requests_pathname_prefix
