 101     def visit_and_snapshot(
 102         self,
 103         resource_path,
 104         hook_id,
 105         wait_for_callbacks=True,
 106         assert_check=True,
 107         stay_on_page=False,
 108     ):
 109         try:
 110             path = resource_path.lstrip("/")
 111             if path != resource_path:
 112                 logger.warning("we stripped the left '/' in resource_path")
 113             self.driver.get("{}/{}".format(self.server_url.rstrip("/"), path))
 114 
 115             # wait for the hook_id to present and all callbacks get fired
 116             self.wait_for_element_by_id(hook_id)
 117             self.percy_snapshot(path, wait_for_callbacks=wait_for_callbacks)
 118             if assert_check:
 119                 assert not self.driver.find_elements_by_css_selector(
 120                     "div.dash-debug-alert"
 121                 ), "devtools should not raise an error alert"
 122             if not stay_on_page:
 123                 self.driver.back()
 124         except WebDriverException as e:
 125             logger.exception("snapshot at resource %s error", path)
 126             raise e

 128     def percy_snapshot(self, name="", wait_for_callbacks=False):
 129         """percy_snapshot - visual test api shortcut to `percy_runner.snapshot`.
 130         It also combines the snapshot `name` with the Python version.
 131         """
 132         snapshot_name = "{} - py{}.{}".format(
 133             name, sys.version_info.major, sys.version_info.minor
 134         )
 135         logger.info("taking snapshot name => %s", snapshot_name)
 136         try:
 137             if wait_for_callbacks:
 138                 # the extra one second sleep adds safe margin in the context
 139                 # of wait_for_callbacks
 140                 time.sleep(1)
 141                 until(self._wait_for_callbacks, timeout=40, poll=0.3)
 142         except TestingTimeoutError:
 143             # API will log the error but this TimeoutError should not block
 144             # the test execution to continue and it will still do a snapshot
 145             # as diff reference for the build run.
 146             logger.error(
 147                 "wait_for_callbacks failed => status of invalid rqs %s",
 148                 self.redux_state_rqs,
 149             )
 150 
 151         self.percy_runner.snapshot(name=snapshot_name)

 153     def take_snapshot(self, name):
 154         """Hook method to take snapshot when a selenium test fails. The
 155         snapshot is placed under.
 156 
 157             - `/tmp/dash_artifacts` in linux
 158             - `%TEMP` in windows
 159         with a filename combining test case name and the
 160         running selenium session id
 161         """
 162         target = "/tmp/dash_artifacts" if not self._is_windows() else os.getenv("TEMP")
 163         if not os.path.exists(target):
 164             try:
 165                 os.mkdir(target)
 166             except OSError:
 167                 logger.exception("cannot make artifacts")
 168 
 169         self.driver.save_screenshot(
 170             "{}/{}_{}.png".format(target, name, self.session_id)
 171         )

 191     def _wait_for(self, method, args, timeout, msg):
 192         """Abstract generic pattern for explicit WebDriverWait."""
 193         _wait = (
 194             self._wd_wait if timeout is None else WebDriverWait(self.driver, timeout)
 195         )
 196         logger.debug(
 197             "method, timeout, poll => %s %s %s",
 198             method,
 199             _wait._timeout,  # pylint: disable=protected-access
 200             _wait._poll,  # pylint: disable=protected-access
 201         )
 202 
 203         return _wait.until(method(*args), msg)

 210     def wait_for_element_by_css_selector(self, selector, timeout=None):
 211         """Explicit wait until the element is present, timeout if not set,
 212         equals to the fixture's `wait_timeout` shortcut to `WebDriverWait` with
 213         `EC.presence_of_element_located`."""
 214         return self._wait_for(
 215             EC.presence_of_element_located,
 216             ((By.CSS_SELECTOR, selector),),
 217             timeout,
 218             "timeout {}s => waiting for selector {}".format(
 219                 timeout if timeout else self._wait_timeout, selector
 220             ),
 221         )

 223     def wait_for_no_elements(self, selector, timeout=None):
 224         """Explicit wait until an element is NOT found. timeout defaults to
 225         the fixture's `wait_timeout`."""
 226         until(
 227             # if we use get_elements it waits a long time to see if they appear
 228             # so this one calls out directly to execute_script
 229             lambda: self.driver.execute_script(
 230                 "return document.querySelectorAll('{}').length".format(selector)
 231             )
 232             == 0,
 233             timeout if timeout else self._wait_timeout,
 234         )

 294     def wait_for_page(self, url=None, timeout=10):
 295         """wait_for_page navigates to the url in webdriver wait until the
 296         renderer is loaded in browser.
 297 
 298         use the `server_url` if url is not provided.
 299         """
 300         self.driver.get(self.server_url if url is None else url)
 301         try:
 302             self.wait_for_element_by_css_selector(
 303                 self.dash_entry_locator, timeout=timeout
 304             )
 305         except TimeoutException:
 306             logger.exception("dash server is not loaded within %s seconds", timeout)
 307             logger.debug(self.get_logs())
 308             raise DashAppLoadingError(
 309                 "the expected Dash react entry point cannot be loaded"
 310                 " in browser\n HTML => {}\n Console Logs => {}\n".format(
 311                     self.driver.find_element_by_tag_name("body").get_property(
 312                         "innerHTML"
 313                     ),
 314                     "\n".join((str(log) for log in self.get_logs())),
 315                 )
 316             )
 317 
 318         if self._pause:
 319             try:
 320                 import pdb as pdb_
 321             except ImportError:
 322                 import ipdb as pdb_
 323 
 324             pdb_.set_trace()

 326     def select_dcc_dropdown(self, elem_or_selector, value=None, index=None):
 327         dropdown = self._get_element(elem_or_selector)
 328         dropdown.click()
 329 
 330         menu = dropdown.find_element_by_css_selector("div.Select-menu-outer")
 331         logger.debug("the available options are %s", "|".join(menu.text.split("\n")))
 332 
 333         options = menu.find_elements_by_css_selector("div.VirtualizedSelectOption")
 334         if options:
 335             if isinstance(index, int):
 336                 options[index].click()
 337                 return
 338 
 339             for option in options:
 340                 if option.text == value:
 341                     option.click()
 342                     return
 343 
 344         logger.error(
 345             "cannot find matching option using value=%s or index=%s", value, index
 346         )

 348     def toggle_window(self):
 349         """Switch between the current working window and the new opened one."""
 350         idx = (self._window_idx + 1) % 2
 351         self.switch_window(idx=idx)
 352         self._window_idx += 1

 354     def switch_window(self, idx=0):
 355         """Switch to window by window index shortcut to
 356         `driver.switch_to.window`."""
 357         if len(self.driver.window_handles) <= idx:
 358             raise BrowserError("there is no second window in Browser")
 359 
 360         self.driver.switch_to.window(self.driver.window_handles[idx])

 388     def _get_chrome(self):
 389         options = self._get_wd_options()
 390 
 391         capabilities = DesiredCapabilities.CHROME
 392         capabilities["loggingPrefs"] = {"browser": "SEVERE"}
 393 
 394         if "DASH_TEST_CHROMEPATH" in os.environ:
 395             options.binary_location = os.environ["DASH_TEST_CHROMEPATH"]
 396 
 397         options.add_experimental_option(
 398             "prefs",
 399             {
 400                 "download.default_directory": self.download_path,
 401                 "download.prompt_for_download": False,
 402                 "download.directory_upgrade": True,
 403                 "safebrowsing.enabled": False,
 404                 "safebrowsing.disable_download_protection": True,
 405             },
 406         )
 407 
 408         chrome = (
 409             webdriver.Remote(
 410                 command_executor=self._remote_url,
 411                 options=options,
 412                 desired_capabilities=capabilities,
 413             )
 414             if self._remote
 415             else webdriver.Chrome(options=options, desired_capabilities=capabilities)
 416         )
 417 
 418         # https://bugs.chromium.org/p/chromium/issues/detail?id=696481
 419         if self._headless:
 420             # pylint: disable=protected-access
 421             chrome.command_executor._commands["send_command"] = (
 422                 "POST",
 423                 "/session/$sessionId/chromium/send_command",
 424             )
 425             params = {
 426                 "cmd": "Page.setDownloadBehavior",
 427                 "params": {"behavior": "allow", "downloadPath": self.download_path},
 428             }
 429             res = chrome.execute("send_command", params)
 430             logger.debug("enabled headless download returns %s", res)
 431 
 432         chrome.set_window_position(0, 0)
 433         return chrome

 435     def _get_firefox(self):
 436         options = self._get_wd_options()
 437 
 438         capabilities = DesiredCapabilities.FIREFOX
 439         capabilities["loggingPrefs"] = {"browser": "SEVERE"}
 440         capabilities["marionette"] = True
 441 
 442         # https://developer.mozilla.org/en-US/docs/Download_Manager_preferences
 443         fp = webdriver.FirefoxProfile()
 444         fp.set_preference("browser.download.dir", self.download_path)
 445         fp.set_preference("browser.download.folderList", 2)
 446         fp.set_preference(
 447             "browser.helperApps.neverAsk.saveToDisk",
 448             "application/octet-stream",  # this MIME is generic for binary
 449         )
 450         return (
 451             webdriver.Remote(
 452                 command_executor=self._remote_url,
 453                 options=options,
 454                 desired_capabilities=capabilities,
 455             )
 456             if self._remote
 457             else webdriver.Firefox(
 458                 firefox_profile=fp, options=options, capabilities=capabilities
 459             )
 460         )

 471     def clear_input(self, elem_or_selector):
 472         """Simulate key press to clear the input."""
 473         elem = self._get_element(elem_or_selector)
 474         logger.debug("clear input with %s => %s", elem_or_selector, elem)
 475         (
 476             ActionChains(self.driver)
 477             .move_to_element(elem)
 478             .pause(0.2)
 479             .click(elem)
 480             .send_keys(Keys.END)
 481             .key_down(Keys.SHIFT)
 482             .send_keys(Keys.HOME)
 483             .key_up(Keys.SHIFT)
 484             .send_keys(Keys.DELETE)
 485         ).perform()

 487     def zoom_in_graph_by_ratio(
 488         self, elem_or_selector, start_fraction=0.5, zoom_box_fraction=0.2, compare=True
 489     ):
 490         """Zoom out a graph with a zoom box fraction of component dimension
 491         default start at middle with a rectangle of 1/5 of the dimension use
 492         `compare` to control if we check the svg get changed."""
 493         elem = self._get_element(elem_or_selector)
 494 
 495         prev = elem.get_attribute("innerHTML")
 496         w, h = elem.size["width"], elem.size["height"]
 497         try:
 498             ActionChains(self.driver).move_to_element_with_offset(
 499                 elem, w * start_fraction, h * start_fraction
 500             ).drag_and_drop_by_offset(
 501                 elem, w * zoom_box_fraction, h * zoom_box_fraction
 502             ).perform()
 503         except MoveTargetOutOfBoundsException:
 504             logger.exception("graph offset outside of the boundary")
 505         if compare:
 506             assert prev != elem.get_attribute(
 507                 "innerHTML"
 508             ), "SVG content should be different after zoom"

 517     def get_logs(self):
 518         """Return a list of `SEVERE` level logs after last reset time stamps
 519         (default to 0, resettable by `reset_log_timestamp`.
 520 
 521         Chrome only
 522         """
 523         if self.driver.name.lower() == "chrome":
 524             return [
 525                 entry
 526                 for entry in self.driver.get_log("browser")
 527                 if entry["timestamp"] > self._last_ts
 528             ]
 529         warnings.warn("get_logs always return None with webdrivers other than Chrome")
 530         return None

 532     def reset_log_timestamp(self):
 533         """reset_log_timestamp only work with chrome webdrier."""
 534         if self.driver.name.lower() == "chrome":
 535             entries = self.driver.get_log("browser")
 536             if entries:
 537                 self._last_ts = entries[-1]["timestamp"]
