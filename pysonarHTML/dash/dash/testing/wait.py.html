  12 def until(
  13     wait_cond, timeout, poll=0.1, msg="expected condition not met within timeout"
  14 ):  # noqa: C0330
  15     res = wait_cond()
  16     logger.debug(
  17         "start wait.until with method, timeout, poll => %s %s %s",
  18         wait_cond,
  19         timeout,
  20         poll,
  21     )
  22     end_time = time.time() + timeout
  23     while not res:
  24         if time.time() > end_time:
  25             raise TestingTimeoutError(msg)
  26         time.sleep(poll)
  27         res = wait_cond()
  28         logger.debug("poll => %s", time.time())
  29 
  30     return res

  54 class contains_text(object):
  55     def __init__(self, selector, text):
  56         self.selector = selector
  57         self.text = text
  58 
  59     def __call__(self, driver):
  60         try:
  61             elem = driver.find_element_by_css_selector(self.selector)
  62             logger.debug("contains text {%s} => expected %s", elem.text, self.text)
  63             return self.text in str(elem.text) or self.text in str(
  64                 elem.get_attribute("value")

  66         except WebDriverException:
