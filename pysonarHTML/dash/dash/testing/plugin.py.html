  19 def pytest_addoption(parser):
  20     dash = parser.getgroup("Dash", "Dash Integration Tests")
  21 
  22     dash.addoption(
  23         "--webdriver",
  24         choices=("Chrome", "Firefox"),
  25         default="Chrome",
  26         help="Name of the selenium driver to use",
  27     )
  28 
  29     dash.addoption(
  30         "--remote", action="store_true", help="instruct pytest to use selenium grid"
  31     )
  32 
  33     dash.addoption(
  34         "--remote-url",
  35         action="store",
  36         default=SELENIUM_GRID_DEFAULT,
  37         help="set a different selenium grid remote url if other than default",
  38     )
  39 
  40     dash.addoption(
  41         "--headless", action="store_true", help="set this flag to run in headless mode"
  42     )
  43 
  44     dash.addoption(
  45         "--percy-assets",
  46         action="store",
  47         default="tests/assets",
  48         help="configure how Percy will discover your app's assets",
  49     )
  50 
  51     dash.addoption(
  52         "--nopercyfinalize",
  53         action="store_false",
  54         help="set this flag to control percy finalize at CI level",
  55     )
  56 
  57     dash.addoption(
  58         "--pause",
  59         action="store_true",
  60         help="pause using pdb after opening the test app, so you can interact with it",

  76 @pytest.hookimpl(tryfirst=True, hookwrapper=True)
  77 def pytest_runtest_makereport(item, call):  # pylint: disable=unused-argument
  78     # execute all other hooks to obtain the report object
  79     outcome = yield
  80     rep = outcome.get_result()
  81 
  82     # we only look at actual failing test calls, not setup/teardown
  83     if rep.when == "call" and rep.failed and hasattr(item, "funcargs"):
  84         for name, fixture in item.funcargs.items():
  85             try:
  86                 if name in {"dash_duo", "dash_br", "dashr", "dashjl"}:
  87                     fixture.take_snapshot(item.name)
  88             except Exception as e:  # pylint: disable=broad-except
  89                 print(e)
