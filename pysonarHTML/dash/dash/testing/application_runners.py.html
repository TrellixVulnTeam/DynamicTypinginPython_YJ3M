  25 def import_app(app_file, application_name="app"):
  26     """Import a dash application from a module. The import path is in dot
  27     notation to the module. The variable named app will be returned.
  28 
  29     :Example:
  30 
  31         >>> app = import_app("my_app.app")
  32 
  33     Will import the application in module `app` of the package `my_app`.
  34 
  35     :param app_file: Path to the app (dot-separated).
  36     :type app_file: str
  37     :param application_name: The name of the dash application instance.
  38     :raise: dash_tests.errors.NoAppFoundError
  39     :return: App from module.
  40     :rtype: dash.Dash
  41     """
  42     try:
  43         app_module = runpy.run_module(app_file)
  44         app = app_module[application_name]
  45     except KeyError:
  46         logger.exception("the app name cannot be found")
  47         raise NoAppFoundError("No dash `app` instance was found in {}".format(app_file))
  48     return app

  51 class BaseDashRunner(object):
  52     """Base context manager class for running applications."""
  53 
  54     def __init__(self, keep_open, stop_timeout):
  55         self.port = 8050
  56         self.started = None
  57         self.keep_open = keep_open
  58         self.stop_timeout = stop_timeout
  59         self._tmp_app_path = None
  60 
  61     def start(self, *args, **kwargs):
  62         raise NotImplementedError  # pragma: no cover
  63 
  64     def stop(self):
  65         raise NotImplementedError  # pragma: no cover
  66 
  67     @staticmethod
  68     def accessible(url):
  69         try:
  70             requests.get(url)
  71         except requests.exceptions.RequestException:
  72             return False
  73         return True
  74 
  75     def __call__(self, *args, **kwargs):
  76         return self.start(*args, **kwargs)
  77 
  78     def __enter__(self):
  79         return self
  80 
  81     def __exit__(self, exc_type, exc_val, traceback):
  82         if self.started and not self.keep_open:
  83             try:
  84                 logger.info("killing the app runner")
  85                 self.stop()
  86             except TestingTimeoutError:
  87                 raise ServerCloseError(
  88                     "Cannot stop server within {}s timeout".format(self.stop_timeout)
  89                 )
  90         logger.info("__exit__ complete")
  91 
  92     @property
  93     def url(self):
  94         """The default server url."""
  95         return "http://localhost:{}".format(self.port)
  96 
  97     @property
  98     def is_windows(self):
  99         return sys.platform == "win32"
 100 
 101     @property
 102     def tmp_app_path(self):
 103         return self._tmp_app_path

 106 class ThreadedRunner(BaseDashRunner):
 107     """Runs a dash application in a thread.
 108 
 109     This is the default flavor to use in dash integration tests.
 110     """
 111 
 112     def __init__(self, keep_open=False, stop_timeout=3):
 113         super(ThreadedRunner, self).__init__(
 114             keep_open=keep_open, stop_timeout=stop_timeout
 115         )
 116         self.stop_route = "/_stop-{}".format(uuid.uuid4().hex)
 117         self.thread = None
 118 
 119     @staticmethod
 120     def _stop_server():
 121         # https://werkzeug.palletsprojects.com/en/0.15.x/serving/#shutting-down-the-server
 122         stopper = flask.request.environ.get("werkzeug.server.shutdown")
 123         if stopper is None:
 124             raise RuntimeError("Not running with the Werkzeug Server")
 125         stopper()
 126         return "Flask server is shutting down"
 127 
 128     # pylint: disable=arguments-differ,C0330
 129     def start(self, app, **kwargs):
 130         """Start the app server in threading flavor."""
 131         app.server.add_url_rule(self.stop_route, self.stop_route, self._stop_server)
 132 
 133         def _handle_error():
 134             self._stop_server()
 135 
 136         app.server.errorhandler(500)(_handle_error)
 137 
 138         def run():
 139             app.scripts.config.serve_locally = True
 140             app.css.config.serve_locally = True
 141             if "port" not in kwargs:
 142                 kwargs["port"] = self.port
 143             else:
 144                 self.port = kwargs["port"]
 145             app.run_server(threaded=True, **kwargs)
 146 
 147         self.thread = threading.Thread(target=run)
 148         self.thread.daemon = True
 149         try:
 150             self.thread.start()
 151         except RuntimeError:  # multiple call on same thread
 152             logger.exception("threaded server failed to start")
 153             self.started = False
 154 
 155         self.started = self.thread.is_alive()
 156 
 157         # wait until server is able to answer http request
 158         wait.until(lambda: self.accessible(self.url), timeout=1)
 159 
 160     def stop(self):
 161         requests.get("{}{}".format(self.url, self.stop_route))
 162         wait.until_not(self.thread.is_alive, self.stop_timeout)

 165 class ProcessRunner(BaseDashRunner):
 166     """Runs a dash application in a waitress-serve subprocess.
 167 
 168     This flavor is closer to production environment but slower.
 169     """
 170 
 171     def __init__(self, keep_open=False, stop_timeout=3):
 172         super(ProcessRunner, self).__init__(
 173             keep_open=keep_open, stop_timeout=stop_timeout
 174         )
 175         self.proc = None
 176 
 177     # pylint: disable=arguments-differ
 178     def start(
 179         self,
 180         app_module=None,
 181         application_name="app",
 182         raw_command=None,
 183         port=8050,
 184         start_timeout=3,
 185     ):
 186         """Start the server with waitress-serve in process flavor."""
 187         if not (app_module or raw_command):  # need to set a least one
 188             logging.error(
 189                 "the process runner needs to start with at least one valid command"
 190             )
 191             return
 192         self.port = port
 193         args = shlex.split(
 194             raw_command
 195             if raw_command
 196             else "waitress-serve --listen=0.0.0.0:{} {}:{}.server".format(
 197                 port, app_module, application_name
 198             ),
 199             posix=not self.is_windows,
 200         )
 201 
 202         logger.debug("start dash process with %s", args)
 203 
 204         try:
 205             self.proc = subprocess.Popen(
 206                 args, stdout=subprocess.PIPE, stderr=subprocess.PIPE
 207             )
 208             # wait until server is able to answer http request
 209             wait.until(lambda: self.accessible(self.url), timeout=start_timeout)
 210 
 211         except (OSError, ValueError):
 212             logger.exception("process server has encountered an error")
 213             self.started = False
 214             self.stop()
 215             return
 216 
 217         self.started = True
 218 
 219     def stop(self):
 220         if self.proc:
 221             try:
 222                 logger.info("proc.terminate with pid %s", self.proc.pid)
 223                 self.proc.terminate()
 224                 if self.tmp_app_path and os.path.exists(self.tmp_app_path):
 225                     logger.debug("removing temporary app path %s", self.tmp_app_path)
 226                     shutil.rmtree(self.tmp_app_path)
 227                 if utils.PY3:
 228                     # pylint:disable=no-member
 229                     _except = subprocess.TimeoutExpired
 230                     # pylint: disable=unexpected-keyword-arg
 231                     self.proc.communicate(timeout=self.stop_timeout)
 232                 else:
 233                     _except = Exception
 234                     logger.info("ruthless kill the process to avoid zombie")
 235                     self.proc.kill()
 236             except _except:
 237                 logger.exception(
 238                     "subprocess terminate not success, trying to kill "
 239                     "the subprocess in a safe manner"
 240                 )
 241                 self.proc.kill()
 242                 self.proc.communicate()
 243         logger.info("process stop completes!")

 246 class RRunner(ProcessRunner):
 247     def __init__(self, keep_open=False, stop_timeout=3):
 248         super(RRunner, self).__init__(keep_open=keep_open, stop_timeout=stop_timeout)
 249         self.proc = None
 250 
 251     # pylint: disable=arguments-differ
 252     def start(self, app, start_timeout=2, cwd=None):
 253         """Start the server with subprocess and Rscript."""
 254 
 255         if os.path.isfile(app) and os.path.exists(app):
 256             # app is already a file in a dir - use that as cwd
 257             if not cwd:
 258                 cwd = os.path.dirname(app)
 259                 logger.info("RRunner inferred cwd from app path: %s", cwd)
 260         else:
 261             # app is a string chunk, we make a temporary folder to store app.R
 262             # and its relevant assets
 263             self._tmp_app_path = os.path.join(
 264                 "/tmp" if not self.is_windows else os.getenv("TEMP"), uuid.uuid4().hex
 265             )
 266             try:
 267                 os.mkdir(self.tmp_app_path)
 268             except OSError:
 269                 logger.exception("cannot make temporary folder %s", self.tmp_app_path)
 270             path = os.path.join(self.tmp_app_path, "app.R")
 271 
 272             logger.info("RRunner start => app is R code chunk")
 273             logger.info("make a temporary R file for execution => %s", path)
 274             logger.debug("content of the dashR app")
 275             logger.debug("%s", app)
 276 
 277             with open(path, "w") as fp:
 278                 fp.write(app)
 279 
 280             app = path
 281 
 282             # try to find the path to the calling script to use as cwd
 283             if not cwd:
 284                 for entry in inspect.stack():
 285                     if "/dash/testing/" not in entry[1].replace("\\", "/"):
 286                         cwd = os.path.dirname(os.path.realpath(entry[1]))
 287                         logger.warning("get cwd from inspect => %s", cwd)
 288                         break
 289             if cwd:
 290                 logger.info("RRunner inferred cwd from the Python call stack: %s", cwd)
 291 
 292                 # try copying all valid sub folders (i.e. assets) in cwd to tmp
 293                 # note that the R assets folder name can be any valid folder name
 294                 assets = [
 295                     os.path.join(cwd, _)
 296                     for _ in os.listdir(cwd)
 297                     if not _.startswith("__") and os.path.isdir(os.path.join(cwd, _))
 298                 ]
 299 
 300                 for asset in assets:
 301                     target = os.path.join(self.tmp_app_path, os.path.basename(asset))
 302                     if os.path.exists(target):
 303                         logger.debug("delete existing target %s", target)
 304                         shutil.rmtree(target)
 305                     logger.debug("copying %s => %s", asset, self.tmp_app_path)
 306                     shutil.copytree(asset, target)
 307                     logger.debug("copied with %s", os.listdir(target))
 308 
 309             else:
 310                 logger.warning(
 311                     "RRunner found no cwd in the Python call stack. "
 312                     "You may wish to specify an explicit working directory "
 313                     "using something like: "
 314                     "dashr.run_server(app, cwd=os.path.dirname(__file__))"
 315                 )
 316 
 317         logger.info("Run dashR app with Rscript => %s", app)
 318 
 319         args = shlex.split(
 320             "Rscript -e 'source(\"{}\")'".format(os.path.realpath(app)),
 321             posix=not self.is_windows,
 322         )
 323         logger.debug("start dash process with %s", args)
 324 
 325         try:
 326             self.proc = subprocess.Popen(
 327                 args,
 328                 stdout=subprocess.PIPE,
 329                 stderr=subprocess.PIPE,
 330                 cwd=self.tmp_app_path if self.tmp_app_path else cwd,
 331             )
 332             # wait until server is able to answer http request
 333             wait.until(lambda: self.accessible(self.url), timeout=start_timeout)
 334 
 335         except (OSError, ValueError):
 336             logger.exception("process server has encountered an error")
 337             self.started = False
 338             return
 339 
 340         self.started = True
