  18     def _filter_resources(self, all_resources, dev_bundles=False):
  19         filtered_resources = []
  20         for s in all_resources:
  21             filtered_resource = {}
  22             if "dynamic" in s:
  23                 filtered_resource["dynamic"] = s["dynamic"]
  24             if "async" in s:
  25                 if "dynamic" in s:
  26                     raise exceptions.ResourceException(
  27                         "Can't have both 'dynamic' and 'async'. "
  28                         "{}".format(json.dumps(filtered_resource))
  29                     )
  30 
  31                 # Async assigns a value dynamically to 'dynamic'
  32                 # based on the value of 'async' and config.eager_loading
  33                 #
  34                 # True -> dynamic if the server is not eager, False otherwise
  35                 # 'lazy' -> always dynamic
  36                 # 'eager' -> dynamic if server is not eager
  37                 # (to prevent ever loading it)
  38                 filtered_resource["dynamic"] = (
  39                     not self.config.eager_loading
  40                     if s["async"] is True
  41                     else (s["async"] == "eager" and not self.config.eager_loading)
  42                     or s["async"] == "lazy"
  43                 )
  44             if "namespace" in s:
  45                 filtered_resource["namespace"] = s["namespace"]
  46             if "external_url" in s and not self.config.serve_locally:
  47                 filtered_resource["external_url"] = s["external_url"]
  48             elif "dev_package_path" in s and dev_bundles:
  49                 filtered_resource["relative_package_path"] = s["dev_package_path"]
  50             elif "relative_package_path" in s:
  51                 filtered_resource["relative_package_path"] = s["relative_package_path"]
  52             elif "absolute_path" in s:
  53                 filtered_resource["absolute_path"] = s["absolute_path"]
  54             elif "asset_path" in s:
  55                 info = os.stat(s["filepath"])
  56                 filtered_resource["asset_path"] = s["asset_path"]
  57                 filtered_resource["ts"] = info.st_mtime
  58             elif self.config.serve_locally:
  59                 warnings.warn(
  60                     (
  61                         "You have set your config to `serve_locally=True` but "
  62                         "A local version of {} is not available.\n"
  63                         "If you added this file with "
  64                         "`app.scripts.append_script` "
  65                         "or `app.css.append_css`, use `external_scripts` "
  66                         "or `external_stylesheets` instead.\n"
  67                         "See https://dash.plot.com/external-resources"
  68                     ).format(s["external_url"])
  69                 )
  70                 continue
  71             else:
  72                 raise exceptions.ResourceException(
  73                     "{} does not have a "
  74                     "relative_package_path, absolute_path, or an "
  75                     "external_url.".format(json.dumps(filtered_resource))
  76                 )
  77 
  78             filtered_resources.append(filtered_resource)
  79 
  80         return filtered_resources
  81 
  82     def get_all_resources(self, dev_bundles=False):
  83         lib_resources = ComponentRegistry.get_resources(self.resource_name)
  84         all_resources = lib_resources + self._resources
  85 
  86         return self._filter_resources(all_resources, dev_bundles)
