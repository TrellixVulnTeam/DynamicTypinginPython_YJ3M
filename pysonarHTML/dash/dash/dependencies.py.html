   4 class _Wildcard:  # pylint: disable=too-few-public-methods
   5     def __init__(self, name):
   6         self._name = name
   7 
   8     def __str__(self):
   9         return self._name
  10 
  11     def __repr__(self):
  12         return "<{}>".format(self)
  13 
  14     def to_json(self):
  15         # used in serializing wildcards - arrays are not allowed as
  16         # id values, so make the wildcards look like length-1 arrays.
  17         return '["{}"]'.format(self._name)

  25 class DashDependency:  # pylint: disable=too-few-public-methods
  26     def __init__(self, component_id, component_property):
  27         self.component_id = component_id
  28         self.component_property = component_property
  29 
  30     def __str__(self):
  31         return "{}.{}".format(self.component_id_str(), self.component_property)
  32 
  33     def __repr__(self):
  34         return "<{} `{}`>".format(self.__class__.__name__, self)
  35 
  36     def component_id_str(self):
  37         i = self.component_id
  38 
  39         def _dump(v):
  40             return json.dumps(v, sort_keys=True, separators=(",", ":"))
  41 
  42         def _json(k, v):
  43             vstr = v.to_json() if hasattr(v, "to_json") else json.dumps(v)
  44             return "{}:{}".format(json.dumps(k), vstr)
  45 
  46         if isinstance(i, dict):
  47             return "{" + ",".join(_json(k, i[k]) for k in sorted(i)) + "}"
  48 
  49         return i
  50 
  51     def to_dict(self):
  52         return {"id": self.component_id_str(), "property": self.component_property}
  53 
  54     def __eq__(self, other):
  55         """
  56         We use "==" to denote two deps that refer to the same prop on
  57         the same component. In the case of wildcard deps, this means
  58         the same prop on *at least one* of the same components.
  59         """
  60         return (
  61             isinstance(other, DashDependency)
  62             and self.component_property == other.component_property
  63             and self._id_matches(other)
  64         )
  65 
  66     def _id_matches(self, other):
  67         my_id = self.component_id
  68         other_id = other.component_id
  69         self_dict = isinstance(my_id, dict)
  70         other_dict = isinstance(other_id, dict)
  71 
  72         if self_dict != other_dict:
  73             return False
  74         if self_dict:
  75             if set(my_id.keys()) != set(other_id.keys()):
  76                 return False
  77 
  78             for k, v in my_id.items():
  79                 other_v = other_id[k]
  80                 if v == other_v:
  81                     continue
  82                 v_wild = isinstance(v, _Wildcard)
  83                 other_wild = isinstance(other_v, _Wildcard)
  84                 if v_wild or other_wild:
  85                     if not (v_wild and other_wild):
  86                         continue  # one wild, one not
  87                     if v is ALL or other_v is ALL:
  88                         continue  # either ALL
  89                     if v is MATCH or other_v is MATCH:
  90                         return False  # one MATCH, one ALLSMALLER
  91                 else:
  92                     return False
  93             return True
  94 
  95         # both strings
  96         return my_id == other_id
  97 
  98     def __hash__(self):
  99         return hash(str(self))

 120 class ClientsideFunction:  # pylint: disable=too-few-public-methods
 121     def __init__(self, namespace=None, function_name=None):
 122 
 123         if namespace.startswith("_dashprivate_"):
 124             raise ValueError("Namespaces cannot start with '_dashprivate_'.")
 125 
 126         if namespace in ["PreventUpdate", "no_update"]:
 127             raise ValueError(
 128                 '"{}" is a forbidden namespace in' " dash_clientside.".format(namespace)
 129             )
 130 
 131         self.namespace = namespace
 132         self.function_name = function_name
 133 
 134     def __repr__(self):
 135         return "ClientsideFunction({}, {})".format(self.namespace, self.function_name)
