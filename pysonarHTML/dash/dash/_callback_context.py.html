   7 def has_context(func):
   8     @functools.wraps(func)
   9     def assert_context(*args, **kwargs):
  10         if not flask.has_request_context():
  11             raise exceptions.MissingCallbackContextException(
  12                 "dash.callback_context.{} is only available from a callback!".format(
  13                     getattr(func, "__name__")
  14                 )
  15             )
  16         return func(*args, **kwargs)
  17 
  18     return assert_context
