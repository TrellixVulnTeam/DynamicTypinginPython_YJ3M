  19 def validate_callback_args(args, cls):
  20     name = cls.__name__
  21     if not isinstance(args, (list, tuple)):
  22         raise exceptions.IncorrectTypeException(
  23             """
  24             The {} argument `{}` must be a list or tuple of
  25             `dash.dependencies.{}`s.
  26             """.format(
  27                 name.lower(), str(args), name
  28             )
  29         )
  30 
  31     for arg in args:
  32         if not isinstance(arg, cls):
  33             raise exceptions.IncorrectTypeException(
  34                 """
  35                 The {} argument `{}` must be of type `dash.dependencies.{}`.
  36                 """.format(
  37                     name.lower(), str(arg), name
  38                 )
  39             )
  40 
  41         if not isinstance(getattr(arg, "component_property", None), _strings):
  42             raise exceptions.IncorrectTypeException(
  43                 """
  44                 component_property must be a string, found {!r}
  45                 """.format(
  46                     arg.component_property
  47                 )
  48             )
  49 
  50         if hasattr(arg, "component_event"):
  51             raise exceptions.NonExistentEventException(
  52                 """
  53                 Events have been removed.
  54                 Use the associated property instead.
  55                 """
  56             )
  57 
  58         if isinstance(arg.component_id, dict):
  59             validate_id_dict(arg)
  60 
  61         elif isinstance(arg.component_id, _strings):
  62             validate_id_string(arg)
  63 
  64         else:
  65             raise exceptions.IncorrectTypeException(
  66                 """
  67                 component_id must be a string or dict, found {!r}
  68                 """.format(
  69                     arg.component_id
  70                 )
  71             )
  72

  74 def validate_id_dict(arg):
  75     arg_id = arg.component_id
  76 
  77     for k in arg_id:
  78         # Need to keep key type validation on the Python side, since
  79         # non-string keys will be converted to strings in json.dumps and may
  80         # cause unwanted collisions
  81         if not isinstance(k, _strings):
  82             raise exceptions.IncorrectTypeException(
  83                 """
  84                 Wildcard ID keys must be non-empty strings,
  85                 found {!r} in id {!r}
  86                 """.format(
  87                     k, arg_id
  88                 )
  89             )

  92 def validate_id_string(arg):
  93     arg_id = arg.component_id
  94 
  95     invalid_chars = ".{"
  96     invalid_found = [x for x in invalid_chars if x in arg_id]
  97     if invalid_found:
  98         raise exceptions.InvalidComponentIdError(
  99             """
 100             The element `{}` contains `{}` in its ID.
 101             Characters `{}` are not allowed in IDs.
 102             """.format(
 103                 arg_id, "`, `".join(invalid_found), "`, `".join(invalid_chars)
 104             )
 105         )

 108 def validate_multi_return(outputs_list, output_value, callback_id):
 109     if not isinstance(output_value, (list, tuple)):
 110         raise exceptions.InvalidCallbackReturnValue(
 111             """
 112             The callback {} is a multi-output.
 113             Expected the output type to be a list or tuple but got:
 114             {}.
 115             """.format(
 116                 callback_id, repr(output_value)
 117             )
 118         )
 119 
 120     if len(output_value) != len(outputs_list):
 121         raise exceptions.InvalidCallbackReturnValue(
 122             """
 123             Invalid number of output values for {}.
 124             Expected {}, got {}
 125             """.format(
 126                 callback_id, len(outputs_list), len(output_value)
 127             )
 128         )
 129 
 130     for i, outi in enumerate(outputs_list):
 131         if isinstance(outi, list):
 132             vi = output_value[i]
 133             if not isinstance(vi, (list, tuple)):
 134                 raise exceptions.InvalidCallbackReturnValue(
 135                     """
 136                     The callback {} output {} is a wildcard multi-output.
 137                     Expected the output type to be a list or tuple but got:
 138                     {}.
 139                     output spec: {}
 140                     """.format(
 141                         callback_id, i, repr(vi), repr(outi)
 142                     )
 143                 )
 144 
 145             if len(vi) != len(outi):
 146                 raise exceptions.InvalidCallbackReturnValue(
 147                     """
 148                     Invalid number of output values for {} item {}.
 149                     Expected {}, got {}
 150                     output spec: {}
 151                     output value: {}
 152                     """.format(
 153                         callback_id, i, len(vi), len(outi), repr(outi), repr(vi)
 154                     )
 155                 )

 158 def fail_callback_output(output_value, output):
 159     valid = _strings + (dict, int, float, type(None), Component)
 160 
 161     def _raise_invalid(bad_val, outer_val, path, index=None, toplevel=False):
 162         bad_type = type(bad_val).__name__
 163         outer_id = (
 164             "(id={:s})".format(outer_val.id) if getattr(outer_val, "id", False) else ""
 165         )
 166         outer_type = type(outer_val).__name__
 167         if toplevel:
 168             location = """
 169             The value in question is either the only value returned,
 170             or is in the top level of the returned list,
 171             """
 172         else:
 173             index_string = "[*]" if index is None else "[{:d}]".format(index)
 174             location = """
 175             The value in question is located at
 176             {} {} {}
 177             {},
 178             """.format(
 179                 index_string, outer_type, outer_id, path
 180             )
 181 
 182         raise exceptions.InvalidCallbackReturnValue(
 183             """
 184             The callback for `{output}`
 185             returned a {object:s} having type `{type}`
 186             which is not JSON serializable.
 187 
 188             {location}
 189             and has string representation
 190             `{bad_val}`
 191 
 192             In general, Dash properties can only be
 193             dash components, strings, dictionaries, numbers, None,
 194             or lists of those.
 195             """.format(
 196                 output=repr(output),
 197                 object="tree with one value" if not toplevel else "value",
 198                 type=bad_type,
 199                 location=location,
 200                 bad_val=bad_val,
 201             )
 202         )
 203 
 204     def _value_is_valid(val):
 205         return isinstance(val, valid)
 206 
 207     def _validate_value(val, index=None):
 208         # val is a Component
 209         if isinstance(val, Component):
 210             # pylint: disable=protected-access
 211             for p, j in val._traverse_with_paths():
 212                 # check each component value in the tree
 213                 if not _value_is_valid(j):
 214                     _raise_invalid(bad_val=j, outer_val=val, path=p, index=index)
 215 
 216                 # Children that are not of type Component or
 217                 # list/tuple not returned by traverse
 218                 child = getattr(j, "children", None)
 219                 if not isinstance(child, (tuple, collections.MutableSequence)):
 220                     if child and not _value_is_valid(child):
 221                         _raise_invalid(
 222                             bad_val=child,
 223                             outer_val=val,
 224                             path=p + "\n" + "[*] " + type(child).__name__,
 225                             index=index,
 226                         )
 227 
 228             # Also check the child of val, as it will not be returned
 229             child = getattr(val, "children", None)
 230             if not isinstance(child, (tuple, collections.MutableSequence)):
 231                 if child and not _value_is_valid(child):
 232                     _raise_invalid(
 233                         bad_val=child,
 234                         outer_val=val,
 235                         path=type(child).__name__,
 236                         index=index,
 237                     )
 238 
 239         # val is not a Component, but is at the top level of tree
 240         elif not _value_is_valid(val):
 241             _raise_invalid(
 242                 bad_val=val,
 243                 outer_val=type(val).__name__,
 244                 path="",
 245                 index=index,
 246                 toplevel=True,
 247             )
 248 
 249     if isinstance(output_value, list):
 250         for i, val in enumerate(output_value):
 251             _validate_value(val, index=i)
 252     else:
 253         _validate_value(output_value)
 254 
 255     # if we got this far, raise a generic JSON error
 256     raise exceptions.InvalidCallbackReturnValue(
 257         """
 258         The callback for property `{property:s}` of component `{id:s}`
 259         returned a value which is not JSON serializable.
 260 
 261         In general, Dash properties can only be dash components, strings,
 262         dictionaries, numbers, None, or lists of those.
 263         """.format(
 264             property=output.component_property, id=output.component_id
 265         )
 266     )

 269 def check_obsolete(kwargs):
 270     for key in kwargs:
 271         if key in ["components_cache_max_age", "static_folder"]:
 272             raise exceptions.ObsoleteKwargException(
 273                 """
 274                 {} is no longer a valid keyword argument in Dash since v1.0.
 275                 See https://dash.plotly.com for details.
 276                 """.format(
 277                     key
 278                 )
 279             )
 280         # any other kwarg mimic the built-in exception
 281         raise TypeError("Dash() got an unexpected keyword argument '" + key + "'")

 284 def validate_js_path(registered_paths, package_name, path_in_package_dist):
 285     if package_name not in registered_paths:
 286         raise exceptions.DependencyException(
 287             """
 288             Error loading dependency. "{}" is not a registered library.
 289             Registered libraries are:
 290             {}
 291             """.format(
 292                 package_name, list(registered_paths.keys())
 293             )
 294         )
 295 
 296     if path_in_package_dist not in registered_paths[package_name]:
 297         raise exceptions.DependencyException(
 298             """
 299             "{}" is registered but the path requested is not valid.
 300             The path requested: "{}"
 301             List of registered paths: {}
 302             """.format(
 303                 package_name, path_in_package_dist, registered_paths
 304             )
 305         )

 308 def validate_index(name, checks, index):
 309     missing = [i for check, i in checks if not re.compile(check).search(index)]
 310     if missing:
 311         plural = "s" if len(missing) > 1 else ""
 312         raise exceptions.InvalidIndexException(
 313             "Missing item{pl} {items} in {name}.".format(
 314                 items=", ".join(missing), pl=plural, name=name
 315             )
 316         )

 319 def validate_layout_type(value):

 321         raise exceptions.NoLayoutException(
 322             "Layout must be a dash component "
 323             "or a function that returns a dash component."

 327 def validate_layout(layout, layout_value):
 328     if layout is None:
 329         raise exceptions.NoLayoutException(
 330             """
 331             The layout was `None` at the time that `run_server` was called.
 332             Make sure to set the `layout` attribute of your application
 333             before running the server.
 334             """
 335         )
 336 
 337     layout_id = stringify_id(getattr(layout_value, "id", None))
 338 
 339     component_ids = {layout_id} if layout_id else set()
 340     for component in layout_value._traverse():  # pylint: disable=protected-access
 341         component_id = stringify_id(getattr(component, "id", None))
 342         if component_id and component_id in component_ids:
 343             raise exceptions.DuplicateIdError(
 344                 """
 345                 Duplicate component id found in the initial layout: `{}`
 346                 """.format(
 347                     component_id
 348                 )
 349             )
 350         component_ids.add(component_id)
