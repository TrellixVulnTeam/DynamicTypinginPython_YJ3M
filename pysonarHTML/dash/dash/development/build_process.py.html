  20 class BuildProcess(object):
  21     def __init__(self, main, deps_info):
  22         self.logger = logger
  23         self.main = main
  24         self.deps_info = deps_info
  25         self.npm_modules = self._concat(self.main, "node_modules")
  26         self.package_lock = self._concat(self.main, "package-lock.json")
  27         self.package = self._concat(self.main, "package.json")
  28         self._parse_package(path=self.package)
  29         self.asset_paths = (self.build_folder, self.npm_modules)
  30 
  31     def _parse_package(self, path):
  32         with open(path, "r") as fp:
  33             package = json.load(fp)
  34             self.version = package["version"]
  35             self.name = package["name"]
  36             self.build_folder = self._concat(self.main, self.name.replace("-", "_"))
  37             self.deps = package["dependencies"]
  38 
  39     @staticmethod
  40     def _concat(*paths):
  41         return os.path.realpath(os.path.sep.join((path for path in paths if path)))
  42 
  43     @staticmethod
  44     def _clean_path(path):
  45         if os.path.exists(path):
  46             logger.warning("🚨 %s already exists, remove it!", path)
  47             try:
  48                 if os.path.isfile(path):
  49                     os.remove(path)
  50                 if os.path.isdir(path):
  51                     shutil.rmtree(path)
  52             except OSError:
  53                 sys.exit(1)
  54         else:
  55             logger.warning("🚨 %s doesn't exist, no action taken", path)
  56 
  57     @job("clean all the previous assets generated by build tool")
  58     def clean(self):
  59         for path in self.asset_paths:
  60             self._clean_path(path)
  61

  73     @job("run the whole building process in sequence")
  74     def build(self, build=None):
  75         self.clean()
  76         self.npm()
  77         self.bundles(build)
  78         self.digest()

  80     @job("compute the hash digest for assets")
  81     def digest(self):
  82         copies = tuple(
  83             _
  84             for _ in os.listdir(self.build_folder)
  85             if os.path.splitext(_)[-1] in {".js", ".map"}
  86         )
  87         logger.info("bundles in %s %s", self.build_folder, copies)
  88 
  89         payload = {self.name: self.version}
  90         for copy in copies:
  91             payload["MD5 ({})".format(copy)] = compute_md5(
  92                 self._concat(self.build_folder, copy)
  93             )
  94 
  95         with open(self._concat(self.main, "digest.json"), "w") as fp:
  96             json.dump(payload, fp, sort_keys=True, indent=4, separators=(",", ":"))
  97         logger.info(
  98             "bundle digest in digest.json:\n%s",
  99             json.dumps(payload, sort_keys=True, indent=4),
 100         )

 102     @job("copy and generate the bundles")
 103     def bundles(self, build=None):
 104         if not os.path.exists(self.build_folder):
 105             try:
 106                 os.makedirs(self.build_folder)
 107             except OSError:
 108                 logger.exception("🚨 having issues manipulating %s", self.build_folder)
 109                 sys.exit(1)
 110 
 111         self._parse_package(self.package_lock)
 112 
 113         getattr(self, "_bundles_extra", lambda: None)()
 114 
 115         versions = {
 116             "version": self.version,
 117             "package": self.name.replace(" ", "_").replace("-", "_"),
 118         }
 119         for scope, name, subfolder, filename, target in self.deps_info:
 120             version = self.deps["/".join(filter(None, [scope, name]))]["version"]
 121             versions[name.replace("-", "").replace(".", "")] = version
 122 
 123             logger.info("copy npm dependency => %s", filename)
 124             ext = "min.js" if "min" in filename.split(".") else "js"
 125             target = (
 126                 target.format(version)
 127                 if target
 128                 else "{}@{}.{}".format(name, version, ext)
 129             )
 130             shutil.copyfile(
 131                 self._concat(self.npm_modules, scope, name, subfolder, filename),
 132                 self._concat(self.build_folder, target),
 133             )
 134 
 135         _script = "build:dev" if build == "local" else "build:js"
 136         logger.info("run `npm run %s`", _script)
 137         os.chdir(self.main)
 138         run_command_with_process("npm run {}".format(_script))
 139 
 140         logger.info("generate the `__init__.py` from template and versions")
 141         with open(self._concat(self.main, "init.template")) as fp:
 142             t = string.Template(fp.read())
 143 
 144         with open(self._concat(self.build_folder, "__init__.py"), "w") as fp:
 145             fp.write(t.safe_substitute(versions))
