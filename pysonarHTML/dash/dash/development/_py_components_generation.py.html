  12 def generate_class_string(typename, props, description, namespace):
  13     """Dynamically generate class strings to have nicely formatted docstrings,
  14     keyword arguments, and repr.
  15 
  16     Inspired by http://jameso.be/2013/08/06/namedtuple.html
  17 
  18     Parameters
  19     ----------
  20     typename
  21     props
  22     description
  23     namespace
  24 
  25     Returns
  26     -------
  27     string
  28     """
  29     # TODO _prop_names, _type, _namespace, and available_properties
  30     # can be modified by a Dash JS developer via setattr
  31     # TODO - Tab out the repr for the repr of these components to make it
  32     # look more like a hierarchical tree
  33     # TODO - Include "description" "defaultValue" in the repr and docstring
  34     #
  35     # TODO - Handle "required"
  36     #
  37     # TODO - How to handle user-given `null` values? I want to include
  38     # an expanded docstring like Dropdown(value=None, id=None)
  39     # but by templating in those None values, I have no way of knowing
  40     # whether a property is None because the user explicitly wanted
  41     # it to be `null` or whether that was just the default value.
  42     # The solution might be to deal with default values better although
  43     # not all component authors will supply those.
  44     c = '''class {typename}(Component):
  45     """{docstring}"""
  46     @_explicitize_args
  47     def __init__(self, {default_argtext}):
  48         self._prop_names = {list_of_valid_keys}
  49         self._type = '{typename}'
  50         self._namespace = '{namespace}'
  51         self._valid_wildcard_attributes =\
  52             {list_of_valid_wildcard_attr_prefixes}
  53         self.available_properties = {list_of_valid_keys}
  54         self.available_wildcard_properties =\
  55             {list_of_valid_wildcard_attr_prefixes}
  56 
  57         _explicit_args = kwargs.pop('_explicit_args')
  58         _locals = locals()
  59         _locals.update(kwargs)  # For wildcard attrs
  60         args = {{k: _locals[k] for k in _explicit_args if k != 'children'}}
  61 
  62         for k in {required_props}:
  63             if k not in args:
  64                 raise TypeError(
  65                     'Required argument `' + k + '` was not specified.')
  66         super({typename}, self).__init__({argtext})
  67 '''
  68 
  69     filtered_props = reorder_props(filter_props(props))
  70     wildcard_prefixes = repr(parse_wildcards(props))
  71     list_of_valid_keys = repr(list(map(str, filtered_props.keys())))
  72     docstring = create_docstring(
  73         component_name=typename, props=filtered_props, description=description
  74     ).replace("\r\n", "\n")
  75 
  76     prohibit_events(props)
  77 
  78     # pylint: disable=unused-variable
  79     prop_keys = list(props.keys())
  80     if "children" in props:
  81         prop_keys.remove("children")
  82         default_argtext = "children=None, "
  83         argtext = "children=children, **args"
  84     else:
  85         default_argtext = ""
  86         argtext = "**args"
  87     default_argtext += ", ".join(
  88         [
  89             (
  90                 "{:s}=Component.REQUIRED".format(p)
  91                 if props[p]["required"]
  92                 else "{:s}=Component.UNDEFINED".format(p)
  93             )
  94             for p in prop_keys
  95             if not p.endswith("-*") and p not in python_keywords and p != "setProps"
  96         ]
  97         + ["**kwargs"]
  98     )
  99     required_args = required_props(props)
 100     return c.format(
 101         typename=typename,
 102         namespace=namespace,
 103         filtered_props=filtered_props,
 104         list_of_valid_wildcard_attr_prefixes=wildcard_prefixes,
 105         list_of_valid_keys=list_of_valid_keys,
 106         docstring=docstring,
 107         default_argtext=default_argtext,
 108         argtext=argtext,
 109         required_props=required_args,
 110     )

 113 def generate_class_file(typename, props, description, namespace):
 114     """Generate a Python class file (.py) given a class string.
 115 
 116     Parameters
 117     ----------
 118     typename
 119     props
 120     description
 121     namespace
 122 
 123     Returns
 124     -------
 125     """
 126     import_string = (
 127         "# AUTO GENERATED FILE - DO NOT EDIT\n\n"
 128         + "from dash.development.base_component import "
 129         + "Component, _explicitize_args\n\n\n"
 130     )
 131     class_string = generate_class_string(typename, props, description, namespace)
 132     file_name = "{:s}.py".format(typename)
 133 
 134     file_path = os.path.join(namespace, file_name)
 135     with open(file_path, "w") as f:
 136         f.write(import_string)
 137         f.write(class_string)
 138 
 139     print("Generated {}".format(file_name))

 142 def generate_imports(project_shortname, components):
 143     with open(os.path.join(project_shortname, "_imports_.py"), "w") as f:
 144         imports_string = "{}\n\n{}".format(
 145             "\n".join("from .{0} import {0}".format(x) for x in components),
 146             "__all__ = [\n{}\n]".format(
 147                 ",\n".join('    "{}"'.format(x) for x in components)
 148             ),
 149         )
 150 
 151         f.write(imports_string)

 154 def generate_classes_files(project_shortname, metadata, *component_generators):
 155     components = []
 156     for component_path, component_data in metadata.items():
 157         component_name = component_path.split("/")[-1].split(".")[0]
 158         components.append(component_name)
 159 
 160         for generator in component_generators:
 161             generator(
 162                 component_name,
 163                 component_data["props"],
 164                 component_data["description"],
 165                 project_shortname,
 166             )
 167 
 168     return components

 171 def generate_class(typename, props, description, namespace):
 172     """Generate a Python class object given a class string.
 173 
 174     Parameters
 175     ----------
 176     typename
 177     props
 178     description
 179     namespace
 180 
 181     Returns
 182     -------
 183     """
 184     string = generate_class_string(typename, props, description, namespace)
 185     scope = {"Component": Component, "_explicitize_args": _explicitize_args}
 186     # pylint: disable=exec-used
 187     exec(string, scope)
 188     result = scope[typename]
 189     return result

 207 def create_docstring(component_name, props, description):
 208     """Create the Dash component docstring.
 209 
 210     Parameters
 211     ----------
 212     component_name: str
 213         Component name
 214     props: dict
 215         Dictionary with {propName: propMetadata} structure
 216     description: str
 217         Component description
 218 
 219     Returns
 220     -------
 221     str
 222         Dash component docstring
 223     """
 224     # Ensure props are ordered with children first
 225     props = reorder_props(props=props)
 226 
 227     return (
 228         """A{n} {name} component.\n{description}
 229 
 230 Keyword arguments:\n{args}"""
 231     ).format(
 232         n="n" if component_name[0].lower() in ["a", "e", "i", "o", "u"] else "",
 233         name=component_name,
 234         description=description,
 235         args="\n".join(
 236             create_prop_docstring(
 237                 prop_name=p,
 238                 type_object=prop["type"] if "type" in prop else prop["flowType"],
 239                 required=prop["required"],
 240                 description=prop["description"],
 241                 default=prop.get("defaultValue"),
 242                 indent_num=0,
 243                 is_flow_type="flowType" in prop and "type" not in prop,
 244             )
 245             for p, prop in list(filter_props(props).items())
 246         ),
 247     )

 250 def prohibit_events(props):
 251     """Events have been removed. Raise an error if we see dashEvents or
 252     fireEvents.
 253 
 254     Parameters
 255     ----------
 256     props: dict
 257         Dictionary with {propName: propMetadata} structure
 258 
 259     Raises
 260     -------
 261     ?
 262     """
 263     if "dashEvents" in props or "fireEvents" in props:
 264         raise NonExistentEventException(
 265             "Events are no longer supported by dash. Use properties instead, "
 266             "eg `n_clicks` instead of a `click` event."

 270 def parse_wildcards(props):
 271     """Pull out the wildcard attributes from the Component props.
 272 
 273     Parameters
 274     ----------
 275     props: dict
 276         Dictionary with {propName: propMetadata} structure
 277 
 278     Returns
 279     -------
 280     list
 281         List of Dash valid wildcard prefixes
 282     """
 283     list_of_valid_wildcard_attr_prefixes = []
 284     for wildcard_attr in ["data-*", "aria-*"]:
 285         if wildcard_attr in props:
 286             list_of_valid_wildcard_attr_prefixes.append(wildcard_attr[:-1])
 287     return list_of_valid_wildcard_attr_prefixes

 290 def reorder_props(props):
 291     """If "children" is in props, then move it to the front to respect dash
 292     convention.
 293 
 294     Parameters
 295     ----------
 296     props: dict
 297         Dictionary with {propName: propMetadata} structure
 298 
 299     Returns
 300     -------
 301     dict
 302         Dictionary with {propName: propMetadata} structure
 303     """
 304     if "children" in props:
 305         # Constructing an OrderedDict with duplicate keys, you get the order
 306         # from the first one but the value from the last.
 307         # Doing this to avoid mutating props, which can cause confusion.
 308         props = OrderedDict([("children", "")] + list(props.items()))
 309 
 310     return props

 313 def filter_props(props):
 314     """Filter props from the Component arguments to exclude:
 315 
 316         - Those without a "type" or a "flowType" field
 317         - Those with arg.type.name in {'func', 'symbol', 'instanceOf'}
 318 
 319     Parameters
 320     ----------
 321     props: dict
 322         Dictionary with {propName: propMetadata} structure
 323 
 324     Returns
 325     -------
 326     dict
 327         Filtered dictionary with {propName: propMetadata} structure
 328 
 329     Examples
 330     --------
 331     ```python
 332     prop_args = {
 333         'prop1': {
 334             'type': {'name': 'bool'},
 335             'required': False,
 336             'description': 'A description',
 337             'flowType': {},
 338             'defaultValue': {'value': 'false', 'computed': False},
 339         },
 340         'prop2': {'description': 'A prop without a type'},
 341         'prop3': {
 342             'type': {'name': 'func'},
 343             'description': 'A function prop',
 344         },
 345     }
 346     # filtered_prop_args is now
 347     # {
 348     #    'prop1': {
 349     #        'type': {'name': 'bool'},
 350     #        'required': False,
 351     #        'description': 'A description',
 352     #        'flowType': {},
 353     #        'defaultValue': {'value': 'false', 'computed': False},
 354     #    },
 355     # }
 356     filtered_prop_args = filter_props(prop_args)
 357     ```
 358     """
 359     filtered_props = copy.deepcopy(props)
 360 
 361     for arg_name, arg in list(filtered_props.items()):
 362         if "type" not in arg and "flowType" not in arg:
 363             filtered_props.pop(arg_name)
 364             continue
 365 
 366         # Filter out functions and instances --
 367         # these cannot be passed from Python
 368         if "type" in arg:  # These come from PropTypes
 369             arg_type = arg["type"]["name"]
 370             if arg_type in {"func", "symbol", "instanceOf"}:
 371                 filtered_props.pop(arg_name)
 372         elif "flowType" in arg:  # These come from Flow & handled differently
 373             arg_type_name = arg["flowType"]["name"]
 374             if arg_type_name == "signature":
 375                 # This does the same as the PropTypes filter above, but "func"
 376                 # is under "type" if "name" is "signature" vs just in "name"
 377                 if "type" not in arg["flowType"] or arg["flowType"]["type"] != "object":
 378                     filtered_props.pop(arg_name)
 379         else:
 380             raise ValueError
 381 
 382     return filtered_props

 386 def create_prop_docstring(
 387     prop_name,
 388     type_object,
 389     required,
 390     description,
 391     default,
 392     indent_num,
 393     is_flow_type=False,
 394 ):
 395     """Create the Dash component prop docstring.
 396 
 397     Parameters
 398     ----------
 399     prop_name: str
 400         Name of the Dash component prop
 401     type_object: dict
 402         react-docgen-generated prop type dictionary
 403     required: bool
 404         Component is required?
 405     description: str
 406         Dash component description
 407     default: dict
 408         Either None if a default value is not defined, or
 409         dict containing the key 'value' that defines a
 410         default value for the prop
 411     indent_num: int
 412         Number of indents to use for the context block
 413         (creates 2 spaces for every indent)
 414     is_flow_type: bool
 415         Does the prop use Flow types? Otherwise, uses PropTypes
 416 
 417     Returns
 418     -------
 419     str
 420         Dash component prop docstring
 421     """
 422     py_type_name = js_to_py_type(
 423         type_object=type_object, is_flow_type=is_flow_type, indent_num=indent_num + 1
 424     )
 425     indent_spacing = "  " * indent_num
 426 
 427     if default is None:
 428         default = ""
 429     else:
 430         default = default["value"]
 431 
 432     if default in ["true", "false"]:
 433         default = default.title()
 434 
 435     is_required = "optional"
 436     if required:
 437         is_required = "required"
 438     elif default and default not in ["null", "{}", "[]"]:
 439         is_required = "default {}".format(default.replace("\n", "\n" + indent_spacing))
 440 
 441     if "\n" in py_type_name:
 442         return (
 443             "{indent_spacing}- {name} (dict; {is_required}): "
 444             "{description}{period}"
 445             "{name} has the following type: {type}".format(
 446                 indent_spacing=indent_spacing,
 447                 name=prop_name,
 448                 type=py_type_name,
 449                 description=description.strip().strip("."),
 450                 period=". " if description else "",
 451                 is_required=is_required,
 452             )
 453         )
 454     return "{indent_spacing}- {name} ({type}{is_required}){description}".format(
 455         indent_spacing=indent_spacing,
 456         name=prop_name,
 457         type="{}; ".format(py_type_name) if py_type_name else "",
 458         description=(": {}".format(description) if description != "" else ""),
 459         is_required=is_required,
 460     )

 463 def map_js_to_py_types_prop_types(type_object):
 464     """Mapping from the PropTypes js type object to the Python type."""
 465 
 466     def shape_or_exact():
 467         return "dict containing keys {}.\n{}".format(
 468             ", ".join("'{}'".format(t) for t in list(type_object["value"].keys())),
 469             "Those keys have the following types:\n{}".format(
 470                 "\n".join(
 471                     create_prop_docstring(
 472                         prop_name=prop_name,
 473                         type_object=prop,
 474                         required=prop["required"],
 475                         description=prop.get("description", ""),
 476                         default=prop.get("defaultValue"),
 477                         indent_num=1,
 478                     )
 479                     for prop_name, prop in list(type_object["value"].items())
 480                 )
 481             ),
 482         )
 483 
 484     return dict(
 485         array=lambda: "list",
 486         bool=lambda: "boolean",
 487         number=lambda: "number",
 488         string=lambda: "string",
 489         object=lambda: "dict",
 490         any=lambda: "boolean | number | string | dict | list",
 491         element=lambda: "dash component",
 492         node=lambda: "a list of or a singular dash component, string or number",
 493         # React's PropTypes.oneOf
 494         enum=lambda: "a value equal to: {}".format(
 495             ", ".join("{}".format(str(t["value"])) for t in type_object["value"])
 496         ),
 497         # React's PropTypes.oneOfType
 498         union=lambda: "{}".format(
 499             " | ".join(
 500                 "{}".format(js_to_py_type(subType))
 501                 for subType in type_object["value"]
 502                 if js_to_py_type(subType) != ""
 503             )
 504         ),
 505         # React's PropTypes.arrayOf
 506         arrayOf=lambda: (
 507             "list"
 508             + (
 509                 " of {}".format(
 510                     js_to_py_type(type_object["value"]) + "s"
 511                     if js_to_py_type(type_object["value"]).split(" ")[0] != "dict"
 512                     else js_to_py_type(type_object["value"]).replace("dict", "dicts", 1)
 513                 )
 514                 if js_to_py_type(type_object["value"]) != ""
 515                 else ""
 516             )
 517         ),
 518         # React's PropTypes.objectOf
 519         objectOf=lambda: "dict with strings as keys and values of type {}".format(
 520             js_to_py_type(type_object["value"])
 521         ),
 522         # React's PropTypes.shape
 523         shape=shape_or_exact,
 524         # React's PropTypes.exact
 525         exact=shape_or_exact,
 526     )

 529 def map_js_to_py_types_flow_types(type_object):
 530     """Mapping from the Flow js types to the Python type."""
 531 
 532     return dict(
 533         array=lambda: "list",
 534         boolean=lambda: "boolean",
 535         number=lambda: "number",
 536         string=lambda: "string",
 537         Object=lambda: "dict",
 538         any=lambda: "bool | number | str | dict | list",
 539         Element=lambda: "dash component",
 540         Node=lambda: "a list of or a singular dash component, string or number",
 541         # React's PropTypes.oneOfType
 542         union=lambda: "{}".format(
 543             " | ".join(
 544                 "{}".format(js_to_py_type(subType))
 545                 for subType in type_object["elements"]
 546                 if js_to_py_type(subType) != ""
 547             )
 548         ),
 549         # Flow's Array type
 550         Array=lambda: "list{}".format(
 551             " of {}s".format(js_to_py_type(type_object["elements"][0]))
 552             if js_to_py_type(type_object["elements"][0]) != ""
 553             else ""
 554         ),
 555         # React's PropTypes.shape
 556         signature=lambda indent_num: "dict containing keys {}.\n{}".format(
 557             ", ".join(
 558                 "'{}'".format(d["key"]) for d in type_object["signature"]["properties"]
 559             ),
 560             "{}Those keys have the following types:\n{}".format(
 561                 "  " * indent_num,
 562                 "\n".join(
 563                     create_prop_docstring(
 564                         prop_name=prop["key"],
 565                         type_object=prop["value"],
 566                         required=prop["value"]["required"],
 567                         description=prop["value"].get("description", ""),
 568                         default=prop.get("defaultValue"),
 569                         indent_num=indent_num,
 570                         is_flow_type=True,
 571                     )
 572                     for prop in type_object["signature"]["properties"]
 573                 ),
 574             ),
 575         ),
 576     )

 579 def js_to_py_type(type_object, is_flow_type=False, indent_num=0):
 580     """Convert JS types to Python types for the component definition.
 581 
 582     Parameters
 583     ----------
 584     type_object: dict
 585         react-docgen-generated prop type dictionary
 586     is_flow_type: bool
 587         Does the prop use Flow types? Otherwise, uses PropTypes
 588     indent_num: int
 589         Number of indents to use for the docstring for the prop
 590 
 591     Returns
 592     -------
 593     str
 594         Python type string
 595     """
 596     js_type_name = type_object["name"]
 597     js_to_py_types = (
 598         map_js_to_py_types_flow_types(type_object=type_object)
 599         if is_flow_type
 600         else map_js_to_py_types_prop_types(type_object=type_object)
 601     )
 602 
 603     if (
 604         "computed" in type_object
 605         and type_object["computed"]
 606         or type_object.get("type", "") == "function"
 607     ):
 608         return ""
 609     if js_type_name in js_to_py_types:
 610         if js_type_name == "signature":  # This is a Flow object w/ signature
 611             return js_to_py_types[js_type_name](indent_num)
 612         # All other types
 613         return js_to_py_types[js_type_name]()
 614     return ""
