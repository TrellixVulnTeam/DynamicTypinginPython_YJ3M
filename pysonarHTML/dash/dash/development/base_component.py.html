  81     def __init__(self, **kwargs):
  82         # pylint: disable=super-init-not-called
  83         for k, v in list(kwargs.items()):
  84             # pylint: disable=no-member
  85             k_in_propnames = k in self._prop_names
  86             k_in_wildcards = any(
  87                 [k.startswith(w) for w in self._valid_wildcard_attributes]
  88             )
  89             # e.g. "The dash_core_components.Dropdown component (version 1.6.0)
  90             # with the ID "my-dropdown"
  91             try:
  92                 error_string_prefix = "The `{}.{}` component (version {}){}".format(
  93                     self._namespace,
  94                     self._type,
  95                     getattr(__import__(self._namespace), "__version__", "unknown"),
  96                     ' with the ID "{}"'.format(kwargs["id"]) if "id" in kwargs else "",
  97                 )
  98             except ImportError:
  99                 # Our tests create mock components with libraries that
 100                 # aren't importable
 101                 error_string_prefix = "The `{}` component{}".format(
 102                     self._type,
 103                     ' with the ID "{}"'.format(kwargs["id"]) if "id" in kwargs else "",
 104                 )
 105 
 106             if not k_in_propnames and not k_in_wildcards:
 107                 raise TypeError(
 108                     "{} received an unexpected keyword argument: `{}`".format(
 109                         error_string_prefix, k
 110                     )
 111                     + "\nAllowed arguments: {}".format(  # pylint: disable=no-member
 112                         ", ".join(sorted(self._prop_names))
 113                     )
 114                 )
 115 
 116             if k != "children" and isinstance(v, Component):
 117                 raise TypeError(
 118                     error_string_prefix
 119                     + " detected a Component for a prop other than `children`\n"
 120                     + "Did you forget to wrap multiple `children` in an array?\n"
 121                     + "Prop {} has value {}\n".format(k, repr(v))
 122                 )
 123 
 124             if k == "id":
 125                 if isinstance(v, dict):
 126                     for id_key, id_val in v.items():
 127                         if not isinstance(id_key, _strings):
 128                             raise TypeError(
 129                                 "dict id keys must be strings,\n"
 130                                 + "found {!r} in id {!r}".format(id_key, v)
 131                             )
 132                         if not isinstance(id_val, _strings + (int, float, bool)):
 133                             raise TypeError(
 134                                 "dict id values must be strings, numbers or bools,\n"
 135                                 + "found {!r} in id {!r}".format(id_val, v)
 136                             )
 137                 elif not isinstance(v, _strings):
 138                     raise TypeError(
 139                         "`id` prop must be a string or dict, not {!r}".format(v)
 140                     )
 141 
 142             setattr(self, k, v)

 144     def to_plotly_json(self):
 145         # Add normal properties
 146         props = {
 147             p: getattr(self, p)
 148             for p in self._prop_names  # pylint: disable=no-member
 149             if hasattr(self, p)
 150         }
 151         # Add the wildcard properties data-* and aria-*
 152         props.update(
 153             {
 154                 k: getattr(self, k)
 155                 for k in self.__dict__
 156                 if any(
 157                     k.startswith(w)
 158                     # pylint:disable=no-member
 159                     for w in self._valid_wildcard_attributes
 160                 )
 161             }
 162         )
 163         as_json = {
 164             "props": props,
 165             "type": self._type,  # pylint: disable=no-member
 166             "namespace": self._namespace,  # pylint: disable=no-member
 167         }
 168 
 169         return as_json

 173     def _get_set_or_delete(self, id, operation, new_item=None):
 174         _check_if_has_indexable_children(self)
 175 
 176         # pylint: disable=access-member-before-definition,
 177         # pylint: disable=attribute-defined-outside-init
 178         if isinstance(self.children, Component):
 179             if getattr(self.children, "id", None) is not None:
 180                 # Woohoo! It's the item that we're looking for
 181                 if self.children.id == id:
 182                     if operation == "get":
 183                         return self.children
 184                     elif operation == "set":
 185                         self.children = new_item
 186                         return
 187                     elif operation == "delete":
 188                         self.children = None
 189                         return
 190 
 191             # Recursively dig into its subtree
 192             try:
 193                 if operation == "get":
 194                     return self.children.__getitem__(id)
 195                 elif operation == "set":
 196                     self.children.__setitem__(id, new_item)
 197                     return
 198                 elif operation == "delete":
 199                     self.children.__delitem__(id)
 200                     return
 201             except KeyError:
 202                 pass
 203 
 204         # if children is like a list
 205         if isinstance(self.children, (tuple, MutableSequence)):
 206             for i, item in enumerate(self.children):
 207                 # If the item itself is the one we're looking for
 208                 if getattr(item, "id", None) == id:
 209                     if operation == "get":
 210                         return item
 211                     elif operation == "set":
 212                         self.children[i] = new_item
 213                         return
 214                     elif operation == "delete":
 215                         del self.children[i]
 216                         return
 217 
 218                 # Otherwise, recursively dig into that item's subtree
 219                 # Make sure it's not like a string
 220                 elif isinstance(item, Component):
 221                     try:
 222                         if operation == "get":
 223                             return item.__getitem__(id)
 224                         elif operation == "set":
 225                             item.__setitem__(id, new_item)
 226                             return
 227                         elif operation == "delete":
 228                             item.__delitem__(id)
 229                             return
 230                     except KeyError:
 231                         pass
 232 
 233         # The end of our branch
 234         # If we were in a list, then this exception will get caught
 235         raise KeyError(id)

 244     def __getitem__(self, id):  # pylint: disable=redefined-builtin
 245         """Recursively find the element with the given ID through the tree of
 246         children."""
 247 
 248         # A component's children can be undefined, a string, another component,
 249         # or a list of components.
 250         return self._get_set_or_delete(id, "get")
 251 
 252     def __setitem__(self, id, item):  # pylint: disable=redefined-builtin
 253         """Set an element by its ID."""
 254         return self._get_set_or_delete(id, "set", item)
 255 
 256     def __delitem__(self, id):  # pylint: disable=redefined-builtin
 257         """Delete items by ID in the tree of children."""
 258         return self._get_set_or_delete(id, "delete")

 260     def _traverse(self):
 261         """Yield each item in the tree."""
 262         for t in self._traverse_with_paths():
 263             yield t[1]
 264 
 265     @staticmethod
 266     def _id_str(component):
 267         id_ = stringify_id(getattr(component, "id", ""))
 268         return id_ and " (id={:s})".format(id_)

 270     def _traverse_with_paths(self):
 271         """Yield each item with its path in the tree."""
 272         children = getattr(self, "children", None)
 273         children_type = type(children).__name__
 274         children_string = children_type + self._id_str(children)
 275 
 276         # children is just a component
 277         if isinstance(children, Component):
 278             yield "[*] " + children_string, children
 279             # pylint: disable=protected-access
 280             for p, t in children._traverse_with_paths():
 281                 yield "\n".join(["[*] " + children_string, p]), t
 282 
 283         # children is a list of components
 284         elif isinstance(children, (tuple, MutableSequence)):
 285             for idx, i in enumerate(children):
 286                 list_path = "[{:d}] {:s}{}".format(
 287                     idx, type(i).__name__, self._id_str(i)
 288                 )
 289                 yield list_path, i
 290 
 291                 if isinstance(i, Component):
 292                     # pylint: disable=protected-access
 293                     for p, t in i._traverse_with_paths():
 294                         yield "\n".join([list_path, p]), t
 295 
 296     def _traverse_ids(self):
 297         """Yield components with IDs in the tree of children."""
 298         for t in self._traverse():
 299             if isinstance(t, Component) and getattr(t, "id", None) is not None:
 300                 yield t
 301 
 302     def __iter__(self):
 303         """Yield IDs in the tree of children."""
 304         for t in self._traverse_ids():
 305             yield t.id

 307     def __len__(self):
 308         """Return the number of items in the tree."""
 309         # TODO - Should we return the number of items that have IDs
 310         # or just the number of items?
 311         # The number of items is more intuitive but returning the number
 312         # of IDs matches __iter__ better.
 313         length = 0
 314         if getattr(self, "children", None) is None:
 315             length = 0
 316         elif isinstance(self.children, Component):
 317             length = 1
 318             length += len(self.children)
 319         elif isinstance(self.children, (tuple, MutableSequence)):
 320             for c in self.children:
 321                 length += 1
 322                 if isinstance(c, Component):
 323                     length += len(c)
 324         else:
 325             # string or number
 326             length = 1
 327         return length

 329     def __repr__(self):
 330         # pylint: disable=no-member
 331         props_with_values = [
 332             c for c in self._prop_names if getattr(self, c, None) is not None
 333         ] + [
 334             c
 335             for c in self.__dict__
 336             if any(c.startswith(wc_attr) for wc_attr in self._valid_wildcard_attributes)
 337         ]
 338         if any(p != "children" for p in props_with_values):
 339             props_string = ", ".join(
 340                 "{prop}={value}".format(prop=p, value=repr(getattr(self, p)))
 341                 for p in props_with_values
 342             )
 343         else:
 344             props_string = repr(getattr(self, "children", None))
 345         return "{type}({props_string})".format(
 346             type=self._type, props_string=props_string
 347         )

 350 def _explicitize_args(func):
 351     # Python 2
 352     if hasattr(func, "func_code"):
 353         varnames = func.func_code.co_varnames
 354     # Python 3
 355     else:
 356         varnames = func.__code__.co_varnames
 357 
 358     def wrapper(*args, **kwargs):
 359         if "_explicit_args" in kwargs.keys():
 360             raise Exception("Variable _explicit_args should not be set.")
 361         kwargs["_explicit_args"] = list(
 362             set(list(varnames[: len(args)]) + [k for k, _ in kwargs.items()])
 363         )
 364         if "self" in kwargs["_explicit_args"]:
 365             kwargs["_explicit_args"].remove("self")
 366         return func(*args, **kwargs)
 367 
 368     # If Python 3, we can set the function signature to be correct
 369     if hasattr(inspect, "signature"):
 370         # pylint: disable=no-member
 371         new_sig = inspect.signature(wrapper).replace(
 372             parameters=inspect.signature(func).parameters.values()
 373         )
 374         wrapper.__signature__ = new_sig
 375     return wrapper
