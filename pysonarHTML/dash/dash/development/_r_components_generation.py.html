 179 # pylint: disable=R0914
 180 def generate_class_string(name, props, project_shortname, prefix):
 181     # Here we convert from snake case to camel case
 182     package_name = snake_case_to_camel_case(project_shortname)
 183 
 184     # Ensure props are ordered with children first
 185     props = reorder_props(props=props)
 186 
 187     prop_keys = list(props.keys())
 188 
 189     wildcards = ""
 190     wildcard_declaration = ""
 191     wildcard_names = ""
 192     default_paramtext = ""
 193     default_argtext = ""
 194     accepted_wildcards = ""
 195 
 196     if any(key.endswith("-*") for key in prop_keys):
 197         accepted_wildcards = get_wildcards_r(prop_keys)
 198         wildcards = ", ..."
 199         wildcard_declaration = wildcard_template.format(
 200             accepted_wildcards.replace("-*", "")
 201         )
 202         wildcard_names = ", wildcard_names"
 203 
 204     # Produce a string with all property names other than WCs
 205     prop_names = ", ".join(
 206         "'{}'".format(p) for p in prop_keys if "*" not in p and p not in ["setProps"]
 207     )
 208 
 209     # Filter props to remove those we don't want to expose
 210     for item in prop_keys[:]:
 211         if item.endswith("-*") or item == "setProps":
 212             prop_keys.remove(item)
 213         elif item in r_keywords:
 214             prop_keys.remove(item)
 215             warnings.warn(
 216                 (
 217                     'WARNING: prop "{}" in component "{}" is an R keyword'
 218                     " - REMOVED FROM THE R COMPONENT"
 219                 ).format(item, name)
 220             )
 221 
 222     default_argtext += ", ".join("{}=NULL".format(p) for p in prop_keys)
 223 
 224     # pylint: disable=C0301
 225     default_paramtext += ", ".join(
 226         "{0}={0}".format(p) if p != "children" else "{}=children".format(p)
 227         for p in prop_keys
 228     )
 229 
 230     return r_component_string.format(
 231         funcname=format_fn_name(prefix, name),
 232         name=name,
 233         default_argtext=default_argtext,
 234         wildcards=wildcards,
 235         wildcard_declaration=wildcard_declaration,
 236         default_paramtext=default_paramtext,
 237         project_shortname=project_shortname,
 238         prop_names=prop_names,
 239         wildcard_names=wildcard_names,
 240         package_name=package_name,
 241     )

 245 def generate_js_metadata(pkg_data, project_shortname):
 246     """Dynamically generate R function to supply JavaScript and CSS dependency
 247     information required by the dash package for R.
 248 
 249     Parameters
 250     ----------
 251     project_shortname = component library name, in snake case
 252 
 253     Returns
 254     -------
 255     function_string = complete R function code to provide component features
 256     """
 257     # make sure the module we're building is available to Python,
 258     # even if it hasn't been installed yet
 259     sys.path.insert(0, os.getcwd())
 260     mod = importlib.import_module(project_shortname)
 261 
 262     alldist = getattr(mod, "_js_dist", []) + getattr(mod, "_css_dist", [])
 263 
 264     project_ver = pkg_data.get("version")
 265 
 266     rpkgname = snake_case_to_camel_case(project_shortname)
 267 
 268     # since _js_dist may suggest more than one dependency, need
 269     # a way to iterate over all dependencies for a given set.
 270     # here we define an opening, element, and closing string --
 271     # if the total number of dependencies > 1, we can concatenate
 272     # them and write a list object in R with multiple elements
 273     function_frame_open = frame_open_template.format(rpkgname=rpkgname)
 274 
 275     function_frame = []
 276     function_frame_body = []
 277 
 278     # pylint: disable=consider-using-enumerate
 279     if len(alldist) > 1:
 280         for dep in range(len(alldist)):
 281             curr_dep = alldist[dep]
 282             rpp = curr_dep["relative_package_path"]
 283 
 284             async_or_dynamic = get_async_type(curr_dep)
 285 
 286             if "dash_" in rpp:
 287                 dep_name = rpp.split(".")[0]
 288             else:
 289                 dep_name = "{}".format(project_shortname)
 290 
 291             if "css" in rpp:
 292                 css_name = "'{}'".format(rpp)
 293                 script_name = "NULL"
 294             else:
 295                 script_name = "'{}'".format(rpp)
 296                 css_name = "NULL"
 297 
 298             function_frame += [
 299                 frame_element_template.format(
 300                     dep_name=dep_name,
 301                     project_ver=project_ver,
 302                     rpkgname=rpkgname,
 303                     project_shortname=project_shortname,
 304                     script_name=script_name,
 305                     css_name=css_name,
 306                     async_or_dynamic=async_or_dynamic,
 307                 )
 308             ]
 309             function_frame_body = ",\n".join(function_frame)
 310     elif len(alldist) == 1:
 311         dep = alldist[0]
 312         rpp = dep["relative_package_path"]
 313 
 314         async_or_dynamic = get_async_type(dep)
 315 
 316         if "css" in rpp:
 317             css_name = "'{}'".format(rpp)
 318             script_name = "NULL"
 319         else:
 320             script_name = "'{}'".format(rpp)
 321             css_name = "NULL"
 322 
 323         function_frame_body = frame_body_template.format(
 324             project_shortname=project_shortname,
 325             project_ver=project_ver,
 326             rpkgname=rpkgname,
 327             script_name=script_name,
 328             css_name=css_name,
 329             async_or_dynamic=async_or_dynamic,
 330         )
 331 
 332     function_string = "".join(
 333         [function_frame_open, function_frame_body, frame_close_template]
 334     )
 335 
 336     return function_string

 339 # determine whether dependency uses async or dynamic flag
 340 # then return the properly formatted string if so, i.e.
 341 # " async = TRUE,". a dependency can have async or
 342 # dynamic elements, neither of these, but never both.
 343 def get_async_type(dep):
 344     async_or_dynamic = ""
 345     for key in dep.keys():
 346         if key in ["async", "dynamic"]:
 347             keyval = dep[key]
 348             if not isinstance(keyval, bool):
 349                 keyval = "'{}'".format(keyval.lower())
 350             else:
 351                 keyval = str(keyval).upper()
 352             async_or_dynamic = ", {} = {}".format(key, keyval)
 353     return async_or_dynamic

 356 # This method wraps code within arbitrary LaTeX-like tags, which are used
 357 # by R's internal help parser for constructing man pages
 358 def wrap(tag, code):
 359     if tag == "":
 360         return code
 361     return "\\{}{{\n{}}}".format(tag, code)

 364 def write_help_file(name, props, description, prefix, rpkg_data):
 365     """Write R documentation file (.Rd) given component name and properties.
 366 
 367     Parameters
 368     ----------
 369     name = the name of the Dash component for which a help file is generated
 370     props = the properties of the component
 371     description = the component's description, inserted into help file header
 372     prefix = the DashR library prefix (optional, can be a blank string)
 373     rpkg_data = package metadata (optional)
 374 
 375     Returns
 376     -------
 377     writes an R help file to the man directory for the generated R package
 378     """
 379     funcname = format_fn_name(prefix, name)
 380     file_name = funcname + ".Rd"
 381 
 382     wildcards = ""
 383     default_argtext = ""
 384     item_text = ""
 385     accepted_wildcards = ""
 386 
 387     # the return value of all Dash components should be the same,
 388     # in an abstract sense -- they produce a list
 389     value_text = "named list of JSON elements corresponding to React.js properties and their values"  # noqa:E501
 390 
 391     prop_keys = list(props.keys())
 392 
 393     if any(key.endswith("-*") for key in prop_keys):
 394         accepted_wildcards = get_wildcards_r(prop_keys)
 395         wildcards = ", ..."
 396 
 397     # Filter props to remove those we don't want to expose
 398     for item in prop_keys[:]:
 399         if item.endswith("-*") or item in r_keywords or item == "setProps":
 400             prop_keys.remove(item)
 401 
 402     default_argtext += ", ".join("{}=NULL".format(p) for p in prop_keys)
 403 
 404     item_text += "\n\n".join(
 405         "\\item{{{}}}{{{}{}}}".format(
 406             p, print_r_type(props[p]["type"]), props[p]["description"]
 407         )
 408         for p in prop_keys
 409     )
 410 
 411     # auto-replace any unescaped backslashes for compatibility with R docs
 412     description = re.sub(r"(?<!\\)%", "\\%", description)
 413     item_text = re.sub(r"(?<!\\)%", "\\%", item_text)
 414 
 415     # scrub examples which begin with **Example Usage**, as these should be
 416     # provided as R code within dash-info.yaml
 417     if "**Example Usage**" in description:
 418         description = description.split("**Example Usage**")[0].rstrip()
 419 
 420     if wildcards == ", ...":
 421         default_argtext += wildcards
 422         item_text += wildcard_help_template.format(accepted_wildcards)
 423 
 424     # in R, the online help viewer does not properly wrap lines for
 425     # the usage string -- we will hard wrap at 60 characters using
 426     # textwrap.fill, starting from the beginning of the usage string
 427 
 428     file_path = os.path.join("man", file_name)
 429     with open(file_path, "w") as f:
 430         f.write(
 431             help_string.format(
 432                 funcname=funcname,
 433                 name=name,
 434                 default_argtext=textwrap.fill(
 435                     default_argtext, width=60, break_long_words=False
 436                 ),
 437                 item_text=item_text,
 438                 value_text=value_text,
 439                 description=description.replace("\n", " "),
 440             )
 441         )
 442     if rpkg_data is not None and "r_examples" in rpkg_data:
 443         ex = rpkg_data.get("r_examples")
 444         the_ex = ([e for e in ex if e.get("name") == funcname] or [None])[0]
 445         result = ""
 446         if the_ex and "code" in the_ex.keys():
 447             result += wrap(
 448                 "examples",
 449                 wrap("dontrun" if the_ex.get("dontrun") else "", the_ex["code"]),
 450             )
 451             with open(file_path, "a+") as fa:
 452                 fa.write(result + "\n")

 481 def write_js_metadata(pkg_data, project_shortname, has_wildcards):
 482     """Write an internal (not exported) R function to return all JS
 483     dependencies as required by dash.
 484 
 485     Parameters
 486     ----------
 487     project_shortname = hyphenated string, e.g. dash-html-components
 488 
 489     Returns
 490     -------
 491     """
 492     function_string = generate_js_metadata(
 493         pkg_data=pkg_data, project_shortname=project_shortname
 494     )
 495     file_name = "internal.R"
 496 
 497     # the R source directory for the package won't exist on first call
 498     # create the R directory if it is missing
 499     if not os.path.exists("R"):
 500         os.makedirs("R")
 501 
 502     file_path = os.path.join("R", file_name)
 503     with open(file_path, "w") as f:
 504         f.write(function_string)
 505         if has_wildcards:
 506             f.write(wildcard_helper)
 507 
 508     # now copy over all JS dependencies from the (Python) components dir
 509     # the inst/lib directory for the package won't exist on first call
 510     # create this directory if it is missing
 511     if os.path.exists("inst/deps"):
 512         shutil.rmtree("inst/deps")
 513 
 514     os.makedirs("inst/deps")
 515 
 516     for rel_dirname, _, filenames in os.walk(project_shortname):
 517         for filename in filenames:
 518             extension = os.path.splitext(filename)[1]
 519 
 520             if extension in [".py", ".pyc", ".json"]:
 521                 continue
 522 
 523             target_dirname = os.path.join(
 524                 os.path.join(
 525                     "inst/deps/", os.path.relpath(rel_dirname, project_shortname)
 526                 )
 527             )
 528 
 529             if not os.path.exists(target_dirname):
 530                 os.makedirs(target_dirname)
 531 
 532             shutil.copy(os.path.join(rel_dirname, filename), target_dirname)

 535 # pylint: disable=R0914, R0913, R0912, R0915
 536 def generate_rpkg(
 537     pkg_data,
 538     rpkg_data,
 539     project_shortname,
 540     export_string,
 541     package_depends,
 542     package_imports,
 543     package_suggests,
 544     has_wildcards,
 545 ):
 546     """Generate documents for R package creation.
 547 
 548     Parameters
 549     ----------
 550     pkg_data
 551     rpkg_data
 552     project_shortname
 553     export_string
 554     package_depends
 555     package_imports
 556     package_suggests
 557     has_wildcards
 558 
 559     Returns
 560     -------
 561     """
 562     # Leverage package.json to import specifics which are also applicable
 563     # to R package that we're generating here, use .get in case the key
 564     # does not exist in package.json
 565 
 566     package_name = snake_case_to_camel_case(project_shortname)
 567     package_copyright = ""
 568     package_rauthors = ""
 569     lib_name = pkg_data.get("name")
 570 
 571     if rpkg_data is not None:
 572         if rpkg_data.get("pkg_help_title"):
 573             package_title = rpkg_data.get(
 574                 "pkg_help_title", pkg_data.get("description", "")
 575             )
 576         if rpkg_data.get("pkg_help_description"):
 577             package_description = rpkg_data.get(
 578                 "pkg_help_description", pkg_data.get("description", "")
 579             )
 580         if rpkg_data.get("pkg_copyright"):
 581             package_copyright = "\nCopyright: {}".format(
 582                 rpkg_data.get("pkg_copyright", "")
 583             )
 584     else:
 585         # fall back to using description in package.json, if present
 586         package_title = pkg_data.get("description", "")
 587         package_description = pkg_data.get("description", "")
 588 
 589     package_version = pkg_data.get("version", "0.0.1")
 590 
 591     # remove leading and trailing commas, add space after comma if missing
 592     if package_depends:
 593         package_depends = ", " + package_depends.strip(",").lstrip()
 594         package_depends = re.sub(r"(,(?![ ]))", ", ", package_depends)
 595 
 596     if package_imports:
 597         package_imports = package_imports.strip(",").lstrip()
 598         package_imports = re.sub(r"(,(?![ ]))", ", ", package_imports)
 599 
 600     if package_suggests:
 601         package_suggests = package_suggests.strip(",").lstrip()
 602         package_suggests = re.sub(r"(,(?![ ]))", ", ", package_suggests)
 603 
 604     if "bugs" in pkg_data:
 605         package_issues = pkg_data["bugs"].get("url", "")
 606     else:
 607         package_issues = ""
 608         print(
 609             "Warning: a URL for bug reports was "
 610             "not provided. Empty string inserted.",
 611             file=sys.stderr,
 612         )
 613 
 614     if "homepage" in pkg_data:
 615         package_url = pkg_data.get("homepage", "")
 616     else:
 617         package_url = ""
 618         print(
 619             "Warning: a homepage URL was not provided. Empty string inserted.",
 620             file=sys.stderr,
 621         )
 622 
 623     package_author = pkg_data.get("author")
 624 
 625     package_author_name = package_author.split(" <")[0]
 626     package_author_email = package_author.split(" <")[1][:-1]
 627 
 628     package_author_fn = package_author_name.split(" ")[0]
 629     package_author_ln = package_author_name.rsplit(" ", 2)[-1]
 630 
 631     maintainer = pkg_data.get("maintainer", pkg_data.get("author"))
 632 
 633     if "<" not in package_author:
 634         print(
 635             "Error, aborting R package generation: "
 636             "R packages require a properly formatted author field "
 637             "or installation will fail. Please include an email "
 638             "address enclosed within < > brackets in package.json. ",
 639             file=sys.stderr,
 640         )
 641         sys.exit(1)
 642 
 643     if rpkg_data is not None:
 644         if rpkg_data.get("pkg_authors"):
 645             package_rauthors = "\nAuthors@R: {}".format(
 646                 rpkg_data.get("pkg_authors", "")
 647             )
 648         else:
 649             package_rauthors = '\nAuthors@R: person("{}", "{}", role = c("aut", "cre"), email = "{}")'.format(
 650                 package_author_fn, package_author_ln, package_author_email
 651             )
 652 
 653     if not (os.path.isfile("LICENSE") or os.path.isfile("LICENSE.txt")):
 654         package_license = pkg_data.get("license", "")
 655     else:
 656         package_license = pkg_data.get("license", "") + " + file LICENSE"
 657         # R requires that the LICENSE.txt file be named LICENSE
 658         if not os.path.isfile("LICENSE"):
 659             os.symlink("LICENSE.txt", "LICENSE")
 660 
 661     import_string = "# AUTO GENERATED FILE - DO NOT EDIT\n\n"
 662     packages_string = ""
 663 
 664     rpackage_list = package_depends.split(", ") + package_imports.split(", ")
 665     rpackage_list = filter(bool, rpackage_list)
 666 
 667     if rpackage_list:
 668         for rpackage in rpackage_list:
 669             packages_string += "\nimport({})\n".format(rpackage)
 670 
 671     if os.path.exists("vignettes"):
 672         vignette_builder = "\nVignetteBuilder: knitr"
 673         if "knitr" not in package_suggests and "rmarkdown" not in package_suggests:
 674             package_suggests += ", knitr, rmarkdown"
 675             package_suggests = package_suggests.lstrip(", ")
 676     else:

 781 def make_namespace_exports(components, prefix):
 782     export_string = ""
 783     for component in components:
 784         if (
 785             not component.endswith("-*")
 786             and str(component) not in r_keywords
 787             and str(component) not in ["setProps", "children"]
 788         ):
 789             export_string += "export({}{})\n".format(prefix, component)
 790 
 791     # the following lines enable rudimentary support for bundling in
 792     # R functions that are not automatically generated by the transpiler
 793     # such that functions contained in the R subdirectory are exported,
 794     # so long as they are not in utils.R.
 795     rfilelist = []
 796     omitlist = ["utils.R", "internal.R"] + [
 797         "{}{}.R".format(prefix, component) for component in components
 798     ]
 799     fnlist = []
 800 
 801     for script in os.listdir("R"):
 802         if script.endswith(".R") and script not in omitlist:
 803             rfilelist += [os.path.join("R", script)]
 804 
 805     for rfile in rfilelist:
 806         with open(rfile, "r") as script:
 807             s = script.read()
 808 
 809             # remove comments
 810             s = re.sub("#.*$", "", s, flags=re.M)
 811 
 812             # put the whole file on one line
 813             s = s.replace("\n", " ").replace("\r", " ")
 814 
 815             # empty out strings, in case of unmatched block terminators
 816             s = re.sub(r"'([^'\\]|\\'|\\[^'])*'", "''", s)
 817             s = re.sub(r'"([^"\\]|\\"|\\[^"])*"', '""', s)
 818 
 819             # empty out block terminators () and {}
 820             # so we don't catch nested functions, or functions as arguments
 821             # repeat until it stops changing, in case of multiply nested blocks
 822             prev_len = len(s) + 1
 823             while len(s) < prev_len:
 824                 prev_len = len(s)
 825                 s = re.sub(r"\(([^()]|\(\))*\)", "()", s)
 826                 s = re.sub(r"\{([^{}]|\{\})*\}", "{}", s)
 827 
 828             # now, in whatever is left, look for functions
 829             matches = re.findall(
 830                 # in R, either = or <- may be used to create and assign objects
 831                 r"([^A-Za-z0-9._]|^)([A-Za-z0-9._]+)\s*(=|<-)\s*function",
 832                 s,
 833             )
 834             for match in matches:
 835                 fn = match[1]
 836                 # Allow users to mark functions as private by prefixing with .
 837                 if fn[0] != "." and fn not in fnlist:
 838                     fnlist.append(fn)
 839 
 840     export_string += "\n".join("export({})".format(function) for function in fnlist)
 841     return export_string

 844 def get_r_prop_types(type_object):
 845     """Mapping from the PropTypes js type object to the R type."""
 846 
 847     def shape_or_exact():
 848         return "lists containing elements {}.\n{}".format(
 849             ", ".join("'{}'".format(t) for t in list(type_object["value"].keys())),
 850             "Those elements have the following types:\n{}".format(
 851                 "\n".join(
 852                     create_prop_docstring_r(
 853                         prop_name=prop_name,
 854                         type_object=prop,
 855                         required=prop["required"],
 856                         description=prop.get("description", ""),
 857                         indent_num=1,
 858                     )
 859                     for prop_name, prop in list(type_object["value"].items())
 860                 )
 861             ),
 862         )
 863 
 864     return dict(
 865         array=lambda: "unnamed list",
 866         bool=lambda: "logical",
 867         number=lambda: "numeric",
 868         string=lambda: "character",
 869         object=lambda: "named list",
 870         any=lambda: "logical | numeric | character | named list | unnamed list",
 871         element=lambda: "dash component",
 872         node=lambda: "a list of or a singular dash component, string or number",
 873         # React's PropTypes.oneOf
 874         enum=lambda: "a value equal to: {}".format(
 875             ", ".join("{}".format(str(t["value"])) for t in type_object["value"])
 876         ),
 877         # React's PropTypes.oneOfType
 878         union=lambda: "{}".format(
 879             " | ".join(
 880                 "{}".format(get_r_type(subType))
 881                 for subType in type_object["value"]
 882                 if get_r_type(subType) != ""
 883             )
 884         ),
 885         # React's PropTypes.arrayOf
 886         arrayOf=lambda: (
 887             "list"
 888             + (
 889                 " of {}s".format(get_r_type(type_object["value"]))
 890                 if get_r_type(type_object["value"]) != ""
 891                 else ""
 892             )
 893         ),
 894         # React's PropTypes.objectOf
 895         objectOf=lambda: "list with named elements and values of type {}".format(
 896             get_r_type(type_object["value"])
 897         ),
 898         # React's PropTypes.shape
 899         shape=shape_or_exact,
 900         # React's PropTypes.exact
 901         exact=shape_or_exact,
 902     )

 905 def get_r_type(type_object, is_flow_type=False, indent_num=0):
 906     """
 907     Convert JS types to R types for the component definition
 908     Parameters
 909     ----------
 910     type_object: dict
 911         react-docgen-generated prop type dictionary
 912     is_flow_type: bool
 913     indent_num: int
 914         Number of indents to use for the docstring for the prop
 915     Returns
 916     -------
 917     str
 918         Python type string
 919     """
 920     js_type_name = type_object["name"]
 921     js_to_r_types = get_r_prop_types(type_object=type_object)
 922     if (
 923         "computed" in type_object
 924         and type_object["computed"]
 925         or type_object.get("type", "") == "function"
 926     ):
 927         return ""
 928     elif js_type_name in js_to_r_types:
 929         prop_type = js_to_r_types[js_type_name]()
 930         return prop_type
 931     return ""
 932

 936     if typestring:
 937         typestring += ". "
 938     return typestring

 941 # pylint: disable=too-many-arguments
 942 def create_prop_docstring_r(
 943     prop_name, type_object, required, description, indent_num, is_flow_type=False
 944 ):
 945     """
 946     Create the Dash component prop docstring
 947     Parameters
 948     ----------
 949     prop_name: str
 950         Name of the Dash component prop
 951     type_object: dict
 952         react-docgen-generated prop type dictionary
 953     required: bool
 954         Component is required?
 955     description: str
 956         Dash component description
 957     indent_num: int
 958         Number of indents to use for the context block
 959         (creates 2 spaces for every indent)
 960     is_flow_type: bool
 961         Does the prop use Flow types? Otherwise, uses PropTypes
 962     Returns
 963     -------
 964     str
 965         Dash component prop docstring
 966     """
 967     r_type_name = get_r_type(
 968         type_object=type_object, is_flow_type=is_flow_type, indent_num=indent_num + 1
 969     )
 970 
 971     indent_spacing = "  " * indent_num
 972     if "\n" in r_type_name:
 973         return (
 974             "{indent_spacing}- {name} ({is_required}): {description}. "
 975             "{name} has the following type: {type}".format(
 976                 indent_spacing=indent_spacing,
 977                 name=prop_name,
 978                 type=r_type_name,
 979                 description=description,
 980                 is_required="required" if required else "optional",
 981             )
 982         )
 983     return "{indent_spacing}- {name} ({type}{is_required}){description}".format(
 984         indent_spacing=indent_spacing,
 985         name=prop_name,
 986         type="{}; ".format(r_type_name) if r_type_name else "",
 987         description=(": {}".format(description) if description != "" else ""),
 988         is_required="required" if required else "optional",
 989     )
