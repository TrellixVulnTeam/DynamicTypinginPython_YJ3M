  40 def generate_components(
  41     components_source,
  42     project_shortname,
  43     package_info_filename="package.json",
  44     ignore="^_",
  45     rprefix=None,
  46     rdepends="",
  47     rimports="",
  48     rsuggests="",
  49 ):
  50 
  51     project_shortname = project_shortname.replace("-", "_").rstrip("/\\")
  52 
  53     is_windows = sys.platform == "win32"
  54 
  55     extract_path = pkg_resources.resource_filename("dash", "extract-meta.js")
  56 
  57     reserved_patterns = "|".join("^{}$".format(p) for p in reserved_words)
  58 
  59     os.environ["NODE_PATH"] = "node_modules"
  60 
  61     cmd = shlex.split(
  62         'node {} "{}" "{}" {}'.format(
  63             extract_path, ignore, reserved_patterns, components_source
  64         ),
  65         posix=not is_windows,
  66     )
  67 
  68     shutil.copyfile(
  69         "package.json", os.path.join(project_shortname, package_info_filename)
  70     )
  71 
  72     proc = subprocess.Popen(
  73         cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE, shell=is_windows
  74     )
  75     out, err = proc.communicate()
  76     status = proc.poll()
  77 
  78     if err:
  79         print(err.decode(), file=sys.stderr)
  80 
  81     if not out:
  82         print(
  83             "Error generating metadata in {} (status={})".format(
  84                 project_shortname, status
  85             ),
  86             file=sys.stderr,
  87         )
  88         sys.exit(1)
  89 
  90     metadata = safe_json_loads(out.decode("utf-8"))
  91 
  92     generator_methods = [generate_class_file]
  93 
  94     if rprefix is not None:
  95         if not os.path.exists("man"):
  96             os.makedirs("man")
  97         if not os.path.exists("R"):
  98             os.makedirs("R")
  99         if os.path.isfile("dash-info.yaml"):
 100             with open("dash-info.yaml") as yamldata:
 101                 rpkg_data = yaml.safe_load(yamldata)
 102         else:
 103             rpkg_data = None
 104         with open("package.json", "r") as f:
 105             pkg_data = safe_json_loads(f.read())
 106         generator_methods.append(
 107             functools.partial(write_class_file, prefix=rprefix, rpkg_data=rpkg_data)
 108         )
 109 
 110     components = generate_classes_files(project_shortname, metadata, *generator_methods)
 111 
 112     with open(os.path.join(project_shortname, "metadata.json"), "w") as f:
 113         json.dump(metadata, f, indent=2)
 114 
 115     generate_imports(project_shortname, components)
 116 
 117     if rprefix is not None:
 118         generate_exports(
 119             project_shortname,
 120             components,
 121             metadata,
 122             pkg_data,
 123             rpkg_data,
 124             rprefix,
 125             rdepends,
 126             rimports,
 127             rsuggests,
 128         )

 131 def safe_json_loads(s):
 132     jsondata_unicode = json.loads(s, object_pairs_hook=OrderedDict)
 133     if sys.version_info[0] >= 3:
 134         return jsondata_unicode
 135     return byteify(jsondata_unicode)

 138 def cli():
 139     parser = argparse.ArgumentParser(
 140         prog="dash-generate-components",
 141         formatter_class=_CombinedFormatter,
 142         description="Generate dash components by extracting the metadata "
 143         "using react-docgen. Then map the metadata to Python classes.",
 144     )
 145     parser.add_argument("components_source", help="React components source directory.")
 146     parser.add_argument(
 147         "project_shortname", help="Name of the project to export the classes files."
 148     )
 149     parser.add_argument(
 150         "-p",
 151         "--package-info-filename",
 152         default="package.json",
 153         help="The filename of the copied `package.json` to `project_shortname`",
 154     )
 155     parser.add_argument(
 156         "-i",
 157         "--ignore",
 158         default="^_",
 159         help="Files/directories matching the pattern will be ignored",
 160     )
 161     parser.add_argument(
 162         "--r-prefix",
 163         help="Specify a prefix for Dash for R component names, write "
 164         "components to R dir, create R package.",
 165     )
 166     parser.add_argument(
 167         "--r-depends",
 168         default="",
 169         help="Specify a comma-separated list of R packages to be "
 170         "inserted into the Depends field of the DESCRIPTION file.",
 171     )
 172     parser.add_argument(
 173         "--r-imports",
 174         default="",
 175         help="Specify a comma-separated list of R packages to be "
 176         "inserted into the Imports field of the DESCRIPTION file.",
 177     )
 178     parser.add_argument(
 179         "--r-suggests",
 180         default="",
 181         help="Specify a comma-separated list of R packages to be "
 182         "inserted into the Suggests field of the DESCRIPTION file.",
 183     )
 184 
 185     args = parser.parse_args()
 186     generate_components(
 187         args.components_source,
 188         args.project_shortname,
 189         package_info_filename=args.package_info_filename,
 190         ignore=args.ignore,
 191         rprefix=args.r_prefix,
 192         rdepends=args.r_depends,
 193         rimports=args.r_imports,
 194         rsuggests=args.r_suggests,
 195     )

 198 # pylint: disable=undefined-variable
 199 def byteify(input_object):
 200     if isinstance(input_object, dict):
 201         return OrderedDict(
 202             [(byteify(key), byteify(value)) for key, value in input_object.iteritems()]
 203         )
 204     elif isinstance(input_object, list):
 205         return [byteify(element) for element in input_object]
 206     elif isinstance(input_object, unicode):  # noqa:F821
 207         return input_object.encode("utf-8")
 208     return input_object
