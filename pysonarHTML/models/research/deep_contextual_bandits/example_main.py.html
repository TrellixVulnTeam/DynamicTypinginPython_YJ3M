  91 def sample_data(data_type, num_contexts=None):
  92   """Sample data from given 'data_type'.
  93 
  94   Args:
  95     data_type: Dataset from which to sample.
  96     num_contexts: Number of contexts to sample.
  97 
  98   Returns:
  99     dataset: Sampled matrix with rows: (context, reward_1, ..., reward_num_act).
 100     opt_rewards: Vector of expected optimal reward for each context.
 101     opt_actions: Vector of optimal action for each context.
 102     num_actions: Number of available actions.
 103     context_dim: Dimension of each context.
 104   """
 105 
 106   if data_type == 'linear':
 107     # Create linear dataset
 108     num_actions = 8
 109     context_dim = 10
 110     noise_stds = [0.01 * (i + 1) for i in range(num_actions)]
 111     dataset, _, opt_linear = sample_linear_data(num_contexts, context_dim,
 112                                                 num_actions, sigma=noise_stds)
 113     opt_rewards, opt_actions = opt_linear
 114   elif data_type == 'sparse_linear':
 115     # Create sparse linear dataset
 116     num_actions = 7
 117     context_dim = 10
 118     noise_stds = [0.01 * (i + 1) for i in range(num_actions)]
 119     num_nnz_dims = int(context_dim / 3.0)
 120     dataset, _, opt_sparse_linear = sample_sparse_linear_data(
 121         num_contexts, context_dim, num_actions, num_nnz_dims, sigma=noise_stds)
 122     opt_rewards, opt_actions = opt_sparse_linear
 123   elif data_type == 'mushroom':
 124     # Create mushroom dataset
 125     num_actions = 2
 126     context_dim = 117
 127     file_name = FLAGS.mushroom_data
 128     dataset, opt_mushroom = sample_mushroom_data(file_name, num_contexts)
 129     opt_rewards, opt_actions = opt_mushroom
 130   elif data_type == 'financial':
 131     num_actions = 8
 132     context_dim = 21
 133     num_contexts = min(3713, num_contexts)
 134     noise_stds = [0.01 * (i + 1) for i in range(num_actions)]
 135     file_name = FLAGS.financial_data
 136     dataset, opt_financial = sample_stock_data(file_name, context_dim,
 137                                                num_actions, num_contexts,
 138                                                noise_stds, shuffle_rows=True)
 139     opt_rewards, opt_actions = opt_financial
 140   elif data_type == 'jester':
 141     num_actions = 8
 142     context_dim = 32
 143     num_contexts = min(19181, num_contexts)
 144     file_name = FLAGS.jester_data
 145     dataset, opt_jester = sample_jester_data(file_name, context_dim,
 146                                              num_actions, num_contexts,
 147                                              shuffle_rows=True,
 148                                              shuffle_cols=True)
 149     opt_rewards, opt_actions = opt_jester
 150   elif data_type == 'statlog':
 151     file_name = FLAGS.statlog_data
 152     num_actions = 7
 153     num_contexts = min(43500, num_contexts)
 154     sampled_vals = sample_statlog_data(file_name, num_contexts,
 155                                        shuffle_rows=True)
 156     contexts, rewards, (opt_rewards, opt_actions) = sampled_vals
 157     dataset = np.hstack((contexts, rewards))
 158     context_dim = contexts.shape[1]
 159   elif data_type == 'adult':
 160     file_name = FLAGS.adult_data
 161     num_actions = 14
 162     num_contexts = min(45222, num_contexts)
 163     sampled_vals = sample_adult_data(file_name, num_contexts,
 164                                      shuffle_rows=True)
 165     contexts, rewards, (opt_rewards, opt_actions) = sampled_vals
 166     dataset = np.hstack((contexts, rewards))
 167     context_dim = contexts.shape[1]
 168   elif data_type == 'covertype':
 169     file_name = FLAGS.covertype_data
 170     num_actions = 7
 171     num_contexts = min(150000, num_contexts)
 172     sampled_vals = sample_covertype_data(file_name, num_contexts,
 173                                          shuffle_rows=True)
 174     contexts, rewards, (opt_rewards, opt_actions) = sampled_vals
 175     dataset = np.hstack((contexts, rewards))
 176     context_dim = contexts.shape[1]
 177   elif data_type == 'census':
 178     file_name = FLAGS.census_data
 179     num_actions = 9
 180     num_contexts = min(150000, num_contexts)
 181     sampled_vals = sample_census_data(file_name, num_contexts,
 182                                       shuffle_rows=True)
 183     contexts, rewards, (opt_rewards, opt_actions) = sampled_vals
 184     dataset = np.hstack((contexts, rewards))
 185     context_dim = contexts.shape[1]
 186   elif data_type == 'wheel':
 187     delta = 0.95
 188     num_actions = 5
 189     context_dim = 2
 190     mean_v = [1.0, 1.0, 1.0, 1.0, 1.2]
 191     std_v = [0.05, 0.05, 0.05, 0.05, 0.05]
 192     mu_large = 50
 193     std_large = 0.01
 194     dataset, opt_wheel = sample_wheel_bandit_data(num_contexts, delta,
 195                                                   mean_v, std_v,
 196                                                   mu_large, std_large)
 197     opt_rewards, opt_actions = opt_wheel
 198 
 199   return dataset, opt_rewards, opt_actions, num_actions, context_dim

 202 def display_results(algos, opt_rewards, opt_actions, h_rewards, t_init, name):
 203   """Displays summary statistics of the performance of each algorithm."""
 204 
 205   print('---------------------------------------------------')
 206   print('---------------------------------------------------')
 207   print('{} bandit completed after {} seconds.'.format(
 208     name, time.time() - t_init))
 209   print('---------------------------------------------------')
 210 
 211   performance_pairs = []
 212   for j, a in enumerate(algos):
 213     performance_pairs.append((a.name, np.sum(h_rewards[:, j])))
 214   performance_pairs = sorted(performance_pairs,
 215                              key=lambda elt: elt[1],
 216                              reverse=True)
 217   for i, (name, reward) in enumerate(performance_pairs):
 218     print('{:3}) {:20}| \t \t total reward = {:10}.'.format(i, name, reward))
 219 
 220   print('---------------------------------------------------')
 221   print('Optimal total reward = {}.'.format(np.sum(opt_rewards)))
 222   print('Frequency of optimal actions (action, frequency):')
 223   print([[elt, list(opt_actions).count(elt)] for elt in set(opt_actions)])
 224   print('---------------------------------------------------')
 225   print('---------------------------------------------------')
