  27 def one_hot(df, cols):
  28   """Returns one-hot encoding of DataFrame df including columns in cols."""
  29   for col in cols:
  30     dummies = pd.get_dummies(df[col], prefix=col, drop_first=False)
  31     df = pd.concat([df, dummies], axis=1)
  32     df = df.drop(col, axis=1)
  33   return df

  36 def sample_mushroom_data(file_name,
  37                          num_contexts,
  38                          r_noeat=0,
  39                          r_eat_safe=5,
  40                          r_eat_poison_bad=-35,
  41                          r_eat_poison_good=5,
  42                          prob_poison_bad=0.5):
  43   """Samples bandit game from Mushroom UCI Dataset.
  44 
  45   Args:
  46     file_name: Route of file containing the original Mushroom UCI dataset.
  47     num_contexts: Number of points to sample, i.e. (context, action rewards).
  48     r_noeat: Reward for not eating a mushroom.
  49     r_eat_safe: Reward for eating a non-poisonous mushroom.
  50     r_eat_poison_bad: Reward for eating a poisonous mushroom if harmed.
  51     r_eat_poison_good: Reward for eating a poisonous mushroom if not harmed.
  52     prob_poison_bad: Probability of being harmed by eating a poisonous mushroom.
  53 
  54   Returns:
  55     dataset: Sampled matrix with n rows: (context, eat_reward, no_eat_reward).
  56     opt_vals: Vector of expected optimal (reward, action) for each context.
  57 
  58   We assume r_eat_safe > r_noeat, and r_eat_poison_good > r_eat_poison_bad.
  59   """
  60 
  61   # first two cols of df encode whether mushroom is edible or poisonous
  62   df = pd.read_csv(file_name, header=None)
  63   df = one_hot(df, df.columns)
  64   ind = np.random.choice(range(df.shape[0]), num_contexts, replace=True)
  65 
  66   contexts = df.iloc[ind, 2:]
  67   no_eat_reward = r_noeat * np.ones((num_contexts, 1))
  68   random_poison = np.random.choice(
  69       [r_eat_poison_bad, r_eat_poison_good],
  70       p=[prob_poison_bad, 1 - prob_poison_bad],
  71       size=num_contexts)
  72   eat_reward = r_eat_safe * df.iloc[ind, 0]
  73   eat_reward += np.multiply(random_poison, df.iloc[ind, 1])
  74   eat_reward = eat_reward.values.reshape((num_contexts, 1))
  75 
  76   # compute optimal expected reward and optimal actions
  77   exp_eat_poison_reward = r_eat_poison_bad * prob_poison_bad
  78   exp_eat_poison_reward += r_eat_poison_good * (1 - prob_poison_bad)
  79   opt_exp_reward = r_eat_safe * df.iloc[ind, 0] + max(
  80       r_noeat, exp_eat_poison_reward) * df.iloc[ind, 1]
  81 
  82   if r_noeat > exp_eat_poison_reward:
  83     # actions: no eat = 0 ; eat = 1
  84     opt_actions = df.iloc[ind, 0]  # indicator of edible
  85   else:
  86     # should always eat (higher expected reward)
  87     opt_actions = np.ones((num_contexts, 1))
  88 
  89   opt_vals = (opt_exp_reward.values, opt_actions.values)
  90 
  91   return np.hstack((contexts, no_eat_reward, eat_reward)), opt_vals

  94 def sample_stock_data(file_name, context_dim, num_actions, num_contexts,
  95                       sigma, shuffle_rows=True):
  96   """Samples linear bandit game from stock prices dataset.
  97 
  98   Args:
  99     file_name: Route of file containing the stock prices dataset.
 100     context_dim: Context dimension (i.e. vector with the price of each stock).
 101     num_actions: Number of actions (different linear portfolio strategies).
 102     num_contexts: Number of contexts to sample.
 103     sigma: Vector with additive noise levels for each action.
 104     shuffle_rows: If True, rows from original dataset are shuffled.
 105 
 106   Returns:
 107     dataset: Sampled matrix with rows: (context, reward_1, ..., reward_k).
 108     opt_vals: Vector of expected optimal (reward, action) for each context.
 109   """
 110 
 111   with tf.gfile.Open(file_name, 'r') as f:
 112     contexts = np.loadtxt(f, skiprows=1)
 113 
 114   if shuffle_rows:
 115     np.random.shuffle(contexts)
 116   contexts = contexts[:num_contexts, :]
 117 
 118   betas = np.random.uniform(-1, 1, (context_dim, num_actions))
 119   betas /= np.linalg.norm(betas, axis=0)
 120 
 121   mean_rewards = np.dot(contexts, betas)
 122   noise = np.random.normal(scale=sigma, size=mean_rewards.shape)
 123   rewards = mean_rewards + noise
 124 
 125   opt_actions = np.argmax(mean_rewards, axis=1)
 126   opt_rewards = [mean_rewards[i, a] for i, a in enumerate(opt_actions)]
 127   return np.hstack((contexts, rewards)), (np.array(opt_rewards), opt_actions)

 130 def sample_jester_data(file_name, context_dim, num_actions, num_contexts,
 131                        shuffle_rows=True, shuffle_cols=False):
 132   """Samples bandit game from (user, joke) dense subset of Jester dataset.
 133 
 134   Args:
 135     file_name: Route of file containing the modified Jester dataset.
 136     context_dim: Context dimension (i.e. vector with some ratings from a user).
 137     num_actions: Number of actions (number of joke ratings to predict).
 138     num_contexts: Number of contexts to sample.
 139     shuffle_rows: If True, rows from original dataset are shuffled.
 140     shuffle_cols: Whether or not context/action jokes are randomly shuffled.
 141 
 142   Returns:
 143     dataset: Sampled matrix with rows: (context, rating_1, ..., rating_k).
 144     opt_vals: Vector of deterministic optimal (reward, action) for each context.
 145   """
 146 
 147   with tf.gfile.Open(file_name, 'rb') as f:
 148     dataset = np.load(f)
 149 
 150   if shuffle_cols:
 151     dataset = dataset[:, np.random.permutation(dataset.shape[1])]
 152   if shuffle_rows:
 153     np.random.shuffle(dataset)
 154   dataset = dataset[:num_contexts, :]
 155 
 156   assert context_dim + num_actions == dataset.shape[1], 'Wrong data dimensions.'
 157 
 158   opt_actions = np.argmax(dataset[:, context_dim:], axis=1)
 159   opt_rewards = np.array([dataset[i, context_dim + a]
 160                           for i, a in enumerate(opt_actions)])
 161 
 162   return dataset, (opt_rewards, opt_actions)

 165 def sample_statlog_data(file_name, num_contexts, shuffle_rows=True,
 166                         remove_underrepresented=False):
 167   """Returns bandit problem dataset based on the UCI statlog data.
 168 
 169   Args:
 170     file_name: Route of file containing the Statlog dataset.
 171     num_contexts: Number of contexts to sample.
 172     shuffle_rows: If True, rows from original dataset are shuffled.
 173     remove_underrepresented: If True, removes arms with very few rewards.
 174 
 175   Returns:
 176     dataset: Sampled matrix with rows: (context, action rewards).
 177     opt_vals: Vector of deterministic optimal (reward, action) for each context.
 178 
 179   https://archive.ics.uci.edu/ml/datasets/Statlog+(Shuttle)
 180   """
 181 
 182   with tf.gfile.Open(file_name, 'r') as f:
 183     data = np.loadtxt(f)
 184 
 185   num_actions = 7  # some of the actions are very rarely optimal.
 186 
 187   # Shuffle data
 188   if shuffle_rows:
 189     np.random.shuffle(data)
 190   data = data[:num_contexts, :]
 191 
 192   # Last column is label, rest are features
 193   contexts = data[:, :-1]
 194   labels = data[:, -1].astype(int) - 1  # convert to 0 based index
 195 
 196   if remove_underrepresented:
 197     contexts, labels = remove_underrepresented_classes(contexts, labels)
 198 
 199   return classification_to_bandit_problem(contexts, labels, num_actions)

 202 def sample_adult_data(file_name, num_contexts, shuffle_rows=True,
 203                       remove_underrepresented=False):
 204   """Returns bandit problem dataset based on the UCI adult data.
 205 
 206   Args:
 207     file_name: Route of file containing the Adult dataset.
 208     num_contexts: Number of contexts to sample.
 209     shuffle_rows: If True, rows from original dataset are shuffled.
 210     remove_underrepresented: If True, removes arms with very few rewards.
 211 
 212   Returns:
 213     dataset: Sampled matrix with rows: (context, action rewards).
 214     opt_vals: Vector of deterministic optimal (reward, action) for each context.
 215 
 216   Preprocessing:
 217     * drop rows with missing values
 218     * convert categorical variables to 1 hot encoding
 219 
 220   https://archive.ics.uci.edu/ml/datasets/census+income
 221   """
 222   with tf.gfile.Open(file_name, 'r') as f:
 223     df = pd.read_csv(f, header=None,
 224                      na_values=[' ?']).dropna()
 225 
 226   num_actions = 14
 227 
 228   if shuffle_rows:
 229     df = df.sample(frac=1)
 230   df = df.iloc[:num_contexts, :]
 231 
 232   labels = df[6].astype('category').cat.codes.as_matrix()
 233   df = df.drop([6], axis=1)
 234 
 235   # Convert categorical variables to 1 hot encoding
 236   cols_to_transform = [1, 3, 5, 7, 8, 9, 13, 14]
 237   df = pd.get_dummies(df, columns=cols_to_transform)
 238 
 239   if remove_underrepresented:
 240     df, labels = remove_underrepresented_classes(df, labels)
 241   contexts = df.as_matrix()
 242 
 243   return classification_to_bandit_problem(contexts, labels, num_actions)

 246 def sample_census_data(file_name, num_contexts, shuffle_rows=True,
 247                        remove_underrepresented=False):
 248   """Returns bandit problem dataset based on the UCI census data.
 249 
 250   Args:
 251     file_name: Route of file containing the Census dataset.
 252     num_contexts: Number of contexts to sample.
 253     shuffle_rows: If True, rows from original dataset are shuffled.
 254     remove_underrepresented: If True, removes arms with very few rewards.
 255 
 256   Returns:
 257     dataset: Sampled matrix with rows: (context, action rewards).
 258     opt_vals: Vector of deterministic optimal (reward, action) for each context.
 259 
 260   Preprocessing:
 261     * drop rows with missing labels
 262     * convert categorical variables to 1 hot encoding
 263 
 264   Note: this is the processed (not the 'raw') dataset. It contains a subset
 265   of the raw features and they've all been discretized.
 266 
 267   https://archive.ics.uci.edu/ml/datasets/US+Census+Data+%281990%29
 268   """
 269   # Note: this dataset is quite large. It will be slow to load and preprocess.
 270   with tf.gfile.Open(file_name, 'r') as f:
 271     df = (pd.read_csv(f, header=0, na_values=['?'])
 272           .dropna())
 273 
 274   num_actions = 9
 275 
 276   if shuffle_rows:
 277     df = df.sample(frac=1)
 278   df = df.iloc[:num_contexts, :]
 279 
 280   # Assuming what the paper calls response variable is the label?
 281   labels = df['dOccup'].astype('category').cat.codes.as_matrix()
 282   # In addition to label, also drop the (unique?) key.
 283   df = df.drop(['dOccup', 'caseid'], axis=1)
 284 
 285   # All columns are categorical. Convert to 1 hot encoding.
 286   df = pd.get_dummies(df, columns=df.columns)
 287 
 288   if remove_underrepresented:
 289     df, labels = remove_underrepresented_classes(df, labels)
 290   contexts = df.as_matrix()
 291 
 292   return classification_to_bandit_problem(contexts, labels, num_actions)

 295 def sample_covertype_data(file_name, num_contexts, shuffle_rows=True,
 296                           remove_underrepresented=False):
 297   """Returns bandit problem dataset based on the UCI Cover_Type data.
 298 
 299   Args:
 300     file_name: Route of file containing the Covertype dataset.
 301     num_contexts: Number of contexts to sample.
 302     shuffle_rows: If True, rows from original dataset are shuffled.
 303     remove_underrepresented: If True, removes arms with very few rewards.
 304 
 305   Returns:
 306     dataset: Sampled matrix with rows: (context, action rewards).
 307     opt_vals: Vector of deterministic optimal (reward, action) for each context.
 308 
 309   Preprocessing:
 310     * drop rows with missing labels
 311     * convert categorical variables to 1 hot encoding
 312 
 313   https://archive.ics.uci.edu/ml/datasets/Covertype
 314   """
 315   with tf.gfile.Open(file_name, 'r') as f:
 316     df = (pd.read_csv(f, header=0, na_values=['?'])
 317           .dropna())
 318 
 319   num_actions = 7
 320 
 321   if shuffle_rows:
 322     df = df.sample(frac=1)
 323   df = df.iloc[:num_contexts, :]
 324 
 325   # Assuming what the paper calls response variable is the label?
 326   # Last column is label.
 327   labels = df[df.columns[-1]].astype('category').cat.codes.as_matrix()
 328   df = df.drop([df.columns[-1]], axis=1)
 329 
 330   # All columns are either quantitative or already converted to 1 hot.
 331   if remove_underrepresented:
 332     df, labels = remove_underrepresented_classes(df, labels)
 333   contexts = df.as_matrix()
 334 
 335   return classification_to_bandit_problem(contexts, labels, num_actions)

 338 def classification_to_bandit_problem(contexts, labels, num_actions=None):
 339   """Normalize contexts and encode deterministic rewards."""
 340 
 341   if num_actions is None:
 342     num_actions = np.max(labels) + 1
 343   num_contexts = contexts.shape[0]
 344 
 345   # Due to random subsampling in small problems, some features may be constant
 346   sstd = safe_std(np.std(contexts, axis=0, keepdims=True)[0, :])
 347 
 348   # Normalize features
 349   contexts = ((contexts - np.mean(contexts, axis=0, keepdims=True)) / sstd)
 350 
 351   # One hot encode labels as rewards
 352   rewards = np.zeros((num_contexts, num_actions))
 353   rewards[np.arange(num_contexts), labels] = 1.0
 354 
 355   return contexts, rewards, (np.ones(num_contexts), labels)

 358 def safe_std(values):
 359   """Remove zero std values for ones."""
 360   return np.array([val if val != 0.0 else 1.0 for val in values])

 363 def remove_underrepresented_classes(features, labels, thresh=0.0005):
 364   """Removes classes when number of datapoints fraction is below a threshold."""
 365 
 366   # Threshold doesn't seem to agree with https://arxiv.org/pdf/1706.04687.pdf
 367   # Example: for Covertype, they report 4 classes after filtering, we get 7?
 368   total_count = labels.shape[0]
 369   unique, counts = np.unique(labels, return_counts=True)
 370   ratios = counts.astype('float') / total_count
 371   vals_and_ratios = dict(zip(unique, ratios))
 372   print('Unique classes and their ratio of total: %s' % vals_and_ratios)
 373   keep = [vals_and_ratios[v] >= thresh for v in labels]
 374   return features[keep], labels[np.array(keep)]
