  25 def sample_contextual_data(num_contexts, dim_context, num_actions, sigma):
  26   """Samples independent Gaussian data.
  27 
  28   There is nothing to learn here as the rewards do not depend on the context.
  29 
  30   Args:
  31     num_contexts: Number of contexts to sample.
  32     dim_context: Dimension of the contexts.
  33     num_actions: Number of arms for the multi-armed bandit.
  34     sigma: Standard deviation of the independent Gaussian samples.
  35 
  36   Returns:
  37     data: A [num_contexts, dim_context + num_actions] numpy array with the data.
  38   """
  39   size_data = [num_contexts, dim_context + num_actions]
  40   return np.random.normal(scale=sigma, size=size_data)

  43 def sample_linear_data(num_contexts, dim_context, num_actions, sigma=0.0):
  44   """Samples data from linearly parameterized arms.
  45 
  46   The reward for context X and arm j is given by X^T beta_j, for some latent
  47   set of parameters {beta_j : j = 1, ..., k}. The beta's are sampled uniformly
  48   at random, the contexts are Gaussian, and sigma-noise is added to the rewards.
  49 
  50   Args:
  51     num_contexts: Number of contexts to sample.
  52     dim_context: Dimension of the contexts.
  53     num_actions: Number of arms for the multi-armed bandit.
  54     sigma: Standard deviation of the additive noise. Set to zero for no noise.
  55 
  56   Returns:
  57     data: A [n, d+k] numpy array with the data.
  58     betas: Latent parameters that determine expected reward for each arm.
  59     opt: (optimal_rewards, optimal_actions) for all contexts.
  60   """
  61 
  62   betas = np.random.uniform(-1, 1, (dim_context, num_actions))
  63   betas /= np.linalg.norm(betas, axis=0)
  64   contexts = np.random.normal(size=[num_contexts, dim_context])
  65   rewards = np.dot(contexts, betas)
  66   opt_actions = np.argmax(rewards, axis=1)
  67   rewards += np.random.normal(scale=sigma, size=rewards.shape)
  68   opt_rewards = np.array([rewards[i, act] for i, act in enumerate(opt_actions)])
  69   return np.hstack((contexts, rewards)), betas, (opt_rewards, opt_actions)

  72 def sample_sparse_linear_data(num_contexts, dim_context, num_actions,
  73                               sparse_dim, sigma=0.0):
  74   """Samples data from sparse linearly parameterized arms.
  75 
  76   The reward for context X and arm j is given by X^T beta_j, for some latent
  77   set of parameters {beta_j : j = 1, ..., k}. The beta's are sampled uniformly
  78   at random, the contexts are Gaussian, and sigma-noise is added to the rewards.
  79   Only s components out of d are non-zero for each arm's beta.
  80 
  81   Args:
  82     num_contexts: Number of contexts to sample.
  83     dim_context: Dimension of the contexts.
  84     num_actions: Number of arms for the multi-armed bandit.
  85     sparse_dim: Dimension of the latent subspace (sparsity pattern dimension).
  86     sigma: Standard deviation of the additive noise. Set to zero for no noise.
  87 
  88   Returns:
  89     data: A [num_contexts, dim_context+num_actions] numpy array with the data.
  90     betas: Latent parameters that determine expected reward for each arm.
  91     opt: (optimal_rewards, optimal_actions) for all contexts.
  92   """
  93 
  94   flatten = lambda l: [item for sublist in l for item in sublist]
  95   sparse_pattern = flatten(
  96       [[(j, i) for j in np.random.choice(range(dim_context),
  97                                          sparse_dim,
  98                                          replace=False)]
  99        for i in range(num_actions)])
 100   betas = np.random.uniform(-1, 1, (dim_context, num_actions))
 101   mask = np.zeros((dim_context, num_actions))
 102   for elt in sparse_pattern:
 103     mask[elt] = 1
 104   betas = np.multiply(betas, mask)
 105   betas /= np.linalg.norm(betas, axis=0)
 106   contexts = np.random.normal(size=[num_contexts, dim_context])
 107   rewards = np.dot(contexts, betas)
 108   opt_actions = np.argmax(rewards, axis=1)
 109   rewards += np.random.normal(scale=sigma, size=rewards.shape)
 110   opt_rewards = np.array([rewards[i, act] for i, act in enumerate(opt_actions)])
 111   return np.hstack((contexts, rewards)), betas, (opt_rewards, opt_actions)

 114 def sample_wheel_bandit_data(num_contexts, delta, mean_v, std_v,
 115                              mu_large, std_large):
 116   """Samples from Wheel bandit game (see https://arxiv.org/abs/1802.09127).
 117 
 118   Args:
 119     num_contexts: Number of points to sample, i.e. (context, action rewards).
 120     delta: Exploration parameter: high reward in one region if norm above delta.
 121     mean_v: Mean reward for each action if context norm is below delta.
 122     std_v: Gaussian reward std for each action if context norm is below delta.
 123     mu_large: Mean reward for optimal action if context norm is above delta.
 124     std_large: Reward std for optimal action if context norm is above delta.
 125 
 126   Returns:
 127     dataset: Sampled matrix with n rows: (context, action rewards).
 128     opt_vals: Vector of expected optimal (reward, action) for each context.
 129   """
 130 
 131   context_dim = 2
 132   num_actions = 5
 133 
 134   data = []
 135   rewards = []
 136   opt_actions = []
 137   opt_rewards = []
 138 
 139   # sample uniform contexts in unit ball
 140   while len(data) < num_contexts:
 141     raw_data = np.random.uniform(-1, 1, (int(num_contexts / 3), context_dim))
 142 
 143     for i in range(raw_data.shape[0]):
 144       if np.linalg.norm(raw_data[i, :]) <= 1:
 145         data.append(raw_data[i, :])
 146 
 147   contexts = np.stack(data)[:num_contexts, :]
 148 
 149   # sample rewards
 150   for i in range(num_contexts):
 151     r = [np.random.normal(mean_v[j], std_v[j]) for j in range(num_actions)]
 152     if np.linalg.norm(contexts[i, :]) >= delta:
 153       # large reward in the right region for the context
 154       r_big = np.random.normal(mu_large, std_large)
 155       if contexts[i, 0] > 0:
 156         if contexts[i, 1] > 0:
 157           r[0] = r_big
 158           opt_actions.append(0)
 159         else:
 160           r[1] = r_big
 161           opt_actions.append(1)
 162       else:
 163         if contexts[i, 1] > 0:
 164           r[2] = r_big
 165           opt_actions.append(2)
 166         else:
 167           r[3] = r_big
 168           opt_actions.append(3)
 169     else:
 170       opt_actions.append(np.argmax(mean_v))
 171 
 172     opt_rewards.append(r[opt_actions[-1]])
 173     rewards.append(r)
 174 
 175   rewards = np.stack(rewards)
 176   opt_rewards = np.array(opt_rewards)
 177   opt_actions = np.array(opt_actions)
 178 
 179   return np.hstack((contexts, rewards)), (opt_rewards, opt_actions)
