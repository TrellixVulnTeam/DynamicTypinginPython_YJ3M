  16 """Define a data buffer for contextual bandit algorithms."""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import numpy as np
  23 
  24 
  25 class ContextualDataset(object):
  26   """The buffer is able to append new data, and sample random minibatches."""
  27 
  28   def __init__(self, context_dim, num_actions, buffer_s=-1, intercept=False):
  29     """Creates a ContextualDataset object.
  30 
  31     The data is stored in attributes: contexts and rewards.
  32     The sequence of taken actions are stored in attribute actions.
  33 
  34     Args:
  35       context_dim: Dimension of the contexts.
  36       num_actions: Number of arms for the multi-armed bandit.
  37       buffer_s: Size of buffer for training. Only last buffer_s will be
  38         returned as minibatch. If buffer_s = -1, all data will be used.
  39       intercept: If True, it adds a constant (1.0) dimension to each context X,
  40         at the end.
  41     """
  42 
  43     self._context_dim = context_dim
  44     self._num_actions = num_actions
  45     self._contexts = None
  46     self._rewards = None
  47     self.actions = []
  48     self.buffer_s = buffer_s
  49     self.intercept = intercept
  50 
  51   def add(self, context, action, reward):
  52     """Adds a new triplet (context, action, reward) to the dataset.
  53 
  54     The reward for the actions that weren't played is assumed to be zero.
  55 
  56     Args:
  57       context: A d-dimensional vector with the context.
  58       action: Integer between 0 and k-1 representing the chosen arm.
  59       reward: Real number representing the reward for the (context, action).
  60     """
  61 
  62     if self.intercept:
  63       c = np.array(context[:])
  64       c = np.append(c, 1.0).reshape((1, self.context_dim + 1))
  65     else:
  66       c = np.array(context[:]).reshape((1, self.context_dim))
  67 
  68     if self.contexts is None:
  69       self.contexts = c
  70     else:
  71       self.contexts = np.vstack((self.contexts, c))
  72 
  73     r = np.zeros((1, self.num_actions))
  74     r[0, action] = reward
  75     if self.rewards is None:
  76       self.rewards = r
  77     else:
  78       self.rewards = np.vstack((self.rewards, r))
  79 
  80     self.actions.append(action)
  81 
  82   def replace_data(self, contexts=None, actions=None, rewards=None):
  83     if contexts is not None:
  84       self.contexts = contexts
  85     if actions is not None:
  86       self.actions = actions
  87     if rewards is not None:
  88       self.rewards = rewards
  89 
  90   def get_batch(self, batch_size):
  91     """Returns a random minibatch of (contexts, rewards) with batch_size."""
  92     n, _ = self.contexts.shape
  93     if self.buffer_s == -1:
  94       # use all the data
  95       ind = np.random.choice(range(n), batch_size)
  96     else:
  97       # use only buffer (last buffer_s observations)
  98       ind = np.random.choice(range(max(0, n - self.buffer_s), n), batch_size)
  99     return self.contexts[ind, :], self.rewards[ind, :]
 100 
 101   def get_data(self, action):
 102     """Returns all (context, reward) where the action was played."""
 103     n, _ = self.contexts.shape
 104     ind = np.array([i for i in range(n) if self.actions[i] == action])
 105     return self.contexts[ind, :], self.rewards[ind, action]
 106 
 107   def get_data_with_weights(self):
 108     """Returns all observations with one-hot weights for actions."""
 109     weights = np.zeros((self.contexts.shape[0], self.num_actions))
 110     a_ind = np.array([(i, val) for i, val in enumerate(self.actions)])
 111     weights[a_ind[:, 0], a_ind[:, 1]] = 1.0
 112     return self.contexts, self.rewards, weights
 113 
 114   def get_batch_with_weights(self, batch_size):
 115     """Returns a random mini-batch with one-hot weights for actions."""
 116     n, _ = self.contexts.shape
 117     if self.buffer_s == -1:
 118       # use all the data
 119       ind = np.random.choice(range(n), batch_size)
 120     else:
 121       # use only buffer (last buffer_s obs)
 122       ind = np.random.choice(range(max(0, n - self.buffer_s), n), batch_size)
 123 
 124     weights = np.zeros((batch_size, self.num_actions))
 125     sampled_actions = np.array(self.actions)[ind]
 126     a_ind = np.array([(i, val) for i, val in enumerate(sampled_actions)])
 127     weights[a_ind[:, 0], a_ind[:, 1]] = 1.0
 128     return self.contexts[ind, :], self.rewards[ind, :], weights
 129 
 130   def num_points(self, f=None):
 131     """Returns number of points in the buffer (after applying function f)."""
 132     if f is not None:
 133       return f(self.contexts.shape[0])
 134     return self.contexts.shape[0]
