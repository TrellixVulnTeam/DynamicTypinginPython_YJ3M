  16 """Define a contextual bandit from which we can sample and compute rewards.
  17 
  18 We can feed the data, sample a context, its reward for a specific action, and
  19 also the optimal action for a given context.
  20 """
  21 
  22 from __future__ import absolute_import
  23 from __future__ import division
  24 from __future__ import print_function
  25 
  26 import numpy as np
  27

  29 def run_contextual_bandit(context_dim, num_actions, dataset, algos):
  30   """Run a contextual bandit problem on a set of algorithms.
  31 
  32   Args:
  33     context_dim: Dimension of the context.
  34     num_actions: Number of available actions.
  35     dataset: Matrix where every row is a context + num_actions rewards.
  36     algos: List of algorithms to use in the contextual bandit instance.
  37 
  38   Returns:
  39     h_actions: Matrix with actions: size (num_context, num_algorithms).
  40     h_rewards: Matrix with rewards: size (num_context, num_algorithms).
  41   """
  42 
  43   num_contexts = dataset.shape[0]
  44 
  45   # Create contextual bandit
  46   cmab = ContextualBandit(context_dim, num_actions)
  47   cmab.feed_data(dataset)
  48 
  49   h_actions = np.empty((0, len(algos)), float)
  50   h_rewards = np.empty((0, len(algos)), float)
  51 
  52   # Run the contextual bandit process
  53   for i in range(num_contexts):
  54     context = cmab.context(i)
  55     actions = [a.action(context) for a in algos]
  56     rewards = [cmab.reward(i, action) for action in actions]
  57 
  58     for j, a in enumerate(algos):
  59       a.update(context, actions[j], rewards[j])
  60 
  61     h_actions = np.vstack((h_actions, np.array(actions)))
  62     h_rewards = np.vstack((h_rewards, np.array(rewards)))
  63 
  64   return h_actions, h_rewards

  67 class ContextualBandit(object):
  68   """Implements a Contextual Bandit with d-dimensional contexts and k arms."""
  69 
  70   def __init__(self, context_dim, num_actions):
  71     """Creates a contextual bandit object.
  72 
  73     Args:
  74       context_dim: Dimension of the contexts.
  75       num_actions: Number of arms for the multi-armed bandit.
  76     """
  77 
  78     self._context_dim = context_dim
  79     self._num_actions = num_actions
  80 
  81   def feed_data(self, data):
  82     """Feeds the data (contexts + rewards) to the bandit object.
  83 
  84     Args:
  85       data: Numpy array with shape [n, d+k], where n is the number of contexts,
  86         d is the dimension of each context, and k the number of arms (rewards).
  87 
  88     Raises:
  89       ValueError: when data dimensions do not correspond to the object values.
  90     """
  91 
  92     if data.shape[1] != self.context_dim + self.num_actions:
  93       raise ValueError('Data dimensions do not match.')
  94 
  95     self._number_contexts = data.shape[0]
  96     self.data = data
  97     self.order = range(self.number_contexts)
  98 
  99   def reset(self):
 100     """Randomly shuffle the order of the contexts to deliver."""
 101     self.order = np.random.permutation(self.number_contexts)
 102 
 103   def context(self, number):
 104     """Returns the number-th context."""
 105     return self.data[self.order[number]][:self.context_dim]
 106 
 107   def reward(self, number, action):
 108     """Returns the reward for the number-th context and action."""
 109     return self.data[self.order[number]][self.context_dim + action]
 110 
 111   def optimal(self, number):
 112     """Returns the optimal action (in hindsight) for the number-th context."""
 113     return np.argmax(self.data[self.order[number]][self.context_dim:])
 114 
 115   @property
 116   def context_dim(self):
 117     return self._context_dim
 118 
 119   @property
 120   def num_actions(self):
 121     return self._num_actions
 122 
 123   @property
 124   def number_contexts(self):
 125     return self._number_contexts
