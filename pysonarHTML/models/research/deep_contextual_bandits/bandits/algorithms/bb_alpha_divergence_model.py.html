  16 """Bayesian NN using expectation propagation (Black-Box Alpha-Divergence).
  17 
  18 See https://arxiv.org/abs/1511.03243 for details.
  19 All formulas used in this implementation are derived in:
  20 https://www.overleaf.com/12837696kwzjxkyhdytk#/49028744/.

  36 tfd = tf.contrib.distributions  # update to: tensorflow_probability.distributions

  39 def log_gaussian(x, mu, sigma, reduce_sum=True):
  40   res = tfd.Normal(mu, sigma).log_prob(x)
  41   if reduce_sum:
  42     return tf.reduce_sum(res)
  43   else:
  44     return res

  47 class BBAlphaDivergence(BayesianNN):
  48   """Implements an approximate Bayesian NN via Black-Box Alpha-Divergence."""
  49 
  50   def __init__(self, hparams, name):
  51 
  52     self.name = name
  53     self.hparams = hparams
  54 
  55     self.alpha = getattr(self.hparams, 'alpha', 1.0)
  56     self.num_mc_nn_samples = getattr(self.hparams, 'num_mc_nn_samples', 10)
  57 
  58     self.n_in = self.hparams.context_dim
  59     self.n_out = self.hparams.num_actions
  60     self.layers = self.hparams.layer_sizes
  61     self.batch_size = self.hparams.batch_size
  62 
  63     self.show_training = self.hparams.show_training
  64     self.freq_summary = self.hparams.freq_summary
  65     self.verbose = getattr(self.hparams, 'verbose', True)
  66 
  67     self.cleared_times_trained = self.hparams.cleared_times_trained
  68     self.initial_training_steps = self.hparams.initial_training_steps
  69     self.training_schedule = np.linspace(self.initial_training_steps,
  70                                          self.hparams.training_epochs,
  71                                          self.cleared_times_trained)
  72 
  73     self.times_trained = 0
  74     self.initialize_model()
  75 
  76   def initialize_model(self):
  77     """Builds and initialize the model."""
  78 
  79     self.num_w = 0
  80     self.num_b = 0
  81 
  82     self.weights_m = {}
  83     self.weights_std = {}
  84     self.biases_m = {}
  85     self.biases_std = {}
  86 
  87     self.h_max_var = []
  88 
  89     if self.hparams.use_sigma_exp_transform:
  90       self.sigma_transform = tfd.bijectors.Exp()
  91     else:
  92       self.sigma_transform = tfd.bijectors.Softplus()
  93 
  94     # Build the graph corresponding to the Bayesian NN instance.
  95     self.graph = tf.Graph()
  96 
  97     with self.graph.as_default():
  98 
  99       self.sess = tf.Session()
 100       self.x = tf.placeholder(shape=[None, self.n_in],
 101                               dtype=tf.float32, name='x')
 102       self.y = tf.placeholder(shape=[None, self.n_out],
 103                               dtype=tf.float32, name='y')
 104       self.weights = tf.placeholder(shape=[None, self.n_out],
 105                                     dtype=tf.float32, name='w')
 106       self.data_size = tf.placeholder(tf.float32, shape=(), name='data_size')
 107 
 108       self.prior_variance = self.hparams.prior_variance
 109       if self.prior_variance < 0:
 110         # if not fixed, we learn the prior.
 111         self.prior_variance = self.sigma_transform.forward(
 112             self.build_mu_variable([1, 1]))
 113 
 114       self.build_model()
 115       self.sess.run(tf.global_variables_initializer())
 116 
 117   def build_mu_variable(self, shape):
 118     """Returns a mean variable initialized as N(0, 0.05)."""
 119     return tf.Variable(tf.random_normal(shape, 0.0, 0.05))
 120 
 121   def build_sigma_variable(self, shape, init=-5.):
 122     """Returns a sigma variable initialized as N(init, 0.05)."""
 123     # Initialize sigma to be very small initially to encourage MAP opt first
 124     return tf.Variable(tf.random_normal(shape, init, 0.05))
 125 
 126   def build_layer(self, input_x, shape, layer_id, activation_fn=tf.nn.relu):
 127     """Builds a layer with N(mean, std) for each weight, and samples from it."""
 128 
 129     w_mu = self.build_mu_variable(shape)
 130     w_sigma = self.sigma_transform.forward(self.build_sigma_variable(shape))
 131 
 132     w_noise = tf.random_normal(shape)
 133     w = w_mu + w_sigma * w_noise
 134 
 135     b_mu = self.build_mu_variable([1, shape[1]])
 136     b_sigma = self.sigma_transform.forward(
 137         self.build_sigma_variable([1, shape[1]]))
 138 
 139     b_noise = tf.random_normal([1, shape[1]])
 140     b = b_mu + b_sigma * b_noise
 141 
 142     # Create outputs
 143     output_h = activation_fn(tf.matmul(input_x, w) + b)
 144 
 145     # Store means and stds
 146     self.weights_m[layer_id] = w_mu
 147     self.weights_std[layer_id] = w_sigma
 148     self.biases_m[layer_id] = b_mu
 149     self.biases_std[layer_id] = b_sigma
 150 
 151     return output_h
 152 
 153   def sample_neural_network(self, activation_fn=tf.nn.relu):
 154     """Samples a nn from posterior, computes data log lk and log f factor."""
 155 
 156     with self.graph.as_default():
 157 
 158       log_f = 0
 159       n = self.data_size
 160       input_x = self.x
 161 
 162       for layer_id in range(self.total_layers):
 163 
 164         # load mean and std of each weight
 165         w_mu = self.weights_m[layer_id]
 166         w_sigma = self.weights_std[layer_id]
 167         b_mu = self.biases_m[layer_id]
 168         b_sigma = self.biases_std[layer_id]
 169 
 170         # sample weights from Gaussian distribution
 171         shape = w_mu.shape
 172         w_noise = tf.random_normal(shape)
 173         b_noise = tf.random_normal([1, int(shape[1])])
 174         w = w_mu + w_sigma * w_noise
 175         b = b_mu + b_sigma * b_noise
 176 
 177         # compute contribution to log_f
 178         t1 = w * w_mu / (n * w_sigma ** 2)
 179         t2 = (0.5 * w ** 2 / n) * (1 / self.prior_variance - 1 / w_sigma ** 2)
 180         log_f += tf.reduce_sum(t1 + t2)
 181 
 182         t1 = b * b_mu / (n * b_sigma ** 2)
 183         t2 = (0.5 * b ** 2 / n) * (1 / self.prior_variance - 1 / b_sigma ** 2)
 184         log_f += tf.reduce_sum(t1 + t2)
 185 
 186         if layer_id < self.total_layers - 1:
 187           output_h = activation_fn(tf.matmul(input_x, w) + b)
 188         else:
 189           output_h = tf.matmul(input_x, w) + b
 190 
 191         input_x = output_h
 192 
 193       # compute log likelihood of the observed reward under the sampled nn
 194       log_likelihood = log_gaussian(
 195           self.y, output_h, self.noise_sigma, reduce_sum=False)
 196       weighted_log_likelihood = tf.reduce_sum(log_likelihood * self.weights, -1)
 197 
 198     return log_f, weighted_log_likelihood
 199 
 200   def log_z_q(self):
 201     """Computes log-partition function of current posterior parameters."""
 202 
 203     with self.graph.as_default():
 204 
 205       log_z_q = 0
 206 
 207       for layer_id in range(self.total_layers):
 208 
 209         w_mu = self.weights_m[layer_id]
 210         w_sigma = self.weights_std[layer_id]
 211         b_mu = self.biases_m[layer_id]
 212         b_sigma = self.biases_std[layer_id]
 213 
 214         w_term = 0.5 * tf.reduce_sum(w_mu ** 2 / w_sigma ** 2)
 215         w_term += 0.5 * tf.reduce_sum(tf.log(2 * np.pi) + 2 * tf.log(w_sigma))
 216 
 217         b_term = 0.5 * tf.reduce_sum(b_mu ** 2 / b_sigma ** 2)
 218         b_term += 0.5 * tf.reduce_sum(tf.log(2 * np.pi) + 2 * tf.log(b_sigma))
 219 
 220         log_z_q += w_term + b_term
 221 
 222       return log_z_q
 223 
 224   def log_z_prior(self):
 225     """Computes log-partition function of the prior parameters."""
 226     num_params = self.num_w + self.num_b
 227     return num_params * 0.5 * tf.log(2 * np.pi * self.prior_variance)
 228 
 229   def log_alpha_likelihood_ratio(self, activation_fn=tf.nn.relu):
 230 
 231     # each nn sample returns (log f, log likelihoods)
 232     nn_samples = [
 233         self.sample_neural_network(activation_fn)
 234         for _ in range(self.num_mc_nn_samples)
 235     ]
 236     nn_log_f_samples = [elt[0] for elt in nn_samples]
 237     nn_log_lk_samples = [elt[1] for elt in nn_samples]
 238 
 239     # we stack the (log f, log likelihoods) from the k nn samples
 240     nn_log_f_stack = tf.stack(nn_log_f_samples)      # k x 1
 241     nn_log_lk_stack = tf.stack(nn_log_lk_samples)    # k x N
 242     nn_f_tile = tf.tile(nn_log_f_stack, [self.batch_size])
 243     nn_f_tile = tf.reshape(nn_f_tile,
 244                            [self.num_mc_nn_samples, self.batch_size])
 245 
 246     # now both the log f and log likelihood terms have shape: k x N
 247     # apply formula in https://www.overleaf.com/12837696kwzjxkyhdytk#/49028744/
 248     nn_log_ratio = nn_log_lk_stack - nn_f_tile
 249     nn_log_ratio = self.alpha * tf.transpose(nn_log_ratio)
 250     logsumexp_value = tf.reduce_logsumexp(nn_log_ratio, -1)
 251     log_k_scalar = tf.log(tf.cast(self.num_mc_nn_samples, tf.float32))
 252     log_k = log_k_scalar * tf.ones([self.batch_size])
 253 
 254     return tf.reduce_sum(logsumexp_value - log_k, -1)

 256   def build_model(self, activation_fn=tf.nn.relu):
 257     """Defines the actual NN model with fully connected layers.
 258 
 259     Args:
 260       activation_fn: Activation function for the neural network.
 261 
 262     The loss is computed for partial feedback settings (bandits), so only
 263     the observed outcome is backpropagated (see weighted loss).
 264     Selects the optimizer and, finally, it also initializes the graph.
 265     """
 266 
 267     print('Initializing model {}.'.format(self.name))
 268 
 269     # Build terms for the noise sigma estimation for each action.
 270     noise_sigma_mu = (self.build_mu_variable([1, self.n_out])
 271                       + self.sigma_transform.inverse(self.hparams.noise_sigma))
 272     noise_sigma_sigma = self.sigma_transform.forward(
 273         self.build_sigma_variable([1, self.n_out]))
 274 
 275     pre_noise_sigma = noise_sigma_mu + tf.random_normal(
 276         [1, self.n_out]) * noise_sigma_sigma
 277     self.noise_sigma = self.sigma_transform.forward(pre_noise_sigma)
 278 
 279     # Build network
 280     input_x = self.x
 281     n_in = self.n_in
 282     self.total_layers = len(self.layers) + 1
 283     if self.layers[0] == 0:
 284       self.total_layers = 1
 285 
 286     for l_number, n_nodes in enumerate(self.layers):
 287       if n_nodes > 0:
 288         h = self.build_layer(input_x, [n_in, n_nodes], l_number)
 289         input_x = h
 290         n_in = n_nodes
 291         self.num_w += n_in * n_nodes
 292         self.num_b += n_nodes
 293 
 294     self.y_pred = self.build_layer(input_x, [n_in, self.n_out],
 295                                    self.total_layers - 1,
 296                                    activation_fn=lambda x: x)
 297 
 298     # Compute energy function based on sampled nn's
 299     log_coeff = self.data_size / (self.batch_size * self.alpha)
 300     log_ratio = log_coeff * self.log_alpha_likelihood_ratio(activation_fn)
 301     logzprior = self.log_z_prior()
 302     logzq = self.log_z_q()
 303     energy = logzprior - logzq - log_ratio
 304 
 305     self.loss = energy
 306     self.global_step = tf.train.get_or_create_global_step()
 307     self.train_op = tf.train.AdamOptimizer(self.hparams.initial_lr).minimize(
 308         self.loss, global_step=self.global_step)
 309 
 310     # Useful for debugging
 311     sq_loss = tf.squared_difference(self.y_pred, self.y)
 312     weighted_sq_loss = self.weights * sq_loss
 313     self.cost = tf.reduce_sum(weighted_sq_loss) / self.batch_size
 314 
 315     # Create tensorboard metrics
 316     self.create_summaries()
 317     self.summary_writer = tf.summary.FileWriter('{}/graph_{}'.format(
 318         FLAGS.logdir, self.name), self.sess.graph)
 319 
 320   def create_summaries(self):
 321     tf.summary.scalar('loss', self.loss)
 322     tf.summary.scalar('cost', self.cost)
 323     self.summary_op = tf.summary.merge_all()
 324 
 325   def assign_lr(self):
 326     """Resets the learning rate in dynamic schedules for subsequent trainings.
 327 
 328     In bandits settings, we do expand our dataset over time. Then, we need to
 329     re-train the network with the new data. Those algorithms that do not keep
 330     the step constant, can reset it at the start of each training process.
 331     """
 332 
 333     decay_steps = 1
 334     if self.hparams.activate_decay:
 335       current_gs = self.sess.run(self.global_step)
 336       with self.graph.as_default():
 337         self.lr = tf.train.inverse_time_decay(self.hparams.initial_lr,
 338                                               self.global_step - current_gs,
 339                                               decay_steps,
 340                                               self.hparams.lr_decay_rate)
 341 
 342   def train(self, data, num_steps):
 343     """Trains the BNN for num_steps, using the data in 'data'.
 344 
 345     Args:
 346       data: ContextualDataset object that provides the data.
 347       num_steps: Number of minibatches to train the network for.
 348     """
 349 
 350     if self.times_trained < self.cleared_times_trained:
 351       num_steps = int(self.training_schedule[self.times_trained])
 352     self.times_trained += 1
 353 
 354     if self.verbose:
 355       print('Training {} for {} steps...'.format(self.name, num_steps))
 356 
 357     with self.graph.as_default():
 358 
 359       for step in range(num_steps):
 360         x, y, w = data.get_batch_with_weights(self.hparams.batch_size)
 361         _, summary, global_step, loss = self.sess.run(
 362             [self.train_op, self.summary_op, self.global_step, self.loss],
 363             feed_dict={self.x: x, self.y: y, self.weights: w,
 364                        self.data_size: data.num_points()})
 365 
 366         weights_l = self.sess.run(self.weights_std[0])
 367         self.h_max_var.append(np.max(weights_l))
 368 
 369         if step % self.freq_summary == 0:
 370           if self.show_training:
 371             print('step: {}, loss: {}'.format(step, loss))
 372             sys.stdout.flush()
 373           self.summary_writer.add_summary(summary, global_step)
