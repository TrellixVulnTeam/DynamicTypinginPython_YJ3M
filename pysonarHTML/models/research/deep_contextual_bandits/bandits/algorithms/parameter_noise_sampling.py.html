  35 class ParameterNoiseSampling(BanditAlgorithm):
  36   """Parameter Noise Sampling algorithm based on adding noise to net params.
  37 
  38   Described in https://arxiv.org/abs/1706.01905
  39   """
  40 
  41   def __init__(self, name, hparams):
  42     """Creates the algorithm, and sets up the adaptive Gaussian noise."""
  43 
  44     self.name = name
  45     self.hparams = hparams
  46     self.verbose = getattr(self.hparams, 'verbose', True)
  47     self.noise_std = getattr(self.hparams, 'noise_std', 0.005)
  48     self.eps = getattr(self.hparams, 'eps', 0.05)
  49     self.d_samples = getattr(self.hparams, 'd_samples', 300)
  50     self.optimizer = getattr(self.hparams, 'optimizer', 'RMS')
  51 
  52     # keep track of noise heuristic statistics
  53     self.std_h = [self.noise_std]
  54     self.eps_h = [self.eps]
  55     self.kl_h = []
  56     self.t = 0
  57 
  58     self.freq_update = hparams.training_freq
  59     self.num_epochs = hparams.training_epochs
  60 
  61     self.data_h = ContextualDataset(hparams.context_dim, hparams.num_actions,
  62                                     hparams.buffer_s)
  63     self.bnn = NeuralBanditModel(self.optimizer, hparams, '{}-bnn'.format(name))
  64 
  65     with self.bnn.graph.as_default():
  66 
  67       # noise-injection std placeholder
  68       self.bnn.noise_std_ph = tf.placeholder(tf.float32, shape=())
  69 
  70       # create noise corruption op; adds noise to all weights
  71       tvars = tf.trainable_variables()
  72       self.bnn.noisy_grads = [
  73           tf.random_normal(v.get_shape(), 0, self.bnn.noise_std_ph)
  74           for v in tvars
  75       ]
  76 
  77       # add noise to all params, then compute prediction, then subtract.
  78       with tf.control_dependencies(self.bnn.noisy_grads):
  79         self.bnn.noise_add_ops = [
  80             tvars[i].assign_add(n) for i, n in enumerate(self.bnn.noisy_grads)
  81         ]
  82         with tf.control_dependencies(self.bnn.noise_add_ops):
  83           # we force the prediction for 'y' to be recomputed after adding noise
  84           self.bnn.noisy_nn, self.bnn.noisy_pred_val = self.bnn.forward_pass()
  85 
  86           self.bnn.noisy_pred = tf.identity(self.bnn.noisy_pred_val)
  87           with tf.control_dependencies([tf.identity(self.bnn.noisy_pred)]):
  88             self.bnn.noise_sub_ops = [
  89                 tvars[i].assign_add(-n)
  90                 for i, n in enumerate(self.bnn.noisy_grads)
  91             ]
  92 
  93   def action(self, context):
  94     """Selects action based on Thompson Sampling *after* adding noise."""
  95 
  96     if self.t < self.hparams.num_actions * self.hparams.initial_pulls:
  97       # round robin until each action has been taken "initial_pulls" times
  98       return self.t % self.hparams.num_actions
  99 
 100     with self.bnn.graph.as_default():
 101       # run noise prediction op to choose action, and subtract noise op after.
 102       c = context.reshape((1, self.hparams.context_dim))
 103       output, _ = self.bnn.sess.run(
 104           [self.bnn.noisy_pred, self.bnn.noise_sub_ops],
 105           feed_dict={self.bnn.x: c,
 106                      self.bnn.noise_std_ph: self.noise_std})
 107       return np.argmax(output)
 108 
 109   def update(self, context, action, reward):
 110     """Updates the data buffer, and re-trains the BNN and noise level."""
 111 
 112     self.t += 1
 113     self.data_h.add(context, action, reward)
 114 
 115     if self.t % self.freq_update == 0:
 116       self.bnn.train(self.data_h, self.num_epochs)
 117       self.update_noise()
 118 
 119   def update_noise(self):
 120     """Increase noise if distance btw original and corrupted distrib small."""
 121 
 122     kl = self.compute_distance()
 123     delta = -np.log1p(- self.eps + self.eps / self.hparams.num_actions)
 124 
 125     if kl < delta:
 126       self.noise_std *= 1.01
 127     else:
 128       self.noise_std /= 1.01
 129 
 130     self.eps *= 0.99
 131 
 132     if self.verbose:
 133       print('Update eps={} | kl={} | std={} | delta={} | increase={}.'.format(
 134           self.eps, kl, self.noise_std, delta, kl < delta))
 135 
 136     # store noise-injection statistics for inspection: std, KL, eps.
 137     self.std_h.append(self.noise_std)
 138     self.kl_h.append(kl)
 139     self.eps_h.append(self.eps)
 140 
 141   def compute_distance(self):
 142     """Computes empirical KL for original and corrupted output distributions."""
 143 
 144     random_inputs, _ = self.data_h.get_batch(self.d_samples)
 145     y_model = self.bnn.sess.run(
 146         self.bnn.y_pred,
 147         feed_dict={
 148             self.bnn.x: random_inputs,
 149             self.bnn.noise_std_ph: self.noise_std
 150         })
 151     y_noisy, _ = self.bnn.sess.run(
 152         [self.bnn.noisy_pred, self.bnn.noise_sub_ops],
 153         feed_dict={
 154             self.bnn.x: random_inputs,
 155             self.bnn.noise_std_ph: self.noise_std
 156         })
 157 
 158     if self.verbose:
 159       # display how often original & perturbed models propose different actions
 160       s = np.sum([np.argmax(y_model[i, :]) == np.argmax(y_noisy[i, :])
 161                   for i in range(y_model.shape[0])])
 162       print('{} | % of agreement btw original / corrupted actions: {}.'.format(
 163           self.name, s / self.d_samples))
 164 
 165     kl = self.compute_kl_with_logits(y_model, y_noisy)
 166     return kl
 167 
 168   def compute_kl_with_logits(self, logits1, logits2):
 169     """Computes KL from logits samples from two distributions."""
 170 
 171     def exp_times_diff(a, b):
 172       return np.multiply(np.exp(a), a - b)
 173 
 174     logsumexp1 = logsumexp(logits1, axis=1)
 175     logsumexp2 = logsumexp(logits2, axis=1)
 176     logsumexp_diff = logsumexp2 - logsumexp1
 177 
 178     exp_diff = exp_times_diff(logits1, logits2)
 179     exp_diff = np.sum(exp_diff, axis=1)
 180 
 181     inv_exp_sum = np.sum(np.exp(logits1), axis=1)
 182     term1 = np.divide(exp_diff, inv_exp_sum)
 183 
 184     kl = term1 + logsumexp_diff
 185     kl = np.maximum(kl, 0.0)
 186     kl = np.nan_to_num(kl)
 187     return np.mean(kl)
