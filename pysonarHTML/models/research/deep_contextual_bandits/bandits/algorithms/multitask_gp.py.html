  32 class MultitaskGP(BayesianNN):
  33   """Implements a Gaussian process with multi-task outputs.
  34 
  35   Optimizes the hyperparameters over the log marginal likelihood.
  36   Uses a Matern 3/2 + linear covariance and returns
  37   sampled predictions for test inputs.  The outputs are optionally
  38   correlated where the correlation structure is learned through latent
  39   embeddings of the tasks.
  40   """
  41 
  42   def __init__(self, hparams):
  43     self.name = "MultiTaskGP"
  44     self.hparams = hparams
  45 
  46     self.n_in = self.hparams.context_dim
  47     self.n_out = self.hparams.num_outputs
  48     self.keep_fixed_after_max_obs = self.hparams.keep_fixed_after_max_obs
  49 
  50     self._show_training = self.hparams.show_training
  51     self._freq_summary = self.hparams.freq_summary
  52 
  53     # Dimensionality of the latent task vectors
  54     self.task_latent_dim = self.hparams.task_latent_dim
  55 
  56     # Maximum number of observations to include
  57     self.max_num_points = self.hparams.max_num_points
  58 
  59     if self.hparams.learn_embeddings:
  60       self.learn_embeddings = self.hparams.learn_embeddings
  61     else:
  62       self.learn_embeddings = False
  63 
  64     # create the graph corresponding to the BNN instance
  65     self.graph = tf.Graph()
  66     with self.graph.as_default():
  67       # store a new session for the graph
  68       self.sess = tf.Session()
  69 
  70       with tf.variable_scope(self.name, reuse=tf.AUTO_REUSE):
  71         self.n = tf.placeholder(shape=[], dtype=tf.float64)
  72         self.x = tf.placeholder(shape=[None, self.n_in], dtype=tf.float64)
  73         self.x_in = tf.placeholder(shape=[None, self.n_in], dtype=tf.float64)
  74         self.y = tf.placeholder(shape=[None, self.n_out], dtype=tf.float64)
  75         self.weights = tf.placeholder(shape=[None, self.n_out],
  76                                       dtype=tf.float64)
  77 
  78         self.build_model()
  79       self.sess.run(tf.global_variables_initializer())
  80 
  81   def atleast_2d(self, x, dims):
  82     return tf.reshape(tf.expand_dims(x, axis=0), (-1, dims))
  83 
  84   def sq_dist(self, x, x2):
  85     a2 = tf.reduce_sum(tf.square(x), 1)
  86     b2 = tf.reduce_sum(tf.square(x2), 1)
  87     sqdists = tf.expand_dims(a2, 1) + b2 - 2.0 * tf.matmul(x, tf.transpose(x2))
  88     return sqdists
  89 
  90   # Covariance between outputs
  91   def task_cov(self, x, x2):
  92     """Squared Exponential Covariance Kernel over latent task embeddings."""
  93     # Index into latent task vectors
  94     x_vecs = tf.gather(self.task_vectors, tf.argmax(x, axis=1), axis=0)
  95     x2_vecs = tf.gather(self.task_vectors, tf.argmax(x2, axis=1), axis=0)
  96     r = self.sq_dist(self.atleast_2d(x_vecs, self.task_latent_dim),
  97                      self.atleast_2d(x2_vecs, self.task_latent_dim))
  98     return tf.exp(-r)
  99 
 100   def cov(self, x, x2):
 101     """Matern 3/2 + Linear Gaussian Process Covariance Function."""
 102     ls = tf.clip_by_value(self.length_scales, -5.0, 5.0)
 103     ls_lin = tf.clip_by_value(self.length_scales_lin, -5.0, 5.0)
 104     r = self.sq_dist(self.atleast_2d(x, self.n_in)/tf.nn.softplus(ls),
 105                      self.atleast_2d(x2, self.n_in)/tf.nn.softplus(ls))
 106     r = tf.clip_by_value(r, 0, 1e8)
 107 
 108     # Matern 3/2 Covariance
 109     matern = (1.0 + tf.sqrt(3.0*r + 1e-16)) * tf.exp(-tf.sqrt(3.0*r + 1e-16))
 110     # Linear Covariance
 111     lin = tf.matmul(x / tf.nn.softplus(ls_lin),
 112                     x2 / tf.nn.softplus(ls_lin), transpose_b=True)
 113     return (tf.nn.softplus(self.amplitude) * matern +
 114             tf.nn.softplus(self.amplitude_linear) * lin)

 116   def build_model(self):
 117     """Defines the GP model.
 118 
 119     The loss is computed for partial feedback settings (bandits), so only
 120     the observed outcome is backpropagated (see weighted loss).
 121     Selects the optimizer and, finally, it also initializes the graph.
 122     """
 123 
 124     logging.info("Initializing model %s.", self.name)
 125     self.global_step = tf.train.get_or_create_global_step()
 126 
 127     # Define state for the model (inputs, etc.)
 128     self.x_train = tf.get_variable(
 129         "training_data",
 130         initializer=tf.ones(
 131             [self.hparams.batch_size, self.n_in], dtype=tf.float64),
 132         validate_shape=False,
 133         trainable=False)
 134     self.y_train = tf.get_variable(
 135         "training_labels",
 136         initializer=tf.zeros([self.hparams.batch_size, 1], dtype=tf.float64),
 137         validate_shape=False,
 138         trainable=False)
 139     self.weights_train = tf.get_variable(
 140         "weights_train",
 141         initializer=tf.ones(
 142             [self.hparams.batch_size, self.n_out], dtype=tf.float64),
 143         validate_shape=False,
 144         trainable=False)
 145     self.input_op = tf.assign(self.x_train, self.x_in, validate_shape=False)
 146     self.input_w_op = tf.assign(
 147         self.weights_train, self.weights, validate_shape=False)
 148 
 149     self.input_std = tf.get_variable(
 150         "data_standard_deviation",
 151         initializer=tf.ones([1, self.n_out], dtype=tf.float64),
 152         dtype=tf.float64,
 153         trainable=False)
 154     self.input_mean = tf.get_variable(
 155         "data_mean",
 156         initializer=tf.zeros([1, self.n_out], dtype=tf.float64),
 157         dtype=tf.float64,
 158         trainable=True)
 159 
 160     # GP Hyperparameters
 161     self.noise = tf.get_variable(
 162         "noise", initializer=tf.cast(0.0, dtype=tf.float64))
 163     self.amplitude = tf.get_variable(
 164         "amplitude", initializer=tf.cast(1.0, dtype=tf.float64))
 165     self.amplitude_linear = tf.get_variable(
 166         "linear_amplitude", initializer=tf.cast(1.0, dtype=tf.float64))
 167     self.length_scales = tf.get_variable(
 168         "length_scales", initializer=tf.zeros([1, self.n_in], dtype=tf.float64))
 169     self.length_scales_lin = tf.get_variable(
 170         "length_scales_linear",
 171         initializer=tf.zeros([1, self.n_in], dtype=tf.float64))
 172 
 173     # Latent embeddings of the different outputs for task covariance
 174     self.task_vectors = tf.get_variable(
 175         "latent_task_vectors",
 176         initializer=tf.random_normal(
 177             [self.n_out, self.task_latent_dim], dtype=tf.float64))
 178 
 179     # Normalize outputs across each dimension
 180     # Since we have different numbers of observations across each task, we
 181     # normalize by their respective counts.
 182     index_counts = self.atleast_2d(tf.reduce_sum(self.weights, axis=0),
 183                                    self.n_out)
 184     index_counts = tf.where(index_counts > 0, index_counts,
 185                             tf.ones(tf.shape(index_counts), dtype=tf.float64))
 186     self.mean_op = tf.assign(self.input_mean,
 187                              tf.reduce_sum(self.y, axis=0) / index_counts)
 188     self.var_op = tf.assign(
 189         self.input_std, tf.sqrt(1e-4 + tf.reduce_sum(tf.square(
 190             self.y - tf.reduce_sum(self.y, axis=0) / index_counts), axis=0)
 191                                 / index_counts))
 192 
 193     with tf.control_dependencies([self.var_op]):
 194       y_normed = self.atleast_2d(
 195           (self.y - self.input_mean) / self.input_std, self.n_out)
 196       y_normed = self.atleast_2d(tf.boolean_mask(y_normed, self.weights > 0), 1)
 197     self.out_op = tf.assign(self.y_train, y_normed, validate_shape=False)
 198 
 199     # Observation noise
 200     alpha = tf.nn.softplus(self.noise) + 1e-6
 201 
 202     # Covariance
 203     with tf.control_dependencies([self.input_op, self.input_w_op, self.out_op]):
 204       self.self_cov = (self.cov(self.x_in, self.x_in) *
 205                        self.task_cov(self.weights, self.weights) +
 206                        tf.eye(tf.shape(self.x_in)[0], dtype=tf.float64) * alpha)
 207 
 208     self.chol = tf.cholesky(self.self_cov)
 209     self.kinv = tf.cholesky_solve(self.chol, tf.eye(tf.shape(self.x_in)[0],
 210                                                     dtype=tf.float64))
 211 
 212     self.input_inv = tf.Variable(
 213         tf.eye(self.hparams.batch_size, dtype=tf.float64),
 214         validate_shape=False,
 215         trainable=False)
 216     self.input_cov_op = tf.assign(self.input_inv, self.kinv,
 217                                   validate_shape=False)
 218 
 219     # Log determinant by taking the singular values along the diagonal
 220     # of self.chol
 221     with tf.control_dependencies([self.input_cov_op]):
 222       logdet = 2.0 * tf.reduce_sum(tf.log(tf.diag_part(self.chol) + 1e-16))
 223 
 224     # Log Marginal likelihood
 225     self.marginal_ll = -tf.reduce_sum(-0.5 * tf.matmul(
 226         tf.transpose(y_normed), tf.matmul(self.kinv, y_normed)) - 0.5 * logdet -
 227                                       0.5 * self.n * np.log(2 * np.pi))
 228 
 229     zero = tf.cast(0., dtype=tf.float64)
 230     one = tf.cast(1., dtype=tf.float64)
 231     standard_normal = tfd.Normal(loc=zero, scale=one)
 232 
 233     # Loss is marginal likelihood and priors
 234     self.loss = tf.reduce_sum(
 235         self.marginal_ll -
 236         (standard_normal.log_prob(self.amplitude) +
 237          standard_normal.log_prob(tf.exp(self.noise)) +
 238          standard_normal.log_prob(self.amplitude_linear) +
 239          tfd.Normal(loc=zero, scale=one * 10.).log_prob(
 240              self.task_vectors))
 241     )
 242 
 243     # Optimizer for hyperparameters
 244     optimizer = tf.train.AdamOptimizer(learning_rate=self.hparams.lr)
 245     vars_to_optimize = [
 246         self.amplitude, self.length_scales, self.length_scales_lin,
 247         self.amplitude_linear, self.noise, self.input_mean
 248     ]
 249 
 250     if self.learn_embeddings:
 251       vars_to_optimize.append(self.task_vectors)
 252     grads = optimizer.compute_gradients(self.loss, vars_to_optimize)
 253     self.train_op = optimizer.apply_gradients(grads,
 254                                               global_step=self.global_step)
 255 
 256     # Predictions for test data
 257     self.y_mean, self.y_pred = self.posterior_mean_and_sample(self.x)
 258 
 259     # create tensorboard metrics
 260     self.create_summaries()
 261     self.summary_writer = tf.summary.FileWriter("{}/graph_{}".format(
 262         FLAGS.logdir, self.name), self.sess.graph)
 263     self.check = tf.add_check_numerics_ops()
 264 
 265   def posterior_mean_and_sample(self, candidates):
 266     """Draw samples for test predictions.
 267 
 268     Given a Tensor of 'candidates' inputs, returns samples from the posterior
 269     and the posterior mean prediction for those inputs.
 270 
 271     Args:
 272       candidates: A (num-examples x num-dims) Tensor containing the inputs for
 273       which to return predictions.
 274     Returns:
 275       y_mean: The posterior mean prediction given these inputs
 276       y_sample: A sample from the posterior of the outputs given these inputs
 277     """
 278     # Cross-covariance for test predictions
 279     w = tf.identity(self.weights_train)
 280     inds = tf.squeeze(
 281         tf.reshape(
 282             tf.tile(
 283                 tf.reshape(tf.range(self.n_out), (self.n_out, 1)),
 284                 (1, tf.shape(candidates)[0])), (-1, 1)))
 285 
 286     cross_cov = self.cov(tf.tile(candidates, [self.n_out, 1]), self.x_train)
 287     cross_task_cov = self.task_cov(tf.one_hot(inds, self.n_out), w)
 288     cross_cov *= cross_task_cov
 289 
 290     # Test mean prediction
 291     y_mean = tf.matmul(cross_cov, tf.matmul(self.input_inv, self.y_train))
 292 
 293     # Test sample predictions
 294     # Note this can be done much more efficiently using Kronecker products
 295     # if all tasks are fully observed (which we won't assume)
 296     test_cov = (
 297         self.cov(tf.tile(candidates, [self.n_out, 1]),
 298                  tf.tile(candidates, [self.n_out, 1])) *
 299         self.task_cov(tf.one_hot(inds, self.n_out),
 300                       tf.one_hot(inds, self.n_out)) -
 301         tf.matmul(cross_cov,
 302                   tf.matmul(self.input_inv,
 303                             tf.transpose(cross_cov))))
 304 
 305     # Get the matrix square root through an SVD for drawing samples
 306     # This seems more numerically stable than the Cholesky
 307     s, _, v = tf.svd(test_cov, full_matrices=True)
 308     test_sqrt = tf.matmul(v, tf.matmul(tf.diag(s), tf.transpose(v)))
 309 
 310     y_sample = (
 311         tf.matmul(
 312             test_sqrt,
 313             tf.random_normal([tf.shape(test_sqrt)[0], 1], dtype=tf.float64)) +
 314         y_mean)
 315 
 316     y_sample = (
 317         tf.transpose(tf.reshape(y_sample,
 318                                 (self.n_out, -1))) * self.input_std +
 319         self.input_mean)
 320 
 321     return y_mean, y_sample

 323   def create_summaries(self):
 324     with self.graph.as_default():
 325       tf.summary.scalar("loss", self.loss)
 326       tf.summary.scalar("log_noise", self.noise)
 327       tf.summary.scalar("log_amp", self.amplitude)
 328       tf.summary.scalar("log_amp_lin", self.amplitude_linear)
 329       tf.summary.histogram("length_scales", self.length_scales)
 330       tf.summary.histogram("length_scales_lin", self.length_scales_lin)
 331       self.summary_op = tf.summary.merge_all()
 332 
 333   def train(self, data, num_steps):
 334     """Trains the GP for num_steps, using the data in 'data'.
 335 
 336     Args:
 337       data: ContextualDataset object that provides the data.
 338       num_steps: Number of minibatches to train the network for.
 339     """
 340 
 341     logging.info("Training %s for %d steps...", self.name, num_steps)
 342     for step in range(num_steps):
 343       numpts = min(data.num_points(None), self.max_num_points)
 344       if numpts >= self.max_num_points and self.keep_fixed_after_max_obs:
 345         x = data.contexts[:numpts, :]
 346         y = data.rewards[:numpts, :]
 347         weights = np.zeros((x.shape[0], self.n_out))
 348         for i, val in enumerate(data.actions[:numpts]):
 349           weights[i, val] = 1.0
 350       else:
 351         x, y, weights = data.get_batch_with_weights(numpts)
 352 
 353       ops = [
 354           self.global_step, self.summary_op, self.loss, self.noise,
 355           self.amplitude, self.amplitude_linear, self.length_scales,
 356           self.length_scales_lin, self.input_cov_op, self.input_op, self.var_op,
 357           self.input_w_op, self.out_op, self.train_op
 358       ]
 359 
 360       res = self.sess.run(ops,
 361                           feed_dict={self.x: x,
 362                                      self.x_in: x,
 363                                      self.y: y,
 364                                      self.weights: weights,
 365                                      self.n: numpts,
 366                                     })
 367 
 368       if step % self._freq_summary == 0:
 369         if self._show_training:
 370           logging.info("step: %d, loss: %g noise: %f amp: %f amp_lin: %f",
 371                        step, res[2], res[3], res[4], res[5])
 372       summary = res[1]
 373       global_step = res[0]
 374       self.summary_writer.add_summary(summary, global_step=global_step)
