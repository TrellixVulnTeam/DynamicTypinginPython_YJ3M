  16 """Contextual algorithm that keeps a full linear posterior for each arm."""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import numpy as np
  23 from scipy.stats import invgamma
  24 
  25 from bandits.core.bandit_algorithm import BanditAlgorithm
  26 from bandits.core.contextual_dataset import ContextualDataset
  27 
  28 
  29 class LinearFullPosteriorSampling(BanditAlgorithm):
  30   """Thompson Sampling with independent linear models and unknown noise var."""
  31 
  32   def __init__(self, name, hparams):
  33     """Initialize posterior distributions and hyperparameters.
  34 
  35     Assume a linear model for each action i: reward = context^T beta_i + noise
  36     Each beta_i has a Gaussian prior (lambda parameter), each sigma2_i (noise
  37     level) has an inverse Gamma prior (a0, b0 parameters). Mean, covariance,
  38     and precision matrices are initialized, and the ContextualDataset created.
  39 
  40     Args:
  41       name: Name of the algorithm.
  42       hparams: Hyper-parameters of the algorithm.
  43     """
  44 
  45     self.name = name
  46     self.hparams = hparams
  47 
  48     # Gaussian prior for each beta_i
  49     self._lambda_prior = self.hparams.lambda_prior
  50 
  51     self.mu = [
  52         np.zeros(self.hparams.context_dim + 1)
  53         for _ in range(self.hparams.num_actions)
  54     ]
  55 
  56     self.cov = [(1.0 / self.lambda_prior) * np.eye(self.hparams.context_dim + 1)
  57                 for _ in range(self.hparams.num_actions)]
  58 
  59     self.precision = [
  60         self.lambda_prior * np.eye(self.hparams.context_dim + 1)
  61         for _ in range(self.hparams.num_actions)
  62     ]
  63 
  64     # Inverse Gamma prior for each sigma2_i
  65     self._a0 = self.hparams.a0
  66     self._b0 = self.hparams.b0
  67 
  68     self.a = [self._a0 for _ in range(self.hparams.num_actions)]
  69     self.b = [self._b0 for _ in range(self.hparams.num_actions)]
  70 
  71     self.t = 0
  72     self.data_h = ContextualDataset(hparams.context_dim,
  73                                     hparams.num_actions,
  74                                     intercept=True)
  75 
  76   def action(self, context):
  77     """Samples beta's from posterior, and chooses best action accordingly.
  78 
  79     Args:
  80       context: Context for which the action need to be chosen.
  81 
  82     Returns:
  83       action: Selected action for the context.
  84     """
  85 
  86     # Round robin until each action has been selected "initial_pulls" times
  87     if self.t < self.hparams.num_actions * self.hparams.initial_pulls:
  88       return self.t % self.hparams.num_actions
  89 
  90     # Sample sigma2, and beta conditional on sigma2
  91     sigma2_s = [
  92         self.b[i] * invgamma.rvs(self.a[i])
  93         for i in range(self.hparams.num_actions)
  94     ]
  95 
  96     try:
  97       beta_s = [
  98           np.random.multivariate_normal(self.mu[i], sigma2_s[i] * self.cov[i])
  99           for i in range(self.hparams.num_actions)
 100       ]
 101     except np.linalg.LinAlgError as e:
 102       # Sampling could fail if covariance is not positive definite
 103       print('Exception when sampling from {}.'.format(self.name))
 104       print('Details: {} | {}.'.format(e.message, e.args))
 105       d = self.hparams.context_dim + 1
 106       beta_s = [
 107           np.random.multivariate_normal(np.zeros((d)), np.eye(d))
 108           for i in range(self.hparams.num_actions)
 109       ]
 110 
 111     # Compute sampled expected values, intercept is last component of beta
 112     vals = [
 113         np.dot(beta_s[i][:-1], context.T) + beta_s[i][-1]
 114         for i in range(self.hparams.num_actions)
 115     ]
 116 
 117     return np.argmax(vals)
 118 
 119   def update(self, context, action, reward):
 120     """Updates action posterior using the linear Bayesian regression formula.
 121 
 122     Args:
 123       context: Last observed context.
 124       action: Last observed action.
 125       reward: Last observed reward.
 126     """
 127 
 128     self.t += 1
 129     self.data_h.add(context, action, reward)
 130 
 131     # Update posterior of action with formulas: \beta | x,y ~ N(mu_q, cov_q)
 132     x, y = self.data_h.get_data(action)
 133 
 134     # The algorithm could be improved with sequential update formulas (cheaper)
 135     s = np.dot(x.T, x)
 136 
 137     # Some terms are removed as we assume prior mu_0 = 0.
 138     precision_a = s + self.lambda_prior * np.eye(self.hparams.context_dim + 1)
 139     cov_a = np.linalg.inv(precision_a)
 140     mu_a = np.dot(cov_a, np.dot(x.T, y))
 141 
 142     # Inverse Gamma posterior update
 143     a_post = self.a0 + x.shape[0] / 2.0
 144     b_upd = 0.5 * (np.dot(y.T, y) - np.dot(mu_a.T, np.dot(precision_a, mu_a)))
 145     b_post = self.b0 + b_upd
 146 
 147     # Store new posterior distributions
 148     self.mu[action] = mu_a
 149     self.cov[action] = cov_a
 150     self.precision[action] = precision_a
 151     self.a[action] = a_post
 152     self.b[action] = b_post
 153 
 154   @property
 155   def a0(self):
 156     return self._a0
 157 
 158   @property
 159   def b0(self):
 160     return self._b0
 161 
 162   @property
 163   def lambda_prior(self):
 164     return self._lambda_prior
