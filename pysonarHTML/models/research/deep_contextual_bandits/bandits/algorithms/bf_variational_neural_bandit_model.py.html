  39 def log_gaussian(x, mu, sigma, reduce_sum=True):
  40   """Returns log Gaussian pdf."""
  41   res = tfd.Normal(mu, sigma).log_prob(x)
  42   if reduce_sum:
  43     return tf.reduce_sum(res)
  44   else:
  45     return res

  48 def analytic_kl(mu_1, sigma_1, mu_2, sigma_2):
  49   """KL for two Gaussian distributions with diagonal covariance matrix."""
  50   kl = tfd.kl_divergence(tfd.MVNDiag(mu_1, sigma_1), tfd.MVNDiag(mu_2, sigma_2))
  51   return kl

  57   def __init__(self, hparams, name="BBBNN"):
  58 
  59     self.name = name
  60     self.hparams = hparams
  61 
  62     self.n_in = self.hparams.context_dim
  63     self.n_out = self.hparams.num_actions
  64     self.layers = self.hparams.layer_sizes
  65     self.init_scale = self.hparams.init_scale
  66     self.f_num_points = None
  67     if "f_num_points" in hparams:
  68       self.f_num_points = self.hparams.f_num_points
  69 
  70     self.cleared_times_trained = self.hparams.cleared_times_trained
  71     self.initial_training_steps = self.hparams.initial_training_steps
  72     self.training_schedule = np.linspace(self.initial_training_steps,
  73                                          self.hparams.training_epochs,
  74                                          self.cleared_times_trained)
  75     self.verbose = getattr(self.hparams, "verbose", True)
  76 
  77     self.weights_m = {}
  78     self.weights_std = {}
  79     self.biases_m = {}
  80     self.biases_std = {}
  81 
  82     self.times_trained = 0
  83 
  84     if self.hparams.use_sigma_exp_transform:
  85       self.sigma_transform = tf.exp
  86       self.inverse_sigma_transform = np.log
  87     else:
  88       self.sigma_transform = tf.nn.softplus
  89       self.inverse_sigma_transform = lambda y: y + np.log(1. - np.exp(-y))
  90 
  91     # Whether to use the local reparameterization trick to compute the loss.
  92     # See details in https://arxiv.org/abs/1506.02557
  93     self.use_local_reparameterization = True
  94 
  95     self.build_graph()

  97   def build_mu_variable(self, shape):
  98     """Returns a mean variable initialized as N(0, 0.05)."""
  99     return tf.Variable(tf.random_normal(shape, 0.0, 0.05))
 100 
 101   def build_sigma_variable(self, shape, init=-5.):
 102     """Returns a sigma variable initialized as N(init, 0.05)."""
 103     # Initialize sigma to be very small initially to encourage MAP opt first
 104     return tf.Variable(tf.random_normal(shape, init, 0.05))

 106   def build_layer(self, input_x, input_x_local, shape,
 107                   layer_id, activation_fn=tf.nn.relu):
 108     """Builds a variational layer, and computes KL term.
 109 
 110     Args:
 111       input_x: Input to the variational layer.
 112       input_x_local: Input when the local reparameterization trick was applied.
 113       shape: [number_inputs, number_outputs] for the layer.
 114       layer_id: Number of layer in the architecture.
 115       activation_fn: Activation function to apply.
 116 
 117     Returns:
 118       output_h: Output of the variational layer.
 119       output_h_local: Output when local reparameterization trick was applied.
 120       neg_kl: Negative KL term for the layer.
 121     """
 122 
 123     w_mu = self.build_mu_variable(shape)
 124     w_sigma = self.sigma_transform(self.build_sigma_variable(shape))
 125     w_noise = tf.random_normal(shape)
 126     w = w_mu + w_sigma * w_noise
 127 
 128     b_mu = self.build_mu_variable([1, shape[1]])
 129     b_sigma = self.sigma_transform(self.build_sigma_variable([1, shape[1]]))
 130     b = b_mu
 131 
 132     # Store means and stds
 133     self.weights_m[layer_id] = w_mu
 134     self.weights_std[layer_id] = w_sigma
 135     self.biases_m[layer_id] = b_mu
 136     self.biases_std[layer_id] = b_sigma
 137 
 138     # Create outputs
 139     output_h = activation_fn(tf.matmul(input_x, w) + b)
 140 
 141     if self.use_local_reparameterization:
 142       # Use analytic KL divergence wrt the prior
 143       neg_kl = -analytic_kl(w_mu, w_sigma,
 144                             0., tf.to_float(np.sqrt(2./shape[0])))
 145     else:
 146       # Create empirical KL loss terms
 147       log_p = log_gaussian(w, 0., tf.to_float(np.sqrt(2./shape[0])))
 148       log_q = log_gaussian(w, tf.stop_gradient(w_mu), tf.stop_gradient(w_sigma))
 149       neg_kl = log_p - log_q
 150 
 151     # Apply local reparameterization trick: sample activations pre nonlinearity
 152     m_h = tf.matmul(input_x_local, w_mu) + b
 153     v_h = tf.matmul(tf.square(input_x_local), tf.square(w_sigma))
 154     output_h_local = m_h + tf.sqrt(v_h + 1e-6) * tf.random_normal(tf.shape(v_h))
 155     output_h_local = activation_fn(output_h_local)
 156 
 157     return output_h, output_h_local, neg_kl

 159   def build_action_noise(self):
 160     """Defines a model for additive noise per action, and its KL term."""
 161 
 162     # Define mean and std variables (log-normal dist) for each action.
 163     noise_sigma_mu = (self.build_mu_variable([1, self.n_out])
 164                       + self.inverse_sigma_transform(self.hparams.noise_sigma))
 165     noise_sigma_sigma = self.sigma_transform(
 166         self.build_sigma_variable([1, self.n_out]))
 167 
 168     pre_noise_sigma = (noise_sigma_mu
 169                        + tf.random_normal([1, self.n_out]) * noise_sigma_sigma)
 170     self.noise_sigma = self.sigma_transform(pre_noise_sigma)
 171 
 172     # Compute KL for additive noise sigma terms.
 173     if getattr(self.hparams, "infer_noise_sigma", False):
 174       neg_kl_term = log_gaussian(
 175           pre_noise_sigma,
 176           self.inverse_sigma_transform(self.hparams.noise_sigma),
 177           self.hparams.prior_sigma
 178       )
 179       neg_kl_term -= log_gaussian(pre_noise_sigma,
 180                                   noise_sigma_mu,
 181                                   noise_sigma_sigma)
 182     else:
 183       neg_kl_term = 0.
 184 
 185     return neg_kl_term

 187   def build_model(self, activation_fn=tf.nn.relu):
 188     """Defines the actual NN model with fully connected layers.
 189 
 190     The loss is computed for partial feedback settings (bandits), so only
 191     the observed outcome is backpropagated (see weighted loss).
 192     Selects the optimizer and, finally, it also initializes the graph.
 193 
 194     Args:
 195       activation_fn: the activation function used in the nn layers.
 196     """
 197 
 198     def weight_prior(dtype, shape, c, d, e):
 199       del c, d, e
 200       return tfd.Independent(
 201           tfd.Normal(loc=tf.zeros(shape, dtype),
 202                      scale=tf.to_float(np.sqrt(2) / shape[0])),
 203           reinterpreted_batch_ndims=tf.size(shape))
 204 
 205     if self.verbose:
 206       print("Initializing model {}.".format(self.name))
 207 
 208     # Compute model additive noise for each action with log-normal distribution
 209     neg_kl_term = self.build_action_noise()
 210 
 211     # Build variational network using self.x as input.
 212     input_x = self.x
 213 
 214     # Create Keras model using DenseLocalReparameterization (prior N(0, 1)).
 215     model_layers = [
 216         tfl.DenseLocalReparameterization(
 217             n_nodes,
 218             activation=tf.nn.relu,
 219             kernel_prior_fn=weight_prior
 220         )
 221         for n_nodes in self.layers if n_nodes > 0
 222     ]
 223 
 224     output_layer = tfl.DenseLocalReparameterization(
 225         self.n_out,
 226         activation=lambda x: x,
 227         kernel_prior_fn=weight_prior
 228     )
 229     model_layers.append(output_layer)
 230 
 231     model = tf.keras.Sequential(model_layers)
 232     self.y_pred = model(input_x)
 233 
 234     # Compute KL term
 235     neg_kl_term -= tf.add_n(model.losses)
 236 
 237     # Compute log likelihood (with learned or fixed noise level)
 238     if getattr(self.hparams, "infer_noise_sigma", False):
 239       log_likelihood = log_gaussian(
 240           self.y, self.y_pred, self.noise_sigma, reduce_sum=False)
 241     else:
 242       log_likelihood = log_gaussian(
 243           self.y, self.y_pred, self.hparams.noise_sigma, reduce_sum=False)
 244 
 245     # Only take into account observed outcomes (bandits setting)
 246     batch_size = tf.to_float(tf.shape(self.x)[0])
 247     weighted_log_likelihood = tf.reduce_sum(
 248         log_likelihood * self.weights) / batch_size
 249 
 250     # The objective is 1/n * (\sum_i log_like_i - KL); neg_kl_term estimates -KL
 251     elbo = weighted_log_likelihood + (neg_kl_term / self.n)
 252 
 253     self.loss = -elbo
 254     self.global_step = tf.train.get_or_create_global_step()
 255     self.train_op = tf.train.AdamOptimizer(self.hparams.initial_lr).minimize(
 256         self.loss, global_step=self.global_step)
 257 
 258     # Create tensorboard metrics
 259     self.create_summaries()
 260     self.summary_writer = tf.summary.FileWriter(
 261         "{}/graph_{}".format(FLAGS.logdir, self.name), self.sess.graph)
 262 
 263   def build_graph(self):
 264     """Defines graph, session, placeholders, and model.
 265 
 266     Placeholders are: n (size of the dataset), x and y (context and observed
 267     reward for each action), and weights (one-hot encoding of selected action
 268     for each context, i.e., only possibly non-zero element in each y).
 269     """
 270 
 271     self.graph = tf.Graph()
 272     with self.graph.as_default():
 273 
 274       self.sess = tf.Session()
 275 
 276       self.n = tf.placeholder(shape=[], dtype=tf.float32)
 277 
 278       self.x = tf.placeholder(shape=[None, self.n_in], dtype=tf.float32)
 279       self.y = tf.placeholder(shape=[None, self.n_out], dtype=tf.float32)
 280       self.weights = tf.placeholder(shape=[None, self.n_out], dtype=tf.float32)
 281 
 282       self.build_model()
 283       self.sess.run(tf.global_variables_initializer())
 284 
 285   def create_summaries(self):
 286     """Defines summaries including mean loss, and global step."""
 287 
 288     with self.graph.as_default():
 289       with tf.name_scope(self.name + "_summaries"):
 290         tf.summary.scalar("loss", self.loss)
 291         tf.summary.scalar("global_step", self.global_step)
 292         self.summary_op = tf.summary.merge_all()
 293 
 294   def assign_lr(self):
 295     """Resets the learning rate in dynamic schedules for subsequent trainings.
 296 
 297     In bandits settings, we do expand our dataset over time. Then, we need to
 298     re-train the network with the new data. The algorithms that do not keep
 299     the step constant, can reset it at the start of each *training* process.
 300     """
 301 
 302     decay_steps = 1
 303     if self.hparams.activate_decay:
 304       current_gs = self.sess.run(self.global_step)
 305       with self.graph.as_default():
 306         self.lr = tf.train.inverse_time_decay(self.hparams.initial_lr,
 307                                               self.global_step - current_gs,
 308                                               decay_steps,
 309                                               self.hparams.lr_decay_rate)
 310 
 311   def train(self, data, num_steps):
 312     """Trains the BNN for num_steps, using the data in 'data'.
 313 
 314     Args:
 315       data: ContextualDataset object that provides the data.
 316       num_steps: Number of minibatches to train the network for.
 317 
 318     Returns:
 319       losses: Loss history during training.
 320     """
 321 
 322     if self.times_trained < self.cleared_times_trained:
 323       num_steps = int(self.training_schedule[self.times_trained])
 324     self.times_trained += 1
 325 
 326     losses = []
 327 
 328     with self.graph.as_default():
 329 
 330       if self.verbose:
 331         print("Training {} for {} steps...".format(self.name, num_steps))
 332 
 333       for step in range(num_steps):
 334         x, y, weights = data.get_batch_with_weights(self.hparams.batch_size)
 335         _, summary, global_step, loss = self.sess.run(
 336             [self.train_op, self.summary_op, self.global_step, self.loss],
 337             feed_dict={
 338                 self.x: x,
 339                 self.y: y,
 340                 self.weights: weights,
 341                 self.n: data.num_points(self.f_num_points),
 342             })
 343 
 344         losses.append(loss)
 345 
 346         if step % self.hparams.freq_summary == 0:
 347           if self.hparams.show_training:
 348             print("{} | step: {}, loss: {}".format(
 349                 self.name, global_step, loss))
 350           self.summary_writer.add_summary(summary, global_step)
 351 
 352     return losses
