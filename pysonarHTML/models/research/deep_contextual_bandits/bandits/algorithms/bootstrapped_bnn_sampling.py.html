  16 """Contextual algorithm based on boostrapping neural networks."""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import numpy as np
  23 
  24 from bandits.core.bandit_algorithm import BanditAlgorithm
  25 from bandits.core.contextual_dataset import ContextualDataset
  26 from bandits.algorithms.neural_bandit_model import NeuralBanditModel
  27 
  28 
  29 class BootstrappedBNNSampling(BanditAlgorithm):
  30   """Thompson Sampling algorithm based on training several neural networks."""
  31 
  32   def __init__(self, name, hparams, optimizer='RMS'):
  33     """Creates a BootstrappedSGDSampling object based on a specific optimizer.
  34 
  35       hparams.q: Number of models that are independently trained.
  36       hparams.p: Prob of independently including each datapoint in each model.
  37 
  38     Args:
  39       name: Name given to the instance.
  40       hparams: Hyperparameters for each individual model.
  41       optimizer: Neural network optimization algorithm.
  42     """
  43 
  44     self.name = name
  45     self.hparams = hparams
  46     self.optimizer_n = optimizer
  47 
  48     self.training_freq = hparams.training_freq
  49     self.training_epochs = hparams.training_epochs
  50     self.t = 0
  51 
  52     self.q = hparams.q
  53     self.p = hparams.p
  54 
  55     self.datasets = [
  56         ContextualDataset(hparams.context_dim,
  57                           hparams.num_actions,
  58                           hparams.buffer_s)
  59         for _ in range(self.q)
  60     ]
  61 
  62     self.bnn_boot = [
  63         NeuralBanditModel(optimizer, hparams, '{}-{}-bnn'.format(name, i))
  64         for i in range(self.q)
  65     ]
  66 
  67   def action(self, context):
  68     """Selects action for context based on Thompson Sampling using one BNN."""
  69 
  70     if self.t < self.hparams.num_actions * self.hparams.initial_pulls:
  71       # round robin until each action has been taken "initial_pulls" times
  72       return self.t % self.hparams.num_actions
  73 
  74     # choose model uniformly at random
  75     model_index = np.random.randint(self.q)
  76 
  77     with self.bnn_boot[model_index].graph.as_default():
  78       c = context.reshape((1, self.hparams.context_dim))
  79       output = self.bnn_boot[model_index].sess.run(
  80           self.bnn_boot[model_index].y_pred,
  81           feed_dict={self.bnn_boot[model_index].x: c})
  82       return np.argmax(output)
  83 
  84   def update(self, context, action, reward):
  85     """Updates the data buffer, and re-trains the BNN every self.freq_update."""
  86 
  87     self.t += 1
  88     for i in range(self.q):
  89       # include the data point with probability p independently in each dataset
  90       if np.random.random() < self.p or self.t < 2:
  91         self.datasets[i].add(context, action, reward)
  92 
  93     if self.t % self.training_freq == 0:
  94       # update all the models:
  95       for i in range(self.q):
  96         if self.hparams.reset_lr:
  97           self.bnn_boot[i].assign_lr()
  98         self.bnn_boot[i].train(self.datasets[i], self.training_epochs)
