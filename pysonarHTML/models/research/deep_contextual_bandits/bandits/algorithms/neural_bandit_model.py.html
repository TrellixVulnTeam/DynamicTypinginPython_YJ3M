  16 """Define a family of neural network architectures for bandits.
  17 
  18 The network accepts different type of optimizers that could lead to different
  19 approximations of the posterior distribution or simply to point estimates.
  20 """
  21 
  22 from __future__ import absolute_import
  23 from __future__ import division
  24 from __future__ import print_function
  25 
  26 import tensorflow as tf
  27 
  28 from absl import flags
  29 from bandits.core.bayesian_nn import BayesianNN
  30 
  31 FLAGS = flags.FLAGS
  32 
  33 
  34 class NeuralBanditModel(BayesianNN):
  35   """Implements a neural network for bandit problems."""
  36 
  37   def __init__(self, optimizer, hparams, name):
  38     """Saves hyper-params and builds the Tensorflow graph."""
  39 
  40     self.opt_name = optimizer
  41     self.name = name
  42     self.hparams = hparams
  43     self.verbose = getattr(self.hparams, "verbose", True)
  44     self.times_trained = 0
  45     self.build_model()
  46 
  47   def build_layer(self, x, num_units):
  48     """Builds a layer with input x; dropout and layer norm if specified."""
  49 
  50     init_s = self.hparams.init_scale
  51 
  52     layer_n = getattr(self.hparams, "layer_norm", False)
  53     dropout = getattr(self.hparams, "use_dropout", False)
  54 
  55     nn = tf.contrib.layers.fully_connected(
  56         x,
  57         num_units,
  58         activation_fn=self.hparams.activation,
  59         normalizer_fn=None if not layer_n else tf.contrib.layers.layer_norm,
  60         normalizer_params={},
  61         weights_initializer=tf.random_uniform_initializer(-init_s, init_s)
  62     )
  63 
  64     if dropout:
  65       nn = tf.nn.dropout(nn, self.hparams.keep_prob)
  66 
  67     return nn
  68 
  69   def forward_pass(self):
  70 
  71     init_s = self.hparams.init_scale
  72 
  73     scope_name = "prediction_{}".format(self.name)
  74     with tf.variable_scope(scope_name, reuse=tf.AUTO_REUSE):
  75       nn = self.x
  76       for num_units in self.hparams.layer_sizes:
  77         if num_units > 0:
  78           nn = self.build_layer(nn, num_units)
  79 
  80       y_pred = tf.layers.dense(
  81           nn,
  82           self.hparams.num_actions,
  83           kernel_initializer=tf.random_uniform_initializer(-init_s, init_s))
  84 
  85     return nn, y_pred
  86 
  87   def build_model(self):
  88     """Defines the actual NN model with fully connected layers.
  89 
  90     The loss is computed for partial feedback settings (bandits), so only
  91     the observed outcome is backpropagated (see weighted loss).
  92     Selects the optimizer and, finally, it also initializes the graph.
  93     """
  94 
  95     # create and store the graph corresponding to the BNN instance
  96     self.graph = tf.Graph()
  97 
  98     with self.graph.as_default():
  99 
 100       # create and store a new session for the graph
 101       self.sess = tf.Session()
 102 
 103       with tf.name_scope(self.name):
 104 
 105         self.global_step = tf.train.get_or_create_global_step()
 106 
 107         # context
 108         self.x = tf.placeholder(
 109             shape=[None, self.hparams.context_dim],
 110             dtype=tf.float32,
 111             name="{}_x".format(self.name))
 112 
 113         # reward vector
 114         self.y = tf.placeholder(
 115             shape=[None, self.hparams.num_actions],
 116             dtype=tf.float32,
 117             name="{}_y".format(self.name))
 118 
 119         # weights (1 for selected action, 0 otherwise)
 120         self.weights = tf.placeholder(
 121             shape=[None, self.hparams.num_actions],
 122             dtype=tf.float32,
 123             name="{}_w".format(self.name))
 124 
 125         # with tf.variable_scope("prediction_{}".format(self.name)):
 126         self.nn, self.y_pred = self.forward_pass()
 127         self.loss = tf.squared_difference(self.y_pred, self.y)
 128         self.weighted_loss = tf.multiply(self.weights, self.loss)
 129         self.cost = tf.reduce_sum(self.weighted_loss) / self.hparams.batch_size
 130 
 131         if self.hparams.activate_decay:
 132           self.lr = tf.train.inverse_time_decay(
 133               self.hparams.initial_lr, self.global_step,
 134               1, self.hparams.lr_decay_rate)
 135         else:
 136           self.lr = tf.Variable(self.hparams.initial_lr, trainable=False)
 137 
 138         # create tensorboard metrics
 139         self.create_summaries()
 140         self.summary_writer = tf.summary.FileWriter(
 141             "{}/graph_{}".format(FLAGS.logdir, self.name), self.sess.graph)
 142 
 143         tvars = tf.trainable_variables()
 144         grads, _ = tf.clip_by_global_norm(
 145             tf.gradients(self.cost, tvars), self.hparams.max_grad_norm)
 146 
 147         self.optimizer = self.select_optimizer()
 148 
 149         self.train_op = self.optimizer.apply_gradients(
 150             zip(grads, tvars), global_step=self.global_step)
 151 
 152         self.init = tf.global_variables_initializer()
 153 
 154         self.initialize_graph()
 155 
 156   def initialize_graph(self):
 157     """Initializes all variables."""
 158 
 159     with self.graph.as_default():
 160       if self.verbose:
 161         print("Initializing model {}.".format(self.name))
 162       self.sess.run(self.init)
 163 
 164   def assign_lr(self):
 165     """Resets the learning rate in dynamic schedules for subsequent trainings.
 166 
 167     In bandits settings, we do expand our dataset over time. Then, we need to
 168     re-train the network with the new data. The algorithms that do not keep
 169     the step constant, can reset it at the start of each *training* process.
 170     """
 171 
 172     decay_steps = 1
 173     if self.hparams.activate_decay:
 174       current_gs = self.sess.run(self.global_step)
 175       with self.graph.as_default():
 176         self.lr = tf.train.inverse_time_decay(self.hparams.initial_lr,
 177                                               self.global_step - current_gs,
 178                                               decay_steps,
 179                                               self.hparams.lr_decay_rate)
 180 
 181   def select_optimizer(self):
 182     """Selects optimizer. To be extended (SGLD, KFAC, etc)."""
 183     return tf.train.RMSPropOptimizer(self.lr)
 184 
 185   def create_summaries(self):
 186     """Defines summaries including mean loss, learning rate, and global step."""
 187 
 188     with self.graph.as_default():
 189       with tf.name_scope(self.name + "_summaries"):
 190         tf.summary.scalar("cost", self.cost)
 191         tf.summary.scalar("lr", self.lr)
 192         tf.summary.scalar("global_step", self.global_step)
 193         self.summary_op = tf.summary.merge_all()
 194 
 195   def train(self, data, num_steps):
 196     """Trains the network for num_steps, using the provided data.
 197 
 198     Args:
 199       data: ContextualDataset object that provides the data.
 200       num_steps: Number of minibatches to train the network for.
 201     """
 202 
 203     if self.verbose:
 204       print("Training {} for {} steps...".format(self.name, num_steps))
 205 
 206     with self.graph.as_default():
 207 
 208       for step in range(num_steps):
 209         x, y, w = data.get_batch_with_weights(self.hparams.batch_size)
 210         _, cost, summary, lr = self.sess.run(
 211             [self.train_op, self.cost, self.summary_op, self.lr],
 212             feed_dict={self.x: x, self.y: y, self.weights: w})
 213 
 214         if step % self.hparams.freq_summary == 0:
 215           if self.hparams.show_training:
 216             print("{} | step: {}, lr: {}, loss: {}".format(
 217                 self.name, step, lr, cost))
 218           self.summary_writer.add_summary(summary, step)
 219 
 220       self.times_trained += 1
