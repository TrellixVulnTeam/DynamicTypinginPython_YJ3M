  16 """Thompson Sampling with linear posterior over a learnt deep representation."""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import numpy as np
  23 from scipy.stats import invgamma
  24 
  25 from bandits.core.bandit_algorithm import BanditAlgorithm
  26 from bandits.core.contextual_dataset import ContextualDataset
  27 from bandits.algorithms.neural_bandit_model import NeuralBanditModel

  30 class NeuralLinearPosteriorSampling(BanditAlgorithm):
  31   """Full Bayesian linear regression on the last layer of a deep neural net."""
  32 
  33   def __init__(self, name, hparams, optimizer='RMS'):
  34 
  35     self.name = name
  36     self.hparams = hparams
  37     self.latent_dim = self.hparams.layer_sizes[-1]
  38 
  39     # Gaussian prior for each beta_i
  40     self._lambda_prior = self.hparams.lambda_prior
  41 
  42     self.mu = [
  43         np.zeros(self.latent_dim)
  44         for _ in range(self.hparams.num_actions)
  45     ]
  46 
  47     self.cov = [(1.0 / self.lambda_prior) * np.eye(self.latent_dim)
  48                 for _ in range(self.hparams.num_actions)]
  49 
  50     self.precision = [
  51         self.lambda_prior * np.eye(self.latent_dim)
  52         for _ in range(self.hparams.num_actions)
  53     ]
  54 
  55     # Inverse Gamma prior for each sigma2_i
  56     self._a0 = self.hparams.a0
  57     self._b0 = self.hparams.b0
  58 
  59     self.a = [self._a0 for _ in range(self.hparams.num_actions)]
  60     self.b = [self._b0 for _ in range(self.hparams.num_actions)]
  61 
  62     # Regression and NN Update Frequency
  63     self.update_freq_lr = hparams.training_freq
  64     self.update_freq_nn = hparams.training_freq_network
  65 
  66     self.t = 0
  67     self.optimizer_n = optimizer
  68 
  69     self.num_epochs = hparams.training_epochs
  70     self.data_h = ContextualDataset(hparams.context_dim,
  71                                     hparams.num_actions,
  72                                     intercept=False)
  73     self.latent_h = ContextualDataset(self.latent_dim,
  74                                       hparams.num_actions,
  75                                       intercept=False)
  76     self.bnn = NeuralBanditModel(optimizer, hparams, '{}-bnn'.format(name))
  77 
  78   def action(self, context):
  79     """Samples beta's from posterior, and chooses best action accordingly."""
  80 
  81     # Round robin until each action has been selected "initial_pulls" times
  82     if self.t < self.hparams.num_actions * self.hparams.initial_pulls:
  83       return self.t % self.hparams.num_actions
  84 
  85     # Sample sigma2, and beta conditional on sigma2
  86     sigma2_s = [
  87         self.b[i] * invgamma.rvs(self.a[i])
  88         for i in range(self.hparams.num_actions)
  89     ]
  90 
  91     try:
  92       beta_s = [
  93           np.random.multivariate_normal(self.mu[i], sigma2_s[i] * self.cov[i])
  94           for i in range(self.hparams.num_actions)
  95       ]
  96     except np.linalg.LinAlgError as e:
  97       # Sampling could fail if covariance is not positive definite
  98       print('Exception when sampling for {}.'.format(self.name))
  99       print('Details: {} | {}.'.format(e.message, e.args))
 100       d = self.latent_dim
 101       beta_s = [
 102           np.random.multivariate_normal(np.zeros((d)), np.eye(d))
 103           for i in range(self.hparams.num_actions)
 104       ]
 105 
 106     # Compute last-layer representation for the current context
 107     with self.bnn.graph.as_default():
 108       c = context.reshape((1, self.hparams.context_dim))
 109       z_context = self.bnn.sess.run(self.bnn.nn, feed_dict={self.bnn.x: c})
 110 
 111     # Apply Thompson Sampling to last-layer representation
 112     vals = [
 113         np.dot(beta_s[i], z_context.T) for i in range(self.hparams.num_actions)
 114     ]
 115     return np.argmax(vals)
 116 
 117   def update(self, context, action, reward):
 118     """Updates the posterior using linear bayesian regression formula."""
 119 
 120     self.t += 1
 121     self.data_h.add(context, action, reward)
 122     c = context.reshape((1, self.hparams.context_dim))
 123     z_context = self.bnn.sess.run(self.bnn.nn, feed_dict={self.bnn.x: c})
 124     self.latent_h.add(z_context, action, reward)
 125 
 126     # Retrain the network on the original data (data_h)
 127     if self.t % self.update_freq_nn == 0:
 128 
 129       if self.hparams.reset_lr:
 130         self.bnn.assign_lr()
 131       self.bnn.train(self.data_h, self.num_epochs)
 132 
 133       # Update the latent representation of every datapoint collected so far
 134       new_z = self.bnn.sess.run(self.bnn.nn,
 135                                 feed_dict={self.bnn.x: self.data_h.contexts})
 136       self.latent_h.replace_data(contexts=new_z)
 137 
 138     # Update the Bayesian Linear Regression
 139     if self.t % self.update_freq_lr == 0:
 140 
 141       # Find all the actions to update
 142       actions_to_update = self.latent_h.actions[:-self.update_freq_lr]
 143 
 144       for action_v in np.unique(actions_to_update):
 145 
 146         # Update action posterior with formulas: \beta | z,y ~ N(mu_q, cov_q)
 147         z, y = self.latent_h.get_data(action_v)
 148 
 149         # The algorithm could be improved with sequential formulas (cheaper)
 150         s = np.dot(z.T, z)
 151 
 152         # Some terms are removed as we assume prior mu_0 = 0.
 153         precision_a = s + self.lambda_prior * np.eye(self.latent_dim)
 154         cov_a = np.linalg.inv(precision_a)
 155         mu_a = np.dot(cov_a, np.dot(z.T, y))
 156 
 157         # Inverse Gamma posterior update
 158         a_post = self.a0 + z.shape[0] / 2.0
 159         b_upd = 0.5 * np.dot(y.T, y)
 160         b_upd -= 0.5 * np.dot(mu_a.T, np.dot(precision_a, mu_a))
 161         b_post = self.b0 + b_upd
 162 
 163         # Store new posterior distributions
 164         self.mu[action_v] = mu_a
 165         self.cov[action_v] = cov_a
 166         self.precision[action_v] = precision_a
 167         self.a[action_v] = a_post
 168         self.b[action_v] = b_post
 169 
 170   @property
 171   def a0(self):
 172     return self._a0
 173 
 174   @property
 175   def b0(self):
 176     return self._b0
 177 
 178   @property
 179   def lambda_prior(self):
 180     return self._lambda_prior
