  16 """Contextual bandit algorithm based on Thompson Sampling and a Bayesian NN."""
  17 
  18 from __future__ import absolute_import
  19 from __future__ import division
  20 from __future__ import print_function
  21 
  22 import numpy as np
  23 
  24 from bandits.core.bandit_algorithm import BanditAlgorithm
  25 from bandits.algorithms.bb_alpha_divergence_model import BBAlphaDivergence
  26 from bandits.algorithms.bf_variational_neural_bandit_model import BfVariationalNeuralBanditModel
  27 from bandits.core.contextual_dataset import ContextualDataset
  28 from bandits.algorithms.multitask_gp import MultitaskGP
  29 from bandits.algorithms.neural_bandit_model import NeuralBanditModel
  30 from bandits.algorithms.variational_neural_bandit_model import VariationalNeuralBanditModel
  31 
  32 
  33 class PosteriorBNNSampling(BanditAlgorithm):
  34   """Posterior Sampling algorithm based on a Bayesian neural network."""
  35 
  36   def __init__(self, name, hparams, bnn_model='RMSProp'):
  37     """Creates a PosteriorBNNSampling object based on a specific optimizer.
  38 
  39     The algorithm has two basic tools: an Approx BNN and a Contextual Dataset.
  40     The Bayesian Network keeps the posterior based on the optimizer iterations.
  41 
  42     Args:
  43       name: Name of the algorithm.
  44       hparams: Hyper-parameters of the algorithm.
  45       bnn_model: Type of BNN. By default RMSProp (point estimate).
  46     """
  47 
  48     self.name = name
  49     self.hparams = hparams
  50     self.optimizer_n = hparams.optimizer
  51 
  52     self.training_freq = hparams.training_freq
  53     self.training_epochs = hparams.training_epochs
  54     self.t = 0
  55     self.data_h = ContextualDataset(hparams.context_dim, hparams.num_actions,
  56                                     hparams.buffer_s)
  57 
  58     # to be extended with more BNNs (BB alpha-div, GPs, SGFS, constSGD...)
  59     bnn_name = '{}-bnn'.format(name)
  60     if bnn_model == 'Variational':
  61       self.bnn = VariationalNeuralBanditModel(hparams, bnn_name)
  62     elif bnn_model == 'AlphaDiv':
  63       self.bnn = BBAlphaDivergence(hparams, bnn_name)
  64     elif bnn_model == 'Variational_BF':
  65       self.bnn = BfVariationalNeuralBanditModel(hparams, bnn_name)
  66     elif bnn_model == 'GP':
  67       self.bnn = MultitaskGP(hparams)
  68     else:
  69       self.bnn = NeuralBanditModel(self.optimizer_n, hparams, bnn_name)
  70 
  71   def action(self, context):
  72     """Selects action for context based on Thompson Sampling using the BNN."""
  73 
  74     if self.t < self.hparams.num_actions * self.hparams.initial_pulls:
  75       # round robin until each action has been taken "initial_pulls" times
  76       return self.t % self.hparams.num_actions
  77 
  78     with self.bnn.graph.as_default():
  79       c = context.reshape((1, self.hparams.context_dim))
  80       output = self.bnn.sess.run(self.bnn.y_pred, feed_dict={self.bnn.x: c})
  81       return np.argmax(output)
  82 
  83   def update(self, context, action, reward):
  84     """Updates data buffer, and re-trains the BNN every training_freq steps."""
  85 
  86     self.t += 1
  87     self.data_h.add(context, action, reward)
  88 
  89     if self.t % self.training_freq == 0:
  90       if self.hparams.reset_lr:
  91         self.bnn.assign_lr()
  92       self.bnn.train(self.data_h, self.training_epochs)
