  17 """Computes Spearman's rho with respect to human judgements.
  18 
  19 Given a set of row (and potentially column) embeddings, this computes Spearman's
  20 rho between the rank ordering of predicted word similarity and human judgements.
  21 
  22 Usage:
  23 
  24   wordim.py --embeddings=<binvecs> --vocab=<vocab> eval1.tab eval2.tab ...
  25 
  26 Options:
  27 
  28   --embeddings=<filename>: the vectors to test
  29   --vocab=<filename>: the vocabulary file
  30 
  31 Evaluation files are assumed to be tab-separated files with exactly three
  32 columns.  The first two columns contain the words, and the third column contains
  33 the scored human judgement.
  34 
  35 """
  36 
  37 from __future__ import print_function
  38 import scipy.stats
  39 import sys
  40 from getopt import GetoptError, getopt
  41 
  42 from vecs import Vecs
  43 
  44 try:
  45   opts, args = getopt(sys.argv[1:], '', ['embeddings=', 'vocab='])
  46 except GetoptError as e:
  47   print(e, file=sys.stderr)
  48   sys.exit(2)
  49 
  50 opt_embeddings = None
  51 opt_vocab = None
  52 
  53 for o, a in opts:
  54   if o == '--embeddings':
  55     opt_embeddings = a
  56   if o == '--vocab':
  57     opt_vocab = a
  58 
  59 if not opt_vocab:
  60   print('please specify a vocabulary file with "--vocab"', file=sys.stderr)
  61   sys.exit(2)
  62 
  63 if not opt_embeddings:
  64   print('please specify the embeddings with "--embeddings"', file=sys.stderr)
  65   sys.exit(2)
  66 
  67 try:
  68   vecs = Vecs(opt_vocab, opt_embeddings)
  69 except IOError as e:
  70   print(e, file=sys.stderr)
  71   sys.exit(1)
  72 
  73 
  74 def evaluate(lines):
  75   acts, preds = [], []
  76 
  77   with open(filename, 'r') as lines:
  78     for line in lines:
  79       w1, w2, act = line.strip().split('\t')
  80       pred = vecs.similarity(w1, w2)
  81       if pred is None:
  82         continue
  83 
  84       acts.append(float(act))
  85       preds.append(pred)
  86 
  87   rho, _ = scipy.stats.spearmanr(acts, preds)
  88   return rho
  89 
  90 
  91 for filename in args:
  92   with open(filename, 'r') as lines:
  93     print('%0.3f %s' % (evaluate(lines), filename))
