  15 """Submatrix-wise Vector Embedding Learner.
  16 
  17 Implementation of SwiVel algorithm described at:
  18 http://arxiv.org/abs/1602.02215
  19 
  20 This program expects an input directory that contains the following files.
  21 
  22   row_vocab.txt, col_vocab.txt
  23 
  24     The row an column vocabulary files.  Each file should contain one token per
  25     line; these will be used to generate a tab-separate file containing the
  26     trained embeddings.
  27 
  28   row_sums.txt, col_sum.txt
  29 
  30     The matrix row and column marginal sums.  Each file should contain one
  31     decimal floating point number per line which corresponds to the marginal
  32     count of the matrix for that row or column.
  33 
  34   shards.recs
  35 
  36     A file containing the sub-matrix shards, stored as TFRecords.  Each shard is
  37     expected to be a serialzed tf.Example protocol buffer with the following
  38     properties:
  39 
  40       global_row: the global row indicies contained in the shard
  41       global_col: the global column indicies contained in the shard
  42       sparse_local_row, sparse_local_col, sparse_value: three parallel arrays
  43       that are a sparse representation of the submatrix counts.
  44 
  45 It will generate embeddings, training from the input directory for the specified
  46 number of epochs.  When complete, it will output the trained vectors to a
  47 tab-separated file that contains one line per embedding.  Row and column
  48 embeddings are stored in separate files.
  49 
  50 Swivel can be run "stand-alone" or "distributed".  The latter involves running
  51 at least one parameter server process, along with one or more worker processes.
  52 """

 117 class Model(object):
 118   """A Swivel model."""
 119 
 120   def __init__(self, input_base_path, hparams):
 121     """Creates a new Swivel model."""
 122     # Read vocab
 123     self.row_ix_to_word, self.row_word_to_ix = self._read_vocab(
 124         os.path.join(input_base_path, 'row_vocab.txt'))
 125     self.col_ix_to_word, self.col_word_to_ix = self._read_vocab(
 126         os.path.join(input_base_path, 'col_vocab.txt'))
 127 
 128     # Read marginals.
 129     row_sums = self._read_marginals_file(
 130         os.path.join(input_base_path, 'row_sums.txt'))
 131     col_sums = self._read_marginals_file(
 132         os.path.join(input_base_path, 'col_sums.txt'))
 133 
 134     # Construct input tensors.
 135     count_matrix_files = glob.glob(
 136         os.path.join(input_base_path, 'shard-*.pb'))
 137 
 138     global_rows, global_cols, counts = self._count_matrix_input(
 139         count_matrix_files, hparams.submatrix_rows, hparams.submatrix_cols)
 140 
 141     # Create embedding variables.
 142     sigma = 1.0 / np.sqrt(hparams.dim)
 143     self.row_embedding = tf.get_variable(
 144         'row_embedding',
 145         shape=[len(row_sums), hparams.dim],
 146         initializer=tf.random_normal_initializer(0, sigma),
 147         dtype=tf.float32)
 148     self.col_embedding = tf.get_variable(
 149         'col_embedding',
 150         shape=[len(col_sums), hparams.dim],
 151         initializer=tf.random_normal_initializer(0, sigma),
 152         dtype=tf.float32)
 153 
 154     matrix_log_sum = np.log(np.sum(row_sums) + 1)
 155     row_bias = tf.constant(
 156         [np.log(x + 1) for x in row_sums], dtype=tf.float32)
 157     col_bias = tf.constant(
 158         [np.log(x + 1) for x in col_sums], dtype=tf.float32)
 159 
 160     # Fetch embeddings.
 161     selected_rows = tf.nn.embedding_lookup(self.row_embedding, global_rows)
 162     selected_cols = tf.nn.embedding_lookup(self.col_embedding, global_cols)
 163 
 164     selected_row_bias = tf.gather(row_bias, global_rows)
 165     selected_col_bias = tf.gather(col_bias, global_cols)
 166 
 167     predictions = tf.matmul(selected_rows, selected_cols, transpose_b=True)
 168 
 169     # These binary masks separate zero from non-zero values.
 170     count_is_nonzero = tf.to_float(tf.cast(counts, tf.bool))
 171     count_is_zero = 1 - count_is_nonzero
 172 
 173     objectives = count_is_nonzero * tf.log(counts + 1e-30)
 174     objectives -= tf.reshape(selected_row_bias, [-1, 1])
 175     objectives -= selected_col_bias
 176     objectives += matrix_log_sum
 177 
 178     err = predictions - objectives
 179 
 180     # The confidence function scales the L2 loss based on the raw
 181     # co-occurrence count.
 182     l2_confidence = (hparams.confidence_base +
 183                      hparams.confidence_scale * tf.pow(
 184                          counts, hparams.confidence_exponent))
 185 
 186     loss_multiplier = 1 / np.sqrt(
 187         hparams.submatrix_rows * hparams.submatrix_cols)
 188 
 189     l2_loss = loss_multiplier * tf.reduce_sum(
 190         0.5 * l2_confidence * tf.square(err))
 191 
 192     sigmoid_loss = loss_multiplier * tf.reduce_sum(
 193         tf.nn.softplus(err) * count_is_zero)
 194 
 195     self.loss_op = l2_loss + sigmoid_loss
 196 
 197     if hparams.optimizer == 'adagrad':
 198       opt = tf.train.AdagradOptimizer(hparams.learning_rate)
 199     elif hparams.optimizer == 'rmsprop':
 200       opt = tf.train.RMSPropOptimizer(hparams.learning_rate, hparams.momentum)
 201     else:
 202       raise ValueError('unknown optimizer "%s"' % hparams.optimizer)
 203 
 204     self.global_step = tf.get_variable(
 205         'global_step', initializer=0, trainable=False)
 206 
 207     self.train_op = opt.minimize(self.loss_op, global_step=self.global_step)
 208 
 209     # One epoch trains each submatrix once.
 210     self.steps_per_epoch = (
 211         (len(row_sums) / hparams.submatrix_rows) *
 212         (len(col_sums) / hparams.submatrix_cols))
 213 
 214   def _read_vocab(self, filename):
 215     """Reads the vocabulary file."""
 216     with open(filename) as lines:
 217       ix_to_word = [line.strip() for line in lines]
 218       word_to_ix = {word: ix for ix, word in enumerate(ix_to_word)}
 219       return ix_to_word, word_to_ix
 220 
 221   def _read_marginals_file(self, filename):
 222     """Reads text file with one number per line to an array."""
 223     with open(filename) as lines:
 224       return [float(line.strip()) for line in lines]
 225 
 226   def _count_matrix_input(self, filenames, submatrix_rows, submatrix_cols):
 227     """Creates ops that read submatrix shards from disk."""
 228     random.shuffle(filenames)
 229     filename_queue = tf.train.string_input_producer(filenames)
 230     reader = tf.WholeFileReader()
 231     _, serialized_example = reader.read(filename_queue)
 232     features = tf.parse_single_example(
 233         serialized_example,
 234         features={
 235             'global_row': tf.FixedLenFeature([submatrix_rows], dtype=tf.int64),
 236             'global_col': tf.FixedLenFeature([submatrix_cols], dtype=tf.int64),
 237             'sparse_local_row': tf.VarLenFeature(dtype=tf.int64),
 238             'sparse_local_col': tf.VarLenFeature(dtype=tf.int64),
 239             'sparse_value': tf.VarLenFeature(dtype=tf.float32)
 240         })
 241 
 242     global_row = features['global_row']
 243     global_col = features['global_col']
 244 
 245     sparse_local_row = features['sparse_local_row'].values
 246     sparse_local_col = features['sparse_local_col'].values
 247     sparse_count = features['sparse_value'].values
 248 
 249     sparse_indices = tf.concat(
 250         axis=1, values=[tf.expand_dims(sparse_local_row, 1),
 251                         tf.expand_dims(sparse_local_col, 1)])
 252 
 253     count = tf.sparse_to_dense(sparse_indices, [submatrix_rows, submatrix_cols],
 254                                sparse_count)
 255 
 256     return global_row, global_col, count

 258   def wordsim_eval_op(self, filename):
 259     """Returns an op that runs an eval on a word similarity dataset.
 260 
 261     The eval dataset is assumed to be tab-separated, one scored word pair per
 262     line.  The resulting value is Spearman's rho of the human judgements with
 263     the cosine similarity of the word embeddings.
 264 
 265     Args:
 266       filename: the filename containing the word similarity data.
 267 
 268     Returns:
 269       An operator that will compute Spearman's rho of the current row
 270       embeddings.
 271     """
 272     with open(filename, 'r') as fh:
 273       tuples = (line.strip().split('\t') for line in fh.read().splitlines())
 274       word1s, word2s, sims = zip(*tuples)
 275       actuals = map(float, sims)
 276 
 277     v1s_t = tf.nn.embedding_lookup(
 278         self.row_embedding,
 279         [self.row_word_to_ix.get(w, 0) for w in word1s])
 280 
 281     v2s_t = tf.nn.embedding_lookup(
 282         self.row_embedding,
 283         [self.row_word_to_ix.get(w, 0) for w in word2s])
 284 
 285     # Compute the predicted word similarity as the cosine similarity between the
 286     # embedding vectors.
 287     preds_t = tf.reduce_sum(
 288         tf.nn.l2_normalize(v1s_t, dim=1) * tf.nn.l2_normalize(v2s_t, dim=1),
 289         axis=1)
 290 
 291     def _op(preds):
 292       rho, _ = scipy.stats.spearmanr(preds, actuals)
 293       return rho
 294 
 295     return tf.py_func(_op, [preds_t], tf.float64)

 297   def analogy_eval_op(self, filename, max_vocab_size=20000):
 298     """Returns an op that runs an eval on an analogy dataset.
 299 
 300     The eval dataset is assumed to be tab-separated, with four tokens per
 301     line. The first three tokens are query terms, the last is the expected
 302     answer. For each line (e.g., "man king woman queen"), the vectors
 303     corresponding to the query terms are added ("king - man + woman") to produce
 304     a query vector.  If the expected answer's vector is the nearest neighbor to
 305     the query vector (not counting any of the query vectors themselves), then
 306     the line is scored as correct.  The reported accuracy is the number of
 307     correct rows divided by the total number of rows.  Missing terms are
 308     replaced with an arbitrary vector and will almost certainly result in
 309     incorrect answers.
 310 
 311     Note that the results are approximate: for efficiency's sake, only the first
 312     `max_vocab_size` terms are included in the nearest neighbor search.
 313 
 314     Args:
 315       filename: the filename containing the analogy data.
 316       max_vocab_size: the maximum number of tokens to include in the nearest
 317         neighbor search. By default, 20000.
 318 
 319     Returns:
 320       The accuracy on the analogy task.
 321     """
 322     analogy_ixs = []
 323     with open(filename, 'r') as lines:
 324       for line in lines:
 325         parts = line.strip().split('\t')
 326         if len(parts) == 4:
 327           analogy_ixs.append([self.row_word_to_ix.get(w, 0) for w in parts])
 328 
 329     # man:king :: woman:queen => king - man + woman == queen
 330     ix1s, ix2s, ix3s, _ = zip(*analogy_ixs)
 331     v1s_t, v2s_t, v3s_t = (
 332         tf.nn.l2_normalize(
 333             tf.nn.embedding_lookup(self.row_embedding, ixs),
 334             dim=1)
 335         for ixs in (ix1s, ix2s, ix3s))
 336 
 337     preds_t = v2s_t - v1s_t + v3s_t
 338 
 339     # Compute the nearest neighbors as the cosine similarity.  We only consider
 340     # up to max_vocab_size to avoid a matmul that swamps the machine.
 341     sims_t = tf.matmul(
 342         preds_t,
 343         tf.nn.l2_normalize(self.row_embedding[:max_vocab_size], dim=1),
 344         transpose_b=True)
 345 
 346     # Take the four nearest neighbors, since the eval explicitly discards the
 347     # query terms.
 348     _, preds_ixs_t = tf.nn.top_k(sims_t, 4)
 349 
 350     def _op(preds_ixs):
 351       correct, total = 0, 0
 352       for pred_ixs, actual_ixs in itertools.izip(preds_ixs, analogy_ixs):
 353         pred_ixs = [ix for ix in pred_ixs if ix not in actual_ixs[:3]]
 354         correct += pred_ixs[0] == actual_ixs[3]
 355         total += 1
 356 
 357       return correct / total
 358 
 359     return tf.py_func(_op, [preds_ixs_t], tf.float64)
 360 
 361   def _write_tensor(self, vocab_path, output_path, session, embedding):
 362     """Writes tensor to output_path as tsv."""
 363     embeddings = session.run(embedding)
 364 
 365     with open(output_path, 'w') as out_f:
 366       with open(vocab_path) as vocab_f:
 367         for index, word in enumerate(vocab_f):
 368           word = word.strip()
 369           embedding = embeddings[index]
 370           print('\t'.join([word.strip()] + [str(x) for x in embedding]),
 371                 file=out_f)
 372 
 373   def write_embeddings(self, config, session):
 374     """Writes row and column embeddings disk."""
 375     self._write_tensor(
 376         os.path.join(config.input_base_path, 'row_vocab.txt'),
 377         os.path.join(config.output_base_path, 'row_embedding.tsv'),
 378         session, self.row_embedding)
 379 
 380     self._write_tensor(
 381         os.path.join(config.input_base_path, 'col_vocab.txt'),
 382         os.path.join(config.output_base_path, 'col_embedding.tsv'),
 383         session, self.col_embedding)

 386 def main(_):
 387   tf.logging.set_verbosity(tf.logging.INFO)
 388 
 389   # If we have ps_hosts, then we'll assume that this is going to be a
 390   # distributed training run.  Configure the cluster appropriately.  Otherwise,
 391   # we just do everything in-process.
 392   if FLAGS.ps_hosts:
 393     cluster = tf.train.ClusterSpec({
 394         'ps': FLAGS.ps_hosts.split(','),
 395         'worker': FLAGS.worker_hosts.split(','),
 396     })
 397 
 398     if FLAGS.job_name == 'ps':
 399       # Ignore the GPU if we're the parameter server. This let's the PS run on
 400       # the same machine as a worker.
 401       config = tf.ConfigProto(device_count={'GPU': 0})
 402     elif FLAGS.job_name == 'worker':
 403       config = tf.ConfigProto(gpu_options=tf.GPUOptions(
 404           visible_device_list='%d' % FLAGS.gpu_device,
 405           allow_growth=True))
 406     else:
 407       raise ValueError('unknown job name "%s"' % FLAGS.job_name)
 408 
 409     server = tf.train.Server(
 410         cluster,
 411         job_name=FLAGS.job_name,
 412         task_index=FLAGS.task_index,
 413         config=config)
 414 
 415     if FLAGS.job_name == 'ps':
 416       return server.join()
 417 
 418     device_setter = tf.train.replica_device_setter(
 419         worker_device='/job:worker/task:%d' % FLAGS.task_index,
 420         cluster=cluster)
 421 
 422   else:
 423     server = None
 424     device_setter = tf.train.replica_device_setter(0)
 425 
 426   # Build the graph.
 427   with tf.Graph().as_default():
 428     with tf.device(device_setter):
 429       model = Model(FLAGS.input_base_path, FLAGS)
 430 
 431       # If an eval path is present, then create eval operators and set up scalar
 432       # summaries to report on the results.  Run the evals on the CPU since
 433       # the analogy eval requires a fairly enormous tensor to be allocated to
 434       # do the nearest neighbor search.
 435       if FLAGS.eval_base_path:
 436         wordsim_filenames = glob.glob(
 437             os.path.join(FLAGS.eval_base_path, '*.ws.tab'))
 438 
 439         for filename in wordsim_filenames:
 440           name = os.path.basename(filename).split('.')[0]
 441           with tf.device(tf.DeviceSpec(device_type='CPU')):
 442             op = model.wordsim_eval_op(filename)
 443             tf.summary.scalar(name, op)
 444 
 445         analogy_filenames = glob.glob(
 446             os.path.join(FLAGS.eval_base_path, '*.an.tab'))
 447 
 448         for filename in analogy_filenames:
 449           name = os.path.basename(filename).split('.')[0]
 450           with tf.device(tf.DeviceSpec(device_type='CPU')):
 451             op = model.analogy_eval_op(filename)
 452             tf.summary.scalar(name, op)
 453 
 454       tf.summary.scalar('loss', model.loss_op)
 455 
 456     # Train on, soldier.
 457     supervisor = tf.train.Supervisor(
 458         logdir=FLAGS.output_base_path,
 459         is_chief=(FLAGS.task_index == 0),
 460         save_summaries_secs=60,
 461         recovery_wait_secs=5)
 462 
 463     max_step = FLAGS.num_epochs * model.steps_per_epoch
 464     master = server.target if server else ''
 465     with supervisor.managed_session(master) as session:
 466       local_step = 0
 467       global_step = session.run(model.global_step)
 468       while not supervisor.should_stop() and global_step < max_step:
 469         global_step, loss, _ = session.run([
 470             model.global_step, model.loss_op, model.train_op])
 471 
 472         if not np.isfinite(loss):
 473           raise ValueError('non-finite cost at step %d' % global_step)
 474 
 475         local_step += 1
 476         if local_step % 10 == 0:
 477           tf.logging.info(
 478               'local_step=%d global_step=%d loss=%.1f, %.1f%% complete',
 479               local_step, global_step, loss, 100.0 * global_step / max_step)
 480 
 481       if FLAGS.task_index == 0:
 482         supervisor.saver.save(
 483             session, supervisor.save_path, global_step=global_step)
 484 
 485         model.write_embeddings(FLAGS, session)
 486 
 487 
 488 if __name__ == '__main__':
 489   tf.app.run()
