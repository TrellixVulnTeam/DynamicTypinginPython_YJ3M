  17 """Prepare a corpus for processing by swivel.
  18 
  19 Creates a sharded word co-occurrence matrix from a text file input corpus.
  20 
  21 Usage:
  22 
  23   prep.py --output_dir <output-dir> --input <text-file>
  24 
  25 Options:
  26 
  27   --input <filename>
  28       The input text.
  29 
  30   --output_dir <directory>
  31       Specifies the output directory where the various Swivel data
  32       files should be placed.
  33 
  34   --shard_size <int>
  35       Specifies the shard size; default 4096.
  36 
  37   --min_count <int>
  38       Specifies the minimum number of times a word should appear
  39       to be included in the vocabulary; default 5.
  40 
  41   --max_vocab <int>
  42       Specifies the maximum vocabulary size; default shard size
  43       times 1024.
  44 
  45   --vocab <filename>
  46       Use the specified unigram vocabulary instead of generating
  47       it from the corpus.
  48 
  49   --window_size <int>
  50       Specifies the window size for computing co-occurrence stats;
  51       default 10.
  52 
  53   --bufsz <int>
  54       The number of co-occurrences that are buffered; default 16M.
  55 
  56 """

  87 def words(line):
  88   """Splits a line of text into tokens."""
  89   return line.strip().split()

  92 def create_vocabulary(lines):
  93   """Reads text lines and generates a vocabulary."""
  94   lines.seek(0, os.SEEK_END)
  95   nbytes = lines.tell()
  96   lines.seek(0, os.SEEK_SET)
  97 
  98   vocab = {}
  99   for lineno, line in enumerate(lines, start=1):
 100     for word in words(line):
 101       vocab.setdefault(word, 0)
 102       vocab[word] += 1
 103 
 104     if lineno % 100000 == 0:
 105       pos = lines.tell()
 106       sys.stdout.write('\rComputing vocabulary: %0.1f%% (%d/%d)...' % (
 107           100.0 * pos / nbytes, pos, nbytes))
 108       sys.stdout.flush()
 109 
 110   sys.stdout.write('\n')
 111 
 112   vocab = [(tok, n) for tok, n in vocab.iteritems() if n >= FLAGS.min_count]
 113   vocab.sort(key=lambda kv: (-kv[1], kv[0]))
 114 
 115   num_words = min(len(vocab), FLAGS.max_vocab)
 116   if num_words % FLAGS.shard_size != 0:
 117     num_words -= num_words % FLAGS.shard_size
 118 
 119   if not num_words:
 120     raise Exception('empty vocabulary')
 121 
 122   print('vocabulary contains %d tokens' % num_words)
 123 
 124   vocab = vocab[:num_words]
 125   return [tok for tok, n in vocab]

 128 def write_vocab_and_sums(vocab, sums, vocab_filename, sums_filename):
 129   """Writes vocabulary and marginal sum files."""
 130   with open(os.path.join(FLAGS.output_dir, vocab_filename), 'w') as vocab_out:
 131     with open(os.path.join(FLAGS.output_dir, sums_filename), 'w') as sums_out:
 132       for tok, cnt in itertools.izip(vocab, sums):
 133         print >> vocab_out, tok
 134         print >> sums_out, cnt

 137 def compute_coocs(lines, vocab):
 138   """Compute the co-occurrence statistics from the text.
 139 
 140   This generates a temporary file for each shard that contains the intermediate
 141   counts from the shard: these counts must be subsequently sorted and collated.
 142 
 143   """
 144   word_to_id = {tok: idx for idx, tok in enumerate(vocab)}
 145 
 146   lines.seek(0, os.SEEK_END)
 147   nbytes = lines.tell()
 148   lines.seek(0, os.SEEK_SET)
 149 
 150   num_shards = len(vocab) / FLAGS.shard_size
 151 
 152   shardfiles = {}
 153   for row in range(num_shards):
 154     for col in range(num_shards):
 155       filename = os.path.join(
 156           FLAGS.output_dir, 'shard-%03d-%03d.tmp' % (row, col))
 157 
 158       shardfiles[(row, col)] = open(filename, 'w+')
 159 
 160   def flush_coocs():
 161     for (row_id, col_id), cnt in coocs.iteritems():
 162       row_shard = row_id % num_shards
 163       row_off = row_id / num_shards
 164       col_shard = col_id % num_shards
 165       col_off = col_id / num_shards
 166 
 167       # Since we only stored (a, b), we emit both (a, b) and (b, a).
 168       shardfiles[(row_shard, col_shard)].write(
 169           shard_cooc_fmt.pack(row_off, col_off, cnt))
 170 
 171       shardfiles[(col_shard, row_shard)].write(
 172           shard_cooc_fmt.pack(col_off, row_off, cnt))
 173 
 174   coocs = {}
 175   sums = [0.0] * len(vocab)
 176 
 177   for lineno, line in enumerate(lines, start=1):
 178     # Computes the word IDs for each word in the sentence.  This has the effect
 179     # of "stretching" the window past OOV tokens.
 180     wids = filter(
 181         lambda wid: wid is not None,
 182         (word_to_id.get(w) for w in words(line)))
 183 
 184     for pos in xrange(len(wids)):
 185       lid = wids[pos]
 186       window_extent = min(FLAGS.window_size + 1, len(wids) - pos)
 187       for off in xrange(1, window_extent):
 188         rid = wids[pos + off]
 189         pair = (min(lid, rid), max(lid, rid))
 190         count = 1.0 / off
 191         sums[lid] += count
 192         sums[rid] += count
 193         coocs.setdefault(pair, 0.0)
 194         coocs[pair] += count
 195 
 196       sums[lid] += 1.0
 197       pair = (lid, lid)
 198       coocs.setdefault(pair, 0.0)
 199       coocs[pair] += 0.5  # Only add 1/2 since we output (a, b) and (b, a)
 200 
 201     if lineno % 10000 == 0:
 202       pos = lines.tell()
 203       sys.stdout.write('\rComputing co-occurrences: %0.1f%% (%d/%d)...' % (
 204           100.0 * pos / nbytes, pos, nbytes))
 205       sys.stdout.flush()
 206 
 207       if len(coocs) > FLAGS.bufsz:
 208         flush_coocs()
 209         coocs = {}
 210 
 211   flush_coocs()
 212   sys.stdout.write('\n')
 213 
 214   return shardfiles, sums

 217 def write_shards(vocab, shardfiles):
 218   """Processes the temporary files to generate the final shard data.
 219 
 220   The shard data is stored as a tf.Example protos using a TFRecordWriter. The
 221   temporary files are removed from the filesystem once they've been processed.
 222 
 223   """
 224   num_shards = len(vocab) / FLAGS.shard_size
 225 
 226   ix = 0
 227   for (row, col), fh in shardfiles.iteritems():
 228     ix += 1
 229     sys.stdout.write('\rwriting shard %d/%d' % (ix, len(shardfiles)))
 230     sys.stdout.flush()
 231 
 232     # Read the entire binary co-occurrence and unpack it into an array.
 233     fh.seek(0)
 234     buf = fh.read()
 235     os.unlink(fh.name)
 236     fh.close()
 237 
 238     coocs = [
 239         shard_cooc_fmt.unpack_from(buf, off)
 240         for off in range(0, len(buf), shard_cooc_fmt.size)]
 241 
 242     # Sort and merge co-occurrences for the same pairs.
 243     coocs.sort()
 244 
 245     if coocs:
 246       current_pos = 0
 247       current_row_col = (coocs[current_pos][0], coocs[current_pos][1])
 248       for next_pos in range(1, len(coocs)):
 249         next_row_col = (coocs[next_pos][0], coocs[next_pos][1])
 250         if current_row_col == next_row_col:
 251           coocs[current_pos] = (
 252               coocs[current_pos][0],
 253               coocs[current_pos][1],
 254               coocs[current_pos][2] + coocs[next_pos][2])
 255         else:
 256           current_pos += 1
 257           if current_pos < next_pos:
 258             coocs[current_pos] = coocs[next_pos]
 259 
 260           current_row_col = (coocs[current_pos][0], coocs[current_pos][1])
 261 
 262       coocs = coocs[:(1 + current_pos)]
 263 
 264     # Convert to a TF Example proto.
 265     def _int64s(xs):
 266       return tf.train.Feature(int64_list=tf.train.Int64List(value=list(xs)))
 267 
 268     def _floats(xs):
 269       return tf.train.Feature(float_list=tf.train.FloatList(value=list(xs)))
 270 
 271     example = tf.train.Example(features=tf.train.Features(feature={
 272         'global_row': _int64s(
 273             row + num_shards * i for i in range(FLAGS.shard_size)),
 274         'global_col': _int64s(
 275             col + num_shards * i for i in range(FLAGS.shard_size)),
 276 
 277         'sparse_local_row': _int64s(cooc[0] for cooc in coocs),
 278         'sparse_local_col': _int64s(cooc[1] for cooc in coocs),
 279         'sparse_value': _floats(cooc[2] for cooc in coocs),
 280     }))
 281 
 282     filename = os.path.join(FLAGS.output_dir, 'shard-%03d-%03d.pb' % (row, col))
 283     with open(filename, 'w') as out:
 284       out.write(example.SerializeToString())
 285 
 286   sys.stdout.write('\n')

 289 def main(_):
 290   # Create the output directory, if necessary
 291   if FLAGS.output_dir and not os.path.isdir(FLAGS.output_dir):
 292     os.makedirs(FLAGS.output_dir)
 293 
 294   # Read the file onces to create the vocabulary.
 295   if FLAGS.vocab:
 296     with open(FLAGS.vocab, 'r') as lines:
 297       vocab = [line.strip() for line in lines]
 298   else:
 299     with open(FLAGS.input, 'r') as lines:
 300       vocab = create_vocabulary(lines)
 301 
 302   # Now read the file again to determine the co-occurrence stats.
 303   with open(FLAGS.input, 'r') as lines:
 304     shardfiles, sums = compute_coocs(lines, vocab)
 305 
 306   # Collect individual shards into the shards.recs file.
 307   write_shards(vocab, shardfiles)
 308 
 309   # Now write the marginals.  They're symmetric for this application.
 310   write_vocab_and_sums(vocab, sums, 'row_vocab.txt', 'row_sums.txt')
 311   write_vocab_and_sums(vocab, sums, 'col_vocab.txt', 'col_sums.txt')
 312 
 313   print('done!')
 314 
 315 
 316 if __name__ == '__main__':
 317   tf.app.run()
