  17 """Converts a Glove binary co-occurrence matrix into Swivel shards.
  18 
  19 Usage:
  20 
  21   glove_to_shards.py --input <coocs> --vocab <vocab> --output_dir <output_dir>
  22 
  23 Options
  24 
  25   --input <coocs>
  26       The Glove co-occurrence file.
  27 
  28   --vocab <vocab>
  29       Path to the vocabulary text file, one token per line.
  30 
  31   --output_dir <directory>
  32       Specifies the touput directory where the various Swivel data
  33       files sohuld be placed.
  34 
  35   --shard_size <int>
  36       Specifies the shard size; default 4096.
  37 """
  38 
  39 from __future__ import print_function
  40 
  41 import itertools
  42 import os
  43 import struct
  44 import sys
  45 
  46 import tensorflow as tf
  47 from six.moves import xrange
  48 
  49 flags = tf.app.flags
  50 
  51 flags.DEFINE_string('input', 'coocurrences.bin', 'Vocabulary file')
  52 flags.DEFINE_string('vocab', 'vocab.txt', 'Vocabulary file')
  53 flags.DEFINE_string('output_dir', '/tmp/swivel_data', 'Output directory')
  54 flags.DEFINE_integer('shard_size', 4096, 'Shard size')
  55 
  56 FLAGS = tf.app.flags.FLAGS
  57 
  58 glove_cooc_fmt = struct.Struct('iid')
  59 shard_cooc_fmt = struct.Struct('if')
  60 
  61 
  62 def make_shard_files(coocs, nshards, vocab_sz):
  63   """Chops the binary Glove co-occurrence matrix into shards.
  64 
  65   This reads the Glove binary co-occurrence file and assigns individual
  66   co-occurrence counts to the appropriate Swivel shard.
  67 
  68   Args:
  69     coocs: the co-occurrnece file to read
  70     nshards: the number of shards along one dimension of the square matrix
  71     vocab_sz: the vocabulary size
  72 
  73   Returns:
  74     A (shard_table, marginals) tuple.  The shard_table maps the row and column
  75     shard ID to a file handle containing the co-occurrences for that shard; the
  76     marginals contain the marginal sums.
  77   """
  78   row_sums = [0] * vocab_sz
  79   col_sums = [0] * vocab_sz
  80 
  81   coocs.seek(0, os.SEEK_END)
  82   ncoocs = coocs.tell() / glove_cooc_fmt.size
  83   coocs.seek(0, os.SEEK_SET)
  84 
  85   shard_files = {}
  86 
  87   for row in range(nshards):
  88     for col in range(nshards):
  89       filename = os.path.join(
  90           FLAGS.output_dir, 'shard-%03d-%03d.bin' % (row, col))
  91 
  92       shard_files[(row, col)] = open(filename, 'w+')
  93 
  94   for ix in xrange(ncoocs):
  95     if ix % 1000000 == 0:
  96       sys.stdout.write('\rsharding co-occurrences: %0.1f%% (%d/%d)' % (
  97           100.0 * ix / ncoocs, ix, ncoocs))
  98 
  99       sys.stdout.flush()
 100 
 101     bits = coocs.read(glove_cooc_fmt.size)
 102     if not bits:
 103       break
 104 
 105     # Glove has 1-indexed IDs.
 106     row_id, col_id, cnt = glove_cooc_fmt.unpack(bits)
 107     if row_id > vocab_sz or col_id > vocab_sz:
 108       continue
 109 
 110     row_id -= 1
 111     row_shard = row_id % nshards
 112     row_off = row_id / nshards
 113 
 114     col_id -= 1
 115     col_shard = col_id % nshards
 116     col_off = col_id / nshards
 117 
 118     shard_pos = row_off * FLAGS.shard_size + col_off  # row major
 119 
 120     shard_files[(row_shard, col_shard)].write(
 121         shard_cooc_fmt.pack(shard_pos, cnt))
 122 
 123     # Accumulate marginals.
 124     row_sums[row_id] += cnt
 125     col_sums[col_id] += cnt
 126 
 127   sys.stdout.write('\n')
 128 
 129   if any(abs(r - c) > 0.1 for r, c in itertools.izip(row_sums, col_sums)):
 130     print('WARNING! Row and column marginals differ; is your matrix symmetric?',
 131           file=sys.stderr)
 132 
 133   return (shard_files, row_sums)
 134 
 135 def main(_):
 136   with open(FLAGS.vocab, 'r') as lines:
 137     orig_vocab_sz = sum(1 for _ in lines)
 138 
 139   shard_sz = FLAGS.shard_size
 140   vocab_sz = orig_vocab_sz - orig_vocab_sz % shard_sz
 141   nshards = vocab_sz / shard_sz
 142 
 143   print('vocab size is %d (originally %d), %d %dx%d-element shards' % (
 144       vocab_sz, orig_vocab_sz, nshards * nshards, shard_sz, shard_sz))
 145 
 146   # Create the output directory, if necessary
 147   if FLAGS.output_dir and not os.path.isdir(FLAGS.output_dir):
 148     os.makedirs(FLAGS.output_dir)
 149 
 150   with open(FLAGS.input, 'r') as coocs:
 151     shard_files, marginals = make_shard_files(coocs, nshards, vocab_sz)
 152 
 153   # Now sort the shards and write the TFRecords.
 154   filename = os.path.join(FLAGS.output_dir, 'shards.recs')
 155   with tf.python_io.TFRecordWriter(filename) as writer:
 156     ix = 0
 157     for (row, col), fh in shard_files.iteritems():
 158       ix += 1
 159       sys.stdout.write('\rwriting shard %d/%d' % (ix, len(shard_files)))
 160       sys.stdout.flush()
 161 
 162       fh.seek(0)
 163       buf = fh.read()
 164       os.unlink(fh.name)
 165       fh.close()
 166 
 167       coocs = [
 168           shard_cooc_fmt.unpack_from(buf, off)
 169           for off in range(0, len(buf), shard_cooc_fmt.size)]
 170 
 171       # N.B. we assume that there aren't any duplicates here!
 172       coocs.sort(key=lambda kv: kv[0])
 173 
 174       def _int64s(xs):
 175         return tf.train.Feature(int64_list=tf.train.Int64List(value=list(xs)))
 176 
 177       def _floats(xs):
 178         return tf.train.Feature(float_list=tf.train.FloatList(value=list(xs)))
 179 
 180       example = tf.train.Example(features=tf.train.Features(feature={
 181           'global_row': _int64s(row + nshards * i for i in range(shard_sz)),
 182           'global_col': _int64s(col + nshards * i for i in range(shard_sz)),
 183           'sparse_local_row': _int64s(pos / shard_sz for pos, _ in coocs),
 184           'sparse_local_col': _int64s(pos % shard_sz for pos, _ in coocs),
 185           'sparse_value': _floats(cnt for _, cnt in coocs)}))
 186 
 187       writer.write(example.SerializeToString())
 188 
 189   print('\nwriting marginals...')
 190 
 191   with open(os.path.join(FLAGS.output_dir, 'marginals.txt'), 'w') as fh:
 192     for cnt in marginals:
 193       fh.write('%0.1f\n' % cnt)
 194 
 195   print('done!')
 196 
 197 if __name__ == '__main__':
 198   tf.app.run()
