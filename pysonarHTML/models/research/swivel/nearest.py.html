  17 """Simple tool for inspecting nearest neighbors and analogies."""
  18 
  19 from __future__ import print_function
  20 import re
  21 import sys
  22 from getopt import GetoptError, getopt
  23 
  24 from vecs import Vecs
  25 
  26 try:
  27   opts, args = getopt(sys.argv[1:], 'v:e:', ['vocab=', 'embeddings='])
  28 except GetoptError as e:
  29   print(e, file=sys.stderr)
  30   sys.exit(2)
  31 
  32 opt_vocab = 'vocab.txt'
  33 opt_embeddings = None
  34 
  35 for o, a in opts:
  36   if o in ('-v', '--vocab'):
  37     opt_vocab = a
  38   if o in ('-e', '--embeddings'):
  39     opt_embeddings = a
  40 
  41 vecs = Vecs(opt_vocab, opt_embeddings)
  42 
  43 while True:
  44   sys.stdout.write('query> ')
  45   sys.stdout.flush()
  46 
  47   query = sys.stdin.readline().strip()
  48   if not query:
  49     break
  50 
  51   parts = re.split(r'\s+', query)
  52 
  53   if len(parts) == 1:
  54     res = vecs.neighbors(parts[0])
  55 
  56   elif len(parts) == 3:
  57     vs = [vecs.lookup(w) for w in parts]
  58     if any(v is None for v in vs):
  59       print('not in vocabulary: %s' % (
  60             ', '.join(tok for tok, v in zip(parts, vs) if v is None)))
  61 
  62       continue
  63 
  64     res = vecs.neighbors(vs[2] - vs[0] + vs[1])
  65 
  66   else:
  67     print('use a single word to query neighbors, or three words for analogy')
  68     continue
  69 
  70   if not res:
  71     continue
  72 
  73   for word, sim in res[:20]:
  74     print('%0.4f: %s' % (sim, word))
  75 
  76   print()
