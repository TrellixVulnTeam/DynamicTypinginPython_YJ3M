  30 def conv_linear(args, kw, kh, nin, nout, rate, do_bias, bias_start, prefix):
  31   """Convolutional linear map."""
  32   if not isinstance(args, (list, tuple)):
  33     args = [args]
  34   with tf.variable_scope(prefix):
  35     with tf.device("/cpu:0"):
  36       k = tf.get_variable("CvK", [kw, kh, nin, nout])
  37     if len(args) == 1:
  38       arg = args[0]
  39     else:
  40       arg = tf.concat(axis=3, values=args)
  41     res = tf.nn.convolution(arg, k, dilation_rate=(rate, 1), padding="SAME")
  42     if not do_bias: return res
  43     with tf.device("/cpu:0"):
  44       bias_term = tf.get_variable(
  45           "CvB", [nout], initializer=tf.constant_initializer(bias_start))
  46     bias_term = tf.reshape(bias_term, [1, 1, 1, nout])
  47     return res + bias_term

  50 def sigmoid_cutoff(x, cutoff):
  51   """Sigmoid with cutoff, e.g., 1.2sigmoid(x) - 0.1."""
  52   y = tf.sigmoid(x)
  53   if cutoff < 1.01: return y
  54   d = (cutoff - 1.0) / 2.0
  55   return tf.minimum(1.0, tf.maximum(0.0, cutoff * y - d), name="cutoff_min")

  58 @function.Defun(tf.float32, noinline=True)
  59 def sigmoid_cutoff_12(x):
  60   """Sigmoid with cutoff 1.2, specialized for speed and memory use."""
  61   y = tf.sigmoid(x)
  62   return tf.minimum(1.0, tf.maximum(0.0, 1.2 * y - 0.1), name="cutoff_min_12")

  66 def sigmoid_hard(x):
  67   """Hard sigmoid."""
  68   return tf.minimum(1.0, tf.maximum(0.0, 0.25 * x + 0.5))

  85 def tanh_cutoff(x, cutoff):
  86   """Tanh with cutoff, e.g., 1.1tanh(x) cut to [-1. 1]."""
  87   y = tf.tanh(x)
  88   if cutoff < 1.01: return y
  89   d = (cutoff - 1.0) / 2.0
  90   return tf.minimum(1.0, tf.maximum(-1.0, (1.0 + d) * y))

  93 @function.Defun(tf.float32, noinline=True)
  94 def tanh_hard(x):
  95   """Hard tanh."""
  96   return tf.minimum(1.0, tf.maximum(0.0, x))

  99 def layer_norm(x, nmaps, prefix, epsilon=1e-5):
 100   """Layer normalize the 4D tensor x, averaging over the last dimension."""
 101   with tf.variable_scope(prefix):
 102     scale = tf.get_variable("layer_norm_scale", [nmaps],
 103                             initializer=tf.ones_initializer())
 104     bias = tf.get_variable("layer_norm_bias", [nmaps],
 105                            initializer=tf.zeros_initializer())
 106     mean, variance = tf.nn.moments(x, [3], keep_dims=True)
 107     norm_x = (x - mean) / tf.sqrt(variance + epsilon)
 108     return norm_x * scale + bias

 111 def conv_gru(inpts, mem, kw, kh, nmaps, rate, cutoff, prefix, do_layer_norm,
 112              args_len=None):
 113   """Convolutional GRU."""
 114   def conv_lin(args, suffix, bias_start):
 115     total_args_len = args_len or len(args) * nmaps
 116     res = conv_linear(args, kw, kh, total_args_len, nmaps, rate, True,
 117                       bias_start, prefix + "/" + suffix)
 118     if do_layer_norm:
 119       return layer_norm(res, nmaps, prefix + "/" + suffix)
 120     else:
 121       return res
 122   if cutoff == 1.2:
 123     reset = sigmoid_cutoff_12(conv_lin(inpts + [mem], "r", 1.0))
 124     gate = sigmoid_cutoff_12(conv_lin(inpts + [mem], "g", 1.0))
 125   elif cutoff > 10:
 126     reset = sigmoid_hard(conv_lin(inpts + [mem], "r", 1.0))
 127     gate = sigmoid_hard(conv_lin(inpts + [mem], "g", 1.0))
 128   else:
 129     reset = sigmoid_cutoff(conv_lin(inpts + [mem], "r", 1.0), cutoff)
 130     gate = sigmoid_cutoff(conv_lin(inpts + [mem], "g", 1.0), cutoff)
 131   if cutoff > 10:
 132     candidate = tanh_hard(conv_lin(inpts + [reset * mem], "c", 0.0))
 133   else:
 134     # candidate = tanh_cutoff(conv_lin(inpts + [reset * mem], "c", 0.0), cutoff)
 135     candidate = tf.tanh(conv_lin(inpts + [reset * mem], "c", 0.0))
 136   return gate * mem + (1 - gate) * candidate

 146 def memory_run(step, nmaps, mem_size, batch_size, vocab_size,
 147                global_step, do_training, update_mem, decay_factor, num_gpus,
 148                target_emb_weights, output_w, gpu_targets_tn, it):
 149   """Run memory."""
 150   q = step[:, 0, it, :]
 151   mlabels = gpu_targets_tn[:, it, 0]
 152   res, mask, mem_loss = memory_call(
 153       q, mlabels, nmaps, mem_size, vocab_size, num_gpus, update_mem)
 154   res = tf.gather(target_emb_weights, res) * tf.expand_dims(mask[:, 0], 1)
 155 
 156   # Mix gold and original in the first steps, 20% later.
 157   gold = tf.nn.dropout(tf.gather(target_emb_weights, mlabels), 0.7)
 158   use_gold = 1.0 - tf.cast(global_step, tf.float32) / (1000. * decay_factor)
 159   use_gold = tf.maximum(use_gold, 0.2) * do_training
 160   mem = tf.cond(tf.less(tf.random_uniform([]), use_gold),
 161                 lambda: use_gold * gold + (1.0 - use_gold) * res,
 162                 lambda: res)
 163   mem = tf.reshape(mem, [-1, 1, 1, nmaps])
 164   return mem, mem_loss, update_mem

 172 def quantize(t, quant_scale, max_value=1.0):
 173   """Quantize a tensor t with each element in [-max_value, max_value]."""
 174   t = tf.minimum(max_value, tf.maximum(t, -max_value))
 175   big = quant_scale * (t + max_value) + 0.5
 176   with tf.get_default_graph().gradient_override_map({"Floor": "CustomIdG"}):
 177     res = (tf.floor(big) / quant_scale) - max_value
 178   return res

 187 def autoenc_quantize(x, nbits, nmaps, do_training, layers=1):
 188   """Autoencoder into nbits vectors of bits, using noise and sigmoids."""
 189   enc_x = tf.reshape(x, [-1, nmaps])
 190   for i in xrange(layers - 1):
 191     enc_x = tf.layers.dense(enc_x, nmaps, name="autoenc_%d" % i)
 192   enc_x = tf.layers.dense(enc_x, nbits, name="autoenc_%d" % (layers - 1))
 193   noise = tf.truncated_normal(tf.shape(enc_x), stddev=2.0)
 194   dec_x = sigmoid_cutoff_12(enc_x + noise * do_training)
 195   dec_x = tf.reshape(dec_x, [-1, nbits])
 196   for i in xrange(layers):
 197     dec_x = tf.layers.dense(dec_x, nmaps, name="autodec_%d" % i)
 198   return tf.reshape(dec_x, tf.shape(x))

 201 def make_dense(targets, noclass, low_param):
 202   """Move a batch of targets to a dense 1-hot representation."""
 203   low = low_param / float(noclass - 1)
 204   high = 1.0 - low * (noclass - 1)
 205   targets = tf.cast(targets, tf.int64)
 206   return tf.one_hot(targets, depth=noclass, on_value=high, off_value=low)

 209 def reorder_beam(beam_size, batch_size, beam_val, output, is_first,
 210                  tensors_to_reorder):
 211   """Reorder to minimize beam costs."""
 212   # beam_val is [batch_size x beam_size]; let b = batch_size * beam_size
 213   # decided is len x b x a x b
 214   # output is b x out_size; step is b x len x a x b;
 215   outputs = tf.split(axis=0, num_or_size_splits=beam_size, value=tf.nn.log_softmax(output))
 216   all_beam_vals, all_beam_idx = [], []
 217   beam_range = 1 if is_first else beam_size
 218   for i in xrange(beam_range):
 219     top_out, top_out_idx = tf.nn.top_k(outputs[i], k=beam_size)
 220     cur_beam_val = beam_val[:, i]
 221     top_out = tf.Print(top_out, [top_out, top_out_idx, beam_val, i,
 222                                  cur_beam_val], "GREPO", summarize=8)
 223     all_beam_vals.append(top_out + tf.expand_dims(cur_beam_val, 1))
 224     all_beam_idx.append(top_out_idx)
 225   all_beam_idx = tf.reshape(tf.transpose(tf.concat(axis=1, values=all_beam_idx), [1, 0]),
 226                             [-1])
 227   top_beam, top_beam_idx = tf.nn.top_k(tf.concat(axis=1, values=all_beam_vals), k=beam_size)
 228   top_beam_idx = tf.Print(top_beam_idx, [top_beam, top_beam_idx],
 229                           "GREP", summarize=8)
 230   reordered = [[] for _ in xrange(len(tensors_to_reorder) + 1)]
 231   top_out_idx = []
 232   for i in xrange(beam_size):
 233     which_idx = top_beam_idx[:, i] * batch_size + tf.range(batch_size)
 234     top_out_idx.append(tf.gather(all_beam_idx, which_idx))
 235     which_beam = top_beam_idx[:, i] / beam_size  # [batch]
 236     which_beam = which_beam * batch_size + tf.range(batch_size)
 237     reordered[0].append(tf.gather(output, which_beam))
 238     for i, t in enumerate(tensors_to_reorder):
 239       reordered[i + 1].append(tf.gather(t, which_beam))
 240   new_tensors = [tf.concat(axis=0, values=t) for t in reordered]
 241   top_out_idx = tf.concat(axis=0, values=top_out_idx)
 242   return (top_beam, new_tensors[0], top_out_idx, new_tensors[1:])

 248   def __init__(self, nmaps, vec_size, niclass, noclass, dropout,

 252                atrous=False, beam_size=1, backward=True, do_layer_norm=False,
 253                autoenc_decay=1.0):
 254     # Feeds for parameters and ops to update them.
 255     self.nmaps = nmaps
 256     if backward:
 257       self.global_step = tf.Variable(0, trainable=False, name="global_step")
 258       self.cur_length = tf.Variable(min_length, trainable=False)
 259       self.cur_length_incr_op = self.cur_length.assign_add(1)
 260       self.lr = tf.Variable(learning_rate, trainable=False)
 261       self.lr_decay_op = self.lr.assign(self.lr * 0.995)
 262     self.do_training = tf.placeholder(tf.float32, name="do_training")
 263     self.update_mem = tf.placeholder(tf.int32, name="update_mem")
 264     self.noise_param = tf.placeholder(tf.float32, name="noise_param")
 265 
 266     # Feeds for inputs, targets, outputs, losses, etc.
 267     self.input = tf.placeholder(tf.int32, name="inp")
 268     self.target = tf.placeholder(tf.int32, name="tgt")
 269     self.prev_step = tf.placeholder(tf.float32, name="prev_step")
 270     gpu_input = tf.split(axis=0, num_or_size_splits=num_gpus, value=self.input)
 271     gpu_target = tf.split(axis=0, num_or_size_splits=num_gpus, value=self.target)
 272     gpu_prev_step = tf.split(axis=0, num_or_size_splits=num_gpus, value=self.prev_step)

 275     if backward:
 276       adam_lr = 0.005 * self.lr

 279       def adam_update(grads):
 280         return adam.apply_gradients(zip(grads, tf.trainable_variables()),
 281                                     global_step=self.global_step,
 282                                     name="adam_update")

 284     # When switching from Adam to SGD we perform reverse-decay.
 285     if backward:
 286       global_step_float = tf.cast(self.global_step, tf.float32)
 287       sampling_decay_exponent = global_step_float / 100000.0
 288       sampling_decay = tf.maximum(0.05, tf.pow(0.5, sampling_decay_exponent))
 289       self.sampling = sampling_rate * 0.05 / sampling_decay
 290     else:
 291       self.sampling = tf.constant(0.0)

 293     # Cache variables on cpu if needed.
 294     if num_replicas > 1 or num_gpus > 1:
 295       with tf.device("/cpu:0"):
 296         caching_const = tf.constant(0)
 297       tf.get_variable_scope().set_caching_device(caching_const.op.device)
 298       # partitioner = tf.variable_axis_size_partitioner(1024*256*4)

 301     def gpu_avg(l):
 302       if l[0] is None:
 303         for elem in l:
 304           assert elem is None
 305         return 0.0
 306       if len(l) < 2:
 307         return l[0]
 308       return sum(l) / float(num_gpus)

 310     self.length_tensor = tf.placeholder(tf.int32, name="length")
 311 
 312     with tf.device("/cpu:0"):
 313       emb_weights = tf.get_variable(
 314           "embedding", [niclass, vec_size],
 315           initializer=tf.random_uniform_initializer(-1.7, 1.7))
 316       if beam_size > 0:
 317         target_emb_weights = tf.get_variable(
 318             "target_embedding", [noclass, nmaps],
 319             initializer=tf.random_uniform_initializer(-1.7, 1.7))
 320       e0 = tf.scatter_update(emb_weights,
 321                              tf.constant(0, dtype=tf.int32, shape=[1]),
 322                              tf.zeros([1, vec_size]))
 323       output_w = tf.get_variable("output_w", [nmaps, noclass], tf.float32)

 325     def conv_rate(layer):
 326       if atrous:
 327         return 2**layer
 328       return 1

 330     # pylint: disable=cell-var-from-loop
 331     def enc_step(step):
 332       """Encoder step."""
 333       if autoenc_decay < 1.0:
 334         quant_step = autoenc_quantize(step, 16, nmaps, self.do_training)
 335         if backward:
 336           exp_glob = tf.train.exponential_decay(1.0, self.global_step - 10000,
 337                                                 1000, autoenc_decay)
 338           dec_factor = 1.0 - exp_glob  # * self.do_training
 339           dec_factor = tf.cond(tf.less(self.global_step, 10500),
 340                                lambda: tf.constant(0.05), lambda: dec_factor)
 341         else:
 342           dec_factor = 1.0
 343         cur = tf.cond(tf.less(tf.random_uniform([]), dec_factor),
 344                       lambda: quant_step, lambda: step)
 345       else:
 346         cur = step
 347       if dropout > 0.0001:
 348         cur = tf.nn.dropout(cur, keep_prob)
 349       if act_noise > 0.00001:
 350         cur += tf.truncated_normal(tf.shape(cur)) * act_noise_scale
 351       # Do nconvs-many CGRU steps.
 352       if do_jit and tf.get_variable_scope().reuse:
 353         with jit_scope():
 354           for layer in xrange(nconvs):
 355             cur = conv_gru([], cur, kw, kh, nmaps, conv_rate(layer),
 356                            cutoff, "ecgru_%d" % layer, do_layer_norm)
 357       else:
 358         for layer in xrange(nconvs):
 359           cur = conv_gru([], cur, kw, kh, nmaps, conv_rate(layer),
 360                          cutoff, "ecgru_%d" % layer, do_layer_norm)
 361       return cur

 366     def dec_substep(step, decided):
 367       """Decoder sub-step."""
 368       cur = step
 369       if dropout > 0.0001:
 370         cur = tf.nn.dropout(cur, keep_prob)
 371       if act_noise > 0.00001:
 372         cur += tf.truncated_normal(tf.shape(cur)) * act_noise_scale
 373       # Do nconvs-many CGRU steps.
 374       if do_jit and tf.get_variable_scope().reuse:
 375         with jit_scope():
 376           for layer in xrange(nconvs):
 377             cur = conv_gru([decided], cur, kw, kh, nmaps, conv_rate(layer),
 378                            cutoff, "dcgru_%d" % layer, do_layer_norm)
 379       else:
 380         for layer in xrange(nconvs):
 381           cur = conv_gru([decided], cur, kw, kh, nmaps, conv_rate(layer),
 382                          cutoff, "dcgru_%d" % layer, do_layer_norm)
 383       return cur

 386     def dec_step(step, it, it_int, decided, output_ta, tgts,
 387                  mloss, nupd_in, out_idx, beam_cost):
 388       """Decoder step."""
 389       nupd, mem_loss = 0, 0.0
 390       if mem_size > 0:
 391         it_incr = tf.minimum(it+1, length - 1)
 392         mem, mem_loss, nupd = memory_run(
 393             step, nmaps, mem_size, batch_size, noclass, self.global_step,
 394             self.do_training, self.update_mem, 10, num_gpus,
 395             target_emb_weights, output_w, gpu_targets_tn, it_incr)
 396       step = dec_substep(step, decided)
 397       output_l = tf.expand_dims(tf.expand_dims(step[:, it, 0, :], 1), 1)
 398       # Calculate argmax output.
 399       output = tf.reshape(output_l, [-1, nmaps])
 400       # pylint: disable=cell-var-from-loop
 401       output = tf.matmul(output, output_w)
 402       if beam_size > 1:
 403         beam_cost, output, out, reordered = reorder_beam(
 404             beam_size, batch_size, beam_cost, output, it_int == 0,
 405             [output_l, out_idx, step, decided])
 406         [output_l, out_idx, step, decided] = reordered
 407       else:
 408         # Scheduled sampling.
 409         out = tf.multinomial(tf.stop_gradient(output), 1)
 410         out = tf.to_int32(tf.squeeze(out, [1]))
 411       out_write = output_ta.write(it, output_l[:batch_size, :, :, :])
 412       output = tf.gather(target_emb_weights, out)
 413       output = tf.reshape(output, [-1, 1, nmaps])
 414       output = tf.concat(axis=1, values=[output] * height)
 415       tgt = tgts[it, :, :, :]
 416       selected = tf.cond(tf.less(tf.random_uniform([]), self.sampling),
 417                          lambda: output, lambda: tgt)
 418       # pylint: enable=cell-var-from-loop
 419       dec_write = place_at14(decided, tf.expand_dims(selected, 1), it)
 420       out_idx = place_at13(
 421           out_idx, tf.reshape(out, [beam_size * batch_size, 1, 1]), it)
 422       if mem_size > 0:
 423         mem = tf.concat(axis=2, values=[mem] * height)
 424         dec_write = place_at14(dec_write, mem, it_incr)
 425       return (step, dec_write, out_write, mloss + mem_loss, nupd_in + nupd,
 426               out_idx, beam_cost)

 428     # Main model construction.
 429     gpu_outputs = []
 430     gpu_losses = []
 431     gpu_grad_norms = []
 432     grads_list = []
 433     gpu_out_idx = []
 434     self.after_enc_step = []
 435     for gpu in xrange(num_gpus):  # Multi-GPU towers, average gradients later.
 436       length = self.length_tensor
 437       length_float = tf.cast(length, tf.float32)
 438       if gpu > 0:
 439         tf.get_variable_scope().reuse_variables()
 440       gpu_outputs.append([])
 441       gpu_losses.append([])
 442       gpu_grad_norms.append([])
 443       with tf.name_scope("gpu%d" % gpu), tf.device("/gpu:%d" % gpu):
 444         # Main graph creation loop.
 445         data.print_out("Creating model.")
 446         start_time = time.time()
 447 
 448         # Embed inputs and calculate mask.
 449         with tf.device("/cpu:0"):
 450           tgt_shape = tf.shape(tf.squeeze(gpu_target[gpu], [1]))
 451           weights = tf.where(tf.squeeze(gpu_target[gpu], [1]) > 0,
 452                              tf.ones(tgt_shape), tf.zeros(tgt_shape))
 453 
 454           # Embed inputs and targets.
 455           with tf.control_dependencies([e0]):
 456             start = tf.gather(emb_weights, gpu_input[gpu])  # b x h x l x nmaps
 457             gpu_targets_tn = gpu_target[gpu]  # b x 1 x len
 458             if beam_size > 0:
 459               embedded_targets_tn = tf.gather(target_emb_weights,
 460                                               gpu_targets_tn)
 461               embedded_targets_tn = tf.transpose(
 462                   embedded_targets_tn, [2, 0, 1, 3])  # len x b x 1 x nmaps
 463               embedded_targets_tn = tf.concat(axis=2, values=[embedded_targets_tn] * height)
 464 
 465         # First image comes from start by applying convolution and adding 0s.
 466         start = tf.transpose(start, [0, 2, 1, 3])  # Now b x len x h x vec_s
 467         first = conv_linear(start, 1, 1, vec_size, nmaps, 1, True, 0.0, "input")
 468         first = layer_norm(first, nmaps, "input")
 469 
 470         # Computation steps.
 471         keep_prob = dropout * 3.0 / tf.sqrt(length_float)
 472         keep_prob = 1.0 - self.do_training * keep_prob
 473         act_noise_scale = act_noise * self.do_training
 474 
 475         # Start with a convolutional gate merging previous step.
 476         step = conv_gru([gpu_prev_step[gpu]], first,
 477                         kw, kh, nmaps, 1, cutoff, "first", do_layer_norm)
 478 
 479         # This is just for running a baseline RNN seq2seq model.
 480         if do_rnn:
 481           self.after_enc_step.append(step)  # Not meaningful here, but needed.
 482           def lstm_cell():
 483             return tf.contrib.rnn.BasicLSTMCell(height * nmaps)
 484           cell = tf.contrib.rnn.MultiRNNCell(
 485               [lstm_cell() for _ in range(nconvs)])
 486           with tf.variable_scope("encoder"):
 487             encoder_outputs, encoder_state = tf.nn.dynamic_rnn(
 488                 cell, tf.reshape(step, [batch_size, length, height * nmaps]),
 489                 dtype=tf.float32, time_major=False)
 490 
 491           # Attention.
 492           attn = tf.layers.dense(
 493               encoder_outputs, height * nmaps, name="attn1")
 494 
 495           # pylint: disable=cell-var-from-loop
 496           @function.Defun(noinline=True)
 497           def attention_query(query, attn_v):
 498             vecs = tf.tanh(attn + tf.expand_dims(query, 1))
 499             mask = tf.reduce_sum(vecs * tf.reshape(attn_v, [1, 1, -1]), 2)
 500             mask = tf.nn.softmax(mask)
 501             return tf.reduce_sum(encoder_outputs * tf.expand_dims(mask, 2), 1)
 502 
 503           with tf.variable_scope("decoder"):
 504             def decoder_loop_fn(state__prev_cell_out__unused, cell_inp__cur_tgt):
 505               """Decoder loop function."""
 506               state, prev_cell_out, _ = state__prev_cell_out__unused
 507               cell_inp, cur_tgt = cell_inp__cur_tgt
 508               attn_q = tf.layers.dense(prev_cell_out, height * nmaps,
 509                                        name="attn_query")
 510               attn_res = attention_query(attn_q, tf.get_variable(
 511                   "attn_v", [height * nmaps],
 512                   initializer=tf.random_uniform_initializer(-0.1, 0.1)))
 513               concatenated = tf.reshape(tf.concat(axis=1, values=[cell_inp, attn_res]),
 514                                         [batch_size, 2 * height * nmaps])
 515               cell_inp = tf.layers.dense(
 516                   concatenated, height * nmaps, name="attn_merge")
 517               output, new_state = cell(cell_inp, state)
 518 
 519               mem_loss = 0.0
 520               if mem_size > 0:
 521                 res, mask, mem_loss = memory_call(
 522                     output, cur_tgt, height * nmaps, mem_size, noclass,
 523                     num_gpus, self.update_mem)
 524                 res = tf.gather(target_emb_weights, res)
 525                 res *= tf.expand_dims(mask[:, 0], 1)
 526                 output = tf.layers.dense(
 527                     tf.concat(axis=1, values=[output, res]), height * nmaps, name="rnnmem")
 528 
 529               return new_state, output, mem_loss
 530             # pylint: enable=cell-var-from-loop
 531             gpu_targets = tf.squeeze(gpu_target[gpu], [1])  # b x len
 532             gpu_tgt_trans = tf.transpose(gpu_targets, [1, 0])
 533             dec_zero = tf.zeros([batch_size, 1], dtype=tf.int32)
 534             dec_inp = tf.concat(axis=1, values=[dec_zero, gpu_targets])
 535             dec_inp = dec_inp[:, :length]
 536             embedded_dec_inp = tf.gather(target_emb_weights, dec_inp)
 537             embedded_dec_inp_proj = tf.layers.dense(
 538                 embedded_dec_inp, height * nmaps, name="dec_proj")
 539             embedded_dec_inp_proj = tf.transpose(embedded_dec_inp_proj,
 540                                                  [1, 0, 2])
 541             init_vals = (encoder_state,
 542                          tf.zeros([batch_size, height * nmaps]), 0.0)
 543             _, dec_outputs, mem_losses = tf.scan(
 544                 decoder_loop_fn, (embedded_dec_inp_proj, gpu_tgt_trans),
 545                 initializer=init_vals)
 546           mem_loss = tf.reduce_mean(mem_losses)
 547           outputs = tf.layers.dense(dec_outputs, nmaps, name="out_proj")
 548           # Final convolution to get logits, list outputs.
 549           outputs = tf.matmul(tf.reshape(outputs, [-1, nmaps]), output_w)
 550           outputs = tf.reshape(outputs, [length, batch_size, noclass])
 551           gpu_out_idx.append(tf.argmax(outputs, 2))
 552         else:  # Here we go with the Neural GPU.
 553           # Encoder.
 554           enc_length = length
 555           step = enc_step(step)  # First step hard-coded.
 556           # pylint: disable=cell-var-from-loop
 557           i = tf.constant(1)
 558           c = lambda i, _s: tf.less(i, enc_length)
 559           def enc_step_lambda(i, step):
 560             with tf.variable_scope(tf.get_variable_scope(), reuse=True):
 561               new_step = enc_step(step)
 562             return (i + 1, new_step)
 563           _, step = tf.while_loop(
 564               c, enc_step_lambda, [i, step],
 565               parallel_iterations=1, swap_memory=True)
 566           # pylint: enable=cell-var-from-loop
 567 
 568           self.after_enc_step.append(step)
 569 
 570           # Decoder.
 571           if beam_size > 0:
 572             output_ta = tf.TensorArray(
 573                 dtype=tf.float32, size=length, dynamic_size=False,
 574                 infer_shape=False, name="outputs")
 575             out_idx = tf.zeros([beam_size * batch_size, length, 1],
 576                                dtype=tf.int32)
 577             decided_t = tf.zeros([beam_size * batch_size, length,
 578                                   height, vec_size])
 579 
 580             # Prepare for beam search.
 581             tgts = tf.concat(axis=1, values=[embedded_targets_tn] * beam_size)
 582             beam_cost = tf.zeros([batch_size, beam_size])
 583             step = tf.concat(axis=0, values=[step] * beam_size)
 584             # First step hard-coded.
 585             step, decided_t, output_ta, mem_loss, nupd, oi, bc = dec_step(
 586                 step, 0, 0, decided_t, output_ta, tgts, 0.0, 0, out_idx,
 587                 beam_cost)
 588             tf.get_variable_scope().reuse_variables()
 589             # pylint: disable=cell-var-from-loop
 590             def step_lambda(i, step, dec_t, out_ta, ml, nu, oi, bc):
 591               with tf.variable_scope(tf.get_variable_scope(), reuse=True):
 592                 s, d, t, nml, nu, oi, bc = dec_step(
 593                     step, i, 1, dec_t, out_ta, tgts, ml, nu, oi, bc)
 594               return (i + 1, s, d, t, nml, nu, oi, bc)
 595             i = tf.constant(1)
 596             c = lambda i, _s, _d, _o, _ml, _nu, _oi, _bc: tf.less(i, length)
 597             _, step, _, output_ta, mem_loss, nupd, out_idx, _ = tf.while_loop(
 598                 c, step_lambda,
 599                 [i, step, decided_t, output_ta, mem_loss, nupd, oi, bc],
 600                 parallel_iterations=1, swap_memory=True)
 601             # pylint: enable=cell-var-from-loop
 602             gpu_out_idx.append(tf.squeeze(out_idx, [2]))
 603             outputs = output_ta.stack()
 604             outputs = tf.squeeze(outputs, [2, 3])  # Now l x b x nmaps
 605           else:
 606             # If beam_size is 0 or less, we don't have a decoder.
 607             mem_loss = 0.0
 608             outputs = tf.transpose(step[:, :, 1, :], [1, 0, 2])
 609             gpu_out_idx.append(tf.argmax(outputs, 2))
 610 
 611           # Final convolution to get logits, list outputs.
 612           outputs = tf.matmul(tf.reshape(outputs, [-1, nmaps]), output_w)
 613           outputs = tf.reshape(outputs, [length, batch_size, noclass])
 614         gpu_outputs[gpu] = tf.nn.softmax(outputs)
 615 
 616         # Calculate cross-entropy loss and normalize it.
 617         targets_soft = make_dense(tf.squeeze(gpu_target[gpu], [1]),
 618                                   noclass, 0.1)
 619         targets_soft = tf.reshape(targets_soft, [-1, noclass])
 620         targets_hard = make_dense(tf.squeeze(gpu_target[gpu], [1]),
 621                                   noclass, 0.0)
 622         targets_hard = tf.reshape(targets_hard, [-1, noclass])
 623         output = tf.transpose(outputs, [1, 0, 2])
 624         xent_soft = tf.reshape(tf.nn.softmax_cross_entropy_with_logits(
 625             logits=tf.reshape(output, [-1, noclass]), labels=targets_soft),
 626                                [batch_size, length])
 627         xent_hard = tf.reshape(tf.nn.softmax_cross_entropy_with_logits(
 628             logits=tf.reshape(output, [-1, noclass]), labels=targets_hard),
 629                                [batch_size, length])
 630         low, high = 0.1 / float(noclass - 1), 0.9
 631         const = high * tf.log(high) + float(noclass - 1) * low * tf.log(low)
 632         weight_sum = tf.reduce_sum(weights) + 1e-20
 633         true_perp = tf.reduce_sum(xent_hard * weights) / weight_sum
 634         soft_loss = tf.reduce_sum(xent_soft * weights) / weight_sum
 635         perp_loss = soft_loss + const
 636         # Final loss: cross-entropy + shared parameter relaxation part + extra.
 637         mem_loss = 0.5 * tf.reduce_mean(mem_loss) / length_float
 638         total_loss = perp_loss + mem_loss
 639         gpu_losses[gpu].append(true_perp)
 640 
 641         # Gradients.
 642         if backward:
 643           data.print_out("Creating backward pass for the model.")
 644           grads = tf.gradients(
 645               total_loss, tf.trainable_variables(),
 646               colocate_gradients_with_ops=True)
 647           for g_i, g in enumerate(grads):
 648             if isinstance(g, tf.IndexedSlices):
 649               grads[g_i] = tf.convert_to_tensor(g)
 650           grads, norm = tf.clip_by_global_norm(grads, max_grad_norm)
 651           gpu_grad_norms[gpu].append(norm)
 652           for g in grads:
 653             if grad_noise_scale > 0.001:
 654               g += tf.truncated_normal(tf.shape(g)) * self.noise_param
 655           grads_list.append(grads)
 656         else:
 657           gpu_grad_norms[gpu].append(0.0)
 658         data.print_out("Created model for gpu %d in %.2f s."
 659                        % (gpu, time.time() - start_time))
 660 
 661     self.updates = []
 662     self.after_enc_step = tf.concat(axis=0, values=self.after_enc_step)  # Concat GPUs.
 663     if backward:
 664       tf.get_variable_scope()._reuse = False
 665       tf.get_variable_scope().set_caching_device(None)
 666       grads = [gpu_avg([grads_list[g][i] for g in xrange(num_gpus)])
 667                for i in xrange(len(grads_list[0]))]
 668       update = adam_update(grads)
 669       self.updates.append(update)
 670     else:
 671       self.updates.append(tf.no_op())
 672 
 673     self.losses = [gpu_avg([gpu_losses[g][i] for g in xrange(num_gpus)])
 674                    for i in xrange(len(gpu_losses[0]))]
 675     self.out_idx = tf.concat(axis=0, values=gpu_out_idx)
 676     self.grad_norms = [gpu_avg([gpu_grad_norms[g][i] for g in xrange(num_gpus)])
 677                        for i in xrange(len(gpu_grad_norms[0]))]
 678     self.outputs = [tf.concat(axis=1, values=[gpu_outputs[g] for g in xrange(num_gpus)])]
 679     self.quantize_op = quantize_weights_op(512, 8)
 680     if backward:
 681       self.saver = tf.train.Saver(tf.global_variables(), max_to_keep=10)
 682 
 683   def step(self, sess, inp, target, do_backward_in, noise_param=None,
 684            beam_size=2, eos_id=2, eos_cost=0.0, update_mem=None, state=None):
 685     """Run a step of the network."""
 686     batch_size, height, length = inp.shape[0], inp.shape[1], inp.shape[2]
 687     do_backward = do_backward_in
 688     train_mode = True
 689     if do_backward_in is None:
 690       do_backward = False
 691       train_mode = False
 692     if update_mem is None:
 693       update_mem = do_backward
 694     feed_in = {}
 695     # print "    feeding sequences of length %d" % length
 696     if state is None:
 697       state = np.zeros([batch_size, length, height, self.nmaps])
 698     feed_in[self.prev_step.name] = state
 699     feed_in[self.length_tensor.name] = length
 700     feed_in[self.noise_param.name] = noise_param if noise_param else 0.0
 701     feed_in[self.do_training.name] = 1.0 if do_backward else 0.0
 702     feed_in[self.update_mem.name] = 1 if update_mem else 0
 703     if do_backward_in is False:
 704       feed_in[self.sampling.name] = 0.0
 705     index = 0  # We're dynamic now.
 706     feed_out = []
 707     if do_backward:
 708       feed_out.append(self.updates[index])
 709       feed_out.append(self.grad_norms[index])
 710     if train_mode:
 711       feed_out.append(self.losses[index])
 712     feed_in[self.input.name] = inp
 713     feed_in[self.target.name] = target
 714     feed_out.append(self.outputs[index])
 715     if train_mode:
 716       # Make a full-sequence training step with one call to session.run.
 717       res = sess.run([self.after_enc_step] + feed_out, feed_in)
 718       after_enc_state, res = res[0], res[1:]
 719     else:
 720       # Make a full-sequence decoding step with one call to session.run.
 721       feed_in[self.sampling.name] = 1.1  # Sample every time.
 722       res = sess.run([self.after_enc_step, self.out_idx] + feed_out, feed_in)
 723       after_enc_state, out_idx = res[0], res[1]
 724       res = [res[2][l] for l in xrange(length)]
 725       outputs = [out_idx[:, i] for i in xrange(length)]
 726       cost = [0.0 for _ in xrange(beam_size * batch_size)]
 727       seen_eos = [0 for _ in xrange(beam_size * batch_size)]
 728       for idx, logit in enumerate(res):
 729         best = outputs[idx]
 730         for b in xrange(batch_size):
 731           if seen_eos[b] > 1:
 732             cost[b] -= eos_cost
 733           else:
 734             cost[b] += np.log(logit[b][best[b]])
 735           if best[b] in [eos_id]:
 736             seen_eos[b] += 1
 737       res = [[-c for c in cost]] + outputs
 738     # Collect and output results.
 739     offset = 0
 740     norm = None
 741     if do_backward:
 742       offset = 2
 743       norm = res[1]
 744     if train_mode:
 745       outputs = res[offset + 1]
 746       outputs = [outputs[l] for l in xrange(length)]
 747     return res[offset], outputs, norm, after_enc_state
