  90 def add(n1, n2, base=10):
  91   """Add two numbers represented as lower-endian digit lists."""
  92   k = max(len(n1), len(n2)) + 1
  93   d1 = n1 + [0 for _ in xrange(k - len(n1))]
  94   d2 = n2 + [0 for _ in xrange(k - len(n2))]
  95   res = []
  96   carry = 0
  97   for i in xrange(k):
  98     if d1[i] + d2[i] + carry < base:
  99       res.append(d1[i] + d2[i] + carry)
 100       carry = 0
 101     else:
 102       res.append(d1[i] + d2[i] + carry - base)
 103       carry = 1
 104   while res and res[-1] == 0:
 105     res = res[:-1]
 106   if res: return res
 107   return [0]

 110 def init_data(task, length, nbr_cases, nclass):
 111   """Data initialization."""
 112   def rand_pair(l, task):
 113     """Random data pair for a task. Total length should be <= l."""
 114     k = int((l-1)/2)
 115     base = 10
 116     if task[0] == "b": base = 2
 117     if task[0] == "q": base = 4
 118     d1 = [np.random.randint(base) for _ in xrange(k)]
 119     d2 = [np.random.randint(base) for _ in xrange(k)]
 120     if task in ["add", "badd", "qadd"]:
 121       res = add(d1, d2, base)
 122     elif task in ["mul", "bmul"]:
 123       d1n = sum([d * (base ** i) for i, d in enumerate(d1)])
 124       d2n = sum([d * (base ** i) for i, d in enumerate(d2)])
 125       if task == "bmul":
 126         res = [int(x) for x in list(reversed(str(bin(d1n * d2n))))[:-2]]
 127       else:
 128         res = [int(x) for x in list(reversed(str(d1n * d2n)))]
 129     else:
 130       sys.exit()
 131     sep = [12]
 132     if task in ["add", "badd", "qadd"]: sep = [11]
 133     inp = [d + 1 for d in d1] + sep + [d + 1 for d in d2]
 134     return inp, [r + 1 for r in res]

 136   def rand_dup_pair(l):
 137     """Random data pair for duplication task. Total length should be <= l."""
 138     k = int(l/2)
 139     x = [np.random.randint(nclass - 1) + 1 for _ in xrange(k)]
 140     inp = x + [0 for _ in xrange(l - k)]
 141     res = x + x + [0 for _ in xrange(l - 2*k)]
 142     return inp, res

 144   def rand_rev2_pair(l):
 145     """Random data pair for reverse2 task. Total length should be <= l."""
 146     inp = [(np.random.randint(nclass - 1) + 1,
 147             np.random.randint(nclass - 1) + 1) for _ in xrange(l/2)]
 148     res = [i for i in reversed(inp)]
 149     return [x for p in inp for x in p], [x for p in res for x in p]

 151   def rand_search_pair(l):
 152     """Random data pair for search task. Total length should be <= l."""
 153     inp = [(np.random.randint(nclass - 1) + 1,
 154             np.random.randint(nclass - 1) + 1) for _ in xrange(l-1/2)]
 155     q = np.random.randint(nclass - 1) + 1
 156     res = 0
 157     for (k, v) in reversed(inp):
 158       if k == q:
 159         res = v
 160     return [x for p in inp for x in p] + [q], [res]

 162   def rand_kvsort_pair(l):
 163     """Random data pair for key-value sort. Total length should be <= l."""
 164     keys = [(np.random.randint(nclass - 1) + 1, i) for i in xrange(l/2)]
 165     vals = [np.random.randint(nclass - 1) + 1 for _ in xrange(l/2)]
 166     kv = [(k, vals[i]) for (k, i) in keys]
 167     sorted_kv = [(k, vals[i]) for (k, i) in sorted(keys)]
 168     return [x for p in kv for x in p], [x for p in sorted_kv for x in p]

 170   def prog_io_pair(prog, max_len, counter=0):
 171     try:
 172       ilen = np.random.randint(max_len - 3) + 1
 173       bound = max(15 - (counter / 20), 1)
 174       inp = [random.choice(range(-bound, bound)) for _ in range(ilen)]
 175       inp_toks = [program_utils.prog_rev_vocab[t]
 176                   for t in program_utils.tokenize(str(inp)) if t != ","]
 177       out = program_utils.evaluate(prog, {"a": inp})
 178       out_toks = [program_utils.prog_rev_vocab[t]
 179                   for t in program_utils.tokenize(str(out)) if t != ","]
 180       if counter > 400:
 181         out_toks = []
 182       if (out_toks and out_toks[0] == program_utils.prog_rev_vocab["["] and
 183           len(out_toks) != len([o for o in out if o == ","]) + 3):
 184         raise ValueError("generated list with too long ints")
 185       if (out_toks and out_toks[0] != program_utils.prog_rev_vocab["["] and
 186           len(out_toks) > 1):
 187         raise ValueError("generated one int but tokenized it to many")
 188       if len(out_toks) > max_len:
 189         raise ValueError("output too long")
 190       return (inp_toks, out_toks)
 191     except ValueError:
 192       return prog_io_pair(prog, max_len, counter+1)

 194   def spec(inp):
 195     """Return the target given the input for some tasks."""
 196     if task == "sort":
 197       return sorted(inp)
 198     elif task == "id":
 199       return inp
 200     elif task == "rev":
 201       return [i for i in reversed(inp)]
 202     elif task == "incr":
 203       carry = 1
 204       res = []
 205       for i in xrange(len(inp)):
 206         if inp[i] + carry < nclass:
 207           res.append(inp[i] + carry)
 208           carry = 0
 209         else:
 210           res.append(1)
 211           carry = 1
 212       return res
 213     elif task == "left":
 214       return [inp[0]]
 215     elif task == "right":
 216       return [inp[-1]]
 217     elif task == "left-shift":
 218       return [inp[l-1] for l in xrange(len(inp))]
 219     elif task == "right-shift":
 220       return [inp[l+1] for l in xrange(len(inp))]
 221     else:
 222       print_out("Unknown spec for task " + str(task))
 223       sys.exit()

 225   l = length
 226   cur_time = time.time()
 227   total_time = 0.0
 228 
 229   is_prog = task in ["progeval", "progsynth"]
 230   if is_prog:
 231     inputs_per_prog = 5
 232     program_utils.make_vocab()
 233     progs = read_tmp_file("programs_len%d" % (l / 10))
 234     if not progs:
 235       progs = program_utils.gen(l / 10, 1.2 * nbr_cases / inputs_per_prog)
 236       write_tmp_file("programs_len%d" % (l / 10), progs)
 237     prog_ios = read_tmp_file("programs_len%d_io" % (l / 10))
 238     nbr_cases = min(nbr_cases, len(progs) * inputs_per_prog) / 1.2
 239     if not prog_ios:
 240       # Generate program io data.
 241       prog_ios = []
 242       for pidx, prog in enumerate(progs):
 243         if pidx % 500 == 0:
 244           print_out("== generating io pairs for program %d" % pidx)
 245         if pidx * inputs_per_prog > nbr_cases * 1.2:
 246           break
 247         ptoks = [program_utils.prog_rev_vocab[t]
 248                  for t in program_utils.tokenize(prog)]
 249         ptoks.append(program_utils.prog_rev_vocab["_EOS"])
 250         plen = len(ptoks)
 251         for _ in xrange(inputs_per_prog):
 252           if task == "progeval":
 253             inp, out = prog_io_pair(prog, plen)
 254             prog_ios.append(str(inp) + "\t" + str(out) + "\t" + prog)
 255           elif task == "progsynth":
 256             plen = max(len(ptoks), 8)
 257             for _ in xrange(3):
 258               inp, out = prog_io_pair(prog, plen / 2)
 259               prog_ios.append(str(inp) + "\t" + str(out) + "\t" + prog)
 260       write_tmp_file("programs_len%d_io" % (l / 10), prog_ios)
 261     prog_ios_dict = {}
 262     for s in prog_ios:
 263       i, o, p = s.split("\t")
 264       i_clean = "".join([c for c in i if c.isdigit() or c == " "])
 265       o_clean = "".join([c for c in o if c.isdigit() or c == " "])
 266       inp = [int(x) for x in i_clean.split()]
 267       out = [int(x) for x in o_clean.split()]
 268       if inp and out:
 269         if p in prog_ios_dict:
 270           prog_ios_dict[p].append([inp, out])
 271         else:
 272           prog_ios_dict[p] = [[inp, out]]
 273     # Use prog_ios_dict to create data.
 274     progs = []
 275     for prog in prog_ios_dict:
 276       if len([c for c in prog if c == ";"]) <= (l / 10):
 277         progs.append(prog)
 278     nbr_cases = min(nbr_cases, len(progs) * inputs_per_prog) / 1.2
 279     print_out("== %d training cases on %d progs" % (nbr_cases, len(progs)))
 280     for pidx, prog in enumerate(progs):
 281       if pidx * inputs_per_prog > nbr_cases * 1.2:
 282         break
 283       ptoks = [program_utils.prog_rev_vocab[t]
 284                for t in program_utils.tokenize(prog)]
 285       ptoks.append(program_utils.prog_rev_vocab["_EOS"])
 286       plen = len(ptoks)
 287       dset = train_set if pidx < nbr_cases / inputs_per_prog else test_set
 288       for _ in xrange(inputs_per_prog):
 289         if task == "progeval":
 290           inp, out = prog_ios_dict[prog].pop()
 291           dset[task][bin_for(plen)].append([[ptoks, inp, [], []], [out]])
 292         elif task == "progsynth":
 293           plen, ilist = max(len(ptoks), 8), [[]]
 294           for _ in xrange(3):
 295             inp, out = prog_ios_dict[prog].pop()
 296             ilist.append(inp + out)
 297           dset[task][bin_for(plen)].append([ilist, [ptoks]])

 299   for case in xrange(0 if is_prog else nbr_cases):
 300     total_time += time.time() - cur_time
 301     cur_time = time.time()
 302     if l > 10000 and case % 100 == 1:
 303       print_out("  avg gen time %.4f s" % (total_time / float(case)))
 304     if task in ["add", "badd", "qadd", "bmul", "mul"]:
 305       i, t = rand_pair(l, task)
 306       train_set[task][bin_for(len(i))].append([[[], i, [], []], [t]])
 307       i, t = rand_pair(l, task)
 308       test_set[task][bin_for(len(i))].append([[[], i, [], []], [t]])
 309     elif task == "dup":
 310       i, t = rand_dup_pair(l)
 311       train_set[task][bin_for(len(i))].append([[i], [t]])
 312       i, t = rand_dup_pair(l)
 313       test_set[task][bin_for(len(i))].append([[i], [t]])
 314     elif task == "rev2":
 315       i, t = rand_rev2_pair(l)
 316       train_set[task][bin_for(len(i))].append([[i], [t]])
 317       i, t = rand_rev2_pair(l)
 318       test_set[task][bin_for(len(i))].append([[i], [t]])
 319     elif task == "search":
 320       i, t = rand_search_pair(l)
 321       train_set[task][bin_for(len(i))].append([[i], [t]])
 322       i, t = rand_search_pair(l)
 323       test_set[task][bin_for(len(i))].append([[i], [t]])
 324     elif task == "kvsort":
 325       i, t = rand_kvsort_pair(l)
 326       train_set[task][bin_for(len(i))].append([[i], [t]])
 327       i, t = rand_kvsort_pair(l)
 328       test_set[task][bin_for(len(i))].append([[i], [t]])
 329     elif task not in ["progeval", "progsynth"]:
 330       inp = [np.random.randint(nclass - 1) + 1 for i in xrange(l)]
 331       target = spec(inp)
 332       train_set[task][bin_for(l)].append([[inp], [target]])
 333       inp = [np.random.randint(nclass - 1) + 1 for i in xrange(l)]
 334       target = spec(inp)
 335       test_set[task][bin_for(l)].append([[inp], [target]])

 338 def to_symbol(i):
 339   """Covert ids to text."""
 340   if i == 0: return ""
 341   if i == 11: return "+"
 342   if i == 12: return "*"
 343   return str(i-1)

 347   """Covert text to ids."""
 348   if s == "+": return 11
 349   if s == "*": return 12
 350   return int(s) + 1

 353 def get_batch(bin_id, batch_size, data_set, height, offset=None, preset=None):
 354   """Get a batch of data, training or testing."""
 355   inputs, targets = [], []
 356   pad_length = bins[bin_id]
 357   for b in xrange(batch_size):
 358     if preset is None:
 359       elem = random.choice(data_set[bin_id])
 360       if offset is not None and offset + b < len(data_set[bin_id]):
 361         elem = data_set[bin_id][offset + b]
 362     else:
 363       elem = preset
 364     inpt, targett, inpl, targetl = elem[0], elem[1], [], []
 365     for inp in inpt:
 366       inpl.append(inp + [0 for _ in xrange(pad_length - len(inp))])
 367     if len(inpl) == 1:
 368       for _ in xrange(height - 1):
 369         inpl.append([0 for _ in xrange(pad_length)])
 370     for target in targett:
 371       targetl.append(target + [0 for _ in xrange(pad_length - len(target))])
 372     inputs.append(inpl)
 373     targets.append(targetl)
 374   res_input = np.array(inputs, dtype=np.int32)
 375   res_target = np.array(targets, dtype=np.int32)
 376   assert list(res_input.shape) == [batch_size, height, pad_length]
 377   assert list(res_target.shape) == [batch_size, 1, pad_length]
 378   return res_input, res_target

 381 def print_out(s, newline=True):
 382   """Print a message out and log it to file."""
 383   if log_filename:
 384     try:
 385       with tf.gfile.GFile(log_filename, mode="a") as f:
 386         f.write(s + ("\n" if newline else ""))
 387     # pylint: disable=bare-except
 388     except:
 389       sys.stderr.write("Error appending to %s\n" % log_filename)
 390   sys.stdout.write(s + ("\n" if newline else ""))
 391   sys.stdout.flush()

 398 def accuracy(inpt_t, output, target_t, batch_size, nprint,
 399              beam_out=None, beam_scores=None):
 400   """Calculate output accuracy given target."""
 401   assert nprint < batch_size + 1
 402   inpt = []
 403   for h in xrange(inpt_t.shape[1]):
 404     inpt.extend([inpt_t[:, h, l] for l in xrange(inpt_t.shape[2])])
 405   target = [target_t[:, 0, l] for l in xrange(target_t.shape[2])]
 406   def tok(i):
 407     if rev_vocab and i < len(rev_vocab):
 408       return rev_vocab[i]
 409     return str(i - 1)
 410   def task_print(inp, output, target):
 411     stop_bound = 0
 412     print_len = 0
 413     while print_len < len(target) and target[print_len] > stop_bound:
 414       print_len += 1
 415     print_out("    i: " + " ".join([tok(i) for i in inp if i > 0]))
 416     print_out("    o: " +
 417               " ".join([tok(output[l]) for l in xrange(print_len)]))
 418     print_out("    t: " +
 419               " ".join([tok(target[l]) for l in xrange(print_len)]))
 420   decoded_target = target
 421   decoded_output = decode(output)
 422   # Use beam output if given and score is high enough.
 423   if beam_out is not None:
 424     for b in xrange(batch_size):
 425       if beam_scores[b] >= 10.0:
 426         for l in xrange(min(len(decoded_output), beam_out.shape[2])):
 427           decoded_output[l][b] = int(beam_out[b, 0, l])
 428   total = 0
 429   errors = 0
 430   seq = [0 for b in xrange(batch_size)]
 431   for l in xrange(len(decoded_output)):
 432     for b in xrange(batch_size):
 433       if decoded_target[l][b] > 0:
 434         total += 1
 435         if decoded_output[l][b] != decoded_target[l][b]:
 436           seq[b] = 1
 437           errors += 1
 438   e = 0  # Previous error index
 439   for _ in xrange(min(nprint, sum(seq))):
 440     while seq[e] == 0:
 441       e += 1
 442     task_print([inpt[l][e] for l in xrange(len(inpt))],
 443                [decoded_output[l][e] for l in xrange(len(decoded_target))],
 444                [decoded_target[l][e] for l in xrange(len(decoded_target))])
 445     e += 1
 446   for b in xrange(nprint - errors):
 447     task_print([inpt[l][b] for l in xrange(len(inpt))],
 448                [decoded_output[l][b] for l in xrange(len(decoded_target))],
 449                [decoded_target[l][b] for l in xrange(len(decoded_target))])
 450   return errors, total, sum(seq)

 453 def safe_exp(x):
 454   perp = 10000
 455   x = float(x)
 456   if x < 100: perp = math.exp(x)
 457   if perp > 10000: return 10000
 458   return perp
