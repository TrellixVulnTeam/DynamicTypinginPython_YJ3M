 102 def zero_split(tok_list, append=None):
 103   """Split tok_list (list of ints) on 0s, append int to all parts if given."""
 104   res, cur, l = [], [], 0
 105   for tok in tok_list:
 106     if tok == 0:
 107       if append is not None:
 108         cur.append(append)
 109       res.append(cur)
 110       l = max(l, len(cur))
 111       cur = []
 112     else:
 113       cur.append(tok)
 114   if append is not None:
 115     cur.append(append)
 116   res.append(cur)
 117   l = max(l, len(cur))
 118   return res, l

 121 def read_data(source_path, target_path, buckets, max_size=None, print_out=True):
 122   """Read data from source and target files and put into buckets.
 123 
 124   Args:
 125     source_path: path to the files with token-ids for the source language.
 126     target_path: path to the file with token-ids for the target language;
 127       it must be aligned with the source file: n-th line contains the desired
 128       output for n-th line from the source_path.
 129     buckets: the buckets to use.
 130     max_size: maximum number of lines to read, all other will be ignored;
 131       if 0 or None, data files will be read completely (no limit).
 132       If set to 1, no data will be returned (empty lists of the right form).
 133     print_out: whether to print out status or not.
 134 
 135   Returns:
 136     data_set: a list of length len(_buckets); data_set[n] contains a list of
 137       (source, target) pairs read from the provided data files that fit
 138       into the n-th bucket, i.e., such that len(source) < _buckets[n][0] and
 139       len(target) < _buckets[n][1]; source and target are lists of token-ids.
 140   """
 141   data_set = [[] for _ in buckets]
 142   counter = 0
 143   if max_size != 1:
 144     with tf.gfile.GFile(source_path, mode="r") as source_file:
 145       with tf.gfile.GFile(target_path, mode="r") as target_file:
 146         source, target = source_file.readline(), target_file.readline()
 147         while source and target and (not max_size or counter < max_size):
 148           counter += 1
 149           if counter % 100000 == 0 and print_out:
 150             print("  reading data line %d" % counter)
 151             sys.stdout.flush()
 152           source_ids = [int(x) for x in source.split()]
 153           target_ids = [int(x) for x in target.split()]
 154           source_ids, source_len = zero_split(source_ids)
 155           target_ids, target_len = zero_split(target_ids, append=wmt.EOS_ID)
 156           for bucket_id, size in enumerate(buckets):
 157             if source_len <= size and target_len <= size:
 158               data_set[bucket_id].append([source_ids, target_ids])
 159               break
 160           source, target = source_file.readline(), target_file.readline()
 161   return data_set

 168 def calculate_buckets_scale(data_set, buckets, problem):
 169   """Calculate buckets scales for the given data set."""
 170   train_bucket_sizes = [len(data_set[b]) for b in xrange(len(buckets))]
 171   train_total_size = max(1, float(sum(train_bucket_sizes)))
 172 
 173   # A bucket scale is a list of increasing numbers from 0 to 1 that we'll use
 174   # to select a bucket. Length of [scale[i], scale[i+1]] is proportional to
 175   # the size if i-th training bucket, as used later.
 176   if problem not in train_buckets_scale:
 177     train_buckets_scale[problem] = []
 178   train_buckets_scale[problem].append(
 179       [sum(train_bucket_sizes[:i + 1]) / train_total_size
 180        for i in xrange(len(train_bucket_sizes))])
 181   return train_total_size

 184 def read_data_into_global(source_path, target_path, buckets,
 185                           max_size=None, print_out=True):
 186   """Read data into the global variables (can be in a separate thread)."""
 187   # pylint: disable=global-variable-not-assigned
 188   global global_train_set, train_buckets_scale
 189   # pylint: enable=global-variable-not-assigned
 190   data_set = read_data(source_path, target_path, buckets, max_size, print_out)
 191   global_train_set["wmt"].append(data_set)
 192   train_total_size = calculate_buckets_scale(data_set, buckets, "wmt")
 193   if print_out:
 194     print("  Finished global data reading (%d)." % train_total_size)

 197 def initialize(sess=None):
 198   """Initialize data and model."""
 199   global MAXLEN_F
 200   # Create training directory if it does not exist.
 201   if not tf.gfile.IsDirectory(FLAGS.train_dir):
 202     data.print_out("Creating training directory %s." % FLAGS.train_dir)
 203     tf.gfile.MkDir(FLAGS.train_dir)
 204   decode_suffix = "beam%dln%d" % (FLAGS.beam_size,
 205                                   int(100 * FLAGS.length_norm))
 206   if FLAGS.mode == 0:
 207     decode_suffix = ""
 208   if FLAGS.task >= 0:
 209     data.log_filename = os.path.join(FLAGS.train_dir,
 210                                      "log%d%s" % (FLAGS.task, decode_suffix))
 211   else:
 212     data.log_filename = os.path.join(FLAGS.train_dir, "neural_gpu/log")
 213 
 214   # Set random seed.
 215   if FLAGS.random_seed > 0:
 216     seed = FLAGS.random_seed + max(0, FLAGS.task)
 217     tf.set_random_seed(seed)
 218     random.seed(seed)
 219     np.random.seed(seed)
 220 
 221   # Check data sizes.
 222   assert data.bins
 223   max_length = min(FLAGS.max_length, data.bins[-1])
 224   while len(data.bins) > 1 and data.bins[-2] >= max_length + EXTRA_EVAL:
 225     data.bins = data.bins[:-1]
 226   if sess is None and FLAGS.task == 0 and FLAGS.num_replicas > 1:
 227     if max_length > 60:
 228       max_length = max_length * 1 / 2  # Save memory on chief.
 229   min_length = min(14, max_length - 3) if FLAGS.problem == "wmt" else 3
 230   for p in FLAGS.problem.split("-"):
 231     if p in ["progeval", "progsynth"]:
 232       min_length = max(26, min_length)
 233   assert max_length + 1 > min_length
 234   while len(data.bins) > 1 and data.bins[-2] >= max_length + EXTRA_EVAL:
 235     data.bins = data.bins[:-1]
 236 
 237   # Create checkpoint directory if it does not exist.
 238   if FLAGS.mode == 0 or FLAGS.task < 0:
 239     checkpoint_dir = os.path.join(FLAGS.train_dir, "neural_gpu%s"
 240                                   % ("" if FLAGS.task < 0 else str(FLAGS.task)))
 241   else:
 242     checkpoint_dir = FLAGS.train_dir
 243   if not tf.gfile.IsDirectory(checkpoint_dir):
 244     data.print_out("Creating checkpoint directory %s." % checkpoint_dir)
 245     tf.gfile.MkDir(checkpoint_dir)
 246 
 247   # Prepare data.
 248   if FLAGS.problem == "wmt":
 249     # Prepare WMT data.
 250     data.print_out("Preparing WMT data in %s" % FLAGS.data_dir)
 251     if FLAGS.simple_tokenizer:
 252       MAXLEN_F = 3.5
 253       (en_train, fr_train, en_dev, fr_dev,
 254        en_path, fr_path) = wmt.prepare_wmt_data(
 255            FLAGS.data_dir, FLAGS.vocab_size,
 256            tokenizer=wmt.space_tokenizer,
 257            normalize_digits=FLAGS.normalize_digits)
 258     else:
 259       (en_train, fr_train, en_dev, fr_dev,
 260        en_path, fr_path) = wmt.prepare_wmt_data(
 261            FLAGS.data_dir, FLAGS.vocab_size)
 262 
 263     # Read data into buckets and compute their sizes.
 264     fr_vocab, rev_fr_vocab = wmt.initialize_vocabulary(fr_path)
 265     data.vocab = fr_vocab
 266     data.rev_vocab = rev_fr_vocab
 267     data.print_out("Reading development and training data (limit: %d)."
 268                    % FLAGS.max_train_data_size)
 269     dev_set = {}
 270     dev_set["wmt"] = read_data(en_dev, fr_dev, data.bins)
 271     def data_read(size, print_out):
 272       read_data_into_global(en_train, fr_train, data.bins, size, print_out)
 273     data_read(50000, False)
 274     read_thread_small = threading.Thread(
 275         name="reading-data-small", target=lambda: data_read(900000, False))
 276     read_thread_small.start()
 277     read_thread_full = threading.Thread(
 278         name="reading-data-full",
 279         target=lambda: data_read(FLAGS.max_train_data_size, True))
 280     read_thread_full.start()
 281     data.print_out("Data reading set up.")
 282   else:
 283     # Prepare algorithmic data.
 284     en_path, fr_path = None, None
 285     tasks = FLAGS.problem.split("-")
 286     data_size = FLAGS.train_data_size
 287     for t in tasks:
 288       data.print_out("Generating data for %s." % t)
 289       if t in ["progeval", "progsynth"]:
 290         data.init_data(t, data.bins[-1], 20 * data_size, FLAGS.vocab_size)
 291         if len(program_utils.prog_vocab) > FLAGS.vocab_size - 2:
 292           raise ValueError("Increase vocab_size to %d for prog-tasks."
 293                            % (len(program_utils.prog_vocab) + 2))
 294         data.rev_vocab = program_utils.prog_vocab
 295         data.vocab = program_utils.prog_rev_vocab
 296       else:
 297         for l in xrange(max_length + EXTRA_EVAL - 1):
 298           data.init_data(t, l, data_size, FLAGS.vocab_size)
 299         data.init_data(t, data.bins[-2], data_size, FLAGS.vocab_size)
 300         data.init_data(t, data.bins[-1], data_size, FLAGS.vocab_size)
 301       if t not in global_train_set:
 302         global_train_set[t] = []
 303       global_train_set[t].append(data.train_set[t])
 304       calculate_buckets_scale(data.train_set[t], data.bins, t)
 305     dev_set = data.test_set
 306 
 307   # Grid-search parameters.
 308   lr = FLAGS.lr
 309   init_weight = FLAGS.init_weight
 310   max_grad_norm = FLAGS.max_grad_norm
 311   if sess is not None and FLAGS.task > -1:
 312     def job_id_factor(step):
 313       """If jobid / step mod 3 is 0, 1, 2: say 0, 1, -1."""
 314       return ((((FLAGS.task / step) % 3) + 1) % 3) - 1
 315     lr *= math.pow(2, job_id_factor(1))
 316     init_weight *= math.pow(1.5, job_id_factor(3))
 317     max_grad_norm *= math.pow(2, job_id_factor(9))
 318 
 319   # Print out parameters.
 320   curriculum = FLAGS.curriculum_seq
 321   msg1 = ("layers %d kw %d h %d kh %d batch %d noise %.2f"
 322           % (FLAGS.nconvs, FLAGS.kw, FLAGS.height, FLAGS.kh,
 323              FLAGS.batch_size, FLAGS.grad_noise_scale))
 324   msg2 = ("cut %.2f lr %.3f iw %.2f cr %.2f nm %d d%.4f gn %.2f %s"
 325           % (FLAGS.cutoff, lr, init_weight, curriculum, FLAGS.nmaps,
 326              FLAGS.dropout, max_grad_norm, msg1))
 327   data.print_out(msg2)
 328 
 329   # Create model and initialize it.
 330   tf.get_variable_scope().set_initializer(
 331       tf.orthogonal_initializer(gain=1.8 * init_weight))
 332   max_sampling_rate = FLAGS.max_sampling_rate if FLAGS.mode == 0 else 0.0
 333   o = FLAGS.vocab_size if FLAGS.max_target_vocab < 1 else FLAGS.max_target_vocab
 334   ngpu.CHOOSE_K = FLAGS.soft_mem_size
 335   do_beam_model = FLAGS.train_beam_freq > 0.0001 and FLAGS.beam_size > 1
 336   beam_size = FLAGS.beam_size if FLAGS.mode > 0 and not do_beam_model else 1
 337   beam_size = min(beam_size, FLAGS.beam_size)
 338   beam_model = None
 339   def make_ngpu(cur_beam_size, back):
 340     return ngpu.NeuralGPU(
 341         FLAGS.nmaps, FLAGS.vec_size, FLAGS.vocab_size, o,
 342         FLAGS.dropout, max_grad_norm, FLAGS.cutoff, FLAGS.nconvs,
 343         FLAGS.kw, FLAGS.kh, FLAGS.height, FLAGS.mem_size,
 344         lr / math.sqrt(FLAGS.num_replicas), min_length + 3, FLAGS.num_gpus,
 345         FLAGS.num_replicas, FLAGS.grad_noise_scale, max_sampling_rate,
 346         atrous=FLAGS.atrous, do_rnn=FLAGS.rnn_baseline,
 347         do_layer_norm=FLAGS.layer_norm, beam_size=cur_beam_size, backward=back)
 348   if sess is None:
 349     with tf.device(tf.train.replica_device_setter(FLAGS.ps_tasks)):
 350       model = make_ngpu(beam_size, True)
 351       if do_beam_model:
 352         tf.get_variable_scope().reuse_variables()
 353         beam_model = make_ngpu(FLAGS.beam_size, False)
 354   else:
 355     model = make_ngpu(beam_size, True)
 356     if do_beam_model:
 357       tf.get_variable_scope().reuse_variables()
 358       beam_model = make_ngpu(FLAGS.beam_size, False)
 359 
 360   sv = None
 361   if sess is None:
 362     # The supervisor configuration has a few overriden options.
 363     sv = tf.train.Supervisor(logdir=checkpoint_dir,
 364                              is_chief=(FLAGS.task < 1),
 365                              saver=model.saver,
 366                              summary_op=None,
 367                              save_summaries_secs=60,
 368                              save_model_secs=15 * 60,
 369                              global_step=model.global_step)
 370 
 371     config = tf.ConfigProto(allow_soft_placement=True)
 372     sess = sv.PrepareSession(FLAGS.master, config=config)
 373 
 374   data.print_out("Created model. Checkpoint dir %s" % checkpoint_dir)
 375 
 376   # Load model from parameters if a checkpoint exists.
 377   ckpt = tf.train.get_checkpoint_state(checkpoint_dir)
 378   if ckpt and tf.gfile.Exists(ckpt.model_checkpoint_path + ".index"):
 379     data.print_out("Reading model parameters from %s"
 380                    % ckpt.model_checkpoint_path)
 381     model.saver.restore(sess, ckpt.model_checkpoint_path)
 382   elif sv is None:
 383     sess.run(tf.global_variables_initializer())
 384     data.print_out("Initialized variables (no supervisor mode).")
 385   elif FLAGS.task < 1 and FLAGS.mem_size > 0:
 386     # sess.run(model.mem_norm_op)
 387     data.print_out("Created new model and normalized mem (on chief).")
 388 
 389   # Return the model and needed variables.
 390   return (model, beam_model, min_length, max_length, checkpoint_dir,
 391           (global_train_set, dev_set, en_path, fr_path), sv, sess)

 394 def m_step(model, beam_model, sess, batch_size, inp, target, bucket, nsteps, p):
 395   """Evaluation multi-step for program synthesis."""
 396   state, scores, hist = None, [[-11.0 for _ in xrange(batch_size)]], []
 397   for _ in xrange(nsteps):
 398     # Get the best beam (no training, just forward model).
 399     new_target, new_first, new_inp, new_scores = get_best_beam(
 400         beam_model, sess, inp, target,
 401         batch_size, FLAGS.beam_size, bucket, hist, p, test_mode=True)
 402     hist.append(new_first)
 403     _, _, _, state = model.step(sess, inp, new_target, False, state=state)
 404     inp = new_inp
 405     scores.append([max(scores[-1][i], new_scores[i])
 406                    for i in xrange(batch_size)])
 407   # The final step with the true target.
 408   loss, res, _, _ = model.step(sess, inp, target, False, state=state)
 409   return loss, res, new_target, scores[1:]

 412 def single_test(bin_id, model, sess, nprint, batch_size, dev, p, print_out=True,
 413                 offset=None, beam_model=None):
 414   """Test model on test data of length l using the given session."""
 415   if not dev[p][bin_id]:
 416     data.print_out("  bin %d (%d)\t%s\tppl NA errors NA seq-errors NA"
 417                    % (bin_id, data.bins[bin_id], p))
 418     return 1.0, 1.0, 0.0
 419   inpt, target = data.get_batch(
 420       bin_id, batch_size, dev[p], FLAGS.height, offset)
 421   if FLAGS.beam_size > 1 and beam_model:
 422     loss, res, new_tgt, scores = m_step(
 423         model, beam_model, sess, batch_size, inpt, target, bin_id,
 424         FLAGS.eval_beam_steps, p)
 425     score_avgs = [sum(s) / float(len(s)) for s in scores]
 426     score_maxs = [max(s) for s in scores]
 427     score_str = ["(%.2f, %.2f)" % (score_avgs[i], score_maxs[i])
 428                  for i in xrange(FLAGS.eval_beam_steps)]
 429     data.print_out("  == scores (avg, max): %s" % "; ".join(score_str))
 430     errors, total, seq_err = data.accuracy(inpt, res, target, batch_size,
 431                                            nprint, new_tgt, scores[-1])
 432   else:
 433     loss, res, _, _ = model.step(sess, inpt, target, False)
 434     errors, total, seq_err = data.accuracy(inpt, res, target, batch_size,
 435                                            nprint)
 436   seq_err = float(seq_err) / batch_size
 437   if total > 0:
 438     errors = float(errors) / total
 439   if print_out:
 440     data.print_out("  bin %d (%d)\t%s\tppl %.2f errors %.2f seq-errors %.2f"
 441                    % (bin_id, data.bins[bin_id], p, data.safe_exp(loss),
 442                       100 * errors, 100 * seq_err))
 443   return (errors, seq_err, loss)

 446 def assign_vectors(word_vector_file, embedding_key, vocab_path, sess):
 447   """Assign the embedding_key variable from the given word vectors file."""
 448   # For words in the word vector file, set their embedding at start.
 449   if not tf.gfile.Exists(word_vector_file):
 450     data.print_out("Word vector file does not exist: %s" % word_vector_file)
 451     sys.exit(1)
 452   vocab, _ = wmt.initialize_vocabulary(vocab_path)
 453   vectors_variable = [v for v in tf.trainable_variables()
 454                       if embedding_key == v.name]
 455   if len(vectors_variable) != 1:
 456     data.print_out("Word vector variable not found or too many.")
 457     sys.exit(1)
 458   vectors_variable = vectors_variable[0]
 459   vectors = vectors_variable.eval()
 460   data.print_out("Pre-setting word vectors from %s" % word_vector_file)
 461   with tf.gfile.GFile(word_vector_file, mode="r") as f:
 462     # Lines have format: dog 0.045123 -0.61323 0.413667 ...
 463     for line in f:
 464       line_parts = line.split()
 465       # The first part is the word.
 466       word = line_parts[0]
 467       if word in vocab:
 468         # Remaining parts are components of the vector.
 469         word_vector = np.array(map(float, line_parts[1:]))
 470         if len(word_vector) != FLAGS.vec_size:
 471           data.print_out("Warn: Word '%s', Expecting vector size %d, "
 472                          "found %d" % (word, FLAGS.vec_size,
 473                                        len(word_vector)))
 474         else:
 475           vectors[vocab[word]] = word_vector
 476   # Assign the modified vectors to the vectors_variable in the graph.
 477   sess.run([vectors_variable.initializer],
 478            {vectors_variable.initializer.inputs[1]: vectors})

 503 def get_bucket_id(train_buckets_scale_c, max_cur_length, data_set):
 504   """Get a random bucket id."""
 505   # Choose a bucket according to data distribution. Pick a random number
 506   # in [0, 1] and use the corresponding interval in train_buckets_scale.
 507   random_number_01 = np.random.random_sample()
 508   bucket_id = min([i for i in xrange(len(train_buckets_scale_c))
 509                    if train_buckets_scale_c[i] > random_number_01])
 510   while bucket_id > 0 and not data_set[bucket_id]:
 511     bucket_id -= 1
 512   for _ in xrange(10 if np.random.random_sample() < 0.9 else 1):
 513     if data.bins[bucket_id] > max_cur_length:
 514       random_number_01 = min(random_number_01, np.random.random_sample())
 515       bucket_id = min([i for i in xrange(len(train_buckets_scale_c))
 516                        if train_buckets_scale_c[i] > random_number_01])
 517       while bucket_id > 0 and not data_set[bucket_id]:
 518         bucket_id -= 1
 519   return bucket_id

 522 def score_beams(beams, target, inp, history, p,
 523                 print_out=False, test_mode=False):
 524   """Score beams."""
 525   if p == "progsynth":
 526     return score_beams_prog(beams, target, inp, history, print_out, test_mode)
 527   elif test_mode:
 528     return beams[0], 10.0 if str(beams[0][:len(target)]) == str(target) else 0.0
 529   else:
 530     history_s = [str(h) for h in history]
 531     best, best_score, tgt, eos_id = None, -1000.0, target, None
 532     if p == "wmt":
 533       eos_id = wmt.EOS_ID
 534     if eos_id and eos_id in target:
 535       tgt = target[:target.index(eos_id)]
 536     for beam in beams:
 537       if eos_id and eos_id in beam:
 538         beam = beam[:beam.index(eos_id)]
 539       l = min(len(tgt), len(beam))
 540       score = len([i for i in xrange(l) if tgt[i] == beam[i]]) / float(len(tgt))
 541       hist_score = 20.0 if str([b for b in beam if b > 0]) in history_s else 0.0
 542       if score < 1.0:
 543         score -= hist_score
 544       if score > best_score:
 545         best = beam
 546         best_score = score
 547     return best, best_score

 550 def score_beams_prog(beams, target, inp, history, print_out=False,
 551                      test_mode=False):
 552   """Score beams for program synthesis."""
 553   tgt_prog = linearize(target, program_utils.prog_vocab, True, 1)
 554   hist_progs = [linearize(h, program_utils.prog_vocab, True, 1)
 555                 for h in history]
 556   tgt_set = set(target)
 557   if print_out:
 558     print("target: ", tgt_prog)
 559   inps, tgt_outs = [], []
 560   for i in xrange(3):
 561     ilist = [inp[i + 1, l] for l in xrange(inp.shape[1])]
 562     clist = [program_utils.prog_vocab[x] for x in ilist if x > 0]
 563     olist = clist[clist.index("]") + 1:]  # outputs
 564     clist = clist[1:clist.index("]")]     # inputs
 565     inps.append([int(x) for x in clist])
 566     if olist[0] == "[":  # olist may be [int] or just int
 567       tgt_outs.append(str([int(x) for x in olist[1:-1]]))
 568     else:
 569       if len(olist) == 1:
 570         tgt_outs.append(olist[0])
 571       else:
 572         print([program_utils.prog_vocab[x] for x in ilist if x > 0])
 573         print(olist)
 574         print(tgt_prog)
 575         print(program_utils.evaluate(tgt_prog, {"a": inps[-1]}))
 576         print("AAAAA")
 577         tgt_outs.append(olist[0])
 578   if not test_mode:
 579     for _ in xrange(7):
 580       ilen = np.random.randint(len(target) - 3) + 1
 581       inps.append([random.choice(range(-15, 15)) for _ in range(ilen)])
 582     tgt_outs.extend([program_utils.evaluate(tgt_prog, {"a": inp})
 583                      for inp in inps[3:]])
 584   best, best_prog, best_score = None, "", -1000.0
 585   for beam in beams:
 586     b_prog = linearize(beam, program_utils.prog_vocab, True, 1)
 587     b_set = set(beam)
 588     jsim = len(tgt_set & b_set) / float(len(tgt_set | b_set))
 589     b_outs = [program_utils.evaluate(b_prog, {"a": inp}) for inp in inps]
 590     errs = len([x for x in b_outs if x == "ERROR"])
 591     imatches = len([i for i in xrange(3) if b_outs[i] == tgt_outs[i]])
 592     perfect = 10.0 if imatches == 3 else 0.0
 593     hist_score = 20.0 if b_prog in hist_progs else 0.0
 594     if test_mode:
 595       score = perfect - errs
 596     else:
 597       matches = len([i for i in xrange(10) if b_outs[i] == tgt_outs[i]])
 598       score = perfect + matches + jsim - errs
 599     if score < 10.0:
 600       score -= hist_score
 601     # print b_prog
 602     # print "jsim: ", jsim, " errs: ", errs, " mtchs: ", matches, " s: ", score
 603     if score > best_score:
 604       best = beam
 605       best_prog = b_prog
 606       best_score = score
 607   if print_out:
 608     print("best score: ", best_score, " best prog: ", best_prog)
 609   return best, best_score

 612 def get_best_beam(beam_model, sess, inp, target, batch_size, beam_size,
 613                   bucket, history, p, test_mode=False):
 614   """Run beam_model, score beams, and return the best as target and in input."""
 615   _, output_logits, _, _ = beam_model.step(
 616       sess, inp, target, None, beam_size=FLAGS.beam_size)
 617   new_targets, new_firsts, scores, new_inp = [], [], [], np.copy(inp)
 618   for b in xrange(batch_size):
 619     outputs = []
 620     history_b = [[h[b, 0, l] for l in xrange(data.bins[bucket])]
 621                  for h in history]
 622     for beam_idx in xrange(beam_size):
 623       outputs.append([int(o[beam_idx * batch_size + b])
 624                       for o in output_logits])
 625     target_t = [target[b, 0, l] for l in xrange(data.bins[bucket])]
 626     best, best_score = score_beams(
 627         outputs, [t for t in target_t if t > 0], inp[b, :, :],
 628         [[t for t in h if t > 0] for h in history_b], p, test_mode=test_mode)
 629     scores.append(best_score)
 630     if 1 in best:  # Only until _EOS.
 631       best = best[:best.index(1) + 1]
 632     best += [0 for _ in xrange(len(target_t) - len(best))]
 633     new_targets.append([best])
 634     first, _ = score_beams(
 635         outputs, [t for t in target_t if t > 0], inp[b, :, :],
 636         [[t for t in h if t > 0] for h in history_b], p, test_mode=True)
 637     if 1 in first:  # Only until _EOS.
 638       first = first[:first.index(1) + 1]
 639     first += [0 for _ in xrange(len(target_t) - len(first))]
 640     new_inp[b, 0, :] = np.array(first, dtype=np.int32)
 641     new_firsts.append([first])
 642   # Change target if we found a great answer.
 643   new_target = np.array(new_targets, dtype=np.int32)
 644   for b in xrange(batch_size):
 645     if scores[b] >= 10.0:
 646       target[b, 0, :] = new_target[b, 0, :]
 647   new_first = np.array(new_firsts, dtype=np.int32)
 648   return new_target, new_first, new_inp, scores

 651 def train():
 652   """Train the model."""
 653   batch_size = FLAGS.batch_size * FLAGS.num_gpus
 654   (model, beam_model, min_length, max_length, checkpoint_dir,
 655    (train_set, dev_set, en_vocab_path, fr_vocab_path), sv, sess) = initialize()
 656   with sess.as_default():
 657     quant_op = model.quantize_op
 658     max_cur_length = min(min_length + 3, max_length)
 659     prev_acc_perp = [1000000 for _ in xrange(5)]
 660     prev_seq_err = 1.0
 661     is_chief = FLAGS.task < 1
 662     do_report = False
 663 
 664     # Main traning loop.
 665     while not sv.ShouldStop():
 666       global_step, max_cur_length, learning_rate = sess.run(
 667           [model.global_step, model.cur_length, model.lr])
 668       acc_loss, acc_l1, acc_total, acc_errors, acc_seq_err = 0.0, 0.0, 0, 0, 0
 669       acc_grad_norm, step_count, step_c1, step_time = 0.0, 0, 0, 0.0
 670 
 671       # For words in the word vector file, set their embedding at start.
 672       bound1 = FLAGS.steps_per_checkpoint - 1
 673       if FLAGS.word_vector_file_en and global_step < bound1 and is_chief:
 674         assign_vectors(FLAGS.word_vector_file_en, "embedding:0",
 675                        en_vocab_path, sess)
 676         if FLAGS.max_target_vocab < 1:
 677           assign_vectors(FLAGS.word_vector_file_en, "target_embedding:0",
 678                          en_vocab_path, sess)
 679 
 680       if FLAGS.word_vector_file_fr and global_step < bound1 and is_chief:
 681         assign_vectors(FLAGS.word_vector_file_fr, "embedding:0",
 682                        fr_vocab_path, sess)
 683         if FLAGS.max_target_vocab < 1:
 684           assign_vectors(FLAGS.word_vector_file_fr, "target_embedding:0",
 685                          fr_vocab_path, sess)
 686 
 687       for _ in xrange(FLAGS.steps_per_checkpoint):
 688         step_count += 1
 689         step_c1 += 1
 690         global_step = int(model.global_step.eval())
 691         train_beam_anneal = global_step / float(FLAGS.train_beam_anneal)
 692         train_beam_freq = FLAGS.train_beam_freq * min(1.0, train_beam_anneal)
 693         p = random.choice(FLAGS.problem.split("-"))
 694         train_set = global_train_set[p][-1]
 695         bucket_id = get_bucket_id(train_buckets_scale[p][-1], max_cur_length,
 696                                   train_set)
 697         # Prefer longer stuff 60% of time if not wmt.
 698         if np.random.randint(100) < 60 and FLAGS.problem != "wmt":
 699           bucket1 = get_bucket_id(train_buckets_scale[p][-1], max_cur_length,
 700                                   train_set)
 701           bucket_id = max(bucket1, bucket_id)
 702 
 703         # Run a step and time it.
 704         start_time = time.time()
 705         inp, target = data.get_batch(bucket_id, batch_size, train_set,
 706                                      FLAGS.height)
 707         noise_param = math.sqrt(math.pow(global_step + 1, -0.55) *
 708                                 prev_seq_err) * FLAGS.grad_noise_scale
 709         # In multi-step mode, we use best from beam for middle steps.
 710         state, new_target, scores, history = None, None, None, []
 711         while (FLAGS.beam_size > 1 and
 712                train_beam_freq > np.random.random_sample()):
 713           # Get the best beam (no training, just forward model).
 714           new_target, new_first, new_inp, scores = get_best_beam(
 715               beam_model, sess, inp, target,
 716               batch_size, FLAGS.beam_size, bucket_id, history, p)
 717           history.append(new_first)
 718           # Training step with the previous input and the best beam as target.
 719           _, _, _, state = model.step(sess, inp, new_target, FLAGS.do_train,
 720                                       noise_param, update_mem=True, state=state)
 721           # Change input to the new one for the next step.
 722           inp = new_inp
 723           # If all results are great, stop (todo: not to wait for all?).
 724           if FLAGS.nprint > 1:
 725             print(scores)
 726           if sum(scores) / float(len(scores)) >= 10.0:
 727             break
 728         # The final step with the true target.
 729         loss, res, gnorm, _ = model.step(
 730             sess, inp, target, FLAGS.do_train, noise_param,
 731             update_mem=True, state=state)
 732         step_time += time.time() - start_time
 733         acc_grad_norm += 0.0 if gnorm is None else float(gnorm)
 734 
 735         # Accumulate statistics.
 736         acc_loss += loss
 737         acc_l1 += loss
 738         errors, total, seq_err = data.accuracy(
 739             inp, res, target, batch_size, 0, new_target, scores)
 740         if FLAGS.nprint > 1:
 741           print("seq_err: ", seq_err)
 742         acc_total += total
 743         acc_errors += errors
 744         acc_seq_err += seq_err
 745 
 746         # Report summary every 10 steps.
 747         if step_count + 3 > FLAGS.steps_per_checkpoint:
 748           do_report = True  # Don't polute plot too early.
 749         if is_chief and step_count % 10 == 1 and do_report:
 750           cur_loss = acc_l1 / float(step_c1)
 751           acc_l1, step_c1 = 0.0, 0
 752           cur_perp = data.safe_exp(cur_loss)
 753           summary = tf.Summary()
 754           summary.value.extend(
 755               [tf.Summary.Value(tag="log_perplexity", simple_value=cur_loss),
 756                tf.Summary.Value(tag="perplexity", simple_value=cur_perp)])
 757           sv.SummaryComputed(sess, summary, global_step)
 758 
 759       # Normalize and print out accumulated statistics.
 760       acc_loss /= step_count
 761       step_time /= FLAGS.steps_per_checkpoint
 762       acc_seq_err = float(acc_seq_err) / (step_count * batch_size)
 763       prev_seq_err = max(0.0, acc_seq_err - 0.02)  # No noise at error < 2%.
 764       acc_errors = float(acc_errors) / acc_total if acc_total > 0 else 1.0
 765       t_size = float(sum([len(x) for x in train_set])) / float(1000000)
 766       msg = ("step %d step-time %.2f train-size %.3f lr %.6f grad-norm %.4f"
 767              % (global_step + 1, step_time, t_size, learning_rate,
 768                 acc_grad_norm / FLAGS.steps_per_checkpoint))
 769       data.print_out("%s len %d ppl %.6f errors %.2f sequence-errors %.2f" %
 770                      (msg, max_cur_length, data.safe_exp(acc_loss),
 771                       100*acc_errors, 100*acc_seq_err))
 772 
 773       # If errors are below the curriculum threshold, move curriculum forward.
 774       is_good = FLAGS.curriculum_ppx > data.safe_exp(acc_loss)
 775       is_good = is_good and FLAGS.curriculum_seq > acc_seq_err
 776       if is_good and is_chief:
 777         if FLAGS.quantize:
 778           # Quantize weights.
 779           data.print_out("  Quantizing parameters.")
 780           sess.run([quant_op])
 781         # Increase current length (until the next with training data).
 782         sess.run(model.cur_length_incr_op)
 783         # Forget last perplexities if we're not yet at the end.
 784         if max_cur_length < max_length:
 785           prev_acc_perp.append(1000000)
 786 
 787       # Lower learning rate if we're worse than the last 5 checkpoints.
 788       acc_perp = data.safe_exp(acc_loss)
 789       if acc_perp > max(prev_acc_perp[-5:]) and is_chief:
 790         sess.run(model.lr_decay_op)
 791       prev_acc_perp.append(acc_perp)
 792 
 793       # Save checkpoint.
 794       if is_chief:
 795         checkpoint_path = os.path.join(checkpoint_dir, "neural_gpu.ckpt")
 796         model.saver.save(sess, checkpoint_path,
 797                          global_step=model.global_step)
 798 
 799         # Run evaluation.
 800         bin_bound = 4
 801         for p in FLAGS.problem.split("-"):
 802           total_loss, total_err, tl_counter = 0.0, 0.0, 0
 803           for bin_id in xrange(len(data.bins)):
 804             if bin_id < bin_bound or bin_id % FLAGS.eval_bin_print == 1:
 805               err, _, loss = single_test(bin_id, model, sess, FLAGS.nprint,
 806                                          batch_size * 4, dev_set, p,
 807                                          beam_model=beam_model)
 808               if loss > 0.0:
 809                 total_loss += loss
 810                 total_err += err
 811                 tl_counter += 1
 812           test_loss = total_loss / max(1, tl_counter)
 813           test_err = total_err / max(1, tl_counter)
 814           test_perp = data.safe_exp(test_loss)
 815           summary = tf.Summary()
 816           summary.value.extend(
 817               [tf.Summary.Value(tag="test/%s/loss" % p, simple_value=test_loss),
 818                tf.Summary.Value(tag="test/%s/error" % p, simple_value=test_err),
 819                tf.Summary.Value(tag="test/%s/perplexity" % p,
 820                                 simple_value=test_perp)])
 821           sv.SummaryComputed(sess, summary, global_step)
 822

 824 def linearize(output, rev_fr_vocab, simple_tokenizer=None, eos_id=wmt.EOS_ID):
 825   # If there is an EOS symbol in outputs, cut them at that point (WMT).
 826   if eos_id in output:
 827     output = output[:output.index(eos_id)]
 828   # Print out French sentence corresponding to outputs.
 829   if simple_tokenizer or FLAGS.simple_tokenizer:
 830     vlen = len(rev_fr_vocab)
 831     def vget(o):
 832       if o < vlen:
 833         return rev_fr_vocab[o]
 834       return "UNK"
 835     return " ".join([vget(o) for o in output])
 836   else:
 837     return wmt.basic_detokenizer([rev_fr_vocab[o] for o in output])

 840 def evaluate():
 841   """Evaluate an existing model."""
 842   batch_size = FLAGS.batch_size * FLAGS.num_gpus
 843   with tf.Session(config=tf.ConfigProto(allow_soft_placement=True)) as sess:
 844     (model, beam_model, _, _, _,
 845      (_, dev_set, en_vocab_path, fr_vocab_path), _, sess) = initialize(sess)
 846     for p in FLAGS.problem.split("-"):
 847       for bin_id in xrange(len(data.bins)):
 848         if (FLAGS.task >= 0 and bin_id > 4) or (FLAGS.nprint == 0 and
 849                                                 bin_id > 8 and p == "wmt"):
 850           break
 851         single_test(bin_id, model, sess, FLAGS.nprint, batch_size, dev_set, p,
 852                     beam_model=beam_model)
 853     path = FLAGS.test_file_prefix
 854     xid = "" if FLAGS.task < 0 else ("%.4d" % (FLAGS.task+FLAGS.decode_offset))
 855     en_path, fr_path = path + ".en" + xid, path + ".fr" + xid
 856     # Evaluate the test file if they exist.
 857     if path and tf.gfile.Exists(en_path) and tf.gfile.Exists(fr_path):
 858       data.print_out("Translating test set %s" % en_path)
 859       # Read lines.
 860       en_lines, fr_lines = [], []
 861       with tf.gfile.GFile(en_path, mode="r") as f:
 862         for line in f:
 863           en_lines.append(line.strip())
 864       with tf.gfile.GFile(fr_path, mode="r") as f:
 865         for line in f:
 866           fr_lines.append(line.strip())
 867       # Tokenize and convert to ids.
 868       en_vocab, _ = wmt.initialize_vocabulary(en_vocab_path)
 869       _, rev_fr_vocab = wmt.initialize_vocabulary(fr_vocab_path)
 870       if FLAGS.simple_tokenizer:
 871         en_ids = [wmt.sentence_to_token_ids(
 872             l, en_vocab, tokenizer=wmt.space_tokenizer,
 873             normalize_digits=FLAGS.normalize_digits)
 874                   for l in en_lines]
 875       else:
 876         en_ids = [wmt.sentence_to_token_ids(l, en_vocab) for l in en_lines]
 877       # Translate.
 878       results = []
 879       for idx, token_ids in enumerate(en_ids):
 880         if idx % 5 == 0:
 881           data.print_out("Translating example %d of %d." % (idx, len(en_ids)))
 882         # Which bucket does it belong to?
 883         buckets = [b for b in xrange(len(data.bins))
 884                    if data.bins[b] >= len(token_ids)]
 885         if buckets:
 886           result, result_cost = [], 100000000.0
 887           for bucket_id in buckets:
 888             if data.bins[bucket_id] > MAXLEN_F * len(token_ids) + EVAL_LEN_INCR:
 889               break
 890             # Get a 1-element batch to feed the sentence to the model.
 891             used_batch_size = 1  # batch_size
 892             inp, target = data.get_batch(
 893                 bucket_id, used_batch_size, None, FLAGS.height,
 894                 preset=([token_ids], [[]]))
 895             loss, output_logits, _, _ = model.step(
 896                 sess, inp, target, None, beam_size=FLAGS.beam_size)
 897             outputs = [int(o[0]) for o in output_logits]
 898             loss = loss[0] - (data.bins[bucket_id] * FLAGS.length_norm)
 899             if FLAGS.simple_tokenizer:
 900               cur_out = outputs
 901               if wmt.EOS_ID in cur_out:
 902                 cur_out = cur_out[:cur_out.index(wmt.EOS_ID)]
 903               res_tags = [rev_fr_vocab[o] for o in cur_out]
 904               bad_words, bad_brack = wmt.parse_constraints(token_ids, res_tags)
 905               loss += 1000.0 * bad_words + 100.0 * bad_brack
 906             # print (bucket_id, loss)
 907             if loss < result_cost:
 908               result = outputs
 909               result_cost = loss
 910           final = linearize(result, rev_fr_vocab)
 911           results.append("%s\t%s\n" % (final, fr_lines[idx]))
 912           # print result_cost
 913           sys.stderr.write(results[-1])
 914           sys.stderr.flush()
 915         else:
 916           sys.stderr.write("TOOO_LONG\t%s\n" % fr_lines[idx])
 917           sys.stderr.flush()

 971         buckets = [buckets[0]]
 972       if buckets:
