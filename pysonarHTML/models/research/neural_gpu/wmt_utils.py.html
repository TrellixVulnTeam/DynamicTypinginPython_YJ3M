  27 # Special vocabulary symbols - we always put them at the start.
  28 _PAD = b"_PAD"
  29 _GO = b"_GO"
  30 _EOS = b"_EOS"
  31 _UNK = b"_CHAR_UNK"
  32 _SPACE = b"_SPACE"
  33 _START_VOCAB = [_PAD, _GO, _EOS, _UNK, _SPACE]
  34 
  35 PAD_ID = 0
  36 GO_ID = 1
  37 EOS_ID = 2
  38 UNK_ID = 3
  39 SPACE_ID = 4

  41 # Regular expressions used to tokenize.
  42 _CHAR_MARKER = "_CHAR_"
  43 _CHAR_MARKER_LEN = len(_CHAR_MARKER)
  44 _SPEC_CHARS = "" + chr(226) + chr(153) + chr(128)
  45 _PUNCTUATION = "][.,!?\"':;%$#@&*+}{|><=/^~)(_`,0123456789" + _SPEC_CHARS + "-"
  46 _WORD_SPLIT = re.compile("([" + _PUNCTUATION + "])")
  47 _OLD_WORD_SPLIT = re.compile(b"([.,!?\"':;)(])")
  48 _DIGIT_RE = re.compile(br"\d")

  55 def maybe_download(directory, filename, url):
  56   """Download filename from url unless it's already in directory."""
  57   if not tf.gfile.Exists(directory):
  58     print("Creating directory %s" % directory)
  59     os.mkdir(directory)
  60   filepath = os.path.join(directory, filename)
  61   if not tf.gfile.Exists(filepath):
  62     print("Downloading %s to %s" % (url, filepath))
  63     filepath, _ = urllib.request.urlretrieve(url, filepath)
  64     statinfo = os.stat(filepath)
  65     print("Successfully downloaded", filename, statinfo.st_size, "bytes")
  66   return filepath

  69 def gunzip_file(gz_path, new_path):
  70   """Unzips from gz_path into new_path."""
  71   print("Unpacking %s to %s" % (gz_path, new_path))
  72   with gzip.open(gz_path, "rb") as gz_file:
  73     with open(new_path, "wb") as new_file:
  74       for line in gz_file:
  75         new_file.write(line)

  78 def get_wmt_enfr_train_set(directory):
  79   """Download the WMT en-fr training corpus to directory unless it's there."""
  80   train_path = os.path.join(directory, "giga-fren.release2.fixed")
  81   if not (tf.gfile.Exists(train_path +".fr") and
  82           tf.gfile.Exists(train_path +".en")):
  83     corpus_file = maybe_download(directory, "training-giga-fren.tar",
  84                                  _WMT_ENFR_TRAIN_URL)
  85     print("Extracting tar file %s" % corpus_file)
  86     with tarfile.open(corpus_file, "r") as corpus_tar:
  87       corpus_tar.extractall(directory)
  88     gunzip_file(train_path + ".fr.gz", train_path + ".fr")
  89     gunzip_file(train_path + ".en.gz", train_path + ".en")
  90   return train_path

  93 def get_wmt_enfr_dev_set(directory):
  94   """Download the WMT en-fr training corpus to directory unless it's there."""
  95   dev_name = "newstest2013"
  96   dev_path = os.path.join(directory, dev_name)
  97   if not (tf.gfile.Exists(dev_path + ".fr") and
  98           tf.gfile.Exists(dev_path + ".en")):
  99     dev_file = maybe_download(directory, "dev-v2.tgz", _WMT_ENFR_DEV_URL)
 100     print("Extracting tgz file %s" % dev_file)
 101     with tarfile.open(dev_file, "r:gz") as dev_tar:
 102       fr_dev_file = dev_tar.getmember("dev/" + dev_name + ".fr")
 103       en_dev_file = dev_tar.getmember("dev/" + dev_name + ".en")
 104       fr_dev_file.name = dev_name + ".fr"  # Extract without "dev/" prefix.
 105       en_dev_file.name = dev_name + ".en"
 106       dev_tar.extract(fr_dev_file, directory)
 107       dev_tar.extract(en_dev_file, directory)
 108   return dev_path

 111 def is_char(token):
 112   if len(token) > _CHAR_MARKER_LEN:
 113     if token[:_CHAR_MARKER_LEN] == _CHAR_MARKER:
 114       return True
 115   return False

 118 def basic_detokenizer(tokens):
 119   """Reverse the process of the basic tokenizer below."""
 120   result = []
 121   previous_nospace = True
 122   for t in tokens:
 123     if is_char(t):
 124       result.append(t[_CHAR_MARKER_LEN:])
 125       previous_nospace = True
 126     elif t == _SPACE:
 127       result.append(" ")
 128       previous_nospace = True
 129     elif previous_nospace:
 130       result.append(t)
 131       previous_nospace = False
 132     else:
 133       result.extend([" ", t])
 134       previous_nospace = False
 135   return "".join(result)

 141 def basic_tokenizer(sentence):
 142   """Very basic tokenizer: split the sentence into a list of tokens."""
 143   words = []
 144   if old_style:
 145     for space_separated_fragment in sentence.strip().split():
 146       words.extend(re.split(_OLD_WORD_SPLIT, space_separated_fragment))
 147     return [w for w in words if w]
 148   for space_separated_fragment in sentence.strip().split():
 149     tokens = [t for t in re.split(_WORD_SPLIT, space_separated_fragment) if t]
 150     first_is_char = False
 151     for i, t in enumerate(tokens):
 152       if len(t) == 1 and t in _PUNCTUATION:
 153         tokens[i] = _CHAR_MARKER + t
 154         if i == 0:
 155           first_is_char = True
 156     if words and words[-1] != _SPACE and (first_is_char or is_char(words[-1])):
 157       tokens = [_SPACE] + tokens
 158     spaced_tokens = []
 159     for i, tok in enumerate(tokens):
 160       spaced_tokens.append(tokens[i])
 161       if i < len(tokens) - 1:
 162         if tok != _SPACE and not (is_char(tok) or is_char(tokens[i+1])):
 163           spaced_tokens.append(_SPACE)
 164     words.extend(spaced_tokens)
 165   return words

 172 def is_pos_tag(token):
 173   """Check if token is a part-of-speech tag."""
 174   return(token in ["CC", "CD", "DT", "EX", "FW", "IN", "JJ", "JJR",
 175                    "JJS", "LS", "MD", "NN", "NNS", "NNP", "NNPS", "PDT",
 176                    "POS", "PRP", "PRP$", "RB", "RBR", "RBS", "RP", "SYM", "TO",
 177                    "UH", "VB", "VBD", "VBG", "VBN", "VBP", "VBZ", "WDT", "WP",
 178                    "WP$", "WRB", ".", ",", ":", ")", "-LRB-", "(", "-RRB-",
 179                    "HYPH", "$", "``", "''", "ADD", "AFX", "QTR", "BES", "-DFL-",
 180                    "GW", "HVS", "NFP"])

 183 def parse_constraints(inpt, res):
 184   ntags = len(res)
 185   nwords = len(inpt)
 186   npostags = len([x for x in res if is_pos_tag(x)])
 187   nclose = len([x for x in res if x[0] == "/"])
 188   nopen = ntags - nclose - npostags
 189   return (abs(npostags - nwords), abs(nclose - nopen))

 192 def create_vocabulary(vocabulary_path, data_path, max_vocabulary_size,
 193                       tokenizer=None, normalize_digits=False):
 194   """Create vocabulary file (if it does not exist yet) from data file.
 195 
 196   Data file is assumed to contain one sentence per line. Each sentence is
 197   tokenized and digits are normalized (if normalize_digits is set).
 198   Vocabulary contains the most-frequent tokens up to max_vocabulary_size.
 199   We write it to vocabulary_path in a one-token-per-line format, so that later
 200   token in the first line gets id=0, second line gets id=1, and so on.
 201 
 202   Args:
 203     vocabulary_path: path where the vocabulary will be created.
 204     data_path: data file that will be used to create vocabulary.
 205     max_vocabulary_size: limit on the size of the created vocabulary.
 206     tokenizer: a function to use to tokenize each data sentence;
 207       if None, basic_tokenizer will be used.
 208     normalize_digits: Boolean; if true, all digits are replaced by 0s.
 209   """
 210   if not tf.gfile.Exists(vocabulary_path):
 211     print("Creating vocabulary %s from data %s" % (vocabulary_path, data_path))
 212     vocab, chars = {}, {}
 213     for c in _PUNCTUATION:
 214       chars[c] = 1
 215 
 216     # Read French file.
 217     with tf.gfile.GFile(data_path + ".fr", mode="rb") as f:
 218       counter = 0
 219       for line_in in f:
 220         line = " ".join(line_in.split())
 221         counter += 1
 222         if counter % 100000 == 0:
 223           print("  processing fr line %d" % counter)
 224         for c in line:
 225           if c in chars:
 226             chars[c] += 1
 227           else:
 228             chars[c] = 1
 229         tokens = tokenizer(line) if tokenizer else basic_tokenizer(line)
 230         tokens = [t for t in tokens if not is_char(t) and t != _SPACE]
 231         for w in tokens:
 232           word = re.sub(_DIGIT_RE, b"0", w) if normalize_digits else w
 233           if word in vocab:
 234             vocab[word] += 1000000000  # We want target words first.
 235           else:
 236             vocab[word] = 1000000000
 237 
 238     # Read English file.
 239     with tf.gfile.GFile(data_path + ".en", mode="rb") as f:
 240       counter = 0
 241       for line_in in f:
 242         line = " ".join(line_in.split())
 243         counter += 1
 244         if counter % 100000 == 0:
 245           print("  processing en line %d" % counter)
 246         for c in line:
 247           if c in chars:
 248             chars[c] += 1
 249           else:
 250             chars[c] = 1
 251         tokens = tokenizer(line) if tokenizer else basic_tokenizer(line)
 252         tokens = [t for t in tokens if not is_char(t) and t != _SPACE]
 253         for w in tokens:
 254           word = re.sub(_DIGIT_RE, b"0", w) if normalize_digits else w
 255           if word in vocab:
 256             vocab[word] += 1
 257           else:
 258             vocab[word] = 1
 259 
 260       sorted_vocab = sorted(vocab, key=vocab.get, reverse=True)
 261       sorted_chars = sorted(chars, key=vocab.get, reverse=True)
 262       sorted_chars = [_CHAR_MARKER + c for c in sorted_chars]
 263       vocab_list = _START_VOCAB + sorted_chars + sorted_vocab
 264       if tokenizer:
 265         vocab_list = _START_VOCAB + sorted_vocab
 266       if len(vocab_list) > max_vocabulary_size:
 267         vocab_list = vocab_list[:max_vocabulary_size]
 268       with tf.gfile.GFile(vocabulary_path, mode="wb") as vocab_file:
 269         for w in vocab_list:
 270           vocab_file.write(w + b"\n")

 273 def initialize_vocabulary(vocabulary_path):
 274   """Initialize vocabulary from file.
 275 
 276   We assume the vocabulary is stored one-item-per-line, so a file:
 277     dog
 278     cat
 279   will result in a vocabulary {"dog": 0, "cat": 1}, and this function will
 280   also return the reversed-vocabulary ["dog", "cat"].
 281 
 282   Args:
 283     vocabulary_path: path to the file containing the vocabulary.
 284 
 285   Returns:
 286     a pair: the vocabulary (a dictionary mapping string to integers), and
 287     the reversed vocabulary (a list, which reverses the vocabulary mapping).
 288 
 289   Raises:
 290     ValueError: if the provided vocabulary_path does not exist.
 291   """
 292   if tf.gfile.Exists(vocabulary_path):
 293     rev_vocab = []
 294     with tf.gfile.GFile(vocabulary_path, mode="rb") as f:
 295       rev_vocab.extend(f.readlines())
 296     rev_vocab = [line.strip() for line in rev_vocab]
 297     vocab = dict([(x, y) for (y, x) in enumerate(rev_vocab)])
 298     return vocab, rev_vocab
 299   else:
 300     raise ValueError("Vocabulary file %s not found.", vocabulary_path)

 303 def sentence_to_token_ids_raw(sentence, vocabulary,
 304                               tokenizer=None, normalize_digits=old_style):
 305   """Convert a string to list of integers representing token-ids.
 306 
 307   For example, a sentence "I have a dog" may become tokenized into
 308   ["I", "have", "a", "dog"] and with vocabulary {"I": 1, "have": 2,
 309   "a": 4, "dog": 7"} this function will return [1, 2, 4, 7].
 310 
 311   Args:
 312     sentence: the sentence in bytes format to convert to token-ids.
 313     vocabulary: a dictionary mapping tokens to integers.
 314     tokenizer: a function to use to tokenize each sentence;
 315       if None, basic_tokenizer will be used.
 316     normalize_digits: Boolean; if true, all digits are replaced by 0s.
 317 
 318   Returns:
 319     a list of integers, the token-ids for the sentence.
 320   """
 321   if tokenizer:
 322     words = tokenizer(sentence)
 323   else:
 324     words = basic_tokenizer(sentence)
 325   result = []
 326   for w in words:
 327     if normalize_digits:
 328       w = re.sub(_DIGIT_RE, b"0", w)
 329     if w in vocabulary:
 330       result.append(vocabulary[w])
 331     else:
 332       if tokenizer:
 333         result.append(UNK_ID)
 334       else:
 335         result.append(SPACE_ID)
 336         for c in w:
 337           result.append(vocabulary.get(_CHAR_MARKER + c, UNK_ID))
 338         result.append(SPACE_ID)
 339   while result and result[0] == SPACE_ID:
 340     result = result[1:]
 341   while result and result[-1] == SPACE_ID:
 342     result = result[:-1]
 343   return result

 346 def sentence_to_token_ids(sentence, vocabulary,
 347                           tokenizer=None, normalize_digits=old_style):
 348   """Convert a string to list of integers representing token-ids, tab=0."""
 349   tab_parts = sentence.strip().split("\t")
 350   toks = [sentence_to_token_ids_raw(t, vocabulary, tokenizer, normalize_digits)
 351           for t in tab_parts]
 352   res = []
 353   for t in toks:
 354     res.extend(t)
 355     res.append(0)
 356   return res[:-1]

 359 def data_to_token_ids(data_path, target_path, vocabulary_path,
 360                       tokenizer=None, normalize_digits=False):
 361   """Tokenize data file and turn into token-ids using given vocabulary file.
 362 
 363   This function loads data line-by-line from data_path, calls the above
 364   sentence_to_token_ids, and saves the result to target_path. See comment
 365   for sentence_to_token_ids on the details of token-ids format.
 366 
 367   Args:
 368     data_path: path to the data file in one-sentence-per-line format.
 369     target_path: path where the file with token-ids will be created.
 370     vocabulary_path: path to the vocabulary file.
 371     tokenizer: a function to use to tokenize each sentence;
 372       if None, basic_tokenizer will be used.
 373     normalize_digits: Boolean; if true, all digits are replaced by 0s.
 374   """
 375   if not tf.gfile.Exists(target_path):
 376     print("Tokenizing data in %s" % data_path)
 377     vocab, _ = initialize_vocabulary(vocabulary_path)
 378     with tf.gfile.GFile(data_path, mode="rb") as data_file:
 379       with tf.gfile.GFile(target_path, mode="w") as tokens_file:
 380         counter = 0
 381         for line in data_file:
 382           counter += 1
 383           if counter % 100000 == 0:
 384             print("  tokenizing line %d" % counter)
 385           token_ids = sentence_to_token_ids(line, vocab, tokenizer,
 386                                             normalize_digits)
 387           tokens_file.write(" ".join([str(tok) for tok in token_ids]) + "\n")

 390 def prepare_wmt_data(data_dir, vocabulary_size,
 391                      tokenizer=None, normalize_digits=False):
 392   """Get WMT data into data_dir, create vocabularies and tokenize data.
 393 
 394   Args:
 395     data_dir: directory in which the data sets will be stored.
 396     vocabulary_size: size of the joint vocabulary to create and use.
 397     tokenizer: a function to use to tokenize each data sentence;
 398       if None, basic_tokenizer will be used.
 399     normalize_digits: Boolean; if true, all digits are replaced by 0s.
 400 
 401   Returns:
 402     A tuple of 6 elements:
 403       (1) path to the token-ids for English training data-set,
 404       (2) path to the token-ids for French training data-set,
 405       (3) path to the token-ids for English development data-set,
 406       (4) path to the token-ids for French development data-set,
 407       (5) path to the vocabulary file,
 408       (6) path to the vocabulary file (for compatibility with non-joint vocab).
 409   """
 410   # Get wmt data to the specified directory.
 411   train_path = get_wmt_enfr_train_set(data_dir)
 412   dev_path = get_wmt_enfr_dev_set(data_dir)
 413 
 414   # Create vocabularies of the appropriate sizes.
 415   vocab_path = os.path.join(data_dir, "vocab%d.txt" % vocabulary_size)
 416   create_vocabulary(vocab_path, train_path, vocabulary_size,
 417                     tokenizer=tokenizer, normalize_digits=normalize_digits)
 418 
 419   # Create token ids for the training data.
 420   fr_train_ids_path = train_path + (".ids%d.fr" % vocabulary_size)
 421   en_train_ids_path = train_path + (".ids%d.en" % vocabulary_size)
 422   data_to_token_ids(train_path + ".fr", fr_train_ids_path, vocab_path,
 423                     tokenizer=tokenizer, normalize_digits=normalize_digits)
 424   data_to_token_ids(train_path + ".en", en_train_ids_path, vocab_path,
 425                     tokenizer=tokenizer, normalize_digits=normalize_digits)
 426 
 427   # Create token ids for the development data.
 428   fr_dev_ids_path = dev_path + (".ids%d.fr" % vocabulary_size)
 429   en_dev_ids_path = dev_path + (".ids%d.en" % vocabulary_size)
 430   data_to_token_ids(dev_path + ".fr", fr_dev_ids_path, vocab_path,
 431                     tokenizer=tokenizer, normalize_digits=normalize_digits)
 432   data_to_token_ids(dev_path + ".en", en_dev_ids_path, vocab_path,
 433                     tokenizer=tokenizer, normalize_digits=normalize_digits)
 434 
 435   return (en_train_ids_path, fr_train_ids_path,
 436           en_dev_ids_path, fr_dev_ids_path,
 437           vocab_path, vocab_path)
