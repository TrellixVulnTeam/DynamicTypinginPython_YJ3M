  27 class ListType(object):
  28   def __init__(self, arg):
  29     self.arg = arg
  30 
  31   def __str__(self):
  32     return "[" + str(self.arg) + "]"
  33 
  34   def __eq__(self, other):
  35     if not isinstance(other, ListType):
  36       return False
  37     return self.arg == other.arg
  38 
  39   def __hash__(self):
  40     return hash(self.arg)

  57 class FunctionType(object):
  58   def __init__(self, args):
  59     self.args = args
  60 
  61   def __str__(self):
  62     return str(self.args[0]) + " -> " + str(self.args[1])
  63 
  64   def __eq__(self, other):
  65     if not isinstance(other, FunctionType):
  66       return False
  67     return self.args == other.args
  68 
  69   def __hash__(self):
  70     return hash(tuple(self.args))

 113 ## Lambdas
 114 # Int -> Int
 115 def plus_one(x): return x + 1
 116 def minus_one(x): return x - 1
 117 def times_two(x): return x * 2
 118 def neg(x): return x * (-1)
 119 def div_two(x): return int(x/2)
 120 def sq(x): return x**2
 121 def times_three(x): return x * 3
 122 def div_three(x): return int(x/3)
 123 def times_four(x): return x * 4
 124 def div_four(x): return int(x/4)
 125 
 126 # Int -> Bool
 127 def pos(x): return x > 0
 128 def neg(x): return x < 0
 129 def even(x): return x%2 == 0
 130 def odd(x): return x%2 == 1
 131 
 132 # Int -> Int -> Int
 133 def add(x, y): return x + y
 134 def sub(x, y): return x - y
 135 def mul(x, y): return x * y

 154 f_scan = Function("c_scan", [ListType("Int")],
 155                             ListType("Int"),
 156                             [FunctionType(["Int", "Int", "Int"])])
 157 def c_scan(f, xs):
 158   out = xs
 159   for i in range(1, len(xs)):
 160     out[i] = f(xs[i], xs[i -1])
 161   return out

 163 @contextlib.contextmanager
 164 def stdoutIO(stdout=None):
 165   old = sys.stdout
 166   if stdout is None:
 167     stdout = StringIO.StringIO()
 168   sys.stdout = stdout
 169   yield stdout
 170   sys.stdout = old

 173 def evaluate(program_str, input_names_to_vals, default="ERROR"):
 174   exec_str = []
 175   for name, val in input_names_to_vals.iteritems():
 176     exec_str += name + " = " + str(val) + "; "
 177   exec_str += program_str
 178   if type(exec_str) is list:
 179     exec_str = "".join(exec_str)
 180 
 181   with stdoutIO() as s:
 182     # pylint: disable=bare-except
 183     try:
 184       exec(exec_str + " print(out)")
 185       return s.getvalue()[:-1]
 186     except:
 187       return default
 188    # pylint: enable=bare-except

 191 class Statement(object):
 192   """Statement class."""
 193 
 194   def __init__(self, fn, output_var, arg_vars, fn_args=None):
 195     self.fn = fn
 196     self.output_var = output_var
 197     self.arg_vars = arg_vars
 198     self.fn_args = fn_args or []
 199 
 200   def __str__(self):
 201     return "%s = %s(%s%s%s)"%(self.output_var,
 202                               self.fn.name,
 203                               ", ".join(self.fn_args),
 204                               ", " if self.fn_args else "",
 205                               ", ".join(self.arg_vars))
 206 
 207   def substitute(self, env):
 208     self.output_var = env.get(self.output_var, self.output_var)
 209     self.arg_vars = [env.get(v, v) for v in self.arg_vars]

 212 class ProgramGrower(object):
 213   """Grow programs."""
 214 
 215   def __init__(self, functions, types_to_lambdas):
 216     self.functions = functions
 217     self.types_to_lambdas = types_to_lambdas
 218 
 219   def grow_body(self, new_var_name, dependencies, types_to_vars):
 220     """Grow the program body."""
 221     choices = []
 222     for f in self.functions:
 223       if all([a in types_to_vars.keys() for a in f.arg_types]):
 224         choices.append(f)
 225 
 226     f = random.choice(choices)
 227     args = []
 228     for t in f.arg_types:
 229       possible_vars = random.choice(types_to_vars[t])
 230       var = random.choice(possible_vars)
 231       args.append(var)
 232       dependencies.setdefault(new_var_name, []).extend(
 233           [var] + (dependencies[var]))
 234 
 235     fn_args = [random.choice(self.types_to_lambdas[t]) for t in f.fn_arg_types]
 236     types_to_vars.setdefault(f.output_type, []).append(new_var_name)
 237 
 238     return Statement(f, new_var_name, args, fn_args)
 239 
 240   def grow(self, program_len, input_types):
 241     """Grow the program."""
 242     var_names = list(reversed(map(chr, range(97, 123))))
 243     dependencies = dict()
 244     types_to_vars = dict()
 245     input_names = []
 246     for t in input_types:
 247       var = var_names.pop()
 248       dependencies[var] = []
 249       types_to_vars.setdefault(t, []).append(var)
 250       input_names.append(var)
 251 
 252     statements = []
 253     for _ in range(program_len - 1):
 254       var = var_names.pop()
 255       statements.append(self.grow_body(var, dependencies, types_to_vars))
 256     statements.append(self.grow_body("out", dependencies, types_to_vars))
 257 
 258     new_var_names = [c for c in map(chr, range(97, 123))
 259                      if c not in input_names]
 260     new_var_names.reverse()
 261     keep_statements = []
 262     env = dict()
 263     for s in statements:
 264       if s.output_var in dependencies["out"]:
 265         keep_statements.append(s)
 266         env[s.output_var] = new_var_names.pop()
 267       if s.output_var == "out":
 268         keep_statements.append(s)
 269 
 270     for k in keep_statements:
 271       k.substitute(env)
 272 
 273     return Program(input_names, input_types, ";".join(
 274         [str(k) for k in keep_statements]))

 277 class Program(object):
 278   """The program class."""
 279 
 280   def __init__(self, input_names, input_types, body):
 281     self.input_names = input_names
 282     self.input_types = input_types
 283     self.body = body
 284 
 285   def evaluate(self, inputs):
 286     """Evaluate this program."""
 287     if len(inputs) != len(self.input_names):
 288       raise AssertionError("inputs and input_names have to"
 289                            "have the same len. inp: %s , names: %s" %
 290                            (str(inputs), str(self.input_names)))
 291     inp_str = ""
 292     for (name, inp) in zip(self.input_names, inputs):
 293       inp_str += name + " = " + str(inp) + "; "
 294 
 295     with stdoutIO() as s:
 296       # pylint: disable=exec-used
 297       exec(inp_str + self.body + "; print(out)")
 298       # pylint: enable=exec-used
 299     return s.getvalue()[:-1]
 300 
 301   def flat_str(self):
 302     out = ""
 303     for s in self.body.split(";"):
 304       out += s + ";"
 305     return out
 306 
 307   def __str__(self):
 308     out = ""
 309     for (n, t) in zip(self.input_names, self.input_types):
 310       out += n + " = " + str(t) + "\n"
 311     for s in self.body.split(";"):
 312       out += s + "\n"
 313     return out

 320 def tokenize(string, tokens=None):
 321   """Tokenize the program string."""
 322   if tokens is None:
 323     tokens = prog_vocab
 324   tokens = sorted(tokens, key=len, reverse=True)
 325   out = []
 326   string = string.strip()
 327   while string:
 328     found = False
 329     for t in tokens:
 330       if string.startswith(t):
 331         out.append(t)
 332         string = string[len(t):]
 333         found = True
 334         break
 335     if not found:
 336       raise ValueError("Couldn't tokenize this: " + string)
 337     string = string.strip()
 338   return out

 341 def clean_up(output, max_val=100):
 342   o = eval(str(output))
 343   if isinstance(o, bool):
 344     return o
 345   if isinstance(o, int):
 346     if o >= 0:
 347       return min(o, max_val)
 348     else:
 349       return max(o, -1 * max_val)
 350   if isinstance(o, list):
 351     return [clean_up(l) for l in o]

 358 def gen(max_len, how_many):
 359   """Generate some programs."""
 360   functions = [f_head, f_last, f_take, f_drop, f_access, f_max, f_min,
 361                f_reverse, f_sort, f_sum, f_map, f_filter, f_count, f_zipwith,
 362                f_scan]
 363 
 364   types_to_lambdas = {
 365       FunctionType(["Int", "Int"]): ["plus_one", "minus_one", "times_two",
 366                                      "div_two", "sq", "times_three",
 367                                      "div_three", "times_four", "div_four"],
 368       FunctionType(["Int", "Bool"]): ["pos", "neg", "even", "odd"],
 369       FunctionType(["Int", "Int", "Int"]): ["add", "sub", "mul"]
 370   }
 371 
 372   tokens = []
 373   for f in functions:
 374     tokens.append(f.name)
 375   for v in types_to_lambdas.values():
 376     tokens.extend(v)
 377   tokens.extend(["=", ";", ",", "(", ")", "[", "]", "Int", "out"])
 378   tokens.extend(map(chr, range(97, 123)))
 379 
 380   io_tokens = map(str, range(-220, 220))
 381   if not prog_vocab:
 382     prog_vocab.extend(["_PAD", "_EOS"] + tokens + io_tokens)
 383     for i, t in enumerate(prog_vocab):
 384       prog_rev_vocab[t] = i
 385 
 386   io_tokens += [",", "[", "]", ")", "(", "None"]
 387   grower = ProgramGrower(functions=functions,
 388                          types_to_lambdas=types_to_lambdas)
 389 
 390   def mk_inp(l):
 391     return [random.choice(range(-5, 5)) for _ in range(l)]
 392 
 393   tar = [ListType("Int")]
 394   inps = [[mk_inp(3)], [mk_inp(5)], [mk_inp(7)], [mk_inp(15)]]
 395 
 396   save_prefix = None
 397   outcomes_to_programs = dict()
 398   tried = set()
 399   counter = 0
 400   choices = [0] if max_len == 0 else range(max_len)
 401   while counter < 100 * how_many and len(outcomes_to_programs) < how_many:
 402     counter += 1
 403     length = random.choice(choices)
 404     t = grower.grow(length, tar)
 405     while t in tried:
 406       length = random.choice(choices)
 407       t = grower.grow(length, tar)
 408     # print(t.flat_str())
 409     tried.add(t)
 410     outcomes = [clean_up(t.evaluate(i)) for i in inps]
 411     outcome_str = str(zip(inps, outcomes))
 412     if outcome_str in outcomes_to_programs:
 413       outcomes_to_programs[outcome_str] = min(
 414           [t.flat_str(), outcomes_to_programs[outcome_str]],
 415           key=lambda x: len(tokenize(x, tokens)))
 416     else:
 417       outcomes_to_programs[outcome_str] = t.flat_str()
 418     if counter % 5000 == 0:
 419       print("== proggen: tried: " + str(counter))
 420       print("== proggen: kept:  " + str(len(outcomes_to_programs)))
 421 
 422     if counter % 250000 == 0 and save_prefix is not None:
 423       print("saving...")
 424       save_counter = 0
 425       progfilename = os.path.join(save_prefix, "prog_" + str(counter) + ".txt")
 426       iofilename = os.path.join(save_prefix, "io_" + str(counter) + ".txt")
 427       prog_token_filename = os.path.join(save_prefix,
 428                                          "prog_tokens_" + str(counter) + ".txt")
 429       io_token_filename = os.path.join(save_prefix,
 430                                        "io_tokens_" + str(counter) + ".txt")
 431       with open(progfilename, "a+") as fp,  \
 432            open(iofilename, "a+") as fi, \
 433            open(prog_token_filename, "a+") as ftp, \
 434            open(io_token_filename, "a+") as fti:
 435         for (o, p) in outcomes_to_programs.iteritems():
 436           save_counter += 1
 437           if save_counter % 500 == 0:
 438             print("saving %d of %d" % (save_counter, len(outcomes_to_programs)))
 439           fp.write(p+"\n")
 440           fi.write(o+"\n")
 441           ftp.write(str(tokenize(p, tokens))+"\n")
 442           fti.write(str(tokenize(o, io_tokens))+"\n")
 443 
 444   return list(outcomes_to_programs.values())
