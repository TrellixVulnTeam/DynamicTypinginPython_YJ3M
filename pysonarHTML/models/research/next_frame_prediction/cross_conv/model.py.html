  29 class CrossConvModel(object):
  30 
  31   def __init__(self, image_diff_list, params):
  32     """Constructor.
  33 
  34     Args:
  35       image_diff_list: A list of (image, diff) tuples, with shape
  36           [batch_size, image_size, image_size, 3] and image_sizes as
  37           [32, 64, 128, 256].
  38       params: Dict of parameters.
  39     """
  40     self.images = [i for (i, _) in image_diff_list]
  41     # Move the diff to the positive realm.
  42     self.diffs = [(d + params['scale']) / 2 for (i, d) in image_diff_list]
  43     self.params = params
  44 
  45   def Build(self):
  46     with tf.device('/gpu:0'):
  47       with slim.arg_scope([slim.conv2d],
  48                           activation_fn=tf.nn.relu,
  49                           normalizer_fn=slim.batch_norm,
  50                           normalizer_params={'is_training':
  51                                              self.params['is_training']}):
  52         self._BuildMotionKernel()
  53         encoded_images = self._BuildImageEncoder()
  54         cross_conved_images = self._CrossConv(encoded_images)
  55         self._BuildImageDecoder(cross_conved_images)
  56         self._BuildLoss()
  57 
  58       image = self.images[1]
  59       diff = self.diffs[1]
  60 
  61       self.global_step = tf.Variable(0, name='global_step', trainable=False)
  62 
  63       if self.params['is_training']:
  64         self._BuildTrainOp()
  65 
  66       diff = diff * 2.0 - self.params['scale']
  67       diff_output = self.diff_output * 2.0 - self.params['scale']
  68       concat_image = tf.concat(
  69           axis=1, values=[image, image + diff_output, image + diff, diff_output])
  70       tf.summary.image('origin_predict_expect_predictdiff', concat_image)
  71       self.summary_op = tf.summary.merge_all()
  72       return self.loss
  73 
  74   def _BuildTrainOp(self):
  75     lrn_rate = tf.maximum(
  76         0.01,  # min_lr_rate.
  77         tf.train.exponential_decay(
  78             self.params['learning_rate'], self.global_step, 10000, 0.5))
  79     tf.summary.scalar('learning rate', lrn_rate)
  80     optimizer = tf.train.GradientDescentOptimizer(lrn_rate)
  81     self.train_op = slim.learning.create_train_op(
  82         self.loss, optimizer, global_step=self.global_step)
  83 
  84   def _BuildLoss(self):
  85     # 1. reconstr_loss seems doesn't do better than l2 loss.
  86     # 2. Only works when using reduce_mean. reduce_sum doesn't work.
  87     # 3. It seems kl loss doesn't play an important role.
  88     self.loss = 0
  89     with tf.variable_scope('loss'):
  90       if self.params['l2_loss']:
  91         l2_loss = tf.reduce_mean(tf.square(self.diff_output - self.diffs[1]))
  92         tf.summary.scalar('l2_loss', l2_loss)
  93         self.loss += l2_loss
  94       if self.params['reconstr_loss']:
  95         reconstr_loss = (-tf.reduce_mean(
  96             self.diffs[1] * (1e-10 + self.diff_output) +
  97             (1-self.diffs[1]) * tf.log(1e-10 + 1 - self.diff_output)))
  98         reconstr_loss = tf.check_numerics(reconstr_loss, 'reconstr_loss')
  99         tf.summary.scalar('reconstr_loss', reconstr_loss)
 100         self.loss += reconstr_loss
 101       if self.params['kl_loss']:
 102         kl_loss = (0.5 * tf.reduce_mean(
 103             tf.square(self.z_mean) + tf.square(self.z_stddev) -
 104             2 * self.z_stddev_log - 1))
 105         tf.summary.scalar('kl_loss', kl_loss)
 106         self.loss += kl_loss
 107 
 108       tf.summary.scalar('loss', self.loss)
 109 
 110   def _BuildMotionKernel(self):
 111     image = self.images[-2]
 112     diff = self.diffs[-2]
 113     shape = image.get_shape().as_list()
 114     assert shape[1] == shape[2] and shape[1] == 128
 115     batch_size = shape[0]
 116 
 117     net = tf.concat(axis=3, values=[image, diff])
 118     with tf.variable_scope('motion_encoder'):
 119       with slim.arg_scope([slim.conv2d], padding='VALID'):
 120         net = slim.conv2d(net, 96, [5, 5], stride=1)
 121         net = slim.max_pool2d(net, [2, 2])
 122         net = slim.conv2d(net, 96, [5, 5], stride=1)
 123         net = slim.max_pool2d(net, [2, 2])
 124         net = slim.conv2d(net, 128, [5, 5], stride=1)
 125         net = slim.conv2d(net, 128, [5, 5], stride=1)
 126         net = slim.max_pool2d(net, [2, 2])
 127         net = slim.conv2d(net, 256, [4, 4], stride=1)
 128         net = slim.conv2d(net, 256, [3, 3], stride=1)
 129 
 130         z = tf.reshape(net, shape=[batch_size, -1])
 131         self.z_mean, self.z_stddev_log = tf.split(
 132             axis=1, num_or_size_splits=2, value=z)
 133         self.z_stddev = tf.exp(self.z_stddev_log)
 134 
 135         epsilon = tf.random_normal(
 136             self.z_mean.get_shape().as_list(), 0, 1, dtype=tf.float32)
 137         kernel = self.z_mean + tf.multiply(self.z_stddev, epsilon)
 138 
 139         width = int(math.sqrt(kernel.get_shape().as_list()[1] // 128))
 140         kernel = tf.reshape(kernel, [batch_size, width, width, 128])
 141     with tf.variable_scope('kernel_decoder'):
 142       with slim.arg_scope([slim.conv2d], padding='SAME'):
 143         kernel = slim.conv2d(kernel, 128, [5, 5], stride=1)
 144         self.kernel = slim.conv2d(kernel, 128, [5, 5], stride=1)
 145 
 146     sys.stderr.write('kernel shape: %s\n' % kernel.get_shape())
 147 
 148   def _BuildImageEncoder(self):
 149     feature_maps = []
 150     for (i, image) in enumerate(self.images):
 151       with tf.variable_scope('image_encoder_%d' % i):
 152         with slim.arg_scope([slim.conv2d, slim.max_pool2d], padding='SAME'):
 153           net = slim.conv2d(image, 64, [5, 5], stride=1)
 154           net = slim.conv2d(net, 64, [5, 5], stride=1)
 155           net = slim.max_pool2d(net, [5, 5])
 156           net = slim.conv2d(net, 64, [5, 5], stride=1)
 157           net = slim.conv2d(net, 32, [5, 5], stride=1)
 158           net = slim.max_pool2d(net, [2, 2])
 159       sys.stderr.write('image_conv shape: %s\n' % net.get_shape())
 160       feature_maps.append(net)
 161     return feature_maps

 163   def _CrossConvHelper(self, encoded_image, kernel):
 164     """Cross Convolution.
 165 
 166       The encoded image and kernel are of the same shape. Namely
 167       [batch_size, image_size, image_size, channels]. They are split
 168       into [image_size, image_size] image squares [kernel_size, kernel_size]
 169       kernel squares. kernel squares are used to convolute image squares.
 170     """
 171     images = tf.expand_dims(encoded_image, 0)
 172     kernels = tf.expand_dims(kernel, 3)
 173     return tf.nn.depthwise_conv2d(images, kernels, [1, 1, 1, 1], 'SAME')
 174 
 175   def _CrossConv(self, encoded_images):
 176     """Apply the motion kernel on the encoded_images."""
 177     cross_conved_images = []
 178     kernels = tf.split(axis=3, num_or_size_splits=4, value=self.kernel)
 179     for (i, encoded_image) in enumerate(encoded_images):
 180       with tf.variable_scope('cross_conv_%d' % i):
 181         kernel = kernels[i]
 182 
 183         encoded_image = tf.unstack(encoded_image, axis=0)
 184         kernel = tf.unstack(kernel, axis=0)
 185         assert len(encoded_image) == len(kernel)
 186         assert len(encoded_image) == self.params['batch_size']
 187         conved_image = []
 188         for j in xrange(len(encoded_image)):
 189           conved_image.append(self._CrossConvHelper(
 190               encoded_image[j], kernel[j]))
 191         cross_conved_images.append(tf.concat(axis=0, values=conved_image))
 192         sys.stderr.write('cross_conved shape: %s\n' %
 193                          cross_conved_images[-1].get_shape())
 194     return cross_conved_images
 195 
 196   def _Deconv(self, net, out_filters, kernel_size, stride):
 197     shape = net.get_shape().as_list()
 198     in_filters = shape[3]
 199     kernel_shape = [kernel_size, kernel_size, out_filters, in_filters]
 200 
 201     weights = tf.get_variable(
 202         name='weights',
 203         shape=kernel_shape,
 204         dtype=tf.float32,
 205         initializer=tf.truncated_normal_initializer(stddev=0.01))
 206 
 207 
 208     out_height = shape[1] * stride
 209     out_width = shape[2] * stride
 210     batch_size = shape[0]
 211 
 212     output_shape = [batch_size, out_height, out_width, out_filters]
 213     net = tf.nn.conv2d_transpose(net, weights, output_shape,
 214                                  [1, stride, stride, 1], padding='SAME')
 215     slim.batch_norm(net)
 216     return net
 217 
 218   def _BuildImageDecoder(self, cross_conved_images):
 219     """Decode the cross_conved feature maps into the predicted images."""
 220     nets = []
 221     for i, cross_conved_image in enumerate(cross_conved_images):
 222       with tf.variable_scope('image_decoder_%d' % i):
 223         stride = 64 / cross_conved_image.get_shape().as_list()[1]
 224         # TODO(xpan): Alternative solution for upsampling?
 225         nets.append(self._Deconv(
 226             cross_conved_image, 64, kernel_size=3, stride=stride))
 227 
 228     net = tf.concat(axis=3, values=nets)
 229     net = slim.conv2d(net, 128, [9, 9], padding='SAME', stride=1)
 230     net = slim.conv2d(net, 128, [1, 1], padding='SAME', stride=1)
 231     net = slim.conv2d(net, 3, [1, 1], padding='SAME', stride=1)
 232     self.diff_output = net
 233     sys.stderr.write('diff_output shape: %s\n' % self.diff_output.get_shape())
