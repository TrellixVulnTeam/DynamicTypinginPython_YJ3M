  16 """Train the cross convolutional model."""
  17 import os
  18 import sys
  19 
  20 import numpy as np
  21 import tensorflow as tf
  22 
  23 import model as cross_conv_model
  24 import reader
  25 
  26 FLAGS = tf.flags.FLAGS
  27 tf.flags.DEFINE_string('master', '', 'Session address.')
  28 tf.flags.DEFINE_string('log_root', '/tmp/moving_obj', 'The root dir of output.')
  29 tf.flags.DEFINE_string('data_filepattern', '',
  30                        'training data file pattern.')
  31 tf.flags.DEFINE_integer('image_size', 64, 'Image height and width.')
  32 tf.flags.DEFINE_integer('batch_size', 1, 'Batch size.')
  33 tf.flags.DEFINE_float('norm_scale', 1.0, 'Normalize the original image')
  34 tf.flags.DEFINE_float('scale', 10.0,
  35                       'Scale the image after norm_scale and move the diff '
  36                       'to the positive realm.')
  37 tf.flags.DEFINE_integer('sequence_length', 2, 'tf.SequenceExample length.')
  38 tf.flags.DEFINE_float('learning_rate', 0.8, 'Learning rate.')
  39 tf.flags.DEFINE_bool('l2_loss', True, 'If true, include l2_loss.')
  40 tf.flags.DEFINE_bool('reconstr_loss', False, 'If true, include reconstr_loss.')
  41 tf.flags.DEFINE_bool('kl_loss', True, 'If true, include KL loss.')
  42 
  43 slim = tf.contrib.slim
  44 
  45 
  46 def _Train():
  47   params = dict()
  48   params['batch_size'] = FLAGS.batch_size
  49   params['seq_len'] = FLAGS.sequence_length
  50   params['image_size'] = FLAGS.image_size
  51   params['is_training'] = True
  52   params['norm_scale'] = FLAGS.norm_scale
  53   params['scale'] = FLAGS.scale
  54   params['learning_rate'] = FLAGS.learning_rate
  55   params['l2_loss'] = FLAGS.l2_loss
  56   params['reconstr_loss'] = FLAGS.reconstr_loss
  57   params['kl_loss'] = FLAGS.kl_loss
  58 
  59   train_dir = os.path.join(FLAGS.log_root, 'train')
  60 
  61   images = reader.ReadInput(FLAGS.data_filepattern, shuffle=True, params=params)
  62   images *= params['scale']
  63   # Increase the value makes training much faster.
  64   image_diff_list = reader.SequenceToImageAndDiff(images)
  65   model = cross_conv_model.CrossConvModel(image_diff_list, params)
  66   model.Build()
  67   tf.contrib.tfprof.model_analyzer.print_model_analysis(tf.get_default_graph())
  68 
  69   summary_writer = tf.summary.FileWriter(train_dir)
  70   sv = tf.train.Supervisor(logdir=FLAGS.log_root,
  71                            summary_op=None,
  72                            is_chief=True,
  73                            save_model_secs=60,
  74                            global_step=model.global_step)
  75   sess = sv.prepare_or_wait_for_session(
  76       FLAGS.master, config=tf.ConfigProto(allow_soft_placement=True))
  77 
  78   total_loss = 0.0
  79   step = 0
  80   sample_z_mean = np.zeros(model.z_mean.get_shape().as_list())
  81   sample_z_stddev_log = np.zeros(model.z_stddev_log.get_shape().as_list())
  82   sample_step = 0
  83 
  84   while True:
  85     _, loss_val, total_steps, summaries, z_mean, z_stddev_log = sess.run(
  86         [model.train_op, model.loss, model.global_step,
  87          model.summary_op,
  88          model.z_mean, model.z_stddev_log])
  89 
  90     sample_z_mean += z_mean
  91     sample_z_stddev_log += z_stddev_log
  92     total_loss += loss_val
  93     step += 1
  94     sample_step += 1
  95 
  96     if step % 100 == 0:
  97       summary_writer.add_summary(summaries, total_steps)
  98       sys.stderr.write('step: %d, loss: %f\n' %
  99                        (total_steps, total_loss / step))
 100       total_loss = 0.0
 101       step = 0
 102 
 103     # Sampled z is used for eval.
 104     # It seems 10k is better than 1k. Maybe try 100k next?
 105     if sample_step % 10000 == 0:
 106       with tf.gfile.Open(os.path.join(FLAGS.log_root, 'z_mean.npy'), 'w') as f:
 107         np.save(f, sample_z_mean / sample_step)
 108       with tf.gfile.Open(
 109           os.path.join(FLAGS.log_root, 'z_stddev_log.npy'), 'w') as f:
 110         np.save(f, sample_z_stddev_log / sample_step)
 111       sample_z_mean = np.zeros(model.z_mean.get_shape().as_list())
 112       sample_z_stddev_log = np.zeros(
 113           model.z_stddev_log.get_shape().as_list())
 114       sample_step = 0
 115 
 116 
 117 def main(_):
 118   _Train()
 119 
 120 
 121 if __name__ == '__main__':
 122   tf.app.run()
