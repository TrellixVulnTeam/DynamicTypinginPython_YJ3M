  16 """Generate the sprites tfrecords from raw_images."""
  17 import os
  18 import random
  19 import re
  20 import sys
  21 
  22 import numpy as np
  23 import scipy.misc
  24 from six.moves import xrange
  25 import tensorflow as tf
  26 
  27 
  28 tf.flags.DEFINE_string('data_filepattern', '', 'The raw images.')
  29 tf.flags.DEFINE_string('out_file', '',
  30                        'File name for the tfrecord output.')
  31 
  32 
  33 def _read_images():
  34   """Read images from image files into data structure."""
  35   sprites = dict()
  36   files = tf.gfile.Glob(tf.flags.FLAGS.data_filepattern)
  37   for f in files:
  38     image = scipy.misc.imread(f)
  39     m = re.search('image_([0-9]+)_([0-9]+)_([0-9]+).jpg', os.path.basename(f))
  40     if m.group(1) not in sprites:
  41       sprites[m.group(1)] = dict()
  42     character = sprites[m.group(1)]
  43     if m.group(2) not in character:
  44       character[m.group(2)] = dict()
  45     pose = character[m.group(2)]
  46     pose[int(m.group(3))] = image
  47   return sprites
  48 
  49 
  50 def _images_to_example(image, image2):
  51   """Convert 2 consecutive image to a SequenceExample."""
  52   example = tf.SequenceExample()
  53   feature_list = example.feature_lists.feature_list['moving_objs']
  54   feature = feature_list.feature.add()
  55   feature.float_list.value.extend(np.reshape(image, [-1]).tolist())
  56   feature = feature_list.feature.add()
  57   feature.float_list.value.extend(np.reshape(image2, [-1]).tolist())
  58   return example
  59 
  60 
  61 def generate_input():
  62   """Generate tfrecords."""
  63   sprites = _read_images()
  64   sys.stderr.write('Finish reading images.\n')
  65   train_writer = tf.python_io.TFRecordWriter(
  66       tf.flags.FLAGS.out_file.replace('sprites', 'sprites_train'))
  67   test_writer = tf.python_io.TFRecordWriter(
  68       tf.flags.FLAGS.out_file.replace('sprites', 'sprites_test'))
  69 
  70   train_examples = []
  71   test_examples = []
  72   for i in sprites:
  73     if int(i) < 24:
  74       examples = test_examples
  75     else:
  76       examples = train_examples
  77 
  78     character = sprites[i]
  79     for j in character.keys():
  80       pose = character[j]
  81       for k in xrange(1, len(pose), 1):
  82         image = pose[k]
  83         image2 = pose[k+1]
  84         examples.append(_images_to_example(image, image2))
  85 
  86   sys.stderr.write('Finish generating examples: %d, %d.\n' %
  87                    (len(train_examples), len(test_examples)))
  88   random.shuffle(train_examples)
  89   _ = [train_writer.write(ex.SerializeToString()) for ex in train_examples]
  90   _ = [test_writer.write(ex.SerializeToString()) for ex in test_examples]
  91 
  92 
  93 def main(_):
  94   generate_input()
  95 
  96 
  97 if __name__ == '__main__':
  98   tf.app.run()
