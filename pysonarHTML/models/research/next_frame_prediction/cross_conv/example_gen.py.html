  16 """Generate examples of two objects moving in different directions."""
  17 import random
  18 import sys
  19 
  20 import numpy as np
  21 from six.moves import xrange
  22 import tensorflow as tf
  23 
  24 
  25 tf.flags.DEFINE_string('out_file', '',
  26                        'Output file for the tfrecords.')

  29 def _add_object(obj_type, image, image2, xpos, ypos):
  30   """Add a moving obj to two consecutive images."""
  31   obj_size = random.randint(8, 10)
  32   channel = random.randint(0, 2)
  33   move = random.randint(6, 10)
  34 
  35   obj = np.zeros([obj_size, obj_size, 3])
  36   if obj_type == 'rectangle':
  37     xpos2 = xpos + move
  38     ypos2 = ypos
  39     for i in xrange(obj_size):
  40       obj[i, 0:i+1, channel] = [1.0 for _ in xrange(i+1)]
  41   elif obj_type == 'square':
  42     xpos2 = xpos
  43     ypos2 = ypos + move
  44     obj[:, :, channel] = 1.0
  45 
  46   for x in xrange(obj_size):
  47     for y in xrange(obj_size):
  48       if obj[x, y, channel] == 1.0:
  49         image[xpos+x, ypos+y, channel] = 1.0
  50         image2[xpos2+x, ypos2+y, channel] = 1.0

  53 def _images_to_example(image, image2):
  54   """Convert two consecutive images to SequenceExample."""
  55   example = tf.SequenceExample()
  56   feature_list = example.feature_lists.feature_list['moving_objs']
  57   feature = feature_list.feature.add()
  58   feature.float_list.value.extend(np.reshape(image, [-1]).tolist())
  59   feature = feature_list.feature.add()
  60   feature.float_list.value.extend(np.reshape(image2, [-1]).tolist())
  61   return example

  64 def generate_input():
  65   """Generate tfrecords."""
  66   writer = tf.python_io.TFRecordWriter(tf.flags.FLAGS.out_file)
  67   writer2 = tf.python_io.TFRecordWriter(tf.flags.FLAGS.out_file + '_test')
  68 
  69   examples = []
  70   for xpos in xrange(0, 40, 3):
  71     for ypos in xrange(0, 40, 3):
  72       for xpos2 in xrange(0, 40, 3):
  73         for ypos2 in xrange(0, 40, 3):
  74           image = np.zeros([64, 64, 3])
  75           image2 = np.zeros([64, 64, 3])
  76           _add_object('rectangle', image, image2, xpos, ypos)
  77           _add_object('square', image, image2, xpos2, ypos2)
  78           examples.append(_images_to_example(image, image2))
  79 
  80   sys.stderr.write('Finish generating examples.\n')
  81   random.shuffle(examples)
  82   for count, ex in enumerate(examples):
  83     if count % 10 == 0:
  84       writer2.write(ex.SerializeToString())
  85     else:
  86       writer.write(ex.SerializeToString())
  87 
  88 def main(_):
  89   generate_input()
  90 
  91 
  92 if __name__ == '__main__':
  93   tf.app.run()
