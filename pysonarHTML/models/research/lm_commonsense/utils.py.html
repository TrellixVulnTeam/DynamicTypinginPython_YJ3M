  29 class Vocabulary(object):
  30   """Class that holds a vocabulary for the dataset."""
  31 
  32   def __init__(self, filename):
  33 
  34     self._id_to_word = []
  35     self._word_to_id = {}
  36     self._unk = -1
  37     self._bos = -1
  38     self._eos = -1
  39 
  40     with tf.gfile.Open(filename) as f:
  41       idx = 0
  42       for line in f:
  43         word_name = line.strip()
  44         if word_name == '<S>':
  45           self._bos = idx
  46         elif word_name == '</S>':
  47           self._eos = idx
  48         elif word_name == '<UNK>':
  49           self._unk = idx
  50         if word_name == '!!!MAXTERMID':
  51           continue
  52 
  53         self._id_to_word.append(word_name)
  54         self._word_to_id[word_name] = idx
  55         idx += 1
  56 
  57   @property
  58   def bos(self):
  59     return self._bos
  60 
  61   @property
  62   def eos(self):
  63     return self._eos
  64 
  65   @property
  66   def unk(self):
  67     return self._unk
  68 
  69   @property
  70   def size(self):
  71     return len(self._id_to_word)
  72 
  73   def word_to_id(self, word):
  74     if word in self._word_to_id:
  75       return self._word_to_id[word]
  76     else:
  77       if word.lower() in self._word_to_id:
  78         return self._word_to_id[word.lower()]
  79     return self.unk
  80 
  81   def id_to_word(self, cur_id):
  82     if cur_id < self.size:
  83       return self._id_to_word[int(cur_id)]
  84     return '<ERROR_out_of_vocab_id>'
  85 
  86   def decode(self, cur_ids):
  87     return ' '.join([self.id_to_word(cur_id) for cur_id in cur_ids])
  88 
  89   def encode(self, sentence):
  90     word_ids = [self.word_to_id(cur_word) for cur_word in sentence.split()]
  91     return np.array([self.bos] + word_ids + [self.eos], dtype=np.int32)

  94 class CharsVocabulary(Vocabulary):
  95   """Vocabulary containing character-level information."""
  96 
  97   def __init__(self, filename, max_word_length):
  98     super(CharsVocabulary, self).__init__(filename)
  99 
 100     self._max_word_length = max_word_length
 101     chars_set = set()
 102 
 103     for word in self._id_to_word:
 104       chars_set |= set(word)
 105 
 106     free_ids = []
 107     for i in range(256):
 108       if chr(i) in chars_set:
 109         continue
 110       free_ids.append(chr(i))
 111 
 112     if len(free_ids) < 5:
 113       raise ValueError('Not enough free char ids: %d' % len(free_ids))
 114 
 115     self.bos_char = free_ids[0]  # <begin sentence>
 116     self.eos_char = free_ids[1]  # <end sentence>
 117     self.bow_char = free_ids[2]  # <begin word>
 118     self.eow_char = free_ids[3]  # <end word>
 119     self.pad_char = free_ids[4]  # <padding>
 120 
 121     chars_set |= {self.bos_char, self.eos_char, self.bow_char, self.eow_char,
 122                   self.pad_char}
 123 
 124     self._char_set = chars_set
 125     num_words = len(self._id_to_word)
 126 
 127     self._word_char_ids = np.zeros([num_words, max_word_length], dtype=np.int32)
 128 
 129     self.bos_chars = self._convert_word_to_char_ids(self.bos_char)
 130     self.eos_chars = self._convert_word_to_char_ids(self.eos_char)
 131 
 132     for i, word in enumerate(self._id_to_word):
 133       if i == self.bos:
 134         self._word_char_ids[i] = self.bos_chars
 135       elif i == self.eos:
 136         self._word_char_ids[i] = self.eos_chars
 137       else:
 138         self._word_char_ids[i] = self._convert_word_to_char_ids(word)
 139 
 140   @property
 141   def max_word_length(self):
 142     return self._max_word_length
 143 
 144   def _convert_word_to_char_ids(self, word):
 145     code = np.zeros([self.max_word_length], dtype=np.int32)
 146     code[:] = ord(self.pad_char)
 147 
 148     if len(word) > self.max_word_length - 2:
 149       word = word[:self.max_word_length-2]
 150     cur_word = self.bow_char + word + self.eow_char
 151     for j in range(len(cur_word)):
 152       code[j] = ord(cur_word[j])
 153     return code
 154 
 155   def word_to_char_ids(self, word):
 156     if word in self._word_to_id:
 157       return self._word_char_ids[self._word_to_id[word]]
 158     else:
 159       return self._convert_word_to_char_ids(word)
 160 
 161   def encode_chars(self, sentence):
 162     chars_ids = [self.word_to_char_ids(cur_word)
 163                  for cur_word in sentence.split()]
 164     return np.vstack([self.bos_chars] + chars_ids + [self.eos_chars])

 167 _SPECIAL_CHAR_MAP = {
 168     '\xe2\x80\x98': '\'',
 169     '\xe2\x80\x99': '\'',
 170     '\xe2\x80\x9c': '"',
 171     '\xe2\x80\x9d': '"',
 172     '\xe2\x80\x93': '-',
 173     '\xe2\x80\x94': '-',
 174     '\xe2\x88\x92': '-',
 175     '\xce\x84': '\'',
 176     '\xc2\xb4': '\'',
 177     '`': '\''
 178 }
 179 
 180 _START_SPECIAL_CHARS = ['.', ',', '?', '!', ';', ':', '[', ']', '\'', '+', '/',
 181                         '\xc2\xa3', '$', '~', '*', '%', '{', '}', '#', '&', '-',
 182                         '"', '(', ')', '='] + list(_SPECIAL_CHAR_MAP.keys())
 183 _SPECIAL_CHARS = _START_SPECIAL_CHARS + [
 184     '\'s', '\'m', '\'t', '\'re', '\'d', '\'ve', '\'ll']

 187 def tokenize(sentence):
 188   """Tokenize a sentence."""
 189   sentence = str(sentence)
 190   words = sentence.strip().split()
 191   tokenized = []  # return this
 192 
 193   for word in words:
 194     if word.lower() in ['mr.', 'ms.']:
 195       tokenized.append(word)
 196       continue
 197 
 198     # Split special chars at the start of word
 199     will_split = True
 200     while will_split:
 201       will_split = False
 202       for char in _START_SPECIAL_CHARS:
 203         if word.startswith(char):
 204           tokenized.append(char)
 205           word = word[len(char):]
 206           will_split = True
 207 
 208     # Split special chars at the end of word
 209     special_end_tokens = []
 210     will_split = True
 211     while will_split:
 212       will_split = False
 213       for char in _SPECIAL_CHARS:
 214         if word.endswith(char):
 215           special_end_tokens = [char] + special_end_tokens
 216           word = word[:-len(char)]
 217           will_split = True
 218 
 219     if word:
 220       tokenized.append(word)
 221     tokenized += special_end_tokens
 222 
 223   # Add necessary end of sentence token.
 224   if tokenized[-1] not in ['.', '!', '?']:
 225     tokenized += ['.']
 226   return tokenized

 229 def parse_commonsense_reasoning_test(test_data_name):
 230   """Read JSON test data."""
 231   with tf.gfile.Open(os.path.join(
 232       FLAGS.data_dir, 'commonsense_test',
 233       '{}.json'.format(test_data_name)), 'r') as f:
 234     data = json.load(f)
 235 
 236   question_ids = [d['question_id'] for d in data]
 237   sentences = [tokenize(d['substitution']) for d in data]
 238   labels = [d['correctness'] for d in data]
 239 
 240   return question_ids, sentences, labels

 246 def cut_to_patches(sentences, batch_size, num_timesteps):
 247   """Cut sentences into patches of shape (batch_size, num_timesteps).
 248 
 249   Args:
 250     sentences: a list of sentences, each sentence is a list of str token.
 251     batch_size: batch size
 252     num_timesteps: number of backprop step
 253 
 254   Returns:
 255     patches: A 2D matrix,
 256       each entry is a matrix of shape (batch_size, num_timesteps).
 257   """
 258   preprocessed = [['<S>']+sentence+['</S>'] for sentence in sentences]
 259   max_len = max([len(sent) for sent in preprocessed])
 260 
 261   # Pad to shape [height, width]
 262   # where height is a multiple of batch_size
 263   # and width is a multiple of num_timesteps
 264   nrow = int(np.ceil(len(preprocessed) * 1.0 / batch_size))
 265   ncol = int(np.ceil(max_len * 1.0 / num_timesteps))
 266   height, width = nrow * batch_size, ncol * num_timesteps + 1
 267   preprocessed = [sent + [PAD] * (width - len(sent)) for sent in preprocessed]
 268   preprocessed += [[PAD] * width] * (height - len(preprocessed))
 269 
 270   # Cut preprocessed into patches of shape [batch_size, num_timesteps]
 271   patches = []
 272   for row in range(nrow):
 273     patches.append([])
 274     for col in range(ncol):
 275       patch = [sent[col * num_timesteps:
 276                     (col+1) * num_timesteps + 1]
 277                for sent in preprocessed[row * batch_size:
 278                                         (row+1) * batch_size]]
 279       if np.all(np.array(patch)[:, 1:] == PAD):
 280         patch = None  # no need to process this patch.
 281       patches[-1].append(patch)
 282   return patches

 285 def _substitution_mask(sent1, sent2):
 286   """Binary mask identifying substituted part in two sentences.
 287 
 288   Example sentence and their mask:
 289     First sentence  = "I like the cat        's color"
 290                        0 0    0   1           0 0
 291     Second sentence = "I like the yellow dog 's color"
 292                        0 0    0   1      1    0 0
 293 
 294   Args:
 295     sent1: first sentence
 296     sent2: second sentence
 297 
 298   Returns:
 299     mask1: mask for first sentence
 300     mask2: mask for second sentence
 301   """
 302   mask1_start, mask2_start = [], []
 303   while sent1[0] == sent2[0]:
 304     sent1 = sent1[1:]
 305     sent2 = sent2[1:]
 306     mask1_start.append(0.)
 307     mask2_start.append(0.)
 308 
 309   mask1_end, mask2_end = [], []
 310   while sent1[-1] == sent2[-1]:
 311     if (len(sent1) == 1) or (len(sent2) == 1):
 312       break
 313     sent1 = sent1[:-1]
 314     sent2 = sent2[:-1]
 315     mask1_end = [0.] + mask1_end
 316     mask2_end = [0.] + mask2_end
 317 
 318   assert sent1 or sent2, 'Two sentences are identical.'
 319   return (mask1_start + [1.] * len(sent1) + mask1_end,
 320           mask2_start + [1.] * len(sent2) + mask2_end)

 323 def _convert_to_partial(scoring1, scoring2):
 324   """Convert full scoring into partial scoring."""
 325   mask1, mask2 = _substitution_mask(
 326       scoring1['sentence'], scoring2['sentence'])
 327 
 328   def _partial_score(scoring, mask):
 329     word_probs = [max(_) for _ in zip(scoring['word_probs'], mask)]
 330     scoring.update(word_probs=word_probs,
 331                    joint_prob=np.prod(word_probs))
 332 
 333   _partial_score(scoring1, mask1)
 334   _partial_score(scoring2, mask2)

 337 def compare_substitutions(question_ids, scorings, mode='full'):
 338   """Return accuracy by comparing two consecutive scorings."""
 339   prediction_correctness = []
 340   # Compare two consecutive substitutions
 341   for i in range(len(scorings) // 2):
 342     scoring1, scoring2 = scorings[2*i: 2*i+2]
 343     if mode == 'partial':  # fix joint prob into partial prob
 344       _convert_to_partial(scoring1, scoring2)
 345 
 346     prediction_correctness.append(
 347         (scoring2['joint_prob'] > scoring1['joint_prob']) ==
 348          scoring2['correctness'])
 349 
 350   # Two consecutive substitutions always belong to the same question
 351   question_ids = [qid for i, qid in enumerate(question_ids) if i % 2 == 0]
 352   assert len(question_ids) == len(prediction_correctness)
 353   num_questions = len(set(question_ids))
 354 
 355   # Question is correctly answered only if
 356   # all predictions of the same question_id is correct
 357   num_correct_answer = 0
 358   previous_qid = None
 359   correctly_answered = False
 360   for predict, qid in zip(prediction_correctness, question_ids):
 361     if qid != previous_qid:
 362       previous_qid = qid
 363       num_correct_answer += int(correctly_answered)
 364       correctly_answered = True
 365     correctly_answered = correctly_answered and predict
 366   num_correct_answer += int(correctly_answered)
 367 
 368   return num_correct_answer / num_questions
