  35 class EnsembleLM(object):
  36   """Ensemble of language models."""
  37 
  38   def __init__(self, test_data_name='wsc273'):
  39     vocab_file = os.path.join(FLAGS.data_dir, 'vocab.txt')
  40     self.vocab = utils.CharsVocabulary(vocab_file, 50)
  41     assert test_data_name in ['pdp60', 'wsc273'], (
  42         'Test data must be pdp60 or wsc273, got {}'.format(test_data_name))
  43     self.test_data_name = test_data_name
  44 
  45     test_data = utils.parse_commonsense_reasoning_test(test_data_name)
  46     self.question_ids, self.sentences, self.labels = test_data
  47     self.all_probs = []  # aggregate single-model prediction here.
  48 
  49   def add_single_model(self, model_name='lm1'):
  50     """Add a single model into the current ensemble."""
  51     # Create single LM
  52     single_lm = SingleRecurrentLanguageModel(self.vocab, model_name)
  53 
  54     # Add the single LM prediction.
  55     probs = single_lm.assign_probs(self.sentences, self.test_data_name)
  56     self.all_probs.append(probs)
  57     print('Done adding {}'.format(model_name))
  58 
  59   def evaluate(self):
  60     """Evaluate the current ensemble."""
  61     # Attach word probabilities and correctness label to each substitution
  62     ensembled_probs = sum(self.all_probs) / len(self.all_probs)
  63     scorings = []
  64     for i, sentence in enumerate(self.sentences):
  65       correctness = self.labels[i]
  66       word_probs = ensembled_probs[i, :len(sentence)]
  67       joint_prob = np.prod(word_probs, dtype=np.float64)
  68 
  69       scorings.append(dict(
  70           correctness=correctness,
  71           sentence=sentence,
  72           joint_prob=joint_prob,
  73           word_probs=word_probs))
  74     scoring_mode = 'full' if self.test_data_name == 'pdp60' else 'partial'
  75     return utils.compare_substitutions(
  76         self.question_ids, scorings, scoring_mode)

  79 class SingleRecurrentLanguageModel(object):
  80   """Single Recurrent Language Model."""
  81 
  82   def __init__(self, vocab, model_name='lm01'):
  83     self.vocab = vocab
  84     self.log_dir = os.path.join(FLAGS.data_dir, model_name)
  85 
  86   def reset(self):
  87     self.sess.run(self.tensors['states_init'])
  88 
  89   def _score(self, word_patch):
  90     """Score a matrix of shape (batch_size, num_timesteps+1) str tokens."""
  91     word_ids = np.array(
  92         [[self.vocab.word_to_id(word) for word in row]
  93          for row in word_patch])
  94     char_ids = np.array(
  95         [[self.vocab.word_to_char_ids(word) for word in row]
  96          for row in word_patch])
  97     print('Probs for \n{}\n='.format(np.array(word_patch)[:, 1:]))
  98 
  99     input_ids, target_ids = word_ids[:, :-1], word_ids[:, 1:]
 100     input_char_ids = char_ids[:, :-1, :]
 101 
 102     softmax = self.sess.run(self.tensors['softmax_out'], feed_dict={
 103         self.tensors['inputs_in']: input_ids,
 104         self.tensors['char_inputs_in']: input_char_ids
 105     })
 106 
 107     batch_size, num_timesteps = self.shape
 108     softmax = softmax.reshape((num_timesteps, batch_size, -1))
 109     softmax = np.transpose(softmax, [1, 0, 2])
 110     probs = np.array([[softmax[row, col, target_ids[row, col]]
 111                        for col in range(num_timesteps)]
 112                       for row in range(batch_size)])
 113     print(probs)
 114     return probs
 115 
 116   def _score_patches(self, word_patches):
 117     """Score a 2D matrix of word_patches and stitch results together."""
 118     batch_size, num_timesteps = self.shape
 119     nrow, ncol = len(word_patches), len(word_patches[0])
 120     max_len = num_timesteps * ncol
 121     probs = np.zeros([0, max_len])  # accumulate results into this.
 122 
 123     # Loop through the 2D matrix of word_patches and score each.
 124     for i, row in enumerate(word_patches):
 125       print('Reset RNN states.')
 126       self.reset()  # reset states before processing each row.
 127       row_probs = np.zeros([batch_size, 0])
 128       for j, word_patch in enumerate(row):
 129         print('Processing patch '
 130               '({}, {}) / ({}, {})'.format(i+1, j+1, nrow, ncol))
 131         patch_probs = (self._score(word_patch) if word_patch else
 132                        np.zeros([batch_size, num_timesteps]))
 133         row_probs = np.concatenate([row_probs, patch_probs], 1)
 134       probs = np.concatenate([probs, row_probs], 0)
 135     return probs
 136 
 137   def assign_probs(self, sentences, test_data_name='wsc273'):
 138     """Return prediction accuracy using this LM for a test."""
 139 
 140     probs_cache = os.path.join(self.log_dir, '{}.probs'.format(test_data_name))
 141     if os.path.exists(probs_cache):
 142       print('Reading cached result from {}'.format(probs_cache))
 143       with tf.gfile.Open(probs_cache, 'r') as f:
 144         probs = pkl.load(f)
 145     else:
 146       tf.reset_default_graph()
 147       self.sess = tf.Session()
 148       # Build the graph.
 149       saver = tf.train.import_meta_graph(
 150           os.path.join(self.log_dir, 'ckpt-best.meta'))
 151       saver.restore(self.sess, os.path.join(self.log_dir, 'ckpt-best'))
 152       print('Restored from {}'.format(self.log_dir))
 153       graph = tf.get_default_graph()
 154       self.tensors = dict(
 155           inputs_in=graph.get_tensor_by_name('test_inputs_in:0'),
 156           char_inputs_in=graph.get_tensor_by_name('test_char_inputs_in:0'),
 157           softmax_out=graph.get_tensor_by_name('SotaRNN_1/softmax_out:0'),
 158           states_init=graph.get_operation_by_name('SotaRNN_1/states_init'))
 159       self.shape = self.tensors['inputs_in'].shape.as_list()
 160 
 161       # Cut sentences into patches of shape processable by the LM.
 162       batch_size, num_timesteps = self.shape
 163       word_patches = utils.cut_to_patches(sentences, batch_size, num_timesteps)
 164       probs = self._score_patches(word_patches)
 165 
 166       # Cache the probs since they are expensive to evaluate
 167       with tf.gfile.Open(probs_cache, 'w') as f:
 168         pkl.dump(probs, f)
 169     return probs

 172 def evaluate_ensemble(test_data_name, number_of_lms):
 173   ensemble = EnsembleLM(test_data_name)
 174   model_list = ['lm{:02d}'.format(i+1) for i in range(number_of_lms)]
 175   for model_name in model_list:
 176     ensemble.add_single_model(model_name)
 177   accuracy = ensemble.evaluate()
 178   print('Accuracy of {} LM(s) on {} = {}'.format(
 179       number_of_lms, test_data_name, accuracy))
