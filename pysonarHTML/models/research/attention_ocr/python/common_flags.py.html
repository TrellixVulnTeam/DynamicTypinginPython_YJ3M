  38 def define():
  39   """Define common flags."""
  40   # yapf: disable
  41   flags.DEFINE_integer('batch_size', 32,
  42                        'Batch size.')
  43 
  44   flags.DEFINE_integer('crop_width', None,
  45                        'Width of the central crop for images.')
  46 
  47   flags.DEFINE_integer('crop_height', None,
  48                        'Height of the central crop for images.')
  49 
  50   flags.DEFINE_string('train_log_dir', '/tmp/attention_ocr/train',
  51                       'Directory where to write event logs.')
  52 
  53   flags.DEFINE_string('dataset_name', 'fsns',
  54                       'Name of the dataset. Supported: fsns')
  55 
  56   flags.DEFINE_string('split_name', 'train',
  57                       'Dataset split name to run evaluation for: test,train.')
  58 
  59   flags.DEFINE_string('dataset_dir', None,
  60                       'Dataset root folder.')
  61 
  62   flags.DEFINE_string('checkpoint', '',
  63                       'Path for checkpoint to restore weights from.')
  64 
  65   flags.DEFINE_string('master',
  66                       '',
  67                       'BNS name of the TensorFlow master to use.')
  68 
  69   # Model hyper parameters
  70   flags.DEFINE_float('learning_rate', 0.004,
  71                      'learning rate')
  72 
  73   flags.DEFINE_string('optimizer', 'momentum',
  74                       'the optimizer to use')
  75 
  76   flags.DEFINE_float('momentum', 0.9,
  77                       'momentum value for the momentum optimizer if used')
  78 
  79   flags.DEFINE_bool('use_augment_input', True,
  80                     'If True will use image augmentation')
  81 
  82   # Method hyper parameters
  83   # conv_tower_fn
  84   flags.DEFINE_string('final_endpoint', 'Mixed_5d',
  85                       'Endpoint to cut inception tower')
  86 
  87   # sequence_logit_fn
  88   flags.DEFINE_bool('use_attention', True,
  89                     'If True will use the attention mechanism')
  90 
  91   flags.DEFINE_bool('use_autoregression', True,
  92                     'If True will use autoregression (a feedback link)')
  93 
  94   flags.DEFINE_integer('num_lstm_units', 256,
  95                        'number of LSTM units for sequence LSTM')
  96 
  97   flags.DEFINE_float('weight_decay', 0.00004,
  98                      'weight decay for char prediction FC layers')
  99 
 100   flags.DEFINE_float('lstm_state_clip_value', 10.0,
 101                      'cell state is clipped by this value prior to the cell'
 102                      ' output activation')
 103 
 104   # 'sequence_loss_fn'
 105   flags.DEFINE_float('label_smoothing', 0.1,
 106                      'weight for label smoothing')
 107 
 108   flags.DEFINE_bool('ignore_nulls', True,
 109                     'ignore null characters for computing the loss')
 110 
 111   flags.DEFINE_bool('average_across_timesteps', False,
 112                     'divide the returned cost by the total label weight')
 113   # yapf: enable
 114

 116 def get_crop_size():
 117   if FLAGS.crop_width and FLAGS.crop_height:
 118     return (FLAGS.crop_width, FLAGS.crop_height)
 119   else:
 120     return None
 121 
 122 
 123 def create_dataset(split_name):
 124   ds_module = getattr(datasets, FLAGS.dataset_name)
 125   return ds_module.get_split(split_name, dataset_dir=FLAGS.dataset_dir)
 126 
 127 
 128 def create_mparams():
 129   return {
 130       'conv_tower_fn':
 131       model.ConvTowerParams(final_endpoint=FLAGS.final_endpoint),
 132       'sequence_logit_fn':
 133       model.SequenceLogitsParams(
 134           use_attention=FLAGS.use_attention,
 135           use_autoregression=FLAGS.use_autoregression,
 136           num_lstm_units=FLAGS.num_lstm_units,
 137           weight_decay=FLAGS.weight_decay,
 138           lstm_state_clip_value=FLAGS.lstm_state_clip_value),
 139       'sequence_loss_fn':
 140       model.SequenceLossParams(
 141           label_smoothing=FLAGS.label_smoothing,
 142           ignore_nulls=FLAGS.ignore_nulls,
 143           average_across_timesteps=FLAGS.average_across_timesteps)
 144   }
 145 
 146 
 147 def create_model(*args, **kwargs):
 148   ocr_model = model.Model(mparams=create_mparams(), *args, **kwargs)
 149   return ocr_model
