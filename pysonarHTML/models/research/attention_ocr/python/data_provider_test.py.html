  16 """Tests for data_provider."""
  17 
  18 import numpy as np
  19 import tensorflow as tf
  20 from tensorflow.contrib.slim import queues
  21 
  22 import datasets
  23 import data_provider
  24 
  25 
  26 class DataProviderTest(tf.test.TestCase):
  27   def setUp(self):
  28     tf.test.TestCase.setUp(self)
  29 
  30   def test_preprocessed_image_values_are_in_range(self):
  31     image_shape = (5, 4, 3)
  32     fake_image = np.random.randint(low=0, high=255, size=image_shape)
  33     image_tf = data_provider.preprocess_image(fake_image)
  34 
  35     with self.test_session() as sess:
  36       image_np = sess.run(image_tf)
  37 
  38     self.assertEqual(image_np.shape, image_shape)
  39     min_value, max_value = np.min(image_np), np.max(image_np)
  40     self.assertTrue((-1.28 < min_value) and (min_value < 1.27))
  41     self.assertTrue((-1.28 < max_value) and (max_value < 1.27))
  42 
  43   def test_provided_data_has_correct_shape(self):
  44     batch_size = 4
  45     data = data_provider.get_data(
  46         dataset=datasets.fsns_test.get_test_split(),
  47         batch_size=batch_size,
  48         augment=True,
  49         central_crop_size=None)
  50 
  51     with self.test_session() as sess, queues.QueueRunners(sess):
  52       images_np, labels_np = sess.run([data.images, data.labels_one_hot])
  53 
  54     self.assertEqual(images_np.shape, (batch_size, 150, 600, 3))
  55     self.assertEqual(labels_np.shape, (batch_size, 37, 134))
  56 
  57   def test_optionally_applies_central_crop(self):
  58     batch_size = 4
  59     data = data_provider.get_data(
  60         dataset=datasets.fsns_test.get_test_split(),
  61         batch_size=batch_size,
  62         augment=True,
  63         central_crop_size=(500, 100))
  64 
  65     with self.test_session() as sess, queues.QueueRunners(sess):
  66       images_np = sess.run(data.images)
  67 
  68     self.assertEqual(images_np.shape, (batch_size, 100, 500, 3))
