   1 """A script to run inference on a set of image files.
   2 
   3 NOTE #1: The Attention OCR model was trained only using FSNS train dataset and
   4 it will work only for images which look more or less similar to french street
   5 names. In order to apply it to images from a different distribution you need
   6 to retrain (or at least fine-tune) it using images from that distribution.
   7 
   8 NOTE #2: This script exists for demo purposes only. It is highly recommended
   9 to use tools and mechanisms provided by the TensorFlow Serving system to run
  10 inference on TensorFlow models in production:
  11 https://www.tensorflow.org/serving/serving_basic
  12 
  13 Usage:
  14 python demo_inference.py --batch_size=32 \
  15   --checkpoint=model.ckpt-399731\
  16   --image_path_pattern=./datasets/data/fsns/temp/fsns_train_%02d.png
  17 """
  18 import numpy as np
  19 import PIL.Image
  20 
  21 import tensorflow as tf
  22 from tensorflow.python.platform import flags
  23 from tensorflow.python.training import monitored_session
  24 
  25 import common_flags
  26 import datasets
  27 import data_provider
  28 
  29 FLAGS = flags.FLAGS
  30 common_flags.define()
  31 
  32 # e.g. ./datasets/data/fsns/temp/fsns_train_%02d.png
  33 flags.DEFINE_string('image_path_pattern', '',
  34                     'A file pattern with a placeholder for the image index.')
  35 
  36 
  37 def get_dataset_image_size(dataset_name):
  38   # Ideally this info should be exposed through the dataset interface itself.
  39   # But currently it is not available by other means.
  40   ds_module = getattr(datasets, dataset_name)
  41   height, width, _ = ds_module.DEFAULT_CONFIG['image_shape']
  42   return width, height
  43 
  44 
  45 def load_images(file_pattern, batch_size, dataset_name):
  46   width, height = get_dataset_image_size(dataset_name)
  47   images_actual_data = np.ndarray(shape=(batch_size, height, width, 3),
  48                                   dtype='uint8')
  49   for i in range(batch_size):
  50     path = file_pattern % i
  51     print("Reading %s" % path)
  52     pil_image = PIL.Image.open(tf.gfile.GFile(path, 'rb'))
  53     images_actual_data[i, ...] = np.asarray(pil_image)
  54   return images_actual_data
  55 
  56 
  57 def create_model(batch_size, dataset_name):
  58   width, height = get_dataset_image_size(dataset_name)
  59   dataset = common_flags.create_dataset(split_name=FLAGS.split_name)
  60   model = common_flags.create_model(
  61     num_char_classes=dataset.num_char_classes,
  62     seq_length=dataset.max_sequence_length,
  63     num_views=dataset.num_of_views,
  64     null_code=dataset.null_code,
  65     charset=dataset.charset)
  66   raw_images = tf.placeholder(tf.uint8, shape=[batch_size, height, width, 3])
  67   images = tf.map_fn(data_provider.preprocess_image, raw_images,
  68                      dtype=tf.float32)
  69   endpoints = model.create_base(images, labels_one_hot=None)
  70   return raw_images, endpoints
  71 
  72 
  73 def run(checkpoint, batch_size, dataset_name, image_path_pattern):
  74   images_placeholder, endpoints = create_model(batch_size,
  75                                                dataset_name)
  76   images_data = load_images(image_path_pattern, batch_size,
  77                             dataset_name)
  78   session_creator = monitored_session.ChiefSessionCreator(
  79     checkpoint_filename_with_path=checkpoint)
  80   with monitored_session.MonitoredSession(
  81       session_creator=session_creator) as sess:
  82     predictions = sess.run(endpoints.predicted_text,
  83                            feed_dict={images_placeholder: images_data})
  84   return [pr_bytes.decode('utf-8') for pr_bytes in predictions.tolist()]
  85 
  86 
  87 def main(_):
  88   print("Predicted strings:")
  89   predictions = run(FLAGS.checkpoint, FLAGS.batch_size, FLAGS.dataset_name,
  90                   FLAGS.image_path_pattern)
  91   for line in predictions:
  92     print(line)
  93 
  94 
  95 if __name__ == '__main__':
  96   tf.app.run()
