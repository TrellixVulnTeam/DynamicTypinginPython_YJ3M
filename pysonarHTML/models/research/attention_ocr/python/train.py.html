  32 FLAGS = flags.FLAGS
  33 common_flags.define()
  34 
  35 # yapf: disable
  36 flags.DEFINE_integer('task', 0,
  37                      'The Task ID. This value is used when training with '
  38                      'multiple workers to identify each worker.')
  39 
  40 flags.DEFINE_integer('ps_tasks', 0,
  41                      'The number of parameter servers. If the value is 0, then'
  42                      ' the parameters are handled locally by the worker.')
  43 
  44 flags.DEFINE_integer('save_summaries_secs', 60,
  45                      'The frequency with which summaries are saved, in '
  46                      'seconds.')
  47 
  48 flags.DEFINE_integer('save_interval_secs', 600,
  49                      'Frequency in seconds of saving the model.')
  50 
  51 flags.DEFINE_integer('max_number_of_steps', int(1e10),
  52                      'The maximum number of gradient steps.')
  53 
  54 flags.DEFINE_string('checkpoint_inception', '',
  55                     'Checkpoint to recover inception weights from.')
  56 
  57 flags.DEFINE_float('clip_gradient_norm', 2.0,
  58                    'If greater than 0 then the gradients would be clipped by '
  59                    'it.')
  60 
  61 flags.DEFINE_bool('sync_replicas', False,
  62                   'If True will synchronize replicas during training.')
  63 
  64 flags.DEFINE_integer('replicas_to_aggregate', 1,
  65                      'The number of gradients updates before updating params.')
  66 
  67 flags.DEFINE_integer('total_num_replicas', 1,
  68                      'Total number of worker replicas.')
  69 
  70 flags.DEFINE_integer('startup_delay_steps', 15,
  71                      'Number of training steps between replicas startup.')
  72 
  73 flags.DEFINE_boolean('reset_train_dir', False,
  74                      'If true will delete all files in the train_log_dir')
  75 
  76 flags.DEFINE_boolean('show_graph_stats', False,
  77                      'Output model size stats to stderr.')

  80 TrainingHParams = collections.namedtuple('TrainingHParams', [
  81     'learning_rate',
  82     'optimizer',
  83     'momentum',
  84     'use_augment_input',
  85 ])

  88 def get_training_hparams():
  89   return TrainingHParams(
  90       learning_rate=FLAGS.learning_rate,
  91       optimizer=FLAGS.optimizer,
  92       momentum=FLAGS.momentum,
  93       use_augment_input=FLAGS.use_augment_input)
  94 
  95 
  96 def create_optimizer(hparams):
  97   """Creates optimized based on the specified flags."""
  98   if hparams.optimizer == 'momentum':
  99     optimizer = tf.train.MomentumOptimizer(
 100         hparams.learning_rate, momentum=hparams.momentum)
 101   elif hparams.optimizer == 'adam':
 102     optimizer = tf.train.AdamOptimizer(hparams.learning_rate)
 103   elif hparams.optimizer == 'adadelta':
 104     optimizer = tf.train.AdadeltaOptimizer(hparams.learning_rate)
 105   elif hparams.optimizer == 'adagrad':
 106     optimizer = tf.train.AdagradOptimizer(hparams.learning_rate)
 107   elif hparams.optimizer == 'rmsprop':
 108     optimizer = tf.train.RMSPropOptimizer(
 109         hparams.learning_rate, momentum=hparams.momentum)
 110   return optimizer

 113 def train(loss, init_fn, hparams):
 114   """Wraps slim.learning.train to run a training loop.
 115 
 116   Args:
 117     loss: a loss tensor
 118     init_fn: A callable to be executed after all other initialization is done.
 119     hparams: a model hyper parameters
 120   """
 121   optimizer = create_optimizer(hparams)
 122 
 123   if FLAGS.sync_replicas:
 124     replica_id = tf.constant(FLAGS.task, tf.int32, shape=())
 125     optimizer = tf.LegacySyncReplicasOptimizer(
 126         opt=optimizer,
 127         replicas_to_aggregate=FLAGS.replicas_to_aggregate,
 128         replica_id=replica_id,
 129         total_num_replicas=FLAGS.total_num_replicas)
 130     sync_optimizer = optimizer
 131     startup_delay_steps = 0
 132   else:
 133     startup_delay_steps = 0
 134     sync_optimizer = None
 135 
 136   train_op = slim.learning.create_train_op(
 137       loss,
 138       optimizer,
 139       summarize_gradients=True,
 140       clip_gradient_norm=FLAGS.clip_gradient_norm)
 141 
 142   slim.learning.train(
 143       train_op=train_op,
 144       logdir=FLAGS.train_log_dir,
 145       graph=loss.graph,
 146       master=FLAGS.master,
 147       is_chief=(FLAGS.task == 0),
 148       number_of_steps=FLAGS.max_number_of_steps,
 149       save_summaries_secs=FLAGS.save_summaries_secs,
 150       save_interval_secs=FLAGS.save_interval_secs,
 151       startup_delay_steps=startup_delay_steps,
 152       sync_optimizer=sync_optimizer,
 153       init_fn=init_fn)
 154 
 155 
 156 def prepare_training_dir():
 157   if not tf.gfile.Exists(FLAGS.train_log_dir):
 158     logging.info('Create a new training directory %s', FLAGS.train_log_dir)
 159     tf.gfile.MakeDirs(FLAGS.train_log_dir)
 160   else:
 161     if FLAGS.reset_train_dir:
 162       logging.info('Reset the training directory %s', FLAGS.train_log_dir)
 163       tf.gfile.DeleteRecursively(FLAGS.train_log_dir)
 164       tf.gfile.MakeDirs(FLAGS.train_log_dir)
 165     else:
 166       logging.info('Use already existing training directory %s',
 167                    FLAGS.train_log_dir)
 168 
 169 
 170 def calculate_graph_metrics():
 171   param_stats = model_analyzer.print_model_analysis(
 172       tf.get_default_graph(),
 173       tfprof_options=model_analyzer.TRAINABLE_VARS_PARAMS_STAT_OPTIONS)
 174   return param_stats.total_parameters

 177 def main(_):
 178   prepare_training_dir()
 179 
 180   dataset = common_flags.create_dataset(split_name=FLAGS.split_name)
 181   model = common_flags.create_model(dataset.num_char_classes,
 182                                     dataset.max_sequence_length,
 183                                     dataset.num_of_views, dataset.null_code)
 184   hparams = get_training_hparams()
 185 
 186   # If ps_tasks is zero, the local device is used. When using multiple
 187   # (non-local) replicas, the ReplicaDeviceSetter distributes the variables
 188   # across the different devices.
 189   device_setter = tf.train.replica_device_setter(
 190       FLAGS.ps_tasks, merge_devices=True)
 191   with tf.device(device_setter):
 192     data = data_provider.get_data(
 193         dataset,
 194         FLAGS.batch_size,
 195         augment=hparams.use_augment_input,
 196         central_crop_size=common_flags.get_crop_size())
 197     endpoints = model.create_base(data.images, data.labels_one_hot)
 198     total_loss = model.create_loss(data, endpoints)
 199     model.create_summaries(data, endpoints, dataset.charset, is_training=True)
 200     init_fn = model.create_init_fn_to_restore(FLAGS.checkpoint,
 201                                               FLAGS.checkpoint_inception)
 202     if FLAGS.show_graph_stats:
 203       logging.info('Total number of weights in the graph: %s',
 204                    calculate_graph_metrics())
 205     train(total_loss, init_fn, hparams)
 206 
 207 
 208 if __name__ == '__main__':
 209   app.run()
