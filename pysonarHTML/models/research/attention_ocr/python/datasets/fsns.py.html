  59 def read_charset(filename, null_character=u'\u2591'):
  60   """Reads a charset definition from a tab separated text file.
  61 
  62   charset file has to have format compatible with the FSNS dataset.
  63 
  64   Args:
  65     filename: a path to the charset file.
  66     null_character: a unicode character used to replace '<null>' character. the
  67       default value is a light shade block 'â–‘'.
  68 
  69   Returns:
  70     a dictionary with keys equal to character codes and values - unicode
  71     characters.
  72   """
  73   pattern = re.compile(r'(\d+)\t(.+)')
  74   charset = {}
  75   with tf.gfile.GFile(filename) as f:
  76     for i, line in enumerate(f):
  77       m = pattern.match(line)
  78       if m is None:
  79         logging.warning('incorrect charset file. line #%d: %s', i, line)
  80         continue
  81       code = int(m.group(1))
  82       char = m.group(2)
  83       if char == '<nul>':
  84         char = null_character
  85       charset[code] = char
  86   return charset

  89 class _NumOfViewsHandler(slim.tfexample_decoder.ItemHandler):
  90   """Convenience handler to determine number of views stored in an image."""
  91 
  92   def __init__(self, width_key, original_width_key, num_of_views):
  93     super(_NumOfViewsHandler, self).__init__([width_key, original_width_key])
  94     self._width_key = width_key
  95     self._original_width_key = original_width_key
  96     self._num_of_views = num_of_views
  97 
  98   def tensors_to_item(self, keys_to_tensors):
  99     return tf.to_int64(
 100         self._num_of_views * keys_to_tensors[self._original_width_key] /
 101         keys_to_tensors[self._width_key])

 104 def get_split(split_name, dataset_dir=None, config=None):
 105   """Returns a dataset tuple for FSNS dataset.
 106 
 107   Args:
 108     split_name: A train/test split name.
 109     dataset_dir: The base directory of the dataset sources, by default it uses
 110       a predefined CNS path (see DEFAULT_DATASET_DIR).
 111     config: A dictionary with dataset configuration. If None - will use the
 112       DEFAULT_CONFIG.
 113 
 114   Returns:
 115     A `Dataset` namedtuple.
 116 
 117   Raises:
 118     ValueError: if `split_name` is not a valid train/test split.
 119   """
 120   if not dataset_dir:
 121     dataset_dir = DEFAULT_DATASET_DIR
 122 
 123   if not config:
 124     config = DEFAULT_CONFIG
 125 
 126   if split_name not in config['splits']:
 127     raise ValueError('split name %s was not recognized.' % split_name)
 128 
 129   logging.info('Using %s dataset split_name=%s dataset_dir=%s', config['name'],
 130                split_name, dataset_dir)
 131 
 132   # Ignores the 'image/height' feature.
 133   zero = tf.zeros([1], dtype=tf.int64)
 134   keys_to_features = {
 135       'image/encoded':
 136       tf.FixedLenFeature((), tf.string, default_value=''),
 137       'image/format':
 138       tf.FixedLenFeature((), tf.string, default_value='png'),
 139       'image/width':
 140       tf.FixedLenFeature([1], tf.int64, default_value=zero),
 141       'image/orig_width':
 142       tf.FixedLenFeature([1], tf.int64, default_value=zero),
 143       'image/class':
 144       tf.FixedLenFeature([config['max_sequence_length']], tf.int64),
 145       'image/unpadded_class':
 146       tf.VarLenFeature(tf.int64),
 147       'image/text':
 148       tf.FixedLenFeature([1], tf.string, default_value=''),
 149   }
 150   items_to_handlers = {
 151       'image':
 152       slim.tfexample_decoder.Image(
 153           shape=config['image_shape'],
 154           image_key='image/encoded',
 155           format_key='image/format'),
 156       'label':
 157       slim.tfexample_decoder.Tensor(tensor_key='image/class'),
 158       'text':
 159       slim.tfexample_decoder.Tensor(tensor_key='image/text'),
 160       'num_of_views':
 161       _NumOfViewsHandler(
 162           width_key='image/width',
 163           original_width_key='image/orig_width',
 164           num_of_views=config['num_of_views'])
 165   }
 166   decoder = slim.tfexample_decoder.TFExampleDecoder(keys_to_features,
 167                                                     items_to_handlers)
 168   charset_file = os.path.join(dataset_dir, config['charset_filename'])
 169   charset = read_charset(charset_file)
 170   file_pattern = os.path.join(dataset_dir,
 171                               config['splits'][split_name]['pattern'])
 172   return slim.dataset.Dataset(
 173       data_sources=file_pattern,
 174       reader=tf.TFRecordReader,
 175       decoder=decoder,
 176       num_samples=config['splits'][split_name]['size'],
 177       items_to_descriptions=config['items_to_descriptions'],
 178       #  additional parameters for convenience.
 179       charset=charset,
 180       num_char_classes=len(charset),
 181       num_of_views=config['num_of_views'],
 182       max_sequence_length=config['max_sequence_length'],
 183       null_code=config['null_code'])
