  16 """Tests for unittest_utils."""
  17 
  18 import numpy as np
  19 import io
  20 from PIL import Image as PILImage
  21 import tensorflow as tf
  22 
  23 from datasets import unittest_utils
  24 
  25 
  26 class UnittestUtilsTest(tf.test.TestCase):
  27   def test_creates_an_image_of_specified_shape(self):
  28     image, _ = unittest_utils.create_random_image('PNG', (10, 20, 3))
  29     self.assertEqual(image.shape, (10, 20, 3))
  30 
  31   def test_encoded_image_corresponds_to_numpy_array(self):
  32     image, encoded = unittest_utils.create_random_image('PNG', (20, 10, 3))
  33     pil_image = PILImage.open(io.BytesIO(encoded))
  34     self.assertAllEqual(image, np.array(pil_image))
  35 
  36   def test_created_example_has_correct_values(self):
  37     example_serialized = unittest_utils.create_serialized_example({
  38         'labels': [1, 2, 3],
  39         'data': [b'FAKE']
  40     })
  41     example = tf.train.Example()
  42     example.ParseFromString(example_serialized)
  43     self.assertProtoEquals("""
  44       features {
  45         feature {
  46           key: "labels"
  47            value { int64_list {
  48              value: 1
  49              value: 2
  50              value: 3
  51            }}
  52          }
  53          feature {
  54            key: "data"
  55            value { bytes_list {
  56              value: "FAKE"
  57            }}
  58          }
  59       }
  60     """, example)
  61 
  62 
  63 if __name__ == '__main__':
  64   tf.test.main()
