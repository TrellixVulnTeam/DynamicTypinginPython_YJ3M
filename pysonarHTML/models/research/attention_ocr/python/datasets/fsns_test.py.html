  29 def get_test_split():
  30   config = fsns.DEFAULT_CONFIG.copy()
  31   config['splits'] = {'test': {'size': 5, 'pattern': 'fsns-00000-of-00001'}}
  32   return fsns.get_split('test', dataset_dir(), config)
  33 
  34 
  35 def dataset_dir():
  36   return os.path.join(os.path.dirname(__file__), 'testdata/fsns')
  37 
  38 
  39 class FsnsTest(tf.test.TestCase):
  40   def test_decodes_example_proto(self):
  41     expected_label = range(37)
  42     expected_image, encoded = unittest_utils.create_random_image(
  43         'PNG', shape=(150, 600, 3))
  44     serialized = unittest_utils.create_serialized_example({
  45         'image/encoded': [encoded],
  46         'image/format': [b'PNG'],
  47         'image/class':
  48         expected_label,
  49         'image/unpadded_class':
  50         range(10),
  51         'image/text': [b'Raw text'],
  52         'image/orig_width': [150],
  53         'image/width': [600]
  54     })
  55 
  56     decoder = fsns.get_split('train', dataset_dir()).decoder
  57     with self.test_session() as sess:
  58       data_tuple = collections.namedtuple('DecodedData', decoder.list_items())
  59       data = sess.run(data_tuple(*decoder.decode(serialized)))
  60 
  61     self.assertAllEqual(expected_image, data.image)
  62     self.assertAllEqual(expected_label, data.label)
  63     self.assertEqual([b'Raw text'], data.text)
  64     self.assertEqual([1], data.num_of_views)
  65 
  66   def test_label_has_shape_defined(self):
  67     serialized = 'fake'
  68     decoder = fsns.get_split('train', dataset_dir()).decoder
  69 
  70     [label_tf] = decoder.decode(serialized, ['label'])
  71 
  72     self.assertEqual(label_tf.get_shape().dims[0], 37)
  73 
  74   def test_dataset_tuple_has_all_extra_attributes(self):
  75     dataset = fsns.get_split('train', dataset_dir())
  76 
  77     self.assertTrue(dataset.charset)
  78     self.assertTrue(dataset.num_char_classes)
  79     self.assertTrue(dataset.num_of_views)
  80     self.assertTrue(dataset.max_sequence_length)
  81     self.assertTrue(dataset.null_code)
  82 
  83   def test_can_use_the_test_data(self):
  84     batch_size = 1
  85     dataset = get_test_split()
  86     provider = slim.dataset_data_provider.DatasetDataProvider(
  87         dataset,
  88         shuffle=True,
  89         common_queue_capacity=2 * batch_size,
  90         common_queue_min=batch_size)
  91     image_tf, label_tf = provider.get(['image', 'label'])
  92 
  93     with self.test_session() as sess:
  94       sess.run(tf.global_variables_initializer())
  95       with slim.queues.QueueRunners(sess):
  96         image_np, label_np = sess.run([image_tf, label_tf])
  97 
  98     self.assertEqual((150, 600, 3), image_np.shape)
  99     self.assertEqual((37, ), label_np.shape)
