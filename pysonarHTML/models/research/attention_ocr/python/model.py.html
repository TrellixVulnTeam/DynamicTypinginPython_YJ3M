  63 def _dict_to_array(id_to_char, default_character):
  64   num_char_classes = max(id_to_char.keys()) + 1
  65   array = [default_character] * num_char_classes
  66   for k, v in id_to_char.items():
  67     array[k] = v
  68   return array

  71 class CharsetMapper(object):
  72   """A simple class to map tensor ids into strings.
  73 
  74     It works only when the character set is 1:1 mapping between individual
  75     characters and individual ids.
  76 
  77     Make sure you call tf.tables_initializer().run() as part of the init op.
  78     """
  79 
  80   def __init__(self, charset, default_character='?'):
  81     """Creates a lookup table.
  82 
  83     Args:
  84       charset: a dictionary with id-to-character mapping.
  85     """
  86     mapping_strings = tf.constant(_dict_to_array(charset, default_character))
  87     self.table = tf.contrib.lookup.index_to_string_table_from_tensor(
  88       mapping=mapping_strings, default_value=default_character)
  89 
  90   def get_text(self, ids):
  91     """Returns a string corresponding to a sequence of character ids.
  92 
  93         Args:
  94           ids: a tensor with shape [batch_size, max_sequence_length]
  95         """
  96     return tf.reduce_join(
  97       self.table.lookup(tf.to_int64(ids)), reduction_indices=1)

 100 def get_softmax_loss_fn(label_smoothing):
 101   """Returns sparse or dense loss function depending on the label_smoothing.
 102 
 103     Args:
 104       label_smoothing: weight for label smoothing
 105 
 106     Returns:
 107       a function which takes labels and predictions as arguments and returns
 108       a softmax loss for the selected type of labels (sparse or dense).
 109     """
 110   if label_smoothing > 0:
 111 
 112     def loss_fn(labels, logits):
 113       return (tf.nn.softmax_cross_entropy_with_logits(
 114         logits=logits, labels=labels))
 115   else:
 116 
 117     def loss_fn(labels, logits):
 118       return tf.nn.sparse_softmax_cross_entropy_with_logits(
 119         logits=logits, labels=labels)
 120 
 121   return loss_fn

 127   def __init__(self,
 128                num_char_classes,
 129                seq_length,
 130                num_views,
 131                null_code,
 132                mparams=None,
 133                charset=None):
 134     """Initialized model parameters.
 135 
 136     Args:
 137       num_char_classes: size of character set.
 138       seq_length: number of characters in a sequence.
 139       num_views: Number of views (conv towers) to use.
 140       null_code: A character code corresponding to a character which
 141         indicates end of a sequence.
 142       mparams: a dictionary with hyper parameters for methods,  keys -
 143         function names, values - corresponding namedtuples.
 144       charset: an optional dictionary with a mapping between character ids and
 145         utf8 strings. If specified the OutputEndpoints.predicted_text will
 146         utf8 encoded strings corresponding to the character ids returned by
 147         OutputEndpoints.predicted_chars (by default the predicted_text contains
 148         an empty vector). 
 149         NOTE: Make sure you call tf.tables_initializer().run() if the charset
 150         specified.
 151     """
 152     super(Model, self).__init__()
 153     self._params = ModelParams(
 154       num_char_classes=num_char_classes,
 155       seq_length=seq_length,
 156       num_views=num_views,
 157       null_code=null_code)
 158     self._mparams = self.default_mparams()
 159     if mparams:
 160       self._mparams.update(mparams)
 161     self._charset = charset

 185   def conv_tower_fn(self, images, is_training=True, reuse=None):
 186     """Computes convolutional features using the InceptionV3 model.
 187 
 188     Args:
 189       images: A tensor of shape [batch_size, height, width, channels].
 190       is_training: whether is training or not.
 191       reuse: whether or not the network and its variables should be reused. To
 192         be able to reuse 'scope' must be given.
 193 
 194     Returns:
 195       A tensor of shape [batch_size, OH, OW, N], where OWxOH is resolution of
 196       output feature map and N is number of output features (depends on the
 197       network architecture).
 198     """
 199     mparams = self._mparams['conv_tower_fn']
 200     logging.debug('Using final_endpoint=%s', mparams.final_endpoint)
 201     with tf.variable_scope('conv_tower_fn/INCE'):
 202       if reuse:
 203         tf.get_variable_scope().reuse_variables()
 204       with slim.arg_scope(inception.inception_v3_arg_scope()):
 205         with slim.arg_scope([slim.batch_norm, slim.dropout],
 206                             is_training=is_training):
 207           net, _ = inception.inception_v3_base(
 208             images, final_endpoint=mparams.final_endpoint)
 209       return net

 211   def _create_lstm_inputs(self, net):
 212     """Splits an input tensor into a list of tensors (features).
 213 
 214     Args:
 215       net: A feature map of shape [batch_size, num_features, feature_size].
 216 
 217     Raises:
 218       AssertionError: if num_features is less than seq_length.
 219 
 220     Returns:
 221       A list with seq_length tensors of shape [batch_size, feature_size]
 222     """
 223     num_features = net.get_shape().dims[1].value
 224     if num_features < self._params.seq_length:
 225       raise AssertionError('Incorrect dimension #1 of input tensor'
 226                            ' %d should be bigger than %d (shape=%s)' %
 227                            (num_features, self._params.seq_length,
 228                             net.get_shape()))
 229     elif num_features > self._params.seq_length:
 230       logging.warning('Ignoring some features: use %d of %d (shape=%s)',
 231                       self._params.seq_length, num_features, net.get_shape())
 232       net = tf.slice(net, [0, 0, 0], [-1, self._params.seq_length, -1])
 233 
 234     return tf.unstack(net, axis=1)

 236   def sequence_logit_fn(self, net, labels_one_hot):
 237     mparams = self._mparams['sequence_logit_fn']
 238     # TODO(gorban): remove /alias suffixes from the scopes.
 239     with tf.variable_scope('sequence_logit_fn/SQLR'):
 240       layer_class = sequence_layers.get_layer_class(mparams.use_attention,
 241                                                     mparams.use_autoregression)
 242       layer = layer_class(net, labels_one_hot, self._params, mparams)
 243       return layer.create_logits()

 245   def max_pool_views(self, nets_list):
 246     """Max pool across all nets in spatial dimensions.
 247 
 248     Args:
 249       nets_list: A list of 4D tensors with identical size.
 250 
 251     Returns:
 252       A tensor with the same size as any input tensors.
 253     """
 254     batch_size, height, width, num_features = [
 255       d.value for d in nets_list[0].get_shape().dims
 256     ]
 257     xy_flat_shape = (batch_size, 1, height * width, num_features)
 258     nets_for_merge = []
 259     with tf.variable_scope('max_pool_views', values=nets_list):
 260       for net in nets_list:
 261         nets_for_merge.append(tf.reshape(net, xy_flat_shape))
 262       merged_net = tf.concat(nets_for_merge, 1)
 263       net = slim.max_pool2d(
 264         merged_net, kernel_size=[len(nets_list), 1], stride=1)
 265       net = tf.reshape(net, (batch_size, height, width, num_features))
 266     return net
 267 
 268   def pool_views_fn(self, nets):
 269     """Combines output of multiple convolutional towers into a single tensor.
 270 
 271     It stacks towers one on top another (in height dim) in a 4x1 grid.
 272     The order is arbitrary design choice and shouldn't matter much.
 273 
 274     Args:
 275       nets: list of tensors of shape=[batch_size, height, width, num_features].
 276 
 277     Returns:
 278       A tensor of shape [batch_size, seq_length, features_size].
 279     """
 280     with tf.variable_scope('pool_views_fn/STCK'):
 281       net = tf.concat(nets, 1)
 282       batch_size = net.get_shape().dims[0].value
 283       feature_size = net.get_shape().dims[3].value
 284       return tf.reshape(net, [batch_size, -1, feature_size])

 286   def char_predictions(self, chars_logit):
 287     """Returns confidence scores (softmax values) for predicted characters.
 288 
 289     Args:
 290       chars_logit: chars logits, a tensor with shape
 291         [batch_size x seq_length x num_char_classes]
 292 
 293     Returns:
 294       A tuple (ids, log_prob, scores), where:
 295         ids - predicted characters, a int32 tensor with shape
 296           [batch_size x seq_length];
 297         log_prob - a log probability of all characters, a float tensor with
 298           shape [batch_size, seq_length, num_char_classes];
 299         scores - corresponding confidence scores for characters, a float
 300         tensor
 301           with shape [batch_size x seq_length].
 302     """
 303     log_prob = utils.logits_to_log_prob(chars_logit)
 304     ids = tf.to_int32(tf.argmax(log_prob, axis=2), name='predicted_chars')
 305     mask = tf.cast(
 306       slim.one_hot_encoding(ids, self._params.num_char_classes), tf.bool)
 307     all_scores = tf.nn.softmax(chars_logit)
 308     selected_scores = tf.boolean_mask(all_scores, mask, name='char_scores')
 309     scores = tf.reshape(selected_scores, shape=(-1, self._params.seq_length))
 310     return ids, log_prob, scores

 312   def encode_coordinates_fn(self, net):
 313     """Adds one-hot encoding of coordinates to different views in the networks.
 314 
 315     For each "pixel" of a feature map it adds a onehot encoded x and y
 316     coordinates.
 317 
 318     Args:
 319       net: a tensor of shape=[batch_size, height, width, num_features]
 320 
 321     Returns:
 322       a tensor with the same height and width, but altered feature_size.
 323     """
 324     mparams = self._mparams['encode_coordinates_fn']
 325     if mparams.enabled:
 326       batch_size, h, w, _ = net.shape.as_list()
 327       x, y = tf.meshgrid(tf.range(w), tf.range(h))
 328       w_loc = slim.one_hot_encoding(x, num_classes=w)
 329       h_loc = slim.one_hot_encoding(y, num_classes=h)
 330       loc = tf.concat([h_loc, w_loc], 2)
 331       loc = tf.tile(tf.expand_dims(loc, 0), [batch_size, 1, 1, 1])
 332       return tf.concat([net, loc], 3)
 333     else:
 334       return net

 336   def create_base(self,
 337                   images,
 338                   labels_one_hot,
 339                   scope='AttentionOcr_v1',
 340                   reuse=None):
 341     """Creates a base part of the Model (no gradients, losses or summaries).
 342 
 343     Args:
 344       images: A tensor of shape [batch_size, height, width, channels].
 345       labels_one_hot: Optional (can be None) one-hot encoding for ground truth
 346         labels. If provided the function will create a model for training.
 347       scope: Optional variable_scope.
 348       reuse: whether or not the network and its variables should be reused. To
 349         be able to reuse 'scope' must be given.
 350 
 351     Returns:
 352       A named tuple OutputEndpoints.
 353     """
 354     logging.debug('images: %s', images)
 355     is_training = labels_one_hot is not None
 356     with tf.variable_scope(scope, reuse=reuse):
 357       views = tf.split(
 358         value=images, num_or_size_splits=self._params.num_views, axis=2)
 359       logging.debug('Views=%d single view: %s', len(views), views[0])
 360 
 361       nets = [
 362         self.conv_tower_fn(v, is_training, reuse=(i != 0))
 363         for i, v in enumerate(views)
 364       ]
 365       logging.debug('Conv tower: %s', nets[0])
 366 
 367       nets = [self.encode_coordinates_fn(net) for net in nets]
 368       logging.debug('Conv tower w/ encoded coordinates: %s', nets[0])
 369 
 370       net = self.pool_views_fn(nets)
 371       logging.debug('Pooled views: %s', net)
 372 
 373       chars_logit = self.sequence_logit_fn(net, labels_one_hot)
 374       logging.debug('chars_logit: %s', chars_logit)
 375 
 376       predicted_chars, chars_log_prob, predicted_scores = (
 377         self.char_predictions(chars_logit))
 378       if self._charset:
 379         character_mapper = CharsetMapper(self._charset)
 380         predicted_text = character_mapper.get_text(predicted_chars)
 381       else:
 382         predicted_text = tf.constant([])
 383     return OutputEndpoints(
 384       chars_logit=chars_logit,
 385       chars_log_prob=chars_log_prob,
 386       predicted_chars=predicted_chars,
 387       predicted_scores=predicted_scores,
 388       predicted_text=predicted_text)

 390   def create_loss(self, data, endpoints):
 391     """Creates all losses required to train the model.
 392 
 393     Args:
 394       data: InputEndpoints namedtuple.
 395       endpoints: Model namedtuple.
 396 
 397     Returns:
 398       Total loss.
 399     """
 400     # NOTE: the return value of ModelLoss is not used directly for the
 401     # gradient computation because under the hood it calls slim.losses.AddLoss,
 402     # which registers the loss in an internal collection and later returns it
 403     # as part of GetTotalLoss. We need to use total loss because model may have
 404     # multiple losses including regularization losses.
 405     self.sequence_loss_fn(endpoints.chars_logit, data.labels)
 406     total_loss = slim.losses.get_total_loss()
 407     tf.summary.scalar('TotalLoss', total_loss)
 408     return total_loss

 410   def label_smoothing_regularization(self, chars_labels, weight=0.1):
 411     """Applies a label smoothing regularization.
 412 
 413     Uses the same method as in https://arxiv.org/abs/1512.00567.
 414 
 415     Args:
 416       chars_labels: ground truth ids of charactes,
 417         shape=[batch_size, seq_length];
 418       weight: label-smoothing regularization weight.
 419 
 420     Returns:
 421       A sensor with the same shape as the input.
 422     """
 423     one_hot_labels = tf.one_hot(
 424       chars_labels, depth=self._params.num_char_classes, axis=-1)
 425     pos_weight = 1.0 - weight
 426     neg_weight = weight / self._params.num_char_classes
 427     return one_hot_labels * pos_weight + neg_weight

 429   def sequence_loss_fn(self, chars_logits, chars_labels):
 430     """Loss function for char sequence.
 431 
 432     Depending on values of hyper parameters it applies label smoothing and can
 433     also ignore all null chars after the first one.
 434 
 435     Args:
 436       chars_logits: logits for predicted characters,
 437         shape=[batch_size, seq_length, num_char_classes];
 438       chars_labels: ground truth ids of characters,
 439         shape=[batch_size, seq_length];
 440       mparams: method hyper parameters.
 441 
 442     Returns:
 443       A Tensor with shape [batch_size] - the log-perplexity for each sequence.
 444     """
 445     mparams = self._mparams['sequence_loss_fn']
 446     with tf.variable_scope('sequence_loss_fn/SLF'):
 447       if mparams.label_smoothing > 0:
 448         smoothed_one_hot_labels = self.label_smoothing_regularization(
 449           chars_labels, mparams.label_smoothing)
 450         labels_list = tf.unstack(smoothed_one_hot_labels, axis=1)
 451       else:
 452         # NOTE: in case of sparse softmax we are not using one-hot
 453         # encoding.
 454         labels_list = tf.unstack(chars_labels, axis=1)
 455 
 456       batch_size, seq_length, _ = chars_logits.shape.as_list()
 457       if mparams.ignore_nulls:
 458         weights = tf.ones((batch_size, seq_length), dtype=tf.float32)
 459       else:
 460         # Suppose that reject character is the last in the charset.
 461         reject_char = tf.constant(
 462           self._params.num_char_classes - 1,
 463           shape=(batch_size, seq_length),
 464           dtype=tf.int64)
 465         known_char = tf.not_equal(chars_labels, reject_char)
 466         weights = tf.to_float(known_char)
 467 
 468       logits_list = tf.unstack(chars_logits, axis=1)
 469       weights_list = tf.unstack(weights, axis=1)
 470       loss = tf.contrib.legacy_seq2seq.sequence_loss(
 471         logits_list,
 472         labels_list,
 473         weights_list,
 474         softmax_loss_function=get_softmax_loss_fn(mparams.label_smoothing),
 475         average_across_timesteps=mparams.average_across_timesteps)
 476       tf.losses.add_loss(loss)
 477       return loss

 479   def create_summaries(self, data, endpoints, charset, is_training):
 480     """Creates all summaries for the model.
 481 
 482     Args:
 483       data: InputEndpoints namedtuple.
 484       endpoints: OutputEndpoints namedtuple.
 485       charset: A dictionary with mapping between character codes and
 486         unicode characters. Use the one provided by a dataset.charset.
 487       is_training: If True will create summary prefixes for training job,
 488         otherwise - for evaluation.
 489 
 490     Returns:
 491       A list of evaluation ops
 492     """
 493 
 494     def sname(label):
 495       prefix = 'train' if is_training else 'eval'
 496       return '%s/%s' % (prefix, label)
 497 
 498     max_outputs = 4
 499     # TODO(gorban): uncomment, when tf.summary.text released.
 500     # charset_mapper = CharsetMapper(charset)
 501     # pr_text = charset_mapper.get_text(

 505     # tf.summary.text(sname('text/gt'), gt_text)
 506     tf.summary.image(sname('image'), data.images, max_outputs=max_outputs)
 507 
 508     if is_training:
 509       tf.summary.image(
 510         sname('image/orig'), data.images_orig, max_outputs=max_outputs)
 511       for var in tf.trainable_variables():
 512         tf.summary.histogram(var.op.name, var)
 513       return None
 514 
 515     else:
 516       names_to_values = {}
 517       names_to_updates = {}
 518 
 519       def use_metric(name, value_update_tuple):
 520         names_to_values[name] = value_update_tuple[0]
 521         names_to_updates[name] = value_update_tuple[1]
 522 
 523       use_metric('CharacterAccuracy',
 524                  metrics.char_accuracy(
 525                    endpoints.predicted_chars,
 526                    data.labels,
 527                    streaming=True,
 528                    rej_char=self._params.null_code))
 529       # Sequence accuracy computed by cutting sequence at the first null char
 530       use_metric('SequenceAccuracy',
 531                  metrics.sequence_accuracy(
 532                    endpoints.predicted_chars,
 533                    data.labels,
 534                    streaming=True,
 535                    rej_char=self._params.null_code))
 536 
 537       for name, value in names_to_values.items():
 538         summary_name = 'eval/' + name
 539         tf.summary.scalar(summary_name, tf.Print(value, [value], summary_name))
 540       return list(names_to_updates.values())

 542   def create_init_fn_to_restore(self, master_checkpoint,
 543                                 inception_checkpoint=None):
 544     """Creates an init operations to restore weights from various checkpoints.
 545 
 546     Args:
 547       master_checkpoint: path to a checkpoint which contains all weights for
 548         the whole model.
 549       inception_checkpoint: path to a checkpoint which contains weights for the
 550         inception part only.
 551 
 552     Returns:
 553       a function to run initialization ops.
 554     """
 555     all_assign_ops = []
 556     all_feed_dict = {}
 557 
 558     def assign_from_checkpoint(variables, checkpoint):
 559       logging.info('Request to re-store %d weights from %s',
 560                    len(variables), checkpoint)
 561       if not variables:
 562         logging.error('Can\'t find any variables to restore.')
 563         sys.exit(1)
 564       assign_op, feed_dict = slim.assign_from_checkpoint(checkpoint, variables)
 565       all_assign_ops.append(assign_op)
 566       all_feed_dict.update(feed_dict)
 567 
 568     logging.info('variables_to_restore:\n%s' % utils.variables_to_restore().keys())
 569     logging.info('moving_average_variables:\n%s' % [v.op.name for v in tf.moving_average_variables()])
 570     logging.info('trainable_variables:\n%s' % [v.op.name for v in tf.trainable_variables()])
 571     if master_checkpoint:
 572       assign_from_checkpoint(utils.variables_to_restore(), master_checkpoint)
 573 
 574     if inception_checkpoint:
 575       variables = utils.variables_to_restore(
 576         'AttentionOcr_v1/conv_tower_fn/INCE', strip_scope=True)
 577       assign_from_checkpoint(variables, inception_checkpoint)
 578 
 579     def init_assign_fn(sess):
 580       logging.info('Restoring checkpoint(s)')
 581       sess.run(all_assign_ops, all_feed_dict)
 582 
 583     return init_assign_fn
