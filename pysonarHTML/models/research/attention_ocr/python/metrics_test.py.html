  16 """Tests for the metrics module."""
  17 import contextlib
  18 import numpy as np
  19 import tensorflow as tf
  20 
  21 import metrics
  22 
  23 
  24 class AccuracyTest(tf.test.TestCase):
  25   def setUp(self):
  26     tf.test.TestCase.setUp(self)
  27     self.rng = np.random.RandomState([11, 23, 50])
  28     self.num_char_classes = 3
  29     self.batch_size = 4
  30     self.seq_length = 5
  31     self.rej_char = 42
  32 
  33   @contextlib.contextmanager
  34   def initialized_session(self):
  35     """Wrapper for test session context manager with required initialization.
  36 
  37     Yields:
  38       A session object that should be used as a context manager.
  39     """
  40     with self.cached_session() as sess:
  41       sess.run(tf.global_variables_initializer())
  42       sess.run(tf.local_variables_initializer())
  43       yield sess
  44 
  45   def _fake_labels(self):
  46     return self.rng.randint(
  47         low=0,
  48         high=self.num_char_classes,
  49         size=(self.batch_size, self.seq_length),
  50         dtype='int32')
  51 
  52   def _incorrect_copy(self, values, bad_indexes):
  53     incorrect = np.copy(values)
  54     incorrect[bad_indexes] = values[bad_indexes] + 1
  55     return incorrect
  56 
  57   def test_sequence_accuracy_identical_samples(self):
  58     labels_tf = tf.convert_to_tensor(self._fake_labels())
  59 
  60     accuracy_tf = metrics.sequence_accuracy(labels_tf, labels_tf,
  61                                             self.rej_char)
  62     with self.initialized_session() as sess:
  63       accuracy_np = sess.run(accuracy_tf)
  64 
  65     self.assertAlmostEqual(accuracy_np, 1.0)
  66 
  67   def test_sequence_accuracy_one_char_difference(self):
  68     ground_truth_np = self._fake_labels()
  69     ground_truth_tf = tf.convert_to_tensor(ground_truth_np)
  70     prediction_tf = tf.convert_to_tensor(
  71         self._incorrect_copy(ground_truth_np, bad_indexes=((0, 0))))
  72 
  73     accuracy_tf = metrics.sequence_accuracy(prediction_tf, ground_truth_tf,
  74                                             self.rej_char)
  75     with self.initialized_session() as sess:
  76       accuracy_np = sess.run(accuracy_tf)
  77 
  78     # 1 of 4 sequences is incorrect.
  79     self.assertAlmostEqual(accuracy_np, 1.0 - 1.0 / self.batch_size)
  80 
  81   def test_char_accuracy_one_char_difference_with_padding(self):
  82     ground_truth_np = self._fake_labels()
  83     ground_truth_tf = tf.convert_to_tensor(ground_truth_np)
  84     prediction_tf = tf.convert_to_tensor(
  85         self._incorrect_copy(ground_truth_np, bad_indexes=((0, 0))))
  86 
  87     accuracy_tf = metrics.char_accuracy(prediction_tf, ground_truth_tf,
  88                                         self.rej_char)
  89     with self.initialized_session() as sess:
  90       accuracy_np = sess.run(accuracy_tf)
  91 
  92     chars_count = self.seq_length * self.batch_size
  93     self.assertAlmostEqual(accuracy_np, 1.0 - 1.0 / chars_count)
  94

  96 if __name__ == '__main__':
  97   tf.test.main()
