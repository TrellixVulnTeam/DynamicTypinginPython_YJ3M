  20 import tensorflow as tf
  21 from contextlib import contextmanager
  22 
  23 from tensorflow.python.ops import variable_scope
  24 
  25 # sanity global state to ensure non recursive.
  26 _is_variable_replacing = [False]
  27 
  28 def in_variable_replace_scope():
  29   return _is_variable_replacing[0]
  30 
  31 @contextmanager
  32 def variable_replace(replacements, no_new=True):
  33   """ A context manager that replaces variables.
  34 
  35   This is a context manager that replaces all calls to
  36   get_variable with the variable in replacements.
  37   This function does not support recursive application.
  38 
  39   Args:
  40     replacements: dict
  41         dictionary mapping a variable to replace (the key), with
  42         the variable one wants to replace this variable with (the value).
  43     no_new: bool
  44         raise an error if variables were created.
  45         This is for sanity checking.
  46   Raises:
  47     ValueError: if a new variable or not all the replacements are used.
  48   """
  49   # TODO(lmetz) This function is a bit scary, as it relies on monkey patching
  50   # the call to get_variable. Ideally this can be done with variable_scope's
  51   # custom_getter attribute, but when initially writing this that was not
  52   # avalible.
  53 
  54   replacements = {k: v for k, v in replacements.items() if not k == v}
  55 
  56   init_vars = tf.trainable_variables()
  57   old_get_variable = variable_scope.get_variable
  58   old_tf_get_variable = tf.get_variable
  59 
  60   names_replace = {}
  61   has_replaced_names = []
  62   tf.logging.vlog(2, "Trying to replace")
  63   for k, v in replacements.items():
  64     tf.logging.vlog(2, k.name + " >> " + v.name)
  65   tf.logging.vlog(2, "===")
  66 
  67   for k, v in replacements.items():
  68     strip_name = k.name.replace("/read:0", "")
  69     strip_name = strip_name.replace(":0", "")
  70     names_replace[strip_name] = v
  71     # TODO(lmetz) is there a cleaner way to do this?
  72   def new_get_variable(name, *args, **kwargs):
  73     #print "Monkeypatch get variable run with name:", name
  74     n = tf.get_variable_scope().name + "/" + name
  75     #print "Monkeypatch get variable run with name:", n
  76     if n in names_replace:
  77       has_replaced_names.append(n)
  78       return names_replace[n]
  79     else:
  80       return old_get_variable(name, *args, **kwargs)
  81 
  82   # perform the monkey patch
  83   if _is_variable_replacing[0] == True:
  84     raise ValueError("No recursive calling to variable replace allowed.")
  85 
  86   variable_scope.get_variable = new_get_variable
  87   tf.get_variable = new_get_variable
  88 
  89   _is_variable_replacing[0] = True
  90 
  91   yield
  92 
  93   if set(has_replaced_names) != set(names_replace.keys()):
  94     print "Didn't use all replacements"
  95     print "replaced variables that are not requested??"
  96     print "==="
  97     for n in list(set(has_replaced_names) - set(names_replace.keys())):
  98       print n
  99     print "Missed replacing variables"
 100     print "==="
 101     for n in list(set(names_replace.keys()) - set(has_replaced_names)):
 102       print n, "==>", names_replace[n].name
 103     raise ValueError("Fix this -- see stderr")
 104 
 105   # undo the monkey patch
 106   tf.get_variable = old_tf_get_variable
 107   variable_scope.get_variable = old_get_variable
 108 
 109   _is_variable_replacing[0] = False
 110 
 111   final_vars = tf.trainable_variables()
 112   assert set(init_vars) == set(final_vars), "trainable variables changed"
