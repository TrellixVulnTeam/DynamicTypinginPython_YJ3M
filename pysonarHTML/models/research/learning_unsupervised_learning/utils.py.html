  38 def eqzip(*args):
  39   """Zip but raises error if lengths don't match.
  40 
  41   Args:
  42     *args: list of lists or tuples
  43   Returns:
  44     list: the result of zip
  45   Raises:
  46     ValueError: when the lengths don't match
  47   """
  48 
  49   sizes = [len(x) for x in args]
  50   if not all([sizes[0] == x for x in sizes]):
  51     raise ValueError("Lists are of different sizes. \n %s"%str(sizes))
  52   return zip(*args)

  55 @contextlib.contextmanager
  56 def assert_no_new_variables():
  57   """Ensure that no tf.Variables are constructed inside the context.
  58 
  59   Yields:
  60     None
  61   Raises:
  62     ValueError: if there is a variable created.
  63   """
  64   num_vars = len(tf.global_variables())
  65   old_variables = tf.global_variables()
  66   yield
  67   if len(tf.global_variables()) != num_vars:
  68     new_vars = set(tf.global_variables()) - set(old_variables)
  69     tf.logging.error("NEW VARIABLES CREATED")
  70     tf.logging.error(10*"=")
  71     for v in new_vars:
  72       tf.logging.error(v)
  73 
  74     raise ValueError("Variables created inside an "
  75                      "assert_no_new_variables context")
  76   if old_variables != tf.global_variables():
  77     raise ValueError("Variables somehow changed inside an "
  78                      "assert_no_new_variables context."
  79                      "This means something modified the tf.global_variables()")

  89 def state_barrier_context(state):
  90   """Return a context manager that prevents interior ops from running
  91   unless the whole state has been computed.
  92 
  93   This is to prevent assign race conditions.
  94   """
  95   tensors = [x for x in nest.flatten(state) if type(x) == tf.Tensor]
  96   tarray = [x.flow for x in nest.flatten(state) if hasattr(x, "flow")]
  97   return tf.control_dependencies(tensors + tarray)

 100 def _identity_fn(tf_entity):
 101   if hasattr(tf_entity, "identity"):
 102     return tf_entity.identity()
 103   else:
 104     return tf.identity(tf_entity)

 107 def state_barrier_result(state):
 108   """Return the same state, but with a control dependency to prevent it from
 109   being partially computed
 110   """
 111   with state_barrier_context(state):
 112     return nest.map_structure(_identity_fn, state)

 115 def train_iterator(num_iterations):
 116   """Iterator that returns an index of the current step.
 117   This iterator runs forever if num_iterations is None
 118   otherwise it runs for some fixed amount of steps.
 119   """
 120   if num_iterations is None:
 121     return itertools.count()
 122   else:
 123     return xrange(num_iterations)

 126 def print_op(op, msg):
 127   """Print a string and return an op wrapped in a control dependency to make
 128   sure it ran."""
 129   print_op = tf.Print(tf.constant(0), [tf.constant(0)], msg)
 130   return tf.group(op, print_op)

 133 class MultiQueueRunner(tf.train.QueueRunner):
 134   """A QueueRunner with multiple queues """
 135   def __init__(self, queues, enqueue_ops):
 136     close_op = tf.group(* [q.close() for q in queues])
 137     cancel_op = tf.group(
 138         * [q.close(cancel_pending_enqueues=True) for q in queues])
 139     queue_closed_exception_types = (errors.OutOfRangeError,)
 140 
 141     enqueue_op = tf.group(*enqueue_ops, name="multi_enqueue")
 142 
 143     super(MultiQueueRunner, self).__init__(
 144         queues[0],
 145         enqueue_ops=[enqueue_op],
 146         close_op=close_op,
 147         cancel_op=cancel_op,
 148         queue_closed_exception_types=queue_closed_exception_types)

 151 # This function is not elegant, but I tried so many other ways to get this to
 152 # work and this is the only one that ended up not incuring significant overhead
 153 # or obscure tensorflow bugs.
 154 def sample_n_per_class(dataset, samples_per_class):
 155   """Create a new callable / dataset object that returns batches of each with
 156   samples_per_class per label.
 157 
 158   Args:
 159     dataset: fn
 160     samples_per_class: int
 161   Returns:
 162     function, [] -> batch where batch is the same type as the return of
 163     dataset().
 164   """
 165 
 166   with tf.control_dependencies(None), tf.name_scope(None):
 167     with tf.name_scope("queue_runner/sample_n_per_class"):
 168       batch = dataset()
 169       num_classes = batch.label_onehot.shape.as_list()[1]
 170       batch_size = num_classes * samples_per_class
 171 
 172       flatten = nest.flatten(batch)
 173       queues = []
 174       enqueue_ops = []
 175       capacity = samples_per_class * 20
 176       for i in xrange(num_classes):
 177         queue = tf.FIFOQueue(
 178             capacity=capacity,
 179             shapes=[f.shape.as_list()[1:] for f in flatten],
 180             dtypes=[f.dtype for f in flatten])
 181         queues.append(queue)
 182 
 183         idx = tf.where(tf.equal(batch.label, i))
 184         sub_batch = []
 185         to_enqueue = []
 186         for elem in batch:
 187           new_e = tf.gather(elem, idx)
 188           new_e = tf.squeeze(new_e, 1)
 189           to_enqueue.append(new_e)
 190 
 191         remaining = (capacity - queue.size())
 192         to_add = tf.minimum(tf.shape(idx)[0], remaining)
 193 
 194         def _enqueue():
 195           return queue.enqueue_many([t[:to_add] for t in to_enqueue])
 196 
 197         enqueue_op = tf.cond(
 198             tf.equal(to_add, 0), tf.no_op, _enqueue)
 199         enqueue_ops.append(enqueue_op)
 200 
 201       # This has caused many deadlocks / issues. This is some logging to at least
 202       # shed light to what is going on.
 203       print_lam = lambda: tf.Print(tf.constant(0.0), [q.size() for q in queues], "MultiQueueRunner queues status. Has capacity %d"%capacity)
 204       some_percent_of_time = tf.less(tf.random_uniform([]), 0.0005)
 205       maybe_print = tf.cond(some_percent_of_time, print_lam, lambda: tf.constant(0.0))
 206       with tf.control_dependencies([maybe_print]):
 207         enqueue_ops = [tf.group(e) for e in enqueue_ops]
 208       qr = MultiQueueRunner(queues=queues, enqueue_ops=enqueue_ops)
 209       tf.train.add_queue_runner(qr)
 210 
 211   def dequeue_batch():
 212     with tf.name_scope("sample_n_per_batch/dequeue/"):
 213       entries = []
 214       for q in queues:
 215         entries.append(q.dequeue_many(samples_per_class))
 216 
 217       flat_batch = [tf.concat(x, 0) for x in zip(*entries)]
 218       idx = tf.random_shuffle(tf.range(batch_size))
 219       flat_batch = [tf.gather(f, idx, axis=0) for f in flat_batch]
 220       return nest.pack_sequence_as(batch, flat_batch)
 221 
 222   return dequeue_batch

 224 def structure_map_multi(func, values):
 225   all_values = [nest.flatten(v) for v in values]
 226   rets = []
 227   for pair in zip(*all_values):
 228     rets.append(func(pair))
 229   return nest.pack_sequence_as(values[0], rets)
 230 
 231 def structure_map_split(func, value):
 232   vv = nest.flatten(value)
 233   rets = []
 234   for v in vv:
 235     rets.append(func(v))
 236   return [nest.pack_sequence_as(value, r) for r in zip(*rets)]

 238 def assign_variables(targets, values):
 239   return tf.group(*[t.assign(v) for t,v in eqzip(targets, values)],
 240                   name="assign_variables")

 243 def create_variables_in_class_scope(method):
 244   """Force the variables constructed in this class to live in the sonnet module.
 245   Wraps a method on a sonnet module.
 246 
 247   For example the following will create two different variables.
 248   ```
 249   class Mod(snt.AbstractModule):
 250     @create_variables_in_class_scope
 251     def dynamic_thing(self, input, name):
 252       return snt.Linear(name)(input)
 253   mod.dynamic_thing(x, name="module_nameA")
 254   mod.dynamic_thing(x, name="module_nameB")
 255   # reuse
 256   mod.dynamic_thing(y, name="module_nameA")
 257   ```
 258   """
 259   @functools.wraps(method)
 260   def wrapper(obj, *args, **kwargs):
 261     def default_context_manager(reuse=None):
 262       variable_scope = obj.variable_scope
 263       return tf.variable_scope(variable_scope, reuse=reuse)
 264 
 265     variable_scope_context_manager = getattr(obj, "_enter_variable_scope",
 266                                              default_context_manager)
 267     graph = tf.get_default_graph()
 268 
 269     # Temporarily enter the variable scope to capture it
 270     with variable_scope_context_manager() as tmp_variable_scope:
 271       variable_scope = tmp_variable_scope
 272 
 273     with variable_scope_ops._pure_variable_scope(
 274         variable_scope, reuse=tf.AUTO_REUSE) as pure_variable_scope:
 275 
 276       name_scope = variable_scope.original_name_scope
 277       if name_scope[-1] != "/":
 278         name_scope += "/"
 279 
 280       with tf.name_scope(name_scope):
 281         sub_scope = snt_util.to_snake_case(method.__name__)
 282         with tf.name_scope(sub_scope) as scope:
 283           out_ops = method(obj, *args, **kwargs)
 284           return out_ops
 285 
 286   return wrapper
