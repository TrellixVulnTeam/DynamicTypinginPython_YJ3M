  39 class LoggingFileWriter(tf.summary.FileWriter):
  40   """A FileWriter that also logs things out.
  41 
  42   This is entirely for ease of debugging / not having to open up Tensorboard
  43   a lot.
  44   """
  45 
  46   def __init__(self, logdir, regexes=[], **kwargs):
  47     self.regexes = regexes
  48     super(LoggingFileWriter, self).__init__(logdir, **kwargs)
  49 
  50   def add_summary(self, summary, global_step):
  51     if type(summary) != tf.Summary:
  52       summary_p = tf.Summary()
  53       summary_p.ParseFromString(summary)
  54       summary = summary_p
  55     for s in summary.value:
  56       for exists in [re.match(p, s.tag) for p in self.regexes]:
  57         if exists is not None:
  58           tf.logging.info("%d ] %s : %f", global_step, s.tag, s.simple_value)
  59           break
  60     super(LoggingFileWriter, self).add_summary(summary, global_step)

  63 def image_grid(images, max_grid_size=4, border=1):
  64   """Given images and N, return first N^2 images as an NxN image grid.
  65 
  66   Args:
  67     images: a `Tensor` of size [batch_size, height, width, channels]
  68     max_grid_size: Maximum image grid height/width
  69 
  70   Returns:
  71     Single image batch, of dim [1, h*n, w*n, c]
  72   """
  73   batch_size = images.shape.as_list()[0]
  74   to_pad = int((np.ceil(np.sqrt(batch_size)))**2 - batch_size)
  75   images = tf.pad(images, [[0, to_pad], [0, border], [0, border], [0, 0]])
  76 
  77   batch_size = images.shape.as_list()[0]
  78   grid_size = min(int(np.sqrt(batch_size)), max_grid_size)
  79   assert images.shape.as_list()[0] >= grid_size * grid_size
  80 
  81   # If we have a depth channel
  82   if images.shape.as_list()[-1] == 4:
  83     images = images[:grid_size * grid_size, :, :, 0:3]
  84     depth = tf.image.grayscale_to_rgb(images[:grid_size * grid_size, :, :, 3:4])
  85 
  86     images = tf.reshape(images, [-1, images.shape.as_list()[2], 3])
  87     split = tf.split(images, grid_size, axis=0)
  88     depth = tf.reshape(depth, [-1, images.shape.as_list()[2], 3])
  89     depth_split = tf.split(depth, grid_size, axis=0)
  90     grid = tf.concat(split + depth_split, 1)
  91     return tf.expand_dims(grid, 0)
  92   else:
  93     images = images[:grid_size * grid_size, :, :, :]
  94     images = tf.reshape(
  95         images, [-1, images.shape.as_list()[2],
  96                  images.shape.as_list()[3]])
  97     split = tf.split(value=images, num_or_size_splits=grid_size, axis=0)
  98     grid = tf.concat(split, 1)
  99     return tf.expand_dims(grid, 0)

 102 def first_layer_weight_image(weight, shape):
 103   weight_image = tf.reshape(weight,
 104                             shape + [tf.identity(weight).shape.as_list()[1]])
 105   # [winx, winy, wout]
 106   mean, var = tf.nn.moments(weight_image, [0,1,2], keep_dims=True)
 107   #mean, var = tf.nn.moments(weight_image, [0,1], keep_dims=True)
 108   weight_image = (weight_image - mean) / tf.sqrt(var + 1e-5)
 109   weight_image = (weight_image + 1.0) / 2.0
 110   weight_image = tf.clip_by_value(weight_image, 0, 1)
 111   weight_image = tf.transpose(weight_image, (3, 0, 1, 2))
 112   grid = image_grid(weight_image, max_grid_size=10)
 113   return grid

 115 def inner_layer_weight_image(weight):
 116   """Visualize a weight matrix of an inner layer.
 117   Add padding to make it square, then visualize as a gray scale image
 118   """
 119   weight = tf.identity(weight) # turn into a tensor
 120   weight = weight / (tf.reduce_max(tf.abs(weight), [0], keep_dims=True))
 121   weight = tf.reshape(weight, [1]+weight.shape.as_list() + [1])
 122   return weight

 125 def activation_image(activations, label_onehot):
 126   """Make a row sorted by class for each activation. Put a black line around the activations."""
 127   labels = tf.argmax(label_onehot, axis=1)
 128   _, n_classes = label_onehot.shape.as_list()
 129   mean, var = tf.nn.moments(activations, [0, 1])
 130   activations = (activations - mean)/tf.sqrt(var+1e-5)
 131 
 132   activations = tf.clip_by_value(activations, -1, 1)
 133   activations = (activations + 1.0) / 2.0 # shift to [0, 1]
 134 
 135   canvas = []
 136   for i in xrange(n_classes):
 137     inds = tf.where(tf.equal(labels, i))
 138 
 139     def _gather():
 140       return tf.squeeze(tf.gather(activations, inds), 1)
 141 
 142     def _empty():
 143       return tf.zeros([0, activations.shape.as_list()[1]], dtype=tf.float32)
 144 
 145     assert inds.shape.as_list()[0] is None
 146     x = tf.cond(tf.equal(tf.shape(inds)[0], 0), _empty, _gather)
 147     canvas.append(x)
 148     canvas.append(tf.zeros([1, activations.shape.as_list()[1]]))
 149   canvas = tf.concat(canvas, 0)
 150   canvas = tf.reshape(canvas, [1, activations.shape.as_list()[0]+n_classes, canvas.shape.as_list()[1], 1])
 151   return canvas

 154 def sorted_images(images, label_onehot):
 155   # images is [bs, x, y, c]
 156   labels = tf.argmax(label_onehot, axis=1)
 157   _, n_classes = label_onehot.shape.as_list()
 158   to_stack = []
 159   for i in xrange(n_classes):
 160     inds = tf.where(tf.equal(labels, i))
 161 
 162     def _gather():
 163       return tf.squeeze(tf.gather(images, inds), 1)
 164 
 165     def _empty():
 166       return tf.zeros([0] + images.shape.as_list()[1:], dtype=tf.float32)
 167 
 168     assert inds.shape.as_list()[0] is None
 169     x = tf.cond(tf.equal(tf.shape(inds)[0], 0), _empty, _gather)
 170     to_stack.append(x)
 171   # pad / trim all up to 10.
 172   padded = []
 173   for t in to_stack:
 174     n_found = tf.shape(t)[0]
 175     pad = tf.pad(t[0:10], tf.stack([tf.stack([0,tf.maximum(0, 10-n_found)]), [0,0], [0,0], [0,0]]))
 176     padded.append(pad)
 177 
 178   xs = [tf.concat(tf.split(p, 10), axis=1) for p in padded]
 179   ys = tf.concat(xs, axis=2)
 180   ys = tf.cast(tf.clip_by_value(ys, 0., 1.) * 255., tf.uint8)
 181   return ys
