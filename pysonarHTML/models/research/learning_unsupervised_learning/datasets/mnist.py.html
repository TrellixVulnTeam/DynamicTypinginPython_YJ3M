  17 import sonnet as snt
  18 import tensorflow as tf
  19 from tensorflow.python.keras.datasets import mnist
  20 from learning_unsupervised_learning.datasets import common
  21 
  22 class Mnist(snt.AbstractModule):
  23   def __init__(self, device, batch_size=128, name="Mnist"):
  24     self.device = device
  25     self.batch_size = batch_size
  26 
  27     self._make_dataset()
  28     self.iterator = None
  29 
  30     super(Mnist, self).__init__(name=name)
  31 
  32   def _make_dataset(self):
  33     (x_train, y_train), (x_test, y_test) = mnist.load_data()
  34 
  35     x_train = x_train.reshape(60000, 784)
  36     x_test = x_test.reshape(10000, 784)
  37 
  38     dataset = tf.data.Dataset.from_tensor_slices((x_train, y_train))
  39     dataset = dataset.repeat()
  40     dataset = dataset.shuffle(self.batch_size * 3)
  41     dataset = dataset.batch(self.batch_size)
  42     def _map_fn(image, label):
  43       image = tf.to_float(image) / 255.
  44       label.set_shape([self.batch_size])
  45       label = tf.cast(label, dtype=tf.int32)
  46       label_onehot = tf.one_hot(label, 10)
  47       image = tf.reshape(image, [self.batch_size, 28, 28, 1])
  48       return common.ImageLabelOnehot(
  49           image=image, label=label, label_onehot=label_onehot)
  50 
  51     self.dataset = dataset.map(_map_fn)
  52 
  53   def _build(self):
  54     if self.iterator is None:
  55       self.iterator = self.dataset.make_one_shot_iterator()
  56     batch = self.iterator.get_next()
  57     [b.set_shape([self.batch_size] + b.shape.as_list()[1:]) for b in batch]
  58     return batch
  59

  61 class TinyMnist(Mnist):
  62   def __init__(self, *args, **kwargs):
  63     kwargs.setdefault("name", "TinyMnist")
  64     super(TinyMnist, self).__init__(*args, **kwargs)
  65 
  66   def _make_dataset(self):
  67     super(TinyMnist, self)._make_dataset()
  68 
  69     def _map_fn(batch):
  70       new_img = tf.image.resize_images(batch.image, [14, 14])
  71       return common.ImageLabelOnehot(
  72           image=new_img, label=batch.label, label_onehot=batch.label_onehot)
  73 
  74     self.dataset = self.dataset.map(_map_fn)
