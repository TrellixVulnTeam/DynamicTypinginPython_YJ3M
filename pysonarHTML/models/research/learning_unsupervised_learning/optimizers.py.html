  18 """Optimizers for use in unrolled optimization.
  19 
  20 These optimizers contain a compute_updates function and its own ability to keep
  21 track of internal state.
  22 These functions can be used with a tf.while_loop to perform multiple training
  23 steps per sess.run.
  24 """

  44 class UnrollableOptimizer(snt.AbstractModule):
  45   """Interface for optimizers that can be used in unrolled computation.
  46   apply_gradients is derrived from compute_update and assign_state.
  47   """
  48 
  49   def __init__(self, *args, **kwargs):
  50     super(UnrollableOptimizer, self).__init__(*args, **kwargs)
  51     self()
  52 
  53   @abc.abstractmethod
  54   def compute_updates(self, xs, gs, state=None):
  55     """Compute next step updates for a given variable list and state.
  56 
  57     Args:
  58       xs: list of tensors
  59         The "variables" to perform an update on.
  60         Note these must match the same order for which get_state was originally
  61         called.
  62       gs: list of tensors
  63         Gradients of `xs` with respect to some loss.
  64       state: Any
  65         Optimizer specific state to keep track of accumulators such as momentum
  66         terms
  67     """
  68     raise NotImplementedError()
  69 
  70   def _build(self):
  71     pass
  72 
  73   @abc.abstractmethod
  74   def get_state(self, var_list):
  75     """Get the state value associated with a list of tf.Variables.
  76 
  77     This state is commonly going to be a NamedTuple that contains some
  78     mapping between variables and the state associated with those variables.
  79     This state could be a moving momentum variable tracked by the optimizer.
  80 
  81     Args:
  82         var_list: list of tf.Variable
  83     Returns:
  84       state: Any
  85         Optimizer specific state
  86     """
  87     raise NotImplementedError()
  88 
  89   def assign_state(self, state):
  90     """Assigns the state to the optimizers internal variables.
  91 
  92     Args:
  93       state: Any
  94     Returns:
  95       op: tf.Operation
  96         The operation that performs the assignment.
  97     """
  98     raise NotImplementedError()
  99 
 100   def apply_gradients(self, grad_vars):
 101     gradients, variables = zip(*grad_vars)
 102     state = self.get_state(variables)
 103     new_vars, new_state = self.compute_updates(variables, gradients, state)
 104     assign_op = self.assign_state(new_state)
 105     op = utils.assign_variables(variables, new_vars)
 106     return tf.group(assign_op, op, name="apply_gradients")

 109 class UnrollableGradientDescentRollingOptimizer(UnrollableOptimizer):
 110 
 111   def __init__(self,
 112                learning_rate,
 113                name="UnrollableGradientDescentRollingOptimizer"):
 114     self.learning_rate = learning_rate
 115     super(UnrollableGradientDescentRollingOptimizer, self).__init__(name=name)
 116 
 117 
 118   def compute_updates(self, xs, gs, learning_rates, state):
 119     new_vars = []
 120     for x, g, lr in utils.eqzip(xs, gs, learning_rates):
 121       if lr is None:
 122         lr = self.learning_rate
 123       if g is not None:
 124         new_vars.append((x * (1 - lr) - g * lr))
 125       else:
 126         new_vars.append(x)
 127     return new_vars, state
 128 
 129   def get_state(self, var_list):
 130     return tf.constant(0.0)
 131 
 132   def assign_state(self, state, var_list=None):
 133     return tf.no_op()
