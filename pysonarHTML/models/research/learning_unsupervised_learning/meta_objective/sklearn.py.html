  40 def build_fit(device, model_fn, num_classes, probs=True):
  41 
  42   def _py_fit_predict(trX, trY, teX):
  43     assert len(np.unique(trY)) == num_classes
  44     model = model_fn()
  45     model.fit(trX, trY)
  46     trP = model.predict(trX)
  47     teP = model.predict(teX)
  48     if probs:
  49       teP_probs = model.predict_log_proba(teX)
  50       return trP.astype(np.int64), teP.astype(np.int64), teP_probs.astype(
  51           np.float32)
  52     else:
  53       teP = model.predict(teX)
  54       return trP.astype(np.int64), teP.astype(np.int64)
  55 
  56   def return_fn(trX, trY, teX):
  57     with tf.device(device):
  58       with tf.device("/cpu:0"):
  59         if probs:
  60           return tf.py_func(
  61               _py_fit_predict,
  62               [tf.identity(trX),
  63                tf.identity(trY),
  64                tf.identity(teX)], [tf.int64, tf.int64, tf.float32])
  65         else:
  66           return tf.py_func(
  67               _py_fit_predict,
  68               [tf.identity(trX),
  69                tf.identity(trY),
  70                tf.identity(teX)], [tf.int64, tf.int64])
  71 
  72   return return_fn

 101   def _build_once(self, dataset, feature_transformer):
 102     with tf.device(self._local_device):
 103       tr_batch = dataset()
 104       te_batch = dataset()
 105       num_classes = tr_batch.label_onehot.shape.as_list()[1]
 106       all_batch = utils.structure_map_multi(lambda x: tf.concat(x, 0),
 107                                             [tr_batch, te_batch])
 108       features = feature_transformer(all_batch)
 109       trX, teX = utils.structure_map_split(lambda x: tf.split(x, 2, axis=0),
 110                                            features)
 111       trY = tf.to_int64(tr_batch.label)
 112       trY_onehot = tf.to_int32(tr_batch.label_onehot)
 113       teY = tf.to_int64(te_batch.label)
 114       teY_shape = teY.shape.as_list()
 115 
 116       def blackbox((trX, trY, teX, teY)):
 117         trY = tf.to_int32(tf.rint(trY))
 118         teY = tf.to_int32(tf.rint(teY))
 119         tf_fn = build_fit(
 120             self._local_device,
 121             self._get_model,
 122             num_classes=num_classes,
 123             probs=self.probs)
 124         if self.probs:
 125           trP, teP, teP_probs = tf_fn(trX, trY, teX)
 126         else:
 127           trP, teP = tf_fn(trX, trY, teX)
 128 
 129         teY.set_shape(teY_shape)
 130         if self.probs:
 131           onehot = tf.one_hot(teY, num_classes)
 132           crossent = -tf.reduce_sum(onehot * teP_probs, [1])
 133           return tf.reduce_mean(crossent)
 134         else:
 135           # use error rate as the loss if no surrogate is avalible.
 136           return 1 - tf.reduce_mean(
 137               tf.to_float(tf.equal(teY, tf.to_int32(teP))))
 138 
 139       test_loss = blackbox((trX, tf.to_float(trY), teX, tf.to_float(teY)))
 140 
 141       stats = {}
 142 
 143       tf_fn = build_fit(
 144           self._local_device,
 145           self._get_model,
 146           num_classes=num_classes,
 147           probs=self.probs)
 148       if self.probs:
 149         trP, teP, teP_probs = tf_fn(trX, trY, teX)
 150       else:
 151         trP, teP = tf_fn(trX, trY, teX)
 152       stats["%s/accuracy_train" % self.name] = tf.reduce_mean(
 153           tf.to_float(tf.equal(tf.to_int32(trY), tf.to_int32(trP))))
 154       stats["%s/accuracy_test" % self.name] = tf.reduce_mean(
 155           tf.to_float(tf.equal(tf.to_int32(teY), tf.to_int32(teP))))
 156       stats["%s/test_loss" % self.name] = test_loss
 157       return test_loss, stats
