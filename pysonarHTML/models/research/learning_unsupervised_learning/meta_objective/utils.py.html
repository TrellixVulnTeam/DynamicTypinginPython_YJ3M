  31 class MultiTrialMetaObjective(snt.AbstractModule):
  32   def __init__(self, samples_per_class, averages, **kwargs):
  33     self.samples_per_class = samples_per_class
  34     self.averages = averages
  35     self.dataset_map = {}
  36 
  37     super(MultiTrialMetaObjective,
  38           self).__init__(**kwargs)
  39 
  40   def _build(self, dataset, feature_transformer):
  41     if self.samples_per_class is not None:
  42       if dataset not in self.dataset_map:
  43         # datasets are outside of frames from while loops
  44         with tf.control_dependencies(None):
  45           self.dataset_map[dataset] = utils.sample_n_per_class(
  46               dataset, self.samples_per_class)
  47 
  48       dataset = self.dataset_map[dataset]
  49 
  50     stats = collections.defaultdict(list)
  51     losses = []
  52     # TODO(lmetz) move this to ingraph control flow?
  53     for _ in xrange(self.averages):
  54       loss, stat = self._build_once(dataset, feature_transformer)
  55       losses.append(loss)
  56       for k, v in stat.items():
  57         stats[k].append(v)
  58     stats = {k: tf.add_n(v) / float(len(v)) for k, v in stats.items()}
  59 
  60     for k, v in stats.items():
  61       tf.summary.scalar(k, v)
  62 
  63     return tf.add_n(losses) / float(len(losses))
  64 
  65   def local_variables(self):
  66     """List of variables that need to be updated for each evaluation.
  67 
  68     These variables should not be stored on a parameter server and
  69     should be reset every computation of a meta_objective loss.
  70 
  71     Returns:
  72       vars: list of tf.Variable
  73     """
  74     return list(
  75         snt.get_variables_in_module(self, tf.GraphKeys.TRAINABLE_VARIABLES))
