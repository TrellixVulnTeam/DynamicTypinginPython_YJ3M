  36 def solve_ridge(x, y, ridge_factor):
  37   with tf.name_scope("solve_ridge"):
  38     # Added a column of ones to the end of the feature matrix for bias
  39     A = tf.concat([x, tf.ones((x.shape.as_list()[0], 1))], axis=1)
  40 
  41     # Analytic solution for the ridge regression loss
  42     inv_target = tf.matmul(A, A, transpose_a=True)
  43     np_diag_penalty = ridge_factor * np.ones(
  44         A.shape.as_list()[1], dtype="float32")
  45     # Remove penalty on bias component of weights
  46     np_diag_penalty[-1] = 0.
  47     diag_penalty = tf.constant(np_diag_penalty)
  48     inv_target += tf.diag(diag_penalty)
  49 
  50     inv = tf.matrix_inverse(inv_target)
  51     w = tf.matmul(inv, tf.matmul(A, y, transpose_a=True))
  52     return w

  55 class LinearRegressionMetaObjective(snt.AbstractModule):
  56   """A meta objective based on training Ridge Regression with analytic solution.
  57 
  58   This is used to evaluate the performance of a given feature set trained in
  59   some other manner.
  60   """
  61 
  62   def __init__(self,
  63                local_device=None,
  64                remote_device=None,
  65                zero_one_labels=True,
  66                normalize_y_hat=True,
  67                normalize_act=False,
  68                averages=1,
  69                ridge_factor=0.1,
  70                center_y=True,
  71                hinge_loss=False,
  72                samples_per_class=10,
  73                test_train_scalar=1.0,
  74               ):
  75     self._local_device = local_device
  76     self._remote_device = remote_device
  77     self.zero_one_labels = zero_one_labels
  78     self.normalize_y_hat = normalize_y_hat
  79     self.normalize_act = normalize_act
  80     self.ridge_factor = ridge_factor
  81     self.averages = averages
  82     self.samples_per_class = samples_per_class
  83     self.center_y=center_y
  84     self.test_train_scalar=test_train_scalar
  85     self.hinge_loss = hinge_loss
  86 
  87     self.dataset_map = {}
  88 
  89     super(LinearRegressionMetaObjective,
  90           self).__init__(name="LinearRegressionMetaObjective")
  91 
  92   def _build(self, dataset, feature_transformer):
  93     if self.samples_per_class is not None:
  94       if dataset not in self.dataset_map:
  95         # datasets are outside of frames from while loops
  96         with tf.control_dependencies(None):
  97           self.dataset_map[dataset] = utils.sample_n_per_class(
  98               dataset, self.samples_per_class)
  99 
 100       dataset = self.dataset_map[dataset]
 101 
 102     stats = collections.defaultdict(list)
 103     losses = []
 104     # TODO(lmetz) move this to ingraph control flow?
 105     for _ in xrange(self.averages):
 106       loss, stat = self._build_once(dataset, feature_transformer)
 107       losses.append(loss)
 108       for k, v in stat.items():
 109         stats[k].append(v)
 110     stats = {k: tf.add_n(v) / float(len(v)) for k, v in stats.items()}
 111 
 112     summary_updates = []
 113     for k, v in stats.items():
 114       tf.summary.scalar(k, v)
 115 
 116     with tf.control_dependencies(summary_updates):
 117       return tf.add_n(losses) / float(len(losses))
 118 
 119   def _build_once(self, dataset, feature_transformer):
 120     with tf.device(self._local_device):
 121       batch = dataset()
 122       num_classes = batch.label_onehot.shape.as_list()[1]
 123 
 124       regression_mod = snt.Linear(num_classes)
 125 
 126       if self.normalize_act:
 127 
 128         def normalize_transformer(x):
 129           unnorm_x = feature_transformer(x)
 130           return tf.nn.l2_normalize(unnorm_x, 0)
 131 
 132         feature_transformer_wrap = normalize_transformer
 133       else:
 134         feature_transformer_wrap = feature_transformer
 135 
 136       # construct the variables of the right shape in the sonnet module by
 137       # calling a forward pass through the regressor.
 138       with utils.assert_no_new_variables():
 139         dummy_features = feature_transformer_wrap(batch)
 140       regression_mod(dummy_features)
 141       reg_w = regression_mod.w
 142       reg_b = regression_mod.b
 143 
 144       batch_test = dataset()
 145       all_batch = utils.structure_map_multi(lambda x: tf.concat(x, 0), [batch, batch_test])
 146       #all_batch = tf.concat([batch, batch_test], 0)
 147       # Grab a new batch of data from the dataset.
 148       features = feature_transformer_wrap(all_batch)
 149       features, features_test = utils.structure_map_split(lambda x: tf.split(x, 2, axis=0), features)
 150 
 151       def center_y(y):
 152         y -= tf.reduce_mean(y)
 153         y *= tf.rsqrt(tf.reduce_mean(tf.reduce_sum(y**2, axis=[1], keep_dims=True)))
 154         return y
 155       def get_y_vec(batch):
 156         y_pieces = []
 157         if hasattr(batch, "label_onehot"):
 158           if self.zero_one_labels:
 159             y_pieces += [batch.label_onehot]
 160           else:
 161             y_pieces += [2. * batch.label_onehot - 1.]
 162         if hasattr(batch, "regression_target"):
 163           y_pieces += [batch.regression_target]
 164         y = tf.concat(y_pieces, 1)
 165         if self.center_y:
 166           y = center_y(y)
 167         return y
 168 
 169       y_train = get_y_vec(batch)
 170 
 171       w = solve_ridge(features, y_train, self.ridge_factor)
 172 
 173       # Generate features from another batch to evaluate loss on the validation
 174       # set. This provide a less overfit signal to the learned optimizer.
 175       y_test = get_y_vec(batch_test)
 176 
 177       def compute_logit(features):
 178         # We have updated the classifier mod in previous steps, we need to
 179         # substitute out those variables to get new values.
 180         replacement = collections.OrderedDict([(reg_w, w[:-1]), (reg_b, w[-1])])
 181         with variable_replace.variable_replace(replacement):
 182           logits = regression_mod(features)
 183 
 184         return logits
 185 
 186       batch_size = y_train.shape.as_list()[0]
 187 
 188       logit_train = compute_logit(features)
 189       logit_test_unnorm = compute_logit(features_test)
 190       if self.normalize_y_hat:
 191         logit_test = logit_test_unnorm / tf.sqrt(
 192             tf.reduce_sum(logit_test_unnorm**2, axis=[1], keep_dims=True))
 193       else:
 194         logit_test = logit_test_unnorm
 195 
 196       stats = {}
 197 
 198       if self.hinge_loss:
 199         # slightly closer to the true classification loss
 200         # any distance smaller than 1 is guaranteed to map to the correct class
 201         mse_test = tf.reduce_sum(tf.nn.relu(tf.reduce_sum(tf.square(logit_test - y_test), axis=1)-1.)) / batch_size
 202       else:
 203         mse_test = tf.reduce_sum(tf.square(logit_test - y_test)) / batch_size
 204 
 205       stats["mse_test"] = mse_test
 206 
 207       mse_train = tf.reduce_sum(tf.square(logit_train - y_train)) / batch_size
 208       stats["mse_train"] = mse_train
 209 
 210       is_correct_test = tf.equal(tf.argmax(logit_test, 1), tf.argmax(y_test, 1))
 211       accuracy_test = tf.reduce_mean(tf.cast(is_correct_test, tf.float32))
 212       stats["accuracy_test"] = accuracy_test
 213 
 214       def test_confusion_fn():
 215         test_confusion = tf.confusion_matrix(tf.argmax(y_test, 1), tf.argmax(logit_test, 1))
 216         test_confusion = tf.to_float(test_confusion) / tf.constant((logit_test.shape.as_list()[0] / float(logit_test.shape.as_list()[1])), dtype=tf.float32)
 217         test_confusion = tf.expand_dims(tf.expand_dims(test_confusion, 0), 3)
 218         return test_confusion
 219       tf.summary.image("test_confusion", test_confusion_fn())
 220 
 221       def train_confusion_fn():
 222         train_confusion = tf.confusion_matrix(tf.argmax(y_train, 1), tf.argmax(logit_train, 1))
 223         train_confusion = tf.to_float(train_confusion) / tf.constant((logit_train.shape.as_list()[0] / float(logit_train.shape.as_list()[1])), dtype=tf.float32)
 224         train_confusion = tf.expand_dims(tf.expand_dims(train_confusion, 0), 3)
 225         return train_confusion
 226       tf.summary.image("train_confusion", train_confusion_fn())
 227 
 228       is_correct = tf.equal(tf.argmax(logit_train, 1), tf.argmax(y_train, 1))
 229       accuracy_train = tf.reduce_mean(tf.cast(is_correct, tf.float32))
 230       stats["accuracy_train"] = accuracy_train
 231 
 232       reg = self.ridge_factor * tf.reduce_sum(tf.square(w[:-1])) / batch_size
 233       stats["ridge_component"] = reg
 234 
 235       stats["total_loss"] = mse_test + reg
 236 
 237       loss_to_train_at = (reg+ mse_test) * self.test_train_scalar + (mse_train + reg)*(1 - self.test_train_scalar)
 238 
 239       loss_to_train_at = tf.identity(loss_to_train_at)
 240 
 241       # Minimizing the test loss should not require regurization because the
 242       # metaobjective is solved for the training loss
 243       return loss_to_train_at, stats

 245   def local_variables(self):
 246     """List of variables that need to be updated for each evaluation.
 247 
 248     These variables should not be stored on a parameter server and
 249     should be reset every computation of a meta_objective loss.
 250 
 251     Returns:
 252       vars: list of tf.Variable
 253     """
 254     return list(
 255         snt.get_variables_in_module(self, tf.GraphKeys.TRAINABLE_VARIABLES))
