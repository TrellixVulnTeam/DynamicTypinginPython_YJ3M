  37 class GradChannelReadout(snt.AbstractModule):
  38   """Perform a linear readout and reshape from input 3 tensor."""
  39 
  40   def __init__(self,
  41                num_grad_channels,
  42                device,
  43                perm=(2, 0, 1),
  44                name='GradChannelReadout'):
  45     """Args:
  46 
  47       num_grad_channels: int
  48         number of channels to readout to.
  49       device: str or callable
  50         devicwe to place weights.
  51       perm: list or tuple
  52         transpose applied.
  53     """
  54 
  55     self.num_grad_channels = num_grad_channels
  56     self.device = device
  57     self.perm = perm
  58     super(GradChannelReadout, self).__init__(name=name)
  59 
  60   def _build(self, h):
  61     with tf.device(self.device):
  62       mod = snt.Linear(self.num_grad_channels)
  63       ret = snt.BatchApply(mod)(h)
  64       # return as [num_grad_channels] x [bs] x [num units]
  65       return tf.transpose(ret, perm=self.perm)

  68 def get_weight_stats(x, axis):
  69   """ Compute weight statistics over the given axis.
  70 
  71   Args:
  72     x: tf.Tensor
  73       a batch of activations.
  74     axis: int
  75       axis to perform statistics over.
  76   Returns:
  77     tf.Tensor
  78       a 3-D tensor with statistics.
  79   """
  80   if x is None:
  81     return []
  82 
  83   stats = []
  84   l1 = tf.reduce_mean(tf.abs(x), axis=axis)
  85   l2 = tf.sqrt(tf.reduce_mean(x**2, axis=axis) + 1e-6)
  86 
  87   mean, var = tf.nn.moments(x, [axis])
  88   stats.extend([l1, l2, mean, tf.sqrt(var + 1e-8)])
  89 
  90   stats = [tf.reshape(s, [-1, 1, 1]) for s in stats]
  91 
  92   return stats

  95 class AddUnitBatchStatistics(snt.AbstractModule):
  96   """Compute some number of statistics over units and concat them on."""
  97 
  98   def __init__(self, name='AddUnitBatchStatistics'):
  99     super(AddUnitBatchStatistics, self).__init__(name=name)
 100 
 101   def _build(self, x):
 102     # [channel, bs, 1]
 103     output = x
 104     for d in [0, 1]:
 105       stats = []
 106       l1 = tf.reduce_mean(tf.abs(x), axis=d, keepdims=True)
 107       l2 = tf.sqrt(tf.reduce_mean(x**2, axis=d, keepdims=True) + 1e-6)
 108 
 109       mean, var = tf.nn.moments(x, [d], keepdims=True)
 110       stats.extend([l1, l2, mean, tf.sqrt(var + 1e-8)])
 111 
 112       to_add = tf.concat(stats, axis=2)  # [channels/1, units/1, stats]
 113       output += snt.BatchApply(snt.Linear(x.shape.as_list()[2]))(to_add)
 114     return output

 117 class ConcatUnitConv(snt.AbstractModule):
 118   """Do a small number of convolutions over units and concat / add them on."""
 119 
 120   def __init__(self, add=True):
 121     self.add = add
 122     super(ConcatUnitConv, self).__init__(name='ConcatUnitConv')
 123 
 124   def _build(self, x):
 125     # x is [units, bs, 1]
 126     net = tf.transpose(x, [1, 0, 2])  # now [bs x units x 1]
 127     channels = x.shape.as_list()[2]
 128     mod = snt.Conv1D(output_channels=channels, kernel_shape=[3])
 129     net = mod(net)
 130     net = snt.BatchNorm(axis=[0, 1])(net, is_training=False)
 131     net = tf.nn.relu(net)
 132     mod = snt.Conv1D(output_channels=channels, kernel_shape=[3])
 133     net = mod(net)
 134     net = snt.BatchNorm(axis=[0, 1])(net, is_training=False)
 135     net = tf.nn.relu(net)
 136     to_concat = tf.transpose(net, [1, 0, 2])
 137     if self.add:
 138       return x + to_concat
 139     else:
 140       return tf.concat([x, to_concat], 2)

 180   def normalize(self, change_w, normalize_epsilon=None):
 181     if normalize_epsilon is None:
 182       normalize_epsilon = self.normalize_epsilon
 183 
 184     # normalize the weights per receptive-field, rather than per-matrix
 185     var = tf.reduce_mean(tf.square(change_w), axis=0, keepdims=True)
 186     change_w = (change_w) / tf.sqrt(normalize_epsilon + var)
 187     return change_w

 192   @snt.reuse_variables
 193   def compute_top_delta(self, z):
 194     """ parameterization of topD. This converts the top level activation
 195     to an error signal.
 196     Args:
 197       z: tf.Tensor
 198         batch of final layer post activations
 199     Returns
 200       delta: tf.Tensor
 201         the error signal
 202     """
 203     s_idx = 0
 204     with tf.variable_scope('compute_top_delta'), tf.device(self.remote_device):
 205       # typically this takes [BS, length, input_channels],
 206       # We are applying this such that we convolve over the batch dimension.
 207       act = tf.expand_dims(tf.transpose(z, [1, 0]), 2)  # [channels, BS, 1]
 208 
 209       mod = snt.Conv1D(output_channels=self.top_delta_size, kernel_shape=[5])
 210       act = mod(act)
 211 
 212       act = snt.BatchNorm(axis=[0, 1])(act, is_training=False)
 213       act = tf.nn.relu(act)
 214 
 215       bs = act.shape.as_list()[0]
 216       act = tf.transpose(act, [2, 1, 0])
 217       act = snt.Conv1D(output_channels=bs, kernel_shape=[3])(act)
 218       act = snt.BatchNorm(axis=[0, 1])(act, is_training=False)
 219       act = tf.nn.relu(act)
 220       act = snt.Conv1D(output_channels=bs, kernel_shape=[3])(act)
 221       act = snt.BatchNorm(axis=[0, 1])(act, is_training=False)
 222       act = tf.nn.relu(act)
 223       act = tf.transpose(act, [2, 1, 0])
 224 
 225       prev_act = act
 226       for i in range(self.top_delta_layers):
 227         mod = snt.Conv1D(output_channels=self.top_delta_size, kernel_shape=[3])
 228         act = mod(act)
 229 
 230         act = snt.BatchNorm(axis=[0, 1])(act, is_training=False)
 231         act = tf.nn.relu(act)
 232 
 233         prev_act = act
 234 
 235       mod = snt.Conv1D(output_channels=self.delta_dim, kernel_shape=[3])
 236       act = mod(act)
 237 
 238       # [bs, feature_channels, delta_channels]
 239       act = tf.transpose(act, [1, 0, 2])
 240       return act

 242   @snt.reuse_variables
 243   def compute_h(self,
 244                 x,
 245                 z,
 246                 d,
 247                 bias,
 248                 W_bot,
 249                 W_top,
 250                 compute_perc=1.0,
 251                 compute_units=None):
 252     """z = [BS, n_units] a = [BS, n_units] b = [BS, n_units] d = [BS, n_units, delta_channels]
 253 
 254     """
 255 
 256     s_idx = 0
 257     if compute_perc != 1.0:
 258       assert compute_units is None
 259 
 260     with tf.device(self.remote_device):
 261       inp_feat = [x, z]
 262       inp_feat = [tf.transpose(f, [1, 0]) for f in inp_feat]
 263 
 264       units = x.shape.as_list()[1]
 265       bs = x.shape.as_list()[0]
 266 
 267       # add unit ID, to help the network differentiate units
 268       id_theta = tf.linspace(0., (4) * np.pi, units)
 269       assert bs is not None
 270       id_theta_bs = tf.reshape(id_theta, [-1, 1]) * tf.ones([1, bs])
 271       inp_feat += [tf.sin(id_theta_bs), tf.cos(id_theta_bs)]
 272 
 273       # list of [units, BS, 1]
 274       inp_feat = [tf.expand_dims(f, 2) for f in inp_feat]
 275 
 276       d_trans = tf.transpose(d, [1, 0, 2])
 277 
 278       if compute_perc != 1.0:
 279         compute_units = int(compute_perc * inp_feat.shape.as_list()[0])
 280 
 281       # add weight matrix statistics, both from above and below
 282       w_stats_bot = get_weight_stats(W_bot, 0)
 283       w_stats_top = get_weight_stats(W_top, 1)
 284       w_stats = w_stats_bot + w_stats_top
 285       if W_bot is None or W_top is None:
 286         # if it's an edge layer (top or bottom), just duplicate the stats for
 287         # the weight matrix that does exist
 288         w_stats = w_stats + w_stats
 289       w_stats = [tf.ones([1, x.shape[0], 1]) * ww for ww in w_stats]
 290       # w_stats is a list, with entries with shape UNITS x 1 x channels
 291 
 292       if compute_units is None:
 293         inp_feat_in = inp_feat
 294         d_trans_in = d_trans
 295         w_stats_in = w_stats
 296         bias_in = tf.transpose(bias)
 297       else:
 298         # only run on a subset of the activations.
 299         mask = tf.random_uniform(
 300             minval=0,
 301             maxval=1,
 302             dtype=tf.float32,
 303             shape=inp_feat[0].shape.as_list()[0:1])
 304         _, ind = tf.nn.top_k(mask, k=compute_units)
 305         ind = tf.reshape(ind, [-1, 1])
 306 
 307         inp_feat_in = [tf.gather_nd(xx, ind) for xx in inp_feat]
 308         w_stats_in = [tf.gather_nd(xx, ind) for xx in w_stats]
 309         d_trans_in = tf.gather_nd(d_trans, ind)
 310         bias_in = tf.gather_nd(tf.transpose(bias), ind)
 311 
 312       w_stats_in = tf.concat(w_stats_in, 2)
 313       w_stats_in_norm = w_stats_in * tf.rsqrt(
 314           tf.reduce_mean(w_stats_in**2) + 1e-6)
 315 
 316       act = tf.concat(inp_feat_in + [d_trans_in], 2)
 317       act = snt.BatchNorm(axis=[0, 1])(act, is_training=True)
 318 
 319       bias_dense = tf.reshape(bias_in, [-1, 1, 1]) * tf.ones([1, bs, 1])
 320       act = tf.concat([w_stats_in_norm, bias_dense, act], 2)
 321 
 322       mod = snt.Conv1D(output_channels=self.compute_h_size, kernel_shape=[3])
 323       act = mod(act)
 324 
 325       act = snt.BatchNorm(axis=[0, 1])(act, is_training=True)
 326       act = tf.nn.relu(act)
 327 
 328       act2 = ConcatUnitConv()(act)
 329       act = act2
 330 
 331       prev_act = act
 332       for i in range(self.compute_h_layers):
 333         mod = snt.Conv1D(output_channels=self.compute_h_size, kernel_shape=[3])
 334         act = mod(act)
 335 
 336         act = snt.BatchNorm(axis=[0, 1])(act, is_training=True)
 337         act = tf.nn.relu(act)
 338 
 339         act = ConcatUnitConv()(act)
 340 
 341         prev_act = act
 342 
 343       h = act
 344       if compute_units is not None:
 345         shape = inp_feat[0].shape.as_list()[:1] + h.shape.as_list()[1:]
 346         h = tf.scatter_nd(ind, h, shape=shape)
 347 
 348       h = tf.transpose(h, [1, 0, 2])  # [bs, units, channels]
 349 
 350       return h

 364   def merge_change_w(self, change_w_terms, global_prefix='', prefix=''):
 365     with tf.device(
 366         self.remote_device), tf.name_scope(global_prefix + '_merge_change_w'):
 367       w_base = change_w_terms['w_base']
 368 
 369       for kk in sorted(change_w_terms.keys()):
 370         name = global_prefix + 'change_w_plane_%s' % kk
 371         delta_w = change_w_terms[kk]
 372         mean, var = tf.nn.moments(delta_w, [0, 1])
 373         root_mean_square = tf.sqrt(tf.reduce_mean(delta_w**2) + 1e-6)
 374 
 375       for kk in sorted(change_w_terms.keys()):
 376         change_w_terms[kk] = self.normalize(change_w_terms[kk])
 377 
 378       initializers = {
 379           'w': tf.constant_initializer(0.1),
 380           'b': tf.zeros_initializer()
 381       }
 382       mod = snt.Linear(
 383           1,
 384           name=global_prefix + '_weight_readout_coeffs',
 385           initializers=initializers)
 386 
 387       change_w_terms_list = [
 388           change_w_terms[kk] for kk in sorted(change_w_terms.keys())
 389       ]
 390       stack_terms = tf.stack(change_w_terms_list, axis=-1)
 391       change_w = tf.squeeze(
 392           snt.BatchApply(mod)(stack_terms), axis=-1) / len(change_w_terms)
 393 
 394       # only allow perpendicular updates, or updates which grow length. don't
 395       # allow length to decay towards zero.
 396       ip = tf.reduce_mean(change_w * w_base)
 397       # zero out any updates that shrink length
 398       ip = tf.nn.relu(ip)
 399       change_w -= w_base * ip
 400       change_w /= tf.sqrt(len(change_w_terms) * 1.)
 401 
 402       change_w = self.normalize(change_w)
 403 
 404       # encourage the receptive field to not collapse to 0
 405       change_w -= w_base / 7.  # This is an arbitrary scale choice
 406 
 407       return tf.identity(change_w)

 429   @utils.create_variables_in_class_scope
 430   def low_rank_readout(self, name, h1, h2, psd=False):
 431     BS = h1.shape.as_list()[0]
 432     r_t = self.get_readout_mod(name + '_top')(h1)
 433     if psd:
 434       r_b = r_t
 435     else:
 436       r_b = self.get_readout_mod(name + '_bottom')(h2)
 437     return tf.reduce_mean(tf.matmul(r_b, r_t, transpose_a=True), axis=0) / BS

 462   @snt.reuse_variables
 463   def compute_next_state(self, grads, learning_rate, cur_state,
 464                          cur_transformer):
 465 
 466     summaries = []
 467     with tf.device(self.local_device):
 468       with tf.control_dependencies(summaries):
 469         new_vars, new_state = self.opt.compute_updates(
 470             cur_state.variables, grads, learning_rate, cur_state.opt_state)
 471         pass
 472 
 473     return OptState(
 474         variables=tuple(new_vars),
 475         opt_state=new_state,
 476         index=cur_state.index + 1)

 478   def assign_state(self, base_model, next_state):
 479     var_ups = [
 480         v.assign(nv) for v, nv in utils.eqzip(base_model.get_variables(),
 481                                               next_state.variables)

 484     opt_ups = self.opt.assign_state(next_state.opt_state)
 485 
 486     return tf.group(opt_ups, *var_ups)

 531   @snt.reuse_variables
 532   def get_shuffle_ind(self, size):
 533     if self.shuffle_ind is None:
 534       # put the shuffle in tf memory to make the eval jobs
 535       # re-entrant.
 536       shuffle_ind_val = np.random.permutation(size)
 537       shuffle_ind = tf.get_variable(
 538           name='shuffle_ind', dtype=tf.int64, initializer=shuffle_ind_val)
 539       unshuffle_ind = tf.scatter_nd(
 540           tf.reshape(shuffle_ind, [-1, 1]), tf.range(size), [size])
 541 
 542     return shuffle_ind, unshuffle_ind
 543 
 544   def _build(self, batch):
 545     image = batch.image
 546     x0 = snt.BatchFlatten()(image)
 547     if self.shuffle_input:
 548       size = x0.shape.as_list()[1]
 549       shuffle_ind, unshuffle_ind = self.get_shuffle_ind(size)
 550       x0 = tf.gather(x0, shuffle_ind, axis=1)
 551 
 552     xs = [x0]
 553     mods = []
 554     zs = []
 555     init = {}
 556 
 557     for i in range(self.n_layers):
 558       mod = common.LinearBatchNorm(
 559           self.inner_size, activation_fn=self.activation_fn)
 560       z, x = mod(xs[i])
 561       xs.append(x)
 562       zs.append(z)
 563       mods.append(mod)
 564 
 565     mod = common.LinearBatchNorm(
 566         self.output_size, activation_fn=self.activation_fn)
 567     z, x = mod(xs[-1])
 568     mods.append(mod)
 569 
 570     xs.append(x)
 571     zs.append(z)
 572 
 573     embedding_x = xs[-1]
 574 
 575     # make a random set of backward mods
 576     backward_mods = []
 577     for i, (x, x_p1) in enumerate(zip(xs[0:-1], xs[1:])):
 578       m = common.LinearBatchNorm(
 579           x_p1.shape.as_list()[1], activation_fn=tf.identity)
 580       _ = m(x)
 581       backward_mods.append(m)
 582 
 583     shape = image.shape.as_list()[1:4]
 584 
 585     for mods_p, prefix in [(mods, 'forward'), (backward_mods, 'backward')]:
 586       if self.shuffle_input:
 587         unshuf_w = tf.gather(mods_p[0].w, unshuffle_ind, axis=0)
 588       else:
 589         unshuf_w = mods_p[0].w
 590       img = summary_utils.first_layer_weight_image(unshuf_w, shape)
 591       tf.summary.image(prefix + '_w0_receptive_field', img)
 592 
 593       for i, m in enumerate(mods_p[0:]):
 594         img = summary_utils.inner_layer_weight_image(m.w)
 595         tf.summary.image(prefix + '_w%d' % (i + 1), img)
 596 
 597     img = summary_utils.sorted_images(image, batch.label_onehot)
 598     tf.summary.image('inputs', img)

 603         img = summary_utils.activation_image(x_vis, batch.label_onehot)

 613   def compute_next_h_d(self, meta_opt, w_bot, w_top, bias, x, z, d, backward_w):
 614     """ Propogate error back down the network while computing hidden state.
 615     """
 616     if z is None:
 617       z = x
 618 
 619     h = meta_opt.compute_h(x, z, d, bias, w_bot,
 620                            w_top)  # [bs x 60 x h_channels]
 621 
 622     # compute the next d
 623     delta = meta_opt.next_delta(z, h, d)
 624 
 625     if backward_w is not None:
 626 
 627       def delta_matmul(w, delta):
 628         d = tf.transpose(delta, [0, 2, 1])  # [bs x delta_channels x n_units)
 629         d = snt.BatchApply(lambda x: tf.matmul(x, w, transpose_b=True))(d)
 630         d = tf.transpose(d, [0, 2, 1])
 631         return d
 632 
 633       # replace the "backward pass" with a random matrix.
 634       d = delta_matmul(backward_w, delta)  # [bs x 60 x delta_channels]
 635       var = tf.reduce_mean(tf.square(d), [2], keepdims=True)
 636       d = d * tf.rsqrt(1e-6 + var)
 637 
 638     return h, d

 640   def weight_change_for_layer(self, meta_opt, l_idx, w_base, b_base, upper_h,
 641                               lower_h, upper_x, lower_x, prefix, include_bias):
 642     """Compute the change in weights for each layer.
 643     This computes something roughly analagous to a gradient.
 644     """
 645     reduce_upper_h = upper_h
 646     reduce_lower_h = lower_h
 647 
 648     BS = lower_x.shape.as_list()[0]
 649 
 650     change_w_terms = dict()
 651 
 652     # initial weight value normalized
 653     # normalize the weights per receptive-field, rather than per-matrix
 654     weight_scale = tf.rsqrt(
 655         tf.reduce_mean(w_base**2, axis=0, keepdims=True) + 1e-6)
 656     w_base *= weight_scale
 657 
 658     change_w_terms['w_base'] = w_base
 659 
 660     # this will act to decay larger weights towards zero
 661     change_w_terms['large_decay'] = w_base**2 * tf.sign(w_base)
 662 
 663     # term based on activations
 664     ux0 = upper_x - tf.reduce_mean(upper_x, axis=0, keepdims=True)
 665     uxs0 = ux0 * tf.rsqrt(tf.reduce_mean(ux0**2, axis=0, keepdims=True) + 1e-6)
 666     change_U = tf.matmul(uxs0, uxs0, transpose_a=True) / BS
 667     change_U /= tf.sqrt(float(change_U.shape.as_list()[0]))
 668 
 669     cw = tf.matmul(w_base, change_U)
 670     cw_scale = tf.rsqrt(tf.reduce_mean(cw**2 + 1e-8))
 671     cw *= cw_scale
 672     change_w_terms['decorr_x'] = cw
 673 
 674     # hebbian term
 675     lx0 = lower_x - tf.reduce_mean(lower_x, axis=0, keepdims=True)
 676     lxs0 = lx0 * tf.rsqrt(tf.reduce_mean(lx0**2, axis=0, keepdims=True) + 1e-6)
 677     cw = tf.matmul(lxs0, uxs0, transpose_a=True) / BS
 678     change_w_terms['hebb'] = -cw
 679 
 680     # 0th order term
 681     w_term = meta_opt.low_rank_readout(prefix + 'weight_readout_0', upper_h,
 682                                        lower_h)
 683     change_w_terms['0_order'] = w_term
 684 
 685     # # rbf term (weight update scaled by distance from 0)
 686     w_term = meta_opt.low_rank_readout(prefix + 'weight_readout_rbf',
 687                                        reduce_upper_h, reduce_lower_h)
 688     change_w_terms['rbf'] = tf.exp(-w_base**2) * w_term
 689 
 690     # 1st order term (weight dependent update to weights)
 691     w_term = meta_opt.low_rank_readout(prefix + 'weight_readout_1',
 692                                        reduce_upper_h, reduce_lower_h)
 693     change_w_terms['1_order'] = w_base * w_term
 694 
 695     # more terms based on single layer readouts.
 696     for update_type in ['lin', 'sqr']:
 697       for h_source, h_source_name in [(reduce_upper_h, 'upper'),
 698                                       (reduce_lower_h, 'lower')]:
 699         structures = ['symm']
 700         if update_type == 'lin' and h_source_name == 'upper':
 701           structures += ['psd']
 702         for structure in structures:
 703           name = update_type + '_' + h_source_name + '_' + structure
 704           if structure == 'symm':
 705             change_U = meta_opt.low_rank_readout(prefix + name, h_source,
 706                                                  h_source)
 707             change_U = (change_U + tf.transpose(change_U)) / tf.sqrt(2.)
 708             change_U = tf.matrix_set_diag(change_U,
 709                                           tf.zeros(
 710                                               [change_U.shape.as_list()[0]]))
 711           elif structure == 'psd':
 712             change_U = meta_opt.low_rank_readout(
 713                 prefix + name, h_source, None, psd=True)
 714           else:
 715             assert False
 716           change_U /= tf.sqrt(float(change_U.shape.as_list()[0]))
 717 
 718           if update_type == 'lin':
 719             sign_multiplier = tf.ones_like(w_base)
 720             w_base_l = w_base
 721           elif update_type == 'sqr':
 722             sign_multiplier = tf.sign(w_base)
 723             w_base_l = tf.sqrt(1. + w_base**2) - 1.
 724 
 725           if h_source_name == 'upper':
 726             cw = tf.matmul(w_base_l, change_U)  # [N^l-1 x N^l]
 727           elif h_source_name == 'lower':
 728             cw = tf.matmul(change_U, w_base_l)
 729           change_w_terms[name] = cw * sign_multiplier
 730 
 731 
 732     if prefix == 'forward':
 733       change_w = meta_opt.merge_change_w_forward(
 734           change_w_terms, global_prefix=prefix, prefix='l%d' % l_idx)
 735     elif prefix == 'backward':
 736       change_w = meta_opt.merge_change_w_backward(
 737           change_w_terms, global_prefix=prefix, prefix='l%d' % l_idx)
 738     else:
 739       assert (False)
 740 
 741     if not include_bias:
 742       return change_w
 743 
 744     change_b = tf.reduce_mean(meta_opt.bias_readout(upper_h), [0])
 745 
 746     # force nonlinearities to be exercised -- biases can't all be increased without bound
 747     change_b_mean = tf.reduce_mean(change_b)
 748     offset = -tf.nn.relu(-change_b_mean)
 749     change_b -= offset
 750 
 751     var = tf.reduce_mean(tf.square(change_b), [0], keepdims=True)
 752     change_b = (change_b) / tf.sqrt(0.5 + var)
 753     return change_w, change_b
 754 
 755   def compute_next_state(self, outputs, meta_opt, previous_state):
 756     zs = outputs.zs
 757     xs = outputs.xs
 758     batch = outputs.batch
 759     mods = outputs.mods
 760     backward_mods = outputs.backward_mods
 761     variables = self.get_variables()
 762 
 763     rev_mods = mods[::-1]
 764     rev_backward_mods = backward_mods[::-1]
 765     rev_xs = xs[::-1]
 766     rev_zs = zs[::-1] + [None]
 767 
 768     to_top = xs[-1]
 769 
 770     # variables that change in the loop
 771     hs = []
 772     d = meta_opt.compute_top_delta(to_top)  # [bs x 32 x delta_channels]
 773 
 774     iterator = utils.eqzip(rev_backward_mods + [None], rev_mods + [None],
 775                            [None] + rev_mods, rev_xs, rev_zs)
 776     for (backward_mod, lower_mod, upper_mod, x, z) in iterator:
 777       w_bot = None
 778       if not lower_mod is None:
 779         w_bot = previous_state.variables[variables.index(lower_mod.w)]
 780       w_top = None
 781       if not upper_mod is None:
 782         w_top = previous_state.variables[variables.index(upper_mod.w)]
 783       backward_w = None
 784       if backward_mod is not None:
 785         backward_w = previous_state.variables[variables.index(backward_mod.w)]
 786       if lower_mod is not None:
 787         bias = previous_state.variables[variables.index(lower_mod.b)]
 788       else:
 789         bias = tf.zeros([x.shape[1]])
 790 
 791       h, d = self.compute_next_h_d(
 792           meta_opt=meta_opt,
 793           w_bot=w_bot,
 794           w_top=w_top,
 795           bias=bias,
 796           backward_w=backward_w,
 797           x=x,
 798           z=z,
 799           d=d)
 800       hs.append(h)
 801 
 802     w_forward_var_idx = [variables.index(mod.w) for mod in rev_mods]
 803     w_backward_var_idx = [variables.index(mod.w) for mod in rev_backward_mods]
 804     b_var_idx = [variables.index(mod.b) for mod in rev_mods]
 805 
 806     # storage location for outputs of below loop
 807     grads = [None for _ in previous_state.variables]
 808 
 809     # over-ride learning rate for perturbation variables
 810     learning_rate = [None for _ in previous_state.variables]
 811 
 812     # This is a map -- no state is shared cross loop
 813     for l_idx, w_forward_idx, w_backward_idx, b_idx, upper_h, lower_h, lower_x, upper_x in utils.eqzip(
 814         range(len(w_forward_var_idx)), w_forward_var_idx, w_backward_var_idx,
 815         b_var_idx, hs[:-1], hs[1:], xs[::-1][1:], xs[::-1][:-1]):
 816 
 817       b_base = previous_state.variables[b_idx]
 818       change_w_forward, change_b = self.weight_change_for_layer(
 819           meta_opt=meta_opt,
 820           l_idx=l_idx,
 821           w_base=previous_state.variables[w_forward_idx],
 822           b_base=b_base,
 823           upper_h=upper_h,
 824           lower_h=lower_h,
 825           upper_x=upper_x,
 826           lower_x=lower_x,
 827           prefix='forward',
 828           include_bias=True)
 829 
 830       if self.identical_updates:
 831         change_w_backward = change_w_forward
 832       else:
 833         change_w_backward = self.weight_change_for_layer(
 834             meta_opt=meta_opt,
 835             l_idx=l_idx,
 836             w_base=previous_state.variables[w_backward_idx],
 837             b_base=b_base,
 838             upper_h=upper_h,
 839             lower_h=lower_h,
 840             upper_x=upper_x,
 841             lower_x=lower_x,
 842             prefix='backward',
 843             include_bias=False)
 844 
 845       grads[w_forward_idx] = change_w_forward
 846 
 847       grads[w_backward_idx] = change_w_backward
 848 
 849       grads[b_idx] = change_b
 850 
 851     cur_transformer = common.transformer_at_state(self,
 852                                                   previous_state.variables)
 853     next_state = meta_opt.compute_next_state(
 854         grads,
 855         learning_rate=learning_rate,
 856         cur_state=previous_state,
 857         cur_transformer=lambda x: cur_transformer(x)[0])
 858     return next_state
