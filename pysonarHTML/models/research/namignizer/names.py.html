  14 """A library showing off sequence recognition and generation with the simple
  15 example of names.
  16 
  17 We use recurrent neural nets to learn complex functions able to recognize and
  18 generate sequences of a given form. This can be used for natural language
  19 syntax recognition, dynamically generating maps or puzzles and of course
  20 baby name generation.
  21 
  22 Before using this module, it is recommended to read the Tensorflow tutorial on
  23 recurrent neural nets, as it explains the basic concepts of this model, and
  24 will show off another module, the PTB module on which this model bases itself.
  25 
  26 Here is an overview of the functions available in this module:
  27 
  28 * RNN Module for sequence functions based on PTB
  29 
  30 * Name recognition specifically for recognizing names, but can be adapted to
  31     recognizing sequence patterns
  32 
  33 * Name generations specifically for generating names, but can be adapted to
  34     generating arbitrary sequence patterns
  35 """

 101 def run_epoch(session, m, names, counts, epoch_size, eval_op, verbose=False):
 102     """Runs the model on the given data for one epoch
 103 
 104     Args:
 105         session: the tf session holding the model graph
 106         m: an instance of the NamignizerModel
 107         names: a set of lowercase names of 26 characters
 108         counts: a list of the frequency of the above names
 109         epoch_size: the number of batches to run
 110         eval_op: whether to change the params or not, and how to do it
 111     Kwargs:
 112         verbose: whether to print out state of training during the epoch
 113     Returns:
 114         cost: the average cost during the last stage of the epoch
 115     """
 116     start_time = time.time()
 117     costs = 0.0
 118     iters = 0
 119     for step, (x, y) in enumerate(data_utils.namignizer_iterator(names, counts,
 120                                                                  m.batch_size, m.num_steps, epoch_size)):
 121 
 122         cost, _ = session.run([m.cost, eval_op],
 123                               {m.input_data: x,
 124                                m.targets: y,
 125                                m.weights: np.ones(m.batch_size * m.num_steps)})
 126         costs += cost
 127         iters += m.num_steps
 128 
 129         if verbose and step % (epoch_size // 10) == 9:
 130             print("%.3f perplexity: %.3f speed: %.0f lps" %
 131                   (step * 1.0 / epoch_size, np.exp(costs / iters),
 132                    iters * m.batch_size / (time.time() - start_time)))
 133 
 134         if step >= epoch_size:
 135             break
 136 
 137     return np.exp(costs / iters)

 140 def train(data_dir, checkpoint_path, config):
 141     """Trains the model with the given data
 142 
 143     Args:
 144         data_dir: path to the data for the model (see data_utils for data
 145             format)
 146         checkpoint_path: the path to save the trained model checkpoints
 147         config: one of the above configs that specify the model and how it
 148             should be run and trained
 149     Returns:
 150         None
 151     """
 152     # Prepare Name data.
 153     print("Reading Name data in %s" % data_dir)
 154     names, counts = data_utils.read_names(data_dir)
 155 
 156     with tf.Graph().as_default(), tf.Session() as session:
 157         initializer = tf.random_uniform_initializer(-config.init_scale,
 158                                                     config.init_scale)
 159         with tf.variable_scope("model", reuse=None, initializer=initializer):
 160             m = NamignizerModel(is_training=True, config=config)
 161 
 162         tf.global_variables_initializer().run()
 163 
 164         for i in range(config.max_max_epoch):
 165             lr_decay = config.lr_decay ** max(i - config.max_epoch, 0.0)
 166             m.assign_lr(session, config.learning_rate * lr_decay)
 167 
 168             print("Epoch: %d Learning rate: %.3f" % (i + 1, session.run(m.lr)))
 169             train_perplexity = run_epoch(session, m, names, counts, config.epoch_size, m.train_op,
 170                                          verbose=True)
 171             print("Epoch: %d Train Perplexity: %.3f" %
 172                   (i + 1, train_perplexity))
 173 
 174             m.saver.save(session, checkpoint_path, global_step=i)

 177 def namignize(names, checkpoint_path, config):
 178     """Recognizes names and prints the Perplexity of the model for each names
 179     in the list
 180 
 181     Args:
 182         names: a list of names in the model format
 183         checkpoint_path: the path to restore the trained model from, should not
 184             include the model name, just the path to
 185         config: one of the above configs that specify the model and how it
 186             should be run and trained
 187     Returns:
 188         None
 189     """
 190     with tf.Graph().as_default(), tf.Session() as session:
 191 
 192         with tf.variable_scope("model"):
 193             m = NamignizerModel(is_training=False, config=config)
 194 
 195         m.saver.restore(session, checkpoint_path)
 196 
 197         for name in names:
 198             x, y = data_utils.name_to_batch(name, m.batch_size, m.num_steps)
 199 
 200             cost, loss, _ = session.run([m.cost, m.loss, tf.no_op()],
 201                                   {m.input_data: x,
 202                                    m.targets: y,
 203                                    m.weights: np.concatenate((
 204                                        np.ones(len(name)), np.zeros(m.batch_size * m.num_steps - len(name))))})
 205 
 206             print("Name {} gives us a perplexity of {}".format(
 207                 name, np.exp(cost)))

 210 def namignator(checkpoint_path, config):
 211     """Generates names randomly according to a given model
 212 
 213     Args:
 214         checkpoint_path: the path to restore the trained model from, should not
 215             include the model name, just the path to
 216         config: one of the above configs that specify the model and how it
 217             should be run and trained
 218     Returns:
 219         None
 220     """
 221     # mutate the config to become a name generator config
 222     config.num_steps = 1
 223     config.batch_size = 1
 224 
 225     with tf.Graph().as_default(), tf.Session() as session:
 226 
 227         with tf.variable_scope("model"):
 228             m = NamignizerModel(is_training=False, config=config)
 229 
 230         m.saver.restore(session, checkpoint_path)
 231 
 232         activations, final_state, _ = session.run([m.activations, m.final_state, tf.no_op()],
 233                                                   {m.input_data: np.zeros((1, 1)),
 234                                                    m.targets: np.zeros((1, 1)),
 235                                                    m.weights: np.ones(1)})
 236 
 237         # sample from our softmax activations
 238         next_letter = np.random.choice(27, p=activations[0])
 239         name = [next_letter]
 240         while next_letter != 0:
 241             activations, final_state, _ = session.run([m.activations, m.final_state, tf.no_op()],
 242                                                       {m.input_data: [[next_letter]],
 243                                                        m.targets: np.zeros((1, 1)),
 244                                                        m.initial_state: final_state,
 245                                                        m.weights: np.ones(1)})
 246 
 247             next_letter = np.random.choice(27, p=activations[0])
 248             name += [next_letter]
 249 
 250         print(map(lambda x: chr(x + 96), name))
