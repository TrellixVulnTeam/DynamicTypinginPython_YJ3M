  14 """RNN model with embeddings"""

  16 from __future__ import absolute_import
  17 from __future__ import division
  18 from __future__ import print_function
  19 
  20 import tensorflow as tf

  23 class NamignizerModel(object):
  24     """The Namignizer model ~ strongly based on PTB"""
  25 
  26     def __init__(self, is_training, config):
  27         self.batch_size = batch_size = config.batch_size
  28         self.num_steps = num_steps = config.num_steps
  29         size = config.hidden_size
  30         # will always be 27
  31         vocab_size = config.vocab_size
  32 
  33         # placeholders for inputs
  34         self._input_data = tf.placeholder(tf.int32, [batch_size, num_steps])
  35         self._targets = tf.placeholder(tf.int32, [batch_size, num_steps])
  36         # weights for the loss function
  37         self._weights = tf.placeholder(tf.float32, [batch_size * num_steps])
  38 
  39         # lstm for our RNN cell (GRU supported too)
  40         lstm_cells = []
  41         for layer in range(config.num_layers):
  42             lstm_cell = tf.contrib.rnn.BasicLSTMCell(size, forget_bias=0.0)
  43             if is_training and config.keep_prob < 1:
  44                 lstm_cell = tf.contrib.rnn.DropoutWrapper(
  45                     lstm_cell, output_keep_prob=config.keep_prob)
  46             lstm_cells.append(lstm_cell)
  47         cell = tf.contrib.rnn.MultiRNNCell(lstm_cells)
  48 
  49         self._initial_state = cell.zero_state(batch_size, tf.float32)
  50 
  51         with tf.device("/cpu:0"):
  52             embedding = tf.get_variable("embedding", [vocab_size, size])
  53             inputs = tf.nn.embedding_lookup(embedding, self._input_data)
  54 
  55         if is_training and config.keep_prob < 1:
  56             inputs = tf.nn.dropout(inputs, config.keep_prob)
  57 
  58         outputs = []
  59         state = self._initial_state
  60         with tf.variable_scope("RNN"):
  61             for time_step in range(num_steps):
  62                 if time_step > 0:
  63                     tf.get_variable_scope().reuse_variables()
  64                 (cell_output, state) = cell(inputs[:, time_step, :], state)
  65                 outputs.append(cell_output)
  66 
  67         output = tf.reshape(tf.concat(axis=1, values=outputs), [-1, size])
  68         softmax_w = tf.get_variable("softmax_w", [size, vocab_size])
  69         softmax_b = tf.get_variable("softmax_b", [vocab_size])
  70         logits = tf.matmul(output, softmax_w) + softmax_b
  71         loss = tf.contrib.legacy_seq2seq.sequence_loss_by_example(
  72             [logits],
  73             [tf.reshape(self._targets, [-1])],
  74             [self._weights])
  75         self._loss = loss
  76         self._cost = cost = tf.reduce_sum(loss) / batch_size
  77         self._final_state = state
  78 
  79         # probabilities of each letter
  80         self._activations = tf.nn.softmax(logits)
  81 
  82         # ability to save the model
  83         self.saver = tf.train.Saver(tf.global_variables())
  84 
  85         if not is_training:
  86             return
  87 
  88         self._lr = tf.Variable(0.0, trainable=False)
  89         tvars = tf.trainable_variables()
  90         grads, _ = tf.clip_by_global_norm(tf.gradients(cost, tvars),
  91                                           config.max_grad_norm)
  92         optimizer = tf.train.GradientDescentOptimizer(self.lr)
  93         self._train_op = optimizer.apply_gradients(zip(grads, tvars))
  94 
  95     def assign_lr(self, session, lr_value):
  96         session.run(tf.assign(self.lr, lr_value))
