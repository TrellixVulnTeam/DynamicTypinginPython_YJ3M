  16 """jpeg2json.py: Converts a JPEG image into a json request to CloudML.
  17 
  18 Usage:
  19 python jpeg2json.py 002s_C6_ImagerDefaults_9.jpg > request.json
  20

  25 import base64
  26 import sys
  27 
  28 
  29 def to_json(data):
  30   return '{"image_bytes":{"b64": "%s"}}' % base64.b64encode(data)
  31 
  32 
  33 if __name__ == '__main__':
  34   file = open(sys.argv[1]) if len(sys.argv) > 1 else sys.stdin
  35   print(to_json(file.read()))
