  30 tf.flags.DEFINE_string('val', '', 'validation dataset, tfrecs')

  39 def main(_):
  40   # Pick up any one-off hyper-parameters.
  41   hparams = path_model.PathBasedModel.default_hparams()
  42 
  43   with open(FLAGS.relations) as fh:
  44     relations = fh.read().splitlines()
  45 
  46   hparams.num_classes = len(relations)
  47   print('Model will predict into %d classes' % hparams.num_classes)
  48 
  49   print('Running with hyper-parameters: {}'.format(hparams))
  50 
  51   # Load the instances
  52   print('Loading instances...')
  53   opts = tf.python_io.TFRecordOptions(
  54       compression_type=tf.python_io.TFRecordCompressionType.GZIP)
  55 
  56   train_instances = list(tf.python_io.tf_record_iterator(FLAGS.train, opts))
  57   val_instances = list(tf.python_io.tf_record_iterator(FLAGS.val, opts))
  58   test_instances = list(tf.python_io.tf_record_iterator(FLAGS.test, opts))
  59 
  60   # Load the word embeddings
  61   print('Loading word embeddings...')
  62   lemma_embeddings = lexnet_common.load_word_embeddings(FLAGS.embeddings)
  63 
  64   # Define the graph and the model
  65   with tf.Graph().as_default():
  66     with tf.variable_scope('lexnet'):
  67       options = tf.python_io.TFRecordOptions(
  68           compression_type=tf.python_io.TFRecordCompressionType.GZIP)
  69       reader = tf.TFRecordReader(options=options)
  70       _, train_instance = reader.read(
  71           tf.train.string_input_producer([FLAGS.train]))
  72       shuffled_train_instance = tf.train.shuffle_batch(
  73           [train_instance],
  74           batch_size=1,
  75           num_threads=1,
  76           capacity=len(train_instances),
  77           min_after_dequeue=100,
  78       )[0]
  79 
  80       train_model = path_model.PathBasedModel(
  81           hparams, lemma_embeddings, shuffled_train_instance)
  82 
  83     with tf.variable_scope('lexnet', reuse=True):
  84       val_instance = tf.placeholder(dtype=tf.string)
  85       val_model = path_model.PathBasedModel(
  86           hparams, lemma_embeddings, val_instance)
  87 
  88     # Initialize a session and start training
  89     best_model_saver = tf.train.Saver()
  90     f1_t = tf.placeholder(tf.float32)
  91     best_f1_t = tf.Variable(0.0, trainable=False, name='best_f1')
  92     assign_best_f1_op = tf.assign(best_f1_t, f1_t)
  93 
  94     supervisor = tf.train.Supervisor(
  95         logdir=FLAGS.logdir,
  96         global_step=train_model.global_step)
  97 
  98     with supervisor.managed_session() as session:
  99       # Load the labels
 100       print('Loading labels...')
 101       val_labels = train_model.load_labels(session, val_instances)
 102 
 103       # Train the model
 104       print('Training the model...')
 105 
 106       while True:
 107         step = session.run(train_model.global_step)
 108         epoch = (step + len(train_instances) - 1) // len(train_instances)
 109         if epoch > hparams.num_epochs:
 110           break
 111 
 112         print('Starting epoch %d (step %d)...' % (1 + epoch, step))
 113 
 114         epoch_loss = train_model.run_one_epoch(session, len(train_instances))
 115 
 116         best_f1 = session.run(best_f1_t)
 117         f1 = epoch_completed(val_model, session, epoch, epoch_loss,
 118                              val_instances, val_labels, best_model_saver,
 119                              FLAGS.logdir, best_f1)
 120 
 121         if f1 > best_f1:
 122           session.run(assign_best_f1_op, {f1_t: f1})
 123 
 124         if f1 < best_f1 - 0.08:
 125           tf.logging.info('Stopping training after %d epochs.\n' % epoch)
 126           break
 127 
 128       # Print the best performance on the validation set
 129       best_f1 = session.run(best_f1_t)
 130       print('Best performance on the validation set: F1=%.3f' % best_f1)
 131 
 132       # Save the path embeddings
 133       print('Computing the path embeddings...')
 134       instances = train_instances + val_instances + test_instances
 135       path_index, path_vectors = path_model.compute_path_embeddings(
 136           val_model, session, instances)
 137 
 138       if not os.path.exists(path_emb_dir):
 139         os.makedirs(path_emb_dir)
 140 
 141       path_model.save_path_embeddings(
 142           val_model, path_vectors, path_index, FLAGS.output_dir)

 145 def epoch_completed(model, session, epoch, epoch_loss,
 146                     val_instances, val_labels, saver, save_path, best_f1):
 147   """Runs every time an epoch completes.
 148 
 149   Print the performance on the validation set, and update the saved model if
 150   its performance is better on the previous ones. If the performance dropped,
 151   tell the training to stop.
 152 
 153   Args:
 154     model: The currently trained path-based model.
 155     session: The current TensorFlow session.
 156     epoch: The epoch number.
 157     epoch_loss: The current epoch loss.
 158     val_instances: The validation set instances (evaluation between epochs).
 159     val_labels: The validation set labels (for evaluation between epochs).
 160     saver: tf.Saver object
 161     save_path: Where to save the model.
 162     best_f1: the best F1 achieved so far.
 163 
 164   Returns:
 165     The F1 achieved on the training set.
 166   """
 167   # Evaluate on the validation set
 168   val_pred = model.predict(session, val_instances)
 169   precision, recall, f1, _ = metrics.precision_recall_fscore_support(
 170       val_labels, val_pred, average='weighted')
 171   print(
 172       'Epoch: %d/%d, Loss: %f, validation set: P: %.3f, R: %.3f, F1: %.3f\n' % (
 173           epoch + 1, model.hparams.num_epochs, epoch_loss,
 174           precision, recall, f1))
 175 
 176   if f1 > best_f1:
 177     save_filename = os.path.join(save_path, 'best.ckpt')
 178     print('Saving model in: %s' % save_filename)
 179     saver.save(session, save_filename)
 180     print('Model saved in file: %s' % save_filename)
 181 
 182   return f1
